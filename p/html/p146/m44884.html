רִבּוֹנוֹ שֶׁלָּעוֹלָם עֲשֵׂה

הכתובת: תפילת הפרנסה


רִבּוֹנוֹ שֶׁלָּעוֹלָם, עֲשֵׂה עִמִּי לְמַעַן שְׁמָךְ הַגָּדוֹל

וּלְמַעַן רַחֲמֶיךָ הָרַבִּים אֲשֶׁר אֵין לָהֶם קֵץ וְתִכְלָה

וּפַרְנְסֵנִי כְּעוֹרֵב וּכְאֶחָד מִשְּׁאָר בִּרְיוֹתֶיךָ אַף עַל פִּי שְׁאֵין אֲנִי רָאוּי וְהָגוּן

כִּי אַתָּה זָן וּמְפַרְנֵס לָרָעִים וְלַטּוֹבִים וּלְהָגוּן וְלִשְׁאֵינוֹ הָגוּן מִמְּקוֹר טוּבְךָ

דְּכַתִיב וַיֹּאמַר אֲנִי אַעֲבִיר כָּל טוּבִי עַל פָּנֶיךָ וְקָרָאתִי בְּשֵׁם יְוֹיָ לְפָנֶיךָ

וְחַנּוֹתִי אֶת אֲשֶׁר אָחוֹן וְרִחַמְתִּי אֶת אֲשֶׁר אֲרַחֵם

כִּי כֵן דַּרְכְּךָ לַעֲשׂוֹת חֶסֶד חִנָּם בְּכָל דוֹר וָדוֹר

אָנָּא יְוֹיָ, יִכְבְּשׁוּ רַחֲמֶיךָ אֶת כַּעֲסֶךָ וְיִגּוֹלוּ רַחֲמֶיךָ עַל מִדּוֹתֶיךָ

וַעֲשֵׂה לְמַעַן י״ג מִדּוֹת שֶׁל רַחֲמִים הַגְּנוּזִים בְּכִתְרָךְ אֲשֶׁר אֵין לָהֶם סוֹף

וּתְעוֹרֵר שֶׁפַע רְצוֹנֶךָ וְתַזְמִין לָנוּ פַרְנָסָה טוֹבָה מֵאוֹצַר הַפַּרְנָסָה

וּמִשָּׁם יְהִי אֶשֶׁד שִׁבְעַת הַנְּחָלִים אֶל הַיָּם הַגָּדוֹל

וּתְפַרְנְסֵנוּ מִשָּׁם פַּרְנָסָה שְׁלֵמָה

שֶׁלֹּא תַצְרִיכֵנוּ לִידֵי אָָדָם אֵלָא לְיָדְךָ הַמְלֵאָה וְהָעֲשִׁירָה

כִּדְכַּתִיב שָׁוְא וּדְבַר כָּזָב הַרְחֵק מִמֶּנִּי

רֵאשׁ וָעֹשֶׁר אַל תִּתֶּן לִי הַטְרִיפֵנִי לֶחֶם חֻקִּי

וְכַתִיב וַתָּבוֹא אֵלָיו הַיּוֹנָה לְעֵת עֶרֶב וְהִנֵּה עֲלֵה זַיִת טֶרֶף בְּפִיהָ

וְכַתִיב מִזְמוֹר לְדָוִד יְוֹיָ רוֹעִי לֹא אֶחְסָר

אָנָּא יְוֹיָ, הַזְמֵן לִי עֲסָק בְּנַחַת וְלֹא בְּצַעַר, בְּהֶתֵּר וְלֹא בְּאִסּוּר

כְּדֵי שֶׁאַרְוִיחַ עַל יְדֵי הַעֲסָק פַּרְנָסָתִי (וּפַרְנָסַת אַנְשֵׁי בֵּיתִי)

כִּי אֵין מַעֲצָר לְפָנֶיךָ לְהוֹשִׁיעַ בְּרָב אוֹ בִּמְעָט

כִּי אַתָּה נוֹתֵן לֶחֶם לְכָל בָּשָׂר כִּי לְעוֹלָם חַסְדּוֹ

וְכַתִיב פּוֹתֵחַ אֶת יָדֶיךָ וּמַשְׂבִּיעַ לְכָל חַי רָצוֹן

אָנָּא יְוֹיָ אַל תִּגְמוֹל לִי כְּפָעֳלִי וְאַל תְּדִינֵנִי כְּמַעֲלִי

אִם גָּבְרוּ חֲטָאַי, חֲסָדֶיךָ מֵעוֹלָם גָּבְרוּ וְעַצְמוּ רָאשֵׁיהֶם

וּבַמֶּה יִוָּדַע אֵפוֹא רַחֲמֶיךָ וַחֲסָדֶיךָ

כִּי אִם בַּעֲשׂוֹתְךָ חֶסֶד עִם שֶׁאֵינָם רְאוּיִים וַהֲגוּנִים

הַיּוֹם (הַלַּיְלָה) עַבְדְּךָ בֶּן אֲמָתְךָ הָעוֹמֵד בִּתְפִלָּה לְפָנֶיךָ

כִּי אֲנִי יוֹדֵעַ בְּעַצְמִי שֶׁאֵין אֲנִי כְּדַאי לְבַקֵּשׁ מִלְּפָנֶיךָ

כִּי אֵין לִי עַל מַה לְהִשָּׁעֵן אוֹ לְהִסָּמֵךְ, אֲבָל עַל רַחֲמֶיךָ הָרַבִּים נִשְׁעַנְתִּי

שֶׁלֹּא תִּתְנַהֵג עִמִּי כַּעֲווֹנוֹתַי וְכַחֲטָאַי וּכְפִשְׁעִי

שֶׁחָטָאתִי וְשֶׁעָוִיתִי וְשֶׁפָּשַׁעְתִי לְפָנֶיךָ

כִּי אַתָּה חוֹקֵר לֵב וּבוֹחֵן כְּלָיוֹת

וְגָלוּי וְיָדוּעַ לְפָנֶיךָ, שֶׁרְצוֹנִי לַעֲשׂוֹת רְצוֹנֶךָ אֶלָא שֶׁהַשְּׂאוֹר שֶׁבָּעִיסָה מְעַכֵּב

לָכֵן, יְוֹיָ, אֱלֹהַי, הָאֵר פָּנֶיךָ אֶל עַבְדְּךָ

וְהַזְמֵן לִי עֵסֶק שֶׁלְּפַרְנָסָה אֲשֶׁר בָּהּ אוּכַל לַחֲזוֹר בִּתְשׁוּבָה שְׁלֵמָה לְפָנֶיךָ

וַעֲשֵׂה לְמַעַן הַשֵּׁם הַגָּדוֹל דִּיקָרְנוּסָא

הַיּוֹצֵא מִפָּסוּק וַהֲרִיקוֹתִי לָכֶם בְּרָכָה עַד בְּלִי דַי

וּמִפָּסוּק נְסָה עָלֵינוּ אוֹר פָּנֶיךָ יְוֹיָ

שֶׁתַּשְׁפִּיעַ לָנוּ בְּמִדַּת טוּבְךָ

(לְקַדֵּשׁ עַצְמִי בַּתְּחִלָּה לִשָּׂא אִשָּׁה

כְּדֵי שֶׁאֶלְמַד תּוֹרָה בְּטַהְרָה וְאֶתְפַּלֵּל בִּקְדֻשָּׁה

וַאֲתַקֵּן מִצְווֹתַי וְאֶזְכֶּה לָשׁוּב בִּתְשׁוּבָה

וּלְטַהֵר הָעוֹלָמוֹת הָעֶלְיוֹנִים אֲשֶׁר פָּגַמְתִּי בַּעֲווֹנוֹתַי הָרַבִּים)

אָנָּא יְוֹיָ, אַל תְּשִׁיבֵנִי רֵיקָם מִלְּפָנֶיךָ

וְאַל תְּבִישֵׁנִי מִשִּׂבְרִי שֶׁאֶזְכֶּה וְאֶחֱזוֹר לְמוּטָב

וְאַל אֶטָּרֵד וְלֹא אֵצֵא מִן הַכְּלָל וְלֹא אֶפָּטֵר מִן הָעוֹלָם (הַזֶּה) בְּלֹא תְשׁוּבָה

כְּמַאֲמַר דָּוִד נְעִים זְמִירוֹתֶיךָ:

אַל תֶּאֱסוֹף עִם חַטָּאִים נַפְשִׁי וְעִם אַנְשֵׁי דָּמִים חַיַּי

אָנָּא יְוֹיָ, הַרְוִיחַ לִי כְּדֵי שֶׁיִּהְיֶה לִי פְּנָאי מִטִּרְדַּת פַּרְנָסָתִי

וּתְשׁוּקָתִי לְחַדֵּשׁ בַּתּוֹרָה וְלֵירֵד לְעוֹמֶק הַהֲלָכָה

וְלִהְיוֹת מִמְּשִׁיבֵי מִלְחֶמֶת הַתּוֹרָה לִשְׁמָהּ וְלָלֶכֶת בְּדַרְכֵי יְשָׁרִים

אָנָּא יְוֹיָ, צָרֵף מַחְשַׁבְתִּי הַטּוֹבָה לַמַּעֲשֶׂה וְהָפֵק שְׁאָלַָתִי

כִּי מַה לְּךָ לִרְדּוֹף קַשׁ נִדָּף, צֵל עוֹבֵר

וְלַהֲבִיאוֹ בְּתַנּוּר עָשָׁן וְלַפִּיד אֵשׁ וּגְזֵרוֹת וְדִינִים

הֲלֹא אַתָּה נִקְרֵאתָ אֶרֶךְ אַפַּיִם וְרַב חֶסֶד וּמַרְבֶּה לְהֵיטִיב

רוֹצֶה בִּתְשׁוּבָתָן שֶׁלָּרְשָׁעִים וְאֵין אַתָּה חָפֵץ בְּמִיתָתָן

הֲשִׁיבֵנִי אֵלֶיךָ וְאָשׁוּבָה וְחַזֵּק יְמִינִי

כִּי אַתָּה מָעוֹז לָאֶבְיוֹן וּמְחַזֵּק יָדַיִם רָפוֹת וּבִרְכַּיִם כּוֹשְׁלוֹת תְּאַמֵּץ

יִהְיוּ לְרָצוֹן אִמְרֵי פִי וְהֶגְיוֹן לִבִּי לְפָנֶיךָ יְוֹיָ צוּרִי וְגוֹאֲלִי.
