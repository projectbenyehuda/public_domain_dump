חֲדַל לָךְ

חתימה: אלשבזי


חֲדַל לָךְ דָּל / אָסִיר נִבְדָּל / וְאַל תִּפּוֹל בְּתַרְדֵּמָה

וְהִתְחַנֵּן / וְהִתְלוֹנֵן / בְּמִגְדַּל עוֹז וְעִיר חוֹמָה

וְהַנְעִימָה / וְהַשְׁלִימָה / לֵב וְנֶפֶשׁ שׁוֹמֵמָה

בְּפִתְחוֹן פֶּה / לְדוֹד יָפֶה / דְּרוֹשׁ אוּלַי יְקוֹמֵמָה

חֲבַצֶּלֶת / אֲשֶׁר גָּלַת / בְּרוֹב חַסְדּוֹ יְרַחֵמָה

יְמִין יְוֹיָ עוֹשָׂה חַיִל יְמִין יְוֹיָ רוֹמֵמָה:


אֲיֻמָּתִי / לְעֶזְרָתִי / הִכּוֹנִי דְרָכַיִךְ

וְהָצִיצִי / וְהָלִיצִי / לְלַמֵּד זְכוּת לְבָנַיִךְ

לְמוּל דּוֹדֵךְ / צְבִי חֶמְדֵּךְ / הֲגִי קוֹל תַּחֲנוּנַיִךְ

וְאִסְפִי עָם / אֲשֶׁר הוּעָם / בְּאֵשׁ אַהֲבַת רְשָׁפַיִךְ

וְעֵת דּוֹדִים / בְּחֵן יַקְדִּים / לְנִדְכָּאִים יְנַחֵמָה:


לְבֵן נִכְאָב / יְרַחֵם אָב / הָרַחֲמָן לְהַצִּילוֹ

יְשַׂמַּח לֵב / מִתְעַלֵּב / מִתִּגְרַת מַבְהִילוֹ

בְּיוֹם גָּזַר / וְנִתְאַכְזַר / וְדָלַק בַּהֲמוֹן חֵילוֹ

אֲדוֹן הַכֹּל / בְּרָע סִכֵּל / עֲצַת רָשָׁע וְהִכְשִׁילוֹ

עֲדֵי יִמַּק / בְּתוֹךְ עֵמֶק / יְהוֹשָׁפָט בְּיוֹם נָפְלוֹ

וְאַכְלוּסָיו / וְכָל סוּסָיו / יִרְדְּפֵיהֶם אַף וְחֵמָה:


שַׁוְעַת דָּל / נִתְדַּלְדָּל / יַאֲזִין צוּר לְכָל נוֹצָר

אֲשֶׁר שָׁקַל / בְּרֶגַע קַל / בְּפֶלֶס הַרְרֵי מִבְצָר

וְיַצִּילָה / לְצוֹלֵלָה / קוֹרְאָה מִן הַמֵּצָר

וְיַצִּיבָה / וְיַשְׂגִּיבָה / וְיוֹשִׁיעַ לְרָשׁ נֶעְצָר

וְיוֹפִיעַ / לְהַרְגִּיעַ / לְבַת נָדִיב דּוֹמֵמָה:


בִּינוּ זֹאת / וּלְהַזּוֹת / לְבַבְכֶם אוֹהֲבַי שִׁיתוּ

וּבִתְשׁוּבָה / אֱנוֹשׁ יוּבָא / שְׂאוֹר וְחָמֵץ תַּשְׁבִּיתוּ

וְלִגְוִיָּה / תְּהֶא מִחְיָה / שְׂמֹאל מַבְאִישׁ הַצְמִיתוּ

וְיִדָּחֵף / בְּרָע יָחֵף / וְכָל חֵילוֹ יַרְתִּיתוּ

וְיַלְדֵי תָם / בְּאַהְבָתָם / בְּבֵית אֵל יָבִיתוּ

וְיִתְוַדֶּה / אֱוִיל מוֹדֶה / בְּלֵב נִשְׁבָּר כְּפוּף קוֹמָה:


זְכוּת תּוֹרָה / תְּבַשֵּׂרָה / שְׂרִידֵי עָם בְּמַקְהֵלוֹת

בִּסְלִיחָה / תַּצְלִיחָה / עֲדַת קֹדֶשׁ בְּחֵן לַעְלוֹת

לְהַר הַמּוֹר / בְּשִׁיר מִזְמוֹר / שִׁיר חָדָשׁ וּתְהִלּוֹת

וְאָז יַחְכִּים / צְעִיר מַשְׁכִּים / וְגַם מַעְרִיב בִּתְפִלּוֹת

וְיִמָּצֵא / כָּל חֶפְצוֹ / בְּנִסְתָּרוֹת וְהַנִּגְלוֹת

וְחֵיל יִבּוֹז / וְלֹא יָבוֹז / עֲנִיֵּי עָם בְּמַשְׂטֵמָה:


יְשׁוּעַת בֵּן / בְּסוֹד חֶשְׁבּוֹן / שְׁנֵי יוֹבֵל וְדָנִיאֵל

אֲבָל סָתוּם / בְּדָת חָתוּם / וְלֹא נִגְלָה לְכָל שׁוֹאֵל

וְכוֹכָב רַךְ / כְּבָר דָּרַךְ / כְּמוֹ מַקֵּל בְּחֶסֶד אֵל

וּבַאֲתַתְקַל / עֲלֵי עָב קַל / שְׁנַיִם כּוֹכְבֵי גוֹאֵל

בֶּן יוֹסֵף / וְיִנּוֹן סוֹף / שְׁנֵי מַלְכֵי יִשְׂרָאֵל

וּבִתְשׁוּבָה / דּוֹדִי בָּא / וְיִבְנֶה עִיר וְגַם חוֹמָה:
