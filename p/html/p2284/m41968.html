[יָסַד יְסוֹד מְיֻסָּד רַב וָרָהַב] (שִׁיר יְדִידֹת)

שִׁיר יְדִידֹת אֶל־בָּרוּךְ הַכֹּהֵן


- - - - - - - (-)

יָסַד יְסוֹד מְיֻסָּד רַב וָרָהַב

אֲדוֹן הַכֹּל וְשָׂם חֶבְיוֹן עֻזּוֹ

אַדֶּרֶת הַחָכְמָה וּלְשׁוֹן זָהָב

וּמִי יוּכַל סַפֵּר הוֹד עֱזוּזוֹ

זֵכֵר נִפְלְאוֹתָיו אֲשֶׁר קָנָה

בְּשִׁיר וּבְקוֹל זִמְרָה שָׂם מָעֻזּוֹ.

חֵךְ וְלָשׁוֹן אֵין לִי לְרַנֲנָה

קְצָת תְּהִלּוֹתָיו וְסוֹד דָּתוֹ

לָכֵן עִמִּי הָרִים פִּצְחוּ רִנָּה.

נְעִים זֶמֶר רֵאשִׁית לְרֵאשִׁיתוֹ

אֶחָד מְשׁוֹרֵר הוּא וְאֵין שֵׁנִי

מְאוֹר עוֹלָם כְּשֶׁמֶשׁ בְּצֵאתוֹ.

נָא דְבָרַי אֶפְנֶה מוּל אֲדֹנִי

בִּשְׁפַל מְלִיצָתִי הַצְּעִירָה

וּמִפְּשָׁעַי נַקֵּה תְנַקֵּנִי.

יְקָרָה הָיְתָה־לִּי הַתְּשׁוּרָה

כְּעַל־כָּל־הוֹן שָׂמַחְתִּי בְקִרְבִּי

עָלָה בְאַפִּי רֵיחַ קְטֹרָה.

הִשְׂכַּלְתָּ־נָּא עֲשׂה חָפַצְתָּ בִּי

עַל־מְלָשְׁנֵי נָשִׁים לִפְרֹשׂ מִכְמָרֶת

קָנִיתָ כִלְיֹתַי וְגַם־לִבִּי

וּמִי יִתֵּן יָדִי לָהֶם גּוֹבָרֶת

לְיַד נָשִׁים אֶמְכֹּר אֶבְרֵיהֶם

הַלֵּב וְהַכָּבֵד וְהַיֹּתָרֶת.

דְּרֹךְ קַשְׁתָּךְ תִּלְעַג עַל־דִּבְרֵיהֶם

הַחֲזֵק בְּצִנָּה וּלְבַשׁ שִׁרְיוֹן

שְׁפֹךְ חֵמָה הַחֲרֵשׁ בַּדֵּיהֶם.

הֵן רַק לְאִישׁ אֶחָד אֵין לוֹ דִמְיוֹן

אֲשֶׁר מְיֻחָד הוּא בְּאַחְדּוּתוֹ

יָרֵאתִי גֶשֶׁת בְּסֵפֶר גִּלָּיוֹן.

אָמַר עִמָּנוּאֵל בִּמְלִיצָתוֹ:

אִשָּׁה סִבַּת הֶעְדֵּר הַקִּנְיָנִים

מָצָא אִשָּׁה מָצָא רֹב חֶרְפָּתוֹ.

לְפַח וּלְצוּר מִכְשׁוֹל וְלִצְנִינִים

שָׂם נֵזֶר הָאִשָּׁה זֶה לְמָנָה

לֹא חָס עַל־חֵץ גָּזַר הַסִּימָנִים.

טָמַן חֶבְלָן בְּשֵׁפֶל הַתַּחְתּוֹנָה

עַל־זֶה הוֹמִים עָלַי לֵב וּבְשָׂרִים

כִּי אֵין אָדָם שָׁלֵם בִּלְתִּי גִנָּה.

רוֹם יָדֶיךָ תִּשָּׂא כְגִבּוֹרִים

קוּמָה עָזְרָה כִּי אֵין בִּלְתָּךְ גּוֹאֵל

בְּאִשָּׁה זְרֹק מַיִם טְהֹרִים.

יִתֶּן לָךְ צוּר עֶלְיוֹן כֹּחַ וְלָאֵל

לִשְׁקֹד יוֹמָם וְלַיִל דַּלְתֵי חָכְמָה

יָרוּם חֶלְקָךְ כְּרַבָּן גַּמְלִיאֵל

נֹעַם תָּמִיד תִּרְדֹּף בְּכָל־עָצְמָה

וְהַשְׁמִיעֵנִי טוּב בְּרִיאוּתְךָ

חָפַצְתִּי זֶה מִכָּל־בִּגְדֵי רִקְמָה.

וְיִיטַב־נָא לִפְנֵי מַעֲלָתְךָ

מִצִּדִּי בִּכְנִיעוּת תִּשְׁתַּחֲוֶה

לְגִבּוֹר מְנַצֵּ"חַ אִישׁ טוֹבָתְךָ.

חֲשַׁקְתִּיךָ אוֹהֵב וַאֲקַוֶּה

תְּכַסֶּה אַהֲבָה עַל־פְּשָׁעִים

יָדַעְתִּי עַל־פָּנַי זֶה הוּא מַסְוֶה

זְכֹר בְּרִית אֲהוּבִים וְרֵעִים

צַוֵּנִי וְתֵעָשׂ כִּי יִיטַב לִי

לְשָׁרֶתְךָ אֲדֹנִי הַנָּעִים.

קַמְתִּי בִּשְׁתֵּי שׁוּרוֹת כְּמִעוּט שִׂכְלִי

אֲהַבְתִּיךָ מִפָּז וְיַהֲלוֹם

זֶה־שְּׁמִי לְעֹלָם וְזֶה זִכְרִי

בָּרוּךְ מַדָּא לְבִישׁ דּוֹבֵר שָׁלוֹם.
