תְּמַהּ אָחִי עֲלֵי יָדְעִי מְשָׁלִים

תְּמַהּ אָחִי עֲלֵי יָדְעִי מְשָׁלִים / מְתוּקִים מִדְּבַשׁ וְנֹפֶת צוּפִים

וּמַעְשֵׂי הֵם אֲנִי אוֹמֵר לַמֶּלֶךְ / בְּדֵעוֹתָיו אֱמֶת טוֹבִים וְיָפִים

בְּצֵאתוֹ עִם צְבָאָיו וַחֲיָלָיו / וְכָל שָׂרָיו וּפַחוֹתָיו צְרוּפִים

הֲכִי מָצָא אֱנוֹשׁ זָקֵן וְקָשִׁישׁ / מְשַׁחֵר לַעֲבֹד אֶרֶץ עֲרָפִים

וְאָז שָׁאַל בְּעָרְמָתוֹ וְאֵיכָה / כְּמִתְעַצֵּל תְּהִי מוֹצִיא אֲלָפִים

הֲלֹא תַקְדִּים וְתַשְׁכִּים לַעֲבוֹדָה / וְכוֹכָבִים לְפָנֶיךָ רְצוּפִים

וְאָז עָנָה וְחֵי רֹאשָׁךְ אֲדוֹנִי / כְּבָר קַמְתִּי וְקִדַּמְתִּי נְשָׁפִים

אֲבָל לָקְחוּ אֲחֵרִים אֶת נְשָׁפַי / וְהוֹסַפְתִּי עֲלֵיהֶם עוֹד כְּסָפִים

וְהַמֶּלֶךְ בְּשָׁמְעוֹ נָם לְיָשִׁישׁ / דְּעֵה הִנֵּה מְנָתָךְ עִם חֲנֵפִים

וְהָלַךְ הוּא לְדַרְכּוֹ מַחֲנָיִם / וְאָז חָנָה כְאִישׁ עוֹשֶׂה כְשָׁפִים

בְּעֵת שָׁאַל סְגָנָיו הֵם וְשָׂרָיו / יְדַעְתֶּם שַׁאֲלוֹתַי הַתְּכוּפִים

לְעוֹבֵד הָאֲדָמָה שָׂב וְיָשִׁישׁ / וְהוּא רָזֶה וּמִשְׁמַנָּיו עֲיֵפִים

וְלֹא יָכְלוּ עֲנוֹת אוֹתוֹ כְּאִלּוּ / בְּשִׂכְלֵיהֶם וְדֵעוֹתָם טְרוּפִים

וְהוּא גָּזַר הֲלֹא אַתֶּם נְגִידִים / מְנָתִי אוֹכְלִים מֵאוֹת אֲלָפִים

וּמַדּוּעַ כְּבַשְׁתֶּם אֶת פְּנֵיכֶם / וְאִם לֹא תֵדְעוּ אַתֶּם הֲדוּפִים

כְּשָׁמְעָם חָגְרוּ עֻזָּם בְּשִׁלְחָם / אֱלֵי יָשִׁישׁ וְהָרָצִים דְּחוּפִים

בְּיָדָם מַתְּנוֹת כֶּסֶף וְזָהָב / וּמִגְדָּנוֹת וְאֵפוֹד עִם תְּרָפִים

בְּהַגִּיעָם אֱלֵי יָשִׁישׁ בְּבֵיתוֹ / וְהִשְׁתַּחְווּ וְכָרָעוּ כְּפוּפִים

לְמַעַן יִפְתְּרָה מִלִּין לְמֶלֶךְ / עֲצָרָם מִלְּפָנָיו וַעֲטוּפִים

וְהַזָּקֵן בְּשׁוּרוֹ מַתְּנוֹתָם / וּפָנָיו צָהֲבוּ כִּפְנֵי שְׂרָפִים

וְאָז הֵשִׁיב שְׁאָלַנִי אֲדוֹנִי / יְמֵי חָרְפִּי הֲלֹא בָנַי צְפוּפִים

וְאָנָה הֵם לְכוֹנֵן אֶת מְלַאכְתִּי / כְּמוֹ עָשׂוּ לְשָׁרָשִׁים עֲנָפִים

עֲנִיתִיו כֵּן אֲנִי הָאָב לְבָנוֹת / נְשָׂאוּם מִבְצְרֵי הָעֹז אֲפוּפִים

וְהַמָּנוֹת אֲשֶׁר גָּרַם בְּשִׂכְלוֹ / לְמַלֹּאת מַחֲסוֹרִי הֵן אֲסוּפִים

וְלָעֶרֶב בְּשַׂר אֵילִים אֲבוּסִים / וְלַבֹּקֶר וְהִיא עוּגַת רְצָפִים

וְאָז הָפְכוּ פְּנֵיהֶם הַשְּׁלוּחִים / לְהַגִּיד לַאֲדוֹנֵיהֶם שְׁטוּפִים

בְּהַגִּידָם בְּשִׂיחַ דִּבְּרוֹתָיו / וְנֶחְשָׁבוּ כְּאִלּוּ הֵם אֲשָׁפִים

וְלֹא יָדְעוּ הֱיוֹת מַלְכָּם כְּמַעְרִים / וְרַעְיוֹנָיו לְמֵרָחוֹק כְּצוֹפִים

לְהַצִּיל מֵהֲמוֹן הוֹנָם לְזָקֵן / אֲסָרוֹ הַזְּמָן כָּלוּא בְּכֵפִים.
