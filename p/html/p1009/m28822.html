בְּנֵי מִשְׂרָה קְחוּ מֶנִּי רְנָנִים

בְּנֵי מִשְׂרָה קְחוּ מֶנִּי רְנָנִים / וְתוֹכָחוֹת לְנָחִים שַׁאֲנַנִּים

לְעוֹרֵר הַמְזִמּוֹת הַהֲגוּנוֹת / מְשָׁלִים עִם מְלִיצוֹת נֶאֱמָנִים

שְׁנוּ בָהֶם וְשַׁלֵּשׁוּ לְחָקְרָם / וְאַל תִּפְנוּ לְדִבְרֵי יִדְּעוֹנִים

בְּאִישׁ פֶּתִי חֲסַר לֵבָב לְתֻמּוֹ / וְלוֹ רַעְיָה כְּמוֹתוֹ אֵם לְבָנִים

יְלָדוֹ אָב בְּיַלְדוּתוֹ כְסִיל הוּא / וְהִתְעָרֵב לְקֵץ יָמָיו לְשׁוֹנִים

בְּמַרְאֵה הַחֲלוֹם עָמַד לְנֶגְדּוֹ / כְּאִישׁ חַרְבּוֹ שְׁלוּפָה צִיר אֱמוּנִים

אֲזַי נִבְעַת וְשָׁאַל לוֹ לְמִי אַתְּ / וְהוּא עָנָה לְאָדוֹן הָאֲדוֹנִים

שְׁלָחַנִי לְנַפְשֶׁךָ לְקַחְתָּהּ / וְהוּכַן לָךְ פְּצָעִים נֶאֱמָנִים

וְהוּא צָעַק בְּמַר אַל נָא תְמַהֵר / אֲדוֹנִי לַהֲמִיתֵנִי חֲרוֹנִים

עֲדֵי אַסְדִּיר לְפָנֶיךָ בְּמִלִּין / וְאוּלַי יֶהֱמֶה לִבָּךְ לְפָנִים

רְאֵה אָבִי רְאֵה רַחֵם בְּנוֹתַי / וּמִי זֶה יִמְצְאָה לָהֶם חֲתָנִים

וְהַבָּנִים בְּלִי אוֹנִים וְעָצְמָה / לְפַת לֶחֶם גְּדוֹלִים עִם קְטַנִּים

וְעַד עַתָּה אֲנִי צָעִיר לְיָמִים / עֲרֹב עַבְדָּךְ לְיָמִים אַחֲרוֹנִים

עֲדֵי אַשְׁלִים יְמֵי מִדַּת צְבָאִי / וּבָנַי אַחֲרַי יִהְיוּ דְשֵׁנִים

לְפִתּוּיָיו וְלַחְלִיקוּת אֲמָרָיו / אֲזַי הֵשִׁיב לְחֶרֶב אֶל נְדָנִים

בְּטֶרֶם יַהֲלֹךְ מַלְאַךְ אֱלֹהִים / תְּקָפָהוּ שְׁמַע מִלַּי נְכוֹנִים

בְּחַיֶּיךָ בְּבוֹאָךְ לַעֲשׂוֹת אֶת / שְׁלִיחוּתָךְ שְׁלַח אֵלַי לְפָנִים

לְמַעַן אֶהְיֶה מֵכִין לְנַפְשִׁי / לְכַפָּרַת פְּשָׁעִים וַעֲווֹנִים

וְהוּא קִבֵּל בְּעָרְמָתוֹ יְהִי כֵן / וְאִם עָלוּ בְּפָנָיו קִמְּשׁוֹנִים

וְהָאִישׁ בַּהֲקִיצוֹ שָׂח לְאִשְׁתּוֹ / וְהִנֶנָּה בְּאֵבֶל כַּיְּעֵנִים

וְלֹא יָדְעוּ לְמָתַי הַפְּרִידָה / וְהוּא נֶעְלָם בְּעֵינֵי הַהֲמוֹנִים

וְהֶאְרִיךְ הַזְּמָן שׁוּלֵי מְעִילוֹ / וְנִשְׁכָּחוּ סְאוֹנִים וַאֲסוֹנִים

וְלֹא פָנָה לְנֶגֶד יוֹם תְּמוּתָה / עֲדֵי הִרְחִיב לְיָתֵר עַל שְׁמֹנִים

וּמִלֵּא מִשְׁאֲלוֹת לִבּוֹ בְּתֵבֵל / וְאָז רָאָה בְּנֵי בָנִים אֱמוּנִים

לְקֵץ יָמִים וּבַעַל הַחֲלוֹם בָּא / לְפִתְרוֹן הַחֲלוֹמוֹת הַיְּשָׁנִים

וְעָמַד מֵעֲלֵי רֹאשׁוֹ לְהַשְׁבִּית / מְעוֹן רִבְצוֹ וְיִשְׁכֹּן בֵּית אֲבָנִים

וְהוּא עָנָה וְלָמָּה תֶּהְדְּפֵנִי / כְּאִישׁ נִדְהָם בְּטוּמְאַת רַעֲיוֹנִים

וְאַיֵּה פֹה שְׁלוּחֶיךָ בְּטֶרֶם / תְּבוֹאֵנִי וְאֶשְׁתֶּה רוֹשׁ פְּתָנִים

וְהַמַּלְאָךְ בְּקִצְפּוֹ וַחֲרוֹנוֹ / בְּמַעְנֵהוּ: וְלֹא כֵן בֶּן זְדוֹנִים

שְׁלוּחַי הֵם לְשִׁבְעָה נֶחֱלָקִים / דְּבוּקִים בָּךְ לְךָ שֹׁוִים וְנִמְנִים

תְּחִלָּתָם לְרוֹאוֹת בָּאֲרֻבּוֹת / וְהֵן כָּהוּ אֲשֶׁר הָיוּ כְּיוֹנִים

וְאָזְנֶיךָ לְקוֹל שׁוֹפָר כְּאַיִן / וּמִקּוֹדֵם בְּלַחַשׁ מַאֲזִינִים

וְשִׁנֶּיךָ כְּבָר נָשְׁרוּ בְּפִיךָ / לְיָמִים שֶׁכְּבָר שָׁחְקוּ אֲבָנִים

שְׂעָרוֹתָךְ שְׁחוֹרוֹת כַּאֲפֵלָה / כְּמוֹ עוֹרֵב וְהֵם עַתָּה לְבָנִים

וְקוֹמָתָךְ זְקוּפָה הִיא כְּאַגְמוֹן / וְנִכְפָּפָה לְאֹרֶךְ הַזְּמַנִים

וְרַגְלֶיךָ שְׁתַּיִם מִלְּפָנִים / וְהֵן עַתָּה לְשָׁלֹשׁ הֵם נְתוּנִים

וְהוּפָרָה אֲבִיּוֹנָה בְּגוּפָךְ / וְעֵדֶיךָ אֱמֶת בַּחוּץ וּבִפְנִים

וְהֶן אֵלֶּה קְצָת עֵדַי לְנֶגְדָּךְ / וְכָהֵנָה יְדוּעִים לַמְּבִינִים

וּמָה אוֹסִיף עֲלֵי אֵלֶּה שְׁלוּחִים / לְגַלּוֹת אָזְנְךָ קוֹרְאִים וְעוֹנִים

וְאָז שָׁלַף לְחַרְבּוֹ מִנְּדָנָהּ / דְּקָרָהוּ וְדִמְעוֹ כַעֲנָנִים

וְהָלַךְ לַאֲבַדּוֹן סָר וְזָעֵף / וְיָמָיו הֵם כְּצֵל אוֹ כַעֲשָׁנִים

דְּעוּ שִׂיחִי וְשִׁירִי הַהֲגוּנִים / דְּעוּ שִׁירִי נְעָרִים עִם זְקֵנִים

לְמַעַן תֵּדְעוּ שַׁדּוּן לְפָנָיו / וְתָעִירוּ נְפָשׁוֹת הַיְּשֵׁנִים.
