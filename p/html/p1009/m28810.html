אֲשֶׁר כִּפַּת שְׁחָקִים לוֹ עֲלִיָּה

לטבריה המעטירה


אֲשֶׁר כִּפַּת שְׁחָקִים לוֹ עֲלִיָּה / פְּלִיאוֹתָיו בְּכָל קַצְוֵי נְשִׁיָּה

שְׁבָחָיו לֹא יְכִילוּן מַלְאֲכֵי רוּם / וְאַף כִּי אִישׁ בְּאֶרֶץ מַאֲפֵלְיָה

וְהוּא בָחַר בְּיִשְׂרָאֵל סְגֻלָּה / וְנַפְשׁוֹתָם לְמֵאָז הִיא קְנוּיָה

וּמִתֵּבֵל אֱמֶת בָּחַר בְּצִיּוֹן / לְמוֹשָׁב לוֹ וְלֹא נָטַשׁ טְבֶרְיָה

בְּיַעַן הִיא לְמִקְדָּשׁוֹ תְּמוּרָה / בְּיוֹם הוּדַשׁ וְיוֹם הֻכַּת שְׁאִיָּה

וְסַנְהֶדְרִין הֲלֹא גָלוּ לְתוֹכָהּ / וּמִתּוֹכָהּ לְמַכָּתִי רְטִיָּה

אֲרוּכָתָהּ אֱמֶת עָלְתָה וּמַרְפֵּא / לְחֶלְאָתָהּ כְּמֵאָז הִיא בְנוּיָה

עֲלֵי יַד הַחֲסִידִים הַקְּדוֹשִׁים / אֲשֶׁר עָמְדוּ לְסוֹעֵרָה עֲנִיָּה

וְהַכֹּהֵן שְׁמוּאֵל הוּא נְגִידָם / וְהַלֵּוִי וְגַם מֹשֶׁה גְדַלְיָה

וְאַבְרָהָם מְשֻׁלָּשִׁים שְׁמוֹתָיו / אֲשֶׁר שָׁלְחוּ לְכַוָּנָה רְצוּיָה

אֲשֶׁר הָפְכוּ נְאוֹת מִדְבָּר לְעֵדֶן / וְגַן בִּיתָן אֲשֶׁר הָיָה לְצִיָּה

צְמָחִים יָבְשׁוּ לוּלֵא זְרָמָם / לְגֶפֶן אֵל בְּאַרְבַּע הִיא זְרוּיָה

וְהוֹמִיָּה בְּגָלוּתָהּ וְדָמְתָה / כְּקִיר נָטוּי וְגָדֵר הַדְּחוּיָה

אֲצִילֶיהָ לְאַכְזָרִים מְסוּרִים / וְשׁוֹעֶיהָ בְּיַד אֻמָּה בְזוּיָה

וְלוּלֵא קִדְּמוּ אוֹתָהּ גְּאוֹנִים / כְּחָכְמָתָם אֲזַי הָיְתָה כְּעוֹטְיָה

וְלֹא בִינָה וְלֹא הַשְׂכֵּל בְּקִרְבָּהּ / וְאֵין דַּעַת וְאֵין חָכְמָה מְצוּיָה

וְהוֹסִיפוּ לְהָאִיר אֶת נְתִיבָהּ / מְתֵי שֵׂכֶל בְּכָל חֶמְדַּת שְׂכִיָּה

בְּהַדְפִּיסָם לְסִפְרֵי הַתְּבוּנָה / עֲדֵי שָׁבָה לְעֵין הַכֹּל גְּלוּיָה

בְּעֵט בַּרְזֶל וְעוֹפֶרֶת חֲקוּקִים / מְלֶאכֶת בִּין בְּתַחְבּוּלוֹת עֲשׂוּיָה

עֲדֵי אָמַר חֲסַר הַלֵּב לְדַעְתּוֹ / וּמִי יִתֵּן בְּלֵב הַיָּם אֳנִיָּה

וְלֹא יָדַע בְּחָכְמַת הָאֱלֹהִים / אֲשֶׁר הִקְדִּים לְנַחֵם הַשְּׁבוּיָה

עֲדֵי קָרְאָה לְמוֹנֶיהָ בְּגָרוֹן / דְּעוּ חֶלְקִי עֲדֵי כוֹסִי רְוָיָה

וְלָאָמָה מְשַׂחֶקֶת בְּכָל עֵת / וּמַה דּוֹדֵךְ וּמָה אִמֵּךְ לְבִיָּא

וְאֵל יַעֲנֶה בְּחִישׁ קוּמִי וְאוֹרִי / הֲלֹא זָרַח כְּבוֹד אוֹרֵךְ צְבִיָּה.
