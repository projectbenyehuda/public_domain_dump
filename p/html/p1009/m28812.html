אֲשֶׁר עוֹלֶה עֲלֵי קִירוֹת לְבָבִי

אֲשֶׁר עוֹלֶה עֲלֵי קִירוֹת לְבָבִי / בְּכָל רֶגַע וְאֶשְׁתַּעְשֵׁעַ בְּזִכְרוֹ

שְׁמוֹ תִשְׁעִים וְדַי לַיּוֹדְעִים חֵן / וְהָאֶלֶף קְחָה לָךְ מַעֲשֵׂרוֹ

וְקָב חֶצְיוֹ בְּמִדָּה הָאֲמִתִּית / מְלֵאָה דַר וְסוֹחֶרֶת לְנִזְרוֹ

וְשֵׁם אָבִיו אֲבִי יִשַׁי בְּשִׁתּוּף / חֲצִי הַשֵּׁם אֲשֶׁר הָלַךְ לְאוֹרוֹ

וְהוּא לָבַשׁ צְדָקוֹתָיו כְּשִׁרְיוֹן / מְלַאכְתּוֹ הִיא בְּכָל עָרֵי מְגוּרוֹ

אֲנִי אֶהְיֶה מְיַחֵל לַשְּׁיַרוֹת / לְאִמְרָתוֹ וְעַד עֵת בּוֹא דְבָרוֹ

וְאֶתְאַו לִרְאוֹת שִׁירוֹת כְּתָבוֹ / אֲשַׁנֵּן בָּם וְאָרִים קוֹל בְּשִׁירוֹ

וּמֵרֹב הַנְּדוֹד אֶהְיֶה בְּעֵינַי / כְּמוֹ גֶבֶר אֲשֶׁר יַיִן עֲבָרוֹ

וְאִם אֶדְאַג לְקוֹרוֹת הַזְּמַנִּים / וְלִבִּי יֶהֱמֶה מִמַּאֲסָרוֹ

הֲכִינוֹתִי בְרֻדִּים וַעֲקֻדִּים / וְכָל טָלוּא וְנָקוֹד עַל שְׂכָרוֹ

וּבוֹ אֶשְׁכַּח לְעָנְיִי עִם מְרוּדִי / וְלֹא יֻתַּן סְגוֹר זָהָב מְחִירוֹ

וְאָגִילָה וְאֶשְׁתַּעְשַׁע וְאֶשְׂמַח / כְּשִׂמְחַת אִישׁ בְּהִוָּלֵד בְּכוֹרוֹ

וְאֶהְגֶּה בוֹ שְׁחָרִים וַעֲרָבִים / לְאַהֲבָתוֹ אֲלַחֵךְ אֶת עֲפָרוֹ

וְשָׂמַחְתִּי בְאִמְרָתוֹ כְּשִׂמְחַת / זְאֵב גָּנַב לְמוֹ טָלֶה בְּעֶדְרוֹ

וְלֹא אֶמְצָא לְדִמְיוֹן הַיְּדִיד זֶה / וְכָלִיל הוּא יְדִידִי בַהֲדָרוֹ

וְלָמָּה הַזְּמָן יֵלֵךְ שְׁחוֹחַ / קְדוֹרַנִּית וְלֹא יָאִיר בְּנֵרוֹ 

הֲלֹא קָם לַזְּמָן מֵאִיר בְּתֵבֵל / הֲלֹא יִכְרַע וְיִקֹּד אֶל גְּבִירוֹ

וּמִי יִתֵּן וְאָעוּפָה כְּנֶשֶׁר / אֱלֵי יָדִיד לְמוֹלַדְתּוֹ וְעִירוֹ

וּמִי יִתֵּן וְאַחֲלַי גַּם וְאַשְׁרַי / וְאוֹסִיף אֶשְׁאֲבָה מֵימֵי מְקוֹרוֹ

וְאִם אֵלֵךְ בְּאַרְבַּע הַקְּצָווֹת / בְּכָל מָקוֹם אֲנִי תָמִיד אֲסִירוֹ 

וְאַל יִתְמַהּ אֱנוֹשׁ עַל טוּב לְשׁוֹנִי / בְּעָבְרוֹ הַגְּבוּל הוֹסִיף בְּגִדְרוֹ 

אֱמֶת כִּי אִישׁ לְפִי שִׂכְלוֹ יְהֻלָּל / וְאִם הִרְבָּה וְאִם הִמְעִיט בְּסִדְרוֹ

וְכָל תָּם יַהֲלֹךְ עִמּוֹ וְיָשָׁר / לְזֶה נִקְרָא אֱמֶת צַדִּיק בְּדוֹרוֹ

לְזֶה רָאוּי אֱנוֹשׁ לָרוּץ לְפָנָיו / הֲלֹא הוּא לַזְּמָן שִׁמְשׁוֹ וְסַהֲרוֹ

אֲבָרֵךְ הַזְּמָן עַל טוּב יְלִידָיו / בְּרוּכָה הָאֲדָמָה בַּעֲבוּרוֹ

אֱלֹהִים חַי יְחָנֵּהוּ בְּכָל עֵת / וּמַלְאָכָיו יְצַו תָּמִיד לְעָזְרוֹ 

וְחַזֵּק אֵל זְרוֹעוֹתָיו וְכֹחוֹ / וְתָמְכֵהוּ וְלֹא תִמְעַד אֲשׁוּרוֹ

וְיִשְׂמַח כָּל יְמֵי חֶלְדּוֹ בְּחַיִּין / וְלֹא יִדְאַג וְלֹא יִכְאַב בְּשָׂרוֹ

בְּעֶזְרַת אֵל יְזַכֵּנִי אֱלֹהִים / רְאוֹת פָּנָיו וְאֶשְׁכֹּן בַּחֲצֵרוֹ

וְנַעֲלֶה אֶל פְּאַת צִיּוֹן חֲבֵרִים / נְקַוֶּה לַחֲזוֹת מִשְׁכַּן דְּבִירוֹ

וְאַשְׁרֵי אִישׁ אֲשֶׁר יָשִׂים אֱלֹהָיו / בְּכָל כִּסְלוֹ וּמִבְטַחוֹ וְשִׂבְרוֹ.
