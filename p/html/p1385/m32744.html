ויכוח על תכונותיהן המסתיים בנצחונו של ״איש השיר״

נְאֻם לְמוּאֵל בֶּן־אִיתִיאֵל. בָּאתִי אֶל בֵּית הַמִּדְרָשׁ / אֲשֶׁר שָׁם כָּל רָז וּדְבַר תְּבוּנָה פֹּרַשׁ / וְהָיָה פָּתוּחַ לְכָל דְּבַר חָכְמָה אוֹגֵר / הָאֶזְרָח וְהַגֵּר / וְהֶחָרָשׁ וְהַמַּסְגֵּר / וְהָיוּ מְדַבְּרִים בְּמִבְחַר אֲמָרִים / וּבְחֶמְדַּת דְּבָרִים / וַיְלַקֵּט פְּנִינִים / מִדִּבְרֵי חֲכָמִים וּנְבוֹנִים. / וְאִם מִעֲטוּ וְעוֹרָם מֵעֲלֵיהֶם / וַיִּקַּח כָּל אֶחָד מֵהֶם / כָּל טוּב מִלָּה / וְהִפְשִׁיט אֶת הָעוֹלָה / וַיֹּאמְרוּ: אֵי־זֶה מֵהַדְּבָרִים יִכְשַׁר / שֶׁיְּדַבֵּר בּוֹ אִישׁ וְאֻשַּׁר. / וַיֹּאמֶר הָאֶחָד: מְעַט מִדָּבָר / וְעִנְיָן רַב בּוֹ מְחֻבָּר. / וַיֹּאמֶר הָאֶחָד: מְדַבֵּר אִישׁ עָנָיו / דָּבָר דָּבוּר עַל אָפְנָיו / יֵשׁ לָשׁוֹן מִלָּה עֲרֵבָה מְדַבֶּרֶת / וְתַחְתֶּיהָ תַּעֲמֹד הַבַּהֶרֶת. / יֵשׁ מִלָּה נֶאֱמֶרֶת בְּיוֹם פְּקִידָה / טוֹבָה מִכָּל כְּלִי חֶמְדָּה / וְכָל הַמְּדַבֵּר בְּתַאֲווֹת לִבּוֹ / דְּבַר בְּלִיַּעַל יָצוּק בּוֹ. / מִדְבָּר הַכְּסִיל יַבְאִישׁ יַבִּיעַ / וְהַמַּשְׂכִּיל יַזְהִירוֹ כְּזֹהַר הָרָקִיעַ / וְהָאֱוִיל כִּי יִפְתֶּה וְדָבָר דִּבֵּר / לֹא לִזְרוֹת וְלֹא לְהָבֵר / וְהַמַּשְׂכִּיל לֹא יְדַבֵּר דָּבָר, כִּי אִם בְּשִׂכְלוֹ יְפָאֲרֵהוּ / וּבַמְּחוּגָה יְתָאֳרֵהוּ / מְקוֹר שִׂכְלוּת וְעֻלּוֹ / בַּעֲלִיַּת הַמִּקְרֶה אֲשֶׁר לוֹ. / לְשׁוֹן חָכָם מִכָּל רַע נְצוּרָה / עֲרוּכָה בַכֹּל וּשְׁמוּרָה. / בִּשְׁלֹשָׁה דְּבָרִים אִמְרַת חָכָם אֲמוּרָה: / בַּמִּדָּה בַּמִּשְׁקָל וּבַמְּשׂוּרָה. / כָּל דִּבְרֵי חֲכָמִים שְׁקוּלִים כְּאֶחָד / בְּמֹאזְנַיִם יִשְׂאוּ יָחַד. / וַיְדַבְּרוּ בְּדִבְרֵי הַשִּׁיר / וּבְיוֹדֵעַ סֵפֶר וּדְבָרָיו בְּמִשְׁקָל מֵשִׁיר / וַיֹּאמְרוּ: חָכְמַת חָכָם בְּלִי מִשְׁקָל וּמִדָּה עֲרוּכָה / הִיא עוּגָה בְּלִי הֲפוּכָה. / וְשָׁם נִקְרָא [אִישׁ] דְּבָרִים / וַיָּרֶב בַּשָּׁר בַּשִּׁירִים / וַיֹּאמַר: מִי הַשִּׁיר וּמִי עִנְיָנוֹ / הֲלֹא שֶׁקֶר בִּימִינוֹ / הִנֵּה מֵיטַב דִּבְרֵי שִׁירָה / דַּבֶּר עֹשֶׁק וְסָרָה / הֲלֹא בָּתֵי הַשִּׁיר לְאַכְזָב / שָׁוְא וּדְבַר כָּזָב. / מְגַנְּבֵי נַפְשׁוֹת אֲנָשִׁים וְלִבּוֹתָם / בְּשִׁקְרֵיהֶם וּבְפַחֲזוּתָם. / וַיִּחַר אַף אִישׁ הַשִּׁיר בַּדְּבָרִים, וַיֹּאמַר: הַאַתֶּם תְּרִיבוּן בַּשִּׁירִים / הַשְּׁכוּנִים בְּבָתֵי הַחֲצֵרִים / וְאַתֶּם נְפוֹצִים עַל הֶהָרִים / וְאֵין בָּכֶם כִּי אִם גּוֹלֶה וְתוֹעֶה / כַּצֹּאן אֲשֶׁר אֵין לָהֶם רֹעֶה. / וְכֻלְּכֶם נִפְרָדִים אִישׁ מֵרֵעֵהוּ / אִישׁ לְבִצְעוֹ מִקָּצֵהוּ / וְאֶת הַנַּחְלוֹת לֹא חִזַּקְתֶּם / וְלַנִּשְׁבֶּרֶת לֹא חֲבַשְׁתֶּם / וַאֲנִי לְכָל שָׁמְעָם שִׁירוֹתַי עֲרֵבוֹת / כְּעֵדֶר הַקְּצוּבוֹת / וְהֵמָּה שְׁלֵמוֹת וּתְמִימוֹת / שֶׁכֻּלָּם מַתְאִימוֹת / וְאֶת הַנַּחְלוֹת אֲחֲלֹק וַאֲחַלְּצָה / אוֹסְפָה הַצּוֹלֵעָה וְהַנִּדָּחָה אֲקַבְּצָה / וּמִלּוֹתַי אֲבַקֵּר וְאֶדְרֹשׁ / וְלַנִּשְׁבָּר אַחֲבֹשׁ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וְאָסִיר מִמֶּנִּי כָל דְּבַר שָׁמִיר וְשָׁיִת

לְחַזֵּק אֶת בֶּדֶק הַבָּיִת

וְכָל דְּבָרַי בְּמֹאזְנֵי צֶדֶק וְאֵיפָה שְׁלֵמָה

וְאִתְּכֶם אֵיפַת רָזוֹן זְעוּמָה.


וַיַעַן אִישׁ דְּבָרִים וַיֹּאמַר: שֶׁקֶר אַתָּה דֹבֵר וְעַל תֹּהוּ אַתָּה בּוֹטֵחַ. הַקֵּץ לְדִבְרֵי רוּחַ וּמַה בַּמִּשְׁקָל בְּמֹאזְנֵי שִׁירֵיכֶם / וְאוּלָם אַתֶּם טוֹפְלֵי־שֶׁקֶר רֹפְאֵי אֱלִיל כֻּלְּכֶם. / הֲלֹא כָל דְּבַר שִׁיר חֲסַר תְּבוּנוֹת / וְחֶסְרוֹן לֹא יוּכַל לְהִמָּנוֹת / וַאֲנִי אִם אֲדַבֵּר דִּבְרֵי חָכְמָה אֲכַלֵּהוּ / מִקְצֵה גְּבוּל עִנְיָן עַד קָצֵהוּ / וּמִלָּתִי תַּגִּיד לָךְ כָּל לִבָּהּ / לֹא תֶחְסַר כֹּל בָּהּ / וְהִיא לַכֹּל בְּנַחַת נִשְׁמַעַת / וּלְבַב נִמְהָרִים יָבִין לָדַעַת / יָרוּץ כָּל פֶּתִי קוֹרֵא בְּמִלּוֹת נְכֹחוֹת / וּלְשׁוֹן עִלְּגִים תְּמַהֵר לְדַבֵּר צַחוֹת. / וְעִנְיְנֵי שִׁירֶיךָ רְחוֹקִים מִלִּבּוֹת אִישִׁים / אֲשֶׁא אָרְחוֹתֵיהֶם עִקְשִׁים./ וְהִנֵּה אִם רָחֲקוּ בָּתֵי הַמְּלָכִים וְהַבְלֵיהֶם / מִי־בַעַל דְּבָרִים יִגַּשׁ אֲלֵיהֶם?



וַיַּעַן הַשִּׁיר: בַּמִּשְׁקָל הוֹדוֹ / וּקְנֵה הַמִּדָּה בְּיָדוֹ / וַיֹּאמֶר: הֲלֹא תֵדַע רֹעַ שְׁבָרֶיךָ / לָמָּה תְּדַבֵּר עוֹד דְּבָרֶיךָ? / הֲלֹא יָדַעְתָּ כִּי כָּל הַדְּבָרִים יְגֵעִים / וּבַעֲלֵי הַשִּׁיר נְבוֹנִים וִידוּעִים. / שִׁירָתָם עֲרֵבָה שִׁירַת חֹפֶשׁ / תַּאֲוָה נִהְיָה תֶּעֱרַב לָנֶפֶשׁ / כַּחֲלִי כָתֶם כָּל מִדְבָּרֶיהָ / וְכַכַּלָּה תַּעְדֶּה כֵלֶיהָ. בַּשִּׁיר יָגִילוּ כָּל יְצוּרִים / בַּחוּרִים גַּם בְּתוּלוֹת / זְקֵנִים עִם נְעָרִים / כִּי הוּא מְחַבֵּשׁ אֲנוּשִׁים / וּמְחַזֵּק חֲלוּשִׁים / הַמְשַׂמֵּחַ אֱלֹהִים וַאֲנָשִׁים / וּמִנַּעַר וְעַד זָקֵן טַף וְנָשִׁים. / הַשִּׁיר יִכְתְבֵהוּ בִּסְפָרָיו / כִּי יִתְהַלֲלוּ עַל חֲבֵרָיו / וּבְיוֹם הַמִּשְׁתֶּה הֱבִיאַנִי הַמֶּלֶ[ךְ] חֲדָרָיו. / וַאֲחַזֵּק לֵב הַמֶּלֶךְ וְלֵב חֵילוֹ אֲשֶׁר סְבִיבוֹ / לַמִּלְחָמָה לָצֵאת וְלָבוֹא. / וּדְבָרְךָ לֹא טַעַם וְלֹא רֵיחַ / לֹא מַכְאִיב וְלֹא מְשַׂמֵּחַ / וְאֵין בּוֹ רְפוּאָה / לְרוּחַ נְכֵאָה / וּמִלֶּיךָ לֹא לָרִיב וְלֹא לְהַגְּעִיל. / הֶבֶל וְאֵין בָּם מוֹעִיל / לֹא תְנַחֵם אָבֵל לֹא תְאַמֵּץ לְבָבוֹ / וּבֵית הַמִּשְׁתֶּה לֹא תָבוֹא. / וְלָמָּה נִפְקְדוּ נִפְרְדוּ דְּבָרֶיךָ / וּמַדּוּעַ נָפוֹצוּ אֲמָרֶיךָ / הָרוּחַ עֲלֵיהֶם יֵעָרֶה / אֲשֶׁר זוֹרֶה בָּרַחַת וּבַמִּזְרֶה / לֹא שְׁקוּלִים וְלֹא מְדוּדִים / כִּי כֻלָּם יְחֵפִים וּקְרוּעֵי בְּגָדִים / וְעַת שׁוֹמְעֵיהֶם יַפִּילוּ תַּרְדֵּמָה / וּקְרָעִים תַּלְבִּישׁ נוּמָה. / כָּל הוֹגֶה בָּם יֶהְגֶּה אַךְ־נְכָאִים / כִּמְלַקֵּט שִׁבֳּלִים בְּעֵמֶק רְפָאִים.

וַיַּעַן אִישׁ דְּבָרִים וַיֹּאמַר: רַב לְךָ וְהֶרֶף מִשִּׁירוֹתֶיךָ / שָׁמַעְתִּי אֶת כָּל נְאָצוֹת שִׁירוֹתֶיךָ. / לָעַגְתָּ לִי וְהִתְלוֹצַצְתָּ / וְרֹאשׁ דְּבָרַי מָחַצְתָּ / וְאִם בַּאדוֹנֶיךָ הֲרִימוֹתָ יָד וְהֵנַפְתָּ / אֶת מִי חֵרַפְתָּ / וְאֶת מִי גִדַּפְתָּ? / וּתְהֹלֵל לְךָ בַּבָּתִים וּבַחֲצֵרִים / בְּשִׂמְחָה וּבְשִׁירִים. / הֲלֹא מִדְּבָרַי בָּנִיתָ בָּתִים וְחוֹמוֹת וּצְרִיחִים / וּמִגְדָּלוֹת וּמֶרְקָחִים / וַעֲלִיּוֹת מְרֻוָּחִים. / וְהַשִּׁירִים הֵם צוֹרֲרַי / מְגַנְּבֵי דְבָרַי / וּמִגְּנֵבַת אֲמָרַי / בָּנִיתָ בַּיִת וּתְחַזֵּק בֶּדֶק / הוֹי בוֹנֶה בֵּיתוֹ בְּלֹא צֶדֶק. / וּמִטּוּבוֹ מִלּוֹתַי גְבוּלוֹת תַּצִּיבוּ / הוֹי מַגִּיעֵי בַּיִת בְּבַיִת וְשָׂדֶה בְשָׂדֶה יַקְרִיבוּ. / וּתְחַבֵּר לְךָ שִׁירִים כְּזָבִים, עֵינֶיךָ בָּם תַּעְלִים / בְּיָדֶיךָ מֵתִים יַחְרִישׁוּ וַתִּלְעַג וְאֵין מַכְלִים. / וַתַּעַשׂ מִדְּבָרַי אֲשֶׁר הֵם מַחֲמַדִּים / בָּתִים לַבַּדִּים / וַתַּקְרֵב מִלַּת שֶׁקֶר לִרְעוּתָהּ / אִשָּׁה אֶל אֲחוֹתָה. / וְאֵיךְ תִּתְפָּאֵר עָלַי לֵאמֹר: מִלִּים אֶמְדֹּד בְּשָׁעֳלֵי רוּחַ מָלֵא יָבוֹא לִי.

וַיִּחַר אַף [אִישׁ] הַשִּׁיר, וַיָּקָם וַיִּקַּח דְּבָרִים מִטּוֹב וּמֵרָע מִשִּׂמְחָה וּמֵאֵבֶל / וַיְמַדֵּד בַּחֶבֶל / וַיֹּאמַר: הִנֵּה קַמְתִּי וַאֲלַקֵּט מִלִּים מִכָּל הַר וְגִבְעָה / מִכָּל עֵמֶק וּבִקְעָה / וּמִכָּל גֵּאָיוֹת וְכָל פֶּרֶץ / וְקִבַּצְתִּים מִיַּרְכְּתֵי אֶרֶץ. / וְלָקַחְתִּי כָל דְּבַר מִלָּה / כַּאֲשֶׁר הֹגֶה מִן הַמְסִלָּה / וְעָשִׂיתִי מֵהֶם רִנָּה / הָיְתָה לְרֹאשׁ פִּנָּה. / וַאֲחַפֵּשׂ דִּבְרֵי אַהֲבָה וְשִׂנְאָה וּבְחַנְתִּים כִּבְחֹן הַזָּהָב. / וָאֶקַּח דְּבָרִים פְּזוּרִים אֲשֶׁר הָיוּ לִי לְמִשְׁלַח רֶמֶשׂ וּלְמִרְמַס־שֶׁרֶץ / וְעָשִׂיתִי אוֹתָם גּוֹי אֶחָד בָּאָרֶץ / וְלִקַּטְתִּי מִלּוֹת מִפִּי כָּל אִישׁ אֲשֶׁר יְדַבֵּר בָּם לְכָל צְרָכָיו וּמַעֲנָיו / וְהַמִּתְפָּרְצִים מִפְּנֵי אֲדוֹנָיו / וְעָשִׂיתִי מֵהֶם שָׁלִישִׁים וּנְסִיכִים / וּסְגֻלּוֹת מְלָכִים. / וַאֲהַלֵּל בָּם סְגָנִים וּנְגִידִים / וַיִּהְיוּ עַל רְבִידִים וְעַל יְדֵיהֶם צְמִידִים / וּנְזָמִים עַל אָזְנַיִם / וְרֵיחַ מוֹר לְאַפַּיִם וְאַתָּה לָמָּה מִלֶּיךָ תְּשַׁבֵּר, הֲקָצוֹר קָצְרָה לְשׁוֹנְךָ / וּפִיךָ כָּבֵד וְהַדִּבֵּר אֵין בְּךָ / וְאִם יוֹעַצְךָ אָבַד, כִּי תָעִיק לְשׁוֹנְךָ בְּכָל מִלָּה / כַּאֲשֶׁר תָּעִיק הָעֲגָלָה / לָמָּה בֹשְׁשׁוּ אִמְרֵי פִיךָ / וְנוֹקְשׁוּ בְּכָל עֵת שְׂפָתֶיךָ / מַדּוּעַ אֶחֱרוּ פַּעֲמֵי מַרְכְּבוֹתֶיךָ? וְאוֹת כָּל לָשׁוֹן חוֹטֵב / כַּנָּהָר שָׁלוֹם וּכְנַחַל שׁוֹטֵף. / וַאֲנִי כָּל לְשׁוֹן הַשָּׁר כָּמוּס אָגוּר / כְּתֹר וְסִיס וְעָגוּר / יִתְפָּרְקוּ מִלַּי עַל לְשׁוֹנָם וְקֹלוֹתָם לֹא יִסָּפוּ / יָרוּצוּ וְלֹא יִכָּשְׁלוּ, יֵלְכוּ וְלֹא יִיעָפוּ.

וַיַּעַן אִישׁ דְּבָרִים וַיֹּאמַר: אָמְנָם הַשִּׁירִים יִתְהַלֲלוּ וּבַכְּזָבִים יִתְהֹלֲלוּ / וַיֵּלְכוּ אַחֲרֵי הַהֶבֶל וַיֶּהֶבָּלוּ. / וַיֹּאמְרוּ: אֲנַחְנוּ מְשַׂמְּחִים מְלָכִים / וּבָנוּ יָשִׂישׂוּ נְגִידִים וּנְסִיכִים / וְאִם תְּדַמּוּ, כִּי דְבָרִים דֹּבֵר תָּמִים עָבַר / לֹא כֵן הַדָּבָר / כִּי בָּחֲרוּ לָהֶם חֲלָקוֹת מִלּוֹת / וַיֶּחֱזוּ לָהֶם מַהֲתַלּוֹת / וַיְפַתּוּם בְּפִיהֶם / וַיַּתְעוּם כִּזְבֵיהֶם / וְיִשְׂמְחוּ בְּדִבְרֵי שִׁקְרֵיהֶם / בְּרָעָתָם יְשַׂמְּחוּ מֶלֶךְ וְשָׂרִים בְּכֹחַ שִׁירֵיהֶם / וְהִנֵּה בְּמִבְחַר הַשִּׁיר / אֲשֶׁר תָּשִׁיר / וּבְמֵיטַב מַחְבֶּרֶת אֲשֶׁר תְּחַבֵּר / שֶׁקֶר אַתָּה דֹּבֵר. / וַיִּקַּח הַשִּׁיר דְּבַר הַמְּלוּכָה / וַיַּעֲשֵׂהוּ פֶּסֶל מַסֵּכָה / וּמֵעֹשֶׁק מִלַּי בָּנָה בֵּיתוֹ וַיְכִינֵהוּ / בַּיִת גָּזַל וַיִּבְנֵהוּ / וַיִּבֶן הַבַּיִת וַיְכַלֵּהוּ. / עֹשֶׁק וּגְנֵבָה בְּשִׁירֵיכֶם / גְּזֵלַת הֶעָנִי בְּבָתֵיכֶם / כִּי אֵין עֵינֶיךָ וְלִבְּךָ כִּי אִם עַל מַאֲמָרַי / עַד אִם דִּבַּרְתִּי דְּבָרַי וּמִמֶּנִּי שִׁירֶיךָ / וּנְגִינוֹת אוֹהֲבֶיךָ / מְלֶאכֶת תֻפֶּיךָ וּנְקָבֶיךָ / תְּחַבֵּר מִלּוֹתַי וּתְנַגְּנֵם בִּנְבָלִים / וּבִמְנַעַנְעִים וּבְצִלְצְלִים / וְעַתָּה מַה לְּךָ וְלִמְחוֹז חֶפְצִי / כִּי בָאתָ אֵלַי לְהִלָּחֵם בְּאַרְצִי?

וַיָּקוּמוּ שָׂרֵי הַחֲכָמִים וְהַנְּבוֹנִים / וּנְגִידֵי הַשֵּׂכֶל וְהַסְּגָנִים / וַיִּגְעֲרוּ בַּמַּאֲמָרִים וְהוּכַח בַּדְּבָרִים. וַיֹּאמְרוּ: כָּל הַמִּלּוֹת / נִבְזוֹת וְנִקְלוֹת / וְהַדָּבָר אִם נַעֲלוּ עִנְיָנָיו וְאִם גָּבְהוּ / הֲלֹא דָּבָר הוּא / וְאִם יָקְרוּ נִפְרָדוֹת / וְלִמְאֹד יִהְיוּ חֲמוּדוֹת / עָבְשׁוּ פְרֻדּוֹת / וַיִּדֹּמּוּ בִּהְיוֹתָם נְפוּצוֹת / כְּאַבְנֵי גִיר מְנֻפָּצוֹת / לֹא נִכְּרוּ בַּחוּצוֹת / וַתִּהְיֶינָה לְמִדְרַךְ כַף רַגְלֵי כְסִילִים / תִּרְמְסֶנָּה רֶגֶל רַגְלֵי עַמִּי פַּעֲמֵי דַלִּים / וּמֵהֶם הַקֹּדֶשׁ מֵהַחֹל לֹא נִבְדַּל / לֹא נִכַּר שׁוֹעַ לִפְנֵי דָל. / וַיִּדְמוּ מְקוֹמוֹתָם טֶרֶם יִהְיוּ בְּרוּרִים וּבְמִשְׁקַל הַשִּׁיר צְרוּפִים / חֹרֵי עָפָר וְכֵפִים / וְלֹא כֵן דְּבַר הַשִּׁיר, כִּי בָּתָיו מֻקָּפִים בְּאַבְנֵי אֶקְדָּח / לְבַל יִדַּח מֶנְהוּ נִדָּח / וּמִלָּיו נִקְרְבוּ אַחַת אֶל אַחַת וַיִּתְאַחֲדוּ / וַיִּתְלַכְּדוּ וְלֹא יִתְפָּרְדוּ / וְהַדְּבָרִים אִם לֹא יִיעָדֵם הַשִּׁיר לֹא נִצְמְדוּ / הֲיֵלְכוּ שְׁנַיִם יַחְדָּו בִּלְתִּי אִם נוֹעָדוּ.

עַד אֲשֶׁר קָם הַשִּׁיר וַיֹּאמַר: כִּי מִמִּלּוֹת

יְקָרוֹת וְנִכְבָּדוֹת / אֶבְנֶה לִי בֵּית מִדּוֹת / וַאֲבִיאֵם בְּכוּר

מַחֲשָׁבָה וְחָנִיתִי לְבֵיתִי מִצָּבָה. וָאֲבִיאֵם בְּמִבְחַר

מֹאזְנַיִם / עַד אֲשֶׁר יֵצֵא הַדָּבָר מֵהַשְּׂפָתַיִם / מְזֻקָּק

שִׁבְעָתַיִם. / וָאֲשִׁיבֵם עַל מַעַרְכוֹת שִׁירָתוֹ / הַבְּכֹר

כִּבְכֹרָתוֹ / וַיִּקַּח כָּל מִלָּה אֲשֶׁר הָיְתָה לְקֶלֶס / וְשָׁקַל

בַּפֶּלֶס.

וַיֹּאמְרוּ הַדְּבָרִים לַשּׁוֹפְטִים: לָמָּה תִּשְׂאוּ פָּנִים? עַל שְׂפַת שֶׁקֶר תַּקְשִׁיבוּ / הֲלֹא דְבָרַי יֵיטִיבוּ. / אִמְרוּ לְמַהַלְלֵי הַשִּׁירָה וּדְבָרַי לָעוּ / עַל־מָה אֲדָנֶיהָ הָטְבְּעוּ / הֲלֹא מֵאִמְרוֹתַי בָּחֲרוּ נִבְחַר כָּל אֹמֶר / וַיָּמֹדוּ בָעֹמֶר / הֲלֹא יְדַעְתֶּם כִּי בִּדְבָרִים דִּבְּרוּ נְבוֹנִים / לְיָמִים קַדְמוֹנִים / וְהֵמָּה טוֹבִים מִשִּׁיר אֵצֶל הַחֲכָמִים / כִּי זְקֵנִים לְיָמִים. / וּנְבִיאֵי הַשִּׁיר תְּמוֹל נִבָּאוּ / חֲדָשִׁים מִקָּרוֹב בָּאוּ / וְקֶדֶם לֹא מָצָא שִׁירִים בְּשִׂפְתּוֹתֵיכֶם / אֲשֶׁר לֹא שִׁעֲרוּם אֲבֹתֵיכֶם / וְהַיּוֹם אִם עַל הַשִּׁיר נִפְתָּה לְבַבְכֶם / רְאוּ כִּי רָעָה נֶגֶד פְּנֵיכֶם. / וְתִבְחֲרוּ לָכֶם מֹאזְנַיִם וּמִשְׁקָל / עַל כֵּן לֹא יֵצֵא מִשְׁפָּט מְעֻקָּל.

וַיַּקָשׁ דְּבַר הַשִּׁירִים / מִדְּבַר אִישׁ דְּבָרִים / וַיֵּצְאוּ מִפְּנֵי הַשִּׁיר זוֹעֲפִים בְּחֵמָה / כִּי עַל פִּי הַשּׁוֹטְרִים הָיְתָה שׁוּמָה / כִּי יִעֲדוּ בָּם הַשּׁוֹטְרִים וְלַעֲבָדִים לַשִּׁיר כְּבָשׁוּם. / וַיָּבוֹאוּ הָרֹעִים וַיְגָרְשׁוּם. / וָיֵאוֹתוּ לְהַמְלִיךְ עַל כָּל מִלָּה וְדִבְרָה / אֶת דִֵּבְרֵי הַשִּׁירָה. / וַיִּהְיוּ הַדְּבָרִים בַּמִּשְׁפָּט עֲשׁוּקִים / כִּי הֵסִירוּ הַשּׁוֹפְטִים מִשְׁפַּט הַצַּדִּיקִים. / וַיִּשְׂמַח הַשִּׁיר עַל כִּי גָבַר שִׁירוֹ וְכַחְשׁוֹ / וְעַל צַדְּקוֹ נַפְשׁוֹ. / וַיָּגֶל כְּבוֹדוֹ עַל הַגְּזֵלָה וְעַל הַנְּאָצָה / וְעַל הָעֹשֶׁק וְעַל הַמְּרוּצָה.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

הִתְעוֹרְרִי נַפְשִׁי וְסוֹד / מֵעֵין [דְּבַר] חָכְמָה גְּלִי

נַפְשִׁי כְאוֹבֵד קָם וּמָךְ / מִטּוּב נְבוֹנִים הַעֲלִי

שִׁיר אִישׁ לָשׁוֹן לִבּוֹ כְּפִי / שִׂכְלוֹ יְכַשֵּׁר בַּכְּלִי

אֵיךְ שִׁיר נְבוֹן־לָחַשׁ אֲרִי / אִזֵּן שָׁר סָכָךְ חֳלִי

לֹא שִׁיר נְבוֹנִים שִׁיר כְּסִיל / אִם תִּכְתְּשֵׁנוּ בָּעֳלִי

כִּי לֵב נְבוֹנִים בֵּית יְקָר / וּלְבַב כְּסִיל בֵּית הָאֱלִי

הַשִּׁיר פְּעָמִים יַעֲלֶה / וּמִבּוֹר וּמִשַּׁחַת בְּלִי

הַשִּׁיר כְּמוֹ רָבִיד עֲלֵי / צַוָּאר וְעַל גָּרוֹן חֲלִי

בִּינַת נְבוֹנִים מִבְּלִי / שִׁיר נֶחְשְׁבָה מַר מִדְּלִי.
