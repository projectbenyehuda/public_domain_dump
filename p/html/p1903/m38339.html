משל פא: חֹלֶה וּבִתּוֹ וְרֹפֵא

חֹשְׁבִים לְהַכְשִׁיל בְּתַחְבֻּלּוֹתָם / יִפְּלוּ בְמוֹעֲצוֹתָם


רֹפֵא הִקִּז דָּם מֵחֹלֶה / וַיֹּאמֶר לוֹ אֶת הַדְּבָרִים הָאֵלֶּה / הַדָּם יְהִי בְיָדְךָ לְמִשְׁמֶרֶת / עַד יוֹם שְׁלִישִׁי בְּאַשְׁמֹרֶת / וְהָיָה בַּיּוֹם הַשְּׁלִישִׁי / אֶרְאֶה אִם תֵּצֵא חָפְשִׁי מִן הֶחֳלִי אֲשֶׁר מֹשְׁלָךְ / וְאֵיךְ יֵקַל מֵעָלֶיךָ / בַּדָּם אַכִּיר כָּל מַכְשֵׁלָה / כָּל נֶגַע וְכָל מְקוֹר מַחֲלָה / כִּי כָל סִבּוֹת הַנְּגָעִים / הַדָּמִים מוֹדִיעִים / יָרֹק שָׁחוֹר לָבָן אֲדַמְדָּם / כִּי נֶפֶשׁ כָּל בָּשָׂר הַדָּם / הַחֹלֶה צִוָּה לְבִתּוֹ / לָשֵׂאת הַדָּם מֵאִתּוֹ / וּלְשָׁמְרוֹ כְּבָבַת עֵינָהּ / פֶּן יִשָּׁפֵךְ אָנֶה וְאָנָה / כִּי כָל אֲשֶׁר הַכְּלִי הֹפֵךְ / בָּאָדָם דָּמוֹ יִשָּׁפֵךְ / הַדָּם לָקְחָה וַתִּשָּׂאֵהוּ / תַּחַת תֵּבָתָהּ שָׂמָתְהוּ / חָתוּל הָלַךְ לְפִי תֻּמּוֹ / וְשָׁפַךְ אֶת דָּמוֹ / אָז רָפוּ יָדֶיהָ כָּהֲתָה רוּחָהּ / וַתֹּאמַר: אוֹי אָבַד נִצְחָהּ / אֵיךְ אֹמַר לְאָבִי אֵינֶנּוּ / וְכִי יִפְקֹד מָה אֲשִׁיבֶנּוּ / אֲשֶׁר שְׁפָכוֹ יִטְרְפֶנּוּ הָאַרְיֵה / אִם בְּהֵמָה אִם אִישׁ לֹא יִחְיֶה / סְעִפֶּיהָ לְחָצוּהָ / כִּלְיוֹתֶיהָ יְעָצוּהָ / לְהָרִיק בַּכְּלִי מִדָּמָהּ / כַּמִּדָּה אֲשֶׁר הָיָה שָׁמָּה / זָמֲמָה מַקִּיז וַתְּבִיאֵהוּ / בַּחֲדָרֶיהָ וַתְּצַוֵּהוּ / לְהַקִּיז מִדָּמָהּ בַּכֶּלִי / וְלָדַעַת מְקוֹר הֶחֳלִי / הָרֹפֵא בָא בִזְמַנּוֹ / וְהַחֹלֶה הִגִּיד לוֹ עִנְיָנוֹ / וְזֶה שָׁאַל אֹתוֹ לֵאמֹר / הַדָּם אֲשֶׁר צִוָּה לִשְׁמֹר / וְהַחֹלֶה הֵשִׁיב כִּי לֹא נִגְרָשׁ / וְגַם דָּמוֹ הִנֵּה נִדְרָשׁ / בִּתּוֹ אֶת הַדָּם הֵבִיאָה / וְכַאֲשֶׁר הָרֹפֵא רָאָה / אֹדֶם הַדָּם וּמַרְאֵהוּ / דָּם יֵחָשֵׁב לָאִישׁ הַהוּא / חָרָה אַפּוֹ וַיִּתְעַבָּר / וַיֹּאמֶר לוֹ: הִנְּךָ מְעֻבָּר / לֹא יִהְיֶה כָזֹאת וְלֹא נִהְיָה / בְּגוּף זָכָר וָלָד וְשִׁלְיָה / לֹא אֶתְמַהּ אִם אַתָּה חֹלֶה / עֲווֹנוֹתֶיךָ הִטּוּ אֵלֶּה / לְאִישׁ כָּמוֹךָ רִשְׁעֶךָ / וְשַׂמְתָּ שַׂכִּין בְּלֹעֶךָ / הָרֹפֵא שָׁב אֶל נְתִיבוֹ / וַיִּתְעַצֵּב אֶל לְבָבוֹ / הַחֹלֶה כִּי לִבּוֹ תָעָה / בִּתּוֹ קָרָא וְהִשְׁבִּיעָהּ / פְּעָמִים רַבּוֹת לָדַעַת / אִם דָּמוֹ בְּלֹא מִגְרַעַת / הֵבִיאָה כַּאֲשֶׁר קִבְּלַתּוּ / אוֹ אִם עַל הָאָרֶץ שְׁפָכַתּוּ / וְדַם אַחֵר שָׂמָה בִּמְקוֹמוֹ / עַל כֵּן לֹא עָמַד בּוֹ טַעְמוֹ / הַנַעֲרָה עָנְתָה: לֹא שָׁפַכְתִּי / הַדָּם וְשֹׁפְכוֹ לֹא יָדַעְתִּי / אַךְ מָצָאתִי הַכְּלִי נֶהְפָּךְ / וְכָל הַדָּם מִתּוֹכוֹ נִשְׁפָּךְ / וּפֶן יֶחֱרֶה אַפְּךָ עִמִּי / תַּחַת דָּמְךָ שַׂמְתִּי דָּמִי / וַיֹּאמַר: חֲשַׁבְתִּיךְ חוֹמָה וְאַתְּ דֶלֶת / מִסֻּבִּין קִוִּיתִי סֹלֶת / הִנֵּה חֲשַׁבְתִּיךְ מְסֻגֶּרֶת / וְאַתְּ מֵאִישׁ מְעֻבֶּרֶת / וַתִּזָּרְעִי מֵאֲנָשִׁים זֶרַע / קִוִּיתִי טוֹב וַיָּבֹא רָע / הַרְבֵּה הִשְׁבַּעְתִּיךְ בַּחֲדָרִים / לְבִלְתִּי לֶכֶת אַחֲרֵי הַבַּחוּרִים / אַךְ בְּחֶבְלֵי עָרְמָתֵךְ נוֹקַשְׁתְּ / הוֹד וְהָדָר לֹא לָבַשְׁתְּ.


וְהַמָּשָׁל בְּחֹשְׁבִים לְהַכְשִׁיל אֲחֵרִים בְּתַחְבֻּלוֹתָם / פִּתְאֹם יִהְיוּ מַכּוֹתָם וּבַמַּאֲרָב אֲשֶׁר הֵם יֶאֱרֹבוּ / יִתָּפְשׂוּ בִּמְזִמּוֹת זוּ חָשָׁבוּ / וְתִגָּלֶה רָעָתָם וּמַעֲלָם / בְּרֶשֶׁת זוּ טָמָנוּ נִלְכְּדָה רַגְלָם / חֹשֵׁב לִלְכֹּד בְּחַבְלֵי שָׁוְא עֲמִיתוֹ / שָׁוְא תִּהְיֶה תְמוּרָתוֹ / כַּאֲשֶׁר קָרָה לַנַּעֲרָה בְמַעֲלָהּ / כִּי אָבִיהָ חֲשָׁבָהּ לִבְתוּלָה / וַיֵּדַע כִּי אִישׁ מְקוֹרָהּ הֶעֱרָה / וַיִּבֶז הַנֶּפֶשׁ אֲשֶׁר הָיְתָה בְעֵינָיו יְקָרָה.
