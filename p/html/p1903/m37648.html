עַכְבָּר אָמַר בְּלִבּוֹ : מקאמה

גָּבֹהַּ מֵעַל גָּבֹהַּ


עַכְבָּר אָמַר בְּלִבּוֹ: זָכָר בְּלֹא אִשָּׁה מַה טִּיבוֹ / וְכָל הַחַי מִכָּל בָּשָׂר רָאִיתִי / וְאִשָּׁה בְּכָל אֵלֶּה לֹא מָצָאתִי / וְהִנֵּה נַפְשׁוֹ נִכְסָפָה / לִדְרֹשׁ וְלָתוּר אִשָּׁה יָפָה / וְטוֹבַת חֵן וּלְבָנָה אֲדַמְדֶּמֶת / וַאֲשֶׁר לֹא מַאֲכָל טוֹעֶמֶת / וּבְעֵת לָלֶכֶת יַפְנֶה שִׁכְמוֹ / וְלֹא יָצְאָה אִשְׁתּוֹ עִמּוֹ / וְלֹא מָצָא בְרַעְיוֹנָיו אִשָּׁה בְּלֹא דֹפִי / רַק הַשֶּׁמֶשׁ אֲשֶׁר לָהּ יֹפִי / וַיֹּאמַר: ׳אִם כָּל־יֹשְׁבֵי הָאָרֶץ חָשְׁכוּ בַעֲרִיפֶיהָ / שֶׁמֶשׁ צְדָקָה וּמַרְפֵּא בִּכְנָפֶיהָ׳ / וְכַאֲשֶׁר זָרְחָה הַשֶּׁמֶשׁ לְפָנָיו / וָתִּשָּׂא חֵן וְחֶסֶד בְּעֵינָיו / וַיֹּאמֶר לָהּ: ׳אַהֲבַת עוֹלָם אֲהַבְתִּיךְ / עַל־כֵּן חֶסֶד מְשַׁכְתִּיךְ / וּמָהֹר אֶמְהָרֵךְ לִי לְאִשָּׁה / וְלִי מִשְׁפַּט הַגְּאֻלָּה וְהַיְרֻשָּׁה׳ / וַתַּעַן הַשֶּׁמֶשׁ בְּמִרְמָה: / ׳הֲזֹאת חָשַׁבְתָּ לְחָכְמָה / לָקַחַת אוֹר אֲשֶׁר חָשַׁךְ אֶמֶשׁ / וְזָרַח הַשֶּׁמֶשׁ וּבָא הַשֶּׁמֶשׁ / הֲתָעִיף עֵינֶיךָ בּוֹ וְאֵינֶנּוּ / תָּמִיד הֶעָנָן יְכַסֶּנּוּ / וְהִנְנִי לֶעָנָן שִׁפְחָה נִכְבֶּשֶׁת / כִּי בִרְצוֹנוֹ אֲנִי מִסּוּת חֹשֶׁךְ מְלֻבֶּשֶׁת / אַךְ קַח לְךָ אִשָּׁה מִמִּשְׁפַּחַת אִמְּךָ וּמִמּוֹלַדְתָּהּ׳ / וְעָמַד וְאָמַר: ׳לֹא חָפַצְתִּי לְקַחְתָּהּ׳ / וַתֹּאמַר: ׳אָז אֶל הֶעָנָן תִּשָּׂא עֵינֶיךָ / יָדַעְתִּי כִּי לֹא תָשִׁיב פָּנֶיךָ׳ / הָעַכְבָּר בְּכִלְיוֹתָיו יִשְׁתּוֹנָן / לָלֶכֶת בְּחָפְזָה אֶל הֶעָנָן / וַיֹּאמֶר לָהּ: ׳הַיּוֹם יָגַעְתִּי וּמָצָאתִי / רַעְיָתִי יָפָתִי תַּמָּתִי / אֲשֶׁר בַּעֲצַת הַשֶּׁמֶשׁ אֶקָּחֵךְ / וְתִהְיִי לִי וְלֹא אֶזְנָחֵךְ׳ / וַתַּעַן הֶעָנָן וַתֹּאמֶר: ׳גָּבֹהַּ מֵעַל־גָּבֹהַּ שֹׁמֵר / בְּיַד הָרוּחַ נְתָנָנִי / בַּאֲשֶׁר תַּחְפֹּץ תַטֵּנִי / צָפוֹן דָּרוֹם מִזְרָח מַעֲרָב / בִּזְרֹעַ נְטוּיָה וּבְכֹחַ רָב / וְאִם אִשָּׁה כָמֹנִי תִרֶץ / וְהָיִיתָ נָע וָנָד בָּאָרֶץ / עֲזֹב שִׁפְחָה וְקַח גְּבֶרֶת / כִּי אֲנִי בְּיַד הָרוּחַ נִמְסֶרֶת / בֹּא אֶל הָרוּחַ וְעִמָּה הַסְכֵּן / תְּפַתֶּה וְגַם תּוּכַל, צֵא וַעֲשֵׂה כֵן׳ / וַיֵּלֶךְ אֶל הָרוּחַ הָעַכְבָּר / יִמְצָאֶהָ בְּאֶרֶץ מִדְבָּר / וַיֹּאמֶר אֶל הָרוּחַ: ׳אַל תִּכָּלְאִי / אִתִּי מִלְבָנוֹן כַּלָּה תָּבֹאִי / כִּי מִכָּל־הַנָּשִׁים אֲשֶׁר לִי אֶחֱזֶה / אֹתָךְ רָאִיתִי לְפָנַי בַּדּוֹר הַזֶּה / וְאִשָּׁה אַחֶרֶת לֹא אֶקַּח עָלַיִךְ / וּתְהִי לִי וַאֲנִי אֵלָיִךְ׳ / וַתַּעַן לוֹ: ׳הָעֵת בָּאתָ לָקַחַת אֹתִי / לֹא יָדַעְתָּ אֶת שִׁפְלוּתִי / כִּי אֵין בִּי כֹחַ וְעָצְמָה / לְהַפִּיל בְּגַעֲרָתִי חוֹמָה / אִם מֵאֶבֶן אוֹ מֵעָפָר / רִיתְךָ מֵאִתִּי תוּפָר / בַּאֲשֶׁר חוֹמָה מִמֶּנִּי גָבְרָה / וְאִם הִיא בְעֵינֶיךָ יָשְׁרָה / וְתוּכַל אֹתָה לְפַתּוֹת / תְּהִי לְךָ מִשְׂגָּב לְעִתּוֹת׳ / וַיֵּלֶךְ אֶל הַחוֹמָה וַיֹּאמַר: ׳שִׁמְעִי / בַּעֲצַת הַשֶּׁמֶשׁ וְהֶעָנָן וְהָרוּחַ דְּעִי / כִּי בָאתִי לְפָקְדֵךְ לִשְׁלוֹמִים / אֵרַשְׂתִּיךְ לִי בְּחֶסֶד וּבְרַחֲמִים׳ / וַתַּעַן הַחוֹמָה בְּחֵמָה: ׳לְהוֹכִיחַ עָלַי חֶרְפָּתִי וּכְלִמָּה / שְׁלָחוּךָ עָדַי לְהַבְאִישֵׁנִי / כִּי בָאתָ אֵלַי לְהַזְכִּיר עֲווֹנִי / בַּאֲשֶׁר הֵמָּה עוֹלִים וְיוֹרְדִים / וַאֲבָנַי וַעֲפָרִי לֹא נָדִים / > וּבִי אֵין כֹּחַ וְלֹא גְבוּרָה / וְכָל עַכְבָּר וְרֶמֶשׂ מְקוֹרִי הֶעֱרָה / וְגִלּוּ יְסוֹדִי לָשׂוּם בִּי מְסִלּוֹת / אֲנִי חוֹמָה וְשָׁדַי כַּמִּגְדָּלוֹת / וַיְנַפְּצוּנִי בְּפִיהֶם וּבְרַגְלֵיהֶם / הָעַכְבָּרִים לְמִשְׁפְּחוֹתֵיהֶם / וַיִּשְׁכְּנוּ בִי אֵם עַל בָּנִים / יוֹתֵר מִשְּׁתֵּי מֵאוֹת קִנִּים / וְלֹא יָכֹלְתִּי נֶגְדָּם לַעֲמֹד / הָאִשָּׁה כָמוֹנִי תָחְמֹד׳ / וּבִרְאוֹת הָעַכְבָּר כִּי נִכְזְבָה תוֹחַלְתּוֹ / וַיִּקַּח לוֹ אִשָּׁה מִמִּשְׁפַּחְתּוֹ / אֲשֶׁר נוֹלְדָה בִּשְׁכוּנָתוֹ בְצִדּוֹ / וַיִּמְצָא לוֹ עֵזֶר כְּנֶגְדּוֹ.


וְהַמָּשָׁל בְּרֹדֵף אַחַר כָּבוֹד לְהַשִּׂיגֶנּוּ / וּשְׂרָרָה וְכָבוֹד בּוֹרְחִים מִמֶּנּוּ / וְהַמִּתְגַּדֵּל יוֹתֵר מִמַּה שֶּׁבְּמַתְכֻּנְתּוֹ / סוֹפוֹ לָשׁוּב עַל עֵקֶב בָּשְׁתּוֹ / כִּי יָשׁוּב כָּל אֶחָד עַל כַּנּוֹ / כָּבוֹד לַמְכֻבָּד וְנִקְלֶה לְמִינוֹ / וְאָמַר הֶחָכָם הַיִּשְׁמְעֵאלִי: צֹבֵעַ שֵׂיבָה יִהְיֶה נִקְלֶה / עֵת גִּדּוּל הַשֵּׁעָר אוֹתוֹ יְגַלֶּה / וְאָמַר שְׁלֹמֹה: עוֹלֶה בְלֹא רְשׁוּת יִקַּח בָּשְׁנָה / כִּי טוֹב אֲמָר לְךָ עֲלֵה הֵנָּה.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר: בְּרֹאשׁ כָּפוּף וְלֵב נִשְׁבָּר / וְעֵין שָׁפָל רְגָלִים יִצְעָדוּ רְאֵה אֵשׁ יַעֲלֶה יֵרֵד / וּמַיִם נִגְּרוּ סוֹף יַעֲמֹדוּ.
