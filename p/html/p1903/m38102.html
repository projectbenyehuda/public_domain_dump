משל מא: טַוָּס, חִטִּים וְעָגוּר

גֵּאֲוַת אָדָם תַּשְׁפִּילֶנּוּ / וְלֹא יוּכַל לָדוּן עִם שֶׁתַּקִּיף מִמֶּנּוּ


טַוָּס הַנִּקְרָא בַּרְבּוּר / מָצָא מִן הַחִטִּים צִבּוּר / וַיָּבֹא הֶעָגוּר גַּם הוּא / וּבַדָּגָן שָׁלַח פִּיהוּ / וַיֹּאמֶר הַבַּרְבּוּר אֶל הֶעָגוּר / לָמָּה מִמֶנִּי לֹא תָגוּר / וּמִי אַתָּה אֲשֶׁר לִי תִתְחַבַּר / מַה לַּתֶּבֶן אֶת הַבָּר / וּמִי כְיָפְיִי בְּכָל הָעוֹפוֹת הֹוֶה / וְאֶל מִי תְדַמְּיוּנִי וְאֶשְׁוֶה / וּמִי יִמְצָא בִּי כָּל שִׁמְצָה / כִּכְתֹנֶת פַּסִּים לִי נֹצָה / וְהַשָּׂרִים וְהַשָּׂרוֹת / זְנָבִי בְּרֹאשָׁם לַעֲטָרוֹת / מִמֶּנּוּ יַעֲשׂוּ כוֹבָעִים / וּרְאֵה אַתָּה אִם לִי רֵעִים / כִּי מִי יוּכַל לְסַפֵּר הוֹדִי / אַשְׁרֵי עוֹף לוֹ אַגִּיד סוֹדִי / בְּרֹאשִׁי כֶתֶם פָּז לַעֲטָרָה / לִי הַכָּבוֹד וְהַמִּשְׂרָה / לְיָפְיִי כָּל עוֹף לֹא דָמָה / לִי הַנֹּצָה לִי הָרִקְמָה / וְלִזְנָבִי לֹא נִמְצָא עֶרְכָּהּ / אַמָּתַיִם וָחֵצִי אָרְכָּהּ / וְאַתָּה דּוֹמֶה לְגַנָּב / אֲשֶׁר כָּרְתוּ מִמֶּנּוּ זָנָב / לָמָּה לֹא תִירָא וְלֹא תֵחָת / לָבֹא עִמִּי בְּמַחֲנֶה אֶחָת / סוּר נָא וְנִהְיֶה לִשְׁנֵי מַחֲנוֹת / חֶסְרוֹן לֹא יוּכַל לְהִמָּנוֹת / וַיַּעַן הֶעָגוּר: אִם סְעִפַּי בִּי הֶחֱשׁוּ / בַּדֶּיךָ מְתִים יַחֲרִישׁוּ / מַה תִּתְהַלֵּל בִּזְנָבְךָ וְנֹצוֹתֶיךָ / יְהַלֶּלְךָ זָר וְלֹא פִיךָ, נָכְרִי וְאַל שְׂפָתֶיךָ / וְאִם אַתָּה בְעֵינֶיךָ יָפֶה / לְפִי גָדְלְךָ אַתָּה שָׁפָל וְרָפֶה / שָׁפָל בְּקוֹל שָׁפָל בְּקוֹמָה / לֹא תֵצֵא בְּיָד רָמָה / מִגַּעֲרַת כָּל עוֹף תִּתְרוֹפֵף / וְלֹא תוּכַל כְּמוֹהֶם לְעוֹפֵף / עַל כֵּן תָּדוּר בֶּחָרָבָה / וְהִנֵּה קָמָה אֲלֻמָּתִי וְגַם נִצָּבָה / צַוָּאר אָרֹךְ וּנְטוּי גָּרוֹן / בֵּין כּוֹכָבִים אַשְׁמִיעַ רוֹן / שׁוֹקַי יְשָׁרוֹת, עָלַי נֶאֱמַר / זֹאת קוֹמָתְךָ דָּמְתָה לְתָמָר / וּזְנָבִי אֲשֶׁר קָצְרָה / לֹא בְרִפְיוֹנִי אַךְ בִּגְבוּרָה / לָתֵת עַל כָּל שֹׂנְאַי אֵימָה / אֶאֱזֹר מָתְנַי לַמִּלְחָמָה / וְתִפְאַרְתְּךָ אַךְ לְךָ לְדֹפִי / שֶׁקֶר הַחֵן וְהֶבֶל הַיֹּפִי / כִּשְׁמֹעַ הַבַּרְבּוּר קָשׁוֹת / וַיִּדֹּם כִּי עֵת לַחֲשׁוֹת.

וְהַמָּשָׁל בְּאִישׁ הַמִּתְהַלֵּל בְּיָפְיוֹ / אִם לִבּוֹ רֵיק מֵחָכְמָה זֶה דָּפְיוֹ / כִּי גַאֲוַת אָדָם תַּשְׁפִּילֶנּוּ / לֹא יוּכַל לָדִין עִם שֶׁתַּקִּיף מִמֶּנּוּ / וְאִמְרַת הֶחָכָם הַיִּשְׁמְעֵאלִי מַה נָּעֲמָה / לֹא הִנְחִיל הָאָב אֶת בְּנוֹ דָּבָר טוֹב מֵחָכְמָה / וּמִתְפָּאֵר בְּיָפְיוֹ וְעָשְׁרוֹ מֻכֶּה בְּסַנְוֵר / כִּי מַה תּוֹעִיל פְּתִיחַת הָעַיִן אִם הַלֵּב עִוֵּר.
