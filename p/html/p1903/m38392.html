משל צו: שְׁפַנִּים וְשָׁפָן

טוֹב שֶׁתִּהְיֶה לְרַגְלֵי אֲצִילִים / מִהְיוֹתְךָ מֶלֶךְ לֶאֱוִילִים

הַשְּׁפַנִּים עַם לֹא עָז / עִמְקֵי שָׂפָה וְעַם נוֹעָז / וַיָּשִׂימוּ בַסֶּלַע בֵּיתָם / וַיִּבְזוּ אֲרִי הַמֶּלֶךְ בֵּינוֹתָם / וַיֹּאמְרוּ: נִבְחֲרָה מִמִּשְׁפְּחוֹתֵינוּ אֲדֹנִים / סְלָעִים מַחְסֶה לַשְּׁפַנִּים / וְאִם נָשִׂים מֶלֶךְ עָלֵינוּ / הָאַרְיֵה מַה יַּעֲשֶׂה לָנוּ / הֲיִרְדֹּף אַחֲרֵינוּ בַחוֹרִים וּבַמְּעָרוֹת / בַּסְּלָעִים וּבַצְּרִחִים וּבַבֹּרוֹת / וַיִּשְׁלְחוּ מַלְאָכִים בְּכָל גְּבוּלָם / וַיִּצָּעֲקוּ לְיוֹם מוֹעֵד כֻּלָּם / וַיִּבְחֲרוּ לָהֶם שָׁפָן גְּבַהּ קוֹמָה / זָקֵן וְגִבּוֹר וּמָלֵא חָכְמָה / וַיֹּאמְרוּ לוֹ: בְּחַרְנוּךָ לְמֶלֶךְ חֹבֵשׁ וְעֹצֵר / לַעֲשׂוֹת כָּל הַיּוֹצֵא מִפִּיךָ לֹא יִבָּצֵר / כִּי זֶה הָאֲרִי אֲשֶׁר תֵּחַת מִקּוֹלוֹ / לֹא יָדַעְנוּ מֶה הָיָה לוֹ / וּבְטֶרֶם יַבְהִילְךָ וְיִהְיֶה עָרֶךָ / וּפָרַקְתָּ עֻלּוֹ מֵעַל צַוָּארֶךָ / וַיַּעַן לָהֶם: חָלִילָה לִי לִהְיוֹת מַלְכְּכֶם / וְלֹא אֶמְשֹׁל אֲנִי בָּכֶם / כִּי לֹא הוֹד מַלְכוּת וְכָבוֹד וּמִשְׂרָה / לַמֹּלֵךְ עַל אֻמָּה שְׁפָלָה וּמִשְׁפָּחָה צְעִירָה / וּמְאֹד יִסְכַּל הַמְבַקֵּשׁ לְהִשְׂתָּרֵר עֲלֵיהֶם / נִשְׁמְטוּ בִּידֵי סֶלַע שֹׁפְטֵיהֶם / וְלָמָּה הִרְגַזְתּוּנִי לַעֲלוֹת / לָלֶכֶת אֳרָחוֹת עֲקַלְקַלּוֹת / כִּי הַמַּלְכוּת אֲשֶׁר תִּתְּנוּ לִי וְהַמֶּמֶשָׁלָה / מִן הַמַּמְלָכוֹת תִּהְיֶה שְׁפָלָה / טוֹב לִי לִהְיוֹת אֶחָד מֵעַבְדֵי הָאֲרִי הַקְּטַנִּים / מִמְּלֹךְ עַל עַם בְּלֹא עָצְמָה וְכֹחַ וְאוֹנִים / וְאַכְפִּי עֲלֵיכֶם לֹא יִכְבָּד / וְיִהְיֶה כְבוֹדֵנוּ בַּד בְּבָד / וַיְשַׁסַּע בִּדְבָרָיו אוֹתָם / וַיְסַכֵּל אֶת עֲצָתָם.

וְהַמָּשָׁל כִּי טוֹב הֱיוֹתְךָ לְרַגְלֵי אֲצִילֵי עָם / מִהְיוֹתְךָ בֵּין אֲנָשִׁים רֵיקִים מַלְכָּם וְרֹעָם / בַּאֲשֶׁר בֵּין הַגְּדוֹלִים שֵׁם לְךָ תִקְנֶה / וּבַחֲבוּרָתָם תִּכּוֹנֵן וְתִבָּנֶה / וּבֵין הַנִּבְזִים אֲשֶׁר תִּשְׂתָּרֵר עֲלֵיהֶם / גַּם אַתָּה כְּאֶחָד מֵהֶם / וּכְבָר אָמְרוּ חֲכָמִים גְּדוֹלִים / הֱוֵי זָנָב לָאֲרָיוֹת וְאַל תְּהִי רֹאשׁ לַשֻּׁעָלִים / וְחָכָם קָרָא לִבְנוֹ וַיֹּאמֶר לוֹ: בְּנִי / בְּכוֹרִי אַתָּה וְרֵאשִׁית אוֹנִי / דַּע כִּי הַצּוּר יָעַץ לְהַגְבִּיהַּ שְׁפָלִים / וּלְהַשְׁפִּיל אֲשֶׁר בִּגְאוֹנָם עוֹלִים / וְהַשְּׂרָרָה בוֹרַחַת מִכָּל מְבַקֵּשׁ גְּדֻלָּה / עַל כֵּן תַּשְׁפִּיל עַצְמְךָ וְתִנָּשֵׂא לְמַעְלָה.

וָאְֶשָּׂא מְשָׁלִי וָאֹמַר:

לִבָּךְ שְׂרָרָה לִרְדֹּף טֹרֵחַ / אַל תִּתְמַהּ אִם נִרְדָּף בֹּרֵחַ

דּוֹד, נַפְשְׁךָ תַּשְׁפִּיל וְתִנָּשֵׂא שָׁם / כִּי לַעֲנָוִים חֵן וְעֹז פֹּרֵחַ

לֹא יִמְצָא כָבוֹד גְּבַהּ לֵב אַךְ דַּע / כִּי עַל נְכֵה רוּחַ יְהִי זֹרֵחַ

תִּבְחַר גְּדֻלָּה תִּבְרַח מִמְּךָ אָז / תַּמְשִׁילְךָ לִשְׁחִין אֲשֶׁר פֹּרֵחַ

תִּדְמֶה הֱיוֹת שֻׁעָל וְהִנָּךְ אַרְיֵה / תַּחְשֹׁב הֱיוֹת אַרְיֵה וְאַתְּ קֵרֵחַ

אִם תֶּאֱהַב לִהְיוֹת בְּיֹשֶׁר וָתֹם / תִּהְיֶה בְעוֹלָמָךְ כְּמוֹ אֹרֵחַ

וּכְפִי אֲשֶׁר תִּשַּׁח וְתִכָּנַע יוֹם / לַיּוֹם הֲדַר כָּבוֹד לְךָ סֹרֵחַ

גֵּאֶה וְגָאוֹן תַּרְחִיק בַּאְשֶׁר רַע / טַעְמוֹ וְעַל דַּרְכּוֹ בְּאוּשׁ הָרֵיחַ

יֵאוֹת לְהִתְגָּאוֹת עֲלֵי כָל מֹשְׁלִים / לִמְשֹׁל וְעַל שֶׁמֶשׁ וְעַל יָרֵחַ.
