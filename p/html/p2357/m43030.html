נְשָׁמָה מִמְּקוֹר

שיר בדרך בקשה ותחינה על חשק העבודה ודבקותו בה


נְשָׁמָה מִמְּקוֹר חַיִּים חֲצוּבָה / זְבַדְתַּנִי לְהִתְעַנֵּג בְּאִבָּה

וְלֹא אַסְפִּיק בְּהוֹדָאָה שְׂגוּבָה / וְלוּ תִהְיֶה מְלֹא עוֹלָם וְרָחְבָּהּ

הֲלֹא חַסְדָּךְ מְאֹד נִשָּׂא וְגָבַהּ / עֲלֵי אַחַת לְאַלְפֵי הָרְבָבָה

וּמַעְנֵה הַלְּשׁוֹן מִמָּךְ שְׁאוּבָה / וּמֵאִתָּךְ עֲרִיכָתָהּ וְנִיבָהּ

גְּמוּלֶךָ לְטוֹב עָלַי נְדָבָה / וְתַשְׁלוּמִי לְרָעָה לֹא לְטוֹבָה

דַּאֲגָתִי מְאֹד גָּדְלָה וְרַבָּה / וְהַבּוּשָׁה בְלִי תַכְלִית וְקִצְבָּה

אֲשֶׁר אֲנִי בְלֹא חַיִל וְצָבָא / בְלֹא טַעַם וְרֵיחַ כָּעֲרָבָה

הֲיֵשׁ לִי פֶה לְהַחְזִיר הַתְּשׁוּבָה / בְעֵת לֶכְתִּי לְמִדְבָּר וַעֲרָבָה

וְאִם אֲנִי בְלִי זְכוּת וְחוֹבָה / הֲלֹא הַלְוַאי לְנַפְשִׁי הָעֲלוּבָה

אֲבָל אֶדְמֶה כְמוֹ סַל וַעֲרֵיבָה / מְלֵאָה תוֹךְ וְהַהַוּוֹת בְּקִרְבָּהּ

וְהַנֶּפֶשׁ אֲשֶׁר בַּשֵּׁם נְקוּבָה / כְגַלְמוּדָה כְמוֹ נִדָּה וְזָבָה

נְסוּגוֹתִי וְשַׂמְתִּיהָ עֲזוּבָה / לְאָן אֶפְנֶה וְעַתָּה אָן אֲנִי בָא

לְבֵיתֶךָ אֲנִי נִבְרַח וְנֶחְבָּה / בְּרַחְמֶיךָ וְיָדָךְ הַנְּדִיבָה

אַהֲבָתְךָ וְעִמָּהּ פַּת חֲרֵבָה / וְלֹא שִׂנְאָה בְשׁוֹר אָבוּס וְאֵיבָה

עֲקִיצָתְךָ מְתוּקָה וַחֲשׁוּבָה / וְאוֹסִיפָה בְךָ אַהְבָה וְחִבָּה

נְדוֹד חִשְׁקָךְ לְהָטַנִי כְלַבָּה / רְצוֹנָךְ אַךְ וְלֹא תֵבֵל וְטוּבָהּ

עֲבוֹדָתָךְ לְנַפְשִׁי הִיא מְשִׁיבָה / וּמַשָּׂאָךְ מְאֹד נַחַת וְשׁוּבָה

וְהַחֹפֶשׁ לְנַפְשִׁי הוּא כְלוּבָהּ / וְכָעֶבֶד וְשׁוֹכֵב מַעֲצֵבָה

וּבַעֲבֹד עֲבוֹדָה הַחֲבִיבָה / בְּעֵינַי אָז כְּמֶלֶךְ בַּמְּסִבָּה

בְּרַוּוֹתִי לְנַפְשִׁי הָרְחָבָה / אֲנִי כַמֵּת וְנַפְשִׁי הִיא שְׁכוּבָה

וְאִם אֶהְיֶה כְאֶרֶץ הַגְּנוּבָה / אֲנִי כֹחִי בְעוֹלָם זֶה וּבַבָּא

יְקַר תֵּבֵל וְכָל חֶמְדַּת זְהָבָהּ / כְּלוּם יוּכְלוּ לְכַבּוֹת הָאַהֲבָה

גְּוִיַּת עַבְדְּךָ וְכָל אֲשֶׁר בָּהּ / אֲנִי מֵכִין לְעָבְדָךְ בַּחֲטִיבָה

וְאֵין תִּקְוָה לְעַבְדֶּךָ וְצָבָא / בְעוֹלָם זֶה וַאֲשֶׁר תּוֹךְ רְחוֹבָהּ

לְבַד רְאוֹת פְּנֵי אָדוֹן וְסִבָּה / בְלִי בוּשָׁה וְלֹא אוֹסִיף לְדַאְבָה

מְאֹד נַפְשִׁי צְמֵאָה גַם רְעֵבָה / לְהִתְקָרְבָהּ בְּתוֹסֶפֶת מְרֻבָּה

בְּדַבְקוּתָךְ קְשׁוּרָה וַעֲנוּבָה / עֲשֵׂה אִתָּהּ כַּאֲשֶׁר עִם לְבָבָהּ

נְעוּרֶיהָ הֲלֹא פָשְׁטָה וְעָזְבָה / עֲדֵי תַחְזֹר לְאָבִיהָ וְשָׁבָה

אֲמָתְךָ הִיא בְכָל שַׁחְרָהּ וְעַרְבָּהּ / וְחֵרוּתָהּ בְּעֵינֶיהָ תְּעוּבָה

וְכָל נְצִירוּת עֲבוֹדָתְךָ עֲרֵבָה / עֲלֵי נַפְשִׁי וְהִיא צֳרִי כְאֵבָה

תְּשׁוּקָתָהּ בְּתוֹךְ לִבִּי כְתוּבָה / וְלֹא יִכְבֶּה לְעוֹלָם שְׁבִיבָהּ

וְלֹא יֵצְרוּ צְעָדַי מִנְּתִיבָה / לְהַלֵּךְ בָּהּ עֲדֵי זִקְנָה וְשֵׂיבָה.
