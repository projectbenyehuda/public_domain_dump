[הֲיִיטַב לַנְּדִיבִים]

ולה קול ירתי בה צדיקא לה תופי ויעזי קראבתה

ולו שיר קינה מקונן בו על ידידו שנפטר ומנחם את קרוביו


הֲיִיטַב לַנְּדִיבִים טוּב פְּעָלִים – / וּמוֹת נָדִיב כְּמוֹ מוֹת הַנְּבָלִים!

וְיִתְהַלֵּל בְּרָב יָמִים חֲכַם לֵב – / וְיוֹם מָחָר יְהוּ יָמָיו צְלָלִים!

בְּחֵי אֵל! בַּהֲבָלִים יַהֲלֹךְ אִישׁ / עֲלֵי אֶרֶץ – וְלֹא יָבִין הֲבָלִים,

כְּאִלּוּ הַכְּסִילוּת סָגְרָה עַל / לְבָבֵינוּ – וְאֵינֶנּוּ כְּסִילִים,

וְהַחַיִּים וְהַמָּוֶת , וְאֵין בָּם / פְּלִילִיָּה, בְּכָל נֶפֶשׁ פְּלִילִים,

וּבֵינָם נִמְצְאָה אֵיבָה, כְּאֵיבַת / שְׁתֵּי צָרוֹת, וְיֵשׁ לָהֶם בְּעָלִים,

יְחַבֵּל זֶה אֲשֶׁר יַעַשׂ חֲבֵרוֹ, / כְּעַוָּלִים – וְאֵינֵימוֹ עֲוִילִים,

וְיֵלֵד אִישׁ וְיֵלֵד יוֹם, וְיַלְדֵי / בְּנֵי אָדָם לְיַלְדֵי יוֹם אֲכוּלִים,

וְאִם לֹא יֶחֱזֶה הַבֵּן בְּמוֹת אָב – / בְּנֵי הָאָב בְּחַיֵּי אָב שְׁכוּלִים,

וְרַע מִזֶּה וּמִזֶּה כִּי פְּעָלָיו, / אֲשֶׁר הִקְדִּים, בְּמֹאזְנַיִם שְׁקוּלִים.

לְךָ, רֵעִי, אֲשֶׁר חֻלָּל וְאֵינוֹ, / לְךָ הוֹמֶה לְבָבִי כַּחֲלִילִים,

לְךָ רֹאשִׁי כְּמוֹ מַיִם, וְעֵינַי / כְּמַעְיָנִים, וְעַפְעַפַּי נְחָלִים,

וְלָךְ מֵעַי, כְּיוֹלֵדָה, יְחִילוּן, / וְיִתְקַצֵּץ לְבָבִי כַפְּתִילִים,

וּבִלְתָּךְ בַּחֲמַת רוּחִי אֲהַלֵךְ / כְּקוֹצֵף אוֹ כְּאַחַד הַסְּכָלִים,

וְאֵתֶּן רֹאשׁ, וְאֵאָסֵר בְּבֵית בּוֹר, / וְאָקוּץ לַחֲנוֹת אֵצֶל אֲצִילִים,

לְמַעַן שִׁלְּחוּ אוֹתָךְ בְּבוֹר, אֵין / לְךָ מֶנְהוּ יְצִיאָה, בַּחֲבָלִים

וְיַעַן לֹא גְזָלוּךָ, וְאוּלָם / כְּבֵדִי וַחֲצִי לִבִּי גְזוּלִים.

יְדַעְתִּיהו ּ, וְיוֹדֵעַ אֲהוּבוֹ / נְהִי אוֹמֵר וְלֹא יֹאמַר הֲתוּלִים,

וְלֻקַּח, וַאֲשֶׁר לֻקַּח יְדִידוֹ – / בְּעֵינָיו כָּל יְמֵי חַיָּיו גְעוּלִים.

קְצַר שָׁנִים – וְזִמּוֹתָיו אֲרֻכּוֹת, / שְׁפַל בֶּרֶךְ – וְלִבּוֹ בַכְּסִילִים,

נְבוֹן דָּבָר – וּמִתְרָאֶה כְּפֶתִי, / גְּבַהּ לֵבָב – וְהוֹלֵךְ אֶל שְׁפָלִים,

וְלֹא נָתַן לְאֵל תִּפְלָה בְּעוֹדוֹ, / וְלֹא נָתַן כְּבוֹדוֹ לַפְּסִילִים.

קְרוֹבָיו, הַקְּרוֹבִים לִי כְּנַפְשִׁי, / יְנַחֶמְכֶם מְנַחֵם הָאֲבֵלִים

וְיוֹעִילְכֶם, בְּצִדְקָתוֹ וְתֻמּוֹ, / בְּשֶׁבֶת אוֹ בְּלֶכֶת עַל שְׁבִילִים.

וְלִידְידִי אֲשֶׁר גָּוַע יְכוֹנֵן / נְוֵה נוֹחַ כְּאַחַד הַגְּדוֹלִים,

וְיָסֵך ְ עַל קְבוּרָתוֹ בְּעָבִים, / וְיַשְׁקֶה אֶת עֲפָרוֹ בַטְּלָלִים!
