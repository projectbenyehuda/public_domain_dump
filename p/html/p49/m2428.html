שְׁמוּאֵל קַדְּמָה

וְאָמַר בְּעָבְרוֹ בְדֶרֶךְ גַיְאֶן בְּמָקוֹם אָרְבוּ לוֹ בּוֹ הַשַּׂר בִּן אַבִּיקֻרָה בְּכָל חֲיָלָיו, פָּרָשִׁים וְרַגְלִים,

וּבָרַח אָבִי, יוֹקִירֵהוּ אֵל, וְנִכְנַס בְּנָהָר גָדוֹל וְעָמֹק אֲשֶׁר שָׁם וְהִצִּילוֹ הַשֵּׁם בּוֹ, וְנִצַּל מֵהֶם,

כִּי הָאוֹיְבִים לֹֹא מְלָאָם לִבָּם לְהִכָּנֵס בַּנָּהָר אַחֲרָיו. וְקָרָא שַׁם זֹאת הַשִּׁירָה:

תְּחִנָּה

שְׁמוּאֵל, קַדְּמָה יוֹשֵׁב כְּרוּבִים / בְּנִדְבַת פֶּה, כְּמָטָר עַל עֲשָׂבִים,

וְיִהְיוּ מַהֲלָלֵי צוּר גְּאָלָךְ / בְּתוֹך פִּיךָ וְלִבָּךְ נֶאֱהָבִים,

וְעֵת תֵּלֵך עֲלֵי נָתִיב – זְכֹר אֵת / אֲשֶׁר עָשָׂה לְךָ עַל הַנְּתִיבִים,

וּבָרֵךְ אֵל גְּמָלָךְ טוֹב, וְשִׂיחַ: / מְבֹרָךְ אַתְּ, מְשַׁלֵּם טוֹב לְטוֹבִים!


זְכֹר אוֹתוֹת אֲשֶׁר עָשָׂה לְךָ פֹה, / כְּאוֹתוֹתָיו לְבָרָיו הַקְּרוֹבִים,

בְּיוֹם רָדְפוּ מְלָכִים אַחֲרֶיךָ / לְהָרְגֶךָ עֲלֵי סוּסִים כְּעָבִים,

וְיָצְאוּ מֵעֲקַלְקַלּוֹת, נְתָנוּם / לְךָ מִפֹּה וּמִפֹּה מַאֲרָבִים,

וּבָרַחְתָּ וְנָפַלְתָּ בְּנָהָר, / גְּדוֹתָיו יִגְּרוּ כִידֵי נְדִיבִים,

וּבשׁוּ מֵאֲשֶׁר קִוּוּ, וְשָׁבוּ / בְּפַח נֶפֶשׁ וּבִמְרִירוּת עֲצָבִים.


הֲזֶה תִזְכֹּר? הֲלֹא תִזְכֹּר בְּהִלּוֹ / לְךָ תִשְׁעָה בְּמַחְשַׁכֵּי קְרָבִים,

וְהִקְפִּיאָךְ בְּבֶטֶן כַּגְּבִינָה, / וְהִתִּיךְ בָּךְ, כְּמוֹ חָלָב, חֲלָבִים,

וְהֵנִיקָךְ לְשָׂבְעָה – לֹא עֲלֵי פֶה, / וְהוֹצִיא רַעֲיָךְ – לֹא עַל נְקָבִים,

וְהֵכִין אֶת יְצִירָתָךְ בְּתוֹכָה, / וְהוֹצִיאָךְ לְקֵץ יָמִים חֲשׁוּבִים.


הֲזֶה תִזְכֹּר? הֲלֹא תִזְכֹּר פְּעָלָיו / בְּיַלְדוּתָךְ, יְמֵי יֶנֶק חֲלָבִים,

וְהַיָּמִים בְּעֵינֶיךָ כְּלֵילוֹת / בְּלִי דַעַת, וּבֹקֶר כַּעֲרָבִים,

וְלֹא תַכִּיר קְרוֹבִים מֵרְחוֹקִים, / וְלֹא תֵדַע אֱמֶת מִן הַכְּזָבִים,

וְלֹא תוּכַל עֲשׂוֹת טוֹבָה לְנַפְשָׁךְ – / וְהִפְקִידָךְ בְּיַד אָבוֹת מְטִיבִים,

וְנָתַן לָךְ בְּלִבּוֹתָם חֲנִינָה / לְהַשְׂבִּיעָךְ, וְלֹא הָיוּ רְעֵבִים,

וְלִשְׁמָרְךָ בְּבוֹא קֶרַח בְּלַיִל, / וְלִשְׁמָרְךָ בְּיוֹמָם מִן שְׁרָבִים,

וְהִשְׂכִּילָךְ אֱלֵי יָמִים לְדַבֵּר – / וְאַתְּ רַךְ וַעֲנָפֶיךָ רְטוּבִים,

וְהִנְהִיגָךְ וְהוֹלִיכָךְ בְּרֶגֶל, / וְהָיִיתָ מְנַתֵּר כַּחֲגָבִים.


הֲזֶה תִזְכֹּר? הֲלֹא תִזְכֹּר בְּשׂוּמוֹ / לְךָ מַדָּע וְשֵׂכֶל טוֹב, כְּשָׂבִים,

וְהֶחְכִּימָךְ בְּמִקְרָאוֹ וְדָתוֹ / אֲשֶׁר הֵמָּה בְּרֹאשׁ חָכְמוֹת נְקוּבִים,

וְהוֹדִיעָךְ תְּבוּנַת הַיְּוָנִים, / וְהִשְׂכִּילָךְ בְּחָכְמַת הָעֲרָבִים,

לְפַתֵּחַ בְּצַחוֹת, בֵּין נְקוּבִים / בְּשֵׁם, פִּתְחֵי תְבוּנָה לֹא נְקוּבִים,

וְעָשָׂה אֶת שְׁמָךְ גָּדוֹל וְעָצוּם / בְּעֵבֶר יָם וְאַרְצוֹת מַעֲרָבִים.


הֲזֶה תִזְכֹּר? הֲלֹא תִזְכֹּר בְּנָתְנוֹ / לְךָ כֵס עַל בְּנֵי עָשׁ הַחֲשׁוּבִים,

וְעשֶׁר רַב כְּיַד הָאֵל, וְכֶסֶף / כְּעֵץ יַעַר, וְכָאֶבֶן – זְהָבִים,

וּבֶגֶד בּוּץ וְאַרְגָּמָן וְרִקְמָה / וּמַכְלוּלִים בְּתוֹך בֵּיתָךְ סְחוּבִים,

וְנָתַן לִבְּךָ רָחָב כְּמוֹ יָם, / וְכַפָּךְ רַעֲנַנָּה כָרְבִיבִים,

וְהוֹרָךְ לַעֲנוֹת, עֵת קוּם מְלָכִים, / נְגִידִים, מִשְּׁאוֹל נֶפֶשׁ מְשִׁיבִים,

וְעִתּוֹתֵי שְׁלוֹמִים – דַּת שְׁלוֹמִים, / וְתַחְבּוּלוֹת, בְּעִתּוֹתֵי קְרָבִים,

לְהַבְעִיר אוּר קְרָבוֹת, אוֹ לְכַבּוֹת / בְּמוֹעֵצוֹת קְרָבוֹת כַּלְּהָבִים,

וְלִדְרֹךְ בַּחֲמַת רוּחַ מְשַׂנְאִים, / כְּמוֹ דוֹרֵךְ עֲנָבִים בַּיְּקָבִים.


הֲזֶה תִזְכֹּר? הֲלֹא תִזְכֹּר בְּקוּמו / לְךָ בַצַּר בְּפִדְיוֹנִים קְרוֹבִים

וְרִוָּה חַרְבְּךָ מִדָּם כְּמַיִם / אֲשֶׁר מִמַּעְיְנֵי שָׂשׁוֹן שְׁאוּבִים,

וְיוֹם צָרָה וּפַחַד, חַם כְּתַנּוּר / אֲשֶׁר מִזֶּה וּמִזֶּה לוֹ שְׁבִיבִים,

וְכָל נֶפֶשׁ שְׁחַרְחֹרֶת, וְכָל לֵב / כְּמוֹ דוֹנַג, וְכָל פָּנִים צְהֻבִּים,

וְחִצִּים וַחֲנִיתוֹת, הַשְּׁלוּחִים / כְּמוֹ מָטָר, בְּאַוִּיר נֶעֱרָבִים,

וְעַל רָאשִׁים וְרִכְבֵי סוּס וְשָׂרִים / חֲרָבוֹת פָּעֲרוּ פִיוֹת רְחָבִים –

גְּאָלָךְ מִיְדֵי צָרָה קְרוֹבָה – / וְאֵין גּוֹאֵל לְךָ מִן הַקְּרוֹבִים,

וְגָדְלוּ מַעֲשָׂיו מִסְפֹר וְסַפֵּר / וּמִשׂוּמָם בְּתוֹךְ סֵפֶר כְּתוּבִים.


אֱלֹהִים, תֵּן לְעַבְדָּךְ פֶּה לְהַגִּיד / כְּבוֹד הוֹדָךְ עֲלֵי חֶסֶד וְטוֹבִים:

עֲלֵי חֶסֶד – בְּטוֹבָתָךְ גְּמַלְתּוֹ / בְּעִתּוֹת צַר, וְהוּא כָבֵד בְּחוֹבִים,

וְעַל טוֹבִים – בְּצִמָּאוֹן, כְּמַיִם, / וּבַמִּדְבָּר מְצָאָם, כַּעֲנָבִים.

אֲחַוֶּה אֶת פְּעָלֶיךָ לְכָל קָם / וְכָל יוֹשֵׁב וְכָל עוֹבֵר וְשָׁבִים,

וְיִמָּלְאוּ בְּסִפְרָתָם לְשׁוֹנוֹת, / וְיִמָּלְאוּ בְּחַקּוֹתָם כְּתָבִים,

וְיֵלְכוּ עַל דְּרָכִים עִם אֲנָשִׁים / אֱלֵי צֹעַן וְאֶל אֶרֶץ לְהָבִים.


אֱלֹהִים הֶעֱוָה עֶבֶד, סְעַדְתּוֹ / בְּהוֹן יָקָר, וּמַרְבַדִּים חֲטוּבִים,

וְשֵׁם גָּדוֹל, כְּשֵׁם זַכִּים, אֲשֶׁר אֶל / דְּבָרֶיךָ חֲרֵדִים וַעֲצֵבִים,

וְזָנָה אַחֲרֵי לִבּוֹ וְעֵינָיו, / כְּמַרְגִּיזִים, וּפָנָה אֶל רְהָבִים,

מְחֵה פִשְׁעוֹ, שְׁמַע שִׂיחוֹ, וְיִהְיוּ / הֲגִיגֵי פִיו לְפָנֶיךָ עֲרֵבִים!

הֲטוֹב לָךְ כִּי תְּהִי פוֹקֵד מְרָדִים / עֲלֵי דוֹדִים, וְחוֹבִים עַל אֲהוּבִים

וְאִם עָנוּ פְּשָׁעִים בּוֹ – יְלִיצוּן / צְדָקָה בַעֲבוּרוֹ הַכְּרוּבִים.


אֱלֹהִים, יֵשׁ בְּשִׁירוֹתַי דְּבָרִים, / עֲלֵיהֶם דִּבְּרוּ עָתָק מְרִיבִים,

וְלֹא דָלוּם כְּמוֹ מַיִם עֲמֻקִּים, / אֲבָל דִּמּוּם אֱלֵי מַיִם גְּנוּבִים,

וְאֵל בּוֹרותֹ מְשֻׁבָּרִים – וְהֵמָּה / מְמֻלָּאִים בְּכָל טוֹבָה, חֲצוּבִים,

וּפִתְרוֹנָם לְךָ גָלוּי וְלִבִּי / כְּמוֹ תֵדַע – וְאַתְּ בּוֹחֵן לְבָבִים.


אֱלֹהִים מַעֲשֶׂיךָ אַל תְּחַבֵּל, / וְאַל תָּמִיר רְפוּאָה בַכְּאֵבִים,

וְאַל תוֹבִישׁ יְדִידַי בִּי, וְאַל נָא / בְּיוֹם אֵיד יִשְׂחֲקוּ עָלַי כְּלָבִים,

וְסֹךְ עָלַי בְּאֶבְרָתָךְ וְצִלָּךְ, / כְּחֶפְצִי, לֹא כְּמוֹ חֵפֶץ יְרִיבִים,

וְהַמְתֵּק לִי עֲדֵי מָוֶת שְׁנוֹתַי, / וְהַמְתֵּק לִי בְּתוֹך קִבְרִי רְגָבִים!
