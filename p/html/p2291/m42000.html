[מַה־לָּךְ צְבִיָּה פֹּה אֲשֶׁר פֹּה תִּשְׁכְּבִי]

קִינָה עַל־מוֹת חַנָּה אֵשֶׁת אָחִיו (נִכְתְּבָה בְּשֵׁם עִמָּנוּאֵל פְרַאנְשֵׁיס)

מַה־לָּךְ צְבִיָּה פֹּה אֲשֶׁר פֹּה תִּשְׁכְּבִי / מַה־לָּךְ וְאֵיךְ אֶל־אָהֳלֵךְ לֹא תִקְרְבִי

הַיּוֹשְׁבָה בִצְפוּן חֲדָרִים בַּדְּבִיר / אֵיךְ כַּנְּבָלוֹת בַּדְּרָכִים תֵּשְׁבִי

הַאַתְּ אֲשֶׁר בִּכְנַף מְעִילִי תֹאחֲזִי / עֵת אֵצְאָה לַשּׁוּק לְבַל־תֵּעָזְבִי

עַתָּה בְּמִי זֶה תִבְטְחִי כִּי יוֹם וְלֵיל / פֹּה תֵשְׁבִי בָדָד בְּמִי תִתְעָרְבִי

קוּמִי צְאִי הִתְעַלְּסִי בָאַהֲבָה / הִתְרַפְּקִי עָלַי וּבִי הִתְחַבְּבִי

אַלּוּף נְעוּרַיִךְ אֱהִי אִישֵׁךְ אֲנִי / מַה־זֶּה וְלִי תִּסָּתְרִי וַתַּחֲבִי

בַּעְלֵךְ אֲנִי גַּם אֵין בְּזֹאת הֶעְדֵּר יְקָר / חֶשְׂפִּי זְרוֹעַיִךְ גְּרוֹנִי סוֹבֲבִי

קוֹדֵר אֲהַלֵּךְ מִשְּׁחוֹר לֵבָב וְסוּת / וּלְדַאֲבוֹנִי אֵיךְ עֲדֶן לֹא תִדְאֲבִי

מֵי נוֹזְלֵי עֵינַי כְּמִנְהָגֵךְ מְחִי / נָא עִם לְחָיַיִךְ לְחָיַי נַגְּבִי

לָמָּה אֲדָמָה תַּעְצְרִי לִי מִנְּשֹׁק / אִשְׁתִּי וְאֵיךְ אוּכַל נְשׂא הֵעָכְבִי

מַה־זֶּה שְׁאוֹל תִּהְיֶה לְעֶדְנִי מַחֲבוֹא / אֶרֶץ פְּצִי לִי פִיךְ וְתַחְתַּי תִּרְחֲבִי

נֻפַּח כְּרֵסֵךְ וּטְבוּרֵךְ רָם מְאֹד / אַךְ מִבְּשָׂרִי תִשְׁמְנִי וַתַּעֲבִי

תִּפֹּל יְרֵכֵךְ הָאֲדָמָה אַחֲרֵי / כִי מִמְּרִירוּת מֵי דְמָעַי תִּצְבִי

קוּמִי צְאִי נָא יַעֲלָה הַאַתְּ אֲשֶׁר / מִלַּי עֲלֵי לוּחַ לְבָבֵךְ תִּכְתְּבִי

בִּתֵּךְ אֲשֶׁר תֹּאמַר “לְאָן אֵם הָלְכָה” / לָהּ אֵעֱנֶה “בָּךְ בָּחֲלָה אִמֵּךְ וּבִי”

לָמָּה עֲזַבְתִּינִי לְעוֹלָם יַעֲלָה / לָמָּה לְאִחוּרֵךְ זְמָן לֹא תִקְצְבִי

מַה־לִּי לְדֹדַיִךְ אֲשֶׁר לוּ תִתְּנֵם / אֵלַי הֲלֹא אַתְּ אָז לְבָבִי תִגְנְבִי

רוּחַ צְבִיָּה נַעֲלֵית מִגַּאֲוָה / בֵּין יוֹשְׁבֵי־חֹמֶר לְבַל תִּתְעָרְבִי

וַתִּיף בְּרֹב גָּבְהֵךְ וּמֵחֶמְדַּת יְקָר / בֵּין כּוֹכְבֵי רוֹם תִּפְרְחִי וַתֵּשְׁבִי

הָהּ בַּת תְּמִימָה אֵיךְ דֳּפִי בָךְ אֶתְּנָה / שֶׁקֶר תְּדַבֵּר פִּי לְשׁוֹנִי תִכְזְבִי

שִׂיחִי וְכַעְסִי יַעֲלָה בִינִי לְטוֹב / זַכִּי רְצוֹנִי רַק הֲגִיגִי חַיְּבִי

נָאָה כְעֶרְכֵּךְ לַשְּׁחָקִים יָאֲתָה / בִּי תִבְחֲלִי יַעַן לְצוּרֵךְ תִּתְאֲבִי

עֵינָהּ יְחִידָתֵךְ בְּקוֹנָהּ נָתְנָה / וּבְאַהֲבָתוֹ תַּחֲלִיפִי אַהֲבִי

מַה־לִּי לְאוֹר תֵּבֵל בְּלִי זִיו זָהֳרֵךְ / הָהּ יַעֲלָה וּמְאֹד בְּעֵינַי תִּיטְבִי

מָנוֹס לְלִבִּי אָן צְבִיָּה אֶמְצְאָה / אִם לוֹחֲמִי חִשְׁקֵךְ וְזִכְרֵךְ אוֹרְבִי

אֵיךְ שׁוֹכְבַת חֵיקִי בְּעֶרֶשׂ רַעֲנָן / מִדְרָס תְּהִי אֶל־כּוֹרְמִי וּלְיוֹגְבִי

הָהּ כִּי אֱמוּנָה אַתּ עֲלֵי תוֹלָע וְאֵיךְ / הַיּוֹם בְּתוֹךְ עָפָר וְכֵפִים תִּרְקְבִי

לֹא נִסְּתָה כַּף פַּעֲמֵךְ הַצֵּג עֲלֵי / אֶרֶץ וְאֵיךְ אוֹי לִי בְּטִיט תִסָּחֲבִי

אַיֵּךְ שְׂפַת עָפְרָה וְאַיֵּה נָעֳמֵךְ / הָהּ כִּי בְמַר כָּזֶה בְּצוּף לִי תַעְרְבִי

הֵבֵאת בְּנִיבֵךְ בַּכְּפוֹר חֹם רַחֲמִים / וּבְלֵב כְּצוּר עַז בּוֹר חֲסָדִים תַּחְצְבִי

בּוֹ כָל־קְשִׁי יֵרַךְ וְיֻתַּר כָּל־סְגוֹר / חוֹרִים בְּאָזְנַיִם אֲטוּמוֹת תִּקְּבִי

צַחוֹת אֲמָרַיִךְ נְפָשׁוֹת יִלְכְּדוּן / וּבְנָעֳמָם שָׁבִית לְבָבוֹת לָךְ שְׁבִי

לֹא נִשְׁמְעָה מִפִּיךְ שְׁאֵלַת תַּאֲוָה / וּלְבַד בְּעָנְגִּי תִפְצְרִי וּתְסָרְבִי

שֶׁקֶר יְהִי הַחֵן וְהֶבֶל הַיֳּפִי / עַל־תֹּם לְבָבֵךְ יֶאֱהָבֵךְ לִבְּבִי

סִקֵּל לְבָבֵךְ מִזְּמָמָּיו כָּל־דְּחִי / מִמַּחְשְׁבוֹתַיִךְ חֲרֻלִּים תַּחְטְבִי

בִּמְאֹד מְאֹד הָיִית שְׁפֵלַת לֵב וְגַם / עַל־בָּמֳתֵי רוּם הַיְחָסִים תִּרְכְּבִי

יֵיקַר בְּעֵינַיִךְ יְקָר הַכֹּל אֲבָל / הַכֹּל בְּצֶדֶק מַעֲשַׂיִךְ תַּעְלְבִי

קַצְתְּ בִּנְעִימוּת הַזְּמָן מִנָּעֳרֵךְ / וּלְבַד לְעֶדְנֵי הַנְּשָׁמָה תִּרְעֲבִי

מִנָּעֳרֵךְ בָּזִית בְּכָל־מַלְבּוּשׁ יְקָר / תִּבְזִי בְלִבֵּךְ כָּל־עֲדִי וּתְתַעֲבִי

פִּזֵּר לְאֶבְיוֹן תִּתְּנִי וּלְשַׁד שְׁדֵי / חַסְדֵּךְ לְדַלִּים וַעֲנִיִּים תַּחְלְבִי

אַיֵּךְ טְהוֹרַת לֵב וְזַכַּת רַעֲיוֹן / אֵלֵךְ לְעֵצָה לָךְ כְּתַלְמִיד אֶל־רְבִי

אִמְרֵי שְׂפָתַי לָךְ לְדָת הָיוּ וְחֹק / לַעְשׂוֹת רְצוֹנִי מַחֲשָׁבוֹת תַּחְשְׁבִי

אֶזְכֹּר גְּבִרְתִּי אֵיךְ כְּאָמָה אֶל־גְּבִיר / תִּתְרַפְּסִי תַחְתַּי וְאוֹתִי תִרְהֲבִי

אַךְ גַּחֲלֵי אֵשׁ צַעֲרִי אֵיךְ אֶחֱתֶה / הַרְפּוּ שְׁבִיבֵי חִיל כְּבִי אִשִּׁי כְּבִי

מָה אֹמְרָה הִנְנִי כְפוּי טוֹבָה כְזֹאת / אֵשׁ עַד שְׁאוֹל תִּיקַד עֲלֵה עוֹד לַהֲבִי

חִילִי גְדַל עִצְמוּ דְוֵי נַפְשִׁי מְאֹד / שֶׁטֶף יְגוֹנוֹתַי פְּרִי תוּגָה רְבִי

אֶל־אַהֲבָתֵךְ עוֹד וְעוֹד אֶתְחַיְּבָה / נִשְׁיִי אֲשַׁלֵּם לָךְ קְצָת חוֹבֵךְ גְּבִי

אֶשְׁתַּגְּעָה עֵת כִּי בְקִרְבִּי אֶחֱשֹׁב / לוּ אַתְּ קְבַרְתִּינִי וּמַה־תִּתְעַצְּבִי

הָלֹךְ וְטָפֹף תֵּלְכִי כָל־עֵת וְלִי / תִּתְאוֹנֲנִי לָעַד לְעוֹלָם תִּכְאֲבִי

לוּ שִׁכְּלוּ בָנוּ יְדֵי הַמָּוְתָה / כַּמָּה צְבִיָּה תִּגְנְחִי וּתְיַבְּבִי

אָז מִקְּרָבַיִךְ אֲנָחוֹת תִּשְׁאֲפִי / אָז דַּם דְמָעוֹת מִלְּבָבֵךְ תִּשְׁאֲבִי

הַמָּוְתָה אֶקְרָא בְּכָל־כֹּחִי אֲבָל / אֵיכָה תְעָלָה אֶשְׁאֲלָה מֵאוֹיְבִי

אוֹי לִי וְאֵיךְ שִׁנָּה לְרַע לִי צַעֲדוֹ / אֵיךְ בָּא לְאִטּוֹ לִי וְלָךְ רָץ כַּצְּבִי

נוּחִי יְחִידָתִי בְּדִי חָלִית בְּדָא / נוּחִי בְצֵל שַׁדַּי וְתוֹדוֹת לוֹ הֲבִי

אֵלָיו תְּחַלִּי־נָא אֲשֶׁר לָךְ אֵלְכָה / מַהֵר וְאֶכְרַע בַּאֲשֶׁר תִּתְיַצְּבִי

וּבְיוֹם יְקִיצַת וַהֲקָמַת כָּל־יְצִיר / נֵחַד שְׁלִישִׁיָּה בְּאֶרֶץ הַצְּבִי

וּבְעַד יְלָדַיִךְ לְאֵל חַלִּי אֲשֶׁר / יֹאמַר לְצוּלַת יָם יְגוֹנָם “תַּחְרְבִי”

תִּזְכִּי לְטוֹב צָפוּן לְכָל־יִרְאוֹת שְׁמוֹ / תַּחַת כְּנָפָיו אוֹר וְיֶשַׁע תִּשְׂגְּבִי

שָׁלוֹם עֲלַיךְ מַחֲצִית נַפְשִׁי וְאִם / אֶצְלֵךְ תְּהִי כֻלָּה בְּשָׁלוֹם תִּשְׁכְּבִי.

(1654)
