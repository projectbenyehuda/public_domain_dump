הַטּוּ אָזְנֵיכֶם

סימן: הפר״ח שלמה יחזקאל יהודה הי״ו (במחרוזת האחרונה)


קדש

הַטּוּ אָזְנֵיכֶם לִדְבָרִים טוֹבִים

וְדֶרֶךְ אֶרֶץ לְאוֹרְחִים וּמְסֻבִּים

לֹא יִהְיוּ קַלִּים בְּמַעֲשֵׂיהֶם וְנֶחְשָׁבִים

פְּחוּתִים רְעַבְתָּנִים וְזוֹלְלִים וְסוֹבִים.


ורחץ

לְסֵדֶר וּרְחַץ רְאֵה מַעֲנֵהוּ

לְטִבּוּל בְּמַשְׁקֶה נְטִילַת יָדַיִם וּמַעֲשֵׂהוּ

וּבִנְטִילָה זוֹ לֹא תְבָרֵךְ הִזָּהֵר וְזָכְרֵהוּ

וְתוֹסֶפֶת הַוָּא״ו רְאֵה וְהָבֵן וְדָרְשֵׁהוּ.


כרפס

לְסֵדֶר כַּרְפַּס דָּת וּמַדָּעִים

פָּחוּת מִכַּזַּיִת קַח בְּדֹק מִן הַתּוֹלָעִים

וּפְטֹר הַמָּרוֹר בְּבִרְכָתוֹ תְכַוֵּן וְתַּטְעִים

שִׁשִּׁים רִבּוֹא וּפֶרֶךְ בּוֹ הֵמָּה נְטוּעִים.


יחץ

מַצָּה הָאֶמְצָעִית קְטַנָּה וּגְדוֹלָה

הַצְנֵעַ חֵלֶק לַאֲפִיקוֹמָן כִּסְגֻלָּה

יָשִׂים הַשֵּׁנִי בֵין הַשְּׁתַיִם כַּדָּת וְכַמְּסִלָּה

הַכֹּל סוֹדוֹת וְלֹא תֵדַע עַל נְקַלָּה.


מגיד

בְּסֵדֶר מַגִּיד מֹר אָרִיתִי

וּדְבָרִים נֶחְמָדִים מִמְּךָ לֹא כִסִּיתִי

י״מ אָמַרְתִּי בָה וְלֹא כִלִּיתִי

יְגַלֶּה הַמַּצָּה וּפְרוּסָה אֱחֹז עָנִיתִי

מוֹצָא שְׂפָתַי הָבֵן וְלָהּ צִפִּיתִי

לֵישֵׁב בְּאֵימָה וְכַמְּעִיל עָטִיתִי.


רחצה

יִרְחַץ יָדָיו לִסְעֻדָּה כַדָּת כַּשּׁוּרָה

לְמַיִם וּכְלִי בִין וְשׁוּרָה

כְּשֵׁרִים יִהְיוּ לִנְטִילָה כְחֻקֵּי הַתּוֹרָה

עַד מְקוֹם חִבּוּר יָד לַזְּרוֹעַ קְשׁוּרָה

אַל תְּדַבֵּר דְּבָרִים וּבְפִיךָ שְׁגוּרָה

בָּרֵךְ בְּנַחַת וְהִתְאַזֵּר בִּגְבוּרָה.


מוציא מצה

יִתְפֹּס הַמַּצּוֹת וִינַשְּׁקֵם בְּגִילָה

בְּיַחַד עִם הַפְּרוּסָה כַמְּסִלָּה

בָּרֵךְ הַמּוֹצִיא כִי הִיא תְדִירָה וּתְחִלָּה

אֲכִילַת מַצַּת בַּסּוֹף וְהַפְרֵד תֵּיבָה וּמִלָּה

וּמִצְוַת בָּעֶרֶב תֹאכְלוּ מַצּוֹת תְּכַוֵּן בָּאֲכִילָה

וַהֲסַבַּת שְׂמֹאל בְּעֵינַן לְכָל קְהִלָּה.


מרור

קַח מָרוֹר כַּזַּיִת וְעָלִים

לַחִים יִהְיוּ וּבַחֲרֹסֶת טְבוּלִים

בִּרְכַּת אֲכִילַת מָרוֹר לֹא תַעֲלִים

אִם לֹא מְרִירוּתָם לֹא הָיִינוּ נִגְאָלִים.


כורך

מַצָּה וּמָרוֹר בּוֹ רָזִים צְפוּנִים

כָּרְכֵם כְּהִלֵּל וְיַחַד יִהְיוּ חוֹנִים

שְׁנֵי זֵיתִים הֵם לְזֵכֶר הַמִּקְדָּשׁ מִלְּפָנִים

בְּלִי בְרָכָה וְכָזֶה אוֹמְרִים וְעוֹנִים

זֵכֶר לַמִּקְדָּשׁ בְּיָמֵינוּ יְחֻדָּשׁ אָבוֹת וּבָנִים

כְּרַבִּי הִלֵּל כּוֹרְכָן וְאוֹכְלָן כְּאֶחָד נִמְנִים

לְקַיֵּם הָאָמוּר מַצּוֹת וּמְרוֹרִים מַעֲשֵׂה זְקֵנִים

יֹאכְלוּהוּ בַהֲסַבָּה וּבְשִׂמְחָה לֹא יִהְיוּ דַבְּרָנִים.


שלחן עורך

שֵׁב לֶאֱכֹל בִּמְנוּחָה וּמְאֹד תִּשְׂמַח

בִּקְדֻשָּׁה וּבְטַהְרָה וּבְיִרְאָה וּבוֹ תִבְטַח

לְעָנִי וּמְשָׁרֵת תֵּן מֵאֲכִילָתְךָ וְלֹא תִשְׁכַּח

אַחֲרֵי כֵן תְּנַגֵּן וּבַתּוֹרָה פִיךָ תִפְתַּח.


צפון

קַח כַּזַּיִת אֲפִיקוֹמַן וֶאֱכֹל בַּהֲסִבָּה

חֵלֶק כְּחֵלֶק כֶּחָתָן תֵּן לְכָל בְּנֵי הַמְּסִבָּה

זֵכֶר לַפֶּסַח הַנֶּאֱכָל עַל הַשֹּׂבַע וְחִבָּה

זְכֹר אַל תִּשְׁכַּח וְאָכְלֵהוּ בְאַהֲבָה רַבָּה.


ברך

לְשֵׁם יֵחוּד אֱמֹר קֹדֵם, כַוֵּן בָאֲמִירָה

בִּרְכַּת הַמָּזוֹן בְּנַחַת וְהִתְאַזֵּר בִּגְבוּרָה

זְכֹר שַׁבָּת וְיוֹם טוֹב וְלֹא תַחְסִירָה

בְּבִרְכַּת הַכּוֹס כַּוֵּן לָרְבִיעִי וְשׁוּרָה

שְׁתֵהוּ בַהֲסִבָּה כִבְנֵי חוֹרִין וּבְנֵי הַגְּבִירָה

לֹא תַפְסִיק בַּשְּׁתִיָּה כְּמוֹ שֶׁכָּתְבוּ שִׁיעוּרָהּ.


הלל

יִמְזֹג כּוֹס רְבִיעִי, וְהַלֵּל יֹאמַר בְּגִילָה

כִּי אֲבוֹתֵינוּ אֲמָרוּהוּ, כְּשֶׁרָאוּ יָדוֹ הַגְּדוֹלָה

אַחֲרָיו לֹא יִשְׁתֶּה, זֹאת תּוֹרַת הָעוֹלָה

הִזָּהֵר לְאָמְרוֹ קֹדֶם חֲצוֹת, כַּדָּת וְכַמְּסִלָּה

וּבִזְמַן הַמִּקְדָּשׁ אֲמָרוּהוּ, בְּקָרְבַּן פֶּסַח וּבָאֲכִילָה

אַחֲרָיו הַלֵּל הַגָּדוֹל, נִשְׁמַת וְיִשְׁתַּבַּח כֻּלָּהּ

שְׁתֵהוּ בַהֲסִבַּת שְׂמֹאל, בְּלֹא בְרָכָה תְחִלָּה

וּבִבְרָכָה אַחֲרוֹנָה כַוֵּן, לִפְטֹר וְהַכֹּל סְלוּלָה.


נרצה

הַכֹּל יַעֲלֶה לְפָנֶיךָ לְרָצוֹן, אֵזוֹר חֲלִי

פְּצֵנִי וְהַצִּילֵנִי מִכָּל רָעָה, וְשָׂא מַעֲלִי

רְאֵה עָנְיִי וְגָלוּתִי, אָנָּא צוּרִי וְגוֹאֲלִי

חַדֵּשׁ יָמֵינוּ, וּשְׁלַח מְשִׁיחִי וְשַׂמְּחֵנִי אֵלִי

שׁוּב לְקַבֵּץ מִכָּל הָעוֹלָם, פְּזוּרִי לְהֵיכְלִי

יָהּ חֶסֶד זְכֹר קַבֵּל אֵל פִּלּוּלִי

יִבָּנֶה הַמִּקְדָּשׁ וְאוּלַם דְּבִיר, הַרְחֵב גְּבוּלִי

הָעֲבוֹדָה יָשִׁיב, וְלַשָּׁנָה הַבָּאָה נִחְיֶה וּקְהָלִי.
