יֵרָצֶה חִין שַׁוְעִי

סליחה


יֵרָצֶה חִין שַׁוְעִי / בְּעַד עַם נִקְרָא סְגֻלָּה,

וְיָחִישׁ קֵץ גְּאֻלָּתִי / עֲבוּר יְשֵׁנֵי מַכְפֵּלָה,

וִיקֻבְּצוּ נְפוּצָיו / יַחַד, בְּשִׂמְחָה וְגִילָה,

אֶל הַמָּקוֹם אֲשֶׁר הָיָה / שָׁם אָהֳלֹה בַּתְּחִלָּה.


אֶבְיוֹנִים אֲשֶׁר דָּפְקוּ / דְּלָתֶיךָ, אֵל נוֹרָא –

לְשַׁוְעָתָם הַקְשִׁיבָה / בְּעָמְדָם בְּרוּחַ שְׁבוּרָה,

וְרִנּוּן שִׂיחָם יִכּוֹן / בְּאַפֶּיךָ כִּקְטוֹרָה,

כִּי בְעֻזְּךָ בָּטָחוּ / וְשִׁמְךָ עֲלֵיהֶם נִקְרָא,

יוֹם הִקְדִּימוּ עֲשִׂיָּה / וְהוֹדְךָ לָמוֹ נִגְלָה.


בִּיטָה מִמְרוֹמֶיךָ, / צוּר מֵישָׁרִים בּוֹטֶה,

כִּי לְשָׁבִים עָדֶיךָ, / יָהּ, חֶסֶד אַתְּ נוֹטֶה,

וּלְזַךְ תְּשַׁלֵּם כְּצִדְקוֹ / וּכְפִי מִרְיוֹ לְשׁוֹטֶה;

חִישׁ יֶשַׁע אֲהוּבֶיךָ, / אֵפוֹד גֵּאוּת עוֹטֶה,

כְּיוֹם בּוֹ חָזוּ עַל יָם / אֶת הַיָּד הַגְּדוֹלָה.


רִנּוּן שִׂיחַ פִּיהֶם – / כְּרֵיחַ נִיחוֹחֵיהֶם

שִׂימוֹ הַיּוֹם, וְעוּרָה / לְקַבֵּץ כָּל נִדְחֵיהֶם

מִכַּנְפוֹת הָאָרֶץ / בְּבוֹא יִנּוֹן מְשִׁיחֵהֶם;

וְיִכָּלְמוּ וְיֵבוֹשׁוּ / יַחַד מַדִּיחֵיהֶם,

וְאֶת גֹּדֶל פִּשְׁעֵיהֶם / הַשְׁלִיכָה לִמְצוּלָה.


הָאֵר נָא נְתִיבוֹתָם, / צוּר עֶזְרָם וּמָגִנָּם,

כִּי נֻדְּדוּ אֶת שְׁנָתָם / וְקִבְּצוּ בְּשִׁירָם הֲמוֹנָם,

וּמֵעֶרֶשׂ תְּנוּמָתָם / עָמְדוּ לְהַלֵּל לְקוֹנָם.

שְׁבֹר מוֹטוֹת עַבְדוּתָם / וַעֲשֵׂה בְחֶמְלָה רְצוֹנָם,

וְתָרוּם בְּעֻזְּךָ יְמִינָם / וִיבֻשָּׂרוּ מְחִילָה.


מְכוּרֵי בְלֹא הוֹן חִנָּם – / בְּלֹא כֶסֶף תִּגְאָלֵם,

וּבְקוֹל שִׂמְחָה וְשִׁירִים / לְבֵית מִקְדָּשָׁם תַּעֲלֵם,

וְרַב טוּבְךָ הַצָּפוּן – / בְּתוֹךְ צִיּוֹן תַּנְחִילֵם,

וְיֵשְׁבוּ שׁוֹקְטִים וּבוֹטְחִים / וְלָעַד לֹא תִגְאָלֵם,

וּמַכְלִימָם יִכָּלֵם / וְיַעֲטוּ מַעֲטֵה תְהִלָּה.
