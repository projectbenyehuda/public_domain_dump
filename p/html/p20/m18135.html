יוצר בחכמה רוחי (בקשה)

יוֹצֵר בְּחָכְמָה / רוּחִי בְקִרְבִּי,

אֱלֹהַי, נְשָׁמָה / שֶׁנָּתַתָּ בִּי.


אַתָּה הָאֵל עוֹשֶׂה / פֶלֶא וְנִסִּים,

אַשְׁרֵי שֶׁבָּךְ חוֹסֶה, / מוֹשִׁיעַ חוֹסִים,

כִּי בַשֶּׂכְוִי תָּשִׂים / בִּינָה וּבְלִבִּי.


בִּפְגָרִים מֵתִים / תַּחְזִיר נְשָׁמָה,

אוֹתוֹת וּמוֹפְתִים / תַּפְלִיא בְחָכְמָה,

לְךָ לִבִּי כָּמַהּ, / צוּרִי מִשְׂגַּבִּי.


רוֹפֵא הַחוֹלִים, / פּוֹקֵחַ עִוְרִים,

מַגְבִּיהַּ שְׁפָלִים, / מַתִּיר אֲסוּרִים,

מַשְׁפִּיל וּמֵרִים – / רַפֵּא נָא כְּאֵבִי!


הָרָם עַל כָּל רָמִים, / זוֹקֵף כְּפוּפִים,

מַלְבִּישׁ עֲרוּמִים, / מַנְעִיל יְחֵפִים –

תֶּן כֹּחַ לַיְּעֵפִים / וּפְדֵם מִשֶּׁבִי!


מִכִּסֵּא כְבוֹדֶךָ / אַתָּה בְרָאתָהּ,

צוּרִי, לְנֶגְדְּךָ, / אַתָּה יְצַרְתָּהּ,

אַתָּה נְפַחְתָּהּ / לְקַיְּמָהּ בִּי.


קָרֵב שְׁנַת פְּדוּת / מִן הַשָּׁמַיִם,

הוֹצִיאֵנִי מֵעַבְדוּת, / רוֹקֵעַ אֶרֶץ עַל מָיִם,

יְחַיֵּנוּ מִיֹּמָיִם / וְיַבִּיט עוֹצְבִי.


טְהוֹר עֵינַיִם, / אָזוּר בִּגְבוּרָה,

יוֹשֵׁב שָׁמַיִם, / עוֹטֶה תִפְאָרָה –

הַמְצֵא כַּפָּרָה / וּסְלַח נָא חוֹבִי!


נָעִים בְּכָל נוֹי, / אַדִּיר בִּקְדֻשָּׁה,

לֹא עָשַׂנִי גוֹי / וְלֹא עָשַׂנִי אִשָּׁה –

יְמַהֵר יָחִישָׁה, / יִדְרֹשׁ לְעוֹלְבִי!


חָן-נָא לְאֻמְּךָ, / מַעֲבִיר שֵׁנָה,

זְכֹר רַחֲמֶיךָ, / הַנּוֹתֵן בִּינָה!

צוּרִי, עַד אָנָה / לֹא תַקְשִׁיב נִיבִי?


זֶבֶד טוֹב / זְבָדַנִי מַלְכִּי,

לֹא עָשַׂנִי עֶבֶד, / עָשָׂה כָּל צָרְכִּי –

יַגִּיהַּ חָשְׁכִּי / וְיָאִיר נְתִיבִי.


קְדוֹשׁ יִשְׂרָאֵל, / גּוֹמֵל חֲסָדִים,

מֵבִיא הַגּוֹאֵל / יִגְאַל עֲבָדִים –

רַחֵם נְדוּדִים, / צוּרִי וְאָבִי!
