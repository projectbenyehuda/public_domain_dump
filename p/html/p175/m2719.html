מאגדות זקני

מֵאַגָּדוֹת זְקֵנִי / שמואל ליב גורדון


עוֹד אֶזְכֹּר הַפָּנִים, הָעֵינַיִם הָהֵם,

אֲשֶׁר לֹא אֶשְׁכָּחֵם;

הַקְּמָטִים הָרַבִּים עַל הַמֵּצַח הָרָם, –

כַּמָּה טוּב יָלֶן שָׁם! –

וּמֵעָל, עוֹד אֶרְאֶה, עַל מִצְחוֹ הַטָּהוֹר

תַּלְתַּלֵּי קַוֵּי-אוֹר;

וּזְקָנוֹ הַלָּבָן הַזּוֹרֵחַ כִּמְעָט

שֶׁסִּלְסַלְתִּי בְּיָד,

כִּי יָשַׁבְתִּי חָבוּק בִּזְרוֹעוֹ עַל בִּרְכָּיו

וָאַבִּיט אֶל עֵינָיו;

וָאַבִּיט אֶל עֵינָיו וָאֶקְרָא שָׁמָּה רָז –

הוֹי, מַה טּוֹב הָיָה אָז! –

וּבְיָדוֹ הָרוֹעֶדֶת הֶחֱלִיק תַּלְתַּלַּי,

וַיְדַבֵּר אָז אֵלַי,

וַיְדַבֵּר – לֹא דְבָרִים, כִּי פְרָחִים וּקְסָמִים

מִקַּדְמֵי הַיָּמִים,

וַיְדַבֵּר – לֹא דְבָרִים, כִּי פְנִינִים נִפְלָאוֹת

רוֹחֲצוֹת בִּדְמָעוֹת…

וַיְדַבֵּר, וַיְסַפֵּר – וּלְבָבִי אָז הָרַךְ

מַה יִּפָּעֶם, מַה יַּךְ,

מַה יֵּצַר, מַה יָּעִיק, מַה יִּבְכֶּה בִי הַלֵּב

מִתַּעֲנוּג וּמִכְּאֵב!

­

אָז אֶשְׁמַע מַעֲשִׂיּוֹת, מַעֲשִׂיּוֹת נוֹרָאוֹת

מִמַּבּוּל שֶׁל דְּמָעוֹת

שֶׁזָּרַם מֵעֵין עָם – לֹא טִפּוֹת, כִּי פְלָגִים;

בָּם חַיִּים כַּדָּגִים

עַלִּיזִים וּמְשַׂחֲקִים לַאֲלָפִים, לִרְבָבוֹת

מִין בְּרוּאִים בְּלִי לְבָבוֹת –

[כִּי הַלֵּב, חֲבִיבַי, לֹא יִשְׂמַח בִּדְמָעוֹת,

לֹא יִחְיֶה בִדְמָעוֹת].

וּשְׂמֵחִים הֵם בְּלִי לֵב, וּבְלִי לֵב אוֹמְרִים שִׁיר

וְשָׁטִים בְּלִי סְנַפִּיר.

כֹּה חַיִּים הַבְּרוּאִים בַּגַּלִּים הַמְּלוּחִים

שָׁם חַיִּים אֲרוּכִים;

וְהַדְּמָעוֹת עוֹד זוֹרְמוֹת – יֵאָנֵחַ זְקֵנִי –

וַתִּדְמַע גַּם עֵינִי…


­

אָז אֶשְׁמַע מַעֲשִׂיּוֹת, מַעֲשִׂיּוֹת לִרְבָבָה

מִיָּם אֵשׁ לֶהָבָה;

וְהַיָּם לֹא יָנוּחַ אַף רֶגַע, וְיֵשׁ

שֶׁיֵּרוֹם גַּל שֶׁל אֵשׁ

וּבָלַע אֲלָפִים וּסְחָפָם וּנְשָׂאָם

אֶל לַהֶבֶת הַיָּם,

וּבַיָּם הַלּוֹהֵט וּבְלַהֶבֶת הַתְּהוֹם

הֵם חַיִּים עוֹד כַּיּוֹם –

[כִּי לֵב אֵשׁ, חֲבִיבַי, לֵב לַהֶבֶת לָהֶם

עַל כֵּן חַיִּים הֵם].


­

אָז אֶשְׁמַע סִפּוּרִים מֵאַחֲרִית הַיָּמִים,

עֵת אֲבִיב עוֹלָמִים:

הַשָּׁמַיִם יַם-אוֹר זַךְ,

כָּל הָאָרֶץ פֶּרַח רַךְ,

וּמִיָּם הָאוֹר מֵעָל

יֵרֵד זִיו בָּלוּל בַּטָּל;

מִסָּבִיב שַׁלְוָה וּדְרוֹר,

שִׁיר וָגִיל וְרֵיחַ מוֹר.

כָּל הָחַי חוֹמֵל, אוֹהֵב,

הָאָדָם – גַּם לוֹ יֶשׁ-לֵב;

וּבַלֵּב לוֹ אֱלֹהִים

וֵאלֹהָיו מָלֵא רַחֲמִים…

אֵין שִׂנְאָה, דִּמְעָה וָדָם

טוּב, חֶמְלָה – כָּל הָאָדָם!

וּבָא הַיּוֹם אָז, יוֹם פְּלָאִים

שֶאֵין לוֹ סוֹף – – –

– – – סִפּוּר נָעִים

סִפֵּר זְקֵנִי אָז מִתְּקִיעַת

שׁוֹפָר גָּדוֹל וּמִבִּיאַת

מְשִׁיחַ צֶדֶק וּתְחִיַּת

מֵתֵי עוֹלָם לְחַיֵּי-עַד.

וִיסַפֵּר עוֹד, וִיסַפֵּר אַט –

לִבִּי יָגִיל וּבְגִיל יִרְעַד,

נַפְשִׁי תִמַּס מִטּוּב, מֵרֹךְ.

וְהוּא דוֹבֵר, דּוֹבֵר – וּלְתוֹךְ

דּבָרָיו נָפַלְתִּי בִּנְשִׁיקָה

חַמָּה עַל פִּיהוּ, – בִּנְשִׁיקָה

שֶׁכָּל נַפְשִׁי נָמַסָּה בָּהּ,

שֶׁכָּל רוּחִי רָעָשָׁה בָּהּ…

וַיִּדֹּם הַזָּקֵן וַיְלַטֵּף תַּלְתַּלַּי,


וַיְדַבֵּר אָז אֵלַי:

'עֵת לָנוּחַ, בְּנִי; לֵךְ לָנוּחַ הַיּוֹם,

וּמָחָר בֹּא הֲלוֹם,

אָז אוֹסִיף אֲסַפֵּר סִפּוּרַי לָךְ, בְּנִי

הֶחָבִיב' – –

– – וַאֲנִי

יוֹם יוֹם אָבוֹא, וַאֲשֶׁר שָׁמַעְתִּי

סַפֵּר כַּיּוֹם עַל פֶּה יָדַעְתִּי.


א

הָיֹה הָיָה בָאָרֶץ אִיש,

הָיָה, בְּנִי, זֶה כְּבָר, –

וַיְהִי הָאִישׁ עָנִי מְאֹד,

עָנִי רוֹכֵל בַּכְּפָר.


רוֹכֵל עָנִי, אֲבָל לַמְדָּן,

לַמְדָּן גָּדוֹל מֵ’רָב';

צַדִּיק נֶעְלָם הָיָה הָאִיש,

אֶחָד מִלַּמֶד-וָו.


אֶחָד מִלַּמֶד-וָו הָיָה

הַצַּדִּיקִים בַּדּוֹר,

שֶׁבִּזְכוּתָם, אַךְ בִּזְכוּתָם

יִרְאוּ חַיִּים וָאוֹר


מֵאִישׁ עַד יֶלֶד עַד צִפּוֹר

אוֹמֶרֶת פִּרְקֵי-שִׁיר

מֵעֵץ עַד פֶּרַח, עַד שׁוֹשָׁן,

עַד הָאֵזוֹב בַּקִּיר.


צַדִּיק נֶעְלָם – וּבְעֵינֵי כֹל

נִבְזֶה הָאִישׁ, נִמְאָס:

'הֵן לֹא יֵדַע גַּם הִתְפַּלֵּל,

גַּם אַלֶף-בֵּית – אִישׁ גַּס!'


יוֹם יוֹם סָבַב מִכְּפָר לִכְפָר,

סָחַר, מָכַר, רָכָל;

בָּא עֶרֶב – אָז יָשׁוּב הָעִיר

אֶל אָהֳלוֹ הַדָּל.


אַךְ עֵת הַחֹשֶׁךְ כִּסָּה עִיר

וּשְׁנָת סָגְרָה עֵין כֹּל,

וּבְכָל חַלּוֹן כָּבָה הַנֵּר,

בָּרְחוֹב אֵין קֶשֶׁב קוֹל,


אָז אוֹר גָּדוֹל, גָּדוֹל מְאֹד

זָרוּעַ שָׁם בִּפְנִים,

בְּתוֹךְ הָאֹהֶל הַשָּׁפָל

וּבְעַד הַחַלּוֹנִים.


וּבְתוֹךְ הָאֹהֶל הָרוֹכֵל

לוֹמֵד סִתְרֵי תוֹרָה,

וּלְכָל מִלָּה יֵצֵא בָרָק

מִפִּיו – וִיהִי אוֹרָה.


מִכָּל מִלָּה יֵצֵא בָרָק,

מִכָּל בָּרָק – כְּרוּב,

כְּרוּב קָטָן עִם כַּנְפֵי-אוֹר,

סוֹבֵב, סוֹבֵב בִּמְעוּף…


וּפְנֵי הַכְּרוּב, בְּנִי, הַקָּטָן

כִּפְנֵי הַבֵּן הַטּוֹב;

אַךְ קַל הוּא, קַל מִקֶּרֶן אוֹר

וָטוֹב הוּא מִכָּל טוֹב…


וּבְיַד כָּל כְּרוּב עֲטֶרֶת פָּז

מְשֻׁבֶּצֶת כּוֹכָבִים,

וּלְכָל כּוֹכָב שְׁלַל צִבְעֵי עֵין

הַקֶּשֶׁת בֶּעָבִים.


וְהוּא לוֹמֵד, לוֹמֵד – וּלְכָל

מִלָּה תֵצֵא מִפִּיו,

יִקְרַב הַכְּרוּב וְעָנַד לוֹ

עַל מֵצַח נֵזֶר זִיו.


וּפְנֵי הָאִישׁ כִּמְאוֹר הַיּוֹם,

וּפִיו מֵפִיק בְּרָקִים,

וּשְׁכִינַת-אֵל עֲלֵי רֹאשׁוֹ

יוֹרְדָה מִמְּרוֹם שְׁחָקִים.


עִם הַשְּׁכִינָה גַּם הָאָבוֹת

יוֹרְדִים מֵעֵדֶן גָּן,

גַּם אֵלִיָּהוּ הַנָּבִיא

יוֹפִיעַ בֶּעָנָן.


וְהוּא הוֹגֶה סִתְרֵי תוֹרָה,

גּוֹלֶה כָּל רָז, כָּל סוֹד,

וְהֵם יוֹשְׁבִים וּמַקְשִׁיבִים

וּמִתְעַנְּגִים מְאֹד.


וָכֹה יִלְמַד, וָכֹה יָאִיר

הָאֹהֶל כָּל הַלֵּיל,

עֲדֵי יִכְבּוּ הַכּוֹכָבִים

מִנֹּגַהּ הַהֵילֵל.


אָז כָּל הָאוֹר עִם כָּל הַהוֹד

יָשׁוּב יֵעָלֶם חִישׁ, –

וּכְדָת הַיּוֹם לַעֲבוֹדָתוֹ

יָצוֹא יֵצֵא הָאִיש.


ב

וּכְבָר זָקֵן רַבִּי אַהֲרֹן

(זֶה שְׁמוֹ), זָקֵן מְאֹד,

וּבֵן אֵין לוֹ וּבַת אֵין לוֹ,

גַּם לֹא הָיָה לוֹ עוֹד.


וּמֶה חָשְׁקָה וּמַה כָּלְתָה

נַפְשׁוֹ כָּל הַיָּמִים

לִרְאוֹת פְּנֵי יֶלֶד, אוֹר עֵינָיו

וּמְלֵא-גִיל צְחוֹק תָּמִים,


וּלְהֵחָבֵק בִּקְטֹן יָדָיו

וּלְחַבֶּק-לוֹ בְּכָל חֹם,

וּלְהִנָּשֵׁק וּלְנַשֵּׁק-לוֹ

בְּלִי הֶרֶף לֵיל וָיוֹם.


וּכְבָר הִרְבָּה תְפִלָּה וָצוֹם

עַד כִּי זָקֵן וָשָׂב,

וּכְבָר שָׁפַךְ נַחֲלֵי דְמָעוֹת –

הַכֹּל, הַכֹּל לַשָּׁוְא.


­

אַךְ שָׁם בִּשְׁמֵי שָׁמַיִם, בְּנִי,

מִתַּחַת כֵּס הָאֵל,

יֵשׁ שְׁנֵי שְׁעָרִים כְּמוּסִים סְגוּרִים,

סְגוּרִים יוֹמָם וָלֵיל;


וּשְׁנֵי כְרוּבִים קְטַנִּים, זַכִּים

כְּאֵגֶל דִּמְעַת עֵין

הַיֶּלֶד, שֶׁתַּזְהִיר מִגִּיל

עֵת עַל שְׂפָתָיו צְחוֹק-חֵן, –


שׁוֹמְרִים תָּמִיד אֶת הַשְּׁעָרִים

הַנִּסְתָּרִים הָהֵם,

גַּם כָּל מַלְאָךְ, בְּנִי, לֹא רָאָם,

שָׂרָף לֹא יֵדָעֵם.


תָּמִיד סְגוּרִים שְׁנֵי הַשְּׁעָרִים,

אַךְ רֶגַע, בְּנִי, קָטָן,

עֵת הַתְּנוּמָה סָגְרָה עֵינָיו

וּפִיו שֶׁל הַשָׂטָן,


אָז נִפְתָּחִים חִישׁ הַשְּׁעָרִים

וְכָל תְּפִלָּה זַכָּה,

וְכָל תְּחִנָּה מִלֵּב נִשְׁבָּר,

דִּמְעָה מֵעֵין דַּכָּא


אָז תַּעַל שָׁם, וּמַלְאֲכֵי

רַחֲמִים רַבִּים מְאֹד

יַעֲלוּהָ חִישׁ בְּכַנְפֵיהֶם

לִפְנֵי כֵס הַכָּבוֹד;


וּלְנֵזֶר-פֶּלֶא מָלֵא אוֹר

תֵּהָפֵךְ כָּל-תְּחִנָּה,

וְעַל הַנֵּזֶר תִּתְנוֹצֵץ

מִכָּל דִּמְעָה – פְּנִינָה;


אָז יִתְעוֹפֵף לִמְרוֹם אֵין-סוֹף

בִּמְעוּף בָּרָק מָהִיר,

וּבְאוֹר גָּדוֹל וּבְאוֹר עוֹלָם

עַל רֹאשׁ הָאֵל יָאִיר;


וְהַתְּפִלָּה וְהַתְּחִנָּה

רֵיקָם לֹא תָשָׁב עוֹד – –

וְאֵלִיָּהוּ הַנָּבִיא

גִּלָּה לוֹ זֶה הַסּוֹד.


אָז יָשַׁב עֵר רַבִּי אַהֲרֹן

וַיִּתְפַּלֵּל לֵיל, לֵיל,

וּתְפִלָּתוֹ וְדִמְעוֹתָיו

עָלֹה עָלוּ לָאֵל.


­

וַיְהִי הַלֵּיל בַּחֲצוֹת הַלֵּיל

נִמְלָא בֵיתוֹ אוֹרָה,

כִּי בָא כִּי רַד אָז הַנָּבִיא

וּבְפִיהוּ הַבְּשׂוֹרָה:


'אַהֲרֹן, אַהֲרֹן, כָּעֵת חַיָּה

חַנָּה חוֹבֶקֶת בֵּן

יָפֶה, חָכָם וָטוֹב'. עָבְרָה

שָׁנָה, וַיְהִי לוֹ כֵן.


כִּי בָא הַיּוֹם נוֹלַד הַבֵּן,

וּבְאָהֳלוֹ הַדָּל

הֵאִיר, הִבְרִיק פִּתְאֹם כִּבְרֹק

בָּרָק בָּעָב מֵעָל;


אַךְ לֹא עָבַר חִישׁ כַּבָּרָק,

הַזֹּהַר לֹא חָדָל;

וּבְלֵב הָאָב וּבְלֵב הָאֵם

הָאוֹר גָּדַל, גָּדָל.


לֹא ‘הָאָבוֹת’ הוֹפִיעוּ שָׁם

לַקְשִׁיב ‘סִתְרֵי תוֹרָה’,

גַּם לֹא כְרוּבִים, שֶׁכַּנְפֵיהֶם

זוֹרְעוֹת סָבִיב אוֹרָה;


כִּי כְרוּב אֶחָד קָטָן וָזַךְ

[וּבְלִי כְנָפַיִם הוּא],

הוֹפִיעַ שָׁמָּה וּמֵאִיר

אוֹר שִׁבְעָתַיִם הוּא.


וּפְנֵי הַבֵּן [אֶפְרַיִם שְׁמוֹ]

כִּפְנֵי חַמָּה קְטַנָּה,

וּשְׁתֵּי עֵינָיו – בְּרָקִים גְּדוֹלִים

בִּשְׁחוֹר הָעֲנָנָה;


וּשְׂעַר רֹאשׁוֹ תַּלְתַּלֵּי שְׁחוֹר

יוֹרְדִים סָבִיב, סָבִיב

וּצְלִיל קוֹלוֹ – שִׁירַת זָמִיר

בְּיַעַר הָאָבִיב.


וִימֵי גִילָה וִימֵי אוֹרָה

בָּאִים, עָפִים כַּחֲלוֹם,

הַיֶּלֶד יִיף, יִגְדַּל, יֶחְכָּם,

יֶחְכַּם מִיּוֹם לְיוֹם.


וּבְטֶרֶם מָלְאָה לוֹ שָׁנָה

יָדַע כְּבָר אַלֶ“ף בֵּי”ת,

מַהֵר יָדַע גַּם הִתְפַּלֵּל;

וְלֹא אָרְכָה הָעֵת


וּכְבָר לָמַד חֻמָּשׁ, רַשִּׁ"י

לָמַד, יָדַע על פֶּה,

מַהֵר הֵחֵל גַּם גִּמָּרָה,

לִמּוּד קָשֶׁה כָּזֶה, –


וִימֵי גִילָה וִימֵי אוֹרָה

בָּאִים, עָפִים כַּחֲלוֹם;

אֶפְרַיִם יִיף, יִגְדַּל, יֶחְכָּם,

יֶחְכַּם מִיּוֹם לְיוֹם.


ג

אַךְ אוֹרָה וְגִילָה, בְּנִי הַטּוֹב,

יַעַבְרוּ חִישׁ כַּצֵּל, –

כֹּה יַחֲלֹף יוֹם בָּהִיר בַּשְּׁחָקִים,

וְאַחֲרָיו חֶשְׁכַת לֵיל.


לֹא רָחוֹק מִן הָעִיר הַקְּטַנָּה,

עַל אַחַת הַגְּבָעוֹת

הִתְנוֹסֵס הָאַרְמוֹן הַגָּדוֹל

שֶׁכֻּלּוֹ נִפְלָאוֹת,


שֶׁגַּגּוֹ עִם רָאשָׁיו הָרַבִּים

הִתְנַשְּׂאוּ עַד הָעָב,

עַמּוּדָיו אַבְנֵי שַׁיִשׁ טָהוֹר

וּמְצֻפִּים בַּזָּהָב.


וּבִזְרֹחַ הַשֶּׁמֶשׁ מִפְּאַת

הַקָּדִים מִמּוּלָם –

פַּלְגֵי אֵשׁ מִשְׁתַּפְּכִים וְנוֹזְלִים,

מַעְיְנֵי נֹגַהּ כֻלָּם.


וּבִשְׁקֹעַ הַשֶּׁמֶשׁ אַט אַט

וְהִתְגַּלְגֵּל לִפְאַת-יָם, –

וְטָבַל הָאַרְמוֹן בְּלַהֲבֵי-אֵשׁ

וּבָעַר בְּזֶרֶם-דָּם.


וַאֲחוֹרֵי זֶה אַרְמוֹן הַקְּסָמִים

הִשְׂתָּרַע גַּן עֲנָק,

בּוֹ צָמַח וּפָרַח כָּל עֵץ פְּרִי,

בּוֹ גָדַל כָּל עֵץ סְרָק.


שָׁם יִשְׂגּוּ אַלּוֹנִים אַדִּירִים

וּבְרוֹשִׁים רַבֵּי צֵל,

וּמֵעַל עֲצֵי פְרִי רַעֲנַנִּים

הִתְנַשְּׁאוּ אַרְזֵי אֵל.


בָּרָקִיעַ הָרָם, הָרַעֲנָן

שָׁם יִרְעַשׁ הָאַוִּיר:

מַקְהֲלוֹת צִפֳּרִים עַלִּיזוֹת

הִתְחָרוּ שָׁם בַּשִּׁיר,


וְהִשְׁתַּפְּכוּ עַד רָחוֹק, עַד רָחוֹק

הַנְּעִימִים הַצְּלִילִים –

וּמִתַּחַת בֵּינוֹת לָעֵצִים

הִתְפַּתְּלוּ הַשְּׁבִילִים;


הִתְפַּתְּלוּ כַּנְּחָשִׁים וְנֶעֶלְמוּ

בַּעֲבִי הַצְלָלִים, –

שָׁם יָנוּם פֶּלֶג-כֶּסֶף בָּהִיר

וְחוֹלְמִים הַגַּלִּים…


וּבְאַרְמוֹן הַפְּלָאוֹת הַמַּבִּיט

מִתְעַנֵּג עַל הַגָּן,

יָשַׁב ‘פַּן’ פּוֹלָנִי מְיֻחָס,

רַדְזִיבִיל שֵׁם הַ’פַּן'.


לוֹ רַבִּים אַרְמוֹנוֹת, הֵיכָלִים,

וּלְמֵאוֹת לוֹ כְּפָרִים,

וַעֲבָדִים וּשְׁפָחוֹת לִרְבָבוֹת,

עֲבָדִים אִכָּרִים,


עֲבָדִים אֻמְלָלִים הַחַיִּים

חַיֵּי עֹנִי, מְרוּדִים,

גַּם עָרִים לוֹ קְטַנּוֹת, וְיוֹשְׁבֵיהֶן

עֲבָדִים-יְהוּדִים.


לוֹ רַבִּים אוֹצָרוֹת בַּזָּהָב

בִּפְנִינִים וְאַבְנֵי-חֵן,

כָּל חֶמְדָּה לֹא יֶחְסַר, אַךְ אֻמְלָל

הוּא, אֻמְלָל – אֵין לוֹ בֵן.


וּמִשְׁתֶּה וְשִׂמְחָה בָּאַרְמוֹן

וּנְעִימוֹת מַנְגִּינוֹת,

וַהֲמוֹנִים שָׁם ‘פַּנִּים בְּהִירִים’

וּ’פַנּוֹת' עֲדִינוֹת;


וּתְרוּעַת עַלִּיזִים שַׁאֲנַנִּים,

מַנְגִּינוֹת וּמְחוֹלוֹת,

וּמִשְׁתֶּה וְשִׂמְחָה וְיוֹם טוֹב

וּתְשׁוּאוֹת הוֹלֵלוֹת.


וְעָבְרוּ הַיָּמִים, הַלֵּילוֹת

כֹּה רַבּוֹת בַּשָּׁנִים;

וְאוֹרָה וּנְגֹהוֹת מִסָּבִיב,

צוֹהֲלִים כָּל פָּנִים.


אַךְ אֶחָד שָׁם הָיָה שֶׁפָּנָיו

עֲצוּבִים-נִזְעָמִים,

כַּעֲנָנָה עֲרִירִית בִּשְׁמֵי אוֹר

בְּלִי מָטָר וּרְעָמִים…


וּלְבָבוֹ בוֹ חָלָל וְכוֹאֵב

לַהֲמוֹן תְּרוּעוֹת גִּיל,

וּשְׂחוֹק מַר כְּמוֹ קָפוּא עַל שְׂפָתָיו, –

הוּא הַפַּן רַדְזִיבִיל.


וּבִסְעֹר בּוֹ לִבּוֹ הַכּוֹאֵב

וְנַפְשׁוֹ אֲבֵלָה

וְקָדְרָה הָעֲנָנָה הַזְּעוּמָה

כִּטְבוּלָה בַאֲפֵלָה,


וּבָעֲרוּ הָעֵינַיִם פִּתְאֹם

וְלִהֲטוּ כַבְּרָקִים –

אָז יִזְעַף הָאָדוֹן וְקוֹלוֹ

כְּקוֹל רַעַם שְׁחָקִים.


וְאוֹי אוֹי לָהֶם אָז לַעֲבָדָיו

הַנִּרְאִים לְעֵינָיו שָׁם,

כִּי יִפְקֹד וְהֻכּוּ בַשּׁוֹטִים

וְהֻכּוּ עַד שְׁפָךְ-דָּם.


וְאוֹי אוֹי לָרוֹכֵל הַנּוֹדֵד

עִם צְרוֹרוֹ שָׁם בַּכְּפָר

וְקוֹרֵא: 'הוֹי, קְנוּ, קְנוּ, אֲדוֹנִים,

בְּלִי מְחִיר כָּל יְקָר!'


וְנֶאֱנָח בִּמְרִירוּת, בִּמְרִירוּת

וְעוֹבֵר דֶּרֶךְ שָׁם –

כִּי יוּבָא וְהֻכָּה בַשּׁוֹטִים

וְהֻכָּה עַד שְׁפָךְ דָּם.


כִּי רָעִים הָיוּ, בְּנִי, וְגֵאִים

‘הַפַּנִּים הַבְּהִירִים’,

אַכְזָרִים כַּיְּעֵנִים, וְגֵאִים

כֵּאלֹהִים אַדִּירִים.


ד

עָבַר יוֹם, יוֹם כֵּהֶה וּמְעֻנָּן

וַיְהִי עֶרֶב אָפֵל,

עָבַר עֶרֶב אָפֵל וּמְצֻנָּן,

עָבַר אַט – וַיְהִי לֵיל.


וַיְהִי לַיִל עָכוּר וְשׁוֹמֵם,

לֵיל דְּמִי אֲפֵלָה, –

הַכֹּל סַר וְנוֹאָשׁ וְדוֹמֵם

כִּפְנֵי אֵם אֲבֵלָה.


גַּם שָׁמַיִם קָדְרוּ וְסָגְרוּ

עֵינֵיהֶם-כּוֹכָבִים,

אַךְ קְטַנִּים שָׁמָּה שְׁנַיִם חָגְרוּ

מִבֵּינוֹת לֶעָבִים.


­

וּבָעֵמֶק הָעִיר הַקְּטַנָּה

כְּבָר סָגְרָה כָּל חַלּוֹן,

וַתִּישַׁן עֲצוּבָה-שַׁאֲנַנָּה;

גַּם הַשָּׂב הָאַלּוֹן,


זֶה הָעֵץ הָעַרְעָר הַבּוֹדֵד,

זְקַן הָעִיר וְשׁוֹמְרָהּ,

מִתְנַמְנֵם, וּלְעִתִּים מִתְנוֹדֵד

כְּרוֹאֶה חֲלוֹם רָע.


אַךְ בִּקְצֵה הָרְחוֹב שָׁם חוּץ לָעִיר

אֹהֶל דַּל עוֹדוֹ עֵר,

בַּחַלּוֹן הַקָּטָן שָׁם יָאִיר

אוֹר שָׂמֵחַ, אוֹר נֵר.


וּמָלֵא כָל הָאֹהֶל אַךְ גִּיל

וּנְעִימוֹת עוֹלָמִים,

וּסְבִיבוֹ יִשָּׁמַע נֹעַם-צְלִיל

שֶׁל שְׂחוֹק יֶלֶד תָּמִים.


וְיָצָא קַו הָאוֹר וּבָקַע

וְגָזַר מַחֲשַׁכִּים,

וְחָדַר בַּעֲלָטָה וְשָׁקַע

וְצָלַל בְּמַעֲמַקִּים.


וְנִדְמָה צְחוֹק הַיֶּלֶד הָרַךְ

בִּשְׁחוֹר לֵיל דּוּמִיָּה

כִּכְרוּב קַל מִתְעוֹפֵף וְתוֹעֶה

בִּתְהוֹם שְׁאוֹל תַּחְתִּיָּה.


וּבָדָד שָׁם בִּשְׁחוֹר דְּמִי הַלֵּיל

יוֹרֵד אִישׁ מִגִּבְעָה,

וְקוֹדְרִים פְּנֵי הָאִישׁ מִנִּי-צֵל

פִּי שִׁבְעִים וְשִׁבְעָה.


וְהוֹלֵךְ הַקּוֹדֵר בַּאֲפֵלוֹת,

וְעֵינָיו – אֵשׁ שְׁחוֹרָה…

כֹּה יִתְעֶה בַלֵּילוֹת, בַּלֵּילוֹת

וִיבַקֵּשׁ קַו-אוֹרָה.


אַךְ אָפֵל מִסָּבִיב וּמֵרוֹם,

וְאָפֵל בּוֹ הַלֵּב…

וְנַפְשׁוֹ בּוֹ צוֹלֶלֶת בִּתְהוֹם

שֶׁל יֵאוֹשׁ וּמְרִי כְאֵב.


וְיָרַד אֶל הָעִיר וְתָעָה

בִּרְחוֹבוֹת שׁוֹמֵמִים,

וְשָׁמַע אַךְ דְּמָמָה, וְרָאָה

אַךְ צְלָלִים דּוֹמֵמִים.


אַךְ בִּקְצֵה הָעִיר שָׁמָּה חוֹדְרָה

וְיוֹצְאָה קֶרֶן זִיו

וּבוֹקְעָה הַחֲשֵׁכָה הַקּוֹדְרָה

הַשְּׁפוּכָה מִסָּבִיב.


וַיַּקְשֵׁב – וּמִדְּמִי מַחֲשַׁכִּים

צְלִילֵי גִיל צוֹלְלִים,

וּנְעִימִים הַצְּלִילִים וְרַכִּים

כַּחֲלוֹמוֹת עוֹלְלִים…


וּבָאֹהֶל יוֹשֵׁב יֶלֶד רַךְ

עַל בִּרְכֵּי אָב וָאֵם,

וּמְפַטְפֵּט בְּקוֹל גַּלֵּי פֶלֶג זַךְ,

וּמְסַפֵּר הוּא לָהֶם


סִפּוּרֵי הַתּוֹרָה נִפְלָאִים

שֶׁיָּדַע כְּבָר עַל פֶּה,

וְשׁוֹמְעִים הַהוֹרִים מִשְׁתָּאִים,

מַבִּיטִים זֶה אֶל זֶה.


מַבִּיטִים – וְעֵינֵיהֶם יָם-אוֹרוֹת

וּלְבָבָם – נְעִימוֹת

וּמְנַשְּׁקִים עֵינֵי בְנָם הַטְּהוֹרוֹת,

עֵינֵי בְנָם הַתְּמִימוֹת.


וִיחַבֵּק אָז אֶפְרַיִם אָבִיו

וְאִמּוֹ חֲלִיפוֹת,

וּתְרוּעָה וּצְחוֹק גִּיל מִסָּבִיב,

וּנְשִׁיקוֹת מַאֲלִיפוֹת.


אַךְ אֶפְרַיִם פִּתְאֹם הִזְדַּעֲזָע

וִיצוּרָיו מַרְעִידִים:

בַּחַלּוֹן שְׁתֵּי עֵינַיִם חָזָה,

שְׁתֵּי עֵינֵי לַפִּידִים:


– הוֹי, אִמִּי! – אֲהָהּ, מִי שָׁמָּה, מִי

הַמַּבִּיט בַּחַלּוֹן

בְּעֵינֵי אֵשׁ? – הָהּ, הֵרָגַע נָא, בְּנִי!

אֵל עַנְפֵי הָאַלּוֹן


הַנָּעִים, – הִיא עוֹנָה נִבְהָלָה.

וְהַיֶּלֶד רוֹעֵד:

–לֹא, אִמִּי… זֶה מַלְאַךְ חַבָּלָה…

עֵינֵי אֵשׁ… עֵינֵי שֵׁד…


וְהַבֵּן עוֹד מַבִּיט בִּרְעָדָה

וְעֵינָיו – יָם פְּחָדִים, –

וּפִתְאֹם וְגַם הוֹרָיו בִּרְעָדָה

שָׁמְעוּ קוֹל צְעָדִים…


­

וּבְמַעֲלֵה הַגִּבְעָה עוֹלֶה אִיש

וְעוֹבֵר שָׁם בַּשְּׁבִיל,

אֶל שַׁעֲרֵי הָאַרְמוֹן יָרוּץ חִישׁ –

הוּא הַפַּן רַדְזִיבִיל.


­

וְיוֹשְׁבִים הַהוֹרִים וּמְחַבְּקִים

אֶת בְּנָם יְחִידָם,

וּפִתְאֹם – וַאֲנָשִׁים מִתְדַפְּקִים

וְקוֹל עֹז הֶחֱרִידָם.


וּבִן רֶגַע – תְּמוּנוֹת נוֹרָאוֹת

– הוֹי מַחֲזֵה בֶהָלָה! –

פָּרְצוּ שָׁם וּפְנֵיהֶן בַּלָּהוֹת

כְּמַלְאֲכֵי חַבָּלָה.


וּדְמִי אֹפֶל שׂוֹרֵר שָׁם בַּכֹּל

כִּבְנִבְכֵי תְהוֹם רַבָּה, –

אַךְ אַט-אַט שָׁם רוֹעֵד לַחַשׁ קוֹל:

הוֹי, אִמָּא! הוֹי, אַבָּא!


וּזְעָקָה אֲיֻמָּה וַאֲרֻכָּה

אָז תַּרְעִישׁ דְּמִי הַלֵּיל,

וְאַחֲרֶיהָ דְּמָמָה עֲמֻקָּה – –

וְהַכֹּל גָּז חִישׁ כְּמוֹ צֵל.


וּדְמִי קֶבֶר וּשְׁחוֹר מִנִּי שְׁחוֹר

אָז כִּסּוּ כָל הַיְקוּם,

גַּם מֵאֹהֶל נֶעְלַם קַו הָאוֹר,

לֹא נִשְׁאַר בּוֹ מְאוּם.


נֶאֱלַם הַפִּטְפּוּט הַנָּעִים

וּצְלִילֵי צְחוֹק הַחֵן,

נֶאֶלְמוּ סִפּוּרִים נִפְלָאִים

וַיֵּעָלֵם הַבֵּן.


וַעֲרִירִי הָאַלּוֹן הַבּוֹדֵד

זְקַן הָעִיר וְשׁוֹמְרָה,

מִזְדַּעְזַע לִרְגָעִים, מִתְנוֹדֵד

כְּרוֹאֶה חֲלוֹם רָע.


­

וַיְהִי אוֹר – אָתָה בֹקֶר בָּהִיר

אַחַר לֵיל עֲבָרוֹת,

וּבִקְצֵה הָרְחוֹב שָׁם חוּץ לָעִיר

מוּבָלִים לִקְבָרוֹת


רַב אַהֲרֹן וְאִשְׁתּוֹ; כִּי מוֹתָם

לֹא יָכֹל הַפְרִידָם, –

הַחַיִּים הֵם גָּזְלוּ מֵאוֹתָם

מַחְמַדָּם יְחִידָם.


וְאַךְ נִתַּק זֶה מִבֵּין זְרוֹעוֹתָם

חִישׁ פָּרְחוּ נִשְׁמוֹתָם –

                                                           - - - - - - - - - - - - - - - - -

וּבַלַּיְלָה, בַּלַּיְלָה שָׁם, בְּנִי

[ כֹּה סִפֵּר לִי זְקֵנִי ],

יִשָּׁמַע מִן הָאֹהֶל קוֹל נְהִי, –

וַתִּדְמַע גַּם עֵינִי.


עָבְרוּ יָמִים,

יָמִים רַבִּים,

שְׁנוֹת עֶשֶׂר כְּבָר עָבָרוּ;

זָרְמוּ זְרָמִים,

זְרָמִים רַבִּים

אֶל תּוֹךְ הַיָּם נָפָלוּ.


וּבְסוֹף הָרְחוֹב

בִּקְצֵה הָעִיר

חָרְבָּה שְׁמֵמָה עוֹמֶדֶת,

וְקוֹל כָּאוֹב

מִפִּנַּת קִיר

עוֹלֶה עוֹד לַיְלָה, לָיְלָה.


וּמַר הַקּוֹל

הַהוּא הַדַּק,

הָעוֹלֶה כְּמוֹ מִקָּבֶר,

כִּמְרִי הַשְּׁכוֹל,

כִּבְכִי נֶחֱנָק

בֶּעֱזוּז תּוּגָה קוֹדָרֶת.


וּבְלֵיל הַדְּמִי

עֵת צֵל קוֹדֵר

כַּעֲנָק הַיְקוּם בּוֹלֵעַ,

אָז קוֹל הַבְּכִי

עוֹלֶה, חוֹדֵר

מִתְּהוֹם עַד שְׁמֵי שָׁמָיִם.


­

וּבְהֵיכַל-הוֹד

תּוֹךְ גַּן רַעֲנָן

כַּעֲנָק עַל רֹאשׁ הַגֶּבַע

שָׁם יֵשֵב עוֹד

וִיחִי שַׁאֲנָן

פַּן רַדְזִיבִיל הַשּׁוֹעַ.


וּכְבָר הִזְקִין

הַגְּרַף וָשָׂב,

אַךְ שָׁב לִימֵי נְעוּרֵיהוּ;

רֹאשׁוֹ הִלְבִּין –

אַךְ שְׁחוֹר הָעָב

עָלָיו לֹא עוֹד יִשְׁכֹּנָה.


זָקֵן וָשָׂב

גְּרַף רַדְזִיבִיל,

אַךְ בִּלְבָבֵהוּ פְנִימָה

כָּל רַחֲמֵי אָב

כָּל נֹעַם גִּיל,

אַהֲבָה, אַהֲבָה וָנֹעַר.


כִּי בֵן לוֹ, בֵּן

רַב פֶּלֶא-הוֹד

וָטוֹב כִּכְרוּב שָׁמָיִם,

וְחָכָם – מֵאֵין

כָּמוֹהוּ עוֹד

בְּכָל חַכְמֵי הָאָרֶץ.


וּפְנֵי הַבֵּן –

יָרֵחַ רַךְ,

עֵינָיו-כּוֹכָבִים שְׁנָיִם,

וּשְׁלַל כָּל חֵן,

זִיו נֹגַהּ זַךְ

שְׁפוּכִים עַל פְּנֵי הַנָּעַר.


וּכְבָר לָמַד

כָּל הַלְּשׁוֹנוֹת –

שִׁבְעִים בָּאָרֶץ הֵנָּה –

וּכְבָר עָמַד

בְּנִסְיוֹנוֹת

בְּכָל הַחָכְמוֹת שֶׁבַע.


כִּי כֵן לִבּוֹ

רָחַב כַּיָּם

מָלֵא חָכְמָה וָדָעַת;

עוֹדוֹ בְאִבּוֹ

זֶה יֶלֶד תָּם

חָכַם מִישִׁישֵׁי אָרֶץ.


וְאוּלָם אַךְ תֹּם

וָטוֹב וָרֹךְ

וָגִיל נַפְשׁוֹ מָלֵאָה;

כִּכְרוּב בֶּן-רוֹם

הָיָה בְתוֹךְ

רֵעָיו יִלְדֵי אֲדָמָה.


וּמָה אָהַב

אָז הַזָּקֵן

בְּכָל-עֹז, בְּכָל חֹם הַנֶּפֶשׁ,

אֶת סְטָנִיסְלַב

[זֶה שֵׁם הַבֵּן]

הַנַּעַר, נַעַר פֶּלֶא!


וּמָה אָהַב

עֵינֵי הַשְּׁחוֹר

שֶׁכֹּה גָדְלוּ, עָמָקוּ;

שָׁאֹב שָׁאַב

כָּל גִּיל, כָּל אוֹר

מִמְּקוֹר עֵינַיִם אֵלֶּה.


וּמֶה חָמַד

יוֹמָם וָלֵיל

הַבִּיט אֶל פְּנֵי הַנָּעַר!

וְכֹה עָמַד

עָלָיו כַּצֵּל,

לֹא זָז מִמֶּנּוּ רָגַע.


יַבִּיט – וְיֵשׁ

שֶׁיֵּצַר לֵב

הַגְּרַף מִשֶּׁפַע אשֶׁר –

וּנְשִׁיקוֹת אֵשׁ-

אַהֲבָה וּכְאֵב

עַל פְּנֵי הַבֵּן תִּבְעַרְנָה;


וְדִמְעָה רוֹתְחָה

יוֹרְדָה מֵעֵין

הַשָּׂב וְתוֹעָה עַל לֶחִי.

וְדִמְעָה קוֹדְחָה

עַל לְחִי הַבֵּן

הַתָּם כָּאֵשׁ צָרָבֶת.


וּזְרוֹעוֹתָיו

חוֹבְקוֹת, לוֹחֲצוֹת

אֶל לֵב הַכְּרוּב-הַיָּלֶד

וְדִמְעוֹתָיו

שׁוֹטפוֹת, רוֹחֲצוֹת

פָּנָיו וּפְנֵי הַיָּלֶד.


וְכֹה עָבְרָה

שָׁנָה שָׁנָה

כְּגַל עַל טֹהַר פֶּלֶג,

וְכֹה גָבְרָה

שָׁנָה שָׁנָה

אַהֲבַת הַגְּרַף לַנָּעַר.


הִנֵּה כִי כֵן

שָׁלֵו, שַׁאֲנָן

כִּרְסִיס טַל זַךְ עַל פֶּרַח

הָיָה הַבֵּן

עַלִּיז, רַעֲנָן

וּשְׂבַע כָּל טוּב וָנֹעַם.


עַלִּיז כָּאוֹר

וְקַל כָּעוֹף

וָטוֹב כְּחֶסֶד-אֵל,

חָפְשִׁי כַדְּרוֹר

מְאֻשָּׁר לֱאֵין-סוֹף

הָיָה גְרַף סְטָנִיסְלַב.
