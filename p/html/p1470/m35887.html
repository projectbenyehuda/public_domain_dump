אוֹנֵי פִטְרִי

מא / שמות יב כט – ויהי בחצי הלילה



א

אוֹנֵי פִטְרִי רַחֲמָתָיִם / בִּתַּקְתָּ בִּמְשׁוּלֵי חֲמוֹרוֹתָיִם

גָּדְלָךְ פָּסַע בִּינְתַּיִם / דֶּבֶר לְהַמְשִׁיל בֵּינוֹתָיִם.


הֵן רֶגַע יָמוּתוּ דִּבָּרְתָּ / וּגְאוֹנָם בַּדֶּבֶר הִדְבָּרְתָּ

זְרוֹעַ צַר בִּבְכוֹרָיו שָׁבָרְתָּ / חֲצוֹת לַיְלָה שָׁם עָבָרְתָּ.


טְמֵאִים זֶה בָּזֶה נִצִּיתָ / יְדִידִים זֶה לָּזֶה רִצִּיתָ

כָּל בְּכוֹר שֶׁהוּא כְּכִלָּיוֹן מִצִּיתָ / לַיְלָה כְּחוּט הַשַּׂעֲרָה בְּעֵת חָצִיתָ.


בְּיָד וּבְיָמִין תִּסְמְכֵנוּ / וּמִנֶּגֶף תְּחַסְּכֵנוּ / וּבְצֵל מָגִנָּךְ תְּגוֹנְנֵנוּ.


ב

מֵתִים בְּכָל בַּיִת בְּנוֹף כְּנֶחֱנָקְנוּ / נַאֲקוֹת חֲלָלִים נָאָקוּ

סֻכְּרוּ וְלִבּוֹתָם נָמָקּוּ / עֻלְּפוּ וְרוּחוֹתָם נָבָקּוּ.


פְּגָרִים מֵתִים לָמוֹ עָשִׂיתָ / צוּרַת כָּל בְּכוֹר גַּם בְּמֵעֵיהֶם לֹא חַסְתָּ

קוּמִי לָךְ לִבְכוֹרָךְ שַׂחְתָּ / רָפוֹא וְנָגוֹף שָׁמָּה שַׂמְתָּ.


שׁוֹאָה וּמְשׁוֹאָה וּבֶהָלָה / שַׁמּוֹת וְשָׁאוֹן וְחַלְחָלָה

תָּקְפָה בָהֶם הַיְלָלָה / תִּמָּהוֹן קָמוּ וְיָצְאוּ בַלַּיְלָה.


שׁוֹאָה וּמְשׁוֹאָה / תַּפִּיל צַר לְשׁוּחָה / וְנִחְיֶה בְּגֶשֶׁם יְשׁוּעָה.


ג

יִשָּׁמַע לֶאֱדוֹם כְּשֵׁמַע מִצְרָיִם / מַשָּׂא דוּמָה כְּמַשָּׂא מִצְרָיִם

נִפְרַעְתָּ מִפַּתְרוֹס בְּתַכְלִית מַכַּת עֲשִׂירִית / וּמֵאֱדוֹם תִּפָּרַע בְּתַכְלִית קֶרֶן עֲשִׂירִית.


יֵרְדוּ רְאֵמִים עִמָּם / בְּכוֹרֵי מַלְכוּת עַמָּם / מִזֵּר תַּעֲרִימָם / בְּאַף תַּחֲרִימָם

יֶשַׁע שָׁמוּר / הוּא לֵיל שִׁמּוּר / אָמַר שׁוֹמֵר אָתָא קֵץ אַשְׁמוּר.


ד

לְיֵשַׁע עַמָּךְ בַּחֲצִי לַיִל יָצָאתָ / וּבְעֶצֶם הַיּוֹם אוֹתָם הוֹצֵאתָ

הַדָּן עַמּוֹ בַּיּוֹם וְעַמִּים בַּלָּיְלָה / כִּי לָךְ יוֹם אַף לָךְ לָיְלָה

לַיְלָה אֲשֶׁר בּוֹ נִפְדּוּ רִאשׁוֹנִים / שִׁמּוּרִים נַמְתָּ יְהִי לָאַחֲרוֹנִים

וּמַה שֶּׁהָיָה בָרִאשׁוֹנָה הוּא שֶׁיִּהְיֶה בָאַחֲרוֹנָה / כְּשֵׁמַע מִצְרַיִם תַּשְׁמִיעַ לְצוֹרִים

תִּצְמַח וְתִצְנַח / תִּצְלַח וְתִרְכַּב / תִּתְקַע וְתִתְקַף / תִּצְרַח וְתִצְוַח

עַל עַם נוֹרָא מָרוֹם וְקָדוֹשׁ.


ה

אָז תֵּשַׁע מַכּוֹת הָיוּ אֲדוֹנִים קָשֶׁה / אֲשֶׁר בָּם סִכַּרְתָּ גּוֹי לִבּוֹ מַקְשֶׁה

בְּעַזּוּת מֶלֶךְ עַז הָיְתָה מַכַּת עֲשִׂירִית / אֲשֶׁר הִמְשַׁלְתָּ בְּנוֹגְשֵׂי בְנֵי נֻסָּה עֲשִׂירִית

גְּדִיעַת בְּכוֹרוֹת הָיְתָה תַכְלִיתָם / אֲשֶׁר שַׁתָּה אַפָּךְ עַל תַּבְלִיתָם

דַּנְתָּ לִפְנֵיהֶם אֶת אֱלֹהֵיהֶם / וְגַם צוּרָתָם הִפַּלְתָּ בְּאָהֳלֵיהֶם

הִכִּיתָ כָּל בְּכוֹר וְכָל רֹאשׁ מִשְׁפָּחָה / כָּל בְּכוֹר לְאִישׁ וְכָל בְּכוֹר לָאִשָּׁה

וּבְכוֹר לְכָל בְּהֵמָה וּבְכוֹר לְכָל דָּבָר / בַּבַּיִת וּבַחוּץ דֶּבֶר שַׁתָּה לְכָל דָּבָר

זַעַף וָנַחַת נִגְלֵיתָ נָגוֹף וְרָפוֹא / לִבְכוֹרֵי צַר נָגוֹף וְלִבְנָךְ בְּכוֹרָךְ רָפוֹא

חֲצוֹת לַיְלָה בְּכֵן יוֹדוּךָ מוֹדֶיךָ / עַל מִשְׁפְּטֵי צִדְקֶיךָ וְעַל צֶדֶק אוֹהֲבֶיךָ

טֶכֶס אֶצְבָּעָךְ עָשְׂתָה תֵּשַׁע מַכּוֹת / יָדָךְ עָשְׂתָה מַכַּת בְּכוֹרוֹת

יְמִין עֻזָּךְ טִבְּעָה צַר וַיְרוֹמְמוּךָ הַכֹּל / וּזְרוֹעַ קָדְשֶׁךָ תִּפְדֵּנוּ וְתָרוּם עַל כֹּל.


ז

אָז רוֹב נִסִּים הִפְלֵאתָ           בַלָּיְלָה

בְּכָל אַשְׁמוּרוֹת זֶה           הַלָּיְלָה

גֵּר צֶדֶק נִצַּחְתָּ כְּנֶחֱלַק לוֹ           לָיְלָה

דַּנְתָּ מֶלֶךְ גְרָר בַּחֲלוֹם           הַלָּיְלָה

הִפְחַדְתָּ אֲרַמִּי בְאֶמֶשׁ           לָיְלָה

וְיִשְׂרָאֵל סָר אֶל מַלְאַךְ וְיָכָל לוֹ           בַּלָּיְלָה

זֶרַע בְּכוֹרֵי פַּתְרוֹס מָחַצְתָּ בַחֲצִי           לָיְלָה

חִלְחַלְתָּ מִדְיָן וַחֲבֵרָיו בִּצְלִיל לֶחֶם בַּחֲלוֹם           לָיְלָה

טִיסַת נְגִיד חֲרֹשֶׁת סִלִּיתָ בְּכוֹכְבֵי           לָיְלָה

יָעַץ מְחָרֵף לְנוֹפֵף יָד לְאִוּוּי וְהוֹבַשְׁתָּ פְגָרָיו           בַּלָּיְלָה

כָּרַע בֵּל וּמַצִּיבוֹ בְּאִישׁוֹן           לָיְלָה

לְאִישׁ חֲמוּדוֹת נִגְלָה רָז חֲזוּת           לָיְלָה

מִשְׁתַּכֵּר בִּכְלֵי קֹדֶשׁ נֶהֱרַג בּוֹ           בַּלָּיְלָה

נוֹשָׁע מִבּוֹר אֲרָיוֹת פּוֹתֵר חֲלוֹמוֹת           לָיְלָה

שִׂנְאָה נָטַר אֲגָנִי וְכָתַב סְפָרִים           בַּלָּיְלָה

עוֹרַרְתָּ נִצְחָךְ עָלָיו בְּנֶדֶר שְׁנַת מֶלֶךְ           בַּלָּיְלָה

פּוּרָה תִדְרֹךְ כְּשׁוֹמֵר מַה           מִלָּיְלָה

צָרַח כְּשׁוֹמֵר וְנָם אָתָא בֹקֶר וְגַם           לָיְלָה

קָרֵב יוֹם אֲשֶׁר הוּא לֹא יוֹם וְלֹא           לָיְלָה

רָם הוֹדִיעַ כִּי לָךְ יוֹם וְגַם לָךְ           לָיְלָה

שׁוֹמְרִים הַפְקֵד לְעִירָךְ כָּל הַיּוֹם וְכָל           הַלָּיְלָה

תָּאִיר כְּאוֹר יוֹם חֲשֵׁכַת           לָיְלָה.


ח

כִּי אֵין לְפָנֶיךָ לָיְלָה / וְהַכֹּל לְפָנֶיךָ יוֹם / וְיוֹם לְיוֹם אֹמֶר יַבִּיעַ / וְלַיְלָה לְלַיְלָה דַעַת יוֹדִיעַ / וְחֹשֶׁךְ לֹא יַחֲשִׁיךְ מִמָּךְ / וּנְהוֹרָה שְׁרֵי עִמָּךְ / בְּקָרִים יַגִּידוּ חַסְדָּךְ / וְלֵילוֹת יְשַׁנֲנוּ אֱמוּנָתָךְ / וְשַׁחַר וְנֹגַהּ וּבֹקֶר וְיוֹם וְצָהֳרָיִם וְעַרְבִית וְעֶרֶב וְנֶשֶׁף וָלַיְלָה וְכָל עֵת וְעִדָּן וּזְמָן וָרֶגַע יַזְכִּירוּ נִסֵּי פְלָאֶיךָ / וְאוֹתוֹת מוֹפְתֵי נִפְלְאוֹתֶיךָ / הַמְּחַדֵּשׁ יְשׁוּעוֹת וּפוֹעֵל נוֹרָאוֹת / הַסּוֹפֵר יָמִים וּמוֹנֶה עִתִּים / וְיוֹמָךְ אֶלֶף שָׁנִים / וּשְׁנוֹתֶיךָ לֹא יִתָּמּוּ / כִּי הַכֹּל בָּלֶה וְאַתָּה לֹא בָלֶה / וְכַאֲשֶׁר אַתָּה הֵם מְשָׁרְתֶיךָ / אַתָּה חָי וְהֵם חַיִּים / טָהוֹר וְהֵם טְהוֹרִים / קָדוֹשׁ וְהֵם קְדוֹשִׁים / וּקְדֻשָּׁה לְקָדוֹשׁ מַקְדִּישִׁים וּמְשַׁלְּשִׁים כֹכ וקרא.
