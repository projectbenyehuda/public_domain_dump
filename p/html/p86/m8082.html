ספר תחכמוני – שַׁעַר אַחַד-עָשָׂר

בְּמַהֲלַל מְלִיצָה בְּכָל מִלּוֹתֶיהָ רֵי"שׁ, וְהַשְּׁנִיָה לָהּ

הָרֵי"שׁ גָּרֵשׁ יְגָרֵשׁ


נְאֻם הֵימָן הָאֶזְרָחִי: נְשָׂאוּנִי בְנֵי מַחֲשָׁבִי. וְרַעְיוֹנֵי לְבָבִי. לִנְסוֹעַ מִנְּוֵה מוֹשָׁבִי. אֶל אֶרֶץ הַצְּבִי. וַאֲנִי עַל הָרֵי בָתֵר קַל מִצְּבִי . רוֹדֵף אַחֲרֵי כָל מְלִיצָה חֲמוּדָה. וְשׁוֹאֵף לִשְׁמֹעַ כָּל מָשָׁל וְחִידָה . וְנָסַעְתִּי לְבֵית לֶחֶם יְהוּדָה. וּבְעוֹדִי בְאַחַת עָרֶיהָ. וּבְשַׁעַר מִשְּׁעָרֶיהָ. אֲחוֹנֵן עֲפָרֶיהָ. וּמִרְעֵה עֳפָרֶיהָ. רָאִיתִי אֲנָשִׁים נִקְבָּצִים. וּמִכָּל פִּנָּה בָאִים וְרָצִים. וְכַאֲשֶׁר שְׁאִלְתִּים עַל עִנְיַן קִבּוּצָם. וְסִיבַּת רוּצָם. אָמְרוּ לִי כִּי אִישׁ חָכָם עָבַר עֲלֵיהֶם . וְחָנָה בִגְבוּלֵיהֶם. וּבִמְלִיצוֹת הַשִּׁיר אֵין מִי יוֹעִידֵהוּ. וּבְמִלְחֲמוֹת תֻּשִׁיּוֹת אֵין מִי יַעַרְכֵהוּ. וְאָמַרְתִּי בְלִבִּי אָרוּצָה נָא בְתוֹכָם . וְאֶשְׁמְעָה מַה בְּפִי הֶחָכָם. וְכַאֲשֶׁר קָרַבְתִּי אֶל הַמַּחֲנֶה. אֲשֶׁר הוּא שָׁם חוֹנֶה. וָאֵרֶא בְתוֹכָם אִיש זָקֵן. מִבְחַר הַמְּלִיצוֹת מְתַקֵּן . וְהוּא אוֹמֵר: שִׁמְעוּ חֲכָמִים מִלָּי. וְיוֹדְעִים הַאֲזִינוּ אֵלָי. כִּי בִלְשׁוֹנִי מְקוֹר הַתּוּשִׁיּוֹת וְהַמְּלִיצוֹת. וּבְרַעְיוֹנִי אוֹצַר הָאִגְּרוֹת הַמְשֻׁבָּצוֹת. וְהַמַּחְבָּרוֹת הַנִּמְרָצוֹת. וְאִלּוּ יְרוּצוּן הַבְּרָקִים בִּמְסִלּוֹתָי. תָּעוּ בִנְתִיבוֹתָי. וְלֹא הִשִּׂיגוּ אֲבַק מַרְכְּבוֹתַי. אָמַר לוֹ רֹאשׁ הַקְּרוּאִים: זְרוֹעַ רָמָה חָשַׂפְתָּ . וְחֶרֶב גַּאֲוָה שָׁלַפְתָּ. וְהִנֵּה שָׁמַעְנוּ נִיבְךָ וְלֹא חֲקַרְנוּךָ . וְכִי יָבוֹא דְּבָרְךָ וְכִבַּדְנוּךָ. אָמַר הַזָּקֵן: הִנְנִי נִצָּב לִפְנֵיכֶם. וְאָנֹכִי עוֹמֵד בֵּין ה' וּבֵינֵיכֶם. בַּחֲרוּ לָכֶם הַיּוֹם הַמְּלִיצוֹת הָעֲמֻקּוֹת. וְהַשִּׁירוֹת הַחֲזָקוֹת. וְנַסּוּ כְפִי אֲשֶׁר תִּרְצוּ. וְחַפְּשׂוּ כַאֲשֶׁר תַּחְפְּצוּ. כִּי כָל דְּבָרַי צֶדֶק. וְלֹא תִמְצְאוּ בְּשׁוֹשַׁנַּי חֶדֶק. אָמְרוּ לוֹ: אִם כֵּנִים דְּבָרֶיךָ. וְצֶדֶק אֲמָרֶיךָ. עֲשֵׂה לָנוּ שִׁירָה בְנוּיָה מִשָּׁלשׁ לְשׁוֹנוֹת. כִּמְלָאכָה עֲמֻקָּה יַכִּירֶנָּה כָל אִישׁ תְּבוּנוֹת.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

דְּבַר אֵל יֵאָמַן / וטאעת אלרחמן / שְׁפַר עוֹבָדַיָּא.

אֲקַדֵּם לוֹ שִׁירִי / ותסביח אלבארי / בְּרֵישׁ כָּל-מִלַּיָּא.

שְׁמוֹ לָעַד נַעֲלָה / הוא אלרב אלעלי / וְרִבּוֹן עָלְמַיָּא.

לְאוֹר צִדְקוֹ זֹהַר / חכי נורא יזהר / בּפַלְגּוּת לֵילַיָּא.

וְעַל כֹּל סוֹד נִשְׁקָף / בֹדאת לם יכשף / לְחַד מִן קָאֲמַיָּא.

חֲסָדָיו גָּבְרוּ / ורות אנהארה/ אֲתַר דִּי לֵית מַיָּא.

בְּתֵבֵל שָׂם מִשְׁטָר / חוי כל אלאקטאר / וְרוּמֵהּ לִשְׁמַיָּא.

וּבוֹ כָל-חַי יִחְיֶה / ומיזאן אלדניא / בְּיַמִּינֵהּ תָּלְיָא.

אֲדוֹן כָּל הַחַיִּים / לה אלמלך אלדאים / וְיוֹבְדוּן מַלְכַיָּא.

פְּלִיאוֹתָיו יָקְרוּ / ואן טאל אלדהר / מְהַשְׁנֵא זִמְנַיָּא.

וְהוּא תֵבֵל יַרְגִיז / ויאתי באלמעג’ז / בְּסוֹף עִדָּנַיָּא.

וְיִשְׁעוֹ יִגָּלֶה / פיעלי מדלולא / וְיָחִית רָמַיָּא.

וְקוֹל שׁוֹפָר יִתְקַע / לנואם אלבלקע / וְיֵיחוּן מֵתַיָּא.

וְסוֹדוֹ יַרְאֵהוּ / פיחיי מותאה / וְיִפְרוּק חַיַּיָא.

וּבֵיתוֹ יִבְנֵהוּ / ויעלי מתואה / עֲלֵי רֵישׁ טוּרַיָּא.

וְשָׁם נַעְרֹךְ שִׁירָה / לדי אלעליא שכרא / דְּחִיל תֻּשְׁבָּחַיָּא.

וְשִׁירִי בוֹ יוּשַׁר / ואמדאחי תנשר / בְּכָל לִשָּׁנַיָּא.

מְקַנְאִים בָּם יֶהְמוּ / סרי מנהא סהמוֹ / לְלֵב חַכִּימַיָּא.

וְעַל צָרִים שָׁרְקּוּ / ויהדיאה אלשרקו / לְאַרַע אַסְפַּמְיָא

בְּמִלִּים צוּף נָטְפוּ / חכי אלמסך אלערפו / וְדִכְרֵהּ לִשְׁמַיָּא.

וְהִיא שִׁירָה יָקְרָה / לקד אצֹחת פכרא / לְרֵישׁ כֹּל אֻמַּיָּא.

וְצוּר חַרְבָּה יוּטַל / עלי האם אלאבטאל / וְכֹל גִּבָּרַיָּא.

וּבָה כָּל-לֵב רוֹהֶה / ויבקא דֹכראהא / לְסוֹפֵי יוֹמַיָּא.

וַיֹּאמֶר לָהֶם: הִנֵּה שִׁירָה יְקָרָה קְשׁוּרָה בְּחֶבֶל שָׁלֹש לְשׁוֹנוֹת בַּל יֵרָתֵק . וְהַחוּט הַמְשֻׁלָּש לֹא בִמְהֵרָה יִנָּתֵק. מִפַּחְדָּהּ חָרְדוּ הַשּׁוֹמְעִים וְנִקְרְצוּ קָרֶץ. וְתַחַת שָׁלֹש רָגְזָה אָרֶץ. בְּנִיתִיהָ בִלְשׁוֹן אֲרָם וְעֵבֶר וַעֲרָב. וְאֵין אֲשֶׁר אֵלֶיהָ יִקְרָב. בַּמִּזְרָח וּבַמַּעֲרָב. אָמְרוּ כָל הָעָם: הֵטַבְתָּ וְלַעֲשׂוֹת הִגְדַּלְתָּ. וְחָכַמְתָּ וְהִשְׂכַּלְתָּ . וְחֲכָמִים סִכַּלְתָּ. וְיָעַדְנוּ כִּי בִמְלֶאכֶת הַשִּׁירִים אֵין עֲרֹךְ אֵלֶיךָ. בְּנֹעַם מִלֶּיךָ. וּבִמְלִיצוֹת מְשָׁלֶיךָ. וְאִם תּּוכַל לְהַרְאוֹת כֹּחֲךָ בִּמְפֹרָד. תִּהְיֶה לְרֹאשׁ לְכָל מְבִינֵי סְפָרָד. וְעַתָּה עֲשֵׂה לָנוּ אִגֶּרֶת. כָּל מְלִיצָה טוֹבָה אוֹגֶרֶת. חֲמוּדָה לְכָל דּוֹרֵשׁ. וְיִהְיֶה בְכָל מִלָּה מִמִּלוֹתֶיהָ רֵי"שׁ. וְתִהְיֶה שֶׁבַח לְאֶחָד מִן הַנִּכְבָּדִים. וְהַשָּׂרִים הַנֶּחְמָדִים. וַיֹּאמַר: הַטּוּ אָזְנְכֶם לְמִלָּי. וְהַאֲזִינוּ אֵלָי.

וַיַּעַן וַיֹּאמַר: קְרָא בְגָרוֹן. אִמְרֵי רֹן. וְעוֹרֵר רַעְיוֹנֶיךָ. וְהַרְחֵב גְּרוֹנֶךָ . וְהַיְשַׁר אֲשׁוּרֵי אֲמָרֶיךָ. וְהָרֵץ בִּרְחוֹב הַמַּאֲמָר רִכְבֵי זְמִירֶיךָ. וְהַרְעֵף רְבִיבֵי דְבָרֶיךָ. לְרַוּוֹת בִּמְטָרֶיךָ. אֶרֶץ הַמִּישׁוֹרָה. וּלְהַרְבִּיץ מִרְבַּץ הַתִּפְאָרָה. לְהְַפִריחַ אֲמִירֵי אֲמָרוֹת טְהוֹרוֹת. וּלְהַפְרוֹת פֹּארוֹת הַשִּׁירוֹת הַמְפֹאָרוֹת. וְלַעֲרֹךְ מְנוֹרוֹת. מְאִירוֹת כַּמְּאוֹרוֹת. לְהָאִיר רוֹאוֹת עִוְרוֹת. בְּסַנְוֵרִים סְגוּרוֹת. וּלְהַזְהִיר מְרוֹמֵי הַיְקָר הַהֲדוּרוֹת. בְּאוֹר גְּבִירֵי הַשְּׂרָרוֹת. וּפְאֵר רָאשֵׁי הַחֲבוּרוֹת. הַכְּרוּב הַנֶּאְדָּר. הַגְּבִיר הַנֶּהְדָּר. אֶדֶר הַמִּשְׂרָה וְהַיֹּשֶׁר. וַהֲדַר הַמּוּסָר וְהַכֹּשֶׁר. נֵצֶר הַשָּׂרִים הָרָמִים. מִזֶּרַע הַפַּרְתְּמִים. אֲמָרָיו סַפִּירִים. בִּסְפָרִים מְפֹאָרִים. וְשִׁירָיו תַּרְשִׁישִׁים בְּטוּרִים מְחֻבָּרִים. גַּעֲרָתוֹ הָאֲרָיוֹת טוֹרֶפֶת. וְאִמְרָתוֹ הַמְּדוּרוֹת שׂוֹרֶפֶת. וּגְבוּרָתוֹ הַנְּהָרוֹת גּוֹרֶפֶת. וְרַעְיוֹנָיו חֲרָבוֹת מְרוּקִים מְרִיקִים. וְדַהֲרוֹת אַבִּירָיו רָצִים כַּבְּרָקִים. גּוֹדֵר הַפֶּרֶץ. וְרוֹפֵא הַחֶרֶץ. וּמֵסִיר הַקֶּרֶץ. לְרַפֵּא הַהֶרֶס. וְלִקְשֹׁר הַקֶּרֶס. וּלְהָאִיר אוֹרָם כְּאוֹר חֶרֶס. וּלְהַרְבִּיץ עֲדָרִים פְּזוּרִים בְּמִבְחַר מִרְעֶה. כְּרוֹעֶה עֶדְרוֹ יִרְעֶה. יְרַפֵּא הַנִּשְׁבֶּרֶת. וְיִדְרֹשׁ הַנִּפְזֶרֶת . וְיַתִּיר הַנִּקְשֶׁרֶת. וְיָרִים רָאשֵׁי יְשֻׁרוּן. וּלְרַנֵּן יְקָרוֹ שַׁחַר יְעִירוּן. זִכְרוֹ מְקֻטָּר קְטֹרֶת מוֹר. וּבַעֲבוּרוֹ יְשׁוֹרֵר הַיְקָר שִׁיר מִזְמוֹר. רְצוֹנוֹ יְעוֹרֵר רְפָאִים. כַּאֲשֶׁר חֲרוֹנוֹ יַהֲרֹג רְפָאִים. עָשְׁרוֹ לְדוֹרְשִׁים יְפַזֵּר. וּרְכוּשׁוֹ לָרָשִׁים יְבַזֵּר. וְלַאֲסִירֵי צָרִים עוֹזֵר. יָרַשׁ הַיֹּשֶׁר מֵהוֹרָיו. וְעָצַר בַּכֹּשֶׁר מִנְּעוּרָיו. וְצָבַר הָעֹשֶׁר לְדוֹרְשָׁיו וּלְשׁוֹחֲרָיו. אַשְׁרֵי הוֹרִים טְהוֹרִים דַּרְכָּם הוֹרוּהוּ. וּבְאֹרַח מִישׁוֹר לָרוּץ עוֹרְרוּהוּ. רָאוּהוּ גְבִירִים וַיְאַשְּׁרוּהוּ. שָׂרִים וְרוֹזְנִים וַיְהַדְּרוּהוּ. רָץ כַּבָּרָק בְּאָרְחוֹתָם. וּפָרַח כַּבְּרוֹשׁ בִּמְקוֹרֵי אֲרֻחָתָם. יוֹצְרֵהוּ יִצְּרֵהוּ. וּבוֹרְאֵהוּ יַגְבִּירֵהוּ וְיַאְדִּירֵהוּ. וּלְשַׁחֵר רְצוֹנוֹ. יְעִירֵהוּ וְיַעַזְרֵהוּ. וּמֵחֲרוֹנוֹ יִשְׁמְרֵהוּ וְיִנְצְרֵהוּ . וְיָרִים לְדוֹר דּוֹרִים בְּרֹאשׁוֹ נִזְרוֹ. וְיָאִיר נֵרוֹ. וְיַזְרִיחַ מְאוֹרוֹ. וְיֹאמַר: הִנֵּה חֶפְצְכֶם אֲשֶׁר דְּרַשְׁתֶּם. וְעוֹד אֶעֱשֶׂה שִׁירִים עַל הָעִנְיָן אֲשֶׁר בִּקַּשְׁתֶּם.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

גְּבִיר הַדּוֹר פְּאֵר מִשְׂרָה וּמוּסָר

וּמִזֶּרַע יְקָר נִבְרָא וְנוֹצַר.

פְּרִי הָדָר וְאוֹר נֶאְדָּר וְנֶהְדָּר

וְדַרְכֵי יוֹצְרוֹ שָׁמַר וְנָצַר.

וּמִשְׂרָתוֹ בְּיִרְאַת אֵל קְשׁוּרָה

וְאַחַר בּוֹרְאוֹ יָרוּץ וְיִפְצַר.

וְקָשַׁר הַיְקָר כֶּתֶר לְרֹאשׁוֹ

וְגָבַר רוֹזְנֵי דוֹרוֹ וְעָצַר.

יְרִימוּ צוּר וְיוֹצְרוֹ יִצְּרֵהוּ

בְּצָרָתוֹ כְּמִבְצָר מִמְּגוֹר צַר.

וְכַאֲשֶׁר הִשְׁלִים אִגַּרְתּוֹ. וַיְכַל פְּלִיאוֹת מַחְשַׁבְתּוֹ. וַיִּשְׁבֹּת מִכָּל מְלַאכְתּוֹ. אָמְרוּ לוֹ: הִבְהַלְתָּ כָל רַעְיוֹן. וְהִדְבַּקְתָּ לְחִכּוֹ כָּל לָשׁוֹן. אַךְ תַּעֲשֶׂה לָנוּ אִגֶּרֶת חֲמוּדָה לְכָל דּוֹרֵשׁ . וּמִמֶּנָּה הָרֵי"שׁ גָּרֵשׁ תְּגָרֵשׁ. וּבַדָּבָר הַזֶּה תְּגַלֶּה יִתְרוֹן מַעֲלָלֶיךָ. וְיַרָאֶה אֶל עֲבָדֶיךָ פָּעֳלֶךָ.

וַיַּעַן וַיֹּאמַר: אֲצִילֵנוּ. גְּאוֹן עֲדָיֵינוּ. וּשְׁבִיס חֲלָיֵינוּ. נֹגַהּ לְעֵינֵינוּ . וְזִיו לְאִישׁוֹנֵנוּ. כּוֹכַב נִשְׁפֵּנוּ. וְהוֹד עַפְעַפֵּינוּ. מְמַלֵּא יְמִינֵנוּ בְּהוֹנֵנוּ. וְכַפֵּנוּ בְכַסְפֵּנוּ. שָׁמְעוֹ יִנְעַם לְכָל אָזְנַיִם . וְיִמְתַּק לְכָל שְׂפָתַיִם. וְצַלְמוֹ תַאֲוָה לָעֵינַיִם. שְׁמוֹ מַחְמַד לְבָבוֹת. וּפָנָיו שֶׁמֶשׁ בְּלִי עָבוֹת. וְיָדָיו גֶּשֶׁם נְדָבוֹת. וּמִידּוֹתָיו עִם מִדּוֹת הָאָבוֹת מְשֻׁלָּבוֹת. הָאֵל מוֹשִיעַ לְעַמּוֹ שְׁלָחָהוּ. וְנָגִיד לְאֻמּוֹ מְשָׁחָהוּ. לְהָכִין וּלְהָפִיק לַנּוֹפְלִים תְּקוּמָה. וּלְאֵין אוֹנִים עָצְמָה. וְלִיגִיעֵי כֹחַ נוֹחַ. וְלַנּוֹדְדִים מָנוֹחַ. לְהַשְׁקוֹת צְמֵאִים בְּצִמְאָה. נַחֲלֵי דְבַשׁ וְחֶמְאָה. וּלְהַחֲזִיק יְדֶי כוֹשְׁלִים . וּלְהָשִׁיב בְּנִדְבוֹתָיו הָעֲבָדִים כְּמוֹשְׁלִים. חַסְדּוֹ סוֹעֵד שְׁפָלִים וְנֶחֱשָׁלִים. וְצִדְּקוֹ מְאַמֵּץ נוֹפְלִים וְנִכְשָׁלִים. וְיָדָיו נוֹתְנוֹת חֵפֶץ שׁוֹאֲלִים. יָחִיד בֶּהֲמוֹנוֹ. וּמוֹפֵת בֵּין אַנְשֵׁי זְמַנּוֹ. וּמוֹשֵׁל בִּקְהִלָּתוֹ. וְנָגִיד וּמְצַוֶּה עַל עֲדָתוֹ. כְּלִיל הַצֶּדֶק. מַחֲזִיק כָּל בֶּדֶק. שׁוֹשַׁן תּוֹךְ חֶדֶק. נְתִיב אֲבוֹתָיו לָמָד. וּבְצֵל תְּבוּנָה לָשֶׁבֶת חָמָד. לְמַעַן הַשִּׂיג כָּבוֹד. וְלִנְחֹל גְּדֻלָּה וָהוֹד. מַה טּוּבוֹ וּמַה יָפְיוֹ. וּמַה נָּעִים יְדִידוּתוֹ וְצִבְיוֹ. וּמַה נֶּחְמָד גְאוֹן עֶדְיוֹ. וּמַה נְּכוֹחִים פְּעָלָיו. וּמַה נְּעִימִים חֲבָלָיו. וּמַה טוֹבוּ אָהֳלֵי מִדּוֹתָיו. וּמַה יְּדִידוֹת מִשְׁכְּנוֹתָיו. כִּנְחָלִים נִטָּיוּ . בְּתוֹכָם יְעֵפִים יִשְׁלָיוּ. וְנוֹדְדִים יֶחֱסָיוּ. וּצְמֵאִים מֵי יֶשַׁע יִשְׁתָּיוּ. וְהַמֵּתִים שָׁמְעוּ זִכְרוֹ וְחָיוּ. וְהַחַיִּים לוּלֵי חַסְדּוֹ אָבָדוּ וְהָיוּ כְלֹא הָיוּ. לְשׁוֹן הַיָּמִים יָשִׂיחַ מַהֲלָלָיו . וּפִי הַזְּמַן יַבִּיעַ טוּב מִפְעָלָיו. וְהַחֵן נָטוּי אֵלָיו. וְהַצֶּדֶק בְּלִבּוֹ תָּקַע אֹהָלָיו. וּבִשְׁמוֹ יִשָּׂא דְגָלָיו. הָאֵל לְנֵס עַמִּים יְשִׂימֵהוּ. וּמִשּׂוֹטְנָיו יַעֲצִימֵהוּ. וְעַל מֵי מְנוּחוֹת יְנַהֲלֵהוּ. וִינַשְּׂאֵהוּ וִינַטְּלֵהוּ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

יְדִיד הָאֵל כְּלִיל מִדּוֹת חֲמוּדוֹת

אֲבָל לִבּוֹ כְמַעְיָן לַתְּעוּדוֹת.

נְגִיד עַם אֵל וְהוּא יָחִיד בְּתֵבֵל

וְכֵן הֵמָּה תְּהִלּוֹתָיו יְחִידוֹת.

וּבָנָה מִגְדְּלוֹת חֶסֶד וְצֶדֶק

וְשָׂם מִדּוֹת אֲבוֹתָיו לוֹ יְסוֹדוֹת.

וְיַעַן כִּי חֲמוּדוֹת הֵם פְּעָלָיו

שְׁמוֹ נוֹדַע בְּעַם אֵל אִישׁ חֲמוּדוֹת.

אָמַר הַמַּגִּיד: וְכִשְׁמֹעַ הָעָם פְלִיאוֹת רַעְיוֹנוֹ . וְנוֹרְאוֹת הֶגְיוֹנוֹ. וְצַחוּת לְשׁוֹנוֹ. נְשָׂאוּהוּ עַל כַּפַּיִם וּבִכְתֵפַיִם . וּפָתְחוּ לוֹ הַיָּדַיִם. עַד עֲמָסוּהוּ בְּרוֹב נִדְבָתָם. וְנִלְאָה לְהוֹדוֹת עַל טוֹבָתָם. וְכַאֲשֶׁר נָסַע מֵהֶם. קָרָא שְׁלוֹמוֹ עֲלֵיהֶם.

וַיַּעַן וַיֹּאמֶר לָהֶם:

אָמְנָם אֲנִי חֶבֶר מְחַבֵּר שִׁיר אֲשֶׁר

נַפְשׁוֹת מְקַנְאִים בַּעֲדוֹ תִּיגַעְנָה.

אֶשְׁקֹט וְאָנוּחַ בְּבֵיתִי אַךְ מְלִי-

צוֹתַי עֲדֵי מִזְרָח וְיָם תִּסַּעְנָה.

שַׂמְתִּים כְּמוֹ חִיצִּים מְרוּקִים בַּזְּמַן

אוּלָם סְגוֹר לֵב אוֹיְבַי תִּקְרַעְנָה.

הוֹלְכוֹת בְּלִי רֶגֶל עֲדֵי מִזְרָח וְיָם

גַּם הַרֲרֵי עוֹלָם בְּעֹז תִּקְרַעְנָה.

אַחְבִּיר לְשׁוֹן עֵבֶר וְקֵדָר וַאֲרָם

כִּי לִי שְׁלָשְׁתָּן בֶּאֱמֶת תִּכְרַעְנָה.

אַבְעִיר מְדוּרָתָם עֲדֵי שַׁחַק. וְיוֹם

וָלֵיל לְרוֹם עוֹלוֹת וְלֹא תִּשְׁקַעְנָה.

מִזִּיו מְאוֹרִי יֶחֱזוּ עִוְרִים וְשִׂפְ-

תוֹת אִלְּמִים טוּב נָאֳמִי תַבַּעְנָה.

נַפְשׁוֹת מְתֵי הַשִּׁיר בְּקִנְאָה יֶהֱמוּ

עָלַי בְּכָל רֶגַע וְלֹא תִרְגַּעְנָה.

הֵם רוֹצְצוֹת אַחְרַי לְהַשִּׂיג אֶל מְרוֹם

שִׁירִי וְתִיגַעְנָה וְלֹא תַגַּעְנָה.
