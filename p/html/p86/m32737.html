מִכִּתְבֵי יָד שׁוֹנִים שֶׁל הַתַּחְכְּמוֹנִי שַׁעַר חֲמִשִׁים

(לפי שד"ל, אגרות, חלק ט', ונייבואר מלאכת השיר, עם תקונים לפי המשקל והענין)

וְעָשִׂיתִי לְאִישׁ קִדַּמְתִּיו בִּתְהִלָּתִי וְלֹא הִכִּיר עֵרֶךְ גְּדֻלָּתִי:

לְבָבִי קַח לְךָ מוּסָר וְשׁוּר כִּי

תְּהִלָּתָךְ נְבָלִים כּוֹסְסוּהָ.

חֲשׂף שִׁירָתְךָ לִפְנֵי חֲכָמִים

וְלֹא לִפְנֵי סְכָלִים יַחְמְסוּהָ.

וְאִם תֶּשְׁלִיךְ לְךָ אֶבֶן יְקָרָה

לְרַגְלֵי הַכְּלָבִים יִרְמְסוּהָ.

וְעָשִׂיתִי לְאַנְשֵׁי הַדּוֹר לִזְכֹּר עִנְיְנֵיהֶם וְרֹעַ מַעֲשֵׂיהֶם:

הָהּ כִּי בְנֵי הַדּוֹר אֲרָיוֹת טוֹרְפִים

כִּי לַעֲשׂוֹת נֵזֶק יְדֵיהֶם פּוֹשְׁטִים.

לָכֵן שְׁמֹר אֶת נַפְשְךָ מֵהֶם, הֲכִי

כָּל הַיְקוּם לִסְטִים וְהַכֹּל שׁוֹחֲטִים.

וְעָשִׂיתִי עַל אִישׁ נָבָל וְכִילַי. וְאֵין בְּפִיהוּ רַק לֹא וְאוּלַי. וְיָמוּת בַּצָּמָא וְהוּא יוֹשֵׁב עַל אוּבַל אוּלַי:

שְׁאֵלוּנִי עַל הַנְּדָבָה וּשְׁלֹמֹה

עֲנִיתִים: הִיא וְהוּא כְּאֵשׁ עִם נוֹזְלֵי עָב.

לְשׁוֹן כּוֹסוֹ דָּבַק לְחִכּוֹ בַּצָּמָא

וְשִׂפְתֵי שֻׁלְחָנוֹ עֲטוּפִים בָּרָעָב.

לְהָסֵךְ מֵי רַגְלָיו מְפַחֵד פֶּן יִצְמָא

וְכֵן יִירָא לִבְדֹּק נְקָבָיו פֶּן יִרְעָב.

וְעָשִׂיתִי עַל עִנְיַן הַדְּמָעוֹת. אֲשֶׁר מִן הַלֵּב נוֹבְעוֹת:

כָּל הַדְּמָעוֹת יִזְּלוּ גַּם יַחְדְּלוּ

עִתִּים, לְבַד דִּמְעוֹת עֲשׁוּקֵי אַהַב.

כִּי אֵד בְּכִי עֵת מִכְּבֵדָם יַעֲלֶה

יִהְיֶה לְמַיִם בַּעֲלוֹת הַלַּהַב.

וְעוֹד עָשִׁיתִי עַל הָעִנְיָן. וְלִשְׁאֹב בְּמַעְיָן.

אִלּוּ כְפִי אֵידִי דְמָעַי יִזְּלוּ

לֹא דָרְכָה רֶגֶל אֱנוֹשׁ יַבֶּשֶׁת.

אָמְנָם דְּעוּ כִּי לֹא לְמֵי נֹחַ לְבַד

כִּי גַם לְדִמְעִי נִרְאֲתָה הַקֶּשֶׁת.

וְעָשִׂיתִי בְחֵרוּף הַנּוֹכְלִים. וְגִדּוּף הַנְּבָלִים:

הֵם כַּבְּהֵמוֹת מִבְּלִי זָנָב אֲבָל

לָהֶם זְנָבוֹת רַק בְּעָרְפָּם נִקְשְׁרוּ.

אוֹת עַל קְפֹץ יָדָם בְּעֵת תַּחְפֹּץ רְאוֹת

שׁוּר מֵאֲחוֹרֵיהֶם יְדֵיהֶם אֻסְּרוּ.

אִם תֶּאֱסֹף מִנְיָן לָךְ מֵהֶם תְּפִ-

לָּתָךְ אֲסוּרָה כִּי עֲשָׂרָה יַחְסְרוּ.

וְעָשִׂיתִי לְאִישׁ מִן הַנְּדִיבִים. וְגֶזַע הַחֲשׁוּבִים:

אָמַר אֱלֹהִים לַנְּדָבָה שַׁחֲרִי

הַשַּׂר לְמַעַן בָּךְ מְאֹד רַב גִּילוֹ.

עִזְבִי בְנֵי תֵבֵל וְלוֹ בִלְבַד לְכִי

כִּי הוּא אֲדוֹנַיִךְ וְהִשְׁתַּחְוִי לוֹ.

וְעָשִׂיתִי עַל עִנְיַן הַנְּדוֹד. הַמַּבְעִיר בְּלִבִּי כִידוֹד:

יְדִידַי יִדּוֹדוּן וְיָרִיעוּ רֵעַי

וְלִנְדוֹדָם אֶמְצָא יְקוֹד אֵשׁ בִּצְלָעָי.

וּמִי יִתֵּן קָרְבוּ חֲשׁוּקַי כִּמְצוּקַי

וְדוֹדַי כִּנְדוֹדַי וְרֵעַי כִּדְמָעָי.

וְעוֹד עָשִׂיתִי בְעִנְיַן הַחֵשֶׁק. בִּצְבִי הַלְּבָבוֹת עוֹשֵׁק:

צְבִי חֵן בּוֹ עֲשָׁקַנִי בְּיָפְיוֹ

וְהִשְׁבִּית כָּל רְנָנַי גַּם שְׂשׂוֹנָי.

וְאֵין לִי מִלְּחָיָיו הַנְּעִימִים

לְבַד הָאוֹר אֲשֶׁר אֶרְאֶה בְעֵינָי.

יְרִיבוּנִי עֲלֵי חִשְׁקִי מְרִיבִים

לְהוֹסִיף אֵיד וְיָגוֹן עַל יְגוֹנָי.

וְנִשְׁבַּעְתִּי לְבַל אֶשְׁמַע בְּקוֹלָם

אֲבָל כִּי אֶסְגְּרָה עֵינַי וְאָזְנָי.

וְהִיא נַפְשִׁי – וְאֵיךְ אֶמְאַס בְּנַפְשִׁי?

הֲלֹא אֶחְטָא לְצוּר שׁוֹכֵן מְעוֹנַי!

וּמָה אֶקֹּב וְלֹא קַבֹּה אֱלֹהִים

וּמָה אֶזְעֹם וְלֹא זָעַם אֲדֹנָי.

וְעָשִׂיתִי בַּעֲדִינָה עַל אִישׁ מֵהַנְּבָלִים וּשְׁמוֹ צְדָקָה בֶּן עֲתִירָה. כְּשֶׁמֵּתָה אִמּוֹ וְנִפְטְרָה:

שְׁאֵלוּנִי סְפֹד אֶל אֵם צְדָקָה –

עֲנִיתִים: לֹא, עֲדֵי יִהְיֶה כְּאִמּוֹ.

אֲקוֹנֵן לוֹ בְּעֵת יִהְיֶה שְׁכֵנָהּ

וְיָשִׂים מוּל מְקוֹמָהּ אֶת מְקוֹמוֹ.

וְאוֹתוֹ תִהְיֶה נוֹשֵׂאת בְּקִבְרָהּ

לְאַט כַּאֲשֶׁר נְשָׂאָהּ הוּא בְשִׁכְמוֹ.

וְעָשִׂיתִי בְעָרֵי אַשּׁוּר עַל עִנְיַן הַקְּהִלוֹת אֲשֶׁר חָזִיתִי. וּבְאַרְצוֹת מִזְרָח רָאִיתִי:

שִׁמְעוּ בְנֵי שֵׂכֶל פְּלִילִיּוֹת

בָּם תִּלְמְדוּ דֵעוֹת וְתֻשִׁיּוֹת.

בָּאתִי לְנֹא אָמוֹן וְחָנִיתִי

בֵּינָם כְּמוֹ הַדָּג בְּתוֹךְ צִיּוֹת.

וָאֶמְצְאָה אִישִׁים בְּמִצְרַיִם

אֵין בָּם לְאַנְשֵׁי הַיְקָר מִחְיוֹת.

וָאֶעֱבֹר מִשָּׁם לְדַמֶּשֶׂק

אֶל עַם אֲשֶׁר תָּעוּ כְמוֹ שֵׂיוֹת.

וָאֶשְׁכְּחָה אֵידָם בְּעַם צוֹבָה

כִּי הֵם כְּחֶרֶב לָהּ שְׁתֵּי פֵיוֹת.

וָאֶמְאֲסָה כַלְנֵה הֲכִי לַעְצֹר

הוֹנָם יְדֵיהֶם תּוֹקְפוֹת הוֹיוֹת.

אוּלָם בְּאַשׁוּר נִשְׁכְּחוּ כֻלָּם

כִּי הֵם בְּהֵמוֹת אוֹ כְמוֹ חַיּוֹת.

אִם אֵרְדָה כֵן כָּל-אֲשֶׁר אֵלֵךְ

אֶמְצָא בְנֵי בָבֶל בְּתַחְתִּיּוֹת.

וְעָשִׂיתִי עַל מוֹנֵעַ טוּבוֹ מִן הַבּוֹטְחִים בּוֹ:

אִם יִצְבֹּר עשֶׁר אִישׁ כִּילַי

וִיהִי עֵץ שָׁתוּל עַל יוּבַל.

בַּחְצִי יָמָיו יַעַזְבֶנּוּ

וּבְאַחְרִיתוֹ יִהְיֶה נָבָל.

וְעָשִׂיתִי עַל מְנוֹרוֹת. בָּהֶם שְׁנֵי נֵרוֹת. וְאוֹרוֹתָם נִפְרָדוֹת וּמְחֻבָּרוֹת:

עוּרָה יְדִידִי וְאַל תִּישַׁן וְקוּמָה שְׂמַח

בִּשְׁנֵי מְאוֹרוֹת בְּלַיִל זֶה בְזֶה דָבְקוּ.

קוּמָה רְאֵה נֵר אֲשֶׁר אוֹרוֹ כְּלֶחְיָךְ וְכִצְ-

לָעֵי פְּתִילָיו וְכֵן דַּלּוּ וְכֵן דָּלְקוּ.

פַּעַם יְנוּדוּן בְּלִי רָצוֹן וְיִפָּרְדוּ

רֶגַע וּפַעַם יְרוּצוּן חִישׁ וְיֵאָבְקוּ.

יִדְמוּ לְחוֹשֵׁק וְחָשׁוּק רָחֲקוּ בַנְּדוֹד

וּבְעֵת אֲשֶׁר נִפְגְּשׁוּ רָצוּ וְהִתְחַבְּקוּ.

וְעָשִׂיתִי עַל הַנָּשִׂיא רַבִּי מָכִיר. יוֹצְרוֹ אוֹתוֹ לְטוֹב יַזְכִּיר:

יוֹם נִבְרֵאתָ אָמַר הָאֵל

הֵן עַבְדִּי אֶתְמָךְ-בּוֹ לָעַד.

בִּדְבַר פִּיהוּ אָפִיק מַרְפֵּא

וּבְמַאְמָרָיו לִבּוֹת אֶסְעָד.

וּפְרִי עֵדֶן וּצְרִי חַיִּים

בּוֹ לִדְוֵה לֵב יִהְיֶה נוֹעָד.

כָּל-אִישׁ לָקַח חֶלְקוֹ וּלְמַ-

כִיר נָתַתִּי אֶת הַגִּיל-עַד.

וְעָשִׂיתִי עַל אִישׁ הוֹלֵךְ בְּמֵישָׁרִים. וּמַרְבֶּה חֲסָדִים בְּמִסְתָּרִים:

יְרוֹצֵץ בְּיוֹמוֹ עֲשׂוֹת טוֹב וְלֵילוֹ

לְשַׁחֵר פְּנֵי אֵל יְעוֹרֵר שְׁחָרִים.

וְיַרְבֶּה חֲסָדָיו בְּסֵתֶר לְדַעְתּוֹ

אֱמֶת כִּי בְיַד אֵל מְגִלַּת סְתָרִים.

וְאֵלֶּה עָשִׂיתִי בִימֵי נַעֲרוּתִי. כְּשֶׁנִּפְטְרָה

הוֹרָתִי ת"מ (=תבורך מנשים. פה המקום האחד אשר יזכיר המשורר מות אמו בעודנו נער):

לְבָבִי, אֵיךְ זְמַן פָּרַץ גְּדֵרָךְ

וְהֵסִיר מֵעֲלֵי רֹאשָׁךְ פְּאֵרָךְ!

בְּמוֹת אֵם הַיְקָר תַּמָּה כְבוֹדָךְ

אֲשֶׁר הָיְתָה כְמוֹ שֶׁמֶשׁ בְּדוֹרָךְ.

תְּאֻשַּׁר בֵּין תְּמִימוֹת לַעֲדֵי עַד

וּמִנָּשִׁים בְּעַד אֹהֶל תְּבֹרָךְ.

וְעָשִׂיתִי עַל עֲדַת נְבָלִים. וְאֵין בָּהֶם אֲצִילִים:

בְּגוֹי נָבָל כְּבוֹדִי שָׁב מְשֻׁסֶּה

וּמַעְלָתִי בְּרַגְלֵיהֶם רְמוּסָה.

אֲשֶׁר יָדָם בְּתוּלָה לַנְּדָבָה

וְלַכִּילוּת בְּעוּלָה גַּם כְּנוּסָה.

בְּמוּמָם נִטְמְאָה תֵבֵל וְלוּלֵא

בְּרִית מַבּוּל – שְׁטָפָם יָם וְכִסָּה.

וְנַפְשָׁם הִיא קְרוּצָה מִנְּבָלוֹת

וּמִכָּל-מוּם וְקַלָּסָה עֲמוּסָה.

וְצִיר מָוֶת בְּעֵת יָבוֹא לְהָרְגָם –

יְבַקֵּשׁ מִבְּאשׁ נַפְשָׁם מְנוּסָה.

וְיוֹרִידָהּ לְשַׁחַת עַל קְצָת עֵץ

לְטֻמְאָתָהּ וְהִיא עָלָיו מְאוּסָה.

וְלוּ יִלְמַד בְּשַׁבָּת יִמְצְאָה שָׁם

הֲכִי נַפְשָׁם מְטַמְּאָה הִיא בְּמַשָׂא.

וְאֵלֶּה עָשִׂיתִי לְאִישׁ בְּיָדוֹ נְדָבוֹת. וּפָנָיו חֶדְוַת לְבָבוֹת:

תְּהִלּוֹתַי שְׁלַחְתִּים אֶל יְדֵי שַׂר

מְרִיקִים מֵי נְדָבוֹת מִהְיוֹתָם.

וְאֶפְחַד אִם יְשַׁו יָדָיו עֲלֵיהֶם

הֲכִי יִמְחַק וְלֹא יוּכַל קְרֹאתָם.

וְאֵלֶּה עָשִׂיתִי בְּמוּסַר הַמַּשְׂכִּילִים. לְהַזְהִירָם מֵעֲלוֹת לְמַעֲלוֹת הַגְּדוֹלִים:

הַמַּעֲלוֹת שָׁלשׁ. וְאִם בָּם תַּחֲשֹׁק

מִן הַשְּׁפָלָה נוּס וְהָעֶלְיוֹנָה.

כִּי הַקְּצָווֹת הֵם מְקוֹם פַּחַד וְהָ-

אֶמְצַע מְקוֹם שָׁלוֹם וְהַתִּיכוֹנָה.

וְעָשִׂיתִי לְאִישׁ נָדִיב אֵין עֲרֹךְ אֵלָיו בְּנֹעַם מִפְעָלָיו:

שׁוֹאֵל בְּעַד נִדְבַת בְּחִיר אֵל הַגְּבִיר –

צוּרֵי סְלָעִים דָּשְׁאוּ מִנּוֹזְלָיו.

מוֹשֵׁל אֲשֶׁר מָשַׁל בְּרֹאשׁ שָׂרִים אֲבָל

נִכְנַע לְכָל-דּוֹרְשָׁיו עֲדֵי הֵם מוֹשְׁלָיו.

מַה נּוֹאֲלוּ הַחוֹשְׁדִים כִּי יִגְמְלֵם

חֶסֶד, וְלֹא יָדְעוּ הֲכִי הֵם גּוֹמְלָיו.

נַפְשׁוֹ לְאֵל מָכַר וְעֵינָיו אֶל יְדֵי

שׁוֹאֵל כְּעֵין עֶבֶד אֶל יַד גּוֹאֲלָיו.

יִתֵּן וְיִתְנַדֵּב כְּנִדְבַת עַנֲנֵי

שַׁחַק וְיִתְחַנֵּן תְּחִנַּת שׁוֹאֲלָיו.

וְאֵלֶּה עָשִׂיתִי עַל אַנְשֵׁי מִזְרָח בְּעֵת אוֹתָם נִסִּיתִי וּבְצִלָּם חָסִיתִי:

יְגַדַּע אֵל בְּאַף קֶרֶן זְמַנִּי

אֲשֶׁר פִּזֵּר חֲפָצַי עֵת אֲסַפְתִּים!

נְדִיבָיו נִבְרְאוּ טֶרֶם הֱיוֹתִי

וְעֵת צֵאתִי לְאוֹר עוֹלָם חֲסַרְתִּים.

כְּאִלּוּ שָׁמְעוּ שִׁמְעִי וּבָרְחוּ

וְעַל כֵּן עֵת דְּרַשְׁתִּים לֹא מְצָאתִים.

וְעוֹד עָשִיתִי עַל הָעִנְיָן הַזֶּה. וְזָכַרְתִּי שְׁלשֶׁת גִּבּוֹרֵי הַשִּׁיר אֲשֶׁר הָיוּ בִּסְפָרַד:

אֲבוֹת הַשִּׁיר שְׁלֹמֹה עִם יְהוּדָה

וּמשֶׁה זָרְחָה שִׁמְשָׁם בְּמַעְרָב.

הֲכִי מָצְאוּ בְּדֹרוֹתָם נְדִיבִים

וּמָכְרוּ אֶת פְּנִינֵיהֶם בְּהוֹן רָב.

וְנִבְרֵאתִי אֲנִי עֵת אָבְדוּ כָל

נְדִיבֵי לֵב וּבָא שִׁמְשָׁם וְעָרָב.

וְהֵמָּה מָצְאוּ נַחְלֵי עֲדָנִים

וּמָצָאתִי אֲנִי חֹרֶב וְשָׁרָב.

וְעָשִׂיתִי בְדַמֶּשֶׂק כְּשֶׁנָּסַע רַבִּי משֶׁה הָרוֹפֵא לְאֶרֶץ חֲמָת:

שְׁאֵלוּנִי, הֲלֹא תִשְׂמַח בְּחֶבְרַת

בְּנֵי מוּסָר. בְּנֵי שֵׂכֶל וְתֻמָּה?

עֲנִיתִים: בַּגְּבִיר משֶׁה מְשׂוֹשִׂי

וּבִלְתּוֹ הַנְּשָׁמָה הִיא נְשַׁמָּה.

וְהוּא נֶפֶשׁ וְאַנְשֵׁי הַזְּמַן גּוּף

וּמַה מּוֹעִיל בְּגוּף מֵאֵין נְשָׁמָה?

וְעוֹד עָשִׂיתִי בְּמִבְחַן הַיְדִידִים:

נַסֵּה יְדִידֶיךָ בְּעֵת מָצוֹר

וּטְעַם בְּחֵךְ שִׂכְלָךְ פְּרִי דָתָם.

אִם לֹא יְהִי לִבָּם לְךָ נֶאְמָן

טוֹב מִיְּדִידוּתָם פְּרִידָתָם.

וְאֵלֶּה עֲשִׂיתִים בְּמַהֲלַל אֲצִילִים. אַנְשֵׁי חַיִל רַבֵּי פְעָלִים:

אִישִׁים אֲשֶׁר אֶל אוֹר כְּבוֹדָם יֵלְכוּ

כָּל-יוֹשְׁבֵי תֵבֵל וְשׁוֹכְנֵי אֶרֶץ.

כִּי בָם שְׂמָחוֹת חִדְּשָׁה תֵבֵל וּבָם

חֶסֶד אֲדֹנָי מָלְאָה הָאָרֶץ.

בַּבִּין וּבַמּוּסָר וּמִדּוֹת כָּל יְקָר –

לָהֶם לְבַדָּם נִתְּנָה הָאָרֶץ.

יִגְאוּ וְאוֹיְבֵיהֶם עֲפָרָם יִלְחֲכוּ

עַד כִּי לְשׁוֹנָם תִּהֲלַךְ בָּאָרֶץ.

רֹאִים גְּדֻלָּתָם וְשִׂכְלָם יֹאמְרוּ

אַךְ יֵשׁ אֱלֹהִים שׁוֹפְטִים בָּאָרֶץ.

וְעוֹד עַל הָעִנְיָן:

1.

אִשֵּׁי לְבָבוֹת הֲלֹא אֶשְׁלַח [בְּיַד] אֶשְלַח

בִּשְׂפַת צְבִי חֵן אֲשֶׁר גּוּפוֹ בְּדֹלַח לַח.

אוֹחִיל תְּשׁוּבַת שְׁלוּחִי לוֹ וְלֹא יִצְלַח

הוּא יֶחֱטַא לוֹ וְאֶתְחַטָּא וְלֹא יִסְלַח.

2.

שָׂרִיג אֲשֶׁר מֵעֲצֵי גַן הַיֳּפִי נִקְצַב

דַּר לֶחֱיוֹ מִמְּקוֹם מֶחְצַב יְקָר חֻצָּב.

כִּנְצִיב בְּדֹלַח עֲלֵי קוֹמַת כְּפוֹר נִצָּב

בִּדְבַשׁ שְׂפָתוֹ מְשׂוֹשׂ לֵבָב אֱנוֹשׁ נֶעְצָב.

3.

לוּ מֵעֲסִיס חִכְּךָ עֹפֶר מְעַט אֶשְאָב

אַחְלִיץ עֲצָמַי אֲשֶׁר בָּלוּ וְלֹא אֶדְאָב.

אֶשְׁתַּחֲוֶה נֶגְדְּךָ כָל-יוֹם וְלָךְ אֶתְאָב

אֶדְמֶה כְּעֶבֶד לְךָ אוֹ בֵן יְכַבֵּד אָב.

4.

נִשְׁבַּע צְבִי עַד יְמִיתֵנִי לְבַל יִשְׁקֹט

נַפְשִׁי עֲלֵי זֹאת [אַף בְּחַיַּי] נָקְטָה נָקֹט.

אוֹחִיל רְצוֹנוֹ וְכִסְלִי יוֹם לְיוֹם יָקֹט

אֶרְאֶה עֲלֵי לֶחֱיוֹ הַמָּן וְלֹא אֶלְקֹט.


וְעוֹד עַל הָעִנְיָן:

עָפְרַת אֲהָבִים בְּתוֹךְ בָּנוֹת מְאֻשֶּׁרֶת

נַפְשִׁי אֲהָהּ בַּעֲבוֹת חִשְׁקָהּ מְקֻשֶּׁרֶת.

חִשְׁקִי צְפַנְתִּיו – וְדִמְעָתִי כְּכִנֶּרֶת

לֹא תַעֲזֹב לֶאֱנוֹשׁ אַהְבָה מְסֻתֶּרֶת.

וְאֵלֶּה עָשִׂיתִי עַל הַנָּשִׂיא רַבִּי יוֹסֵף בֶּן שׁוֹשָׁן זַ"ל כְּשֶׁנִּפְטַר:

יִכְלוּ זְמַנִּים וְלֹא יִכְלוּ יְגוֹנֵינוּ

הַאִם לְבָבוֹת כְּצוּר אוֹ יָם בְּעֵינֵינוּ?

שָׂרֵי זְמַן נִבְהֲלוּ עַל מוֹת גְּבִירָם אֲשֶׁר

מָשְׁחוּ לְמֶלֶךְ וְאָזְלַת יַד גְּאוֹנֵינוּ.

בּוֹשֵׁשׁ בְּחִיר אֵל וְלִזְמַן יֹאמְרוּ קוּם עֲשֵׂה

לָנוּ אֱלֹהִים אֲשֶׁר יֵלְכוּ לְפָנֵינוּ.

וְעָשִׂיתִי לְאִישׁ יוֹדֵעַ נַגֵּן בְּכִנּוֹר:

שְׂאוּ נָא אֶת שְׁלוֹמִי אֶל יְשַׁעְיָה

אֲשֶׁר לָקַח תְּעוּדָה לוֹ לְרַעְיָה.

מְעוֹרֵר מֵיתְרֵי כִנּוֹר לְשׁוֹרֵר

כְּמוֹ לֵוִי בְּמִשְׁמֶרֶת לְוִיָּה.

וְכַיּוֹנֵק בְּחֵק אִמּוֹ מְצַחֵק

וְיָרִים קוֹל בְּרִנָּה לֹא בְּבִכְיָה.

וְלוּ עַל מֵת יְנַגֵּן עוֹרְרָהוּ

וְעָלָיו צָלְחָה רוַּח וְחָיָה.

(בסוף כל הספר, אחרי המלים “להעלות נר תמיד” עמוד 433, נמצאו לפי נייבואר בכ"י חרוזי-סיום אלה, אבל נקל לראות שאינם מהמחבר).


שָׁלְמוּ מְזִמּוֹת בִּין אֲשֶׁר חִבֵּר

הָרַב יְהוּדָה הוּא בְחָכְמָתוֹ.

יָקְרוּ פְנִינֵיהֶם וְרִקְמָתָם

מִכָּל זְהַב אוֹפִיר וְזִמְרָתוֹ.

יִחְיוּ נְפָשִׁים אַחֲרֵי מוֹתָם

הָעֵת שְׁמֹעַ טוּב מְלִיצָתוֹ.

אֶל כָּל-נְדִיב לֵבָב וְגַם שׁוֹעַ

שָׁלַח נְגִיד שִׁירָיו תְּמוּרָתוֹ.

בֶּאְמֶת כְּיוֹצֵא (?) בַּגְּבִיר הָרַב

רַבִּי יְשׁוּעָה שָׂם מְגַמָּתוֹ.

לַעְשׂוֹת רְצוֹן קוֹנוֹ וְלִשְׁתַּעְשֵׁעַ

תָּמִיד בְּחֻקֵּי אֵל וְתוֹרָתוֹ.

אֵל יַעְזְרֵהוּ גַּם יְשַׁלֶּם לוֹ

טוּב מַעֲשָׂיו וּשְׂכַר פְּעֻלָּתוֹ.

יִזְכֶּה עֲלוֹת אֶל הַר קְדוֹשׁ עֶלְיוֹן

לִהְיוֹת פְּקִידִים תּוֹךְ חֲצַר בֵּיתוֹ.

גַּם בַּחֲמוּד לִבּוֹ (=בבנו) מְאֹד יִשְׂמַח תַּחת אֲבוֹתָיו שַׂר יְשַׁו אוֹתוֹ.

וְשָׁלוֹם עַל יִשְׂרָאֵל. וּבָא לְצִיּוֹן גּוֹאֵל.

תם
