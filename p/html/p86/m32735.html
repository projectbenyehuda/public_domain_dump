ספר תחכמוני - שַׁעַר אַרְבָּעִים וְתִשְׁעָה

בְּשֶּׁבַח פֵּרוֹת אִילָנֵי הַגָּן


נְאֻם הֵימָן הָאֶזְרָחִי: נִכְסְפָה וְגַם כָּלְתָה נַפְשִּׁי. בִּימֵי תַעֲנוּגִי וְנַפְשִׁי. וּבִהְיוֹתִי כְפֶרֶא חָפְשִׁי. לְהִתְעַלֵּס בְּצִלְלֵי הַגַּנִּים. וְלִרְעוֹת בַּשּׁוֹשַׁנִּים. וַיְהִי בִקְצָת הַיָּמִים הִשְׁכַּמְתִּי בְּטֶרֶם יַשְׁכִּימוּ הַמְּאוֹרִים. וְעוֹרַרְתִּי שְׁחָרִים. וְיָצָאתִי בַכְּפָרִים. לְהִתְעַלֵּס בַּאֲהָבִים. תַּחַת צִלְלֵי הָעֲרָבִים. וְהִנֵּה רוּחַ שְׁחָרִים פָּנַי מְשַׁחֶרֶת. תְּגַלֶּה לִי אַהֲבָה מְסֻתֶּרֶת. וּבְלִי לָשׁוֹן מְדַבֶּרֶת. קָרְאָה לִי לְשָׁלוֹם. וְאָמְרָה לִי: גְּשָׁה נָא עַד הֲלֹם. אַבִּיעַ סוֹדוֹת אֵלֶיךָ. וְאָצַלְתִּי רוּחַ אַהֲבָתִי עָלֶיךָ. וְהִנִּה הֵבֵאתִי לְךָ מִפְּרִי אַדְמָתִי. וּמֵרֵאשִׁית תְּבוּאָתִי. רֵיחַ עֲרוּגָתִי. וְרֶקַח אַבְקָתִי. קַח נָא אֶת בִּרְכָתִי. וְאִם תָּסוּר אֵלַי לְהִתְעַדֵּן בְּאֶשְׁנַבֵּי צְלָלַי. וּבְמַשְאַבֵּי נְחָלַי. תִּרְוֶה בְנוֹזְלֵי טְלָלַי. וְהָיִּיתָ קָרוֹב אֵלַי. וְכַשִּׁפְחָה אֵחָרֵף אֵלֶיךָ. וְאֶשָּׂא מַשְׂאוֹת מֵאֵת פָּנַי אֵלֶיךָ. וּכְשָׁמְעִי אִמְרָתָהּ. שָׂמַחְתִּי בְשִׂמְחָתָהּ וּבְשׂוֹרָתָהּ. וְאָמַרְתִּי אֶל לִבִּי עַתָּה רוּץ נָא לִקְרָאתָהּ. לִרְאוֹת הֲיֵאָמְנוּ דְבָרֶיהָ. וְיִצְדְּקוּ אֲמָרֶיהָ. וְכַאֲשֶׁר נָדְרָה כֵּן עָשְׂתָה. וְכַאֲשֶׁר דִּמִּיתִי כֵּן הָיְתָה. כִּי הֱבִיאַתְנִי לְגִנָּה. דְּשֵׁנָה וְרַעֲנַנָּה. הִסְתַּבְּכוּ שִׂיחֶיהָ. וְהִתְלַכְּדוּ צְמָחֶיהָ. וּמִינֵי גֶרֶשׁ יְרָחֶיהָ. וְכַפְתּוֹרֶיהָ וּפְרָחֶיהָ. וּלְחִי נִצָּנֶיהָ. כִּלְחִי עֲלוּמֶיהָ. מְשֻׁבָּצִים מַעֲשֵׂה רִקְמָה וְהָעֲרוּגָה מִתְעַטֶּפֶת בְּבֶגֶד עֲדָנֶיהָ. וּמִשְׁבְּצוֹת שָׁנֶיהָ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

פְּרָחֶיהָ יְאִירוּן עַד חֲשַׁבְתִּים

כְּמוֹ נֵרוֹת בְּרָאשֵׁי הָאֲמִירִים.

וְעַל שַׁחַק אֲשֶׁר יֵרֵד וְיִפְצַר

לְכַבּוֹתָם וְהֵם יוֹתֵר מְאִירִים.

וּבְתוֹךְ הַגִּנָּה יִבְלֵי מַיִם. וּמְקוֹם נְהָרִים יְאוֹרִים רַחֲבֵי יָדַיִם. וּמֵימֵיהֶם נִגָּרִים בַּמּוֹרָד. כַּעֲנַק בְּדֹלַח לַח מְפֻזָּר וּמְפֹרָד.

וַיַּעַן הַמְשׁוֹרֵר וַיֹּאמַר:

פְּלָגִים יוֹרְדִים עַל רִצֲפַת אֵשׁ

וְנִגָּרִים כְּמוֹ דִמְעִי בְּמוֹרָד.

וְנִטְפֵי נוֹזְלִים יִדְמוּ בְרִדְתָּם

בְּדֹלַח לַח מְפֻזָּר וּמְפֹרָד.

וְעַל הַמַּשְׁאַבִּים. גֻּלּוֹת וְאֶשְנַבִּים. וּסְבִיב הַתְּעָלוֹת. אֹהָלִים מֵעֲצֵי אֲהָלוֹת. וּבֵין גֻּלּוֹת עִלִּיּוֹת. טִירוֹת יְפֵה-פִיּוֹת. וְחַדְרֵי מַשְׂכִּיּוֹת. וּסְבִיב הָעֲרוּגוֹת הָעֲרוּכוֹת. שִׁקֲתוֹת מַיִם וּבְרֵכוֹת. סְבִיבָם נִצָּנִים וּפְרָחִים. וּמִגְדְּלוֹת מֶרְקָחִים. וְכָל מִינֵי צְמָחִים. וְיַעַר צוֹמֵחַ עֲלֵי שִׂיחִים. מֵהֶם תַּפּוּחִים. בְּשֶׁמֶן הַחֵן רְקוּחִים. כְּאִלּוּ מִשְּׁדֵי עַלְמָה קְטוִּפים. אוֹ מִלְּחִי צְבִי חֵן אֲסוּפִים.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וְתַפּוּחִים כְּמוֹ דָמִים אֲדֻמִּים

מְקֻטָּרִים בְּכָל רָאשֵׁי בְשָׂמִים.

בְּעֵת אִישׁ יֹאחֲזֵם וַיֹּאכְלֵם אָז

חֲבָלִים נָפְלוּ לוֹ בַנְּעִימִים.

וְאֶצְלָם רִמּוֹנִים. צַחִים וְאַדְמוֹנִים. בָּם לְצִירִים נִכְאָבִים. רְפוּאוֹת נֶעֱצָבִים. מִמֶּחְצַב הַנֹּעַם נֶחְצָבִים. וּמֵעֲצֵי עֵדֶן נִקְצָבִים. וְכָל רִמּוֹן. צַח וְאַדְמוֹן. וּבְתוֹכוֹ מִבְּנוֹת הַיֳּפִי הָמוֹן. לָהֶם מִפְּנִינֵיהֶם פַּעֲמוֹן. וּמִחוּץ עֲלֵיהֶם טַל חֶרְמוֹן. וּמִבַּיִת מֵימֵיהֶם כְּמֵי דִימוֹן. כְּאִלּוּ מִפָּנֵינוּ הָיוּ נִכְלָמִים. וְעַל כֵּן שָׁבוּ אֲדֻמִּים. מְגוֹלָלִים בְּדָמִים.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וּבֶן-רִמּוֹן יְפֵה מַרְאֶה וְאַדְמוֹן

אֲשֶׁר יָנוּד לְכָל-רוּחַ כְּאַגְמוֹן.

וְרִמּוֹנָיו כְּפַעְמוֹנִים סְבִיבָיו

בְּכָל-עֵת פַּעֲמֹן זָהָב וְרִמּוֹן.

וְנְגָּדם אֱגוֹזִים. בְּגִנְזֵי נֹעַם גְּנוּזִים. לוֹבְשִׁים בְּגָדִים יְרֻקִּים. וּבְטַעְמָם לְכָל פֶּה מְתוּקִים. וְכָל אֱגוֹז נָאֶה בְּתַבְנִיתוֹ. וְנֶחְמָד בְּהַדְרָתוֹ. אַךְ הוּא כִילַי לְכָל מְבַקֵּשׁ טוֹבָתוֹ. וְלֹא יִתֵּן לְךָ בִרְכָתוֹ. עַד תּוֹסִיף לְהַכּוֹתוֹ. וְתִתְקַע אֶת הַיָּתֵד בְּרַקָּתוֹ.

וָאֶשָּׂא מְשָׁלִי:

אֱגוֹזִים לוֹבְשִׁים בִּגְדֵי נְעִימוֹת

עֲלֵיהֶם לַיֳּפִי מַלְבּוּשׁ רְקָמוֹת.

אֲבָל יַקְשׁוּ מְאֹד עָרְפָּם לְדוֹרְשָׁם

עֲדֵי יַכֵּם וְאָז יִמְצָא נְקָמוֹת.

וְלִפְנֵיהֶם עֲצֵי תְמָרִים. בְּקוֹמַת עֳפָרִים. עַל קַו ישֶׁר מְיֻשָּׁרִים. כְּאִלּוּ גָזְלוּ מִן הָעֲלָמוֹת. יְפִי הַקּוֹמוֹת.

וָאֶשָּׂא מְשָׁלִי:

אָמַרְתִּי אֶעְלֶה עַל תָּמָר

וּבְצוּף מַעְדַנִּיו אֶתְיַמָּר.

עֵת אֶטְעַם אֹתוֹ אֶתְעַדֵּן

עַד נֹפֶת צוּף נֶגְדּוֹ יֵמַר.

וּבְפִגְעִי יַעְלַת חֵן אֹמַר:

קוֹמָתֵךְ דָּמְתָה לַתָּמָר.

וְנִכְחָם עֲנָבִים עֲרֵבִים. לְהַשְבִּיעַ רְעֵבִים. וּמִצּוּף וְנֹפֶת מָתוֹק טַעְמָם. וְנֶגְדָּם נִשְׁכְּחוּ כַל הַמַּמְתַּקִּים וְלֹא יִזָּכְרוּ עוֹד בִּשְׁמָם.

וָאֶשָּׂא מְשָׁלִי:

עֳרֵבִים הֵם לְכָל לָשׁוֹן עֲנָבִים

וְלִמְאֹד נָעֲמוּ לִטְעֹם וְלֶאְכֹל.

וּבָם שִׂמְחַת אֱלֹהִים וַאֲנָשִׁים

יְבֹרַךְ אֵל אֲשֶׁר הִפְלִיא וְיָכוֹל.

וּמֵרֹב אַהֲבַת הָאֵל בְּאֶשְׁכּוֹל

לְזֹאת תָּלָה מְכוֹן עוֹלָם בְּאֶשְׁכּוֹל.

וּלְצִדָּם תְּאֵנִים מְתוּקִים. לְכָל פֶּה חֲשׁוּקִים. וּבְכוּר הַנֹּעַם יְצוּקִים. פִּרְיָם חַיֵּי נְפָשִׁים. וְטַעְמָם צְרִי צִיר אֲנוּשִׁים.

וַיַּעַן וַיֹּאמַר:

תְּאֵנִים מִשְּׂפַת עֹפֶר גְּנוּבִים

וְאִם לֹא מִשְּׁדֵי עַלְמָה עֲשׁוּקִים.

כְּאִלּוּ הַתְּאֵנִים אוֹמְנִים הֵם

וְאוֹתָנוּ דְבַשׁ עִם צוּף מְנִיקִים.

וּסְבִיבָם שְׁקֵדִים. בִּקְנֵה הַיֳּפִי מְשֻׁקָּדִים. וּמֵרְאֵיהֶם זַכִּים. וְאִם גַּבֵּיהֶם קָשִׁים לִבּוֹתָם רַכִּים. כַּעֲלָמוֹת בְחֶדֶר בָּאוּ. וּמֵעֵין רוֹאֵיהֶם נֶחְבָּאוּ. וְאִם אִישׁ יִשְׁבֹּר דַּלְתוֹתָם יֵצְאוּ לַחוּץ וּמִשְּׁבָרִים יִתְחַטָּאוּ.

וָאֶשָּׂא מְשָׁלִי:

שְׁקֵדִים שׁוֹקְדִים דַּלְתֵי נְעִימוֹת

וּמִי יוּכַל לְסַפֵּר אוֹ לְדַמּוֹת.

וּמִתּוֹכָם פְּרִי סָגוּר וְאָגוּר

כְּעַלְמוֹת חֵן בְּחֶדֶר נַעֲלָמוֹת.

וּבֵינוֹתָם עֲרוּגוֹת. בְּחוּטֵי זָהָב אֲרוּגוֹת. וְהִּצִּצים נוֹצְצִים כִּשְׁבִיסִים. וְאִם יַד רוּחַ תְּנִיפֵם יִתְנַעֲרוּ מִן הָרְסִיסִים.

וָאֶשָּׂא מְשָׁלִי:

עֲרוּגָה אֲרוּגָה בְּשׁוֹשָׁן וְסוּגָה

וּמִלֵּב דְּוֵה לֵב תְּגָרֵשׁ דְּאָגָה.

וְדִמְעַת טְלָלִים בְּלֶחְיָהּ שְׁחָרִים

וְתִשְׂחַק וְתִבְכֶּה וְאֵין לָהּ הֲפוּגָה.

וּבַת הַגְּפָנִים בְּמוֹתָהּ חֲיוֹתִי

וְחַיֵּי שְׂשׂוֹנִי בְּעֵת הִיא הֲרוּגָה.

וּבְתוֹךְ הָעֲרוּגָה אַיֶּלֶת אֲהָבִים. בַּת נְדִיבִים. לֶחְיָהּ וּשְׂעָרָהּ שְׁחָרִים וַעֲרָבִים. בָּרָה כַלְּבָנָה. יָפָה נַעֲמָנָה. בְּקוֹל עָרֵב מְרַנֶּנֶת. וְעַל כִּנּוֹר מְנַגֶּנֶת. וְהוּא כְיוֹנֵק בְּחֵיקָהּ וַתְּהִי לוֹ לְאוֹמֶנֶת.

וָאֶשָּׂא מְשָׁלִי:

יְפַת מַרְאֶה מְנַגֶּנֶת בְּכִנּוֹר

וְשַׂעְרָה לֵיל אֲבָל לֶחְיָהּ שְׁחָרִים.

בְּאֶצְבָּעוֹת כְּשַׁרְבִיטִים לְבָנִים

כְּעֵין מָן אוֹ בְּדֹלָחִים וְדָרִים.

תְּעוֹרֵר מֵיתְרֵי כִנּוֹר לְשִׁירִי

וְתִמְשֹׁךְ לֵב בְּמָשְׁכָהּ הַיְתָרִים.

וְעֵת תַּחְפֹּץ גְּזוֹר לֵב חוֹשְׁקֶיהָ

תְּחַנֵּן קוֹל וְיִגָּזֵר גְּזָרִים.

וְאֵשׁ חִשְׁקָהּ יְפַלַּח הַלְּבָבוֹת

כְּלַפִּיד יַעֲבֹר בֵּין הַבְּתָרִים.

וּלְפָנֶיהָ גְבִיעִים מְאָדָּמִים. בִּדְמֵי בַת הַכְּרָמִים. אֲשֶׁר בָּהּ יָשִׂישׂוּ דוֹאֲגִים. וְיִשְׂמְחוּ נוּגִים. וְיִחְיוּ הַהֲרוּגִים.

וָאֶשָּׂא מְשָׁלִי:

יַיִן כְּאוֹר עַיִן אֲשֶׁר תָּמִיד

יִתְחַדְּשׁוּ גִילָיו וְהוּא נוֹשָׁן.

הוּא אֵשׁ אֲבָל יֶעְרַב וְלֹא יִבְעַר

כָּאֵשׁ וְלֹא יִשְׂרֹף וְלֹא יֶעְשָׁן.

וְנֶגְדָּהּ בַּחוּרֵי חֶמֶד מַחֲמַדֵּי עֵינַיִם. מַחְפִּירִים בְּיָפְיָהּ כּוֹכְבֵי שָׁמַיִם. וְהֵם מִשְׁתַּעַשְׁעִים בְּשִׁירֵי נְגִינוֹת. וְדִבְרֵי רְנָנוֹת. מְקוֹר הַשֵּׂכֶל בְּהֶגְיוֹנֵיהֶם. וּמַעְיַן הַשִּׁיר בְּרַעְיוֹנֵיהֶם. וּמוֹצָא הַבְּדֹלַח בֵּין שִׁנֵּיהֶם. וּמִזְרַח הַשֶּׁמֶשׁ עַל פְּנֵיהֶם. וְכַאֲשֶׁר קָרַבְתִּי אֲלֵיהֶם. חָרְדוּ לִקְרָאתִי. וְקִדְּמוּ בְשָׁלוֹם אוֹתִי. וְאָמְרוּ לִי: שָׁלוֹם בּוֹאֶךָ. מֵאַיִן מוֹצָאֶךָ. אָמַרְתִּי: סְפָרַד אַדְמָתִי. וְאֶרֶץ הַצְּבִי מְגַמָּתִי. אְמרוּ לִי: אַל יֶחֱשַׁך מְרוֹמֶךָ. וּבַל יִפָּקֵד מְקוֹמֶךָ. וַיָּשִׂימוּ לִי מָקוֹם בְּתוֹךְ הַקְּרוּאִים. וְהִנֵּה רֻבָּם חֶבֶל נְבִיאִים. וּבְתוֹכָם אֲבִי הַשִּׁירִים הַנּוֹרָאִים. חֶבֶר הַקֵּינִי. וּבְהַכִּירוֹ אוֹתִי רָץ וַיְחַבְּקֵנִי. וַיִּפֹּל עַל צַוָּארִי וַיִּשָּׁקֵנִי. וַיֹּאמַר לַחֲבֵרָיו: עַתָּה תִשְׁלַם לָכֶם הַשִּׂמְחָה וְהַצָּהֳלָה. כִּי לֹא בָאתֶם עַד עַתָּה אֶל הַמְּנוּחָה וְאֶל הַנַּחֲלָה. וָאֶעֱמֹד חֹדֶשׁ יָמִים לְהִתְעַלֵּס בָּהֶם. וְלִלְקֹט מַעֲדַנַיִּם מִפְּרִי פִיהֶם. עַד נָטָה הַזְּמַן שׁוֹט הַפְּרִידָה עָלֵינוּ. וַיִּתְנַכֵּר אֵלֵינוּ. וְכָחַל בְּפוּךְ הַנְּדוֹד אֶת עֵינֵינוּ. וְהִפְרִיד בֵּינֵינוּ.
