ספר תחכמוני -שַׁעַר חֲמִישִׁי

בְּשֶׁבַח י"ב מְשׁוֹרְרִים, אֲשֶׁר כָּל אֶחָד עַל חֹדֶשּ אֶחָד קוֹלוֹ יָרִים.

נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי בְּמַעֲמַד יְדִידִים. חֲמוּדִים. בְּיַד הַיֳפִי צְמוּדִים. וְהָיוּ לַאֲחָדִים. כֻּלָּם אֲצִילִים וּנְגִידִים. יָפִיצוּ מִלְּשׁוֹנָם אִמְרֵי נְגִידִים. וּמְלִיצוֹתָם כְּרִשְׁפֵּי יְקוֹדִים. בּוֹעֲרוֹת כְּמַרְאֵה הַלַּפִּידִים. וְהָיָה הַזְּמַן חֹדֶשׁ הָאָבִיב. הוּא חֹדֶשׁ נִיסָן הָאָהוּב וְהֶחָבִיב. וּמִדִּמְעוֹת הַטַּל בִּלְחִי הַשּׁוֹשָׁן רָבִיב. וְהַגַּן מֵרִקְמַת הַשּׁוֹשַׁנִּים שָׂפָה יִהְיֶה לְפִיו סָבִיב. וְהָאֲדָמָה לָבְשָׁה מִן הַצִּצִּים בִּגְדֵי רִקְמָתָהּ. וַתַּעַד נִזְמָהּ וְחֶלְיָתָהּ. וְהַשּׁוֹשָׁן כַּנֵּר יַזְהִיר לְעֵינֶיהָ.

וְהֵאִיר אֶל עֵבֶר פָּנֶיהָ. וְעַל גַנֶּיהָ. פְּרָחֶיהָ וְנִצָּנֶיהָ. וְעַל שָׁרוֹנֶיהָ שְׂשׂוֹנֶיהָ. וְיָרָק יָרַק בְּפָנֶיהָ. וּבָשְׂמֵי הַגַּן לְאַפֵּינוּ יִמָּשֵׁכוּ. אֶל אֲשֶׁר יִהְיֶה שָׁמָּה הָרוּחַ לָלֶכֶת יֵלֵכוּ. וְהַחַמָּה הוֹסִיפָה אוֹרָהּ וַיִּגָּלֶה. וְחָנְתָה בְּמַזַּל טָלֶה. וְהַצְּמָחִים יָצְאוּ מֵעִמְקֵי אָרֶץ. וּפָרְצוּ פָרֶץ. וְשָׂמוּ לָהֶם רֹאשׁ אֶחָד וְעָלוּ מִן הָאָרֶץ. וְכִסּוּ פְּנֵי תֵבֵל בְּמִפְרְשֵׂי נֹעַם. וְהָעָב צֻוְּתָה לְבַשֵּׂר הָאֲדָמָה שְׁלוּחֵי הָרַעַם. וַתִּשְׁלַח וַתִּקְרָא לְבָרָק בֶּן אֲבִינֹעַם. וַתֹּאמֶר לוֹ: שְׁלֹף עַל הָאָרֶץ חַרְבְּךָ וְתֶלְיְךָ. קוּם בָּרָק וּשְׁבֵה שֶׁבְיְךָ. וְלֵךְ לְבַשֵּׂר הָאֲדָמָה כִּי חֵילִי עָלֶיהָ עָרַכְתִּי. וַיֹּאמֶר אֵלֶיהָ בָּרָק: אִם תֵּלְכִי עִמִּי וְהָלַכְתִּי. וַתֹּאמֶר לוֹ: לֹא אֶרֶף יָדִי מִמְּךָ. וְהָלוֹךְ אֵלֵךְ עִמְּךָ. וַיִּשְׂאוּ קוֹלָם הָרְעָמִים וְהָאָרֶץ תַּעֲנֶה. וּגְדוּד הַגֶּשֶׁם לְמִלְחַמְתָּהּ חוֹנֶה. וּבָרָק רָדַף אַחֲרֵי הָרֶכֶב וְאַחֲרֵי הַמַּחֲנֶה. וְהַגַּנָּה לָבְשָׁה בִּגְדֵי יָפְיָהּ. וְהַשּׁוֹשַׁנָּה מִשִּׁמְחָתָהּ בְּשִׁכְבַת הַטַּל אֲשֶׁר הֶחְיָהּ. בָּכוֹ תִבְכֶּה בַּלַּיְלָה וְדִמְעָתָהּ עַל לֶחְיָהּ. וְהַגַּנָּה הִשְׁקִיפָה כְעַלְמָה מֵאֶשְׁנַבֵּי הֵיכָלֶיהָ. וַתָּסַר צְעִיפָהּ מֵעָלֶיהָ. וּמִשְׁבְּצוֹת זָהָב מְעִילֶיהָ.

וּמֶרְקְחוֹת הַמּוֹר בְּשׁוּלֶיהָ. וּלְשׁוֹנוֹת הַהֲדַס רְכִילֶיהָ. כִּי בְכָל פִיּוֹת יְרַגְּלוּ עָלֶיהָ. וְהַיַּיִן יַחֲלִיץ הָעֲצָמוֹת. אַחֲרֵי מוֹת. וִישַׂמַּח הַנְּשָׁמוֹת הַנְשַׁמּוֹת. וְהָיָה עִמָּנוּ רֹאשׁ הַחֲכָמִים וְהַגְּאוֹנִים. וּמֵסִיר שָׂפָה לְנֶאֱמָנִים. וְלוֹקֵחַ בְּשִׂכְלוֹ טַעַם זְקֵנִים. מִזֶּרַע הַקֵּינִים. הוּא חֲבֵרֵנוּ חֶבֶר הַקֵּינִי. הַמְחַבֵּר טוּרֵי הַסַּפִּירִים. בְּצַוְּארֵי הַסְּפָרִים. וּמוֹשֶׁה הַפְּנִינִים. מִמְּצוּלֹות הָרַעְיוֹנִים.

וַיַּעַן וַיֹּאמֶר לָהֶם: הִנְּכֶם כְּעֵץ חַיִּים וּמַה טּוֹב צִלְּכֶם. וְעַל־כִּסֵּא הַיְקָר אַתֶּם נִצָּבִים הַיּוֹם כֻּלְּכֶם. וְהַיָּמִים בְּאוֹר פְּנֵיכֶם יָאִירוּ. וּכְזֹהַר הָרָקִיעַ יַזְהִירוּ. וְהַזְּמַנִּים צָהֲלוּ הַיּוֹם פְּנֵיהֶם. וְנִפְקְחוּ עֵינֵיהֶם. וְעַתָּה הֵחָלְצוּ מֵאִתְּכֵם גִּבּוֹרֵי חָיִל. כְּכוֹכְבֵי לָיִל. בַּעֲלֵי שֵׂכֶל וּתְבוּנָה. וּמְסַפְּרִים בְּחָדְשֵׁי הַשָּׁנָה. וְכָל אֶחָד מֵהֶם יְשַׁבַּח חָדְשׁוֹ. וְיָשֵׂם נֵזֶר תְּהִלּוֹת עַל רֹאשׁוֹ. יַתְחִיל הִראשׁוֹן לְשַׁבֵּחַ חֹדֶשׁ נִיסָן הַנֶּאְדָּר. וְיַשְׁלִים הָאַחֲרוֹן בְּחֹדֶשׁ אֲדָר. וּבַדָּבָר הַזֶּה יִוָּדַע כֹּחַ שִׁירְכֶם. וְיִבָּחֲנוּ דִבְרֵיכֶם. הַאֱמֶת אִתְּכֶם. אָמְרוּ לוֹ כָל הָעָם: אַתָּה הַחִלּוֹתָ לְהַרְאוֹת אֶת עַבְדְּךָ חָכְמָתְךָ הָעֲמֻקָּה. וְאֶת יָדְךָ הַחֲזָקָה. וְעַתָּה הָחֵל אַתָּה בִּמְלִיצוֹת אֲמָרֶיךָ.

וַאֲנַחְנוּ נָבוֹא אַחֲרֶיךָ. וּמִלְּאֵנוּ אֶת דְּבָרֶיךָ.

וַיִּגַּשׁ חֶבֶר הַקֵּינִי וַיֹּאמַר:

חֹדֶשׁ נִיסָן רֹאשׁ הָרָאשִׁים. וּמֶלֶךְ עַל הֶחֳדָשִׁים. כִּי בוֹ הַזְּמַן מְקֻדָּשׁ וְהָעוֹלָם מְחֻדָּשׁ. וַיָּקָם מֶלֶךְ חָדָשׁ. וּבוֹ יְצַפְצְפוּ הַסִּיסִים. וְיִטְּפוּ הָרְסִיסִים. וְיִפְרְחוּ הַהֲדַסִּים. וְהַזְּמַן יַזְהִירוּ שְׁחָרָיו. וְיָאִירוּ מְאוֹרָיו. וִיחַדֵּשׁ כַּנֶּשֶׁר נְעוּרָיו. וְיָסִיר מֵעָלָיו הַשַּׂקִּים הַשְּׁחוֹרִים. וּפָשַׁט אֶת בְּגָדָיו וְלָבַשׁ בְּגָדִים אֲחֵרִים. וְרָחַץ בְּמַיִם טְהוֹרִים. וְאַחֲרֵי זִקְנָתוֹ וְאֹרֶךְ יָמָיו. יָשׁוּב לִימֵי עֲלוּמָיו. וְאָז הַגַּנִּים יִפְרָחוּ. וְהַלֵּילוֹת יִזְרָחוּ. וְהַיְגוֹנִים יִבְרָחוּ. וְהַשְּׂשׂוֹנִים יִצְמָחוּ. וְהַשּׁוֹשַׁנִּים יִשְׂמָחוּ. וְהָאָרֶץ תּוֹצִיא צִמְחָהּ. וּתְגַלֶּה פִּרְחָהּ. וְתוֹסֵף תֵּת כֹּחָהּ. וַאֲחֵרי הֱיוֹת הָאֲדָמָה לַסְּתָו בְּעוּלָה. וּבְעֶרֶשׂ דּוֹדִים כְּלוּלָה. תְּגָרֵשׁ בַּעְלָהּ אֲשֶׁר נוֹשָׂן יָשָׁן. וְתִכְסֹף לְחֹדֶשׁ נִיסָן. אֲשֶׁר הוּא מַחְמַד כָּל אִישׁוֹן. וְאָמְרָה אֵלְכָה וְאָשִׁיבָה אֶל אִישִׁי הָרִאשׁוֹן. וְאַחֲרֵי נִתְגָּרְשָׁה מִזְּמַן הַקֹּר אֲשֶׁר הָיָה אִתָּהּ. יוּכַל בַּעְלָהּ הָרִאשׁוֹן אֲשֶׁר שִׁלְּחָהּ לָשׁוּב לְקַחְתָּהּ. וְהוּא חֹדֶשׁ הַפְּלָאִים. וְהַנִּסִּים הַנּוֹרָאִים. וְרֹאשׁ לְחָדְשֵׁי הַמּוֹעֲדִים. כִּי בּוֹ נִגְאֲלוּ יְדִידִים. מִבֵּית עֲבָדִים. וְיָצְאוּ לְכוּדִים. מִידֵי לוּדִים. וְאוֹיְבֵיהֶם רָדְפוּ אַחֲרֵיהֶם גְּדוּדִים. וְנִבְקַע לָהֶם הַיָּם בְּרוּחַ קָדִים. וְשָׁקְעוּ הַמִּצְרִים בִּמְצוּלִים. וְשִׁירָה חֲדָשָׁה שִׁבְּחוּ גְאוּלִים. וּפַרְעֹה טָבַע בְּיַם־סוּף וְנָפַל מִשָּׁמַיִם. לְמַעֲמַקֵּי מָיִם. וּמִצּוֹאָתוֹ רֻחַץ בְּלִי יָדַיִם. וּמֹרַק וְשֻׁטַּף בַּמָּיִם. וְהַקֶּרֶב וְהַכְּרָעַיִם יִרְחַץ בַּמַּיִם. וְזֶרַע יְשֻׁרוּן עַל בָּמֳתֵי אֶרֶץ. רָכַב וַיַּהֲלֹךְ. כִּי מִבֵּית הָאֲסוּרִים יָצָא לִמְלֹךְ. וְאָז הַשֶּמֶשׁ תִּקְרַב. וְהַיַּיִן יֶעֱרַב. וּבַת הָאֲשִׁישָׁה. הִיא בְּיָפְיָהּ נַעֲרָה וּבְאֹרֶךְ יָמֶיהָ יְשִׁישָׁה. וָאֶקַּח אוֹתָהּ לִי לְאִשָּׁה. וְכָל־הַלַּיְלָה לְצִדִּי שׁוֹכֶנֶת. וַתְּהִי לִי סוֹכֶנֶת. וּבַעֲלוֹת הַכּוֹס בְּכַפִּי. תִּשְׁלַח בְּשָׂמֶיהָ לְאַפִּי. וְנוֹשֶׁקֶת עַל פִּי. וּבְשׁוּבָהּ רֵיקָם מִלְפָנַי. אוֹמֶרֶת בְּאָזְנַי. אֲנִי מְלֵאָה הָלַכְתִּי וְרֵיקָם הֱשִׁיבַנִי אֲדֹנַי.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בֵּינוֹת עֲרוּגוֹת שֵׁב יְמֵי נִיסָן

עֵת יַעֲלֶה נִצָּן וְרָם נִסָּן

וַעְלֹז לְקוֹל זָמִיר עֲלֵי אָמִיר

גַּם שִׂישׂ לְהוֹד הַצִּיץ וְהַסַּנְסָן.

שִׂמְחַת נְפָשׁוֹת בַּאֲשִׁישׁוֹת, כִּי

מֵהַזְּמָן מַתָּן מְנָת כּוֹסָן.

וַיַּעַן אֶחָד מִן הַחֲבֵרִים וְהוּא שֵׁנִי לְחֶבֶר הַקֵּינִי וַיֹּאמַר. חֹדֶשׁ אִיָּר מַחְמַד הֶחֳדָשִׁים. כִּי בוֹ רִפְאוּת אֲנָשִׁים. וְּגבוּרַת חַלָּשִׁים. הַמְשַׂמַּחַת אֱלֹהִים וַאֲנָשִׁים. וְאִםִ חֹדֶשׁ נִיסָן עַל הֶחֳדָשִׁים מֶלֶךְ. הֲלֹא חֹדֶשׁ אִיָּר מִשְׁנֶה לַמֶּלֶךְ. וְעַל כֵּן לְצִדּוֹ יֵשֵׁב וְיַחֲנֶה. וַיַּרְכֵּב אוֹתוֹ בְּמִרְכֶּבֶת הַמִּשְׁנֶה. וְאָז הַגַּן יִפְרַח. וְהַיּוֹם יִזְרַח. וְהַחֹם יִסְרַח. וְהַקֹר יִבְרַח. וְלִפְנֵי גְדוּדֵי הַחֹם מַר יִצְרַח. וְחֵיל הַחֹם יִרְדְּפֵהוּ. עַדִ יִגֳפֵהוּ. וְאָז יְדֵי הַגֶּשֶׁם יִפְתְּחוּ גִנְזֵי אֲדָמוֹת. וְאוֹצְרֵי תְהוֹמוֹת. לְהוֹצִיא מִתַּחַת הָאָרֶץ בִּגְדֵי רְקָמוֹת. וּלְכַסּוֹת פְּנֵי אֲדָמָה בְּמִשְׁבְּצוֹת נְעִימוֹת. שְׁקַעֲרוּרוֹת יְרַקְרְקּוֹת אוֹ אֲדַמְדַּמּוֹת. לְהַלְבִּישׁ הָעֲרוּגוֹת. בְּכָתְנוֹת מֶשִׁי אֲרוּגוֹת. וְאַחֲרֵי הֶיוֹת הָאֲדָמָה לוֹבֶשֶׁת בֶּגֶד אַלְמְנוּתָה. וּמִתְאַבֶּלֶת בְּרֹב עֱנוּתָהּ. עַל מוֹת זְמַן הַסְּתָו בְּעָלֶיהָ. יִקָּחֶנָּה זְמַן הַחֹם לְאִשָּׁה וְיִתְחַבֵּר אֵלֶיהָ. יְבָמָהּ יָבוֹא עָלֶיהָ. וְאָז תִּפְשֹׁט שַׂקִּים בְרֹב גִּילֶיהָ. וְתָסֵר בִּגְדֵי אַלְמְנוּתָהּ מֵעָלֶיהָ. וְתָשֵׂם בְּרֹאשָׁהּ כְּלִילָהּ וּבְאְזְנֶיהָ עֲגִילֵיהָ. וּבִזְרוֹעֶיהָ פְּתִילֶיהָ. וְהַיַּיִן יִקְרָא לְשׁוֹתָיו: הָכִינוּ בַגַּן סֻכִּי. וּמַהֲרוּ לִשְׁתּוֹתִי כִּי צוּף תִּמְצְאוּ בְחִכִּי. וְאַל תְּאַחֲרוּ אוֹתִי וַה' הִצְלִיחַ דַּרְכִּי.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

רְאֵה הַגַּן בְּטוּרֵי צִיץ מְצֻיָּר

וְרוּץ לִשְׁתּוֹת לְאוֹר יָפְיוֹ בְּאִיָּר.

וְשׁוּר כִּי הַזְּמַן כָּתַב רְקָמוֹת

עֲלֵי הַצִּיץ בְּלִי קֶלֶף וְנַיָּר.

וְאָז יִגְבַּר גְּדוּד מָשׂוֹשׂ. וְנִתְפַּשׂ

גְּדוּד יָגוֹן וְלֹא יִהְיֶה מְתֻיָּר.

וַיַּעַן הַשְּׁלִישִׁי וַיֹּאמַר:

חֹדֶשׁ סִיוָן רֹאשׁ הַשָּׁלִישִׁים. וּמַחְמַד הֱחֳדָשִׁים בּוֹ יָאִירוּ פְּנֵי עוֹלָם. וְיִגָּלֶה כָּל נֶעְלָם. וְהַשָּׁמַיִם יִתְּנוּ טַלָּם. וְנִצָּנֵי הַגַּן יִשְׂאוּ חֵילָם. וְהַיּוֹם יָפוּחַ. וְיִשָּׂא מֶרְקְחוֹתָיו לְאַפֵּנוּ עַל כַּנְפֵי רוּחַ. וְהַתַּפּוּחַ בְּשֶׁמֶן הַמּוֹר מָשׁוּחַ. וְהַשּׁוֹשָׁן כְּאִלּוּ בְּמַחְתַת אַהֲרֹן רָקוּחַ. אוֹ מִגַּן אֱלֹהִים לָקוּחַ. וְהַפֵּרוֹת לְכָל פֶּה יֶעֱרָבוּ. וּמִזֶּרֶם הָרִים יִרְטָבוּ. וְהַזְּמַן יְקַדְּמֵנוּ בְּמַעֲדַנָּיו וְהוֹדוֹ. וְכָל טוּב אֲדוֹנָיו בְּיָדוֹ. וּבוֹ מִבְחַר מוֹעֲדֵי ה‘. וְהַיּוֹם אֲשֶׁר בָּחַר ה’. וַיֵּרֶד עַל הַר סִינַי. וְהִנְחִיל דָּתוֹ. לְלוֹמְדֵי תְעוּדָתוֹ. וּכְשָׁמְעָם הוֹד קוֹלוֹ יָצְאָה נִשְׁמָתָם. וְהוֹרִיד טַל אוֹרוֹת וְהֶחֱלִיץ עַצְמוֹתָם. לְהַצִּיל מִמָּוֶת נַפְשָׁם וּלְחַיּוֹתָם. עַל כֵּן בָּזֶה הַחֹדֶשׁ יִשְׂמַח כָּל לֵב וְיָגִיל. בְּעֵת הָעֲרוּגָה כְּכַלָּה לוֹבֶשֶׁת פְתִיגִיל. וְעָלֶיהָ טַבַּעַת וְעָגִיל. וְכָל עֵץ פְּרִי פּוֹרֵחַ. וְהַגְּפָנִים סְמָדַר נָתְנוּ רֵיחַ. וְהַזְּמַן סוֹרֵחַ. וְהַלֵּב שָׂמֵחַ. וְהַיַּיִן מְקֻטָּר בְּרָאשֵׁי בְּשָׂמִים מַעֲשֵׂה רוֹקֵחַ. וְהַכּוֹס תִּקְרָא לֵאמֹר: אִם תִּרְצֶה לְכַבּוֹת אֵשׁ צְלָעֶיךָ. וּלְרַפֵּא נְגָעֶיךָ. שִׂימֵנִי כַּחוֹתָם עַל לִבֶּךָ. וְכַחוֹתָם עַל זְרוֹעֶךָ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

מֵעַל פְּנֵי כוֹסוֹת מְכֻסּוֹת דַּר

הָסֵר לְאַט מַסְוֶה בְּבוֹא סִיוָן.
