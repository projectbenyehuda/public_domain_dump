ז אודרו בעוז

יום שני שלעצרת


א

אֻדָּרוּ בְעוֹז [… …. … … טים]

[ב…] בְכוֹרִים בשמ[….] טים

גְּנַיי ח[….. ….. …ו מלהטים

דְּבָרִים מְתוּקִים […] ו[…טים]


הש[….ה ל[…..] דִּבְּרוֹת עשר

ו[..]ך ל[…ד וְלָהֶם מָסַר

זֶבֶד מֶזֶג בַּל יֶחְסַר

חַיִּים לְמוֹצְאֵיהֶם תּוֹכְחוֹת מוּסָר


טַעְמֵי דָת הִזְהַרְתָּ […. ל..] ור

יְקָרָה מִפְּנִינִים לִקְנוֹת וְ[לֹא] לִמְכּוֹר

כְּנִגְלֵיתָ בְנוֹף בְּכוֹרֵימוֹ לַעְכּוֹר

לְבֵן בְּכוֹר לִמַּדְתָּ לְהַקְדִּישׁ כָּל הַבְּכוֹר


א1

מוּסָר / אִלַּפְתָּ לַבְּחוּרִים

לְיַדְּעָם מִצְוַת בִּכּוּרִים

עֲנַקְתָּם זְבַדְתָּם עֲשֶׂרֶת הַדְּבָרִים

רוֹמַמְתָּם גְּנַנְתָּם בְּיוֹם הַבִּכּוּרִים


ב

מִקּוֹל הַדִּבֵּר נִבְעֲתוּ עֲמוּסִים

מִהֲרוּ לְהַקְדִּים לְמִשְׁמָע מַעֲשִׂים

נָמוּ לְצִיר דַּבֵּר וְאָנוּ עוֹשִׂים

נָאוֹר אַל יְדַבֵּר פֶּן נָמוּת קְנוּסִים


שְׂמֵחִים שָׂשִׂים נָסְעוּ מֵרְפִידִים

שִׂמְחָה וְצָהֳלָה אֲפוּדִים

עֲטֶרֶת תִּפְאֶרֶת חֲמוּדִים

עֲנָוָה וִיקָר זְבוּדִים


פַּצְתָּה לְנֶאֱמָן בֶּאֱמוּנָה

פִּקּוּדֶיךָ לְלַמֵּד לְזֶרַע מִי מָנָה

צַו כָּל הַבְּכוֹר לֶאֱכֹל בְּנֶאֱמָנָה

צֹאן וּבָקָר לְהַגִּישׁ בַּמָּקוֹם שָׁנָה בְּשָׁנָה


ב1

עֵינַיִם / אִלַּפְתָּ דָּת שְׁבוּעוֹת

לְיַדְּעָם דִּבְרֵי שַׁעֲשׁוּעוֹת

עֲנַקְתָּם זְבַדְתָּם חֹסֶן יְשׁוּעוֹת

רוֹמַמְתָּם וְהֶחֱיִיתָם בְּטַל בְּחַג הַשָּׁבוּעוֹת


ג

קוֹל לֹא תִשָּׂא

קָרָא צִיר בְּמַשָּׂא

רְדוּמֵי אוֹם עֲמוּסָה

רֹאשָׁם בּוֹ נָשָׂא


שִׁתְלֵי אָבוֹת תְּמִימִים

כְּעָמְדוּ לְקַבֵּל רְשׁוּמִים

לֹא הָיוּ בָהֶם בַּעֲלֵי מוּמִים

כִּי כֻלָּם נְקִיִּים וּמְסֻיָּמִים


תְּמִימִים בְּאֵין מוּם

בְּלֹא אִלֵּם וְחֵרֵשׁ וְחִגֵּר וְעָמוּם

עַל כֵּן נִצְטַוּוּ יָפִים בְּאֵין בָּם מוּם

קָרְבָּן בְּלִי לְהַקְרִיב אִם יְהִי בוֹ מוּם


ד

אָז מֵרֹאשׁ צוּרִים

לְעַמְּךָ כִּנִּיתָ בְכוֹרִים

וּבַעֲבוּרָם נָגַפְתָּ בְּכוֹרֵי מִצְרַיִם

וְהִזְהַרְתָּם עַל מִצְוַת בְּכוֹרִים

אֲשֶׁר לְךָ מְבֻכָּרִים

בְּכוֹרֵי אָדָם וּפִטְרֵי חֲמוֹרִים

וּבְכוֹרֵי כְבָשִׂים וְעִזִּים נפרים

לְהַקְדִּישׁ לְךָ אַדִּיר בָּאַדִּירִים

מֵהֶם לִפְדּוֹת בְּכֶסֶף כִּפּוּרִים

ומֵהֶם אֵין לִפְדּוֹת כִּי אִם לְהַחֲרִים

וּכְחָפַצְתָּ יוֹדֵעַ נִסְתָּרִים

לְגַלּוֹת לָמוֹ עֲשֶׂרֶת הַדְּבָרִים

נִגְלֵיתָ עַל הַר תָּלוּשׁ מִכָּל הָרִים

וְדִבַּרְתָּ עִמָּם מִשְּׁמֵי שְׁפָרִים

וְהִנְחַלְתָּם טַעְמֵי אֲמָרִים

מְתוּקִים וְנֶחֱמָדִים וּמְיֻשָּׁרִים

עַל יַד אֲבִי צִירִים

בְּחַג שָׁבוּעוֹת בְּיוֹם הַבִּכּוּרִים
