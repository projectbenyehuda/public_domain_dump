[אזי בשעיף]

וממ ّ א חוא פיה כל כואכב אלת͘אבתה מת͘ל אל ّ ד͘י לם יטורהא כ͘יאל ולא ילחקהא ארתג͘אל קצידה או ّ להא [וממה שמכיל את כל כוכבי-השבת (ר"ל: המזהירים) ואשר לכמותה לא יקרב הדמיון ולא ישיגנה כוח האימפרוביזאציה זו שירה שראשיתה]

אֲזַי בִּשְׂעִיף חֲלוֹם רוּחַ נְשָׂאוֹ / וְעָבַר כַּעֲבֹר בָּרָק בְּפִלְאוֹ

וְלֹא שָׁר אַחֲרָיו שֶׁמֶץ לְבָבִי / וְלוּ מִדֵּי רְאוֹת עָמַד רְפָאוֹ

וְקָץ לָשׁוּר דְּמוּתוֹ מִשְּׂעִפָּיו / וְלֹא שָׁעַר לְבָבִי כִּי בְזָאוֹ

וְיַסְתִּיר מֵאֲחֵי נֹעַר אֲמָרָיו / לְבִלְתִּי יֹֽאמְרוּ חֵשֶׁק מְלָאוֹ

הֲכִי נִלְכַּד בְּחַבְלֵי הַתְּלָאוֹת / וְשֶׁבֶר אַחֲרֵי שֶׁבֶר קְרָאוֹ

וּמִשָּׁבְרָךְ תְּמוֹל הֵנִיא זְמַנּוֹ / הֲיִתְמַלֵּא וְהוּא עַז לַהֲנִיאוֹ

כְּאִלּוּ אָזְנְךָ הָעָם אֲשֶׁר צַר / לְהָכִיל מַחְשְׁבוֹתָיו יוֹם צְבָאוֹ

וְאָחַז לִבְּךָ כַּנְפוֹת אֲדָמָה / וְהֵשַׁח בָּמֳתֵי-שַׁחַק בְּשִׂיאוֹ

וּבָקַע שִׂכְלְךָ מִדְבַּר דְּבָרִים / וְתַחַת רַגְלְךָ תַּדְבֵּר לְבִיאוֹ

וְנָגַהּ מִמְּךָ אוֹר עַל-אֲנָשִׁים / כְּשֶׁמֶשׁ עִמְּךָ הָיָה מְבוֹאוֹ

וְלֹא נִזְכָּר לְשִׁירְךָ שִׁיר מְנַחֵם / וְשִׁיר דּוּנָשׁ וְשִׁיר אַבּוּן קְרִיאוֹ

כְּצַח יַסְתִּיר בְּעוֹדוֹ עַל-אֲדָמָה / בְּנֵי-עַיִשׁ וְיֵרָאוּן בְּבוֹאוֹ

וְאֵיכָה תַחֲרִישׁ כַּיּוֹם לְהָפִיץ / יְאוֹר בּוֹ כָל-יְאוֹר שָׁבַר צְמָאוֹ

וְנֶפֶשׁ בַּהֲמוֹת יַמִּים תְּדַכֵּה / ועטרם כַּחֲנִית גָּלְיַת בְּדַכְּאוֹ

וְלָשׁוֹן כַּחֲצַץ בָּרָק יְלַהֵט / וְיָפִיק כַּאֲפִיק נַחַל בְּשׂוֹאוֹ

וְנָטַשְׁתָּ גְּדוּד הַשִּׁיר מְפֻזָּר / בְּלִי-עָרוּךְ וְאַתָּה שַׂר צְבָאוֹ

לְכֹל נִתְעֶה לְבָבוֹ בַּכְּסִילוּת / כְּהִתָּעוֹת אֱנוֹשׁ שִׁכּוֹר בְּקִיאוֹ

וְכָל-שֹׁגֶה וְנָקֵל מִמְּשׁוּגָה / אֲשֶׁר הַשִּׁיר בְּמַצְרֵף לֹא הֱבִיאוֹ

וּמִקֹּר נוֹזְלָיו כִּי-נִצְּבוּ נֵד / עֲדֵי כִי-קָפְאוּ יַמִּים לְקָפְאוֹ

זְכֹר משֶׁה אֲשֶׁר הָיָה כְבַד-פֶּה / וְעַל-כֵּן אַהֲרֹן הָיָה נְבִיאוֹ

עֲנִיתִיו אֵיךְ אֲשׁוֹרֵר שִׁיר וְהַשִּׁיר / לְעֻמַּת כָּל-כְּסִיל יִרֶז בְּרִיאוֹ

וְנִפְשַׁט מִבְּגָדָיו הַחֲמֻדוֹת / וְנוֹתַר כֶּאֱנוֹשׁ עָנִי בְּכִסְאוֹ

וְנִמְאַס מִכְּלִי אֵין בּוֹ תְּעָלָה / וּמִמֶּלֶךְ אֲשֶׁר הוּרַד לְכִסְאוֹ

וְנָקֵל מִכְּתָב מִדְרָשׁ בְּיַד מִין / וּמִפֶּסֶל אֲשֶׁר נָבִיא מְצָאוֹ

וְלֹא נִשְׁאַר אֱנוֹשׁ מִבַּעֲלֵי כֵן / וְנִמְלָא הַזְּמָן מִבַּעֲלֵי אוֹ

וְאֵינָם אֹהֲבֵי שֵׂכֶל וּבִינָה / אֲבָל הֵם אֹהֲבֵי שֵׁמַע וְרָאוֹ

כְּפֶתִי כַּחֲכַם לֵבָב כְּזוֹלֵל / כְּיָקָר כַּאֲרַז דֶּלֶת כְּתָאוֹ

הֲכָשׁוֹל בַּאֲחֵי שִׁפְלוּת וְסִכְלוּת / וּפָגוֹשׁ בַּהֲמוֹת יַעַר וּפִרְאוֹ

כְּזִיזֵי יַעֲרוֹת יָנוֹק שְׁדֵי הָר / כְּעַתּוֹדֵי רְעִי לָחוֹךְ לְדִשְׁאוֹ

עֲיָרִים לוּ אֱנוֹשׁ יִדְקֹר בְּשָׂרָם / בְּפַטִּישׁ גַּב אֲחֹרָם לֹא יְרֵאוּ

וְיִשְׂתָּרֵר אֲשֶׁר יַשִּׂיג מְעַט בִּין / כְּהִשְׂתָּרֵר נְבַל-נֶפֶשׁ בְּמָצְאוֹ

וְהֵם לוּ אֻסְּפוּ עַל-כַּף נְמָלָה / נְקַלִּים לִהְיוֹתָם דֵּי מְלֹאוֹ

הֲכִי מוֹפֵת בְּאִוַּלְתָּם לְחֶסֶד / אֲדוֹן עוֹלָם אֲשֶׁר לָהֶם בְּרָאוֹ

וְאֹמֵר נִשְּׂאוּ כִּי חִבְּרוּ שִׁיר / עֲנִיתִיו לֹא כְּשִׁיר זֶמֶר וְכִתְאוֹ

וְהַחֹגֵר לְבַד יִבֶז גְּבוּרָה / וּמִלְחֶמֶת פְּרָז אֶפֶס בְּגֵאוֹ

וְהָעַז בַּקְּרָב יָבֹא וְיִגְבַּר / וְיַעְשֶׁה מַעֲשֵׂה פִינְחָס בְּקַנְאוֹ

וְהֶחָפֵץ בְּעוֹדוֹ לַעֲלוֹת עָב / בְּתוֹךְ בֶּטֶן שְׁאוֹל תַּחְתִּית כְּלָאוֹ

וְשִׁיר כָּל-בֹּחֲנֵי מִלִּין יְמוֹלֵל / וְיִיבַשׁ כַּחֲצִיר שָׂדֶה וְגָמְאוֹ

וְכָתַב הַזְּמָן שִׁירִי בְּפָנָיו / וְאִם-הוּא לֹא יְדָעוֹ לֹא שְׂנֵאוֹ.
