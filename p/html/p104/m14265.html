סִפּוּר הַחוֹשֵׁק וְהַחֲשּׁוּקָה.

א. חֲתֻנַּת הַמֶּלֶךְ שׂוּלַיְמָן.

וַיִּפְתַּח הַוָּזִיר דִּנְדָּן פִּיו וַיְסַפֵּר לֵאמֹר:
דַּע לְךָ, אֲדוֹנִי הַמֶּלֶךְ הַמְאֻשָּׁר, כִּי סֻּפַּר לִי מִסִּפּוּר “הַחוֹשֵׁק וְהַחֲשׁוּקָה” וּמֵהַמְדֻבָּר בֵּינֵיהֶם וּמֵהַפְּלָאוֹת וְהַנוֹרָאוֹת אֲשֶׁר קָרוּם, דְּבָרִים יָסִירוּ כָל-דְּאָגָה מִלֵּב שׁוֹמְעֵיהֶם, וְגַם צַעַר אֵבֶל יַעֲקֹב יפָיִגוּ מִיּוֹדְעֵיהֶם, לֵאמֹר:

הָיֹה הָיְתָה בַיָּמִים הַקְּדוּמִים מִאַחֲרֵי הָרֵי אִסְפָּהָן עִיר גְּדוֹלָה לֵאלֹהִים וּשְׁמָהּ “הָעִיר הַיְרֻקָּה”. וּבָעִיר הַהִיא מֶלֶךְ וּשְׁמוֹ שׂוּלַיְמָן, וְהוּא נָדִיב קָם עַל נְדיִבוֹת, וּבַעַל צֶדֶק וְיֹשֶׁר וּדְרָכִים טוֹבוֹת, וַיִּפְנוּ אֵלָיו הָאֲנָשִׁים מִכָּל עֵבָר וּפִנָּה, וְשָׁמְעוּ הוֹלֵךְ בְּכָל מְדִינָה וּמְדִינָה. וַיֵּשֶׁב עַל כִּסְאוֹ יָמִים רַבִּים בְּכָבוֹד וּבִגְדֻלָּה, וּשְׁמוֹ מְרוֹמָם עַל כָּל-בְּרָכָה וּתְפִלָּה. אֶפֶס כִּי הָיָה הָאִישׁ עֲרִירִי מִבְּלִי אִשָּׁה וּבָנִים, וְלוֹ וְזִיר קָרוֹב אֵלָיו בִּתְכוּנוֹתָיו, וּבִנְדִיבוֹתָיו וּמַתְּנוֹתָיו.

וַיְהִי הַיּוֹם וַיִּשְׁלַח הַמֶּלֶךְ לִקְרֹא אֶת-וְזִירוֹ וַיֹּאֶמר אֵלָיו: וְזִירִי, צַר לִי מְאֹד וְכֹחִי רָפֶה, כִּי הִנְנִי כָל-יָמַי בְּלִי אִשָּׁה וּבְלִי בָנִים; וְאֵין זֹאת מִדֶּרֶךְ הַמְּלָכִים הַשַּׁלִיטִים עַל כָּל-שׁוֹעַ וָדָל; כִּי חֲפֵצִים הֵם לְהַשְׁאִיר אַחֲרֵיהֶם זֶרַע, אֲשֶׁר בּוֹ יְדַבְּרוּ אֶת-אוֹיְבֵיהֶם בַּשָּׁעַר. וּכְבָר אָמַר הַנָּבִיא שְׁלוֹם הָאֵל וּצְלוּתוֹ עָלָיו: קְחוּ נָשִׁים וְהוֹלִידוּ בָנִים וּפְרוּ וּרְבוּ, כִּי בָכֶם אֶתְפָּאֵר בַּגּוֹיִם בְּיוֹם תְּחִיַּת הַמֵּתִים. וְעַתָּה הַגֵּד נָא עֲצָתְךָ וְאֶשְׁמְעֶנָּה. וַיְהִי כִשְׁמֹעַ הַוָּזִיר אֶת-דְּבָרָיו וַתִּזַלְנָה עֵינָיו דְּמָעוֹת וַיֹּאמֶר אֵלָיו: חָלִילָה לִי, מֶלֶךְ הַזְּמַן, מִדַּבֵּר בַּדְּבָרִים שֶׁהֵם מֵעִנְיְנֵי אָבִינוּ הָרַחֲמָן, הֶחָפֵץ אַתָּה, כִּי אָבוֹא בְאֵשׁ קֶצֶף הָאֵל הַכַּבִּיר? וַיֹּאמֶר אֵלָיו הַמֶּלֶךְ: דַּע נָא, הַוָּזִּיר, כִּי בִרְכֹּשׁ לוֹ הַמֶּלֶךְ צְעִירָה, אֲשֶׁר לֹא יֵדַע יַחֲשָׂהּ, יֵשׁ אֲשֶׁר תֵּעָלֵם מִמֶּנוּ זֻלּוּת מוֹצָאָהּ וְלֹא יֵדַע לְהִתְרַחֵק מִמֶּנָּה, אוֹ כְבוֹד מִשְׁפַּחְתָּהּ וְלֹא יִתְעַלֵּס בַּאֲהָבֵיהָ. וְאִם הָרֹה תֶהֱרֶה לוֹ, יֵשׁ אֲשֶׁר יִהְיֶה הַיֶּלֶד מָשְׁחָת, עָרִיץ וְשׁוֹפֵךְ דָּם, וְהָיָה דִּמְיוֹנָהּ כָּאָרֶץ הַמְּלֵחָה אֲשֶׁר יִזְרְעוּ בָהּ זֶרַע וְהָיָה פִרְיוֹ מְקֻלְקָל, וְקִיּוּמוֹ מְדֻלְדָּל, וְיֵשׁ אֲשֶׁר יָבִיא עָלָיו הַיֶּלֶד הַהוּא קֶצֶף הָאֵל, וְכָל-אֲשֶׁר יְצַוֵּהוּ לֹא יַעֲשֵׂהוּ. וּמִכָּל-אֲשֶׁר יַזְהִירֵהוּ לֹא יִרְחַק. כִּי עַל כֵּן מֵאַנְתִּי לָקַחַת לִי צְעִירָה, אַךְ חָפַצְתִּי כִּי תְאָרֵשׂ לִי בַת מִבְּנוֹת הַמְּלָכִים אֲשֶׁר יִהְיֶה יַחֲשָׂהּ יַחַשׂ כָּבוֹד וּגְדֻלָּה, וְיָפְיָהּ נַעֲלֶה מֵעַל כָּל-סְגֻלָּה. וְהָיָה אִם תִּמְצָא לִי בַת יַחַשׂ וֶאֱמוּנָה מִבְּנוֹת מַלְכֵי הַמֻּשְׂלָמִים וְאֵרַשְׂתִּיהָ וּנְשָׂאתִיהָ, בִּפְנֵי קָהָל וָעֵדָה, וּמָצָאתִי חֵן בְּעֵינֵי אֱלֹהִים וְאָדָם. וַיֹּאמֶר אֵלָיו הַוָּזִיר: הִתְבַּשֵׂר, אֲדנִֹי הַמֶּלֶךְ, כִּי מִלֵּא הָאֱלֹהִים חֶפְצֶךָ. וַיֹּאמֶר אֵלָיו: בַּמֶּה? וַיֹּאמֶר: שָׁמַעְתִּי כִי לַמֶּלֶךְ זָהֵר-שָׁהּ אֲדוֹנֵי הָאָרֶץ-הַלְּבָנָה בַת יָפָה וְנֶחְמָדָה, לֹא תְּתֹאַר בְּאֹמֶר וּדְבָרִים, וְאֵין דּוֹמָה לָהּ בִּזְמַנֵּנוּ, יִשְׁרַת קוֹמָה, הוֹלֶכֶת רוֹמָה, כְּחֻלַּת הָעֵינַיִם, דַּקַּת הַמָּתְנַיִם, בְּקָרְבָהּ תְּפַתֶּה הַלְבָבוֹת, וּבְרָחֲקָהּ תִּשְׁבֶּה הַמַּחֲשָבוֹת, מַבָּטֶיהָ לַהֲטֵי חֲרָבוֹת, אוֹ חִצֵּי קְרָבוֹת, שֶׁמֶשׁ וְיָרֵחַ לֹא יִדְמוּ לְתָאָרָהּ, וְכָלֹ-כּוֹכְבֵי מָרוֹם יִכְרְעוּ לְזָהֳרָהּ. וְעַתָּה לוּ לַעֲצָתִי תִשְׁמָע, נִשְׁלְחָה אֶל אָבִיהָ צִיר נָבוֹן וּבַעַל נִסָּיוֹן, וְדִבֵּר עַל לִבּוֹ לְתִתָּהּ לְךָ לְאִשָּׁה, וְהָיִיתָ מְאֻשָּׁר בְּחַיֶּיךָ וְהָאֵל יִרְצֶה מַעֲשֶׂךָ, הֵן כֹּה אָמַר הַנָּבִיא: אֵין פְּרִישׁוֹת בָּאִשְׂלָם. וַיִּשְׂמַח הַמֶּלֶךְ מְאֹד וַיֹּאמֶר אֶל הַוָּזִיר: דַּע לְךָ, וְזִירִי הַטּוֹב, כִּי לֹא יִכּוֹן לַדָּבָר הַזֶּה אִישׁ בִּלְעָדֶיךָ, לְשִׂכְלְךָ וּלְדַעְתְּךָ לְהִתְהַלֵּך עִם בְּנֵי הָאָדָם. וְעַתָּה קוּמָה לְכָה אֶל בֵּיתְךָ וְסִדַּרְתָּ עִנְיָנֶיךָ, וֹמָחָר בַּבֹּקֶר תִּצְטַיֵּד לְדַרְכֶּךָ, וְאֵרַשְׂתָּ לִי אֶת-הַנַּעֲרָה אֲשֶׁר בְּתֵאוּרָהּ שָׁבִיתָ לִבִּי שֶׁבִי, וְלֹא תָשׁוּב עַד אִם הֲבֵאתָהּ אִתְךָ. וַיֹּאמֶר הַוָּזִיר: כִּדְבָרֶיךָ, אֲדוֹנִי הַמֶּלֶךְ.
וַיֵלֶךְ הַוָּזִיר אֶל בֵּיתוֹ, וַיְצַו וַיָּבִיאוּ אֵלָיו תְּשׁוּרוֹת הַמְּלָכִים, אַבְנֵי יְקָר וְאוֹצְרוֹת פְּלָאוֹת, כָּל-הַדָּבָר הַקַּל בְּמִשְׁקָלוֹ וְכָבֵד בְּעֶרְכּוֹ, וַיִטְעֲנוּ אֶת-הַכֹּל עַל הַגְּמָלִים וְהַפְּרָדִים, וַיְנוֹפְפוּ הַדְּגָלִים מֵעַל לְרֹאשׁוֹ, וַיֵּלֵכוּ.

וְהַמֶּלֶךְ שׂוּלַיְמָן יָשַׁב בְּהֵיכָלוֹ כְּיוֹשֵׁב עַל הַגֶּחָלִים הַבּוֹעֲרוֹת, וְלִבּוֹ לוֹהֵט בְּאַהֲבַת הַנַּעֲרָה יוֹמָם וָלַיְלָה.

וְהַוָּזִיר נָסַע לְמַסָּעָיו וַיַּעֲבֹר אֲרָצוֹת וּמִדְבָּרוֹת, עַד אֲשֶׁר לֹא נִשְׁאַר בֵּינוֹ וּבֵין הָאָרֶץ אֲשֶׁר הָלַךְ אֵלֶיהָ בִלְתִּי אִם מַהֲלַךְ יוֹם אֶחָד.

וַיִּחַן עַל שְׂפַת נָהָר, וַיִּקְרַא אֵלַיו אַחַד אֲנָשָׁיו, וַיְּצַוֵּהוּ לָלֶכֶת אֶל הַמֶּלֶךְ זָהֵר-שָׁהּ וּלְהַגִּיד לוֹ דְבַר בּוֹאוֹ. וַיְמַהֵר הָאִישׁ וַיָבֹא אֶל עִיר הַמֶּלֶךְ זָהֵר-שָׁהּ. וַיְהִי בְהַגִּיעוֹ אֵלֶיהָ וְהַמֶּלֶךְ זָהֵר-שָׁהּ יוֹשֵׁב בְּאַחַד גַּנֵּי-חֶמְדוֹ לְיַד שַׁעַר הָעִיר, וַיִּרְאֵהוּ וַיֵּדַע כִּי אִישׁ נָכְרִי הִנֵּהוּ, וַיְצַו וַיְבִיאוּהוּ לְפָנָיו. וַיְהִי בְּבוֹאוֹ וַיֻּגַּד לַמֶּלֶךְ, כִּי וְזִיר הַמֶּלֶךְ הַגָּדוֹל שׂוּלַיְמָן-שָׁהּ אֲדוֹנֵי הָאָרֶץ הַיְרֻקָּה וְהָרֵי אִסְפָּהָן הוֹלֵךְ וּבָא אֵלָיו, וַיִּשְׂמַח הַמֶּלֶךְ זָהֵר-שָׁהּ וַיֹּאמֶר אֵלָיו: אֵיפֹה נִפְרַדְתָּ מֵהַוָּזִיר? וַיֹּאמֶר: נִפְרַדְתִּי מִמֶּנוּ הַיּוֹם בַּבֹּקֶר עַל יַד שְׂפַת הַנָּהָר פְּלוֹנִי, וּמָחָר יַגִּיעַ אֵלֶיךָ, יָרִים הָאֵל הוֹדֶיךָ וְיָרִיק נָעֳמוֹ עָלֶיךָ. וַיְצַו זָהֵר-שָׁהּ אֶת-וְזִירוֹ לָקַחַת אֶת גְּדוֹלֵי שָׂרָיו וְחֹרָיו אִתּוֹ וְלָלֶכֶת לִקְרַאת הַוָּזִיר לִכְבוֹד מַלְכּוֹ שׂוּלַיְמָן-שָׁהּ. כִּי שִׁלְטוֹנוֹ נָפוֹץ בָּאָרֶץ.

וּוְזִיר הַמֶּלֶךְ שׂוּלַיְמָן-שָׁהּ נָח עַד חֲצוֹת הַלַּיְלָה, וְאַחַר קָם לָלֶכֶת אֶל הָעִיר. וַיְהִי כִי הֵאִיר הַבֹּקֶר וְהַשֶּׁמֶשׁ זָרְחָה עַל הֶהָרִים וְהָעֲמָקִים, וְהִנֶּה וְזִיר הַמֶּלֶךְ זָהֵר-שָׁהּ וְשָׂרָיו וְחֹרָיו בָּאִים לִקְרָאתוֹ לְקַדֵּם פָּנָיו, וַיִפְגְּשׁוּהוּ דֶרֶך מִילִים אֲחָדִים לִפְנֵי הָעִיר. וַיֵּדַע הַוָּזִיר כִּי אָכֵן הִצְלִיחַ אֱלֹהִים דַּרְכּוֹ, וַיְבָרֶךְ אֶת-הַבָּאִים לְשָׁלוֹם, וַיֵּלְכוּ מִשַּׁעַר הַמֶּלֶךְ עַד הַמָּבוֹא הַשְּׁבִיעִי, הוּא הַמָּקוֹם אֲשֶׁר אֵלָיו לֹא יָבוֹא כָל-רוֹכֵב כִּי קָרוֹב הוּא לְמוֹשַב הַמֶּלֶךְ, וַיֵּרֶד הַוָּזִיר מֵעַל סוּסוֹ וַיֵּלֶך בָּרֶגֶל עַד הַגִּיעוֹ אֶל אוּלָם. וּבְרֹאש הַאוּלָם כִּסֵא שַׁיִשׁ מְשֻׁבָּץ פְּנִינִים וְאַבְנֵי יָקָר. וְלַכִּסֵּא אַרְבַּע רַגְלֵי שֵׁן, וְעַל הַכִּסֵּא מַצַּע קְטִיפָה יְרֻקָּה רְקוּמָה זָהָב טָהוֹר, וּמִמַּעַל לוֹ חֻפָּה מְשֻׁבֶּצֶת אַבְנֵי יְקָר. וְהַמֶּלֶךְ זָהֵר-שָׁהּ יוֹשֵׁב עַל הַכִּסֵּא, וְרַבֵּי הַמְּלוּכָה עוֹמְדִים לְפָנָיו.

וַיְהִי בְּבוֹא הַוָּזִיר לְפָנָיו וַיְאַמֵּץ לִבּוֹ וַיִּפְתַּח פִּיו, וַיְּדַבֵּר נְגִידִים בִּשְׂפַת נֶאֱמָנִים. וְַיָקְרְבֵהוּ הַמֶּלֶךְ אֵלָיו וַיּוֹשִׁיבֵהוּ עַל יָדוֹ וַיָּאֶר פָּנָיו אֵלָיו וַיְדַבֶּר אִתֹּו טוֹבוֹת. וְאַחַר הִגִּישׁוּ אֶת-הַשֻּׁלְחָן הֶעָרוּך, וַיֹּאכְלוּ וַיִּשְׂבָּעוּ. וַיֵצְאוּ כָל-הַנִּצָּבִים עַל הַמֶּלֶךְ וַיִּשָּׁאֲרוּ רַק הַקְּרוֹבִים אֵלַיו. וַיְהִי בִרְאוֹת הַוָּזִיר זֹאת, וַיָּקָם עַל רַגְלָיו וַיְשַׁבַּח אֶת-הַמֶּלֶךְ וַיִכְרַע לְפָנָיו וַיֹאמֶר: אֲדוֹנִי הַמֶּלֶךְ הַגָּדוֹל, בּוֹא בָאתִי אֵלֶיךָ בְדָבָר אֲשֶׁר בּוֹ אָשְׁרְךָ וְטוֹבֶךָ. בָּאתִי אֵלֶיךָ שָׁלוּחַ מֵאֵת הַמֶּלֶךְ שׂוּלַיְמָן-שָׁהּ אִישׁ הַיֹשֶׁר וְהָאֱמוּנָה, וְהַחָכְמָה וְהַתְּבוּנָה, מֶלֶךְ הָאָרֶץ הָיְרֻקָּה וְהָרֵי אִסְפָּהָן, לְדַבֵּר נִכְבָּדוֹת בְּבִּתְּךָ רַבַּת הַיַּחַשׂ. וְהִנֵּה שָׁלַח אֵלֶיךָ תְּשׁוּרוֹת רַבּוֹת וּסְגֻלּוֹת מְלָכִים, וְהוּא אוֹמֵר לְהִתְחַתֵּן בְּמִשְׁפַּחְתְּךָ, וּמְחַכֶּה לְמוֹצָא שְׂפָתֶיךָ. וְאַחַר שָׁתַק מַחֲרִישׁ מְחַכֶּה לְמַעֲנֵהוּ. וַיָּקָם זָהֵר-שָׁהּ עַל רַגְלָיו וַיִקֹּד. וַיִּשְׁתּוֹמְמוּ הַנִּצָּבִים לְעַנְוַת הַמֶּלֶךְ בִּפְנֵי הַמַּלְאָךְ וַיִּתְפַּלָּאוּ. וַיָּקָם הַמֶּלֶךְ עַל רַגְלָיו וַיֹּאמֶר: שְׁמַע נָא, הַוָּזִיר הַגָּדוֹל וְהַשַּׂר הַמְּפֹאָר אֶת אֲשֶׁר אֹמַר: כֻּלָּנוּ פֹה נְתִינִים סָרִים לְמִשְׁמַעַת הַמֶּלֶךְ שׂוּלַיְמָן-שָׁהּ וּמִתְפָּאֲרִים בְּיַחֲשׂוֹ, וּבִתִּי אַחַת מִשִּׁפְחוֹתָיו.

וַיִּשְׁלַח הַמֶּלֶךְ וַיִּקְרָא לַשּׁוֹפְטִים וְלָעֵדִים, וַיָּעִידוּ הָעֵדִים כִּי מִלֵּא הַמֶּלֶךְ שׂוּלַיְמָן-שָׁהּ יְדֵי וְזִירוֹ לָקַחַת לוֹ לְאִשָּׁה אֶת-בַּת הַמֶּלֶךְ זָהֵר-שָׁהּ, וְהַמֶּלֶךְ זָהֵר-שָׁהּ אֶרַשׂ לוֹ אֶת בִּתּוֹ בִרְצוֹנוֹ הַטּוֹב, וְהַשׁוֹפְטִים חָרְצוּ מִשְׁפָּטָם כִּי הָאֵרוּשִׂים נַעֲשׂוּ כַּדִּין וְכַדָּת. וַיְבָרְכוּ הַקְּרוּאִים אֶת-הַמֶּלֶךְ וְאֶת-בִּתּוֹ בְּהַצְלָחָה. אָז יָקוּם הַוָּזִיר וַיָבֵא אֶת-הַתְּשׁוּרוֹת הַיְקָרוֹת, וַיִתְּנֵן עַל יַד הַמֶּלֶךְ זָהֵר-שָׁהּ.

וְהַמֶּלֶךְ הֵחֵל לְהָכִין אֶת-שִׁלוּחֵי בִתּוֹ, וַיַּעַשׂ יוֹם יוֹם מִשְׁתֶּה לַגְּדוֹלִים וְלַקְּטַנִּים. וַתֶּרֶב הַשִׂמְחָה בְּהֵיכָלוֹ מֶשֶׁךְ חָדְשַׁיִם יָמִים, וְלֹא הָיָה דָבָר מְשַׂמֵּחַ לֵב וּמַרְהִיב עַיִן אֲשֶׁר לֹא כוֹנְנָהוּ. וַיְהִי כַהֲתִמּוֹ לְהָכִין כָּל-חֶפְצֵי הַכַּלָּה, וַיְצַו וַיוֹצִיאוּ אֶת-הָאֹהָלִים וַיִּתְקְעוּם מִחוּץ לָעִיר. וַיְמַלְּאוּ אֶת-הַתֵֹבוֹת וְהָאַרְגָּזִים בְּכָל-אֲרִיגֵי יְקָר, וַיָּכִינוּ אֶת-הַצְּעִירוֹת הָרוֹמִיּוֹת וְהָעֲלָמוֹת הַתֻּרְכִּיּוֹת, וְעִם הַכַּלָּה שָׁלְחוּ כָל-מִכְמַנֵּי הָאוֹצָרוֹת וְכָל-אַבְנֵי הַיְקָר. וְאַחַר עָשָׂה לָהּ אָבִיהָ אַפִּרְיוֹן זָהָב סָגוּר מְשֻׁבָּץ אֲבָנִים טוֹבוֹת וְנִשָּׂא עַל גַּבֵּי עֲשָׂרָה פְרָדִים. וַיְהִי הָאַפִּרְיוֹן כְּהֵיכַל חֶמֶד וְהַיּוֹשֶׁבֶת בּוֹ כִּבְתוּלָה מִבְּתוּלוֹת גַּן-הָעֵדֶן. וַיַּאַרְזוּ אֶת-הַסְּגֻלּוֹת וְהַגְּנָזִים וַיִּשָּׂאוּם עַל הַפְּרָדִים וְהַגְּמָלִים. וְהַמֶּלֶךְ זָהֵר-שָׁהּ הוֹלֵך אִתָּם לְשַׁלְּחָם דֶּרֶךְ שָׁלֹשׁ פַּרְסָאוֹת; וְאַחַר בֵּרַך אֶת-בִּתּוֹ וְאֶת-הַוָּזִיר וְכָל-אֲשֶׁר לוֹ לְשָׁלוֹם, וַיָשָׁב אֶל בֵּיתוֹ שָׂמֵחַ וְטוֹב לֵב.

וְהַוָּזִיר נָסַע עִם בַּת הַמֶּלֶךְ, וַיַעֲבֹר מְדִינוֹת וּמִדְבָּרוֹת, וַיִּסַּע יוֹמָם וָלַיְלָה עַד אֲשֶׁר נִשְׁאַר בֵּינוֹ וּבֵין אַרְצוֹ דֶרֶךְ שְׁלֹשֶׁת יָמִים, וַיִּשְׁלַח אֶל הַמֶּלֶךְ שׂוּלַיְמָן-שָׁהּ לְבַשְּׂרֵהוּ, כִּי הוּא בָא עִם הַכַּלָּה. וַיְמַהֵר הַמַּלְאָךְ וַיָבאֹ עַד לִפְנֵי הַמֶּלֶךְ וַיְבַשְּׂרֵהוּ. וַיִּשְׂמַח הַמֶּלֶךְ וַיִּתֵּן לַמַּלְאָךְ מַחֲלָצוֹת, וְאֶת-צְבָאוֹ צִוָּה לָצֵאת בִּכְבוּדָה גְּדוֹלָה לִקְרַאת הַכַּלָּה לְכַבְּדָהּ, וְאֶת-נוֹשְׂאֵי הַדְּגָלִים צִוָּה לָלֶכֶת אִתָּם וּלְנוֹפֵף דִּגְלֵיהֶם עַל רֹאשָׁם. וְהַכָּרוֹז מַכְרִיז, כִּי לֹא תִשָּׁאֵר נַעֲרָה יוֹשֶׁבֶת בַּיִת, וְלֹא בַת חֹרִים כְּבוּדָה, וְלֹא זְקֵנָה בָלָה, אֲשֶׁר לֹא תֵצֵא לְקַדֵּם אֶת-פְּנֵי הַכַּלָּה. וַיֵּצְאוּ כֻלָּם לִקְרָאתָהּ, וְהַגְּדוֹלִים מִתְאַמְּצִים לְשָׁרְתָהּ. וַיֹּאמְרוּ לְהֲבִיאָהּ אֶל הֵיכַל הַמֶּלֶךְ לַיְלָה, וּלְקַשֵּׁט אֶת-הַדֶּרֶךְ וְלַעֲמֹד עַד אֲשֶׁר תַּעֲבֹר הִיא וְהַמְּשָׁרְתִים לְפָנֶיהָ וְהַשְּׁפָחוֹת עַל יָדָהּ. וַיְהִי בְקָרְבָהּ וְהִיא לְבוּשָׁה הַמַּחֲלָצוֹת אֲשֶׁר נָתַן לָהּ אָבִיהָ, וַיַּקִּיפָהּ הַצָּבָא מִימִינָהּ וּמִשְּׂמֹאלָהּ, וְהָאַפִּרְיוֹן הוֹלֵךְ וּמִתְקַדֵּם עַד אֲשֶׁר קָרְבוּ אֶל הַהֵיכָל, וַיֵּצְאוּ כָל-אַנְשֵׁי הָעִיר לַחֲזוֹת בַּמַּחֲזֶה, וַיַּכּוּ בַּתֻּפִּים. וַיְשַׂחֲקוּ בָרְמָחִים, וַיִּתְקְעוּ בַשׁוֹפָרוֹת. וְרֵיחוֹת הַבְּשָׂמִים נוֹדְפִים, וְהַדְּגָלִים מְנַפְנְפִים, וְהַסּוּסִים דּוֹהֲרִים, וְהַהֲמוֹנִים נוֹהֲרִים, עַד הַגִּיעָם אֶל שַׁעַר הַהֵיכָל, וַיִּשְׂאוּ הַבַּחוּרִים אֶת-הָאַפִּירְיוֹן עַד שַׁעַר-הַסְּתָרִים, וַיֵּאוֹר הַמָּקוֹם בְּיִפְעָתָהּ, וְכָל-פִּנּוֹתָיו הִזְהִירוּ בְהַדְרָתָהּ.

וַיְהִי בְהַגִּיעַ הַלַּיְלָה וַיִּפְתְּחוּ הַמְּשָׁרְתִים אֶת-שַׁעַר הַהֵיכָל, וַיַּעֲמְדוּ עַל יַד הַשַּׁעַר בְּמַעֲרָכָה, וַתָּבוֹא הַכַּלָּה נְהוּגָה בֵין נַעֲרוֹתֶיהָ כְּיָרֵחַ בֵּין כּוֹכְבֵי רוֹם, וְכַמַּרְגָּלִית הַתִּיכוֹנָה בֵּין הַחֲרוּזוֹת, וַתָּבוֹא הַחַדְרָה, וְהִנֵּה שָׁם עָרוּך כִּסֵּא הַשַׁיִשׁ הַמְּשֻׁבָּץ בְּאַבְנֵי חֵן, וַתֵּשֶׁב עָלָיו. וַיָּבֹא הַמֶּלֶךְ אֵלֶיהָ, וַיִּתֵּן הָאֱלֹהִים אֶת-חִנָּהּ בְּעֵינָיו, וַיֵּשֶׁב אִתָּה יֶרַח יָמִים וַתַּהַר לוֹ.

וַיְהִי כַּעֲבֹר הַחֹדֶשׁ וַיֵּצֵא הַמֶּלֶךְ וַיֵּשֶׁב עַל כִּסֵּא מַלְכוּתוֹ, וַיִּשְׁפֹּט בְּצֶדֶק בֵּין נְתִינָיו, עַד עֲבֹר תִּשְׁעַת יַרְחֵי הֶרְיוֹנָהּ. וַתֵּשֶׁב עַל הָאָבְנָיִם וַתֵּקַל בְּלִדְתָּהּ וַתֵּלֶד בֵּן. וַיִּשְׂמַח הַמֶּלֶךְ לַבְּשׂוֹרָה, וַיִּתֵּן לַמְּבַשֵׂר רְכוּשׁ גָּדוֹל וַיֵּלֶךְ לִרְאוֹת אֶת-הַיֶּלֶד, וַיִּשָּׁקֵהוּ וַיִּתְפַּלֵּא עַל יָפְיוּ.

וְהַמְיַלְּדוֹת לָקְחוּ אֶת-הָרַךְ הַנּוֹלָד וַתִּכְרֹתְנָה טַבּוּרוֹ, וַתִּכְחַלְנָה עֵינָיו, וַתִּקְרֶאנָה לוֹ בְשֵׁם תָּג'-אַלְמְלוּךְּ1. וַיִּינַק הַיֶּלֶד מִשְּׁדֵי הַתַּפְנוּקִים, וְהוּא הוֹלֵךְ וְגָדֵל בְּחֵיק הַתַּעֲנוּגִים, וַיַּעַבְרוּ הַיָּמִים, וַתַּחֲלפְֹנָה הַשָּׁנִים, וַיְהִי הַיֶּלֶד לְבֶן שֶׁבַע. אָז קָרָא הַמֶּלֶךְ שׂוּלַיְמָן-שָׁהּ לַחֲכָמִים וְלַיּוֹדְעִים וַיְצַוֵּם לְלַמֵּד אֶת-בְּנוֹ הַכְּתָב וְהַמִּכְתָב וְהַחָכְמָה וְהַמוּסָר, וַיַעֲשׂוּ כֵן שְׁנָתַיִם יָמִים. וַיִּלְמַד הַנַּעַר כָּל-הַדָּרוּשׁ לוֹ לְדַעְתּוֹ. וַיְהִי כְּכַלּוֹתוֹ לִלְמֹד כָּל-אֲשֶׁר דָּרַשׁ הַמֶּלֶךְ אָבִיו וַיָּבֵא לוֹ מוֹרֶה לְלַמְּדֵהוּ הַפָּרָשׁוֹת, וַיְלַמְּדֵהוּ עַד מְלֹאת לוֹ אַרְבַּע עֶשְׂרֶה שָׁנָה. וַיְהִי בְכָל-צֵאתוֹ בַקָּהָל וַיִשְׁבְּ אֶת-לֵב כָּל-רוֹאָיו, וְכָל-לֵב נָטָה לְאַהֲבָתוֹ לְיָפְיוֹ הַמַּזְהִיר. וַיְהִי בְהַגִּיעוֹ לִשְׁנַת הַשְּׁמוֹנֶה עֶשְׂרֵה וַתִּהְיֶינָה לְחָיָיו כַּעֲרוּגוֹת הַבֹּשֶׂם, וַעֲלֵיהֶן נְקֻדַּת-יֳפִי שְׁחוֹרָה כְטִפַּת הָעַנְבָּר, וְהוּא כֻלּוֹ כְדִבְרֵי הַמְשׁוֹרֵר:
יֻלַּד חֲלִיף יוֹסֵף בְּיָפְיֵהוּ

יֶחֶרְדוּ הַחוֹשְׁקִים לְמַרְאֵהוּ,

סוּרָה הֲלוֹם, אָחִי, וְרָאִיתָ

אוֹת מַלְכוּת נוֹצֵץ בְּלֶחְיֵהוּ.

וּכְדִבְרֵי הַמְשׁוֹרֵר:

שְׂפַת אֹדֶם בְּטוּרִים מִפְּנִינִים

וְעֵינַיִם כְּמוֹ חִצִּים שְׁנוּנִים

וְרַקַּת הוֹד כְּשׁוֹשַׁנָה אֲדֻמָּה

וּפָנִים נִטְּעוּ בָהֶם עֲדָנִים.

וַיִהְיוּ לְתָג'-אַלְמְלוּךְ יְדִידִים וְרֵעִים רַבִּים וְכָל-הַקָּרֵב אֵלָיו אָמַר: מִי יִתֵּן וּרְאִיתִיו יוֹשֵׁב עַל כִּסֵּא הַמְּלוּכָה וְאָנֹכִי אֶהְיֶה אַחַד מְשָׁרְתָיו.

וְהַצַּיִד הָיָה אֲהוּב שַׁעֲשׁוּעָיו, וַיִּשְׁעֶה וַיִּשְׁגֶּה בוֹ תָמִיד. וְהֶמֶּלֶךְ אָבִיו הִתְאַמֵּץ לְמָנְעוֹ מִמֶּנּוּ, כִּי יָרֵא לְנַפְשׁוֹ מִפְּנֵי חַיּוֹת הַטֶּרֶף וְסַכְּנוֹת הַמִּדְבָּרוֹת, אַךְ הָעֶלֶם לֹא שָׁעָה אֶל דְּבָרָיו.

וַיְהִי הַיּוֹם וַיֹּאמֶר תָּג‘-אַלְמְלוּךְ אֶל מְשָׁרְתָיו: קְחוּ עִמָּכֶם צֵידָה וּמִסְפּוֹא לַעֲשָׂרָה יָמִים; וַיַעֲשׂוּ כַאֲשֶׁר צִוָּם, וַיֵּצֵא עִם בְּנֵי-לִוְיָתוֹ לָצוּד צָיִד. וַיֵּלְכוּ דֶרֶךְ אַרְבַּעַת יָמִים עַד הַגִּיעָם אֶל אֶרֶץ פּוֹרִיָּה, וַיִרְאוּ שָׁם חַיּוֹת רוֹעוֹת, וְעֵצִים פּוֹרְחִים, וַעֲיָנוֹת נוֹבְעִים. וַיֹּאמֶר תָּג’-אַלְמְלוּךְּ אֶל בְּנֵי לִוְיָתוֹ: מִתְחוּ פֹה אֶת-הַחֲבָלִים, וְהַרְחִיבוּ אֶת גְּבוּלוֹת הַמִּכְלָא, וּבְמָקוֹם פְּלוֹנִי אַלְמוֹנִי נִוָּעֲדָה יָחַד. וַיַּעֲשׂוּ כָּכָה. וַיַּקִּיפוּ אֶת-הַמָּקוֹם בַּחֲבָלִים וַיַּרְחִיבוּ אֶת-גְּבוּלָם. וַתִּקָּבַצְנָה בַּמָּקוֹם הַהוּא חַיוֹת רַבּוֹת לְמִינֵיהֶן: צְבָאִים וָעֳפָרִים, וְתוֹאִים וּזְמָרִים, וַיְהִי רַעַשׁ גָּדוֹל, וַתָּנֹסְנָה הַחַיּוֹת מְנוּסַת חֶרֶב מִפְּנֵי הַסּוּסִים וְרוֹכְבֵיהֶם, וַיְגָרוּ בָּהֶן אֶת-כַּלְבֵי הַצַיִּד, וְאַחַר יָרוּ עֲלֵיהֶן בְּחִצֵּיהֶם וַיִּירוּ אֶל הַשַּׂעֲרָה וְלֹא הֶחְטִיאוּ, וְעוֹד טֶרֶם יַגִּיעוּ אֶל קְצֵה הַמִּכְלָא הַשֵׁנִי רַב הַצַּיִד וְהַמַּלְקוֹחַ. אָז יָצָא תָּג‘-אַלְמְלוּךְּ אֶל מַעְיָן הַמַּיִם, וַיָּבִיאוּ לְפָנָיו אֶת-הַצַּיִד וַיְחַלְּקֵהוּ. וַיָּרֶם לְאָבִיו שׂוּלַיְמָן-שָׁהּ מִמֵּיטַב צֵידוֹ וַיִּשְׁלָחֵהוּ אֵלָיו, וְחֵלֶק שָׁלַח אֶל רַבֵּי הַמְּלוּכָה. וַיָּלֶן בַלַּיְלָה הַהוּא בַּמָּקוֹם, וַיְהִי בְּאוֹר הַבֹּקֶר וַיַּרְא וְהִנֵּה אֹרְחָה הוֹלֶכֶת וּקְרוֹבָה, וְהָאֹרְחָה גְדוֹלָה, וּבָהּ סוֹחֲרִים וַעֲבָדִים וַעֲלָמִים, וַתִּחַן הָאֹרְחָה עַל יַד הַמַּיִם וְהַיָּרָק. וַיְהִי כִרְאוֹת תָּג’—אַלְמְלוּךְּ אֶת-הָאֹרְחָה וַיֹּאמֶר לְאֶחָד מֵרֵעָיו: לֵךְ וְהוֹדִיעֵנִי דָבָר עַל אוֹדוֹת הָאֹרְחָה הַזֹּאת, וּשְׁאַלְתָּם לָמָּה חָנוּ בַמָּקוֹם הַזֶּה. וַיְהִי בְּבוֹא הַמַּלְאָךְ אֲלֵיהֶם וַיֹּאמֶר לָהֶם: הַגִּידוּ נַא לִי מִי אַתֶּם, וַיְמַהֲרוּ וַיַּעֲנוּ וַיֹּאמְרוּ: סוֹחֲרִים אֲנָחְנוּ, וַנִּחַן בַּמָּקוֹם הַזֶּה לָנוּחַ, כִּי דַרְכֵּנוּ רְחוֹקָה, וְיָדוֹעַ יָדַעְנוּ כִּי הִנְנוּ פֹה בְצֵל הַמֶּלֶךְ שׂוּלַיְמָן-שָׁהּ וּבְנוֹ, וְכָל-הַחוֹסֶה בְצִלָּם לִבּוֹ נָכוֹן בָּטוּחַ. וְאִתָּנוּ אֲרִיגִים יְקָרִים אֲשֶׁר הֵבֵאנוּ לִבְנוֹ תָּג’—אַלְמְלוּךְּ וַיָּשָׁב הַמַּלְאָך אֶל בֶּן-הַמֶּלֶךְ וַיּוֹדִיעֵהוּ כָּל-אֲשֶׁר שָׁמַע מִפִּי הַסּוֹחֲרִים. וַיֹּאמֶר תָּג’-אַלְמְלוּךְּ: אִם כֵּנִים הָאֲנָשִׁים הָאֵלֶּה וְהֵבִיאּו אִתָּם חֲפָצִים לְמַעֲנִי, לֹא אָשׁוּב הָעִירָה עַד אִם יִפְרְשׂוּם לְנֶגֶד עֵינָי. וַיִּרְכַּב עַל סוּסוֹ וּמְשָׁרְתָיו אַחֲרָיו עַד הַגִּיעוֹ אֶל הָאוֹרְחָה. וַיָּקוּמוּ הַסּוֹחֲרִים מִפָּנָיו וַיְבָרְכוּהוּ לְאֹשֶׁר וּלְהַצְלָחָה, וַיִּתְקְעוּ לוֹ אֹהֶל הַקְּטִיפָה הָאֲדֻמָּה הָרְקוּמָה בְאַבְנֵי יְקָר, וַיִפְרְשׂוּ לוֹ מַרְבַדֵּי תִפְאָרָה, וּשְׁטִיחֵי רִקְמָה נֶהְדָּרָה. וַיִּשְׁלַח וַיִּקְרָא אֶת-הַסּוֹחֲרִים וַיְצַוֵּם לְהָבִיא לְפָנָיו כָּל-אֲשֶׁר אִתָּם. וַיָּבִיאוּ הַסּוֹחֲרִים סְחוֹרוֹתֵיהֶם וַיִּתְבּוֹנֵן בְּכֻלָּן, וַיִּקַּח מֵהֶן כָּל-הַכָּשֵׁר בְּעֵינָיו וַיְשַׁלֵּם לָהֶם מְחִירוֹ, וַיִּרְכַּב עַל סוּסוֹ וַיֹּאמֶר לָלֶכֶת.

עוֹד הוּא מִתְכּוֹנֵן לָלֶכֶת וְהִנֵּה הֵעִיף עֵינָיו בָּאוֹרְחָה וַיַּרְא בֵּינֵיהֶם אִישׁ צָעִיר יְפֵה-תֹאַר, אַךְ יָפְיוֹ הוּעַם וְעַל פָּנָיו חִוְרוֹן הָאֲהוּבִים הַנִּפְרָדִים וּמֵעֵינָיו עוֹרְפוֹת דְּמָעוֹת, וְהוּא שָׁר לֵאמֹר:

נֶהְמֶה הֲמוֹת יוֹנִים עֲלֵי עַנְפֵי הֲדַס / כִּי יִזְכְּרוּ פֵרוּד וְיִתְאוֹנֵנוּ,

יִבְכּוּ בְלִי דֶמַע וְיִטּוּ מִבְּלִי / יַיִן כְּשִׁכּוֹרִים וְיִתְרוֹנֵנוּ

הַיּוֹם אֲשֶׁר עָלָה נְדוֹד בָּנוּ הֲלֹא / טוֹב לַעֲלוֹת מָוֶת בְּחַלּוֹנֵנוּ,

נִבְכֶּה לְשֹׁד פֵּרוּד וְנִשְׂחַק לַהֲמוֹן / יַלְדֵי יְמוֹתֵינוּ וְלִזְמַנֵּנוּ.

וְאַחַר בָּכָה בְכִי רָב וַיִתְעָלָּף. וְתָג‘-אַלְמְלוּךְּ מַבִּיט אֵלָיו מַחֲרִישׁ מִשְׁתָּאֶה. וַיְהִי בַהֲקִיצוֹ וַיַּרְא וְהִנֵּה בֶן-הַמֶּלֶךְ עוֹמֵד עַל יָדוֹ, וַיֶּחֱרַד וַיָּקָם וַיַּעֲמֹד עַל רַגְלָיו וַיִּקֹּד לְפָנָיו. וַיֹאמֶר אֵלָיו תָּג’-אַלְמְלוּךְּ: מַדּוּעַ זֶה לֹא פָרַשְׂתָּ סְחוֹרוֹתֶיךָ לְפָנָי? וַיֹּאמֶר: אֲדוֹנִי, אֵין בִּסְחוֹרוֹתַי דָּבָר יִכְשַׁר לְמַעֲלָתֶךָ. וַיֹּאמֶר: פְּרָשׂ-נָא לְפָנַי סְחוֹרוֹתֶיךָ וְהוֹדִיעֵנִי עִנְיָנֶךָ, כִּי הִנְנִי רוֹאֲךָ דּוֹמֵעַ הָעֵינַיִם וְנִכְאֵה לֵבָב. וְעַתָּה אִם עָשׁוּק הִנְּךָ נָסִיר עָשְׁקֶךָ, וְאִם נוֹשִׁים נוֹשִׁים בְּךָ נְשַלֵּם נִשְׁיֶךָ, כִּי נִשְבַּר לִבִּי בְקִרְבִּי עָלֶיךָ לְמֵעֵת רְאִיתִיךָ. וַיְצַו תָּג‘-אַלְמְלוּךְּ וַיַּעַרְכוּ שְׁנֵי כִסְאוֹת עֲצֵי הָבְנִים וַיִּפְרְשׂוּ עֲלֵיהֶם מַצַּע מֶשִׁי. וַיֵּשֶׁב תָּג’-אַלְמְלוּךְּ עַל כִּסְאוֹ, וְאֶת-הַצָּעִיר צִוָּה לָשֶׁבֶת עַל הַשָּׁטִיחַ. וַיֹּאמֶר אֵלָיו פְּרָשׂ-נָא לְפָנַי סְחוֹרָתֶךָ. וַיֹּאמֶר הַצָּעִיר: בִּי אֲדוֹנִי, אַל נָא תָאִיץ בִּי, כִּי אֵין בִּסְחוֹרָתִי דָּבָר יִסְּכָּן-לָךְ. וַיֹּאמֶר אֵלָיו תָּג‘-אַלְמְלוּךְּ: וִיהִי מָה, פָּרְשֶַׂנָּה לְפָנָי. וַיְצַו הַצָּעִיר אֶת-אֶחָד מֵעֲלָמָיו וַיְבִיאֶנָּה. וַיְהִי אַךְ רָאֹה רָאָהּ וַתִּזַּלְנָה דִמְעוֹתָיו וַיֵבְךְ וַיִּתְאוֹנֵֹן וַיֶּאֱנֹק דֹּם. וְאַחַר הִתִּיר צְרוֹר סְחוֹרָתוֹ וַיִּפְרְשֶׂנָּה לִפְנֵי תָג’-אַלְמְלוּךְּ אַחַת לְאֶחָת. וּבֵין הַסְּחוֹרוֹת שִׂמְלַת קְטִיפָה אֲרוּגָה חוּטֵי זָהָב, אַלְּפָּיִם דֵּינָר עֶרְכָּהּ. וַיְהִי בְגוֹלְלוֹ אֶת-הַשִּׂמְלָהּ וְהִנֵּה נָפְלָה מִמֶּנָּה מִטְפַּחַת. וַיְמֵַהר הַצָּעִיר וִַיִּקָּחֶנָּה וַיְשִׂימֵנָּה תַּחַת יְרֵכוֹ, וּבִינָתוֹ הִסְתַּתְּרָה וַיָּשָׁר:

אִם הַפְּרִידָה עַל שְׁנֵינוּ נִגְזְרָה / עִמְדִי מְעַט עַד אֶחֱזֶה פָנַיִךְ,

לֹא אֵדְעָה אִם בֵּין צְלָעַי נֶעֱצַר / לִבִּי, וְאִם יֵלֵךְ לְמַסָּעַיִךְ,

בֵּינִי וּבֵינֵךְ יָם דְּמָעוֹת יֶהֱמוּ / גַּלָּיו, וְלֹא אוּכַל עֲבֹר אֵלָיִךְ,

לוּ אַחֲרֵי מוֹתִי בְאָזְנָי יַעֲלֶה / קוֹל פַּעֲמוֹן זָהָב עֲלֵי שׁוּלַיִךְ,

אוֹ תִשְׁאֲלִי לִשְׁלוֹם יְדִידָךְ מִשְּׁאוֹל / אֶשְׁאַל בְּדוֹדַיִךְ וּבִשְׁלוֹמַיִךְ.

מַה תַּחְפְּצִי מוֹתִי וְהֵן אֶחְפֹּץ אֲנִי / שָׁנַי לְהוֹסִיף עַל שְׁנֵי חַיָּיִךְ.

וַיִּתְפַּלֵּא תָּג‘-אַלְמְלוּךְ עַל שִׁירָתוֹ תַּכְלִית פֶּלֶא, וְלֹא יָדַע מָה הַדָּבָר, וַיְהִי בְּקַחַת הַצָּעִיר אֶת-הַמִּטְפַּחַת וַיְשִׂימֶנָּה תַּחַת יְרֵכוֹ, וַיֹּאמֶר אֵלָיו תָּג’-אַלְמְלוּךְ: מָה-הַמִּטְפַּחַת הָזֹּאת? וַיֹאמֶר אֵלָיו: אֲדוֹנִי, מַה-לְךָ וְלַמִּטְפַּחַת? וַיֹּאמֶר: הַרְאֵנִי אוֹתָהּ. וַיֹּאמֶר אֵלָיו: אֲדוֹנִי, הֵן רַק בִּגְלָלָהּ מֵאַנְתִּי לִפְרֹשׂ לְפָנֶיך סְחוֹרוֹתַי, וְלֹא אוּכַל תִּתְּךָ לִרְאוֹתָהּ. וַיֹּאמֶר תָּג‘-אַלְמְלוּךְ: רָאֹה אֶרְאֶנָּה! וַיִּפְצַר בּוֹ וַיִּקְצֹף. וַיּוֹצִיאֶנָֻּה הַצָּעִיר מִתַּחַת יְרֵכוֹ וַיֵבְךְּ וַיֵּאָנֵק. וַיֹּאמֶר אֵלָיו תָּג’-אַלְמְלוּךְ: רוֹאֶה הִנְנִי אֶת-עִנְיָנֶיךָ מִתַּעְתְּעִים; וְעַתָּה הַגִּידָה-נָא לִי מַה-דָּבַר בִּכְיְךָ בְּהַבִּיטְךָ אֶל הַמִּטְפַּחַת הַזֹּאת? וַיְּהִי כִּשְׁמֹעַ הַצָּעִיר אֶת-זֵכֶר הַמִּטְפַּחַת וָיֵּאָנַח וַיֹּאמַר: אֲדוֹנִי, סִפּוּרִי פֶּלֶא וְעִנְיָנִי מוּזָר בַּמִּטְפַּחַת הַזֹּאת וּבְבַעֲלַת הַצּוּרוֹת הָאֵלֶּה אֲשֶׁר עָלֶיהָ. וַיִּפְרֹשׂ אֶת-הַמִּטְפַּחַת וְהִנֵּה בָהּ צוּרַת צְבִי רְקוּמָה מֶשִׁי וּמְקֻשָּׁטָה זָהָב סָגוּר, וְלִפְנֵי הַצּוּרָה צוּרַת צְבִי שֵׁנִי רְקוּמָה בְכֵֶסֶף וּבְצַוָּארֹו עֲנַק זָהָב וּשְׁלֹשֶׁת קְנֵי בָרֶקֶת. וַיְהִי בְהִתְבּוֹנֵן תָּג'-אַלְמְלוּךְ בַּמִּטְפַּחַת וּבִיפִי מְלַאכְתָּהּ וַיֹּאמֶר: בָּרוּךְ הַמְּלַמֵד לֶאָדָם אֶת-אֲשֶׁר לֹּא יָדָע! וַיִּשׁתּוֹקֵק מְאֹד לָדַעַת מַה-דְּבַר הַצָּעִיר וְעִנְיָנוֹ, וַיֹּאמֶר אֵלָיו: סַפְּרָה נָא לִי סִפּוּרְךָ עִם בַּעֲלָת הַצְבָעִים הָאֵלֶּה.
