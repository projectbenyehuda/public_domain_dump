מלחמת היהודים ברומאים וחורבן ירושלים

סקירה היסטורית

ברגשי כבוד ויקר.

למר נח ב"ר יונה יהודה מושער

1

עַמִּים רַבִּים נִכְבְּשׁוּ מֵאֵת הָרוֹמָאִים מִיּוֹם הֱיוֹתָם לְעָם. עַמִּים גְּדוֹלִים וַעֲצוּמִים נָפְלוּ וְלֹא יָסְפוּ קוּם תַּחַת מַכּוֹת גִּבּוֹרֵי רוֹמָא. הַיּוֹנֵק הַקָּטֹן, אֲשֶׁר עָלָה עַל שְׂפַת נְהַר טִבֶּר בְּאִיטַלִּיָּה, גָּדַל וַיְהִי לְאֶרֶז אַדִּיר, אֲשֶׁר כִּסָּה בַעֲנָפָיו אֶת כָּל הָאֲרָצוֹת הַיְּדוּעוֹת לַקַּדְמוֹנִים, אֶת כָּל הָאֲרָצוֹת הַמַּקִּיפוֹת אֶת הַיָּם הַגָּדוֹל הוּא הַיָּם הַתִּיכוֹן. בְּחָכְמָתָם וּבְאוֹפַן מֶמְשַׁלְתָּם יָדְעוּ הָרוֹמָאִים לְאַחֵד אֶת כָּל הָעַמִּים הַנִּכְבָּשִׁים לְעַם אֶחָד אֲשֶׁר שָׂפָה אַחַת וְרוּחַ אַחַת לוֹ. בְּרַק הַשְׂכָּלַת הַיְּוָנִים אֲשֶׁר הָרוֹמָאִים הָיוּ מְבִיאִים אִתָּם לְכָל הָאֲרָצוֹת אֲשֶׁר כָּבְשׁוּ לָקַח אֶת לֵב הַנִּכְבָּשִׁים וַיָחֵלוּ גַם הֵם לִדְגֹל בַּהַשְׂכָּלָה הַזֹּאת וְלַחֲשֹׁב אֶת עַצְמָם לְרוֹמָאִים גְּמוּרִים.

רַק עַם אֶחָד נִמְצָא בְתוֹךְ הֲמוֹן הַגּוֹיִם, אֲשֶׁר בָּאוּ בְמַגָּע עִם הָרוֹמָאִים, אֲשֶׁר לֹא חָפֵץ לְהִתְבּוֹלֵל בְּתוֹךְ הָעָם הָעֲנָק הַזֶּה וַאֲשֶׁר הוֹסִיף לִשְׁמֹר עַל קִנְיָנָיו הָרוּחָנִים וְהַמּוּסָרִים וּלְהִתְפַּתֵּחַ בָּרוּחַ אֲשֶׁר הָיְתָה לוֹ מוֹרָשָׁה מֵאֵת אֲבוֹתָיו וַאֲבוֹת אֲבוֹתָיו. הָעָם הַזֶּה הָיָה עַם הַיְּהוּדִים. אַדְמָתוֹ הַקְּטַנָּה הָיְתָה כְבָר לְשָׁלָל לָרוֹמָאִים יַחַד עִם כָּל אַרְצוֹת סוּרִיָּה, צִבְאוֹתֵיהֶם הָיוּ עוֹבְרִים בָּהּ מֵאַסִּיָּה לְמִצְרַיִם וּמִמִּצְרַיִם לְאַסִּיָּה. בִּירוּשָׁלַיִם עָמַד מַצַּב הַלִּגְיוֹנוֹת וּבְקֵסָרִיָּה יָשָׁב נָצִיב. וּבְכָל זֹאת הוֹסִיף הָעָם הַקָּטֹן לִהְיוֹת עָם לְבָדָד יִשְׁכֹּן וּבְגוֹיֵי רוֹמָא לֹא יִתְחַשָּׁב. אוֹהֲבֵי הַחֵרוּת בָּעָם הַזֶּה לֹא חָדְלוּ אַף רֶגַע לַחֲשֹׁב מַחְשָׁבוֹת לְגָרֵשׁ אֶת הַזָּרִים כָּלָה מֵאַרְצָם. מַצַּב הַמִּלְחָמָה הָיָה לְמַצָב רָגִיל, לְמַצָב תְּמִידִי, בֵּין הָרוֹמָאִים וְהַיְּהוּדִים. כָּל הָר, כָּל סֶלַע, כָּל מְעָרָה נֶהְפְּכוּ לְמִבְצָרִים, אֲשֶׁר מֵהֶם הָיוּ יוֹצְאִים כְּפַעַם בְּפַעַם גְּדוּדִים גְּדוּדִים לְהַחֲרִיד אֶת חֵיל אוֹיְבֵיהֵם הָרוֹמָאִים. לַשָּׁוְא הָיוּ דָנִים הָרוֹמָאִים אֶת אַנְשֵׁי הַגְּדוּדִים הָאֵלֶּה דִינֵי שׁוֹדְדִים עֲלֵי הַדֶּרֶךְ וַיִהְיוּ מַעֲנִישִׁים אוֹתָם קָשֶׁה. הַשּוֹדְדִים הָאֵלֶּה הָיוּ בֶאֱמֶת גִּבּוֹרֵי הָעָם, אֲשֶׁר נִשְׁבְּעוּ לְהִלָּחֵם עִם אוֹיְבֵיהֶם עַד טִפַּת דָּמָם הָאַחֲרוֹנָה וַיַּעֲבִידוּ אֶת חֵיל הָרוֹמָאִים עֲבוֹדָה גְדוֹלָה. חָסֵר הָיָה לָאֲנָשִׁים הָאֵלֶּה רַק גִבּוֹר אֶחָד, גִבּוֹר, כְּמוֹ יְהוּדָה הַמַּכְבִּי וְאֶחָיו, אֲשֶׁר יָכֹל לְאַחְדֵם יַחַד וּלְהַצִּית אֶת אֵשׁ הַמֶּרֶד בְּכָל הָאָרֶץ בְּפַעַם אַחַת.


2

אִם נִתְבּוֹנֵן אֶל תְּכוּנַת נֶפֶש שְׁנֵי הָעַמִּים, הַיְּהוּדִים וְהָרוֹמָאִים, נָבִין עַל נְקַלָּה אֶת סִבַּת הַשִּׂנְאָה הָעֲמוּקָה אֲשֶׁר הָיְתָה בֵינֵיהֶם. מִלְחֶמֶת הַיְּהוּדִים וְהָרוֹמָאִים לֹא הָיְתָה רַק מִלְחֶמֶת שְׁנֵי עַמִּים הַנִּלְחַמִים בֵּינֵיהֶם בְּעַד חֵרוּת מְדִינִית, כְּדֶרֶךְ כָּל הָעַמִּים, הִיא לֹא הָיְתָה מִלְחָמָה בְּעַד אֵיזֶה רֶוַח גַּשְׁמִי, מִלְחָמָה אֲשֶׁר לְפִי חֻקֶּיהָ הַקַּדְמוֹנִים הָיָה הַמְּנֻצָח מְקַבֵּל עָלָיו אֶת מָרוּת הַמְּנַצֵחַ. הַמִּלְחָמָה הַזֹּאת הָיְתָה מִלְחֶמֶת שְׁנֵי עוֹלָמוֹת הֲפוּכִים וְזָרִים זֶה לָזֶה, מִלְחָמָה רוּחָנִית, מִלְחֶמֶת שְׁנֵי אִידִיאַלִּים אֲשֶׁר לֹא יָכְלוּ לְהִתְקַיֵּם שְׁנֵיהֶם יַחַד. גַּם הָרוֹמָאִים גַּם הַיְהוּדִים שָאֲפוּ לְבִנְיַן מַמְלָכָה אַחַת גְּדוֹלָה, אֲשֶׁר תָּכִיל בְּקִרְבָּהּ אֶת כָּל חֶלְקֵי תֵבֵל, תְּאַחֵד אֶת כָּל הָעַמִּים. אֲבָל הָרוֹמָאִים שָׁאַפוּ אֶל מֶמְשָׁלָה גַשְׁמִית וְהַיְהוּדִים אֶל מֶמְשָׁלָה רוּחָנִית, הָרוֹמָאִים חָפְצוּ לִבְנוֹתָה בְּכֹחַ הַזְּרֹעַ וְהַיְהוּדִים בְּכֹחַ הָרוּחַ. אִידִיאַל הָרוֹמָאִים הָיָה לְהָסִיר גְּבוּלוֹת עַמִּים, לְבַלְבֵּל אֶת הָאֻמּוֹת וְלַעֲשׂוֹתָן לְעַם אֶחָד תַּחַת מֶמְשֶׁלֶת הָרוֹמָאִים, מֶמְשָׁלָה גַשְׁמִית בִּמְלֹא מוּבַן הַמִּלָה. וְאִידִיאַל הַיְהוּדִים הָיָה כִּי כָל הָעַמִּים יֵשְׁבוּ אִישׁ אִישׁ עַל נַחְלָתוֹ אֲשֶׁר קִבֵּל מֵאֵת אֲדוֹן כָּל הָאָרֶץ, בְּהַנְחִילוֹ גוֹיִם, בְּהַפְרִידוֹ בְנֵי אָדָם, בְּהַצִּיבוֹ גְבוּלוֹת עַמִּים1. וְרַק בֵּין כָּל הָעַמִּים יִהְיֶה עַם יִשְׂרָאֵל לְעַם סְגוּלָה, לְמַמְלֶכֶת כֹּהֲנִים2, אֲשֶׁר לוֹ תִהְיֶה הַמֶּמְשָׁלָה הָרוּחָנִית, אֲשֶׁר מִמֶּנוּ יִלְמְדוּ כָל הַגּוֹיִם לָדַעַת אֶת ה' וְלָלֶכֶת בְּאֹרְחוֹתָיו3. לָרוֹמָאִים מֶרְכַּז כָּל הָעוֹלָם הָיָה בְּרוֹמָא עַל גִּבְעַת קַפִּיטוֹלִיוּס, וְלַיְהוּדִים הָיָה מֶרְכַּז כָּל הָעוֹלָם בִּירוּשָׁלַיִם עַל גִּבְעַת צִיּוֹן, כִּי מִשָּׁם תֵצֵא תוֹרָה לְכָל עַמֵּי הָעוֹלָם. בֵּין רוֹמָא וִירוּשָׁלַיִם לֹא יָכֹל לִהְיוֹת שָׁלוֹם, כִּי שְׁתֵּיהֶן הִתְנַּשְׂאוּ לִמְלֹךְ עַל כָּל הָעוֹלָם, שְׁתֵּיהֶן הָיוּ צָרוֹת אִשָּׁה לִרְעוּתָהּ. הָרוֹמָאִים אֲשֶׁר עַל חַרְבָּם חָיוּ וּבְכֹחַ יָדָם בָּטָחוּ, לֹא יָכְלוּ לְהָבִין אֶת עַם יִשְׂרָאֵל, אֶת רוּחוֹ וְאֶת שְׁאִיפוֹתָיו הַנַּעֲלוֹת וַיָבוּזוּ לוֹ, וְהַיְּהוּדִים מִצִּדָּם בִּרְאוֹתָם אֶת מְשׁוּבַת הָרוֹמָאִים אֶת אֲמוּנַת הַבְלֵיהֶם אֶת כְּבוֹד אֱלֹהִים אֲשֶׁר הָיוּ נוֹתְנִים לְפִסְלֵי עֵץ וְאֶבֶן וּלְקִסְרֵיהֶם הַמֵּתִים הָיוּ גַם הֵם בָּזִים לָהֶם. וּבִרְאוֹתָם אֶת תַּאֲוָתָם לִבְלֹעַ חֵיל זָרִים וְלָרֶשֶׁת מִשְׁכָּנוֹת לֹא לָהֶם, וּבִפְרַט בִּרְאוֹתָם אֶת מֶמְשַׁלְתָּם בִּיְהוּדָה הָעַקֻבָּה מִדָּם, שָׂנְאוּ אוֹתָם תַּכְלִית שִׂנְאָה.

3

הָרוֹמָאִים הִצְטַיְּנוּ תָמִיד בִּידִיעָתָם לִמְשֹׁךְ אֲלֵיהֶם אֶת לִבּוֹת הָעַמִּים הַנִּכְבָּשִׁים, וְאַחַד הָאֶמְצָעִים הַיּוֹתֵר מְנֻסִּים הָיְתָה הָאֱמוּנָה. בַּיָמִים הָהֵם בְּעֵת שֶׁהָאֱמוּנָה הָיְתָה הַצִּיר הָרֹאשִׁי אֲשֶׁר עָלָיו סָבְבוּ חַיֵי כָּל עַם וְעָם, הָיָה הָאֶמְצָע הַזֶּה בֶאֱמֶת הַיּוֹתֵר טוֹב. וְלָכֵן אַחֲרֵי אֲשֶׁר כָּבְשׁוּ הָרֹומָאִים אֵיזֶה עַם מֵהָעַמִּים, הָיוּ מְקַבְּלִים אֶל תּוֹךְ הֲמוֹן אֱלִילֵיהֶם גַּם אֶת אֱלִילֵי הָעָם הַהוּא וַיִהְיוּ מְפַנִּים לוֹ מָקוֹם עַל מְרוֹם הַקַפִּיטוֹלִיוֹס. בְּרוֹמָא הָיָה הֵיכָל הַיָּדוּעַ בְּשֵׁם “פַּנְתֵּאוֹן” לֵאמר “כָּל הָאֱלֹהִים”, וְשָׁם נֶאֶסְפוּ בְמֶשֶׁךְ הַזְּמַן כִּשְׁלשִׁים אֶלֶף אֱלִיל מִכָּל הָעַמִּים הָרַבִּים, אֲשֶׁר כָּבְשׁוּ. בְּעַד הַכָּבוֹד, אֲשֶׁר הָיוּ נוֹתְנִים לֶאֱלִילָיו הָיָה הָעָם הַנִּכְבָּשׁ מְקַבֵּל גַּם אֶת אֱלִילֵי רוֹמָא וְזוֹבֵחַ לָהֶם זְבָחִים. כִּי סוֹף סוֹף הֲלֹא אֵין הֶבְדֵּל גָּדוֹל בֵּין יֻפִּיתֶר הָרוֹמָאִים, צִיאוֹס הַיְּוָנִים, בֵּל הַכַּשְׂדִים, בַּעַל הַכְּנַעֲנִים אוֹ אָמוֹן אֱלֹהֵי מִצְרַיִם. וְכַאֲשֶׁר הָיָה לְמִנְהָג כִּי קִסְרֵי רוֹמָא יְקַבְּלוּ אַחֲרֵי מוֹתָם “אַפּוֹתֵאוֹזָה” לֵאמֹר כְּבוֹד אֱלֹהִים, לֹא הִתְנַגְּדוּ לָזֶה כְּלַל, כִּי הֲלֹא וְאֶת רוּחָם וְלֹא לִהְיוֹת עַם מִתְגָּרֵה מִלְחָמוֹת בְּכָל אַרְצוֹת תֵּבֵל, פַּטִּיש כָּל הָאָרֶץ וְאֶת הַדָּבָר הַזֶּה לֹא יָכְלוּ הָרוֹמָאִים לַעֲשׂוֹת, כְּמוֹ לֹא יַחְלִיף כּוּשִׁי עוֹרוֹ וְנָמֵר כְּמוֹ לְכָל הָעַמִּים אֲשֶׁר נִגְזְרוּ אַחֲרֵי כֵן מֵאִתָּם מִתַּחַת הַשָּׁמַיִם נוֹדְעוּ הָרוֹמָאִים לַיְּהוּדִים בָּרִאשׁוֹנָה בְתוּר אוֹהֲבִים. בָּעֵת אֲשֶׁר הֵחֵלָה הַמִּלְחָמָה בֵינֵיהֶם וּבֵין מַלְכֵי סוּרִיָּה, שָׂמְחוּ הָרוֹמָאִים לְצִירֵי הַיְּהוּדִים, אֲשֶׁר נִשְׁלְחוּ מֵאֵת יְהוּדָה הַמַּכְבִּי לְבַקֵּש מֵהֶם עֶזְרָה נֶגֶד הַסּוּרִים, וַיִקְרְאוּ לַיְּהוּדִים “יְדִידֵי רוֹמָא”. מִפְּנֵי רִחוּק הַמָּקוֹם לֹא יָכְלָה הַבְּרִית עִם הָרוֹמָאִים לְהָבִיא תוֹעֶלֶת גְּדוֹלָה לְעַם הַיְּהוּדִים. בְּכָל זֹאת חִדֵּשׁ שִׁמְעוֹן הַחַשְׁמוֹנָאִי אֶת הַבְּרִית הַזֹּאת וְגַם הַפַּעַם בְּלִי תוֹעֶלֶת מַמָּשִׁית. הַפְּגִישָׁה הָאֲמִתִּית שֶׁל שְׁנֵי הָעַמִּים הָאֵלֶּה הָיְתָה בִשְׁנַת 133 לִפְנֵי הַחָרְבָּן, בָּעֵת אֲשֶׁר הִגִּיעַ פּוֹמְפִּיּוּס שַׂר צְבָא רוֹמָא עַד שַׁעֲרֵי דַמֶשֶׂק, אַחֲרֵי הַכּוֹתוֹ אֶת מֶלֶךְ אַרְמֵנִיָה וְאֶת מֶלֶךְ סוּרִיָּה. וּמִן הַפְּגִישָׁה הָרִאשׁוֹנָה הִרְגִּישׁוּ הַיְּהוּדִים אֶת צִפָּרְנָיו הַחַדּוֹת שֶׁל נֶשֶׁר רוֹמָא5.

לִמְצֹא חֵן בְּעֵינֵי הָרוֹמָאִים וּלְהַגִּיעַ בְעֶזְרָתָם לִמְלוּכָה, כִּי כָל הַכָּבוֹד, אֲשֶׁר נָחַל בְּתוּר מֶלֶךְ עַל כָּל אֶרֶץ יִשְׂרָאֵל, מֵאֵת הָרוֹמָאִים, שְׂנוּאֵי נֶפֶשׁ הָעָם, בָּא לוֹ, וְכִי כָּל יְמֵי שִׁבְתּוֹ עַל כִּסֵּא הַמְּלוּכָה, לֹא חָדַל לְהַחְנִיף לָרוֹמָאִים, לְמַעַן יְחַזְקוּ אֶת יָדָיו נֶגֶד הָעָם הַסּוֹעֵר. גְּדוּלַת מַלְכוּת הוּרְדוּס הָיְתָה צֵל גְּדוּלָה וְהַצֵּל הַזֶּה לא יָכֹל לְסַמֵּא אֶת עֵינֵי אוֹהֲבֵי הַחֵרוּת. וְלָכֵן בְּעֵת אֲשֶׁר הִשְׁתַּדֵל הוּרְדּוּס בְּאֶמְצָעִים שׁוֹנִים לִקְנוֹת לוֹ אֶת אַהֲבַת הָעָם, בְּעֵת אֲשֶׁר חָשַׁב לִדְמוֹת לִשְׁלֹמֹה הַמֶּלֶךְ בִּבְנוֹתוֹ אֶת בֵּית הַמִּקְדָּשׁ הַמְפֹאָר, בְּעֵת אֲשֶׁר אָגוּסְטוּס קֵסַר רוֹמָא קָרָא לוֹ “יְדִידוֹ ואוֹהֲבוֹ”, לֹא חָדְלָה בִפְנִים הָאָרֶץ תְּסִיסָה חֲזָקָה נֶגֶד הַמֶּלֶךְ הַזֶּה וְנֶגֶד כָּל בֵּיתוֹ וְיַחַד עִמָהֶם גַּם נֶגֶד הָרוֹמָאִים. כַּאֲשֶׁר מֵת הוּרְדּוּס וְעַל כִּסְּאוֹ יָשַׁב בְּנוֹ אַרְכִילַאוֹס, מָאַס הָעָם בְּצֵל הַגְדוּלָה הַזֹּאת וַיְבַקְּשׁוּ מֵאֵת הַקֵּסַר אָגוּסְטוּס, כִּי יָסִיר אֶת הַמֶּלֶךְ הַזֶּה מֵעַל כִּסְּאוֹ וְיוֹשִׁיב נְצִיבִים עַל מְקוֹמוֹ. מִנִי אָז הָיְתָה אֶרֶץ יְהוּדָה לִפְּרוֹבִינְצִיה6 רוֹמָאִית תַּחַת מֶמְשֶׁלֶת נְצִיבִים. אֲבָל הָעָם, אֲשֶׁר קִוָּה לָנוּחַ מֵעֲרִיצוּת הַמְּלָכִים הָאֲדוֹמִים תַּחַת מֶמְשֶׁלֶת נְצִיבִים, הַנִּשְׁלָחִים מִטַּעַם הַקֵּסַר, שָׁגָה שְׁגִיאָה גְדוֹלָה. הַנְּצִיבִים שָׂמוּ אֶל לִבָּם לְמַלֵּא זָהָב חוֹרֵיהֶם וּלְהִבָּנוֹת מֵהַפְּרוֹבִינְצִיָּה, אֲשֶׁר נִתְּנָה לָהֶם, וְלָכֵן לָחֲצוּ אֶת הָעָם בְּחָזְקָה וַיָמֹצוּ אֶת כָּל לְשַׁדּוֹ. הַפְּרוֹבִינְצִיּוֹת הַקְּרוֹבוֹת לְרוֹמָא הָיוּ יוֹתֵר תַּחַת הַשְׁגָחַת הַקֵּסָרִים וְלָכֵן הָיָה אֶפְשָׁר לִפְעָמִים לְהָבִיא לְמִשְׁפָּט נָצִיב עָרִיץ. אֲבָל בַּפְּרוֹבִינְצִיּוֹת הָרְחוֹקוֹת, כְּמוֹ שֶׁהָיְתָה אֶרֶץ יִשְׂרָאֵל, הָיָה הַנָּצִיב הַשַּׁלִיט בְּלִי גְבוּל, בְּלִי פַחַד הַקֵּסָרִים עָלָיו. מִלְּבַד זֶה הָיְתָה אַחֶרֶת גַּם תְּכוּנַת הַנְּצִיבִים. לַפְּרוֹבִינְצִיּוֹת הַקְּרוֹבוֹת הָיוּ נִשְׁלָחִים מִטּוּבֵי הָרוֹמָאִים, אֲנָשִׁים יוֹתֵר נִכְבָּדִים וּנְשׂוּאֵי פָנִים, אֲנָשִׁים אֲשֶׁר יָדְעוּ לְכַלְכֵּל אֵת עִנְיָנֵי הַמְּדִינָה, אֲשֶׁר תַּחַת רְשׁוּתָם בְּצֶדֶק וּבְמִשְׁפָּט. לַפְּרוֹבִינְצִיּוֹת הָרְחוֹקוֹת, רָחוֹק מֵרוֹמָא, רָחוֹק מֵהַחַיִּים הָרוּחָנִים וְהַמְדִינִיִּים, נִשְׁלְחוּ הַפְּחוּתִים בַּנְּצִיבִים, אֲנָשִׁים אֲשֶׁר רַק תַּאֲוַת בֶּצַע מַעֲשַׁקּוֹת הוֹבִילָה אֶת רַגְלֵיהֶם לַעֲבֹר אָרְחוֹת יַמִּים וּלְהִתְגּוֹרֵר רָחוֹק מִשַּׁעֲשׁוּעֵי תֵבֵל, אֲשֶׁר יָכְלוּ לִמְצֹא בְרוֹמָא. אֲנָשִׁים כָּאֵלֶה הָיוּ נְצִיבֵי רוֹמָא בְיהוּדָה וְהַנְּצִיבִים הָאֵלֶּה לא יָכְלוּ לְהַרְגִּיעַ אֶת עַם יְהוּדָה. הַפְּרָעוֹת הַבִּלְבּוּלִים וְהַמְּרִידוֹת לֹא חָדְלוּ וְאֶת אֵשׁ הַמֶּרֶד הָיוּ הַנְּצִיבִים מְכַבִּים בְּדַם אַלְפֵי אֲנָשִׁים. לִפְעָמִים הָיָה הַמֶּרֶד מִתְפַּשֵּׁט בְּמִדָּה יוֹתֵר גְּדוֹלָה וְהָיָה צָרִיךְ כִּי הַנָּצִיב הָעֶלְיוֹן, נְצִיב סוּרְיָּה, יָבֹא לְעֶזְרַת הַנָּצִיב אֲשֶׁר בִּיהוּדָה, וְאָז הָיָה מִסְפַּר הַמּוּמָתִים לִרְבָבוֹת, הֶעָרִים שְׂרוּפוֹת אֵשׁ וְהָאָרֶץ שׁוֹמֵמָה.

5

קֶרֶן אוֹר בְּתוֹךְ הַחֲשֵׁכָה הַזֹּאת הָיְתָה מֶמְשֶׁלֶת אַגְרִפַּס הָרִאשׁוֹן, אֲשֶׁר יָשַׁב עַל כִּסֵּא הַמְלוּכָה בִיְרוּשָׁלַיִם בְּמֶשֶׁךְ אַרְבַּע שָׁנִים. הוּא לֹא הָלַךְ בִּגְדוֹלוֹת כְּמוֹ אָבִיו הַזָּקֵן, הוּרְדוּס, וַיָשֶׂם אֶת לִבּוֹ לְעִנְיָנֵי אַרְצוֹ הַפְּנִימִים. מְגַמָּתוֹ הָיְתָה לְחַזֵּק אֶת הָעָם וְאֶת הָאָרֶץ לְמַעַן יוּכְלוּ לַעֲמֹד נֶגֶד הָרוֹמָאִים, כִּי רָאָה, כִּי מַצָּב כָּזֶה לֹא יָכֹל לְהִמָּשֵׁךְ זְמַן רַב, וְכִי סוֹף סוֹף עֲתִידָה מִלְחָמָה נוֹרָאָה לִפְרֹץ בֵּין הַיְּהוּדִים וְהָרוֹמָאִים. מוֹצָאוֹ מִצַּד אֵם אָבִיו מִמִּשְׁפַּחַת הַחַשְׁמוֹנָאִים מָשְׁכָה אֵלָיו אֶת לֵב הָעָם, אֲשֶׁר אָהַב אוֹתוֹ מְאֹד, וְלוּ הֶאֱרִיךְ יָמִים עַל כִּסְּאוֹ, כִּי אָז אוּלַי הָיָה יָכֹל לְרַפֵּא אֶת בְּקִיעֵי אֶרֶץ יְהוּדָה. הוּא בִּצֵּר אֶת יְרוּשָׁלַיִם בְּחוֹמָה חֲדָשָׁה וְגַם בָּא בִּדְבָרִים עִם רַבִּים מִמּוֹשְׁלֵי סוּרִיָּה לִכְרֹת יַחַד בְּרִית נֶגֶד הָרוֹמָאִים. אֲבָל נְצִיב סוּרִיָּה, יִבִּיוּס מַרְסוּס הֵרִיחַ אֶת הַקֶשֶׁר מֵרָחוֹק וַיִּגְעַר בַּמּוֹשְׁלִים הָאֵלֶה אֲשֶׁר הִתְאַסְּפוּ לְעִיר טִבֶּרִיָה, וַּיְּצַוֵּם לָשׁוּב אִישׁ אִישׁ לְאַרְצוֹ. וְגַּם לִגְמֹר אֶת חוֹמַת יְרוּשָׁלַיִם לֹא נָתַן לוֹ הַנָּצִיב הַזֶּה. לַאֲסוֹן עַם יְהוּדָה מֵת הַמֶּלֶךְ אַגְרִפַּס פִּתְאֹם בִּהְיוֹתוֹ בְקֵסָרִיָּה, וְיֵשׁ לַחֲשֹׁב, כִּי הָרוֹמָאִים הִשְׁקוּהוּ רַעַל אַחֲרֵי אֲשֶׁר לֹא יָכְלוּ לְהִתְאוֹנֵן עָלָיו בְּאָזְנֵי הַקֵּסָר קְלוֹדִיוס, אֲשֶׁר הָיָה אוֹהֲבוֹ.

אַחֲרֵי מוֹתוֹ שָׁבוּ הַנְּצִיבִים לִמְשֹׁל בִּיהוּדָה וְהַיָמִים הָרָעִים הֵחֵלוּ בְיֶתֶר שְׂאֵת. הָרוֹמָאִים, הַיְּוָנִים, הַשֹּׁמְרוֹנִים וְכָל צָרֵי יִשְׂרָאֵל בְּאֶרֶץ יְהוּדָה, אֲשֶר כָּל יְמֵי מַמְלֶכֶת אַגְרִיפַּס הָיוּ נֶאֱלָצִים לְהַסְתִּיר אֶת שִׂנְאָתָם לְעַם הַיְּהוּדִים, הֵרִימוּ עַתָה אֶת רֹאשָׁם וַיָחֵלוּ לְהַצִּיק לָהֶם וְלֶאֱכֹל אֶת יִשְׂרָאֵל בְּכָל פֶּה. בְּכָל הֶעָרִים אֲשֶׁר יָשְׁבוּ בָּהֶן יְהוּדִים יַחַד עִם יְוָנִים, הֵחֵלו מְהוּמוֹת וּפְרָעוֹת וְדָמִים בְּדָמִים נָגָעוּ. הַנְּצִיבִים לֹא עָשׂוּ מְאוּמָה לְהָגֵן עַל הָעָם, וּלְהָשִׁיב אֶת הַסְּדָרִים עַל מְכוֹנָם. בִּפְרַט הִצְטַיֵּן בְּרִשְׁעָתוֹ הַנָּצִיב הָאַחֲרוֹן אֲשֶׁר בִּיהוּדָה, גֶסִּיוּס פְלוֹרוּס. בְּיָמָיו מָלְאָה סְאַת צָרוֹת יִשְׂרָאֵל, וְהָיָה בָּרוּר, כִּי רַק הַחֶרֶב תּוּכַל לִשְׂפֹּט בֵּין הַיְּהוּדִים וּמַצִּיקֵיהֶם הָרוֹמָאִים וְכָל בְּנֵי הַנֵכָר הַנִלְוִים עֲלֵיהֶם. בַּמִּדָּה אֲשֶׁר הִתְרַבּוּ צָרוֹת הַיְּהוּדִים, הִתְחַזְּקוּ בְתוֹכָם אוֹהֲבֵי הַחֵרוּת, הַקַּנָּאִים, אֲשֶׁר הָיוּ מוּכָנִים לִשְׁפּוֹך אֶת דָּמָם עַד הַטִּפָּה הָאַחֲרוֹנָה בְעַד חֵרוּת הָאָרֶץ. וְהַמְּתוּנִים, אֵלֶּה אֲשֶׁר לֹא הֶאֱמִינוּ בְּהַצְלָחַת הַמִּלְחָמָה, נָסֹגוּ אָחוֹר וְהַשְׁפָּעָתָם עַל הָעָם הָלְכָה הָלֹךְ וְקָטֹן. הָיָה צָרִיךְ מִקְרֶה נִכְבָּד, אֲשֶׁר יָעִיר אֶת הָעָם לֶאֱחֹז בְּחֶרֶב וּבַחֲנִית וְלָצֵאת בְּיָד רָמָה נֶגֶד הָרוֹמָאִים. הַמִּקְרֶה הַזֶּה לֹא אֵחַר לָבֹא. מֵעִיר קֵסָרִיָה נִפְתְּחָה הָרָעָה, מֵהָעִיר, אֲשֶׁר הַמֶּלֶך הוּרְדוס יְסָדָהּ וַיִּבֶן בָּהּ הֵיכָלֵי תִפְאָרָה וְחוֹמָה בְצוּרָה מִסָּבִיב לָהּ בְּכֶסֶף הַיְּהוּדִים. הַמְּרִיבָה, אֲשֶׁר פָּרְצָה שָׁם בֵּין הַיְּהוּדִים וְהַיְּוָנִים, הֵבִיאָה מִלְחָמָה בְּתוֹךְ חוּצוֹת הָעִיר וְהַיְּוָנִים הִפִּילוּ חֲלָלִים רַבִּים מֵהַיְּהוּדִים עַד כִּי הָיוּ הַנִּשְׁאָרִים נֶאֱלָצִים לִבְרֹחַ מִן הָעִיר. גֶסִּיוּס פְלוּרוּס, אֲשֶׁר אֵלָיו הִגִּישׁוּ יְהוּדֵי קֵסָרִיָּה אֶת מִשְׁפָּטָם, לֹא שָׁעָה לָהֶם וְנַהֲפֹךְ הוּא, כִּי עוֹד הֶעֱלָה עָלָיו אֶת חֲמַת יושְׁבֵי יְרוּשָׁלַיִם עַד לְהַשִׁחית וַיַּצְלִיחַ לְהַצִּית אֶת אֵשׁ הַמֶּרֶד גַּם בָּעִיר הַזֹּאת. לַשָּׁוְא נִסּוּ אוֹהֲבֵי הַשָּׁלוֹם לַעֲמֹד נֶגֶד הָעָם. שִׁבְעָה יָמִים אָרְכָה הַמִּלְחָמָה בּתוֹךְ הָעִיר בֵּין אוֹהֲבֵי הַשָּׁלוֹם וּבֵין אוֹהֲבֵי הַחֵרוּת וּלְאַחֲרוָנה גָבְרָה יַד אוֹהֲבֵי הַחֵרוּת, אֲשֶׁר הִצְלִיחוּ לְהַשְׁמִיד גַּם אֶת מַצַּב הָרוֹמָאִים בְּתוֹךְ הָעִיר. הַנִּצָחוֹן הַזֶּה חִזֵּק אֶת לֵב הָעָם, אֲשֶׁר נִסְפַּח כֻּלּוֹ עַל אוֹהֲבֵי הַחֵרוּת וַיַּחְלִיט לְהוֹסִיף לְהִלָּחֵם נֶגֶד הָרוֹמָאִים עַד אֲשֶׁר יִפְרֹק אֶת עֻלָּם מֵעַל צַוָּארוֹ.

וְאֵשׁ הַמֶּרֶד, אֲשֶׁר הִתְלַקְּחָה בִיְרוּשָׁלַיִם, הִתְפַּשְּׁטָה בִמְהִירוּת נִפְלָאָה בְכָל אֶרֶץ יִשְׂרָאֵל, וּבְכָל הֶעָרִים הֵחֵלוּ מִלְחָמוֹת בֵּין הַיְהוּדִים וּבֵין בְּנֵי הַנֵּכָר. גַּם הַפַּעַם הָיְתָה הָעִיר קֵסָרִיָּה הָרִאשׁוֹנָה. כַּאֲשֶׁר שָׁמְעוּ יוֹשְׁבֶיהָ הַיְּוָנִים אֶת אֲשֶׁר נַעֲשֶׂה בִיְרוּשָׁלַיִם, הִתְנַפְּלוּ יַחַד עִם חֵיל הָרוֹמָאִים, אֲשֶׁר עָמַד בָּעִיר הַהִיא, עַל הַיְּהוּדִים וַיָּמִיתוּ מֵהֶם עֶשְׂרִים אֶלֶף אִישׁ וְאֶת הַנִשְׁאָרִים מָכַר גֶסִּיוּס פְלוֹרוּס, אֲשֶׁר נִמְצָא אָז שָׁם, לְעַבְדֵי עוֹלָם. הַמַּעֲשֶׂה הַזֶּה הֵעִיר עוֹד יוֹתֵר אֶת קִנְאַת הָעָם וַיּוֹסִיף עֵצִים עַל הַמְּדוּרָה. הַיְהוּדִים גַּם הֵם הִתְנַפְּלוּ בִּמְקוֹמוֹת רַבִּים עַל הַיְּוָנִים וַיַּרְבּוּ אֶת חַלְלֵיהֶם וַיִּשְׂרְפוּ עֲלֵיהֶם אֶת בָּתֵּיהֶם לִנְקֹם אֶת דַּם אַחֵיהֶם הַשָּׁפוּךְ. וְלֹא רַק בְּאֶרֶץ יְהוּדָה הִתְחוֹלְלָה הַמִּלְחָמָה. בְּכָל עָרֵי הַמִּזְרָח: בְּאַנְטִיוֹכִיָּה, בְּצִידוֹן, בְּעַכּוֹ, וְגַם בְּאַלֶכְסַנְדְרִיָּה אֲשֶׁר בְּמִצְרַיִם הִתְנַפְּלוּ הֲמוֹנֵי הָעָם עַל הַיְּהוּדִים, אֲשֶׁר יָשְׁבוּ לָבֶטַח אִתָּם וַיַּגִּירוּם לְפִי חֶרֶב.

וּנְצִיב סוּרִיָּה, קֶסְטִיוּס גַּלּוּס, בְּשָׁמְעוֹ, כִּי מָרְדוּ הַיְּהוּדִים בְּיָד רָמָה בְמֶמְשֶׁלֶת רוֹמָא, הֶחְלִיט לְהַשְׁקִיט אֶת הַמֶּרֶד בְּפַּעַם אַחַת. בְּרֹאשׁ שְׁלשִׁים אֶלֶף אִישׁ גִּבּוֹרֵי הַחַיִל עָלָה הוּא עַל אֶרֶץ יְהוּדָה וַיָשֶׂם פָּנָיו יְרוּשָׁלָיְמָה. בְּכָל מָקוֹם אֲשֶׁר עָבַר הֵבִיא עִמּוֹ שַׁמָּה וּשְׁאִיָּה. עָרִים גְּדוֹלוֹת הָיוּ לְעִיֵּי חֲרָבוֹת, הַשָּׂדוֹת וְהַגַּנִּים הוּשַׁמּוּ וּפִגְרֵי אָדָם הָיוּ כְדֹמֶן עַל פְּנֵי הַשָּׂדֶה. עַל כָּל הַדֶּרֶךְ לֹא פָגַשׁ חֵיל הָרוֹמָאִים מַעֲצָר וּמִכְשֹׁל וְלָכֵן חָשַׁב קֶסְטִיּוּס, כִּי בְאֶפֶס יָד יוּכַל לַיְּהוּדִים. אַךְ כַּאֲשֶׁר הִגִּיעַ עַד יְרוּשָׁלַיִם, יָצְאוּ גְדוּדֵי יְהוּדִים מִן הָעִיר וַיִתְנַפְּלוּ עַל הָרוֹמָאִים בְּחֵמָה שְׁפוּכָה, הִפִּילוּ בָּהֶם חֲלָלִים רַבִּים וַיְּאַלְּצוּם לָסֹג אָחוֹר עַד בֵּית-חוֹרֹן. גִּבּוֹרֵי הַיְּהוּדִים רָדְפוּ אַחֲרֵיהֶם וַיַּכּוּ אֶת חֵיל הַמְּאַסֵּף וַיָשׁוּבוּ בְשָׁלָל רַב וּבִתְרוּעַת נִצָּחוֹן יְרוּשָׁלָיְמָה. בְּעֵת אֲשֶׁר עָמַד קֶסְטִיּוּס בְּבֵית חוֹרֹן, נִסּוּ עוֹד פַּעם אוֹהֲבֵי הַשָּׁלוֹם לְדַבֵּר עַל לֵב הַקַנָּאִים בִּיְרוּשָׁלַיִם לְהַשְׁלִים עִם הָרוֹמָאִים, אַךְ דִּבְרֵיהֶם לֹא מָצְאוּ אֹזֶן קַשֶׁבֶת וְקֶסְטִיּוּס צִוָּה לְאַנְשֵׁי חֵילוֹ לִלְכֹּד אֶת הָעִיר בְּרַעַשׁ מִלְחָמָה. חָמֵש פְּעָמִים הֶעְפִּילוּ גִבּוֹרֵי רוֹמָא לַעֲלוֹת עַל הַחוֹמָה, אַךְ בְּכָל פַּעַם וּפַעַם הִפְלִיאוּ בָהֶם הַיְּהוּדִים אֶת מַכּוֹתֵיהֶם וַיַכּוּם אָחוֹר, עַד כִּי נוֹכַח קֶסְטִיּוּם, כִּי לֹא עַל נְקַלָּה יִלְכֹּד אֶת הָעִיר וְלֹא עַל נְקַלָּה יִגְבַּר עַל הַיְהוּדִים הַנִּלְחָמִים בְּחֵרֶף נֶפֶשׁ בְּעַד חֵרוּתָם. וּבִּרְאוֹתוֹ, כִּי הַמִּלְחָמָה תִמָּשֵׁךְ לְאֹרֶךְ יָמִים וִיְמֵי הַגְּשָׁמִים קְרוֹבִים לָבֹא, הֶחְלִיט לַעֲלוֹת מֵעַל הָעִיר וְלָשׁוּב אֵלֶיהָ בִּיְמֵי הָאָבִיב עִם חַיִל יוֹתֵר גָּדוֹל, יוֹתֵר חָזָק. אַךְ הַיְּהוּדִים רָדְפוּ אַחֲרָיו בִּגְבוּרָה נִמְרָצָה וַיִּתְנַפְּלוּ עַל חֵילוֹ מִכָּל צַד וַיַפִּילוּ בוֹ חֲלָלִים רַבִּים. בִּפְרַט חַזְקָה הַמִּלְחָמָה בַדֶּרֶךְ הַצָּרָה בֵין הֶהָרִים, אֲשֶׁר בָּהּ הָלְכוּ הָרוֹמָאִים. מֵעַל רָאשֵׁי הֶהָרִים הִמְטִירוּ הַיְּהוּדִים עֲלֵיהֶם חִצִּים וְאַבְנֵי מַשְׁחִית, וּכְפֶשַׂע הָיָה בֵינֵיהֶם וּבֵין כִּלָיוֹן חָרוּץ. לְמַעַן הַצִּיל אֶת חֵילוֹ, עָזַב קֶסְטִיּוּס גְּדוּד שֶׁל אַרְבַּע מֵאוֹת אִישׁ לְהִלָּחֵם בַּיְּהוּדִים. וּבְעֵת אֲשֶׁר הַיְּהוּדִים בְחֶשְׁכַּת הַלַּיְלָה חָשְׁבוּ, כִּי נִלְחָמִים הֵם נֶגֶד כָּל חֵיל הָרוֹמָאִים, הִשְׁתַּמֵּש קֶסְטִיּוּס בִּשְׁגִיאָתָם וַיְּמַהֵר לָנוּס, בְּחָפְזוֹ עָזַב אֶת כָּל הַכְּבוּדָה, אֶת כָּל הַנֶּשֶק וְגַם אֶת אוֹצַר הַמִּלְחָמָה עִם דִּגְלֵי הַלִּגְיוֹנוֹת. בְּאוֹר הַבֹּקֶר רָאוּ הַיְּהוּדִים אֶת שְׁגִיאָתָם וַיַּכְרִיעוּ לָטֶבֶחַ אֶת אַרְבַּע מֵאוֹת הָאִישׁ וַיְּמַהֲרוּ לִרְדֹּף אַחֲרֵי יֶתֶר הַחַיִל, אַךְ כְּבַר לֹא הִשִׂיגוּהוּ. עֲמוּסִים בְּשָׁלָל רָב שָׁבוּ הַמְּנַצְּחִים בְּשִׂמְחָה וְשָׂשׂוֹן ירוּשָלָיְמָה.

הַנִּצָחוֹן הַגָּדוֹל הַזֶּה הִלְהִיב אֶת לִבּוֹת כָּל הָעָם, אֲשֶׁר כְּבָר רָאָה, כִּי שָׁבוּ יְמֵי הַחַשְׁמוֹנָאִים כִּי עוֹד פַּעַם תִּהְיֶה יַד גִּבּוֹרֵי יְהוּדָה בְעֹרֶף אוֹיבֵיהֶם וְהָרוֹמָאִים, כְּמוֹ הַסּוּרִים, גָּרֵשׁ יְגֹרָשׁוּ מֵהָאָרֶץ. מִסְפַּר אוֹהֲבֵי הַשָּׁלוֹם נִמְעַט מְאֹד וְכָל הָעָם נָתַן אֶת יָדוֹ לַקַּנָּאִים הַנִּלְחָמִים אֶת מִלְחֶמֶת הַחֵרוּת. מִכָּל פִּנּוֹת אֶרֶץ יִשְׂרָאֵל וּמִכָּל הָאֲרָצוֹת, אֲשֶׁר שָׁם יָשְׁבוּ הַיְּהוּדִים נָהֲרוּ הֲמוֹנִים הֲמוֹנִים יְרוּשָׁלַיְמָה לְהִוָסֵף עַל חֵיל הַמִּלְחָמָה, לָקַחַת חֵלֶק בְּמִלְחֶמֶת הַמִּצְוָה נֶגֶד הָרֹוֹמָאִים וְיֶתֶר אֻמּוֹת הָעוֹלָם, אֲשֶׁר הָיוּ בִּבְרִית עִמָּם. בֵּינֵיהֶם הָיוּ רַבִּים מִן הָאִסִּיִּים, אֲשֶׁר בְּמֶשֶׁך זְמַן רַב עָמְדוּ מֵרָחוֹק לְחַיֵּי הַמְּדִינָה וְאֲשֶׁר שְׁפִיכַת דָּמִים גַּם בְּמִלְחָמָה הָיְתָה בְעֵינֵיהֶם לְתוֹעֵבָה גְדוֹלָה. גַּם הֵם הִתִּירוּ הַפַּעַם לְעַצְמָם לֶאֱחֹז בְּחֶרֶב וּלְהִלָּחֵם נֶגֶד הָרוֹמָאִים. גַם הַשֹּׁמְרוֹנִים שָׁכְחוּ בָּעֵת הַהִיא אֶת שִׂנְאָתָם לְעַם יְהוּדָה, וַיִּהְיוּ נְכוֹנִים לְהִלָּחֵם יַחַד אִתּוֹ נֶגֶד מַחְרִיבֵי אַרְצָם. בֵּין רָאשֵׁי הַגִּבּוֹרִים בִּיְרוּשָׁלַיִם הִצְטַיְנוּ יוֹתֵר מִכֻּלָם שִׁמְעוֹן בַּר גְּיוֹרָא, פַּחַד הָרוֹמָאִים, אֶלְעָזָר בֶּן חַנַּנְיָה, וְאֶלְעָזָר בֶּן שִׁמְעוֹן, גַּם בְּנֵי בֵית הַמְּלוּכָה בְהַדְיֵיבּ נֶכְדֵּי הִילֵנִי הַמַּלְכָּה, אֲשֶׁר הִתְיַהֲדָה יַחַד עִם בְּנָה אִיזַטִּיס הַמֶּלֶךְ, הָיוּ זְרֹעַ לְגִבּוֹרֵי יְהוּדָה בְמִלְחַמְתָּם בָּרוֹמָאִים עַד יוֹם יְרושָׁלַיִם הָאַחֲרוֹן. בִּזְמַן מְאֻחָר מְעַט אַחֲרֵי מִלְחֶמֶת הַגָּלִיל נוֹסַף עַל הַגִּבּוֹרִים הָאֵלֶּה גַם יוֹחָנָן מִגּוּשׁ-חָלָב.

וּבִירוּשָׁלַיִם שָׁבָה מֶמְשֶׁלֶת הָעָם עַל כַּנָּה. בְּראֹש הַמֶמְשָׁלָה עָמְדוּ הַסַּנְהֶדְרִין, אֲשֶׁר הָיוּ שׁוֹאֲלִים אֶת פִּי כָל הָעָם בְּכָל דָּבָר נִכְבָּד יוֹצֵא מִן הַכְּלָל. וְרֹאשׁ הַסַּנְהֶדְרִין בָּעֵת הַהִיא הָיָה רַבָּן שְׁמְעוֹן בֶּן רַבָּן גַּמְלִיאֵל הַנִּקְרָא הַזָּקֵן. לְהַרְאוֹת, כִּי חָדַל הָעָם לִהְיוֹת תַּחַת עוֹל הָרוֹמָאִים, הֵחֵלוּ לְהַטְבִּיעַ מַטְבֵּעוֹת, וַעֲלֵיהֶן כָּתוּב בְּעִבְרִית: "הַשָּׁנָה הָרִאשׁוֹנָה לְחֵרוּת יְרוּשָׁלַיִם ". אֲסֵפַת הָעָם הִפְקִידָה פְקִידִים עַל כָּל חֶלְקֵי הָאָרֶץ וַיְּמַלְּאוּ אֶת יְדֵיהֶם לְהָכִין הַכֹּל לַמִּלְחָמָה נֶגֶד הָרוֹמָאִים.


7

וְנִירוֹן קֵסָר בְּשָׁמְעוֹ אֶת הַשְמוּעוֹת, אֲשֶׁר בָּאוּ לוֹ מֵאֶרֶץ יְהוּדָה, נִבְהַל מְאֹד. הוּא פָחַד פְּן יָקוּמוּ יוֹשְׁבֵי יֶתֶר הַפְּרוֹבִינְצִיוֹת וְעָשׂוּ כְמַעֲשֵׂי הַיְּהוּדִים וְאֵשׁ הַמֶּרֶד תּאחַז בְּאַרְבַּע פִּנוֹת קֵסָרוּת רוֹמָא, בִּפְרַט בְּאַרְצות הַמִּזְרָח. גַּם הַפַּרְתִּים, שְׁכֵנֵי הָרוֹמָאִים בַּמִּזְרָח, הִתְעַתְּדוּ לָבֹא לִגְבוּל רוֹמָא וְיָכְלוּ לְהָבִיא בְסַכָּנָה גְדוֹלָה אֶת שִׁלְטוֹן הָרוֹמָאִים בְּמַעֲרַב אַסִּיָּה. לָכֵן הֶחְלִיט נִירוֹן לִשְׁלֹחַ אַרְצָה יְהוּדָה אֶת מִבְחַר גִּבּוֹרֵי רוֹמָא וְאֶת מִבְחַר שָׂרֵי צְבָאוֹ, פְלַוִּיוּס אַסְפָּסִיאַנּוּס, בְּיָדְעוֹ, כִּי רַק הוּא לְבַדּוֹ יוּכַל לְהַשְׁקִיט אֶת הַמֶּרֶד אֲשֶׁר פָּרַץ בִּיְהוּדָה.

אַסְפָּסִיאַנוּס בָּא לְעַכּוֹ וַיַּרְא, כִּי לֹא לְהַשְׁקִיט אֶת מֶרֶד הַיְּהוּדִים בָּא, כִּי אִם לִכְבֹּש אֶת הָאָרֶץ מֵחָדָשׁ. לְעַכּוֹ נִקְבְּצוּ וּבָאוּ כָּל מוֹשְׁלֵי סוּרִיָּה, אֲשֶׁר נִבְהֲלוּ מִפְּנֵי תְנוּפַת יַד הָרוֹמָאִים וַיְּמַהֲרוּ לָבֹא לְעֶזְרָתָם נֶגֶד הַיְּהוּדִים. בָּהֶם הָיָה מַלְכוֹס מֶלֶךְ עֲרָב עִם חֵיל נוֹשְׁקֵי רוֹמֵי קֶשֶׁת, אַנְטִיוֹכוֹס מֶלֶךְ קוֹמַגֶנָה, שׁוֹהַם מֶלֶךְ אֶמֶסָה וְאַגְרִפַּס הַשֵּׁנִי, אֲשֶׁר נָשָׂא אֶת שֵׁם מֶלֶךְ יְהוּדָה, עִם אֲחוֹתוֹ בֶרֶנִיקָה. גַּם בְּנוֹ טֵיטוּס בָּא מֵאַלֶכְּסַנְדְּרִיָה בְּרֹאשׁ שְׁנֵי לִגְיוֹנוֹת. תַּחַת פְּקוּדַת אַסְפָּסִיַאנּוּס נֶאֱסְפוּ יוֹתֵר מֵחֲמִשִׁים אֶלֶף אִישׁ מִלְבַד הַכְּבוּדָה הָרַבָּה.

לְמַעַן לֹא תִהְיֶה לוֹ הַמִּלְחָמָה מִפָּנִים וּמֵאָחוֹר הֶחְלִיט שַׂר צְבָא הָרוֹמָאִים לִכְבּשׁ אֶת אֶרֶץ הַגָּלִיל טֶרֶם יָשִׂים פָּנָיו יְרוּשָׁלָיְמָה. הַגָּלִיל הָיָה אֶרֶץ בְּרוּכָה, אֲשֶׁר לֹא חָסַר כֹּל בָּהּ וּמִסְפַּר יוֹשְׁבֶיהָ יוֹתֵר מִשְׁלשָׁה מִילְיוֹנִים נָפֶשׁ. כֻּלָּם הָיוּ מֵאוֹהַבֵי הַחֵרוּת וַיִּהְיוּ נְכוֹנִים לְהַקְרִיב בַּעֲדָהּ אֶת רְכוּשָׁם וְאֶת חַיֵּיהֶם. מֵהֶם יָצְאוּ הַקַּנָּאִים הָרִאשׁוֹנִים, אֲשֶׁר עוֹד בִּיְמֵי הוּרדוּס הִרְגִיזוּ תָּמִיד אֶת הָרוֹמָאִים. וְעַתָּה, כַּאֲשֶׁר הֵחֵלָה מִלְחֶמֶת הַחֵרוּת, הִתְלַהֲבוּ הֵמָה הָרִאשׁוֹנִים בְּעַד מִלְחֶמֶת הַמִּצְוָה הַזֹּאת. לוּ נִמְצָא בְּתוֹכָם אִישׁ אֲשֶׁר יָכֹל לַעֲמֹד בְּרֹאשׁ הָעָם בְּרוּחַ עָצה וּגְּבוּרָה וּלְנַהֵל אֶת הַתְּנוּעָה בְּדֶרֶך יְשָׁרָה, אָז לֹא עַל נְקַלָּה גָבְרוּ עֲלֵיהֶם הָרוֹמָאִים וְאוּלַי קִבְּלָה הַמִּלְחָמָה צוּרָה אַחֶרֶת. אַךְ הַפָּקִיד, אֲשֶׁר הָפְקַד עַל הָאָרֶץ הַזֹּאת מִטַּעַם הַסַּנְהֶדְרִין אֲשֶׁר בִּיְרוּשָׁלַיִם, יוֹסֵף בֶּן מַתַּתְיָהוּ, הָיָה מִן הַנּוֹטִים אַחֲרֵי הָרוֹמָאִים וּבוֹגֵד בְּעַמּוֹ וּבְמוֹלַדְתּוֹ. הוּא לֹא עָשָׂה מְאוּמָה לְהָכִין בַּגָּלִיל אֶת צָרֵכי הַמִּלְחָמָה, לְהָכִין חַיִל וּלְסַדּרוֹ לַמִּלְחָמָה. הוּא הֵפִיחַ מְדָנִים בְּכָל הָאָרֶץ וַיְּסַכְסֵךְ עִיר בְּעִיר וְסִיעָה בְּסִיעָה. וְתוֹצָאַת הַדָּבָר הָיָה, כִּי נִלְחֲמוּ הַיְּהוּדִים בִּגְבוּרָה נִמְרָצָה אַךְ בְּלִי סֵדֶר וּמִשְׁטָר כְּלָל. וְלָכֵן נָקֵל הָיָה לָרוֹמָאִים לְהַכּוֹת כָּל גְּדוּד וּגְדוּד לְבַדּוֹ וְלִלְכֹּד כָּל עִיר וָעִיר לְבַדָּהּ.


8

אָמְנָם, בִּמְקוֹמוֹת רַבִּים הִפְלִיאוּ הַיְּהוּדִים אֶת מַכּוֹתֵיהֶם בָּרוֹמָאִים וַיַּפִּילוּ מֵהֶם רָב. בִּפְרָט הִצְטַיְּנוּ יוֹשְבֵי הָעִיר תָּבוֹר, גַמַּלָא וְיוּדֶפֶת. הָעִיר תָּבוֹר הָיְתָה בְנוּיָה עַל הַר תָּבוֹר, הוּא הָהָר, הַמִּתְנַּשֵּׂא בְּתוֹךְ עֵמֶק יִזְרְעֶאל. שַׂר הַגְדוּד הָרוֹמָאִי, פְּלַקִּידַס, אֲשֶׁר נִשְׁלַח מֵאֵת אַסְפָּסִיאַנּוּס לִכְבֹּש אֶת הָעִיר, לֹא יָכֹל לְלָכְדָהּ בְּרָעָשׁ מִלְחָמָה; וְלָכֵן שָׂם אֶל הָעָרְמָה פָנָיו. הוּא עָשָׂה אֶת עַצְמוֹ כְנָס מִפְּנֵי [[23]] הַיְּהוּדִים. אַנְשֵׁי הַמִּלְחָמָה יָצְאוּ כֻּלָּם מִן הָעִיר לִרְדֹף אַחֲרָיו וַיֵּרְדּוּ מֵעַל הָהָר אֶל הָעֵמֶק. כַּאֲשֶׁר הָנְתְּקוּ הַרְבֵּה מֵעִירָם, שָׁבוּ הָרוֹמָאִים וַיָּסֹבּוּ עֲלֵיהֶם מִסָבִיב וְלֹא נָתְנוּ לָהֶם לָשׁוּב הָעִירָה. רַבִּים מֵהַיְּהוּדִים הִצִליחוּ לְהִמָּלֵט יְרוּשָׁלַיְמָה, אֲבָל הַנִשְׁאָרִים נָפְלוּ בַּמִּלְחָמָה וָהִעיר נָפְלָה בִיְדֵי הָרוֹמָאִים.

יוֹתֵר כָּבֵד הָיָה לָרוֹמָאִים כִּבּוּשׁ גַּמַּלָא, אֲשֶׁר מִמִּזְרָח לְיַם כִּנֶּרֶת. הָעִיר הָיָתה בְנוּיָה עַל רֹאשׁ הַר גָּבֹהַּ הַדּוֹמֶה לְדַבֶּשֶׁת גָּמָל וְלָכֵן נִקְרְאָה בָּשֵׁם הַזֶּה. עֲבוֹדָה רַבָּה הֶעֱבִידוּ גִבּוֹרֵי גַמַּלָּא אֶת חֵיל הָרוֹמָאִים. חֲלָלִים רַבִּים נָפְלוּ מִתּוֹכוֹ עַד אֲשֶׁר הִצְלִיחַ לִלְכֹּד אֶת הָעִיר. כָּל יוֹשְׁבֶיהָ נָפְלוּ בְחֶרֶב הָרוֹמָאִים, כִּי אִישׁ מֵהֶם לֹא חָפֵץ לְבַקֵּשׁ חֲנִינָה מֵאֵת מַחֲרִיבֵי אַרָצם וַיִּבְחֲרוּ מָוֶת מֵחַיֵּי עַבְדוּת וְקָלוֹן.

יוֹתֵר מִכָּל הֶעָרִים הִצְטַיְּנָה יוּדֶפֶת בְּמִלְחַמְתָּהּ נֶגֶד הָרוֹמָאִים. אַרְבָּעִים יוֹם צָר כָל חֵיל אַסְפָּסִיאַנּוּס עַל הָעִיר הַזאת. כָּל תַּחְבּוּלוֹת הָרוֹמָאִים לָקַחַת אֶת הָעִיר הַזֹּאת בְּמִלְחָמָה לֹא הִצְלִיחוּ, כָּל גְבוּרָתָם נָשָׁתָה. הָיוּ יָמִים, אֲשֶׁר גִּבּוֹרֵי הַיְּהוּדִים הָיוּ יוֹצְאִים מִתּוֹךְ הָעִיר הוֹרסִים אֶת כָּל סוֹלְלוֹת הָאוֹיֵב וְשׂורְפִים אֶת כָּל מְכוֹנוֹת הַמָּצוֹר. פְּעָמִים רַבּוֹת הָיְתָה מַחֲנֶה הָאוֹיֵב בַּמָּצוֹר וְהַיְּהוּדִים צָרִים עָלֶיהָ. כַּאֲשֶׁר הֶעְפִּילוּ פַּעַם צִבְאוֹת הָרוֹמָאִים לַעֲלוֹת עַל חוֹמַת הָעִיר, שָׁפְכוּ הַיְּהוּדִים עֲלֵיהֶם מֵעַל הַחוֹמָה שֶׁמֶן רוֹתֵח וַיַּפִּילוּ מֵהֶם עַם רַב וַיְּגָרְשׁוּם. אֲבָל בְּתוֹךְ הָעִיר הָיָה חָזָק מְאֹד בַּעַל בְּרִית הָרוֹמָאִים: הֲלֹא הוּא הָרָעָב וְהַצָמָא. כָּל מְבוֹאוֹת הָעִיר הָיוּ בִיְדֵי הָרוֹמָאִים וְהֵם לֹא נָתְנוּ לְהָבִיא הָעִירָה לֹא לֶחֶם וְלֹא מַיִם, וַיִּתַּם לָרִיק כֹּחַ אַנְשֵׁי הַמִּלְחָמָה כִּי יוֹתֵר מֵאֲשֶׁר נָפְלוּ בְחֶרֶב הָרוֹמָאִים, מֵתוּ בָרָעָב וּבַצָּמָא. לְאַחֲרוֹנָה נִמְצָא גַם בּוֹגֵד אֶחָד (אוּלַי אַחַד מֵאֵלֶּה, אֲשֶׁר לֹא יָכְלוּ לִסְבֹּל אֶת מְצוּקַת הָרָעָב אוֹ אַחַד מִיְּדִידֵי הָרוֹמָאִים וְעַבְדֵי יוֹסֵף…), אֲשֶׁר הֶרְאָה לָאוֹיֵב מָקוֹם עַל חוֹמַת הָעִיר, אֲשֶׁר הַנְּצוּרִים לֹא שְׁמָרוּהוּ כָּרָאוּי. הָרוֹמָאִים הִשְׁתַּמְּשׁוּ בְחֶשְׁכַּת הַלַּיְלָה, בְּעֵת אֲשֶׁר כָּל מֵגִינֵי הָעִיר, עֲיֵפִים מִמִּלְחֶמֶת הַיּוֹם, נָמוּ שְׁנָתָם וַיָּבֹאוּ הָעִירָה. בִּתְשׁוּקָתָם הָעַזָּה לִנְקֹם אֶת נִקְמַת אַנְשֵׁי חֵילָם, אֲשֶׁר נָּפְלוּ בַּמִּלְחָמָה הַזֹּאת, עָשׂוּ הָרוֹמָאִים טֶבַח נוֹרָא בָּעִיר וַיַּגִּירוּ לְפִי חֶרֶב יוֹתֵר מֵאַרְבָּעִים אֶלֶף נָפֶשׁ. יוֹסֵף בֵּן מַתַּתְיָהוּ, פְּקִיד הַגָלִיל, הָלַךְ בַּשֶׁבִי לִפְנֵי הָרוֹמָאִים. אַסְפָּסִיאַנּוּס נָטָה לוֹ חֶסֶד, כִּי יָדַע, כִּי אוֹהֵב הָרוֹמָאִים הוּא וְאוּלַי הוֹעִיל הוּא לְנִצְחוֹנוֹ יוֹתֵר מִגְבוּרַת לִגְיוֹנוֹתָיו. מִן הָעֵת הַהִיא הָיָה יוֹסֵף, אֲשֶׁר קִבֵּל אֶת שֵׁם מִשְׁפַּחַת אַסְפָּסִיאַנּוּס וַיִּקָרֵא יוֹסֵף פְלַוִּיוּס, לְמוֹרֵה דֶרֶךְ לָרוֹמָאִים בְּמִלְחַמְתָּם בִּיְהוּדָה וּלְיוֹעֵץ לְאַסְפָּסִיאַנּוּס בְּכָל עִנְיָנֵי הַמִּלְחָמָה.

בִּנְפֹל הֶעָרִים הָאֵלֶה וּבִנְפֹל גּוּשׁ-חָלָב בִּקְצֵה הַגָּלִיל הַצְּפוֹנִי נָפְלָה כָל אֶרֶץ הַגָּלִיל בִּיְדֵי הָרוֹמָאִים. כָּל הָאָרֶץ הַיָּפָה וְהַפּוֹרִיָה הָיְתָה לִשְמָמָה, הֶעָרִים נֶחְרְבוּ וְתַחַת חָרְבוֹתֵיהֶם נִקְבְּרוּ רִבְבוֹת אַלְפֵי יְהוּדִים, חַלְלֵי חֶרֶב אוֹ מֵתֵי רָעָב. לֹא הָיְתָה עִיר אֲשֶׁר נִשְׁאֲרָה עַל תִּלָהּ זוּלָתִי הָעִיר צִפּוֹרִי, אֲשֶׁר בָּהּ הָיְתָה עַל הָעֶלְיוֹנָה יַד יְדִידֵי הָרוֹמָאִים, כִּי עוֹד קֶסְטִיוּס גַלּוּס לָקַח רַבִּים מִבְּנֵי הַמִּשְׁפָּחוֹת הַנִּכְבָּדוֹת וַיִּשְׁלְחֵם בְּתוּר בְּנֵי תַעֲרוּבוֹת לְעִיר דּוֹר (הִיא טַנְטוּרָה הַנּוֹכְחִית). גַּם הַשֹּׁמְרוֹנִים, אֲשֶׁר הִתְבַּצְּרוּ עַל הַר-גְּרִיזִּים לֹא הִצְלִיחוּ, כִּי עַל רֹאשׁ הָהָר חָסְרָה לָהֶם צֵידָה, וְגַם מֵי-שְׁתִיָּה לֹא הָיוּ לָהֶם וְהָרוֹמָאִים סָבִיב שָׁתוּ עַל הָהָר וְלֹא נָתְנוּ לְהָבִיא שָׁמָה מָזוֹן וּמִחְיֶה. שְׁנֵי יָמִים אַחֲרֵי חָרְבַּן יוּדֶפֶת הֵגִירוּ הָרוֹמָאִים לְפִי חֶרֶב אֶת כָּל הָעָם, אֲשֶׁר מָצְאוּ עַל הַר-גְּרִיזִּים.


9

דֶּרֶךְ עֵמֶק הַשָּׁרוֹן הִגִּיעוּ צִבאוֹת הָרוֹמָאִים עַד הָעִיר יָפוֹ. הָעִיר הַזֹּאת נֶחְרְבָה עוֹד מֵאֵת קֶסְטִיוּס גַלּוּס, אֲבָל הַקַּנָּאִים שָׁבוּ וַיִּבְנוּהָ וַיִּתְבַּצְּרוּ בָהּ. כַּאֲשֶׁר נִלְכְּדָה הָעִיר מֵאֵת הָאוֹיֵב, חָשׁוּ יוֹשְׁבֶיהָ מִפָלט לָהֶם עַל הָאֳנִיוֹת, אַךְ לַאֲסוֹנָם הֵחֵלָה עַל הַיָּם סְעָרָה חֲזָקָה, אֲשֶׁר שָׁבְרָה אֶת אֳנִיוֹתֵיהֶם וַתַּשְׁלִיכֵן אֶל הַיַּבָּשָׁה. וְשָׁם עָמְדוּ הָרוֹמִאים וַיָּמִיתוּ אֶת כָּל אֲשֶׁר נָפַל בְּיָדָם. רַבִּים טָבְעוּ בַיָּם וְרַבִּים מֵתוּ בְחֶרֶב הָרוֹמָאִים, אֲשֶׁר שָׂרְפוּ אֶת הָעִיר בָּאֵשׁ. אַחֲרֵי יָפוֹ נָפְלָה גַם יַבְנֶה בִיְדֵי הָרוֹמָאִים.

אַסְפָּסִיאַנּוּס נָתַן לְחֵילוֹ לָנוּחַ בְּעֵת הַחֹרֶף וּלְהַחְלִיף כֹּחַ מִן הַמַּכּוֹת, אֲשֶׁר הֻכָּה גַם הוּא בַּמִּלְחָמָה הַנּוֹרָאָה בְמֶשֶׁךְ כָּל הַקַּיִץ. רַבִּים מִפְּלִיטֵי יְרוּשָׁלַיִם מִיְּדִידֵי הָרוֹמָאִים הֵאִיצוּ בוֹ, שֶׁיָבֹא מַהֵר לִלְכֹּד אֶת הָעִיר הַזֹּאת, אֲבָל הוּא שָׁמַע כִּי בָהּ כְּבָר הֵחֵלוּ מִלְחָמוֹת בֵּין הַמְּתוּנִים וְהַקַּנָּאִים וְגַם בֵּין רָאשֵׁי הַקַנָּאִים בְּעַצְמָם. לָכֵן הֶחְלִיט לְחַכּוֹת עַד אֲשֶׁר יִתַּם כֹּחַ גִבּוֹרֵי יְרוּשָׁלַיִם בַּמִּלְחָמָה בֵינֵיהֶם וָאז יוּכַל בְּאֶפֶס יַד לִלְכֹּד אֶת הָעִיר. וַיָּבִיא אֶת הַמִלְחָמָה לְעֵבֶר הַיַּרְדֵּן מִזְרָחָה. בְּחֶרֶב נוֹקֶמֶת עָבְרוּ צִבְאוֹת רוֹמָא בְכָל הָאָרֶץ וַיַּהֲפְכוּהָ לְמִדְבַּר שְׁמָמָה וַחֲלָלִים לָאֲלָפִים נָשְׂאוּ גַלֵּי הַיַּרְדֵּן אֶל תּוֹךְ יַם הַמֶּלַח. מִכָּל עָרֵי עֵבֶר הַיַּרְדֵּן הַמִּזְרָחִי נִשְׁאַר רַק הַמִּבְצָר מָכֵרוּס בְּמִזְרַח יַם הַמֶּלַח עַל תִּלוֹ, כִּי אוֹתוֹ לֹא יָכְלוּ הָרוֹמָאִים לָקַחַת בַּמִּלְחָמָה.

וּבְעֵת אֲשֶׁר הָיְתָה הַמִּלְחָמָה בְעֵבֶר הַיַּרְדֵּן מִזְרָחָה, נִלְחַם חֵלֶק אֶחָד מִצְּבָא הָרוֹמָאִים בִּדְרוֹם הָאָרֶץ, בְּאֶרֶץ אֱדוֹם, וְאַחֲרֵי מִלְחָמָה קָשָׁה הֵשַׁם גַּם אֶת הָאָרֶץ הַהִיא וַיַּחְרִיב אֶת כָּל עָרֶיהָ מִלְּבַד הַמִּבְצָרִים הֵרוֹדִיוּם ומסדה. בָּאוֹפָן הַזֶּה נָפְלָה בִיְדֵי הָרוֹמָאִים כָּל אֶרֶץ יְהוּדָה וַתִּשָׁאֵר רַק יְרוּשָׁלַיִם וְעִמָּהּ הַמִּבְצָרִים מָכֵרוּס, מַסַּדָּה וְהֵרוֹדִיוּם.

לַעֲלוֹת עַל יְרוּשָׁלַיִם בְּעַצְמָהּ עוֹד לֹא מָצָא אַסְפָּסִיאַנוּס אֶת הַשָּׁעָה מְכֻוֶּנֶת מִפְּנֵי שְׁתֵּי סִבּוֹת. רֵאשִׁית, יָדַע הוּא אֶת הַנַּעֲשֶׂה בָּעִיר הַזֹּאת, אֶת הַמִּלְחָמוֹת הַחֲזָקוֹת בֵּין הַסִּיעוֹת וַיָּבֶן, כִּי יוֹתֵר מִמַּה שֶׁתֹּאכַל בַיְּהוּדִים חֶרֶב הָרוֹמָאִים, תֹּאכַל בָּהֶם חֶרֶב אִישׁ בְּרֵעֵהוּ, יוֹתֵר מִמַּה שֶׁיַזִּיקוּ לָעִיר וּלְיוֹשְׁבֶיהָ הָרוֹמָאִים, יַזִּיקוּ לָהֶם הַיְּהוּדִים בְּעַצְמָם. הוּא הֵבִין, כִּי אַךְ יִקְרַב עִם חֵילוֹ אֶל חוֹמַת הָעִיר, יַשְׁלִימוּ בֵינֵיהֶם רָאשֵׁי הַסִּיעוֹת וְיֵצְאוּ כֻלָּם כְּאִישׁ אֶחָד לִקְרָאתוֹ, וְלָכֵן הֶחְלִיט לַעֲמֹד מֵרָחוֹק לְמַעַן לֹא יַכִּירוּ בְנֵי יְרוּשָׁלַיִם אֶת הַסַּכָּנָה הַגְּדוֹלָה הַמְּרַחֶפֶת עַל רֹאשָׁם וְיוֹסִיפוּ לְהַשְׁחִית אֶת הָעִיר בִּמְרִיבוֹתֵיהֶם וּמְדָנֵיהֶם. שֵׁנִית, הִגִּיעוּ לוֹ שְׁמוּעוֹת מֵרוֹמָא, כִּי הוּמָת נִירוֹן וְכִי קֵסָרִים חֲדָשִׁים מִתְנַשְּׂאִים לָשֶׁבֶת עַל כִּסֵּא הַקֵּסָרוּת בְּעֶזְרַת הַצְּבָאוֹת, אֲשֶׁר תַּחַת פְּקוּדָתָם. וַיִּוָּלֵד הָרַעֲיוֹן בְּלִבּוֹ לְנַסּוֹת גַּם הוּא אֶת מַזָּלוֹ וּלְהִתְנַשֵֹּא לְקֵסָר בְּעֶזְרַת הַלִּגְיוֹנוֹת אֲשֶׁר הָיָה מַצְבִּיאָם. לָכֵן הִסִּיחַ אֶת דַעֲתּוֹ מֵעִנְיָנֵי הַמִּלְחָמָה בִיְהוּדָה, לְמַעַן יוּכַל לְהַקְדִּישׁ אֶת כָּל כֹּחוֹ לַמַּטָּרָה, אֲשֶׁר הִצִּיג לוֹ. וּבָאוֹפָן הַזֶּה שָבְתָה הַמִּלְחָמָה בִיהוּדָה בְמֶשֶׁךְ שְׁתֵּי שָׁנִים.


10

וְחֵפֶץ אַסְפָּסִיאַנּוּס הִצְלִיחַ בְּיָדוֹ. בְּהִשְׁתַּדְלוֹת לִיקִינִיוּס מוּקִיאַנוּס, נְצִיב סוּרִיָּה וְטִבֶּרִיוּס אַלֶכְּסַנְדֶּר (יְהוּדִי אֲשֵׁר הֵמִיר אֶת דָתוֹ בְדַת עוֹבֵדי אֱלִילִים. הוּא הָיָה בֵן אֲחִי הַפִּילוֹסוֹף הַיְּהוּדִי פִילוֹן, הַנִּקְרָא יְדִידְיָה הָאַלֶכְּסַנְדְּרוֹנִי), נְצִיב מִצְרַיִם, הִסְכִּימוּ כָּל הַלִּגְיוֹנוֹת, אֲשֶׁר עָמְדוּ בְאֶרֶץ הַמִּזְרָח לְהַכִּירוֹ לְקֵסָר, וַיָשָׁב לְרוֹמָא לָשִׂים אֶת כֶּתֶר הַקֵּסָרִים עַל רֹאשׁוֹ וְאֶת כִּבּוּשׁ יְרוּשָׁלַיִם מָסַר לְטִיטוּס בְּנוֹ.

בְּרֹאשׁ חַיִל גָּדוֹל שֶׁל שְׁמוֹנִים אֶלֶף אִישׁ עָלָה טִיטוּס בְּחֹדֶשׁ נִיסָן עַל יְרוּשָׁלַיִם וַיִּחַן עַל רֹאשׁ הַמִּצְפָּה נֹכַח הָעִיר, בָּרִאשׁוֹנָה חָשַׁב כִּי מַרְאֵה חֵילוֹ הַגָּדוֹל יַטִּיל אֵימָה בְלֵב הַיְּהוּדִים וְיִּכָּנְעוּ לְפָנָיו. אַךְ הַיְּהוּדִים יָצְאוּ מִתּוֹךְ הָעִיר וַיִּתְנַפְּלוּ עַל חֵילוֹ בְחֵמָה שְׁפוּכָה וַיְּאַלְּצוּהוּ לָסֹג אָחוֹר. גַּם טִיטוּס בְּעַצְמוֹ הָיָה בְסַכָּנָה גְדוֹלָה וּכְפֶּשַׁע הָיָה בֵינוֹ וּבֵין הַמָּוֶת. אָז רָאָה כִּי לֹא יַצְלִיחַ לִלְכֹּד אֶת הָעִיר בְּרַעַשׁ מִלְחָמָה, כִּי שְׁתֵּי שָׁנִים שֶׁל מִלְחָמוֹת בֵּינֵיהֶם לֹא הֶחְלִישׁוּ אֶת כֹּחַ הַיְּהוּדִים וְלֹא שָׁלְלוּ מֵהֶם אֶת הַבִּטָּחוֹן לְהַכּוֹת אֶת הָרוֹמָאִים וּלְגָרְשֵׁם מִן הָאָרֶץ, וַיָחֵל אֶת הַמָּצוֹר הַגָּדוֹל, אֲשֶׁר אֵין כִּמְעַט דּוּגְמָתוֹ בְכָל דִּבְרֵי יְמֵי הָעוֹלָם. הָרוֹמָאִים הִתְבַּצְּרוּ בְתוֹךְ מַחְנֵיהֶם וַיָחֵלוּ לִשְׁפֹּךְ סוֹלְלָה, לִבְנוֹת דָּיֵק סְבִיב הָעִיר, לְמַעַן הַקִּיף אוֹתָה מִכָּל צַד לְבִלְתִּי לָתֵת לָהּ יוֹצֵא וָשָׁב. עַל הַסּוֹלְלוֹת הֶעֱמִידוּ אֵילֵי בַרְזֶל וְכָל מְכוֹנוֹת הַמָּצוֹר לְנַגַח אֶת חוֹמַת הָעִיר וּלְהַמְטִיר עָלֶיהָ חִצִּים וְאַבְנֵי מַשְׁחִית. אֲבָל לֹא עַל נְקַלָּה יָכְלוּ לַעֲשׂוֹת אֶת כָּל הַדְּבָרִים הָאֵלֶּה. פְּעָמִים רַבּוֹת הָיוּ גִבּוֹרֵי יְהוּדָה יוֹצְאִים מִתּוֹךְ הָעִיר, שׂוֹרְפִים אֶת הַמְּכוֹנוֹת וְאֶת הַמִּגְדָּלִים וּמַכִּים אָחוֹר אֶת חֵיל הָאוֹיֵב. לֹא אַחַת וְלֹא שְׁתַּיִם הָיוּ הָרוֹמָאִים כְּמִתְיָאֲשִׁים בִּרְאוֹתָם אֶת גְּבוּרַת הַיְּהוּדִים וְהָיָה צָרִיךְ, כִּי טִיטוּס וְיֶתֶר שָׂרֵי צְבָאוֹ יַזְכִּירוּ לְאַנְשֵׁיהֶם אֶת גְּדוּלַת הָרוֹמָאִים וְאֶת כָּל הַמִּלְחָמוֹת הַגְּדוֹלוֹת, אֲשֶׁר עָשׂוּ הֵם וְאֲבוֹתֵיהֶם, לְהָשִׁיב לָהֶם אֶת אֹמֶץ רוּחָם, לְמַעַן יוֹסִיפוּ לְהִלָּחֵם בַּיְּהוּדִים.

אַרְבָּעָה חֳדָשִׁים אָרַךְ הַמָּצוֹר, אַרְבָּעָה חֳדָשִׁים לֹא חָדְלָה הַמִּלְחָמָה עַל חוֹמַת הָעִיר, וּמִי יוֹדֵעַ כַּמָּה יָכְלָה עוֹד לְהִמָּשֵׁךְ לוּלֵא הָרָעָב הֶחָזָק, אֲשֶׁר שָׁלַט בָּעִיר בְּכָל תָּקְפוֹ. הַמָּצוֹר הֵחֵל בְּחדֶשׁ נִיסָן, בְּעֵת אֲשֶׁר הָמוֹן רָב נֶאֱסַף יְרוּשָׁלַיְמָה לָחֹג אֶת חַג הַפֶּסַח וְכָל הֶהָמוֹן הַזֶּה לֹא יָכֹל לָצֵאת מִן הָעִיר וַיֹּאכַל אֶת כָּל הַלֶּחֶם, אֲשֶׁר הָיָה מוּכָן בָּהּ. מִלְּבַד זֶה נִשְׂרְפוּ אוֹצְרוֹת בַּר רַבִּים בְּעֵת הַמְּהוּמוֹת וְהַמִּלְחָמוֹת בֵּין הַסִּיעוֹת. וַיֶּרֶב מִסְפַּר הַמֵּתִים בָּרָעָב עַד מְאֹד. רַבִּים, אֲשֶׁר לֹא יָכְלוּ לִסְבּוֹל אֶת מְצוּקוֹת הָרָעָב, הִתְגַּנְּבוּ מִתּוֹךְ הָעִיר לִנְפֹּל אֶל הָרוֹמָאִים, אַךְ הֵם הָיוּ מֵמִיתִים אוֹתָם בְּאַכְזָרִיוּת נוֹרָאָה. יַחַד עִם הָרָעָב בָּאָה הָעִירָה גַם הַמַּגֵּפָה, אֲשֶׁר הִפִּילָה גַם הִיא חֲלָלִים רַבִּים עַד כִּי פִגְרֵי אָדָם הִתְגּוֹלְלוּ בַחוּצוֹת וְאֵין קוֹבֵר.

הָרָעָב וְהַמַּגֵּפָה וְגַם הַמְּדָנִים, אֲשֶׁר עוֹד לֹא חָדְלוּ, הֶחְלִישׁוּ מְאֹד אֶת כֹּחַ אַנְשֵׁי הַמִּלְחָמָה וְסוֹף סוֹף הִצְלִיחוּ הַיְּהוּדִים קוֹרְאִים לָהּ עַתָּה “מְצוּדַת דָּוִד”, וְחוֹקְרֵי דִבְרֵי הַיָּמִים חוֹשְׁבִים כִּי הִיא מְצוּדַת “אַנְטוֹנִיָּה”. טִיטוּס לֹא הֶחֱרִיב אוֹתָה לְמַעַן לְהַרְאוֹת לַדּוֹרוֹת הַבָּאִים אַחֲרָיו מַה-בְּצוּרָה הָיְתָה יְרוּשָׁלַיִם וּמַה-קָּשָׁה הָיְתָה הַמִּלְחָמָה עָלֶיהָ.

הָרוֹמָאִים לִפְרֹץ פֶּרֶץ בְּחוֹמַת הָעִיר. עוֹד פַּעַם חָגְרוּ הַיְּהוּדִים אֶת שְׁאֵרִית כֹּחָם וַיֵּצְאוּ מִן הָעִיר לְהִתְנַפֵּל עַל הָאוֹיֵב, אֲבָל הַפַּעַם הֻכּוּ אָחוֹר, וְהָרוֹמָאִים בְּרָדְפָם אַחֲרֵיהֶם בָּאוּ הָעִירָה, וַתִּתְלַקַּח מִלְחָמָה עֲצוּמָה בַחֲצַר בֵּית הַמִּקְדָּשׁ. לַמְּרוֹת פְּקוּדַת טִיטוּס לְבִלְתִּי לָגַעַת בְּבֵית הַמִּקְדָּש, הִשְׁלִיךְ אֶחָד מֵאַנְשֵׁי צְבָא רוֹמָא לַפִּיד אֵשׁ אֶל תּוֹךְ הַבַּיִת דֶרֶךְ “חַלּוֹן הַזָּהָב” וְהָאֵשׁ אָחֲזָה בְאַרְבַּע פִּנּוֹתָיו. כַּאֲשֶׁר רָאוּ הַיְּהוּדִים אֶת מִקְדָּשָׁם בּוֹעֵר בָּאֵשׁ, מָלְאוּ מָתְנֵיהֶם חַלְחָלָה וַתִּרְפֶּינָה יְדֵיהֶם. חָדְלוּ גִבּוֹרֵי יְהוּדָה לְהִלָּחֵם, נָשְׁתָה גְבוּרָתָם וְהָרוֹמָאִים שׁוֹאֲפֵי נְקָמָה וָרֶצַח עָשׂוּ בָהֶם טֶבַח נוֹרָא. לֹא חָמְלוּ עַל נָשִׁים וְעַל עוֹלָלִים וּפְנֵי זְקֵנִים לֹא נָשָׂאוּ וַיַּגִּירוּ לְפִי חֶרֶב אֶת כָּל אֲשֶׁר מָצָאוּ. רַבִּים הִשְׁלִיכוּ אֶת עַצְמָם אֶל תּוֹךְ לַהֲבַת בֵּית הַמִּקְדָּשׁ. יְרוּשָׁלַיִם עִם כָּל מַחֲמַדֶּיהָ הָיְתָה לִמְעִי מַפָּלָה וְתַחַת חָרְבוֹתֶיהָ נִקְבְּרוּ מֵגִינֶיהָ הָאַחֲרוֹנִים יַחַד עִם חֵרוּת יִשָׂרֵאל וְחָסְנוֹ הַלְּאֻמִּי.

אַחֲרֵי שְׂרֵפַת בֵּית הַמִּקְדָּשׁ וְאַחֲרֵי הַהֶרֶג הָרַב, אֲשֶׁר עָשׂוּ הָרוֹמָאִים בְּלָכְדָם אֶת יְרוּשָׁלַיִם, נִגְמְרָה הַמִּלְחָמָה. טִיטוּס יָצָא עִם חֵילוֹ מִמְּקוֹם הַמַהְפֵּכָה וַיֵּלֶךְ רוֹמָאָה לָחֹג יַחַד עִם אָבִיו בְּרֹב פְּאֵר וְהָדָר אֶת נִצְחוֹנוֹ הַגָּדוֹל. בְּכָל זֹאת הוֹסִיפוּ עוֹד גְּדוּדֵי קַנָּאִים אֲחָדִים לְהִלָּחֵם בְּמֶשֶׁךְ שָׁלשׁ שָׁנִים נֶגֶד הָרוֹמָאִים מַחֲרִיבֵי אַרְצָם, אַף כִּי הָיָה בָּרוּר כִּי אֵין בְּכֹחָם לְהָשִׁיב לְעַמָּם אֶת חֵרוּתוֹ וְאֶת מַצָּבוֹ הַקוֹדֶם. בִּיְרוּשָׁלַיִם בְּעַצְמָהּ בָּעִיר הָעֶלְיוֹנָה עָמְדוּ רַבִּים מֵהַקַּנָּאִים עַל נַפְשָׁם כְּחֹדֶש יָמִים וְרַק בַּשְּׁמִינִי בֶּאֱלוּל נָפַל חֵלֶק הָעִיר הַזֶּה בִיְדֵי הָרוֹמָאִים וְשִׁמְעוֹן בַּר גִיוֹרָא הָלַךְ בַּשֶּׁבִי לִפְנֵיהֶם. רַבִּים חָשְׁבוּ לְהִנָּצֵל בְּבָרְחָם אֶל עֵבֶר הַיַּרְדֵּן מִזְרָחָה, אַךְ חֵיל הָרוֹמָאִים תַּחַת פְּקוּדַת בַּסּוּס רָדַף אַחֲרֵיהֶם וּבַמִּלְחָמָה אֲשֶׁר הָיְתָה בַּיַּעַר בְּקִרְבַת הַיַּרְדֵּן נָפְלוּ כֻּלָּם חֲלָלִים. בַּסּוּס עָבַר אֶת הַיַּרְדֵּן וַיָצָר עַל מָכֵרוּס. הַמִּבְצָר הַזֶּה הָיָה בָּנוּי עַל רֹאשׁ סֶלַע וְהָרוֹמָאִים לֹא יָכְלוּ לְלָכְדוֹ, וְנַהֲפֹךְ הוּא, כִּי הַיְהוּדִים, אֲשֶׁר הִתְבַּצְּרוּ בוֹ, הָיוּ יוֹצְאִים כְּפַעַם בְּפַעַם וּּמַכִּים בְּאוֹיבֵיהֶם. בֵּינֵיהֶם הִצְטַיֵּן יוֹתֵר מִכֻּלָּם אִישׁ גִּבּוֹר חַיִל וּשְׁמוֹ אֶלְעָזָר וַיְּהִי לִמְפַקֵּד כָּל חֵיל הַמִּבְצָר. אֲבָל פַּעַם אַחַת לֹא נִזְהַר אֶלְעָזָר וַיִּפֹּל בַּשֶׁבִי בִיְדֵי הָרוֹמָאִים. בַּסּוּס צִוָּה לַעֲנוֹתוֹ בְּעִנּוּיִם קָשִׁים לְעֵינֵי כָּל אַנְשֵׁי הַמִּבְצָר, אֲשֶׁר עָמְדוּ עַל הַחוֹמָה. וְלֹא יָכְלוּ הַיְּהוּדִים לִרְאוֹת בְּצָרַת אָחִיהֶם וַיַּבְטִיחוּ לְבַסּוּס לִמְסֹר לוֹ אֶת הַמִּבְצָר, אִם רַק יִתֵּן לָהֶם לָצֵאת בְּשָׁלוֹם. בַּסּוּס נֵאוֹת לָהֶם וַיִּתֵּן לָצֵאת לְאַנְשֵׁי הַמִּלְחָמָה. אַךְ אֶת יֶתֶר הָעָם הִגִּיר לְפִי חֶרֶב וְאֶת הַמִּבְצָר הָרָס.

הַמִּבְצָר מַסַּדָּה נָפַל בָּאַחֲרוֹנָה. אַנְשֵׁי הַמִּבְצָר הַזֶּה בִרְאוֹתָם, כִּי כְבָר עָשׂוּ הָרוֹמָאִים פֶּרֶץ בְּתוֹךְ הַחוֹמָה וְכִי סוֹף סוֹף יִּפֹּל הַמִּבְצָר בִּידֵיהֶם, הֵמִיתוּ אֶת נְשֵׁיהֶם וִיְלָדֵיהֶם וְאַחֲרֵי כֵן הֵמִיתוּ אִישׁ אֶת רֵעֵהוּ, בַּעֲצַת אֶלְעָזָר בֶּן יָאִיר אֶחָד מֵרָאשֵׁי סִיעַת הַקַּנָּאִים. וְלֹא מָצְאוּ הָרוֹמָאִים בְּתוֹךְ הַמִּבְצָר בִּלְתִּי אֶת גּוֹפוֹת הַחֲלָלִים. רַק שְׁתֵּי נָשִׁים וְחַמִשָּׁה יְלָדִים נִשְׁאֲרוּ בַחַיִּים כִּי הִתְחַבְּאוּ, וַיְסַפְּרוּ בְּאָזְנֵי הָרוֹמָאִים אֶת כָּל אֲשֶׁר קָרָה לַגְּדוּד הַקָּטֹן אֲשֶׁר הָיָה בַמִּבְצָר, וְאֶת דִּבְרֵי יָמָיו הָאַחֲרוֹנִים.

הַמִּבְצָר הֵרוֹדִיוּם נָפַל עוֹד לִפְנֵי זֶה בִיְדֵי הָרוֹמָאִים בְּלִי מִלְחָמָה.

וְקַנָּאֵי הַחֵרוּת, אֲשֶׁר תְּבוּסָתָם בְּאֶרֶץ יְהוּדָה הָיְתָה שְׁלֵמָה, עוֹד מָצְאוּ מָקוֹם לְהָרֵעַ לָרוֹמָאִים. רַבִּים מֵהֶם נִמְלְטוּ לְאֶרֶץ מִצְרַיִם, לְקִירֵינִי (מִמַּעֲרָב לְמִצְרַיִם עַל שְׂפַת הַיָּם הַגָּדוֹל), לְאֶרֶץ לוּב וַיָּסִיתוּ אֶת הַיְּהוּדִים הָרַבִּים, אֲשֶׁר יָשְׁבוּ שָׁם, לְהָרִים אֶת נֵס הַמֶּרֶד נֶגֶד מֶמְשֶׁלֶת רוֹמָא. עוֹד פַּעַם נִשְׁפְּכוּ נַהֲרוֹת נַחְלֵי דָם כַּמַּיִם וַתִּרְוֶינָה גַם הָאֲרָצוֹת הָהֵן מִדַּם יִשְׂרָאֵל. בָּעֵת הַהִיא צִוָּה אַסְפָּסִיאַנּוּס לִסְגֹּר אֶת מִקְדַּשׁ חוֹנְיוֹ אֲשֶׁר בְּמִצְרַיִם לְמַעַן לֹא יִהְיֶה הוּא לְמֶרְכַּז הָאֻמָּה הַיִּשְׂרְאֵלִית אַחֲרֵי חָרְבַּן בֵּית הַמִּקְדָּש בִּירוּשָׁלַיִם (בִּשְׁנַת 3 אַחֲרֵי הַחָרְבָּן). יוֹתֵר מְאֻשָּׁרִים הָיוּ הַיְּהוּדִים, אֲשֶׁר נִמְלְטוּ לְאֶרֶץ עֲרָב. הָאֶרֶץ הַזֹּאת לֹא נִכְבְּשָׁה לְעוֹלָם מֵאֵת הָרוֹמָאִים וְהָעֲרָבִים הָיוּ חַיִּים חַיֵּי שְׁבָטִים בּוֹדְדִים וּבֵין הַשְּׁבָטִים הָאֵלֶה נֶאֶחְזוּ בָאָרֶץ בְּתוּר שְׁבָטִים מְיוּחָדִים גַּם יְהוּדִים וַיִּחְיוּ שָׁם בְּחֵרוּת גְּמוּרָה בְמֶשֶׁך חֲמֵשׁ מֵאוֹת וְחֲמִשִּׁים שָׁנָה.

12

בּדִבְרֵי יְמֵי הָעוֹלָם רוֹאִים אֲנַחְנוּ לֹא אַחַת וְלֹא שְׁתַּיִם גִּבּוֹרִים נוֹפְלִים בְּיַד חַלָּשִׁים וְרַבִּים בְּיַד מְעַטִּים, בְּעֵת אֲשֶׁר הַחַלָשִׁים הָאֵלֶּה, הַמְּעַטִּים הָאֵלֶּה נִלְחָמִים בְּעַד רַעֲיוֹן רוּחָנִי, נִלְחָמִים בְּעַד קִנְיָנֵיהֶם הַמּוּסָרִים וְהָרוּחָנִים, אוֹ בְּעַד חֵרוּת עַמָּם וְאַרְצָם הַחֲבִיבָה לָהֶם מֵחַיֵּיהֶם. כֵּן נִצְּחוּ הַיְּהוּדִים אֶת הַיְּוָנִים בְּעֵת הַחַשְׁמוֹנָאִים. גִּבּוֹרֵי הַחֵרוּת הָאַחֲרוֹנִים הִתְבָּרְכוּ גַם הֵם בְּלִבָּם, כִּי יָשׁוּבוּ לָהֶם יְמֵי הַחַשְׁמוֹנָאִים, כִּי “הִתְרוֹמְמוּת הָרוּחַ” תִּתֵּן לָהֶם כֹּחַ וָעֹז לִגְבֹּר עַל אוֹיבֵיהֶם, כַאֲשֶׁר גָּבְרוּ אֲבוֹתֵיהֶם. אֲבָל תְּנָאֵי הַמִּלְחָמָה הַשְּׁנִיָּה הָיוּ שׁוֹנִים מִתְּנָאֵי הַמִּלְחָמָה הָרִאשׁוֹנָה, מֶמְשֶׁלֶת הַיְּוָנִים אוֹ הַסּוּרִים כְּבָר נָטְתָה לִנְפֹּל בִּיְמֵי הַחַשְׁמוֹנָאִים. הַפַּרְתִּים מִמִּזְרָח וְהָרוֹמָאִים מִמַּעֲרָב הִפְלִיאוּ בָהֶם אֶת מַכּוֹתֵיהֶם וְהַמְּרִיבוֹת וְהַמִּלְחָמוֹת הַפְּנִימִיוֹת עָזְרוּ לֹא מְעַט לְנִצְחוֹנוֹת הַיְּהוּדִים. מִלְבַד זֶה עָמְדוּ אָז בְּרֹאשׁ הַתְּנוּעָה הַלְּאֻמִּית אֲנָשִׁים, אֲשֶׁר הִצְטַיְּנוּ לֹא רַק כְּגִבּוֹרֵי חַיִל וְאַמִּיצֵּי לֵב הַיּוֹדְעִים לְדַבֵּר עִם אוֹיבֵיהֶם בַּשָׁעַר, כִּי אִם גַּם אֲנָשִׁים בַּעֲלֵי כִּשְׁרוֹנוֹת גְּדוֹלִים בְּתוּר חַכֵמי מְדִינָה, אֲנָשִׁים הַיּוֹדְעִים לְכַלְכֵּל אֶת מַעֲשֵׂיהֶם בְּחָכְמָה וּבִתְבוּנָה הַיוֹדְעִים לְאַחֵד עָצה עִם גְּבוּרָה לַמִּלְחָמָה. לֹא כֵן הָיָה הַמַּצָּב בִּיְמֵי הַמִּלְחָמָה הָאַחֲרוֹנָה. מֶמְשֶׁלֶת הָרוֹמָאִים הִגִּיעָה בָעֵת הַהִיא עַד מְרוֹם הַצְלָחָתָהּ. כָּל עַמֵּי הָעוֹלָם כְּבָר נִכְנְעוּ לִפְנֵיהֶם, לֹא הָיְתָה מְדִינָה, אֲשֶׁר שָׂגְבָה מֶהם. וְלוּ הָיָה צָרִיךְ יָכְלוּ לִשְׁלֹחַ אַרְצָה יְהוּדָה הֲמוֹן לִגְיוֹנוֹת, כִּי בִיְדֵיהֶם הָיוּ כֹּחוֹת כָּל הָעוֹלָם. אֶת טַכְסִיסֵי הַמִּלְחָמָה וְכָל תַּחְבּוּלוֹתֶיהָ הַשּׁוֹנוֹת הֵבִיאוּ הָרוֹמָאִים עַד תַּכְלִית הַשְּׁלֵמוּת וְאַנְשֵׁי צִבְאוֹתֵיהֶם הָיוּ מְלֻמָּדִים וּמְנֻסִּים בְּכָל הַמִּלְחָמוֹת הָרַבּוֹת נֶגֶד הַפַּרְתִּים, הַגֶּרְמַנִּים, הַבְּרִיטִים וְיֶתֶר הָעַמִּים עַל גְּבוּלוֹת אַרְצָם. בְּתּוֹךְ חֵילָם שָׂרְרוּ סְדָרִים טוֹבִים וּמִשְׁמַעַת חֲזָקָה. נֶגֶד אוֹיֵב כָּזֶה לֹא הָיְתָה דַיָּה הִתְרוֹמְמוּת הָרוּחַ לְבָד!

אֶמֶת הַדָּבָר, כִּי כְבָר הָיָה אִי אֶפְשָׁר לַיְּהוּדִים לָהִכין לָהֶם חַיִל טוֹב מְלֻמָּד וּמְנֻסֶּה, לְהָכִין לָהֶם מְכוֹנוֹת וּכְלֵי מִלְחָמָה, אַחֲרֵי אֲשֶׁר הָיוּ בְמֶשֶׁךְ מֵאָה וּשְׁלשִׁים שָׁנָה מְשֻׁעְבָּדִים לָרוֹמָאִים וַיַּחְדְּלוּ לִחְיוֹת חַיֵּי מְדִינָה הַדּוֹאֶגֶת בְּעַד שְׁלוֹמָהּ מֵחוּץ וְהַמֵּכִינָה לָהּ צָרְכֵי מִלְחָמָה. אֲבָל יֶשְׁנוֹ עוֹד כֹּחַ אֶחָד הַנּוֹתֵן נִצָּחוֹן בְּיוֹם קְרָב וּמִלְחָמָה וְגַם הוּא חָסֵר כָּלִיל לַיְּהוּדִים בְּמִלְחַמְתָּם הָאַחֲרוֹנָה. הַכֹּחַ הַזֶּה הוּא “הָאַחְדוּת”. לֹא נִמְצָא בָעֵת הַהִיא אִישׁ כְּאַחַד הַחַשְׁמוֹנָאִים, אִישׁ, אֲשֶׁר יוּכַל לַעֲמֹד בְּרֹאשׁ הָעָם כֻּלּוֹ, אֲשֶׁר יוּכַל לְכַלְכֵּל אֶת כָּל הַמִּלְחָמָה, אֲשֶׁר יוּכַל לְאַחֵד אֶת כָּל כֹּחוֹת הָעָם הַמְּפֻזָרִים לְכֹחַ גָדוֹל אֶחָד, אֲשֶׁר יוּכַל לַעֲמֹד נֶגֶד הָאוֹיֵב. לַמְּרוֹת כָּל גְּבוּרָתָם לֹא יָכֹל אַף אֶחָד מֵרָאשֵׁי הַקַּנָּאִים בָּעֵת הַהִיא לְהִתְנַשֵּׂא עַל חֲבֵרָיו וְלָקַחַת בְּיָדוֹ אֶת הַשִּׁלְטוֹן הָעֶלְיוֹן. בֵּין הָרָאשִׁים הָאֵלֶּה שָׁלְטָה קִנְאָה וְשִׂנְאָה. כָּל אֶחָד חָשַׁב רַק אֶת עַצְמוֹ רָאוּי לִשְׂרָרָה וְלָכֵן לֹא יָכְלוּ לָבֹא לְעֵצָה אַחַת לְסֵדֶר אֶחָד. אַחַד הַפְּקִידִים הַגְּדוֹלִים, אֲשֶׁר בּוֹ שָׂמוּ הַיְּהוּדִים חֵלֶק גָּדוֹל מִתִּקְוָתָם, מִפְּנֵי כִי יָדַע אֶת הָרוֹמָאִים וְאֶת סִדְרֵי מִלְחַמְתָם, וַיַּפְקִידוּ בְיָדוֹ אֶת הַחֵלֶק הַיּוֹתֵר גָּדוֹל וְהַיּוֹתֵר חָשׁוּב מֵאַרְצָם, אֶת אֶרֶץ הַגָלִיל, הוּא יוֹסֵף בֶּן מַתַּתְיָהוּ, בָּגַד בְּעַמּוֹ וַיִּמְכֹּר אֶת אַרְצוֹ לָאוֹיֵב.

וְהֶעְדֵּר הָאַחְדוּת הַנְּחוּצָה הוּא הוּא אֲשֶׁר פָּעַל, כִּי כָל עִיר וָעִיר, כָּל גְּדוּד וּגְדוּד נִלְחַם לְבַדּוֹ נֶגֶד הָאוֹיֵב; כִּי בְעֵת אֲשֶׁר הָיוּ הָרוֹמָאִים מַחֲרִיבִים אֶת הַגָּלִיל, שָׁקְטוּ יוֹשְׁבֵי עֵבֶר הַיַּרְדֵּן מִזְרָחָה עַל שְׁמָרֵיהֶם וּבִירוּשָׁלַיִם נִלְחֲמוּ בֵינֵיהֶם רָאשֵׁי הַסִּיעוֹת בְּעַד כָּבוֹד מְדֻמֶּה, בְּעַד הַשְּׂרָרָה בָעִיר. אַחֲרֵי אֲשֶׁר הֶחְרִיבוּ הָרוֹמָאִים אֶת הַגָּלִיל, עָבְרוּ לְעֵבֶר הַיַּרְדֵּן מִזְרָחָה וּלְאֶרֶץ אֱדוֹם וְהַקַּנָּאִים בְּירוּשָׁלַיִם עוֹד הוֹסִיפוּ לְהִלָּחֵם בֵּינֵיהֶם וּלְמַלֵּא אֶת הָעִיר חַלְלֵי חַרְבָּם. וּכְבָר הֶחְרִיבוּ הָרוֹמָאִים אֶת כָּל אֶרֶץ יְהוּדָה, כְּבָר הָיְתָה חַרְבָּם הַחַדָּה מוּנַחַת עַל צַוְּאֵרי יוֹשְׁבֵי יְרוּשָׁלַיִם וְהַמִּלְחָמוֹת בָּעִיר עוֹד לֹא חָדֵלוּ. בִּקְרוֹב הָרוֹמָאִים אֶל חוֹמַת הָעִיר הָיוּ כָל מֵגִינֶיהָ מִתְאַחְדִּים וּמַכִּים אֶת הָאוֹיֵב אָחוֹר. אֲבָל אַךְ סָרוּ הָרוֹמָאִים מֵעַל הַחוֹמָה, אַךְ שָׁבוּ אֶל מַחְנֵיהֶם הַבְּצוּרָה וְהִנֵּה עוֹד פַּעַם הָיְתָה בִירוּשָׁלַיִם חֶרֶב אִישׁ בְּרֵעֵהוּ. בִּתְּנָאִים כָּאֵלֶּה כָּבֵד הָיָה לְקַוֹת לִגְבֹּר עַל הָאוֹיֵב וּלְהַצִּיל מִמֶּנּוּ אֶת הָאָרֶץ וְאֶת חֵרוּתָה.

לֹא נִשְׁאַר אֵפוֹא לַנִּלְחָמִים כִּי אִם דָּבָר אֶחָד: לְהַצִּיל אֶת כְּבוֹדָם וְלָמוּת מוֹת גִבּוֹרִים, וְאֶת הַדָּבָר הַזֶּה עָשׂוּ בָּאוֹפָן הַיּוֹתֵר נַעֲלֶה. שֶׁבָע שָׁנִים אָרְכָה הַמִּלְחָמָה בֵין הָעָם הַקָּטֹן וְהַדַּל, כְּמוֹ הָיוּ הַיְּהוּדִים, וּבֵין הָעָם הָעֲנָקִי, כְּמוֹ הָיוּ הָרוֹמָאִים, וְאַף רַגע לֹא עָלָה עַל דַּעַת הַנִּלְחָמִים לְבַקֵּש עַל חַיֵּיהֶם מֵאֵת הָאוֹיֵב. כְּבָר הָיְתָה לִשְׁמָמָה כָל הָאָרַץ וְהָעִיר יְרוּשָׁלַיִם עוֹד עוֹמֶדֶת מוּכֶנֶת לַמִּלְחָמָה וְגִבּוֹרֶיהָ מַשְׁלִיכִים מֵעַל הַחוֹמָה אֶת הַמּצִיעִים לְהִכָּנֵעַ לִפְנֵי הָרוֹמָאִים. בְתוֹךְ הָעִיר חַלְלֵי רָעָב וּמַגֵּפָה, מֵחוּץ לָהּ חַלְלֵי חֶרֶב וּמוּקָעֵי עֵצִים וְגִבּוֹרֵי יְהוּדָה, עֲיֵפִים וּרְעֵבִים, עוֹד עוֹמְדִים עַל הַחוֹמָה כְסֶלַע אֵיתָן וּמֵשִׁיבִים אָחוֹר אֶת כָּל דִּבְרֵי הַשָּׁלוֹם אֲשֶׁר שׁוֹלֵחַ לָהֶם טִיטוּס, אֶת כָּל דִּבְרֵי סְלִיחָה וַחֲנִינָה הַמַּבְטִיחַ לָהֶם שַׂר צְבָא הָאוֹיֵב עַל יְדֵי הַבּוֹגֵד יוֹסֵף. וּכְבָר נָפְלָה יְרוּשָׁלַיִם, הַמִּקְדָּשׁ חָרֵב, וְהַר צִיּוֹן שָׁמֵם וְהַקַּנָּאִים הָאַחֲרוֹנִים עוֹד מוֹסִיפִים לְהִלָּחֵם בְּמָכֵרוּס, בְּמַסַּדָּה, בְּמִצְרַיִם, בְּקִירֵינִי וּבְכָל הַמְּקוֹמוֹת אֲשֶׁר נְשָׂאוּם רַגְלֵיהֶם לָבֹא שָׁמָה. הַקַּנָּאִים נִשְׁאֲרוּ נֶאֱמָנִים לַשְּׁבוּעָה, אֲשֶׁר נִשְׁבְעוּ לָמוּת וֹלא לְהַשְׁלִים עִם הָרוֹמָאִים. בֵּין כָּל הֶעָרִים הַגְדוֹלוֹת, אֲשֶׁר הֶחְרִיבוּ הָרוֹמָאִים, כְּמוֹ קַרְתַּגוֹ בִצְפוֹן אַפְרִיקָה, קוֹרִינְתְּ בְּאֶרֶץ יָוָן, הָיְתָה יְרוּשָׁלַיִם הָאַחֲרוֹנָה. נֶגֶד קַרְתַּגוֹ וְקוֹרִינְתְּ נִלְחֲמה רוֹמָא, בְּעֵת אֲשֶׁר הָיְתָה רַק גְּבֶרֶת אִיטַלִּיָה, וְנֶגֶד יְרוּשָׁלַיִם רוֹמָא אֲשֶׁר הָיְתָה כְּבָר גְּבֶרֶת כָּל הָעוֹלָם. וּבְכָל זֹאת הֶעֱבִידוּ גִבּוֹרֵי הַיְּהוּדִים אֶת צִבְאוֹת רוֹמָא עֲבוֹדָה יוֹתֵר גְּדוֹלָה, יוֹתֵר קָשָׁה מִגִבּוֹרֵי קַרְתַּגוֹ וּמִגִבּוֹרֵי יָוָן הָאֲחֲרוֹנִים. אִם לֹא יָכְלוּ לְהַצִּיל אֶת הָעִיר וֶאת הָאָרֶץ, מִלְּאוּ הֵם בִּמְלֹא מוּבַן הַמִּלָּה אֶת חוֹבָתָם בְּתוּר אַנְשֵׁי מִלְחָמָה וַיָּמוּתוּ כֻלָּם מוֹת גִבּוֹרִים בַּעד עַמָּם וְדָתָם, אַרְצָם וְחֵרוּתָם!


הָבוּ גֹדֶל לְגִבּוֹרֵי יִשְׂרָאֵל!

הכותל המערבי.
