מתוך המחברת "משא גיא החזיון"

[מרוֹמָא, חי באמצע המאה הי"ג]


מִתּוֹךְ הַמַּחְבֶּרֶת “מַשָּׂא גֵּיא חִזָּיוֹן” (עַל־יְהִירוּת הָעֲשִׁירִים)


עִקַּר כָּל-הַיְחָסִים / עשֶׁר וּנְכָסִים / וּמַה-בֶּצַע בְּשֹׁרֶשׁ לֹא יַעֲשֶׂה פֶּרִי / וְהוּא בֵּין אַחִים יַפְרִיא / וְאִם הַחֲכָמִים גָּדְלוּ / וְעַל-עֲשָׂרָה יוּחָסִין עָלוּ / הֲלֹא הֲלָכָה כְּדִבְרֵי הָאוֹמְרִים הַכֶּסֶף מְטַהֵר אֶת-הַמַּמְזֵרִים / וּמַעֲלִין לְיוּחָסִין עַל-פִּי עֵד אֶחָד / כָּל-שֵּׁכֵּן בְּהִתְחַבֵּר יַחַד / שְׁנֵיהֶם / בְּהַצַּד הַשָּׁוֶה שֶׁבָּהֶם.

וּמוּדַעַת זֹאת בְּכָל-הָאָרֶץ / כִּי אֵלֶּה תּוֹלְדוֹת הַזְּמָן: פֶּרֶץ הוֹלִיד אֶת-רָם וְחֶסְרוֹן הוֹלִיד אֶת-עוֹבֵד / וַיְצַו אוֹתוֹ הִכָּבֵד וְשֵׁב בְּבֵיתֶךָ / אוֹ הָכֵן בַּחוּץ מְלַאכְתֶּךָ / וְעִם-שׁוֹנִים אַל-תִּתְעָרֵב / וַאֲלֵיהֶם אַל-תִּקְרָב / וּפֶרֶץ לָכַד אֶת-הַמְּלוּכָה עַל-כָּל-הַמְּלָאכָה / לַעֲשׂוֹת אוֹתָהּ וְהוֹתֵר / וַאֲשֶׁר נִתַּן כֶּתֶר מַלְכוּת בְּרֹאשׁוֹ / כִּי רַב הוֹנוֹ וּרְכוּשׁוֹ / וּמִשְׁפַּחַת הַחֶסְרוֹנִי יָשַׁב לוֹ מִמּוּל / מִשְׁפַּחַת הָעָמוּל / וַיְשַּׂגֵב אֶבְיוֹן מֵעוֹנִי / לְחֶסְרוֹן מִשְׁפַּחַת הַחֶסְרוֹנִי / לְעָמוּל מִשְׁפַּחַת הָעָמוּלִי / לְשָׁאוּל מִשְׁפַּחַת הַשָּׁאוּלִי / וַיְהִי מוֹשָׁבָם בַּשְּׁפֵלָה / מִשְׁפַּחַת הָרִיב לָהּ / אֶרֶץ כָּבוּל / וְנָסַב לָהֶם הַגְּבוּל / מִנֶּגֶב / אֶרֶץ הַנֶּגֶב / וּמִמִּשְׁפַּחַת פֶּרֶץ / עָלָה הַפֹּרֵץ / לִפְנֵיהֶם / וַיִּתְקְעוּ אֶת-אָהֳלֵיהֶם / מוּל הַגִּלְגָּל / אֲשֶׁר בַּמַּעִגָּל / וַיִּבְנוּ לָהֶם בָּתִּים / וַיִּטְעוּ כְרָמִים וְזֵיתִים / וַיִּקְחוּ לָהֶם נָשִׁים מִכֹּל אֲשֶׁר בָּחָרוּ / בְּכַסְפָּם אֲשֶׁר אָצָרוּ / וַיִּסְּעוּ מִגְּבוּרֹות / הַתַּאֲוָה וַיַּחֲנוּ בַּחֲצֵּרוֹת / וְתָאַר לָהֶם הַגְּבוּל / בֵּין מִגְדֹּל וּבֵין יָם יָפוֹ / מִפּוֹ וּמִפּוֹ / וּתְשֻׁבָתָם הָרָמָתָה / בְּדֶרֶךְ אֶפְרָתָה / וַיָּשִׂימוּ כַצֹּאן / מִשְׁפְּחוֹת לַעַג וְלָצוֹן / הַלַּעַג הַשַּׁאֲנַנִּים / הַבּוּז לִגְאֵיוֹנִים / עַל-כֵּן קָרְאוּ לַמָּקוֹם הַזֶּה / גֵּיא הַחִזָּיוֹן עַד-יוֹם הַזֶּה.

וְהַמַטִּים עֲקַלְקַלּוֹתָם / וּנְלוֹזִים בְּמַעְגְּלוֹתָם / זֶה דַּרְכָּם בְּאָרְחוֹתָם / עַל-עָשְׁרָם אֲשֶׁר גָּדָל / וּמַהֲלָכָם עָקֵב בְּצַד גּוּדָל / זוֹקְפִים קוֹמָתָם / וּמִתְהַפְּכִים בִּמְסִלּוֹתָם / כַּאֲשֶׁר רוּחַ רָעָה מְבַעִתָּם / פַּעַם עַל-יְמִינָם / פַּעַם עַל-שְׂמֹאלָם / מִתְנַעְנְעִים בְּמַעְגָּלָם / וְרַגְלֵיהֶם רַגְלַיִם יְשָׁרוֹת / וּשְׁתֵּי כְתֵפֹת חֹבְרֹת / לָהֶם / בְּכָל-דַּרְכֵיהֶם / הָרֶגֶל מְהַלֶּכֶת רִאשׁוֹנִית / וְהַכָּתֵף הַשֵּׁנִית / וְעֵינֵיהֶם רָמוֹת / עַל-רְקָמוֹת / כָּל-דְּמוּת מַלְבּוּשֵׁיהֶם / אֲשֶׁר עֲלֵיהֶם / מִנְעָל דָּחוּק וּכְתֹנֶת דַּקָּה / גֹמֶד אָרְכָּהּ / וְאֵפוֹד וּמְעִיל וּבְגָדִים אֲחֵרִים / מִפָּז יְקָרִים / רְחָבִים מִלְּמַטָּה עַל-שׁוּלֵיהֶם / וּקְצָרִים מִלְּמַעֲּלָה עַל-חֲזֵיהֶם / וְרִמּוֹנֵי תְכֵלֶת וְאַרְגָּמָן סָבִיב רָאשֵׁיהֶם / וּפְתוּחִים מִצִּדֵּיהֶם / לְאַרְבַּעַת רִבְעֵיהֶם / וַאֲזוֹרֵיהֶם מְשֻׁבָּחִים / עַל-רִקֻּעֵי פַחִים / וְהֵם מְחֻשָּׁקִים כֶּסֶף / וָוֵיהֶם כֶּסֶף / וַחֲגוּרִים עַל-הָאַצִּילִים / כִּי מָתְנֵיהֶם אָחֲזוּ חִלְחוּלִים / וְנִשְׁתַּבְּרוּ רֹב צַלְעוֹתֵיהֶם / וְנִלְאוּ מֵהָכִיל אֲזוֹרֵיהֶם / אָכֵן אֶל-אֲחוֹרֵיהֶם / תְּלוּיִם עַד-שְׁתוֹתֵיהֶם / וְכִיס תָּלוּ לְצִדּוֹ / אִישׁ אִישׁ כַּאֲשֶׁר תַּשִּׂיג יָדוֹ / וְכֶסֶף וְזָהָב יְמַלְּאוֹ / וְנִשְׁמַע קוֹלֹו בְבֹאוֹ / גַּם אֶצְבְּעוֹתָיו / נוֹשְׂאוֹת טַבְּעוֹתָיו / מוּסַבֹּת מִשְׁבְּצֹת זָהָב / בְּמִלֻּאֹתָם / פִּתּוּחֵי חוֹתָם.

וּבְרֹאשָׁם כֹּבַע שָׁשְׁרָה / דַּקָּה וּקְצָרָה / וְעָלֶיהָ מַאֲחִיזִים / מַטְוֵה הָעִזִּים / עַל-חֲצִי הָרֹאשׁ שׂוּמָה / אֲשֶׁר שָׁם מְלוֹן הַכְּנִימָה / וְאֵין חִשְׁקָם / רַק לְכַסּוֹת נִתְקָם / אַךְ שַׂעֲרוֹתָם גְּדוֹלִים / קְוֻצּוֹתָיו תַּלְתַּלִּים / וְסֶרַח הָעֹדֶף / בְּצִיצִית רֹאשָׁם נוֹטֵף / עַל- הַכָּתֵף / וְאִם בֹּהַק פָּרַח בְּצַוְּרוֹנוֹ / שְׂעַר רֹאשׁוֹ יְכַסֶּנּוּ / קְצוּצֵי פֵאָה וּבְלוֹרִית / וְהֵם בַּעֲלֵי בְרִית / לַשְּׂעִירִים אֲשֶׁר הֵם / זֹנִים אַחֲרֵיהֶם / וְאוּלַי יָעוּפוּ / בְּשַׂעֲרוֹתָם אֲשֶׁר עָדָפוּ / יֵלְכוּ וְלֹא יִיעָפוּ / וְכָל-אֶחָד מֵהֶם מַחֲזִיק בְּאִוַּלְתּוֹ / אֵין אִישׁ נִחַם עַל-רָעָתוֹ / וְאִם אֶשְׁאָלֶנּוּ עַל-דֶּרֶךְ הַמּוּסָר / “דָּבָר לִי אֵלֶיךָ הַשָּׂר” / אֶקְרָא אֵלָיו וְלֹא יַעֲנֵנִי / אֲשֶׁר בְּשַׂעֲרָה יְשׁוּפֵנִי / וּשְׁבוּעַת אָלָה אִם יַבְטִיא / אֶל-הַשַּׂעֲרָה וְלֹא יַחְטִא / כִּי פְרָעוֹ לְשִׁמְצָה / לֵאמֹר אִם אֶבְחַר לְהִתְלוֹצֲצָה / אוֹ אִם עַוְלָה בִשְׂפָתַי תִּמְצָא / יִפֹּל מִשַּׂעֲרַת רֹאשִׁי אָרְצָה / וְלוּלֵי רְאִיתִים לָעָיִן / הַשֹּׁתִים בְּמִזְרְקֵי יָיִן / וּמְטַמְּאִים רֹאשׁ נִזְרָם / בְּמַעֲשֵׂיהֶם וּבְמוּסָרָם / אָמַרְתִּי עֲלֵיהֶם / כִּי נְזִירִים הֵם / וְאַשְׁרֵי אֶרֶץ צָדְקוּ שָׂרֶיהָ / זַכּוּ נְזִירֶיהָ / וְאוּלַי נִשְׁבְּרוּ הַמִּסְפְּרַיִם וְהַתַּעַר / אוֹ הָאוּמָנִים אָחֲזוּ שָׂעַר / עַל-כֵּן יַעַר שַׂעֲרָם גָּדֵל וְסוֹעֵר / וְחַיָּתוֹ אֵין דֵּי בָּעֵר / וְכִי יָרִים כִּנָּיו / לֶאֱחֹז בִּסְבַךְ קַרְנָיו / וְהִנֵּה הַשְּׁוָרִים מַלִּינִים עֲלֵיהֶם / כִּי לָקְחוּ קַרְנוֹתֵיהֶם / לַעֲשׂוֹת לָהֶם מַסְרֵקִים קָשִׁים וַחֲזָקִים / כִּי בִסְבַךְ שַׂעֲרוֹתָם תָּמִיד מִשְׁתַּבְּרִים / כְּשֶׁבֶר נֵבֶל יוֹצְרִים / וּמַדּוּעַ נֶחְתְּכָה הָאֵלָּה / אֲשֶׁר בָּהּ אַבְשָׁלוֹם נִתְלָה / אוּלַי תוֹרִיד לַמִּצְעָר / בַּעֲלֵי הַשֵּׂעָר. וְאִם הַכֹּל הוּא בְּטֶבַע הַזְּמָן וְגַלְגַּלָּיו / אֲשֶׁר יוּשַׁת עָלָיו / אוֹ הַיָּמִים יְמֵי אִוָּלֶת / וּשְׁנֵיהֶם מְלֵאִים פְּסֹלֶת / מַה-לִּי עוֹד לַעֲצַבִּים / אוֹ לִדְאָגוֹת וּמַכְאוֹבִים / לְבַקֵּשׁ חִשְּׁבֹנוֹת רַבִּים / בְּתַהֲפֻכוֹת הַזְּמָן וְהַיָּמִים הַבּוֹגְדִים / אֵלְכָה-לִי אֶל- הַיְדִידִים וְאֵצֵא אַחֲרֵיהֶם / וְאֶעֱשֶׂה כְמַעֲשֵׂיהֶם / כִּי טוֹב לִי הֱיוֹת זָנָב לַאֲרָיוֹת / וְלֹא רֹאשׁ לְשׁוּעָלִים קְטַנִּים / מְחַבְּלִים כְּרָמִים.
