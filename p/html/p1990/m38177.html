אֶבְכֶּה וַאֲקוֹנֵן וּדְמָעוֹת מֵעֵינַי נוֹזְלִים

*


קינה על מאורעות תרי״א

חתימה: א״ב, קומה, יחיה צארם, חזק יהוה


אֶבְכֶּה וַאֲקוֹנֵן וּדְמָעוֹת מֵעֵינַי נוֹזְלִים

בְּנֵי צִיּוֹן הַיְּקָרִים בָּאוּ עֲלֵיהֶם שׁוֹלְלִים

גֹּרְשׁוּ מִבָּתֵיהֶם וְעַל זֹאת מִתְאַבְּלִים

דְּגוּלֵי מַעְלָה אֶרְאֶלִּים וְחַשְׁמַלִּים.


הוֹ הוֹ נִקְרָא בְיוֹם הַשִּׁשִּׁי בְּכָל חוּצוֹת עִירֵנוּ

וְקוֹל צְעָקָה נִשְׁמַע בִּרְחוֹבוֹת בֵּיתֵנוּ

זָעַקְנוּ בְקוֹל תַּחֲנוּן אֶל ה׳ אֱלֹהֵינוּ:

חֲמוֹל עָלֵינוּ וְאָזְנֶיךָ אַל תַּעְלִים.


טֹרְפָה תְּפִלָּתֵנוּ וְשַׁעֲרֵי תְּפִלָּה נִנְעֲלוּ

יָשִׁישׁ בְּחָכְמָה וְזָקֵן בְּשָׁנִים עֲלֵיהֶם לֹא חָמְלוּ

כִּי עֲוֹנוֹתֵינוּ וּפְשָׁעֵינוּ עַד לַשָּׁמַיִם גָּדְלוּ

לְשֹׁד בֵּית הַתְּפִלָּה רַגְלֵי אוֹיֵב הָיוּ קַלִּים.


מִמָּרוֹם שָׁלַח אֵשׁ בְּעַצְמוֹת עַמּוֹ בְּחִירוֹ

נִשְּׂקָה בְיַעֲקֹב וְקִינָה חֵלֶף שִׁיר״וֹ

שַׂק וָאֵפֶר יֻצַּע לְכָל אֶחָד תְּמוּרַת צְנִיף פְּאֵרוֹ

עוֹלָלִים שָׁאֲלוּ לֶחֶם, וְאִישׁ לְאָחִיו וְגֵרוֹ

פַּת לֶחֶם וּמַיִם לֹא יֵשׁ כִּי הָיִינוּ דַלִּים.


צַדִּיק הוּא ה׳ כִּי פִיהוּ מָרִינוּ

קוֹל חָתָן וְקוֹל כַּלָּה נִשְׁבַּת בְּיוֹם זֶה בַעֲוֹנֵינוּ

רְדָפוּנוּ אוֹיְבֵינוּ וְקָרְעוּ סִפְרֵי תּוֹרָתֵנוּ

שַׁעֲרֵי הַהֵיכָלוֹת בָּקְעוּ בְּכַשִּׁילֵי בַּרְזִלִּים.


תּוֹרָה חֲמוּדָה וּצְנוּעָה כְּבֻדָּה בַת מֶלֶךְ

שָׁלְחוּ צָרִים יְדֵיהֶם בָּהּ סָחוֹב וְהַשְׁלֵךְ

וְהִיא מְקוֹנֶנֶת בְּקוֹל מַר וְאוֹמֶרֶת: אָנָה אֵלֵךְ?

כִּי הִפְשִׁיטוּהָ עֲרֻמָּה וּבָהּ הָיוּ מְזַלְזְלִים.


תּוֹרָה תְמִימָה וּנְעִימָה, אֵיךְ הִיא גוֹלָה וְסוּרָה

בְּיַד זָרִים אַכְזָרִים עוֹשֵׂי רָע וַעֲבֵירָה

לְמִרְמַס רַגְלַיִם וְנַעֲלַיִם לְהַכְעִיס וּלְהוֹסִיף מָרָה

עַל זֹאת תֶּאֱבַל הָאָרֶץ וּמִסְפֵּד חֵלֶף נְבָלִים.


תּוֹרָה טְהוֹרָה וּקְדוֹשָׁה נִקְרֵאת שַׁעֲשׁוּעִים

אֵיךְ הָיְתָה לְחֶרְפָּה בְּיַד צָרִים וְרָעִים

עַל זֹאת כָּל אִישׁ יִבְכֶּה וְיִקְרַע לִבּוֹ לִקְרָעִים

כִּי נִשְׁבָּה אֲרוֹן ה׳ וּבְנֵי אֵל גּוֹלִים.


תּוֹרָה יְקָרָה מִפְּנִינִים לֹא יִשְׁווּ כָל חֲפָצִים לִכְבוֹדֵךְ

חַלִּי נָא פְנֵי הָאֵל יִרְאֶה בְעָנְיֵךְ וּמְרוּדֵךְ

וְיֹאמַר דַּי קִינָה לְצָרוֹתֵינוּ וְיַרְאֵנוּ הֲדָרֵךְ וְהוֹדֵךְ

וְיִשְׁבֹּר מַטֵּה רְשָׁעִים וְגַם שֵׁבֶט מוֹשְׁלִים.


תָּמְהוּ מָתְנֵינוּ וְכָשְׁלוּ דְּרָכֵינוּ

מֵאֵין כְּסוּת בַּקָּרָה כָּחַשׁ בְּשָׂרֵנוּ

לְחֶרְפָּה וּלְקֶלֶס הָיִינוּ וְחָשְׁכוּ עֵינֵינוּ

וּלְרַחֲמֵי שָׁמַיִם הִנְנוּ מְיַחֲלִים.


קוּמָה ה׳ עֶזְרָתָה לָּנוּ וּלְמַעַן חַסְדְּךָ פְּדֵנוּ

וְרַחֵם עַל מִקְדָּשְׁךָ וִירוּשָׁלַיִם בֵּית תִּפְאַרְתֵּנוּ

מֵעֵינֶיךָ לֹא נֶעְלָם כָּל הַבָּא עָלֵינוּ

הֵעָתֵר לָנוּ וְתָקִים מֵעָפָר דַּלִּים.


יְהִי נָא חַסְדְּךָ לְנַחֲמֵנוּ כְּאִמְרָתְךָ לָנוּ

חוֹזֶה נֶאֱמַן בֵּיתְךָ מֹשֶׁה נְבִיאֵנוּ

יִחַלְתָּנוּ עַל יָדוֹ שֶׁלֹּא תַעֲשֶׂה כָלָה אוֹתָנוּ

הָחִישׁ יְשׁוּעָתְךָ לָנוּ כִּי אַתָּה סוֹמֵךְ נוֹפְלִים.


צוֹפֶה אַתָּה כִי עֲנִיּים אֲנַחְנוּ

אֵין לָנוּ עַל מִי לְהַשְׁלִיךְ יְהָבֵנוּ

רְאֵה בְעָנְיֵנוּ וְרִיבָה רִיבֵנוּ

מִמָּרוֹם תָּחִישׁ יְשׁוּעָה לְעָם נִקְרָאִים בְּנֵי אֵלִים.


חַי חַי הוּא יוֹדֶךָ כָּמוֹנוּ זֶרַע אֵיתָנִים

זֵכֶר רַב טוּבְךָ יַבִּיעוּ וְיוֹדִיעוּ אָבוֹת לְבָנִים

קְרָא שְׁנַת רָצוֹן וּגְאֻלָּה לְגֶזַע אֱמוּנִים

יְמִינְךָ הוֹשִׁיעָה וּבְרִית הָאֵיתָנִים.


מִמַּעֲמָקִים תִּדְלֵנוּ וְלֹא נוֹסִיף לִהְיוֹת כּוֹשְׁלִים

וִיהַלְלוּךָ דְּגוּלֵי מַעְלָה אֶרְאֶלִּים וְחַשְׁמַלִּים.
