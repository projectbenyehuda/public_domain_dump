בָּרְכוּ עַמִּים אֱלֹהֵינוּ

ברכת המזון


בָּרְכוּ עַמִּים אֱלֹהֵינוּ / וְהָבוּ לִשְׁמוֹ בִרְכוֹת טוֹב

כִּי בְרֹב טוּבוֹ הֶחֱיָנוּ / וְנֶפֶשׁ רְעֵבָה מִלֵּא טוֹב


אֱמוּנָתוֹ וְחַסְדּוֹ זִכְרוּ / בִּקְהַל אֱמוּנָיו וְהַשְׁמִיעוּ

כִּי יְדֵי טוּבוֹ לֹא קָצְרוּ / וַחֲסָדָיו לֹא נִמְנְעוּ

וּלְכָל בְּרוּאָיו אֲֲשֶׁר נוֹצְרוּ / הוּא זָן לְבִלְתִּי יִיגָעוּ

יֶהֱמָיוּן וְלֹא יִשְׁקֹטוּ / וְלָךְ אֵל יְשַׁוְּעוּ לִמְצֹא טוֹב

תִּתֵּן לָהֶם יִלְקֹטוּ / תִּפְתַּח יָדְךָ יִשְׂבְּעוּן טוֹב:


בְּהַנְחִילְךָ אֶת אֲבוֹתֵינוּ / אֶרֶץ חֶמְדָּה וִיקָרָה

וּמִבֵּית עֲבָדִים פְּדִיתָנוּ / וְהִנְחַלְתָּנוּ בְרִית וְתוֹרָה

נוֹדֶה שִׁמְךָ אֱלֹהֵינוּ / וּנְבָרֶכְךָ בְשִׁיר וְזִמְרָה

כִּי לַהוֹלְכִים לְפָנֶיךָ / בְּתָם לְבָבְךָ לֹא תִמְנַע טוֹב

וְכָל הַדּוֹרְשִׁים רְצוֹנֶךָ / יוי, לֹא יַחְסְרוּ כָל טוֹב:


רַחֵם עַל בֵּית מְנוּחֶךָ / וְעַל עִיר קָדְשָׁךְ אֵל עֶלְיוֹן

וְעַל שְׁאֵרִית עַם זְנוּחֶיךָ / הַחוֹכִים מִמָּךְ פִּדְיוֹן

וּמַלְכוּת דָּוִד מְשִׁיחֶךָ / תַּחְזִיר לִמְקוֹמָהּ לְצִיּוֹן

וּסְגֻלָּה אֲשֶׁר פִּזַּרְתָּ / אֱסוֹף בְּכוֹחֲךָ אֶל אֶרֶץ טוֹב

וּבְנֵה כַּאֲשֶׁר דִּבַּרְתָּ / בְּיָמֵינוּ הָהָר הַטּוֹב:


מְלוֹךְ עָלֵינוּ בְמַלְכוּתְךָ / כִּי אֵל טוֹב וּמֵטִיב אַתָּה

וְשַׂבְּעֵנוּ מִבִּרְכָתְךָ / וּמִטּוּבְךָ אֲשֶׁר צָפַנְתָּ

וְזַכֵּנוּ לְבִנְיַן בֵּיתְךָ / וְלִמְשִׁיחֲךָ אֲשֶׁר נָסַכְתָּ

וּמַלֵּא שְׁאֵלַת עָם אוֹמְרִים: / גַּם יוי יִתֵּן הַטּוֹב

וְיֶעֱרַב לְךָ שִׁיר הַמְזַמְּרִים: / הוֹדוּ לַיוי כִּי טוֹב:
