לחתן וכלתו בת יצחק

יִשְׂמַח חָתָן בָּאֲדֹנָי

[חתן]


יִשְׂמַח חָתָן בַּיוי / כִּי הִצִּיב אֶת מַעֲמָדוֹ

וּבְמוֹט הָרִים לֹא יִמּוֹט / כִּי יוי סוֹמֵךְ יָדוֹ

שִׂמְחַת עוֹלָם עַל רֹאשׁוֹ / תִּהְיֶה וְיַזְרִיחַ כְּבוֹדוֹ

מַהֵר יֵלְכוּ יוֹנְקוֹתָיו / וִיהִי כַזַּיִת הוֹדוֹ:


אֶל מִי אֲדַמֵּהוּ בְצֵאתוֹ / כַּעֲטֶרֶת פָּז מִנְּוָתוֹ

אֶל זֹהַר הָרָקִיעַ / אוֹ אֶל סַהַר בְּמִלֵּאתוֹ

אוֹ לִמְאוֹר כּוֹכְבֵי שַׁחַק / אוֹ לַשֶּׁמֶשׁ בַּעֲלוֹתוֹ

כִמְעַט קָט עָשַׁק שֶׁמֶשׁ / אֶת הוֹד פָּנָיו וַחֲמָדוֹ:


בַּעֲבוּר שְׁמוֹ נְתַנְאֵל נִקְרָא / סוֹד אֵל מֵבִין בּוֹ נִגְלָה

לֵאמֹר: נָתַן אֵל אוֹתוֹ / לְתִפְאֶרֶת לְשֵׁם וְלִתְהִלָּה

וּבַעֲבוּר שֵׁם אָבִיו אַב הֲמוֹן מִזַּרְעוֹ עוֹלָם נִמְלָא

וְלָכֵן כָּל רוֹאָיו יֹאמְרוּ: / אַשְׁרֵהוּ וְאַשְׁרֵי יוֹלְדוֹ:


בְּמִשְׁפַּחְתּוֹ דָבְקָה נַפְשׁוֹ / וְהִיא לוֹ מֵרֹאשׁ נִדְרָשָׁה

וְכַאֲשֶׁר הֶחֱזִיק בְּבַת דּוֹדוֹ / הוּא יִצְחָק לוֹ לְמוֹרָשָׁה

צְחוֹק עָשָׂה לוֹ הָאֱלֹהִים / וְשִׂמְחָתוֹ לָעַד חֲדָשָׁה

וְהַשָּׁלוֹם שָׁכַן בְּאָהֳלוֹ / וְרָבַץ, וְאֵין מִי יַחֲרִידוֹ:


רָחֲבָה לָךְ יַד הָאַהֲבָה / הַאֲרִיךְ לָהּ מֵיתָרֶיךָ

הִנֵּה עִתָּךְ עֵת דּוֹדִים / וּבַת דּוֹדָךְ הִיא עֶזְרֶךָ

בְּאַהֲבָתָהּ תִּשְׁגֶּה תָמִיד / וּבָרוּךְ יִהְיֶה מְקוֹרֶךָ

וְלֹא יְכַזְּבוּ לָעַד מֵימָיו / כִּי אֵל מָלֵא הֶעֱמִידוֹ:


הִנֵּה הַזְּמָן לָךְ נָתוּן / אֶל כָּל אֲשֶׁר תַּחְפּוֹץ יִפְנֶה

כָּל אֲשֶׁר בִּלְבָבָךְ קוּם עֲשֵׂה / וּקְרָא אֵלָיו וְהוּא יַעֲנֶה

אֵיכָה תֶחֱרַד מִיָּמִים / וּסְבִיבָךְ מַלְאָךְ חוֹנֶה

לִשְׁמָרְךָ מִפֶּגַע רָע / תָּמִיד, פֶּן יִשְׁלַח יָדוֹ:


מִשְׂרָתָם לָעַד תִּכּוֹן / וְלָעַד יִבְנוּ וְיַצְלִיחוּ

וּכְבוֹד עוֹלָם יִירָשׁוּ / וּפְרִי מַיִם יַצְמִיחוּ

וּבִימֵיהֶם יָחִישׁ גּוֹאֵל / הָאֵל לְגוֹלִים נִשְׁכָּחוּ

וְרָצָה בְעַמּוֹ וְלֹא יְאַחֵר / קֵץ יִשְׁעוֹ, כִּי בָא מוֹעֲדוֹ:
