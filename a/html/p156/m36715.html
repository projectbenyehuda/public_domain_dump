אִגֶּרֶת עִמָּנוּאֵל לְרִבִּי הִלֵּל

זֶה הַכְּתָב שֶׁשָּׁלַח ר׳ עִמָּנוּאֵל בְּר׳ שְׁלֹמֹה בַּר יְקוּתִיאֵל ז״ל לְרִ׳ הִלֵּל


זְמַן הָרַע תְּכוּנַת אִישׁ גָּרַע / וְסִבּוֹתָיו וְעִתָּיו לוֹ לְמוֹקֵשׁ

וְהַחָכְמוֹת מְאֹד רָמוֹת, וְאַתָּה / תְּבַקֶּשׁ לָךְ גְּדוֹלוֹת? – אַל תְּבַקֵשׁ!


אַל תְּבַהֵל עַל פִּיךָ וְלִבְּךָ אַל יְמַהֵר / בְּלֹם פִּיךָ מִלְּדַבֵּר וְלִבְּךָ מִלְּהַרְהֵר / כִּי יֵשׁ אֱלֹהִים שׁוֹפְטִים / עַל כֵּן יִהְיוּ דְבָרֶיךָ מְעַטִּים / וְהַחָכְמוֹת רָמוֹת / וְהַבִּינוֹת נַעֲלָמוֹת / וְהַלְּבָבוֹת נֶאֱלָמוֹת / וְהַהַשָּׂגוֹת דּוֹמְמוֹת / כְּגֶחָלִים עוֹמְמוֹת / וַיִּסְעוּ מִיָּטְבְתָה וַיַּחֲנוּ בִּישִׁימוֹת / וְהַשֵּׂכֶל הָיָה לְחָרְבָּה / כְּסוֹמֵא בָאֲרֻבָּה / אֲשֶׁר לֹא יָדַע בֵּין יְמִינוֹ לִשְׂמֹאלוֹ וּבְהֵמָה רַבָּה / וְהַיּוֹם קָצָר וְהַמְּלָאכָה מְרֻבָּה / וְהַנֶּפֶשׁ עֲווֹנָה בָּהּ / הִיא שׁוֹבֵבָה / בָּעֶרֶב הִיא בָאָה וּבַבֹּקֶר הִיא שָׁבָה / וּתְבוּאָתָהּ מְרֻבָּה / לְאֵל נֵכָר אֲשֶׁר בְּקִרְבָּהּ / לַעֲשׂוֹת כַּוָּנִים לְהַעֲצִיבָהּ / וְהַזְּמַן בּוֹגֵד וְהַמִּקְרֶה מוֹקֵשׁ / יִפֹּל בָּהֶם נִפְתָּל וְעִקֵּשׁ / וְאַתָּה תְּבַקֵּשׁ לְךָ גְּדוֹלוֹת? – אַל תְּבַקֵּשׁ! /

הִשָּׁמֶר לְךָ פֶּן תִּוָּקֵשׁ / וְעַתָּה, מַה יִקָּחֶךָ לִבֶּךָ וּמַה יִּרְמְזוּן עֵינֶיךָ / כִּי תַּטְרִיד רַעְיוֹנֶיךָ / וְתִשָּׂא מָרוֹם עֵינֶיךָ / וְאֶל קְדוֹשׁ יִשְׂרָאֵל / רֹאשׁ גּוֹלַת אֲרִיאֵל / אִישׁ הַבֵּינַיִם / נֶחְמָד הוּא לְהַשְׂכִּיל וְתַאֲוָה הוּא לָעֵינַיִם / שָׁקַל בַּפֶּלֶס שִׁירִים וּמְלִיצוֹת בְּמֹאזְנַיִם / פְּלוֹנִי נֵר הַמַּעֲרָבִי אֲשֶׁר בִּילִידֵי הָרָפָה / אֲשֶׁר עָלֵהוּ לִתְרוּפָה / וּמִפִּרְיוֹ יֹאכְלוּ כָּל בַּעֲלֵי שָׂפָה / וְיִחְיוּ הַמֵּתִים בַּמַּגֵּפָה וְאַתָּה חֹק הַמּוּסָר פָּרַעְתָּ / וְקָלוֹן מִכָּבוֹד שָׂבַעְתָּ / הִגַּדְתָּ וְהוֹדַעְתָּ וְהִשְׁמַעְתָּ / מַה לְּךָ כִּי נִזְעָקְתָּ / כָּל קֳבֵל דִּי דְּנָה יְדַעְתָּ / וּפָעַרְתָּ פִיךָ לִבְלִי חֹק / וְהִשְׁמַעְתָּ קוֹלְךָ לְמֵרָחוֹק / כְּקוֹל הַסִּירִים תַּחַת הַסִּיר / תְקִילָא בְמֹאזְנַיָּא וְאִשְׁתְּכַח הַסִּיר / וַתֵּלֶךְ דְּרָכִים לֹא סְלוּלִים / בְּרֹב חֲלוֹמוֹת וַהֲבָלִים / וְהוֹצֵאתָ מִפִּיךָ מִלִּין / וַתִּלְעַג וְאֵין מַכְלִים / וְדִבְרֵי תוֹרָה הַמַּצִּילִים / הִכִּיתָ לְשִׁבְעָה נְחָלִים / וְהִדְרַכְתָּ בַּנְּעָלִים / דֶּרֶךְ לַעֲבֹר מְגֹאָלִים / תִּרְמְסֶנָּה רָגֶל, רַגְלֵי עָנִי, פַּעֲמֵי דַלִּים / וְעַתָּה, אַל תִּפֹּל עָלֶיךָ אֵימָתָה וָפַחַד / הַזְכִּירֵנִי, נִשָּׁפְטָה יָחַד / יַחְדָּו לַמִּשְׁפָּט נִקְרָבָה / הִנֵּה קָמָה אֲלֻמָּתִי וְגַם נִצָּבָה / אִם תוּכַל – הֲשִׁיבֵנִי, עָרְכָה לְפָנַי הִתְיַצָּבָה / הִדְרַכְתָּ בְיָם הַחָכְמָה סוּסֶיךָ / וְשִׁפְעַת הַתְּבוּנָה תְּכַסֶּךָ / אוֹ תִבְטַח אֶל מַעֲשֶׂיךָ / לָצֵאת בְּיָד רָמָה / וְלִבְקֹעַ יָם בְּחָכְמָה / בְּמַטֵּה הַחֵמָה וּבְחֶרֶב שְׁנוּנָה / לֹא בֶאֱמֶת וְלֹא בֶאֱמוּנָה / הַגֵּד אִם יָדַעְתָּ בִינָה / וְכָל אִישׁ אֲשֶׁר מַחֲזֵה שַׁדַּי יֶחֱזֶה / מִשְׁמַן בְּשָׂרוֹ יֵרָזֶה / בְּשָׁמְעוֹ הַדָּבָר הַזֶּה / וְאַתָּה עַל לִבְּךָ וְכָל עִנְיָנָיו / אֲשֶׁר הוּא חָכָם בְּעֵינָיו / אֵיךְ לֹא הִרְגִּישׁ בַּעֲווֹנוֹ / כַּאֲשֶׁר פָּנָה לִימִינוֹ / וְעָשָׂה יֶשְׁנוֹ אֵינוֹ / וַיֵּשְׁתְּ מִן הַיַּיִן וַיִּשְׁכָּר / חִלֵּל אֶת קֹדֶשׁ ה׳ אֲשֶׁר אָהֵב וּבָעַל בַּת אֵל נֵכָר / וְיָצָא בְּחֵמָה גְדוֹלָה לְהַשְׁמִיד וּלְהַחֲרִים רַבִּים / וַיִּפְּלוּ אַרְצָה מִן הַצָּבָא וּמִן הַכּוֹכָבִים / וְצַלְעוֹת הַבַּיִת וְהַעֻבִּים / וְעַמּוּדֵי הַתָּוֶךְ אֲשֶׁר הַבַּיִת נָכוֹן עֲלֵיהֶם / הֵסִיר מִמְּקוֹמָם וְשָׂם שִׁיחוֹת תַּחְתֵּיהֶם / וְתָלָה יְסוֹדֵי הַחָכְמָה / וּמֶחְקְרֵי תַעֲלוּמָה / עַל בְּלִימָה / לֹא דִקְדֵּק וְלֹא חִפֵּשׂ / וְחָצַב מֵעַמּוּדִים גְּדוֹלִים אֲוִיר שֶׁאֵינוֹ נִתְפָּשׂ / וְנָטַשׁ אֱמוּנַת הַעִבְרִיּים וְהִסְפִּיק בְּיַלְדֵי נָכְרִים / אָהֳלֵי אֱדוֹם וְיִשְׁמָעֵאל, מוֹאָב וְהַגְרִים / וְדִבֵּר מִדְבַּר זָרִים / וַאֲשֶׁר תַּחַת מֶגְרְפוֹתֵיהֶם עָבֵשׁוּ / וּמַעֲטֵה גַאֲוָה לָבָשׁוּ / מִן גֵּו יְגֹרָשׁוּ / לֹא זֹרוּ וְלֹא חֻבָּשׁוּ / גַּם נָשַׁף בָּהֶם וַיִּיבָשׁוּ / הִיא הָאִגֶּרֶת הָרִאשׁוֹנָה / הַשּׁוֹנָה בִלְשׁוֹנָה / אֲשֶׁר דְּבָרֶיהָ רָשׁוּ וְרָעֵבוּ / עָמָל כִּתֵּבוּ / וְדוֹבֵר דְּבָרִים תַּמִּים יְתָעֵבוּ / וַיֵּלְכוּ אַחַר הַכָּזָב אֲשֶׁר אָהֵבוּ / וַאֲנִי, בִּרְאוֹתִי הַקְדָּמָתָהּ / שָׂמַחְתִּי עַל יְפִי נְעִימוּתָהּ / וַיֹּאמֶר לִבִּי בְּקִרְבִּי: וְאֶל בְּשׂוֹרָה טוֹבָה תָבוֹא; וְכַאֲשֶׁר יָרַדְתִּי אֶל עֹמֶק כַּוָּנָתָהּ / וְהִכַּרְתִּי קִלְקוּל אֲמָנָתָהּ / עָלְתָה בָּאְשָׁהּ וְצַחֲנָתָהּ / וְהִנֵּה רֵיקָה יוֹצֵאת / וְלָהּ אֵין בְּשׂוֹרָה מוֹצֵאת / וַעֲווֹנָהּ צָרוּר עַל שִׁכְמָהּ / נִטְּשָׁה עַל אַדְמָתָהּ, אֵין מְקִימָהּ / וְגַם אַנְשֵׁי הַמָּקוֹם אָמְרוּ: לֹא הָיְתָה בָזֶה קְדוֹשָׁה / אוֹי לָהּ לְאוֹתָה בּוּשָׁה! / וֵיִּקָּרֵא שְׁמָהּ: כָּזְבִּי קְטַנַּת אֲמָנָה / אֲשֶׁר מִכְּזִיב הַמְּדִינָה / וְהָאִגֶּרֶת הַשְּׁנִיָּה / תַּאֲנִיָּה וַאֲנִיָּה / וְדִמְעָתָהּ עַל לֶחְיָהּ / וְכָל חַכְמֵי לֵב בְּצֵל שַׁדַּי יִתְלוֹנְנוּ עָלֶיהָ / יִלּוֹנוּ וְיִתְאוֹנָנוּ / כִּי אֲשֶׁר לֹא סֻפַּר לָהֶם רָאוּ וַאֲשֶׁר לֹא שָׁמְעוּ הִתְבּוֹנָנוּ וְכוּ׳.
