[צְבִי הַחֵן בְּרָקָיו מִמְּקוֹמָם]

שִׁיר לִכְבוֹד הַמְשׁוֹרֵר יְהוּדָה צִיצִילְיַאנוֹ

͝ – – – ͝ – – – ͝ – –

צְבִי הַחֵן בְּרָקָיו מִמְּקוֹמָם / יְיָרְאוּנִי וְזָחַלְתִּי לְרַעְמָם

יְיָרְאוּנִי לְבַד שִׁירָיו וְאֵין שִׁיר / אֲשֶׁר שִׁירַי לְעַד עוֹלָם עֲמָמָם

וְאֵין זִכְרוֹן אֱלֵי רָאמוֹת וְגָבִישׁ / עֲנַדְתִּים מוּל מְאוֹרֵי יַהֲלֹמָם

אֲבוֹי שִׁירַי אֲשֶׁר שָׁרוּ לְכָל־שָׂר / וְהִנֵּה שֻׁדְּדוּ הֵם כַּהֲתִמָּם

תְּמוֹל כָּל־חֵיל לְשִׁירוֹתַי יְיַחֵל / וְהַיּוֹם אֵין אֲשֶׁר יִדְרֹשׁ שְׁלוֹמָם

זְמִירַי הָפְכוּ כַיּוֹם וְגַם יַד / יְהוּדָה הָיְתָה בָהֶם לְהֻמָּם

וְהֵפֵר הַזְּמָן כָּל־מוֹעֲצוֹת שָׁר / וְהוּא יַעַשׂ אֲשֶׁר חָשַׁב וְזָמָם

וְאֵיכָה נַחֲשֹׁב לַחְקֹר נְבוּאוֹת / חֲרוּזוֹתָיו וְאֵין פּוֹתֵר חֲלוֹמָם

וְאֵיכָה נֵֽשְׁבָה עַל־כֵּס זְמִירָיו / וְלָאִינוּ לְהַשִּׂיג אֶת־הֲדוֹמָם

וְשִׁיר אָסָף וְהֵימָן נִטְּשׁוּ עַל / עֲפַר־אֶרֶץ כְּאִלּוּ אֵין מְקִימָם

וְשָׁרִים אָחֲזוּ שַׂעַר לְפָנָיו / וְנָסִים כִּי אֱלֹהֵינו הֲדִמָּם

וְכָל־חוֹזִים הֲכִי הוֹזִים וְשׁוֹכְבִים / וְלֹא יֵדְעוּ בְּשִׁכְבָם אוֹ בְקוּמָם

וְשִׁירָיו בִּֽגְבוּרָה נֶאֱזָרִים / וְשִׁירֵינוּ עֲטוּיִם עַל־שְׂפָמָם

וְלֹא נַשִּׂיג גְּלוּיֵי מַאֲמָרָיו / וְאֵיכָה נַחֲקֹר סוֹדוֹת עֲלוּמָם

וְהוּא יַחַשׂ לְכָל־נָגִיד וְטִפְסָר / וְהוּא עֶדְיָם וְחֶלְיָתָם וְנִזְמָם

וְאֵין בַּחַי וּבִמְדַבֵּר כְּעֶרְכּוֹ / וְאַף כִּי לֹא בְצוֹמֵחַ וְדוֹמָם

וְכָל־שָׁרִים כְּנֶגְדּוֹ כַעֲרוּמִים / וּבִזְמִרָיו יְכַסּוּ מַעֲרוּמָם

וְאֵיכָה נַחֲזִיק מַעֲמָד לְפָנָיו/ וְהוּא לוֹכֵד חֲכָמֵינוּ בְּעָרְמָם

וְאֵיךְ אַקְרִיב לְפָנָיו אֶת־קְטָרְתִּי / וְיָפִיחַ לְכָל־רוֹכְלִים וּבָשְׂמָם

וְכָל־שָׁרִים וְנוֹגְנִים יַחֲזִיקוּ / בְּכַנְפוֹת שִׁיר וְהוּא יִשְׁכֹּן מְרוֹמָם

וְלוּ פָּגַע זְמִירוֹ אֶת־לְשׁוֹנוֹת / נְשִׂיאִי הַמְּלִיצָה־הֶחֱרִימָם

וְלוּ דָרַךְ בְּרַגְלָיו עַל־בְּאֵרוֹת / חֲפָרוּמוֹ נְדִיבֵי עָם סְתָמָם

וְלוּ עִמּוֹ יְרוּצוּן כּוֹֽכְבֵי־אֵל / בְּמַעֲרֶכֶת תְּבוּנוֹת הוּא קְדָמָם

וְלוּ יִלָּחֲמוּ עִמּוֹ צְבָא רוֹם / אֲזַי מִדַּהֲרוֹת שִׁירָיו הֲלָמָם

וְלוּ יִגְעַר בְּיַרְדֵּן הַתְּעוּדוֹת / אֲזַי יִבָּקְעוּ מֵימָיו וְזִרְמָם

וְלוּ יֹאמַר “הַשֶּׁמֶשׁ דּוֹם בְּגִבְעוֹן” / אֲזַי מִגַּעֲרָתוֹ זָע וְדָמָם

וְאוּלַי הוּא אֲחִי שֶׁמֶשׁ וְסַהַר / וְעַל־כֵּן הֵם קְרָאוּהוּ תְאוֹמָם

צְבִי הַחֵן יְדֵי אֵל כּוֹנֲנוּהוּ / וְשַׂרְפֵי רוֹם יְלָדוּהוּ כְצַלְמָם

וּבוֹ יִתְפָּאֲרוּ רָאשֵׁי תְעוּדוֹת / וְהַשָּׁרִים קְרָאוּהוּ נְעִימָם

וְהָיוּ הֵם בְּעֵינֵיהֶם חֲגָבִים / וְכֵן הָיוּ וְהוּא קֶרֶן רְאֵמָם

וְהָיוּ כָּל־יְמֵי נוֹדוֹ עֲרָבִים / וְלֵילוֹת נִהְיוּ עִמּוֹ כְּיוֹמָם

וְלֹא נִרְאֶה לְבַד עַל־פִּיו תְּעוּדוֹת / וְאוּלַי כִּי בְּטַבַּעְתּוֹ חֲתָמָם

וְשָׁרִים יִשְׁמְרוּ מַחְסוֹם לְפִיהֶם / עֲדֵי אֶחְשֹׁב בְּשִׂכְלִי כִּי חֲסָמָם

וְזִיו פָּנָיו הֲכִי יַחְפִּיר מְאוֹרוֹת / וְעֵינָיו מוֹֽשְׁכִים קֶשֶׁת לְתֻמָּם

וְאִמְרָתוֹ לְקָחוּהָ מְלִיצִים / לְמֵנִקְתָּם וְהוֹרָתָם וְאִמָּם

וְשִׁירָתוֹ לְכָל־שִׁירִים לְצוּרָה / וְשִׁיר כָּל־שָׁר יְהִי לָהֶם לְגָלְמָם

לְמִי חֶמְדַּת זְמָן כִּי אִם לְזֵכֶר / פְּלִיאוֹת מַעֲלוֹתֵהוּ וְלִשְׁמָם

וְהֵאִירוּ פְּנֵי תֵּבֵל כְּהַיּוֹם / צְבִי הַחֵן בְּרָקָיו מִמְּקוֹמָם.
