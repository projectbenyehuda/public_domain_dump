מִי שַׁלִּיט

מִי שַׁלִּיט בְּקַצְוֵי מְעוֹנָי / מִי לוֹ יִכְרְעוּן כָּל הֲמוֹנָי

מִי מוֹשֵׁל רוּם וְתַחַת / מִי הַמֵּבִין כָּל צְפוּנָי

מִי זֶה מֶלֶךְ הַכָּבוֹד הֲלֹא אַתָּה מִקַּדְמוֹנָי / מִי כָמוֹכָה בָּאֵלִים יְיָ

מִי אַדִּיר עוֹטֶה אוֹרוֹת / מִי בָהִיר אוֹזֵר גְּבוּרוֹת

מִי גִּלָּה עֶשֶׂר סְפִירוֹת / מִי דָלָה שָׁלֹשׁ בְּצוּרוֹת

מִי הִכְפִּיל שֶׁבַע נְצוּרוֹת / מִי וּשְׁנֵים עָשָׂר עֲצוּרוֹת

מִי פָעַל וְעָשָׂה קוֹרֵא הַדּוֹרוֹת / מִי כָמוֹכָה בָּאֵלִים יְיָ


מִי זָמַם חֲקֹר סוֹד רִכֵּן / מִי חֲתֹם בְּשֵׁשׁ טַבְּעוֹת סִכֵּן

מִי טִפַּח שָׁלֹשׁ לְתַכֵּן / מִי יְעָדָם מִשָּׁלֹשׁ כֵּן

מִי כִּידוֹד אֵשׁ בְּרוּם שִׁכֵּן / מִי לִהֵק רוּחַ לְהַסְכֵּן

[מִי מָדַד בְּשָׁעֳלוֹ מַיִם וְשָׁמַיִם בַּזֶּרֶת תִּכֵּן / מִי כָמוֹכָה בָּאֵלִים יְיָ]


מִי מְפַלֵּג שֶׁבַע אֲחוּזוֹת / מִי נְתָנָם לִשְׁתֵּי אֲחֻזּוֹת

מִי שָׂם שֵׁשׁ צְלָעוֹת פְּרוּזוֹת / מִי עִטְּרָם הֵיכָל רֹאשׁ מְזוּזוֹת

מִי פִזַּר חָמֵשׁ גְּרוּזוֹת / מִי צֵרְפָם לִהְיוֹת חֲרוּזוֹת

מִי הָאִישׁ הֶחָכָם וְיָבֵן אֶת זֹאת / מִי כָמוֹכָה בָּאֵלִים יְיָ


מִי קָבַע שֵׁשׁ בְּשֵׁם גְּאוֹנוֹ / מִי רִצֵּף בָּהֶם קְצוֹת מְעוֹנוֹ

מִי שֵׁם יוֹד הֵי וָו הֱכִינוֹ / מִי תִּכֵּן בּוֹ רוּם רְצוֹנוֹ

מִי שִׁכְלֵל מַטָּה בְאוֹנוֹ / מִי לָז בְּיוֹד וָו הֵי צְפָנוֹ

מִי הֵקִים כָּל אַפְסֵי אָרֶץ מַה שְּׁמוֹ וְשֶׁם בְּנוֹ / מִי כָמוֹכָה בָּאֵלִים יְיָ.
