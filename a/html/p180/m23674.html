אֲרַנֵּן לְאֵל חַי

אֲרַנֵּן לְאֵל חַי / לְמֵכִין אֳרָחַי / אֲסַדֵּר שְׁבָחַי / לְעוֹטֵה מְלוּכָה

בְּיוֹם דִּין וְזִכְרוֹן / אֶקְרָא בְגָרוֹן / תֵּת הַלֵּל וְרֹן / לִפְנֵי כְבוֹד מְלוּכָה

גַּלֵּה יְקָרָךְ / וְהַרְאֵה הֲדָרָךְ / לְעַם לָךְ עָרַךְ / סֵדֶר מְלוּכָה

דִּגְלָךְ הָרֵם / וְאוֹיְבָךְ הַחֲרֵם / וְלַעַן הַבְרֵם / לְמַכְחִישֵׁי מְלוּכָה

הַרְאֵה פֶלֶא / לַאֲסִיר כֶּלֶא / וּמִבּוֹר הַעֲלֵה / בֵּן הִנְחַלְתּוֹ מְלוּכָה

וּמִיּוֹם הָגְלָם / וּמִיּוֹם נָפְלָם / יְחַכּוּ לְגוֹאֲלָם / אֲדוֹן הַמְּלוּכָה

זְרֵה נָא פִשְׁעָם / וְקָרֵב יִשְׁעָם / וְנַקֵּה זַרְעָם / וְחַדֵּשׁ מְלוּכָה

חִישׁ נָא יֶשַׁע / לְשָׁבֵי פֶשַׁע / וּמֵחֵטְא וְרֶשַׁע / נַקֵּה זֶרַע מְלוּכָה

טַהֵר מַחֲסֶה / לְעַם בָּךְ חוֹסֶה / וּמִיַּד שׁוֹסֶה / חַלֵּץ בְּנֵי מְלוּכָה

יֶשְׁעֲךָ הַרְאֵה / אֶל עַם נִלְאֶה / מֻכֶּה וְנִכְאֶה / מְיַחֵל קֵץ מְלוּכָה

כֹּחֲךָ יִגְדַּל / עֲלֵי רָשׁ וָדָל / וְאַל יְהִי נִבְדָּל / מִיֶּשַׁע מְלוּכָה

לְךָ רָם וְתָלוּל / אֶעֱמֹד בְּפִלּוּל / וְאַשְׁמִיעַ הִלּוּל / וַאֲיַחֵד מְלוּכָה

מֵקִים בְּרִיתוֹ / לְעַם נַחֲלָתוֹ / וּמַכְתִּיר עֲדָתוֹ / בְּכֶתֶר מְלוּכָה

נַחֵם בְּיוֹם זֶה / עַם דַּל וְנִבְזֶה / וְיֶשְׁעֲךָ נֶחֱזֶה / בְּחַדְּשָׁךְ מְלוּכָה

סְמָךְ נָא לְעָטוּף / בְּיַד צַר קָטוּף / בְּכָל עֵת חָטוּף / בְּיַד מַרְגִּיזֵי מְלוּכָה

עוֹנֶה בַצָּר / עֲנֵה נָא לְנֶעְצָר / וְנַפְשׁוֹ תִקְצַר / מִכֹּבֶד עֹל מְלוּכָה

פּוֹדֶה וְקוֹנֶה / שְׁעֵה לִמְעֻנֶּה / בְּחַסְדְּךָ תִפְנֶה / לִמְחַכֵּה מְלוּכָה

צַפֵּה בְעָנְיוֹ / וְהַתֵּר שִׁבְיוֹ / וְהָשֵׁב עֶדְיוֹ / לְקִרְיַת מְלוּכָה

קְרָא נָא דְרוֹר לוֹ / וְהָסֵר סֻבָּלוֹ / וּשְׁבֹר עֻלּוֹ / וְהָקֵם כֶּתֶר מְלוּכָה

רִנְנוֹ תִרְצֶה / וְחָפְשִׁי יֵצֵא / אֲזַי לְךָ יִפְצֶה / בְעֹז הַמְּלוּכָה

שְׁעֵה נָא רַחְשׁוֹ / וְהַעֲבֵר כַּחְשׁוֹ / וְהַרְחֵק מוֹקְשׁוֹ / וְעַטְּרֵהוּ מְלוּכָה

תִּדְלֶה נֶאֱסָר / אֲשֶׁר כֹּל יֶחְסַר / מִיּוֹם מֶנְהוּ סָר / כֶּתֶר מְלוּכָה

תִּזְכֹּר בְּרַחֲמִים / לְעַם נִכְלָמִים / פְּנֵי כָל עַמִּים / כּוֹנֵן קִרְיַת מְלוּכָה.
