[שירי זרח]

[303]: [חזון תחכמוני הזקן]

אָמַר תַּחְכְּמֹנִי הַמֻּפְלֶא:/"אֲנִי טֶרֶם אֲכַלֶּה / לְדַבֵּר אֶל לִבִּי – וְרַעְיוֹנַי רוֹהִים/ וָאֶרְאֶה מַרְאוֹת אֱלֹהִים:/ וְהִנֵּה הַשֶּׁמֶשׁ וְהַיָּרֵח קָדְרוּ אוֹרָם,/ חָשָׁךְ מִשְּׁחוֹר תָּאֳרָם, וְהַכֹּוכָבִים מִמְּסִלּוֹתָם / שַׂק הוּשַׁת כְּסוּתָם, / וְהַגַּלְגַּלִּים הִתְגַּלְגָּלוּ / וְהַגְּבָעוֹת הִתְקַלְקָלוּ וְהָאֲרָצוֹת הִתְחַלְחָלוּ, / וְאָדָם וּבְהֵמָה חָרָדוּ / וְכָל עוֹף הַשָּׁמַיִם נָדָדוּ. / וָאֹמַר בְּלִבִּי; עָזַב אֱלֹהִים אֶת הָאֲדָמָה / וְהָאָרֶץ תִּשָּׁאֶה שְׁמָמָה, / כָּל הָאָדָם כּוֹזֵב / וְהָאָרֶץ תֵּעָזֵב, / וָאָקוּם כְּמִתְאוֹנֵן / וָאֶעֱמֹד לְהִתְבּוֹנֵן / הֶהָיְתָה זֹאת לְכָלָה וָחָרֶץ – / וְהִנֵּה אוֹפַן אֶחָד בָּאָרֶץ, / קוֹל קוֹרֵא / וּמִי כָמוֹהוּ מוֹרֶה / "בֶּן-אָדָם! עֲמֹד עַל רַגְלֶיךָ / וַאֲדַבְּרָה אֵלֶיךָ. / שָׁמַעְתִּי אֶת תְּלֻנּוֹתֶיךָ, / רָאִיתִי אֶת דִּמְעוֹתֶיךָ, / הָסֵר כַּעַס מִלִּבֶּךָ / וּבִעַרְתָּ הַיָּגוֹן מִקִּרְבֶּךָ, / וְאַל יַבְהִילְךָ זְמָן הַמַּרְבֶּה עַוְלָה / וְכָל פֶּה דוֹבֵר נְבָלָה: / כִּי כָל הָאֲנָשִׁים נָבְקָה עֲצָתָם / טַחוּ מֵרְאוֹת עֵינֵיהֶם מֵהַשְׂכִּיל לִבּוֹתָם. / וְאִם גָּדַל פִּשְׁעָם וְחוֹבָם – / נְשֵׁיהֶם הִטּוּ לְבָבָם! / כִּי מִי יַתִּיר עֲבוֹתוֹת / אֲשֶׁר הַנָּשִׁים עוֹשׂוֹת? / דֶּרֶךְ נְכוֹחָה יְגָרֵשׁוּ / וְאֶת כָּל הַיְשָׁרָה יְעַקֵּשׁוּ, / חֶבְרָתָן וְאַהֲבָתָן כְּרֶגַע, / עַל פִּיהֶן יִהְיֶה כָל רִיב וְכָל נֶגַע. / כַּמָּה אֲנָשִׁים / כְּגֹבַהּ בְּרוֹשִׁים / הָיוּ לְחֶרְפָּה וּלְבוּשָׁה / עַל אוֹדוֹת הָאִשָּׁה! / בִּי נִשְׁבַּעְתִּי: לֹא הַנָּחָשׁ הֶעֱרִים עַל חַוָּה / וְלֹא עָלֶיהָ צִוָּה – / הִיא שָׂמָה אִישָׁהּ נוֹדֵד מִמְּקוֹמוֹ / וְנָחָשׁ עָפָר לַחְמוֹ, / וְאִם בָּנֶיהָ פִּשְׁעָם יְדַכְּאֵם – / הִיא הָיְתָה הָאֵם! / וַהֲלֹא עַל הַתְּרָפִים / יַעֲקֹב וּבָנָיו נִרְדָּפִים / וְעַל עוֹרוֹת גְּדָיֵי הָעִזִּים / אַתֶּם לְחוּצִים וְנִבְזִים. / וַהֲלֹא עַל דִּינָה / הָיְתָה לְאָבִיהָ וּלְאַחֶיהָ שִׂטְנָה, וְלוּלֵי הַנְּזָמִים – / לֹא נַעֲשָׂה הָעֵגֶל לִקְסֹם קְסָמִים. / וְעַל מִי נִפְתּוּ יַלְדֵי אֱמוּנַי / וַתְּהִי הַמַּגֵּפָה בַּעֲדַת אֲדֹנָי? / וּמֵאֲשֶׁר גָּבְרוּ עַל צְלָפְחָד דִּבְרֵי עֲוֹנוֹת / לֹא הָיוּ לוֹ בָנִים כִּי אִם בָּנוֹת. / וְאַבְרָם עַל יֹשֶׁר פָּעֳלוֹ / בַּת לֹא הָיְתָה לוֹ, / וּבָזֹאת תּוֹלֶה אֶרֶץ כְּאֶשְׁכּוֹל / בֵּרַךְ אֶת אַבְרָהָם בַּכֹּל. / וְהוֹסִיף לְאִיּוֹב מִשְׁנֶה עַל כָּל הַטּוֹבוֹת – / מִלְּבַד הַנְּקֵבוֹת, / וְצִוָּה לַמָּשִׁיחַ עַל עַם קְדוֹשִׁים: / לֹא יַרְבֶּה לוֹ נָשִׁים. / וְעַל מִי נִגְרַע שֵׁבֶט בִּנְיָמִין / צָפוֹן וְיָמִין? / וְעַל מִי כָשַׁל כֹּחַ / שִׁמְשׁוֹן בֶּן-מָנוֹחַ / וַיִּרְדְּפוּהוּ עַד הַשְּׁבָרִים / וַיְהִי טוֹחֵן בְּבֵית הָאֲסוּרִים? / וּבְעֵת עִוָּה עַמִּי נְתִיבוֹ – / נָשִׁים מָשְׁלוּ בוֹ!

וַיַּעַן הַמַּלְאָךְ הַדּוֹבֵר בִּי וַיֹּאמַר:

רְאֵה שֵׂכֶל פַּרְעֹה וְהִתְבּוֹנֵן דֵּעוֹ

בְּעֵת הִפְלִיא עֵצוֹת לְעַנּוֹת עַם בִּנוֹת

אֲשֶׁר עִנָּה אוֹתָם וְהִגְדִּיל תּוּגָתָם

וְהֵמִית הַבָּנִים – וְחִיָּה הַבָּנוֹת.

וּבַעֲבוּר זֹאת הֶעֱמַדְתִּיךָ לְגַלּוֹת אֵלֶיךָ רָז כָּל נָבִיא וְחוֹזֶה, / כִּי אוֹתְךָ רָאִיתִי צַדִּיק לְפָנַי בַּדּוֹר הַזֶּה, / וְנִגְלֹה נִגְלֵיתִי לְמַעֲנֶךָ / וּלְמַעַן תְּסַפֵּר בְּאָזְנֵי בְנֶךָ.

וַיְכַל תַּחְכְּמֹנִי לְצַוּוֹת אֶת בְּנוֹ וְאֶת כָּל אַנְשֵׁי שְׁלוֹמָיו / וַיִּגְוַע וַיֵּאָסֶף אֶל עַמָּיו. / וַיִּסְפְּדוּ לוֹ כָל יוֹדְעָיו לְפָנִים / בְּאֵבֶל גָּדוֹל וּבְקִינִים / וְכָבְדָה עָלָיו הָאֲנָקָה / וְהִקִּיפָה הַזְּעָקָה. / וַיִּקְבֹּר אוֹתוֹ בְּנוֹ זֶרַח / בִּגְבוּל נַחֲלָתוֹ בְּתִמְנַת סֶרַח / אֲשֶׁר מִצְּפוֹן לְהַר גַּעַשׁ, / שְׁנָתַיִם אַחַר הָרַעַש. / בֶּן מֵאָה שָׁנָה הָיָה תַחְכְּמֹנִי בְּהֵאָסֵף רוּחוֹ, / לֹא כָהֲתָה עֵינוֹ וְלֹא נָס לֵחוֹ, / מִלְּבַד עֶשְׂרִים שָׁנָה הָיָה נָשׂוּי בִּימֵי עֲלוּמָיו – / עַל כֵּן לֹא בָאוּ בְמִסְפַּר יָמָיו! / כִּי בְקַחְתּוֹ אִשָּׁה הָיָה צָעִיר / אֲשֶׁר לֹא יָדַע לָלֶכֶת אֶל הָעִיר, / וְכַאֲשֶׁר גָּדַל וְרָאָה אֶת עֶרְוָתָה / נָתַן לָהּ אֶת כָּל אֲשֶׁר לוֹ וַיְגָרֶשׁ אוֹתָהּ / וְלֹא יָסְפָה שׁוּב אֵלָיו עוֹד / וְלֹא יָדַע בַּפַּשׁ מְאֹד.

וַיִּכְתֹּב זֶרַח אֶת כָּל דִּבְרֵי אָבִיו וְאֶת כָּל אֲשֶׁר אָמַר / וַיִּתֵּן אוֹתָם בְּמִשְׁמָר, / מִן הַיּוֹם הַהוּא וָהָלְאָה כָּל הַמַּזְכִּיר לוֹ אִשָּׁה – עָלָיו יִלְעַג וְיִשְׂחַק / וְגָעַר בּוֹ וְנָס מִמֶּרְחָק, / וְתֹאחֲזֵהוּ רְעָדָה וְחַלְחָלָה / כִּי יִבְעַל בָּחוּר בְּתוּלָה. / וַיִּבְחַר לוֹ שְׁלֹשָׁה רֵעִים / נְדִיבִים וְשׁוֹעִים, / כִּי רָדְפוּ אַחֲרֵי הַשֵּׂכֶל מִימֵי נְעוּרִים / לָקַחַת מוּסַר הַשְׂכֵּל, צֶדֶק וּמִשְׁפָּט וּמֵישָׁרִים./ וְאֵלֶּה שְׁמוֹתָם / לְמִשְׁפְּחוֹתָם לְבֵית אֲבוֹתָם: / נְתַנְאֵל בֶּן-מַהֲלַלְאֵל, אֲבִינֹעַם בֶּן- אֲבִינָדָב, אֲחִיטוּב בֶּן-אֲבִישׁוּעַ, וַיִּלָּוֶה אֲלֵיהֶם / וַיִּתְחַבֵּר עִמָּהֶם. / וַיֹּאמְרוּ לוֹ “הִנֵּה הַמָּקוֹם צַר מִמֶּנוּ / וְהַנָּשִׁים יוֹשְׁבוֹת בְּקִרְבֵּנוּ, / וַאֲנחְנוּ פֹּה יְרֵאִים מִתַּרְמִיתָן, / פֶּן נִלָּכֵד בִּשְׁחִיתוֹתָן. / וְעַתָּה נִבְחֲרָה לָּנוּ מָקוֹם פֶּן נֶאְשָׁם – / אֶרֶץ אֲשֶׁר לֹא עָבַר בָּהּ אִישׁ וְלֹא יָשַׁב אָדָם שָׁם”. / וַיֹּאמֶר אֲלֵיהֶם “חֶפְצִי כְּחֶפְצְכֶם / וּרְצוֹנִי כִרְצוֹנְכֶם, / פְּנוּ וּסְעוּ לָכֶם!”

מִשָּׁם הָלְכוּ הָלוֹךְ וְנָסוֹעַ / לִמְצֹא לְנַפְשָׁם מַרְגּוֹעַ, / וְעָבְרוּ מַעְבָּרוֹת / וְחָלְפוּ נְהָרוֹת / וְדָרְכוּ מְסִלּוֹת, / אֳרָחוֹת עֲקַלְקַלּוֹת, / עַד אֲשֶׁר בָּאוּ אֶל נַחַל הַבְּשׂוֹר, / מָקוֹם אֲשֶׁר אֵין שָׁם מַחְסוֹר: / עַל שְׂפָתוֹ כָּל מִינֵי מְגָדִים, כְּפָרִים עִם נְרָדִים, / וְכָל עֲצֵי פְרִי וַעֲצֵי עֵדֶן עָלָיו שְׁתוּלִים / מַיְמִינִים וּמַשְׂמִאלִים, / וְכָל אֵלֶּה עֲבוֹת וּפֹארוֹת בַּבַּדִּים תְּשַׁלַּחְנָה / וַיֵּחָצוּ הֵנָּה וָהֵנָּה. / וְכָל עֹפֶר וּצְבִי יִתְלוֹנָן / תַּחַת כָּל עֵץ רַעֲנָן, / וְצִפּוֹר כָּל כָּנָף יִסְתּוֹפֵף בְּעַנְפֵיהֶם / וְעַל עֲלֵיהֶם יִפְרְשׂוּ כַנְפֵיהֶם, / פַּעַם יֶהְגּוּ וּפַעַם עִלְּגִים, / הַמְצַפְצְפִים וְהַמַּהְגִּים./ וַיִּרְאוּ הַמָּקוֹם כִּי טוֹב וְכִי שָׁם כָּל הַנְּעִימוֹת / וַיִּמְצָא חֵן לִפְנֵיהֶם מִכָּל הַמְּקוֹמוֹת./ וַיִּשְׂאוּ מְשָׁלָם וַיֹּאמְרוּ:

רְאוּ, מַה נָּעֲמוּ סִיסִים בְּצִצִּים / יְצַפְצֵפוּ וְכַנְפֵיהֶם פְּרוּשִׂים,

וְגֻלּוֹת יוֹרְדוֹת עַל הַחֲצָצִים / וְהָרוּחַ תְּרַחֵף בֵּין הֲדַסִּים,

וְכָל עֵץ וְכָל אִישׁ שָׂשׂ בְּצִלּוֹ, / וּפֹארוֹתָיו בְּפֵרוֹתָיו עֲמוּסִים!

וַיֹּאמְרוּ “עַד כֹּה דָרְכָה רַגְלֵנוּ, / כִּי בָאָה נַחֲלָתֵנוּ אֵלֵינוּ, / פֹּה תַחֲנוֹתֵנוּ / כִּי הִיא מְקוֹם מְנוּחָתֵנוּ”. / וַיֵּשְׁבוּ בֵין הַהֲדַסִּים / שְׂמֵחִים נֶעֳלָסִים / פַּעַם יוֹרוּ חֻקּוֹת, פַּעַם יוֹרוּ מִשְׁפָּטִים, / פַּעַם לֶאֳכֹל וְלִשְׁתּוֹת נוֹטִים, / פַּעַם מְשַׂחֲקִים, פַּעַם מַלְעִיגִים, / כִּי כֵן מִשְׁפַּט יְמֵי הַתַּעֲנוּגִים. / חֹדֶשׁ בַּשָּׁנָה יַעֲלֶה זֶרַח בְּכָל שְׁכֵנָיו / לְהוֹכִיחַ שְׁלוּמֵי אֱמוּנָיו, / וּבְכָל מָקוֹם אֲשֶׁר יִפְנֶה / כֹּה יַעֲנֶה: / “גּוּרוּ לָכֶם מִפְּנֵי חֶרֶב נְטוּשָׁה / וְאַל תִּגְּשׁוּ אֶל אִשָּׁה! / בַּקְּשׁוּ רְוָחָה / וּמִצְאוּ הֲנָחָה, / הַרְחִיקוּ בַעֲלֵי נְזָמִים – / לְמַעַן תַּאֲרִיכוּ יָמִים, / לְהַבְאִישׁ רֵיחָם – / תִּפְאֶרֶת בַּחוּרִים כֹּחָם: / כִּי לֹא יַעְטוּ כְלִמָּה וּבוּשָׁה, / אִם נִשְׁמְרוּ הַנְּעָרִים אַךְ מֵאִשָּׁה. / רֹע הַנִּשׂוּי לָעַד לֹא יִבְלֶה / מִנֶּפֶשׁ וְעַד בָּשָׂר יְכַלֶּה. / כָּל הַנּוֹשֵׂא – / בַּחֹשֶׁךְ שְׁמוֹ יְכֻסֶּה, / כְּאֵבוֹ יִגְדַּל וּכְבוֹדוֹ יִדַּל, / הַשּׁוֹמֵעַ יִשְׁמַע וְהֶחָדֵל יֶחְדַּל. / כָּל בַּעַל-אִשָּׁה גָּרֵשׁ יְגָרֵשׁ / פֶּן יִוָּרֵשׁ, / וְהַמַּשְׂכִּיל יָבִין וְיָסֹר, / וּדְבַר שְׂפָתַיִם אַךְ לְמַחְסוֹר”.

[305]: [כינוס נשים]

כֹּה יַעֲשֶׂה זֶרַח, וְכֹה רַעְיוֹנָיו פְּתָאיִם מַחְכִּימִים / וְכֹה מִשְׁפָּטוֹ כָּל הַיָּמִים. / הַמַּשְׂכִּילִים שָׁמְרוּ תוֹרָתוֹ לֹא עֲזָבוּהָ כָּל יְמֵיהֶם / וְרַבִּים מֵעַמֵּי הָאָרֶץ מִתְיַהֲדִים כִּי נָפַל פָּחַד זֶרַח עֲלֵיהֶם, / וְכָל מְדִינָה וּמְדִינָה מְקוֹם אֲשֶׁר דְּבַר זֶרַח וְדָתוֹ מַגִּיעַ – / לַעֲנָה וָרוֹשׁ לְכָל הַנָּשִׁים מַשְׂבִּיעַ, / וַתִּוָּעַדְנָה כֻלָּן קְטַנּוֹת וּגְדוֹלוֹת, / אַלְמָנוֹת וּבְתוּלוֹת, / רוֹגְזוֹת וְחוֹלוֹת, / לָקַחַת עֵצָה לְהִתְעוֹלֵל עֲלִילוֹת, / וְצִירִים אֲחָזוּן וְחַלְחָלָה / וְנַפְשָׁם מְאֹד נִבְהֲלָה. / וַיִּתְגּוֹדֲדוּ כְמִשְׁפָּטָן, / כִּי הוֹבִישׁ מַבָּטָן, / וְגָדְלָה צַעֲקָתָן וְאֶנְקָתָן / וְעָצְמָה אַנְחָתָן / וְרַבָּה מְהוּמָתָן / וַתַּעַל שַׁוְעָתָן / וַתֹּאמַרְנָה “מַה לַּעֲשׂוֹת בָּאִישׁ אֲשֶׁר שֹׁד וָשֶׁבֶר הִנְחִילָנוּ / אֲשֶׁר כִּלָּנוּ וַאֲשֶׁר דִּמָּה לָנוּ? / הִנֵּה נוֹדְעָה חֶרְפָּתֵנוּ וּמַעֲלָלֵינוּ – / וְאֵין אִישׁ בָּאָרֶץ לָבוֹא עָלֵינוּ, / כִּי מִי הוּא אֲשֶׁר עָרַב לִבּוֹ לָגַעַת אֵלֵינוּ / וַיִּתֵּן אוֹתָנוּ לַחֲרָפוֹת? / תִּתְעַלַּפְנָה הַבְּתוּלוּת הַיָּפוֹת, / כָּל בַּת אָבַד פְּרִי מַטָּעָהּ, / בְּתוּלָה – וְאִישׁ לֹא יְדָעָהּ. / הִנֵּה זֶרַח שָׂמָנוּ לְמוֹרָא וָפַחַד / וְהֶחֱזִיקוּ שֶׁבַע נָשִׁים בְּאִישׁ אֶחָד! / דַּעְנָה וּרְאֶינָה כִּי רָעָתֵנוּ רַבָּה: / כָּל בְּתוּלָה סְגוּרָה וּמְסֻגֶּרֶת אֵין יוֹצֵא וְאֵין בָּא. / לָמָּה נִשְׁכְּבָה בְּבָשְׁתֵּנוּ / וּתְכַסֵּנוּ כְלִמָּתֵנוּ? / קֹמְנָה וְהִתְיָעַצְנָה – / וְהַנָּשִׁים תִּשְׁכַּבְנָה, / וְנַעֲמֹד עַל זֶה הַשֶּׁבֶר: / נְקֵבָה תְּסוֹבֵב גֶּבֶר! / וְאַל יְהִי בְעֵינֵכֶם הַדָּבָר הַזֶּה נָקֵל וְנִבְזֶה – / מִשְׁפַּט מָוֶת לָאִישׁ הַזֶּה, / סָקוֹל יִסָּקֵל וְאַל יִגָּאֵל, / כִּי הוֹצִיא שֵׁם רַע עַל בְּתוּלַת יִשְׂרָאֵל. / וְאַיֵּה בַעֲלֵי מִרְמָה / אֲשֵׁר בִּידֵיהֶם זִמָּה? / וְאַיֵּה תַרְמִית הַנָּשִׁים וְנִכְלֵיהֶן / וְאַיֵּה מוֹעֲצוֹתֵיהֶן וְתַחְבּוּלוֹתֵיהֶן? / וְאַיֵּה עָרְמָתָן / וְאַיֵּה חָכְמָתָן? / וְאַיֵּה מִרְמָתָן / וְאַיֵּה תַרְמִיתָן? וּמַדּוּעַ אָבְדָה חָכְמָתֵנוּ / וְתוּשִׁיָּה נִדְּחָה מִמֶּנוּ? / עַל אִישׁ אֶחָד נֵרֵנוּ יִכְבֶּה / וְחוֹטֵא אֶחָד יְאַבֵּד טוֹבָה הַרְבֵּה / הָבָה נִתְחַכְּמָה לוֹ וּמִגּוֹי נַכְחִידֶנּוּ / וְנִקְחָה נִקְמָתֵנוּ מִמֶּנוּ, / וְחָכְמַת זֶרַח נְאַבֵּד מְהֵרָה – / וְאִישׁ וְאָבִיו יֵלְכוּ אֶל הַנַּעֲרָה!”

וַתָּקָם מִבֵּינֵיהֶם אִשָּׁה חֲכָמָה / בַּעֲלַת אוֹב וּמִרְמָה, / בְּרֹב כְּשָׁפֶיהָ וְנַאֲפוּפֶיהָ / נְהָרוֹת לֹא יִשְׁטְפוּ רְשָׁפֶיהָ, / וּשְׁמָהּ כָּזְבִּי בַת-יְרֵשָׁה, / כִּי אָחֲזָה הַתַּרְמִית לְמוֹרָשָׁה, / וְלָהּ אִישׁ זָקֵן וּשְׁמוֹ שֶׁקֶר בֶּן-חֵפֶר / וְרָעָתוֹ לֹא יְכַלְכֵּל סֵפֶר. / וַתֹּאמֶר לָהֶן “שְׁמַעְנָה, נָשִׁים שַׁאֲנַנּוֹת, / וְאַל תַּגְבֵּרְנָה לְשׁוֹנוֹת! / כִּי שְׁחִין אֲסוֹנִים / לֹא יְחַיֶּנוּ דְבֶלֶת תְּאֵנִים, / וְנָבִיא עֵצוֹת מֵרָחוֹק / וְתַחְבּוּלוֹת לִבְלִי חֹק. / כָּשְׁלוּ גִבּוֹרִים / וְנָמַסּוּ הֶהָרִים – / בִּשְׁתַּיִם קַרְנֵי שִׂכְלוֹ אֲגַדַּע / וַעֲצָתוֹ אֲבַלַּע, / לְבַל יִרְאֶה בְעֵינָיו / וְיִשְׁמַע בְּאָזְנָיו. / צְאֶינָה וּרְאֶינָה בְכָל הַמְּקוֹמוֹת / וּבַקְּשׁוּ בְכָל הָעֲלָמוֹת / נַעֲרָה תְמִימָה כְּלִילַת יֹפִי / אֲשֶׁר אֵין בָּהּ דֹּפִי, / יָפָה לְכָל מוֹצְאֶיהָ / נוֹשֵׂאת חֵן בְּעֵינֵי כָּל רוֹאֶיהָ / בָּרָה כַּחַמָּה יָפָה כְתִרְצָה / כַּשֶּׁמֶשׁ מְצִיצָה, / מְקַבֶּלֶת כָּל רוֹאֶיהָ בְּנֶפֶשׁ חֲפֵצָה, / בַּעֲלַת מוּסָר וְעֵצָה, / מִתְבּוֹנֶנֶת מָשָׁל וּמְלִיצָה, / נְגִידִים תְּדַבֵּר / וְשִׁירִים תְּחַבֵּר, / אֲמָרֶיהָ כְּנֹפֶת מְתוּקִים / כִּרְאִי מוּצָק חֲזָקִים, / מִצָפוֹן תָּעִיר צְפוּנָיו / וּמִמֶּרְחָק תָּבִיא עִנְיָנָיו / וּבְכוּר הַשֵּׂכֶל תִּצְרְפֶנּוּ – / גֶּחָלִים בָּעֲרוּ מִמֶּנּוּ, / יוֹדַעַת נַגֵּן עַל כִּנּוֹר וְעַל נֵבֶל, / מְעוֹרֶרֶת הַשִּׂמְחָה וְהָאֵבֶל. / בַּנַּעֲרָה הַזֹּאת יִמַּסּוּ אֲסוּרָיו, / כִּי יֵצֶר לֵב הָאָדָם רַע מִנְּעוּרָיו. / וְאִם לֹא יָשׁוּב כַּאֲשֶׁר יִכּוֹן וְיֵאוֹת, / לַמָּוֶת תּוֹצָאוֹת, / הַזָּהָב יְפַתֶּנוּ / וְלִבְאֵר שַׁחַת יוֹרִידֶנּוּ, / הוּא יְנַתֵּק מוֹסְרוֹתָיו / וְהוּא יָפֵר מַחְשְׁבוֹתָיו. / הוּא מוֹרִיד הַמְּלָכִים מִכִּסְאוֹתָם / וְיוֹצִיאֵם מֵחַצְרֵיהֶם וּמִטִּירוֹתָם, / הוּא מַחְכִּים הַפְּתָאִים / וְהוּא מְכַסֶּה עַל הַחֲטָאִים, / וְהוּא מַשְׁפִּיל בְּנֵי עֲנָקִים / וּמְסַלֵּף דִּבְרֵי צַדִּיקִים, / וְהוּא מְיַשֵּׁר הַמַּעֲקַשִּׁים / וְהוּא מִשְׁעָן לְיוּקָשִׁים / וְעֵינַיִם לָעִוְרִים וְאָזְנַיִם לַחֵרְשִׁים, / הַמְּשַׂמֵּחַ אֱלֹהִים וַאֲנָשִׁים, / וְהוּא לְכָל אֶחָיו / צִיר נֶאֱמָן לְשׁוֹלְחָיו”.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

שַׂחְתִּי “לְמִי כָבוֹד, לְמִי יָקָר?” / וָאֶשְׁמְעָה אוֹמְרִים "לְמִי זָהָב –

בּוֹ הַכְּסִיל יָעֹז וְיִתְרוֹמָם, / בּוֹ כָל אֱנוֹשׁ יַשִּׂיג אֲשֶׁר יֶאְהַב“.

וְעַתָּה אַחְיוֹתַי וְרַעְיוֹתַי, / אֵלֶּה מוֹעֲצוֹתַי וְאֵלֶּה מְזִמּוֹתַי: / נָשִׂים הָעַלְמָה לְנֶגְדּוֹ וְהַזָּהָב לְמוּלוֹ / אוּלַי יְפֻתֶּה וְנוּכְלָה לוֹ”. / וַיֹּאמְרוּ אֵלֶיהָ “עָלֵינוּ לָתֶת לָךְ אֶת כָּל אֲשֶׁר תַּחְפֹּצִי, / אוּלַי תּוּכְלִי אוּלַי תַּעֲרֹצִי”.

וַתִּיקַד אֵשׁ הָאַהֲבָה בְקִרְבּוֹ / וּשְׁבִיב הַחֵשֶׁק בּוֹעֵר בּוֹ, / וַיִּקְרָא לְשֶׁקֶר וּלְכָזְבִּי הַמִּרְשַׁעַת – / וְהוּא בִּבְלִי דַעַת! – / וַיֹּאמֶר לָהֶם “הַרְבּוּ מְאֹד מֹהַר וּמַתָּן עָלַי – / וְאֶעֱשֶׂה כַאֲשֶׁר תֹּאמְרוּ אֵלַי! / וְהַשְׁמִיעוּ זֹאת בְּכָל אִיֵּי אֱלִישָׁה / וּתְנוּ לִי אֶת הַנַּעֲרָה הַזֹּאת לְאִשָּׁה”. / וַיְהִי כִשְׁמֹעַ שֶׁקֶר אֶת דְּבָרָיו, / נָעֲמוּ אָזְנָיו וְצָהֲלוּ פָנָיו / וַיֵּאוֹרוּ עֵינָיו / וְכָל תּוּגוֹתָיו נִשְׁכָּחוּ, / וַיִּקֹד אַרְצָה וַיִּשְׁתַּחוּ. וַתְּמַהֵר כָּזְבִּי לְבַשֵּׂר אֶת הָעֲלָמוֹת הַבְּתוּלוֹת / בְּגִבְעַת הָעֲרָלוֹת, / וַתֹּאמֶר לָהֶן “הַיְדַעְתֶּן זֶרַח הַמּוֹכִיחַ / אֲשֶׁר לְבָבוֹ עָלֵינוּ הִקְשִׁיחַ – / הִסְגִּירוֹ יְיָ בְּיַד אִשָּׁה / וְהִנֵּה הוּא לְחֶרְפָּה וּלְבוּשָׁה! / נָתַן אֱלֹהִים בְּיָדֵינוּ / אֶת זֶרַח אוֹיְבֵנוּ, / פָּרַשְׂתִּי עָלָיו רִשְׁתִּי / וְנִלְכַּד בִּמְצוּדָתִי, / נִפְתָּה לִבּוֹ עַל אַיָּלָה שְׁלוּחָה / וְהַיּוֹם יֵרֵד כְּצֹאן לְטִבְחָה. / לְעֵת עֶרֶב אַחֲלִיפֶנָּה בְאַחֶרֶת – / וְתַחְתֶּיהָ תַּעֲמֹד הַבַּהֶרֶת!” / אָז תִּשְׂמַחְנָה הַנְּעָרוֹת – / הַשְּׁפָחוֹת וְהַגְּבִירוֹת / עַל דְּבַר הַבְּשׂוֹרוֹת: / זֹאת מְהַלֶּלֶת וּמְשַׂחֶקֶת, / וְזֹאת עוֹבְרָה חוֹמֶקֶת, / וְזֹאת לְכָזְבִּי חוֹבֶקֶת, / וְזֹאת יָדֶיהָ מְנַשֶּׁקֶת, / וְזֹאת לְרַגְלָהּ סוֹגֶדֶת, / וְזֹאת מֵעֲלֹז רוֹקֶדֶת / – וְזֹאת עַרְשָׂהּ רוֹבֶדֶת. / וַיְּתְּנוּ לָהּ עֲשָׂרָה זָהָב וַחֲגוֹרָה / הַבְּכִירָה וְהַצְּעִירָה, / עַל כֵּן קָרָא לַמָּקוֹם הַהוּא נַחַל הַבְּשׂוֹר / עַל הַבְּשׂוֹרָה שֶׁנִּתְבַּשֵּׂר בְּאוֹתוֹ מִישׁוֹר.

וַיֶאֱסֹף הָאִישׁ הַזָּקֵן אֶת כָּל אַנְשֵׁי הַמָּקוֹם לְרִבּוֹא וְרִבּוֹתַיִם / וְסֻגְּרוּ דְלָתַיִם, / וַיָּבוֹאוּ הָאֲנָשִׁים / עַל הַנָּשִׁים, / וַיַּעַשׂ זֶרַח מִשְׁתֶּה גָדוֹל לָרָחוֹק וְלָקָרוֹב / וַיִּזְבַּח שׁוֹר וּמְרִיא וְצֹאן לָרֹב. / וַיַּעַשׂ שָׁם שִׂמְחָה גְדוֹלָה – / כָּל שִׁבְעַת הַיָּמִים שְׁתִיָּה וַאֲכִילָה / בְּתֻפִּים וּבִנְבָלִים / וּבִמְנַעַנְעִים וּבְצֶלְצְלִים. / וְאֲחַרֵי כֵן קָם הָאִישׁ הַזָּקֵן מִבֵּינֵיהֶם / וַיִּקַּח סֵפֶר הַבְּרִית וַיִּקְרָא בְאָזְנֵיהֶם:

בָּרְבִיעִי בְשַׁבָּת / הָיְתָה לָאִישׁ הַזֶּה סִבַּת / וּמְשׂוֹשׂ לִבּוֹ נִשְׁבַּת / לְגַלּוֹת עֶרְוָתוֹ / וּלְהַרְאוֹת חֶרְפָּתוֹ. / בְּאַרְבַּעַת אֲלָפִים וּתְשַׁע מֵאוֹת וְשִׁבְעִים וְשִׁבְעָה / הִבְדִּילוֹ יְיָ לְרָעָה. / בִּשְׁלֹשָה עָשָׂר לְחֹדֶשׁ אֲדָר / לֹא תֹאַר לוֹ וְלֹא הָדָר, / אֵיךְ זֶרַח הֶחָתָן / אֲשֶׁר כְּבוֹדוֹ לְאַחֵר נָתַן / אָמַר לָהּ לְמָרַת-רוּחַ רִצְפָּה בַת-אַיָּה מִבְּנוֹת פּוּתִיאֵל / מִטֻּמְאוֹת בְּנֵי יִשְׂרָאֵל: / יָמִים רַבִּים תִּהְיֶה-לִּי לַעֲבֹד אֶת עֲבוֹדַת בְּנֵי יִשְׂרָאֵל / וְכָל מַחְסוֹרֵךְ עָלַי אֵין לְהוֹחִיל עַד בּוֹשׁ, / לֶחֶם לֶאֱכֹל וּבֶגֶד לִלְבֹּשׁ, / וְכִלְכַּלְתִּי אוֹתָךְ כְּחֹק כָּל אִישׁ וּמִפְעָלוֹ / וְאָנֹכִי אַעֲלֵךְ גַּם עָלֹה. / וִיהִיבְנָא לִיכִי / מֹהַר בְּתוּלַיְכִי / מֵאָה עֲרָלוֹת / כְּמֹהַר הַבְּתוּלוֹת, / וַיִּיטַב הַדָּבָר בְּעֵינֵי הַקְּדֵשָׁה / וַתְּהִי לוֹ לְאִשָּׁה. / וּצְבֵי הוּא וְאוֹסִיף לָהּ מִדִּילֵהּ: נִקְיוֹן שִׁנַּיִם / וּפִיק בִּרְכַּיִם / וְחַלְחָלָה בְּכָל מָתְנַיִם. / וְעוֹד רָצָה וְנָתַן לָהּ בְּמַתָּנָה / אֶת אַרְבַּעַת אֵלֶּה לְמָנָה: / תַּחַת הַמְּעִילִים – / נְגָעִים גְּדוֹלִים, / וְתַחַת הַטַּבָּעוֹת – / צָרוֹת רַבּוֹת וְרָעוֹת, / וְתַחַת פְּתִיגִיל – מַחֲגֹרֶת שַׂק / וְתַחַת בֹּשֶׂם – מַק. / וְדָא נְדוּנְיָא דְּהַנְעַלְתְּ לֵהּ הַכַּלָּה: / חֶרְפָּה וְשַׁמָּה וְשַׁעֲרוּרָה וְכָלָה. / נִמְצָא סַךְ תּוֹסֶפֶת כְּתֻבָּה וּמַתָּנָה וּנְדוּנְיָא – / כְּוִיָּה תַּחַת כְּוִיָּה / וּפֶצַע וְחַבּוּרָה וּמַכָּה טְרִיָּה!

וְכָך קִבֵּל זֶרַח עַל עַצְמוֹ / שֶׁלֹּא יִהְיֶה אֱלֹהָיו עִמּוֹ, / בְּטֶרֶם יְמַלֵּא שָׁנָה / יִהְיֶה לְמָשָׁל וְלִשְׁנִינָה. / וַיִּקְנוּ מִיַּד זֶרַח קִנְיָן שָׁלֵם, / כִּי יִהְיֶה כְּחֵרֵשׁ לֹא יִשְׁמַע וּכְאִלֵּם, / וַיָּעִידוּ עֵדִים נֶאֱמָנִים וַיִּכְתְּבוּ בַסֵּפֶר: / בֶּן-טָבְאֵל עֵיפָה וָעֵפֶר, / וְנַעַל בֶּן-עֶבֶד וְעַבְדּוֹן, / וּבִלְהָן וְזַעֲוָן וְאֵסַרְחַדּוֹן.

וּבַיוֹם הָרְבִיעִי נִקְהֲלוּ אֶל עֵמֶק הַבְּרָכָה, / כִּי שָׁם בֵּרְכוּ אֶת זֶרַח – עַל כֵּן קָרָא שֵׁם הַמָּקוֹם עֵמֶק הַבְּרָכָה. / וַיְהִי כָל הָעָם נָדוֹן / “הוֹי אָדוֹן / וְהוֹי הוֹדוֹ! / עַתָּה יִדַּל כְּבוֹדוֹ, / בָּאָה פְקֻדָּתוֹ / וְנָבְקָה עֲצָתוֹ. / אַיֵּה כָל נִפְלָאוֹתָיו / וְאַיֵּה כָל תּוֹכְחוֹתָיו / וְאַיֵּה מוֹעֲצוֹתָיו / וְאַיֵּה מַחְשְׁבוֹתָיו? / הֲזֶה הָאִישׁ מַרְעִיד הָאֲנָשִׁים / וּמַרְעִישׁ הַנָּשִׁים? / הַיּוֹם כָּלוּ זְמָמָיו / וְנִכְרַת מֵעַמָּיו!”

וַתִּקַּח כָּזְבִּי אֶת הַנַּעֲרָה / וַתִּתֵּן תְּמוּרָתָהּ אִשָּׁה מְצֵרָה / כָּעוֹרֵב שְׁחֹרָה, / שְׂפָתֶיהָ כְּנֹאד עַל נֹאד נְפוּחִים / אֲשֶׁר כָּל רוֹאֶיהָ נֶאֱנָחִים. / וּשְעַר בְּשָׂרָהּ כְּקוֹצִים כְּסוּחִים, / כֻּסּוּ פָנֶיהָ חֲרֻלִּים, / מַבְרַחַת יוֹנְקִים וְעוֹלֲלִים, / הָיְתָה בְעֵינֵי כֹל בְּזוּיָה, / וּשְׁמָהּ רִצְפָּה בַת-אַיָּה. / וַתִּקַּח כָּזְבִּי אֶת הַשְּׁחֹרָה / וַתְּבִיאֶהָ אֶל זֶרַח הַחַדְרָה, / וַיִּפְנוּ וַיֵּלְכוּ שֶׁקֶר וְכָזְבִּי בְשִׂמְחָה, / כִּי כָרוּ לִפְנֵי זֶרַח שׁוּחָה, / וַיִּתְנַכְּלוּ עַל הַכֶּסֶף אֲשֶׁר הֵכִינוּ / וַיֵּלְכוּ וַיַּטְמִינוּ.

וַיִּשָּׁאֵר זֶרַח עִם רִצְפָּה בַת-אַיָּה / עֵירֹם וְעֶרְיָה, / וַיְהִי עַד כֹּה וְעַד כֹּה / וַיִּזְרַח לוֹ הַשֶּׁמֶש וְהוּא צוֹלֵע עַל יְרֵכוֹ, / וְהִנֵּה רִצְפָּה – מְקוֹם אַיָּלָה שְׁלוּחָה, / וְתַחַת מַעֲשֶׂה מִקְשֶׁה – קָרְחָה! / וַיִּקְרַע אֶת לְבוּשׁוֹ / וַיָּשֶׂם עָפָר עַל רֹאשׁוֹ / וַיִּקְרָא “קֶשֶׁר שֶׁקֶר / רַב-שֶׁקֶר, / עַל זֹאת אֲקוֹנֵן וְאֵילִילָה /וּמִסְפֵּד מַר אַגְדִּילָה. / הַכֶּסֶף אָזַל וְנִשְׁאַר הַדֹּפִי / אֵין זָהָב וְאֵין יֹפִי! / כִּי הָיָה הַקֶּשֶׁר אַמִּיץ וְעָלַי גָּבַר – / פֶּתִי יַאֲמִין לְכָל דָּבָר. / עַד כֹּה רַבּוּ שְׂשׂוֹנַי / וְעַד הֵנָּה עֲזָרַנִי אֲדֹנָי. / אֲזַי קַרְתִּי / וְאֶת הָאִשָּׁה הַזֹאת לָקַחְתִּי, / וְלֹא אֶעֱנֶה עַזּוֹת – / חָפַרְתִּי אֶת הַבְּאֵר הַזֹּאת / וְלָקַחְתִּי לְנַפְשִׁי מְהָרְסֶיהָ וּמַחֲרִיבֶיהָ / וְנָתַתִּי נַפְשִׁי בְּכַף אוֹיְבֶיהָ. / אוֹי לִי, מִי יוֹעִילֵנִי / וּמִי יַצִּילֵנִי? / וּלְחִצֵּי מָוֶת לֹא יוֹעִיל שִׁרְיָן / כִּי בָא הַחֲלוֹם בְּרֹב עִנְיָן”. / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

חֲמָסִי עַל זְמָן הִרְבָּה / מְצוּקוֹתַי וְאַנְחוֹתַי

וּפָרַשׂ פַּח בְּמַעְגָּלִי / וְשָׁלַח אֵשׁ בְּעַצְמוֹתַי,

וְהַיָּמִים הֲדָפוּנִי / וְהֵפֵרוּ מְזִמּוֹתַי.

לְזֹאת אָהִים וְאֶשְׁתּוֹמֵם / וְאָגֹז אֶת קְוֻצּוֹתַי,

וְלִבְנוֹת יַעֲנָה אֶקְרָא / לְהִתְקַבֵּץ סְבִיבוֹתַי

וְתִשֶּׂנָה נְהִי עָלַי / וְאֶבְכֶּה עַל בְּחוּרוֹתַי.

אֲהָהּ הַיּוֹם מְרִי שִׂיחִי / וְזֶה מִסְפַּר חֲלוֹמוֹתַי!

וַתֹּאמֶר לוֹ אִשְׁתּוֹ “חֲגֹר מָתְנֶיךָ / וְאַל תִּתֵּן שֵׁנָה לְעֵינֶיךָ / וְאַל תַּמְרֶה דְבַר פִּי וְקוֹלִי / וְלֵךְ קַח לִי: / כְּלֵי כֶסֶף וּכְלֵי זָהָב וּשְׂמָלוֹת / הַנְּטִיפוֹת וְהַשֵּׁרוֹת וְהָרְעָלוֹת / וּמִשְׁכָּן וְדִירָה / וְכִסֵּא וּמְנוֹרָה / וְשֻלְחָן וְכַפּוֹת / וֶעֱלִי וְרִיפוֹת / וּמַכְבֵּר וְכִישׁוֹר / וּמַחְצֶלֶת וְכִיּוֹר / וְטֶנֶא וָפֶלֶךְ וּפָרוּר וּבַקְבּוּק וְקַלַּחַת / וּמַכְתֵּשׁ וּמַטְאֲטֵאת וּמִטְפַּחַת / וְסִיר וְתַנּוּר וְכִירַיִם / וְחָבִית לַמַּיִם / וּמִזְרָקוֹת וְיָעִים / וְכֵלִים וּגְבִיעִים / וּבָתֵּי הַנֶּפֶשׁ וְהַמַּעֲטָפוֹת / וּרְדִידִים וּצְנִיפוֹת / וּמַחֲלָצוֹת וּסְדִינִים / וְנִזְמֵי הָאַף וְגִלְיוֹנִים / וַחֲרִיטִים וְשַׂהֲרוֹנִים / וּלְחָשִׁים וְקִשּׁוּרִים / וְרִקְמָה וּבוּץ וּפְאֵרִים,/ הַטַּבָּעוֹת וּשְׁבִיסִים / וּצְעָדוֹת וַעֲכָסִים – / וּמִלְּבַד אֵלֶּה: בְּגָדִים חֲמוּדִים / לְשַׁבָּתוֹת חַגִּים וּמוֹעֲדִים, / וְכָהֵנָּה וְכָהֵנָּה / יָדֶיךָ תְבִיאֶינָה / וְאִם מִנְּשׂא כָל אֵלֶּה תִלְאֶה / וְהָיִיתָ מְשֻׁגָּע מִמַּרְאֵה עֵינֶיךָ אֲשֶׁר תִּרְאֶה – / וְזֶה-לְּךָ הָאוֹת: הַשָּׁנָה הַזֹּאת תִּהְיֶה לְךָ לְעִצָּבוֹן / וְהַשֵּׁנִית לִדְאָבוֹן, / לְעֶבֶד תִּהְיֶה אַחֲרֵי הֱיוֹתְךָ שַׂר / וְדֹמֶן עַל רֹאשְׁךָ אַל יֶחְסַר, / כִּי לֹא יֵאוֹת לְךָ לִהְיוֹת אֶחָד מֵעֲבָדַי / לָצֶקֶת מַיִם עַל יָדַי”!

וַיְהִי כִשְׁמֹע זֶרַח אֶת דִּבְרֵי הָאָלָה / וַיִּצְעַק צְעָקָה גְדוֹלָה / וַיֹּאמַר “הַעֶבֶד יִשְׂרָאֵל? / אִם אֵין לוֹ גוֹאֵל? / עַתָּה קָמוּ הַשְּׁפָחוֹת / עַל הַסְּגָנִים וְהַפַּחוֹת, / וְאָזְנַיִם רְצוּעוֹת / יִמְשְׁלוּ עַל הַמִּגְבָּעוֹת”. / וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר:

שְׁחָקִים וַאֲרָצוֹת עִם גְּבָעוֹת, / בְּכוּ עַל זֹאת וְהוֹרִידוּ דְמָעוֹת!

הֲבֶן-חוֹרִין וְיֶלֶד שַׁעֲשׁוּעִים / יְהִי עֶבֶד לְמִי אָזְנָיו רְצוּעוֹת?

וַיּוֹסֵף עוֹד שְׂאֵת מְשָׁלוֹ וַיֹּאמַר:

שְׁתַּיִם מֵרֲרוּ חַיַּי וְאוֹתָם / לְדוֹר וָדוֹר אֲסַפֵּר וַאֲחַוֶּה:

יְלִיד חָכָם אֲשֶׁר עָבַד בְּאִשָּׁה / וּבֵן מַשְׂכִּיל לְיַד סָכָל יְקַוֶּה.

וַתֹּאמֶר לוֹ אִשְׁתּוֹ “אַל תִּשָּׂא מְשָׁלֶיךָ / וְאַל תַּגְבִּיהַּ קוֹלֶךָ! / אֵין לִי חֵפֶץ בְּשֵׂכֶל וּבְמוּסָר, / כִּי אִם בְּלֶחֶם וּבְבָשָׂר. / מְשָׁלֶיךָ וְשִׁירֶיךָ נֶגְדִּי כְאַיִן, / בַּשִּׁיר לֹא יִשְׁתּוּ יַיִן. / אָמַרְתָּ אַךְ דְּבַר שְׂפָתַיִם / לָבוֹא עַל רַחַם רַחֲמָתַיִם? / קוּם קַח מַחֲרֵשָׁתְךָ וְאִתְּךָ, / שָׂא נָא תֶלְיְךָ וְקַשְׁתְּךָ, / וְהַקֵּף אֶת הָעִיר מֵעֶרֶב עַד בֹּקֶר / גָּנוֹב וְרָצוֹחַ וְהִשָּׁבֵעַ לַשֶּׁקֶר! / וּבְמַרְבִּית בֵּיתְךָ תַּטְרִיף אֶת לֶחֶם חֻקָּם / וְלֹא יֵרָאוּ פָנַי רֵיקָם, / כִּי הָאִשָּׁה – עָלֶיךָ מַחְסוֹרָהּ / לְעָבְדָה וּלְשָׁמְרָהּ. / וְאִם לֹא תָבִיא אֶת כָּל אֲשֶׁר אֶחְמֹד – / בַּחוּץ תַּעֲמֹד! / וְכָל זֶה נָקֵל עַד בּוֹא עֵת הַלֵּדָה – / אָז תִּכְבַּד הָעֲבוֹדָה: / אַחַי וּקְרוֹבַי לְנֶגֶד עֵינֶיךָ / סָבִיב לְשֻׁלְחָנֶךָ, / וְתַעֲשֶׂה לָהֶם בְּכָל יוֹם מַטְעַמִּים – / שֶׂה תָמִים, / וְלַמְיַּלֶּדֶת לֹא יֶחְסַר / יַיִן לֶחֶם וּבָשָׂר, / וּלְבָנֶיךָ תִקְרָא אִשָּׁה מֵינֶקֶת – / הֵם מַגִּישִׁים אֵלֶיהָ וְהִיא מוֹצֶקֶת – / וַאֲמָהוֹת וּשְׁפָחוֹת תִּקְנֶה לָהֶם / לֶאֱכֹל אֶת צוֹאָתָם וְלִשְׁתּוֹת אֶת מֵימֵי רַגְלֵיהֶם, / וְיָדִי לֹא תִגַּע בָּהֶם!”

וַיִּפֹּל זֶרַח מְלוֹא קוֹמָתוֹ אַרְצָה / וְנַפְשׁוֹ בַּחַיִּים קָצָה, / וַיֵּבְךְּ עַל בְּחוּרָיו / וְעַל יְמֵי נְעוּרָיו, / כִּי יָרְדוּ לַמִּצְעָר / וְכִי הִלְבִּין שְׂעַר רֹאשׁוֹ – וְהוּא נַעַר. / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

רְאוּ בִי פְלָאוֹת וְנוּדוּ לְיֶלֶד / אֲשֶׁר שָׂב כְּאִלּוּ שְׁנוֹתָיו שְׁמוֹנִים,

וְהִלְבִּין שְׂעָרוֹ וְקָדַר מְאוֹרוֹ, / וְזִמֵּן שְׁחֹרִים – וּמָצָא לְבָנִים!

וַיִּשְׁמְעוּ שְׁלֹשֶׁת רֵעָיו / הַקְּרוֹבִים אֵלָיו / אֶת כָּל הָרָעָה הַבָּאָה עָלָיו, / וַיֵּצְאוּ אִישׁ מִמְּקוֹמוֹ / לָבוֹא לָנוּד לוֹ וּלְנַחֲמוֹ. / וַיִּשְׂאוּ עֵינֵיהֶם מֵרָחוֹק וְלֹא הִכִּירוּהוּ / כִּי מִשְׁחַת מֵאִישׁ מַרְאֵהוּ. / וַיַּעַן נְתַנְאֵל בֶּן-מַהֲלַלְאֵל וַיֹּאמַר “הַצְּבִי, אֵיך נִלְכַּדְתָּ / וּמִכִּסְאֲךָ אֵיךְ הוּרַדְתָּ / וְעִם יוֹרְדֵי בוֹר אֵיךְ נִמְשַׁלְתָּ? / כִּנְפֹל לִפְנֵי בְנֵי עַוְלָה נָפַלְתָּ / וּדְבָרֶיךָ הַנְּעִימִים שִׁחַתָּ – / מָה הַבִּטָּחוֹן הַזֶּה אֲשֶׁר בָּטַחְתָּ! / הִנְּךָ מֵת עַל הָאִשָּׁה אֲשֶׁר לָקַחְתָּ, / חַטֹּאותֶיךָ לְרָעָתְךָ הִקְרִיבוּךָ, / עַל כֵּן עֲלָמוֹת אֲהֵבוּךָ”. / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אוֹי לְךָ, זֶרַח, וְאוֹי לָךְ! / דַּעְתְּךָ אַיֵּה וְשִׂכְלָךְ?

צָמְחוּ שַׂעְרֵי פְאֵרָךְ – / עֻזְּךָ חָלַף וְהָלַךְ!

שֹׁד וְשֶׁבֶר עִם יְגוֹנִים / נָפְלוּ הַיּוֹם בְּחֶבְלָךְ,

נִבְהֲלוּ בָךְ רַעֲיוֹנִים / בִּי- וּמָה עוֹד לַעֲשׂוֹת לָךְ?

וַיַּעַן אֲבִינֹעַם בֶּן-אֲבִינָדָב וַיֹּאמַר “כְּעֹצֶם מַעֲלָתְךָ וְתִפְאַרְתֶּךָ / הָיְתָה מַפַּלְתֶּךָ. / נָפְלָה פִנַּת יִקְרָתֶךָ / וְעָרְבָה כָּל שִׂמְחָתֶךָ, / מְהֵרָה נִתְּקוּ זְמָמֶיךָ / וְנִגְדַּעְתָּ בַּחֲצִי יָמֶיךָ. / זֹהַר נְעוּרֶיךָ עָטָה שַׁחֲרוּת / וּשְׁמֵי עֻזְּךָ לָבְשָׁה קַדְרוּת. / דְּבַר שֶׁקֶר וְכָזְבִּי הִקְשַׁבְתָּ / וְדִבְרֵי אֱמוּנִים עָזַבְתָּ / וּמִגְדַּל עֹז בָּנָה אָבִיךָ יָדְךָ הֲרָסוֹ, / עַתָּה הִסְכַּלְתָּ עֲשׂוֹ. / הַאִישׁ כָּמוֹךָ נוֹקַשׁ וְנִכְשַׁל / כִּי הָיִיתָ הַיּוֹם לְמָשָׁל? / עָלֶיךָ אֶשָּׂא בְּמַר רוּחִי: / צַר לִי עָלֶיךָ, אָחִי!” / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

שֶׁמֶשׁ נְעוּרִים אֲשֶׁר חָשְׁכוּ מְאוֹרֶיהָ

תִּצְרַח בְּמָרָה לְמַכְאוֹבָהּ וְצִירֶיהָ,

תִּצְעַק: אֲהָהּ, נָפְלָה קוֹמַת בְּרוֹשֶׁיהָ

עַד יָם רְחוֹקִים תְּשַׁלַּחַ אֶת קְצִירֶיהָ.

יוֹדְעֵי נְהִי, מַהֲרוּ סִפְדוּ לְאִישׁ אָסְרָה

כָזְבִּי לְהָרְגוֹ וְקָמוּ כָל אֱסָרֶיהָ.

זֶרַח בְּכוּ נָא וְהִתְאַסְפוּ לְתַנּוֹת וְכָל

נֶפֶשׁ חֲכָמָה מְמָרֶרֶת אֲמָרֶיהָ,

מַה לִּי בְּחַיִּים וּמַה לִּי עוֹד בְּטוּב הַזְּמָן?

תֵּבֵל – חֲלוֹם שָׁוְא וְזֶה חֶזְיוֹן שְׁקָרֶיהָ!

וַיַּעַן אֲחִיטוּב בֶּן-אֲבִישׁוּעַ וַיֹּאמַר “אַל תִּבְכּוּ לְמֵת וְאַל תָּנוּדוּ לוֹ, / בְּכוּ בָכֹה לַאֲשֶׁר בְּרֶשֶׁת הַנִּשּׂוּי נִלְכְּדָה רַגְלוֹ. / יוֹנָה יָצָא מִמְּעֵי הַדָּגָה – וְאֵין לַאֲסִיר אִשָּׁה תִּקְוָה וְלֹא פוּגָה! / הִרְכַּבְתָּ אֱנוֹשׁ עַל רֹאשֶׁךָ / וְלָקַחְתָּ מוֹקֵשׁ לְנַפְשֶׁךָ. / זְדוֹן לִבְּךָ הִשִּׁיאֶךָ / וּבְפַח יָקוּשׁ הֱבִיאֶךָ, / אַיֵּה מוֹעֲצוֹתֶיךָ / וְאַיֵּה חָכְמָתְךָ וּתְבוּנוֹתֶיךָ? / תְּמוֹל הָיִיתָ לְמוֹשֵׁל בָּאָרֶץ – / וְהַיּוֹם לְכָלָה וָחָרֶץ, / תְּמוֹל הָיִיתָ כְּמֶלֶךְ בַּגְּדוּד – / וְהַיּוֹם בָּזוּז וְשָׁדוּד, / תְּמוֹל הָיִיתָ נָגִיד וּמְצַוֵּה לְאֻמִּים – / וְהַיּוֹם חֶרְפַּת עַמִּים, / תְּמוֹל הָיִיתָ גִבּוֹר כְּאַרְיֵה – / וְהַיּוֹם לֹא תָקוּם וְלֹא תִהְיֶה, / תְּמוֹל הָיִיתָ כְגֹבַהּ בְּרוֹשִׁים – / וְהַיּוֹם נִבְזֶה וַחֲדַל אִישִׁים, / תְּמוֹל הָיָה דְבָרְךָ נָחוּץ – / וְהַיּוֹם לֹא שֵׁם לְךָ עַל פְּנֵי חוּץ. / בֶּאֱמֶת נִקְרָא שִׁמְךָ זֶרַח, כִּי זָרְחָה בְמִצְחֲךָ הַצָּרַעַת / וְהִקְשַׁבְתָּ לְכָזְבִּי הַמִּרְשַׁעַת! / הֲמִירוֹתָ דִבְרֵי תַחְכְּמֹנִי בְּדִבְרֵי שֶׁקֶר / לָכֵן תִּהְיֶה כַעֲנַן בֹּקֶר”.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

כָּל אִישׁ אֲשֶׁר יַמְרֶה עֲצַת אָבִיו / יָמוּת בְּיַד אִשָּׁה כְּמוֹת נָבָל. לוּ יִהְיֶה בִימֵי בְנוֹ עַמְרָם – / בּוֹ הַקְּלָלָה לוֹ בְהַר עֵיבָל!

וְזֹאת לִיהוּדָה וַיֹּאמַר:

גַּם אֲנִי לֹא אֶחֱשָׂךְ-פִּי / בְּעוֹד בְּקִרְבִּי שְׂעִפִּי, / לָזֹאת אַכֶּה כַפִּי / וְאֶבְכֶּה יְמֵי חָרְפִּי / וְעָלֶיךָ אֶשָּׂא בְּמַר קוֹלִי, / הֵיטֵב חָרָה לִי. / מְאֹד קָשָׁה עָלַי תּוּגָתֶךָ / וְעָצְמָה לִי תְלָאָתֶךָ, / זֵרִיתָ אָרְחֲךָ וְרִבְעֶךָ – / וּבַחֹשֶׁךְ רִפַּדְתָּ יְצוּעֶךָ, / אַיָּלָה שְׁלוּחָה חָשַׁקְתָּ – / וְעִם רִצְפָּה בַת-אַיָּה דָבַקְתָּ! / וְלֹא שָׁמַעְתָּ לְקוֹל מוֹרֶיךָ – / כָּזֶה וְכָזֶה יִקְרֶךָ, / עָזַבְתָּ דֶרֶךְ הַמִּישׁוֹר – / כָּכָה יֵעָשֶׂה לַשּׁוֹר, / כָּל הַמַּמְרֶה דְבַר אָבִיו וְקוֹלוֹ – / כַּמִּשְׁפָּט הַזֶּה יֵעָשֶׂה-לּוֹ! / שֶׁקֶר וְכָזְבִּי לְחֶרְפָּה שָׂמוּךָ – / כָּמוֹנִי כָמוֹךָ, / הַנִּשּׂוּי שָׂמְךָ לַחֲרָדָה / וַיְגַל אֶת מָסַךְ יְהוּדָה". / וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר:

אֲהָהּ זֶרַח, אֲבוֹי לָךְ – וַאֲבוֹי לִי, / לְכָה נִבְכֶּה וְנָנוּד לַאֲחָדִים לִזְמָן הָרַע אֲשֶׁר רָדַף כְּאוֹיֵב / וְשָׂמָנוּ בְיַד נָשִׁים עֲבָדִים.
