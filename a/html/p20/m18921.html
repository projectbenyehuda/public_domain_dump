אָבִינוּ הַיּוֹם בְּלִי אָב דָּמִינוּ

זולת (לשבת איכה)


אָבִינוּ, / הַיּוֹם בְּלִי אָב דָּמִינוּ,

בָּזִינוּ / דְבָרְךָ וְכִפְלַיִם לָקִינוּ,

רָאִינוּ / יוֹם נִשְׂרְפָה עִיר שֶׁשָּׁם עָלִינוּ,

הָיִינוּ / חֵרְשִׁים וְאִלְּמִים וְלֹא נִפְתַּח פִּינוּ,

מָצִינוּ / קֻבַּעַת כּוֹס שָׁתִינוּ.


עָלֶיךָ הֹרַגְנוּ וְאֵלֶיךָ קִוִּינוּ,

זְכֹר חַסְדְּךָ וְאַל תִּזְכֹּר אֶת אֲשֶׁר הֶעֱוִינוּ,

רְאֵה, כִּי אַרְבַּע גָּלֻיּוֹת גָּלִינוּ.


גָּלוּת מִצְרַיִם רִאשׁוֹנָה, / שֵׁנִית גָּלוּת שֹׁמְרוֹנָה,

שְׁלִישִׁית גָּלוּת צִיּוֹנָה, / רְבִיעִית גָּלוּת אַחֲרוֹנָה;


בְּצֹעַן בַּיְאוֹר מֵתוּ בָנַי, / בְּחָבוֹר נִפְזְרוּ עֲדַת אֶבְיוֹנָי,

בְּבָבֶל נִשְׁבּוּ כְּלֵי בֵית אֲדֹנָי, / בֶּאֱדוֹם נִפְזְרוּ מְתֵי אֵיתָנָי –

יַחַד כֻּלָּם יְצָאוּנִי, / צָרוֹת רַבּוֹת מְצָאוּנִי.


שְׁתַּיִם הָיוּ קֹרְאוֹתַי, / שְׁתַּיִם הַיּוֹם רָעוֹתָי:

אַחַת – הַשֹּׁד וְהַשֶּׁבֶר, / שֵׁנִית – הָרָעָב וְהַדֶּבֶר;


הֲמוֹנַי / מַמְרוֹר שָׂבָעוּ,

זְקֵנַי / בָּעִיר גָּוָעוּ,

כֹּהֲנַי / יַחַד כְּשִׁכּוֹר תָּעוּ,

נְבִיאַי / זָעֲקוּ וְלֹא נִשְׁמָעוּ.


הִנֵּה בְּגָלוּת מִצְרַיִם / פַּרְעֹה הִכָּה אֲחוֹרָיִם,

הֶחְשִׁיךְ פָּנַי כַּכִּירַיִם, / הֶעֱבִיד בְּפֶרֶךְ אֶפְרָיִם;

הִנֵּהבְּתִגְלַת פִּלְאֶסֶר / לָקַח בְּנוֹתַי הָעְשֶׂר,

נְתָנָם לָרָעָב וְלַחֹסֶר / וְעוֹדָם צוֹעֲקִים מֵאֹסֶר;


הִנֵּה בְּגָלוּת בַּת בָּבֶל / מַעֲשֵׂי נְבוּכַדְנֶאצַּר חִבֵּל,

כִּלָּה שְׁאֵרִית נַחֲלָה חֶבֶל, / עִנָּה רַגְלִי בַּכֶּבֶל;


הִנֵּה בְּגָלוּת הַחֵל הַזֶּה / מָשַׁל בִּי אֱדוֹם הַלָּזֶה,

יִבְזֶה אוֹתִי – וְהוּא נִבְזֶה; / כֻּלָּם יַחַד, זֶה כְמוֹ זֶה,

לְחָצוּנִי, / מְחָצוּנִי, / הֱלִיצוּנִי, / הֱרִיצוּנִי,

וְעַד קְצֵה הָאָרֶץ הֱפִיצוּנִי;

נֶאֱלַמְתִּי, / נֶהֱלַמְתִּי, / נֶעֱלַמְתִּי, נִכְלַמְתִּי,

אָבִי, מְצוּקִי מִמְּךָ לֹא נֶעְלָם –

כִּי עֶזְרַת אֲבוֹתֵינוּ אַתָּה הוּא מֵעוֹלָם.
