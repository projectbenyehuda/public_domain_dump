יי אתנפל ואשפיל (תחינה)

יְיָ, אֶתְנַפֵּל וְאַשְׁפִּיל / וּתְחִנָּתִי אַפִּיל

וְתִתְעַטֵּף רוּחִי; / אוּלַי שִׂיחִי

חֲמָתָךְ יִכְפֶּה.


בִּנְךָ – אַרְצוֹ הֱקִיאוֹ / וְהַצָּר הֱנִיאוֹ,

בְּכוּר עֹנִי הֶפְאוֹ / וְהֵחֵל לְהַשִּׂיאוֹ

בְּחֵלֶק שְׂפָתוֹת / לְשַׁנּוֹת דָּתוֹת:

כִּי אֶחֱרוּ הָאוֹתוֹת / וּמֵאָה שַׁבָּתוֹת

וְשִׁשִּׁים וְשֵׁשׁ נִסְפֶּה.


רָחַב בֹּעֲלִי, / כִּי נָטָה צִלֲלִי

וְיָבְשׁוּ שָׁרְשִׁי וְעָלִי / וְהוּא אָמַר לִי:

׳לִצְמִיתֻת נִמְכַּרְתָּ, / כִּי הָיִיתָ אַתָּה,

מֵאָז וְעַד עַתָּה,

בִּמְלֶאכֶת יְיָ מִתְרַפֶּה.׳


הַעַל אֵלֶּה תַעְלִים / עַיִן – וְלֹא תַכְלִים

פְּנֵי הַנְּבָלִים / הַמִּתְהַלֲלִים בֶּאֱלִילִים

וּלְקוֹסְמִים פּוֹנִים / וְדוֹרְשֵׁי יִדְעוֹנִים

וּמְנַחְשִׁים וּמְעוֹנֲנִים?

וַאֲנִי בָּייָ אֲצַפֶּה!


מָתַי תַּאֲשִׁימֵם / וְהַחֲרֵם תַּחֲרִימֵם –

וְנִסְּךָ תְּרוֹמֵם / וְהַנּוֹפֵל תְּקוֹמֵם,

וְתַרְבֶּה תִפְאַרְתִּי / וְתַשְׂגִּיא מַשְׂכֻּרְתִּי

וְתִתֵּן שִׁבְתִּי

וְרַפֹּא יְרַפֵּא?


עֲדֶנָה נֶעְכַּרְתִּי, / בְּלִי יַיִן שָׁכַרְתִּי

עַד לֹא נִכַּרְתִּי – / כִּי בְלִי מְחִיר נִמְכַּרְתִּי.

וְהֶעֱוִיתִי מַהֲלָכִי / וְאֵין צֶדֶק בְּתוֹכִי,

כִּי עֵירֹם אָנֹכִי –

בְּחַסְדְּךָ מַעֲרֻמַי תְחַפֶּה!


זָכְרָה בְרִיתְךָ / וְקַנֵּא לְבֵיתְךָ

וְצֵא כְּמֵאָז בְּצֵאתְךָ / וְאַל תְּאַבֵּד שְׁאֵרִיתְךָ

אֲשֶׁר שַׂק חוֹגֶרֶת / כְּשִׁפְחָה נֶאֱסֶרֶת –

וּבְמַלְבּוּשׁ הַגְּבֶרֶת

בֶּן הָאָמָה מִתְיַפֶּה.


רֹאשׁ לֹא נָשָׂאתִי / מִיּוֹם בַּגּוֹלָה יָצָאתִי,

בְּאֵשׁ וּבְמַיִם בָּאתִי / וּמְנוּחָה לֹא מַָצָאתִי,

שַׁמָּה הֱשִׁמָּנִי / וֶאֱדוֹם הֱדִמָּנִי

זֶה לַגַּי הוֹרִידָנִי / וְזֶה הוֹשִׁיבָנִי

עַל הַר נִשְׁפֶּה.
