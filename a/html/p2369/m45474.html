יחזקאל ב״ר סלימאן ב״ר עבדאללה חנין

קרובו (דודו, אחי אביו?) של עבדאללה חנין הנ“ל, היגר להודו ויסד דפוס ככלכתה, The Jewish Press, ונדפסו בו בשנים תרל”א-תרנ“ג (1893–1871) עשרה ספרים ושבועון בשם “מבשר”, שהוא עצמו ערך והוציא לאור. בשבועון זה פירסם שירים משלו. בשנת תר”מ הדפיס שיר בשבועון “פרח”, שיצא בכלכתה אחרי שה“מבשר” חדל לצאת. שיר אחד על צרות הגלות וערגה לציון, והיתר שירי אישים, בני משפחות שיצאו מבגדאד והתיישבו בהודו והיתה להם השפעה על הקהילה.


סימן: יחזקאל ח׳זק. לחן: אליך קוני.


יְדִידִי חֶלְקִי / חִשְׁקִי / לָמָּה שַׂמְתַּנִי

לְמַכִּים גֵּוִי / כְּוִי / וַעֲזַבְתַּנִי

בְּיַד צַר מוּזָר / אַכְזָר / וַיִּשְׂטְמֵנִי

צֵא לְעֶזְרָתִי, עֹז יְשׁוּעָתִי, נָא לְקַדְמוּתִי, אוֹתִי, הָשֵׁב עַל כַּנִּי.


חִשְׁקֵךְ לִי טָמוּן / אָמוּן / מֵאָז עֲזַבְתִּיךְ

עַל יַד יְמִינִי / אֲנִי / כְּחוֹתָם שַׂמְתִּיךְ

שׁוּבִי רַעְיָתִי / אִתִּי / חֶסֶד רִחַמְתִּיךְ

אֶבְנֶה מִקְדָּשַׁי, וְאֶשְׁלַח לָךְ שַׁי, מִגֶּזַע יִשַׁי, יִשַׁי, קוּמִי וָרֹנִּי.


זְכֹר אֶת עָנְיִי / דַיִּי / מֵי רֹאשׁ, לַעֲנָה

עַד אָן בְּגָלוּת / דַּלּוּת / אֶהְמֶה כְיוֹנָה

שְׁעֵה לְשַׁוְעִי / יִשְׁעִי / עֵת לְחֶנְנָה

צוּרִי מִשְׂגַּבִּי, שַׂמַּח לְבָבִי, וְגוֹאֵל תָּבִיא, תִּשְׁבִּי, יְבַשְּׂרֵנִי.


קוּמִי נָא עוּרִי / שִׁירִי / וְזִמְרָה פִצְחִי

נַפְשִׁי נִכְסְפָה / יָפָה / אֶשְׁלַח מְשִׁיחִי

יִפְדֵּךְ מִשִּׁבְיָה / רַעְיָה / שִׂישִׂי וְשִׂמְחִי

בִּגְדֵי שֵׁשׁ רִקְמָה, תִּלְבְּשִׁי רָמָה, בָּרָה כַּחַמָּה, תַּמָּה, זִיוֵךְ הַרְאִינִי.


אֵלֶיךָ צוּרִי / עֶזְרִי / תָּמִיד אֲיַחֵל

יְרוֹמֵם דִּגְלִי / אֵלִי / יְשׁוּעוֹת וָחֵל

שׁוֹכֵן רוּם חֶבְיוֹן / צִיּוֹן / אוֹתִי הַנְחֵל

מְקוֹם מִקְדָּשִׁי, אִוְּתָה נַפְשִׁי, אַקְרִיב לָךְ אִשִּׁי, אִישִׁי, תּוֹדַת קָרְבָּנִי.


חִזְקִי וְאִמְצִי / עִלְצִי / חָזָק גּוֹאֲלֵךְ

אֶת אוֹיְבֵךְ יִגְמֹר / יִזְמֹר / סָחֹב וְהַשְׁלֵךְ

וּפִצְחִי גָרוֹן / בְּרֹן / צַהֲלִי קוֹלֵךְ

זִמְּרָה הַלְלִי, צִיּוֹן תַּעֲלִי, מִשְׁכָּן זְבוּלִי, גִּילִי, בְּשִׁיר רַנְּנִי.
