הֲשֶׁמֶשׁ בַּגְּבִיעִים

אל משה

הֲשֶׁמֶשׁ בַּגְּבִיעִים אוֹ לְהָבִים / וּמֵאֹדֶם, אֱמֹר, אוֹ דַם עֲנָבִים,

וּמִמַּיִם אֲשֶׁר קָפְאוּ אֲשִׁישִׁים / וְאִם מִמַּחֲצַב שֹׁהַם חֲצוּבִים?

יְלִיד שֹׂרֵק בְּתוֹכָמוֹ כְרוּחַ / לְחַיּוֹתָם וְהֵמָּה כַקְּרָבִים.

שְׁתִינוּהוּ וְעִמָּנוּ אֲנָשִׁים / קְרָאנוּם עֵת פְּקֹד בָּנִים אֲהוּבִים,

וּמִי צֵל הַהֲדַס נֶעְדַּר לְפָנָיו / יְהִי יֹשֵׁב בְּצִלֵּי הָעֲרָבִים,

וְאַרְיֵה לֹא יְהִי מֹצֵא נְמֵרִים / לְבַל יֹאבַד – יְהִי טֹרֵף כְּלָבִים,

וּמִתְחַקֶּה עֲלֵי שָׁרְשֵׁי פְעָמִים / לְאִישִׁים – כָּל יְמוֹתָיו הֵם עֲצָבִים,

וְעֵין רָצוֹן תְּהַרְאֵהוּ נְכֹחוֹת / וְעָבֵי אַף מְאוֹר צֶדֶק מְעִיבִים,

וְדוֹדֵי הָאֱמֶת הַיּוֹם מְעַטִּים / וְאִם הֵמָּה בְשֵׁם אֵמֻן נְקֻבִים,

אֲסַפֵּר בַּעֲדָם אִמְרֵי אֱמוּנָה / וְהֵמָּה דִבְּרוּ עָלַי כְזָבִים,

יְרַמּוּנִי בְצוּף אֹמֶר לְהַעְלִים / הֲכִי הֵם לִי מְרֵעִים אוֹ מְטִיבִים.

הֲלֹא יֶחְטָא זְמָן פַּעַם לְהַקְרִיב / אֲשֶׁר הִרְחִיק וְיַרְחִיק הַקְּרוֹבִים,

אֲשֶׁר תָּעוּ נְדִיבֵי הַחֲסָדִים / וְנָסוֹגוּ וּפָנוּ אֶל רְהָבִים.

נְגִידֵי מַאֲמָר אִמְרֵי חֲכָמִים / וְשָׂרֵי נֹזְלִים כַּפֵּי נְדִיבִים,

וְשִׁיר מֹשֶׁה, אֱמֶת, הוֹוֶה לְמֶלֶךְ / לְשִׁיר עֶרֶב בְּמִלָּיו הָעֲרֵבִים.

גְּבִיר יֹשֵׁב עֲלֵי חוּג הַתְּעוּדָה / וְיֹשְׁבֶיהָ לְפָנָיו כַּחֲגָבִים,

בְּפָנָיו זָרְחָה שֶׁמֶשׁ צְדָקָה / וְלָבְשָׁה עַל פְּנֵי כָל־שַׂר עֲרָבִים.

קְשֵׁה לֵב יוֹם קְרָב שֵׂכֶל וְרַכּוּ / אֲמָרָיו עַד לְבָב אַרְיֵה מְדִיבִים.

לְכָל מַחְלָה בְּפִיו מַרְפֵּא, וּמַלְקוֹשׁ / בְּעֵת יִמָּנְעוּ לַשְׁקוֹת רְבִיבִים,

וְהוּא יַכְאִיב לְבַב צָרָיו וּפִיהוּ / צֳרִי גִלְעָד לְרַפֵּא אֶת כְּאֵבִים,

וְגָנְבוּ יֹעֲצִים מֵי מַחְשְׁבוֹתָיו / וְלָהֶם מָתְקוּ מַיִם גְּנוּבִים,

וְיוֹם נִצָּב לְבָאֵר תַּעֲלוּמוֹת – / מְתֵי חָכְמָה לְעֻמָּתוֹ נְצִיבִים,

וְיוֹנֶה נֹהֲלֵי חָכְמַת חֲכָמִים / וְיַעְרִיב כָּל מְאוֹר צַחוֹת עֲרָבִים.

וְעֵטוֹ מִבְּלִי כָנָף יְעוֹפֵף / בְּתִתּוֹ אֶצְבְּעוֹתָיו לוֹ כְרֻבִים.

יְרִיבוּנִי בְּאַהְבָתוֹ פְתָאִים / עֲדֵי אֶשְׁחַק לְרַגְלוֹ הַמְּרִיבִים.

וְיוֹם מִתַּאֲוָה יֵצֵא לְבָבִי / כְּתָבָיו הֵם אֲשֶׁר אוֹתוֹ מְשִׁיבִים.

וְהֵא הֵלֶךְ דְּבַשׁ דּוֹדַי, יְחַו כִּי / יְאוֹרֵי הָאֲהָבִים לֹא חֲרֵבִים,

חֲרוּתִים הֵם עֲלֵי לֻחוֹת לְבָבִי / וְאַךְ מִזֶּה וּמִזֶּה הֵם כְּתֻבִים.
