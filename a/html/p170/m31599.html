לִגְבִיר דֵּעוֹת, וְחֹסֶן יְשׁוּעוֹת

לִגְבִיר דֵּעוֹת, וְחֹסֶן יְשׁוּעוֹת, וְחַיִל וְצָבָא, וְחוֹמָה נִשְׂגָּבָה, וְשֶׁמֶשׁ בְּעֵיפָה, וְצֵל סֶלַע כָּבֵד בְּאֶרֶץ עֲיֵפָה, אִישׁ חֲמוּדוֹת, וְצוּר מָעֹז וּבֵית מְצוּדוֹת, וְצִיץ הַנֵּזֶר, אֶבֶן הָעֶזֶר, יְסוֹד הָאֱמוּנָה, וְסוֹד הַדַּעַת וְהַתְּבוּנָה, הַר הַמּוֹר גִּבְעַת הַלְּבוֹנָה, מַשְׂכִּיל בְּכָל חָכְמָה, גִּבּוֹר חַיִל וְאִישׁ מִלְחָמָה, הֶחָכָם הַגָּדוֹל, הַמָּעֹז וְהַמִּגְדּוֹל, מ״ר וְר׳ חֲנַנְאֵל, אֱלֹהִים יְמַלֵּא מַאֲוַיָּיו, מִסְפַּר יְמֵי חַיָּיו, בֶּן כְּבוֹד ר׳ יְשׁוּעָה, בְּגַן אֱלֹהִים יִהְיֶה חֶבְלוֹ, וְכִסֵּא כָבוֹד יַנְחִילוֹ, מִכְּנַף הָאָרֶץ נִשְׁמַע שָׁמְעוֹ, וּמֵאֲפָסֶיהָ נוֹדַע דֵּעוֹ, כִּי הַזְּמָן יְשׁוֹרֵר חַסְדּוֹ, וּלְשׁוֹן יְמוֹתָיו תְּחַוֶּה כְּבוֹדוֹ, וּשְׂפַת לֵילוֹתָיו תָּשִׂיחַ הוֹדוֹ. הָרִים יִפְצְחוּ רִנָּה בְּטוּבוֹ, יַעַר וְכָל עֵץ בּוֹ, אֶבֶן מִקִּיר צִדְקוֹ תְּתַנֶּה, וְכָפִיס מֵעֵץ יַעֲנֶה, הוֹלְכֵי מִדְבָּרוֹת יְזַמְּרוּ לוֹ, רוֹכְבֵי יַמִּים יְסַפְּרוּ גָדְלוֹ, עוֹבְרֵי אֳרָחוֹת יַבִּיעוּ מַהֲלָלוֹ, וְהָעֵת אוֹתוֹ יַזְכִּירוּ, יִתְרוֹעֲעוּ אַף יָשִׁירוּ, כֻּלָּם יַעֲנוּ יַחַד, וְיֹאמְרוּ שְׁכֶם אֶחָד, עֵץ חַיִּים פָּרַח – בִּפְאַת מִזְרָח, וְכוֹכַב שֵׂכֶל זָרַח, וּלְחוֹף יַמִּים אָרַח, יָפִיץ אוֹרוֹ פְּרִי עֶרֶץ, לֶאֱחֹז בְּכַנְפוֹת הָאָרֶץ. פָּנָיו שֶׁמֶשׁ מַכְלִימִים, וְאוֹרָיו סַהַר מַעֲלִימִים, וְיָדָיו תֵּבֵל מַגְשִׁימִים, וּתְבוּנָיו פְּתָאִים מַחְכִּימִים, וּמִלָּיו כֹּשֵׁל מְקִימִים, וְנִיבָיו חוֹלִים מַחְלִימִים, דְּבַר עַוְלָתָה לֹא נִמְצָא בוֹ וָדֹפִי, מָלֵא חָכְמָה וּכְלִיל יֹפִי, אֹהֵב חָכְמָה מִכָּל מִסְחָר, וְדַעַת מֵחָרוּץ בָּהָר, וַיִּתֵּן אֶת לִבּוֹ בְעָצְמָה, לִדְרוֹשׁ וְלָתוּר בְּחָכְמָה, וַיָּצַר עָלֶיהָ וַיִּשְׂכָּל, וַיָּשַׁר אֵלֶיהָ וַיּוּכָל. גִּלָּה יְסוֹדוֹתֶיהָ, וְהֶעֱרָה אָשְׁיוֹתֶיהָ, עַד רֶדֶת חוֹמוֹתֶיהָ, וַיִּלְכְּדֶנָּה עִם בְּנוֹתֶיהָ. קְרָאוֹ לִבּוֹ לִשְׁכּוֹן גְּבוֹהִים, וַיְּמַלֵּא אוֹתוֹ רוּחַ אֱלֹהִים, וּמִמַּעַל לְכוֹכְבֵי אֵל הוּקַם מַצָּבוֹ, וְלֹא גָבַהּ לִבּוֹ, וּבֵין כַּנְפֵי כְרוּבִים יְעוּפוּן רַעְיוֹנָיו, וְלֹא רָמוּ עֵינָיו, וְעַל פְּנֵי עָבִים נִתְכְּנוּ לוֹ מַעֲלוֹת, וְלֹא הָלַךְ בְּנִפְלָאוֹת וּבִגְדוֹלוֹת, עַד כִּי חֲכָמִים לְפָנָיו נוֹאָלוּ, וּנְבוֹנִים מִפָּנָיו נִבְהָלוּ, וּמַשְׂכִּילִים לְעֻמָּתוֹ נִסְכְּלוּ, כַּף שָׂמוּ לְפִיהֶם, וְכַחֲגָבִים הָיוּ בְעֵינֵיהֶם, וְלֹא מָצְאוּ כָל אַנְשֵׁי חַיִל יְדֵיהֶם. וְאַחֲרֵי הוֹדִיעַ יְיָ אוֹתִי אֶת כָּל זֹאת, עוֹרַרְתִּי רוּחִי לַחֲזוֹת, אֵיךְ אֲפָאֵר בַּאֲהָבָיו נַפְשִׁי, וְאַכְתִּיר בִּכְתָבָיו רֹאשִׁי. אָז שַׂמְתִּי עֲצוֹת בְּקִרְבִּי, לִשְׁמֹר עֵצוֹת בִּלְבָבִי, וָאֹמַר בְּחַבְלֵי שִׁירִים אֶמְשְׁכֵם, וּבַעֲבוֹתוֹת אֲמָרִים אֶתְמְכֵם, נִכְבְּדוֹת אֲדַבֵּר בּוֹ, עַד אָבוֹא כַמַּיִם בְּקִרְבּוֹ, וּלְשׁוֹנִי תֶהְגֶּה צִדְקוֹתָיו, עַד אָבוֹא כַּשֶּׁמֶן בְּעַצְמוֹתָיו. אַךְ בְּבוֹאִי כִסְאֵי חֲסָדָיו לְהַלְאוֹת, רָאִיתִי נִפְלָאוֹת, וְלָחֹק אֶת הַתְּבוּנָה, אֵין קֵצֶה לַתְּכוּנָה. וְעוֹד אֲנִי מְדַבֵּר עִם לִבִּי, וּמִתְיָעֵץ עִם יַלְדֵי מַחְשָׁבִי, וְהִנֵּה צָפוֹן תָּעִיר סַמֶּיהָ, וְתֵימָן תָּבִיא בְשָׂמֶיהָ, וְאוֹר יְרוֹצֵץ, כְּבָרָק יְנוֹצֵץ, וְקוֹל נְגִינוֹת, יָשִׁיר רְנָנוֹת, וְהַמְּבַשֵּׂר יֹאמַר וְיִשְׂחַק, שְׁמוּעָה טוֹבָה מֵאֶרֶץ מֶרְחָק, וְהַקּוֹרֵא יָרוּץ וְיִקְרָא: הִנֵּה יוֹם בְּשֹׂרָה, בָּאָה תְשׁוּרָה, מִפְּנִינִים יְקָרָה, מִכְתָּב מְפֹאָר, בִּמְחוּגַת הַשֵּׂכֶל מְתֹאָר, וּבְעֵט הַדַּעַת מְבֹאָר, מַכְפִּיר לִזְהַב אוֹפִיר, מֵפִיר יְפִי שֹׁהַם וְסַפִּיר, טוֹב מִקָּנֶה וְקִדָּה, יָקָר מִכָּל כְּלִי חֶמְדָּה. אֲמָרָיו כְּמַיִם קָרִים, וְחֹזֶק צוּרִים, דְּבָרָיו כְּנֹפֶת צוּף מְתוּקִים, וְכִרְאִי מוּצָק חֲזָקִים. מִלָּיו כְּשֶׁמֶן רַכִּים, וּמִשֶּׁלֶג זַכִּים. כְּשֶׁלֶג בְּיוֹם קָצִיר נַחְשְׁבֶנּוּ, וְגֶחָלִים בָּעֲרוּ מִמֶּנּוּ, מַעֲנָיו לִמְאֹד נִפְלָאִים, וְטוּרָיו בַּתַּרְשִׁישׁ מְמֻלָּאִים, כְּנֹפֶךְ עַל סַפִּיר וְאַחְלָמָה, ושֵׁשׁ וּמֶשִׁי וְרִקְמָה. מִפְּאַת שְׁעָרָיו – הֱרִיחוּ כְּפָרָיו, וּמֵעֵבֶר טְעָמָיו – נָזְלוּ בְשָׂמָיו, וּמֵרוּחַ לִקְחוֹ – נִרְדּוֹ נָתַן רֵיחוֹ, וְהָעֵת אוֹתִי אָחֲזָה יְמִינִי, כָּל יְקָר רָאֲתָה עֵינִי, וַתִּגְבַּהּ נַפְשִׁי בְעֵינַי – מִכֹּל אֲשֶׁר הָיוּ לְפָנַי, וְכָל אֲשֶׁר הֻגַּד לִי חָזִיתִי, וְכַאֲשֶׁר שָׁמַעְתִּי כֵּן רָאִיתִי, כְּמוֹר עַל הַכְּפוֹר בְּלִי שֶׁקֶר, כְּמַרְאֵה הָעֶרֶב וְהַבֹּקֶר, גַּן חֲמוּדוֹת – עַל פַּלְגֵּי תְּעוּדוֹת, כָּל עֵץ בּוֹ נָשָׂא פִרְיוֹ, כִּי מַה־טּוּבוֹ וּמַה־יָּפְיוֹ. אָרִיתִי מוֹרִי, אָכַלְתִּי יַעְרִי, שָׁתִיתִי יֵינִי וַחֲלָבִי וְנִסְכִּי, וּפִרְיוֹ מָתוֹק לְחִכִּי. וְאַרְבַּע פַּעֲמוֹתָיו – נַחַל מָלֵא עַל כָּל גְּדוֹתָיו, בְּקַיִץ וּבְחֹרֶף לֹא יִיבַשׁ, עַל פְּנֵי אֶרֶץ כְּהֵלֶךְ דְּבָשׁ. לֹא רַעַם לוֹ וְלֹא נַהַם, שָׁם הַבְּדֹלַח וְאֶבֶן הַשֹּׁהַם, שֻׁלְחָן עָרוּךְ – וְיַיִן מָסוּךְ. וְקוֹל מֵרִיעַ, אֶל קְצֵה הָאָרֶץ יַגִּיעַ! לְכוּ לַחֲמוּ בְלֶחֶם פָּרַשְׂתִּי, וּשְׁתוּ בְיַיִן מָסַכְתִּי, לֶחֶם מְזִמָּה – וְלֹא לֶחֶם אֲדָמָה, טֶבַח מוּסָר – וְלֹא טֶבַח בָּשָׂר, יֵין אֱמוּנִים – וְלֹא יֵין גְּפָנִים, מִגְדֵּי שֵׂכֶל – וְלֹא מִגְדֵּי אֹכֶל, בִּשְׂמֵי חֲסָדִים – וְלֹא בִּשְׂמֵי נְרָדִים, שִׁירֵי תְבוּנוֹת – וְלֹא שִׁירֵי נְגִינוֹת. וַיּוֹדִיעֵנִי נִפְלְאוֹת מְזִמָּה, וַיַגֶּד־לִי תַעֲלוּמוֹת חָכְמָה, עֲנַדְתִּיו עַל גַּרְגְּרוֹתַי, קְשַׁרְתִּיו עַל אֶצְבְּעוֹתַי, שַׂמְתִּיו עַל זְרוֹעוֹתַי, לִהְיוֹת לִי לְמִשְׁמֶרֶת, לְכָבוֹד וּלְתִפְאֶרֶת, הוּא שְׂשׂוֹנִי – בְּעִתּוֹת יְגוֹנִי, וּלְרִגְעֵי חִילִי – שִׂמְחַת גִּילִי, אֲדַבֵּר בּוֹ וְעֶדְיוֹ יְשִׁיבֵנִי, אַבִּיט אֵלָיו וְיָפְיוֹ יַרְהִיבֵנִי, בֵּין צְלָעַי הִפַּלְתִּי גּוֹרָלוֹ, וּבָנִיתִי בֵּית זְבוּל לוֹ, וּבְקֶרֶב קְּרָבַי הַמְּהִימִים – מָכוֹן לְשִׁבְתּוֹ עוֹלָמִים. וּמָה הַגְּמוּל אֲשֶׁר אָקִים לְנֶגְדּוֹ, וְהַתַּשְׁלוּם אֲשֶׁר אֲשַׁלֵּם לְחַסְדּוֹ? לְשׁוֹנִי תִקְצַר מֵהַגִּיד מַעֲשֵׂהוּ, וְיָדִי תִקְצַר לִנְגֹּעַ בְּקָצֵהוּ. לָכֵן בִּרְאוֹתִי כִּי בְתַגְמוּל יָדִי – אֵין לְאֵל יָדִי, שַׂמְתִּי נַפְשִׁי בְּכַפִּי, וְקָנִיתִי בְּכָל כַּסְפִּי – לְאָמָה לְךָ נַפְשִׁי, לְבַל תֵּצֵא חָפְשִׁי, וְלִבִּי לַעֲבוֹד רְצוֹנָיו, וְלַעֲשׂוֹת לוֹ כַטּוֹב בְּעֵינָיו. וּמִי יִתֵּן לִי כֹחַ – לִדְאוֹת עַל כַּנְפֵי רוּחַ, לָבֹא אֶל מְגוּרָיו – וְלִנְפֹּל עַל צַוָּארָיו, לְהוֹדִיעוֹ סוֹדֵי חֻבִּי, וּלְהַגִּיד לוֹ כָל לִבִּי, וּלְסַפֵּר אֹדוֹת מִסְתָּרִי – וְתַהֲפוּכוֹת דּוֹרִי, דּוֹר לֹא שָׂם יְיָ נִכְחוֹ, וְלֹא נֶאֶמְנָה אֶת אֵל רוּחוֹ, דּוֹר מֶה רָמוּ עֵינָיו, וַחֲרָבוֹת שִׁנָּיו, פָּרַשׂ לְרַגְלַי מוֹקְשָׁיו, וְהִנְחִילַנִי יַרְחֵי שָׁוְא, הִרְחִיק מֵעָלַי גִּילַי, וְלֵילוֹת עָמָל מִנָּה לִי, לְמַשְׁחִית הָפַךְ הוֹדַי, וְצוּקִי חָדָשׁ עִמָּדִי, חִצָּיו אָכְלוּ כֹחִי, וַחֲמָתָם שׁוֹתָה רוּחִי, וְלֹא נֶעְצַבְתִּי אֶל הַחַיִל כִּי נִשְׁלַל, וְלֹא דָאַגְתִּי כִּי חָלַף וְאֻמְלַל, וְלֹא נִהְיֵיתִי כִּי סָפָה הַכְּבוּדָה, וְלֹא נֶחְלֵיתִי כִּי עָפָה הָעֲבוֹדָה. כִּי בְעֵין רַעְיוֹנִי – שַׂרְתִּי בֶגֶד זְמַנִּי, וּבְאִישׁוֹנֵי זְמָמַי – רָאִיתִי תַהֲפֻכוֹת יָמַי. זְמָן נַעֲווּ מִפְעֲלוֹתָיו, וְתָעוּ מַחְשְׁבוֹתָיו, נֶעְקַשׁ בְּאֹרְחוֹתָיו, וְנָלוֹז בְּמַעְגְּלוֹתָיו, יִשַּׁק – וְהוּא יִשַּׁךְ, כַּנָּחָשׁ יַחֲרִישׁ – וְכַצִּפְעוֹנִי יַפְרִישׁ, יַתְאִים לְהַפְרִיד, יִשָּׂא לְהוֹרִיד, וּלְעַקֵּשׁ יַיְשִׁיר, וּלְהוֹרִישׁ יַעֲשִׁיר. עָשְׁרוֹ כְרָקָב יִכֶל, וְחֵילוֹ יִשָּׂא רוּחַ יִקַּח הֶבֶל וְכַסְפּוֹ וּזְהָבוֹ – כִּכְלִי אֵין חֵפֶץ בּוֹ.


הִנֵּה מְעַט מַהֲלָכָיו, וְאֵלֶּה קְצוֹת דְּרָכָיו, וְאֵיךְ לֹא אָבוּז לְמִפְעָלָיו, וְאֶלְעַג לְמַעֲלָלָיו, וַאֲנִי מֵעוֹדִי – כַבִּיר מָצְאָה יָדִי, וְיָעוּף הָעֹשֶׁר – כַּאֲשֶׁר יִדְאֶה הַנֶּשֶׁר, וְכָל עֲמַל יָדַיִם – יַעֲשֶׂה לוֹ כְנָפַיִם, לְמַעַן דַּעַת יְיָ כִּי חֲזָקָה הִיא.


אֲבָל יִזְּלוּ עֵינַי, וְאֶשָּׂא בְשָׂרִי בְּשִׁנַּי, כִּי נִשְׁאַרְתִּי בוֹדֵד בְּאֶרֶץ מוֹלַדְתִּי, וְאֵין אִישׁ אִתִּי, וַאֲנִי בָהּ נֶחְשָׁב – כְּגֵר תּוֹשָׁב, וְלֹא אֶחֱזֶה גֶבֶר סְבִיבָי – מִמִּשְׁפַּחְתִּי וּמִבֵּית אָבִי. סָפוּ בָהּ מוֹרֵי דֵעָה, וְאָפְסוּ מְבִינֵי שְׁמוּעָה, גַּם נִבְנוּ עוֹשֵׂי רִשְׁעָה. עַל יְיָ הֶעְתִּירוּ, וְגָדְלוּ וַיַּעֲשִׁירוּ, וְלֹא יָדְעוּ מַחְשְׁבוֹתַי, וְלֹא הֵבִינוּ עֲצוֹתַי, כִּי טַח מֵרְאוֹת רוֹאוֹתָם, מֵהַשְׂכִּיל לִבּוֹתָם, וּפְלֵיטֵי בִין אֲשֶׁר נוֹדָעוּ, וּשְׂרִידֵי יַחַשׂ אֲשֶׁר נִשְׁמָעוּ, דַּלּוּ מֵאֱנוֹשׁ נָעוּ. עַל זֹאת אֵאָנֵק דּוּמָם, וְאֵשְׁבָה מְשׁוֹמָם, נָגוּעַ וּמְעֻשֶּׂה, כִּי רַע עָלַי הַמַּעֲשֶׂה. אָכֵן עֵת רָעָה וְהַמַּשְׂכִּיל יִדֹּם, כִּי מָה הָאָדָם – שֶׁיָּבֹא אַחֲרֵי הַמֶּלֶךְ אֵת אֲשֶׁר כְּבָר עָשׂוּהוּ.


לַיְיָ תוֹחַלְתִּי עַד בּוֹשׁ, וְאֵדַע כִּי לֹא אֵבוֹשׁ, וְחָסִיתִי בְשׁוֹכֵן עֶרֶץ, וְהַחוֹסֶה בַּיְיָ יִנְחַל אָרֶץ, וְעוֹד לַיּוֹם אֲשֶׁר אֳנִי בוֹ סוֹעֵר, כְּתַנּוּר בּוֹעֵַר, לִהְיוֹת כָּל זֵד נוֹקֵשׁ, וְכָל עוֹשֵׂי רִשְׁעָה קַשׁ, יוֹם אֲשֶׁר יַחֲמֹל עַל סְגֻלָּתוֹ – וּמַחֲזִיקִים בִּבְרִיתוֹ, כַּאֲשֶׁר יַחְמֹל אִישׁ עַל בְּנוֹ הָעוֹבֵד אוֹתוֹ, וְאַף עַל פִּי בֶן סְגֻלַּת הַבְּרוּאִים, וְרֹאשׁ הַקְּרוּאִים, הַמֻּפְלָא בְמִידוֹתָיו – וְהַיָּחִיד בְּמִפְעֲלוֹתָיו, רַבָּנָא אָבוּן יְחִי לָעַד, וַאֲשׁוּרוֹ לֹא תִמְעַד, הוּא לִי מִבְצָר – לְרִגְעִי הַצָּר, וּמִגְדָּל – אִם הַכְּאֵב יִגְדָּל, וְדֶלֶת וְחוֹמָה – מֵחֶרְדֶת מְהוּמָה, וְשֶׁמֶשׁ צְדָקָה – בְּאֹפֶל הַמְּצוּקָה, מֵרָחוֹק בְּשָׂמָיו יָפִיחוּ, מֵאַפְסֵי אֶרֶץ מוֹרָיו יָרִיחוּ. מִפִּי סְפָרָיו – אֶשְׁמַע דְבָרָיו, וּבִפְנֵי כְתָבָיו – אֶרְאֶה אֲהָבָיו, וּמִבֵּית מְגוּרָיו – יְבוֹאוּנִי יְצוּרָיו. וּפָנַי יְרוּצוּן לְחַלּוֹת, עַל מִשְׁכָּבִי בַּלֵּילוֹת, וְאֶקְרָא בְּעִתּוֹת צָרָתִי – מֵאֶרֶץ מֶרְחָק אִישׁ עֲצָתִי, וְעַתָּה מִכְּבוֹד אַלּוּפִי – יִשְׁאֲלוּ אִמְרֵי פִי. אִם יִיטַב בְּעֵינָיו – וְכָשֵׁר הַדָּבָר לְפָנָיו, לְחַזֵּק חַבְלֵי הָאַחֲוָה – וּלְהַתִּיר קִשְׁרֵי הַתַּאֲוָה, וְעַל יְדֵי אֹמֶר וְעִנְיָנִים – נִתְרָאֶה פָנִים, וּבְגַן עֵטִים וּכְתָבִים – נִתְעַלְּסָה בַאֲהָבִים. אוּלַי כְּתָבוֹ הַנָּעִים וְהַנָּאֶה – יִתְמֹךְ לְבָבִי הַנִּכְאֶה. כִּי הִנֵּה בַּשָּׁמַיִם עֵדִי, וּבַמְּרוֹמִים שָׂהֲדִי, כִּי הַנֶּפֶשׁ אֵלָיו נִכְסֶפֶת, וְהָעַיִן עָלָיו דּוֹלֶפֶת, וּמַרְאֵהוּ שׁוֹאֶפֶת, וּמִתְּשׁוּרָתוֹ דִמְעָה עוֹרֶפֶת. וְאֵין לַיְיָ מַעְצוֹר לִזְכֹּר הַנִּשְׁכָּחִים – יִקְבֹּץ הַנִדָּחִים. וְחֶבֶל הַפְּרִידָה יֵרָתֵק, וְחוּט הַתְּשׁוּקָה יִנָּתֵק, לְחַבֵּר אֶת הָאֹהֶל לִהְיוֹת אֶחָד.


אָז אֲשֶׁר פָּצוּ שְׂפָתַי אֲקַיֵּמָה, וַאֲשֶׁר נָדַרְתִּי אֲשַׁלֵּמָה, וִיבָאֵר לִי הֵיטֵב שְׁלוֹמָיו – וּמַצְפּוּנֵי עֲלוּמָיו. וּשְׁלוֹם צִיץ הַתְּעוּדָה, הַפִּטְדָּה הַחֲמוּדָה, צִיצַת הַבְּשָׂמִים – וְחֶמְדַּת הַחֲכָמִים, ר׳ יְשׁוּעָה, לֹא תְאֻנֶּה אֵלָיו רָעָה, וִיבַלֶּה שְׁנוֹתָיו בַּנְּעִימִים, וְכַחוֹל יַרְבֶּה יָמִים, יִתְבָרְכוּ בוֹ גוֹיִם וְיִתְגַדָּלוּ, וּבִשְׁמוֹ יִתְהַלָּלוּ, וְעוֹד יוֹדִיעֵנִי בְהַגִּיעַ כְּתָבִי אֵלָיו, וְיִפֹּל עַל רַגְלָיו, הֲיִמְצָא חֵן בְּעֵינָיו, אוֹ הֲיִשָּׂא פָנָיו. וְאִם יִגְדַּל אֶצְלוֹ אוֹ יִדַּל, וְהַאוֹסִיף עוֹד אִם אֶחְדָּל, וְשָׁלוֹם רַב – כְּמִמִּזְרָח לְמַעֲרָב, וּמֵעֲגָלָה לְעַקְרָב, עָלָיו וְעַל בְּנֵי מִשְׁפַּחְתּוֹ, וּלְאֶחָיו וְיַלְדֵי מַחְבַּרְתּוֹ, וּלְדוֹדָיו וְהַסָּרִים אֶל מִשְׁמַעְתּוֹ, וְגִיל פּוֹרֵחַ, וְאוֹר זוֹרֵחַ, וְרֹב שָׁלוֹם עַד בְּלִי יָרֵחַ. לְמַעַן יְיָ אֱלֹהֶיךָ וְלִקְדוֹשׁ יִשְׂרָאֵל כִּי פֵאֲרָךְ.
