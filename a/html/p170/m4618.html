לגביר דעות וחוסן ישועות / ר' משה אבן עזרא

אל רבנו חננאל בן ישועה


לִגְבִיר דֵּעוֹת

וְחֹסֶן יְשׁוּעוֹת

וְחַיִל וְצָבָא

וְחוֹמָה נִשְׂגָּבָה

וְשֶׁמֶשׁ בְּעֵיפָה

וְצֵל סֶלַע-כָּבֵד בְּאֶרֶץ עֲיֵפָה

אִישׁ-חֲמֻדוֹת

צוּר-מָעוֹז וּבֵית מְצוּדוֹת

צִיץ הַנֵּזֶר

אֶבֶן הָעֶזֶר

יְסוֹד הָאֱמוּנָה

סוֹד הַדַּעַת וְהַתְּבוּנָה

הַר הַמּוֹר גִּבְעַת הַלְּבוֹנָה

מַשְׂכִּיל בְּכָל-חָכְמָה

גִּבּוֹר חַיִל וְאִישׁ מִלְחָמָה

הֶחָכָם הַגָּדוֹל

הַמָּעוֹז הַמִּגְדּוֹל

מר' ור' חֲנַנְאֵל

אֱלֹהִים יַרְאֵהוּ מַאֲוַיָּיו

מִסְפַּר יְמֵי חַיָּיו

בֶּן-כְּבוֹד ר' יְשׁוּעָה

בְּגַן-אֱלֹהִים יִהְיֶה חֶבְלוֹ

וְכִסֵּא כָבוֹד יַנְחִילוֹ

מִכְּנַף הָאָרֶץ נִשְׁמַע שִׁמְעוֹ

וּמֵאֲפָסֶיהָ נוֹדַע טִבְעוֹ

פִּי הַזְּמָן יְשׁוֹרֵר חַסְדּוֹ

וּלְשׁוֹן יְמוֹתָיו– תְּחַוֶּה כְבוֹדוֹ

וּשְׂפַת לֵילוֹתָיו–תָּשִׂיחַ הוֹדוֹ

הָרִים יִפְצְחוּ רִנָּה בְּטוּבוֹ

יַעַר וכָל-עֵץ בּוֹ

אֶבֶן מִקִּיר צִדְקוֹ תְּתַנֶּה

וְכָפִיס מֵעֵץ יַעֲנֶה

הֹלְכֵי מִדְבָּרוֹת יְזַמְּרוּ-לוֹ

רֹכְבֵי יַמִּים יְסַפְּרוּ גָדְלוֹ

עֹבְרֵי אֳרָחוֹת יַבִּיעוּ מַהֲלָלוֹ

וְהָעֵת אֹתוֹ יַזְכִּירוּ

יִתְרוֹעֲעוּ אַף-יָשִירוּ

כֻּלָּם יַעֲנוּ יַחַד

וְיֹאמְרוּ שְׁכֶם אֶחָד

עֵץ חַיִּים פָּרַח

בִּפְאַת מִזְרָח

וְכוֹכַב שֵׂכֶל זָרַח

וְּלְחוֹף יָם אָרַח

יָפִיץ אוֹרוֹ עֲדֵי עֶרֶץ

לֶאֱחֹז בְּכַנְפוֹת הָאָרֶץ

פָּנָיו שֶׁמֶשׁ מַכְלִימִים

וְאוֹרָיו סַהַר מַעְלִימִים

וְיָדָיו תֵּבֵל מַגְשִׁימִים

וּתְבוּנָיו פְּתָאִים מַחְכִּימִים

וּמִלָּיו כּוֹשֵׁל מְקִימִים

וְנִיבָיו חוֹלִים מַחְלִימִים

דְּבַר עַוְלָתָה לֹא נִמְצָא-בוֹ וְדֹפִי

מָלֵא חָכְמָה וּכְלִיל יֹפִי

אָהַב חָכְמָה מִכָּל-מִסְחָר

וְדַעַת מֵחָרוּץ בָּחַר

וַיִּתֵּן אֶת-לִבּוֹ בְּעָצְמָה

לִדְרוֹשׁ וְלָתוּר בַּחָכְמָה

וַיָּצַר עָלֶיהָ וַיִּשְׂכַּל

וַיָּשַׂר אֵלֶיהָ וַיֻּכַל

גִּלָּה יְסֹדוֹתֶיהָ

וְהֶעֱרָה אָשְׁיוֹתֶיהָ

עַד רֶדֶת חוֹמוֹתֶיהָ

וַיִּלְכְּדָהּ עִם-בְּנוֹתֶיהָ

קְרָאוֹ לִבּוֹ לִשְׁכֹּן גְּבֹהִים

וַיְמַלֵּא אֹתוֹ רוּחַ אֱלֹהִים

וּמִמַּעַל לְכוֹכְבֵי-אֵל הוּקַם מַצָּבוֹ

וְלֹא-גָבַהּ לִבּוֹ

וּבֵין כַּנְפֵי כְרוּבִים יְעוּפוּן רַעְיוֹנָיו

וְלֹא-רָמוּ עֵינָיו

וְעַל-פְּנֵי עָבִים נִתְכְּנוּ-לוֹ מַעֲלוֹת

וְלֹא-הִלֵּךְ בְּנִפְלָאוֹת וּבִגְדֹלוֹת

עַד כִּי-חֲכָמִים לְפָנָיו נִסְכָּלוּ

וּנְבֹנִים מִפָּנָיו נִבְהָלוּ

וּמַשְׂכִּילִים לְעֻמָּתוֹ נֹאָלוּ

כַּף שָׂמוּ לְפִיהֶם

וְכַחֲגָבִים הָיוּ בְעֵינֵיהֶם

וְלֹא-מָצְאוּ כָל-אַנְשֵׁי-חַיִל יְדֵיהֶם. – וְאַחֲרֵי הוֹדִיעַ אֱלֹהִים אֹתִי אֶת-כָּל-זֹאת

עוֹרַרְתִּי אֶת-רוּחִי לַחֲזוֹת

אֵיךְ אֲפָאֵר בַּאֲהָבָיו – נַפְשִׁי

וְאַכְתִּיר בִּכְתָבָיו – רֹאשִׁי

אָז שַׂמְתִּי בְנַפְשִׁי לִשְׁמֹר

עֵצוֹת בִּלְבָבִי לֵאמֹר

בְּחַבְלֵי שִׁירִים – אֶמְשָׁכֵם

וּבַעֲבֹתוֹת אֲמָרִים – אֶתְמָכֵם

נִכְבָּדוֹת אֲדַבֶּר-בּוֹ

עַד-אָבוֹא כַמַּיִם בְּקִרְבּוֹ

וּלְשׁוֹנִי תֶהְגֶּה צִדְקוֹתָיו

עַד-אָבוֹא כַשֶּמֶן בְּעַצְמוֹתָיו

אַךְ בְּבֹאִי חֲסָדָיו לְהַתְאוֹת

רָאִיתִי נִפְלָאוֹת

וְלָחֹק אֶת-הַתְּבוּנָה

וְאֵין קֵצֶה לַתְּכוּנָה.

וְעוֹד אֲנִי מְדַבֵּר עִם-לְבָבִי

וּמִתְיָעֵץ עִם-יַלְדֵי מַחֲשָׁבִי

וְהִנֵּה צָפוֹן תָּעִיר סַמֶּיהָ

וְתֵימָן תָּבִיא בְשָׂמֶיהָ

וְאוֹר יְנוֹצֵץ

כַּבָּרָק יְרוֹצֵץ

וְקוֹל נְגִינוֹת

וְשִׁיר רְנָנוֹת

וְהַמְבַשֵּׂר יֹאמַר וְיִשְׂחַק

שְׁמוּעָה טוֹבָה מֵאֶרֶץ מֶרְחָק

וְהַקֹּרֵא יָרוּץ וְיִקְרָא

הִנֵּה יוֹם-בְּשׂוֹרָה

בָּאָה תְשׁוּרָה

מִפְּנִינִים יְקָרָה

מִכְתָּב מְפֹאָר

בִּמְחוּגַת הַשֵּׂכֶל מְתֹאָר

וּבְעֵט הַדַּעַת מְבֹאָר

מַחְפִּיר

זְהַב אוֹפִיר

מֵפִיר

יְפִי שֹׁהַם וְסַפִּיר

טוֹב מִקָּנֶה וְקִדָּה

יָקָר מִכֹּל כְּלִי חֶמְדָּה

אֲמָרָיו כְּמַיִם קָרִים

וְחֹזֶק צֻרִים

דְּבָרָיו כְּנֹפֶת צוּפִים מְתוּקִים

וְכִרְאִי מוּצָק חֲזָקִים

מִלָּיו כַּשֶּׁמֶן רַכִּים

וּמִשֶּׁלֶג זַכִּים

כְּשֶׁלֶג בְּיוֹם קָצִיר תַּחְשְׁבֶנּוּ

וְגֶחָלִים בָּעֲרוּ מִמֶּנּוּ

מַעֲנָיו מְאֹד נִפְלָאִים

וְטוּרָיו בְּתַרְשִׁישׁ מְמֻלָּאִים

כְּנֹפֶךְ עַל-סַפִּיר וְאַחְלָמָה

וְשֵׁשׁ וָמֶשִׁי וְרִקְמָה

מִפְּאַת טוּרָיו

הֵפִיחוּ כְפָרָיו

וּמֵעֵבֶר טְעָמָיו

נָזְלוּ בְשָׂמָיו

וּמֵרוּחַ לִקְחוֹ

נִרְדּוֹ נָתַן רֵיחוֹ

וְהָעֵת אֹתוֹ אָחֲזָה יְמִינִי

כָּל-יְקָר רָאֲתָה עֵינִי

וַתִּגְבַּהּ נַפְשִׁי בְעֵינַי

מִכֹּל אֲשֶׁר-הָיוּ לְפָנַי

וְכָל-אֲשֶׁר הֻגַּד לִי חָזִיתִי

וְכַאֲשֶׁר שָׁמַעְתִּי כֵּן רָאִיתִי

כְּמֹר עַל-הַכְּפֹר בְּלִי-שֶׁקֶר

כְּמַרְאֵה הָעֶרֶב וְהַבֹּקֶר

גַּן חֲמוּדוֹת

עַל-פַּלְגֵי תְעוּדוֹת

כָּל-עֵץ בּוֹ נָתַן פִּרְיוֹ

מַה-טּוּבוֹ וּמַה-יָּפְיוֹ

אָרִיתִי מוֹרִי

אָכַלְתִּי יַעְרִי

שָׁתִיתִי יֵינִי וַחֲלָבִי וְנִסְכִּי

וּפִרְיוֹ מָתוֹק לְחִכִּי

וְאַרְבַּע פַּעֲמוֹתָיו

נַחַל מָלֵא עַל-כָּל-גְּדוֹתָיו

בַּקַּיִץ וְבַחֹרֶף לֹא-יִיבַשׁ

עַל-פְּנֵי אֶרֶץ כְּהֵלֶךְ דְּבַשׁ

לֹא-רַעַם לוֹ וְלֹא-נַהַם

שָׁם הַבְּדֹלַח וְאֶבֶן הַשֹּׁהַם

שֻׁלְחָן עָרוּךְ

וְיַיִן מָסוּךְ

וְקוֹל מֵרִיעַ

לִקְצֵה-אֶרֶץ יַגִּיעַ

לְכוּ לַחֲמוּ בְּלֶחֶם פָּרַסְתִּי

וּשְתוּ בְּיַיִן מָסַכְתִּי

לֶחֶם מְזִמָּה

וְלֹא לֶחֶם אֲדָמָה

טֶבַח מוּסָר

וְלֹא טֶבַח בָּשָׂר

יֵין אֱמוּנִים

וְלֹא יֵין גְּפָנִים

מִגְדֵי שֵׂכֶל

וְלֹא מִגְדֵי אֹכֶל

בִּשְׂמֵי חֲסָדִים

וְלֹא בִשְׂמֵי נְרָדִים

שִׁירֵי תְבוּנוֹת

וְלֹא שִירֵי נְגִינוֹת

וַיּוֹדִיעֵנִי נִפְלְאוֹת מְזִמָּה

וַיַּגֶּד-לִי תַּעֲלֻמוֹת חָכְמָה

עֲנַדְתִּיו – עַל-גַּרְגְּרוֹתַי

קְשַרְתִּיו – עַל-אֶצְבְּעוֹתַי

שַׂמְתִּיו – עַל-זְרֹעוֹתַי

לִהְיוֹת לִי לְמִשְׁמֶרֶת

לְכָבוֹד וּלְתִפְאָרֶת.

הוּא שְׂשׂוֹנִי

בְּעִתּוֹת יְגוֹנִי

וּלְרִגְעֵי חִילִי

שִׂמְחַת גִּילִי

אֲדַבֶּר-בּוֹ וְעֶדְיוֹ יְשִׁיבֵנִי

אַבִּיט אֵלָיו וְיָפְיוֹ יַרְהִיבֵנִי

בֵּין צְלָעַי הִפַּלְתִּי גוֹרָלו

ֹוּבָנִיתִי בֵּית זְבֻל לוֹ

וּבְקֶרֶב קְרָבַי הַמְּהִימִים

מָכוֹן לְשִׁבְתּוֹ עוֹלָמִים.

וּמָה הַגְּמוּל אֲשֶׁר אָשִׁיב לְנֶגְדּוֹ

וְהַשִּׁלֻּם אֲשֶׁר אֲשַׁלֵּם לְחַסְדּו

ֹוּלְשׁוֹנִי נֶעְצָר – מֵהַגִּיד מַעֲשֵׂהוּ

וְיָידִי תִקְצַר – לִנְגֹּעַ בְּקָצֵהוּ

לָכֵן בִּרְאֹתִי כִּי לְתַגְמוּל יְדִידִי

אֵין לְאֵל יָדִי

שַׂמְתִּי בְכַפּוֹ

מִקְנַת כַּסְפּוֹ

לְאָמָה נַפְשִׁי

לְבַל-תֵּצֵא חָפְשִׁי

וְלִבִּי לְעֶבֶד רְצוֹנָיו

וְלַעֲשׂוֹת בּוֹ כַּטּוֹב בְּעֵינָיו.

וּמִי יִתֶּן-לִי כֹחַ

לִדְאוֹת עַל כַּנְפֵי-רוּחַ

לָבוֹא אֶל-מְגוּרָיו

וְלִנְפֹּל עַל-צַוָּארָיו

לְהוֹדִיעוֹ סוֹדֵי חֻבִּי

וּלְהַגִּיד לוֹ אֶת-כָּל-לִבִּי

וּלְסַפֵּר אֹדוֹת מִסְתּוֹרִי

וְתַהְפֻּכוֹת דּוֹרִי

דּוֹר לֹא-שָׂם אֱלֹהִים נִכְחוֹ

וְלֹא-נֶאֶמְנָה אֶת-אֵל רוּחוֹ

דּוֹר מָה-רָמוּ עֵינָיו

וַחֲרָבוֹת שִׁנָּיו

פָּרַשׂ לְנַפְשִׁי מֹקְשָׁיו

וְהִנְחִילַנִי יַרְחֵי-שָׁוְא

הִרְחִיק מֵעָלַי גִּילִי

וְלֵילוֹת עָמָל מִנָּה-לִּי

לְמַשְׁחִית הָפַךְ הוֹדִי

וְצוֹקוֹ חָדָשׁ עִמָּדִי

חִצָּיו אָכְלוּ כֹחִי

וַחֲמָתָם שֹׁתָה רוּחִי

וְלֹא נֶעְצַבְתִּי – אֶל-הַחַיִל כִּי נִשְׁלַל

וְלֹא דָאַגְתִּי – כִּי-חָלַף וְאֻמְלַל

וְלֹא נִהְיֵיתִי – עַל כִּי-סָפָה – הַכְּבוּדָּה

וְלֹא נֶחֱלֵיתִי – כִּי-עָפָה –הָעֲבֻדָּה

כִּי בְעֵין רַעְיוֹנִי

שַׁרְתִּי בֶּגֶד זְמַנִּי

וּבְאִישׁוֹנֵי זְמָמַי

רָאִיתִי תַּהְפֻּכוֹת יָמַי

זְמָן נַעֲווּ מִפְעֲלוֹתָיו

וְתָעוּ מַחְשְׁבוֹתָיו

נֶעְקָשׁ בְּאֹרְחוֹתָיו

וְנָלוֹז בְּמַעְגְּלוֹתָיו

יִשַּׁק

וְהוּא כְּנָחָשׁ יִשָּׁךְ

יַחְרִישׁ

וּכְצִפְעֹנִי יַפְרִישׁ

יַתְאִים לְהַפְרִיד

יִשָּׂא לְהוֹרִיד

וּלְעַקֵּשׁ יַיְשִׁיר

וּלְהוֹרִישׁ יַעֲשִׁיר

עָשְׁרוֹ כְּרָקָב יִבֶל

וְחֵילוֹ יִשָּׂא-רוּחַ יִקַּח הָבֶל

וְכַסְפּוֹ וּזְהָבוֹ

כִּכְלִי אֵין-חֵפֶץ בּוֹ.

הִנֵּה מְעַט מִמַּהֲלָכָיו

וְאֵלֶּה קְצוֹת דְּרָכָיו

וְאֵיךְ לֹא אָבוּז לְמִפְעָלָיו

וְאֶלְעַג לְמַעֲלָלָיו

וַאֲנִי מֵעוֹדִי

כַּבִּיר מָצְאָה יָדִי

וְיָעוּף הָעשֶׁר

כַּאֲשֶׁר יִדְאֶה הַנָּשֶׁר

וְכָל-עֲמַל יָדַיִם

יַעֲשֶׂה-לּוֹ כְנָפַיִם

לְמַעַן דַּעַת יַד אֲדֹנָי כִּי חֲזָקָה הִיא.

אֲבָל יִזְּלוּ עֵינַי

וְאֶשָּׂא בְשָׂרִי בְשִׁנַּי

כִּי נִשְׁאַרְתִּי בוֹדֵד בְּאֶרֶץ מוֹלַדְתִּי

וְאֵין אִישׁ אִתִּי

וַאֲנִי בָהּ נֶחְשָׁב

כְּגֵר תּוֹשָׁב

וְלֹא אֶחֱזֶה גֶבֶר סְבִיבִי

מִמִּשְׁפַּחְתִּי וּמִבֵּית אָבִי

סָפוּ – בָהּ מוֹרֵי דֵעָה

וְנֶאֱסָפוּ – מְבִינֵי שְׁמוּעָה

גַּם-נִבְנוּ עֹשֵׂי רִשְׁעָה

עַל-אֲדֹנָי דִּבְרֵיהֶם הֶעְתִּירוּ

וְגָדְלוּ וַיַּעֲשִׁירוּ

וְלֹא יָדְעוּ מַחֲשַׁבְתּוֹ

וְלֹא-הֵבִינוּ עֲצָתוֹ

כִּי טַח מֵרְאוֹת רֹאוֹתָם

מֵהַשְׂכִּיל לִבּוֹתָם

וּפְלִיטֵי בִין אֲשֶׁר נוֹדָעוּ

וּשְׂרִידֵי יַחַשׂ אֲשֶׁר נִשְׁמָעוּ

דַּלּוּ מֵאֱנוֹשׁ נָעוּ.

עַל-זֹאת אֵאָנֵק דּוֹמֵם

וְאֵשְׁבָה מְשׁוֹמֵם

נָגוּעַ וּמְנֻסֶּה

כִּי רַע עָלַי הַמַּעֲשֶׂה

אָכֵן עֵת רָעָה הִיא וְהַמַּשְׂכִּיל יִדֹּם, כִּי מֶה הָאָדָם שֶׁיָּבוֹא אַחֲרֵי הַמֶּלֶךְ אֵת אֲשֶׁר-כְּבָר עָשׂוּהוּ. לַאדֹנָי הוֹחַלְתִּי עַד-בּוֹשׁ

וָאֵדַע כִּי-לֹא אֵבוֹשׁ

וְחָסִיתִי בְּשֹׁכֵן עֶרֶץ

וְהַחוֹסֶה בוֹ יִנְחַל-אָרֶץ.

וְעוֹד לַיּוֹם אֲשֶׁר אַפּוֹ בּוֹ סֹעֵר

כַּתַּנּוּר בֹּעֵר

לִהְיוֹת כָּל-זֵד נוֹקָשׁ

וְכָל-עֹשֵׂי רִשְׁעָה קַשׁ

יוֹם אֲשֶׁר יַחְמֹל עַל-סְגֻלָּתוֹ

וּמַחֲזִיקִים בִּבְרִיתוֹ

כַּאֲשֶׁר יַחְמֹל אִישׁ עַל-בְּנוֹ הָעֹבֵד אֹתוֹ

וְאַף כִּי-סְגֻלַּת הַבְּרוּאִים

וְרֹאשׁ הַקְּרוּאִים

הַמֻּפְלַא בְמִדּוֹתָיו

הַיָּחִיד בְּמִפְעֲלוֹתָיו

רַבָּנָא אָבוּן יְחִי לָעַד

וַאֲשֻׁרוֹ לֹא תִמְעַד

הוּא לִי מִבְצָר

לְרִגְעֵי הַצַּר

וּמִגְדָּל

אִם-הַכְּאֵב יִגְדַּל

וְדֶלֶת וְחוֹמָה

מֵחֶרְדַּת הַמְּהוּמָה

וְשֶׁמֶשׁ צְדָקָה

בְּאֹפֶל הַמְּצוּקָה

מֵרָחוֹק בְּשָׂמָיו יָפִיחוּ

וּמֵאַפְסֵי-אֶרֶץ מוֹרָיו יָרִיחוּ

מִפִּי סְפָרָיו

אֶשְׁמַע דְּבָרָיו

וּבִפְנֵי כְתָבָיו

אֶרְאֶה אֲהָבָיו

וּמִבֵּית מְגוּרָיו

יְבֹאוּנִי יְצֻרָיו

וּפָנַי יְרֻצוּן לְחַלּוֹת

עַל-מִשְׁכָּבִי בַּלֵּילוֹת

וְאֶקְרָא בְּעִתּוֹת צָרָתִי

מֵאֶרֶץ מֶרְחָק אִישׁ עֲצָתִי.

וְעַתָּה מִכְּבוֹד אַלּוּפִי

יִשְׁאֲלוּ אִמְרֵי-פִי

אִם-יִיטַב בְּעֵינָיו

וְכָשֵׁר הַדָּבָר לְפָנָיו

לְחַזֵּק חַבְלֵי הָאֲחֲוָה

וּלְהַתִּיר קִשְׁרֵי הַתַּאֲוָה

וְעַל-יְדֵי אֹמֶר וְעִנְיָנִים

נִתְרָאֶה פָנִים

וּבְגַן עֵטִים וּכְתָבִים

נִתְעַלְּסָה בָּאֳהָבִים

אוּלַי כְּתָבוֹ הַנָּעִים וְהַנָּאֶה

יִתְמֹךְ לְבָבִי הַנִּכְאֶה

כִּי הִנֵּה בַשָּׁמַיִם עֵדִי

וּבַמְּרוֹמִים שָׁהֲדִי

כִּי-הַנֶּפֶשׁ אֵלָיו נִכְסֶפֶת

וְהָעַיִן אֵלָיו דֹּלֶפֶת

וּמַרְאֵהוּ שֹׁאֶפֶת

וּמִתְּשׁוּקָתוֹ דִּמְעָה עֹרֶפֶת

וְאֵין לַאדֹנָי מַעְצוֹר לִזְכֹּר הַנִּשְׁכָּחִים

וְלִקְבֹּץ הַנִּדָּחִים

וְחֶבֶל הַפְּרִידָה יֵרָתֵק

וְחוּט הַתְּשׁוּקָה יִנָּתֵק

לְחַבֵּר אֶת-הָאֹהֶל לִהְיוֹת אֶחָד, אָז אֲשֶׁר-פָּצוּ שְׂפָתַי אֲקַיֵּמָה

וַאֲשֶׁר נָדַרְתִּי אֲשַׁלֵּמָה.

וִיבָאֵר לִי הֵיטֵב שְׁלוֹמָיו

וּמַצְפֻּנֵי עֲלֻמָיו

וּשְׁלוֹם צִיץ הַתְּעוּדָה

הַפִּטְדָה הַחֲמוּדָה

צִיצַת הַבְּשָׂמִים

וְחֶמְאַת הַחֲכָמִים

ר' יְשוּעָה

לֹא תְאֻנֶּה אֵלָיו רָעָה

וִיבַלֶּה שְׁנוֹתָיו בַּנְּעִימִים

וְכַחוֹל יַרְבֶּה יָמִים

יִתְבָּרְכוּ בוֹ גוֹיִם וְיִתְגַּדָּלוּ

וּבִשְׁמוֹ יִתְהַלָּלוּ.

וְעוֹד יוֹדִיעֵנִי בְּהַגִּיעַ כְּתָבִי אֵלָיו

וְיִפֹּל עַל-רַגְלָיו

הֲמָצָא חֵן בְּעֵינָיו

אוֹ הֲנָשָׂא פָנָיו

וְאִם יִגְדַּל

אֶצְלוֹ אוֹ יִדַּל

וְהַאוֹסִף עוֹד אִם-אֶחְדָּל.

וְשָלוֹם רָב

כְּמִמִּזְרָח לְמַעֲרָב

וּמֵעֲגָלָה לְעַקְרָב

עָלָיו וְעַל-בְּנֵי מִשְׁפַּחְתּוֹ

וּלְאֶחָיו וְיַלְדֵי מַחְבַּרְתּוֹ

וּלְדוֹדָיו וְהַסָּרִים אֶל-מִשְׁמַעְתּוֹ

וְגִיל פּוֹרֵחַ

וְאוֹר זוֹרֵחַ

וְרֹב שָלוֹם עַד-בְּלִי יָרֵחַ

לְמַעַן אֲדֹנָי אֱלֹהֶיךָ וְלִקְדוֹשׁ יִשְׂרָאֵל כִּי פֵאֲרָךְ.
