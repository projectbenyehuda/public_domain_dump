[בַּצָּרָתָה לִי בְּקָרְאִי עֲצָרוֹת (תְחִנָּה)]

תְּחִנָּה

בַּצָּרָתָה לִי בְּקָרְאִי עֲצָרוֹת

נִגַּשׂ נַעֲנֶה בִּתְלָאוֹת וְצָרוֹת

יָהּ יַעֲנֶה בַמֶּרְחָב וְיִקְרָא דְרוֹרוֹת

מִמַּסְגֵּר לְהוֹצִיא אֲסִירִים בַּכּוֹשָׁרוֹת

נְקַדְּמָה פָנָיו בְּתוֹדָה וּבִזְמִירוֹת

עַל-הַנִּסִּים וְעַל-הַגְּבוּרוֹת.


אֵימוֹת מָוֶת עָלַי נָפָלוּ

בִּקְרֹב עָלַי מְרֵעִים שָׁלוּ

בְּעַד הַחַיִּים לַמֵּתִים יִשְׁאָלוּ

אֲזַי חַיִּים בְּלָעוּנוּ בַּחֲרוֹת.


גּוֹיִם בְּנָפַת שָׁוְא לַהֲנִיפֵהֶם

בְּמֶתֶג וּבְרֶסֶן מַתְעֶה עַל-לְחֵיהֶם

דָּתֵיהֶם שׁוֹנוֹת תָּמִיד בְּהַבְלֵיהֶם

וַיֵּלְכוּ בְּמֹעֵצוֹת בִּשְׁרִירוּת.


הֲמוֹן עָרִיצִים עַד לַשָּׁמָיִם

יַעֲלֶה עַל-עַם מְיַחֲדֵי פַעֲמָיִם

וְיַעֲרִימוּ סוֹד לַעֲזֹב מְקוֹר מַיִם

חַיִּים לַחְצֹב לָהֶם בֹּארוֹת.


זִמַּת אִוֶּלֶת חַטַּאת נְבָלִים

מַעֲשֵׂה תַּעְתֻּעִים מוּסַר הֲבָלִים

חֻקּוֹתָם חִקְקֵי אָוֶן וַעֲמָלִים

אַחֲרֵי הַבְּעָלִים וְהַעָשְׁתָּרוֹת.


טוֹעֲנִים וְנוֹשְׂאִים עֲצֵי פִסְלָם

נוֹהֲמִים כַּהֲמוֹת יַמִּים בְּקוֹלָם

יְשַׁוְּעוּ וְאֵין מוֹשִׁיעַ וְאוּלָם

מַעַלְלֵיהֶם אֶל-יְיָ לַמְרוֹת.


כְּגֶבֶר עֲבָרוֹ יַיִן לְהָנִיעָה

בָּעֲרוּ נוֹאֲלוּ מִבְּלִי דֵעָה

לַעֲשׂוֹת חֹנֶף וּלְדַבֵּר תֹּעָה

כִּי חָלְפוּ חֹק עָבְרוּ תוֹרֹת.


מִקּוֹל שְׁאוֹנָם לְבָבִי יֵחָת

רוֹגְזִים וְשׂוֹחֲקִים וְאֵין-לִי נָחַת

נָתְנוּ פַח-לִי פַּחַד וָפָחַת

שׁוּחָה עֲמֻקָּה פִּי זָרוֹת.


סֵתֶר וּמָנוֹס אָבַד מִפָּנָי

אֵין-לִי מַכִּיר בְּצִבְאוֹת מוֹנָי

עַל-זֹאת עוּרִי זְרוֹעַ יְיָ

עוּרִי כִּימֵי קֶדֶם דּוֹרוֹת.


פֶּן-יֹאמְרוּ רָמָה יָדָם אֱוִילִים

בּוֹעֲרִים הַמִּתְהַלֲלִים בָּאֱלִילִים

צַוֵּה וְיִמּוֹטוּ עֲלֵיהֶם גֶּחָלִים

בָּאֵשׁ יַפִּלֵם בְּמַהֳמֹרוֹת.


קַנּוֹא וְנוֹקֵם הֲתֶחֱשֶׁה עַד-מָה

רַבַּת צְרָרוּנוּ זֶה כַמָּה

רְאֵה כִּי לְבַד-בְּךָ נַזְכִּיר וְקוּמָה

יְיָ בְּאַפֶּךָ הִנָּשֵׂא בְּעַבְרוֹת.


שֵׁם קָדְשְׁךָ הַמְחֻלָּל בָּעַמִּים

הוֹדַע וְקַדֵּשׁ לְעֵין כָּל-אֻמִּים

תִּתֵּן לְעַמְּךָ תְּשׁוּעַת עוֹלָמִים

תַּחַת אֲשֶׁר הִרְאִיתָנוּ צָרוֹת.


בִּיהוּדָה נוֹדָע אֱלֹהִים אֶחָד

מֵאָז בְּיִשְׂרָאֵל גָּדוֹל וּמְיֻחָד

נִמְצָא בְּכָל-עֵת לְדוֹרְשָׁיו בְּלִי-כָּחַד

מַחֲסֶה וָעֹז עֶזְרָה בְצָרוֹת.


יֵבשׁ כָּל-פֶּסֶל וְיִּכָּלְמוּ עוֹבְדָיו

מוֹשִׁיעַ חוֹסִים בְּהַפְלִיא חֲסָדָיו

נֶגֶד יְרֵאָיו יוֹדְעֵי סוֹדָיו

אֵלֶּה הַחֻקִּים וְהַמִּשְׁפָּטִים וְהַתּוֹרֹת.
