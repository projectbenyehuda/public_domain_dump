מֵעֵבֶר לַיַּמִּים

יֵשׁ מִי 

יֵשׁ מִי שֶׁקָּנַנִי לוֹ, אַחַי, וְשׁוּב לֹא יֶרֶף לִי,

עָרִיץ מִכָּל שַׁלִּיט וְנֶאֱמַן פֻּרְקָן מִכָּל עָמִית –

מִשְּׁאוֹן דַּאֲגַתְכֶם הוּא הִפְקִיעַנִי, רַק מֵעִצְּבוֹנְכֶם

הַגָּדוֹל סִנֵּן רֹן תַּמְצִיתוֹ לִי מְסֹעַר הֲמִית.


כֹּה רַבַּת חַסְדּוֹ עָלַי וְעַל רֹאשִׁי הִרְכִּין

פֶּלֶא אֳפָקִים, זִיו אֲשָׁדוֹת וְקַיִץ בְּאִבּוֹ,

וְחֻפְשָׁה לֹא יְדַעְתֶּם אַתֶּם – – – עַד כִּי פִּתְאֹם, פִּתְאֹם

בְּשׁוּקֵי עוֹלָם יְנַהֲגֵנִי: רְאוּ עַבְדִּי בָחַרְתִּי בוֹ.


יֵשׁ מִי שֶׁפָּדַנִי לְמִשְׁתֵּה שִׁפְעוֹ וּבִי לֹא יִתְחָרֵט –

עוֹד שִׁבְעָתַיִם יְגַבֵּר יְצָרַי וְחֻבֵּי גַנַּי יַרְעִישׁ;

אַךְ בְּפָקְדוֹ טַנְאֵי חֶמְדִּי – בִּפְרוֹטְרוֹט יִמְנֵם לִמְלוֹא מִכְסוֹ,

אָז רִנַּת אָסְפִּי יֵשׁ תִּתְקַטַּע בְּעֹמֶק לֵב מַר רִישׁ.


מִי רַחוּם־אַכְזָר הִזְמִין אֶצְלִי יָמַי וְלֵילוֹתַי?…

אֵלֶּה יָמַי־לִי, שֶׁלֹּא אוּכַל קָטְפָם כְּדִין אָדוֹן,

כְּבַזְבֵּז אוֹתָם כָּל אָח לִי בָּאָדָם, שֶׁאֵין לִבּוֹ נוֹקְפוֹ,

וְלֹא יַנְמִיךְ אֶת פָּנָיו אַרְצָה כְמוֹעֵל בְּפִקָּדוֹן – – –


שִׁיר הָעֶלְבּוֹן

אֲכַתֵּף עוֹד סִבְלוֹת יוֹם קֵיצִי, כִּמְאַסֵּף בְּמַפַּל אֲלֻמִּים,

וְעַל דַּרְכִּי שַׁלֶּכֶת כְּבָר תּוֹלָה בַסְּבָךְ טַנְאֵי אִשָּׁהּ;

וְלָמָּה עַל מַחֲשׂוֹף שְׁלָפַי יְלַקְּשׁוּ דְבַר־מָה עוֹד עֲלוּמִים,

שֶלֹּא נֶאֱסַף גֹּרְנָה – שֶׁנִּשְׁכַּח עַל תַּלְמֵי הַכְּמִישָׁה?


וּצְנוּעָה־עֲצוּרָה כָּל נַעֲרָה שֶׁאֶקְרֶה עַל שִׁקֲתוֹת נְדוּדִים, 

וּבְחִיּוּךְ נְזִירוֹת שְׂרַף בָּסְרָהּ אֶל כַּדִּי תְקַלַּח,

וְלֹא תֵדָע: בְּיוֹם מָתְקָהּ יִגָּדֵשׁ צְמֵא הֶפְקֵר – בְּאֹרְחַת גַּלְמוּדִים

אֲנִי בִשְׁרַב דָּמַי אֶסְתַּגָּף – – – הָהּ, אֵל טוֹב וְסַלָּח.


יֵשׁ יוֹם וְכֹה מַר אָז לִשְׁלִיחֲךָ לְכַתֵּף בְּכָל בַּיִת דַּם יֵינוֹ, 

וּלְסַנֵּן לְכָל זָר אֶת כָּל מִשְׁרַת גַּעְגּוּעִים חֲרִיפָה,

וּלְקַדֵּשׁ כָּל יֹפִי לָאִשָּׁה – וּלְשַׁכֵּךְ בְּכִי מֶרֶד,


בְּבוֹא פִּתְאֹם צַוְּךָ בְּחַג אָסְפּוֹ, כִּי קֻשַּׁט כָּל שְׁבִיל וְכָל זֶרֶד,

לְנָטְשָׁם לָזֶה שֶׁלֹּא חָלָה כָל חֶבְלֵי הַשִּׁפְעָה,

שֶׁבְּעָבְרוֹ רַק יָנִיף עַל פִּרְיָהּ גַּס לֵב לַהַב זֵינוֹ…


שִׁיר תּוֹדָה

וְכֹה לְךָ אַרְנִין תּוֹדַת נַפְשִׁי בְהַשְׁלִימָהּ

עַל גְּבוּל עֲלוּמַי עַתָּה אַחֲרוֹן מַסְלוּלָהּ:

עַל שֶשִּׁחְרַרְתַּנִי, אֵלִי, מִתַּרְמִית כָּל גְּאֻלָּה

שֶׁל קְדֻשָּׁה בִרְיַת שְׂכָרָהּ וּכְפִירָה רְזַת בְּלִימָה.


וְעַל שֶׁאִלַּפְתַּנִי תַכְלִית בּוֹא וָצֵאת, וַתָּחֹן

אֶת עַבְדְּךָ בְּשַׁלְוַת חֶדְוָה כְמוּסָה:

לָשֵׂאת לִי חִידַת סִבְלִי עַל רֻכְסֵי כָל נִצָּחוֹן,

וּפִתְרוֹן פְּדוּת לִלְכֹּד בְּשׁוּלֵי דַרְכֵי תְבוּסָה.


וְעַל שֶׁאֲמַנְתַּנִי תָּם, צָנוּעַ, וּבְדָמַי רִטְטֵי בֹסֶר,

אֲשֶׁר אֲכַבֵּד כֹּה כָּל אִישׁ שֶׁעַל בַּגְרוּתוֹ יֵלֶא:

שֶׁקַּרְדֹּם לוֹ וְתַחֲרוּת קִיּוּם וְדַאֲגַת חֹסֶר –

וַאֲנִי עֲנוּג תַּפְנוּקִים לֹא נֻסֵּיתִי עוֹד בָּאֵלֶּה.


וְעַל שֶׁעִצַּבְתַּנִי, אֵלִי, נָבוֹן, גֵּא וְאַמִּיץ כָּכָה,

וּמִכֻּלָּם בָּגִיר לְצַעַר רַב וּלְבִטְחָה:

כְּאָז בִּימֵי עֲלוּמַי גַּם עַתָּה עַל מֶרְחַקֵּי גִתְּךָ

עוֹד אֶכְפֹּן אַשְׁכְּלוֹת עִצְּבוֹנִי בְשֵׁלֵי בְרָכָה…


שִׁיר הַמַעְגָּל 

בִּמְחוֹל אֵיתָנִים רַב סְחָפַנִי רֵעַ פֶּלִאי,

כִּי גָדַל מַעְגַּל הַגַּלְמוּדִים עַל שִׁלְפֵי מֶרְחָק כָּחֹל; 

וַיִּלְחָשׁ: לֹא לְךָ הָעֶצֶב כֻּלּוֹ, נִחוּמֵי שִׁיר וּמָחוֹל –

כָּל הָעוֹלָם יְמָרֵר וִירוֹנֵן פֹּה בַשֶּׁלִי.


וְלֹא מִטְעָנְךָ לְבַד פֹּה גָרַף אֵד אֶל חוֹף חִדָּלוֹן,

וְלֹא הָאֶחָד, אָח, רֻמֵּיתָ אֶל סֻבְּכֵי גִיל וָטֶרֶף…

הַאִם רַק לִבְּךָ הוּא הַמְטַפֵּס פֹּה עַל אַחֲרוֹן מִצְפֵּה עֶרֶב 

לִרְאוֹת: הֲיֵשׁ יִשָּׁקֵף שָׁם, בֵּין יָם וָיָם, עוֹד מִקְלַט־מָלוֹן?…


וּבְיוֹם יִתְבַּדֶּה לְךָ כָל פֶּלֶא וּמְרִי יְגוֹנִים יִנְהָם…

וְעַל דַּרְכֵי חִידָה רְחוֹקָה יְפַגֵּר פִּתְרוֹן־קָרוֹן –

לַמַּעְגָּל פְּלֹט!… מִיַּיִן מְשֻׁמָּר שָׁם כָּל נָזִיר יָרֹן,

וְסָטִיר רַחֲמָן יְגַל כָּל סוֹד לְמֶרֶד חִנָּם.
