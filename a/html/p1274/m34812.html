עַל הַדָּנוּב / אִידִילְיָה

דָּכְיֵי מַיִם מְזַנְּקִים אֶל מַעֲרֻמֵּי כֵּפִים,

חוֹבְקִים גַּרְמֵי־צְלָעוֹת — שִׁלְדֵי־פְּגָרִים זְעוּמִים,

וּמִתְנַפְּצִים לְבָנִים, קֶצֶף פּוֹלֵט קֶצֶף,

חוֹזְרִים עַל עִקְּבוֹתָם, גַּלִּים נוֹשְׂאִים גַּלִּים,

נְמוֹגִּים וּמִתְפַּשְּׁטִים, יְרֻקִּים, צְלוּלִים, שְׁלֵוִים

לְעָרוֹת הַדְּלָיַיִם עַל סַף הָאֹהָלִים.


הָאֹהָלִים פְּזוּרִים עַל הָהָר וּכְתָלָיו,

כְּמַרְאֵה הַכַּוָּרוֹת בְּחֵיק יְבוּל הַשְּׁדֵמוֹת

בְּעֶצֶם עֵת הַלִּבְלוּב; אֹהֶל וְנוֹף־אִילָן

פּוֹרֵשׂ עָלָיו כַּפּוֹת, מַשְׁרֶה צֵל עַל חֶצְיוֹ,

וְעַל פְּנֵי הַשְּׁבִילִים, עוֹלִים וּמִתְפַּתְּלִים,

זְרוּעֵי חוֹל אַדְמוּמִי — רִשְׁתוֹת־צֵל רוֹעֲדוֹת.

רַק אֲחָדִים דַּלִּים, שִׁפְלֵי גַג וַאֲרֻבָּה,

מֻטִּים עַל הַדָּנוּב, מְצִיצִים אֶל מַעֲמַקָּיו


מִבֵּין הַצַּפְצָפוֹת, חֲרֵדִים וּמְשֻׁעְבָּדִים —

יְרֵאִים אֶת שְׁחוֹר מֵימָיו וְנִמְשָׁכִים אֵלָיו.


גִּנּוֹת חַסְרוֹת־מְשׂוּכָה, חֲתוּכוֹת תְּלָמִים־תְּלָמִים

יְשָׁרִים וַאֲלַכְסוֹנִים, זְרוּעוֹת פּוֹל וְדוּרָה

עוֹלִים בְּמַעֲלֵה קָנִים; בָּצָל עִם קִשּׁוּאִים,

אִישׁ בִּזְרוֹעוֹת אָחִיו; דְּלוּעִים וַאֲבַטִּיחִים

צָצִים רַחֲבֵי־טְרָפִים, מְלִיטִים צִיצַת־גְּזָרִים,

קִלְחֵי־כַרְפֲּס חִוְרִים, עוֹדָם עוּלֵי־יָמִים,

רַכִּים וּבַיְשָׁנִים. — בֵּין הַיְרָקוֹת מְדַדּוֹת,

נוֹפְלוֹת וְנִתְקָלוֹת תְּמוּכוֹת מֻטּוֹת־אֶבְרָה,

מְבֻדְּרוֹת נוֹצַת־זָנָב, תַּרְנְגוֹלוֹת בְּרֻדּוֹת,

אִשָּׁה וּמִשְׁפַּחְתָּהּ — כַּת אֶפְרוֹחִים מְצַיְּצִים.

וּבְרֹאשׁ כֻּלָּם צוֹעֵד גֶּבֶר אֲדֹם־כַּרְבֹּלֶת,

מוֹרֶה תוֹרַת־נִקּוּר בְּאַשְׁפָּה וּבִירָקוֹת

אַגַּב קִרְקוּר־גַּאֲוָה.


עַל כְּלוֹנְסָאוֹת־לִבְנֶה טוֹבְעִים בֵּין עֲשָׂבִים

מְתוּחִים חַבְלֵי־פִּשְׁתָּן, וַעֲלֵיהֶם פְּרוּשׂוֹת

רְשָׁתוֹת וּמִכְמָרוֹת תַּבְנִית סַל וַעֲרִיסָה,

תְּפוּחוֹת וּמִתְבַּדְּרוֹת בִּנְשֹׁב רוּחַ חֲרִישִׁית,

מְנַדְּפוֹת רֵיחַ דָּגִים. שְׁיָרֵי־יְרוֹקָה יְבֵשִׁים

עִם קַשְׂקֶשֶׂת־כֶּסֶף נֶאֶחְזוּ בְּנִקְבֵיהֶן

בְּעֵת דַּיִג אַחֲרוֹן. וּבֵין אֵלּוּ תְּלוּיִים

לְבוּשִׁים מִכָּל הַמִּינִים: צְנִיפִים עִם סִינָרִים

רְקוּמִים פִּרְחֵי־אֹדֶם, שְׁבִיסִים וְסוּדָרִים,

עֵרֶב־צְבָעִים מְשֻׁנֶּה, שְׂמָלוֹת עִם קִשּׁוּרִים,

קְפָלִים עַל גַּב קְפָלִים, סְדִינִים אַף חֲלוּקִים

תְּלוּיֵי זוּג שַׁרְווּלִים פְּרוּשִׂים כְּלַפֵּי מָטָּה.

לְעֻמַּת אֵלֶּה מְדַלְּגָה, שְׁתֵּי קַרְנֶיהָ שׂוּמוֹת

בַּעֲבוֹת וּבַיָּתֵד, עֵז אֲחוּזַת מְשׁוּבָה,

גִּדְלַת כְּחָל וְזָקָן, וְדַבְלוּלֵי צֶמֶר

תְּלוּיִים דְּבֵקֵי־דֹמֶן עַל רְזוֹן־שׁוֹקֶיהָ

לְפַיֵּס מִעוּט זְנָבָהּ. פִּיהָ קָץ בִּכְסִיסָה

מִנִּי אוֹר הַבֹּקֶר רַק בַּתְּחוּם הַמְסֻיָּם,

פּוֹעָה פְּעִיַּת־שִׁעֲמוּם, כּוֹרְעָה וּמִשְׂתָּעֲרָה

כְּלַפֵּי טוּר הַלְּבוּשִׁים, נְכוֹנָה כְּבָר לִנְגֹּחַ —

אַךְ הַחֶבֶל מְעַכֵּב: נִצְּבָה בְּקוֹמָה זְקוּפָה,

רֹאשָׁהּ אֲחוֹרַנִית וְעֵינֶיהָ לְטוּשׁוֹת.

עַל־יַד אֹהֶל לְבַנְבַּן זָקוּף עַמּוּד־שׁוֹבָךְ

שָׁחוֹר מִנִּי מָטָר וּמִשָּׁרָב מְבֻקָּע;

וּמִסְגֶּרֶת, מַרְאֵה כַּרְכֹּב־צַעֲצוּעִים,

שְׁלוּחִים מִן הַפְּתָחִים, פֶּתַח — פֶּה וּלְשׁוֹנוֹ.

יוֹנִים לְמוּדוֹת־מַיִם טָסוֹת אַט בִּמְחוּגָה,


יָשָׁר וּבְשִׁפּוּעַ, כָּנָף אַחַת סְרוּחָה

וַחֲבֶרְתֶּהּ נָעָה, מְלַטְּפָה אֶת הָאֲוִיר;

מְרַחֲפוֹת עַל הַנָּהָר, נוֹגְעוֹת וְאֵינָן נוֹגְעוֹת,

וְנִבְלָעוֹת חֶרֶשׁ אֶל תּוֹךְ חֲלַל הַשּׁוֹבָךְ

יוֹצְאוֹת צְנוּעוֹת, שְׁלֵווֹת, מְטַיְּלוֹת עַל הַכַּרְכֹּב,

עוֹגְבוֹת וּמִזְדַּוְּגוֹת אַגַּב הֶגֶה כָּשֵׁר.


כְּנֻפְיַת יְלָדִים רַכִּים, לְבוּשִׁים כָּתְנוֹת פִּשְׁתָּן,

עוֹסְקִָים בִּדְמִי־חֲרִיצוּת עֲלֵי כְּבִיר־הַדֶּשֶׁא:

חוֹבְרִים כָּפִיס לְכָפִיס, מְצָרְפִים יָד לְנֶסֶר,

מְהַדְּקִים בְּחוּּטֵי־קַנְבּוּס וּמַפְלִיגִים “סִירוֹת” —

דּוֹבְרוֹת־שַׁעֲשׁוּעִים — עַל פְּנֵי רְאִי הַמָּיִם,

שׁוֹלְחִים אַחֲרֵי כְּלֵיהֶם עֵינֵי־מַיִם עוֹרְגוֹת,

מְפִיקוֹת סַעֲרַת־גַּלִּים, וְשָׁטִים עִמָּם יַחַד

אֶל אִי־זָהָב רָחוֹק מְנַצְנֵץ בְּדִמְיוֹנָם.


כְּעֵין גֻּלְגֹּלֶת־כֶּתֶם עַל חֹד עַמּוּד שָׁחוֹר,

נִתְלָה עַל שֶׁן־סֶלַע שֶׁמֶשׁ בֵּין־עַרְבָּיִם,

וּמִלְּבָבוֹ זָבִים זִרְמֵי־דָם לוֹהֲטִים,

שׁוֹתְתִים וּמִתְקַלְּחִים לְרִבּוֹא רְמָחִים יוֹקְדִים,

צוֹנְחִים עַל הַכֵּפִים וּכְמִלֻּאַת־שְׁהָמִים

נִקְוִים בַּמִּשְׁקָעִים וּבִשְׁחוֹר הַחֲגָוִים,

יוֹרְדִים עַד תַּחְתִּיתָם וּמִשְׁתַּפְּכִים רְצוּעוֹת,

רְצוּעוֹת שֶׁל שַׁלְהֶבֶת, גַּם עַל פְּנֵי הַמָּיִם —

מִתּוֹךְ שְׁלִיטַת־דְּמָמָה טְעוּנָה זֹהַר־שְׁקִיעָה,

עוֹטָה עַלְטַת־הָרִים וְשַׁלְוַת־מַיִם רַכָּה,

עֹז הַנּוֹי מִסְתַּלֵּק, נוֹשֵׁק נְשִׁיקַת־פְּרִידָה

יוֹקְדָה לְרֹאשׁ הֶהָרִים וְנָמוֹג בְּחֵיק טָמִיר;

וְיֹפִי אַחֵר מְזֻמָּן רַךְ וּמְשׁוּחַ בֶּשֶׂם

נִכְנָס בַּחֲשָׁאִי. שׁוּלָיו מְגַלִּים טֶפַח

וּמְכַסִּים טִפְחֵי־טְפָחִים. עַל הַדָּנוּב מְרַפְרֵף

אֵד חֲוַרְוָר שָׁקוּף, עוֹלֶה וּמִתְבַּדֵּר

כִּצְעִיף־מֶשִׁי צָחוֹר מְנַפְנֵף בִּנְשֹׁב רוּחַ.

עֻמַּם תֵּל־הַזָּהָב צִפָּה אֶת הַסְּלָעִים,

מַרְאָם הָפַךְ אַט־אַט תִּלֵּי־רֶמֶץ כְּבוּיִים.

רַק עַל גַּג־הַנְּחשֶׁת שֶׁעַל דְּבִיר־הַתְּפִלָּה

וְעַל כִּפַּת־רֹאשׁוֹ בּוֹעֵר עוֹד וּמַבְהִיק

צְלָב־שַׁלְהֶבֶת מֻצָּל. נֶחְבְּאָה כָּל הַשְּׁפֵלָה

תַּחַת גַּף־אַפְלוּלִית, נִבְלַע כָּל הַחַי בָּהּ,

פָּסַק כָּל הָרוֹחֵשׁ וְלֹא נוֹדַע מָתָי.

רַק הַנָּהָר מְעָרֶה דְלָיָיו בִּדְמִי חֲרָדָה,

קוֹבֵל עַל נַחְשׁוֹלָיו הָרוֹחֲצִים אֶת הַכֵּפִים

בְּקוֹל שִׁכְשׁוּךְ צָלוּל, וְעַל כָּל פִּרְכּוּס דָּגָה.


אֶת הַדְּמָמָה פִּלַּח פִּתְאֹם כְּעֵין נֵפֶץ:

קוֹל הַפַּעֲמוֹן יָרַד מֵעַל דְּבִיר־הַתְּפִלָּה


אַדִּיר וּמִתְגַּלְגֵּל אֶל חֵיק דְּמִי הַשָּׁפֶל.

זֶה עוֹד טֶרֶם יִגְוַע — וְזֶה מִתְעוֹפֵף אַחֲרָיו

נִרְדָּף מִמִּשְׁנֵהוּ — וְהִנֵּה נִתַּךְ מָטָר

שֶׁל צִלְצוּלִים רַכִּים, נָאוִים וַעֲגוּמִים,

גּוֹלְשִׁים מִן הַגִּבְעָה. בָּאַפְלוּלִית צָצוּ

בְּרִיּוֹת עַל הַשְּׁבִילִים. תַּחַת כִּפּוֹת עֵצִים

נִפְנוּף רְדִידִים לְבָנִים, טִיסַת כּוֹבְעֵי־תֶבֶן

עַל קָדְקֳדֵי־צְלָלִים יוֹרְדִים טוּרִים־טוּרִים,

סִיעֹות, כְּנֻפִיוֹת, זוּגוֹת קְרֵבִים וּמִתְדַּבְּקִים,

פּוֹרְשִׁים וּמִתְפַּזְּרִים מִתּוֹךְ מְנוֹד־גֻּלְגָּלוֹת

וְנַעֲנוּעֵי־זְרוֹעוֹת — בְּרִיּוֹת עוֹלָם אַחֵר.

מִבֵּין הָאִילָנוֹת בּוֹקְעִים שִׁיר וּרְנָנָה.

שִׁיר וָשִׁיר לְעַצְמוֹ דּוֹבֵב עוֹלַם־רְגָשׁוֹת,

שַׁלְוַת לִבּוֹת נוּגִים וְלַַחַשׁ תְּפִלָּה תַמָּה,

זוֹלְפִים אֶל הַתְּכֵלֶת שֶׁל הַלֵּיל הַחֲגִיגִי,

גַּם נִפְגָּשִׁים דּוּמָם: זֶמֶר נוֹשֵׁק זֶמֶר

וּכְעֵין נְחִיל פַּרְפָּרִים יוֹרְדִים וּמשְׁתַּטְּחִים

עַל רְאִי מֵי הַדָּנוּב, רוֹתְתִים עַל בְּדָלְחוֹ,

שׁוֹקְעִים, צוֹלְלִים, אוֹבְדִים בִּשְׁאוֹן דְּכִי־הַסְּלָעִים.


עַל שְׁחוֹר הָאֶשְׁנַבִּים נִצְנְצוּ שַׁלְהָבוֹת

מִן הָאָח נִגָּרוֹת, מְפַזְּזוֹת וּמִשְׁתַּקְּפוֹת

בְּתוֹךְ רְאִי הַמַּיִם — שְׁלַל כּוֹכָבִים “נְפִילִים”.

הָאֲרֻבּוֹת פּוֹלְטוֹת עֲתַר עָשָׁן לְבַנְבַּן

עוֹלֶה וּמִתַּמֵּר אֶל רוֹם תְּכֵלֶת כֵּהָה.

סָמוּךְ אֶל סַף אֹהֶל, עַל גּוּשׁ אֶבֶן אָפֹר,

מֵסֵב בְּלִי־נוֹעַ סָבָא לְבֶן־תַּלְתַּלִּים,

מוֹצֵץ לוֹ מִקְטַרְתּוֹ, וּסְבִיבוֹתָיו חוֹנִים

גְּבָרִים, טַף וְנָשִׁים, עִרְבּוּב יָד וָרֶגֶל.

הַגֻּלְגָּלוֹת תְּמוּכוֹת עַל כְּתֵפַיִם מְגֻלּוֹת

וּבְהַקְשָׁבַת־דְּבֵקוּת תְּפוּשׂוֹת כָּל הָרוּחוֹת.

מִן הַגַּן הַסָּמוּךְ עוֹלָה וּמִשְׁתַּפְּכָה

פְּרִיטַת מַנְדּוֹלִינָה צְלוּלָה וַעֲרֵבָה,

נוֹסְכָה תּוֹךְ הַלְּבָבוֹת יַם גַּעְגּוּעִים חֲרִישִׁים.

וְיֵשׁ שֶׁתֵּרֶא עַיִן עַל הָרְאִי הַמְלֻטָּשׁ

הַנָּח בִּיפִי גְאוֹנוֹ מֵעֵין פִּזוּז בּוּעוֹת

כְּחֻלּוֹת וְחַכְלִילוֹת — כָּכָה עֵין הַפֶּרֶט.

בְּאֹפֶק מִזְרַח־נֶגְבָּה הֻצַּת פִּתְאֹם לַהַב;

גַּלְגַּל מְלֻבָּן אָיֹם יוֹצֵא וּמִזְדַּקֵּר

מִן הַתְּהוֹם הַחֲבוּיָה וְנִתְלֶה עַל הַצְּרִיחִים

וַעֲלֵי מְעִי־הֶחֳרָבוֹת שֶׁל פַּלַּטִּין עַתִּיק,

נִצַּב בְּדוֹר־קְדוּמִים עַל הָהָר בַּמֶּרְחָק —

נוֹלָד שָׁם הַסַּהַר. הָעֵינַיִם תְּלוּיוֹת

אֶל אֵשׁ הַמֶּרְחַקִּים וְהָאֹזֶן קוֹלְטָה

סִיּוּם דְּבַר הַסָּבָא מוֹצֵץ הַמִּקְטֶרֶת:


"…וּבְכָל לֵילֵי־שַׁבָּת וּמוֹעֲדֵי־קֹדֶשׁ

אָסוּר לָדוּג דָּגִים מִשּׁוּם חֲשָׁשׁ־סַכָּנָה:

בֶּן־הַגְּרַף אָז עוֹלֶה סָמוּךְ לַחֲצוֹת־לַיְלָה

מִתְּהוֹם מֵי הַדָּנוּב הָדוּר בִּלְבוּשׁ גִּבּוֹר,

קוֹבַע פָּז בְּרֹאשׁוֹ. אַף הָרִיבָה עוֹלָה,

בַּת הַדַּיָּג פֶרֶנְץ, תַּלְתַּלֶּיהָ פְּזוּרִים,

מְלֵאִים שְׁלַל־קִשּׁוּרִים וְעֵינֶיהָ שׂוֹחֲקוֹת.

מְטַיְּלִים עַל הַמַּיִם כְּטַיֵּל עַל יַבָּשָׁה.

פּוֹרֵשׁ לוֹ הַגִּבּוֹר, מְטַפֵּס עַל הַסְּלָעִים,

מֵטִיל דֹּם אֲבוּקָה אֶל מְעִי הַמַּפֹּלֶת —

וְשׁוּב הָאֵשׁ מִתְלַקְּחָה, מְצִיפָה הַשָּׁמַיִם

וְכָל אַפִיק הַדָּנוּב יוּצַת בְּשַׁלְהָבֶת.

זְקַן־הַגְּרַף הַקָּבוּר תַּחַת גַּל הַהֵיכָל

קוֹפֵץ נְשׁוּךְ־יִסּוּרִים, מְשַׁוַּע בַּלֶּהָבָה,

וּמִזַּעַם־מַכְאוֹב זוֹרֵק הוּא עַמּוּדִים,

קוֹרוֹת, גְּלִילֵי־אוּרִים מְנַתְּזִים עִנְקֵי־רְשָׁפִים

לְאַרְבַּע פִּנּוֹת עוֹלָם. וּשְׁנֵי הַנֶּאֱהָבִים

עוֹמְדִים עַל הַמַּיִם שְׁלוּבֵי זְרוֹעוֹת־רוּחַ,

פְּנֵיהֶם מוּל הַבְּעֵרָה, קוֹרְנִים בְּאוֹר הָאשֶׁר

וּמִסָּבִיב עוֹלוֹת בְּנוֹת־הַמַּיִם זוֹרְחוֹת,

פּוֹרְטוֹת עַל עֻגָבִים, מַכּוֹת בְּתֻפֵּי־צֶדֶף,

שָׁרוֹת בְּמַקְהֵלָה וְשָׁטוֹת מְחוֹל־הַחֻפָּה

סְבִיב הַזּוּג הַחוֹגֵג…זֶהוּ חַג הַטְּבוּעִים,

בֶּן־הַגְּרַף הַגִּבּוֹר וּבַת הַדַּיָּג פֶרֶנְץ

בְּכָל לֵילֵי־שַׁבָּת וּמוֹעַדֵי־קֹדֶשׁ…"


נִפְסַק קוֹל הַסָּבָא, נֶאֶנְקָה וְנִשְׁתַּתְּקָה

גַּם הַמַּנְדּוֹלִינָה. רוּחַ חֲרָדוֹת לָבְשׁוּ

גְּבָרִים, טַף וְנָשִׁים, מְגַשְּׁשִׁים בָּאַפְלוּלִית.

מִצְטַלְּבִים דּוּמָם, יָד אֶל יָד נִצְמֶדֶת

וְהָעֵינַיִם תְּלוּיוֹת אֶל רוֹם הָרָקִיעַ:

כּוֹכְבֵי־זָהָב קוֹרְצִים וְעִגּוּל־סַהַר מְזֻקָּק

יוֹצֵק אוֹרַת־קֶסֶם אֶל חֵיק סוּת־הַלָּיְלָה. —

לְאַחַר שְׁהִיַּת־רֶגַע כָּל הַכַּת מִזְדַּעְזְעָה,

חוֹפְזִים וּמִתְחַמְּקִים אִישׁ אִישׁ לְאָהֳלוֹ.

נֶאֱלַם רִשְׁרוּשׁ אַחֲרוֹן, כָּבוּ הָאֶשְׁנַבִּים;

כָּל הַדָּנוּב נִמְסַר עִם הֶהָרִים סָבִיב

וְעִם הֲמוּלַת דָּכְיוֹ כְּבָר לִרְשׁוּת הַלָּיְלָה.

“חֲצוֹת” מִתְקָרְבָה חֶרֶשׁ וְהַסַּהַר מְיַחֵל,

צוֹפֶה מִמִּשְׁמַרְתּוֹ לַחֲזוֹת חֲזוֹן־הַפְּלָאוֹת

יִבְקַע מִמַּעֲמַקִּים…


תרע"ב
