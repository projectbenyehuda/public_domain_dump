ספר תחכמוני - שַׁעַר אַרְבָּעִים וּשְׁלשָׁה

מְרִיבַת הַיָּם וְהַיַּבָּשָׁה. מִי הִיא יוֹתֵר חֲשׁוּבָה וּדְרוּשָׁה


נְאֻם הֵימָן הָאֶזְרָחִי: הָלַכְתִּי מֵעֶקְרוֹן. לְשֹׁמְרוֹן. וּפָגַעְתִּי בָהּ אֲנָשִׁים עוֹרְכִים מַעֲרֶכֶת תּוֹכָחוֹת. וְלוֹחֲמִים מִלְחֶמֶת אֲמָרוֹת נְכוֹחוֹת. וְרַכּוּ דִבְרֵיהֶם מִשֶּׁמֶן וְהֵמָּה פְתִחוֹת. וְהָיָה בְתוֹכָם אִישׁ יָקְדָה הָאֵשׁ מִנִּיצוֹצֵי אֲמָרָיו. וִיְפוֹצֵץ הַסֶּלַע בְּפַטִּישׁ דְּבָרָיו. וַיַּעַן וַיֹּאמַר: אַנְשֵי לֵבָב שִׁמְעוּ לִי. וְהַאֲזִינוּ נְגִידֵי שִׂכְלִי. אָחוּדָה נָא לָכֶם חִידָה חֲמוּדָה. תִהְיֶה לָכֶם לְתוֹרָה וְלִתְעוּדָה. וַיִּתְקַבְּצוּ הָעָם כֻּלּוֹ סְבִיבָיו. וְהֶחֱרִישׁוּ לְנִיבָיו. וַיֹּאמֶר לָהֶם: דְּעוּ כִּי הָאֵל בּוֹרֵא שָׁמַיִם. בְּעֵת בָּרָא הָאֲדָמָה כַדּוּר חִלֵּק אוֹתָה לִשְׁנַיִם.חֶצְיָהּ יַבָּשָׁה וְחֶצְיָהּ מַיִם. וְהִקִּיף הַיָּם עַל כָּל הָעוֹלָם. וְעַל בְּרוּאֵי תֵבֵל כֻּלָּם. וַתְּהִי הָאָרֶץ סְגוּרָה בְּמַאֲסָרוֹ. וּמִתַּחַת לְזֵרוֹ. וְהִיא בְתוֹכוֹ עֲצוּרָה. כְּאִלּוּ הַיָּם מֶלֶךְ וְהִיא שִׁפְחָה אֲסוּרָה. וְאֵין לָהּ מָנוֹס הֵנָּה וָהֵנָּה. כִּי הוּא מִכָּל צַד יִכְבְּשֶׁנָּה וְיַקִּיפֶנָּה. וְכִרְאוֹת הַיָּם כִּי הוּא כוֹבֵשׁ הָאֲדָמָה. וּמַקִּיף עָלֶיהָ כַּחוֹמָה. אָמַר בְּלִבּוֹ יָדִי רָמָה. וַיִּבֶז כָּל הָעוֹלָם בְּעֵינָיו. וְגָאוּ עַל הָאָרץ רַעְיוֹנָיו. וַיַּעַן וַיֹּאמַר: לִי יָאַתָה הַמְּלוּכָה. וּמַעֲלָתִי נְסוּכָה. וְכָל הָאָרֶץ בְּמָסֹרֶת בְּרִיתִי. וְכָל שׁוֹכְנֶיהָ בִּרְשׁוּתִי. וּבְעֵת תַּעֲלֶה עָלֶיהָ חֲמָתִי. יִשְׁטְפוּהָ מֵימֵי עֶבְרָתִי. וְאֵין לָהּ מָנוֹס מִתּוֹךְ כְּבָלַי. וְלֹא חֹפֶשׁ מֵעֻלַּי. וְכַאֲשֶׁר הִגְדִּיל הַיָּם לְדַבֵּר בִּגְאוֹנוֹ. וְהֶאֱרִיךְ לְשׁוֹנוֹ. לָבְשָׁה הָאֲדָמָה גָאוֹן וְגֵאָה. וַתַּעַט כַּמְּעִיל קִנְאָה. וְכִמְעַט עָלְתָה לְרוּם עָרֶץ. וַתִּגְעַשׁ וַתִּרְעַשׁ הָאָרֶץ.

וַתַּעַן וַתֹּאמַר: מִי הוּא זֶה וְאֵי זֶה הוּא אֲשֶׁר מְלָאוֹ לִבּוֹ לְהִתְנַשֵּׂא עָלַי. וְלִנְגֹּעַ בְּשׁוּלַי. וְהוּא מִדְרָךְ לְרַגְלַי. וּשְׂרוֹךְ לִנְעָלַי. וַאֲנִי כְמֶלֶך וְהַיַּמִּים מַרְכְּבוֹתַי. וַאֲנִי כְאָדוֹן וְהַנְּחָלִים עֲבָדִים סְבִיבוֹתַי. וְכָל בְּנֵי אָדָם בְּתוֹכִי נִבְרָאוּ. וְהַנְּבִיאִים בִּי נִבָּאוּ. וְהַתּוֹרָה בִי נְתוּנָה. וּבִי שָׁכְנָה הַשְּׁכִינָה. וְהַיָּם תִּשְׁכֹּן עָלָיו עֲנָנָה. וְאַל תָּבֹא בוֹ רְנָנָה. וּבַעֲבוּרִי יִפְתַּח הָאֵל מְרוֹמִים. לְהוֹרִיד מִטְרוֹת רַחֲמִים. וּלְהַזִּיל גְּשָׁמִים. וּלְרַוּוֹת בָּם תְּלָמִים. וּלְהַצְמִיחַ צְמָחִים. וּלְהַפְרִיחַ פְּרָחִים. מִמֶּגֶד גֶּרֶשׁ יְרָחִים. וּבִי הַמְּאוֹרִים יֵאוֹרוּ. וּבִי שָׂרִים יָשׂרוּ. וּמַה תַּרְחִיב פִּיךָ וְאַתָּה נִמְאָס וְנִבְזֶה. לְכָל שׁוֹמֵעַ וְחוֹזֶה. וְכָל הַהוֹלֵךְ מִמֶּנִּי אֵלֶיךָ יֵצֵא מִבֵּית נְעִימוּת. וְהוּא הוֹלֵךְ לָמוּת. וְכָל הַיָּמִים אֲשֶׁר יֶהְגֶּה בְךָ. וְיִשְׁכֹּן בְּקִרְבְּךָ. חַיָּיו תְּלוּאִים מִנֶּגֶד לוֹ. וְהַמָּוֶת נִצָּב בְּכָל עֵת לְמוּלוֹ. גַּם בַּלַּיְלָה לֹא שָׁכַב לִבּוֹ. וּבְכָל הַיּוֹם לֹא יִשְׁקֹט עָצְבּוֹ. וְלֹא יַאֲמִין בְּחַיָּיו עַד אֲשֶׁר יֵצֵא מֵעִמְךָ. וְיִמָּלֵט מִמְּךָ. וּבְבוֹאוֹ אֵלַי אָז יִמְצָא מְנוּחָה. וְיֵדַע כִּי הָיְתָה הָרְוָחָה.

וַיְהִי כִשְׁמֹעַ הַיָּם מִלָּיו. הִתְגָּעֲשׁוּ גַלָּיו. וְנִצְבוּ כְּמוֹ נֵד נוֹזְלָיו. וְגָבְרוּ שְׁאוֹן מִשְׁבָּרָיו בְּקוֹל שׁוֹאָה. וְכִמְעַט יִשְׁטֹף הָאָרֶץ וּמְלוֹאָהּ. וַיֹּאמַר אֶל הַיַּבָּשָׁה: אֶרֶץ אֶרֶץ אָרֶץ הֲתִדְמְי אֵלַי. וְתַעַרְכִי אֵלַי. בַּתְּחִלָּה מִי אָנֹכִי הֲלֹא אֲנִי מְחוֹלֵל הָרְפָאִים. וּמַחֲרִיד הַבְּרוּאִים. וְכָל הָרוֹאֶה שְׁאוֹן גַּלַּי. יִפְחַד מִקְּרֹב אֵלַי. וְלוּלֵי כִי יוֹצְרִי וּמְחוֹלְלִי. נָתַן הַחוֹל גְּבוּלִי. וְאָמַר לִי מְקוֹמְךָ לֹא תַחֲלִיף. וְעַד פֹּה תָבֹא וְלֹא תוֹסִיף. יָצְאוּ גַלַּי עָלַיִךְ. וְשָׁטַפְתִּי כָל גְּבוּלַיִךְ. וְאָסִירָה כָּל בְּדִילַיִךְ. וְשָׁבַרְתִּי כָּל סְפִינוֹתַיִךְ. וְהִטְבַּעְתִּי בִתְהוֹם רַבָּה אֳנִיוֹתַיִךְ. וּבִמְצוּלָתִי תִּתְחַבְּאִי. וּתְבֻקְּשִׁי וְלֹא תִמָּצְאִי. הֲלֹא תֵדְעִי הַנִּפְלָאוֹת אֲשֶׁר עָשָׂה הָאֵל בִּי. בְּבֹא עַם הַקֹּדֶשׁ בְּקִרְבִּי. וְהָאוֹיֵב נִגַּשׁ עִמָּהֶם לָרִיב. וּפַרְעֹה הִקְרִיב. וְכִרְאוֹתִי צָרָתָם מֵאוֹיְבֵיהֶם. הָמוּ רַחֲמַי עֲלֵיהֶם. וּבְעֵת צוֹרְרֵיהֶם אָמְרוּ אַיָּם. וַיֵּט משֶׁה אֶת יָדוֹ עַל הַיָּם. נָשְׁאוּ מִשְׁבָּרֵי דָּכְיָם. וְנִבְקְעוּ לָהֶם גַּלַּי. וְנִצְּבוּ כַחוֹמָה נוֹזְלַי. וּבְתוֹכִי הֲבֵאתִים. וּבֵין גַּלַּי הֶחְבֵּאתִים. וּמִן הָאוֹיְבִים בְּצִלִּי כִסִּיתִים. וּבְבוֹא הַתַּנִּין הַגָּדוֹל לִטְרֹף עֶדְרֵיהֶם. וַיִּרְדְּפוּ מִצְרַיִם אַחֲרֵיהֶם. כְּמַעֲשֵׂיהֶם גְּמַלְתִּיהוּ. וּבַמִּדָּה אֲשֶׁר מָדַד שְׁפַטְתִּיהוּ. וּבְמִשְׁבְּרֵי גַּלַּי שְׁטַפְתִּיהוּ. וְהֵבֵאתִי עָלָיו יוֹם קְפָדָה. וְעָשִׂיתִי מִמֶּנּוּ לִדְגֵי הַיָּם סְעֻדָּה. וְאִלּוּ נִשְׁאַר עַם הַקֹּדֶשׁ בַּיַּבָּשָׁה. תִּהְיֶה הָאַחֲרוֹנָה מִן הָרִאשׁוֹנָה מָרָה וְקָשָׁה. וְחֶרֶב הַצָּר אֲכָלָם. וְאֵשׁ זַעְמוֹ כִלָּם. וְלֹא יָכְלוּ הֲרָרַיִךְ וּמִדְבָּרַיִךְ לְהַצִּילָם.

וַתִּפְתַּח הָאָרֶץ אֶת פִּיהָ וַתֹּאמֶר לַיָּם: הֲתִמְצָא בְּפִיךָ מַעֲנֶה. וְעַד אָנָה תְהִי בְאִוַּלְתְּךָ שׁוֹנֶה. הֲלֹא רָעָתְךָ רַבָּה וְאֵין לָהּ גְּבוּל. הֲלֹא אַתָּה שְׁאֵרִית מֵימֵי הַמַּבּוּל. הַמַּיִם הַמָּרִים הַמְאָרְרִים. אֲשֶׁר טָבְעוּ בָהֶם הַיְצוּרִים. מֵימֶיךָ לֹא זַכִּים וְלָא בָרִים. לֹא צַחִים וְלֹא קָרִים. כִּי אִם רְפָשִׁים וּשְׁמָרִים. מְלוּחִים וּמָרִים. לֹא יַשְקוּ מִמְּךָ אֲדָמָה. וְלֹא יִשְבְּרוּ מִמְּךָ צָמָא. וְלֹא תִקְרַב אֵלֶיךָ רֶגֶל בְּהֵמָה. וְכָל הַשּׁוֹתֶה מֵימֶיךָ יִתְהַפְּכוּ קְרָבָיו לִרְגָעָיו. עַד יָמוּת בְּתַחְלוּאֵי מֵעָיו. וּמִלְּבַד זֶה אַתָּה אַכְזָר וְעָרִיץ. וּבוֹגֵד וּפָרִיץ. תַּהֲרֹג נְקִיִּים. וְתָמִית אֶבְיוֹנִים וַעֲנִיִּים. כַּמָּה סְפִינוֹת הִטְבַּעְתָּ.וְכַמָּה חֲסִידִים הָרַגְתָּ. וְכַמָּה יוֹנְקִים מִשְּׁדִי אִמָּם גָּמַלְתָּ. וְטָבַחְתָּ וְלֹא חָמַלְתָּ. אִם יָבֹא בְךָ אָדָם יַעֲמֹד בְּסַכָּנַת מָוֶת. וְיִרְאֶה פְתוּחִים לְפָנָיו שַׁעֲרֵי מָוֶת. וְכִרְאוֹתוֹ מֵימֶיךָ נִצְּבוּ עָלָיו כְּחוֹמוֹת. יְסוֹבְבוּהוּ מְהוּמוֹת. וְרַעְיוֹנָיו יַעֲלוּ שָׁמַיִם יֵרְדוּ תְהוֹמוֹת. וְאִם יִתְעוֹרְרוּ עָלֶיךָ רוּחוֹת. יִתְעוֹרְרוּ בְלִבּוֹ אֲנָחוֹת. וְאִם יִתְגָּעֲשׁוּ עָלָיו גַּלֶּיךָ. וְנִצְּבוּ כְמוֹ נֵד נוֹזְלֶיךָ. אוֹי לְשׁוֹכֵב עָלֶיךָ. כִּי יִשְׂבַּע עִנּוּיִים עֲצוּמִים. וְיִטְעַם הַמָּוֶת כַּמָּה פְעָמִים. וּבְצִאתוֹ מִמְּךָ יַעֲרֹךְ לָאֵל תְּהִלָּה וּבְרָכָה. אֲשֶׁר הִצִּילָהוּ מִן הַהֲפֵכָה. וְהַחֵמָה הַשְּׁפוּכָה.

אָמַר הַיָּם לַיַּבָּשָׁה: רֹב הֶגְיוֹנַיִךְ. לְאוֹת עַל עַזּוּת פָּנַיִךְ. כִּי אִם הַחֲרֵשׁ תַּחֲרִישִׁי תַּסְתִּירִי קְלוֹנֵךְ. וּבְרֹב דְּבָרַיִךְ תְּגַלִּי עֳוֹנֵךְ. הֲלֹא תֵדְעִי כִּי אַתְּ נְוֵה הָעֲבֵרוֹת. וּבֵית הַמִּפְעָלוֹת הַמְּכֹעָרוֹת. וּמוֹשַׁב הָאַכְזָרִים הֶעָרִיצִים. וּמְעָרַת פָּרִיצִים. כַּל שׁוֹכְנַיִךְ הֵם אַנְשֵׁי רֶשַׁע וְאַשְׁמָה. וְכָל יוֹשְׁבַיִךְ בַּעֲלֵי חֵטְא וְזִמָּה. וְהַיּוֹצֵא מִמֵּךְ וְיָבוֹא בְתוֹכִי. יָשׁוּב חָסִיד כִּי חָסִיד אָנֹכִי. וּבְעוֹד אֲשֶׁר יִשְׁכֹּן בִּי יַרְבֶּה לָאֶל תְּפִלוֹתָיו. וְאַל יֶחְדַּל עֶרֶב וָבֹקֶר מִתְּחִינוֹתָיו. יַעֲזֹב רָשָׁע דַּרְכּוֹ וְאִישׁ אָוֶן מַחְשְׁבוֹתָיו. וּבְעֵת יִמָּלֵט אָדָם מֵרִשְׁתִּי. וְיֵצֵא הָאֶחָד מֵאִתִּי. יָשׁוּב מְהֵרָה לְרִשְׁעָתוֹ. וְיַעֲזֹב וְיַחֲזֹר לְקַדְמָתוֹ. וְיִשְׁכַּח אֱסָרָיו. וְיָפֵר נְדָרָיו. וְיַעֲזֹב הַתְּפִלָּה. וִיְתַעֵב עֲלִילָה. וְיָשׁוּב לַעֲשׂוֹת כָּל הַנְּבָלָה. כַּאֲשֶׁר בַּתְּחִלָּה.

וְכַאֲשֶׁר הִרְבּוּ לְדַבֵּר דִּבְרֵיהֶם. עָשׂוּ זִקְנֵיהֶם שָׁלוֹם בֵּינֵיהֶם. וְשָׁבוּ לְקַדְמָתָם שְׁנֵיהֶם.

אָמַר הַמַּגִּיד: וּכְשָׁמְעִי מְלִיצוֹתָיו. וּמִשְׁלֵי חֲמוּדוֹתָיו. הִתְבּוֹנַנְתִּי אֵלָיו וְהִנֵּה הוּא אַבִּיר הָרוֹעִים. וּגְבִיר הַשּׁוֹעִים. אַלּוּפֵנוּ וּמוֹרֵנוּ. חֶבֶרֶ הַקֵּינִי חֲבֵרֵנוּ.

וָאֶשֶּׂא מְשָׁלִי וָאֹמַר:

אֲשֶׁר יִדְרשׁ חָכְמוֹת בְּחִידוֹת וּמְזִמּוֹת

וְיִשְׁאַל מִי אֶחָד וּמֵבִין מֵאֵין שֵׁנִי

בְּכָל חָכָם יָקוּץ וְכָל מַשְׂכִּיל חָרוּץ

וּבֵין כֻּלָּם יָרוּץ לְחֶבֶר הַקֵּינִי.
