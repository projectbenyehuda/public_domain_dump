מִתּוֹךְ הַשִּׁיר “כּוֹס תַּנְחוּמִין”: יוֹם יוֹם תַּהַר תֵּבֵל

[חי בוֵנֶצְיָה לערך שנת 1700]


מִתּוֹךְ הַשִּׁיר “כּוֹס תַּנְחוּמִין” (עַל־פְּטִירַת משֶׁה לִֵוי מוּייָא)


תרעומת נרגן על תבל המרטשת עולליה ושליתה היוצֵת מבין רגליה.

– – – – – – – –

יוֹם יוֹם תַּהַר תֵּבֵל תּוֹלִיד בָּנִים

עֵת וָעֵת כּוֹרָה שַׁחַת קִבְרוֹתָם

הַאִם עָצְמָה אֵין לָהּ אִם אֵין אוֹנִים

שָׁוִים לִשְׁמִירָתָם כִּבְרִיאָתָם

אוֹ פֶן־רַחְמָהּ הָרָתָה לִזְנוּנִים

שֶׁתַּעְלִים נַבְלוּתָהּ בִּשְׁחִיתוֹתָם

תֵּרַע תַּקָּנָתָהּ עַל־קִלְקוּלָהּ

חַטָּאָה רַבָּה לֹא תִּמְחֶה קַלָּה.


אַךְ אִם תֹּאכַל אֶת־כָּל־בָּשָׂר חַרְבָּהּ

מַה־לִּי וָלָהּ לִתְבֹּעַ עֶלְבּוֹנָם

אַלְפֵי רִבּוֹא אִישׁ וּבְהֵמָה רַבָּה

הֵמָּה יָרִיבוּ אֶת־מִשְׁפַּט דִּינָם.

אֵת שֶׁהֵיטֵב חָרָה לִי אַפִּי בָהּ

עַל־רֶשֶׁת זוּ טָמָנָה לִי חִנָּם

עַל־לֹא חָמָס עָלַי עַל־לֹא עַוְלָה

אוֹי גַּם מָחָר אַף־אָנִי קָרוּא לָהּ.


אֵיזֶה פִשְׁעִי אֵיזֶה חֶטְאִי אֶצְלָהּ

שֶׁתִּשְׁתּוֹנֵן שִׁנָּהּ בִּי בִּבְשָׂרִי

אִלּוּ תֶחְסַר בָּשָׂר מִדֵּי אָכְלָהּ

שׁוֹר אוֹ כֵשֶׂב אוֹ עֵז, רַב לִי עֶדְרִי.

תֵּלֶךְ־נָא אֶל־הַצֹּאן, שָׁם תִּקַּח לָהּ

בָּשָׂר שַׂגִּיא פִּדְיוֹן נַפְשִׁי כָּפְרִי

לֹא גָזַר מִמִּכְלָא צֹנֶה מִקְנָי

עִזָּא טַבָּחָא שָׁמֵינָא מִנָּי.


תַּרְתִּי חָתַרְתִּי לַחְשֹׂף מַצְפּוּנָהּ

עַל־אֹדוֹת מֶה דַּרְכָּהּ זֶה כֵּסֶל לָהּ

וָאֶתְכּוֹנֵן רֵאשִׁית כָּל־עִנְיָנָהּ

גָזוֹר אָחוֹר מֵאֵת סוֹף מִפְעָלָהּ

שֶׁלֹּא תִלְטשׁ מַחְשֶׁבֶת רַעְיוֹנָהּ

רַק לִמְשָׁל בָּנוּ שֵׁן בִּרְיָה קַלָּה

שִׁנֵּי תּוֹלַעַת, מַלְתַּעַת רִמָּה

מֵאֵין אִתָּנוּ יוֹדֵעַ עַד־מָה.


תִּמְהוֹן לִבִּי עַל־זֶה בִּי מִשְׁתָּאֶה

אָפוּנָה אִם חוֹלֵם הוּא אוֹ הוֹזֶה

אֵיךְ תַּשְׁחִית מִין נֶחְמָד וִיפֵה מַרְאֶה

לַבְרוֹת לַרְווֹת לַצְבּוֹת בֶּטֶן נִבְזֶה

חָלִילָה לָהּ חָלִילָה לֹא יָאֶה

לִיקָר הַדְרָהּ לָתֵת זֶה תַּחַת זֶה

בַּעַר וּבְעִיר מַבְעִיר מֵיטַב כַּרְמוֹ

לִצְבֹּר רַק שֶׁמֶץ רֶמֶץ פֶּחָמוֹ.


מִתְגּוֹלֵל עוֹד בִּמְסִלַּת רַעְיוֹנָי

לִמְצֹא חִידַת סוֹדָהּ סוֹד חִידָתָהּ

וַתִּתְגַּל כִּתְמוּנָה נֶגֶד עֵינָי

אִשָּׁה כוֹפֶפֶת קֶשֶׁת זִקְנָתָהּ

תִּשְׁלַח אֶצְבַּע רָגְזָהּ אֶל־מוּל פָּנָי

בּוֹעֶטֶת כַּף רַגְלָהּ עַל־אַדְמָתָהּ

שִׂפְתוֹתֶיהָ נָעוֹת קוֹל נָמוּךְ לָהּ

וָאֶשְׁמַע כָּאוֹב מֵאֶרֶץ קוֹלָהּ.


תשובת תבל הומיה היא וסוררת וזאת אומרת:

הִנֵנִי הִנֵּה אָנֹכִי תֵבֵל

מִי בַעַל מִשְׁפָּטִי יִגַּשׁ אֵלַי

הָאִישׁ כָּמוֹךָ דַּל רַב הַחוֹבֵל

חוֹתֵר בִּמְשׁוֹט שִׂכְלוֹ עַל־שָׁוְא גַּלָּי

מִי אַתָּה מֹץ נִדָּף עָלֶה נוֹבֵל

שֶׁתִּשְׂתָּרֵר סִכְלוּת כִּסְלָךְ עָלָי

אִם תַּבְחִין מַה עֶרְכָּךְ מָךְ מֵעֶרְכִּי

בּוּז דַּרְכָּךְ לֹא יִשְׁפֹּךְ בּוּז עַל־דַּרְכִּי.


קָלַעְתָּ אַבְנֵי־דֹפִי אֶל־מִצְחִי

וַתּוֹצֵא כִסְלוּתְךָ אֶת־דִּבָּתָהּ

וַתּוֹכִיחֵנִי עַל־חֻלְשַׁת כֹּחִי

לִנְצֹר אֶת־בֶּן־בִּטְנָה בִּבְרִיאָתָהּ

כַּזּוֹלֵלָה הִבְאַשְׁתָּ אֶת־רֵיחִי

כַּנּוֹאֶפֶת חוֹנֶקֶת שִׁלְיָתָהּ

אַךְ שָׁוְא לִי עֶלְבּוֹנְךָ שָׁוְא הִלּוּלָךְ

אִם־אֶנְאַף אִם־אֶרְעַב לֹא אֹמַר לָךְ.


הַגֶּד־נָא אִם יָדַעְתָּ אֶת־עַצְמָךְ

בִּנְיַן גּוּפְךָ קִשְׁרֵי הַרְכָּבָתוֹ

אַלְפֵי פִלְאֵי אֵבָרֶיךָ דָּמָךְ

רוּחוֹ נוּעוֹ נוּחוֹ וּתְקוּפָתוֹ

כִּי חוּצָה לָךְ תִּדְרֹשׁ מֻפְלָא מִמָּךְ

וּבְמַה־שֶׁעִמָּךְ לֹא תַּשִּׂיג אוֹתוֹ

לָרִיק נֶעְלָם עַצְמוֹ אֵצֶל עַצְמוֹ

יַחְקֹר סִתְרֵי סִדְרֵי סוֹד עוֹלָמוֹ.


לוּלֵא חַסְתִּי חָשַׁשְׁתִּי לִכְבוֹדִי

פֶּן־דַּם פִּשְׁפֵּשׁ יַבְאִישׁ אֶת־צִפָּרְנָי

הִנֵּה עַתָּה שָׁלַחְתִּי אֶת־יָדִי

וָאֶטְבַּח פִּגְרָךְ אֶל־כַּלְבֵי צֹאנִי

אַךְ אִם חֶטְאָךְ יָרֵט דַּרְכּוֹ נֶגְדִּי

לֹא אֶשְׁקֹל חֵטְא לָחֵטְא עַל־כַּף מֹאזְנָי

אִם רַע כָּל־רַע הוּא. רַע זֶה הַבּוֹחֵר

לִגְמֹל רֵעוֹ רַע תַּחַת רַע אַחֵר.


אִלּוּלֵי מָעוֹז חֶסֶד חֶמְלָתִי

עוֹדֵף לִסְלֹחַ כָּל־עֲוֹן אַשְׁמָה

שֵׁבֶט מוּסָרִי שֵׁבֶט עֶבְרָתִי

כִּכְבָר הֵטִילוּ עָלֶיךָ אֵימָה

אַךְ מָה־אֶתְנַכֵּל אֶקְחָה נִקְמָתִי

מִפֶּתִי נִבְעַר דַּעַת וּמְזִמָּה

הַפֶּתִי סִרְחוֹנוֹ מֵחֶסְרוֹנוֹ

טִפֵּשׁ לֹא יִתָּפֵשׂ עַל־שִׁגְעוֹנוֹ.


זוּלַת רַבַּת רָעַת נִקְמַת חֶטְאוֹ

יֵאוֹת לָחוּס אַךְ עַל־טֵרוּף דַּעְתּוֹ

נָגִיד הַשֵּׂכֶל שָׂם בָּנוּ כִּסְאוֹ

לִפְנֵי לִפְנִים בָּמֹחַ רַק שִׁבְתּוֹ

שָׁם נֶעְלַם מִן־הָעַיִן מַחְבוֹאוֹ

שָׁם לֹא תִּגַּע בּוֹ יָד אֶל־מִסְגַּרְתוֹ

חָלְיוֹ כִּמְעַט נוֹאַשׁ מִתַּקָּנָה

כָּל־מַכַּת חָלָל חֶזְקַת סַכָּנָה.


(1707)
