ואמר עוד, יוקירהו אל, מתאר הצרות ההן ומשבח האל יתעלה על אשר שמר וגונן, והחלים ורפא

שִׁירָה


תְּנוּמָה, בְּעֵין מַכִּיר אֱלֹהָיו – עֲצַלְתַּיִם, / וְשִׁפְלוּת – שְׁתֹק מִשִּׁיר לְיוֹשֵׁב בְּשָׁמַיִם,

וּמִי שָׁר בְּעֵין לִבּוֹ דְּרָכָיו וּמִתְעַלֶּם, / לְבַד מִנְּלוֹז לֵבָב וְנֶעְקַשׁ דְּרָכַיִם

וְאִלֵּם כְּבַד פֶּה, לֹא כְמוֹתִי, לְטוּשׁ גָּרוֹן / וְחֵךְ, לֹא כְּבַד שָׂפָה וְעֶרֶל שׂפָתָים?

וְאִם אַחֲרִישׁ, אַחַר שְׁנוֹתוֹ עֲשֹׂה לִי אֵל / יְצַוֶּה וְאֶבָּתֵר לְשָׁלֹשׁ וְלִשְׁתָּיִם!

אֲדַבֵּר בְּמַקְהֵלוֹת לְמֵאוֹת וְרִבּוֹאוֹת / וְאֹמַר בְּהִפָּרְדָם לְאֶחָד וְלִשְׁנָיִם:

אֲנִי בָא כְּמִישָׁאֵל בְּכִבְשָׁן – וְלֹא שֹׂרַף, / וּמָטְבָּע כְּמוֹ יוֹנָה וְלֹא מֵת בְּלֶב מַיִם,

וְחַי – אַחֲרֵי עָבְרוּ מְהוּמוֹת עֲלֵי נַפְשִׁי / בְּצִירִים וּמַחְלוּיִים כְּנַחַל וְנַחְלַיִם,

וְלָכֵן לְשׁוֹנִי – תֹּף מְרַנֵּן, וּפִי – כִּנּוֹר / מְנַגֵּן, וְעַל כָּכָה שְׂפָתַי מְצִלְתַּיִם,

מְחַוּוֹת כְּבוֹד צוּרָם וְחַסְדּוֹ בְּשִׁיר נָכוֹן / כְּהִכּוֹן כְּלִי יוֹצֵר מְיֻשָּׁר בְּאָבְנַיִם

וּמוֹשְׁכוֹת לְלִבּוֹת מַאֲזִינִים, בְּהִקָּרְאָם / בְּטוּרִים מְכֻבָּדִים, כְּמַזְלֵג בְּשִׁנַּיִם,

שְׁקוּלֵי דְבָרִים וַעֲבָרִים, כְּמוֹ לַיִל / וְיוֹם, יוֹם עֲלוֹת חֶרֶס בְּטָלֶה וּמֹאזְנַיִם

בְּבֹאִי קְרָב לֹא הוּא קְרָב אַךְ שְׁאוֹל עָמְקָה / וְנָגְעִי לְמָוֶת אוֹ לְכֹבֶד נְחֻשְׁתַּיִם

וְעָשָׂה אֱלֹהִים לִי עֲשׂוֹתוֹ בְּמֵאָה מִן / שְׁאוֹנִים אֲשֶׁר עָבְרוּ כְּמוֹ עָב, וּמָאתַיִם

וְלָכְדִי מְצָדוֹת בַּשְּׁחָקִים גְּבוֹהִים, לֹא / צְרִיכִים אֱלֵי מַנְעוּל וְלֹא אֶל דְּלָתַיִם

וְשׁוּב לוֹחֲמַי אֵלַי בְּשָׁלוֹם, כְּאַוָּתִי / וְלִבִּי, ולֹא בִקְרָב וְגַם לֹא בְּאַפַּיִם,

וְשׁוּבִי אֱלֵי בֵיתִי בְּשִׂמְחָה, סְבִיבִי מַלְ־/אֲכֵי אֵל וְסִתְרָתוֹ, כְּיַעְקֹב בְּמַחְנַיִם –

אֲזַי בָּא חֳלִי הָפוּךְ וְקָשֶׁה אֲשֵׁר לֹא בָא / כְּמוֹתוֹ לְדָוִים נֶאֱנָשִׁים בְּמִצְרַיִם,

בְּצִיר יַךְ עֲלֵי צַלְעִי כְּאוֹיֵב בְּלָכְדוֹ מַ־/עֲרֶכֶת לְנִלְחָמָיו, וְלָכְדוֹ אֲגַפַּיִם,

כְּאֵב הֶאֱבִיד לֵחִי וּמֹחִי וְכָל כֹּחִי, / וְשָׁלַל תְּקוּמָה מֵעֲקֵבוֹת וּפַעֲמַיִם,

אֲדַבֵּר – וְאֵין מֵבִין, וְאָשׁוּר – וְלֹא אַכִּיר, / וְאַבִּיט אֱלֵי עוֹרִי – וְהִנֵּה כְּכִירַיִם,

ואֶעְטֹף וְאֶתְגּוֹלֵל בְּצִירִי, עֲדֵי יִגַּע / שְׂמֹאל אֶל מְקוֹם יָמִין וְרֹאשׁ אֶל כְּרָעַיִם

וְאָקוּם כְּפוּף קוֹמָה כְּאַגְמוֹן בְּתוֹךְ מֵימָיו / וְאֵשֵׁב כְּיוֹלֵדָה בְּיָד עַל חֲלָצַיִם

וְאֶשְׁכַּב, כְּמוֹ שִׁכּוֹר בְּלִי לֵב, וּמִמַּכְאוֹב / עֲצָמַי וּמֵחֻמָּם, כְּאוּד בֵּין שְׁפַתַּיִם,

וְתִכְבַּד בְּלַיִל מַחֲלָתִי וְאֶנְקָתִי / כְּאֶנְקַת חֲלַל מוֹאָב בְּיַהַץ וְאֶגְלַיִם,

וְיֵאוֹר – וְלֹא אַאְמִין בְּבֹקֶר וְסוּר לַיִל, / וְיֶחְשַׁךְ – וְלֹא אָבִין בְּעֶרֶב וְעַרְבַּיִם,

וְלִבִּי בְּבֵין זֵעָה לְזֵעָה כְּצָף בֵּינוֹת / יְאוֹרִים שְׁנַיִם נֶהֱדָפִים, וְנַהְרַיִם,

וְהָרוֹפְאִים בָּאִים חֲרֵדִים וְיֹאמְרוּ כִי / מְעַט הֵם אֲשֶׁר חָיוּ בְּכָזֶה שְׁבוּעַיִם,

וְרָצִים בְּנִי וּשְׁאָר קְרוֹבַי בְּעַד מַרְפֵּא, / וְאִם הֵם מְלֵאִים פִּיק מְנַקֵּשׁ לְבִרְכַּיִם

וְגוֹעִים בְּתוּגָתָם וּבוֹכִים לְרָחוֹק מִן / מְקוֹמִי, וְיֵשׁ גּוֹעוֹת וּמַכּוֹת לְשָׁדַיִם,

וְאִם יִנְאֲמוּ בֹקֶר: כְּבָר בָּא אֱלֵי דַיּוֹ / בְּחָלְיוֹ – בְּעֶרֶב יֶחֱזוּנִי בְּכִפְלָיִם.

סְבַלְתִּיו וְאֵין לִי גוּף לְבַד מִמְּלֹא עוֹר מִן / עֲצָמוֹת, בְּפֶה, וָלֵב, וְאֹזֶן, וְעֵינַיִם,

לְזֶה נִיב, וְזֶה רוּחַ אֱלֹהִים, וְזֶה שָׁמַע / יְרֵאִים, וְזֶה רוֹאֶה בְּפָנָיו לְחָיַיִם,

וְעָשׂוּ יְמֵי תַחְלוּא שְׁנַיִם חֳדָשִׁים בִּי / אֲשֶׁר יַעֲשׂוּ כָכָה לְחוֹלֶה שְׁנָתַיִם,

לְבַד מִן יְמֵי חָלְשִׁי אֲשֶׁר נֶאֱסַר בָּם עַל / אֲשׁוּרַי לְהַלֵּךְ עַל אֲדָמָה זְרָתַיִם

וְהִקְצוּ שְׁתֵּי רַגְלַי בְּתוֹךְ אָהֳלִי מִצֵּאת / לְשַׁעַר, וְהַחֹלֶשׁ מְקַצֶּה לְרַגְלַיִם,

וּמַשְׁפִּיל מְרוֹם קָדְקֹד וּמַפִּיל אֱיָל כָּל יָד / וּמֵקֵל דְּמֵי נֶפֶשׁ, וּמַכְבִּיד כְּתֵפַיִם,

וְהִצִּיל יְחִידָתִי בְּעֶשֶׂר, אֲשֶׁר בָּאוּ / רְצוּפוֹת בְּרַע, שָׁלֹשׁ פְּעָמִים וּפַעְמַיִם:

כְּאֵב צַד, וְנֶגַע בַּחֲדַר צַד, וְצֵאת לֵחַ, / וְחֹם גּוּף, וְצִיר בִּשְׁתֵּי כְּלָיוֹת וּמֵעַיִם,

וְרֶגֶשׁ בְּלוּחַ לֵב מְמוֹגָג, וְיֵרָקוֹן, / וְתַעַב לְכָל נִגָּר וְהָרוּס בְּשִׁנַּיִם,

וְהָדַר פְּנֵי עַבְדּוֹ בְּחַיִּים מְחֻדָּשִׁים / אֱלֹהִים, סְבִיבָיו הוֹד וְהָדָר וְקַרְנַיִם,

וְהִתִּיר אֲגֻדַּת חִיל יְדִידוֹ אֲשֶׁר הָיָה / אֲסוּר לֵב־אֱלֵי־מַחְלָה וְתָפוּשׂ בְּיָדַיִם

אֱלֹהִים, אֲשֶׁר עוֹנִים קְדֻשָּׁה מְכַסִּים אֶת / פְּעָמִים וּפָנִים, לוֹ, מְתֵי שֵׁשׁ כְּנָפַיִם,

וּמֶלֶךְ, אֲשֶׁר לוֹ יַעֲנוּ כֵן, חֲרֵדִים מִן / חֲטָאתָם וְרִשְׁעָתָם, יְהוּדָה וְאֶפְרַיִם,

מְעוֹן כֹּל וְנוֹשֵׂא כֹל בְּתָקְפּוֹ וְלֹא נִסְמָךְ / בְּמָקוֹם וְקוֹמָה, אוֹ בְּמַלְבּוּשׁ וְנַעְלַיִם,

וּבוֹרֵא אֲשֶׁר אֵין לוֹ גְּבוֹהָה, כְּנִבְרָאִים, / וּמַטָּה וְאַרְבָּעָה עֲבָרִים, וּבֵינַיִם,

מְתַכֵּן בְּזֶרֶת אֶת שְׁחָקָיו, וְכָל אֶרֶץ / בְּשָׁלִישׁ, וְשׁוֹקֵל אֶת גְּבָעוֹת בְּמֹאזְנַיִם,

וּמֵסֵב רְקִיעָיו הָעֲגֻלִּים עֲלֵי אֶרֶץ, / כְּהָסֵב כְּלִי מִתּוֹךְ מְחוּגָה וְרֵחַיִם,

וְעַל כֵּן בְּקַו עָמְדָה בְּתָוֶךְ, בְּלִי בֶרֶךְ / וְרֶגֶל מְקוֹמֶמֶת, וּבִלְתִּי יְרֵכַיִם,

וְשָׂם זֶה לְעֻמַּת זֶה, כְּאוֹרָה וְחֹשֶׁךְ, אוֹ / כְּקֶדֶם וְאָחוֹר, אוֹ כְּפָנִים וְשׁוּלַיִם,

וְשָׁת אֶת יְמֵי אָדָם טְפָחוֹת, יְמֵי קֵץ זֶה / לְנֶגְדּוֹ כְּמוֹ טֶפַח, וְקֵץ זֶה – טְפָחַיִם,

וְנוֹתֵן לְכָל בָּשָׂר בְּעִתּוֹ כְּפִי אָכְלָם, / וְנַעַל לְרַגְלֵיהֶם, וְחָגוֹר לְמָתְנַיִם,

מְחַיֶּה נְכֵאִים, יוֹם שְׁלִישִׁי, וְנִדְכָּאִים, / בְּמָצוֹק, נְשָׂאוּהוּ כְּחוֹל יָם בְּיוֹמַיִם,

וּמוֹשֵׁךְ לְהֶרֶג עַם מְמֻשָּׁךְ בְּעֵבֶר נַ־/הֲרֵי כוּשׁ וְגוֹי קַו־קָו וְצִלְצַל כְּנָפַיִם,

וּמֵבִישׁ עֲצַת מַגְבִּיר־לְשׁוֹנוֹ, כְּאוֹמֶרֶת: / ‘הֲלֹא יַחְלְקוּ שָׁלָל, לְאִישׁ רַחֲמָתַיִם’,

וְאוֹמֵר: ‘הֲיַצִּילְכֶם אֱלֹהִים – וְאַיֵּה נָא / אֱלֹהֵי חֲמָת, אַיֵּה אֱלֹהֵי סְפַרְוַיִם?’

וּפוֹצֶה עֲנִיֵּי עַם בְּצָרָה כְּמַבְכִּירָה, / אֲשֶׁר בָּא, וְלֹא נָפַל, יְלוּדָהּ בְּאָבְנַיִם,

וְסָר מֵאֱרֹב עַל בֵּית חֲבֵרוֹ, וְלֹא נִפְתָּה / בְּאִשָּׁה, וְהִיא זִמָּה כְּגֹמֶר בְּדִבְלַיִם,

וּמִכָּל עֲווֹן חַטָּאת, אֲשֶׁר יוֹם פְּנוֹת זַכִּים / לְקַבֵּל שְׂכַר צִדְקָם מְאַחֵז עֲקֵבַיִם,

מְמַלֵּא לְפִי זַכִּים כְּצִדְקָם, וּמַשְׂבִּיעַ / לְזֵדִים וְזֵדוֹנִים כְּרִשְׁעָם בְּחָפְנַיִם,

וְעוֹבֵר עֲלֵי פֶשַׁע לְעוֹבֵר עֲלֵי פֶשַׁע / אֲנָשִׁים, וְאֶרֶךְ אַף לְמַאְרִיךְ בְּאַפַּיִם,

וּמוֹדֵד לְכַף רָשָׁע בְּרִשְׁעוֹ סְאָה בִסְאָה / וְגוֹמֵל בְּחֵיק צַדִּיק לְעֹמֶר בְּסָאתַיִם,

וּפוֹדֶה לְמַעְרִיצוֹ, כְּמוֹתִי, וּבַל יִפְדֶּה / לְעָרִיץ מְנַגֵּחַ כְּבַעַל קְרָנַיִם

אֲנַשֵּׂא בְּשִׁירָתִי מְנַשֵּׂא כְּבוֹדִי מִן / שְׁאוֹלָה, וְאֶשָּׂא לוֹ לְבָבִי בְּכַפָּיִם,

בְּשִׁירָה אֲשֶׁר תִּנְעַם בְּמִלִּים, וְתִישַׁר עַל / לְבַב כָּל נְבוֹנֵי לֵב, וְתִיטַב בְּאָזְנַיִם,

וְיֵלְכוּ אֲנָשִׁים בָּהּ לְחַמַּת וְשַׁעְרַיִם / וְשִׁנְעָר, וְתַגִּיעַ לְאֶרֶץ מְרָתַיִם,

וְתִקְרָא, בְּלִי גָרוֹן, בְּגַפֵּי מְרוֹם קָרֶת, / וְתֵלֵךְ לְמֶרְחַקִּים בְּלִי שׁוֹק וְרַגְלַיִם,

וְאֶשַּׁח לְאֵל הֵשַׁח כְּאֵבִי, עֲדֵי אֶגַּע / לְאָרֶץ וְעַד אַדְבְּיק לְקָדְקֹד בְּאַפְסַיִם

זְכַרְתִּיו – וְרֹב חַסְדּוֹ בְּשִׁבְעִים מְתֻקָּנִים / שְׁקַלְתִּים וְלוּ אוּכַל זְכַרְתִּים בְּאַלְפַּיִם

וְאוּלָם אֲמַלֵּא מַחֲסוֹרָם בְּתַחְנוּנוֹת / וְרִנּוֹת בְּעֶרֶב־יוֹם וּבֹקֶר וְצָהֳרַיִם,

וְאָקוּם וְאֶשְׁתַּחְוֶה לְפָנָיו בְּנוּם אִישִׁים, / וְאֶכְרַע עֲלֵי בִרְכַּי וְאֶקֹּד לְאַפָּיִם.
