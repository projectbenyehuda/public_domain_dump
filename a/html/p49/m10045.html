זְמַן רַע

וְחָלָה אֶלְיָסָף אָחִי מֵחֳלִי הַשְּׁחִין הַיּוֹצֵא בַּנְּעָרִים בְּסִיוָן שֶׁל שְׁנַת תתי"ג [4813; 1063] וְהָיָה צֵאתוֹ עָלָיו קֹדֶם הַשָּׁבוּע מִסֵּדֶר וַיִּקַּח קֹרַח יוֹם אֶחָד וְאָמַר כָּזֶה, יוֹקִירֵהוּ אֵל, בַּחֲיוֹתוֹ מֵחָלְיוֹ:

זְמַן רַע יְצִיקֶךָ בְּפַחְזוֹ וְחֶפְזוֹנָיו, וְשָׂמָךְ כְּמַטָּרָה לְחִצָּיו וְכִידוֹנָיו,

וְשָׁת אֶת לְבָבָךְ בֵּית יְגוֹנָיו, וְגוּפָתָךְ נְגוּעָה וְנִגֶּפֶת בְּצוֹמָיו וְצִמְאוֹנָיו,

וּפָצַע וְנָשַׁךְ אֶת בְּשָׂרָךְ, כְּאִלּוּ אֵל נְתָנָךְ בְּמַתָּנָה לְצִפְעוֹ וְצִפְעוֹנָיו,

וְאִם יַחֲרִישׁ רֶגַע כְּמַשְׁלִים אֲזַי יִשְׁאֶה כְּלוֹחֵם וְיִתֶּן קוֹל כְּפַעְמוֹן בְּרִמּוֹנָיו,

וּמַה טּוֹב לְךָ בִזְמַן – וְשָׂבְעוֹ כְּרַעְבוֹנָיו, וְטוּבוֹ כְּרָעָתוֹ, וְעָשְׁרוֹ כְּחֶסְרוֹנָיו,

יְשַׁלֵּח לְךָ הַיּוֹם בְּרָקָיו – וְיוֹם מָחָר יְגוֹרֵר בְּשָׂרָךְ עַל אֲטָדָיו וּבַרְקָנָיו,

וְיַלְבִּין פְּנֵי נַעַר וּבָחוּר בְּשַׁחְרוּתוֹ, וְיַשְׁחִיר פְּנֵי זָקֵן וְיָשִׁישׁ בְּלִבּוּנָיו,

וְיִסֹּב, וּבִי יָשׁוּב, וְיִתֵּן סְבִיבוֹתַי פְּקֻדּוֹת לְהָשִׁיב מַתְּנוֹתָיו וּפִקְדוֹנָיו.

אֲדַמֶּה אֲשֶׁר אֵין לַזְּמַן שִׂיג וְלֹא שִׂיחַ לְבַד בִּי, וְעָלַי מַחְשְׁבוֹתָיו וְחֶשְׁבּוֹנָיו.

הֲיֵשׁ מִזְּמַן גּוֹאֵל, יְסוֹכֵךְ יְחִידָתִי, בְּעָרְכוֹ קְרָב אֵלַי, בְּשִׁלְטוֹ וְשִׁלְטוֹנָיו?

אֱלֹהַי מְאַמֵּץ אֶת זְרוֹעִי בְּרַפּוֹתוֹ זְרוֹעִי, וְאֵין רָעָה לְנַפְשִׁי בְרִפְיוֹנָיו,

וּמִי כָאֱלֹהִים לַהֲרֹס אֶת אֲשֶׁר בָּנָה זְמַן רַע, וְחִזֵק אֶת אֲבָנָיו וּבִנְיָנָיו?

זְמַן, הַמְחַבֵּל טוֹב וְרָעָה לְעִתּוֹתָיו וְיָמָיו, הֲדִמַּנִי בְחֶבְלוֹ וְהֵרוֹנָיו.

זְמַן לִי בְּיוֹם אֶחָד – וְעָלַי בְּיוֹם שֵׁנִי, וְחָבֵר לְבֶן אָדָם – וְחָבֵר לְמוֹ שׁוֹנָיו,

וְיוֹם זֶה יְמַלֵּא פִי בְטוֹבָה וְיוֹם מָחָר יְמָרֵר בְּפִי אֶת טוּב מְגָדָיו וּמִגְדָּנָיו,

וְאִם אֶת יְהוֹסֵף יַעֲלֶה מֵחֳלָיָיו – אָז יְשַׁלַּח בְּאֶלְיָסָף חֲמָתוֹ וְחַרְבוֹנָיו,

וְעַתָּה הֱנִיאַנִי בְּאֵשׁ שָׂם בְּעַצְמוֹתָיו, וּבִשְׁחִין נְעָרִים שָׁת בְּגוּפוֹ וְעַל פָּנָיו,

כְּאִלּוּ יְקוֹד כִּבְשָׁן בְּגוּפוֹ, וְחַבּוּרוֹת נְפוּחוֹת עֲלִי גִלְדּוֹ כְּפִיחַ בְּכִבְשָׁנָיו,

וּמַרְאֵה נְגָעָיו עַל בְּשָׂרוֹ כְּמוֹ מִחְיַת פְּצָלוֹת יְפַצֵּל אִישׁ בְּלוּזָיו וְעַרְמוֹנָיו,

וְאוּלָם אֱלֹהִים סַךְ בְּצִלּוֹ וְאֶבְרָתוֹ, וְכִסָּה בְּיָדוֹ עַל בְּבוֹתָיו וְעַל עֵינָיו,

וְשָׁלַח אֲרוּכָה לוֹ מְהֵרָה, וְלֹא אָרַךְ בְּחָלְיוֹ, וְשָׁת זִָרים פְּדוּתוֹ וּפִדְיוֹנָיו,

וְכֵן מַעֲשֵׂה צוּרִי לְעַבְדּוֹ אֲשֶׁר יִחֵל לְחַסְדּוֹ וּבוֹ שָׁת אֶת חֲסוּתוֹ וּבִטְחוֹנָיו

מְצָאוֹ חֳלִי פִתְאוֹם וּמִנָּעֳמוֹ הָעְתַּק, וּמִשַּׁאֲנַנּוֹ אֶל הֲמוֹנָיו וְלִשְׁאוֹנָיו,

וּבָא לוֹ וְהוּא קוֹרֵא בְּלוּחוֹ בְּבֵית סֵפֶר וְהוּמַר אֳלֵי הֶגֶה הֲגִיגוֹ וְהֶגְיוֹנָיו,

קְרָאוֹ בְּשָׁבוּעַ, דְּבַר מַעֲוַת קֹרַח בְּסֵפֶר מְקֹרָא בוֹ, וְעָצַב לְעִצְבוֹנָיו,

כְּאִלּוּ בְּנִי צִוָּה חֳלָאָיו לְעַנּוֹתוֹ וְלֹא יַעֲנֶה רַע בַּאֲבוֹתָיו וּבִזְקֵנָיו,

וּבָחַר לְהִמָּלֵא שְׁחִין רַע וְלָשֵׂאת אֵשׁ עֲצָמוֹת וְלֹא יִשָּׂא חֲרָפוֹת בְּקַדְמוֹנָיו.
