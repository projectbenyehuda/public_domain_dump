שובי יפיפיה חלי פני דודך

חתימה: שלם [שבזי] ש״צ [שמרו צורו]


שׁוּבִי יְפֵיפִיָּה / חַלִּי פְנֵי דּוֹדֵךְ

אַל תִּישְׁנִי רַעְיָה / רַב לָךְ צְפוּן סוֹדֵךְ

עֵצָה וְתוּשִׁיָּה / בָּם חַזְּקִי יָדֵךְ

אָז תִּמְצְאִי מִחְיָה / אִם יִגְבְּרָה חַסְדֵּךְ

שׁוּבִי בְרוֹן / הַסֵּהֲרוֹן

לִבְשִׁי לְחֵן / וַעְשִׂי רְצוֹן

הָאֵל שְׁהוּא קוֹנֵךְ

לוֹ תִשְׁאֲלִי / יִרְפָּא חֳלִי

אָז תִּצְהֲלִי / גַּם תַּעֲלִי

תַּמָּה לְאַרְמוֹנֵךְ

מַה טּוֹב עֲלוֹת / הַמִּגְדְּלוֹת

שָׁם מִגְבְּלוֹת / כָּל מִקְהֲלוֹת

עַמֵּךְ וְהָמוֹנֵךְ

דּוֹד שֻׁלְחֲנוֹ עָרַךְ / יִגְדַּל וְיִתְבָּרַךְ


לִבְשִׁי לְבוּשׁ הָדָר / כִּי בָא זְמָן יִשְׁעֵךְ

זִיו בַּת אֱדוֹם קָדָר / כִּי הִרְבְּתָה מִרְעֵךְ

הוֹדִי לְאֵל נֶאְדָּר / מוֹחֶה לְכָל פִּשְׁעֵךְ

עוֹל הַזְּמָן נֶעְדָּר / וַיֶּאֱתֶה שִׁפְעֵךְ

שִׁפְעַת כְּבוֹד / נֵצַח וְהוֹד

גַּם הַיְסוֹד / כָּל אָח וְדוֹד

לָךְ שׁוֹפְעִים כֻּלָּם

לֹא תִירְאִי / הִמָּלְאִי

אַל תִּדְכְּאִי / חֵן תִּמְצְאִי

לִפְנֵי אֲדוֹן עוֹלָם

מִיּוֹם זְמָן / כִּלְכַּל בְּמָן

עָם נֶאֱמָן / וַיְקוֹמְמָן

וַיְסֹבְבָה דִּגְלָם

חִזַּק לְכָל מֹרֶךְ / זָנָם בְּכָל צֹרֶךְ


מָשַׁךְ זְמָן עָלַי / הַחֵן וְהַטּוֹבָה

אַרְצִי וּמִגְדָּלַי / נִשְׁלַל בְּמַעְצֵבָה

פִּשְׁעִי וּמַעְלָלַי / גָּרַם וְהַחוֹבָה

וָאֶסְבְּלָה, אוּלַי / אֶשְׁמַע בְּדוֹדִי בָא

שַׁוְעִי שְׁעֵה / צוּרִי דְּעֵה

צֹאנָךְ רְעֵה / גַּם הַסְּעוֹ

עַל מַחֲמַד הָרִים

לִבְנֵי סְגוּל / זָכְרָה גְמוּל

עָלָיו חֲמוּל / וּבְיוֹם הֲמוּל

אַל יִמְשְׁלוּ זָרִים

מַלְכָּךְ שְׁלַח / עָם נֶאֱלַח

דַּרְכּוֹ צְלַח / אוֹרוֹ זְרַח

יֵלֵךְ בְּמֵישָׁרִים

גַּם חַזְּקָה בֶּרֶךְ / עָם נֶעֱבָד פֶּרֶךְ


שׁוּר מִשְּׁמֵי מָרוֹם / וּרְאֵה עֳנִי עַמָּךְ

דִּגְלוֹ בְחֵן הָרֵם / קַנֵּא אֲדוֹן לִשְׁמָךְ

וּרְצֵה וְהַחְזִירֵם / לִרְאוֹת בְּמַנְעַמָּךְ

וּמְחֹל לְחֵטְא גּוֹרֵם / וּמְרַחֲקֵם מִמָּךְ

שׁוּב אֶל מְקוֹם / קָדְשָׁךְ, נְקֹם

מִדּוֹחֲקֵם / יַד צָר עֲקֹם

וּשְׁפֹט לְהַר עֵשָׂו

לוֹ שַׁלְּמָה / אַף וְחֵמָה

אָז יִיהֲמָה / אַל יִשְׂטְמָה

עַמָּךְ וְיַכְעִיסָיו

מֵחַסְדָּךְ / צַו שִׁפְעֲךָ

אֶל עַמְּךָ / תֶּן יִשְׁעֲךָ

אָז יֵבְשׁוּ שׁוֹסָיו

כִּי הַזְּמָן אָרַךְ / עָלַי וְלִבִּי רַךְ


צוּרִי וּמָגִנִּי / עֶזְרַת בְּנֵי אִישׁ תָּם

אַל תִּרְחֲקָה מֶנִּי / גַּלֵּה לְאוֹר נֶעְתָּם

אָשׁוּב מְקוֹם גַּנִּי / מִקֵּץ אֲשֶׁר נִסְתָּם

וַתִּשְׁמְעָה אָזְנִי / שֶׁמֶץ וְלֹא נֶחְתָּם

מַה יַּעֲנֶה / לֵב נַעֲנֶה

לֹא יֶהֱנֶה / בַּמַּחֲנֶה

מִזִּיו כְּבוֹד מַעְלָה

שָׁמַם וְסָר / עָלַי כְּשָׂר

וַיֶּחֱסָר / בַּמַּאֲסָר

עִם בַּעֲלֵי עַוְלָה

תָּמַהּ מְאֹד / וַיַּחֲמֹד

לֵידַע בְּסוֹד / וַיַּעֲמֹד

זְכָר עָנִי חָלָה

אֵלִי רְאֵה / צוּרִי דְּעֵה

שַׁוְעִי שְׁעֵה / אוֹתִי פְּדֵה

מִיַּד בְּנֵי עַוְלָה

בָּא צַד שְׂמֹאל עָרַךְ / לִלְחֹם וְחֵץ דָּרַךְ


אֶחְתֹּם שְׁבָח אֵל חַי / בִּלְשׁוֹן מְצֻחְצַחַת

לָשׁוֹן כְּמַלְקוֹחַי / לִשְׁמוֹ מְשַׁבַּחַת

שָׁלוֹם לְכָל אַחַי / תַּזְכִּיר וְשׂוֹמַחַת

וּבְסוֹד לְהַפְצִיחִי / תָּמִיד מְפַעְנַחַת

מַלְכִּי מְחוֹל / עָוֹן כְּחוֹל

עַמָּךְ נְחוֹל / טָבְעוֹ שְׁחוֹל

נִשְׁמַע בְּבוֹא תִּשְׁבִּי

שַׁוְעַת עָלוּב / שִׁמְעָה וְשׁוּב

רַפֵּא מְשׁוּב / לִבּוֹ נְבוּב

הָסֵר לְכָל עָצְבִי

תַּקְשִׁיב לְקוֹל / דָּל מִנְּפוֹל

חֶסֶד גְּמוֹל / יָמָיו כְּצֵל

וִימִינְךָ תַּצְבִּיא

אללה וחיד / איצא וסיד

ירבל עביד / יסלי אלנכיד

ואתוב עין דנבי

[הָאֵל יָחִיד / וְגַם אָדוֹן

מְקַבֵּל תְּשׁוּבַת יְרֵאִים / מְנַחֵם נִדְכָּאִים

אֶחֱזֹר בִּתְשׁוּבָה מֵחֶטְאִי]

כִּי אֵין לְךָ עֵרֶךְ / תִּגְדַּל וְתִתְבָּרַךְ.
