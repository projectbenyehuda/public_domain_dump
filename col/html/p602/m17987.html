הַשְׁלִישִׁיָה הַזֹאת (this threefold kindness) הִטִילָה עָלַי חוֹב קָדוֹשׁ לְשַלֵם לָךְ תּוֹדוֹת וּבְרָכוֹת אֵין סְפֹרוֹת, אָדוֹן רַב חֶסֶד וּצְדָקָה! וְאוּלָם יַעַן קָצְרָה לְשׁוֹנִי מִלְהַבִּיעַ לָךְ כָּל מוֹרָשֵׁי לְבָבִי אֲשֶׁר יִתְרֹצְצוּ בְּקִרְבִּי, מָצָאתִי לְנָכוֹן לְהוֹדוֹת לָךְ בְּדֶרֶךְ אַחֶרֶת כַּאֲשֶׁר יָאֲתָה לָךְ, וְגָמַרֵתִּי לְהַקְדִיש אֶת מִנְחָתְךָ הַיְקָרָה לְהַדְפִיס בִּמְחִירָה זֹה הַחוֹבֶרֶת אֲשֶׁר תָּכִיל הֶגְיוֹנַי שֶׁכָּתַבְתִּי לִכְבוֹדְךָ וְלִכְבוֹד אַנְשֵׁי שֵׁם, הַנוֹדָעִים לִתְהִלָה בְּסַפְרוּתֵנוּ הַקְדוֹשָׁה, וּלְשַלְחָה לְהוֹד כְּבוֹדְךָ לְמַזְכֶּרֶת תּוֹדָה וּבְרָכָה לְיוֹם הֻלַדְתֶּךָ הַנִכְבָּד, חֵלֶף הַכָּבוֹד הַגָדוֹל אֲשֶׁר עָשִׂיתָ לִי בְּרֹב חַסְדְךָ, וְאֶת פִּדְיוֹם הָעֶקְסֶעמְפְּלַארֶע אֲשֶׁר אֶמְכֹּר מֵחוֹבֶרֶת הַזאֹת, נָדַרְתִּי נֶדֶר לִשְׂלֹחַ לְמַעַן כְּבוֹדְךָ לַעֲנִיֵי אַדְמַת הַקֹדֶשׁ. וְעַתָּה, אָדוֹן רָם וְנִשָׂא! צְבִי תִפְאֶרֶת עַם אֱלֹהֵי אַבְרָהָם! אַחֲרֵי אֲשֶׁר אֲחַלֶה פְּנֵי כְּבוֹדֶךָ, כִּי זֹה תְּשׁוּרָתִי הַקְטַנָה תִהְיֶה לָךְ לְרָצוֹן וּבְנֶפֶשׁ חֲפֵצָה תְקַבְּלֶנָהּ מִיָדִי, אֲבָרְכֶךָּ שֶׁאֵל נַעֲרָץ בְּסוֹד קְדושִׁים רַבָּה יוֹסֶף לִכְבוֹדְךָ אוֹרֶךְ יָמִים וּשְׁנוֹת חַיִים טוֹבִים וּנְעִימִים, גַם עֹז וְתַעֲצֻמוֹת, שֶׁתָּנוּב בְּשֵׂיבָה טוֹבָה, דָשֵׁן וְרַעֲנָן, וְתַּשִׂיג עַד מְהֵרָה מַטְרַת חֶפְצְךָ הַקְדוֹשָׁה, וּבִזְמַן קָרוֹב נִזְכֶּה כֻלָנוּ לִרְאוֹת בְּהוֹד תִּפְאֶרֶת צִיוֹן וִירוּשָׁלַיִם. כַּעֲתִירַת עַבְדְךָ הַמִשְׁתַּחֲוֶה מוּל הַדְרַת כְּבוֹדְךָ בִּתְּשׁוּאוֹת תּוֹדוֹת יהושׁע אטלאס.

אֵם הָיְתָה לִישֻׁרוּן לְעִתּוֹת בַּצָרָה,

וּמַטְרַת חֶפְצָהּ רַק חֶמְלָה וַחֲנִינָה!

וְזֹה הַנֶפֶשׁ – זֹה הַנְשָׁמָה הַנִשְׂגָבָה,

לְעִצְבוֹן שִׁבְטֵי יַעֲקֹב הַתֵּבֵל עָזָבָה! –

4.

אֲבָל הַסוּ מוֹרָשַׁי! עוֹדֶנָהּ בַּחַיִים!

רוּחָהּ גַם נַפְשָהּ עוֹדֶנָם בֶּחָלֶד!

מְהֻלָלִים הֵמָה בְּפִי שָׂב וָיָלֶד,

וְעַל כָּל תִּשְׁבָּחוֹת נִפְלָאִים שִׁבְעָתַיִם!

כֹּה בַחַיִים אֲשׁוּרֶנָהּ גַם בְּסִפְרָתָהּ,

וּכְהוֹד צַלְמְךָ1 תּוֹפִיעַ בְּעֶצֶם תֻּמָתָה!

5.

וְסִפְרָתָהּ זֹאת לִי הִפַּלְתָּ לַחֶבֶל? –

לִי לְמָנָה שִׁלַחְתָּהּ? הֲלֹא שָׁגִיתִי?

לִסְגֻלָה כָזֹאת, הֶאָח לֹא קִוִיתִי,

וּלְעֻמָתָה מִכְמַנֵי פָּז אַךְ הֶבֶל!

הֶבֶל לְעֻמָתָהּ כָּל גִנְזַכֵּי מְלָכִים,

וַעֲלֵיהֶם אֲרוֹמְמֶהָּ עַד נֶצַח נְצָחִים!

6.

כִּי מִמֶנָהּ אֶחֱזֶה כְּבוֹד הוֹדֶךָ!

אֶחֱזֶה אֵיךְ עָזַבְתָּ הֵיכְלֵי מַחֲשַׁקִים,

לָנוּעַ לְמַעַן עַמְךָ לְאֶרֶץ מֶרְחַקִים –

לְאֶרֶץ הַקְדוֹשָׁה, בָהּ רָבוּ חֲסָדֶיךָ!

וּמִנִטְעֵי שַׁעֲשׁוּעִים אֲשֶׁר שָׁמָה שָׁתַלְתָּ,

אַכִּיר וְאֶתְבּוֹנֵן מַה לִישֻרוּן פָּעַלְתָּ!

7.

אַךְ לֹא רַק שָמָה הִפְלֵאתָ מִפְעָלִים!

לֹא רַק שָׁמָה הָיִיתָ מְרַחֵם!

לֹא רַק שָׁמָה הוֹפַעְתָּ כִּמְנַחֵם!

גַם בְּפַאֲתֵי קֶדֶם2 הוֹשַׁעְתָּ אֻמְלָלִים!

וּבְצָפוֹן3 וּבְנֶגֶב4 וּבְכָל יַרְכְּתֵי הָאָרֶץ,

חִלַצְתָּ דַכְּאֵי־רוּחַ מִשׁוֹד וָקָרֶץ!

8.

וּלְמִי אֲדַמְךָ אֵיפֹה, מַלְאָךְ מוֹשִׁיעַ?

רַק לְבֶן עַמְרָם רַב הָעֲלִילִיָה!

כָּמוֹהוּ גַם אַתָּה הִגְדַלְתָּ תֻּשִׁיָה,

וּכִּשִׁמוֹ גַם שִׁמְךָ לָנֶצַח יוֹפִיעַ!

לִפְנֵי שֶׁמֶשׁ יִנוֹן עֲלֵי אֲדָמוֹת,

וּלְדוֹר אַחֲרוֹן יוֹרֶה נִשְׂגָבוֹת וְרָמוֹת!

9.

וּמִי יִתֵּן אֵל הַמַגְבִּיהִי לָשָׁבֶת,

יְחַדֵשׁ כַּנֶשֶׁר יְמֵי שְׁנֵי חַיֶיךָ!

בְּשָׂשׂוֹן וּבְרִנָה תִקְצוֹר פְּרִי מִפְעָלֶיךָ,

וְעִמְךָ יַצְלִיחַ הַלֵוִי בְּרוּחַ נִשְׂגָבֶת!

אַתֶּם, הַשָׂרִים! תַּשְׁבִּיתוּ יָגוֹן וָאֲנָחָה,

וְעִמְכֶם יַעַל יְהוּדָה לִמְרוֹם הַהַצְלָחָה!!

יהיוּ נא לרצון אמרי פי, והגיון לבי לפניךָ, כַּאֲוַת נפש עבדךָ, המוקירךָ כערכך הרם, והמשתחוה לך בענות כבוד.


מִכְתָּם לְמשֶׁה.

לכבוד נשיא ישראל וצבי תפארתנו, רודף צדקה וחסד, ופועל ישועות בקרב הארץ, האומר לירושלים: תוּשב! ולערי יהודה: תבנינה! וברוב חסדיו יתנדב להשלים עצתו, כש“ת מוהר”ר סיר משׁה מונטיפיורי5, צדיק כתמר יפרח, ויתענג על רב טוב עד עולם!

1.

עוֹד כְּצִלְצְלֵי שֵׁמַע בִּתְּשֻׁאוֹת שָׂשׂוֹן וָגִילָה

תּוֹדוֹת חַסְדְךָ הָרִאשׁוֹן לְךָ הוֹמֶה לְבָבִי,

עוֹד רוּחִי וְנַפְשִׁי לְךָ תַּבַּעֲנָה תְהִלָה,

וּמֵהֵד רִגְשוֹתֵיהֶן יִרְעֲשׁוּ מֵיתְרֵי עֻגָבִי,

וְהִנֵה, הֶאָח! עוֹד הִרְבֵּיתָ לִי שַׁעֲשׁוּעַ,

וּמֵחָדָש מִמְרוֹם הוֹדְךְ לִי אוֹר זָרוּעַ!

2.

מֵחָדָשׁ שֶׁמֶשׁ חַסְדְךָ עָלַי הוֹפַעְתָּ,

וּפִי שְׁנַיִם כִּבַדְתָּנִי בִּסְפָרִים יְקָרִים,

וּבָמוֹ מֵחָדָשׁ רַב טוּבְךָ הוֹדַעְתָּ

גַם הִפְלֵאתָ לַעֲשׂוֹת, פּוֹעֵל מֵישָרִים!

כִּי מֵהֶם אֶחֱזֶה שִׂיאֲךָ בֶּהָדָר וְתִפְאֶרֶת,

וְשַׁלְהֶבֶת אַהֲבַת אָדָם בְּקִרְבְּךָ בֹּעָרֶת.

3.

אַהֲבַת אָדָם? הֶאָח, מַה נְעִימָה וִיקָרָה,

מַה יְקָרָה וּנְעִימָה זֹה בַת הַשָׁמַיִם!

שָׁם בְּהֵיכַל קוֹנָם, שָׁם בַּקֹדֶשׁ נֶאֱדָרָה,

וּמִתוֹעֲפוֹת כֵּס הוֹדוֹ שֻׁלְחָה לִמְתֵי הַחַיִים.

מִשָׁם הוֹרִידָה אֵל רוֹכֵב עֲרָבוֹת,

לִנְטוֹעַ שָׁלוֹם בַּתֵּבֵל, וּלְחַבֵּר הַלְבָבוֹת.

4.

אַךְ זֹה הַתֵּבֵל, בָּהּ קִנְאָה פָּרָחָה,

בָהּ חֵמָה וּמַשְׁטֵמָה דָגְרוּ חֵיל מְרֹרוֹת,

וּמִתִּגְרַת יָדָן בַּת הַשָמַיִם נִדָחָה,

וַתֵּלֶךְ לָנוּע שָׁנִים אֵין סְפֹרוֹת;

כֹּה רִבֲבוֹת דֹרוֹת עֶרְכָּהּ לֹא הוֹקִירוּ,

עֲדֵי קַרְנֵי זִיז כְּבוֹדְךָ מִמְרוֹמְךָ הֵאִירוּ.

5.

וּמִמְרוֹמְךָ, משֶׁה! שָׁלַחְתָּ צְבָא מַלְאָכֶיךָ:

יוֹשֶׁר, צֶדֶק וָחֶסֶד, חֶמְלָה וַחֲנִינָה,

וַיָבִיאוּ הָאַהֲבָה בְצֵל כַּנְפֵי הוֹדֶךָ

וּבְצֵל כַּנְפֵי הוֹדֶךָ חָסָיָה הָעֲדִינָה.

בָּךְ מָצְאָה מִשְׂגַבָּהּ, סַלְעָהּ וּמְצוּדָתָהּ,

וַתַּאֲהִיל מִקְדַש הֶחָלֶד בִּבְרַק יִפְעָתָהּ!

6.

כֵּן הִצַתָּ אִשָהּ בְּלִבּוֹת אַנְשֵׁי רֶשַׁע,

כֵּן נִדְחֵי יְשֻׁרוּן מִשוֹד וָעֹשֶׁק הִצַלְתָּ,

וּבְכָל פַּאֲתֵי תְפוּצוֹתֵיהֶם הִלְבַּשְתָּם יֶשַׁע,

וְכֵן בָּךְ נִתְפָּאֵר, כִּי לַעֲשׂוֹת הִגְדַלְתָּ,

וְתִקְוָה טוֹבָה וּמְתוּקָה תְחַיֵנוּ וּתְנַחֲמֵנוּ:

שֶׁעוֹד נִשְׁבַּע עֲדָנִים בְּאֶרֶץ נַחֲלָתֵנוּ!

7.

שָׁם בְּאָרֶץ יְהוּדָה, בְּשַדְמוֹת הַרְרֵי קֹדֶשׁ,

שָׁם, מַלְאָךְ מוֹשִיעַ! בָּךְ נִשְׂמַח שָׂמֹחַ!

שָׁם נֶחֱזֶה אוֹר חָדָש כַּסַהַר בַּחֹדֶשׁ,

וּכְבוֹד צִיוֹן וִירוּשָלַיִם עוֹד יִפְרַח פָּרֹחַ,

כִּי אַתָּה רָצִיתָ אוֹתָן וַתַּחְפֹּץ בִּיקָרָן,

וּבְיוֹשֶׁר לְבָבְךָ הִתְנַדַבְתָּ לְחוֹנֵן עֲפָרָן.

8.

בְּחֶמְלָתְךָ שֶׁבַע פְּעָמִים נָסַעְתָּ שָמָה

וַתִּבֶן בִּמְקוֹם גַלִים מִקְדָשִׁים חֲדָשִים;

אַרְמְנוֹת כּוֹנַנְתָּ בִּמְקוֹם חָרְבָה וּשְׁמָמָה,

גַם בָּתֵּי מַרְפֵּא, וּמִשְׁכְּנוֹת דַלִים וְרָשִים,

וּסְבִיבוֹתָם נָטַעְתָּ שָׂדוֹת, גַנִים וּכְּרָמִים,

כְּמַטָעֵי יְמֵי קֶדֶם דוֹרוֹת עוֹלָמִים.

9.

לָזֹאת כָּל נֶפֶש אַחֶיךָ בְּאַלְפֵי בְרָכוֹת

תּוֹדוֹת וּתְהִלוֹת אֵין מִסְפָּר לְךָ רֹחֶשֶׁת,

תּוֹדָתִי לְךָ תַּקְרִיב בִּבְרָכָה מְשֻׁלֶשֶׁת:

אוֹרֶךְ יָמִים תִּשְׂבַּע וּלְמַטְרָתְךָ תַּנְחֵנוּ,

לְמַעַן נָשִׂישׂ בָּךְ מְהֵרָה בְּעָרֵי מוֹעֲדֵנוּ!

מי יתן, אדון רם ונשא! שהטורים האלה יפיקו ממך רצון כחפץ וַאֲוַת נפש עבדך, המוקירך ומכבדך כערכך הנעלה, והמשתחוה להוד כבודך בתשואות רחשי תודה.

רַחֲשֵׁי לִבִּי לִתְקוּפַת הַשָׁנָה

שנת תרל"ט.

לכבוד ידיד נפשי ולבבי, המליץ הנשגב והמשורר הנעים כש“ת מוהר”ר יעקב ביברינג נ"י!

ע“מ י”א וי"ב תנועות.

1.

בִּתְנוּמוֹת עֲלֵי מִשְׁכָּב הַתֵּבֵל נִלְכָּדָה,

וְתַּחַת כַּנְפֵי מַרְגוֹעַ שַׁאֲנָן יָשַנְתִּי,

וְהִנֵה קוֹל קוֹרֵא בַּחֲלוֹמִי הֶאֱזַנְתִּי,

וּשְׁנָתִי מֵעֵינַי בְדֵי רֶגַע נָדָדָה.

2.

חִישׁ בַּחֲדָרִי הִשְׂלַכְתִּי מַבָּט עֵינָיִם,

פָּנִיתִי לָדַעַת מֶה אָזְנִי לָקָחָה,

וְהִנֵה רוּחַ הַמְלִיצָה עָלַי צָלָחָה,

וַתִּמְשֹׁךְ לְבָבִי כִמְחֹלַת הַמַחֲנָיִם.

3.

“הִתְעוֹרֵר נָא גֶבֶר!” אֵלַי דִבֵּרָה,

"קַח כִּנוֹרְךָ, כִּי עֵת הַזָמִיר הִגִיעָה!

"עַל כְּנָפַי עוּפָה, רַחֲשֵׁי לִבְּךָ הַשְׁמִיעָה,

“וְלִתְקוּפַת הַשָנָה שִׁיר נָעִים זַמֵרָה!”

4.

“בְרוּכָה אַתְּ גְבִרְתִּי!” קָרָאתִי בְּלֵב שָׂמֵחַ,

וּבִמְרוּצַת בֶּן־קֶשֶת מֵעַרְשִׁי קַמְתִּי,

שִׁנַסְתִּי מָתְנַי, כִּנוֹרִי בְיָדִי שַׂמְתִּי,

וְעַד אַרְגִיעָה נְשָׂאַתְנִי כְנֶשֶר פּוֹרֵחַ!

5.

עַפְתִּי רַחֲבֵי תֵבֵל, עַל הָרִים וָעֲמָקִים,

עַל נַהֲרוֹת אֵיתָן וְגִבְעוֹת עוֹלָם דָאִיתִי,

בָּאתִי לְיַם הַנִצְחִי וְשָׁנָה זוּ רָאִיתִי,

וָאֵרֶא סְבִיבוֹתֶיהָ הֲמוֹן צִבְאוֹת שְחָקִים –

6.

אֶרְאֶלָם וּשְׂרָפִים נִקְבְּצוּ בִּרְנָנָה,

וְרִבֲבוֹת אַלְפֵי שִׁנְאָן בַּמְקְהֵלָה שָרוּ;

זְעֵיר שָׁם, זְעֵיר שָם אַחֲרֶיהָ הָלְכוּ וְאָמָרוּ:

“לְכִי לָךְ בַּשָׁלוֹם! לְכִי בַשָׁלוֹם שָׁנָה!”

7.

פִּתְאוֹם רָעֲדָה הָאָרֶץ, רָגְזָה הַשָׁפָה,

רָעֲשׁוּ גַלֵי הַיָם, הַמִשְׁבָּרִים הָמוּ,

הַמַיִם כַּסִיר רֻתְּחוּ, נֶחֱצוּ, נֶעֱרָמוּ,

וּכְהֶרֶף־עַיִן הַשָׁנָה נִסְחָפָה! –

8.

רָאִיתִי חָזוּת קָשָׁה וְרוּחִי חֻבָּלָה,

תָּעָה לְבָבִי, בַּדֵי עוֹרִי רָעָדוּ,

כָּרַעְתִּי, קָרַסְתִּי, רַגְלַי מָעָדוּ,

וּמָתְנַי, הָהּ מָתְנַי! מָלְאוּ חַלְחָלָה!

9.

אַךְ חִיש הִתְאַמַצְתִּי, אָסַפְתִּי אוֹנִי,

וְנָדַדְתִּי לַמֶרְחַקִים בְּקוֹל רֶגֶשׁ דְמָמָה;

מֻעָף בִּיעָף רָקַעְתִּי גָבְהֵי רָמָה,

גַם עַפְתִּי לַעֲרָבוֹת בְּכֹּחַ דִמְיוֹנִי.

10.

בָּאתִי הֵיכַל קֹדֶשׁ, מִשְׁכַּן אֱלֹהִים צְבָאוֹת,

חָזִיתִי אַלְפֵי מַזָלוֹת גַם מַזָרוֹת,

רִבֲבוֹת נְגֹהוֹת שַׁחַק, שְׁבִיבֵי אוֹרוֹת יְקָרוֹת,

וְנִדְהַמְתִּי, דוֹמַמְתִּי, נִפְלֵיתִי נוֹרָאוֹת!

11.

עַל שֻׁלְחָן מאֹד גָדוֹל וּרְחַב יָדַיִם

סֵפֶר הַזִכְרֹנוֹת שָׁם לְעֵינַי פָּתוּחַ;

בּוֹ כָל מַעֲלָל יָשָר, כָּל שָׁוְא וּמַדוּחַ,

מִפְעֲלוֹת כָּל בְּנֵי אָדָם בְּאֶרֶץ הַחַיִים.

12.

וּבְראֹשׁ הַשֻלְחָן אֵל יוֹשֵׁב לָדִין אָרֶץ,

וּקְדוֹשִׁים מַלְאֲכֵי רום סְבִיבוֹ יִדֹדוּן,

כָּל צְדָקָה וָפֶשַע בַמִדָה יִמֹדוּן,

וְיִכְתְּבוּ בַּסֵפֶר מִי לְחַיִים אוֹ לַקָרֶץ!

13.

כִּרְאוֹתִי זֹאת רַעַד אֲחָזַנִי שָׁמָה,

וּבְרֶגֶש קֹדֶשׁ עַל פָּנַי נָפָלְתִּי!

נֶאֱלַמְתִּי דוּמִיָה, דַבֵּר לֹא יָכָלְתִּי,

וּבְרוּחַ נִשְבָּרָה שַׁבְתִּי עֲלֵי אֲדָמָה!

14.

לָכֵן, אֵל נַעֲרָץ! לָךְ הַפַּעַם אָשִׁירָה,

וְלִכְבוֹדְךָ מֵיתְרֵי כִנוֹרִי אָנִיעָה;

עַל כַּנְפֵי רְנָנִים אֶת קוֹלִי אַשְׁמִיעָה,

וְרַחֲשֵי נַפְשִי וּלבָבִי לְךָ אָקִירָה!

15.

יָדַעְתִּי, אֵלִי! כִּי מְקוֹר הַחַיִים אַתָּה,

וְתֵּבֵל וּמְלֹאָהּ רַק חָכְמָתְךָ עָשָׂתָה;

בְּצַלְמְךָ הָאָדָם בִּגְדֹל זְרֹעֲךָ בָּרָאתָ,

וְתִפְאֶרֶת וְהָדָר, חֵן וְכָבוֹד לוֹ נָתַתָּה –

16.

אֵדַע כִּי כֹּחֲךָ גָדוֹל, וְיָדְךָ רוֹמֵמָה,

תּוּכַל לִפְעוֹל גְדֹלוֹת וּנְצֻרוֹת בֶּחָלֶד;

בִּתְבוּנָתְךָ תַּנְחֶה גַם שָׂב גַם יָלֶד,

וּכְמִשְׁפָּטֶיךָ תְּחַיֶה אָדָם וּבְהֵמָה.

17.

וּבְעֵת אֵלֶיךָ נִקְרָא תִשְׁמַע הַצְעָקָה,

תַּפִּיל צָרִים, תַּדְבֵּר תַּחְתֵּינוּ קָמִים,

כִּי לֹא תְאַחֵר הַטוֹב לַהוֹלְכִים בְּתָמִים,

וְעִמְךָ מְקוֹר הַמִשְׁפָּט, חֶסֶד וּצְדָקָה.

18.

לָכֵן אַחֲלֵי! אֹזֶן קַשֶׁבֶת לִי תֵנָה!

הַאֲזִינָה, הַקְשִׁיבָה קוֹל תַּחֲנוּנוֹתַי!

בְּעַד עַמִי וְדָתִי תִטֹפְנָה אִמְרוֹתַי,

בְּעַד שִׁבְטֵי נַחֲלָתֶךָ לְךָ אֶתְחַנֵנָה!

19.

פְּתַח חַלוֹנֵי רָקִיעַ, הַשְׁקִיפָה מָטָה,

וְהַבִּיטָה אֵיךְ בְּכָל הָאֲרָצוֹת נִפְזָרִים!

רְאֵה אֵיךְ נִרְדָפִים בֵּין לְאֻמִים אַכְזָרִים,

וְאֵיךְ נַפְשָׁם בְּחַיֵיהֶם שָׁמָה נָקָטָה!

20.

רְאֵה אֵיךְ בַּת צִיוֹן תְּפָרֵשׂ בְּיָדֶיהָּ,

וּשְׁלַח לָהּ עֶזְרָתְךָ מִשְׁמֵי קֹדֶשׁ!

כָּתְבֵנוּ בְּסֵפֶר הַחַיִים בְּזֶה הַחֹדֶשׁ,

וְתָחֵל הַשָׁנָה הַחֲדָשָׁה בְּבִרְכוֹתֶיהָ!!

21.

וּבְתוֹךְ חֲסִידֶיךָ לְיַעֲקֹב אִיש הָרוּחַ,

הוֹסִיפָה אוֹרֶךְ יָמִים בְּגִיל וּשְׂמָחוֹת!

הַשְׂבִּיעָה נָא נַפְשׁוֹ שַׁעֲשֻׁעִים וְצַחֲצָחוֹת

לֵאוֹר בְּאוֹר הַחַיִים כְּאוֹר שֶׁמֶשׁ זָרוּחַ!

22.

מַדוּעַ, אֵל שַׁדַי, חָכְמָה לוֹ אָצַלְתָּ,

וְרוֹמַמְתָּהוּ רַק בְּמַשְׂכִּיוֹת הַדָעַת?

לָמָה יְצַרְתּוֹ פַּרְדְסֵי רַעֲיוֹנִים לָטָעַת,

וְחֶבֶל בִּנְעִימֵי תֵבֵל לוֹ לֹא הִפַּלְתָּ? –

23.

נְסָה עָלָיו אוֹר פָנֶיךָ לַשָׁנָה חֲדָשָׁה,

וְהַחֲלֵף גוֹרָלוֹ לְטוֹב לוֹ כָל הַיָמִים!

יְחִי חַיֵי נַחַת, חַיֵי שָׂשׂוֹן לְעֹלָמִים,

וְשָׁלוֹם וּבְרָכָה יִהְיוּ לוֹ לְמוֹרָשָׁה!

24.

שֶׁמֶש הַהַצְלָחָה לוֹ תִזְרַח הָאהֱלָה,

וּבְצֵל הָעֹשֶׁר וְהַכָּבוֹד יִשְׁכֹּן שַׁלְאֲנָן!

יִפְרַח כַּתָּמָר, יִתְעָרֶה כְאֶזְרָח רַעֲנָן,

וְיִתְעַנַג עַל רַב טוֹב עַד עוֹלָם סֶלָה!

בִּרְכַּת תּוֹדָה

לראשית השנה תרל"ט. (נכתבה על־ידי מאת החותמים)

לכבוד המליץ הנשגב מנעים זמירות כש“ת מוה”ר יעקב עהרליך נשיא חברתנו, יזרח נרו סלה!

1.

עַל אֵם הַדֶרֶךְ מִבְּאֵר שֶבַע חָרָנָה,

בִּמְקוֹם קֹדֶש יַעֲקֹב אָבִינוּ חָנָה,

וַיַרְא בְּתַרְדֵמָתוֹ, בַּחֲלוֹם חֶזְיוֹן לָיִל,

סֻלָם גָבֹהָ רֹאשׁוֹ לַשָמַיִם מַגִיעַ,

וּבִּשְׁלַבָּיו אֶרְאֶלָם בְּעֹז וָחָיִל

יַעְפִילוּ לַעֲלֹת לִמְרוֹמֵי הָרָקִיעַ;

אָז נִרְאָה לוֹ עֶלְיוֹן וּבְרָכוֹת הֶעֱנִיקֵהוּ,

וְהוּא קָם וַיִצֹק שֶׁמֶן עַל אֶבֶן מְרַאשֹׁתֵהוּ.

2.

גַם לְךָ יַעֲקֹב! כַּמַחֲזֶה הַלָזֶה קָרָה,

בִּרְחֹבוֹת עִירֵנוּ עֵת רַגְלְךָ עָבָרָה!

פִּתְאוֹם חָזִיתָ סֻלָם שָׁפָה הָעִבְרִיָה,

בּוֹ נִצְרֵי עַם יְשֻׁרוּן עָלוּ מִשָׁפֶל,

וַתַּעַט בִּרְכַּת שַׁדַי, וַתַּסֵךְ שֶׁמֶן תֻּשִׁיָה,

לְהַעֲלֹת נֵר חָכְמָה פֹה בָּעֲרָפֶל,

וְכָּכָה, גֶבֶר נַעֲלָה, יָסַדְתָּ אֲגֻדָתֵנוּ,

וּבְלֶקַח יְקָר רוּחֲךָ, הֶאָח, תְּחַיֵינוּ!

3.

לָכֵן, נְשִׂיא רֹאשֵׁנוּ! לִתְקוּפַת הַשָׁנָה,

רַחֲשֵׁי לְבָבֵנוּ לָךְ נוֹבִילָה לְמָנָה:

אֹרֶךְ יָמִים תּוֹסֶף כְּגֶבֶר צוֹלֵחַ!

תִּחְיֶה חַיֵי עֹנֶג, חַיֵי שַׁלְוָה וָנַחַת!

לִפְנֵי שֶׁמֶשׁ תִּנוֹן עַד בְּלִי יָרֵחַ,

וְהַצְלָחָתְךָ תַּעַל כַּחֲבַצֶלֶת פּוֹרַחַת!

אַךְ לָאֹשֶׁר וּבְרָכָה יְהִי פָּתוּחַ בֵּיתֶךָ,

וִישַׂגֶבְךָ עַד עוֹלָם שֵם אֱלֹהֵי שְמֶךָ!!

כַאֲוַת נפשׁ מוקיריךָ ומכבדיךָ כערכךָ הרם:

“חברי: דורשי דעת”

לַשָׁנָה הַחֲדָשָׁה

אל המעלה נשיאים ירוממך, וירכיבך על במתי האשר, ויכתבך לחיים טובים ונעימים, ידיד נפשי, שארי ובשרי החכם השלם והמשורר הנפלא כש“ת מוה”ר מתתיהו שמחה ראבענער נ"י לנצח!

AKROSTICHON.

מָחָר עֵת הַבֹּקֶר אוֹרוֹ יַבְקִיעַ

תֵּרָאֶה שָׁנָה חֲדָשָׁה בְּאַרְצוֹת הֶחָלֶד;

תִּקְוָה טוֹבָה וּמְתוּקָה בָּה עָלֵינוּ תוֹפִיעַ,

יַחַד יוֹחִילוּ, יָגִילוּ יָשִׁישׁ גַם יָלֶד,

הֶאָח, מִי יִתֵּן, גֶבֶר יְקַר רוּחַ!

וְתִקְוָה זֹאת תְּמַלֵא כָּל מִשְאֲלוֹת לִבֶּךָ!

שֶׁמֶשׁ הָאֹשֶׂר כְּאוֹר חֶרֶס זָרוּחַ,

מִמְרוֹם גָבְהֵי שְׁחָקִים תָּאִיר נָוֶךָ!

חֶדְוָה וָשִׂמְחָה, שָׂשׂוֹן, גִיל וּרְנָנָה,

הַשָׁנָה הַחֲדָשָׁה לָךְ תָּבִיא הָאֹהֱלָה;

רִבֲבֹת בְּרָכוֹת לָךְ תִּפֹּלְנָה לְמָנָה,

אַךְ הֵנָה תְּסֹבְבוּךָ עַד עוֹלָם סֶלָה!

בְּמִשְׁכַּן כְּבוֹדְךָ עִם מִשְׁפַּחְתְּךָ הַנְעִימָה,

עַל רַב טוֹב תִּתְעַנַג וְלֹא תֵדַע אֲנָחָה!

נַחַת תִּשְׂבַּע, וְקַרְנְךָ בְּכָבוֹד תָּרִימָה,

עֲדֵי תִסַק שְמֵי הַתִּפְאֶרֶת וְתַעֲלֶה כָּכָה

רוּם בָּמֳתֵי הָעֹשֶר לִמְחוֹז הַהַצְלָחָה!

ערב ר“ה תרמ”א.

כַּאֲוַת אוהבך ומכבדך הנאמן.

לַמְנַצֵחַ עַל אַיֶלֶת הַשַחַר,

לכבוד גבר חכם בעוז, הנותן אמרי שפר, המשורר היקר והנעים המשביר לבני ציון “זמרת הארץ” כש“ת מוה”ר מתתיהו שׂמחה ראבענער נ"י ויזרח!

AKROSTICHON.

מִיוֹם גֹרַשְׁנוּ מִשַׁדְמוֹת אֶרֶץ יְהוּדָה,

תִּפְאֶרֶת לְשׁוֹן עֵבֶר יָרְדָה פְלָאִים.

תַּחַת שוֹאָה הִתְגַלְגְלָה הַחֲמוּדָה,

יָצְאָה בַגוֹלָה עִם בְּנֵי צִיוֹן הַמְדֻכָּאִים.

וּבִרְתֻקוֹת אַהֲבָה כָזֹאת וּכְּאֵבִי נֶעְכָּר שִׁבְעָתַיִם
לְיַעֲקֹב8 אָחִינוּ נִקְשַרְתִּי. בְּהִתְבּוֹנְנִי עַלָ עֲמָלְךָ
לוּלֵא אֶשְׁגֶה בָרוֹאֶה וַעֲמַל כָּל סוֹפְרֵינוּ!
יָשָׁר הוּא כִשְמוֹ אֲבוֹי כָּמוֹךָ כָּמוֹהֶם,
גַם בִּידִידוּתוֹ אֵלַי, יוֹמָם גַם לָיִל
כִּי הִרְאַנִי אוֹתָהּ בְּלִי שֶׁקֶט וּמְנוּחָה
כְּעֶצֶם הַשָׁמַיִם לָטוֹהַר כְּנִשְׁרֵי שַׁחַק תָּעוּפוּ
וּכְצֵאת הַשֶׁמֶשׁ בִּגְבוּרָתוֹ. עַל יַם הַדִמְיוֹנוֹת.
אַתָּה תָבִין לְרֵעַי… וּבָעֵת מֶלֶךְ הַמַרְגוֹעַ
עַל מֶה הָטְבָּעוּ, עַל בִּרְכֵּי הַתּוֹלָדָה
וְלִבִּי נָכוֹן בָּטוּחַ, יְיַשֵׁן מְתֵי תֵבֵל,
כִּי כַיָשָׁר הַלָזֶה עַל מִשְׁמַרְתְּכֶם תַּעֲמֹדוּ
כָּל תְּמִימֵי דֵעִים לְהִלָחֵם כָּל מִלְחֲמֹתֵינוּ! –
מִמַכִּירַי וְיוֹדְעֵי שְׁמִי וּמַה גְמוּל לָכֶם?
אֱמֶת יֶהְגֶה חִכָּם, הָהּ! נֶאֱלַמְתִּי דוּמִיָה! –
וְלֹא יִרְהֲבוּ בְּנַפְשָׁם בֵּית עַכָּבִיש תִּקְוָתֵכֶם!
טָפֹל עָלַי שֶׁקֶר… לֹא רַק כֶּסֶף,
הַשְׁמוּעָה בִּדְבַר שׁוֹשַׁנֶיךָ אַךְ גַם תּוֹדָה
נָגְעָה עַד לִבִּי, לֹא תִנְחֲלוּ בָּאָרֶץ!
וּלְפִי רְאוֹת עֵינַי אֲבָל, יַעֲקֹב יְקִירִי!
אֲשֶׁר יָגֹרְתִּי בָא. נַס וּדְרוֹךְ קַסְתֶּךָ
וּמָה אִיעָצְךָ הַפָּעַם? לָאֵלֶה אֲשֶר יָדַעְתָּ!…
מָה עֲשׂת אוּכָלָה? – הַזְכִּירֵם! בַּקְשֵׁם! הוֹכִיחֵם!
הָהּ אָבְדוּ עֶשְׁתּנוֹתַי אוּלַי תַשְׂכִּיל וְתַּצְלִיחַ
וְשַׂרְעַפַּי מֶנִי וָהָלְאָה! לְהַשִׂיג שְׂכַר פָּעָלְךָ.
הַיְאֻמָן כִּי יְסֻפָּר כֹּה יִתֵּן שַׁדַי!
כִּי מַשְׂכִּילֵי אַחֵינוּ, חֲיֵה בַּטוֹב וּבַּנְעִימִים,
צֶאֱצָאֵי עַם רַחְמָנִי וְתָחֹג אֶת חַגֵנוּ
יִטְמְנוּ שׁוֹשַׁנִים כְּשׁוֹשַׁנֶיךָ בַּהֲמוֹן עֲלִיצוֹת נָפֶשׁ,
מִבְּלִי שַׁלֵם מְחִירָן? וְשָלוֹם שַלְוָה וָעֹנֶג
אֲבָל הִסְכַּלְתִּי לִשְׁאוֹל! לְךָ וּלְכָל בֵּיתְךָ,
אוֹיָה! לְדַאֲבוֹן נַפְשִי, מִמֶנִי, מֵאֵת אֲבוֹתַי,
זֶה כְּבַר בַּיָמִים וּמִיַעֲקֹב אָחִינוּ הַיָקָר
לָדַעַת כָּזֹאת נוֹכַחְתִּי! עַד עוֹלָם סֶלָה!
יָדַעְתִּי גַם יָדַעְתִּי

דבר אל הקורא!

בספרי הלזה אשׁלם חוב קדוש לכבוד הוד אדוננו הרם והנשא סיר משׁה מונטיפיורי נ"י, ואקוה כי פרחי שׂרעפי יהיו לך לרצון, קורא נעים! וביושׁר לבבך תִתְנַדֵב לכבדם כערכם למען תבוא עליך ברכת אומללים עצוּמים, אשׁר למו נָדַרְתִי את הַפִדיוֹם (Der Erlös ist für Arme)
