ספר תחכמוני - שַׁעַר אַרְבָּעִים וְשִׁבְעָה

בְּשֶׁבַח כָּל מְדִינָה. זֹאת אוֹמֶרֶת אֲנִי אֶעֱבֹר רִאשׁוֹנָה

נְאֻם הֵימָן הָאֶזְרָחִי: נָסַעְתִּי מִפְּתוֹר. לְאֶרֶץ כַּפְתּוֹר. בְּעֵת נִשְׁמָע קוֹל הַתּוֹר. וּבְעוֹדִי הוֹלֵךְ בַּדֶּרֶךְ. וַאֲנִי חֲזַק מַכְאוֹב רְפֵה בֶרֶךְ. וְחֹם הַיּוֹם. נוֹרָא וְאָיוֹם. יִשְׂרֹף בְּאִשּׁוֹ הַגְּוִיּוֹת. וְיִטְרֹף בְּכֹחוֹ הָאֲרָיוֹת. וְיוֹצִיא מִמְּעוֹנֵיהֶם הַנְּחָשִׁים וְהַחַיּוֹת. עָדַ קָצְרָה רוּחִי. וְיָרַדְתִּי מֵעַל מֶרְכַּבְתִּי. לָתוּר מָקוֹם לְשִׁבְתִּי. וּלְהָנִיחַ אֶת בְּהֶמְתִּי. וָאֶרְאֶה תַּחַת צִלְלֵי הַצְּמָחִים. בֵּין אַחִים. אִישׁ נִלְכַּד בְּחַבְלֵי שְׁנָתוֹ. וְנִקְשַׁר בַּעֲבוֹת תְּנוּמָתוֹ. וְכַאֲשֶׁר קָרַבְתִּי אֵלָיו וְנִצַּבְתִּי עָלָיו הֵקִיץ וְהוּא חָרֵד מִתְּנוּמָתוֹ. וַיִּבְרַח מִלְּפָנַי. וְנָטָה מִפָּנַי. וָאֹמַר לוֹ: שָׁלוֹם לְךָ אַל תִירָא. מָצָאתָ אָח לְצָרָה. וּפָגַשְׁתָּ אִישׁ חֶבְרָה. וְעַתָּה הַגִּידָה לִי מַה לְךָ פֹה. וְלֶכְתְּךָ אֵיפֹה. וְאָנָא עִיר מוֹלַדְתֶּךָ. וּמַה מְּלַאכְתֶּךָ. אָמַר: הוֹרַי זֶרַע אֱמוּנִים. מִבְּנֵי הַקֵּינִים. וּמְעוֹנִי אֵלוֹן בְּצַעֲנַנִּים. וְכַאֲשֶׁר הֲרִיחוֹתִי הֲדַסֵּי אֲמָרָיו. וְטָעַמְתִּי צוּף דְּבָרָיו. הִכַּרְתִּיהוּ. וְיָדַעְתִּי כִּי הוּא חֲבֵרֵנוּ חֶבֶר הַקֵּינִי אֲשֶׁר דְּרַשְתִּיהוּ. וָאֵשֵׁב עִמּוֹ. לִלְקֹט בְּדָלְחֵי נָאֳמוֹ. וָאֹמַר לוֹ: סַפֵּר לִי מִן הַגְּדוֹלוֹת אֲשֶׁר רָאוּ עֵינֶיךָ. וְשָׁמְעוּ אָזְנֶיךָ. וַיֹּאמַר: כֵּן דִּבַּרְתָּ. וַאֲנִי מְזֻמָּן לְכָל אֲשֶׁר שָׁאָלְתָּ. שָׁמַעְנוּ כִּי מִימֵי קֶדֶם אֲשֶׁר מֵעוֹלָם. הִתְפָּאֲרוּ הַמְּדִינוֹת הָעֲדִינוֹת כֻּלָּם. אֲשֶׁר מִמִּצְרַיִם וְעַד עֵילָם. וּפָרְקוּ עֻלָּם. וְהִתְהַלְּלָה כָּל מְדִינָה בְיָפְיָהּ וְהַדְרָתָהּ. וַתְּקַנֵּא כָּל עִיר בַּאֲחוֹתָהּ. וַתַּעַן נֹא אָמוֹן. בְּרֹאשׁ הֶהָמוֹן. וַתֹּאמַר: לִי יָאֲתָה מְלוּכָה. כִּי בְּרִית עוֹלָם שְׁמוּרָה לִי וַעֲרוּכָה. וְהָאֵל קָרָא שְׁמִי נֹא אָמוֹן. כִּי אֲנִי לוֹ לְבַת וָאֶהְיֶה אֶצְלוֹ אָמוֹן. וּבְצַוָּארִי עֲנַק הַיָּם הַנּוֹתֵן לִי עֹז וְעָצְמָה. וּמַלְבִּישׁ צַוָּארִי רַעְמָה. וְיָפְיִי כַּזָּהָב הַטּוֹב לֹא יוּעַם. וּדְרָכַי דַּרְכֵי נֹעַם. וַאֲנִי בֵּית נְתִיבוֹת נִצָּבֶת. וְעַל מְבוֹאוֹת יָם יוֹשָׁבֶת. וּמִי זֹאת מִן הַמְּדִינוֹת אֲשֶׁר תִּגַּע בְּשׁוּלַי. וְתַגִּיעַ אֵלַי. וְעַל רֹאשָׁהּ יְרוּמוּן דְּגָלַי. וִיכַסּוּהָ הֲמוֹן גַּלָּי.

וַתִּשָּׂא מְשָׁלָהּ וַתּאֹמַר:

לוּ תַעֲטֶה כָל-עִיר מְעִיל זֹהַר

וּצְבָא שְׁחָקִים לָהּ כְּמוֹ פַעְמוֹן.

אוֹ נָתְנָה כִסְאָהּ בְּחוּג שַׁחַק

אוֹ בָנְתָה מֵעַל זְבוּל אַרְמוֹן.

כִּסְאִי עֲלֵי רֹאשָׁם וְאִם יִגְאוּ

יָד יִתְּנוּ כֻלָּם לְנֹא אָמוֹן.

וַתַּעַן מִצְרַיִם וַתֹּאמַר: אֲנִי תַּאֲוָה לָעֵינַיִם. נָאוָה כִּירוּשָׁלַיִם. אֲנִי עִיר הוֹמִיָּה. וְלִי תְהִלָּה דוּמִיָּה. וְלִי עֹז וְתוּשִׁיָּה. וְכָל אֲנָשַׁי אֲצִילֵי אָרֶץ. וְסוֹחֲרֵי נִכְבַּדֵּי אָרֶץ. וְאֵלֵי גּוֹיִם יָבוֹאוּ מֵאַפְסֵי אָרֶץ. וּסְבִיבוֹתַי הַיְאוֹר. הַנָּאוֹר. אֲשֶׁר מִנִּפְלְאוֹתָיו זָרַח בַּחשֶׁךְ אוֹר. אֲשֶׁר מִגַּן עֵדֶן מוֹצָאוֹ. וְעַד מִצְרַיִם מוֹבָאוֹ. וּבֹו נִרְאוּ נוֹרָאוֹת. אֲשֶׁר הָיוּ לָאֵל לְעֵד וּלְאוֹת.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

לוּ הָאֲרָצוֹת יַעֲלוּ שָׁמַיִם

לִי נֶחְשְׁבוּ מִדְרָךְ לְכַף רַגְלָיִם.

אוֹ עָלְתָה מַעְלַת יְקָרָם עַד זְבוּל

מַעְלַת יְקָרִי עָלְתָה כִפְלָיִם.

כִּי הַמְּדִינוֹת כַּשְּׁפָחוֹת אוֹ כְמוֹ

בָּנוֹת וְאִמָּם נֶחְשְׁבָה מִצְרָיִם.

וַתַּעַן יְרוּשָׁלַיִם וַתֹּאמַר: בִּהְיוֹתִי בִּימֵי חָרְפִּי. הַשְּכִינָה שָׁכְנָה בְסִפִּי. וְחָמְדוּ אַרְצִי מַלְאֲכֵי גְבוֹהִים. וְהָיִיתִי מַחֲנֵה אֱלֹהִים. וַיְקַנְאוּ בִי כּוֹכְבֵי נְגוֹהִים. כִּי כְבוֹד אֵל בְּהֵיכָלַי. וְהַקְּדֻשָּׁה בְּאָהֳלִי. וְאוֹר עוֹלָם עַל דִּגְלִי. וְהַחֲסִידוּת בַּאֲצִילַי. וְהַנְּבוּאָה בִּזְקֵנַי וְעוֹלָלַי. וְעַתָּה בַּעֲוֹן מְתֵי סוֹדִי. גָּלָה כְבוֹדִי. וְהָעֳרָה יְסוֹדִי. וְהַמָּקוֹם אֲשֶׁר שָׁכְנָה בוֹ הַשְּׁכִינָה. תִּשְׁכֹּן עָלָיו עֲנָנָה. וּבִמְקוֹם הַקְּדֻשּׁה וְהַנְּבוּאָה. נִגְאָלָה מֹרְאָה. וְרוּחַ הַטֻּמְאָה. וּבִנְוֵה הָאוּלַמִּים. אֱלִילִים אִלְּמִים. וְצוּרַת צְלָמִים. וּבְכָל זֹאת יֵשׁ לִי בְּעֵינֵי הָעַמִּים מַעֲלָה וְיִתְרוֹן. וּשְׁמִי עַל כָּל הָאֲרָצוֹת לִפְאֵר וּלְזִכָּרוֹן.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

תִּדְבַּק לְשׁוֹן כָּל-אִישׁ לְמַלְקוֹחַיִם

בִּנְוֵה אֲדֹנָי עִיר יְרוּשָׁלָיִם.

מַחְנֵה אֱלֹהִים הִיא וְצוּר פָּתַח שְׁעָ-

רֶיהָ לְנֹכַח שַׁעֲרֵי שָׁמָיִם.

כִּי הִיא מְאוֹר עוֹלָם לְכָל עַיִן הִיא

חַיֵּי לְבָבוֹת תַּאֲוַת עֵינָיִם.

וַתַּעַן עַכּוֹ וַתֹּאמַר: אָנכֹיִ הָאֶבֶן הָרֹאשָׁה. וְאַרְצִי אֶרֶץ קְדוֹשָׁה. כִּי אֲנִי חֲבַצֶּלֶת הַשָּׁרוֹן. וּסְבִיבוֹתַי הַכַּרְמֶל וְהַלְּבָנוֹן. וְתָבוֹר וְחֶרְמוֹן.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

אֲנִי עַכּוֹ אֲנִי הוֹד כָּל-מְדִינָה

כְּלִילַת חֵן וְאֵין קֵץ לַתְּכוּנָה.

וְהַכַּרְמֶל סְבִיבִי עִם לְבָנוֹן

וְשָׁם נִגְלָה כְּבוֹד שׁוֹכֵן מְעוֹנָה.

וְאַרְצִי מִיְּמוֹת עוֹלָם קְדוֹשָׁה

וְאָנֹכִי שְׁכֵנָה לַשְּׁכִינָה.

וַתַּעַן דַּמֶּשֶׂק וַתֹּאמַר: לִי יָאֲתָה הַמְּלוּכָה. וּמֲעֲלָתי מִקַּדְמֵי אֶרֶץ נְסוּכָה. יָצָא שְׁמִי בְּכָל הַעוֹלָם. עַד נִשְׁכְּחוּ נֶגְדִּי שִׁנְעָר וְעֵילָם. לְנֹעַם חֲמוּדוֹתַי. וְרֹב עֲרוּגוֹתַי. וְרֹחַב מְסִלּוֹתַי. וְהַדְרַת חוּצוֹתַי. הַשֶּׁמֶשׁ צְנִיף מְרַאֲשׁוֹתַי. וְהַכּוֹכָבִים יִשְּׁקוּ מַרְגְּלוֹתַי. וְכָל דְּבִירַי עֲלִיּוֹת וְאֶשְׁנַבִּים. וְכָל דְּרָכַי גֻּלּוֹת וּמַשְׁאַבִּים. וּבְתוֹכִי כָּל תַּעֲנוּג יִנְעַם וְיִשְׁפַּר. וּסְבִיבוֹתַי אֲמָנָה וּפַרְפַּר. וְכָל גִּבְעוֹתַי וְהָרַי גַּנּוֹת וּפַרְדֵּסִים. כַּמַּיִם לַיָּם מְכַסִּים. וּמֵהֶם יִטְּפוּ הֶהָרִים עֲסִיסִים.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

לִי יָאֲתָה מַלְכוּת לְמַעַן בִּי

שָׂם הַיְּקָר כִּסֵּא תְהִלָּתוֹ.

הֵן כָּל-כְּבוֹד עוֹלָם וְכָל-הוֹדוֹ

תִּשְׁפַּל לְמַעְלָתִי גְדֻלָּתוֹ.

כִּי בוֹרְאִי הִשְׁכִּין כְּבוֹדוֹ תוֹךְ

חַדְרָךְ וְדַמֶּשֶׂק מְנוּחָתוֹ.

וַתַּעַן צוֹבָה וַתֹּאמַר: אֲנִי נֵר מַעֲרָכָה. וְעִיר מַמְלָכָה. וּמַעֲלָתִי מִדּוֹר לְדוֹר נִמְשָׁכָה. וְעַל עַמּוּד הַמְּלוּכָה נִסְמָכָה. וְאַרְצִי לְבָנָה כַלְּבָנָה. וּבְמִינֵי מַעֲדַנִּים דְּשֵׁנָה וְרַעֲנָנָה. וַאֲנִי כְּלִילַת יֹפִי, בִּלי דֹפִי. אִם תֵּרֵד עֲמָקִים אוֹ תֵלֵךְ שֶׁפִי. וּבִי מִמְּתֵי הַחָכְמָה אֲצִילִים וְרָמִים. וְהֵמָּה חֲכָמִים מְחֻכָּמִים. מֵהֶם רוֹפְאֵי רְפָאִים. וּמֵהֶם מַחְכִּימֵי פְתָאיִם. וּמֵהֶם סוֹפְרִים מַעַרְכוֹת פָּז יַעֲרֹכוּ. וּבְשֵׁבֶט סוֹפֵר יִמְשֹׁכוּ.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

עֵת נִזְכְּרוּ אַרְצוֹת זְמַן יַחַד

הֵן נוֹתְנוֹת יָד לַאֲרַם צוֹבָה.

כִּסְאִי בְּרֹאשׁ כֻּלָּם וּמוּל שִׂיאִי

שָׁפְלוּ וְגַם שָׁכְבוּ לְמַעְצֵבָה.

וּבְעֵת אֲלֻמּוֹת אִלְּמוּ כֻלָּם

קָמָה אֲלֻמָּתִי וְנִצָּבָה.

וַתַּעַן אַשּׁוּר וַתֹּאמַר: בְּאָשְׁרִי כִּי אִשְּׁרוּנִי בָנוֹת. וְשִׁבְּחוּנִי עֲדִינוֹת. וְקִנְאוּ בִי הַמְּדִינוֹת. אֲנִי אַשּׁוּר מְאֻשֶּׁרֶת בְּמַעֲלָתִי. וְכִשְׁמִי כֵּן תְּהִלָּתִי. יִמְעַט כָּל מַהֲלָל לַעֲרֹךְ גְּדֻלָּתִי. כִּי אִני אוֹסֶפֶת הַחֲמוּדוֹת. וְכוֹלֶלֶת הַמִּדּוֹת הַנִּכְבָּדוֹת.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

עֲלֵי כָל-עִיר אֱלֹהִים כִּבְּדַנִי

וְשָׂם לִי כֵס יָקָר וַיְנַטְּלֵנִי.

וְכָל-רוֹאַי בְּחשֶׁך יֶחֱזוּ אוֹר

וְאַמְתִּיק פִּי אֱנוֹשׁ עֵת יִזְכְּרֵנִי.

אֲנִי אַשּׁוּר מְאֻשֶּׁרֶת בְּבָנוֹת

וְאֹזֶן שָׁמְעָה וַתְּאַשְּׁרֵנִי.

וַתַּעַן בָּבֶל וַתֹּאמַר: אֲנִי עֲדִינָה. פְּאֵר כָּל מְדִינָה. וְנֶגֶד הוֹדִי וְגָדְלִי. הֵן גּוֹיִם כְּמַר מִדְּלִי. וַאֲנִי אָחוֹת לְעִיר הֹּקֶדׁש. כִּי מִיּוֹם גָּלוּ עַם הַקֹּדֶשׁ. מֵעַל אַדְמַת הַקֹּדֶשׁ. אֲנִי אֵלַי אֲסַפְתִּים. וּבְצִלִּי הִסְתַּרְתִּים. וּמֵאָז וְעַד עַתָּה לֹא מְאַסְתִּים. וְלֹא גְעַלְתִּים. וּמִמֶּנִּי יָצְאוּ גְּאוֹנֵי הַתּוּשִׁיָּה. וְחַכְמֵי פְלִילִיָּה.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

אֲנִי כֶתֶר זְמָן. אֶרֶץ עֲדִינָה

וְאַדְמָתִי פְאֵר עַל כָּל-מְדִינָה.

הֲכִי כָל-הָאֳרָצוֹת כַּאֲבָנִים

בְּמַרְגֵּמָה. וְאָנֹכִי פְנִינָה.

וְחַכְמֵי הָאֱלֹהִים שָכְנוּ בִי

וְשָׁכְנָה תוֹךְ לְבָבָם הַשְּׁכִינָה.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר כִלָּה מְשָׁלָיו. וְנֹעַם מִלָּיו. אָמַרְתִּי לוֹ: מַה יָּקְרוּ לִי מְלִיצוֹתֶיךָ. וּמַה מָּתְקוּ חִידוֹתֶיךָ. וּמֶה עָרְבוּ מִשְׁלֵי תְעוּדוֹתֶיךָ. הָאֵל יֵיטִיב גְּמוּלֶיךָ. וְיִפְרֹשׂ סֻכַּת שְׁלוֹמוֹ עָלֶיךָ. וְקָרָא בִשְׁלוֹמִי. וְהָלַךְ מֵעִמִּי.
