ספר תחכמוני - שַׁעַר אַרְבָּעִים וּשְׁנַיִם

מְרִיבַת הַכִּילוּת וְהַנְּדִיבוּת. מִי יוֹתֵר נִכְבָּד בֵּין אַנְשֵׁי הַחֲשִׁיבוּת


נְאֻם הֵימָן הָאֶזְרָחִי: וַיְהִי הַיּוֹם וַאֲנִי יוֹשֵׁב בְּחֶבְרַת מַשְׂכִּילִים. וְעִם נְדִיבִים אֲצִילִים. יָפִיצוּ מֵעֲנַק שִׂכְלָם טוּרֵי לֶשֶׁם. וּמֵעֲנַן לִקְחָם נוֹזְלֵי גֶשֶׁם. נְשִׂיאֵי עֵדָה קְרִיאֵי מוֹעֵד אַנְשֵׁי שֵׁם. וּבְעוֹדֵנוּ מְדַבְּרִים בְּעִנְיְנֵי אֲמָרִים. מְעֻלָּפִים סַפִּירִים. יְקָרִים מִפְּנִינִים וְדָרִים. עָבַר עָלֵינוּ אִישׁ מִן הַשָּׂרִים הַנִּכְבָּדִים. וּמִגּוֹמְלֵי חֲסָדִים. וְכַאֲשֶׁר שָׁמַע מְלִיצוֹת דְּבָרֵינוּ. נִכְסַף לְמֶתֶק מַאֲמָרֵינוּ. וַיֵּשֶׁב עִמָּנוּ. וְהֵחֵל לְדַבֵּר אִתָּנוּ. בְּעִנְיְנֵי שֵׂכֶל וּבִינָה. וּלְגַלּוֹת צְפוּנֵי הַתְּבוּנָה. וַיְגַלֶּה לָנוּ מִשִּׁירָיו פְּלָאוֹת. וְהֶרְאָה נוֹרָאוֹת. לֹא שָׁמְעוּ הַשּׁוֹמְעִים כָּהֵם. וְלֹא הִגִּיעוּ אֲלֵיהֶם. וְאַחֲרֵי אֲשֶׁר הֶאֱרִיךְ לָשֶׁבֶת. וּלְהַרְבּוֹת מַחֲשֶׁבֶת. קָם וְנָסַע מֵאִתָּנוּ. אַחֲרֵי אֲשֶׁר נָתַן אֲרוּחָה וּמַשְׂאֵת לְכֻלָּנוּ. וְקָרָא עָלֵינוּ שְׁלוֹמוֹ. וְשָׁב לִמְקוֹמוֹ. וְכַאֲשֶׁר הָלַךְ מִעִמָּנוּ. תָּמְהוּ אֲנְשֵׁי הַחֲבוּרָה עַל הַכָּבוֹד אֲשֶׁר עָשָׂה לָנוּ. וְאָמְרוּ: רְאוּ נָא הָאִישׁ הַלָּזֶה. הַרְאִיתֶם כָּמוֹהוּ בְּהָקִיץ אוֹ בַמַּחֲזֶה. רְאוּ אֵיךְ מָשַׁךְ לִבּוֹתֵינוּ בְטוֹבוֹתָיו. וְכָבַשׁ נַפְשׁוֹתֵינוּ בַּעֲבוּר נִדְבוֹתָיו. אָמַר אִישׁ מֵהֶם: בָּזֹאת תֵּדְעוּ כִּי אֵין בְּכָל הַמִּדּוֹת כְּמוֹ הַנְּדָבָה. וְכָל חֲפָצִים לֹא יִשְׁווּ בָהּ. וְכָל הַמִּדּוֹת לְרַגְלָהּ תִּכְרַעְנָה. רַבּוֹת בָּנוֹת עָשׂוּ חַיִל וְהִיא עָלְתָה עַל כֻּלָּנָה. וְהָיָה בְתוֹכָם אִישׁ. הָרֵי חָכְמָה יַרְעִישׁ. וּכְצִפְעוֹנִי יַפְרִישׁ. וַיֹּאמֶר לָהֶם: אַתֶּם אֲמַרְתֶּם כִּי הַנְּדָבָה עַל כָּל הַמִּדּוֹת הִיא עֶלְיוֹנָה. וְכִי הִיא עוֹלָה בָרִאשׁוֹנָהַ. וַאֲנִי אִם אֶחְפֹּץ אֲשִׁיבֶנָּה אַחֲרוֹנָה. וְאוֹרִידֶנָּה עַד הַמַּעֲלָה הַתַּחְתּוֹנָה. אָמְרוּ לוֹ כָל הָעָם: בֶּאֱמֶת תִּהְיֶה בֶּן חַיִל. אִם תּוּכַל לַעֲשׂוֹת מִן הַיּוֹם לַיִל. וּמִן הַכְּפִיר אַיִל. וַיִּגַּשׁ בָּחוּר מִמַּשְׂכִּילֵיהֶם. וּמֵאֲצִילֵיהֶם. וֶיֹּאמֶר לוֹ: הִנֵּה אָנֹכִי בָא אֵלֶיךָ. לַחְקֹר מִלֶּיךָ. וַאֲנִי אֲחַזֵּק טַעֲנַת הַנְּדָבָה וַאֲשַׁבְּחֶנָּה. וְעַל כָּל מִדָּה אַגְדִּילֶנָּה. וְאַתָּה אִם תּוּכַל תַּשְׁפִּילֶנָּה. וּמִמַּצָּבָהּ תַּפִּילֶנָּה. אָמַר הֶזָּקֵן: אֲנִי אְזִני קַשֶּׁבֶת לְמַאֲמָרֶיךָ. וְעַתָּה הַשְׁמִיעֵנִי דְבָרֶיךָ.

וַיָּחֶל הַבָּחוּר וַיֹּאמַר: אֲלֵיכֶם אִישִׁים אֶקְרָא. אַנְשֵׁי עֵצָה וּגְבוּרָה. הֲלֹא יְדַעְתֶּם. אִם לֹא שְׁמַעְתֶּם. מִיּוֹם נוֹלַדְתֶּם. כִּי הַנְּדָבָה הִיא בְכָל הַמִּדּוֹת גְּבֶרֶת. וּלְרֹאשָׁם עֲטֶרֶת. וְכִי הִיא פְּאֵר הַמְּלָכִים. וְנֵזֶר הַנְּסִיכִים. וְנוֹתֶנֶת רוּחַ לַנְּדִיבִים. וְתָשִׁיב יְדֵיהֶם כְּעָבִים. וּפְנֵיהֶם כְּכוֹכָבִים. וְלוּלֵי הַנְּדָבָה גָבְרָה יַד הַזֻּלּוּת. וְשָׁכְנוּ הֶנִּכְבָּדִים בַּשִּׁפְלוּת. וְרָמָה יַד אַנְשֵׁי הַכִּילוּת. וְהָיָה הַיּשֶׁר לָמַס. וְתִמָּלֵא הָאָרֶץ חָמָס. וְשָׁבוּ הַנִּכְבָּדִים. עֲבָדִים. וְהָאֲצִילִים. נְבָלִים. וְהָיוּ הַחֲזָקִים בְּלִי אֱיָלוּת. וְשָׁכְנוּ הָרָמִים בְּבוֹר מַעֲצֵבָה וְדַלּוּת. וּמַה יַמְרִיץ בִּנְדָבָה מְחוֹקֵק. וְהִיא כְגֶפֶן שׂוֹרֵק וְזוּלָתָהּ גֶּפֶן בּוֹקֵק.

וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר:

אֱמֶת כִּי הַנְּדָבָה הִיא גְבִירָה

לְכָל מִדָּה וְזוּלָתָהּ צְעִירָה.

וְלוּלֵא הִיא יְקָר תֵּבֵל יְהִי סָף

וּבַעְדָּה כָּל יְקָר נוֹצָר וְנִבְרָא.

וּבָהּ יִכְבּשׁ אֱנוֹשׁ זַעַף מְלָכִים

וְיַשְׁקִיט יָם חֲמָתָם בַּתְּשׁוּרָה.

וְהַמִּדּוֹת לְעֻמָּתָהּ שְׁפָלוֹת

לְמַעַן לָהּ זְרוֹעַ עִם גְּבוּרָה.

וְכָל מַעְלָה לְמַעְלָתָהּ חֲסֵרָה

וְכָל לָשׁוֹן בְּמַהְלָלָה קְצֵרָה.

וַיֹּאמֶר הַזָּקֵן: הֲרֹב דְּבָרִים לֹא יֵעָנֶה. וְאִם לַהֲתֻלֶיךָ לֹא תִשְׁמַע מַעֲנֶה. וּמַה מַּעֲלַת הַנְּדָבָה. וְהִיא תַשְׁכִּיב בְּעָלֶיהָ לְמַעֲצֵבָה. וּתְרוֹשֵׁשׁ בֵּיתוֹ מִכָּל טוֹבָה. הוֹנוֹ מְפַזֶּרֶת. וּגְאוֹנוֹ מְשַׁבֶּרֶת. וְאוֹתוֹ בְּחַיָּיו מְקַבֶּרֶת. וְלִבּוֹ בְּיֵין יְגוֹנוֹ מְשַׁכֶּרֶת. תַּשְׁפִּיל גְּדוֹלֶיהָ. וְתַפִּיל דְּגוּלֶיהָ. וְתַכְנִיעַ אֲצִילֶיהָ. לִשְׁפָלֶיהָ. וּנְדִיבֶיהָ לִנְבָלֶיהָ. וְתַעֲזֹב הַיָּדַיֵם רֵקוֹת. בּוּקוֹת וּמְבוּקוֹת. וְכָל קוֹפֵץ יְמִינוֹ יְמַלֵּא הוֹן אוֹצְרוֹתָיו. וְיִרְבּוּ טוֹבוֹתָיו. וְיִגְדַּל עָשְׁרוֹ. וְיָאִיר אוֹרוֹ. וְתוֹסִיף טוֹבָתוֹ. וְתִכָּפֵל בִּרְכָתוֹ. וְאִלּוּ הַכִּילַי הָיָה נָדִיב. אָז הַיָּגוֹן לִבּוֹ הֶאֱדִיב. וְנוֹתַר אִישׁ רָשׁ וְנִקְלֶה. וְנִדָּח וְגוֹלֶה. וּבִקְפִיצַת יָד. יְהִי הָאָדָם נִכְבָּד. וְיִמָּלְאוּ הוֹן כֵּלָיו. וּבְרָכָה אֹהָלָיו. וְנֵרוֹ לֹא יִכְבֶּה. וְעָלָיו אָמַר בְּחִיר ה': וְקֹבֵץ עַל-יָד יַרְבֶּה.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֲשֶׁר יִקְפֹּץ יְמִינוֹ יֶאֱסֹף הוֹן

וְיִהְיֶה לוֹ בְּיוֹם צָרָה לְעֶזְרָה.

וְיִמָּלְאוּ עשֶׁר אוֹצְרוֹתָיו

וְטוֹבָתוֹ עֲרוּכָה גַּם שְׁמוּרָה.

וְהַנָּדִיב יְרוֹשֵׁשׁ בֵּית נְכֹתוֹ

וְלֹא יַשְׁאִיר בְּיָדוֹ עַד אֲגוֹרָה.

וְהַכִּילַי בְּרָכָה בַאֲסָמָיו

וְהַנָּדִיב בְּאוֹצְרוֹתָיו מְאֵרָה.

וְזֶה בָתָּיו בְּרֹב עָשְׁרוֹ מְלֵאִים

וְזֶה יָדוֹ לְכָל טוֹבָה חֲסֵרָה.

וַיַּעַן הַבָּחוּר וַיֹּאמַר: הַחֵרְשִׁים שִׁמְעוּ נִפְלָאוֹת. וְהַעִוְרִים הבֶיִּטוּ לִרְאוֹת. הַרְאִיתֶם עֶבֶד יִתְפָּאֵר עֶל אֲדוֹנָיו וְיִלְעָג. וְאִם שׁוּעָל עַל הַכְּפִיר יִשְׁאָג. דִּמִּיתָ בְּאֹרֶךְ לְשׁוֹנֶךָ. וְרוֹב הֶגְיוֹנֶיךָ. לְהָשִׁיב הַיֹּפִי. דֹּפִי. וְהַחֶסֶד. חֶסֶד. וְהַגְּדֻלָּה. נְבָלָה. וְהָאֱיָלוּת. שִׁפְלוּת. וְהַנְּדָבָה. דִּבָּה. תִּתְפָּאֵר עָלַי בְּנַבְלוּת הַנְּבָלִים. וְתִתְגָּאֶה בְּמִדַּת הַשְּׁפָלִים. הֲלֹא יָדַעְתָּ כִּי בַעֲבוּר מַתַּן אָדָם לֹא תֶחְסַר בִּרְכָתוֹ. וּבִגְלַל נִדְבָתוֹ. לֹא תֹאבַד טוֹבָתוֹ. כְּמוֹ מַעְיַן הַמַּיִם אֲשֶׁר אִם יִשְׁתּוּ מִמֶּנּוּ יִרְבּוּ נוֹזְלָיו. וְיִגְבְּרוּ נְחָלָיו. וְלֹא יִיבְשׁוּ זְרָמָיו. וְלֹא יְכַזְּבוּ מֵימָיו. וְאִם לֹא יִקְחוּ מִמֶּנּוּ יַחְסְרוּ לִפְעָמִים מֵימֵיהֶם. וְיֶחֶרְבוּ אַגְמֵיהֶם. וְכֵן בִּרְבוֹת הַמָתן וְהַנְּדָבָה. תִּרְבֶּה הַטּוֹבָה. וּבִקְפִיצַת הַיָּד. יִכְלֶה הַהוֹן וְיֹאבַד. וְאִם תֹּאמַר כִּי יֵשׁ נְבָלִים אֲשֶׁר יִצְבְּרוּ עשֶׁר. כָּל עָשְׁרָם בְּלִי כשֶׁר. וְסוֹפוֹ לָעוּף כַּנֶּשֶׁר. וְאִם יֵשׁ נְדִיבִים אֲשֶׁר אָבַד הוֹנָם. וְנֶחְסָר מָמוֹנָם. לֹא הִתְרוֹשְׁשוּ בַּעֲבוּר נִדְבָתָם. אֲבָל הָאֵל רָצָה לְנַסּוֹתָם.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

נְדָבָה הִיא לְרֹאשׁ נָדִיב עֲטָרָה

וְהַכִּילַי וּבִרְכָתוֹ אֲרוּרָה.

וְטוֹבַת כָּל אֱנוֹשׁ כִּילַי כְּאַיִן

וְהִיא עָלָיו וְעַל כָּל אִישׁ אֲסוּרָה.

וְהוֹנוֹ רַב וְהוּא יָמוּת בְּרָעָב

וְיָדוֹ מֵעֲשׂוֹת לוֹ טוֹב עֲצוּרָה.

כְּאִישׁ עוֹמֵד עֲלֵי נַחַל עֲדָנִים

וְהוּא יִרְוֶה בְּמֵי בִצּוֹת וּמִשְׁרָה.

וְטוֹבָתוֹ לְפִקָּדוֹן בְּיָדוֹ

וְהִיא עִמּוֹ לְאִישׁ נָכְרִי שְׁמוּרָה.

וְהַנָּדִיב בְּעֵין כֹּל אִישׁ מְכֻבָּד

וְהַכִּילַי זְעוּם הָאֵל בְּעֶבְרָה.

וַיַּעַן הַזָּקֵן וַיֹּאמַר: הֲיֵשׁ בְּפִיךָ מַעֲנֶה. וְאֵל מִי מִקְּדוֹשִׁים תִּפְנֶה. הֲלֹא הִנְּךָ רוֹאֶה כִּי בְכָל אַרְצוֹת הָעוֹלָם. אֲשֶׁר מִמִּצְרַיִם עַד עֵילָם. לֹא תִמְצָא נָדִיב בְּלִי סֶלֶף. כִּי אִם אֶחָד מֵאֶלֶף. וְהַנִּשְׁאָרִים הֵם קוֹפְצֵי יָד כֻּלָּם. מִקְּטַנָּם וְעַד גְּדוֹלָם. וּבַדָּבָר הַזֶּה הֶעֱשִׁירוּ בְנֵי עֵילָם. וְזֶה לְךָ הָאוֹת כִּי הַנְּדָבָה הִיא מִדָּה רָעָה וּגְרוּעָה. כִּי כָל בְּנֵי אִישׁ יִמְאָסוּהָ. וּבְכַף רַגְלָם יִרְמְסוּהָ. וְאִלּוּ הָיְתָה מִדָּה טוֹבָה לְמָדוּהָ גְדוֹלִים וּקְטַנִּים. וְהֶחֱזִיקוּ בָהּ שְׁפָלִים וּגְאוֹנִים. וּמֵאַחַר אֲשֶׁר יִמְאָסוּהָ בְּנֵי הָעוֹלָם כֻּלָּם. בְּזֹאת תֵּדַע כִּי הִיא מְאוּסָה. וּבְכָל מוּם עֲמוּסָה. וְכִמְעַט אֹמַר כִּי הִיא אֲסוּרָה. וְכָל הַדָּבֵק בָּהּ כְּאִלּוּ דָבַק בַּעֲבוֹדָה זָרָה. וּמִי שֶׁאוֹהֵב אוֹתָהּ הוּא נֶחְשָׁב בְּעֵינֵי בְּנֵי אָדָם מְהוֹלָל וּמְשֻׁגָּע. וּבְנֶגַע סִכְלוּת מְנֻגָּע.

אָמַר הַבָּחוּר: הֲתַחְשֹׁב כִּי בַעֲבוּר נַבְלוּתָהּ. יִמְאֲסוּ בְּנֵי אִישׁ אוֹתָהּ. וְיִרְחֲקוּ מֵחֶבְרָתָהּ. לֹא כֵן הַדָּבָר רַק מִפְּנֵי גְדֻלָּתָהּ. יִלְאוּ לְהַשִּׂיג אוֹתָהּ. וְתִקְטַן נַפְשָׁם לַעֲלוֹת לְמַעֲלָתָהּ. וְּאֶתֵּן לְךָ מוֹפֵת עַל דְּבָרַי כִּי הַדָּבָר הַנִּכְבָּד אֵינוֹ נִמְצָא בְּיַד כָּל אָדָם רַק בִּידֵי הַיְחִידִים. הַנִּכְבָּדִים. כְּמוֹ הָאֲבָנִים הַיְקָרוֹת אֲשֶׁר אֵינָן נִמְצָאוֹת בְּיַד כָּל אָדָם רַק בִּידֵי הַשָּׂרִים וְהַסְּגָנִים. וְהַמְּלָכִים וְהַגְּאוֹנִים. וּכְמוֹ הַחָכְמָה. שֶׁאֵינָהּ מְצוּיָה בַחַיָּה וּבַבְּהֵמָה. וְלֹא בְכָל עַמֵּי הָאֲדָמָה. רַק בַּמַּשְׂכִּילִים הַגְּבוֹהִים. אֲשֶׁר בָּם רוּחַ אֱלֹהִים. וּכְמוֹ הַחֲסִידוּת הַקְּדוֹשָׁה. אֲשֶׁר לֹא נִתְּנָה לְכָל יְלוּד אִשָּׁה. רַק לְיִשְׁרֵי לֵב הַקְּדוֹשִׁים. אֲשֶׁר הֵם לַעֲבֹד הָאֵל נְגָּשִׁים וּמִתְקַדְּשִׁים. וּשְׁאָר הַמִּדּוֹת הַטּוֹבוֹת כְּמוֹ הַעֲנָוָה וְהָאֱמוּנָה. וְכָל מִדָּה הֲגוּנָה. שֶׁאֵינָם מְצוּאוֹת אֵצֶל הָרֵיקִים וְהַסְּכָלִים. וְלָא בַבְּזוּיִים הַשְּׁפָלִים. רַק בְּמִי שֶׁהוּא בַּעַל נֶפֶשׁ טְהוֹרָה. וְרוּחַ יְקָרָה וּבָרָה. כֵּן הִיא הַנְּדָבָה אֵינָהּ מְצּויָה בִידֵי בְנֵי בְלִיַּעַל. אֲשֶׁר כִּחֲשׁוּ לָאֵל מִמַּעַל. וְהַבִּרְכַּיִם אֲשֶׁר כָּרְעּו לַבַּעַל. רַק לְאַנְשֵׁי הַיּשֶׁר וְהַנְּכוֹחָה. אֶחָד מֵעִיר וּשְׁנַיִם מִמִּשְׁפָּחָה. וְהַיּוֹדְעִים עֶרְכָּהּ. הֵם יֵלְכוּ בְדַרְכָּה. וְהַמַּכִּירִים מַעֲלָתָהּ. הֵם יִתְפָּאֲרוּ בְחֶבְרָתָהּ. כִּי לֹא נִבְרְאָה לַנְּבָלִים הַחֲנֵפִים רַק לִקְדוֹשֵׁי עֶלְיוֹן. אֲשֶׁר בָּנוּ בְלִבָּם לְיִרְאַת הָאֵל אַפִּרְיוֹן.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

לוּ הַנְּדָבָה דִבְּרָה וַתַּעֲנֶה

אָז גִּלְּתָה סוֹדָהּ לְאַנְשֵׁי מַעֲנֶה.

לָכֵן מְכֹר בִּלְתָּהּ וְאָז תִּהְיֶה לְרֹאשׁ

וּבְכָל סְגֻלַּת הוֹנְךָ אוֹתָהּ קְנֵה.

אָמַר הַמַּגִּיד: וַכַאֲשֶׁר הִרְבּוּ דְבָרִים. לַעֲרֹךְ מִלְחֶמֶת אֲמָרִים. יָדְעוּ כָל הָעָם כִּי דִבְרֵי הַבָּחוּר חֲזָקִים כְּמִגְדָּל בָּנוּי לְתַלְפִּיּוֹת. וְכִי טַעֲנוֹת הַזָּקֵן דְּחוּיּוֹת. וְאָמְרוּ לַזָּקֵן: אִם אַתָּה נִמְרָץ בְּמִלֶּיךָ וּמְשָׁלֶיךָ. הֲלֹא הַבָּחוּר גָּבַר עָלֶיךָ. אָמַר הַזָּקֵן: שֶׁקֶר חֳשַׁבְתֶּם. וְתָפֵל טַחְתֶּם. וּבַשָּׁוְא בְּטַחְתֶּם. כִּי בִלְשׁוֹנִי יָכֹלְתִּי לִשְׁבֹּר מַלְתְּעוֹת טַעֲנוֹתָיו. וְלַהֲרֹס מִגְדְּלֵי תְּשׁוּבוֹתָיו. רַק לָא נָכוֹן לְהַגְבִּיר הַנְּבָלָה עַל הַנְּדִיבוּת הַנְּעִימָה. פֶּן יֹאמְרוּ הַנְבָלִים יָדֵנוּ רָמָה. וִיהִי לָהֶם פִּתְחוֹן פֶּה עַל הַנְּדִיבִים. וְלֹא יִשָּׁאֵר זִכָּרוֹן לַמַּעֲשִׂים הַטּוֹבִים. אָמְרוּ לוֹ: הֵיטַבְתָּ בְשִׂכְלְךָ וְטַעְמֶךָ. וְעַתָּה הַגִּידָה נָא שְׁמֶךָ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בְּאֵלוֹן צַעֲנַנִּים מְגוּרַי

וְשָׁמָּה נוֹדְעָה אַרְצִי וְעִירִי.

אֲנִי חֶבֶר וְכִשְׁמִי מַעֲשַׂי כִּי

אֲחַבֵּר נִפְזְרֵי מוּסָר בְּשִׁירִי.
