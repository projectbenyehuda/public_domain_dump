ספר תחכמוני - שַׁעַר שְׁלֹשִׁים וּשְׁמוֹנָה

בִּסְפִינָה מִסְתָּעֶרֶת בַּיָּם. מִשְּאוֹן גַּלָּיו וְדָכְיָם


נְאֻם הֵימָן הָאֶזְרָחִי: נְשָׂאַנִי עַל אֶבְרָתוֹ הַנְּדוֹד. וְהִבְעִיר בְּלִבִּי כִּידוֹד. וְהִרְחִיקַנִי מִכָּל רֵעַ וָדוֹד. עַד שָׁכְנָה נַפְשִׁי דוּמַה וְדוּמִיָּה. וְכַאֲשֶׁר אָרְכוּ לִי הַיָּמִים בְּאֶרֶץ נָכְרִיָּה. נִכְסְפָה נַפְשִׁי. לָשׁוּב לְנָוֶה חָפְשִׁי. וַיְהִי עִם לְבָבִי. לַחֲזֹר לְאֶרֶץ מוֹשָׁבִי. וְגָדַל חֶפְצִי. לִרְאוֹת אַרְצִי. וּנְוֵה רִבְצִי. אֲבָל הָיָה בֵינִי וּבֵין עִירִי כַּמָּה חֳדָשִים וְיָמִים. וְכַמָּה גְבָעוֹת וּתְלָמִים. וְכַמָּה אֲגַמִּים וְיַמִּים. וָאִירָא לָבוֹא בַיָּם. מִפְּנֵי שְׁאוֹן גַּלָּיו וְדָכְיָם. אַחֵרי כֵן שַׂמְתִּי נְפִׁשי בְּכַפִּי. וּבָטַחְתִּי עַל הָאֵל נִשְמָתוֹ בְאַפִּי. וּתְהִלָּתוֹ בְתוֹךְ פִּי. וְשָׂכַרְתִּי לִי אֳנִיָּה יְפֵה-פִיָּה נָאוָה שְׁחַרְחֹרֶת. כְּבָרָק מְמַהֶרֶת. כּוּשִׁית מִבְּנוֹת הַמַּיִם. תַאֲוָה הִיא לָעֵינַיִם. לָהּ מִן הַקְּלָעִים כְּנָפַיִם. וּמִן הַמְּשׁוֹטִים רַגְלַיִם. וְהִיא הָרָה וְלֹא יָשְבָה עַל אָבְנָיִם. וַתִּשָּׂאֵנִי רוּחַ בֵּין הָאָרֶץ וּבֵין הַשָּׁמָיִם. וְרַצְנוּ בְשָלוֹם אַךְ יוֹם אוֹ יוֹמַיִם. וּבְעוֹד הַיָּם כַּשֶּׁמֶן הַטּוֹב. לַח וְרָטֹב. וּמֵימָיו לְכֶל נֶפֶשׁ נְכֵאָה. נַהֲרֵי נַחֲלֵי דְבַש וְחֶמְאָה. וְהָרוּחַ חוֹלָה וְדַלָּה. בַּלָּאט נוֹשֶׁבֶת. כְּאִלּוּ הִיא מִתְגַּנֶּבֶת. וְהַנְּפָשׁוֹת מְשׁוֹבֶבֶת. וְהַמַּיִם כְּדִמְעַת חוֹשֵׁק זַכִּים וְצַחִים. וְהַגַּלִּים שׁוֹקְטִים וְנָחִים. וְכֻלָּנוּ שְׂמֵחִים. עַד רָעֲמוּ לוֹ פְנֵי הַשָּׁמַיִם. וְנֶהֶפְכוּ לָנוּ לְאוֹיְבִים גַּלֵּי הַמַּיִם. וּבָאָה בֵינֵיהֶם הָרוּחַ כְּאִישׁ הַבֵּינַיִם. לְהַעֲלוֹת הַסְּפִינָה עַד לֵב הַשָּׁמַיִם. וּלְהוֹרִידָהּ לְמַעֳמַקֵּי מַיִם. וְכָל הָעָם צוֹעֲקִים בְּיָדַיִם עַל חֲלָצַיִם. וְעַפְעַפֵּיהֶם יִזְּלוּ מַיִם. וּשְׁלוּחֵי הָרוּחַ. מִהֲרוּ כְצִיר שָׁלוּחַ. וְהָפְכוּ הַיָּם כְּסִיר נָפוּחַ. וְהַסְּפִינָה. בְּנִבְכֵי הַמַּיִם סְפוּנָה. וּלְשׁוֹן הַיָּם יְכַחֵשׁ בָּהּ. וְהִיא אִתּוֹ גְנוּבָה. וְלִפְעָמִים תִּמָּצֵא בְיָדוֹ הַגְּנֵבָה. וַתָּנַח הַתֵּבָה. וְלִפְעָמִים יַסְתִּירוּהָ הַגַּלִּים. וְיַהַפְכוּהָ כְגַלְגַּלִּים. וְהַמַּיִם סְבִיבָהּ יוֹרְדִים וְעוֹלִים. וְהַמְּשׁוֹטִים. נִשְׁמָטִים. וְעַל פְּנֵי הַיָּם שָׁטִים. וְהַקְּלָעִים. נִקְרָעִים. וְהַתֹּרֶן.וְהָאֹרֶן. כַּמּוֹץ יְסֹעַר מִגֹּרֶן. וְכָל אִישׁ מִתְגּוֹלֵל בְּצִירוֹ. וּמִתְהוֹלֵל בְּשִׁבְרוֹ. כְּאִישׁ יַיִן עֲבָרוֹ. וְהָיָה בַסְּפִינָה אִישׁ. רוֹאֶה סַעַר הַיָּם וּמַחֲרִישׁ. וְכָל אֶחָד מִמֶּנּוּ בּוֹכֶה וְצוֹרֵחַ. וְהוּא שׁוֹחֵק וְשָׂמֵחַ. וְכָל אִישׁ מֵאַנְשֵׁי הַסְּפִינָה מִתְיַפֵּחַ וּמִשְׁתַּטֵּחַ. וְצוֹעֵק וְצֹוֵחַ. וְשָׁבָה הָאֳנִיָּה כְּבֵית מַרְזֵחַ. וְהוּא דוֹמֵם וּמַחֲרִישׁ. כְּצוּר הַחַלָּמִישׁ. מִמְּקוֹמוֹ לֹא יָמִישׁ. וָאֶקְרָב אֵלָיו וְאָמַרְתִּי לוֹ: הֲמִסֶּלַע נַפְשְׁךָ חֲרוּשָׁה. וְרַעְיוֹנֶיךָ בַּרְזֶל אוֹ נְחוּשָׁה. וְאֵיךְ לֹא פָחַדְתָּ לְפַחְדָּתֵנוּ. וְלֹא חָרַדְתָּ לְחֶרְדָּתֵנוּ. כִּי כֻלָּנוּ יוֹשְׁבִים בְּשַׁעֲרֵי מָוֶת. וְרַגְלֵינוּ יוֹרְדוֹת אֶל חַדְרֵי מָוֶת. וְאַתָּה תִּתְיַצֵּב מִנֶּגֶד. וַתַּעַט הַגַּאֲוָה כַּבֶּגֶד. וְעַתָּה עַל מִי בָטַחְתָּ. כִּי שָׂמַחְתָּ. וְאֵיךְ יָדַעְתָּ. כִּי נִמְלַטְתָּ. וַיִצְחַק לְמִלַּי. וְהֵנִיד רֹאשׁוֹ עָלַי. וַיֹּאמֶר אֵלַי: כָּל מִי שֶׁיֵּדַע אֲשֶׁר אֵדַע. לֹא יִירָא לְמִקְרֶה וְלֹא לְפֶגַע.

אָמַר הַמַּגִּיד: וְכִשְׁמֹעַ הָעָם דְּבָרָיו. לֹא שָׁתוּ לֵב לְמַאֲמָרָיו. וְחֲשָׁבוּהוּ מְהוֹלָל חֲסַר דַּעַת. וּבַעַל נֶפֶשׁ מִשְׁתַּגַּעַת. עַד אֲשֶׁר שָׁקַט הַיָּם מִזַּעְפּוֹ. וְהִרְבָּה לְהָשִׁיב אַפּוֹ. וְרָאוּ כָל הָעָם כִּי צָדְקוּ אֲמָרָיו. וַיֵּצְאוּ לָאוֹר דְּבָרָיו. וְנֵזֶק לֹא עָבַר עֲלֵיהֶם. וְיַד הַמָּוֶת לֹא נָגְעָה בְּאַחַד מֵהֶם. וְשָׁב הַיָּם לִנְעִימוּתוֹ. כִּבְקַדְמָתוֹ. וְשָׁתְקוּ גַלָּיו. וְנָחָה רוּחַ הַשָּׁלוֹם עָלָיו. אָז יָקְרָה בְעֵינָם מַעֲלָתוֹ. וַיֵּדְעוּ כִּי הָאֵל חָלַק לוֹ מִתְּבוּנָתוֹ. וְהִתְקַבְּצוּ סְבִיבוֹ. לִבְחֹן נִיבוֹ. וַיֹּאמְרוּ לוֹ: הֲתַעְלִים מִמֶּנּוּ תַעֲלוּמוֹת שִׂכְלָךְ. הֲלֹא תַגִּיד לָנוּ מָה אֵלֶּה לָךְ. הֲלֹא תוֹדִיעֵנוּ חָכְמָתְךָ הָרְחָבָה. הֲלֹא תַאֲצֵל עָלֵינוּ רוּחֲךָ הַטּוֹבָה. אָמַר לָהֶם: אַתֶּם חֲשַׁבְתֶּם דִּבְרֵי לַעַג וּשְׂחוֹק. וְעַתָּה אֲגַלֶּה לָכֶם סוֹד עָמֹק וְרָחֹק. וְהוּא יָקָר לִבְלִי חֹק. דְּעוּ כִּי בְיִדי סֵפֶר. וּבוֹ אִמְרֵי שֶׁפֶר. וְכָל מַחֲזִיקוֹ יִמָּלֵט מֵרֶדֶת שַׁחַת וְיִמְצָא כֹפֶר. כִּי בוֹ שְׁמוֹת מַלְאַכֵי הַיָּם. אֲשֶׁר בּוֹ יִנָּצֵל מִשְּׁאוֹן גַּלָּיו וְדָכְיָם. וְכָל מִי יִשָּׁאֶנּוּ עַל זְרוֹעוֹ. בְּאָרְחוֹ וְרִבְעוֹ. וְשִׁבְתּוֹ וְנָסְעוֹ. אִלּוּ הָשְׁלַךְ בָּאֵשׁ לֹא תִשְׂרְפֵהוּ. אוֹ בַיָּם לֹא יִשְטְפֵהוּ. אוֹ בְּמַלְתְּעוֹת אַרְיֵה לֹא יִטְרְפֵהוּ. אָמְרוּ לוֹ: אִם תְּגַלֶּה לָנוּ סוֹד חָכְמְתָך. יֵשׁ שָׂכָר לִפְעֻלָּתְךָ. וְלֹא תִהְיֶה לָרִיק יְגִיעָתְךָ. אָמַר לָהֶם: נִשְׁבַּעְתִּי בְחֵי הַסּוֹד אֲשֶׁר בּוֹ כָתוּב. וְהוּא לַבּוֹרֵא חָשׁוּק וְאָהוּב. כִּי לֹא אֲגַלֶּה אוֹתוֹ עַד יִתֵּן לִי כַל אִישׁ זָהוּב. וְאָז אֶכְתֹּב לוֹ אִגֶּרֶת. תִּהְיֶה לוֹ בַצָּר שׁוֹמֶרֶת. חוֹצֶבֶת לַהַב. מַחֲצֶבֶת רַהַב. לָהּ מֶמְשֶׁלֶת וָרַהַב. וְלֹא יַעַרְכֶנָּה זָהָב.

אָמַר הַמַּגִּיד: וּכְשָׁמְעֵנוּ שְׁבוּעָתוֹ. מִהֲרוּ קְצָת אֲנָשִׁים לָתֵת לוֹ אֶת שְׁאֵלָתוֹ. אָז מָשַׁךְ מִתּוֹךְ חֵיקוֹ. שַׂקּוֹ. וְהוֹצִיא סֵפֶר וְכָתַב לְכָל אֶחָד אִגֶּרֶת. סוֹגֶרֶת וּמְסֻגֶּרֶת. וּבָהּ חֲקוּקִים שְׁמוֹת הַשְּׁבוּעוֹת. וְחוֹתָמוֹת וְטַבָּעוֹת. וַיֹּאמֶר לָנוּ: הִנֵּה לָכֶם בְּרָכָה. אֲשֶׁר בָּהּ תַּעֲלֶה לָכֶם אֲרוּכָה. וְכֹל אֲשֶׁר עִמּוֹ יִשָּׂאֶנָּה. וְעַל זְרוֹעוֹ יִתְלֶנָּה. הַגִּבּוֹרִים יִפְחֲדוּ מִצִּלּוֹ. אַפַּיִם אֶרֶץ יִשְׁתַּחֲווּ לוֹ. וְכִי יַעֲבֹר בַּיָּם יַשְׁבִּיחַ שְׁאוֹן גַּלָּיו. וְנִצְּבוּ כְמוֹ נֵד נוֹזְלָיו. וַיְהִי כַאֲשֶׁר יָצָאנוּ אֶל הַיַבָּשָׁה. וְשׁוֹרַרְנוּ לָאֵל שִׁירָה חֲדָשָׁה. שָׁמַעְתִּי הַזָּקֵן יְזַמֵּר. וְהוּא בְלַחַשׁ אוֹמֵר:

אָמְנָם אֲנִי חֶבֶר יְלִיד קֵינִי

שִׂכְלִי חֲלֵב לִקְחוֹ יְנִיקֵנִי.

בָּרוּךְ אֲשֶׁר לֹא יַעֲזֹב חַסְדּוֹ

מִכָּל מְתֵי חֶסְרוֹן בְּנֵי עֹנִי.

לוּלֵי עֲצָתִי קִבְּצָה לִי הוֹן

אֶצְעַק בְּחֹסֶר כֹּל וְאֵין עוֹנִי.

אָמַר הַמַּגִּיד: וּכְשָׁמְעִי מִפִּיו אֵלוּ הַמִּלּוֹת. יָדַעְתִּי כִּי הוּא חֲבֵרֵנוּ אֲבִי הַנְּבָלוֹת. וְרֹאשׁ הַתַּחְבּוּלוֹת. וְיָדַעְתִּי כִּי כָל דִּבְרֵי הָאֳנִיָּה. כָּזָב וּרְמִיָּה. וְעֵצוֹת לְבַקֵּשׁ הַמִּחְיָה. אַךְ חָמַלְתִּי עָלָיו. וְהִסְתַּרְתִּי רֹעַ פְּעָלָיו. וּמִטּוֹב עַד רָע לֹא דִבַּרְתִּי אֵלָיו. פֶּן תִּוָּדַע עָרְמָתוֹ. וְתִגָּלֶה בַקָּהָל רָעָתוֹ. וְהָלַכְתִּי מֵאִתּוֹ. וַאֲנִי בַלֵּב תָּמֵהַּ עַל עָרְמַת מְלַאכְתּוֹ. וְעֹמֶק מַחְשַׁבְתּוֹ.
