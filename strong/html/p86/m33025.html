ספר תחכמוני - שַׁעַר שְׁלשִׁים

הָרוֹפֵא הָאוֹרֵחַ. וּמִינֵי תְּרוּפוֹתָיו בָּרְחוֹבוֹת צוֹרֵחַ


נְאֻם הֵימָן הָאֶזְרָחִי. נָסַעְתִּי מִנַּחַל אַרְנוֹן. לְבַעַל גָּד אֲשֶׁר בְּבִקְעַת הַלְּבָנוֹן. וּבְעוֹדִי מְשׁוֹטֵט בְּחוּצוֹתֶיהָ. לִלְקָט מִלְּשׁוֹן חֲכָמֶיהָ פְנִינֵי מְלִיצוֹתֶיהָ. רָאִיתִי עַם רַב בָּאִים. וְנִצְבָּאִים. מִכָּל מְסִלָּה. וְנַעֲשׂוּ כַעֲגֻלָּה. וָאֵרָא בְתוֹךְ הֶהָמוֹן. זָקֵן כָּפוּף כְּאַגְמוֹן. וּסְבִיבָיו קָהָל רַב נִצָּבִים. וְהוּא נִצָּב עַל שַׁעַר בַּת רַבִּים. וּלְפָנָיו כַּפּוֹת מְלֵאוֹת. בְּמִינֵי רְפוּאוֹת. בָּאֵלֶּה מֶרְקָחוֹת. וּבָאֵלֶּה מְשִׁיחוֹת. וּלְפָנָיו רְטִיּוֹת. לְמַכּוֹת טְרִיּוֹת. וּתְעָלוֹת וַאֲרוּכוֹת. לְכָל מַחֲלָה עֲרוּכוֹת. מַעֲרָכוֹת מוּל מַעֲרָכוֹת. וְאֶצְלוֹ כְּלֵי בַרְזֶל וּמֶלְקָחַיִם. וְהַמַּזְלֵג שְׁלשׁ הַשִּׁנַּיִם. וּכְלֵי הַהַקָּזָה וְהַכְּוִיוֹת. וּכְלֵי חִתּוּךְ הַבָּשָׂר חַדִּים כְּחֶרֶב פִּיּוֹת. וְהוּא קוֹרֵא בְּמִלָּיו. לְכָל הַנִּצָּבִים עָלָיו.


וַיַּעַן וַיֹּאמַר:

שִׁמְעוּ אֵלַי עַמִּים. וְהַקְשִׁיבוּ הֲמוֹן לְאֻמִּים. אָנֹכִי הַבָּא מִמַּעַרְכוֹת הַזְּמַנִּים. וְהַנִּמְלָט מִיַּד מְקוֹרוֹת הַשְּׁאוֹנִים. וְהַמַּיִם הַזֵּידוֹנִים. עָבְרוּ עָלַי כַּמָּה תְּלָאוֹת. וְרָאוּ עֵינַי פְּלָאוֹת. וְשָׁמְעוּ אָזְנַי נוֹרָאוֹת. נָסַעְתִּי מֵאֶרֶץ עֵילָם. וְשׁוֹטַטְתִּי בְּאַפְסֵי עוֹלָם. לַחְקֹר חַכְמֵי תֵבֵל כֻּלָּם. וְלָמַדְתִּי חָכְמַת הָרְפוּאָה לִפְנֵי רוֹפְאִים עֲצוּמִים. וְיָצַקְתִּי מַיִם עַל יְדֵי חֲכָמִים מְחֻכָּמִים. וְשָׁאַבְתִּי בְצִמְאָה אֶת דִּבְרֵיהֶם. וְקִבַּלְתִּי מִפִּיהֶם וְלֹא מִפִּי סִפְרֵיהֶם. וְעָמַדְתִּי עַל סוֹדוֹתֵיהֶם. וְנִגְלוּ לִי יְסוֹדֵי חָכְמוֹתֵיהֶם. אֲנִי בְעֶזְרַת הָאֵל אֶרְפָּא מְחָצִים. וְאֶגְדֹּר פְּרָצִים. וְאֶחְבּשׁ מַעֲצָבִים. וְאַשְׁקִיט מַכְאוֹבִים. וְאִתִּי רְפוּאָה מְצוּיָה. לְכָל כְּאֵב וּכְוִיָּה. אֲשֶׁר בַּגְּוִיָּה. וְלִנְשִׁיכַת צִפְעוֹן וְחַיָּה. וּבְיָדִי רְטִיָּה. לְכָל מַכָּה טְרִיָּה. וּמִי שֶׁנְּשָׁכָהוּ נָחָשׁ יָבוֹא אֵלַי. וְיִשְׁמַע מִלָּי. וְהָיָה כָּל הַנָּשׁוּךְ וְרָאָה אוֹתִי וָחָי. וְלִי אָבָק. שְׁחַקְתִּיהוּ עַד אֲשֶׁר דָּק. לְהָאִיר רוֹאוֹת סְגוּרוֹת. וְלִפְקֹחַ עֵינַיִם עִוְרוֹת. וְאִתִּי רְפוּאָה לְהַזְרִיעַ הָעֲקָרוֹת. וְלַהֲשִׁיבָן הָרוֹת. וּלְחַמֵּם הַנָשִׁים הַקָּרוֹת. וְאִתִּי מָזוֹר. לְכָל מָזוֹר. וְכֶל נֶגַע אֲשֶׁר לֹא חֻבַּשׁ וְלֹא זוֹר. אֲנִי אֶהְיֶה לָכֶם מֵעִיר לַעְזֹר. אֲנִי בְּעֶזְרַת הָאֵל אֲנַהֵל פִּסְחִים. וְאָשִׁיב הַעִוְרִים פִּקְחִים. עַד יְדַלֵּג כָּאַיָּל פִּסֵּחַ. וְיָמָּה וְצָפוֹנָה יְנַגֵּחַ. גַּם כָּל חֳלִי וְכָל מַכָּה. אֲשֶׁר הַלֵּב מְדַכָּה. וְתָשִׁיב אֱנוֹשׁ עַד דַּכָּא. אֲנִי אֲסִירֶנָּה וְאַעֲבִירֶנָּה. וּבְשַׁלְוָה אֲמִירֶנָּה. וּבְיָדִי פּוּךְ. לְכָל דֶּמַע שָׁפוּךְ. וְכָל עַיִן הָפוּךְ. לְהוֹבִישׁ מֵימַיו. וְלַעְצֹר גְּשָׁמָיו בְּשָׁמָיו. וְלִי אֲרוּכָה. וּרְפוּאָה עֲרוּכָה. לְכָל רֶגֶל נִקְשֶׁרֶת. וּלְכָל יָד נִשְׁבֶּרֶת. וְכָל צוֹעֵק מִכְּאֵב אִישׁוֹן. בַּאֲפֵל לַיְלָה וְאִישׁוֹן. וְאֵינֶנּוּ מַנִּיחַ לוֹ לִישׁוֹן. בְּיָדִי רְפוּאוֹת חָלְיוֹ. עַד רוֹאָיו יֹאמְרוּ אַיּוֹ. וְכָל מִי הוּא נְכֵה רַגְלַיִם וְאֵין לוֹ מַעֲמָד. וְאִישׁ כִּי יִהְיֶה בוֹ שֶׁבֶר רֶגֶל אוֹ שֶׁבֶר יָד. אוֹ חוֹלֶה מַכְאוֹב צַוְּארוֹנוֹ. אוֹ תְבַלּוּל בְּעֵינוֹ. אוֹ גָרָב אוֹ יַלֶּפֶת בְּיָדִי רְפוּאָתוֹ. וְלִי רְפוּאָה לְכָל מַחֲלָה נִצַּחַת. וְלַשַּחֶפֶת וְלַקַּדַּחַת. וּלְאֵשׁ מִתְלַקַּחַת. אֲשֶׁר בֵּין הָעֲצָמוֹת קוֹדַחַת. וּלְצָרַעַת פּוֹרַחַת. וְלַשְּׂאֵת וְלַסַּפַּחַת. כִּי יִהְיֶה בַקָּרַחַת. אוֹ בַגַּבַּחַת. וְלִכְאֵב הַשִּׁנַּיִם. וּבֶקַע הַשְּׂפָתַיִם. וְרִפְיוֹן הַיָּדַיִם. וְשִׁלְשׁוּל הַמֵּעַיִם. וְחֹלֶשׁ הַמָּתְנַיִם. וּפִיק הַבִּרְכַּיִם. אֲנִי אֶרְפָּאֵם בְּעֶזְרַת שׁוֹכֵן שָׁמַיִם. וְכָל הַמִּשְׁתֵגַע וְנוֹפֵל עַל גְּחוֹנוֹ. וְיִמְצָא מַחֲנָק בְּגְרוֹנוֹ. וְיוֹצִיא מִפִּיו לְשׁוֹנוֹ. וַיּוֹרֶד רִיר עַל זְקָנוֹ. אֲנִי אָסִיר שִׁגְעוֹנוֹ. וְלִי מַרְפֵּא. לְכָל אִישׁ נִכְפֶּה. וּלְכָל לֵב נִרְפֶּה וְנִסְפֶּה. וּלְתַחְלוּאֵי הַפֶּה. וְלִנְשׁוּךְ אֶפְעֶה אוֹ שְׁפִיפוֹן. וּלְיֵרָקוֹן וּלְשִׁדָּפוֹן. וּלְגִידִים הַנְּתוּקִים. וּלְאֵבָרִים הַמִּתְפָּרְקִים. וְלָאִישׁ אֲשֶׁר שׁוֹקָיו הִתְעַוְּתוּ. וְרַגְלָיו מֵהֲלֹךְ שָׁבְתוּ. וּלְכָל אִישׁ צָרוּעַ. וּמֻכֵּה אֱלֹהִים וְנָגוּעַ. וְאִשָּׁה כִּי תִהְיֶה זָבָה אֲטַהֲרֶנָּה מִטֻּמְאָתָהּ. וְלַדָּוָה בְנִדָּתָהּ. בְּיָדִי רְפוּאָתָהּ. וְלַזָּב אֶת זוֹבוֹ. אָסִיר יְגוֹן לִבּוֹ. וְכָל הֶחֳלָאִים הַנִּזְכָּרִים. וַאֲשֶׁר אֵינָם נִזְכָּרִים. רְפוּאָתָם בְּיָדִי. וְגַם עַתָּה הִנֵּה בַשָּׁמַיִם עֵדִי. וּבַמְרוֹמִים שָׂהַדִי. וְכַל אֲשֶׁר דְּבָרִי יַאֲזִין. וְלֹא יַאֲמִין. יָבוֹא יְנַסֵּנִי. כִּי כַאֲשֶׁר דִּבַּרְתִּי כֵן יִמְצָאֵנְי.

אָמַר הַמַּגִּיד: וְכִשְׁמֹעַ הָעָם אִמְרָתוֹ. נִמְשְׁכוּ אֵלָיו בְּמֶתֶק מִלָּתוֹ. וְנִתְפְּשׂוּ כְדָגִים בִּמְצוּדָתוֹ. וְהִתְקַבְּצוּ אֲלָפִים וְרִבּוֹאוֹת. וְצִבְאֵי צְבָאוֹת. וּבָאוּ אֵלָיו הֲמוֹנִים הֲמוֹנִים. בְּמִינֵי נְגָעִים מִשְׁתַּנִּים. וָחֳלָאִים רָעִים וְנֶאֱמָנִים. וְנִגְּשׁוּ אֵלָיו גְּדוֹלִים וּקְטַנִּים. וַעֲשִׁירִים וְאֶבְיוֹנִים. וַעֲבָדִים עִם אֲדוֹנִים. בָּם עִוֵּר וּפִסֵּחַ. וְקֵרֵחַ וְגִבֵּחַ. וּבַעַל שְׁחִין פּוֹרֵחַ. זֶה בוֹכֶה וּמִתְיַפֵּחַ. וְזֶה צוֹעֵק וּמַר צוֹרֵחַ. וּכְלֵי הַשֶּׁתֶן בָּאִים עַל כַּמָּה מִינִים. וְכֵלִים מִכֵּלִים שׁוֹנִים. מֵהֶם אֲשֶׁר מַרְאֵהוּ שָׁחוֹר וּמֵהֶם מְאָדָּם. וְהָיָה הַנֶּגַע יְרַקְרַק אוֹ אֲדַמְדָּם. וַיִּתְקַבְּצוּ לְפָנָיו אֳנָשִׁים וְנָשִׁים. וּנְעָרִים וִישִׁישִׁים. זֶה צוֹעֵק בִּכְאֵבוֹ. וְזֶה יַגִּיד נֶגַע לְבָבוֹ. וְזֹאת תְּסַפֵּר מַחֲלָתָהּ וְאַנְחָתָהּ. וּמַחַץ מַכָּתָהּ. וְהוּא מַאֲזִין אֲלֵיהֶם. וּמְחַלֵּק רְפוּאוֹתָיו אֲלֵיהֶם. וְנוֹתֵן דִּבְרֵי הֲבָלִים לָהֶם. וְלוֹקֵחַ כְּסָפִים מִידֵיהֶם. עַד נִמְלָא כְרֵסוֹ וְכִיסוֹ. וְגָדַל מְשׂוֹשׂוֹ. וְסָר כַּעְסוֹ. וּמָצָא שׁוּק גָּדוֹל לִסְחוֹרָתוֹ. וּמָכַר בְּדָמִים יְקָרִים הַבְלֵי רְפוּאָתוֹ.

אָמַר הַמַּגִּיד: וְלֹא יָכֹלְתִּי לְהַגִּיעַ אֵלָיו. עַד נָפוֹצוּ הָעָם מֵעָלָיו. אָז קָרַבְתִּי אֶצְלוֹ. וְהִתְבּוֹנַנְתִּי לוֹ. וְאָמַרְתִּי: אַשְׁבִּיעֲךָ בְּאֵל אֶחָד וְאֵין שֵׁנִי. הַאַתָּה חֶבֶר הַקֵּינִי. וַיִצְחַק לְמִלָּי. וַיֹּאמֶר אֵלָי: אָנֹכִי מִבְחַר יְדִידֶיךָ. וְאִישׁ חֲסִידֶךָ. וַיִּשְׂמַח לִבִּי בְּחֶבְרָתוֹ. וְיָשַׁבְתִּי לְעֻמָּתוֹ. לְהוֹכִיחַ אוֹתוֹ. וְאָמַרְתִּי לוֹ: הַאִישׁ כָּמוֹךָ יְחַלֵּל עַל הַדְּרָכִים הוֹדוֹ. וִינַבֵּל בְּעִסְקֵי הָרַמָּאִים כְּבוֹדוֹ. וְיִכְעַס לְמִלַּי.


וַיֹּאמֶר אֵלָי:

אַתָּה יְדִיד נַפְשִי טְהָר-לֵב נֶאֱמָן

עַל כֵּן דְּבָרֶיךָ לְחִכִּי צוּף וּמָן.

אוּלָם אֲנִי סוֹבֵב לְבַקֵּשׁ מִחְיָה

אוּלַי אֱלֹהִים מֵחֲסָדָיו לִי יְמַן.

כִּי הַזְּמַן הָרַע מְצָאתִיהוּ כְצוּר

קָשֶׁה וְאֵין לִבּוֹ לְעָנְיִי רַחֲמָן.

לָכֵן בְּעֵת תִּרְצֶה תְּלוּנָה לַעֲרֹךְ

אַל תַּאֲשִׁימֵנִי וְהַאְשֵׁם הַזְּמָן.


וַיִּתְמַהּ לִבִּי לְשִׂכְלוֹ וְנִכְלוֹ. וְלֹא מָצָאתִי מַעֲנֶה לְהָשִׁיב לוֹ. וַיְהִי עַד כֹּה וְעַד כֹּה בִּקַּשְׁתִּיהוּ וְהִנֵּה סָר צִלּוֹ. וְחָלַף הָלַךְ לו.
