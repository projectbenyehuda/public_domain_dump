אֶרֶץ יִשְׂרָאֵל

ספר למוד הגאוגרפיה של ארץ ישראל (עם מפה בעברית).


פרק א: מְקוֹם אֶרֶץ יִשְׂרָאֵל, שְׁמוֹתֶיהָ, גְּבוּלוֹתֶיהָ, גָּדְלַָהּ וַחֲלוּקָתָהּ.


א)

עַל חוֹף הַיָּם הַגָּדוֹל, הוּא הַיָּם הַתִּיכוֹן אֲשֶׁר בְּמַעֲרַב אַסִּיָה, יוֹשֶׁבֶת הָאָרֶץ הַנִּקְרֵאת אֶרֵץ יִשְׂרָאֵל אוֹ אֶרֶץ יְהוּדָה, וְהָאָרֶץ טוֹבָה וּפוֹרִיָה, אֶרֶץ מִרְעָה וַעֲצֵי פֶרִי, אֶרֶץ הָרִים וּבְקָעוֹת, אֶרֶץ מַעֲיָנוֹתוּנְחָלִים.

אֶרֶץ יִשְׂרָאֵל יוֹשֶׁבֶת בֵּין הַמַּעֲלָה הַשְּׁלשִׁים וָאַחַת וּבֵין הַשְּלשִׁים וְׁשָׁלשׁ וָחֵצִי מִמַּעֲלוֹת רֹחַב הַצָּפוֹן, וּבֵין הַמַּעֲלָה הַחֲמִשִּׁים וּשְׁתַּיִם וּבֵין הַחֲמִשִּׁים וְחָמֵשׁ מִמַּעֲלוֹת אֹרֶךְ הַמִּזְרָח.

לְפָנִים נִקְרְאָה הָאָרֶץ הַזֹּאת גַם בְּשֵׁם אֶרֶץ כְּנַעַן עַל שֵׁם שִׁבְטֵי הַכְּנַעֲנִים שֶׁיָּשְׁבוּ בָה, וְהָעַמִּים יִקְרְאוּ לָהּ פַּלֶּסְתִּינָה עַל שֵׁם הַפְּלִשְׁתִּים שֶׁיָשְׁבוּ בָהּ לְפָנִים; וְיֵשׁ גַם אֲשֶׁר יִקְרְאוּ לָהּ אֶרֶץ הַעִבְרִים, הָאָרֶץ הַקְּדוֹשָׁה אוֹ אֵֶרֶץ יְיָ.

ב)

מִצְּפוֹן לָאָרֶץ יִשְׂאוּ הָרֵי הַלְּבָנוֹן רָמִים אֶת רָאשֵׁיהֶם הַמְכֻסִּים בְּשֶׁלֶג עוֹלָמִים, מִיָּם לָהּ יָשׂקּוּ גַלֵּי מֵי הַיָּם הַתִּיכוֹן, מִנֶּגֶד לָהּ יִשְׂתָּרֵעַ מִדְבַּר עֲרַב וּמִדְבַּר פָּארָן הַנּוֹגֵעַ בְּמִפְרַץ עֶצְיוֹן גָּבֶר אֲשֶׁר בְּיַם סוּף, וּמִקֶּדֶם מִדְבַּר סוּרִיָה.

בְּמִפְרַץ עֶצְיוֹן גָּבֶר הָיָה חוֹף לְמַעֲמַּד אֳנִיּוֹת מַלְכֵי יְהוּדָה בִּימֵי הַבַּיִת הָרִאשׁוֹן.

וְאֵלֶּה הֵם גְּבוּלוֹת הָאָרֶץ: גְּבוּל צָפוֹן – מִצִּידוֹן וְעַד הַר חֶרְמוֹן, גְּבוּל יָם – הַיָּם הַתִּיכוֹן מִנַּחַל מִצְרַיִם (בְּעַרְבִית אֵל עַרִישׁ) וְעַד צִידוֹן, וּגְבוּל נֶגֶב – מִנַּחַל מִצְרַיִם עַד מִדְבַּר פָּארָן, וּגְבוּל מִזְרַָח – הָיָה בִּימֵי דָוִד וּשְׁלֹמה וּבִימֵי מַלְכֵי הַחַשְׁמוֹנָאִים הַנָּהָר פְּרָת, וּבַיָּמִים הָהֵם רָחֲבוּ גְבוּלוֹת הָאָרֶץ גַּם מִצָּפוֹן.

ג)

מִדַּת אֹרֶךְ הָאָרֶץ מִגְּבוּלָהּ הַצְּפוֹנִי עַד גְּבוּלָהּ הַדְּרוֹמִי כְּמָאתַיִם וַחֲמִשִּׁים קִילוֹמֵתֶּר1, וּמִדַּת רָחְבָּהּ הַבֵּינוֹנִי כְּמֵאָה וַחֲמִשִּׁים קִילוֹמֵתֶּר, וּמִדַּת כָּל שִׁטְחָהּ כִּשְׁלשִׁים וְשִׁשָּׁה אֶלֵֶף קִילוֹמֵתֶּר מְרֻבָּע.

בִּזְמַנִּים רַבִּים הָיָה שִׁטְחָהּ גָּדוֹל הַרְבֵּה יוֹתֵר.

ד)

אֶרֶץ יִשְׂרָאֵל נֶחְלֶקֶת עַל יַד נַחַל הַיַּרְדֵּן, הָעוֹבֵר כִּמְעַט לְאֹרֶךְ כָּל הָאָרֶץ מִצָּפוֹן נֶגְבָּה, לִשְׁנָיִם: לְעֵבֶר הַיַּרְדֵּן יָמָּה וּלְעֵבֶר הַיַּרְדֵּן מִזְרָחָה.


בִּימֵי הַבַּיִת הָרִאשׁוֹן נֶחְלְקָה הָאָרֵֶץ לְמִסְפַּר שְׁנֵים עָשָׁר שִׁבְטֵי בְנֵי יִשְׂרָאֵל אֲשֶׁר יָרְשׁוּ אֶת הָאָרֶץ בְּגוֹרַל נַחֲלָתָם, וּבִימֵי הַבַּיִת הַשֵּׁנִי – לְאַרְבָּעָה: א) אֶרֶץ יְהוּדָה (שֵׁבֶט יְהוּדָה, דָּן, בִּנְיָמִין וְשִׁמְעוֹן), ב) אֶרֶץ שֹמְרוֹן (שֵׁבֶט אֶפְרַיִם וּמְנַשֶּׁה), ג) אֶרֶץ הַגָּלִיל (שֵׁבֶט יִשָּׂשׂכָר, זְבֻלוּן, אָשֵׁר וְנַפְתָּלִי), ד) אֶרֶץ עֵבֶר הַיַּרְדֵּן אוֹ הַגִּלְעָד (שֵׁבֶט רְאוּבֵן, גָד וּמְנַשֶּׁה). שְׁלשָׁה הָרִאשׁוֹנִים בְּעֵבֶר הַיַַּרְדֵּן יָמָּה וְהָאַחֲרוֹן בְּעֵבֶר הַיַּרְדֵּן מִזְרָחָה.

וְהָעַמִּים יִקְרְאוּ לְאֶרֶץ עֵבֶר הַיַּרְדֵּן בְּשֵׁם “פֵּרֵאָה”.


פרק ב: הֶהָרִים, הָעֲמָקִים וְהַמִּדְבָּרוֹת.


א)

אֶרֶץ יִשְׂרָאֵל הִיא אֶרֶץ הָרִים גְּבוֹהִים וּבְקָעוֹת עֲמֻקּוֹת מְאֹד.

1) צָפוֹנָה לְאֶרֶץ יִשְׂרָאֵל יִשָּׂא הַר חֶרְמוֹן אֶת רֹאשׁוֹ הַגָּבוֹהַ מְאֹד (כְּאַלְפַּיִם וּשְׁמֹנֶה מֵאוֹת מֵעַל מֵי הַיָּם), וּלְרַגְלָיו מְקוֹרֵי הַיַּרְדֵּן, הוּא הַנָּהָר הַגָּדוֹל בְּאֶרֶץ יִשְׂרָאֵל.

הַצֶּמַח רַב עַל יַרְכְּתַי הַר הַחֶרְמוֹן וְהַגָּפֶן תִּתֵּן פְרִי תִפְאָרָה. וְעֲצֵי פְרִי רַבִּים וְאִילָנֵי סְרַק שׁוֹנִים יִשְׂאוּ שָׁם רַָאשֵׁיהֶם.

2) נֶגְבָּה לְחֶרְמוֹן יֵרָאוּ מֵעֵבֶר לַיַּרְדֵּן יָמָּה הָרֵי נַפְתָּלִי וְהַגָּלִיל (גָּבְהָם כְּאֶלֶף מֵתֶּר2

אַדְמַת הֶהָרִים הָאֵלֶּה דְּשֵׁנָה מְאֹד, כִּי מַיִם רַבִּים יַשְׁקוּהָ וְהִצְמִיחוּ שָׁם עֲצֵי יַעַר לָרֹב.

3) נֶגְבָּה לָהֶם הַר תָּבוֹר עַל יַד עֵמֶק יִזְרְעֶאל (גָּבְהוֹ כְּשֵׁשׁ מֵאוֹת מֵתֶּר).

אַדְמָתוֹ אַדְמַת גִּיר וְיַעֲרֵי עֵצִים שׁוֹנִים מְכַסִּים אֶת צַלְעוֹתָיו.

4) נֶגְבָּה לְהַר תָּבוֹר – הָרֵי אֶפְרַיִם וְשֹׁמְרוֹן (בְּאֶרֶץ שֹׁמְרוֹן) אֲשֶׁר לִשְׁלֻחוֹתֵיהֶם מִקֶּדֶם יִקְרְאוּ הָרֵי הַגִּלְבֹּעַ וְלַאֲשֶׁר מִיָּם יִקְרְאוּ הָרֵי הַכַּרְמֶל (גָּבְהָם קָרוֹב לְשֵׁשׁ מֵאוֹת מֵתֶּר).

צַלְעוֹת הֶהָרִים הָאֵלֶּה דְּשֵׁנוֹת וּפוֹרִיּוֹת מְאֹד, וּמַעֲיְנוֹת מַיִם רַבִּים נוֹבְעִים בְּכָל הֶהָרִים הָאֵלֶּה וְהִשְׁקוּ אֶת אַדְמָתָם וְהִפְרוּהָ, וְכַרְמֵי זַיִת רַבִּים עַל צַלְעוֹת הֶהָרִים יָמָּה.

בִּקְצֵה הָרֵי אֶפְרַיִם נֶגְבָּה מִתְנַשְּׂאִים שְׁנֵי הָרִים גְּדוֹלִים, הַר עֵיבָל (גָּבְהוֹ כִּתְשַׁע מֵאוֹת וְשִׁשִּׁים מֵתֶּר) וְהַר גְּרִזִּים (גָּבְהוֹ קָרוֹב לִתְשַׁע מֵאוֹת מֵתֶּר).

5) נֶגְבָּה לָהֶם – הָרֵי יְהוּדָה (גָּבְהָם הַבֵּינוֹנִי הוּא כִּשְׁמֹנֶה מֵאוֹת מֵתֶּר). בֵּינֵיהֶם יְדוֹעִים הַר הַמּוֹרִיָה וְהַר הַזֵּיתִים בִּירוּשָׁלָיִם.

צַלְעוֹת הָרֵי יְהוּדָה כִּמְעַט עֲרֻמּוֹת עַתָּה מִכָּל צֶמַח וָעֵץ, אוּלַי לְפָנִים הָיוּ פוֹרִיוֹת כִּי הָיוּ נֶעֳבָדוֹת הֵיטֵב בִּידֵי יוֹשְׁבֵי הָאָרֶץ.

6) וּמֵעֵבֶר לַיַּרְדֵּן מִזְרָחָה יָרִימוּ הָרֵי הַגִּלְעָד אֶת רָאשֵּׁיהֶם (גָּבְהָם יַגִּיעַ עַד אַלְפַּיִם מֵתֶּר). הַיַּבּוֹק הוּא גְבוּל הֶהָרִים הָאֵלֶּה מִנֶּגֶּב.

הֶהָרִים הָאֵלֶּה מְכֻסִּים בְּיַעֲרֵי עֵצִים שׁוֹנִים.

7) וְנֶגְבָּה לָהֶם – הָרֵי מוֹאָב עַל שְׂפַת יָם הַמֶּלַח (גָּבְהָם יַגִּיעַ עַד שְׁבַע מֵאוֹת וַחֲמִשִּׁים מֵתֶּר). בֵּינֵיהֶם נִמְצָא הַר הָעֲבָרִים הוּא הַר נְבוֹ.

אַדְמַת הֶהָרִים הָאֵלֶּה כְּחוּשָׁה, אַך בְּכָל זֹאת יֵשׁ עֲלֵיהֶם צְמַח.

8) נֶגְבָּה לָהֶם – הָרֵי שֵׂעִיר הַנִּמְשָׁכִים עַד לְשׁוֹן יַם סוּף. הֶהָרִים הָאֵלֶּה עֲרֻמִּים מִכָּל צֶמַח.

ב)

1) הַעְמֶק הַגָּדוֹל בְּאֶרֶץ יִשְׂרָאֵל הִיא בִּקְעַת הַיַּרְדֵּן, הָעוֹבֶרֶת לְאֹרֶךְ הָאָרֶץ מִצָּפוֹן נֶגְבָּה, וְקָרְאוּ לַחֶלְקָה הַצְּפוֹנִית אֲשֶׁר עַל יַד יָם כִּנֶרֶת שִׁפְלַת גִּינֹסַר, וְלַחֶלְקָה הַדְּרוֹמִית יִקְרְאוּ הָעֲרָבָה אוֹ עַרְבַת הַיַרְדֵּן. וְהָעַרְבִיאִים יִקְרְאוּ לָעֲרָבָה הַזֹּאת אַלְגּוֹר אוֹ גוֹר, לֵאמֹר עֵמֶק מְסֻגָּר בֵּין הָרִים, וְכֵן יִקְרְאוּ לָּה הַגּוֹיִם:

מַעְיָנוֹת וְנַחֲלֵי מַיִם רַבִּים מַשְׁקִים אֶת שִׁפְלַת גִּינסַר וּמַפְרִים אוֹתָהּ, וְהַשְּׁפֵלָה הַזֹּאת הָיְתָה לְפָנִים כְּגַן עֵדֶן וַתְּהִי אָז מְלֵאָה אָדָם רָב אֲשֶׁר עֲבָדוּהַָ וַיִּשְׂבְּעוּ מִטּוּבָהּ. גַם בָּעֲרָבָה נִמְצָאִים מַעָיְנוֹת מַיִם אֲשֶׁר יְדַשְּׁנוּהַָ וְהִצְמִיחוּ בָהּ דֶּשֶׁא עֵשֶׂב לְמִרְעֵה הָעֲדָרִים וְנָתֹן תִּתֵּן גַם יְבוּל לְעוֹבְדֵֶיהָ.

2) וְעֵמֶק שֵׁנִי עוֹבֵר בָּאָרֶץ מִצָּפוֹן נֶגְבָּה עַל שְׂפַת הַיָּם הַתִּיכוֹן, וְקָרְאוֹ לֶחֶצְיוֹ הַצְפוֹנִי (עַד יָפוֹ) הַשָּׁרוֹן, וּלְחֶצְיוֹ הַדְּרוֹמִי הַשְּּׁפֵלָה. אַדְמַת הָעֵמֶק הַזֶּה פּוֹרִיָּה וְטוֹבָה לַעֲבוֹדָה.

3) וּלְרַגְלֵי הַר חֶרְמוֹן נִמְצַָא עֵמֶק מֵי מֵרוֹם. אַדְמָתוֹ דְּשֵׁנָה מְאֹד אַךְ יוֹשְׁבָיו הֵם בִּידוִּאים נוֹסְעִים בָעֵדֶר אֲשֶׁר לֹא יָשִׂימוּ לֵב לַעֲבוֹדַת הָאֲדָמָה וַתַּעָזֵב הָאָרֶץ אֲשֶׁר נָתְנָה לְפָנִים תְּנוּבָה רַבָּה מְאֹד.

4) נֶגְבָּה לּוֹ עֶמֶק זְבֻלוּן הַמִּשְׂתָּרֵעַ עַד עַכּוֹ. אַדְמָתוֹ אַדְמַת חוֹל דְּשֵׁנָה וּפוֹרִיַָה.

5) נֶגְבָּה לוֹ עֶמֶק יִזְרְעֶאל הַמִּשְׂתָּרֵעַ עַד הַיַּרְדֵּן, וְנָגַע הַעֵמֶק הַזֶּה בְּקָצֵהוּ יָמָּה אֶת עֵמֶק הַחוֹף הוּא הַשָּׁרוֹן וּבְקָצֵהוּ קֵדְמָה בְּבִקְעַת הַיַּרְדֵּן. אַדְמַת הָעֵמֶק הַזֶּה דְּשֵׁנָה מְאֹד מְאֹד וְנָתֹן תִּתֵּן תְּנוּבָה הַרְבֵּה מְאֹד אִם כִּי לֹא תַעָבֵד כֻּלָּהּ בִּידֵי אָדָם, וּבִמְקוֹמוֹת הַנֶּעֱזָבִים יִרְעוּ הַבִּידוּאִים אֶת עֶדְרֵיהֶם הָרַבִּים.

הָעֵמֶק הַזֶּה הוּא הַיּוֹתֵר יָפָה בְּכָל עִמְקֵי הָאָרֶץ וְהָרִים סָבִיב לוֹ: הָרֵי הַגִּלְגָּל מִצָּפוֹן, הַר תָּבוֹר מִקֶּדֶם, הָרֵי אֶפְרַיִם מִנֶּגֶב, וְהָרֵי הַכַּרְמֶל מִיָּם.

6) וּבֵין הָרֵי יְהוּדָה נִמְצֵאת רָמַת חֶבְרוֹן. אַדְמָתָהּ דְּשֵׁנָה וּפוֹרִיָּה מְאֹד, בָּה תִּשְׂגֶה הַגֶּפֶן, וַעֲנָבֶיהָ נוֹדָעִים לְהַלֵּל וּמֵהֶם יַעֲשׂוּ גַּם דְּבַשׁ עֲנָבִים. בָּהּ יִגְדְּלוּ גַּם הַתְּאֵנָה וְהַשָּׁקֵד וְהָאֱגוֹז.

וּבְעֵבֶר לַיַּרְדֵּן מִזְרָחָה לֹא נִמְצָא עֲמָקִים עֲמֻקִּים, כִּי הִיא אֶרֶץ רָמוֹת רַבּוֹת אֲשֶׁר נַחֲלֵי מַיִם יַפְרִידוּ בֵינֵיהֶן.

7) מִצָּפוֹן, לְרַגְלֵי הַחֶרְמוֹן, תִּמָּצֵא רָמַת הַגּוֹלָן וְהַבָּשָׁן (עַד נַחַל יַרְמוּךְ הַנּוֹפֵל אֶל הַיַּרְדֵּן).

8) נֶגְבָּה לָּהּ רָמוֹת גִּלְעָד וְחַוְרָן.

9) וְנֶגְבָּה לָהֶם רָמַת בְּנֵי עַמּוֹן (מוּל יָם הַמֶּלַח).

ג)

1) מִדְבָּרוֹת שְׁנַיִם הֵם בְּעִבְרִית. יֵשׁ מִדְבָּר אֲשֶׁר אַדְמָתוֹ לֹא תִצְלַח לְכָל זֶרַע הוּא הַיְשִׁימוֹן. כַּמִּדְבָּר הַזֶּה אֵין בְּתוֹךְ הָאָרֶץ, וְאַךְ עַל גְּבוּלוֹתֶיהָ נִמְצָאִים: מִדְבַּר פָּארַָן וּמִדְבַּר מִצְרַיִם מִנֶּגֶב, וּמִדְבַּר עֲרַב מִקֶּדֶם. וְיֵשׁ מִדְבָּר אֲשֶׁר אַדְמָתוֹ לֹא רָזָה וּמַצְמַחַת דֶּשֶׁא טוֹב לְמִרְעֵה הָעֲדָרִים, אַךְ כָּל עֵץ אֵין בּוֹ וְאַדְמָתוֹ לֹא נֶעֱבָדָה. מִדְבָּרוֹת כָּאֵלֶּה יֵשׁ בְּאֶרֶץ יִשְׂרָאֵל, וְאֵלֶּה הֵם הַגְּדוֹלִים בָּהֶם:

2) מִדְבַּר יְרִיחוֹ, צָפוֹנָה־יָמָּה לְיָם הַמֶּלַח.

3) מִדְבַּר יְהוּדָה, עַל שְׂפַת יָם הַמֶּלַח יַָמָּה.

4) מִדְבַּר בְּאֵר שֶׁבַע, בִּקְצֵה הָאָרֶץ נֶגְבָּה.


פרק ג: הַיָּמִּים, הַנְּחָלִים וְהַיֵמִים.


א)

1) מִיָּם לְאֶרֶץ יִשְׂרָאֵל יִמָּצֵא הַיָּם הַתִּיכוֹן, הוּא הַיָּם הַגָּדוֹל.

לְפָנִים קָרְאוּ לוּ גַם יָם פְּלִשְׁתִּים וְגַם הַיָּם הָאַחֲרוֹן. בְּרוֹב הַיָּם הַזֶּה אֵין בִּגְבוּל אֶרֶץ יִשְׂרָאֵל מִפְרָצִים לְמַעֲמַד אֳנִיּוֹת, כִּי אִם מִפְרַָץ עַכּוֹ וְהוּא עַל יַד חִיפָּה הָעִיר; גַּם כָּל חוֹף בָּטוּחַ עָשׂוּי בִּידֵי אָדָם אֵין לְאֹרֶךְ שְׂפַת הַיָּם, אִם כִּי עֵרֶךְ הַיָּם הַזֶּה רַב מְאֹד לַאָרֶץ, וַחֲכָמִים וּנְבוֹנִים מְנַבְּאִים לָאָרֶץ עֲתִידוֹת טוֹבוֹת בִּגְלַל מוֹשָׁבָהּ עַל חוֹף הַיָּם הַתִּיכוֹן.

2) בִּקְצֵה עַרְבַת הַיַּרְדֵּן נֶגְבָּה יֵשׁ מִקְוֵה מַיִם גָּדוֹל, אֲשֶׁר אָרְכּוֹ יַעֲלֶה כִּשְׁמוֹנִים קִילוֹמֵתֶּר וְרָחְבּוֹ הַבֵּינוֹנִי כִּשְׁלֹש עֶשְׂרֵה קִילוֹמֵתֶּר, הוּא יָם הַמֶּלַח.

אֶת הַיָּם הַזֶּה יִקְרְאוּ גַם יָם הַמָּוֶת, כִּי כָל נֶפֶשׁ חַיָּה אֵין בּוֹ, וְיֵשׁ גַּם אֲשֶׁר יִקְרְאוּ לוֹ יָם הָעֲרָבָה עַל שֵׁם עַרְבַת הַיַּרְדֵּן אֲשֶׁר עַל יָדוֹ: גַּם קָרְאוּ לוֹ הַיָּם הַקַּדְמוֹנִי כִּי הוּא מִקֶדֶם לַיָּם הַגָּדוֹל, גַּם יִקְרְאוּ לוֹ יָם הַחַמָר עַל הַחֵמָר הָרַב אֲשֶׁר עַל שְׂפָתוֹ וְעַל קַרְקָעוֹ.

וּמְשְּׂפַת הַיָּם הַמִּזְרָחִית נִכְנֶסֶת לְשׁוֹן יַבָּשָׁה אֶל תּוֹךְ הַיָּם הַנִּקְרֵאת בְּעַרְבִית אֵל לִישַׁאן, לֵאמֹר הַלָּשׁוֹן.

בִּימֵי הַבַּיִת הַשֵּׁנִי הָלְכוּ עַל פְּנֵי הַיָּם הַזֶּה אֳנִיוֹת רַבּוֹת לְאָרְכּוֹ וּלְרָחְבּוֹ, וְתַחַת הַשֶּׁקֶט הַשּׁוֹרֵר כַּיּוֹם שָׁם נִשְׁמַע בַּיָּמִים הָהֵם קוֹל עוֹשֵׁי מְלָאכָה בַּמָּיִם.

בְּמֵי הַיָּם הַזֶּה נִמְצָא בְּרוֹם בְּמִדָּה מְרֻבָּה מְאֹד, וּכְאֶבֶן הַשְּׁחוֹרָה הַנִּמְצֵאת עַל שְׂפָתוֹ מִסָּבִיב יְכוֹלִים לַעֲשּׂוֹת חֹמֶר3 אָסְפַלְט טוֹב, וְגַם מֶלַח בָּרוּד4 נִמְצָא עַל שְׂפַת הַיָּם לָרֹב, וְהַחֲכָמִים מְעִידִים כִּי גַם מַעֲיְנוֹת נֵפְטְ יֶשְׁנָם שָׁם מִסָּבִיב, וְעַל שְׂפַת הַיָּם נִמְצָאוֹת אֲבָנִים בּוֹעֲרוֹת בָּאֵשׁ כְּפָחָמִים וְהָעוֹבְרִים שָׁם מְבַשְׁלִים עֲלֵיהֶן אֶת אָכְלָם, וּבַלָּשׁוֹן יֵשׁ גַם גָפְרִית לָרֹב. פְנֵי הַיָּם הַזֶּה שְׁפָלִים מִפְּנֵי הַיָּם הַגָּדוֹל כְּאַרְבַּע מֵאוֹת מֵתֶּר, וְלָכֵן יִגְּבַּר שָׁם הַחֹם תָּמִיד.


3) צָפוֹנָה לוֹ כְּמֵאָה קִילוֹמֵתֶּר נִמְצָא מִקְוֵה מַיִם שֵׁנִי הַקָּטֹן פִּי שְׁלשָׁה מִיָּם הַמֶּלַח, הוּא יָם כִּנֶרֶת אוֹ כִּנְרוֹת, אָרְכּוֹ כְּעֶשְׂרִים וּשְׁנַיִם קִילֹומֵתֶּר וְרָחְבּוֹ כַּעֲשָׂרָה קִילוֹמֵתֶּר.

וְקָרְאוּ לוֹ גַּם יָם גִּינֹסַר עַל שֵׁם שְׁפְלַת גִּינֹסַר הַדְשֵׁנָה הַגּוֹבֶלֶת בּוֹ, וְיֵשׁ גַּם אֲשֶׁר יִקְרְאוּ לוֹ יָם טְבֶרְיָה עַל שֵׁם הָעִיר טְבֶרְיָה הָעוֹמֶדֶת עַל שְׂפָתוֹ.

הַדָּגָה רַבָּה מְאֹד בַּיָּם הַזֶּה, וּמֵימָיו טוֹבִים לִשְׁתִיָה. אַדְמַת הַחוֹף מִסָּבִיב אַדְמַת חוֹל הַמְכֻסָּה בִּימֵי הָאָבִיב בִּצְמָחִים נֶהְדָּרִים וְיָפִים, וְהֶהָרִים אֲשֶׁר מִסָּבִיב שְׁפָלִים וּנְחָלִים רַבִּים יוֹצְאִים מֵהֶם. פְּנֵי מֵי הַיָּם הַזֶּה שְׁפָלִים מִפְּנֵי מֵי הַיָּם הַגָּדוֹל בְּמָאתַיִם מֵתֶּר.

4) צָפוֹנָה לּוֹ כְּשִׁבְעָה עָשָׂר קִילוֹמֵתֶּר נִמְצָא עוֹד מִקְוֵה מַיִם קָטָן הוּא מֵי מֵרוֹם, כִּי הוּא הָרָם בֵּין שְׁלשֶׁת הַיַּמִּים הַנֶּאֱחָדִים בַּיַּרְדֵּן. אָרְכּוֹ כְּשִׁשָּׁה קִילוֹמֵתֶּר.

יֵשׁ אֲשֶׁר יִקְרְאוּ לוֹ יָם חֲוִילָה. בִּימֵי הַחֹרֶף יִגְדַּל הַיָּם אוֹ הָאֲגַם הַזֶּה מִמֵּי הֲמָסִיס אֲשֶׁר יֵרְדוּ מֵהֶהָרִים וּבִימֵי הַקַּיִץ יִדְלוּ מֵימָיו וְהָיָה לְבִצָּה וְעָלָה קָנִים גְּבֹהִים אֲשֶׁר מֵהֶם יַעֲשׂוּ הָעַרְבִיִּים מַחְצְלָאוֹת, וּבֵין הַקָּנִים הָאֵלֶּה יִשְׁכְּנוּ עוֹפוֹת הַמַּיִם, וְגַם חֲזִיר הַיַּעַר וְהַזְּאֵב וְהַשּׁוּעַָל וְהַצָּבוֹעַ יִבְחֲרוּ שָׁם מִשְׁכָּן לָהֶם.

הַדָּגָה רַבָּה מְאֹד בַּיָּם הַזֶּה וּמֵימָיו מְתוּקִים לִשְׁתִיָּה. אַדְמַת חוּפוֹ שְׁמֵנַָה מְאֹד, אַךְ שְׁמָמָה הִיא כִּי לֹא יַעַבְדוּהָ הָעַרְבִים יוֹשְׁבֵי הַמָּקוֹם, הֵם הַבִּידוּאִים הַנּוֹסְעִים בָּעֶדֶר.

5) וּמֵעֵבֶר לַיַּרְדֵּן קֵדְמָה, צָפוֹנָה־מִזְרָחָה מִמֵּי מֵרוֹם יֵשׁ אֲגַם עָמֹק, הַמָּלֵא תָמִיד מַיִם הוּא בְּרֵכַת רָם, אוֹ בְּרֵכַת הַמִּזְרָק, וּלְפָנִים קָרְאוּ לוֹ יָם פִילָא. הַחֲכָמִים אוֹמְרִים כִּי הוּא פִי הַר שְׂרֵפָה שֶׁכָּבָה.

ב)

בְּאֶרֶץ יִשְׂרָאֵל אֵין שְׁלָגִים זוּלָתִי עַל הָרֵי הַלְּבָנוֹן, וְגַם מַעְיָנוֹת גְּדוֹלִים אֵין בָּהּ הַרְבֵּה,

וְלָכֵן לֹא יִרְבּוּ בָהּ נְחָלִים גְּדוֹלִים וּרְחָבִים.


מֵימֵי אֶרֶץ יִשְׂרָאֵל מִתְחַלְּקִים לִשְׁתֵּי בְרֵכוֹת: בְּרֵכַת יָם הַתִּיכוֹן וּבְרֵכַת יָם הַמֶּלַח.

I. הַנְּהָרִים וְהַנְּחָלִים הַנּוֹפְלִים אֶל הַיָּם הַתִּיכוֹן הֵם:

1) נְהַר לִיטַנִּי, רֵאשִׁיתוֹ בַּבִּקְעָה אֲשֶׁר בֵּין הָרֵי הַלְּבָנוֹן וְהָלַךְ דְּרוֹמָה וּפָנָה מַעֲרָבָה וּמֵימֵי נְהָרִים וּנְחָלִים בָּאִים בּוֹ מִשְּׁנֵי עֲבָרָיו, וְתוֹצְאוֹתָיו הַיָּמָה צָפוֹנָה מִצּוֹר.

2) נְהַר נַעֲמָן (שִׁיחוֹר לִבְנָת) אוֹ בֶּלוּס, רֵאשִׁיתוֹ בְּהָרֵי זְבֻלּוּן, וְנוֹפֵל אֶל מִפְרַץ עַכּוֹ עַל יַד הָעִיר עַכּוֹ. הַחוֹל אֲשֶׁר בְּקַרְקַע הַנָּהָר הַזֶּה וּבְכָל הַכִּכָּר מִסָּבִיב דַּק מְאֹד וְטוֹב לִמְלֶאכֶת הַזְּכוּכִית, וְהָאֲגָדָה מְסַפֶּרֶת כִּי עַל יַד הַנָּהָר הַזֶּה מָצְאוּ הַצּוֹרִים אֶת רִגְבֵי הַזְּכוּכִית מֵהַחוֹל אֲשֶׁר נִתַּךְ בָּאֵשׁ, וְעַל פְּנֵי כָל חוֹף הַנָּהָר הַזֶּה נְפוּצוֹת קְלִפּוֹת תּוֹלַעַת הַחִלָּזוֹן אֲשֶׁר מִמֶּנוּ הָיוּ מוֹצִיאִים הַכְּנַעַנִים אֶת צֶבַע הָאַרְגָּמָן וְהַתְּכֵלֶת.

3) נַחַל קִישׁוֹן מְקוֹרָיו בְּהָרֵי הַגִּלְבֹּעַ וְעָבַר בַּגַּיְא אֲשֶׁר בֵּין הָרֵי הַגִּלְגָּל וְהַכַּרְמֶל וְנוֹפֵל לְמִפְרַץ עַכּוֹ לֹא רָחוֹק מֵחִיפָּה צָפוֹנָה. בְּעַרְבִית יִקְרְאוּ לַנַּחַל הַזֶּה מְקַטַּע, מֵי הַנַּחַל הַזֶּה יִרְבּוּ בַחֹרֶף וּבַקַּיִץ יִמְעָטוּ.

4) נְהַר הַתְּכֵלֶת אוֹ זַרְקָא יוֹרֵד מֵרַגְלֵי הַר הַכַּרְמֶל וְנוֹפֵל אֶל הַיָּם הַגָּדוֹל מִצְּפוֹן לְעִיר קֵסַרְיָה. לְפָנִים קָרְאוּ לַנָּהָר הַזֶּה נְהַר הַתַּנִּינִים כְּרוֹכוֹדִלְיוֹן עַל הַתַּנִּינִים הַקְּטַנִּים שֶׁנִּמְצָאִים בּוֹ, וְהַתַּנִּינִים כְּתַּנִּינֵי יְאוֹר מִצְרַיִם וְגַם אַוִּיר הַמָּקוֹם הַזֶּה כְּאַוִיר מִצְרַיִם הוּא.

5) נְהַר הַיָּרוֹק אוֹ אַל אַחְדָּר, נוֹפֵל אֶל הַיָּם נֶגְבָּה מִקֵּסַרְיָה. עַל שְׂפָתוֹ עוֹמֶדֶת הַמּושָׁבָה חִידְרָה.

6) נְהַר אַבּוּזְבוּרָה, נָהָר קָטֹן וּבִמְקוֹם נִפְלוֹ אֶל הַיָּם יֵשׁ מִפְרָץ קָטָן וְעָמְדוּ שָׁם בִּימֵי הַקַּיִץ אֳנִיוֹת סוֹחֵר קְטַנּוֹת לְהוֹצִיא מִשָּׁם אֲבַטִּיחִים לְמִצְרַיִם וְלַאֲרָצוֹת אֲחֵרוֹת.

7) נְהַר הַקָּנֶה אוֹ פַאלֵק עַל הַגְּבוּל בֵּין מְנַשֶּׁה וְאֶפְרַיִם, וְקָנִים רַבִּים גְּבֹהִים מְכַסִּים אֶת מֵימָיו וְאֶת שְׂפָתוֹ.

8) נְהַר הַיַּרְקוֹן אוֹ אַלְעוֹגָה נוֹפֵל אֶל הַיָּם מִצָּפוֹן לְיָפוֹ. עַל יַד הַנַּחַל הַזֶּה עוֹמֶדֶת הַמּוֹשָׁבָה פֶּתַח תִּקְוָה, וְעַל הַנַּחַל הַזֶּה בְּנוּיוֹת טַחֲנוֹת מַיִם רַבּוֹת.

9) נַחַל שׂוֹרֵק אוֹ רוּבִּין יוֹצֵא מֵהָרֵי יְהוּדָה מִצְּפוֹן לִירוּשָׁלַיִם וְנוֹפֵל אֶל הַיָּם מִצְרַיִם לְיָפוֹ.

10) נַחַל הַבְּשֹר אוֹ שַׂרִיעָה מוֹצָאוֹ בְּהָרֵי יְהוּדָה עַל יַד חֶבְרוֹן וְיָרַד נֶגְבָּה וּפָנָה מַעֲרָבָה וְתוֹצְאוֹתָיו הַיָמָּה.

11) נַחַל מִצְרַיִם הוּא הַגָּדוֹל בֵּין אֶרֶץ יִשְׂרָאֵל וּמִצְרָיִם.


II. הַנְּהָרִים וְהַנְּחָלִים אֲשֶׁר תּוֹצְאוֹתֵיהֶם לְיָם הַמֶּלַח.

1) הַנָּהָר הַגָּדוֹל אֲשֶׁר בְּאֶרֶץ יִשְׂרָאֵל הוּא הַיַּרְדֵּן, מְקוֹרָיו בְּרַגְלֵי הַר הַחֶרְמוֹן, וְהָלַךְ וְיָרַד דְּרוֹמָה בְּכָל דַּרְכּוֹ וְעָבַר אֶת מֵי מֵרוֹם וְאֶת יָם כִּנֶּרֶת וְנָפַל אֶל יָם הַמֶּלַח מִצָּפוֹן לוֹ, וּמֵי נְחָלִים רַבִּים בָּאִים בּוֹ מִשְּׁנֵי עֲבָרָיו. עַל שְׂפַת הַיַּרְדֵּן בְּנוּיוֹת טַחֲנוֹת מַיִם רַבִּים.

וְאֵלֶּה הֵמָּה הַנְּחָלִים הַנּוֹפְלִים אֶל הַיַּרְדֵּן מֵעֶבְרוֹ הַמִּזְרָחִי:

א) הַיַּרְמוּךְ, הַנּוֹפֵל אֶל הַיַּרְדֵּן מִיָּם כִּנֶּרֶת נֶגְבָּה. מֵי הַנַּחַל הַזֶּה רַבִּים מְאֹד וְהַדָּגָה רַבָּה בוֹ.

ב) נַחַל יַבּוֹק, הַנּוֹפֵל אֶל הַיַּרְדֵּן, בֵּין יָם כִּנֶּרֶת וּבֵין יָם הַמֶּלַח בַּתָּוֶךְ.

וְעוֹד נְחָלִים קְטַנִּים אֲחָדִים.

וּמִצַּד מַעֲרָב נוֹפְלִים אֶל הַיַּרְדֵּן:

ג) נְהַר בֵּית שְׁאָן הוּא נְהַר גַּלוּת הָעוֹבֵר בָּעֵמֶק בֵּין הָרֵי הַגִּלְגָּל וְהַגִּלְּבּעַ.

ד) נַחַל פָּרִיעַ הַנּוֹפֵל אֶל הַיַּרְדֵּן מִמַּעֲרָב לֹא רָחוֹק מִתּוֹצְאוֹת נַחַל יַבּוֹק.

וְעוֹד נְחָלִים קְטַנִּים אֲחָדִים.


2) וְאֶל יָם הַמֶּלַח מִקֶּדֶם נוֹפְלִים.

א) נְהַר תְּכֵלֶת מָעוֹן, תּוֹצְאוֹתָיו אֶל הַיָּם כְּשֵׁשׁ עֶשְׂרֵה קִילוֹמֵתֶּר נֶגְבָּה מִקְצֵה שְׂפַת הַיָּם הַצְפוֹנִית.

ב) נַחַל אַרְנוֹן, לַאדִי מוּגָב, הוּא הַגְּבוּל בֵּין אֶרֶץ יִשְׂרָאֵל לְאֶרֶץ מוֹאָב.

ג) נַחַל זֶרֶד, וַאדִי כֶּרֶךְ, הַנּוֹפֵל אֶל יָם הַמֶּלַח מִקְצֵה לְשׁוֹן הַיָּם נֶגְבָּה.

ד) וּמִיָּם נוֹפֵל אֶל יָם הַמֶּלַח נַחַל קִדְרוֹן, וַאדִי אַַנַאר, מְקוֹרָיו מִקֶּדֶם לִירוּשַָׁלַיִם

וְעוֹד נְחָלִים קְטַנִים אֲחָדִים.

לְבַד הַנְחָלִים הָאֵלֶּה הַנִזְכָּרִים יֵשׁ בְּכַל הָאָרֶץ שְׁלוּלִיוֹת אוֹ וַאדִים לָרֹב אֲשֶׁר בָּהֶם יִתְאַסְּפוּ מַיִם בִּימֵי הַגְּשָׁמִים וְשָׁטְפוּ בָאָרֶץ וּבִימֵי הַקַּיִץ יִיבָשׁוּ וְיֶחֱרָבוּ.

גַּם מַעְיָנוֹת נִמְצָאִים בְּכָל הָאָרֶץ וּבִפְרַט בֵּין הֶהָרִים.


III. יֵשׁ בָּאָרֶץ גַּם יֵמִים הֵם מַעְיְנוֹת מַיִם חַמִּים אוֹ מַעְיְנֵי הָרְפוּאַָה, בֵּינֵיהֶם יְדוּעִים:

1) חַמֵּי טְבַרְיָה, הֵם הַמַּעְיָנוֹת הַחַמִּים הַנִּפְלָאִים אֲשֶׁר עַל יַד הָעִיר טְבֶרְיָה מִנֶּגֶב. וַחֲכָמִים יוֹדְעֵי דָבָר אוֹמְרִים כִּי מֵי הַמַּעְיָנוֹת הָאֵלֶּה דּוֹמִים לְמֵי מַעְיְנוֹת קַרֶלְסְבַּד הַמְּפֻרְסָמִים, וְהֵם אוֹמְרִים כִּי לוּ הָיוּ הַמַּעֲיָנוֹת הָאֵלֶּה וְהַמָּקוֹם אֲשֶׁר מִסָּבִיב לָהֶם בְּאַרְצוֹת הַמַּעֲרָב, אוֹ לוּ תִקְנוּ אֶת הַמָּקוֹם וַיְשַׁכְלָלוּהוּ, כִּי עַתָּה הָיָה אַחַד הַמְּקוֹמוֹת הַיָּפִים בְּכָל כַּדּוּר הָאָרֶץ וְרִבְבוֹת אֲנָשִׁים הָיוּ בָאִים לְהֵרָפֵא בְּמֵימָיו וּלְהִתְעַנֵּג עַל נֹעַם אַוִירוֹ וְהָיָה לְמִחְיָה לְרִבְבוֹת אֲנָשִׁים.

2) גַּם מִצְּפוֹנָה שֶׁל טְבֶרְיָה נִמְצָאִים מַעְיְנוֹת חַמַיִּם טוֹבִים.

3) חַמֵּי גְדֵרָה הֵם הַמַּעְיָנוֹת הַחַמִּים הָרַבִּים אֲשֶׁר עַל שְׁנֵי עֶבְרֵי נְהַר יַרְמוּךְ. וְטֶבַע מֵימֵיהֶם כְּטֶבַע מֵי טְבַרְיָה.

4) וּמַעְיָנוֹת רַבִּים חַמִּים נִמְצָאִים עַל שִׂפְתוֹת הַנַּחַל תְּכֵלֶת מַעְיָן הַנּוֹפֵל אֶל יָם הַמֶּלַח.

בְּמֵימֵי הַמַַּעְיָנוֹת הָאֵלֶּה יֵשׁ מֶלַח וְגָפְרִית וְחֹמֶץ בַּרְזֶל בְמִדָה מְרֻבָּה מְאֹד.

וְכֹל הַמַּעיָנוֹת הָרַבִּים הַטּוֹבִים הָאֵלֶּה, אֲשֶׁר יָכְלוּ לְהָבִיא רְפוּאוֹת וְחַיִּים לְאָדָם רָב, נֶעֱזָבִים עַתָּה מִבְּלִי אֲשֶׁר יִשְׁתַּמְּשׁוּ כִמְעַט בָּהֶם.


פרק ד: שִׁלְטוֹן הָאָרֶץ, יוֹשְׁבֵֶיהָ, עָרֶיהָ הַגְּדוֹלוֹת, וּמוֹשְׁבוֹת הַיְהוּדִים.


§ 10.

1) אֶרֶץ יִשְׂרָאֵל נֶחְשֶׁבֶת לְאַחַת מִמְּדִינוֹת מַמְלֶכֶת תּוֹגַרְמָה, הִיא תּוֹרְקִיָה.

הַשּׁוּלְטַן מֶלֶךְ תּוֹרְקִיָה מוֹשֵׁל בָּאָרֶץ מֶמְשָׁלָה בִּלְתִּי מָגְבְּלֶת, הוּא גַּם נֶחְשָׁב לְרֹאשׁ כָּל הַמֻּשְׁלְמָנִים. לְכָל יוֹשְׁבֵי תוֹגַרְמָה מִשְׁפָּט אֶחָד וּזְכוּת אַחַת בְּלִי הַבְדֵּל דָּת, וְאַךְ בַּצָּבָא לֹא יַעַבְדוּ כִּי אִם הַמֻּשְׁלְמָנִים בְּנֵי דַת מַחְמַד, וּבְנֵי יֶתֶר הַדָּתוֹת מְשַׁלְמִים מֶכֶס הַצָבָא כָּל זָכָר לַגֻּלְגֹּלֶת מִדֵּי שָׁנָה בְשָׁנָה.

2) אֶרֶץ יִשְׂרָאֵל נֶחְלֶקֶת לִפְלָכִים (פשאליק או סַנְגַּק) בְּאֹפָן כָּזֶה:

א) פֶּלֶךְ יְרוּשָׁלַיִם, וּבָהּ יוֹשֵׁב הַפֶּחָה אֲשֶׁר יָבֹא בְּמַשָּׂא וּמַתָּן יָשָׁר עִם הַמֶּמְשָׁלָה הָעֶלְיוֹנָה בְּקוֹנְסְטַנְטִינוֹפּוֹל עִיר בִּירַת תּוֹרְקִיָה.

ב) פֶּלֶךְ שְׁכֶם אוֹ נֶפְלוּס, וּמְקוֹם מוֹשַׁב הַפֶּחָה שַׂר הַפֶּלֶךְ בִּשְׁכֶם.

ג) פֶּלֶךְ עַכּוֹ וּמְקוֹם מוֹשַׁב הַפֶּחָה בְּעַכּוֹ. שְׁנֵי הַפַּחוֹת הָאֵלֶּה סָרִים לְמִשְׁמַעַת הַנְּצִיב שַׂר הַגָּלִיל – וַאלִי – הַיּוֹשֵׁב בְּבַיְרוּתּ.

ד) פֶּלֶךְ חַוְרָן בְּעֵבֶר לַיַּרְדֵּן מִזְרָחָה בַּצָּפוֹן וּמְקוֹם מוֹשַׁב הַפֶּחָה בְּשִׁיךְ סַעַד.

ה) פֶּלֶךְ כֶּרֶךְ, בְּעֵבֶר לַיַּרְדֵּן מִזְרָחָה בַּדָּרוֹם וּמְקוֹם מוֹשַׁב הַפֶּחָה בְּכֶרֶךְ הִיא קִרְיַת מוֹאָב. שְׁנֵי הַפַּחוֹת הָאֵלֶּה סָרִים לְמִשְׁמַעַת הַנְּצִיב שַׂר הַגָּלִיל הַיּוֹשֵׁב בְּדַמֶּשֶׂק.


§ 11.

בְּאֶרֶץ יִשְׂרָאֵל נִמְצָאִים כִּשְׁלשֶׁת רִבְעֵי מִילְיוֹן נֶפֶשׁ אָדָם, וְהֵם מִתְחַלְּקִים לְעִירוֹנִים, פְּלָחִים וּבִידוּאִים.

הָעִירוֹנִים הֵם עַל פִּי רֹב אַנְשֵׁי שַׁחַץ גֵּאִים וַעֲרוּמִים: הָעֲשִׁירִים בָּהֶם עוֹסְקִים גַּם בְּמִסְחָר, אַךְ יָדָם לֹא תַעֲשֶׂה תּוּשִׁיָּה רַבָּה וְדַרְכֵי הַמִּסְחָר עֲזוּבוֹת, וְיֵשׁ בֵּינֵיהֶם גַּם בַּעֲלֵי אֲחֻזּוֹת, וְהָעֲנִיִּים רֻבָּם פּוֹעֲלִים שְׂכִירֵי יוֹם, אֳמָנִים מְעַטִּים בֵּינֵיהֶם, וּמַצַּב הַשְׂכָּלַת כֻּלָּם יַחַד שָׁפֵל מְאֹד.

וְהַפְּלָחִים הֵם יוֹשְׁבֵי הַכְּפָרִים הָעוֹסְקִים בַּעֲבוֹדַת הָאֲדָמָה עַל פִּי אָפְנֵי הָעֲבוֹדָה שֶׁנָּהֲגוּ אֲבוֹתֵיהֶם לִפְנֵי מֵאוֹת שָׁנִים וְאֵינָם מֵיטִיבִים אֶת הָאֲדָמָה, וְאֵינָם מַשְׁקִים אוֹתָהּ גַּם בִּמְקוֹמוֹת הַשְּׁפָלִים שֶׁיָּכְלוּ עַל נְקַלָּה לְהַרְבּוֹת, עַל יְדֵי בְאֵרוֹת וּמְכוֹנוֹת לְהַעֲלוֹת הַמַּיִם, אֶת מְקוֹמוֹת הַהַשְּׁקָאָה הַנּוֹתְנִים פְרִי רַב מְאֹד. הַפְּלָחִים אֵינָם רָעִים בְּטִבְעָם: מַצַּב הַשְׂכָּלָתָם שָׁפֵל מְאֹד וְכִמְעַט אֵין בֵּינֵיהֶם אַף אֶחָד הַיּוֹדֵעַ לִכְתֹּב אוֹת אֶחָת.

וְהַבִּידוּאִים הֵם יוֹשְׁבֵי אֹהֶל וּמִקְנֶה וְנוֹסְעִים בָּעֵדֶר. הֵם לֹא יִבְנוּ לָהֶם בָּתִּים לְשִׁבְתָּם וְלֹא יֵשְׁבוּ בְמָקוֹם אֶחָד יָמִים רַבִּים, כִּי אִם יַעַבְרוּ בָאָרֶץ מִמָּקוֹם לְמָקוֹם וְנָטוּ אָהֳלֵיהֶם בִּמְקוֹם אֲשֶׁר ייִשַׁר בְּעֵינֵיהֶם לְמִרְעֶה לְצֹאנָם וְלִבְקָרָם וּלְעָדְרֵי גְמַלֵּיהֶם אֲשֶׁר מֵהֵֶם יְחַיוּ אֶת נַפְשָׁם, מִבְּשָׂרָם יֹאכֵלוּ מֵחֶלְבָּם יִשְׁתּוֹ וּמִגֵּזָם יִתְחַמָּמוּ וְעָשׂוּ יְרִיעוֹת לְאָהֳלֵיהֶם וְהַשְּׁאָר יִמְכֶּרוּ: וְסָבַב הַבִּידוּאִי מִמָּקוֹם לְמָקוֹם וְשָׁב אֶל מְקוֹמוֹ הַָרִאשׁוֹן. אַךְ יֵשׁ גַם בִּידוּאִים אֲשֶׁר יֵשְׁבוּ זְמַן רַב עַל חוּפֵי הַנְּהָרוֹת בָּאֳהָלִים אוֹ גַּם בְּסֻכּוֹת־קָנִים וְעָבְדוּ גַם מְעַט אֶת הָאֲדָמָה לְהוֹצִיא מִמֶּנָּה לֶחֶם לְאָכְלָה, אֵלֶה הֵם חֲצִי בִידוּאִים, וְכָאֵלֶּה כֵּן אֵלֶּה יִחְיוּ עַל חַרְבָּם וְעַל קַשְׁתָּם. וְהֵם נֶחְלָקִים לִשְׁבָטִים וּלְמִשְׁפָּחוֹת וּלְכָל שֵׁבֶט וּמִשְׁפָּחָה זָקֵן – שֵׁיךְ – אֲשֶׁר יִמְשֹׁל בָּהֶם. מַצַּב הַשְּׂכָּלָתָם שָׁפַל מְאֹד, קְרֹא וּכְתֹב לֹא יֵדְעוּ, גַּם לְכָל מְלָאכָה וַחֲרשֶׁת לֹא יָשִׂימוּ לֵב וְלֹא יַחְפְצוּ לְדַעְתָּן. הַבִּידוּאִים נֶאֱמָנִים מְאֹד בִּבְרִיתָם לְאַנְשֵׁי שְׁלוֹמָם. מִסְפָּרָם לֹא נוֹדָע בְּדִיוּק. חוֹשְׁבִים, כִּי יֶשְׁנָם יוֹתֵר מִמָּאתַיִם אֶלֶף בְּכָל הָאָרֶץ. מוֹשָׁבָם הָעִיקְרִי בְּעֵבֶר לַיַּרְדֵּן מִזְרָחָה, כְּנֶגֶד הַשְּׁפַלָה, וּבִמְקוֹמוֹת רַבִּים שׁוֹכְנִים בְּתוֹךְ הָאָרֶץ וּבִפְרַט בַּגָּלִיל.


יוֹשְׁבֵי הָאָרֶץ הֵם בְּרב מוּשְׁלִימִים, אַךְ יֵשׁ גַּם נוֹצְרִים־יְוָנִים, אַרְמִינִים וְגַם עַרְבִיאִים – בְּמִסְפָּר לֹא קָטֹן, וְהֵם יוֹשְׁבִים רֻבָּם בְּעָרֵי הַחוֹף וְעוֹסְקִים בְּמִסְחָר. אוֹלָם יֵשׁ מֵהֶם גַם בִּפְנִים הָאָרֶץ וּבִפְרַט בֶּעָרִים הַקְּדוֹשׁוֹת לָהֶם: בְּבֵית לֶחֶם בִּיהוּדָה וּבְנָצָרֶת בַּגָּלִיל. וּבִסְבִיבוֹת יְרוּשָׁלַיִם וּבַגָּלִיל נִמְצָאִים גַּם כְּפָרִים אֲחָדִים אֲשֶׁר יוֹשְׁבֵיהֶם הֵם נוֹצְרִים הָעוֹסְקִים בַּעֲבוֹדַת הָאֲדָמָה וַעֲבוֹדָתָם יוֹתֵר מְסֻדָּרָה וְיוֹתֵר מְתֻקֶּנֶת מֵעֲבוֹדַת הַפְּלָחִים הַמּוּשְׁלִימִים וּבִכְלַל נַעֲלִים הֵם הַפְּלָחִים הַנּוֹצְרִים בְּאֹפַן חַיֵּיהֶם וְהַשְׂכָּלָתָם עַל הַפְּלָחִים הַמּוּשְׁלִימִים.

בָּאָרֶץ יֵשׁ יְהוּדִים כְּשִׁבְעִים אֶלֶף נָפֶשׁ, רֻבָּם יוֹשְׁבִים בְּאַרְבַּעַת הֶעָרִים הַקְּדשׁוֹת: יְרוּשָׁלַיִם, חֶבְרוֹן, טְבֵרְיָה וּצְפַת, וּבִזְמַן הָאַחֲרוֹן גָּדַל הַיִּשּׁוּב גַּם בְּיָפוֹ עִיר הַחוֹף, וְגַם בְּיֶתֶר עָרֵי הַחוֹף יֶשְׁנָן קְהִלּוֹת יִשְׂרָאֵל, וּבְעֶשְׂרִים הַשָּׁנִים הָאַחֲרוֹנוֹת הִתְיַשְּׁבוּ הַיְהוּדִים בַּמּוֹשָׁבוֹת אֲשֶׁר כּוֹנְנוּ בָאָרֶץ לַעֲבוֹדַת הָאֲדָמָה.


§ 12.

וְאֵלֶּה שְׁמוֹת הֶעָרִים הַגְּדוֹלוֹת וְהַנִּכְבָּדוֹת בְּאֶרֶץ יִשְׂרָאֵל:

I. מֵעֵבֶר לַיַּרְדֵּן מַעֲרָבָה.

1) בְּפֶלֶךְ יְרוּשָׁלַיִם.

א) יְרוּשָׁלַיִם5, עַל הָרֵי יְהוּדָה, מִסְפַּר יוֹשְׁבֶיהָ כַּחֲמִשִּׁים אֶלֶף אִישׁ, בָּהֶם יְהוּדִים כִּשְׁלשִׁים אֶלֶף וְהַמּוֹתָר חֶצְיָם נוֹצְרִים וְחֶצְיָם מוּשְׁלִימִים. הַמִּסְחָר לֹא רַב בָּעִיר. רַבִּים מֵהַיְּהוּדִים יוֹשְׁבֵי הָעִיר הַזֹּאת וְכֵן גַּם רַבִּים מִיּוֹשְׁבֵי יֶתֶר עָרֵי הַקֹּדֶשׁ. וּבִפְרַט הַזְּקֵנִים וְהַחַלָּשִׁים, מְקַבְּלִים חֲלוּקָה הִיא תְמִיכָה אֲשֶׁר יִשְׁלְחוּ אַחֵינוּ מִכָּל קְצוֹת הָאָרֶץ, אַךְ יֵשׁ בָּהֶם גַם בַּעֲלֵי מְלָאכָה וָאֳמָנִים רַבִּים. הַחַיִּים בַּעֲמַל כַּפֵּיהֶם, גַּם יֵשׁ הָעוֹסְקִים בְּחָנְוָנוּת וּמִסְחָרִים קְטַנִּים,

הַתַּעֲשִׂיָה לֹא פָרְצָה בָּעִיר. קוֹרוֹת יְרוּשָׁלַיִם עַד חֻרְבַּן הַבַּיִת הַשֵּׁנִי יְדוֹעוֹת לְכָל אִישׁ יִשְׂרָאֵל. אַחֲרֵי הַחֻרְבָּן נִשְׁאָרָה יְרוּשָׁלַיִם שׁוֹמֵמָה וַעֲזוּבָה מִבָּנֶיהָ אֲשֶׁר הָלְכוּ בַּגּוֹלָה, וְאַךְ בִּימֵי קוֹנְסְטַנְטִין הַגָּדוֹל הֵחֵלּוּ הַיְּהוּדִים לָשׁוּב אֵלֶיהָ בְּרִשְׁיוֹנוֹ. בַּמֵּאָה הַשִּׁשִּׁית לַחֻרְבָּן עָלוּ הַפַּרְסִים עָלֶיהָ וַיַּהַרְסוּהָ, אַךְ בְּבוֹא הָאָרֶץ לִידֵי הַמּוּשְׁלִימִים הוּטַב שָׁם מַצַּב הַיְּהוּדִים, וּבַעֲלוֹת נוֹסְעֵי הַצְלָב הָרְגוּ אֶת כָּל הַיְּהוּדִים אֲשֶׁר מָצְאוּ בִּירוּשָׁלַיִם; אוּלָם אַחֲרֵי כֵן הֵחֵלּוּ לְהִתְיַשֵּׁב בָּהּ מְעַט מְעָט וּבְשׁוּב הָאָרֶץ לִידֵי מוֹשְׁלֵי הַמַּחְמָדִּים רַב מִסְפַּר הַיְּהוּדִים בָּהּ.

בִּירוּשָׁלַיִם נִמְצָא שְׂרִיד הַכּוֹתֵֶל הַמַּעֲרָבִי אֲשֶׁר לְבֵית יְיָ, שְׂרִידֵי מִגְדַּל דָּוִד, קִבְרֵי הַמְּלָכִים, וְעוֹד.

בִּירוּשָׁלַיִם אִי־אֶפְשָׁר לַחְפֹר בְּאֵרוֹת וְגַם מַעְֲֲיָנוֹת אֵין בָּהּ, וְשָׁתוּ יוֹשְׁבֶיהָ מַיִם מְהַבּוֹרוֹת הַחֲצוּבוֹת בָּאֲדָמָה אֲשֶׁר בָּהֶם יִתְאַסְּפוּ מֵימֵי הַגְּשָׁמִים בַּחֹרֶף. בִּימֵי הַבַּיִת הַשֵּׁנִי הֵבִיאוּ מַיִם לִירוּשָׁלַיִם דֶרֶך צִנּוֹרוֹת מִמַּעֲיְנוֹת אֲשֶׁר מִחוּץ לָעִיר.

ב) יָפוֹ6, עַל שְׂפַת יָם הַתִּיכוֹן, מִסְפַּר יוֹשְׁבֶיהָ כְּעֶשְׂרִים וְשִׁשָּׁה אָלֶף. בָּהֶם יְהוּדִים כִּשְׁלשֶׁת אֲלָפִים וְנוֹצְרִים כְּאַרְבַּעַת אֲלַָפִים וְהַשְּׁאָר מוּשְׁלִימִים. בָּהּ יוֹשֵׁב שַׂר הַמָּחוֹז.

יָפוֹ הִיא עִיר מִסְחָר. דֶּרֶךְ הַחוֹף יָבִיאוּ אֶת כָּל הַסְּחוֹרוֹת מִחוּץ לָאָרֶץ וְיוֹצִיאוּ הַרְבֵּה תְבוּאָה וְשֶׁמֶן זַיִת וְשֻׁמְשְׁמִין וְסַבּוּן וּבִפְרַט תַּפּוּחֵי־זָהָב (כַּחֲמֵשׁ מֵאוֹת אֶלֶף תֵּבוֹת תַּפּוּחֵי זָהָב יוֹצְאוֹת מִיָפוֹ בְּשָׁנָה וּמְחִיר הַתֵּבָה הוּא בְּעֶרֶךְ שִׁשָּׁה פְרַנְקִים) וְאֶתְרוֹגִים (כְּמֵאָה וַחֲמִשִּׁים אָלֶף בְּשָׁנָה) מֵהַפַּרְדֵּסִים הָרַבִּים אֲשֶׁר סָבִיב לָעִיר. בָּהּ יֵשׁ בָּתֵּי בַד, בָּתֵי מְלֶאכֶת סַבּוּן וּבָתֵּי חֲרשֶׁת בַּרְזֶל אֲחָדִים. מְקַבְּלֵי חֲלוּקָה מְעַטִּים בְּיָפוֹ. רֹב יוֹשְׁבֵֶיהָ הַיְהוּדִים הֵם בַּעֲלֵי מְלָאכָה, חֶנְוָנִים, סוֹחֲרִים, פּוֹעֲלִים בַּמּוֹשָׁבוֹת.

יָפוֹ הִיא עִיר עַתִּיקָה מְאֹד, הִיא הָיְתָה עַל גְּבוּל נַחֲלַת דָּן. חֻפָּהּ הָיָה יָדוּעַ בִּימֵי קֶדֶם, וּבְעֵת מִלְחֲמוֹת הַחֵרוּת לַָקְחָה הָעִיר הַזֹּאת חֵלֶק גָּדוֹל, אַךְ מִימֵי נוֹסְעֵי הַצְּלַב בָּטֵל הַיִּשּׁוּב בָּהּ עַד אַחֲרִית הַמֵּאָה שֶׁעָבְרָה, וּבַשָׁנִים הָאַחֲרוֹנוֹת מֵעֵת אֲשֶׁר הֵחֵלּוּ הַיְּהוּדִים לְיַסֵּד מוֹשָׁבוֹת גָּדְלָה, קְהִלַּת הַיְּהוּדִים בְּיָפוֹ.

ג) עַזָּה7, כַּחֲצִי שָׁעָה מֵחוֹף הַיָּם, מִסְפַּר יוֹשְׁבֶיהָ כְּעֶשְׂרִים אֶלֶף אִישׁ, בָּהֶם מֵאוֹת אֲחָדוֹת נוֹצְרִים, וְכִשְׁמֹנִים נֶפֶשׁ יְהוּדִים, בָּהּ יוֹשֵׁב שַׂר הַמָּחוֹז.

דֶּרֶך חוֹף עַזָּה יוֹצִיאוּ תְבוּאָה רַבָּה מִשְּׂדוֹת הַשְּׁפֵלָה. לְפָנִים הָיְתָה יְדוּעָה גַּם בִּכְרָמֶיהָ, וְעַד הַיּוֹם יֵשׁ בִּכְפָר אֶחָד קָרוֹב לְעַזָּה כְּרָמִים נִפְלָאִים, שָׁם יַעֲשׂוּ גַם שֶׁמֶן זַיִת וְשֻׁמְּשְׁמִין וְסַבּוּן. בְּעֵת הָאַחֲרוֹנָה הֵחֵלּו לִנְטֹעַ גַּם סְבִיבָה פַּרְדֵּסִים.

ד) חֶבְרוֹן8, מִסְפַּר יוֹשְׁבֶיהָ כַּחֲמִשָּׁה עָשָׂר אָלֶף, בָּהֶם כְּאֶלֶף וַחֲמֵשׁ מֵאוֹת יְהוּדִים, נוֹצְרִים מְעַטִּים בָּהּ, וּשְׁאָר הַתּוֹשָׁבִים הֵם מוּשְׁלִימִים. בָּהּ יוֹשֵׁב שַׂר מָחוֹז.

בָּעִיר הַזֹּאת יֵשׁ בָּתֵּי בַד לְמַעֲשֵׂה שֶׁמֶן, גַּם סַבּוּן יַעֲשׂוּ שָׁם, גַּם כְִּלֵי זְכוּכִית פְּשׁוּטִים יַעֲשׂוּ הָעַרְבִים שָׁם. יֵשׁ שָׁם יְהוּדִים הָעוֹסְקִים בְּמַעֲשֵׂה הַיַּיִן מֵהַכְּרָמִים הַנִּפְלָאִים וְהַמְּשֻׁבָּחִים הָרַבִּים אֲשֶׁר מִסָּבִיב לָעִיר וְיֵשׁ גַּם אֲשֶׁר יַעֲשׂוּ דְּבַשׁ מֵהָעֲנָבִים. הָעִיר הַזֹּאת נוֹשָׁנָה הִיא בְּאֶרֶץ יִשְׂרָאֵל, בָּהּ יָשַׁב הַמֶּלֶךְ דָּוִד בְּרֵאשִׁית מַלְכוּתוֹ; עוֹלֵי בָּבֶל הִתְיַשְׁבוּ בָהּ, וְהָרוֹמָאִים הֲרָסוּהָ כָּלָה, וְאַחֲרֵי אֲשֶׁר לָקְחוּ הַמַּחְמַדִּים אֶת הָאָרֶץ מִידֵי נוֹסְעֵי הַצְּלַב רָבוּ בָהּ הַיְּהוּדִים וּמֵאָז לֹא חָדַל יִשּׁוּב יִשְׂרָאֵל בָּעִיר הַזֹּאת.

עַל יַד הָעִיר נִמְצֵאת מְעָרַת הַמַּכְפַּלָה אֲשֶׁר בָּהּ קְבוּרִים אֲבוֹתֵינוּ הָרִאשׁוֹנִים, וְעוֹד קִבְרוֹת קְדוֹשִׁים שׁוֹנִים.

ה) בְּאֵר־שֶׁבַע9, בִּדְרוֹם הָאָרֶץ, עִיר חֲדָשָׁה הַנּוֹסְדָה מִטַּעַם הַמֶּמְשָׁלָה לְמוֹשַׁב שַׂר הַמָּחוֹז אֲשֶׁר יִשְׁלַט עַל הַבִּידוּאִים הָרַבִּים אֲשֶׁר מִסָּבִיב.

ו) לוּד10, מִסְפַּר יוֹשְׁבֶיהַָ כִּשְׁמֹנָה אֲלָפִים אִישׁ, בָּהֶם נוֹצְרִים כְּאַלְפַּיִם וַחֲמֵשׁ מֵאוֹת, יְהוּדִים אֵין בָּהּ. בָּעִיר הַזֹּאת יֵשׁ בָּתֵּי בַד לְמַעֲשֵׂה הַשֶּׁמֶן וּבָתֵּי מַעֲשֵׂה הַסַּבּוּן. סָבִיב לָעִיר יֵשׁ כַּרְמֵי זַיִת לָרֹב.

אַחֲרֵי חֻרְבַּן יְרוּשָׁלַיִם הָיָה בָהּ יִשּׁוּב יְהוּדִים נִכְבָּד וַתְּהִי גַם מֶרְכָּז לְתוֹרַת יִשְׂרָאֵל. בַּיָּמִים הָהֵם הָיְתָה גַם עִיר מִסְחָר וְסוֹחֲרֶיהָ הָיוּ יְדוּעִים בַּחֲרִיצוּתָם וְעֻרְמָתָם, אוּלָם אַחֲרִי כֵן יַָרְדָה מְעַט מְעַט וּמִימֵי מַסְעֵי הַצְּלַב בָּטַל יִשּׁוּב הַיְהוּדִים בָּעִיר הַזֹּאת.

ז) רַמְלָה11, מִסְפַּר יוֹשְׁבֶיהָ כַּעֲשֶׂרֶת אֲלָפִים אִישׁ, וּבָהֶם כְּאֶלֶף נוֹצְרִים: יְהוּדִים מְעַטִּים בָּהּ, וְהֵם אָמָנִים וּפוֹעֲלִים בַּמּוֹשָׁבוֹת הַקְּרוֹבוֹת.

בְּרַמְלָה יֵשׁ בָּתֵּי בַד לְשֶׁמֶן זַיִת וְשֻׁמְשְׁמִין וּבָתֵּי מַעֲשֵׂה סַבּוּן. סָבִיב לָעִיר יֵשׁ כַּרְמֵי זַיִת הַרְבֵּה. הָעִיר הַזֹּאת נִבְנְתָה אֵיזֶה מֵאוֹת שָׁנָה אַחֲרֵי הַחֻרְבָּן, וְיִשּׁוּב יְהוּדִים נִכְבָּד לֹא הָיָה בָהּ מִיּוֹם הִוָּסְדָהּ.

ח) בֵּית לֶחֶם12, מִסְפַּר יוֹשְׁבֶיהָ כְּשִׁבְעַת אֲלָפִים, כֻּלָּם נוֹצְרִים. בָּעִיר הַזֹּאת יַעַסְקוּ הַרְבֵּה בַּעֲשִׂיַּת כְּלֵי צְדֶף13 קְטַנִּים וְיָפִים. עַל יַד הָעִיר הַזֹּאת נִמְצָא קֶבֶר רָחֵל אִמֵּנוּ.

ט) יְרִיחוׂ14, עִיר קְטַנָּה, וְיוֹשְׁבֶיהָ הֵם פְּלָחִים מוּשְׁלִימִים. עַל יַד הָעִיר הַזֹּאת נִמְצָא מַעְיָן גָּדוֹל וּמִמֶּנּוּ יוֹצֵא נַחַל הַמַּשְׁקֶה אֶת הָאֲדָמָה מִסָּבִיב, וְהַמְּסוֹרָה תִּקְרָא לַמַּעְיָן הַזֵֶּה “עֵין אֱלִישָׁע”, כִּי לְפִי דְבָרֶיהָ הוּא הַמַּעְיָן אֲשֶׁר רִפָּא הַנָּבִיא הַזֶּה אֶת מֵימָיו.

2) בְּפֶלֶךְ שְׁכֶם:

א) שְׁכֶם15, – נַפְלוּס – בֵּין הַר גְּרִזִּים וּבֵין הַר עֵיבָל. מִסְפַּר יוֹשְׁבֶיהָ כְּעֶשְׂרִים וְאַרְבָּעָה אֶלֶף אִישׁ, בָּהֶם כִּשְׁבַע מֵאוֹת נוֹצְרִים, כְּמֵאָה יְהוּדִים וּכְמֵאָה וְעֶשְׂרִים שֹׁמְרוֹנִים.

סְחַר הָעִיר רַב, כִּי אֵלֶיהָ יָבִיאוּ הַבִּידוּאִים יוֹשְׁבֵי עֵבֶר הַיַּרְדֵּן יְבוּל אַדְמָתָם וְגֵז עַשְׁתְּרוֹתֵיהֶם, וְגַם הָאֲדָמָה מִסָּבִיב דְּשֵׁנָה וּפוֹרִיָה כִּי מַעְיָנוֹת רַבִּים בָּהּ. בָּהּ יֵשׁ גַּם בָּתֵּי מְלָאכָה רַבִּים לְמַעֲשֵׂה הַסַּבּוּן.

בָּעִיר הַזֹּאת נִמְצֵאת קְהִלַּת הַשּׂמְרוֹנִים הַיְחִידָה בְּכָל הָעוֹלָם. בָּעִיר הַגְּדוֹלַָה הַזֹּאת הִתְאַחֲזוּ הַכּוּתִים אוֹ הַשּׁׂמְרוֹנִים בִּימֵי גָלוּת בָּבֶל וּבִימֵי הַבַּיִת הַשֵּׁנִי הָיְתָה הָעִיר הַזֹּאת לָהֶם לְעִיר הַמֶרְכָּז וְגַם לְעִיר הַקְּדוֹשָׁה כִּי עַל יָדָהּ בָּנוּ לָהֶם הֵיכָל. יוֹחָנָן הוֹרְקְנוֹס לָכַד אֶת הָעִיר וַיַּהֲרֹס אֶת הַהֵיכָל, וְאַסְפַּסְיָנוּס בָּנָה אוֹתָהּ וַיִּקְרָאֶהָ פְלַבִיָּה נְיָאפּוּלִיס, וְלָכֵן יִקְרָאוּהָ הָעַרְבִים נַפְלוּס אוֹ נַבְלוּס. קְהִלוֹת יְהוּדִים לֹא הָיוּ בָהּ בְּכָל הַזְּמַָן.

ב) גַּנִּין16 הִיא עֵין גַּנִּים, כִּשְׁלֹשֶׁת אֲלָפִים יוֹשְׁבִים, בָּהֶם נוֹצְרִים מְעַטִּים וִיהוּדִים אֵין שָׁם כְּלָל. בָּהּ יוֹשֵׁב שַׂר מָחוֹז.

מֵהַמַּעְיָן הַגָּדוֹל אֲשֶׁר מֵימָיו יָבֹאוּ תּוֹךְ הָעִיר וְהָיוּ לְנַחַל קָטֹן יַשְׁקוּ אֶת הַגַּנִּים הָרַבִּים הַסּוֹבְבִים אֶת הָעִיר.

ג) תֵּל כֶּרֶם17, (בני סַעַב) כִּשְׁלשֶׁת אֲלָפִים יוֹשְׁבִים מוּשְׁלִימִים. בָּהּ יוֹשֵׁב שַׂר מָחוֹז. הָאֲדָמָה מִסָּבִיב פּוֹרִיָּה וְטוֹבָה לְמִזְרַע תְּבוּאָה.

3) בְּפֶלֶךְ עַכּוֹ.

א) עַכּוֹ18, עַל חוֹף הַיָּם, מִסְפַּר יוֹשְׁבֶיהָ יוֹתֵר מֵעֲשֶׂרֶת אֲלָפִים אִישׁ, בָּהֶם כִּשְׁלשֶׁת אֲלָפִים נוֹצְרִים וּכְמָאתַיִם יְהוּדִים וְהַיּוֹתֵר מוּשְׁלִימִים.

הָעִיר הַזֹּאת הָיְתָה מְהֻלָּלָה בְּחֻפָּהּ עוֹד בִּימֵי קֶדֶם, וָאֳנִיּוֹת רַבּוֹת בָּאוּ אֵלֶיהָ מֵרָחוֹק, וְהִיא מוּקֶפֶת חוֹמָה בְצוּרָה. דֶּרֶךְ הַחוֹף יוֹצִיאוּ תְבוּאָה רַבָּה הַמּוּבָאָה מֵהַכְּפָרִים מִסָּבִיב וּמִבִּקְעַת יִזְרְעֵאל וְגַם מִהָרֵי חַוְּרָן. בַּזְּמַן הָאַחֲרוֹן הֵחֵלּוּ לִנְטֹעַ שָׁם פַּרְדֵּסִים.

ב) חֵיפָה19, עַל שְׂפַת הַיָּם לְרַגְלֵי הַר הַכַּרְמֶל, מִסְפַּר יוֹשְׁבֶיהָ כַּעֲשֶׂרֶת אֲלָפִים אִישׁ, בָּהֶם יְהוּדִים כְּאֶלֶף וַחֲמֵשׁ מֵאוֹת, וּכְאַרְבַּעַת אֲלָפִים נוֹצְרִים וְהַיּוֹתֵר מוּשְׁלִימִים. בָּהּ יוֹשֵׁב שַׂר מָחוֹז.

חוֹף חֵיפָה הוּא הַיּוֹתֵר טוֹב בְּכָל הָאָרֶץ לְמַעֲמַדאֳנִיּוֹת. דֶּרֶךְ הַחוֹף יוֹצִיאוּ תְבוּאָה רַבָּה וְשֶׁמֶן, גַּם בָּתֵּי מַעֲשֵׂב סַבּוּן יֵשׁ שָׁם רַבִּים, וּמִסָּבִיב לָעִיר יֵשׁ עֲצֵי פְרִי רַבִּים, וּבִפְרַט עֲצֵי זָיִת.

ג) צוֹר20, עַל שְׂפַת הַיָּם. מִסְפַּר יוֹשְׁבֶיהָ כַּחֲמֵשֶׁת אֲלָפִים אִישׁ, חֶצְיָם מוּשְׁלִימִים, וְחֶצְיָם נוֹצְרִים, יְהוּדִים מְעַטִּים בָּהּ. בָּהּ יוֹשֵׁב שַׂר מָחוֹז.

צוֹר הַמְהֻלָּלָה בַּגּוֹיִם בְּיָמִים מִקֶּדֶם הִנֶּהָ עַתָּה עִיר קְטַנָּה וַַעֲנִיָּה, סַחְרָהּ עֲזָבָהּ, דֶּרֶךְ הַחוֹף יוֹצִיאוּ מְעַט כִּתְנָה וְטַבַּק.

הָעִיר הַזֹּאת נִבְנְתָה עַל יְדֵי הוֹרְדָס לִכְבוֹד הַקֵּיסַר אוֹגוֹסְטוֹס, וַיִּבֶן בָּהּ הֵיכָלוֹת וְאַרְמנוֹת. בָּהּ הָיָה מוֹשַׁב נְצִיבֵי יְהוּדָה. אַחֲרֵי הַחֻרְבָּן הָיְתָה עִיר מֶרְכָּז לְהַדַּת הַנּוֹצְרִית. נוֹסְעֵי הַצְּלָב לְכָדוּהָ וְאַחֲרֵי כֵן נֶהֶרְסָה עַד הַיְסוֹד עַל יְדֵי אַחַד הַשּׁוּלְטָנִים וּבַָעֵת הָאַחֲרוֹנָה הִתְיַשְּׁבוּ בָהּ מוּשְׁלִימִים. סָבִיב לָעִיר תִּשְׂתָּרֵעַ כִּכַָּר נִרְחָבָה מְלַאָה אַבְנֵי חֳרָבוֹת וּשְׂרִידֵי עַמּוּדֵי שַׁיִּשׁ נִפְלָאִים, גַּם שְׁאֵרִית קִיר הַחוֹף תֵּרָאֶה עוֹד שָׁם.


II. מֵעֵבֶר לַיַּרְדֵּן מִזְרַָחָה.

1) בְּפֶלֶךְ חַוְרָן:

א) שיֵׁךְ סַעַד, עִיר קְטַנָּה וּבָהּ יוֹשֵׁב הַפֶּחָה.

2) בְּפֶלֶךְ כֶּרֶךְ:

א) כֶּרֶךְ הִיא קִיר מוֹאָב, מִסְפַּר יוֹשְׁבֶיהָ כַּחֲמֵשֶׁת אֲלָפִים, מֵהֶם כְּאֶלֶף נוֹצְרִים וִיהוּדִים אֲחָדִים, וְהַנּוֹתָרִים מוּשְׁלִימִים. בָּהּ יוֹשֵׁב הַפֶּחָה.

ב) סַלְטְ, הִיא רָמַת גִּלְעָד, מִסְפַּר יוֹשְׁבֶיהָ כְּאַרְבַּעַת אֲלָפִים, מֵהֶם כִּשְׁמוֹנֶה מֵאוֹת נוֹצְרִים, וִיהוּדִים אֵין שָׁם כְּלָל.

הָאָרֶץ מִסָּבִיב דְּשֵׁנָה וּפוֹרִיָּה מְאֹד, יֵשׁ שָׁם גַּם עֲצֵי פְרִי טוֹב לְמַכְבִּיר.

וְאֵלֶּה הֵן מוֹשְׁבוֹת הַיְּהוּדִים בְּאֶרֶץ יִשְׂרָאֵל:

I. בְּפֶלֶךְ יְרוּשָׁלַיִם.

1) בִּמְחוּז יְרוּשָׁלַיִם:

א) מוֹצָא אוֹ קוֹלוֹנִיָּה, שְׁלשֶׁת רִבְעֵי שָׁעָה מִירוּשָׁלַיִם יָמָּה עַל דֶּרֶךְ יָפוֹ, כְּפַר עַרְבִי וּבוֹ כְרָמִים וְגַנֵּי יֶרֶק אֲשֶׁר שְׁנֵי מַעְיְנוֹת מַיִם יַשְׁקוּם. לַיְּהוּדִים

שָׁם כְּשֵׁשׁ מֵאוֹת דּוּנַם 26אֲדָמָה. הָאֲדָמָה הִיא נַחֲלַת אַרְבָּעָה עָשָׂר אֲנָשִׁים וְאַךְ חֶצְיָם יוֹשְׁבִים שָׁם, שְׁלִיש שֶׁטַח אַדְמַת הַיְּהוּדִים נָטוּעַ בְּכַרְמֵי גָפֶן נוֹסְדָה בִּשְׁנַת א' תתכ"ה.

ב) אַרְטוּף – אֶרֶץ צוּף – עַל דֶרֶךְ יָפוֹ־יְרוּשָׁלַיִם, שֶׁטַח אַדְמָתָהּ כַּחֲמֵשֶׁת אֲלָפִים דּוּנַם, חֵלֶק גָּדוֹל מֵהָאֲדָמָה טוֹב לִזְרִיעָה. אֶת הָאֲדָמָה קָנוּ חֶבֶר אַחֵינוּ יְלִידֵי בוֹלְגַרְיָה וַאֲחָדִים מֵהֶם בָּאוּ לְהִתְיַשֵׁב בָּהּ. מַצָּבָם אֵינֶנּוּ טוֹב, כִּי חֲסֵרִים לָהֵֶם דְּבָרִים רַבִּים עִקְרִים הַדְּרוּשִׁים לָאִכָּר, הָאַקְלִים שָׁם טוֹב. הַמּוֹשָׁבָה נוֹסְדָה בִּשְׁנַת א' תתכ"ו.

2) בִּמְחוֹז יָפוֹ:

א) בֵּית סֵפֶר לַעֲבוֹדַת הָאֲדָמָה. נוֹסַד בִּשְׁנַת א' תת"א עַל יְדֵי קַרְל נֶטֶר בִּפְקֻדַּת חֲבֵרָיו כָּל יִשְׂרָאֵל חֲבֵרִים. שֶׁטַח אַדְמָתָהּ הוּא כְּאַלְפַּיִם וְשֵׁשׁ מֵאוֹת דּוּנַם, כְּמֵאָה וַחֲמִשִּׁים אֶלֶף גֶּפֶן בָּהּ, כְּאַלְפַּיִם עֲצֵי שְׁקֵדִים וּכְמִסְפָּר הַזֶּה עֲצֵי תַפּוּחֵי זָהָב וּכְאֶלֶף עֲצֵי פְרִי שׁוֹנִים. בִּנְיָנִים רַבִּים יָפִים בָּהּ גַם יֶקֶב חָצוּב בַּסֶּלַע. וְשֵׁשׁ בְּאֵרוֹת מַיִם יַשְׁקוּ אֶת גַּנֶּיהָ.

ב) רִאשׁוֹן לְצִיּוֹן, נוֹסְדָה עַל יְדֵי חֶבְרַת יְחִידִים שֶׁבָּאוּ מֵאֶרֶץ הַצָּפוֹן בִּשְׁנַת א' תתנ"ג כְּשָׁעָה וָחֵצִי מִיָּפוֹ נֶגְבָּה מִזְּרָחָה, שֶׁטַח אַדְמָתָהּ כְּשִׁבְעַת אֲלָפִים דּוּנַם, וּמִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ מַגִּיעַ עַד שֵׁשׁ מֵאוֹת. בְּנֵי הַמּוֹשָׁבָה הַזֹּאת עוֹסְקִים בְּעִקָר בִּכְרָמִים וְחָמֵשׁ שִׁשִּׁיּוֹת אַדְמָתָהּ מְכֻסּוֹת בְּכַרְמֵי גָפֶן, וְיֵשׁ בָּהּ גַּם עֲצֵי פְרִי אֲחֵרִים. בְּרִאשׁוֹן לְצִיוֹן יֵשׁ יֶקֶב גָּדוֹל וּרְחַב יָדַיִם מְאֹד, שָׁם יֵשׁ גַּם “בֵּית־עָם” אֲשֶׁר בָּנוּ בְנֵי הַמּוֹשָׁב לְבֵית הַצִּבּוּר.

ג) עֵמֶק שׁוֹשַׁנִּים אוֹ לַאדִי־חַנִּין, כִּשְׁלשֶׁת רִבְעֵי שָׁעָה מֵרִאשׁוֹן לְצִיוֹן נֶגְבָּה, נוֹסְדָה עַל יְדֵי יְחִידִים בִּשְׁנַת א' תתי"ג; שֶׁטַח אַדְמָתָהּ כְּאֶלֶף וַחֲמֵשׁ מֵאוֹת דּוּנַם, הָאִכָּרִים עוֹסְקִים בַּכְּרָמִים, יֵשׁ בָּהּ גַּם פַּרְדֵּסִים אֲחָדִים, פִּי אַדְמָתָהּ כִּמְעַט כֻּלָּהּ רְאוּיָה לְמַטַּע פַּרְדֵּסִים, אֲחָדִים מֵהָאִכָּרִים עוֹסְקִים גַּם בְּגִדּוּל דְּבוֹרִים וְדִבְשָׁם מְשֻׁבָּח מְאֹד. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ עוֹלֶה לְמֵאָה.

ד) רחֹבוֹת (דּוּרַאן־שַׁעֲרַיִם) כַּחֲצִי שָׁעָה מֵעֵמֶק שׁוֹשַׁנִּים נֶגְבָּה מִזְרָחָה, וּכְשָׁעָה מֵרַמְלָה הָעִיר, נוֹסְדָה עַל יְדֵי יְחִידִים בִּשְׁנַת א' תתב"א, שֶׁטַח אַדְמָתָהּ עֲשֶׂרֶת אֲלָפִים וְשֵׁשׁ מֵאוֹת דּוּנַם, עַל חֲצִי אַדְמַת הַמּוֹשָׁב נְטוּעִים כְּרָמִים, אַךְ יֵשׁ בָּהּ גַם עֲצֵי זֵיתִים וּשְׁקֵדִים בְּמִסְפָּר הָגוּן. רַבִּים מִבַּעֲלֵי הָאֲחֻזּוֹת בִּרְחֹבוֹת יוֹשְׁבִים עוֹד בְּחוּץ לָאָרֶץ. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ עוֹלֶה לְמָאתַיִם וּשְׁלשִׁים.

ה) עֶקְרוֹן אוֹ מַזְכֶּרֶת בִּתְיָה, כִּשְׁלשֶׁת רִבְעֵי שָׁעָה מֵרְחֹבוֹת נֶגְבָּה מִזְרָחָה, נוֹסְדָה עַל יְדֵי הַנָּדִיב בִּשְׁנַת א' תתט"ו, שֶׁטַח אַדְמָתָהּ כְּאַרְבַּעַת אֲלָפִים וּמָאתַיִם דּוּנַם, וְכֻלָּהּ דְּשֵׁנָה וְטוֹבָה לְמִזְרַע תְּבוּאוֹת. בַּשָּׁנִים הָאַחֲרוֹנוֹת נָטְעוּ שָׁם עֲצֵי פְרִי רַבִּים וְשׁוֹנִים, וַעֲצֵי הַזַּיִת הָרַבִּים אֲשֶׁר נָטְעוּ נוֹתְנִים תִּקְוָה טוֹבָה. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ עוֹלֶה לְמָאתַיִם וְעֶשְׂרִים.

ו) פֶּתַח תִּקְוָה וִיְהוּד, נוֹסְדוּ עַל יְדִֵי יְחִידִים בִּשְׁנַת א' תתי"ג, כִּשְׁתֵּי שָׁעוֹת מִיָּפוֹ צָפוֹנָה מִזְרָחָה. מְיַסְּדֵי פֶתַח תִּקְוָה בָּנוּ לָהֶם בָּתִּים בִּיהוּד מִפְּנֵי הָאַוִּיר הַטּוֹב אֲשֶׁר שָׁם, אַךְ אַחֲרֵי כֵן הָכְרְחוּ לַעֲזֹב אֶת הַמָּקוֹם הַזֶּה הָרָחוֹק מִשְּׂדוֹתֵיהֶם וַיִּבְנוּ לָהֶם בָּתִּים עַל אַדְמַת פֶּתַח תִּקְוָה. שֶׁטַח אַדְמָתָהּ כְּאַרְבָּעָה עָשָׂר אֶלֶף דּוּנַם, רֶבַע הָאֲדָמָה מְכֻסֶּה בְּכַרְמֵי גֶפֶן, וְעַל שֶׁטַח אֶלֶף דּוּנַם יֵשׁ שָׁם פַּרְדֵּסִים, וְעַל שֶׁטַח אֲשֶׁר כָּזֶה נֹטְעִים גַּם זֵיתִים, וְיֶתֶר הָאֲדָמָה הִיא אַדְמַת זְרִיעָה. מִסְפַּר יוֹשְׁבֶיהָ מַגִּיעַ עַד שְׁמֹנֶה מֵאוֹת וַחֲמִשִּׁים נָפֶשׁ.

2) בִּמְחוֹז עַזָּה.

א) גְּדֵרָה, כִּשְׁלשֶׁת רִבְעֵי שָׁעָה מֵעֶקְרוֹן נֶגְבָּה מַעֲרָבָה, נוֹסְדָה עַל יְדֵי חֶבְרַת חוֹבְבֵי צִיּוֹן בְּאֶרֶץ הַצָּפוֹן בִּשְׁנַת א' תתט"ו, שֶׁטַח אַדְמָתָהּ כִּשְׁלשֶׁת אֲלָפִים וּשְׁלשׁ מֵאוֹת דּוּנַם. עִיקַר עֲבוֹדַת הָאִכָּרִים מַטַּע הַכְּרָמִים, אַךְ יֵשׁ לְכָל אֶחָד גַּם חֵלֶק קָטָן אַדְמַת זְרִיעָה וְגַם פְּרִי עֵץ אֲחֵרִים שׁוֹנִים יֶשְׁנָם שָׁם. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ עוֹלֶה לְמֵאָה וְעֶשְׂרִים.

ב) קַסְטִינָה הִיא בְּאֵר טוּבִיָּה, כִּשְׁתֵּי שָׁעוֹת מִגְּדֵרָה נֶגְבָּה, נוֹסְדָה עַל יְדֵי חֶבְרַת חוֹבְבֵי צִיּוֹן בִּשְׁנַת א' תתכ"ז. שֶׁטַח אַדְמָתָהּ כְּשֵׁשֶׁת אֲלָפִים דּוּנַם וְהִיא טוֹבָה וּדְשֵׁנָה וּרְאוּיָה לְמִזְרַע תְּבוּאוֹת. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ הוּא מֵאָה עֶשְׂרִים וְחָמֵשׁ. כָּל הָאִכָּרִים עוֹסְקִים בִּזְרִיעָה.


II. בְּפֶלֶךְ שְׁכֶם,

בִּמְחוֹז תֵּל כֶּרֶם נִקְנְתָה זֶה לֹא כְבָר חֶלְקַת אֲדָמָה בַּת שִׁבְעַת אֲלָפִים וַחֲמֵשׁ מֵאוֹת דּוּנַם הִיא כְּפַר סָבָא, אַךְ עוֹד לֹא הוֹשִׁיבוּ עָלֶיהֵ אִכָּרִים.


III. בְּפֶלֶךְ עַכּוֹ.

1) בִּמְחוֹז חֵיפָה.

א. חִידֶרָה, נוֹסְדָה עַל יְדֵי יְחִידִים בִּשְׁנַת א' תת"ךְ עַל שְּׂפַת נָהָר הַיָּרוּק, כַּחֲצִי שָׁעָה מֵהַיָּם: שֶׁטַח אַדְמָתָהּ כִּשְׁלשִׁים אֶלֶף דּוּנַם, רֹב אַדְמָתָהּ טוֹבָה לִזְרִיעָה, וְחֵלֶק רַב מִמֶּנָּה רְאוּיָה גַּם לְמַטַּע פַּרְדֵּסִים. הַבִּצּוֹת אֲשֶׁר בְּאַדְמָתָהּ וּמִסָּבִיב לָהּ וַאֲשֶׁר הָיוּ לִמְקוֹר מַחֲלוֹת רַבּוֹת הוֹלְכוֹת וּמִתְיַבְּשׁוֹת עַל יְדֵי הַָעֵצִים הָרַבִּים אֲשֶׁר נַָטְעוּ עַל שְׂפוֹת הַבִּצּוֹת וְהָאַוִּיר הוּטַב כְבָר מְעָט, וּבְהֵעָשׂוֹת הַתְּעָלוֹת וְהַבִּצּוֹת תִּיבַשְׁנָה וְיָכְלוּ כָל בְּעָלֶיהָ לְהִתְיַשֵּׁב בָּהּ, וְהַמָּקוֹם טוֹב מְאֹד לְגִדּוּל בְּהֵמוֹת. כְּאֶלֶף דּוּנַם מֵאַדְמָתָהּ מְכֻסִּים בְּכַרְמֵי גֶפֵֶן, כְּמָאתַיִם דּוּנַם פַּרְדֵּסִים, גַּם יֵשׁ בָּהּ עֲצֵי פְרִי אֲחֵרִים. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ מֵאָה וַחֲמִשִּׁים.

ב. זִכְרוֹן יַעֲקֹב, נוֹסְדָה עַל יְדֵי יְחִידִים, שֶׁעָלוּ מֵרוּמֶנְיָה בִּשְׁנַת א' תתי"ב, עַל שֵׁן סֶלַע גָּבהַּ הַנִּשְׁקָף מִקַּרְנֵי הַר הַכַּרְמֶל. שֶׁטַח אַדְמָתָהּ כְּשִׁבְעַת אֲלָפִים דּוּנַם, אֲדָמָה רְאוּיָה לִזְרִיעָה מְעַטָּה מְאֹד, וְכִשְׁלשֶׁת רִבְעֵי אַדְמָתָהּ מְכֻסִּים בַּכְּרָמִים, אַךְ יֵשׁ בָּהּ גַּם עֲצֵי פְרִי אֲחֵרִים, גַּם כַּוְרוֹת דְּבוֹרִים נִמְצָאוֹת שָׁם. מִסְפַּר יוֹשְׁבֶיהָ קָרוֹב לִתְשַׁע מֵאוֹת נָפֶשׁ. בְּזִכְרוֹן יַעֲקֹב יֵשׁ יֶקֶב גָּדוֹל מְאֹד.

ג. שְׁוֵיָה, מֵאִיר שְׁוֵיָה, נוֹסְדָה עַל יְדֵי הַנָּדִיב בִּשְׁנַת א' תתכ"ב, כַּחֲצִי שָׁעָה מִזִכְרוֹן יַעֲקֹב מִזְרָחָה צָפוֹנָה עַל רֹאשׁ הָר. שֶׁטַח אַדְמָתָהּ הוּא כִּשְׁמנָה אֲלָפִים וַחֲמֵשׁ מֵאוֹת דּוּנַם. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ עוֹלֶה לְמֵאָה וּשְׁלשִׁים. רֶבַע הָאֲדָמָה מְכֻסֶּה בִּגְפָנִים וְרֶבַע בַּעֲצֵי פְרִי אֲחֵרִים.

ד. אֵם אֵל גָּמָל – בַּת שְׁלמֹה – נוֹסְדָה עַל יְדֵי הַנָּדִיב בִּשְׁנַת א' תת"ךְ כְּשָׁעָה מִזִּכְרוֹן יַעֲקֹב מִזְרָחָה, שֶׁטַח אַדְמָתָהּ אַלְפַּיִם וַחֲמֵשׁ מֵאוֹת דּוּנַם, וְרֻבָּהּ מְכֻסָּה בַּכְּרָמִים. מִסְפַּר יוֹשְׁבֶיהָ כְּמֵאָה וְעֶשְׂרִים.

ה. אַטְלִיט, בְּאֶמְצַע הַדֶּרֶךְ בֵּין זִכְרוֹן יַעֲקֹב וְחֵיפָה עוֹמְדִים עַל חוֹף הַיָּם שְׂרִידֵי מִבְצָר עַתִּיק חָזָק וְעַל יָדוֹ אַדְמַת מִישׁוֹר טוֹבָה וּפוֹרִיָה. הָאֲדָמָה הַזֹּאת, אֲשֶׁר שִׁטְחָהּ מַגִּיעַ עַד עֲשֶׂרֶת אֲלָפִים דּוּנַם, קָנָה הַנָּדִיב, אַךְ עַד הַיּוֹם לֹא הוֹשִׁיבוּ אֶת הַנַּחֲלָה הַזֹּאת.

ו. גַּם בִּכְפַר תִּנְתּוּרָה – דּוּר – אֲשֶׁר עַל שְׂפַת הַיָּם יֵשׁ נַחֲלַת אֲדָמָה הַשַּׁיֶּכֶת לַנָּדִיב וְשִׁטְחָהּ כְּאַרְבַּע מֵאוֹת דּוּנַם.


2) בִּמְחוֹז טְבַרְיָה.

א) סֵגֵרָה, נוֹסְדָה בִּשְׁנַת א' תתב“ט עַל יְדֵי חֶבְרַת יק”א, שֶׁטַח אַדְמָתָהּ קָרוֹב לְאַרְבָּעִים אֶלֶף דוּנַם, כִּשְׁנֵי שְׁלִישִׁית מִמֶּנָה טוֹבָה לִזְרִיעָה, וְהַמּוֹתָר אַדְמַת טָרָשִׁים הַטּוֹבָה לִנְטִיעָה. מְיַסְּדֶיהָ עָשׂוּ שָׁם אֲחֻזַּת נַחֲלָה 27– לַעֲבוֹדַת הַָאֲדָמָה וּכְשִׁבְעִים פּוֹעֲלִים יְהוּדִים עוֹבְרִים שָׁם. בַּיָּמִים הָאַחֲרוֹנִים קָנוּ שָׁם עוֹד אֲחֻזּוֹת גְּדוֹלוֹת עַל יַד אַדְמַת הַמּוֹשָׁבָה הַזֹּאת.

ב) בִּכְפַר חַטִּין – הַצִּדִּים – יֵשׁ פַּרְדֵּס אֶתְרוֹגִים גָּדוֹל אֲחֻזַּת כּוֹלֵל וַרְשָׁה בִּירוּשָׁלַיִם, וּבֲזְמַן הָאַחֲרוֹן קָנָה שָׁם אֶחָד מֵאַחֵינוּ אֲחֻזַּת נַחֲלָה גְדוֹלָה שֶׁשֶּׁטַח אַדְמָתָהּ יוֹתֵר מֵאַלְפַּיִם דּוּנַם. ג) הַמּוֹשָׁבוֹת הַחֲדָשׁוֹת. בַּיָּמִים הָאַחֲרוֹנִים קָנָה הַנָּדִיב נַחֲלָאוֹת אֲחָדוֹת לְהוֹשִׁיב בָּהֶן אִכָּרִים וּבְנֵי אִכָּרִים מִמּוֹשָׁבוֹת אֲחֵרוֹת שֶׁבָּהֶן צַר הַמָּקוֹם מֵחֹסֵֶר אֲדָמָה וְאֵלֶּה הֵן: א) עַבַּבִּיָּה וּמִלְחַמִּיָּה עַל שְׂפַת הַיַּרְדֵּן כְּשָׁעָה נֶגְבָּה מִטְּבַרְיָה: שֶׁטַח אַדְמַת שְׁתֵּי הַנַּחְלָוֹת הָאֵלֶּה כְּתִשְׁעָה עָשָׂר אֶלֶף דּוּנַם. ב) דַּלַּיְקָה וְיַמַּה כִּשְׁעָתַיִם מִטְּבַרְיָה דְרוֹמִית מַעֲרָבִית; שֶׁטַח אַדְמַָתָן כִּשְׁמוֹנָה עָשָׂר אֶלֶף דּוּנַם. ג) לֻבִּיָּה כְּשָׁעָה וָחֵצִי מִטְּבַרְיָה מַעֲרָבָה, שֶׁטַח אַדְמָתָהּ כַּחֲמִשָּׁה עָשָׂר אֶלֶף דּוּנַם. ד) מַסְאֶה לְרַגְלֵי הַר תָּבוֹר מִצָּפוֹן. שֶׁטַח אַדְמָתָהּ כְּשִׁשָּׁה אֶלֶף דּוּנַם. וּכְבָר הֵחֵלּוּ לִבְנוֹת שַָׁם בָּתִּים וּכְפָרִים לָאִכָּרִים הַחֲדָשִׁים אֲשֶׁר הִתְיַשְּׁבוּ שָׁם.

3) בִּמְחוֹז צְפַת.

א. עֵין־זֵיתִים, בְּיַרְכְּתֵי הָהָר כַּחֲצִי שָׁעָה מִצְּפַת צָפוֹנָה, נוֹסְדָה עַל יְדֵי חֶבְרָה אַחַת מֵאֶרֶץ הַצָּפוֹן בִּשְׁנַת י“ח תתכ”ב, שֶׁשֶׁטַח אַדְמָתָהּ חֲמֵשֶׁת אֲלָפִים וּשְׁבַע מֵאוֹת דּוּנַם, וְיֵשׁ בָּהּ מַעְיְנוֹת מַיִם טוֹבִים. הָאֲדָמָה אַדְמַת סִיד וְלָכֵן לֹא הִצְלִיחוּ הַכְּרָמִים הָרַבִּים שֶׁנָּטְעוּ שָׁם. עַתָּה יוֹשְׁבִים שָׁם פּוֹעֲלִים יְהוּדִים אֲחָדִים וּמַשְׁגִּיחִים עַל הָעֲבוֹדָה, כִּי הֵחֵלּוּ לְהַרְבּוֹת שָׁם עֲצֵי זַיִת. גַּם זוֹרְעִים אֶת רֶבַע הָאֲדָמָה הָרַָאוּי לִזְרִיעָה. מִסְפַּר יוֹשְׁבֶיהָ כְּאַרְבָעִים נָפֶשׁ.

ב. רֹאשׁ פִּנָּה, כְּשָׁעָה מִצְפַת צָפוֹנָה מִזְרָחָה עַל הַדֶּרֶךְ הַהוֹלֶכֶת לְדַמֶּשֶׂק. הַמּוֹשָׁבָה הַזֹּאת נוֹסְדָה עַל יְדֵי יְחִידִים מִיהוּדֵי רוּמֶנְיָה בִּשְׁנַת א' תתי"ב. שֶׁטָח אַדְמָתָהּ עַתָּה הוּא אַרְבָּעִים וּשְׁלשָׁה אֶלֶף דּוּנַם; שְׁלִישׁ הָאֲדָמָה טוֹבָה לִזְרִיעָה, שְׁלִיש הוּא אַדְמַת הָרִים הָרְאוּיִם לְמִרְעֵה הָעֲדָרִים וְהַשְּׁלִישׁ הוּא אַדְמַת מַטָּעִים שׁוֹנִים. בִּפְרַט הִרְבּוּ שָׁם לִנְטֹעַ עֲצֵי תות לְגִדּוּל הַמֶּשִׁי. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ עוֹלֶה לַחֲמֵשׁ מֵאוֹת וַחֲמִשִּׁים בְּעֶרֶךְ. שָׁם יֵשׁ בֵּית מַטְוֶה מֶשִׁי גָּדוֹל הָעוֹבֵד בְּכֹחַ מְכוֹנוֹת הַקִּיטוֹר וּכְמֵאָה אִישׁ עוֹבְדִים בַּבַּיִת הַזֶּה. בַּזְּמַן הָאַחֲרוֹן הֵבִיאוּ שָׁמָּה גַּם נוּלִים 28לְלַמֵּד אֶת הָאִכָּרִים וּנְשֵׁיהֶם וּבְנוֹתֵיהֶם אֶת מַעֲשֵׂה הָאֵֶרֶג, וְהָאָּכִרים עוֹסְקִים שָׁם גַּם בְּמִזְרַע הַתַּבַּק.

ג. מִשְׁמַר הַיַּרְדֵּן, כְּשָׁעָה מֵרֹאשׁ פִּנָּה צָפוֹנָה מִזְרָחָה עַל שְׂפַת הַיַּרְדֵן, נוֹסְדָה עַל יְדֵי אֲחָדִים מִיּוֹשְׁבֵי הָאָרֶץ בִּשְׁנַת א' תתכ"א. שֶׁטַח אַדְמָתָהּ אַרְבָּעָה אֲלָפִים וַחֲמֵשׁ מֵאוֹת דוּנַם; מִסְפַּר יוֹשְׁבֵֶיהָ כְּשִׁבְעִים וַחֲמִשָּׁה אִישׁ, אַדְמָתָהּ אַדְמַת זְרִיעָה וְכָל אִכָּרֶיהָ עוֹסְקִים בִּזְרִיעַת תְּבוּאָה.

ד. מַחֲנַיִם, נוֹסְדָה בִּשְׁנַת א' תתכ"ט עַל יְדֵי חֶבְרָה אַחַת מִגַּלִּיצִיָּה. שֶׁטַח אַדְמָתָהּ כְּשֵׁשֶׁת אֲלָפִים דּוּנַם וְהִיא גוֹבֶלֶת לְמִשְׁמַר הַיַּרְדֵּן. הַמּוֹשָׁבָה הַזֹּאת לֹא הִשְׁתַּכְלְלָה עוֹד.

ה. יְסֻד הַמַּעֲלָה, יוֹשֶׁבֶת עַל מֵי מֵרוֹם, כְּשָׁעָה וָחֵצִי מֵרֹאשׁ פִּנָּה, נוֹסְדַָה עַל יְדֵי יְחִידִים מֵעוֹלֵי אֶרֶץ הַצָּפוֹן בִּשְׁנַת א' תתנ"ב. שֶׁטַח אַדְמָתָהּ אַחַד עָשָׂר אֶלֶף דּוּנַם, כֻּלָּהּ טוֹבָה לִזְרִיעָה. יֵשׁ בָּהּ גַּם כִּשְׁלשׁ מֵאוֹת דּוּנַם פַּרְדֵסִים, וְכַחֲמֵשׁ מֵאוֹת דּוּנַם מַטָּעֵי עֵץ פְּרִי שׁוֹנִים. מִסְפַּר יוֹשְׁבֶיהָ כְּמָאתַיִם וַחֲמִשִּׁים נָפֶשׁ. בַּמּוֹשָׁבָה הַזֹּאת יֵשׁ בֵּית־מַעֲשֵׂה שְׁמָנִים נוֹתְנִים רֵיחַ הַנַּעֲשִׂים מִפִּרְחֵי עֵצִים שׁוֹנִים.


IV. בְּפֶלֶךְ בַּיְרוּת, בִּמְחוֹז גְּדֵידָה נִמְצֵאת הַמּוֹשָׁבָה:

מְטוּלָה, לְרַגְלֵי הַר חֶרְמוֹן. נוֹסְדָה עַל יְדֵי הַנָּדִיב בִּשְׁנַת א' תתכ"ו. שֶׁטַח אַדְמָתָהּ כִּשְׁלשָׁה עָשָׂר אֶלֶף דּוּנַם. שְׁתֵּי שְׁלִישִׁיּוֹתֶיהָ טוֹבוֹת לִזְרִיעָה. מִסְפַּר יוֹשְׁבֶיהָ כִּשְׁלשׁ מֵאוֹת נָפֶשׁ. הָאִכָּרִים עוֹסְקִים בִּזְרִיעַת תְּבוּאָה וְגַם בְּמַטַּע הַתַּבַּק.


VI בְּפֶלֶךְ חַוְרָן:

א) בְּנֵי יְהוּדָה, מוּל יָם כִּנֶּרֶת, כְּשָׁעָה קֵדְמָה. שֶׁטַח אַדְמָתָהּ שְׁלשֶת אֲלָפִים וַחֲמֵשׁ מֵאוֹת דּוּנַם. הִיא נוֹסְדָה עַל יְדֵי אֲחָדִים מִילִידֵי הָאָרֶץ בִּשְׁנַת א' תתכ"א. אַדְמָתָהּ טוֹבָה וּפוֹרִיָּה. מִסְפַּר נַפְשׁוֹת יוֹשְׁבֶיהָ כַּחֲמִשִּׁים נָפֶשׁ.

ב) נַחֲלָאוֹת גְּדוֹלוֹת אֲשֶׁר קָנָה הַנָּדִיב וְעוֹד לֹא הוֹשִׁיבוּ עֲלֵיהֶן אִכָּרִים וְהֵן: גּוֹלָן, וְסַחַם גּוֹלָן וּבֵיטִימָה. שֶׁטַח אַדְמָתָם יַחַד יוֹתֵר מִמֵּאָה אֶלֶף דּוּנַם.

§ 14.

I .

הַמִּישׁוֹרִים וְהָרָמוֹת וְהָעֲמָקִים מוֹצִיאִים הַרְבֵּה חִטִּים, שְׂעוֹרִים, שֻׁמְשְׁמִין, דּוֹרָה, כַּרְשִׁינִים, עֲדָשִׁים, חִטֵּי תִירָס, פּוֹל, אֲבַטִּיחִים וְעוֹד מִינֵי תְבוּאוֹת וִירַק הַגָּן; וְהֶהָרִים נוֹתְנִים פְּרִי עֲצֵי הַגֶּפֶן, הַזַּית, הַתְּאֵנָה וְהָרִמּוֹן וְעוֹד עֲצֵי פֶרִי. בִּמְקוֹמוֹת הַשְּׁפָלִים יִמָּצְאוּ פַרְדְּסֵי תַפּוּחֵי זָהָב, לִימוֹנִים וְאֶתְרוֹגִים. הַחִטָּה רַבָּה וְטוֹבָה בְּכָל הָאָרֶץ וּבִפְרַט בִּסְבִיבוֹת עַזָּה וּבְכִכַּר הַיַּרְדֵּן וּבְעֵבֶר לַיַּרְדֵּן מִזְרָחָה, וְהוֹבִילוּ חִטִּים מִשָּׁם לְכָל אֶרֶץ עֲרַב וְהוֹצִיאוּ גַם הַרְבֵּה אֶל מִחוּץ לָאָרֶץ דֶּרֵֶךְ יָפוֹ, עַכּוֹ, חֵיפָה, בַּיְרוּתְּ וְעַזָּה.

הַשְּׂעוֹרָה רַבָּה בִּפְרַט בַּשְּׁפֵלָה עַל יַד עַזָּה וּבְכִכַּר הַיַּרְדֵּן.הַגֶּפֶן רַבָּה בְּהָרֵי הַלְבָנוֹן וּבְהָרֵי יְהוּדָה. הַפַּרְדֵּסִים רַבִּים בִּסְבִיבוֹת יָפוֹ, צוֹר וְצִידוֹן: גַּם בִּסְבִיבוֹת עַזָּה, חִידְרָה וּפֶתַח תִּקְוָה הֵחֵלּוּ לִנְטֹעַ פַּרְדֵּסִים.

II.

בֵּין עַנְפֵי הַתַּעֲשִׂיָּה עוֹמְדִים בְּשׁוּרָה הָרִאשׁוֹנָה: א) מַעֲשֵׂה הַסַּבּוּן בִּשְׁכֶם, בִּירוּשָׁלַיִם, בְּרַמְלָה, בְּלוּד, בְּעַזָּה וּבְחֵיפָה. ב) מַעֲשֵׂה שֶׁמֶן שֻׁמְשְׁמִין בְּרַמְלָה, בְּלוּד, בְּיָפוֹ, בְּחֵיפָה, וְהוֹצִיאוּ מִמֶּנּוּ הַרְבֵּה גַּם לְחוּץ לָאָרֶץ. ג) בָּתֵּי הַבַּד לְמַעֲשֵׂה שֶׁמֶן זַיִת נִמְצָאִים כִּמְעַט בְּכָל עָרֵי הָאָרֶץ וְהוֹצִיאוּ מִמֶּנּוּ הַרְבֵּה לְאֵירָפָּה. ד) מַעֲשֵׂה הַיַּיִן בְּרִאשׁוֹן לְצִיוֹן, בְּזִכְרוֹן יַעֲקֹב, בִּירוּשָׁלַיִם, בְּחֶבְרוֹן בְּסָרוֹנָה, וְעוֹד בִּמְקוֹמוֹת אֲחָדִים. ה) מַעֲשֵׂה טְוִיַּת הַמֶּשִׁי בְּרֹאשׁ פִּנָּה. ו) מַעֲשֵׂה הַזְּכוּכִית

בְּחֶבְרוֹן. ז) מַעֲשֵׂה צַעֲצֻעִים מִדַּר30 בְּבֵית לֶחֶם וּבִירוּשָׁלָיִם. ח) בָּתֵּי אֶרֶג לְבַד וְלִירִיעוֹת בִּירוּשָׁלָיִם. ט) טַחֲנוֹת קִיטֹר בֶּעָרִים וּבִכְפָרִים רַבִּים. י) בָּתֵּי מְלָאכָה וְהַתָּכַת בַּרְזֶל בְּיָפוֹ. יא) בָּתֵּי בֻרְסִים לְעִבּוּד עוֹרוֹת לִנְעָלִים בְּיָפוֹ וּלְעִבּוּד עוֹרוֹת לְנֹאדוֹת בְּחֶבְרוֹן. יב) מַעֲשֵׂה מַחְצְלָאוֹת בִּסְבִיבוֹת יָפוֹ וּבְקִרְבַת מֵי־מֵרוֹם. יג) מַעֲשֵׂה הַסַּלִּים בִּמְקוֹמוֹת רַבִּים בָּאָרֶץ.

III.

בֵּין מַעֲדַנֵּי31 הָאָרֶץ עוֹמְדִים בְּשׁוּרָה הָרִאשׁוֹנָה: א) אַבְנֵי שַׁיִשׁ לָבָן וְשָׁחוֹר הַנִּמְַצָא בֶּהָרִים. ב) בַּזֶּלֶת, הִיא אֶבֶן הַבָּשָׁן הַשְּׁחוֹרָה וְהַקָּשָׁה. ג) הַמֶּלַח יִמָּצֵא בֶהָרִים מִנֶּגֶב לְיָם הַמֶּלַח וּבִגְבָאָיו אֲשֶׁר עַל שְׂפַת יָם הַמֶּלַח וְיָם הַתִּיכוֹן. ד) חֲתִיכוֹת גָּפְרִית רַבּוֹת מְכַסּוֹת אֶת שְׂפַת יָם הַמֶּלַח צָפוֹנָה־מַעֲרָבָה. ה) חֵמָר נִמְצָא לָרֹב בְּקַרְקַע יָם הַמֶּלַח וְעַל שְׂפָתוֹ. ו) מַעֲיְנֵי מֵי מַדְעֶן אוֹ יֵמִים נִמְצָאִים עַל יַד טְבַרְיָה הֵם חַמֵּי טְבַרְיָה הַיְדוּעִים עוֹד מִיָּמִים קַדְמוֹנִים. הַמַּיִם בּוֹעֲרִים וְרֵיחַ גָּפְרִית לָהֵֶם וְטוֹבִים הֵם לְמַחֲלוֹת רַבּוֹת שׁוֹנוֹת. (לְפִי עֵדוֹת הַחֲכָמִים אֵינָם נוֹפְלִים הַמַּיִם בְּטוּבָם מִמֵּי כַּרְלְסְבַּד הַיְדוּעִים, וְיַָכֹל יָכְלוּ אֵפוֹא לִהְיוֹת לִמְקוֹר חַיִּים לְאַלְפֵי מִשְׁפָּחוֹת). גַּם עַל שְׂפַת נְהַר יַרְמוּךְ יֵשׁ כְּעֶשְׂרִים יַמִּים וּמֵימֵיהֶם טוֹבִים עוֹד מִמֵּי חַמֵּי טְבַרְיָה. ז) אַבְנֵי הַגִּיר רַבּוֹת מְאֹד בָּאָרֶץ. ח) גַּם אַדְמַת חֹמֶר יֵשׁ לָרֹב בָּאָרֶץ וְעָשׂוּ מֵהֶם כְּלֵי יוֹצֵר, אַךְ בָּתֵּי חֲרשֶׁת לִלְבֵנִים וְלִרְעָפִים לֹא נִבְנוּ בָאָרֶץ אִם כִּי הַסּוֹחֲרִים מַרְבִּים לְהָבִיא אֶת חָמְרֵי הַבִּנְיָן הָאֵלֶּה מֵאֵירָפָּה.

VII.

בְּאֶרֶץ הַגִּלְעָד רַבִּים מְאֹד עֶדְרֵי הַצֹּאן, וְהַבִּידוּאִים הַמִּתְגּוֹרְרִים שָׁם עוֹסְקִים בְּגִדּוּלָם, גַּם עֶדְרֵי גְמַלִּים נִמְצָאִים שָׁם לָרֹב. הַבָּקָר לֹא רַב בָּאָרֶץ.

עֶדְרֵי הַתּוֹא נִמְצָאִים בְּבִצּוֹת אֲשֶׁר בְּכִכַּר הַיַּרְדֵּן וְהַשָּׁרוֹן.


§ 15.

I. הַשָּׁנָה מִתְחַלֶּקֶת בְּאֶרֶץ יִשְׂרָאֵל לִשְׁתֵּי עִתִּים: לִימֵי הַחַמָּה – הַקַּיִץ, וְלִימֵי הַגְּשָׁמִים – הַחֹרֶף. הַגְּשָׁמִים יוֹרְדִים אַךְ בַּחֹרֶף, בְּחֶשְׁוָן יָחֵלּוּ הַגְּשָׁמִים הָרִאשׁוֹנִים, הוּא הַיּוֹרֶה אֲשֶׁר יַרְוֶה אֶת הָאֲדָמָה וְיַרְטִיבָהּ וְיַכְשִׁירֶהָ לַחֲרִישָׁה, אַחֲרֵי כֵן יָבֹאוּ יָמִים בְּהִירִים וְצַחִים וּנְעִימִים וְאָז יֵצֵא הָאִכָּר לְפָעֳלוֹ בַּשָּׂדֶה וְחָרַש אֶת אַדְמָתוֹ לִזְרַע הַחֹרֶף, אַחֲרֵי כֵן בָּאִים עוֹד הַפַּעַם גְּשָׁמִים יוֹתֵר מְרֻבִּים וְיוֹתֵר עַזִּים וְרוּחַ קָרַָה וְחֲזָקָה, וְאַחֲרֵיהֶם עוֹד הַפַּעַם אַוִּיר טוֹב וְנָעִים וְיָמִים בְּהִירִים וְכֵן פְּעָמִים אֲחָדוֹת חֲלִיפוֹת. בְּרֵאשִׁית נִיסָן בְּעֶרֵֶךְ יֵרְדוּ הַגְּשָׁמִים הָאַחֲרוֹנִים הוּא הַמַּלְקוֹשׁ הַיּוֹרֵד בְּזַעַף וְזַעַם וּמַרְטִיב אֶת הַשִּׁבֳּלִים, וְאַז יִזְרַע הָאִכָּר אֶת תְּבוּאוֹת הַקַּיִץ. אַחֲרֵי הַמַּלְקוֹשׁ יָבוֹאוּ יְמֵי הַקַּיִץ הָרִאשׁוֹנִים הַנְּעִימִים מְאֹד, הֵם יְמֵי הָאָבִיב, וּבַקַּיִץ שׁוֹרֵר חֹם גָדוֹל אֲשֶׁר יֶחֱזַק בְּאַחֲרִית יְמֵי הַקַּיִץ, יְמֵי הָאָסִיף.

II.

הָרוּחוֹת בָּאוֹת בָּאָרֶץ עַל פִּי רֹב בִּזְמַנִּים קְבוּעִים, וְטֶבַע כַָּל רוּחַ הוּא לְפִי הַמָּקוֹם אֲשֶׁר מִשָּׁם תָּבוֹא. רוּחַ קָדִים תָּבוֹא מִמִּדְבַּר עֲרַב, וְהִיא עַזָּה וְקָשָׁה, מְיַבֶּשֶׁת הַצְּמָחִים וְהַמַּיִם, וְהִיא מְצוּיָה בְּעֵת הָאָסִיף, בַּחֹרֶף וּבָאָבִיב, הָרוּחַ הַזֹּאת נִקְרֵאת גַּם בְּשֵׁם “שַׁרְקִיָּה”. רוּחַ צָפוֹן קָרָה נוֹשֶׁבֶת כְּאַרְבָּעִים יוֹם בַּשָּׁנָה, וְרוּחַ יָם לַחָה וּמְבִיאָה גֶשֶׁם בַּחרֶף נוֹשֶׁבֶת כַּחֲמִשִּׁים יוֹם בַּשָּׁנָה.

שְׁתֵּי הָרוּחוֹת הָאֵלֶּה מַרְבּוֹת לִנְשֹב בִּימֵי הַקַּיִץוּמְצַנְנוֹת מְעַט אֶת חֹם הָאַוִּיר, וְרוּחַ תֵּימָן הַנּוֹשֶׁבֶת בַּיָּמִים שֶׁבֵּין פֶּסַח לְשָׁבוּעוֹת בְּסִירוֹגִין הִיא חַמָּה וִיבֵשָׁה: אוּלָם גַּם בְּיָמִים הַיּוֹתֵר חַמִּים יִקְטַן הַחֹם אַחַר הַצָּהֳרָיִם, כִּי אָז תִּשֹּׁב בְּכָל יוֹם רוּחַ יָם חֲרִישִׁית, וּבַלַּיְלַָה יַעֲלֶה הַטַּל וִיכַסֶּה אֶת הָאָרֶץ וִיצַנֵּן אֶת הָאַוִּיר.

III.

בְּאֶרֶץ יִשְׂרָאֵל יֵשׁ בִּצּוֹת גְדוֹלוֹת וּקְטַנּוֹת וְרֻבָּן עַל שְׂפַת הַיָּם הַתִּיכוֹן בַּשָּׁרוֹן וְגַם בַּשְּׁפֵלָה. רֹב הַבִּצּוֹת אֵינָן עֲמֻקּוֹת וְהֵן נִהְיוּ בְּמֶשֶׁךְ הַזְּמָן מְעַט מְעַט מִפְּנֵי הָעֲזוּבָה אֲשֶׁר רָבְתָה בָאָרֶץ מִזְּמַן קָדוּם, כִּי בְמֶשֶׁךְ דּוֹרוֹת רַבִּים לֹא עָבְדוּ אֶת הָאֲדָמָה כָּרָאוּי, וְכֹה נִקְבְּצוּ הַמַּיִם וְעָמְדוּ בִּמְקוֹמוֹת הַשְּׁפָלִים וְהָיוּ לְבִצּוֹת, וְגַם גְּדוֹת הַיָּם אֲשֶׁר הוּרְמוּ עַל יְדֵי הַחוֹל עָצְרוּ אֶת הַמַּיִם הַיּוֹרְדִים מֵהֶהָרִים וְלֹא נְתָנוּם לִשְׁטף אֶל הַיָּם.

מִלְּבַד הַבִּצּוֹת אֲשֶׁר עַל שְׂפַת הַיָּם הַתִּיכוֹן יֵשׁ עוֹד בִּצּוֹת אֲחָדוֹת בִּפְנִים הָאָרֶץ. אֶת הַבִּצּוֹת הַקְּטַנּוֹת יְכוֹלִים לְיַבֵּשׁ עַל יְדֵי עֲרוּגָה מְתֻקָּנָה אִם יַחֲרשׁוּן הֵיטֵב וְיִזְרְעוּן וְיִטְעוּ בָהֶן עֵצִים שׁוֹנִים וּבִפְרַט אֶת עֲצֵי הָאֵיקְלִיפְטוּס הַגְּדֵלִים בָּאָרֶץ הַזֹּאת בִּמְהִירוּת נִפְלָאָה כִּמְעַט בְּלִי כָל עֲרוּגָה; אוּלָם הַבִּצּוֹת הַגְּדוֹלוֹת דּוֹרְשׁוֹת תִּקּוּן גָּמוּר לְהַעֲבִיר בָהֶן תְּעָלוֹת עֲמֻקּוֹת וּלְהוֹרִיד אֶת הַמַּיִם שֶׁבָּהֶן אֶל הַנְּהָרוֹת אוֹ אֶל הַיָּם. הַבִּצוֹת הָאֵלֶּה בְּמַצָּבָן עַתָּה, מִלְּבַד שֶׁהֵן מְכֻסּוֹת כִּכְּרוֹת דּוּנַמִּים אֲדָמָה טוֹבָה וּפוֹרִיָה מְאֹד, מְבִיאוֹת גַּם מַחֲלוֹת שׁוֹנוֹת וּבִפְרַט אֶת הַקַּדַּחַת – מַלָּרִיָה

גַּם מַחֲלוֹת הָעֵינַיִם מְצוּיָה בָאָרֶץ וּבִפְרַט בֵּין יְלָדִים וְנָשִׁים, אוּלָם גַּם הַמַּחֲלוֹת הָאֵלֶּה תַּעֲברְנָה עַל יְדֵי עֲרוּגָה וְקֵדְמַָה, עַל יְדֵי יְבשֶׁת הַבִּצּוֹת, הִתְרַבּוּת הַצְּמָחִים וְהָעֵצִים וְהֲטָבַת אוֹפַן הַחַיִּים בִּכְלָל.

מִלְּבַד שְׁתֵּי הַמַּחֲלוֹת הַנִּזְכָּרוֹת מִתְהַלְּכוֹת בָּאָרֶץ, כִּבְכָל הָאֲרָצוֹת, עוֹד מַחֲלוֹת רְגִילוֹת, אַךְ רֻבָּן אֵינָן מְסֻכָּנוֹת פֹּה כְּבַאֲרָצוֹת אֲחֵרוֹת; מַחֲלַת הֶחָזֶה לֹא תֵרָאֶה כִמְעַט פֹּה בֵּין יְלִידֵי הָאָרֶץ.

עַל פִּי הַמַּבָּט הַגֵּאוֹלוֹגִי כָּל הַחֵלֶק הַהֲרָרִי הוּא שִׂידִי קַרְטוּנִי, וְהַשִּׂיד הַזֶּה שׁוֹנֶה מְאֹד עַל פִּי תָכְנִיתוֹ: יֶשׁ רַךְ וְתָלוּל וְיֵשׁ קָשֶׁה וּמוּצָק כְּשַׁיִשׁ וְשִׂיד הַלִּיתֻּגְרַפִי. וּבֵין הֶהָרִים הָאֵלֶּה נִפְגּשׁ רָמוֹת וּבְקָעוֹת פּוֹרִיוֹת. הַבִּקְעָה הַמִּשְׂתָּרַעַת לְאֹרֶךְ חוֹף הַיָּם הַתִּיכוֹן טוֹבָה מְאֹד לַעֲבוֹדַת הָאֲדָמָה; אַדְמָתָהּ הִיא שִׁכְבַת הַחוֹף הַמְכֻסָּה בִּמְקוֹמוֹת רַבִּים בְּגִבְעוֹת חוֹל חַלָּמִישִׁי, וְחֵלֶק הַבִּקְעָה הַדְּרוֹמִי מְצֻיָּנָה בְּטִיבָהּ לִזְרִיעַת הַדָּגָן. וּכְכֹל אֲשֶׁר תִּתְרַחֵק מִן הַחוֹף וְתִקְרַב אֶל הָרֵי יְהוּדָה אַדְמָתָהּ אַדְמַת חֵמָר.

הַדָּבָר הַיּוֹתֵר מַעֲצִיב הוּא לִרְאוֹת אֶת גִּבְעוֹת הַחוֹל הַמִּתְרַחֲבוֹת מְעַט מְעַט וּמְכַסּוֹת חֶלְקוֹת אֲדָמָה אֲשֶׁר נֶעֶבְדוּ קֶדֶם, וּבִפְרַט הֵבִיאוּ הַגְּבָעוֹת נֵזֶק בְּחֵלֶק הַבִּקְעָה הַצְפוֹנִי מִיַּפוֹ וְעַד חֵיפָה וּפְעֻלָּתָן שָׁם מְסֻכָּנָה.

מֵאַחֲרֵי הַגְּבָעוֹת הָאֵלֶּה נִמְצָאוֹת בְּקָעוֹת פּוֹרִיּוֹת אֲשֶׁר אַדְמָתָן מַשְׁקֶה עַל יְדֵי נַחֲלֵי מָיִם, וְהַמַּיִם מִתְקַבְּלִים יַחַד בְּמָקוֹם שָׁפֵל וְגִבְעוֹת הַחוֹל סוֹגְרוֹת בַּעֲדָן אֶת הַדֶּרֶךְ לַעֲבֹר אֶל הַיָּם וְנֶהֶפְכוּ הַבְּקָעוֹת הַפּוֹרִיוֹת לְבִצוֹת מַעֲלוֹת אֵדִים מְסֻכָּנִים לַבְּרִיאוּת.

לוּ הִפְסִיקוּ אֶת מַהֲלַךְ גִּבְעוֹת הַחוֹל עַל יְדֵי נְטִיעוֹת יְדוּעוֹת וְלוּ יִדְשׁוּ אֶת הַבִּצּוֹת עַל יְדֵי אוֹפַנִּים שׁוֹנִים, כִּי עַתָּה הִגְדִּילוּ אֶת פּוֹרִיַּת הָאֲדָמָה וְהֵיטִיבוּ אֶת מַצַב הַבְּרִיאוּת מִסָּבִיב. בִּסְבִיבוֹת יָם הַמֶּלַח מְכֻסָּה הָאֲדָמָה בְּמֶלַח גָּפְרִית וּבְעוֹד חֳמָרִים אֲשֶׁר יָכְלוּ לִהְיוֹת כְּמַעְיָן לְתוֹצָאוֹת חֲרשֶׁת הַמַּעֲשֶׂה, וּמִמֵּי יָם הַמֶּלַח יָכְלוּ לְהוֹצִיא גַם כֵּן חֳמָרִים רַבִּים בְּיִחוּד יוֹד, כְלָרִים שׁוֹנִים וּמֶלַח הַמַּגְנִיזָה.

חֲכָמִים אֲחָדִים מְשַׁעֲרִים, כִּי לוּ חִפְּשׂוּ בְּעַרְבוֹת הַיַּרְדֵּן בְּיִחוּד מִיָּם כִּנֶרֶת וְעַד יָם הַמֶּלַח כִּי אָז מָצְאוּ בְּלִי כָל סָפֵק מַעְיְנוֹת שֶׁמֶן אֲדָמָה. כָּל הָאָמוּר בָּזֶה הוּא אַךְ בְּנוֹגֵעַ לְחֵלֶק הָאָרֶץ הַדְּרוֹמִי הוּא “יְהוּדָה”, אוּלָם בַּגָּלִיל וְגַם בְּשׁוֹמְרוֹן הָאֲדָמָה יוֹתֵר עֲשִׁירָה. שָׁם הָאֲדָמָה מְכֻסָּה בֶּהָרִים גַּבְנוּנִים עֲגוּלִים, וְאַדְמָתָם הִיא אַדְמַת חֵמַָר הַטּוֹבָה לִזְרִיעָה בְּכָל מָקוֹם.

וְאַדְמַת עֵבֶר הַיַּרְדֵּן הִיא אֲדָמָה בוּלְקַנִּית, מַרְאֵה אֲדַמְדָּם אוֹ שְׁחַרְחַר, מֵהִתְפּוֹצְצוֹת הַסְּלָעִים הַבַּזַלְתִּים; הָאֲדָמָה הַהִיא טוֹבָה לִזְרִיעַת הַדָּגָן כִּי אַדְמַת חֵמַָר הִיא וּבְיִחוּד, אִם יִתְּנוּ לָהּ מְעַט שִׂיד הֶחָסֵר לָהּ.

§ 17.

1) הַמֶּמְשָׁלָה לוֹקַחַת מֵעוֹבְדֵי הָאֲדָמָה מַעְשַׂר הַתְּבוּאָה, וְהַמּוֹכְסִים הַלּוֹקְחִים אֶת הַמַּעֲשֵׂר הַזֶּה בַּחֲכִירָה מַכְבִּידִים אֶת יָדָם לִפְעָמִים קְרוֹבוֹת עַל הָאִכָּרִים עוֹבְדֵי הָאֲדָמָה הַמְשַׁלְּמִים לְבַד הַמַּעֲשֵׂר גַּם מֶכֶס הָאֲדָמָה, וּמַס הַדְּרָכִים.

2) מִכָּל סְחוֹרָה הַבָּאָה אֶל הָאָרֶץ מִחוּץ מְשַׁלְּמִים שְׁמוֹנָה לְמֵאָה, וְהַסּוֹחֲרִים כִּמְעַט פְּטוּרִים מִכָּל מַס וּמֶכֶס, וְאֵין הַמֶּמְשָׁלָה מִתְעָרֶבֶת בְּעִנְיָנֵי הַסּוֹחֵר.

3) מִסִּים מְיֻחָדִים יֶשְׁנָם עַל הַתַּבַּק וְעַל הַיַּיִן וְעַל פַּקּוּעוֹת הַמֶּשִׁי.

4) שְׂדוֹת הַכְּפָרִים הֵם נִכְסֵי־הַצִּבּוּר שֶׁל כָּל הַכְּפָר וְלֹא נֶחְלְקוּ בֵין עוֹבְדֵיהֶם לַצְּמִיתוּת, כִּי אִם בְּכָל שָׁנָה וְשָׁנָה מְחַלְּקִים אוֹתָם הַזְּקֵנִים לַחֲלָקִים שָׁוִים וּמוֹסְרִים לְכָל אֶחָד חֵלֶק יָדוּעַ לָעוֹבֵד בַּשָּׁנָה הַזֹּאת וְאֵין אֶחָד זוֹרֵעַ מַעֲנִית אַחַת שְׁתֵּי שָׁנִים רְצוּפוֹת, וְלָכֵן אֵין אִישׁ מֵהֶם חָפֵץ לְהֵיטִיב וּלְהַשְׁבִּיחַ אֶת הָאֲדָמָה שֶׁהוּא עוֹבֵד. אַךְ בְּמוֹשְׁבוֹת הַיְּהוּדִים נֶחְלְקָה הָאֲדָמָה בְּעֶרְכָּאוֹת לְכָל אִכָּר וְאִכָּר לְבַדּוֹ.
