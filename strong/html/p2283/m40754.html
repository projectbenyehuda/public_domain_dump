[תְּהלָּה יְדַבֶּר־פִּי לְשׁוֹאֵל צְרוֹר כַּסְפִּי]

Samuel Archevolti

[נולד בצַ’זֶנָה (Cesena) בפלך רוֹמַנְיָה בשנת 1515, חי בבּוֹלוֹנְיָה ובוֵנֶצְיָה ומשנת 1568 ואילך בפָּדוֹבָה. נפטר בשנת 1611]


שִׁיר עַל־חָנֵף הַמִּשְׁתַּבֵּחַ בַּחֲנֵפוּתוֹ


ں – – ں – – ­­–

תְּהלָּה יְדַבֶּר־פִּי / לְשׁוֹאֵל צְרוֹר כַּסְפִּי / לְהָסִיר דְּבַר מִכְשׁוֹל / וּמַסְגֵּר לְאַמְתַּחְתִּי

וְאַקְהִיל קְהִלַּת עָם / לְדַבֵּר בְּלֵב נִפְעָם / וְאַטְמִין בְּתוֹךְ קִרְבִּי / שְׂעִפַּי וּמַחְשַׁבְתִּי

בְּמוֹ־פִי יְהִי שָׁלוֹם / לְדַלִּים וְהוּא חָלוֹם / וְאוֹי לוֹ לְמִי יִצְפֶּה / לְשִׁבְטִי וּמִשְׁעַנְתִּי

וְאֶפְרֹשׂ מְצוּדָתִי / לְעַמִּי בְּאִמְרָתִי / וְאַסְתִּיר צְפוּן לֵבָב / וְלִי יָד בְּאִישׁ עִתִּי

לְהַרְבּוֹת תְּלֻנָּתוֹ / לְנֶגְדִּי בְסִפְרָתוֹ / וְחֶלְקוֹ יְהִי עִמִּי / בְּיֵינִי וְגַם פִּתִּי

וְסַלְעִי יְהִי סַלְעִי / וּמַה־לִי לְמוֹדָעִי / יְדִידִי וּמַכָּרִי / וְלִבְנִי וּבֶן־בִּתִּי

וְאַרְבֶּה לְהַרְאוֹת אוֹת / וּמוֹפֵת וְנִפְלָאוֹת / לְקַיֵּם דְּבַר כָּזָב / בְּאַרְצִי וּמוֹלַדְתִּי

וְצַדִּיק בְּדוֹרוֹתָיו / אֲסַלֵף בְּאִמְרוֹתָיו / וּפָנַי פְּנֵי אֶסְתֵּר / וְאֶרְשַׁע כְּמוֹ וַשְׁתִּי

בְּמוֹדִים אֶכְרַע אֶקֹּד / בְּבִרְכִּי אֲשַׁו קָדְקֹד / וְאַבִּיט סְבִיבוֹתַי / לְרוֹאִי וּמִי אִתִּי

אֲחַוֶּה לְכָל־רֵעַי / קְרוֹבַי מְיֻדָּעַי / כְּבוֹד כָּל־קְרוּאֵי שֵׁם / בְּבֵיתִי וּמִשְׁפַּחְתִּי

דְּבַשׁ מָן וְכָל־מִשְׁמָן / בְּחִכִּי וְעֵד נֶאְמָן / עֲלֵי כָל־מְזִמּוֹתַי / יְהִי לִי דְּבַר אִשְׁתִּי

וְלִי כָל־כְּלִי מִרְמָה / וּבִכְלָל בְּכָל־עָרְמָה / וְכָל־מִין חֲלָקוֹת פֶּה / אֲנִי הוּא וְאֵין בִּלְתִּי.
