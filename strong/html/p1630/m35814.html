אָמְרוּ כִּי בְּאַחַת הָאִיִּים


מתוך ספר אחיטוב וצלמון

אָמְרוּ כִּי בְּאַחַת הָאִיִּים הָרְחוֹקִים / אֲשֶׁר מִתַּחַת לַמְּצוּקִים / מֵעֵבֶר לְאֶרֶץ הַחֲוִילָה / הָיְתָה אִי אַחַת גְּדוֹלָה, / וּמִימִינָהּ וּמִשְׂמֹאלָה / חוֹמַת אֵשׁ סָבִיב לָהּ: / סוֹבֵב אוֹתָה נַחַל / בּוֹעֵר יוֹם וָלַיְלָה כְּגַחַל. / וְדָרֶיהָ – בַּעֲלֵי שְׁנֵי יְסוֹדִים: / אֵשׁ וַאֲוִיר מְצֻמָּדִים / וּמִמַּיִם וְעָפָר נִבְדָּלִים, / עַל כֵּן בִּתְנוּעָתָם קַלִּים. / וְשָׁם מָלְכָה אִשָּׁה חֲשׁוּבָה, / כְּבוּדָּה בַת מֶלֶךְ וּבַת מַלְכַּת שְׁבָא, / וְהָיוּ עֲבָדֶיהָ סְגָנִים וְשָׂרִים / וְהִיא שׁוֹפְטָה אֶת עַמָּהּ בְּצֶדֶק וּבְמֵישָׁרִים. / וְהָיָה לָה עֹשֶׁר וְכָבוֹד וּנְסִיכוּת / וּמַכְרַעַת עַצְמָהּ לְכַף זְכוּת, / דּוֹבֶרֶת שָׁלוֹם לְכָל זַרְעָהּ וְעַמָּהּ / וּפִיהָ פָּתְחָה בְחָכְמָה, / מְנַהֶגֶת אֶת עַמָּהּ הַנְהָגָה טוֹבָה / וְשׁוֹפֶטֶת בְּרוּחַ נְדִיבָה. / וְעַמָּהּ כֻּלָּם צַדִּיקִים, / אֲבָל בְּלִי נִימוּסִים וְחֻקִּים, / כִּי מִקְצָתָם מַעְגְלוֹתָם נָעוּ / וְאֶת יוֹצְרָם לֹא יָדָעוּ, / וּמִקְצָתָם עוֹבְדִים לְשַׂר שֶׁל מַיִם / וּמִקְצָתָם לִצְבָא הַשָּׁמַיִם / וּמִקְצָתָם עוֹבְדִים לַחַמָּה – / וְלֹא הָיְתָה בֵינֵיהֶם הַסְכָּמָה. / וְהַמְעַט שֶׁבָּהֶם אֲשֶׁר בְּעֵין הַשֵּׂכֶל רוֹאִים / אֶת הָאֱלֹהִים הָיוּ יְרֵאִים. / גַּם הָיְתָה הַמַּלְכָּה אִשָּׁה מַשְׂכֶּלֶת / מַשְׁכֶּמֶת וּמַעֲרֶבֶת וּמִתְפַּלֶּלֶת, / בְּכָל יוֹם פַּעֲמַיִם / נָשְׂאָה יָדָהּ לְרוּם שָׁמַיִם, / אַךְ רַעְיוֹנֶיהָ הָיוּ תְמֵהִים / בַּמֶּה תַעֲבֹד אֶת הָאֱלֹהִים.

וַיְהִי כִּי אָרְכוּ הַיָּמִים וְהַמַּלְכָּה שׁוֹכֶבֶת עַל מִטָּתָהּ / וּמֵעֵינֶיהָ נָדְדָה שְׁנָתָהּ / וְעַל מִשְׁכָּבָהּ שְׁנָתָהּ לֹא עָרְבָה / מֵרֹב נְבִיכוּת הַמַּחֲשָׁבָה, / כִּי שִׁנּוּי הַדֵּעוֹת הֱצִיקָהּ – / וַתַּעַל בְּלִבָּהּ עֵצָה עֲמֻקָּה / לְהַמְצִיא לְעַמָּהּ תַּקָּנוֹת / לְבִלְתִּי יִהְיוּ דֵעוֹת שׁוֹנוֹת. / וַיְהִי בַבֹּקֶר וַתִּשְׁלַח וַתִּקְרָא לַשָּׂרִים וְלַסְּגָנִים / מַרְבִּים הַמִּשְׂרָה וְלַחֲכָמִים וְהַנְּבוֹנִים / יוֹעֲצֵי נְדִיבוֹת / וְלַמְּבִינִים אַנְשֵׁי לְבָבוֹת. / וַיָּבוֹאוּ וַיִּשְׁתַּחֲווּ לְפָנֶיהָ בְאֵימָה, / וַתְּדַבֵּר אֲלֵיהֶם בְּחָכְמָה.

וַתֹּאמַר ״אַתֶּם שָׂרַי וּסְגָנַי / עֲבָדַי וְנֶאֱמָנַי, / הִנֵּה אַרְצֵנוּ בְּכָל טוּב מְבֹרֶכֶת / וְעַל כָּל אִקְלִימִים נֶעֱרֶכֶת, / וְיוֹצְרֵנוּ בֵּרַךְ אֶת גְּבוּלֵנוּ – / מַה טּוֹב וּמַה נָּעִים גּוֹרָלֵנוּ! / כִּי יְיָ הִשְׁפִּיעַ עָלֵינוּ טוֹבָתוֹ / וְצִוָּה לָנוּ אֶת בִּרְכָּתוֹ. / וְעָלֵינוּ לְשַׁבֵּחַ וְלִתֵּן כָּבוֹד – / וַאֲנַחְנוּ לֹא נֵדַע מַה נַּעֲבֹד, / וַאֲנִי רוֹאָה בֵינֵינוּ דִבְרֵי עֲווֹנוֹת, / אֲשֶׁר בֵּינֵינוּ דֵעוֹת שׁוֹנוֹת. / לַדָּבָר הַזֶּה עֵצָה עוּצוּ / וּמַעְיְנוֹתֵיכֶם חוּצָה יָפוּצוּ!״

וַיָּקָם אֶחָד מִן הָאֲנָשִׁים / נִכְבָּד וְשַׂר חֲמִשִּׁים, / וַיֹּאמַר ״אֲדוֹנָתִי הַמַּלְכָה תִחְיִי, / שְׁלֵוָה וְרַעֲנָנָה תִהְיִי! / הִנֵּה דָבָר עָמֹק עוֹלֶה בְמַחֲשָׁבֵךְ, / הֱטִיבוֹת כִּי הָיָה זֹאת עִם לְבָבֵךְ: / כִּי הָאֵל בַּשָּׁמַיִם כְּבוֹדוֹ / אֵין אֱלוֹהַּ מִבַּלְעָדוֹ / וְעַל כָּל בִּרְיָה לַעֲבֹד בִּשְׁמוֹ, / לְהוֹדוֹת לוֹ וּלְרוֹמֲמוֹ. / אֲבָל קַבָּלָה הִיא בְיָדֵינוּ מִישִׁישִׁים / שֶׁאֵין עֲבוֹדָה בְּלִי נִימוּסִים, / וְהַמִּשְׁתַּחֲוֶה לְזוּלָתוֹ / מַשְׁחִית נַפְשׁוֹ וְנַחֲלָתוֹ. / וַאֲנִי שָׁמַעְתִּי כִּי בְקַצְוֵי אֶרֶץ הָרְחוֹקִים / יֵשׁ לָהֶם נִימוּסִים וְחֻקִּים. / וְעַתָּה אִם טוֹב בְּעֵינֵי אֲדוֹנָתִי / קֹצֶר עֲצָתִי וּמַעַן מִלָּתִי, / נַמְתִּיק סוֹד בֵּינֵינוּ / וְנִשְׁלְחָה אֲנָשִׁים לְפָנֵינוּ / וְעָבְרוּ יַמִּים וּנְהָרוֹת / לָתוּר לָנוּ אֶת הַתּוֹרוֹת / וְיַחְפְּרוּ לָנוּ אֶת כָּל הָאֲרָצוֹת / מִקְצֵה אֶרֶץ וְעַד קְצוֹת / וְיוֹדִיעוּ לָנוּ בְּנִימוּסֵי הָעַמִּים חֲדָשׁוֹת / לִלְמֹד וּלְלַמֵּד לִשְׁמֹר וְלַעֲשׂוֹת, / וְנִהְיֶה כֻלָּנוּ לָאֵל עוֹבְדִים / שָׂפָה אֶחָת וּדְבָרִים אֲחָדִים״. / וַיִּיטַב הַדָּבָר בְּעֵינֵי הַמַּלְכָה וּבְעֵינֵי הַשָּׂרִים / וַיַּסְכִּימוּ לְבַקֵּשׁ אֲנָשִׁים יְשָׁרִים, / בַּעֲלֵי מִדּוֹת וּנְבוֹנִים וּתְמִימִים / נִכְבַּדֵּי אֶרֶץ וַחֲכָמִים / מַשִּׂיגֵי הָאֱמֶת בִּתְבוּנָתָם, / וַיְבֻקְשׁוּ וְאֵלֶּה שְׁמוֹתָם: / הָרִאשׁוֹן צַלְמוֹן הָאֲחוֹחִי, / וְהַשֵּׁנִי עֵקֶר הַקָּרְחִי, / וְהַשְּׁלִישִׁי הָיָה אָב בְּחָכְמָה – / אֲחִיטוּב מֵאֶרֶץ תּוֹגַרְמָה. / וַתִּשְׁלַח אַחֲרֵיהֶם הַמַּלְכָּה / בְּרוּחַ נְדִיבָה וּבְלָשׁוֹן רַכָּה / וַתַּשְׁבִּיעֵם הַמַּלְכָּה בָּאֵל אֲשֶׁר בְרָאָם / בַּשָּׁמַיִם וּבָאָרֶץ וּבְכָל צְבָאָם.



וַתֹּאמֶר אֲלֵיהֶם ״עִבְרוּ אֶת הַנָּהָר / עֲלוּ זֶה בַּנֶּגֶב וַעֲלִיתֶם אֶת הָהָר, / וְשׁוֹטֲטוּ בְּכָל מְחוֹזוֹת וּלְשׁוֹנוֹת / וּבְכָל הָאִקְלִימִים וְהַמְּדִינוֹת, / וְחִקְרוּ אַחֲרֵי כָל הָאֱמוּנוֹת, / אִם הֵם אַחַת אִם הֵם שׁוֹנוֹת, / וְהָשִׁיבוּ לָנוּ דְּבַר אֱמֶת בִּלְתִּי נִסְתָּם / מִנִּימוּסֵי הָעַמִּים וַעֲבוֹדָתָם״. / וַיַּעֲנוּ פֶה אֶחָד ״אֲדוֹנָתֵנוּ הַגְּבֶרֶת, / נַעֲשֶׂה כַּאֲשֶׁר אַתְּ דּוֹבֶרֶת״. / וַיִּקְחוּ הָאֲנָשִׁים רְשׁוּת מִן הַמַּלְכָּה וַיֵּלְכוּ הֲלֹם / וַתֹּאמֶר אֲלֵיהֶם ״לְכוּ לְשָׁלוֹם!״


שְׁלוּחַי הַנְּעִימִים! / יִתֶּנְכֶם לְרַחֲמִים

שׁוֹכֵן בַּמְּרוֹמִים / אֲשֶׁר לוֹ תּוֹעָפוֹת

וְחֶפְצֵיכֶם יַשְׁלִים / וְחַסְדּוֹ לֹא יַעְלִים,

תֵּקַלּוּ כְאֵילִים / וּכְצִפֳּרִים עָפוֹת.

תְּסוֹבֲבוּ הַגְּבוּלִים / וְיַמִּים וּנְחָלִים

וְחַפְּשׂוּ כָל אִקְלִים / מְדִינוֹת וְנָפוֹת,

וְעַל כָּל הַמַּעֲשִׂים / תַּחְקְרוּ מִישִׁישִׁים

וְדִרְשׁוּ נִימוּסִים / וְדָתוֹת הַיָּפוֹת.

פְּנוּ לָכֶם וּסְעוּ / לְמַעַן תִּשְׁמְעוּ,

בְּשׁוּבָה תִוָּשְׁעוּ, / יְדֵיכֶם לֹא רָפוֹת.


וַיִּקְחוּ הָאֲנָשִׁים צֵידָה לַדֶּרֶךְ, / כִּי לֹא יָדְעוּ לָהּ קִצְבָּה וָעֵרֶךְ. / וַיַּעַבְרוּ אֶת הַנַּחַל / אֲשֶׁר הָיָה בּוֹעֵר כְּגַחַל, / וַיֵּלְכוּ דֶרֶךְ אֶרֶץ הַחֲוִילָה / וַיָּבוֹאוּ עַד כֶּלַח הִיא הָעִיר הַגְּדוֹלָה. / וַיְדַבְּרוּ הָאֲנָשִׁים יוֹם אֶחָד / ״לָמָּה נֵלֵךְ שְׁלָשְׁתֵּנוּ יַחַד? / כִּי אֵין אָנוּ וּמְדִינָתֵנוּ נִכָּרִים / וּמַשְׂפִּיקִים אָנוּ בֵּין יַלְדֵי נָכְרִים. / יֶשׁ לָנוּ לָחוּשׁ מֵעַמֵּי הָאֲרָצוֹת, / בַּעֲלֵי מִרְמוֹת וּמוֹעֵצוֹת, / פֶּן יִהְיוּ עָלֵינוּ מַעֲרִימִים וּמַעֲלִילִים / וְיִתְּנוּ אוֹתָנוּ כִּמְרַגְלִים / וְיִהְיוּ עָלֵינוּ מִתְוַעֲדִים / לָקַחַת אוֹתָנוּ לַעֲבָדִים. / וְעַתָּה נִקַּח עֵצָה טוֹבָה / וְנִפָּרֵד אִישׁ מֵעַל רֵעֵהוּ בְּאַהֲבָה / וְלֹא יַרְגִּישׁ בָּנוּ אִישׁ כִּי יִכְרֶה הַבּוֹר / וּנְחַלְּקָה לָנוּ אֶת הָאֲרָצוֹת לַעֲבֹר. / וּזְמַן תְּשׁוּבָתֵנוּ יַחַד נִקְבַּע – / עַד קֵץ שֶׁבַע״. / וַיַּסְכִּימוּ הָאֲנָשִׁים לַדָּבָר הַהוּא / וַיִּפָּרְדוּ אִישׁ מֵעַל רֵעֵהוּ. / וַיַּעֲבֹר הָרִאשׁוֹן / בְּאֶרֶץ הַנֶּגֶב מוּל נַחַל פִּישׁוֹן, / וְהַשֵּׁנִי אָרַח לוֹ לִקְרַאת / הַנָּהָר הַגָּדוֹל נְהַר פְּרָת, / וְהַשְּׁלִישִׁי לְצַד אִסְתָּן שָׁת עֶַרְכּוֹ / לִנְהַר חִדֶּקֶל דַּרְכּוֹ.

וַיַעַבְרוּ הָאֲנָשִׁים נְהָרוֹת וְיַמִּים / וַיְשׁוֹטְטוּ אֲרָצוֹת וְאִקְלִימִים / וַיַּעַבְרוּ מְדִינוֹת וְנָפוֹת, / נוֹרָאוֹת וְיָפוֹת, / וּבְכָל מְדִינָה וּמְדִינָה עִיר וָעִיר / חוֹקְרִים מִן רַב וְצָעִיר / עַל אֱלֹהֵיהֶם וְעַל דָּתָם / עַל נִימוּסֵיהֶם וְעַל אֱמוּנָתָם. / וַיִּמְצְאוּ כִּי הָאֱמוּנוֹת / לִשְׁלֹשָׁה חֲלָקִים מִשְׁתַּנּוֹת, / וַיֵּדְעוּ כִּי הַיְּהוּדִים לְמִסְפָּרָם / שׁוֹמְרִים תּוֹרַת מֹשֶׁה בֶן־עַמְרָם, / אַךְ הֵם הַמְעַט מִכָּל הָעַמִּים / וּשְׁפָלִים מִכָּל הָאֻמִּים. / וְהָאֻמָּה הַיִּשְׁמְעֵאלִית בִּכְלָלָהּ / מַאֲמִינִים כִּי מֻחַמַּט רְשוּת הַשֵּׁנִי לֶאֱלוֹהַ / וֶאֱלוֹהַּ שָׂמוֹ בָעוֹלָם מְמֻנֶּה / וַיַּרְכֵב אוֹתוֹ בְּמִרְכֶּבֶת הַמִּשְׁנֶה. / וּמַלְכוּת רוֹמִי הֵם הַנּוֹצְרִים / אֱמוּנַת יֵשׁוּ הֵם נוֹטְרִים / וּמַאֲמִינִים גְּדוֹלָם וּקְטַנָּם / שֶׁסָּבַל מִיתָה כְּדֵי לִפְדּוֹתָם מֵהַגֵּיהִנָּם.

וְצַלְמוֹן עָבַר בְּכָל אֶרֶץ הַצִּידוֹן / וַיָּבוֹא עַד הַר שִׂיאוֹן / וּבְכָל אֶרֶץ שבילייא עַד הַשִּׁיחוֹר / וְהַיַּרְדֵן יִסֹּב לְאָחוֹר, / וַיַּעֲבֹר עַד בֵּית חֹרוֹן / וַיָּבוֹא עַד חֶבְרוֹן. / וַיַּרְא שָׁם הַמְּעָרָה הַגְּדוֹלָה / הַנִּקְרֵאת מְעָרַת הַמַּכְפֵּלָה / אֲשֶׁר שָׁמָּה הָיוּ הָאָבוֹת קְבוּרִים, / בַּעֲלֵי נְטִיעוֹת הֵמָּה הַיּוֹצְרִים, / וְעַל פְּנֵי כֻלָּהּ בִּנְיָן נָאֶה / וּבַיִת גָּדוֹל לְמַרְאֶה, / וְנוֹתְנִים שָׁם לְדַלִּים צְדָקוֹת / וְשָׁם הָיוּ תָמִיד עֲשָׁשִׁיּוֹת דּוֹלְקוֹת, / יוֹם וָלַיְלָה לֹא יִכְבּוּ / וְרָבוּ כְּמוֹ רָבוּ, / וּמְקַטְּרִים שָׁם לְבוֹנָה וְסַמִּים / וְכָל רָאשֵׁי בְשָׂמִים. / וַיַּרְא צַלְמוֹן בְּנִימוּסֵי הָעֵדָה / וַתִּכָּנֵס בְּלִבּוֹ הָעֲבוֹדָה / וַיֹּאמַר: אַךְ זֶה הַיּוֹם שֶׁקִּיוִיתִי / יָגַעְתִּי וְגַם מָצָאתִי! / זֶהוּ עִקַּר הָאֱמוּנָה וְהַמַּעֲשֶׂה / וְאַחֲרֵי אֱמוּנָה אַחֶרֶת לֹא אֶשְׁאַל וְלֹא אֲנַסֶּה, כִּי וַדַּאי נִימוּסֵיהֶם וְדָתֵיהֶם / הֵן אֲשֶׁר יַעֲשֶׂה אוֹתָם הָאָדָם וְחַי בָּהֶם. / וַיִּפְשֹׁט צַלְמוֹן אֶת בְּגָדָיו / וַיִּלְבַּשׁ מֵעַל לְמַדָּיו אַדֶּרֶת שֵׂעָר עַד רַגְלָיו, / וְהָיָה אֲזוֹרוֹ מוֹכִיחַ עָלָיו, / וּבַעֲטִיפַת יִשְׁמְעֵאלִים עַל שָׂפָם יַעְטֶה, / עוֹד מִנִּימוּסֵי יִשְׁמְעֵאלִים לֹא יִטֶּה, / וְשֶׁבַע פְּעָמִים בַּיּוֹם נוֹפֵל עַל בֶּרֶךְ, / כּוֹרֵעַ לְמֻחַמַּט וּמְבָרֵךְ.

וְעֵקֶר גַּם הוּא נוֹסֵעַ דֶּרֶךְ עֲרַבְיָא / וַיָּבוֹא עַד יָם אַנְטוֹכְיָא / וַיָּשֶׂם פָּנָיו אֶל אֶרֶץ אַרְמֵנְיָא / וַיָּבוֹא עַד קוֹשְׂטַנְטִינָא, וְהִיא הָיְתָה עִיר גְּדוֹלָה מֵאֱמוּנַת נוֹצְרִים / אֲשֶׁר סוֹחֲרֶיהָ שָׂרִים. / וְאֵחַר שָׁם יָמִים שְׁלֹשָׁה / לַעֲשׂוֹת חֲקִירָה וּדְרִישָׁה / עַל אֱמוּנָתָם וְעַל דָּתָם. / וַיַּרְאוּהוּ בְּבֵית נְכוֹתָם / עֵץ אֲשֶׁר יֵשׁוּ בּוֹ נִתְלָה / וְאוֹמְרִים שֶׁהוּא טוֹב לְכָל מַחֲלָה / וְכִי יְרַפֵּא אֲשֶׁר חֵץ מְחָצוֹ / כִּי יִתֵּן בַּכּוֹס עֵצוֹ. / וְהֶרְאוּהוּ כֶּתֶר קוֹצִים שֶׁעַל רֹאשׁוֹ / יוֹם הֶעֱרָה לַמָּוֶת נַפְשׁוֹ, / וּמַסְמְרוֹת אֲשֶׁר נִתְקְעוּ בֵּין רַגְלָיו / כַּאֲשֶׁר הַנּוֹצְרִים מַאֲמִינִים עָלָיו. / וְעֵקֶר טָפַשׁ כַּחֵלֶב לִבּוֹ / וַתָּבוֹא הָאֱמוּנָה כַּמַּיִם בְּקִרְבּוֹ / וּבְעַצְמוֹתָיו כְּמוֹ שֶׁמֶן – / כִּי פֶתִי לְכָל אֲשֶׁר יִשְׁמַע יַאֲמֵן! / וַיְסַפְּרוּ לוֹ אֵיךְ הַיְּהוּדִים עִטְּרוּהוּ וַיְקַצְּרוּ יָמָיו, / וַיְבַקֵּשׁ עֵקֶר לִבְכּוֹת כִּי נִכְמְרוּ רַחֲמָיו. / וַיְבַקֵּשׁ עֵקֶר לְהוֹדִיעֵהוּ וּלְהוֹרוֹת / אֶת הַחֻקִּים וְאֶת הַתּוֹרוֹת, / וַיֹּאמְרוּ ״אִם תְּבַקֵּשׁ לָדַעַת עִקַּר הָאֱמוּנָה, / לֵךְ אֶל רוֹמִי הַמְּדִינָה: / תַּלְמִידֵי יֵשׁוּ שָׁמָּה נִתְגַדְּלוּ / וְתוֹרַת יֵשׁוּ שָׁמָּה הֶעֱלוּ״.

וְהָאִישׁ עֵקֶר לֹא נָתַן לְנַפְשׁוֹ דוּמָה / עַד הַגִּיעוֹ לִמְדִינַת רוֹמָה, / כִּי הָיָה מַאֲמִין לְכָל הַדְּבָרִים / וּמִתְפַּתֶּה לְכָל הַסִּפוּרִים, / וְלֹא הָיָה מַעֲמִיק בַּמַּחֲשָׁבָה / לְהַבְחִין בֵּין דֶּרֶךְ רְחוֹקָה לִקְרוֹבָה, / לְהַבְחִין בֵּין הָאֱמֶת לְבֵין הַשֶּׁקֶר. / וַיְהִי כִּי הִגִּיעַ שָׁמָּה עֵקֶר / וַיָּבוֹא עֵקֶר בְּתוֹךְ הַבָּאִים / וַיַּרְא שָׁם בִּנְיָנִים מֻפְלָאִים / בְּנוּיִים לְשֵׁם וְלִתְהִלָּה. / וַיָּבוֹא עַל הַבָּמָה הַגְּדוֹלָה / אֲשֶׁר נִקְרֵאת עַל שֵׁם קָדֵשׁ פְּיֶטְרִי, / תַּלְמִידוֹ שֶׁל יֵשׁוּ הַנּוֹצְרִי. / וַיַרְאוּהוּ שָׁם מִטְפַּחַת / בְּבֵית גִּנְזֵיהֶם מֻנַּחַת, / קַבָּלָה בְּיָדָם אָב לְבָנִים / שֶׁהַנּוֹצְרִי קִנַּח בָּהּ פָּנִים. / וַיֵּלֶךְ עֵקֶר מִכּוּמָרַיָּא לְכוּמָרַיָּא / מִבֵּית כַּרְיָא לְבֵית גַּלְיָא, / וַיְהִי הוֹלֵךְ וּמְפַשְׁפֵּשׁ / אַחֲרֵי אֱמוּנַת אוֹתוֹ יֵשׁוּ מְחַפֵּשׂ – / עַד כִּי הָיְתָה בְלִבּוֹ שְׁמוּרָה / וְנַעֲשָׂה כֹמֶר לַעֲבוֹדָה זָרָה. / וַיְגַלַּח וַיְחַלֵּף שִׂמְלוֹתָיו / וַיִּתֵּן עֲלֵיהֶם אוֹתוֹתָיו / שְׁתִי וָעֵרֶב עַל לְבוּשׁוֹ, / וַיִּקְרַח קָרְחָה בְּרֹאשׁוֹ, / וַיִּלְבַּשׁ לְבוּשֵׁי הַכְּמָרִים / וַיִּדְבַּק בְּדָת הַנּוֹצְרִים.

וַאֲחִיטוּב גַּם הוּא עָבַר מַעְבָּרוֹת / וַיְסוֹבֵב יַמִּים וּנְהָרוֹת, / לֹא הִנִּיחַ גּוֹי וּמַמְלָכָה / אֲשֶׁר רַגְלוֹ לֹא דָרְכָה. / וַיֵּלֶךְ בַּמְּדִינוֹת כְּמִין גַּאם / לָדַעַת בָּאֱמוּנוֹת פִּתְגָּם: / כִּי אֱמוּנַת מֻחַמַּט בְּלִבּוֹ לֹא עָלָה / הֱיוֹת בָּשָׂר וָדָם בְּזוֹ הַמַּעֲלָה – / וַיַּחְשְׁבֶהָ לְחוּכָה וְִלִטְלוּלָא, / וְלֶאֱמוּנַת יֵשׁוּ לֹא פָנָה, / כִּי אָמַר: בִּלְתִּי יְסוֹד וּפִנָּה / הֱיוֹת אֱלוֹהַּ נוֹלָד מֵאִשָּׁה / וְיָצָא מִמְּקוֹם טִנֹּפֶת וּבָאְשָׁה, / וִהְיוֹת אַחַר כָּךְ אֱלוֹהַּ מֵת – / קָרוֹב לַשֶּׁקֶר וְרָחוֹק מִן הָאֱמֶת!

וַיֵּלֶךְ וּנְבוֹכִים רַעְיוֹנָיו / וַיָּשֶׁת אֶל הַמִּדְבָּר פָּנָיו, / וַיֵּלֶךְ מַהֲלָךְ גָּדוֹל וְחָשׁוּב / עַד שֶׁבָּא לִכְלַל יִשּׁוּב. וְלֹא עָבַר דֶּרֶךְ שָדוֹת וּכְרָמִים / מַהֲלַךְ עֲשָׂרָה יָמִים, / עַד הַגִּיעוֹ לְעִיר גְּדוֹלָה בָּרֶמֶשׁ, / וַיָּלֶן שָׁם כִּי בָא הַשֶּׁמֶשׁ. / בַּלַּיְלָה הַהוּא חָלַם אֲחִיטוּב / וְהִנֵּה הוּא עוֹמֵד בְּגַן רָטֹב, / וּבַגַּן הָיָה אִילָן סָמוּךְ, / שָׁפָל מְאֹד וְנָמוּךְ, / אֲבָל הָיוּ רַבִּים פֹּארוֹתָיו / וְהָיוּ מְרֻבִּים פֵּרוֹתָיו. / וַיִּקְרַב אֶל הָאִילָן לִטְעֹם טַעְמָן / וַיִּמְצָא בָהֶם טַעַם מָן. / וַיְהִי כִרְאוֹתוֹ כִּי אוֹרוּ עֵינָיו / וַיִּלְקֹט מֵהֶם מְלוֹא חָפְנָיו / וַיְמַלֵּא מֵהֶן תַּרְמִילוֹ / וַיִּתְאַב לָשֶׁבֶת בְּצִלּוֹ. / וַיִּיקַץ אֲחִיטוּב וְהִנֵּה חֲלוֹם / וַיֹּאמַר: חֶלְמָא טָבָא חֲזַאי, יְהִי שָׁלוֹם!

וַיַשְׁכֵּם בַּבֹּקֶר וְרַעְיוֹנָיו נְבוּכִים / וְלִבּוֹ לָצֵאת אֶל הַדְּרָכִים, / וַיַּעֲבֹר לִפְנֵי בַיִת גְּדוֹלָה / וַיִּשְׁמַע שָׁם הֲמֻלָּה. / וַיֹּאמֶר בְּלִבּוֹ: נִבְהֲלוּ רַעְיוֹנַי / מָה הַקּוֹל הַזֶּה בְּאָזְנַי? / עַתָּה אָסוּרָה נָּא וְאֶרְאֶה. / וַיַּעֲמֹד פֶּתַח הַבַּיִת מַחֲרִישׁ וּמִשְׁתָּאֵה, / וַיַּרְא שָׁם עֲדַת אֲנָשִׁים / כֻּלָּם מַעֲרִיצִים וּמַקְדִּישִׁים / וּמִתְפַּלֲלִים בְּאֵימָה וּבְיִרְאָה, / וַיִּשְׁתּוֹמֵם עַל הַמַּרְאָה.> וַיַּרְא עוֹד כִּי אַחַר הַתְּפִלָּה / עָמַד זָקֵן עַל גַּב מַעֲלָה / וּבְהִלְכוֹת הַיִּחוּד לָהֶם דּוֹרֵשׁ / וּפֶרֶק אֶחָד שׁוֹנֶה וּפוֹרֵשׁ. / וַיְהִי כְּכַלּוֹתוֹ לִדְרֹשׁ לִקְהָלוֹ / וַיֵּלֶךְ כָּל אֶחָד אֶל אָהֳלוֹ, / וְהַזָּקֵן לַחֲזֹר לִמְעוֹנוֹ פָנָה / עָבַר בַּשּׁוּק אֵצֶל כָּל פִּנָּה. / וְהִנֵּה לִקְרָאתוֹ עֲדַת דְּבוֹרִים / מִקְּטַנֵּי בְנֵי הַנּוֹצְרִים / וַיִּסְקְלוּ בוֹ שׁוּעָלִים קְטַנִּים / וַיִּרְגְּמוּהוּ בָאֲבָנִים / וַיִּקְרְאוּ ״כֶּלֶב כֶּלֶב!״ / אֶל הַזָּקֵן עָלוּב וְלֹא עוֹלֵב. / וַיַּרְא אֲחִיטוּב וַיְבַטְּלֵם / וַיִּגְעַר בָּהֶם וַיְקַלֲלֵם. / וַיְהִי כִּי הִקְרִיב הַזָּקֵן אֶל בֵּיתוֹ – / וְהִנֵּה אִישׁ בְּלִיַּעַל לִקְרָאתוֹ / וַיִּתְמַרְמֵר לְנֶגְדּוֹ / וַיִּתְפְּשֵׂהוּ בְּבִגְדוֹ / וַיְחָרְפֵהוּ בְּחֵץ לְשׁוֹנוֹ, / וַיִּגְזֹל מִמֶּנּוּ מִטְרוֹנוֹ / וַיַּךְ אוֹתוֹ בַּחֲרוֹן קִצְפּוֹ./ וַיָּסַר מִמֶּנּוּ כַּפּוֹ / וַיֹּאמַר ״הָבָה לִי דִּינַר כֶּסֶף / אוֹ רְאוֹת מִטְרוֹנְךָ אַל תּוֹסֶף!״ / וַיִּפְתַּח הַזָּקֵן כִּיסוֹ וְהַדִּינָר יִצְבֹּט / וַיָּשֶׁב לוֹ הָאִישׁ אֶת הָעֲבוֹט.

וַיָּשָׁב הַזָּקֵן לְבֵיתוֹ / וַיְהִי מְעַט בַּבַּיִת שִׁבְתּוֹ – / וַיִּקַּח הַשּׁוֹטֵר עֲבוֹטוֹ / וְתָמִיד שִׁכְמוֹ לִסְבֹּל יַטּוֹ, / כִּי נִתְחַיֵּב לְשַׂר הַבִּירָה מַס, / וְלִבּוֹ לֵב אֲרִי לֹא הִמֵּס יִמַּס. / אַךְ אֹמֶר לַעֲלוּקָה ״הַב הַב״, / וַיִּפְדֶּה אֶת עֲבוֹטוֹ בִּשְׁלֹשָׁה שִׁקְלֵי זָהָב / וְשִׁכְמוֹ לִסְבֹּל נָטָה / וּבְלִבּוֹ וּבִשְׂפָתָיו לֹא חָטָא, / רַק אָמַר ״יְהִי רָצוֹן שֶׁיְּהֵא גָלוּתֵנוּ / כַּפָּרָה לְנַפְשׁוֹתֵינוּ!״ / וַיָּשָׁב הַזָּקֵן אֶל הַמְּסִבָּה / וְהִנֵּה אֲחִיטוּב בָּא / וַיִּתְיַצֵּב שָׁם עִמּוֹ / וְזֶה דְבַר נְאוּמוֹ:


אַתָּה אֱנוֹשׁ יָשִׁישׁ אֲשֶׁר אֱלֵי מַכִּים

נוֹתֵן לְחִי וָהוֹן לְכֹל אֲשֶׁר יִתְבַּע.

לִי נִסְתְּרָה דַרְכָּךְ – מַה נִּשְׁתַּנִּיתָ

אֲשֶׁר מִכָּל אִישׁ בּוּז וְגַם חֶרְפָּה תִשְׂבַּע?

אֶחְשֹׁב הֱיוֹתְךָ בָּא מִתּוֹלְדוֹת כֹּהֵן

מִדְיָן אֲשֶׁר רוֹעִים גֵּרְשׁוּ בְנוֹת שֶׁבַע.


וַיֹּאמֶר לוֹ ״יְהוּדִי אַתָּה?״ / וְיֹּאמֶר הַזָּקֵן ״כֵּן דִּבַּרְתָּ״. / וַיֹּאמַר ״עוֹד מַצְפּוּן לִבְּךָ מִמֶּנִּי אַל תַּכְמִין / הוֹדִיעֵנִי בְּמִי אַתָּה מַאֲמִין״. / וַיֹּאמֶר לוֹ ״אֲנִי מַאֲמִין בֵּאלֹהֵי אַבְרָהָם יִצְחָק וְיַעֲקֹב״.
