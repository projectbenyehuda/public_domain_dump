בָּרְכִי אֱמוּנָה בְּיַד

ד. ראש חודש

חתימה: א״ב, שלם אלשבזי משתא


בָּרְכִי אֱמוּנָה בְּיַד אֱלֹהַיִךְ שׁוֹמֵר אֱמוּנִים

אֶת שֵׁם אֵל יָחִיד וּמְיוּחָד בְּפִי עַם קוֹדֶשׁ הֲגוּנִים

אוֹהֲבִים לַהֲגוֹת בְּתוֹרָתוֹ וּלְשֻׁלְחָנוֹ זְמוּנִים

אַשְׁרִי אִישׁ יָרֵא אֶת יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי בַּעֲלַת חֲמִשָּׁה שֵׁמוֹת קְבוּעִים

אֶת שֵׁם בּוֹרֵא מְצִיאוּת הַנִּסְתָּרִים וְהַיְּדוּעִים

בָּחַר בְּזֶרַע יַעֲקֹב גִּנַּת כֶּרֶם נְטוּעִים

בּוֹ בָּחַר יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי גְּבֶרֶת הַמִּתְקַשֶּׁרֶת בְּאַרְבַּע יְסוֹדוֹת

אֶת שֵׁם גּוֹזֵר וּמְצַוֶּה בְּחֻקִּים וְעֵדוֹת

גְּדוֹל הָעֵצָה וְלִשְׁמוֹ טוֹב לְהוֹדוֹת

גָּדוֹל וְקָדוֹשׁ יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי דּוֹבֶרֶת צֶדֶק לַעֲשׂוֹת מֵישָׁרִים

אֶת שֵׁם דַּיַּן אַלְמָנוֹת עֶלְיוֹן עַל גַּלְגַּלִּים וְשָׂרִים

דִּבַּר לְמֹשֶׁה לְצַוּוֹתֵנוּ בְּחֻקִּים יְשָׁרִים

דָּבָר שָׁלַח יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי הֲדוּרַת קוֹמָה בַּת מֶלֶךְ חֲשׁוּקָה

אֶת שֵׁם הַמּוֹשִׁיעֵךְ מֵהַבְלֵי זְמָן וּמְצוּקָה

הָעִירִי שֵׂכֶל אֱנוֹשׁ מִשּׁוּחָה עֲמוּקָה

הַלְלוּ אֶת שֵׁם יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי וְעוּדָה לַעֲרוֹךְ מַעֲשַׂיִךְ יוֹם דִּין וְחֶשְׁבּוֹן

אֶת שֵׁם וַתִּיק דַּיָּן עַל כָּל יְצוּרָיו וְרִבּוֹן

וְאִם תְּעוֹרְרִי יֵצֶר טוֹב וְתִתְּנִי עֵרָבוֹן

וְהִקְרַבְתֶּם אִשֶּׁה לַיוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי זְבוּדָה בְּזֶבֶד טוֹב מִיְּקָר עֶלְיוֹן

אֶת שֵׁם זוֹכֵר הַבְּרִית לְעַם עָנִי וְאֶבְיוֹן

זַכִּי פְּעֻלּוֹת הַגּוּף וְשֵׂכֶל הָרַעְיוֹן

זִכָּרוֹן לִפְנֵי יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי חֲכָמָה הַמְסַכֶּכֶת עַל בִּנְיַן הַגּוּף

אֶת שֵׁם חַי הָעוֹלָמִים מַמְצִיא בְּלִי שִׁתּוּף

חוֹקֵר כְּלָיוֹת וָלֵב וְסוֹד מַאֲמָרוֹ צָרוּף

חוֹנֶה מַלְאַךְ יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי טוֹבַת מַעֲשֶׂה הַמַּחְזֶקֶת בַּמִּצְווֹת

אֶת שֵׁם טְהוֹר עֵינַיִם וְדֶרֶךְ רְשָׁעִים יְעַוֵּת

טוּבוֹ וְחַסְדּוֹ לְעוֹלְמֵי עַד וּבְרִיּוֹתָיו לְטוּבוֹ מְקַוּוֹת

טוֹב לְקוֹיָיו יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי יְקָרָה מִטִּנּוּפֵי גּוּף וְרוֹב הֲבָלָיו

אֶת שֵׁם יוֹצֵר הַכֹּל וְנוֹתֵן לְאִישׁ כְּמַעֲלָלָיו

יוֹדְעִים סוֹד וּמִשְׂרָה וְחוֹכִים וְנוֹהֲרִים אֵלָיו

יִשְׂרָאֵל בְּטַח בַּיוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי כְּמוּסָה בִּיסוֹד צַדִּיק שְׁמוּרָה

אֶת שֵׁם כַּבִּיר כּוֹחַ הַקּוֹשֵׁר גּוֹלֶם וְצוּרָה

כִּתְבִי עַל לוּחַ לִבֵּךְ תּוֹרַת אֱלֹהַיִךְ תְּשׁוּרָה

כִּי יָשָׁר דַּעַת יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי לוֹבֶשֶׁת הוֹד שָׁפוּעַ מֵהַר הַכַּרְמֶל

אֶת שֵׁם לוֹקְחֵךְ עֶלְיוֹן מִגַּלְגַּלִּים וְחַשְׁמַל

לִקְרַאת אֱלֹהַיִךְ תָּשׁוּבִי וְחַסְדּוֹ אֵלַיִךְ יִגְמַל

לִרְאוֹת בְּטוּב יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי מְשׁוּכָה בְּחוּט שֶׁלְּחֶסֶד קְשׁוּרָה

אֶת שֵׁם מְחַיֵּיךְ בְּמִזְגֵּי הַגּוֹלֶם עֲצוּרָה

מִצְאוּ לָכֶם סוֹפֵר בִּתְשׁוּבָה גְּמוּרָה

מִנְחָה הִיא לַיְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי נְקִיָּה לְשֻׁלְחַן אֱלֹהַיִךְ קְרוּאָה

אֶת שֵׁם נֶאֱמָן בִּבְרִיתוֹ וְכִסְאוֹ לְמַעְלָה נְשׂוּאָה

נֶעֱלָם מִכָּל עַיִן וְטוֹבָתוֹ לַכֹּל מְצוּאָה

נַפְשֵׁנוּ חִכְּתָה לַיוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי סוֹעֶדֶת הַגּוּף מִזִּיו כְּבוֹד הַשְּׁכִינָה

אֶת שֵׁם סוֹבֵל הָעוֹלָם כְּמוֹ הַיָּם לַסְּפִינָה

שְׂאִי שָׁלוֹם וּבַשְּׂרִי בַּעֲלֵי דַּעַת נְכוֹנָה

שִׂמְחוּ צַדִּיקִים וְגִילוּ בַּיוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי עוֹסֶקֶת בַּמִּצְווֹת לִשְׁמוֹר לַעֲשׂוֹת

אֶת שֵׁם עוֹשֶׂה רְצוֹן יְרֵאָיו וּבוֹ טוֹב לַחֲסוֹת

עַנְנֵי כְּבוֹדוֹ מִתְפַּרְשְׁזוֹת וּמִתְרַכְּסוֹת

עוֹשֵׂה כֻלָּם יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי פּוֹצַחַת שִׁיר לְהַלֵּל לְקוֹנֵךְ

אֶת שֵׁם פּוֹעֲלֵךְ וּמִזִּיו שְׁכִינָתוֹ מְזוֹנֵךְ

פַּתִּי גּוֹלֶם וְיָשׁוּב מֵחֵטְא וְאָמַרְתְּ לִלְשׁוֹנֵךְ

פָּצִיתִי פִּי אֶל יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי צוֹפָה נוֹעַם אֱלֹהַיִךְ לִרְאוֹת

אֶת שֵׁם צוּר הָעוֹלָמִים אֱלֹהֵי הַצְּבָאוֹת

צַפִּי לִישׁוּעָתוֹ וּקְחִי תּוֹרָתוֹ אוֹת

צוּרִי וְגוֹאֲלִי יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי קְרוּאָה לְהֵיכַל מֶלֶךְ פְּנִימָה

אֶת שֵׁם קָדוֹשׁ וּמְקֻדָּשׁ בְּשִׁיר וּבִנְעִימָה

קָבַע שְׁנֵי הַמְּאוֹרוֹת וְכוֹכָבִים וּכְסִיל וְכִימָה

קַוֹּה קִוִּיתִי יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי רְצוּיָה לַעֲשׂוֹת רְצוֹן אֱלֹהַיִךְ

אֶת שֵׁם רָם וְנִשָּׂא וּמַשְׁגִּיחַ עַל כָּל פָּעֳלַיִךְ

רַבַּת בָּנִים בַּשְּׂרִי כִּי יָחִישׁ גּוֹאֲלַיִךְ

רוּחַ אַפֵּנוּ מְשִׁיחַ יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי שְׁמוּרָה מִכָּל חֵטְא וְסוּרֵי הֲבָלִים

אֶת שֵׁם שַׁדַּי מְרוֹמָם עַל כָּל מַהֲלָלִים

שְׁמוֹ מְיוּחָד בְּפִי זְקֵנִים וּבַחוּרִים וְעוֹלָלִים

שְׁמַע יִשְׂרָאֵל יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


בָּרְכִי תְּלוּיָה בְּחוּט שֶׁלְּחֶסֶד מְשׁוּכָה

אֶת שֵׁם תָּמִים אֲשֶׁר אַתְּ בִּשְׁמוֹ בְרוּכָה

תְּנִי חֵן וְהָעִיצִי בֶּן גְּבֶרֶת בַּמְּלוּכָה

תָּמִים תִּהְיֶה עִם יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:


שִׁוַּעְתִּי לְטוּבָךְ מַמְצִיא הַכֹּל וּמְחַיֵּם

אֱלֹהַי לְעַבְדְּךָ שְׁלַח בְּפִדְיוֹם

זְכוּת יְשָׁרִים מַפְצִיחֵי שְׁבָח תִּפְאַרְתְּךָ אָיוֹם

הוֹשִׁיעֵנוּ יְוֹיָ

בָּרְכִי נַפְשִׁי אֶת יְוֹיָ:
