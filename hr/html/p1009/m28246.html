אוֹדֶה לְךָ צוּרִי

אוֹדֶה לְךָ צוּרִי וְהִנֵּנִי / אֶזְכֹּר חֲסָדִים שֶׁגְּמַלְתַּנִי

בַּנְתָּ לְמוֹדָעִי לְמֵרָחוֹק / מִשֹּׁד וּמֵחָמָס פְּדִיתַנִי

גָּלְמִי הֲלֹא תִרְאֶה בְּעֵינֶיךָ / בַּדִּין וּבַחָכְמָה יְצַרְתַּנִי

דִּבְרֵי עֲוֹנוֹת גָּבְרוּ מֶנִּי / הוֹאֵל סְלַח נָא לִי וְחָנֵּנִי

הוֹאֵל וְכַפֵּר אֶת עֲוֹנוֹתַי / אֶעֱמֹד וְיַד צָר אַל תְּנִידֵנִי

וּפְתַח לְעַבְדָּךְ שַׁעֲרֵי צֶדֶק / אַחַר בְּחָנֵּנִי וְנַסֵּנִי

זָכְרָה לְבִנְךָ חַסְדֵי אָבוֹת / עֵת בּוֹא בְמִשְׁפָּט אַל תְּדִינֵנִי

חֶלְקִי יי אָמְרָה נַפְשִׁי / לָכֵן גְּוִיָּתִי תְּרִיבֵנִי

טָבְעָה לְטוּמְאָתָהּ בְּשׁוּלֶיהָ / רוּחַ בְּקִרְבִּי הִיא הֱצִיקַתְנִי

יָמַי הֲלֹא קַלּוּ מְרוּצָתָם / אֶגְוַע וְעַיִן לֹא תְשׁוּרֵנִי

כָּלוּא בְיָגוֹן כָּל יְמֵי חַיַּי / רָעוֹת בְּלֹא מִסְפָּר סְבָבוּנִי

לֹא תַעֲמֹד יִצְרִי עֲלֵי עָמְדָךְ / חִנָּם לְבַלֵּעַ תְּסִיתֵנִי

מַהֵר זְכֹר זְמַנִּי אֲשֶׁר עָבַר / בַּיָּם וְכִמְעַט קָט וְאֵינֶנִּי

נֶגֶד לְעֵינֶיךָ זְכֹר בָּבֶל / יוֹם תַּחֲלוּאֶיהָ הֲלָמוּנִי

סוֹרֵר זְכֹר לֶכְתָּךְ בְּאֶרֶץ כּוּשׁ / יוֹם הָאֳנִיָּה נִדְּחָה מִנִּי

עָצֵל זְכֹר חוֹמַת מְדִינָתָךְ / יוֹם הָבְקְעָה הָעִיר וְקָם מוֹנִי

עַיִן לְךָ תִּלְעַג בְּתוֹךְ אַרְמוֹן / כָּלוּא בְמַחְלוֹנִי וְכִלְיוֹנִי

פֶּן תֹּאמְרָה יִצְרִי לְכַדְתַּנִי / לָכֵן סְעִפַּי הֵם יְשִׁיבוּנִי

פָּרוּד וְגַם פּוֹחֵד בְּמִגְדָּלוֹת / בָּהֶם בְּנֵי אָמָה אֲסָרוּנִי

צַדִּיק בְּכָל אֵלֶּה אֱלֹהִים חַי / גָּבְרוּ חֲסָדָיו לֹא עֲזָבַנִי

קָצַר לְשׁוֹנִי מִלְּסַפֵּר עוֹד / כָּל נִפְלְאוֹתָיו רָאֲתָה עֵינִי

רַגְלִי הֲלֹא הֶעֱמִיד בְּתוֹךְ מִישׁוֹר / לוּלֵי יי שֶׁעֲזָרַנִי

שַׁתִּי בְּאֵל בִּטְחִי לְשֶׁעָבַר / מִן הָעֲתִידוֹת אַל תְּבִישֵׁנִי

תַּכְלִית דְּבָרַי אֶשְׁאֲלָה תָּמִיד / בֵּין הַחֲסִידִים אֵל יְזַכֵּנִי.
