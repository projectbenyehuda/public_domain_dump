אֶל הַכּוֹכָבִים

אֶל פַּאֲתֵי הַיָּמָּה בָּאָה הַשֶּׁמֶשׁ

וּבְשַׁחַק וָאָרֶץ לַיִל יַצִּיעַ,

וּלְרַגְלוֹ פָּשְׁטוּ דֻמִיָּה וָאֶמֶשׁ,

וַיֵּצְאוּ חוֹצֵץ כָּל צִבְאוֹת רָקִיעַ!


וּכְאִיֵּי זָהָב עַל יָם הַשָּׁמַיִם

יִשְׁאָפוּ יִזְרָחוּ, יַחֲנוּ יִסָּעוּ,

יֵלְכוּ יִדְאוּ וִישׁוֹטְטוּ כָּעֵינַיִם

עַד אֹרַח אֵין-קֵץ וּנְתִיב לֹא יָדָעוּ.


אָז חַם בִּי לִבִּי, אַךְ רוּחִי נָהָרָה:

"עִמְדוּ, קָרָאתִי, עִמְדוּ עַד אַרְגִּיעַ!

(מִנַּהֲמַת לִבִּי עַל פִּי אֵשׁ בָּעָרָה)

הַגִּידוּ מִי אַתֶּם, צִבְאוֹת רָקִיעַ?


מִי זֶה אַתֵּמָה, חֵיל נוֹרָא, חֵיל נָאוֹר,

אֵין קֵצֶה, אֵין סוֹף, אֵין רֵאשִׁית וּתְחִלָּה?

הַאֲרָצוֹת הֵמָּה כָּל כּוֹכָב וּמָאוֹר?

אָבְדוּ עֶשְׁתֹּנוֹתַי וּבְפִי אֵין מִלָּה!


אוֹ רַעְיוֹנֵי-אֵל וּשְׂעִפֵּי אֲלֹהַּ

עַל סִפְרֵי שַׁחַק חֲקוּקִים אַתֵּמָה?

הַחֵקֶר עֶלְיוֹן וּבִינַת גָּבֹהַּ

יִמְצָא הַיּוֹדֵעַ לִקְרֹא בָהֵמָּה?

אוֹ גָדְלוֹ תָרֹנּוּ שִׁירֵי אֲהָבִים,

וּמְזַמְּרֵי אֵל אַתֶּם, שָׁרֵי-שָׁמָיִם!

וּפִיתָגוֹר וַאֲפַלְטוֹן, חוֹלְמִים נִשְׂגָבִים,

לַאֲזִין שִׁירֵיכֶם אָז כָּרוּ אָזְנָיִם?


אוֹ כִּי יִנְווּ בָכֶם נַפְשׁוֹת הַיְשָׁרִים

בֵּין עַרְבוֹת אוֹרְכֶם, בֵּין זָהֳרֵי צַחְצָחוֹת

כִּי יִטְּשׁוּ אֶרֶץ בָּהּ מָוֶת וּקְבָרִים,

הִתְעַלֵּס בִּנְעִימוֹת נֶצַח וּשְׂמָחוֹת!


הוֹי, מִי אַתֵּמָה? לִפְנֵיכֶם אוֹחִילָה!

וּצְעִיף הַסֵּתֶר נָא קִרְעוּ מֵעֲלֵיכֶם!

עַל חֶבְיוֹן עֻזְּכֶם לוּ רֶגַע אָגִילָה,

לוּ שָׁדוּד אַף אֶפֹּל אֶל זִיק קַוֵּיכֶם!


הֵן אֶזְכּוֹר עֵת חָלְפוּ לִי שָׁנִים שֶׁבַע

אַךְ דָּרְכָה רַגְלִי עַל מִפְתַּן הַחַיִּים,

אָז עָמַד הַיֶּלֶד עַל שִׂיא הַגֶּבַע

וּכְבַר אֲלֵיכֶם הוּא נָשָׂא עֵינַיִם.


לִבִּי בִּי הִשְׁתּוֹמֵם, נַפְשִׁי תָשׁוּחַ,

נָזְלוּ עֵינַי לָכֶם דִּמְעוֹת עֲדָנִים,

וּכְאִלּוּ עַל פָּנַי אָז יַחֲלוֹף רוּחַ,

וּלְאָזְנַי לָאַט סוֹד שִׂיחַ שִׁנְאַנִּים.


בֶּן עֶשֶׂר וּשְׁתַּיִם הָיִיתִי נַעַר,

אָז “מִי אַתֶּם?” לָכֶם מִלִּים עָרַכְתִּי,

וּשְׂפָתַי צָלְלוּ, עָלַי סִמֵּר שַׂעַר,

בַּהֲגִיגִי אֵשׁ לָכֶם לִבִּי שָׁפַכְתִּי!


עַתָּה קוֹל עֶלֶם נָא שִׁמְעוּ שָׁמֹעַ,

וּמְעַטִּים אִם יָמַי פֹּה עֲלֵי אָרֶץ,

אַךְ לִבִּי בִּי זָקֵן יִגְוַע גָּוֹעַ

אַחֲרֵי רָאָה עֳנִי חֳלִי וָפָרֶץ!


עֲנוּנִי נָא עֲנוּנִי, שׁוֹכְנֵי רָמָה!

הֲלִיכוֹת עוֹלָם נָא עִצְרוּ כְרֶגַע!

אֲהָהּ, קָצָה נַפְשִׁי בִמְלֹא הָאֲדָמָה

בָּהּ יֻלַּד אֱנוֹשׁ לְעָמָל וָפֶגַע!


פֹּה תִנְוֶה רִשְׁעָה גַּם תָּגוּר עַוְלָתָה,

פֹּה דִבָּה נִתְעָבָה, הָהּ, תַּחֲלִיק פִּיהָ –

אֱמוּנַת שָׁוְא תָּפִיץ עֶבְרוֹת אֵימָתָה,

לִימִינָהּ נִצַּב הַשֶּׁקֶר אָחִיהָ!


אֲהָהּ, פֹּה שִׁנְאַת דָּת מוֹרָאָה נִגְאָלָה,

עַל פִּיהָ אֵל חַנּוּן, בִּימִינָהּ חֶרֶב

וּתְפַלֵּל אַף תִּכְרַע עֵת תַּעַשׂ כָּלָה,

וּבְשֵׁם אֵל סַלָּח תִּטְרוֹף כִּזְאֵב עֶרֶב!..


תֵּבֵל זוּ בָּרָא אֵל בַּחֲרוֹן אַפַּיִם

אָז הִשְׁלִיכָהּ מִפָּנָיו בַּחֲמַת קֶצֶף,

וַיַּרְא הַמָּוֶת וַיִּלְטוֹשׁ עֵינַיִם,

וַיַּעַט אֵלֶיהָ בָּחֳרִי וָשֶׁצֶף!


וּבֵין צִפָּרְנָיו יֹאחֲזֶנָּה לָנֶצַח,

אָז גַּם הָאָסוֹן לָהּ חָרַק שִׁנַּיִם,

אֶל הָאָדָם מִהֵר חִישׁ בַּחֲמַת רֶצַח

וִיעַנֶּה אוֹתוֹ כָּל עוֹדוֹ בַּחַיִּים!


הוֹי, מָוֶת אָיֹם, אֱלִיל כָּל הָאָרֶץ!

כֹּהֲנוֹ הַזְּמַן יַקְרִיב חִישׁ לוֹ אֲשָׁמִים –

מִזְבְּחוֹתָיו כָּל קֶבֶר עַל אַדְמַת-קֶרֶץ

וּמְלֹא הַיְקוּם זִבְחֵי אֱלִיל הַדָּמִים!


הוֹי, אֶזְכּוֹר, אֶשְׁתּוֹלֵל! הוֹי, שׁוֹד וָרֶצַח!

גַּם הַזְּמַן פֹּה אָיִן! אֵין כֹּל וּמְאוּמָה!

הַמָּוֶת הוּא הַזְּמָן, לוֹ גַם הַנֶּצַח.

הוֹי קָצָה נַפְשִׁי בַּתֵּבֵל הַזְּעוּמָה!


אֲלֵיכֶם שׁוֹכְנֵי עָל, נַפְשִׁי כָלָתָה!

פִּתְחוּ שִׂפְתוֹתֵיכֶם, שִׂפְתֵי עוֹלָמִים!

הוֹרוּ נָא הָעֶלֶם עָדֵיכֶם אָתָה,

אֶל אוֹרְכֶם וַאֲמִתְּכֶם מוֹרָשָׁיו צָמִים!


הֲגַם בֵּינֵיכֶם רִשְׁעָה מִתְקַדֶּשֶׁת,

עוֹשְׁקָה וּבוֹכָה, חוֹמְסָה וּמִתְפַּלֶּלֶת?

הֲגַם שָׁם הַחֲנוּפָה יִרְאָה לוֹבֶשֶׁת

וֹבָרֵי לֵבָב בַּשַּׁחַת טוֹבֶלֶת?

הֲתִרְאוּ שָׁם אֱמֶת עַיִן לָעָיִן?

תֶּחֱזוּ סוֹד הַבְּרִיאָה מֵאִישׁ נִשְׂגָבָה?

לָדַעַת אֵיךְ רִקְמָה יֵשׁ רַב מֵאַיִן,

חקֹ הַנֵּצַח – בּוֹ הִיא רָצָה אַף שָׁבָה!


אִמְרוּ אִם עוֹלֶה מָוֶת גַּם אֲלֵיכֶם?

הֲתִגְוַע גַם עַיִשׁ שָׁם עַל בָּנֶיהָ?

אוֹ הֲיֵשׁ גַּם שָׂב גַּם יָשִׁישׁ בֵּינֵיכֶם?

אִם זָקְנָה שֶׁמֶשׁ, אִם כָּהוּ עֵינֶיהָ?


אוֹ אִם חַיֵּי עַד לָכֶם נִתְּנוּ מָנָה

וּמִסְּפוֹר חֲדַלְתֶּם שָׁנִים גַּם יָמִים?

וּבְנֵי אָדָם מָה הֵם, אִמְרוּ נֶאֱמָנָה!

גַּם מָה הֵמָּה לָהֶם חַיֵּי עוֹלָמִים?


הוּי, אֶת מִי זֶה אֶשְׁאַל? הוֹי, רַעְיוֹן יֶלֶד!

לָכֶם גַּם מוֹרָשָׁי לָרִיק אַךְ חָלוּ –

זֶה שָׁנִים אֵין חֵקֶר כָּל חַכְמֵי חֶלֶד

אֶתְכֶם יִדְרוֹשׁוּ, יַחְקוֹרוּ, יִשְׁאָלוּ!


גַּם קוֹרְאֵי סִפְרֵי רוֹם, חוֹבְרֵי שָׁמַיִם,

לָרִיק, אַךְ לָרִיק לָכֶם נָשְׂאוּ מֵצַח –

וּלְאָדָם הֶבֶל לֹא תַטּוּ אָזְנַיִם,

אִלְּמִים חֵרְשִׁים תֵּלְכוּ הֲלִיכוֹת נֶצַח!


גַּם עוֹד אֲזַי בִּימֵי יַלְדוּת הָאָרֶץ

עֵת לִבְנֵי אָדָם הֱיִיתֶם לֵאלֹהִים,

שַׁלִּיטִים עַל כֹּל, עַל חַיִּים וָקָרֶץ

וַעֲשַׁן בָּמוֹתֵיכֶם עוֹלֶה לִגְבוֹהִים;


גַּם אַחֲרֵי כֵן עֲצַת שַׁדַּי כִּי קָמָה

וַיּוֹכַח דַּעַת כִּי אֵין אֵל עִמָּדוֹ –

אַתֶּם הֶחֱרַשְׁתֶּם וּלְשׁוֹנְכֶם נֶאֱלָמָה,

כִּי שָׂם חֲסִין-יָהּ עַל פִּיכֶם אֶת יָדוֹ!
