ספר תחכמוני - שַׁעַר עֶשְׂרִים וְאֶחָד

בְּעָרְמַת חֶבֶר הַקֵּינִי. עִם הַכָּפְרִי הֶעָנִי


נְאֻם הֵימָן הָאֶזְרָחִי: נָטִיתִי לִבִּי. בְּעוֹדִי רַעֲנָן בְּאִבִּי. לְשׁוֹטֵט עַל הַמְּדִינוֹת הָעֲדִינוֹת. וְלִלְמֹד מִפִּי נְבוֹנִים תְּבוּנוֹת. וְלִלְקֹט מִפְּרִי לְשׁוֹנָם מָנוֹת. וַיְהִי הַיּוֹם וָאָבוֹא לְעִיר רְחוֹבוֹת. וּבְעוֹדִי מְסוֹבֵב בַּשְּׁוָקִים וּבָרְחוֹבוֹת. וְהִנֵּה לִקְרָאתִי חֶבֶר הַקֵּינִי וְכִרְאוֹתוֹ אוֹתִי רָץ אֵלַי וּפִיו מָלֵא שְׂחוֹק. וְנִשְׁמַע קוֹלוֹ לְמֵרָחוֹק. וָאָסוּר אֵלָיו. לִקְרֹא שְׁלוֹמִי עָלָיו. וְאָמַרְתִּי: מִפְּנֵי מָה צָחָקְתָּ. וּמַה לְךָ כִּי נִזְעָקְתָּ. אָמַר: הֶרֶף וְאַגִּידָה לְךָ דָבָר יְמַלֵּא פִיךָ שְׂחוֹק. לִבְלִי חֹק. אָמַרְתִּי לוֹ: הַגֶּד נָא לִי הֲבָלֶיךָ. וְדִבְרֵי הֲתוּלֶיךָ. אָמַר: הִשְׁכַּמְתִּי הַיּוֹם וְנַפְשִׁי שׁוֹקֵקָה. וְיָדִי רֵיקָה. וְאִוִּיתִי לְאֶכֹל בָּשָׂר. וְלִבִּי בְּשׁוֹט רָעָב מְיֻסָּר. וְאֵין בְּיָדִי אֲגוֹרַת כֶּסֶף. לִמְצֹא אֲשֶׁר אֲנִי כוֹסֵף. וְכַאֲשֶׁר דָּלַק בְּלִבִּי יְקוֹד הָרָעָב הַמַּבְעִיר. אָמַרְתִּי: אָקוּמָה נָא וַאֲסוֹבְבָה בָּעִיר. אוּלַי הָאֵל רַחֲמָיו יָעִיר. וָאָקוּם וַאֲסוֹבֵב בְּכָל עֵבֶר. אוּלַי יֵשׁ תִּקְוָה וָשֵׂבֶר. וְאֵין מָזוֹן וָשֶׁבֶר. כִּי אִם יָגוֹן וָשֶׁבֶר. יוֹרִיד אֶל הַקֶּבֶר. וְכַאֲשֶׁר נוֹאַשְׁתִּי. מֵאֲשֶׁר דָּרַשְׁתִּי. וְנוֹאַלְתִּי. מֵאֲשֶׁר הוֹחַלְתִּי. פָנִיתִי אַחֲרֵי צֵאתִי. לָשׁוּב לְבֵיתִי. בְּשׁוּרִי כִּי אָבְדָה תִקְוָתִי. וְלָרִיק יְגִיעָתִי. אֲנִי טֶרֶם אֲכַלֶּה לְדַבֵּר אֶל לִבִּי. וָאֶשָּׂא עֵינַי וָאֶרְאֶה וְהִנֵּה אִישׁ עֲרָבִי. יָבֹא מֵעִירוֹ וּמִכְּפָרוֹ. וְהוּא רוֹכֵב עַל חֳמוֹרוֹ. וְכַאֲשֶׁר אוֹתוֹ רָאיתִי. אָמַרְתִּי בְלִבִּי: מָצָאתִי אֲשֶׁר אִוִּיתִי. וְזֶה הַיּוֹם שֶׁקִּוִּיתִי. וּמִהַרְתִּי אֵלָיו. לִקְרֹא שְׁלוֹמִי עָלָיו. וְאָמַרְתִּי אֵלָיו. שָׁלוֹם בּוֹאֲךָ אֲדוֹנִי. אֲבִידָן בֶּן גִּדְעוֹנִי. מֵאַיִן מוֹצָאָךְ. וְאָנָה מוֹבָאָךְ. בָּרוּךְ אֲשֶׁר זִכַּנִי לִרְאוֹת מַלְאָךְ. וְעַל כֵּן יָצָאתִי לִקְרָאתֶךָ. לְשַׁחֵר פָנֶיךָ וָאֶמְצָאֶךָ. כִּי כָל הַיָּמִים הָיִיתִי נִכְסָף אֵלֶיךָ. וְשׁוֹאֵל לְכָל עוֹבֵר וָשָׁב עָלֶיךָ. אָמַר הַכָּפְרִי: שָׁלוֹם עָלֶיךָ אֲדוֹנִי. אֲבָל אֵינִי אֲבִידָן בֶּן גִּדְעוֹנִי. רַק אֲבִידָע הַגִּבְעוֹנִי. אָמַרְתִּי לוֹ: שְׁנֵי הַשֵּׁמוֹת נִמְשָׁלִים יָחַד. וְהַכֹּל הוֹלֵךְ אֶל מָקוֹם אֶחָד. וְעַתָּה הוֹדִיעֵנִי שְׁלוֹמְךָ. וְאֵי זֶה מְקוֹמְךָ. וְהוֹדִיעֵנִי שְׁלוֹם אָבִיךָ וְאִמְּךָ. אִם הֵם כַּאֲשֶׁר עֲזַבְתִּים בְּכֹחָם. אוֹ כָהֲתָה עֵינָם וְנָס לֵחָם. וַיֵאָנַח הַכָּפְרִי. בְּשָׁמְעוֹ דְבָרִי. וְאָמַר: נֶאֶסְפוּ בְרֹב עֹנִי לְעַמָּם. וְנָסְעוּ לְבֵית עוֹלָמָם. סָעוּ הֵמָּה לִמְנוּחוֹת. וְעָזְבוּ אוֹתִי לַאֲנָחוֹת. וּכְשָׁמְעִי דְבָרוֹ שָׁלַחְתִּי יָדִי. לִקְרֹעַ בִּגְדִי. וַיַּחֲזֵק הַכָּפְרִי בְיָדִי וַיֹּאמֶר אֵלַי: בְּחַיֵּי יוֹלְדֶיךָ. אַל תִּקְרַע בְּגָדֶיךָ. כִּי הַדְּאָגָה לֹא תוֹעִיל. וּמִמָּוֶת לֹא תַצִּיל. אָמַרְתִּי לוֹ: אִלוּ עָשִׂיתִי כָרָאוּי קָרַעְתִּי כְבֵדִי. בִּמְקוֹם בִּגְדִי.

וְשָׁפַכְתִּי דְמֵי הַצְּלָעוֹת. תַּחַת הַדְּמָעוֹת. עַל אָח נֶאֱמָן. אֲשֶׁר הָיָה לִי לְמִגְדַּל עזֹ מִקּוֹרוֹת הֵזְּמָן. כַּמָּה טוֹבוֹת גְּמָלַנִי. וְכַמָּה חֲסָדִים הֶעֱנִיקַנִי. אֲבָל רָאוּי לְכָל מֵבִין. לְהַצְדִּיק אֶת הַדִּין. וְעַתָּה יְדִידִי הִנָּחֵם. וּבֹא הַבַּיְתָה לֶאֱכֹל פַּת לֶחֶם. אוֹ נֵלֵךְ לַשּׁוּק. וְאֶקְנֶה לְךָ צְלִי אֵשׁ נֶחְמָד וְחָשׁוּק. וְכִשְׁמֹעַ הַכָּפְרִי זֵכֶר הַבָּשָׂר גָּרְסָה נַפְשׁוֹ לְתַאֲוָה. וְרָץ אַחֲרַי בְּאַהֲבָה. וַיְמַהֵר אִתִּי. וְלֹא יָדַע כִּי נָפַל בְּרִשְׁתִּי. וְכַאֲשֶׁר בָּאנוּ אֶל הַשּׁוּק מָצָאנוּ צְלִי אֵשׁ תַּאֲוָה לָעֵינַיִם. יוֹשֵׁב בֵּין תַּנּוּר וְכִירַיִם. יִטְּפוּ מֵימָיו בְּחֻמָּם. וְכִמְעַט מִיקוֹד הָאֵשׁ יָזוּב בְּשָׂרָם עַל עַצְמָם. וּלְשׁוֹן הַלַּהַב בִּשֵּׁל נְתָחָיו. וְרִתַּח רְתָחָיו. וְהִפְרִיחַ פְּרָחָיו. וּבְחֻמּוֹ הֶאֱדִיבָם. וּלְמַעֲצֵבָה הִשְׁכִּיבָם. עַד הֱשִׁיבָם. זַכִּים וְצַחִים. וְרַכִּים וְלַחִים. כְּגִזְרוֹת בְּדֹלָחִם. מִמַּעַל אָדְמוּ עֶצֶם מִפְּנִינִים. וּמִתַּחַת כַּשֶּׁלֶג לְבָנִים. כְּלִבְנַת הַכְּפוֹר לִבְנָתָם. סַפִּיר גִּזְרָתָם. וַעֲלֵיהֶם שֹׁמֶן בַּר וָלַח. וְעֵינוֹ כְּעֵין הַבְּדֹלַח. טָהוֹר קֹדֶשׁ מְמֻלָּח. לֵב רוֹאָיו יִפְלַח. וְאֵשׁ בְּצַלְעוֹתָיו יְשַׁלַּח. וְכַאֲשֶׁר קָרַבְנוּ לְבַעַל הַבָּשָׂר. אָמַרְתִּי לוֹ: שְׁקֹל מִן הַבָּשָׂר. לַאֲבִידָן חֲבֵרֵנוּ. כִּי הוּא אָחִינוּ בְּשָׂרֵנוּ. וּבְחַר לָנוּ הַנְּתָחִים הַטּוֹבִים. הַשְּׁמֵנִים וְהָעֳרֵבִים. הָרַכִּים וְהַדְּשֵׁנִים. הָאֲדֻמִּים וְהַלְּבָנִים. הַנָּאִים לְמַרְאֵה עֵינַיִם. הָעֲרֵבִים לַשְּׂפָתַיִם. הַמְּתוּקִים לַחֵךְ וְלַשִּׁנַּיִם.

וַיִּשְׁלַח הָאִישׁ אֶת יָדוֹ וַיִקַּח אֶת הַמַּאֲכֶלֶת. לַחְתֹּךְ מִן הַחַיָּה הַנֶּאֱכֶלֶת. וְהִנֵּה הִיא אֲדֻמָּה כְּגַחֶלֶת. וְכִלְבַב מְקַנְּאִים מְבֻשֶּׁלֶת. וּבְטַעְמָהּ לְנֹפֶת צוּף נִמְשֶׁלֶת. וְרֵיחָהּ כְּקִנָּמוֹן וּשְׁחֵלֶת. וְהֵבִיא לָנוּ לֶחֶם סֹלֶת. וַיִתֵּן לְפָנֵינוּ. וְאָכַלְנוּ שְׁנֵינוּ. עַד קָהוּ שִׁנֵּינוּ. וְדָבְקָה לָאָרֶץ בִּטְנֵנוּ. וְכַאֲשֶׁר כִּלִּינוּ לֶאֱכֹל. וְלִלְחֹךְ הַכֹּל.

אָמַרְתִּי לְבַעַל הַבָּשָׂר: תֵּן לַאֲבִידָן חֲבֵרֵנוּ מִן הַשְּׁקֵדִים הַמְבֻשָּׁלִים בִּדְבָשׁ. אֲשֶׁר טַעְמָם כָּל צִיר יֶחְבָּשׁ. וְיִהְיֶה מַה שֶּׁתִּתֵּן חָדָשׁ בֶּן לֵילוֹ. כַּיָּרֵחַ בְּהִלּוֹ. טַעְמוֹ מָתוֹק וְדָשֵׁן. יְעוֹרֵר אֵלָיו הַיָּשֵׁן. וְיִשְׁכַּב בַּפֶּה עַל מִטּוֹת שֵׁן. יִמַּס לְמַרְאֵה הָעֵינַיִם. וְיָזוּב בֵּין הַיָּדַיִם. וּבְטֶרֶם יִכְתְּשׁוּהוּ הַשִּׁנַּיִם. יִמְשְׁכוּהוּ הַמַּלְקוֹחַיִם. בְּלִי מֶלְקָחַיִם. וְיִצְפְּנוּהוּ בְחַדְרֵי הַמֵּעַיִם. וַיְמַהֵר הָאִישׁ בְּשָׁמְעוֹ הַקּוֹל. לָתֵת וְלִשְׁקֹל. וְלָשׂוּם עַל יְדֵי הַכָּפְרִי הַכֹּל. וַנֹּאכַל גַּם אָכוֹל. וְשַׁבְנוּ שְׁנֵינוּ. לִנְשֹׁךְ בְּשִׁנֵּינוּ. וְלִמְשֹׁךְ בִּלְשׁוֹנֵנוּ. וְכַאֲשֶׁר הִשְׁלַמְנוּ הָעֲבוֹדָה. וְעָרַכְנוּ לָאֵל תּוֹדָה. אָמַרְתִּי אֶל הַכָּפְרִי: אֲנַחְנוּ צְרִיכִים לְמַיִם קָרִים. לְכַבּוֹת אֵלּוּ הַמַּאֲכָלוֹת הַבּוֹעֲרִים. וְאַתָּה פָּה עֲמֹד עַד אֶקְרַב לְזֶה הַשַּׁעַר. וְאֶקְרָא אֶל הַנַּעַר. וְיָבִיא לָנוּ מַיִם טוֹבִים. כְּגִזְרוֹת שֶׁלֶג עֲרֵבִים. מֵהַר שְׂנִיר חֲצוּבִים. יְכַבּוּ מִן הַקְּרָבִים. מוֹקְדֵי לְהָבִים. אָמַר הַכָּפְרִי: עֲשֵׂה כַּאֲשֶׁר דִּבַּרְתָּ. וְיָפֶה גָּזַרְתָּ. וָאֵרֵד מִן הַמַּצֵּבָה. וְעָמַדְתִּי אַחֲרֵי הַקִּיר נֶחְבָּא. לְדֵעָה מַה יֵעָשֶׂה לוֹ. וּמַה יַעֲלֶה בְגוֹרָלוֹ. וְכַאֲשֶׁר רָאָה כִּי בּוֹשַׁשְׁתִּי לָבוֹא. וַיִּתְעַצֵּב אֶל לִבּוֹ. וְקָם לִקְשֹׁר אֲזוֹרוֹ. וְלִרְכֹּב עַל חֲמוֹרוֹ. וַיְמַהֵר בַּעַל הַבָּשָׂר אֵלָיו. וְהֶחֱזִיק בְּשׁוּלָיו. וְאָמַר לוֹ: אָנָה תֵלֵךְ וְחוֹבְךָ לֹא פָרַעְתָּ. תֵּן מְחִיר מַה שֶּׁאָכַלְתָּ. אָמַר הַכָּפְרִי: סוּר מֵעָלַי. וְאַל תִּקְרַב אֵלַי. כִּי אֲנִי דָבָר לֹא קָנִיתִי. כִּי אַכְסְנַי הָיִיתִי. וַיַּחֲזֵק הָאִישׁ בְּעָרְפּוֹ. וְהִכָּהוּ בְכַפּוֹ. וְאָמַר לוֹ: בֶּן הַזּוֹנָה מָתַי קְרָאנוּךָ. אוֹ מֵאַיִן יְדַעְנוּךָ.

תֵּן שְׁלשָׁה כְסָפִים וְאִם אַיִן סְחַבְנוּךָ. עַד תָּקִיא כֹּל אֲשֶׁר בָּלַעְתָּ. וְתִרְעַב כִּפְלַיִם מֵאֲשֶׁר שָׂבַעְתָּ. וַיֵּבְךְּ הַכְּפִרי וַיֹאמַר: אָרוּר יִהְיֶה אוֹתוֹ הַנָּבָל. אֲשֶׁר עַל אָבִי הִתְאַבָּל. כִּי אֲנִי אָמַרְתִּי לוֹ שְׁמִי אֲבִידָע הַגִּבְעוֹנִי. וְהוּא מִתְעַקֵּשׁ וְקוֹרֵא שְׁמִי אֲבִידָן בֶּן גִּדְעוֹנִי. עַד נְשָׁכַנִי כְצִפְעוֹנִי. וֶהֱפִיצוּנִי יְמֵי עֹנִי. וְלֹא יָכוֹל לְהִמָּלֵט עַד שָׁקַל הַמָּעוֹת. וַיֵלֶךְ וּבִלְחָיָיו פַּלְגֵי דְמָעוֹת. וְכַאֲשֶׁר הָלַךְ לְעִירוֹ. שְׁמַעְתִּיהוּ בּוֹכֶה בְצִירוֹ.

וְאוֹמֵר בְּשִׁירוֹ:

נָבָל אֲשֶׁר הִבְעִיר בְּאֵשׁ לִבִּי

מִי יִתְּנֵהוּ מַאֲכַל חַרְבִּי.

כִּי מִהְיוֹתִי לֹא רְאִיתִיהוּ

עַד בָּא לְרַמּוֹתִי וְהֵתֵל בִּי.
