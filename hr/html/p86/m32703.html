ספר תחכמוני-שַׁעַר רִאשׁוֹן

מוֹדִיעַ הֲוָיַת הַסֵּפֶר, וְעַל שֵׁם מִי חִבְּרוֹ בְּאִמְרֵי שֶׁפֶר


נְאֻם הֵימָן הָאֶזְרָחִי הַגֵּד הִגִּיד לִי מְחַבֵּר הַסֵּפֶר הַזֶּה מִפְּנִינֵי רַעְיוֹנוֹ. וּמִמְּלִיצוֹת לְשׁוֹנוֹ. וַיֹּאמַר: נִקְרֹא נִקְרֵיתִי בְקִרְיַת יְעָרִים. עִם חֲבֵרִים גְבָרִים. מִיַּלְדֵי הָעִבְרִים. מִלֵּיהֶם סַפִּירִים. וִידֵיהֶם נְהָרִים. וּפְנֵיהֶם מְאוֹרִים. בִּרְקִיעַ הַתְּהִלָּה מְאִירִים. וּבִמְרוֹם הַגְּדֻלָּה מַזְהִירִים. בְּאוֹרָם כּוֹכְבֵי שַׁחַק יַעְלִימוּ. וּפְנֵי שַׁחַר יַכְלִימוּ. וְלַיְלָה לְיוֹם יָשִׂימוּ. עָלוּ בְסֻלַּם שִׂכְלָם עַד חוּג עֶרֶץ. וַיִּתֵּן אוֹתָם אֱלֹהִים בִּרְקִיעַ הַשָּׁמַיִם לְהָאִיר עַל הָאָרֶץ. לְאוֹרָם הַשַּׁחַר נִדְמֹה נִדְמָה. וְחָפְרָה הַלְּבָנָה וּבוֹשָׁה הַחַמָּה. בְּלֶכְתָּם בְּמַחֲשַׁכִּים עַמּוּד אֵשׁ הַשֵּׂכֶל יָאִיר לְעֵינֵיהֶם. וַעֲנַן הָעֲנָוָה עוֹמֵד עֲלֵיהֶם. וַאֲרוֹן הַצְּדָקָה נוֹסֵעַ לִפְנֵיהֶם. וּבִמְקוֹם אֲשֶׁר יִשְׁכֹּן עֲנַן כְּבוֹדָם שָׁם הַתְּהִלּוֹת יִרְגָּעוּ. וּבַעֲלוֹתוֹ יִסָּעוּ. מַחְשְׁבוֹתָם הָרוֹת וְעַל מַשְׁבֵּר לֹא עָמָדוּ. בְּטֶרֶם תָּבוֹא אֲלֵיהֶן הַמְיַלֶּדֶת וְיָלָדוּ. יְלָדִים אֵין בָּהֶם כָּל מוּם וְכָל דֹפִי. מְלֵאִים חָכְמָה וּכְלִיל יֹפִי. הֵכִינוּ לַתּוֹרָה בְלִבָּם מַחֲנֶה. וְהַשֵּׂכֶל עַל פִּיהֶם יִסַּע וְיַחֲנֶה. וְהָיָה טֶרֶם יִקְרָאוּ וְהוּא יַעֲנֶה. אִמְרָתָם צְרוּפָה. וּמְלִיצָתָם רְצוּפָה. וְנִדְבָתָם עֲדוּפָה. וּלְשׁוֹנָם חֶרֶב שְׁלוּפָה. מִזֹּהַר עִנְיָנֶיהָ לַפִּידִים יְהַלֵּכוּ. וּמִנִּצּוֹצֵי נִיבֶיהָ תְּשַׁלַּח בְּרָקִים וְיֵלֵכוּ. וּמִלַּהַט יְקוֹדֶיהָ אוֹחֶזֶת לְבָבוֹת וּבָבוֹת. בְּנִיבוֹת עֲרֵבוֹת. מִלֶּהָבוֹת חֲצוּבוֹת. פַּעַם עוֹרֶכֶת קְרָבוֹת. וּפַעַם מְעוֹרֶרֶת אֳהָבוֹת. בְּמִלּוֹת הַנְפָשׁוֹת מְשִׁיבוֹת. וְהַלְּבָבוֹת מְדִיבוֹת. וְשָׁם אִתָּנוּ נַעַר עִבְרִי. בִּמְרוֹם הַתְּבוּנָה יַמְרִיא. וְעַל נַחֲלֵי אֱמוּנָה יַפְרִיא. הַגִּבּוֹרִים יֵחַתּוּ לְקוֹל שְׁאוֹנָיו. וְהָעֲנָקִים נִבְהֲלוּ מִפָּנָיו. הַשֶּׁמֶשׁ בְאוֹרוֹ תְּקַנֵּא. וְהַיֹּפִי עַל פָּנָיו יַחֲנֶה. וּבְאֵימָתוֹ לֵב הַכְּפִירִים יַמְסֶה. וְהַחָכְמָה תַחת כְּנָפָיו תֶּחֱסֶה. וְיַשְׂכִּיל בְּכָל אֲשֶׁר יַעֲשֶׂה. פַּעַם דְּמֵי הַכּוֹס יִמְצֶה. וּפַעַם יַבִּיעַ נוֹרָאוֹת בְּעֵת פִיהוּ יִפְצֶה. אִם יִהְיֶה בְּבֵית מִשְׁתֵּה הַיַּיִן. תִּרְאֶה בּוֹ כָּל מַחְמַד עַיִן. וְאִם יְנַגֵּן. הַיָּגוֹן בְּיַד הַשִּׂמְחָה יְמַגֵּן. וּלְשׁוֹמְעָיו עֲטֶרֶת תִּפְאֶרֶת יְמַגֵּן. וְאִם שִׁיר יְשׁוֹרֵר. מֵתִים יְעוֹרֵר. וּסְלָעִים יְפוֹרֵר. וְאִם יָבוֹא בְּמִלְחֶמֶת תְּעוּדוֹת. יַלְבִּישׁ הַזְּמַן בִּגְדֵי חֲרָדוֹת. וְיַפְשִׁיט גִּבּוֹרֵי הַחָכְמָה בִּגְדֵיהֶם הַחֲמֻדּוֹת. וְעָלָיו תְּהִלּוֹת עוֹלָם. יִשְׂאוּ קוֹלָם:

הַלֵּב יוֹרֶה. אֶל קוֹל קוֹרֵא:

מִי הָאִישׁ הֶחָפֵץ חַיִּים.

יָרוּץ לַגְּבִיר. חַיִל מַגְבִּיר.

יוֹדִיעֶנּוּ אֹרַח חַיִּים.

כָּל לֵב הוֹמֶה. אֶל בִּין צָמֵא.

נָתַן עָלָיו מַיִם חֶיִּים.

תַּחַת שִׂיאוֹ. יִכָּנַע שׂוֹנְאוֹ.

אַךְ כָּל מוֹצְאוֹ מָצָא חַיִּים.

כָּל רוֹאֶנּוּ. יֶאֱהָבֶנּוּ.

כִּי מִמֶּנּוּ תוֹצְאוֹת חַיִּים.

כִּי בִלְשׁוֹנוֹ. וּבְהֶגְיוֹנוֹ

צִירֵי מָוֶת וּצְרִי חַיִּים.

לַהַט חֶרֶב מִתְהַפֶּכֶת

לִשְׁמֹר דֶּרֶךְ עֵץ הַחַיִּים.

וַיַּעַן וַיֹּאמֶר. אֶל הָעוֹמְדִים לְפָנָיו לֵאמֹר: אֲלֵיכֶם אִישִׁים אֶקְרָא. מְשִׁיבֵי מִלְחֲמוֹת הַחָכְמוֹת שָׁעְרָה. הַרְאִיתֶם.

בְּכָל אֲשֶׁר חֲזִיתֶם. וְאִם יְדַעְתֶּם. בְּכָל אֲשֶׁר שְׁמַעְתֶּם. אוֹ מְצָאתֶם. בְּכָל אֲשֶׁר קְרָאתֶם. בִּמְלִיצוֹת הַיִּשְׁמְעֵאלִי אֲבִי הַחִידוֹת הַחֲמֻדוֹת. וְהַמְּלִיצוֹת הַנִּמְרָצוֹת. הַנִּקְרָא אַלְחֲרִירִי. הֲשַׂמְתֶּם לְבַבְכֶם אֵלָיו. וְאִם טְעַמְתֶּם דְּבַשׁ מִלָּיו. כִּי אֵין מְזַנֵּב נֶחֳשָׁלָיו. וְאֵין מַשִּׂיג בִּמְרוּצָתוֹ אֲבַק רַגְלָיו. מַה מָּתְקוּ מְשָׁלָיו. וּמַה נָּעֲמוּ מַהֲלָלָיו. צֵל עֵדֶן צְלָלָיו. וְטַל אוֹרוֹת טְלָלָיו. וּבְהַטִּיף עַל אִישׁ טַל מִלָּיו. יֵרֵד הַמָּן עָלָיו. וְאִלּוּ הַסֵּפֶר הַהוּא יְחֻבַּר בְּלָשׁוֹן אַחֶרֶת חוּץ מִלְּשׁוֹן עֲרָב. נָפַל מִמֶּנּוּ רַב. וְהָיָה אֲסַפְסוּף נֶעֱרָב. וְגַם עֵרֶב רָב. וְהָיְתָה מְלִיצָתוֹ נַחְלָה וְנִשְׁבֶּרֶת. וְלֹא יֶחְסַר בָּהּ כָּל מוּם רַע שָׁבוּר אוֹ עַוֶּרֶת. וּשְׂאֵת אוֹ סַפַּחַת אוֹ בַהֶרֶת. וְצָרַעַת מַמְאֶרֶת. כִּי לְשׁוֹן עֲרָב אֵין בְּכָל הַלְּשׁוֹנוֹת עֲרוֹךְ אֵלֶיהָ. בְּמֶתֶק מִלֶּיהָ. וְנֹעַם מְשָׁלֶיהָ. וְהִיא לְכָל שׁוֹמְעֶיהָ עֲרֵבָה וַאֲהוּבָה. מֵאֵשׁ חֲצוּבָה. וּמִנִּי יָם רְחָבָה.

אָמַר מְחַבֵּר הַסֵּפֶר הַזֶּה: וּכְשָׁמְעִי דְבָרָיו נֶהְפַּךְ עָלַי לִבִּי. וּנְשָׁמָה לֹא נוֹתְרָה בי. אַךְ הִתְאַפַּקְתִּי. וּבְדַבְּרוֹ עִמִּי הִתְחַזַּקְתִּי. וְאָמַרְתִּי לוֹ: הֵיטַבְתָּ. בְּכָל אֲשֶׁר אָמַרְתָּ. וְכֵן דִּבַּרְתָּ. כִּי לְשׁוֹן עֲרָב מִבְחַר כָּל הַלְּשׁוֹנוֹת. הָאַחֲרוֹנוֹת וְהָרִאשׁוֹנוֹת. וְכֻלָּם לְנֶגְדָּהּ מִשְׁתַּנּוֹת. וְדָתֵיהֶם שׁוֹנוֹת. וִּמלּוֹתָם בְּזוּיוֹת וּמְשֻׁנּוֹת. חוּץ מִלְּשׁוֹן הַקֹּדֶשׁ. כִּי בִּלְשׁוֹן הַקֹּדֶשׁ דִּבֶּר הָאֵל עִם נְבִיאָיו. וְהִקְדִּישׁ בְּסִינַי קְרוּאָיו. וְנָתַן דָּתוֹ. לְלוֹמְדֵי תְעוּדָתוֹ. וְחָלִילָה לְאֵל בּוֹרֵא כָל הַלְּשׁוֹנוֹת. לִבְחֹר מֵהֶן לְעַצְמוֹ הַשָּׂפָה הַחֲסֵרָה. וּלְדַבֵּר בַּלָּשׁוֹן הַצָּרָה הַקְּצֵרָה. אֲבָל מִן הַיָּדוּעַ כִּי לְשׁוֹן הַקֹּדֶשׁ מִימֵי קֶדֶם עָדְפָה וְגָדְלָה. וְרָחֲבָה וְנָסְבַה לְמָעְלָה. וּמֵעֵת גָּלִינוּ מֵאַרְצֵנוּ. לָמַדְנוּ לְשׁוֹנוֹת הַגּוֹיִם וְעָזַבְנוּ לְשׁוֹנֵנוּ. וְזוֹ הִיא סִבַּת קֹצֶר לְשׁוֹנֵנוּ וְחֶסְרוֹנָה. אַחֲרֵי תְמִימוּתָה וְיִתְרוֹנָהּ. וּמִיּוֹם הִתְעָרְבוּ בְּנֵי עַמֵּנוּ בֵּין הַגּוֹיִם וְשָׁכְנוּ בֵינָם. וְלָמְדוּ לְדַבֵּר בִּלְשׁוֹנָם. עָזְבוּ לָשׁוֹן עִבְרִיָּה. וּפָקוּ פְּלִילִיָּה.

וְנַפְשָׁם מָאֲסָה אֶת בֶּן שָׂרָה הָעִבְרִית. וְחָשְׁקָה אֶת־בֶּן־הָגָר הַמִּצְרִית. וְלָכֵן לְשׁוֹנֵנוּ אָבַד. וְרֻבּוֹ לִשְׁאֹל יָרַד. וּקְצָתוֹ שָׂרַד. אֲשֶׁר הִשִׁאִיר הַבָּרָד. אֲבַל בַּמְעַט הַהוּא נַפְלִיא לַעֲשׂוֹת. לַעֲרֹךְ אַבְנֵי נֵזֶר מִתְנוֹסְסוֹת. וּלְמַלֹּאת אֲמָרִים. מִכְּלֵי פַז יְקָרִים. וּלְחַבֵּר מְלִיצוֹת. בִּזְהַב הַיֳּפִי מְשֻׁבָּצוֹת. וְלִכְתּוֹב אִגְּרוֹת. וּלְשׁוֹרֵר שִׁירוֹת. וְלִשְׁלֹחַ סִפְרֵי תְשׁוּרוֹת. וְאִמְרֵי בְשׂוֹרוֹת. וְלַעֲשׂוֹת חִבּוּרִים. וּלְסַדֵּר סְדָרִים. בְּמִלִּים כְּמִלּוּאֵי תַּרְשִׁישִׁים. יְחַיּוּ הַנְּפָשִׁים. מְפֻתָּחִים עַל הַלֵּב פִּתּוּחֵי חוֹתָם. מְשֻׁבָּצִים זָהָב יִהְיוּ בְּמִלֻּאוֹתָם. וַתִּמָּלֵא הָאָרֶץ אוֹתָם. וְזֶה לְךָ הָאוֹת. כִּי לְשׁוֹנֵנוּ לְשׁוֹן נִפְלָאוֹת. וּדְבָרֶיהָ דִּבְרֵי נְבוּאוֹת. כִּי הִיא צָרָה – וְתִתְרַחֵב לָנוּ. וּקְצָרָה – וְתַסְפִּיק לְכֻלָּנוּ.

אָמַר הָאִישׁ: וְאִם אֱמֶת דְּבָרֶיךְ. וְצֶדֶק מַאֲמָרֶיךָ. מַדּוּעַ לֹא נִמְצָא בִּבְנֵי יִשְׂרָאֵל. אִישׁ אֲשֶׁר יְחַבֵּר כַּדְּבָרִים הָאֵל. אֲשֶׁר בִּלְשׁוֹן יִשְׁמָעֵאל. וְיִהְיֶה לִלְשׁוֹן הַקֹּדֶשׁ מוֹשִׁיעַ וְגוֹאֵל? אָמַרְתִּי לוֹ: וְכָל זֶה אֵינֶנּוּ רָחוֹק. וְהַחֵפֶץ הַזֶּה אֵינֶנּוּ לְדוֹרְשָׁיו עָמֹק. וּבְעֵת נִרְצֶה לְחַבֵּר כְּזֹאת הַמְּגִלָּה. עַל פִּי תָקְפָּה וְגָדְלָהּ. עָלֹה נַעֲלֶה וְיָרַשְׁנוּ אוֹתָהּ כִּי יָכוֹל נוּכַל לָהּ. אֲבָל הַמְחַבֵּר לְמִי יְחַבֵּר. וְעַל אָזְנֵי מִי יְדַבֵּר. וְהָאָזְנַיִם חֵרְשׁוֹת. וְהַיָּדַיִם קָשׁוֹת. וְהַזְּמַנִים סָגְרוּ עֵינֵי הַיְצוּרִים. אֲשֶׁר בְּבֵית הַתַּאֲוָה עֲצוּרִים. וְאֶת הָאֲנָשִׁים אֲשֶׁר פֶּתַח הַבַּיִת הִכּוּ בַּסַּנְוֵרִים. וְאֵין בָּהֶם רוֹאֶה וְשׁוֹמֵעַ. וְאֵין שָׂם עַל לֵב וְאֵין יוֹדֵעַ. וְאִם תַּחְקֹר קְהִלוֹת תֵּבֵל. אֲשֶׁר מִמִּצְרַיִם וְעַד בָּבֶל. לֹא תִמְצָא אוֹהֵב חָכְמָה וּמְכַבֵּד בְּעָלֶיהּ. וְלֹא מֵשִׁיב גְּמוּל טוֹב עָלֶיהָ. וְאַתָּה יָדַעְתָּ כִּי הַסְּפָרִים הַיְקָרִים לֹא יְחֻבְּרוּ פְּנִינֵיהֶם. רַק לִמְבִינֵיהֶם. אוֹ לְמֵשִׁיב גְּמוּל טוֹב עֲלֵיהֶם. כִּי לֹא נַעֲשׂוּ לִכְסִילִים הַלוֹעֲגִים עֲלֵיהֶם. מִן הַמְתַעְתְּעִים בָּהֶם. וְכָל חָכְמָה נְעִימָה בִּשְׁלשָׁה תְנָאִים סוֹדָהּ יֵחָשֵׂף. בְּמָצְאָהּ נָדִיב אוֹ חָכָם אוֹ נִכְבָּד אֲשֶׁר אֵלֶיהָ כּוֹסֵף. תִּמָּכֶר־לוֹ לְאָמָה וְאוֹרָה יִזְרַח וְלֹא יֵאָסֵף. אִם שְׁלָשׁ־אֵלֶּה לֹא יֵעָשֶׂה לָהּ וְיָצְאָה חִנָּם אֵין כָּסֶף. וְעַל כֵּן בְּדוֹרֵנוּ זֶה תִּשְׁתַּפֵּכְנָה אַבְנֵי קֹדֶשׁ וְהַכֶּתֶם הַטּוֹב יִשְׁנֶה. וְהַשִּׁירִים וְהַמְּלִיצוֹת יִתְמַכְרוּ לַעֲבָדִים וְלִשְׁפָחוֹת וְאֵין קוֹנֶה. וְנִדְבַת הַנְּדִיבִים הִיא כְּמִשֶׁעֶנת קָנֶה. לָהּ בִּלְשׁוֹנוֹת מַעֲנֶה. וְאֵין לָהּ בִּלְבָבוֹת מַחֲנֶה. וּבְכָל מָקוֹם אֲשֶׁר אַחֲנֶה. אֶקְרָא: הֲיֵשׁ מַשְׂכִּיל? וְאֵין עוֹנֶה. וְהִנֵּה הַמָּקוֹם מְקוֹם מִקְנֶה. וּבְכָל עִיר וָעִיר נִרְאֶה כָל אֶחָד מֵהֶם בְּחֵיק הַסִּכְלוּת נִרְדָּם. וְהִנֵּה אֵין אִישׁ שָׁם וְקוֹל אָדָם. כִּי אִם הַסּוּס אָסוּר וְהַחֲמוֹר אָסוּר. בְּעוֹל הַתַּאֲוָה קָשׁוּר וּמָסוּר. וּבַדָּבָר הַזֶּה אֵיךְ תִּתְעוֹרֵר הַנְּשָׁמָה. אוֹ תִּכְסֹף הַנֶּפֶשׁ הַחֲכָמָה. לְדַבֵּר דָּבָר מִדִּבְרֵי הַחָכְמָה. אוֹ לְדַבֵּר בְּעִנְיָן מֵעִנְיְנֵי הַמְּזִמָּה. אוֹ לְמַלֹּאת אִגֶּרֶת נְעִימָה. וְהַחָכְמָה בְעֵינֵי בְנֵי עַמֵּנוּ כִּצְרוֹר אֶבֶן בְּמַרְגֵּמָה. אוֹ כְזוֹרֵק פְּנִינָיו לְרַגְלֵי הַבְּהֵמָה.

אָמַר הָאִישׁ: הִנֵּה יָדַעְתִּי כִּי כָל דְּבָרֶיךָ צֶדֶק. וְאֵין בִּבְנֵי הַדּוֹר מַחֲזִיק הַבֶּדֶק. וְטוֹבָם כַּחֶדֶק. אֲבָל הַעְלֵם עֵינֶיךָ מֵעַמְּךָ וְרֹעַ פְעֻלָּתָם. וְחֶסְרוֹן נִדְבָתָם. וְאַל תֵּפֶן אֶל מִנְחָתָם. וּבִגְלַל הָרַע אַל יֹאבַד הַטּוֹב. וּבַעֲבוּר קוֹץ וְדַרְדַּר אַל תַּעֲקֹר גַּן רָטֹב. כִּי אִם תּוּכַל לְחַבֵּר סֵפֶר כָּזֶה בְּנִיבִים עֲרֵבִים וְחִידוֹת חֲמוּדוֹת. וְשִׂיחִים צַחִים וְשָׂפָה יָפָה. וְלֶקַח מְרֻקָּח וְתוּשִׁיָּה יְפֵה־פִיָּה. וְשִׁירִים מְאֻשָּׁרִים וּמְלִיצוֹת נִמְרָצוֹת. תִּתֵּן פִּתְחוֹן פֶּה לְזֶרַע הַקֹּדֶשׁ. וּתְגַלֶּה לְעֵינֵי הַגּוֹיִם מַעֲלַת לְשׁוֹן הַקֹּדֶשׁ. וְעַתָּה לֵךְ וְאָנֹכִי אֶהְיֶה עִם פִּיךָ. וּבְכָל מַחְבֶּרֶת אֲשֶׁר תְּחַבֵּר. אֶהְיֶה עִמְּךָ וְהוֹרֵיתִיךָ אֲשֶׁר תְּדַבֵּר. וְאִם תְּבַקֵּשׁ אָדוֹן לְפָאֵר סִפְרְךָ בְּמַהֲלָלוֹ. וּלְהַרְאוֹת לְעֵינֵי הָעַמִּים אֶת כְּבוֹדוֹ וְאֶת גָּדְלוֹ. הִנֵּה רָאִיתִי בֵן לְיִשַׁי. לֹו יוֹבִילוּ מְלָכִים שַׁי. עָלָיו הוֹד הַמְּלוּכָה. כִּי בְּרִית עוֹלָם לוֹ שְׁמוּרָה וַעֲרוּכָה. וּבְיוֹם נוֹלַד בִּשַּׂר הָאֵל מִמְּרוֹמוֹ. לְעֵינֵי עַמּוֹ. הִנֵּה בֵן נוֹלַד לְבֵית דָּוִד יֹאשִׁיָּהוּ שְׁמוֹ. וַתְּהִי הַמִּשְׂרָה עַל שִׁכְמוֹ. וְנָפַח רוּחֹו בְּפִגְרֵי הַתְּהִלּוֹת וַיִּקְרְבוּ עֲצָמוֹת עֶצֶם עַל עַצְמוֹ. וְצֶלֶם הַיְקָר נִבְרָא בְּצַלְמוֹ. מַשְׂכִּיל בְּכָל דְּרָכָיו וַה' עִמּוֹ. וּמַפְתֵּחַ בֵּית דָּוִד עַל שִׁכְמוֹ.

נְשִׂיאוּת נִבְרְאָה אֶצְלוֹ לְעֵזֶר

וְצֶלֶם הַיְקָר נִבְרָא בְצַלְמוֹ;

וְשָׂם רוּחוֹ בְּפִגְרֵי הַתְּהִלּוֹת

עֲדֵי כִי קָרְבוּ עֶצֶם לְעַצְמוֹ.

וְלוֹ שְׁנֵי עֵדִים מְעִידִים עַל תַּעֲצוּמוֹ. כֶּתֶר רֹאשׁוֹ דָּוִד וְחוֹתַם יְמִינוֹ שְׁלֹמֹה. וְלָאִישׁ אֲשֶׁר אֵלֶּה לוֹ. עַל כּוֹכְבֵי שַׁחַק יָרִים רַגְלוֹ. וְיִיף בְּגָדְלוֹ. וּלְפָנָיו יִקְרְאוּ: אַבְרֵךְ נְשִׂיא דוֹרוֹ. כָּכָה יֵעָשֶׂה לָאִישׁ אֲשֶׁר הַמֶּלֶךְ חָפֵץ בִּיקָרוֹ. כִי הוּא אֵדֶר הַיְקָר. וּמַהֲלָלוֹ בְּלִי מֶחְקָר. בְּלִבּוֹ יָם תְּעוּדוֹת מֵאֵין מִדָּה. וּבִלְשׁוֹנוֹ אוֹצַר כָּל כְּלִי חֶמְדָּה. וּבְפִיהוּ מַעְיַן הַדֵּעוֹת. וּבִימִינוֹ חֹסֶן יְשׁוּעוֹת. כְּרוּב מִמְשָׁח. בְּשֶׁמֶן תְּהִלוֹת נִמְשַׁח. פָּנָיו נֵר מַעֲרָכָה. וְרַעְיוֹנָיו נִיר מַמְלָכָה. נְשִׂיאוּתוֹ מֵאָב לְבָנִים נִמְשָׁכָה. לְשֵׁם טוֹב וְיַחַס זָכָה. הָאֵל יַפִּיל בַּנְּעִימִים חַבְלוֹ. וְיָרֵם דִּגְלוֹ וְגָדְלוֹ. וּבְכִסֵּא הַמַּמְלָכָה בִּלְעָדָיו לֹא יָרִים אִישׁ אֶת יָדוֹ וְאֶת רַגְלוֹ. וְעַתָּה שְׁלַח סִפְרְךָ תְשׁוּרָה. לְמַרְבֵּה הַמִּשְרָה. וּלְשָׁלוֹם אֵין קֵץ עַל כִּסֵּא דָוִד וְעַל מַמְלַכְתּוֹ.


בְּעֹז שַׁדַּי יָעֹז. נְשִׂיא אֵל. הַמָּעֹז.

אֲשֶׁר הוּא מַטֵּה עֹז. וּמַקֵּל תִּפְאָרָה.

נְשִׂיא אֵל יֹאשִׁיָּה. לְרֹאשׁ הַיְקָר לִוְיָה.

בְּמַחֲשַׁכֵּי שִׁבְיָה. שְׁלָחוֹ צוּר אוֹרָה.

וְיַרְבֶּה הַשָּׁלוֹם. לְמַרְבֵּה הַמִּשְׂרָה.

כַּכָּתוּב: לְמַרְבֵּה הַמִּשְׂרָה (ישעיה ט, ו).

נְשִׂיא אֵל יֹאשִׁיָּה. פְּאֵר הַתּוּשִׁיָּה.

שְׁלָחוֹ צוּר מִחְיָה. וְגוֹדֵר כָּל פֶּרֶץ.

פְּאֵר עַם הַצֶּדֶק. כְּשׁוֹשןָׁ תּוֹךְ חֶדֶק.

מְחַזֵּק הַבֶּדֶק. בְּחִיר תּוֹלְדוֹת פֶּרֶץ.

מְלָכִים וּמְתֵיהֶם. לְפָנָיו כּוֹרְעִים הֵם.

בְּעָבְרוֹ לִפְנֵיהֶם. וְעָלָה הַפּוֹרֵץ.

כַּכָּתוּב: עָלָה הַפֹּרֵץ (מיכה ב, יג).

יְמֵי הוֹד וַאֲרִיכוּת. לְשַׂר ישֶׁר וּזְכוּת.

בְּצַוַּאר הַמַּלְכוּת. נְתָנוֹ אֵל רָבִיד.

וְהֵרִים מִשְׂרָתוֹ. אֲשֶׁר בָּחַר אֹתוֹ.

וְלָעַד מַעְלָתוֹ. כְּיָרֵחַ הֶעְמִיד.

וְגָזַר וַיַּבַּע. הֱיוֹת כִּסְאוֹ נִקְבָּע.

וּבִגְלַל זֶה נִשְׁבַּע. אֲדֹנָי אֶל דָּוִד.

כַּכָּתוּב: נִשְבַּע ה' לְדָוִד (תהלים קלב, ב).

בְּחִירֵי עַם הָאֵל. אֱמוּנֵי יִשְׂרָאֵל.

תְּנוּ תוֹדָה לָאֵל. וְשִׁירוּ לַאדֹנָי.

אֲשֶׁר הַיְקָר הֶחְיָה. בְּמִשְׂרַת יֹאשִׁיָה.

פְּאֵר הַתּוּשִׁיָּה. וְנָשִׂיא לַאדֹנָי.

וְדַלִּים נֶעְצָבוּ. בְּיָמָיו יָנוּבוּ.

וְאַחַר יָשׁוּבוּ. וּבִקְשׁוּ לַאדֹנָי.

כַּכָּתוּב: אַחַר יָשֻׁבוּ בְּנֵי יִשְׂרָאֵל וּבִקְשׁוּ (הושע ג, ה).

בְּרָכוֹת נִשְׂגָּבוֹת. וְטוֹבוֹת לִרְבָבוֹת.

כְּמוֹ נֵד נִצָּבוֹת. לְרֹאשׁ פִּנַּת יִקְרַת.

וְהַמִּשְׂרָה לִוְיָה. עֲלֵי רֹאשׁ יֹאשִׁיָּה.

כְּמַעְיַן תּוּשִׁיָּה. הֲלֹא הוּא בֵן פֹּרָת.

צְעִיר אַחִים רֹדֵם. בְּסוֹד הַיְקָר אֹדֶם.

וְהוֹרָיו מִקֶּדֶם. בְּבֵית לֶחֶם אֶפְרָת.

כָּכָּתוּב וְאַתָּה בֵּית לֶחֶם (מיכה ה, א).

הוּא נֵזֶר הַחֲסִידִים וְהַנְּשִׂיאִים. וְגֶזַע הַמְּלָכִים וְהַיְרֵאִים. נְשִׂיאֵנוּ וַאֲדוֹנֵנוּ. עַמּוּד הָאוֹר הַמֵּאִיר עָלֵינוּ. וַאֲרוֹן הַבְּרִית הַנּוֹסֵעַ לְפָנֵינוּ. לָשׂוּם הֶעָקֹב לְמִיׁשֹור לְעֵינֵינוּ. וְלִהְיוֹת קָדוֹשׁ מַחֲנֵנוּ. כְּג"ק מוֹרֵנוּ וְרַבֵּנוּ וַאֲדוֹנֵנוּ. נְשִׂיאֵנוּ נְסִיכֵנוּ מַלְכֵּנוּ.

יֹאשִׁיָּהוּ הַנָּשִיא הַגָּדוֹל נְשִׂיא גָלוּת כָּל־יִשִׂרָאֵל שק“ל בן כג”ק נְשִׂיאֵנוּ יִשַׁי הַנָּשִׂיא הַגָּדוֹל נְשִׂיא גָלוּת כָּל־יִשְׂרָאֵל זק"ל בֶּן־שְׁלֹמֹה בֶּן יֹאשִׁיָּהוּ בֶּןּ נֶחֶמְיָה בֶּן הוֹדִיָּה נֶכֶד זְרֻבָּבֶל בֶּן שְׁאַלְתִּיאֵל בֶּן הַמֶּלֶךְ יְכָנְיָה בֶּן יֹאשִׁיָּה בֶּן חִזְקִיָּהוּ בֶּן יוֹתָם בֶּן עֻזִּיָּה בֶּן אָסָא בֶּן אֲבִיָּה בֶּן שְׁלֹמֹה בֶּן דָּוִד הַגֶּבֶר הֻקַם עָל. מְשִׁיחַ אֱלֹהֵי יַעֲקֹב וּנְעִים זְמִירוֹת יִשְׂרָאֵל. וְאִישׁ אֲשֶׁר אֵלֶּה לוֹ לְאָבוֹת וְכַמִּדּוֹת הָאֵלֶּה מִדּוֹתָיו. רָאוּי לְפָאֵר סִפְרְךָ בִּתְהִלּוֹתָיו. כִּי הוּא בְחָכְמָתוֹ יַכִּיר מַעֲלַת הֶגְיוֹנֶךָ. וְנוֹרְאוֹת לְשׁוֹנֶךָ. וְיִקְנֶה בְּכָל הוֹן פְּנִינֶיךָ. כִּי לְנֶגֶד מַעֲלַת סִפְרְךָ הָעֲצוּמָה. אֵין כֶּסֶף נֶחְשָׁב בְּעֵינָיו לִמְאוּמָה.

וַיִּשָּׂא מְשָׁלוֹ:

תִּשְׁלַח לְיֹאשִׁיָּהוּ תְשׁוּרָתְךָ

מֵאַשְׁכְּלוֹת שִׂכְלְךָ וְטוּב פִּרְיָם.

לוֹ יֶאֱתָיוּ כָּל־תְּהִלּוֹתָיו

כִּי הַנְּחָלִים הוֹלְכִים אֶל יָם.

נָשִׂיא אֲשֶׁר לוֹ כָּל נְשִׂיאֵי רוּם

נָתְנוּ שְׂרוֹךְ נַעֲלוֹ עֲלֵי לֶחְיָם.

סָפוּ בְרוּחַ פִּיו מְקַנְּאָיו. עַד

אָבְדוּ וְרוֹאִים אָמְרוּ אַיָּם.

בִּשְׁמוֹ בְנֵי גוֹלָה יְרַנֵּנוּ

בִּגְאוֹן אֲדֹנָי צָהֲלוּ מִיָּם.

נִרְפָּא חֳלִי לִבָּם וְרָם שִׂיאָם

כִּי הוּא צֳרִי חֵילָם וְצִיר חָלְיָם.

כִּי הַגְּדֻלָּה בָנְתָה מִגְדָּל

לִשְׁמוֹ עֲלֵי כּוֹכְבֵי תְּלִי קַיָּם.

טָווּ בְנוֹת שִׁיר מַחְלְצוֹת הוֹדוֹ

מִמִּשְׁבְּצוֹת עֶדְיָם וְטוּב מִשְׁיָם

עַד הֶעֱבִיר הַקּוֹל בְּמַחְנֵיהֶם

כִּי הַמְּלָאכָה הָיְתָה דַיָּם.

אָמַר הַמַּגִּיד: וָאֵרָא כִּי עֲצָתוֹ הִיא הָעֵצָה הַטּוֹבָה. וְאֶל הָאֱמֶת קְרוֹבָה. וָאַעַשׂ הסַפֵּרֶ הַזֶּה עַל שֵׁם הַנָּשִׂיא הַגָּדוֹל הַנִּזְכָּר. וְיָדַעְתִּי כִּי יֵשׁ לִפְעֻלָּתִי שָׂכָר. וְשִׁירִי גְּאֻלָּה תִהְיֶה לוֹ אַחֲרֵי נִמְכַּר. וְנָפַחְתִּי רוּחַ תְּהִלּוֹתָיו בְּרֹב עִנְיְנֵיהֶם. בְּשִׁירוֹתַי אֲשֶׂר גָּוְעוּ לְחֶסְרוֹן יוֹדְעֵיהֶם. וַתָּבוֹא בָם רוּחַ וַיִּחְיוּ וַיַּעַמְדוּ עַל רַגְלֵיהֶם. וּבַעֲבוּר כִּי לְשׁוֹנִי בִשְׁבָחָיו צָרָה. אָחַזְתִּי דֶרֶךְ קְצָרָה. וְעָזַבְתִּי הַנָּהָר הֶעָמֹק. וְעָבַרְתִּי מַעְבָּרָה. וְכָתַבְתִּי רָאשֵׁי תְהִלּוֹתָיו לְהָשִׁיב קְצָת חוֹבוֹ אֲשֶׁר הוּא בִי נוֹשֶׁה. וּבַמֶּה יִתְרַצֶּה זֶה אֶל אֲדוֹנָיו הֲלֹא בְרָאשֵׁי.

אָמַר הַמְחַבֵּר: וּכְשָׁמְעִי אֲמָרָיו. שָׁקַלְתִּי בְּפֶלֶס שִׂכְלִי דְבָרָיו. וּמָצָאתִי עֲצָתוֹ נְכוֹנָה וּצְרוּפָה בְּמַצְרֵף הָאֱמוּנַה. וְחִבַּרְתִּי הַסֵּפֶר הַזֶּה כַּאֲשֶׁר מִן הַשָּׁמַיִם הָרְאֵיתִי. וְהִנַּבֵּאתִי כַּאֲשֶׁר צֻוֵּיתִי. וְרוּחַ מִבִּינָתִי הֶעִירַתְנִי לִלְקֹט מִפְּרִי עֵץ הַשֵּׂכֶל. הִיא נָתְנָה לִי מִן הָעֵץ וָאֹכֵל. וְעָשִׂיתִי מִתְּכֵלֶת הַמְּלִיצוֹת הַנּוֹרָאוֹת. חֲמִשִּׁים לֻלָאוֹת. שַׂמְתִּים בְּמִשְׁכַּן הַמַּדָּע לְעֵד וּלְאוֹת. וְקַרְסֵיהֶם בְּפַז הַמּוּסָר מְסֻלָּאוֹת. וְהֵבֵאתִי אֶת הַקְּרָסִים בַּלֻּלָאוֹת. וְהָיָה הַמִּשְׁכָּן אֶחָד.

בְּמִלִּים חֲשׁוּקִים. כְּנֹפֶת מְתוּקִים.

וְאֶל רוֹדְפֵיהֶם. קְרוֹבִים רְחוֹקִים.

וְנִלְאוּ לְהַשִּׂיג. אֲבָקָם בְּרָקִים.

תְּדַמֵּם כְּרוּבִים. חֲצוּבֵי לְהָבִים.

וְרָצִים וְשָׁבִים. כְּמַרְאֵה בְּזִקִּים.

מְעוֹנִי. בִּמְקוֹמוֹת רַבִּים תִּפְגְּשֵׁנִי. וּבְכָל מַחְבֶּרֶת אֲשֶׁר תְּחַבֵּר תִּמְצָאֵנִי. וְאֶעֱנֶה אוֹתְךָ בְּטֶרֶם תִּקְרָאֵנִי. וַיְהִי בְּשָׁמְעִי אִמְרָתוֹ. שָׂמַחְתִּי בְּחֶבְרָתוֹ. וְנִקְשַׁר לִבִּי בַּעֲבוֹתוֹת אַהֲבָתוֹ. וַיִּסֹּב מֵאִתִּי לָלֶכֶת. וְנַפְשִׁי אַחֲרָיו נִמְשֶׁכֶת. וּבִפְנוֹתוֹ לָלֶכֶת מֵאִתִּי לְעִירוֹ. שְׁמַעְתִּיו וְהוּא אוֹמֵר בְּשִׁירוֹ:

אֲשֶׁר יִשְׁמַע עֲצַת אוֹהֵב טְהָר־לֵב

עֲלֵי אוֹיְבָיו כְּחֶרֶב יִשְׁלְפֵהוּ

וַכַזָּהָב יְהִי מוֹצֵא דְבָרוֹ

בְּעֵת תּוֹךְ כּוּר לְבָבוֹ יִצְרְפֵהוּ.
