ספר תחכמוני - שַׁעַר עֶשְׂרִים וְתִשְׁעָה

הָאָב וּבְנוֹ לְרֹעַ זְמַנָּם. שׁוֹאֲלִים לִמְלֹאת סִפְקָם.


נְאֻם הֵימָן הָאֶזְרָחִי. קָנִיתִי לִי סְחוֹרָה וּמִקְוֵה. לִמְצֹא בָהּ רֶוַח וּמִקְוֶה. וְנָסַעְתִּי לְנִינְוֵה. וְהָיְתָה בָּעֵת הַהִיא מְאִירָה כְּאוֹרֵי נְגוֹהִים. עִיר גְדוֹלָה לֵאלֹהִים. וּבָאתִי אֵלֶיהָ בְּחֹדֶשׁ טֵבֵת. בְּעֵת רוּחַ הַסְּעָרָה נוֹשֶׁבֶת. וְהַזְּמַן קַר מְאֹד. וְלִפְנֵי קָרָתוֹ מִי יַעֲמֹד. וְהַגֶּשֶׁם הִצְבִּיא חֲיָלָיו. וְהַשֶּׁלֶג נָשָׂא בְּרֹאשׁ הֶהָרִים דְּגָלָיו. וַיְהִי הַיּוֹם וַאֲנִי בְּבֵית הַתְּפִלָּה. עִם הַקְּהִלָּה. וְהִנֵּה אִישׁ מָלֵא שֵׂכֶל וְדֵעִים. עָלָיו בְּגָדִים קְרוּעִים. וְהוּא בוֹכֶה לִקְשִׁי יוֹמוֹ. וְנַעַר קָטָן עִמּוֹ. וְהַקֹּר מְבַקֵּעַ מְקוֹרֵי עֵינָיו. וִיפוֹצֵץ סַלְעֵי שִׁנָּיו. וְהָרוּחַ בְּסַעֲרָתוֹ. יְסַמֵּר שַׂעֲרָתוֹ. וְהַשֶּׁלֶג יְדַבֵּק לְחִכּוֹ לְשׁוֹנוֹ. וִיבַלְבֵּל הֶגְיוֹנוֹ. וְהוּא עוֹמֵד בְּאֵבָרִים רוֹעֲדִים. בֵּין הָעוֹמְדִים. וְכַאֲשֶׁר הִשְׁלַמְנוּ הַתְּפִלָּה. עָרַךְ לָאֵל תְּהִלָּה. וַיִּקְרָא שָׁלוֹם בַּתְּחִלָּה. אַחֲרֵי כֵן אָמַר: אֲלֵיכֶם אֶקְרָא עוֹטֵי הַמִּשְׁבָּצוֹת. וְעוֹטְפֵי הַמַּחֲלָצוֹת. לוֹבְשֵׁי בִגְדֵי בְרוֹמִים. וּמִתְכַסִּים בְּעוֹרוֹת אֵילִים מְאָדָּמִים. הַלּוֹעֲגִים עַל הַקֹּר וַחֲיָלָיו. הַשּׂוֹחֲקִים לַשֶּׁלֶג בְּעֵת יָרִים דְּגָלָיו. הַנּוֹתְנִים בִּגְדֵיהֶם צִנָּה. מִפְּנֵי הַצִּנָּה. וְסוֹחֵרָה. מִפְּנֵי הַקָּרָה. וְהַקֶּרַח הַנּוֹרָא. הַשּׁוֹכְנִים בְּבֵית מִדּוֹת. הַשּׁוֹכְבִים בֵּין שִׁדָּה וְשִׁדּוֹת. חֲשַׁבְתֶּם כִּי אֵין עֲשִׁירִים בִּלְעֲדֵיכֶם. וְאֵין כָּבוֹד אֶלָּא כְבוֹדְכֶם. פַּחֲדוּ מִן הַזְּמַן אֲשֶׁר כַּמַּיִם מִשְׁתַּפֵּךְ. וְהַגַּלְגַּל הַמִּתְהַפֵּךְ. כִּי כַמָּה לָבַשְנוּ עֲדָנִים. וְשָׂבַעְנוּ מַעֲדַנִּים. וְרָכַבְנוּ סוּסִים מְיֻזָּנִים. וְסָחַבְנוּ מִשְׁבְּצוֹת שָׁנִים. וְהָיִינוּ נְשׂוּאִים בְּחֹצֶן וּבִכְתֵפַיִם. וּמַלְבּוּשֵׁנוּ צֶבַע רִקְמָתַיִם. וּמַצָּעֵנוּ רַחַם רַחְמָתַיִם. וּבְעֵת הָפַךְ הַזְּמַן טַעְמוֹ. וְהוֹצִיא עָלֵינוּ כְלֵי זַעְמוֹ. שָׁבוּ הַמַּעֲדַנִּים מְרוֹרִים. וְנֶהֱפַךְ מִשְׁתֵּה שְׁמָנִים לְמִשְׁתֵּה שְׁמָרִים. וְהָיוּ כְּרֶגַע הַמַּחֲלָצוֹת חֲלִיצוֹת. וְהַמִּשְׁבְּצוֹת מְגוֹלָלוֹת בְּבִצּוֹת. וְשָׁבוּ כְלֵי הָרְקָמוֹת. כְּלֵי נְקָמוֹת. וְהָעֲכָסִים. נִרְמָסִים. וּפָשַׁטְנוּ רִקְמָה וְתוֹלַעַת. וְלָבַשְׁנוּ רִמָּה וְתוֹלַעַת. וְעַתָּה הִזָּהֲרוּ בְּעוֹד יֵשׁ לְאֵל יֶדְכֶם. בְּטֶרֶם תַּעַזְבוּ לַאֲחֵרִים כְּבוֹדְכֶם. וְתֵלְכוּ עֲרֻמִּים מֵהוֹדְכֶם. כִּי הִנֵּה עֵינֵיכֶם רוֹאוֹת אֵיךְ קָצַר הַמַּצָּע. וְשַׂק וָאֵפֶר לְרַבִּים יֻצַּע. וְיָרַדְנוּ פְלָאוֹת. וְרָוִינוּ תְלָאוֹת. וְעֵינֵינוּ כְּעֵינֵי יְתוֹמִים לַמְרַחֲמִים. אוֹ כְעֵינֵי אֲבֵלִים לַמְנַחֲמִים. וְעַתָּה הֲיֵשׁ בָּכֶם בֶּן חוֹרִים. וְאִישׁ מִגֶּזַע טְהוֹרִים. יֵרַךְ לְבָבוֹ לְדַלּוּתֵנוּ. וְיֶהֱמוּ רַחֲמָיו לְשִׁפְלוּתֵנוּ. וְיִפְחַד פֶּן יִמְצָאֵהוּ אֲשֶׁר מְצָאָנוּ. וְיַשְׁכִּירֵהוּ בְּיֵין הַתַּרְעֵלָה הַזְּמַן אֲשֶׁר הִשְׁכִּירָנוּ. וְיַעֲבֹר עָלָיו כּוֹס הַיָגוֹן אֲשֶׁר עֲבָרָנוּ. וְכַאֲשֶׁר הִגִּיעַ עֶד כֹּה דָּבַק לְשׁוֹנוֹ אֶל חִכּוֹ. וְהִתְבּוֹנֵן אוּלַי יֵשׁ חוֹמֵל בְּלֵב הוֹמֶה וְנִשְׁבָּר. וְאֵין דּוֹבֵר אֵלָיו דָּבָר. וּבִרְאוֹתוֹ כִּי לֹא הוֹעִילוּ דְבָרָיו. וְהָעָם כְּמוֹ פֶתֶן חֵרֵשׁ לַחֲבָרָיו. יָשַׁב עַל הָאָרֶץ וְהוּא רוֹעֵד. וּמָתְנוֹ מוֹעֵד. וַיֹּאמֶר לְנַעֲרוֹ: עֲשֵׂה לְךָ אֲשֶׁר תִּמְצָא יָדֶיךָ. כִּי אֵין חוֹמֵל לְאֵידִי וְלֹא לְאֵידֶךָ. וַיָּקָם הַנַּעַר. וּלְשׁוֹנוֹ מְלֻטָּשׁ כַּתַּעַר.

וַיֹּאמַר: בִּי אֲדֹנָי מָה אֹמַר. וְחָסִיד גָּמַר. וְרֵיחַ הַצְּדָקָה נָמָר. וְשַׁדַּי נַפְשִׁי הֵמַר. וְלִבִּי חֳמַרְמָר. וְאִלּוּ בַדְּבָרִים אֲשֶׁר הַיּוֹם שְׁמַעְתֶּם יְדַבֵּר אָדָם עַל הַיָּם וְנִקְרַע. וְעֶל הַסֶּלַע וְנִבְקַע. וְעַל הָהָר וְנִרְקַע. וְעַל הָאֵשׁ וְתִשְׁקַע. וְעַל הַנְּחָשִׁים. אֲשֶׁר אֵין לָהֶם לְחָשִׁים. וְשָׁמְעוּ לְקוֹל מְלַחֲשִׁים. וְנִקְרַע לְבָבָם לִשְׁנַיִם. וְנָמַס וִיהִי לְמַיִם. וּמַדּוּעַ לְבַבְכֶם כְּצוּר הַחַלָּמִישׁ. מִמְּקוֹמוֹ לֹא יָמִישׁ. הָאֵל יַצִּילֵנוּ מִלְּבָבוֹת קָשׁוֹת. וּמִנְּפָשׁוֹת חֳלוּשׁוֹת. וּמִיָּדַיִם סְגוּרוֹת. וְאָזְנַיִם חֵרְשׁוֹת. מֵאֵלֶּה הַלְּבָבוֹת. אֲשֶׁר מִסֶּלַע חֲצוּבוֹת. וְהָרַעְיוֹנִים. הַקָּשִׁים מֵאֲבָנִים. וְהַיָּד הַחֲזָקָה. אֲשֶׁר הִיא מֵחֶסֶד רֵיקָה. וְשָׁב מְקוֹרָהּ יַבֶּשֶׁת. וּמֵצַח בְּעָלֶיהָ נְחשֶׁת. הִנֵּה שָׁמְעוּ הַיּוֹם אָזְנֵיכֶם. אֲשֶׁר לֹא שְׁמַעְתֶּם כָּמוֹהוּ בִּזְמַנֵּיכֶם. וְאַתֶּם מַעֲלִימִים מֵרְאוֹת עֵינֵיכֶם וְהוֹפְכִים לְצַד אַחֵר פְּנֵיכֶם. כְּאִלּוּ אֵין הַמִּצְוָה הַזֹאת אֲלֵיכֶם. אוֹ הַתּוֹכָחָה הַזֹּאת לְזוּלַתְכֶם. וְעַתָּה יִזְכֹּר כָּל אִישׁ מִכֶּם יוֹצְרוֹ. וְיִשְׁבֹּר קְשִׁי יִצְרוֹ. וְיֵרַךְ לְבָבוֹ עָלָיו וּכְבֵדוֹ. וְאַל יִקְפֹּץ מִמֶּנִּי יָדוֹ. וְיַעֲשֶׂה הַיּוֹם מַה שֶׁיִמְצָא מָחָר יַלְדּוֹ. וְיִפְחַד פֶּן יִקְרֶה מָחָר לִבְנוֹ אֲשֶׁר קְרָאֵני. וְיָשׁוּב לִשְׁאֹל פַּת לֶחֶם כָּמוֹנִי. וְיַצְרִיכֵהוּ הַזְּמַן לַאֲנָשִׁים קָשִׁים כַּאֲשֶׁר הִצְרִיכָנִי.

אָמַר הַמַּגִּיד: וְכִשְׁמֹעַ הָאֲנָשִׁים דִּבְרֵי הַבֵּן. נִבְקַע לִבָּם אֲשֶׁר הָיָה כְמוֹ אֶבֶן. וְתָמְהוּ לְתֹקֶף לְשׁוֹנוֹ. וְהִתְנַדְּבוּ כֻלָּם עַל הָאָב וְעַל יְלִידוֹ. אִישׁ כְּמַתְּנַת יָדו. וְכַאֲשֶׁר הִתְמַלֵּא שַׂקּוֹ. וְנִמְלָא סִפְקוֹ.


נָשָׂא הָאָב מְשָׁלוֹ וַיֹּאמַר:


יְרַחֵם אֵל אֲצִילִים רִחֲמוּנִי

בְּנִדְבוֹתָם וְרַךְ לִבָּם לְאֵידִי.

וְהָיָה מִתְּחִלָּה לִי לְבָבָם

כְּמוֹ שָׁמִיר וְאֵין מַחְזִיק בְּיָדִי.

וְצוּר הָפַךְ לְבָבָם לִי וְחָמְלוּ

לְזִקְנָתִי וְעַל קַטְנוּת יְלִידִי.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר כִּלָּה דְבָרָיו.

וְהִשְׁלִים שִׁירָיו. הִתְבּוֹנַנְתִּי אֵלָיו. וְהִנֵּה הוּא חֶבֶר הַקֵּינִי בְּעֵינוֹ. וְהַנַּעַר בְּנוֹ. וּמַטַּע גַּנּוֹ. וְיָדַעְתִּי עָרְמָתוֹ וְנִכְלוֹ. וְהֵבַנְתִּי כִּי הַנַּעַר רִשְתּוֹ וְחַבְלוֹ. אַךְ לָא הִתְוַדַּעְתִּי אֵלָיו. פֶּן תִּגָּלֶה בַקָּהָל עָרְמַת פְּעָלָיו. וַעֲזַבְתִּיהוּ וְנָטִיתִי מַעָלָיו.
