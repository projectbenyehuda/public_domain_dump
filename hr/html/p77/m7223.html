בלט ממללת

כתובת: תשובת דון וידאל בנבנשת ז"ל

בַּלָּט מְמַלֶּלֶת וְכִמְצַחֶקֶת / בַּת שִׁירְךָ חִצֵּי זְמִיר זוֹרֶקֶת

בַּלָּט מְמַלֶּלֶת וְתַגְבִּיר מַעֲנֶה / וַתַּעֲנֶה רַכּוֹת וְשֵׁן חוֹרֶקֶת

תָּעִיר דְּבַר רִיבוֹת וְשָׁלוֹם קָרְאָה / מִתְקָרְבָה לַקְּרַב וּמִתְרַחֶקֶת

תִּשְׂטֹם אֲבָל תַּחְתֹם וּפִיהָ תֶּאֱטֹם / תַּעְצֹר חֲמַת לָשׁוֹן וְהִיא מוֹצֶקֶת

תַּגֵּשׁ בְּרַק חַרְבָּהּ אֲבָל לֹא שָׁלְפָה / אַט תַּהֲלֹךְ וּלְאַחֲרַי דוֹלֶקֶת

לֹא תֶחֱרַד רוּחִי לְקוֹל נֶשֶׁק וְלֹא / מִקּוֹל הֲמֻלַּת חוֹלְלִים נִתֶּקֶת

תִִּּלְעַג לְזַעַם שׁוֹלְפֵי חֶרֶב וְאֶל / רַעַם וְרַעַשׁ נוֹגְנִים שׂוֹחֶקֶת

לֹא פָחֲדָה מִמַּלְתְּעוֹת לָבִיא וְלֹא / תָּגוּר וּפִיהָ גוּר אֲרִי נוֹשֶׁקֶת

תִּשְׁטֹף צְבַא נוֹגְנִים בְּרִגְשַׁת נָאֳמָהּ / לוּ חֵיל צְבָאֶיהָ תְּהִי שׁוֹרֶקֶת

כִּי יַד יְמִינָהּ רוֹמְמָה גַם עָשְׂתָה / חַיִל וְלֹא מִיָּדְךָ נֶחְזֶקֶת

לֹא לָקְחָה מִיָּדְךָ הַתֹּף וְלֹא / מִסִּפְרְךָ מִשְׁלֵי זְמִיר מַעְתֶּקֶת

שִׁירָה כְאַיָּלָה שְׁלוּחָה עַל פְּנֵי / שָׂדַי וְלִמְצוֹא מַחֲנֶה שׁוֹקֶקֶת

עַל פִּי קְדוּמֵי הַזְּמִיר הִתְחַבְּרָה / אֵלַי וּבִי תִדְבַּק וְלִי נִזְקֶקֶת

וּכְיַעֲלַת חֵן עַל זְרוֹעוֹת אוֹהֲבָה / הִתְנַפְּלָה אֵלַי וּמִתְרַפֶּקֶת

בָּאָה בְחַצְרוֹתַי וּבִידֵי אַהֲבָה / עַל קִיר לְבָבִי נָטְעָה שׂוֹרֶקֶת

בָּאָה וְלֹא אַתָּה נְפַחְתָהּ בִּי וְלֹא / מֵאִתְּךָ אֶל אָהֳלִי נוֹסֶקֶת

אֵיך תַּחֲשֹׁב כִּי מִבְּאֵרָךְ שׁוֹאֲבָה / רוּחִי וּמֵימֵי שִׁירְךָ לוֹקֶקֶת

יוֹנְקָה שְׁדֵי לֵוִי גְבִירוֹל שָׂר בְשִׁיר / שׁוֹד בֶּן מְשֻלָּם שִׁיר תְּהִי יוֹרֶקֶת

דַּי לָהּ בְּפַת חָלָה וְזָר לֹא שָׁאֲלָה / דַּי לָהּ לְפִי אָכְלָהּ וּמִסְתַּפֶּקֶת

הַהוּא נְזִיר אַחִים אֲשֶׂר רוּחוֹ עֲלֵי / קֹטֶב גְּלִילֵי שִׁירְךָ חוֹמֶקֶת

עַל מֶלְתְּחוֹת שִׁירָךְ זְרוֹעוֹ מוֹשְׁלָה / וּצְפוּן טְמוּנֵי לִבְּךָ בּוֹדֶקֶת

שָׁלַח יְמִינוֹ עַל נְאוֹת שִׁירָךְ וְזָר / קָרֵב לְקִיר הֵיכָלְךָ דּוֹחֶקֶת

פִּיהוּ שְׁאַל אִם הֶאֱצִיל מֵרוּחֲךָ / עָלַי וּמִלָּתוֹ תְהֵא מַצְדֶּקֶת

אוּלַי לְחַרְחֵר רִיב קְנָאוֹת בֵּין שְׁנֵי / אַחִים וְרֵעִים נַפְשְׁךָ חוֹשֶקֶת

בַּל תַּחֲשֹׁב לָפִיר בְּרִית דּוֹדִים אֲשֶׁר / נַפְשָׁם בְּחוּט הָאַהֲבָה מֻדְבֶּקֶת

כִּי לֹא לְדוֹר דּוֹרִים רְתוּקוֹת אַהֲבָה / וַעְבוֹת יְדִידוּת תִּהְיֶה נִרְתֶּקֶת

נַפְשִׁי עֲצוּבָה עַל חֲלִיפוֹת הַזְּמָן / וּלְיוֹם תְּמוֹל כִּי יַעֲבֹר נֶאְנֶקֶת

תֵּבְךְּ עַל זְמַן נֹעַר אֲשֶׁר חָלַף וְיַד / שֵׂיבָה בְאֵיבָה רֹאשְׁךָ מוֹחֶקֶת

כִּי יָרְאָה פֶן יֹאמְרו הַמּוֹשְׁלִים / יָדָהּ עֲלֵי יָשִׁישׁ וְשָׂב עַתֶּקֶת


אַחְלַי וְתָשׁוּב נַפְשְׁךָ לִימֵי שְׁנֵי / נֹעַר וְעֹל זֹקֶן תְּהִי פוֹרֶקֶת

תָּשׁוּב לְיָד בַּחְרוּת בְּעוֹד לֹא נִתְּקוּ / הַחוּט וְהַשִּׁדְרָה וְהַמַּפְרֶקֶת

מִיָּדְךָ אֶדְרֹשׁ חֲמַס שִׁירָה אֲשֶׁר / נִקְרֵאת מְשַׁקֶּרֶת וְשָׁוְא חוֹקֶקֶת

זֶה פַעֲמַיִם נִתְּנָה עַל יָדְךָ / לבִזָּה וְלִמְשִׁיסָה וְהִיא שׁוֹתֶקֶת

טוּב סַחֲרָהּ תִּקְרָא רְכֻלַּת שָׁוְא וְאֶת / חֶמְדַּת מְלַאכְתָהּ תַעֲשֶׂה בוֹקֶקֶת

אַתָּה זְקַן בֵּיתָהּ הֲלַשְׁמִיד הוֹד שְׁמָהּ / טַעַם וְעֵצָה רוּחֲךָ מַעְמֶקֶת

אַתָּה מְנַת חֶבְלָהּ וְאֵיךְ בָּהּ גָּעֲלָה / נַפְשָׁךְ וּבִפְנֵי נָעֳמָהּ יוֹרֶקֶת

רוּחַ יְדִידוּתָהּ אֲשֶׁר נָחָה בְךָ / אֵיךְ מִמְּךָ סָרָה וּמִסְתַּלֶּקֶת

עַל חֵיק מְזִמָּתָהּ גְּמָלַתְךָ כְּבֵן / וַחְלֵב רְנָנֶיהָ לְךָ מֵינֶקֶת

וּכְאֵם גְּדָלַתְךָ וְאֵיכָה נַפְשְׁךָ / פִּיךָ לְבֹשֶׁת אִמְּךָ פּוֹשֶׂקֶת

הַאִם לְחַלֵּל אֶת צְבִי תִּפְאַרְתְּךָ / מִשְׁתַּדְּלָה נַפְשָׁךְ וּמִתְעַסֶּקֶת

שִׁירָה לְקוֹלָךְ רָגְזָה וַתַּהֲלֹֹךְ / הָלוֹךְ וְזוֹעֶקֶת וְכַף סוֹפֶקֶת

כִּי הֹרְסָה מֵאָהֳלָהּ שֵׂיָה וּמִן / שׁוֹכֵב בְּחֵיקָהּ הָיְתָה נֶעְשֶׁקֶת

הָיְתָה כְמַטָּרָה לְחֵץ שִׁיר עוֹרְרָה / מִפְגָּע לְפִי חַרְבוֹת זְמִיר מֵיפֶקֶת

לִירוֹת לְנַפְשָׁהּ חֵץ הֲלֹא מֵאָהֳלָהּ / קֶשֶׁת רְמִיָּה יָדְךָ נוֹשֶׁקֶת

הִיא עָרְכָה הָאֵשׁ וְהָעֵצִים וְהִיא / זִיקוֹת מְאַזֶּרֶת וּבָם נִשֶּׂקֶת

בַּהְגוּת שְׂפָתֶיהָ לְקֶלֶס נִתְּנָה / וּבְשׁוֹט לְשׁוֹנָהּ הָיְתָה נִזֶּקֶת

מַן נָאֳמָהּ נָמֵס בְּחֹם שִׁמְשָׁהּ וּמִן / אֵשׁ דַּת זְמִירוֹת קוֹדְחָה נִמֶּקֶת

לִמְצוֹא פְדוּת נַפְשָׁהּ כְּאַיָּל תַּעֲרֹג / מִי לִי בְיַלְדֵּי הַזְּמָן צוֹעֶקֶת

לִפְקוֹד פְּקוּדָתָהּ בְּאָזְנַי לָחֲשָׁה / וַתַּעֲמִיק עֵצָה וְסוֹד מַמְתֶּקֶת

וַתַּעֲמִיק עֵצָה וּמִדֵּי דַּבְּרָהּ / יָד נָתְנָה אֵלַי וּמִתְאַפֶּקֶת

יָד נָתְנָה אֵלַי וְלַעְמֹד עַל כְּבוֹד / יִקְרַת שְׁמָהּ אוֹתִי כְבֵן חוֹבֶקֶת

תֶּחְזַק יְמִינָהּ בִּי וְלִבּוֹת דּוֹבְרֵי / עָתָק בְּחַרְבוֹת נָאֳמִי עוֹתֶקֶת

אִם הַזְּמִיר הוּא מָר דְּרוֹר לִבִּי צְרוֹר / אוֹ הוּא כְיֵין רֶקַח זְמָמִי שֹׁקֶת

בֵּין מַחֲנֶה שָׁרִים וְצִבְאוֹת נוֹגְנִים / יִגַּל בְּרַק שִׁירִי כְעֵין בָּרֶקֶת

אֶל דּוֹבְרֵי זָדוֹן בְּיוֹם מָדוֹן בְּפִי / לָשׁוֹן כְּעֵין כִּדּוֹן וְלֵב חוֹלֶקֶת

וּלְשׁוֹחֲרֵי רָצוֹן וְדוֹרְשֵׁי אַהֲבָה / בַּלָּט מְמַלֶּלֶת וְכִמְצַחֶקֶת


סימן

כתובת: ועוד לו

רוּחִי בְקִרְבִּי קוֹל זְמִיר נָתָנָה / תִּשְׂחַק כְּבֵן יָאִיר לְאֵיד בִּגְתָּנָה

וּכְבַת אֲבִיחַיִל גְּאוֹנִי יַעֲלֹז / כִּי בֶאֱלֹהֶיהָ לְבָד נִשְׁעָנָה

יָשְׁבָה בְצֵל שִׁירִי וּבוֹ הִתְלוֹנְנָה / וּבְחֵיל תְּלָאוֹת נָשְׂאָה אוֹתָנָה

יֶתֶר שְׂאֵת נִשֵּׂאת הֲדַסָּה עַל בְּנוֹת / הַחֵן וְהִיא עָלְתָה עַל כֻּלָּנָה

הִיא בַת אֲבִיחַיִל וְחַיִל אָזְרָה / כִּי עִם אֲגָגִי לַקְּרָב הוּכָנָה

נַפְשִׁי לְזֵכֶר פָּעֳלָהּ יָצְאָה בְיוֹם / הַפּוּר וְחִצֵּי נָאֳמָהּ שִׁנָּנָה

יָצְאָה בְחַרְבוֹת מַעֲנֶה לִנְקוֹם נְקַם / בַּת שִׁיר לְחֶרְדַּת נוֹגְנִים נִתָּנָה

קֶשֶׁת בְּיָד עֵט דֹּרְכָה עַד יֹאמְרוּ / עַד אָן תְּשַׁלַּח חֵץ הזמיר עַד אָנָה
