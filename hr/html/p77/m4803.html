במתק הזמיר

כתובת: לידידי דון וידאל בן לביא

בְּמֶתֶק הַזְּמִיר אֶכְתֹּב מְרוֹרוֹת / וּמִבֵּין הָחֳרִי אָעִיר זְמִירוֹת

יְקַר תֵּבֵל אֲחַלֵּל בַּחֲלִילִי / וְעַל סִדְרֵי זְמָן אֶשְׁפֹּךְ עֲבָרוֹת

וְהוֹד עִתִּי ביפִי עֵטִי אֲנַבֵּל / וְעֹז פָּנָיו אֲשַׁנֶּה בִגְעָרוֹת

בְּמַעְנֶה רַךְ אֲנִי קָשׁוֹת אֲדַבֵּר / וּבִשְׁפַל נָאֳמִי אֶחְשֹׂף יְעָרוֹת

לְאַט אָרִיב אֲבָל חָזְקוּ דְבָרַי / וּמִשָּׁמִיר אֲמָרוֹתַי וְשׁוּרוּת / 5

וְקוֹלִי אֶל בְּנֵי אִישׁ קוֹל חֲלוּשָׁה / וְיִשָּׁמַע כְּתוֹקֵעַ חֲצוֹצְרוֹת

זְמָן עַוָּל יְדַכֵּא כֹל וְיִסְבֹּל / וְיִתְגַּלְגֵּל בְּיַד מִקְרִים וְקוֹרוֹת

יְסוֹדוֹתָיו בְּתַהְפּוּכוֹת יְצוּקִים / וְהָטְבָּעוּ אֲדָנָיו עַל תְּמוּרוֹת

וְאֵיךְ יִשְׁבֹּת אֲשֶׁר כֻּלּוֹ חֲלִיפוֹת / וּמִדֵּי יוֹם בְּיוֹם יִשְׁגֶּה בְזָרוֹת

וְאִם פַּעַם לְךָ קוֹלוֹ יְחַנֵּן / נְטֵה לָךְ פֶּן יְקַרְקֵר לֵב וְקִירוֹת / 10

וְטוּבוֹ כַחֲלָקוֹת בַּאֲפֵלָה / וְצוּפוּ רוֹשׁ וּבִרְכּוֹתָיו מְאֵרוֹת

קְשֵׁה רוּחַ וּפִיו חָלָק וְקוֹלוֹ / כְקוֹל יַעְקֹב אֲבָל יָדָיו שְׂעִירוֹת

וְלֹא יִשָּׂא פְנֵי נָגִיד וְנָשִׂיא / וְלֹא יֶהְדַּר אֲשֶׁר פָּנָיו הֲדוּרוֹת

וְכָל רָאשָׁיו וְדוֹר דּוֹרְשָׁיו מְתֵי שָׁוְא / וְחַלָּשָׁיו יְשַׁו שָׂרֵי עֲשָׂרוֹת

לְעוֹבֵר דָּת וְדִין יִתֵּן בְּרָכוֹת / וּמִגְדָּנוֹת לְחוֹלְפֵי חֹק וְתוֹרוֹת / 15

וְהוּא מִשְׁעָן לְכָל גֶּבֶר חֲסַר לֵב / וְכַפָּיו מִתְּמֹךְ אִישׁ תָּם נְעוּרוֹת

וְהוּא תוֹמֵךְ בְּיַד שׁוֹגֵג וּמַשְׁגֶּה / וְיַגֶּה הַנְּפָשׁוֹת הַטְּהוֹרוֹת

וְיִבְנֶה לִפְתָאִים בֵּית מְלוּכָה / וְיַחְצֹב אֶל נְבוּבֵי לֵב בְּאֵרוֹת

וְכָל עָרוּם יְשַׁו שׁוֹלָל וְעָרוֹם / וְטַרְפֵּי הַתְּאֵנָה לוֹ חֲגוֹרוֹת

יְשַׁו אָדָם כְּמוֹ חַרְשֵׂי אֲדָמָה / וְאַבְנֵי גִיר מְנֻפָּצוֹת יְקָרוֹת / 20

וְאֵיךְ יִשְׂמַח אֱנוֹשׁ בִּמְעַט עֲדָנָיו / וְאַחְרִיתוֹ וְסוֹפוֹ לִקְבָרוֹת

הֲיָשִׂישׂ אִישׁ וּמָוֶת בֵּין פְּשָׁעָיו / וְאִם יִחְיֶה שְׁמוֹנִים עַל גְבוּרוֹת

כְּרָץ קַלּוּ יְמֵי חַיָּיו בְּעֵינָיו / וְטוֹבוֹתָיו כְּצֵל חוֹלְפוֹת וְעוֹבְרוֹת

וְלוּלֵי יַד יְהוֹסֵף הָיְתָה בוֹ / שְׂעִפָּיו יָרְדוּ אֶל מַהֲמוּרוֹת

וּבוֹ נִחַם גְּאוֹנִי מִיְגוֹנִי / וּמִמַּדְוֵי זְמָן מָצָא מְזוֹרוֹת / 25

יְהוֹסֵף מַחֲנוֹת בִּינוֹת מְאַסֵּף / וּבָהֶם יַעֲשֶׂה שֶׁבַע חֲקִירוֹת

בְּנֵר שִׂכְלוֹ צְפוּנֵי בִין יְחַפֵּשׂ / וְנִגְלוּ לוֹ כְקַלּוֹת כַּחֲמוּרוֹת

וְהִשְׁלִים בּוֹ זְמַנּוֹ מַחֲסוֹרוֹ / בְתֹם מַעְשִׂים וּבִפְעוּלוֹת בְּרוּכוֹת

וְתִקֵּן אֶת מְעָנוֹתָיו בְּכִשְׁרוֹן / תְּכוּנוֹתָיו וּמִדּוֹתָיו יְשָׁרוֹת

וְעַל כֵּן נִקְשְׁרָה נַפְשִׁי בְנַפְשׁוֹ / וְטוּחוֹתַי בְיֵין חִשְׁקוֹ שְׁכוּרוֹת / 30

יְבַלֶּה אֶת שְׁנוֹתָיו בַּנְּעִימִים / וְזִכְרוֹ בַזְּמָן יַעְמֹד לְדוֹרוֹת

וְיִשָּׂא מִפְּרִי יָדָיו בְּרָכוֹת / כְּחוֹל הַיָּם וְשָׁלוֹם כַּנְּהָרוֹת
