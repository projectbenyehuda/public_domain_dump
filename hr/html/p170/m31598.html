אֱמֶת עַל הַזְּמָן

לר׳ חננאל, הוא אלחנן, בן ר׳ ישועה

אֱמֶת עַל הַזְּמָן אֵין לִי תְּנוּאוֹת / וְלֹא אֶפְקֹד עֲלֵי יָמִים חֲטָאוֹת,

וְאֶשְׁכַּח כָּל אֲשֶׁר עָשָׂה לְנַפְשִׁי / וְאֶחְשֹׁב כָּל זְדוֹנוֹתָיו שְׁגִיאוֹת.

וְאֵיכָה לֹא אֲזַמֵּר לוֹ וְהִנֵּה / בְּנוֹתָיו כָּל חֲפָצַי אַט מְבִיאוֹת.

הֲרִימוֹתִי יְמִינִי עַל חֲיָלֵי / מְצוּקִים גָּבְרוּ עָלַי וְשׁוֹאוֹת.

וְחַיִּים מָתְקוּ לִי אַחֲרֵי כִי / שְׁנוֹת חַיַּי אֱלֵי נַפְשִׁי שְׂנוּאוֹת.

וּמַה־לִּי עוֹד וְדוֹדֵי רַב חֲנַנְאֵל / תְּשׁוּרָה הוּבְאוּ עָדָי וּמַשְׂאוֹת.

גְּבִיר נוֹצָר עֲלֵי אַוַּת לְבָבוֹ / וְנִבְרָא כַאֲשֶׁר יִכּוֹן וְיֵאוֹת.

וּמֵאוּר חַם לֵבָב שֶׁמֶשׁ בְּשׁוּרוֹ / וְעָלָיו עָבְרָה רוּחַ קְנָאוֹת.

וְלוֹ נַהַם יְעִידוּהוּ אֲרָיוֹת / וְלוֹ חֵן חָמְדוּ אוֹתוֹ צְבָאוֹת.

וְנִקְרָא מִבְּנֵי אָרֶץ, וְדַרְכֵי / בְנֵי שַׁחַק בְּמַהְלָכָיו מְצוּאוֹת.

כְּפִיר־אֹמֶר, וּמַשְׂכִּילִים לְפָנָיו / לְשׁוֹנוֹתָם בְּפִיּוֹתָם כְּלוּאוֹת.

הֲמוֹנֵיהֶם בְּצוּר לָשׁוֹן יְרַדֵּף / וְיָנִיס יוֹם קְרַב־הַבִּין צְבָאוֹת,

וְאִלּוּ יִהְיוּ שְׂרֵי עֲשָׂרוֹת, / לְרִבּוֹאוֹת יְהִי שַׂר, לֹא לְמֵאוֹת.

בְּשָׁאְגוֹ כַּכְּפִיר אֹמַר: אֲבוֹתָיו / לְבָאִים הֵם וְאִמּוֹתָיו לְבָאוֹת.

וְכֵן נָעִיד לְצִדְקוֹ, כִּי נְבִיאִים / אֲמָנוּהוּ וְאֹמְנוֹתָיו נְבִיאוֹת.

עֲצָתוֹ רוֹפְפָה גֹבַהּ שְׁחָקִים / בְּנוֹתֶיהָ עֲצַת יָמִים מְנִיאוֹת,

וּמַעְלָתוֹ לְכָל מַעְלָה גְבָרֶת / וְהַחָכְמוֹת בְּשֵׁם שִׂכְלוֹ קְרוּאוֹת.

וְנַפְשׁוֹ דָרְכָה עַל מִפְלְשֵׁי עָב / וְאָכֵן גָּעֲלָה גָאוֹן וְגֵאוֹת.

יְמִינוֹ כָל פְּזוּרֵי בִין תְּקַבֵּץ / וְתִתְחַר כָּל קְבוּצֵי הוֹן לְהַפְאוֹת.

כְּעָגוּר צִפְצְפוּ רַבִּים חֲסָדָיו / וּמַהְלָלָיו לְבַד לָקְחוּ מְנָאוֹת.

וְכָל עֵת סִפְּרוּ טֹהַר פְּעָלָיו / בְּגִיל פִּיּוֹת בְּנֵי שֵׂכֶל מְלֵאוֹת,

וְכִשְׁמֹעַ תְּהִלּוֹתָיו נְפָשׁוֹת / לְמוּלוֹ פֹרְחוֹת יַחַד וְדָאוֹת,

וּבִשְׁמוֹ תַעֲלֹז תֵּבֵל, וְטֶרֶם / הֱיוֹתוֹ דָמְתָה אֶל עִיר מְשׁוֹאוֹת.

וּמִי זָרַע בְּכָל לֵבָב חֲסָדִים, / אֱמֶת, כִּי מַהֲלָלִים לוֹ תְבוּאוֹת,

וְאֵיךְ קָרְאוּ שְׁמוֹ פֶּלֶא לְבַדּוֹ / וְהִנֵּה כָל פְּעֻלּוֹתָיו פְּלִיאוֹת,

וְלוֹ חֶסֶד לְדוֹר אַחְרוֹן יְסֻפָּר / יְהִי מָשָׁל בְּפִי בָאִים וּבָאוֹת.

יְצָרוֹ אֵל לְבַב רֵעִים לְסָעֵד / בְּיוֹם רִפְיוֹן, וְלֵב חוֹמֵד לְהַכְאוֹת.

אֱמֹר לָאִישׁ אֲשֶׁר יֵצַר מְעוֹנוֹ / וְאַף כָּל הָאֲרָצוֹת לוֹ מְקִיאוֹת:

לְרֹחַב בֵּית חֲנַנְאֵל מַהֲרָה־נָא / וְתַרְגִּיעַ וְתִשְׁכַּח הַתְּלָאוֹת,

וְלָאֹהֶל אֲשֶׁר הֵלֶךְ מְאַהֵל / וְאֶל רוֹעֶה מְחַבֵּשׁ נַהֲלָאוֹת.

בְּנֵי אֹרַח, לְיֵין מִשְׁתָּיו עֲלוּ, כִּי / מְאֹד נִרְקָח וּמַאֲכָלָיו בְּרִיאוֹת.

לְכוּ מֵאֵין מְחִיר שִׁבְרוּ וְרַוּוּ / בְּלֹא כֶסֶף נְפָשׁוֹת הַצְּמֵאוֹת,

וּפֶצַע הַנְּדוֹד יֶחְבַּשׁ, לְמַעַן / בְּמִדְבְּרָיו לְכָל נֶגַע רְפוּאוֹת.

עֲרֵבִים הֵם וּמִשֶּׁמֶן לְכָל חֵךְ / מְאֹד רַכּוּ וְחָלְקוּ מֵחֲמָאוֹת.

אֲהוֹדֶה לַגְּבִיר אָבוּן אֲשֶׁר שָׂם / בְּרִית בֵּינִי ובֵין חֶמְדַּת בְּרִיאוֹת.

יְסוֹד חָכְמָה, יְלָדוּהוּ נְשִׂיאִים, / נְשָׂאוּהוּ בְתוֹךְ חֵיקָן נְשִׂיאוֹת.

וּמִי לוֹ יַעֲרֹךְ מַשְׂכִּיל כְּעוֹרֵךְ / מְעוֹן קֹדֶשׁ אֱלֵי בָמוֹת טְלֻאוֹת,

וְהָאֹדֶם יְדַמֶּה לָאֲדָמָה / וְסוּס עָגוּר לְכוֹס מִדְבָּר וְקָאוֹת.

וְכָל חֶסֶד בְּיַד שָׂרִים כְּפֶגֶר / וְאָבוּן לוֹ לְאָזְנַיִם וְרוֹאוֹת.

לְשׁוֹנִי יוֹם תְּכַל לִסְפֹּר חֲסָדָיו / אֲזַי לָהּ מַחְשְׁבוֹתַי…

יְשִׁיבוּן: אִם תְּאַו לַשִּׂיג קְצָתָם / בְּמִדְבָּרִים אֲלֵיהֶם אִישׁ לְהַלְאוֹת.

כְּתָבִי! מִבְּלִי לָשׁוֹן בְּאָזְנֵי / חֲנַנְאֵל תִּקְרָא אֶת הַקְּרִיאוֹת,

אֱמָר־לוֹ, כִּי בְעֵין שִׂכְלִי רְאִיתִיו / – וְשֵׂכֶל לַחֲזוֹת בָּרִים כְּמַרְאוֹת –

וְזֵכֶר מַעֲשָׂיו לִבִּי יְשׁוֹבֵב / בְּעֵת יֵצֵא דְרוֹשׁ אָמוֹן בְּפֵאוֹת,

וְהָעֵת אָחֲזָה יָדִי כְתָבוֹ – / עֲלֵי כָל יַד מְאֹד עָלְתָה פְלָאוֹת.

כְּתָב הֵן בַּעֲדוֹ עֵינַי מְצַפִּים / לְיַד דֶּרֶךְ וְלִרְאוֹתוֹ תְלֻאוֹת,

שְׂעִפַּי נִבְהֲלוּ בוֹ, כִּי אֲמָרָיו / וְעִנְיָנָיו כְּמוֹ דִבְרֵי נְבוּאוֹת.

הֲרִימוֹתִיו לְהִתְנוֹסֵס כְּמוֹ נֵס / וְאֶת־יָפְיוֹ לְכָל עַמִּים לְהַרְאוֹת.

נְתַתִּיהוּ בְהָקִיץ שַׁעֲשׁוּעָי / וְעַל עַרְשִׂי בְנוּם לַיִל וּמַרְאוֹת.

עֲלֵה, הַשַּׂר, מְעוֹן צֶדֶק לְיַשֵּׁב / וּבֵית עָוֶל בְּרוּחַ פֶּה לְהַשְׁאוֹת.

וְהִתְגַּדֵּל לְהַכְנִיעַ גְּבוֹהִים / וְלַגְּבִּיהַּ נְפָשׁוֹת הַנְּכֵאוֹת

וְלָשִׂים אֶת־פְּעָלֶיךָ כְּאוֹתוֹת / לְכֹל יָבֹא, כְּמוֹ אַתְּ בַּזְּמָן אוֹת.
