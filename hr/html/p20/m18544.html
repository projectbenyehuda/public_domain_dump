וּבְכֵן יְיָ קְרָאתִיךָ

וּבְכֵן יְיָ קְרָאתִיךָ: חוּשָׁה לִּי, / הַאֲזִינָה קוֹלִי בְּקָרְאִי לָךְ!

תִּכּוֹן תְּפִלָּתִי קְטֹרֶת לְפָנֶיךָ, / מַשְׂאַת כַּפַּי מִנְחַת עֶרֶב.


שִׂימָה, שָׁמְרָה לְפִי וַאֲמִתְּךָ תִּדְבֹּר,

אַל תַּט לִבִּי לְרַע עִם פְּתָאִים לַחְבֹּר.

יֶהֶלְמֵנִי דְּבַר עָוֹן אֲשֶׁר אֶצְבֹּר,

נִשְׁמְטוּ יְמֵי עֶדְנָה וּבָא הָעֵת לִקְבּוֹר,

כְּמוֹ פוֹלֵחַ נִפְזְרוּ עֲצָמַי לְפִי הַבּוֹר,

כִּי אֵלֶיךָ אֵל שַׁבְתִּי בְּתֹם וָבֹר.

שָׁמְרֵנִי מִפַּח הִצִּיבוֹ גּוֹי גִּבּוֹר –

יִפְּלוּ בְּמַכְמֹרָיו רְשָׁעִים יַחַד, אָנֹכִי עַד אֶעֱבוֹר.


קוֹלִי כְמַבְכִּירָה מִפְּנֵי אֵימֶיךָ,

אֶשְׁפֹּךְ אֶת נַפְשִׁי וְאֶדְרֹשׁ רַחֲמֶיךָ.

זָעַקְתִּי מִפְּנֵי תִגְרַת קָמֶיךָ –

הַקְשִׁיבָה שַׁוְעִי מִמְּכוֹן שָׁמֶיךָ.

בְּהִתְעַטֵּף רוּחִי אֶזְכְּרָה נָעֳמֶךָ,

הַבֵּט לִכְבוֹדְךָ יוֹם בּוֹ אֲרוֹמִמְךָ –

הוֹצִיאָה מִמַּסְגֵּר נַפְשִׁי לְהוֹדוֹת אֶת שְׁמֶךָ.


יְיָ, שְׁמַע שִׂיחִי בְּיוֹם עָשׂוֹר לְחָדְשִׁי

וְאַל תָּבוֹא בְמִשְׁפָּט אִתִּי עַל כַּחְשִׁי –

כִּי רָדַף אַחֲרַי פֶּרֶא שָׁב לְחָפְשִׁי.

בְּהִתְעַטֵּף עָלַי נַפְשִׁי בְּהִנָּפְשִׁי,

זָכַרְתִּי יָמִים מִקֶּדֶם אֲשֶׁר אַשִּׂיא.

פֵּרַשְׂתִּי יָדַי מִתּוֹךְ פַּח מוֹקְשִׁי –

מַהֵר עֲנֵנִי, כִּי בְרִיתְךָ לֹא אֶשִׁי.

הַשְׁמִיעֵנִי גִיל אֲשֶׁר יָרִים רֹאשִׁי,

לַמְּדֵנִי רְצוֹנְךָ וְהָסֵר כָּל מַשִּׁיא.

לְמַעַן שֵׁם כְּבוֹדְךָ תִּבְנֶה הַר קָדְשִׁי –

וּבְחַסְדְּךָ תַּצְמִית אֹיְבָי וְהַאֲבַדְתָּ כָּל צֹרֲרֵי נַפְשִׁי.


בָּרוּךְ צוּר, לְבָבִי שְׁמוֹ שָׂם גּוֹרָלוֹ;

חַסְדִּי – יְיָ הוּא לְבַדּוֹ, וְאָנֹכִי – חַבְלוֹ.

יְיָ, מָה אָדָם? – וַתֵּדַע שִׂכְלוֹ,

אָדָם לַהֶבֶל דָּמָה – וּמַה פָּעֳלוֹ?

יְיָ, הַט זְבוּלְךָ וְחֵיל אֶרֶץ – כַּלּוֹ,

וּבְרֹק בָּרָק וְהָפֵץ כֻּלּוֹ!

שְׁלַח יָדֶיךָ לְהַצִּילוֹ –

אֲשֶׁר פִּיהֶם פָּעֲרוּ צָרִים לְהַבְהִילוֹ.

אֱלֹהִים, אַל דֳּמִי לָךְ עַד לְךָ תַּבְדִּילוֹ.

הַנּוֹתֵן תְּשׁוּעָה לְבֶן יִשַׁי אֲצִילוֹ –

פְּצֵנִי וְהָשֵׁב אֶת כְּבוֹדְךָ לְהֵיכָלוֹ.

אֲשֶׁר בָּנֵינוּ שָׂם מְגֻדָּלִים בְּגָדְלוֹ,

מְזָוֵינוּ מְלֵאִים וְיִרֶב צֹאן קְהָלוֹ,

אַלּוּפֵינוּ שָׂם מְסֻבָּלִים אֶצְלוֹ –

אַשְׁרֵי הָעָם שֶׁכָּכָה לוֹ.
