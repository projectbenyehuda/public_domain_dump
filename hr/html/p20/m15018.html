אֵל אֱלֹהֵי הָרוּחוֹת

אֵל אֱלֹהֵי הָרוּחוֹת / שָׁת חָכְמָה בַּטּוּחוֹת, / הָרַחֲמִים וְהַסְּלִיחוֹת.

אָמְנָם הִסְכַּלְתִּי עֲשׂה / בִּמְרֹד לִבִּי בְּעוֹשׂוֹ, / גָּדוֹל עֲוֹנִי מִנְּשׂוֹא!

נְעוּרַי כִּימֵי שִׁכּוֹר / וְשֵׂיבָה רֹאשִׁי תַּעְכֹּר – / וּרְאִיתִיהָ לִזְכֹּר.

יְשׁוּבוּן עוֹד רַעְיוֹנָי / בְּתֹם לְהַעֲמִיד בִּנְיָנָי, / לִפְנֵי שַׁחֵת יְיָ.

אַשְׁרֵי מוֹאֵס הֲבָלָיו / וּמְיַשֵּׁר מַעְגָּלָיו, / בְּחֶמְלַת יְיָ עָלָיו.

בְּחֹק שָׂם לִי הִשִּׂיאַנִי, / יוֹם בּוֹ הוֹכִיחַנִי – / בַּדֶּרֶךְ נָחַנִי.

רֹאשׁ בּוֹאִי וּמַהֲלָכִי / אֵיךְ לֹא אָבִין בְּתוֹכִי, / כִּי עֵירֹם אָנֹכִי?

הֵיךְ לִבִּי הִכְבַּדְתִּי / וְתַאֲוָה, בָּהּ נִלְכַּדְתִּי, / בְּכָל כֹּחִי עָבָדְתִּי?

מַרְגּוֹעַ אֶמְצָא, כִּי / אֶשָּׂא קוֹלִי בִּבְכִי – / כִּי יָרֵא אָנֹכִי.

בַּנְתִּי וָאֵדָעָה, / כִּי מָחָר אֶגְוָעָה – / וְאֵיךְ אֶעֱשֶׂה הָרָעָה?

חֶלֶד כַּצֵּל עָבָר – / וְיֵשׁ תִּקְוָה לְלֵב נָבָר, / כִּי נָכוֹן הַדָּבָר:

בְּשׁוּב הַגֵּו אֶל שָׁרְשׁוֹ, / שָׁב רוּחוֹ לִקְדוֹשׁוֹ / וְנַפְשׁוֹ קְשׁוּרָה בְנַפְשׁוֹ.

לְאֵל חַיַּי נְתִיבוֹתַי, / עַד אִם תַּמּוּ שְׁנוֹתַי / וְשָׁכַבְתִּי עִם אֲבֹתַי.

אֲנִי בְּצֵל שַׁדַּי אֶחְסֶה, / לֹא אֶשְׁאַל וְלֹא אֲנַסֶּה / לְדֵעָה מַה יֵּעָשֶׂה!

יְחִידָתִי חֻפָּשָׁה / מִכָּל חֶמְדָּה חֲדָשָׁה / וּמֵעֲבֹדָה קָשָׁה;

וְיוֹם גּוּף מִנְּשָׁמָה יֶחְסָר – / לְתוֹלַעַת יְהִי נִמְסָר / וְאָכְלוּ אֶת הַבָּשָׂר.

אַשְׁרֵי דּוֹרֵשׁ תְּבוּנָה, / בְּפִיהוּ דִּבֶּר נְכוֹנָה / וַיְהִי יָדָיו אֱמוּנָה.

בּוֹעֲרִים כְּסִילוּת יִפְרְשׂוּן – / וּמֵחֲבֵרִים יְכַסּוּן / הַמַּעֲשֶׂה אֲשֶׁר יַעֲשׂוּן.

מָה אֱנוֹשׁ וּמַחֲשָׁבוֹ? / כִּי כָל סֵתֶר לְבָבוֹ / עַד הָאֱלֹהִים יָבֹא.

מִי רוּחַ אֱלֹהִים בּוֹ / יָרִים תְּרוּמָה בְּחֻבּוֹ, / אֲשֶׁר יִדְּבֶנּוּ לִבּוֹ;

תְּרוּמָתוֹ נִשְׁמָרֶת / לְנַפְשׁוֹ הַנִּשְׁמָרֶת / לְכָבוֹד וּלְתִפָאֶרֶת,

בְּהַגִּיעָהּ לַגְּבֹהִים / עִם כּוֹכְבֵי נְגוֹהִים, / כְּתֻבִים בְּאֶצְבַּע אֱלֹהִים.

הָיָה לְכָל עָוֶל מְלַאכְתּוֹ – / מִי יַעֲנֶה, בְּעֵת בָּשׁתּוֹ, / אִישׁ אֲשֶׁר נִמְצָא אִתּוֹ?

לֹא יוֹסִיף בְּרִשְׁעָתוֹ / לְכַבּוֹת נֵר נִשְׁמָתוֹ / עַד יוֹם הֵעָלֹתוֹ!

יַעֲזֹב דֶּרֶךְ נָטָה / וְיִתְבּוֹנֵן כִּי שָׂטָה – / וְהִתְוַדָּה אֲשֶׁר חָטָא.

וְרַע לְהַקְשִׁיב לֹא יֹאבֶה, / אֶת תַּאֲוָתוֹ מַרְבֶּה, / תּוּקַד בּוֹ, לֹא תִכְבֶּה –

פָּעֳלוֹ לְשֶׁרֶץ נִדְמָה / וְלִבּוֹ מָלֵא מִרְמָה, / כֹּל אֲשֶׁר בְּתוֹכוֹ יִטְמָא!

כָּל יְגִיעָיו שָׁוְא וְתֹהוּ / וְחָלְיוֹ אֵין כָּמֹהוּ – / נֶגַע צָרַעַת הוּא!

יִיגַע אִישׁ בְּכָל מְאֹדוֹ / וְהַכֹּל יְבַקֵּשׁ בַּעֲדוֹ, / אֲשֶׁר לֹא תַשִּׂיג יָדוֹ;

וְאֶת אֲשֶׁר עָלָיו יִשְׁפָּר, / שִׁחֵת, קִבְרוֹ חָפָר / וְכִסָּהוּ בֶּעָפָר.

וְאֵיךְ בְּיִצְרוֹ יִתְגּוֹלֵל / וּבִכְבוֹדוֹ יִתְהוֹלֵל? / קֹדֶשׁ יְיָ חִלֵּל!

מַשְׁחִיתִים דַּרְכֵיהֶם – / יִמְאָסֵם אֱלֹהֵיהֶם, / כִּי מָשְׁחָתָם בָּהֶם;

וּמַשְׂכִּיל סָר בְּחָכְמָתוֹ, / כִּי הוּא בְּחַיָּיו וּבְמוֹתוֹ / בְּיַד הַקֹּנֶה אֹתוֹ.

אֲשֶׁר בּוֹטְחִים בְּקוֹנָם / לְהָסִיר מַחְשְׁבוֹת אוֹנָם – / וְהִתְוַדּוּ אֶת עֲוֹנָם!

וּבוֹגְדִים בָּגְדוּ בֶגֶד – / וְהוּא יוֹשֵׁב מִנֶּגֶד, / וְכִסּוּ אוֹתוֹ בְּבֶגֶד;

וְנַפְשָׁם הַנִּשָּׂאָה / חֶמְדַּת תֵּבֵל מְלֵאָה, / וְנִסְתְּרָה וְהִיא נִטְמָאָה.

אָשִׁיב, בְּטֶרֶם מוֹתִי, / לְטַהֵר אֶת טֻמְאָתִי – / וְאַל אֶרְאֶה בְּרָעָתִי!

מַה לְּאִישׁ נַפְשׁוֹ רָזָה, / וְסוֹד הָאֱמֶת לֹא חָזָה, / כִּי דְבַר יְיָ בָּזָה?

סָפוּ מְתִים בְּהַבְלֵיהֶם, / וְנוֹסְרוּ מַשְׂכִּילֵיהֶם – / וְלֹא יָמוּתוּ גַּם שְׁנֵיהֶם.

אֵיךְ תִּמְעַד אֲשׁוּרֵנוּ – / כִּי הַמָּוֶת יִקְרֵנוּ, / אֲנַחְנוּ וּבְעִירֵנוּ?

אִישׁ יְעַוֵּת מַסְלוּלוֹ / וְיִצְרוֹ מַסִּית שִׂכְלוֹ – / בַּדֶּרֶךְ לְשָׂטָן לוֹ.

שׁוֹבָבִים, מַה לָּכֶם / רְדוֹף אַחֲרֵי עֵינֵיכֶם? / כִּי צֹרֲרִים הֵם לָכֶם!

תָּכִין נַפְשָׁם פָּעֳלָהּ / לְהִנָּקוֹת מִמַּעֲלָהּ – / וַייָ יִסְלַח לָהּ.

בִּינוּ, יְהִירִים, בְּסוֹדְכֶם, / כִּי לְכָבוֹד אֵל בְּרַאֲכֶם – / וְהִשְׁתַּחֲוִיתֶם לָכֶם!

לְהוֹשִׁיבְכֶם מְקוֹם תְּהִלּוֹת / סִבֵּב בַּעְדְכֶם מַעֲלוֹת – / וְלֹא אֲבִיתֶם לַעֲלֹת.

בְּטֶרֶם יִכְלוּ שָׁנָי / אוֹדֶה וְאֶעֱזֹב זְדוֹנָי, / פֶּן יֶחֱרֶה אַף יְיָ.

יוֹדֵעַ רֵאשִׁיתוֹ / יָבִין, כִּי בְּאַחֲרִיתוֹ / יְיָ הוּא נַחֲלָתוֹ.

לָנֶצַח הוּא בְמַעֲלָה – / וּבָאֲדָמָה הַשְּׁפָלָה / אֵין לוֹ חֵלֶק וְנַחֲלָה.

יְעִירֵהוּ מַחֲשָׁבוֹ / סוֹד מֵאַיִן יָבוֹא, / לְבִלְתִּי רוּם לְבָבוֹ.

פֶּתִי, יְיַסֵּר אוֹתוֹ / אֵל – יָרוּץ לְרַצּוֹתוֹ, / פֶּן יֹסִיף לְהַכֹּתוֹ.

יְמַהֵר, עַד יֹאבַד, / וִיכַבֵּד בְּכָל מַעֲבָד / אֶת הַשֵּׁם הַנִּכְבָּד;

מַה לְּרוּחוֹ וּלְעָצְבָּהּ, / בְּסוֹד חֳלָאִים בְּקִרְבָּהּ / אֲשֶׁר חִלָּה יְיָ בָּהּ?

עֲוֹנִי טָמוּן בְּחֻבִּי – / וְאֵיךְ יַעֲלֶה עַל לִבִּי, / כִּי אֵין אֱלֹהַי בְּקִרְבִּי?

סוֹד חַיִּים וּשְׁבִילוֹ / וְסוֹד מָוֶת הֶרְאָה לוֹ / הַצּוּר תָּמִים פָּעֳלוֹ.

נְהַלֶּל יָהּ כֻּלָּנוּ / עַל חֶסֶד גְּמָלָנוּ – / תּוֹרָה צִוָּה לָנוּ.
