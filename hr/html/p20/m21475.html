אָשִׁיר בְּקוֹלִי שִׁיר

חרוז על הסקאקי לר״א


אָשִׁיר בְּקוֹלִי שִׁיר וְלֹא סִיר יִהְיֶה

קוֹלִי, אֲשֶׁר אָרִים כְּמוֹ רַב גִּילָה,

אֶכְתֹּב עֲלֵי סֵפֶר דְּבַר שֶׁפֶר אֲשֶׁר

לִי שַׁעֲשׁוּעִים הוּא, עֲדֵי אֵין תִּכְלָה,

פֶּן יִהְיֶה מִנִּי כְּרֶגַע נֶעְלָם,

פֶּן יֹאבְדוּ כָּל דֹּרְשִׁים הַסְּגֻלָּה;

לָכֵן בְּקוֹל שִׁירָה וְגַם זִמְרָה הֲלֹא

אַגִּיד לְכָל גֶּבֶר: קְרָב בַּתְּחִלָּה

בִּשְׁנֵי מְלָכִים נִלְחֲמוּ יַחַד, וְהֵן

אֶחָד שְׁמוֹ כּוּשִׁי, וְיָדוֹ גָדְלָה

אוֹיְבוֹ אֲשֶׁר נֶגְדּוֹ הֲלֹא יוֹצֵא בְעַם

לָבָן מְאֹד יָפֶה, לְבָבָם גָּעֲלָה

כִּי יַחְשֹׁב אוֹיְבוֹ לְקַשׁ כָּחוּשׁ, וְגַם

קָנֶה רָצוּץ תּוֹךְ קְרָב וּרְעָלָה,

יַחְדָּו יְבֹאן הֵם לְהִלָּחֵם, וְיֵשׁ

לָהֶם גְּדוּד שָׁוֶה וְרָב, וּכְפוּלָה

רַגְלִים, וְגַם סוּסִים וּמִבְצָרִים, וְגַם

מִשְׁנֶה לְכִסְאוֹתָם, וְלָהֶם מַעֲלָה

מֶלֶךְ וְהַמַּלְכָּה, הֲלֹא הֵם יֹשְׁבִים

אֶל תּוֹךְ גְּדוּדֵיהֶם בְּשִׁיר גַּם צָהֲלָה

הֵם עָרְכוּ יַחַד לְהִלָּחֵם בְּתוֹךְ

עֵמֶק שְׂדֵה שָׁוֶה, בְּלִי שׁוּם חֶמְלָה,

הֵן הוּא שְׁמוֹנָה עַל שְׁמוֹנָה כֵּן לְכָל

רוּחַ, לְכָל גֶּבֶר בְּקִרְבּוֹ נַחֲלָה,

הֵן כַּאֲשֶׁר יַחְדָּו יְנוּעוּן לַעֲרוֹךְ

הַמִּלְחָמָה, אָז בְּקוֹל חַלְחָלָה

נִשְׁמַע תְּרוּעָתָם, יְנוּעוּן אָז בְּקוֹל

שָׁאוֹן, וְיִשָּׁמַע אָז קוֹל חִילָה,

כִּי מִמְּקוֹמָם נוֹסְעִים רַגְלִים

וְיָבֹאוּ לְהִלָּחֵם בְּחֶרֶץ וְכָלָה,

אַךְ בַּתְּחִלָּה יוּכְלוּ כִּי אִם שְׁנֵי

בָּתִים בְּקַו לַהֲלוֹךְ בְּתוֹךְ הַסְּלוּלָה,

מִכַּאן לְהַבָּא יוּכְלוּ הֵם לַהֲלוֹךְ

תָּמִיד בְּבֵית אֶחָד וְלֹא כְּבַתְּחִלָּה.

הֵן כַּאֲשֶׁר לְנֶגְדָּם אוֹיְבָם

נִתַּן אֲזַי לָהֶם רְשׁוּת וּתְהִלָּה

לִתְפֹּשׂ אֲזַי אוֹיְבָם בַּאֲלַכְסוֹן, וְאָז

יִהְיֶה לְתִפְאֶרֶת אֲבָל לֹא כִסְלָה.

אִם יַעֲלֶה רַגְלִי וְיִשְׁבֹּר מַחֲנֵה

אוֹיְבָיו וְיַגִּיעַ בְּרֶגֶל קַלָּה

רֹאשׁ מַחֲנֵה הַצָּר בְּלִי פֶגַע, אֲזַי

יוּשַׂם עֲלֵי רֹאשׁוֹ עֲטֶרֶת גְּדוֹלָה;

לוֹ יִהְיֶה כֹּחַ כְּמוֹ מַלְכָּה, אֲשֶׁר

תָּמִיד בְּכָל פִּנָּה הֲלֹא הִיא עוֹלָה,

סוּסִים בְּעֵת יִרְאוּ גְּדוּדִים יַעֲלוּ,

הִנֵּה אֲזַי נִשְׁמַע שְׁאוֹן בַּמְּסִלָּה,

מַהֵר יְרוּצוּן לַעֲזוֹר הָרַגְלִים,

בָּתִים שְׁלֹשָׁה רֹקְדִים בַּמְּצוּלָה,

דֶּרֶךְ עֲקַלָּתוֹן כְּגִבּוֹרִים יְרוּצוּן

אָז בְּלִי אֵחוּר וְגַם לֹא עַצְלָה.

מִבְצָר וּמִגְדָּל עֹז וּמָצוֹר כַּאֲשֶׁר

יִרְאֶה פְּנֵי הַקְּרָב לְנֶגְדּוֹ כָּלָה,

יִסַּע וְיָקוּם מִמְּקוֹמוֹ אָז וְיַהֲלֹךְ

עַל דְּרָכִים הֵם יְשָׁרִים סֶלָה.

יֵשׁ לוֹ רְשׁוּת לַהֲלֹךְ בְּכָל צַד הַקְּרָב.

מִצַד אֱלֵי פִנָּה, כְּרֶגַע יַעֲלֶה

מִשְׁנֶה לְמוֹ מֶלֶךְ, הֲלֹא גַם הוּא אֲזַי

יָרוּץ כְּמוֹ גִבּוֹר בְּרֹב תַּחְבֻּלָּה,

לִרְאוֹת פְּנֵי הַקְּרָב וְגַם לַעֲזוֹר לְכָל

נוֹפֵל, עֲלֵי מַכָּה טְרִיָּה נֶחֱלָה

יוּכַל הֲלֹךְ דֶּרֶךְ עֲקַלָּתוֹן כְּמוֹ

רָצוֹן לְבָבוֹ תּוֹךְ קְרָב וּשְׁבִילָה;

מַלְכָּה לְכָל אַרְבַּע קְרָנוֹת תַּעֲלֶה,

אֵין מִי אֲשֶׁר יֹאמַר לְנֶגְדָּהּ מִלָּה,

קַלָּה כְּמוֹ הַצְּבִי וַחֲזָקָה כַּאֲרִי,

תִּלְחַם בְּלֵב חָזָק בְּלִי לָהּ עֲמֵלָה.

מֶלֶךְ בְּעֵת יִרְצֶה לְדַלֵּג בַּתְּחִלָּה,

אָז רְשׁוּת אֵלָיו בְּדַת וּפְלִילָה,

לַעֲלוֹת לְבֵית מִבְצָר וְשָׁם יַעֲמֹד, וְאִי

לוֹ יִהְיֶה מִפְלָט לְיוֹם צוּק וְחֶבְלָה.

תָּמִיד יְשַׁלַּח מִבְצָר חִישׁ אֶל מְקוֹם

שִׁבְתּוֹ לְהִלָּחֵם בְּחוֹמָה נָפְלָה,

אַךְ אִם יְבוֹאוּן אוֹיְבִים נֶגְדּוֹ וְיוֹרוּ

אָז הֲלֹא יִבְרַח אֱלֵי קוֹל חִילָה

אֶחָד בְּכָל עוֹנָה הֲלֹא אֵלָיו רְשׁוּת

יִדְלַג בְּכָל בַּיִת בְּאֶחָד סֶלָה,

וְיַחֲזִיקוּ אִישׁ בְּרֹאשׁ אוֹיְבוֹ בְּרֹב

סוֹאֵן סְאוֹן וּמְגוֹלְלָה הַשִּׂמְלָה.

וַיִּתְּנוּ נַפְשָׁם בְּעַד נֶפֶשׁ

אֲדוֹנֵיהֶם וְנֶהֶרְגוּ וְדָם הִגְדִּילָה,

אֵלֶּה כְאֵלֶּה נָפְלוּ אַָרְצָה חֲלָלִים,

עַד אֲשֶׁר רוּחָם הֲלֹא הֵן סֶּלָה.

הִנֵּה פְּעָמִים יִגְבְּרוּ כוּשִׁים וְהֵן

כֹּחַ לְבָנִים אוֹיְבֵיהֶם נָפְלָה,

יִנָּגְפוּ כֻלָּם וְגַם מֶלֶךְ הֲלֹא

יִפּוֹל בְּתוֹךְ רִשְׁתָּם בְּקַבְּצָם כֻּלָּהּ,

גַּם יִשְׁפְּכוּ דָמוֹ וְיוֹרִידוּ אֲזַי

אוֹתוֹ בְּרֹב מַכּוֹת כְּמֵת לִשְׁאֹלָה.

גַּם יֵשׁ פְּעָמִים כִּי לְבָנִים יְגַבְּרוּ

גַם הֵם עֲלֵי כוּשִׁים, בְּרֹב חַלְחָלָה

נִשְׁמַע בְּאִי כוּשִׁים הֲכִי הֵם יִצְעֲקוּ

גַּם יִתְּנוּ קוֹלָם בְּמַר כְּמוֹ חוֹלָה,

דֶּרֶךְ לְהַצִּילָם הֲלֹא אָיִן, וְלֹא

נִמְצָא לְמוֹ נַפְשָׁם צְרִי וּתְעָלָה,

רַק כִּי יְשׁוּבוּן יִלְחֲמוּ, אוּלַי

לְנַפְשָׁם תִּהְיֶה כָבוֹד יְקָר וּגְדֻלָּה.
