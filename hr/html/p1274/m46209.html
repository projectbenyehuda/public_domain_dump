רַשִׁ"י וְהַדֻּכָּס גוֹטְפְרִיד

רַשִׁ“י, הוּא רַבִּי שְׁלֹמֹה יִצְחָקִי, מְפָרֵשׁ הַתּוֹרָה שֶׁבִּכְתָב וְהַתּוֹרָה שֶׁבְּעַל־פֶּה, נוֹלַד בְּצָרְפַת, אַךְ בָּא לָגוּר בְּווֹרְמֵיזָא שֶׁבְּאַשְכְּנַז, וְשָׁם הָיָה לוֹ בֵּית מִדְרָשׁ שֶׁהָיָה מְלַמֵּד בּוֹ לְתַלְמִידִים הַרְבֵּה. בֵּית מִדְרָשׁ זֶה הָיָה מְשַׁמֵּשׁ גַם בֵּית כְּנֶסֶת לִתְפִלָּה, הוּא בֵּית הַכְּנֶסֶת שֶׁל רַשִׁ”י בְּווֹרְמֵיזָא הַנִּקְרָא כֵּן עַד הַיּוֹם הַזֶּה.

וּבַיָּמִים הָהֵם בִּקֵּשׁ הַדֻכָּס הַנָּצְרִי שֶׁל לוֹתְרִינְגֶן, גוֹטְפְרִיד מִבּוּיוֹן, לַעֲלוֹת עִם חֵילוֹ לְמִלְחָמָה עַל יְרוּשָׁלַיִם עִיר הַקֹּדֶשׁ וּלְכָבְשָׁהּ מִידֵי מוֹשְׁלֶיהָ הַמֻּסְלְמִים, הִיא הַמִּלְחָמָה הַנּוֹדָעָה בְּשֵׁם מִלְחֶמֶת “מַסְעֵי הַצְּלָב”. רַבּוֹת שָׁמַע הַדֻּכָּס עַל רַַשִׁ"י הֶחָכָם וְהַצַּדִּיק, שֶׁהוּא מְכֻבָּד עַל יְהוּדִים וְנָצְרִים וְרַבִּים יַחְשְׁבוּהוּ לְנָבִיא. אָמַר הַדֻכָּס בְּלִבּוֹ: “לִפְנֵי שֶׁאֲנִי יוֹצַא לְדֶרֶךְ רְחוֹקָה וּלְמִלְחָמָה קָשָׁה, אֶשְאַל דַעָה וְעֵצָה מִפִּי הֶחָכָם הַיְּהוּדִי”.

שָׁלַח שָׁלִיחַ לִקְרֹא לְרַשִׁ“י, שֶׁיָבוֹא לְבֵיתוֹ וְיִתֶּן לוֹ עֵצָה בְּדָבָר חָשוּב. סֵרַב רַשִׁ”י לָבוֹא אֵלָיו, שֶׁמָּא הוּא מִתְאַנֶּה לוֹ.

כְּשֶׁשָּׁמַע הַדֻכָּס שֶׁרַשִׁ“י מְסָרֵב לָבוֹא לְפָנָיו, נִמְלָא חֵמָה עַל הַיְּהוּדִי שֶׁאֵינוֹ מְצַיֵּת לוֹ, בָּא עִם חֵילוֹ לָעִיר ווֹרְמֵיזָא וְהָלַךְ לְבֵיתוֹ שֶׁל רַשִׁ”י.> מָצָא אֶת דַלְתוֹת הַבַּיִת פְּתוּחוֹת לִרְוָחָה, הַסְּפָרִים מֻנָּחִים עַל הַשֻּׁלְחָן וְאֵין אִישׁ בַּבָּיִת.

קָרָא הַדֻּכָּס בְּקוֹל גָדוֹל: שְׁלֹמֹה, שְׁלֹמֹה, אַיֶּךָּ?

הֵשִׁיב קוֹל מִן הָאֲוִיר: מָה רְצוֹנְךָ כְּבוֹד הַדֻּכָּס?

שָׁאַל הַדֻּכָּס שׁוּב: שְׁלֹמֹה אַיֶּךָּ?

וְקוֹלוֹ שֶׁל רַשִׁ"י עָנָה: כָּאן אֲנִי, אֲדוֹנִי הַדֻּכָּס.

פַּעֲמַיִם וְשָׁלֹש קָרָא הַדֻּכָּס לְרַשִׁ"י וּבְכָל פַּעַם הֵשִיב הַקּוֹל, אַךְ הַדֻּכָּס לֹא רָאָה כָּל דְמוּת הַגּוּף.

יָצָא הַדֻּכָּס מִן הַבַּיִת מֻכֵּה תִמָּהוֹן. בָּרֶגַע שֶׁיָּצָא עָבַר עָלָיו אֶחָד מִתַּלְמִידֵי רַשִׁ“י. שְׁאֵלוֹ הַדֻּכָּס: כָּאן דָר רַשִׁ”י?

עָנָה הַתַּלְמִיד: כֵּן, הוּא רַבִּי וּמוֹרִי.

אָמַר לוֹ הַדֻּכָּס: אֱמֹר לְרַבְּךָ שֶׁיָבוֹא אֶצְלִי, אֵצֶל הַדֻּכָּס גוֹטְפְרִיד. נִשְׁבַּעְתִּי בְּרֹאשִׁי שֶׁלֹּא אֶגַּע בּוֹ לְרָעָה.

כְּשֶׁשָּׁמַע רַשִׁ"י כָּךְ מִיָּד נִתְגַלָּה לַדֻּכָּס וְנָפַל לְרַגְלָיו מִפְּנֵי הַכָּבוֹד. הֶעֱמִידוֹ הַדֻּכָּס עַל רַגְלָיו וְאָמַר: הִנֵּה רָאִיתִי כִּי אִישׁ חָכָם וְקָדוֹשׁ אַתָּה וְעַתָּה אַגִּיד לְךָ לָמָּה בָּאתִי לְבֵיתְךָ. עוֹמֵד אֲנִי בְּעִנְיָן גָדוֹל: אָסַפְתִּי חֵיל רַגְלִים וּפָרָשִׁים רַב כְּחוֹל הַיָּם וַאֲנִי הוֹלֵך לִכְבֹּש אֶת יְרוּשָלָיִם. בּוֹטֵחַ אֲנִי בֵּאלֹהִים, שֶׁיִּהְיֶה עִמִּי וְיַעַזְרֵנִי לְהַכּוֹת אֶת הַתֻּרְכִּים הַכּוֹפְרִים מַכָּה רַבָּה אֲשֶׁר לֹא יָקוּמוּ עוֹד. אַךְ לִפְנֵי צֵאתִי לַדֶּרֶך חָפַצְתִּי לִשְׁמֹעַ אֶת דַעְתְּךָ וַעֲצָתְךָ. דַבֵּר דְבָרְךָ בְּגָלוּי וְאַל תִּירָא אִם לְנִצָּחוֹן תִּהְיֶה נְבוּאָתְךָ אוֹ לְמַפָּלָה. יָדַעְתִּי כִּי נְבִיא אֶלֹהִים אַתָּה בְּתוֹכֵנוּ וְתוּכַל לְהַגִּיד לְאָדָם מַה צָּפוּי לוֹ בַּמִּלְחָמָה. הַגִּידָה לִי, אֵפוֹא, אֶת הָאֱמֶת, הַאֲנַצֵּחַ בַּמִּלְחָמָה אִם לֹא?

הֵשִׁיב לוֹ רַשִׁ"י קְצָרוֹת: אֲדוֹנִי הַדֻּכָּס הַגָּדוֹל! לֹא אֲכַחֵד מִמְךָ אֶת הָאֶמֶת: בַּתְּחִלָּה תִּהְיֶה הַצְלָחָתְךָ שְׁלֵמָה וְתִכְבּשׁ אֶת יְרוּשָלַיִם וְתִמְלֹךְ בָּהּ שְׁלשֶׁת יָמִים. אַךְ בַּיּוֹם הָרְבִיעִי יֵאָסְפוּ עָלֶיךָ חֵילוֹת הַתֻּרְכִּים שֵׁנִית וִיגָרְשׁוּךָ מִן הָעִיר וְאַתָּה תִּמָּלֵט עַל נַפְשְׁךָ. רֹב חֵילְךָ יֵהָרֵג בַּחֶרֶב וְהַנִּשְׁאָרִים יָמוּתוּ בַּדֶּרֶךְ, וְאַתָּה תָּשׁוּב לָעִיר הַזֹּאת עִם שְׁלשָׁה סוּסִים בִּלְבָד. הִנֵּה זֹאת חָזִיתִי וְהִגַּדְתִּי לְךָ, וְאַתָּה עֲשֵׂה כִּרְצוֹנְךָ.

כְּשֶׁשָּׁמַע הַדֻּכָּס אֶת דִבְרֵי רַשִׁ"י הִכָּהוּ לִבּוֹ, אַךְ הִתְחַזֵּק וְאָמַר: יִתָּכֵן כִּי אֱמֶת יֶהְגֶה חִכְּךָ וְיָבוֹא עָלַי כָּל מַה שֶּׁנִּבֵּאתָ לִי, אֲבָל זֹאת תֵּדַע לְךָ: אִם אָשׁוּב אֶל עִירִי עִם אַרְבָּעָה סוּסִים בִּמְקוֹם הַשְּׁלשָׁה שֶׁאָמַרְתָּ, אַשְׁלִיךְ אֶת בְּשָׂרְךָ לַכְּלָבִים וְאֱהֶרֹג אֶת כָּל הַיְהוּדִים הַיּוֹשְׁבִים בְּאַרְצִי.

וְהַדֻּכָּס יָצָא לְמַסָּעוֹ עִם רֹב חֵילוֹ.

כָּל מַה שֶּׁנִּבָּא רַשִׁ"י נִתְקַיֵּם בּוֹ בְּאוֹתוֹ דֻכָּס. אַחֲרֵי שְׁלשֶׁת יָמִים לְכִבּוּשׁ יְרוּשָׁלַיִם הֻכָּה גוֹטְפְרִיד מַכָּה רַבָּה וּשְׁאֵרִית חֵילוֹ מֵתוּ בַּדֶּרֶךְ, וּמִקֵּץ אַרְבַּע שָׁנִים שָׁב לווֹרְמֵיזָא עִם שְׁלשָׁה פָּרָשִׁים רוֹכְבִים עַל שְׁלשָׁה סוּסִים, וְהַדֻּכָּס עַצְמוֹ רוֹכֵב עַל סוּסוֹ, בְּיַחַד אַרְבָּעָה.


איור: הנריק הכטקופף


כְּשֶׁהִגִּיעַ אֶל שַׁעַר הָעִיר זָכַר מַה שֶּׁנִּבָּא לוֹ רַשִׁ“י, שֶׁסּוֹפוֹ לָשׁוּב עִם שְׁלשָׁה סוּסִים, וְהִנֵּה הוּא שָׁב וְהַסּוּסִים הֵם אַרְבָּעָה. הָיָה לִבּוֹ מָלֵא כַּעַס וְזָמַם לַהֲרֹג אֶת רַשִׁ”י, כִּי כֵן הִבְטִיחוֹ, שֶׁאִם יָשׁוּב עִם אַרְבָּעָה סוּסִים יַשְׁלִיךְ אֶת בְּשָׂרוֹ לַכְּלָבִים וְיַהֲרֹג אֶת כָּל הַיְּהוּדִים הַיּוֹשְבִים בְּאַרְצוֹ.

כָּךְ חָשַׁב הַדֻּכָּס, אֲבָל הַקָּדוֹשׁ בָּרוּךְ הוּא מְסַכֵּל עֲצַת רְשָׁעִים. בְּשָׁעָה שֶׁהַדֻּכָּס בִּקֵּשׁ לְהִכָּנֵס בְּשַׁעַר הָעִיר, נָפַל כְּלוֹנַס הַשַּׁעַר הֶעָשׂוּי בַּרְזֶל עַל הַסּוּס הַרְבִיעִי הַהוֹלֵךְ בָּאַחֲרוֹנָה וְהִתִּיז רֹאשׁוֹ וְרוֹכְבוֹ נִשְׁאַר מִבַּחוּץ. וּבְכֵן נִכְנַס הַדֻּכָּס בְּשַׁעַר הָעִיר עִם שְׁלֹשָה סוּסִים וְלֹא עִם אַרְבָּעָה, כְּמוֹ שֶׁנִּבָּא לוֹ רַשִׁ"י הַקָּדוֹשׁ.

נָפַל פַּחַד עַל הַדֻּכָּס וְהוֹדָה שֶׁנְּבוּאָתוֹ שֶׁל רַשִׁ"י הָיְתָה אֱמֶת. יָרַד מֵעַל סוּסוֹ וּמִהֵר לְבֵית הַצַּדִּיק לְהִשְׁתַּחֲווֹת לוֹ וּלְהוֹדוֹת לוֹ עַל נְבוּאָתוֹ.

אַךְ כְּשֶׁבָּא לְבֵית רַשִׁ"י מְצָאוֹ מֻטָּל מֵת בְּתוֹךְ הָאָרוֹן מוּכָן לִקְבוּרָה וְעַם רַב בָּא לְלַוּוֹתוֹ לְבֵית עוֹלָמוֹ. נִלְוָה גַם הַדֻּכָּס אֶל הַמְלַוִּים וְיָמִים רַבִּים הָיָה מִתְאַבֵּל עַל מוֹת הַצַּדִּיק.
