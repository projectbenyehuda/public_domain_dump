בְּלֵילוֹת־זִיו

א

בְּלֵילוֹת־זִיו אֵתַע אִלֵּם

מֻצָּף רוּחוֹת וּמַזָּלוֹת.

תּוֹעָה יַתְמוּת עַל מִשְׁעוֹלִי –

הֵד נְשָׁמוֹת לֵאוֹת, כָּלוֹת.


הָרוֹם יִקְרָא. יָרִיד הַגֵּיא

וְיִרְעֲדוּן דִּמְדּוּמֵי־צֵל.

רֹךְ־בִּכְיְךָ, הַגֵּיא, אֶשָּׂא

אֶזְרֶה בְּמֶרְחַבֵּי־הַלֵּיל.


לַפִּיד־כּוֹכָב, מַשַּׁק־חֲלוֹם

יִפּוּנִי, נִשְּׂאוּנִי דֹם.

בְּלַהַב נִשְׁמָתִי אַתִּיךְ

כִּלְיוֹן־הַגֵּיא, תִּפְאֶרֶת רוֹם.


בּ

מָה־עַזִּים הַמֶּרְחַקִּים.

מֵיתָרַי רָנִים.

זֹהַר רַד – וָאַרְא שָׁם

חֹרֶשׁ־תְּרָנִים.


לִי סְפִינוֹתַי, שְׁלַל

נֹעַר וְשִׂיאִים.

גֵּא הַשֶּׁטֶף, בּוֹ אַפְלִיג

וַאֲגַל אִיִּים.


שַׁעֲרֵי־חֲלוֹם וָסוֹד

חֶרֶשׁ נִפְתָּחִים.

בֹּקֶר זַךְ חוֹפֵף בַּלֵּב.

מֵיתָרַי נָחִים.


ג

עַל גִּבְעַת־לֵילִי יִשְׂגֶא אֶרֶז־גַּעְגּוּעָי.

כִּתְרְךָ, אֵין הַסּוֹף, הֵצִיף עָלַי נִצְחוֹ

וַיַּצֶּת תִּפְאֶרֶת מַלְכוּתִי.

עוֹרֵג אֶעֱלֶה עֲנָנִים וְכוֹכָבִים,

בַּסְּפֵירוֹת אֶנָּשֵׂאָה.

סוֹעֲרוֹת בִּנְתִיבִי נִשְׁמוֹת־חוֹזִים

חוֹלְמֵי־קְדוּמִים, צְמֵאֵי־מַעְיְנוֹת־הַהֲוָיָה.

יֵשׁ עָמְדוּ עַל אֵם־דַרְכָּן וְקָפְאוּ עִם חֲלוֹמָן,

יֵשׁ הוֹמוֹת וּמְשַׁוְּעוֹת קְצוּצוֹת־כְּנָפַיִם…

אֶל כִּתְרְךָ, אֵין־הַסּוֹף, אָבֹאָה.

יָדַעְתִּי, כָּמוֹנִי הֵן תִּיצֶר רְקִיעִים

וְאֵלִים וְיַמִּים וּסְעָרוֹת.

יָדַעְתִּי, דִּמְעוֹתֶיךָ – הַיְּאוֹרוֹת הָעֲגוּמִים,

הֵדֵי יְגוֹנְךָ – מֶרְחַקֵּי־הָאֹפֶל

וּמַעְיְנוֹתֵיהֶם הָאִלְּמִים. – – –

גַּם אַתָּה, אֵין הַסּוֹף, מִתְגַּעְגֵּעַ

כָּמוֹנִי, הָעוֹרֵג עַל גִּבְעַת־לֵילִי?

אַלְלַי, אֵי יַחֲנוּ, אָן תִּשְׁלַח גַּעְגּוּעֶיךָ

הַיָּפִים, הָעֲצוּמִים, כִּשְׁלֹחַ עֵץ עֲנָפָיו

וְשֶׁמֶש קַרְנָיו?…

עַל גִּבְעַת־לֵילִי, יִשְׂגֶא אֶרֶז־גַּעְגּוּעָי.


ד

בְּלֵילוֹת־זִיו בַּדּוּמִיָּה

תִּשַּׁק בִּכְיָה מְסִלּוֹתַי.

קַו אֵין־הַסּוֹף הוֹמֶה שָׁט

בְּמֶרְחַבֵּי־חֲלוֹמוֹתַי.


הוֹמֶה שָׁט בְּלֶב־הָרִים,

בְּרַחַשׁ־צֵל, בְּהֵד־גַּלִּים.

שָׁט נֶהִי רַךְ בַּמֶּרְחַקִּים

וַעֲזוּבִים הָעַרְפַּלִּים…


ה

הַחֹרֶשׁ עַל נִבְלוֹ יָרֹן.

הַיְּאוֹר יִזְרַע חֵן־אַגָּדוֹת.

בָּשֵׁל גָּמֵל הַהֵרָיוֹן

בְּשִׁבֳּלֵי־שָׂדוֹת.


בְּנִשְׁמָתִי נִצּוֹת עוֹלוֹת,

מַגַּל־הַסַּהַר יִקְצְרֵן;

לְנוֹף עָבִים וּמְחוֹלוֹת

הָרוּחַ יִשָּׂאֵן.


ו

נִגְלֵית, עֲנָנָה, וְרֻדַּת־עֲלוּמִים,

וַיָּנַע נִצְחֵךְ.

יִפְעַת־דְּמוּתֵךְ – רֹאשׁ־עַלְמָה זַךְ, קָדוֹשׁ.

כּוֹכָב עַל מִצְחֵךְ.


כְּיָד עַל עוּגָב כֵּן תִּרְעַד עַל נִימַי

עֶדְנַת־שְׁבִיבֵךְ.

עַינִי עֲצוּמָה וְדוֹפֶקֶת נַפְשִׁי

בְּשַׁעַר זִיוֵךְ…


כְּעָשָׁן־עוֹלָה אִנָּשֵׂא לָךְ כֻּלִי

בְּלֵיל יְצִירָה,

וְהֵד כְּבָלַי נִתָּקִים יִתָּלֶה

בְּרוֹם־סְפֵירָה…


ז

(הלילית והסהר).

יוֹם יוֹם יִנָּשֵׂא הַדְּבִיר

זוֹעֵף וַעֲיֵף־שִׁמָּמוֹן.

זָרִים לוֹ הַגִּיל וְהָאוֹר,

צָצִים יְגוֹנָיו בֶּהָמוֹן.

בַּלַּיְלָה, עֵת סַהַר יָהֵל,

וְצֵל כְּדִמְעָה שֶׁפִי נָח –

נִיד־סֵתֶר עוֹבֵר אָז סָבִיב,

נִנְעָר הַדְּבִיר וְיִפְרָח.

וִיהִי כְנֵד־קֶסֶם נוֹהֵר,

זִיוִים רוֹעֲדִים נִתָּכִים.

כֵּפָיו מוּפָזִים. מַשְׁחִירִים

פְּרָצָיו כִּדְמוּת־סְבָכִים.

נֵד־פֶּלֶא, אַרְמוֹן מְכֻשָּׁף,

מַלְכָּה צְנוּעָה בּוֹ בַּלָּט.

עוֹד רֶגַע תּוֹפַע הַמַּלְכָּה,

הוֹדָהּ עִם עָבֵי־שַׁחַק שָׁט…

עוֹד רֶגַע… עַל רֹאשׁ־הַדְּבִיר

הוֹפִיעָה לִילִית רַכַּת־צֵל.

נָפַל חִוָּרוֹן בַּמֶּרְחָב,

לֹא נָע וְלֹא זָע רוּחַ־לֵיל…

אַשְׁמֹרֶת־עֵינָהּ עֲגֻלָּה,

לוֹהֵט אִישׁוֹנָהּ וְנוֹהֶה.

רֹאשָׁהּ אַט יַרוֹם. מִמּוּלָהּ

הַסַּהַר פּוֹרֵחַ, תּוֹעֶה.


"הוֹי, סַהַר! אוֹרְךָ הֵן יַזִּיל

גִּיל־יֹפִי בְּאֵבֶל־יוֹמָהּ.

אֵלֶיךָ סוּפַת־נִשְׁמָתָהּ,

עָלֶיךָ רֻקַּם חֲלוֹמָהּ"…


נָדַמָּה רִנַּת־הָאֲגָם,

נֶאְלַם נִיד בְּסֵתֶר־סְבָךְ.

יָשׁוּט לוֹ הַסַּהַר גֵּא, קַר,

כְּנַף־הַלִּילִית חֶרֶשׁ תַּךְ…


ח

הִדְלַקְתִּי נֵרִי וַיַּכְחֵל חַלּוֹנִי.

יָדַעְתִּי: תִּתְעִי שְׁבוּרָה;

נוֹטַעַת נַפְשֵׁךְ צִיץ־שְׁכֹל חֲשׂוּךְ־נִיד

עַל רֹאשׁ הִרְהוּרָהּ.


כְּסַהַר פָּגוּם הַצָּמֵא לְאָחוֹת

בְּלֵב שְׁחָקִים,

תִּתְעִי, תַּעַבְרִי חֻדֵּי־הָר אַכְזָרִים

וְתוּגַת עֲמָקִים.


גִּבְעוֹת־כְּרָמִים מִימִינֵךְ, מִשְּׂמֹאלֵךְ.

הָאשֶׁר שָׁם נָם.

נוֹטֵר רַע יַכֵּךְ, מְשׁוּכַת־הַקּוֹצִים

תִּדְקְרֵךְ עַד זוֹב דָּם.


תָּנוּחִי עַל אֶבֶן. הָאֶבֶן הוֹגָה.

דּוֹבֵב הַחֲלוֹם:

"בְּסֵתֶר נָוֵהוּ חַיָּה מַלְכוּתֵךְ,

עֶדְנָהּ עוֹד תִּרְקֹם.


וְאוֹר־עֲלוּמַיִךְ זוֹלֵף בְּלִבּוֹ

וִיפִי עֻנּוֹתֵךְ

יַכְחִיל נֵר־הַקֶּסֶם אֶת אֵלֶם־נַפְשׁוֹ;

נַפְשׁוֹ – אֲחוֹתֵךְ"…


חוּט־סֵתֶר נִמְשָׁךְ וּבוֹכֶה וְלוֹאֵט

לְנַפְשִׁי: “אֲחוֹתֵךְ”…

אֲגַל סְפֵירוֹת וּגְאוֹן הֲוָיָה

בְּסוֹד עֻנּוֹתֵךְ.
