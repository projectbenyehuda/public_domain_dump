אִישׁ תָּם

אִישׁ תָּם יוֹשֵׁב אָהֳלֵי תוּשִׁיָּה

בֹּחַן פִּנַּת יִקְרַת מוֹסְדוֹת נְשִׁיָּה

גָּשׁ וַיִּפְגַּע בְּאֶבֶן הַשְּׁתִיָּה / אֶת כָּל הַתְּפִלָּה וְהַתְּחִנָּה הַזֹּאת


דָּפַק לָלוּן בְּמָקוֹם נִפְרָז

הִפְלִיא שַׁדַּי לְגַלּוֹת לוֹ רָז

וַיַּרְא סֻלָּם עַד זְבוּל נִגְרָז / לֹא נִהְיְתָה וְלֹא נִרְאֲתָה כָּזֹאת


זָמַם לְבַשְּׂרוֹ בְּמַלְאֲכֵי מְעוֹנַי

חֲזוֹת אַחִים צִירֵי אֱמוּנַי

טַעֲמֵי סֻלָּם סְפוּרִים בְּסִינַי / מֵאֵת יְיָ הָיְתָה זֹאת


יָזַם לְהַנְחִילוֹ בְּפַעַם שִׁשִּׁית

כַּנַּת חֶמְדַּת תְבוּאַת רֵאשִׁית

לְחוּט הַמְּשֻׁלָּשׁ יָעַד שְׁלִישִׁית / לְזַרְעֲךָ אֶתֵּן אֶת הָאָרֶץ הַזֹּאת


מִצְעֲדֵי אֲשׁוּרֶיךָ אֲכוֹנֵן בְּכָל פֶּלֶךְ

נִינֶיךָ יִרְבְּיוּן בְּתוֹךְ קִרְיַת מֶלֶךְ

סְבִיבְךָ אֶהְיֶה בְכָל אֲשֶׁר תֵּלֵךְ / וַהֲשִׁיבוֹתִיךָ אֶל הָאֲדָמָה הַזֹּאת


עִטֵּר פִּנַּת יִקְרַת הַהֶדֶר

פָּצוּ שְׂפָתָיו וַיִדַּר נֵדֶר

צוּרִי אִם תַּעַשׂ דְּבָרִי כְּסֵדֶר / אַף אֲנִי אֶעֱשֶׂה זֹאת


קָפַץ לַעֲבֹד בְּתֹם סַרְעַפָּיו

רְכוּשׁוֹ פָּרַץ וְרַבּוּ טַפָּיו

שָׁפַט וַיּוֹכַח צוּר מוֹשִׁיעוֹ / הֲלֹא אֱלֹהִים יַחֲקָר זֹאת


תָּאַב אֲרַמִּי לְהַצִּיל יְגִיעוֹ

שָׁפַט וַיּוֹכַח צוּר מוֹשִׁיעוֹ

לְעַבְדִי אִם תָּרַע וְתִפְצַר לְהַרְשִׁיעוֹ / וְרָעָה לְךָ זֹאת


מֵאִישׁ דָּמִים לִקְרָאתוֹ רָץ

הִרְעִיד וְשִׁוַּע בְּפֶלֶל נִמְרָץ

בְּחֶסֶד מְשַׁכְתַּנִי אֵל נַעֲרָץ / וּנְקַלּוֹתִי עוֹד מִזֹּאת


רְעָצַנִי אָח וְקַרְנִי גוֹדֵעַ

בְּשִׁמְךָ בָטַחְתִּי פּוֹדֶה וּפוֹדֵעַ

יָגֹרְתִּי חֵטְא כִּי מִי יוֹדֵעַ / בַּמָּה הָיְתָה הַחַטָּאת הַזֹּאת


יְלָדָיו חָצָה מִפֹּה וּמִפֹּה

הִקְרִיב שׂוֹטֵם כְּאַרְיֵה עַל טַרְפּוֹ

וָתִיק עָמַד לְהָשִׁיב אַפּוֹ / וְלֹא שָׁת לִבּוֹ גַּם לָזֹאת


דָּרוֹשׁ תִּדְרֹשׁ עֲבוּרוֹ עַם נִלְאֶה

הַיּוֹם תְּחוֹנֵן גִּזְעוֹ הַנִּכְאֶה

חֲסִין הַבֵּט מִשָּׁמַיִם וּרְאֵה / וּפְקֹד גֶּפֶן זֹאת


חָמַלְתָּ עַל אָב חֲמֹל עַל בֵּן

זַעֲקוֹ תַאֲזִין הֲגִיגוֹ תָבֵן

קִשְׁרֵי כְתָמָיו תְבָרֵר וְתַלְבֵּן / לְעֵינֵי הַשֶּׁמֶשׁ הַזֹּאת


וְכַאֲשֶׁר חִלַּצְתָּ אִישׁ תָּם מִנִּקְדָּם

זֵד אֱדוֹם בְּצֵאתוֹ עָלָיו לִשְׁפֹּךְ דָּם

קְהָלְךָ תְחַלֵּץ מֵעָווֹן הָאֲדַמְדָּם / וְתִסְלַח לְמִי זֹאת בְּקָרְאָהּ בְּזֹאת.
