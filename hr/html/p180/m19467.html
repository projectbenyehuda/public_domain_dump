[אֲשֶׁר תֵּשֵׁב שְׁכוּלָה כַּנְּהָרוֹת]

ומן מטאולאת קצאידה ואג͘זאלה קצידה ירת͘י פיהא בעץ͘ אלמפקודין [ומשירותיו הארוכות והנמלצות ביותר שירה שבה קונן על אחד הנפטרים]


אֲשֶׁר תֵּשֵׁב שְׁכוּלָה כַּנְּהָרוֹת / דְּמָעוֹת עַל לְחָיֶיהָ פְזוּרוֹת

וְתֵאָנַח וְיָד תָּשִׂים לְלֶחְיָהּ / וְהִיא מָרָה כְּהָמֵר עַל בְּכֹרוֹת

וּפָנֶיהָ יְדַעְתִּים כַּמְּאוֹרוֹת / וְנֶהֶפְכוּ בְּבִכְיָתָה שְׁחֹרוֹת

לְמַה תִּבְכִּי כְּלִילַת הַנְּעָרוֹת / וְלֹא מִטַּל עֲנָנַיִךְ עֲצוּרוֹת

עֲלֵי פֵרוּד יְדִידַיִךְ כְּאִלּוּ / גְוִיֹּתָם לְפָנַיִךְ קְבוּרוֹת

וּמִדֵּי חַי אֱנוֹשׁ רֵעָיו יְקַוֶּה / וְאֵין מִקְוֶה לְשׁוֹכֵב בְּקְּבָרוֹת

בְּכִי בָכֹה עֲלֵי חַיִּים וּמָוֶת / אֲשֶׁר נָפַל בְּאַחַד הַיְּעָרוֹת

וְאִלּוּ מֵת כְּמוֹת רַבִּים בְּבֵיתָם / וְיִקָּבֵר בְּכָבוֹד וַהֲדָרוֹת

אֲזַי הָיָה לְלִבִּי קָט הֲנָחָה / וְנִחֻמִים וְחֹבֶשׁ לַשְּׁבוּרוֹת

וְלֹא יָמוּת מְטֹעָן בַּחֲרָבוֹת / וּבִמְרוּצוֹת אֲשֶׁר כָּאֵשׁ בְּעוּרוֹת

וְלֹא תֵעָשׂ גְּוִיָּתוֹ כְּבָרָה / וְיִדָּקֵר בְּכָל גּוּפוֹ דְּקָרוֹת

אֲשֶׁר נִשְׁלַח הֱיוֹת עֶזְרָה בְצָרוֹת / וְהָלַך לַהֲדֹף חֵילֵי מְצוּרוֹת

טְרָפוּהוּ הֲדָפוּהוּ לְבָאִים / וְהָארָיוֹת וְגוֹרֵי הַכְּפִירוֹת

אֲשֶׁר עָמְדוּ וְרָמְחֵיהֶם מְרוּקִים / וְחַרְבוֹתָם שְׁלוּפוֹת מִתְּעָרוֹת

חֲרָבוֹת נֽשְׁכוֹת כָּאֵשׁ תְּלַהֵט / תְּגֹרֵר הָאֲבָנִים כַּמְּגֵרוֹת

אֲדֹמִים שָֽׁפְכוּ דַם אִישׁ נְקִי כָף / וְנַפְשׁוֹ הָֽרְגוּ מֵהַבְּחוּרוֹת

וְהֶאָח אָֽמְרוּ עָלָיו בְּהָרְגוֹ / וְלִבִּי סָר וְעוֹפֵף בַּאֲוִירוֹת

שְׁחָטוּהוּ כְּאַחַת הָעֲגָלוֹת / וְגַם הוּבַל אֱלֵי טֶבַח כְּפָרוֹת

וְהִתְגֹּלֵל בְּדָמוֹ בַֽמְסִלָּה / וְהִרְתִּיחַ כְּבָשָׂר בַּקְּדֵרוֹת

לְדָמוֹ רָעֲשָׁה אֶרֶץ בְּשָׁפְכוֹ / גְּבָעוֹת רָֽגְזוּ עָלָיו וְהָרוֹת

וְלוּ רָאָה וְיָדַע מַחְשְׁבֹתָם / אֲזַי בָּא מִפְּנֵיהֶם בַּמְּעָרוֹת

אֲבָל עֵת־בֹּא פְקֻדַּת אִישׁ וְיוֹמוֹ / הֲיִמָּלֵט וְאִם יִמְצָא עֲזָרוֹת

וְאֵין שִׁלְטוֹן בְּיוֹם מָוֶת לְאָדָם / וְלֹא יַעְבֹר אֱנוֹשׁ מִיּוֹם עֲבָרוֹת

אֲהָהּ אֵיךְ הֻתְּשָׁה חָכְמָה כְּחוֹמָה / אֲשֶׁר הָיְתָה בְצוּרַת הַבְּצוּרוֹת

וְחוֹמַת־אֵשׁ סְבִיב כָּל הַגְּבִירוֹת / וּמִגְדַּל עֹז עֲלֵי כָל הָעֲיָרוֹת

וְהַר תָּבוֹר מְחַזֵּק לֵב חֲשׁוּלִים / וְלִצְמֵאִים בְּצַחְצָחוֹת מְקוֹרוֹת

וְצוּף אֹהֵב וְרוֹשׁ אוֹיֵב אֲבָל כִּי / בְּכָל כֹּחוֹ אֱמֶת הִמְתִּיק מְרֹרוֹת

תְּמִים לֵבָב וּבַר מִכָּל נְבַר לֵב / אֲשֶׁר אָחַז אֲשׁוּרוֹ בַּֽיְשָׁרוֹת

לְמַעַן יֹֽדְעָיו חָלוּ לְשִׁמְעוֹ / וְאָחְזוּ צִיר כְּצִירֵי הַֽמְצֵרוֹת

וְקִינָתָם בְּדִמְעָתָם עֲשׂתָם / בְּחַיָּתָם תְּמוּר יַיִן וְשִׁירוֹת

וְתַחַת מַעֲטֵה כָבוֹד וְהָדָר / חֲגוּרִים מִלְּבוּשׁ שַׂקִּים חֲגֹרוֹת

חֲפוּיֵי רֹאשׁ מְקוֹם הוֹד הָעֲטָרוֹת / וְשִׁית אֵפֶר עֲלֵיהֶם וַעֲפָרוֹת

וְשֶׁבֶת דֹּם בְּלֵב יֵחַם כְּתַנּוּר / מְקוֹמָם מֵעֲלִיּוֹת הַֽמְקֵרוֹת

וְכָל בָּנוֹת עֲדִינוֹת שַׁאֲנַנּוֹת / אֲשֶׁר בָּרֹךְ וּבַתַּפְנוּק זְכוּרוֹת

מְקוֹנְנוֹת הֵן וְאַפֵּיהֶן גְּדוּדוֹת / וְכָל עֶרֶב וְכָל שַׁחַר שְׁחֹרוֹת

וּמַר צָרְחוּ וְרוֹשׁ לַעַן בְּמוֹת רֹאשׁ / לְכָל רָאשִׁים וְשָׂבְעוּ מַמְּרֹרוֹת

וְכוֹס רַעַל תְּמוּרַת הָֽרְעָלוֹת / חֲבשׁ חֶבֶל מְקוֹם שָׁרִים וְשָׁרוֹת

וְהֵן הוֹלְכוֹת וְזוֹ אֶל זוֹ סְמוּכוֹת / כְּיוֹלֵדוֹת בְּצִירִים לֹא עֲצֻרוֹת

וּבֹכוֹת עַל אֲדֹנֵיהֶן אֲבִיהֶן / וְהוֹמִיּוֹת זְקֵנוֹת עִם נְעָרוֹת

וְכֻלָּן נֹֽשְׂאוֹת קוֹלָן בְּחִילָן / כְּאִלּוּ יָֽלְדוּ אֹתוֹ עֲקָרוֹת

וְאֵיכָה לֹא יְבַכּוּ הַמְּאוֹרוֹת / לְאוֹר תֵּבֵל אֲשֶׁר כִּבָּה נְהָרוֹת

וְכִמְעַט קָט וְתִבְכֶּינָה אֲבָנִים / וְכִמְעַט קָט תְּפוּצֶינָה נְהָרוֹת

בְּכוּ עָלָיו קְהִלּוֹת הַֽיְשָׁרוֹת / לְהוֹצִיא מַעְיְנֵיכֶם כַּֽיְאוֹרוֹת

וְשִׂימוּ יוֹם כְּיוֹם גָּלוּת בְּכִיתוֹ / וְיוֹם אֵבֶל יְהִי לָכֶם לְדוֹרוֹת

וּמַהְרוּ נָא וְקִרְאוּ לַחֲכָמוֹת / וְשִׁלְחוּ עוֹד וְהוֹסִיפוּ אֲחֵרוֹת

וְסִפְדוּ לוֹ כְּמִסְפַּד הַֽמְדֻקָּר / מְחַדֵּשֹ בֵּית אֱלֹהִים בַּטְּהָרוֹת

וְכִתְבוּ אֶל פְּלִילֵי דָת גְּאוֹנִים / עֲגֻלּוֹת כַּחֲצִי גֹרֶן וְשׁוּרוֹת

וְשִׁלְחוּ שִׁיר אֲלֵיהֶם חִבְּרוֹ שָׁר / דְּבָרָיו מִבְּדֹלָחִים יְקָרוֹת

אֲשֶׁר בָּחַן אֲמָרוֹת הַצְּרוּפוֹת / וְגַם הִלְבִּין וּבֵרַר הַבְּרוּרוֹת

וְנִשְׁקָלוֹת בְּפֶלֶס מַחֲשָׁבוֹת / וּמֵאָסְפֵי תְעוּדָה הֵן אֲגוּרוֹת

תְּמִימוֹת הֵן כְּמִשְׁפָּטָן וְחֻקָּן / כְּמַתְכֻּנְתָּן וְאֵינֵיהֶן שְׁבוּרוֹת

יְשִׂיחוּהֶן קְצִינֵי הַחֲבוּרוֹת / בְּשִׂיחָתָן כְּחֹק דָּתוֹת וְתוֹרוֹת

אֲשֶׁר יַטּוּ בְחִנָּם הַלְּבָבוֹת / וְכָל נֶפֶשׁ יְשֵׁנָה הֵן מְעִירוֹת

בְּכוֹת עַל אִישׁ אֲשֶׁר הֵסִיר מְשׂוּכָּה / בְּיוֹם סוּרוֹ וּפָרַץ הַגְּדֵרוֹת

וְעַל אֶרֶז אֲשֶׁר הָיָה מְגוּרוֹת / אֲשֶׁר הֶאְרִיךְ סְעַפֹּת לוֹ וּפֹארוֹת

וְכָל עוֹף שָֽׁכְנוּ תַחְתָּיו וְקִנְנוּ / וְשָׂמוּ בֵּין סְעַפֹּתָיו מְדוּרוֹת

כְּזַיִת רַעֲנָן הָיָה וְנִכְרַת / בְּיַד אֹיֵב וְשָׁרָשָׁיו עֲקוּרוֹת

יְעַקֵּר עֹֽקְרָיו הָאֵל מְעַקֵּר / חֲמֵשֶׁת הַמְּלָכִים בַּמְּעָרוֹת

נְתָשׁוּהוּ נְתָצוּהוּ סְעָרוֹת / אֲחָזוּנוּ לְשֹׁד שִׁבְרוֹ שְׂעָרוֹת

לְמַעְנוֹ צָלֲלוּ אָזְנוֹת סְגוּרוֹת / וְעָלָיו רָעֲשׁוּ בָתִּים וְקֹרוֹת

אֲסָפוּ אֵל וְרוּחוֹ הַטְּהוֹרָה / וְשָׂמוֹ בַנְּפָשׁוֹת הַטְּהֹרוֹת

וְנֵרוֹ יַעֲלֶה תָמִיד כְּמוֹ אוֹר / עֲלוֹת שֶׁמֶשׁ עֲלֵי כָל הַמְּאֹרוֹת.
