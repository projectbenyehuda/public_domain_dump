וָאָרֶץ אַשְׁפִּיל

וָאָרֶץ אַשְׁפִּיל וְאַפִּיל תְּחִנָּתִי

אֶקֹּד וְאֶשְׁקֹד בְּזִמְרַת רִנָּתִי

אָעִירָה שַׁחַר וְאֶזְכְּרָה נְגִינָתִי / אוֹדְךָ אֲדֹנָי אֱלֹהַי


אֱלֹהַי בְּקָרְאִי סְעִפַּי נִשְׁפָּלוּ

בְּזָכְרִי אֲשָׁמַי יְגוֹנַי נִכְפָּלוּ

בְּנָשְׂאִי יָדַי פָּנַי נָפָלוּ / בֹּשְׁתִּי וְנִכְלַמְתִּי לְהָרִים אֱלֹהַי


אֱלֹהַי גָדַלְתָּ בְּשַׁפְרִיר זְבוּלֶךָ

גּוּשׁ עָפָר הֲיַעֲרֹךְ אֵלֶיךָ

גָמְלֵנִי חֶסֶד בְּבִטְחִי עָלֶיךָ / גַּם זֹאת זָכְרָה לִי אֱלֹהַי


אֱלֹהַי דְּרַשְׁתִּיךָ יָרֵא וְזָוֵעַ

דַּל אַל תִּבְזֶה בְּעָמְדוֹ מְשַׁוֵּעַ

דַּלּוּ עֵינַי כְּעָנִי וְגוֹוֵעַ / הַבִּיטָה עֲנֵנִי יְיָ אֱלֹהַי


אֱלֹהַי הַיּוֹם פָּנֶיךָ שִׁחַרְתִּי

הֲרִיצוֹתִי פַעַם וְלֹא אֵחַרְתִּי

הֱיוֹתִי חוֹסֶה בְצִלְּךָ בָּחַרְתִּי / הִסְתּוֹפֵף בְּבֵית אֱלֹהַי


אֱלֹהַי וְאַל תַּחֲקֹר אוֹתִי וְאַל תְּנַסֶּה

וּבְמֵי דְמָעַי פְּשָׁעַי הַמְסֵה

וּלְקוֹל שַׁוְעִי הַקְשִׁיבָה וַעֲשֵׂה / וְאַל תְּאַחַר לְמַעַנְךָ אֱלֹהַי


אֱלֹהַי זֻלְעַפְתִּי עֵת כִּי דְרַשְׁתִּיךָ

זַעְתִּי וְנַעְתִּי עֵת כִּי שְׁחַרְתִּיךָ

זָחַלְתִּי וְחַלְתִּי יוֹם זְכַרְתִּיךָ / זָכְרָה לִי אֱלֹהַי


אֱלֹהַי חָמַלְתָּ עַל אֹם נֶאֱנַחַת

חַקּוֹתָ יוֹם כֹּפֶר לְטַהֵר נֶאֱלַחַת

חִלַּצְתָּ נַפְשִׁי וַתַּעַל מִשַּׁחַת / חַיַּי יְיָ אֱלֹהָי


אֱלֹהַי טִכַּסְתָּ עַם זוּ לְהַבְדִּילוֹ

טֹהַר וְצֹהַר יוֹם זֶה לְהַנְחִילוֹ

טוּבְךָ לְהוֹדוֹת לֹא דוֹמַמְתִּי וְלֹא / יִדֹּם יְיָ אֱלֹהָי


אֱלֹהַי יָגֹרְתִּי מֵרֹב זְדוֹנוֹתַי

יָגוֹן נוֹסַפְתִּי עַל יְגוֹנוֹתַי

יָשׁוּב יְרַחֵם יִכְבֹּשׁ עֲווֹנוֹתַי / יִמְאָסֵם אֱלֹהָי


אֱלֹהַי כַּפֵר עָווֹן יִרְטֵנִי

כָּלָה אַל תַּעַשׂ פֶּן תַּמְעִיטֵנִי

כְּבוֹאֲךָ עִמִּי בַּמִּשְׁפָּט שָׁפְטֵנִי / כְּצִדְקְךָ יְיָ אֱלֹהָי


אֱלֹהַי לְשִׁמְךָ פְּצֵנִי פְּדֵנִי

לְהַלְבִּין כֶּתֶם עָקֵב יְגוּדֵנִי

לֵב טָהוֹר בְּרָא לִי אֱלֹהִים לַמְּדֵנִי / לַעֲשׂוֹת רְצוֹנֶךָ כִּי אַתָּה אֱלֹהָי


אֱלֹהַי מַלֵּט סְגֻלָּה מֵרֶשֶׁף

מִי זֹאת הַנִּשְׁקָפָה כַּשַּׁחַר בַּנֶּשֶׁף

מְשַׁכְתָּהּ אַחֲרֶיךָ מִיַּד מְתֵי אֶשֶׁף / מִקֶּדֶם יְיָ אֱלֹהָי


אֱלֹהַי נִגְרַשְׁתִּי מִדְּבִיר וּמִלֶּשֶׁךְ

נוֹדֵד וּבוֹדֵד בְּאָהֳלֵי מֶשֶׁךְ

נַמְתִּי כִּי אֵשֵׁב בַּחֹשֶׁךְ / נֵרִי יְיָ אֱלֹהָי


אֱלֹהַי סְבָבוּנִי נוֹשְׁכִים בְּלִי לָחַשׁ

סָבְרוּ לְהַכְחִישׁ דָּתִי בְכָחַשׁ

סִפְּרָה לְשׁוֹנִי וַתִּרְחַשׁ רָחַשׁ / סוּרוּ מִמֶּנִּי מְרֵעִים וְאֶצְּרָה מִצְווֹת אֱלֹהָי


אֱלֹהַי עַד מֶה יַעֲלֹז רָשָׁע

עֵינַי הָאֵר וְעֵינָיו הָשַׁע

עַל עָבְדוֹ שָׁוְא וּתְרָפִים הָשַׁע / עַבְדְּךָ אַתָּה אֱלֹהָי


אֱלֹהַי פְּקַדְתִּיךָ בְעֵת צַר מִתְגַּבֵּר

פָּרַשְׂתִּי כַף נֶגְדְּךָ וָאֲשַׂבֵּר

פָּעַרְתִּי פִי לַאֲשֶׁר דִבֶּר / פִּי יְיָ אֱלֹהָי


אֱלֹהַי צַַמֵּת זָדוֹן מִתְקַשֵּׁר

צֶדֶק תְּאַשֵּׁר וְעָקֹב תְּיַשֵּׁר

צְבָאֲךָ קִדֵּשׁ צוֹם כַּאֲשֶׁר / צִוַּנִי יְיָ אֱלֹהָי


אֱלֹהַי קָדַרְתִּי שְׁחוֹחַ וְנִחַמְתִּי

קָדוֹשׁ בְּיוֹם צַמְתִּי שִׂבְרִי בְּךָ שַׂמְתִּי

קַחַת דְּבָרִים קִדַּמְתִּי וְקַמְתִּי / קוּמָה יְיָ הוֹשִׁיעֵנִי אֱלֹהָי


אֱלֹהַי רְדָפוּנִי נְגָעַי לִרְגָעַי

רַבּוּ פְצָעַי עַל רֹב בְּצָעַי

רָם אִם עָצְמוּ וְרַבּוּ פְּשָׁעַי / רַבּוֹת עָשִׂיתָ אַתָּה יְיָ אֱלֹהָי


אֱלֹהַי שַׁוְעַת עַמְּךָ תַעֲרֹב

שְׁבִיבָם יִגַּהּ וְצִדְקָם לָרֹב

שַׁדַּי יִשְׁלַח לָרַָחוֹק וְלַקָּרוֹב / שָׁלוֹם אָמַר אֱלֹהָי


אֱלֹהַי תַּאֲוַת עֲנָוִים לְפָנֶיךָ

תִּשָּׂא עָלֵימוֹ אוֹר פָּנֶיךָ

תָֻּכִין לִבָּם תַּקְשִׁיב אָזְנֶךָ / תַּעֲנֶה יְיָ אֱלֹהָי.
