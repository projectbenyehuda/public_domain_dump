הֲתִלְעַג לֶאֱנוֹשׁ

הֲתִלְעַג לֶאֱנוֹשׁ יָחִיד בְּדוֹרוֹת / וְתִבְזֶה אִישׁ אֲשֶׁר לָבַשׁ יְקָרוֹת

אֲשֶׁר הַתֹּם וְהַיֹּשֶׁר לְבוּשׁוֹ / וּבָחַר בָּאֱמוּנָה מִבְּחוּרוֹת

אֲשֶׁר לִבּוֹ כְּחוֹל יִכְבַּד בְּיוֹם בּוּז / וְיוֹם שָׁאוֹן וְיוֹם עֶבְרָה וְצָרוֹת

אֲרָזִים לֹא עֲמָמוּהוּ וְכַפָּיו, / כְּכָל אַנְשֵׁי יְקָר וָשֵׁם, נְעוּרוֹת

בְּלִבּוֹ נִפְלְאוֹת חָכְמָה וּמוּסָר / וְהַדֵּעוֹת בְּנַפְשׁוֹ הֵם סְתוּרוֹת

לְךָ אֹמַר, לְאִישׁ שִׂפְתֵי מְחִתָּה / אֲשֶׁר הֶבֶל וְרוּחַ בָּם צְרוּרוֹת

שְׁמַע מִלַּי וְהַטֵּה אָזְנְךָ לִי / וְקוּם וּבְרַח וְעָבְרָה הָעֲבָרוֹת

בְּטֶרֶם לֹא אֲשִׂימָךְ כָּעֳבָרָה / בְּצִיר יַכְאִיב לְבָבָךְ כַּמְּצֵרוֹת

וְתָחִיל בַּחֲמַת חִצֵּי אֲמָרַי / כְּמַבְכִּירָה עֲצוּרָה בַנְּעוּרוֹת

וְאִם הִנֵּה אֲנִי יָמַי מְעוּטִים / וְשֵׁשׁ עֶשְׂרֵה שְׁנוֹתַי בָּם סְפוּרוֹת -

דְּעֶה כִּי גִדְּלוּ עָבִים נְתִיבַי / וּבַל אֹחֵז, אֲבָל כִּי בַמְּקֵרוֹת

וְהִנֵּה בָם יְבֹרַךְ כָּל חֲכַם לֵב / וְכָל נֶפֶשׁ תְּבַקֵּשׁ הַבְּצוּרוֹת

וְלִי שִׁיר שֶׁיְּפוֹצֵץ צוּר בְּעֻזּוֹ / וְיוֹצִיא מִסְּלָעִים הַיְּאוֹרוֹת

וְדַלְתֵי הַתְּבוּנָה נִפְתְּחוּ לִי / וְהֵם עַל כָּל בְּנֵי עַמִּי סְגוּרוֹת

וְלִי דֶרֶךְ מְסֻקָּל בַּשְּׁחָקִים / וְגַם לִי מַחֲנֶה עִם הַמְּאוֹרוֹת

אֲבָל עוּרָה לְךָ יָשֵׁן וְנִרְדָּם / וְנֶפֶשׁ מִנְּפָשׁוֹת הַשְּׁבוּרוֹת

בְּלִי דַעַת, בְּלִי שֵׂכֶל, אֲבָל כִּי / לְחֶרְפָּה תּוֹךְ גְּוִיָּתָם אֲסוּרוֹת

וְאִם עַל חִכְּךָ מַר הַדְּבַשׁ - קוּם / אֱכֹל וּשְׁתֵה לְשָׂבְעָה מַמְּרוֹרוֹת

וְהָעִוֵּר יְסֻגַּר בַּחֲשֵׁכָה / לְמַעַן מָאֲסוּ עֵינָיו בָּאוֹרוֹת

וְתֵלֵךְ כַּאֲבֶל אֵם וַחֲפוּי רֹאשׁ / וְיוּסַר מֵעֲלֵי רֹאשָׁךְ עֲטָרוֹת

קְרָא שִׁירִי וְאַל תִּדְאַג לְמַעַן / חֲרָפוֹת בּוֹ בְּךָ יִהְיוּ קְשׁוּרוֹת

לְמַעַן כִּי אֲנִי אָשׁוּב וְאֶשְׁכַּח / נְטֹר אֵיבָה וְאַקְשִׁיב הָעֲתָרוֹת

וְלִי מַרְפֵּא לְכָל מַכָּה אֲנוּשָׁה / וּמִדְבָּרִי צְרִי עַל כָּל מְזוֹרוֹת

אֲבָל כִּי עַל מְנָת תָּשׁוּב, וְכָל יוֹם / תְּלַחֵךְ מִנְּעָלַי הָעֲפָרוֹת

וְכֵן אָשׁוּב אֲנִי אָז לָךְ וְיִהְיוּ / אֲמָרַי הַנְּעִימִים לָךְ עֲטָרוֹת

וְאָשִׂים מַעֲלוֹתֶיךָ גְּבוֹהוֹת / עֲלֵי שָׂרִים, וְיִהְיוּ בָם יְתֵרוֹת

וְשָׂרֵי הָאֲלָפִים יִכְרְעוּ לָךְ / וְהַמֵּאוֹת וְשָׂרֵי הַעֲשָׂרוֹת

וְאִמְרוֹתַי אֲשֶׁר עַתָּה חֲקַקְתִּים / יְהוּ בִלְשׁוֹן מְתֵי מוּסָר זְכוּרוֹת

וּבִידֵיהֶם יְהוּ לָהֶם שְׁבָטִים / לְיַסֵּר הַפְּתָאִים בָּם לְדוֹרוֹת

וְחִבַּרְתִּי בְךָ אֵלֶּה לְיַסֵּר / לְבַב נִמְהָר בְּאֵלֶּה הַהֲדוּרוֹת

לְעֵת תָּבוֹא בְמִלְחָמוֹת תְּשַׁנֵּס / חֲלָצֶיךָ וְתַחְגֹּר עֹז חֲגוֹרוֹת

וְעוֹד אָשִׁיר לְבַד אֵלֶּה בְּשִׁמְךָ / וְאֶשְׁלַח אֶל קְצֵה תֵבֵל אֲחֵרוֹת

וְאִם בִּלְבָבְךָ שֶׁמֶץ תְּבוּנָה / הֲלֹא תֵדַע יְקָר שָׁרִים וְשָׁרוֹת

וְאִם אַתָּה חֲסַר שֵׂכֶל וּבִינָה / וְדֵעוֹת מִלְּבָבֶךָ עֲדוּרוֹת -

בְּעֵינֵי כָל אֱנוֹשׁ יַשְׂכִּיל וְיָבִין / אֲנִי פָטוּר וְשִׁירוֹתַי פְּטוּרוֹת.
