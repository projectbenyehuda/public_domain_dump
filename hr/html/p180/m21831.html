קוּם הַזְּמַן

ולו מהלל את רבנו שמואל

קצידה לשלמה הקטן פי אלנגיד (שירה לשלמה הקטן על הנגיד)

קוּם הַזְּמַן, וּלְבַשׁ עֲדָנֶיךָ / וּקְשֹׁר שְׁבִיסֵי שַׂהֲרוֹנֶיךָ!

עַד אָן תְּנַצֵּל עֶדְיְךָ, עַד אָן / תִּפְשֹׁט לְפָנֵינוּ סְדִינֶיךָ?

אִם אֵין לְךָ נֵזֶר לְהִתְנוֹסֵס, / כִּי חֻלְּלוּ נִזְרֵי צְפוּנֶיךָ –

מֵעִם שְׁמוּאֵל קַח רְבִידֵי פַז, / יִהְיוּ לְךָ עַל צַוְרוֹנֶיךָ

יָצִיץ בְּךָ שִׁירוֹ כְּמוֹ אֵטוּן, / קוּם וַעֲלֹץ וּמְשֹׁךְ אֲטוּנֶךָ

דָּרַךְ בְּךָ כּוֹכָב, צְלַח וּרְכַב, / כִּי בָא אֲשֶׁר יָאִיר אֱשׁוּנֶךָ

גָּבַהּ בְּרוֹשׁ תִּדְהָר בְּבֵית יִצְהָר / עַל נַעֲצוּצֵי קִמְּשׁוֹנֶיךָ

צָמַח לְךָ חֹרֶשׁ מְאֹד מֵצַל – / פַּנֵּה לְפָנָיו קִיקְיוֹנֶךָ

נִקְרָה לְךָ הַיּוֹם יְאוֹר אֵיתָן - / רַוֵּה בְּמֵימָיו צִמְאוֹנֶךָ

עוּרָה, לְבָבִי, מַה לְּךָ נִרְדָּם, / עוּרָה וְהָקִיץ רַעֲיוֹנֶיךָ!

כַּמָּה גְעַרְתִּיךָ לְבַל תּוֹסִיף / דַּעַת, לְבַל יִרְבּוּ יְגוֹנֶיךָ

הָסֵר תְּלֻנָּתְךָ וְאַל תִּזְכֹּר / הַיּוֹם לְפָנַי עִצְּבוֹנֶיךָ

הִכּוֹן לְךָ, כִּי יֶשׁ לְךָ הַיּוֹם / רִיב עִם בְּנֵי לֵוִי צְפוּנֶיךָ

הַקְרֵב לְעֻמָּתָם שְׂעִיר חַטָּאת – / אוּלַי יְכֻפַּר בּוֹ עֲווֹנֶךָ

אָנָּא, כְּרוּב מִמְשַׁח, לְאַט עָלַי, / עַד אֶעֱרֹךְ מִשְׁפָּט לְפָנֶיךָ!

מַה סּוֹד מְשׁוּגָתִי וּמַה פִּשְׁעִי, / כִּי תַעֲבִיר עָלַי חֲרוֹנֶיךָ?

תִּכְתֹּב לְךָ סֵפֶר כְּאִישׁ מָדוֹן / גַּם תַּאֲרִיךְ עָלַי לְשׁוֹנֶךָ

תַּרְעֵם בְּקוֹל שָׁאוֹן אֲשֶׁר לֹא כֵן, / לִהְיוֹת סְאוֹן רַעַשׁ סְאוֹנֶךָ

אַל תַּחֲשֹׁב כִּי צָלְלוּ אָזְנַי / יוֹם הַעֲלוֹת בָּם שַׁאֲנָנֶךָ:

לוּ רָצְתָה נַפְשִׁי לְהִתְבָּאֵשׁ - / אָז שָׁחֲקוּ מֵימֵי אֲבָנֶיךָ

הִנֵּה בְּחַר מִשְׁפָּט וְדַע מַה טּוֹב, / דּוּן, כִּי אֲנִי חָפֵץ בְּדִינֶךָ

עַל מַה תְּרִיבֵנִי, פְּלִיל צֶדֶק, / כִּי תַאֲרִיךְ בִּי קִלְּשׁוֹנֶךָ?

כִּמְעַט חֲשַׁבְתִּיךָ לְמִגְדַּל עֹז - / עַד נָפְלוּ עָלַי אֲבָנֶיךָ

כִּמְעַט קְרָאתִיךָ זְמוֹרַת זָר – / עַד נְֶהְפְּכוּ לִי נַעֲמָנֶיךָ

תִּצְמָא לְךָ נַפְשִׁי – וְתַשְׁקֵנִי / תַּחַת צְמָאִי רֹאשׁ פְּתָנֶיךָ

אֶזְרַע לְפִי חֶסֶד - וְתָשִׂים אֶת / רֵאשִׁית תְּבוּאָתִי בְּגָרְנֶךָ

מַה זֶּה, גְּבִיר, מַה זֶּה וְעַל מַה זֶּה / תִּצְפֹּן כְּמוֹ אֵלֶּה לְבָנֶיךָ?

תָּרוּץ בְּלִי עָווֹן וְתִכּוֹנֵן / לִקְרַאת שְׁלוּמֶיךָ אֱמוּנֶיךָ

תִּגְאֶה וְכַשַּׁחַל תְּצוּדֵנִי / עַד כִּי תְכַלֶּה בִּי שְׁנוּנֶיךָ?

אִם הָאֱלֹהִים הוּא הֱסִיתְךָ בִּי – / יָרַח קְטֹרֶת עַד רְצוֹנֶךָ

אוֹ אִם בְּנֵי אָדָם - אֲרוּרִים הֵם / כִּי גֵרְשׁוּנִי מִלְּפָנֶיךָ

יַכְרֵת אֱלֹהִים דּוֹבְרֵי כָזָב / הַקּוֹרְצִים בֵּינִי וּבֵינֶךָ

כִּי פָרְשׂוּ רֶשֶׁת לְיַד מַעְגָּל / וַיְבָעֲרוּ עָלַי עֲשָׁנֶיךָ

זָדוּ כְּמוֹ צִיבָא בְּבֶן שָׁאוּל / וַיְדַבְּרוּ שֶׁקֶר בְּאָזְנֶיךָ.

מִי זֶה אֲשֶׁר יָרִים בְּךָ יָדוֹ, / אֵי זֶה זְבוּב יַבְאִישׁ שְׁמָנֶיךָ?

מֵרִים בְּךָ יָדוֹ - כְּמֵרִים יָד / בִּשְׁנֵי בְנֵי עַמְרָם קְצִינֶיךָ.

הַשִּׁיר הֲלֹא לָכֶם, בְּנֵי לֵוִי, / וּלְךָ לְבַד מִתַּחְכְּמוֹנֶיךָ!

אִם עָמְדוּ שָׁרִים בְּמִשְׁמַרְתָּם - / הָיוּ נְגִידֵיהֶם סְגָנֶיךָ

יִשְׁכַּל אֲבִי נָבָל אֲשֶׁר יַחְשֹׁב / לַעְקֹר כְּצִיץ נוֹבֵל אֳרָנֶיךָ!

אֶשְׁלֹף לְךָ חַרְבִּי וְאֶגְעַר בּוֹ: / סוּר, פֶּן תְּגַלֶּה יַד קְלוֹנֶךָ

תִּפַּח בְּאֵשׁ פֶּחָם וְגַם תִּזְרַע / בִּמְקוֹם חֲרֵרִים זֵרְעוֹנֶיךָ

תִּיגַע וְלֹא תוּכַל עֲלוֹת אֵלָיו / עַד תֶּאֱסֹף רוּחַ בְּחָפְנֶיךָ

נָפַל לְךָ חֶבֶל בְּסוֹד הֵימָן / וּבְסוֹד בְּנֵי אָסָף מְבִינֶיךָ

הִנֵּה לְךָ עָלָיו שְׁנֵי עֵדִים / גַּם עוֹד שְׁלֹשָׁה נֶאֱמָנֶיךָ:

הַתֹּם וְהַיֹּשֶׁר וְהַבִּינָה - / זֶה מִלְּבַד שֵׁבֶט אֱמוּנֶיךָ

קוּם וַעֲטֵה, הַשַּׂר, מְעִיל כָּבוֹד - / הִנֵּה אֲמָרַי פַּעֲמוֹנֶיךָ

הַעַל תְּהִלָּתִי אֲשֶׁר שָׂמָה / רָבִיד מְקֻשָּׁר עַל גְּרוֹנֶךָ!

אַחַת נְשָׂאתִיהָ כְּמוֹ תֹרֶן / לָשֵׂאת בְּרֹאשׁ כָּל הַר תְּרָנֶיךָ

קוּם וַעֲדֵה גָאוֹן וְהִנָּשֵׂא, / כִּי עִמְּךָ הִנֵּה בְּמֹאזְנֶיךָ

חֶבְרוֹן וְעֻזִּיאֵל וּמִישָׁאֵל / וּשְׁאָר בְּנֵי לֵוִי זְקֵנֶיךָ

הַנּוֹשְׂאִים אָרוֹן עֲלֵי כָתֵף, / הַשּׁוֹמְרִים מִקְדַּשׁ אֲדוֹנֶיךָ

תָּרוּם בְּךָ קֶרֶן סְפָרַד עוֹד / מִיּוֹם הֱיוֹתָךְ עַד זְקוּנֶיךָ

תַּרְחִיב בְּךָ פֶה עַל בְּנֵי שִׁנְעָר, / הַשּׁוֹאֲבִים מִמַּעֲיָנֶיךָ

מַה תִּגְבְּהִי, בָּבֶל, וּמַה תִּגְאִי, / שִׁנְעָר, בְּמָעֻזֵּי גְאוֹנֶיךָ?

דַּע, כִּי בְאַרְצֵנוּ חֲבַצֶּלֶת – / לֹא פָרְחָה בִנְאוֹת שְׁרוֹנֶיךָ

שָׂרֵי פְלִילִיָּה לְךָ עֵד, כִּי / פֶה פָעֲרוּ נֶגֶד עֲדָנֶיךָ.

אַל תַּחְשְׁבֵנִי כִּי בְרֹב תִּקְוָה / אַפִּיל תְּחִנָּתִי לְפָנֶיךָ

אוֹ מֵאֲשֶׁר יִירָאֲךָ לִבִּי – / לוּ כוֹכְבֵי שַׁחַק מְעוֹנֶךָ.

הָאֵל יְהִי עִמָּךְ, וְאַל יַעְמֹד / שָׂטָן לְעוֹלָם עַל יְמִינֶךָ

וּמְצָא בְעֵינָיו חֵן וְשֵׂכֶל טוֹב / וּרְאֵה בְנֵי בָנִים לְבָנֶיךָ!
