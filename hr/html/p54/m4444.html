נגהות

שַׁחַר


בָּקַע מִקַּצְוֵי הַמִּזְרָח שִׁיר רְנָנִים,

שִׁיר נֹגָהּ וָזִיו –

נִתְּזוּ רְשָׁפִים מִכַּנְפֵי עֲנָנִים.


נָשְׁרוּ זַרְזִיפִים אֶל חַכְלִיל -הַשּׁוֹשָׁן,

וַיִלְהַט וַיִיף

בְּזֹהַר שֶׁל בְּרֵאשִׁית וּבְאֵשׁ רַעֲנָן -נוֹשָׁן.


וַיִּסַּע רְוֵה -נְגֹהוֹת הֶעָנָן הָרִאשׁוֹן

לְמֶרְחַב הַגִּיל,

וַיַצֶּת אֵשׁ -פְּלָדוֹת בְּנִבְכֵי כָל -אִישׁוֹן.


וּבְכָל תְּהוֹם עֲמֻקָּה וּבְכָל נֶפֶשׁ גּוֹלָה

אָז רִטֵּט הֵד -צְלִיל –

תְּפִלַת גְּאֻלָּה לַשֶּׁמֶש הַעוֹלָה.


בֹּקֶר


יֵשׁ אֶרֶץ רְחוֹקָה, שָׁם כַּרְמִי שָׁתַלְתִּי,

דָּרַכְתִּי יְקָבַי,

וְאַשְׁכֹּלוֹת עֲנָבַי

עִם כָּל -בֹּקֶר וָבֹקֶר לְבַדִי עוֹלַלְתִּי.


יֵשׁ הָרִים רְחוֹקִים בְּעַרְפִלֵי בְקָרִים,

בְּמֶרְחָב חַכְלִילִי,

שָׁם אֶחְצֹב בַּרְזִלִּי

וְאֶחְפֹּשׂ צְמֵא -אֹשֶׁר מַטְמֻנֵי -מִסְתָּרִים.


עִיר יֶשְׁנָהּ רְחוֹקָה עִם לֹבֶן מִגְדָּלִים

וּמִקְדַּש -אֲרָזִים

וְרַעַד שֶל רָזִים –

שָׁם אֶלְחַשׁ תְּפִלָּתִי בִשְׂדֵרוֹת -הַצְּלָלִים.


יֵשׁ שְדוֹת -קְרָב רְחוֹקִים, שָׁם אֵצֵא בַצּוֹבְאִים

וְיִלְהַט דַּם חִשְקִי,

וְאֶלְטֹשׁ אֶת נִשְׁקִי

בִּזְהַב זֹהַר בֹּקֶר וּדְמֵי -אֵשׁ הָאוֹיְבִים.


צָהֳרַיִם


עַל שָפְרִיר הַתְּכֵלֶת בָּרוֹם

שָׁט שֶׁמֶשׁ -רַהַב –

רְוֵה בֶשֶׂם וָלַהַב

עַל -פְּנֵי שָׂדוֹת לוֹ יָשֵן הַיוֹם.


וְחוֹלֵם חֲלוֹם שַׁחֲרֵי -עַד

בַּמִזְרָח הַלוֹהֵט –

וְלֹא יִרְאֶה, מִי צוֹעֵד

בַּמַּעֲרָב וּצְלָלִים שָׁם יַט.


הָּה יוֹמִי! אַל תִּירָא לֵיל -סוֹד,

עוֹד טֶרֶם נְטוֹת צְלָלִים

עַל אַרְגְּמַן -עָלִים

בַּחֲלוֹמְךָ דֹּם תִּגְוַע בִּיקוֹד.


עֶרֶב


עֶדְנָה עֲנֻגָּה נִתֶּכֶת בַּכֹּל

וְרֹךְ וַחֲלוֹם –

יָרְדָה לָעוֹלָם שְׁעַת קֶסֶם וּכְחוֹל.


נֶעְלַה הַיַּעַר בְּמֶרְחַק -הָאֵד,

שֶׁהָרָה הַחֹם –

מִכִּנּוֹר -מֶרְחַקִּים הִתְמַלֵּט צְלִיל-הֵד:


הֵד שִׁירָה אַדִּירָה, שֶׁגְּרָפוֹ כָל-גָּל,

הִסְתִּירוֹ כָל צֵל,

וּסְפָגוֹ בִדְמָמָה כָּל -אֵגֶל שֶׁל טָל.


וִיְבַשֵׂר הָרוּחַ בֵּין תַּלְמֵי הַנִּיר

דְּמִי עֶצֶב וָלֵיל –

וַתֵּכֶל אָז נַפְשִׁי לִתְפִלָּה, לְשִׁיר…


לַיְלָה


כְּבָר יָרַד הַלַּיְלָה. מִצִּלְלֵי הֶהָרִים

מִיַרְכְּתֵי הַסְּבָךְ

דּוּמָם לוֹ גָח…

בְּנַפְשִׁי – פְּלֵטַת זִיו -שְׁחָרִים.


מָה אֶעֱשֶׂה הַלַּיְלָה? – רֹן -בֹּקֶר עוֹד רוֹעֵד.

טַל -נְעוּרִים יְפַךְ

בִּפְרָחַי, בִּסְבַךְ

יְרַק -צַמְרוֹת יַעֲרִי הַלּוֹהֵט.


בְּזִיו -סַהַר אֶרְחַץ, בִּשְבִילָיו אֲטַיְלָה.

אָשוּטָה בַחֲלוֹם –

עַד יָבוֹא הַיוֹם,

עַד יָנוּס וְיִפְנֶה הַלַּיְלָה.
