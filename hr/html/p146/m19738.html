יוֹם אֶזְכְּרָה עַלְמָה יָפָה מְהֻדָּרָה

החתימה: שלם שץ


יוֹם אֶזְכְּרָה עַלְמָה / יָפָה מְהֻדָּרָה

אָבִין בְּכָל חָכְמָה / נִגְלָה וְנִסְתָּרָה

מַרְוָה לְכָל נִצְמָא / תַּמְתִּיק לְמֵי מָרָה

כִּי הִיא מְפֻרְסָמָה / דַּרְכָּהּ מְיֻשָּׁרָה

שִׁפְעָהּ מְקוֹר

חַיִּים וְאוֹר

בֹּשֶׂם וּמוֹר

מִמָּה צְרוֹר

בָּשְׂמִי וְקִנְמוֹנִי

לָהּ יִנְהֲרוּ

עָם נִבְחֲרוּ

יִתְחַבְּרוּ

גַּם יִשְׁמְרוּ

כָּל סוֹד מְכֻוָּנִי

מִמַּחְמְדֵי

כָּל נִכְבְּדֵי

סוֹד לוֹמְדֵי

דָּת, שׁוֹקְדֵי

הֵיכָל וְאַרְמוֹנִי

יִזְרַח כְּמוֹ כִימָה / שִׂכְלָם וְיָאִירָה

אִם יִמְצְאוּ סִימָה / צַחָה וְגַם בָּרָה:


שֶׁפַע וְרֹב שַׁלְוָה / לִכְלַל בְּנֵי יֶשַׁע

שַׁלְוָה וְגַם תִּקְוָה / מֵאֵין דְּבַר פֶּשַׁע

רִנָּה וְקוֹל חֶדְוָה / נֶרֶב לְהִשְׁתַּעְשַׁע

מַה טּוֹב וּמַה נָּעִים / לָעָם אֲשֶׁר נוֹשַׁע

שִׂמְחוּ בְּנֵי

תָּם, שׁוֹכְנֵי

תּוֹךְ מַחֲנֵה

כָּל סוֹגְנֵי

חָכְמָה וְתוּשִׁיָּה

לִמְדוּ דְּבַר

תּוֹרָה בְּבַר

לֵבָב, חֲבַר

עִם כָּל סְבַר

פָּנִים וְטוֹב מִחְיָה

מוּסָר קְנוּ

סוֹד תִּצְפְּנוּ

הִתְבּוֹנְנוּ

אַל תִּישְׁנוּ

אָז תִּמְצְאוּ רַעְיָה

מַלְכָּה מְאֹד תַּמָּה / דּוֹדָהּ שְׁמָהּ קָרָא

תַּבְהִיל פְּנֵי חַמָּה / הַדְרַת בְּנֵי שָׂרָה:


לִי תַעֲרֹךְ שֻלְחָן / וּלְכָל בְּנֵי חֶלֶד

הַחֵן בְּפִתְחָהּ חָן / תָּמִיד וְלֹא יֵרֵד

בֵּין זֶה לְזֶה תִבְחָן / צַדִּיק וְאִישׁ מֶרֶד

תִּרְצֶה לְהוֹכִיחָן / תַּשְׁפִּיעַ כְּמוֹ חוֹמֶד

שָׁם עָרְכָה

שֻׁלְחָן, וְכָ־

בוֹד תִּמְשְׁכָה

וּתְנַסְּכָה

נִסְכָּהּ עֲלֵי הַכֹּל

לָהּ עוֹרְכוֹת

הַלִּשְׁכוֹת

הַנִּמְשְׁכוֹת

גַּם סוֹכְכוֹת

תֵּשַׁע בְּרֹב הַשְׂכֵּל

מֵימָהּ מְקוֹר

סוֹד, חֹם וְקֹר

הוּא יַחֲקוֹר

לֹא יַעֲקוֹר

לְכָל בְּנֵי כַלְכֹּל

נוֹבֵעַ כְּמוֹ יַמָּה / בָּרוּךְ אֲשֶׁר בָּרָא

בִּשְׁבָח וּבִנְעִימָה / לוֹ אֶעֱנֶה שִׁירָה:


מַה טּוֹב וּמַה נָּעִים / חֶבְרַת בְּנֵי חֶמְדָּה

נְסִיכִים וְהָרוֹעִים / נָאִים בְּכָל מִדָּה

אֶחָד וְגַם שִׁבְעִים / הֶן סַנְדְּרֵי נֶגְדָּהּ

דָּנִים וּמַשְׁמִיעִים / כָּל סוֹד וְכָל חִידָה

שָׂרִים עֲלֵי

דִּין פּוֹעֲלֵי

כָּל בַּעֲלֵי

דִּין, שׁוֹאֲלֵי

מִשְׁפָּט וְדִין צֶדֶק

לָהֶם יְסוֹד

פֶּשֶׁר, בְּסוֹד

בַּעְלֵי כָבוֹד

כִּי זֶה לְעֵד

לְמְחַזְּקֵי בֶּדֶק

מִתְלַבְּשִׁים

הוֹד חוֹבְשִׁים

מִתְקַדְּשִׁים

וּמְפַשְׁפְּשִׁים

סוֹד רוֹם וְהָעֹמֶק

לָהֶם מְאֹד אֵימָה / בַּדִּין וּבַמּוֹרָא

וְכָל גְּבַהּ קוֹמָה / דַּרְכָּהּ תְּיַשֵּׁרָה:


שׁוּר מִמְּעוֹנֶךָ / צוּרִי וְהַב צַהְלָה

קַבֵּץ לְעַמֶּךָ / יַחַד אֲשֶׁר גָּלָה

וּזְכֹר בְּרִיתֶךָ / רַפֵּא לְלֵב חָלָה

זָכְרָה שְׁבָטֶיךָ / יוֹם דִּגְלְךָ עָלָה

שְׁלַח רְצוּץ

מִנַּעֲצוּץ

מַהֵר וְרוּץ

קַבֵּץ חֲלוּץ

לִנְוֵה מְחוֹז חֶפְצוֹ

לָךְ יִדְרְשׁוּ

כִּי חָלְשׁוּ

וַיְבַקְּשׁוּ

עֵת יִגְּשׁוּ

לַעְלוֹת לְהִמָּצֵא

מַהֵר עֲלוֹת

בֵּין מִקְהֲלוֹת

הַנִּסְגְּלוֹת

יוֹם צַהֲלוֹת

עַמָּךְ לְתוֹךְ אַרְצוֹ

עַל צָר שְׁלַח חֵמָה / יִיגַע בְּיוֹם צָרָה

יִדְאַג וְיָהֹמָּה / בַּבּוֹר אֲשֶׁר כָּרָה:


צַוֵּה לְבִרְכָתָךְ / וּפְדֵה לְיוֹנָתִי

עָם בּוֹחֲרֵי דָתָךְ / זֶרַע אֲיֻמָּתִי

חַדֵּשׁ לְמַלְכוּתָךְ / רַפֵּא מְשׁוּבָתִי

וּבְצֵל שְׁכִינָתָךְ / תַּלְבִּישׁ לְרַעְיָתִי

חֶשְׁקִי בְּךָ

אֶקְרָאֲךָ

הַט אָזְנְךָ

עִם עַמְּךָ

וּשְׁלַח לְבֶן דָּוִד

זָכְרָה גְּמוּל

אַבְרָם סְגוּל

וּזְכוֹר חֲמוּל

נֶעְקָד לְמוּל

קָדְשָׁךְ וְהוּא יָחִיד

קַבֵּץ נְדוּד

עַם הַבְּדוּד

יַחַד צְמוּד

בִּזְכוּת חֲמוּד

אִישׁ תָּם לְךָ יָדִיד

בָּנָיו תְּרַחֵמָה / עָוֹן תְּכַפֵּרָה

נִדְרִי אֲשַׁלֵּמָה / אַקְרִיב וְאַקְטִירָה:
