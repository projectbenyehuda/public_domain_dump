אֵין לָנוּ כֹּהֵן גָּדוֹל

אֵין לָנוּ כֹּהֵן גָּדוֹל / לְכַפֵּר בַּעֲדֵנוּ / וְאֵיךְ יִתְכַּפֵּר לָנוּ / עַל שִׁגְגוֹתֵינוּ

אֵין לָנוּ אוּרִים וְתֻמִּים / לִשְׁאוֹל בָּהֶם / וְאֵיךְ יֵאוֹר לָנוּ / וְאִוּוּיֵינוּ מְאֻפָּל.


בָּטְלָה עֲבוֹדָה / מִבֵּית עֲבוֹדָה / וְאֵיךְ נַעֲבוֹד לְזָךְ / וְזָר מַעֲבִיד בָּנוּ

בָּטְלוּ מִמֶּנּוּ / אִשִּׁים וְאָשָׁם / וְאֵיךְ נַעַשׂ אִשִּׁים / וְאֵשׁ מִזְבֵּחַ כָּבָה.


גִּיל גּוֹרָלוֹת / חָדַל מִמֶּנּוּ / וְאֵיךְ נַעַל בְּגִילָה / וְאָנוּ בַּגּוֹלָה

גּוֹדְרֵי פְרָצוֹת / נִפְרְצוּ בַאֲשָׁמֵינוּ / וְאֵיךְ נִגְדּוֹר פֶּרֶץ / וְאֵין עוֹמֵד בַּפֶּרֶץ.


דִּמְעָה וּמְלֵאָה / דָּלְלוּ מִמֶּנּוּ / וְאֵיךְ נִתְרוֹם דֶּמַע / וְדִמְעָה בְעֵינֵינוּ

דַּם זְרִיקַת זֶבַח / פַּס מִבֵּית זֶבַח / וְאֵיךְ נִזְרוֹק דָּם / וְנִשְׁפַּךְ דָּמֵינוּ.


הַקְטָרַת רֵיחַ סַמִּים / הוּסַר מִמֶּנּוּ / וְאֵיךְ נַקְטִיר רֵיחַ / וְהֻבְאַשׁ רֵיחֵינוּ

הֲטָבַת נֵרוֹת / הֻדְעֲכוּ מֵאִוּוּיֵינוּ / וְאֵיךְ נֵטִיב נֵרוֹת / וְכָבָה נֵרֵנוּ.


וְנִיחֹחַ חָדַל / מִמְּנוּחָתֵינוּ / וְאֵיךְ נָבִיא מִנְחָה / וְאֵין בֵּית מְנוּחָה

וּבִכּוּרִים / חָדְלוּ לְבַכֵּר / וְאֵיךְ נַעַל בְּבִכּוּר / וְאֵין בִּכּוּרָה.


זִבְחֵי כָלִיל / פַּסּוּ מִמִּכְלַל יוֹפִי / וְאֵיךְ נַעַשׂ זֶבַח / וְחָרַב בֵּית זֶבַח

זוֹלֵלָה / זִלְזְלוּהָ זָרִים / וְאֵיךְ נָבוֹא לְנַחֲלָה / וְנַחֲלָתֵינוּ לְזָרִים.


חֲלָבִים / פַּסּוּ מֵהַקְטִירָם / וְאֵיךְ נַקְטִיר חֵלֶב / וְאֵין חֵלֶב חִטָּה

חָדְלוּ מִמֶּנּוּ / שְׂעִירֵי חַטָּאת / וְאֵיךְ נַעַשׂ חַטָּאת / וְגָבְרוּ חַטָּאֵינוּ.


טְבִילַת טָהֳרָה / אֵין בַּפַּרְוָה / וְאֵיךְ נִטְבּוֹל בְּטֹהַר / וּפָסְקוּ טְבִילוֹת

טוּרֵי אֶבֶן / פַּסּוּ מִמִּלּוּאַת אָבֶן / וְאֵיךְ נָסִיר לֵב אֶבֶן / וְיִדּוּ בָנוּ אָבֶן.


יַעַר הַלְּבָנוֹן / הוּשַׁת לְמִרְמָס / וְאֵיךְ תְּלַבֵּב לְדוֹד / כַּלַּת הַלְּבָנוֹן

יָפְיֵנוּ הוּשַׁם / וְאוֹרֵינוּ אֻפַּל / וְאֵיךְ תְּיַפֶּה לְדָגוּל / יְפַת עֵינַיִם.


כֹּפֶר פָּר / פַּס מֵאַפִּרְיוֹן / אֵיךְ נַקְרִיב פָּר / וְאֵין מִי יְכַפֵּר

כַּפָּרוֹתֵינוּ / פַּסּוּ מֵאִוּוּיֵינוּ / וְאֵיךְ נַגִּישׁ שַׁי / וַתֶּמֶס פְּגָרֵינוּ.


לְבוֹנָה לֹא עָשִׂינוּ / בְּגִבְעַת הַלְּבָנוֹן / וְאֵיךְ יְלֻבַּן חֵטְא / יָפָה כַלְּבָנָה

לוֹג שֶׁמֶן פַּס / מִקֶּרֶן בֶּן שָׁמֶן / וְאֵיךְ נצוח שֶׁמֶן / עַל בְּלוּלָה בַשָּׁמֶן.



מָר דְּרוֹר אָפַס / מִמְּקֻטֶּרֶת מוֹר / וְאֵיךְ בְּהַר הַמּוֹר / יָלִין צְרוֹר הַמּוֹר

מַחְתָּה הוּחַתָּה / מִלַּחְתּוֹת גֶּחָלִים / וְאֵיךְ נַעַשׂ מַחְתָּה / וְאָנוּ בִמְחִתָּה.


נֵרְדְּ וְכַרְכּוֹם / נָדַד רֵיחוֹ / וְאֵיךְ נַעַן: נִרְדִּי / נָתַן רֵיחוֹ

נִסְכֵּי שֵׁכָר / פַּסּוּ מֵהַסַּךְ / וְאֵיךְ נְנַסֵּךְ / כְּנִסַּכְתּוֹ מֵרֹאשׁ.


סוֹלֶת סֻלְּפָה / מִסֵּדֶר חַלּוֹת / וְאֵיךְ נְיַפֶּה סוֹלֶת / וְנִלְכַּדְנוּ מִסּוֹלְלוֹת

סַמִּים הוּסַר / מִמְּלוֹא חָפְנַיִם / וְאֵיךְ נַעַשׂ סַמִּים / וְהוּמַסְנוּ בְּמִסִּים.


עוֹלָה עָתְקָה / מִיַּעַר עוֹלוֹת / וְאֵיךְ נַעַשׂ עוֹלוֹת / וְהֻכְּבַּד עֻלֵּינוּ

עֲצֵי עֵרֶךְ / חָדְלוּ מִמִּזְבַּח עֵץ / וְאֵיךְ נִטְהַר בָּעֵץ / וְכָשַׁלְנוּ בָעֵץ.


פָּר וְשָׂעִיר / פַּסּוּ מִלְּהִשָּׂרֵף / וְאֵיךְ נִשְׂרוֹף פָּר / וּמִקְדָּשׁ נִשְׂרַף

פָּרֹכֶת פּוֹרְכָה / וְכַפּוֹרֶת כּוֹרְתָה / וְאֵיךְ נַפְצִיחַ פֶּלֶל / וְאָנוּ גַּם בְּפָרֶךְ.


צִיּוֹן צִיָּה / יְרוּשָׁלַיִם לְעִיִּים / וְאֵיךְ נַצִּיב לָנוּ / נַצִּיב צִיּוּנִים

צִיץ נֵזֶר נֻצַּל / מִמֵּצַח מְרַצֶּה / וְאֵיךְ נֵרָצֶה בְצִיץ / וְחַסְדֵּנוּ כְּצִיץ.


קוֹל מִדַּבֵּר / חָדַל מֵאֹהֶל / וְאֵיךְ נִנְטֶה אֹהֶל / וְשֻׁדַּד אָהֳלֵינוּ

קָרְבָּן חָדַל / מִטְּלֵה קָרְבָּן / וְאֵיךְ נַקְרִיב קָרְבָּן / וְשָׁלֵם בְּחֻרְבָּן.


רְבָבָה הֻרְחָקָה / רְבוּכָה הֻקָּרָה / וְאֵיךְ נַעַשׂ מֻרְבֶּכֶת / וְאָנוּ מְרֻבָּכִים

רָחֲבָה הֻצְרָה / וְנָסְבָה הֻסְגָּרָה / וְאֵיךְ נַרְחִיב פֶּה / וְאוֹיְבֵינוּ פְלִילִים.


שְׁלֵמִים הָאָנְחוּ / שְׁלָמִים שָׁבָתוּ / וְאֵיךְ נְשַׁלֵּם שַׁי / וּמִשָּׁלוֹם זָנַחְנוּ

שִׁבְעָה תְמִימִים / תַּמּוּ מֵאַרְצֵנוּ / וְאֵיךְ נְהַלֵּל שֶׁבַע / וְיֻסַּרְנוּ בְשֶׁבַע.


תְּפִלָּה פָסְקָה / מִבֵּית תְּפִלָּה / וְאֵיךְ נַעַן תְּפִלָּה / וְסָךְ בְּעַד תְּפִלָּה.

תְּמִידִים כָּלוּ / מִיַּעַר תְּמִידִים / וְאֵיךְ נַעַשׂ תְּמִידִים / וּבָטֵל הַתָּמִיד.
