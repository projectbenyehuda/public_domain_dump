וַיַּרְא הַזְּמָן כִּי נִסְתַּחֲפָה : מקאמה

חֲתֻנַּת זְקוּנִים


– – – וַיַּרְא הַזְּמָן כִּי נִסְתַּחֲפָה שָׂדִי / וַיָּבֹא לָנוּד לִי וּלְסַעֲדִי / וּלְדַבֵּר עַל לִבִּי וּלְנַחֲמִי / כִּי עוֹרִי חָרָה עַל עַצְמִי / וַיַּרְא כִּי גָּדַל הַכְּאֵב מְאֹד / וַיֹּאמַר: ׳זָכוֹר אֶזְכָּרְךָ עוֹד / אֶת בִּתִּי אַהֲבָה הַמְרֻצָּה / אוֹתְךָ עָזְבָה וַתֵּצֵא הַחוּצָה / חַיָּתְךָ לַמָּוֶת אַל תִּתֵּן / בִּשְׁתַּיִם בִּי תִּתְחַתֵּן / בַּת אַחַת יֵשׁ לִי שְׁמָהּ שֵׂיבָה / חַיָּתְךָ תְּבַלֶּה בָּהּ / וּמְאֵרָתְךָ וְדִמְעֲךָ לֹא תְאַחֵר / וְנֶהְפַּכְתָּ לְאִישׁ אַחֵר / קַלְקָלָתְךָ עַתָּה אֲתַקֵּן / לֹא תִּקָּרֵא עוֹד פֶּרֶא כִּי אִם זָקֵן / וְכַבֵּד אֲכַבֶּדְךָ עוֹד כְּמֶלֶךְ בַּגְּדוּד / תֵּשֵׁב רֹאשׁ לִסְעוֹד / בְּרֹאשׁ עַם קֹדֶשׁ תִּמְנֵי / סֵּפֶר תִּקְרָא בְּנִהְנֵי / אֵלֶיךָ אֶל עֵצָה יָבֹאוּ / יִרְאוּךָ נְעָרִים וְיֵחָבְאוּ / מִצְוַת ׳וְהָדַרְתָּ׳ בְּךָ תְקֻיָּם / וְתִהְיֶה יָראוּי וּמְאֻיָּם׳.


וָאֹמַר: ׳אָנָּא אֲדֹנִי אִם תַּשְׁאִיבֵנִי מַיִם בְּשָׂשׂוֹן / הֵיטַבְתָּ חַסְדְּךָ הָאַחֲרוֹן מִן הָרִאשׁוֹן / אֲבָל דַּאֲגָתִי וְכָל מְגוּרָי / אוּלַי לֹא תֹאבֶה הָאִשָּׁה לָלֶכֶת אַחֲרָי׳ / וַיִּשָּׂא יָדָיו הַשָּׁמַיִם וַיִּשָּׁבַע בְּחֵי הָעוֹלָמִים / כִּי לֹא תַעֲזֹב אוֹתִי כָל הַיָּמִים / וַיִּקַּח הַזְּמָן שֵׂיבָה בִתּוֹ וַיִּתְנָהּ לִי לְאִשָּׁה / תַּחַת זַיִת יֵצֵא חוֹחַ וְתַחַת שְׂעֹרָה אְשָׁה / כִּי מֵהַשֵּׁם הָיְתָה סִבָּה / וַיֹּאמֶר לַעֲשׂוֹת לָהּ כְּתֻבָּה / בָּרְבִיעִי בְּשַׁבָּת / מְשׂוֹשִׂי שָׁבַת / וְנֵרִי כָּבָת / בַּחֲמִשָּׁה יָמִים לְיֶרַח בּוּל / שְׁנָתַיִם אַחַר הַמַּבּוּל / מָאתַיִם וְעֶשֶׂר לִבְרִיאַת עוֹלָם / אִשְׁתְּקַל מִלּוּלִי וָאֵשֵׁב נֶאֱלָם / הִגִּיעוּ יָמִים אֵין בָּהֶם חֵפֶץ נָשִׁיתִי טוֹבָה / בָּא זָקֵן וְאָמַר לְמָרַת שֵׂיבָה / הֱוִי לִי לְאִנְתּוּ וַאֲנָא אֶפְלַח וְאוֹקִיר / נְגַשְּׁשָׁ כְּהִלְכוֹת גּוּבְרִין / דִּמְרִירָן נַפְשִׁין יְהוּ כְּגִדִּין מְרָרִין / דְּמוֹקְרִין וְזָנִין בְּקוּשְׁטָא / בִּקְטָטָה חֲרָטָה וַעֲלָטָה / וְדָא נְדוּנְיָא דְּהַנְעֲלָה לֵיהּ / הַפִּסֵּחַ וְהָעִוֵּר וְהַחוֹלֶה / הִשִּׁעוּל וְהַקִּיא וְהַבּוּשָׁה / דְּהַנְעַלַת לֵיהּ מִבֵּי נַשָׂא / וּצְבִי ר׳ זָקֵן וְאוֹסֵיף לַהּ מָנָה אַחַת אַפַּיִם / פִּיק בִּרְכַּיִם / כִּהְיוֹן שִׁנַּיִם / כְּאֵב עֵינַיִם וְרִפְיוֹן יָדַיִם / וּשְׁחִין עַל הַבִּרְכַּיִם וְעַל הַשּׁוֹקַיִם / סַךְ הַכֹּל קִבֵּל עָלָיו הֶחָתָן / עִם זוּזֵי קֶצֶף מָאתַן / מָאתַיִם זְקוּקִים לְמִשְׁקַל עוֹל בַּרְזֶל מְנֻקִּים / מִכָּל גֹּדֶל וְרֶחֶם מַשְׂכִּיל וְשָׁדַיִם צוֹמְקִים / וּצְבָאַת מָרַת שֵׂיבָה וַהֲוָת לֵיהּ לְאִנְתּוּ / בְּכוֹס אֶחָד יִשְׁתּוּ / כְּדַת מֹשֶׁה וְיִשְׂרָאֵל / וְלֹא יַפְרִידֵם קָרוֹב אוֹ גוֹאֵל / עַד עֲמוֹד הַשַּׂר מִיכָאֵל.


וְזֹאת הַבְּרָכָה אֲשֶׁר בֵּרַךְ הַזְּמָן זָקֵן וְשֵׂיבָה / וַיֹּאמַר: ׳הֱיוּ לְאַלְפֵי רְבָבָה / מַרְבִּים לְהוֹצִיא מַטְבֵּעַ / רֹק כִּיחַ פִּיעַ וְנִיעַ / וְרוּחוֹת פּוֹרְחוֹת לְעֵת הַשֵּׁינָה / תַּחְתִּיִּים וְעִלִּיִּים תַּעֲשֶׂנָּה / וְתֶמֶד מֵימֵיכֶם מְצוּיוֹת / גֻּלּוֹת עִלִּיּוֹת וְגֻלּוֹת תַּחְתִּיּוֹת / וּבְצִנּוֹרוֹת מְקַלְּחִין בִּמְרוּצָה / יָפוּצוּ מַעְיְנוֹתֵיהֶם חוּצָה / פִּיכֶם תִּתְּנוּ לַחֲטִיא אֶת בְּשַׂרְכֶם / בְּלֶכְתְּךָ בַדֶּרֶךְ וּבְשָׁכְבְּכֶם / וְשֶׂרֶט תִּתְּנוּ בִּבְשַׂרְכֶם / וּמִן הֶחָזֶה הַשִּׁיר יִהְיֶה לָכֶם / אִסְטוּמְכָּה מְלֵאָה וּמֵעַיִם רוֹעֲמוֹת / מְעַט שֵׁינָה וּמְעַט תְּנוּמוֹת / תִּזְרְעוּ גָּרָב וּשְׁחִינִים תַּצְמִיחוּ / תְּנוּבוּן בְּבָנִים וּבְשֵׂיבָה תַּפְרִיחוּ.


וַיְהִי לְעֵת עֶרֶב / בְּבֹא הַשֶּׁמֶשׁ וְלַיִל קָרַב / וְיָרֵחַ לֹא נָתְנָה אוֹרָהּ / וַיְבִיאֶהָ הַזְּמָן אֵלַי הַחֶדְרָה / וָאֶקְרַב אֶל הַנְּכֵאָה / וַיְהִי בַבֹּקֶר וְהִנֵּה הִיא לֵאָה / וְעֵינֵי לֵאָה רַכּוֹת / אֲפֵלוֹת כֵּהוֹת חֲשׁוּכוֹת / שִׁנֶּיהָ עֲקוּמוֹת וַחֲרוּקוֹת / צְנוּמוֹת דַּקּוֹת / מֵחוּשׁ הַשָּׁנוּי שִׂפְתּוֹתֶיהָ / וְהִלְבִּינוּ שַׂעֲרוֹתֶיהָ / כְּפֶלַח הָרִמָּה רַקָּתָהּ / מִבַּעַד לְצַמָּתָהּ / וְצַלְעוֹתֶיהָ וּמִפְרַקְתָּהּ / אַחַת גְּבוֹהָה מֵחֲבֶרְתָּהּ / קְמָטִים קְמָטִים בְּפָנֶיהָ / וְגָדְלוּ רִיסֵי עֵינֶיהָ / חֲצִי נָשְׁרוּ / וְעַד הַיְסוֹד עָרוּ / פִּיהָ בָּנוּי לְתַלְפִּיּוֹת / בָּרוּךְ מְשַׁנֶּה הַבְּרִיּוֹת / כְּמוֹתָהּ לֹא רָאִיתִי כְעוּרָה / עַל עַצְמָהּ צָפַד עוֹרָהּ / וַיֵּאָכֵל חֲצִי בְּשָׂרָהּ / וְעַל רֹעַ פָּנֶיהָ הַדַּלִּים / כְּבָר אָמְרוּ הַמּוֹשְׁלִים:


מְאֹד נִכְאָה אֲשֶׁר יִרְאֶה בְּעוֹד יוֹם

פְּנֵי שֵׂיבָה וְתִבְעַר בּוֹ חֲמָתוֹ

אֲזַי יָקוּם חֲצוֹת לַיְלָה לְדַלֵּג

חֲלוֹן בֵּיתָהּ וְיִשְׁבֹּר מִפְרַקְתּוֹ.


וַיְהִי כַּאֲשֶׁר קָרַבְתִּי אֶל הַמְּשֻׁנָּה / נִתְעַבְּרָה מִבִּיאָה רִאשׁוֹנָה / וַתַּהַר לִי בְּאוֹתָהּ לַיְלָה / אֶת חוּלְשָׁה וְאֶת מַחֲלָה / שְׁתֵּי בָנוֹת אוֹמְרוֹת הַב הַב / הַזָּלוֹת מִכִּיס זָהָב / וַיְהִי בַבֹּקֶר וַתִּפָּעֶם רוּחִי / וָאֹמַר: אָבַד נִצְחִי / כִּי נִרְאָה בְעֵינַי הַנֶּגַע / וְעֵת וָפֶגַע / קְרָאַנִי וְרֹאשׁ לָחֳלִי / כִּי בָאָה שֵׂיבָה בְגוֹרָלִי / מַה חֶפְצִי עוֹד בְּבֵיתִי / וְאָנָה אוֹלִיךְ אֶת חֶרְפָּתִי / הַעַיִט צָבוּעַ הָיְתָה לִי נַחֲלָתִי / וְלֹא עָצַרְתִּי עוֹד כֹּחַ / כִּי תַחַת הַחִטָּה עָלָה חוֹחַ / וָאֶצְעַק אֶל הַזְּמָן וָאֹמַר: ׳אֲהָהּ אֲדֹנִי / הֵן בְּצֶדֶק עָבַדְתִּי עִמָּךְ וְלָמָּה רִמִּיתָנִי / וַתִּגְנֹב אֶת לְבָבִי / וַתֹּסֶף מַכְאוֹב עַל מַכְאוֹבִי / כִּי מֵאָז עֶרְכִּי נִגְרַע / וְהֶחֱלַפְתִּי טוֹב בְּרָע / אֵיךְ תַּשְׁוֶה שֵיבָה עִם קְמָטֶיהָ / לְחֶפְצִי בָהּ וְיֹפִי לְהָטֶיהָ / וְאֵין עֶדְנֵי בָשָׂר לְשֵׂיבָה / כְּמֹתֶק עֶדְנַת חֶפְצִי בָהּ / וְעַתָּה אֲדֹנִי הוֹשִׁיעָה / וּמַלְּטֵנִי מִן הָרָעָה׳ / וַיַּעַן הַזְּמָן: ׳הַהֵיטֵב חָרָה לָּךְ / הַס אַל תַּשְׁמִיעַ עוֹד אֶת קוֹלָךְ / בִּתִּי שֵׂיבָה אַל תְּגַנֶּה / אַל תָּבוֹז לַקְּמָטִים שֶׁבְּפָנֶיהָ / וְאַל תֵּבוֹשׁ שֵׂיבָה לַעֲבֹד / כִּי נֶגֶד זְקֵנַי כָּבוֹד / וְאֵיךְ תִּהְיֶה כְּפוּי טוֹבָה / הֵן תִּפְאֶרֶת זְקֵנִים שֵׂיבָה / וְעַתָּה רַב לָךְ אַל תּוֹסֶף דַּבֵּר / עִם בִּתִּי שֵׂיבָה תִּתְחַבֵּר / תִּדְבַּק בָּךְ וְהָיְתָה עִמָּךְ / לֹא תוּכַל שַׁלְּחָהּ כָּל יָמָךְ׳. – – –
