מִי לֹא יִרָאֲךָ מֶלֶךְ הַגּוֹיִם (סְלִיחָה עַל הָרַעַשׁ)

[חי בשנת 1329 בלָרִפָּה (Larippa) בפלך מַרְקָה (Marche d’Ancona) ואחר כך בנוֹרְצִי]


סְלִיחָה עַל־הָרַעַשׁ בְּפֶלֶךְ מַרְקָה


מִי לֹא יִרָאֲךָ מֶלֶךְ הִגּוֹיִם

מִגַּעֲרָתְךָ תָּנוּט הָאָרֶץ יִרְעֲשׁוּ הָאִיִּים

יִרְגְּזוּ יָחִילוּ לְפָנֶיךָ יִכְרְעוּ צִיִּים

כָּל־בָּשָׂר אֲשֶׁר־בּוֹ רוּחַ חַיִּים.


מִשָּׁמַיִם הִשְׁמַעְתָּ דִּין אֶרֶץ רָעָשָׁה

פּוֹר הִתְפּוֹרְרָה אֶרֶץ הִתְנוֹדֲדָה כַּמְּלוּנָה הִתְגָּעָשָׁה

עִם־כָּל־צֶאֱצָאֶיהָ קַדְרוּת לָבָשָׁה

הִרְאִיתָ עַמְּךָ קָשָׁה.


תַּחַת אָוֶן אָהֳלֵי עוֹלָם רָאִיתִי

נִפְלָאִים מַעֲשֶׂיךָ עַל־נוֹרְאוֹתֶיךָ נִפְלֵיתִי

קָטֹנְתִּי בְעֵינַי וְגָדַלְתָּ מְאֹד אֵמֶיךָ נָשָׂאתִי

יְיָ שָׁמַעְתִּי שִׁמְעֲךָ יָרֵאתִי.

תֵּבֵל רָגְזָה וַתִּרְעַשׁ הָאָרֶץ

נָטִיתָ יְמִינְךָ מַה־פָּרַצְתָּ עָלֶיךָ פָּרֵץ

תְּמֹךְ אוֹתָהּ בִּימִין צִדְקְךָ וְיֶחְדַּל הַקָּרֶץ

אַתָּה הִצַּבְתָּ כָּל־גְּבוּלוֹת אָרֶץ.


תַּרְגִּיז אֶרֶץ מִמְּקוֹמָהּ וְעַמּוּדֶיהָ זָעוּ

אָז נִבְהֲלוּ אַלּוּפֵי אֱדוֹם כִּי נָצוּ גַּם־נָעוּ

אַבִּירֵי לֵב חַתּוּ וּמִגְּאוֹנָם נִכְנָעוּ

וַיִּרְא הָעָם וַיָּנֻעוּ.


תֹּלֶה אֶרֶץ עַל־בְּלִימָה בְּחָכְמָה

הַעֲבֵר עָלֶיהָ רוּחַ חֵן וְנֶחָמָה

דַּבֵּר עַל־לִבָּהּ כִּי כִמְעַט שָׁכְנָה דוּמָה

לֹא אֹסִיף לְקַלֵּל עוֹד אֶת־הָאֲדָמָה.


יוֹנָתְךָ חָרְדָה לִרְגָּעִים לֹא מָצְאָה מָנוֹחַ

אַתָּה זָכַרְתָּ מִתּוֹךְ הַמַּבּוּל הִצַּלְתָּ נֹחַ

זָכַרְתָּ אֶת־אַבְרָהָם מִתּוֹךְ הַהֲפֵכָה לִשְׁלֹחַ

זָכְרָה־לָּנוּ לְטוֹבָה גָּדוֹל אֲדוֹנֵינוּ וְרַב־כֹּחַ.


יֶעֱרֵב עָלֶיךָ כְּרֵיחַ נִיחֹחַ שִׂיחַ יְלָדֶיךָ

וּשְׁחָקִים יִזְּלוּ־צֶדֶק עַל־צֹאן יָדֶיךָ

מִמֲתִים יָדְךָ יְיָ מִמֲתִים מֵחֶלְדֶּךָ

זְכֹר לְאַבְרָהָם לְיִצְחָק וּלְיִשְׂרָאֵל עֲבָדֶיךָ.


הִרְעַשְׁתָּה אֶרֶץ פְּצַמְתָּהּ אָבְלָה נָבְלָה

קִרְיָה בְצּוּרָה לְמַפֵּלָה

כָּל־לְבַב אֱנוֹשׁ יִמַּס בְּכָל־מָתְנַיִם חַלְחָלָה

וְחָרְדוּ אִישׁ לְנַפְשׁוֹ קוּמִי רֹנִּי בַלַּיְלָה.


הַמַּבִּיט לָאָרֶץ וַתִּרְעַד וַיִּתְפֹּצְצוּ הַרְרֵי־עַד

מֶלֶךְ רָם וְנִשָּׂא שֹׁכֵן עַד

בְּרֹגֶז רַחֵם תִּזְכֹּר וְלֹא נִמְעָד

הָרוֹפֵא לִשְׁבוּרֵי לֵב וּלְבַב אֱנוֹשׁ יִסְעָד.


הָאֲדָמָה חַתָּה וַיִּרְעֲשׁוּ כָּל־צֶאֱצָאֶיהָ

קָרְאָה בַת־אֱדוֹם בְּחָזְקָה אֶל־אֱלֹהֶיהָ

טְמֵאַת הַשֵּׁם פְּשֹׁטָה וְעֹרָה וַחֲגוֹרָה עַל־חֲלָצֶיהָ

נְמֹגִים־אֶרֶץ וְכָל־יֹשְׁבֶיהָ.


חוּסֶה עַל־עַמְּךָ וּשְׁמַע קוֹלָם

מִתִּגְרַת יָדְךָ וּמִכְּלֵי זַעַמְךָ לְהַצִּילָם

עַל־יַד יְמִינָם הֱיֵה סִתְרָם וְצִלָּם

וּמִתַּחַת זְרֹעֹת עוֹלָם.


זַלְעָפָה אָחֲזָה רְשָׁעִים עֹזְבֵי תּוֹרָתֶךָ

כָּל־כָּתֵף מְרוּטָה חֲבוּטָה לְאֵל זָר בִּלְתֶּךָ

לְבַד־בְּךָ נַזְכִּיר שִׁמְךָ כִּי אֵין זוּלָתֶךָ

אֲשֶׁר־יַעֲשֶׂה כְמַעֲשֶׂיךָ וְכִגְבוּרֹתֶךָ.


קַבֵּץ נְפוּצוֹת יְהוּדָה וְנִדְחֵי יִשְׂרָאֵל

יוּבֵל לָטֶבַח כַּכֶּבֶשׂ אַלּוּף מַגְדִּיאֵל

וְלֹא יִרַשׁ בֶּן־הָאָמָה אֶת־אֲרִיאֵל

וְשָׁבוּ בָנִים לִגְבוּלָם וּבָא לְצִיּוֹן גּוֹאֵל.
