משל סז: שְׁנֵי קוֹפִים וַאֲרִי

אַנְשֵׁי תְכָכִים מְלֵאִים קִנְאָה / זֶה עַל זֶה יָעִירוּ שִׂנְאָה


שְׁנֵי קוֹפִים הִתְהַלְּכוּ לִפְנֵי הָאֲרִי הַמֶּלֶךְ / וּלְבַקֵּשׁ מַתָּת מִמֶּנּוּ כָּל אֶחָד הֹלֵךְ / וַיָּבֶן הָאֲרִי כַּאֲשֶׁר זֶה יְדַבֵּר וְזֶה יַעֲנֶה / כִּי הָאֶחָד חֶמְדָּן וְהַשֵּׁנִי מְקַנֵּא / וַיֹּאמֶר לָהֶם: אִם לִשְׁאֹל מִשֶּׁלִּי עֵינֵיכֶם עָלַי / אֶתְּנָה לָכֶם כַּאֲשֶׁר תֹּאמְרוּ אֵלָי / אַךְ בְּזֹאת יֵאוֹת לִי / אִם שָׁמֹעַ תִּשְׁמְעוּ בְקוֹלִי / אֶחָד מִכֶּם יְגַלֶּה לְאָזְנִי בַקָּשָׁתוֹ / וְאַחֲרֵי דְבָרוֹ לֹא אֶשְׁנֶה לָתֵת אֶת שְׁאֵלָתוֹ / וְהַשֵּׁנִי אֲשֶׁר יַחֲרִישׁ בְּפֶה וּבְלָשׁוֹן / עַד תֹּם בַּקָּשַׁת הָרִאשׁוֹן / הִנֵּה הוּא לִי תַאֲוַת עֵינַיִם / לָתֶת לוֹ פִּי שְׁנַיִם / אֲשֶׁר אֶתֵּן לַחֲבֵרוֹ / כִּי אֶחְפֹּץ בִּכְבוֹדוֹ וּבִיקָרוֹ / וְהַחֶמְדָּן אָמַר בְּלֵב תֹּךְ וּמִרְמָה / הַחֲרֵשׁ אַחֲרִישׁ וּתְהִי לִי חָכְמָה / וְאֶשְׁמַע מָה יִּשְׁאַל חֲבֵרִי וּמַה יַּשְׁמִיעַ בְּקוֹלוֹ / וִיהִי לִי פִּי שְׁנַיִם כְּכָל אֲשֶׁר יִמָּצֵא לוֹ / וַיְהִי כִּי הֶחֱרִישׁ וְהִקְשִׁיב קָשֶׁב / וַיֹּאמֶר בְּלִבּוֹ הַמְקַנֵּא: חֲבֵרִי הַמַּחֲרִישׁ חָכָם לֹא יֵחָשֵׁב / גַּם לִי לֵבָב כָּמֹהוּ אֲשֶׁר מִתַּחְבֻּלּוֹת לֹא יִגְרָע / וְגַם הוּא חָכָם וַיָּבֵא רָע / וַיִּקְרָא הַמְקַנֵּא אֶל הַמֶּלֶךְ לֵאמֹר / מוֹצָא שְׂפָתֶיךָ תִּשְׁמֹר / וְעָשִׂיתָ כַּאֲשֶׁר דִּבַּרְתָּ בְאָזְנָי / שֹׁאֵל אֲנִי מִמְּךָ לְנַקֵּר אַחַת מֵעֵינַי / הִנֵּה גִלִּיתִי בְאָזְנְךָ אֶת תַּאֲוָתִי / מִי יִתֵּן תָּבֹא שֶׁאֱלָתִי / וַיֹּאמֶר הָאֲרִי: לַעֲשׂוֹת בַּקָּשָׁתְךָ אֶשְׁמַע בְּקוֹלְךָ / כִּי לֹא אֶעֱזָבְךָ עַד אֲשֶׁר אִם עָשִׂיתִי אֶת אֲשֶׁר דִּבַּרְתִּי לְךָ / וּלְחֶמְדָּן אָמַר: הִנְנִי לֹקֵחַ מִמְּךָ שְׁתֵּי עֵינֶיךָ / בִּין תָּבִין אֶת אֲשֶׁר לְפָנֶיךָ / וַיָּשֶׁב לָהֶם כִּגְמוּל תַּחְבֻּלּוֹתָם / וַיִּתֵּן לָהֶם אֶת שֶׁאֱלָתָם.


וְהַמָּשָׁל בְּאַנְשֵׁי תְכָכִים מְלֵאִים קִנְאָה / אֲשֶׁר זֶה עַל זֶה יָעִירוּ שִׂנְאָה / זֶה יֹאמַר בְּלִבּוֹ: אֵיךְ שְׁכֵנִי הוּרַם וְהוּנָף / אַכֶּנּוּ בַלָּשׁוֹן וְלֹא אַשְׁאִיר לוֹ שֹׁרֶשׁ וְעָנָף / כִּי סָרָבִים וְסַלּוֹנִים הוּא אִתִּי / וּבְטֶרֶם יִפֹּל מֵאֲשֶׁר זָמַמְתִּי / אֶתֵּן שָׂדוֹת כְּרָמִים וּבָתִים / תָּמֹת נַפְשִׁי עִם פְּלִשְׁתִּים / וְלַעֲשׂוֹת רָע לִשְׁכֵנוֹ יִתְנַכֵּר / וְאֶל צְרוֹר נָקוּב מִשְׂתַּכֵּר / וִיחַפְּשׂוּ בְלִבָּם מַהֲתַלּוֹת / וְסִבּוֹת וְעָרְמוֹת וְתַחְבֻּלּוֹת / גַּם אִם יֹאמַר לָהֶם הַמֶּלֶךְ: הַמְתִּיקוּ סוֹד יַחַד / לָמָּה אֶשְׁכַּל גַּם שְׁנֵיכֶם יוֹם אֶחָד / אָזְנַיִם לָהֶם וְלֹא יִשְׁמָעוּ / כִּי מִדֶּרֶךְ הַשֵּׂכֶל תָּעוּ / עַד אֲשֶׁר זֶה לְהוֹן שְׁכֵנוֹ שִׁבֵּר וְאִבַּד / יִרְהֲבוּ הַנַּעַר בַּזָּקֵן וְהַנִּקְלֶה בַּנִּכְבָּד / וְאָמַר הֶחָכָם: דַּיֶּךָ מִן הַמְקַנֵּא בִרְאוֹתְךָ / אֲשֶׁר יֶאֱבַל לְשִׂמְחָתְךָ וּלְטוֹבָתְךָ.
