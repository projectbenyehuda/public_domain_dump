משל כג: אַרְיֵה וְשֻׁעָל

נִסְתָּר בְּעֵת רָעָה / מִשְּׁמוּעָה לֹא יִירָא


אַרְיֵה חוֹשֵׁב בְּתַחְבֻּלּוֹת / בְּאִישׁוֹן לַיְלָה וַאֲפֵלוֹת / וּדְבַר בְּלִיַּעַל יָצוּק בּוֹ / וַיֹּאמֶר הָאֲרִי בְּלִבּוֹ / כָּל מֶלֶךְ יוֹשֵׁב בְּבֵיתוֹ חָפְשִׁי / וְלָמָּה זֶה אֲנִי עָמֵל וְלָרִיק מְחַסֵַּר נַפְשִׁי / בְּנֹעַ וְנֹד אַחַר טַרְפִּי / וְלֹא יָבֹא דָבָר תּוֹךְ פִּי / כִּי אִם בְּמִקְרֶה וָפֶגַע / לָמָּה זֶה הֶבֶל אִיגַע / לִשְׁבֹּר גּוּפִי בִמְרוּצָתִי / וּמְנוּחָה לֹא מָצָאתִי / וּמִי בְּכָל עֲבָדַי אֲשֶׁר עָלַי חָמָל / טוֹב מְלֹא כַף נַחַת מִמְּלֹא חָפְנַיִם עָמָל / וַיַּגֵּד לַכֹּל בְּעָרְמָתוֹ / כִּי מְאֹד עָצְמָה מַחֲלָתוֹ / וַיַּזְעֵק בְּהֵמָה וְחַיָּה / בַּחֲלָקוֹת וּבִשְׂפַת רְמִיָּה / מַהֲרוּ וּבֹאוּ לִרְאוֹתִי / לֹא יָדַעְתִּי יוֹם מוֹתִי / כִּי אֵין דָּן דִּינִי לְמָזוֹר / רְכוּשׁ וְשָׁלָל לָכֶם אֶבְזוֹר / וְכָל מַחְסוֹרְכֶם עָלַי / וְאֶתְּנָה כַּאֲשֶׁר תֹּאמְרוּ אֵלַי / וְאָשִׁיב לְאִישׁ וָאִישׁ כְּצִדְקָתוֹ / לָרַב תַּרְבּוּ נַחֲלָתוֹ / אִם לֹא עַכְשָׁו אֵימָתַי / הִנְנִי שׁוֹכֵב עִם אֲבֹתַי / וַיִּסְעוּ לָלֶכֶת כֻּלָּם / מִקְּטַנָּם וְעַד גְּדוֹלָם / וַיְהִי בְעוֹד כִּבְרַת אֶרֶץ לָבֹא / וַיֻּגַּד מִטַּעַם הַמֶּלֶךְ וְנִיבוֹ / הַבָּא לִרְאוֹתוֹ יָבֹא אֶחָד אֶחָד / וְאַל יֵלְכוּ שְׁנַיִם יַחַד / לִרְאוֹת חוֹלֶה כָּמוֹהוּ / פֶּן בְּקוֹלָם יַבְהִילוּהוּ / וַיֵּלֶך הָאֶחָד לְבַדּוֹ / וַיִּמְצָאֵהוּ שֹׁכֵב עַל צִדּוֹ / הָאֲרִי בְסֻכּוֹ כְּמֵת / וַיֹּאמֶר לוֹ: אֲדֹנִי בֶאֱמֶת / לִרְאוֹת שְׁלוֹמְךָ מִהַרְתִּי אֲשׁוּרָי / וַיַּעַן הָאֲרִי: מַה לְּךָ וּלְשָׁלוֹם, סֹב אֶל אֲחֹרָי / הֶלְאֵתַנִי כַּמָּה פְעָמִים / הֲלֹא תִכָּלֵם שִׁבְעַת יָמִים / בְּשָׂרְךָ אֹכַל וַעֲצָמֶיךָ אֶשְׁבֹּר / אַל יָנוּס הַקַּל וְאַל יִמָּלֵט הַגִּבּוֹר / וַיַּעַשׂ מִמֶּנּוּ כִדְבָרוֹ / וַיְשַׁסְּעֵהוּ וַיֹּאכַל בְּשָׂרוֹ / עוֹד זֶה אוֹכֵל וְזֶה בָּא וַיֹּאמַר / אֲדֹנִי הַמֶּלֶךְ מֵחָלְיְךָ בְּשָׂרִי סָמָר / וַיְפַתֵּהוּ בְתוֹךְ סֻכּוֹ בְּחֵלֶק לָשׁוֹן / וַיַעַשׂ לוֹ כְּמִשְׁפַּט הָרִאשׁוֹן / גַּם הַשְּׁלִישִׁי גַּם כָּל הַהֹלְכִים אַחֲרֵי הָעֲדָרִים / לֹא זָז מִשָּׁם עַד שֶׁנַּעֲשָׂה אֲבָרִים אֲבָרִים / וְכֵן כֻּלָּם עַד תֻּמָּם / לֹא נִשְׁאַר רַק הַשֻּׁעָל אֲשֶׁר בְּלִבּוֹ זָמַם / לָדַעַת וְלִשְׁמֹעַ מָה הַמֶּלֶךְ בְּלִבּוֹ דָרַשׁ / מַה יֵּעָשֶׂה לוֹ כִּי לֹא פֹרָשׁ / וַיָּפָג לִבּוֹ כִּי לֹא הֶאֱמִין אֲשֶׁר יֹאמְרוּ בָעַמִּים / לְכֻלָּם יִתֵּן הַמֶּלֶךְ שָׂדוֹת וּכְרָמִים / וְאִם אֱמֶת אִתָּם, לָמָּה הֵעִיד בַּעֲבָדָיו / לְבִלְתִּי לֶכֶת שְׁנַיִם יַחַד עָדָיו / לֹא אֵלֵךְ פֶּן אֶפֹּל בְּפַח מִרְמָה וָרֶשֶׁת / וַיֵּלֶךְ וַיֵּשֶׁב מִנֶּגֶד הַרְחֵק כִּמְטַחֲוֵי קֶשֶׁת / וַיִּתְחַבֵּא בָדָד בְּמִסְתָּרִים / לִשְׁמֹעַ שְׁרִיקוֹת עֲדָרִים / וַיֹּאמַר: בִּרְאוֹתִי כִּי הֹלֵךְ הָעַכְשׁוּב / רוּחַ הֹלֵךְ וְלֹא יָשׁוּב / וְהִנֵּה כֻּלָּם הֹלְכִים בְּלִי פַחַד / אַךְ לֹא יָשׁוּב אַף אֶחָד / כִּי לֹא יָדַעְתִּי מָה הוּא / וְהַמִּנְהָג כְּמִנְהַג יֵהוּא / וְהָיְתָה לָהֶם הַמְסִלָּה / כַּחֲלַקְלַקּוֹת בַּאֲפֵלָה / כִּי מִתְאַנֶּה הַמֶּלֶךְ לָהֶם / וְלֹא יָשׁוּבוּ לְאָהֳלֵיהֶם / וּבָא יוֹם אֵיד לַאֲשֶׁר קִוּוּ לִרְאוֹת אֵידוֹ / וְלֹא יִסְפְּדוּ לוֹ: הוֹי אָדוֹן וְהוֹי הוֹדוֹ / וְעַל כִּי לֵב הַשֻּׁעָל מִפַּחַד יִתָּר / עָרוֹם רָאָה רָעָה וְנִסְתָּר / וַיְהִי כַּאֲשֶׁר כִּלָּה הָאֲרִי לִטְרֹף כָּל הַבָּא / וַיִּתְבּוֹנֵן כִּי הַשֻּׁעָל בְּעָרְמָתוֹ נֶחְבָּא / וַיִּקְרָא: לָמָּה זֶה וְעַל מַה זֶּה אֶחֱרוּ פְעָמֶיךָ / הֲלֹא שָׁלַחְתָּ יָד בִּשְׁלוֹמֶיךָ / וַיַּעַן לוֹ: דֶּרֶךְ הַהֹלְכִים רָאִיתִי / וְעִקְבוֹת הַשָּׁבִים לֹא מָצָאתִי.

וְהַמָּשָׁל עַל כֵּן יֹאמְרוּ הַמֹּשְׁלִים / צוֹק הָעִתִּים בְּסִבּוֹת הַגַּלְגַלִּים / וּבְבוֹא דְּבֵַר פֶּלֶא כָּל אִישׁ יִדְרֹשׁ שְׁלוֹמוֹ / שְׁבוּ אִישׁ תַּחְתָּיו אַל יֵצֵא אִישׁ מִמְּקוֹמוֹ / כִּי לֹא טוֹבָה הַשְּׁמוּעָה / יוֹסֵף אֶת דִּבָּתָם רָעָה / וְיֵשׁ נִסְתָּר בְּיוֹם עֶבְרָה / מִשְּׁמוּעָה רָעָה לֹא יִירָא / דַּיָּה צָרָה בְשָׁעָתָהּ בְּבוֹאָה אֵלֶיךָ / וְיֵשׁ שֶׁבֶר בְּאָהֳלֶךָ.
