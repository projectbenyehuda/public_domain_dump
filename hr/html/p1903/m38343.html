משל פה: אֲרִי וּזְאֵב וְדֹב וְשֻׁעָל

חֹרֵשׁ עַל רֵעֵַהוּ רָעָה / קִלְלַת חִנָּם עָלָיו בָּאָה


הָאֲרִי חָלָה חָלְיוֹ אֲשֶׁר יָמוּת בּוֹ / וַיָּבֹאוּ הַזְּאֵב וְהַדֹּב לְדַבֵּר עַל לִבּוֹ / וְכָל חַיְתוֹ אֶרֶץ לְמִינוֹ / רַק שֻׁעָל אֲשֶׁר מִסְתַּתֵּר הִנּוֹ / לָדַעַת סוֹף דָּבָר לִבּוֹ נִפְעָם / וּמַה יִּהְיֶה בְּאַחֲרִית הַזָּעַם / וְאִם הוּא מֵאֵת הַמֶּלֶךְ יִפָּקֵד / וְלִשְׁמֹעַ מַה יְּדַבְּרוּ בְמִסְתָּרִים שֹׁקֵד / וַיִּשְׁמַע בֶּאֱמֹר הַזְּאֵב אֶל הָאֲרִי / אֲדֹנִי, שְׁמַע נָא אֶת דְּבָרִי / שְׁלַח נָא הַשֻּׁעָל לְהָבִיא צֳרִי מִגִּלְעָד / וּבְשָׂמִים וּזְנִים אֲשֶׁר לִבְּךָ מֵהֶם יִסְעָד / כִּי לוֹ אֹזֶן מִלִּין תִּבְחָן / וְיָדוֹ הַמִּרְקַחַת תִּטְחָן / וְהוּא יֵדַע מִכֻּלָּנוּ הַשְּׁבִילִים / וּבְגִלְעָד מַכִּיר אֶת הָרֹכְלִים / חָכָם וְנָבוֹן אֵין כָּמֹהוּ /וַיֹּאמֶר הַמֶּלֶךְ: קְרָאֵהוּ / וְהַזְּאֵב עָשָׂה בְעָקְבָה / לְהוֹצִיא עַל הַשֻּׁעָל דִּבָּה / לְהַטְרִיחוֹ וּלְהַעֲצִיבוֹ / וּלְהַרְחִיקוֹ מֵעַל נְתִיבוֹ / וְשָׂם לוֹ עֲלִילוֹת כִּי לֹא בָא אֶל הַמֶּלֶךְ לְבַקֵּר / וַיַּגְדֵּל עָלָיו שְׂפַת שֶׁקֶר / וְהַשֻּׁעָל שֹׁמֵעַ קוֹלוֹ / פֹּשֵׂק שְׂפָתָיו מְחִתָּה לוֹ / בֶּאֱמֹר לַמֶּלֶךְ: הַשֻּׁעָל אֵינֶנּוּ / כְּמוֹ חַי כְּמוֹ חָרוֹן יִשְׂעָרֶנּוּ / וּכְשָׁמְעוֹ דִבְרֵי הַזְּאֵב / מִלִּבּוֹ עָרְמָתוֹ שֹׁאֵב / לָתֵת בְּלֵב הַזְּאֵב עַצֶּבֶת / וְעַל שְׂפָתָיו כְּאֵשׁ צָרָבֶת / לַעֲשׂוֹת לוֹ כַּאֲשֶׁר זָמָם / בְּכִלְיוֹתָיו יִשְׁתּוֹנָן וְיֵשֵׁב דּוּמָם / עַד אֲשֶׁר רָאָה פְּנֵי הַמֶּלֶךְ זֹעֲפִים / הָרָצִים יָצְאוּ דְחוּפִים / אָנֶה וְאָנָה הַשֻּׁעָל לְבַקֵּשׁ / וַיִּירָא הַשֻּׁעָל פֶּן יִנָּקֵשׁ / וַיָּבֹא בַלָּאט שׁוֹק אֶל שׁוֹק כַּף אֶל כַּף נֹגֵעַ / כַּאֲשֶׁר הָיָה עָיֵף וְיָגֵעַ / וַיֹּאמַר: יְחִי אֲדֹנִי הַמֶּלֶךְ לְעוֹלָם / אֲשֶׁא מִמֶּנּוּ נִסְתָּר וְנֶעְלָם / טֹרַח דַּרְכִּי וְהַתְּלָאָה / לְבַקֵּשׁ לְךָ אֲרוּכָה וּרְפוּאָה / מִיּוֹם אֲשְֶׁר שָׁכַב אֲדֹנִי וַיִּתְחַל / חָדַל נֹחִי וְנֹעִי הוּחַל / עַד אֲשֶׁר בָּאתִי בְגִלְעָד וְשַׁבְתִּי / לֹא נַחְתִּי וְלֹא שָׁקַטְתִּי / זֶה אַרְבָּעִים יוֹם שָׂבַעְתִּי נְדֻדִים / וְכֹל לֶחֶם צֵידִי יָבֵשׁ הָיָה נִקֻּדִּים / עָבַרְתִּי עַל הַר וְגִבְעָה / וְעַל רְכָסִים וּבִקְעָה / בְּחֹשֶׁךְ וַאֲפֵלָה לֹא יָדַעְתִּי אַיַּי / קִפַּדְתִּי כָאֹרֵג חַיָּי / אַךְ הָרֹפְאִים נָתְנוּ לִי וָאֹכֵל / מִכֹּל אַבְקַת רוֹכֵל / וְכֻלָּם פֶּה אֶחָד אָמָרוּ / כִּי לְחָלְיְךָ רְפוּאָה בָחָרוּ / לְהַפְשִׁיט עוֹר זְאֵב חַי מֵעַל בְּשָׂרוֹ / וְיֻקַּח חַם אֶת עוֹרוֹ / לִהְיוֹת עַל רֹאשׁ הַמֶּלֶךְ לְמִצְנֶפֶת / וְהָיָה לְאוֹת וּלְמוֹפֵת / וְיָבֹאוּ אֹהֲבֶיךָ / וּמִדָּמוֹ יַזּוּ סְבִיבֶיךָ / וְלֹא תֹאמַר אָבַד נִצְחִי / וְיִמְרְחוּ עַל הַשְּׁחִין וְיֶחִי / וְלֹא אֵחַר הַמֶּלֶךְ לַעֲשׂוֹת כִּדְבָרוֹ / וַיִּקְחוּ אֶת הַזְּאֵב וַיַּפְשִׁיטוּ אֶת עוֹרוֹ / אַךְ נִמְלַט בְּעוֹר שִׁנָּיו / וְהָיָה חֶרְפָּה לִשְׁכֵנָיו / כִּי עָרֹם אֹתוֹ הִשְׁאִירוּ / וְעַצְמוֹתָיו כְּמוֹקֵד נִחָרוּ / הַשֻּׁעָל תְּפָשׂוֹ בִמְצוּדָתוֹ / וַיַּסַּע כָּעֵץ תִּקְוָתוֹ / וּבְלַעְגּוֹ יִתְיַצֵּב לְמֻלּוֹ / וַיַּחֲרֹק שֵׁן וַיֹּאמֶר לוֹ / הֲלֹא אַתָּה אֲשֶׁר הַמֶּלֶךְ חָפֵץ בִּיקָרוֹ / הֲיַהֲפֹךְ כּוּשִׁי עוֹרוֹ / אַךְ בְּאִמְרֵי פִיךָ נוֹקַשְׁתָּ / וּבְרֶשַׁע אֲשֶׁר עָלַי חָרַשְׁתָּ / וְרִשְׁעָתְךָ עָלֶיךָ תִתָּךְ / כִּי יָשַׁבְתִּי לָבֶטַח אִתָּךְ.


וְהַמָּשָׁל כִּי שָׁלוֹם לְנֶפֶשׁ לֹא דֹרֵשׁ / אֲשֶׁר עַל רֵעֵהוּ רַע חֹרֵשׁ / לְדַבֵּר עָלָיו בְּזָדוֹן / פּוֹטֵר מַיִם רֵאשִׁית מָדוֹן / וְאַחַת דָּתוֹ לְהָמִית / מְלָשְׁנִי בַסֵּתֶר רֵעֵהוּ אֹתוֹ אַצְמִית.
