משל ו: שֻׁעָל וְדָגִים

רַבּוּ יוֹעֲצִים לְפִי דַרְכָּם / לְהַעֲלוֹת זוּלָתָם בְּחַכָּם

שֻׁעָל הָלַךְ עַל שְׂפַת הַנָּהָר / וַיַּרְא כִּי דָּג אֶחָד מִהַר / לָנוּס וְדָג אַחֵר רְדָפוֹ / עַד כִּי הִשִּׂיגוֹ וּבְאֵיבָה הֲדָפוֹ / וַיִּנָּצוּ שְׁנֵיהֶם / וְאֵין מַצִּיל בֵּינֵיהֶם / זֶה אֶל זֶה יָרוּץ בְּחֵמָה / וְלֹא דִּבֵּר אִתָּם מְאוּמָה / אָכֵן נִגַּשׁ אֶל הַמַּיִם / אוּלַי יִשְׁלַח בָּם שִׁנַּיִם / אוֹ יִלְכְּדֵם בְּרֶשֶׁת עָרְמָתוֹ / אַךְ הַמַּיִם חוֹצְצִים בִּפְנֵי תַאֲוָתוֹ / וַיִפֶן מִשָּׁם אֶל מָקוֹם אַחֵר / וּלְדַבֵּר אֲלֵיהֶם לֹא יְאַחֵר / וּבִרְאוֹתוֹ שָׁם אְֶת הַדָּגִים נִצִּים / וְהַגְּדוֹלִים אֶת הַקְּטַנִּים לוֹחֲצִים / וְאַדִּירֵיהֶם נָשְׁכוּ צְעִירֵיהֶם / וַתִּכְבַּד הַמִּלְחָמָה בֵינֵיהֶם / וַיִּקְרָא לָהֶם: הֲכָל אֶחָד מִכֶּם סָכָל / כִּי הַשְׁקֵט לֹא יוּכַל / הֲזֹאת הַבְּרִית אֲשֶׁר בֵינֵיכֶם / הַקָּהָל חֻקָּה אַחַת לָכֶם / כִּי כָל אֶחָד יַשְׁחִית רֵעוֹ / הַגָּדוֹל מֵחֲבֵרוֹ בּוֹלְעוֹ / וּבְשֶׁצֶף קֶצֶף אוֹתוֹ יְשַׁסֵּף / אִם אֶת כָּל דְּגֵי הַיָּם יֵאָסֵף / לֵאמֹר לִי: עָלֵינוּ מָלְכָה / לֹא אֵצֵא מִן הָמַּעֲרָכָה / כַּאֲשֶׁר כָּל הַיּוֹם תִּפְחֲדוּ וְתִרְהוּ / וְנִלְחֲמוּ אִישׁ בְּאָחִיו וְרֵעֵהוּ / וְדַרְכָּם לְנֶגְדָּם יָרַט / מְמֻשָּׁךְ וּמְמֹרָט / וְעַל זֶה שָׁנוּ תָעוּ / דֶּרֶךְ שָׁלוֹם לֹא יָדָעוּ / שִׁמְעוּ נָא אֶת טַעַם מִלָּי / הַטּוּ אָזְנְכֶם וּלְכוּ אֵלָי / צְאוּ מִשָּׁם וּבוֹאוּ הֲלוֹם / בְּשִׂמְחָה תֵצֵאוּ וּבְשָׁלוֹם / וְאִם תִּשְׁמְעוּ לַעֲצָתִי / וּבֵרַכְתֶּם גַּם אֹתִי / צְאוּ מִן הַיָּם לַיַּבָּשָׁה / וְיַחְדָּו אֶת הָאָרֶץ נִירָשָׁה / וְאָז שָׁלוֹם לָכֶם יִרֶב / וְלֹא יִשָּׂא גּוֹי אֶל גּוֹי חֶרֶב / כִּי אֵין בָּנוּ פּוֹרֵץ פָּרֶץ / נָחָה שָׁקְטָה כָּל הָאָרֶץ / הֵן אִתָּנוּ מְשׂוֹשׂ פְּרָאִים / לְהַחֲיוֹת לֵב נִדְכָּאִים / וְכָל שֹׁכְנֵי אֶרֶץ בְּחֶדְוָתָם / יְרַנְנוּ עַל מִשְׁכְּבוֹתָם / יוֹמָם וָלַיְלָה לֹא יִשְׁבֹּתוּ / לֹא יָרֵעוּ וְלֹא יַשְׁחִיתוּ / וַיַּעַן הָאֶחָד: אִם מָלֹךְ תִּמְלֹךְ עָלֵינוּ / הַאִם תִּשְׁפֹּת שָׁלוֹם לָנוּ / הֵן בִּהְיוֹתֵנוּ בְמֵי מְנוּחוֹת / וּסְבִיבֵנוּ הַמִּשְׁפָּחוֹת / בַּשָּׁלוֹם שׁוֹדֵד יְבוֹאֶנּוּ / יַעֲלֶה הַכּוֹרֵת עָלֵינוּ / וּמִמְּצוּדוֹת בְּנֵי אָדָם אֲנַחְנוּ נְמוֹגִים / וְדִיגוּנוּ רַבִּים דַּיָּגִים / וְצָדוּנוּ צַיָּדִים רַבִּים / וְנֶהֱמֶה כֻלָּנוּ כַדֻּבִּים / וְאִם יָדַעְתָּ מְקוֹם חֲנוֹתֵנוּ / גַּם אַתָּה חֻלֵּיתָ כָּמוֹנוּ / אִם אַתָּה בְעָרְמָתְךָ תֹאכַל שַׁד / כַּהֲתִמְךָ שׁוֹדֵד תּוּשַׁד / כִּי פֶתַע יָקוּמוּ נֹשְׁכֶיךָ / וְלֹא תַצְלִיחַ אֶת דְּרָכֶיךָ / וְאֵיךְ אַתָּה רֶשֶׁת מִרְמָה עָלֵינוּ שֹׁטֵחַ / וּבְאֶרֶץ שָׁלוֹם אַתָּה בֹטֵחַ / הֲלֹא לְעוֹף הַשָּׁמַיִם וְלִדְגֵי הַיָּם וּלְחַיַּת הַשָּׂדֶה / לְכֻלָּם יֵשׁ אֹרֵב וְצֹדֶה / כִּי גַּם בְּנֵי אָדָם נִצִּים מִקִּנְאָה אֲשֶׁר בֵּינֵיהֶם / אֲבָל גָּבֹהַּ מֵעַל גָּבֹהַּ שֹׁמֵר וּגְבֹהִים עֲלֵיהֶם.

וְהַמָּשָׁל לְהִזָּהֵר מִן הַיּוֹעֲצִים לְפִי דַרְכָּם / וּלְהִשָּׁמֵר מִמֶּתֶק חִכָּם / כִּי בְּפֶשַׁע הָאָרֶץ רַבּוּ חַנְפֵי לֵב / וּפְנֵי הַדּוֹר פְּנֵי כֶלֶב / וּלְהִזָּהֵר כָּל אִישׁ מִמּוֹקְשׁוֹ / הֶחָכָם עֵינָיו בְּרֹאשׁוֹ.

וָאֶשָּׂא מְשָׁלִי:


פְּנֵי הַדּוֹר פְּנֵי כֶלֶב / וְדִבְרֵיהֶם בְּלֵב וָלֵב

כְּמוֹ חֶמְאָה דְבַר פִּיהֶם / וְהַמַּחְשָׁב כְּמוֹ חֵלֶב

וּמָצוּ אֶת שְׁדֵי גִילִי / כְּמוֹ יוֹנֵק בְּעֵת חוֹלֵב

מְרַגְלִים הֵם, אֲבָל אֵין בָּם / יְהוֹשֻׁעַ וְלֹא כָלֵב.
