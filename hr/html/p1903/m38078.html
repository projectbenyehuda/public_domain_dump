משל יז: צִרְעָה וּנְמָלָה

הָסֵר מְסוּכַת הָעַצְלוּת / וְלֹא יִהְיוּ עֵינֶיךָ רוֹאוֹת וְכָלוֹת


צִרְעָה בָאָה אֶל נְמָלָה / לִשְׁאֹל מִמֶּנָּה מַאֲכָלָהּ / כִּי חֹסֶר גָּדוֹל יֵשׁ לָהּ / מִמַּאֲכָל נֶגֶד הַחֹרֶף / וְהַנְּמָלָה הָפְכָה עֹרֶף / וּבָזָה מַאֲמָרָהּ וְלַעֲגָה / וַתֹּאמַר: אַחֲרִיתֵךְ תּוּגָה / וּרְאִי נָא מִמִּי נָעַמְתְּ / כִּי בִימֵי הַקָּצִיר נִרְדַּמְתְּ / וְלַחֹרֶף לֹא הֵכַנְתְּ מַאֲכָל / כַּמֶּה יִהְיֶה הַלֵּב סָכָל / הַנּוֹתֵן לָךְ אֲשֶׁר אָצַר / דְּעִי כִּי יָדִי לָךְ תִּקְצָר / כִּי מֶה עָשִׂית עֵת הַקָּצִיר / נִרְאָה דֶשֶׁא גָּלָה חָצִיר / וְכָל הַיְקוּם לִלְקֹט נָטוּ / אִישׁ לְפִי אָכְלוֹ לָקָטוּ / וּלְהָכִין לַחֹרֶף כָּל נָחוּץ / אָמַר עָצֵל: אֲרִי בַחוּץ / כִּי אֲשֶׁר יִרְצֶה חִטִּים לֶאֱצֹר / זְרוֹעוֹ שִׁבֳּלִים יִקְצֹר / כָּמוֹנִי אֲשֶׁר נְשָׂאתִים עַל שִׁכְמִי / וְהֵכַנְתִּי בַּקַּיִץ לַחְמִי / וַתַּעַן הַצִּרְעָה: כָּל עֵת הַזָּמִיר / בְּעֵת יְמַלְּאוּ הָעֲגָלָה עָמִיר / לָמַדְתִּי לָשִׁיר שִׁיר נָעִים / וְהִרְבֵּיתִי עָלַי רֵעִים / וְהִבְרַחְתִּי מִלִּבִּי יְגוֹנַי / כִּי יָשְׁרוּ הַשִּׁירִים בְּעֵינָי / וְהֵם אָמְרוּ כִּי קוֹלִי עָרֵב / וְעַתָּה אֲכָלַנִי חֹרֶב / וְקֹר וְחֹרֶף וְקֶרַח / הָיוּ עָלַי לַטֹּרַח / וַתֹּאמֶר הַנְּמָלָה: הַגֹּרֶן וְהַיֶּקֶב / לֹא יוֹשִׁיעוּךְ עַתָּה עֵקֶב / לִמַּדְתְּ בַּשִּׁירִים קוֹלֵךְ / וְלֹא חָשַׁבְתְּ עַל מַאֲכָלֵךְ / וְאִם אַתְּ מְקֻנַּנְתְּ בָּאֲרָזִים / וְאִם צַוָּארֵךְ בַּחֲרוּזִים / תֵּלְכִי נָא אֶל בֵּית עָשִׁיר / הֵיטִיבִי נַגֵּן הַרְבִּי שִׁיר / לְמַלֵּא נַפְשֵׁךְ כִּי תִרְעָב / הֵן מַאֲמָרֵךְ אֵלַי נִתְעָב / לְכִי מֵעָלַי וְהִפָּרְדִי / כִּי תֵעֲבוּךְ כָּל מְתֵי סוֹדִי / בַּאֲשֶׁר אֲנִי שִׁפְלַת קוֹמָה / וּבִי אֵין כֹּחַ וְעָצְמָה / מִלֵּאתִי כָל טוּב אָהֳלִי / אַךְ אִם חָכַמְתִּי חָכַמְתִּי לִי / וְאַתְּ לֹא תִּתְפַּרְנְסִי בַשִּׁירִים / בְּבֹא הַקָּרָה מִמְּזָרִים / כִּי קְרָעִים תַּלְבִּישׁ נוּמָה / וְעַצְלָה תַפִּיל תַּרְדֵּמָה.


וְהַמָּשָׁל לְמִתְעַצֵּל לְהַמְצִיא טֶרֶף בֵּיתוֹ / הֲיִשְׁמְעוּ אֲחֵרִים צַעֲקָתוֹ / יָשֵׁן בַּקָּצִיר וְנִרְדָּם / לָאִישׁ הַהוּא יֵחָשֵׁב דָּם.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


בְּנִי, צְלַח רְכַב עֲבֹר / רְחוֹב וְעִיר בְּחֹם, צְבֹר

לְעֵת סְתָיו כְּסוּת וְסוּת / וְשׁוֹר וְשֶׂה וְאַךְ בְּבֹר

וְעֵין יְמִין אֱוִיל נְקֹר / וּפִיו בְּלֹם וְשֵׁן שְׁבֹר.
