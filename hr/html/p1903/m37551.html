משל ט: כַּלְבָּה הָרָה וְכַלְבָּה

הַמַּשְׁלִיט זוּלָתוֹ עָלָיו / יַרְכִּיבֵהוּ עַד יִהְיֶה בְעָלָיו

כַּלְבָּה אַחַת בַּיִת לָלוּן חוֹמֶדֶת / כִּי קָרְבוּ יָמֶיהָ לָלֶדֶת / וְהַסְּתָו הוֹלֵךְ וְקָרֵב / וַתָּבֹא אֶל רְעוּתָה בָּעֶרֶב / אֲשֶׁר בַּיִת וָחֶדֶר בִּגְבוּלָהּ / וַתֵּבְךְּ וַתִּתְחַנֵּן לָהּ / לִשְׁמֹעַ אֶת צַעֲקָתָהּ / וְלֶאֱסֹף אוֹתָהּ הַבַּיְתָה / עַד אֲשֶׁר תִּפָּלֵט מִן הַתְּלָאָה / וְתָרִיק בֶּטֶן הַמְּלֵאָה / וַתַּחְמֹל עָלֶיהָ וַתֹּאמַר: הִנֵּה בֵּיתִי לְפָנַיִךְ / עֲשִׂי כַטּוֹב בְּעֵינַיִךְ / וְצֶאֱצָאֵי מֵעַיִךְ בְּקַטְנוּתָם / שְׁבִי עַד גָּמְלֵךְ אוֹתָם / וְהַכַּלְבָּה שָׁם נָחָה וְשָׁקְטָה / עַד אֲשֶׁר יָלְדָה וְהִמְלִיטָה / בִּדְמוּתָהּ כְּצַלְמָהּ כְּלָבִים / וּבְעֵינֵי גְבֶרֶת הַבַּיִת לֹא טוֹבִים / כִּי לַשָּׁמַיִם עָלְתָה צַעֲקָתָם / וְלֹא זֹרוּ מִתַּאֲוָתָם / וַיִּרְבּוּ וַיַּעַצְמוּ / בָּשָׂר אָכְלוּ וְעֶצֶם עָצְמוּ / שָׂבְעוּ וַיָּרָם לִבָּם / וּבְקִרְבָּם יָשִׂימוּ אָרְבָּם / וְלֹא יַשְׁאִירוּ בַבַּיִת מִחְיָה / וַתֹּאמֶר הַגְּבֶרֶת אֶל הַנָּכְרִיָּה: רַעְיָה / צְאִי מִבֵּיתִי וּמֵעָלַי סוּרִי / וְגוּרִי בַאֲשֶׁר תָּגוּרִי / וְאִם אֲנִי קִרְבָתֵךְ אוֹהֶבֶת / לֹא נוּכַל יַחְדָּו לָשֶׁבֶת / כִּי אֵין לִי עִם יְלָדַיִךְ נֹפֶשׁ / וְהַכְּלָבִים עַזֵּי נֶפֶשׁ / וְתָמִיד מַכְעִיסִים אוֹתִי / וּלְזַעֲקָתָם נִגְזְלָה שְׁנָתִי / אוֹיָה לִי כִּי גָרוּ מֶשֶׁךְ עִמִּי / שֹׁתֵי מֵימִי אֹכְלֵי לַחְמִי / וַתִּקְצַר נַפְשִׁי בַעֲמָלָם / וְאֵין לִי מְנוּחָה בָעוֹלָם / וַתַּעַן לָהּ בִּשְׂפַת חֲלָקוֹת הַנָּכְרִיָּה / גְּבִרְתִּי הַכְּבֻדָּה וְהַנְּקִיָּה / תִּנָּתֵן לִי נַפְשִׁי בִשְׁאֵלָתִי / וְעַמִּי בְּבַקָּשָׁתִי / אִם קִרְבָתִי תִּשְׂנְאִי אֶצְלָךְ / שַׁלַּח תְּשַׁלַּח אֶת הָאֵם וְאֶת הַבָּנִים תִּקַּח לָךְ / כִּי אָמְרוּ לִי צֶאֱצָאַי וַהֲמוֹנַי / שׁוּבִי אֶל גְּבִרְתֵּךְ וְהִתְעַנִּי / עוֹד לֹא יֶחֱרַץ כֶּלֶב בְּאָזְנַיִךְ לָשׁוֹן / וְתֵיטִיבִי לִי חַסְדֵּךְ הָאַחֲרוֹן מִן הָרִאשׁוֹן / וְהַנִּיחִי לִי עַד יַעֲבֹר הַחֹרֶף וְהַקֹּר / וּבְעֵת הַזָּמִיר בַּיִת אַחֵר נַחְקֹר / וְאַל תּוֹרִידִי שֵׂיבָתִי בְדָם שְׁאוֹלָה / פֶּן יֹאכְלֵנוּ חֹרֶב בַּיּוֹם וְקֶרַח בַּלַּיְלָה / אִם לְעֵת כָּזֹאת בָּנַי מִבְּלִי אֹהֶל יִשָּׁאֲרוּ / וְכַלְבַּת הַבַּיִת עָלֶיהָ רַחֲמֶיהָ נִכְמָרוּ / וַתֹּאמַר: שֶׁלִּי שֶׁלָּךְ וּבֵיתִי בֵיתֵךְ / כִּשְׁאֵלָתֵךְ וּבַקָּשָׁתֵךְ / וּבְבֹא עֵת הַזָּמִיר / אֶת לִבָּם שָׂמוּ שָׁמִיר / הַנָּכְרִיָּה וּבָנֶיהָ / וְהַגְּבֶרֶת אוֹמֶרֶת בְּאָזְנֶיהָ / הֵימִינִי אוֹ הַשְׂמִילִי / צְאִי אַתְּ וּבָנַיִךְ מֵאָהֳלִי / לֹא תָקוּם פַּעֲמַיִם צָרָה / לָמָּה אַתְּ מִתְנַכְּרָה / לְשַׁלֵּם רָעָה תַּחַת טוֹבָה / וְהַנָּכְרִיָּה בָּזָה אֶת נִיבָהּ / וֵַתִּוָּעֵץ בְִּלִבָּהּ וְשִׂכְלָהּ / לְמַעַן הַרְחִיקָהּ מֵעַל גְּבוּלָהּ / וַתֹּאמֶר לָהּ: הַבַּיִת שֶׁלִּי / צְאִי אַתְּ מִמִּגְרָשִׁי וּגְבוּלִי / וְהִפָּרְדִי נָא וְהַנִּלְוִים עָלַיִךְ מֵעִמִּי / מַהֲרוּ צְאוּ מִתּוֹךְ עַמִּי / וַיַּבְהִילוּ לְהוֹצִיא אוֹתָהּ / וַתֵּלֶךְ נִכְאֶבֶת מִבֵּיתָהּ / וּבְנֵי הַנָּכְרִיָּה יִירָשׁוּם / וַיָּבֹאוּ הָרֹעִים וַיְגָרְשׁוּם.

וְהַמָּשָׁל בְּאִישׁ נָעִים וְעָרֵב / אֲשֶׁר בַּעַל תַּהְפּוּכוֹת יְקָרֵב / בְּקֶרֶב בֵּיתוֹ עֹשֶׂה רְמִיָּה / וּמַעֲרִים לְהוֹנוֹתוֹ בִצְדִיָּה / וּבְעֵת הָלֹךְ וָשׁוֹב / דִּבְרֵי מִרְמוֹת יַחֲשֹׁב / וּלְהַתְעוֹת אֲדֹנָיו יֵאָנַח / כַּאֲשֶׁר מִשָּׁלוֹם נַפְשׁוֹ תִזְנַח / וְעֵינָיו יִתְּנוּ מֵימָיו / וְהָאָדוֹן עָלָיו יִכָּמְרוּ רַחֲמָיו / לְמַלְּאוֹת כָּל תַּאֲוָתוֹ / וְלִהְיוֹת שׂוֹרֵר בְּבֵיתוֹ / וְהָאֶחָד אֲשֶׁר בָּא לָגוּר יַעֲרִים / לִי הַבַּיִת וְלִי הַחֲדָרִים / וְאַתָּה בְּבֵיתִי אוֹרֵחַ / וְזֶה בְּקוֹלוֹ מַר צוֹרֵחַ / וְאִם יוֹם שָׂמַח בַּחֲבוּרָתוֹ / שְׁנַיִם יֶאֱבַל לְמָחֳרָתוֹ / וְאָמַר הֶחָכָם בְּחָכְמָתוֹ / יֵשׁ צַדִּיק אֹבֵד בְּצִדְקָתוֹ.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


בָּתֵי אֱמוּנִים וְתִלָּם / שָׁמְמוּ וְאֵין אִישׁ בְּכֻלָּם

דֹּבֵר נְכֹחָה וְיֹשֶׁר / הַזְּמָן הֲדִמָּם וְכִלָּם

חֹנֶף וְשֶׁקֶר יְחַיֶּה / תֹּם מֵת וְנִסְתָּר וְנֶעְלָם

יָרְדוּ שְׁאוֹלָה יְשָׁרִים / עָלוּ נְלוֹזִים בְּסֻלָּם

מָתוֹק כְּמַר, אוֹר כְּחֹשֶׁךְ / מַיִם סְרוּחִים כְּטַלָּם

נִקְלֶה כְּנִכְבָּד לְכַבֵּד / שִׁמְשָׁם לְהָאִיר בְּצִלָּם

אִם אִישׁ עֲמִיתוֹ יְקָרֵב / לוֹ רַע עֲבוּר טוֹב יְשֻׁלָּם

וַאְשֶׁר יְגָרֶה בְּחָנֵף / מִי הִקְשָׁה לוֹ וְיִשְׁלָם

מֵשִׁיב בְּטוֹב רָע יְעֻוַּר / מֵרְאוֹת מְאוֹרוֹת בְּהִלָּם.
