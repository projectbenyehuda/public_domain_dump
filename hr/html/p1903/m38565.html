אַריֵה וּבְנוֹ

טוֹב לְהִזָּהֵר מֵהָרְשָׁעִים עַל כָּל פָּנִים / כִּי מַעֲשֵׂי אָבוֹת יַעֲשׂוּ בָנִים

אַרְיֵה מֶלֶךְ בַּחַיּוֹת / שֹׁכֵן בִּמְעוֹנוֹת אֲרָיוֹת / לַעֲשׂוֹת מִשְׁפַּט עַמּוֹ / בְּנַפְשׁוֹ יָבִיא לַחְמוֹ / כַּאֲשֶׁר עָשׂוּ אֲבוֹתָיו / וּמְחַנֵּק לְלִבְאֹתָיו / וּבְחָלְיוֹ אֲשֶׁר יָמוּת בּוֹ / וְהִנֵּה נַפְשׁוֹ בָחֲלָה בוֹ / וַיִּקְרָא לִבְנוֹ בְכוֹרוֹ / אֲשֶׁר הָיָה בְעִירוֹ / הַמֹּשֵׁל בְּכָל אֲשֶׁר לוֹ / וַיִּזָּעֵק הַכְּפִיר לְקוֹלוֹ / וַיִּקְרָא אֵלָיו: בְּנִי / וַיֹּאמֶר לוֹ: הִנֵּנִי / וַיִּקֹּד וַיִּשְׁתַּחֲוֶה לְמוּלוֹ / וַיְצַוֵּהוּ וַיֹּאמֶר לוֹ / בְּנִי, שְׁמֹר מִצְוָה וֶהְיֵה / מְשֹׁל בְּשֵׂיְךָ וְהֱיֵה / גִּבּוֹר וּבֶן חַיִל / רְאֵה כֶּבֶשׂ וְאַיִל / לְכַף רַגְלְךָ מִדְרָךְ / בְּעוֹדְךָ נַעַר וָרָךְ / וּכְשֶׁיִּגְדַּל כֹּחֲךָ / לֹא תִקְצַר יָדְךָ וְרוּחֲךָ / וְכָרַעְתָּ וְרָבַצְתָּ כְּלָבִיא / לָצוּד צַיִד לְהָבִיא / תֵּקַל אֲלֵיהֶם בְּרֶגֶל / אֶל מַעֲשֵׂה הָעֵגֶל / וְאֶל עַיִר בֶּן אֲתוֹנוֹת / לֹא תַעֲזֹב חֲנוֹת / כִּי בְאַף תִּקְפֹּץ / כַּאֲשֶׁר תַּחְפֹּץ / וּמִסֹּבֶךְ תַּעַל כְּאַרְיֵה / וְכָל אֲשֶׁר יַמְרֶה אֶת פִּיךָ לֹא יִחְיֶה / אָז תִּהְיֶה מֹלֵךְ / עַל כָּל אַרְבַּע הֹלֵךְ / תִּשְׂמַח וְתָגֵל לִמְשַׁלֵּחַ רֶגֶל הַשּׁוֹר וְהַחֲמוֹר / וְכָל בְּהֵמָה לִשְׁמוֹר / דֶּרֶךְ מְבֹא הַשֶּׁמֶשׁ מוֹצָאָהּ / תָּמִיד לִהְיוֹתָהּ לְאָכְלָה / וְלֹא לִסְחוֹרָה רַק לַאֲחֻזַּת נַחֲלָה / וְכַאֲשֶׁר תָּרִיד וְרָבַצְתָּ לֹא יִבָּצֵר מִמְּךָ מְזִמָּה / מִכָּל חַיָּה וּבְהֵמָה / מִן הַטְּהוֹרָה / לְךָ אֲסוּרָה / הָאַרְנֶבֶת וְהַשָּׁפָן / תִּשְׂחַק לְשֹׁד וּלְכָפָן / וְגַם מִן הַחֲזִיר / אַל תֵּעָשֶׂה נָזִיר / כִּי בְשָׂרוֹ שָׁמֵן / וְעָנִיתָ וְאָמַרְתָּ: אָמֵן / וּמִמְּךָ לֹא יֶחְסָר / וּבְכָל אַוַּת נַפְשְׁךָ תֹּאכַל בָּשָׂר / וּמִכָּל הַבְּהֵמָה / תּוֹצִיא בְיָד רָמָה / וּבְכָל חַיָּה / יָדְךָ הוֹיָה / וְתַשְׁחִית אֶת הוֹדָם / אַךְ הִשָּׁמֵר מִבְּנֵי אָדָם / כִּי רַבּוֹת תַּחְבֻּלוֹתָיו / יִלְכָדְךָ בִּשְׁחִיתוֹתָיו / וְאִם עָצַמְתָּ מִמֶּנּוּ / בְּמִרְמָה לֹא תַצִּילֶנּוּ / כִּי יְמַלֵּא יָדָיו בְּקֶשֶׁת / אוֹ מִכְבָּר מַעֲשֵׂה רֶשֶׁת / לָשִׂים רַגְלְךָ בַּסַּד / פִּנַּת יִקְרַת מוּסָד / פַּח וָפַחַד וָּפָחַת / תֵּרֵד לִבְאֵר שָׁחַת / אַתָּה לֹא תוּכַל לוֹ / כִּי אֵין עַל עָפָר מָשְׁלוֹ / וּמִי לוֹ דָמָה / כִּי כָל הַבְּהֵמָה / כֻּלָּם לָאָדָם עֲבָדִים / גַּם הַסּוּסִים וְהַפְּרָדִים / חָמַד אוֹתָם לוֹ לְשִׁבְתּוֹ / יָשִׂים לְמֶרְכַּבְתּוֹ / וְיוֹם יוֹם יַעֲמֹס עַל הַחֲמוֹרִים / גַּם יִטְבַּח טֶבַח וְהָכֵן וְיַחֲרֹשׁ בַּבְּקָרִים / חָזָק בְּגִזַּת הַצֹּאן / שׁוֹר וַאֲרִי וְצֹאן / נַפְשָׁם עֲלֵיהֶם תֶּאֱבָל / וְשֶׂה לַטֶּבַח יוּבָל / עַל כֵּן אוֹתְךָ צִוִּיתִי / לִשְׁמֹעַ לַעֲצָתִי / וְלִשְׁמֹר מִצְוָתִי / פֶּן תַּשְׁחִית אַחֲרֵ מוֹתִי / אִם תָּפֵר דִּבְרֵי אֱמֶת / וְהִנֵּה אָנֹכִי מֵת / כְּדֶרֶךְ כָּל הָאָרֶץ / וּפָרַצְתָּ עָלֶיךָ פָּרֶץ / אַל תַּעֲרֹץ וְאַל תֵּחָת / אַךְ בְּלִי רֶדֶת שַׁחַת / תְּהִי נַפְשְׁךָ שֹׁמֵר / הַשְׁקֵט וְהִשָּׁמֵר / מֵחֹפֵר גּוּמָץ / וַחֲזַק וֶאֱמָץ / וְאַחֲרֵי כֵן צִוָּה הַמֶּלֶךְ אֶת בְּנוֹ / לְהוֹשִׁיבוֹ עַל כַּנּוֹ / לְעֵינֵי כָל שָׂרָיו / לְמַעַן יִהְיוּ בְעֹזְרָיו / וְאַחֲרֵי אֲשֶׁר צִוָּהוּ / נָפַל עָלָיו וַיִּשָּׁקֵהוּ / וַיְהִי אַחֲרֵי דַבְּרוֹ יָצְאָה רוּחוֹ / וְכָהֲתָה עֵינוֹ וְנָס לֵחוֹ / וְלִבּוֹ עָלָיו נִשְׁפָּךְ / וּלְמַשְׁחִית הוֹדוֹ נֶהְפָּךְ / וַיֵּאָסֵף אֶל עַמָּיו / וְהַבֵּן נִכְמְרוּ עָלָיו רַחֲמָיו / וַיִּצְעַק: אָבִי אָבִי / רֹכֵב אַיָּל וּצְבִי / וְהוֹד הַמְּלוּכָה / רֹאשׁ הַמַּעֲרָכָה / כִּי תִּסָּגֵר בְּאֶרֶץ צִיָּה / אוֹיָה לָנוּ אוֹיָה / וַיִּשָּׂא קוֹלוֹ וַיֵּבְךְּ עָלָיו וַיִּשַּׁק לוֹ / וּלְקוֹלוֹ חָרְדוּ מִשְׁפַּחְתּוֹ / וְכָל עֹזְרָיו אִתּוֹ / וַיָּבֹאוּ כֻלָּם לְסָפְדוֹ / הוֹי אָדוֹן וְהוֹי הוֹדוֹ / אִם יְשֻׁנֶּה עֹז פָּנָיו / מַה יַּעֲשׂוּ יְתוֹמִים בָּנָיו / אַחֲרָיו כְּעִוְרִים נְגַשֵּׁשׁ / וּמַדּוּעַ בֹּשֵׁשׁ רִכְבּוֹ לָבוֹא / וַיָּנָס בְּמַחֲבֹא / לָמָּה מֵעָלֵינוּ סָר צִלֵּנוּ / לוּ יוֹסֵף יִשְׂטְמֵנוּ / וְהָשֵׁב יָשִׁיב לָנוּ / אֶת כָּל הָרָעָה אֲשֶׁר גָּמַלְנוּ / אוֹי נָא לָנוּ כִּי חָטָאנוּ / גָּוַעְנוּ אָבַדְנוּ כֻּלָּנוּ אָבַדְנוּ / אִם נָפַל מַלְכֵּנוּ / אֲשֶׁר יִתְהַלֵּל בָּאֻמִּים / וַיִּסְפְּדוּ לוֹ הָעַמִּים / וּבְנוֹ לֹא קִבֵּל תַּנְחוּמִים / וַיַּעַשׂ לְאָבִיו אֵבֶל שִׁבְעַת יָמִים / וַיְהִי בַּעֲבֹר הָאֵבֶל / מִכֹּשֶׁל כֹּחַ הַסֵּבֶל / כִּי עוֹד לֹא יַכִּירוּהוּ / וַיִּשְׂאוּ הַמֵּת וַיִּקְבְּרוּהוּ / בְּסֶלַע בֵּיתוֹ הָרָמָה / וַיִּקְרְאוּ שְׁמָהּ בָּמָה / כִּי הִיא לְמַצֵּבָה תִהְיֶה / לִרְאוֹת אֶת מַפֶּלֶת הָאַרְיֵה / וַיְהִי אַחֲרֵי קָבְרָם / כְּבוֹדָם וִיקָרָם / וַיָּשָׁב הָעָם וַיֵּלֶךְ / וַיָּשִׁיבוּ בֶּן הַמֶּלֶךְ / וַיְנַחֲמוּ אֹתוֹ בְּבֵיתוֹ / וַיִּתְקְעוּ כָף לְאַהֲבָה וּלְעָבְדָה אֹתוֹ / וּכְשָׁב הַמֶּלֶךְ לְבֵיתוֹ / יָשַׁב עַל כִּסֵּא מַלְכוּתוֹ / כַּדָּת וְכַהֲלָכָה / וַתִּצְלַח בְּיָדוֹ הַמְּלוּכָה / וַיִּפֶן אָנֶה וְאָנָה בַהֲמוֹנָיו / וְכָל הַחַיּוֹת לֹא יַעַמְדוּ לְפָנָיו / יְרוֹפְפוּ וְיִתְמְהוּ מִגַּעְרָתוֹ / וּכְטוֹב לִֵב הַמֶּלֶךְ בְּשִׂמְחָתוֹ / וַיִּגְבַּהּ מְאֹד רוּחוֹ / וַיִּפְתַּח אֶת יָדָיו / וַיַּעַשׂ מִשְׁתֶּה לְכָל עֲבָדָיו / וְהַשְּׁתִיָּה כַדָּת אֵין אֹנֵס / וַיִּשְׂאוּ עַל הֶהָרִים נֵס / לְהַשְׂבִּיעַ כָּל חַי רָצוֹן / חֶמְאַת בָּקָר וַחֲלֵב צֹאן / וַיְכַסּוּ פָנִים בְּחֶלְבָּם / עַד כִּי הָיָה לִבָּם / אֶל מַלְכָּם נִפְתֶּה / וַיְהִי בִימֵי הַמִּשְׁתֶּה / וַיִּקְרָא הַמֶּלֶךְ לִמְשָׁרְתָיו / וַיִּפְקֹד שָׂרֵי חֵילוֹתָיו / לִשְׂמֹחַ הַבַּחוּרִים / לִבְטֹחַ בַּגִּבּוֹרִים / לָשִׂים שָׂרֵי צְבָאוֹת / שָׂרֵי אֲלָפִים וְשָׂרֵי מֵאוֹת / כֻּלָּם נִקְבְּצוּ יַחַד / וַיִּהְיוּ לְקֶשֶׁר אֶחָד / דֻּבִּים וּנְמֵרִים / אֲרָיוֹת וַחֲזִירִים / סוּסִים וּפִילִים / הָרֶכֶשׁ וְהַגְּמַלִּים / וַיֹּאחֲזוּ שֻׁעָלִים / לָבֹא יַחַד בְּמַקְהֵלִים / שֶׁנְהַבִּים וְקוֹפִים / לְאַלְפֵי אֲלָפִים / עַד אֵין מָקוֹם לַעֲמֹד / וַיְהִי הַמַּחֲנֶה כָּבֵד מְאֹד / וְהַמֶּלֶךְ לִרְאוֹת חִכָּה / וַיַּעַשׂ שָׁם סֻכָּה / וַיִּקְרָא לְמִשְׁפַּחְתּוֹ / מְשָׁרְתָיו וּבְנֵי בֵּיתוֹ / וַיֹּאמֶר לָהֶם: רְאוּ / אֵיזֶה מֵהֶם לְפָנַי תַּרְאוּ / בַּחֲרוּ לִי הַגִּבּוֹרִים / לִהְיוֹת בַּצָּבָא שָׂרִים / לְהַשְׁחִית עִיר וָפֶלֶךְ וַיֹּאמְרוּ: הָאִישׁ נָסַע מִזֶּה / בַּיַּעַר לַחְטֹב עֵץ / כִּי בַעֲצָתוֹ יִוָּעֵץ / מִשְׁפַּחְתּוֹ רָכְבוּ עִמּוֹ / עוֹדָם מְדַבְּרִים עִמּוֹ / וְהִנֵּה הַשֻּׁעָל לִקְרָאתוֹ / וַיִּשְׂמַח הַמֶּלֶךְ לִרְאוֹתוֹ / וַיֹּאמַר: זֶה מְבַשֵּׂר / וְאָנֹכִי לֹא אֲפַשֵּׁר / וּבְהַגִּיעַ הַשֻּׁעָל אֵלָיו / וַיִּקֹּד לְרַגְלָיו / וַיִּשְׁאַל: הַשָּׁלוֹם אֲדֹנִי / וַיַּעַן: שָׁלוֹם בְּנִי / וַיֹּאמַר: הַמֶּלֶךְ, יָצָאתִי קַן־צִפֹּר וְאֶפְרֹחָיו אָרַבְתִּי / לְבַעֲלֵי כָנָף / לֹא אַשְׁאִיר עָנָף / וַיַּעַן הַמֶּלֶךְ וַיֹּאמַר: וְהָאָדָם לֹא נִשְׁמָר / שָׁם אֵינֶנּוּ / אוֹ חָזַקְתָּ מִמֶּנּוּ / וַיֹּאמַר: אָרַבְתִּי / וְלֹא בְחָזְקָה לָקַחְתִּי / כִּי מִי אֲנִי וּמִי עַמִּי / וַיֹּאמֶר הַמֶּלֶךְ: שֵׁב עִמִּי / עַד אֶמְצָאֶנּוּ / אִם לֹא בַחֲמָתִי אַכֶּנּוּ / פַּעַם אַחַת וְלֹא אֶשְׁנֶה / וְהַמָּקוֹם מְקוֹם מִקְנֶה / תִּרְעֶה בְמִרְעֶה שָׁמֵן / וַיֹּאמֶר הַשֻּׁעָל: אָמֵן / כִּדְבָרְךָ אֲדֹנִי הַמֶּלֶךְ / אֲנִי אַחֲרֶיךָ אֵלֵךְ / אַךְ הִשָּׁמֵר מִמּוֹקֵשׁ / הָאִישׁ אֲשֶׁר אַתָּה מְבַקֵּשׁ / בְּסִבְכֵי יַעַר תִּמְצָא / כִּי עַתָּה מִבֵּיתוֹ יָצָא / וְאַתָּה לְהִלָּחֵם תִּקְרָב / לִשְׁלֹל שָׁלָל רָב / וַיֵּלֶךְ הָאַרְיֵה בְּיָד רָמָה / וְהַשֻּׁעָל נִשְׁאָר שָׁמָּה.


וְהַמָּשָׁל לְהִזָּהֵר מִן הָרְשָׁעִים עַל כָּל פָּנִים / כִּי מַעֲשֵׂה אָבוֹת יַעֲשׂוּ בָנִים / כָּאַרְיֵה אֲשֶׁר הָרַג בָּקָר וְשָׁחַט צֹאן / הֵן בְּטֶבַע הֵן בְּרָצוֹן.
