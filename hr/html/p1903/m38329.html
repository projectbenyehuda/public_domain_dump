משל פ: אִשָּׁה וּבַעֲלָהּ וּפָרַָשׁ וּמִשְׁנֶה

מְעַט מֵהַחַיִּים יַעֲלוּ עַל לִבְבֵיהֶם / הַנֶּאֱהָבִים וְהַנְּעִימִים בְּחַיֵּיהֶם


אִשָּׁה סֹפֶקֶת כַּפֶּיהָ / יִזְּלוּ מַיִם עַפְעַפֶּיהָ / וּבְמַר נַפְשָׁהּ בָּדָד יָשְׁבָה / מְתֹפֶפֶת עַל לְבָבָהּ / עַל בַּעֲלָהּ אֲשֶׁר מֵת וְנִקְבָּר / וְהִנֵּה אַחַר הַמִּדְבָּר / פָּרָשׁ הָיָה תָלוּי עַל עֵץ / וּמִשְׁנֶה לַמֶּלֶךְ וְיוֹעֵץ / מֵעִיר אֶל עִיר מִתְהַלֵּךְ / וְיִזְעַק מִטַּעַם הַמֶּלֶךְ / כִּי כָל אֲשֶׁר יַפִּיל הַתָּלוּי / גּוּפוֹ יִהְיֶה בָּאֵשׁ קָלוּי / וַיַּרְא פָּרָשׁ אַחֵר אָחִיו הַנִּתְלָה / כַּאֲשֶׁר הֹנָה מִן הַמְסִלָּה / זֶה הַדָּבָר מְאֹד הִכְלִימוֹ / אָחִיו הוּא מֵאָבִיו וּמֵאִמּוֹ / וְלֹא יְפַלֵּס נָתִיב לְאַפּוֹ / וַיָּשֶׂם בַּלַּיְלָה אֶת נַפְשׁוֹ בְּכַפּוֹ / וַיְנַתֵּק אֶת מוֹסְרוֹתָיו / וַיִּקְבְּרֵהוּ עִם אֲבוֹתָיו / וַיִּירָא מְאֹד לְנַפְשׁוֹ / פֶּן יָסִירוּ מֵעָלָיו רֹאשׁוֹ / כִּי מִצְוַת הַמֶּלֶךְ לֹא שָׁמַר / מִפַּחְדּוֹ כָּל בְּשָׂרוֹ סָמַר / וַיִּבְרַח לְהֵחָבֵא בֵּית הָאִשָּׁה / הַמְבַכָּה עַל בַּעְלָהּ בְּמַר נַפְשָׁהּ / וַיְדַבֵּר עַל לִבָּהּ בְּמַאֲמָרִים נְעִימִים / דְּבָרִים טוֹבִים דְּבָרִים נִחֻמִּים / לֵאמֹר: לִי תְהִי לְמָנָה / וְאֵרַשְׂתִּיךְ לִי בֶּאֱמוּנָה / שִׁכְחִי יְגוֹנֵךְ וּמִבַּעְלֵךְ תִּנָּחֲמִי / וּבֹאִי שִׁכְבִי עִמִּי / וְהִיא גַּם הִיא אָמְרָה / מַזָּלְךָ לְפָנֶיךָ הִקְרָה / הָאִשָּׁה אֲשֶׁר עַתָּה נָפְלָה לְגוֹרָלָךְ / שָׁלוֹם תַּעַנְךָ וּפָתְחָה לָךְ / וְלָדַעַת כָּל מַעֲשָׂיו הֱצִיקַתְהוּ / וַיַּגֵּד לָהּ אֶת כָּל אֲשֶׁר קָרָהוּ / כִּי שָׁחָה נַפְשׁוֹ עַד עָפָר / בַּאֲשֶׁר בַלַּיְלָה הַהוּא מִצְוַת הַמֶּלֶךְ הֵפַר / עַל כֵּן יָמוּת בְּלֹא עִתּוֹ / וְלֹא יָשׁוּב עוֹד אֶל מִשְׁפַּחְתּוֹ / וַתֹּאמֶר לוֹ הָאִשָּׁה: אַל תֵּחָת / בַּעְלִי הַנִּקְבָּר נִתְלָה תַחַת / הַתָּלוּי עַל הָעֵץ הַהוּא / בְּטֶרֶם יַכִּיר אִישׁ אֶת רֵעֵהוּ / וּמִי יֵדַע אֲשֶׁר נַעֲשָׂה בַמִּסְתָּרִים / אֵין אֹמֶר וְאֵין דְּבָרִים / עַתָּה תִשְׁמַע לַעֲצָתִי / כִּי עַל כֵּן בָּאתָ בְּצֵל קֹרָתִי / וַתּוֹלִיכֵהוּ חִישׁ אֶל הַקְּבוּרָה / אַהֲבָה מְקַלְקֶלֶת הַשּׁוּרָה / וְהַקָּבוּר מִקְּבוּרָתוֹ הֵסִירוּ / וַיַּעֲשׂוּ כַּאֲשֶׁר דִּבֵּרוּ / וַתֵּלֶךְ וַתֵּתַע בַּמִּדְבָּר / לִתְלוֹת אֶת בַּעְלָהּ הַנִּקְבָּר.


וְהַמָּשָׁל מְעַט מִן הַחַיִּים יַעֲלוּ עַל לִבְבֵיהֶם / הַנֶּאֱהָבִים וְהַנְּעִימִים בְּחַיֵּיהֶם / וִיכַתְּתוּ כָל אַהֲבָתָם טָחוֹן / לְכָל אֲשֶׁר יָמוּת אֵין בִּטָּחוֹן / מֵאֲשֶׁר חִבְּקוּהוּ וַיְנַשְּׁקוּהוּ / כַּמֵּת מִלֵּב יִשְׁכָּחוּהוּ / כִּי לֹא יִתְּנוּ לֵב לְהַזְכִּיר אֶת צִדְקָתוֹ / וְלֹא לִשְׁמֹר אֶת מִצְוָתוֹ / הָאֹמֵר: זְכָר נָא אֶת חַסְדִּי / אַל תִּשְׁקֹר לִי וּלְנִינִי וּלְנֶכְדִּי / מֵת הָאִישׁ וַיְכַסּוּהוּ בֶעָפָר / וְהַחַי אֶת מִצְוָתוֹ הֵפַר.
