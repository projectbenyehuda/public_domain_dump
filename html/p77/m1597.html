לחולי הזמן

כתובת: לאחד מן החברים התחיל להתעסק במלאכת הרפואה

לְחוֹלֵי הַזְּמָן זֶה יוֹם בְּשׂוֹרָה / לְהַצְמִיחַ אֲרוּכָתָם מְהֵרָה

וְיוֹם רָצוֹן לְחַבֵּשׁ נִשְׁבְּרֵי לֵב / וּמַרְגֹּעַ לְכָל רוּחַ שְׁבוּרָה

וּמַרְגֹּעַ לְכָל נֶחְלֶה כְאִלּוּ / אֲרוּכָה עַל פְּנֵי הַיְּקוּם מְזוֹרָה

וְכָל נֶפֶשׁ נְכֵאָה אֶל מְזוֹרָהּ / וְצִירָהּ מָצְאָה אֶת מְזוֹרָהּ

הֲכִי קָם בַּזְּמָן אַבְרָם לְרוֹפֵא / בְּרב בִּינָה הֱכִינָהּ גַּם חֲקָרָהּ

וְהָיְתָה הַתְּבוּנָה לוֹ לְמִקְנָה / מְכוּרָה לוֹ וְלֹא יִמְשֹׁל לְמָכְרָהּ

בְּרֹחַב לֵב יְגַלֶּה הַעֲמֻקּוֹת / וְתוֹפַע עַל פְּנֵי שִׂכְלוֹ נְהָרָה

וְיוֹדֵעַ יְסוֹד כָּל סוֹד וְכָל רָז / וְכָל סָתוּם בְּטֶבַע הַיְּצִירָה

יְנוֹפֵף יָד בְּטִבְעֵי הַיְּצוּרִים / וְכָל מָרְכָּב בְּמוֹ חֹמֶר וְצוּרָה

וְיַכִּיר בַּמְּזָגִים אִישׁ כְּעֶרְכּוֹ / בְּיַד מוֹפְתָיו וּבִרְאִיָּה בְרוּרָה

וְאֵיךְ יִתְמַזְּגוּ אַרְבַּע הֲפָכִים / לְהִתְאַחֵד בְּמַתְכּנֶת יְשָׁרָה

בְּהִתְלַכֵּד יְסוֹד אֲוִיר וְעָפָר / וְגַם מַיִם וְאֵשׁ אָרְחוּ לְחֶבְרָה

וְהֵם הַיְּסוֹד לְכָל הוֶה וְנִפְסָד / כְּפִי הַחוּשׁ וּמִשְׁפַּט הַסְּבָרָה

וּבַגּוּף יוּלְדוּ אַרְבַּע כְּנֶגְדָּם / וְכָל אַחַת יְדַמֶּה לַחֲבֵרָהּ

וְהֵם הַדָּם וּמִינֵי הַלְּבָנָה / וְהָאְדֻמָּה וּמִינֵי הַשְּׁחוֹרָה

וְהַכָּבֵד מְקוֹרָם גַּם יְסוֹדָם / וּבָאוּ שָׁם כְּמֵימֵי הַשְּׂעוֹרָה

וְלִשְׁמוֹר הַבְּרִיאוּת יַעֲרִיכָם / חֲכָם לְבָד בְּמַעְרֶכֶת שְׁמוּרָה

וְהַלֵּיחָה אֲשֶׁר תִּמְשֹׁל וְתִגְבַּר / בְּעוֹז סָמָיו יְסִירָהּ מִגְּבִירָה

וְיִפְעַל בָּם בְּכַמּוּת אוֹ בְּאֵיכוּת / וּבִסְגֻלָּה מְאֹד נִפְלֵאת וְזָרָה

וְהִתְחַכֵּם בְּאַרְבַּע מַעֲלוֹתָם / וּמַזְכֶּרֶת שְׁמָם עַל פִּיו שְׁגוּרָה

וְשָׁת לבוא בְּמִשְׁקַל הַמְּתֻקָּן / וְהַמּוֹבִיל וּמִשְקַל הַתְּמוּרָה

וְיַעְבִיר כָּל חֳלִי עָצוּם כְּהַעְבִיר / אֱנוֹשׁ שַׂעְרוֹ בְתַעַר הַשְּׂכִירָה

כְּאִלּוּ גָנְזוּ סִפְרֵי רְפוּאוֹת / בְּתוֹךְ רוּחוֹ וְהוֹצִיאוּ לְאוֹרָה

וְכֹחוֹ רַב בְּכֹחַ כָּל פְּעוּלָה / וְאִם הַלֵּב וְהַמּוֹחַ מְקוֹרָהּ

בְּכֹחַ רַעֲיוֹן צִיּוּר וְדִמְיוֹן / וְזִכָּרוֹן וְכֹחַ הַשְּׁמִירָה

וְחוּשֵׁי הָרְאוּת טַעַם וּמִשּׁוּשׁ / וְהֵרִחַ וְשָׁמַע קוֹל הֲבָרָה

וְיוֹדֵעַ בְּנִתּוּחַ עֲצָמִים / וּבַעְצַבִּים וּמֵיתְרֵי הַקְּשִׁירָה

וּבַדּפֶק יְמִישֶׁנּוּ אצבעוֹתָיו / לְהִתְבּוֹנֵן בְּסוּגָיו הָעֲשָׂרָה

וְעָלָיו נִפְתְּחוּ שַׁעְרֵי תְּבוּנָה / וְאֵין חָכְמָה לְפָנָיו נֶעֱצָרָה

וְיַרְגִּישׁ מַעְגְּלוֹת קָבוּץ כְּפָשׁוּט / וְלֹא עָמַד זְמָן רַב בַּחֲקִירָה

וְכָל צָפוּן יְבוֹנֵן עַל נְקַלָּה / וְלֹא יֵשֵׁב כְּסֹקְרָאט בַּמְּעָרָה

וּבִכְלִי הַזְּכוּכִית יֶחֱזֶה כָל / חֳלִי נִסְתָּר וְכָל מָרָה סְגוּרָה

וְכָל מַחְלָה אֲשֶׁר עַל הַכְּלָיוֹת / וְכָל סָתוּם בְּמַעְבַר הַמְּרֵרָה

וְיַנִּיעַ בְּיַד שִׂכְלוֹ קְרָבִים / כְּהַנִּיעַ אֱנוֹשׁ בָּר בִּכְבָרָה

וְיִמְצָא בוֹ אֱנוֹשׁ אָנוּשׁ תְּעָלָה / לְכָל עִלָּה גְדוֹלָה אוֹ צְעִירָה

וְכָל אִשָּׁה מְצֵרָה וַאֲנוּשָׁה / בְרוּחוֹ מָצְאָה רִפְאוּת לְשֵׁרָהּ

וְיוֹדֵעַ בְּיוֹלֶדֶת וְהָרָה / אֲשֶׁר לֹא יָדְעָה פוּעָה וְשִׁפְרָה

לְכָל מַדְוֶה צֳרִי גִלְעָד יְשַׁוֶּה / וּבוֹ מִקְוֶה לְמַלְאוֹת מַחֲסוֹרָהּ

וְאִם תִּשְׁכַּב לְמַעְצֵבָה יְשִׁיבָהּ / כְּאֶזְרַח רַעֲנָן בִּיבוֹשׁ קְצִירָהּ

וְאִלּוּ אַב הֲמוֹן גּוֹיִים יְדָעוֹ / אֱמֶת לֹא הָיְתָה שָׂרַי עֲקָרָה

וְשֵׁם אַבְרָם לְלֹא עֵזֶר יְמִירוּן / בְּאַבְרָהָם וְלֹא שָׂרַי בְשָׂרָה

וּבִסְגֻּלּוֹת לְבָד רוֹעֶה עֲקָרוֹת / כְּעֶשֶׁת שֵן מְגֹרָר בַּמְּגֵרָה

הֲכִי נִגְלֵית לְפָנָיו כָּל מְזִמָּה / וְכָל חָכְמָה כְקַלָּה כַחֲמוּרָה

וְכַרְמֵי בִין זְמָן נָטַע בְּרוּחוֹ / וְאֵין בִּלְתּוֹ אֲשֶׁר יִכְרֹת זְמוֹרָה

וּמָצָא חֵן בְּעֵינֵי אֵל וְאָדָם / וְקָנָה הַלְּבָבוֹת בַּאֲמִירָה

וְנִקְשָׁר כָּל אֱנוֹשׁ בַּעְבוֹת אֲהָבָיו / כְּשַׁלְהֶבֶת בְּגַחֶלֶת קְשׁוּרָה

וְהוּא מִשְׁעָן לְכָל כּוֹשֵׁל וְנֶחְשָׁל / וְלַנִּכְשָׁל זְרוֹעַ עִם גְּבוּרָה

וְהוּא מַרְפֵּא לְכָל נִסְפֶּה וְנִדְכֶּה / וְהוּא לַכּוֹשְׁלִים חוֹמָה בְצוּרָה

וּמָצְאוּ בוֹ מְיֻדָּעָיו תְּשׁוּעָה / אֲהָה וְאֵיךְ רוּחִי לֹא נֶעֱזָרָה

וּמִפִּיו צוּף דְּבָשׁ יִרְדּוּ יְדִידָיו / אֲהָה לִבִּי וְאֵיךְ יִרְדֵּה מְרוֹרָה

וְלִבּוֹת יוֹדְעָיו שָׂבְעוּ שְׂמָחוֹת / וְאֵיךְ רוּחִי שְׂבַע רֹגֶז וְעֶבְרָה

יְנַהֵל אוֹהֲבָיו עַל מֵי מְנוּחוֹת / וְנַפְשִׁי תַּהֲלֹךְ גּוֹלָה וְסוּרָה

אֲהָה נַפְשִׁי הֲיֵשׁ רִפְאוּת לְשִׁבְרָהּ / וְרוֹפֵא כָּל נְכֵה רוּחַ עֲכָרָהּ

הֲיֵשׁ חִתּוּל לְחַבֵּשׁ עִצְּבוֹנָהּ / וְהוּא יָרִיק סְגוֹר לִקְרַאת סְגוֹרָהּ

הֲיֵשׁ מָזוֹר לְמַחַץ חֵץ נְדוּדָיו / אֲשֶׁר לֹא יִחְבְּלוֹ שִׁרְיוֹן וְתַחְרָה

הִמְטִיר בִּי נְדוֹדוֹ אֵשׁ וְגָפְרִית / כְּעַל אַנְשֵׁי סְדוֹמָה וַעֲמוֹרָה

וְשָׁת לִבִּי כְלַבַּת אֵשׁ וְגָלְמִי / כְקַשׁ יָבֵשׁ בְּתוֹךְ רוּחַ סְעָרָה

וְאֵין זוּלָתְךָ עוֹזֵר וְסוֹמֵךְ / וְאֶל מִי בִלְתְּךָ אָנוּס לְעֶזְרָה

תְּמוֹךְ רוּחִי בְסַמִּים מַחֲזִיקִים / וּמֹר עוֹבֵר לְאַפִּי שִׂים קְטוּרָה

וְתַחְבֹּשֹׁת כְּפָרִים עִם נְרָדִים / בְּמֵי וֶרֶד לְכַבּוֹת הַמְּדוּרָה

וּמַחְלַת הַנְּדוֹד אִכְּלָה קְרָבַי / בְּשׁוֹשַׁנֵּי נְדוֹדָךְ הַעֲבִירָה

וְתַפּוחַ זְמִירָךְ תִּסְמְכֵנִי / הֲכִי יַד רוֹפְאֵי גִלְעָד קְצָרָה

וְכָל דּוֹרֵשׁ לְזוּלָתָךְ אֲרוּכָה / כְעוֹזֵב אֵל וְדוֹרֵשׁ לַאֲשׁרָה

וְיוֹם קַמְתָּ לְרוֹפֵא בֵין חֲכָמִים / לְחוֹלֵי הַזְּמָן הוּא יוֹם בְּשׂוֹרָה


סימן

בְּכָל לֵב נֶחֱלֶה תַעַשׂ פְּלָאִים / בְּיַד שִׂכְלָךְ בְּלֹא לַחַשׁ וְנַחַשׁ

הֲתָמְכָה יַד בְּנוֹ עַמְרָם יְמִינָךְ / וְֹהַֹמָֹּטֶֹהֹ אֲשֹר נֶהְפַּךְ לְנָחָשׁ


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
