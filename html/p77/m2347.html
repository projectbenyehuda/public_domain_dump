קרבת ידידך

קרבת ידידך / וידאל בנבנשת


קִרְבַת יְדִידָךְ כְּמוֹ אֶשְׁכֹּל כְּפָרֶיךָ / לִבִּי אֲבָל רָחֳקוֹ אֶשְׁכֹּל מְרוֹרֶיךָ

אִם אָז בְּאוֹר תָּאֳרוֹ זָרַח מְעוּף חֶשְׁכְּךָ / הַיּוֹם בְּהֵעָלְמוֹ חָשְׁכוּ מְאוֹרֶיךָ

יָצִיץ בְּאֶשְׁנַב יְדִידוּתוֹ וְתָצִיץ כְּעֵץ / רַעְנָן וְתָשִׂישׂ כְּסִיס עַל צִיץ אֲמִירֶךָ

חָלַף וְהָלַךְ וְתַחַת אוֹנְךָ אוֹנְךָ / הֵבִיא וּבִמְקוֹם מְזוֹרֶיךָ מְזוֹרֶיךָ

זִיווֹ וְהוֹדוֹ אֲרוּכָה לָךְ וּמַרְפֵּא אֲבָל / חֶרֶב נְדוֹד נָעֳמוֹ קָרַע סְגוֹרֶךָ           5

אוֹהֵב חֲשַׁבְתּוֹ אֲבָל יוֹם בָּא נְדוֹד נֶהֱפָךְ / אוֹיֵב וְשָׁב שׂוֹנְאָךְ וַיְהִי כְעָרֵךָ

כִּי אֵיךְ יְהִי אוֹהֲבָךְ גֶּבֶר אֲשֶׁר חֵילְךָ / יַגְבִּיר וְיַכְבִּיד חֲבָלֶיךָ וְצִירֶיךָ

עָרָה יְסוֹדוֹת עֲלֵי סוּתָךְ וְהָרַס וְלֹא / חָמַל וְרָדַף מְשׂוֹשָׂךְ עַד שְׁבָרֶךָ

אִבֵּד וְשִׁבֵּר בְּרִיחֵי גִיל וְקִצֵּץ עֲבוֹת / פֹּארוֹת פְּאֵר שִׁירְךָ פָּרַץ גְּדֵרֶיךָ

אֵיכָה לְבָבִי תְחִי רֶגַע וְנָד מִמְּךָ / נִשְׁמַתְּךָ רוּחֲךָ חַיֵּי בְשָׂרֶךָ           10

חַיֵּי בְשָׂרָךְ וְרוּחַ אַפְּךָ הוּא וְלוֹ / נָאוָה תְהִלָּה וְהוֹדָךְ וַהֲדָרֶךָ

צִיץ מִצְחֲךָ הוּא וְהוּא נֵזֶר לְרָאשָׁךְ וְהוּא / לוּחוֹת בְּרִיתָךְ וְתֻמֶּיךָ וְאוּרֶיךָ

אוֹצַר כְּלֵי הוֹד וְלֹא יִשְׁווּ חֲפָצִים בְּחִין / עֶרְכּוֹ וְנֶגְדּוֹ כְטִיט חוּצוֹת סְגוֹרֶךָ

גֶּבֶר יְגַבֵּר חֲיָלֵי בִין וְלֹא נִתְכְּנוּ / צִבְאוֹת זְמִירוֹת נְעִים אִמְרֵי שְׁפָרֶיךָ

גֶּבֶר אֲשֶׁר לוּ יְשִׂימָךְ זְמַנָּך הֲדוֹם / רַגְלָיו וְכַף נַעֲלוֹ אָז יַאֲמִירֶךָ           15

הֵן הוּא יְדִידָךְ יְהוֹסֵף הַמְאַסֵּף לְכָל / מַחְנֶה תְבוּנוֹת כְּרוֹעֶה צֹאן עֲדָרֶיךָ

יוֹסֵף שְׁמָךְ כִי דְּגַן חָכְמָה וּבַר בִּין לְךָ / תִּצְבֹּר וְתִשְׁבֹּר לְלֹא כֶסֶף מְחִירֶךָ

תִּשְׁמַע וְתַאְזִין חֲלוֹם חָכְמָה וְאַתָּה תְרַו / שִׁבְרוֹ וְיָקָם לְךָ תִּצְדַּק בְּדַבְּרֶךָ

אַתָּה תְאַלֵּם אֲלֻמּוֹת הַמְּזִמּוֹת וְכָל / דּוֹרֵשׁ רְשׁוּמוֹת יְלַקֵּט בֵּין עֳמָרֶיךָ

יוֹדְעִים אֲשֶׁר מַחֲזֶה חָכְמָה וּבִין יָדְעוּ / הָלְכוּ לְנֹגַהּ מְאוֹרֶךָ וְנֵרֶךָ           20

רֹאשׁ בַּחֲשִׁיבוּת תְּהִי אֶל כָּל בְּנֵי חֶלְדְּךָ / תִּקְדַּם בְּסִבָּה וְהֵם יִהְיוּ מְאַחְרֶיךָ

תֵּבֵל עֲקָרָה וְגַלְמוּדָה אֱמֶת הָיְתָה / קֹדֶם וְהָיָה זְמָן הוֹלֵךְ עֲרִירֶךָ

אֶל מִי אֲדַמָּךְ וּבִן יָקֶה לְנֶגְדָּךְ כְּמוֹ / אָדָם בְּשָׁשָׁרְךָ חָקוּי וְקִירֶךָ

גָּבָהּ זְמִירָךְ עֲדֵי דִּבְרֵי לְמוּאֵל לְאַל / תָּשִׂים וְנֶגְדָּךְ אֲגוּרֶךָ עֲגוּרֶךָ

כִּי אִשְׁרוּךָ בְּנֵי חֶלְדָּךְ וְכֵן דִּבְּרוּ / אַשְׁרֵי אֲשֶׁר כּוֹנְנָךְ אַשְׁרָיו וְאַַשְׁרֶיךָ           25

עֵט קַסְתְּךָ מַטְּךָ בּוֹ תַעֲשֶׂה מִפְלְאוֹת / אוֹתוֹת וְתַחְזִיר נְשָׁמוֹת אֶל פְּגָרֶיךָ

אַתָּה עֲטֶרֶת לְרֹאשׁ חָכְמָה וְנֶזֶם עֲלֵי / אַפָּהּ וְנָאווּ לְחָיֶיהָ בְּתוֹרֶיךָ

הָיוּ תְהִלּוֹת בְּךָ עֶצֶם וְעַצְמוּתְךָ / יוֹרוּ וְאֶל בִּלְתְּךָ הֵמָּה כְמִקְרֶךָ

קוֹסֵם אֲכַנָּךְ בְּשֵׁם חַרְטֹם וְאָשָׁף וְאוֹב / שׁוֹאֵל וְאֶקְרָאֲךָ חוֹבֵר חֲבָרֶיךָ

יַעַן בְּלַהְטֵי יְדִידוּתָךְ וְעִם שַׁרְשְׁרוֹת / חִנָּךְ תְּקַשֵּׁר כְּאִישׁ אֶחָד חֲבֵרֶיךָ           30

תַּבִּיט בְּמַבַּט אֲהָבֶיךָ פְּנֵיהֶם וְהֵם / כָּרְתוּ בְרִית לָךְ וְעָבְרוּ בֵין בְּתָרֶיךָ

תֹּאחַז לְבָבוֹת כְּיַעְלַת חֵן לְבַב חוֹשְׁקִים / תַּסְגִּיר סְגוֹרָם בְּבֵית כִּלְאָךְ וְסֹהְרֶךָ

מַלְכֵי תְּבוּנָה בְזִיקֵי שִׂכְלְךָ אָסְרוּ / רָצְעוּ לְבָבָם עֲלֵי דַל בֵּית אֲסוּרֶיךָ

שָׂרֵי יְקָר נוֹעֲדוּ יַחְדָּיו לְעָבְדָּךְ וְלֹא / יָצְאוּ בְיוֹבְלָךְ וְלֹא בִקְּשׁוּ דְרוֹרֶךָ

רוֹכֵב [.]קרי לְמֵישָׁרִים יְהַלֵּךְ בְּעֵת / אֶקְרַב לְנֹעַם הֲדָרֶךָ וְתוֹרֶךָ           35

וּבְעֵת אֲשֶׁר אֶרְחֲקָה מֵעִם נְעִים הוֹדְךָ / יִסֹּב לְאָחוֹר כְּמוֹ כוֹכְבֵי נְזוֹרֶךָ

בֵּינוֹת מְאוֹר שִׁמְשְׁךָ וּפְנֵי סְהָרִי זְמָן / אַדְמַת נְדוּדִים יְשַׁו וּמְרוֹם הֲרָרֶיךָ

עַד כִּי מְאוֹר סַהֲרִי נֶעְתָּם וְאָפֵס וְתָם / נֶעְדַּר וְקָדַר וְחָשַׁךְ מִשְּׁחוֹרֶיךָ

קֶשֶׁת יְדִידוּת וְקַו הָאַהֲבָה נִפְגְּשׁוּ / נִתְמַשְּׁשׁוּ עַל גְּבוּל וּקְצֵה קְטָרֶךָ

אֵיךְ קַו נְדוֹד בָּא לְהַפְרִיד בֵּין שְׁנֵיהֶם עֲלֵי / יֹשֶׁר וְזָוִית אֲהָבִים מוּל בְּצוּרֶךָ           40

תּוֹרֶה בְכַף קִלְעֲךָ לַיְלָה וְיוֹמָם לְמוּל / לִבִּי וְלֹא תַחֲטִיא אֶל חוּט שְׂעָרֶךָ

כִּמְעַט וּמַתִּי וְגָוַעְתִּי לְרֹב חִשְׁקְךָ / לוּלֵי בְּקִרְבִּי בְּעֵין לִבִּי אֲשׁוּרֶךָ

יוֹדְעַי אֲשֶׁר טָעֲמוּ צוּף נָעֳמִי יִשְׁאֲלוּ / אֵיךְ נֶאֱלַמְתָּ וְאֵיךְ נָם עֵט זְמִירֶךָ

אָשִׁיב וְהֵן יַד נְדוֹד הִיא אָמְרָה דּוֹם וְהַס / כִּי לֹא לְהַזְכִּיר בְּשׁיר בֹּא בַחֲדָרֶיךָ

יִרְחַב לְבָבִי וְגִיל יָגִיל בְּעֵת אֶחֱנֶה / נִכְחָךְ וְלוּ אֶשְׁכְּנָה בֵינוֹת מְצָרֶיך           45

הָיָה בְעֵינַי כְשׁבְתִּי בֵית אֲדֹנָי וְלֹא /אֶזְכֹּר גְּלוֹתִי בְיוֹם אֶשְׁכֹּן חֲצֵרֶיךָ

מִי יִתְּנֵנִי וְאֶסְתֹּפֵף בְּצֵל סֻכְּךָ / וָלֹא וְאַשִּׂיג הֱיוֹת צִפּוֹר דְּרוֹרֶךָ

אֶחְשֹׁק רְאוֹתָךְ עֲלֵי עַרְשִׂי בְחֶזְיוֹן חֲלוֹם / לַיְלָה וְלֹא אֶשְׁאֲלָה כַנְפֵי שְׁחַרֶךָ

אִם עוֹר פְּנֵי עוֹרְךָ קָרַן וְטָח מֵרְאוֹת / עֵינִי וְאַחְלַי וְהַרְאֵנִי אֲחוֹרֶיךָ

מִי יִתְּנֵנִי וְאֶרְאֶה צִלְּךָ אוֹ כְנַף / מַעְטֶה תְהִלָּתְךָ אֶדֶר יְקָרֶךָ           50

אוֹ אֶשְׁמְעָה צִלְצְלֵי שֶׁמַע תְּהִלָּתְךָ / אוֹ פַעֲמוֹן שִׁירְךָ כִּנּוֹר אֲמָרֶיךָ

מִי יִתְּנֵנִי אֲעוֹפֵף אֶל מְקוֹמָךְ עֲלֵי / אֶבְרַת חֲסִידָה וְעַל כַּנְפֵי נְשָׁרֶיךָ

נִשְׁבַּע לְבָבִי בְחֵי הָאַהֲבָה כִי בְדוֹד / לֹא אַחֲלִיפָךְ וְלֹא לָעַד אֲמִירֶךָ

דּודִי קְחָה לָךְ אֲרוּחַת שִׁיר וּמִנְחַת זְמִיר / מֵאֵת אֲסִיר חִשְׁקְךָ תּוֹךְ מַהֲמוֹרֶךָ

מֵאֵת מְשָׁרֵת יְדִידוּתָךְ וּמַעְלָתְךָ / נִמְכָּר לְאַהְבָתְךָ בִּשְׁטַר שְׁרִירֶךָ           55

צָמֵא וְדוֹמֶה לְמֵי שִׁירָךְ וְהוֹמֶה כְמוֹ / אַיָּל אֲשֶׁר תַּעֲרֹג עַל פִּי יְאוֹרֶךָ

צָמֵא לְנַחַל עֲדָנֶיךָ וְיֵין רִקְחֲךָ / רָוֶה נְדוּדָךְ וְשׁוֹתֶה כוֹס שְׁמָרֶיךָ

מַעְתִּיר פְּנֵי אֵל יְחַיֶּךָּ וְיַשְׂבִּיעֲךָ / כָּבוֹד וְעֹז לַעֲמֹד שִׁמְךָ וְזִכְרֶךָ

כִּרְצוֹן לְבָבָךְ וְלֵב מֻכֶּה בְשׁוֹט נוּדְךָ / נִרְצָח בְּשֶׁלַח פְּרִידָתָךְ וְתַעְרֶךָ

מַקְרִיב כְּתַב רוֹן וּמִזְמֹר שִׁיר עֲלֵי אַף נְדוֹד / הַמַּחֲרִיב יַם יְדִידוּת בַּחֲרָרֶךָ           60

נִכְתָּב בְּעֵט אַהֲבָה מַגִּיד מְעַט מֵאֲשֶׁר / חָקוּק בְּלִבּוֹ בְצִפֹּרֶן שֹמִירֶךָ

נִכְתַּב וְנֶחְתַּם בְּטַבַּעַת יְדִידוּת בְּאֶ-/ רֶץ נוֹד בְּשִׁשה וְעֶשְׂרִים לַאֲדָרֶךָ
