איך שאלתם דודים

אֵיךְ שְׁאֶלְתֶּם דּוֹדִים לְמִי הַתְּבוּנָה / לֹא יְדַעְתֶּם כִּי הִיא לְיוֹסֵף לְמִקְנָה

הוּא יְצָרָהּ בִּגְאוֹן יְקָרָהּ וְאֵיכָה / הִיא לְבִלְתּוֹ תִּכְרֹת בְּרִית וָאֲמָנָה

אִם בְּאַחֵר תִּבְחָר תְּמָאֵן כְּבוֹדָהּ / בָּחֲרָה בוֹ תַחְזִיר עֲטָרָה לְיָשְׁנָהּ

מִי בוֹחֵר בִּלְתּוֹ בְתֵבֵל לְשַׁכֵּן / בּוֹ הֲדָרָהּ אִם הוא לְבָדוֹ מְעוֹנָהּ

כֵּס לְשִׁבְתָּהּ הוּכַן לְבָבוֹ וְטֶרֶם / יַעֲמֹד הוּא הִתְנוֹדְדָה כַמְּלוּנָה 5

נָדְדָה עַל קַצְוֵי אֲרָצוֹת כְּבוֹדָהּ / נָדְדָה כַצִּפּוֹר וְשָׁבָה לְקִנָּהּ

לוֹ לְבָדוֹ בִמְתֵּי זְמָן יָאֲתָה כִי / הוּא לְכֻלָּם יַחַד כְּיָתֵד וּפִנָּה

הוּא בְּתוֹכָם כַּלֵּב בְּתוֹךְ הַגְּוִיָּה / הֵם כְּעַיִן אַךְ הוּא לְבַדּוֹ אֱשׁוּנָהּ

אֵל יְצָרוֹ בֵינָם לְהַשְׂכִּיל לְהֵיטִיב / הַפְּתָיוּת בִּעֵר בְּשִׂכְלוֹ וּפִנָּה

בֵּין אֲנָשֹים יִדְמֶה לְמַלְאַךְ אֱלֹהִים / הוּא כְאִישׁ אַךְ אֵין בּוֹ לְבָד הַתְּמוּנָה 10

מַעֲלוֹתָיו רַבּוּ לְאֵין קֵץ וּלְכָל / תַּעֲצוּמוֹ תֵבֵל עָלֶיהָ קְטַנָּה

וַאֲנִי לוֹ עַל כֵּן כְּעֶבֶד וְנַפְשִׁי / לַעֲבוֹדַת חִשְׁקוֹ נְתוּנָה נְתוּנָה

אֶתְּנָה רוּחִי לוֹ וְאִם יַחְפְּצֵנָהּ / אֶדְעָה אָז כִּי אֱלֹהִים חֲנָנָהּ

אֹחֲזָה לִי כַנְפֵי רְנָנִים וְלַחְזוֹת / הוֹד יְקָרוֹ אָבֹא לְפָנָיו בְּרִנָּה

אֶשְׁלְחָה אֶת זִמְרַת זְמִירַי וְשִׁירַי / לוֹ כְמֵבִיא עָפָר לְגִבְעַת לְבוֹנָה 15

אוֹ כְמֵבִיא אָטָד לְאַרְזֵי לְבָנוֹן / אוֹ כְמַקְרִיב אֶל אֵל בְּנֵי תוֹר וְיוֹנָה

אֶבְטְחָה בוֹ יַעְלוּ לְרָצוֹן לְפָנָיו / אַחֲרֵי כִי לוֹ הַתְּהִלָּה לְמָנָה

אֵין עֲרוֹךְ לוֹ בִיקָר וְחָכְמָה וּמַדָּע / פָּעֳלוֹ הוּא הֵעִיד בְּפָנָיו וְעָנָה

בַּחֲזוֹתְכֶם דּוֹדִים פְּאֵר נָעֳמוֹ אֵיךְ / לֹא יְרֵאתֶם לִשְׁאֹל לְמִי הַתְּבוּנָה.

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
