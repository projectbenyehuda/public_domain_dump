אחות קטנה

אָחוֹת קְטַנָּה / תְּפִלּוֹתֶיהָ

עוֹרְכָה, וְעוֹנָה / תְּהִלּוֹתֶיהָ.

אֵל, נָא רְפָא נָא / לְמַחֲלוֹתֶיהָ –

תִּכְלֶה שָׁנָה וְקִלֲלוֹתֶיהָ.


בְּנֹעַם מִלִּים / לְךָ תִּקְרָאֶה,

וְשִׁיר וְהִלּוּלִים, כִּי לְךָ נָאֶה.

עַד מָה תַעְלִים / עֵינְךָ – וְתִרְאֶה

זָרִים אוֹכְלִים / נַחֲלוֹתֶיהָ?

תִּכְלֶה שָׁנָה וְקִלֲלוֹתֶיהָ.


רְעֵה אֶת צֹאנְךָ / אֲרָיוֹת זֵרוּ,

וּשְׁפֹךְ חֲרוֹנְךָ / בְּאוֹמְרִים ״עָרוּ!״

וְכַנַּת יְמִינְךָ / פָּרְצוּ וְאָרוּ,

לֹא הִשְׁאִירוּ / עוֹלֲלוֹתֶיהָ –

תִּכְלֶה שָׁנָה וְקִלֲלוֹתֶיהָ.


מָתַי תַּעֲלֶה / בִּתְּךָ מִבּוֹר,

וּמִבֵּית כֶּלֶא / עֻלָּהּ תִּשְׁבֹּר,

וְתַפְלִיא פֶלֶא / בְּצֵאתְךָ כְּגִבּוֹר

לְהָתֵם וְכַלֵּה / מִכְלוֹתֶיהָ?

תִּכְלֶה שָׁנָה וְקִלֲלוֹתֶיהָ.


חֵילָהּ קָבְעוּ / הַגּוֹי כֻּלּוֹ

וְטוּבָהּ שָׂבְעוּ / וּבָזְזוּ אִישׁ לוֹ

וְלִבָּהּ קָרְעוּ / וּבְכָל זֹאת לֹא

מִמְּךָ נָעוּ / מַעְגְּלוֹתֶיהָ.

תִּכְלֶה שָׁנָה וְקִלֲלוֹתֶיהָ.


זְמִירָהּ שָׁבַת / וְחֶשְׁקָהּ תַּגְבִּיר

לַחְפֹּץ קִרְבַת / דּוֹדָהּ, וְתַעֲבִיר

מִלֵּב דַּאֲבַת / נַפְשָׁהּ וְתָסִיר,

לְבַקֵּשׁ אַהֲבַת / כְּלוּלוֹתֶיהָ –

תִּכְלֶה שָׁנָה וְקִלֲלוֹתֶיהָ.


נְחֶהָ בְנַחַת / לִנְוֵה רִבְצָהּ,

רַב נִזְנַחַת / מִדּוֹד חֶפְצָהּ,

וְהִיא כְפוֹרַחַת / עָלְתָה נִצָּהּ,

כִּי הִבְשִׁילוּ /אַשְׁכְּלוֹתֶיהָ –

תִּכְלֶה שָׁנָה וְקִלֲלוֹתֶיהָ.


חִזְקוּ וְגִילוּ / כִּי שֹׁד גָּמַר,

לְצוּר הוֹחִילוּ / בְּרִיתוֹ שָׁמַר

לָכֶם, וְתַעֲלוּ / לְצִיּוֹן, וְאָמַר:

סֹלּוּ סֹלּוּ / מְסִלּוֹתֶיהָ!

תִּכְלֶה שָׁנָה וְקִלֲלוֹתֶיהָ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
