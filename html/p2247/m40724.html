נֶפֶשׁ יְקָרָה אֶת־יְיָ הַלֲלִי (תחנה)

[מִמוֹנְטַלְצִ֗ינוֹ (Montalcino) בפלך טוֹסְקָנָה, חי לערך 1300]


תְּחִנָּה


– – ͝ – – – ͝ – – – ͝ –

נֶפֶשׁ יְקָרָה אֶת־יְיָ הַלֲלִי / הִשָּׁעֲנִי אֵלָיו וְלוֹ הִתְחוֹלֲלִי

אֵלָיו שְׂאִי לִבֵּךְ בְּכַפַּיִךְ וְאֶת־ / הוֹד שֵׁם כְּבוֹדוֹ הַמְּהֻלָּל גַדְּלִי

מַה־תֶּחֱשִׁי עוּרִי זְמִירוֹת זַמְּרִי / לָאֵל אֲשֶׁר תָּמִיד בְּחַסְדוֹ גוֹֽמְלֵי

דִּרְשִׁי אֱלהַיִךְ בְּכָל־לִבֵּךְ וְגַם / עֹצֶם גְּבוּרוֹת נִפְלְאוֹתָיו מַלְּלִי

נֹעַם דְבָרָיו עַל־לְבָבֵךְ יִֽהְיוּ / שִׁמְעִי קְחִי עֵצָה וּמוּסָר קַבְּלִי

יִהְיוּ לְטֹטָפוֹת עֲלֵי מִצְחֵךְ וְעַל־ / לִבֵּךְ כְּמוֹ חוֹתָם וְגַם כֶּתֶם חֲלִי

אֹרַח נְתִיבָתֵךְ בְּחָכְמָה יַשְּׁרִי / פַּנִּי מְסִלָּתֵךְ וְאוֹתָהּ סַקְּלִי

לִבֵּךְ בְּבֹרִית הַתְּבוּנָה כַּבְּסִי / נַפְשֵׁךְ בְּיַרְדֵּן הַתְּעוּדָה תִּטְבְּלִי

בִּגְדֵי לְבוּשֵׁךְ מַחֲלָצוֹת יִֽהְיוּ / תַּעְדִּי בְּיִרְאַת אַל לְאַט תִּתְנַהֲלִי

רִפְאוּת לְשָׁדֵּךְ הִיא וְגַם שִׁקּוּי תְּהִי / אֶל־כָּל־עֲצָמַיִךְ לְרַפֵּא כָל־חֳלִי

יָשִׂישׂ לְבַב שִׂכְלֵךְ בְּהַשְׂכִּילֵךְ וְיָ־ / דֹעַ אֱלהַיִךְ בְּזֹאת תִּתְהַלֲּלִי

חֶלְקִי יְיָ תֹּאמְרִי נַפְשִׁי וְגַם / הִתְחַנְּנִי אֵלָיו לְחַסְדּוֹ יַחֲלִי

יִנְעַם לְחִכֵּךְ צוּף אֲמָרָיו נִמְלְצוּ / בָּם תִּגְאֲלִי נַפְשֵׁךְ וּמִשַּׁחַת בְּלִי

אַל־תַּחְמְסִי נַפְשֵׁךְ וְאַל־נָא תִּשְׁלֲלִי / אַל־נָא בְּטֻמְאַת הַזְּמָן תִּתְגָּעֲלִי

לִבְשִׁי אֱפוֹד חָכְמָה וְעַל־מִצְחֵךְ תְּנִי / צִיץ הוֹד וְעָלַיִךְ יְחִידָה תַּחְמְלִי

חוּסִי עֲלֵי נַפְשֵׁךְ וְגוּפֵךְ פֶּן בְּיוֹם / אֶחָד יְחִידָה גַּם שְׁנֵיהֶם תִּשְׁכְּלִי

זִכְרִי שְׁלשֶׁת הַזְּמַנִּים, יַשְּׁרִי / לִבֵּךְ אֱלֵי צוּרֵךְ וְאַל תִּתְרַשְּׁלִי

קוּמִי וְהָכִינִי קְחִי צֵדָה לְיוֹם / בֹּא עֵת פְּרִידָתֵךְ לְדַרְכֵּךְ תֵּֽזְלִי

הִתְעוֹרֲרִי נָא מַהֲרִי טֶרֶם אֲשֶׁר / בִּיוֵן מְצוּלוֹת טִיט זְמַנֵּךְ תִּצְלֲלִי

בִּינִי לְאַחְרִיתְךָ בְּטֶרֶם בֹּא יְמֵי / עֵת קֵץ פְּקֻדָּתֵךְ וְאַל תִּתְעַצְּלִי

שִׂימִי אֱלֵי שִׂכְלֵךְ תְּשׁוּקָתֵךְ וְעַל־ / יִצְרֵךְ יְחִידָתִי בְּחָכְמָה תִמְשְׁלִי

אִם תִּשְׁמְעִי מוּסָר וְתַקְשִׁיבִי לְקוֹל /…………..
