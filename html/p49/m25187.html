[נשמה מאשר תתאַו]

ולה פי אלצבא ענד אנתקאלה מן קרטבה

ולו (שיר) בנעוריו בצאתו מקרטבה


U – – – U – – – U //– – U – – – U – – – U – –

נְשָׁמָה מֵאֲשֶׁר תִּתְאַו גְּרוּעָה // וְנֶפֶשׁ מֵאֲשֶׁר תִּשְׁאַל מְנוּעָה,

וְגוּף – שָׁמֵן וְשָׂבֵעַ וְדָשֵׁן, // וְנַפְשׁוֹ הַכְּבוּדָה לֹא שְׂבֵעָה,

וְאִישׁ עָנָו מְהַלֵּך עַל אֲדָמָה – // וּמַחְשַׁבְתּוֹ עֲלֵי שַׁחַק שְׂרוּעָה,

וּמַה יוֹעִיל לְאִישׁ־בָּשָׂר בְּשָׂרוֹ // וְטוֹבָתוֹ – וְהַנֶּפֶשׁ בְּרָעָה!

וְיֵשׁ רֵעִים מְרֵעִים, לֹא מְטִיבִים, // גְּדוֹלֵי גוּף – וְזִמָּתָם גְּרוּעָה,

יְדַמּוּ כִי לְהוֹסִיף הוֹן אֲנִי סָר // וְנֶעְתָּק מֵחֲנִיָּה אֶל נְסִיעָה,

וְאֵם־רֹאשִׁי בְּסוּפַת יוֹם פְּרוּעָה, // וּבַת־עֵינִי בְּפוּךְ לֵיל קְרוּעָה –

וְלֹא יָדְעוּ מְרֵעַי סוֹד לְבָבִי, // אֲבָל לֹא דִבְּרוּ רֵעַי בְּדֵעָה,

וְנַפְשָׁם מִבְּלִי הַכֵּר וְהָבִין, // כְּנֶפֶשׁ הַבְּהֵמָה הַשְּׁסוּעָה!

הֲיִתְאַפַּק אֲשֶׁר נַפְשׁוֹ לְבָנָה, // וְכִלְבָנָה לְהִתְנַשֵּׂא יְגֵעָה,

וְיִשְׁקֹט עַד אֲשֶׁר יֶאְזֹר בְּמָתְנָיו // כְּנָפֶיהָ, כְּאִישׁ אוֹזֵר יְרִיעָה,

וְעַד יִפְעַל וְיִשָּׁמְעוּ פְּעָלָיו, // וְיוֹסִיף עוֹד כְּיָם עַל הַשְּמוּעָה?

בְּחֵי הָאֵל וְחֵי עַבְדֵי אֱלֹהִים! // – וְכָמוֹנִי יְהִי שׁוֹמֵר שְׁבוּעָה –

בְּרַגְלַי אֶעֱלֶה סֶלַע, וְאֵרֵד // אֱלֵי שַׁחַת, בְּמַעְמַקִּים תְּקוּעָה,

וְאֶתְפֹּר אֶת שְׂפַת מִדְבָּר בְּמִדְבָּר, // וְאֶקְרַע יָם בְּכָל שׂוֹחָה קְלוּעָה,

וְאָשׁוּט עַד אֲשֶׁר אֵרוֹם וְאַעַל // אֱלֵי שִׂיאָה, תְּהִי לָעַד יְדוּעָה,

וְתִמָּצֵא לְצָרַי בִּי חֲרָדָה, // וְתִמָּצֵא לְדוֹדַי בִּי תְּשׁוּעָה,

וְאֶרְצַע אֶת בְּנֵי חוֹרִים בְּאֹזֶן, // וְלִי אֹזֶן לְמוֹדָעַי רְצוּעָה,

וְלִי נֶפֶשׁ לְרֵעִים מַחֲזֶקֶת, // וְלִי נֶפֶשׁ לְמַשְׂטִינִים מְנִיעָה,

וְלָךְ תּוֹכָה עֲרוּגָה, מִיְּדִידוּת // מְלֵאָה, עַל נְהַר אַהַב נְטוּעָה,

יְדִידוּת הַשְּׁמוּרָה מִנְּעוּרִים, // כְּחוֹתֶמֶת בְּטַבַּעַת קְבוּעָה,

מְפֻתַּחַת כְּפִתּוּחֵי יְרַקְרַק // עֲלֵי שִׁמְשָׁה, בְּסַף הֵיכָל קְרוּעָה!

יְהִי לָךְ כְּאַהְבָתָךְ, וְנַפְשָׁךְ // אֲשֶׁר אֹהַב תְּהִי מִצַּר פְּדוּעָה,

וְיִשְׁלַח אֵל־לְמוֹשָׁעוֹת לְךָ, עַד // בְּלִי שֶׁמֶשׁ וְיָרֵחַ, יְשׁוּעָה!


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
