רכא. [אשר לו]

אֲשֶׁר לוֹ יָם וְחָרָבָה / וְכָל מוֹשָׁב וּמַצָּבָה!

לְךָ אֶקְרָא בְּאַשְׁמוּרָה / וְעֵינִי מִבְּכִי זָבָה

לְךָ אֹמַר: פְּדֵנִי מִ־ / יְּדֵי חַטָּאת וּמֵחוֹבָה


הֲשִׁיבֵנוּ יְיָ אֵלֶיךָ וְנָשׁוּבָה!


בְּזָכְרִי אֶת עֲווֹנוֹתַי / יְקוֹד יֵקַד בְּכִלְיוֹתָי

אֲנִי זַדְתִּי, אֲנִי לַצְתִּי / וְהֶעְוֵיתִי נְתִיבוֹתָי

אֲהָהּ לִי, מָה אֲדַבֵּר עַל / אֲשֶׁר גָּרְמוּ עֲווֹנוֹתָי

וְאִם אֶלּוֹן — עֲלֵי נַפְשִׁי / לְבַדָּה כָּל תְּלוּנוֹתָי

בְּשָׁכְחָהּ כִּי בְּיוֹם מָחָר / אֱלֵי אֵלִי תְּשׁוּבוֹתָי

וּמָה אַעַשׂ בְּיוֹם יִשְׁקֹל / בְּמֹאזְנַיִם חֲטָאוֹתָי

וְאֵין מֵלִיץ וְאֵין זוֹכֵר / לְטוֹב כִּי אִם פְּעֻלּוֹתָי

וּמָה אֹמַר וְאֶצְטַדָּק, / וּמָה אַעַן וְאָשִׁיבָה?


הֲשִׁיבֵנוּ יְיָ אֵלֶיךָ וְנָשׁוּבָה!


גְּדוֹל עֵצָה, תְּמִים דֵּעָה, / אֲנִי רִמָּה וְתוֹלֵעָה

וְנַפְשִׁי מֵעֲפַר אֶרֶץ / וְאֶל עָפָר תְּהִי סוֹעָה

וְלָךְ אֶרֶץ וְשָׁמַיִם, / וְלָךְ טוֹבָה וְלָךְ רָעָה,

הֲתִפְקֹד עַל עֲווֹן נַפְשִׁי / אֲשֶׁר חָטְאָה וְהִרְשִׁיעָה —

וְאֵיךְ יִפְקֹד אֲבִיר רוֹעִים / עֲלֵי רָזָה וְצוֹלֵעָה?

הֲרֵעוֹתִי וְלֹא שַׁרְתִּי / בְּהָלְכִי עַל נְתִיב תּוֹעָה

אֱלֹהַי, גַּל כְּסוּת מֵעַל / שְׁתֵּי עֵינַי וְאֵיטִיבָה!


הֲשִׁיבֵנוּ יְיָ אֵלֶיךָ וְנָשׁוּבָה!


אֲנִי חוֹטֵא בְּנִבְרָאִים / וִבֵין רָעִים וְחַטָּאִים

וְאֵין לִי פֶה וְלֹא מֵצַח / לְהָרִים רֹאשׁ כְּמוֹ גֵאִים

וְאֵין עִמִּי בְּשָׁקְדִי עַל / מְזוּזוֹת פִּתְחֲךָ כִּי אִם

נְפָשׁוֹת עַד תְּהוֹם שַׁחוּ / בְּתַחְנוּנִים מְנֻשָּׂאִים

וְסָלַחְתָּ כְּמוֹ שַׂחְתָּ / וְהִנֵּנוּ לְךָ בָאִים

וּמִתְוַדִּים בְּפִלּוּלִים / בְּפַז יָקָר מְסֻלָּאִים

וְנַפְשׁוֹת עַם מְחֻלָּאִים / וְחֹלִי רָב מְמֻלָּאִים

רְפָא נִגְעָם, סְלַח פִּשְׁעָם,/ שְׁעֵה שַׁוְעָם וְהַקְשִׁיבָה!


הֲשִׁיבֵנוּ יְיָ אֵלֶיךָ וְנָשׁוּבָה!


נְתַתַּנִי בְּגָלוּתִי / וְטָבַעְתִּי בְּאַשְׁמָתִי

וְהָיִיתִי כְּשֶׂה אוֹבֵד / וְאֵין מֵשִׁיב אֲבֵדָתִי

וְלִי לָעְגוּ אֲשֶׁר רָאוּ / חַכוֹתִי יוֹם גְּאֻלָּתִי

וְאָמְרוּ כִי כְּעָב נִמְחָה, / וְכַצֵּל, יוֹם יְשׁוּעָתִי

וְכִי אֵין קֵץ לְדַלּוּתִי / וְאֵין תִּכְלָה לְצָרָתִי

וְאֶשְׁמָעֵם וְאֶכָּלֵם / וְתִשְׁתַּפֵּךְ מְרֵרָתִי,

וְאֵיךּ תִּשְׁקֹט, וְאֵיכָה לֹא / תְּקַנֵּא מִכְּלִמָּתִי?

זְכָר עַמָּךְ וּפָקְדֵהוּ / וְהַעְלֵהוּ בְּמַחְשָׁבָה!


הֲשִׁיבֵנוּ יְיָ אֵלֶיךָ וְנָשׁוּבָה!


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
