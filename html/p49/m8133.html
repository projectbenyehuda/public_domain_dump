[התזכיר אל]

וֶהֱשִׁיבוּ אָבִי, יוֹקִירֵהוּ אֵל [לְרַב יוֹסֵף בֶּן חַסְדַאי], בָּזֶה.


הֲתַּזְכִּיר אֶל יְפַת מַרְאֶה לְבוֹנָה – / וְגֻלְגָּלְתָּךְ כְּמוֹ חָלָב לְבָנָה!

וְהָעֶדְנָה שְׂנֵאָתָךְ בְּזָדוֹן – / וְאֵיךְ תִּשְׁגֶה בְּאַהֲבָתָךְ עֲדִינָה?

– שְׁמָּעֵנִי, מְרִיבִי, עַל בְּכוֹתִי / יְמֵי נֹעַר, וְאַחַר כֵּן עֲנֵה נָא:

חֲדָשָׁה נִבְרְאָה עַל קָדְקָדִי, לֹא / אֲהַבְתִּיהָ, וְאֹהַב הַיְּשָׁנָה,

וְרֶשֶׁת הַשְּׁחִיתוֹת, הַמְחִישׁוֹת / אֱלֵי שַׁחַת, עֲלֵי רֹאשִׁי נְתוּנָה, [5]

וְהָפְכוּ לִי לְבָנִים, לֵב לְבָנוֹת / אֲהֵבוּנִי, לְאֶבֶן אוֹ לְבֵנָה,

וְלָכֵן מִלְּשַׂחֵק נָא חֲדַלְנָה, / מְשַׂחְקוֹתַי, וְנוּס לֵחִי בְּכֶינָה,

וְשָׁרוֹתַי, שְׁבֹרְנָה תֹף וּמָחוֹל, / וְעַל נַפְשִׁי כְּלֵי תַנּוֹת עֲשֶׂינָה.

אֲנִי גֶבֶר, יְקִיצֵנִי בְּלַיִל / יְגוֹן שֵׂיבָה אֲשֶׁר פָּשְׂתָה עֲדֶנָה,

וְיַזְכִּירוּ יְמֵי רֵאשִׁית וְשַׁחְרוּת, / יְמֵי רַע שִׁחֲרוּנִי אַחֲרוֹנָה, [10]

וּמִי אֶקְרָא אֱלֵי אַהַב – וְיַעַן, / וְלֹא יִגְרַע שְׁאֵר וּכְסוּת וְעוֹנָה,

וְלִבִּי סָג אֱלֵי סוּגָה כְּשׁוֹשָׁן, / וְלִבִּי בַר לְבָרָה כַלְּבָנָה,

וְהָיִיתִי לְמָנָה לָאֲהָבִים, וְהָיוּ הָאֲהָבִים לִי לְמָנָה.

תְּבוֹאֵנִי אֲחוֹת שֶׁמֶשׁ, וְרֹאשׁ לָהּ / כְּמוֹ אֹפֶל וְכִפָּה רַעֲנָנָה,

לְהִתְעַלֵּס בְּמֵי אַהַב וְלִרְווֹת / בְּיֵין דּוֹדִים וְגוּפָה רַעֲנָנָּה, [15]

וְתֵּדַע כִי אֲהָבַי נֶאֱמָנִים, / וְכִי לִבִּי, כְּמוֹ יָדַי, אֱמוּנָה,

וְלֹא תִירָא מְצֹא אֶצְלִי כְּמִפְעָל / אֲשֶׁר פָּעַל שְׁכֶם הָרָע לְדִינָה,

וְעַל כָּכָה תְּבוֹאֵנִי לְאוֹר יוֹם, / וְלֹא תִשְׁלַח יְצוּרֶיהָ בְּשֵׁנָה,

וְלֹא תַרְאֶה דְּמוּתָהּ בַּתְּמוּנָה, / וְלֹא תַרְאֶה בְּדִמְיוֹנוֹת תְּמוּנָה,

וְלֹא אֹמַר בְּעֵת הָקִיץ: אֲהָהּ עַל / אֲשֶׁר הָיָה כְּלֹא הָיָה וּפָנָה! [20]

וְיֵשׁ תַּעְנוּג בְּצֶלֶם שָׁט יְשִׁימוֹן / וּמִסְתַּתֵּר בְּצַלְמָוֶת נְכוֹנָה,

וְלוֹבֵשׁ, מִפְּנֵי מֵרִיב, עֲרָפֶל, / כְּסַהַר שָׁכְנָה עָלָיו עֲנָנָה,

וְגוֹזֵר בַּחֲצִי לַיִל דְּרָכִים אֲשֶׁר לֹא יִגְזְרוּ סוּסִים בְּשָׁנָה,

לְהִתְרָאוֹת בְּתַרְדֵּמָה לְלֵב עֵר / וְנֶפֶשׁ מֵאֲהָבִים לֹא יְשֵׁנָה –

אֲבָל יִתְרוֹן, עֲלֵי רוֹאוֹת בְּנוּמָה, / לְעֵינַיִם בְהָקִיץ תֶּחֱזֶינָה, [25]

כְּיִתְרוֹן אוֹהֲבֵי יוֹסֵף עֲלֵי כָל / בְּנִי דוֹרִי וְעַמִּי בַתְבוּנָה,

הֲבִינוֹתִי תְהִלָּתָךְ וְעַל כֵּן, / נְשִׂיא עַמִּי, דְּבָרַי הַאֲזִינָה,

וְתֵדַע כִּי יְדַעְתִּיךָ, וְאִמְרַת / אֲשֶׁר יָדַע יְדִידוֹ – נֶאֱמָנָה,

וְלָמָּה זֶה אֱהִי, אִם לֹא אֲסַפֵּר / אֲשֶׁר אֵדַע בְּךָ אַחַר בְּחִינָה,

כְּאִישׁ יִטְרַח וְיָבִיא שֶׂה לְפִסְחוֹ – / וְיֹאכְלֵהוּ בְּקֶרֶב מִקְדְּשׁוֹ נָא, [30]

וְכַחוֹלֵל אֲשֶׁר יָגַע וּלָמַד – / וְשִׁבֵּר אֵת חֲלִילוֹ עֵת נְגִינָה.

לְךָ מוֹרָשׁ וְקִנְיָן בַּגְּדֻלָּה – / וְאָן תִּמְצָא אֲשֶׁר יָרַשׁ וְקָנָה?

וְאַתָּה מַעֲנֶה לַדָּת, וְעִנְיָן לְכָל נוֹצָר, וְלַמּוּסָר מְעוֹנָה,

וְלָךְ תָּבִיא תְּבוּנָתָךְ וְדַעְתָּךְ / חֲפָצֶיךָ, וְכָל טוֹבָה סְפוּנָה,

וְכָל יִיגַע לְהַשִּׂיג הוֹן – וְתִיגַע / לְהַשִּׂיג חֹק בְּמוֹעֵצוֹת וּבִינָה, [35]

וְרֹב שָׂרִים עֲלֵי בוּז שַׁאֲנַנִּים – / וְאֵין נַפְשָׁךְ בְּיוֹם בּוּז שַׁאֲנַנָּה,

וְאַתְּ מוֹסִיף בְּיוֹם מָחָר לְהַפְלִיא / עֲלֵי מִפְעַל נְכוֹנָה תַעֲשֶׂה נָּא,

וְתִפְגֹּשׁ אֶת מְיֻדָּעִים וְרֵעִים / וְכָל דּוֹרֵשׁ מְקוֹם מִסְעָד וְלִינָה

בְּפָנִים יִצְהֲלוּ נֶגְדָּם, כְּאִלּוּ / מְאוֹר הַיּוֹם אֲלֵיהֶם סָר וּפָנָה,

וְיָדַיִם יְבִישׁוּן הָעֲנָנִים, כְּיוֹם מַבּוּל מְלֻמָּדוֹת נְתִינָה, [40]

וּבִכְלָיוֹת לְמוֹעֵצוֹת, עֲלֵי כָל / כְּלָיוֹת נַעֲלוּ וַתִּגְבְּהֶנָה,

וְרֹחַב לֵב, וְרֹחַב כַּף, כְּיַרְדֵּן, / גְּדוֹתָיו יִגְּרוּ אָנֶה וְאָנָה,

וְאִם תֹּאמַר – סְעָדוּךָ יְחָסִים, / וְאִם תַּעַשׂ – עֲזָרַתְךָ אֱמוּנָה,

וְיָדֶיךָ יְדוּעוֹת לַמְּבַקֵּשׁ / וְלַשּׁוֹאֵל, כְּפַרְפַּר וַאֲמָנָה,

וְכָל דּוֹרְשֵׁי מְקוֹם מַרְבֵּץ לְחָכָמָה – / בְּבֵיתָךְ יִמְצָאוּ חָכְמָה שְׁכוּנָה, [45]

וְהִגִּיעַ לְשִׁכְנָךְ חֹק מְשׁוֹטֵט – / וְשָׁבַר אֶת כְּלֵי מַסָּע וְחָנָה,

וְלֹא נִמְצֵאת לְחָכְמָתָךְ תְּכוּנָה – / וּמִי יִמְצָא לְיָם סוֹבֵב תְּכוּנָה?

כְּמוֹ פִיךָ, כְּמוֹ עֵטָךְ, יְנוּבוּן / פְּרִי צֶדֶק וְתַחְבּוּלוֹת וּבִינָה:

לְזֶה יִתְרוֹן בְּשֶׁבֶת שַׁעֲרָה עִם / מְתֵי שֵׂכֶל, בְּעֵת אָמַר וְעָנָה,

וְלָזֶה קוֹל בְּלִי אֹמֶר לְחַרְחַר / מְרִיבָה אוֹ לְהַשְׁקִיט אֶת רְגִינָה, [50]

יְרֻקָּה, מִבְּנוֹת סוּף, גָּדְלָה בֵין / פְּלָגִים, וַעֲלֵי יוּבַל אֱמוּנָה,

מְצַפְצֶפֶת כְּמוֹ יוֹנָה – וְתַחְרִישׁ, / בְּהִבָּלְעָה בְּתוֹךְ קֶסֶת כְּיוֹנָה,

גְּבִירָה תַּעֲמִיד אֶרֶץ, וְאוֹתוֹת – / חֲיָלִים לָהּ, וְאִגֶּרֶת – מְדִינָה,

וְהָרֹאשָׁה, כְּפִיחַ לָהּ בְּרֹאשָׁהּ, / בְּיַד פַּחוֹת וּבִקְוֻצּוֹת קְצִינָה,

וְחֵרֶשֶת – וְתִשְׁמַע מֶחְקְרֵי לֵב, / וְעִוֶּרֶתּ – וְתִרְאֶה כָל טְמִינָה, [55]

וְאִלֶּמֶת – מְדַבֶּרֶת בְּרוֹמָה / עֲלֵי אֶצְבַּע, כְּכִיּוֹר עַל מְכוֹנָה,

וְתִשְׁפֹּךְ סוֹד קְרָבִים עַל כְּתָבִים, / וְלֹא תָכִיל, בְּהִנָּשְׂאָהּ, טְמוּנָה,

וּפִסַּחַת – אֲשֶׁר תָקוּם עַלֵי גַב / סְפָרִים מִבְּלִי יָרֵךְ וּקָנָה,

וְתִשָּׂא קוֹל עֲלֵיהֶם מִסְּפָרַד / וְהַחוֹנִים בְּאֶרֶץ נֹף מְבִינָה

וְאוֹמֶרֶת בְּלִי לָשׁוֹן וְעוֹנָה, / וְסוֹפֶרֶת בְּלֹא אֶצְבַּע וּמוֹנָה, [60]

וְיַלְדָּהּ בַעֲשׂוֹתָהּ אֶת מְלַאכְתָה, וְעֵת תִּיעַץ וְתַמְתִּיק סוֹד – זְקֵנָה,

וּפַעַם תַּעֲנֶה תוֹרוֹת כְּתַמָּה, / וּפַעַם תַּעֲנֶה זָרוֹת כְּזוֹנָה.

לְךָ שָׂרִי וְנִכְבָּדִי בַּחָכְמָה אֲשֶׁר תָּנוּב לְךָ מוֹנִים שְׁמוֹנָה,

וְטוֹבָתָהּ יְדוּעָה לְךָ, וְכָכָה / תְּבוּאַת מַעֲשֶׂיהָ לָךְ צְפוּנָה,

וְצִוִּיתָ בְּשִׁירָךְ עַל בְּנֵי אָח / אֲהַבְתָּם, וַעֲלֵיהֶם לְךָ חֲנִינָה, [65]

וַתֵּדַע כִּי חֲשַׁבְתִּימוֹ כְּבָנַי / וְכִי עַצְמָם, בְּתוֹךְ צִלִּי, דְּשֵׁנָה,

וְיוֹם יִרְאוּ תְאוּנִים מֵעֲלֵיהֶם – / הֲשִׁיבוֹתִי בְּחֶזְקַת יָד תְּאוּנָה,

וְיוֹם בָּאוּ בְּמֵי צָרָה וְחָבַשׁ / לְרֹאשָׁם סוּף – אֲנִי לָהֶם סְפִינָה,

וְאִם נָשָׂא אֱוִיל רֹאשׁ לְהֲנִיאָם – / אֲנִי חֶרֶב עֲלֵי רֹאשׁוֹ שְׁנוּנָה,

וְאוֹסִיף עוֹד, לְמַעַן כִּי יְדִידוּת / בְּתוֹךְ נַפְשִׁי לְךָ מֵאֵין תְּלֻנָּה, [70]

וּמֵחִשְׁקִי בְּךָ נַפְשִׁי שְׁמֵנָה, / כְּמוֹ אָשֵׁר אֲשֶׁר לַחְמוֹ שְׁמֵנָה,

וְיִשָּׁמַע, כְּשָׁמְעִי אוֹמְרִים כִּי / בְּטוֹבָה אַתְּ, עֲלֵי פִּי קוֹל רְנָנָה,

וְאַהְבָתִי מְזֻקֶּקֶת כְּכֶסֶף / לְךָ מֵאָז, וְכַזָּהָב בְּחוּנָה,

וּמִי יָבֹא לְכַסּוֹתָהּ – יְכַסֶּה / בְּחוּט מֵעַל אֲדָמָה אוֹר לְבָנָה!

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
