[הַאֶעֱצֹר נַחֲלֵי]

וְאָמַר עוֹד בְּצֵאת מַחֲנוֹת אִשְׁבִּילִיָּה וְרוֹנְדָה וְכָל קְצֵה הַמַּעֲרָב אֶל רֵיָה הַמָּקוֹם, בֶּאֱלוּל שְׁנַת תתי"ב [ 4012; אוֹגוּסְט 1052]. וְכַאֲשֶׁר הִגִּיעוּ חֲיָלָיו בְּקָרוֹב מֵהֶם נִגְּשוּ כֻּלָּם אֶל הֶהָרִים וְעָלוּ בַגְּבָעוֹת, וּמַחֲנוֹתֵינוּ יָשְׁבוּ בְתַחְתִּית הָהָר לִקְרָאתָם שְׁנֵי שָׁבוּעוֹת וְלֹא יָרַד אֲלֵיהֶם אִישׁ, לֹא פָרָשׁ וְלֹא רַגְלִי, וְאַחַר כֵּן הָלְכוּ לְדַרְכָּם בְּבשֶׁת פָּנִים. וְאָמַר, יַעַזְרֵהוּ אֵל:

תְהִלָּה

הַאֶעֱצֹר נַחֲלֵי עֵינַי וְאָנוּחָה – / וּבְנֵי חֲמָת נָחֲלוּ חַמַּת וְיָנוֹחָה,

תִּשְׁכֹּן גְּבַל תּוֹךְ גְּבוּל צִיּוֹן וּבֵית / מַרְאֵה נֹגַהּ שְׁכִינָה – וּבַת צִיּוֹן מְנֻדָּחָה.

אֵשֶׁת כְּרִיתוּת חֲשָׁבָה בַּעֲלָהּ, מֵאֵין / סֵפֶר כְּרִיתוּת גְּרוּשָׁתוֹ, וְנִשְׁכָּחָה.

מַה לִּי בְחַיִּים – וְאֵין אָרוֹן וְכַפֹּרֶת / וַהְדַר מְנוֹרָה מְשֻׁקֶּדֶת וּפֹרָחָה.

וּדְבִיר וְאֵלָיו, ובֵית קָדְשֵׁי קֳדָשִים – / אֵשׁ זָרָה בְתוֹכוֹ, וְאֵשׁ קֹדֶשׁ מְלֻקָּחָה

וּפְאֵר מְכוֹנוֹת וְכוֹתָרוֹת וּתִמּוֹרוֹת / דּוֹמוֹת לְרִמּוֹן, וְהֵם תִּשְׁעִים וְשֵׁשׁ רוּחָה,

אֵשֵׁב נְדוּד נוּם, בְּטוּב נוּמָה לְעַרְלֵי לֵב, / אֶשְׁאַף וְאָשִׂים עֲלֵי לִבִּי וְאָשִׂיחָה.

אָשׁוּר שְׁחָקִים עֲלֵי רֹאשִׁי, וְכוֹכָבִים, / כִּדְמוּת עֲרוּגָה בְּנִצָּנִים מְפֻתָּחָה,

כִּפְנֵי פְרָחִים, כְּעֵין כּוֹסוֹת בְּיַיִן, לוֹ / נֹגַהּ, וְרֵיחַ צְרוֹר הַמֹּר בְּמֶרְקָחָה,

וּדְמוּת כְּסִיל, בַּחֲדָרָיו נָח לְבַדּוֹ, בֵּן / סוֹרֵר כְּסִיל, יֵשְׂטְ מִשֵּׁבֶט וּמִשְׁפָּחָה

וּפְנֵי בְנֵי עָשׁ וְעָשׁ – אִשָּׁה אֲשֶׁר עוֹטָה / בָּנוֹת בְּחֶמְלָה, וּבִנְוֵה שַׁאֲנָן נָחָה.

תִּסֹּב מְהֵרָה בְּלִי רֶגֶל בְּקַו שַׁחַק / כִּימָה בְּכָל יוֹם, כְּאַיָּלָה מְשֻׁלָּחָה,

פֶּלֶךְ יְרוֹצֵץ בְּלִי רִפְיוֹן, וְאֶרֶץ תּוֹךְ / פֶּלֶך תְּלוּיָה בְּלִי חֶבֶל וְלֹא שָׁחָה,

כֻּלָּם נְכוֹחִים לְסוּרֵי רַע מְבִינִים לוֹ – / וְקְהַל פְּתָאִים בְּטוּב לֵבָב וּבִמְנוּחָה.

הַלֵּךְ בְִּנִי עַל נְתִיב טוֹבָה, לְמַעַן כִּי / אֵין טוֹב לְבַד עַל נְתִיב טוֹבָה וּבִנְכוֹחָה,

וּשְׁמַע פְּלִיאָה אֲשֶׁר עָשָה אֱלֹהִים, יוֹם / הִרְבּוּ מְשַׂנְאִים בְּמַשְׂאוֹת שָׁוְא וּמַדּוּחָה:

נִקְווּ נְדִיבִים וְטַפְסָרִים וְשַׁלִּיטִים, / שָׁרֵי צְבָאוֹת וּמַמְלָכוֹת וְכָל פֶּחָה.

בָּאנוּ עֲלֵיהֶם בְּעֵדָה מִהֲרָה לָרִיב, / זִיקוֹת קְרָב אִזְּרָה מֵאָז וְקָדָחָה.

זָכְרוּ מְשַׂנְאִים פְּעֻלָּתָהּ – וְקֻלָּלָה / שָׁמְעוּ מְלָכִים גְּבוּרָתָה – וְשֻׁבָּחָה.

נַפְשָׁם שְׂבֵעָה לְהָרֵעַ, וּמִדָּמִים / אֶרֶץ מְרַוָּה וּמוֹלִידָה וּמַצְמִיחָה,

שָׁמְעָה עֲדַת רִיב תְּרוּעָתָה –וְרָעָה לָהּ / לִמְאֹד, וְשָׁמְעָה בְהַצְרִיחָהּ – וְצָרָחָה.

אָמְרָה בְּלִבָּה: אֲנִי אַחַת וְאַפְסִי עוֹד! – / הָיְתָה כְאֶפֶס בְּבוֹאֵנוּ, וְנִדָּחָה;

הָיְתָה בְמִישׁוֹר כְּמוֹ לֵישִׁים מְמֻשָּׁכָה – / עָלְתָה אֱלֵי הַר כְּאַיָּלִים וּפָסָחָה.

שַׂחְנוּ אֲלֵיהֶם: הֲזֶה מִפְעַל עֲדַת אִישִׁים, / שָׁמְעָה שְׁמוּעַת עֲדַת אִישִׁים וּבָרָחָה?

אָנָּא, רְדוּ מֵרְכָסִים עַד אֲשֶׁר נָשִׂים / אֶרֶץ כְּמוֹ סִיר בְּדַם שָׂרִים מְרֻתָּחָה.

בּוֹאוּ גְשׁוּ בַּקְּרָב, נַפְשׁוֹת שְׁנֵי לוֹחְמִים, / אַחַת תְּשַׁוַּע וְרַעְיָתָהּ מְשֻׂמָּחָה.

בּוֹאוּ נְבַכֶּה נְשֵׁיכֶם דֹּם, וְנַגְבִּירָה / תּוֹךְ אָהֳלֵיכֶם מְמֹרָטָה וְנִקְרָחָה,

עַד יַעֲבִיר קוֹל נְשִׂיא כָל עוֹף וְכָל חַיָּה / עַל כָּל שְׁכוּנָה-בְתַלְאוּבוֹת וּפוֹרָחָה,

לָבֹא וְלָלֹק דְּמֵי גוּפוֹת גְּבִירִים, רֹאשׁ / מַכָּה בְּרֹאשׁ אוֹ בְּפִי חֶרֶב מְנֻתָּחָה,

לָשׂוּם גְּבִירִים, אֲשֶׁר שַׂמְנוּ מְכַסֵּיהֶם / רִמָּה, לְמִשְׁטָח כְּמוֹ דֹּמֶן וְכַסּוּחָה,

לָשׂוּם נְתָחִים, וְלֹא נִתְחֵי כְבָשִׂים, אֶל / כָּל עוֹף בְּכָל קֵן וְאֶל כָּל גּוּר בְּכָל שׁוּחָה.

אִישִׁים אֲשֶׁר נִבְאֲשׁוּ בָּנוּ – אֲזַי נַבְאִישׁ / עוֹרָם, וְהֵם בַּעֲלֵי רֹקַח וְרַקָּחָה,

נַכֵּם בְּמִפְעָל וְשׁוֹט יָד, לֹא בְשֹוֹט לָשׁוֹן, / כִּי רַךְ לְבָבוֹ יְגַבֵּר פִּיו וְיָפִיחָה.

נִקְרַב – וּמִפֹּה וּמִפֹּה נַעֲלָה, אָכֵן / עֵת רָחֳקֵנוּ בְּשֵׁבֶט פֶּה יְנַגֵּחָה.

לָמָּה גְבוֹהָה גְבוֹהָה תִנְאֲמוּ רִיק, רַק / עֵת בּוֹא אֱמֶת תֶּחֱשׁוּ מֵרִיב וְתוֹכֵחָה

אַיֵּה שְׁקָרִים אֲשֶׁר טַחְתֶּם בְּפָה? הִנֵּה / הַמַּעֲרָכָה – וְאֵין תִּפְלוּת בְּקִיר טוּחָה!

בִּפְנֵי נְשֵׁיכֶם תְּשִׂיחוּן רַב, וְיֵשׁ נָשִׁים / לָהֶם בְּעָלִים, בְּבוֹאָם יִגְרְעוּ שִׂיחָה.

צוֹוְחִים בְּרֹאשׁ הָר רְדוּ הֵנָּה נְמַלֵּא פֶה / נָשִֹים, מְמֻלָא שְֹחוֹק, בִּנְהִי וּבִצְוָחָה.

נַרְעֵם בְּקוֹלוֹת וְאין עוֹנֶה בְּקוֹל, כִּי אֵין / מַלְכּוֹד בְּשִׂיחָה לְעוֹף רָאָה מְקוֹם שִׁיחָה.

בּשׁוּ וְשָׁבוּ מְתֹעָבִים וְנֶאְלָחִים / מִכָּל חֲבוּרָה מְתֹעָבָה וְנֶאְלָחָה.

הָיוּ בְחֶרְפָּה בְּדַבֵּר אִישׁ לְאִישׁ: אַיֵּה / אֵיפֹה דְּבָרָךְ? וְיֵאָלֵם וְיִשְׁתֶּחָה.

שָׁבוּ רְצוּחִים – וְהֵם חַיִּים, וְיֵשׁ אִישִׁים / מֵתֵי כְלִמָּה – וְנַפְשָׁם לֹא מְרֻצָּחָה.

יֵשׁ אִישׁ וְלוֹ שַׂעֲרַת זָקָן וְרֹאשׁ רַבָּה – / מִבּוּז זְקָנוֹ וְשַׂעְרָתוֹ מְגֻלָּחָה;

יֵשׁ אִישׁ מְהַלֵּךְ וְהוּא בָּרִיא – וּמֵחִצֵּי / לַעַג-בְּשַׁעַר מְרוֹרָתוֹ מְפֻלָּחָה.

נַפְשָׁם שְׂבֵעָה יְגוֹנִים מֵחֲפוֹת רֹאשׁ, אַךְ / נַפְשִׁי, בְּרֹאשׁ נַעֲלֶה, שָׂשָׂה וְשָׂמֵחָה.

זֶה סוֹף קְהִלָּה אֲשֶׁר אָמְרָה וְלֹא עָשְׂתָה, / זֶה סוֹף קְהִילָּה אֲשֶׂר רָכְבָה וְצָלֵחָה;

זֶה אַחֲרִית עַם בְּנוּס רוּחַם – וְזֶה אַחֲרִית / חוֹשְׁבֵי תְמוּתָה כְּמוֹ חַיִּים וְכִרְוָחָה.

כֵּן כָּל מְפִירַת בְּרִית בַּיּוֹם תְּהִי שָׁבָה, / כִּי מָאֲסָה חֹק, וְדִין עָזְבָה וְשָׁכָחָה. <>br>
הִנֵּה לְשׁוֹנִי כְּחֶרֶב חַד, וְלַגּוֹמֵל / כָּזֹאת לְנַפְשִׁי – בְּשִׁיר אָחֵל וְאֶפְתָּחָה,

אָחֵל וְאַשְׁלִים בְּהַלֵּל אֵל-יְשׁוּעוֹת עַל / כּוֹסוֹת יְשׁוּעָה אֲשֶׁר אֶתֵּן וְאֶקָּחָה,

עַל הַיְּשׁוּעָה אֲשֶׁר מָתְקָה וְעָרְבָה, כִּי / עַל כָּל תְּשׁוּעַת אֱנוֹשׁ עָדְפָה וְסָרָחָה.

לוּלֵי אֳלֹהִים אֲשֶׁר תָּמַךְ בְּרוּחִי – אָז / מֵרֹב שְׂמָחוֹת וְגִיל עָפָה וּפָרָחָה.

אֹמַר וְאֶצְלַח בְּשִׁירִי, כִּי נְתִיבָתִי / יוֹם הַקְּרָב יִשְּׁרָה רַגְלַי וְהִצְלִיחָה.

אָשִׂים אֱלֵי לֵב אֲשֶׁר יַעַשׂ בְּכָל יוֹם לִי / הָאֵל פְּלִיאוֹת וְאֶתְבּוֹנֵן וְאַשְׁגִּיחָה,

וּכְמוֹ בְּקֶדֶם וְיָם כָל יוֹם אֲנִי נוֹשָׁע – / כָּכָה בְקֶדֶם וְיָם יָדָךְ אֲשַׁבֵּחָה,

וּבְנוֹת זְמִירוֹת, לְפִי בָנִים וּבָנוֹת, אֶל / צָפוֹן וְיָמִין, בְּיַד שָׁרִים אֲשַׁלֵּחָה.
