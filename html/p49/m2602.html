חֳדָשִׁים מִשְּׁנֵי

וקאל ירתי בעץ אקארבה מן רהט לוי

ואמר קינה שבה קונן על אחד מקרוביו משבט לוי


חֳדָשִׁים מִשְּׁנֵי רֹגֶז חֲרוּצִים / וְחַיִּים כַּאֲשֶׁר צָרַי חֲפֵצִים,

וְיִרְאַת כָּל אֲשֶׁר יִרְאֶה, כְּאִלּוּ / אֲנִי טָלֶה וְכָל רוֹאַי שְׁחָצִים,

וְנוּם נָפוֹג, וְלֵב רַגָּז וְנָמוֹג / כְּאִלּוּ מָלְאָה עַרְשִׂי בְּקוֹצִים,

וְשׂבַע מֵאֲשֶׁר אָבוּס, וְשֵׁמַע / אֲשֶׁר אָקוּץ, וְלֹא שֵׁמַע עֲלָצִים.

שְׁמַע, לִבִּי, מְדָנִים מִלְשׁוֹנִי, / שְׁחוּטִים מַאֲמָרֵיהֶם כְּחִצִּים:

הֲתוּכַל לַעֲטוֹת בּוּשָׁה וְלִשְׁכַּב / בְּבֵית קָלוֹן עֲלֵי עֶרֶשׂ לְחָצִים –

וְהָאָרֶץ לְפָנֶיךָ רְחָבָה / וְלָךְ סוּסִים מְיֻזָּנִים אֲמֻצִּים!

הֱשִׁיבַנִי: הֲתִרְחַב לִי אֲדָמָה / וְאִם אֶרְכַּב בְּפָרָשִׁים מְאִיצִים –

וְהַנּוֹשְׂאִים בְּכָתֵף יַהְלְכוּן אַט / בְּסוּתִים מֵעֲלֵי כָתֵף חֲלוּצִים,

וְהָרוֹחְצִים בְּדַם אִשֶּׁה יְדֵיהֶם / פְּנֵיהֶם מִדְּמֵי אִישׁוֹן רְחוּצִים.

הֲכָל שָׁנָה בְּמִשְׁפַּחְתִּי מְחִתָּה / וּמִיָּמִים בְּבֵית אָבִי פְּרָצִים

וְתָמִיד אֶחֱזֶה אַחַי, וְהֵמָּה / בְּלִבָּם, לֹא בְּגוּפוֹתָם, מְחוּצִים?

לְךָ דוֹדִי וּבֶן-דּוֹדִי דְמָעַי / בְּעַפְעַפַּי לְקִינוֹת פִּי מְרִיצִים

וְאֶבָּהֵל וְאֶבָּעֵת כְּאִלּוּ / סְעִפַּי כַּעֲדַת אַחְאָב נְפוֹצִים,

בְּעַד צַדִּיק אֲשֶׁר אָחַז דְּרָכָיו / וְהוֹסִיף, בַּאֲשֶׁר יִכְשַׁר, אֳמָצִים.

וְהָיְתָה כָל עֲבוֹדָתוֹ בְּחַיָּיו / נְתֹק מוֹטוֹת וְשַׁלֵּחַ רְצוּצִים

וְלִנְתֹץ אֶת נְתִיב רִשְׁעָה, וְלִבְנוֹת / יְסוֹדֵי דָת אֱלֹהָיו הַנְּתוּצִים,

וְלִזְרֹעַ, כְּמוֹ זַרְעוֹ, צְדָקָה / וְלָקוּם, כַּאֲבוֹתָיו, בַּפְּרָצִים.

בְּנוֹ, קוּמָה וְקַח מִפִּי דְּבָרִים, / הֲגִיתִימו ֹ בְּמַר נַפְשִׁי, נְחוּצִים

וְהֵם חַמִּים, לְמַעַן כִּי כְּתַבְתִּים / וְלִבִּי בִי כְּלַפִּיד אֵשׁ בְּעֵצִים.

וְדַע כִּי יֵשׁ עֲלֵי לִבִּי וְלֵב כָּל / נְבוֹן דָּבָר, כְּעַל לִבָּךְ, מְחָצִים,

וְכִי אֶבְלָךְ יְבַכֶּה אוֹהֲבֵי אֵל / וְכִי אֶבְלָךְ יְבַכֶּה כָל נְאוּצִים,

וּמִי שִׂמַּח בְּנֵי אֶרֶץ וּבָכָה – / לְבִכְיוֹ יִבְכְּיוֹן תַּמִּים וְלֵצִים.

קְרָא סִפְרִי וְהִתְנַחֵם, לְמַעַן / לְצוּרָךְ אֵין, בְּרֹב אֵבֶל, חֲפֵצִים,

וְלֹא יִסּוֹר לְהָרִיב עִם אֱלֹהִים / עֲלֵי רָעִים וּמִשְפָּטִים קְצוּצִים.

וְאוּלָם יֵשׁ לְכָל מַחְשָׁב מְחַשֵּׁב / וְכָל מִסְפָּר בְּתוֹךְ סֵפֶר קְבוּצִים

וְיוֹם יָבוֹא לְכָל אֶרֶז וְיִפֹּל, / וְעֵת קָצוֹר לְכָל חָצִיר, וְקִצִּים,

וְיֵשׁ שָׂכָר לְמִתְאַפֵּק בְּאֶבְלוֹ / וְיֵשׁ שָׂכָר לְנוֹהֶה עַל מְלִיצִים

וְלַהוֹלֵךְ יְשִׂיחוּן ‘בּוֹא בְּשָׁלוֹם’ / מְשָׁרְתֵי אֵל וְשָׂרִים נַעֲרָצִים.

וְלֹא הָיְתָה בְּךָ שִׁמְצָה – וְכָכָה, / בְּכֵס הוּכַן לְךָ, אֵין בּוֹ שְׁמָצִים,

וְהָיִיתָ בְּכָל פָּרִיץ מְמָאֵס – / וְלָכֵן אֵין בְּמוֹשָׁבָךְ פְּרִיצִים,

וְהָיִיתָ לְכָל חָמוֹץ מְאַשֵּׁר – / שְׁבָה עִם כָּל מְאַשֵּׁר הַחֲמוֹצִים!
