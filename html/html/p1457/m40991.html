ו [בחדש אשר ]

בַּחֹדֶשׁ / אֲשֶׁר אֻמָּה

כַּנִּדְגָּלוֹת אֲיֻמָּה

בּוֹ הָנְחֲלָה מֵרוּמָה

תּוֹרַת יְיָ תְּמִימָה


ובכן בחדש השלישי


הַשְּׁלִישִׁי / גָּבַר לְחֹפֶשׁ

גּוֹי יָצְאוּ מִכֶּפֶשׁ

דָּחָה מִלִּבָּם טֶפֶשׁ

בִּמְשִׁיבַת נֶפֶשׁ


לְצֵאת / הֲמוֹן מִי מָנָה

שִׁשִּׁים רֹאשׁ נִמְנָה

וְכֻלָּם קִבְּלוּ בֶאֱמוּנָה

עֵדוּת יְיָ נֶאֱמָנָה


בְּנֵי / זֶרַע יוֹנָתִי

זָכַרְתִּי בְּחַסְדִּי וֶאֱמוּנָתִי

חִנְּכָם נֶאֱמַן בֵּיתִי

בְּמַחְכִּימַת פֶּתִי


יִשְׂרָאֵל / טָסוּ בַכּוֹשָׁרִים

כְּעַל כַּנְפֵי נְשָׁרִים

יְשׁוּרוּן יַחַד מְאַשְּׁרִים

פִּקּוּדֵי יְיָ יְשָׁרִים


מֵאֶרֶץ / כְּחָשׁוּ בְכָל לֵב

קָרְאוּ לְבוֹחֵן כְּלָיוֹת וָלֵב

לָכֵן פִּקּוּדֵיהֶם מַפְגִּינֵי לֵב

וּמְשַׂמְּחֵי לֵב


מִצְרַיִם / מֵעֵת עָבְרָה

אוֹם יָפָה וּבָרָה

נָחֲלָה חָכְמָה וּגְבוּרָה

מִצְוַת יְיָ בָּרָה


בַּיּוֹם / [ס… ע… רה]

פְּאֵר שָׂם בְּרֹאשׁ נוֹטֵירָה

צָנִיף לְרֹאשׁ מוקטרה

[יֵרְאַת יְיָ טְהוֹרָה]


[הַזֶּה] בָּאוּ / קְהוּלִים בְּצַעַד

וְאִישׁ מֵהֶם לֹא נִמְעַד

[רָשַׁמְתָּ בְּעִנְיְנֵיהֶם] עֲדֵי עַד

עוֹמֶדֶת לָעַד


מִדְבַּר / שׁוּר מִתְקוֹמֶמֶת

וּמַלְכָּהּ מְרוֹמֶמֶת

שְׁבוּיַת שָׁמַיִם מְחֻכֶּמֶת

מִשְׁפְּטֵי יְיָ אֱמֶת


סִינַי / תָּאֵב בְּלַפִּידָיו

תֹּקֶף עוֹז לְהַנְחִיל יְדִידָיו

תָּקַף וְתִקֵּף י[….] גְּדוּדָיו

צָדְקוּ יַחְדָּיו
