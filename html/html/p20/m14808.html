אֵיךְ הָיָה כְּמוֹ צִיָּה

אֵיךְ הָיָה / כְּמוֹ צִיָּה / בֵּית קָדְשִׁי וְאוּלַמִּי,

וְסָר הוֹדִי / וּכְבוֹדִי / וְגַם עֻזִּי וְתַעְצוּמִי?

בְּרֹב פִּשְׁעִי / וְגַם רִשְׁעִי / וּזְדוֹנִי וַאֲשָׁמִי

אֻסַּרְתִּי / וְנִמְסַרְתִּי / בְּכַף קָמַי וְגַם ל[…]

אֲכָלוּנִי / וְכִלוּנִי / וְהֶגְלוּנִי מִמְּקוֹמִי

וּבָאֲפִיקִים / בְּכָל אֲרָקִים / נִזְּרוּ אָהֳלֵי לְאֻמִּי –

אַל תָּאִיצוּ לְנַחֲמֵנִי עַל שֶׁבֶר בַּת עַמִּי.


בְּיוֹם צָרָה / [וְיוֹם] עֶבְרָה / וגם אֲיֵלִילָה,

ואת אב[לי] / כְּמו אבל / עַל עִיר הַמְהֻלָּלָה,

[…] שָׁמַם / וְגַם דָּמַם / מְקוֹם שִׁירָה וּתְ[הִלָּה],

וְסָר נֵרִי / וּמְאוֹרִי / וּפַס זֶבַח וְגַם עוֹלָה,

וְכָל שִׂמְחָה / אֲזַי בָּרְחָה / וְהֵשַׁמּוּ שִׁיר גִּילָה,

וְיַד צָרַי / בְּכָל עָרַי / וַאֲנִי בא[ת…..] לה –

וּמְשַׂנְאַי / וּמַקְנִיאַי / נָתְנוּ [עֹל עַל] שִׁכְמִי.


[רָם], בְּאַפּוֹ / וּבְזַעְפּוֹ, / אֵשׁ הֶעֱבִיר בְּחוֹמוֹתַי,

[וי.ש.] / וְגַם הֵשַׁח / אֶל אֶרֶץ דִּמְעוֹתָי.

וְצַר טָרַף / וְגַם חֵרַף / וְהִרְבָּה כְלִמּוֹתַי,

וַיְחַלֵּל / מְקוֹם הַלֵּל / וְגַם הִרְבָּה שׁוֹמֲמוֹתַי,

וַיִּגְדַּע / מְקוֹם מַדָּע / וְכָל אַנְשֵׁי מְזִמּוֹתַי,

וְדִכְּאַנִי / וְהֶלְאַנִי / וְהִגְדִּיל אֶת נִקְמוֹתָי –

וְהֵן כִּלָּה / וְגַם בִּלָּה / אֱלֹהַי בְּשָׂרִי וַעֲצָמִי.


הֲמוֹן שָׂ[רַ]י / וְאַדִּירַי / בְּאַף סָפוּ וְגַם כָּלוּ,

גַּם חֲכָמַי / וּתְמִימַי / בְּמוֹ חֶרֶב שֻׁכָּלוּ,

וּבֶן תְּרַדְיוֹן – / בְּרֹב קִשְׁיוֹן / עִנּוּיָיו הִגְדִּילוּ.

וְעֵת נִתְפַּשׂ / בְּיַד נִמְאָס / וְצָרִים בּוֹ מָשָׁלוּ –

הוּא וְאִשְׁתּוֹ / וְגַם בִּתּוֹ / בְּיַד אוֹיֵב נָפָלוּ

בְּרֹב צָרוֹת, / וּגְזֵרוֹת / עֲלֵיהֶם אָז הִטִּילוּ;

וּשְׁלָשְׁתָּם, / בְּעֵת צֵאתָם, / קִבְּלוּ וְגַם סָבָלוּ

דִּין יוֹצְרָם / וַאֲבִירָם, / וְהוֹדוֹת לְאֵל מִלֵּלוּ –

לְזֹאת אֲקוֹנֵן / וְאֶתְאוֹנֵן / בְּעֵת שִׁבְתִּי וְעֵת קוּמִי.


מֵרֹב דִּוְיִי / וְגַם שִׁבְיִי, / אֱלֹהַי, שׁוּר וְהַבִּיטָה,

וְכָל אוֹיְבַי / וּמְרִיבַי – / אוֹתָם בְּאַף הַמְעִיטָה,

וְאֲלֵיהֶם / כְּפָעֳלֵיהֶם / הָשֵׁב וְגַם תִּשְׁפּוֹטָה,

וְכוֹס חֵמָה / בְּיוֹם נְקָמָה / עוֹד לָהֶם הַלְעִיטָה,

לְעַנּוֹתָם / לְהַכּוֹתָם / ורום למא[…] תנקיטה.

וְכָל אוֹנֵן / אֲזַי יְרַנֵּן / בְּעֵת אוֹתִי תַשְׁקִיטָה,

וְאָז נִשְׂמַח / בְּעֵת […] / תּוֹגַרְמִי וַאֲדוֹמִי.
