הֲלַכּוֹשֵׁל וְלַנּוֹפֵל

הֲלַכּוֹשֵׁל וְלַנּוֹפֵל תְּקוּמָה / וְלַנּוֹדֵד וְלָאוֹבֵד תְּנוּמָה?

וְאִם יֵשׁ לִי, בְּמוֹט רַגְלִי, וְאִם יֵשׁ / לְקַרְסֻלִּי אֲשֶׁר מָעַד, תְּרוּמָה?

בְּנַפְשִׁי כָּל אֲשֶׁר יָבוֹא בְּעֶזְרָה / וְנֶחָמָה לְכָל נֶפֶשׁ הֲלוּמָה,

וּבִנְפָשִׁים אֲשֶׁר בָּאוּ בְּיוֹם בּוֹא / מְהוּמָתִי וְרָאוּ בִי מְהוּמָה,

יְדַמּוּ כִי לְעוֹלָם אֵין תְּמוּרָה / לְיוֹם צָרָה, וְלֹא תֵשֵׁב שְׁמָמָה –

וְאֵל יָרִים אֲשֶׁר הֵשַׁח וְהִשְׁפִּיל, / וְלוֹ סוֹדוֹת וְדַרְכּוֹ נֶעֱלָמָה!

שְׁמַע מִלָּה, וְדַע כִּי לֹא לְקַלָּה / יְחַיֵּינִי אֱלֹהִים עַל אֲדָמָה,

וְקַבֵּל אֶת יְמֵי רָעָה בְּלֵב טוֹב / וּפֶה סָגוּר וּמַחְשָׁבָה חֲסוּמָה,

וּמִשּׁוֹנִים תְּהִי אָזְנָךְ כְּבֵדָה, / וּמִזּוֹנִים תְּהִי עֵינָךְ עֲצוּמָה,

כְּמוֹ חָפְנִי בְּשִׁלֹּה, אוֹ כְזִמְרִי / בְּמִדְיָנִית, וְאוֹנָן בַּיְבָמָה.

שְׁמַע מִלָּה, וְהַגְדֵּל יָהּ בְּבַיִת / וּבַדֶּרֶךְ וּבִשְׁכִיבָה וְקִימָה,

וְהִבָּהֵל וְהֵרָחֵף וְגוּרָה, / וְהֵחָרֵד וְהִמָּלֵא בְּאֵימָה

לְאֵל רִאשׁוֹן בְּאֵין מוֹלִיד וְנוֹלָד וְחַי לָעַד בְּלִי דוֹמֶה וְדוֹמָה,

וְלִמְצֹא אֶת רְצוֹנוֹ לֵךְ וְשׁוּבָה, / וּמִירֹא גַעֲרָתוֹ שֵׁב וְקוּמָה.

אֲנִי הָאִישׁ אֲשֶׁר יָרַד בְּתוֹךְ יָם, / וְיֵשׁ לַיָּם בְּרֹב שָׁנָה נְהָמָה,

וּבָחַרְתִּי לְרִדְתּוֹ אָב, לְמַעַן / בְּאָב לֹא יִשְׁבְּרוּ גַּלָּיו בְּחֵמָה,

וְנִמְשַׁכְנוּ בְּמִשּׁוֹטוֹת, וְרוּחַ / תְּנוֹפֵף שִׁבֳּלֵי שָׂדֶה וְקָמָה,

וְהִנֵּה יָם כְּעַבְדִּי יַעֲשֶׂה אֵת / אֲשֶׁר אֶחְפֹּץ, וְהָרוּחַ כְּאָמָה,

וְשָׁמַיִם כְּסַפִּירִים טְהוֹרִים, / וְהַמַּיִם כְּשֶׁמֶן טוֹב וְחֵמָה.

וּמַלָּחִים יְחַוּוּ זֶה לְזֶה אֶת / אֲשֶׁר רָאוּ בְּיָם מִתַּעֲלוּמָה,

וְזֶה יַזְכִּיר דְּבַר חַיָּה גְדוֹלָה / וְנִפְלָאָה, וְזֶה יַזְכִּיר בְּהֵמָה,

וְרַב חוֹבֵל יְשִׁיבֵם: ״לֹא יְדַעְתֶּם / פְּלִיאַת יָם, וְלֹא שַׂחְתֶּם מְאוּמָה!

וְשָׁם חַיָּה, שְׁמָהּ קָרְחָה, אֲשֶׁר הִיא / בְּכָל חַיּוֹת מְצוּלָה לֹא עֲמוּמָה,

וְכָל חַיָּה אֲשֶׁר חַיּוֹת שְׁאֵרָהּ – / בְּפִגְעָהּ בָּהּ אֲכוּלָה אוֹ הֲמוּמָה,

וְהוֹפֶכֶת לְפִיהָ כָּל סְפִינָה, / וְצִי אַדִּיר כְּמֶרְכָּבָה רְתוּמָה״.

וְלֹא כִלָּה לְסַפֵּר פָּעֳלָהּ עַד / אֲשֶׁר בָּאָה, וְהִיא גוֹעָה וְהוֹמָה.

שְׁמַעְתִּיהָ – וְהָיָה קוֹל הֲמֻלָּה, / וְקוֹל רַעַם בְּאָזְנַי קוֹל דְּמָמָה,

וְהִבַּטְתִּי אֱלֵי צוּרָה אֲיֻמָּה, / אֲבָל הָיְתָה בְּתֹכֶן דָּג רְקוּמָה,

כְּלִוְיָתָן אֲשֶׁר סֻפַּר, כְּנָחָשׁ / עֲקַלָּתוֹן, כְּמוֹ תָמָר בְּקוֹמָה,

וְרֹאשׁ לָהּ כָּאֳנִי שַׁיִט בְּמִדָּה, / עֲלֵי פָנִים מְנֻשָּׁאִים כְּרָמָה,

וְעֵינַיִם כְּמַעְיָנִים, וְנָחִיר / כְּמוֹ כִבְשָׁן, וְלָהּ רַקָּה כְּחוֹמָה,

וּפֶה עָמֹק וְרָחָב כַּמְּעָרָה, / תְּעַר נָהָר אֱלֵי תוֹכוֹ בְּצָמָא,

וּבִשְׂפָתוֹת כְּנֹאד עַל נֹאד נְפוּחִים, / וּבֵינֵיהֶם כְּפִי תַחְרָא פְּרוּמָה,

וְגוּף לָבָן, וְגַב יָרֹק, וְצַוָּאר / כְּמוֹ מִגְדָּל, וּבֶטֶן כַּעֲרֵמָה,

וּבִסְנַפִּיר כְּמוֹ חֶרֶב לְטוּשָׁה, / וְקַשְׂקֶשֶׂת כְּסוֹחֵרָה אֲדֻמָּה,

וְגִזְרָתָהּ לְמַשְׁגִּיחִים כְּמוֹ צוֹר / אֲשֶׁר הָיְתָה בְּלֶב הַיָּם כְּדֻמָּה.

וְשָׁטָה מִסְּבִיבוֹת לַסְּפִינָה, / וְאֵין מֵנִיד וְאֵין נוֹאֵם נְאִימָה,

וְהִגִּיעָה לְצִדֶּיהָ וְעָמְדָה / עֲלֵי זָנָב כְּאֶרֶז אוֹ שְׁדֵמָה.

אֲזַי נָמֵס לְבַב כָּל אִישׁ כְּדוֹנַג / וְכַמַּיִם וְשִׁבֹּלֶת זְרוּמָה.

וְשִׁוִּיתִי וְדוֹמַמְתִּי לְנַפְשִׁי / כְּרָחֵל יוֹם טְבֹחַ נֶאֱלָמָה.

וְזָכַרְתִּי אֱלֹהַי – וַאֲחֵרִים / לְנֶגְדִּי יִזְכְּרוּ אֵל כַּאֲשִׁימָה.

וְאָמַרְתִּי: ״אֱמֶת, כָּזֹה וְכָזֶה / עֲווֹנוֹת יִלְכְּדוּ נֶפֶשׁ אֲשֵׁמָה,

וְאִם נִבְלַע בְּיָם יוֹנָה וְהוּקָא / בְּיַבָּשָׁה לְצִדְקָתוֹ – אֲנִי מָה?

אֱלֹהַי, הַאֲרֶךְ לִי, כִּי אֲשֶׁר בָּא / הֱקִיצַנִי – וְהָיִיתִי בְּנוּמָה,

וְאִם זֶהוּ גְּמוּל יָדִי, תְּהִי בוֹ / סְלִיחַת מַעֲוָתִי הָעֲצוּמָה!״

אֲזַי שָׁקְעָה בְּלֶב יַמִּים, כְּחַיִל / אֲשֶׁר הָאֵל בְּיָם יָרָה וְרָמָה,

וְתַחַת לָאֳנִי בָאָה לְהָפְכוֹ, / וּמֵת כָּל לֵב וְאָבְדָה כָל נְשָׁמָה –

אֲבָל גָּעַר אֱלֹהִים בָּהּ, וְשָׁבָה / כְּרֶגַע אֶל מְצוּלוֹת יָם פְּנִימָה,

וְהוֹשִׁיעַ לְתוֹלָעִים קְטַנִּים / עֲלֵי שׂוֹכָה בְּמוֹ זֶפֶת קְרוּמָה,

וְהֵשִׁיב מִשְּׁאוֹל מֵתִים, וְהִצִּיל / מְבֻלָּעִים בְּיָד נִשֵּׂאת וְרָמָה.

וְתָמְהוּ כָל נְבוֹנֵי יָם, וְאָמְרוּ: / ״בְּמַה זֶּה נִמְלְטוּ אֵלֶּה, וְלָמָּה?

וְלֹא הָיְתָה פְּלֵטָה לָאֳנִיָּה / לְפָנִים מִיְּדֵי קָרְחָה זְעוּמָה!״

עֲנִיתִים: ״כֵּן גְּאֻלַּת אֵל לְיוֹדְעֵי / כְּבוֹד הוֹדוֹ מְהֻדֶּרֶת שְׁלֵמָה,

וְהוּא עוֹשֶׂה לְמַכִּירָיו תְּשׁוּעָה, / וְהוּא עוֹשֶׂה בְּמַרְגִּיזָיו נְקָמָה,

אֲשֶׁר לוֹ יָם וְחַיָּתוֹ, וּמֵימֵי / תְהוֹם רַבָּה, וְתוֹלָם עַל בְּלִימָה.

וּמַה חַיָּה לְנֶגֶד צוּר נְסָכָהּ / וְשָׂם לָהּ עֹז, וְשָׂם לָהּ תַּעֲצוּמָה?

אֲנִי אוֹדֶה לְאֵל תּוֹדַת גְּאוּלִים, / תְהִי עַל פִּי מְשֻׁנֶּנֶת וְשׂוּמָה,

וְאוֹדֶה כִּי לְצוּרִי אֵין תְּחִלָּה / כְּנִבְרָאִים, וְכִי אֵין לוֹ חֲתִימָה,

וְאוֹדֶה כִי אֱמֶת הִיא הַתְּחִיָּה, / וְכִי מֵתִים יְקִיצוּן מֵאֲדָמָה,

וְכִי מֹשֶׁה וְתוֹרָתוֹ, אֲשֶׁר הִיא / בְּיָדֵינוּ, אֱמֶת וּבְתֹם רְשׁוּמָה,

וְכִי דִבְרֵי חֲכָמֵינוּ יְשָׁרִים, / וְתַלְמוּדָם וּמִשְׁנָתָם נְעִימָה,

וְיֵשׁ שָכָר בְּעוֹלָם בָּא לְזַכִּים, / וְתַגְמוּלִים לְמֵתִים עַל עֲלוּמָה,

וְאֵל שַׁלִּיט בְּיַבֶּשֶׁת וְיַמִּים / וּבִשְׁחָקִים וְעַל עַיִשׁ וְכִימָה,

וְיִרְאָתוֹ עֲלֵי פָנַי נְתוּנָה, / וְתוֹרָתוֹ בְּתוֹךְ לִבִּי מְשׂוּמָה!״
