רהיט [בְּחָכְמָתְךָ אָזַרְתָּ הוֹד בְּאֵין חוֹצֶה]

סימן: אלף־בית

כה

         בְּחָכְמָתְךָ אָזַרְתָּ הוֹד בְּאֵין חוֹצֶה וּפוֹצֶה

         וְיַעַן לֹא יִמְצָאֲךָ מוֹצֵא

         וְהַחָכְמָה מֵאַיִן תִּמָּצֵא.

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


5         בְּחָכְמָתְךָ בִּנְיָינֵי תַּחַת וּמַעַל מֵאֵין מוּסַד בְּנוּיִים

         בְּחַבְלֵי שְׂעָרָה בִּזְרוֹעֲךָ תְּלוּיִים

         לֹא תִמָּצֵא בְאֶרֶץ הַחַיִּים.

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ גֵּאֶה בְּאֵין מָקוֹם תָּקַעְתָּ אֳהָלֶךָ

10       וּמוּסַד מָקוֹם כָּל מוּסַד עָלֶיךָ

         בְּרֹב חָכְמָתְךָ הִרְבֵּיתָ חֵילֶיךָ

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ דֹּק וְחֹג כַּיְרִיעָה נָטִיתָ

         וּבְאַוֵּירָם חוּג אֲדָמָה כְּחַרְדַּלָּה נָחִיתָ

15       בְּרֹב חָכְמָתְךָ וּבִתְבוּנָתְךָ עָשִׂיתָ

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ הִצַּבְתָּ מִסְתּוֹרִים מְקוֹם מְנוּחָתֶךָ

         וּבְחַדְרֵי חֲדָרִים נָתַתָּ מִשְכְּנוֹתֶיךָ

         עַל יְפִי חָכְמָתֶךָ

20      מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ וּבִתְבוּנָתְךָ תִּכַּנְתָּ כָּל לְהַרְאוֹת נוֹרְאוֹתֶיךָ

         לְסַפֵּר דוֹר לְדוֹר מַלְאֲכוֹתֶיךָ

         הַשּׁוֹמְעִים אֶת חָכְמָתֶךָ

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


25       בְּחָכְמָתְךָ זַע כָּל נָבוֹן וְעָמַד מִבֹּא חֶקֶר תְּכוּנָתֶךָ

         וְנִבְהַל כָּל חָכָם מֵהַשִּׂיג תְּהִלָּתֶךָ

         עַל דְּבָרְךָ וְעַל חָכְמָתֶךָ

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ חֲכָמִים יוֹדוּ כִּי אֵין זוּלָתֶךָ

301  וְאֵיךְ יוּכְלוּ לְהַשִׂיג וְהֵמָּה בְּרִיוֹתֶיךָ

         חֲצִי מַרְבִּית חָכְמָתֶךָ

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ טִהַרְתָּ וְזִקַּקְתָּ שִׁבְעָה בְשִׁבְעָה מוּצָקִים נְכוֹנִים

         פְּנִינֵי גַב לֶכְתָּם לְטוּשִׁים לְבָנִים

35       וּמֶשֶׁךְ חָכְמָה מִפְּנִינִים

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ יָצַרְתָּ שְׁלשָׁה מִמַּעַל וּשְלשָׁה מִתַּחַת מַהֲלָכָם

         יַעֲבֹר מַלְכָּם בֵּינֵיהֶם וְתוֹכָם

         וְהַחָכְמָה תָעֹז לְחָכָם

40       מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ כְּסִילֵי שַׁחַק וּגְדוּדָיו כּוֹנַנְתָּ פְּלִיאֵי תְמוּנָה

         מֵהֶם שָׁבִים וּמֵהֶם מְהַלְּכִים בֶּאֱמוּנָה

         בְרוּחַ חָכְמָה וּבִינָה

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


45       בְּחָכְמָתְךָ לַהַב וְזֶרֶם וְרוּחַ וְעָפָר גִּזְרָתָם גְּזוּרָה

         בְּהִתְחַלְּקָם בְּבָתֵּי מַלְכוּתָם וְשִׁפְלוּתָם בְּחֶבְרָה

         עִמּוֹ עֵצָה וּגְבוּרָה

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ מִדַּת קַו זְמַנֵּיהֶם שַׁתָּה מְשֻׁונָּה

50       מֵהֶם חָשִׁים וּמֵהֶם שׁוֹהִים וְעוֹשִׂים מְלַאכְתָּם בֶּאֱמוּנָה

         בְּחָכְמָה וּבִתְבוּנָה

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ נִהוֹלֵי חֵילָם מִיָּם לְקָדִים יֻסְּדוּ

         עַל פִּיךָ נוּחָם וּתְנוּדוֹת יָנִידוּ

55       אֲשֶׁר חֲכָמִים יַגִּידוּ

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ סוֹבֵב חֶרֶס עַל פְּנֵי צָפוֹן שִׁשָּׁה וְאֵין נִסְתָּר מֵחַמָּתוֹ

         וְהִלּוּכוֹ עַל פְּנֵי דָרוֹם שִׁשָּׁה וּמִי יַעְמֹד לִפְנֵי קָרָתוֹ

         מֵכִין תֵּבֵל בְחָכְמָתוֹ

60       מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ עֲשׂוֹתוֹ שְׁלשִׁים בְּכָל מַזָּל וּמַזָּל יַעֲבִירֵם

         לְמַלְּאֹת שָׁנָה כֻּלָּם יִגְמְרֵם

         שִׂפְתֵי חֲכָמִים תִּשְׁמְרֵם

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


65          בְּחָכְמָתְךָ פְּקוּדוֹת אַרְבַּע תְּקוּפוֹת בְּאַרְבַּעַת רְבָעָיו נוֹדַעַת

         לַעֲשׂוֹת אַרְבָּעָה זְמַנִּים קֹר וָחֹם וְקַיִץ וָחֹרֶף חוֹנָה וְנוֹסַעַת

         בְּחָכְמָה וּבִתְבוּנָה וּבְדַעַת

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ צָעֲדָה בְכָל יוֹם מַעֲלָה יְדוּעָה תְמִימָה

70       זָורְחָה וּבוֹאָהּ לַעֲשׂוֹת חֹק יוֹם וְלֵיל בְּתוּמָה

         גַּם זוֹ רָאִיתִי בַחָכְמָה

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ קְבִיעוּת הִלּוּךְ חַמָּה וּלְבָנָה לְמוֹעֲדִים וְשָׁנִים מְסֻויָּמִים

         וַחֲמֵשֶׁת כּוֹכָבִים בִּזְמַנִּים מְשֻׁונִים בְּמַזָּלוֹת רוּם מַשְׁלִימִים

75          וְהֵמָּה חֲכָמִים מְחֻכָּמִים

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ רְדוֹתְךָ בַּגִּלְגַּל וְנֹחָם וְנֹעָם

         בַּחֲזוֹת סוֹבָם אֲחוֹרַנִּית וְעָומְדָם וְנָוסְעָם

         תָּבֹא חָכְמָה בְלִבֶּךָ וְדַעַת לְנַפְשְׁךָ יִנְעָם

80      מֵעֵין כָּל חַי נֶעְלַמְתָּ.


         בְּחָכְמָתְךָ שְׁבִילֵי לְבָנָה פְּעָמִים בַּאֲרוּכָה פְּעָמִים בִּקְצָרָה

         אַךְ מַזָּלוֹת מֵעַל בְּחָדְשָׁה גּוֹמְרָה

         הֲלֹא חָכְמָה תִקְרָא

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.


85       בְּחָכְמָתְךָ תַּם הַכֹּל וְקָם בְּחֶפְצְךָ וְתַשְׁלִים מְזִמָּה

         וּמִי בִלְתְּךָ יַשִּׂיג לְהַגִּיד נְתִיבוֹת חָכְמָה וְעָרְמָה

         מִי יְסַפֵּר שְׁחָקִים בְּחָכְמָה

         מֵעֵין כָּל חַי נֶעְלַמְתָּ.
