[איגרת אל חסדאי בן שפרוט]

אַפְקִיד חֲמָסִי בְּיַד אֵל / וְאֵלָיו עֵינַי תִּבְכֶּינָה,

בְּמוֹ־פִי אֶתְחַנֵּן לְדוֹרֵשׁ הָעֲשׁוּקִים

בְּעוֹד נִשְׁמָתִי בִי עַל רִיבִי אֲשַׁוֵּעַ.

וְאִם סֶלֶף תְּדַבֵּרְנָה שְׂפָתַי – / הַוָּתִי הֱצִיקַתְנִי,

וְאִם יֶהְגֶּה חִכִּי נָלוֹז – / בְּלִבִּי רוֹבֶצֶת שַׁלְהֶבֶת.

עַל כֵּן לֹא טוֹב לְרוֹזְנִים / לִבְחֹן אִמְרֵי מָרֵי נָפֶשׁ,

כִּי קֶטֶב תּוּגָתָם / יְשַׁכְּרֵם לִרְגָעִים,

הֲלֹא עֲלֵיהֶם לִירֹא / מִגְּזֵרַת נִסָּיוֹן!

משֶׁה אִישׁ הָאֱלֹהִים / אֲשֶׁר לֹא קָם כָּמוֹהוּ בִּנְבִיאֵי צֶדֶק

הֲלֹא בְקֹצֶר נַפְשׁוֹ / וּבִמְרִירוּת לִבּוֹ

פָּצוּ שְׂפָתָיו / אֲשֶׁר לֹא כַדָּת

תָעָה לְבָבוֹ / וְתוּשִׁיָּה נִדְּחָה מִמֶּנּוּ,

כַּאֲשֶׁר דִּבֶּר מְשִׁיחַ יְיָ:

וַיַּקְצִיפוּ עַל מֵי מְרִיבָה / וַיִּרַע לְמשֶׁה בַּעֲבוּרָם

כִּי הִמְרוּ אֶת רוּחוֹ / וַיְבַטֵּא בִּשְׂפָתָיו.

וְאַף כִּי הַכֶּלֶב הַמֵּת / אֲשֶׁר כָּמוֹנִי –

לְהַבְדִּיל בֵּין הַקֹּדֶּשׁ וּבֵין הַחֹל / וּבֵין הַטָּמֵא וּבֵין הַטָּהוֹר


שְׁמַע נָא דְבָרַי, / אֲדוֹנִי הַנָּגִיד הַגָּדוֹל!

מֵחֹמֶר קֹרַצְתָּ כָמוֹנִי / וְעוֹשֵׂנִי עוֹשֶׂךָ,

וְעָפָר יְסוּדָתֵנוּ / וְצֵל יָמֵינוּ עֲלֵי אָרֶץ,

וְכַבִּירִים אֲשֶׁר יְשֻׂגָּבוּ / וְדַלִּים אֲשֶׁר יִדְאָבוּ

יַחַד עַל עָפָר יִשְׁכָּבוּ!

אֲנִי הַגֶּבֶר הַנִּדְאָב, / בְּלֹא עֵת יוֹמִי עָרַב

וְעוֹד לֹא יוֹמִי – מָלְאוּ יָמַי.

מָה אָרְכוּ רִגְעֵי לֵילִי / אֲשֶׁר לֹא נִבְקַע בָּם שַׁחַר!

גַּם יְרֵחִי הוּעַם / בַּעֲבִי מַאֲפֵל

וְכוֹכְבֵי נִשְׁפִּי / אָסְפוּ נָגְהָם.

עַל אֵלֶּה לְפוֹעֲלִי / אֶתֵּן צֶדֶק.

אֲנִי הַגֶּבֶר הַנִּדְוֶה / אֲשֶׁר בְּלִי בֹסֶר קָהוּ שִׁנַּי –

לִי עָרְכוּ טַעֲמֵי מָוֶת / וּלְרַגְלַי כָּרוּ שַׁחַת.

אָסְרוּ רַגְלַי בְּכַבְלֵי חָרוֹן / וְהִשִּׂיקוּ קְרָבַי בְּמוֹקְדֵי עוֹלָם

וּמוֹגְגַי בְּיוֹם אֵידִי / אָמְרוּ “הֶאָח!”

וְאִם שַׁדַּי הִרְחִיק יֶשַׁע – / אֵין נָקִי מִפֶּשַׁע,

וְאִם בְּעֵת צָרָה שָׁמָּה יְיָ.

הֲלֹא יוֹם וָלַיִל / מְמֻנִּים אֲנְשֵׁי חַיִל

כְּעוֹרְבֵי מַשָּׂאוֹת.

וְאִם זְמָן יַאֲרִיכוּ – / מִשְׁפָּט יַעֲרִיכוּ

וְנִגְלָה צֶדֶק עוֹלָמִים.

לֹא יִמָּשְׁכוּ יְמֵי אֱנוֹשׁ / עַד בּוֹא עֵת לַעֲנשׁ

וְיִצְדַּק פֹּעַל אֱלֹהִים.

וְלוּ לֹא נִקְבַּר / עַד רְאוֹת אֵיךְ יִפֹּל דָּבָר

וּמַה מִּפְעַל הַגַּלְגַּל!

וְאִם אֵין מְחַוֶּה / מַה הוּא וּמֶה הֹוֶה –

דַּיָּן בַּשַּׁחַק נִצָּב סֶלָה!

עַל כָּל כְּלִמּוֹתַי / אֵלֶּה תַנְחוּמוֹתַי.

מָה אָעִיד וּמָה אֲדַמֶּה / לְלִבִּי אֲשֶׁר יֶהֱמֶה –

וְיִשְׁתֹּק מִשַּׁאֲגוֹתָיו?

חָשַׁבְתִּי כָל פֶּרֶץ / מִנִּי שׂוּם אָדָם עֲלֵי אָרֶץ –

וְכָמוֹנִי בְּכָל כּוֹאֲבִים / לֹא מָצָאתִי.

לְ“אֵין כָּל חָדָשׁ” / נוֹצַרְתִּי מַכְחִישׁ –

כִּי כָמוֹנִי לֹא הָיָה / גַּם לֹא גָז חִישׁ!

הֲלֹא עֵינַי רוֹאוֹת, / כִּי לִהְיוֹת לְאוֹת

בָּאתִי לָעוֹלָם.

לָכֵן מִתֹּךְ וּמֵחָמָס / לְבָבִי יִמַּס

וְכַיָּם יֶהֱמֶה

וְאִם אַחַי תַּנִּינִים / וְרֵעַי יְעֵנִים –

לֹא הִשִּׂיגוּ בְנִיהֶם,

כִּי מִתַּמְרוּרָם אַגְדִּיל תַּמְרוּרִים / וּמִבְּכִי יַעְזֵר אֲמָרֵר בַּבֶּכִי.

גַּם אֲנִי אַצְרִיחַ / וְאַרְחִיב קָרְחָתִי,

עֲצָמַי יִבְכָּיוּן / אֵלֶּה עַל אֵלֶּה

וּנְתָחַי יֶהֱמָיוּן / קְצָתָם עַל קְצָתָם.

וְאַף אֲנִי בְּמוֹ פִי / אֲקוֹנֵן עַל כֻּלָּם.

מֵחִי מִלְּחוּמִי אֲנִי אֶגְרֹד / וּמָגִנַּי אֲנִי אֲגַדֵּעַ,

אֶזְרָעֵם בְּרִגְבֵי עָפָר / וְאַמְטִיר עָלֵימוֹ מֵימֵי עַפְעַפַּי

וְתָמִיד אֲרַוֵּם בְּדִמְעָתִי:

אוּלַי חֲמָסִי יַצְמִיחַ פֶּרִי / וְיַשְׁבִּית מְרִי מִבֵּית הַמֶּרִי.

וּבְעֵת תַּלְמֵי צִיָּה בְּדִמְעָתִי יִרְוָיוּן – / עַל חֲמַס לְשַׁדִּי יִבְכָּיוּן.

הֵן מִדֵּי בַת עֵינִי דָּמֹעַ תִּדְמַע / הֲלֹא שָׂהֲדִי בַמְּרוֹמִים שָׁמֹעַ יִשְׁמַע.

אֲנִי אִישׁ דְּוֵה־לֵב, / הַיָּגֵעַ תָּמִיד עַל חֲמָסִי,

וְלֹא אָשׁוּב, עַד יִכְלֶה שְׁאֵרִי / וְיָשׁוּב לַעֲפָרוֹ.

אֶשָּׂא מוּסַר כְּלִמָּתִי, אֶסְבֹּל מַדּוּחַי – / וַאֲנִי יָדַעְתִּי: גּוֹאֲלִי חַי!

הוּא יָרִיב אֶת רִיבִי / וְיַעֲשֶׂה מִשְׁפָּטִי

וְאִם דִּינִי עַתָּה יִתְמַהְמַהּ – / אֲקַוֶּה לְיוֹם תּוֹכֵחָה,

לְיוֹם קוּמִי לְעַד, / יוֹם אֲשֶׁר נַעַמְדָה יַחַד,

יוֹם אֲשֶׁר בַּעַל מִשְׁפָּטִי יִגַּשׁ אֵלַי,

יוֹם אֲשֶׁר לֹא יָנוּס נָס / וְלֹא יוֹעִיל בּוֹ כָל הוֹן,

יוֹם אֲשֶׁר יִשְׁווּ בוֹ שׁוֹעַ וְדַל,

וְרֹאשׁ לְזָנָב לֹא יֹאמַר: חֲדַל,

יוֹם אֲשֶׁר לֹא יִתָּכְנוּ בוֹ תְנוּאוֹת

וְלֹא יָרִיבוּ כַבִּירִים בְּכֹחַ הַזְּרוֹעַ.


הֶאָנֹכִי לְךָ אֶהֱמֶה? / וְאִם עֵינַי לְךָ תִבְכֶּינָה?

וְאִם נַפְשִׁי לְךָ תִשְׁתַּפֵּךְ / וְרוּחִי לְךָ יִכָּנַע?

הֲלֹא עַל חֲמָסִי / אֲנִי בוֹכֶה

וְעַל מִשְׁפָּטִי / עֵינִי דֶּמַע דּוֹלֶפֶת

וְעַל סֶלֶף דִּינִי / תִּתְעַלַּפְנָה עֲצָמַי,

גַּם לְבָבִי יִמַּס / וְיִחַר כְּבֵדִי,

מֵעַי יְחִילוּן / וְכִלְיוֹתַי תִּכְלֶינָה

וְעַפְעַפַּי יִזְּלוּ מַיִם / וּבָבוֹתַי לַיְלָה וְיוֹמָם לֹא תְפוּגֶינָה,

עַד אֲשֶׁר עַנְנֵי שַׁחַק / לֹא הִשִּׂיגוּנִי בְנִטְפֵיהֶם

וְלֹא נְצָחוּנִי / בְּרַעַף רְבִיבֵיהֶם.

כִּי עֵינַי תִּטֹּפְנָה כַגִּתּוֹת – / וּרְסִיסֵי מָרוֹם מְחַלְּקִים לְעִתּוֹת:

עֵת טֹהַר – וְעֵת בֹּהַר / עֵת בֹּקֶר – וְעֵת עָבוֹת.

אֶפֶס לֹא כֵן אָנֹכִי, / כִּי אִם בְּכִי לִקְרַאת בֶּכִי,

וַתִּהְיֶינָה עֵינַי כִּנְחָלִים אֲשֶׁר לֹא יִבְגֹּדוּ

וְכַעֲיָנִים אֲשֶׁר לֹא יְכַזְּבוּ מֵימֵיהֶם.

וְאִם אָמַרְתִּי תֵּת לִי דֳמִי / וּלְעֵינַי הֲפוּגוֹת –

עִצְּבוֹנֵי חֲמָסִי / יָעִירוּ דִמְעוֹתַי.

וְאִם בֶּאֱמֶת תִּבְכֶּינָה / וּבְתָמִים תִּטֹּפְנָה,

הֲלֹא יוֹצֵר עַיִן / יַשְׁקִיף וְיֵרֶא,

הֲלֹא דָלֹה יִדְלֵם / וְיוֹפִיעַ לַעֲשׂוֹת מִשְׁפָּט מִדָּמִי.

זֹאת נוֹדַע מִנִּי עַד, / כִּי כָל מְשַׁוֵּעַ נֶעְתָּר

וְכָל שׁוֹפֵךְ שִׂיחַ יִשָּׁמַע.

מִי הוּא גָזוּל נֶעֱזָב? / וְאֵיפֹה דִמְעוֹת עֲשׁוּקִים נִכְחָדוּ?

זֹאת תִּקְוָתִי בְעָנְיִי / וְאֵלֶּה תַּנְחוּמוֹתַי.


אֲנִי אֶעֱרֹךְ מִשְׁפָּט / וְיָדַעְתִּי כִּי אֲנִי אֶצְדַּק.

כֹּה חַקּוֹתָ לִי בִּתְשׁוּבַת אִגַּרְתִּי / אֲשֶּר שָׁלַחְתָּ אֵלַי לֵאמֹר:

"אִם הֶעֱוֵיתָ – / כְּבָר הֲבִיאוֹתִיךָ לְמוּסָר,

וְאִם לֹא הֶעֱוֵיתָ – / כְּבָר הֲבִיאוֹתִיךָ לְחַיֵּי הָעוֹלָם הַבָּא".

הֲזֹאת חָשַׁבְתָּ לְמִשְׁפָּט? / הַלְהוֹכַח מִלִּין תַּחְשֹׁב?

הֲלֹא אֱלֹהִים יַחֲקָר־זֹאת, / הַנּוֹטַע אֹזֶן הֲלֹא יִשְׁמַע,

אִם יוֹצֵר עַיִן הֲלֹא יַבִּיט!

וְכִי תַחְסֹם אֶת פִּי / בְּרוּחַ קָדִים

וְתַאֲלִים אֶת לְשׁוֹנִי / בְּהַבְלֵי מַהֲתַלּוֹת?

הֲלֹא מֵעֲפָרְךָ עֲפָרִי / וֵאלֹהֶיךָ אֱלֹהַי,

חוֹפֵשׂ כָּל תַּעֲלוּמוֹת / וְחוֹפֵשׂ כָּל חַדְרֵי בָטֶן.


וְעַתָּה שְׁמַע נָא, בֶּן אָדָם,

הֲלֹא אִישׁ אַתָּה / וּמִי כָמוֹךָ בְיִשְׂרָאֵל?

הֲלֹא לְךָ לָדַעַת אֶת הַמִּשְׁפָּט!

יֵשׁ עַל הַמַּשְׂכִּיל – / מֵאֹרַח הַמּוּסָר וּמִדֶּרֶך הָאֱמוּנָה,

בְּשָׁמְעוֹ נְכוֹחָה / שֶׁיּוֹדֶה עָלֶיהָ – / הֵן לוֹ, הֵן עָלָיו:

זֹאת תִּפְאֶרֶת נְדִיבִים / וְהִיא תְהִלָּתָם.

וְכִי תֹאמַר “אִם הֶעֱוֵיתָ – / וְאִם לֹא הֶעֱוֵיתָ”,

הֲנָכוֹן לִשְׁפֹּט עַל “אִם”, / הַיִּיטַב בְּעֵינֵי יְיָ?

וְאִם אַחֲרֵי לִבְּךָ נָטִיתָ / וּבְשִׁקּוּל דַּעְתְּךָ דַּנְתַּנִי,

לֹא נָכוֹן לַעֲשׂוֹת כֵּן.

הַאַף בְּסוֹד יְיָ עָמַדְתָּ / אִם תַּחַת אֱלֹהִים אַתָּה?

הֲבָאתָ עַד תַּעֲלוּמוֹת הָאָדָם / וְסֵתֶר מַצְפּוּנָיו תִּרְאֶה

הֲנִגְלוּ לְךָ סוֹדֵי לְבָבוֹת / וְחִקְרֵי כְלָיוֹת תַּחְפֹּשׂ?

הֲתִדְרשׁ אֶת קֶרֶב הָאִישׁ / וְיֵצֶר מַחֲשָׁבוֹת תִּרְאֶה

אֵיזֶה הוּא וּמִי הוּא זֶה?

הֲלֹא הוּא אֶחָד / אֲשֶׁר יֵאוֹתוּ לוֹ כָל אֵלֶּה,

אֲשֶׁר בְּכָל מָקוֹם יֶשְׁנוֹ / וְלֹא יִפָּלֵא מִמֶּנּוּ כָּל דָּבָר!


הֲלֹא עָכָן בֶּן־זֶרַח אֲשֶׁר מָעַל מַעַל בַּחֵרֶם וְנִכְשַׁל בְּלָקְחוֹ מִמֶּנּוּ —

הֵן אֵל כַּבִּיר עָנָה בוֹ וַיְעִידֵהוּ וְגַם נִלְכַּד בְּגוֹרְלוֹת צֶדֶק.

וּבְכָל זֹאת לֹא לֹא שָׁלְחוּ בוֹ יָד שׁוֹפְטֵי יִשְׂרָאֵל,

וְלוּלֵי יְהוֹשֻׁעַ בִּן־נוּן אֲשֶׁר בִּקֵּשׁ וַיִּתְחַנֵּן עֲדֵי תִתּוֹ תוֹדָה, לֹא הוּמַת.

כַּאֲשֶׁר נִמְצָא כָתוּב בַּסֵּפֶר "בְּנִי שִׂים נָא כָבוֹד לַיָי אֱלֹהֵי יִשְׂרָאֵל

וְתֶן לוֹ תוֹדָה!"

הִנֵּה בְאִישׁ חַיָּב לַהֶרֶג / לֹא נָגְעָה בוֹ יָד,

לֹא עַל פִּי יְיָ / וְלֹא עַל פִּי גוֹרָלוֹת,

כִּי אִם בְּהוֹדָאַת פִּיו.

וְאַף כִּי אֲנִי – / הָיוּ לִי עֵדֵי אֱמֶת / מַצְדִּיקִים אוֹתִי,

וְלוּלֵי אֲשֶׁר מִהֲרוּ רוֹדְפַי / לִנְקֹם נָקָם

וְלִשְׁפֹּךְ דַּם נָקִי בְחֵיקְךָ / לַעֲשׂוֹת לָהֶם שֵׁם –

קָמוּ עֵדַי וְהֵעִידוּ.

אֲבָל רְדָפוּנִי בְקִנְאָה וּבְתַאֲנָה / וָאֱהִי לָהֶם לְמַאֲכָל,

וָאֱהִי כְעוּגָה בְּלִי הֲפוּכָה / וּכְבִכּוּרָה בְּטֶרֶם קַיִץ

אֲשֶׁר יִרְאֶה הָרוֹאֶה אוֹתָהּ, / בְּעוֹדָה בְּכַפּוֹ יִבְלָעֶנָּה,

וְכִתְאֵנִים עִם בִּכּוּרִים, / אִם יִנּוֹעוּ – וְנָפְלוּ עַל פִּי אוֹכֵל.

יִשְׁפֹּט יְיָ הַשּׁוֹפֵט / הַיּוֹם בֵּינִי וּבֵין עוֹשְׁקַי

וְיֵרֶא וְיָרִיב אֶת רִיבִי!


שְׁמַע נָא, בֶּן אָדָם, / כִּי עוֹד יֵשׁ לְעַבְדְּךָ מִלִּים.

דַּע אַתָּה, אֲדוֹנִי, / כִּי אֱלֹהִים הָיִיתָ בְמִשְׁפָּטִי,

מִשְׁפַּט אֱלֹהִים שְׁפַטְתַּנִי, / דִּין עֶלְיוֹן דַּנְתַּנִי:

אֲשֶׁר הוּא הָעֵד – / הוּא הַדַּיָּן

וְלוֹ הַנִּסְתָּרוֹת / וְהוּא יוֹדֵעַ תַּעֲלוּמוֹת לֵב

וְלֹא צִוָּה כֵן לְנוֹחֲלֵי מִשְׁפָּט / לַעֲשׂוֹת פְּלִילָה.

וַאֲנִי יָדַעְתִּי / כִּי לֹא יְיָ הֱסִיתְךָ בִי, / כִּי אִם בְּנֵי הָאָדָם,

כִּי גֵרְשׁוּנִי הַיּוֹם מֵהִסְתַּפֵּחַ בְּנַחֲלַת יְיָ.

שִׂימָה נָא כִדְבָרֶךָ:

אִם שָׁגִיתִי – / יִסַּרְתַּנִי,

וְאִם לֹא שָׁגִיתִי – / הֶחֱיִיתַנִי.

וְאִם הֶחֱיִיתַנִי – / הַדָּן דִּינִי מַה יְּהִי עָלָיו?

וְאִם עֲשָׁקַנִי – / מַה תְּהֵא סְתִירָתוֹ?

וְאֵיךְ יִשָּׂא פָנִים / לְאֵין פָּנִים נוֹשֵׂא?


לֹא דִין תּוֹרָה דַנְתַּנִי / וְלֹא מִשְׁפַּט תַּחְכְּמוֹנִי שְׁפַטְתַּנִי

וּכְמִשְׁפְּטֵי הַגּוֹיִים לֹא חִיַּבְתַּנִי:

כִּי אָמְנָם הָתּוֹרָה לֹא מְסָרַתְנִי בְיָדֶךָ

וְכֵן הַבָּאִים אַחֲרֶיהָ לֹא סְגָרוּנִי לְמוּסָרֶךָ.

כִּי עִבְרוּ אִיֵּי כִתִּיִים וּרְאוּ / וְקֵדָר שִׁלְחוּ וְהִתְבּוֹנְנוּ מְאֹד –

וּרְאוּ הֵן הָיְתָה כָּזֹאת: / הֲתִשְׁפֹּט הֲתִשְׁפֹּט בְּלִי רְאוֹת?

הֲלֹא עֵינֵי בָשָׂר לְךָ, / הֲלֹא כִרְאוֹת אֱנוֹשׁ תִּרְאֶה?

כִּי מִי כָל בָּשָׂר – / מֵאִיֵּי הָעַמִּים / וּמִכָּל הַלְּשׁוֹנוֹת

מֵעוֹבְדֵי אֵשׁ / וּמִמְּשָׁרְתֵי מַיִם

וּמִן הַמִּשְׁתַּחֲוִים לִצְבָא הַשָּׁמַיִם –

אֲשֶׁר עָשָׂה כָזֹאת / וּמִי מֵהֶם אֲשֶׁר דָּנוּ כֹה?

וְלוּלֵי הָיוּ שׁוֹפְטַי הֲלוּמֵי יַיִן,

לֹא הֶחֱנִיפוּ לְעֵינֵי הַשֶּׁמֶשׁ / וְלֹא הִטּוּ דִינִי

וְלֹא הֶעֱווּ רִיבִי נֹכַח פְּנֵי יְיָ.

וְאִם אֲנִי אֶגְוַע – / חֲמָסִי לֹא יֵאָסֵף,

וְאִם אֲנִי כְּרָקָב אֶבְלֶה – / חֲמָסִי לַמָּרוֹם יַעֲלֶה!


וְאַתָּה, בֶּן אָדָם,

שְׁמַע נָא אֶת דִּבְרֵי עַבְדְּךָ, / כִּי יֶשׁ לוֹ מִלִּים.

אִם תּוּכַל הֲשִׁיבֵנִי — / אֲנִי אַחֲרִישׁ וְאֶגְוַע.

אִם לַיָי פָּעַלְתָּ / וְלַעֲשׂוֹת נְכוֹחָה נָטִיתָ,

מַדּוּעַ לֹא קִנֵּאתָ בַּהוֹלֲלִים / אֲשֶׁר חִלֲלוּ עָלַי מוֹעֵד וְשַׁבָּת?

הֲלֹא לְעֵינֶיךָ הֲלָמוּנִי / לִפְרֹם בְּיוֹם מְנוּחַת קֹדֶשׁ מַדִּי

וְלִמְרֹט שְׂעָרִי בְּיוֹם שַׁבָּת קֹדֶשׁ, / לַמְרוֹת עֶלְיוֹן לְעֵינֵי כָל חַי.

גַּם בְּיוֹם מוֹעֵד, / מִבְחַר כָּל הַמּוֹעֲדִים, / צִוִּיתָ לְהַחֲרִיב בִּנְיָנִי,

וּבִידֵיהֶם הֲלָמוּהוּ / בְּכַשִּׁיל וְכֵילַפּוֹת –

לֹא עַל יְדֵי נָכְרִים / כִּי אִם בִּידֵיהֶם!

הֲלֹא אֵלֶּה יִזְכֹּר יְיָ / וְיַעֲלֶה עַל לִבּוֹ,

כִּי חִלֵּל יְהוּדָה קֹדֶשׁ יְיָ אֲשֶׁר אָהֵב.

אִם יֵשׁ מִלִּין – / הֲשִׁיבֵנִי וְיַסְּרֵנִי עוֹד,

אַךְ בְּמִשְׁפָּט / וְלֹא בְכֹחַ הַזְּרוֹעַ!

כִּי מִי יוּכַל לָדִין עִם שֶׁתַּקִּיף מִמֶּנּוּ?

הֲלֹא יָכֹלְתָּ לְחַמֵּץ דִּינִי / וּלְמַהְמֵהַּ מִשְׁפָּטִי

עַד אֲשֶׁר יִגָּלֶה צֶדֶק עוֹלָמִים / וְיֵרָאֶה לָעַיִן.


נָוֶה אֲשֶׁר הִנְחִילוּנִי אֲבוֹתַי

צִוִּיתָ לְחַלֵּל עָלָיו יוֹם מוֹעֵד

וּבָאוּ בָה פָּרִיצִים וְחִלְּלוּהוּ.

אֲהָהּ, אֲדוֹנִי, הֶחָצֵר אֲשֶׁר צִוִּיתָ לְהַחֲרִיב

לִי וּלְאַחַי הַיְתוֹמִים הָיְתָה לְמוֹרָשָׁה –

הֲלֹא אֱלֹהִים יָרִיב רִיבָם!

הֵן מִקּוֹרוֹתַי וְדַלְתוֹתַי וְלוּחוֹתַי / הוּקַם בִּנְיָנָם

וְהִנָּם בַּבִּנְיָן – / כָּל רוֹאֵיהֶם יַכִּירוּם,

וְקֶצֶב גִּזְרָתָם וּמִדָּתָם / יוֹרֶה עֲלֵיהֶם,

וְכָפִיס מֵעֵץ יַעֲנֶה: / הֲלֹא אֱלֹהִים יַחֲקָר־זֹאת!

וְאִלּוּ הִסְכַּנְתָּ תַעֲלוּמַי

וְעָמַדְתָּ עַל אֲמִתַּת דִּבְרֵי חַטָּאתִי בְּעֵדֵי רְאִיָּה,

לֹא הָיָה נָכוֹן לַעֲשׂוֹת כֵּן / לִבְזוֹת יוֹם קֹדֶשׁ יְיָ

תְּחִלַּת מוֹעֲדָיו / וְרֵאשִׁית קָדָשָׁיו.


וְאִם אָמֹר תֹּאמַר: / יוֹם הוּא רָאוּי לְחַלֵּל –

חָלִילָה לְהַצְדִּיק / כָּל דּוֹבְרֵי פֶה!

כִּי יֵשׁ עִמִּי עוֹד תְּשׁוּבָה,

מִלִּים שְׁנוּנִים מֵחִצִּים, / פְּתוּחִים מֵחֲרָבוֹת,

מַחֲסִמֵי פֶה, / מַאֲלִימֵי לָשׁוֹן.

וְאִם עַתָּה יֶשׁ לָהּ דְּמָמָה – / אַחֲרֵי יוֹם תִּשְׁמַע שְׁאוֹנָהּ:

כִּי הִנֵּה יָמִים בָּאִים – / אִם אֱלֹהִים יִלְוֵנִי / וְלֹא יֶאֱסֹף רוּחִי –

קָנִים יִמְעֲטוּ / וְיַמִּים יֶחְסָרוּ וּקְלָפִים יִצְעָרוּ

בְּחוֹקְקִי מִשּׂוֹנְאַי כְּרֹעַ מִפְעָלָם – / וְכִסֵּא אֲדוֹנִי הַמֶּלֶךְ נָקִי!


עַל חֲמַס עוֹשְׁקַי אָשִׂיחַ אֶל אָרֶץ: / אֶרֶץ, אַל תְּכַסִּי דָמִי!

גַּם לְהַרֲרֵי עַד דָּלְפָה עֵינִי: / חֲמָסִי וּשְׁאֵרִי שִׁמְעוּ, הָרִים!

גַּם לִגְבָעוֹת בְּשִׂיחִי אָהִימָה: / אָנָּא, גְּבָעוֹת, שְׁמַעַן קוֹלִי!

חֲמָסִי עֲלֵיכֶם, אִם תַּכְחִידוּהוּ:

הֲלֹא לְנֶגְדְּכֶם סְבָבוּנִי בְכַחַשׁ / וְלֹא לֶאֱמוּנָה עֲשָׁקוּנִי מְרִיבַי.

אֲבַדּוֹן וָמָוֶת! שִׁמְרוּ יוֹם חֲמָסִי, / וְאַל יְהִי סֵתֶר לִמְרִי שִׂיחִי:

הֲלֹא מִמַּחְצַבְכֶם קֹרַצְתִּי גַּם אָנִי, / כִּי תִקְרְאוּ לִי בְּצִלְּכֶם תְּסֻכּוּנִי.

בְּעוֹד יְמֵי פְרִידָה חָנּוּנִי / וּבַעֲדִי זַעֲקוּ חָמָס

עַד שׁוּבִי אֲלֵיכֶם – בְּעוֹדֶנִּי חַי – / צַעֲקוּ עִמִּי!

אוּלַי לֹא לָאֹרֶךְ אַף, / אַךְ יֶחֱרֶה בְלוֹחֲמַי.

מֵאַרְבַּע הָרוּחוֹת בּוֹאִי, רוּחַ, / אֲשֶׁר מִמֶּנָּה רוּחִי נֻפָּחָה!

הִתְאַבְּלִי – הֵימִינִי / הַאֲנִינִי – הַשְׂמִילִי,

לִבְשִׁי עָלַי שַׂק / וְהִתְפַּלְּשׁי בָאֵפֶר,

עַל תּוּגַת מֶחֱצָתֵךְ / הַרְחִיבִי קָרְחָתֵךְ

וְעַל מַר רוּחִי / חוּלִי וָגֹחִי.

אֱלִי נָא עָלַי כָּל יְמֵי הָאָרֶץ

וְחָלִילָה לָךְ תֵּת לְזַעֲקָתֵךְ דֹּמִי.

שְׂאִי עָל כַּפַּיִךְ – / אוּלַי תּוּכְלִי הוֹעִיל

וְגַם שִׁפְכִי שִׂיחַ – / אוּלַי תַּעֲרֹצִי:

כִּי גוֹאֲלִי חַי / לֹא שָׁכַח חַנּוֹת,

וּלְהַסְתִּיר פָּנִים / לֹא גָמַר אֹמֶר.

גַּם קוֹלִי אֱלֵי מַיִם עַל יַבֶּשֶׁת לֵחוּתִי –

הֲלֹא בְמֵימֵי מוּצָק בְּצֵקִי הוּלַשׁ.

גַּם אַתֶּם צַעֲקוּ בְּקוֹל שָׁאוֹן / וְאַל תִּדֹּם הֶמְיַת צִוְחַתְכֶם

גַּם אַתֶּם לוּ שְׁמָעוּנִי,

הֵן לָכֶם יְכֹלֶת / לִנְקֹם נִקְמָתִי:

בִּשְׁתוֹת אֶתְכֶם חוֹמְסַי – / הֱיוּ נָא לְמָרִים,

בְּבוֹאֲכֶם בְּמֵעֵימוֹ – / הֱיוּ לַצְבּוֹת בֶּטֶן,

וְכָל מַשְׁקֶה אֲשֶׁר יִשָּׁתֶה / יֵמַר לְשׁוֹתָיו,

וְעוֹלְלוּ לָמוֹ – / וְיִהְיוּ כָמוֹנִי!

וְלִשְׁמֵי קֶדֶם / אָמַרְתִּי: שְׁמָעוּנִי!

וְלִצְבָא מָרוֹם / חַנֹּתִי בְּמוֹ פִי,

וּלְגִבּוֹרֵי כֹחַ / אֵלֵי פֶלֶא,

אֲשֶׁר מִשְׁכְּנוֹתָם / גָּבְהֵי מְרוֹמִים –

לְכֻלָּם אֶפְרֹשׂ כַּפַּי / וְאֶשְׁפֹּךְ שִׂיחִי:

אוּלַי יָעִידוּ חֲמָתִי / לְשָׂהֲדִי בַּמְּרוֹמִים / הַשַּׁלִּיט עַל כֻּלָּם!

וְגַם לְשׁוֹכְנֵי עָפָר / עֵינִי דוֹמַעַת

הָעֲתִידִים לַעֲמֹד בַּדִּין / לִפְנֵי יוֹצֵר כֹּל

הָעֲשׁוּקִים כָּמוֹנִי הַיּוֹם / דִּמְעוֹת אַנְחוֹתַי יַטּוּ לְדִמְעוֹתָם

וְעַל חֲמָסִי אֶל אֵל יְשַׁוֵּעוּ:

אוּלַי מִמֶּנִּי אוֹ מֵהֶם יַקְשִׁיב אֹזֶן.


עוֹד אֲנִי מְדַבֵּר / וְאֶעֱרֹךְ מִלִּים – וְלֹא לְהוֹכִיחַ

וְאַזְכִּיר קַדְמוֹנִיּוֹת – / לֹא לְהוֹעִיל.

אַךְ עַתָּה הֱצִיקַתְנִי רוּחִי:

עוּרָה, אֲדוֹנִי, וְהִתְבּוֹנֵן / וְהַט אֹזֶן לְשַׁוְעַת מִתְחַנֵּן,

וְלֵב עָגוּם תַּרְנִין, / אִם פָּנֶיךָ לֹא תַעֲנִין!

מְאוּמָה לֹא אֲבַקֵּשׁ בִּלְתִּי קֶשֶׁב,

וְאִם אֶל כָּמוֹנִי יִפְנֶה כָמוֹךָ – / אֵינֶנּוּ תִמָּהוֹן,

וְאִם תֵּט אֵלַי סְכוֹת חִין עֶרְכִּי – / כְּבוֹדְךָ לֹא יֵחַל

וְאֵין לְמַעֲלָתְךָ גֵרָעוֹן.

אָעִיר זִכְרוֹנֵי קֶדֶם / אֲשֶׁר צִדְקִי בָם,

וְאִם תִּבְחַר אֱמוּנָה, / תּוֹדֶה לִי עֲלֵיהֶם.

תְּחִלַּת אוֹדוֹתַי / הָיָה עָלֶיךָ לְכַבְּדֵנִי

עָמְדִי לִפְנֵי אֲדוֹנִי אָבִיךָ – / יוֹצְרוֹ יַרְגִּיעַ רוּחוֹ הַנְּדִיבָה! –

וָאֲשָׁרְתֵהוּ וָאֶצֹּק מַיִם עַל יָדָיו, / אֲהֵבַנִי וַיִּמְשְׁכֵנִי בְחַבְלֵי יְדִידוּת,

וְכָל יְמַי אֲשֶׁר פָּרַשׂ עָלַי צִלּוֹ / שֶׁמֶשׁ לֹא יֵךְ עָלַי.

הֲלֹא לְפִי כֹחַ מִזְעַר בִּינָתִי / סִפַּרְתִּי מַהֲלָלָיו

וְהִנָּם כְּתוּבִים עַל אֲרוֹן הָעֵדוּת / בְּבֵית זְבוּל אֲשֶׁר בָּנָה לְשֵׁם אֱלֹהָיו,

וְהַמִּכְתָּב מִכְתַּב יָדִי הוּא / וְכָמוֹהוּ כָּהֵנָּה וְכָהֵנָּה.


אַחֲרֵי זֹאת זְכֹר נָא, אֲדוֹנִי, לֵיל תַּמְרוּרִים

אֲשֶׁר בּוֹ נֶאֶסְפָה גְבִרְתִי יוֹלַדְתְּךָ – / יְרַחֲמֶנָּה עוֹשָׂהּ!

כֹּה יַעֲשֶׂה לִי אֱלֹהִים / וְכֹה יוֹסִיף מֵחֲרוֹנְךָ –

אִם לֹא בַחֲצוֹת לַיְלָה / רַגְלִי בָּאתָ אֵלַי

לְתַקֵּן סֵפֶד / וּלְהַחְרִיז נֶהִי!

מְצָאתַנִי – וַאֲנִי כוֹתֵב / וְנִזְהַרְתִּי טֶרֶם תַּזְהִירֵנִי.

וַיְהִי כִּי רָאִיתָה כֹה, / נִשְׁבַּעְתָּ בְאֵל עֶלְיוֹן

“שְׁמוּרָה זֹאת תִּהְיֶה לְךָ כָּל הַיָּמִים!” –

וְאַתָּה זָנַחְתָּ וַתִּמְאַס!

הֲלֹא כֹה קְרָאַנִי / בְּהֵאָסֵף אֲדוֹנִי אָבִיךָ:

כּוֹנַנְתִּי רַב סֵפֶד / וְחַקֹּתִי דִבְרֵי קִינוֹת

אֲשֶׁר קוֹנֲנוּ כָל יִשְׂרָאֵל / דְּבַר יוֹם בְּיוֹמוֹ / כָּל יְמֵי הָאֵבֶל

וְעַל קַל מִזֹּאת צִוָּה דָוִד מְשִׁיחַ יְיָ לִפְנֵי מוֹתוֹ

לִהְיוֹת בְּנֵי בַרְזִלַּי הַגִּלְעָדִי בְּאוֹכְלֵי שֻׁלְחָנוֹ.

וּמָה הַגְמוּל אֲשֶׁר גְּמַלְתַּנִי / וּמָה הַשִּׁלּוּם אֲשֶׁר שִׁלַּמְתָּ לִי –

אַחַר אֲשֶׁר שַׂמְתִּי אִגְּרוֹת מַהֲלָלֶיךָ / אֲחַשְׁתְּרָנֵי רֶכֶב

וָאָשִׁית גִּלְיוֹנֵי שְׁבָחֶיךָ / רוֹכְבֵי רֶכֶשׁ

וָאָשִׂים סִפְרֵי הוֹדְךָ / גַּלְגִּלֵּי כִרְכָּרוֹת / בְּכָל עִיר וָעִיר?

סִדַּרְתִּי כָל רָז, / עֲדֵי הֲתִמּוֹתִי כָל חָרָז

הֶאֱלַמְתִּי כָל מְצַחְצֵחַ / וְאֶחְסֹם פֶּה כָּל מַחֲרִיז

לְבִלְתִּי הוֹתִיר לָמוֹ מָקוֹם לְדַבֵּר / וְלֹא מַעֲנֶה לְהוֹסִיף.

וְעַד הֵנָּה לֹא הֲסִירוֹתִי תֻמָּתִי מִמֶּנִי –

וְעוֹדֶנִּי זוֹרֵעַ מַהֲלָלֶיךָ / וּמְזָרֶה שְׁבָחֶיךָ כְּטַל עַל הָאֲדָמָה.

וְאִם יִסַּרְתַּנִי בְּרֹעַ מוּסָר – / צִדְקִי מִמֶּנִּי לֹא סָר,

וְאִם גְּמַלְתַּנִי לֹא כַדָּת – / לֹא תִטֶּה אַשּׁוּרִי מִנִּי הַדֶּרֶךְ!


וְאַחֲרֵי זֹאת זְכֹר נָא, אֲדוֹנִי, / כִּי הֲבִיאוֹתַנִי מִמֶּרְחָק, מִנְּוֵה מַרְגּוֹעַ,

מְשַׁכְתַּנִי מִמִּרְעֶה טוֹב / וְעוֹרַרְתַּנִי מֵרִבְצִי.

אַצְתִּי חַשְׁתִּי וָאָבוֹא –

לֹא אִחֲרוּ רַגְלַי חַרְבוֹנֵי קַיִץ / וְלֹא מִהְמְהוּ פְעָמַי סוּפוֹת סַעַר

וְלֹא בוֹשְׁשׁוּ צְעָדַי חֲגוּרֵי שֶׁלַח / וְלֹא אֲחוּזֵי חֶרֶב.

וְאוּלָם פַּחְדְּךָ לֹא הִסִּיעַנִי / וּמַתַּן יָדְךָ לֹא מְשָׁכַנִי

וּגְמוּלַת יָדְךָ לֹא הֱנִיעַתְנִי / וְנִדְבוֹתֶיךָ לֹא הֶעִירוּנִי –

בִּלְתִּי אַחֲוָתְךָ עוֹרֲרַתְנִי / וְעֵקֶב יְדִידוּת הוֹדְךָ מִהֲרוּ רַגְלַי!


וַיְהִי כַאֲשֶׁר בָּאתִי

צִוִּיתַנִי עַל מְלֶאכֶת לִמּוּדִים / עֲבוֹדַת לְשׁון הַקֹּדֶשׁ

וָאֹמַר “שָׁמַעְתִּי, אֲדוֹנִי!” / וָאָעִיר לְבָבִי מִקִּנּוֹ וָאֲחַזְּקֵהוּ

וָאָשִׁית כִּלְיוֹתַי אַנְשֵׁי עֲצָתִי / וַיִּהְיוּ לִי כִּמְשִׁיבֵי טַעַם

וָאָנִיד עִמִּי עַפְעַפַּי / וַתְּמָאֵן נַפְשִׁי תֵּת שֵׁנָה לְעֵינַי,

עַד אֲשֶׁר נָכוֹנוּ עִנְיְנֵי אֲדוֹנִי / וַיִּשְׁלַם חֵשֶׁק מַאֲוַיָּיו.

ןְחַי יְיָ וְחֵי נַפְשְׁךָ, / אִם לֹא הֲקִימוֹתִי מִצְוָתְךָ

מִדְּאָגָה וּמֵרְעָבוֹן / וּמִתּוֹךְ עֵירֹם וְחֹסֶר כֹּל!

וְעֵין יְמִינִי כָּהֹה תִכְהֶה, / אִם הִשִּׂיגָה יָדִי דֵּי מַחְסוֹרִי

וְאִם לֹא הֻתַּכְתִּי / בְּמַצְרֵף כּוּר רְעָבוֹן!

וֶאֱלוֹהַּ הִקְשִׁיחַ אֶת לֵב אֲדוֹנִי

וְהִרְחִיק מִמֶּנּוּ חֲנִינָה מֵחֲמֹל עָלַי

וְאִמֵּץ אֶת לְבָבוֹ מִדַּעַת אֶת עֲמָלִי

לֵאמֹר: זֶה הַטּוֹרֵחַ אֲשֶׁר הִטְרַחְתִּיו בִּמְלַאכְתִּי

אֲשֶׁר הֲבִיאוֹתִיו מֵאֶרֶץ מַרְחַקִּים / וַהֲשִׁיבֹתִיהוּ מִסְּחוֹרָתוֹ

וְאֵין בְּיָדוֹ מְאוּמָה בִּלְתִּי אֵלַי עֵינָיו –

מֵאַיִן מִחְיָתוֹ וּמַכֹּלֶת לְבֵיתוֹ? / מִי יִרְעֵם וּמִי יְכַסֵּם?

וַיְהִי כַאֲשֶׁר צָפַד עוֹרִי עַל עַצְמִי,

הָיָה מֵאֵת אֲדוֹנִי תְחִנָּה / כִּמְעַט רֶגַע וְעוֹנָה

וּמִנָּה אֵלַי קְצָת מֵאָכְלִי וּשְׁתִיָּתִי

בְּמִדָּה בְּמִשְׁקָל וּבִמְשׂוּרָה / מִדֵּי חֹדֶשׁ בְּחָדְשׁוֹ.

כֹּה יַעֲשֶׂה לִי אֱלֹהִים / וְכֹה יוֹסִיף מֵחֲרוֹנְךָ,

אִם לָקַחְתִּי מִכָּל אֲשֶׁר צִוָּה לִי מְאוּמָה,

וְהִנְנִי וְהֵם – / אִם עַל פָּנַי יַכְזִיבוּנִי!

וּבְכָל זֹאת לֹא סַרְתִּי מִנִּי דֶרֶךְ וָאֹמַר: אֲקַוֶּה רְצוֹנוֹ.

וְקָנִיתִי חֲרוֹנוֹ / וַתְּהִי מַשְׂכֻּרְתִּי שִׁמֲמוֹת עוֹלָם

וְחֵלֶף עֲבוֹדָתִי – / כּוֹס תַּרְעֵלָה הֹעַלְתָה לִי

לִהְיוֹת לְדִרְאוֹן עוֹלָם.

וְהָיְתָה תוֹחַלְתִּי נִכְזָבָה / וַיִּכֶל שִׂבְרִי בְּאֶפֶס תִּקְוָה.

הֶעֱבַדְתַּנִי חִנָּם עֲבוֹדַת עֶבֶד / וַתְּהִי יְגִיעָתִי לָרִיק וְזֵעָתִי לַהֶבֶל,

וְלוּלֵי פָנֶיךָ – לֹא קְרָאוּנִי אֵלֶּה!

הֲכָזֶה יִהְיוּ תַשְׁלוּמֵי אִישׁ שֵׂכֶל / וְאִם אֵלֶּה גְמוּלָיו?

מַדּוּעַ זָרַעְתִּי חִטִּים / וְקוֹצִים קָצַרְתִּי,

וְאֶרֶץ אֲשֶׁר מְאֹד נָעֲמָה / הֶעֶלְתָה לִי בָאְשָׁה?

וְאִלּוּ חָרַשְׁתִּי תַלְמֵי מְלֵחָה / וְזָרַעְתִּי בְּאֶרֶץ לֹא זְרוּעָה,

הֶחֱרַשְׁתִּי – כִּי אֵין פְּרִי לִמְלֵחָה.


וְאַיֵּה אֵפוֹא פִיךָ / אֲשֶׁר נִשְׁבַּעְתָּ בַּיָי אֱלֹהֶיךָ

לְמַלּאֹות מַחְסוֹרִי / וּלְשַׁלֵּם גְּמוּלִי?

מַחְסוֹרִי נָתוּן לְךָ / וּגְמוּלִי לְאַחֵר תֵּן –

וְהָעֶבֶד לַעֲשׂוֹת בּוֹ כַּטּוֹב בְּעֵינֶיךָ.

אֲדוֹנִי יוֹדֵעַ כִּי מֵעֵת יוֹם חֲמָסִי עַד הֵנָּה

לוּלֵי רַחֲמֵי אֲדוֹנִי אָבִיךָ / – יְהִי אֱלֹהָיו בְּעֶזְרוֹ! –

אֲשֶׁר חָסִיתִי בְצִלּוֹ / וַיִּפְרֹשׂ עָלַי כְּנָפָיו

וַיְרַחֲמֵנִי וַיְכַבְּדֵנִי – / וַיְשִׂימֵנִי בְּאוֹכְלֵי שֻׁלְחָנוֹ

וְלוּלֵי חֲסָדָיו – / אָבַדְתִּי בְעָנְיִי.

יְשַׁלֵּם יְיָ פָּעֳלוֹ / וּתְהִי מַשְׂכֻּרְתּוֹ שְׁלֵמָה

מֵאֵת יְיָ אֱלֹהֵי יִשְׂרָאֵל.


וְעַתָּה, אֲדוֹנִי,

אָנָה אוֹלִיךְ אֶת דַּלּוּתִי / וְאֶל מִי מִקְּדוֹשִׁים אֶפְנֶה

וְעַל מִי אָנוּסָה לְעֶזְרָה / וְאֶל מִי אֶשָּׂא עַיִן וְאֵיךְ אֶחְיֶה?

הָבָה לִי עֵצָה – / אֵיךְ תִּגָּדֵר הַפִּרְצָה / אַחֲרֵי כָלָה וְנֶחֱרָצָה?

זְכֹר, אֲדוֹנִי, כִּי בְכָל כֹּחִי עֲבַדְתִּיךָ / וּבְכָל נַפְשִׁי הִלַּלְתִּיךָ

וּבְכָל פֵּאָה זָרַעְתִּי מַהֲלָלֶיךָ / וְחַי יְיָ אֱלֹהֶיךָ,

אִלּוּ הָיְתָה לְאֵל יָדִי / וּמָצָאתִי אוֹן לִי –

הָיִיתִי מְרַצְּךָ בְהוֹנִי / כַּאֲשֶׁר רִצִּיתִיךָ מִפְּרִי לְשׁוֹנִי:

אֲבָל מֵהַנִּמְצָא בְיָדִי נָתַתִּי לְךָ,

אָבִי אָבִי, רֶכֶב יִשְׂרָאֵל וּפָרָשָׁיו!

אָמְנָם מְהִיר צֶדֶק / וּנְשִׂיא אֱלֹהִים אַתָּה,

וֶאֱמוּנָתְךָ אִמְּצַתְנִי / וְעַל אֲמִתְּךָ נִשְׁעַנְתִּי,

וְלוּלֵא חֲסָדֶיךָ / לֹא דִבַּרְתִּי עַד כֹּה.

יָדַעְתִּי כִּי נְכוֹנָה תַחְפֹּץ / וּמֵישָׁרִים תִּרְצֶה –

אֲבָל מֵרֹב שִׂיחִי וְכַעְסִי / דִּבַּרְתִּי עַד הֵנָּה.

וַאדוֹנִי יוֹדֵעַ כִּי לֹא נָכוֹן / לְהוֹכִיחַ מָרֵי נֶפֶשׁ

וְלֹא לִגְעֹר בִּדְוֵי לֵבָב / וְלֹא לִדְחֹף טְחוּנֵי פָנִים,

כִּי בְמָצוֹק יְדַבֵּרוּ / וּבִמְרִירוּת יַבִּיעוּ,

כִּי בָחֲרוּ מָוֶת מֵחַיִּים.


וְעַתָּה הִשְׁבַּעְתִּיךָ בֶאֱמוּנַת הַצֶּדֶק

אֲשֶׁר הִיא נְתִיבָתְךָ / הַסְּלוּלָה הַמְיֻשֶּׁרֶת

וּבְתִפְאֶרֶת הָאֱמוּנָה / אֲשֶׁר הִיא אֵזוֹר לַחֲלָצֶיךָ –

שֶׁתִּפְנֶה אֶל טַעֲנָתִי / בְּקַו אֱמוּנָה / וּבְמִשְׁקֹלֶת צֶדֶק

וְאֶל בִּינָתְךָ אַל תִּשָּׁעֵן.

וְאִם אֱמֶת הִיא – / תּוֹדֶה לִי עָלֶיהָ,

וְאִם יִמָּצֵא בָהּ תַּרְמִית – / תּוֹרֶה לִי עָלֶיהָ בַּלָּאט.


וְיוֹדֵעַ אֲדוֹנִי כִּי אִיּוֹב / עֵת הוּבָא בִפְלִילֵי נִסָּיוֹן

וְהֻכָּה בְרֹעַ מַדְוֶה / וּבְנִגְעֵי בְנֵי אָדָם –

הֲלֹא כְגַעַת בּוֹ יַד אֱלוֹהַּ, / הִכְבִּיר מִלִּים,

וְכַאֲשֶׁר חָזְקָה מַחֲלָתוֹ, / הֶחֱזִיק וַיּוֹכַח עִם אֱלוֹהַּ

כְּבֶן אָדָם לְרֵעֵהוּ.

וַיִּתְּנֵהוּ כְּמַטֶּה דִין / לַעֲשׂוֹת מִשְׁפָּט מְעֻקָּל

וּמִנִּי אֹרַח צֶדֶק מַתְעֶה,

וּלְעֵת נוּעַ לְבָבוֹ / נָטָיוּ רַגְלָיו.

הֲלֹא זֶה חֹמֶר נָלוֹז / וּמֵמֵר אֶת יוֹצְרוֹ!

אֲבָל אֵל כַּבִּיר בְּאֹרַח טוּבוֹ / אֶת לְזוּת שְׂפָתָיו לֹא חָשַׁב לְחַטָּאת,

עַל כִּי בְנֶפֶשׁ מָרָה שְׂפָתָיו מִלֵּלוּ / וּבְתוּגַת לֵב דִּבֶּר עָתָק,

רִיבוֹת שְׂפָתָיו לֹא שָׂם תִּפְלָה, / כִּי הוּא יָדַע נֶגַע לְבָבוֹ:

לְמַעַן יַכִּירוּ כָל יוֹרְדֵי עָפָר

כִּי שׁוֹפֵט כָּל הָאָרֶץ / כֹּחַ מִשְׁפָּטוֹ עִם עָפָר וָאֵפֶר.

וַיְהִי אַחֲרֵי זֹאת

שָׁב יְיָ אֶת שְׁבוּתוֹ / וּמֵרֵאשִׁיתוֹ הֵיטִיב אַחֲרִיתוֹ.

יִלְמַד אֲדוֹנִי מִדֶּרֶךְ אֱלֹהָיו / וְדַיּוֹ לְעֶבֶד שֶׁיְּהֵא כְרַבּוֹ!

הֲלֹא עַל יְלוּד אִשָּׁה לִלְמֹד מִדְּרָכָיו / וּבִדְרָכָיו יִנְהַג עִם נְמוּכָיו –

וֵאלֹהֵי יִשְׂרָאֵל / הַנּוֹתֵן תְּשׁוּעָה לַמְּלָכִים

יַעְטֶה אֲדוֹנִי מַעֲטֵה תְהִלָּה / וְיַלְבִּישׁוֹ בִגְדֵי יֶשַׁע

וְיָמָיו יְבַלֶּה בַנְּעִימִים / יִרְאֶה זֶרַע יַאֲרִיךְ יָמִים,

וְיִמְצָא חֵן וְשֵׂכֶל טוֹב / בְּעֵינֵי אֱלֹהִים וְאָדָם.

וְתַאֲוַת לִבּוֹ יִתֶּן לוֹ / וַאֲרֶשֶׁת שְׂפָתָיו בַּל יִמְנַע סֶלָה,

וְאוֹיְבָיו יְכַלֶּה בְּשֵׁבֶט עֶבְרָתוֹ / וְאוֹהֲבָיו כְּצֵאת הַשֶּׁמֶשׁ בִּגְבוּרָתוֹ.

הָאֵל הַנֶּאֱמָן / הַמְנַהֵג בְּעֻזּוֹ תֵימָן / יָשִׂים מוֹשָׁבוֹ מִשְׁמָן

עַד תַּכְלִית כָּל זְמָן / לְמַעַן יְיָ אֲשֶׁר נֶאֱמָן

קְדוֹשׁ יִשְׂרָאֵל וַיִּבְחָרֶךָ.

שָׁלוֹם לָרוֹם / כְּמִיָּם אֶל דָּרוֹם

וּכְחוֹל יַמִּים וּצְבָא מָרוֹם

אָמֵן נֶצַח סֶלָה.


כְּתָב הַמּוּשָׁב מֵאֵת יְדִידִי שַׂמְתִּיהוּ לְטוֹטָפוֹת

וַיְהִי לִי לְמֵשִׁיב נֶפֶשׁ וּמֶתֶק צוּפוֹת.

וּבִרְאוֹתִי כִּי בוֹשֵׁשׁ – בָּעֲרוּ בְקִרְבִּי רְשָׁפוֹת

וְיָגֹרְתִּי פֶּן נֶהְפַּךְ לִהְיוֹת לִי חֲלִיפוֹת,

כְּמַהְפֵּכַת אִיּוֹב לְאוֹיֵב וְטִפָּה בְּטִפּוֹת.

וְאִם נֶהְפַּךְ – אַיֵּה הַתְּרוּפוֹת?

הֲלֹא כְעֶבֶד אֶל אֲדוֹנָיו עֵינַי לְךָ צוֹפוֹת,

עַד יָקוּם עֵד הַנֶּאֱמָן כִּי נוֹתַרְתִּי כְּיוֹנָה קְצוּצַת אֲגַפּוֹת

וּכְשָׂרִיג הַמַּלְבִּין אֲשֶׁר נֶחֱשַׂף חֲשָׂפוֹת

וֶאֱלוֹהַּ מָעֻזִּים שָׁלוֹם לְךָ יִשְׁפֹּת.

אָמֵן!
