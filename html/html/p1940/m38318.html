אָרְכוּ הַיָּמִים

&&& אָרְכוּ הַיָּמִים

סליחה. סימן: א״ב כפול וחתימה: שמעון בר יצחק חזק.


אָרְכוּ הַיָּמִים וּדְבַר כָּל חָזוֹן,

אַרְמוֹן נֻטַּשׁ וְחָדַל פְּרָזוֹן,

בְּקַוּוּי מַרְפֵּא בְּעָתָה וְרָזוֹן,

בִּטְחָה וְהַשְׁקֵט לִמְנוּסָה וְחִפָּזוֹן;


גְּאֻלָּה נֶאֱחֶרֶת וְתוֹחֶלֶת מְמֻשָּׁכָה,

גּוֹלָה וְסוּרָה קְדוֹרַנִּית, חֲשׁוּכָה.

דִּמְעָתָהּ מְצוּיָה בְּלֶחֱיָהּ לְמָשְׁכָה,

דְּרוּסָה, בְּעוּטָה, נְגוּחָה, נְשׁוּכָה;


הוּבְאָה בַחֹשֶׁךְ וְיוֹשֶׁבֶת דּוּמָם,

הוֹמֶה לִבָּהּ וּבְקִרְבָּהּ יִשְׁתּוֹמָם,

וְעֵינֶיהָ מְיַחֲלוֹת לַיְלָה וְיוֹמָם,

וְעָצוּר בְּעַצְמוֹתֶיהָ כְּגַחֶלֶת לַחְמָם;


זְמַנֵּי קִצֶּיהָ מְאֹד נֶחְתָּמוּ,

זָהֳרוֹת נַהֲרוֹתֶיהָ בְּעֶבְרָה נֶעְתָּמוּ,

חֻבְּאוּ נִדָּחֶיהָ וּבְמַסְגֵּר נִסְתָּמוּ,

חֲטָאֶיהָ הִשִּׂיגוּהָ כִּי נִכְתָּמוּ!


טוֹפְלֵי שֶׁקֶר יִשְׁלָיוּ בְשֶׁקֶט,

טִירוֹתָם בְּטוּחוֹת כְּעַל שִׁמְרֵי פֶקֶט;

יַעֲצִיבוּ רַעְיָתְךָ בְּרֹגֶז לְמַקֵּט,

יְגוֹזְזוּהָ כְרָחֵל, גִּזּוֹתֶיהָ לְלַקֵּט;


כָּשַׁל כֹּחָהּ מִכֹּבֶד סֵבֶל,

כִּנּוֹרָהּ לִבְכִי וְעוּגָבָהּ לְאֵבֶל;

לְנֶגְדָּהּ כְּלִמָּתָהּ מִצָּרֶיהָ לְהִתְנַבֵּל,

לֵאמֹר: ״נוֹאַשְׁתֶּם לָצֵאת מִסֵּבֶל;


מִיּוֹם נִמְאַסְתֶּם בַּגּוֹיִם לְנִדָּח

מְנוּסְכֶם אָבַד וְכִיקוֹד מֻקְדָּח,

נוֹתַרְתֶּם כַּנֵּס וְכִצְבִי מֻדָּח.

נִהְיָה יְקָרְכֶם כְּאֹפֶל מְנֻדָּח!״


סַלְעֵנוּ וּמְצוּדָתֵנוּ בְּשִׁמְךָ נָרוּץ,

סָבוֹל עָלֶיךָ כִּלָּיוֹן חָרוּץ,

עֲנוּשִׁים וּמֻכִּים, כְּפֶרֶץ פָּרוּץ,

עֱזוּזְךָ מֵהָמִיר בְּחֹמֶר קָרוּץ!


פְּנֵה לְהָחִישׁ, אֱיָלוּתִי, לְעֶזְרָתִי,

פֶּן יֹאמַר הָאוֹיֵב: יָכֹלְתִּי!

צָרַי יָגִילוּ כִּי שָׁכַלְתִּי –

צְדָקָה עֲשֵׂה כִּי כִמְעַט כָּלִיתִי;


קוּמָה וְהִנָּשֵׂא בְּעַבְרוֹת צוֹרֲרִים,

קָרֵב יוֹם קָרְאוּ נוֹצְרִים,

רָעֵי גוֹיִם לִהְיוֹת מְצֵרִים,

רְאוֹתָם דְּרִיכַת פּוּרַת בּוֹצְרִים!


שׁוּר בְּעַמְּךָ לְהוֹסִיף אָמְצוֹ

שְׁקֹד לְטַהֲרוֹ וּלְנַקּוֹתוֹ מִשִּׁמְצוֹ,

תִּשְׁמֹר עֶדְרְךָ כְּרוֹעֶה לְרִבְצוֹ,

תַּשְׁמִיעַ: מְזָרֵה יִשְׂרָאֵל קִבְּצוֹ!


שַׁנֵּן בְּרַק חֶרֶב בְּזַעְמֶךָ,

שַׁכֵּר חִצֶּיךָ מִגּוֹי מַזְעִימֶךָ,

מֵרֹאשׁ פַּרְעוֹת אוֹיֵב בְּהִנָּקְמֶךָ,

מִמַּה שֶּׁעָשׂוּ בְּחַלְלֵי עַמֶּךָ!


עֲבוּר נִקְמַתְךָ גּוֹיִם הִרְנִינוּ,

עִלִּיּוֹת וְתַחְתִּיּוֹת הֵרִיעוּ וְרָנּוּ,

וְהַר וָגֶבַע וְאִילָנוֹת יְרַנֵּנוּ,

וְגַם יוֹשְׁבֵי סֶלַע יָרֹנּוּ!


נְקֹם הַדָּם וּנְקֹם הֶחָמָס,

נִמְסַךְ לְצָרִים לְהַגִּירָם נִכְמָס,

בְּכוֹס עָמֹק וְרָחָב לְהַעְמַס,

בִּשְׂחוֹק וָלַעַג כֹּחָם יִמָּס;


רִאשׁוֹנָה תְשַׁלֵּם מִשְׁנֵה עֲווֹנוֹתֵיהֶם,

רְאֵה חַלְּלָם אַרְצְךָ בִּשְׁחִיתוֹתֵיהֶם,

יֵבוֹשׁוּ זֵדִים בְּשֶׁקֶר עַוְּתוֹתֵיהֶם,

יִפָּקְדוּ מֵרֹב יָמִים בְּחַטֹּאתֵיהֶם!


צֹרַפְנוּ כְּכֶסֶף בְּמוּעָקָה לְגוֹלֲלָה,

צֻמַּתְנוּ וְנוֹאַשְׁנוּ מִן הַגְּאֻלָּה,

חוֹלֲלָה לְנַפֵּץ יַד עַם סְגֻלָּה,

חֵפֶץ פְּרוּטָה מִכִּיס אָזְלָה;


קַנֵּא לְשִׁמְךָ וְלִקְדֻשָּׁתְךָ חָס,

קָרְבֵנוּ אֵלֶיךָ וּבְצִלְּךָ נָחַס,

חֲזֵה אֵין מַפְגִּיעַ עָלֵינוּ לְהָחָס,

חַלּוֹתְךָ כְּמֹשֶׁה וְאַהֲרֹן וּפִנְחָס;


זְכֹר חֶסֶד וּשְׁבוּעַת קְדוּמִים,

זַעֲקָתֵנוּ קְשֹׁב מִשְּׁמֵי מְרוֹמִים,

קַיֵּם לְעַמְּךָ בְּרִית שְׁלוֹמִים,

אֵל מֶלֶךְ יוֹשֵׁב עַל כִּסֵּא רַחֲמִים!


&&& אָתִיתִי לְחַנֶּנְךָ

קרובה לראש השנה. סימן: א״ב וחתימה: שמעון


אָתִיתִי לְחַנֶּנְךָ בְּלֵב קָרוּעַ וּמֻרְתָּח / בַּקֵּשׁ רַחֲמִים כְּעָנִי בַפֶּתַח

גַּלְגַּל רַחֲמֶיךָ וְדִין אַל תִּמְתַּח / אֲדֹנָי שְׂפָתַי תִּפְתָּח.


דָּבָר אֵין בְּפִי וּבִלְשׁוֹנִי מִלָּה / הֵן יְיָ יָדַעְתָּ כֻלָּהּ

וּמִמַּעֲמַקֵּי הַלֵּב לְפָנֶיךָ אוֹחִילָה / אֶחֱסֶה בְסֵתֶר כְּנָפֶיךָ סֶּלָה.


זַלְעָפָה וּפַלָּצוּת אֲחָזוּנִי בְּמוֹרָא / חַלּוֹת פְּנֵי נוֹרָא בְּנֶפֶשׁ יְקָרָה

טוּב טַעַם וָדַעַת קָטֹנְתִּי לְחַסְּרָה / עַל כֵּן זָחַלְתִּי וָאִירָא.


יָגַעְתִּי בְאַנְחָתִי אֵיךְ לַעֲמֹד לְפָנֶיךָ / כִּי אֵין מַעֲשִׂים לִזְכּוֹת בְּעֵינֶיךָ

לְחַלּוֹתְךָ שְׁלָחוּנִי מַקְהֵלוֹת הֲמוֹנֶיךָ / תָּכִין לִבָּם תַּקְשִׁיב אָזְנֶךָ.


מָה אֲנִי וּמֶה חַיַּי תּוֹלֵעָה וְרִמָּה / נִבְעָר מִדַּעַת בְּאֶפֶס מְזִמָּה

סָמַכְתִּי יְתֵדוֹתַי בְּסֵפֶר הַחָכְמָה / מַעֲנֶה רַךְ יָשִׁיב חֵמָה.


עֻזִּי אֵלֶיךָ אֶשְׁמְרָה לְסַעֲדִי / פֵּתַח דְּבָרֶיךָ הָאֵר לְהַגִּידִי

צַדְּקֵנִי וְאַמְּצֵנִי וְתֵן לְאֵל יָדִי / כִּי אֱלֹהִים מִשְׂגַּבִּי אֱלֹהַי חַסְדִּי.


קְהָלֶיךָ עוֹמְדִים לְבַקֵּשׁ מְחִילָתֶךָ / רַחֲמֶיךָ יִכָּמְרוּ לְרַחֲמָם בְּחֶמְלָתֶךָ

שׁוֹפְכִים לֵב כַּמַּיִם לְעֻמָּתֶךָ / וְאַתָּה תִּשְׁמַע הַשָּׁמַיִם מְכוֹן שִׁבְתֶּךָ.


תְּחַזֵּק לְעַמְּךָ יָדָם הָרָפָה / שְׁלַח מֵאִתְּךָ עֵזֶר וּתְרוּפָה

נָעָמְךָ יַשִּׂיגוּ לְחַזֵּק וּלְתָקְפָה / כָּל אִמְרַת אֱלוֹהַּ צְרוּפָה.
