אָרְחוֹת יוֹשֶׁר תְּעִירֵנוּ

חתימה: אלשבזי


אָרְחוֹת יוֹשֶׁר תְּעִירֵנוּ / בַּעֲלוֹת שַׁחַר בְּקוּמֵנוּ

מִצַּד שְׂמֹאל תַּזְהִירֵנוּ / בְּיֵין הֲבָלָיו יַטְעֵנוּ

וּבְצַד יָמִין תִּסְעָדֵנוּ / לְהַצְלִיחַ אֶת דְּרָכֵינוּ

וְנָשׁוּבָה אֶל אֱלֹהֵינוּ / צוּרִי שְׁעֵה שַׁוְעָתֵנוּ

נוֹרָאוֹת בְּצֶדֶק תַּעֲנֵנוּ / אֱלֹהֵי יִשְׁעֵנוּ:


לֵאלֹהִים צָמְאָה נַפְשִׁי / וְאֶתְבּוֹנֵן בְּסוֹד רַחְשִׁי

כִּי צַד שְׂמֹאל סִבַּת יָקְשִׁי / וּבְקָו אֶמְצָעִי שָׁרְשִׁי

עוּרִי יְחִידָתִי דִּרְשִׁי / צוּרִי, וְגִוְיָתִי הַמְשִׁי

כִּי שֶׁמֶשׁ צְדָקָה שִׁמְשִׁי / וּמִי צוּר זוּלָתִי אֱלֹהֵינוּ:


שׁוֹכֵן שְׁחָקִים יוֹפִיעַ / וְעָלֵינוּ חֵן יַשְׁפִּיעַ

עֲנִיֵּי עַמּוֹ יוֹשִׁיעַ / וְאוֹר זָרוּעַ יַזְרִיעַ

טוֹב וְחֶסֶד לֹא יַגְרִיעַ / בְּשׂוֹרָה טוֹבָה יַשְׁמִיעַ

בְּבוֹא מַלְכִּי הַמּוֹשִׁיעַ / כִּי בוֹ יִשְׂמַח לִבֵּנוּ:


בִּזְכוּת אָבוֹת רִאשׁוֹנִים / בְּחוּרִים נִקְרְאוּ אֵיתָנִים

יְרַחֵם עַל אָבוֹת וּבָנִים / וְיַשְׁפִּיל אוֹיְבִים עַזְפָּנִים

דּוֹבְרֵי שֶׁקֶר וְחַמָּנִים / יִהְיוּ לְטֶבַח מוּכָנִים

וְיֵרָאוּ הַנִּצָּנִים / וְקוֹל הַתּוֹר נִשְׁמַע בְּאַרְצֵנוּ:


זֶה הַיּוֹם עָשָׂה יְוֹיָ / מְכַפֵּר פְּשָׁעַי וַעֲווֹנָי

וְאֶתְבּוֹנֵן בְּסוֹד הֶגְיוֹנָי / כִּי זוֹ דֶּרֶךְ טוֹבָה לְפָנַי

לַעֲלוֹת הֵיכַל אַרְמוֹנָי / וְלַחֲזוֹת זְבוּל בֵּית מַצְפּוּנָי

בִּזְכוּת מַעֲמַד הַר סִינַי / אַשְׁרֵנוּ מַה טּוֹב חֶלְקֵנוּ:


יְשׁוּרוּן יַשֵּׁר מַעֲשֶׂיךָ / וְיֵרָצֶה אֵל חַי עָלֶיךָ

וְאַל יַשִּׂיאֲךָ לִבֶּךָ / כִּי לֹא יוֹעִיל בְּכָשְׁלֶךָ

בְּצֶדֶק תִּשְׁפּוֹט עֲמִיתֶךָ / וְתֵלֵךְ לָבֶטַח דְּרָכֶיךָ

וּמַה טּוֹבוּ אוֹהָלֶיךָ / בְּחַצְרוֹת בֵּית אֱלֹהֵינוּ:
