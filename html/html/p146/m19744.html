אֲרוֹךְ עַיִן בְּכוֹס יַיִן

חתימה: אלשבזי


אֲרוֹךְ עַיִן / בְּכוֹס יַיִן / לְבַל יִשְׁגֶּה לְבָבֶךָ

וְאַל תִּלְמֹד / יְדִיד לַחְמֹד / לְכָל יֶשְׁנוֹ לְרֵעֶךָ

וְסוּר מֵרָע / וְטוֹב תִּזְרָע / צְדָקָה לַעֲנִיֶּךָ

וְאַל תַּכְלִים / פְּנֵי דַּלִּים / וְאַל תִּקְפֹּץ לְיָדֶיךָ

וְאַל תִּגְאֶה / וְתִתְגָּאֶה / וְאַל תִּבְטַח בְּעָשְׁרֶךָ

בְּלֵב גָּמוֹר / לְדָת לִשְׁמֹר / וְאָז תִּצְדַּק בְּדָבְרֶךָ

זְכוֹר קוֹנָךְ / בְּיוֹם אוֹנָךְ / אֱמֹר נָא כֵן בְּעָמְדֶךָ

וְטוֹב אַתָּה וּמֵטִיב / לַמְּדֵנִי חֻקֶּיךָ:


לְלֵב נִמְהָר / יְהֶא נִזְהָר / לַעֲשׂוֹת אֶת רְצוֹן קוֹנוֹ

וְיִתְיַסֵּד / לְהָבִין סוֹד / בְּתוֹרָתוֹ וְהֶגְיוֹנוֹ

וְדָת מוּסָר / וְלֹא יֶחְסָר / וְלֹא יָמוּת בְּשִׁגְיוֹנוֹ

בְּעֵת יָשׁוּב / יְהֶא חָשׁוּב / וְאָז יַחְזֹר לְצִבְיוֹנוֹ

שְׁמַע מִלִּין / וְאַל תַּלִּין / לְבָבָךְ בַּחֲטָאֶיךָ:


שְׁחִי רוּחִי / בְּמַר שִׂיחִי / לְיוֹם שׁוּבֵךְ תְּנִי צֵידָה

וְהִכּוֹנִי / וְתִשְׁכֹּנִי / בְּתוֹךְ קָהָל וְגַם עֵדָה

תְּנִי כוֹפֶר / לְגוּף הֵפַר / בְּרִית אַחִים וְגַם זָדָה

וְהָבִינִי / בְּיוֹם דִּינִי / וְעָלַי שַׁלְּמִי תוֹדָה

בְּבוֹא צוֹרֵף / כְּדוֹב טוֹרֵף / בְּיוֹם הַדִּין וְיַעְמִידָה

מְנָת כּוֹסִי / וְצוּר מַחְסִי / מְחֹל נָא לַעֲבָדֶיךָ:


בְּגָלוּתִי / וְדַלּוּתִי / אֲנִי הוֹמֶה בְּמַר בִּכְיִי

וְלִבִּי סָר / וְנִתְיַסָּר / אֱלֹהַי רַפְּאָה חָלְיִי

וְחוֹן גּוּף דַּל / אֲשֶׁר חָדַל / וְחַזֵּק יוֹצְרִי רִפְיִי

רְצֵה עָלַי / בְּמִשְׁאָלַי / רְאֵה שִׁפְלִי וְרֹב עָנְיִי

בְּךָ בִּטְחִי / סְבוֹל טָרְחִי / בְּיוֹם דָּפְקִי דְלָתֶיךָ:


זְמָן אָתָה / בְּהַשְׁחָתָה / וְהִכְבִּיד עֹל וְרֹב דַּלּוּת

וּבֶן אָמָה / מְאֹד הָמָה / וּמִתְגָּאֶה כְּרוּם זֻלּוּת

וְעֵינָיו צָר / בְּעַם נֶעְצָר / שְׁהוּא צוֹלֵל בְּבוֹר גָּלוּת

וְרָם לִבּוֹ / בְּתוֹךְ קִרְבּוֹ / בְּעַמָּךְ יִנְהֲגָה שִׁפְלוּת

אֱלֹהַי חוּן / לְדַל בָּחוּן / בְּחֵשֶׁק אַהֲבָתֶךָ:


יְשׁוּרוּן לֵךְ / וְהִמָּלֵךְ / בְּשִׂכְלָךְ לַעֲשׂוֹת טוֹבָה

וְהִתְיַשֵּׁר / בְּרֹב הֶכְשֵׁר / בְּלִי עָוֹן וְלֹא חוֹבָה

גְּבוּל תָּשִׂים / בְּטוּב מַעְשִׂים / וְאֶת דַּעְתָּךְ תְּיַשֵּׁבָה

הֲדַר זָקֵן / בְּחֵן תַּקֵּן / וְתָקוּם מִפְּנֵי שֵׂיבָה

עֲשֵׂה לָךְ רָב / בְּיוֹם תִּקְרָב / וְלֹא תִּשְׁכַּב לְמַעְצֵבָה

בְּיָהּ הַשְׁלֵךְ / יְהָבֶךָ / וְהוּא יִהְיֶה מְכַלְכֶּלְךָ:
