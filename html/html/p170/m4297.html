היתאפק אנוש כלו עלומיו

הֲיִתְאַפַּק אֱנוֹשׁ כָּלוּ עֲלוּמָיו / וְכָלוּ בַנְּדֹד שָׁנָיו וְיָמָיו

וְדָמוּ לַעֲנַן־בֹּקֶר מְשׂוֹשָׂיו / וְאֵיךְ יִתָּֽפְשׂוּ הַבְלֵי חֲלֹמָיו

וּמֵעָיו יֶהֱמוּ כַיָּם בְּזָכְרוֹ / זְמָן עָבַר כְּבָרָק עִם־נְעִימָיו

זְמָן הָיָה בְּיַד חֵפְצוֹ כְּחֹמֶר / בְּיַד יוֹצֵר עֲדֵי רָכְבוֹ מְרוֹמָיו

וְחִישׁ נֶהְפַּךְ בְּלִי־פֶשַׁע לְאַכְזָר / וּפָרַשׂ לַחֲבֹל חַבְלֵי חֲרָמָיו

וְאֹהֶל יֹֽדְעָיו הֶחְרִיב וְאֶחָיו / בְּלִי שָׂרִיד וְכָל־אַנְשֵׁי שְׁלֹמָיו

וְלָמֹד פַּאֲתֵי עוֹלָם קְרָאוֹ / וְלַשְׁקוֹת מִדְּמֵי עֵינָיו תְּלָמָיו

וּמִבּוֹר הַנְּדֹד יִצְעַק יְחִידִי / וְאֵין מַקְשִׁיב וּמֵעִמְקֵי תְהֹמָיו

וְלוּלֵי לַהֲבֵי־אֵשׁ בֵּין צְלָעָיו / מְגוּרָיו הֶחֱרִיבוּ מֵי זְרָמָיו

וְעוֹד יָדוֹ נְטוּיָה עַל־לְבָבוֹ / לְהַרְגִּיזוֹ בְּפַחְדָּתוֹ וְאֵימָיו

וּמִכְתָּב מִיְּדֵי עֶזְרָא עֲזָרוֹ / הֲכִי מִצּוּף־דְּבַשׁ מָתְקוּ טְעָמָיו

וְחָלְקוּ מַחֲמָאֹת דַּבְּרֹתָיו / וְהֵם שִׂכִּים בְּעֵין חֹמְדָיו וְקָמָיו

צְעִיר שָׁנִים וְלוֹ טַעַם זְקֵנִים / וְעוּל יָמִים וְנִקְרָא רֹאשׁ בְּעַמָּיו

כְּתָב יָעִיד לְיֹשֶׁר מַעֲלָלָיו / וּמַהְלָלָיו וְתִפְאֶרֶת קְדוּמָיו

בְּתָמְכוֹ נָֽטְפוּ־מֹר אֶצְבְּעֹתַי / וְנָשָׂא אֶל־נְחִירַי רֹאשׁ בְּשָׂמָיו

כְּנֹפֶךְ עַל־פְּנֵי שֹׁהַם מְצָאתִיו / וְכִתְכֵלֶת עֲלֵי חוֹרֵי גְלוֹמָיו

כְּיָם גָּדוֹל חֲשַׁבְתִּיהוּ וְגַלָּיו / כְּנֹפֶת־צוּף וְקוֹל כִּנּוֹר רְעָמָיו

בְּשָׂמִים יִגְרְשׁוּ מֵימָיו וְאֹדֶם / גְּבֻלָתוֹ וְעַל־חוֹפוֹ לְשָׁמָיו

מְעִיל עֹז יָעֲטוּ כָל־מַאֲמָרָיו / וְגַם צָנְפוּ צְנִיף הוֹד כָּל־נְאֻמָיו

וְאָרִיתִי לְאַט מִגְדֵי אֲהָבָיו / וְרָוִיתִי בְּיֵין דֹּדֵי כְרָמָיו

בְּשִׁיר הוּשַׁת צְמִיד יַד הַתְּבוּנַה / וְעַל־אַף הַֽיְקָר כַּעְדִי נְזָמָיו

לְכָה כּוֹכָב אֲשֶׁר יַזְהִיר בְּאֶרֶץ / וְהוֹדוּ לוֹ שְׁנֵי אוֹרִים בְּשָׁמָיו

לְכָה אִיעָֽצְךָ עֵצָה וּמַה־טּוֹב / יְדִיד יוֹעֵץ בְּתָם־לֵבָב תְּמִימָיו

דְּרֹשׁ מוּסָר בְּעוֹדְךָ לַח וְתִצְלַח / וְתָרוּם עַל־מְתֵי־שֵׂכֶל וְרָמָיו

וְעַל־דַּלְתֵי אֱמֶת תִּשְׁקֹד וְתִקֹּד / לְכָל־מַשְׂכִּיל לְהִתְבּוֹנֵן עֲלֻמָיו

וְהֶהָדָר תְּיַשֵּׁר אֶת־הֲדוּרָיו / וְהַמַּדָּע תְּפַתֵּחַ סְתֻמָיו

לְיַמֵּי הַֽמְזִמּוֹת רֵד בְּכָל־עֵת / וְיָם אַחֵר זְנַח לִשְׂחוֹת בְּמֵימָיו

וְאַל־תִּלְעַג לְאִישׁ חוֹבֵר חֲבָרָיו / וְאַל־תָּבֻז בְּעֵת יִקְסֹם קְסָמָיו

הֲכִי יֵשׁ לַנְּבוֹנִים רַעֲיוֹנִים / וְכָל־עָם יַחֲשֹׁק גִּנְזֵי בְרֹמָיו

וְהַסֶּכֶל יְחַרְחַר רִיב פְּתָאיִם / וְהַשֶּׂכֶל יְהַפְרִיד בֵּין עֲצוּמָיו

וְיָגֵעַ בְּיַלְדוּתוֹ יְהִי נָח / בְּשַׁחְרוּתוֹ וְיַחְלִיץ אֶת־עֲצָמָיו

וְלֹא יִרְדֶּה בְזֹקֶן צוּף תְּבוּנָיו / עֲדֵי יֹאכַל בְּנֹעַר אֶת־רְתָמָיו

וְאַל־תַּחְמֹד מְתֵי מִדּוֹת וּמִדּוֹת / וְאֵיךְ יַחְמֹד אֱנוֹשׁ תַּבְנִית צְלָמָיו

וּמַה־בֶּצַע בְּאִישׁ תֹּאַר בְּלִי־בִין / וְלוּ אוֹפִיר יְהִי־לוֹ עִם־כְּתָמָיו

כְּלֶחֶם־בִּין יְהִי שֹׂבַע לְרָעֵב / וְסוּת־שֵׂכֶל יְכַסֶּה מַעֲרֻמָיו

וְאַל יַשִּׁאֲךָ מִדְבַּר מְהַלֵּל / וְלוּ יַקְרִיב לְךָ זִבְחֵי שְׁלָמָיו

הֲכִי כָל־הָאֱנוֹשׁ יַחְלִיק לְשׁוֹנוֹ / וּמַצְפֻּנָיו שְׁטָר שֶׁקֶר חֲתוּמָיו

וְתוּר אַחַר דְּבַר אַנְשֵׁי מְלִיצָה / וְחִידֹתָם בְּכָל־דּוֹר וַחֲכָמָיו

וְהִתְרָע אֶל־מְתֵי אֹמֶן וְצֶדֶק / וּמַה־נֶּחְמָד שְׁכֹן כָּל־אִישׁ לְדוֹמָיו

וְהַקְטֵן עַצְמְךָ לִפְנֵי עֲצוּמִים / הֲכִי גֵאִים בְּעֵינֵי אֵל זְעוּמָיו

וְאִישׁ מַשְׂכִּיל יְהִי נֹשֶׁה חֲסָדָיו / וְכָל־רֶגַע יְהִי זֹכֵר אֲשָׁמָיו

וְשָׂא שָׁלוֹם כְּמֵי־יָם אוֹ מְעֹתָיו / וְגוּשׁ עָפָר וּמִסְפַּר כָּל־יְקוּמָיו

וְעַל־הָאָב אֲבָל אַב הַחֲסָדִים / כְּחֵיל־שַׁחַק וְכָל־נִטְפֵי גְשָׁמָיו

יְהִי שָׁלֵו יְמֵי חַיָּיו וְשָׁלוֹם / וְאֵל יָשִׂים אֱלֵי־דֶרֶךְ פְּעָמָיו.
