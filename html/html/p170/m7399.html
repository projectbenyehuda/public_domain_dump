נטפי דמעות מערף נגרעו

נטפי דמעות מערף נגרעו / ר' משה אבן-עזרא


נִטְפֵי דְמָעוֹת מֵעֲרֹף נִגְרָעוּ / אוֹ גַחֲלֵי־אֵשׁ מִשְּׂרֹף שָׁקָעוּ

כִּי־אֶמְצְאָה אֵלֶּה נְבֻכִים בֵּין שְׁנֵי / צִדַּי וְאֵלֶּה בֵּין לְחָיַי תָּעוּ

אָרִיב יְקֹד אִשִּׁי וְאוֹכִיחַ הֲמוֹן / דִּמְעִי הֲכִי בוֹ מִשְּׁלֹט נִמְנָעוּ

וַיַּעֲנוּ גָּבַר מְאֹד הַשֹּׁד עֲדֵי / חֻקָּם אֲשֶׁר־לוֹ נִבְרְאוּ פָּרָעוּ

לֹא־מָֽצְאוּ עַזִּים יְדֵיהֶם לֶאֱלוֹת / לֹא־אָֽמְצוּ לָנוּד וְדוּמָם פָּעוּ

אֵיךְ עָֽמְדוּ לִבּוֹת וְאֵיךְ לֹא־שָֽׁלְפוּ / וַיֵּֽצְאוּ מִגֵּו לְשֹׁד שָׁמָעוּ

אֵיכָה עֲלֵי צוּרֵי שְׁלָחִים יִפְּלוּ / לִבּוֹת נְפִלֵיהֶם וְלֹא יִבְצָעוּ

הֹוָה לְחִילָהּ הַנְּפָשִׁים שָֽׁכְנוּ / רָעָה לְחוּלָהּ הַקְּרָבִים נָעוּ

יִתְלַחֲשׁוּ אִישִׁים בְּהַגִּידָהּ הֲכִי / רָהוּ לְהָרִים קוֹל וְגַם־יָרָעוּ

מִי הֶאֱמִין כִּי מִגְדְּלֵי עֹז נָֽפְלוּ / אוֹ כִי עֲצֵי־עֵדֶן בְּאַף גֻּדָּעוּ

אוֹ בָא מְאוֹר־אֶרֶץ אֲשֶׁר בֵּין כּֽוֹכְבֵי / אוֹר עִקְּבוֹתָיו מֵאֱנוֹשׁ נוֹדָעוּ

אָחִי אֲבָל אָח לַמְּאוֹרִים אַךְ בְּיוֹם / יִתְנוֹסֲסוּ יוֹסִיף וְיִגָּרָעוּ

צֹעֵד עֲלֵי־אֶרֶץ וְיַלְדֵי מַחְשְׁבוֹת / שִׂכְלוֹ עֲדֵי כַנְפֵי כְרוּב נָגָעוּ

סוֹדָיו עֲלֵי כַנֵּי אֱמוּנָה נֽוֹסְדוּ / מִלָּיו עֲלֵי אַדְנֵי אֱמֶת הָטְבָּעוּ

מֵאָז זְמָמָיו נִתְכְּנוּ לָפִיק רְצוֹן / צוּרוֹ וְכֹל הַיּוֹם בְּחֻקָּיו שָׁעוּ

אֶצְלוֹ אֲצִילֵי מַאֲמָר שַׁחוּ כְּמוֹ / נֶגְדּוֹ נְגִידֵי מַעֲנֶה כָּרָעוּ

שָׂמוּ לְפִיהֶם כַּף מְתֵי צַחוֹת בְּיוֹם / אֹמֶר וְדִבְרֵיהֶם לְפָנָיו לָעוּ

הִרְבּוּ גְּבֹהָה לַעֲנוֹת יוֹם אָֽלְמוֹ / אַךְ נֶאְלְמוּ עִמּוֹ וְאֵלָיו שָׁעוּ

סָרוּ לְמִשְׁמַעְתּוֹ מְתֵי־סֵפֶר וְלוֹ / אָזְנֵי מְבִינֵי הַסְּפָר רֻצָּעוּ

אֶתְמַהּ עֲלֵי עֵטָיו אֲשֶׁר לֹא־פָֽרְחוּ / מִנַּחֲלֵי יָדָיו אֲשֶׁר נָבָעוּ

נָדְבוּ עֲדֵי נֶגְדָּם יְדֵיהֶם אָֽסְפוּ / עָבִים וּמֵהַגְשִׁים עֲנָנִים שָׁעוּ

טוּרָיו לְמֹר עַל־הַכְּפֹר דָּמוּ וְאִם / נֹפֶךְ עֲלֵי־שֹׁהַם בְּיָד קָלָעוּ

הוֹרָיו לְפִי־חֶסֶד יְדֵיהֶם זָֽרְעוּ / וַיְרַו בְּמֵי צִדְקוֹ אֲשֶׁר זָרָעוּ

הָהָר אֲשֶׁר חָמַד אֱלֹהִים אֵיךְ בְּנֵי / אִישׁ נָֽשְׂאוּ גָבְהוֹ וּבוֹ פָשָׁעוּ

נָגְהָם צְבָא־שַׁחַק בְּמוֹתוֹ אָֽסְפוּ / כִּי־דַּל מְאוֹר שֶׁמֶשׁ וּפָנָיו רָעוּ

אִלּוּ לְכוֹכָבִים קְרָבִים נָהֲמוּ / כַיָּם וְעֵינַיִם אֲזַי דָּמָעוּ

הִפְלִיא זְמָן לִמְרֹד וּבָנָיו נָֽשְׂאוּ / יָד בַּאֲדֹנֵיהֶם וּבוֹ פָשָׁעוּ

לָקְחוּ אֲבִי־לֶקַח וְסוֹד הַמַּאֲמָר / שָׁבוּ וּמוֹפֵת הַֽיְקָר קָבָעוּ

לָקְחוּ אֲדֹנִי מֵעֲלֵי רֹאשִׁי וְרֹאשׁ / הַדּוֹר בְּאַף פָּרְעוּ וְלֹא פָרָעוּ

אָבִי וּמַחְכִּים רַעֲיוֹנַי יוֹם אֲשֶׁר / יִשְׂתָּעֲרוּ לַבִּין וְיִשְׁתַּעְשָׁעוּ

לֹא־מָֽצְאוּ נוֹחַ עֲלֵי תֵבֵל עֲדֵי / בֵינוֹת צְלָעַי אָהֳלָם תָּקָעוּ

יִדּוּ אֲבָנִים בִּי בְּבוֹר גֵּרוּת וְלֹא / חָטְאוּ סְגוֹר לִבִּי בְּעֵת קָלָעוּ

אָסְרוּ לְהַשְׁמִיד אֶת־בְּנֵי אָבִי אֲבָל / לַפְרִיד בְּנֵי־עַיִשׁ הֲלֹא נִשְׁבָּעוּ

דָּמִי לְאַט שָׁתוּ וְלֹא־רָווּ כְּמוֹ / אָכְלוּ שְׁאֵר גּוּפִי וְלֹא שָׂבָעוּ

אֹתִי בְּפִצְעֵי הַנְּדֹד עִנּוּ עֲדֵי / יָזְמוּ לְהָמִית אַחֲרֵי פָצָעוּ

אֶחָיו אֲשֶׁר מִיּוֹם כְּבוֹדוֹ פָּֽקְדוּ / לֹא נֶעְזְרוּ מִשֹּׁד וְלֹא נוֹשָׁעוּ

חָלוּ לְחוּל אֵידוֹ וְעֹרֹתָם עֲלֵי / עַצְמָם כְּרֶגַע נִמְאֲסוּ רָגָעוּ

לֹא־חָֽגְרוּ שַׂק כִּי־לְבוּשָׁם מִשְּׁחוֹר / הַלֵּב וּבִדְמֵי הַבְּכִי צָבָעוּ

לֹא־אָֽסְפוּ אֵלָיו לְהָשִׁית יָד עֲלֵי / עֵינָיו וְלֹא־פָרְעוּ וְלֹא־קָרָעוּ

אִמְרוּ הֲנַפְשׁוֹ בָּחֲלָה בָהֶם וְאִם / הֵם מָאֲסוּ אֹתוֹ וּבוֹ נָקָעוּ

אוֹ אִם־זְמָן בֹּגֵד הֱפִיצָמוֹ בְּלִי / חֶמְלָה וּבִמְכוֹנֵי נְדֹד גָּוָעוּ

חָלַף וְלֹא־חָלְפוּ סְעַפֹּתָיו אֲשֶׁר / יָדָיו יְמֵי הַשַּׁחֲרוּת נָטָעוּ

הוּא אִם־עֲרִירִי נֶאֱסַף לֹא־נִכְחֲדוּ / יַלְדֵי תְהִלּוֹתָיו וְלֹא נִכְנָעוּ

יִבְלֶה עֲפַר־אֶרֶץ וְהֵם יִתְחַדְּשׁוּ / יֵצְרוּ עֲרִיפֶיהָ וְיִשְׂתָּרָעוּ

אֵיךְ אֶתְּנָה שֵׁנָה לְעַפְעַפַּי וְלֹא / יִתְחַבְּרוּ לָנוּם וְלֹא יִתְרָעוּ

אוֹ אֵיךְ אֱחִי אִם־לֹא אֲקַוֶּה לַעֲלוֹת / עוֹד שַׁחֲרֵי פָנָיו וְיִבָּקָעוּ

מִיקוֹד צְלָעַי אֶהֱמֶה תָמִיד וְאִם / רַגְלַי בְּצוּלַת הַבְּכִי טָבָעוּ

לֹא־חָֽמְדָה נַפְשׁוֹ שְׁכָן־לוֹ עִם־מְתֵי / רֹעַ אֲשֶׁר רוּחַ הֲתֻלִים רָעוּ

דּוֹר נֶאֱלָח עֵינָיו מְאֹד רָמוּ אֲבָל / אֹתָם לְהַבִּיט הָאֱמֶת הֵשָׁעוּ

רָפוּ יְדֵיהֶם לַעֲשׂוֹת חֶסֶד וְתֹם / אָכֵן בְּרָב־כֹּחַ לְהָרַע צָעוּ

קוֹל הֶעֱבִיר מָוֶת בְּכָל־נוֹצָר שְׁכֹן / כֵּפִים בְּכַף־רֶגֶל תְּמוֹל רָקָעוּ

מִקְרֵה כְסִיל יִקְרֶה חֲכָמִים נֽוֹדְעוּ / סָפוּ וְחָכְמָתָם וְהִתְבַּלָּעוּ

אַיֵה מְלָכִים יִסְּדוּ חֶפְצָם שְׁאַל / אִם־נִמְשְׁכוּ כִמְעַט עֲדֵי בִצָּעוּ

יָלְדוּ לְבֶהָלָה וְחַיִל עָֽתְקוּ / עָזְבוּ וְלָרִיק כָּל־אֲשֶׁר יָגָעוּ

כֶּסֶף כְּחוֹל צָבְרוּ וּפַח הָיָה לְיַד / מַעְגַּל תְּבוּסָתָם וְהִתְגַּלָּעוּ

שָׂמוּ אֲדָמָה מַאֲבֻסֵיהֶם וְהֵם / סָבִיב בְּלִי־לֵב כַּשְּׁוָרִים גָּעוּ

אָכְלוּ וְשָׂבֵעוּ וְהִשְׁמִינוּ עֲדֵי / קָיוּ לְפִי־רֶגַע אֲשֶׁר בָּלָעוּ

מָוֶת אֲשֶׁר הִשְׁוָה נְבָרִים צָֽדְקוּ / בִּמְעֵי אֲדָמָה עִם־אֲשֶׁר רָשָׁעוּ

לוּ יִֽהְיֶה הֹדֵר פְּנֵי שָׂרִים אֲזַי / יָדָיו לְאָחִי מִשְּׁאוֹל פָּדָעוּ

תָּמְהוּ בְנֵי־יוֹם לַחֲיוֹתִי אַחֲרֵי / הַמֹּֽצְאוֹת אֹתִי וְהִתְרֹעָעוּ

לֹא־יָֽדְעוּ כִּי־לִי מְזִמּוֹת יִפְגְּעוּ / בָהֶם בְּטֶרֶם בִּי בְאַף יִפְגָּעוּ

מָתַי אֲרָיוֹת פָּחֲדוּ מִקּוֹל אֱנוֹשׁ / אוֹ מֵעֲבֹר רוּחַ הֲרָרִים זָעוּ

שָׁלוֹם לְאוֹר מַעְרָב אֲשֶׁר עָרַב כְּפִי / שֹׁאַת לְבָבוֹת בַּעֲדוֹ שֻׁסָּעוּ

יִבְלֵי סְלִיחָה יַעֲנוּ קִבְרוֹ לְבַל / יִבְלֶה וְתַחְתָּיו רַחֲמִים יֻצָּעוּ

יַעְמֹד לְגוֹרָלוֹ בְּיוֹם יִמָּֽשְׁכוּ / בוֹ שֽׁוֹפְרוֹת יֶשַׁע וְיִתָּקָעוּ

לִבְכּוֹת עֲלֵי אַחַי פְּעָמַי אֶחֱרוּ / כִּמְעַט וְאֶסַּע אֶל־אֲשֶׁר נָסָעוּ.
