משל עה: זְאֵב וְרֹעֶה

כַּאֲשֶׁר תַּעֲשֶׂה לְאִישׁ טוֹבָה / אַל תִּרְאֶה לוֹ חוֹבָה


זְאֵב הָיָה נִרְדָּף בַּמְסִלָּה / בָּעֲרָבָה בָּהָר וּבַשְּׁפֵלָה / וּמֵרֹב פַּחְדּוֹ בַדֶּרֶךְ תֹּעֶה / וַיָּבֹא עַד הָרֹעֶה / וַיֹּאמַר:קוֹל שָׁאוֹן אַחֲרַי צֹרֵחַ / וַאֲנִי מִפָּנָיו בֹּרֵחַ / מִקּוֹל פָּרָשׁ וְרֹמֵה קֶשֶׁת / כִּי לְלָכְדֵנִי פָרְשׂוּ רֶשֶׁת / וַיַּאֲמִינוּ וַיַּשְׂמְאִילוּ / בָּאוּ בֶּעָבִים וּבַכֵּפִים עָלוּ / לְבַעֲבוּר כִּי זְאֵב עֲרָבוֹת יְשָׁדְדֵם / אָמְרוּ: לְכוּ וְנַכְחִידֵם / וְלֹא יִשָּׁאֵר מֵהֶם פַּרְסָה / מִי יִתֵּן וְעֵינְךָ עָלַי חָסָה / וּבְנֶפֶשׁ צָרַי אַל תִּתְּנֵנִי / יִסְעֲרוּ לַהֲפִיצֵנִי / וּמִן הַיַּעַר אֹתִי הִדִּיחוּ / תַּחַת חָרוּל יְסֻפָּחוּ / וְאַתָּה מְעוֹדֵד לְנֹדֵד / הֱיֵה סֵתֶר לִי מִפְּנֵי שֹׁדֵד / וּתְהִי נַפְשִׁי בְעֵינֶיךָ יְקָרָה / אַל תַּגְדֵּל פִּיךָ בְּיוֹם צָרָה / וְנִסְתַּרְתִּי עַד שֹׁב הָרֹדְפִים / אֲשֶׁר בְּאַוַּת נַפְשָׁם אֹתִי שֹׁאֲפִים / וּכְבִיאָתִי אִתְּךָ כָּעֵת אַרְבֶּה עָצְבָּם / וְשֶׁן־בְּהֵמוֹת אֲשַׁלַּח בָּם / וַיֹּאמֶר הָרֹעֶה: לֵךְ וְהִסָּתֵר / בְּנִקְרַת הָרֵי בָתֶר / וַאֲנִי אֲסַכֵּל עֲצַת הָאֹרֵב / וַאֲמַלֶּטְךָ מֵאִבְחַת חָרֶב / וַיָּנָס הַזְּאֵב לְהֵחָבֵא בְּנִקְרַת הַצּוּר / יִדְכֶּה יָשֹׁחַ וַיְהִי עָתוּר / וְהָרֹדְפִים אֲשֶׁר אַחֲרָיו תָּעוּ / עַד אֵצֶל הָרֹעֶה הִגִּיעוּ / לֵאמֹר לוֹ: הֲרָאִית הַזְּאֵב אֲשֶׁר עַל עָרֵינוּ יִשְׁקֹד / וְטָרַף זְרוֹעַ אַף קָדְקֹד / וַיַּעַן לָהֶם: רְאוּ אֶת נְתִיבוֹ / זֶה הַדֶּרֶךְ תֵּלְכוּ בוֹ / כַּתְּרוּ אֹתוֹ הַרְדִּיפוּהוּ / מְנוּחָה הַדְרִיכוּהוּ / מַהֲרוּ אַחֲרָיו לְהַשִּׂיגוֹ / אָרוּר מֹנֵעַ חַרְבּוֹ מִלְּהָרְגוֹ / כָּל אֶחָד לָרוּץ אַחֲרָיו תָּעָה / וַתְּהִי מְרוּצָתָם רָעָה / וְכַאֲשֶׁר הֵמָּה נִלְאוּ / חֲדָשִׁים מִקָּרוֹב בָּאוּ / וְכָל אֶחָד חַרְבּוֹ יָרִיק / וְיִגְעוּ עַמִּים בְּדֵי רִיק / כִּי הָרֹעֶה בְפִיו וּבְיָדוֹ יַסִּיעֵם / לְהַרְאוֹתָם הַדֶּרֶךְ אֲשֶׁר יַתְעֵם / אָכֵן בַּזְּאֵב יִתֵּן עֵינָיו / וְאֶת פָּנָיו אֶל מוּל פָּנָיו / וְכַאֲשֶׁר הָרֹדְפִים שָׁבוּ / מִזֶּרֶם הָרִים יִרְטָבוּ / הָרֹעֶה עַד הַזְּאֵב הָלַךְ / וַיֹּאמֶר אֵלָיו:קוּם לֶך לְךָ / הֲקִימוֹתִי אֶת דְּבָרֶיךָ / לֹא נָתַתִּי אֶת פִּי לַחֲטִיא אֶת בְּשָׂרֶךָ / וְכָל אַנְחָתִי הִשְׁבַּתִּי / לַחֲטֹא חִכִּי לֹא נָתַתִּי / מִלַּטְתִּיךָ מֵרֶדֶת שָׁחַת / חֲזַק וֶאֱמַץ אַל תַּעֲרֹץ וְאַל תֵּחָת / וַיַּעַן הַזְּאֵב: הִגְדַּלְתָּ לִי חֲסָדֶיךָ / מִדְּבַר פִּיךָ וּמִנְּטִיַּת יָדֶיךָ / בַּאֲשֶׁר לִמַּדְתָּ הָרֹדְפִים מְסִלּוֹת / לָלֶכֶת אֳרָחוֹת עֲקַלְקַלּוֹת / אַךְ עַל עֵינֶיךָ תְּלֻנָּתִי / כִּי תָמִיד הִבִּיטוּ אֹתִי / וְלֹא נָחוּ וְלֹא שָׁקְטוּ / עֵינֶיךָ לְנֹכַח הִבִּיטוּ / וְהֵמָּה הָיוּ בְּעוֹכְרַי / לָתֵת אֹתִי בְיַד צָרַי / מֵהֶם הָיִיתָ גֹּדֵר פָּרֶץ / כִּי מִנֶּגֶד תִּרְאֶה אֶת הָאָרֶץ / אֲשֶׁר שָׁמָּה נִסְתַּרְתִי / כִּמְעַט אֶל מַבָּטְךָ נוֹקַשְׁתִּי.


וְהַמָּשָׁל פֶּן יִהְיוּ עֵינֶיךָ רְכִילִים / אִם יִהְיוּ דְבָרֶיךָ פְלִילִים / לְדַבֵּר לְרֵעֲךָ עַל אִישׁ מִמַּעֲלָלָיו / הִשָּׁמֵר אַל תֵּפֶן אֵלָיו / לְמַרְאֵהוּ וּלְקוֹמָתוֹ / לְצַלְמוֹ וְלִדְמוּתוֹ / כִּי עֵינֶיךָ יַגִּידוּ לוֹ / דְּבָרֶיךָ בִּלְבַב שִׂכְלוֹ / וְאִמְרַת הֶחָכָם שָׁמַעְתִּי בְאָזְנַי / בְּרִית כָּרַתִּי לְעֵינַי / מֵהַבִּיט עַל אִישׁ בְּדַבְּרִי בוֹ / כִּי מִמַּרְאֵה עֵינַי יָבִין בְּלִבּוֹ / אִם אַאֲרִיךְ לְשׁוֹנִי מִקְּלוֹנוֹ אוֹ מֵחַסְדּוֹ / עֵת עַפְעַפַּי יֵישִׁירוּ לְנֶגְדּוֹ / אִם יָבִין הַקָּלוֹן יֵהָפֶךְ לוֹ לְחֶרְפָּה / וְאִם הַחֶסֶד יֵחָשֵׁב לַחֲנֻפָּה / וְאָמַר שְׁלֹמֹה בְּחָכְמָתוֹ הַנִּשְׂגֶּבֶת / קֹרֵץ עַיִן יִתֵּן עַצֶּבֶת.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


חָכָם הוּא אַךְ כַּאְשֶׁר יֹעֵץ / עַל אִישׁ שָׁם אֶת עֵינוֹ יַפְנֶה

הַן הַמְדַבֵּר מֵחְרִיץ חָלָב / לֹא יַבִּיט אֶל כָּל הַגְּבִינָה.
