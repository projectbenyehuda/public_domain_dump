משל יב: אַרְיֵה, עֵז, רָחֵל וּפָרָה

מִתְחַבֵּר עִם מִי שֶׁמִּמֶּנּוּ גָדַל / לְסוֹף כְּבוֹדוֹ יִדָּל


אַרְיֵה הָלַךְ לִטְרֹף טֶרֶף / לַנְפִּיל יָרֵךְ וְלִכְרֹת עֹרֶף / לָצוּד צַיִד לְהָבִיא / וְעִם חֶבְרָתוֹ בְּנֵי לָבִיא / עֵז וְרָחֵל וּפָרָה / וּבְבֹאָם אֶל הַיַּעְרָה / אָמְרוּ: כָּל אֲשֶׁר יִפְגָּעֵנוּ נַפִּיל חָלָל / הֲלֹא יִמְצְאוּ יְחַלְּקוּ שָׁלָל / וַיִּרְאוּ צְבִיָּה וַיִּלְכְּדוּהָ בִשְׁחִיתוֹתָם / וַיֹּאמְרוּ לָאֲרִי לְהַרְאוֹתָם / מַה יִּרְצֶה מִמֶּנָּה לְמַלֵּא סִפְקוֹ / וּמַה יִּקַּח כָּל אֶחָד לְחֶלְקוֹ / וְהָאֲרִי חִקְרֵי לִבּוֹ גְּדוֹלִים / וִימִינוֹ לְהַלְמוּת עֲמֵלִים / וַיֹּאמֶר לָהֶם: שִׁמְעוּ נָא עֵצָה / לֹא תוֹצִיאוּ מִן הַבָּשָׂר חוּצָה / הַצְּבִיָּה נָפְלָה בְגוֹרָלִי / אֲנִי מְצָאתִיהָ, כֻּלָּהּ שֶׁלִּי / חֵלֶק רִאשׁוֹן, כִּי אֲנִי מֶלֶךְ / וְהַשֵּׁנִי, בִּהְיוֹתִי הוֹלֵךְ / אַחֲרֶיהָ בִמְרוּצָה וְהִשַּׂגְתִּיהָ / חֵלֶק שְׁלִישִׁי, לִי חִלַקְתִּיהָ / כִּי אֲנִי גִבּוֹר מִכָּל הַיְקוּם / חֵלֶק הָרְבִיעִי, מִי יָקוּם / לְהַפְרִידוֹ מִן הַשְּׁלֹשָׁה / מִי יַעֲנֵנִי עַל זֹאת קָשָׁה / לָקַחַת אוֹתָהּ שֶׁלֹּא בִרְצוֹנִי / לֹא אַכְזָר כִּי יְעוּרֵנִי / וּמִי הוּא לְפָנַי יִתְיַצָּב / לְהָפֵר דִּין מִפִּי נֶחְצָב / וַיְהִי כְּשָׁמְעָם מַעֲנָיו / רָגְזוּ וְחָלוּ מִפָּנָיו / וּלְכָל הַשּׁוֹאֵל חֵלֶק יִשְׂחָק / וְגָעַר בּוֹ וְנָס מִמֶּרְחָק / וְאַחַר אֲשֶׁר נָסוּ לְקוֹלוֹ / כָּל הַצַּיִד נִשְׁאַר לוֹ.

וְהַמָּשָׁל לְאָדָם הַמִּתְחַבֵּר / עִם חָזָק מִמֶּנּוּ אִם יְדַבֵּר / מִשֻּׁתָּפוּת שֶׁבֵּינֵיהֶם / הֶעָשִׁיר אָז כַּאֲרִי נוֹהֵם / וְיָשִׁיב לוֹ דָבָר בְּזַעְמוֹ / אֵין לוֹ חֵלֶק וְנַחֲלָה עִמּוֹ / וִיאַבֵּד הוֹנוֹ וְאַהֲבָתוֹ / בַּיּוֹם הַהוּא אוֹ לְמָחֳרָתוֹ / וְהַטּוֹב תִּגָּלֶה צִדְקָתוֹ / כְּצֵאת הַשֶּׁמֶשׁ בִּגְבוּרָתוֹ.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


וְאִם דָּרֹשׁ דָּרַשׁ / לְחֶבְרַת עָשִׁיר רָשׁ / בְּלֹא גוֹרָל נִגְרָשׁ / כְּמוֹ רֵעֵי הָאְרִי

בְּעֵת שָׁאְלוּ חֶלְקָם / בְּגוֹרָל אֶל חֵיקָם / הֱשִׁיבָם חִישׁ רֵיקָם / בְּזַעַף אַך וָחְרִי

וְתֶבֶן בִּמְקוֹם בָּר / לְחֵלֶק אִישׁ חָבַר / / לְאִישׁ עָלָיו יִגְבָּר / וְתַחַת מֹר מֶרִי

מְבַקֵּשׁ שַׂר אִתּוֹ / לְעֻמַּת מַחְבַּרְתּוֹ / בְּתוֹךְ סִכְלוּת שִׁבְתּוֹ / וְגַם יֵלֵךְ קֶרִי.
