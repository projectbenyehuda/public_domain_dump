משל טו: אַרְיֵה, עַכְבָּר וָרֶשֶׁת

טוֹב לִהְיוֹת טוֹב וּמֵטִיב לַכֹּל / כִּי יֵשׁ עֵת וּזְמָן לְכָל חָפֵץ לִהְיוֹת יָכֹל


אַרְיֵה נִרְדַּם בִּמְקוֹם מִדְבָּר / וַיִּדְרֹךְ עָלָיו עַכְבָּר / עַד כִּי הֵקִיץ מִשְּׁנָתוֹ / וַיִּפֶן לִרְאוֹת מִי אוֹתוֹ / אֲשֶׁר מֶלֶךְ גָּדוֹל כָּמוֹהוּ / בְּרַגְלַיִם יִרְמְסֵהוּ / וַיִּתְבּוֹנֵן כִּי בְּצִדּוֹ / הָעַכְבָּר עֹמֵד לְבַדּוֹ / הוּא הָיָה אֵין זָר אִתּוֹ / וַיֹּאמַר: הִגִּיעָה עִתּוֹ / וַיִּקְרָא לוֹ: עַכְבָּר נִבְזֶה / מִשְׁפַּט מָוֶת לְךָ אֶחֱזֶה / כִּי הֱקִיצוֹתַנִי וּבְזִיתַנִי / וְלָמָּה זֶה עֲכַרְתַּנִי / זְדוֹן לִבְּךָ הִשִּׂיאֲךָ לִהְיוֹת עָלַי צֹעֵד / וְאָנֹכִי הַיּוֹדֵעַ וָעֵד / וְלֹא תוּכַל לְהִתְעַלֵּם / עַתָּה גְמוּלְךָ אֲשַׁלֵּם / וַיַּעַן לוֹ הָעַכְבָּר / בְּלֵב מַר מְאֹד וְנִשְׁבָּר / הֲלֹא אָנֹכִי נָבָר וָבָר / אֲחַלֶּה נָא בְּכָל אוֹנִי / יִכָּנְסוּ דְבָרַי בְּאָזְנֵי אֲדֹנִי / וְאַל יִחַר אַפְּךָ בְּעַבְדֶּךָ / וְשַׁכֵּךְ חֲמָתְךָ בְּחַסְדֶּךָ / כִּי אַתָּה יָכֹל לַעֲשׂוֹת מַה שֶּׁלֹּא נִסִּיתָ / וְלֹא תוּכַל לְהָשִׁיב מַה שֶּׁעָשִׂיתָ / וְאִם לִבְּךָ חָם / מוֹדֶה וְעֹזֵב יְרֻחָם / יָדַעְתִּי כִּי הִסְכַּלְתִּי / וְשָׁגִיתִי אֲבָל לֹא זַדְתִּי / כִּי מִי שָׁלַח יָד בַּאדֹנִי וְנִקָּה / וְלֹא הָיָה לוֹ לְפוּקָה / חָלִילָה לְמֶלֶךְ גָּדוֹל וְגִבּוֹר / לִשְׁלֹחַ יָד בְּעַכְבָּר נִבְזֶה לִשְׁבּוֹר / וּבְאָמְרוֹ כָּכָה נִכְמְרוּ עָלָיו רַחֲמָיו / וַיִּכְרֹת לוֹ אֶת בְּרִית שְׁלוֹמָיו / וּבְשִׂפְתֵי רָצוֹן אֵלָיו קָרָא / הִשָּׁמֵר וְהִשָּׁקֵט אַל תִּירָא / וַיְהִי בַחֲצִי הַלַּיְלָה / בְּאִישׁוֹן חֹשֶׁךְ וַאֲפֵלָה / עֵת שַׁאֲגַת הַכְּפִירִים / הָאֲרִי שָׁאַג לִטְרֹף בְּאַחַד הַיְּעָרִים / וַיִּפֹּל בְּרֶשֶׁת מְזוֹרָה / וַיִּזְעַק זְעָקָה גְדוֹלָה וּמָרָה / וְכַאֲשֶׁר הִתְחַזֵּק לָקוּם הָרֶשֶׁת הֶחֱזִיקָתְהוּ / תָּעָה לְבָבוֹ פַּלָּצוּת בִּעֲתָתְהוּ / וַתִּדַּד שְׁנַת הָעַכְבָּר לְקוֹלוֹ / וַיֵּלֶךְ עָדָיו וַיִּמְצָאֵהוּ בַסְּבָךְ מֵרֹאשׁוֹ וְעַד רַגְלוֹ / וַיֹּאמֶר לָאֲרִי: מִי אַתָּה כִּי הֱקִיצוֹתַנִי / וַיַּעַן לוֹ הָאֲרִי: הִנֵּנִי / וַיֹּאמֶר הָעַכְבָּר: הָעָם אֲשֶׁר מָאַסְתָּ בּוֹ / צֵא נָא עַתָּה וְהִלָּחֵם בְּקִרְבּוֹ / אַךְ בָּא הָעֵת וְהִגִּיעַ הַזְּמָן / אֲשֶׁר לְשַׁלֵּם גְּמוּלְךָ אֶהֱיֶה נֶאֱמָן / כַּחֶסֶד אֲשֶׁר עָשִׂיתָ עִמָּדִי / לְהַצִּילְךָ מִפַּח יָקוּשׁ יֵשׁ לְאֵל יָדִי / כִּי אֲקַבֵּץ עִמִּי חֲבֵרִים / לִכְרֹת מִן הָרֶשֶׁת מֵיתָרִים / וּנְנַתְּקָה אֶת מוֹסְרוֹתֶיךָ / אָז תַּצְלִיחַ אֶת אֹרְחוֹתֶיךָ / וַיֹּאמֶר לוֹ הָאֲרִי: אֲדֹנִי מַהֵרָה / טוֹבָה חָכְמָה מִגְּבוּרָה / וְצַדִּיק אַתָּה מִמֶּנִּי / אִם מֵרֶשֶׁת זוֹ תּוֹצִיאֵנִי / וְהָעַכְבָּר אֶת חֲבֵרָיו חִבֵּר / וַיַּעַשׂ לוֹ כַּאֲשֶׁר דִּבֵּר / וַיִכְרְתוּ בְּשִׁנֵּיהֶם הָרְשָׁתוֹת / וַיִּהְיוּ כְרֶגַע כְּרוּתוֹת / וַיֹּאמֶר לוֹ הָאֲרִי: עַתָּה גְמוּלְךָ עָלָי / כִּי הֶעֱמַדְתָּ בַמֶּרְחָב רַגְלָי / וַיִּפֶן הָעַכְבָּר שִׁכְמוֹ לָלֶכֶת לְדַרְכּוֹ / וְהָאֲרִי שָׁב שָׁלֵם אֶל סֻכּוֹ.


וְהַמָּשָׁל בְּאֶבְיוֹנִים וַעֲנִיִּים / אֲשֶׁר מִפֶּשַׁע וּמִמֶּרֶד נְקִיִּים / נֶגֶד עֲשִׁירִים מִשְּׁכֵנוּתָם / אַךְ אִם חָטְאוּ בְשִׁגְגָתָם / בִּבְלִי דָעַת, וְהָעֲשִׁירִים / עַל שִׁגְגָתָם מְכַפְּרִים / בְּבֹא לָעֲשִׁירִים יוֹם אֵידָם / הָעֲנִיִּים בְּמַאֲמַצֵּי יָדָם / בָּהֶם יְדֻבְּקוּ לַעֲזוֹר / כַּאֲשֶׁר יִדְבַּק הָאֵזוֹר / כִּי יֵשׁ אֹהֵב דָּבֵק מֵאָח / וְיֵשׁ אִישׁ אִם אָחִיו תּוֹךְ אָח / לֹא יַצִּיל נַפְשׁוֹ מִשְּׂרֵפָה / וְאִמְרַת הֶחָכָם צְרוּפָה / בְּאָמְרוֹ: אָדָם גָּדוֹל כַּעֲנָק / לִִפְעָמִים בִּזְבוּב קָטָן נֶחֱנָק / וְשָׁאֲלוּ לֶחָכָם, אֲשֶׁר חָפֵץ בִּיקָרוֹ / וּמִי אֹהֵב יוֹתֵר, אָחִיו אוֹ חֲבֵרוֹ / וַיַּעַן לָהֶם: שִׁמְעוּ נָא דְבָרִי / אֵינֶנִּי אֹהֵב אָחִי עַד יִהְיֶה חֲבֵרִי.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


מֵטִיב לְךָ טוּבוֹ גְּמֹל / אִם יִשְׁגֶּה בָרָע חֲמֹל

כִּי יֵשׁ בְּעֵת צָר יַעֲזוֹר / מָחָר וְאִם בָּזוּי תְּמֹל

לִבָּךְ בְּךָ הַשְׁפֵּל וּמֹל / לַתָּם וְרֹאשׁ עִקֵּשׁ קְמֹל.
