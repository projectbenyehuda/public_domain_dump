משל קיא: שְׂמָמִית וּמֶלֶךְ וְעֶבֶד

אֲשֶׁר הָעֲשִׁירִים עֲלֵיהֶם לֹא חָמְלוּ עָלוּ לְמַעְלָה / וְהֵמָּה יָרְדוּ מַטָּה שְׁאוֹלָה

שְׂמָמִית הִיא בְּהֵיכְלֵי מֶלֶךְ / יָדָהּ תָּמְכָה פָלֶךְ / לִטְווֹת וְלַעֲשׂוֹת מַטְוֶה / אֶל עֵמֶק שָׁוֶה / וּבְכָל קַרְנֵי הַכְּתָלִים / מִן הַבָּתִים וּמִן הָאֹהָלִים / וַיִּבֶן הַבַּיִת מֵאַבְנֵי אֶלְגָּבִישׁ / בֵּית הַמֶּלֶךְ תּוֹכוֹ עָשְׂתָה בֵית עַכָּבִישׁ / עַד שֶׁהַמֶּלֶךְ בִּמְסִבּוֹ / וַחֲמַת הַמֶּלֶךְ בָּעֲרָה בוֹ / וְהָעֶבֶד נִקְרַב כְּדַרְכּוֹ / לְכַבֵּד הַבַּיִת בְּרָחְבּוֹ וְאָרְכּוֹ / וַיַּבֵּט אֶת הָעַכָּבִישׁ בַּקְּרָנוֹת / וַיַּעַל אַרְבַּע פִּנּוֹת / מִזְרָחִית צְפוֹנִית מַעֲרָבִית דְּרוֹמִית / וַיֹּאמֶר אֶל הַשְּׂמָמִית / לֹא נוּכַל לָשֶׁבֶת יַחְדָּו / אֶת הַמֶּלֶךְ עִם עֲבָדָיו / וְטֵאטֵאתִיךְ בְּמַטְאֲטֵא הַשְׁמֵד / וַתֵּלֶךְ הִיא אֶל הַיַּיִן וְאֶל הַמַּיִם וְאֶל הַתֶּמֶד / לְהַטִּיל אַרְסָהּ בַּמַּשְׁקֶה / וְהָאִישׁ נַקֵּה לֹא יִנָּקֶה / לַצְבּוֹת בִּטְנוֹ וְלַנְפִּל יְרֵכוֹ / וְהָעֶבֶד יֵלֵךְ הָלֹךְ וּבָכֹה / לְהַגִּיד לַאדֹנָיו הַדְּבָרִים / וַיִּהְיוּ הַמַּיִם הַמְאָרְרִים לְמָרִים / וַיְצַו הַמֶּלֶךְ לִשְׁפֹּךְ לַחוּץ הַמַּשְׁקִים / בָּרְחוֹבוֹת וּבַשְּׁוָקִים / וַיֹּאמֶר הָעֶבֶד: מַר מִמָּוֶת אֲנִי מוֹצֵא / וַיְבֻקַּשׁ הַדָּבָר וַיִּמָּצֵא / כַּאֲשֶׁר בָּאוּ מִן הַיְּעָרִים / מִן הַשָּׂדֶה הָעִזִּים וְהַחֲזִירִים / יִשְׁתּוּ יֵין נִסְכֵּיהֶם / וְצָבְתוּ בִטְנֵיהֶם וְנָפְלוּ יַרְכוֹתֵיהֶם / מִן הַמַּשְׁקִים אֲשֶׁר שָׁפְכוּ / שָׁתוּ וְלָעוּ וְכַפִּשְׁתָּה דָעֲכוּ / וְהָאֲנָשִׁים הִרְגִּישׁוּ בַדָּבָר / אֶת אֲשֶׁר נַעֲשָׂה כְבָר / כִּי בִזְבוּב קָטָן נֶחֱנָק / אָדָם גָּדוֹל כַּעֲנָק / לִשְׁתּוֹת כּוֹס הַתַּרְעֵלָה / בְּלִי רְפוּאָה אֶרֶךְ וּתְעָלָה / וְכַאֲשֶׁר רָאָה הַמֶּלֶךְ הַנֵּס בְּלִבּוֹ עָלַץ / כִּי מִצָּרָה גְדוֹלָה נֶחֱלָץ / מִמִּיתָה מְשֻׁנָּה לְחַיִּים / וְהָיוּ חַיָּיו מִנֶּגֶד תְּלוּיִים / וַיַּעַשׂ מִשְׁתֶּה לְכָל אֹהֲבָיו / וּלְאַנְשֵׁי בֵיתוֹ וְלִקְרוֹבָיו / וַיֹּאמַר, כִּי טוֹב לִי עַתָּה / לְהַעֲבִיר הַשְּׂמָמִית מֵרָעָתָהּ / וַיָּבֹא אֵלֶיהָ הָאֹהֱלָה / וַיְצַו אֶל עֲבָדָיו לְהַגִּיד לָהּ / לָבוֹא בִּדְבַר הַמֶּלֶךְ / וַיֹּאמַר: שַׂמְתִּי שָׁלוֹם בִּגְבוּלֵךְ / כִּי בָךְ מָרֹה מָרִיתִי / מַרְבַדִּים עַרְשִׂי רָבַדְתִּי / לוּלֵא יָגֹרְתִּי חָרְבֹתַיִךְ וְשֹׁמְמוֹתַיִךְ / הַרְחִיבִי מְקוֹם אֹהָלַיִךְ וִירִיעוֹת מִשְׁכְּנוֹתַיִךְ / וּבֹאִי לִינִי בַאֲשֶׁר אָלִין / מַה יָּפוּ פְעָמַיִךְ בַּנְּעָלִים / וְהִנֵּה בֵּית זְבוּל לָךְ בָּנִיתִי / תַּחַת אֲשֶׁר לָךְ הֲרֵעוֹתִי / וְלֹא תוֹסִיפִי לְדַאֲבָה עוֹד עוֹלָמִית / וְלֹא תַעֲלֶה בְלִבִּי עוֹד עָלַיִךְ רָעָה וְתַרְמִית / כִּי הֲבִיאֹתַנִי עַד הֲלוֹם / וְשַׂמְתִּי פְקֻדָּתֵךְ שָׁלוֹם / וְאִם אַתְּ עֹשָׂה לִי כָּכָה / וַחֲמָתֵךְ מֶנִּי שָׁכָכָה / וַתַּעַן הַשְּׂמָמִית קָשׁוֹת / עֵת לְדַבֵּר וְעֵת לַחֲשׁוֹת / גֵּרְשׁוּנִי מֵהִסְתַּפֵּחַ בְּנַחֲלָתֶךָ / הִנְנִי מַשְׁלִיחַ בְּךָ וּבְבָתֶיךָ / עָרֹב כָּבֵד מְאֹד / וּמִפָּנַי לֹא תוּכַל לַעֲמֹד / וְלֹא תוּכַל לְהֵרָפֵא / כִּי מִמָּוֶת תִּהְיֶה נִסְפֶּה / וַתֵּלֶךְ וַתִּצְבֹּר כָּל הַשְּׂמָמִיּוֹת / מִן הַבָּתִים וּמִן הַגֵּאָיוֹת / וַתִּצָּבַרְנָה כֻלָּנָה יַחְדָּו / לָבֹא אֶל הַמֶּלֶךְ וְאֶל עֲבָדָיו / וַתִּגַּשְׁנָה לְאַבְּדָם וּלְהֻמָּם / הָאֲנָשִׁים עַד תֻמָּם / וַתַּעֲשֶׂינָה בָהֶם אֹרְבִים וְנֹגְשִׂים / בִּנְפֹל תַּרְדֵּמָה עַל אֲנָשִׁים / וַיִּפְּלוּ וַיָּמוּתוּ לָרֹב מֵהֶם / וְלֹא נִשְׁאַר עַד אֶחָד בָּהֶם.

וְהַמָּשָׁל בְּאֶבְיוֹנִים וְדַלִּים / וְהָעֲשִׁירִים עֲלֵיהֶם לֹא חֹמְלִים / וַאֲחֵרִים הֱסִיתוּם לְהָרַע לָהֶם / כַּאֲשֶׁר הָיוּ מִקֶּדֶם אֹיְבֵיהֶם / וְהָעֲנִיִּים יָבִינוּ לְאַחֲרִיתָם / וְיַעֲשׂוּ חַרְצֻבּוֹת לְמוֹתָם / וְלִנְקֹם נִקְמָתָם יַשְׂכִּילוּ / לְשַׁלֵּם לָהֶם מַעֲלָם אֲשֶׁר מָעֲלוּ / וְהָיוּ לְסִכִּים בְּעֵינֵיהֶם / וְלִצְנִינִים בְּצִדֵּיהֶם / עַד אֲשֶׁר בָּאוּ וְכָרְעוּ וְנָפְלוּ לְרַגְלֵיהֶם / לַעֲבֹר עַל פִּשְׁעָם וְלִמְחֹל חַטָּאתָם / וַיִּכָּנְעוּ וַיָּשׁוּבוּ עַל עֵקֶב בָּשְׁתָּם / וַיֵּרְדוּ מַטָּה מַטָּה שְׁאוֹלָה / וְהָעֲנִיִּים עָלוּ מַעְלָה מָעְלָה / וְאָמַר שְׁלֹמֹה בְחָכְמָתוֹ / יֵשׁ מִתְרוֹשֵׁשׁ וְהוֹן רָב מֵסִיר אָזְנוֹ מִשְּׁמֹעַ דַּל בְּזַעֲקָתוֹ.
