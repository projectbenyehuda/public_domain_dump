ספר תחכמוני - שַׁעַר אַרְבָּעִים וְאֶחָד

מַחֲלֹקֶת הָאִישׁ וְהָאִשָּׁה. מִי הוּא יוֹתֵר קִיּוּם הָעוֹלָם וְאִשָּׁהּ


נְאֻם הֵימָן הָאֶזְרָחִי: נִתְחַבַּרְתִּי עִם אֲנָשִׁים סוֹחֲרִים. מֵעִיר אֶל עִיר עוֹבְרִים. נוֹשְׂאִים מִינֵי סְחוֹרוֹת. בְּשָׂמִים וְרִקְמָה וַאֲבָנִים יְקָרוֹת. וּבְעוֹדֵנוּ הוֹלְכִים בַּדֶּרֶךְ אֶל אַחַת הֶעָרִים. אֲשֶׁר בַּכְּפָרִים. רָאִיתִי אֲנָשִׁים נֶאֱסָפִים. וּמִכָּל גִּבְעָה נִשְׁקָפִים. וּמְשָׁכַתְנִי תַאֲוִתי. לַעֲזֹב חֶבְרָתִי. וְלָרוּץ עִמָּהֶם. אֶל מְקוֹם קִבּוּץ הֲמוֹנֵיהֶם. עַד בֹּאֵנוּ אֶל תּוֹךְ רְחָבָה. וְשָׁם הָמוֹן רַבָּה. וְהִנֵּה אִישׁ זָקֵן עוֹרֵךְ מְרִיבָה. עִם אִשָּׁה עַזָּה וְקָשָׁה. לְבוּשׁ רֶשַׁע לְבוּשָׁהּ. וּפָנֶיהָ כַנְּחוּשָׁה.

וַיֹּאמֶר לָהּ הָאִישׁ: תְּהִלָּה לָאֵל הַמַּגְדִּילֵנוּ בֵּין יְרֵאָיו. וְהַמַּמְשִׁילֵנוּ עַל בְּרוּאָיו. וַיְשִׂימֵנוּ רֵאשִׁית קְרוּאָיו. וְהוֹצִיא מִמֶּנּוּ גֶזַע נְבִיאָיו. וְשָׂם יִתְרוֹן עַל הַנָּשִׁים לָאֲנָשִׁים. בַּעֲבוּר כִּי הֵם בִּקְדֻשָּׁתוֹ נִקְדָּשִׁים. וּמָנַע מִצְוֹתָיו מִן הַנָּשִׁים. וְלֹא חִיֵּב עֲלֵיהֶם תּוֹרָה וּמִצְוָה. מִפְּנֵי שֶׁהֵם עֶרְוָה. וְצִוָּה עֲלֵיהֶם לְהִסָּתֵר בְּבָתֵּיהֶם. לְמַעַן בּשֶׁת פְּנֵיהֶם. לְהוֹדִיעַ כִּי הָעוֹלָם וְכֹל אֲשֶׁר בּוֹ לִכְבוֹד הָאִישׁ נִבְרָא. וְעַל שְׁמוֹ נִקְרָא. גַּם הָאִשָּׁה לוּלֵי הָאִישׁ לֹא נִבְרְאָה. וּבְשֵׁם לֹא נִקְרְאָה.

אָמְרָה הָאִשָּׁה: הוֹי גּוֹי חוֹטֵא. כְּמַדְקְרוֹת חֶרֶב בּוֹטֶה. וְאַחַר הַשֶׁקֶר נוֹטֶה. בְּמַחֲסֵה שָׁוְא נִסְתַּרְתָּ. וּבְמִבְצַר כָּזָב נִבְצַרְתָּ. וְצֶדֶק לֹא דִבַּרְתָּ. וְיֵשׁ לְךָ לָדַעַת כִּי לוּלֵי הַנָּשִׁים. לֹא נִבְרְאוּ הָאֲנָשִׁים. כְּי אֲנַחְנוּ שֹׁרֶשׁ וְאַתָּם עָנָף וְאֵיךְ יִצְמְחוּ עֲנָפִים בְּלִי שָׁרָשִׁים. אֲבָל אַתֶּם עַזֵּי מֵצַח וְקָשִׁים. וְאֵינְכֶם מַכִּירִים הָאֱמֶת. וְהַנְּכוֹחָה מִכֶּם נֶעְלֶמֶת. וְאִלּוּ בְצֶדֶק שְׁפַטְתֶּם. בְּמַעֲלַת הַנָּשִׁים נִזְהַרְתֶּם. וְאוֹתָם עֲבַדְתֶּם. וְאִם יֵשׁ לָכֶם עָלֵינוּ יִתְרוֹן גַּם לָנוּ יִתְרוֹן עֲלֵיכֶם. כִּי צָרְכְּכֶם אֵלֵינוּ כְּצָרְכֵּינוּ אֲלֵיכֶם.

אָמַר הָאִישׁ: שִׂימִי יָד לְפֶה בְּזוּיָה וּמְאוּסָה. כְּדֹמֶן בְּאַשְׁפָּה רְמוּסָה. וּמִכָּל חֶרְפָּה עֲמוּסָה. וּמְלֵאָה בוּז וְקַלָּסָה. הֲלַזְּכָרִים תַּמְשִׁילִי עַצְמֵךְ. וְלֹא תַכִּירִי מוּמֵךְ. כִּי הָאִשָּׁה לֹא נִבְרְאָה רַק לְשָרֵת הָאִישׁ כְּאָמָה. וְלִלְחֹךְ עֲפָרוֹ כְּתוֹלֵעָה וְרִמָּה. וְלֹא נִגְזְרָה מִצַּלְעוֹתָיו. רַק לְמַעַן תֵּדַע כִּי הִיא עֶצֶם מֵעַצְמוֹתָיו. וְאֵבֶר מִתּוֹלְדוֹתָיו. וְשִׁפְחָה מִשִּׁפְחוֹתָיו.

אָמְרָה הָאִשָּׁה: הֲלֹא תֵבוֹשׁ כְּסִיל וָבַעַר. כִּי פִיךָ עָלַי תִּפְעַר. וְכֻלְּךָ מָלֵא חֶלְאָה. וּמְקוֹר הַצּוֹאָה. וְשֹׁרֶשׁ הַטֻּמְאָה. וּבָדְבָרֶיךָ הֶרְאֵיתָ מוֹרְאוֹתֶיךָ. וְגִלִּיתָ רֹעַ מִדּוֹתֶיךָ. וְהוֹצֵאתָ מִפִּיךָ חֶלְאָתְךָ. וְעַתָּה שׁוּב וְכִסִּיתָ אֶת צֵאָתֶךָ. וְהַסְתֵּר דִּבָּתֶךָ. הֲלֹא יָדַעְתָּ כִּי אִם אַתֶּם נִכְבָּדִים בָּנוּ כָּל כְבוֹדְכֶם. וַאֲנַחְנוּ מְנוּחַתְכֶם בְּעוֹדְכֶם. אֲבָל אַתֶּם בְּעֵת צָרְכְּכֶם אֵלֵינוּ. תִּשְׁתַּחֲווּ אֶל כַּפוֹת רַגְלֵינוּ. וּבְעֵת תַּשִּׂיגוּ חֶפְצְכֶם תִּתְגָּאוּ עָלֵינוּ. וְתִרְצוּ לְהַשְׁפִּילֵנוּ. כֵּן דֶּרֶךְ בְּנֵי בְלִיַּעַל כִּי הֵם מִתְעַדְּנִים בְּטוֹבַת הָאֵל וְלֹא יוֹדוּהוּ עָלֶיהֶ. וְלֹא יָשִׂימוּ לֵב אֵלֶיהָ. וּבַמֶּרְחַב יִתְגָּאוּ עַל הָאֵל וְיֹאמְרוּ מִי יַכְנִיעֵנוּ. וּבְעֵת רָעָתָם יֹאמְרוּ קוּמָה וְהוֹשִׁיֵענוּ.

אָמַר הָאִישׁ: אִם הָאִשָּׁה יְסוֹד הָאִישׁ. הוּא יְסוֹד מַחְפִּיר וּמַבְאִישׁ. כֵּן מוֹצָא הַפְּנִינִים הָרָמִים. מִמַּעֲמַקֵּי תְהוֹמִים. וּמִמְצוּלוֹת הַיֶּמִּים. וְכֵן הָאֲדָמָה הִיא יְסוֹד הַנִּבְרָאִים. וְעַל שְׁמָהּ בְּנֵי אִישׁ נִקְרָאִים. וּמִמֶּנָּהּ יָצְאוּ הַנְּבִיאִים. וּמֵעֲפָרָהּ נִבְרְאוּ הַצַּדִּיקִים וְהַיְרֵאִים. וְאֵין רָאוּי לוֹמַר בַּעֲבוּר כִּי הֶעָפָר יְסוֹדָם. כִּי הֶעָפָר טוֹב מִן הָאָדָם.

אָמְרָה הָאִשָּׁה: הֲלֹא יָדַעְתָּ כִּי בָנוּ תִשְׁלַם שִׂמְחַתְכֶם. וְזוּלָתֵנוּ תִּגְדַּל אַנְחַתְכֶם. וּבַמֶּה תִתְנַשְּׂאוּ עָלֵינוּ. וְהַצֹּרֶךְ יַכְנִיעֲכֶם לְרַגְלֵינוּ. וְאִם עִמָּנוּ תִתְנַהֲגוּ בְגַאֲוָה. תַּכְנִיעֲכֶם אֵלֵינוּ הַתַּאֲוָה.

אָמַר הָאִישׁ: תַּאֲוַת הָאִישׁ אֶל הָאִשָּׁה הִיא תַאֲוָה רָעָה. וְצֹרֶךְ הַשָּׁעָה. כְּתַאֲוַת הַחוֹלֶה אֶל הַבָּשָׂר. אֲשֶׁר בּוֹ חָלְיוֹ יוֹסִיף וְלֹא יֶחְסָר. וְהַחוֹלֶה יוֹדֵעַ כִּי בַאֲכִילָתוֹ. תִּגְדַּל מַחֲלָתוֹ. וְתִכָּפֵל רָעָתוֹ. וְלֹא יוּכַל לְהִמָּנַע מִמֶּנּוּ. כִּי הַתַּאֲוָה אֵלָיו תִּמְשְׁכֶנּוּ. וְכֵן הָאִשָּׁה הִיא סִבָּה לְנֵזֶק הָאִישׁ. וּכְצִפְעוֹנִי תַפְרִישׁ. וְלֹא יוּכַל לְהִמָּנַע מִמֶּנָּה. כִּי הַתַּאֲוָה אֵלָיו תִּמְשְׁכֶנָּה. וְהוּא יַכִּירֶנָּה. כִּי בְעֵת יִקַּח אִישׁ אִשָּׁה. נָפַל בִּמְצוּדָה רָעָה וְקָשָׁה. כִּי בִגְלָלָהּ יָבִיא נַפְשׁוֹ בְצָרוֹת. וְיִשְׂבַּע מְרוֹרוֹת. כְּדֵי לְהַלְבִּישָׁהּ וּלְהַשְׁקוֹתָהּ. וְלֹא יִגְרַע שְׁאֵרָהּ כְּסוּתָהּ וְעוֹנָתָהּ. וּמִי הִצְרִיכָנוּ לְזֹאת הַצָּרָה הַמָּרָה. וְהַתְּלָאָה הַנּוֹרָאָה. הֲלֹא טוֹב לָאָדָם הַקֶּבֶר. מִזֶּה הַיָּגוֹן וְהַשֶּׁבֶר. וְטוֹב לָשֶׁבֶת בְּאֶרֶץ מִדְבָּר. מֵאֵשֶׁת מִדְיָנִים וּבֵית חֶבֶר.

אָמְרָה הָאִשָּׁה: יַכְרַת אֱלֹהִים נוֹרָא עֲלִילוֹת. לָשׁוֹן מְדַבֶּרֶת גְּדוֹלוֹת. וְשָׂפָה מְעוֹלֶלֶת עֲלִילוֹת. כִּי לְשׁוֹנְךָ מִנִּי יָם אָרוּךְ. כְּחֶרֶב עָרוּךְ. לְהַגְבִּיר הַשֶּׁקֶר עַל הָאֱמוּנָה. וּלְהָרִים יַד הָעָוֶל עַל הַנְּכוֹנָה. וְאִם הִאיׁש בְּמַחֲזֵה הַנְּבוּאָה יִתְגַּבֵּר. וְעַל אֲשֶׁר דִּבֶּר הָאֵל עִמּוֹ גְדוֹלוֹת יְדַבֵּר. הֲלֹא גַם בָּנוּ דִבֵּר. וּמִמֶּנּוּ הָיוּ כַמָּה נְבִיאוֹת. עוֹמְדוֹת בְּמַרְאוֹת הַצּוֹבְאוֹת. וְצָלְחָה עֲלֵיהֶם רוּחַ נְבוּאוֹת. וְהָיוּ אֶל עָל קְרוּאוֹת.

אָמַר הָאִישׁ: נְבוּאַת הַנָּשִׁים. הִיא לְרֹב עֲוֹנוֹת הָאֲנָשִׁים. כִּי בִהְיוֹת הָאֲנָשִׁים בַּעֲלֵי חֲסִידוּת וּקְדֻשָּׁה. לא צַלְחָה רוּחַ נְבוּאָה עַל אִשָּׁה. רַק בְּעֵת יַרְבּוּ רִשְׁעָתָם. תִּהְיֶה לְיַד אִשָּׁה תְשׁוּקָתָם. כְּמוֹ שֶׁהִתְנַבְּאָה דְבוֹרָה. בְּעֵת הֱיוֹת יִשְׂרָאֵל עוֹבְדִים עֲבוֹדָה זָרָה. וּכְמוֹ שֶׁהִתְנַבְּאָה חֻלְדָּה. כְּשֶׁעָצְמוּ עֲוֹנוֹת יִשְׂרָאֵל בְּלִי מִדָּה. וְהָאִשָּׁה אִלוּ נָתְנָה בַשַּׁחַק מְעוֹנֶיהָ. וְהִשְׁתַּחֲווּ אוֹרִים לְפָנֶיהָ. הָאִישׁ לְשִׁפְחָה יִקְנֶהָ. וְיִשְׁכַּב אוֹתָהּ וִיעַנֶּהָ. וְהָאִשָּׁה וִילָדֶיהָ תִּהְיֶה לַאדֹנֶיהָ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בָּרוּךְ אֲשֶׁר חָלַק לְאִישׁ כָּבוֹד

וַיְשָׁו פְּאֵר לְאִישׁ עֲלֵי אִשָּׁה.

וַיִּתְּנֵהוּ רֹאשׁ וְאָדוֹן לָהּ

אַךְ הִיא חֲרוּפָה לוֹ וְנִכְבָּשָׁה.

נַחְנוּ אֱמֶת זֶרַע קְדוֹשׁ אֵל חַי

הַלּוֹקְחִים דָּתוֹ לְמוֹרָשָׁה.

גַּם הַנְּבִיאִים נָבְרְאוּ מֶנוּ

הַהוֹפְכִים הַיָּם לְיַבָּשָׁה.

כִּי אֵין דְּבַר ישֶׁר בְּנָשִׁים רַק

שֶׁהֵם לְאִישׁ עֶרְוָה וְגַם בּוּשָׁה.


אָמַר הַמַּגִיד: וְכַאֲשֶׁר הִשְׁלִימוּ אֵיבָתָם. וּמְרִיבָתָם. שָׂמְתִּי אֶל הָאִישׁ לִבִּי. וּפָנִיתִי אֵלָיו בְּאִישׁוֹן מַחֲשָׁבִי. וָאֵרֶא וְהִנֵּה הוּא חֶבֶר הַקֵּינִי אֲבִי הַתַּחְבֻּלוֹת הַגְּדוֹלוֹת. וראשׁ הָעָרְמוֹת וְהַנְּבָלוֹת. וְיָדַעְתִּי כִּי כָל דְּבָרָיו עָרְמָה. וְכָל מַעֲשָׂיו מִרְמָה. אַךְ הָיִיתִי נֶחְפָּז לָלֶכֶת. וְנַפְשִׁי לַחֲבֵרַי נִמְשֶׁכֶת. וְעַל כֵּן לֹא נוֹדַעְתִּי לוֹ וַעֲזַבְתִּיהוּ. וְעַל דְּבַר הָאִשָּׁה לֹא שְׁאִלְתִּיהוּ.
