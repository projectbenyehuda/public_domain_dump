ספר תחכמוני - שַׁעַר שְׁלשִׁים וְשִׁשָּׁה

בִּשְׁאֵלוֹת וּתְשׁוּבוֹת. זֶה שׁוֹאֵל וְזֶה מֵשִׁיב בְּמִלִּים עֲרֵבוֹת


נְאֻם הֵימָן הָאֶזְרָחִי: שָׁאֲפָה רוּחִי. בִּימֵי תַעֲנוּגִי וְחָרְפִּי. לְהָאִיר בְּאוֹר הַשְׂמָחוֹת נַפְשִׁי וְלָתֵת אֵצֶל סִפָּם סִפִּי. וָאֶתְּנָה לִבִּי לְשׁוֹטֵט בְּתֵבֵל. וְלָקַחַת הַשְּׂמָחוֹת לְחֶבֶל. וּלְהִתְעַלֵּס בְּכִנּוֹר וָנֵבֶל. עַל עֲסִיסֵי נֵבֶל. וַיְהִי הַיּוֹם וַאֲנִי בְאֶרֶץ סְבָאִים. וְעָבַרְתִּי עַל בֵּית הַסּוֹבְאִים. וְשָׁם בַּחוּרִים נִצְבָּאִים. וְעַל בֵּית הַגְּפָנִים צוֹבְאִים. וּבְתוֹכָם חֶבֶר הַקֵּינִי. וְכוֹס יַיִן כְּסַהַר לְנֶגְדּוֹ. וְעֹפֶר כַּשֶּׁמֶשׁ לְצִדּוֹ. וְהוּא בֵינֵיהֶם. מִתְעַדֵּן בִּשְׁנֵיהֶם. וְכַאֲשֶׁר רָאַנִי מִמֶּרְחָק. נִכְלַם מִפָּנַי וְשָׂחָק. וַיֹּאמֶר לִי: קְרַב וְאַל תִּרְחָק. וָאֵשֵׁב אֶצְלוֹ. וָאֹמַר לוֹ: הֲלָעַד תִּרְדֹף הַתַּאֳווֹת וְאוֹתָם תְּבַקֵּשׁ. וְעַד מָתַי יִהְיֶה זֶה לְךָ לְמוֹקֵשׁ. וַיִּכְעַס לְנִיבִי. וְהֵסֵב פָּנָיו מִשְּׁמֹעַ רִיבִי. וַיֹּאמֶר לִי: אֵין זֶה יוֹם תּוֹכָחָה. רַק יוֹם תַּעֲנוּג וְשִמְחָה. וְעַתָּה הָסֵר חַטָּאתִי מִנֶּגֶד עֵינֶיךָ. וְהַטֵּה לְחִידָתִי אָזְנֶיךָ. אָמַרְתִּי: הִנְנִי שׁוֹמֵעַ אִמְרָתְךָ וְהֶגְיוֹנֶךָ. וַיֹּאמֶר לִי: הָיִיתִי בִּימֵי נְעוּרִים בְּאֶרֶץ כִּתִּים. וְעָלֶיהָ אֲרָם נְחִתִּים. וְעַל גְּדוּדֵי אֶרֶץ הַחִתִּים. עִם מַעַרְכוֹת פְּלִשְׁתִּים. וְעָמַדְתִּי יָמִים בְּתוֹךְ הָעִיר בַּמָּצוֹר. עַד קָצְרָה נַפְשִׁי לְרֹב הַמַּחְסוֹר וְהַמַּעְצוֹר. וּבִרְאוֹתִי כִּי אָזְלַת יָדִי. וְכִמְעַט מִן הָרָעָב נִשְׁפַּךְ לָאָרֶץ כְּבֵדִי. וְכִי אֲנִי בְתוֹךְ הָעִיר לָחוּץ. נוֹעַצְתִּי עֵצָה וְדִלַּגְתִּי בַלַּיְלָה מִן הַחוֹמָה לַחוּץ.

וְעָרְכִּתי לָאֵל תּוֹדָה. כִּי הִצִּילַנִי מִן הַקְּפָדָה. וְיָשַׁנְתִּי בַּמַּחֲנֶה עַד הַבֹּקֶר. וְכַאֲשֶׁר הִשְׁכַּמְתִּי. וּמִשְּׁנָתִי קַמְתִּי. וּבְנֵי אָדָם יָצְאוּ מֵאָהֳלֵיהֶם הֲמוֹנִים הֲמוֹנִים. גְּדוֹלִים וּקְטַנִּים. עָלִיתִי עַל רֹאשׁ גִּבְעָה רָמָה. וְאָמַרְתִּי אַיֵּה אוֹהֲבֵי הַחָכְמָה. יָבוֹאוּ וְיֶחֱזוּ פְלָאוֹת. וְיִשְׁמְעוּ נוֹרָאוֹת. וַיִּקָּבְצוּ אֵלַי קָהָל גָּדוֹל. כֶּעָפָר וְכַחוֹל. וָאַעַן וָאֹמַר: אַנְשֵׁי לֵבָב שִׁמְעוּ לִי. וְהַקְשִׁיבוּ מְלִיצוֹת שִׂכְלִי. אָנֹכִי הַבָּא מִמַּעֲרֶכֶת הַזְּמַנִּים סָבַבְתִּי אַפְסֵי עוֹלָם. וְחָקַרְתִּי כָּל סוֹד נֶעְלָם. וּבָחַנְתִּי כָל בְּנֵי אָדָם. אֲשֶׁר בֶעָפָר יְסוֹדָם. וְהֵבַנְתִּי כָל חָכְמָה סְתוּמָה. וַתְּהִי טַבַּעְתִּי חֲתוּמָה. כִּי בְּסוֹד הָרְפוּאוֹת. אֲגַלֶּה נִפְלָאוֹת. וּבְחָכְמַת הַמִּדּוֹת. בָּנִיתִי בֵית מִדּוֹת. וְחָכְמַת הַחֶשְׁבּוֹן. אֶחְשֹׁב לְעֵץ רִקָּבוֹן. וּבִמְרוֹם הַהִגָּיוֹן. בָּנִיתִי לִי אַפִּרְיוֹן. וּבְדִינֵי הַכּוֹכָבִים. אָשִׁיב הָעֲנָקִים חֲגָבִים. וּבְטַעֲמֵי הַתּוֹרוֹת. אֲגַלֶּה נִסְתָּרוֹת. וּלְאֹהֶל הַתַּלְמוּד. שִׂכְלִי כְעַמּוּד. וְרֵיחַ הַמִּשְׁנָה. לֹא נָמַר בִּי וְלֹא שָׁנָה. וְהַגְּמָרָא וְהַסְּבָרָה. בִּלְשׁוֹנִי סְדוּרָה. וּבְסִפְרָא וְסִפְרֵי. מִי כָמוֹנִי מוֹרֶה. וּבְפֵרוּשִׁים. אֲגַלֶּה חִדּוּשִׁים. וְסוֹד הַדִּקְדּוּק. עַל לִבְּי חָקוּק. וּבִמְלִיצוָת לְשׁוֹן עֲרָב. מִי זֶה לָגֶשֶׁת אֵלַי לִבּוֹ עָרָב. וּבִמְלֶאכֶת הַשִּׁירִים. אֲחַבֵּר בְּדוֹלָחִים וְדָרִים.

וַיִּגַּשׁ אֵלַי בָּחוּר מַשְׂכִּיל. וְטוּב זִכְרוֹ עָלָיו הוֹלֵךְ רָכִיל. וַיֹּאמֶר לִי: עַתָּה יִבָּחֲנוּ מְלִיצוֹתֶיךָ, בִּשְׁאֵלוֹתַי וּבִתְשׁוּבוֹתֶיךָ.אָמַרְתִּי: הִנְנִי עוֹמֵד לְפָנֶיךָ. קְרָא אֵלַי וְאֶעֱנֶךָ. אָמַר לִי: בַּמֶּה תִהְיֶה חֶבְרָה נְעִימָה וַעֲרֵבָה. אָמַרְתִּי: בְּאַהֲבָה. אָמַר: מִי הוּא הַמַּשְׂכִּיל בֵּין עַמּוֹ. אָמַרְתִּי: הַמַּכִּיר מוּמוֹ. אָמַר: מָה הַמִּדָּה הַחֲשׁוּקָה. אָמַרְתִּי: הַשְּׁתִיקָה. אָמַר: בַּמֶּה יִתְיַפֶּה הָעשִׁר. אָמַרְתִּי: בְּישֶׁר. אָמַר: מַה הַשֵּׂכֶל וּגְבוּלוֹ. אָמַרְתִּי: לָשׂוּם כָּל דָּבָר בְּמקוֹמוֹ הָרָאוּי לוֹ. אָמַר: מָתַי הַתְּפִלָּה נִשְׁמַעַת. אָמַרְתִּי: בְּעֵת הַנֶּפֶשׁ נִכְנַעַת. וְהָעַיִן דּוֹמַעַת. וְהַתְּלָאָה לְקִירוֹת הַלֵּב נוֹגַעַת. אָמַר: מַה יְעַוֵּר עֵינֵי הַשֵׂכֶל. אָמַרְתִּי: תַּאֲוַת הָאֹכֶל. אָמַר: מָתַי נִבְחָר מָוֶת מֵחַיִּים. אָמַרְתִּי: כְּשֶׁיִּתְגָּאוּ עַל הַנִּכְבָּדִים בְּזוּיִים. אָמַר: מָה הַמַּחֲלָה הַמְיַגַּעַת. אָמַרְתִּי: חֶסְרוֹן הַדַּעַת. אָמַר: מָתַי יֹאבְדוּ הַקְּהִלּוֹת. אָמַרְתִּי: כְּשֶהֵם חֲבִילוֹת חֲבִילוֹת. אָמַר: כַּמָּה הֵם מִדּוֹת הָרִשְׁעוּת. אָמַרְתִּי: סִכְלוּת וְגַאֲוָה וּשְׁטוּת וְעַזּוּת. וּבְאַחַת מֵהֶם יֶחֱרַב הָעוֹלָם. כָּל שֶׁכֵּן בְּהִתְקַבְּצָם כֻּלָּם. אָמַר: מִי הוּא הַגִּבּוֹר בִּבְנֵי דוֹרוֹ. אָמַרְתִּי: הַיָּרֵא לְצוּרוֹ. אָמַר: מִי הוּא הַמֵּבִיא עַצְמוֹ בְסַכָּנָה. אָמַרְתִּי: כָּל מְשָׁרֵת הַמְּלָכִים בְּלֹא אֱמוּנָה. אָמַר: מִי הַמַּשִּׁיג מִן הָעוֹלָם סִפְקוֹ. אָמַרְתִּי: הַשָּׂמֵחַ בְּחֶלְקוֹ. אָמַר: מִיִ יָבִיא לָאָדָם לְבוּשׁ קָלוֹן לִלְבּשׁ. אָמַרְתִּי: שְׂחוֹק וְקַלּוּת רֹאשׁ. אָמַר: מִי הָאָדוֹן בִּבְנֵי אַנְשֵׁי סוֹדוֹ. אָמַרְתִּי: הַמּוֹחֵל עַל כְּבוֹדוֹ. אָמַר: מִי קְרוֹבָה מַפַּלְתּוֹ. אָמַרְתִּי: הַמִּתְרַפֶּה בִמְלַאכְתּוֹ. אָמַר: מִי הַיַּחַס הַטּוֹב. אָמַרְתִּי: הַשֵּׁם הַטּוֹב. אָמַר: מִי הַנָּדִיב מִכָּל מִשְׁפַּחְתּוֹ. אָמַרְתִּי: הַמִּתְנַדֵּב כְּפִי יְכָלְתּוֹ. וְנוֹתֵן מְעֶט מִן הַמְעַט אֲשֶׁר אִתּוֹ. אָמַר: מִי הַכְּסִיל אֲשֶׁר אֵין כָּמוֹהוּ. אָמַרְתִּי: הָרוֹאֶה מִיתַת אָחִיהוּ וְאָבִיהוּ. וְלֹא יִפְחַד פֶּן יָמוּת גַּם הוּא. אָמַר: מָתַי יִיקַר הָאִישׁ הַנִּקְלֶה. אָמַרְתִּי: כְּשֶׁיִּמָּצֵא בְּמִדּוֹת שֵׂכֶל מָלֵא. אָמַר: מָתַי תִּהְיֶה הַשְּׁתִיקָה כְלִמָּה. אָמַרְתִּי: כְּשֶׁיִּהְיֶה הַדִּבּוּר חָכְמָה. אָמַר: מָתֵי תִּתְגַּל מַעֲלַת הַמַּשְׂכִּילִים. אָמַרְתִּי: בְּחֶבְרַת הַכְּסִילִים. אָמַר: מָתַי יִהְיוּ נִמְאָסִים הַמַּהֲלָלִים הַטּוֹבִים. אָמַרְתִּי: בְּמוֹת הַנְּדִיבִים. אָמַר: בַּמֶּה תְּהֵא הָעֲנָוָה נָאוָה. אָמַרְתִּי: בְּהַכְנִיעַ הַגַּאֲוָה וְהַתַּאֲוָה. אָמַר: מָה הַמִּדָּה אֲשֶׁר אֵין אַחֶרֶת כְּנֶגְדָּהּ. אָמַרְתִּי יִרְאֶת ה' לְבַדָּהּ. וְכַאֲשֶׁר הִשְׁלִים שְׁאֵלוֹתָיו.וּפֵרַקְתִּי כָל קֻשְׁיוֹתָיו. הֶעֱנִיקַנִי כָּל אִישׁ מִטּוֹבוֹתָיו. וְאָצַל עָלַי מִבִּרְכוֹתָיו:

וָאַעַן וָאֹמַר:

אֲשֶׁר יִסֹּב זְמַן עָלָיו בְּצִירָיו

יְבַקֶּשׁ לוֹ צְרִי מַרְפֵּא לְצִירָיו.

וְעֵת יִסָּגְרוּ שַׁעְרֵי חֲפָצָיו

בְּיַד שִׂכְלוֹ לְאַט יִפְתַּח שְׁעָרָיו.

וְהֶעָצֵל יְאַבֶּד הוֹן אֲבוֹתָיו

וְהֶחָרוּץ יְמַלֵּא טוּב חֲדָרָיו.

וּכְשָׁמְעִי אֲמָרָיו וּמִלָּיו. חֲקַקְתִּים עַל לִבִּי.

וּצְפַנְתִּים בְּחַדְרֵי קִרְבִּי. וְהִשְׁתַּחֲוֵיתִי אֵלָיו. וּפָנִיתִי

לְדַרְכִּי מֵעָלָיו. וַעֲזַבְתִּיהוּ וְלֹא מְצָאתִיִהוּ.
