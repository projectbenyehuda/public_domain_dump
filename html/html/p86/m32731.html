ספר תחכמוני - שַׁעַר שְׁלֹשִׁים וּשְׁנַיִם

בְּבָחוּר בְּשִׁירִים חֲיָלִים יְגַבֵּר. וְהַזָּקֵן מֵשִׁיב סְלָעִים יְשַׁבֵּר.


נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי בִימֵי עֲלוּמִים. בְּמַעַמד חֲכָמִים. יְשִׁיבוּן מְתֵי שֵׂכֶל נִכְלָמִים. וּבַעֲלֵי לָשׁוֹן נֶאֱלָמִים. וּבְעוֹדֵנוּ מְדַבְּרִים דִּבְרֵי מְלִיצוֹת. וְחִידוֹת נִמְרָצוֹת. וְאוֹרְגִים מִן הַשִּׁירִים בִּגְדֵי חֲלִיצוֹת. בִּזְהַב הַיֳּפִי מְשֻׁבָּצּוֹת. נִצַּב עָלֵינוּ אִישׁ בַּעַל לָשׁוֹן נִמְרָץ. וְעַל פִּיו חֲזוֹן הַשִּׁיר נִפְרָץ. וְהָיָה מִתְפָּאֵר. כִּי בְשִׁירוֹ כָל סָתוּם יְבָאֵר. וְכָל מַחְשָׁךְ יָאֵר. וּבְהֶגְיוֹנוֹ יִקְרַע יָם הַשִּׁיר קְרָעִים. וּבְצוּר לְשׁוֹנוֹ יְפוֹצֵץ סְלָעִים. וַיִּגַּשׁ אֵלָיו אֶחָד מִן הַבַּחוּרִים הַבְּחִירִים. מַשְׂכִּיל בִּמְלֶאכֶת הַשִּירִים. וְכָל דְּבָרָיו מְמֻלָּאִים בִּפְנִינִים מְיֻסָּדִים בְּסַפִּירִים. וַיֹּאמֶר לוֹ: אִם תִּצְדַּק בִּדְבָרֶיךָ. וְתִזְכֶּה בְּמַאֲמָרֶיךָ. לְכָה נָא וְנָרוּץ בְּמַעֲרֶכֶת הַשִּׁיר שְׁנֵינוּ. וְנַחֲלֹק שְׁנֵי חֲרוּזִים בֵּינֵינוּ. אֶבְנֶה אֲנִי בִּית אֶחָד וְאֲיַסֵּד הַבִּנְיָן. וְאַתָּה תַעֲשֶׂה בַיִת שֵׁנִי בְּמָשָׁל אֲשֶׁר בּוֹ יִשְׁלַם הָעִנְיָן. וְאִם תְּמַלֵּא הַתְּנַאי הַזֶּה. שְׁאֵלָתְךָ תִּמְצָא וְחֶפְצְךָ תֶּחֱזֶה. אָמַר הַזָּקֵן: טוֹב הַדָּבָר אֲשֶׁר דִּבַּרְתָּ. וְהִנְנִי מְזֻמָּן לְךָ לְכָל אֲשֶׁר גָּזַרְתָּ.

וַיִּגַּשׁ הַבָּחוּר וַיְדַבֵּר עַל הָעֵט וַיֹּאמַר:

מַה נֶּחְמָד עֵט סוֹפֵר יְרַקְרַק וּמֻצְהָב. נָעִים וְנֶאֱהָב. בַּגִּבּוֹרִים יִרְהָב. וְלַעֲרֹךְ קְרָב מַאֲמָר יֶאֱהָב.

אָמַר הַזָּקֵן:

יִדְמֶה שַׁרְבִיט זָהָב. וְטוּרָיו כִּמְדוּרַת לַהַב. בָּהֶם חֵן וָאַהַב. וְחֵמָה וָרַהַב.

וַיְשׁוֹרֵר הַבַּחוּר וַיֹּאמַר:

וְעֵט סוֹפֵר יְרוֹצֵץ עַל מְגִלָּה

וְיִרְקֹם בַּשְּׁחָרִים טוּר אֲפֵלָה.

אָמַר הַזָּקֵן:

כְּמוֹ נָחָשׁ אֲשֶׁר יָרוּץ בְּעָפָר

וְיַשְׁאִיר אַחֲרָיו דֶּרֶךְ סְלוּלָה.

וַיְדַבֵּר הַבָּחוּר עַל הָאִגֶּרֶת. וַיֹּאמַר: מַה נָּעֲמָה אִגֶּרֶת. הַנֹּעַם אוֹגֶרֶת. כְּאִלּוּ בְאַבְקַת רוֹכֵל מְקֻטֶּרֶת. וְהִיא עַל מִשְׁבְּצוֹת הַיֳּפִי שַׁרְשֶׁרֶת.

אָמַר הָּזֵקן:

כְּעַלְמָה נָאוָה שְׁחַרְחֹרֶת. גְּלוּיָה וְחוֹדֶרֶת. אִלֶּמֶת מְדַבֶּרֶת. וּבְלִי לָשׁוֹן רָזִים מְגַלָּה וּמְסַפֶּרֶת. וּתְהִלַּת ה' מְבַשֶּׂרֶת.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

וְאִגֶּרֶת מְמֻלָּאָה תְבוּנָה

וְעַל חֵיק הָאֱמוּנָה הִיא אֲמוּנָה.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

כְּמוֹ עַלְמָה אֲשֶׁר לֶחְיָהּ לְבָנָה

וְשַׂעַרָהּ עַל מְאוֹרָהּ כַּעֲנָנָה.

וַיְדַבֵּר הַבָּחוּר עַל הַחֶרֶב וַיֹּאמַר: רְאוּ חֶרֶב שְׁלוּפָה. בְּגִזְרַת כֶּסֶף צְרוּפָה. וּבִצְנִיף הַיֳּפִי צְנוּפָה. פַּעַם מִסְתַּתֶּרֶת וּפַעַם חֲשׂוּפָה. כְּאִלּוּ מִן הַבְּרָקִים חֲטוּפָה. אוֹ מִן הַמְּאוֹרִים קְטוּפָה.

אָמַר הַזָּקֵן:

תִּדְמֶה כְּשׁוֹשַׁנָּה אֲדֻמָּה וְיָפָה. בִּדְמֵי חֲלָלִים פִּרְיָהּ וַעֲנָפָהּ. וּמִבְּלִי כָנָף עָפָה. לְהִלָּחֵם מִלְחֶמֶת תְּנוּפָה.

וַיְשׁוֹרֵר הַבָּחוּר:

רְאוּ חֶרֶב בְּיַד גִּבּוֹר שְׁלוּפָה

לְעֵין עִוְרִים בְּמַחְשַׁכִּים תְּנוֹצֵץ.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

חֲשַׁבְתִּיהָ בְּתוֹךְ הַמַּעֲרָכָה

לְשׁוֹן בָּרָק עֲלֵי שַׁחַק יְרוֹצֵץ.

וַיְדַבִּר הַבָּחוּר עַל הַשִּׁרְיָן וַיֹּאמַר:

רְאוּ שִׁרְיָן עָמְקוּ מַחְשְׁבוֹתָיו. וְרָחֲבוּ חָכְמוֹתָיו. וְצָרוּ טַבְּעוֹתָיו. וְכָל הַלּוֹבֵשׁ מַחְלְצוֹתָיו. יִשְׂחַק לַקְּרָב וּתְשֻׁאוֹתָיו. וְלַמֶּות וְצִבְאוֹתָיו.

אָמַר הַזָּקֵן:

יִדְמֶה לְמִגְדַּל עֹז סְגוּרוֹת דַּלְתוֹתָיו. פְּתוּחוֹת פִּתּוּחֵי מִסְגְּרוֹתָיו. כְּמוֹ גוּף הַגְּמַלִּים טַבְּעוֹתָיו. וּכְעֵין הַזְּבוּבִים מַסְמְרוֹתָיו.

וַיְשׁוֹרֵר הַבָּחוּר:

רְאוּ שִׁרְיָן עֲצוּמוֹת מַחְשְׁבוֹתָיו

בְּיַד חָכְמָה עֲשׂוּיִים מִסְגְּרוֹתָיו.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

כְּמוֹ גּוּף הַגְּמַלִים טַבְּעוֹתָיו

וְעֵינֵי הַחֲגָבִים מַסְמְרוֹתָיו.

וַיְדַבֵּר הַבָּחוּר עַל הַסּוּס וַיֹּאמַר: מַה נֶּחְמָד סוּס יָרוּץ וְיִדְהָר. לְאוֹרוֹ הוֹלֵךְ חשֶׁךְ יִנְהָר. יִלָּבֵב כָּל מָהִיר וְנִמְהָר. וְיִדְאֶה כַּנֶּשֶׁר מֵהַר אֶל הָר.

אָמַר הַזָּקֵן:

כַּבָּרָק יִזְהָר. וְכָרוּחַ רָץ וּמִהֵר. יָפִיץ לְכָל עֵבֶר זִיו וָזֹהַר. כְּעֶצֶם הַשָּׁמַיִם לָטֹהַר.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

וְסוּס יָרוּץ כְּרוּץ בָּרָק וְכִמְעַט

לְחֹם לִבּוֹ יְהִי נָמֵס כְּדוֹנַג.

וַיַּעַןַ הַזָּקֵן וַיֹּאמַר:

בְּרֹאשׁ הַר כַּצְּבִי הוּא אַךְ בְּמִדְבָּר

כְּמוֹ נֶשֶׁר וּבַנָּהָר כְּמוֹ דָג.

וַיְדַבֵּר הַבָּחוּר עַל הָאֲבוּקָה וַיֹּאמַר: מַה נָּעֲמָה אֲבוּקָה דוֹלֶקֶת. וּמִנְּיצוֹצֶיהָ חִצִּים מוֹרֶקֶת. וַחֲנִיתוֹת זוֹרֶקֶת.

אָמַר הַזָּקֵן:

בְּתוּלַת אֲהָבִים חוֹשֶׁקֶת. דַלָּה יְרַקְרֶקֶת. הֻכְּתָה בְדַלֶּקֶת. בּוֹכָה וְצוֹחֶקֶת. וִידֵי הַלַּהַב יְמַהֲרוּ דִמְעָה בְלֶחְיָהּ לָצֶקֶת. הֵם מַגִּישִׁים אֵלֶיהָ וְהִיא מוֹצֶקֶת.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

הַבֵּט אֲבוּקַת שַׁעֲוָה דוֹלֶקֶת

וּלְעֵין יְדִידִים זָהֳרָהּ חוֹלֶקֶת.

וַיַּעַן הַזָּקֵן וֶיֹּאמַר:

תִּדְמֶה לְעָנָף מִזְּהַב אוֹפִיר אֲשֶׁר

מֵעַל לְרֹאשׁוֹ צָמְחָה בָרֶקֶת.

וַיְדַבֵּר הַבָּחוּר עַל צְרוֹר הַמּוֹר וַיֹּאמַר: מַה נֶּחְמָד צְרוֹר הַמּוֹר בְּעִנְיָנָיו. הֶעָשׂוּי לְהַסְתִּיר רָקַח שְׁמָנָיו. וְלִצְפֹּן רֵיחַ עֲדָנָיו. וְטוּב רֵיחוֹ יְגַלֶּה מַצְפּוּנָיו.

אָמַר הַזָּקֵן:

הוּא כְחוֹשֵׁק בְּדִמְיוֹנָיו. יַסְתִּיר חִשְׁקוֹ בְרַעְיוֹנָיו. וְיִתְגַלֶּה סוֹדוֹ מִמַּרְאֵה עֵינָיו. וּדְמוּת פָּנָיו.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

צְרוֹר נַעְשֶׂה לְהַסְתִּיר מֹר דְּרוֹר אַךְ

יְגַלֵּהוּ נְעִים רֵיחַ שְׁמָנָיו.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

כְּמוֹ רָכִיל אֲשֶׁר כָּרְתוּ לְשׁוֹנוֹ

וְלֹא יֶחְדַּל אֲבָל קוֹרֵץ בְּעֵינָיו.

וַיְדַבֵּר הַבָּחוּר עַל הַיַּיִן וַיֹּאמַר: רְאוּ בֶן שׂוֹרֵק. לִשְׂמָחוֹת שׁוֹרֵק. וְלַאֳנָחוֹת שִׁנּוֹ חוֹרֵק. חַיֵּי נְפָשִׁים נָעֳמוֹ. וְהַשִּׂמְחָה נָפְלָה בְשִׁכְמוֹ. יוֹרֶה הַיָּגוֹן לְתֻמּוֹ. וְיוֹצִיא עָלָיו כְּלֵי זַעְמוֹ.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

הוּא כַשֶּׁמֶשׁ בְּחֻמּוֹ. וְכַסַּהַר בְּתֻמּוֹ. וְכַנֹּפֶת בְּטַעְמוֹ. וְכַמּוֹר בְּבָשְׂמוֹ.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

יְדֵי מַשְׁקֶה יְרִיקוּן יֵין אֲדַמְדָּם

יְחַיּוּ לֵב אֱנוֹש עֵת יִשְׁפְּכוּ דָם.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

נְעִים טַעְמוֹ כְּמוֹ רֹק פִּי צְבִיָּה

וּמַרְאֵהוּ כְּלֶחְיָהּ הַמְאָדָּם.

וַיְדַבֵּר הַבָּחוּר עַל הַנֵּץ וַיֹּאמַר: רְאוּ נֵץ מִתְנוֹצֵץ. כָּרוּחַ יְרוֹצֵץ. כְּנָפָיו מִשְׁבְּצוֹת שָׁנִים. וְאֶבְרוֹתָיו כְּטוּרֵי זָהָב מְתֻקָּנִים. לְהַשִׂיג הַבְּרָקִים מוּכָנִים.

אָמַר הַזָּקֵן:

כְּאִלּוּ הוּא מַבִּיט בִּשְׁנֵי פְנִינִים. וְנוֹשֵׁךְ בִּשְׁנֵי צְנִינִים. וְתוֹםֵשׂ בִּשְׁנֵי חִצִּים שְׁנוּנִים. וּמְעוֹפֵף בְּשְׁנֵי עֲנָנִים.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

וְנֵץ לִפְאַת זְבוּל יִפְרֹשׂ כְּנָפָיו

וְיוֹרִיד הַנְּשָׁרִים מִשְּׁחָקִים.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

שְׁתֵּי עֵינָיו חֲצוּבִים מִפְּנִינִים

וְעָשָׂה אֶת כְּנָפָיו מִבְּרָקִים.

וַיְדַבֵּר הַבָּחוּר עַל הַזְּאֵב וַיֹּאמַר:

זְאֵב טוֹרֵף. אַכְזָר וּקְשֵׁה עֹרֶף. מֵעֲשׂוֹת רַע לֹא יֶחְדַּל וְלֹא יֶרֶף.

אָמַר הַזָּקֵן:

הוּא כָּאֵשׁ שׂוֹרֵף. וְכַנַּחַל גּוֹרֵף. בַּקַּיִץ וּבַחֹרֶף. לֹא יִשְׁכַּב עַד יֹאכַל טֶרֶף.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

זְאֵב יִטְרֹף אֲשֶׁר יִמְצָא וְיַחְטֹף

וְעֵינָיו כַּבְּרָקִים עֵת יְרוֹצֵץ.

וַיַּעַן הַזָּקֵן וַיּאֹמַר:

חֲשַׁבְתִּיהוּ כְגַנָּב בָּא בְלַיִל

וְלַפִּיד בֵּין שְׁתֵּי עֵינָיו יְנוֹצֵץ.

וַיְדַבֵּר הַבָּחוּר עַל הַבָּרָק וַיֹּאמַר:

וְאוֹר בָּרָק כְּחָרוּץ פַּז יְרַקְרַק. לֻטַּשׁ וּמֹרַק. אוֹחֵז קְצוֹת שַׁחַק. יִקְרַב עֵת וְעֵת יִרְחַק.

אָמַר הַזָּקֵן:

כְּטוּר אֹדֶם בְּנֹפֶךְ הוּחַק. וּבְטֶרֶם נִכְתַּב נִמְחַק. כְּאִלּוּ הוּא עַל פְּנֵי שַׁחַק. כּוּשִׁי רָגָז וְשָׂחַק.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

וְהַבָּרָק מְשַׂחֵק עַל עֲנָנִים

כְּגִבּוֹר רָץ וְלֹא יִיעַף וְיִיגַע.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

כְּשׁוֹמֵר לֵיל אֲשֶׁר נִרְדַּם וְיִפְקַח

קְצָת עֵינוֹ וְתִסָּגֵר כְּרֶגַע.

וַיְדַבֵּר הַבָּחוּר עַל מֵימֵי הַנְּחָלִים וַיֹּאמַר:

רְאוּ נַחַל כְּטוּרֵי שֹׁהַם רָצוּף. אוֹ כְּצוּר חֶרֶב חָשׂוּף. מְקוֹרוֹ עַל פְּנֵי אֶרֶץ יָצוּף. וְטַעְמוֹ נֹפֶת צוּף.

אָמַר הַזָּקֵן:

יִדְמֶה לְזִרְמֵי גֶשֶׁם עָרוּף. אוֹ רְבִיבֵי טַל נָטוּף. אוֹ שַׁרְשְׁרוֹת כֶּסֶף צָרוּף. אוָ בְרַק חֶרֶב שָׁלוּף. אוֹ בְכִי חוֹשֵׁק אָסוּף. יוֹסִיף וְלֹא יָסוּף.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

וְנַחַל יֵלְכוּ מֵימָיו לְאִטָּם

כְּאִלּוּ שַׁרְשֻרְוֹת כֶּסֶף זְרָמָיו.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

ְוַיד רוּחַ כְּיַד חָרָשׁ וְאָמָּן

תְּחַבֵּר טַבְּעוֹת שִׁרְיָן בְּמֵימָיו.

וַיְדַבֵּר הַבָּחוּר עַל הָרִמּוֹנִים וַיֹּאמַר:

רְאוּ רִמּוֹנִים. זַכִּים אַדְמוֹנִים. בְּחֵיק הַחֵן אֲמוּנִים. וּבְאוֹצְרוֹת נֹעַם טְמוּנִים.

אָמַר הַזָּקֵן:

יִדְמוּ אֲרוֹנִים. מְלֵאִים מַעֲדַנִים. בְּתוֹכָם פְּנִינִים. לֶאֱכֹל מְזֻמָּנִים. וְהָיוּ נְכוֹנִים.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

וְרִמּוֹנִים לְחֵיהֶם כַּעֲלָמוֹת

וְהֵמָּה מִמְּאוֹר עֵין אִישׁ צְפוּנִים.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

בְּעֵת כִּי תִפְתְּחָה אוֹתָם תְּדַמֶּה

אֲרוֹן זָהָב וְהוּא מָלֵא פְנִינִים.

וַיְדַבֵּר הַבָּחוּר עַל הָאֱגוֹזִים וַיֹּאמַר:

מַה נָּעֲמוּ אֱגוֹזִים. בְּגִנְזֵי הַחֵן גְּנוּזִים. הֵם תַּאֲוַת מְחַכִּים. וְשִׂמְחַת מְבַכִּים.

אָמַר הַזָּקֵן: הֵם קָשִׁים וְרַכִּים. נְעִימִים וְזַכִּים. כַּעֲבָדִים מֻכִּים. וּבְכָל יָד מְדֻכִּים. וְיִתְּנוּ לְחִי לְמַכִּים.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

וּמַה נָּעִים אֱגוֹז תּוֹכוֹ פְּרִי חֵן

וְיִמְנַע הַפְּרִי עַד יִשְׁבְּרוּהוּ.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

כְּאִלּוּ הוּא כְסִיל מֵאֵן קְחִת בִּין

וּמוּסָר עַד בְּמַכּוֹת יִסְּרוּהוּ.

וַיְדַבֵּר הַבָּחוּר עַל הַכִּנּוֹר וַיֹּאמַר: רְאוּ

כִנּוֹר אִלֵּם. וְיַבִּיעַ אִמְרֵי נֹעַם. וְחֵרֵשׁ וְקוֹלוֹ קוֹל רַעַם.

בְּפִיהוּ טוּב טַעַם. וְקוֹלוֹ לְשׁוֹמְעָיו יִנְעָם.

אָמַר הַזָּקֵן:

דְּמוּתוֹ כְּיָרֵךְ נִצְמַד אֱלֵי פַעַם. יְעוֹרֵר הַטַּעַם. וְיַשְׁקִיעַ הַזַּעַם. וְיַכֶּה הַיְגוֹנִים. כְּפַטִּישׁ וְהוֹלֵם פָּעַם.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

רְאוּ כִנּוֹר בְּחֵיק עַלְמָה יְרַנֵּן

וְיִמְשֹׁךְ לֵב בְּהוֹד קוֹלוֹ וְצִבְיוֹ.

וַיַּעַן הַזֵּקן וַיֹּאמַר:

כְּמוֹ יוֹנֵק בְּחֵיק אִמּוֹ מְבַכֶּה

וְהִיא תָשִׁיר וְצוֹחֶקֶת לְבִכְיוֹ.

וַיְדַבֵּר הַבָּחוּר עַל הָעֲגֻלָּה הַמְסוֹבֶבֶת בַּמַּיִם וַיֹּאמַר:

מַה נָּעֲמָה עֲגֻלָּה. הִיא לְכָל חֶמְדָּה סְגֻלָּה. מְסוֹבֶבֶת כְּגֻלָּה. וְשׁוֹאֶבֶת מִתּוֹךְ הַגֻּלָּה.

אָמַר הַזָּקֵן:

תִּדְמֶה לְיַעֲלַת אֲהָבִים דַּלָּה. בְּצִירֵי חֵשֶׁק

מְחֻלָּה. וְצוֹעֶקֶת קוֹל כְּחוֹלָה. וְיוֹרֶדֶת וְעוֹלָה. וּמִמַּעְיַן אֲהָבִים דְּמָעוֹת דּוֹלָה.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

וְנוֹשְׂאָה לְמַיִם וְהִיא בָם נְשׂוּאָה

טְהוֹרָה וְתִטְבֹּל בְּמֵימֵי נְהָרִים.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

תְּרַנֵּן פְּעָמִים כְּרִנַּת נְבָלִים

וְתִשְׁאַג פְּעָמִים כְּשַׁאְגַת כְּפִירִים.

וַיְדַבֵּר הַבָּחוּר עַל הָאֶתְרוֹג וַיֹּאמַר:

רְאוּ אֶתְרוֹג נָעִים וְנֶחְמָד. בּוֹ לְכָל עַיִן מַחְמָד. וְיָפְיוֹ לֹא יִמַּד.

אָמַר הַזָּקֵן:

כְּאִלּוּ דַרְכֵי חוֹשְׁקִים לָמַד. וְלִבְנֵי הַחֵן נִצְמַד. לֹא סָר יָפְיוֹ וְלֹא נִשְׁמַד. רַק בְּעֵינוֹ עָמַד.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

פְּרִי הָדָר יְפֵה מַרְאֶה וְנֶהְדָּר

הֲכִי הָאֵל רְבִיד חֵן הֶעֱנִיקוֹ.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

יְרַקְרַק מִבְּלִי מַחְלָה כְּדִמְיוֹן

פְּנֵי חוֹשֵק בְּעֵת יִפְגּשׁ חֲשׁוּקוֹ.

וַיְדַבֵּר הַבָּחוּר עַל הַנֵּר וַיֹּאמַר:

מַה נֶּחְמָד אוֹר דּוֹלֵק. אַחַר גְּדוּד הָאֹפֶל דּוֹלֵק. וְחִצָּיו לְהַשְׁמִיד דּוֹלֵק. וּלְכָל סְבִיבָיו אוֹר חוֹלֵק.

אָמַר הַזָּקֵן:

בּוֹ לְרוֹאָיו חֶמְדָּה וָחֵשֶׁק. וְנִיצוֹצָיו כִּכְלֵי נֶשֶׁק. כְּאִלּוּ הַכּוֹכָבִים עוֹשֵׁק. אוֹ לְחִי סַהַר נוֹשֵׁק.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

הַבֵּט לְנֵר יָאִיר בְּתוֹךְ חשֶׁךְ

וִיגַל אֲשֶׁר הַמַּאֲפֵל כָּמָס.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

עֵת בַּמְּנוֹרָה יַעֲלֶה אוֹרוֹ

יִדְמֶה לְשׁוֹן זָהָב אֲשֶׁר נָמָס.

וַיְדַבֵּר הַבָּחוּר עַל אוֹר הַשֶׁמֶשׁ וַיֹּאמַר: מַה נָּעֲמָה שֶׁמֶשׁ בְּרוּם שַׁחַק זְבוּלֶיהָ. וּבָאָרֶץ שְׁבִילֶיהָ. לְהָאִיר מַאֲפֵלֶיהָ. וּלְהָסִיר בִּגְדֵי קַדְרוּתָהּ מֵעָלֶיהָ.

אָמַר הַזָּקֵן:

תִּדְמֶה לִגְבֶרֶת תְּגַבֵּר חֲיָלֶיהָ. וּבְעֵת תָּסִיר צְעִיפָה מֵעָלֶיהָ. תְּשַׂמַּח כָּל הַנִכְסָפִים אֵלֶיהָ. וְיָשִׂישׂוּ אִתָּהּ מָשׂוֹשׂ כָּל הַמִּתְאַבְּלִים עָלֶיהָ.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

רְאוּ שֶׁמֶשׁ אֲשֶר יִפְרֹשׂ כְּנָפָיו

עֲלֵי אֶרֶץ לְהָאִיר אֶת נְשָׁפָיו.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

כְּעֵץ רַעְנָן אֲשֶׁר צָמַח בְּשַחַק

וְהִגִּיעוּ עֲדֵי אֶרֶץ עֲנָפָיו.

וַיְדַבֵּר הַבָּחוּר עַל הַיָּרֵחַ. אֲשֶׁר עַל פְּנֵי הַמַּיִם זוֹרֵחַ. וַיֹּאמַר: רְאוּ לְבָנָה לָהּ בַּשְּׁחָקִים מַעֲרֶכֶת. וְעַל הָאָרֶץ מוֹלֶכֶת. וְכָאֳנִיָּה עַל פְּנֵי הַמַּיִם מְהַלֶּכֶת.

אָמַר הַזָּקֵן:

תִּדְמֶה לְשׁוֹן זָהָב נִתֶּכֶת. אוֹ לַהַט הַחֶרֶב הַמִּתְהַפֶּכֶת. תַּחְשְׁבֶנָּה עוֹמֶדֶת וְהִיא הוֹלֶכֶת. וּבַעֲבוֹת אַהֲבָה נִמְשֶׁכֶת.

וַיְשׁוֹרֵר הַבָּחוּר וַיֹּאמַר:

רְאוּ סַהַר עֲלֵי מֵימֵי נְחָלִים

כְּגִזְרַת פָּז עֲלֵי תַרְשִׁישׁ רְצוּפָה.

וַיַּעַן הַזָּקֵן וַיֹּאמַר:

כְּאִלּוּ הַלְּבָנָה גִלְּתָה עַל

פְּנֵי מַיִם דְּמוּת חֶרֶב שְׁלוּפָה.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר כִּלּוּ שְׁנֵיהֶם מְלִיצות שִׁירוֹתָם. וְנֹעַם תְּשׁוּבָתָם. אָמַר הַבָּחוּר לָּזֵקן: נִשְבַּעְתִּי בְּחֵי הַמְלַמֵּד אָדָם דַּעַת. וְנַפְשׁוֹ כָּל סוֹד יוֹדַעַת. כִּי לֹא רָאִיתִי עֲרֹךְ אֵלֶיךָ. וְאֵין מַשִּׂיג אֲבַק רַגְלֶיךָ. וּמְלִיצוֹת מְשָׁלֶיךָ.

אָמַר הַזָּקֵן:

וְגַם אֲנִי לֹא רָאִיתִי שְׂרִיג אֶזְרָח כְּיָפְיֶךָ. הַחָכְמוֹת מְגָדֶיךָ. וְהַתְּבוּנוֹת פִּרְיֶךָ. וְעֵינֶיךָ קַשְׁתְּךָ וְתֶלְיֶךָ. וְהַלְּבָבוֹת שִׁבְיֶךָ. וְהַמְּלִיצוֹת הַיָּפוֹת כְּאִלּוּ גְנַבְתָּם מֵהֲדַר צִבְיֶךָ. וְהַמְּשָׁלִים הַחֲמוּדִים כְּאִלּוּ גְזַלְתָּם מִזֹּהַר לֶחְיֶךָ. וַיָאצֶל עָלָיו הַבָּחוּר וַחֲבֵרָיו כָּל אִישׁ מִנִּדְבוֹתָיו. עַד נִלְאָה לָשֵׂאת רֹב מַתְּנוֹתָיו. אַחֲרֵי כֵן אָמַר לוֹ הַבָּחוּר: הִשְׁבַּעְתִּיךָ בָאֵל הַחוֹלֵק כָּבוֹד וְחֶמְדָּה. וְנָתַן לְךָ שֵׂכֶל בְּלִי מִדָּה. וַיָּשֶׂם פִּיךָ כְּחֶרֶב חַדָּה. לְהוֹדִיעֵנִי שְׁמֶךָ. כִּי הִבְהִילַנִי נָאֳמֶךָ.

וַיַּעַן וַיֹּאמַר:

אֲנִי פֶלֶא בְּעֵינֵי הַמְּבִינִים

וְלִוְיַת חֵן לְרָאשֵׁי הַתְּעוּדוֹת.

הֲכִי קָרְאוּ שְׁמִי חֶבֶר לְמַעַן

אֲחַבֵּר כָּל פְּזוּרֵי הַחֲמוּדוֹת.

וּכְשָׁמְעִי נָאֳמוֹ. יְדַעְתִּיו בִּשְׁמוֹ. וְהֵרַחְתִּי רֵיחַ בָּשְׂמוֹ. וְנָפַלְתִּי עַל שִׁכְמוֹ. וְנָשַׁקְתִּי שְׂפָמוֹ. וְעָמַדְתִּי יָמִים אֲחָדִים עִמּוֹ. לְהָסִיר יְגוֹן לִבִּי וְזַעְמוֹ. עַד הִפְרִידוּ בֵינֵינוּ קוֹרוֹת הַזְּמָן. וְעָזַב נְדוֹדוֹ בְּלִבִּי צִיר נֶאֱמָן. וְגַם הִשְׁאִיר זִכְרוֹ בְּפִי כְּצִנְצֶנֶת הַמָּן.
