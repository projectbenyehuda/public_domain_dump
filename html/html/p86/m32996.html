ספר תחכמוני - שַׁעַר עֶשְׂרִים

מִשֶּׁבַע הַבְּתוּלוֹת וְתַחְבּוּלוֹתֵיהֶן


נְאֻם הֵימָן הָאֶזְרָחִי: מְשָׁכוּנִי תַעֲנוּגֵי הַשַּׁחַרוּת וְהַיַּלְדוּת. וְעָלַי מִזִּיו הַנְּעוּרִים הַנֵּזֶר וְהָעֵדוּת. לִרְכֹּב אֳנִיּוֹת. וְלַעֲבֹר נְהָרוֹת וְצִיּוֹת. בִּמְעוֹנוֹת אֲרָיוֹת. וּפְרִיצֵי חַיּוֹת. וְרָתַמְתִּי פִּרְדֵי הַפְּרִידָה. וְדָאִיתִי בְכַנְפֵי הַחֲסִידָה. וְלָקַחְתִּי הַתְּשׁוּקָה לְצֵידָה. וְנָסַעְתִּי מִבֵּית לֶחֶם יְהוּדָה. בְּעוֹד עֲנַף הַיַּלְדוּת פּוֹרֵחַ. וְדוּדָאֵי הַשַּׁחֲרוּת נָתְנוּ רֵיחַ. וְאוֹר הַנַּעֲרוּת זוֹרֵחַ. וְלֹא כִסְּתָה שֶׁמֶשׁ הַשֵּׂיבָה זִיו הַיָּרֵחַ. וּבְעוֹדִי מִתְהַלֵּךְ לְרוּחַ הַיּוֹם עַל פְּנֵי הַשָּׂדֶה. וָאֶשָּׂא עֵינַי וָאֵרֶא עַל פְּנֵי הַמִּדְבָּר שֶׁבַע נְעָרוֹת. כְּשִׁבְעָה מְאוֹרוֹת. הָאָרֶץ הֵאִירָה מֵהִלָּם. וְשָכַן עֲלֵיהֶם אוֹר נֶעְלָם. וְכִרְאוֹתָן דְּמוּתִי. קִדְּמוּ בְשָׁלוֹם אוֹתִי. וְהֵן מְכֻסּוֹת בִּצְעִיפֵיהֶן. וּמִתְעַטְּפוֹת בִּרְדִידֵי צְנִיפֵיהֶן. אֲבָל זֹהַר יָפְיָן שָׁלַח בְּרָקִים לְעֵינַי מִפְּנֵיהֶן. וַיּוֹרֶה זִקִּים לְלִבִּי מֵעֵינֵיהֶן. וְכֻלָּן בְּקוֹמוֹת נְעִימוֹת. כַּנִּדְגָּלוֹת אֲיֻמּוֹת. וְלָהֶן שָׁדַיִם נִצְּבוּ כְּחוֹמוֹת וְקוֹמוֹת. כַּאֲרָזִים נִצָּבוֹת.

מִמַּחְצַב הַיֹפִי חֲצוּבוֹת. וּמֵעֲצֵי עֵדֶן קְצוּבוֹת. וְכָל אַחַת מְקֻטֶּרֶת בְּקָנֶה וְקִדָּה. לוֹבֶשֶׁת שְׂכִיוֹת הַחֶמְדָּה. מְטוֹפְפֶת בַּשָּׂדֶה. כְּאִלּוּ הַיֳּפִי עֲגֻלָּה וְהִיא בְתוֹכוֹ נְקֻדָּה. בִּלְחִי בְרֻדָּה. נִקְבַּץ בָּהּ הַחֵן וְהַיֳּפִי וְהָיוּ לַאֲגֻדָּה. וְאַחַת בֵּינֵיהֶן. מִשִּׁכְמָהּ וָמַעְלָה גְּבוֹהָה מֵהֶן. בְּעֵינַיִם. כְּיוֹנִים עַל אֲפִיקֵי מַיִם. וְעַפְעַפַּיִם. יְכֶשְׁפוּ לִמְכַשְפֵי מִצְרַיִם. וְשָׁדַיִם. יִקְרְעוּ הַלְּבָבוֹת לִשְׁנַיִם. וּמַתְנַיִם. לְחִשְקָם נָמֵס כָּל לֵב וְרָפוּ כָּל יָדַיִם. וְשׁוֹקַיִם. כְּעַמּוּדֵי שֵׁשׁ שְׁנַיִם. וְיָדַיִם. כְּאִלּוּ נִקְרְצוּ מִכְּפוֹר הַשָּׁמַיִם. וּלְחָיַיִם. יְשִׁיבוּן הַעֲרָבִים צָהֳרַיִם. וְשִׁנַּיִם. כְּטוּרֵי בְדֹלַח שְׁנַיִם. סְגוּרִים מִן הַשְּׂפָתַיִם. כִּבְרִיחַ וּדְלָתַיִם. וּמִן הַבְּדֹלָחִים הָהֵם יָזוּבוּ נִטְפֵי מַיִם. כְּנֹפֶת צוּף לְמַלְקוֹחַיִם. וְכָל הַשּׁוֹאֵב מֵהֶם לֹא רָעַב לַלֶּחֶם וְלֹא צָמֵא לַמַּיִם. וַתֹּאמֶר לִי: שָׁלוֹם עָלֶיךָ כְּפִיר אֲרָיוֹת. מְזַנֵּק מֵהַרְרֵי תוּשִׁיּוֹת. וָאַעַן וָאֹמַר לָהּ: צִבְאוֹת הַשָּׁלוֹם יִלָּווּ עָלַיִךְ. וִיקַוּוּ אֵלַיִךְ. אֲבָל הִשְׁבַּעְתִּי אֶתְכֶם בְּנוֹת יְרוּשָׁלַיִם. הַנְּטוּיוֹת גָּרוֹן וּמְשַׂקְּרוֹת עֵינַיִם. בְּחִין הַשְּׂפָתַיִם. הָאֲדֻמּוֹת בְּדַם קִרְבִּי. וְהָעֵינַיִם הַכְּחֻלֹּות בְּפוּךְ קַדְרוּת לִבִּי. וְהָעַפְעַפַּיִם אֲשֶׁר יוֹרוּ וְלֹא יַחְטִיאוּ אֶל צְלָעַי. וְהַלְּחָיַיִם אֲשֶׁר יְקוֹד גַּחֳלֵיהֶם יַצִּית אֵשׁ בְּמֵעַי. וְהַשִּׁנַּיִם אֲשֶׁר קָפְאוּ טוּרוֹתָם מִבְּדֹלַח דְּמָעַי. וְהָאֶצְבָּעוֹת הַקְּצוּבוֹת מֵעֵץ הַחַיִּים. וְהַמַּלְקוֹחַיִם אֲשֶׁר יָזוּב מֵהֶם מַיִם חַיִּים. וְזֹהַר הַפָּנִים אֲשֶׁר הָפְכוּ הַשַּׁחַר שָׁחוֹר. וְהֵשִׁיבוּ בְאוֹרָם צֵל הַמַּעֲלוֹת לְאָחוֹר – לְהָשִׁיב גְּזֵלַת לִבִּי אֲשֶׁר גְּזַלְתֶּן. וְרוּחִי אֲשֶׁר חִבַּלְתֶּן. וּלְהוֹדִיעֵנִי מֵאַיִן בּוֹאֲכֶן. וְאָנָה מְגַמַּת פְּנֵיכֶן.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

יַעְלָה אֲשֶׁר לִבִּי כַּחוּט מְשָׁכַתְהוּ

וּבְגַחֲלֵי חִשְׁקָהּ כַּיּוֹם שְׂרָפַתְהוּ.

חִמְלִי עֲלֵי חוֹשֵׁק נַפְשׁוֹ מְכָרַתְהוּ

וּבְמַאֲסַר אַהַב עֵינוֹ נְתָנַתְהוּ.

נָא רַחֲמִי עָלָיו וּפְנִי בְּחֵן אֵלָיו

פֶּן יֹאמְרוּ עָלָיו אִשָּׁה הֲרָגַתְהוּ.


וַתַּעַן אַחַת מֵהֶן וְהַבְּדֹלָחִים הַלַּחִים יִתְפַּזְרוּ מִבֵּין שִׁנֶּיהָ. וְהַבְּרָקִים יְרוֹצְצוּ מִבֵּין אִישׁוֹנֶיהָ. וְהַחֵן יִקְרָא אַבְרֵךְ לְפָנֶיהָ. אַךְ הָיָה הַמַּסְוֶה בֵין עֵינֶיהָ. וְנֶעֶלְמָה מֵעֵינֵי כָל חַי כִּי כִּסְּתָה פָנֶיהָ. וַתֹּאמֶר לִי: חָלִילָה חָלִילָה לָנוּ מֵעֲשׂוֹת לְךָ רָעָה כִּי לֹא נְגַעְנּוּךָ. וְלֹא הוֹגַעְנוּךָ. אֲבָל אַתָּה בָאתָ אֶל מוּל פְּנֵי הַמִּלְחָמָה הַחֲזָקָה וְעַל כֵּן פְּגַעְנוּךָ. וְאַתָּה בֶּן-אָדָם מַה לְּךָ לַחְקֹר עַל סוֹדוֹת הַגְּבִירוֹת. וְלִדְפֹּק עַל דַּלְתֵי הַחֵשֶׁק הַסְּגוּרוֹת. וְאֵיךְ בָּאתָ אֶל הֵיכַל מְלָכִים וְלֹא יָרֵאתָ. וְקָרַבְתָּ אֶל הֶחָצֵר הַפְּנִימִית וְלֹא נִקְרֵאתָ. וְאֵיךְ לֹא פָחַדְתָּ פֶּן יִפְגְּעוּ בְךָ חִצֵּי הָעֵינַיִם. וְחַרְבוֹת הָעַפְעַפַּיִם. וְיִגְזְרוּ כְבֵדְךָ לִשְׁנַיִם. וְיִגְזְלוּ לִבְּךָ בְּלִי יָדַיִם. וְתָעוּף נַפְשְׁךָ בְּלִי כְנָפַיִם. וּמַדּוּעַ לֹא יָגֹרְתָּ פֶּן יִשְּׁכוּךָ מִן הָעֵינַיִם הַנְּחָשִׁים הַשְּׂרָפִים. וְיִשְׂרְפוּךָ מִן הַלְּחָיַיִם גַּחֲלֵי רְשָׁפִים. וְאֵיךְ לֹא נִבְהַלְתָּ מִזִּיו הַשְּׁחָרִים וְקַדְרוּת הַנְּשָׁפִים. וְלֹא חָרַדְתָּ מֵרִמּוֹנֵי הַסְּעִיפִים. אֲשֶׁר תַּחַת הַצְּעִיפִים. וְלֵילֵי הַמַּחְלָפוֹת אֲשֶׁר תַּחַת הַצְּנִיפִים. וָאַעַן וָאֹמַר: לְאַט לִי הַצְּבִיָּה. הַטּוֹרֶפֶת כִלְבִיָּא. הַמּוֹשֶׁכֶת בְּקֶשֶת רְמִיָּה. הַהוֹרֶגֶת בְּלִי צְדִיָּה. כִּי מִיּוֹם נוֹלַדְתִּי. בִּשְׁמֵי הַיְקָר זָרַחְתִּי. וְעַל נַהֲרֵי שַׁלְוָה פָרַחְתִּי. וּמִנְּעוּרַי נְשָׂאוּנְי נְדִיבֵי עָם עַל כַּפַּיִם. וּבִכְתֵפַיִם. וְשֵׁעַשְׁעוּנִי הַשּׁוֹעִים עַל בִּרְכַּיִם. כִּי אֲבוֹתַי הָיוּ אֲצִילֵי אֶרֶץ יְקָרִים. וְהוֹרַי אַמִּיצֵי לֵב וְגִבּוֹרִים. לֵב הָאֳרִי יִמַּס מִפְּנֵיהֶם. וְיֵחַת הַזְּמַן מֵהִתְיַצֵּב לִפְנֵיהֶם. וּבֵין צְלָעַי נֶפֶשׁ יְקָרָה נֶאֱמֶנֶת. בְּצֵל הָעַנְוָה מִתְלוֹנֶנֶת. וְעִם כּוֹכְבֵי רוּם שׁוֹכֶנֶת. וַתְּהִי לָהֶם לֶאוֹמֶנֶת. מִמַּחְצַב כָּבוֹד חֻצְּבָה. וְעַל כֵּס הַיְקָר הֻצְּבָה. וְכַמָּה לְבָאוֹת צַדְתִּי. וּצְבָאוֹת לָכַדְתִּי. בְּרֶשֶׁת אַהֲבָתִי. וְעִנִּיתִי לִבָּם בִּפְרִידָתִי. וְהִבְעַרְתִּי בְקִרְבָּם אֵשׁ תְּשׁוּקָתִי. וּמִהְיוֹתִי חָקַרְתִי וְחָתַרְתִּי. עַל סוֻד נִטְמָן. וְלִבִּי רָד עִם אֵל וְעִם קְדוֹשִׁים נֶאֱמָן. וְלִי נָאֶה לַחְקֹר עַל כָּל סוֹד וְלִשְׁאֹל. כִּי אֵין זוּלָתִי לִגְאֹל.

וַתַּעַן וַתֹּאמֶר לִי: מַה נָּעֲמוּ אִמְרוֹתֶיךָ. וּמַה מָּתְקוּ דִבְרוֹתֶיךָ. הֻצַּק חֵן בְּשִׂפְתוֹתֶיךָ. וְנִפְלֵאת לָנוּ אַהֲבָתֶךָ. וְעַל כֵּן רָאוּי לָנוּ לָתֵת אֶת שְׁאֵלָתֶךָ. וְלַעֲשׂוֹת אֶת בַּקָּשָׁתֶךָ. יָדוֹעַ תֵּדַע כִּי זֹאת הַגְּבֶרֶת הָעֲדִינָה. הִיא פְאֵר כָּל מְדִינָה. בְּזֹהַר תְּמוּנָה. וּלְחִי כַלְּבָנָה. וְשֵׂעָר כַּעֲנָנָה. וְאֵין קֵצֶה לַתְּכוּנָה. וְהִיא יְחִידָה בְּבָנוֹת. וְתִפְאֶרֶת בֵּין הָעֲדִינוֹת. וְשָׂרָתִי בַמְּדִינוֹת. וְהַיּוֹם הַזֶּה לָבְשָׁה בִגְדֵי עֶדְיָהּ. וְקָשְׁרָה שְׁבִיסֵי חֶלְיָהּ. וְיָצְאָה לְהַרְאוֹת הָעַמִּים וְהַשָּׂרִים אֶת יָפְיָהּ. וְגַם זָבַת חָלָב וּדְבַשׁ הִיא וְזֶה פִּרְיָהּ.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

צְבִיַּת חֵן לְבָנָה כַלְּבָנָה

וְשַׂעְרָהּ עַל מְאוֹר לָחְיָהּ עֲנָנָה.

קְצָת הַחֵן לְבַד לָקְחוּ עֲלָמוֹת

וְלָהּ כָּל הַיֳּפִי הָיָה לְמָנָה.

כְּאִלּוּ קִנְּאוּ אוֹרִים לְיָפְיָהּ

וְלִלְחֹם שָׁלְפוּ חֶרֶב שְׁנוּנָה.

וְכוֹכְבֵי רוֹם כְּלַהְבֵי-הַחֲנִיתוֹת

וְהַסַּהַר כְּסוֹחֵרָה וְצִנָּה.

וְאִלּוּ אִישׁ אֲסִיר שֵׂיבָה יְשׁוּרָהּ

יְהִי שָׁב מִזְּמַן זֹקֶן לְעֶדְנָה.

שְׂפָתָה צוּף וְלֶחְיָהּ מָן וְרֻקָּה

כְּטַל אוֹרוֹת וּפָנֶיהָ כְּגַנָּה.

וְהָיָה הַיֳּפִי נוֹדֵד וְגוֹלֶה

עֲדֵי פָגַע מְעוֹנוֹ בָהּ וְחָנָה.

לְאוֹרָהּ נָהַרוּ עִוְרִים חֲצוֹת לֵיל

וְנִשְׁמַע מִלְּשׁוֹן אִלֵּם רְנָנָה.

וְאִלּוּ יֶחֱזֶה יָפְיָה חֲכַם לֵב

עֲזָבַתְהוּ בְלִי שֵׂכֶל וּבִינָה.

וּבָהּ כָּל הוֹד וְכָל כָּבוֹד מְחֻבָּר

וְכָל יֹפִי וְאֵין קֵץ לַתְּכוּנָה.

וְאֵיכָה תִדְרְשׁוּ עֵרֶךְ לְיָפְיָהּ

וְהוּא נֶעְדָּר בְּכָל-גּוֹי גַּם מְדִינָה.

וְאִם תֹּאמְרוּ הַכִי כָזֶה רְאִיתֶם

דְּעוּ כִּי לֹא רְאִיתֶם כָּל תְּמוּנָה.


אַחֲרֵי כֵן אָמְרָה לִי: וְגַם אַתָּה הַגֵּד לָנוּ עִנְיָנֶיךָ. וּמֵאֵיזֶה מָקוֹם מְעוֹנֶךָ. אָמַרְתִּי: אֲנִי הַגֶּבֶר. מִמִּבְחַר יַלְדֵי עֵבֶר. שׁוֹטַטְתִּי בְכָל עֵבֶר. וְעוֹפַפְתִי מִיָּם וְעַד מִזְרָח בְּלִי אֵבֶר. מִסְּפָרַד נִפְרַדְתִּי. וְאֶל אֶרֶץ יָוָן חָרַדְתִּי. וּמִשְׁבָּרִים שָׁבַרְתִּי. וּמַעֲבָרִים עָבַרְתִּי. וּבָאתִי מִנֹּא אָמוֹן. עַד הַר חֶרְמוֹן. וְצָעַנְתִּי מִצֹּעַן. לְאֶרֶץ כְּנָעַן. וּמִנּוֹף. עַד יְפֵה נוֹף. וּמִמִּצְרַיִם. עַד יְרוּשָׁלַיִם. וּמִצּוֹר. עַד חָצוֹר. וּמִדִּיבוֹן. עַד חֶשְׁבּוֹן. וּמֵאַרְנוֹן. עַד הַלְּבָנוֹן. וּמִדָּן. עַד דְּדָן. וּמִן הָרָמָתַיִם. עַד אֶרֶץ מְרָתַיִם. וּמִקְעִילָה. עַד חֳוִילָה. וּמֵאַשּׁוּר. עַד גְּשׁוּר. וּמֵאַרְבֵּל. עַד בָּבֶל. וּבְכָל אֲשֶׁר חָזִיתִי. כַּמַּרְאֶה הַזֶּה לֹא רָאִיתִי. וְעַתָּה אִם נָא מָצָאתִי חֵן בְּעֵינַיִךְ. הָסִירִי הַמַּסְוֶה אֲשֶׁר עַל פָּנַיִךְ. וְהָשִׁיבִי רוּחִי. וְהַחְלִיפִי כֹחִי. וּבַשְׁאֵלָה הַזֹּאת אֵצֵא מִמַּאֲסָרִי חָפְשִׁי. הֲלֹא מִצְעָר הִיא וּתְחִי נַפְשִׁי. וּכְשָׁמְעָה נִיבִי צָחֲקָה לְמִלַּי וַתַּחֲזֵק בִּי וַתֹּאמַר: הִנֵּה שָׁמַעְתִּי בְקוֹלֶךָ. וְעָשִׂיתִי מִשְׁאָלֶךָ. וַתָּסַר הַמַּסְוֶה מֵעַל עֵינֶיהָ. וְגִלְּתָה פָנֶיהָ. וָאֶרְאֶה וְהִנֵּה אִישׁ וּזְקָנוֹ אָרֹךְ. כְּאִלּוּ נָחָשׁ כָּרוּךְ. וְהוֹצִיא מִתַּחַת בִּגְדוֹ. חַרְבּוֹ שְׁלוּפָה בְיָדוֹ. וָאֵרֶא וָאֶפֹּל עַל פָּנַי. וְאָבְדוּ רַעְיוֹנַי. וּבִרְאוֹתוֹ כִּי סָר מִמֶּנִּי כֹּחִי. וְכִמְעַט יָצְאָה רוִּחי. הֵחֵל לִצְחוֹק. וְנִשְׁמַע קוֹלוֹ לְמֵרָחוֹק. וְהָעֲלָמוֹת נִמְלָא שְׂחוֹק פִּיהֶם. וְנָפְלוּ עַל פְּנֵיהֶם. וַיֹּאמֶר לִי: פְּקַח עֵינֶיךָ וּרְאֵנִי. אוּלַי תַּכִּירֵנִי. וָאֶתְבּוֹנֵן אֵלָיו וְהִנֵּה הוּא חֶבֶר הַקֵּינִי. אֲבִי הַתַּחְבֻּלוֹת הַגְּדוֹלוֹת. וְרָאשׁ הַנְּאָצוֹת וְהַנְּבָלוֹת. וָאֹמַר לוֹ: יַעְכָּרְךָ ה' כִּי עֲכַרְתַּנִי. וְהַכְרֵעַ הִכְרַעְתַּנִי. וְלַמָּוֶת הִגַּעְתַּנִי. אָמַר לִי: שָׁלוֹם שָׁלוֹם לְךָ בְעוֹדְךָ. חֲזַק וַעֲמֹד עַל עָמְדְךָ. וַתָּשָׁב רוּחִי אֵלַי בְּשָׁמְעִי אֲמָרָיו. וְהִתְחַזַּקְתִּי בִדְבָרָיו. וָאֹמַר לוֹ: מִי לְךָ כָּל הַמַּחֲנֶה הַזֶּה אֲשֶׁר פָּגַשְׁתִּי. וּבְחִשְׁקָם נֶעֱנַשְׁתִּי. וּבְרֶשֶׁת אַהֲבָתָם נוֹקַשְׁתִּי. אָמַר: אֵלֶּה הֵם צִבְאוֹת אַהֲבָתִי. אֲשֶׁר בָּהֶם חֲשׁוּקָתִי. וְלָהֶם תְּשׁוּקָתִי. וְהָיִיתִי מִתְהַלֵּךְ עִמָּהֶם. לִלְקֹט פְּנִינֵי אִמְרֵיהֶם. וּלְהִשְׁתַּעֲשֵׁעַ בְּמֶתֶק דִּבְרֵיהֶם. וּבִרְאוֹתֵנוּ אוֹתְךָ מֵרָחוֹק הִגַּדְתִּי לָהֶם. כִּי אַתָּה מִבְחַר יְדִידַי וְרֵעַי. וּסְגֻלַּת אֲהוּבַי וּמְיֻדָּעַי. וְחִלּוּ פָנַי וּפִיְּסוּנִי. וּבְחַיֵּי אַהֲבָתְךָ הִשְׁבִּיעוּנִי. לְהָתֵל בְּךָ וּלְרַמּוֹתְךָ. וְלִבְחֹן כֹּחֲךָ וּלְנַסּוֹתְךָ. וַיְמַהֵר לִקְרֹא בִשְׁלוֹמִי. וְהָלַךְ מֵעִמִּי. וְהָעֲלָמוֹת נֶעֶלְמוּ מֵעֵינָי. וְנָשְׂאָה אוֹתָם רוּחַ אֲדֹנָי. וְנִשְׁאַרְתִּי לְבַדִּי מִשְׁתּוֹמֵם בְּחִילִי. עַל הַמִּקְרֶה אֲשֶׁר קָרָה לִי. וּפָנִיתִי לִמְקוֹמִי וְלִבִּי נֶעֱצָב לְקוֹרוֹתָיו. וַאֲנִי מְקַלֵּל חֶבֶר הַקֵּינִי וּתְשׁוּקוֹתָיו.
