שׁוּבִי נַפְשִׁי

שׁוּבִי נַפְשִׁי לִמְנוּחָיכִי / כִּי יְיָ גָּמַל עָלַיכִי

הֲלֹא מֵרוּחַ פִּיו נְפָחֵךְ / וּמִכַּף רֶגֶל וְעַד רֹאשׁ מְתָחֵךְ

וּמִכֹּחַ תְּכוּנָתוֹ בִּמְעַט הֱכִינֵךְ / וּמֵעֹצֶם תְּבוּנָתוֹ בְּמִקְצָת הֱבִינֵךְ

וְהִשְׂכִּילֵךְ לָתוּר חָכְמוֹת / וְהֶחְכִּימֵךְ לָדַעַת מְזִמּוֹת

וְהִכְתִּירֵךְ בְּדַעַת וּתְבוּנָה / וְהִדְרִיכֵךְ עַל דֶּרֶךְ נְכוֹנָה

וְאַתְּ שָׂטִית תַּחַת צוּרֵךְ / וּפָרַקְתְּ עֻלּוֹ מֵעַל צַוָּארֵךְ

וַתְּתַעֲבִי אֶת שִׂכְלֵךְ / וַתִּמְשְׁכִי בְכִסְלֵךְ יָקָר הוֹלֵךְ

וַתִּבְעֲטִי בְחָכְמָתֵךְ / וַתָּשִׂימִי הָרַע מְגַמָּתֵךְ

וַתִּפְשְׁטִי בִגְדֵי אֲהָבוֹת / וְתִּלְבְּשִׁי תַחְתָּם בְּלוֹיֵי סְחָבוֹת

וַתְּכַבִּי אֶת נֹגַהּ שִׁמְשֵׁךְ / וַתַּשְׁלִיכִי אֶת עֲטֶרֶת רֹאשֵׁךְ

וַתַּחַרְשִׁי רָעוֹת לַבְּקָרִים / וַתָּשׂוּכִי אֶת דַּרְכֵּךְ בַּסִּירִים

וַתִּרְעִי עֲווֹנוֹת עֲדָרִים עֲדָרִים / וַתִּצְבְּרִי מִרְמוֹת חֳמָרִים חֳמָרִים

וַתַּשְׁחִיתִי אֶת נֹעַם יָפְיֵךְ וַתִּתְחַבָּלִי / וַתֵּלְכִי אַחֲרֵי הַָהֶבֶל וַתֶּהְבָּלִי

עַל מַה נִּאַצְתְּ אֱלֹהַיִךְ כָּל זֹאת / וַתְּשַׁחֲרִיהוּ הַיּוֹם בְּפָנִים עַזּוֹת

עַל מִי בָטַחַתְּ / כִּי נִסְרַחַתְּ

וַתַּרְבִּי חֲמָסֵךְ / וַתִּשְׁכְּחִי עוֹשֵׂךְ

וְלֹא אָמַרְתְּ אַיֵּה אֱלוֹהַּ עוֹשָׂי / צוֹפֶה עֲלוּמַי וּבוֹחֵן מַעֲשָׂי

מָה אֲמֻלָּה לִבָּתֵךְ / מַה טְּרִיָּה מַכָּתֵךְ

מָה אֲנוּשָׁה מַחֲלָתֵךְ / מַה צְּרוּבָה מִכְוָתֵךְ

אֵין חוֹבֵשׁ לְשִׁבְרֵךְ / אֵין צָרִי לְצִירֵךְ

עַל מִי תָנוּסִי לְעֶזְרָה / יוֹם אֱלֹהַיִךְ אֵלַיִךְ יִקְרָא

מַה תָּשִׁיבִי וְרָעָתֵךְ רַבָּה / מַה תַּעֲנִי וְחַטָּאתֵךְ כְּתוּבָה

בְּעֵט בַּרְזֶל וְעוֹפֶרֶת / עַל סְגוֹר לִבֵּךְ לְמִשְׁמֶרֶת

לָכֵן שִׁמְעִי בַת וּרְאִי וְהַטִּי אָזְנֵךְ / לְהִנָּצֵל מִפַּח פִּקְחִי נָא עֵינֵךְ

הִנָּזְרִי מִפֶּשַׁע טֶרֶם יֶאֱמָץ / וְרַחֲקִי מֵרֶשַׁע טֶרֶם יֶחְמָץ

לַחֲמִי אֶת יִצְרֵךְ בְּחַרְבוֹת תּוֹרָה / וְשׁוּבִי לְיוֹצְרֵךְ בְּדַעַת טְהוֹרָה

הִנָּחֲמִי נֶצַח עַל עֲשׂוֹת זִמָּה / וּבְחַטָּאתֵךְ שִׁלְחִי חֶרְמֵשׁ בַּקָּמָה

מַהֲרִי הִתְנַעֲרִי מֵרֹב עֲווֹנָיְכִי / וְלִפְנִי תְשׁוּבָה שִׂימִי פָנָיְכִי

וְכַבְּסִי מֵרָעָה לִבֵּךְ / וְרַחֲצִי בְנִקָּיוֹן קִרְבֵּךְ

הִטַּהֲרִי מֵעֲווֹנוֹת עֲלוּמָיְכִי / וְהִתְכַּבְּסִי מִמִּרְמוֹת זְמָמָיְכִי

פַּלְּסִי בְצֶדֶק מַעְגָּלָיְכִי / וְיַשְּׁרִי בְמִשְׁפָּט מַסְלוּלָיְכִי

וְהֵיטִיבִי בְּחֶסֶד מַעְבָּדָיְכִי / וְגַאֲלִי מִשַּׁחַת חַיָּיְכִי

וְהַמְתִּיקִי מִנֹּפֶת זְמִירוֹת חִכֵּךְ / וְקוּמִי רֹנִּי בַלַּיְלָה לְמַלְכֵּךְ

וּשְׂאִי אֵלָיו כַּפַּיִךְ / וְהִשְׁתַּחֲוִי לוֹ עַל אַפַּיִךְ

וְהִכּוֹנִי לִקְרָאתוֹ בְּזֶמֶר נָעִים / וְשַׁחֲרִי אֶת פָּנָיו בְּשִׁירֵךְ לִרְגָעִים

וְהַמְלִיכִיהוּ עָלַיִךְ בֶּאֱמוּנָה / וְיַחֲדִי אֶת זִכְרוֹ בְּדַעַת נְכוֹנָה

וְרַעֲדִי מִפַּחְדּוֹ / וְהִתְחַנְנִי נֶגְדּוֹ

וְרַעֲשִׁי מִכֹּחוֹ / וְהִתְנַפְּלִי נִכְחוֹ

זַעֲקִי אֵלָיו / וְהִתְיַצְּבִי עָלָיו

וְיַחֲלִי רְצוֹנָיו / וְהִשְׁתַּחֲוִי לְפָנָיו

וְעַל מִשְׁמַרְתֵּךְ עִמְדִי וְהִתְוַדִּי לִשְׁמוֹ / וְהַלְלִי לְכֵס מְרוֹמוֹ.
