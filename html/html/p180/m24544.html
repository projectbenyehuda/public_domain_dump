אֵל נַעֲרָץ

אֵל נַעֲרָץ בִּקְדֻשּׁוֹת שְׁלֹשָׁה

בְּסוֹד חַיָּה וְאוֹפַן וְשָׂרָף שְׁלֹשָׁה

גְּל[וּמֵ]י שֵׁשׁ פְּרוּדֵי שְׁלֹשָׁה

דְּגוּלוֹת אַרְבַּע בַּהֲלִיכוֹת שְׁלֹשָׁה

הֵן רוּצָם וְלֶכְתָּם וְהִנָּשְׂאָם שְׁלֹשָׁה

וִעוּד אַרְבָּעָה דְמוּיֵי שְׁלֹשָׁה

זִיוָם כְּחַשְׁמַל וְקָלָל וְתַרְשִׁישׁ שְׁלֹשָׁה

חֹזֶק [כַּנְפֵי]הֶם כְּקוֹלוֹת שְׁלֹשָׁה

טוֹעֲנֵי כֵס [אֵל הָרוֹכֵ]ב שְׁלֹשָׁה

יָהּ דּוֹרְשִׁים בְּאוֹתוֹת שְׁלֹשָׁה

כּוֹלְלִים שֵׁם נֶחְלָק [שְׁלֹשָׁה]

לוֹהֲגִים [שִׁ]ין דָּלֶת יוֹד שְׁלֹשָׁה

מְיַסֵּד [עוֹ]לָם בִּסְפָרִים שְׁלֹשָׁה

נוֹתֵן [שִׁשָּׁ]ה כְּנֶגֶד שְׁלֹשָׁה

סָפַר שִׁבְעָה וְהֵטִיל שְׁלֹשָׁה

עָצַר שְׁנֵים עָשָׂר בְּעֵדִים שְׁלֹשָׁה

פִּלַּג עוֹלָם וְשָׁנָה וְנֶפֶשׁ שְׁלֹשָׁה

[צִ]וָּ[ה] בְרִי[תוֹת לִכְר]וֹת שְׁלֹשָׁה

קְ[הָ]לוֹ חִלֵּק מַחְלְקוֹת שְׁלֹשָׁה

[רוֹמֵם] קַרְנָם בְּ[ר]וֹעִי[ם] שְׁלֹשָׁה

שָׁלַח עַבְדּוֹ בְאוֹתוֹת שְׁלֹשָׁה

תִּכֵּן שְׁלֹשָׁה לְיָמִים שְׁלֹשָׁה

תִּנָּה יְצוּרָיו בִּסְפָרִים שְׁלֹשָׁה.


אִמֵּץ יוֹם זֶה לְכַפֵּר שְׁלֹשָׁה

בְּפַר וָאַיִל וְשָׂעִיר שְׁלֹשָׁה

גָּדוֹל זוֹכֶה לְמִדּוֹת שְׁלֹשָׁה

דָּתוֹ לְהַפְרִישׁ אַחַר שְׁלֹשָׁה

הָגוּן לְהַרְאוֹת זְבָחִים שְׁלֹשָׁה

וְלַשְׁמִיעוֹ לַיְלָה קוֹלוֹת שְׁלֹשָׁה

זְרִיזִים יְדַשְּׁ[נוּ] וְצָלוּ שְׁלֹשָׁה

חֲזוֹתָם שַׁחַר יְמַהֵר שְׁלֹשָׁה

טָבַל וְ[סָפַ]ג וְלָבַשׁ קֹדֶשׁ שְׁלֹשָׁה

יַקְטִיר וְיֵיטִיב וְיַקְרִיב שְׁלֹשָׁה

כְּפָשְׁטוֹ כְלֵי פַז יְקַיֵּם שְׁלֹשָׁה

לְפָרוֹ יַקְרִיב וְיַתְחִיל שְׁלֹשָׁה

מְבָ[רֵךְ] וּמוֹדֶה בְמִלִּ[ים] שְׁלֹשָׁה

נָטַל שְׂעִירִים וְקַלְפֵי שְׁלֹשָׁה

סָבַל כְּלֵי פוּר וְהֵקִים שְׁלֹשָׁה

עָמַד [בַּ]בַּיִת לְהוֹדוֹת שְׁלֹשָׁה

פָּרוֹ יִשְׁחַט וְיִקַּ[ח] שְׁלֹשָׁה

צוֹעֵד לְהַ[קְטִי]ר [אֵי]לִים שְׁלֹשָׁה

קָפַץ לְ[הַ]זּוֹת הֵיכָל שְׁלֹשָׁה

רָצַח שָׂעִיר אֶחָד שְׁלֹשָׁה

שָׁחַר לְהַצְלִיף אֶמְצַע שְׁלֹשָׁה

תָּקַף וְיָצָא לְכַלּוֹת שְׁלֹשָׁה

שֶׁבַע יַזֶּה… הֵ[י]כָל שְׁלֹשָׁה

לַשָּׂעִיר יַקְרִיב וְיוֹ[דֶ]ה שְׁלֹשָׁה

מִדְבָּר יְשַׁלַּח נוֹשֵׂא שְׁלֹשָׁה

הִקְטִיר וְהִכְלִיל וְשָׂרַף שְׁלֹשָׁה

הִקְרִיב תָּמִיד וְהַ[שָּׂעִיר וְהָעוֹלָ]ה שְׁלֹשָׁה

קִדֵּם לְהֵטִיב כְּמִצְוַת שְׁלֹשָׁה

טָהוֹר נִרְאֶה בְשֵׁמוֹת שְׁלֹשָׁה

נָחוֹץ יִנְחַץ לְקַדֵּשׁ שְׁלֹשָׁה

חַסְדּוֹ נְרַנֵּן בְּגִילִים שְׁלֹשָׁה

זִמְרוֹ נְשַׁ[לֵּשׁ] בִּרְ[נָנִ]ים שְׁלֹשָׁה

קָדְשׁוֹ נְשַׁנֵּן פְּעָמִים שְׁלֹשָׁה

[קָדוֹ]שׁ [קָדוֹ]שׁ קָדוֹשׁ שְׁלֹשָׁה.


שְׁלֹשָׁה קְדֻשּׁוֹת לְאֵל אֶ[חָ]ד בָּאָ[רֶץ] מְסִבָּה / אֵל נַעֲרָץ בְּסוֹד קְדוֹשִׁים רַבָּה

שְׁמוֹ קָדוֹשׁ כְּעִנְיַן מְשׁוֹרֵר בְּרַחֲשׁוֹ / וִיבָרֵךְ כָּל בָּשָׂר שֵׁם קָדְשׁוֹ

[לָכֵן] רוּחוֹ קָדוֹשׁ כְּעִנְיַן חוֹזֶה בְּבַקְשׁוֹ / אַיֵּה הַשָּׂם בְּקִרְבּוֹ אֶת רוּחַ קָדְשׁוֹ

מְכוֹן כִּסְאוֹ קָדוֹשׁ כְּעִנְיַן מְשׁוֹרֵר בְּהַ[קְדִּישׁוֹ] / אֱלֹהִים יָשַׁב עַל כִּסֵּא קָדְשׁוֹ

הֵן זְרוֹעוֹ קָדוֹשׁ כְּעִנְיַן חוֹזֶה בְּאָרְשׁוֹ / חָשַׂף יְיָ אֶת זְרוֹעַ קָדְשׁוֹ

קָדוֹשׁ זִכְרוֹ קָדוֹשׁ כְּעִנְיַן מְשׁוֹרֵר בְּחַדְּשׁוֹ / וְהוֹדוּ לְזֵכֶר קָדְשׁוֹ

[טָהוֹר] מְעוֹנוֹ קָדוֹשׁ כְּעִנְיַן מְשׁוֹרֵר בְּ[דָרְ]שׁוֹ / אֱלֹהִים בִּמְעוֹן קָדְשׁוֹ

נ…ר יְשׁוּרוּן אֵין כָּאֵל / [וּ]בָאָרֶץ [מִי כְּעַמְּךָ יִשְׂרָאֵל].


אֱ[מ]וּנִים מְקֹרָאִים קֹדֶשׁ / כִּי כֵן נֶאֱמַר [בֶּאֱמוּנָתָם יִתְ]קַדְּשׁ[וּ] קֹדֶשׁ

בְּחִירֵי כֹהֲנִים נִקְרְאוּ קֹדֶשׁ / כִּי כֵן נִבְאַר לֶחֶם אֱלֹהֵיהֶם הֵם מַקְרִיבִים וְהָיוּ קֹדֶשׁ

גַּם לְוִיִּים מְשָׁרְתִים נִקְרְאוּ קֹדֶשׁ / כִּי כֵן נִגְלַף וַאֲחַלֵּל שָׂרֵי קֹדֶשׁ

דָּת מְלֶאכֶת הַ[מִשְׁכָּ]ן קֹדֶשׁ / כִּי כֵן נִדְרַשׁ לַעֲשׂוֹת אֶת כָּל מְלֶאכֶת עֲבוֹדַת הַקֹּדֶשׁ

הֵן אֹהֶל מוֹעֵד וְכָל כֵּלָיו קֹדֶשׁ / [כִּי כֵן נֶהְגָּה] וְקִדַּשְׁתָּ אוֹתָם וְהָיוּ קֹדֶשׁ

וּשְׁמוֹנַת בִּגְדֵי כְהֻנָּה קֹדֶשׁ / [כִּי כֵן נוֹ]דַע וְעָשִׂיתָ בִגְדֵי קֹדֶשׁ

זַךְ שֶׁמֶן הַמִּשְׁחָה קֹדֶשׁ / כִּי כֵן נִ[זְכַּר] שֶׁמֶן מִשְׁחַת קֹדֶשׁ

חֶמְדַּת קְטֹרֶת הַסַּמִּים קֹדֶשׁ / כִּי כֵן נֶחְקַק בְּמַתְכֻּנְתָּהּ לֹא תַעֲשׂוּ לָכֶם קֹדֶשׁ

טַעַם אֵיל מִלּוּאִים קָרוּי קֹדֶשׁ / כִּי כֵן נַטִּיף וְזָר לֹא יֹאכַל קֹדֶשׁ

יֶתֶר הַמִּנְחָה קָרוּי קֹדֶשׁ / כִּי כֵן נִירַשׁ וְהַנּוֹתֶרֶת מִן הַמִּנְחָה לְאַהֲרֹן וּלְבָנָיו קֹדֶשׁ

כְּלִיל הַחַטָּאת גַּם הוּא קֹדֶשׁ / כִּי כֵן נִכְ[תַּב] תִּשָּׁחֵט הַחַטָּאת לִפְנֵי יְיָ קֹדֶשׁ

לַמְּדוּ כִּי הָאָשָׁם קֹדֶשׁ / כִּי כֵן נִלְהַג וְזֹאת תּוֹרַת הָאָשָׁם קֹדֶשׁ

מַלְּלוּ כִּי הַשְּׁלָמִים קֹדֶשׁ / כִּי כֵן נִמְרַץ וְאוֹכְלָיו עֲווֹנוֹ יִשָּׂא כִּי אֶת קֹדֶשׁ

נֹעַ[ם] תְּרוּמוֹת קָדָשִׁים קֹדֶשׁ / כִּי כֵן נִנְ[אַ]ם כָּל זָכָר יֹאכַל אוֹתוֹ קֹדֶשׁ

סְפִיחַ [שָׁנָ]ה רְבִיעִית קֹדֶשׁ / כִּי כֵן נִסְדַּר תֹּאכְלוּ אֶת פִּרְיו קֹדֶשׁ

ע     דה קֹדֶשׁ / כִּי כֵן נֶעֱרַךְ מַחֲצִית הַשֶּׁקֶל בְּשֶׁקֶל הַקֹּדֶשׁ

פ    קֹדֶשׁ / כִּי כֵן נִפְ[צָ]ה וּבַיּוֹם הַשְּׁבִיעִי יִהְיֶה לָכֶם קֹדֶשׁ

צ    קֹדֶשׁ / כִּ כֵן נִצְטַוָּה בַּיּוֹם הָרִאשׁוֹן מִקְרָא קֹדֶשׁ

ק    גַם הֵם קֹדֶשׁ / כִּי כֵן נִקְרָ[א] בַּיּוֹם הַשְּׁמִינִי מִקְרָא קֹדֶשׁ

ר    קֹדֶשׁ / [כִּי כֵן נִרְשַׁם] וּקְרָאתֶם בְּעֶצֶם הַיּוֹם הַזֶּה מִקְרָא קֹדֶשׁ

שָׁ[בוּעוֹת] גַּם הֵם קֹדֶשׁ / כִּי כֵן [נִשְׁ ] עַל שְׁנֵי כְבָשִׂים קֹדֶשׁ

תְּרוּעָה גַם הִיא [קֹדֶשׁ] / [כִּי כֵן נִתְ ] זִכְרוֹן תְּרוּעָה מִקְרָא קֹדֶשׁ

קֹדֶשׁ / כִּי כֵן נֶ[אֱלַ]ף יוֹם הַכִּפּוּרִים הוּא מִקְרָא קֹדֶשׁ.
