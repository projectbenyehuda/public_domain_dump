תְּחִלַּת הַחָכְמָה

תְּחִלַּת הַחָכְמָה / וְרֵאשִׁית הָעָרְמָה / לְיַד אֵל הָרָמָה / וְיוֹצֵר הַיְצוּרִים

וְרִאשׁוֹן בִּתְחִלָּה / וְאַחְרִית כָּל תִּכְלָה / מְרוֹמָם בִּתְהִלָּה / וּמַשְׁפִּיל אַף מֵרִים

אֲשֶׁר לֹא יִתָּוֶה / בְּמָעוֹן גַּם נָוֶה / וְאֵיכָה יִשְׁתַּוֶּה / בְּשֶׁבֶת נוֹצָרִים?

וְכָל דַּעַת נוֹצָר / לְהַגְבִּיל אֵל קָצַר / וְלִבָּם כַּיָּם צַר / וְדֵעוֹת כִּשְׁקָרִים

אֲשֶׁר מֵאָז קָדַם / וְהַכֹּל סָר נִרְדָּם / וְעַד לֹא שָׂם אָדָם / וְלֹא יָצַר הָרִים

וְשָׂם אַרְבַּע יָדוֹת / נְכוֹנוֹת מוּעָדוֹת / וְלָהֵנָּה סוֹדוֹת / צְפוּנִים וּסְתוּרִים

וְצוּרָם אַבִּירָם / אֱנוֹשׁ מֵהֶם קָרַם / וְיַחַד הֶחְבִּירָם – / וְאֵינָם נֶחְבָּרִים

וְיָזַם וְיַשְׁלִים / עֲשׂוֹת הַגַּלְגַּלִּים / כְּתַבְנִית אֹהָלִים / כְּמַעְשֵׂה סַפִּירִים

פְּלָאָיו לָדַעַת / וְאוֹר מֵאַרְבַּעַת / שְׁחָקִים לָגַעַת / לְעֵינֵי הַיְצוּרִים

וְהִצִּיג כֹּחַ בָּם / כְּפִי מַתַּת רַבָּם / לְבַל יִמּוֹט רַבָּם / לְדוֹרֵי הַדּוֹרִים

וְשָׁת בָּהֶם תֵּבֵל / תְּלוּיָה בַחֶבֶל / וְשָׁת תִּלָּהּ סֵבֶל / בְּמֹאזְנֵי מֵישָׁרִים

וּבִמְרוּצַת מֶרֶץ / יְרוּצוּן בַּחֶרֶץ / לְהַעְמִיד תֵּל אֶרֶץ / יְצוּקָה מֵעֲפָרִים

וְרוּחַ עוֹשֵׂיהֶם / תְּרַחֵף בֵּינֵיהֶם / וְיֹאחֵז כַּנְפֵיהֶם / וְיִהְיוּ נִנְעָרִים

וּפָרָשׂ אֶת צִלָּם / עֲלֵי כָל הָעוֹלָם / וְחָצַץ אֶת חֵילָם / לְמַחְנוֹת וּסְדָרִים

וְהוּא הוּא הַתּוֹאֵם / שְׁנַיִם אָב וָאֵם / בְּשֵׁמוֹת יִקְרָאֵם / וְאֵינָם נֶעְדָּרִים

וְחוֹזֵי כוֹכָבִים / לְהוֹדוֹת סָרָבִים / וְאֵינָם קַשָּׁבִים / וּמִמּוּסָר סָרִים

וְכִחְדוּ אֵל שַׁדַּי / בְּאָמְרָם כִּי יֵשׁ דַּי / בְּאַרְבַּע מוֹסָדַי / וְהֵם הָעִקָּרִים

לְהַמְצִיא הַנֶּעְדָּר / בְּאֵל חַי הַנֶּאְדָּר – / וְהוּא קוֹץ אַף דַּרְדַּר / לְרַגְלֵי הָהוֹבְרִים

הֲלֹא הֵנָּה אוֹתוֹת / לְרַגְלָם כִּרְשָׁתוֹת / וְאַל יִהְיוּ שְׂפָתוֹת / עֲלֵי אֵל מַגְבִּירִים

הֲיוּכַל תֵּת צוּרָה / אֲשֶׁר עוֹד לֹא נִבְרָא / אֲבָל הָאֵל קָרָא / וְגִלָּה נִסְתָּרִים

וְחִדֵּשׁ חֹם יִפְצַר / עֲלֵי קִרְיַת מִבְצָר / וְרוּחַ מֵאוֹצָר / וְקָרָה מִמְּזָרִים

בְּשׂוּם הָאֵד נִפְרָד / וּמֵאֶרֶץ נִטְרָד / וְיָשׁוּב בַּמּוֹרָד / עֲלוֹת הַקִּיטוֹרִים

וְעוֹד אוֹת כִּי נָטַר / אֱמֶת, כִּי לֹא אָטַר / עֲנָנִים מִמָּטָר / וְהָיוּ נֶעְצָרִים

אֲשֶׁר הֵם נֶחָמָה / וְאַף הֵם לִנְקָמָה / וּבָהֶם לִמְהוּמָה / וּמוּסָר לַסּוֹרְרִים

וּבָם יָדִין עַמִּים / לְקֶצֶף וּלְרַחְמִים / וְאֹכֶל מַטְעַמִּים / יְשַׁו לַמַּכְבִּירִים

וְהָעֵת יִמָּנְעוּ / רְבִיבִים – יִכָּנְעוּ / יְצוּרִים, כִּי פָשְׁעוּ / וְהָלְכוּ נִקְדָּרִים

וְעָרְבָה כָּל שִׂמְחָה / בְּיָגוֹן וַאְנָחָה / וְשִׂיחָה וּצְוָחָה / וּמִסְפַּד תַּמְרוּרִים

וּמַשְׁקֶה כּוֹס רַעַל / בְּקֶצֶף וּבְגֹעַל / וְיָגוֹן נִקְשַׁר עַל / לְבָבוֹת נִתָּרִים

וְאִישׁ כַּשּׁוֹר גּוֹעֶה / וְהַכֹּל סָר תּוֹעֶה / וְלֹא מָצְאוּ מִרְעֶה / לְצֶמֶד אִכָּרִים

וְאַנְשֵׁי תוּשִׁיָּה / יְשִׂיחוּן בִּשְׁאִיָּה / וְקוֹל כִּנְהִי נִהְיָה / לְשָׂבִים וּנְעָרִים

וְעָרְגוּ אַיָּלִים / וְעָטְפוּ עוֹלָלִים / וְשָׁבוּ עַוָּלִים / תְּשׁוּבַת הַבָּרִים

וְעָלְתָה שַׁוְעָתָם / לְאֵל רָאָה אוֹתָם / וְשָׁמַע אֶנְקָתָם / וְהֶרְאָה נִסְתָּרִים

וְהַחֶסֶד לָבַשׁ / וְהֶעָווֹן כָּבַשׁ / וְחָתַל אַף חָבַשׁ / מְזוֹרִים לִמְזוֹרִים

וְנָשָׂא הַזַּעַם / וְנָפַל הָרַעַם / וּבָא בֶּן אֲבִינֹעַם / בְּמַיִם כַּבִּירִים

וְקוֹלוֹ עֹז קוֹלוֹת / יְחוֹלֵל אַיָּלוֹת / וְלָנוּ הַנִּגְלוֹת / וְלוֹ הַנִּסְתָּרִים

וּמִלֵּא הֶעָבִים / בְּמַלְקוֹשׁ וּרְבִיבִים / וְהֵרִיק לַשָּׁבִים / בְּדַיֵּי מַחְסוֹרִים

וּבִלַּע הַמָּוֶת / בְּאֶרֶץ צַלְמָוֶת / וְלִמְאֹד עַז מָוֶת / בְּעֵינֵי נִקְבָּרִים

וְשִׂמַּח כָּל יוֹגֵב / וְכָל אַנְשֵׁי רֶגֶב, / וְאֶרֶץ הַנֶּגֶב / יְרַוֶּה בִיאוֹרִים

וְשָׁקַט לֵב הוֹמֶה / וְלֵב רָעֵב, צָמֵא / וּבַמֶּה יֵשׁ, בַּמֶּה / לְהַקְרִיב אֶשְׁכָּרִים?

וּמַה לִּי קָרְבָּנוֹת / כְּנֹעַם הֶגְיוֹנוֹת / וְשִׁירֵי שִׁגְיוֹנוֹת / בְּפִי רֹאשׁ הַשָּׂרִים

שְׁמוּאֵל, הַמּוֹרֶה / פְתָאִים, הַפּוֹרֶה / תְבוּנוֹת, הַיּוֹרֶה / חֲנִיתָיו כַּמּוֹרִים

אֲשֶׁר בּוֹ נִכְבָּדוֹת / נְכוֹנוֹת מוּעָדוֹת / וְלוֹ עֶשֶׂר יָדוֹת / עֲלֵי כָל הַמּוֹרִים

וְכִשְׁמוּאֵל הָיָה / בְּדָת וּפְלִילִיָּה / וְאַנְשֵׁי תוּשִׁיָּה / לְמוּלוֹ נִמְהָרִים

וּפִתַּח גַּם הִשְׁרָה / אֲגֻדּוֹת הַמִּקְרָא / וּבָדַק בַּתּוֹרָה / מְלֵאִים וַחְסֵרִים

וְהֵשִׁיב לִנְכוֹחוֹת / קְמוּצוֹת וּפְתוּחוֹת / כְּמוֹרִיד עַל לוּחוֹת / עֲשֶׂרֶת הַדְּבָרִים

וְאָסַף נַעְרָצוֹת / בְּדַעַת מוֹעֵצוֹת / עֲלֵי כָל הָאֲרָצוֹת / דְּבָרָיו נִזְכָּרִים

וּמִדְבָּרוֹ לַהַב / וּמִכְתָּבוֹ רַהַב / כְּעֶדְיֵי הַזָּהָב / יְשׁוֹרֵר בַּשִּׁירִים

יְגַלֶּה עִנְיָנִים / צְפוּנִים וּטְמוּנִים / וְהָיוּ נִבְחָנִים / כְּכֶסֶף בַּכּוּרִים

וְהִתִּיר נֶאְסָרִים / וְאָסַף נִפְזָרִים / וְהוֹרִיד נִבְצָרִים / וּפָתַח נִסְגָּרִים.
