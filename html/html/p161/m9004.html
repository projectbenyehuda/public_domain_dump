אֶל ר' דָּוִד נַרְבּוֹנִי (ב)

אָבִי אָבִי עֵץ הַדַּעַת, מָלֵא רוּחַ בְּחָכְמָה בִתְבוּנָה וּבְדַעַת, רֹאשׁ הַמְּדַבְּרִים, אַבִּיר הָרֹעִים, גְּבִיר הַמַּשְׂכִּילִים, וּמְזַכֶּה אֶת הָרַבִּים, מוֹכִיחַ לְמוֹכִיחִים, מוֹרֵה הַבּוֹטְחִים וְהַמַּבְטִיחִים, יִבְטַח בַּה‘, גּוֹל עַל ד’, דּוֹם לַד‘, קַוֵּה אֶל ד’, הַשְׁלֵךְ עַל ד‘, וְהִתְעַנֵּג עַל ד’, כִּי יֵשׁ לִפְעֻלָּתְךּ שָׂכָר וּלְתִקְוָתְךָ תַּגְמוּל, וּלְבִטְחוֹנְך, תַּשְׁלוּם, וּלְלֵילְךָ שַׁחַר, וּלְאָפְלְךָ צַחַר, וּלְעַרְבְּךָ בֹּקֶר, וּלְעָצְבְּךְ חֵקֶר, וּלְצִירְךָ תַּכְלִית, וּלְצוּקָתְךּ אַחֲרִית, וּלְמַחְלָתְךָ קֵצֶה, וּלְתַאֲלָתְךָ קֶצֶב. עַתָּה יָקוּם יֹאמַר ד‘, עַתָּה יָעִיר עָלֶיךָ, וְיִהְיֶה בְכִסְלֶךָ, וְיָרִים דְּגָלֶיךָ, וְיָאֵר פָּנָיו אֵלֶיךָ, וְיַעֲמֵד בַּמֶּרְחָב רַגְלֶיךָ, וְיָקֵל מֵעָלֶיךָ, וְשָׁלְמוּ יְמֵי אֶבְלְךָ, כְּדִכְתִיב לֹא יָבֹא עוֹד שִׁמְשְׁךָ וגו’.

חָזוּת קָשָׁה הֻגַּד לִי וגו‘. הַשּׁוֹמֵעַ יִשְׁמַע וְהֶחָדֵל יֶחְדַּל וגו’. שָׁמַעְתִּי וְנֶאֱנַחְתִּי, לֹא שָׁקַטְתִּי וְלֹא נַחְתִּי, וְנַפְשִׁי הוֹמִיָּה, קוֹדֶרֶת וּבוֹכִיָּה, סֹעֶרֶת כָּאֳנִיָּה, כִּי הַחָכְמָה יוֹשֶׁבֶת שׁוֹמֵמָה, וְהַמְּזִימָּה, – כְּרָחֵל נֶאֱלָמָה, וּמַלְכֵי הַתּוֹרָה נִמְהֲרוּ, וְשַׂק חָגְרוּ, וְלָאָרֶץ סָחֲרוּ, וְכֹחַ לֹא עָצְרוּ, לְמַעַן הַיָּמִים, אֲשֶׁר קָמוּ כְקָמִים, וּפָרְשׂוּ חֲרָמִים, וְלָכְדוּ תְמִימִים, וְרִחֲקוּ נְעִימִים, וְקֵרְבוּ זְעָמִים, וְהִשְׁפִּילוּ רָמִים, וְהִקְדִּירוּ מְרוֹמִים, וּזְמַן שָׁלַח יָד וּפִלַּח כָּבֵד, וְהֵשַׂח – כְּרוּב מִמְשַׁח, וְהִגִּיעַ לֶעָבִים, וַיַּפֵּל אַרְצָה מִן הַצָּבָא וּמִן הַכּו­ֹכָבִים, וּבָא עַד הַצְּבִי, וַיִּשְׁבְּ מִמֶּנוּ שְׁבִי, בָּאָה שְׁמוּעָה, וְאֶרֶץ הִתְרוֹעֲעָה, וְעַיִן דָּמְעָה, וְנַפְשִׁי יָרְעָה, וּמַמְרוֹרִים שָׂבְעָה, וְרֹאשׁ פָּרְעָה, וּמַדָּהּ קָרְעָה, וּמִסְפֵּד קָבְעָה, לְהֵילֵל נָפַל, וּמָאוֹר אָפֵל, וְגָבֹהַּ שָׁפֵל, וְהַמֶּלֶךְ הִתְאַבֵּל, כִּי חָרַב סִפּוֹ, וְדָעַךְ רִשְׁפּוֹ, וְחָשְׁכוּ כוֹכְבֵי נִשְׁפּוֹ, הַיּוֹם הַהוּא יִקָּחֵהוּ אֹפֶל, וְיִשְׁכֹּן עָלָיו עֲרָפֶל, וְתִשְׁכֹּן עָלָיו עֲנָנָה, וְאַל יֵחַד בִּימֵי שָׁנָה. נְדִיבַי אַל תָּאִיצוּ לְנַחֲמוֹ, כִּי מִכָּל עֲבָרָיו, בָּאוּ מְגוֹרָיו, וּבְכָל אֲשֶׁר יָצָא, מְנוּחָה לֹא מָצָא, וְאֵין מְתוֹם בִּלְבָבוֹ, וְאֵין מַכְאוֹב כְּמַכְאוֹבוֹ, אֲבֵלוּת – בַּגָּלוּת, וְתַאֲנִיָּה – בְּאֶרֶץ נָכְרִיָּה, וְיִסּוּרִים חֲזָקִים, בְּאִיִּים רְחוֹקִים, הוּרַד מִמַּצַּבְתּוֹ, וְנִפְרַד מִמַּחְבַּרְתּוֹ, וְשֻׁדַּד מֵאֶרֶץ מוֹלַדְתּוֹ, וְנִמְשְׁכָה תוֹחַלְתּוֹ, וְנֶהֶפְכָה מַחֲשַׁבְתּוֹ, וְנֶעֶקְשׁוּ דְרָכָיו, וְנוֹקְשׁוּ מַהֲלָכָיו, וּמַעֲשָׂיו הִפְרִיעַ, וּבַגּוֹיִם לֹא הִרְגִּיעַ.

זֹאת וְעוֹד אַחֶרֶת, הִיא הַחוֹדֶרֶת, הַמָּרָה וְהַנִּמְהֶרֶת, כִּי מִסְּבִיבָיו, קָמוּ מַחֲרִיבָיו, וּבְתוֹךְ קְרָבָיו, אָרְבוּ מַאֲרָבָיו, וּמִבֵּית מִבְטֶחָיו, יָצְאוּ מַדִּיחָיו, שׁוּעָלִים קְטַנִּים, לְצֶלַע נְכוֹנִים. עוֹלְלוּ כַרְמוֹ, וְעוֹרְרוּ זַעְמוֹ, וְגָרְמוּ עַצְמוֹ, וּמָצְצוּ דָמוֹ, וּמֵרֲרוּ נְעִימוֹ, ד' נָתַן וַד' לָקַח, יְהִי שֵׁם ד' מְבֹרָךְ.

אָבִי אָבִי רֶכֶב יִשְׂרָאֵל, שַׂר וְגָדוֹל בְּיִשְׂרָאֵל, אֵלֶיךָ עֵינֵי יִשְׂרָאֵל, לָדַעַת מַה יַּעֲשֶׂה יִשְׂרָאֵל, אִם קָטוֹן אַתָּה בְעֵינֶיךָ, גָּדוֹל אַתָּה בְעִנְיָנֶיךָ, וּכְצוּר אֵיתָן מַחֲנֶיךָ, וּבִמְרוֹם עֹז מִשְׁכָּנֶךָ, רַבִּים אֲשֶׁר יִסַּרְתָּ, וַעֲצוּמִים אֲשֶׁר הוֹכַחְתָּ, וְיָדַיִם רָפוֹת חִזַּקְתָּ, וּבִרְכַּיִם כּשְׁלוֹת אִמַּצְתָּ, וְעַתָּה מַדוּעַ נֶחְשַׁלְתָּ, וּמַה לְּךָ כִּי נִבְהַלְתָּ, הֲלֵב אַרְיֵה יִמַּס, אִם נֶפֶשׁ כְּפִיר תִּרְפֶּה, אוֹ הַר תָּלוּל יִפּוֹל, וְצוּר חָזָק יִבּוֹל, הָפֵץ עֶבְרוֹת אַפֶּךָ, וְהָסֵר זַעַמְךָ וְקִצְפֶּךָ, וּגְעַר בְּיָמִים וְיִתְחַטְּאוּ, וְהָרִיע עַל הָרָעוֹת וְיִדַּכְּאוּ, צִדְקָתְךָ תְּפִיצֵם, וְכִכְלִי יוֹצֵר תְּנַפְּצֵם, אֵיךְ תִּתְרַפֶּה בְּיוֹם צָרָה, וְתַשְׁכִּים וְתַעֲרִיב בְּנֶפֶשׁ מָרָה, רַב לְךָ כֶּתֶר וּלְמַלְאוֹת, לְהַסְתִּיר וּלְגַלּוֹת, לְרַפְּאוֹת וּלְחַלּוֹת, לְהָחֵל וּלְכַלּוֹת, וְהַכֹּל בִּכְתָב מִיַּד ד‘, לְמַלְאֹת אֶת דְּבַר ד’, וְרָזֵי הַיָּמִים סְתוּמִים וַחֲתוּמִים, מְכֻסִּים יוֹצֵא, זֶה עוֹשֶׂה וְזֶה מְעַשֶּׂה, זֶה נוֹשֵׂא וְזֶה מְנַשֵּׂא, אֵין לְהִתְהַלֵּל בְּיוֹם מָחָר, וְאֵין לְהִתְעַצֵּב לְצָרַת מָחָר, כִּי לֹא נֵדַע מַה יֵּלֶד יוֹם, פְּלִיאָה דַעַת מִמֶּנּוּ, רָחוֹק יָבוֹא הַנֹּפֶש, וּבְהִתְאַחֵר הַתִּקְוָה, תְּמַהֵר הַחֶדְוָה, וּבְאֶפֶס עָצוּר, יֶהֱמוּ רַחֲמֵי הַצּוּר. נוֹאָשׁ אַב הָמוֹן מִבֶּן חַיִל, וַיַּרְא וְהִנֵּה אַיִל, נוֹאָשׁ מִבְּנוֹ תָּמִים אֶזְרָחָי, וְיַגִּידוּ לוֹ עוֹד יוֹסֵף חָי, גָּלָה עָנָו מֵאַרְצוֹ וַיִתְיָאֵשׁ, וַיַּרְא וְהִנֵּה הַסְּנֶה בוֹעֵר בָּאֵשׁ, גְּדוֹלִים מַעֲשֵׂי ד‘, כִּי אֵל דֵּעוֹת ד’, כִּי לֹא יִזְנַח לְעוֹלָם ד‘. הֵן אֵלֶּה קְצוֹת דְּרָכָיו וְכָהֵנָּה רַבּוֹת עִמּוֹ, אֵין צוּר בִּלְעָדוֹ, וְאֵין לְהִתְיָאֵשׁ מֵאֲשֶׁר בְּיָדוֹ, כִּי רַחֲמָיו מְרֻבִּים וַחֲסָדָיו עֲצוּמִים, וְיָדוֹ רְחָבָה, וְחֶמְלָתוֹ קְצוּבָה, יוֹכִיחַ אֶת אֹהֲבָיו, וְיִפְקֹד עַל קְרֹבָיו, וִידַקְדֵּק עִם אֲשֶׁר סְבִיבָיו, כְּדִכְתִיב וּסְבִיבָיו נִשְׂעֲרָה מְאֹד, לְהַנְעִים חֶלְקָם, וּלְהַשְׁלִים חֻקָּם, וּלְמַלְאוֹת שִׂפְקָם, וְלָתֵת מִשְׁאָלָם, וּלְשַׁלֵּם גְּמוּלָם, וּלְהַשְׂגִּיא פָּעֳלָם, לְהַרְבּוֹת אוֹרָם וּלְהֵיטִיב אַחֲרִיתָם, כְּדִכְתִיב וְהָיָה רֵאשִׁיתְךָ מִצְעָר וגו’.
