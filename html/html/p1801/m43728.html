יְשִׂיחוּנִי סְעִפַּי

יְשִׂיחוּנִי סְעִפַּי מַאֲמָרִים / וְאַט יִתְלַחֲשׁוּ לִי בַדְּבָרִים

וְנִיבָם צַח וּמִלָּה אֵין בְּפִיהֶם / בְּעֵינֵיהֶם לְבַד קוֹרְצִים וּמוֹרִים

יְרֵאִים פֶּן אֲנָשִׁים יֵדְעוּן סוֹד / וְ״הֶאָח!״ יֹאמְרוּ זָרִים וְעָרִים

וְזֶה נִיבָם בְּכָל עוֹנָה וְכָל עֵת / וְכָל רֶגַע, עֲרָבִים אוֹ שְׁחָרִים:

״הֲכָכָה יַעֲשֶׂה רֵעַ לְרֵעוֹ? / הֲיֵשׁ כָּכָה בְּמִשְׁפַּט הַחֲבֵרִים?

הֲתִהְיֶה אַהֲבָה עַזָּה עֲזוּבָה / נְטוּשָׁה עַל פְנֵי מִדְבַּר חֲרֵרִים?

וּפוֹרִיָּה, אֱמֶת, הָיְתָה לְדוֹדָהּ / וּמֵאָז שִׁלְּחָה עַד יָם קְצִירִים

מְגָדֶיהָ כְצִיצֶיהָ וּפִרְחָהּ / תְּבַשֵּׁל בַּעֲמָקִים אוֹ בְהָרִים״.

וְכָזוֹ דִבְּרוּ אֶחָד אֱלֵי חַד / בְּמִלִּים יָקְרוּ מִפָּז וְדָרִים

וְאָמַרְתִּי: אֲנִי אֶצְפֹּן תְנוּאוֹת / וְלָעַד יִהְיוּ בַלֵּב שְׁמוּרִים

עֲדֵי כִי נִצְבְּאוּ עָלַי אֲהָבִים / וְרוֹבִים בַּחֲצָצֵיהֶם וּמוֹרִים

צְדִיָּה פִלְּחוּ בָם אֶת כְּבֵדִי / וּמִדַּם לֵב לְבַד רָוִים וּמוֹרִים

לְכָה נִשָּׁפְטָה יַחַד, וְנָשִׂים / עָלֵינוּ יוֹדְעֵי מִשְׁפָּט וּמוֹרִים

מְבִינֵי דִין וְדָת חֵשֶׁק וּמִשְׁפַּט / אֲהָבִים מִיְּמֵי אָבוֹת וְהוֹרִים

וְנַקְרִיב עַצְמוּתֵנוּ לְנֶגְדָּם / לְבִלְתִּי נִהְיֶה סוֹרְרִים וּמוֹרִים

אֲבָל זַעְתִּי לְבַל נִהְיֶה לְמִלָּה / בְּפִי הַיְקוּם, זְקֵנִים עִם נְעָרִים

וְנִעַרְתִּי לְזֹאת חָצְנִי, וְכַפַּי / לְיִרְאַת בּוּז, אֱמֶת, שַׂמְתִּים נְעוּרִים

לְזֹאת אֶדֹּם, וּפִי אֶבְלֹם בְּמִתְגִים / וְשִׂפְתוֹתַי אֲשִׂימֵם נֶאֱסָרִים

וְאֶבְחַר מִשְׁפְּטֵי פִיךָ לְנַפְשִׁי / הֲכִי לָךְ מִשְׁפְּטֵי צֶדֶק יְשָׁרִים

וְשָׁלַחְתִּי אֲנִי זֹאת הַמְּגִלָּה / לְךָ תָעוּף כְּעַל כַּנְפֵי נְשָׁרִים

תְּמַלֵּא תוֹכְחוֹת פִּיהָ, וְאִם הִיא / מְנַגֶּנֶת בְקוֹל זִמְרָה וְשִׁירִים

וְאִם רַכּוֹת לְךָ תַעְנֶה, רְאֵה כָל / דְּבָרֶיהָ חֲזָקִים מִשְּׁמִירִים

תְּבוּנוֹת מַעֲנֶיהָ מֹר אֲהָלוֹת / וְעֵץ בֹּשֶׂם נְרָדִים עִם כְּפָרִים

וּמוֹסֶכֶת בְּרִיב, יֵין אַהֲבָה לָךְ / בְּטוּרֵי שִׁיר וּבָתֵי הַזְּמִירִים

כְּסִירִים בַּעֲצֵי בֹשֶׂם צְמֵחִים / וְצוּף וּדְבַשׁ בְּיַעְרַת הַדְּבוֹרִים

וּמַגֶּדֶת אֲשֶׁר הָיָה לְפָנִים / בְּצַחוֹת שִׁיר וְכִמְאוֹרוֹת מְאִירִים

וְזֹאת הִיא הַתְּלוּנָה הַצְּפוּנָה / תִּצְעַק מַר וְתִשָּׂא קוֹל וְתָרִים:

הֲמִלִּכְתֹּב אֵלַי יָדָךְ כְּבֵדִים / וְעָבֶיךָ כְהַיּוֹם נֶעֱצָרִים?

וּמֵאָז עָרְפוּ גֶשֶׁם נְדָבוֹת / עֲלֵי אַרְצָם, רְבִיבִים אוֹ שְׂעִירִים

וּמִטְרוֹת חֵן בְּלִי רַעַם וּבָרָק / וּמֵימֵיהֶם מְאֹד זַכִּים וּבָרִים

לְתוֹכַחַת מְגֻלָּה קוּם אֱזָר־נָא / חֲלָצֶיךָ, כְּגֶבֶר שִׂים חֲגוּרִים

קְרָא אַתְּ וַאֲנִי כֵן אֶעֱנֶךָ / וְאֶשְׁאַל וַעֲנֵה מִלִּים בְּרוּרִים

שְׁאַל־נָא בַעֲדָהּ אָרְחוֹת דְּדָנִים / וֶאֱדֹם, אָהֳלֵי מוֹאָב וְהַגְרִים

וְיִשְׁמָעֵאל, וְנָא חַל הָעֲרָבִים / וְקֵדָר הַשְּׁכוּנִים בַּיְּעָרִים

בְּנֵי יֶפֶת וְגַם כָּל מִשְׁפְּחוֹת שֵׁם / וְתוֹלְדוֹת חָם, צְחוֹרִים אוֹ שְׁחוֹרִים,

שְׁמַע אֵלַי וְקַבֵּל מוֹעֲצוֹתַי / הֲכִי לָךְ, לֹא לְזָרִים, הֵם מְסוּרִים

וְחֵן חֵן תַּעֲשֶׂה עִמִּי וְחֶסֶד / שְׁלַח קַל חִישׁ לְאֵלֶּה הַסְּפָרִים

קְנֵה לִי בַחֲמִשָּׁה אוֹ בְשִׁשָּׁה / זְהוּבִים אִם תְּרַבֶּה בַמְּחִירִים

תְּכֻפַּר הַמְּשׁוּבָה בַתְּשׁוּבָה / וְיִיטַב לִי מְקוֹם אֵילִים וּפָרִים

הֲשִׁיבֵנִי וְאָשׁוּבָה וְשׁוּבָה / וְנָשִׁיב אַהֲבָה לִימֵי נְעוּרִים

וְרֵעַ יַעֲנֶה רֵעוֹ בְרִנָּה / יְשִׂיחוּנִי סְעִפַּי מַאֲמָרִים

וְאָמְרוּ: יֵשׁ הֲנָחוֹת בַּאֲנָחוֹת / נְשִׂיאֵי רוֹזְנִים וּבְנֵי נְזִירִים

וְהֵן לוּ הַדְּאָגוֹת אִישׁ יְשִׂיחֵם / אֲזַי לֹא יִהְיוּ לוֹ עוֹד מְצֵרִים

לְפִצְעֵי אַהֲבָה הַעְלֵה אֲרוּכָה / וּמַרְפֵּא צַו וּמָזוֹר לַמְּזוֹרִים

עֲשֵׂה חֶפְצִי וְאֶחְדַּל מִמְּרִיבָה / וְיֶחְדַּל רִיב וּמַצּוֹת עִם מְרוֹרִים

פְּשָׁעַי הַנְּקַלִּים לֹא נְקַלִּים / וְנִכְבָּדִים שְׁגָגוֹתָם חֲמוּרִים

וּמֵאָז כֵּן וְעַתָּה מַחְשְׁבוֹתַי / בְּצֶדֶק וֶאֱמֶת וּבְלִי שְׁקָרִים

וּבִקְצֵה הַיְּרִיעָה שָׁם חֲקַקְתִּים / בְּעֵט בַּרְזֶל, וְלָךְ שַׂמְתִּים סְדוּרִים.
