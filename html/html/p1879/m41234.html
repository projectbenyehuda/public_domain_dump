צְבִי הַחֵן

זה השיר אשר שלח להחכם השלם כהר יעקב אלישר שליח חברון תוב״ב ש׳ התקכ״ג

סימן: צאלח


צְבִי הַחֵן קְחָה שִׁירָה שְׁקוּלָה / בְּמֹאזְנֵי בִין וּמִלִּבִּי אֲצוּלָה

אֲכַבֶּה אֶת שְׁבִיב חִשְׁקִי בְשִׁירִים / בְּטֶרֶם אֶחֱלֶה אוּרַד שְׁאוֹלָה

לְךָ הִשְׁתּוֹקְקָה נַפְשִׁי וְחָשְׁקָה / דְּבָרְךָ לָהּ יְרַפֵּא עַד נְקַלָּה

חָקַרְתִּי כוֹכְבֵי שַׁחַק וְהִנֵּה / בְלִי נֹגַהּ וְצֶדֶק גַּם בְּתוּלָה

וְשָׁאַלְתִּי וּמִי עָלָה שְׁמֵי רוֹם / וּמִי גָזַל וְהֵשִׁיב הַגְּזֵלָה

הֱשִׁיבוּנִי אֲשֶׁר הָלְכוּ לְהָאִיר / פְּנֵי דוֹדְךָ וְהֵמָּה לוֹ סְגֻלָּה

וְעֵת הֵאִיר וְהוֹפִיעַ כְּבוֹדְךָ / לְאַרְצֵנוּ וְהִיא נָתְנָה יְבוּלָהּ

וְעֵת הֵכִין לְדֶרֶךְ אֶת פְּעָמָיו / עֲצוּבָה וַעֲרָפֶל לָהּ חֲתֻלָּה

וּמַה מָּתְקוּ אֲמָרֶיךָ וְדִבְרֵי / מְלִיצָתְךָ בְּעָמְדָךְ תּוֹךְ קְהִלָּה

וְאִם אָבוֹא לְהַקִּישׁ שִׂכְלְךָ דּוֹד / וְאֵיךְ תִּדְמֶה לְעִלָּה הָעֲלוּלָה

וְחֵן תֹּאַר יְדִידִי הוּא בְעֶצֶם / וְחֵן זוּלָתְךָ דֶרֶךְ שְׁאֵלָה

וְאִם יִהְיֶה זְמַנֵּנוּ מְנוֹרָה / כְּבוֹדְךָ הוּא עֲלֵי רֹאשָׁהּ לְגֻלָּהּ

וְאִם דּוֹדִי כְקֹטֶב הַצְּפוֹנִי / לְךָ נַפְשִׁי תְסַבֵּב כַּעֲגָלָה

וּבָחַרְתִּי לְהִסְתּוֹפֵף בְצִלָּךְ / וְאִם תֵּימִין אֲנִי לֹא אַשְׂמְאִילָה

קְחָה מֶנִּי תְשׁוּרַת מַעֲנֵה פִי / כְּשַׁי הוֹן רַב בְּלִי חֵקֶר וְתִכְלָה

וְאִם נִמְצָא שְׁגִיאַת חֵטְא בְּשִׁירִי / בְּפִי, מִמָּךְ אֲבַקֵּשׁ הַמְּחִילָה

וְעוֹד מָה אַאֲרִיךְ לָשׁוֹן, לְטוֹב לִי / שְׁתִיקָתִי וְדוּמִיָּה תְהִלָּה

אֱלֹהַי אֶשְׁאֲלָה מִמָּךְ תְּחִנָּה / תְּקֻבַּל נָא תְפִלָּתִי כְעוֹלָה

וְתַשְׁלִים אֶת רְצוֹן צִירְךָ כְּחֶפְצוֹ / וְדַרְכּוֹ תִהְיֶה דֶרֶךְ סְלוּלָה

וְיָבֹא יַעֲקֹב שָׁלֵם לְחֶבְרוֹן / וְעַל רֹאשׁוֹ עֲטֶרֶת הוֹד גְּדוֹלָה

וְיַחְזֹר אֶל נְוֵה קָדְשׁוֹ בְּכָבוֹד / בְּהוֹן עָתֵק וְרֹב חֶדְוָה וְגִילָה.
