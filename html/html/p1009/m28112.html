אֲהָהּ עַל מִשְׁפְּטֵי תוֹרָה נְבוּקִים

הגנה על הרמב״ם כנגד מבקרו הראב״ד


אֲהָהּ עַל מִשְׁפְּטֵי תוֹרָה נְבוּקִים / כְּסַפִּירִים רְאִי מוּצָק חֲזָקִים

אֲהָהּ שָׁבוּ כְּלַעְנָה מַמְּרוֹרִים / אֲשֶׁר הָיוּ כְּנֹפֶת צוּף מְתוּקִים

אֲהָהּ לִי עַל שְׂדֵי חֶמֶד גְּאוּלִים / וְהֵן הֵנָּה בְּכָל פֵּיוֹת מְרוּקִים

יְסוּדָתָם בְּהַר קֹדֶשׁ בְּבִנְיַן / גְּאוֹן הַדּוֹר וְגִבּוֹר בָּעֲנָקִים

הֲלֹא הוּא רָם בְּמַזָּלוֹ עֲלֵי כָל / גְּאוֹנִים עָצְרוּ מִלִּים דְּחוּקִים

בְּנוֹ מַיְמוּן מְאֻשָּׁר הוּא לְעֵין כֹּל / וְיִסַּד עֹז בְּפִי עוֹלְלִים וְיוֹנְקִים

וּמִשְׁכָּן בִּין קְרָשָׁיו עִם בְּרִיחָיו / הֲלֹא הִצִּיב יְרִיעוֹתָיו וְהֵקִים

וְהַדִּינִים לְמִינֵיהֶם מְשָׁכָם / בְּמַתְכֻּנְתָּם עֲלֵי סִפְרוֹ חֲקוּקִים

וְנִשְׁמַע מִכְּנַף אֶרֶץ זְמִירוֹת / וְאַפְסֵי יָם קְרוֹבִים עִם רְחוֹקִים

יְשַׁנֵּנוּ תְּהִלּוֹתָיו וְשִׂיחָיו / בְּכָל מָקוֹם רְחוֹבוֹת עִם שְׁוָקִים

וְהִסְכִּימוּ בְּנֵי עֶלְיוֹן לְקוֹלוֹ / בְּחֶזְקַת יָד יְעִידוּן בַּשְּׁחָקִים

אֲבָל אִישׁ קָם בְּעִיּוּנוֹ לְנִכְחוֹ / בְּמִלְחַמְתּוֹ וְסַרְעַפָּיו עֲסוּקִים

וְשָׂם חֶרֶב בְּפִיו לוּטָה וּמְרוּטָה / וְשִׂיחָיו הֵם רְמָחָיו כַּבְּרָקִים

שְׁמוֹ רַאבַּ״ד וְכִמְעַט קָט וְאִבַּד / לְעַמּוּדִים בְּמַעְבֵה בִין יְצוּקִים

וְלוּלֵי רַחֲמֵי אֵל קִדְּמוּהוּ / בְּמִגְדַּל עֹז לְהַצִּיל הָעֲשׁוּקִים

וְהַמַּגִּיד לְמִשְׁנֵהוּ יְכוֹנֵן / הֲלָכוֹתָיו לְעֻמַּת הַפְּרָקִים

דְּבַר הָאֵל בְּסֵפֶר מִלְחֲמוֹתָיו / בְּאֶת וָהֵב בְּסוּפָה מַחֲזִיקִים

וּבֵית הִלֵּל וּבֵית שַׁמַּאי מְעִידִים / לְבָרֵר הַשְּׁמוּעוֹת מַעֲמִיקִים

וְכָל כָּבוֹד וּמוּסָר וַעֲנָוָה / בְּחַצְרֵיהֶם וְטִירוֹתָם דְּבוּקִים

וְלֹא נִשְׁמַע דְּבַר חֶרְפָּה בְּפִיהֶם / וְלֹא קִנְאָה וְנִקְרְאוּ מְצִיקִים

כְּמוֹ רַאבַּ״ד אֲשֶׁר מִלֵּא חֲדָרָיו / לְלַעַג עִם הֲבָלִים נֶחֱלָקִים

עֲלֵי גָאוֹן אֲשֶׁר הֵאִיר לְתֵבֵל / בְּפֵרוּשָׁיו לְמִשְׁפָּטָיו וְחֻקִּים

וְהִשְׂבִּיעַ לְנֶפֶשׁ הָרְעֵבָה / וְחָכְמוֹתָיו לְכָל צָמֵא מְנִיקִים

וְתִפְאֶרֶת לְאִישׁ בִּשְׁבָח אֲנָשִׁים / יְרֵאִים הֵם וּמִכָּל חֵטְא מְנֻקִּים

לְזֹאת יִתַּר לְבָבִי מִמְּקוֹמוֹ / וְעַצְמוֹתַי בְּשִׁבְרָם נֶאֱנָקִים

לְמַעְנֵהוּ הֲלֹא יִסְלַח אֱלֹהִים / בְּנֶפֶשׁ קוֹרְאָה מִמַּעֲמַקִּים

וְהִיא תַעְרֹג לְפָנֶיךָ וְתִדְמֶה / כְאַיָּל תַּעֲרֹג עַל הָאֲפִיקִים.
