[אִם הַיְּגוֹנִים]

אִם הַיְּגוֹנִים כְּאֵשׁ עֵצִים עֲצָמֵינוּ / עֵצִים אֲבָל נַעֲשׂוּ גַחְלֵי רְתָמֵינוּ

אוֹ אִם נְהָרוֹת בְּכִי עֵינִי וְרֹאשִי מְקוֹר / דִּמְעָה לְבַכּוֹת יְקוֹדֵנוּ וְחֻמֵּנוּ

לַיִל וְיוֹם יִזְּלוּ נַחְלֵי דְמָעַי עֲדֵי / נֶגְדָּם כְּמַר מִדְּלִי נַחַל קְדוּמֵינוּ

לַבַּת לְבָבִי בְלֵב יַמֵּי דְמָעוֹת תְּחִי / תִּבְעַר כְּיַעַר יְאוֹרֵנוּ וְיַמֵּנוּ

לוּלֵי אֲשֶׁר הֶחֱרִיב אִשִּׁי הֲמוֹן נַחֲלֵי / דִמְעִי כְּבָר טֻבְּעוּ בִּבְכִי יְקוּמֵינוּ [5]

הֶחְשִׁיךְ מְאוֹרִי זְמָן וַיַּךְ לְבָבִי בְחִיל / נִמְרָץ וְנֶעְרָץ וְרָץ בָּחְרִי לְהוּמֵנוּ

טָרַף וְנָגַף בְּאַף מַחְמַד לְבָבִי וְלֹא / נִמְנַע דְּבַר יוֹם בְּיוֹם מִבֹּא בְדַמֵּנוּ

יוֹם יוֹם יְסַעֵף פְּאֵר פֹּארוֹת אֲרָזַי וְלֹא / חָמַל וּבָנָה מְצוֹר צִירֵי כְרָמֵינוּ

לִבִּי כְפַעַם וְהוּא פַּטִּישׁ וְכָשִׁיל אֲשֶׁר / הוֹלֵם כְּפַעַם בְּפַעַם אֶת פְּעָמֵינוּ

אִבֵּד וְשִׁבֵּר שְׁאַר רוּחִי וְכֹחִי וְיֵז / נִצְחִי וְנִפֵּץ כְּאַבְנֵי גִיר לְשָׁמֵינוּ [10]

הוּא הוּא שְׁלֹמֹה שְׁלוֹם נַפְשִׁי וְחֶמְדַּת צְבִי / עֶדְיִי בְסוּר נָעֳמוֹ הוּסַר שְׁלוֹמֵנוּ

הוּסַר הֲדָרִי וְעֹז פָּנַי יְשֻׁנֶּה עֲדֵי / לִפְנֵי זְבוּב נִמְאֲסוּ נִבְזוּ צְלָמֵינוּ

לַיִל וְיוֹם אֶבְחֲרָה מָוֶת לְנַפְשִׁי וְלֹא / אָבָה לְהָפִיק זְמָנֵנוּ זְמָמֵנוּ

הָיו יְתוֹמִים בְּלִי אָב הַיְּגוֹנִים עֲדֵי / שַׁקַּם לְבָבִי וּמָצְאוּ אָב יְתוֹמֵינוּ

אָב לַחֲבָלִים אֲנִי הַיּוֹם וְרֹאשׁ לָחֳלִי / רֹאשׁ לָחֳלִי הַזְּמָן הַיּוֹם יְרִימֵנוּ [15]

בִּקְּשׁוּ יְעֵנִים וְלֹא מָצְאוּ לְנִיהֶם דְּמוּת / אָמְרוּ לְמִי הַזְּמָן בִּנְהִי יְדַמֵּנוּ

אַךְ בַּהֲרִימִי מְרִי שִׂיחִי בְכִיתָם זְמִיר / נֶחְשַׁב וְקוֹל תּוֹר לְעֻמַּת קוֹל רְעָמֵינוּ

אִם אֶשְׁכְּנָה אֶל פְּאַת צָפוֹן הֲלֹא קוֹל שְׁאוֹן / צִיר מַחֲצִי יִשְׁמְעוּ בִפְאַת דְּרוֹמֵנוּ

תַּנִּים לְתַנּוֹת חֳלִי לִבִּי יְלָדָם זְמָן / גַּם שָׁת לְזֹאת אֶת בְּנוֹת יַעְנָה תְאוֹמֵינוּ

אַךְ נִתְּנָה לַבְּכוֹר כַּדַּת בְּכוֹרָה וְעַל / נַחְלַת חֲבָלִים וְחִיל לִבְכוֹר יְשִׂימֵנוּ [20]

לִסְבּוֹל תְּלָאוֹת יְשִׁיבֵנוּ זְמָן אַחֲרֵי / שֵׂיבָה וְאַחְרֵי בְלוֹת לִימֵי עֲלוּמֵינוּ

לָשֵׂאת שְׂאֵת מַחֲלַת הֶגֶה וְהוֹ הוֹ וְהִי / לוּ אֶגְוְעָה אַחֲרֵי מָוֶת יְקִימֵנוּ

אַךְ בַּחֲזוֹתִי מְעַט טוּבוֹ יְבַלֶּה יְמֵי / נֹעַר וִיכַל שְׁנוֹתֵינוּ וְיָמֵינוּ

חָלְפוּ כְמוֹ צֵל שְׁנוֹת טוֹבָה וְהִתְעוֹפְפוּ / כַחְלוֹם יְמֵי גִיל וְהָיִינוּ כְגָלְמֵנוּ

נַפְשִׁי בְחֶלֶד כְּאַלְמָנַת זְמִיר יָשְׁבָה / בִבְכִי עֲדֵי קָם נְהִי לִהְיוֹת יְבָמֵנוּ [25]

יָבַם אֲבָל לֹא כְדַת יָבָם וְנִשְׁבַּע אֲשֶׁר / לֹא יִזְכְּרוּ עוֹד בְּשֵׁם הַשִּׁיר בְּעַמֵּנוּ

נוֹאָשׁ אֲנִי מֵחֲזוֹת עוֹד עֹז וְחֶדְוָה וְלֹא / אוֹחִיל אֲשֶׁר הַזְּמָן יָשׁוּב יְרַחְמֵנוּ

כִּי אַחֲרֵי דַכְּאוֹ לִבִּי וְאֵין בּוֹ מְתוֹם / בַּמֶּה יְרֻצֶּה וּבַמֶּה עוֹד יְנַחְמֵנוּ

הָאֶמְצְאָה אֶל שְׁבוּרֵי לֵב אֲרוּכָה וְאֶל / נִסְפֶּה וּמֵת יִתְּנוּ מַרְפֵּא בְשָׂמֵינוּ

יִשַּׁךְ זְמָנִי כְצִפְעוֹנִי וְלֹא יַאֲזִין / אֶל קוֹל חֲבָרִים וְאֶל לַחַשׁ חֲכָמֵינוּ [30]

מָחַץ לְבָבִי בְבֵן קַשְׁתּוֹ וְהִשְׁלִיךְ בְּאַף / אֶבֶן נְגָפָיו וְנָפְלָה בַעֲצוּמֵינוּ

אִשּׁוֹ כְקַשׁ אוֹכְלָה רוּחִי וְלִיקוֹד יְשַׁו / הָאֵשׁ יְגוֹנִים וְהָעֵצִים עֲצָמֵינוּ


סימן

יוֹדְעֵי יְגוֹנִי וְאוֹנִי / אֶל צַעֲקָתִי תְנוּ לֵב

כִּי רָחֲבָה יַם חֶבְלִי / וָאֶצְעֲקָה מִכְּאֵב לֵֹבׁ

על הכתב

בִּכְאֵב וְחִיל מֵעֵינַי / עֵינוֹת דְּמָעַי יָצְאוּ

וַיִּתְמְהוּ רוֹאַי כִי / עַיִן בְּעַיִן יִרְאוּ
