מֶטַמוֹרפוֹזוֹת

(קטעים נבחרים)

ספר א'

פתח דבר

רוּחִי לִי יָשִׁיר אֶת שִׁיר-הַחֲלִיפוֹת, זוֹ שִׁנּוּ הַדְּבָרִים

אַף לַחֲדָשִׁים הֲפָכוּם הָאֵלִים! הֶן יֶדְכֶם תְּחוֹלֵל

אֶת הַהֲפֵכוֹת, – הֱיוּ-נָא עִם שִׁירִי, וְיֵלֵךְ מֵישָׁרִים

שִׂיחוֹ – מֵרֹאשׁ קַדְמֵי-תֵבֵל עַד דּוֹרִי זֶה חַי עַל אֲדָמוֹת.

בריאת העולם

5 טֶרֶם יַבֶּשֶׁת וְיָם יִבָּרֵאוּ וּשְׁחָקִים מִסָּבִיב,

אֶחָד אַךְ הָיָה וְיָחִיד יְסוֹד הַיְקוּם עֲלֵי-תֵבֵל, –

כָּאוֹס זֶה שְׁמוֹ, וַיְהִי אַךְ גֹּלֶם אֵין תֹּאַר וּפָנִים,

נֵטֶל לֹא יָמוּשׁ מִמְּקוֹמוֹ, מוּצָק, בּוֹ נִקְבְּצוּ בָאוּ

זֶרַע וִיסוֹד לְכָל דָּבָר, נִפְרָדִים וְשׁוֹנִים גַּם יָחַד.

10 טֶרֶם עוֹד יִתֵּן הַטִּיטַן אֶת אוֹרוֹ לַתֵּבֵל וּמְלוֹאָהּ;

טֶרֶם גַּם תַּעֲלֶה פֶבֵּי לְחַדֵּשׁ חֲלִיפוֹת קַרְנֶיהָ;

טֶרֶם הָאָרֶץ, נְשׂוּאַת כֹּבֶד מְטִילָהּ, תְּרַחֵף

תְּלוּיָה עַל בְּלִימָה בָאַוִּיר; גַּם אַמְפִיטְרִיטֵי לֹא שָׁלְחָה

עוֹד זְרוֹעוֹתֶיהָ לְחַבֵּק יַרְכְּתֵי גְּבוּלוֹת הָאָרֶץ.

15 אָמְנָם גַּם אַוִּיר וָיָם גַּם אַדְמַת יַבֶּשֶׁת שָׁם הָיוּ,

אָכֵן אֵין עֶמְדָה בָאָרֶץ, הַגַּלִים לֹא יִשְּׂאוּ הַשֹּוֹחֶה,

תֹּהוּ הָאַוִּיר אֵין אוֹר; גַּם טֶרֶם תְּכוּנָתָם נָכוֹנָה:

אֶחָד בְּאֶחָד נִפְגָּשׁוּ, וּבְתוֹךְ גּוּף אֶחָד יִלָּחֲמוּ

חֹם – בְּצָרָתוֹ הַקָּרָה, לֵחַ – בְּחֹרֶב יַבֶּשֶׁת,

20 קָשָׁה – בָּרַךְ, וְהַכָּבֵד בַּאֲשֶׁר כָּבֵד אֵינֶנוּ.

יָד אֱלֹהִים בַּיְקוּם הִיא כוֹנְנָה הַשָּׁלוֹם וַתַּבְדֵּל

בֵּין הַשָּׁמַיִם לָאָרֶץ, בֵּין אֶרֶץ וּמִקְוֵה-הַמַּיִם,

גַּם בֵּין הַשַּׁחַק הַזַּךְ ומַעֲבֵה הָאַוִּיר, וַתּוֹצֵא

אוֹתָם מֵעֵרֶב הַתֹּהוּ: הִפְרִידָה בֵינֵיהֶם, אַף שָׁבָה

25 יַד אֱלֹהִים וַתְּקַבְּצֵם וַתְּכוֹנֵן בְּרִית-שָׁלוֹם בֵּינֵיהֶם.

אֵשׁ מִשָּׁמַיִם פָּרָצָה, אֵשׁ חִדְלַת-מִשְׁקָל מִכִּפַּת

שְׁחָקִים, וַתַּעַל אֶל מְרוֹמֵי-שִׂיאָם – שָׁם אִוְּתָה לָשֶׁבֶת.

כָּבֵד מֵאֵשׁ הָאַוִּיר – וַיָּבֹא וַיִּשְׁכֹּן עַל יָדָהּ;

עָבָה מֵהֶם הָאֲדָמָה, וְכָל יְסוֹד עָצוּם וְכָבֵד

30 בָּהּ הוּא, וַתִּכְרַע תַּחְתֶּיהָ וַתַּשְׁפֵּל לָשֶׁבֶת, וְנִקְּוּוּ

בַּיַּרְכָּתַיִם הַמַּיִם וּגְבוּל לַיַּבֶּשֶׁת הִצִּיבוּ.

אָז אֵל פַּלְמוֹנִי – יְהִי אֲשֶׁר יִהְיֶה, שְׁמוֹ לֹא יָדַעְנוּ –

יָבֹא וְגָזַר אֶת כָּל הַקְּבוּצָה הַזֹּאת לִגְזָרֶיהָ:

רֵאשִׁית מַעֲשֵׂהוּ וְצָנַף אֶת הָאֲדָמָה לִצְנֵפַת

35 כַּדּוּר, לְמַעַן הֱיוֹתָהּ שָׁוָה אֶל כָּל עֲבָרֶיהָ;

יָפִיץ אֶת מִקְוֵה-הַמַּיִם וְהָיוּ לְיַמִּים, יַקִּיפוּ

רַחֲבֵי אֲרָצוֹת וְסַעַר מִתְחוֹלֵל מַכֶּה גַּלֵּיהֶם;

גַּם הַמַּעְיָנוֹת יָפֻצוּ, הַיְּאוֹרוֹת וְאַגְמֵי-הַמָּיִם;

שׁוֹטְפִים הַנְּהָרוֹת גַּם הֵם, עֲקַלְקַלּוֹת יִלָּפְתוּ דַּרְכֵיהֶם

40 יֵשׁ וְאֶל חֵיק הָאֲדָמָה יִשָּׁפְכוּ וּבְלָעָם עֲפָרָהּ,

יֵשׁ וְעַד מֶרְחַקֵּי יָם זֶה רְחַב-הַיָּדַיִם יָבֹאוּ,

יִשְׂאוּ שָׁם דָּכְיָם וּלְחוֹף הַסֶּלַע יְנֻפְּצוּ גַּלֵּיהֶם.

צִוָּה הָאֵל וַיִּשְׂתָּרְעוּ הַשָּׂדוֹת, עֲמָקִים הֶעֱמִיקוּ,

עָטְפוּ יְעָרוֹת עֲלֵיהֶם וְהַרְרֵי-סְלָעִים הָגְבָּהוּ.

45 אָזַר הַשַּׁחַק אֲזוֹרָיו – שְׁנַיִם יְמִינָה וּשְׁנַיִם

שְׂמֹאלָה, וְלֹהֵט מְאֹד הָאֵזוֹר הַחֲמִישִּׁי בַתָּוֶךְ.

כֵּן גַּם בִּצְנֵפַת הָאָרֶץ רַבַּת-הַכֹּבֶד הִבְדִילָה

זְרוֹעַ הָאֵל אֲזוֹרֶיהָ וַיִּהְיוּ גַם הֵמָּה חֲמִשָּׁה;

שָׁמֵם הָאֵזוֹר הַתִּיכוֹן מִשָּׁרָב, לֹא יֵשֵׁב שָׁם אָדָם,

50 שֶׁלֶג יְכַסֶּה הַשְּׁנַיִם, וּשְׁנַיִם אֲשֶׁר בֵּינוֹתֵיהֶם

מָזְגוּ קָרָתָם וְחֻמָּם חֵלֶק כְּחֵלֶק גַּם יָחַד.

יַעֲטֶה אַוִּיר אֶת כָּל הָאֲזוֹרִים, וְהָיָה מִשְׁקָלוֹ

כָּבֵד מֵאֵשׁ, כִּכְבֹד הֶעָפָר מִכֹּבֶד הַמָּיִם.

יִדַּע הָאֵל לַעֲרָפֶל מְקוֹמוֹ וּמִשְׁכָּן לֶעָבִים,

55 דֶּרֶךְ לְקוֹלוֹת וּבְרָקִים כִּי יִתְּנוּ חִתִּיתָם בָּאָדָם.

אָז גַּם לְרוּחַ וָסַעַר, מְחוֹלְלֵי שֹׁאָה וְקָרָה,

יִיעַד אָרְחוֹתָם, לְבִלְתִּי יִפְרְצוּ פֶּרֶץ בָּאַוִּיר,

כִּי גַּם כָּעֵת, אַף אִם יֵדַע רוּחַ וָרוּחַ נְתִיבוֹ,

אַךְ בְּדֵי-עָמָל יֵעָצֵר שֶׁטֶף זִרְמָתָם, הַשּׁוֹאֵף

60 לַעֲרוֹץ אֶרֶץ ומְלוֹאָהּ, כִּי רַבָּה מַשְׂטֵמַת הָאַחִים:

אֶוְרוּם בִּזְבוּלֵי אוֹרוֹרָה יִשְׁכֹּן וַיֵּט עַד מַמְלֶכֶת

פָּרַס וּבְנֵי-הַנְּבָטִי וְהַרְרֵי בֹקֶר זוֹרֵחַ;

זָפִיר יְבַקֵּשׁ אֶת קִרְבַת וֶסְפֶּר וְאֶרֶץ נוֹשֶׁקֶת

קַרְנֵי הַשֶּׁמֶשׁ בְּבֹאוֹ; יַרְגִּיז בּוֹרֵיאַס הַזּוֹעֵם

65 סְקִתְּיָה וּמִשְׁכְּנוֹת עַיִשׁ, וְאוֹסְתֵּר בְּיַרְכְּתֵי תֵימָן

יַעַל נְשִׂיאִים וָעָנָן וְהִמְטִיר רְבִיבֵי נְדָבָה.

מָרוֹם מִמַּעַל לָאַוִּיר שָׁם שַׁחֲקֵי-הָאֶתֶּר נִטָּיוּ,

חִדְלֵי-הַכֹּבֶד וּשְׁקוּפִים וּטְהוֹרִים מֵחֶלְאַת אֲדָמָה.

כָּכָה יְחַלֵּק הָאֵל וַיַּצֵּב לַכֹּל אֵת גְּבוּלוֹתָיו.

70 אָז גַּם הֲמוֹן הַכּוֹכָבִים יִתְפָּרְצוּ מִתַּחַת מַשָּׂאָם,

הִכְבִּיד עֲלֵיהֶם הַתֹּהוּ, וְהִבְרִיקוּ בַשָׁחַק. וַיִּתֵּן

יוֹצֵר הָאֵלִים גַּם נֶפֶשׁ חַיָּה לְכָל מֶמְשְׁלוֹתָיו:

אֵלִים וּצְבָא הַכּוֹכָבִים יִזְבְּלוּ אֵת הַשָּׁמַיִם,

דְּגֵי קַשְׂקֶשֶׂת נוֹצֵצָה יֶחֱצוּ הַגַּלִים, וְתִשְׁכֹּן

75 חַיַּת הַשָּׂדֶה בָאָרֶץ וְעוֹף בָּאַוִּיר יִתְעוֹפֵף.

בריאת האדם

אָכֵן לֹא הָיְתָה עוֹד נֶפֶשׁ זַכָּה מֵאֵלֶּה, שׁוֹאֶפֶת

דַּעַת וְשִׁלְטוֹן בְּכָל הַחַי אֲשֶׁר עַל הָאֲדָמָה.

אָז הָאָדָם יִוָּלֵד. הֲיֵשׁ כִּי אֲדוֹן כָּל הַמַּעֲשִׂים,

יוֹצֵר תֵּבֵל וּמְכוֹנְנָהּ, מִזֶּרַע הָאֵלִים הִצְמִיחוּ,

80 אוֹ בָאֲדָמָה, אֲשֶׁר מִזְּרוֹעוֹת הָאֶתֶּר לֻקֳּחָה,

עוֹד עִקְּבוֹת שַׁחַק, אָחִיהָ מִקֶּדֶם, נִשְׁאָרוּ, – וַיִּקַּח

עָפָר מִמֶּנָּה בֶּן-יַפֶּתּוּס וּבָלוֹל בַּמַּיִם וַיִּיצֶר

אָדָם בְּצֶלֶם הָאֵלִים הַמּוֹשְׁלִים בְּתֵבֵל וּמְלוֹאָהּ.

הַשְׁפֵּל אִם תַּשְׁפִּיל כָּל נֶפֶשׁ רֹאשָׁהּ וְתַבִּיט לָאָרֶץ,

85 מָרוֹם יַגְבִּיהַּ הָאָדָם רֹאשׁוֹ וְעֵינָיו אֶל נוֹכַח

שַׁחַק תַּשְׁקֵפְנָה וְהָיָה עַד שִׂיא הַכֹּכָבִים מַבָּטָן.

כָּכָה אֶת אַדְמַת הַתֹּהוּ, גֹּלֶם אֵין תֹּאַר וְהָדָּר,

הָפְכָה יָד אֵלִים וַתְּהִי תַּרְבּוּת אֲנָשִׁים בְּאֶרֶץ.

ארבעה דורות

הָיָה בְּרֵאשִׁית דּוֹר-זָהָב, אֲשֶׁר בְּאֵין נוֹגֵשׂ, וְטֶרֶם

90 מִשְׁפָּט וְחֹק יְלַמְּדוּהוּ, אֱמוּנִים וְתֹם שָׁמַר לִבּוֹ.

שְׁפָטִים בַּל נַעֲשׂוּ עוֹד, אַף טֶרֶם עַל לוּחוֹת נְחוֹשֶׁת

דִּבְרֵי הָאָלָה יֵחָרְתוּ, וְטֶרֶם יָבֹאוּ בִּרְעָדָה

דַבֵּר תַּחֲנוּנִים לַשּׁוֹפֵט, אַךְ יָשְׁבָה הָאָרֶץ בְּאֵין מַחֲרִיד.

עוֹד לֹא יִכָּרֵת הָאֹרֶן וְעָזַב אֵת תֹּעֲפוֹת הָרָיו

95 לַחְתּוֹר בְּנִבְכֵי הַגַּלִּים אֶל אֶרֶץ נָכְרִיָּה, וְטֶרֶם

יֵדְעוּ בְנֵי-אָדָם שְׂפַת-יַמִּים זוּלָתִי חוֹף אֶרֶץ-מוֹלֶדֶת.

שׁוּחוֹת עֲמֻקּוֹת עוֹד טֶרֶם מֵזַח לֶעָרִים תִּהְיֶנָה,

שׁוֹפַר נְחשֶׁת וַחֲצוֹצְרַת-מִקְשֶׁה לֹא הָיוּ עֲדֶנָּה,

אַף לֹא חֲרָבוֹת וְקוֹבַע; לֹא לָמְדוּ תַּכְסִיסֵי מִלְחָמָה,

100 יָשְׁבוּ לָבֶטַח לְאֻמִּים וַיִּרְאוּ מְנוּחָה נָעֵמָה.

עוֹד הָאֲדָמָה לֹא נָשְׂאָה בְעֹל וְלֹא קָרְעוּ סְגוֹר-לִבָּהּ

אֵת וּמַחֲרֵשָׁה, אַךְ נָתְנָה לְמַכְבִּיר תְּנוּבָתָהּ נְדָבָה.

חָמְדוּ בְּנֵי-אָדָם צְמָחֶיהָ, לֹא עָמְלוּ יָדַיִם בָּהֵמָּה,

גַּרְגְּרֵי-שָׂדֶה יִלְקֹטוּ, בּוּל-הָרִים וּפְרִי-דֻבְדְּבָנִים,

105 יִקְטְפוּ עִנְבֵי מַלּוּחַ, הָעוֹלִים בֵּין קוֹצֵי הַשִּׂיחִים,

אוֹ אֶת אֱגוֹזֵי הָאַלּוֹן, עֵץ-חָמַד לוֹ עֶלְיוֹן בָּאֵלִים.

פָּרַח הָאָבִיב עוֹלָמִים, וַיְרַחֲפוּ זֶפִירִים שַׁאֲנַנִּים,

רַחֵף וְסַלְסֵל בִּכְנָפָם הַחַמָּה נִצָּנֵי סָפִיחַ.

תָּחִישׁ גַּם שְׁדֵמָה לֹא עֻבְּדָה וְנָשְׂאָה פְרִי-תְּנוּבָה, וְעָטַף

110 זְהַב הַשִּׁבֳּלִים הַמְּלֵאוֹת נִיר, לֹא הֶאֱרִיכוּ מַעֲנִיתוֹ;

יִשְׁטְפוּ נַהֲרוֹת חָלָב אוֹ יִשְׂאוּ צוּף נֶקְטַר הַנְּהָרוֹת,

יִטְּפוּ אַלּוֹנֵי-הָאָטָד גַּם הֵם עֲסִיסֵי דְבַשׁ-זָהָב.

הוּרַד סַטוּרְנוּס מַחֲשַׁכֵּי הַטַּרְטַר וְיָשַׁב עַל כִּסְאוֹ

לִמְשֹׁל בְּתֵבֵל יֻפִּיטֵר, – מֵאָז תּוֹר-הַכֶּסֶף הִגִּיעַ,

115 מָךְ חִין-עֶרְכּוֹ מִזָּהָב וְיָקָר מִנְחֹשֶׁת נוֹצֵצָה.

נָטָה יֻפִּיטֵר אֶת יָדוֹ וַיַּקְצֵר עֲלוּמֵי הָאָבִיב,

חָלַק הַשָּׁנָה לִתְקוּפוֹת, שָׂם מוֹעֵד לְחוֹרֶף וָקַיִץ,

אָסִיף הַמְּשַׁנֶּה אֶת טַעֲמוֹ וְאָבִיב מְהִיר-הַכְּנָפָיִם.

חֹרֶב וָחֹם אָז רִאשׁוֹנָה הִרְתִּיחוּ הָאַוִּיר, וְרִגְבֵי

120 קֶרַח מִנִּשְׁמַת-הַסַּעַר הַקָּרָה לַמּוּצָק דֻבָּקוּ.

אָז גַּם בֵּית-מוֹשָׁב רִאשׁוֹנָה יִהְיֶה בָאָרֶץ, – בִּמְּעָרוֹת

צוּרִים בְּסִבְכֵי יְעָרוֹת אוֹ מְלוּנַת-נְצָרִים עֲבֻתָּה.

קֶרֶס אָז תִּזְרֶה רִאשׁוֹנָה אֶת חָפְנֵי זְרָעֶיהָ בִּתְלָמִים

רִפְּדָה הָאֵת, וּבְנֵי-בָקָר מִסֵּבֶל הָעֹל יֵאָנֵחוּ.

125 חָלַף תּוֹר-כֶּסֶף גַּם הוּא וַיָּקָם דּוֹר חָדָש, נְחֻשְׁתָּן.

רוּחַ אַבִּירִים בְּלִבּוֹ וְיָדָיו לוֹ רַב בַּמִּלְחָמָה,

לֹא יָדַע עוֹד פֶּשַׁע. דּוֹר-בַּרְזֶל נוּקְשֶׁה אָז קָם בָּאַחֲרוֹנָה.

נִקְבְּצוּ בָאוּ גַם יַחַד כָּל חֵטְא וְכָל עָוֹן פְּלִילִיָּה:

תַּמּוּ אֱמוּנִים מֵאֶרֶץ וְיֹשֶׁר וְתֻמַּת-עֲנָוִים,

130 תַּחְתִָּם יָבֹאוּ וְיָרְשׁוּ מְקוֹמָם אַךְ מִרְמָה וָרֶשַׁע,

חָמָס וְנִכְלֵי-בְלִיַּעַל וְאַהֲבַת בֶּצַע –מַעֲשַׁקּוֹת.

יוֹרְדֵי-הַיָּם מִפְרְשֵׂיהֶם צָרְרוּ בְכַנְפֵי-הָרוּחַ,

טֶרֶם יְדָעוּהוּ עַד-הֵנָּה, וְאֹרֶן מִתְעָרֶה בֶּהָרִים

יָרַד מִמְּרוֹמָיו, יִלָּפֵת דַּרְכּוֹ בֵין גַּלֵּי-מֶרְחַקִּים.

135 שְׁדֵמָה, זוּ הָיְתָה נַחֲלַת כָּל-אָדָם, כַּשֶּׁמֶשׁ וְאַוִּיר,

נָטָה עָלֶיהָ הַמּוֹדֵד אֶת קַוּוֹ וַיַּצֵּב גְּבוּלֶיהָ.

דַּי עוֹד לֹא יֹאמַר הָאָדָם, כִּי יִדְרֹשׁ מֵחֵיק הָאֲדָמָה

תְּרוּמַת תְּנוּבָתָהּ וּפִרְיָהּ לְמַכְבִּיר, אָז יִקְרַע סְגוֹר-לִבָּהּ,

הָלֹךְ וְהַעֲמֵק שְׁאֹלָה, עַד יַרְכְּתֵי אֶרֶץ-רְפָאִים,

140 לַחְפֹּר מַטְמוֹנִים הָחְבָּאוּ וְאוֹצָרוֹת, מַכְאוֹבִים אַחֲרִיתָם.

יֶאֱתֶה בַּרְזֶל מְשַׁכֵּל וְזָהָב מֵהָרֵס מִמֶּנּוּ,

אִתָּם-מִלְחָמָה הָעֹשָׂה בִשְׁנֵיהֶם גַּם יָחַד מְלַאכְתָּהּ,

זְרוֹעָהּ חֲמוּצָה בְדָם וּבְצִלְצְלֵי-נֶשֶׁק רוֹעֶשֶׁת.

חָמָס הָאָרֶץ מָלֵאָה: קָם אִישׁ עַל רֵעֵהוּ בָאֹהֶל,

145 חָתָן עַל חוֹתְנוֹ יִתְגַּבֵּר, גַּם אַהֲבַת אַחִים נֶעְדָּרָה.

אִישׁ יַחֲרֹשׁ רָעָה עַל אִשְׁתּוֹ וְאִשָּׁה עַל בָּעַל-נְעוּרֶיהָ,

תִּמְסֹךְ הָאֵם הַחוֹרֶגֶת סַף-רַעַל וְדָרְשׁוּ הַבָּנִים

אֶל הַיִּדְּעוֹנִים, עַל מִסְפַּר שְׁנוֹת אֲבוֹתֵיהֶם יִשְׁאָלוּ.

כָּרַע גַּם נָפַל הַצֶּדֶק, אָז תָּקוּם הָעַלְמָה אַסְטְרֵיאָה,

150 שְׂרִידַת הָאֵלִים בָּאָרֶץ, וְעָלְתָה מֵעֵמֶק הַדָּמִים.

הגיגנטים

אָכֵן כְּבָר אָרַב הַשֹּׁד גַּם לְשַׁפְרִיר מְרוֹמֵי הָאֶתֶּר:

קָמוּ גִיגַנְטִים לַכְרִיעַ מַמְלֶכֶת הָאֵלִים בַּשְּׁחָקִים,

הָרִים עַל הָרִים הִרְכִּיבוֹ וּגְבָעוֹת עַד שִׂיא הַכּוֹכָבִים.

אָז אֵל-הָאֵלִים נֶאְדָּרִי-בַכֹּחַ חֵץ-בְּרָקָיו יְשַׁלַּח,

155 יַרְעִישׁ אוֹלִימְפּוּס וְהָדַף הַר פֶּלְיוֹן מִבָּמֳתֵי הָאוֹסָה.

נָפְלוּ הִתְגּוֹלְלוּ גִּיגַנְטִים אֶל תַּחַת מַשּׂוּאוֹת הָרֵיהֶם,

דְּמֵיהֶם כַּנָּהָר הִתְפָּרְצוּ וַיְכַסּוּ פְנֵי אִמָּם אֲדָמָה.

חָמְלָה עֲלֵיהֶם הַאֵם, בַּל זִכְרָם מֵאֶרֶץ יִכָּחֵד,

נָפְחָה נְשָׁמָה וְחַיִּים בִּדְמֵיהֶם הַחַמִּים וַתְּשַׁוֶּה

160 צֶלֶם בְּנֵי-אָדָם לָהֵמָּה. אַךְ הָיוּ גַּם הֵם בְּנֵי-מְרֵעִים,

זֵדִים מְנָאֲצֵי-אֵל וְשׁוֹאֲפֵי דָמִים וָרֶצַח:

עָנְתָה הַכָּרַת פְּנֵיהֶם לֹא כִחֲשָׁה, כִי דָמִים מְקוֹרָם.

סוד האלים

בֶּן סַטוּרְנוּס אַךְ רָאָה הַדָּבָר מִזְּבוּלוֹ בַמְּרוֹמִים,

נֶאֱנַח קָשֶׁה, אַף זָכַר תּוֹעֵבָה חֲדָשָׁה נַעֲשָׂתָה,

165 תוֹעֲבַת מִשְׁתֵּה לִיקָאוֹן, עוֹד טֶרֶם יְדָעוּהָ הָאֵלִים.

אָז אֵשׁ חֲמָתוֹ תִּתְלַקַּח, חֲמַת-אַף יֻפִּיטֶר, וַיִּקְרָא

אֶת מְתֵי-סוֹדוֹ הָאֵלִים, יְמַהֲרוּ, לֹא יִתְמַהְמָהוּ.

יֶשְׁנוֹ בַמְּרוֹמִים נָתִיב וּבְזֹךְ הָרָקִיעַ יֵרָאֶה,

שְׁמוֹ נְתִיב-חָלָב יִקָּרֵא, כִּי צַח הוּא מִלִּבְנַת-הַזֹּהַר:

170 שָׁם לִבְנֵי-אֵלִים מְסִלָּה אֶל דְּבִיר אֱלֹהֵי-הַחֲזִיזִים.

יָמִין וֹּשְמֹאל שָׁם יִתְנוֹסְסוּ זְבוּלֵי הָאֵלִים הַגְּדוֹלִים,

פְּתוּחִים לִרְוָחָה וּמְלֵאִים הֲמוֹנִים, לְשִׁכְנָם יִדְרשׁוּ;

דַּלַּת הָאֵלִים מִסָּבִיב זְעֵיר שָׁם זְעֵיר שָׁם יִשְׁכֹּנוּ;

שָׁם גַם אַדִּירֵי הַשַּׁחַק וּקְצִינָיו עַל-גַּפֵּי מְרוֹמִים

175 בָּנוּ בָתֵּיהֶם. אָכֵן נוֹרָא הַמָּקוֹם וְאֵין זֶה –

אַרְהִיב וְאֹמַר – כִּי אִם גִּבְעַת פַּלַטְיוּם לָאֵלִים.

אִוּוּ הָאֵלִים לָהֶם אוּלָם-שַׁיִשׁ בְּיַרְכְּתֵי הַהֵיכָל,

נִשְׁעָן עַל מַטֵּה-הַשֵּׁן יוֹפִיעַ בְּרֹאשָׁם יֻפִּיטֶר,

רֹאשׁוֹ יָנִיעַ וְרָעֲדוּ קְוֻצּוֹת-תַּלְתַּלָּיו, וְאִתָּם

180 יִרְגְּזוּ מוֹסְדוֹת הָאָרֶץ, הַיָּם וְכָל צְבָא-הַשָּׁמָיִם.

אַחַר יִפְתַּח אֵל שְׂפָתָיו וְדִבֵּר בְּרוּחוֹ הַקָּשָׁה:

"אָז בְּהִתְקוֹמֵם גִּיגַנְטִים, צִפְעוֹנִים רַגְלֵיהֶם, אַף הָדוּ

מֵאוֹת יְדֵיהֶם וְלִלְכֹּד אֶת כֵּס הַשָּׁמַיִם הֶעְפִּילוּ,

רֹגֶז לֹא יָדַע לְבָבִי כָּרֹגֶז זֶה-עַתָּה בָאָנִי.

185 כִּי אִם אַכְזָרִי וְנוֹרָא אָז הָיָה צָרֵנוּ, אַךְ הָיְתָה

אַחַת מִשְׁפַּחְתּוֹ וְאַחַת הָעֵדָה, הִתְגָּרוּ מִלְחָמָה.

עַתָּה כִּי אָקוּם, וְאָבְדָה תֵבֵל וּמְלוֹאָהּ עַד אַפְסֵי

יָמִּים רְחוֹקִים, בְּשָׁאוֹן שָׁם יִשָּׂא מִשְׁבָּרָיו נֵרֵוְס.

הוֹי פַּלְגֵי שַׁחַת, מְפַכִּים בְּעִמְקֵי תַחְתִּיּוֹת! נִשְׁבַּעְתִּי,

190 רַבּוֹת נִסֵּיתִי וָאוֹחִיל, אַךְ כָּלָה נֶחְרָצָה מֵאִתִּי, –

הָתֵז הַפֶּצַע, פֶּן אִתּוֹ הַגְּוִיָּה גַּם כֻּלָּה תִּסָּפֶה!

אֵלֵי-תַחְתִּיּוֹת הֵן יֵשׁ, אֱלִילֵי-הַכְּפָר, בָּהֶם נִימְפוֹת,

פַאוּנִים, סַטִּירִים וּקְהַל סִלְוַנִּים הַשּׁוֹכְנִים בֶּהָרִים.

טֶרֶם לָשֶׁבֶת בְּשַׁחֲקֵי-הָרֹם אַחַר כָּבוֹד לְקַחְנוּם,

195 יֵשְׁבוּ לְמִּצְעָר שַׁאֲנַנִּים בָאָרֶץ נַחְלָתָם מִתָּחַת.

אָכֵן, הוֹי שִׂימוּ אֶל לִבְּכֶם, הֲיֵשׁ כִּי יֵשְׁבוּ לָבֶטַח,

אִם גַּם עָלַי, יֹרֵה-בְרָקִים וּמוֹשֵׁל בְּאֵלֵי-שָׁמַיִם,

חָרַשׁ מְזִמּוֹת לִיקָאוֹן אִישׁ-דָּמִים, זֶה שְׁמוֹ עֲלֵי אָרֶץ".

נִרְגְּשׁוּ הָמוּ הָאֵלִים וּשְׁפָטִים בַּחֲמָתָם יִדְרשׁוּ,

200 שְׁפָטִים לְראֹשׁ בֶּן-הַמֶּרִי. כָּכָה הִתְרַגְּשׁוּ לְאֻמִּים,

אָחֲזָה שַֹעַר גַּם תֵּבֵל וּמְלוֹאָהּ לִשְׁמוּעַת-בַּלָּהוֹת,

עֵת עֲדַת זֵדִים הִתְקוֹמְמָה לִשְׁלֹחַ יַד-רֶשַׁע בְּנֶפֶשׁ

קֵיסָר וּלְהַאֲבִיד שֵׁם וְזֵכֶר לְרוֹמָא בָאָרֶץ.

אַהֲבַת-אֵמוּן רֵעֶיךָ אִם נָעֲמָה לְךָ אָז, אַבְגוּסְטוּס,

205 כֵּן לְיֻפִּיטֶר אֱמוּנֵי אֵלָיו נָעֵמוּ. אָז יִתֵּן

אוֹת וַתָּקָם לִדְמָמָה סַעֲרַת רוּחָם, נָדַמּוּ

קוֹלוֹת הַשָּׁאוֹן, מִפְּנֵי גְאוֹן יֻפִּיטֶר, בְּפָתְחוֹ

שֵׁנִית שִׂפְתוֹתָיו, וְכָכָה יִשָּׂא אֲלֵיהֶם נְאוּמוֹ:

"אַל-נָא תֵרָגְזוּ: כְּבָר נָשָׂא הָעַוָּל אֶת מוּסַר-עֲוֹנוֹ.

210 עַתָּה דְּעוּ-נָא, מַה פָּעַל הָאִישׁ וְתוֹכַחְתּוֹ מֵאִתִּי.

זַעֲקַת-חָמָס מֵאֶרֶץ הִגִּיעָה אָזְנֵינוּ, וָאֹמַר,

אַךְ דִּבַּת-שֶׁקֶר שָׁמַעְנוּ, וָאֵרֵד מִמְּרוֹמֵי אוֹלִימְפּוּס,

מַרְאֵה אֲנָשִׁים לֹא-אֵל לִי שַׂמְתִּי וָאָשׁוּט בָּאָרֶץ.

תִּקְצַר הָעֵת לִי מִסְּפוֹר הַשֹּׁד וְהָאָוֶן מָצָאתִי,

215 רַבּוּ אַף גָּבְרוּ מְאֹד גַּם מִן הַשְּׁמוּעָה שָׁמָעְנוּ.

הַרְרֵי מֵנַלִים עָבַרְתִּי, רֵבֶץ בְּנֵי-שָׁחַץ, קִילֶנֵי,

גַּם הַר לִיקֵאוּס, הַר-קֶרַח, יֵעֲרוֹת-אֹרֶן כִּסּוּהוּ.

אָז קִרְיַת הָאַרְקַדִּי אָבוֹא וָאֵפֶן אֶל אַרְמוֹן הַמֶּלֶךְ.

נָטָה הַיּוֹם וַיְּהִי הַנֶּשֶׁף וַתֵּרֶד אֲפֵלָה.

220 אוֹתִי נָתַתִּי, כִּי אֵל בָּא לִשְׁכֹּן בְּתוֹכָם, וַיִּקְּדוּ

בָּאֵי הַבָּיִת. אָז יִלְעַג לִיקָאוֹן לִמְכַבְּדֵי אֱלֹהַּ.

"הָבָה – יִתְקַלֵּס – אֲנַסֶּה, אִם אָמְנָם אֵל שׁוֹכֵן אִתָּנוּ

אוֹ בֶן-תְּמוּתָה, אַף אָשִׂים אוֹת לִי וָאֵדַע דְּבַר-אֱמֶת".

אָמַר בְּלִבּוֹ הַלָּז: עֵת תִּפֹּל תַּרְדֵּמָה עַל עֵינַי,

225 יָקוּם עֲלֵי לְהָרְגֵנִי, – כֹּה בָחֹן יִבָּחֵן הַדָּבָר.

אָכֵן לֹא יֹאמַר עוֹד דַּי, וְאִתּוֹ מֵעַם הַמּוֹלוֹסִים

אִישׁ לְעֵרָבוֹן שָׁלוּחַ – וַיָּקָם וַיַּכּוֹ בֶחָרֶב.

רוֹעֲדִים עוֹד כָּל יְצוּרָיו, וַיְבַשֵּׁל בַּסִּיר הָרוֹתֵח

חֶצְיָם, וְיִתְרָם לוֹ יִצְלֶה בָאֵשׁ עַל הָאָח הַמְּבֹעָרֶת.

230 אַךְ אֶת מַטְעַמָּיו לַשֻּׁלְחָן יַגִּישׁ – וָאֶשְׁלַח אֶת בְּרָקִי,

בָּרָק מְפוֹצֵץ, וְהֻכָּה הַבָּיִת עַל יוֹשְׁבָיו בְּנֵי-מֶרִי.

נִדְהָם לִיקָאוֹן וְנָס עַל נַפְשׁוֹ הַשָּׂדֶה, שָׁם יִנְהַק,

יֵילִיל, אַף דַּבֵּר יְנַסֶּה, אַךְ נֶעְתְּקוּ מִלִּים מֵאִתּוֹ.

פִּיהוּ כִּמְשַׁחֵר לַטֶּרֶף, וְנַפְשׁוֹ, הַצְּמֵאָה לָרֶצַח,

235 מִרְבַּץ עֲדָרִים תְּבַקֵּשׁ, אַף יָשִׂישׂ אִם דָּמִים יָרִיחַ.

הָיוּ שִׂמְלוֹתָיו לְצֶמֶר שָׂעִיר וְיָדָיו – כְּרָעַיִם,

כֻּלוֹ כִּזְאֵב מַרְאֵהוּ, אַךְ נוֹדְעוּ עִקְּבוֹתָיו מִקֶּדֶם:

שֵׂיבָה בְּשַׂעֲרוֹת עוֹרוֹ וְעַזּוּת יָפִיקוּ הַפָּנִים,

עֵינָיו כִּידוֹדֵי-הָאֵשׁ וְזֹעֵם וְזָעֵף מַרְאֵהוּ.

240 הָשַּׁם הַבַּיִת הָאֶחָד, אָכֵן לֹא אֶחָד הַבַּיִת

חָטָא שְׁאוֹל, – וֶאֱרִינִיס מַמְאֶרֶת מוֹשֶׁלֶת בָּאָרֶץ

כְּמוֹ עָשָׂה חֹזֶה הָאָדָם אֶת רֶשַׁע. אַךְ בָּא יוֹם פְּקוּדָתִי

לִפְקוֹד עַל כָּל מַעֲלָלָיו, כִּי כָּלָה נֶחְרָצָה מֵאִתִּי!"

המבוּל

יָשְׁרָה בְעֵינֵי הָאֵלִים עֲצַת יֻפִּיטֶר וְאָמֵן

245 יַעֲנוּ אַחֲרָיו וְהָאֵץ יָאִיצוּ בָאֵל הַמִּתְעַבֵּר.

אָכֵן עַל אָבְדַן הָאָדָם נַפְשָׁם דּוֹאָבֶת וְשָׁאֹל

יִשְׁאֲלוּ יַחְדָו: מַה יִהְיֶה לָאָרֶץ, כִּי תֵשַׁם מֵאָדָם?

מִי עוֹד לָאֵלִים יַקְטִיר לְבוֹנָה? אִם אָמְנָם הָאָרֶץ

מִרְמָס לִפְרִיצֵי-הַחַיּוֹת תִּנָּתֵן וְהָיְתָה לְבָעֵר?

250 רוּחָם הָרַגָּז יֻפִּיטֶר יַשְׁבִּיחַ וְאָמַר, כִּי דָאֹג

יִדְאַג לַכֹּל וְכִי יַפְלִיא אוֹתוֹתָיו וּבָרָא דּוֹר חָדָשׁ,

זֶרַע אֲנָשִׁים אֲשֶׁר לֹא יִדְמֶה לְאָדָם מִקֶּדֶם.

אָז אֵל-הָאֵלִים יְכוֹנֵן אֶת בְּרָקָיו לִשְׁלֹחַ בָּאָרֶץ,

אָכֵן בָּא פַחַד בְּלִבּוֹ, פֶּן יֹאחַז הַלַּהַב בְּשַׁפְרִיר

255 שַׁחֲקֵי-הָרֹם וְאֵשׁ מִתְלַקַּחַת צִיר תֵּבֵל תְּלַהֵט.

יִזְכֹּר הָאֵל גַּם חֲזוֹן-עֲתִידוֹת בְּסֵפֶר הַחַיִּים:

יָבֹא יוֹם פְּקוּדָה וְלִהֲטָה אֵשׁ נְאוֹת-אֶרֶץ וְיַמִּים,

יֻצַּת גַּם זְבוּל הַשָּׁמַיִם, וּמוֹסְדוֹת הָאָרֶץ יְמוּטוּן.

בְּרָקָיו, זוּ כוֹנְנוּ חַרְשֵׁי קִיקְלוֹפִּים, אָז יִשְׁלַח מִפָּנָיו,

260 יִבְחַר לוֹ שְׁפָטִים אֲחֵרִים – הֲמוֹן שִׁפְעַת מַיִם לְשַׁחֵת

יַחַד אֶת כָּל הַיְקוּם, מִטְּרוֹת-עֹז מֵאֲרֻבּוֹת שָׁמַיִם.

יָקוּם וְשָׂם אֶת אַקְוִילוֹ בְּמִשְׁמָר בִּמְעָרוֹת אֵאוֹלוּס

גַּם אֶת כָּל רוּחַ נוֹשֶׁבֶת, בַּל תָּפִיץ הֶעָבִים, וַיִּשְׁלַח

נוֹטוּס לַחָפְשִׁי. בִיעָף יִפְרֹץ נוֹטוּס וּכְנָפָיו תִּרְעַפְנָה,

265 צִלְלֵי-עֲרָפֶל יַקְדִּירוּ אֶת פָּנָיו, פְּנֵי אֵימָה וּמָגוֹר;

זְקָנוֹ כְּבַד-גֶשֶׁם, וּפְלָגִים מִשַּׂעֲרוֹת הַשֵּׂיבָה יִזְרֹמוּ.

מִצְחוֹ מְעוֹנֵי עֲנָנִים וְשׁוּלָיו יַרְעִיפוּ רְבִיבִים.

יִשְׁלַח אֶת זְרוֹעוֹ וְהִכָּה בְּמִפְלְשֵׂי-עָבִים – אָז יַרְעִים

רַעַם בַּגַּלְגַּל וְזִרְמָם הַסּוֹחֵף שָׁמַיִם יַמְטִירוּ.

270 אִירִיס, הַצִּיר לְיוּנוֹ, תַּעֲדֶה שְׁלַל-רִקְמָתַיִם,

תִּצְרֹר לָהּ מַיִם וְנָשְׂאָה מַשְׁקֶה לְפִי הָעֲנָנִים.

תֻּצַּע לָאָרֶץ הַקָּמָה וְנִכְזְבוּ תִּקְווֹת הָאִכָּר,

יֹאבְדוּ לַהֶבֶל וְהָיוּ לָאַל יַרְחֵי עָמָל וָעֹצֶב.

אָכֵן לֹא מָצָא עוֹד דַּיּוֹ יֻפִּיטֶר בְֹּשָמָיו, אָז אָחִיו

275 לוֹבֵשׁ-הַתְּכֵלֶת אֶת גַּלָיו יָבִיא לְעֶזְרָה, יְמַהֵר

יַקְהִיל אֶת אֵלֵי-הַנְּהָרוֹת, וְאַךְ אֶל הֵיכָלוֹ יָבוֹאוּ,

יִפְנֶה אֲליהֶם וְאָמַר: "לֹא עֵת הִיא לְהַכְבִּיר בְּמִלִּים,

אָכֵן זֹאת עַתָּה עֲשׂוּ: יִתְפָּרֵץ מַעְיַנְכֶם הַחוּצָה,

פִּתְחוּ בָּתֵּיכֶם וְגֹלּוּ אֶת אַבְנֵי-הַסְּגוֹר מֵאֲפִיקֵי

280 נַהֲרוֹתֵיכֶם, וְשָׁטְפוּ פָּרֹץ וְשָטֹף בְּאֵין-מַעֲצוֹר".

שָׁמְעוּ הָאֵלִים וְיַחַד יָחִישׁוּ וּפָתְחוּ לִרְוָחָה

מַעְיְנוֹת הַנְּהָרוֹת וְשָׁטְפוּ קַלֵּי-הַמֵּרוּץ הַיָּמָה.

קָם גַּם נֶפְּטוּנוּס וַיַּךְ הָאָרֶץ בַּקִּלְשׁון. הִתְפּוֹרְרָה

פִּתְאֹם הָאָרֶץ וַתְּפַלֵּג לְשֶׁטֶף הַמַּיִם תְּעָלוֹת.

285 פָּרֹץ הִתְפָּרְצוּ הַנְּהָרוֹת וְיִסְחֲפוּ אָחוּ וּשְׁדֵמָה,

יִגְרְפוּ קָמָה וָשִׂיחַ, עַשְׁתְּרוֹת-צֹאן וּבְנֵי-אָדָם,

בָּתִּים וּמִקְדְשֵׁי-אֵל יַחַד עִם פְּסִילֵי תַרְפֵיהֶם.

גַּם אִם יִמָּלֵט מֵהֶרֶס אַחַד הַבָּתִּים וְעָמַד

הָכֵן עַל כַּנּוֹ, עַל גַּגּוֹ יֶהֱמוּ יֶחְמְרוּ גַלִּים;

290 שַׁחוּ גַּם מִגְדְּלֵי-רֹם וְשֶׁטֶף מְצוּלָה יְכַסֵּם.

שֻׁדַּד כָּל גְּבוּל, לֹא עוֹד יַפְרִיד בֵּין מַיִם וְאֶרֶץ-יַבֶּשֶׁת,

הָיָה אַךְ יָם עַל אַפְסָיִם, אֵין חֹק לוֹ, אֵין חוֹף לוֹ מִסָּבִיב.

יֵשׁ כִּי יִמָּלֵט לוֹ אִישׁ בַּגִּבְעָה וּבְסִירַת-הַדּוּגָה

יֵתַע מִשְׁנֵהוּ בִמְקוֹם הֶאֱרִיכוּ מַעֲנִיתָם הַחוֹרְשִׁים.

295 יֵשׁ וְיָשׁוּטוּ וּפָגְעָה סִירָתָם בְּקָמָה אוֹ בַיִת

נוֹפֵל וְתִרְזָה, בִּסְבַךְ צַמַּרְתָּהּ דְּגֵי-יָם נֶאֱחָזוּ.

עֹגֶן כִּי תַשְׁלִיךְ אֳנִיָּה וְנָפַל בָּאָחוּ, וִירֵכָהּ,

יֶרֶךְ-גַּבְנוּנִים, תְּבוֹסֵס גֶּפֶן-אַדֶּרֶת נִסְחָפָה.

שָׁם בִּמְקוֹם גְּדִיּוֹת נֶחְמָדוֹת לִחֲכוּ עֵשֶׂב יְרַקְרַק,

300 יַלְדֵי-הַתַּחַשׁ מָשְׁחֲתֵי-הַמַּרְאֶה יָנוּחוּ יִרְבָּצוּ.

עַלְמוֹת-הַיָּם תִּשְׁתּוֹמַמְנָה לַבָּתִּים, לַחֹרְשׁוֹת, לַקְּרָיוֹת,

יֶחֱצוּ הַגַּלִּים; בְּסִבְכֵי הַיַּעַר יְקַנֵּנוּ דֹולְפִינִים,

שָׁם בֵּין עֳפָאִים יְטַפְּסוּ, וְרָעַד תַּחְתֵּיהֶם הָאַלּוֹן.

יִשְֹחֶה זְאֵב בֵּין רְחֵלִים וְנָשָׂא הַזֶּרֶם אֲרָיוֹת,

305 יִשָּׂא נְמֵרִים; אֵין מוֹעִיל לַחֲזִיר בְּשִׁנָּיו כִּבְרָקִים,

חִנָּם גַּם רַגְלֵי הַצְּבִי, כִּי טֻבְּעוּ בְמַיִם זֵדוֹנִים.

צִפּוֹר נוֹדֶדֶת לַשָּׁוְא אַךְ תֵּתַע וַתְּבַקֵּשׁ מָנוֹחַ,

תַּשְׁפִּיל כְּנָפֶיהָ עֲיֵפוֹת וְנָפְלָה אֵין-אוֹנִים הַיָּמָּה.

יִזְעַף הַיָּם וְלֹא יֵדַע מַעֲצוֹר לְרוּחוֹ וְכִסָּה

310 גְבָעוֹת, וְשֶׁצֶף הַגַּלִּים עַל בָּמֳתֵי הָרִים יִתְגָּעֵשׁ.

הָיְתָה כָל נֶפֶשׁ לְטֶרֶף גַּלֵּי הַתְּהוֹם, וְאִם יִהְיֶה

שָׂרִיד וְנָפַל גַּם הוּא בֵין חַלְלֵי רָעָב וְכָפָן.

דֵוְקַלְיוֹן וּפִירָה

בֵּין אֲאוֹנְיָה וְחֶבֶל אַקְטֶיָה שְׂדוֹת פוֹקִים יִשְׂתָּרְעוּ,

אֶרֶץ פּוֹרִיָּה מֵאָז, אַךְ בָּאוּ כְפֶתַע הַגַּלִּים,

315 שָׁדְדוּ אַדַּרְתָּהּ וַתְּהִי לְמִפְרַץ הַיָּם הַמִּתְגָּעֵשׁ.

שָׁם הַר פַּרְנַסוּס יִתְנוֹסֵס וְנָשָׂא שְׁנֵי רָאשֵׁי גִבְעוֹתָיו

עַד הַשָּׁמַיִם וְחָצוּ פִּסְגוֹתָיו אֶת חַשְׁרַת הֶעָבִים.

הֵנָּה, מִשּׁוּט בָּאֳנִיָּה קַלָּה בֵין גַּלִּים, יְכַסּוּ

אֶרֶץ וּמְלוֹאָהּ מִסָּבִיב, בָּאוּ דֵוְקַלְיוֹן וְאִשְׁתּוֹ.

320 תְּפִלָּה לְנִימְפוֹת הֶהָרִים נָשְּׂאוּ בְרֶגֶשׁ, לָאֵלִים

שׁוֹכְנֵי הָהָר וּלְתֶמִיס, הַחוֹזָה אֲזַי שָׁם בָּאוּרִים.

הָיָה דֵוְקַלְיוֹן אִישׁ תָּם וַישַׁר-הַלֵּבָב אֵין כָּמֹהוּ,

אֵין גַּם כְּפִירָה בָאָרֶץ אִשָּׁה יִרְאַת-הָאֵלִים.

רָאָה יֻפִּיטֵר, כִּי אַגְמֵי-מַיִם פְּנֵי-תֵבֵל יְכַסּוּ,

325 וּמֵרִבְבוֹת אָדָם אִישׁ אֶחָד אַךְ נִשְׁאַר לִפְלֵטָה,

אַחַת הָאִשָּׁה גַם הִיא מֵרְבָבוֹת לִפְלֵטָה נִשְׁאָרָה,

שְׁנֵיהֶם נְקִיִּים מֵחֵטְא וְיִרְאֵי הָאֵלִים גַּם שְׁנֵיהֶם, –

הֵפִיץ הֶעָבִים אַף שָׁלַח בָּהֶם אֶת אַקְוִילוֹ לְגָרְשָׁם,

גִּלָּה פְנֵי שָׁמָיו לָאָרֶץ וּשְׁחָקִים הִבִּיטוּ פְּנֵי תֵבֵל.

330 עָמַד מִזַּעְפּוֹ הַיָּם, וּמוֹשֵׁל בְּגֵאוּת הַגַּלִּים

יִשְׁלַח מִפָּנָיו אֶת נִשְׁקוֹ שְׁלָשׁ-הַשִּׁנַּיִם, וּדְמָמָה

תְהִי עַל הַמָּיִם. אָז יִקְרָא לִטְרִיטוֹן וְעָלָה מִמְּצוּלָה,

תְּכֵלֶת לְבוּשׁוֹ וּבְנֵי שַׁבְּלוּלִים יְכַסּוּ הַשֶּׁכֶם.

יִקְרָא הָאֵל וְצִוָּהוּ לִתְקֹעַ בְּקוֹנְכָה הוֹמִיָּה

335 תְּרוּעָה לְמַסַּע הַגַּלִּים וּלְהָשִׁיב הַיָּם אֲחוֹרַנִּית.

יִקַּח אָז טְרִיטוֹן שׁוֹפָרוֹ, סוֹחֶרֶת-הַשַּׁבְּלוּל הָעוֹלָה

עָלֹה וְרָחֹב, אֲשֶׁר אִם בְּטַבּוּר הַיָּם בָּהּ יִתָּקַּע,

תַּקִּיף הַתְּרוּעָה שְׁתֵּי גְדוֹתָיו מִמִּזְרַח הַשֶּׁמֶשׁ עַד בֹּאוֹ.

לַחוֹת שִׂפְתֵי הָאֵל מֵרְסִיסֵי הַזָּקָן, וְהָיָה

340 אַךְ בִּשְׁפוֹפֶרֶת תִּגַּעְנָה לִתְקֹעַ וּלְהָשִׁיב הַמָּיִם –

צָלְלָה הַתְּרוּעָה וַיֵּלֶךְ שִׁמְעָהּ בְּגַלִּים רְחוֹקִים,

גַּלֵּי יַבֶּשֶׁת וָיָם וַהֲמוֹתָם וּגְאוֹנָם הִשְׁבִּיחָה.

קַלּוּ הַמַּיִם וְנִרְאוּ רָאשֵׁי הֶהָרִים וְעָלוּ

חוֹפֵי הַיָּם וּמְלֹא אֲפִיקֵיהֶם הַנְּהָרוֹת יִזְרֹמוּ.

345 רַחֲבֵי אֶרֶץ יִגָּלוּ מִתַּחַת הַמַּיִם כִּי דַלּוּ,

גַּם הַיַּעַר, זֶה נֶחְבָּא יָמִים כַּבִּירִים, יַגְבִּיהַּ

רֹאשׁוֹ וְהוֹצִיא לַחָפְשִׁי עֲנָפָיו מְכֻסֵּי-הָרֶפֶשׁ.

חֻדְּשׁוּ פְנֵי-תֵבֵל. וַיְּהִי כִּי רָאָה דֵוְקַלְיוֹן הָאָרֶץ

בָּדָד יוֹשֶׁבֶת שׁוֹמֵמָה וּדְמָמָה רוֹבֶצֶת מִסָּבִיב,

350 נָזְלוּ דְּמָעוֹת מֵעֵינָיו וּלְאִשְׁתּוֹ אֲמָרִים יַבִּיעַ:

"הוֹי הַשְּׂרִידָה בָאָרֶץ, אֲחוֹתִי וְאֵשֶׁת נְעוּרַי!

יַחַד מַקֶּבֶת בּוֹר אֶחָד נֻקַּרְנוּ, אַחַר קְשָׁרַתְנוּ

בְּרִית-נְעוּרֵינוּ, וְעַתָּה אִחְדוּנוּ הַשֹּׁד וְהַשֶּׁבֶר!

נַחְנוּ רַק נַחְנוּ הַשְּׁנַיִם יוֹשְׁבֵי הַתֵּבֵל הִנֵּנוּ, –

355 בָּלַע הַיָּם אֶת הַכֹּל מִמִּזְרַח הַשֶּׁמֶשׁ עַד מְבוֹאוֹ.

אַף גַּם אֲנַחְנוּ עוֹד תְּלוּאִים חַיֵּינוּ מִנֶּגֶד וְטֶרֶם

שָׁלֵו נֵדָעָה, כִּי עוֹד יַרְגִּיזוּ הֶעָבִים רוּחֵנוּ.

הוֹי, לוּ נֶחְלַצְתְּ מִכִּלָּיוֹן רַק אַתְּ וְאֵין אֲנִי אִתָּךְ,

מַה הִשְׁתּוֹחֲחָה עַתָּה נַפְשֵׁךְ, עֲזוּבָה! מַה פָּחַד

360 לִבֵּךְ וְרָגַז בְּקִרְבֵּךְ וְאֵין לָךְ מְנַחֵם בַּצָּרָה!

כִּי אֲנִי הֵן נִשְׁבַּעְתִּי, לוּ שָׁטֹף שְׁטָפָתֵךְ שִׁבֹּלֶת,

אָז גַּם אָנֹכִי אַחֲרַיִךְ הָלָכְתִּי וָאֵרֶד מְצוּלָה!

הוֹי, מִי יִתְּנֵנִי כְאָבִי בּוֹרֵא נְשָׁמוֹת, כִּי עַתָּה

חַיִּים נָפַחְתִּי בְּגָלְמֵי אֲדָמָה – וְחָיוּ בְנֵי-אָדָם!

365 אָכֵן כֹּה יָעַץ הָאֵל וּמִי יְשִׁיבֶנּוּ: הָיִינוּ

פְּלֵטַת בְּנֵי-תְמוּתָה רַק שְׁנֵינוּ וּשְׁאֵרִית הָאָדָם בָּאָרֶץ!"

כָּכָה יְדַבֵּר וַיִּבְכּוּ שְׁנֵיהֶם. אָז נוֹעֲצוּ יַחְדָּו

לָשֵׂאת תְּפִלָּתָם לְאֵל הַשָּׁמַיִם וְלִשְׁאוֹל בָּאוּרִים.

מַהֵר יְמַהֲרוּ לֶכְתָּם וּבָאוּ אֶל נְהַר הַקֶּפִיסוּס,

370 עֲדַיִּן דְּלוּחִים מֵימָיו, אַךְ שָׁבוּ אֶל גְבוּלָם מִקֶּדֶם –

יִשְׁאֲבוּ שָׁם מִן הַמַּיִם וּזְרָקוּם עַל רֹאשָׁם וִיחַטְּאוּ

כְסוּתָם, וְאַחַר יָרִימוּ פַעֲמֵיהֶם וְהָלְכוּ אֶל מִקְדַּשׁ

תֶּמִיס הָאֵלָה הַקְּדוֹשָׁה וְאֵזוֹב מְגֹאָל אֶת גַּגּוֹ

יְכַסֶּה, אַף נִדְעֲכָה אֵשׁ הַמִּזְבֵּחַ הַשָּׁמֵם.

375 הִנֵּה אֶל מַעֲלוֹת הַהֵיכָל קָרְבוּ וַיִּפְּלוּ עַל אַפַּיִם

אָרְצָה וְלָאֶבֶן הַקָּרָה יְנַשְּׁקוּ בְיִרְאָה וָרֶגֶש.

"הוֹי, אִם יֵשׁ כִּי יֵעָתְרוּ – קָרְאוּ דֵוְקַלְיוֹן וְאִשְׁתּוֹ –

אֵלִים לִתְחִנָּה צוֹדֶקֶת, יִנָּחְמוּ וְשָׁב חֲרוֹן-אַפָּם, –

גַּלִּי-נָא, תֶּמִיס, אָזְנֵינוּ, בַּמֶּה יִוָּשַׁע הָאָדָם,

380 אַף, רַבַּת-חֶסֶד, הַחִישִׁי עֶזְרָתְךָ לְתֵבֵל הַמוּצֶפֶת!"

נָגְעוּ הַדְּבָרִים עַד לֵב הָאֵלָה וַתִּתֵּן אֶת דְּבָרָהּ:

"עִזְבוּ הַהֵיכָל, רָאשֵׁיכֶם כַּסּוּ וּפִתְחוּ חֲגוֹרְכֶם,

עַצְמוֹת הוֹרַתְכֶם הַגְּדוֹלָה אַחֲרֵי גֵוְכֶם הַשְׁלִיכוּ".

נִבְעֲתוּ שְׁנֵיהֶם, נָדַמּוּ. אָז פִּירָה הַדְּמָמָה תַפְרִיעַ,

385 קוֹלָהּ תְּחַנֵּן לָאֵלָה וּסְלִיחָה תְבַקֵּשׁ, כִּי מָרְתָה,

מֵאֲנָה שְׁמֹר פִּקּוּדֶיהָ, אָכֵן הֲיַעֲרֹב לִבָּהּ

הַשְׁלֵךְ אֶת עַצְמוֹת הוֹרָתָהּ וְהַכְלֵם אֶת צִלָּה בַּקֶּבֶר?

עוֹד דְּבַר הָאֵלָה עַל לְשׁוֹנָם וְהָגוּ בוֹ יַחְדָּו לִרְגָעִים,

הָגֹה וְחָקֹר אֶת דִּבְרֵי הֶחָזוֹן הַחֲתוּמִים וּסְתוּמִים.

390 אָז עַל לֵב אִשְׁתּוֹ יְדַבֵּר בֶּן-פְּרוֹמֶתֶּוְס וְאָמַר:

"רוּחִי אִם נָכוֹן בְּקִרְבִּי וְיֶחֱזֶה-לִי לִבִּי נְכֹחוֹת,

טְהוֹרוֹת הֵן אִמְרוֹת הֶחָזוֹן, טְהוֹרוֹת לֹא יָעֲצוּ אָוֶן:

כִּי הָאֲדָמָה – אִמֵּנוּ הַגְּדוֹלָה, וְאַבְנֵי הַשָּׂדֶה

הֵן עַצְמוֹתֶיהָ, כִּי יָדּוֹת אֲבָנִים צִוָּנוּ הֶחָזוֹן".

395 יָשְׁרוּ בְּעֵינֵי טִיטַנְיָה פִּתְרוֹנִי דֵוְקוֹלְיוֹן, אַךְ לִבָּהּ

עוֹד יְסִיתֶנָּה בְֹּשֶפֶק, כִּי זָרִים נִשְׂגָּבִים מִשְּׁנֵיהֶם

דִּבְרֵי הָאֵלִים, אֵין אֵמוּן בְּלִבָּם. אַךְ מַה מֵהֶם יַהֲלֹךְ?

יַחַד אָז יֵצְאוּ, יְכַסּוּ הָרֹאשׁ וַחֲגֹרָתָם יְפַתְּחוּ,

אַף לַאֲחוֹרֵיהֶם, כִּמְצֻוָּה, אַבְנֵי-אֲדָמָה יַשְׁלִיכוּ.

400 פִּתְאֹם – מִי יַאֲמִין לַשְּׁמוּעָה לוּלֵי הַדּוֹרוֹת מִקֶּדֶם

יִתְּנוּ עֵדֵיהֶם? – הִתְמוֹגֵג מְטִיל-הָאֲבָנִים הַמּוּצָק,

הָלַךְ קִפְאוֹנָן וָרַךְ וְתֹאַר וּדְמוּת לָמּוֹ יִהְיוּ.

הָלֹךְ וְנָמֹג תֵּלַכְנָה, וּכְהִמֵּס רִגְבֵיהֶן וְרַכּוּ

תֵּלֵךְ גַם דְּמוּתָן הַכֵּהָה הָלֹךְ וָאוֹר, וּכְמוֹ מַרְאֵה

405 אָדָם מַרְאֵיהֶן, וְהָיוּ כְגֹלֶם הַשַּׁיִשׁ בְּהָחֵל

חָרָשׁ מְלַאכְתּוֹ, כַּפֶּסֶל בְּטֶרֶם עִצְּבָהוּ הַיּוֹצֵר.

הוֹיָה יַד-פְּלָאִים בָּאֶבֶן וְרִגְבֵי הֶעָפָר, כִּי רַכִּים

הִנָּם וְלַחִים, וַיִּהְיוּ – לְמַפְּלֵי הַבָּשָׂר בַּגְּוִיָּה,

חַלְמִישׁ-הָאֶבֶן הֶחָזָק וּמוּצָק – לְעֶצֶם בַּבָּשָׂר

410 גִּידֵי הָאֶבֶן גַּם הֵם – לְגִידִים בִּגְוִיַּת הָאָדָם.

מִהֲרָה קָמָה עֲצַת הָאֵלִים, וּבְהַשְׁלֵךְ הַגֶּבֶר

גָּלְמֵי הָאֶבֶן וְהָיָה צֶלֶם זָכָר לָהֵמָּה,

וּמֵאֲבָנִים הִשְׁלִיכָה הָאִשָּׁה – נְקֵבָה נִבְרָאָה.

לָכֵן גַּם גֶּזַע מוּצָק וְחָזָק הוּא גִּזְעֵנוּ וְעָשׂוּי

415 הוּא לִבְלִי חָת, לֹא יְנַכֵּר אוֹתוֹתָיו, כִּי אֶבֶן מְקוֹרוֹ.

פתון

יֶתֶר כָּל חַיָּה וְעוֹף מְעוֹפֵף יָלְדָה הָאָרֶץ,

אַחֲרֵי יֶחֱמַתָּה לַהַב הַשֶּׁמֶשׁ וַיֵּחַם לְשַׁדָּהּ:

תִּבְעֶה הָאֵשׁ אֶת מֵי הָאֲגַמִּים וְצָבָה הָרֶפֶשׁ,

אָז גַּם יִתְעוֹרְרוּ זַרְעִי הַחַיִּים הַמַּפְרִים בְּקֶרֶב

420 חֵיק הוֹרָתָם, מִלְּשַׁד אֲדָמָה יִינְקוּ וְגָדְלוּ

הָלֹךְ וְגָדֹל וּדְמוּת יְשַׁוּוּ לָהֶם לְמִינֵיהֶם.

כָּכָה הַחוֹרֵשׁ גַּם הוּא, עֵת נִילוּס וְשִׁבְעַת נְחָלָיו

יַשְׁקוּ הַשְּׁדֵמוֹת וְשָׁבוּ אֶל חֵיק אֲפִיקֵיהֶם וְיַחֵם

שֶׁמֶשׁ-הַשְּׁחָקִים הַלּוֹהֵט מִשְׁמַנֵּי הָרֶפֶשׁ, – אָז מָצֹא

425 יִמְצָא בֵין מִגְרְפוֹת אֵתוֹ תַּרְבּוּת רְמָשִׂים וָשָׁרֶץ.

יֶשְׁנָם וְהֵם אַךְ נוֹלְדוּ וְרָמְשׂוּ מִתּוֹךְ הָאֲדָמָה,

יֶשְׁנָם – עוֹד טֶרֶם נָכוֹנוּ יְצוּרֵי גָלְמֵיהֶם, וְהָיָה

חֵלֶק הַגְּוִיָּה – נְשָׁמָה בְקִרְבָּהּ וְיִתְרָהּ עֲדֶנָה

רִגְבֵי אֲדָמָה, אֲשֶׁר אֵין חַיִּים וְרוּחַ בָּהֵמָּה.

430 לֵח וָחֹם כִּי יִתְחַבְּרוּ יַחְדָּו כַּחֹק – וְהִזְרִיעוּ

זֶרַע וְיֻלְּדוּ מֵהֶם כָּל תּוֹלְדוֹת הַחַיִּים בָּאָרֶץ.

אָמְנָם מִלְחָמָה לָאֵשׁ בַּמַּיִם, אַךְ חֹרֶב וָלֵחַ –

יוֹצְרֵי הַכֹּל הֵם, וְיֻכַּן הַשָּׁלוֹם בְּצֶאֱצָאֵיהֶם.

כָּכָה גַם אַדְמַת-הַבֹּץ, בְּיַחֲמָהּ מֵחֹם וּמִשָּׁרָב,

435 רָעֲפוּ שַׁחֲקֵי-הָרֹם כַּעֲבוֹר שׁוֹט-הַמַּבּוּל, וַתּוֹצֵא

חַיָּה וְעוֹף לְמִינֵיהֶם, אֲשֶׁר לֹא יִסָּפְרוּ, וּתְחַדֵּשׁ

צַלְמָם וּדְמוּתָם כְּקֶדֶם אוֹ בָרְאָה מִפְלָצוֹת חֲדָשׁוֹת.

אָז – וְאִם הָיָה לְמוֹרַת-רוּחָהּ הַדָּבָר – גַּם אוֹתְךָ,

פִּתּוֹן, נְחַשׁ-הָעֲנָק, הָאָרֶץ חוֹלְלָה; הִקִּיפוּ

440 מֻטּוֹת בַּדֶּיךָ הֶהָרִים וּמָגוֹר לָאָדָם הָיִיתָ.

אֵל חֲמוּשׁ-קֶשֶׁת הִצְמִיתְךָ – אֲשֶׁר כּוֹנֵן חִצָּיו עַד הֵנָּה

רַק אֶל עֳפָרֵי-הַצְּבָאִים, אֶל גְּדִיּוֹת-הַשָּׂדֶה הַקַּלוֹת –

הֵרִיק עָלֶיךָ אֶת בְּנֵי-אַשְׁפָּתוֹ, וַיֵז מִפְּצָעֶיךָ

רִיר מְרֵרָתְךָ הַשְּׁחוֹרָה וְחָלָל נָפַלְתָּ לָאָרֶץ.

445 חַגִּים וּמוֹעֲדֵי-מִשְׂחָק אָז הוֹעִיד הָאֵל לְזִכָּרוֹן,

זֵכֶר מִפְעָלוֹ כִּי עָז, וְקֹרָא לְחַגֵּי-הַקֹּדֶשׁ

מִשְׂחֲקֵי “פִתִּים”, כְּשֵׁם הַנָּחָשׁ, הִכְרִיעָה אֶזְרוֹעוֹ.

שָׁמָּה בְּאֶגְרוֹף יִתְחָרוּ, בְּמֵרוּץ, בָּרֶכֶב, וְהָיָה

הָאַלּוֹן הַיָרֹק עֲטֶרֶת-נִצָּחוֹן לְרֹאשׁ הַבַּחוּרִים.

450 טֶרֶם עוֹד יִהְיוּ אֲזַי הַדְּפַנִים, וְעֵצִים אֲחֵרִים

יִתְּנוּ לְפֶבּוּס עֲלֵיהֶם לְזֵר עַל תַּלְתַּלֵי-הַחֶמֶד.

דפני

רֵאשִׁית אֲהָבִים לְפֶבּוּס הָיְתָה דַפְנֵי בַּת פֶּנֶוְס.

לֹא יַד הַמִּקְרֶה, אַךְ זַעַם קֻפִּידוֹ לְפֶבּוּס נְתָנָהּ.

גָּבַהּ לֵב דֶלְיוּס, כִּי רִצְצוּ חִצָּיו הַנָּחָשׁ, וְהִנֵּה

455 רָאָה בְאַחַד הַיָּמִים – קֻפִּידוֹ גַם הוּא יִדְרֹךְ קֶשֶׁת.

"מַה לְךָ, נַעַר מִתְהוֹלֵל, לְךָ וּלְנֶשֶׁק אַבִּירִים?

"לִי הוּא לְפָאֵר אֶת כְּתֵפִי, לִי קֶשֶׁת-גִּבּוֹרִים יָאָתָה –

"לַצְמִית מְשַׂנְּאַי מִפָּנַי, לִמְחֹץ פְּרִיץ-חַיּוֹת כִּי יָקוּם!

"הִנֵּה אָנֹכִי חֲצָצַי פָּלְּחוּ לֵב פֶּתֶן אַכְזָרִי,

460 "רוֹבֵץ עַל מֶרְחֲבֵי אֶרֶץ וּמְזָרֶה חֲמָתוֹ מִסָּבִיב.

"אַתָּה דַיֶּךָּ כִּי לַפִּיד בְּיָדְךָ לְעוֹרֵר אֲהָבִים,

“לָמָּה-זֶה תַסִּיג גְּבוּל כְּבוֹדִי וּבְמַעֲשֵׂי-יָדַי תִּתְפָּאֵר?”

עָנָה בֶּן וֵנוּס: "לוּ יִנְחֲתוּ, פֶבּוּס, בַּכֹּל חֲצָצֶיךָ,

"חִצַּי – אֶת לִבְּךָ יִמְחֹצוּ! כִּגְבֹהַּ חִין-עֵרֶךְ הָאֵלִים

465 “מִן כָּל הַחַיָּה בָאָרֶץ, כֵּן כְּבוֹדִי אֲנִי – מִכְּבוֹדֶךָ!”

אָמַר וּתְכֵלֶת-הַזֹּהַר חָצָה בִכְנָפָיו הָרַכּוֹת

קַל עַל-פְּנֵי רוּחַ יִנָּשֵׂא אֶל רֹאשׁ הַפַּרְנַסוּס הַמּוּצָל.

שָׁמָּה מֵחִצֵּי אַשְׁפָּתוֹ יִבְחַר לוֹ שְׁנַיִם: הָאֶחָד

עוֹרֵר יְעוֹרֵר אֲהָבִים, מִשְׁנֵהוּ – אֵשׁ-אַהֲבָה יְכַבֶּה.

470 שָׁנוּן הָאֶחָד וְזָהָב מִתְנוֹצֵץ בְּשִׁנּוֹ הַחַדָּה,

קֵהֶה מִשְׁנֵהוּ וּמְטִיל-עוֹפֶרֶת בַּקָּנֶה הַנָּבוּב.

אוֹתוֹ יִשְׁלַח בִּלְבַב בַּת פֶּנֶוְס הַנִּימְפָה, וְחִצּוֹ

נוֹשֵׂא-הָאַהֲבָה יַךְ בְּאַפּוֹלוֹ וּפִלַּח אֶת לִבּוֹ.

אָהֹב אָז יֹאהַב הָאֵל, וָהִיא – לִשְׁמוֹ וּלְזִכְרוֹ

475 תֶּחְרַד וְתָנוּס עַל נַפְשָׁהּ, וְחָמְדָה לָהּ צִלְלֵי-חֹרֶשׁ,

בַּהֲמוֹת יַעַר וְצַיִד, כְּפֶבֵּי תַמָּתִי בָאֵלִים,

תַּעֲנֹד צָנִיף לְרֹאשָׁהּ וְצָנַף קְוֻצָּתָהּ הַפְּזוּרָה.

רַבִּים יְבַקְּשׁוּ קִרְבָתָהּ, אַךְ תַּהֲפֹךְ עֹרֶף לְכֻלָּם,

תֵּתַע בְּסִבְכֵי יְעָרוֹת, וְתִרְחַק וְתִפְחַד מִגֶּבֶר,

480 זָרוּ לָהּ דּוֹדִים וְאַהֲבָה וּמַעֲדַנּוֹת אֵל הִימֵן.

פַּעַם בְּפַעַם לָהּ יֹאמַר אָבִיהָ: “הוֹי, חָתָן לִי הָבִי!”

פַּעַם בְּפַעַם לָהּ יֹאמַר: “בִּתִּי, הוֹי, הָבָה לִי נֶכֶד!”

אֶפֶס כְּחֵטְא בְּעֵינֶיהָ חֲתוּנָה וְדוֹדִים, וְאָדַם

זֹּךְ לְחָיֶיהָ הָרַכּוֹת וְלָהַט מִבֹּשֶׁת-עֲלוּמִים,

485 תִּפּוֹל עַל צַוְּארֵי אָבִיהָ, תְּחַבְּקוֹ וְרַכּוֹת תְּדַבֵּר:

"אֲבִי מַחְמַלִּי, הוֹי תֵּן לִי, כְּתֵת גַּם יֻפִּיטֶר לְבִתּוֹ,

אֵשֵׁב בִּבְתֻלַּי כָּל יָמַי וְאֶשְׁמְרָה תֻּמַּת עֲלוּמָי".

נֵאוֹת אָבִיהָ, אַךְ הוֹי, זֶה יָפְיֵךְ לְמוֹקֵשׁ לָךָ יִהְיֶה,

הוֹדֵךְ וְחִנֵּךְ, כִּי רַב הוּא, יָשִׂים לְאַל נְדָרָיִךְ.

490 חָמַד אַפּוֹלוֹ אֶת הוֹדָהּ, אֵלֶיהָ תְּשׁוּקָתוֹ בוֹעֶרֶת,

יַחֲמֹד וְיֶחְשַׁק וִיקַו, אַךְ שׁוֹלָל תּוֹלִיכוֹ תִקְוָתוֹ.

אֵשׁ כִּי תֵצֵא וְאָכְלָה מְלִילוֹת-הַקָּמָה בַשָּׂדֶה

אוֹ בִמְשׁוּכַת-נְצָרִים כִּי תֹאחֵז לַהֶבֶת הַמְּדוּרָה,

הִבְעִיר הַהֵלֶךְ וְהִשְׁכִּים לָלֶכֶת וְאִשָּׁהּ לֹא כִבָּה,

495 כָּכָה בְּנֶפֶשׁ אַפּוֹלוֹ כִּידוֹדֵי-שַׁלְהֶבֶת הִתְלַקְּחוּ,

תִּיקַד בּוֹ תִּקְוָה כּוֹזֶבֶת וּתְלַהֵט לַשָּׁוְא אַהֲבָתוֹ.

מַחְלְפוֹת-רֹאשָׁהּ כִּי יִרְאֶה פְּרוּעוֹת תַּלְתַּלִים וְקָרָא:

“הוֹי, לוּ תִּקָּלַע קְוֻצָּתֵךְ!” אוֹ יַבִּיט אֶל בְּרַק עֵינֶיהָ

נוֹצֵץ כִּשְׁבִיב הַכֹּכָבִים; אֶל חֶמְדַת שְׂפָתֶיהָ וְאֵין דַּי

500 רַק לַחֲזוֹת בָּן; אוֹ יַחֲמֹד מֵרָחוֹק אֶת עֶדְנַת הָאֶצְבַּע,

לִבְנַת כַּפֶּיהָ וְעֵירוֹם אַצִּילֵי יָדֶיהָ הָרַכּוֹת,

וַאֲשֶׁר כֻּסָּה מֵעֵינָיו – וְצוֹדֵד פִּי-שֶׁבַע אֶת לִבּוֹ.

אָכֵן כְּרוּחַ קַל-כָּנָף הִיא נָסָה מִפָּנָיו, לֹא תִשְׁמַע

רַכֵּי אֲמָרָיו: "הוֹי עִמְדִי, הַנִימְפָה, עֲמֹדִי! שַׁלָּמָּה

505 תִּהְיִי כְעוֹטְיָה, תָּנוּסִי כְּכִבְשָׂה מִזְּאֵב, כָּאַיֶּלֶת

מִן הָאֲרִי אוֹ כְיוֹנַת-אֵלֶם חֲרֵדָה מִנֶּשֶׁר,

מִפְּנֵי רוֹדְפֵיהֶן הֵן נָסוֹת, אֲנִי – אַהֲבָתִי רוֹדַפְתֵּךְ!

הוֹי, אַל תִּכָּשְׁלִי בַמּוֹרָד! הִשָּׁמְרִי, בַּל יִשְׂרֹט הַסִּרְפָּד

רַגְלֵךְ הָרַכָּה, וְאַל-נָא אָנֹכִי אֱהִי בְעוֹכְרָיִךְ.

510 הִנֵּה חַתְחַתִּים בַּדֶּרֶךְ, אַל-נָא תְמַהֲרִי מִנְעִי

רַגְלֵךְ מֵרוּץ וְאֶתְנַהֵל לְאִטִּי אַחֲרַיִךְ גַּם אָנִי!

נִימְפָה! הֲתֵדְעִי, הֲשַׂמְתְּ אֶל לִבֵּךָ, מִי חָמַד אֶת חִנֵּךְ?

הֵן לֹא הֲרָרִי בַשָּׂדֶה, לֹא פֶרֶא הָרוֹעֶה בָעֵמֶק

עֵדֶר הַצֹּאן! הוֹי, יָדֹעַ לֹא תֵדְעִי, הַבַּת הַסּוֹרֵרָה,

515 מִי הוּא הַלָּז,כִּי עַל כֵּן תִּתְחַמָּקִי: אֲנִי – יַעַבְדוּנִי

דֶלְפִי וּקְלַרוֹס, טֶנֶדוֹס גַם קִרְיַת-מְלָכִים פַּטַרָה.

אָבִי – יֻפִּיטֶר; אָנֹכִי רִאשׁוֹנוֹת יָדַעְתִּי וּנְצוּרוֹת

אַגִּיד בְּחָזוֹן; בִּשְׁמִי מֵיתָרִים זְמִירָתָם יַנְעִימוּ.

שָׁנוּן חֵץ קַשְׁתִּי, אַךְ יֶשְׁנוֹ חֵץ אֶחָד – וַיִגְבַּר מִמֶּנִּי,

520 לִבִּי הַשָּׁלֵו כִּי חָלַף וְיִּמְחַץ וַיֶּרֶב הַפֶּצַע.

חָכְמַת הָרוֹפְאִים אָנֹכִי יָצַרְתִּי, וּשְׁמִי עֲלֵי-אָרֶץ

רַב לְהוֹשִׁיעַ, וְלִי כָל צֳרִי וְכָל עִֹשְבֵי רְפוּאָה.

"אוֹי לִי, אֵין צֳרִי לְמַחֲלָתִי, כִּי חוֹלֵה-אַהֲבָה אָנִי,

“רַבִּים אִם כִּי יִוָּשֵׁעוּ, לִי לֹא תוֹשִׁיעַ חָכְמָתִי”.

525 עוֹד הוּא מְדַבֵּר וְהִיא חֲרֵדָה תִּתְחַמֵק מִפָּנָיו,

עֹרֶף לוֹ תִפְנֶה וְכָתֵף סוֹרֶרֶת לְשֶׁטֶף אֲמָרָיו.

תָּנוּס וְיָפְיָהּ מָה רַב: מֵרוּחַ נוֹשֶׁבֶת יִתְנוֹפְפוּ

שׁוּלֵי שִׂמְלָתָהּ הַקַּלָּה וְחֵן מַעֲרוּמֶיהָ יֵחָֹשֵף;

פִּזְּרָה וַתְּסַלְסֵל הָרוּחַ תַּלְתַּלֵי צַמָּתָהּ, וְגָדַל

530 חִנָּהּ פִּי-שֶׁבַע בְּרוּצָהּ, יִנָּחֵם אַפּוֹלוֹ כִּי שִׁחֵת

חֶלְקַת אֲהָבָיו לַשָּׁוְא, וְנִרְגָּשׁ וְנִרְדָּף מִסַּעַר

אַהֲבָה, תָּקַף לְבָבוֹ, יָרוּץ בְּעִקְבֵי הַיָּפָה.

כֶּלֶב בֶּן-גַּלְיָה כִּי יִפְגַע אַרְנֶבֶת בַּשָּׂדֶה וְרָגְזוּ

שְׁנֵיהֶם גַּם יַחַד, – הִיא תָחִישׁ לָהּ מִפְלָט, הוּא יָטוּשׂ אֶל צֵידוֹ;

535 הִנֵּה כִּמְעַט יִתְפְּשֶׂנָּה וְתָקַע בָּה שִׁנָּיו, שָׁלוּחַ

לוֹעוֹ הַפָּעוּר אַחֲרֶיהָ וְאַפּוֹ יְגַמֵּא עֲקֵבָהּ, –

תּוֹעֶה לְבַב הָאַרְנֶבֶת וְחָשְׁבָה בַלֶּכֶד חַיָּתָהּ,

רֶגַע – וַתִּפְרֹץ וַתָּנָס, לַשָּׁוְא שִׁנֵּי-טֶרֶף קִדְּמוּהָ; –

כָּכָה הָאֵל וְהַבְּתוּלָה – אוֹתוֹ הַתִּקְוָה תִשָּׂא

540 וְאוֹתָהּ – הַפַּחַד, וְעֹז-אַהֲבָתוֹ לוֹ יִתֵּן כְּנָפַיִם,

יָטוּשׂ וְהָיָה כְפֶשַׂע בֵּינוֹ וּבֵין צֵידוֹ, וְנָשְׁבָה

רוּחַ-נִשְׁמָתוֹ בִּקְוֻצּוֹת-הַחֵן עֲלֵי חֶלְקַת צַוָּארָהּ.

עָיְפָה הַנִּימְפָה וְחָוְרוּ פָּנֶיהָ, מֵרוּץ וּבָרֹחַ

כֹּחָהּ עֲזָבָהּ, בִּרְעָדָה אֶל גַּלֵי פֶּנֶוְס תִּתְחַנֵּן:

545 "חוּשָׁה, אוֹי, אָבִי, עָזְרֵנִי! אִם יֶשְׁנָהּ יַד אֵלִים בִּנְהָרוֹת

“תַּרְאֶה נִפְלָאוֹת וְאָסְפָה אֶת יָפְיִי, לִי נֶהְפַּךְ לְמַשְׁחִית!”

עוֹד אֲמָרֶיהָ בְּפִיהָ וַתְּהִי תֹּמֶר-מִקְשֶׁה גְּוִיָּתָהּ,

תַּעֲטֶה קְלִיּפַת-עֵץ רַכָּה אֶת עֶדְנַת עֲלוּמֵי חָזֶהָ,

הָיְתָה הַזְּרוֹעַ לְעָנָף וְשַׂעֲרוֹת-רֹאשָׁהּ לְעָלִים,

550 שֹׁרְשׁוּ בָאָרֶץ וְנִצְּבוּ בְּלִי-נוֹעַ רַגְלֶיהָ הַקַּלוֹת,

רֹאשָׁהּ כָּאָמִיר עָלֶיהָ – רַק עוֹרָהּ עוֹד יִקְרַן כְּקֶדֶם.

אָכֵן לֹא יֶחֱדַל הָאֵל מֵאֱהֹב וְהִתְרַפֵּק עָלֶיהָ,

הַלְמוּת לְבָבָהּ הֶחָרֵד יַקְשִׁיב מִתּוֹךְ עֲבִי-גִּזְעָהּ,

חַבֵּק יְחַבֵּק אֶת עַנְפֵי-הָאֵלֶם וְעֵצָהּ יְנַשֵּׁק,

555 אוּלָם מַצֶּבֶת-הָעֵץ גַּם הִיא מִנְּשִׁיקוֹתָיו תִּתְחַמֵּק.

רַכּוֹת יְדַבֵּר לָהּ פֶבּוּס: "אִם אֵשֶׁת-נְעוּרִים אֲהוּבָה

"לִי לֹא הָיִית וַתְּמָאֵנִי, יְהִי נָא עֵצֵךְ לִי קֹדֶשׁ,

"יִהְיוּ הַדְּפַנִּים זֵר רֹאשִׁי, עֲדִי לְכִנּוֹרִי וְתֶלְיִי,

"יָצִיץ עַל מֵצַח גִּבּוֹרִים נִזְרֵךְ הָרַעֲנָן, עֵת הֲמוֹן

560 "חוֹגֵג יוֹלִיכֵם אֶל מְרוֹם קָפִּיטוֹליוּם בִּתְרוּעָה וּרְנָנִים.

"שַׁעֲרֵי הֵיכַל אוֹגוּסְטוּס שָׁם תַּעֲמֹדִי, כְּשׁוֹמֵר

"נֶאֱמָן סוֹכֵךְ עַל בֵּיתוֹ ובְצִלֵּךְ הָאַלּוֹן יִתְלוֹנָן.

"תְּהִי רַעֲנַנָּה כִפָּתֵךְ וּמְלֵאָה זִיו-נֵצַח כְּרֹאשִׁי

“עוֹטֵה-עֲלוּמִים, אֲשֶׁר אֶת נִזְרוֹ לֹא חִלֵּל הַתַּעַר”.

565 דִּבֵּר אֵל פֵּיאַן וְנַעֲנוּ עַנְפֵי הַדְּפַנִים, וּבְנַחַת

הָנֵד יָנִיד הָעֵץ אֶת רֹאשׁוֹ יְפֵה-הַצַּמָּרֶת.

איוֹ

יֶשְׁנָהּ בְּאֶרֶץ הֵימוֹנְיָה בִּקְעָה וְטֶמְפֵּי תִּקָּרֵא,

תּוֹעֲפוֹת-יַעַר לָהּ סָבִיב, שָׁם גַּלֵּי פֶּנֶוְס מִבֶּטֶן

פִּינְדוּס יָגִיחוּ, יַרְתִּיחוּ לִבְנַת-הַקֶּצֶף, בְּרַעַשׁ

570 יֵרְדוּ אַשְׁדוֹתָיו וְהֶעֱלוּ אֵדִים וְעַרְפְּלֵי-עָנָן,

יִזּוּ עַל מְרוֹמֵי הַיַּעַר רְסִיסֵי-הַזֶּרֶם וּסְבִיבָיו

נַהַם וּתְשׁוּאוֹת וְרָעַשׁ וְהָלַךְ הַקּוֹל לְמֵרָחוֹק.

פֹּה אֶל-הַנְּהָרוֹת יִסְתּוֹפֵף, פֹּה זְבוּלוֹ וְהֵיכָל אַפַּדְנוֹ:

חֶבְיוֹן מְעָרָה בֵין כֵּפֵי הַסֶּלַע – שָׁם יֵשֵׁב לַכִּסֵא,

575 יַפְקִיד אֶת צִבְאוֹת מִשְׁבָּרָיו וִיצַוֶּה עַל נִימְפוֹת בַּגַּלִּים.

נִקְבְּצוּ הֵנָּה הַנְּהָרוֹת. בְּרֹאשָׁם הַקְּרוֹבִים יָבֹאוּ –

סְפֶרְכֵי גְדָל-תְּרָזוֹת, אֶנִיפֶוְס הַרַגָּז וְגַם אֶרִידַנּוּס,

יָשִׁישׁ בַּנְּהָרוֹת, אַמְפְרִיסוּס הַשָּׁקֵט גַם אֵיאַם – וַיָּפָן

לִבָּם: הַיְקַדְּמוּ בְּרִנָּה אוֹ נַחֵם יְנַחֲמוּ אֶת פֶּנֶוְס.

580 יֶתֶר הַנְּהָרוֹת גַּם הֵם יָבֹאוּ מִמְּקוֹם שָׁם יְשָׂרְכוּ

דַּרְכָּם וְנָשְׂאוּ אֶל חֵיק הַיָּם אֶת גַּלֵּיהֶם הַתּוֹעִים.

אִינָךְ רַק הוּא לֹא יָבֹא בַבָּאִים. בְּסֵתֶר מְעָרָה

יִבְכֶּה תַמְרוּרִים וְגָאוּ מֵימָיו מִדְּמָעוֹת: עַל בִּתּוֹ

אִיּוֹ יֵבְךְּ כִּי אֵינֶנָּה. לֹא יֵדַע, אִם עוֹדָהּ בַּחַיִּים

585 אוֹ עִם רְפָאִים תָּנוּחַ. הוֹי אָבְדָה מִמֶּנּוּ, אָב שַׁכּוּל,

לֹא עוֹד יִרְאֶנָּה לָנֶצַח! וּדְאָגָה נוֹרָאָה תְקָפַתְהוּ.

פָּגַע יֻפִּיטֶר בַּנַעֲרָה בְּשׁוּבָה מִנַּחַל אָבִיהָ,

פָּגַע וַיִּקְרָא: "הַבַּת בָּהּ בָּחַר יֻפִּיטֵר, וְאַשְׁרֵי

אִישׁ כִּי יִרְוֶה מִנֹעַם דּוֹדָיִךְ! הוֹי, בֹּאִי אֶל תַּחַת

590 צִלְלֵי הַחֹרֶשׁ הָרָם (הוּא נָטָה יָדוֹ לְיָּעַר),

הִנֵּה עוֹד גָּדוֹל הַיּוֹם וּבוֹעֵר הַשֶּׁמֶשׁ בִּמְרוֹמָיו!

אַל-נָא תֵחַתִּי מֵחַיַּת-הַיַּעַר רוֹבֶצֶת מַחְשַׁכִּים,

בֶּטַח לְכִי לָךְ, כִּי אִתָּךְ אֲנִי, אֵל-מָגֵן עָלַיִךְ,

אֵל וְעֶלְיוֹן עַל אֵלִים, הַמּוֹשֵׁל בִּגְאוֹן הַשָּׁמַיִם,

595 כִּסְאִי – הַשַּׁחַק וְיָדִי חֲזִיזִים וָרַעַם תְּחוֹלֵל.

אַל-נָא תָּנוּסִי, הַיָּפָה" אַךְ בָּרְחָה הַנַּעֲרָה. כְּבָר עָבְרָה

רִבְצֵי עֲדָרִים עִם לֶרְנָה וִיעָרוֹת לְרַגְלָי הַלִּירְקֵאוּס,

תָּרוּץ, – אַךְ פָּרַשׂ הָאֵל עֲנָנָיו, עֲלָטָה רוֹבֶצֶת

סָבִיב, וְהִנֵּה – נִלְכָּדָה, וְתֹם-עֲלוּמֶיהָ מְחֻלָּל.

600 תָּמַהּ הִשְׁתּוֹמֵם לֵב יוֹנוֹ כִּי רָאֲתָה שְׂדוֹת הָעֲמָקִים

לוֹטִים בָּאֹפֶל וְחַשְׁרַת-עֲרָפֶל תִּפְרשֹׁ כְּנָפֶיהָ,

תַּחְשִׁיךְ אוֹר שֶׁמֶשׁ, וְלֹא מִן הַנָּהָר יִתְקַבְּצוּ הָאֵדִים

לֹא מֵעַל פְּנֵי הָאֲדָמָה יִתַּמְּרוּ, כִּי שָׁתְתָה לִרְוָיָה.

אָז עַל סְבִיבֶיהָ תִּתְבּוֹנֵן וּבִקְּשָׁה אֶת אִישָׁהּ, כִּי יָדְעָה

605 רַבּוּ מְאֹד תַּעֲלוּלָיו, רַבּוּ מִתְּמוֹל וּמִשִּׁלְשׁוֹם.

אָכֵן אַיֵּהוּ? אֵינֶנּוּ! "הַאֻמְנָם גְּאוֹנִי יְחַלֵּל?

אוֹ אַךְ שָּׁגִיתִי?" תִּתְמַרְמֵר הָאֵלָה וּתְמַהֵר וְיָרְדָה

אָרְצָה, שָׁם תִּגְעַר בְּעַב-הָעֲרָפֶל וְהָלַךְ וְנָמֵס.

רָאָה יֻפִּיטֶר אֶת יוֹנוֹ וְהִנֵּה הִיא קְרוֹבָה וּבָאָה,

610 מִהֵר וַיַּהְפֹּךְ אֶת פְּנֵי בַת אִינָךְ לְעֶגְלָה נֶחְמָדָה.

יָפָה הָעֶגְלָה וּמָלֵא פִי יוֹנוֹ תְּהִלָּתָהּ, אִם אָמְנָם

לַמְרוֹת לְבָבָה תְּהַלְּלֶנָּה וְשָׁאֲלָה יוֹנוֹ בְעָרְמָה,

מָה הִיא הָעֶגְלָה, מֵאַיִן בֹּאָהּ וּמִי בְעָלֶיהָ.

כִּחֵשׁ כִּי יָלְדָה הָאָרֶץ אוֹתָהּ לְמַעַן לֹא תּוֹסֵף

615 עוֹד חֲקוֹר. אָז תִּשְׁאַל מִמֶּנּוּ אֶת עֶגְלַת-הַחֶמֶד לְמִנְחָה.

נָבוֹךְ יֻפִּיטֶר. הֲיַסְגִיר יָפָתוֹ בִידֵי צוֹרַרְתָּהּ?

אוֹ הֲיֶחְדַל – וְרַב הַחֶשֶׁד! יִתְרוֹצְצוּ בְקִרְבּוֹ

בָשְׁתּוֹ וְעֹז-אַהֲבָתוֹ וַתִּגְבַּר הָאַהֲבָה בְלִבּוֹ.

אֲבָל הֲיִמְנַע מִנְחָתוֹ מֵאֵשֶׁת-נְעוּרָיו? – הֵן הָבֵן

620 תָּבִין כִּי דְבַר בְּלִיַּעַל יָצוּק בָּעֶגְלָה הַיָּפָה!

הִנֵּה צָרָתָהּ בְּיָדָהּ, אַךְ טֶרֶם עוֹד תִּשְׁלַו הָאֵלָה,

פַּחַד תְּפַחֵד מִנִּכְלֵי יֻפִּיטֶר. אָז תִּתֵּן לְאַרְגוּס

בֶּן אַרֶסְטָרוּס עֶגְלָתָהּ וּתְהִי לְמִשְׁמֶרֶת בְּיָדוֹ.

מֵאָה עֵינַיִם לְאַרְגּוּס וְעָטְרוּ אֶת רֹאשׁוֹ מִסָּבִיב.

625 שְׁתַּיִם שְׁתַּיִם מֵהֶן תֵּרָדַמְנָה בְרֹאשׁוֹ חֲלִיפוֹת,

יִתְרָן צוֹפִיּוֹת וּפְקוּחוֹת תָּמִיד וְשָׁמְרוּ מִשְׁמַרְתָּן.

בַּאֲשֶׁר יַעֲמֹד הַשּׁוֹמֵר, עֵינָיו אֶת אִיּוֹ תִרְאֶינָה,

אָחוֹר כִּי יִפְנֶה, וְהָיְתָה גַם אָז לִפְנֵי עֵינָיו הַפְּקוּחוֹת.

רָעֹה יִרְעֶנָּה עִם שַׁחַר, עִם שֶׁמֶשׁ – וּסְגָרָהּ עַל מַסְגֵּר,

630 כְּבָלִים יָעִיקוּ מוּעָקָה עַל צַוְּארֵי הָעֶגְלָה הָרַכָּה.

עָלֶה וָצִיץ מַאֲכָלָהּ אוֹ תִקְטֹף מַלּוּחַ בַּשָּׂדֶה,

תִּצְמָא לְמַיִם – וְהָיוּ אַגְמֵי מֵי-רֶפֶשׁ מִשְׁתֶּהָ,

צְחִיחַ-אֲדָמָה אֵין חָצִיר שָׁם תִּשְׁכַּב לָהּ לְמַעֲצֵבָה.

גַּם אִם אָמְרָה תִּתְחַנֵּן לְאַרְגוּס, בְּמוֹעַל-כַּפֶּיהָ

635 אֵלָיו תִּתְפַּלֵּל, – אֲבוֹי, לֹא הָיוּ כַפַּיִם לָעֶגְלָה!

וּבְהִתְעַטֵּף בָּהּ נַפְשָׁהּ וְלִבְכּוֹת תְּבַקֵּשׁ, וְהָיְתָה

נַאֲקַת הַבָּקָר נַאֲקָתָהּ וְהֶחְרִיד הַקּוֹל אֶת לְבָבָהּ.

יֵשׁ וְהִיא רֹעָה עַל גְּדוֹת הָאִינָךְ, שָׁם יָדְעָה מֵעוֹדָהּ

שְׂחוֹק וְשַׁעֲשׁוּעִים, וְהָיָה כִּי תִרְאֶה בְּגַלֵּי הַנָּהָר

640 שְׁתֵּי הַקַּרְנַיִם לְרֹאשָׁהּ – וְחָרְדָה וְנָסָה מִפָּחַד.

זָרָה הִיא לִבְנוֹת-הַגַּלִּים, וְאִינָךְ אָבִיהָ

לֹא יַכִּירֶנָּה גַם הוּא. וְשָׁאֹף תִּשְׁאַף אֲלֵיהֶם,

תִּדְבַּק בָּהֶם אַף תִּתְרַפֵּק, וְהֵם אַךְ תָּמֹהַּ יִתְמָהוּ.

צִיצִת עֲשָׂבִים לָהּ יוֹשִׁיט אִינָךְ הַיָּשִׁישׁ, אִיּוֹ

645 זְרוֹעוֹת אָבִיהָ תְּלַחֵךְ, אֶת כַּפָּיו תְּנַשֵּׁק, וּדְמָעוֹת

זוֹלְגוֹת עֵינֶיהָ. לוּ הָיוּ מִלִּים בְּפִיהָ וַתַּגֵּד

שְׁמָהּ וְהַקּוֹרוֹת אֲשֶׁר עֲבָרוּהָ וְקָרְאָה לְעֶזְרָה!

אָכֵן רַק תָּחֹק בְּרַגְלָהּ אֶת שְׁמָהּ בֶּעָפָר וְהָיָה

עֵד לַהֲפֵכָה נוֹרָאָה אֲשֶׁר הֲפָכַתָּהּ יָד-אֵלִים.

650 "אוֹי לִי, נִדְמֵיתִי! –יֵאָנַח מָרָה הָאָב וְהִתְרַפֵּק

עַל הַקַּרְנַיִם הָרַכּוֹת, עַל לִבְנַת צַוְּארֵי הָעֶגְלָה –

"אוֹי לִי – יֵאָנַח לִרְגָעִים – הַאַתְּ הִיא הַבַּת לֹא נֻחָמָה?

בַּקֵּשׁ בִּקַּשְתִּיךְ בְּכָל רַחֲבֵי הָאָרֶץ, מִי יִתֵּן

וְלֹא עוֹד נִמְצֵאת וּכְאֵבִי לֹא יִּרֶב הַיּוֹם… הוֹי, שַׁלָּמָה

655 הַחֲרֵשׁ תַּחֲרִישִׁי, אֵין אֹמֶר וּדְבָרִים, רַק תִּקְרַע אֲנָחָה

לִבֵּךְ הַסּוֹעֵר וְגָעֹה אַךְ תִּגְּעִי בְדַבְּרִי אֵלָיִךָ?

שָׁלֵו הָיִיתִי וָאֶשָּׂא אֶת נַפְשִׁי לְעִתֵּךְ עֵת-דּוֹדִים,

הָיָה חֲתָנִי – תִּקְוָתִי, וְנֶכְדִּי – תּוֹחֶלֶת-לְבָבִי.

עַתָּה בָּעֵדֶר תִּשְׁגְּלִי, בָּעֵדֶר פְּרִי-בִטְנֵךְ תַּמְלִיטִי.

660 הוֹי, מִי יִתְּנֵנִי אָמוּתָה וְאֶשְׁכַּח עֲמָלִי בַקֶּבֶר,

אָכֵן לֹא אָדָם, כִּי אֵל אָנֹכִי, וַיָּסֶךְ הַמָּוֶת

שַׁעֲרֵי-שַׁחַת בְּפָנַי, וְהָיָה לָנֶצַח כְּאֵבִי".

עוֹדוֹ מִתְלוֹנֵן וְאַרְגוּס הַמְכֻכָּב מִזְּרוֹעוֹת הָאָב

יִגְזֹל אֶת אִיּוֹ וְנִהֵל אוֹתָהּ אֶל עֵמֶק-הַמִּרְעֶה.

665 שָׁם עַל רֹאשׁ גִּבְעָה לוֹ יֵשֵׁב אַרְגּוּס וְעֵינָיו הַפְּקוּחוֹת

אוֹרְבוֹת צוֹפִיּוֹת מֵרָחוֹק, נִבָּטוֹת עַל-פְּנֵי כָל הַכִּכָּר.

נִלְאָה אֵל-מָרוֹם, לֹא יוּכַל רְאוֹת בֶּעֱנוּת אֲהוּבָתוֹ,

יִקְרָא לִבְנוֹ, זוּ פְּלֶיַס רַבַּת-הַזִּיו יְלָדַתּוּ,

אַף יְצַוֵּהוּ עַל אַרְגּוּס, לְהַכְחִיד מֵאֶרֶץ חַיָּתוֹ.

670 מִהֵר מֶרְקוּרְיוּס וְשָׂם כְּנָפַיִם לְרַגְלָיו וַיַּעֲנֹד

קוֹבַע לְרֹאשׁוֹ וּבְיָדוֹ שַׁרְבִיטוֹ הַמַּפִּיל תַּרְדֵּמָה.

קַל-הַכְּנָפַיִם יִתְעוֹפֵף מֵהֵיכַל אָבִיהוּ וְיָרַד

אָרְצָה, שָׁם יָסִיר מֵעָלָיו קוֹבָעוֹ וּשְׁתֵּי כַנְפֵי רַגְלָיו,

אִתּוֹ רַק שַׁרְבִיט-הַפְּלָאִים וְהֶחֱזִיק בַּכָּף לֹא יַרְפֶּנּוּ.

675 גְּדִיוֹת כִּי יִפְגֹשׁ וְדָלַק בַּשַּׁרְבִיט אַחֲרֵיהֶן, כְּרֹעֶה

יִרְעֶה אֶת עֶדְרוֹ בַשָּׂדֶה, הָלֹךְ וְחַלֵּל בֶּחָלִיל.

קֶסֶם בֶּחָלִיל, אֶת נֶפֶשׁ אַרְגּוּס תְּצוֹדֵד זִמְרָתוֹ.

"בֹּא-נָא, תְּהִי אֲשֶׁר תֶּהִי – יְקַדְמֵהוּ אַרְגּוּס – שֵׁב אִתִּי

פֹּה עַל הָאֶבֶן. כָּר נִרְחָב וְדָשֵׁן, רְאֵה, לְפָנֶיךָ,

680 חָצִיר לְעֵדֶר הַצֹּאן וּמַחֲסֶה מִשֶּׁמֶשׁ לָרֹעֶה".

יֵשֵׁב מֶרְקוּרְיוּס לָנוּחַ, וְשִׁיחוֹת וּדְבָרִים לְמַכְבִּיר

יַרְבֶּה עַל אַרְגּוּס שׁוֹמְעֵהוּ, וְשָׁפַךְ עַל עֵינָיו הָעֵרוֹת

קֶסֶם שִׁירָתוֹ הַנְּעִימָה, שִׁירַת הַגֹּמֶא הַנָּבוּב.

אָכֵן יִתְאַמֵּץ עוֹד אַרְגּוּס, יִתְגַּבֵּר עַל קֶסֶם הַשֵּׁנָה,

685 רַבּוֹת מֵעֵינָיו אִם סֻגְּרוּ בְחַבְלֵי תַּרְדֵּמַת-הַפֶּלֶא,

יִתְרָן עוֹד פְּקוּחוֹת וְעֵרוֹת. אָז יִשְׁאַל אֶת רֵעוֹ הַמְנַגֵּן,

מָה הִיא מְלֶאכֶת הֶחָלִיל (כִּי טֶרֶם עוֹד הָיָה בָאָרֶץ).

יָשִׁיב אֲמָרָיו הָאֵל: "עַל הָרֵי אַרְקַדְיָה הַקָּרִים

יַעַר נוֹנַקְרִיס יִתְנוֹסֵס, שָׁם מִשְׁכַּן דְּרִיַּדּוֹת, וַתְּהִי שָׁם

690 אַחַת וּשְׁמָהּ יוֹצֵא לְפָנֶיהָ וְסִירִינְכְּס תִּקָּרֵא.

פַּעַם בְּפַעַם, בִּרְדֹף אַחֲרֶיהָ סַטִּירִים אוֹ אֵלִים

שׁוֹכְנֵי הַיַּעַר הַמֵּצַל וְשַׁדְמוֹת-הַבָּר, וְהִתְחַמְּקָה

סִירִינְכְּס מִפְּנֵיהֶם, כִּי אֲחֹז תֹּאחַז בְּדַרְכֵי דִיאַנָּה,

תִּשְׁמֹר תֻּמָּתָהּ כָּמוֹהָ, כָּמוֹהָ גַּם תִּלְבַּשׁ מַדֶּיהָ,

695 אִישׁ כִּי יִרְאֶנָּה וַחֲשָׁבָהּ לָאֵלָה לַטּוֹנְיָה, רַק קַשְׁתָּהּ –

קֶשֶׁת עֲשׂוּיָה מִקֶּרֶן, וְקֶשֶׁת הָאֵלָה – כְּלִיל זָהָב.

אָכֵן בְּאַחַד הַיָּמִים וְהִיא מֵעַל הָרֵי לִיקֵיאוּס

תֵּרֵד וְהִנֵּה בָא פַּן לִקְרָאתָהּ, עַל רֹאשׁוֹ זֵר-קוֹצִים,

קוֹצֵי הָאֹרֶן, וַיֹאמַר…"וְסַפֵּר יְסַפֵּר הַמְנַגֵּן:

700 בָּזְתָה הַנַּעֲרָה לְדִבְרֵי פַּן וּבִמְנוּסָה תִמָּלֵט,

תָּרוּץ לָהּ תֹּהוּ לֹא-דֶרֶךְ עַד נַחַל הַלַּדּוֹ, הַמּוֹלִיךְ

זִרְמוֹת מֵימָיו בֵּין חוֹפֵי הַחוֹל, וְתִתְחַנֵּן אֲלֵיהֶן,

אֶל אַחְיוֹתֶיהָ הַזַּכּוֹת, לַהֲפֹךְ צֶלֶם תַּבְנִיתָהּ.

יָדוֹ כְּבָר שָׁלַח אֵלֶיהָ פַּן לְתָּפְשָׂהּ וְהִנֵּה –

705 תַּחַת הַנִּימְפָה הָרַכָּה – גִּבְעוֹל הַגֹּמֶא בְיָדוֹ.

נֶאֱנַח מָרָה הָאֵל. וְרוּחַ תְּנַשֵּׁב בַּגֹּמֶא,

נַשֵּׁב וְהָמֹה בַּקָּנִים, וּתְלוּנָה חֲרִישִׁית הֶמְיָתָהּ.

חָרַד אֵל-פַּן וַיִּשְׁתּוֹמֵם עַל נֹעַם הַקּוֹלוֹת וַיִּקְרָא:

“יִהְיוּ הַקּוֹלוֹת הָאֵלֶּה מָסֹרֶת בְּרִית-אַהֲבָה לָנוּ!”

710 אֵלֶּה הֵן תוֹלְדוֹת הֶחָלִיל, אֲגֻדַּת קְנֵי-סוּף, – אֲשֶׁר דֻבְּקוּ

חַד אֶל אֶחָד בַּדּוֹנַג, וְקָרָא שֵׁם סִירִינְכְּס עֲלֵיהֶם.

עוֹדוֹ מְדַבֵּר וַיַּרְא קִילֶנְיוּס וְהִנֵּה – כְּבָר עָצַם

אַרְגּוּס אֶת עֵינָיו כֻּלָּהַן, בְּחַבְלֵי תַּרְדֵּמָה סֻגָּרוּ.

הֶחֱרִישׁ מֶרְקוּרְיוּס מִדַּבֵּר, וּלְמַרְבֵּה הַשֵּׁנָה וְנוֹפֵף

715 שַׁרְבִיט הַקֶּסֶם עַל עֵינֵי אַרְגּוּס אֲחוּזוֹת-הַתְּנוּמָה.

אַחַר יְמַהֵר וְשָׁלַף אֶת חַרְבּוֹ וְהִכָּה בְאַרְגּוּס,

הַכֵּה בַּצַּוָּאר וְהָדֹף אוֹתוֹ מֵאֶבֶן-מוֹשָׁבוֹ

נִפְצָע וְנִשְׁבָּר, וְדָמָיו עַל-פְּנֵי כָל הַסֶּלַע נִשְׁפָּכִים.

נָפֹל נָפַלְתָּ, אַרְגּוּס! חָשְׂכוּ מְאוֹרוֹת עֵינֶיךָ,

720 אֶחָד הַמָּוֶת – וַיְכַס בְּמַשָּׁאוֹן עֵינֶיךָ הָרַבּוֹת.

יוֹנוֹ תִקָּחֵן אֵלֶיהָ וְהָיוּ עֲדִי לְטַוָּסָהּ,

רִקְמַת נוֹצָתוֹ וְתוֹרֵי-כֹכָבִים לִזְנָבוֹ הַנֶּחְמָד.

אָכֵן רַב קֶצֶף הָאֵלֶּה וַתְּשַׁלַּח בְּאִיּוֹ צָרָתָהּ

זָעַם אֵרִינִיס נוֹקֶמֶת, מְכַלָּה עֵינַיִם וָנֶפֶשׁ.

725 אֵימִים יַרְגִּיזוּ לְבָבָהּ וּמְנוּסַת-בֶּהָלָה הִיא נָסָה,

תֵּלֵךְ, תָּנוּעַ בָּאָרֶץ וְשָׂבְעָה אַךְ פַּחַד וּנְדוּדִים.

רַק אֶל הַנִּילוּס לֹא בָּאָה עוֹד כָּל יְמֵי-נְדוּדֶיהָ,

וְכַאֲשֶׁר הִגִּיעָה עַד גְּדוֹתָיו וַתִּכְרַע שָׂם עַל בִּרְכֶּיהָ,

גָהֲרָה אַרְצָה וּמָרוֹם הוֹשִׁיטָה צַוָּארָהּ הַקָּצָר,

730 זָקְפָה אֶת רֹאשָׁהּ – עוֹד יָכְלָה זָקְפֵהו ּ– אֶל מוּל הַכֹּכָבִים.

מַר תֵּאָנֵחַ וּדְמָעוֹת דּוֹלְפוֹת עֵינֶיהָ וְגָעֹה

תִגְעֶה תַמְרוּרִים וְהָיְתָה כִמְבַכָּה וּמְבַקְּשָׁה עַל נַפְשָׁהּ

לִפְנֵי יֻפִּיטֶר. אָז יָקוּם, יְחַבֵּק בְּאַהֲבָה צַוְּארֵי

אִשְׁתּוֹ וְחִלָּה פָנֶיהָ לַחְדֹּל מֵאִיּוֹ הַנְּגוּעָה,

735 אַף יִשָּׁבַע אֱמוּנִים וְהָעֵד אֶת סְטִיכְּס נְהַר-שַׁחַת:

“אַל-נָא תִּפְחָדִי, לֹא תְהִי עוֹד הַנַּעֲרָה לִמְגִנַּת-לְבָבֵךְ!”

שָׁכַךְ חֲרוֹן-אַף הָאֵלָה, וּכְרֶגַע וַיָּשָׁב לְאִיּוֹ

צֶלֶם תַּבְנִיתָהּ מִקֶּדֶם: סָר מֵעַל עוֹרָהּ הַצֶּמֶר,

נִגְדְּעוּ קַרְנֵי-הָעֵגֶל וַתִּקְטַן גַּם בָּבַת הֶעָיִן,

740 שָׁבוּ לָה לִהְיוֹת גַּם פֶּה, גַּם כָּתֵף וְיָד כְּמִקֶּדֶם,

תַּחַת הַפַּרְסָה – אֶצְבָּעוֹת, חָמֵשׁ אֶצְבָּעוֹת נִבְקָעוּ,

נִכְחַד צֶלֶם הָעֶגְלָה, לֹא נוֹתַר אַךְ לֹבֶן פָּנֶיהָ,

גַּם עוֹד לֹא תֵלֵךְ עַל אַרְבַע וּזְקוּפָה קוֹמָתָהּ הַיְשָׁרָה.

רַק עוֹד תְּפַחֵד לִפְתֹּחַ שְׂפָתֶיהָ בִּדְבָרִים, פֶּן תִּגְעֶה

745 גָעֹה כְעֶגְלָה, וְחֶרֶֹש אַךְ תִּפְטֹר, תִּלְחַשׁ מִלֶּיהָ.

עַתָּה לְאֵלָה תֵּחָשֵׁב וַיְכַהֲנוּ פְאֵר כֹּהֲנֶיהָ.

פאתון

אָז – כֵּן יְסֻפַּר – גַּם יָלְדָה אִיּוֹ אֶת בְּנָהּ, אֶת אֶפַּפוּס,

בֶּן לְיֻפִּיטֶר, וְלוֹ הֵיכְלֵי-קֹדֶשׁ בֶּעָרִים הַנִּבְנוֹת

יַחַד עִם מִקְדְּשֵׁי אִמּוֹ. וְאִתּוֹ פַאֵתוֹן בֶּן-שֶׁמֶשׁ

750 נַעַר כְּגִילוֹ וּכְלִבּוֹ הַגֵּאֶה. וַיְהִי בְּהִתְפָּאֵר

עָלָיו פַאֵתוֹן, כִּי הוּא לֹא נוֹפֵל מִמֶּנּוּ וּפֶבּוּס

אָבִיו הִנֵהוּ, וַיִּכְעַס אֶפַּפוּס וְיֹאמַר: "לַשָּׁוְא אַךְ

תַּאֲמִין, סָכָל, בְּאִמְּךָ, לַשָּׁוְא עַל אָבִיךָ תִּתְגָּאֶה"…

אָדְמוּ פְנֵי רֵעוֹ, אַךְ עָצַר פַאֵתּוֹן בְּרוּחוֹ, אֶל אִמּוֹ

755 יֵלֵךְ, יִתְאוֹנֵן בְּאָזְנֶיהָ, כִּי לָעַג לוֹ, בָּז לוֹ אֶפַּפוּס:

"הוֹי, הֲלֹא יִכְאַב לְבָבֵךְ, פִּי שֶׁבַע הֵן יִכְאַב, כִּי בְנֵךְ הוּא

בְּנֵךְ זֶה גְבַהּ-הַלֵּבָב, וַיִּשְׁמַע בְּחַלֵּל מְנָאֵץ

כְּבוֹדוֹ וְדָבָר לֹא הֵשִׁיב אֶל חֵיק מְחָרְפֵהוּ וַיִּדֹּם!

עַתָּה אִם אָמְנָם מוֹלֶדֶת-אֵלִים נוֹלַדְתִּי, הוֹי תְּנִי לִי

760 אוֹת וְיָדַעְתִּי, אַף יֵחַד בֵּין אֵלֵי-הַשַּׁחַק גַּם כְּבוֹדִי!"

כָּכָה יְדַבֵּר וּזְרוֹעוֹ עַל צַוְּארֵי קְלִימֶנֵי תִּתְרַפֵּק,

אַף יַשְׁבִּיעֶנָּה בְּחַיֵּי מֵרֹפְּס, בַּחֲתוּנַת אַחְיוֹתָיו,

לָתֵת לוֹ אוֹת יְבַקְּשֶׁנָּה, אוֹת וּמוֹפֵת-אֱמוּנִים.

נָגְעָה עַד לִבָּהּ הַתְּחִנָּה, תְּחִנַּת פַאֵתוֹן, אוֹ אוּלַי

765 דִּבַּת-הַדֹּפִי הִרְגִּיזָה אֶת נַפְשָׁהּ, – וַתָּקָם קְלִימֶנֵי,

פֵּרְשָׂה כַפֶּיהָ אֶל עָל וְעֵינֶיהָ נְשׂוּאוֹת אֶל שֶׁמֶשׁ:

"הִנֵּה נִשְׁבַּעְתִּי לְךָ, בְּנִי, הַיּוֹם בַּשֶּׁמֶשׁ וּבְנֹגַהּ

קַרְנָיו, אֲשֶׁר מִמְרוֹמוֹ יִרְאֵנוּ אַף יַקְשִׁיב קוֹלֵנוּ,

אֵלָיו, בְּנִי, תִּשָּׂא עֵינֶיךָ וְחַיֵּי-תֵבֵל מִיָּדוֹ –

770 אָמְנָם בֶּן שֶׁמֶשׁ הִנֶּךָּ! וְהָיָה אִם שֶׁקֶר דִּבַּרְתִּי,

עֵת רוּחַ-מֶרֶד יְמַלְּאֵם וּבָעֲטוּ בְּכֶפֶל-הָרֶסֶן?

שׁוּב אֵיפוֹא, בְּנִי, מִדְּבָרֶיךָ וּדְרֹשׁ, בְּעוֹד מוֹעֵד, מִיָּדִי,

דְּרֹשׁ לְךָ אַחֶרֶת, כִּי לָמָּה תֵהָפֵךְ מִנְחָתִי לְמַשְׁחִית.

אָמְנָם עֲרֻבָּה תְּבַקֵּשׁ, לְמַעַן תִּוָּכַח וְתֵדַע,

90 כִּי מֵחֲלָצַי יָצָאתָ, – יְהִי אֵפוֹא פַחְדִּי מוֹפֶתְךָ,

פַּחְדִּי וְרֹגֶז-לְבָבִי – הוּא עֵדִי הַנֶּאֱמָן! וּרְאֵה-נָא

פָּנַי אֲשֶׁר חֳמַרְמָרוּ! מִי-יִתֵּן וְחָדְרוּ עֵינֶיךָ

גַּם חַדְרֵי-נַפְשִׁי וְתִרְאֶה בַדְּאָגָה אֲשֶׁר תִּתְקְפֶנָּה!

שׁוּר גַּם פְּנֵי תֵבֵל מִסָּבִיב וּרְאֵה אוֹצְרוֹתֶיהָ מָה-רַבּוּ,

95 שׁוּר וְהַבִּיטָה וּדְרשׁ מִכָּל אֲשֶׁר תִּמְצָא בָאָרֶץ,

יָם וְשָׁמַיִם – וְאֶתֵּן, לֹא אֶמְנַע כָּל-טוֹבָה מֵאִתָּךְ.

לוּ רַק שְׁאֵלָתְךָ הָאַחַת, בְּנִי, אַחֲלַי, שְׁכָחֶנָּה, כִּי קְלָלָה

צְפוּנָה בְתוֹכָהּ. הוֹי, בְּנִי, לֹא מִנְחָה, אַךְ קְלָלָה שָׁאָלְתָּ!

מַה-לְךָ, בֵּן פּוֹתֶה, תִּתְרַפֵּק, תְּחַבֵּק כַּיֶלֶד צַוָּארִי?

100 אַל-נָא תָפוּנָה – רְאֵה, בִּנְהַר-הַתַּחְתִּיּוֹת נִשְׁבַּעְתִּי! –

כָּל אֲשֶׁר-תִּשְׁאַל וְתֵעָשׂ, רַק אַל תְּבַהֵל עַל פִּיךָ!"

כָּכָה יוֹכִיחַ הָאָב, אַךְ אָטַם בְּנוֹ אָזְנוֹ מִשְּׁמֹעַ,

אַחַת יִשְׁאַל, לֹא יָשׁוּב, וְתִבְעַר כָּאֵשׁ תַּאֲוָתוֹ.

יָקוּם אָז פֶבּוּס מִמְּקוֹמוֹ אַחֲרֵי הִתְמַהְמַהּ דַּיּוֹ,

105 יָקוּם וְהוֹלִיךְ אֶת בְּנוֹ אֶל רִכְבּוֹ, מְלֶאכֶת וֻלְקַנּוּס:

זָהָב הַמּוֹט בַּמֶּרְכָּבָה וְזָהָב הַצִּיר לָאוֹפַנִּים,

גַּם חִשׁוּקֵיהֶם – כְּלִיל-זָהָב, וְכֶסֶף – יִתְדוֹת הָאוֹפָן.

פִּטְדָה וְאַבְנֵי-יְקָר בְּעַנְקוֹת הַסּוּסִים יִתְנוֹסְסוּ

וּמִנֹּגַהּ הַשֶּׁמֶשׁ זֹהַר וְיִפְעָה יָפִיצוּ.

110 עוֹד עַל הַמַּרְאֶה יִשְׁתּוֹמֵם הָעֶלֶם וְעֵינָיו תִּבְחַנָּה

פִּלְאֵי הָרֶכֶב, וְהִנֵּה – אֲדַמְדַּם הַמִּזְרָח, אוֹרוֹרָה

תָקוּם שָׁם, תִּפְתַּח דְּלָתֶיהָ, דַּלְתוֹת-אַרְגָּמָן, וְנִגְלָה

אוּלָם-הַשּׁוֹשָׁן. יָנוּסוּ אָז כּוֹכְבֵי-הַלַּיִל, אַחֲרֵיהֶם

יָחוּשׁ לוּקִיפֶר גַּם-הוּא וְעָזַב מִשְׁמַרְתּוֹ בַשְּׁחָקִים.

115 פֶבּוּס יִרְאֵהוּ וְהִנֵּה הוּא עָף אֶל הָאָרֶץ, מִפָּנָיו

אָדְמוּ פְנֵי-תֵבֵל, וְקַרְנֵי הַסַּהַר הַפָּגוּם הָיוּ

כְלֹא-הָיוּ, – אָז יִקְרָא הַטִּיטַן לַהוֹרוֹת הַמְּהִירוֹת וִיצַוֵּן

לֶאֱסֹר רִכְבּוֹ. חִישׁ-קַל תְּמַהֵרְנָה הָאֵלוֹת, תּוֹצֶאנָה

סוּסֵי-הָאֵשׁ מִן אֻרְוָתָם, שְׂבֵעֵי אַמְבְּרוֹסְיָה וְנִשְׁמַת-

120 אַפָּם – לְהָבִים, אַף שָׂמוּ אֶת רִסְנָם הַמְצַלְצֵל עֲלֵיהֶם.

מָשַׁח הָאָב פְּנֵי הַנַּעַר בְּמִשְׁחַת-הַקֹּדֶשׁ לְבִלְתִּי

תִּשְׁלַט הָאֵשׁ הַיּוֹקֶדֶת בִּבְשָׂרוֹ, אַף עָנַד לְרֹאשׁוֹ

צְפִירַת-הַזֹּהַר, וַיֹּאמֶר (וְלִבּוֹ הַסּוֹעֵר בְּקִרְבּוֹ

שִׁבְּרוּ אֲנָחוֹת מְבַשְּׂרוֹת יָגוֹן וְשֹּׁאָה כִּי תָּבֹא):

125 "לוּ אַךְ הַפַּעַם, בְּנִי, תִּשְׁמַע בְּקוֹלִי: הִשָּׁמֵר מְעוֹרֵר

שׁוֹט עַל סוּסֶיךָ, וְהַחֲזֵק אַל תַּרְפֶּה מֵרִסְנָם יָדֶיךָ.

כִּי עַל-נְקַלָּה יֵלֵכוּ, אַךְ מַעֲצֹר אֵין לְזִרְמָתָם.

אַף אַל-נָא תַעֲבֹר דֶּרֶךְ חֲמֵשֶׁת אֵזוֹרֵי-שָׁמַיִם,

יָסוּר הַצִּדָּה נְתִיבְךָ, יִנָּטֶה כְקֶשֶׁת רְחָבָה,

130 שְׁלֹשֶׁת אֲזוֹרֵי-הַשַּׁחַק הֵם יִהְיוּ גְבוּלָיו, שָׁם יֵלֵךְ

הַרְחֵק מֵאַפְסֵי הַנֶּגֶב, מֵעַיִשׁ וְסַעֲרוֹת-צָפוֹן.

זֹאת הִיא הַדֶּרֶךְ בָּהּ תֵּלֵךְ: שָׁם תִּמְצָא גַם עִקְּבוֹת אוֹפַנָּי.

אַךְ אַל-נָא תַשְׁפִּיל הָרֶכֶב, גַּם הָרֵם אַל תָּרִים לְמָעְלָה:

חֵלֶק כְּחֵלֶק תֵּן חֻמְּךָ לָאָרֶץ וּנְאוֹת-הַשָּׁמַיִם, –

135 פֶּן לַשָּׁמַיִם תַּרְקִיעַ וְנִצְּתוּ זְבוּלֵיהֶם, אוֹ תַשְׁפִּיל

אֶרֶץ – תִּשָּׂרֵף: בִּשְׁבִיל-הַבֵּינַיִם לֵךְ בֶּטַח דַּרְכֶּךָ.

סוּר-נָא מִיָּמִין וּרְחַק – עֲקַלָּתוֹן שָׁם רוֹבֵץ הַנָּחָשׁ,

אַף אֶל מִזְבֵּחַ בַּשֵּׁפֶל אַל-נָא תַשְׂמְאִילָה: בַּתָּוֶךְ

לֵךְ אֶת דַּרְכֶּךָ. וּתְחָנְּךָ פוֹרְטוּנָה, – בְּיָדָהּ הִפְקַדְתִּי,

140 בְּנִי, אֶת גּוֹרָלְךָ, הִיא תֵיטִיב מִמֶּךָּ לְהָכִין צְעָדֶיךָ.

הִנֵּה עוֹדֵנוּ מְדַבְּרִים – וּבְפַאֲתֵי מַעֲרָב הַלַּיִל

שָׁב כְּבָר אֶל גְּבוּלוֹ, לֹא נוּכַל עוֹד הִתְמַהְמֵהַּ, הִגִּיעָה

עֵת הָעֲבוֹדָה! אוֹר עָטְתָה אוֹרוֹרָה וַתָּנֶס הָאֹפֶל.

קַח-נָא הַמּוֹשְׁכוֹת וּנְהָג! – אוֹ אוּלַי יֵהָפֵךְ לְבָבְךָ,

145 תָּשׁוּב תִּתְעַשֵּׁת וְתַחַת חָשְׁקְךָ בְּרֶכֶב אָבִיךָ

תִּשְׁמַע עֲצָתוֹ, וְטֶרֶם עָזַבְתָּ מְקוֹם מְנוּחָתְךָ

לַעֲלוֹת גָּבְהֵי מֶרְכַּבְתִּי, חָמַדְתָּ לְךְ לְהַוָּתְךָ, –

תֶּן-נָא וְאָאִיר אָנֹכִי לָאָרֶץ וְאַתָּה הַבִּיטָה".

מִהֵר הַנַּעַר וַיַּעַל, וְקַל הוּא, לֹא יַכְבִּיד הָרֶכֶב,

150 מָרוֹם שָׁם יַעֲמֹד וְשִׁעְשְׁעוּ מוֹשְׁכוֹת הָרֶסֶן אֶת נַפְשׁוֹ,

תּוֹדוֹת יַבִּיעַ לְאָבִיו, אַךְ מֵאֵן אָבִיהוּ בְתוֹדָה.

פִּתְאֹם וְסוּסֵי-הַשֶּׁמֶשׁ קַלֵּי-הַכָּנָף, פִּירוֹאִיס,

אֵאוֹאוּס וְאֵתוֹן וּפְלֶגוֹן, יִתְעוֹרְרוּ וְנַחֲרַת-לְהָבִים

תַּרְעִישׁ הָרוּחַ, וּבָעֲטָה רַגְלָם בִּבְרִיחַ-הַשָּׁעַר.

155 תֶּטִיס לֹא תֵדַע עֲתִידוֹת צָפַן הַגּוֹרָל לְנֶכְדָּהּ, –

נֶחְפְּזָה, פָּתְחָה אֶת סְגוֹר הַבָּרִיחַ, וּמֶרְחֲבֵי תֵבֵל

נִגְלוּ כְרָגַע, – וַיִּדְאוּ הַסּוּסִים, בְּרַגְלָם יְשַׁסְּעוּ

חַשְׁרַת הֶעָבִים, יַמְרִיאוּ, יַעֲלוּ אֵבֶר וְדָלְקוּ

אַחַר הָרוּחַ הַקַּל, יִתְעוֹפֵף מִפַּאֲתֵי-קָדִים.

160 אָכֵן מַה-יֵּקַל הַפַּעַם הָרֶכֶב, לֹא יָעִיק לַסּוּסִים

כֹּבֶד מַשָּׂאוֹ, וְעֻלָּם לֹא יַכְבִּיד גַּם-הוּא כִּתְמוֹל שִׁלְשׁוֹם.

קַלָּה, בְּלִי נֵטֶל כַּחֹק, כִּי-תִּתְנוֹדֵד בַּיָּם הָאֳנִיָּה,

תָּחֹג, תִּנּוֹעַ וּמַעֲמָד לֹא תִמְצָא בְשֶׁטֶף-הַגַּלִּים, –

כָּכָה הִתְנוֹדֵד הָרֶכֶב, כִּי-נֶעְדַּר נֵטֶל מַשָּׂאוֹ,

165 מָרוֹם יְרַקֵּד, יְכַרְכֵּר,וּכְמוֹ אַךְ רֵיק הוּא וְעָזוּב.

רָאוּ הַסּוּסִים וַיִּפְרְצוּ פֶרֶץ אַרְבַּעְתָּם אַף-עָזְבוּ

דַּרְכָּם הַסְּלוּלָה וְעָפוּ בְלִי מִשְׁטַר וְסֵדֶר בַּשְּׁחָקִים.

חָרַד פַאֵתוֹן, לֹא יֵדַע הַדֶּרֶךְ אַיָּהָ, וְאָנָה

יוֹלִיךְ אֶת סוּסָיו, גַּם לִמְשֹׁל בָּהֶם לֹא עוֹד יַעֲצֹר כֹּחַ.

170 עָשׁ וּבָנֶיהָ רִאשׁוֹנָה אָז חַמּוּ מִקַּרְנֵי הַשֶּׁמֶשׁ,

חַמּוּ וְהָרְסוּ הַיָּמָּה אֶל גְּבוּלוֹת לֹא עָבְרוּ מֵעוֹדָם.

נָחָשׁ, זֶה קָפָא כַכְּפוֹר מִנִּשְׁמַת הַצָּפוֹן, בִּכְבֵדוּת

זָחַל אֶת דַּרְכּוֹ וַיֶּחְדַּל מֵהָפִיץ בַּלָהוֹת וָפַחַד, –

הִנֵּה הִתְעוֹרֵר גַּם-הוּא וְחֵמָה בוֹעֶרֶת מְלָאַתּוּ.

175 הוֹי, כִּי גַם-אַתָּה, בּוֹאוֹטֶס, נִבְהַלְתָּ, מִהַרְתָּ לִבְרֹחַ,

אַף כִּי הִסְכַּנְתָּ הִתְנַהֵל לְאִטְּךָ לְרֶגֶל עֶגְלָתְךָ.

נִשְׁקַף פַאֵתוֹן הָאֻמְלָל מִמְּרוֹמָיו לָאָרֶץ, וְהִנֵּה

רָחֲקָה הִרְחִיקָה מִמֶּנּוּ, שָׁם תִּשְׁכֹּן בְּעִמְקֵי-תַחְתִּיּוֹת, –

חִיל אֲחָזָהוּ וּפִיק-בִּרְכַּיִם וּפָנָיו חָוָרוּ,

180 חָשַׁךְ אוֹר-עֵינָיו, לֹא יִרְאֶה בַנְּהָרָה תּוֹפִיעַ מִסָּבִיב.

הוֹי, מִי-יִתְּנֶנּוּ לֹא יִגַּע בְּסוּסֵי-הַשֶּׁמֶשׁ מֵעוֹדוֹ!

לוּ גַם מוֹצָאוֹ לֹא יֵדַע וְחֵפֶץ לְבָבוֹ לֹא יֻתַּן,

לוּ רַק בֶּן-מֵרֹפְּס יִקָּרֵא וְהָיָה זֶה שְׁמוֹ כָל-הַיָּמִים!

יֵדֶא פַאֵתוֹן וְעָף, כָּאֳנִיָּה אֲשֶׁר תְּסָעֲרֶנָּה

185 סוּפָה, וְאָבְדָה כָל-עֵצָה מֵחוֹבֵל וְרַק עַל הָאֵלִים

יַשְׁלִיךְ יְהָבוֹ. כְּבָר הִרְחִיק לָלֶכֶת, וְשֶׁבַע עוֹד תִּגְדַּל

דַּרְכּוֹ לְפָנָיו: יִתְבּוֹנֵן פָּנִים וְאָחוֹר חֲלִיפוֹת –

קֵדְמָה אֶל מִזְרַח-הַשֶּׁמֶשׁ יִשָּׂא מַבָּטָיו, אוֹ יָשׁוּב

יַבִּיט מַעֲרָבָה אֶל גְּבוּל, אֲשֶׁר עוֹד לֹא יַעַבְרֶנּוּ.

190 נִתְּקוּ זִמּוֹתָיו, לֹא יֵדַע, הַיַּחֲזִיק בְּמוֹשְׁכוֹת, גַּם-כֹּחַ

אֵין עוֹד בְּיָדוֹ, גַּם שְׁמוֹת הַסּוּסִים לֹא יֵדַע יְכַנֵּם.

חָרֵד וְנִדְהָם יִתְבּוֹנֵן עַל-סְבִיבָיו וְעֵינָיו תֶּחֱזֶינָה

צַלְמֵי-בַלָּהוֹת בַּשְּׁחָקִים, מִפְלְאוֹת חַיּוֹת כַּבִּירוֹת.

מָקוֹם לוֹ יָדַע הָעַקְרָב בַּשַּׁחַק, שָׁם יִרְבַּץ וְכָפַף

195 כַּפָּיו כָּאַגְמוֹן וּזְנָבוֹ יַחְפֹּץ וְהָיוּ כְקֶשֶׁת,

קֶשֶׁת נְטוּיָה וּגְדוֹלָה, אֲשֶׁר שְׁנֵי כוֹכָבִים בְּתוֹכָהּ.

רָאָה פַאֵתוֹן וְהִנֵּה הֵזִיעַ הָעַקְרָב וּשְׁחוֹרָה

זֵעַת-הָרַעַל, וְאֶרֶס-חֲמָתוֹ שָׁלוּחַ לִפְצֹעַ, –

רָאָה וְלִבּוֹ הִתְפַּלֵּץ וַיַּשְׁמֵט הַמּוֹשְׁכוֹת מִיָּדוֹ…

200 נָפְלוּ הַמּוֹשְׁכוֹת, עַל-גַּבֵּי הַסּוּסִים תִּצְנַחְנָה, אָז נִתְּרוּ

סוּסֵי-הָאֵשׁ וַיִּתְפָּרְצוּ אִישׁ אִישׁ אֶל עֶבְרוֹ, בְּלִי-מַעְצֹר

יִדְאוּ מֵעַל לַאֲרָצוֹת לֹא יָדְעוּ עַד-הֵנָּה, יָעוּפוּ

אֶל אֲשֶׁר רוּחָם יִשָּׂאֵם, יִכָּשְׁלוּ בְּכוֹכְבֵי-הַמָּרוֹם,

תֹּהוּ לֹא-דֶרֶךְ יְטַלְטְלוּ רִכְבָּם, וּפַעַם יַגְבִּיהוּ

205 מַעְלָה מַעְלָה, יַמְרִיאוּ, וּפִתְאֹם יַשְׁפִּילוּ וְיֵרְדוּ,

דֶּרֶךְ חַתְחַתִּים וְשׁוּחוֹת מַטָּה יַשְׁפִּילוּ עַד-אָרֶץ.

תָּמַהּ, הִשְׁתּוֹמֵם הַסַּהַר, כִּי יִרְאֶה, הִשְׁפִּילוּ מִמֶּנּוּ

מֶרְכְּבוֹת אֶחָיו, כִּי נִצְּתוּ הֶעָבִים וַיַּעַל עֲשָׁנָם.

לִהֲטָה אֵשׁ אֶת-כָּל-שְׁפָיִים בָּאָרֶץ, הִתְבַּקַּע, הִתְפּוֹרֵר

210 חֵיק הָאֲדָמָה, נָס לֵחָהּ וַתֶּחֱרַב, לִצְחִיחַ נֶהְפָּכָה.

אֻמְלְלוּ רִבְצֵי-הַצֹּאן, אֵשׁ תִּצַּת בְּסִבְכֵי הָעֵצִים,

קָמַל הַשָּׂדֶה וְטֶרֶף אַךְ יִתֵּן לַלַּהַב הַמַּשְׁחִית.

תִּגְדַּל הָרָעָה: תֶּחֱרַבְנָה גַם עָרִים עִם חוֹמוֹת-מִשְֹגַּבָּן,

מַמְלְכוּת אֶרֶץ וְעַמִּים תְּלַהֵט הַמְּדוּרָה וְהָיוּ

215 עָפָר וָאֵפֶר. אֵשׁ תֹּאכֵל גַּם יַעֲרוֹת-עַד עַל הָרֵיהֶם:

אַתּוֹס וְטוֹרוּס אֲשֶׁר בְּקִילִיקְיָה וּטְמוֹלוּס וְאֶטֵּי,

אִדֵּי, מָה-רַבּוּ מַעְיְנוֹתָיו לְפָנִים, אַךְ יָבְשׁוּ, הֶלִיקוֹן,

הַר-הַעֲלָמוֹת, וְהֵמוּס עוֹד טֶרֶם יִקָּרֵא אֵאַגְרִי;

אֶטְנָה אֵין-גְּבוּל לְשַׁלְהֶבֶת אִשָּׁהּ, שַׁלְהֶבֶת-כִּפְלַיִם,

220 יִבְעַר גַּם אֶרִיקְס וְקִנְתּוּס, פַּרְנַסּוּס וְתָאֳמֵי-רֹאשׁוֹ,

אוֹתְּרִים וּמִימַס וִיזֹרַב הַשֶּׁלֶג עַל בָּמֳתֵי רוֹדֹפֵּי,

יֵקַד מִיקַלֵּי, דִּנְדִּימִים, קִיתֵרוֹן הַר-זֶבַח לְאֵלִים.

סְקִתִּיָּה, גַּם אוֹתָהּ לֹא תַּצִּיל קָרָתָהּ: כַּלַּהַב – קַאוּקַזּוּס,

אוֹסָה אַף פִּנְדּוּס – בָּאֵשׁ גַּם אוֹלִימְפּוּס הָעוֹלֶה עַל שְׁנֵיהֶם;

225 בָּמֳתֵי הָאַלְפִּים וּמְרוֹם אַפֶּנִּינוּס הַשּׁוֹכֵן בֵּין עָבִים.

נִבַּט פַאֵתוֹן וְהִנֵּה בוֹעֶרֶת תֵּבֵל מִסָּבִיב,

גָּדוֹל הַחֹם וְלֹא יוּכַל שֵׂאתוֹ, וְהָיָה כִּי-יִשְׁאַף

רוּחַ וְתִבְעַר כְּאֵֹש הַכִּבְֹשָן בְּקִרְבּוֹ, גַּם רִכְבּוֹ

הָיָה כָּאֵֹש הַיּוֹקֶדֶת וַיִּצְרֹב צָרֶבֶת בִּבְשָׂרוֹ.

230 נִלְאָה הַנַּעַר נְשֹׂא אֶת תִּימְרוֹת הָאֵפֶר וְלַהַט

שְׁבִיבֵי-הָאֵשׁ, וַעֲשַׁן-לֶהָבָה מִסְּבִיבוֹ יִתְאַבֵּךְ,

קַדְרוּת אֲפֵלָה תְּכַסֵּהוּ, לֹא יִרְאֶה אַיֵּהוּ וְאָנָה

יֵלֵךְ אֶת דַּרְכּוֹ, וַיְהִי לְשָׁלָל לְזִרְמַת הַסּוּסִים.

אוֹמְרִים, אֲזַי בְּנֵי-כוּשִׁיִּים חָֹשְכוּ מִשְּׁחוֹר. כִּי-דְמֵי-לִבָּם

235 נִכְמְרוּ קִרְבָּם וַיִּפְרְצוּ פֶרֶץ אֶל עוֹרָם וּבְשָׂרָם.

לוּב אָז הוֹבִישָׁה מֵחֹרֶב וָחֹם, כִּי נִֹשְּתוּ מֵימֶיהָ.

נִזְרָן תָּגֹזְנָה הַנִּימְפוֹת, כִּי שֻׁדְּדוּ יְאוֹר וּמַבּוּעַ:

עָזְבוּ מֵי דִרְקֵי בֵאוֹטְיָה, וּמֵי אֲמֵימוֹנֵי אֶת אַרְגּוּס,

לֹא עוֹד בְּאֶרֶץ אֱפִירֵי גַלֵּי פִירֵנֵי יֶהֱמָיוּ.

240 עָבַר הַשּׁוֹט גַּם עַל נַהֲרוֹת עֹז וּמֻטּוֹת גְּדוֹתֵיהֶם:

יִבְעַר טַנַּאִיס וְגַלָּיו הַשּׁוֹטְפִים יֶעֱשָׁנוּ, כָּמוֹהוּ

כֵּן גַּם פֶּנֶוְס הַיָּשִׁישׁ וְקַיְקוּס בְּחֶבֶל טֶוְתְרַנְיָה,

זֶרֶם-אַבִּירִים אִיסְמֵנוּס וּנְהַר אֱרִיַמַנְטוּס בְּפֶגְיָה,

קְסַנְתּוּס, זֶה שֵׁנִית תְּבוֹאֵהוּ הָאֵשׁ, וְלִיקוֹרְמַס הַצָּהֹב,

245 גַּלֵּי מֵיַאְנְּדְּרוֹס, אֲשֶׁר עֲקַלְקַלּוֹת יִתְהַלְּכוּ יִתְהוֹלְלוּ,

מֶלַּס בְּאֶרֶץ מִגְדּוֹנְיָה וְאֶוְרוֹטַס אֲשֶׁר בְּטֵינַרְיוּם.

אָחֲזָה הָאֵשׁ גַּם בִּפְרָת וַתִּצַּת בְּמֵימֵי אוֹרוֹנְטֶס,

גַּם בְּתֶרְמוֹדוֹן הַמָּהִיר, בְּגַנְגָּס וּפַסִּיס וְהִסְטֵר.

הָיָה אַלְפֶוְס – מֶרְקָחָה, וּגְדוֹת הַסְפֶּרְכֵאוּס – לְמוֹקֵד,

250 נָמֵס, בִּקְדֹחַ בּוֹ אֵשׁ, הַזָּהָב בְּזֶרֶם הַטַּגּוּס,

גַּם הַבַּרְבּוּרִים, אֲשֶׁר הֲמוֹנָם יְמַלֵּא מֵאוֹנְיָה,

נִשְׂרְפוּ לְחוּמֵי-הָאֵשׁ בִּשְׂחוֹתָם בְּמֵימֵי קַאִיסְטְרוּס.

נִבְהַל הַנִּילוּס וַיָּנָס אֶל אַפְסֵי הָאָרֶץ, שָׁם יַסְתִּיר

מוֹצָא מַעְיָנוֹ וְאִישׁ לֹא יֵדָעֵהוּ; אָז הֻכָּה לְשִׁבְעָה

255 נַחֲלֵי-יְשִׁימוֹן, לְשֶׁבַע בְּקָעוֹת אֵין מָיִם בָּהֵנָּה.

יַחַד עִם הֶבְּרוּס וּסְטְרִימוֹן, אֲשֶׁר בְּאִיסְמַרוּס, הֶחֱרִימָה

יַד הַלֶּהָבָה אֶת נַהֲרוֹת הֶסְפֶּרְיָה – רוֹדַנּוּס וְרֵינוּס,

פַּדוּס גַּם טִבֶּרִיס, לוֹ מִנּוּ הַמְשֵׁל וְשִׁלְטוֹן בָּאָרֶץ.

נִבְקְעָה כָל הָאֲדָמָה וְיַחְדֹּר אוֹר-שֶׁמֶשׁ שְׁאוֹלָה,

260 חָדֹר וְהַחֲרֵד אֶת מֶלֶךְ-תַּחְתִּיּוֹת וְעִמּוֹ הַשֵּׁגָל.

רֻשְּׁשׁוּ גְּבוּלוֹת הַיָּם וַיְהִי תַּחַת גַּלָּיו הַזּוֹרְמִים

צְחִיחַ יַבֶּשֶׁת וָחוֹל, וְעָלוּ מֵעִמְקֵי מְצוּלָה

הַרְרֵי-עַד וְעַל אִיֵּי קִקְלַדִּים הַנְּפוֹצִים יִלָּווּ.

תִּשְׁאַף הַדָּגָה אֶל תַּחְתִּית הַיָּם, וְדֹוְלִפינִים גַּבְנֻנִּים

265 עוֹד לֹא יִתְרוֹמְמוּ עַל-פְּנֵי הַמַּיִם וְשָׁאֲפוּ רוּחַ.

חַלְלֵי תְחָשִׁים עַל-פְּנֵי הַתְּהוֹם מְעֻטָּפִים יָצוּפוּ,

גַּם– כֹּה תַגִּיד הַשְּׁמוּעָה – נֵרֵאוּס וְדוֹרִיס וּבְנוֹתָם

מִפְלָט יָחִישׁוּ בִמְעָרוֹת-חֶבְיוֹנָם, אַךְ חַמּוּ גַם-הֵנָּה.

יֵשׁ וִינַסֶּה נֶפְּטוּנוּס לְהוֹצִיא מִתַּחַת הַמַּיִם

270 יָדָיו וּפָנָיו פְּנֵי-זַעַם, אַךְ לַהַט הָאֵשׁ יְנִיסֵהוּ.

אָז אֵם-כָּל-חַי, הָאֲדָמָה, אֲשֶׁר גַּלֵּי-יַמִּים סַבּוּהָ,

נַהֲרוֹת-מַיִם וּמַעְיָנוֹת כָּל-תְּהוֹם מֵעֲבָרִים, תִּתְעוֹרֵר,

תָּרִים פָּנֶיהָ, אֲשֶׁר חֳמַרְמָרוּ – כִּי יָבְשָׁה עַד-צַוָּאר –

כַּסֵּה תְכַסֶּה פָנֶיהָ בְּיָדָהּ, וּבְרֹגֶז וָקֶצֶף

275 תִּגְעַשׁ אַף-תִּרְעַשׁ מִסָּבִיב, תִּשְׁקַע אַף-תַּשְׁפִּיל לָשֶׁבֶת,

קוֹלָהּ נִחָר וְיָבֵשׁ וְכָכָה תַּבִּיעַ מִלֶּיהָ:

"הוֹי, חֲסִין אֵלִים! אִם אָמְנָם חָטָאתִי וְתַחְפֹּץ דַּכְּאֵנִי,

לָמָּה בְרָקֶיךָ לֹא תִשְׁלַח? אִם אֵשׁ וַאֲבַדּוֹן – גּוֹרָלִי,

אַתָּה שְׁלַח אִשְּׁךָ וּנְגַע בִּי וּתְהִי זֹאת נֶחָמָתִי בְּעָנְיִי.

280 הֵן בְּדֵי-עָמָל אֶת שְׂפָתַי אֶפְתַּח בְּדַבְּרִי אֵלֶיךָ!

מַחֲנָק לִי יָשִׂים הֶעָשָׁן, וּרְאֵה – שַׂעֲרוֹתַי נֶחְרָכוּ,

פִּיחַ יְכַסֶּה עַפְעַפַּי וְאֵפֶר קוֹדֵחַ עַל שְׂפָתָי.

הוֹי, הַאִם זֹאת הִיא מַשְׂכֻּרְתִּי מֵאִתְּךָ וּגְמוּל עֲבוֹדָתִי

תַּחַת הֱיוֹתִי נְגוּעָה וְשֵׁן הַמַּחֲרֵשָׁה הַכְּפוּפָה

285 תִּפְצַע, תְּשַׁסַּע לְבָבִי וָאֶשָּׂא בָעֹל כָּל-הַשָּׁנָה,

חָצִיר לַבְּהֵמָה אַצְמִיחַ וּתְנוּבַת פְּרִי-מַאֲכָל לָאָדָם,

מִנְחַת לְבוֹנָה לָאֵלִים לְרֵיחַ-נִיחוֹחַ בְּאַפָּם?

"אַף אִם אָמְרִי חָטָאתִי וּפְרִי מַעֲלָלִי אָכַלְתִּי,

לָמָה עַל גַּלֵּי אָחִיךָ יָצָא הַקֶּצֶף? שַׁלָּמָה

290 רֻשְּׁשׁוּ יַמִּים נַחֲלָתוֹ, מִשְּׁחָקִים מֵימֵיהֶם הָרְחָקוּ?

לוּ אַךְ שָׁמֶיךָ תְּרַחֵם, אִם אוֹתִי לֹא תָּחֹן וְתִשְׁכַּח

אַהֲבַת אָחִיךָ! הַבִּיטָה, אֵשׁ אָחֲזָה בַשַּׁחַק, וְתִימְרוֹת

עָשָׁן הִתְאַבְּכוּ סְבִיבוֹתָם: אִם מוֹט יָמוּטוּ מוֹסְדֵיהֶם,

אָז גַּם זְבוּלֵיכֶם יֻכַּתּוּ! רְאֵה-נָא אֶת צִיר-הַשָּׁמַיִם

295 יוֹקֵד כָּאֵשׁ וּכְבָר נִלְאָה אַטְלַס לְשֵׂאתוֹ עַל שִׁכְמוֹ.

הֵן אִם יַבֶּשֶׁת וָיָם וּמְכוֹן-הַשָּׁמָיִם יֹאבֵדוּ,

יַחַד לְתֹהוּ וָבֹהוּ נָשׁוּב כֻּלָּנוּ! הוֹי, קוּמָה

פְּלֵטַת נַחֲלָתְךָ הַצִּילָה וְהָאֵר לְתֵבֵל פָנֶיךָ!"

דִּבְּרָה דְּבָרֶיהָ וַתֶּחְדַּל, כִּי עָיְפָה מֵאֵשׁ וּמֵעָֹשָן

300 נַפְשָׁהּ וְדַבֵּר לֹא תוּכַל. אָז תַּפְנֶה פָנֶיהָ, תַּסְתִּירֵם

הַרְחֵק בְּעִמְקֵי-תַּחְתִּיּוֹת עַד יַרְכְּתֵי אֶרֶץ-רְפָאִים.

אָז אֵל-הָאֵלִים יִתְעוֹרֵר וְהֵעִיד כָּל שׁוֹכְנֵי-שָׁמַיִם,

גַּם אֶת אֲדוֹן-הַמֶּרְכָּבָה בְתוֹכָם, לְעֵדִים, כִּי צְפוּיָה

תֵבֵל וּמְלוֹאָהּ לְשֹׁד וַאֲבַדּוֹן, אִם עֶזְרוֹ לֹא יָחִישׁ, –

305 יַעֲלֶה מְרוֹם-הָרָקִיעַ, מִשָּׁם עָב-עֲנָנָיו יְשַׁלַּח,

יַרְעִים בַּגַּלְגַּל אוֹ שׁוֹאַת חֲזִיזִים מִתְרוֹצְצִים יְחוֹלֵל.

אָכֵן לֹא הָיוּ הַפַּעַם מִפְלְשֵׂי-עָבִים בְּיָדוֹ,

יָבְשׁוּ גַם נִבְלֵי-שָׁמַיִם מֵהַמְטִיר לָאָרֶץ גִּשְׁמֵיהֶם,–

הַרְעֵם אָז יַרְעִים הָאֵל, בִּימִינוֹ הַנְּטוּיָה יְשַׁלַּח

310 חִצָּיו בַּנַּעַר, יְמוֹתְתוֹ, מֵרִכְבּוֹ יַפִּילוֹ, וְיָצְאָה

אִשּׁוֹ הַזּוֹעֶמֶת וְזֶרֶם הַלַּהַב הַסּוֹעֵר תַּכְנִיעַ.

חָרְדוּ, הִתְפַּלְּצוּ הַסּוּסִים, יִתְהוֹלְלוּ, יִתְפָּרְצוּ מִתַּחַת

עוֹל צַוְּארֵיהֶם, וּמוֹסְרוֹת רִסְנָם יְנַתְּקוּ וְנָסוּ.

הִנֵּה נְטוּשׁוֹת הַמּוֹשְׁכוֹת, שָׁם נִשְׁבָּר מִמּוֹטוֹ יִתְגּוֹלֵל

315 צִיר-הָאוֹפַנִּים, נָפוֹצוּ חִשּׁוּקֵי הָאוֹפַן הַשְּׁבוּרִים,

נִפְזְרוּ, נִטְּשׁוּ מִסָּבִיב שִׁבְבֵי הָרֶכֶב הָרָצוּץ.

הָשְׁלַךְ פַאֵתוֹן גַּם-הוּא, בִּיעָף יָרַד תְּהוֹמוֹת, אֲדֻמּוֹת

שַׂעֲרוֹת רֹאשׁוֹ מֵאֵשׁ הָאוֹכֵלָה, וְאַחֲרָיו נְתִיבַת-

זֹהַר, וְהָיָה כְמַרְאֵה כוֹכָב, אֲשֶׁר יֵשׁ נִרְאֵהוּ

320 נוֹפֵל מִתְּכֵלֶת הַשְּׁחָקִים, אַף אִם נָפֹל לֹא יִפֹּל הַכּוֹכָב.

הַרְחֵק מֵאֶרֶץ-מוֹלַדְתּוֹ בְּיַרְכְּתֵי-תֵבֵל קִדְּמָהוּ

נְהַר-אֱרִידַנּוּס וּפִיחַ הֶעָשָׁן מִפָּנָיו הִדִּיחַ.

נִימְפוֹת בְּאֶרֶץ הֶסְפֶּרְיָה לִקְבָרוֹת הוֹבִילוּ גּוּפָתוֹ,

מֻכַּת חַצְצֵי אֵל, וְעַל אֶבֶן הַקֶּבֶר חָקָקוּ:

325 "קֶבֶר פֹּה מָצָא פַאֵתוֹן, רוֹכֵב מֶרְכֶּבֶת אָבִיהוּ;

קָצְרָה מִמְּשֹׁל בָּהּ יָדֵהוּ, אַךְ דָּרְכָה עֹז נַפְשׁוֹ – וַיָּמֹת".

זוֹעֵף וְקוֹדֵר הָאָב וַיָּלֶט בְּמַעֲטֵה-עֲרָפֶל

פָּנָיו הַנּוּגִים; אַף-הָיָה יוֹם אֶחָד – מִי יַאֲמִין לַשְּׁמוּעָה? –

יוֹם לְלֹא שֶׁמֶשׁ: רַק מוֹקֵד הָאֵשׁ הַבּוֹעֶרֶת, הוּא נָתַן

330 אוֹרוֹ, וַיְהִי בַשֶּׁבֶר הַנּוֹרָא גַם שֶׁמֶץ תּוֹעֶלֶת.

מִסְפֵּד תְּעוֹרֵר קְלִימֵנֵי, הָאֵם הָעֲזוּבָה וּשְׁכוּלָה,

קִינָה וְאֵבֶל כַּמִּשְׁפָּט, וְקָרְעָה שִׂמְלָתָהּ עָלֶיהָ,

סוּרָה וְגוֹלָה תְשׁוֹטֵט בְּקַצְוֵי הָאָרֶץ וּתְחַפֵּשׂ

גּוּפַת מַחְמַדָּהּ אוֹ לוּ רַק פְּלֵטַת עַצְמוֹתָיו הַפְּזוּרוֹת.

335 מָצֹא גַם מָצְאָה הַקֶּבֶר בְּאֶרֶץ נָכְרִיָּה וּרְחוֹקָה,

צָנְחָה וַתִּפֹּל לְפָנָיו, וּבְקָרְאָהּ שֵׁם בְּנָהּ עַל הַשַּׁיִשׁ,

תִּבְכֶּה תַּמְסֶנּוּ בִדְמָעוֹת, אֶל לִבָּהּ הַחַם תְּאַמְּצֵהוּ.

בָּכֹה תִבְכֶּינָה גַם בְּנוֹת-הַשֶּׁמֶשׁ וּדְמָעוֹת תִּשְׁפֹּכְנָה,

דְּמָעוֹת אֵין הוֹעִיל בָּהֶן, תְּתוֹפֵפְנָה בְלִבָּן, תִּגְהַרְנָה

340 יוֹמָם וָלַיִל עַל קֶבֶר אֲחִיהֶן, תִּקְרֶאנָה לַנַּעַר,

אַךְ לֹא תַקְשֵׁבְנָה עוֹד אָזְנָיו וּמְרִי-קִינוֹתֵיהֶן לֹא יִשְׁמָע.

אַרְבַּע פְּעָמִים כְּבָר חִדְּשָׁה הַלְּבָנָה אֶת זֹהַר קַרְנֶיהָ,

אָכֵן עוֹד הֵנָּה כְדַרְכָּן – לְחֹק לָהֶן הָיָה הַדָּבָר –

קִינוֹת תִּשֶּׂאנָה. וְהִנֵּה – הַגְּדוֹלָה בָהֵן פַאֵטוּזָה

345 תִּבְכֶּה, כִּי אָחַז הַשָּׁבָץ רַגְלֶיהָ בְּכָרְעָהּ לָאָרֶץ!

קָמָה אֲחוֹתָהּ לַמְפֵּטְיָה הַיָּפָה לָגֶשֶׁת אֵלֶיהָ,

אַךְ בְּעוֹד רֶגַע כָּעֵץ בָּאָרֶץ רַגְלֶיהָ שֹׁרָשׁוּ;

תּוֹשִׁיט הַשְּׁלִישִׁית יָדֶיהָ לָגֹז אֶת רֹאשָׁהּ וְהִנֵּה –

עָלִים תִּקְטֹפְנָה, וְעָמְדוּ בְלִי-נוֹעַ כְּגֶזַע עֵץ יָבֵשׁ

350 רַגְלֵי הָאַחַת, וִידֵי אֲחוֹתָהּ – דָּלִיּוֹת נְטוּשׁוֹת.

עוֹד תִּשְׁתּוֹמַמְנָה הַנְּעָרוֹת, וְהִנֵּה כִקְלִיפַּת הָעֵצִים

עָלְתָה עַל מַעֲרֹם הַבֶּטֶן, מִשַּׁד וְעַד כָּתֵף וּזְרוֹעַ

עָטְתָה אֶת בְּשָׂרָן, – עַד דַּל-שִׂפְתֵיהֶן, לָאֵם תְּשַׁוַּעְנָה.

סֹעֲרָה רוּחַ הָאֵם, וְאָנֶה וָאָנָה תִשְׂתָּעֵר,

355 מַהֵר תְּמַהֵר לְהַשִּׁיק שְׂפָתֶיהָ אֶל שִׂפְתֵי בְנוֹתֶיהָ,

לְחַלֵּץ אֶת גֵּו מַחְמַדֶּיהָ מִמֵּיצַר הַגֶּזַע, וּבְיָדָהּ

תִּמְרֹט, תְּשַׁבֵּר אֶת סִבְכֵי הֶעָנָף הָרַכִּים, וְנִטְפֵי

דָם יַאְדִּימוּ פִּצְעֵיהֶם וְנָטְפוּ רְסִיסִים רְסִיסִים.

"הַרְפִּי, הוֹי הַרְפִּי, אִמֵּנוּ – תִּקְרֶאנָה הַנְּעָרוֹת, כִּי גָ דוֹל

360 מַכְאוֹב פִּצְעֵיהֶן. – הֲלֹא אֶת בְּשָׂרֵנוּ הוּא אֲשֶׁר תִּמְרֹט

יָדֵךְ, הֲיִי שָׁלוֹם…" וְסָגַר הָעֵץ גַּם עַל פִּיהֶן.

אָז הַדָּלִיּוֹת הָרַכּוֹת דְּמָעוֹת תִּטֹּפְנָה, וְהָיוּ

אֶגְלֵי הַדְּמָעוֹת לִפְנִינֵי אֱלֶקְטְרוֹן, בְּמֵימֵי הַנָּהָר

רוֹחֲצוֹת הַפְּנִינִים וְהָיוּ חֲרוּזִים עַל צַוַּאר בְּנוֹת-לַטְיוּם.

פירמוס ותסבה

עוֹד אַלְקִתּוֹאֶה בַת מִינִיַּס עוֹמֶדֶת בְּמִרְיָהּ, בַּל תִּשְׁמוֹר

מוֹעֲדֵי בַכְּכוּס, גַּם כַּחֵשׁ תְּכַחֵשׁ, סוֹרֵרָה, כִּי נוֹלַד

בַכְּכוּס עַל בִּרְכֵּי יֻפִּיטֶר. כָּמוֹהָ כֵּן גַּם אַחְיוֹתֶיהָ

נִאֲצוּ אֵל אַף כִּחֵשׁוּ. – צַוֵּה יְצַוֶּה הַכֹּהֵן:

5 חַג וַעֲצָרָה! וְעָזֹב כִּגְּבִירָה כָאָמָה מְלַאכְתָּהּ,

כַּסֵּה הֶחָזֶה בְּעוֹרוֹת, אַף הָסֵר קִשּׁוּרֵי-הַפֵּאָה,

עָנֹד זֵר רַעֲנָן לָרֹאשׁ, וּבַכַּף – שַׁרְבִיט-קִיסוֹס יְרַקְרַק!

אוֹי לַמַּכְעִיסִים אֱלוֹהַּ, כִּי תִתַּךְ חֲמָתוֹ עֲלֵיהֶם!…

כָּכָה יְצַוֶּה הַכֹּהֵן, וְחָרְדוּ עֲלָמוֹת וּגְבָרוֹת,

10 כִּישוֹר וָפֶלֶךְ תַּעֲזֹבְנָה, בְּטֶרֶם מְלַאכְתָּן נִשְׁלָמָה.

קְטֹרֶת לְבַכְּכוּס תַּקְטֵרְנָה. "הוֹי בְּרוֹמְיוּס, לִיאֵיוּס – תִּקְרֶאנָה –

בְּרָקִים חִבְּלוּךָ, יְלוּד-פַּעֲמַיִם וּבֶן-אִמּוֹתָיִם!"

עוֹד תְּזַמֵּרְנָה: "נִסֶוְס, תִּיאוֹנֶוְס יְפֵה-הַתַּלְתַּלִּים,

אַתָּה לִינֶוְס, אֲבִי כָל גֶפֶן תְּשַׂמַּח לֵב אֱנוֹשׁ,

15 אַתָּה אָבִינוּ נִקְטֶלִּי, אֱלֶלִּי וְיַכְּכוּס וְאֵוְהַן, –

רַבִּים, הוֹי לִבֶּר, עוֹד רַבִּים שְׁמוֹתֶיךָ בְּפִי עַמֵּי יָוָן!

רַעֲנָן זִיו-עֲלוּמֶיךָ, לֹא יֲחלוֹף לֹא יִבּוֹל לָנֶצַח,

נַעַר-עוֹלָמִים, מְאֹד יָפְיָפִיתָ וּלְמַרְאֶה בִּמְּרוֹמֵי

שְׁחָקִים, אוֹ תָסִיר קַרְניֶךָ – וְהָיוּ פָנֶיךָ פְּנֵי עַלְמָה.

20 אַפְסֵי הַמִּזְרָח תַּחְתֶּיךָ הִדְבַּרְתָּ עַד יַרְכְּתֵי הֹדּוּ,

אֶרֶץ שְׁחוּמֵי הַפָּנִים, שָׁם מֵימֵי הַגַּנְגֵּס יְפַכּוּ.

אַתָּה, הַנַּעֲרָץ, יִסַּרְתָּ מֵחַלְלֵי שְׁמֶךָ – פֶּנְתֶּוְס

גַּם אֶת לִיקוּרְגוּס, הָאוֹזֵר קַרְדֹּם-פִּיפִיוֹת, וְאַפְּךָ

רָמָה בַיָּם בְּנֵי טִירֶנְיָה. תִּרְכַּב עַל שְׁתֵּי חֻלְדּוֹת-יָעַר,

25 מוֹשְׁכוֹת בְּרֶסֶן-הָרִקְמָה, וּסְבִיבְךָ סָטִירִים וּבַכְּכוֹת,

אִתָּם אִישׁ-שֵׂיבָה יִתְנַהֵג בִּכְבֵדוּת, עַל מַקְלוֹ יִשָּׁעֵן,

אוֹ עַל דַּבֶּשֶׁת אֲתוֹנוֹ יִתְנוֹדֵד הֲלוּם הַשִּׁכָּרוֹן.

בַּאֲשֶׁר תֵּלֵךְ, אֱלוֹהַּ, יָרִיעַ קוֹל רִנַּת בַּחוּרִים,

יַחַד עֲלָמוֹת תִּצְהַלְנָה וְתוֹפְפָה יָדָן בְּתֻפִּים,

30 תִּרְעַם הַחֲצוֹצְרָה, וְחָלִיל אֶרֶךְ-הַגָּרוֹן יְחַלֵּל.

“הָאֵר פָּנֶיךָ, אֵל חַנּוּן!” תִקְרֶאנָה נְשֵׁי הָאִיסְמֵנוּס

וּתְפָאֵרנָה כִמְצֻוֶּה אֶת מִזְבְּחֵי אֵלָן בַּקֹּדֶשׁ.

רַק בְּנוֹת מִינִיַּס לְבַדָּן הֵפֵרוּ הַחַג בַּעֲבוֹדָה;

טֹווֹת אֶת מַטְוֶה הַצֶּמֶר, וְסוֹבֵב הַכִּישׁוֹר בְּיָדָן,

35 שׁוֹקְדוֹת עַל מַעֲשֵׂה הָאֶרֶג וּמְנַצְּחוֹת עַל נַעֲרוֹתֵיהֶן,

נִדְבְּרָה אַחַת מֵהֵנָּה וְשִׁלְּחָהּ הַחוּט מֵאֶצְבָּעָהּ:

"יִבְחֲרוּ מוֹעֲדֵי-בַּדִּים הֲמוֹנֵי הַנָּשִׁים הַפּוֹתוֹת,

נַחְנוּ – מוֹפְתֵנוּ רַק פַּלַּס, הִיא רַבַּת-הַחֶסֶד בָּאֵלִים.

הָבָה בְּדִבְרֵי שְׂפָתָיִם יְגִיעַ כַּפֵּינוּ נָקֵלָּה,

40 סַפֵּר נְסַפֵּר חֲלִיפוֹת הָאַחַת בְּאָזְנֵי אֲחוֹתָהּ,

כָּכָה נְבַלֶּה עִתֵּנוּ וְעָבְרָה עָלֵינוּ בַּנְּעִימִים".

נֵאוֹתוּ לָהּ אַחְיוֹתֶיהָ וְאוֹתָהּ תִּדְרֹשְׁנָה כִּי תָחֵל.

רַבּוֹת מַחֲשָׁבוֹת בְּלִבָּהּ, מַה תִּבְחַר מֵרֹב סִפּוּרֶיהָ:

אִם עַל דֶּרְקֵטִיס בַּת-בָּבֶל, זוּ סַפֵּר יְסַפְּרוּ עָלֶיהָ

45 בְּנֵי פָּלֶשְׂתִּינָה, כִּי פְלָאִים נֶהֱפַךְ צֶלֶם תַּבְנִיתָהּ,

עָטְפָה אֶת בְּשָׂרָהּ קַשְׂקֶשֶׂת, וּבְנִבְכֵי אֲגַמִּים תִּתְלוֹנָן;

אוֹ עַל-דְּבַר בִּתֵּךְ, דֶּרְקֵטִיס, זוּ צָמְחוּ לָהּ נוֹצָה וְכָנָף

וּבָעֲלִיּוֹת מִגְדָּלִים שָׁם שָׁכְנָה עַד בֹּא חֲלִיפָתָהּ;

אוֹ עַל בַּת-יָם, אֲשֶׁר קֶסֶם שִׁירֶיהָ וְלַהֲטֵי-עֵשֶׂב

50 הָפְכוּ בַחוּרִים לְדָגִים נֶאֱלָמִים, עַד הָיְתָה כְמוֹהֶם.

עוֹד לָהּ גַּם סִפּוּר הָעֵץ, אֲשֶׁר הָיוּ לְבָנִים עֲנָבָיו,

הָפְכוּ וַיִּהְיוּ שְׁחוֹרִים, אַךְ נָגַע דַּם אָדָם בָּהֵמָּה.

יָשַׁר הַסִּפּוּר הַזֶּה בְּעֵינֶיהָ, כִּי טֶרֶם יְדָעוּהוּ

רַבִּים. אָז תֵּיטִיב הַחוּטִים וּפָתְחָה בְמָשָׁל שְׂפָתֶיהָ:

55 פִּירָמוּס וְתִסְבֵּה – הוּא עֶלֶם-מַחֲמַדִּים, תִּפְאֶרֶת בְּנֵי-גִילוֹ,

נָאוָה גַם הִיא בַּעֲלָמוֹת, נֶחְמָדָה מִכָּל בְּנוֹת הַקֶּדֶם –

בַּיִת אֶל בַּיִת שָׁם יָשְׁבוּ בְעִיר זוֹ שְׂמִירָמִית הַמַּלְכָּה

אִוְּתָה לְמוֹשָׁב וַתִּבְנֶה לָהּ חוֹמַת-לְבֵנִים מִסָּבִיב.

קִרְבַת בָּתֵּיהֶם קֵרְבָתַם וַתָּכֶן צַעֲדֵיהֶם רִאשׁוֹנָה,

60 אַחַר וַתִּשְׂגֶּה הָאַהֲבָה, אַךְ טֶרֶם בְּנִשּׂוּאִים יָבֹאוּ:

מִצְוַת הוֹרֵיהֶם מְנָעָתַם. אָכֵן מִי יוּכַל וְיִמְנָע

אֵשׁ אַהֲבָתָם, כִּי תֹּאחַז בְּלִבָּם וּבָעֲרוּ יַחְדָּו?

מֵלִיץ לֹא יָבוֹא בֵינֵיהֶם: אוֹתוֹת וּרְמָזִים – לְשׁוֹנָם,

כְּלוּאָה הָאֵשׁ בְּלִבּוֹתָם וּפָרֹץ תִּתְפָּרֵץ כִּפְלָיִם.

65 הָיָה קִיר אֶחָד לְבָתִּים, וּפֶרֶץ קַל נִבְעָה בְתוֹכוֹ,

נִבְעָה מֵאָז וּמִקֶּדֶם, מִיוֹם אֶת הַכֹּתֶל הֵקִימוּ;

יָמִים עַל שָׁנִים יִנְקוֹפוּ, וְאִישׁ לֹא יְגַלֶּה עִקְבוֹתָיו.

אוּלָם הֲיֵשׁ כִּי יֵעָלֵם מֵעֵינֵי הָאַהֲבָה דָּבָר?

אַתֶּם, הַדּוֹדִים, רִאשׁוֹנָה הַבֶּדֶק מְצָאתֶם, וַיִּהְיֶה

70 דֶּרֶךְ לְסוֹד שִׂיחוֹתֵיכֶם וְהֶמְיַת דּוֹדֵיכֶם הָרַכָּה.

פַּעַם בְּפַעַם וְהֵמָּה נִצָּבִים מֵעֶבְרֵי הַכֹּתֶל,

נִשְׁמַת אַפֵּיהֶם בּוֹ תַחֲלֹף עִם לַחַשׁ-תְּלוּנָתָם גַּם יָחַד:

"מַה לְּךָ, קִיר זוֹעֵם, כִּי תַפְרִיד, כְּשָׂטָן כִּי תַעֲמוֹד בֵּין רֵעִים?

תֶּן נָא וְיַחַד נִצָּמֵד, וּבָשָׂר אֶל בָּשָׂר נִדְבָּקָה,

75 אוֹ לוּ לַמִּצְעָר תִּתְּנֵנוּ וְשָׂפָה אֶל שָׂפָה נִשָּׁקָה!

אָכֵן גַּם בָּרוּךְ הֱיֵה נָא, כִּי לָנוּ נְתִיבָה נָתַתָּ

לַעֲבוֹר שִׂיחַ דּוֹדֵנוּ מִפֶּה אֱלֵי אֹזֶן קַשּׁוּבָה".

כָּכָה יָשִׂיחוּ בֵינֵיהֶם, וְהָיָה כִּי יָבוֹא הַלָּיִל,

אִישׁ אִישׁ מֵעֶבְרוֹ יַבִּיעוּ בִרְכָתָם וְאִישׁ מִקָּצֵהוּ

80 יִשְׁלַח עֲתֶרֶת נְשִׁיקוֹת, זוּ סָגַר עֲלֵיהֶן הֶחָיִץ…

עָלֹה אַךְ יַעַל הַשַּׁחַר וְנָסוּ כוֹכָבִים מִפָּנָיו,

שֶׁמֶשׁ הַבֹּקֶר אַךְ תֵּצֵא וּרְסִיסֵי-הָעֵשֶׂב יִיבָשׁוּ,

שָׁם אֶל מְקוֹמָם יָשׁוּבוּ הָרֵעִים: בִּתְלוּנָה חֲרִישִׁית

יִשְׁפְּכוּ שִׂיחָם בֵּינֵיהֶם, וְכָכָה יִוָּעֲצוּ יַחְדָּו:

85 דּוּמָם בְּאִישׁוֹן הַלַּיְלָה, בִּנְפֹל עַל הַשּׁוֹמְרִים תַּרְדֵּמָה,

יֵצְאוּ אֶת פֶּתַח הַבַּיִת וְעָזְבוּ גַם עִיר וַהֲמוֹנָהּ;

תָּעֹה אִם יִתְעוּ בַשָּׂדֶה, וְנוֹעֲדוּ אֵצֶל הַקָּבֶר,

קֶבֶר לְנִינוּס הַמֶּלֶךְ, וְיַחַד בְּצֵל יֵחָבֵאוּ –

תַּחַת הַבָּכָא הָרַעֲנָן, זֶה צַחוּ עֲנָבָיו מִשֶּׁלֶג

90 וּמֵעַל מַעְיָן הַמַּיִם שָׁם יִשָּׂא אֶת צַמַּרְתּוֹ הָרָמָה.

כָּכָה בֵינֵיהֶם נִדְבָּרוּ, בִּכְבֵדוּת יִתְנַהֵג הַשָּׁמֶשׁ,

יֵרֵד לְאִטּוֹ הַיָּמָּה וְעָלָה הַלַּיְלָה מִתּוֹכוֹ.

חֹשֶׁךְ מִסָּבִיב, וְחֶרֶשׁ תִּסּוֹב הַדֶּלֶת עַל צִירָהּ;

דּוּמָם תִּתְחַמֵּק לָהּ תִּסְבֵּה, תְכַסֶּה פָּנֶיהָ וְיָצְאָה,

95 תֵּלֵךְ אֶל גְּדִישׁ הַקָּבֶר, וְתַחַת הַבָּכָא הַמֵּצֵל

תֵּשֵׁב, אַף פַּחַד לֹא יֵדַע לִבָּהּ הָאוֹהֵב. וְהִנֵּה –

בָּאָה לְבִיאָה מִטַּרְפָּהּ, יֵז קֶצֶף וָדָם מִן הַלֹּעַ, –

לִשְׁבֹּר צִמְאוֹנָהּ בַּמַּעְיָן הַקָּרוֹב מְגַמַּת פָּנֶיהָ.

רָאֲתָה תִסְבֵּה מֵרָחוֹק לְאוֹר הַיָּרֵחַ כִּי יָהֵל,

100 רָעֲדָה כֻלָּהּ, וַתְּמַהֵר הִמָּלֵט אֶל סֵתֶר מְעָרָה, –

תָּרוּץ, וּבְחָפְזָהּ וְצָנַח מְעִילָהּ מִשִּׁכְמָהּ לָאָרֶץ,

שָׁתְתָה גַם שָׁבְרָה צִמְאוֹנָהּ וְשָׁבָה הַלְּבִיאָה הַיַּעְרָה,

תֵּלֵךְ וְדָרְסוּ רַגְלֶיהָ אֶת שִׂמְלַת הַנַּעֲרָה הַקַּלָּה,

אַף תְּשַׁסְּעֶנָּה לִקְרָעִים בְּלֹעָהּ הַמְגֹאָל בְּדָמִים.

105 הוֹלֵךְ גַּם פִּירָמוּס וּבָא אֶל הַשָּׂדֶה, וּפִתְאֹם הוּא רוֹאֶה

עִקְבוֹת פְּרִיץ-חַיּוֹת בְּעָפָר. חָוְרוּ פְנֵי פִירָמוּס, וְהִנֵּה –

שִׂמְלָה בְדָם מְגוֹלָלָה… וּמַר יְשַׁוַּע הֶעָלֶם:

"שַׁכֵּל תְּשַׁכֵּל, הַלַּיְלָה, שְׁתַּיִם נְפָשׁוֹת אֲהוּבוֹת!

רַכָּה עוֹד הָיְתָה הָאַחַת וְתֹּאבַד בְּלִי עֵת, וְאָנֹכִי –

110 גָּדוֹל עֲוֹנִי מִנְּשֹא, כִּי אָנֹכִי הֲבֵאתִיךְ, אֻמְלָלָה

הֵנָּה בֵּין מָגוֹר וָפַחַד בְּאִשׁוֹן אֲפֵלָה וָלָיִל,

אַף לֹא מִהַרְתִּי לִקְרָאתֵךְ… הוֹי, בֹּאוּ כָל פְּרִיצֵי אֲרָיוֹת,

זַנְּקוּ מִסֵּתֶר הַסְּלָעִים וְשַׁלְּחוּ בִי שִׁנֵּי הַטָּרֶף,

שַׁסְּעוּ אֶת בְּשָׂרִי לִקְרָעִים, הוֹי, שַׁסְּעוּ חַיָּתִי הַחֹטֵאת,

115 אָכֵן, רַק רַךְ-הַלֵּבָב לַמָּוֶת יִתְפַּלֵּל!"… אָז יָרִים

שִׂמְלַת הַנַּעֲרָה וּנְשָׂאָהּ אֶל תַּחַת הַבָּכָא הַמֵּצֵל.

נַשֵּׁק יְנַשֵּׁק הַשִּׂמְלָה, נְשִׁיקוֹת וְדִמְעָה יַרְוֶנָּה,

“עַתָּה, הַשִּׂמְלָה, גַּם דָּמִי, דַּם-לִבִּי הַחַם אֲרַיָּוֶךְ!”

קָרָא הָעֶלֶם, וְחַרְבּוֹ הוּא יִתְקָע אֶל בִּטְנוֹ, וְטֶרֶם

120 תֵּצֵא נִשְׁמָתוֹ, וּמַהֵר מִפִּצְעוֹ הַחַם יוֹצִיאֶנָּה.

נָפֹל אָז יִפֹּל לָאָרֶץ, וּפָרַץ וְיֵז מִן הַפֶּצַע

דָּמוֹ רוֹתֵחַ לְמַעְלָה, כִּפְרוֹץ שֶׁטֶף מַיִם מִצִּנּוֹר,

צִנּוֹר-עוֹפֶרֶת מִתְפּוֹצֵץ, וּבְשָׁאוֹן יְסֹעַר מִפִּיהוּ

זֶרֶם מִתְגָּעֵשׁ וְנִגְרָשׁ, מִתְפָּרֵץ וְשׁוֹאֵף מְרוֹמִים.

125 הוּעַמּוּ עִנְבֵי הַבָּכָא מֵרְסִיסֵי הַדָּם וַיִּשְׁחָרוּ,

יָנַק הַשֹּׁרֶשׁ קֻבַּעְתּוֹ, וְאֹדֶם-אַרְגָּמָן חַכְלִילִי

כִּסָּה וְהָפַךְ לְתוֹלָע אֶת לֹבֶן הַתּוּתִים הַצַּחִים.

פַּחַד עוֹד מָלֵא לֵב תִּסְבֵּה, אַךְ לָשׁוּב תְּמַהֵר אֶל דּוֹדָהּ,

שָׁם לָהּ יְחַכֶּה, וּבְכִלְיוֹן עֵינַיִם וָנֶפֶשׁ תְּבַקֵּשׁ

130 אוֹתוֹ וּלְסַפֵּר הִיא תִּשְׁאַף, אֵיךְ נִמְלְטָה פְלָאִים מִמָּוֶת.

הִנֵּה הַמָּקוֹם הַנּוֹעָד וְהִנֵּה הָעֵץ, אַף מַה מוּזָר

צֶבַע עֲנָבָיו! הֲזֶה הוּא הַבָּכָא, אוֹ תִשְׁגֶּה בָרֹאֶה?

עוֹדָהּ פִּתְרוֹנִים תְּבַקֵּשׁ וְהִנֵּה הִיא רוֹאָה – בֶּעָפָר

גְּוִיָּה בְדָם מִתְבּוֹסֶסֶת… וַתִּתַּר הַנַּעֲרָה מִמְּקוֹמָהּ,

135 חָוְרוּ מִדּוֹנַג פָּנֶיהָ וְרָעֲדוּ כָל יְצוּרֶיהָ,

רָעֹד כַּיָּם, עֵת יְרַחֵף הָרוּחַ וְרָגַע אֶת מֵימָיו.

אַךְ עַד אַרְגִּיעָה – וְהַכֵּר הִיא תַכִּיר אֶת גּוּפַת אֲהוּבָהּ,

אָז – וִילָלָה תִּתְפָּרֵץ מִלִּבָּהּ, תְּפָרֵשׁ כַּפֶּיהָ,

נִפְזְרוּ מַחְלְפוֹת רֹאשָׁהּ, אֶת גְּוִיַּת מַחְמַדָּהּ תְּחַבֵּק,

140 תַּרְוֶה אֶת פִּצְעוֹ דִּמְעָתָהּ, שָׁם דְּמָעוֹת וָדָם יִתְבּוֹלָלוּ,

וּנְשִׁיקוֹתֶיהָ הַחַמּוֹת עַל פְּנֵי הַחֵן תִּתְרַפֵּקְנָה.

"פִּירָמוּס – הִיא תֵילִיל – הֶאָסוֹן בָּאֲךָ כַּשֹּׁד וַיְשֻׁדֵּנִי?

פִּירָמוּס, עֲנֵנִי, מַחֲמַדִּי! הֵן תִּסְבֵּה, אֲנִי, יְחִידָתְךָ,

קוֹרְאָה אֵלֶיךָ! הוֹי, פְּקַח נָא וְאֶרְאֶה בְּנֹעַם עֵינֶיךָ!"

145 רֶגַע לְזֵכֶר שֵׁם תִּסְבֵּה עֵינָיו הַסְּגוּרוֹת נִפְקָחוּ,

דּוּמָם בָּהּ יָעִיף מַבָּטָן וָשָׁב וַעֲצָמָן לָנֶצַח…

תּוֹעָה עֵין תִּסְבֵּה וְרוֹאָה: קִרְעֵי שִׂמְלָתָהּ וְחֶרֶב

שְׁלוּפָה מִתַּעַר-הַשֵּׁן… אָז תִּקְרָא הַנַּעֲרָה: "יָדַעְתִּי,

זוֹ אַהֲבָתְךָ שִׁחֶתְךָ וְיָדְךָ הִיא קִפְּדָה חַיֶּיךָ!…

150 אָכֵן גַּם יָדִי בְעֶזְרִי וְתוֹסִיף הָאַהֲבָה חָיִל –

אֵרֵד אֶל שַׁחַת גַּם אָנִי, כִּי שְׁאוֹלָה אֲנִי הִדְרַכְתִּיךָ,

עַתָּה בִשְׁאוֹל אֲלַוְּךָ וְאֶשְׁמוֹר צְעָדֶיךָ בַקָּבֶר,

יַחַד הָלַכְנוּ בַחַיִּים, אַל יַפְרִיד גַּם מָוֶת בֵּינֵינוּ!

אַחַת רַק אֶשְׁאַל מֵאִתְּכֶם, הוֹי אֲבוֹתֵנוּ שַׁכּוּלִים,

155 חֶסֶד כִּי תַעֲשׂוּ אִתָּנוּ, וְתַחַת הַגָּדִישׁ הָאֶחָד

קֶבֶר תָּכִינוּ לַשְּׁנָיִם, אֲשֶׁר לַאֲחָדִים קְשָׁרָתַם

אֵשׁ אַהֲבָתָם הָעַזָּה וְרִגְעָם הָאַחֲרוֹן בַּחַיִּים!…

אַף, עֵץ הַבָּכָא, גַּם אַתָּה, אֲשֶׁר בִּצְלָלֶיךָ תִּתְלוֹנָן

נֶפֶשׁ מַחֲמַדִּי, וּכְרֶגַע גַּם שְׁנֵינוּ בְּצִלְּךָ נֶחְסָיָה,

160 שְׁמָר נָא לָנֶצַח אֶת עִקְּבוֹת מוֹתֵנוּ, וְיַעֲטֹף אֵבֶל

שָׁחוֹר אֶת פְּרִי עֲנָבֶיךָ, וְהָיוּ מַזְכֶּרֶת בְּרִית-דָּמִים!…"

עוֹד אֲמָרֶיהָ בְּפִיהָ – וַתַּכֶּה אֶל לִבָּהּ בֶּחָרֶב,

חֶרֶב לֹא קָפְאוּ עָלֶיהָ עוֹד נִטְפֵי הַדָּם הָרוֹתֵחַ.

שָׁמְעוּ אֶל לַחַשׁ תְּפִלָּתָהּ הָאֵלִים, גַּם שָׁמְעוּ הָאָבוֹת:

165 שָׁחוֹר וְעוֹטֶה צֵל-אֵבֶל הַבָּכָא עֵת יַבְשִׁיל עֲנָבָיו,

אֶחָד אֲרוֹן הַנֶּאֱהָבִים, שָׁם אֵפֶר עַצְמוֹתָם יָנוּחַ".

כִּלְתָה הַנְּסִיכָה סִפּוּרָהּ, אַךְ עוֹד תַּעֲבֹדְנָה בְנוֹת מִינִיַּס,

בָּזוּ לְאֵל בְּלִבּוֹתָן וְקֹדֶשׁ הַחַג תְּחַלֵּלְנָה.

פִּתְאֹם וְהֶמְיָה חֲרִישִׁית תֻּפִּים נַעֲלָמִים יַשְׁמִיעוּ,

170 נַגֵּן יַנְעִימוּ חֲלִילִים וַחֲצוֹצְרוֹת בִּתְרוּעָה תִצְהַלְנָה.

רֵיחַ מוֹר עוֹבֵר מִסָּבִיב וּבָשְׂמֵי הַכַּרְכֹּם יִזָּלוּ,

אַף – וְאִם נִפְלָא הַדָּבָר – הָפַךְ הַצֶּמֶר יְרַקְרַק,

פָּרְחוּ מְעִילֵי הַנְּסִיכוֹת וַיַּעֲלוּ צִיצֵי הַקִּיסוֹס,

הָיָה הָאֶרֶג שָׂרִיגִים, זַלְזַלִּים מִחוּטָיו יִצְמָחוּ,

175 שְׂרוּקֵי הַגֶּפֶן יִשְׂתָּרְגוּ בִשְׁתִי הַמַּסֶּכֶת, וְהִבְרִיק

זֹהַר-אַרְגָּמָן חַכְלִילִי בְּאַשְׁכְּלוֹת עִנְּבֵי-הַחֶמֶד.

חָלַף כְּבַר עָבַר הַיּוֹם וַיִּהְיֶה אוֹר כֵּהֶה בַּבָּיִת,

אֵין זֹאת אֲפֵלַת הַנֶּשֶׁף, אַף נֹגַהּ הַיּוֹם עוֹד לֹא יָאִיר,

חֹשֶׁךְ וָאוֹר יְסֻבָּכוּ, וְלַיִל יְנַכֵּר אוֹתוֹתָיו.

180 רֶגַע וְאַמּוֹת הַסִּפִּים יָנוּעוּ, לַפִּידִים מִסָּבִיב

עוֹלִים וְלַהַב אֲדַמְדַּם יַגִּיהַּ הַבָּיִת, וּפְרִיצֵי

חַיּוֹת, צַלְמֵיהֶם בַּלָּהוֹת, בְּיִלְלַת אֵימָה יֵילִילוּ.

יַחַד הַנְּסִיכוֹת נָבוֹכוּ וְתָעוּ בְגֵאוּת הֶעָשָׁן,

מִפְלָט לָהֶן תְּבַקֵּשְׁנָה מִנֹּגַהּ הָאֵשׁ הַנּוֹרָאָה.

185 עוֹד הֵן תִּתְעֶינָה וְהִנֵּה – מֵאֶצְבַּע לָאֶצְבַּע הַקְּטַנָּה

נִקְרַם עוֹר דַּק, וַתְּכַסֶּה נוֹצָה אַצִּילֵי יְדֵיהֶן, –

שֻׁנָּה מַרְאֵיהֶן מְאֹד וְלֹא יָדְעוּ, כִּי כָבְדָה עֲלֵיהֶן

חֶשְׁכַּת הַבָּיִת. גַּם הָרֵם לֹא תָּרִים הַנּוֹצָה אוֹתָנָה,

רַק תְּרַחֵפְנָה כִמְעַט בַּכְּנָפַיִם הַשְּׁקוּפוֹת וְהָיָה

190 אִם תְּדַבֵּרְנָה, וְנֶחְבָּא קוֹל דַּבְּרוֹתֵיהֶן: בִּשְׁרִיקָה

כֵּן תֶּהֱמֶינָה, וְיָדְעוּ אַךְ צַפְצֵף וְהָגֹה נְכָאִים.

בַּיִת, לֹא יַעַר מוֹשָׁבָן, וּלְזָרָא אוֹר יוֹם בְּעֵינֵיהֶן,

לַיְלָה תִּבְחַרְנָה לִמְעוּפָן, וְנֶשֶׁף שְׁמוֹ יִקְרָא עֲלֵיהֶן.
