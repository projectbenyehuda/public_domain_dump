חֶבְלֵי־מָשִׁיחַ

א

"נִשְׂגַּבְתָּ סוֹד הַגְּאוּלָה! בְּעֹז וָגִיל

אֶחְתּוֹר עֲדֵי אָבוֹאָה סֵתֶר זְבוּלְךָ.

בְּסַעֲרוֹת־תְּפִלּוֹתַי אָסֹל שְׁבִיל,

בְּאֵשׁ לִבִּי אַבְעִיר, אַפִּיל קִיר גְּבוּלְךָ.

שַׁדַּי! תּוֹלַעַת אֲדָמָה רְפַת־אוֹנִים

אֲנִי…רְאֵה נַפְשִׁי! בָּהּ זִיק שְׁאוֹל יִלְהַט…

הַלֵּב נִשְׁבַּר וַיֵּהָפֵךְ תֵּל יְגוֹנִים…

כָּלָה הַקֵּץ. שַׁדַּי הוֹפַע מִיַּד, מִיַּד!"

כֹּה הִתְפַּלֵּל, הָגָה רַבִּי מֵאִיר הַשָּׂב

דּוּמָם סוֹדוֹ נָשָׂא, רַךְ־עַיִן, כְּפוּף־גַּב.

עוֹד שָׁם בְּמֶרְחַקֵּי־גוֹלָה,

צָץ בְּלִבֵּהוּ הַחֲלוֹם.

זִרְמֵי־גַעְגּוּעִים עַזִּים

עָטוּהוּ, הֲמָמוּהוּ דֹּם.

צַר, צַר הָיָה לוֹ בַּנֵּכָר…

הָמָה לִבּוֹ עֲיֵף רָזִים…

וַיַּעַל אֶרֶץ אֲבוֹתָיו

עִם הָרוּחוֹת הַנֶּחְפָּזִים.

שְׁמֵי־תְכֵלֶת עֲמֻקִּים,

יְפִי־גִבְעָה טָהוֹר וָרַךְ — —

הִרְגִּישׁ לִבּוֹ זִיו קְדוּשָׁה

אֶל תּוֹךְ נַפְשׁוֹ יְפַךְ, יְפַךְ…

יוֹם יוֹם יֵתַע הוֹמֶה, סוֹעֵר

בֵּין הֶהָרִים, בַּשְּׁבִילִים.

כְּלַהֲקוֹת־יוֹנִים הוֹמִים

קִרְבּוֹ שִׁירֵי הַתְּהִלִּים,

וּמַלְאָכִים זַכִּים רָנִים

נִגְלוּ לוֹ עַל כָּל גַּג וָתֵל.

דּוּמָם, בַּחֲרָדָה יִלְחַשׁ:

“אֶת כְּבוֹדְךָ רָאִיתִי, אֵל”.

בָּעִיר הָעַתִּיקָה, בְּחֵיק הַסִּמְטָאוֹת,

בֵּית הַמִּדְרָשׁ עוֹמֵד, קַן סוֹד וּפְלָאוֹת,

שָׁם “שַׁעַר הַשָּׁמַיִם”, קַן־הַקַּבָּלָה,

בְּשֶׁקֶט חֲלָלוֹ זִיו אֱלֹהִים עָלָה…

בַּלֵּיל בַּלֵּיל יָבוֹאוּ זְקֵנִים שַׁחִים,

רוֹעֵד אוֹר פַּנָּסָם בְּסוֹד רוּחוֹת לַחִים.

בֵּית הַמִּדְרָשׁ נֵעוֹר אָז. הַנֵּרוֹת חַיִּים.

הוֹמִים הַלְּבָבוֹת, הוֹמִים מַאֲוַיִּים.

עוֹלְמוֹת־חֵן אֵיתָנִים וַעֲדִינִים שָׁטִים,

גַּלֵּי הַסְּפִירוֹת בָּאִים וְנִמְלָטִים…

בְּחֶדֶר צַר, בּוֹדֵד רַבִּי מֵאִיר יֵשֵׁב

שָׁם — וְנַפְשׁוֹ רָוָה זִיו־חֶסֶד, זִיו־כְּאֵב…

בְּשׁוּט עָבֵי־הַטַּל רַכֵּי צִנָּה בָּרוֹם —

אָז תַּחֲרֹג נַפְשׁוֹ כְּרַעַם בַּתְּהוֹם…

"קָרוֹב אֲנִי אֵלֶיךָ, אֵל! שְׁמַע הַצָּו:

לֹא אַרְפְּךָ, הוֹרֵד אוֹר יִשְׁעֲךָ הָרָב!

בָּנֶיךָ נִשְׁחָטִים, אוֹר תּוֹרָתְךָ יִדְעָךְ,

שֵׁם כְּבוֹדְךָ גָּלָה מִלֵּב, נִסָּח"…

בַּחֶדֶר הֵד שׁוֹקֵט צוֹלֵל. הַס בַּפִּנָּה.

נִשְׁמַת־הַלֵּיל רוֹעֶדֶת וּמַאֲזִינָה…

חֲדַל־אֱיָל, רַבִּי מֵאִיר יִצְנַח,

עִם סַהַר־גִּיל צְחוֹק קַל עַל פָּנֵיהוּ נָח…

רַק הוּא, רַבִּי מֵאִיר, רַק הוּא יֵדַע הַסּוֹד:

הַקֵּץ קָרוֹב. קֵץ לְשִׁפְלוּת לְבוּז וְנוֹד.

תִּגְבַּרְנָה הַצָּרוֹת עַל הָאֻמָּה כֻלָּה —

הֵן, אֵלֶּה הַצָּרוֹת, תָּחֵשְׁנָה גְּאוּלָה.

בְּחִיל וְגִיל יַקְשִׁיב לְשֶׁטֶף הַדָּמִים…

יֵדַע, יֵדַע רַבִּי מֵאִיר: לְקֵץ יָמִים

סוּפַת־עֹז תַּחֲלֹף עַל יִשְׂרָאֵל הַנָּד.

הַשַּׁחַק אָז יִמֹּט, הַהֲוָיָה תִּרְעַד,

יִשְׁטֹף הַתֹּהוּ אֶת הַיְּקוּם הָעֵר, הֶחָי…

קָרוֹב הַיּוֹם! בָּא יוֹם הַדִּין. יוֹם אֲדֹנָי…

אָז הַיְּצִירָה תּוֹפַע טַל אשֶׁר כְּלוּלָה,

אָז יַעֲנֹד מָשִׁיחַ זֵר הַגְּאוּלָה.


ב

אָבִיב צָעִיר בַּשַּׁחַק אָז רִחֵף.

שִׁירַת צִיצִים בַּגַּיא, עַל צֶלַע־כֵּף

הֵד בְּרָכָה, עֲנֹג אֶבְרָה, נָהַר

עַל עִיר וָנִיר, וְשָׁרוּ תֵל וָהָר.

וְשָׁרוּ לְבָבוֹת שִׁיר עֲלוּמִים.

דִּשְׁאֵי־חוֹמָה הִזִּילוּ נִחוּמִים.

אָבִיב צָעִיר בַּשַׁחַק דֹּם יְפַךְ —

בָּרְחוֹבוֹת דַּם־יְהוּדִים נִשְׁפַּךְ…

זָעִים רוּחוֹת רָעִים, שֻׁדַּד הַלֵּב —

אָבִיב צָעִיר בַּשַּׁחַק אָז רִחֵף…


רָאִיתִי צִלְלֵי־אָבוֹת דּוּמָם צָפִים —

פֹּה נִינֵיהֶם מִתְרוֹצְצִים וְנִגָּפִים.

אַל, אַל תִּכָּלְמוּ, אָבוֹת! לֹא הֶרֶג כְּלָבִים

הָרְגוּ נִינֵיכֶם. כְּסַעַר לְהָבִים

הֵם נִלְחָמִים עַזִּים, מָרִים,

לִבּוֹת אָבִיב, לִבּוֹת־זִיו נִסְעָרִים.

חוֹמוֹת יְרוּשָׁלַיִם רָאוּם הַבִּרְיוֹנִים, הַגְּאֵיוֹנִים!


עִם לֵיל מֵרוֹם הַשַּׁחַק רַד הָעֶלְבּוֹן,

עֶלְבּוֹן־עֲנָק, זוֹעֵם, גֵּא כֶּעָוֹן.

מֵת הָאָבִיב. קִבְרוֹ נִשָּׂא בָרוֹם.

עֻלַּף הַיְּקוּם וַיִּתְכַּנֵּס לוֹ דֹּם…

בָּאֹפֶל כֶּתֶם־דָּם יַבְרִיק, עוֹלָם קָטָן —

בּוֹ נִשְׁקָפִים הָעֶלְבּוֹן וְהַשָּׂטָן.

הַבִּרְיוֹנִים שָׁרִים שִׁירֵי־חָרוֹן, שִׁירֵי־קְרָב,

הַיְּשִׁישִׁים תּוֹהִים עַל עוֹלָמָם חָרָב…

אִלֵּם הַלֵּיל. תּוֹעֶה הָעֶלְבּוֹן הַמָּר…

הַשַּׁחַק שְׁכוּל־אוֹר וְצַר לוֹ, צַר…


שָׁט תֹּהוּ בַּחֲלַל־הַמְּבוֹאוֹת

עִמּוֹ נִשְׁמוֹת־הַחֲלָלִים תּוֹעוֹת.

הִי־סֵתֶר יַחֲלֹף כְּלֵב, כְּגַל:

“צִיץ הַתִּקְוָה. צִיץ־הַתִּקְוָה קָמָל”….


בְּמַאֲפֵל־סִמְטָה קוֹל דֶּלֶת נִפְתָּחָה

חוֹלֵף כְּנַאֲקַת־חַיָּה, כַּאֲנָחָה.

קוֹל עַז עוֹלֶה: “בָּא יוֹם הַדִּין, יוֹם־אֲדֹנָי!”

וַיְהִי הַהֵד, כִּיצִיר אֵשׁ, עֵר וָחָי.

אִישׁ צַח, לָבָן, כַּמֵּת בְּתַכְרִיכָיו, עָבַר

בָּאֹפֶל, כְּכוֹכָב בְּמַחֲשַׁכִּים יִנְהָר.

הַשּׁוֹמְרִים נִדְהָמוּ. חֻבְּאוּ בַּצֵּל.

הָעֶלְבּוֹן חָוַר, גָּנַח הַלֵּיל.


דֹּם אֶל הַר־הַזֵּיתִים דַּרְכּוֹ אָחַז,

אַט אַט צָעַד, חֶזְיוֹן גֵּאוּת וָרָז.

שָׁם מֵחֵיקוֹ הוֹצִיא שׁוֹפָר, תָּקַע

בְּכָל אוֹנוֹ — הַדּוּמִיָּה בָּקַע…

פָּרַשׂ כַּפָּיו, קָרָא בְּקוֹל עַז, מַר:

"קוּמוּ, שׁוֹכְנֵי עָפָר!

בָּא יוֹם־הַדִּין, יוֹם אֲדֹנָי

וְשָׁב הַמֵּת וָחָי.

הִתְנָעֲרוּ, תִּלִּים!

בַּמִּשְׁעוֹלִים, בַּשְּׁבִילִים

צְאוּ, צְאוּ בַּמְּחוֹלוֹת.

הַמַּקְהֵלוֹת עוֹלוֹת,

הַמַּקְהֵלוֹת הַלְּבָנוֹת,

כַּעֲנָנוֹת.

חֲיוּ, מֵתִים זַכִּים!

מִמֶּרְחַקִּים

הָבִיאוּ הַמָּשִׁיחַ הַגּוֹאֵל…

אֲנִי, מֵאִיר בֶּן… אֶקְרָא לָכֶם הַלֵּיל.

דּוֹרוֹת דּוֹרוֹת

טָווּ זֵר תְּפִלּוֹת־אוֹרוֹת…

כָּלָה הַקֵּץ, גּוֹסֵס הַיְּקוּם הֶחָי…

בָּא יוֹם־הַדִּין, יוֹם אֲדֹנָי"…


בֵּינוֹת לַקְּבָרִים רַבִּי מֵאִיר יִצְעַד,

לִבְנַת־מַדָּיו לְסַהַר־פָּז תִּלְהַט…

קוֹל הַשּׁוֹפָר יָשׁוּט לַמֶּרְחַקִּים,

קוֹלוֹ יִקְרָא: "חֲיוּ, מֵתִים זַכִּים!

אַיֶּכָּה, הַמָּשִׁיחַ הַגּוֹאֵל?…

הוֹפַע, הוֹפַע בְּמַאֲפֵל הַלֵּיל!"…

— — — — — — — — —

בִּקְצֵה־הָעִיר, מֻצַּף־עֳפָאֵי־גַן,

בֵּית־תִּפְאָרָה נִשָּׂא, עַלִּיז וָרָן.

זוֹעֶמֶת חוֹמָתוֹ. מִבַּעַד לַשְּׂבָכוֹת

קוֹל נְשָׁמוֹת בָּא, טְרוּפוֹת וְנִדָּחוֹת.

בְּכִי וָשִׁיר הוֹמִים,

שִׁיר־יְשִׁימוֹן, שִׁיר־יְתוֹמִים.

“בֵּית־הַמְשֻׁגָּעִים” שָׁם. יוֹם וָלֵיל

רַב הַשָּׁאוֹן בּוֹ, יְלַל זָר, הוֹלֵל.

מֵאַחַד הַחַלּוֹנוֹת פּוֹרֵץ קוֹל מַר:

"קוּמוּ, שׁוֹכְנֵי עָפָר!

בָּא יוֹם הַדִּין, יוֹם אֲדֹנָי"…

יָטוּס הַקּוֹל לַמֶּרְחַקִּים, לְהַר וָגַיְא,

תּוֹעֶה הֵדוֹ בִּיגוֹן־הַסְּלָעִים.

וּבְהַקְשִׁיב אִישׁ לַקּוֹלוֹת הַנִּשָּׂאִים —

וְהִשְׁתּוֹמֵם, נִדְהָם יִצְעַד לְאַט

וּבְעֵינוֹ דִּמְעָה תִרְעָד…


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
