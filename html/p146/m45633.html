אַיֶּלֶת אַהֲבִים רַצִּי

החתימה: אלשבזי


אַיֶּלֶת אַהֲבִים / רַצִּי עַל הֲמוֹנָי

כִּי לְקָחוּם שׁוֹבִים / חֲבוּרַת מַשְׂטִינָי

וַיִּכְתְּבוּ כְתָבִים / לְהָעִיץ עַל צְפוּנָי

עוֹרְרִי שׁוֹבָבִים / נְעָרַי וּזְקֵנָי

נַחְפְּשָׂה דְּרָכֵינוּ / וְנָשׁוּבָה עַד יְוֹיָ:


לָמָּה נִתְעַלֵּם / בְּבוֹר גָּלוּת הוֹמִים

וְאֵיךְ נִסְבּוֹל עֻלֵּם / מְרוּדִים וַאֲשֵׁמִים

נַעֲלֶה לְעִיר שָׁלֵם / מִגְּבוּלוֹת עַמִּים

וְנָשׁוּב וּנְשַׁלֵּם / פָּר וְאַיִל תָּמִים

לְחֵי הָעוֹלָמִים / הֵ[…] יְוֹיָ

נַחְפְּשָׂה דְּרָכֵינוּ / וְנָשׁוּבָה עַד יְוֹיָ:


שְׁבוּת עַמְּךָ תָּשִׁיב / אֱלֹהֵי יִשְׁעֵנוּ

בְּרַחֲמִים תַּקְשִׁיב / לְקוֹל תְּפִלָּתֵנוּ

בַּעֲבוּר בַּת נָדִיב / עֲטֶרֶת רֹאשֵׁנוּ

לְעִיר צִיּוֹן תֵּיטִיב / שָׁם מְחוֹז חֶפְצֵנוּ

תֵּן מַאֲוַיֵּינוּ / הָהָ[…] חָמָה יְוֹיָ

נַחְפְּשָׂה דְּרָכֵינוּ / וְנָשׁוּבָה עַד יְוֹיָ:


בְּחַסְדְּךָ וְרַחֲמֶיךָ / תַּעְזְרֵנוּ מְהֵרָה

תִּכְבּוֹשׁ כַּעֲסֶךָ / וְתָסִיר קְטִיגוֹרָה

תְּנַהֵג בָּנֶיךָ / בְּתוֹךְ קֶרֶן אוֹרָה

בַּעֲבוּר כְּבוֹדֶיךָ / וּבְגִין אָבוֹת וְתוֹרָה

עַמְּךָ תְּעוֹרְרָה / הַבּוֹטְחִים בַּיְוֹיָ

נַחְפְּשָׂה דְּרָכֵינוּ / וְנָשׁוּבָה עַד יְוֹיָ:


זְכוֹר יוֹם גְּאַלְתָּנוּ / בְּיַד רוֹעֶה נֶאֱמָן

עֲבוּרוֹ פַלְּטֵנוּ / בְּעֵת רָצוֹן נְזֻמָּן

בְּכָבוֹד פַּרְנְסֵנוּ / בִּזְכוּת אוֹכְלֵי מָן

תְּקַבֵּץ פְּזוּרֵנוּ / אֲשֶׁר נִפְרָד בְּתֵימָן

וְזֶה יִהְיֶה סִימָן / טוֹב לַחֲסוֹת בַּיְוֹיָ

נַחְפְּשָׂה דְּרָכֵינוּ / וְנָשׁוּבָה עַד יְוֹיָ:


יְשׁוּעָה לִבְנֵי תָם / יוֹם זֶה וְהַרְוֵחַ

תִּתְבַּשֵּׂר עֲדָתָם / בְּרָצוֹן וְהַצְלֵחַ

יִרְצֶה תְּשׁוּבָתָם / מוֹחֵל וְסוֹלֵחַ

וְתַעֲלֶה תְפִלָּתָם / כְּמוֹ קָרְבְּנוֹת רֵיחַ

מַעֲשֵׂה רֹקֵחַ / קוֹדֶשׁ לַיְוֹיָ

נַחְפְּשָׂה דְּרָכֵינוּ / וְנָשׁוּבָה עַד יְוֹיָ:


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
