אִם דְּרָכַי

חתימה: אלשבזי


אִם דְּרָכַי / בְּמַהְלָכַי / בְּיוֹם שׁוּבִי חָקַרְתִּי

לֹא אוּכַל / בְּעֵין שֵׂכֶל / לְהַעֲלוֹתָן עֲלֵי דַּעְתִּי

כִּי אָצוּ / לְחֵטְא רָצוּ / פְּעָמַי בְּיוֹם מָרַדְתִּי

וְלֵב הוֹמֶה / כְּמוֹ סוֹמֶא / בַּעֲצָתוֹ הָלַכְתִּי

וְעֵינַיִם / כְּמוֹ מַיִם / בְּמַרְאֵיהֶן נִמְשַׁכְתִּי

אֵלֶיךָ יְוֹיָ / נַפְשִׁי אֶשָּׂא / אֱלֹהַי בְּךָ בָּטַחְתִּי:


לֵב טָמֵא / בַּאֲשָׁמוֹ / בִּימֵי נְעוּרַי מָרַד

חֵטְא חָמַד / וְלֹא לָמַד / וּבְיוֹם הַדִּין חָרַד

חֲבָל עַל דָּל / הֵיךְ יֶחְדָּל / וְיוֹם אֵידוֹ יִתְפָּרַד

וּבִכְלִמָּה / וְרוֹב אַשְׁמָה / יָשׁוּב לַעֲפָרוֹ יֵרַד

וְהַנֶּפֶשׁ / בְּיוֹם נוֹפֶשׁ / תִּדְאַג אִם נִזְכַּרְתִּי:


שְׁמַע קוֹלִי / צוּר אֵלִי / רְצֵה שִׂיחִי וְקוֹל נִיבִי

כִּי שַׁחְתִּי / בְּעֵת לֶכְתִּי / וְחָלַשְׁתִּי בְּרוֹב חוֹבִי

וְאֵיךְ יִצְדַּק / עָפָר דַּק / וְצַד שְׂמֹאל מָשַׁל בִּי

כְּפוּף קוֹמָה / בְּרוֹב אֵימָה / וְאֵיךְ אַקְבִּיל פְּנֵי רִבִּי

וְלִבִּי סָר / וְנִתְיַסָּר / וּבַהֲבָלָיו נוֹקַשְׁתִּי:


בְּכִי עֵינִי / לְבֶן אוֹנִי / אֲשֶׁר חָטָא בְּמַעְלָלָיו

וְהִתְבּוֹנָן / וְהִתְאוֹנָן / עֲלֵי רוֹעַ מִפְעָלָיו

חֲבַצֶּלֶת / כְּאַיֶּלֶת / בְּלֶכְתֵּךְ בַּקְּשִׁי עָלָיו

בְּיוֹם יַחְזוֹר / לְהֵעָזֵר / לְרַצּוֹת דּוֹד בְּמַהְלָלָיו

וְטַל אוֹרוֹת / בְּאַשְׁמוּרוֹת / תְּנִי עָלַי בְּיוֹם לֶכְתִּי:


זָרַח שֶׁמֶשׁ / לְדָל טָמַשׁ / בְּמֵי עָכוֹר וְנִצְטַנֵּן

וְהוּא צוֹפֶה / לְדוֹד יָפֶה / אֲשֶׁר דַּעְתּוֹ בְּטוֹב חוֹנֵן

וּבֶן פֶּרֶא / לְרֹאשׁ פּוֹרֶה / מְאֹד גּוֹאֶה וּמִתְרוֹנֵן

וְהִרְבָּה מַס / וְדָל חָמַס / וּפִי חַרְבּוֹ לְחֵטְא שׁוֹנֵן

וּמִתְיָהֵר / שְׁהוּא טָהוֹר / וּבַחֲמָסוֹ שָׁפַלְתִּי:


יְקָר חֶפְצִי / בְּבֶן פַּרְצִי / בְּיוֹם יָבוֹא לְהֵיכָלוֹ

וְעֵת יָבוֹא / לְמוֹשָׁבוֹ / וְשָׁב עַמִּי לִגְבוּלוֹ

בְּאַהְבַת עָם / מְאֹד יִנְעָם / וְנִקָּבֵץ בִּגְלָלוֹ

וְאֵשׁ תַּבְעִיר / לְאִישׁ שָׂעִיר / וּבֶן אָמָה וְכָל חֵילוֹ

וּבַת צִיּוֹן / בְּאַפִּרְיוֹן / תְּעִירֵנִי וְשָׂמַחְתִּי:


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
