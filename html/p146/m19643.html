אֵיכָה יְחִידָתִי תֶחֱזִי צוּרֵךְ

החתימה: אלשבזי


אֵיכָה יְחִידָתִי תֶחֱזִי / צוּרֵךְ וְתַגְבִּיהִי דִגְלִי

וַאֲנִי הִגְעַלְתִּיךְ בְּחָפְזִי / וְהִבְדַּלְתִּיךְ מֵעֲצַת שִׂכְלִי

וָאֶפְחֲדָה יוֹם בּוֹא חוֹזִי / וְשָׁקַל פְּעֻלּוֹתַי אֶצְלִי

יְיֹיָ עֻזִּי וּמָעֻזִּי / וּמְפַלְטִי לִי:


לִבִּי חֲזוֹר בָּךְ וְהִתְוַדֶּה / כִּי אֵין תְּשׁוּבָה לָךְ בִּשְׁאוֹל

כִּי טוֹב אִישׁ עוֹזֵב וּמוֹדֶה / כִּי כָּל הַיּוֹם חֶסֶד אֵל

וְנֶאְמַן רוּחַ יִפָּדֶה / יוֹם בּוֹא תְשׁוּעַת הַגּוֹאֵל

עוֹזְבֵי חֲבוּרַת אִישׁ שָׂדֶה / הֵם עַם אֱלֹהֵי יִשְׂרָאֵל

צִוָּה עֲלֵיהֶם לֹא תִרְדֶּה / עַם זוּ יָצַרְתִּי לִי:


שַׁעֲרֵי רַחֲמִים וּתְשׁוּבָה / יַחַד בְּרָצוֹן נִפְתָּחִים

וְסֵפֶר וּמְגִלָּה כְּתוּבָה / בְּיָמִין וּשְׂמֹאל מֻנָּחִים

בְּיָמִין הַטּוֹבָה קְצוּבָה / וּבִשְׂמֹאל רָעוֹת נִמְתָּחִים

וְהַכֹּל בַּמִּשְׁקָל יוּבָא / וְעָוֹן וּזְכוּת מְפֻתָּחִים

אוֹי לִי כִּי הַנּוֹשֶׁה בָא / לוּלֵי יְיֹיָ עֶזְרָתָה לִּי:


בַּעֲוֹן יִצְרִי מָרִיתִי / הִשִּׁיאַנִי וַיְפַתֶּה

בְּהַבְלֵי עוֹלָם שָׁגִיתִי / בְּטוֹבַת מַאֲכָל וּבְמִשְׁתֶּה

וּבְאַחֲרִית יוֹם רָאִיתִי / הַכֹּל בְּיוֹם הַדִּין יֶאֱתֶה

לְהַזְכִּיר כָּל אֲשֶׁר עָשִׂיתִי / וְעָרוּךְ מֵאֶתְמוּל תָּפְתֶּה

בַּייֹיָ חָסִיתִי / וַיְהִי יְיֹיָ לְמִשְׁעָן לִי:


זְכוּת אָבוֹת תִּסְעָדֵנִי / אוּלַי תְּשׁוּבָתִי תְקֻבָּל

חֶסֶד וְרַחֲמִים יַנְחֵנִי / פֶּן בַּהֲבָלַי אֶתְבַּלְבָּל

וּמֹשֶׁה זְכוּתוֹ תַמְשֵׁנִי / מֵעוֹל בְּנֵי עַמּוֹן וּגְבָל

אֶשְׂמַח בַּעֲבוֹדַת קוֹנִי / וְאַל יִפְרְכָה בִי עַם נָבָל

בְּגִין יְפֵה עַיִן וְאַדְמוֹנִי / אַל יַעַלְצוּ אוֹיְבַי לִי:


יָחִיד בְּטוּבְךָ נַחֲמֵנוּ / וְחַזֵּק בִּרְכַּיִם כּוֹשְׁלִים

בְּסֵפֶר חַיִּים כָּתְבֵנוּ / חַסְדְּךָ וְיִשְׁעֶךָ מְיַחֲלִים

בְּחֶבְלֵי אַהֲבָה תִמְשְׁכֵנוּ / בְּחַיּוֹת וְאוֹפָן וְאֶרְאֶלִּים

וְחַשְׁמַל וְגַלְגַּל יְבַשְּׂרֵנוּ / צַהֲלִי קוֹלֵךְ בַּת גַּלִּים

אַתָּה תָּשׁוּב תְּרַחֲמֵנוּ / יְיֹיָ צוּרִי וְגוֹאֲלִי:

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
