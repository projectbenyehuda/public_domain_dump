לִבִּי מְצַפֶּה רַעְיָה וְדוֹד צַח וְיָפֶה

החתימה: לאלשבזי


לִבִּי מְצַפֶּה / רַעְיָה וְדוֹד צַח וְיָפֶה / תַּשְׁפִּיעַ לְחַמָּה וְסַהֲרִי

חוּן דּוֹד וְהַרְפֶּה / עַל דָּל בְּגָלוּת מְעֻלְפָה / אַתָּה כְבוֹדִי וְיוֹקְרִי

אוֹיֵב מְחַפֶּה / עָלַי דְּבָרִים וְנִכְפֶּה / מַחְרִיץ כְּנַאנָא עֲלֵי אֲרִי

עַצְמוֹ יְשֻׁפֶּה / נֵרוֹ תְכַבֶּה וְיִטְפֶּה / וַאנִי תְהוֹסִיף בְּזֹהֲרִי

גּוֹלֶה צְפוֹנִי

הָחִישׁ גְּאוֹנִי

מֶלֶךְ הֲמוֹנִי

מַהֵר עֲנֵינִי / שִׁמְךָ מְנָתִי וְשׁוֹמְרִי

נָבִיא וְצוֹפֶה / חָלְיֵי כְאֵבִי יְרַפֵּא / אַנְעִים לְקוֹלִי בְזַמְּרִי:


אַצפִי פֻוַאדִי / תַוחִיד רַבּ אַלעִבַּאדִי / מַא דֻמת פִי דַאר חַיוְתִי

גַאיַת מֻרַאדִי / אַסרַאר עִלמִי רְשַׁאדִי / אַלַלה יֻבַּלִּג לִנִיְּתִי

סִרּ אִעתִקַאדי / בִּאלרַּאי וַאלאִעתִמַאדִי / סֻבּחַאן כַּאפִל מְעִישְׁתִי

לִאלכַׄלק הַאדִי / דׄוּ אַלכַּוְן וְשַׁיְ דׄוּ פְסַאדִי / נַאטַק וְאַבכַּם וְצַאמִתִי

פַצַּׄל בִּאִסמֻה

אַסבַּאט עִלמֻה

קַדהֻם בִּקִסמֻה

מִן תַחת חֻכּמֻה / קַומֻן מֻקַדַּס וְטַאהִרִי

אַסרַאר תֻכׄפַא / אַלקַן לְמוּסַי בִּלֻטפַא / פִי כַּתבּ לַוחַין גַׄוְהִרִי:


לָבַשׁ יְקָר הוֹד / מֵאֵשׁ סְנֶה חָשׁ לְוַעַד / שִׁבְטֵי הֲמוֹנַי בְּאַהֲבָה

רַעְיָה וְגַם דּוֹד / נִתְחַבְּרוּ שָׁם לְסַעַד / אַרְבַּע דְּגָלַי לְסוֹבְבָה

וּמֹשֶׁה בְּכָבוֹד / מַנְעִים בְּשִׁירָה וְצוֹעֵד / עִמּוֹ מְטַטְרוֹן לְשַׂר צְבָא

נִצָּב כְּמוֹ נֹאד / הַיָּם כְּיָרֵא וּפוֹחֵד / אֵשׁ יִבְּשַׁתּוּ וְהִחְרְבָה

נַעֲשָׂה דְרָכִים

טוּרִים כְּרוּכִים

לִבְנֵי מְלָכִים

כַּנּאֹד סְבוּכִים / לִפְנֵי שְִׁבַָטַי בְּשַׁחֲרִי

עָמַד וְקָפֶה / טִבַּע לְפַרְעֹה וְנִסְפֶּה / חֵילוֹ בְתוֹכוֹ מְנוֹעֲרִי:


שַׁאפוּן אַלאַעדַא / פַרעוּן וְתִלךּ אַלגֻׄנוּדַא / אַלמַוְג אַוְצַל נְפַדׄ בְּהֻם

נַאלוּן כַּיְדַא / מִן שַׁוְר עִמלַאק, בִּזהַדַא / שַׁדּוּ רִיַאנַאת כַׄילְהֻם

מִן גִַיר גַׄהדַא / מַא נַאלְהֻם רַאי רַשׁדַא / קַד אַחבַּט אַללַּה חַאלְהֻם

בִּלעָם אַבּדַא / אַסחַארְהוּ תֻׄם תִעַדַּא / וְקֻתִל בִּמִדיָן בַּיְנְהֻם

וַאליַאוְנוּסִי

וַאליוּמבַּרוּסִי

אַבּנַא אַלמְגׄוּסִי

נַאלוּ אַלנְחוּסִי / הֻם מַדׄהַבּ אַלזּוּר סַאמִרִי

בִּאלסַּיְף אַנפַא / לַיוַאן דַׄאךּ אַלשְּׁרִיפַא / פִי יַוְם צַׄרבּ אַלבְּוַאתִרִי:


בִּזְכוּת אֲבוֹתַי / יִתְכּוֹנְנוּ אָרְחוֹתַי / נַעְלֶה לְצִיּוֹן וְנִשְׂמְחָה

וּבְמִשְׁכְּנוֹתַי / אֶרְאֶה סְגֻלַּת נְאוֹתַי / יָבוֹא מְשִׁיחִי וְיִזְרְחָה

רֹאשׁ מַחֲנוֹתַי / וּגְאוֹן הֲדַר מִקְהֲלוֹתַי / יָהּ יַעְזְרֵהוּ וְיַצְלְחָה

בֵּין רִבְבוֹתַי / אֶרְאֶה לְנֵס נִפְלְאוֹתַי / אֶחְסֶה בְצוּרִי וְאֶבְטְחָה

יָשׁוּב צְבִי חֵן

עָלַי, וְיָחוֹן

אֶרְאֶה בְּפִתְחוֹן

מַיִם דְּגִיחוֹן / אַשְׁקֶה לְצֹאנִי בְנַהֲרִי

לִבִּי אֲצַפֶּה / וְאֵשׁ כְּבֵדִי כְאוֹפֶה / תַּחְלִישׁ שְׂמֹאל הַמְּאַחֲרִי:


זַיַּן אִלַיהִי / מִן כַּאן זְכִּי עַקל בַּאהַי / לֻהּ עִזּ וְנַאמוּס דַאִימַא

וַאהְל אַלמְלַאהִי / אַפכַּארְהֻם לַא תְּנַאהִי / מִן כַּאן מֻכׄטִא וְאַתׄמַא

דַׄאת אִכּתִדַאהִי / וַאלעַקל שַׁארֵד וְסַאהִי / לַא יַסתְמִע קַול עַאלִמַא

יַבּקַא מֻצַׄאהִי / לְאלגַׄהל בַּיְן אַלסִּפַאהִי / וְלַא יְכַׄאף אַמר חַאכִּמַא

סַאלִבּ לִעַקלֻהּ

תַאבַּע לִגַׄהלֻהּ

רַאבִּי בִּוַהלֻהּ

מַא בַּיְן אַהלֻהּ / תֻּשׁתַּם בְּהוּ אַלעְשַׁאיִרִי

לַא עַאד יֻכׄפַא / פִעלֻה רְדִיֻּן כְּתִׄיפַא / גַׄהַּאר אַעיַבּ מֻנַאכִּרִי:


יַכׄתַץּ לַחנִי / מֻנטִק שִׁפַאתִי וַלִסַאנִי / אַלַּלה אְחַד גַּלּ אִסמְהוּ

הוּ דִׄי כְׄלַקנִי / וַאפִנִי אַלמְהִמַּאת עַנִּי / וַאגִׄבּ עְלַיְּ דִׄכּר שֻׁכּרְהוּ

יַעלַם בִּצַׄנִּי / וִאלַּא דְּעַוְתֻה אֲגַׄבּנִי / וְלַא יֻכַיִּׄבּ לִעַבּדְהוּ

רוּחִי תְמַנִּי / תַזכּוּ תְנַאל מַא וְעַדנִי / וַאזוּר אִלַי בַּית קֻדסְהוּ

אַגנַם זְמַאנִי

פִי קַלבּ הַאנִי

חַיתׄ אַלמְעַאנִי

וַאנַאל אֲמַאנִי / אַסמַע לְזַיְן אַלבְּשַׁאיִרִי

כַּאסִי אַלמֻצַפַא / מֻצהִר לִאַסרַאר תֻכׄפַא / אֻפתִי לִמַן כַּאן חַאיִרִי:

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
