אֲיֻמָּתִי בְּהֶגְיוֹנַי תְּעִירֵנִי

חתימה: אלשבזי


אֲיֻמָּתִי בְּהֶגְיוֹנַי / תְּעִירֵנִי בְּסוֹד יוֹקֶר

וְחֵן לִבִּי וְרַעְיוֹנַי / מְהַרְהֲרִים בְּדִבְרֵי שֶׁקֶר

כִּי רַבּוּ מַשְׂטִינַי / בַּעֲרָב וּבְנֵי עֵקֶר

וּבְהַזְכִּירִי יְגוֹנַי / לְאָחוֹר גּוּפִי נִזְקַר

נַפְשִׁי לָה׳ מִשֹּׁמְרִים לַבֹּקֶר:


וְסוּרִי מֵחֵטְא וְאַשְׁמָה / וַעֲלִי אֶל אַרְמוֹנִי

אָז תִּזְרְחִי כְּחַמָּה / בְּאוֹר חַיִּים קַדְמוֹנִי

לָמָּה יְסִיתֵךְ לָמָּה / נָחָשׁ הַצִּפְעוֹנִי

שְׁתִי מִצְלוּל הַמַּיְמָה / וְאַל יִטְעֵךְ הַצְּפוֹנִי

וְרַצִּי עָלַי קוֹנִי / בְּכָל עֶרֶב וָבֹקֶר:


שׁוּרִי אוֹרַח חַיִּים, וְכִתְבִי נַפְשׁוֹת עֲנִיִּים

אֲשֶׁר הֵם לָאֵל נְקִיִּים / בְּצִדְקָם יְרַוּוּם צִיִּים

[….] / […..]

מִבְּאֵר שֶׁבַע וְאִיִּים / יָשוּבוּ לְבִצָּרוֹן

וְיִהְיֶה לָהֶם יִתְרוֹן / בְּאַשְׁמֹרֶת הַבֹּקֶר:


בְּחֵן אַהֲבַת חֲבַצֶּלֶת / תִּתְנַהֵג יְחִידָתִי

וְתָרוּצָה כְּאַיֶּלֶת / לְבֵית הֵיכַל מְנוּחָתִי

אֲבָל רַגְלִי צוֹלֶלֶת / בְּגָלוּתִי בְּאַשְׁמָתִי

וּבַת שֵׂעִיר צוֹהֶלֶת / מְשַׁחֶקֶת עֲלֵי דָתִי

סְלַח צוּרִי לְחַטָּאתִי / לְהַגִּיד חַסְדְּךָ בַּבֹּקֶר:


זֶה הַיּוֹם קִוִּינוּ / מֵעֲשֶׂרֶת יְמֵי תְּשׁוּבָה

מִקֶּדֶם אִוִּינוּ / לְרַפֹּאת נֶפֶשׁ עֲלוּבָה

בַּיוֹיָ חָסִינוּ / לְהָסִיר לַחַץ מְשׁוּבָה

בְּנֵי עֶלְיוֹן הַשְׂגִּיבוּ / בְּנֵי רָחֵל הָאֲהוּבָה

וְאִם יִשְׁעֵנוּ יָבוֹא / וְחִכִּינוּ לְאוֹר בֹּקֶר:


יוֹם בְּשׂוֹרָה לְעַם קֹדֶשׁ / מֵרֵאשִׁית הַשָּׁנָה

וְקוֹל שׁוֹפָר בַּחֹדֶשׁ / יְעִירֵנִי בְּהַזְמָנָה

לְשׁוֹבֵב אֶת הַנֶּפֶשׁ / בִּתְשׁוּבָה נִבְחָנָה

וְתִנָּצֵל מֵעוֹנֶשׁ / וְהִרְהוּר אָנָה וְאָנָה

עוּרִי עֲדַת מִי מָנָה / יָעִיר בַּבֹּקֶר בַּבֹּקֶר:


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
