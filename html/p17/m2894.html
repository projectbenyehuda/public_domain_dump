בת שבע

וַיְהִי לְעֵת הָעֶרֶב וַיָּקָם דָּוִד מֵעַל מִשְׁכָּבוֹ וַיִּתְהַלֵּךְ עַל-גַּג בֵּית-הַמֶּלֶךְ וַיַּרְא אִשָּׁה רֹחֶצֶת מֵעַל הַגָּג וכו' (שמואל ב' י"א ב')

א.

הַשֶּׁמֵשׁ מִמַּסָּעֶיהָ,

לָלוּן נָטָתָה;

וְזַעַת לְחָיֶיהָ,

בַּעֲנַן-קַל מָחָתָה.

מִכָּל הַר וָגֶבַע,

בָּרְחוּ לָהֶם קַרְנִָים,

לִרְחוֹץ אָז בַּת שֶׁבַע,

יָרְדָה אֶל הַמָּיִם

שָׁם עַל רִצְפַּת סֶלַע

הֵסִירָהּ בְּגָדֶיהָ;

אַף פִּתְּחָהּ מֵהַצֵּלַע,

לוּלָאֵי קִשּׁוּרֶיהָ.

וְאַדְנֵי שֵׁשׁ כְּתֵפֶיהָ,

מִכֻּתָּנְתָּהּ נֶחֱשָׂפוּ,

וּתְאוֹמֵי שָׁדֶיהָ,

מִבַּעַד נִשְׁקפוּ.

בְּרַח לְךָ גֶּבֶר לְנַפְשֶׁךָ!

אַל תִּרְהַב הֵנָּה גֶּשֶׁת;

כִּי יֵצֶר-רַע לְלָכְדֶךָ,

פֹּה טָמַן פַּח וָרֶשֶׁת.

ב.

שָׁם נִשְׁקָף עַל יַד מָיִם,

אַרְמוֹן דָּוִד הַמֶּלֶךְ,

חֵן חֵן לוֹ שִׁבְעָתַיִם,

מִכָּל אַרְמוֹנֵי פֶלֶךְ.

גַּם גַּג מָצוּק כַּנְּחֹשֶׁת,

מִסְּבִיבוֹ יַעַטְרֶנּוּ;

מַעֲשֵׂה אֳמָן בַּחֲרֹשֶׁת,

וּבְזָהָב יְצַפֶּנוּ.

וּמִמִּשְׁכָּבוֹ הַמֶּלֶךְ

קָם טוֹב-לֵב וָרוּחַ;

עַל גַּג זֶה לָשׂוּחַ יֵלֶךְ,

אַף יָתוּר אֶל כָּל רוּחַ.

פֹּה גַן נֶחְמָד לָעֵינַיִם,

שָׁם רִמּוֹן עִם תַּפּוּחַ;

פֹּה יִזְּלוּ יִבְלֵי מָיִם,

שָׁם רוּחַ צַח יָפוּחַ.

גַּם שָׁם בְּנַחַל הַשִּׁבֹּלֶת,

רוּם, תַּחַת מְקַפֶּצֶת;

וּבְתֹכוֹ כַּגִּלְגֹּלֶת,

בְּמֵימָיו מִי רֹחֶצֶת?….

"מִי זֹאת יָפָה כַּשַׁחַר,

מִן הָרַחְצָה עָלָתָה?

וּבְשָׂרָהּ כְּצֶמֶר צַחַר,

בַּשְּׂמִיכָה חִישׁ כִּסָּתָה?"

כָּכָה אֶל לִבֵּהוּ,

אָמַר דָּוִד עָתָּה,

עֵת מִמְּרוֹם שִׁבְתֵּהוּ,

בַּת שֶׁבַע רָאָה מָטָּה.

אוֹ-אָז בְּעָצְמָהּ תֶּרֶב,

אֵשׁ תַּאֲוָה בוֹ לָפוּחַ;

וַתְּלַהֵט סָבִיב קֶרֶב,

נֶפֶשׁ, לֵב, וָרוּחַ.

אָז גַּם יַד הַיֵּצֶר,

לָחֲמָה בוֹ בְכֹחַ;

וְשֹׁרֶשׁ רַע וָנֵצֶר,

הֵחֵל בּוֹ לִצְמֹחַ.

" – כֵּן" – אָמַר אֶל לִבֵּהוּ

יָעַצְתִּי וַאֲקִימֶנָּה;

אוֹ שׁוֹט נָקָם בַּחֲרוֹנֵהוּ,

אוֹ בְחֶסֶד אַמְצִיאֶנָּה!"

וּלְעַבְדוֹ כֹה הִבִּיעַ:

נָא רוּץ כַּחֵץ מִקֶּשֶׁת,

וּמַהֵר לִי תוֹדִיעַ

מִי הִיא זֹאת הָאֵשֶׁת? –

– שְׁמָהּ? עַבְדְּךָ יוֹדֵעַ,

"כִּי זֶה פְעָמִים שֶׁבַע;

"מֵאִישׁ אִישׁ אֲנִי שׁוֹמֵעַ.

“הַיְּפֵהפִיָּה בַּת שֶׁבַע!”

בַּת שֶׁבַע? – פֹּה לָעֶרֶב? –

אֲיָלוֹת שִׂים רַגְלֶיךָ!

גּוּר לְךָ מִפְּנֵי חֶרֶב,

תָּבוֹא רֶגַע אִתֶּךָ.

בֹּשֶׁת, גִּיל וָפַחַד,

בְּדָוִד אָז נִלְחָמוּ;

עֵת רָאָה שְׁנֵיהֶם יַחַד,

פְּעָמֵיהֶם אֵלָיו שָׂמוּ.

ג.

"– פֹּה בּוֹאִי אֶל הַחֶדֶר,

"לָמָּה זֶה תִּירָאִי;

"פֹּה בַת הַחֵן וָהֶדֶר,

“אֲדַבֵּר וְאַתְּ תִּשְׁמָעִי–”

"מִיּוֹם בְּחֶלֶד שָׁאַפְתִּי,

"רוּחַ הַשְׂכֵּל וְיָדוֹעַ;

"לִמְאֹד נִכְסֹף נִכְסַפְתִּי,

“דִּבְרֵי מַלְכִּי לִשְׁמוֹעַ!”

וְלָמָּה כְנִכְנָעַת,

"יָפָתִי כֹּה נִצֶּבֶת?

"וּרְצוֹנִי אַתְּ שׁוֹמָעַת?

“פֹּה כִסֵּא לָךְ לָשֶׁבֶת –!”

"תַּרְבֵּנִי עַנְוָתֶךָ –.

"וּתְמוּרָתָהּ חֵלֶף,

"תִּרְצֶה מֵאֲמָתֶךָ

"קַחַת תֹּדוֹת אֶלֶף –.

"מִמַּלְכֵּךְ מַה תִּדְרוֹשִׁי.

"יָפָתִי נָא אֱמוֹרִי;

"גַּם עֲדִי פָּז תַּחֲבוֹשִׁי,

"וְקִשּׁוּרִים לָךְ תִּקְשֹׁרִי–.

"הֵן אַתָּה שַׁלִּיט פֶּלֶךְ,

"וַאֲמָתְךָ אָנִי;

"וּמַה מָצָאתָ בִּי מֶלֶךְ

"כִּי כֹה כִבַּדְתָּנִי –?

"כִּי מִכְלָל מִכָּל יֹפִי,

"מִנָּשִׁים אַתְּ עֲטֶרֶת;

"חֵן וָתֹם תִּצְנֹפִי,

“כָּל הֶדֶר צְבִי תִפְאֶרֶת.”

"שֶׁקֶר חֵן וָצֶבִי,

"וָהֶבֶל גַּם הַיֹפִי;

"כִּי יַחַד הוֹלְכִים שֶׁבִי,

“בְּיַד פֶּשַׁע וָדֹפִי –.”

–חֵן הוּצַק עַל שְׂפָתָיִם –

חִשֵּׁב דָּוִד כָּכָה;

אֲדַבֵּר אַחַת וְלֹא שְׁתַּיִם,

מַה-בֶּצַע כִּי אֶתְוַכָּחָה.

"נָא שִׁמְעִי אֵשֶׁת חָיִל.

"הַצְּלָלִים כְּבָר עָבָרוּ;

"נָא לִינִי פֹּה הַלָּיִל……!

“אַךְ פָּנַיִךְ מֶה חָוָרוּ? –”

"כִּי נַפְשִׁי הָהּ נִדְהָמָה,

"שָׁלוֹם לְךָ – אֵלְכָה בָית –.

"וּבוֹצֵר זָר הֵן שָׁמָה,

“לֹא יָבֹא אֶל הַזָּיִת……….!”

"לְלִבֵּךְ מַה יִכָּסֶף,

בַּחֲרִי אַל תִּמְנָעִי;

בַּל אֶמְנַע פָּז גַּם כָּסֶף,

בְּקוֹלִי אִם תִּשְׁמָעִי."

"בִּסְגֻלּוֹת בַּקֵּשׁ אֹשֶׁר…?

"תֹּעֵבָה יִבְחַר בָּמוֹ –!

"לֵב טָהוֹר בְּתֹם וָיֹשֶׁר,

"עֹלֶה עַל כֻּלָּמוֹ–….! –

"אַךְ אִמְרִי נָא יָפָתִי,

"הֵן נָטְפוּ שִׂפְתוֹתַיִךְ,

"כִּי תָמִיד אֶל אִמְרָתִי.

“תַּטִּי חִישׁ אָזְנַיִךְ –?”

"יִסְלַח אֲדוֹן הַפֶּלֶךְ,

"אֲכַבֶּדְךָ עוֹד עִם שָׁמֶשׁ –.

"וְעַתָּה – יְחִי הַמֶּלֶךְ –

"אֵלְכָה כִּי קָרוֹב אָמֶשׁ –.

אֵין קֵץ לְדִבְרֵי בָלַע,

חִשֵּׁב דָּוִד הַפָּעַם; –

אָשִׂים מִצְחִי סָלַע,

וּלְשׁוֹנִי לִכְלִי זָעַם:

"עִמְדִי בַת מוֹרֶדֶת,

"מִבֵּיתִי לֹא תֵלֵכִי:

"כִּי חֶרֶב לָךְ שֹׁקֶדֶת,

“אַף תָּבוֹא בְתוֹכֵכִי!”

"גַּם בֵּיתֵךְ וּשְׂדוֹתַיךְ

"בְּעוֹד רֶגַע נִשְׂרָפוּ;

"גַּם אִישֵׁךְ וּבָנַיִךְ

“כַּמֵּתִים חִישׁ נִסְחָפוּ –!”

"אוּרִיָּה –!!" וּמִתְעַלֶּפֶת

נָפְלָה עַל פָּנֶיהָ;

נַפְשָׁהּ הָהּ רֹחֶפֶת

אַף כָּשְׁלוּ בִּרְכֶּיהָ –…

ד.

הוֹי תַאֲוָה קִצְפֵּךְ פֶּתֶן,

לִבֵּךְ עַז כַּנְּחֹשֶׁת;

מִפִּיךְ תְּשַׁלְחִי רֶסֶן,

הָלְאָה תִזְרִי בֹשֶׁת –.

אַתְּ אַתְּ אֵם כָּל פֶּשַׁע,

לַמַּר תָּשִׂימִי נֹפֶת;

לֵב זַךְ תִּהַפְּכִי לָרֶשַׁע,

גַּן עֵדֶן הָהּ – לַתֹּפֶת.

הֲגַם בְּלֵב חַף לָטֹהַר,

בְּלֵב דָּוִד יְקַר עֶרֶךְ,

לֹא יָדַע שַׁחֲרֵךְ מִנֹּעַר,

גַּם שָׁם סַלּוֹת לָךְ דֶּרֶךְ? –

פִּיו מַנְעִים לָאֵל זֶמֶר,

הָפַכְתְּ לוֹ לִכְלִי זָעַם;

בַּעֲבוּרֵךְ אֵיכָה יֶמֶר

מִצְוָתוֹ גַם הַפָּעַם.

כֹּה מֵחֶלְאַת כְּשָׁפַיִךְ,

נֶפֶשׁ דָּוִד נִכְתָּמָה,

כָּכָה עַל בִּרְכַּיִךְ,

גַּם צִדְקָתוֹ הָהּ נִרְדָמָה –.

ה.

הַיָּמִים בֶּחָלֶד,

כִּצְלָלִים יִבְרָחוּ;

וְכָל חָדָשׁ יִוָּלֵד,

עִמָּהֶם יִקָּחוּ.

וּכְכָל אַשְׁמַת רֶצַח,

חַטַּאת דָּוִד נִשְׁכָּחָה;

אַךְ מִסֵּפֶר אֵל נֶצַח,

עֲדֶן לֹא נִמָּחָה.

עוֹד בְּעֵט עֹפֶרֶת,

בְּסִפְרוֹ נֶחֱקֶקֶת;

עוֹד רוּחַ נִשְׁבֶּרֶת,

אֵלָיו נֶאֱנֶקֶת –.

הֵן הַתַּאֲוָה עֹז לָבָשָׁה,

לֹא שָׁקְטָה לֹא נָחָה;

וְכַאֲשֶׁר יָרָשָׁה,

הֲלֹא גַּם רָצָחָה –….

דְּמֵי אוּרִיָּה רָעָשׁוּ,

מִכָּל אַפְסָיִם;

נָקָם דָּרָשׁוּ

מִשּׁוֹכֵן שָׁמָיִם –!!

אָז הִתְאַנַּף אֱלוֹהַ,

בְּמַלְכּוֹ וּבְמָשִׁיחַ;

וְדַרְכּוֹ מִגָּבֹהַּ,

עַל פָּנָיו הוֹכִיחַ:

"קוּם בְּרַח לְךָ כֶבֶד פֶּשַׁע,

נְטוֹשׁ אַרְמוֹן וָפֶלֶךְ!

כִּי לֹא עַל כִּסֵּא רֶשַׁע,

בְּאַרְצִי יִמְלוֹךְ מֶלֶךְ –."

"יַעַן בְּזִיתָנִי,

תִּפָּרֵד מִנָּשֶׁיךָ!….

אַף עָלֶיךָ מֵקִים אָנִי

רָעָה מִבֵּיתֶךָ! –"

אֵין שֶׁבֶר כְּשִׁבְרֵהוּ,

הוֹי לָאָב שֶׁכֹּה קָרָהוּ;

כִּי יוֹצֵא חֲלָצֵיהוּ,

אַבְשָׁלוֹם, רְדָפָהוּ –.

נֶעֱזָב מֵאָח, מֵרֵעַ,

מִבָּנָיו, מִכָּל נָשָׁיו;

כֹּה נָדַד הַפֹּשֵׁעַ,

עִם מִסְפָּר חֵיל אֲנָשָׁיו,

בִּישִׁימוֹן, בְּעַרְבוֹת שְׁאִיּהָ,

תֹּעֶה הוּא כַּהֵלֶךְ;

כֹּה נָדַד בִּמְקוֹם צִיָּה,

דָּוִד – בִּישֻׁרוּן מֶלֶךְ –.

פַּעַם עֲלֵי דֶרֶךְ,

שָׁדוּד מִכָּל כֹּחַ;

מֵרוֹב עָמָל וָפֶרֶךְ,

נַפְשׁוֹ שָׁאֲלָה נֹחַ.

עַנְנֵי אֹפֶל סָכָכוּ,

פְּנֵי כָל הַשָּׁמָיִם;

כֹּכְבֵי רוּם יֶחֱשָׁכוּ,

גַּם הֵם עָצְמוּ עֵינָיִם.

אַךְ בָּדָד שָׁם עֵין סַהַר,

בֵּין מִפְרְשֵׂי עָב צֹפִיָּה,

עַל עִיר, שָׂדֶה גַם יַעַר,

עַל נַהֲרֵי-חֹל וְצִיָּה.

וְלַמְּעָרָה מִתַּחַת,

אוֹר כֵּהָה עוֹד שֹׁלֵחַ;

וְעַל הַיְשֵׁנִים שָׁם בְּנַחַת,

לְבֶן-סוּתוֹ הוּא שֹׁטֵחַ.

אַנְשֵׁי דָּוִד הֵמָּה,

כָּל הַיּוֹם נִרְדָפוּ;

וּבַמְּעָרָה הַשּׁוֹמֵמָה

יִשְׁנוּ, כִּי יִיעָפוּ.

אַךְ לֹא יָנוּם פֶּשַׁע-יֶרֶב,

הוּא לְבַדּוֹ עֵר יִוָּתֶר,

הוּא צָפוּי עֲלֵי חֶרֶב,

שָׁם יֶאֶרְבוּ לוֹ בַסָּתֶר.

– כָּכָה נִבְהָל מֵרוֹדְפֶיהָ,

בִּמְעוֹנָהּ תִּשְׁכַּב הָאַרְנֶבֶת;

וּנְטִיּוֹת עוֹד אָזְנֶיהָ,

תְּרוּעַת אוֹיֵב כְּמוֹ קַשֶּׁבֶת –

מִפַּחַד כְּאֵב וּמַחַץ,

יֵשֵׁב דָּוִד עָתָּה;

וּמִנֵּטֶל פִּיד וָלַחַץ,

רֹאשׁוֹ הוּרַד מָטָּה.

וּבְכִנֹּרוֹ בֵּין בִּרְכָּיִם,

כָּל רֶגַע קוֹל הִשְׁמִיעַ,

לִבְלִי תֵּת שֵׁנָה לָעֵינַיִם,

לִבְלִי יָנוּם יַרְגִיעַ.

וּבְרֹאשׁוֹ אָז תִּנְקֹפְנָה,

לִרְבָבוֹת מַחְשְׁבוֹתָיו;

וּלְפָנָיו אָז תַּחֲלֹפְנָה,

גַּם כָּל עֲוֹנוֹתָיו.

יִזְכּוֹר כִּי הוּא מֶלֶךְ,

מְאֻשָּׁר אֵין כָּמֹהוּ;

וּבְפִשְׁעוֹ תֶמֶס יֵלֶךְ,

מֻרְדָף בִּמְקוֹם תֹּהוּ.

וּבְסֵתֶר לִבּוֹ יְיַסְּרֵהוּ:

–בַּצַּר לָעַד כֹּה תֶחִי! –

כְּאֵב עָצוּר בְּקִרְבֵּהוּ,

יַחְפּוֹץ דֶּמַע, יַחְפּוֹץ בֶּכִי…

אָז – תַּגַּע יָדוֹ עַל הַנֵּבֶל,

מֵיתָרָיו חִישׁ יִרְעָשׁוּ;

וּכְמוֹהֶם בִּנְהִי וָאֵבֶל,

מוֹרָשָׁיו רָחָשׁוּ.

וּבְשִׁיר עֵת מְנַצֵּחַ,

עֵינוֹ גַּם נִגֶּרֶת;

כִּי לָאֵל אָז הוּא זֹבֵחַ

כָּלִיל רוּחַ נִשְׁבֶּרֶת:

"חָטָאתִי אֵל נֶגְדֶּךָ,

מֵעֲוֹנִי נָא כַבְּסֵנִי;

וּכְרוֹב חֲסָדֶיךָ,

חָנֵּנִי אֵל חָנֵּנִי.

הָרַע עָשִׂיתִי בְעֵינֶיךָ,

נִחַמְתִּי – נָא שְׁמָעֵנִי;

אַל תַּשְׁלִיכֵנִי מִלְּפָנֶיךָ,

חָנֵּנִי אֵל חָנֵּנִי.

לְךָ חָטָאתִי לְבַדֶּךָ,

אַל בְּאַפְּךָ תוֹכִיחֵנִי

הֲלֹא זֹאת מִיָּדֶךָ,

חָנֵּנִי אֵל חָנֵּנִי.

אֱצֹל לִי מֵאוֹרֶךָ,

וּבְרוּחַ נָכוֹן תְּחַדְשֵׁנִי,

אָז אֵלֶךְ בִּדְרָכֶיךָ,

חָנֵּנִי אֵל חָנֵּנִי.

הָשִׁיבָה לִי שְׂשׂוֹן יִשְׁעֶךָ,

אוֹת לְטוֹבָה נָא הַרְאֵנִי;

תִּשְׁלַח לִי רוּחַ קָדְשֶׁךָ,

חָנֵּנִי אֵל חָנֵּנִי.

אָז חַטָּאִים יָשׁוּבוּ אֵלֶיךָ,

בִּרְאוֹתָם כִּי תְחַטְּאֵנִי;

אֲלַמְּדָה פוֹשְׁעִים דְּרָכֶיךָ,

חָנֵּנִי אֵל חָנֵּנִי

הֵן גָּבְרוּ חֲסָדֶיךָ,

נָא – מִדָּמִים הַצִּילֵנִי……

אָז אֲרַנֵן תְּהִלָּתֶךָ,

חָנֵּנִי אֵל חָנֵּנִי.

לֵב נִשְׁבָּר יָבוֹא עָדֶיךָ,

הוֹאֵל נָא וּרְצֵנִי;

סְפוֹר דִּמְעוֹתַי בְנֹאדֶךָ,

חָנֵּנִי אֵל חָנֵּנִי.

וְתַחַת מִשְׁפָּטֶיךָ,

שָׂשׂוֹן תַּשְׁמִיעֵנִי;

כִּי אֶשְמוֹר פִּקּוּדֶיךָ,

חָנֵּנִי אֵל חָנֵּנִי."

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
