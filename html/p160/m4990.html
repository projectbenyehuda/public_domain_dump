תלונות החיים

שיר למודים 1

כְּמוֹ שַׁחַר עָלָה גַּם לִי אוֹר הַחַיִּים

כְּשׁוֹשַׁנֵּי חֶמְדָּה יְמֵי נְעוּרַי פָּרָחוּ

תִּקְוָה נִצָּבָה לִימִינִי בְּיִפְעַת עֵינָיִם

נְגֹהוֹת עֵדֶן עַל נְתִיבוֹתַי זָרָחוּ;

אוֹר-יְקָרוֹת אוֹר עוֹלָם כְּמוֹקֵד מִזְבֵּחַ

עַל הֲדַר שְׁנוֹתִי לִרְגָעִים זוֹרֵחַ;

בְּסֵתֶר לְבָבִי הָגוּת-עֹז מִתְהַלֶּכֶת

בִּנְעִימוֹת פֶּה תְּחַדֶּה יָמִים נִצְפָּנוּ

הִיא בְּרָב כֹּחַ לִמְעוֹן עֶלְיוֹן דּוֹרֶכֶת

שִׂמְחוֹת עוֹלָם תְּחָנֵנִי אֲשֶׁר לֹא יִמָּנוּ;

וּבְכֵן יִדּוֹדוּן מַלְאֲכֵי גִיל חָלֶד

כְּגַלֵּי יָם לְעֵת בּוֹקֶר מִמִּזְרָח יִוָּלֶד

אַךְ פִּתְאוֹם הֱקִיצוֹתִי – וְהִנֵּה שַׁמָּה וּשְׁאִיָּה!

וְעַל מַעְגַּל דְּרָכַי מִנְהֲרוֹת תַּחְתִּיָּה!

בְּחֵקֵר לְבָבִי נָשָׂאתִי עֵינַי שָׁמַיִם

אֱמֶת קִוִּיתִי כְּמַלְאַךְ אֵל תְּבוֹאֵנִי

אָמְנָם אַךְ חֶזְיוֹן רוּחִי תֵּחֱזֶינָה עֵינָיִם

אָמַרְתִּי אֶחְכָּמָה וְהִיא רְחוֹקָה מִמֶנִּי!

לָמָּה לִי אוֹר עַל נְתִיבוֹתַי זוֹרֵחַ

אִם כִּבְרַק לַיִל יָהֵל – אַךְ בַּלְהוֹת-אֲפֵלָה;

לָמָּה לִי אוֹר מִנִּי דַּעַת בּוֹרֵחַ

וַיַּסַּע כָּל-תִּקְוָה בְּנֶפֶשׁ יִחֵלָה?

מַה-תִּסְכּוֹן מַה-תּוֹעִיל לִי בִּין! גַּבְהוּת חָלֶד!

אִם תַּעְכּוֹר נַפְשִׁי דָּבְקָה בֵּאלֹהֶיהָ?

קַח לָךְ שַׁעֲשׁוּעֶיךָ! שָׁגִיתִי כְּמוֹ יָלֶד

וְהָשֵׁב לִי אֱמוּנָתִי וְשׁוֹד תַּנְחוּמֶיהָ!

כִּי רַבָּה אַהֲבָתָהּ בְּצוּף לִקְחָהּ תְּשִׂיחֵנִי

לְבָבִי בְּלַחַץ צָרָה בָּהּ הִיא בָטוּחַ

בְּחֵיקָהּ אֶשְׁכָּבָה בְּצֵל-כְּנָפָהּ תַּסְתִּירֵנִי

לֹא תִתֵּן לִנְגּוֹעַ בִּי שֹׁד פִּתְאוֹם שָׁלוּחַ! –

כְּמוֹ מִקִּלְלַת-אֱלֹהִים נֶפֶשׁ אָדָם נִבְהָלָה

עֵת מִגַּן גְּרָשָׁהוּ לַהַט חֶרֶב מִתְהַפֶּכֶת

הָהּ! כֵּן אֲנִי גֹוֵעַ כֵּן נַפְשִׁי אֻמְלָלָה

לִרְאוֹת חֲמוּדִי נָמֵס וְעֵץ-חַיַּי בְּשַׁלֶּכֶת.

שָׁלוֹם הָיָה סְבִיבִי פְּעָמַי בְּאוֹר הָלָכוּ

לֹא צָרוּ צְעָדַי לֶכֶת דֶּרֶךְ הַחַיִּים

כְּמוֹ עַל בָּמֳתֵי מָרוֹם רַגְלֵי אָבִיב יִדְרָכוּ

וּבְשַׂלְמַת אוֹר יְבוֹאֵנִי בַּעֲדִי עֲדָיִים.

הָהּ חֶמְדַּת עֲלוּמַי עָשׂוּ לָהֶם כְּנָפַיִם!

רַגְלַי עָמְדוּ בְּמִישׁוֹר, עַתָּה בְּנִבְכֵי עֲמָקִים

מֵאַיִן יָבוֹא עֶזְרִי אֶל הָרִים אֶשּׂא עֵינַיִם

אָבוֹא אֵלֶיךָ אֵל? אַרְקִיעַ עִמְךָ לִשְׁחָקִים? –

יָמַי נִזְעָכוּ וְאֵינָם לָעַד לֹא יְשׁוּבוּן

כְּיֶרֶק דֶּשֶׁא יִבּוֹלוּן לָעַד לֹא יְנוּבוּן

כְּאוֹר נִתְעֶה בַּשּׁמַיִם יָעוּף וָאָיִן

עַד עוֹלְמֵי עַד לֹא יָשׁוּב לֹא תְשׁוּרֶנּוּ עָיִן.

רַבִּים אוֹמְרִים: דִּמְעוֹתֵינוּ מְאֹד יִמָּנוּ

נְעִימוֹת נֶצַח בְּסֵתֶר עֶלְיוֹן נִצְפָּנוּ!

אוּלָם אוֹר-חַיִּים אַחֲרֵי יֶחְדַּל הוֹפִיעַ

מָה הָאוֹת כִּי לַבֹּקֶר עוֹד יָשׁוּב יַגִּיהַּ?

הֵן בִּימֵי טוֹבָה מְאֹד נַפְשֵׁנוּ יוֹדַעַת:

כִּי בְּאוֹצְרוֹת שַׁדַּי שְׁמוּרָה מַשְׂכּוֹרֶת תְּמִימִים

לֵב נִפְתֶּה עַל טוֹב בְּרוּחַ תִּקְוָה נִטַּעַת

כָּל עוֹד נִטָּיוּ נַחֲלֵי עֲדָנִים נְעִימִים;

אוּלָם הַלֵּב יִפּוֹל רֶגַע תָּבוֹא צָרָה

תִּקְוָה תִּסּוֹג אָחוֹר תַּעֲזוֹב נֶפֶשׁ מָרָה!

כֹּל תֶּחֱזֶה עֵינִי לוּטָה בְּצַלְמָוֶת דּוּמִיָּה

הַשּׂפֶק 2 נֵעוֹר מִזְּבוּל חָשְׁכּוֹ רֵבֶץ שְׁאִיָּה

יַעֲמוֹד יִתְבּוֹנֵן בִּי יְסוֹבְבִנִי חִיל וּרְעָדָה

לְיַד מַעְגָּלַי יִפְרוֹשׂ פַּחִים וְנַפְשִי נִלְכָּדָה,

יִשְׁאָלֵנִי: אֵי מִזֶּה בָּאוּ חַיִּים וּמָוֶת?

חַיִּים כִּסָּה חֹשֶׁךְ, בְּחָשְׁכּוֹ יִתְכָּס צַלְמָוֶת

חַיִּים מִן עֲרָבוֹת יָבוֹאוּ מִמִּסְתָּרִים

כְּמִקֶּדֶם אוֹר נבְרָא מִנִּי חַדְרֵי בֹהוּ

מָוֶת עֲרָפֶל חֲתוּלָתוֹ בְּמִנְהָרִים

כְּהִנְדּוֹף עָשָׁן יִנְדּוֹף וְיַעַל בָּתֹּהוּ!

בְּחֵקֶר לְבָבִי כִּי אֵלֵךְ בּגֵיא הַחַיִּים

אָז תָּעָה לְבָבִי בֵּין חָרְבוֹת שְמָמָה

לְכָל תִּכְלָה רָאִיתִי קֵץ תַּחַת שָׁמַיִם

צֶאֱצָאַי כִּלְבוּשׁ יַחֲלוֹפוּ תֹּאמַר אֲדָמָה.

כִּי אֶשּׂא שָׁמַיִם עֵינַי בְּגִיל רְעָדָה

אֶתְבּוֹנֵן מִפְלְאוֹת מִקְדְּשֵׁי כּוֹכְבֵי אוֹרָה

אָז נַפְשִׁי מִמְרוֹם עוּפָהּ מָטָה יָרָדָה

נִלְאָה חֲזוֹת שָׁמָּה חַפֶץ אֵל נוֹרָא.

כְּאַיָּל עַל אֲפִיק נְחָלִים נַפְשִׁי מְמהֶרֶת

שְׁתוֹת מִפֶּלֶג אֵל בָּהּ שַׁלְהֶבֶת יָהּ בּוֹעֶרֶת

הֲלָעַד תֶּאֱנַף אֵל? תִּמְסוֹךְ כּוֹסִי רְוָיָה

לְהַפִּילֵהוּ מִיָּדִי עֵת לֶאֱמֶת אֶהֱמְיָה?

קוֹלִי לְאֵל וְאצְעָקָה, אֶזעַק רֹאשׁ כּוֹכָבִים

אָשִׂיחָה וְאֶהֱמֶה אֵת פְּלִיאָה מֶנִּי נִשְׂגָּבָה,

אַךְ מִי זֶה יַאַסְפֵנִי? יְחוֹנֵן מְטִיל רְגָבִים,

לְהַשְׁקִיט אֵת תִּשְאַל נַפְשִׁי, נַפְשִׁי הַנֶּעֱזָבָה?

בָּדָד בְּאִישׁוֹן לַיְלָה נַפְשִׁי מִתְהַלֶּכֶת

רוּחִי תִּרְכַּב עֲרָבוֹת עַל בָּמֳתֵי עָל דּוֹרֶכֶת

מִי יְבִינֵנִי תָּו לְהָבִים אוֹר שִׁבְעָתָיִם

מִכְתַּב-אֱלֹהִים חָרוּת עַל לוּחַ שָׁמַיִם?

הֲיִתְהַלֵךְ אֱלֹהִים בְּאוֹר כּוֹכְבֵי רָקִיעַ?

עַל גּלֵּי יָם בְּגַאֲוָתוֹ לִוְיָתָן יַשְׁקִיעַ?

אַכִּיר הוֹד תִּפְאַרְתּוֹ בְּעָשׁ זוֹחֵל אֲדָמָה

אוֹ בִיקָר אָדָם לִבְנֵי אלֹהִים דָּמָה?

אֶשְׁמַע קוֹל פַּחְדּוֹ עֵת יֶהֱמֶה בַּגַּלְגַּל רָעַם?

אַאֲזִין הוֹד קוֹלוֹ בְּפֶלֶג יִזַּל הַפָּעַם?

הַבֵּט כּוֹכְבֵי אוֹר רִבּוֹ רִבּוֹתָיִם

יְנַהֲלוֹ רוּחַ נָכוֹן אֶל אֵל בַּשּׁמָיִם;

אוּלָם מִבָּמֳתֵי עֶלְיוֹן כִּי אַשְׁקִיף בַּשּׁפֶל

אָז אֶחְזֶה חֶלֶד חֲתוּלָתָהּ עֲרָפֶל.

פֹּה סוּפָה תֶּהֱמֶה פֹּה יָחוּל סַעַר

וְהָגוּת שָׂפֶק בְּלֵב אֱנוֹשׁ שְׁנָתָהּ נָמָה

לְקוֹל הַסַּעַר תֵעוֹר מִגַּעֲרָתוֹ קָמָה

שַׁעֲרֵי שְׁאוֹל תְּרָדְפֵהוּ וִינָגֶף בַּשּׁעַר. –

עַל עָב קַל תִּרְכַּב שֶׁמֶשׁ גִּיל נֶאֱזֶרֶת!

אוּלָם לְהוֹלְכֵי נְתִיב חֹשֶׁךְ לֹא הִיא מַזְהֶרֶת.

רְאֵה נְלוֹזֵי-לֵב בָּמֳתֵי אֶרֶץ יִדְרָכוּ

יַצְלִיחוּ דַרְכָּם כִּי יֵשׁ אֱלֹהִים יִשְׁכָּחוּ

רְאֵה שָׁמָּה אוֹרְבֵי דַּם נָקִי בְּנֵי עַוְלָתָה

עַל טַרְפָּם יִלְטְשׁוּ עָיִן וְרַבָּה הַצַעֲקָתָה;

עַד אָנָה תִּזְנַח אֵל תֶּחֱרַשׁ לָנֶצַח?

עַד אָנָה על טְהוֹרֵי לֵב יִמְשׁוֹל זָדוֹן וָרֶצַח?

שָׁם בִּנְוַת חֹשֶׁךְ דַּכְּאֵי רוּחַ מַר נֶאֱנָחוּ

צֶדֶק יִמּוֹט כְּשַׁל תּוֹם וְרִשְׁעָה אַךְ פּוֹרִיָּה

קוֹמְמִיּוּת יֵלֵךְ עַוָּל תְּמִימִים כַּצֵּל נֶהֱלָכוּ,

וְהֶעָנִי בְּשֵׁבֶט עֶבְרָה יַךְ תֵּבֵל שְׁאִיָּה;

שָׁם תַּחַת אַלּוֹן בָּכוּת יֵשֵׁב שַׁח עֵינָיִם

יֵשֵׁב בָּדָד וְיִדּוֹם יִסְבֹּל עֶבְרַת הַחַיִּים;

רִשְׁעֵי אֶרֶץ בְּלַעֲגֵי מָעוֹג יְחָרְפוּ תְּמִימִים

עַל הוֹלְכֵי תוֹם חֶרְפַּת רְשָׁעִים נָסַבָּה

אוֹיְבֵי אֵל יִשְׂמָחוּ יְבַלּוּ שְׁנֵיהֶם בַּנְּעִימִים

וְיָשָׁר בּוֹטֵחַ בַּאדֹנָי – הוּא יִמּוֹט רַבָּה!

אוֹ הֲכִי פֶּגַע יוֹלִיךְ כֹּל דֶּרֶךְ נִסתָּרָה?

עַל פִּיו יֵלְכוּ כּוֹכְבֵי אוֹר לְרגְלוֹ זֹאת אֲדָמָה?

הוּא אָמַר וַיְהִי טוֹב, צִוָּה וַתָּבוֹא צָרָה?

לֹא-סְדָרִים בְּפָעֳלוֹ אֵין חֵפֶץ בַּמָּרוֹם שָׁמָּה?

אִם כָּל עוֹלָמוֹת כְּעוֹלָלִים בְּלִי אֵם יִבְכָּיוּן?

מֵרֶחֶם דֹּחוּ וּבְאֵין מְרַחֵם יֶהֱמָיוּן?

אָמְנָם חֲזֵה! עַל כָּל נֶטַע חָכְמָה נִשְׂגָּבָה

כַּנֵּר אֱלֹהִים תָּהֵל הִיא הִיא תְּעִידֵהוּ!

עֲצַת אֲדֹנִי הֲלֹא מִנִּי יָם רְחָבָה,

אֶרֶז יְפָאֵר אֵל גַּם הָאֵזוֹב יְרוֹמְמֵהוּ!

אוּלָם הֵן לֹא סְדָרִים עוֹד רַבּוּ בַּחַיִּים

אֵי פֹּה חָכְמַת אֵל? הֲלֹא נִסְתְּרָה לָעֵינָיִם?

אַךְ זְרוֹעַ קָשָׁה בַּמִּסְתָּרִים עוֹבֶרֶת

כָּל חֶמְדָּה בַּחַיִּים בְּשֵׁבֶט אַף מְיַסֶּרֶת

כָּל מַטַּע לְהִתְפָּאֵר בְּכַפָּהּ הִיא נוֹבֶלֶת

תְּמָרֵר נֹעַם אַהֲבָה תְּכַזֵּב חֲזוֹן תּוֹחֶלֶת.

קוֹל שִׂמְחָה קוֹל נֶהִי כָּל הַיּוֹם יְסוֹבְבֵנִי

עַל אָהֳלִי מִתְדַפְּקִים וַתֹּאחְזֵנִי חֲרָדָה

זֶה יִקְרָא לֶאֱמוּנָה וְזֶה בְּשֶׂפֶק יֶהֶלְמֵנִי

וְרוּחִי תֵלֵךְ שׁוֹלָל נָעָה גַּם נָדָה!

"הַגִּידוּ לִי! אֵיפֹה מְנוּחָה? אֵיפֹה דּוּמִיָּה?

כֵּן תִּשְׁאַל נֶפֶשׁ דַּכָּא וַתְּמָרֵר בַּבֶּכִי

"אֶל הֵיכְלֵי אוֹר בַּשּׁמַיִם עֵינִי צוֹפִיָּה

הֲכִי שָׁמָּה יִשְׁלָיוּ כָּל הוֹלְכֵי דֶחִי?

אוֹ גַם בַּמְּרוֹמִים אֵין שָׁלוֹם וְשַׁלְוָה נֶעְדָּרָה

אֵין חֲלִיפוֹת לְלֵב נִדְכֶּה רוּחַ נִשְׁבָּרָה?"

קַוֵּה בֶּן אֱנוֹשׁ! יַחֵל עַד קֵץ יָמֶיךָ

אֱמוֹר לְנַפְשְׁךָ: יֵשׁ אַחֲרִית בָּעֲרָבוֹת שָׁמָּה!

אַךְ תִּקְוָה מְנַת חֶבְלָךְ בַּמֵּיצַר אֲדָמָה,

עִמְךָ עַד שַׁעֲרֵי מָוֶת תֵּלֵךְ לְשַׁלְחֶךָ. –

תִּקְוָה! מַה נָעַמְתְּ! מַלְאֲכֵי שָׁלוֹם כִּנּוֹר יָעִירוּ

אָז עַל בִּרְכֵּךְ תְּיַשּׁנֵנִי וְרוּחַ טוֹב מְרַחֶפֶת

עַל פָּנַי תַּחֲלִיף תֵּבֵל כָּל יְקָר מְעֻלֶּפֶת

מְאוֹרֵי עוֹלָם מַחֲשַׁכֵּי אֶרֶץ יַזְהִירוּ,

שׁוֹד מִשּׁדַּי נָסוֹג אָחוֹר יוּשַׁב גַּן עֲדָנִים

הָאָדָם יִתְהַלֵּךְ כְּקֶדֶם בֵּין נִטְעֵי נַעֲמָנִים

רוּחַ אֱלֹהִים בְּרָגֶשׁ בַּגָּן מִתְהַלֶּכֶת,

עֵץ חַיִּים לֹא תִשְׁמוֹר עוֹד חֶרֶב מִתְהַפֶּכֶת.

הָהּ! מִי יִתְּנֵנִי לְעוֹלָם מִתְהַלֵּךְ בִּיקְרוֹתֶיךָ

כִּי נָעַמְתְּ לִי תִּקְוָה! מְאֹד שָׁפְרָה נַחֲלָתֶךָ!
