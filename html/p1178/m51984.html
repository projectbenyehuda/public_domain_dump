דניאל בגוב האריות

מֻגָּשׁ

למעלת אדון רם ונשא, שר וגדול בישראל, זורע צדקות ומצמיח ישועות, יועץ וחכם חרשים, אזרח נכבד לדֹרותיו,

כהר"ר

לאזאר בראָדסקי.

הי"ו.

בהגות כבוד ויקר,

מאת

המחבר.


שְׁפַר קֳדָם דָּרְיָוֶשׁ וַהֲקֵים עַל־מַלְכוּתָא לַאֲחַשְׁדַּרְפְּנַיָּא מְאָה וְעֶשְׂרִין דִּי לֶהֱוֺן, בְּכָל־מַלְכוּתָא: וְעֵלָּא מִנְּהוֹן סָרְכִין תְּלָתָה דִּי דָנִיֵּאל חַד מִנְּהוֹן דִּי לֶהֱוֺן אֲחַשְׁדַּרְפְּנַיָּא אִלֵּין יָהֲבִין לְהוֹן טַעֲמָא וּמַלְכָּא לָא לֶהֱוֵא נָזִק: אֱדַיִן דָּנִיֵּאל דְּנָה הֲוָה מִתְנַצַּח עַל סָרְכַיָּא וַאֲחַשְׁדַּרְפְּנַיָּא כָּל קְבֵל דִּי רוּחַ יַתִּירָא בֵּהּ וּמַלְכָּא עֲשִׁית לַהֲקָמוּתֵהּ עַל כָּל מַלְכוּתָא: אֱדַיִן סָרְכַיָּא וַאֲחַשְׁדַּרְפְּנַיָּא, הֲווֹ בָעַיִן עִלָּה לְהַשְׁכָּחָה לְדָנִיֵּאל מִצַּד מַלְכוּתָא וְכָל עִלָּה וּשְׁחִיתָה לָא יָכְלִין לְהַשְׁכָּחָה כָּל קְבֵל דִּי מְהֵימַן הוּא וְכָל שָׁלוּ וּשְׁחִיתָה לָא הִשְׁתְּכַחַת עֲלוֹהִי: אֱדַיִן גֻּבְרַיָּא אִלֵּךְ אָמְרִין דִּי לָא נְהַשְׁכַּח לְדָנִיֵּאל דְּנָה כָּל עִלָּה לָהֵן הַשְׁכַּחְנָא עֲלוֹהִי בְּדָת אֱלָהֵהּ:

וגומר.

(דניאל ו', ב־ו.)


שָׁאוֹן מִקָּצֶה, קוֹל פְּחָדִים מֵעֲבָרִים,

תִּרְעַד הָאָרֶץ, יִרְגְּזוּ מוֹסְדֵי הָרִים,

כְּמוֹ תִבָּקַע בִּטְנָהּ קְרָבֶיהָ יֻכַּתּוּ

וּתְהֹמוֹת יֶהֱמָיוּ תַּחְתִּיּוֹת חַתּוּ –

הָרְפָאִים תַּפִּיל אֶרֶץ, תָּקִיא רֶשַׁע?

אוֹ כִי פוֹר תִּתְפּוֹרֵר מִכֹּבֶד פֶּשַׁע?

כִּי רַבָּה רָעַת הָאָדָם מִכָּל עֲבָרָיו

לֹא תוּכַל הָאָרֶץ לְהָכִיל אֶת כָּל דְּבָרָיו!

לֹא! אִם חֹזִים מֵאָז כֵּן לָנוּ הִגִּידוּ

אַךְ בִּמְלִיצָה חִידוֹת חָזוֹן הֶעֱמִידוּ

כִּי חֻקּוֹת שָׁמַיִם וָאָרֶץ יִשָּׁבֵרוּ

אִם חֻקּוֹת הַחַיִּים בְּנֵי אָדָם יָפֵרוּ.

רָאוֹת רַבּוֹת נִרְאֶה מֵאָז וָעָתָּה

כִּי מֵרָעוֹת רַבּוֹת אֶרֶץ לֹא מָטָה,

מִשֹּׁאַת רְשָׁעִים שָׁמַיִם לֹא שָׁאוּ

מֵרוּחַ עָרִיצִים מְאֹרוֹת לֹא כָהוּ,

כּוֹכַב בֹּקֶר יְשׁוֹרֵר בְּחַלּוֹן רָקִיעַ

שַׁוְעַת עֲשׁוּקִּים שָׁמָּה לֹא תַגִּיעַ,

יָשִׂישׂ הַשֶּׁמֶשׁ לָרוּץ אֹרַח שָׁמָיו

עֵת קָדְקֹד שֵׂעָר מִתְהַלֵּךְ בַּאֲשָׁמָיו,


לֹא תָנוּט הָאָרֶץ לֹא תֵט אַף שַׁעַל

אִם חָמָס מִלְּאוּהָ כָּל רָע וּבְלִיַּעַל.

הִנֵּה גַם עַתָּה הָאָרֶץ שֹׁקֶטֶת

עֵת שָׁמָּה כָּאֵשׁ הָרִשְׁעָה לֹהֶטֶת,

שָׁם חֶבֶר מְרֵעִים צַדִּיק כַּבִּיר הִרְשִׁיעוּ

לִמְנוֹתוֹ אֶת פּשְׁעִים בַּמֶּלֶךְ הִפְגִּיעוּ

וּבֶאֱשׁוּן לַיְלָה אִישׁ חֲמֻדוֹת הוֹבִילוּ

וּלְבוֹר הָאֲרָיוֹת אוֹתוֹ הִפִּילוּ

וּפִי הַבּוֹר אָטְרוּ, בּוֹ אֶבֶן שָׂמוּ,

בְּטַבַּעַת הַמֶּלֶךְ וּבְטַבַּעְתָּם חָתָמוּ,

וּבְנָפְלוֹ הַבּוֹרָה בֵּין לָבִיא וּלְבִיָּא

רָעֲשׁוּ הָאֵיתָנִים וַתְּהִי שַׁעֲרֻרִיָּה

יִרְעַם הַבּוֹר וּמְלֹאוֹ מִקּוֹל צֹרֵחַ –

הוּא הַקּוֹל תַּחְתִּיּוֹת אֶרֶץ פֹּלֵחַ.


מִשַּׁאֲגַת אַרְיֵה, נַהַם כְּפִיר וְקוֹל שָׁחַל

תָּמְהוּ שָׁמַיִם אֶרֶץ רָגְזָה וַתָּחַל

וַיְזַנֵּק הָאַרְיֵה, כְּפִירִים דָּהָרוּ,

יֶהְגּוּ הַלְּבָאִים, גּוּרֵיהֶם נָעָרוּ,

שׁוֹת שָׁתוּ אֶל דָּנִיֵּאל בַּיַּרְכָתָיִם

פָּצוּ עָלָיו פִּיהֶם חָרְקוּ שִׁנָּיִם

וּפְרִיץ חַיּוֹת מִהֵר לַעֲבוֹר מִשְׁנֵהוּ

נֵתַח טוֹב לוֹ לִבְחוֹר מִבְּשַׂר טַרְפֵּהוּ,


עוֹד רֶגַע וְדָנִיֵּאל נִקְרַע לִגְזָרִים

בְּבֶטֶן אֲרָיוֹת לוֹ נָכוֹנוּ קְבָרִים

וְהִנֵּה אַרְיֵה, הַזָּקֵן שֶׁבַּחֲבוּרָה

קָם וַיַּחְפֹּץ זְנָבוֹ נֶאְזָר בִּגְבוּרָה

וּבְעֹז פְּנֵי מֶלֶךְ רַעְמָתוֹ הֵנִיעַ

וַיַּהַס הָעֵדָה, אֹמֶר הִבִּיעַ:

– הַרְפּוּ וָדֹמוּ! אַל תָּזִידוּ לִנְגֹּעַ

אַחְוָתִי בְאָזְנֵיכֶם שִׁמְעוּ שָׁמֹעַ:

אוֹיֵב וּמִתְנַקֵּם אֲנִי מֵעוֹלָם

לִבְנֵי הָאָדָם מִקְּטַנָּם עַד גְּדוֹלָם,

פְּנֵי אָדָם לִי מִפְלֶצֶת מַרְאָיו זַעֲוָה

מְתוּקִים לִי דָמָיו בְּשָׂרוֹ מַאֲכַל תַּאֲוָה,

עַל אָדָם כִּי אֶפְגַּע יִתָּפֵשׂ בְּכַפִּי

בְּאַכְזְרִיּוּת חֵמָה אָפִיץ עֶבְרוֹת אַפִּי,

כִּי תַכְלִית שִׂנְאָה אֲיֻמָּה כַמָּוֶת

אֶשְׂנָא זֶה הָאָדָם – רַהַב הוּא שָׁבֶת

הָאוֹמֵר בִּלְבָבוֹ: אֲנִי וְאַפְסִי עוֹד!

וְהוּא חַלָּשׁ וְנִקְלֶה, קָטֹן וּבָזוּי מְאֹד,

וְזֶה הַחַלָּשׁ הַנִּקְלֶה – נוֹרָא מִשְּׁמֹעַ! –

אֵיתָנִים יְסַלֵּף כַּבִּירִים יָרֹעַ

כָּל חַיְתוֹ יַעַר וְשָׂדֶה לוֹ סִיר רַחַץ

וּמִתְאַמֵּר לְמֶלֶךְ עַל כָּל בְּנֵי שַׁחַץ


וּבְעָצְמַת כְּשָׁפָיו זֶה הוֹלֵךְ עַל שְׁתַּיִם

כֹּל הוֹלֵךְ עַל אַרְבַּע יַגִּישׁ לִנְחֻשְׁתַּיִם

מְשַׁדֶּד אָב, יַבְרִיחַ אֵם, יַפְרִיד בֵּין אַחִים

יִתְפֹּשׂ בְּשַׁחְתּוֹ, יִתֵּן בַּסּוּגַר בַּחַחִים.

גַּם אוֹתִי שָׁדַד זֶה מַשְׁחִית הַיַּעַר

וַיְגוֹרֵנִי בְחֶרְמוֹ עוֹדֶנִּי נַעַר

וּלְמֶרְחַקִּים הוֹבִילַנִי בַכֶּלֶא

וַיַּפְלִיא מַכּוֹתַי לְאַלְּפֵנִי פֶלֶא

לִצְנוֹף גֵּוִי הִתְהַלֵּךְ עַל פִּי חֶבֶל

לָלֶכֶת רוֹמָה לִפְרֹט עַל פִּי נֵבֶל.

לְרַאֲוָה בִי נְתָנַנִי לִשְׂחֹק וְשַׁעֲשֻׁעִים

לְעֵינֵי הֲמוֹנָיו עָשִׂיתִי תַעְתֻּעִים.

עַל מִשְׁבַּתַּי שָׂחֲקוּ מָחֲאוּ כַפָּיִם

וַתִּצַּת נִקְמָתִי נְקַם אַחַת אַפָּיִם

וַיְהִי כִּי חִבַּקְתִּיו לֶחְיוֹ לָקַקְתִּי

פִּתְאֹם רֹאשׁוֹ מִמּוּל עָרְפּוֹ מָלַקְתִּי

בַּלָּהוֹת הִפַּלְתִּי וַהֲמוֹנִים רָמַסְתִּי

וּבֵין הַפְּגָרִים לִי נָתִיב פִּלַּסְתִּי

וָאָסֹב בָּעִיר מֵעֶרֶב עַד נֶשֶׁף

וּלְפָנַי יֵלֵךְ דֶּבֶר לְרַגְלַי רֶשֶׁף,

הִרְבֵּיתִי חֲלָלִים, רַבָּה נִקְמָתִי,

לְאָדָם, הָהּ, נִדְמֵיתִי וַתְּהִי מְבוּכָתִי!


עַד כֹּה וָכֹה צַיָּדִים הִתְמַלָּאוּ

כִּתְּרוּנִי הִדְרִיכוּנִי רַגְלִי פָצָעוּ

עִנּוּהָ בַּכֶּבֶל גֵּוִי בַזִּקִּים

וָאֵלֵךְ מַר לִפְנֵי שׁוֹבִים מְצִיקִים

וַיָּבֹאוּ עַד הֲלוֹם וַיְפַתְּחוּנִי

וְאִתְּכֶם, אַחִים לַצָּרָה, פֹּה שָׂמוּנִי. –

עַתָּה הֲלֹא תָבִינוּ שׁוֹמְעֵי דְבָרִי

כִּי לִשְׂנוֹא אָדָם, זֶה אוֹיְבִי, זֶה צָרִי,

לִי הַצְּדָקָה מִכֻּלְּכֶם שִׁבְעָתַיִם,

אֶפֶס שִׂנְאָתִי לֹא עִוְּרָה לִי עֵינַיִם,

אִם לֵב בְּנֵי אָדָם מָלֵא רָע וָסֶלֶף

גַּם טוֹב בָּם מָצָאתִי, אֶחָד מֵאֶלֶף,

כִּי לֹא כֻלָּם סַלּוֹנִים קוֹץ וָחֵדֶק

וּכְמוֹ לֹא כָל חַיָּתֵנוּ אַךְ צֶדֶק;

יֵשׁ בִּבְנֵי אָדָם כָּמֹנוּ אֻמְלָלוּ

מֵרָעַת הָאָדָם כַּקַּשׁ יִמָּלוּ

וּמִשְׁנֶה שִׁבָּרוֹן מֶנּוּ נִשְׁבָּרוּ

כִּי מִידֵי אֲחֵיהֶם הֵמָּה נִגְזָרוּ –

נַחְנוּ לִבְנֵי מִינֵנוּ לֹא נָרֵעַ

וּבְנֵי הָאָדָם – אִישׁ אָחיו בֹּלֵעַ

רַבּוֹת רָאִיתִי כָּל אֲשֶׁר עוֹלֵלוּ

בֵּן מְנַבֵּל אָב, אָבוֹת בָּנִים יֹאכֵלוּ


עַל חֲלָּשִׁים נְקִיִּים תַּקִּיפִים יִגְבָּרוּ

יִרְעֲצוּם וִירוֹצְצוּם דָּמָם יִשְׁכָּרוּ

וּמְשׁוּבָתָם רָעָה מִכָּל חַיְתוֹ טֶרֶף

כִּי לֹא רַק לִרְעָבָם לֹא לִמְצוֹא טֶרֶף

אֲנָשִׁים יִלְכֹּדוּ דָּמָם יִשְׁפֹּכוּ

כִּי אִם לִשְׂחֹק שַׁעֲשֻׁעִים חָזוֹן יַעֲרֹכוּ

בֵּין אָדָם וְחַיָּה מִלְחֶמֶת בֵּינַיִם

וַאֲחִיהֶם כִּי יִטָּרֵף יִמְחֲאוּ כַפַּיִם

מַשַּׁק זִרְמַת דָּמָיו לָמוֹ שַׁעֲשֻׁעַ

וּרְאוֹת יִתְעַנְּגוּ סְגוֹר לִבּוֹ קָרוּעַ.

וַאֲנִי זֶה דַרְכִּי עִם אָדָם כִּי גַרְתִּי

מִטְּרֹף אֹבְדִים אֵלֶּה כַּפִּי נָעַרְתִּי

וּמִקְּהַל הָרוֹאִים בִּהְיוֹתָם נְכוֹנִים

עַל דַּם רֵעָם לַעֲמוֹד שִׁחַתִּי הֲמוֹנִים,

וּפֹה כִּי הִסְּגִּירוּנִי – זֶה מְעַט בֹּאִי

גַּם הֲלֹם רָאִיתִי אַחֲרֵי רֹאִי

כִּי לַרְנִין שׁוֹבֵינוּ בְּעֹנֶג וָנַחַת

אֻסַּפְנוּ פֹה אֲסֵפָה אֶל בּוֹר שַׁחַת

וַאֲסִירֵנוּ זֶה הַמֻּשְׁלָךְ, זֶה הָאָדָם

אֻמְלָל הוּא כָּמֹנוּ, נִרְדָּף מִיָּדָם,

לָכֵן עֲצָתִי כִּי בוֹ לֹא נִגָּעָה

שֹׁרֶשׁ דָּבָר בּוֹ נִמְצָא עַד נִשְׁמָעָה.


וַיִּדֹּם הַזָּקֵן וּסְבִיבָיו נִשְׂעָרָה

לַכְּפִירִים וּלְבָאִים אָמְנָם הֵיטֵב חָרָה

מִשִּׁנֵּיהֶם הָשְׁלַךְ טֶרֶף עֵת יִרְעָבוּ

אַךְ לַמְרוֹת פִּי גִבּוֹרָם לֵב לֹא עָרָבוּ

קִפְּלוּ זַנְבוֹתָם וּפִיהֶם קָפָצוּ

שָׁבוּ עַל עָקֵב אֶל הַקִּיר נִלְחָצוּ.

וַיִּסֹּב הָאַרְיֵה וַיָּעֶף עַיִן

וּדְמָמָה שׂרֶרֶת פֹּצֶה פֶה אַיִן

וַיִּפֶן אֶל דָּנִיֵּאל וַיְנִיעֵהוּ

הֶחֱלִיק אֵלָיו בְּעֵינָיו וַיִּשְׁאָלֵהוּ:

– מִי אַתָּה, בֶּן אָדָם? קוּמָה הַגִּידָה,

אַךְ קשְׁט אִמְרֵי אֱמֶת – עֲדָתִי בְךָ אָעִידָה!

אָזְנַי מִלִּין תִּבְחַן, אֶצְרָפְךָ כַבֹּר –

מֶה עָשִׂיתָ כִּי שָׂמוּ אוֹתְךָ בַּבּוֹר?


דָּנִיֵּאל הִתְאוֹשֵׁשׁ וַתְּחִי בוֹ הָרוּחַ

וַיַּעַן אֲמָרָיו שָׁקֵט בָּטוּחַ:

– אֱמֶת מְשׂוֹשׂ דַּרְכִּי, קשֶׁט מִסְעָדִי

כִּי אֱמֶת יֶהְגֶּה חִכִּי זֶה הַבּוֹר שָׂהֲדִי

תֻּמָּתִי וּבֹרִי זֶה בוֹרִי יַעַן

אֵלָיו הָשְׁלַכְתִּי לֹא יַעַן וּבְיַעַן

עָוֶל פָּעַלְתִּי, חַטָּאתִי גָדֵלָה

כִּי אִם עַל אֲמִתִּי – מִפְּנֵי קשֶׁט סֶלָה!


– בֶּן אָדָם! – נָהַם הָאֲרִי בְּזַעַף וּרְתֵת –

הֲמָשֹׁל תִּמְשֹׁל? הֲתָחוּד? לֹא אוּכַל שְׂאֵת!

אַל בִּלְשׁוֹן סֵתֶר, לְשׁוֹן אָדָם הָאֲרוּרָה

מִי וָמָה אַתָּה סַפֵּר – וּבְשָׂפָה בְּרוּרָה!


– בֶּן חֹרִים אָנִי מִשּׁוֹעֵי יְרוּשָׁלַיִם

הָעִיר הַהֻלָּלָה בְּכָל פֶּה וּשְׂפָתַיִם

בִּגְאוֹן תִּפְאַרְתָּהּ וּסְגֻלַּת כָּל צֶבִי,

אַךְ שֹׁד מִשַּׁדַּי בָּא וָאֵלֵךְ שֶׁבִי

לִפְנֵי צָר אֲשֶׁר כּוֹנֵן לְהַשְׁחִית לְכָלָה

לַעֲרֹץ אַרְצִי וּלְמַשְּׁאוֹת הַמֶּמְשָׁלָה

וַיֶּאְסֹר אֶת מַלְכִּי בִּנְחֻשְׁתַּיִם

וּסְגָנִים כְּמוֹ חֹמֶר רָמַס בְּרַגְלַיִם

וַאֲנִי – נַעַר וָרַךְ בְּעֶצֶם תֻּמִּי וְגִילִי

לְקָחוּנִי עִם הַיְלָדִים אֲשֶׁר כְּגִילִי

מִן הַפַּרְתְּמִים וּמִזֶּרַע הַמְּלוּכָה

הַמֻּגְלִים עִמִּי בְּיוֹם מְבוּסָה וּמְבוּכָה

וּשְׁבוּיֵי חֶרֶב נִהֲגוּנוּ וַנֵּלֶךְ

וַנָּבֹא בָבֶלָה לַעֲמוֹד בְּהֵיכָל מֶלֶךְ.

וּלְמַלְכִּי זֶה שׁוֹבִי אַחֲרֵי נִשְׁבַּעְתִּי

בֶּאֱמוּנָה עֲבַדְתִּיו, לֹא רָשַׁעְתִּי,

וּלְעַמִּי כִּי הֵקִים צָרָה פַּעֲמָיִם

בַּעֲלוֹתוֹ כַשֹּׁאָה עַל יְרוּשָׁלָיִם


וּמַשְׁחִית גּוֹיִם עָבַר וְרָמַס וְטָרַף

הֶעֱבִיר הֶדֶר מַלְכוּת מִקְדַּשׁ אֵל שָׂרַף

וַיִּשְׁחַט וַיִּטְבַּח וַיַּהֲרֹג לְפִי חֶרֶב

הֲמוֹנִים הֲמוֹנִים בֹּקֶר וָעֶרֶב

וַיְרַצֵּץ וַיְנַתֵּץ וַיַּעַשׂ כָּלָה

מִירוּשָׁלַיִם עִיר עָז – מְעִי מַפָּלָה,

גַּם אָז בֶּאֱמוּנָתִי לְפָנָיו עָמַדְתִּי

וּלְהַשְׁלִים חֶפְצוֹ בְּתָם לֵבָב שָׁקַדְתִּי

כִּי כֵן בְּיַד נְבִיאוֹ אֱלֹהִים צִוָּנוּ

לִדְרשׁ אֶת שְׁלוֹם הָעִיר שָׁמָּה הֶגְלָנוּ

הִתְפַּלֵּל בַּעֲדָהּ, בְּעַד מַלְכָּהּ וַאֲשֶׁר אִתּוֹ

כִּי לִמְשֹׁל בָּנוּ הָאֱלֹהִים הִשְׁלִיטוֹ

וְנֶאֱמָן לְפָנָיו גַּם הַמֶּלֶךְ מְצָאָנִי

וּלְרֹאשׁ בִּמְדִינוֹת מַלְכוּתוֹ הִשְׁלִיטָנִי –

אִם עַל שִׁכְמִי, אִישׁ זָר, מִשְׂרָה זֹאת נִתָּנָה

מָה אוֹת כִּי רוּחִי לְמַלְכִּי נֶאֱמָנָה!

וָאֱעֶבוֹד אֶת מַלְכִּי בֶּאֱמֶת וָצֶדֶק

וָאֶשְׁמֹר אֶת אַרְצוֹ חַזֵּק כָּל בֶּדֶק

וּבִהְיוֹתִי תָמִים לְמֶלֶךְ וּלְאָרֶץ

בְּחוֹמַת דָּתִי לֹא פָרַצְתִּי פָּרֶץ,

דַּת אֱלֹהֵי הָעִבְרִים, אֱלֹהֵי אָבִי,

רוּחוֹ בְקִרְבִּי, אֱמוּנָתוֹ בִּלְבָבִי,


וּבְפַתְבַּג הַמֶּלֶךְ כִּי לֹא הִתְגָּאָלְתִּי

לֵאלֹהַי עֹשָׂי כִּי יוֹם יוֹם הִתְפַּלָּלְתִּי

כָּל אֵלֶּה לֹא הָיוּ לְמִפְגָּע לִי וְשָׂטָן

לִגְרֹעַ מִדְּבַר הַמֶּלֶךְ גָּדוֹל וְקָטָן.

וּמַלְכִּי זֶה, אָמְנָם, – כַּחֵד לֹא אוּכָלָה –

מֵהַבְלֵי שָׁוְא לֹא נִקָּה וַיָּבֵא בְאָלָה

בִּמְדִינוֹת מַלְכוּתוֹ שָׂרָיו וַעֲבָדָיו

הִשְׁתַּחֲוֹת לַצֶּלֶם עָשׂוּ יָדָיו

וּבַאֲשֶׁר אֲנִי יְהוּדִי לֹא הִשְׁתַּחֲוֵיתִי

אַךְ שָׁלַוְתִּי, מֵחֲמַת מֶלֶךְ נִקֵּיתִי

כִּי חָכַמְתִּי בְעֵינָיו מִשִּׁבְעָה מְשִׁיבֵי טַעַם

וַיְהִי לִי יִתְרִי סִתְרִי הֵחָבֵא מִזַּעַם.

וּשְׁלשֶׁת חֲבֵרַי – אֵל אֶחָד לָנוּ

לְמֶלֶךְ וּלְאַרְצוֹ כָּמֹנִי נֶאֱמָנוּ

וַיְשִׁימֵם לְמִשְׁמַעְתּוֹ לִשְׁלֹט בַּמְּדִינָה

וּבְיוֹם זַעַם הֵם לֹא מָצְאוּ חֲנִינָה

לֹא סָגְדוּ לַצֶּלֶם וּלְכִבְשָׁן הָשְׁלָכוּ

וֵאלֹהֵי יִשְׂרָאֵל, עָלָיו נִסְמָכוּ,

עָשָׂה פֶלֶא וּבְמוֹ אֵשׁ לֹא נִכָּווּ

וַיִּתְמְהוּ כָל הַגּוֹיִם לִרְאוֹת נִקָּווּ

וַיֶּחֱרַד הַמֶּלֶךְ וַיִּכָּנַע לְבָבוֹ

וַיֵּבוֹשׁ מִפִּסְלוֹ, אֱלִילֵי זְהָבוֹ,


וַיְצַו לְכָל עַמּוֹ לָתֵת כָּבוֹד בִּתְהִלָּה

לֵאלֹהֵי הָעִבְרִים נוֹרָא עֲלִילָה

תָּמִיד כָּל הַיּוֹם רֹמְמוֹת אֵל בִּגְרוֹנוֹ

אֵלָיו פִּיו יִקְרָא וְרוֹמָם תַּחַת לְשׁוֹנוֹ.

וַאֲנִי כִּי לֵאלֹהֵי אָמֵן הִנְנִי עָבֶד

בִּשְׂאֵת יֶתֶר נַעֲלֵיתִי וָאִכָּבֶד.

וּבְמוֹת זֶה מַלְכִּי וּבָנָיו מָלָכוּ

גַּם הֵם אוֹתִי אַחַר כָּבוֹד לָקָחוּ

וּכְמוֹ אֲבוֹתָם עָבַדְתִּי גַם אוֹתָם

וּבְאֵמוּן רוּחַ בַּעֲבוֹדָתִי אֵיתָם,

וּמִשְׁפַּט אֱלֹהַי כִּי לֹא עָזַבְתִּי

לֹא יוֹכִיחַ בִּי כִּי לְמַלְכִּי נִכְזַבְתִּי, –

עָם עָם אֶל אֱלֹהָיו יִדְרשׁ בִּשְׁמוֹ יֵלֶךְ,

וַאֲנִי תְפִלָּתִי לָאֵל מַעֲשַׂי לְמֶלֶךְ.

הֵן אֶרֶץ כַּשְׂדִּים מְקֻבֶּצֶת מֵעַמִּים

נִפְרְדוּ לִפְסִילֵיהֶם וּבְאַחַת הֵם תַּמִּים

בֶּאֱמוּנַת רוּחַ לַמֶּלֶךְ לָאָרֶץ

כְּאִישׁ אֶחָד חֲבֵרִים לַעֲמוֹד בַּפָּרֶץ,

גַּם בִּיהוּדִי לֹא יִתְּנוּ מִגְרַעַת

כִּי יַעֲבֹד אֱלֹהָיו, נַפְשׁוֹ יוֹדַעַת

אִם אַךְ לַמַּמְלָכָה נֶאֱמָן הִנֵּהוּ

לִנְצֹר אָרְחָהּ בְּאָשְׁרָה תָּלַן עֵינֵהוּ.


וּבְאֹרַח זוּ אֲהַלֵּךְ זָקַנְתִּי וְשַׂבְתִּי

עִם תֻּמִּי אֶגְוַע בִּלְבָבִי חָשַׁבְתִּי

וּמִי מִלֵּל לִי, הָהּ, כִּי בַעֲרוֹב יוֹמִי

פִּתְאֹם יָבֹא אֵיד וִישָׂעֲרֵנִי מִמְּקוֹמִי

לִסְגֹּר חַיָּתִי לְחַיָּה לֶאֱכוֹל בְּשָׂרִי

וּכְמוֹת נָבָל אָמוּתָה מוֹת אַכְזָרִי!

וּפִשְׁעִי מָה הוּא, מֶה עָוֶל פָּעָלְתִּי?

כִּי לֵאלֹהַי, כְּמוֹ הִסְכַּנְתִּי, הִתְפַּלָּלְתִּי!

הֵן זֶה דַרְכִּי תָמִיד מֵאָז כָּל הַיָּמִים

יוֹם יוֹם שֵׂאת רִנָּה וּתְפִלָּה שָׁלשׁ פְּעָמִים

וּתְפִלָּתִי עַד כֹּה גְּדוּלָתִי תֶרֶב

עַתָּה לַחֲטָאָה עֲוֹן אַשְׁמָה וָחֶרֶב

וַאֲנִי הוֹלֵךְ לָמוּת בְּנֶפֶשׁ מָרָה

לַמְתִּיק מַר הַמָּוֶת נֹחַם חָסָרָה:

לוּ רַק עַל תְּפִלָּתִי לָמוּת לֻקַּחְתִּי

כַּהוֹלֵךְ בֶּחָלִיל לִקְרָאתוֹ הָלַכְתִּי,

בְּעַד אֱלֹהַי קְדוֹשִׁי וּכְבוֹד תּרוֹתָיו

יָקָר לִי הַמָּוְתָה מָתְקוּ מְרֹרוֹתָיו

וּבְנַהֲמַת חַיּוֹת עַל בְּשָׂרִי תִּקְרַעְנָה

אָעִיר כְּבוֹדִי שְׂפָתַי תְּהִלָּה תַבַּעְנָה

כִּי בִי יָרוּם צוּרִי יִגְדַּל הֲדָרוֹ

לְהַקְדִּישֵׁהוּ בַעֲבָדָיו עֹשֵּׂי דְבָרוֹ,


אַךְ בְּמוֹתִי עַתָּה כְּבוֹד אֵל לֹא יַכִּירוּ

כִּי עַל חֵטְא הַתְּפִלָּה מֶרֶד יַחְבִּירוּ,

הֵן זֵדִים, קָמַי, אַךְ שׁוּחָה לִי כָרוּ

בְּקִנְאָתָם בִּי הַאֲשִׁימֵנִי חָתָרוּ

לְשֹׁאָה בִּקְשׁוּ נַפְשִׁי הַטּוֹת מִנִּי אֹרַח

לִמְצֹא נְתִיבָתָם בְּלִי עָמָל וָטֹרַח

וּתְפִלָּתִי אַךְ תֹּאֲנָה נִכְלֵי אָוֶן

לִמְצֹא עֲוֹנִי לִשְׂנוֹא יַדּוֹת בִּי אָבֶן

וְאֶל הָעָם יַגִּידוּ יוֹדִיעוּ פֵשֶׁר

כִּי לִמְרֹד חָשַׁבְתִּי, נִמְצָא בִי קֶשֶׁר,

וּמוֹתִי – מֹות נָבָל רָשָׁע בְּלִיַּעַל

הוֹרֵס חֻקֵּי אֶרֶץ בְּמֶרֶד וּבְמַעַל,

וּמַר הַמָּוֶת לָכֵן לִי עוֹד יֵמַר

כִּי גוֹי חוֹטֵא כָּל עַמִּי אוֹיֵב יֹאמַר.

– בֶּן אָדָם! – נָהַם הָאֲרִי בְּזָעַם –

אַךְ זָרוֹת וְתַהְפֻּכוֹת תְּדַבֵּר הַפָּעַם!

אֵיכָה נֶהְפַּךְ פִּתְאֹם לֵב עַם הָאָרֶץ

הִתְנַפֵּל עָלֶיךָ וּלְהָבִיא קָרֶץ?

עַם זוּ אֲהֵבְךָ מִנֹּעַר עַד שֵׂיבָה

מָה רָאָה עַל כָּכָה לַהֲדָפְךָ בְּאֵיבָה?

אִם פִּתְאֹם הִשְׁתּוֹבֵב מְשׁוּבָה נִצַּחַת

עַל לֹא חָמָס בְּכַפְּךָ נַפְשְׁךָ לָקַחַת


אֱהִי מַלְכְּךָ אֵפוֹא אֲשֶׁר אָמַרְתָּ

נֶאֱמָן לוֹ הִנְּךָ וּבְעֵינָיו יָקַרְתָּ?

אֵיךְ לָמוּת נְתָנְךָ בִּידֵי אַנְשֵׁי רֶשַׁע

אִם חַף אַתָּה, לֹא עָוֹן לָךְ, זַךְ בְּלִי פֶשַׁע?

אֵין זֹאת כִּי עִוַּלְתָּ גָּדוֹל עֲוֹנְךָ

וּבְתוֹךְ חִכְּךָ תִּמְנָעֶנּוּ תַּחַת לְשׁוֹנְךָ,

אַךְ סְגוֹר לִבְּךָ אֶקְרַע עַד אוֹצִיאֶנּוּ

וּבִמְתַלְעוֹתַי מִבִּטְנְךָ אוֹרִישֶׁנּוּ –

אִם יֵשׁ מִלִּין הֲשִׁיבֵנִי – וַאֲחַכֶּה

וְעָנְתָה בְךָ צִדְקָתְךָ – וַאֲנַקֶּה.

אֵימָתְךָ לֹא תְבַעֲתֵנִי לֹא נְפוּגוֹתִי

הֵן לָמוּת בָּאתִי לַמָּוֶת נְכוּנוֹתִי

לִבְשָׂרִי אִם תִּתְאָו בְּכַפְּךָ הִנֶּנִּי

פְּצֵה פִיךָ וֶאֱכוֹל שִׁנְּךָ בִי וְאֵינֶנִּי

וּלְמַעַן תִּצְדַּק פִּתְחוֹן פֶּה לְךָ לָמָּה

לִמְצֹא עָלַי תְּנוּאוֹת לְמַעַן אֶאְשָׁמָה?

הֵן עֹז וּגְבוּרָה לָךְ, יָדְךָ חֲזָקָה,

וְלֶאֱכוֹל עֲנִיֵּי אֶרֶץ לְךָ הַצְּדָקָה. –

אַךְ בִּדְמֵי נְקִיִּים אִם לֹא הִתְגָּאָלְתָּ

כִּי לֹא אָדָם הִנְּךָ – וּכְמוֹ הִתְהַלָּלְתָּ –

לִמְדַבֵּר תַּהְפֻּכוֹת אַל נָא תִתְּנֵנִי

כַּתַּר לִי זְעֵיר אֲחַוְּךָ וּשְׁמָעֵנִי.


לִבְלִי הַלְאוֹתְךָ סִפַּרְתִּי בִקְצָרָה

רַק שֹׁרֶשׁ דָּבָר מִכֹּל אֲשֶׁר קָרָה

וּפָרָשַׁת דָּבָר אִם תַּחְפֹּץ דַּעַת

הָבָה אֲסַפֵּר עַל אֹזֶן שֹׁמַעַת:

בֵּלְשַׁאצַר מֶלֶךְ כַּשְׂדִּים נִרְצַח בַּלָּיִל

כִּי הִתְקַשְּׁרוּ עָלָיו אַנְשֵׁי הֶחָיִל

וַתִּבָּקַע הָעִיר בָּא כַנָּהָר צָר

וְעַל מַלְכוּת כַּשְׂדִּים הָמְלַךְ אִישׁ זָר

דָּרְיָוֶשׁ הַמָּדִי זֶה מַלְכִּי עָתָּה

אֶפֶס אֶרֶץ כַּשְׂדִּים טֶרֶם שָׁקָטָה

עוֹד נִכְּרוּ בַחֻצוֹת מֶרֶד וָמַעַל

דַּם הַמֶּלֶךְ הַמּוּמָת עוֹד קֶצֶף יַעַל

וִיגֹעֲשׁוּ עָם יִתְגּוֹדְדוּ בַסֵּתֶר

וְעַל רֹאשׁ הַמָּשׁוּחַ יָנַע הַכֶּתֶר

וְהוּא אִישׁ נָכְרִי עַתָּה יַעֲשֶׂה מְלוּכָה

עַל גּוֹי מַר וְנִמְהָר בִּמְהוּמָה וּמְבוּכָה

וַיָּשֶׂם הַמֶּלֶךְ נְצִיבִים בָּאָרֶץ

לִבְצֹר רוּחַ מַשְׁחִית לְהַשְׁקִיט מִפָּרֶץ

מֵאָה וְעֶשְׂרִים שַׁלִּיטִים – הֵם הָאֲחַשְׁדַּרְפְּנִים

לְכָל סֵבֶל הַמַּמְלָכָה מִחוּץ וּמִפְּנִים

וּגְבוֹהִים עֲלֵיהֶם שְׁלשֶׁת שָׂרֵי שָׂרִים

הֵמָּה לַכֹּל לְרֹאשׁ וּלְמִשְׁמַעְתָּם סָרִים

אֲחַשְׁדַּרְפְּנֵי הַמֶּלֶךְ הָרוֹדִים עַל יָדָם

לָשִׂים מִשְׁטָרָם וּלְבַקֵּר מַעְבָּדָם

וְהִקִּיפוּ עַל הַמֶּלֶךְ לְבַל יִקְרַב נֶגַע

וִיהִי שָׁקֵט וְשַׁאֲנָן מִנֵּזֶק וָפֶגַע.

וּלְפָנָיו נֶאֱמָן הַמֶּלֶךְ מְצָאָנִי

וּבִשְׁלשָׁה הָרָאשִׁים נִמְנֵיתִי גַם אָנִי,

כִּי שָׁמַע שִׁמְעִי מַעֲשַׂי וּמַעֲמָדִי

כִּי רוּחַ אֱלֹהִים עָלַי וְשַׁדַּי עִמָּדִי

לַמֶּלֶךְ לְפָנָיו מְרִי קִצּוֹ הִשְׁמַעְתִּי

הָאֹתִיּוֹת לְאָחוֹר בְּלִי חָת הוֹדַעְתִּי

כִּי מְאָסוֹ אֱלֹהִים וּמַלְכוּתוֹ הָרָס

לַטּוֹב מִמֶּנּוּ נְתָנָהּ – לְמָדַי וּפָרָס

וּדְבָרִי הֵן בָּא נְבוּאָתִי נֶאֱמָנָה

וּמַלְכוּת בָּבֶל לְדָרְיָוֶשׁ נִתָּנָה

וּבִי לָכֵן יִרְאֶה צִיר מִגָּבוֹהַּ

לְכוֹנֵן מַלְכוּתוֹ בְּמַלְאֲכוּת אֱלוֹהַּ,

כִּי נֶאֱמָן לוֹ אֶהְיֶה לִבּוֹ בֹטֵחַ

כְּמוֹ לָאֵל עַל יָדִי דְּבָרִי שׁוֹלֵחַ.

וָאֶקַּח פְּקֻדָּתִי לָהּ הִתְמַכַּרְתִּי

וּכְאִישׁוֹן עֵינִי אֶת מַלְכִּי נָצַרְתִּי

וָאָעֶר כֹּחִי וּלְבָבִי יוֹם וָלָיִל

וּבְדַעַת וּבְכִשְׁרוֹן עָשִׂיתִי חָיִל

בְּצֶדֶק וּבְמִשְׁפָּט הֶעֱמַדְתִּי אָרֶץ

רִפֵּאתִי שְׁבָרֶיהָ גָּדַרְתִּי פָרֶץ.

וַיַּרְא הַמֶּלֶךְ כִּי דְרָכַי הִשְׂכַּלְתִּי

וּמִכָּל הַשָׂרִים מַעֲשַׂי הִגְדַּלְתּי

וַיּוֹסֶף כַּבְּדֵנִי בִּיקָר וּגְדֻלָּה

הַשְׁלִיטֵנִי אָמַר בְּמַמְלַכְתּוֹ כֻלָּהּ

נַשְּׂאֵנִי לַכֹּל לְרֹאשׁ בִּשְׂאֵת יֶתֶר

וִיחַסְּרֵנִי מִמֶּנּוּ רַק הַכֶּתֶר.


מֶלֶךְ זָקֵן וּכְסִיל! – הָאַרְיֵה נָהַם –

אַךְ הֶבֶל בְּנֵי אָדָם קוֹץ מֻנָּד כֻּלָּהַם!

כִּי אֵיכָכָה נוֹאַל לִפְרֹשׂ אִוֶּלֶת

לִפְקֹד עַל אַרְצוֹ וּבְעֹז מֶמְשֶׁלֶת

אִישׁ מֵעַם זָר אֲשֶׁר לֹא מִבְּנֵי מִינוֹ

לִבְטֹחַ בּוֹ וּלְהוֹשִׁיבוֹ לִימִינוֹ?

מֶלֶךְ גַּם אָנִי, לִי כָל חַיְתוֹ יָעַר

מִפְּנֵי שְׁמִי יֵחַתּוּ יֹאחֲזוּ שָׂעַר.

גַּם לִי מִשְׁנִים, שָׁלִישִׁים, שָׂרֵי מִסִּים

כֻּלָּם גִּבּוֹרִים בַּמַּסָּה נִסִּיתִים

נָמֵר שׁוֹקֵד דֹּב שֹׁקֵק וּזְאֵב עֶרֶב

עַד בֹּקֶר לִי הֵבִיאוּ וּשְׁלַל עֶרֶב.

אִם אָמְנָם יָדַעְתִּי כִּי מָלְאוּ כַחַשׁ

פִּיהֶם שָׁוְא רַגְלָם עַל מִרְמָה תַחַשׁ

כִּי כָל נֵתַח טוֹב בְּחֹרֵיהֶם יַטְמִינוּ

מִמֵּיטַב הַצֹּאן וּמֵחֶלְבֵהֶן יַשְׁמִינוּ

יָרֵךְ וְכָתֵף יֹאכְלוּ זְרֹעַ וּלְחָיַיִם

וּתְרוּמָתָם לִי בְּדַל אֹזֶן וּכְרָעַיִם,

מַעְלָם וּזְדוֹנָם יָדַעְתִּי לֹא בְסֵתֶר

וְלֹא אַחַת יִסַּרְתִּים שִׁלַּמְתִּי עַל יֶתֶר,

אַךְ טוֹבִים לִי הֵמָּה וּלְפָנַי עָמָדוּ

אִם כָּל אַשְׁמוֹתָם מֶנִּי לֹא נִכְחָדוּ,

כִּי בְנֵי יַעְרִי הֵמָּה, בַּעֲלֵי מִשְׁפַּחְתִּי

וּמוּמָם בָּם נִגְעִי הוּא, מִסְפַּחְתִּי,

וּכְבוֹדָם לֹא אֶתֵּן לַאֲחֵרִים, לְזָרִים,

לוּ גַם מִמִּשְׁפַּחַת רָם, מִן הַנְּשָׁרִים.


אִם זַךְ וְאִם יָשָׁר פָּעָלְךָ, אֲרִיאֵל!

לֹא לְאָדָם הַמִּשְׁפָּט – עָנָה דָנִיֵּאל –

אָמְנָם צָדַקְתָּ אִם אַתָּה לְמִינֶךָ

תִּבְחַר עֹשֵׂי דְבָרְךָ לַעֲמֹד לִימִינֶךָ;

לִטְרֹף וּלְחַנֵּק לִנְשֹׁךְ וּלְשַׁסֵּעַ

אַךְ בֶּן מִינְךָ מִמִּשְׁפַּחְתְּךָ יוֹדֵעַ

חֲרָבוֹת שִׁנָּיו מַאֲכָלוֹת מְתַלְעוֹתָיו

וּמַה יִּתְרוֹן לַזָּר עִם כָּל מַעֲלוֹתָיו?

לֹא כֵן הָאָדָם מַלְכוּתוֹ בָאָרֶץ

לִבְנוֹת לִנְטֹעַ לִהְיוֹת גּוֹדֵר פָּרֶץ

וּדְרוּשִׁים לְחֶפְצוֹ סָרִים אֶל מִשְׁמַעַת

שָׂרִים נֱאֱמָנִים בְּמוֹעֵצוֹת וָדַעַת

וְאֶל זֶה יַבִּיט מֶלֶךְ חָכָם מֶלֶךְ נָבָר

אֶל נֶאֱמַן רוּחַ וּמַשְׂכִּיל עַל דָּבָר

וִיחַפְּשֵׂהוּ כַמַּטְמוֹן בַּאֲשֶׁר יִמְצָאֶנּוּ

וּלְמִשְׁפַּחְתּוֹ אִם מִתְיַחֵשׂ אֵינֶנּוּ. –

וּמַלְכִּי בִי הֶאֱמִין וְעֵינוֹ צֹפִיָּה

כִּי אַפְלִיא עֵצָה אַף אַגְדִּיל תּוּשִׁיָּה

אָבָה הַשְׁלִיטֵנִי הָרִים מַעֲמָדִי

לִהְיוֹת שַׂר שָׂרִים וְכָל אֲשֶׁר לוֹ בְּיָדִי

וִיהוּדִי כִי אָנִי זֹאת תַּמְרִיצֵהוּ

כִּי שָׁפְרָה עֲצָתוֹ בִי לִמְצֹא חֶפְצֵהוּ

לִנְטֹעַ אָהֳלֵי אַפַּדְנוֹ לַעֲקֹר שָׁיִת

וּלְהָנִיחַ מִסָּבִיב מִחוּץ מִבָּיִת,

זֶה אַדִּיר חֶפְצוֹ וּמִי אִישׁ כִּלְבָבוֹ

עָלָיו יִשָּׁעֵן, יַשְׁלִיךְ עָלָיו יְהָבוֹ?

בַּכַּשְׂדִּי לֹא יַאֲמִין, כִּי מִי יֶעֶרְבֶנּוּ

אִם תָּמִים לִבּוֹ בָּרֹגְנִים אֵינֶנּוּ?

וּבְנֵי מָדַי, עַם קְרוֹבוֹ, מִקָּרוֹב בָּאוּ

הָאָרֶץ וַאֲנָשֶׁיהָ לֹא יֵדָעוּ,

לָכֵן רַק הָאִישׁ מִזֶּרַע הַיְּהוּדִים

יַעֲשֶׂה חֶפְצוֹ בְּבָבֶל וּזְרֹעוֹ כַּשְׂדִּים. –

הַיְּהוּדִי הַגֹּלֶה מֵאֶרֶץ מוֹלֶדֶת

וּבְתוֹכָהּ רוּחוֹ, לָהּ נַפְשׁוֹ נִצְמֶדֶת,

הוּא בְאֶרֶץ נוֹדוֹ יְבַקֵּש אַךְ מָנוֹחַ

לְהַשְׁקִיט לוֹ מִימֵי רָע, רִישׁוֹ לִשְׁכֹּחַ.

מֶה הֹוֶה מַה שֹּׁוֶה לַגֵּר לָאֹרֵחַ

עִם שׁוֹנִים הִתְעָרֵב אֶרֶץ הָפֵחַ?

אַךְ שֶׁקֶט מַאֲוַיָּיו, שַׁלְוָה מַחְשְׁבוֹתָיו

כִּי יַנִּיחוּ לוֹ, אִישׁ אַל יָנַע עַצְמוֹתָיו,

וּבְאֶרֶץ שָׁלוֹם בְּעֹז מֶלֶךְ עֲרוּכָה

רַק בָּהּ, יֵדַע לֵב יְהוּדִי, יִרְאֶה מְנוּחָה,

וּפְרָעוֹת בִּמְדִינָה מוֹקְשֵׁי עָם וּזְוָעָה

לוֹ מָקוֹר נִפְתָּח אֶל כָּל צָרָה וְרָעָה

בַּעֲדַת מְרֵעִים כִּי נִצָּב בְּקֶרֶב הֲמוֹנָהּ

יַד הָעֵדָה תִּהְיֶה בּוֹ בָרִאשׁוֹנָה,

לָכֵן אֶל מַלְכּוֹ מְאַסְּפוֹ נֶאֱמָן הוּא תָמִים

שְׁלוֹמוֹ וּשְׁלוֹם אַרְצוֹ יִדְרשׁ כָּל הַיָּמִים

בְּמָעֻזּוֹ יַחֲזִיק וּבְעָצְמוֹ שָׂמֵחַ

וִישֻׁגַּב אָז גַּם הוּא שֹׁקֵט בֹּטֵחַ,

זֹאת הֵשִׁיב אֶל לִבּוֹ מַלְכִּי, רַב תְּבוּנוֹת,

וּבְמָצְאוֹ אוֹתִי לְפָנָיו אִישׁ אֱמוּנוֹת

וַאֲנִי לֹא תְמוֹל בֹּאִי, בָּאָרֶץ נוֹשַׁנְתִּי

מוֹסְדֶיהָ מְכוֹנֶיהָ תַּרְתִּי בָחַנְתִּי


וּתְכוּנַת ישְׁבֶיהָ מַעֲלוֹת כָּל רוּחַ

יָדַעְתִּי כֻלָּה, חֲרוּשָׁה עַל לוּחַ,

הִנֵּה לִי יָאֲתָה לִהְיוֹת מִשְׁנֵהוּ

וּבִי יָפֵק זְמָמוֹ פֹּעַל לַמַּעֲנֵהוּ.


בְּהִשָּׁמַע דְּבַר הַמֶּלֶךְ לַעֲשׂוֹת יָזָם

רָעֲמוּ פְנֵי הַשָּׂרִים רַעַד אֲחָזָם,

הָאֲחַשְׁדַּרְפְּנִים וַחֲבֵרַי הַשְּׁנַיִם

בַּחֲמַת רוּחַ קִנְאָה חָרְקוּ שִׁנַּיִם

וַיַּעֲרִימוּ סוֹד הִתְיָעֲצוּ הִתְמַרְמָרוּ

וַיְהוֹתְתוּ עָלַי וּבְכָל עֹז חָתָרוּ

וַיְחַבְּלוּ תַחְבֻּלוֹת וַיִּתְנַכָּלוּ

עָלַי לִמְצֹא תְנוּאוֹת אוּלַי יוּכָלוּ,

לְמַעַן הַבְאִישֵׁנִי מִשְּׂאֵתִי לְהַדִּיחַ

בִּקְּרוּ כָל מַעֲשַׂי כָּל שִׂיג כַּל שִׂיחַ,

כָּל פֹּעַל כָּל מִקְרֶה הָעֵת קָבָרָה

הֶעֱלוּ מִנְּשִׁיָּה הֵנִיעוּ בַכְּבָרָה,

אַךְ לֹא דָבַק מְאוּם וּמוּם לֹא מָצָאוּ

לָשִׂים עֲלִילוֹת דְּבָרִים לָרִיק יָגָעוּ

בְּמַעֲשַׂי וִיגִיעִי אֵין מִדְחֶה וָסֶלֶף

שַׁכֻּלָּה אֵין בָּמוֹ, בְּאֶחָד מֵאֶלֶף,

וּלְחַפֵּשׂ כִּי הִרְבּוּ לִמְצֹא עֲוֹנוֹת

מִהֲרוּ גַם לִסְגֹּר סֵפֶר הַזִּכְרֹנוֹת

פֶּן יָשׁוּב עֲמָלָם יֵרֵד עַל קָדְקֳדָם

לַחְשׂף מַצְפּוּנֵיהֶם כָּל חֲמַס יָדָם

אוֹ לִרְאוֹת מַעְרָם פֶּן תִּפָּקַח עַיִן

לִרְאוֹת פָּעֳלָם מֵאָפַע נֶגְדִּי כְאַיִן.

וַיִּגְדַּל קִצְפָּם, הֵיטֵב לָמוֹ חָרָה

עַל כִּי נִקֵּיתִי לֹא עִקַּשְׁתִּי הַיְשָׁרָה

עַל כִּי חָכַמְתִּי וְלִי עֵצָה וְתוּשִׁיָּה

וַאֲנִי מַפְלִיא לַעֲשׂוֹת וְרַב הָעֲלִילִיָּה

אֵלֶּה עָשָׁן בְּאַפָּם רָקָב בְּעַצְמוֹתָם

כִּי יִבָּצֵר מֵהֶם לַחְמֹס מְזִמּוֹתָם –

עַד נוֹכְחוּ לָדַעַת הִתְבּוֹנְנוּ בִינָה

כִּי שָׁוְא יְחַפְּשׂוּ עֹלוֹת בְּמַעֲשַׂי לַמְּדִינָה

וַיִּגְעָשׁוּ וַיִּתְחוֹלְלוּ כַסַּעַר

וַיֵּלְכוּ קַחַת תּוֹרָה מֵחַיְתוֹ יַעַר

הִיא תוֹרָתְךָ, אֲרִיאֵל! לִי בֵאַרְתָּ

עַל פִּיהָ מִיַּעַרְךָ זָרִים בִּעַרְתָּ.

הֵן גַּם בֶּן הָאָדָם לְעִתִּים מְזֻמָּנִים

יִלְמַד דַּרְכְּךָ פִּקּוּדֶיךָ לוֹ נֶאֱמָנִים,

מִשְׁפַּט חַיְתוֹ יַעַר גַּם לְאָדָם יִיקָר

אִם לִמְצֹא בוֹ חֵפֶץ מִקְרֵהוּ יִקָּר.

בָּאָדָם נֶפֶשׁ חַיָּה, בְּקִרְבּוֹ מְסֻתֶּרֶת

וּלְעֵת מְצוֹא בִּמְשׁוּבָה מִתְעוֹרֶרֶת

אָז יִהְיֶה הָאָדָם גַּם לַיִשׁ גַּם לָבִיא

וְעָבַר וְרָמַס וְטָרַף אֶל חֹרָיו לָבִיא

וּפְרִיץ חַיּוֹת זֶה הוֹלֵךְ עַל שְׁתַּיִם

לוֹ אֵין מְתַלְעוֹת, פַּרְסוֹת אַבִּירִים, קַרְנַיִם,

וּבְכֹחוֹ יַגְדִּיל אַךְ רוּחוֹ כִי יַעַר

גַּם מִמְּךָ, אֲרִיאֵל! וְכָל חַיְתוֹ יַעַר,

כֹּחוֹ בְמֹחוֹ וּמְזִמּוֹתָיו צִפָּרְנָיו

יִשְׂתָּעֵר – וְכָל חַיּוֹת לֹא יַעַמְדוּ לְפָנָיו,

וְלֹא בְמַחְשָׁךּ מַעֲשֵׂהוּ, בְּסִבְכֵי הַיַּעַר,

כִּי אִם בְּרֹאשׁ הֹמִיּוֹת, יוֹמָם, בַּשַּׁעַר!

יָדַעְתִּי, יָדַעְתִּי, הָאַרְיֵה קָרָא –

אַךְ סַפֵּר הוֹסִיפָה מָה אוֹתְךָ קָרָה?

בִּרְאוֹת הַשָּׂרִים כִּי מִצַּד הַמְּלוּכָה

אֵל חִנָּם לִי צָדוּ לִכְרוֹת לִי שׁוּחָה

הִתְהַפְּכוּ מְסִבּוֹת הָסֵב כְּלֵי הַמִּלְחָמָה

אֶל מְקוֹם פְּלֹנִי אַלְמוֹנִי, שָׁם יָדָם רָמָה:

קִנְאַת עָם עוֹרְרוּ קִנְאַת דָּת וֶאֱמוּנָה

הִיא תְעַוֵּר פִּקְחִים וּתְסַלֵף תְּבוּנָה.


מַה קָּדוֹשׁ מִדָּת, מַה מֶּנָּה גָּבוֹהַּ?

מָה רָם מַה נִּשְׂגָּב לֶאֱנוֹשׁ מֵאֱלוֹהַּ?

וּבְנֵי הָאָדָם מַה שָּׁפְלוּ נֶאֱלָחוּ

כִּי דָת וֵאלֹהִים אֶל חֶפְצָם יִקָּחוּ

אֶל חֵפֶץ תַּרְמִיתָם וּמַחְשְׁבוֹת אוֹנָם

לִכְלֵי חָמָס אֶל רָעָתָם וּזְדוֹנָם!

יִרְאוּ כִי אָזְלַת יָד מַעֲשֵׂיהֶם בַּצֵּעַ

כָּלוּ הַחִצִּים הַכְרִיעַ בָּם רֵעַ

אֶל מִקְדָּשָׁם יָבֹאוּ בְּחֹזֶק מֵצַח

וּבְבִגְדֵי הַשְּׂרָד יָלִיטוּ כְלֵי רֶצַח

וַחֲמֻשִׁים נֶשֶׁק קֹדֶשׁ יֵצְאוּ בְנֵי נָבָל,

מַפֵּץ לָמוֹ הַדָּת אֱלֹהִים מְחִי קָבָל,

הַקֹּדֶשׁ מָלֵא חָמָס שֹׁד עֲנִיִּים

וְהָאֱמוּנָה עֲקֻבָּה מִדְּמֵי נְקִיִּים!


גַּם שׂטְנַי הָרָעִים אוֹן לָמוֹ מָצָאוּ

זָכְרוּ אֱמוּנָתִי וּלְפָעֳלָם יָצָאוּ

לְקוֹמֵם לְאוֹיֵב לִי אֶת עַם הָאָרֶץ

לְהוֹכִיח לוֹ וּלְהַרְאוֹת כִּי בָא קָרֶץ

וְאִם יִתְמַהְמֵהַּ בֹּא יָבֹא לֹא יְאַחֵר

אִם יָבֹא הַיְּהוּדִי הַזָּר בֶּן עַם אַחֵר

הִשְׂתָּרֵר בִּמְדִינָה בְּגַאֲוָתוֹ וּגְאוֹנוֹ

וּמָשַׁל מִמְשָׁל רָב וְעָשָׂה כִרְצוֹנוֹ –:

הַיְּהוּדִי הַגּוֹלֶה אַךְ מָצָא מְנוּחָה

מִתְעָרֶה כאֶזְרָח וְעוֹד לוֹ אַךְ הַמְּלוּכָה!

הַיְּהוּדִי בָא לָגוּר וַיִּשְׁפֹּט שָׁפוֹט

קָרְחִי וָגֹזִּי, שֵׁשַׁךְ, הָמֵר וְסָפוֹד!


תִּפְאֶרֶת גְּאוֹן כַּשְׂדִּים צִיצַת נוֹבֵל

נָבְלָה הָאָרֶץ, אֻמְלְלָה נָבְלָה תֵבֵל!

עַרְעֵר תִּתְעַרְעַר חוֹמַת בָּבֶל הָרְחָבָה

וּבְבֶטֶן הַיְּהוּדִי לָהּ קֶבֶר חָצָבָה!

תַּחַת עֶבֶד כִּי יִמְלֹךְ יִמְשֹׁל בְּשָׂרִים

בְּמִכְמַנֵּי הָרוּחַ וּבְחַיֵּי בְשָׂרִים

הֲלֹא תִרְגַּז אֶרֶץ יָשֹׁמּוּ יְשָׁרִים

הֲבָנִים אֵין לְבָבֶל כִּי נָשִׂיר זָרִים?

הֵן בָּשְׁנָה נִקַּח וְתוֹכַחְתֵּנוּ לַבְּקָרִים

נִמָּשֵׁל כַּבְּהֵמוֹת נִדְמֵה לַבְּקָרִים

כֵּן מְשַׁל קַדְמוֹנִי שָׁתָה לַעַג לָרְוָיָה:

“הֵן אֶרֶץ כַּשְׂדִּים זֶה הָעָם לֹא הָיָה!”

וּבִמְשֹׁל הַיְּהוּדִי וְעָמַד עַל כַּנּוֹ

עַל כֹּל יִתְגַּדֵּל – הֵן נוֹקֵם הוּא וְקַנּוֹא!

עַל גָּלוּתוֹ בָנוּ נִקְמָתוֹ יָנִיחַ

וּלְרַגְלָיו הֲדֹם רָאשֵׁינוּ יַנִּיחַ,

וּבְקִנְאָתוֹ לֵאלֹהָיו יָבִיא קָרֶץ

עַל מַחְמַד עֵינֵינוּ, אֱלֹהֵי הָאָרֶץ.

כָּרַע בֵּל, קָרַס נְבוֹ, יֻכַּתּוּ שְׁאִיָּה

מִגְּאוֹן הַכְּרוּב – הַחַיָּה הַיְּהֻדִיָּה!

עֲשׂוּ פְלִילָה, הָבוּ עֵצָה וְתַקָּנָה

כִּי אָבְדָה הָאֱמוּנָה הַדָּת בְּסַכָּנָה!…

כֵּן הָמוּ רָעָשׁוּ בְּקוֹל חֹצֵב לֶהָבוֹת

אַךְ לֹא עָצְרוּ כֹחַ לִתְפֹּשׂ לְבָבוֹת,

כִּי קִנְאַת דָּת וֶאֱמוּנָה לָעָם זָרָה

וּלְצוֹדֵד בָּהּ נְפָשׁוֹת יָדָם קָצָרָה.

גַּם מִגָּבוֹהַּ מִלִּים מֵהֶם הֶעְתִּיקוּ

לִבְלִי לֵב הָעָם מִזָּרִים יַרְחִיקוּ,

הֵן עַתָּה אֶרֶץ וּמְלֹאָהּ בְּיַד זָרִים:

מִבְּנֵי מָדַי הֵם הַמֶּלֶךְ וְהַשָּׂרִים

וּמְגַמַּת פְּנֵי הַמֶּלֶךְ וּתְשׁוּקָתוֹ

לִשְׁפֹּךְ עַל בָּבֶל רוּחַ אֶרֶץ מְכֹרָתוֹ,

עַל כָּל מַחֲמַדֶּיהָ לָכֵן יָדוֹ פָרָשׂ

לְתַכְּנֵם לְעָרְכֵם בְּדָתֵי מָדַי וּפָרָס

הַס, כִּי לֹא לְהַזְכִּיר בְּשֵׁם זָרִים אוֹ גֵרִים

מֶלֶךְ אֶחָד לַכֹּל – כָּל עַמּוֹ חֲבֵרִים! –

רָאוּ צֹרְרַי כִּי עֲצָתָם הוּפָרָה

וַיְחַבְּלוּ תַחְבֻּלָה וַיַּעֲמִיקוּ סָרָה:

כִּתְבֵי שִׂטְנָה עָרְכוּ חֶרֶשׁ הֵפִיצוּ

בְּיָד חָרוּצִים לְכָל בְּנֵי הָעָם הֵרִיצוּ

כָּתְבוּ מְרֹרוֹת בִּמְזִמּוֹת וְחִשְׁבֹנוֹת

הֹרוֹ וְהֹגוֹ מִלֵּב דִּבְרֵי עֲוֹנוֹת

כִּי הַיְּהוּדִים הַמְעַטִּים פָּרְצוּ פָרֶץ

וִילַחֵךְ הַקָּהָל כָּל טוּב הָאָרֶץ,


יַד הַיְּהוּדִים בַּכֹּל, אֵין עוֹד מִלְּבַדָּם

וִינַצְּלוּ אֶת הָעָם יַחֲרִיבוּ לְשַׁדָּם

כִּי הַיְּהוּדִים כֻּלָּם חֲבֵרִים יַחַד

וְלָתֵת חִתִּיתָם עִמָּם הַמְשֵׁל וָפַחַד. –:

– שִׁמְעוּ עַמִּים כֻּלָּם וּלְאֻמִּים הַאֲזִינוּ

עַל מָה אָבְדָה הָאָרֶץ הֲלֹא תָבִינוּ

בַּכְּפָרִים אִכָּרִים רָשׁוּ רָעֵבוּ

בַּקְּרִיּוֹת הַחֲנֻיּוֹת שָׁמְמוּ חָרֵבוּ

הַקּוֹנֶה לֹא יִשְׂמַח הַמּוֹכֵר יִתְאַבָּל

וְאָנוּ וְאָבְלוּ: כָּשַׁל כֹּחַ הַסַּבָּל!

וְסָבְבוּ בַשּׁוּק סוֹפְדִים מַחֲזִיקֵי בַפֶּלֶךְ

שְׂכַר הָאָדָם לֹא נִהְיָה בָעִיר בַּפֶּלֶךְ

וְהַמִּשְׂתַּכֵּר מִשְׂתַּכֵּר אֶל צְרוֹר נָקוּב, –

הֲלֹא עַל הַיְּהוּדִי רֵאשִׁית הַגּוֹיִם נָקֻב

כִּי רֵאשִׁית תְּבוּאָתָם לוֹ רֵאשִׁית הַחֵרֶם

וַיַּבְעֵר מִגָּדִישׁ וְעַד קָמָה וְעַד כֶּרֶם,

לִבּוֹ תְּהוֹם רַבָּה לוֹ מַחְשְׁבוֹת חָרוּץ

כֶּעָפָר יִצְבֹּר כֶּסֶף וּכְטִיט חָרוּץ,

דִּמְיֹנוֹ כְאַרְיֵה יָגִיחַ מִן הַמַּאֲרָב

וְנָפַל בַּעֲצֻמָיו כָּל הַמִּפְעָל כָּל הַמַּעֲרָב.

אַיֵּה גְבוּרָתֵךְ, כַּשְׂדִּים! אַיֵּה קִנְאָתֵךְ?

עִזְבוֹנֵךְ נֶעְזָב נֶאֶסְפָה כִּנְעָתֵךְ. –


אִי לָךְ – אֶרֶץ שִׁנְעָר! אָבַדְתָּ, עַם מְרֹדָךְ!

מִי יָנוּד לִמְרוּדָךְ, מִי יִפְדְּךָ מֵרֹדָךְ?

עָלָה הַכּוֹרֵת, מֵפִיץ עַל פָּנֶיךָ,

כִּי אִישׁ יְהוּדִי יִהְיֶה רֹאשׁ משְׁלֶיךָ

וּלְאֶחָיו הוּא סֵתֶר וְאֵין מִיָּדוֹ מַצִּיל

כִּי גַם עַל הַמֶּלֶךְ מֵרוּחוֹ מַאֲצִיל

עַתָּה מִיהוּדִים לֹא תִהְיֶה לָנוּ תְּקוּמָה

אֶת בָּבֶל הַשְּׁדוּדָה יַפְשִׁיטוּ עֲרֻמָּה

לִפְשֹׁט עוֹרֵנוּ וּלְגָרֵם עַצְמוֹתֵינוּ

כּוֹס זָהָב לָמוֹ בָּבֶל לִשְׁתּוֹת דָּמֵינוּ. –

הָבוּ, אַחִים! מָרֵי נֶפֶשׁ וּקְשֵׁי יוֹם כֻּלְכֶם

תּוֹשַׁע לָכֶם יְמִינְכֶם לִשְׁבֹּר זֶה עֻלְּכֶם!

אִישׁ רֵעֵהוּ עִזְרוּ וְדִבֶּר חַד אֶת אַחַד

וִהְיוּ לַאֲגֻדָּה לָצֵאת חוֹצֵץ יַחַד

לְקַצּוֹת בַּיְּהוּדִים וּרְכוּשָׁם תָּשֹׁסּוּ

קַדְּשׁוּ יוֹם קִרְאוּ מוֹעֵד עֲלֵיהֶם תָּכֹסּוּ

בֹּזוּ כֶסֶף בֹּזּוּ זָהָב וְהָיָה שִׁלֻּמִים

לַעֲמַלְכֶם עָשְׁקוּ הַיְּהוּדִים הָאֲיֻמִּים

הֵן מֵאֲשֶׁר לָכֶם עָשׂוּ כָל הָעשֶׁר

וְשַׁבְתֶּם וּלְקַחְתֶּם בִּצְדָקָה וּבְישֶׁר.

וּלְאַנְשֵׁי הַצָּבָא לֵב אַל תָּשִׂימוּ

כִּי לִנְגֹּעַ בָּכֶם יָד לֹא יָרִימוּ


כִּי אַתֶּם עָם – לֹא יִלָּחֲמוּ עִם רַבִּים

וּלְמַרְאֵה עַיִן רַק יָסֹבּוּ רַבִּים

וּבְמַעֲשֵׂיכֶם אֵלֶּה נִפְלָאוֹת תְּחוֹלֵלוּ

כִּי אֶת צוֹרְרֵינוּ מֵאֶרֶץ יְנַשֵּׁלוּ,

הֵן מַלְכֵּנוּ עֵינָיו יַבִּיטוּ לְנֹכַח

וּפֹעַל פָּעַל הָעָם יִרְאֶה וְנוֹכַח

כִּי נִתְּכָה הַחֵמָה עַל בְּנֵי הַגּוֹלָה

בִּגְלָלָם יְגֹעֲשׁוּ עָם וְהַמְּהוּמָה גְּדוֹלָה

וְהָעָם כַּיָּם נִגְרָשׁ הַשְׁקֵט לֹא יוּכַל

עוֹד מְעַט וְנָשָׂא נֵס, קֶשֶׁר לִקְרוֹא יוּחַל

וּמַלְכֵּנוּ חָכָם הָעִתִּים יוֹדֵעַ

וּלְהַקְדִּים הָרָעָה מֵהִשְׂתָּרֵעַ

לְכַפֵּר פְּנֵי הָעָם יָחִישׁ גָּרֵשׁ כָּלָה

עַם חֶרְמָם, הַיְּהוּדִים, מִגְּבוּל הַמֶּמְשָׁלָה

וְעָזְבוּ חֵילָם וְהִתְנַחַלְתֶּם וּשְׂבַעְתֶּם

בָּתִּים לֹא בְנִיתֶם וּכְרָמִים לֹא נְטַעְתֶּם

אָז יָחִיל טוּבְכֶם כִּי תִפְרְצוּ פָרֶץ

וְהוּשַׁבְתֶם לְבַדְּכֶם בְּקֶרֶב הָאָרֶץ!

כֵּן כָּתְבוּ הַשָּׂרִים בִּמְגִלּוֹת סְתָרִים

הֵפִיצוּ לִרְבָבוֹת בֶּעָרִים בַּכְּפָרִים,

כֵּן כָּתְבוּ הַשָּׂרִים שׁוֹמְרֵי הַסְּדָרִים

הַנִּשְׁבָּעִים לַמֶּלֶךְ לַעֲשׂוֹת מֵישָׁרִים


וּבְנִכְלֵיהֶם עוֹרְרוּ חָמָס וָמֶרִי

וּדְבָרָם הִכָּה שֹׁרֶשׁ עָשָׂה פֶרִי

כִּי תַאֲוַת בֶּצַע בְּכָל נֶפֶשׁ שׂרֶרֶת

בָּזֹה הִיא גְלוּיָה וּבָזֹה מְסֻתֶּרֶת –

אִם יָבִיא הַמִּקְרֶה מִי יְמָאֵן קַחַת?

מִי נֹעֵר כַּפָּיו מִתֵּת בַּצַּלַּחַת?

אִם אֵין מַפְרִיעַ אִם אָפֵס מֹנֵעַ

מִי יֶחְדַּל הִבָּנוֹת מֵחֻרְבַּן רֵעַ?

אַף כִּי הָאֲסַפְסֻף, הַנֶּפֶשׁ הָרֵיקָה,

אֶבְיוֹנֵי אָדָם, הַנֶּפֶשׁ הַשּׁוֹקֵקָה!

שָׁמְעוּ וַתְּחִי נַפְשָׁם צָהֲלוּ וַיַּעֲלֹזוּ

וּבְדֵי שָׁלָל זְרֹעֵי יְדֵיהֶם יָפֹזּוּ

וַיִּתְחָרְשׁוּ לִהְיוֹת לַיּוֹם עֲתִידִים

וּפְרָצִים יֵצְאוּ לִפְרֹץ בַּיְּהוּדִים

וּכְבָר שָׂמוּ אֹתוֹתָם אֹתוֹת זַעַם

וַיִּתְהַלֵּךְ קוֹל נֶעְלָם בְּסֵתֶר רַעַם. –

וְהַשְּׁמוּעָה בָאָה עַד הַמֶּלֶךְ בִּמְסִבּוֹ

וְהַיָּמִים יְמֵי מְבוּכָה – וַיֵּצֵא לִבּוֹ

וַיְמַהֵר לִכְלוֹא הָרוּחַ הָרָעָה

בְּטֶרֶם תִּתְחוֹלֵל וּתְהִי לִזְוָעָה

וּבְעֹז מֶלֶךְ וּדְבַר שִׁלְטוֹן בְּמַפְגִּיעַ

נִתְּנָה הַדָּת לְהוֹדִיעַ וּלְהַשְׁמִיעַ


כִּי לֹא יִנָּקֶה בַּיְּהוּדִים נֹגֵעַ

וְחִיַּב אֶת רֹאשׁוֹ כֹּל מוּסָר פֹּרֵעַ.

הָרָצִים יָצְאוּ דְחוּפִים, הַדָּת פָּשָׁטָה

נָמֹג הָעָם, אֶרֶץ יָרְאָה וְשָׁקָטָה. –


רָאוּ הַשָּׂרִים כֵּן תָּמְהוּ נִבְהָלוּ

כִּי בַשּׁוּחָה כָרוּ כִּמְעָט נָפָלוּ

וּלְהַשְׁקִיט שָׁאוֹן חָתְרוּ בְּכָל כֹּחַ

לִבְלִי יֵחָקֵר דָּבָר וְיִשָּׁכֵחַ

וּבְלִי גַם יִפָּקֵד עַל לֵב לֹא יַעַל

וְלֹא תִגָּלֶה יָדָם הָיְתָה בַמַּעַל. –


וּלְדַרְיוֹשׁ דָּבָר הַמֶּלֶךְ צִוָּנִי

אַךְ בְּמַשָּׁאוֹן כַּסּוֹת חַשְׁתִּי גַם אָנִי.

כִּי עָלַי מַחְשְׁבוֹתָם הֵיטֵב חָקַרְתִּי

לְהַפְגִּיעַ בִּי אֶת הָאוֹיֵב וְנִגְזַרְתִּי,

כִּי בִפְרֹץ הֲמוֹנִים בִּפְרֹעַ פְּרָעוֹת

יְרֻצוּן וְיִכּוֹנָנוּ אֵלַי, מְקוֹר הָרָעוֹת,

כֵּן יוֹרוּם הַשָּׂרִים, יָד לֹא יַטְמִינוּ

וּלְכַלְכֵּל דָּבָר עַל אָפְנָיו יָבִינוּ;

אִם זֹאת לֹא תִצְלָח וְיָד לֹא תַשִּׂיגֵנִי

אֶל הַמֶּלֶךְ יַקְשׁוּ מִשַּׁלִּיט הֲסִירֵנִי

וְלֹא אֶהְיֶה לְמוֹקֵשׂ לָשִׂים עָם כְּמֶרְקָחָה

כִּי בְשֶׁלִּי הַסַּעַר – וְהָיְתָה הָרְוָחָה.


זֹאת חֲקַרְתִּיהָ וְהַדָּבָר הֶעֱלַמְתִּי

וּכְבַעַר לֹא יֵדַע הָיִיתִי וְנֶאֱלַמְתִּי

לִבְלִי יֵעָנְשׁוּ וִיחַסְּדֵנִי שֹׁמֵעַ

כִּי עָוִיתִי מִשְׁפָּט עָדַי נֹגֵעַ.

כִּי תָם אָנִי לֹא אֵדַע רָאוּ הַשָּׂרִים

וַיִּשְׂמְחוּ לָשׁוּב לִירוֹת בַּמִּסְתָּרִים,

בִּמְזִמָּתָם כִּי נוֹקְשׁוּ לֹא נוֹאָשׁוּ

וְאַחֶרֶת לִמְצוֹא תִקְוָה לֹא נָטָשׁוּ –

רָעַת הָאָדָם רַבָּה לֹא תֵחָקֵר

וּכְהָקִיר בַּיִר מַחְשְׁבוֹת אָוֶן תָּקֵר

וּפִתְגָּם נוֹדַע מִכְּבָר הַיָּמִין:

“יָגַעְתִּי וְלֹא מָצָאתִי – אַל תַּאֲמִין!”

יָגְעוּ צוֹרְרַי בְּתַחְבֻּלוֹת הִתְהַפָּכוּ

וַיְחַבְּלוּ עֵצָה וּמְזִמָּה עָרָכוּ

וַיָּבֹאוּ אֶל הַמֶּלֶךְ וַיְדַבֵּרוּ:

נַחְנוּ עֲבָדֶיךָ חֲיָלִים גִּבֵּרוּ

לִבְצֹר רוּחַ עָם וּשְׁאוֹנוֹ הַשְׁבִּיחַ

וּלְבָעֵר הָרָע כָּל מַסִּית מַדִּיחַ,

הָרוּחוֹת הָרָעוֹת הָמוּ וַיָּשֹׁקּוּ

תָּמְהוּ מִגְּעָרָה דָּמוּ וַיָּשֹׁכּוּ

וְאַנְשֵׁי תָמִיד יַבְדִּילוּ עוֹבְרִים בָּאָרֶץ

לְהַשְׁקִיט מִסָּבִיב מִצְּוָחָה וָפָרֶץ


אֹסֶף בְּלִי יָבֹא וּגְדוּדִים לֹא יֵרָאוּ

וּמְפִירֵי חֹק בְּבָתֵּי כְלָאִים הָחְבָּאוּ

גַּם אָזְנַיִם כָּרִינוּ לִשְׁמוֹר מַסָּח

מוֹצָא פִי הָאָדָם, מַה דִּבֵּר מַה־שָּׂח,

כָּל אֵלֶּה עָשִׂינוּ וְיָדֵנוּ רָמָה.

לִגְדֹר פֶּרֶץ הָקֵם סְעָרָה לִדְמָמָה,

אַךְ לִבֵּנוּ יֶהְגֶּה אֵימָה וָשֶׂפֶק

אִם בִּפְעֻלָּתֵנוּ מִלֵּאנוּ הַסֶּפֶק,

אַף כִּי הִרְבִּינוּ, יָדֵנוּ לֹא קָצָרָה

לִכְלוֹא רוּחַ מַשְׁחִית כַּבּוֹת אֵשׁ זָרָה

אַךְ שֶׁמֶץ מֶנָּה אוּלַי יִשָּׁאֵר

עוֹד בָּהּ עֲשִׂירִיָּה וְשָׁבָה וְהָיְתָה לְבָעֵר!

לָכֵן רָחַשׁ לִבֵּנוּ הַשְׁלֵם הַמְּלָאכָה

לָתֵת לָעָם מוֹרָא, חֹזֶק לַמַּמְלָכָה,

בַּעֲזוֹז עֹז מֶלֶךְ לְמַעְלָה הַגְבֵּהַּ

וְעַל פְּנֵי כָל הָעָם יִכָּבֵד כֵּאלוֹהַּ. –

לְזֹאת, אֲדוֹנֵינוּ הַמֶּלֶךְ לְעוֹלָם יֶחִי!

לְמַלֵּט עַמְּךָ מֵרָע אַרְצְךָ מִדֶּחִי

נוֹעֲצוּ לֵב יַחַד הַשָּׂרִים וְהַסְּגָנִים

הָאֲחַשְׁדַּרְפְּנִים הַפַּחוֹת וְהַסְּרָנִים

לְהִנָּתֵן דַּת הַמֶּלֶךְ כָּלָה וְנֶחֱרָצָה

לֶאְסֹר אִסָּר בְּתֹקֶף פְּקֻדָּה נִמְרָצָה


לִבְלִי יָזִיד אִישׁ בָּעִיר בַּמְּדִינָה

אֲשֶׁר לוֹ דָבָר, כָּל תְּפִלָּה, כָּל תְּחִנָּה,

חַלּוֹת פְּנֵי גֶבֶר לַעֲשׂוֹת שְׁאֵלָתוֹ

אַף לֹא לֵאלֹהָיו יָבִיא תְפִלָּתוֹ

כִּי אִם אֵלֶיךָ מַלְכֵּנוּ, רֹאשׁ הַתְּחִלָּה,

עָדֵיךָ כָל בָּשָׂר יָבֹאוּ בִתְפִלָּה.

וּמְבַקֵּשׁ דָּבָר מֵאֱנוֹשׁ, מֵאֱלוֹהַּ

חוּץ מִמְּךָ גָּבוֹהַּ מֵעַל גָּבוֹהַּ

בֶּן מָוֶת הוּא וְהֻשְׁלַךְ לְגֹב הָאֲרָיוֹת

וּלְעַמְּךָ תִוָּרֵא בַּעֲשׂוֹתְךָ נוֹרָאוֹת.

עַתָּה, מַלְכֵּנוּ! בִּימִינְךָ רֹמֵמָה,

הִנֵּה כְתָב הַדָּת הוֹאֵל קַיֵּמָה

וּתְהִי זֹאת לְחֹק מֵהַיּוֹם עַד חֹדֶשׁ יָמִים

עַד יִסְכֹּן עָלָיו עָם בֶּאֱמֶת וּבְתָמִים. –


כֹּה שָׁנְנוּ לְשׁוֹנָם לֵב מֶלֶךְ חֹדֶרֶת

וְהוּא נִפְעָם וְנִדְהָם בּוֹ נַפְשׁוֹ סֹעֶרֶת

וּלְהַרְעִישׁוֹ יוֹסְפִים בִּשְׁמֻעוֹת יְבַהֲלֻהוּ

וּבְחֹק זֶה, יֹאמְרוּ, מִכָּל רָע יַצִּילֻהוּ.

וּבְדַבְּרָם – בִּפְנֵי הַמֶּלֶךְ נִחֵשׁוּ

וּבְהַכִּירָם אוֹת רָצוֹן הִתְאוֹשֵׁשׁוּ

מִהֲרוּ וַיַּחְלְטוּ וּכְתָב הַדָּת שָׂמוּ

וּבְטַבַּעַת הַמֶּלֶךְ חִישׁ חָתָמוּ –


וַאֲנִי כְּתָב הַדָּת בִּרְאוֹתִי הֶאֱמַנְתִּי

כִּי כֵן חָפֵץ מַלְכִּי וּכְמוֹ הִסְכַּנְתִּי

לִרְאוֹת מַלְכִּי הָרִאשׁוֹן נְבֻכַדְנֶאצַּר

וּבְגַאֲוָתוֹ גַּם דָּרְיָוֶשׁ לֹא נֶעְצַר

מֵהִתְאַוֹּת תַּאֲוָה כֵּאלֹהִים הוֹפִיעַ

לֵאמֹר לַכֹּל: אֵין מִבַּלְעָדַי מוֹשִׁיעַ!

וְעַל כֵּן זֹאת הַפַּעַם לֹא עִמִּי נוֹעַץ

בַּאֲשֶׁר יָדַע זֶה חֶפְצוֹ לִי מִנֹּאָץ

וּלְדָת כָּזֹאת לָכֵן לִבִּי לֹא שַׁתִּי

וּמִצְוַת הַמֶּלֶךְ מֵעֲבוֹר לֹא חַתִּי

כִּי לֹא אֵלַי הִיא נַפְשִׁי מְאֹד יֹדַעַת

הֵן מִכְּבָר עָשׂוּ חָפְשִׁי מִכָּל מִשְׁמַעַת

אֱמוּנָתִי עֲבוֹדָתִי לְאֵל אֲבוֹתַי

וּבְחֵקֶר אֵל אִישׁ אַל יָנִיא מַחְשְׁבוֹתַי

וּבְתָם לֵבָב לָכֵן שָׁקֵט בָּטוּחַ

אֶל חַדְרִי בָאתִי בּוֹ חַלּוֹן פָּתוּחַ

נֹכַח עִיר קָדְשִׁי הַצְּבִי יְרוּשָׁלַיִם

שָׁם יוֹם יוֹם עֶרֶב וָבֹקֶר וְצָהֳרַיִם

לֵאלֹהַי אֶתְפַּלֵּל אֶכְרְעָה אֶבְרָכָה

וָאָבוֹא גַם הַפַּעַם לַעֲשׂוֹת כָּכָה

וּבְעוֹדֶנִּי מִתְפַּלֵּל עַל בִּרְכָּיִם

מִתְחַנֵּן וּמַעְתִּיר אֶל אֵל שָׁמָיִם


שָׁאֲפוּ שֹׁרְרַי, הָמוּ רָגְשׁוּ הַשָּׂרִים

כַּחֶתֶף לִי אָרְבוּ דֶּרֶךְ הָאֲתָרִים

וּבְשָׁמְעָם קוֹלִי, מִלָּתִי עַל לְשׁוֹנִי

מִהֲרוּ אֶל הַמֶּלֶךְ בִּי לַעֲנוֹת עֲוֹנִי.

אִמְרוֹתֶיךָ – אָמַר הָאַרְיֵה – מְזֻקָּקוֹת

לְשׁוֹנְךָ תַּעַר מְלֻטָּשׁ, לְךָ שְׂפַת חֲלָקוֹת

אַךְ דַּעַת אֶחְפֹּץ אֶל מָה הִתְפַּלַּלְתָּ

וּמָה הַדָּבָר מֵאֱלֹהֶיךָ שָׁאַלְתָּ?

אוּלַי בַּתְּפִלָּה וּדְבַר הַתְּחִנָּה

תִּפְלָה נָתַתָּ בַּמֶּלֶךְ בַּמְּדִינָה!

הָהּ! – עָנָה דָנִיֵּאל – תְּפִלָתִי זַכָּה

וּדְבָרִי הִתְחַנַּנְתִּי מִלֵּב דַּכָּא

אִם תִּשְׁמַע, תַּחַת לְשׁוֹנִי לֹא אַכְחִידָה

אִם הָבֵן אַךְ תּוּכַל, הָבָה, אַגִּידָה:

יִרְמְיָהוּ הַנָּבִיא הוֹדִיעַ נֶאֱמָנָה

כִּי לִמְלֹאת לְבָבֶל שִׁבְעִים שָׁנָה

מִיּוֹם לִנְבֻכַדְנֶאצַּר בָּאָה הַמֶּמְשָׁלָה

לְמִיּוֹם כָּבַשׁ אַרְצִי וּמַלְכִּי גָלָה,

אָז תִּשָּׁבֵר בָּבֶל יֶחֱרַב עַם כֶּשֶׂד

צִיּוֹן בְּמִשְׁפָּט תִּפָּדֶה וְעַמִּי בְּחֶסֶד,

מִמָּדַי וּפָרַס יָקוּם מֶלֶךְ מְנַצֵּחַ

הוּא יִבְנֶה עִירִי גָּלוּתִי יְשַׁלֵּחַ,


עַתָּה אֹתוֹתָיו הָרְאֵיתִי לָדַעַת

בֶּהָרֵג בֵּלְשַׁאצַּר וּבָבֶל נִכְנַעַת,

הֵקִיץ הַקֵּץ עָלֶיהָ בָּא בָא קָרֶץ

וְדָרְיָוֶשׁ הַמָּדִי הָמְלַךְ בָּאָרֶץ,

אָז תַּרְתִּי בְלִבִּי בִּינוֹתִי בַסְּפָרִים

אַחַת לְאַחַת לַחְשֹׁב שָׁנִים מִסְפָּרִים

וְהִנֵּה שְׁנַת הַשִּׁבְעִים הַשָּׁנָה מָלָאָה

הִנֵּה קֵץ הַפְּלָאוֹת, שְׁנַת גְּאוּלַי בָּאָה,

וָאֶתְּנָה אֶת פָּנַי אֶל נוֹרָא עֲלִילָה

עַל הַר קֹדֶשׁ אֱלֹהַי לְבַקֵּשׁ תְּפִלָּה

וַאֲנִי מִתְוַדֶּה וּמִתְחַנֵּן לָאֵל

עַל שְׁבִיָּה בַת צִיּוֹן עַל גָּלוּת הַחֵל

כִּי יָקִים דְּבַר עַבְדּוֹ נָבִיא שָׂמָהוּ

לִבְלִי חַטֹּאתֵינוּ הַטּוֹב יִמְנָעוּ

וּבְרַחֲמָיו הָרַבִּים כְּרַחֵם אָב עַל בָּנִים

יָשִׁיב אַפּוֹ וַחֲמָתוֹ וְיָאִיר פָּנִים

עַל מִקְדָּשׁוֹ הַשָּׁמֵם בֵּיתוֹ הֶחָרֵב

עַל נִדְחֵי יִשְׂרָאֵל עַם שְׂרִידֵי חָרֶב

כִּי יָשׁוּב וְנִחָם וִישִׁיבֵם לְקַדְמָתָם

יָשׁוּב וְנָחָם וִינִיחֵם עַל אַדְמָתָם

בִּירוּשָׁלַיִם הַבְּנוּיָה לְתַלְפִּיּוֹת

כִּימֵי עוֹלָם וּכְשָׁנִים קַדְמֹנִיּוֹת!


הַמְצָאתִיךָ?! – שָׁאַג הָאַרְיֵה בְזַעַם –

דְּבַר בְּלִיַּעַל יָצוּק בָּךְ נֶחְשַׂף הַפַּעַם!

פִּיךָ יַרְשִׁיעֲךָ! שָׁמְעָה אָזְנִי וַתָּבֶן

כִּי קִרְבְּךָ הַוּוֹת לִבְּךָ יִקְבָּץ אָוֶן

בֶּן נַעֲוַת הַמַּרְדּוּת! טוֹבָה שֹׁכֵחַ

לִמְעֹל בַּמֶּלֶךְ בְּךָ לִבּוֹ בֹטֵחַ!

וּבְהַאֲמִינוֹ בְךָ וַיְנַטְּלְךָ בִּשְׂאֵת יֶתֶר

וּבְיָדְךָ עֵת יַפְקִיד אֶרֶץ וָכֶתֶר

תִּשָּׂא עֵינֶיךָ אֶל אֶרֶץ אַחֶרֶת

הִיא לְךָ אֶרֶץ חֵפֶץ לְגָאוֹן וּלְתִפְאֶרֶת

וּתְשׁוּקָתְךָ עָלֶיהָ שַׁוּוֹת עֵזֶר

בְּעָם לְבָדָד יִשְׁכֹּן עָלָיו יָצִיץ נֵזֶר!

אִם לִבְּךָ צִיּוֹנָה פֹּה הִנְּךָ עֵקֶר

וַעֲבוֹדָתְךָ לַמֶּלֶךְ רְמִיָּה וָשֶׁקֶר

תֻּמָּתְךָ אַךְ מִכְסֶה, בֶּגֶד מִמַּעַל,

מִרְמוֹת וָתֹךְ תּוֹכְךָ, בֶּגֶד וָמַעַל –

לִמְרֹד בַּמֶּלֶךְ חָשַׁבְתָּ מַחֲשָׁבֶת

נִכְתַּם עֲוֹנְךָ וּלְךָ מִשְׁפַּט מָוֶת!


הָאַרְיֵּה הָגָה – וַיְהִי דְבָרוֹ לְקוֹל הָמוֹן

וְהִנֵּה קוֹל רַעַשׁ גָּדוֹל: אָמֵן וְאָמֵן!

לַיִשׁ אֹבֵד מִבְּלִי טֶרֶף הִתְנָעֵר:

“כֵּן, מוֹרֵד הוּא, בֶּן־מָוֶת וְהָיָה לְבָעֵר!”

וּבְנֵי לָבִיא הִתְפָּרְדוּ חֻבְּרוּ פֶתַע:

“דָּתוֹ לְהִטָּרֵף, הוֹ, נְשַׁסְּעֵהוּ שֶׁסַע!”

בַּעֲלִיצוּתָם אַחֲרֵיהֶם הֶחֱרוּ הֶחֱזִיקוּ

גּוּרֵי אֲרָיוֹת וּמְתַלְעוֹתָם הֵרִיקוּ.

וּמֵחֵק אֲרִי מְסֻבָּל לְבִיאָה קָמָה

וּבְצַוָּאר עָתָק הִתְנַשְּׂאָה רָעָמָה:

– לֹא חִנָּם צֵָרנוּ צִיּוֹנָה יִכְסֹף

שָׁם הַרְרֵי טֶרֶף יָמִישׁ, יְעָרוֹת יֶחֱשׂף,

יַשְׁמִיד חַיְתוֹ לְבָנוֹן אַבִּירֵי בָשָׁן

יַכְרִית חַיַּת כְּפִירִים וְכָלוּ כֶעָשָׁן.

כֹּל יָשִׁית תַּחַת רַגְלָיו זֶה הָאַכְזָר

הֵן נַפְשׁוֹ אִוְּתָה רָע, חֶפְצוֹ, הָהּ, אַךְ זָר:

הָמֵר אֶרֶץ יֹאמַר זֶה אִישׁ אָוֶן וּמְרִי

לָשׂוּם אַרְיֵה כַּבָּקָר כְּעֵגֶל וּמְרִיא

וְשִׁעֲשַׁע בִּזְקָנוֹ שֶׂה, יִרְהַב בַּלָּיִשׁ

עַל שַׁחַל תִּדְרֹךְ רָחֵל, יִרְמְסֶנּוּ תָיִשׁ,

וּבֶן אָדָם זֶה בְּחָכְמָתוֹ יִתְאַמֵּר

יִכְבּשׁ כָּל אַחֵינוּ וּבָם יִתְעַמֵּר,

בְּתֶלֶם עֲבוֹתוֹ יִקְשֹׁר דֹּב עִם שׁוֹרוֹ

עָלוֹת יְנַהֵל זְאֵב יָבוֹל לַחֲמוֹרוֹ

לְנָמֵר יִרְתֹּם מֵרְכַּבְתּוֹ וְיִבְלְמֶנּוּ

וִיזַנֵּב הַשּׁוּעָל רַכָּב יְשִׂימֶנּוּ –


הֵן כָּל אֵלֶּה יִפְעַל זֶה אֱנוֹשׁ רִמָּה

הָבָה נַכְרִיעֵהוּ טֶרֶם בִּצַּע מְזִמָּה!

הַס! – קָרָא אַרְיֵה הָרֹאשׁ מָלֵא חֵמָה –

מָה עִמֵּךְ, לְבִיָּא! מַה תַּפִּילִי אֵימָה?

הֶאָנֹכִי אָגוּר מִבֶּן אָדָם חָצִיר

כִּי יָעֹז בְּהַוָּתוֹ עַל יַעַר הַבָּצִיר?

שָׁם מַרְבֵּץ לַחַיָּה מִדֹּרוֹת עוֹלָמִים

בְּטֶרֶם נוֹצַר זֶה הָאָדָם קְצַר יָמִים.

אָמְנָם כֵּן, הוּא מִתְהַפֵּךְ בְּתַחְבֻּלוֹתָיו

וּכְשַׁךְ יְקוּשִׁים נִלָּכֵד בִּשְׁחִיתוֹתָיו

וּמְתֵי מְעָט נִתְפָּשׂוּ הָיוּ לְמַס עוֹבֵד

בִּידֵי זֶה הַסָּס, הַשָּׂשׂ לָמַס אוֹבֵד

אַךְ לַעֲשׂוֹת כָּלָה אֶת חַיַּת הַלְּבָנוֹן

וּלְהוֹרִיד עֹז מִבְטֶחָה לִהְיוֹת מָנוֹן

מֶנְּהוּ יִבָּצֵר לוֹ קָצְרוּ יָדָיִם

וְאִלּוּ חָיָה אֶלֶף שָׁנִים פַּעֲמָיִם!

לָכֵן הֵרָגְעוּ, אַחַי! אַל תִּשְׂעַרוּ שַׂעַר

אַל תִּדְאֲגוּ לִשְׁלוֹם אַחֵינוּ בַּיַּעַר,

אִם כָּל יַעְרֵי עַד יֵהָפְכוּ מִשֹּׁרֶשׁ

אִם יַעֲלֶה הַכּוֹרֵת וְנִקַּף כָּל חֹרֶשׁ

אַדִּיר הוּא הַלְּבָנוֹן דָּגוּל מֵרְבָבָה

וְחַיְתוֹ יַעַר בּוֹ תִרְמשׂ אֵיתָן מוֹשָׁבָהּ


וְאַתָּה, בֶּן אָדָם! – אֶל דָּנִיֵּאל פָּנָה –

יְאַלֵּף עֲוֹנְךָ פִיךָ וּבְךָ עָנָה

כִּי נָקִי יָשָׁר בָּאָדָם אֵינֶךָ

לִמְרֹד בַּמֶּלֶךְ נָשָׂאתָ עֵינֶךָ!

וּמוֹתָר לְךָ אַיִן מִכָּל הָאָדָם

שְׂנֵאתִים עַל תַּרְמִיתָם וַחֲמַס יָדָם

וְאַתָּה – מֶלֶךְ רִמִּיתָ, לֹא רַק עֲמִיתְךָ,

הִצְטַדֵּק אִם תּוּכָל, וָלֹא – אֲמִיתְךָ!

לָמָה הֶבֶל אִיגָע? – עָנָה דָנִיֵּאל –

וּמִיָּדְךָ הִגָּזֵר דָּנַנִי אֵל!

הַחֵפֶץ לְךָ כִּי אֶצְדַּק וּתְנַקֵּנִי?

לִבְשָׂרִי הֵן תִּתְאָו – הִנְנִי, אָכְלֵנִי!

כַּמָּה לֹא תִשְׁעֶה מֶנִּי וְכַעַשׂ תֶּרֶב

תִּדְרשׁ לַעֲוֹנִי תִּבְחַן לֵב וָקֶרֶב,

חַטָּאתִי לְפָנֶיךָ – עַצְמִי וּבְשָׂרִי,

וּמֵחֶטְאִי זֶה אַךְ בִּטְנְךָ מִטְהָרִי.

כִּמְדַבֵּר בִּצְדָקָה לָמָּה תִתָּבָר?

לָמָּה לְךָ מַסְוֶה כַּסּוֹת עֶרְוַת דָּבָר?

אַתָּה חוֹתַם תָּכְנִית: חַנֵּק, טָרוֹף וּשְׁכוֹל

אֹתוֹתֶיךָ אַל תְּנַכֵּר – חֲטוֹף וֶאֱכוֹל!

רַק לִבְנֵי הָאָדָם, בִּלְבָבָם מֹרֶךְ

בִּכְסוּת עֵינַיִם רַק לָמוֹ יֵשׁ צֹרֶךְ


וּבְשַׂר רֵעַ לֶאֱכוֹל מְזִמּוֹת יַחְמֹסוּ

עַל שָׁרְשֵׁי רַגְלָיו יִתְחַקּוּ עֹלוֹת יַחְפֹּשׂוּ

שֵׂעָר דַּק יִמְצָאוּ בְּהַוָּתָם יְעַבְּתוּהוּ

“הֶאָח, רָאֲתָה עֵינֵנוּ!” – וְיִבְלָעוּהוּ.

כֵּן אֹכְלֵי עַמִּי עַל נַפְשׁוֹ יָגֹדּוּ

לְהַשְׁמִיד בִּלְבָבָם כִּי חֶלְקוֹ יַחְמֹדוּ

אַךְ לָקוּם לִרְצֹחַ בְּכֹחַ הַזְּרֹעַ

יֶחֱרַף לְבָבָם מָלֵא זָדוֹן וָרֹעַ

לָכֵן יִתְחַפְּשׂוּ – הָהּ, הֹלְלִים יָהֹלּוּ! –

כִּמְבַקְשֵׁי מִשְׁפָּט אָרְחוֹת צֶדֶק יָסֹלּוּ

וַעֲוֹן עַמִּי יְבַקְּשׁוּ יִיעֲצוּ לַהֲדִיחוֹ

“גּוֹי חוֹטֵא הוּא, אֵין שֹׁוֶה לְהַנִּיחוֹ!”

וַחֲטָאִים אִם תַּמּוּ, אִם כָּלָה פֶשַׁע

לֹא יַחְדְּלוּ הִתְעוֹלֵל עֲלִילוֹת בְּרֶשַׁע

כִּי גַם צִדְקַת עַמִּי תְּהִי לַחֲטָאָה

כִּי יִתְרֹון לוֹ בַּכֹּל – וְהִנֵּה מַתְּלָאָה!

כֵּן יֶהֱמוּ יִרְעָשׁוּ צֹרְרֵי יְהוּדָה

לִמְצוֹא עֲוֹנוֹ לִשְׂנֹוא יָסְדוּ אֲגֻדָּה

הַאֲבִיד עַם לֹא עָז יִתְנַכְּלוּ יִתְאַנּוּ

בְּרָב כֹּחַ יִתְחַפְּשׂוּ עֹז פָּנִים יְשַׁנּוּ.

דַּרְכָּם כֶּסֶל לָמוֹ – זֶה כָּל הָאָדָם

כִּי יָפוּג לִבָּם וּכְמוֹ חָזְקָה יָדָם,


וְאַתָּה – אֲרִיאֵל! לֹא לְאָדָם דַּרְכֶּךָ

לִבְּךָ לֵב אַרְיֵה – יִגְדַּל נָא כֹחֶךָ!

אַל תִּשְׁעֶה בְהַבְלֵי בְנֵי אָדָם אֱוִילִים

לְהַצִּיקֵנִי בִשְׁאֵלוֹת לְדַכְּאֵנִי בְמִלִּים,

הֵן בִּטְנְךָ חֲסֵרָה – כַּפְּךָ נָא הָרֵם

בְּקָעֵנִי אָכְלֵנִי וְעַצְמוֹתַי גָּרֵם!

לֹא! – קָרָא הָאַרְיֵה כְּנָח מִזַּעְפּוֹ

וְעַל שֶׁכֶם דָּנִיֵּאל נִחֲתָה כַפּוֹ –

לֹא לִמְצוֹא תֹאֲנָה מִבְּשָׂרְךָ לִשְׂבֹּעַ

הֶעֱמַדְתִּיךָ עַד כֹּה מִבְּלִי לִנְגֹּעַ,

לִטְרֹף אִם אֶכְסֹף פִּי וְלִבִּי נִשְׁתָּווּ

לֹא כִבְנֵי אָדָם חַבְלֵי שָׁוְא לָמוֹ טָווּ

לֹא אָדָם אָנִי – זֹאת נְכוֹנָה דִבַּרְתָּ

וּבְעֵינַי עַל כֵּן נֵחַנְתָּ יָקַרְתָּ

כִּי רָעַת הָאָדָם הֵיטֵב חָקַרְתָּ

וּמַר בַּחֲמַת רוּחַ בּוֹ הִתְעַבַּרְתָּ

וּנְתַתִּיךָ עַל כָּכָה הַרְבּוֹת לַהַג

לִשְׁמוֹעַ אֲמָרֶיךָ בִּשְׁתוֹתְךָ לַעַג –

לֹא אָדָם אָנִי, הַוֹּת לֹא אָבִינָה

אֱמֶת חָפַצְתִּי בָהּ צַעֲדִי אָכִינָה

וּדְבָרֶיךָ יִהְיוּ אֱמֶת בְּךָ אָעִידָה

לְמִשְׁפָּטִי אֵפוֹא כִּי אוֹתְךָ אוֹעִידָה!


הוֹכִיחָה אֵיךְ צָדְקוּ תַהְפֻּכוֹת שְׁתָּיִם:

עֶבֶד בָּבֶל וּבוֹחֵר בִּירוּשָׁלָיִם!

וַיַּעַן דָּנִיֵּאל: – אִם אֱמֶת חָפַצְתָּ

אַל תְּהִי כִבְנֵי אָדָם דַּרְכָּם נִאַצְתָּ

הֵמָּה כִי נִכְּלוּ לִי וּלְנַפְשִׁי זָדוּ

עֲוֹן מֶרֶד עָלַי מִלִּבָּם בָּדוּ.

בִּינָה הֲגִיגִי, בְּחָנֵנִי דַּע שַׂרְעַפָּי

כִּי לֹא אָוֶן בְּלִבִּי לֹא עָוֶל בְּכַפָּי

אִם לֵאלֹהַי בִּתְפִלָּה הִתְחַנַּנְתִּי

עַל עַמִּי אֵהָב, עַל אַרְצִי חוֹנַנְתִּי

כִּי לֹא יֶחֱטָא הַיְּהוּדִי אִם מַחְשְׁבוֹתָיו

בִּמְקוֹרוֹ, בְּקַדְמָתוֹ, בְּאֶרֶץ אֲבוֹתָיו.

הַיְּהוּדִי הַגּוֹלֶה יָבִיא צַוָּארוֹ

בְּעֹל מַלְכּוֹ מְחוֹנְנוֹ. וְלֹא יַחֵל דְּבָרוֹ,

דִּבְרַת שְׁבוּעַת אֱלֹהִים יִשְׁמוֹר תָּמִים

לַמֶּלֶךְ לַמְּדִינָה נֶאֱמָן כָּל הַיָּמִים;

הַיְּהוּדִי הַנּוֹדֵד טוֹבָה לֹא שֹׁכֵחַ,

וּבְאֶרֶץ גָּלוּתוֹ כִּי יִמְצָא מָנוֹחַ

חֶסֶד מֶלֶךְ יִזְכֹּר חָרוּת עַל לוּחַ,

וּבֶאֱמוּנָה יַעֲבֹד וּבְכָל תֹּם רוּחַ,

גַּם דָּמוֹ כִי נִדְרָשׁ גַּם דַּם זַרְעֵהוּ,

כֵּן אֵל צִוָּהוּ כֵּן לִבּוֹ יוֹרֵהוּ.


אַךְ תֻּמָּתוֹ לְמַלְכּוֹ בְּאֶרֶץ מִקְלָטוֹ

לֹא תִהְיֶה לוֹ לְפוּקָה הוֹבִישׁ מַבָּטוֹ

כִּי עוֹד תָּרוּם קַרְנוֹ, עוֹד יִגַּהּ אוֹרוֹ

לָשׁוּב אֶל אַרְצוֹ, לִכְבוֹדוֹ, לִמְקוֹרוֹ,

לֹא בְתַחְבֻּלוֹת אָוֶן וְלֹא בְּיָד חֲזָקָה,

לֹא בְמֶרֶד וְלֹא בְמַעַל – כִּי אִם בִּצְדָקָה,

עֵת רָצוֹן כִּי תָבֹא מֵאֱלֹהִים וָמֶלֶךְ

וְגַם הוּא אֶל אַרְצוֹ וְאֶל מוֹלַדְתּוֹ יֵלֶךְ.

הֵן כָּל גּוֹיֵי הָאֲרָצוֹת לְאֻמּוֹתָם

אִישׁ אֶל עַמּוֹ יִפְנוּ בְּהִתְיַחֵשׂ לְתֹלְדֹתָם,

וְאֵין מַכְלִים דָּבָר לַחְשֹׁב לַחֲטָאָה

מַשָּׂא נֶפֶשׁ עָם אִם לֹא יַחֲרשׁ רָעָה.

גַּם לִיהוּדִי לֵב, גַּם הוּא בֶּן עַמּוֹ,

וּבַגּוֹלָה כִי הָלַךְ בּוֹ עָמַד טַעְמוֹ,

לֹא נִכְחַד עָצְמוֹ יִצְרֶנְהוּ כְּאִישׁוֹן

וְאֶל מְקוֹמוֹ שׁוֹאֵף לִכְבוֹדוֹ הָרִאשׁוֹן –

וְזֶה כָּל הַיְּהוּדִי, זֶה חוֹתַם תָּכְנִיתוֹ:

אַוָּתוֹ תִקְוָתוֹ לָשׁוּב אֶל רֵאשִׁיתוֹ

וּכְעָם בְּצִיּוֹן יֵשֵׁב וּשְׂאֵתוֹ מִמֶּנּוּ,

וּמִי לֹא כֵן יוֹחִיל – יְהוּדִי אֵינֶנּוּ,

וּכְמוֹ הַכַּשְׂדִּי זֶה שְׁמוֹ יִהְיֶה לְחָגָא

אִם אֶל עַמּוֹ וְאַרְצוֹ לֹא נַפְשׁוֹ עָרָגָה.


יִבָּדֵל מִקָּהָל וְנִכְתַּם עֲוֹנוֹ

אִם לֹא בָבֶל חִשְׁקוֹ תִּפְאֶרֶת גְּאוֹנוֹ;

וּלְשֵׁם גַּם יְהוּדִי אִם לֹא נָשׂוּא לַשָּׁוְא

צִיּוֹן הִיא צִיּוּן וּמִקְדָּשָׁהּ הַתָּו –

זוֹ צִיּוֹן עֶרֶשׂ עַם עֵבֶר וּמְכֹרָתוֹ,

רוּחוֹ הִיא תִכְּנָה, לוֹ נָתְנָה נִשְׁמָתוֹ

כִּי בָהּ הָאֱלֹהִים נִגְלָה בִכְבוֹדוֹ

בְּחֵיקָהּ הִנִּיחַ רוּחוֹ, צָפַן סוֹדוֹ,

וּלְעַם קְרוֹבוֹ, עַם סְגֻלָּה, הֶעֱרָה מְקוֹרָהּ

וּלְיִשְׂרָאֵל מִמֶּנָהּ יָצְאָה תוֹרָה –

מֵאָז, שְׁלשֶׁת אֵלֶּה יַחַד יִתְלַכָּדוּ,

צִיּוֹן יִשְׂרָאֵל וְתוֹרָה לֹא יִתְפָּרָדוּ,

כְּמוֹ חֲלָאִים מַעֲשֶׂה יְדֵי אָמָן נִקְשָׁרִים

הָאֶחָד שֶׁיִּפֹּל אֵין תְּקוּמָה לַנִּשְׁאָרִים;

וּמֵאַרְצֵנוּ עַתָּה כִּי נִדַּחְנוּ

אוֹצָרָהּ הַטּוֹב אִתָּנוּ לָקַחְנוּ:

טָהֳרַת הַקֹּדֶשׁ וּשְׂפֻנֵי סְגֻלָּתָהּ

צְרוּרוֹת בִּצְרוֹר חַיֵּינוּ – בְּסִפְרָתָהּ!

זֶה סֵפֶר הַתּוֹרָה, בּוֹ מַחֲבֵא רוּחַ,

רוּחַ אֱלֹהִים שֶׁבְּאַרְצוֹ תָפוּחַ,

וְתוֹרָתֵנוּ נִשְׁמוֹר רוּחֵנוּ שֹׁמֶרֶת

לִבְלִי תִצְלַח עָלֵינוּ רוּחַ אַחֶרֶת,


וְאִם נִדָּחֵנוּ בִּקְצֵה הַשָּׁמַיִם

נַפְשֵׁנוּ בְצִיּוֹן, לִבֵּנוּ בִירוּשָׁלַיִם,

וּבְתִקְוָתֵנוּ זֹאת בַּגּוֹלָה נֶחִי

כִּי מְזָרֵה יִשְׂרָאֵל יְחַלְּצֶנּוּ מִדֶּחִי

וִיכַנֵּס אֶל הָאֲדָמָה מֶנָּהּ גָּלִינוּ

וִיקִימֵנוּ לָעָם לִהְיוֹת אֲשֶׁר הָיִינוּ,

וּלְנִשְׁמַת הַתּוֹרָה נִהְיֶה הַחֹמֶר

אֶל רוּחָהּ חַיִּים וּלְרַעְיוֹנָהּ אֹמֶר –

וְאִם לֹא זֹאת לָנוּ כָּל יֶשַׁע כָּל חֵפֶץ

וּבְאַדְמַת נֵכָר נִבְחַר בְּפִזּוּר וָנֵפֶץ;

עַל תִּקְוָתֵנוּ לִהְיוֹת עַם הָאָרֶץ

אִם יַעֲלֶה הַכּוֹרֵת יָבֹא הַקָּרֶץ,

אֲזַי תּוֹרָתֵנוּ תְלוּיָה עַל בְּלִימָה

וְאִמְרָתָהּ הַצְּרוּפָה אַךְ אִמְרָה נְעִימָה,

אֲזַי אַךְ מִצְוַת אֲנָשִׁים מְלֻמָּדָה

מִצְוָתָהּ הָרְחָבָה עָלֵינוּ הָעֳמָדָה. –

וּמָה אֵפוֹא הַיְּהוּדִי אִם הוּא כְּמוֹ?

וּבְשֵׁם יִשְׂרָאֵל כִּי יְכַנֶּה – יִמַּח שְׁמוֹ!

אִם יֹאמַר הָעִבְרִי: צִיּוֹן אֹבֶדֶת

כַּשְׂדִּים עַמִּי, בָּבֶל אֶרֶץ מוֹלֶדֶת,

וְיַחֲנִיף בַּחֲלָקוֹת בַּגּוֹיִם הִתְעָרֵב

נָבָל הוּא, לֶחֶם הַקְּלֹקֵל לוֹ עָרֵב,


וְאִם בֶּאֱמֶת וּבְתָמִים כֵּן לִבּוֹ יוֹרֶנּוּ

מָה הוּא יְהוּדִי יוֹדֵעַ אֵינֶנּוּ,

אִם כֹּה אוֹ כֹה וּבָנִים יִוָּלֵדוּ

מִבְּלִי מֵשִׂים מִיִּשְׂרָאֵל יִכָּחֵדוּ…

וּבְתֻמָּתִי כִי הֶחֱזַקְתִּי וָאֵלֶךְ

בְּעָמְדִי לְמַחֲזִיק וּמָעוֹז לַמֶּלֶךְ

תָּמִים לוֹ וְנֶאֱמָן לֹא הָיָה עוֹד בִּלְתִּי,

וְזֶה הָאוֹת כִּי הִתְרוֹמַמְתִּי וְהִתְגַּדִּלְתִּי,

דֹּר לְדֹר שִׁבַּח מַעֲשַׂי הִלֵּל פָּעֳלִי

וּבִרְכַּת מֶלֶךְ תִּלְוֵנִי בַעֲמָלִי;

אֲבָל אֲשֶׁר עִם שַׁדַּי לֹא כִחַדְתִּי,

אֶת אֱלֹהַי אֶת עַמִּי וּמוֹלַדְתִּי

לֹא הִסְתַּרְתִּי, יַחַד פִּי וְלִבִּי תַמִּים,

וִיהוּדִי אָנִי גָּלוּי לְכָל הָעַמִּים,

וּמַשָּׂא נַפְשִׁי, מַשָּׂא גֵּיא חִזָּיוֹן,

לֹא טָמַנְתִּי לֹא כִסִּיתִי בְּמַשָּׁאוֹן.

עוֹד בִּימֵי בֵלְשַׁאצַּר הִבַּעְתִּי חִידוֹת

וּלְשִׁיבַת צִיּוֹן אָז חַשְׁתִּי עֲתִידוֹת

עַל עַמִּי עַל אַרְצִי חָזוֹן הִרְבֵּיתִי

וְלֹא כִהָה בִי, אֶת פּשְׁעִים לֹא נִמְנֵיתִי,

וּבְדַעְתּוֹ תֻמָּתִי נִשֵּׂא מַעֲמָדִי

וּשְׁלִישִׁית הַמַּמְלָכָה נָתַן בְּיָדִי.


וּמַלְכִּי עָתָּה – הֵן לְךָ כְּבָר סִפַּרְתִּי

עַד כַּמָּה בְעֵינָיו נִכְבַּדְתִּי יָקַרְתִּי

וּמַעֲלוֹת רוּחִי הוּא מֵבִין שׁוֹמֵעַ,

מַעַרְכֵי לִבִּי וּמְגַמַּת פָּנַי יוֹדֵעַ,

וּבְהִנַּבֵּאתִי עַל בָּבֶל הַנְּבוּכָה

כִּי אֶל מַלְכוּת פָּרַס תִּסֹּב הַמְּלוּכָה,

וּבְרֹאשׁ דָּרְיָוֶשׁ אַךְ הוּשַׂם הַכֶּתֶר

חָזוֹן מָצָאתִי דִּבַּרְתִּי לֹא בְסֵתֶר,

חָזוֹן לַמּוֹעֵד כִּי יָשׁוּב אֵל לְמוֹשָׁעוֹת

וּלְעַמּוֹ הַמְחַכֶּה יָבִיא קֵץ הַפְּלָאוֹת,

וּלְמַלְכִּי עַל כָּכָה לִמְאֹד נֶאֱמַנְתִּי,

כִּי לֹא נַחְבֵּאתִי, הֶגְיוֹן לִבִּי לֹא צָפַנְתִּי,

כִּי הִנְנִי לְפָנָיו כְּמוֹ אָנֹכִי עִמָּדִי:

יְהוּדִי יְרוּשַׁלְמִי, לֹא כַשְׂדִּי, לֹא מָדִי, –

עֶבֶד משְׁלִים, גֵּר בְּאֶרֶץ נָכְרִיָּה,

אַךְ לֵאלֹהִים יִשְׁעִי נַפְשִׁי דֻמִיָּה

כִּי אֲסִירֵי הַתִּקְוָה יָשִׁיב לְבִצָּרוֹן

לְאֶרֶץ אָבוֹת, בָּהּ לָמוֹ חֵלֶק וְזִכָּרוֹן;

גַּם צִפּוֹר מָצְאָה בַיִת, קֵן לָהּ דְּרוֹר

וּמְקַבֵּץ יִמְצָא נוֹדֵד, שָׁבוּי דְּרוֹר –

וּבְצִפִּיָּתֵנוּ זֹאת נַחֲלִיף כֹּחַ

כִּי יִקְרָא אֱלֹהִים לְעַמּוֹ פְּקַח־קוֹחַ


וּבְרַחֲמִים גְּדוֹלִים לֵב מֶלֶךְ יָעִיר

לֵב מֶלֶךְ חָסִיד, מִשְׁפָּטֵנוּ יָאִיר

וּלְטוֹבָה אֶת גָּלוּת יְהוּדָה יַכִּיר,

לְאָבִיו וּלְאִמּוֹ יָשִׁיב בֵּן יַקִּיר,

עֲשׁוּקִים לְנַחֲלָתָם יוֹנִים אֶל אֲרֻבָּתָם

וּמְנוּחָה יִמְצְאוּ נוֹדְדִים, כּשְׁלִים עֲרֻבָּתָם.

וְעַד עֵת בֹּא דְבָרוֹ גֵּר אָנֹכִי בָאָרֶץ

הִיא עָלַי סִתְרָה מֵעָוֶל וָפָרֶץ

וְתֹדוֹת אֲשַׁלֵּם – זֹאת תְּכוּנָתִי מֵעוֹדִי –

לִדְרשׁ שְׁלוֹמָהּ בְּכָל נַפְשִׁי וּבְכָל מְאֹדִי,

בְּכָל מַאֲמַצֵּי כֹחַ לִגְמוֹל לָהּ אוֹאִיל

וּבְכֹל תִּמְצָא יָדִי אֵיטִיב לָהּ, אוֹעִיל,

וּבָנֶיהָ הַטּוֹבִים לִי אַחִים וְרֵעִים

בְּחָנוּנִי גַם רָאוּ כִּי הִנְנִי תְמִים דֵּעִים,

כִּי עִבְרִי אָנֹכִי עִמִּי הִסְכִּינוּ

וַאֲמִתִּי יוֹקִירוּ בְּיָשְׁרִי יַאֲמִינוּ,

וּשְׂנוּאִים רַק אֵלֶּה אֹבְדִים נִדָּחִים

הַמְבַקְּשִׁים בְּאֶרֶץ נוֹד תּוֹרַת אֶזְרָחִים,

וּלְמַעַן יְחָנּוּם וִישׁיתוּם בַּבָּנִים

לִמְצֹא חֵן הֵם מִתְיַפִּים בְּסֵתֶר פָּנִים,

לִסְתֹּם מְקוֹר יִשְׂרָאֵל לִשְׂטֹם דֹּר עַמֵּנוּ

בַּגּוֹיִם יִתְחַשְּׁבוּ לְמַעַן יֵאָמֵנוּ,


אַךְ יִפְּלוּ מִמֹּעֲצוֹתֵיהֶם מְתֵי שָׁוְא

וּבְלֹא אֵמוּן יְחַסְּדֵם כָּל אֶזְרַח וְתוֹשָׁב –

לִשְׂנוֹא אַרְצוֹ וּלְהִתְעַלֵּם מִלְּאֻמּוֹ

לֹא יָבִין הָאֶזְרָח לֹא יַאֲמִין לְתֻמּוֹ;

וּמִשְׁפָּטוֹ אֵפוֹא עַל אֵל הָאֲנָשִׁים

כִּי בְמִרְמָה יְדַבְּרוּ בָּנִים כֶּחָשִׁים,

וּבִגְלָלָם קוֹל גָּדוֹל יַעֲבוֹר בָּעֵדָה

כִּי יַעֲקֹב יַעְקֹב וִיהוּדָה בֹּגֵדָה…

וּבְעַם הָאָרֶץ נִבְאַשׁ וַיִּפָּרַע

לְשִׁמְצָה וּנְאָצָה, חִין עֶרְכּוֹ יְגָרַע –

וּבָנִים כֶּחָשִׁים חַתּוּ נִבְהָלוּ

אָחֲזָה רְעָדָה חֲנֵפִים כִּי יֵקָלוּ,

וַיּוֹסִיפוּ אֹמֶץ חֲיָלִים לְגַבֵּר,

הִתְבּוֹלֵל בָּעַמִים, עִמָּם הִתְחַבֵּר,

וּלְמַעַן יוֹתִירוּם לְטוֹב, הָשֵׁב חִנָּם,

עַל עַמָּם יַעֲטוּ חֶרְפָּה, הַשְׂנִיאוֹ חִנָּם,

לֹא עַם בִּינוֹת הוּא – יֹאמְרוּ – רוּחַ רוֹעֶה

סָכָל אֵין לֵב מִדֶּרֶךְ הַשְׂכֵּל תֹּעֶה,

כֻּלּוֹ סָג יַחְדָּו פָּנָיו אֲחֹרַנִּית

וְאַחֲרֵי הַהֶבֶל הוֹלֵךְ קְדֹרַנִּית,

מַעֲשָׂיו תַּעְתֻּעִים, מִנְהָגוֹ שִׁגָּעוֹן –

וּבִקְלוֹנוֹ יִכָּבְדוּ יַעְדּוּ גָאוֹן,


כִּי הֵם הַמַּשְׂכִּילִים לָרַבִּים יָבִינוּ,

אֶל דֶּרֶךְ הַגּוֹיִם צַעֲדָם יָכִינוּ,

וִידֵיהֶם שִׂכְּלוּ לִירוֹת חִצִּים שְׁנוּנִים

עַל אֲחֵיהֶם בְּנֵי עַמָּם שׁוֹמְרֵי אֱמוּנִים,

וַיִּתְקַלְּסוּ בָם,לַעַג כַּמַּיִם שָׁתוּ

בִּשְׁאָט נֶפֶשׁ אוֹתָם תֹּעֵבוֹת שָׁתוּ,

וּשְׂחוֹק גָּדוֹל אָז פָּרַץ מֵאַפְסָיִם

וְהַיְּהוּדִי לִשְׁנִינָה וּלְזוּת שְׂפָתַיִם,

וַיְהִי עֶצֶב נִבְזֶה שְׁרֵקָה בָּעַמִּים

וְשֵׁם “יְהוּדִי” וּגְדוּפָה יַחְדָּו הָיוּ תַמִּים –

הַעַל זֹאת לֹא תֶאֱבַל נֶפֶשׁ וְתָשׁוּחַ,

לֹא יִתְחַמֵּץ לֵב, לֹא תִרְגַּז כָּל רוּחַ,

כִּי לִקְרוֹא בְשֵׁם “יְהוּדִי” – חֶרְפָּה וְקַלָּסָה

וְנוֹשְׂאָיו יִשָּׂרְטוּ כִּבְאֶבֶן מַעֲמָסָה!

כָּל עַם וָעָם בְּאֻמָּתוּ מִסְתַּפֵּחַ

בִּשְׁמָהּ יִדְגֹּל בִּתְהִלָּתָהּ מִשְׁתַּבֵּחַ:

זֶה שְׁמִי וְזֶה זִכְרִי, אֲנִי בֶּן עַמִּי! –

יִתְגַּדֵּל כַּשְׂדִּי יִתְאַמֵּר אֲרַמִּי

וּבִשְׁמוֹ “יְהוּדִי”, זֶה עֵדוּת עֲדָתוֹ

בּוֹ נִסְמָן נִסּוֹ, אוֹת דָּתוֹ יְחִידָתוֹ –

בּוֹ יֵבוֹשׁ יַעֲקֹב, מִפְּנֵי שְׁמוֹ הוּא נִחַת,

פֶּן יִשְׁמַע צָר וְרָגַז וְשָׂחַק וְאֵין נָחַת!


אַךְ מִי הֵסִית צָר, מִי אוֹיֵב הִפְגִּיעַ?

מִי בְּפִיהֶם שָׂם דָּבָר עָתָק הִבִּיעַ?

לֹא עָלָה עַל לֵב, לֹא חָשַׁב אִישׁ, לֹא דִמָּה

כִּי שֵׁם עָם – קָלוֹן לָשֵׂאת עָלָיו כְּלִמָּה

וּבְעוֹכְרֵינוּ אַחֵינוּ – הַמַּשְׂכִּילִים

הַשֹּׁגִים בְּזָרִים וְאֶת עַמָּם מַשְׁפִּילִים,

רָעָתָם הֵקֵרָה לִפְרֹץ הֵחִישָׁה

וְעֵדָה רָעָה הַגּוֹי כֻּלּוֹ הִבְאִישָׁה,

כֵּן בִּכְרָה קַלָּה דְּרָכֶיהָ מְשָׂרֶכֶת

עַל כָּל הָעֵדֶר בַּלָּהוֹת הֹפֶכֶת!


הָרָעָה הַזֹּאת רָעוֹת רַבּוֹת תַּהַר

וּבְאַחֲרִית הַיָּמִים תְּחוֹלֵל סַעַר –

רָזִי לִי, רָזִי לִי – הָהּ, חָזוּת קָשָׁה

רָעָה גְדוֹלָה נִשְׁקָפָה רָעָה חֲדָשָׁה:

עֵת יִרְבּוּ הֲמוֹנִים הָאָרֶץ יִסְחָרוּ

וְכֻתְּתוּ גוֹי בְּגוֹי וּלְהָרֵעַ יִתְחָרוּ

וְאֹכְלֵי עַמִּי בִּימֵי רְעָבוֹן יֶהֱמָיוּ

לְאָכְלוֹ בְּכָל פֶּה כְּפֶרֶץ רָחָב יֶאֱתָיוּ

וּלְמַעַן לֹא יֶאְשָׁמוּ סוֹד יַעֲרִימוּ

וַעֲלִילוֹת דְּבָרִים לִבְנֵי עַמִּי יָשִׂימוּ

לִזְרֹעַ שִׂנְאָה וּמְדָנִים לִשְׁלוֹחַ

וִיאַזְּרוּ זִיקוֹת, אֵשׁ קִנְאָה לִקְדֹּחַ –


וְהַיְּהוּדִים הַנֶּחֱשָׁלִים יִבָּהֵלוּ

וּמִן הַמַּשְׂכִּילִים רַבִּים יִכָּשֵׁלוּ

וְחִלְּלוּ הַמִּקְדָּשׁ הַמָּעוֹז וְהֵסִירוּ

קָדְשֵׁי יִשְׂרָאֵל וּבְזָרִים יָמִירוּ

כָּל יְרֻשַּׁת פְּלֵיטָה מְבַעֲרִים וּמְטַאְטְאִים

בְּתִפְאֶרֶת אָבוֹת מַלְעִיבִים וּמִתַּעְתְּעִים

לְמַעַן יֹאמְרוּ בַגּוֹיִם: הִשְׂכַּלְתֶּם,

אֵלֵינוּ נִדְמֵיתֶם אֵלֵינוּ נִמְשַׁלְתֶּם!

וְכַאֲשֶׁר יְעַנּוּ אוֹתָם כַּן יִפְרֹצוּ

חוֹמַת בַּת צִיּוֹן, עַד הַיְסוֹד יִתֹּצוּ –

וּבְהִכָּשְׁלָם עֵזֶר מְעָט יֵעָזֵרוּ

וִינַחֲשׁוּ וְיַחְלְטוּ כִּי בַאֲשֶׁר הֵפֵרוּ

בְּרִית אָבוֹת, בְּרִית עָם, עַל כֵּן נוֹשָׁעוּ

וּבְעֵינֵי עַם הָאָרֶץ חֵן מָצָאוּ, –

מִמַּרְשִׁיעֵי בְרִית אֵלֶּה תֵּצֵא חֲנוּפָה:

הֵיטֵב דֶּרֶךְ, בַּקֵּשׁ אַהֲבָה בְּנֶפֶשׁ חֲרוּפָה,

וְהַמַּשְׂכִּילִים יַזְהִירוּ – יַרְבּוּ הִזַּכּוֹת

וְנִלְווּ עֲלֵיהֶם רַבִּים בַּחֲלַקְלַקּוֹת –

וּבְבֵית יִשְׂרָאֵל אֶרְאֶה שַׁעֲרוּרִיָּה,

קֹדֶשׁ וְצָבָא מִרְמָס, צִיּוֹן אֶרֶץ נְשִׁיָּה

וְעַמִּי תְלוּאִים עַל כְּנַף שִׁקּוּצִים מְשׁוֹמֵם

אֶל עַל יִקְרָאֻהוּ יַחַד לֹא יְרוֹמֵם –


קוֹל מֵהֵיכָל קוֹל שַׁדַּי בִּלְבָבוֹ יַהַם

וּלְמִבְצְרֵי מָעֻזִּים עַם נֵכָר יַעַשׂ!

וּבָעֵת הַהִיא כִּי נִשָּׂא וְרָם לְבָבוֹ

כִּי בְשַׁלְוָה בְּמִשְׁמַנֵּי מְדִינָה יָבוֹא

פִּתְאֹם – וְרוּחַ מַשְׁחִית בְּנֵפֶץ וָפָרֶץ

בָּא מִן הַמַּעֲרָב עַל פְּנֵי כָּל הָאָרֶץ,

עַל בֵּית יְהוּדָה יִשְׁתָּעֵר יִשְתָּרֵעַ

מִמַּסַּד עַד הַטְּפָחוֹת לְקַעֲקֵעַ

וְהָלַךְ בְּסַעֲרוֹת תֵּימָן, כַּשֹּׁאָה יַעַל

יַזֶּה גוֹיִם רַבִּים וְהִשְׁקָם סַף רַעַל

מִבְּאֵר שַׁחַת, מִן הַמַּיִם הַזֵּידוֹנִים

וּבְעֵמֶק הֶחָרוּץ הֲמוֹנִים הֲמוֹנִים…

עַל עַם מְמֻשָּׁךְ וּמוֹרָט, עַם הַגּוֹלָה

יִתְגָּעֲשׁוּ יִתְהֹלְלוּ בַּהֲמֻלָּה גְדוֹלָה

בְּרַעַם וּבְרַעַשׁ יִפָּקֵד בְּיָד רָמָה

וּבְיָמִים אֲחָדִים יִשָּׁבֵר – וְלֹא בְמִלְחָמָה.

וּזְרֹעוֹת הַשֶּׁטֶף יָפֹזּוּ בַכֹּחַ

לִרְעוֹץ וּלְרוֹצֵץ לַהֲדֹף לִנְדֹּחַ

עַם עֵבֶר, עַם עִוֵּר, עַל זָרִים עָגָבוּ

עַם עֵבֶר, עַם עֶבְרָה, כְּמוֹ זָר נֶחְשָׁבוּ –

וְהָיְתָה עֵת צָרָה כָּמֹהָ לֹא נִהְיָתָה

וְאֵין אֶחָד מִתְחַזֵּק לָקוּם לִישׁוּעָתָהּ.


עַל עַם נִבְלָע, לֹא מָעֳמָד לוֹ, לֹא אָרֶץ,

מִי יָחוּס מִי יַחְמֹל לַעֲמֹד בַּפָּרֶץ?

צֹאן אֹבְדוֹת עַמִּי – לֹא עַם כִּי אִם נְפָשׁוֹת, –

צֹאן קָדָשִׁים בְּתוֹךְ צֹאן אָדָם נִפְרָשׁוֹת,

מִי יָנוּד לַצֹּאן יִשְׁגֶּה בְּכָל הֶהָרִים?

מִי יַחְמֹל מִי יָשִׁית לִבּוֹ לַעֲדָרִים

אֲשֶׁר קוֹנֵיהֶם יַהַרְגוּם וְלֹא יֶאְשָׁמוּ

וּבְלֹא הוֹן יִמָּכְרוּ לְמַעַן יִתָּמוּ!

מִי יֶאֱסֹף צֹאן הַהֲרֵגָה הַנִּדַּחַת

אִם צָמְקוּ שָׁדֶיהָ וּבְצַמְרָהּ קָרַחַת?

אוֹתָהּ יִרְעוּ קָדְקֹד כִּי מִרְעֶה רֹמֶסֶת

יֶהֶלְמוּהָ כִּי מִשְׁקַע מַיִם רֹפֶשֶׁת. –

וְעַמִּי נִגַּשׂ וְנַעֲנֶה יֹרֵד בַּבֶּכִי

מֵהַר אֶל גִּבְעָה מִדֶּחִי אֶל דֶּחִי,

הוֹלֵךְ וּבוֹכֶה עַם זָקֵן, עַם סָכָל,

לָתוּר לוֹ מְנוּחָה עַד הֵנָּה לֹא שָׂכָל.

עַם זָקֵן עַם עוֹלָם בּוֹכֶה כַיֶּלֶד

אֵין אָב וָאֵם, אֵין אָח, עָזוּב בַּחֶלֶד.

סֻגַּר כָּל בַּיִת לִמְצֹא פֶתַח יֵלֶא –

וּמִי הֵסֵב בְּרָעָתוֹ, מִי פָעַל כָּל אֵלֶּה?

הֲלֹא רֹב עֲוֹנוֹ כִּי אָהַב לָנוּעַ

מְרוּצָתוֹ רָעָה וְהוּא לְשִׁמְצָה פָרוּעַ


מִמְּקוֹמוֹ שֵׂעֲרוּהוּ לֹא אַחַת וּשְׁתַּיִם

וַיְבַקֵּשׁ מָנוֹחַ בִּקְצֵה הַשָּׁמַיִם

וּלְאֶרֶץ אֲבוֹתָיו לֹא נָשָׂא עֵינַיִם

לֹא זָכַר צִיּוֹן שָׁכַח יְרוּשָׁלַיִם

עַם לֹא בִינוֹת עָלֶיהָ הִשְׁלִיךְ נַעַל

וּמִכַּסְפּוֹ וּזְהָבוֹ נָתַן לַבַּעַל

אֵלָיו הִשְׁתַּחֲוָה לוֹ שַׁעַר לִפְתֹּחַ

תֵּת לוֹ חֲנִינָה מִפְלָט וּמָנוֹחַ

וּמִכֹּחוֹ כִי שִׁחֵד וַיְחָרֶף נֶפֶשׁ

וַיִּקֶן לוֹ זָוִית בִּמְקוֹם הָרֶפֶשׁ

עַל כֶּסֶף מִכְרֵהוּ עוֹד שָׁת נוֹסָפוֹת:

חֶמְדַּת יִשְׂרָאֵל, הַסְּגֻלּוֹת הַיָּפוֹת,

אוֹתָן זִבְחֵי תוֹדָה הִקְרִיב וַיַּעַל

עַל מִזְבְּחוֹ לַבּשֶׁת מִזְבַּח הַבַּעַל,

וְהַבּשֶׁת אָכְלָה כָּל יְגִיעַ אֲבוֹתָיו

וַיְנַתֵּק מוֹסְרוֹתָיו הִשְׁלִיךְ עֲבוֹתָיו

וּבְבֹא יְמֵי הַפְּקֻדָּה כִּי מָלְאָה סְאָתוֹ

קָאַתְהוּ הָאָרֶץ וְהִנּוֹ בְּרָעָתוֹ,

יָחֹג יָנוּעַ – לֹא יַיִן וְלֹא שֵׁכָר –

אֶרֶץ חֲדָשָׁה יָתוּר, אַדְמַת נֵכָר,

כִּי לֶאֱהֹב נָכְרִיּוֹת שָׁפַל וַיֵּזֶל

וְהוּא תֹעֶה אֵצֶל הָאֶבֶן הָאָזֶל…


וּבָעֵת הַהִיא עֵת מְהוּמָה וּמְבוּכָה

רַבִּים יָקִיצוּ מִתַּרְדֵּמָה הַנְּסוּכָה

צִיּוֹן יִשְׁאָלוּ, לִנְתִיבוֹת עוֹלָם,

“שׁוּבוּ וְנָשׁוּבָה!” – כַּיָּם יֶהֱמֶה קוֹלָם,

"אֶל אִמֵּנוּ, אֶל עַמֵּנוּ – לְקַדְמָתֵנוּ!

אֶל צִיּוֹן חֶמְדָּתֵנוּ – עַל אַדְמָתֵנוּ!"

וְרַבִּים מִיְּשֵׁנֵי אֶרֶץ נְשִׁיָּה יָקִיצוּ

לְחַיֵּי עוֹלָם יָעִירוּ בָּעָם יָאִיצוּ

לָתֵת גְּאֻלָּה לָאָרֶץ חַיִל לַחֵלֶךְ

מָעֳמָד לַנּוֹדֵד וּמְנוּחָה לַהֵלֶךְ

נֶפֶשׁ לְגַבֵּי חֹמֶר וּלְבָשָׂר רוּחַ

אֶל נֵר יִשְׁרָאֵל אוֹר יְקָרוֹת זָרוּחַ

לִבְזֹה נֶפֶשׁ לִמְתָעֵב גּוֹי וָעֶבֶד

תִּפְאֶרֶת אָדָם כְּבוֹד גּוֹיִם וָזֶבֶד. –

קוֹל קוֹרֵא בַכֹּחַ כֹּרֶה אָזְנַיִם

וּלְנֵס צִיּוֹנָה יֵאָסְפוּ מַחֲנַיִם

שֵׁם צִיּוֹן וּשְׂפָתָהּ כָּל אֹזֶן שֹׁמַעַת

בָּמוֹ יְשׁוֹטְטוּ רַבִּים וְתִרְבֶּה הַדַּעַת

וּבְסוֹד עַמִּי הַבָּאִים לִמְצֹא חֵקֶר

יִתְעַנְּגוּ לְמַרְאֵה הָעֶרֶב וְהַבֹּקֶר:

מְעוּף צוּקָה מְנֻדָּח הֵאִיר מִמִּזְרָח

עֵץ יָבֵשׁ עֵץ שָׁפֵל מִתְעָרֶה כְּאֶזְרָח –


וּבָעֵת הַהִיא – עַל זֹאת, הָהּ, אֵילִילָה –

בְּעֵת לַעֲשׂוֹת, עֵת חַיָּה רַבַּת עֲלִילָה,

יִנַּשְּׂאוּ פָּרִיצֵי עַמִּי וְהִרְשִׁיעוּ

אֶת בּוֹנֵי צִיּוֹן וַעֲלֵיהֶם יָרִיעוּ

כִּי יִשְׁעוּ בַהֶבֶל, חֲלֹמוֹת יַחֲלֹמוּ,

יֶחֱזוּ שָׁוְא וּמַדּוּחִים כָּזָב יִקְסֹמוּ:

יִשְׂרָאֵל לֹא עָם – הֵמָּה יִתְּנוּ קוֹלָם –

שֶׁכְּבָר מִהְיוֹת גּוֹי שָׁבַת וְחָדַל לְעוֹלָם,

וּמִי שִׁיבַת צִיּוֹן עַל רוּחוֹ יַעַל

אַךְ חוֹשֵׁב רָעָה הוּא יוֹעֵץ בְּלִיַּעַל

לָשׁוּב הַבְאִישֵׁנוּ בְּיוֹשֵׁב הָאָרֶץ

עַל חֵטְא לִסְפּוֹת מֶרִי עַל אָוֶן – פָּרֶץ. – –

כֹּה יֶהֱמוּ יַבִּיעוּ מַשְׂכִּילֵי הָרַבִּים

בְּנֵי הַתַּעֲרֻבוֹת, חֲבוּרֵי עֲצַבִּים,

עוֹבְדֵי הַבַּעַל וּבוֹ שֵׁם מוּסַבִּים

וְהֵם נוֹשְׁקֵי הַיָּד תְּדוּשֵׁם בָּעַקְרַבִּים

וּבְנֵצַח יִשְׂרָאֵל תָּהֳלָה יָשִׂימוּ

אַחֲרִיתוֹ יַכְרִיתוּ וּנְבִיאָיו יַכְלִימוּ

יִמְאֲסוּ בְּאֶרֶץ אָבוֹת, צִיּוֹן לָמוֹ צָיוֹן

וְכָל הַדְּבָרִים וְכָל הַחִזָּיוֹן – קִיקָיוֹן. –

אִם אָמְנָם כָּל דִּבְרֵיהֶם יִשָּׂא רוּחַ

וּדְבַר אֱלֹהֵינוּ יָקוּם – לִבִּי בָטוּחַ


חָזוֹן לַמּוֹעֵד יָבֹא – אִם יִתְמַהְמֵהַּ –

וְנִצְדַּק קֹדֶשׁ, הַשָּׁפָלָה הַגְבֵּהַּ

וְהֵקִיץ הַקֵּץ – אַשְׁרֵי הַמְחַכֶּה וְיַגִּיעַ –

וְעִם “עֲנָנֵי שְׁמַיָּא” יָבֹא מוֹשִׁיעַ

לַחְתֹּם חָזוֹן צֶדֶק עוֹלָמִים לְהָבִיא

יֹאמַר לַצָּפוֹן: תֵּנִי, בָּנַי לִי הָבִי!

וְהָיָה לָאָב אֶל הַשָּׁבִים מִן הַשֶּׁבִי

וְעָפוּ כָעָב אֶל הַמִּזְרָח וְאֶל הַצֶּבִי

לְשׁוֹמְרֵי אֱמוּנִים שַׁעֲרֵי צֶדֶק לִפְתֹּחַ

בְּשֶׁמֶן שָׂשׂוֹן קֹדֶשׁ קָדָשִׁים לִמְשֹׁחַ –

דְּבַר אֱלֹהֵינוּ יָקוּם, בְּזֹאת אֲנִי בֹטֵחַ,

כָּרָשׁוּם בִּכְתַב אֱמֶת לַקֵּץ יָפֵחַ,

אַךְ רָקָב בַּעֲצָמַי, מַכָּה אֲנוּשָׁה,

אֵלֶּה בָנִים כֶּחָשִׁים, מִצְחָם נְחוּשָׁה,

לֵאמֹר לְעַם נִבְחָר: מִכָּל עָם הוּא שָׁפֵל

וּנְבִיאֵנוּ חָזוּ לָנוּ שָׁוְא וְתָפֵל,

לֵאמֹר לְעַם עוֹלָם כִּי יֹאבַד בַּגּוֹלָה

וּמִקְוֵה יִשְׂרָאֵל – חֲטָאָה גְדוֹלָה!

וְכָזֹאת יֹאמְרוּ לֹא זָרִים לֹא צָרִים,

אַחַי בְּנֵי עַמִּי, בְּנֵי צִיּוֹן הַיְקָרִים! –

וּבְעֵת צָרָה עַתָּה כִּי רַבִּים הֵחֵלּוּ

לָשׁוּב אֶל עַמָּם וּלְאוֹר חָדָשׁ יְיַחֵלוּ


כִּי יִתֵּן אֱלֹהִים לַעֲדָתוֹ הַשְּׁדוּדָה

יָתֵד בִּמְקוֹם קָדְשׁוֹ גָּדֵר בִּיהוּדָה

וְהִנֵּה הֵם חַתִּים יֵצְרוּ צַעֲדֵי אוֹנָם

כִּי כֹהֲנֵי הַבַּעַל בַּהֲמוֹנָם וּשְׁאוֹנָם

מְבַהֲלִים אֶת הָעָם יִתְּנוּהוּ לְזַעֲוָה

כִּי שִׁיבַת צִיּוֹן מְשׁוּבָה, אַוַּת עָם עַוָּה!

עַל זֹאת אֵלְכָה שׁוֹלָל, אֶקְרָא אֶל אֵבֶל

עַל רוּחַ עִוְעִים רוּחַ טֻמְאָה וָתֶבֶל

הִיא הָיְתָה בְעַמִּי, בְּיוֹדְעָיו חָזוּ יָמָיו,

וּבְלֹא עָם יַקְנִיאוּהוּ לִפְזֹר עֲצָמָיו!

לָעָם לֹא נֶהִי מֵאָפַע פָּעֳלֵנוּ –

וּמָה אֵפוֹא אָנוּ? מַה גּוֹרָלֵנוּ?

הַאִם עִם אַבְנֵי הַשָּׂדֶה נוֹלַדְנוּ

וּצְרוּרִים בְּמַרְגֵּמָה לִהְיוֹת נוֹעַדְנוּ?

אוֹ לִמְשָׁרְתִים נִמְכַּרְנוּ, בִּכְרֹעַ בֶּרֶךְ

לִלְאֻמִּים בָּעֲרָפֶל הָאִיר הַדֶּרֶךְ?

וּלְאוֹר גּוֹיִם לִהְיוֹת בְּאֶרֶץ נָכְרִיָּה

חֲשֵׁכִים כִּי נֵלֵךְ עַד נֻכַּת שְׁאִיָּה? –

עַם הָיִינוּ – אַף נָשׁוּב לִהְיוֹת כָּכָה!

וְכֹל אֲשֶׁר לָעָם מִיַּד אֵל נִקָּחָה.

עַם – לֹא רַק בַּסֵּפֶר, לֹא רַק בָּרוּחַ,

עַם נַחֲלָה עַל אַדְמָתוֹ נָכוֹן בָּטוּחַ!


עוֹד יִשְׂמַח יִשְׂרָאֵל אֶל גִּיל כָּעַמִּים

וּבְנֵי אָב אֶחָד יַחְדָּו יִהְיוּ תַמִּים –

זֹאת אֱמוּנָתִי, זֹאת תְּפִלָּתִי, תִּקְוָתִי

וּבָהּ חַיֵּי רוּחִי, נַפְשִׁי, נִשְׁמָתִי!

הֲזֶה לִי פֶשַׁע, עֲוֹן מֶרֶד וָפֶרֶץ,

אִם לֹא אֶחְשֹׁב עַמִּי רֶמֶשׂ וָשֶׁרֶץ

וּכְכֹל הָעַמִּים עַל כָּל פְּנֵי הָאֲדָמָה

לִהְיוֹת גַּם הוּא אֶשְׁאַף – כִּי בְאַפּוֹ נְשָׁמָה!

אִם כֵּן מִשְׁפָּטְךָ, אַדִּיר מֵהַרְרֵי טֶרֶף!

הָבָה, פְּרִיץ חַיּוֹת, מַעֲשֶׂה יָדְךָ אַל תֶּרֶף!

לֹא אוֹסִיף, הִצְטַדֵּק לְזַכּוֹת מַאֲוַיָּי

קוּמָה אָכְלֵנִי – טוֹב מוֹתִי מֵחַיָּי!

– הֵרָגַע, בֶּן אָדָם! – הָאַרְיֵה קָרָא

וּלְרַגְלֵי דָנִיֵּאל בִּמְנוּחָה כָּרָע –

אַל תִּירָא אִישׁ חֲמֻדוֹת, לְךָ הָשְׁלַמְתִּי,

לִבַּבְתַּנִי בִדְבָרֶיךָ וְאוֹתְךָ רָחַמְתִּי,

אִם הִכְבַּרְתָּ מִלִּין מֶנִּי נִשְׂגָּבוּ

דֵּעוֹת נָשָׂאתָ כְּמוֹ זָר לִי נֶחְשָׁבוּ

אַךְ בִּדְבָרְךָ כָאֵשׁ לִבְּךָ צָרַפְתִּי

בָּחַנְתִּי כִלְיוֹתֶיךָ יָשְׁרְךָ חָשַׂפְתִּי

צֶדֶק לִי נִשְׁקָף מִטֹּהַר עֵינַיִם

ישֶׁר דִּבְרֵי אֱמֶת – מִמֶּתֶק שְׂפָתַיִם


עַתָּה אַל תִּשְׁתָּע, כִּסִּיתִיךָ בְּצֵל יָדִי,

הִתְאוֹשֵׁשׁ, – מִשְׁמֶרֶת אַתָּה עִמָּדִי,

וּלְסַפֵּר הוֹסִיפָה, בָּאֵר הוֹאִילָה

אֵיךְ מִכַּף צָרִים יַד מֶלֶךְ לֹא הִצִּילָה?

מַלְכִּי – עָנָה דָנִיֵּאל – לְעוֹלָם יֶחִי,

חוֹמָה הָיָה עָלַי לְחַלְּצֵנִי מִדֶּחִי

מִמִּתְקוֹמְמַי הוֹשִׁיעַ גִּבֵּר חֲיָלִים

אַךְ מַלֵּט לֹא יָכֹל מֵעֲצַת נִפְתָּלִים.

יֵשׁ דָּבָר גַּם מֶלֶךְ לֹא יוּכַל עֲשׂהוּ

כָּל תָּקְפּוֹ וּרְצוֹנוֹ יַעֲלוּ בַתֹּהוּ –

הַשָּׂרִים הִרְעִימוּ בִּזְעָקָה גְדוֹלָה:

– דָנִיֵּאל אִישׁ יְהוּדָה מִבְּנֵי הַגּוֹלָה…

לָמָּה – תָּמַהּ הָאַרְיֵה – דִּבְּרוּ כָכָה?

וְאַתָּה שַׁלִּיט רַב רֹדֶה בַמַּמְלָכָה!

הֲטֶרֶם נוֹדַעְתָּ מִי וָמָה אָתָּה?

וּמַה שֵּׁם עִירְךָ וְגָלוּתְךָ הֶעֱלוּ עָתָּה?

אָמְנָם – עָנָה דָנִיֵּאל – בְּזָדוֹן דִּבֵּרוּ

דָּבָר מָר שָׁלְחוּ, לֵב מֶלֶךְ עוֹרֵרוּ

כִּי לֵב יְהוּדִי בִּיהוּדָה, אֵלֶיהָ עֵינוֹ

בְּקִרְבּוֹ תָּלִין הַמְרוֹתוֹ, מַחְשְׁבוֹת אוֹנוֹ,

הִנֵּה כֵן זָעֲקוּ הָאֲנָשִׁים הָאֵל:

אִישׁ יְהוּדָה מִבְּנֵי הַגּוֹלָה, דָנִיֵּאל,

בַּמֶּלֶךְ יִתְקַלָּס וּבִשְׁאָט נֶפֶשׁ

יָשִׂים חֻקָּיו מִרְמָס מִצְוָתוֹ רֶפֶשׁ.

הֵן נִתְּנָה הַדָּת מוּדַעַת בַּמְּדִינָה

כִּי רַק אֵלֶיךָ יָבִיאוּ כָל תְּחִנָּה

וּלְגֹב הָאֲרָיוֹת עוֹבְרִים יוֹרִידוּ

וַעֲבָדֶיךָ הֶאֱזִינוּ וְלֹא הֵזִידוּ

וְרַק זֶה דָנִיֵּאל, הַיְּהוּדִי הַחֹזֶה

מִשְׂחָק לוֹ הַמֶּלֶךְ וּדְבָרוֹ בֹזֶה

כִּי לֹא אַחַת וּשְׁתַּיִם הַדָּת מְחַלֵּל

יוֹם יוֹם שָׁלשׁ פְּעָמִים לֵאלֹהָיו מִתְפַּלֵּל!

וַיִּתְחַלְחַל מַלְכִּי, פַּלָּצוּת בִּעֲתַתּוּ

הוֹדוֹ נֶהְפַּךְ עָלָיו צַלְמָוֶת כִּסַּתּוּ

עָרְמַת שׂטְנַי עַתָּה רָאָה וַיָּבֶן

לִכְלִי חֵפֶץ הָיָה בְּיַד חֹרְשֵׁי אָוֶן

וַיֵּצֶר לוֹ וַיִּתְמַרְמַר וַיַּצְרִיחַ

בַּעֲצֻמוֹת עֹז בִּדְבָרִים הוֹכִיחַ

כִּי לֹא עָלַי הַדָּת, בַּאֲשֶׁר יְהוּדִי אָנִי

וּכְבָר חָפְשִׁי בֶאֱמוּנָתִי עָשָׂנִי

לֹא לִי הִיא, לֹא עָלַי תִּטֹּף מִלָּתָהּ

זֹאת לֹא דִמָּה, עַל לִבּוֹ לֹא עָלָתָה, –

אָז נִסּוּ הַשָּׂרִים הַתְעֵב עֲלִילָה

סַפֵּר לַמֶּלֶךְ פָּרָשַׁת הַתְּפִלָּה


כִּי דְבַר בְּלִיַּעַל בָּהּ יָצוּק וָמַעַל

בָּבֶל סִיר רַחְצִי, עַל כַּשְׂדִּים אַשְׁלִיךְ נַעַל

וּמְגַמַּת פָּנַי לָשׁוּב יְרוּשָׁלַיְמָה

וְעַל זֹאת תְּפִלָּתִי נָשָׂאתִי שָׁמַיְמָה –:

– הֵן זֶה דָנִיֵּאל הַיְּהוּדִי הַגּוֹלֶה

עַל מַלְכּוֹ עַל אַרְצוֹ אֵינֶנּוּ חוֹלֶה,

הִיא לוֹ אַךְ מִקְלָט מַחֲסֶה לְעִתּוֹת בַּצָּרָה

לִשְׂבֹּעַ מִטּוּבָהּ וּלְהָרִיק אוֹצָרָהּ

וּבְקִרְבּוֹ אָרְבּוֹ לְעֵת שְׁמוּרָה וַעֲרוּכָה

לָנוּס יְרוּשָׁלַיְמָה שָׁם לַעֲשׂוֹת מְלוּכָה,

לִמְרֹד בְּךָ יֹאמַר הַיְּהוּדִי גִּדַּלְתָּ

בֶּאֱמוּנָתְךָ בוֹ עַל אַרְצְךָ הִמְשַׁלְתָּ,

מוֹרֵד הוּא בוֹגֵד, בּוֹ עָנְתָה תְפִלָּתוֹ

וּלְמַרְבֵּה הַמִּשְׂרָה כֵּן תִּגְדַּל אַשְׁמָתוֹ

מַהֲרָה הֲמִיתֵהוּ טֶרֶם יִפְרָץ פָּרֶץ

לְהַדִּיחַ כָּל עַמּוֹ וְעָלָה מִן הָאָרֶץ!

כִּשְׁמֹעַ הַמֶּלֶךְ כָּל דִּבְרֵי בָלַע

שִׁנְּנוּ שׂטְנַי, חִזְּקוּ פְנֵיהֶם מִסָּלַע,

וַיִּסְפֹּק אֶת כַּפָּיו וַיִּמְרֹט שַׂעַר

וַתִּבְעַר חֲמָתוֹ וַיִּשְׂעַר שַׂעַר

וַיַּהַס אֶת כָּל עֲדַת הַנִּרְגָּנִים

וַיָּרֶב אֶת הַחֹרִים וְאֶת הַסְּגָנִים


עַל שֶׁלְחָם פִּיהֶם וּלְשׁוֹנָם חָרָצוּ

וּבְחֶרְפַּת נָבָל בְּחִיר יֹעֲצָיו נָאָצוּ

וּבְמִישׁוֹר הוֹכִיחַ אֶת לֶכְתִּי בְתֻמִּי

לַחְמֹד אֶרֶץ אֲבוֹתַי לֶאֱהֹב לְאֻמִּי

וּלְחַכּוֹת לַיּוֹם כִּי אֱלֹהִים יְחוֹנֵן

וּבְמִשְׁפָּט יִפְדֶּה בִּצְדָּקָה יְכוֹנֵן. –

נַפְתּוּלֵי אֱלֹהִים נִפְתַּל מְלִיצִי רֵעִי

וַיֵּאָבֵק עִם צוֹרְרַי וַיְזַכֶּה דֵעִי,

עִמָּם יִתְוַכַּח אֲמָרָיו לָמוֹ יֶרֶב

גָּנוֹן וְהַצֵּל לֹא נִלְאָה עַד הָעֶרֶב

וּבְאִמְרֵי ישֶׁר עִם הַשָּׂרִים שָׂרָה

כִּבְּסַנִי מֵעָוֹן חַטָּאתִי סָרָה

וּמְרִיבַי חַתּוּ כִּי יָדָם קָצָרָה

לַחְרֹךְ צֵידָם וּמְזִמָּתָם נִבְצָרָה

וַיִּשְׂתָּעֲרוּ אֶל הַמֶּלֶךְ בְּתַרְמִיתָם

לַהֲלוֹךְ עָלָיו אֵמִים לָתֵת חִתִּיתָם,

רָגְשׁוּ וַיִּרְעָשׁוּ דִּבְרֵיהֶם הֶעְתִּירוּ

וּבְרוּחַ כַּבִּיר לִלְשׁוֹנָם הִגְבִּירוּ:

– מֶלֶךְ גָּדוֹל! כָּמוֹךָ גַם אֲנַחְנוּ

כִּי נָקִי הָאִישׁ לָדַעַת נוֹכַחְנוּ,

אַךְ מִשְׁפָּט בּוֹ לַעֲשׂוֹת בְּחָזְקָה נַפְגִּיעַ

פֶּן יְבֻלַּע לַמֶּלֶךְ אִם מְחֻקָּק יַפְרִיעַ,


הֵן בְּמָדַי וּפָרַס הַדָּת מוּדַעַת

כִּי בִכְתוֹב הַמֶּלֶךְ וְנַחְתּוֹם בְּטַבַּעַת

לְקַיֵּם דָּבָר בְּכֹל אֲשֶׁר יָשִׂים מִשְׁטָרוֹ

לֹא יְשֻׁנֶּה, לֹא יַעֲבוֹר – לֹא יַחֵל דְּבָרוֹ,

וּבָאָרֶץ זֶה מְעָט הָרוּחוֹת הֲשִׁכֹּנוּ

וּלְרֹאשׁ מַלְכֵּנוּ מֵרֹגְנִים סַכֹּנוּ

בְּצִפֹּרֶן שָׁמִיר עַל לִבָּם חַקֹּנוּ

לִשְׁמוֹר חֻקֵּי אֶרֶץ יַחְדָּו יִכֹּנוּ,

עַתָּה אִם יוּפַר חֹק אִם שֶׁמֶץ יִגָּרַע

וְנִקְלוּ הַחֻקִּים כֻּלָּם – עָם יִפָּרַע,

וּכְפֶרֶץ נוֹפֵל פָּרִיצִים יֶהֱמָיוּ

עַד שֶׁהַמֶּלֶךְ בִּמְסִבּוֹ יֶאֱתָיוּ

וּמְנוּחָתוֹ כָּבוֹד וּשְׁלוֹם הָאָרֶץ

לַעֲרוֹב לֹא נוּכַל עֵת יִגְדַּל הַפָּרֶץ!

לָכֵן, מַלְכֵּנוּ! מַלְכוּתְךָ הַצִּילָה

וּמִשְׁפָּט חָרַצְתָּ לַעֲשׂוֹת הוֹאִילָה.

יִתְאַפְּקוּ רַחֲמֶיךָ מִתֵּת חֲנִינָה

יוּמַת נָא הַיְּהוּדִי לְטוֹבַת הַמְּדִינָה!

יוּמַת הַיְּהוּדִי – וְהִשְׁאִיר אַחֲרָיו בְּרָכָה

שָׁלוֹם לַמֶּלֶךְ שַׁלְוָה לַמַּמְלָכָה!

כֵּן הָמוּ רָעָשׁוּ וּמַלְכִּי רָאָה

כִּי יֹעֲצֵי בְלִיַּעַל חוֹשְׁבִים עָלַי רָעָה,


הֵבִין כִּי הַוַּת נַפְשָׁם הִיא וַיְעַבְּתוּהָ

מִרְמַת בִּטְנָם בְּאֵזוֹר צֶדֶק יְכַסּוּהָ

אַךְ הַצֵּל לֹא יוּכַל, עָלָיו חָזָקוּ

דִּבְרֵי הָאֲנָשִׁים לְעֵינֵי כֹל יִצְדָּקוּ

וְהָאֲנָשִׁים קָשִׁים בִּזְדוֹנָם לֹא יַרְגִּיעוּ

וַחֲתַת בָּהּ יֵרְאוּהוּ עָדָיו יַגִּיעוּ;

חֹק אִם יוּפַר וְיַד הַמֶּלֶךְ בַּמַּעַל

בָּעֹז יִתְאַמְּצוּ אֵל בְּנֵי הַבְּלִיַּעַל

וּנְפָשׁוֹת יַרְעִישׁוּ לִפְרֹעַ פְּרָעוֹת

לְמַעַן יִצְדָּקוּ כִּי נִבְּאוּ הָרָעוֹת –

רָאָה מַלְכִּי נִכְלֵי זֵדִים אֲרוּרִים

וּלְהַצֵּל לֹא יוּכַל וַיֵּבְךְ תַּמְרוּרִים

וּבְנֶפֶשׁ מָרָה בְּלֵב נִקְרַע לִגְזָרִים

לַעֲשׂוֹת מִשְׁפָּט כָּתוּב נֵאוֹת לַשָּׂרִים

וַאֲנִי נִקְרֵאתִי וּבְבֹאִי נִבְעַתִּי

אֲשֶׁר נִדְבָּר וַאֲשֶׁר נִגְזַר שָׁמַעְתִּי…

וַיֹּאמֶר אֵלַי הַמֶּלֶךְ: דָּנִיֵּאל!

קָצְרָה יָדִי מִפְּדוּת, יוֹשִׁיעֲךָ הָאֵל,

אֱלֹהֶיךָ בּוֹ תִדְבַּק בֶּאֱמֶת וּבְתָמִים

וּבְאֵמוּן תַּעַבְדֶנּוּ כָּל הַיָּמִים

בְּאֶבְרָתוֹ יָסֶךְ־לָךְ בִּמְעֹנוֹת אֲרָיוֹת

וּבְשַׂר חֲסִידוֹ לֹא יִתֵּן מַאֲכָל לַחַיּוֹת.


הֵן עָלָיו הִנְּךָ לַמָּוֶת לָקוּחַ

וּמָגֵן לְךָ יִהְיֶה לִבִּי בָּטוּחַ.

הַלַּיְלָה הַזֶּה לֵיל שֹׁאָה וּמְשֹׁאָה

גַּלְמוּד בּוֹ אֶהִי, לֹא יְרֻנָּן לֹא יְרֹעָע,

בְּדִמְעוֹתַי בְּאַנְחוֹתַי רְגָעָיו אֶסְפֹּר

וּמִפְעֲלוֹת אֱלֹהִים לִרְאוֹת אֶצְפֹּר

יַעֲבוֹר לֵיל עֲבָרוֹת יִבָּקַע שַׁחַר

אֶשָּׂא כְנָפָיו וּפָנֶיךָ אֲשַׁחֵר. –

וּבְדַבֵּר הַמֶּלֶךְ – וַיִּשָּׁקֵנִי

אוֹתִי נִהֲגּו חשֶׁךְ – וּפֹה הִנֵּנִי.

יָגֵל כְּבוֹדִי – הָגָה הָאַרְיֵה וַיִּחַדְּ –

כִּי בִקְהַל מְרֵעִים הַמֶּלֶךְ לֹא יֵחַד.

עֹז מֶלֶךְ – מִשְׁפָּט, צֶדֶק תִּפְאַרְתּוֹ

דִּגְלוֹ אַהֲבָה, חֶסֶד וֶאֱמֶת עֲטַרְתּוֹ,

מִפִּי מֶלֶךְ עֶלְיוֹן לֹא תֵצֵא רָעָה

רַק הַשָּׂרִים סֹרְרִים, בְּנַפְשָׁם הָרְשָׁעָה

רַק הֵם בְּהַוָּתָם הַיְשָׁרָה יְעַקֵּשׁוּ

בְּקִנְאָתָם בְּשִׂנְאָתָם לִנְפָשׁוֹת יְנַקֵּשׁוּ

יַצִּיבוּ מַשְׁחִית בִּטְנָם מִרְמָה תָכִין

וּבְדַרְכֵיהֶם יַכְשִׁילוּ לַמְחוֹת מְלָכִין

כִּי כָל תֹּעֲבוֹתָם מִקִּרְבָּם יָצָאוּ

אֶל הַמֶּלֶךְ יִגֹּלּוּ בִּשְׁמוֹ יִקְרָאוּ. –


וּבְשֵׁם הַמֶּלֶךְ צָרֶיךָ הַשָּׂרִים

עַתָּה הִשְׁמִיעוּ וְנִשְׁלֹחַ סְפָרִים

כִּי דָנִיֵּאל הַיְּהוּדִי לִמְרֹד פָּרַץ

וּמִשְׁפַּט מָוֶת עָלָיו הַמֶּלֶךְ חָרַץ,

יִשְׁמְעוּ הֲמוֹנִים וְקָרְאוּ מִכָּל עֵבֶר:

“כֵּן הָעָם הַזֶּה – מָוֶת לִבְנֵי עֵבֶר!”

וִישָׁרִים יָשֹׁמּוּ, וְשָׁרְקוּ עַל הָרָעוֹת

וְהָיָה מֶלֶךְ הַכָּבוֹד לְמֶלֶךְ בַּלָּהוֹת – –

אַךְ הֵרָגַע, עַד אַרְגִּיעָה לְשׁוֹן שֶׁקֶר

קֵץ לַחשֶׁךְ, יֹאבַד לַיְלָה וְאָתָא בֹקֶר

תִּגָּל צִדְקָתְךָ וִיהִי לְעוֹלָם עִתֶּךָ

וּמַרְשִׁיעֵי צַדִּיק תַּצְמִית אֲמִתֶּךָ.

לָכֵן הִשָּׁקֵט וֶהְיֵה שַׁאֲנַן מִפַּחַד

וְחִכִּינוּ עַד הַבֹּקֶר בְּשָׁלוֹם יַחַד

הָאֵל מָגֵן לָךְ, כִּי לוּ בְיַד בְּנֵי אָדָם

מִגְּנוּךָ צָרֶיךָ – נִגְזַרְתָּ מִיָּדָם,

וּבְאָשְׁרְךָ הִסְגִּירוּךָ לְחַיְתוֹ יַעַר,

עַל זֹאת לֵאלֹהֶיךָ תּוֹדָתְךָ תַעַר!


בַּבּוֹר הָשְׁלָךְ הָס, חַיָּתוֹ שֹׁקֶטֶת

תּוֹכוֹ רָצוּף אַהֲבָה, שַׁלְוָה שַׁלֶּטֶת,

כַּחֲבֵרִים מַקְשִׁיבִים הָאֲרָיוֹת יִרְבָּצוּ

אֵין נַהַם וָמַשַּׁק – פִּיהֶם קָפָצוּ,


אֶל הַשָּׂב הַנֶּהְדָּר, כַּצִּנָּה עָטָרוּ,

בִּידִידוּת חֲרִישִׁית יַבִּיטוּ וְנָהָרוּ,

וְהוּא, דָּנִיֵּאל, בֵּינֵיהֶם עַל בִּרְכָּיִם,

מִתְפַּלֵּל – וְכַפָּיו פְּרוּשׂוֹת הַשָּׁמָיִם:

– אֱלֹהִים, נוֹרָא תְהִלּוֹת עֹשֵׂה פֶלֶא!

מִמַּחֲשַׁכִּים אֲהוֹדְךָ, מִבּוֹר כֶּלֶא,

בְּמוֹ אֹפֶל יוֹשֵׁב הַכְּרוּבִים תּוֹפִיעַ

וּבְבוֹר שָׁאוֹן אֶרְאֶה זֹהַר רָקִיעַ

כִּי אַתָּה, אֵל נָאוֹר! עֵינְךָ תֶחֱזֵנִי

וּבְתַחְתִּיּוֹת אֶרֶץ יְמִינְךָ תֹאחֲזֵנִי

מֵחֶבְיוֹן עֻזְּךָ עָלַי שַׂכּוֹתָ כַפֶּיךָ

וּבְתוֹךְ לְבָאִים אֶשְׁכְּבָה בְּסֵתֶר כְּנָפֶיךָ. –

מִשִּׁנֵי אֲרָיוֹת מִי הִשְׁלִיךְ טֶרֶף?

כְּפִירִים רָעֵבוּ אֵיךְ פָּנוּ לוֹ עֹרֶף?

אֵיךְ לַיִשׁ אֹבֵד לֹא יַעַט אֶל הַשָּׁלָל

וּבְנֵי לָבִיא יִתְפָּרְדוּ סְבִיבוֹת הֶחָלָל?

אֶשְׁתֹּלְלוּ אַבִּירֵי לֵב אָחוֹר נָסֹגוּ

מִגַּעֲרָתְךָ, אֱלֹהִים! חַתּוּ נָמֹגוּ

כִּי אַתָּה יוֹצְרָם וּלְפָעָלְךָ תּוֹצִיאֵם

אִם לְשֵׁבֶט אִם לְחֶסֶד לְמִשְׁפָּטְךָ תַּמְצִיאֵם.

שַׁלַּח פֶּרֶא חָפְשִׁי תְּשַׁלַּח חֲרוֹנְךָ

מֶתֶג וָרֶסֶן עֶדְיוֹ לִבְלֹם – רְצוֹנְךָ.


וּבִי כִי רָצִיתָ, אֵל אֱלֹהֵי אָבִי!

יִתְרֹעֵעַ אַרְיֵה יִלָּבֵב לָבִיא,

זוֹלְלֵי בָשָׂר מִלֵּאתָ רוּחַ טוֹבָה

מַאֲכַל תַּאֲוָה שַׂמְתָּ לָמוֹ תֹּעֵבָה –

נִפְלָאִים מַעֲשֶׂיךָ וְנַפְשִׁי יֹדַעַת

כִּי לְמַעַן שִׁמְךָ הַגָּדוֹל אוֹתָם תַּעַשׂ,

זֶה שִׁמְךָ בּוֹ אֶדְגֹּל, בּוֹ חַיַּי גָּאַלְתָּ,

יִוָּדַע בַּגּוֹיִם בְּעֻזְּךָ פָעַלְתָּ

כִּי גַם חַיָּה רָעָה לִטְרֹף נוֹצָרָה

שִׁנְּתָה דַרְכָּהּ, תַּאֲנָתָהּ הוּפָרָה,

כִּי עֶלְיוֹן הִנְּךָ יוֹצֵר כָּל חֹמֶר

עַל כָּל יֵצֶר וָקֶרֶב תִּגְזַר אֹמֶר;

עַל כֵּן, אֱלֹהִים! בִּישׁוּעָתְךָ אָגִילָה

כִּי בְחַרְתַּנִי וּכְבוֹד שִׁמְךָ אַגְדִּילָה

וַאֲנִי בְּרֹב חַסְדְּךָ, אֵל נוֹרָא עֲלִילָה!

אֶשְׁתַּחֲוֶה אֶכְרָעָה וָאֲבַקֵּשׁ תְּפִלָּה

כִּי עוֹד תַּפְלִיא חַסְדְּךָ וְיִגְדַּל עוֹד שְׁמֶךָ

שִׁמְךָ הַנִּקְרָא עַל עִירְךָ וְעַל עַמֶּךָ –

עִירְךָ נְוֵה קָדְשְׁךָ – מַהְפֵּכַת זָרִים,

עַמְּךָ שֵׁבֶט נַחֲלָתְךָ – חֶרְפַּת צָרִים,

עַל מָה יְדַכְּאוּ עַמְּךָ, נַחֲלָתְךָ שֹׁסִים,

וְדַם עֲבָדֶיךָ לְשִׁכָּרוֹן צַמִּים שׁוֹתִים?


עַל מָה יִנְאַץ אוֹיֵב וּמְעַנִּים עֹשִׂים?

עַל כִּי עַמְּךָ הֵמָּה וּדְבָרְךָ עֹשִׂים!

עָלֶיךָ בָּנֶיךָ יִשְׂאוּ יִסְבֹּלוּ

כִּי בְרִיתְךָ יִנְצֹרוּ, בְּשִׁמְךָ יִדְגֹּלוּ –

תַּעֲשֶׂה נָא קִנְאָתְךָ, אֵל גָּדוֹל וּמְהֻלָּל!

חֲמוֹל עַל שֵׁם קָדְשְׁךָ בַּגּוֹיִם מְחֻלָּל,

אֶל מִקְדָּשְׁךָ כִי נִחַל פָּנִים הָאִירָה

מֵעַמְּךָ כִי נוֹחַל חֶרְפָּה הָסִירָה.

עַל חַיְתוֹ יַעַר רוּחַ חֵן שָׁפַכְתָּ

לַאֲרָיוֹת וּלְבָאִים לֵב אַחֵר הָפַכְתָּ, –

גַּם עַל חַיְתוֹ גוֹי, כָּל אֹכְלֵי עַמֶּךָ

יֵעָרֶה נָּא הָרוּחַ מִמְּרוֹמֶיךָ

רוּחֲךָ, אֱלֹהִים! רוּחַ אֱמֶת וָדַעַת

לַעֲקֹר לֵב אֶבֶן, לֵב בָּשָׂר לָטַעַת

וּמָלְאָה הָאָרֶץ דֵּעָה, אֱמֶת בַּטּוּחוֹת,

הַשְׂכֵּל וְיָדוֹעַ אוֹתְךָ אֱלֹהֵי הָרוּחוֹת!

וּמִשְׁפַּט עַמְּךָ בָּחַרְתָּ מֵעוֹלָם

יַכִּירוּ וְיֵדְעוּ קְטַנָּם וּגְדוֹלָם

מִשְׁפָּט הֻסַּג אָחוֹר יָשׁוּב עַד צֶדֶק

וּשְׁבָרָיו יֵרָפְאוּ וִיחֻזַּק בֶּדֶק. –

וּבַעֲצָמוֹת הַיְבֵשׁוֹת עַצְמוֹת יְהוּדָה

פְּחָה נָא הָרוּחַ וְיֵעָשׂוּ אֲגֻדָּה


אִישׁ בְּאָחִיהוּ יְדֻבָּקוּ, לֹא עוֹד יֵחָצוּ

בַּאֲשֶׁר רָצְתָה נַפְשְׁךָ כֻּלָּם יֶחְפָּצוּ

הַצֵּב גְּבוּל אַלְמָנָה וְנָוָהּ יִוָּסֵד

וּמִמֶּנּוּ פִנָּה מִמֶּנּוּ יָתֵד…

לָתֵת בְּצִיּוֹן תְּשׁוּעָה לְיִשְׂרָאֵל תִּפְאֶרֶת

וְכַנָּה נָטְעָה יְמִינְךָ – לְגֶפֶן אַדֶּרֶת

אֲסִירֵי הַתִּקְוָה יָשׁוּבוּ לְבִצָּרוֹן

וְשָׁב לָמוֹ הֲדַר הַכַּרְמֶל וְהַשָּׁרוֹן

וּמְדֻכָּאִים לִפְנֵי עָשׁ כִּמְסֹס נֹסֵס

יַגְבִּיהוּ כַנֶּשֶׁר וְהָיוּ נֵס לְהִתְנוֹסֵס,

אָז יִגְדַּל שִׁמְךָ, אֱלֹהִים עֹשֵׂה פֶלֶא

יֹאמְרוּ בַגּוֹיִם: הִגְדִּיל לַעֲשׂוֹת עִם אֵלֶּה;

אָז לִרְאוֹת תִּפָּקַחְנָה עֵינֵי עִוְרִים

כִּי גָדוֹל הָאֱלֹהִים, אֱלֹהֵי הָעִבְרִים,

וּלְאוֹר גָּדוֹל יִהְיֶה עַמְּךָ, עַם סְגֻלָּה,

נֹגַהּ זַרְחָךּ יָפִיץ בַּתֵּבֵל כֻּלָּהּ,

בִּימִינוֹ אֵשׁ דָּת שָׁמַיִם לִנְטֹעַ

וּמִיָּדוֹ קַרְנַיִם אוֹר לִזְרֹעַ…

הַס, אוֹר נָגַהּ עָלַי כְּמַרְאֵה הַקֶּשֶׁת

מֵחוּטֵי זָהָב לְרַגְלַי פָּרַשׂ רֶשֶׁת

מַה קַּוֵּי זֹהַר יִתְהֹלְלוּ יִשְׂחָקוּ

וּלְיַרְכְּתֵי בוֹרִי אַדְנֵי פָז הוּצָקוּ!


הֶחָזוֹן נִרְאֶה אֵלַי וְשַׁדַּי יַעֲנֵנִי

כִּי שָׁמַע תְּפִלָּתִי אֹתוֹת יַרְאֵנִי? –

הֶאָח, אוֹר גָּדוֹל, רוּחַ צַח יָפוּחַ

צֵל אֲנָשִׁים אֶרְאֶה – הַבּוֹר פָּתוּחַ!

הִנֵּה קוֹל קוֹרֵא בִּמְרִירוּת צֹרֵחַ –

הַקּוֹל – הַס – קוֹל מַלְכִּי הוּא – מִתְיַפֵּחַ:

– אֲהָהּ, דָּנִיֵּאל! הָאָדָם הַמַּעֲלָה

דָּנִיֵּאל, עֶבֶד אֵל חָי, מַה מִּלַּיְלָה?

אֱלֹהִים צַדִּיק בֹּחֵן לִבּוֹת וּכְלָיוֹת

הֵהִצִּיל חֲסִידוֹ מִשִּׁנֵּי אֲרָיוֹת?

– יְחִי מַלְכִּי לָנֶצַח! – דָּנִיֵּאל עָנָה –

אֱלֹהַי מִבְטַחִי סָבִיב לִי חָנָה

שַׁדַּי אֵל רֳאִי מַגְבִּיהִי לָשָׁבֶת

שָׁלַח אוֹרוֹ וַאֲמִתּוֹ בְּגֵיא צַלְמָוֶת

לִהְיוֹת לִי סֵתֶר מַלְאַךְ פָּנָיו דָּאָה

סָגַר פִּי אֲרָיוֹת, לֹא נָגְעוּ לְרָעָה

כִּי יוֹדֵעַ הָאֵל תְּמִימָיו לוֹ נֶאֱמָנוּ

וּלְפָנָיו יָשְׁרִי, תֻּמָּתִי, בִּי עָנוּ.

קֶרֶב אִישׁ וְלֵב עָמֹק עֵינוֹ צֹפִיָּה

לֹא רָאָה אָוֶן בִּי, בְּרוּחִי רְמִיָּה,

גַּם עִמְּךָ, מַלְכִּי, כֵּן תָּמִים הָיִיתִי

יָשָׁר לֹא הֶעֱוִיתִי, מִדְחֶה לֹא עָשִׂיתִי.


– בָּרוּךְ אֵל עֶלְיוֹן! – הַמֶּלֶךְ הֵרִיעַ

וּשְׂשׂוֹן לְבָבוֹ עַל פָּנָיו הוֹפִיעַ

וַיְצַו וּכְרֶגַע דָּנִיֵּאל הֹעָלָה

וַיֶּחֱרַד אֵלָיו הַמֶּלֶךְ בְּבֶהָלָה

בִּזְרֹעוֹתָיו קִדְּמָהוּ וַיְחַבְּקֵהוּ

וַיֵּבְךְּ עַל צַוָּארָיו וַיִּשָּׁקֵהוּ,

אֵלָיו הִתְבּוֹנָן וְלִבּוֹ הָמָה כְּמוֹ יָיִן

כִּי מִכַּף רֶגֶל עַד רֹאשׁ מוּם בּוֹ אָיִן –

– בָּרוּךְ אֵל עֶלְיוֹן! – הַמֶּלֶךְ הֵרִיעַ –

נוֹצֵר אֱמוּנִים בִּלְתּוֹ אֵין מוֹשִׁיעַ

אֱלֹהֵי דָנִיֵּאל מְרוֹמָם עַל כָּל בְּרָכָה

הוּא הָאֱלֹהִים – אַשְׁרֵי הָעָם שֶׁלּוֹ כָכָה!

וַיִּסֹּב הַמֶּלֶךְ אֶל הָעֹטְרִים אוֹתוֹ

וַיְצַוֵּם דָּבָר וַיְמַהֲרוּ לַעֲשׂוֹתוֹ

וְהִנֵה רַעַשׁ – הַרְרֵי עַד יִתְפּוֹצָצוּ

נָרֹץ הַגַּלְגַּל, רְעָמִים יִתְרֹצָצוּ

יַמִּים יֶהֱמָיוּן, גַּלִּים יִתְגַּלְגָּלוּ

בַּהֲמֻלָּה גְדוֹלָה קוֹלוֹת יִתְבֹּלָלוּ:

זַעֲקַת שֶׁבֶר וִילָלָה, זְעָקָה מָרָה,

וּשְׁאָגָה מְשׁוּבָה וּתְרוּעָה נִשְׂעָרָה

יַכּוּ בְתִמָּהוֹן אָזְנַיִם יַחֲרִישׁוּ

וִיסַמְּרוּ בָשָׂר וּנְפָשׁוֹת יַרְעִישׁוּ –


מַה קּוֹלוֹת אֵלֶּה וּמִי שָׁאוֹן קָרָא?

מַה פֵּשֶׁר דָּבָר, מַה נַּעֲשָׂה, מַה קָּרָה?

מִטַּעַם הַמֶּלֶךְ הַשָּׂרִים הוּבָאוּ

מַלְשִׁינֵי דָנִיֵּאל לִסְפּוֹתוֹ יָגָעוּ

גַּם זַרְעָם וּנְשֵׁיהֶם אִתָּם הוּבָלוּ

וּלְגֹב הָאֲרָיוֹת יַחַד הֻטָּלוּ.

אֶל קַרְקַע הַבּוֹר הֵם טֶרֶם הִגִּיעוּ

הִשְׂתָּעֲרוּ הָאֲרָיוֹת, הִתְעַלּוּ, הִרְקִיעוּ

וּבְנֶפֶשׁ שֹׁקֵקָה וּבְחֵמָה שְׁפוּכָה

עֲלֵיהֶם הִתְנַפְּלוּ בִּמְהוּמָה וּמְבוּכָה,

יְחַנְּקוּ וִישַׁסְּעוּ, יִקְרָעוּ, יִבְלָעוּ

וְרָמְסוּ וְפָרְשׂוּ וִיזוֹלְלוּ וְיִסְבָּאוּ

זְרֹעַ אַף קָדְקֹד יְרַטְּשׁוּ וִיכַרְסֵמוּ

וִיפַצְּחוּ עַצְמוֹתֵיהֶם וִיגָרֵמוּ.

וּלְצִוְחַת נִטְרָפִים אֶרֶץ קֹרַעַת

שַׁאֲגַת אֲרָיוֹת שָׁמַיִם בֹּקַעַת,

שַׁוְעַת חֲלָלִים כִּי תִגְדַּל מִכְּאֵב נֶצַח

תִּגְדַּל כֵּן הַתְּרוּעָה בַּחֲמַת נְקַם רֶצַח,

אֶל זֵדִים מַלְשִׁינִים נָקָם יָשִׁיבוּ

עַל כִּי לֶאֱכוֹל נָקִי אוֹתָם הִרְעִיבוּ –

נָקָם וְשִׁלֵּם הוּא, נְקַם בֶּטֶן רְעֵבָה

אָיֹם הוּא וְנוֹרָא נְקַם חַיָּה שׁוֹבֵבָה!


מִנְּקַם רֶצַח וָדָם נָסֵב עֵינָיִם

בּוֹ לֹא יִשְׂמַח צַדִּיק בַּר לֵבָב וְכַפָּיִם,

לַעֲזוֹב גֵּיא הַהֲרֵגָה הִתְחַנֵּן לַמֶּלֶךְ

נֵפֶן גַּם אָנוּ, אַחֲרֵיהֶם נֵלֶךְ,

מִקְדַּשׁ מֶלֶךְ נָבֹא, בְּהֵיכָלוֹ פְּנִימָה

שָׁם יָרֹן לֵב צַדִּיק בִּנְקָמָה נְעִימָה:

נִקְמַת ישֶׁר וָצֶדֶק בְּרֶשַׁע כֶּסֶל

נִקְמַת אֵל אֱמֶת בֶּאֱלִיל וָפֶסֶל,

נִקְמַת תֹּם וָקשֶׁט בְּשָׁוְא וּמַדּוּחַ

לֹא בְחַיִל וְלֹא בְכֹחַ כִּי אִם בָּרוּחַ!

מֶלֶךְ עֶלְיוֹן, שְׂאֵת עַמִּים רַבִּים מֶנְהוּ

אֶת אֵל יִשְׂרָאֵל יַעֲרִיץ וִירֹמְמֶנְהוּ

וּסְפָרִים שָׁלַח לְכָל הָעַמִּים וְהַלְּשׁוֹנוֹת

לְכָל ישְׁבֵי תֵבֵל, שׁוֹכְנֵי אֲרָצוֹת שׁוֹנוֹת:

– אֲנִי, דָּרְיָוֶשׁ, שָׁלוֹם רָב מַשְׁמִיעַ

וּלְכָל הָעַמִּים לִהְיוֹת גָּלוּי מוֹדִיעַ

כִּי בְכָל מֶמְשַׁלְתִּי, מְקוֹם דְּבָרִי מַגִּיעַ

חֹק וּמִשְׁפָּט שַׂמְתִּי וָאֲצַו בְּמַפְגִּיעַ

לְיִרְאָה אֶת הַשֵּׁם הַנִּכְבָּד וְהַנּוֹרָא

אֱלֹהֵי דָנִיֵּאל, בִּשְׁמוֹ הוּא קֹרָא,

שַׁדַּי שַׂגִּיא כֹחַ חַיִּים בִּרְצוֹנוֹ

וְרָגְזוּ וְחָלוּ וּפָחֲדוּ מֵהֲדַר גְּאוֹנוֹ,


הוּא אֱלֹהִים חַיִּים בְּאוֹרוֹ נִרְאֶה אוֹר

מַלְכוּתוֹ עוֹלָמִים, מֶמְשַׁלְתּוֹ בְּכָל דּוֹר,

מָגֵן לַחוֹסִים בּוֹ, לְקֹוָיו אֵל לְמוֹשָׁעוֹת

נוֹתֵן אֹתוֹת וּמוֹפְתִים עֹשֶׂה נִפְלָאוֹת

בַּשָּׁמַיִם, בָּאָרֶץ, בַּיַּמִּים וְכָל תְּהֹמוֹת

כִּי עֶלְיוֹן הוּא, מֵעַל עִתִּים וּמְקוֹמוֹת.

וְהוּא הָאֱלֹהִים רַב הָעֲלִילִיָּה

עַתָּה הִפְלִיא לַעֲשׂוֹת הִגְדִּיל תּוּשִׁיָּה

עַל דָּנִיֵּאל חֲסִידוֹ עֵינוֹ חָסָה

מֵאֲרָיוֹת הִצִּילָהוּ – כִּי בוֹ חָסָה,

מֵאֲרָיוֹת מְזֵי רָעָב וּלְחֻמֵי רֶשֶׁף

עִמָּם נֶעְצַר מֵהַנֶּשֶׁף עַד הַנֶּשֶׁף!


* * *


וַיַּעַשׂ הַמֶּלֶךְ מִשְׁתֵּה שְׁמָנִים

לְכָל שָׂרָיו וַעֲבָדָיו הַנֶּאֱמָנִים

אֶת מִשְׁתֵּה דָנִיֵּאל בִּיקַר תִּפְאֶרֶת

וַיִּכָּתֵב בְּסֵפֶר זִכָּרוֹן לְמִשְׁמֶרֶת

וֵאלֹהֵי יִשְׂרָאֵל נוֹדַע בַּגּוֹיִם

לוֹ כָּבוֹד יָשִׂימוּ תְּהִלָּתוֹ בָּאִיִּים

בְּנִפְלְאוֹתָיו יָשִׂיחוּ כָל פֶה וּשְׂפָתָיִם

וַיִּגְדַּל שְׁמוֹ וַיִּקָּדֵשׁ מֵאַפְסָיִם.

וְדָנִיֵּאל הֻקַּם עַל, רֹאשׁ לְכָל הַשָּׂרִים

וַיַּצְלַח וַיַּשְׂכֵּל לַעֲשׂוֹת מֵישָׁרִים


וּבְהוֹרִישׁ דַּרְיָוֶשׁ אֶת הַמֶּמְשָׁלָה

אֶל כּוֹרֶשׁ הַפַּרְסִי, עַל כִּסְאוֹ עָלָה,

הוּא מָשִׁיחַ נָגִיד, הָאֱלֹהִים יְעָדוֹ

לִבְנוֹת נָוֵהוּ וּלְהַרְבִּיץ צֹאן יָדוֹ,

רָחַב לֵב דָּנִיֵאל, נַפְשׁוֹ נֻחָמָה,

כִּי הֵקִיץ הַקֵּץ, עַצַת עֶלְיוֹן קָמָה

וּדְבָרוֹ בֶחָזוֹן, בּוֹ גָּלָה סוֹדוֹ,

הִנֵה בָא, הִנֵּה נֵעוֹר וְנִגְלָה יְסוֹדוֹ,

כִּי שִׁיבַת צִיּוֹן הָשֵׁב פָּקַד כּוֹרֶשׁ

וּשְׁאֵרִית יְהוּדָה עַל אַדְמָתָהּ תַּךְ שֹׁרֶשׁ

וַתְּהִי אָז עַל דָּנִיֵּאל מִשְׂרָה כְבוּדָה

וְקֹרָא לוֹ: שֵׁשְׁבַּצַּר הַנָּשִׂיא לִיהוּדָה,

וּבְשֵׁשׁ צָרוֹת צוּרוֹ כִּי הִצִּילָהוּ

נִחֻמִּים לוֹ שִׁבְעָתַיִם גְּמָלָהוּ:

לָשֵׂאת כְּלֵי הַקֹּדֶשׁ, צָרִים נִאֵרוּ

לַהֲשִׁיבָם אֶל הָאֱלֹהִים וְיִטָּהֵרוּ,

לִרְאוֹת רֵאשִׁית לִיסֻד הַמַּעֲלָה מִבָּבֶל

לְרְאוֹת אֶבֶן הָרֹאשָׁה בְּיַד זְרֻבָּבֶל,

אֶבֶן מָאֲסוּ הַבּוֹנִים הָיְתָה לְרֹאשׁ פִּנָּה

וּפְדוּיִם יְשֻׁבוּן לָבֹא צִיּוֹן בְּרִנָּה!
