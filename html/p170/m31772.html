כִּמְעַט אֲשֻׁרֵי הַיְשָׁרִים

אל שלמה בן בתאת


כִּמְעַט אֲשֻׁרֵי הַיְשָׁרִים מָטוּ / עֵת רַעֲיוֹנֵיהֶם לְדֵעָה שָׁטוּ,

שָׁטוּ רְאוֹת עָמָל אֲשֶׁר עָשׂוּ בְנֵי / יָמִים וְדִין עָוֶל אֲשֶׁר שָׁפָטוּ,

לוּלֵי אֱמוּנָה תָמְכָה יַד בַּעֲלֵי / תֹם – מִנְּתִיב צֶדֶק כְּאַיִן שָׂטוּ,

אֵיךְ הוֹלְלִים שָׁלוּ וְנַפְשׁוֹת כָּל־מְתֵי / שֵׂכֶל בְּחַיֵּיהֶם מְאֹד נָקָטוּ.

אִמְרוּ: הֲנֶעֶקְשׁוּ שְׁבִילֵי רוֹם וְאִם / דַּרְכֵי צְבָאָיו מֵהֲלֹךְ עֻבָּטוּ,

לָבְשׁוּ לְבוּשׁ הֵפֶךְ וּבֶגֶד בּוֹגְדִים / עָטוּ וְכָתְנוֹת הָאֱמֶת פָשָׁטוּ,

נָשְׁכוּ בְשִׁנֵּיהֶם וְשָׁלוֹם קָרְאוּ / וּכְמַדְקֵרוֹת חֶרֶב הֲתֻלִּים בָּטוּ,

לוּ בָם זְמָן יִשְׁפֹּט וְדִינָיו יִצְדְּקוּ / אָז עָרְפוּ אוֹתָם וְלֹא שָׁחָטוּ.

מַה־מָּצְאוּ יָמִים בְּאַנְשֵׁי הָאֱמֶת / כִּי לָעֲגוּ לָהֶם וּבָהֶם עָטוּ,

מִלְאוּ יְדֵי רָעִים בְּלִבּוֹתָם בְּהוֹן / עָתָק וְלַטּוֹבִים מְעַט צָבָטוּ,

לֹא אֵדְעָה אִם הַזְּמָן נִשְׁחַת וְאִם / דַּרְכֵי אֲנָשָׁיו לַחֲתֹף יָרַָטוּ.

לִי מַר מְאֹד מִכָּל־אֲשֶׁר נִסּוּ, הֲכִי / אֹתִי בְשֵׁבֶט הַנְּדֹד חָבָטוּ.

הָעֵת אֲשֶׁר שֶׁמֶשׁ נְעוּרִים פָּנְתָה / גַּם צִלְלֵי הַשַּׁחֲרוּת יִנָּטוּ, –

הַעוֹד יְשׁוּבוּן אוֹ יְשִׁיבוּן אֶל מְתֵי / זֹקֶן עֲבֹט נֹעַר אֲשֶׁר עָבָטוּ?

אַחַי חֲמַת תַּנִּין בְּלַחְמִי נָתְנוּ / אוֹ עִנְבֵי רוֹשׁ בַּכְּלִי שָׁחָטוּ,

יֵמַר לְפִי שֵׁכָר וְאֶשְׁתּוֹמֵם לְשִׁיר / עַל פִּי נְבָלִים נֹגְנִים פָּרָטוּ,

רַצְתִּי לְאוֹר כָּל־יוֹם מְצוֹא חֵפֶץ עֲדֵי / עָלָיו גְּדוּדֵי מַאֲפֵל פָּשָׁטוּ,

תִּדַּד שְׁנַת עֵינַי בְּלֵילִי עַד יְדֵי / שַׁחַר קְוֻצּוֹת לַיְלָה מָרָטוּ,

אַךְ צַעֲדֵי חֶפְצִי זְמָן אָסַר וְגַם / רִכְבֵי רְצוֹנִי מִקְּרֹב קֻמָּטוּ,

אֶתְאַב לְאַחִים בֵּין צְלָעַי שָׁכְנוּ, / אָכֵן בְּיַד סֶלַע נְדֹד נִשְׁמָטוּ,

וּלְנֹד שְׁלֹמֹה לָבְשׁוּ קַדְרוּת שְׁמֵי / גִילַי וְשַׂקֵּי שַׁחֲרוּת יָעָטוּ,

אִישׁ הֶחֱיָה מֵתֵי אֱמוּנִים אַחֲרֵי / אֹתָם יְדֵי הַשֹּׁקְרִים חָנָטוּ,

מִמַּעֲלָלָיו יֶאֱרוּ בֹשֶׂם כְּמוֹ / מִפִּיו פְּנִינֵי מַאֲמָר לָקָטוּ,

מִמֹּעֲצוֹתָיו סוֹד אֱמוּנָה חָטְפוּ / מִדַּבְּרוֹתָיו הָאֱמֶת חָלָטוּ.

כַּמָּה אֲרָיִים יָרְאוּ עֵת נָעֲרוּ / גּוּרֵי מְזִמּוֹתָיו וְגַם שָׁקָטוּ,

חָכְמָה הֱנִיקַתְהוּ חֲלֵב דַּדָּהּ וְהֵם / מִפַּת פְּתַיּוּת שָׁמְנוּ בָעָטוּ,

בּוֹ אָמְצוּ לִבּוֹת בְּנֵי חָכְמָה וּבוֹ / שָׂגְבוּ פְלִיטֶיהָ אֲשֶׁר פָּלָטוּ.

אָחִי, הֲתִלּוֹן עַל יְלִידֵי יוֹם אֲשֶׁר / בָּךְ יֶחֱרֶה אַפָּם וְיִתְקוֹטָטוּ?

אַיֵּה יְשָׁרִים מֵחֲמָתָם נִצְּלוּ? / אֵיפֹה נְקִיִּים בַּזְּמָן נִמְלָטוּ?

אַתָּה שְׁאֵרִית עָם פְּעָלָיו יָדְעוּ / וּבְעֵין אֱמוּנָה מַעֲשָׂיו נִבָּטוּ,

לֹא הֶאֱמִינוּ בוֹ וְלֹא טָשׁוּ עֲלֵי / אֹכֶל שְׁלַל תֵּבֵל וְלֹא יָעָטוּ,

הַשְׁקֵט מְעַט, כִּי הֵן בְּשִׁלְטֵי גָדְלְךָ / רַבִּים עֲלֵי רַבֵּי זְמָן שָׁלָטוּ,

וּבְלַהֲטֵי שֶׁלַח דְּבָרְךָ הָדְפוּ / חַרְבוֹת מְתֵי אֹמֶר אֲשֶׁר מֹרָטוּ;

מַלֵּא בְיִתְרוֹן שִׂכְלְךָ בוֹ מַחֲסֹר / וּרְפָא שְׁבָרָיו כִּי פְלָאִים מָטוּ,

וּרְבֵה לְרַבֵּי הַזְּמָן כִּי צָעֲרוּ / וּכְבַד לְיַלְדֵי בִין אֲשֶׁר מָעָטוּ.

הֵא שִׁיר, אֲבָל נִטְעֵי אֲהָבִים בּוֹ, אֱרֵה / פַגֵּי תְבוּאָתָם אֲשֶׁר חָנָטוּ,

שִׁיר נִכְלְמוּ לִפְנֵי מְאוֹרָיו כּוֹכְבֵי / שֶׂכֶל וּמִבֹּשֶׁת פְּנֵיהֶם לָטוּ,

נוֹזְלֵי חֲרוּזִים יִרְדוּ לוֹ עַל אֱנוֹשׁ / לִבּוֹ רְבִיבֵי הַבְּכִי לָהָטוּ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
