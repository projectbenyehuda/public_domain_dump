קראני גביר הקדיש קראיו

קראני גביר הקדיש קראיו / ר' משה אבן-עזרא


קְרָאַנִי גְבִיר הִקְדִּישׁ קְרֻאָיו / וְהִקְרַנִי לְחֶבְרָה עִם־יְדֻעָיו

לְדוֹדִים מָֽתְקוּ אֶל־פִּי כְּדוֹדָיו / וְרֵעִים יָֽקְרוּ עָלַי כְּרֵעָיו

בְּגַן־בִּיתָן אֲשֶׁר הַחֵן סְבִיבוֹ / וְהַיֹּפִי לְאַרְבַּעַת רְבָעָיו

מְלֻבָּשׁ כָּֽתְנוֹת מֶשִׁי וְעָטוּי / מְעִיל תּוֹלָע וְשֵׁש וּשׁלַל צְבָעָיו

עֲרֻגוֹתָיו בְּלִי רֹקֵם וְחֹשֵׁב / מְשֻׁבָּצִים וְשׁוֹשַׁנִּים יְצוּעָיו

כְּאִלּוּ הַזְּמָן קָלַע פְּנֵיהֶם / בְּתִמֹרָיו וּמִקְלַעַת פְּקָעָיו

כְּזָהָב מַעֲשֵׂה מוֹרָד עֲלֵיהֶם / מָאוֹר שֶׁמֶשׁ וְהַשִּׂיחִם רְקִיעָיו

וּמֵימָיו יִגְּרוּ אַרְצָה כְּדִמְעִי / עֲלֵי מִפְקַד אֲחֵי לִבִּי וְרֵעָיו

וְתוֹר יָתוּר לְזִמְרָתוֹ הֲדַסִּים / וְסִיס יָשִׂישׂ עֲלֵי גַפֵּי נְטָעָיו

יְרַנֵּן בֵּין עֲצֵי שֹׂרֵק וְשֹׁרֵק / וְיִתְרוֹנֵן כְּשִׁכּוֹר מִדְּמָעָיו

וְעוֹפוֹת פָּֽרְצוּ עַל־כָּל־פְּרָצָיו / וְשָׁם דָּגְרוּ וּבָקְעוּ בֵּין בְּקָעָיו

בְּיוֹם עֵרוּ יְשֵׁנֵי גִיל וְעָצַם / זְמָן עֵינָיו וְנָזֹרוּ פְגָעָיו

וְעִתָּיו מִהֲרוּ לָעוּת יְעֵפָיו / עֲלִיצֻתָם וְלַרְגִּיעַ רְגָעָיו

וְהַשֻּׁלְחָן דְּמוּת גַּלְגַּל וְאַגָּן / כְּמוֹ־סַהַר וְכוֹכָבִים גְּבִיעָיו

קְעָרוֹתָיו מְלֵאוֹת מַעֲדַנִּים / וּמַחְתֹּתָיו בְּשָׂמִים רֹאשׁ וְיָעָיו

וְכוֹס שֹׁהַם אֲשֶׁר עָלָה כְּכוֹכָב / בְּיַד מַשְׁקֶה וּמֵי־אֹדֶם בְּמֵעָיו

כְּחֹשֵׁק קָֽפְאוּ מֵימֵי דְמָעָיו / עֲלֵי לֶחְיוֹ וְהָאֵשׁ בֵּין צְלָעָיו

יְקַצֵּץ מִזְּהַב אוֹפִיר פְּתִילִם / עֲלֵי יָדָיו לְהָאִיר אֶת־זְרֹעָיו

לְהָסִיר מִלְּבַב כֹּאֵב עֲצָבָיו / וְלַרְגִּיעַ בְּלֵב רַגָּז רְגָעָיו

לְאַט יֶהְגֶּה וְרֹאשׁ יַגֶּה וְיִשְׁגֶּה / בְּאַהְבָתוֹ וְיֵאָמְנוּ פְצָעָיו

חֲמֻדוֹתָיו לְאֵין קֶצֶה וְקֶצֶב / כְּאַבְרָהָם בְּמַהְלָלָיו וְדֵעָיו

אֲשֶׁר יִזַּל כְּטַל אִמְרוֹ וְחֶסֶד / בְּיָדוֹ מַחֲנוֹתָיו הֵם וּמַסְעָיו

רְחַב לֵבָב רְחַב־צַעַד לְהֵיטִיב / וְלָרַע מַה־מְּאֹד צָרוּ פְשָׂעָיו

וְלֶכְתּוֹ עִם־אֱלֹהִים וַאֲנָשִׁים / בְּהַצְנֵעַ וְחָכְמָה אֶת־צְנוּעָיו

וְלָאֹבֵד בְּמַחְשַׁךְ הַֽיְגוֹנִים / כְּתֹרֶן אָהֳלוֹ נָטַע לְתֹעָיו

וּבוֹ יִנָּֽפְשׁוּ מָרֵי נְפָשִׁים / וְיָנוּחוּ יְעֵפֵי־אוֹן יְגֵעָיו

וּמִי נִקְרַע סְגוֹר לִבּוֹ בְּחִצֵּי / נְדֹד פִּיהוּ יְהִי תֹפֵר קְרָעָיו

הֲכִי רַכּוּ כְשֶׁמֶן דַּבְּרֹתָיו / וְיוֹם אֹמֶר יְפֹצֵץ בָּם סְלָעָיו

אֲנָשִׁים כָּֽבְשׁוּ לָהֶם עֲבָדִים / וּמֵאֵין הוֹן בְּנֵי חוֹרִים רְצוּעָיו

עֲדֵי בֹשׁוּ לְנִדְבָתוֹ עֲנָנִים / וְהוֹדוּ לוֹ נְדִיבֵי עָם וְשׁוֹעָיו

וּמִי זָרַע יְמֵי חַיָּיו חֲסָדִים / יְהִי קוֹצֵר לְפִי־חֶסֶד זְרָעָיו

יְהִי שָׁלֵו עֲדֵי יַנְחִיל כְּבוֹדוֹ / וּמַעְלָתוֹ לְיַלְדֵי שַׁעֲשׁוּעָיו.
