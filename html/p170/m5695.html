[מֵאֶרֶץ שָׁפַלְנוּ בְּרֻבֵּי עֲוֹנֵינוּ]

כמנהג קהלת ארגיל(אלג’יר)


מֵאֶרֶץ שָׁפַלְנוּ / בְּרֻבֵּי עֲוֹנֵינוּ

שַׁחְנוּ מֵהָרִים / אֵלֶיךָ פָּנֵינוּ

הֵרַעְנוּ וְאֵיךְ נוֹחִיל / לִרְאוֹת בְּעֵינֵינוּ

בְּכָל אֲפָסֵינוּ – אֵת יְשׁוּעַת אֱלהֵינוּ: [ישע' נב, י]

אֱלֹהֵינוּ, אֵיךְ / עָפָר וָאֵפֶר

אֶדֶר יְקָרְךָ / יְבָאֵר עַל סֵפֶר

אֱנוֹשׁ כֶּחָצִיר / יָצִיץ וְלֹא יִפֶר

יְסַפֵּר – / אֶת מַעֲשֵׂה אֱלֹהֵינוּ: [ירמ' נא, י]

אֱלֹהֵינוּ, בְּךָ / אִם פָּשָׁעְנוּ

בִּיוֵן מְצֻלוֹת / מַעֲשֵׂינוּ טָבָעְנוּ –

בָּאנוּ לְהִתְוַדּוֹת / כִּי לֹא נִכְנָעְנוּ

וְלֹא שָׁמַעְנוּ – / בְּקוֹל אֲדֹנָי אֱלֹהֵינוּ: [דנ' ט, י]

אֱלֹהֵינוּ, גּוֹי / בְּךָ יִתְפָּאָר

גֵּווֹ בְמוֹסָר / אַכְזָרִי מְתֹאָר

גּוֹיִים כִּחֲשׁוּ / בְגָדְלְךָ הַמְפֹאָר

וְנִשְׁאַר – / גַּם הוּא לֵאלֹיהֵנוּ: [זכר' ט, ז]

אֱלֹהֵינוּ, דִּין / עָנִי וְנִכְלָם

דּוֹן, וְהֶרֶף / כִּי נֶגְדְּךָ נֶאֱלָם

דְּבָרוֹ מֵאֶרֶץ / שַׁח וְנֶעֱלָם

וְיָקוּם לְעוֹלָם – / דְּבַר אֱלֹהֵינוּ: [ישע' מ, ח]

אֱלֹהֵינוּ, הַקְרֵב נָא / אֶת רְנָנִי

הַפְלֵא חֲסָדֶיךָ / לְעֵין כֹּל וְעֵינָי

הַשְׁמִיעַ מְיַחֲלִים / יִרְאוּ מִמְּעוֹנָי

כְּבוֹד אֲדֹנָי – / הֲדַר אֱלֹהֵינוּ: [ישע' לה, ב]

אֱלֹהֵינוּ, וּבוֹרֵא / כָּל נְשָׁמָה

וָתִיקִים אַמֵּץ / לוֹבְשֵׁי חִיל וְאֵימָה

וּלְאַט כּיִ יָגֹרְנוּ / מִפְּנֵי הָאַף וְהַחֵמָה

וַנָּצוּמָה – / וַנְּבַקְשָׁה מֵאֱלֹהֵינוּ: [ישע' לה, ב]

אֱלֹהֵינוּ, זֶה / עָנִי עָזוּב וְעָצוּר

זְדוֹנָיו הֱצִיקוּהוּ / לְהִטָּמֵן בַּצּוּר

זִכְרְךָ לֹא שָׁכַח / כִּי בְלִבּוֹ נָצוּר

וּמִי צוּר – / זוּלָתִי אֱלֹהֵינוּ: [תה' יח, לב]

אֱלֹהֵינוּ, חַנּוּן / וְרַחוּם וּמַקְשִׁיב

חֶרְפָּה תֶאֱסֹף / וְחָרְבָּה תוֹשִׁיב

חַסְדְּךָ תִּשְׁלַח / רוּחַ חֵן מַשִּׁיב

לְהָשִׁיב – / חֲרוֹן אַף אֱלֹהֵינוּ: [ע.' י, יד]

אֱלֹהֵינוּ, טוּב זִכְרְךָ / אֶל חִכִּי

טָעַמְתִּי מִצּוּף / דִּבְשִׁי וְנִסְכִּי

טַעַם זְמִירְךָ / נֵר בֶּאֱשׁוּן חָשְׁכִּי

כִּי – / טוֹב זַמְּרָה אֱלֹהֵינוּ: [תה' קמז, א]

אֱלֹהֵינוּ, יִסַּרְתִּי / לֵב קֹשֵׁר

יְדֵי מַעְגָּלָיו / בְּדָתְךָ לְאַשֵּׁר

יֵבֹשׁ כִּי מָאַס / שׂוּם שֵׂכֶל וְכֹשֶׁר

וְכָל אֲשֶׁר – / יֹאמַר אֱלֹהֵינוּ: [דב’ה, כד]

אֱלֹהֵינוּ, כּוֹאֲבִים / וַעֲנִיִּים וַעֲצֵבִים

כְּהַיּוֹם בְּעֵינֵינוּ / נֶחְשַׁבְנוּ כַּחֲגָבִים

כֻּלָּנוּ אוֹמְרִים / הַאִם לְנִבְזֵי עֲצַבִּים

אֱלֹהִים קְרֹבִים / כַּאדֹנָי אֱלֹהֵינוֹ: [שם ד, ז]

אֱלֹהֵינוּ, לָקַחְנוּ / עִמָּנוּ תְּחִנּוֹת

לָבוֹא עָדֶיךָ / עִם בָּרִים לְהִמָּנוֹת

לְהוֹדוֹת קַמְנוּ / וּזְדוֹנֵנוּ לַעֲנוֹת

וּלְהִתְעַנּוֹת – / לִפְנֵי אֱלֹהֵינוּ: [עז' ח, כא]

אֱלֹהֵינוּ, מִדְבְּרֵי / דָת זָנַחְנוּ

מִדֶּרֶךְ סַרְנוּ / וְתוֹרָתְךָ שָׁכָחְנוּ

מוֹדִים בָּאנוּ / וּמִבֹּשֶׁת נֶאֱנָחְנוּ

אֲנַחְנוּ – / מָעַלְנוּ בַּאדֹנָי אֱלֹהֵינוּ: [שם י, ב]

אֱלֹהֵינוּ / נְבוֹכִים מֵחוֹבָם

נְסָה עֲלֵיהֶם / אוֹר סְלִיחָה סְבִיבָם

נַעֲווּ מִשְּׁמֹעַ / וְלֹא אָמְרוּ בִלְבָבָם

נִירָא נָא / אֶת אֲדֹנָי אֱלֹהֵינוּ: [ירמ' ה, כד]

אֱלֹהֵינוּ, שְׂרִיד חֶרֶב / וּלְבָבוֹ יָחִיל

שַׂק יַצִּיעַ / תַּחַת פְּתִיגִיל

סָפוּ בְהִתְנַצֵּל / עֲדִי טַבַּעַת וְעָגִיל

גִּיל – / שִׂמְחָה מִבֵּית אֱלֹהֵינוּ: [יוא' א, טז]

אֱלֹהֵינוּ, עָוִינוּ / וּפָשַעְנוּ, אָכֵן

עִוַּתְנוּ דַרְכֵּנוּ / וְדִבַּרְנוּ לֹא כֵן

עֵינֵי עֶבֶד / לְיַד אָדוֹן וּמִסְכֵּן

כֵּן – / עֵינֵינוּ אֶל אֲדֹנָי אֱלֹהֵינוּ: [תה' קכג, ב]

אֱלֹהֵינוּ, פָּשַׁעְנוּ בְךָ / עָוִינוּ וּמָרִינוּ

פָּעַלְנוּ אָוֶן / וְעָוֹן לֹא כִסִּינוּ

פִּקּוּדֵי עֵדוֹתֶיךָ / זָנַחְנוּ וְשָׁגִינוּ

וְלֹא חִלִּינוּ – / פְּנֵי אֲדֹנָי אֱלֹהֵינוּ: [דנ' ט, יג]

אֱלֹהֵינוּ, צְבָא / חוֹבַי וַעֲוֹנִי

צַוֵּה לְהַפְאוֹת / מִנֶּגֶד עֵינָי

צוֹפְנֵי מְצוּקוֹת / יְחַדְּשׁוּ אֶת רְנָנָי

בְּעִיר אֲדֹנָי – / צְבָאוֹת בְּעִיר אֱלֹהֵינוּ: [תה' מח,]

אֱלֹהֵינוּ, קְרָא דְּרוֹר / לָנוּ מֵחוֹבִים

קִצְרֵי רוּחַ / מֵעֶרֶב נִצָּבִים

קָרְבַּן אִמְרֵיהֶם / אֲשֶׁר הֵם מַקְרִיבִים

קְרֹבִים – / יְהוּ אֶל אֲדֹנָי אֱלֹהֵינוּ: [מל"א ח,]

אֱלֹהֵינוּ, רַחֲמֶיךָ / מְאֹד יִתְאַוּוּ

רוֹהִים, יְשַׁבֵּרוּ / אֵלֶיךָ וִיקַוּוּ

רוֹחֲשִׁים לַכֹּל יְבוֹאוּן / לְבֵיתְךָ וְיִלָּווּ

הִשְׁתַּחֲווּ – / רוֹמְמוּ אֲדֹנָי אֱלֹהֵינוּ: [תה' צט,]

אֱלֹהֵינוּ, שֶׁקֶר נָחֲלוּ / אַךְ נְכָאִים

שָׁמַעְנוּ אַַחֲרֵינוּ / הִתּוּלֵי חַטָּאִים

שׁוֹפֵט, הֲלֹא תַחְקֹר / לִבּוֹת נִבְרָאִים

אִם – / שָׁכַחְנוּ שֵׁם אֱלֹהֵינוּ: [שם מד, כא]

אֱלֹהֵינוּ, תֶּן עֹז / לְבֵן נוֹקָשׁ הוּדָשׁ

תְּשׁוּעַת כַּפָּרָה / עָלָיו תִּתְחַדָּשׁ

תְּמִימֵי סְגֻלָּתְךָ / יְשׁוֹרְרוּ בַמִּקְדָּשׁ

ׁשִיר חָדָשׁ / תְּהִלָּה לֵאלֹהֵינוּ: [שם מ, ד]

אֱלֹהֵינוּ, חוֹבֵינוּ / כְדוֹנַג יִדְאָבוּ

זְמִירֵינוּ לְךָ לְרָצוֹן / יַעֲלוּ וְיִיטָבוּ

קְדוֹשֵׁי עֶלְיוֹן / יִתְבַּשְּׂרוּ כִּי נִקְשָׁבוּ

הָבוּ גֹדֶל לֵאלֹהֵינוּ: [דב' לנ, ג]

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
