לְבָבִי יֶחֱרַד

אל אבו־איוב [שלמה] בן אלמעלם ואחיו


לְבָבִי יֶחֱרַד לָצֵאת אֲלֵיהֶם / וְעֵינַי יִזְּלוּ מַיִם עֲלֵיהֶם

וְיֵלְכוּ יֶחְמְרוּ, כִּי מִמְּקוֹרֵי / כְּבֵדִי נִמְשְׁכוּ מֵי נַחֲלֵיהֶם.

אֲדַמֶּה כִּי יְכַבּוּ אֵשׁ קְרָבַי, / וְאָכֵן נוֹקְדוּ בָם גַּחֲלֵיהֶם.

בְּיָדָם נִפְקְדָה נַפְשִׁי בְּצֵאתָם / וְאֵיךְ אוֹתָהּ אֲמַשֵּׁשׁ בֵּין כְּלֵיהֶם,

אֲחוֹנֵן אֶת־עֲפַר רַגְלָם וְאֶשַּׁק / בְּדֶרֶךְ צָעֲדוּ בָהּ שַׁעֲלֵיהֶם,

וְאֵיךְ אֶתְאַב לְמַרְאֵיהֶם וּבֵינוֹת / קְרָבַי תָּקְעוּ אֶת־אָהֳלֵיהֶם,

וְאֵיךְ מִקֹּר וְחֹם אִירָא עֲלֵיהֶם / וְהֵם בֵּינוֹת צְלָעַי צִלֲלֵיהֶם,

וְרוּחָם אֵיךְ אֱהִי שֹׁאֵף וְזֵכֶר / שְׁמָם יִשָּׂא לְאַלְפֵי מֹשְׁלֵיהֶם,

לְאַחִים מָשְׁלוּ עַל הַלְּבָבוֹת – / הֲכִי צֶדֶק וְחֶסֶד מֹשְׁלֵיהֶם,

מְאוֹרִים, גָּזְלוּ שֶׁמֶשׁ וְסַהַר / מְאוֹרֵיהֶם, וְאַךְ הֵם גּוֹזְלֵיהֶם,

וְהָרִים עָרְכוּ תָבוֹר וְחֶרְמוֹן, / וְאִם גָּדְלוּ עֲלֵיהֶם מִגְדְּלֵיהֶם,

וְעֵת עִמָּם יְהוּ עוֹלִים בְּמֹאזְנֵי / אֱמוּנָה – חָסְרוּ אֶת־מִשְׁקְלֵיהֶם.

עֲצֵי עֵדֶן עֲלֵי פַלְגֵי תְעוּדָה / וְנַחְלֵי הַחֲסָדִים נַחְלֵיהֶם.

עֳפָרִים, אַךְ בְּמִלְחֶמֶת חֲכָמִים / בְּפִיהֶם יִטְרְפוּ אֶת־שַׁחֲלֵיהֶם,

וּבִמְרוּצַת בְּנֵי הַשְׂכֵּל עֲיָרִים / הֲיַשִּׂיגוּ לְאִטָּם יַעֲלֵיהֶם?

בְּפִיהֶם רִפְּאוּ חוֹלֵי יְגוֹנִים / וּמִלֵּיהֶם יְקִימוּן כּוֹשְׁלֵיהֶם,

וְהִשְׁאִילוּ לְכָל דּוֹרֵשׁ חֲפָצָיו / עֲדֵי הָיוּ עֲנָנִים שֹׁאֲלֵיהֶם,

יְדֵיהֶם לִמְּדוּ מַתָּן לְיָמִים / עֲדֵי הִתְנַדְּבוּ כָּל־נֹכְלֵיהֶם.

יְאַו סַהַר הֱיוֹת כֶּתֶר לְרֹאשָׁם / וְעַיִשׁ עִם יְלָדֶיהָ חֲלֵיהֶם,

וּבִבְלוֹיֵי חֲלִיצוֹתָם יְפָאֵר / זְמָן שָׂרָיו וּבִשְׂרוֹךְ נַעֲלֵיהֶם,

בְּאֵין יָד הַלְּבָבוֹת שֹׁלְלֵיהֶם / וְעֵינַיִם בְּלִי פֶה אֹכְלֵיהֶם.

כְּאִלּוּ אֵל עֲלֵי חֶפְצָם יְצָרָם / וְעַל פִּיהֶם הֱכִינָם פֹּעֲלֵיהֶם.

דְּמֵי עֵינַי לֹא יִתְּנוּ עַד / בְּחַסְדָּם יִהְיוּ הֵם גֹּאֲלֵיהֶם,

וְאִמְרֵי פִי לְהַזּוֹת מֵי רְצוֹנָם / עֲלֵיהֶם מִלְּבַד חֵן מִשְׁאֲלֵיהֶם,

וְלִפְנֵיהֶם לְאַט צֶדֶק יְהַלֵּךְ / וְגַם יָשִׂים לְדֶרֶךְ מַעְגְּלֵיהֶם.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
