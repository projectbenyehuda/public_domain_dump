את חטאי נוכח פני שמתי

לקרובה א'

אֶת חֲטָאַי נֹכַח פָּנַי שַׂמְתִּי כִּי אַחֲרֵי שׁוּבִי נִחַמְתִּי: ירמ' לא,יח

מְחֵה שַׁדַּי הַיּוֹם בְּקוּמִי חַטָּאתִי וְחַטַּאת עַמִּי: דנ' ט,כ

המלך הקדוש

בָּאתִי לִשְׁפֹּךְ נֶפֶשׁ נְכֵאָה בְּעַד הַשְּׁאֵרִית הַנִּמְצָאָה: ישע' לז, ד

זְדוֹנֵנוּ כְמוֹ דָוָה תִּזְרֵם פְּשָׁעֵינוּ אַתָּה תְכַפְּרֵם: תה' סה,ד

המלך הקדוש

גְּבֹהָה דִּבֶּר צָר בְּעַד צוּר מַפְגִּיע גִּבּוֹר לֹא יוּכַל לְהוֹשִׁיעַ: ירמ' יד,ט

מחה

דִּינִי הָאֵר וְיִרְאוּ בְעֵינֵיהֶם דְּבַר מִי יָקוּם מִמֶּנִּי וּמֵהֶם: ירמ' מד,כח

זדוננו

הַעַל רִמָּה גָּדֵר תִּפְרֹץ הֶעָלֶה נִדָּף תַּעֲרוֹץ: אי' יג,כה

מחה

וּנְקַלֹּתִי עוֹד מִפְּנֵי זְדוֹנִי וְהָיִיתִי שָׁפָל בְּעֵינָי: ש"ב ו,כב

זדוננו

זְדוֹנוֹתַי שָׂמוּנִי מְשִׁסָּה זַעַף אֲדֹנָי אֶשָּׂא: מי' ז, ט

מחה

חָנֵּנִי אֵל כִּי נִבְהֲלוּ עֲצָמָי חֲדַל מִמֶּנִּי כִּי הֶבֶל יָמָי: איו' ז, טז

זדוננו

טַהֲרֵנִי כִּי חַטָּאתִי גְדוֹלָה טָבַעְתִּי בִּיוֵן מְצוּלָה: תה' סט,ג

מחה

יָהּ נֶגְדְּךָ בָּאתִי שַׁח עֵינַיִם יָגֵעַ וּרְפֵה יָדַיִם: ש"ב יז,ב

זדוננו

כְּנֹאד בְּקִיטוֹר לִבִּי בִּמְעָלָיו כִּי אֲדֹנָי סָר מֵעָלָיו: שופ טז,כ

מחה

לָעֲגוּ עָלַי אַנְשֵׁי דָמִים וּמִרְמָה לֹא יָדַעְתִּי אֵי מִזֶּה הֵמָּה: ש"א כה, יא

זדוננו

מָאֲסוּ בִי וְשֵׁן חָרְקוּ עָלַי חָרֹק מִפָּנַי לֹא חָשְׂכוּ רֹק: איו' ל,י

מחה

נִחַמְתִּי עַל עֲוֹנוֹת סְתָרָי נָשָׂאתִי חֶרְפַּת נְעוּרָי: ירמ' לא,יח

זדוננו

סִבַּבְתִּי פָנַי ונַפְשִׁי לֹא אָמִישׁ שַׂמְתִּי פָנַי כַּחַלָּמִישׁ: ישע' נ,ו

מחה

עַל חַטֹּאתַי דָּלְפָה נַפְשִׁי עָצְמוּ מִשַּׂעֲרוֹת רֹאשִׁי: תה' ט, יג

זדוננו

פָּחַדְתִּי וּבְזָכְרִי עֲוֹנִי פַּלְגֵי מַיִם יָרְדוּ עֵינָי: שם קיט,קלו

מחה

צָמֵאתִי וּלְטַל סְלִיחָה מְחַכֶּה צַמְתִּי וָאֶבְכֶּה: ש"ב יב, כב

זדוננו

קְרָבַי הָמוּ לִי כִּי נִכְלַמְתִּי קַמְתִּי מִתַּעֲנִיתִי: עז' ט, ה

מחה

רַחוּם אַל תִּהְיֶה יוֹם צָרָתִי רָחוֹק מִישׁוּעָתִי: תה' כב, ב

זדוננו

שׁוֹבְבֵנִי לָשֶׁבֶת בָּאֳהָלִים שֹׁקֵט וּבֹטֵחַ וְאֵין מַכְלִים: שופ' יח, ז

מחה

תְּבַשֵּׂר כִּי רָצִיתָ בָנֶיךָ תָּכִין לִבָּם תַּקְשִׁיב אָזְנֶךָ: תה' י, יז

זדוננו

מְצַעֵק: אָבִי! בִּכְאֵב לִבּוֹ מִבּוֹר אֵין מַיִם בּוֹ: זכר' ט, יא

מחה

שְׁלַח אֵלָיו פֶּן בְּאָפְלוֹ יִסָּפֶה שֶׁמֶשׁ צְדָקָה וּמַרְפֵּא: מלא' ג, כ

זדוננו

הַשְׁמִיעַ שָׁלוֹם מִשְּׁחָקִים לַקְּרֹבִים וְלָרְחֹקִים: דנ' ט, ז; ישע' נז,יט

מחה

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
