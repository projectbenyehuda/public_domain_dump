עיני עננים בבכי נמאסו

עיני עננים בבכי נמאסו / ר' משה אבן-עזרא

עֵינֵי עֲנָנִים בַּבְּכִי נִמְאָסוּ / עַד עִקְּבֵי גַן־הַהֲדַס נֶחְמָסוּ

בָּכוּ בְחֹזֶק נֽוֹזְלִים וַיִּשְׂחֲקוּ / שִׂפְתֵי עֲרֻגוֹתָיו אֲשֶׁר כָּעָסוּ

עָנָן אֲשֶׁר הִגְשִׁים וְשַׁחַק רָעֲמוּ / פָנָיו וְאַפֵּי הָאֲדָמָה שָׂשׂוּ

הִרְעִים וְאָכֵן בַּעֲבוּר הַרְעִים פְּנֵי / שַׁחַר וְעַפְעַפָּיו לְקוֹלוֹ נָסוּ

כַּנְפֵי בְרָקָיו אָחֲזוּ כַנְפוֹת פְּנֵי / חֶלֶד וְצִבְאֵי לַֽיְלָה פָרָשׂוּ

רָצִים וְשָׁבִים כַּנְּשָׁרִים נֶחְפְּזוּ / הָעֵת עֲלֵי־אֹכֶל בְּאֶרֶץ טָשׂוּ

לוּלֵי בְנֵי־עָב יָֽרְדוּ לֹא־נַעֲלוּ / צִצֵּי אֲמִירֵי־גַן וְהִתְנוֹסָסוּ

הָיוּ אֲפִילוֹת עַד מְטָר־גֶּשֶׁם וְחֹם / שֶׁמֶשׁ לְהַרְאוֹת זָהֳרָם נָגָשׂוּ

בִּשְׁמֵי עֲרֻגֹת זָֽרְחוּ אָכֵן יְדֵי / צָפוֹן צְפוּנֵי מֶרְקְחֵיהֶם שָׁסוּ

רֵיחַ אֲבַק רוֹכֵל וְהָרוּחַ כְּמוֹ / רָכִיל בְּסוֹדֵיהֶם אֲשֶׁר כָּמָסוּ

הֶרְאוּ יְפִי־פָנִים אֲשֶׁר בַּאְפֵר סְתָיו / חָפוּ יְמֵי הַקֹּר וְהִתְחַפָּשׂוּ

הוֹדוּ בְּלִי לָשׁוֹן לְעָבִים יָֽנְקוּ / אֹתָם וְעֵינֵיהֶם עֲלֵיהֶם חָסוּ

כָּרִים מְעִיל תּוֹלָע וְחוּר עָטוֹ עֲלֵי־ / בֶגֶד תְּכֵלֶת בַּעֲדָם פָּרָשׂוּ

עֵינַי בְּעֵינֵיהֶם פְּלָאִים יֶחֱזוּ / כִּי מַעֲשֵׂה תַשְׁבֵּץ פְּנֵיהֶם עָשׂוּ

בָּם גַּרְגְּרֵי שֹׁהַם וְהֵם לַחִים הֲכִי / אֹתָם רְסִיסֵי לַֽיְלָה רָסָסוּ

מַה־יָּֽקְרוּ לוּ הַחֲרוּצִים לָֽקְטוּ / אֹתָם וְכַפֵּי רֹֽכְלִים כָּנָסוּ

נִצְהַל בְּמַשְׂכִּיתָם וְקוֹלָם יִתְּנוּ / עוֹף בֵּין עֳפָאֵיהֶם וְיִתְעַלָּסוּ

נָשִׂישׂ בְּקוֹל הַסִּיס וְתֹר אִם־יַעֲנֶה / כִּנּוֹר וְעוּגָב עֹֽגְבִים יִמְאָסוּ

חֵי סֹֽבְאִים סָבְאוּ בְּבַת־כֶּרֶם אֲשֶׁר / אֹתָהּ נְדִיבִים לַעֲלֹץ אֵרָשׂוּ

תִּמְשׁוֹל בְּרֹאשׁ שָׂרִים וּמִן־אָז פַּעֲמֵי / דַלִּים גְּאוֹנָהּ דָּֽרְכוּ עָסָסוּ

זָקְנָה וְאוֹר לֹא־רָאֲתָה מִיּוֹם אֲשֶׁר / אֹתָהּ בְּאוֹצְרֹתָם יְדֵיהֶם עָשׂוּ

שִׁפָּה זְמָן עַצְמָהּ וְכִמְעַט נֶעְלְמָה / לוּלֵי גְבִיעִים זָהֳרָהּ עָמָסוּ

אֵשׁ בֹּעֲרָה אַךְ הַֽיְגוֹנִים בִּעֲרָה / מֵעָם בְּרֶשֶׁת הַנְּדֹד נִתְפָּשׂוּ

עֶדְנֵי בְנֵי עֶדְנָה וְאוֹן נַפְשׁוֹת מְתֵי / אוֹנִים לְחוּל זֹקֶן מְאֹד גָּרָסוּ

בִּי מִזְרְקֵי סַפִּיר כְּמַיִם קָֽפְאוּ / בָּם גַּחֲלֵי הָאֵשׁ אֲבָל נָמָסּוּ

אֵיךְ גָּֽמְלוּ חֶסֶד עֲלֵי כָל־כֹּאֲבִים / וַאְנִי לְבַדִּי מֵאֱנוֹשׁ חָמָסוּ

אֶשְׁתֶּה וְאֶשְׁתָּאֶה וְאֶמְהֹל אֶת־עֲסִיס / כּוֹסִי בְּמֵי עֵינַי אֲשֶׁר נִמְאָסוּ

אֵיכָה לְפִי־רֶגַע גְּבִיעִים יֽוּכְלוּ / לִרְפּוֹא אֲשֶׁר לַיִל וְיוֹם הָרָסוּ

יָלְדוּ בְנוֹת יָמִים עֲלֵי בִרְכַּי וְלֹא / גָמְלוּ וְיַלְדֵיהֶם בְּצִלִּי חָסוּ

עַד עַל־שְׁמִי יִקָּֽרְאוּ קֹרֹת וְלִי / יִתְיַלְּדוּ תוּגוֹת וְיִתְיַחָסוּ

עָדַי לְבַד עָדוּ לְנַצֵּל אֶת־צְבִי / עֶדְיִי וְחַיָּתִי בְּאַף רָמָסוּ

בָּאתִי בְאַחְרִית הַזְּמָן עֵת גָּֽמְרוּ / כָל־הַחֲסִידִים וֶאֱמוּנִים פָּסּוּ

וּבְנֵי בְלִי־שֵׁם גָּֽבְהוּ כַּאְשֶׁר מְתֵי / הַשֵּׁם פְּלָאִים כָּֽרְעוּ קָרָסוּ

סָפוּ כְפִירֵי דָת וְתַחְתָּם נִבְרְאוּ / חֶרְפַּת בְּנֵי־אָדָם אֲבָל רָמָשׂוּ

שֻׁלְחַן כְּזָבִים עָֽרְכוּ בּוֹ מַאֲכַל / קֶשֶׁר וְלֶחֶם לֹחֲמִים פָּרָסוּ

זַכּוּ בְעֵינֵיהֶם וְהֵם יִטַּמְּאוּ / בִּדְמֵי כְסִילוּתָם וְיִתְבּוֹסָסוּ

רֻכְסֵי נְתִיבִים בֶּאֱרֹג שִׁיר דָּֽרְכוּ / לֹא־סֻקְּלוּ לָהֶם וְלֹא־פֻלָּסוּ

קוּרָיו כְּעַכָּבִישׁ וְלָזֹאת יִלְעֲגוּ / לוֹ שֹֽׁמְעֵי שִׁמְעוֹ וְיִתְקַלָּסוּ

לוּ חָֽכְמוּ בַּצּוּר קְלוֹנָם טָֽמְנוּ / וּסְחִי חֲרוּזֵיהֶם בְּעָפָר כָּסוּ

אֶבְכֶּה פְּלִיטֵי תֹם אֲלֵיהֶם נָֽגְעוּ / נִגְעֵי בְנֵי־הֵפֶךְ וּבָהֶם פָּשׂוּ

וַיְגֹאֲלוּ בָהֶם וְנַפְשָׁם טִנְּפָה / חֶלְאַת פְּעָלֵיהֶם וְלֹא־כֻבָּסוּ

עַד־נִדְלְחוּ מֵי מַעְיְנֵי־צֶדֶק אֲשֶׁר / יִשְׁתּוּ וְנַחְלֵי הָאֱמֶת נִרְפָּשׂוּ

וּשְׁדֵי אֲהָבִים מָעֲכוּ שׁוֹנִים בְּיַד / בֶּגֶד וְדַדֵּי הַֽיְדִידוֹת עָשׂוּ

קַמְתִּי לְהָשִׁיב אֶת־גְּמוּלֵיהֶם אֲבָל / אֹתִי מְלִיצֵי אַהֲבָה אָנָסוּ

קַח מִגְדְּלֵי בֶצֶר לְהִבָּצֵר הֲכִי / שׁוּרֵי בְנוֹת־שִׁיר בֹּעֲרִים נָתָסוּ

אוֹ טַבְּעוֹת כֶּסֶף יְדֵי אָמָּן לְיַד / שֶׁמֶשׁ בְּחוּטֵי מַאֲמָר רָכָסוּ

שָׁרִים לְדַבֵּר אִם־יְעוּשׁוּן יִקְרְאוּ / לָהֶם בְּנֵי־עַיִשׁ פְּתָאיִם הָסּוּ.
