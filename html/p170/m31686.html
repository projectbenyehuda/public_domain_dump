פְּנֵי שִׂמְחַת מְתֵי תֵבֵל

לאבא יחיי – יהודה – בן אלרב על מות אשתו


פְּנֵי שִׂמְחַת מְתֵי תֵבֵל דְּאָגָה / וְחֶמְאַת מִיץ מְשׂוֹשֶׂיהָ תְּלָאָה,

וְכָל גֶּבֶר אֲשֶׁר יִזְרַע שְׂשׂוֹנִים / יְגוֹנִים יִהְיוּ לָהֶם תְּבוּאָה,

וְיֵשׁ יָשֵׁן עֲלֵי מִטַּת הֲנָחָה / תְּעִירֵהוּ לְעֵת בֹּקֶר תְּשׁוּאָה.

וְתֵבֵל תַּעֲשֶׂה תֶבֶל בְּזָדוֹן, / וְאִם תֵּיטִיב מְעַט תִּהְיֶה שְׁגִיאָה,

כְּחֶפְצָהּ תַּעֲשֶׂה, אַךְ לֹא כְחֵפֶץ / יְלָדֶיהָ וְלֹא כִרְצוֹן בְּרִיאָה.

וּמִי קִוָּה הֱיוֹת חַיָּיו כְּלִבּוֹ – / יְקַו כִּי אֵשׁ בְּתוֹךְ מַיִם מְצוּאָה.

וְנַפְשָׁהּ מֵאֲשֶׁר תּוֹלִיד תְּשַׁכֵּל, / וְאֵיךְ חַיִּים לְיַד נָכְרִי מְבִיאָה?

וְרוֹפֵא לֹא יְהִי חוֹבֵשׁ פְּצָעָיו – / לְחוֹלֶה אֵיךְ יְהִי מַמְצִיא רְפוּאָה?

אֱמֶת, חַיִּים וְהַמָּוֶת כְּצָרוֹת, / אֲהוּבָה זֹאת וְזֹאת מֵאָז שְׂנוּאָה,

וְאָכֵן הַזְּמָן עִוָּה שְׁפָטָיו / וְנָתַן פִּי־שְׁנַיִם לַשְּׂנִיאָה.

וְהַלֵּילוֹת כְּמוֹ הָרוֹת, וּמִי זֶה / אֲשֶׁר בָּן מַה בְּבֶטֶן הַמְּלֵאָה?

בְּיָדִי כוֹס בְּמֵי עֵינִי רְוָיָה / וְלֶחְיִי מִיְקוֹד לִבִּי צְמֵאָה,

לְעֶבְרָה עָבְרָה עַל הַנְּפָשׁוֹת – / וְאֵיךְ נִפְלֵאת לְשׁוֹמַעַת וְרוֹאָה!

לְבָנָה עָרְבָה, אַךְ בָּעֲרָבָה, / וְחָפְרָה בַעֲנַן שַׁחַת וּבָאָה,

וְנוֹשֵׂאת חֵן לְכָל עַיִן – בְּפֶתַע / אֱלֵי בַדֵּי שְׁאוֹל הָיְתָה נְשׂוּאָה,

וְאֵיךְ עָפָר יְהִי אֹכֵל עֳפָרִים / וּמָוֶת אֵיךְ יְהִי טֹרֵף לְבִיאָה?

וְשֶׁמֶשׁ עִם בְּנוֹת עַיִשׁ יְנוּדוּן, / וְאִם בָּהּ הָיְתָה שֶׁמֶשׁ מְקַנְּאָה.

עֲדִינָה לָבְשָׁה מַלְבּוּשׁ עֲנָוָה / וּפָשְׁטָה כֻּתֳּנוֹת גָּאוֹן וְגֵאָה,

וְרַכּוּ דַבְּרוֹתֶיהָ לְאַלּוּף / נְעוּרֶיהָ כְּנֹפֶת צוּף וְחֶמְאָה.

אֲהָהּ, בִּכְלוּא יְגוֹנִים הַנְּפָשִׁים / לְבָרֶקֶת בְּבוֹר קֶבֶר כְּלוּאָה.

אֱמֹר לְאִישׁ אֲשֶׁר יִקְרָא בְאָזְנֵי / יְהוּדָה זֹאת – וְאֵלָיו הַקְּרִיאָה:

לְאַט, כִּי הָיְתָה נַפְשׁוֹ קְשׁוּרָה / בְּמוֹ נַפְשָׁהּ וְעֵינוֹ לָהּ תְּלוּאָה,

אֲבָל יֵאוֹת לְהִתְכַּבֵּד, וְאַל נָא / כְּבוֹד נַפְשׁוֹ תְּהִי הֹוָה מְנִיאָה.

וְאַל עַל הַזְּמָן יַרְבֶּה תְלוּנוֹת, / וּמַה תּוֹעִיל עֲלֵי סָכָל תְּנוּאָה?

וּמַה בֶּצַע בְּהַכְאִיב הַנְּפָשִׁים / וּמִי יוּכַל נְשֹׂא רוּחַ נְכֵאָה?

וּפֶגַע כָּל־אֱנוֹשׁ מָוֶת, וּמִקְרֵה / מְכַשֵּׁפָה כְּמִקְרֵה הַנְּבִיאָה,

וְיָמוּת רָשׁ וּמִרְעֵהוּ חֲצִירוֹ / כְּמוֹת עָשִׁיר וּמַאְכָלוֹ בְרִיאָה.

וְהִיא תֵבֵל אֲנָשֶׁיהָ תְּבַלֶּה / וְאוֹתָם אֶל מְקוֹם קָאָה מְקִיאָה,

וְכָל נוֹצָר עֲפַר מָוֶת יְלַחֵךְ / וְכָל נֶפֶשׁ בְּיֵין אָסוֹן סְבוּאָה.

וְהַמַּשְׂכִּיל – אֲשֶׁר אִזֵּן וְחִקֵּר / וְשָׂם לֵב לַחֲזוֹת בָּאוֹת בְּמַרְאָה,

וְקִבֵּץ עִנְיְנֵי שֵׂבֶר לְנַפְשׁוֹ / וְהֵפִיץ מַחְבְּרֵי שֶׁבֶר וְהִפְאָה,

וְהִזְרָה אֶת־כְּאֵב לִבּוֹ וְשָׂם אֶת / מְגִנָּתוֹ בְּכַף רֶגֶל מְחוּאָה.

וְשָׁלוֹם אִם יְחוֹפֵף עַל יְהוּדָה – / אֲזַי חַטַּאת זְמָן תִּהְיֶה נְשׂוּאָה.

וְצוּר הִפְלִיא מְאֹד אֵידוֹ יְצַוֶּה / חֲסָדָיו לַעֲשׂוֹת עִמּוֹ פְלִיאָה,

וְתַמָּה הַמְהַלֶּכֶת נְכוֹחוֹ / תְּהִי אֶל גַּן עֲדָנֶיהָ קְרוּאָה,

וְצוּר קִבְרָהּ כְּמֵי עָנָן יְרַוֶּה / וּמִשֹּׁד לֹא תְהִי לָעַד יְרֵאָה.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
