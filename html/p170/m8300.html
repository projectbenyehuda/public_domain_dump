לנטע השכל ונצר היושר

לנטע השכל ונצר היושר / ר' משה אבן עזרא

אל רבי יאשיה בן בזאז

לְנֶטַע הַשֵּׁכֶל

וְנֵצֶר הַיּשֶׁר

וְגֶזַע הַבִּין וְחֹטֶר הַכּשֶׁר

שְׁתִיל הַדַּעַת וְהַמְּזִמָּה

נַחַל נֹבֵעַ מְקוֹר חָכְמָה

עֵץ נֶחְמָד לְמַרְאֶה וְטוֹב לְמַאֲכָל, וּפִרְיוֹ מָתוֹק לִשְׂפָתַיִם

וְתַאֲוָה-הוּא לָעֵינַיִם

שָׁתוּל עַל-פַּלְגֵי מָיִם

לֹא יִרָא חֹם וּמִגְעֶרֶת וְלֹא יִדְאַג בִּשְׁנַת בַּצֹּרֶת

עַל-מֵי תְהוֹם מַטַּעְתּוֹ

וּבֵין עֲבוֹתִים צַמַּרְתּוֹ

מַטַּע אֲדֹנָי לְהִתְפָּאֵר בְּצִבְיוֹ

לֹא-יִבּוֹל עָלֵהוּ וְלֹא יִתֹּם פִּרְיוֹ

כָּל-עֵץ בְּגַן-אֱלֹהִים לֹא-דָמָה אֵלָיו בְּיָפְיוֹ

הַכּוֹכָב הַמַּזְהִיר

הָאוֹר הַבָּהִיר

מר' ור' יֹאשִׁיָּה הֶחָכָם וְהַנָּבוֹן

לִפְנֵי שֶמֶשׁ שְׁמוֹ יִנּוֹן

וּמִשְׂרָה עַל-שִׁכְמוֹ

בִּדְרָכָיו מַשְׂכִּיל וַאדֹנָי עִמּוֹ

בֶּן-כְּבוֹד אַבִּיר הָרֹעִים

וְרֹאשׁ הַיְדֻעִים

רַבִּי יַעֲקֹב, יְבַלֶּה בַטּוֹב יְמוֹתָיו

וּבַנְּעִימִים כָּל-שְׁנוֹתָיו.

מַה-נִּכְבָּד הַיּוֹם מִיָּמִים

וּמַה-נֶּחְמָד בְּעִתּוֹת עוֹלָמִים

וּמַה-יָּקָר בְּעֵינֵי כָל-יְקוּמִים

יוֹם בָּאוּ כּוֹכְבֵי פִידוֹ

וְזָרְחוּ – כְּסִילֵי הוֹדוֹ

וּבָרְחוּ – אָפְלֵי תוּגוֹתָיו

וְצָהֲלוּ פְּנֵי עֲלִיצוֹתָיו

יוֹם אֲנִי יָשֵׁן וְלִבִּי עֵר

כְּתַנּוּר בֹּעֵר

וְכַיָּם הוֹלֵךְ וְסֹעֵר

וְעָלַי אֵינֶנּוּ מִתְרַפֵּק

וְלֹא-יָכֹל לְהִתְאַפֵּק

וְקוֹל דּוֹדִי דוֹפֵק

אָז נִפְקְחוּ עֵינֵי שְׂשׂוֹנַי

וְרָחֲשׁוּ שִׂפְתֵי רְנָנַי

וּמוֹרָשֵׁי לְבָבִי הַשֹּׁאֲפִים

יָצְאוּ מְבֹהָלִים וּדְחוּפִים

כִּי רָאוּ הַיּוֹם שֶׁקִּוּוּ

וּמָצְאוּ הָעֵת אֲשֶׁר אִוּוּ

כִּי לְשִׁקּוּי הָעֲצָמוֹת בְּשׂרָה נִפְלָאָה

וְעֵץ חַיִּים תַּאֲוָה בָאָה

וְטֶרֶם רְאוֹת סִפְרוֹ – הֵפִיחַ רֵיחֵי בְשָׂמָיו

וְטֶרֶם קְרֹא שִׁירוֹ – נָשְׁבוּ רוּחוֹת שְׁלוֹמָיו

וְהָעֵת אֲחַזְתִּיו

כַּחוֹתָם עַל-לִבִּי שַׂמְתִּיו

חֻקָּיו לִשְׁמֹר

וּמִלָּיו לִגְמוֹר

כִּי יָדַי נָטְפוּ-מוֹר

וְעֵת בָּאתִי פְתָחָיו

וְשָׁבַרְתִּי בְרִיחָיו

לֶאֱרוֹת תַּפּוּחָיו

וְלִשְׁתּוֹת יֵינֵי מֶרְקָחָיו

לִרְעוֹת בַּגַּנִּים

וְלִלְקֹט שׁוֹשַנִּים

וְהִנֵּה הוּא תַאֲוָה לָעֵינַיִם

יָם גָּדוֹל וּרְחַב יָדַיִם

פַּחַד גָּדוֹל יְסוֹבֲבֶנּוּ

וְצִי אַדִּיר לֹא יַעַבְרֶנּוּ

נְבוֹנִים זָעוּ מִגַּלָּיו

וַיִּירְאוּ מִגֶּשֶׁת אֵלָיו

וְעֵת עָמַדְתִּי עַל-שְׂפָתוֹ

וּבָאתִי עַד-תְּכוּנָתוֹ

מָצָאתִי מַעְיַן גַּנִּים

מֵימָיו נֶאֱמָנִים

כְּנַחַל עַל-פְּנֵי-אֶרֶץ נִפְשַׁט

וּמֵי הַשִּׁלֹחַ הַהֹלְכִים לְאַט

מָתְקוּ מֵימֵי תְהוֹמָיו

וְנָפְצוּ אַבְנֵי לְשָׁמָיו

וּמֹר וְנֹפֶת יִגְרְשׁוּ מֵימָיו

קוֹל סַעֲרוֹתָיו – כְּקוֹל צִנּוֹר

וְרַעַם גְּבוּרוֹתָיו – כְּנֵבֶל וְכִנּוֹר

וְרֵיחָיו רֵיחֵי סַמִּים

וְרוּחוֹתָיו כַּעֲרוּגַת הַבְּשָׂמִים

וְכִרְאֹתִי פְנִינָיו צָהֲלוּ פָנַי

וְכַאֲשֶׁר שָׁמַעְתִּי נְאֻמוֹ נָעֲמוּ אָזְנַי

וְהָעֵת טָעַמְתִּיּ מֶתֶק דִּבְשׁוֹ אֹרוּ עֵינַי

וְאָז קַרְתִּי וְשָׁתִיתִי מַיִם בְּצִמְאָה

נַהֲרֵי נַחֲלֵי דְּבַשׁ וְחֶמְאָה

וָאָבוֹא אֶל-אוֹצְרוֹת מַטְמוֹנִים

וָאֶלְקֹט מַחֲמַדֵּי אֲבָנִים

וּמֶשֶׁךְ חָכְמָה מִפְּנִינִים

וָאֹמַר בַּמָּה אֲקַדֵּם אֶת-חֲסָדָיו – וְהֵם הִקְדִּימוּנִי

וְאִכַּף לִפְנֵי מַעְבָּדָיו – וּמֵחָלְיִי הֶחֱלִימוּנִי

וַאֲשַׁלֵּם יֹשֶר פְּעָלָיו – וְעַל-עָמְדִי הֱקִימוּנִי

וְאָז יְעָצוּנִי לִבִּי וְכִלְיוֹתַי

וְיִסְּרוּנִי מַחְשְׁבוֹתַי

לָשׂוּם רַגְלַי

תַּחַת מִלַּי

וּרְכָבַי

תַּחַת מִכְתָּבַי

וּמִצְעָדַי

בִּמְקוֹם יָדַי

וְלָרוּץ אֶל-שְׁפָיִם

כְּאַחַד הַצְּבָיִם

וּנְקַלֹּתִי עוֹד מִזֹּאת

נֹעַם פָּנָיו לַחֲזוֹת

וְהָיָה שָׁפָל בְּעֵינָיו

לִרְאוֹת אֶת-פָּנָיו

אֲבָל הַזְּמָן חִלֵּק תֹּלְדוֹתָיו

וְהָרֶגַע מָנַע אֹדוֹתָיו

וְיֹתֵר מֵהֵמָּה כִּי יוֹם אֲשֶׁר הִגִּיעַ כְּתָבוֹ אֵלַי – נִתְגַּלְגַּל עָלַי – מְאֹרָע

וְנִסְתַּבֵּב מִקְרֶה רָע

וְעוֹד אֲנִי בוֹ שׁוֹבֵר

הָיָה כְצֵל עוֹבֵר

וּבְעוֹד עֵינִי שְׁזָפוֹ

לֹא עָמַד כִּי אֲדֹנָי הֲדָפוֹ

מַצְדִּיק אֲנִי אֶת-הַדִּין עַל-רָעַת הַזְּמָן וְטוּבוֹ

וְעַל-זֶה וְכַיּוֹצֵא בוֹ

וּבָטוּחַ אֲנִי כִּי יֵשׁ בְּרָע תַּמְרוּק פֶּשַׁע

וּסְלִיחַת רֶשַׁע.

וְכִרְאֹתִי כֹּחַ יְדִידִי וּגְאוֹנוֹ

וְעֹצֶם שִׂכְלוֹ וְאוֹנוֹ

וְצַחוֹת פִּיו וּלְשׁוֹנוֹ

וְכִי כָל-מִדְבְּרֵי עִנְיָנָיו

דָּבָר דָּבֻר עַל-אָפְנָיו

שָׂמַח לִבִּי

וַיָּגֶל כְּבוֹדִי

וְחֻקָּה עַל-אִישׁ לִשְׂמֹחַ בְּבֶן-גִּילוֹ

וְעַל-כָּל-מִין לָשִׂישׂ עִם הַדּוֹמֶה-לּוֹ

וְכָל-שֶׁכֵּן לְבָבִי הַנִּכְאָב

אֲשֶׁר לִבְנֵי שֵׂכֶל יִתְאַב

וּמִתְּשׁוּקָתָם יִדַּל וְיִדְאַב

כִּי-נִשְׁאַרְתִּי בְרִמֹּן פֶּרֶץ

כְּגֵר תּוֹשָׁב בָּאָרֶץ

וּבְעִיר יָרַד הֲדָרָהּ וַהֲמוֹנָהּ

כְּצִפּוֹר נוֹדֶדֶת מִקִּנָּהּ

וּבְדוֹר סָג וְנֶאֱלַח

כְּעוֹף-נוֹדֵד קֵן מְשֻׁלָּח

אָבַד מָנוֹס מִמֶּנִּי וְאֵין דּוֹרֵשׁ וּמַזְכִּיר

אַבִּיט עַל-יָמִין וְאֶרְאֶה וְאֵין-לִי מַכִּיר

וְאַף-עַל-פִּי שֶׁאֲנִי בְּאֶרֶץ תֹּלְדוֹתַי

וְעִיר קִבְרוֹת אֲבוֹתַי

צֹעֵק אֲנִי מִן-הַמֵּצַר

כִּי-הָיְתָה עָלַי כְּחוֹתָם צַר

וְכִי עוֹד יַד-אֲדֹנָי נְטוּיָה עָלֶיהָ

וְלֹא-שָׁב חֲרוֹן אַפּוֹ מֵאֲצִילֶיהָ

עַד כִּי-סָפוּ קְרִיאֵי מוֹעֵד, וְתַמּוּ אַנְשֵׁי הַשֵּׁם, וְכָלוּ בְנֵי הַיַּחַשׂ יַחַד

וְהַשְּׂרִידִים לֻקְּטוּ לְאַחַד אֶחָד

וּמִי יִתֵּן הֱיוֹתִי נָע וְנָד בָּאֲדָמָה

וְיַעֲבֹר עָלַי מָה.

וְאֹתוֹ מְאֹרָע שֶׁקִּדַּמְתִּי אֵחַר תְּשׁוּבָתִי

וְעִכֵּב אִמְרָתִי

לָכֵן אֲנִי דֹרֵשׁ מִיקָּרוֹ

וְשֹׁאֵל מֵהֲדָרוֹ

שֶׁאַל יַאֲשִׁימֵנִי, כִּי הָאֵל יֹדֵעַ כִּי מִתּוֹךְ קֹצֶר רוּחַ כָּתַבְתִּי טוּרִים אֵלּוּ, וְלָזֹאת אָחַזְתִּי דֶּרֶךְ קְצָרָה

וְלֹא-חִבַּרְתִּי מִלָּה יְשָׁרָה

וְנִסְמַכְתִּי עַל-אַחֲוָתוֹ

וְנִשְׁעַנְתִּי עַל-תֻּמָּתוֹ

שֶׁיָּדִין אֹתִי לְכַף זְכוּת, כִּי-הַזְּכוּת תְּלוּיָה בְמִדּוֹתָיו

וְהָעֲנָוָה אַחַת מִצַּלְעוֹתָיו

וּלְפָאֲרֵנִי –בְּמִלָּיו הַמְּתוּקִים

וּלְכַבְּדֵנִי – בְּשִׁירָיו הַחֲזָקִים

כְּדֵי שֶׁתִּכָּפֵל הָאַהֲבָה

וְתֶעֱדַף הַיְדִידֹת וְהַחִבָּה

וּלְהוֹדִיעֵנִי בְּמִסְתּוֹרֵי לִבּוֹ

וּמַצְפֻּנֵי חֻבּוֹ.

וַאֲשֶׁר הִשְׂבִּיעַנִי –מִפְּרִי אֲהָבָיו

יְחַדֵּנִי – בְשִׂמְחָה אֶת-פָּנָיו

וַאֲשֶׁר הִרְוַנִי – מִנַּחֲלֵי דֹדָיו

יֱזַכֵּנִי – לַמְתִּיק סוֹד יַחְדָּיו.

וְחֶדְוַת יְדִידִי וְאַלּוּפִי עַל-הַיֶּלֶד אֲשֶׁר-חָנַן אֱלֹהִים אֹתִי, אָמְנָה הִיא לְהִתְעָרֵב בְּשִׂמְחָה זוּ, כִּי-הֵם לִי יְלָדִים / וְלוֹ עֲבָדִים

וְהָאֵל יִתֵּן יֹצְאֵי יְרֵכוֹ כְמוֹתוֹ

לִנְחַל תּוֹרָתוֹ

וְלִירַשׁ תְּהִלָּתוֹ

וְיָשִׂים כַּיָּם תֹּלְדוֹתָיו

וֱצֶאֱצָאֵי מֵעָיו כִּמֱעוֹתָיו

לְבָאֵר דַּעַת וּמוֹעֵצָה

לְהָבִין מָשָׁל וּמְלִיצָה

וּלְלַמֵּד בְּיִשְׂרָאֵל חֹק וּמִשְׁפָּט.

וְהָאֹמֵר עֲצָתִי תָקוּם וְכָל-חֶפְצִי אֶעֱשֶׂה, יְאַמֵּץ יְדִידִי וּגְבִירִי וִיחַזֵּק יָדָיו

וְיִנְצֵרהוּ וִיכוֹנֵן צְעָדָיו

וְיִהְיֶה-לּוֹ חוֹמָה וּמִבְצָר

וְעֶזְרָת מִצָּר

צַעֲדוֹ אַל-יֵצַר

וּמַצָּעוֹ אַל-יִקְצָר.

וְיִשָׂא שָׁלוֹם לְאֵין קֶצֶה וְקֶצֶב

וּבְאֵין יָגוֹן וְעֶצֶב

וּמַשְׂאַת שְׁלוֹמִים

כְּמֵימֵי אֲגַמִּים

וְשֶׁפַע יַמִּים

אֶל-אָבִיו הָרֹאשׁ הָאַדִּיר ש"צ וַעֲלֵי אֶחָיו וּקְרוֹבָיו – הַנֶּחֱמָדִים הַחֲשׁוּקִים

וְעַל-דּוֹדָיו וַאֲהוּבָיו –הַקְּרֹבִים וְהָרְחֹקִים

וַעֲנָוִים יִירְשׁוּ אָרֶץ וְהִתְעַנְּגוּ עַל-רֹב שָׁלוֹם.
