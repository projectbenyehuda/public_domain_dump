לְבָבִי יֶהֱמֶה לִזְמָן

לר׳ חננאל, הוא אלחנן, בן ר׳ ישועה

לְבָבִי יֶהֱמֶה לִזְמָן שְׁאָפוֹ / וְעֹז מִשְׁבַּר שְׁאוֹנָיו כִּי אֲפָפוֹ,

יְעוֹפֵף בֵּין צְלָעָיו לוּ כְצִפּוֹר / יְהִי נָס מִפְּנֵי פֶרֶס רְדָפוֹ,

וְאִם יָרוּץ אֱלֵי אֹהֶל שְׂמָחוֹת – / יְלָדָיו יֹאמְרוּ: עוֹד מִי לְךָ פֹה?

בְּהֵעָצֵר מְטַר גֶּשֶׁם דְּמָעָיו / אֲזַי רוּחַ יְגוֹנוֹתָיו תְּנִיפוֹ.

כָּעָב תָּרִיק לְאַט, אוֹ יַד חֲנַנְאֵל / בְּעֵת יָגִיל לְהֵלֶךְ אִם שְׁזָפוֹ.

גְּבִיר יִקְרָא לְכָל חָרֵד וְיָרֵא / לְמַהֵר לַחֲסוֹת תַּחַת כְּנָפוֹ,

וְאִם יַמְרֶה זְמָן אֶת־מַאֲמָרוֹ – / בְּיָדוֹ הַחֲזָקָה רֹאשׁ יְשׁוּפוֹ.

כְּסִילִים נִכְלְמוּ נֶגְדּוֹ וְשֶׁמֶשׁ / מְאוֹרוֹ מִפְּנֵי פָנָיו אֲסָפוֹ.

וְהַחֶסֶד כְּמוֹ פֶגֶר וְהוּא רֹאשׁ, / וְאִם הוּא רֹאשׁ – חֲנַנְאֵל הוּא צְנִיפוֹ.

מְכַסֶּה סוֹד, וְאָכֵן סוֹד תְּעוּדָה / לְעֵינֵי כֹל כְּמוֹ מַיִם חֲשָׂפוֹ.

וְלֹא יִשְׁנוּ הֲלִיכוֹת מִפְעֲלוֹתָיו / וְרֹאשׁ חַסְדּוֹ בְּהִתְנַדֵּב כְּסוֹפוֹ,

וְאֵיךְ יִשְׁנֶה זְהַב אוֹפִיר וְכֶתֶם / בְּמַצְרֵף הָאֱמוּנָה אֵל צְרָפוֹ.

קְרָאַנִי גְבִיר שֹׁד, וְכִמְעַט / לְבָבִי מִבְּכִי מִגֵּו שְׁלָפוֹ

מְבַשֵּׂר כִּי אֳנִי דֹרֵךְ בְּלֵב יָם / בְּשָׁלוֹם – כַּאֲרִי הַיָּם טְרָפוֹ,

וְצִי אַדִּיר, אֲשֶׁר פָּרַשׁ לְנִסּוֹ / לְיַד מִזְרָח – לְשׁוֹן גַלָּיו חֲטָפוֹ.

וְהָיָה לוֹ לְיוֹם בֹּאוֹ לְאוֹמֵן / וְהוּא נִשָּׂא כְיוֹנֵק עַל כְּתֵפוֹ,

וְכִמְעַט קָט וְרוּחַ גְּדוֹלָה / וְחָזָק אֶל תְּהוֹמוֹתָיו דְּחָפוֹ,

וְגַם נִסְחַף לְפִי רֶגַע לְפָנָיו / וְלֹא עָמַד, לְמַעַן אֵל הֲדָפוֹ.

לְרֵע נֶאֱמָן עַל הָאֲהָבִים / אֲנִי בוֹכֶה אֲשֶׁר תּוֹכוֹ נְגָפוֹ,

וְעֵץ נֶחְמָד אֲשֶׁר נִשְׁבַּר וְהוּא לַח / וְעוֹדֶנּוּ בְאִבּוֹ צוּר קְטָפוֹ.

בְּיָדוֹ שָׂם יְדִידֶךָ בֶּן־אֲהָבַָיו / וְשָׁלַח מִשְּׁתִיל דּוֹדָיו עֲנָפוֹ,

שְׁלָחָהוּ אֱלֵי יָם הַתְּהִלּוֹת / וְאַךְ הַיָּם בְּיַד אֵיבָה גְרָפוֹ.

חֲמָדָהוּ וְשָׁלַל מַהֲלָלָיו / וְכַדּוּר אֶל קְצֵה אֶרֶץ צְנָפוֹ

וְאֵיכָה יַחֲמֹד הַיָּם לְגֶבֶר / שְׁתֵּי יָדָיו כְּמֵימֵי סוּף וְיָפוֹ,

וְאֵיךְ יִתְחַר בְּאִישׁ רָחַב לְבָבוֹ / מְאֹד מֶנְהוּ וְיֹאמַר כִּי תְקָפוֹ,

וְרֶפֶשׁ יִגְרְשׁוּ מֵימָיו וְהַחוֹל, / וְיַד זֶה מוֹר וְהָאֹדֶם לְחוֹפוֹ,

וּמִשְׁבָּרָיו מְאֹד מָרוּ, וּמִתּוֹךְ / יְמִין זֶה יַהֲלֹךְ כִּדְבַשׁ וְצוּפוֹ.

וְהַגַּרְזֶן עֲלֵי חֹצֵב הֲיוּכַל / לְהִתְפָּאֵר וּמַשּׂוֹר עַל מְנִיפוֹ?

וְלוּ רִשְׁפֵּי סְעִפֵּי רַעֲיוֹנָיו / יְשַׁלַּח בּוֹ אֲזַי כַּקַּשׁ שְׂרָפוֹ,

וְלוּ בוֹ מֵאֱנוֹשׁ יִתְגָּר חֲנַנְאֵל / בְּמֵי נִדְבַת יְמִינוֹ חִישׁ שְׁטָפוֹ.

אֲיַחֵל, הַגְּבִיר, לִזְמָן וְאוּלַי / יְהִי נִכְלָם וְיָשִׁיב אֶת־צְעִיפוֹ,

וּבִכְתָב יָדְךָ יַחְבֹּשׁ פְּצָעָיו / וְיִהְיוּ מַעֲשָׂיו נָגוֹף וְרָפוֹא.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
