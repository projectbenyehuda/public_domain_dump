רִכְבֵי נְדוּדִים בַּיְּדִידִים

אל אבו־אסחק [אברהם] בן אבו אלפרנ


רִכְבֵי נְדוּדִים בַּיְּדִידִים אָצוּ / אוֹ בַנְּפָשִׁים, כִּי לְאַבְּדָם רָצוּ,

אוֹ חָבְלוּ אוֹתָם, לְמַעַן בָּחֲלוּ / לָגוּר בְּפִגְרֵיהֶם וּבָהֶם קָצוּ.

הִתְעַבְּרוּ עִמָּם בְּנֵי יוֹם בַּעֲבוּר / פּוּצָם בְּכָל עֵבֶר וּבָם פָּרָצוּ.

סוֹד הֶעֱרִימוּ, אַךְ עֲלֵי סוֹד אֹהֲבִים, / לַרְחִיק קְרוֹבֵיהֶם, וְהִתְיֶעְצוּ.

הָיוּ בְהִתְחַבֵּר כְּמוֹ כִימָה וְהֵם / הַיּוֹם כְּיַלְדֵּי עָשׁ אֲשֶׁר נָפוֹצוּ.

רָפוּ יְדֵי כֹחָם, וּבַהְמוֹן מֵי בְכִי / יֵעָזְרוּ כִמְעָט וְיִתְאַמְּצוּ.

יִגַּר נְהַר דִּמְעִי וְיִתְגָּר בִּי כְאֵב / לִבִּי וְעָלַי לַחֲתֹף יִנָּצוּ.

יִשְׁתּוֹנְנוּ כִלְיוֹת אֲהָבִים בַּעֲדָם, / יִתְנוֹדְדוּ לִבּוֹת וְיִתְחַמָּצוּ.

אֵיכָה נְפָשִׁים חָבְרוּ יֵחָלְקוּ, / אֵיכָה אֲחָדִים נוֹצְרוּ יֵחָצוּ?

פְּקַח זְמָן עֵינֵי שְׂמָחוֹת כָּל יְמֵי / נֹעַר, וְעֵינֵיהֶם בְּנוֹתָיו עָצוּ,

עֵרוּ לְעֵת שֵׂיבָה, וְלוּ בִי נוֹעֲצוּ / יָשְׁנוּ שְׁנַת עוֹלָם וְלֹא הוּקָצוּ.

עָפוּ יְמֵי חֶבְרָה וְנִמְשָׁכוּ שְׁנֵי / עֶבְרָה בְּפִשְׁעֵיהֶם אֲשֶׁר לֹא רָצוּ.

יַלְדֵי זְמָן לַשְׁאוֹת מְגוּרֵיהֶם, אֲבָל / לַפְאוֹת שְׁכוּנֵיהֶם מְאֹד לָחָצוּ.

דִּמְעִי כְּמֵי עָב עָרְפוּ מֵאֵין דֳּמִי / עַל אַרְמְנוֹתֵיהֶם בְּאַף נָתָצוּ.

יִשֹּׁם וְיִשְׁרֹק כֹּל עֲלֵיהֶם יַעֲבֹר / וּלְתַחֲלוּאֵיהֶם אֱנוֹשׁ רָעָצוּ.

קוֹרֵא אֲלֵיהֶם יַעֲנֶה קוֹרֵא אֲשֶׁר / דָּגָר וְאִיִּים לַעֲנוֹת רָבָצוּ.

בִּלְתִּי פְנֵיהֶם אֶבְחֲרָה מָוֶת, הֲכִי / חַיַּי בְּחַיַּי אַחֲרֵיהֶם קָצוּ.

אַחַי וּמֵרֵעַי כְּצֵל נָטוּי עֲדֵי / לֵילוֹת בְּחוּרוֹתַי בְּלִי צִיץ נָצוּ.

שַׁבְתִּי וּמוֹרָשֵׁי לְבָבִי שָׁחֲרוּ, / אוֹחִיל הֲמוֹת מֵעַי וְיִתְפַּלָּצוּ.

גַּרְתִּי בְאֶרֶץ נֹד וּמֵאָז שָׁחֲחוּ / הָרֵי מְזִמּוֹתַי וְהִתְפּוֹצָצוּ.

אִמְרוּ: הֲכִי יִתְקַלָּסוּ יָמִים וְאִם / לִי יִלְעֲגוּ לֵילוֹת וְיִתְלוֹצָצוּ,

אוֹ אִם כְּמִקְרֵה רוֹזְנִים יִקְרֶה וְאִם / כֵּן רָעֲצוּ בָרִים וְגַם רוֹצָצוּ?

אָמְנָם מְתֵי חֶסֶד לְבַדָּם אָיְבוּ / וּבְעֵין תְּנוּאוֹת מֵאֱנוֹשׁ קֹרָצוּ.

אֶרְאֶה פְנֵיהֶם רָעֲמוּ לִקְרַאת מְתֵי / עָרְמָה וּבִרְאוֹת הַפְּתָאִים דָּצוּ.

שָׁלוֹם מְגוּרֵיהֶם וְלֹא הִתְנַכְּלוּ / יָמִים לְהָמִית בָּם וְלֹא חָרָצוּ,

יִתְנַפְּלוּ אַךְ עַל אֲשֶׁר יִתְנַפְּלוּ / לָבֹא בְמֵי צֶדֶק וְהִתְרֶחָצוּ,

מִיּוֹם בְּאַבְרָהָם בְּלִי חֵטְא פָשְׁעוּ / קַלּוּ פְעָלֵימוֹ, וְאִם נִמְרָצוּ.

אִם בַּעֲצֵי אֵל נָגְעוּ – אֵיךְ יִרְגְּזוּ / חָצִיר וְעֵשֶׂב לַעֲבוֹדָה צָצוּ?

אִם בִּקְדוֹשִׁים נָשְׂאוּ יָדָם, אֱמֹר / אֵיךְ יִצְעֲקוּ חָמָס אֲשֶׁר שֻׁקָּצוּ?

הַשַּׂר אֲשֶׁר אוֹתוֹ מְלָכִים בָּחֲרוּ / כַּאֲשֶׁר אֲבוֹתָיו מִגְּבִירִים פָּצוּ.

מִמָּר־דְּרוֹר קֹרַץ בְּיוֹם כָּל־הַיְּקוּם / מֵהָאֲדָמָה לִהְיוֹת קֹרָצוּ.

יָדָיו נְפוֹצֵי הַתְּעוּדָה אָסְפוּ / יוֹם כָּל־אֲצִילֶיהָ מְעַט קָמָצוּ.

אַלּוּף אֱדוֹם וּנְשִׂיא עֲרָב וַחְכַם בְּנֵי / יָוָן לְפָנָיו פֶּה עֲנוֹת קָפָצוּ.

סָרוּ לְמִשְׁמַעְתּוֹ וְהֵזִינוּ נְאֻם / פִיו וַעֲצַת נַפְשָׁם כְּבָר נָאָצוּ.

לוֹ לֵב סְעִפָּיו עַל כְּרוּבִים נַעֲלוּ / אַךְ רֹאשׁ יְלִידֵי הַזְּמָן מָחָצוּ.

לוֹ מוֹעֲצוֹת שֵׂכֶל יְסַכֵּל וַעֲצַת / שֵׂטִים כְּחֶפְצוֹ כָּעֲסוּ אוֹ רָצוּ.

הוּא גַן חֲסָדִים חָרְפוּ עָלָיו מְתֵי / חֹרֶף וְתַחַת צֵל כְּנָפָיו קָצוּ.

רֵעָיו כְּרֵעַי יָקְרוּ וּכְצוּף לְפִי / אִמְרֵי מְלִיצוֹתָיו מְאֹד נִמְלָצוּ.

עַל מָה יְהִי נֶעְצָב לְיָמִים חָשְׁבוּ / רָעָה עֲלֵי נַפְשׁוֹ וּבוֹ פָרָצוּ.

קֶדֶם הֲלֹא רַבּוּ עֲבָדִים עָבְרוּ / עַל פִּי אֲדוֹנֵיהֶם וְהִתְפָּרָצוּ.

לֹא הֶעֱשִׁירוּ עַד לְהוֹרִישׁ יָזְמוּ / וַיַּחְשְׁבוּ לַפְרִיד אֲשֶׁר נִקְבָּצוּ.

הֵם הֵם אֲשֶׁר בָּכוּ לְצֶלַע בּוֹגְדִים / אָכֵן לְחַפִּים יֶאֱלוּ עָלָצוּ.

אִם רַחֲבֵי נֶפֶשׁ וְשֵׂכֶל הוּבְאוּ / לָגוּר בְּמָקוֹם צָר וְגַם אֻלָּצוּ,

הֵמָּה כְּשֶׁמֶשׁ מַעֲלוֹת אִם יַחֲלִיף / לֹא חָסְרוּ אוֹרָיו וְלֹא פֹרָצוּ,

אוֹ מוֹר אֲשֶׁר נֶחְתַּם וְרֵיחוֹ יֶאֱחֹז / כָּל צַד וּמֵאֵין פֶּה שְׂפָתָיו פָּצוּ.

גַּלְגַּל מְסוֹבֵב הוּא וְנֵצַח יַעֲשׂוּ / בְּנוֹת תְּקוּפוֹתָיו אֲשֶׁר יֶחְפָּצוּ.

צֵל הֵם יְמוֹתֵינוּ עֲלֵי אֶרֶץ, אֲבָל / הֶבֶל חֳדָשֵׁינוּ אֲשֶׁר חֻצָּצוּ.

הֵא לָךְ נְאֻם רֵעַ אֲשֶׁר יִתְגּוֹדְדוּ / עָלָיו בְּנֵי יָמִים וְיִתְרוֹצָצוּ,

וַיְעַבְּתוּ נִטְעֵי תְלָאוֹתָיו וְכָל / פֹּארוֹת תְּהִלּוֹתָיו לְאַט קֻצָּצוּ.

וּמְעַט שְׁאָר נִשְׁאַר חֲמֻשִּׁים יָצְאוּ / אֵלָיו וְלַצִּית אֵשׁ קְרָב נֶחְלָצוּ.

יָנְקוּ חֲלֵב הוֹדוֹ וְחֵלֶב נָעֳמוֹ / אָכְלוּ וְשֹׁד גָּדְלוֹ בְּכָל־פֶּה מָצוּ.

שַׂרְפֵי נְדוּדִים יִשְּׁכוּ לִבּוֹ הֲכִי / בֵּינוֹת צְלָעָיו יוֹם לְיוֹם יִשְׁרָצוּ.

אַךְ אָהֳלֵי דוֹדָיו לְדוֹדָיו רָחֲבוּ / לֹא יֵצְרוּ לָעַד וְכֵן לֹא אָצוּ.

קִירוֹת בְּרִיתוֹ מִנְּטוֹת חָזְקוּ וְלֹא / נִגְלוּ יְסוֹדוֹתָם וְלֹא נִפְרָצוּ.

קַח־נָא נְגִידֵי שִׁיר וְכוֹכְבֵי מַאֲמָר / מֵהַשְּׁחָקִים אַרְצָה חָלָצוּ.

נָחוּ בְחֵיק חַסְדָּךְ, וְאוּלָם אֶל קְצֵה / תֵבֵל לְסַפֵּר מַהֲלָלָךְ אָצוּ,

כָּתְנוֹת תְּהִלּוֹת אָרְגוּ אוֹתָם יְדֵי / אָמָן וּבִזְהַב אַהֲבָה שֻׁבָּצוּ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
