תם אהל צדק

לקרובה א'

תָּם אֹהֶל צֶדֶק וּמְגוּרֵי יֹשֶר

שָׁכַן, וּמֵעוֹדוֹ מַהֲלָכָיו אֲשֶׁר רָחַשׁ בְּנִכְבַּדֵּי מְזִמָּה וָפֵשֶׁר

וְכֹל אֲשֶׁר הָיָה בָרוּחַ:

דהי"א כח,יב

קִרְבַת אֱלֹהִים חָמַד אִישׁ חָלָק

צַיִד בְּפִיהוּ לוֹ אַדְמוֹנִי חָלָק

פֶּרֶא אָדָם אַחֲרֵי צֵידוֹ דָלָק

רֹדֵף קָדִים וְרֹעֶה רוּחַ:

הו' יב,ב


עַל דְּבַר הַבְּכֹרָה רַבָּה מַשְׂטֵמָה

שָׂעִיר הִתְנֶחָם וּלְהָרְגוֹ דִמָּה

נַפְשׁוֹ שָׂמַתְהוּ לִרְדֹּף בְּחֵמָה

אֶל עָנִי וּנְכֵה רוּחַ:

ישע' סו,ב


מֵאָבִיו יָצָא אֶל אֲרַם נַהֲרָיִם

לָלוּן נָטָה אֶל שַׁעַר הַשָּׁמָיִם

כְּבוֹד אֵל חָזָה וּלְהַשְׁמָעוּת אָזְנָיִם

וַתִּצְלַח עָלָיו הָרוּחַ:

ש"א י,י

יוֹרְדִים וְעוֹלִים רָאָה בַסֻּלָּם

טַעֲמֵי מוֹצָאָם וּמוֹבָאָם לוֹ גִּלָּם

חָזוֹן נִרְאָה לְסַרְעַפָּיו וְלֹא נֶעֱלָם

כְּנוֹחַ עָלָיו הָרוּחַ:

במד' יא,כה


זֶבֶד טוֹב זְבָדָהוּ לִשְׁמֹר הֲלִיכָתוֹ

וּזְמָמָיו לְהָפֵק לַהֲשׁיִבוֹ לְאַדְמָתוֹ

הֱיוּתָם כָּל צָרָיו וְכָל אַנְשֵׁי מַצּוּתוֹ

כְּקַשׁ לִפְנֵי רוּחַ:

תה' פג,יד

דָּר עִם אֲרַמִּי וַיֶּרֶב טוּבוֹ

גָּדַל וַיִתְאַו לָשׁוּב אֶל מַצָּבוֹ

בִּגְבוּל שׂוֹטֵם בָּא וַיָּנַע לְבָבוֹ

כְּנוֹעַ עֲצֵי יַעַר מִפְּנֵי רוּחַ:

ישע' ז,ב

אוֹתוֹ בִּשֵּׂר אֵל כִּי לֹא יִשְׂכָּל

מַלְאָךְ הִתְאַבֵּק עִמּוֹ וַיִּשְׂכָּל

שָׂרָה עִם אֱלֹהִים וְעִם אֲנָשִׁים וַיֻּכָל

אָדָם שַׁלִּיט בָּרוּחַ:

קה' ח,ח

הִשְׁכִּים וַיַּחַץ מַחֲנוֹתָיו לְרָאשֵׁיהֶם

בָּנָיו וַעֲבָדָיו נָשָׁיו וְאַמְהוֹתֵיהֶם

רָעַד מֵחֲיָלֵי אָחִיו וּמַרְכְּבוֹתֵיהֶם

וְסוּסֵיהֶם – בָּשָׂר וְלֹא רוּחַ:

ישע' לא,ג

בָּטַח בְּמֵבִין צְפוּנֵי סַרְעַפָּיו

יָדַע כִּי בַּדָּיו יְנוּבוּן וַעֲנָפָיו

יְלִידֵי שׂוֹטֵם וּמוֹטוֹת כְּנָפָיו

וְכָל אֲגַפָּיו – אֱזָרֶה לְכָל רוּחַ:

יחז' יב,יד

עֱנוּת יְלָדָיו יִשְׁמַע וְיָחִישָׁה

קוֹל דְּרוֹר יַאֲזִין בְּפַח גָּלוּת יְקוּשָׁה

בְּמַסְגֵּר עֲוֹנָהּ נַפְשָׁהּ יָבְשָׁה

מֵעֲבֹדָה קָשָׁה וּמִקֹּצֶר רוּחַ:

שמ' ו,ט

אֶת מֵי עֵינֵיהֶם יַזִּילוּן כַּמָּיִם

בְּדַאֲבוֹן נֶפֶשׁ וְכִלְיוֹן עֵיָנִים

נוֹשְׂאֵי עֹל בְּצַוָּאר וּמוּעָקָה בְמָתְנָיִם

וְרָפוּ כָל יָדַיִם וְכִהֲתָה כָל רוּחַ:

יחז' כא,יב

עוּרָה וְהָפֵק צֳרִי לִכְאֵבִי

זַעַף תְּשַׁלַּח לְצָר יִשְׁאַג כְּלָבִיא

רַחוּם בַּשֵּׂר הֲרוּגֵי הַשְּׁבִי

הִנְנִי מֵבִיא – בָהֶם אֶת הָרוּחַ:

שם לז,ה

חִין עֶרְכֵיהֶם יַעֲלֶה וִישֻׁפַּר

זְדוֹנָם בְּיוֹם עַנּוֹת נַפְשָׁם יְכֻפַּר

קְרָא כִּי צָר לְכָל צַד יְזֹרֶה וִיעֻפַּר

כְּעָפָר עַל פְּנֵי רוּחַ:

תה' יח,מג

בְּזָכְרָם בְּזֹאת יָבֹא רֹאשׁ מִנְּזָרִים

בְּפַר בֶּן בָּקָר וְאַיִל מִכָּרִים

חֻקָּה מֵאֵת מַשְׁפִּיל וּמֵרִים

יוֹצֵר הָרִים וּבֹרֵא רוּחַ:

עמ' ד,יג
