יָזַמְתָּה מִיתָה

לא / בראשית מח א – ויאמר ליוסף הנה אביך חלה


ג

יָזַמְתָּה מִיתָה לַכֹּל / וְיָפֶה בְעִתּוֹ עָשִׂיתָה כֹּל / כּוֹס מָסוּךְ לַכֹּל / וְהַכֹּל כַּאֲשֶׁר לַכֹּל

נֶעֱצָבִים בְּחוֹלִי……. תה / בַּעֲבוּר כִּי עַל חַיֵּיהֶם יְכַפֵּר יוֹם מִיתָה

יַד שִׁלְטוֹן אֵין בְּיוֹם הַמָּוֶת / וּמִי לֹא יִרְאֶה מָוֶת / עַד יְבֻלַּע מָוֶת

יְקוֹשׁ חוֹלִי רִאשׁוֹן וְחוֹלִי שֵׁנִי הֶחֱלֵיתָה / לִנְבִיאֵי צֶדֶק וּמֵהֶם הַחִלּוֹתָה.


ד

יְמֵי תְמִימִים אַתָּה יוֹדֵעַ / וְיוֹם מִיתָה לָהֶם לֹא מוֹדִיעַ

לְמַעַן אִם יֶחֱלוּ וְאִם יַזְקִינוּ / מְלַאכְתָּם יַתְקִינוּ / וְיוֹרְשֵׁמוֹ יַתְקִינוּ

כְּאָב וִיחִידוֹ אֲשֶׁר בָּטוּ / כִּי לִימוֹת זִקְנָה נָטוּ

דָּאֲגוּ פֶּן יוֹם מִיתָה יִהְיֶה / וְצִוּוּ עַל בָּתֵיהֶם מַה יִהְיֶה

וְתָם בְּעֵת חָלָה / קָרָא לִבְנוֹ לְיוֹסֵף / עַד לֹא יֵאָסֵף

עָלָיו לְהוֹסִיף / תּוֹסֶפֶת בְּרָכוֹת

אֶפְרַיִם כִּרְאוּבֵן מְנַשֶּׁה כְשִׁמְעוֹן / וְאָז הִסְכַּמְתָּה קָדוֹשׁ.


ה

אָז בְּזִקְנָה הִתְחִיל אָב בָּא בַיָּמִים / לְהוֹדִיעַ כִּי זִקְנָה הוּא סוֹף יָמִים

בֶּן אָדָם בְּהַגִּיעוֹ לְזִקְנַת יְמוֹתָיו / יְפַחֵד תָּמִיד מִפְּנֵי יוֹם מוֹתָיו

גַּם יְחִידוֹ הֵחֵל בְּיִסּוּרִים / כִּי הֵם לְאָדָם לְמִיתָה מוֹסְרִים

דָּאוֹג עֲבוּר יִדְאַג כָּל אִישׁ אִם יִוָּסֵר / פֶּן תִּקְצַר נַפְשׁוֹ וְלַמָּוְתָה יִמָּסֵר

הֵן מֵאִישׁ תָּם הַחוֹלִי הִתְחִיל / כִּי אִם יֶחֱלֶה אָדָם יֵיחַת וְיָחִיל

וְיָשׁוּב בְּחָלְיוֹ כִּי יִירָא מִמָּוֶת / כִּי הַחוֹלִי קוֹרֵא לַמָּוֶת

זָעַק תָּם לְנֶחֱסַךְ מִלְּטַמֵּא יָרֵךְ / הִשָּׁבְעָה לִי בְּחוֹתַם יָרֵךְ

חָלִילָה לָּךְ מִלִּקְבֹּר שֶׂה בְּצַד חֲמוֹרִים / כִּי אִם בַּאֲשֶׁר קָנִינוּ מֵאֱמוֹרִים

טִפּוּחָיו הִגִּישׁ לוֹ אַבְרֵךְ / וַיְבָרְכֵם וַיֹּאמֶר בָּךְ יְבָרֵךְ

יַרְבִּיב אֶפְרַיִם / וְיַאֲלִיף מְנַשֶּׁה / לְעוֹלָם לִי אֶפְרַיִם וְּלָעַד לִי מְנַשֶּׁה.


ו

אָב אֲשֶׁר נַפְשׁוֹ לָךְ אֲהֵבָה / חָלָה וג………….

בַּעֲבוּר אַהֲבָתָךְ אֲשֶׁר כְּלֶהָבָה / נִמְשַׁךְ בַּעֲבוֹתוֹת אַהֲבָה.


גּוֹוֵעַ וְנָטָה לְמִיתָה / מְהַלֵּךְ וְנָם הַפַּעַם אָמוּתָה

דַּיָּין הַמּוֹתִיר בְּנֵי תְמוּתָה / יָקָר בְּעֵינָיו לַחֲסִידָיו הַמָּוְתָה.


הוֹרָךְ הֵן נֶאֱסָף אֶל עַמּוֹ / וְטוֹב בְּרָכָה נֶאֱסֶפֶת עִמּוֹ

וְקָח אֶת שְׁנֵי בָנָיו עִמּוֹ / כִּי יְבָרְכֵם בְּנָאֳמוֹ וְנָעֳמוֹ.


זַעַם חוֹלִי עַל אָב חֻזַּק / וְרִיפְיוֹן יָדַיִם חִזַּק

חוֹלְלוּ בְּשָׁמְעוֹ הִנּוֹ נִתְחַזֵּק / בְּבִרְכוֹתָיו אוֹתוֹ לְחַזֵּק.


טוֹעֲנָךְ וְנוֹשְׂאָךְ עַל בִּרְכָּיו / טָעַן חוֹלִי וְכָרַע עַל בִּרְכָּיו

יוֹסֵף בְּקָשְׁבוֹ אִמֵּץ בִּרְכָּיו / לְהַטְעִין בְּרָכָה לְנוֹלְדוּ עַל בִּרְכָּיו.


כָּמַהּ מְחוֹלְלָךְ לָךְ לְחַלּוֹת / נַפְשׁוֹ בְּהִתְעַטֵּף וּבְהֵחָלוֹת

לָכֵן לֹא נִתְמַהְמַהּ לַעֲלוֹת / וְאֶת בָּנָיו עִמּוֹ לְהַעֲלוֹת.


מַאֲהִיבָךְ בֵּין הַבָּנִים / חָלָה וְלָךְ קָרָא מִכָּל בָּנִים

נִחֵל בָּנָיו בְּרָכָה כִּגְדוֹלֵי בָנִים / כִּי בְּנֵי בָנִים כְּבָנִים.


סָמַךְ הוֹרָךְ לְסוֹף יְמוֹתָיו / וְנָטָה לָלוּן עִם אֲבוֹתָיו

עָלָה הוּא וְיוֹצְאֵי יְרֵכוֹתָיו / לְכוֹנֵן עֲלֵיהֶם כְּלַל בִּרְכוֹתָיו.


פָּקַד הָאֵל פּוֹקֵד כֹּל / לְאָבִיךָ פְּקֻדַּת כֹּל

צוּר הַכֹּל / כֹּל כַּאֲשֶׁר לַכֹּל / יָפֶה בְעִתּוֹ עָשָׂה כֹּל.


קוֹרְאָךְ קָרָא מִקְרָאוֹ לָמוּת / וּמִי גֶבֶר יִחְיֶה וְלֹא יָמוּת

רָץ לְחַלּוֹתוֹ בְּסֵבֶר דְּמוּת / כִּמְחַלֶּה זה שְׁכִינַת דְּמוּת.


שׁוֹמֵר מִצְוָה שְׂכָרָהּ חַיֵּי נָפֶשׁ / לְכִבּוּד אָב אֲהֵבָךְ אַהֲבַת נָפֶשׁ

תָּכַף לְקַבֵּל מֶנּוּ נָפֶשׁ / וְלָתֵת תְהִלָּה לַאֲדוֹן כָּל נָפֶשׁ.


ז

אֱמֶת אָמַרְתָּה לִמְנַשֶּׁה בְּמָעוֹז רֹאשׁ נִכְתַּר אֶפְרַיִם

גִּדּוּל בְּכוֹר הָיָה מְנַשֶּׁה דִבַּרְתָּה בְכוֹרִי הוּא אֶפְרַיִם

הַכֹּל מְבָרְכִים בִּמְנַשֶּׁה וּמִתְבָּרְכִים בְּשֵׁם אֶפְרַיִם

זֻמַּן כְּשִׁמְעוֹן מְנַשֶּׁה חֻשַּׁק כִּרְאוּבֵן אֶפְרַיִם

טֶרֶם נוֹלַד מְנַשֶּׁה יֻדְּעָה תּוֹלְדוֹת אֶפְרַיִם

כְּבוֹד מִילָה עָלָה לִמְנַשֶּׁה לִיווּי שַׁבָּת נָחַל אֶפְרַיִם

מְקוֹם מִילָה בִּגְבוּל מְנַשֶּׁה נִתַּן כֵּן בְּהַר אֶפְרַיִם

שֻׂכְּלָה יָד עַל רֹאשׁ מְנַשֶּׁה עֻטְּרָה יָמִין בְּרֹאשׁ אֶפְרַיִם

פְּלָאִים הִפְלֵאתָ לִבְנֵי מְנַשֶּׁה צֶדֶק הִצְמַחְתָּ לְמַטֵּה אֶפְרַיִם





את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
