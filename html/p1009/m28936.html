הִנֵּה יְמֵי מוֹעֵד כְּבָר הִגִּיעוּ

הִנֵּה יְמֵי מוֹעֵד כְּבָר הִגִּיעוּ / לָעָם אֲשֶׁר בֵּין אוֹיְבִים נִקְלָעוּ

מַה לִּי לְיוֹם שִׂמְחָה אֲיַחֵל עוֹד וְאָגִיל / אֶחְגֹּר וּבַחוּצוֹת כְּעִוְרִים נָעוּ

הָעָם אֲשֶׁר נִגְאַל בְּנֵס עַל יַד בְּחִיר / פַּרְעֹה וּמִצְרַיִם בְּיָם הָטְבָּעוּ

הָעָם אֲשֶׁר הָאֵל נְשָׂאָם עַל כְּנַף / נֶשֶׁר וּבַמִּדְבָּר יַחֲנוּ גַּם יִסָּעוּ

מֵימֵי בְאֵר יִשְׁתּוּ וְיֹאכְלוּ הַשְּׂלָו / עֶרֶב וּבַבֹּקֶר בְּמָן שָׂבָעוּ

עִם עַנְנֵי כָבוֹד בְּעַמּוּד לַעֲשׂוֹת / שָׁפָל וְגָבוֹהַּ כְּמוֹ נוֹדָעוּ

הָעָם אֲשֶׁר בָּחַר אֱלֹהִים לוֹ לְעַם / קָדוֹשׁ וְלִדְבָרוֹ אֱמֶת נִכְנָעוּ

שָׁבוּ לְמַטָּרָה לְחִצֵּי הַזְּמָן / תּוֹךְ בּוֹר חֲשֵׁכִים בֶּאֱמֶת נִשְׁקָעוּ

עֵת יִזְכְּרוּ חָג יֶהֱמוּ יִשְׁתּוֹחֲחוּ / עָבַר וְעֵינָם בַּבְּכִי דָמָעוּ

וּבְבוֹא יְמֵי פֶסַח כְּיַמִּים יֶהֱמוּן / כָּל מַעְיְנוֹת דִּמְעָה אֲזַי בָקָעוּ

פִּתָּם בְּדִמְעוֹתָם מְסוּכָה בַּעֲבוּר / לֹא מָצְאוּ נַחַת וְלֹא רָגָעוּ

לֹא מָצְאוּ יַיִן וּמַצָּה נִשְׁמְרָה / נִמְצָא לְהַזְכִּירָם מְרוֹרִים לָעוּ

שָׁכְרוּ בְיֵין גָּלוּת בְּרֹאשׁ צוּרִים וְתוֹךְ / בֵּית הָאֲסוּרִים אָהֳלָם תָּקָעוּ

הָהּ לִי וְאוֹיָה לִי וְאֵין לִי מָעֳמָד / עַל מַחֲמַדֵּיהֶם אֲשֶׁר גָּוָעוּ

קֵדָר אֲשֶׁר הֶחֱרִיב נְוָתָם וֶאֱדוֹם / חוֹנִים בְּאָהֳלֵיהֶם וְלֹא מָנָעוּ

חִגֵּר וְסִיעָתוֹ שְׁאוֹלָה יֵרְדוּ / בַּל שֹׁרְשׁוּ לָעַד וּבַל נִטָּעוּ

אָמָה וְנִינֶיהָ נְשִׂיאֶיהָ יִהֲיוּ / כִּכְפִיר אֲשֶׁר שִׁנָּיו כְבָר נִתָּעוּ

הֵמָּה וְחַצְרֵיהֶם וְטִירוֹת שָׁכְנוּ / כֻּלָּם כְּמִזְבָּחוֹת אֲשֶׁר גֻּדָּעוּ

הַטַּף וְהַנָּשִׁים עֲזָבוּם צוֹעֲקִים / עַל אָב אֲשֶׁר גָּלָה כְסוּת קָרָעוּ

אֶל עוֹבְרֵי יַמִּים יְקַוּוּ יוֹם לְיוֹם / עֵת יֵלְכוּ מֵהֶם וְעֵת יַגִּיעוּ

גַּם נָשְׁקוּ יָדָם וְרַגְלָם בַּעֲבוּר / יִשְׂאוּ שְׁלוֹם בָּנִים לְאָבוֹת נָעוּ

יִתְמַרְמְרוּ יִשְׁתּוֹחֲחוּ אַף יִצְעֲקוּ / לָאֵל אֲשֶׁר גָּזַר וּבִבְכִי גָּעוּ

יִזְכֹּר בְּרִית אָבוֹת לְבָנִים סוֹרְרִים / פָּקוּ פְלִילִיָּה וְלִמְאֹד תָּעוּ

שָׁגוּ בְהַבְלֵיהֶם וְתוֹרוֹת עָבְרוּ / חֹק חָלְפוּ בַיָּם וְלֹא נִרְתָּעוּ

לָכֵן רְאֵה כַמָּה תְלָאוֹת קָצְרוּ / מֵאָה שְׁעָרִים עַל אֲשֶׁר זָרָעוּ

וּשְׁקֹל תְּלָאוֹתָם לְנֶגֶד מַעֲלָם / מַעַל לְרוּם יַעַל וְהֵם יַכְרִיעוּ

אַל יֹאמְרוּ צָרִים לְקַחְתָּם לַהֲרֹג / אוֹתָם בְּרֹאשׁ צוּרִים וְלֹא נוֹשָׁעוּ

רֵעַי שְׁעוּ מֶנִּי אֲמָרֵר בַּבְּכִי / עַל מַעֲשֵׂי יָדַי אֲשֶׁר יִבְצָעוּ

עֵת אֶזְכְּרָה חֶטְאִי כְתַנִּים אֶהֱמֶה / לֵב אֶקְרְעָה עַל כֵּן דְּבָרַי לָעוּ

אֵיכָה אֲבַקֵּשׁ עַל עֲנִיָּה סוֹעֲרָה / לַעְרֹךְ תְּפִלּותַי וְלֹא נִשְׁמָעוּ

לָכֵן אֱלֹהִים חַי בְּכִסֵּא גָדְלְךָ / וּבְמַלְאֲכֵי מַעְלָה אֲשֶׁר יָרִיעוּ

שִׁמְעָה תְפִלָּתִי וְרִנַּת עַמְּךָ / מִשְׁתַּחֲוִים נֶגְדָּךְ וְלָךְ יִכְרָעוּ

וּרְצֵה תְפִלָּתָם וְשַׁוְעָתָם שְׁעֵה / וּלְעָבְדְךָ לָעַד הֲלֹא נִרְצָעוּ

וּבְעֵת גְּאֻלָּתָם בְּךָ יִשְׁתַּבְּחוּ / תָּמִיד וְזֵכֶר טוּבְךָ יַבִּיעוּ

עַל הַרְרֵי צִיּוֹן בְּרִנָּה יֹאמְרוּ / הִנֵּה יְמֵי מוֹעֵד כְּבָר הִגִּיעוּ.
