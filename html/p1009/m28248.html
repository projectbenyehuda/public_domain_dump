אֱזֹר גֶּבֶר חֲלָצֶיךָ

אֱזֹר גֶּבֶר חֲלָצֶיךָ מְהֵרָה / וְקוּם עַיֵּן בְּדִבְרֵי הַגְּמַרָא

הֱיֵה חָכָם בְּמִלְחַמְתּוֹ לְמַעַן / תְּהִי בָּקִי וְתִהְיֶה לָךְ סְבַרָא

וְדַע אִינִי וּמַתְקִיף וַהֲתִיבוּן / לְקוּשְׁיוֹת הֵם לְתַנָּא וַאֲמוּרָא

מְקוֹם תֵּימַהּ פְּשִׁיטָא דַּע וְהָבֵן / לְאַבַּיֵּי וְרַבָא הַבְּכוֹרָה

וְאַף עַל פִּי וְאַף עַל גַּב תְּתַקֵּן / בְּעִיּוּנָךְ לְךָ דֶּרֶךְ יְשָׁרָה

כְּדֵי נִיסְבֵּא בְּפֵרוּשָׁהּ וְעִנְיַן / תְּנָא נַמֵּי לְהַךְ דֶּרֶךְ עֲבָרָה

מְקוֹם תִּיק״וּ וְאַלִיבָּא דְּרִבִּי / תְּרֵין אִנּוּן וְאִנּוּן חַד בְּחִזְרָא

וּבַמִּשְׁנָה בְּחַסַּוְרֵי מְחַסֵּר / תְּהִי מַשְׂכִּיל שְׁהִיא דֶּרֶךְ קְצָרָה

וְהַמִּדּוֹת מְקוֹם הַקַּל וְחֹמֶר / וּבִנְיַן אָב וְאִם תַּשְׁוֶה גְּזֵרָה

וְאַדְרַבָּה מְתַקֶּנֶת לְהֵפֶךְ / בְּנוּטְרִיקוֹן לְקַלָּה וַחֲמוּרָה

וְטִיב מַשָּׂא וּמַתָּן בַּהֲוָיוֹת / לְרַבִּינָא וְרָב אַשֵׁי מְסוּרָה

וּמֵהֶם לָמְדוּ דֶּרֶךְ גְּאוֹנִים / בְּפִסְקֵיהֶם לְכָל דֵּעָה חֲסֵרָה

לְבִלְתִּי יִיגְעוּ לִמְצֹא דְּרָכִים / בְּתַלְמוּד בָּבְלִי פִּים הַפְּצִירָה

וְהַמַּסְלוּל אֲשֶׁר תִּמְצָא בְּפִיהֶם / וְכָל תִּיקוּ דְּאִסּוּרָא לְחוּמְרָא

וְטִיב גִּטִּין וְקִדּוּשִׁין מְתֻקָּן / בְּחָכְמָתָם וְדִין עֵדֵי מְסִירָה

וּבִכְתֻבּוֹת וּמֵאוּן וַחֲלִיצָה / וְדִין פֶּסַח וּמַצָּה הָעֲשִׁירָה

וְאֵין בִּינָה בְּבָנִים עוֹד וְיֶשׁ אֵם / לְמָסֹרֶת וְגַם יֶשׁ אֵם לְמִקְרָא

וְעוֹד דִּינֵי נְגָעִים וָאָהֳלוֹת / טְבוּל יוֹמוֹ וְטוּמְאָה הַחֲמוּרָה

וְדִין הַשּׁוֹר וְהַמַּבְעֶה וְהַבּוֹר / וְאַרְבַּעְתָּם כְּדִין מַבְעִיר בְּעֵרָה

וְהַמִּשְׁפָּט בְּבַיִת וַעֲלִיָּה / וְאִם קַל הוּא כְּדִין תַּנּוּר וְכִירָה

וְהַשּׂוֹכֵר וְהַשּׁוֹמֵר לְמִינָיו / וְהַמְּקַבֵּל וְדִין חוֹכֵר חֲכִירָה

וְהַמִּיתוֹת בְּאַרְבַּעְתָּן לְבֵאוּר / וְכָרֵיתוֹת וְדִין עוֹבֵר עֲבֵרָה

וְחִלּוּקִים מְדֻיָּקִים וְדַקִּים / בְּעֵרוּבִין לְשַׁבָּת הַחֲמוּרָה

וְדִין מַמְרֶה וְדִין סוֹרֵר וּמוֹרֶה / וְדִין עֶבֶד וְדִין אָמָה מְכוּרָה

וְדִין לֶקֶט וְדִין שִׁכְחָה וּפֵאָה / וּמַעַשְׂרוֹת וְדִין בֵּית הַבְּחִירָה

וְעוֹד נוֹסָף עֲלֵי אֵלֶּה בְּרַבִּים / וּמָה אַעַשׂ וְאֵין לִי עוֹד זְכִירָה

וְלֹא אוּכַל שְׂאֵת מַהֲלָל לְנַפְשִׁי / בְּיָדְעִי בָּם בְּתַכְלִית הַשְּׁמִירָה

אֲבָל חֶשְׁקִי יְדִיעָתָם וְלִבִּי / לְפֵרוּדָם כְּלֵב אִשָּׁה מְצֵרָה

וְלֹא עָלַי גְּמִירַת הַמְּלָאכָה / וְלֹא אֹמַר לְנַפְשִׁי אַתְּ פְּטוּרָה

וְאַשְׁרֵי אִישׁ אֲשֶׁר זִכָּה לְנַפְשׁוֹ / בְּקֵץ יָמָיו וּמִשְׁנָתוֹ סְדוּרָה.

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
