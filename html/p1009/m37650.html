חָכְמַת הַיְּהוּדִי: מקאמה

– – – וַיֹּאמֶר לִי: דַע אֲדוֹנִי

בִּתְחִלַּת עִנְיָנִי / כִּי אֲנִי הָיִיתִי בְּאֶרֶץ אַשּׁוּר / אֲשֶׁר מֵרֹאשׁ אֲמָנָה תָשׁוּר / וַיִּהְיוּ לִי חֲבֵרִים שְׁנַיִם / נוֹצְרִי וְיִשְׁמְעֵאלִי בַּעֲלֵי עֵינַיִם / וַיֹּאמְרוּ לִי: ׳הַיְּהוּדִים חֲלָשִׁים כֻּלָּם / מִקְּטַנָּם וְעַד גְּדוֹלָם / וְאֵינָם יוֹדְעִים לַעֲשׂוֹת דְּבַר גְּבוּרָה / וְלֹא דְּבַר גְּנִיבוּת גְּמוּרָה / אֲנַחְנוּ יֶשׁ לָנוּ שֵׁם / אִם יֶעֶרְבוּ לָנוּ כַמָּה דְבָרִים בַּשּׁוּק נִירָשֵׁם / כְּרֶגַע מִבַּעֲלֵיהֶן נְגָרְשֵׁם׳ / וָאֹמַר אֲלֵיהֶם: ׳נַסּוּ / אֲבָל הַטּוֹעֲנִים מִקּוֹדֵם עֲשׂוּ / וַאֲנִי יִהְיֶה מַעֲשִׂי אַחֲרֵיכֶם / לְבַעֲבוּר תַּפְרִישׁוּ בֵינִי וּבֵינֵיכֶם / עַל מְנָת שֶׁכָּל אֶחָד מִמֶּנּוּ זוֹכֶה בְחֶלְקוֹ / כְּפִי עִנְיָנוֹ וּפִרְקוֹ׳ / וַיֹּאמְרוּ לִי: ׳הֶחֱיִיתָנוּ / לֹא עֲשַׁקְתָּנוּ וְלֹא רַצּוֹתָנוּ׳ / וַיַּפִּילוּ גוֹרָלוֹת בֵּינֵיהֶם / וּשְׁבוּעַת הַשֵּׁם בֵּין שְׁנֵיהֶם / וַיִּפֹּל הַגּוֹרָל בַּתְּחִלָּה עַל הַנּוֹצְרִי / וְאַחֲרָיו הַהַגְרִי.

וַנֵּלֶךְ לַשּׁוּק כֻּלָּנוּ / עַל פִּי הַדְּבָרִים / אֲשֶׁר בֵּינֵינוּ / וַנִּרְאֶה בַשּׁוּק שְׁתֵּי חֵמוֹת / מְלֵאוֹת מֵחֶמְאַת הַבְּהֵמוֹת וַיֹּאמֶר הַיִּשְׁמְעֵאלִי אֶל הַנּוֹצְרִי: ׳אִם אַתָּה רוֹצֶה לְרַפֵּא שִׁבְרִי / עָלֶיךָ לִגְנֹב אֶת הַחֵמֶת הַגְּדוֹלָה / מִלִּפְנֵי בְעָלֶיהָ בְתַחְבּוּלָה׳ / וַיֹּאמֶר הַנּוֹצְרִי: ׳עָלַי מִיָּדוֹ לְהַפְקִיעֶנָּה / וְאַתָּה תַּגְבִּיהֶנָּה׳ / וַיֹּאמֶר הַיִּשְׁמְעֵאלִי: ׳אֲנִי אֶעֱשֶׂה / לֹא תִשְׁאַל וְלֹא תְנַסֶּה׳ / וַיֵּלֶךְ הַנּוֹצְרִי וּבְיָדוֹ צֶבֶת / וְהוּא צוֹעֵק מִכְּאֵב שִׁנּוֹ לֹא יוּכַל לָשֶׁבֶת / וַיֹּאמֶר אֶל מוֹכֵר הַחֶמְאָה: ׳בְּחַיֶּיךָ הָסֵר מִמֶּנִּי חֳלָאָה / כִּי אֲנִי שָׁמַעְתִּי עָלֶיךָ / בְּנֹעַם מִלֶּיךָ / תְּרַפֵּא בְנֵי אָדָם / עַל יְדֵי הַקָּזַת דָּם / וְתִלְחַשׁ עַל הַמַּכָּה וְתִתְרַפֵּא / וּבְיָדָךְ סְגֻלַּת תַּרְשִׁישׁ וְיָשְׁפֵה / עֲשֵׂה לִי כְחַסְדֶּךָ / וְקַח הַצֶּבֶת בְּיָדֶךָ / וְהָסֵר אֶת שִׁנָּי / כִּי לֹא טָעַמְתִּי שֵׁנָה בְעֵינָי׳ / וַיַּאֲמֵן בַּעַל הַחֶמְאָה / כִּי עַל יָדוֹ תִּהְיֶה רְפוּאָה / בְּיָדוֹ אֶת הַצֶּבֶת / וַיַּגְבִּיהַּ אֶת הַנּוֹצְרִי לָשֶׁבֶת / וַיִּתְקַבְּצוּ לְמוּלוֹ מִן הַשּׁוּק אֲנָשִׁים / נְעָרִים וִישִׁישִׁים / וְהַיִּשְׁמְעֵאלִי בָּא לָקַחַת אֶת הַחֵמֶת הַקְּטַנָּה / וְהוּא נִבְהָל וְשִׂכְלוֹ בִמְגִנָּה / וּבַעַל הַחֶמְאָה מְמַשֵּׁשׁ / בְּשִׁנֵּי הַנּוֹצְרִי מְגַשֵּׁשׁ / וַיֹּאמֶר הַנּוֹצְרִי בְּרִשְׁעוֹ / כְּאִלּוּ הוּא מַרְאֶה נִגְעוֹ: / ׳קַח הַגְּדוֹלָה שֶׁבָּהֶם / כִּי שִׁנַּי אָמַרְתִּי אַפְאֵיהֶם / וַיָבֶן הַיִּשְׁמְעֵאלִי כִּי הוּא רוֹמֵז אֵלָיו / בְּרֹעַ מַעֲלָלָיו / וַיִּקַּח אֶת הַחֵמֶת הַגְּדוֹלָה / וַיֵּלֶךְ לְדַרְכּוֹ הַסְּלוּלָה / וְהַנּוֹצְרִי יָדַע כִּי הַיִּשְׁמְעֵאלִי בָרַח / וַיֹּאמֶר לְבַעַל הַחֶמְאָה: ׳הַכְּאֵב מִמֶּנִּי פָרַח / בָּרוּךְ יִהְיֶה אֲשֶׁר אֵלֶיךָ הִנְחַנִי / הַיּוֹם רָאִיתִי בָךְ מַה שֶּׁלֹּא רָאֲתָה עֵינִי / וּמֵעֵת אֲשֶׁר מִשַּׁשְׁתָּ בְשִׁנָּי / סָר נִגְעִי מִלְּפָנַי׳ / וַיִּקַּח אֶת הַצֶּבֶת וַיֵּלֶךְ לְדַרְכּוֹ / וּמוֹכֵר הַחֶמְאָה צוֹעֵק כָּל צָרְכּוֹ / כִּי לֹא יָדַע מִי גְנָבוֹ / וַיִּבְעַר לִבּוֹ בְקִרְבּוֹ / זֹאת עֲצַת הַנּוֹצְרִי הַבְּלִיַּעַל / הַנִבְזֶה וְהַנִּגְעָל.

וַיְהִי לְמָחָר וַיֹּאמֶר הַיִּשְׁמְעֵאלִי: ׳זֶה הַיּוֹם לְמוּלִי מַצָּבוֹ / נָגִילָה וְנִשְׂמְחָה בוֹ׳ / וַנֵּלֶךְ בְּשׁוּק הַצּוֹרְפִים / אֲשֶׁר בַּעֲלֵי הַכֶּסֶף עוֹדְפִים / וַנִּרְאֶה שֻׁלְחָנִי אֶחָד זָקֵן וּלְפָנָיו תֵּיבָה / מְלֵאָה מִכָּל אֶבֶן יְקָרָה וַחֲשׁוּבָה / וַיֹּאמֶר הַיִּשְׁמְעֵאלִי אֶל הַנּוֹצְרִי: ׳עָלַי לָקַחַת הַתֵּיבָה / וְתַחְבּוּלָתְךָ עַל לוּחַ לִבִּי כְתוּבָה׳ / וַיֹּאמֶר הַנּוֹצְרִי: ׳וְאֵיךְ תּוּכַל לְהַעְלִימֶנָּה / וְשֶׁמָּא תָּבוֹא לִידֵי סַכָּנָה? / וַיֹּאמֶר יִּשְׁמְעֵאלִי: ׳עַתָּה תִרְאֶה / הֲנִסָּה דָבָר אֵלֶיךָ תִלְאֶה / וַיֵּלֶךְ אֵצֶל הַשֻּׁלְחָנִי / הָאִישׁ מְלָשְׁנִי / וַיֹּאמֶר: ׳שָׁלוֹם עָלֶיךָ, בַּעַל הָאֱמוּנָה / כָּל יְמֵי עֶדְנָה וְזִקְנָה / וַיַּעֲנֵהוּ הַשֻּׁלְחָנִי בִּכְפִילַת הַשָּׁלוֹם / וְלֹא יָדַע כִּי הוּא כְּחוֹלֵם חֲלוֹם / וַיֹּאמֶר לוֹ הַיִּשְׁמְעֵאלִי: ׳אֲנִי תָּמֵהַּ עָלֶיךָ / אֵיךְ תַּאֲמִין עַל זֹאת הַתֵּיבָה לְהַנִּיחָהּ בֵּין רַגְלֶיךָ / וְאַתָּה צָרִיךְ לְהַצְנִיעָהּ עַד אֲשֶׁר לַכֶּסֶף תְּבַקְּשֶׁנָּהּ / וּמִן הַמַּטְמוֹנִים תַּחְפְּשֶׁנָּהּ / כִּי יֵשׁ בְּזֹאת הָעִיר לִסְטִים מְזֻיָּנִים / גּוֹנְבִים מִקַּרְנֵי רֵאמִים וְעַד בֵּיצֵי כִנִּים / וְאַתָּה הִזָּהֵר מֵהֶם / פֶּן יִרְאוּ מָמוֹנָךְ בְּעֵינֵיהֶם / כִּי אֶפְשָׁר לָהֶם לַעֲשׂוֹת כָּךְ / וּלְהַנִּיחָךְ מָךְ מֵעֶרְכָּךְ׳ / וַיּוֹצִיא הַתֵּיבָה אֶל פֶּתַח הֶחָנוּת מִבַּחוּץ / וַיִּהְיוּ דְּבָרָיו אֶל הַשֻּׁלְחָנִי נָחוּץ / וַיֹּאמֶר לוֹ: כָּךְ עוֹשִׂים / וּמְמוֹן בְּנֵי אָדָם תַּחַת כַּנְפֵיהֶם מְכַסִּים / וַיִּסְגֹּר הַדֶּלֶת בַּעֲדוֹ / כִּי קָרוֹב יוֹם אֵידוֹ / וְאָסַר אֶת הָעֲנִיבָה / וְהַשֻּׁלְחָנִי יוֹשֵׁב עַל הָאַרְכֻּבָּה / מְקַוֶּה לִפְתֹּחַ / וְהַיִּשְׁמְעֵאלִי הָלַךְ בַּשּׁוּק שְׁחוֹחַ / וַיֹּאמֶר אֶל הַנּוֹצְרִי: כִּמְעַט נִשְׁמָתִי לֹא נוֹתְרָה בִי / כִּי הִנַּחְתִּי הַשֻּׁלְחָנִי עַל קֶרֶן הַצְּבִי / וַיִּצְעַק הַשֻּׁלְחָנִי מִתּוֹךְ הֶחָנוּת / מַתְחִיל בִּשְׁבָח וּמְסַיֵּם בִּגְנוּת / וַיִּתְמְהוּ בַעֲלֵי הַשּׁוּק עָלָיו / אֵיךְ הַלִּסְטִים בַּחֲצִי הַיּוֹם גָּבְרוּ עָלָיו / וַיֹּאמְרוּ אִישׁ אֶל אָחִיו: ׳וְשֶׁמָּא הַשֻּׁלְחָנִי נִשְׁתַּטָּה / וּמִדֶּרֶךְ הַפִּקְחִים נָטָה / כִּי בְעֵינָיו נִגְנַב מָמוֹנוֹ כְאִלּוּ הוּא בַעֲלָטָה׳ / וַיַּעַזְבוּ אוֹתוֹ צוֹעֵק / וְכָל עוֹבֵר עָלָיו שׂוֹחֵק / זֹאת תַּחְבּוּלַת הַיִּשְׁמְעֵאלִי הַמַּמְזֵר / אֲשֶׁר עַל בַּעַל דָּתוֹ מִתְאַכְזֵר.

וַיְהִי בַיּוֹם הַשְּׁלִישִׁי וַיָּבוֹאוּ שְׁנֵיהֶם אֵלָי / לְהַרְאוֹת לָהֶם מֵיטַב מִפְעָלָי / וּמַה תִּהְיֶה גְבוּרָתִי / וְעַל אֵיזֶה אֹפֶן תַּחְבּוּלָתִי / וָאֹמַר לָהֶם: ׳אֵינִי כְמוֹתְְכֶם עוֹשֶׂה / כִּי כָתוּב בְּתוֹרָתֵנוּ אִם נָבַלְתָּ בְהִתְנַשֵּׂא / וְאַתֶּם לֹא תִּמָּצְאוּ / עַל רָאשֵׁי הָעֵצִים אֶתְכֶם יִשְּׂאוּ / אוֹ אִם יֻכַּר דָּבָר מִן הַנִּגְנָב אֶצְלְכֶם / רְאוּ כִּי רָעָה נֶגֶד פְּנֵיכֶם / וְאַתֶּם בּוֹאוּ וְנֵלֵךְ בְּשׁוּקֵי הָעִיר / וּבַחֲרוּ לִי אֶחָד מִן הַסּוֹחֲרִים בְּמָמוֹן מַרְחִיב וּמַשְׂעִיר / וְהַרְאוּנִי חֲצֵרוֹ / וְאֵיפֹה כְבוֹדוֹ מֻנָּח וִיקָרוֹ / וּלְכוּ לְדַרְכְּכֶם / וַאֲנִי אַרְאֶה אֶתְכֶם / אֵיךְ אַשְׁבִּית מִטֳּהָרוֹ / וְהוּא רוֹאֶה בְעֵינָיו סִכְלוּת דְּבָרוֹ / וַנֵּלֶךְ בִּמְבוֹאוֹת הַמְּדִינָה / אֲשֶׁר הִיא לְכָל הַמְּדִינוֹת עֲדִינָה / וַיַּרְאוּנִי בַיִת גָּדוֹל וְדִירָה נָאָה / לְסוֹחֵר אֶחָד וּסְחוֹרָתוֹ פְלִיאָה / וַאֲנִי חִפַּשְׂתִּי עַל שְׁמוֹ וְשֵׁם אָבִיו וְשֵׁם זְקֵנוֹ / בְּכִנּוּיוֹ כְתִקּוּנוֹ / וָאֵלֵךְ בַּשּׁוּק וָאֶקְנֶה מַפְתְּחוֹת / עַל כָּל מִין מְפֻתָּחוֹת / וְקָנִיתִי נְיָר לְתַקֵּן עִנְיָנָי / וְקֶסֶת הַסּוֹפֵר בְּמָתְנָי.

וַיְהִי בָעֶרֶב / יָצָאתִי אֲנִי וַחֲבֵרַי כְּמוֹ חַיָּה לְמוֹ אָרֶב / וַנַּגִּיעַ בַּחֲצוֹת הַלַּיְלָה / אֶל פֶּתַח בֵּית הַסּוֹחֵר בָּאֲפֵלָה / וָאֶפְתַּח בְּאַחַד הַמַּפְתֵּחוֹת אֶת הַדֶּלֶת / אָז יָדַעְתִּי כִּי לִי תִּקְוָה וְתוֹחֶלֶת / וַנִּכָּנֵס שְׁלָשְׁתֵּנוּ / וְאִם יֵדַע בָּנוּ הַסּוֹחֵר נִשְׁכְּבָה בְּבָשְׁתֵּנוּ / וַנַּדְלִיק אֶת נֵרֵנוּ / בְּצִנְעָה לְהַעֲלִים חֶרְפָּתֵנוּ / וַנִּפְתַח אֶת הָאוֹצָרוֹת / לְמַעַן נֵדַע אֵיזֶה בָהֶם גְּנוּזִים אֲבָנִים יְקָרוֹת / וַנַּגִּיעַ עַד אוֹצָר הַסְּחוֹרָה / מָצִינוּ בוֹ אַרְגָּז בְּתוֹכוֹ כָל אֶבֶן יְקָרָה / עִם זָהָב שָׁחוּט / נִטְוָה כַחוּט / עִם כְּלֵי חֶמְדָּה וְאַבְנֵי פָז / קְבוּעִים בְּזָהָב מוּפָז / עִם אֹדֶם פִּטְדָה וּבָרֶקֶת / לְבָנָה אֲדֻמָּה וִירַקְרֶקֶת / עִם מְעַט מִבִּגְדֵי שֵׁשׁ וָמֶשִׁי וְאַרְגָּמָן / בְּיַרְכְּתֵי הָאַרְגָּז נִטְמָן / וָאוֹצִיא אֶת הַנְּיָר וְהַקֶּסֶת / וָאֶכְתֹּב כָּל אֲשֶׁר בָּאַרְגָּז אֶבֶן יְקָרָה מִתְנוֹסֶסֶת / עִם הַזָּהָב כָּל כְּלִי בְּצוּרָתוֹ / וְאֵיךְ תְּכוּנַת מְלַאכְתּוֹ / לֹא הִנַּחְתִּי בָּאַרְגָּז דָּבָר אֲשֶׁר לֹא כָּתַבְתִּי / וּבְתוֹךְ הָאַרְגָּז הִטְמַנְתִּי / הַשְּׁטָר אֲשֶׁר עַל שְׁמִי יִסַּדְתִּי / וְכָתַבְתִּי אַחֵר כְּמוֹתוֹ / וְשַׂמְתִּיו בְּיָדִי בְאֵיכוּתוֹ וְכַמּוּתוֹ / הַכֹּל עַל שְׁמִי נִכְתַּב וְנֶחְתָּם / נִתְקַבֵּל בְּיַד הַסּוֹחֵר אֲשֶׁר עֲווֹנוֹ נִכְתָּם / וַנֵּצֵא מִן הָאוֹצָר וְהָאַרְגָּז סָגוּר / וְכָל אֲשֶׁר בְּתוֹכוֹ אָגוּר / וַיִּתְמְהוּ חֲבֵרַי עַל אֲשֶׁר לֹא לָקַחְתִּי מְאוּמָה / וַיִּשְׂבְּעוּ חֶרְפָּה וּבוּז וּכְלִמָּה / וָאֹמַר לָהֶם: ׳אִם לֹא תֵצְאוּ מִזֶּה כַּאֲשֶׁר בָּאתֶם / דְּעוּ כִּי אַתֶּם חֲטָאתֶם / וּבְנַפְשְׁכֶם סִכַּנְתֶּם׳ / וַיִּשְׁמְעוּ לַעֲצָתִי / אֲבָל לֹא יָדְעוּ סוֹף דַּעְתִּי / וַנֵּצֵא מִן הַבַּיִת וְלֹא לָקַחְנוּ מְאוּמָה / לְמַעַן נוּכַל בְּתַחְבּוּלָה וְעָרְמָה.

וַיְהִי בַבֹּקֶר וָאֵלֵךְ אֵצֶל הַסּוֹחֵר / וָאֹמַר לוֹ: ׳יֶשׁ לִי מִשְׁפָּט עִמָּךְ אֵצֶל הַקָּצִין לֹא תְאַחֵר׳ / וַאֲנִי לָבוּשׁ לְבוּשׁ הַסּוֹחֲרִים / הַבָּאִים מֵאֶרֶץ הַהַגְרִים / וַיָּקָם הַסּוֹחֵר וַיֵּלֶךְ עִמִּי / אֵצֶל הַקָּצִין וְהוּא הָיָה אִישׁ חֶרְמִי / וַנַּעֲמֹד לְפָנָיו שְׁנֵינוּ / אֲנִי וְהַסּוֹחֵר הַמַּשְׂבִּיעַ בַּטּוֹב עֶדְיֵנוּ / וָאֹמַר אֶל הַקָּצִין: ׳יֵדַע אֲדוֹנֵנוּ / בַּעַל מִשְׁפָּטֵנוּ / כִּי אֲנִי נָתַתִּי לָאִישׁ הַזֶּה פִקָּדוֹן / וְהוּא עַל זֶה לְפָנֶיךָ נִדּוֹן / כִּי לֹא רָצָה לְהָשִׁיב בְּעָרְמָתוֹ / הַפִּקָּדוֹן אֲשֶׁר הָפְקַד אִתּוֹ׳.

וַיֹּאמֶר הַקָּצִין אֶל הַיִּשְׁמְעֵאלִי: ׳מַה תֹּאמַר בְּטַעֲנַת זֶה הָאִישׁ אֲשֶׁר עוֹמֵד לְמוּלִי?׳ / וַיֹּאמֶר הַיִּשְׁמְעֵאלִי: ׳לֹא פִקָּדוֹן אִתִּי, לֹא נִגְלֶה וְלֹא נֶעְלָם / וְלֹא הָיוּ דְּבָרִים מֵעוֹלָם׳ / וָאֹמַר אֶל הַקָּצִין: ׳אֲדוֹנִי / אִם אֶתֵּן רְאָיָה עַל פִּקְדוֹנִי / מַה יִּהְיֶה תַּכְלִית עִנְיָנִי?׳ / וַיֹּאמֶר הַקָּצִין: ׳זֶה גָלוּי וְיָדוּעַ בְּכָל קַצְוֵי נְשִׁיָּה / כָּל הַמּוֹצִיא מֵחֲבֵרוֹ עָלָיו הָרְאָיָה׳ / וַיּוֹסֶף הַיִּשְׁמְעֵאלִי עָלַי לִטְעוֹן / כִּי אָמַר: ׳לִי הַדָּבְרֵי וְרַבְרְבָנַי בְּעוֹן / אֲנִי נֶאֱמָן בְּכָל הַיִּשְׁמְעֵאלִים / וְאַתָּה תִּלְעַג וְאֵין מַכְלִים׳ / וַיֹּאמֶר לוֹ הַקָּצִין: ׳אַל תַּגְבִּיהַּ קוֹלָךְ / שְׁמַע הַטּוֹעֵן לְמוּלָךְ / וַהֲשִׁיבֵהוּ כְּפִי טַעֲנוֹתָיו / אַל תֵּט יָמִין וּשְׂמֹאל מִדַּבְּרוֹתָיו / וּמִקִּלְקוּל הַדִּין עֵינֵי אַנְשֵׁי הַמִּשְׁפָּט יִכְהוּ / כִּי הַמִּשְׁפָּט לֵאלֹהִים הוּא׳ / אָז נִתְחַזֵּק לִבִּי / וְסָר מִמֶּנִּי עָצְבִּי / וָאֹמַר אֶל הַקָּצִין: ׳דַּע, אֲדוֹנִי, כִּי הַשֵּׁם הֱקִימָךְ עַל הַיִּשְׁמְעֵאלִים / לִשְׁפֹּט בֵַּינֵיהֶם וְלַעֲזֹר הָאֲמֵלָלִים / וַאֲנִי בָאתִי מִיָּמִים קַדְמוֹנִים אֶל זֹאת הַמְּדִינָה / זֶה לִי שְׁתֵּים עֶשְרֵה שָׁנָה / וְנִזְדַמֵּן מִכָּאן דֶּרֶךְ עַל יְדֵי מְעַט עֵסֶק / לִמְדִינַת דַּמֶּשֶׂק / וְהָיָה בְיָדִי אַרְגָּז מָלֵא חֲמוּדוֹת / הֱבִיאוֹתִיו מִמְּקוֹם הַמְּצוּדוֹת / קִבַּצְתִּי בוֹ כָל מָמוֹנִי / וּלְהוֹלִיכוֹ בְאוֹתָהּ הַדֶּרֶךְ חָסָה עֵינִי / לְבַל אוֹלִיכֶנּוּ / וְלִפְנֵי הַלִּסְטִים הַמְזֻיָּנִים אַשְׁלִיכֶנּוּ / אָז אָמַרְתִּי לְעַצְמִי כִּי לֹא אֶמְצָא נֶאֱמָן / כְּמוֹ זֶה הָאִישׁ בְּסוֹחֲרֵי הַזְּמָן / וָאֶתֵּן בְּיָדוֹ הָאַרְגָּז לְמִשְׁמֶרֶת / וּבְתוֹכוֹ אִגֶּרֶת / כָּל אֲשֶׁר בּוֹ מַזְכֶּרֶת / וּכְמוֹתָהּ הִנַּחְתִּי בְיָדִי / תִּהְיֶה עֵדִי וְשָׂהֲדִי׳ / וַיֹּאמֶר הַסּוֹחֵר: ׳אִם אֱמֶת דְּבָרֶיךָ אֲשֶׁר דִּבַּרְתָּ / טוֹל מִמְּקוֹם שֶׁהִנַּחְתָּ׳ / וָאֹמַר אֶל הַקָּצִין: ׳יִשְׁלַח אֲדוֹנֵנוּ עִמָּנוּ אֶחָד / לְבַל מִן הַסּוֹחֵר אֶפְחָד / וְנֵלֵךְ וְנָבִיא הָאַרְגָּז שְׁנֵינוּ / מִמְּקוֹם שֶׁהִנַּחְנוּ׳.

וַיִּשְׁלַח הַקָּצִין עִמָּנוּ אֶת נַעֲרוֹ / לְמַעַן הָקִים אֶת דְּבָרוֹ / וַנֵּלֶךְ אֶל בֵּית הַסּוֹחֵר / עִם הַנַּעַר הַשּׁוֹחֵר / וַנִּכָּנֵס דֶּרֶךְ הָאוֹצָר / אֲשֶׁר בְּתוֹכוֹ הָאַרְגָּז בְּמָקוֹם צָר / וַנִּקַּח אוֹתוֹ וַנִּשָּׂאֶנּוּ לִפְנֵי הַשּׁוֹפֵט הַיִּשְׁמְעֵאלִי / וְכַאֲשֶׁר פָּתַח אוֹתוֹ עָמַד נֵס דִּגְלִי / כִּי מָצָא בוֹ אִגֶּרֶת כְּתוּבָה / כָּל הַמַּחֲמַדִּים בְּתוֹכָהּ חֲשׁוּבָה וַאֲנִי נָתַתִּי גַּם כֵּן הָאִגֶּרֶת הַשְּׁנִיָּה / וְהִנֵּה הִיא לְתַלְפִּיּוֹת בְּנוּיָה / לֹא פָחֲתָה וְלֹא הוֹתִירָה / מִכָּל אֲשֶׁר בָּאִגֶּרֶת הָרִאשׁוֹנָה בְּתוֹכָהּ זְכוּרָה / וַיִּתְמַהּ הַסּוֹחֵר וַיִּתְמַרְמַר וַיִּתְאַדָּם / וַיֹּאמַר: ׳הוֹשִׁיעָה אֲדוֹנִי כִּי גָמַר חָסִיד כִּי פַסּוּ אֱמוּנִים מִבְּנֵי אָדָם / חָלִילָה לַסּוֹחֲרִים וֶאֱמוּנָתָם / כִּי כָכָה שֶׁקֶר תַּרְמִיתָם / וַיֵּלֶךְ הַסּוֹחֵר מִלְּפָנֵינוּ בְּבוּשָׁה וּכְלִמָּה / כִּי הָיְתָה מַכָּתוֹ עֲצוּמָה / וְהָאַרְגָּז מִלִּפְנֵי הַקָּצִין לָקַחְתִּי / בְּחַרְבִּי וּבְקַשְׁתִּי / וָאֵלֵךְ אֵצֶל חֲבֵרַי וַיִּתְמְהוּ מִמַּעֲשַׂי הַגְּדוֹלִים / וַיֹּאמְרוּ לִי: ׳בֶּאֱמֶת, כִּי אַתָּה אִישׁ חַיִל רַב פְּעָלִים / לְךָ הוֹדִינוּ / וּלְךָ רָאוּי לִכְרֹעַ עַל בִּרְכֵּנוּ / וְתִהְיֶה מֵעַתָּה אֲדוֹנֵנוּ׳.



את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
