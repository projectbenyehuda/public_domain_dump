אלף בית

אָלֶף הִיא הָאות הַגְּבֶרֶת

אֵין כָּמוֹהָ לְתִפְאֶרֶת,

אָלֶף לִי תַּזְכִּיר יָפֶה

אֶת הַכֹּל אֲשֶׁר אֶרְצֶה,

אִמָּא, אַבָּא, אָח, אָחות,

אֹרֶן, אֶרֶז, אֵת וְאוֹת,

אַף אָרֹךְ אוֹ אֶצְבַּע זֶרֶת,

אָלֶף הִיא הָאוֹת הַגְּבֶרֶת.


בֵּית הִיא הַשְּׁנִיָּה בַּתּוֹר

בְּשִׂמְחָה אוֹתָה אֶזְכֹּר,

בִּנְיָנָהּ הוּא לֹא קָשֶׁה

בַּיִת קָט וְגַם יָפֶה.

בֵּיצָה, בֻּבָּה, בֶּגֶד, בַּד,

בָּחוּר בָּרִיא, בֵּן אוֹ בַּת,

בְּזִכְרוֹנִי יָפֶה אֶשְׁמור:

בֵּית הִיא הַשְּׁנִיָּה בַּתּוֹר.


גִּמֶּל הִיא מִסְפָּר שָׁלֹשׁ

גַּם עִם רֶגֶל, גַם עִם רֹאשׁ.

גּוּף זָקוּף לָהּ וְתָמִיר,

גַּם הִיא הַרְבֵּה דְּבָרִים תַּזְכִּיר.

גַּן, גַּנּון וְגַם גִּנָּה,

גָּמָל, גְּדִי וְגַם גְּדִיָּה,

גִּדְעוֹן גָמַר הַיּוֹם לַחֲרוֹשׁ,

גִּמֶּל אוֹת מִסְפָּר שָׁלֹשׁ.


דָּלֶת הִיא מִסְפָּר אַרְבַּע

דֶּלֶת מַמָשׁ פְּתוּחָה,

דַּע וּזְכוֹר יַלְדִי הָרַךְ,

דְּמוּת הַדָּלֶת אַל תִּשְׁכַּח!

דָּג, דָגָן, דַחְלִיל וְדֹב,

דֶּגֶל, דְּבָשׁ מָתוֹק וְטוֹב,

דֶּקֶל, דֶּרֶךְ, דּוֹד, דּוֹדָה,

דָּלֶת אוֹת מִסְפָּר אַרְבַּע.


הֵא הָאוֹת הַחֲמִישִׁית,

הִיא מְאֹד מְאֹד פִּקְחִית,

הֶחָבִיב, זְכוֹר וּשְׁמוֹר:

הֵא תָּאִיר כְּמִגְּדַּלוֹר.

הַשָּׁמַיִם, תְכוֹל הַיָּם,

הַמוֹלֶדֶת הִיא לָעָם,

הָאָרֶץ הִיא הֵן הַתַּכְלִית,

הֵא הָאוֹת הַחֲמִישִׁית.


וָו שִׁשִׁית הִיא בַּשּׁוּרָה,

וְהִיא וַדָּאִי מְכוּבָּדָה.

וּבְּרֹאשׁ מִלִים רַבּוֹת,

וָו נָחוּץ מְאֹד מְאֹד.

וֶרֶד, וֵשֶׁט, גַּם וִילוֹן,

וָוִית סוֹגֶּרֶת עַל חַלוֹן,

וְשֻׁלְחָן לִי וְסַפָּה,

וָו שִׁשִׁית הִיא בַּשּׁוּרָה.


זַיִן הִיא הַשְׁבִיעִית

זִיוָנִית וְזַנְבָנִית,

זְכור בְּנִי וְאַל תִּטְעֶה:

זוֹ הָאוֹת הַבֵּט וּרְאֵה!

זָרְחָה חַמָּה, הַזְּבוּב זִמְזֵם,

זָמִיר זָרִיז הֵקִיץ הַשְׁכֵּם.

זִכְרוֹנְךָ כְּשֶׁמֶן זָךְ,

זַיִן בְּנִי אַל נָא תִּשְׁכַּח.


חֵית שְׁמִינִית הִיא בַסִּדְרָה

חֲכָמָה וּנְבוֹנָה.

חַיָּב אַתָּה לִזְכּוֹר תָּמִיד:

חֵית הָאוֹת הַגְּרוֹנִית.

חָכָם אַתָּה חָבֵר חָבִיב,

חַרְגל שָׂמַח בְּחֹם אָבִיב,

חֲזוֹר בְּנִי, עוֹד פַּעַם, וְדַע:

חֵית שְׁמִינִית הִיא בַסִּדְרָה.


טֵית הָאוֹת הַתְּשִׁיעִית

טוֹבָה מְאֹד הִיא וְטִבְעִית.

טוֹב לָדַעַת, טוֹב לִלְמוֹד,

טֵית לִזְכּוֹר זֶה קַל מְאֹד.

טָלֶה טִפֵּס לָאֲוִירוֹן

טָעָה וְטָס לְתֵיאַטְרוֹן.

טַל שָׁמַיִם הוּא בְּרָכָה

טֵית תְּשִׁיעִית הִיא בַּשּׁוּרָה.


יוֹד הָעֲשִׂירִית בַּתוֹר,

יֵשׁ לָהּ רֹאשׁ קָטָן, שָׁחוֹר,

יֶלֶד קָט, שְׁמוֹר תָּמִיד:

יוֹד קְטַנָה זוֹ אַל תַּשְׁמִיט.

יַלְדָה יָפָה יָשְׁבָה הַיּוֹם,

יַלְקוּט יָרֹק יָדָה תִּרְקֹם,

יוֹאָב יָצָא הַיּוֹם לִשְׁמוֹר

יוֹד הָעֲשִׂירִית בַּתוֹר.


כָּף הָאוֹת הָאַחַת־עֶשְׂרֵה,

כֶּרֶס לָהּ שָׁמֵן, עָבֶה,

כָּף תּוֹצִיא אֶת הַנְקוּדָה,

כִּי אָז לְכָף הִיא נֶהֶפְכָה.

כָּבוֹד, כֹּכָב, כִּבְשָׂה, כַּוֶרֶת,

כּוֹתֵּב אֲנִי מִכְתָּב לִגְבֶרֶת,

כַּרְבֹּלֶת לוֹ לַתַּרְנְגֹל,

כָּף הִיא נְחוּצָה לַכֹּל.


לָמֶד הִיא הַשְׁתֵּים עֶשְׂרֵה,

לָה הַכֹּבַע הַיָּפֶה.

לְמַד בְּנִי הֵיטֵב הֵיטֵב,

לָמֶד בַּל נָא יֵעָלֵב.

לִוְיָתָן לָמַד לָלֶכֶת,

לְטָאָה לָבְשָׁה מַסֶּכֶת,

לֵיצָן לָשׁ לֶחֶם וְאָכַל,

לָמֶד הוּא נָחוּץ וְקַל.


מֵם הָאוֹת הַשְּׁלֹש עֶשְׂרֵה

מְפֹאֶרֶת לְמַרְאֶה,

מִי חָכָם וְלֹא יֵדַע:

מֵם מְאֹד הִיא נְחוּצָה.

מִרְיָם הַיּוֹם שׁוֹתָה רַק מַיִם,

מְאֹד כְּחֻלִּים הֵם הַשָּׁמַיִם,

מַה מְאֹד טוֹבָה אִמִּי

מֵם נְחוּצָה לָכֶם וְלִי.


נוּן הָאַרְבַּע עֶשְׂרֵה בַּתֹּור

נָעִים מְאֹד אוֹתָהּ לִזְכּוֹר,

נֶאֱמָן הֱיֵה תָּמִיד

נוּן יָקָר כְּבֶן יָחִיד.

נוֹסְעִים הִגִּיעוּ לַנָּמֵל,

נַחְמָן נָסַע לְיַבְנְאֵל,

נֶחְמָד, נָעִים, נִפְלָא הָאוֹר,

נוּן הָאַרְבַּע עֶשְׂרֵה בַּתֹּור.


סָמֶךְ הִיא הַחֲמֵשׁ עֶשְׂרֵה,

סִימָן גַּם טוֹב וְגַּם יָפֶה.

סֵדֶר, בְּנִי, חָשׁוּב מְאֹד,

סָמֶךְ הִיא תַּזְכִּיר אֶת זֹאת.

סוּס, סְנוּנִית, סֻכָּה, סַדְרָן,

סֻלָּם, סֻכָּר, סַבָּל, סַבְלָן,

סוֹדוֹת אַף פַּעַם אַל תְּגַלֶּה,

סָמֶךְ הִיא הַחֲמֵשׁ עֶשְׂרֵה.


עַיִן הַשֵּׁשׁ עֶשְׂרֵה בַּתּוֹר

עַדִינָה הִיא מְלֵאַת אוֹר.

עמוֹד נָא רֶגַע וַחֲשׁוֹב,

עַיִן הִיא עוֹלָם אֵין סוֹף.

עָלֶה, עָדִין, עוֹרֵב, עִוֵּר,

עָגוּר עֲנָק כְּבָר הִתְעוֹרֵר.

עַד מֵאָה תִלְמַד לִסְפּוֹר,

עַיִן הַשֵּׁשׁ עֶשְׂרֵה בַּתּוֹר


פֵּא הָאוֹת הַשְׁבַע עֶשְׂרֵה,

פֹּה צוּיְרָה, הַבֵּט וּרְאֵה.

פִּי מוֹרֶה אִם רַק תִּשְׁאַל,

פֵּא לִשְׁכֹּחַ לֹא תּוּכַל.

פָּרָה, פְּסַנְתֵּר וְגַּם פִּילוֹן,

פָּנִים, פַּרְדֵס וּפַעֲמוֹן,

פִּנְחָס לָמַד הַיּוֹם יָפֶה

פֵּא הָאוֹת הַשְׁבַע עֶשְׂרֵה.


צָדֵי הַשְׁמוֹנָה עֶשְׂרֵה בַּתּוֹר,

צָרִיךְ אַתָּה אוֹתָהּ לִזְכּוֹר.

צִלְצוּלָה עָדִּין וְרַךְ,

צָדֵי בְּנִי אַל נָא תִּשְׁכָּח.

צַד, צַרְצוּר, צִנוֹר, צִיּוֹן,

צַלָּם, צַיָּר, צְמָחִים וְצֹאן,

צִיְצָה, צָחְקָה לָהּ הַצִפּוֹר,

צָדֵי הַשְׁמוֹנָה עֶשְׂרֵה בַּתּוֹר.


קוֹף הַתְּשַׁע עֶשְׂרֵה בַּתּוֹר

קַל מְאֹד אוֹתָהּ לִזְכּוֹר,

קַח בַיָּד הָעֵט וּרְשׁוֹם:

קוֹף תִּמְצָא בְּכָל מָקוֹם.

קַו, קִפּוֹד, קַלְמָר וְקִיר,

קָרוֹן קָטָן קָנָה יָאִיר.

קַלְמָן קָפָא הַיּוֹם מִקּוֹר,

קוֹף הַתְּשַׁע עֶשְׂרֵה בַּתּוֹר.


רֵישׁ הָעֶשְׂרִים הִיא בַּמִסְפָּר

רֹאשָׁהּ רָחָב, גּוּפָהּ קָצָר.

רָצוֹן אִם יֵשׁ לְךָ מְעַט

רֵישׁ מֵאָה פְּעָמִים תִּלְמַד.

רוֹפֵא, רוֹכֵל, רִקּוּד, רָמָה,

רִאשׁוֹן לְחַגִּים רֹאש־הַשָּׁנָה.

רָחוֹק מִמֶּנִי הוּא הַכְּפָר

רֵישׁ הָעֶשְׂרִים הִיא בַּמִסְפָּר.


שִׁין הָאוֹת הָעֶשְׂרִים וְאַחַת

שְׁלֹשָׁה רָאשִׁים לָהּ כְּאֶחָד,

שׁוּר יַלְדִי, הַבְחֵן בַּקוֹל:

שִׁין יָמִין וְשִׂין שֶׁל שְׂמֹאל.

שֶׁמֶן, שַׁעַר וְשִׁנַּיִם,

שָׁט הַשֶּׁמֶשׁ בַשָּׁמַיִם,

שֶׁפַע גֶּשֶׁם בְּחֹדֶשׁ שְׁבָט,

שִׁין הָאוֹת הָעֶשְׂרִים וְאַחַת.


תָּו מִסְפָּר עֶשְׂרִים וּשְׁתַּיִם,

תְּמִירָה, עַל שְׁתֵּי רַגְלַיִם,

תָּמִיד תִּזְכּוֹר הֵיטֵב, וְדַע:

תָּו הָאוֹת הָאַחֲרוֹנָה.

תִּינוֹק, תַּלְתָּל, תֻּכִּי, תָּמָר,

תַּנִין, תָּמִים, תַּלְמִיד, תַּיָּר,

תֵּל־אָבִיב הִיא עִיר יָפָה,

תָּו הָאוֹת הָאַחֲרוֹנָה.



הָאָלֶף־בֵּית בָּזֶה נִגְמַר,

עֶשְׂרִים וּשְׁתַּיִם בְּמִסְפָּר.

אַךְ עָלֶיךָ גַּם לִלְמוֹד:

הָאוֹתִיוֹת הַסוֹפִיוֹת.


לֹא קָשֶׁה עִבְרִית לִלְמוֹד

תְּצָרֵף רַק אוֹת לְאוֹת.

וְלִקְרֹא גַּם קַל מְאֹד,

אִם תִּלְמַד אֶת הַנְקוּדוֹת.

וְאֵלּוּ הֵן:


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
