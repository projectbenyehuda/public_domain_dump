תחכמוני - שער עשירי

תחכמוני – שַׁעַר עֲשִׂירִי / יהודה אלחריזי

בַּחֲסִידוּת הַכָּפְרִי, כְּעֵת בַּמָּרוֹם יַמְרִיא

נְאֻם הֵימָן הָאֶזְרָחִי: נָסַעְתִּי מֵעֵמֶק הַשִּׂדִּים. אֶל אֶרֶץ כַּשְׁדִּים. וְכַאֲשֶׁר זָרַחְתִּי בִּמְרוֹמֶיהָ. וְדָרַכְתִּי הֲדוֹמֶיהָ. יָצָאתִי לְהִתְהַלֵּךְ בָּאָרֶץ לִרְאוֹת הָעוֹבְרִים וְהַשָּׁבִים. וְהָאֶזְרָחִים וְהַתּוֹשָׁבִים. וְהַמִּזְרָחִים וְהַמַּעֲרָבִים. וָאֶשָּׂא עֵיַני וָאֵרֶא אִישׁ יוֹשֵׁב עַל מַצֵּבָה לְבַדּוֹ. וָאֵשֵׁב לְצִדּוֹ. כְּמִשְׁפַּט האוֹרְחִים וְהַגֵּרִים. אֲשֶׁר אִישׁ לְאָחִיו מִתְחַבְּרִים. וְקוֹרוֹת זְמַנָּם מְסַפְּרִים. וָאֹמַר לוֹ: מָה עִנְיַן שִׁבְתֶּךָ. וּמַה מְלַאכְתֶּךָ. אָמַר: אֲנִי אִישׁ נָכְרִי הוֹלֵךְ מֵחַיִל אֶל חָיִל. פַּעַם אָצוּד שׁוּעָל וּפַעַם אֶטְרֹף אָיִל. אָמַרְתִּי לוֹ: הַגִּידָה לִי מִקְצָת עִנְיָנֶיךָ. וְהַשְׁמִיעֵנִי מֵאֲשֶׁר רָאוּ בְתֵבֵל עֵינֶיךָ.

אָמַר: רָאִיתִי בַדְּרָכִים אֲשֶׁר הָלַכְתִּי. וּבַנְּתִיבוֹת אֲשֶׁר דָּרַכְתִּי. כַּמָּה נוֹרָאוֹת. וְכַמָּה נִפְלָאוֹת. וְלֹא רָאוּ עֵינַי דָּבָר נִפְלָא לְשׁוֹמְעָיו. וְנֶחְמָד לְיוֹדְעָיו. כַּאֲשֶׁר רָאִיתִי אֶמֶשׁ. לִפְנֵי בוֹא הַשֶּׁמֶשׁ. וָאֶשְׁאָלֵהוּ עַל הַפְּלִיאָה הַנּוֹרָאָה. אֲשֶׁר רָאָה.

אָמַר: הָיִיתִי עִם חֲבֵרִים נְעִימִים. מִזֶּרַע הַמְּלוּכָה וּמִן הַפַּרְתְּמִים. נִדְבוֹתָם גְּשָׁמִים. וּבִרְכוֹתָם אֲסָמִים. וְתִימְרוֹתָם בְּשָׂמִים. וְאִמְרוֹתָם לְשָׁמִים. וּפְנֵיהֶם מְאוֹרִים. וּמַרְאֵיהֶם שְׁחָרִים. וִידֵיהֶם נְהָרִים. בְּאוֹרָם הַשֶּׁמֶשׁ וְהַיָּרֵחַ יַכְלִימוּ. וְכוֹכְבֵי לַיְלָה יַעֲלִימוּ. וְלַיְלָה לְיוֹם יָשִׂימוּ. וְכַאֲשֶׁר עָלָה אֶתְמוֹל הַשַּׁחַר קַמְנוּ. וְהִשְׁכַּמְנוּ וְהִסְכַּמְנוּ. לָצֵאת לִכְפָרִים. וּלְהִתְעַלֵּס בֵּין נְרָדִים וּכְפָרִים. אוֹ בְאַחַד הַיְּעָרִים. וְצִוִּינוּ הַנְּעָרִים לֶאֱסֹר לָנוּ מַרְכְּבוֹתֵינוּ. וְיָצָאנוּ וּנְעָרֵינוּ סְבִיבוֹתֵינוּ. וַאֲנַחְנוּ עַל סוּסִים מְזֻיָּנִים. לְהַשִּׂיג הַבְּרָקִים מוּכָנִים. וַיֹּאמֶר אֵלֵינוּ. רֹאשׁ חֲבוּרָתֵנוּ. הֲלֹא תֵלְכוּ עִמִּי לְמִבְחַר הַכְּפָרִים. עַל אַחַד הֶהָרִים. כִּי יֶשׁ לִי שָׁם כְּפָר נֶחְמָד. יָפְיוֹ לֹא יִמָּד. וּבוֹ כָּל נֹעַם נִצְמָד. וְכָל אֲנָשָׁיו הֵם עֲבָדַי. וּגְמוּלֵי חֲסָדַי. אָמַרְנוּ לוֹ: כֹּל אֲשֶׁר תִּגְזֹר עָלֵינוּ נַעֲשֶׂה. וּבְצִלְּךָ נֶחֱסֶה. וְהָלַכְנוּ כָּל הַיּוֹם עַד עֲבֹר הַצָּהֳרָיִם. וְהִגַּעְנוּ עַד הַכְּפָר בֵּין הָעַרְבָּיִם. וְהַשֶׁמֶשׁ נָטְתָה לָרֹב. פָּנְתָה לַעֲרֹב. וְהַלַּיְלָה קָרוֹב. וְכַאֲשֶׁר נִסְתְּרָה הַחַמָּה בִּמְעוֹנֶיהָ. וְכִסְּתָה פָנֶיהָ. בְּמִשְׁבְּצוֹת שָׁנֶיהָ. בָּאנוּ אֶל הַכְּפָר וְהִנֵּה הַכָּפְרִים עוֹבְדֵי הָאֲדָמָה יָצְאוּ בְשִׂמְחָה לִקְרָאתֵנוּ. וְהִשְׁתַּחֲווּ לְעֻמָּתֵנוּ. וּבָא הַגָּדוֹל מֵהֶם וְהוֹרִידָנוּ מֵעַל מַרְכְּבוֹתֵנוּ. וְלָקַח הַבְּהֵמוֹת. וֶהֱבִיאָן לְמִבְחַר אֲדָמוֹת. לְהַשְׂבִּיעָם וּלְכַלְכְּלָם. וּבִנְאוֹת דֶּשֶׁא יַרְבִּיצֵם וְעַל מֵי מְנוּחוֹת יְנַהֲלָם. וְהוּא שָׁב אֵלֵינוּ בְּשִׂמְחָתוֹ. וַיְבִיאֵנו אֶל בֵּיתוֹ. וְהִנֵּה דִּירָה רְחָבָה. כַּמִּדְבָּר וְכָעֲרָבָה. וּבְתוֹכָהּ הַתֶּבֶן פְּנִימִי וְחִיצוֹן. וּצְפִיעֵי הַבָּקָר וְהַצֹּאן. וַיְמַהֵר לְפַנּוֹת הַבַּיִת. וּלְהָסִיר הַשָּׁמִיר וְהַשַּׁיִת. וְאֶת הַבַּית יַקְצִיעַ מִבַּיִת. וְטִהַר צַחֲנָתוֹ וְחֶלְאָתוֹ. וְהֵסִיר אֶת מֻרְאָתוֹ. וְלָקַח נָשָׁיו וּבְנוֹתָיו וּבַבַּיִת הַפְּנִימִי הֶחְבִּיאָם. וְחֶדֶר בְּחֶדֶר הֱבִיאָם. וְהוּא בָא לְפָנֵינוּ. וַיֹּאמֶר אֵלֵינוּ: אִם טוֹב בְּעֵינֵיכֶם. אַגִּישׁ לִפְנֵיכֶם. גְּבִינָה נִקְפָּאָה. וְחָלָב וְחֶמְאָה. אָמַרְנוּ: אֵיפֹה הוּא. כִּי אֵין כָּמוֹהוּ. וַיְמַהֵר אֶל הָעֲדָרִים. לְמַלֹּאת בַּחֲלֵב הָעִזִּים סֵפֶל אַדִּירִים. וּבְרֶגַע הִקְפִּיאָהוּ. וּלְפָנֵינוּ הֱבִיאָהוּ. וַיֹּאמֶר לָנוּ: אִכְלוּ מַשְּׁמַנִּים. וּרְווּ

מַעֲדַנִּים. כִּי אֲנַחְנוּ שׁוֹכְנֵי הֶהָרִים. וְיוֹשְׁבֵי הַכְּפָרִים. נַפְשֵׁנוּ נְדִיבָה. וְיָדֵנוּ רְחָבָה. לְקַבֵּל כָּל אֹרֵחַ בְּאַהֲבָה. וְאֵין עֵינֵינוּ רָעָה. וְלֹא נִדְבָתֵנוּ כְּשׁוֹכְנֵי הַמְּדִינוֹת גְּרוּעָה. כִּי מַתְּנָתָם חֲסֵרָה. וְיָדָם קְצֵרָה. וְעֵינָם צָרָה. וַאֲנַחְנוּ כָּל שִׂמְחָתֵנוּ לְקַבֵּל הָאֹרְחִים. וְלַהֲבִיאָם בַּעֲלִיוֹת מְרֻוָּחִים. וּלְהַשְׂבִּיעָם עוֹלוֹת מֵחִים. וְלָשׂוּם לִפְנֵיהֶם דְּמֵי הַלֵּב וְהַנְּתָחִים.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

לְהַכְנִיס אֹרְחִים יִמְתַּק לְחִכִּי

כְּמִיץ חָלָב בְּחֶמְאַת הַבְּקָרִים.

וְכַפִּתִּים בְּתוֹךְ חֶמְאָה שְׁרוּיִם

בְּעֵת כִּי יִצְבְּרוּ אוֹתָם חֳמָרִים.

וְעֵת קוֹל אֹרְחִים אֶשְׁמַע חֲצוֹת לֵיל

לְבָנַי אֶקְרְאָה פִּתְחוּ שְׁעָרִים

וְשִׂמְחָתִי בְּאוֹרֵחַ כְּשִׂמְחַת

זְאֵב גָּנַב גְּדִי מִן הָעֲדָרִים.

וְכָל-יוֹם אֶחֱרשׁ אַדְמַת נְדָבוֹת

כְּחֹרֵשׁ הָאֲדָמָה בַּבְּקָרִים.

וְיִיטַב לִי עֲשׂוֹת חֶסֶד כְּרָעֵב

בְּמָצְאוֹ פַּת יְבֵשָׁה עִם חֲצִירִים.

וְעֵינִי לַחֲזוֹת אֹרְחִים תְּקַוֶּה

כְּעֵינֵי הַחֲמוֹרִים לַשְּׂעוֹרִים.


אַחַר כָּל זֶה לָקַח כֶּבֶשׂ בֶּן שְׁנָתוֹ. וַיְמַהֵר לַעֲשׂוֹת אוֹתוֹ. וַיְשִׂימֵהוּ לְפָנֵינוּ. וְהוּא עוֹמֵד עָלֵינוּ. וְכַאֲשֶׁר הִשְׁלַמְנוּ לֶאֱכֹל הֵכִין לָנוּ. מִטּוֹת רַעֲנַנּוֹת. וּמְנוּחוֹת. שַׁאֲנַנּוֹת. וְשָׁכַבְנוּ וְרָגַעְנוּ. מִן הַיְגִיעָה אֲשֶׁר יָגָעְנוּ.

וַיְהִי בַבֹּקֶר וְהַשֶּׁמֶשׁ הִשְׁקִיפָה מֵאֶשְׁנַבֵּי גְלִילֹותֶיהָ. וְגִלְּתָה מְעִילֶיהָ. וַתָּסַר צְעִיפָהּ מֵעָלֶיהָ. קַמְנוּ מִמִּטּוֹתֵינוּ. לַעֲרֹךְ לָאֵל תְּפִילָּתֵנוּ. וְרָכַבְנוּ עַל סוּסֵינוּ. וְיָצָאנוּ מִמְּכוֹנֵינוּ. וְשַׂמְנוּ אֶל הַמִּדְבָּר מְגַמַּת פָּנֵינוּ. לִלְטשׁ בְּמַרְאֵה הַצִּצִּים וְהַצְּמָחִים עֵינֵינוּ. וּלְעוֹרֵר יְשֵׁנֵי רַעְיוֹנֵינוּ. וְהִתְהַלַּכְנוּ בֵּין צִלְלֵי גַנִּים. דְּשֵׁנִים וְרַעֲנַנִּים. מְלֵאִים מַעֲדַנִּים. וְכַאֲשֶׁר הִגִּיעָה עֵת הָאֹכֶל וְהִנֵּה הַכָּפְרִי בָא לִקְרָאתֵנוּ בְּשִׂמְחָתוֹ. וַיְבִיאֵנוּ אֶל בֵּיתוֹ. מָצָאנוּ מְקוֹמוֹת מוּכָנִים. וּמַטְעַמִּים מְזֻמָּנִים. וְהַקְּדֵרוֹת יַרְתִּיחוּ. וּמֵרָחוֹק יָרִיחוּ. וְהַיְרָקוֹת מָלְאוּ הַסִּיר. וְכָל הַבָּשָׂר חָצִיר. וַיָּשֶׂם לְפָנֵינוּ קְעָרוֹתָיו. מְלֵאוֹת בְּמַטְעַמּוֹתָיו.


אָמַר הַמַּגִּיד: וְרָאִיתִי בְּבֵית הַכָּפְרִי תַּרְנְגוֹל. אֶרֶךְ הָאֵבֶר וְגָדוֹל. רַךְ וָטוֹב. דָּשֵׁן וְרָטֹב. וְאִוִּיתִי לְאָכְלוֹ. כִּי כָל הַלַּיְלָה הֶעִירַנִי מִשְּׁנָתִי בְּקוֹלוֹ. וָאֹמַר אֶל הַכָּפְרִי: אַתָּה הִגְדַּלְתָּ טוֹבָתְךָ עָלֵינוּ. וְלַעֲרֹךְ תּוֹדָה לְךָ חוֹבָה עָלֵינוּ. אֲבָל אֲנִי חוֹלֶה וְלִבִּי זָעֵף וָסַר. וְצִוּוּנִי הַרוֹפְאִים לִבְלִי אֱֹל בָּשָׂר. וַאֲנִי מִתְאַוֶּה לֶאֱכֹל. מִבְּשַׂר זֶה הַתַּרְנְגוֹל. אָמַר לִי הַכָּפְרִי: אִלּוּ הִתְאַוִּיתָ בְּשַׂר בְּנִי. לֹא חָסָה עָלָיו עֵינִי. וַיְמַהֵר הַכָּפְרִי הוּא וְאִשְׁתּוֹ. אַחֲרֵי הַתַּרְנְגוֹל לְקַחְתּוֹ. וְהַתַּרְנְגוֹל בָּרַח בְּיָגוֹן וּמְגִנָּה. מִגַּג אֶל גַּג וּמִפִּנָּה אֶל פִּנָּה. וְכַאֲשֶׁר יָגַע הַכָּפְרִי וְלֹא רָגַע. וְרָדַף אַחַר תַּרְנְגוֹלוֹ. וַיַּרְא כִּי לֹא יָכֹל לוֹ. וַיִּקְרָא בְּנֵי בֵיתוֹ לַעֲזֹר לוֹ. וַיָּרוּצוּ אַחֲרָיו הַנְּעָרִים. בְּשִׂמְחָה וּבְשִׁירִים. וְנַעַר אֶחָד מֵהֶם הִשִּׂיגָהוּ. וּבְרַגְלוֹ הֶחֱזִיקָהוּ. וְהַתַּרְנְגוֹל בִּמְרִירוּת הַמָּוֶת נְשָׁכָהוּ. עַד עֲזָבָהוּ. וּבָרַח הַתַּרְנְגוֹל וְנִמְלַט מִיַּד כֻּלָּם. וְעָלָה עַל רֹאשׁ הַסֻּלָּם. וְנָתַן תְּהִלָּה וָשֶׁבַח לַאֲדוֹן עוֹלָם. בְּיוֹם הִצִּיל ה' אוֹתוֹ מִכַּף כָּל אוֹיְבָיו כֻּלָּם.


אַחֲרֵי כֵן עָלָה עַל גַּג בֵּית הַתְּפִילָּה. וְהָיָה שָׁם קְהִלָּה גְדוֹלָה. וְקוֹל הֲמוּלָה. וְנָשָׂא קוֹלוֹ בִּנְגִינָה. וְאָמַר: בָּרוּךְ אַתָּה ה' הַנּוֹתֵן לַשֶּׂכְוִי בִינָה. וַיִּתְקַבְּצוּ כָּל הָעָם סְבִיבוֹ. לִשְׁמֹעַ נִיבוֹ. וַיַּעַן וַיֹּאמֶר לָהֶם: שִׁמְעוּ אֵלַי רוֹדְפֵי צֶדֶק אַנְשֵׁי תְהִלּוֹת. הַבָּאִים לַעֲרֹךְ תְּפִלּוֹת. לְנוֹרָא עֲלִילוֹת. וְלִשְׁפֹּךְ לְבָבוֹת. לְשׁוֹכֵן עֲרָבוֹת. וָלָשֵׂאת כַּפַּיִם. אֶל אֵל בַּשָּׁמַיִם. מַה יּוֹעִילוּ הַתְּפִלּוֹת. וְנַפְשׁוֹתֵיכֶם בַּדָּם מְגֹאָלוֹת. וּמַדּוּעַ תַּרְבּוּ תַחֲנוּנִים. וְאַתֶּם שׁוֹפְכִים דַּם נְקִיִּים וְאֶבְיוֹנִים. וְאֵיךְ לְהִתְפַּלֵּל תַּחֲשֹׁבוּ. וּבְכָל יוֹם לַהֲרֹג נְקִיִּים תֶּאֱרֹבוּ. לֹא תוֹסִיפוּ הָבִיא מִנְחַת שָׁוְא וּבֵית אֵל אַל תָּבוֹאוּ. יְדֵיכֶם דָּמִים מָלֵאוּ. תְּפִלַּתְכֶם לָצוֹן. וְלֹא לְרָצוֹן. וּבָהּ לֹא יְכֻפַּר עָוֹן. כִּי הִיא מִנְחַת זִכָּרוֹן וּמַזְכֶּרֶת עָוֹן. וַיִּתְמְהוּ הָעָם לִתְלוּנָתוֹ. וַיִּשְׁאֲלוּ אִישׁ לְאָחִיו עַל הַמִקְרֶה אֲשֶׁר קָרָה אוֹתוֹ.

וַיַּעַן וַיֹּאמֶר לָהֶם: אֲלֵיכֶם אִישִׁים אֶקְרָא. מְשִׁיבֵי מִלְחָמָה שָׁעְרָה. הַאֵין יִרְאַת ה' עַל פְּנֵיכֶם. וְאִם אֵין מִשְׁפַּט צֶדֶק בֵּינֵיכֶם. וְאִם תִּבְגְּדוּ בְּכָל הַבָּא בְּצֵל קוֹרַתְכֶם. מִי יִבְטַח בְּאַהֲבַתְכֶם. וְאִם תָּרֵעוּ לְבַעֲלֵי בְרִיתְכֶם. מִי יַאֲמִין עוֹד בָּכֶם. הַנְקַלָּה בְעֵינֵיכֶם לִבְגֹּד בְּאַנְשֵׁי הָאַהֲבָה. וּלְשַׁלֵּם רָעָה תַּחַת טוֹבָה. הֲלֹא מִיּוֹם בָּאתִי בְּצֵל קוֹרַתְכֶם. עָבַדְתִּי בְּכָל כֹּחִי אֶתְכֶם. וְכָל הַלַּיְלָה אָעִיר יְשֵׁנֵיכֶם. לְהִתְפַּלֵּל לֵאלֹהֵיכֶם. וּבְכָל הַיּוֹם אֲרַנֵּן לִפְנֵיכֶם. לְשַׂמֵּחַ לְבַבְכֶם. וְהִרְבֵּיתִי זֶרַע בְּצִלְּכֶם. וְעָשִׂיתִי בָנִים וּבָנוֹת לְהַמְתִּיק בְּטַעְמָם חִכְּכֶם. וְאַתֶּם לְהָרַע לִי מַחְשְׁבוֹתֵיכֶם. וּלְהָרְגֵנִי כָּל תַּאֲוַתְכֶם. מַמְרִים הֱיִיתֶם עִם ה' מִיּוֹם דַּעְתִּי אֶתְכֶם. וְכָל מַחְשְׁבוֹתֵיכֶם לִשְׁחֹט וְלֶאֱכֹל אוֹתִי. הֵן בְּעוֹדֶנִּי חַי עִמָּכֶם מַמְרִים הֱיִיתֶם וְאַף כִּי אַחֲרֵי מוֹתִי. הִשְׁבַּעְתִּי אֶתְכֶם: מַה מְּצָאתֶם בִּי עָוֶל כָּל יָמַי וְשָׁנַי. הִנְנִי עֲנוּ בִי נֶגֶד ה‘. הֵן אַתֶּם כֻּלְּכֶם חֲזִיתֶם. וּמַעֲשַׂי רְאִיתֶם. כִּי אֵין מְלַאכְתִּי בְּבֵיתִי. רַק תְּחִנָּתִי וְשִׁירָתִי. וְאֵין עִסְקִי רַק לְבַקֵּש מִחְיָתִי. וּלְנַגֵּן לָאֵל בְּשִׁבְתִּי וְקוּמִי. וּלְשׁוֹרֵר בְּלֵילִי וְיוֹמִי. יוֹמָם יְצַוֶּה ה’ חַסְדּוֹ וּבַלַּיְלָה שִׁירוֹ עִמִּי. וְעַתָּה אַחֲרֵי אֲשֶׁר זָקַנְתִּי. וּבַעֲבוֹדַתְכֶם נוֹשַׁנְתִּי. וְחָלְפוּ יָמַי וְשָׁנַי. וְהִגִּיעוּ יְמֵי זְקוּנַי. תְּשַׁלְּמוּנִי רָעָה תַּחַת טוֹבָה. וְתִגְמְלוּנִי תַּחַת אַהֲבָה אֵיבָה. וְתִרְצוּ מֵעַל בָּנַי לְהַפְרִידֵנִי. וּמֵחֵיק נָשַׁי לְגָרְשֵׁנִי. לַעֲזֹב יְתוֹמוֹת הַבָּנוֹת. וְנָשַׁי עֲגוּנוֹת אַלְמָנוֹת. וְאִלּוּ מְצָאתֶם בִּי מוֹעִיל הֶחֱרַשְׁתִּי. וַחֲמָסִי מִכֶּם לֹא דָרַשְׁתִּי. אֲבָל בְּשָׂרִי מִצּוּר קָשֶׁה. לֹא יֵרָצֶה לְקָרְבַּן אִשֶּׁה. וְכָהֲתָה עֵינִי וְנָס לֵחִי. וְיָבֵשׁ כַּחֶרֶשׂ כֹּחִי. וְשָׁבוּ כַבַּרְזֶל עֲצָמַי. לְאֹרֶךְ שְׁנוֹתַי וְרֹב יָמַי. וְרֵיחִי נָמָר. וּבְשָׂרִי סָמָר. וְטַעְמִי מָר. וֹנְתָחַי קָשִׁים לְטוֹעֲמֵיהֶם. כִּי מָרִים הֵם. וּמְרַק פִּגּוּלִים כְּלֵיהֶם. וּבָטְלוּ טוֹחֲנוֹתַי. וְחָשְׁכוּ רֹאוֹתַי. וְלֹא נִשְׁאַר בָּשָׂר בְּצַלְעוֹתַי. וְלֹא חֵלֶב בְּכִלְיוֹתַי. כַּמַּיִם נִשְׁפַּכְתִּי וְהִתְפָּרְדוּ כָל עַצְמוֹתַי. אִם יֹאכְלֵנִי אִישׁ חוֹלֶה יָמוּת בְּיוֹמוֹ. וְיִשָּׂא עֲוֹנוֹ עַל שִׁכְמוֹ. וְלֹא יְכֻפַּר אֲשָׁמוֹ. וְאִם יֹאכְלֵנִי אִישׁ בָּרִיא לֹא יַחֲלִיף בִּי כֹּחַ. וְלֹא יִמְצָא מָנוֹחַ. וְעַתָּה לָמָּה תִגְמְלוּנִי הַגְּמוּלָה הַזֹּאת. וְתַשְׁאִירוּ לִבְנֵיכֶם דִּבָּה כָזֹאת. הֲלֹא יֵשׁ לָכֶם בְּבָנַי תְּמוּרָה. וּמַחְמַד תְּשׁוּרָה. כִּי הֵם נְעִימִים וְזַכִּים. וְטוֹבִים וְרַכִּים. וְכָל אֶחָד מֵהֶם רַבּוּ שְׁבָחָיו. וְרַכּוּ נְתָחָיו. וְכַבֹּשֶׂם רְקָחָיו. וּמָלְאוּ שֶׁמֶן צַלְעוֹתָיו. וּמֹחַ עַצְמוֹתָיו. וּבְנוֹתַי כֻּלָּן שְׁמֵנוֹת וּמְתוּקוֹת. וּלְכָל לָשׁוֹן עֲרֵבוֹת וְלַלֵּב חֲשׁוּקוֹת. וְטַעְמָן מַמְתִּיק הַפֶּה. וְאִם יֹאכְלֵם הַחוֹלֶה יִתְרַפֵּא. וְלֹא יִצְטָרֵךְ לְרוֹפֵא. עָלַי אָמְרָה הַתּוֹרָה בַּזְּמַן שֶׁחָלַף וְהָלָךְ. שַׁלַּח תְּשַׁלַּח אֶת-הָאֵם וְאֶת-הַבָּנִים תִּקַּח-לָךְ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

מְיֻדָּעַי וְרֵעַי רַחֲמוּנִי

וְעַל גֹּדֶל יְגוֹנִי נַחֲמוּנִי.

הֲכָזֶה יִהְיֶה תַגְמוּל אֱמוּנִים

אֲשֶׁר עַל טוֹב בְּרָעָה יִגְמְלוּנִי.

יְדִידִים שָׁכְחוּ חֶסֶד גְּמַלְתִּים

וְהֵם תָּמִיד לְהָרְגִי יִדְרְשׁוּנִי.

אֲנִי אֶרְדֹּף שְׁלוֹמוֹתָם וְטוּבָם

וְתַחַת אַהֲבָתִי יִשְׂטְנוּנִי.

וְתָמִיד אֶשְׁמְעָה עָלַי עֲצָתָם

וְכָל הַיּוֹם כְּלוֹחֵם יִלְחֲצוּנִי.

וְקִדַּמְתִּים בְּבָנִים גַּם בְּבָנוֹת

וְהֵם בַּמַּאֲכֶלֶת קִדְּמוּנִי.

וְהֶאְכַלְתִּים בְּשַׂר בָּנַי. וְהֵמָּה

מְבַקְשִׁים תֹּאֲנָה עַד יֹאכְלוּנִי.

אֲנִי זָקֵן וְאֵין בָּשָׂר בְּגוּפִי

וְלָמָּה זֶה לְאָכְלִי יַחְמְדוּנִי.

רְדָפוּנִי בְרִשְׁעָתָם וְאִלּוּ

אֲחָזוּנִי כְּרֶגַע אִבְּדוּנִי.

אֲבָל הָאֵל עֲזָרַנִי בְּחַסְדּוֹ

וְנִמְלַטְתִּי בְּעֵת כִּי בִקְשׁוּנִי.

וְלוּלֵא צוּר כְּתָבַנִי לְחַיִּים

לְהָרְגִי נִקְבְּצוּ וַיִּשְׁחֲטוּנִי.

וְנָשַׁי גֵּרְשׁוּ מִבֵּין זְרוֹעַי

וּמֵעַל תַּרְנְגוֹלַי רִחֲקוּנִי.


אָמַר הַמַּגִּיד: וְכִשְׁמֹע הָעָם תְּלוּנָתוֹ. יָדְעוּ צִדְקָתוֹ וֶאֱמוּנָתוֹ. וְחָרָה מְאֹד לָהֶם. וַיֵּרַע הַדָּבָר בְּעֵינֵיהֶם. וְשָׁלְחוּ לִקְרֹא לִבְעָלָיו. וְדִבְּרוּ קָשׁוֹת אֵלָיו. וְאָמְרוּ לוֹ: אֵיךְ חָשַׁבְתָּ לַעֲשׂוֹת. נְבָלָה כָּזֹאת. לִשְׁלֹחַ יָד בִּזְקֵנֵנוּ וּמוֹרֵנוּ. אֲשֶׁר לְדַרְכֵי אֱלֹהִים יוֹרֵנוּ. וּמִשְּׁנָתֵנוּ לִתְפִלָּה יְעִירֵנוּ. אָמַר הַכָּפְרִי: אֱמֶת דִּבַּרְתֶּם כִּי מִיּוֹם קְנִיתִיהוּ. יָרֵא אֶת ה' רְאִיתִיהוּ. מִתְפַּלֵּל בְּכָל עֵת מְצָאתִיהוּ. אֲבָל בָּאוּ הֵנָּה הַלַּיְלָה מִבְּנֵי יִשְׂרָאֵל חֶבְרַת נְדִיבִים. וּגְבִירִים חֲשׁוּבִים. וְאִישׁ מֵהֶם חָלָה וְצִוּוּ הָרוֹפְאִים עָלָיו. לְהַרְחִיק הַבָּשָׂר וְאַל יִקְרַב אֵלָיו. וְשָׁאַל מִמֶּנִּי בְּפִיּוּס גָּדוֹל. לַעֲשׂוֹת לוֹ זֶה הַתַּרְנְגוֹל. וְלֹא אוּכַל לְהָשִׁיב שׁוֹאֲלַי רֵיקָם. עַד אֲמַלֵּא סִפְקָם. אָמְרוּ לוֹ כָל הַזְּקֵנִים: הֵיטַבְתָּ. בַּאֲשֶׁר הִתְנַדַּבְתָּ. אַךְ בִּשְׁבִילֵנוּ תְּפַיְּסֵהוּ. לְהַנִּיחוֹ וּבְזוּלָתוֹ יְמִירֵהוּ. וְהִשְׁבִּיעוּ הַכָּפְרִי לְבַל יַעֲשֶׂה בּוֹ רָע. וּבוֹ יָדוֹ אַל תִּגָּע. אָמַר לָהֶם: דְּעוּ כִּי לֹא אֶעֱשֶׂה לוֹ נֵזֶק לְעֵילוֹם. וְהִנְנִי נוֹתֵן לוֹ אֶת בְּרִיתִי שָׁלוֹם. וְכִשְׁמֹעַ הַתַּרְנְגוֹל הָאָלָה וְהַשְּׁבוּעָה. נָשָׂא קוֹלוֹ בְּשִׁירָה וַיָּרַע תְּרוּעָה. עַל הַנִּסִּים וְעַל הַפֻּרְקָן וְעַל הַפְּדוּת וְעַל הַתְּשׁוּעָה. בִּרְאוֹתוֹ כִּי נִחַם ה' עַל הָרָעָה.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

יְרַחֵם אֵל אֲצִילִים רִחֲמוּנִי

וּבִנְעִימוּת דְּבָרִים נִחֲמוּנִי.

וְלֹא אָבְדָה אֱמוּנָה מִלְּבָבָם.

וְעֵת צַר לִי בְּחַסְדָּם קִדְּמוּנִי.

וְלוּלֵי הֵם. מְשַׂנְאַי מָשְׁלוּ בִי

וְעַצְמִי רִצֲצוּ וַיְעַצְּמוּנִי.


אָמַר הַמַּגִּיד: וְכַאֲשֶׁר כִּלָּה הָאִישׁ דְּבָרוֹ וְנָאֳמוֹ. שְׁאִלְתִּיהוּ עַל שְׁמוֹ:


וַיַּעַן וַיֹּאמַר:

אֲנִי חֶבֶר מְחַבֵּר שִׁיר יְדִידוֹת

מְמֻלָּא מִפְּנִינֵי הַתְּעוּדוֹת.

וְאַחְבִּיר מִלְּבָבִי נִיב הֲתֻלִּים

בְּמֶתֶק שִׁיר וְטוּב מָשָׁל וְחִידוֹת.


אָמַר הַמַּגִּיד: וּכְשָׁמְעִי שִׁירוֹ הִכַּרְתִּיהוּ. וְיָדַעְתִּי כִּי חֶבֶר הַקֵּינִי הוּא. וְאָמַרְתִּי לוֹ: מַה נָּעֲמוּ שַׁעֲשׁוּעֶיךָ. וּמֶה עָצְמוּ דִבְרֵי תַּעְתּוּעֶיךָ. חַי ה' אֵלֶּה הַדְּבָרִים לֹא שְׁמַעְתָּם מִבָּשָׂר וָדָם. כִּי מִלִּבְּךָ אַתָּה בּוֹדְאָם. וַיִּשְׂחַק לְמִלָּי. וַיִּסֹּב לְדַרְכּוֹ מֵעָלַי.
