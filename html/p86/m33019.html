ספר תחכמוני - שַׁעַר עֶשְׂרִים וְאַרְבָּעָה

בִּשְׁלִיחַ צִבּוּר וּתְפִלּוֹתָיו. וְסִפּוּר שִׁבּוּשָׁיו וְטָעֻיוֹתָיו.


נְאֻם הֵימָן הָאֶזְרָחִי: הַגֵּד הִגִּיד לִי חֶבֶר הַקֵּינִי. וְאָמַר: אַגִּידָה נָא לְךָ מִן הַנִּפְלָאוֹת אֲשֶׁר רָאֲתָה עֵינִי. כִּי נָסַעְתִּי מֵאֶרֶץ גְּשׁוּר לְאַשּׁוּר. בְּעֵת חָרְבוּ יְדֵי נְדִיבֶיהָ כְּחַלְמִישׁ צוּר. וְאֶפֶס עָזוּב וְעָצוּר. וְהָעִיר נָאָה וַחֳשׁוּבָה. גְּדוֹלָה מְאוֹד וּרְחָבָה. וְהָמוֹן רָב וְעשֶׁר נִמְצָא בְּקִרְבָּהּ. וְהָאֵמִים לְפָנִים יָשְׁבוּ בָהּ. וְאֵין בְּכָל הָאֲרָצוֹת כְּיָפְיָהּ. וְלֹא כַהֲדָרָהּ וְצִבְיָהּ. וְגַם זָבַת חָלָב וּדְבַשׁ הִיא וְזֶה פִרְיָהּ. אֶּפֶס כִּי עַז הָעָם הַיּוֹשֵׁב עָלֶיהָ. הַשּׁוֹכְנִים מִבְּנֵי עַמֵּנוּ בִּגְבוּלֶיהָ. כִּי לֹא יִתְנַדֵּב אַחַד מֵהֶם לְנִכְבָּד אוֹ בְלִיַּעַל. מִחוּט וְעַד שְׂרוֹךְ נַעל. וְכֻלָּם צְרוֹרוֹת כַּסְפֵּיהֶם. צְרוּרוֹת בְּכַפֵּיהֶם. בְּלִי יוֹצֵאת וָפָרֶץ. וְלֹא יִפֹּל צְרוֹר אָרֶץ. וּבְבוֹאִי אֵלֶיהָ בְּנֶפֶשׁ נְבוֹכָה. עֶרֶב שַׁבָּת עִם חֲשֵׁכָה. אָמַרְתִּי: אוּלַי בֵּית הַתְּפִלָּה קְרוֹבָה. וְאֵלְכָה אֵלֶיהָ וְאֶדְרְשָׁה בָהּ. וְכַאֲשֶׁר בָּאתִי לְבֵית הַתְּפִלָּה. עִם הַקְּהִלָּה. רְאִיתִיהָ יָפָה כְּהֵיכָל וְאַרְמוֹן. וְיָשַׁבְתִּי בְּתוֹךְ הֶהָמוֹן. וְהָיוּ לְצִדִּי שְׁנֵי זְקֵנִים. אֲרֻכֵּי זְקָנִים. חֲסוֹנִים כָּאַלּוֹנִים. מִמּוֹכְרֵי הַתְּהִלּוֹת וְלֹא מִן הַקּוֹנִים. וּמֵהוֹרְסֵי בָתֵּי צֶדֶק וְלֹא מִן הַבּוֹנִים. בַּעֲלֵי קוֹמוֹת. וּבִטְנָם כַּעֲרֵמוֹת. לָהֶם נְפָשׁוֹת בְּזוּיוֹת וְעֵינֵיהֶם רָמוֹת. וְכַאֲשֶׁר שְׁאִלְתִּים עַל גְּדוֹלֵיהֶם. וְיֶתֶר קַהֲלֵיהֶם. וְהַחַזָּן אֲשֶׁר לָהֶם. אָמְרוּ: יֵשׁ לָנוּ רָאשֵׁי גָלֻיּוֹת. וּפְנִינֵי מַרְגָּלִיּוֹת. וּמִלְּבַד זֶה יֵשׁ לָנוּ חַזָּן דַּרְשָׁן. עָנָו וּבַיְשָׁן. דָּשֵׁן וְשָׁמֵן כְּמוֹ אֵילִים בְּנֵי בָּשָׁן. תְּפִלָּתוֹ עֲרֵבָה. וְחֶבְרָתוֹ אֲהוּבָה. וּנְגִינָתוֹ טוֹבָה. יוֹרֶה בַתּוֹרָה. וְיִקְרָא בַמִּקְרָא. וְיִסְבֹּר בַּנְּבִיאִים כַּמָּה סְבָרָא. וְיוֹדֵעַ פִּיּוּטִים יְקָרִים. וְשָׁר בַּשִּירִים.

וּכְשָׁמְעִי דִבְרֵיהֶם אָמַרְתִּי: תְּהִלָּה לָאֵל וּשְׁבָח. כִּי זִכַּנִי לִרְאוֹת זֶה הַחַזָּן הַמְשֻׁבָּח. וּבְעוֹדִי בְזֹאת הַמַּחֳשָׁבָה. וְהִנֵּה הַחַזָּן בָּא. וְעַל מִצְחוֹ טוֹטֶפֶת. וּבְרֹאשׁוֹ מִצְנֶפֶת. לְבָנָה וּנְעִימָה. בַּת מָאתַיִם אַמָּה. וְשַׂעֲרַת זְקָנוֹ עַד טַבּוּרוֹ הֶאֱרִיכָה. וַתְּכַסֵּהוּ בַשְּׂמִיכָה. וְהוּא יִסְחֹב עַל הָאָרֶץ גְּדִילֶיהָ. וְכִמְעַט יִכָּשֵׁל בְּשׁוּלֶיהָ. וּבִרְאוֹתֵנוּ אוֹתוֹ. חָרַדְנוּ כֻלָּנוּ לִקְרָאתוֹ. וְהִשְׁתַּחֲוִינוּ לְעֻמָּתוֹ. וְהֶחֱרַשְׁנוּ לְאֵימָתוֹ. עַד הֵחֵל בִּנְעִימוּתוֹ. וּפָתַח תְּפִלָּתוֹ. וּמָנִיתִי לוֹ בַּתְּפִלָּה יוֹתֵר מִמֵּאָה טָעֻיּוֹת. בְּרוּרוֹת וּגְלוּיוֹת. מִלְבַד הָאֲחֵרוֹת שֶׁאֵינָם לִזְכֹּר רְאוּיוֹת. אַךְ לֹא עֲנִיתִיהוּ. כִּי אָמַרְתִּי אוּלַי מִקְרֶה הוּא. אוֹ טִרְדַּת הַשַּׁבָּת הִבְהִילַתְהוּ. אוֹ שֵׁנָה אֲנָסַתְהוּ. וּכְמוֹ הַשַּׁחַר עָלָה. הִשְׁכַּמְתִּי לְבֵית הַתְּפִלָּה. וְהִנֵּה הַחַזָּן בָּא וְיָשַׁב בְּמוֹשַׁב הַגְּדֻלָּה. וּפָתַח בְּמֵאָה בְרָכוֹת. אֲשֶׁר עַל לְשׁוֹנוֹ סְדוּרוֹת וַעֲרוּכוֹת. וְאָמַר בְּקוֹל צְעָקָה רָמָּה. בָּרוּךְ אַתָּה יְיָ' אֲשֶׁר יָצַר אֶת הָאָדָם בְּהֵמָה. וּבִפְסוּקֵי זִמְרָה. טָעָה טָעֻיּוֹת רַבּוֹת וְאֵין לָהֶם סְפִירָה.

וּבִמְקוֹם גַּם מִזֵּדִים חֲשׂךְ עַבְדֶּךָ. אָמַר: גַּם מִזֵּיתִים חֲנֹךְ עַבְדֶּךָ. וּבִמְקוֹם חֵלֶב חִטִּים יַשְׂבִּיעֶךָּ. אָמַר: חֶרֶב חַדִּים יַשְׂבִּיעֶךָּ. וּבִמְקוֹם הַמְכַסֶּה שָׁמַיִם בֶּעָבִים. אָמַר: הַמְכַסֶּה שָׁמַיִם בְּגָדִים. וּבִמְקוֹם יִשְמַח יִשְׂרָאֵל בְּעוֹשָׂיו. אָמַר: יִשְׂמַח יִשְׁמָעֵאל בְּעֵשָׂו. וּבִמְקוֹם הַלְלוּהוּ בְּמִנִּים וְעוּגָב. אָמַר: הַלְלוּהוּ גְבִינִים וְעוּגָה. וּבִמְקוֹם וּבְיָדְךָ לְגַדֵּל וּלְחַזֵּק לַכֹּל. וּבְיָדְךָ לְגַדֵּף וּלְהַזִּיק לַכֹּל. עַד אֲשֶׁר נִכְלַמְתִּי. וְעַל בּוֹאִי לְבֵית הַכְּנֶסֶת נִחַמְתִּי. וְנֶאֱלַמְתִּי. וְיָדִי לְפִי שַׂמְתִּי. וְכַאֲשֶׁר כִּלָּה זְמִירוֹתָיו. עָמַד עַל עָמְדוֹ לַעֲרֹךְ תְּהִלּוֹתָיו. וְכִסָּה פָנָיו לֹא מֵעֲנָוָה. וְעָמַד בְּגַאֲוָה. וְהִרְבָּה תְנוּעוֹתָיו. וְהֵנִיעַ כִּתְפוֹתָיו. וְהֵרִים רַגְלוֹ הַיְמָנִית. וְהוֹרִיד הַשֵּׁנִית. וְחָזַר מְעַט לַאֲחֹרַנִּית. וּפָתַח לָהֶם גִּנְזֵי חָכְמָתוֹ. וְהֶרְאָה בֵית נְכֹתֹה. וְהִתְחִיל לוֹמַר פִּיּוּטִים וְשִׁירִים. כֻּלָּם שְׁבוּרִים. וּפִסְּחִים וְעִוְרִים. וְדַרְכָּם מְעֻקָּל. בְּלֹא חֶרֶז וּבְלֹא מִשְׁקָל. וּבְלֹא בִנְיָן. וּבְלֹא עִנְיָן. וְכַאֲשֶר הֶאֱרִיךְ פִּיּוּטָיו. לְשׁוֹטָיו. וְשִׁירֵי הֲתֻלָיו לִכְסִילָיו. וּמִזְמוֹרָיו לַחֲמוֹרָיו. ישֵ מִׁקְצָת הָעָם אֲשֶׁר יָשְבוּ. וּמֵהֶם יָשְׁנוּ שְׁנַת עוֹלָם וְשָׁכְבוּ. וּקְצָתָם בָּרְחוּ וְלֹא שָׁבוּ. וּבֵית הַכְּנֶסֶת עָזָבוּ. וְנָפוֹצוּ מֵעַל הָרוֹעֶה הַשְּׁוָרִים. וּבָרְחוּ הַמִּקְנֶה וְהָעֲדָרִים. וְלֹא נִשְׁאֲרוּ כִּי אִם אַרְבָּעָה חֲמוֹרִים. צוֹעֲקִם עִם הַחַזָּן וְנוֹעֲרִים. וְהֵם יַחְשְבוּ כִּי הֵם מְשׁוֹרְרִים. וְהוּא הֶאֱרִיךְ לָהֶם בְּפִיּוּטָיו עַד הַצָּהֳרַיִם. עַד דָּבְקָה לְשׁוֹנָם לַמַּלְקוֹחַיִם. וְכַאֲשֶׁר הִשְׁלִים פִּיּוּטָיו וְחָזַר לְהַשְׁלִים הַתְּפִלָּה. לֹא הָיָה אִישׁ בְּבֵית הַכְּנֶסֶת כּי נִמְצְאוּ בְּבָתֵיהֶם יְשֵׁנִים כָּל הַקְּהִלָּה. וְהָיָה בְתוֹכָם אִישׁ מִן הָאֲצִילִים הַחֲסִידִים. בַּעַל מַעֲשִׂים חֳמוּדִים. וְכַאֲשֶׁר שָׁמַע הַדָּבָר הַזֶּה וַיֵּרַע בְּעֵינָיו. וַיִּחַר לוֹ מְאֹד וַיִּפְּלוּ פָנָיו. וַיֹּאמַר: הַיִּיטַב בְּעֵינֵי יְיָ‘. לְחַבֵּר דְּבָרִים בְּטֵלִים לִתְפִלַּת יְיָ’. וּלְהַצְרִיךְ הַקָּהָל לָצֵאת בְּלֹא תְפִלָּה. הֲלֹא זֹאת רָעָה חוֹלָה. וַחֲטָאָה גְדוֹלָה. כִּי לֹא נִשְׁאַר לָנוּ בִּמְקוֹם הַקָּרְבָּן וְהָעוֹלָה. כִּי אִם תְּפִלָּה. וְתַחַת הַכְּבָשִׂים וְהַפָּרִים. תַּחֲנוּנֵי אֲמָרִים. כְּמָה שֶׁנֶּאֱמַר: וּנְשַׁלְּמָה פָרִים שְפָתֵינוּ. וַאֲנַחְנוּ בָהּ מִתְרַשְּׁלִים. וְאוֹתָה מְבַטְּלִים. בַּעֲבוּר שִׁירֵי הֲבָלִים. וְהָיָה עוֹמֵד סְבִיבוֹ. אִישׁ רִיבוֹ. וּכְשָׁמְעוֹ נִיבוֹ. אָמַר: לֹא כֵן הַדָּבָר כַּאֲשֶׁר דִּבַּרְתָּ. וּבְמַעֲנֵה פִיךָ שִׁקַּרְתָּ. כִּי עִקַּר הַתְּפִלָּה. הִיא הַתְּהִלָּה. וְכִי הַנִּגוּנִים. הֵם יוֹתֵר טוֹבִים מִן הַתַּחֲנוּנִים. כִּי בַזְּמַן שֶׁבֵּית הַמִּקְדָּשׁ קַיָּם הָיוּ הַלְוִיִּם בַּמִּקְדָּשׁ. מְשׁוֹרְרִים לָאֵל שִׁיר חָדָשׁ. וְאוֹתָם הַשִּׁירִים. הָיוּ בְעֵינֵי הָאֵל יְקָרִים. כִּיקַר כָּרִים וְאֶשְׁכָּרִים. וְדַם אַבִּירִים עִם חֵלֶב פָּרִים. וְעַתָּה בְּגָלוּתֵנוּ הֵן הַשִּׁירוֹת וְהַקְּרוֹבוֹת עִקַּר תְּפִלָּתֵנוּ. וְעַמּוּד לַעֲבוֹדָתֵנוּ וְדָתֵנוּ. וְעוֹד כִּי מִנְהַג כָּל הַגָּלֻיּוֹת. אֲשֶׁר בְּכָל הַמַּלְכֻיּוֹת. לוֹמַר בְּבֵית הַכְּנֶסֶת פִּיּוּטִים וּקְרוֹבָה. וּמְלָאכָה רַבָּה. וְדִבְרֵי סְלִיחוֹת. וְשִׁירֵי שְׁבָחוֹת. וַאֲנַחְנוּ אֵיךְ נְשַׁנֶּה מִנְהָגָם. וְנָסוּר מִמַּהֲלָכָם. אֵין זֶה מַעֲשֵׂה חָכָם. אָמַר לוֹ הַטּוֹעֵן: אֱמֶת דִּבַּרְתָּ כִּי בַזְּמַן שֶׁבֵּית הַמִּקְדָּשׁ קַיָּם הָיוּ הַזְּמִירוֹת וְהַתּוֹדָה. עִקַּר הָעֲבוֹדָה. בְּעוֹד שֶׁהָיְתָה הַשְּׁכִינָה. שָׁמָּה שְׁכוּנָה. וְהָיוּ הַמְשׁוֹרְרִים. צַדִּיקִים גְּמוּרִים. וְהָיוּ הַלְוִיִּם מְנַגְּנִים בְּשִׁירֵי דָוִד לֹא בְזוּלָתָם. וְהָיוּ כָל הַשּׁוֹמְעִים מְבִיניִם אוֹתָם. אֲבָל עַתָּה אֵין הֵיכָל וְאֵין מִקְדָּשׁ. וְלֹא תְהִלָה וְלֹא שִׁיר חָדָשׁ. וְלֹא לְוִיִּם וְלֹא כֹהֲנִים וְלֹא מְשׁוֹרְרִים. וְלֹא מְנַגְּנִים וְלֹא צַדִּיקִים וְלֹא יְשָׁרִים. כִּי הַמְנַגְּנִים. נִרְגָּנִים.

וְהַמְשׁוֹרְרִים. סָרֵי סוֹרְרִים. וְהַחַזָּנִים לֹא יָבִינוּ מַה יְּדַבֵּרוּ. וְהָעָם לֹא יֵדְעוּ מַה שֶׁהַחַזָּנִים יֹאמֵרוּ. וּבִמְקוֹם שִׁירֵי דָוִד הַקְּדוֹשִׁים. יְשׁוֹרְרוּ לָהֶם שִׁירֵי הַקְּדֵשִׁים. וְיַמְשִׁילוּ דִבְרֵי הֲתֻלִים לְדִבְרֵי אֱלֹהִים חַיִּים. יְדַמּוּ עִדִּים. לַעֲדָיִים. וְהַמֵּתִים אֶל הַחַיִּים. וּבְעוֹד יֹאמַר הַחַזָּן קְרוֹבוֹתָיו. לֹא יַכִּירוּ הָעָם בֵּין שִׁירוֹתָיו לְקִינוֹתָיו. וְלֹא יֵדְעוּ אִם הוּא מְשַׁבֵּחַ. אוֹ כֶלֶב נוֹבֵחַ. וְאִם הוּא מְשׁוֹרֵר. אוֹ חֲמוֹר נוֹעֵר. וְאֵיךְ יִישַׁר בְּעֵינֵי הָאֱלֹהִים כַּדָּבָר הַזֶּה כִּי אִישׁ מֵעַם הָאָרֶץ אֲשֶׁר אֵינֶנּו יוֹדֵעַ בֵּין יְמִינוֹ לִשְׂמֹאלוֹ. וְהוּא תוֹעֶה בְּמַחְשְׁכֵּי שִׂכְלוֹ. וְלֹא יָבוֹא לְבֵית הַכְּנֶסֶת רַק שָׁעָה אַחַת בַּשַּׁבָּת. כְּשֶׁיִּהְיֶה מִמְּלַאכְתּוֹ נִשְׁבָּת. וְיֵצֵא בַהֶבֶל כַּאֲשֶׁר בָּא. כִּי הוּא לֹא יֵדַע לְהִתְפַּלֵּל וְהַחַזָּן לֹא הוֹצִיאָהוּ יְדֵי חוֹבָה.


אָמַר לוֹ בַּעַל רִיבוֹ: הֲתַחְשֹׁב כִּי בַעֲבוּר טַעֲנוֹתֶיךָ הַחֲלוּשׁוֹת וְהַדַּלוֹת. נַעֲזֹב אֲנַחְנוּ מִנְהַג כָּל הַקְּהִלּוֹת. כְּגוֹן קְהִלּוֹת סְפָרַד וּקְהִלּוֹת מַעֲרָב וּקְהִלַּת יָוָן אֲשֶׁר אֵין אֶרֶץ מֵאַרְצוֹתָם. וְלֹא קְהִלָּה מִקְּהִלּוֹתָם. שֶׁלֹּא יֹאמְרוּ בָם קְרוֹבוֹת. וְשִׁירוֹת רַבּוֹת. וְיֵשׁ בִּקְהִלּוֹתָם סְכָלִים רַבִּים וּפְתָאיִם. דּוֹמִים לִבְהֵמוֹת וְלִפְרָאִים. אֲבָל בַּעֲבוּרָם לֹא יְשַׁנּוּ הָעָם מִנְהָגָם. וְאַתָּה רוֹצֶה לְחַדֵּשׁ דָּבָר בָּעוֹלָם. שֶׁלֹּא הָיָה מֵעוֹלָם. אָמַר לוֹ הַטּוֹעֵן: אֵין עִיר מִכֶּל הֶעָרִים אֲשֶׂר זָכַרְתָּ. וּקְהִלָּה מִן הַקְהִלּוֹת אֲשֶׁר סָפַרְתָּ. שֶׁיִהְיוּ כֻּלָּם סְכָלִים וְעַמֵי אָרֶץ. כְּמוֹ אַנְשֵׁי זֹאת הָאָרֶץ. כִּי בְכָל קְהִלָּה מִן הְקְּהִלּוֹת. תִּמְצָא קְצָתָם מַשְׂכִּיליִם בַּתְּהִלּוֹת. וּמְבִינִים הַתְּפִלּוֹת. וּבְבוֹאָם אֶל בֵּית הַתְּפִלָּה. אִם יאֹמַר הַחַזָּן שִׁירֵי תְהִלָּה. יָבִינוּ דְבָרָיו מִקְצָת הַקְּהִלַּה. וְיַעֲנוּ אוֹתוֹ עַל כָּל מִלָּה וּמִלָּה. אֲבָל בָּעִיר הַזֹּאת כֻּלָּם שְׁוָרִים. וּמִדַּעַת נִבְעָרִים. גַּם הַחַזָּן לֹא יָבִין מַה שֶּׁהוּא אוֹמֵר. וְלֹא בַמֶּה הוּא מַתְחִיל הוּא גוֹמֵר. וְכֵן הֶהָמוֹן אֲשֶׁר סְבִיבָיו וְאַחֲרָיו.

לֹא יָבִין דְּבָרָיו. אִם [טָעָה] אֵין יוֹדֵעַ. וְאִם יִמְצָא אֵין שׁוֹמֵעַ. וְאִם יִסְכַּל אֵין מִי יְרִיבֵהוּ. וְאִם יֶחֱטָא אֵין מִי יְשִׁיבֵהוּ. וְהוּא מְנַגֵּן בְּשִׁירָיו עַל בְּהֵמוֹת מְלֻבָּשִׁים בִּבְגָדִים לְבָנִים. וִידַבֵּר עַל הָעֵצִים וְעַל הָאֲבָנִים. וּלְמִי יְנַגֵּן הַחַזָּן בְּרֶגֶל מְעַכֶּסֶת. וְיָד מְפַרְכֶּסֶת. רַק לְקִירוֹת בֵּית הַכְּנֶסֶת. כְּמוֹ שֶׁנֶּאֱמַר: אֶת מִי יוֹרֶה דֵעָה. וְאֶת מִי יָבִין שְׁמוּעָה. וְאִלּוּ נִמְצָא בְכָל הַקְּהִלּוֹת קְהִלָּה כָזֹאת. בָּטְלוּ מֵהֶם הַקְּרוֹבוֹת וְהַשִּׁירוֹת. וְהַפִּיּוּטִים וְהַזְּמִירוֹת. וְכָל הַמִּתְפַּלֵּל בְּדִבְרֵי פִּיוּטִים לְזֹאת הַקְּהִלָּה. תְּפִלָתוֹ תִּפְלָה. וּתְהִלָּתוֹ תָּהֳלָה. וְטוֹב לָעָם לָשֶׁבֶת בּיוֹם הַשַּׁבָּת בְּבֵיתָם. מִלָּבוֹא לְבֵית הַכְּנֶסֶת לְהוֹסִיף חֵטְא עַל חַטָּאתָם. כִּי כָל אִישׁ מֵהֶם בַּאֲפֵלָה מְמַשֵּׁשׁ וּמְהַלֵּךְ. כִּי בַהֶבֶל בָּא וּבַחשֶׁך יֵלֵךְ. וְעַל כֵּן אֲנִי אוֹמֵר כִּי הַקְּרוֹבוֹת וְהַשִּׁירוֹת. אֵינָם לְכָל הַקְּהִלּוֹת אֲסוּרוֹת רַק מֻתָּרוֹת. אֲבָל הֵן אֲסוּרוֹת לַקָּהָל הַזֶּה. כִּי כֻלָּם כִּבְהֵמוֹת וְאֵין מַקְשִׁיב בָּהֶם וְאֵין פּוֹנֶה. וְהִנֵּה הַמָּקוֹם מְקוֹם מִקְנֶה. וְכִשְמֹעַ אִישׁ רִיבוֹ. דְּבָרוֹ וְנִיבוֹ. הִכִּיר כִּי דְבָרָיו נְכוֹנִים. וְטַעֲנוֹתָיו חֲזָקִים וְנֶאֱמָנִים. וַיָּשֶׂם יָדוֹ עַל פִּיו הָאִישׁ. וַיְהִי כְמַחֲרִישׁ.


וְהָיָה בְעִיר אַרְבֵּל גֶּבֶר. מִבְּנֵי עֵבֶר. וּשְׁמוֹ בִנְיָמִין הֶחָבֵר. וְהוּא חוֹבֵר חָבֶר. וְהֵשִׁיב עַל דִּבְרֵי זֶה הַטּוֹעֵן. וְהָאֵל עֲנָשָׁהוּ בְּזֶה הֶעָוֹן. וַיִּצָּמֵת. כִּי בְחֶטְאוֹ מֵת. וְהַרְבֵּה טְעָנוֹת בַּאִגֶּרֶת אֲשֶׁר לוֹ. כָּל הַשּׁוֹמֵעַ יִצְחַק לוֹ. וְהָיָה פְתַיּוּתוֹ. נִסְתָּר בִּשְׁתִיקָתוֹ. עַד זָרַק הַקִּיתוֹן. וַיִּפְתַּח ה' אֶת פִּי הָאָתוֹן. כִּי הֵבִיא טְעָנוֹת בּזוּיוֹת. כָּלָּם בְּאֶפֶס יָד דְּחוּיוֹת. וְעַל לֹא יְסוֹד בְּנוּיוֹת. וְכָל דְּבָרָיו תַּעְתּוּעִים. וְטַעֲנוֹת הַמִּשְׁתַּגְּעִים. וְאָמַר בִּדְבָרָיו כִּי יֵשׁ בְּפָרָשַׁת וַיְחַל משֶׁה רֶמֶז לִסְלִיחוֹת מִפְּנֵי שֶׁיֹּאמְרוּ וַיַּעֲבֹר ה' עַל פָּנָיו בְּכָל סְלִיחָה וּסְלִיחָה. וְאָמַר כִּי הַתְּפִלָּה אֵינָהּ עָלֵינוּ חוֹבָה מְחֻיֶּבֶת. מִפְּנֵי שֶׁאֵינָהּ בַתּוֹרָה נִכְתֶּבֶת. וּמַה שֶּׁנֶּאֱמַר וַעֲבַדְתֶּם אֶת ה' אֱלֹהֵיכֶם וְאָמרוּ חֲכָמִים עֲבוֹדָה זוֹ תְּפִלָּה. אָמַר זֶה הֶחָבֵר כִּי הִיא סְבָרָה קַלָּה. וּרְאָיָה דַלָּה. וְכָל הָאוֹמֵר כִּי יֵשׁ לִסְלִיחוֹת רְאָיָה בַתּוֹרָה. וְאֵין לִתְפִלָּה רְאָיָה מִן הַתּוֹרָה. אֵין לַחֲטָאָיו כַּפָּרה. כִּי דִבֶּר סָרָה. עַל הָאֵל הַגָּדוֹל הַגִּבּוֹר וְהַנּוֹרָא.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

אֲשֶׁר יֹאמַר רְשׁוּת הִיא הַתְּפִלָּה

וְחוֹבָה הַקְּרוֹבָה לַקְּהִלָּה.

אֱמֶת הֻכָּה בְהוֹלֵלוּת וְסִכְלוּת

וְלֹא נִמְצֵאת לְמַכָּתוֹ תְעָלָה.

לְמַעַן כֹּי אֱמוּנָתוֹ שְׁבוּרָה

וְאֶת שִׁבְרָהּ יְרַפֵּא עַל נְקַלָּה.

אֱמֶת כִּי הַשְּׁתִיקָה הִיא יְקָרָה

בְּכַל מִדּוֹת חֲמוּדוֹת הִיא שְׁקוּלָה.

וְאִלּוּ הַכְּסִיל שָׁתַק יְכַסֶּה

אֱוִילוּתוֹ אֲשֶׁר חָשַׂף וְגִלָּה.

בְּלִבּוֹ הָיְתָה סִכְלוּת עֲצוּרָה

וּפָתַח פִּיו וְקָרָא לָהּ גְּאֻלָּה.

וְכָל מַרְבֶּה דְבָרִים הֵם לְרַע לוֹ

וְרֹב שֶׁמֶן יְכַבֶּה הַפְּתִילָה.

וְרָאִיתִי כְסִיל חָכָם בְּעֵינָיו

וְלֹא יָבִין וְלִבָּתוֹ אֲמֻלָּה.

וְאֵיךְ יוּכַל רְאוֹת מֻכֶּה בְסַנְוֵר

וְעֵין לִבּוֹ בְּפוּךְ סִכְלוּת כְּחוּלָה.

וְחֹלֶשׁ טַעֲנָתוֹ הוּא לְאוֹת

הֲכִי טַעְנַת מְרִיבוֹ הִיא מְעֻלָּה.

וְהֵעִיד כִּי חֲכָמִים אָמְרוּ אֵין

תְּפִלָּה חוֹב וְעֵדוּתוֹ פְּסוּלָה.

וְלִגְזֹל הָאֱמֶת יַחְשֹׁב בְּמִלָּיו

וְלֹא יָדַע הֲכִי נַפְשׁוֹ גְזוּלָה.

וְיִיגַע רִיק לְהַצְדִּיק דִּין מְרִיבָיו

וְיֵשׁ נֶפֶשׁ לְזוּלָתָה עֲמֵלָה.

וְכָל אוֹמֵר קְרוֹבָה לַסְּכָלִים

כְּאִישׁ דוֹבֵר עֲלֵי אֹזֶן עֲרֵלָה.

וְכָאוֹמֵר לְעֵץ הָקֵץ וְעוּרָה

וְכַדּוֹפֵק עֲלֵי דֶלֶת נְעוּלָה.

וְכָעוֹזֵב נְתִיב שָׁלוֹם וְיֵלֵךְ

סְבִיב שׁוּחַה וְיִקְרֶב לַנְּפִילָה.

וְהַחַזָּן אֲשֶׁר יֹאמַר סְלִיחוֹת

לְעַם זֶה אֵין לְחַטָּאתוֹ מְחִילָה.

וּבִקְהִלָּה אֲשֶׁר כֻּלָּם פְּתָאִים

לְמִי יֹאמַר אֱנוֹשׁ שִׁירֵי תְהִלָּה.

וְאֵין שָׂכָר לְשׁוֹמֵעַ קְרוֹבָה

וְדִבְרֵי שִׁיר לְבַד אִבּוּד תְּפִלָּה.

הֲכִי עֵת יִשְׁמְעוּ הָעָם דְּבָרָיו:

יְנוּסוּן חִישׁ וְהִתְפָּרְדָה חֲבִילָה.

וְכָל אוֹמֵר הֲלֹא זֶה חוֹב וּמִצְוָה

בְּמוֹתוֹ יִמְצְאָה מֵאֵל גְּמוּלָה.

וּמִלָּיו יִהְיוּ רֶשֶׁת לְרַגְלָיו

וְיֵעָנֵשׁ בְּיוֹם רִדְתּוֹ שְׁאוֹלָה.

לְמַעַן כִּי עֲזִיבַת הַתְּפִלָּה

לְדֶרֶךְ חֵטְא וּמִינוּת הִיא מְסִלָּה.

אֱלֹהִים חַלְּצֵנוּ מֵעֲצַת כֹּל

אֲשֶׁר הִשְׁחִית וְגַם הִתְעִיב עֲלִילָה.

וְעָזְרֵנוּ לְהַשִׂיג אֶת רְצוֹנָךְ

בְּעוֹדֵנוּ וְאַל נָשׁוּב לְכִסְלָה.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
