ספר תחכמוני -שַׁעַר רְבִיעִי

בְּשֶׁבַח שְׁנֵי מְשׁוֹרְרִים, מִי יָשִׁיר שִׁירִים יוֹתֵר יְקָרִים


(שִׁיר הַנְּמָלָה וְשִׁיר הַפַּרְעשׁ)


נְאֻם הֵימָן הָאֶזְרָחִי: רָאֲתָה עֵינִי בִּימֵי נַעֲרוּתִי. וּזְמַן יַלְדוּתִי. שְׁנֵי אֲנָשִׁים עִבְרִים נִצִּים. כְּגִבּוֹרִים חֲלוּצִים. בְּפִיהֶם עִנְיָנִים נִמְרָצִים. וּבִלְשׁוֹנָם חֲנִית וְחִצִּים. וּלְרִיבָם בְּנֵי אִישׁ נִקְבָּצִים. וּמִכָּל פִּנָּה רָצִים. וְאִישׁ בְּאָחִיו נִלְחָצִים. הָאֶחָד מֵהֶם זָקֵן סָר מִמֶּנּוּ כֹּחוֹ. וְהַשֵּׁנִי בָּחוּר לֹא כָהֲתָה עֵינוֹ וְלֹא נָס לֵחֹה. וּשְׁנֵיהֶם כִּשְׁנֵי אֲרָיוֹת נִלְחָמִים בְּחֹזֶק יָדַיִם. אוֹ כִשְׁנֵי נְשָׁרִים יִתְמַרְמְרוּ זֶה אֶל זֶה בַּחֲצִי הַשָּׁמַיִם. זֶה יֹאמַר לַה' אָנִי. וּמִבֶּטֶן קְרָאָנִי. וּלְחַבֵּר הַמְּלִיצוֹת בְּרָאָנִי. וְזֶה יִתְגָּאֶה וְעַל חוּג הַשָּׁמַיִם יַהֲלֹךְ. וִּמִתְנַשֵּׂא לֵאמָר אֲנִי אֶמְלֹךְ. וְכַאֲשֶׂר דָּלְקוּ מְדוּרוֹת הַמְּרִיבוֹת בֵּינֵיהֶם. וְעָלוּ עַד לֵב הַשָּׁמַיִם הַלַּפִּידִים וְהַקּוֹלוֹת בֵּין שְׁנֵיהֶם. נִגְּשׁוּ אֶל הַשּׁוֹפֵט וּפְנֵי לְהָבִים פְּנֵיהֶם. וְכִידוֹדֵי אֵשׁ יִתְמַלְּטוּ מֵעֵינֵיהֶם. וְעַד הָאֱלֹהִים יָבֹא דְּבַר שְׁנֵיהֶם.

וּבְעָמְדָם לְפָנָיו אָמרוּ לוֹ: אִם טוֹב בְּעֵינֵי אֲדוֹנֵנוּ, יַחְקֹר לְמִי הָאֱמֶת בֵּינֵינוּ. וְיָשֵׁת יָדוֹ עַל שְׁנֵינוּ. אָמַר הַשּׁוֹפֵט: הַגִּידוּ עִנְיַן מְרִיבוֹתֵיכֶם. וְהַגִּישׁוּ עַצֻּמוֹתֵיכֶם. וְהַצַּדִיק יַעַזְרֵהוּ צִדְקוֹ. וְהָרָשָׁע יָשׁוּב זְדוֹנוֹ לְחֵיקוֹ. אָמַר הַגָּדוֹל מֵהֶם: בִּי אֲדוֹנִי הַטֵּה אָזְנְךָ לְנִיבִי. וְאֶצְעַק אֵלֶיךָ חֲמַס אִישׁ רִיבִי. הָרוֹצֶה לָעוּף הַשָּׁמַיִם. בְּלִי כְנָפַיִם. הַדּוֹרֵשׁ לָבוֹא בְּמִלְחַמְתִּי וְהוּא יָגֵעַ וּרְפֵה יָדַיִם. אָמַר לוֹ בַּעַל רִיבוֹ: הִנֵּה עֲנָקַתְךָ גַאֲוָה עַל כֻּלָּנוּ. וּמִמָּרוֹם תְּדַבֵּר לָנוּ. וְעַתָּה הוֹדִיעֵנוּ כֹחֲךָ זֶה. וְנֵדַע הֲשָׁמֵן הוּא אִם רָזֶה. וְהַרְאֵנוּ אִם גְּדוּדֶיךָ חַלָּשִׁים אוֹ גִבּוֹרִים. וּמִשְׁכְּנוֹתֶיךָ הַבְּמַחֲנִים אִם בְּמִבְצָרִים. וּבְעֵת יִגָּלוּ סְתָרֶיךָ. אָז יִבָּחֲנוּ דְּבָרֶיךָ.

וְאָז יִבָּקַע כַּשַּׁחַר אוֹרֶךָ. אָמַר חֲבֵרוֹ: וְאֵיךְ תַּעֲרֹך אֵלַי וּבִלְשׁוֹנִי מְקוֹר הַשֵּׂכֶל וְהַמּוֹעֵצוֹת. וּבְלִבִּי מַעְיַן הַמְּלִיצוֹת. כִּי יֶשׁ לִי יְכֹלֶת בְּתֹקֶף הֶגְיוֹנִי. וּבִמְלִיצוֹת לְשׁוֹנִי. לְרוֹמֵם כָּל דָּבָר נִבְזֶה וְנִקְלֶה. וּלְהַשְׁפִּיל כָּל דָבָר נַעֲלֶה. וְאַתָּה אִלּוּ הָיוּ הַסְּעָרוֹת מַחֲנוֹתֶיךָ. וְהָרוּחוֹת מַרְכְּבוֹתֶיךָ. וְהָעֲנָנִים צִבְאוֹתֶיךָ. וְהַבְּרָקִים נוֹשְׂאִים כְּלֵי מִלְחֳמוֹתֶיךָ. וְהַכּוֹכָבִים לְאוֹר חִצֶּיךָ. יֵלְכוּ לְנֹגַהּ בְּרַק חֲנִיתֶיךָ. יִמַּס לִבָּם מֵהִתְיַצֵּב לְפָנַי. וְיִרְפּוּ יָדֶיךָ לַעֲרֹךְ לְהֶגְיוֹנַי. עַד תּוּכַל לְשַׁבֵּחַ נְמָלָה. אַף עַל פִּי שֶׁהִיא בְּזוּיָה וְדַלָּה. וּתְשִׂימֶיהָ לְשֵׁם וְלִתְהִלָּה. אָמַר לוֹ בַעַל רִיבוֹ: וּמֶה עָזַרְתָּ לְלֹא כֹחַ. הֲתַפְחִיד אַרְזֵי לְבָנוֹן בְּקִמּוֹשׁ וָחוֹחַ? כַּמָּה נִקְלָה שְׁאֵלָתֶךָ. וְקַלָּה חִידָתֶךָ. וְעַתָּה שְׁמַע בַּקָּשָׁתֶךָ.

וַיַּעַן וַיֹּאמֶר עַל הַנְּמָלָה: שִׁפְחָה חָרוּצָה. מֵעָפָר קְרוּצָה. כְּגִבּוֹר חֲלוּצָה. וּלְבַקֵּשׁ מְזוֹנָהּ נְחוּצָה. הִיא דַלַּת חֲלָצַיִם. קַלַּת רַגְלַיִם. יוֹשֶבֶת בְּפֶתַח עֵינַיִם. שְׁחוֹרָה וְנָאוָה. וְאֵינֶנָּה מִבְּנוֹת חַוָּה. לוֹבֶשֶׁת שְׁחוֹרִים. וְיוֹצֵאת מִן הַחוֹרִים. וּבְכָל יוֹם תָּעִיר שְׁחָרִים. וְתַשְׁכִּים בְּפִתְחֵי שְׁעָרִים. לְסוֹבֵב נְתִיבוֹת. בִּגְבָעוֹת וּבַעֳרָבוֹת. פַּעַם בַּחוּץ פַּעַם בָּרְחוֹבוֹת. וּבְכָל יְמֵי הַסְּתָו תִתְעַלֵּם אִם יִדְרְשׁוּנְהָ. וְלָשׁוּט אַחֲרֵי מַאֲכָלֶיהָ. בְּבֵיתָהּ לֹא יִשְׁכְּנוּ רַגְלֶיהָ. בְּכָל יוֹם תַּעֲלֶה גְבָעוֹת. וְתֵרֵד בְּקָעוֹת. וְתִיגַע. וְלֹא תֵרָגַע. וְתָרוּץ. וְלֹא תָקוּץ. וְתִלְקֹט וְלֹא תִשְׁקֹט. לְבַקֵּשׁ מִחְיָתָהּ. וְלִמְצֹא טֶרֶף לְבֵיתָהּ. הִיא דַּלָּה מִן הַדַּלּוֹת. וְנֹוׂשֵׂאת מַשָּׁאוֹת גְּדוֹלוֹת. וּמִן הַשָּׂדוֹת תִּקְטֹף מְלִילוֹת. וְתָכִין תַּחַת הָאָרֶץ מְעָרוֹת וּמְחִלוֹת. לְמַלֹּאת אוֹתָם תְּנוּבוֹת. וְשִׁבֳּלִים מְלֵאוֹת וְטוֹבוֹת. וּבְכָל יוֹם בֵּית נְתִיבוֹת נִצָּבָה. בָּעֶרֶב הִיא בָאָה וּבַבֹּקֶר הִיא שָׁבָה. וְכָל יְמֵי הַקֹּר תִּסְתַּתֵּר בְּבָתֵּי כְלָאִים. וְתֵרֵד פְּלָאִים. וּבִימֵי הַקַּיִץ תֵּצֵא עִם הַיּוֹצְאִים. לִשְׁבֹּר בְּתוֹךְ הַבָּאִים. וּבְהַגִּיעַ זְמַן הַקָּצִיר. וּבַא צִיר הַבָּצִיר. חָגְרָה בְּעֹז מָתְנֵיהָ. וְתִסַּע מִמְּעוֹנֵיהָ. וַתָשֵׂם אֶל הַגֹּרֶן מְגַמְתַ פַּנֶיהָ. וְתֵצֵא עִם הַפּוֹעֲלִים. לַחְבֹּט מִן הַשִּׁבֳּלִים. וְתֵלֵךְ מִצַּד הַקּוֹצְרִים. לֶאֱסֹף מִן הָעֳמָרִים. וּבְתוֹךְ הַגְּרָנוֹת תְּשׁוֹטֵט עַד יְרַחֲמוּהָ. וּבֵין הָעֳמָרִים תְּלַקֵּט וְלֹא יַכְלִימוּהָ. וּמִבִּרְכָתָם יַעֲנִיקוּן לָהּ. וִּמן הַצְּבָתִים שׁוֹל יָשֹׁלּוּ לָהּ. וַעֲזָבוּהָ וְלִקְּטָה וְלֹא יִגְעֲרוּ בָהּ. וְיֹּאמַר אִישֻ אֶל אָחִיו הַרְפֵּה לָהּ. כִּי נַפְשָׁהּ מָרָה לָהּ. וְכִי בֵיתָהּ רֵיקָם וְזֶרַע אֵין לָהּ. וּבְכָל יוֹם תְּשׁוֹטֵט כְּשְׁפִחָה. אַחַר לֶקֶט וְשֵׁכְחָה. וּבְשִׂכְלָהּ וְטוּב טַעְמָהּ. מִמֶּרְחָק תָּבִיא לַחְמָהּ. וְתָכִין שֶׁבֶר וּמָזוֹן. לְעֵת בַּצֹּרֶת וְרָזוֹן. וְכָל שְׂעוֹרָה אֲשֶׁר תִּמְצָא כְּבָבַת עֵינָהּ תִנְצְרֶנָּה. וְתַחַת הָאָרֶץ תִּשְׁמְרֶנָּה. וּמִן הַזֶּרֶם וְהַגֶּשֶׁם תַּסְתִּירֶנָּה. וְהַחִטָּה וְהַכֻּסֶּמֶת לֹא נֻכּוּ כִּי אֲפִילוֹת הֵנָּה. וּבִרְאוֹתָהּ כִּי בָאוּ יְמֵי הַקָּרָה. וְהַקֶּרַח הַנּוֹרָא. וּמִן הַגֶּשֶׁם כָּל אִישׁ נֶחְבָּא. וְאֵין יוֹצֵא וְאֵין בָּא. אָז תִּפְתַּח אוֹצְרוֹתֶיהָ וּבֵית נְכֹתָהּ. וְתִתֵּן טֶרֶף לְבֵיתָהּ. מִפְּרִי יָדֶיהָ. וּמִיגִיעַת צְעָדֶיהָ. וּבִפְעָלֶיהָ תָּעִיד עַל חָכְמַת עוֹשֶׂיהָ. וַיְּהַלְלוּ בַשְּׁעָרִים מַעֲשֶׂיהָ. וְאַשְׁרֵי מַשְׂכִּיל אֲשֶׂר יִוָּסֵר בְּמִפְעָלֶיהָ. וְאוֹי לֶעָצֵל אֲשֶׁר לֹא יִלְמַד מִמַּעֲלָלֶיהָ. כִּי יִשָׁאֵר נָעוּר וָרֵיק. כְּגֶפֶן בּוֹקֵק. שְׂפַת הַיָּגוֹן מְנַשֵּׁק. וְחֵיק הַחֶסְרוֹן מְחַבֵּק.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וּבַת חַיִל שְׁחַרְחֹרֶת

בְּתוֹךְ בָּנוֹת מְאֻשֶּׁרֶת.

חֲגוּרַת עֹז בְּמָתְנֶיהָ

בְּלִי אֵזוֹר וּמַחְגֹּרֶת.

שְׁחוֹרָה הִיא כְּמוֹ הַמּוֹר

וּבְמוֹר לֹא מְקֻטֶּרֶת

בְּכָל בֹּקֶר מְבַקֶּרֶת

וְכָל שַׁחַר מְשַׁחֶרֶת.

וְתַשְׁכִּים לֶאֱסֹף טֶרֶף

וְהַלֵּילוֹת מְקַצֶּרֶת.

וְעִם הַפּוֹעֲלִים תַּעֲמֹד

שְׁחָרִים עַל מְרוֹם קֶרֶת.

וְתִצְבֹּר בָּר וְרֹב מָזוֹן

לְעֵת רָזוֹן וּבַצֹּרֶת.

וְכָל חִטָּה אֲשֶׁר תִּצְבֹּר

בְּעָפָר הִיא מְקַבֶּרֶת.

וְתוֹךְ בֶּטֶן שְׁאוֹל תַּחְצֹב

מְעָרוֹת לָהּ וְחוֹפֶרֶת.

וּבֵין חַדְרֵי שְׁאוֹל תִּצְפֹּן

אֲשֶׁר אוֹסְפָה וְצוֹבֶרֶת.

וְאֵינֶנָּה מְגַנֶּבֶת

וְהִיא נִמְצֵאת בְּמַחְתֶּרֶת.

וּבַקַּיִץ גְּלוּיָה הִיא

וְהִיא בַקֹּר מְסֻתֶּרֶת.

וְתִצְפֹּן לַסְּתָו לַחְמָהּ

וְתַנִּיחוֹ לְמִשְׁמֶרֶת.

וְלָהּ חִטָּה מְזֻמֶּנֶת

וּמַצָּתָהּ מְשֻׁמֶּרֶת.

וְעֵת יַחְלֹף זְמַן קָצִיר

בְּחַיִּים הִיא מְקֻבֶּרֶת.

וְתִנְעֹל שַׁעֲרֵי עִירָהּ

וְכִירִיחוֹ מְסֻגֶּרֶת.

וְעֵת יָבוֹא זְמַן קָצִיר

דְּלָתֶיהָ מְשַׁבֶּרֶת.

וְתִגְלֶה מִמְּעוֹנָהּ רַק

לְבֵית לֶחֶם מְמַהֶרֶת.

לְדַעְתָּהּ כִּי בְּגָלוּתָהּ

עֲוֹנֶיהָ מְכַפֶּרֶת.

שְׁחָרִים יִישְׁנוּ אִישִׁים

וְהִיא תֵעוֹר בְּאַשְׁמֹרֶת.

בְּכֶל עֵבֶר וְכָל דֶּרֶךְ

מְשׁוֹטֶטֶת כְּסוֹחֶרֶת.

וְהִיא לֹא לָמְדָה מוּסָר

וְלִבְנֵי אִישׁ מְיַסֶּרֶת.

וְהֶעָצֵל בְּמִטָּתוֹ

וְהִיא שַׁחַר מְעוֹרֶרֶת.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר כִּלָּה שִׁירוֹ. אָמַר לַחֲבֵרוֹ. אֵלֶּה הֵם הַמְּלִיצוֹת אֲשֶׁר יִקְרעו כָּל לֵב לִשְׁנַיִם. וְיִמְשְׁכוּ הַלְּבָבוֹת בְּלִי יָדַיִם. וְיַחְפִּירוּ פְּנֵי חַמָּה בְיוֹם תִּהְיֶה לְשִׁבְעָתָיִם. וְנֶגְדָּם אִישׁ כָּמוֹךָ יַחֲרִישׁ וְאַל יְמַלֵּל. כִּי אִם בְּזֹאת יִתְהַלֵּל הַמִתְהַלֵל. הֲתַעֲרֹךְ אֵלַי וְזַעְמִי עָלֶיךָ נִתָּךְ. וְאֵי זֶה עָבַר רוּחַ ה' מֵאִתִּי לְדַבֵּר אִתָּךְ. הֲלִפְנִינִים תַּעֲרֹךְ נִבְלֵי חֶרֶשׂ. וְלַחֲשֵׁכִים תַּמְשִׁיל אוֹר חֶרֶס. וּמָתַי צָלְחָה רוּחַ אֱלֹהִים עָלֶיךָ. וְהַחָכְמָה מֵאַיִן תָּבוֹא לְהַשְׂכִּילְךָ? הֲרִימוֹתִי יָדִי אֶל ה' אִלּוּ קָרַעְתָּ שְׁחָרִים. וְרָקַעְתָּ הֶהָרִים. וּבָקַעְתָּ נְהָרִים. וְהָיוּ צְבָא הַשָּׁמַיִם גְּדוּדֶיךָ. וְעַיִשׁ עִם בָּנֶיהָ עֲבָדֶיךָ. וְתָרִים עַל רֹאשׁ תְּלִי דְּגָלֶיךָ וְנִסֶּיךָ. וְנִשְׂרְפָה הָאֵשׁ מֵאֵש הֲמָסֶיךָ. יֶחֱרַד לִבְּךָ מֵהִתְיַצֵּב נֶגְדִּי. וְתֶחֱלַשׁ רוּחֲךָ לְקוֹל פַּחְדִּי. וְלָעוּ דְבָרֶיךָ. וְחַתּוּ גִּבּוֹרֶיךָ.

וְעָיְפוּ דַּהֲרוֹת אַבִּירֶיךָ. וְכָשְׁלוּ סוּסֵי אֲמָרֶיךָ. לְהַשִּׂיג אֲבַק רִכְבִּי. וְלַעֲלוֹת לִמְרוֹם נִיבִי. עַד תּוּכַל כָּמוֹנִי לְבָאֵר וְלִפְרשׁ. עִנְיְנֵי הַפַּרְעשׁ. אָמַר לוֹ חֲבֵרוֹ: לֹא יֶחֶרְדוּ הַנְּשָׁרִים. מִקּוֹל הַצִּפֳּרִים. וְלֹא יִפְחֳדוּ הַזְּאֵבִים. לְקוֹל כַּנְפֵי הַזְּבוּבִים. הֲתַחְשֹׁב כִּי הֶחֱרַדְתַּנִי בְחִידָתְךָ? וְעַתָּה הַט אָזְנְךָ לִשִׁאֵלָתֶךָ. וְתִשְׁמַע תְּשׁוּבָתֶךָ… וַיַּעַן לְבָאֵר וְלִפְרשׁ. עַל עִנְיְנֵי הַפַּרְעשׁ.

וַיַּעַן וַיֹּאמֶר עַל הַפַּרְעשׁ: כּוּשִׁי וְלֹא מִבְּנֵי חָם. שָׁחוֹר כְּפֶחָם. לֶחֶם רֶשַׁע יִלְחַם. וּבְלִי שֶׁלַח יִלְחַם. וְכַתַּנּוּר יֵחַם. יֶאֱרֹב כְּגַנָּב בְּבֵיתֶךָ. וְיִשְׁכֶּב בְּשִׂמְלָתֶךָ. וּבַחֲדַר מִשְׁכָּבְךָ וְעַל מִטָּתֶךָ. וּבְשָׂרְךָ יִקַּח כָּל הַלַּיְלָה לְאָכְלוֹ. וְאִם בְּכָל הַיּוֹם תִּמְנַע מִמֶּנּוּ מַאֲכָלוֹ. עַד בּוֹא הַשֶּׁמֶשׁ תְּשִׁיבֶנּוּ לוֹ. וּבְהִלָּכֵד בְּחַבְלֵי שֵׁנָה הַנְּפָשִׁים. וּבִנְפֹל תַּרְדֵּמָה עַל אֲנָשִׁים. יָבוֹא בַלָּאט אֵלֶיךָ. לְהִתְגּוֹלֵל עָלֶיךָ. וּלְהִתְנַפֵּל עָלֶיךָ. וְיִמְצֶה דָמְךָ בְּהָקִיץ וּבַחֲלוֹם. וְיָשֵׂם דְּמֵי מִלְחָמָה בְּשָׂלוֹם. וְכִי תִדְרְשֶׁנּוּ. וְיַעֲבֹר וְהִנֵּה אֵינֶנּוּ. וְאִם תֹּאמַר: אַחֲרָיו אָרוּצָה. וַיָּנָס וַיֵּצֵא הַחוּצָה. וְאִם תִּתְפְּשֵׂהוּ פַעַם וּשְׁתַּיִם. יִמָּלֵט מִבֵּין הַיָּדַיִם. וְעשׂה יַעֲשֶׂה לוֹ כְּנָפַיִם. כְּנֶשֶׁר יָעוּף בַּשָּׁמַיִם. וְכַמָּה פְעָמִים יִתְחַבֵּא עִם הָעֲלָמוֹת תַּחַת צֶבַע רִקְמָתַיִם. וְיָבוֹא מִמָּתְנַיִם עַד יְרֵכַיִם. וְיִסְתַּתֵּר עִמָּם בֵּין הַשָּׁדַיִם.

וְיִקְרָא שֵׁם הַמָּקוֹם הַהוּא מַחֲנַיִם. וְכִי יִמְצָא נַעֲרָה בְתוּלָה. אוֹ בְעוּלָה. יִדְבַּק בָּהּ וְיִשְׁכַּב אֶצְלָהּ. עַד לְרֹעַ מַעֲשָׁיו תִּתֵּן קוֹלָהּ. וְצָעֲקָה הַנַּעֲרָה וְאֵין מוֹשִׁיעַ לָהּ. וְכִי יִשְׁאַלוּהָ לָמָּה תִבְכִּי וְלֹא תֶחֱשִׁי. תֹּאמַר: בָּא אֵלַי הָעֶבֶד הַכּוּשִׁי. וּמִשְׁכָּבוֹ בְּחֵיקִי הֵכִין. צְרוֹר הַמּוֹר דּוֹדִי לִי בֵּין שָׁדַי יָלִין. כָּל הַלַּיְלָה הִשְׁכִּיבַנִי לְמַעֲצֵבָה. וְלָקַח לוֹ לְמִשְׁכָּב וּלְמַצֵּבָה. הַזְּרוֹעַ וְהַלְּחָיַיִם וְהַקֵּבָה. הוּא כְּמוֹ כֹהֵן מַזֶּה בֶּן מַזֶּה. לָקַח לְאָכְלוֹ הַשּׁוֹק וְהֶחָזֶה. וְנָפַל בְּחֶלְקוֹ חֲזֵה הַתְּנוּפָה וְשׁוֹק הַתְּרוּמָה. וְחֶלְבּוֹ הָאַלְיָה תְמִימָה. וּבְכָל הַלַּיְלָה אֵשׁ קְרָב יַחְתֶּה. וְלֹא יִשְׁכַּב עַד יֹאכַל טֶרֶף וְדַם חֲלָלִים יִשְׁתֶּה. בְּלִי קֶשֶׁת יוֹרֶה אֶת עוֹרִי. וּבְלִי חֲנִית יִדְקֹר בְּשָׂרִי. הוּא קָטָן וּמַעֲשָׂיו גְּדוֹלִים. וְחַלָּשׁ וְיַחֲלשׁ נְפִילִים. וְנִבְזֶה וְהוּא רַב פְּעָלִים. לֹא תְצוּדֶנּוּ רָשֶׁת. וְלֹא יַבְרִיחֵנּוּ בֶּן קֶשֶׁת. שָׂרֵי הַצָּבָא בְּמִלְחַמְתּוֹ חֲלָשָׁם. וְהַגִּבּוֹרִים בְּמַאֲסָרוֹ כְּבָשָׁם. גַּם בְּנֵי עֲנָקִים רָאִינוּ שָׁם. אִם תִּסְתַּתֵּר מִמֶּנּוּ בִבְרִיחַ וּדְלָתַיִם. יָבֹא אֵלֶיךָ בֵּין הַחוֹמוֹתַיִם. וְיֵרֵד עָלֶיךָ מִן הַשָּׁמַיִם. וּבְחָשְׁבְךָ כִּי נִמְלַטְתָּ מִמֶּנּוּ. בְּחֵיקְךָ תִמְצָאֶנּוּ. וְעַל שִׁכְמְךָ תִּשָּׂאֶנּוּ. וְאִם תַּהֲרֹג חֲיָלָיו וְנִתְחֵיהֶם תְּנַתֵּחַ. יָבוֹאוּ קְרוֹבֵיהֶם לִשְׁפֹּךְ דָּמְךָ וְאַתָּה בוֹטֵחַ. גּוֹאֵל הַדָּם הוּא יָמִית הָרוֹצֵחַ. וְאִם לֹא יִמְצָא דֶרֶךְ לְהֵיכַל הַמֶּלֶךְ. יִתְחַבֵּא בִבְגָדָיו וְתַחַת יְצוּעָיו מִתְהַלֵּךְ. וְיָבוֹא אֵלָיו בַּלַּיְלָה כְּאֹרֵחַ וְהֵלֶךְ. בַּלַּיְלָה הַהוּא נָדְדָה שְׁנַת הַמֶּלֶךְ. וַיִּבֶז בְּעֵינָיו כָּל אֲשֶׁר הוּא מוֹלֵךְ. וְיִקְרָא לַעֲבָדָיו לַיְלָה. וּפָרְשׂוּ הַשִּׂמְלָה. וְהַשְּׁפָחוֹת זוֹ אֶל זוֹ צוֹעֲקוֹת בְּקוֹל. וְהַמְּנוֹרוֹת חָמֵשׁ מִיָּמִין וְחָמֵשׁ מִשְּׂמֹאל. וְהוּא מִתְחַבֵּא מֵהֶם וְהֵם אֵלָיו חוֹכִים. וְהוּא שׂוֹחֵק עֲלֵיהֶם וְהֵם בּוֹכִים. וְאִלּוּ נִמְצָא. דָּמוֹ נִמְצָה. אַךְ בְּכָל מָקוֹם לוֹ מָנוֹס וּמִפְלָט. וּבְכָל בֶּגֶד יִמְצָא עִיר מִקְלָט. וְהַמֶּלֶךְ אִם תַּשִּׂיגֵהוּ יָדוֹ. וְיִתְפְּשֵׂהוּ בְּבִגְדוֹ. לֹא יִשְׁקֹט עַד יָשִׂים לָאָרֶץ מְרֵרָתוֹ שְׁפוּכָה. וַחֲמַת הַמֶּלֶךְ שָׁכָכָה. הֵן אֵלֶּה קְצוֹת דְּרָכָיו. וְשֶׁמֶץ מַהֲלָכָיו. כִּי בְכָל יוֹם יִתְעַנֵּג מִכָּל מַאֲכָל. וּבָא הַשֶּׁמֶשׁ וְטָהֵר וְאַחַר יֹאכַל. יַזִּיק בְּלִי עָוֹן נְקִיִּים. וְיָלֹק דְּמֵי עֲשִׁירִים וַעֲנִיִּים. וְיִבְלָעֵם כִּשְׁאוֹל חַיִּים. כֹּה מִשְׁפָּטוֹ כָּל הַיָּמִים. לְהַגִּיעַ דָּמִים בְּדָמִים. וַיְהִי כִּי אָרְכוּ לוֹ שָׁם הַיָּמִים. וְיִרְאֶה כִּי קָצַר הַיּוֹם. וְחָסַר הַחֹם. וְהָרוּחַ תִּדְּפֶנּוּ. וְהַקֹּר יִגֳּפֶנּוּ.

וְהַגֶּשֶׁם יִשְׁטְפֶנּוּ. יִתְחַבֵּא בִמְחִלּוֹת עָפָר. וּמְעוֹנוֹ תַחַת לָאָרֶץ חָפָר. וְכָל גְּדוּדָיו יִבְרְחוּ וְיַעֲשׂוּ לָהֶם כְּלֵי גוֹלָה. וְיֵרְדוּ הֵם וְכָל אֲשֶׁר לָהֶם חַיִּים שְׁאוֹלָה. וְכָל יְמֵי הַקּוֹר יָשִׂימוּ בִשְׁאוֹל בֵּיתָם. וּבְרִגְבֵי עָפָר מְעוֹנָתָם. וְיֵשְׁבוּ תַחְתָּם בַּמַּחֲנֶה עַד חֲיוֹתָם. וּבִרְאוֹתָם הַגֶּשֶׁם יֶחְדַּל. וְהַקֹּר יִדַּל. וְהַחֹם יִגְדַּל. יָבוֹא עִתָּם. וְיִתְעוֹרְרוּ מִמְּכוֹן שִׁבְתָּם. וּמֵעֲפַר אֶרֶץ יִפְרְחוּ וְיָצִיצוּ. וְרַבִּים מִיְּשֵׁנֵי אַדְמַת עָפָר יָקִיצוּ.

וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר:

יְלִיד לַיִל אֱשׁוּן חשֶׁךְ יְשׁוּפוֹ

וּבוֹר הַמַּאֲפֵל שֶׁבַע צְרָפוֹ.

וְלוּלֵי רִחֲפוּ עָלָיו כְּנָפָיו

יְקֹד אִשּׁוֹ כְּפִי רֶגַע שְׂרָפוֹ.

תְּדַמֶּה כִּי מְלֶאכֶת אֵשׁ מְלַאכְתּוֹ

לְזֹאת קָדַר וְהַלַּהַב שְׁזָפוֹ.

יְעוֹפֵף רַק בְּכַנְפֵי הָאֲפֵלָה

וְיִתְגַּנֵּב וְלֹא נוֹדַע בְּעוּפוֹ

נְגָפַנִי וְלִי מַרְפֵּא בְהָרְגוֹ

הֲכִי בוֹ אֶמְצְאָה נָגוֹף וְרָפוֹא.

וְצוּר חַרְבּוֹ בְשִׁנָּיו גַּם בְּפִיהוּ

וְעַל דָּמִי בְּכָל רֶגַע שְׁלָפוֹ.

וְדָמוֹ הוּא כְמוֹ שׁוֹשָׁן מְאָדָּם

וְאוּלָם הוּא בְטֶרֶם צָץ קְטָפוֹ.

יְבוֹאֵנִי לְעַרְשִׂי בַחֲצוֹת לֵיל

וְלֹא יוֹעִיל לְאִישׁ דֶּלֶת יְגִיפוֹ.

וּמִשְׁפָּטוֹ כְּמִשְׁפַּט הָעֲטָלֵף

אֲשֶׁר בַּלַּיְלָה יִפְרֹשׂ כְּנָפוֹ.

וְתַשְׁכִּיבוֹ בְלַיְלָה הָאֲפֵלָה

עֲלֵי גַבִּי וְהַשַּׁחַר יְעִיפוֹ

יְקַצֵּר אֶת תְּנוּמָתִי וְאַךְ יַ־

אֲרִיךְ לֵילִי וְשָׂם רֹאשׁוֹ בְּסוֹפוֹ

כְּאִלּוּ יִמְצָא בִי נַחֲלֵי צוּף

בְּבוֹאוֹ לִי צְמֵא הַלֵּב עֲיֵפוֹ.

וְהוּא קָטָן צְעִיר יָמִים צְעִיר גּוּף

וְיַד הַיּוֹם בְּלִי רוּחַ תְּנִיפוֹ.

וְאֶחְשֹׁב כִּי נְקֻדָּה מִדְּיוֹ הוּא

וְנִמְלַט מִלְּשׁוֹן הָעֵט בְּנָטְפוֹ.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר הִשְׁלִימוּ שְׁנֵיהֶם אִמְרֵיהֶם. וְכִלּוּ שִׁירֵיהֶם. חָרַד לֵב הַשּׁוֹפֵט לְתֹקֶף דִּבְרֵיהֶם. וְאָמַר: הֲרִימוֹתִי יָדִי לְאֵל עֶלְיוֹן. הַשּׁוֹכֵן בְּרוּם חֶבְיוֹן. אִם רָאוּ כַמַּרְאֶה הַזֶּה עֵינַי. אוֹ שָׁמְעוּ הַדָּבָר הַזֶּה אָזְנַי. הִנֵּה יָדַעְתִּי כִּי אֵין עֲרֹךְ אֲלֵיכֶם. בְּכָל בְּנֵי דוֹרְכֶם. כִּי כָּל אִישׁ מִכֶּם גִּבּוֹר בְּמִלְחַמְתּוֹ. וּמָהִיר בִּמְלַאכְתּוֹ. וְעַתָּה כִּרְתּוּ שְׁנֵיכֶם בְּרִית אַהֲבָה. וְהָסִירוּ הָאֵיבָה. וְכַאֲשֶׁר הֶחְבִּיר הָאֵל הַמְּלִיצוֹת בֵּינֵיכֶם. הִתְחַבְּרוּ בִידִידוּת שְׁנֵיכֶם. וַיַּעֲנִיקֵם הַשּׁוֹפֵט מִבִּרְכָתוֹ. וַיָּאצֶל עֲלֵיהֶם נִדְבָתוֹ. וַיֵּצְאוּ מִלְפָנָיו שְׂמֵחִים בִּתְהִלָתוֹ. עֲמוּסִים בִּנְדִיבוּתוֹ. וּבְצֵאתָם מִלְפָנָיו יָצָאתִי אַחֲרֵיהֶם. וְשָׁאַלְתִּי הַגָּדוֹל מֵהֶם אֶת שְׁמוֹ לְהוֹדִיעֵנִי. וְסוֹדוֹ לַהֲבִינֵנִי.

וַיַּעַן וַיֹּאמַר:

אֲנִי חֶבֶר וְזֶה פִּרְיִי וְאוּלָם

אֲנִי וְהוּא כְּפִירֵי הַמְּלִיצוֹת.

אֲנִי אֶבְנֶה נְתִיבוֹת בִּין נְתוּצוֹת

וְהוּא יָקִים הֲרִיסוֹת שִׁיר פְּרוּצוֹת.

וּכְשָׁמְעִי שִׁירוֹ יָדַעְתִּי כִּי הוּא חֲבֵרֵנוּ חֶבֶר הַקֵּינִי. וּבְנוֹ הַשֵּׁנִי. וְנִפְרַדְתִּי מֵהֶם. וַאֲנִי תָמֵהַּ עֲלֵיהֶם. וְנָסַעְתִּי מֵאִתָּם וְעָזַבְתִּי לִבִּי עִמָּהֶם.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
