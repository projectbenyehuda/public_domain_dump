ספר תחכמוני - שַׁעַר שְׁלשִׁים וּשְׁלשָׁה

בְּשִׁירִים עַל אַלְפָ“א בֵיתָ”א יְקָרִים. מֵהַרֲרֵי שֵׂכֶל גְּזוּרִים.


נְאֻם הֵימָן הָאֶזְרָחִי: נָטִיתִי לִבִּי. בְּעוֹדִי רַעֲנָן בְּאִבִּי. וְעֶזְרָתִי בִי. לִשְמֹעַ חִידוֹת חֲכָמִים. וְלִלְקֹט מִפִּיהֶם לְשָׁמִים. וְלִרְקֹחַ מֵאֲבַק שִׂכְלָם רֶקַח בְּשָׂמִים. לְבַלּוֹת יָמַי בַּטּוֹב וּשְׁנוֹתַי בַּנְּעִימִים. וַיְהִי הַיּוֹם וַאֳנִי בְּהַר שֵׂעִיר. מִתְהַלֵּךְ בָּעִיר. וְרָאִיתִי חֶבְרַת מְשׁוֹרְרִים. מְמַלְּאִים פְּנִינֵי אֲמָרִים. וּמְפִיצִים מִלְּשׁוֹנָם נַחֲלֵי שִׁירִים. וְקָרַבְתִּי אֲלֵיהֶם. לִרְווֹת בְּנַחֲלֵי מֵימֵיהֶם. וְלַחֲזוֹת בְּמַנְעַמֵּיהֶם. וַיֹּאמֶר אֶחָד מֵהֶם לַחֲבֵרָיו: הֲיֵשׁ בָּכֶם מִי יְחַבֵּר שִׁיר נִצְמָד. יִהְיֶה בוֹ לְכָל לֵב מַחְמָד. וְיָפְיוֹ לֹא יִמַּד. וַיִּשְאָׁלוּהוּ מִקְצָתָם עַל הַשִּׁיר הַנִּצְמָד. אָמַר לָהֶם: דְּעוּ כִּי הַשִּׁיר הַנִּצְמָד. הוּא אֲשֶׁר כָּל סוֹף שְׁנֵי בָתִּים מִלָּה אַחַת וְהַמִּלָּה לֹא תִשְׁתַּנֶּה. אֲבָל הָעִנְיָן יְשֻׁנֶּה. וּכְשָׁמְעָם דְּבָרוֹ. וְהֵבִינוּ תְכוּנַת מַאֲמָרוֹ. רָץ כָּל אֶחָד מֵהֶם בְּמַעֲרֶכֶת נִיבוֹ. וַיְחַבֵּר אֲשֶׁר נָתַן אֱלֹהִים בְּלִבּוֹ. וְהָיָה שָׁם אִישׁ זָקֵן שׁוֹמֵעַ וּמַחֲרִישׁ. וּמִמְּקוֹמוֹ לֹא יָמִישׁ. כְּצוּר הַחַלָּמִישׁ. וְכִרְאוֹתוֹ כִי יָגְעוּ. וְלֹא נָגְעוּ. וְהַמְּלָאכָה לֹא יָדְעוּ. אָמַר לָהֶם: דְּעוּ כִּי דֶרֶךְ קָשָׁה רְכַבְתֶּם. וּמִבּוֹר עָמֹק שְׁאַבְתֶּם. וּמֵהַר חָזָק חֲצַבְתֶּם. וְאִם תִּרְצוּ לִשְׁמֹעַ הַשִּׁיר הַנִּצְמָד. וְהַנִּיב הַנֶּחְמָד. הַטּוּ אָזְנְכֶם אֵלָי. וְהַקְשִׁיבוּ לְמִלָּי.

וַיַּעַן וַיֹּאמַר: תְּהִלָּה לָאֵל אֲשֶׁר עָצְמָה תְהִלָּתוֹ. וּמָלְאָה הָאָרֶץ תְּהִלָּתוֹ. וְהָאוֹר וְהַכָּבוֹד כְּתָרָיו וְגֻלָּתוֹ. וּצְמֵאֵי טוּבוֹ יִרְווּ בְּמַעְיַן גֻּלָּתוֹ. הַנּוֹתֵן רְפוּאַת כָּל חוֹטֵא תְשׁוּבָתוֹ. לַעֲלוֹת לְמַעְלָה רוּחוֹ וְהָרָמָתָה תְּשׁוּבָתוֹ. לְהִסָּתֵר מֵאַף ה' וַחֲמָתוֹ. וְלִרְחֹץ הֲלִיכָיו בַּחֲמָתוֹ. הַנּוֹתֵן טוֹבַת הַיֵּצֶר פְּרִידָתוֹ. וְחֶבְרַת הַיּשֶׁר פְּרִי דָתוֹ. לִשְׁאֹב בְּמַעְיַן הַשֵּׂכֶל וּתְעָלָתוֹ. וְלִמְצֹא בוֹ צְרִי הַלֵּב וּתְעָלָתוֹ. לְהָרִים נַפְשׁוֹ מֵעֲפַר הַבּוּז וְאַשְׁפָּתוֹ. וּלְהָבִיא בְּכִלְיוֹת הַתַּאֲוָה בְּנֵי אַשְׁפָּתוֹ. וְלִהְיוֹת חֶבֶל אֱלֹהָיו וּסְגֻלָּתוֹ. וְלָשׂוּם דָּתוֹ מִבְחַר חֲפָצָיו וּסְגֻלָּתוֹ. וּבְמִלְחֲמוֹת הַצָּרוֹת יְהִי צִדְקוֹ צִנָּתוֹ. לְהִסָּתֵר בְּמַחֲסֵהוּ מִפְּנֵי חֲמָתוֹ וְצִנָּתוֹ. לָרוּץ אַחֲרֵי אֱלֹהָיו וְאַל יֶחְדַּל מִמְּרוּצָתוֹ. לְהַשְׁבִּית עשֶׁק הַיֵּצֶר וּמְרוּצָתוֹ. וְלִלְקֹט מִפְּרִי הַמָּדָּע תְּבוּנָתוֹ וְזִמְרָתוֹ. וְלַעֲרֹךְ עָלָיו שִׁירוֹ וְזִמְרָתוֹ. וְלַעֲלוֹת מִשְּׁאוֹן הַחֵטְא וְנַחְלָתוֹ. וְלִהְיוֹת עִסְקוֹ בְּעוֹד יְחִידָתוֹ. וּלְהָשִׁיב אֶל אֵל יְחִידָתוֹ. וְלִגְדֹּר כֶּרֶם הַיּשֶׁר וְאַל יָסֵר מְשׂוּכָּתוֹ. וְלִהְיוֹת בְּגַן אֱלֹהִים כָּל אֶבֶן יְקָרָה מְסֻכָּתוֹ. וְלִכְבּשׁ בְּעֹל הַשֵּׂכֶל תַּאֲוָתוֹ. וְלִהְיוֹת בִּנְוֵה הַשָּׁלוֹם גְּבוּלוֹ וְתַאֲוָתוֹ. וּלְהָסִיר מוּם הָרֶשַׁע וּמָשְׁחָתוֹ. וּבְשֶׁמֶן הַשֵּׁם הַטּוֹב יָשִׂים מִשְׁחָתוֹ. וּבִידֵי הָרַעְיוֹן יְסַקֵּל מְסִלָּתוֹ. וְיָשִׂים הַתֹּם מִסְעָד לְבֵיתוֹ וּמְסִלָּתוֹ. אָז יִרְאֶה זִיו יוֹצְרוֹ עַיִן בְּעַיִן. וְיִפְרַח כְּבֵן פֹּרָת עֲלֵי עָיִן. בְּחָקְרוֹ מֵאֵי זֶה מוֹבָאוֹ וּמֵאַיִן. וְיַכִּיר כִּי סוֹפוֹ אָיִן. וְיֵדַע כִּי הוּא בְעוֹלָמוֹ אוֹרֵחַ וָהֵלֶךְ. וְאַל יְסִיתֵהוּ מִדְּבַשׁ תַּעֲנוּגָיו הֵלֶךְ. וּבְחִצֵּי שִׂכְלוֹ יַכֶּה הַיֵּצֶר וְיוֹרֵהוּ. וּבְדֶרֶךְ טוֹבָה יִצְעַק אֶל הָאֵל וְיוֹרֵהוּ. וְאָז אֱלֹהָיו יָאִיר אוֹרוֹ. וְעָלָיו יָפִיץ עֲנַן אוֹרוֹ. וּמַלְאַךְ פָּנָיו יְהִי שְׁלוּחוֹ וְצִירוֹ. לַחֲבשׁ נִגְעוֹ וְצִירוֹ. וְאִם יַזִּיר מִן הַתַּאֲוָה יִשְׂמַח בְּסוֹף יְמֵי נִזְרוֹ. וְעָלָיו יָצִיץ נִזְרוֹ.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר הִשְׁלִים אִגַּרְתּוֹ הָרְצוּפָה. וְאִמְרָתוֹ הַצְּרוּפָה. וְחֶרֶב מְלִיצָתוֹ הַשְּׁלוּפָה. וּתְהִלָּתוֹ אֲשֶׁר בִּצְנִיף כָּבוֹד צְנוּפָה. תָּמְהוּ כָל הָעָם לִתֹקֶף מִלָּיו וְעִנְיָנָיו. וּלְנֹעַם הֶגְיוֹנָיו. אַחֲרֵי כֵן אָמַר לָהֶם: וְעוֹד אַגִּיד לָכֶם שִׁירוֹת חֲמוּדוֹת. וּמְלִיצוֹת צְמוּדוֹת. בְּרֹאשָׁם וְסוֹפָם מִלּוֹת נִצְמָדוֹת. מְלֵאוֹת שֵׂכֶל וְחִידוֹת.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אוֹת אָלֶ"ף

אֱמוּנָה הִיא עֲדִי רוֹזְנִים וְשָׂרִים

וְהִיא מַלְכָּה וְכָל מֶלֶךְ אֲסִירָהּ.

אֲמוּנָה הָיְתָה נַפְשִׁי בְחֵיקָהּ

וְעַל כֵּן מִלְבָבִי לֹא אֲסִירָהּ.

אוֹת בֵּי"ת

בּוֹחֵן לְבָבוֹת יַחֲקֹר סוֹד לִבְּךָ

לָכֵן בְּישֶׁר הַחֲזֵק תָּמִיד בְּנִי.

בּוֹ חֵן וְטוֹב תִּמְצָא בְּעֵת כִּי הַזְּמַן

יָמִיר בְּאֵיד גִּילָךְ וְשִׂמְחָתָךְ בְּנִי.

אוֹת גִּימֶ"ל

גִּלְעָד צֳרִי אֵין בּוֹ לְצִירֶיךָ

עַד תִּהְיֶה חֵיל חֶטְאֲךָ גוֹרֵשׁ.

גִּיל-עַד אֲזַי תִּמְצָא בְּעֵת יִהְיֶה

יָם הַזְּמַן רֶפֶשׁ וְטִיט גּוֹרֵשׁ.

אוֹת דֶּל"ת

דּוֹר שָׁוְא יְבַקְשׁוּ תַעֲנוּג לָכֵן דְּרשׁ

הָאֵל וְסוּרָה מֵהַתֻלֵי דוֹרְךָ.

דּוֹרְשָׁיו בְּכָל-לֵב מָצְאוּ יִשְׁעוֹ בְּעֵת

יַד הַזְּמַן קֶשֶׁת אֲלֵיהֶם דּוֹרְכָה.

אוֹת הֵ"א

הַצֵּל יְחִידָתָךְ בְּצִדְקָךְ רַק

עַל יִצְרְךָ צוּר חַרְבְּךָ הָרִיק.

הַצֵּל כְּרֶגַע סָר וְכֵן רוֹדְפֵי

טוֹבַת זְמַן רָע יִרְדְּפוּ הָרִיק.

אוֹת וָי"ו

וְשָׁנִים חוֹלְפִים כַּצֵּל וְנֶפֶשׁ

אֱנוֹשׁ תָּעוּף כְּצִפּוֹר וַחֲסִידָה.

וְשָׁנִים פּוֹשְׁטָה רַק תַּעֲטֶה שַׂק

בְּבוֹא יוֹם דִּין לְחוֹטְאָה וַחֲסִידָה.

אוֹת זַיִ"ן

זֶרַע אֱמֶת בֶּן אִישׁ זְרוֹעַ

תִקְצֹר אֲלֻמּוֹת עַל זְרוֹעַ.

זֶה רַע אֲשֶׁר חֵטְא יִזְרְעָה

אֵיד יִקְצְרָה אַחַר זְרוֹעַ.

אוַת חֵי"ת

חֲנִי-נָא בֵית יְקָר נַפְשִׁי וְאַשְׁרֵי

אֱנוֹשׁ נַפְשׁוֹ עֲדִי צֶדֶק חֲלִי לָהּ.

חֲנִינָה תִהְיֶה מוֹצְאָה וְלָאֵל

בְּשִׁיר חָדָשׁ תְּנַגֵּן עַל חֲלִילָּהּ.

אוֹת טֵי"ת

טוֹבְעִים בְּיָם רֶשַׁע דְּעוּ נָא כִי

אַשְׁרֵי אֱנוֹשׁ בִּנְהַר אֱמֶת טָבְלוֹ

טוֹב עִם אֱלֹהָיו הוּא וְכֵן עִם כָּל

אָדָם לְזֹאת לֹא יֶחֱסַר טוֹב לוֹ.

אוֹת יוּ"ד

יִמָּאֲסוּ דִבְרֵי חֲכָמִים לַכְּסִיל

רַק יֶאֱהַב תֵּבֵל וְיִדְרשׁ כָּל-יְקָר.

יִמָּאֲסוּ יָמָיו כְּמוֹ מַיִם וְשַׁח

לִשְׁאוֹל וְלֹא בִזְבוּל עֲלִיּוֹתָיו יְקָר.

אוֹת כָּ"ף

כִּידוֹד יְקוֹד תֹּפֶת לְךָ הוּכַן לְזֹאת

עוּרָה וְלַעֲבוֹדַת אֱלֹהֶיךָ כְסוֹף

כִּי דוֹד וְאָח לוֹ יַעַזְרוּ אוֹתָךְ בְּעֵת

מוֹתָךְ וְרֹאשׁ עוֹלָמְךָ יָשׁוּב כְּסוֹף.

אוֹת לָמֶ"ד

לוֹחֵץ לְבָבוֹ אַחֲרֵי הַתַּאֲוָה

לֵךְ אַחֲרֵי שִׂכְלָךְ וְאָרְחוֹתָיו לְמַד.

לוֹ חֵץ תְּפַלַּח בּוֹ בְּלֵב הָאוֹיְבִים

וּבְעֵת תְּהִי עָרֹם לְךָ יִהְיֶה לְמַד.

אוֹת מֵ"ם

מוֹרִים דְּבַר הָאֵל וּמִתְגָּאִים

שׁוֹכְנֵי קְבָרוֹת יוֹשְׁבֵי מִדִּין.

מוֹרִים הֱיוּ אֶת לִבְּכֶם לָאֵל

אוּלַי נְקִיִּים תֵּצְאוּ מְדִּין.

אוֹת נוּ"ן

נְצַפֶּה לָךְ אֱלֹהֵינוּ הֲכִי לָךְ

גְּדֻוָּלה עַל בֵּני אָדָם נְסוּכָה.

נְצַפֶּה מַאֲמָרֵינוּ זְהַב שָׁוְא

וּמַסֵּכָה עֲלֵי כָל-לֵב נְסוּכָה.

אוֹת סָמֶ"ךְ

סוּגָה עֲרוּגַת תַּאֲוָה שׁוֹשָׁן

הַרְחֵק וְאַל תָּבוֹא בְּעַם סוֹדָהּ.

סוּגָה מְאֹד נַפְשָׁם מְלֵאָה חֵטְא

אָכֵן לְאֵל לֹא נֶעֱלָם סוֹדָהּ.

אוֹת עַיִ"ן

עָרֹם בְּלִי צֶדֶק הֲלֹא תֵבוֹשׁ לְמוּל

הָאֵל קְחָה סוּת תֹּם אֲבָל אוֹתוֹ עֳדֵה

עָרוּם תְּהִי אִם תַּחֲלשׁ חֵיל יִצְרְךָ

לָכֵן שְׁאַג עָלָיו וְכַשַּחַל עֲדֵה.

אוֹת פֵ"א

פִּנָּה לְבַב מַשְׂכִּיל מְסִלָּתוֹ

מֵחֵטְא. וְעֻלּוֹ מִזְּמַן פּוֹרֵק.

פִּנָּה וּמִגְדָּל יִהְיֶה לוֹ אֵל

מֵחֵיל זְמַן עַצְמוֹת אֱנוֹשׁ פּוֹרֵק.

אוֹת צָדִ"י

צָדוּ צְעָדַי מוֹקְשֵׁי חֶטְאִי

עָלַי כְּמֵימֵי נַחֲלָה צָפוּ.

צָדוּ נְאוֹת גִּילִי עֲדֵי אוֹיְבַי

רָאוּ בְיוֹם אֵידִי וְגַם צָפוּ.

אוֹת קוֹ"ף

קְרָב יָמִים בְּלִי תַכְלִית. וְצִירֵי

תְּלָאוֹת רִחֲפוּ עָלָיו קְלָעִים.

קְרַב לִבִּי לְעַבְדוּת אֵל וְאַל תַּט

לְבֵית מִדּוֹת פְּטוּרֵי צִיץ קְלָעִים.

אוֹת רֵי"ש

רָאשִׁים לְעָפָר שָׁפְלוּ כֻלָּם

אַחַר אֲשֶׁר עַל הַזְּמַן רָדוּ.

רָשִׁים וְדַלִּים נֶהְפְּכוּ בַּסּוֹף

כִּי מַר לְחִכָּם צוּף אֲשֶׁר רָדוּ.

אוֹת שִׁי"ן

שָׁעַר לְבָבִי בַּעֲבוֹדַת אֵל

כִּי טוֹב וְנוֹחַ אֶמְצְאָה שָׁמָּה

שַׁעַר דְּבִיר הַתַּאֲוָה אֶסְגֹּר

עַד יֶחֱרַב גַּם יִהְיֶה שָׁמָּה.

אוֹת תָּי"ו

תַּבַל יְגוֹנַיהָ בְלִי קֶצֶב

כַּאֲשֶׁר מְצוּקֶיהָ בְלִי תִכְלָה.

תֶּבֶל עֲזֹב בֶּן אִישׁ וְשׁוּב אֶל אֵל

וַעְרֹךְ תְּהִלָּה לוֹ וְלֹא תִכְלָה.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר הִשְׁלִים אֲמָרָיו.

וּמְלִיצוֹת דְּבָרָיו. נִפְלְאוּ כָל הַיּוֹשְׁבִים בְּטוּב טַעְמוֹ.

וּמֶתֶק נָאֳמוֹ. וַיִּשְׁאָלוּהוּ עַל שְׁמוֹ.


וַיַּעַן וַיֹּאמַר:

אֲנִי חֶבֶר וְתוֹךְ פי נֶחְבְּאוּ כָּל

פְּזוּרֵי שִׁיר וְנִדְחֵי הַמְּלִיצוֹת.

וְאִם חָכְמוֹת לְכָל רוּחַ נְפוֹצוֹת

הֲלֹא הֵמָּה בְּרוּחַ פִּי קְבוּצוֹת.
