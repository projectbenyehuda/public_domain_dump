ספר תחכמוני - שַׁעַר עֶשְׂרִים וּשְׁנַיִם

עִנְיַן הָאִישׁ הַחוֹזֶה בַכּוֹכַבִים וּמַזָּלוֹת וּבְעָרְמָתוֹ וְשִׂכְלוֹ יְדַבֵּר גְּדוֹלוֹת


נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי בִימֵי חָרְפִּי. וּבִזְרֹחַ כּוֹכְבֵי נִשְׁפִּי. וְהַגַּלְגַל סוֹבֵב עַל פִּי. מִתְעַלֵּס בַּגַּנִּים. וְרוֹעֶה בַשּׁוֹשַׁנִים. עַל מַשְׁאַבֵּי נְחָלִים. וְתַחַת סִבְכֵי הַצְּלָלִים. אֶשְׁאַף רוּחוֹת. בֶּשֶׁמֶן הַמּוֹר רְקוּחוֹת. בָּהֶם לַנֶּפֶשׁ מְנוּחוֹת. וְלַלֵּב שְמָחוֹת. וַיְהִי בִּקְצָת הַיָּמִים. יָצָאתִי לְשׁוֹטֵט בַּמִּדְבָּר וּבָעֲרָבָה. וָאֶתְהַלְּכָה בָרְחָבָה. וָאֵרָא לְשַׁעַר הָעִיר אִישׁ נִצָּב. דּוֹאֵג וְנֶעֱצָב. וְכַאֲשֶׁר שַׂמְתִּי עָלָיו עֵינִי. מְצָאתִיהוּ חֶבֶר הַקֵּינִי. וִיְגַּדל עַל עָצְבּוֹ יְגוֹנִי. וּמִהַרְתִּי אֵלָיו לְשָׁאֳלוֹ. עַל הַמִּקְרֶה אֲשֶׁר קָרָה לוֹ. אָמַר לִי: הָיִיתִי אֶתְמוֹל עִם חֲבֵרִים. מִיַּלְדֵי הָעִבְרִים. וּבְעוֹדֵנוּ בְשַׁעַר הָעִיר רָאִינוּ עַם רַב נִקְבָּצִים. וּמִכָּל עֵבֶר בָּאִים וְרָצִים. וְאִישׁ בְּאָחִיו נִגָּשִׁים וְנִלְחָצִים. וּמִהַרְתִּי אֲנִי וַחֲבֵרַי אֲלֵיהֶם.

וְשָׁאַלְנוּ לְאִישׁ מֵהֶם. לְהוֹדִיעֵנוּ מְגַמַּת פְּנֵיהֶם. וַיֹּאמֶר לָנוּ כִּי יֵשׁ בְּשַׁעַר הָעִיר חָכָם מֵחַכְמֵי הָעֲרָבִים. חוֹזֶה בַכּוֹכָבִים. יְגַלֶּה הָעֲתִידוֹת בְּטֶרֶם הֱיוֹתָם. וְיַגִּיד הַסּוֹדוֹת עַד לֹא חֲנוֹתָם. וַיֹּאמְרוּ לִי חֲבֵרַי: לְכָה נָא וְנִשְמְעָה נְבוּאוֹתָיו. וְנִרְאֶה מַה יִהְיוּ חֲלוֹמוֹתָיו. וְכַאֲשֶׁר בָּאנוּ אֶל הֶהָמוֹן רָאִינוּ זָקֵן בָּא בַיָּמִים. וּמֵהֲדַר הַחָכְמָה עָלָיו אֵימִים. וּבְיָדָיו כְּלִי נְחשֶׁת. וְעָלָיו מַעֲשֵׂי רֶשֶׁת. וְהוּא שׁוֹקֵל הַשֶּמֶשׁ בְּגַלְגַּלֶּיהָ. וּמְרוּצָתָהּ בְּמַעְגָּלֶיהָ. וּמֶרְחֲבֵי גְלִילֶיהָ. וּנְטִיַּת צְלָלֶיהָ. וְהוּא אוֹמֵר לְכָל הַנִּצָבִים עָלָיו: הָאוֹתִיּוֹת שֶׁאָלוּנִי. וְעַל הָעֲתִידוֹת חִקְרוּנִי. וּבְסוֹדוֹת הָאָרֶץ אַל תִּבְחָנוּנִי. רַק בְּסוֹדוֹת הַשָּׁמַיִם נַסּוּנִי. כִּי לֹא בַסֵּתֶר דִּבַּרְתִּי וְלֹא אָמַרְתִּי תֹהוּ בַקּשׁוּנִי. אֲנִי אַגִּיד לָכֶם חֻקּוֹת שָׁמַיִם. וְסוֹד הַמְּאוֹרוֹת הַשְּׁנַיִם. וְלַחֲמֵשֶׁת כּוֹכְבֵי לֶכֶת. בְּלִבִּי מַעֲרֶכֶת. יָדַעְתִּי סוֹדוֹת מַזְּלוֹתָם. וְחַצְרֵיהֶם וְטִירוֹתָם. וְרִדְתָּם וַעֲלוֹתָם. שְׁנֵים עָשָׂר נְשִיאִים לְאֻמּוֹתָם. אִם יֵשׁ לָהֶם בַּשַּׁחַק דְּרָכִים. לָהֶם בְּלִבִּי מַהְלָכִים. וְאִם בַּמָּרוֹם מַעְגְּלֵיהֶם. בְּרַעְיוֹנִי גַּלְגַּלֵּיהֶם. וְלֹא נֶעְלַם מִמֶּנִּי שַׁבְּתַי וְרָעוֹתָיו. וְצֶדֶק וְטוֹבוֹתָיו. וּמַאְדִּים וּמִלְחֲמוֹתָיו. וְהַשֶּׁמֶשׁ וּמֶמְשְׁלוֹתָיו. וְנֹגַהּ וַחֲמוּדוֹתָיו. וְכוֹכַב חַמָּה וְחָכְמוֹתָיו. וְסַהַר וְתוֹלְדוֹתָיו. וְסוֹד הַגַּלְגַּל וּמַעֲלוֹתָיו. וְהַצּוֹמֵחַ וּמַזְּלוֹתָיו. וּמֵיתָרָיו וְיִתְדוֹתָיו. עַל אַרְבַּע פַּעֲמוֹתָיו. וּמַזַּל טָלֶה. בְּחַבְלֵי שִׂכְלִי יִתָּלֶה. וְקַרְנֵי הַשּׁוֹר. בְּלִי עֲבוֹת אֶקְשֹׁר. וְעָזַבְתִּי תְאוֹמִים. בְּחַיֵּי הָאֵם יְתוֹמִים. וְגֹבַהּ סַרְטָן. לְמַעֲלָתִי יִקְטַן. וּמַזַּל אַרְיֵה. הוּא לְאֵימָתִי חוֹלֶה וְנִהְיֶה. וְאִתִּי בְתוּלָה. בַּחֲדַר לִבִּי בְעוּלָה. וְחָשַׁבְתִּי מֹאזְנַיִם. כְּשַׁחַק מֹאזְנַיִם. וְנָשַׁךְ לְעַקְרָב. מִלְּשׁוֹנִי עַקְרָב. וְתוֹפֵשׂ הַקֶּשֶׁת. יִירָא אֵלַי מִגֶּשֶׁת. וְשִׁסַּעְתִּי מַזַּל הַגְּדִי. כְּשַׁסַּע הַגְּדִי. וּלְמֵימֵי שִׂכְלִי. נֶחְשְׁבוּ מֵי הַדְּלִי. כְּמַר מִדְּלִי. וּמַזַּל דָּגִים. נִתְפַּס בִּמְצוּדָתִי כְדָגִים. יָדַעְתִּי עֲתִידוֹתֵיהֶם. וְחָקַרְתִּי צְפוּנוֹתֵיהֶם.

וְהוֹצֵאתִי לָאוֹר מִשְׁפְּטֵיהֶם. אֶשְׁפֹּט עֲלֵיהֶם בְּצֶדֶק. וַאֲחַזֵּק מֵהֶם כָּל בֶּדֶק. וְאַשְׁמִיעַ לִבְנֵי אִישׁ נוֹרָאוֹת. וּדְּבָרִים כְּדִבְרֵי נְבוּאוֹת. וְאַגִּיד לְכָל אִישׁ אֲשֶׁר יִמְצָא אוֹתוֹ. מִיּוֹם הֱיוֹתוֹ. וְעַד יוֹם מוֹתוֹ. אִם יַצְלִיחַ אוֹ יִגְבַּר. וְאִם יָמוּת בַּיָּם אוֹ בַמִּדְבָּר. וְאוֹדִיעֵהוּ מַזָּלוֹ וְהַצְלָחָתוֹ. וְרִדְתּוֹ וַעֲלוֹתוֹ. וּפָרָשַׁת גְּדֻלָּתוֹ. וְכָל מַעֳשֵׂה תָקְפוֹ וּגְבוּרָתוֹ. וּמִסְפַּר שְׁנוֹתָיו בְּלִבִּי חֻצָּצוּ. וּבִלְשׁוֹנִי נֶחֱרָצוּ. וְיָדַעְתִּי מָתַי יַעֲשִׂיר אוֹ יִדַּל.

וְיִשְּׁפַּל אוֹ יִגְדַּל.

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

אֲנִי לָשׁוֹן לְקוֹרוֹת הַזְּמַנִּים

אֲגַלֶּה לֶאֱנוֹשׁ רָזִים צְפוּנִים.

וְאַגִּיד לוֹ מוֹצְאוֹתָיו בְּטֶרֶם

הֱיוֹתָם, וַאֲשֶׁר הָיָה לְפָנִים.

וְאִם יִצְלַח וְאִם יִדַּל וְיִגְדַּל

וְאִם יֵלֵךְ לְאָחוֹר אוֹ לְפָנִים.

וְאִם יַעְשִׁיר וְיִוָּרֵשׁ וּמָתַי

וְאִם יִזְכֶּה לְבָנוֹת אוֹ לְבָנִים.

וְאִם יִחְיֶה וְאִם יָמָיו אֲרֻכִּים

וְאִם יָמוּת קְצַר יָמִים וְשָׁנִים.

וְאִם יַרְחִיק נְדוֹד מֵעִיר מְגוּרָיו

וְאִם יִצְבֹּר כְּחוֹל עשֶׁר וְהוֹנִים.

וְאוֹדִיעוֹ אֲשֶׁר צָפַן בְּלִבּוֹ

וּמִי יָבִין צְפוּנֵי רַעְיוֹנִים.

וְאַגִּיד לוֹ עֲתִידוֹת קוֹרוֹתָיו

לְאוֹתוֹת מוֹעֲדִים יָמִים וְשָׁנִים.

לְכוּ-נָא שַׁאֲלוּ בַּעְדִי וְחִקְרוּ

הֲנִמְצָא אִישׁ כְּמוֹתִי בַזְּמַנִּים.


אָמַר הַמַּגִּיד: וְכִשְׁמֹעַ הָעָם פְּלִיאוֹת אֲמָרָיו. וְתֹקֶף דְּבָרָיו. כָּרְעוּ לְפָנָיו. וְנִבְהֲלוּ מִפָּנָיו. וּבָאוּ לִשְׁאֹל כָּל אִישׁ עַל סודותיו. וְלִרְאוֹת עֲתִידוֹתָיו. זֶה יִשְׁאַל עַל יְגוֹנוֹ. וְזֶה יִדְרשׁ עַל מְאֹרְעוֹת זְמַנּוֹ. וְזֶה יַחְקֹר עַל מַזַּל בְּנוֹ. וְזֶה יִדְרֹש עַל מְלַאכְתּוֹ וּמִפְעָלוֹ. וְזָה יַגִּיד נֶגַע לְבָבוֹ וְחִילוֹ. וְהוּא מֵשִׁיב לְכָל אִישׁ מַעֲנֶה עַל שְׁאֵלָתוֹ. וְהֵם נוֹתְנִים לוֹ אִישׁ אֲשֶׁר כְּבִרְכָתוֹ. וְאֲנִי לְרֹע מַזָלי. וְהַיָּגוֹן אֲשֶׁר עָלָה בְּגוֹרָלִי. אָמַרְתִּי לַחֲבֵרַי: בַּמֶה נוּכַל לְנַסּוֹתוֹ. וְלָדַעַת שְׁקָרָיו מֵאֱמוּנָתוֹ. וְלִבְחֹן שִׂכְלוֹ מִסִּכְלוֹ. וּלְהַבְדִּיל זְהָבוֹ מִבְּדִילוֹ. אָמְרוּ לי: לְכָה נָא וְנִשְׁתַּתֵּף כֻּלָּנוּ בִּשְׁאֵלָה אַחַת וְנֵלֵךְ אֵלָיו יַחד.

ולבּוֹתֵינוּ לֵב אֶחָד. וְנִשְׁאַל עַל עִנְיַן הַגְּאֻלָּה. וּמָתַי תִּהְיֶה הַתְּשׁוּעָה לִבְנֵי הַגּוֹלָה. וְאִם יֵשׁ תְּקוּמָה לְזֹאת הַנְפִילָה. אָמַרְתִּי: טוֹב הַדָּבָר. וְכַאֲשֶׁר עָמַדְנוּ לְפָנָיו אָמַרְנוּ לוֹ: הַגֶּד לָנוּ שְׁאֵלָתֵנוּ. וְתִמְצָא גְמוּל טוֹבוֹתֵינוּ. אָמַר: יָשִׂים אִישׁ מִכֶּם יָדוֹ על הַחוֹל. כִּי בוֹ יִוָּדַע הַקֹּדֶשׁ מִן הַחֹל. וַנַּעֲשֶׂה כַּאֲשֶׁר צִוָּה. וְחָקַק עַל הַחוֹל טוּרִים וְהִתְוָה. וְכַאֲשֶׁר הִפִּיל נְקֻדּוֹתָיו. וְטוּרוֹתָיו. וְעָשָׂה תָכְנִיוֹתָיו. וְסָפַר סְפִירוֹתָיו. וְתָלָה כְּלִי הַנְּחשֶׁת בְּאֶצְבְּעוֹתָיו. וְשָׁקַל הַשֶּׁמֶשׁ בְּמַעְגְלוֹתָיו. וְהוֹשִׁיב הַצּוֹמָח עַל אַרְבַּע יִתְדוֹתָיו. וְחָקַק כָּל כּוֹכָב אִישׁ עַל דִּגְלוֹ בְאוֹתוֹתָיו. וְכִמְעַט כָּרְעוּ לוֹ כּוֹכְבֵי עָרֶץ1. וְיָשְׁבוּ אִתּוֹ לָאָרֶץ. וַיִשְׁתּוֹמֵם כְּפִי שָׁעָה וְהוּא חוֹשֵׁב בְּנַפְשׁוֹ . אַחֲרֵי כֵן נָשָׂא רֹאשׁוֹ. וְהוּא תָמֵהַּ עָלֵינוּ. וּמַבִּיט בְּכַעַס אֲלֵינוּ. וַיֹּאמֶר לָנוּ: נִשְׁבַּעְתִּי בְיוֹצֵר הָאוֹר הַזּוֹרַחַ. וְהַשֶּׁמֶשׁ וְהָיָרֵחַ. וְכָל מַזָּל צוֹמֵחַ, וּמְסֵב כַּדּוּר הַגַּלְגַּלִּים. כִּי אֵינְכֶם יִשְׁמְעֵאלִים. וְלֹא עֲרֵלִים . רַק מֵעַם נִבְזִים וּשְׁפֵלִים. וְאוּלַי אַתֶּם יִשְׂרָאֵלִים. אָמַרְנוּ: כֵּן דִּבַּרְתָּ. אָמַר: הֲרִימוֹתִי יָדִי לְאֵל עֶלְיוֹן. הַחוֹקֵר כָּל רַעְיוֹן הַמַּבְדִּיל בֵּין קדֶשׁ לְחֹל וּמְלַמֵּד לַעֲבָדָיו סוֹדוֹת הַכּוֹכָבִים וְהַחוֹל. כִּי שְׁאֶלְתֶּם עַל סוֹד גָּדוֹל וְהִקְשִׁיתֶם לִשְׁאוֹל. וְהֶעֱמַקְתֶּם עַד שְׁאוֹל.

שְׁאֶלְתֶּם עַל מִגְדָּל נָפַל הֲיִבָּנֶה לְתַלְפִיּוֹת וְעַל שֶׁה פְזוּרָה הֲתִמָּלֵט מִשִּׁנֵי אֲרָיוֹת. וְהִיא מִתְהַלֶּכֶת בֵּין הַחַיוֹת. שְׁאֶלְתֶּם עַל אֲנִיָה טוֹבַעַת הֲתַעֲלֶה מִבּוֹר תַּחְתִּיוֹת. וְעַל עוֹלָם חָדָשׁ הֲיִתָּכֵן לִהְיוֹת. שְׁאֶלְתֶּם עַל קִבּוּץ גָלֻיּוֹת. וְעַל חֻרְבַּן מַלְכיֻּוֹת. וְעַל הַמֵּתִים לְהַחֲיוֹת. חַי ה' כִּי בְנֵי מָוֶת אַתֶּם. אֲשֶׁר בְּחֻרְבַּן הָעוֹלָם שְׁאֶלְתֶּם. וְעַל נֵזֶק הַמְּלָכִים חֲקַרְתֶּם. וְיָד בַּמַּלְכוּת שְׁלַחְתֶּם.

אָמַר הַמַּגִיד: וַיַּקְהֵל כָּל הָעָם עָלֵינוּ הֲמוֹנִים הֲמוֹנִים. וַיֹּאמְרוּ כָל הָעֵדָה לִרְגֹם אוֹתָם בַּאֲבָנִים . אַחֲרֵי כֵן הִסְכִּימוּ לְהוֹלִיכֵנוּ אֶל הַשּׁוֹפֵט וַיִּקְחוּ אוֹתָנוּ. וְהָעָם יִסְחָבוּנוּ. עַל פָּנֵינוּ. וְעַל עֵינֵינוּ. וְהמַּכּוֹת בְּגַבֵּנוּ. וְהָרֹק בְּפָנֵינוּ. וְלֹא הִגַּעְנוּ אֶל הַשּׁוֹפֵט כִּי אִם רוּחֵנוּ בֵּין שִׁנֵּינוּ. וְהַשּוֹפֵט הָיָה מֵחֲסִידִי אֻמּוֹת הָעוֹלָם. וְכָל סוֹד מִמֶּנּוּ לֹא נֶעְלַם. וַיֹּאמְרוּ לוֹ: הִנֵּה בְנֵי נַעֲוַת הַמַּרְדוּת. שָׁלְחוּ יָד בַּמַּלְכוּת. וַיִּקָחֵנוּ הַשּׁוֹפֵט וּבִמְקוֹם סֵתֶר הֶבִיאָנוּ - וְשָׁאַל אוֹתָנוּ. עִנְיַן הַמִּקְרִים הָאֵלֶּה. וַנַּגֶד לוֹ עַל פִּי הַדְּבָרִים הָאֵלֶּה. אָמַר לָנוּ: שָׁלוֹם לָכֶם אַל תִּירָאוּ. כִּי נֵזֶק לֹא תִמְצָאוּ. וַיִּקְרָא לְעֶבֶד מֵעֲבָדָיו וַיֹּאמַר: שִׂים אֶת אֵלֶּה אֶל בֵּית הַכֶּלֶא. וְעָמַדְנוּ שָׁם כָּל הַלַּיְלָה. עַד נִתְפָּרְדָה כָּל הַקְּהִלָּה. וּכְמוֹ הַשַּׁחַר עָלָה. צִוָּה לְשַׁלְּחֵנוּ בְּחֶמְלַת בּוֹרְאֵנוּ. וַיִּשְׁלַח מַלְאָךְ וַיוֹצִיאֵנוּ . וְעתָּה נִמְלַטְתִּי מִשְׁאוֹנֵיהֶם. וּמִמַּלְתְּעוֹת הֲמוֹנֵיהֶם. בָּרוּךְ ה' אֲשֶׁר לֹא נְתָנָנוּ טֶרֶף לְשִׁנֵיהֶם.


  1. עֶרֶץ – כִּנּוּי פִּיּוּטִי לַשָּׁמַיִם. הערת פב"י.  ↩
