עִיר כְּלִילַת

סימן: ע׳יר ציון נעמת האשיגה עוזך. לחן: ידידי השכחת.


עִיר כְּלִילַת יֹפִי / מְשׂוֹשׂ כָּל הָאָרֶץ / לָךְ אִוְּּתָה נַפְשִׁי / מִפַּאֲתֵי מִזְרָח

צָעוֹק אֶצְעַק מַר וְ/אֶהֱמֶה כַיּוֹנָה / בִּרְאוֹת כִּי זִיו הוֹדֵךְ / וַהֲדָרֵךְ אָרַח

יֶחֱרַד לְבָבִי / יִתַּר מִמְּקוֹמוֹ / עַל כִּי מַלְכֵּךְ אֵין בָּךְ / מִמֵּךְ כִּצְבִי בָרַח

וְאֵיךְ אַחֲרִישׁ אֶתְ/אַפָּק בִּרְאוֹתִי דְ/רָכַיִךְ אֲבֵלוֹת / שָׁבַת עוֹבֵר אֹרַח

נְטוּשָׁה רְטוּשָׁה / בְּיַד זָרִים כְּבוּשָׁה / אֶת כָּל עֶדְיֵךְ הִצִּיל / וְרֹאשׁ נִזְרֵךְ קָרַח

נַחֲלַת אֵל עֶלְיוֹן / וְגֵיא הַחִזָּיוֹן / אֵיךְ נֶהְפַּכְתְּ לְזָרִים / לָזֹאת בְּמַר אֶצְרַח

עֵינִי נִגְּרָה, לֹא / תִדְמֶה בִרְאוֹתִי כִי / מַטֵּה עֻזֵּךְ נִשְׁבַּר / וּמַטֵּה זָר יִפְרַח

מִי יִתֵּן וְהָיָה / אֲחוֹנֵן עֲפָרֵךְ / וְאֶת אֲבָנַיִךְ / כְּמָר דְּרוֹר אָרַח

תָּאַבְתִּי לִגְוֹעַ / בַּעְפָרֵךְ לִרְגֹּעַ / כִּי גֵוִי שָׁם יָצִיץ / צִיץ וַיּוֹצֵא פֶרַח

הֲמוֹן בְּחִירַיִךְ / הֵם שׁוֹכְנֵי עֲפָרֵךְ / כָּל אֲשֶׁר גָּר אִתָּךְ / וְכָל הָאֶזְרָח

אַשְׁרֵי יִבְחַר יִקְרַב / יִשְׁכֹּן חֲצֵרָיִךְ / כִּי שׁוֹכֵן בָּךְ נְשׂוּי / עֲווֹנוֹ אֵל יִמְרַח

שָׁם יִמְצָא מָנוֹחַ / וְשָׁם יַרְגִּיעַ / כְּמוֹ בְשָׂמִים רֹאשׁ / וַאֲהָלוֹת יָרַח

יָחִיד רָם וְנִשָּׂא / בָּךְ בָּחַר וְנִסָּה / לְהַקְרִיב יְחִידוֹ / אַבְרָהָם בֶּן תֶּרַח

גַּם מָכוֹן לִזְבוּלוֹ / בָּחַר לְנַחְלָה לוֹ / וּכְבוֹדוֹ שָׁם נִגְלָה / וְאוֹרוֹ שָׁם זָרַח

הִתְנַחֲמִי צִיּוֹן / יוֹשֶׁבֶת בְּצָיוֹן / כִּי מַלְכֵּךְ יָבֹא לָךְ / וְחַסְדּוֹ לָךְ יִסְרַח

עוֹד יֶאְצֹר בָּךְ שָׂשׂוֹן / יִשְׂרָאֵל בָּךְ יֶחְסוּן / לֹא יָמֵר עוֹד רֵיחוֹ / לֹא יִבְאַשׁ וְיִסְרַח

וְלַשָּׁאטִים אוֹתָךְ / יָטַח אֶת יְמִינוֹ / תִּבְלָעֵמוֹ אֶרֶץ / כְּמוֹ עֲדַת קֹרַח

זְרִיקַת דַּם זִבְחֵךְ / וְרֵיחַ נִיחוֹחֵךְ / יַעֲלוּ לְרָצוֹן / בַּל יִהְיוּ לָטֹרַח

כָּל מִדְבָּר וְצִיָּה / יָשִׁית גַּן רְוָיָה / וְתָגֵל עֲרָבָה / וּפָרוֹחַ תִּפְרַח

עוֹד יִבְנֵךְ וְנִבְנֵית / קִרְיָה נֶאֱמֶנֶת / אוֹר שֶׁמֶשׁ לֹא יָאִיר / לָךְ וְלֹא אוֹר יָרֵחַ

לֹא יָבוֹא עוֹד שִׁמְשֵׁךְ / כִּי אֵל צוּר קְדוֹשֵׁךְ / יֹאמַר קוּמִי אוֹרִי / וּכְבוֹדוֹ לָךְ יִזְרַח.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
