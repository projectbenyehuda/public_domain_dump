מגילת שמיטה

תמונה 1

נמצאה בגנזי החורבה.

נדפסה פעם ראשונה בשושן ועתה יצאה לאור במהדורא תנינא ומתוקנת לקראה כדין וכמצוה בכל שנה של שמטה

ועם שני פרושים בעזרת החונן לאדם דעת ממני הק' קַל מִסֻּבִּין.


א

וַיְהִי בִּימֵי1 ר' דָוְצִי הוּא ר' דָוְצִי הַמּוֹלֵךְ עַל חוּרְבָה וּמְזוּזָה שִׁבְעָה וְעֶשְׂרִים וּמֵאָה וְיוֹתֵר חֲסִידִים: בַּיָּמִים הָהֵם כְּשֶׁבֶת הָרַב ר' דָוְצִי עַל כִּסֵּא רַבָּנוּתוֹ אֲשֶׁר בִּירוּשָׁלַיִם הַבִּירָה: בִּשְׁנַת עֶשֶׂר לְמָלְכוֹ עָשָׂה מִדְחֶה לְכָל הַיִּשׁוּב הֶחָדָשׁ חֵיל הַפְּרוּשִׁים וְהַחֲסִידִים הַפַּרְתְּמִים וְשָׂרֵי הַמְּדָנִים לְפָנָיו: בְּהַרְאוֹתוֹ אֶת עשֶׁר כְּבוֹד תּוֹרָתוֹ וְאֶת יְקָר תִּפְאֶרֶת חוּמְרָתוֹ יָמִים רַבִּים אִסּוּרִים בְּכָל יוֹם: וּבִמְלֹאת הָאִסּוּרִים הָאֵלֶה עָשָׂה ר' דָוְצִי לְכָל הָעָם הַנִּמְצְאִים בִּירוּשָׁלַיִם הַבִּירָה אִסּוּר שְׁמִיטָה לְשֶׁבַע שָׁנִים2 בְּחוּר הַחוּרְבָה שֶׁל יְהוּדָה הֶחָסִיד: חוּר יַרְמוּלְקָה וּתְכֵלֶת אָחוּז בְּחַבְלֵי־בוֹץ וְקַיְמָא־לָן עַל גְלִילֵי כֶסֶף וְעַמּוּדֵי שַׁס שִׁטּוֹת הָרִיא וְהָרִיד3 עַל רִצְפַּת בַּטְלָנוּת וְדוּק וְסֹחָרֶת: וְהַקְשׁוֹת מִכְּלִי יָקָר4 וְכֵלִים מִכֵּלִים שׁוֹנִים וּפִלְפּוּל יֵשׁ לָרָב כְּיַד הַמֶּלֶךְ: וַיֵּצֵא הָאִסּוּר הַשְּׁתִיָה לֹא כַדָּת אֲנִי אוֹנֵס כֵּן יִסַּד ר' דָוְצִי עִם כָּל רַבָּנֵי בֵּיתוֹ שֶׁלֹא לַעֲשׂוֹת כִּרְצוֹן אִישׁ וָאִישׁ: ס וַיְמָאֵן הָרַב קוּק5 לְמַלֵּא תּוֹרַת ר' דָוְצִי אֲשֶׁר בְּיַד הַסָּרִיסִים6 וַיְחַבֵּר וַיְפָרֵשׁ הֵיתָר לַשְׁמִיטָה וְתוֹרָתוֹ בָּעֲרָה בוֹ: וַיֹּאמֶר ר' דָוְצִי לַבַּטְלָנִים לֹא־יֹדְעֵי הָעִתִּים כִּי כֵן דְבַר ר' דָוְצִי לִפְנֵי כָּל יוֹדְעֵי דָת וָדִין: וְהַקָּרֹב אֵלָיו כַּרְסָנִי7 שׁוֹטֵר אַדְמוֹנִי בַקְשִׁישׁ מֶרֶץ חַרְסִינָה מְהוּמָן שִׁבְעַת שָׂרֵי פְרוּשִׁים וַחֲסִידִים רוֹאֵי פְּנֵי ר' דָוְצִי הַיּוֹשְׁבִים רִאשׁוֹנָה בַּמַּלְכוּת8: כְּדָת מַה לַעֲשׂוֹת בָּרַב קוּק עַל אֲשֶׁר לֹא עָשָׂה כְּמַאֲמַר הָרַב ר' דָוְצִי בְּיַד הַסָּרִיסִים: ס וַיֹאמֶר מְהוּמָן לִפְנֵי ר' דָוְצִי וְהַדַּיָנִים לֹא עַל ר' דָוְצִי לְבַדּוֹ עָוָה הָרַב קוּק כִּי עַל־כָּל־הָאֲסֵפוֹת וְעַל־כָּל הָעִתּוֹנִים אֶת עַצְמוֹ יַחַז לְרֹאשׁ: וְהַיּוֹם הַזֶּה תֹּאמַרְנָה קְהִלוֹת הַפְּרוּשִׁים וְהַחֲסִידִים לְהַבְזוֹת בַּעֲלֵי הַתּוֹרָה בְּאָמְרָן הָרַב קוּק נָאוֹר9 וּמְדַבֵּר דְבָרִים שֶׁל טַעַם וּכְדַי בִּזָּיוֹן וָקָצֶף: אִם־עַל־רַ' דָוְצִי טוֹב יֵצֵא דְבַר־הַגְּזֵרָה מִלְפָנָיו בְּכָל תֹּקֶף וְיִכָּתֵב בְּחוֹתַם הַפְּרוּשִׁים וְהַחֲסִידִים בִּטּוּל הַהֵיתָר שֶׁל קוּק וְכָל הָאֲנָשִׁים יִתְּנוּ יְקָר לְבַעֲלֵי הַתּוֹרָה לְמִגָּדוֹל וְעַד קָטָן10: וַיִּיטַב הַדָּבָר בְּעֵינֵי רַ' דָוְצִי וְהַדַּיָּנִים וַיַּעַשׂ רַ' דָוְצִי כִּדְבַר מְהוּמָן: וַיִּשָּׁלְחוּ סְפָרִים אֶל־כָּל־חֲנוּיוֹת יְרוּשָׁלַיִם אֶל חֲנוּת וַחֲנוּת כִּכְתָבָהּ וְאֶל מוֹזֵג וּמוֹזֵג כִּלְשׁוֹנוֹ11 לִהְיוֹת כָּל אִישׁ סוֹרֵר בְּבֵיתוֹ וּמִתְנַגֵּד לְעִנְיָנֵי עַמּוֹ: ס


ב

בְּעֵת הַדְּבָרִים הָאֵלֶה גִדַּל הָרַב רַ' דָוְצִי אֶת־יַדְלֶר בֶּן־הַמְּדָנִים הַדֵּימַגּוֹגִי12 וַיְנַשְּׂאֵהוּ וַיָּשֶׂם אֶת כִּסְאוֹ אֵצֶל הַבַּדַּצְ אֲשֶׁר אִתּוֹ: וַיִּבֶז בְּעֵינָיו לִשְׁלֹחַ יָד בַּשְׁתִיָּה לְבַדָּהּ וַיְבַקֵּשׁ יַדְלֶר לְהַשְׁמִיד אֶת כָּל יִשּׁוּב הַיְּהוּדִים וַעֲתִידָם13 אֶת הַסַּבּוֹן וְהַבֻּלְבּוּס14 וְאֶת הָעוֹלְשִׁין וְהַמַּמְתַּקִּים: וַיֹאמֶר יַדְלֶר לָרַב רַ' דָוְצִי יֶשְׁנוֹ עַם־אֶחָד מְפֻזָּר וּמְפֹרָד15 בֵּין הַמּוֹשָׁבוֹת וְדָתֵיהֶם לְקוּחוֹת אֵצֶל קוּק וְאֶת דָּתֵי רַ' דָוְצִי אֵינָם עוֹשִׂים וּלְרַ' דָוְצִי אֵין שׁוֶֹה לְהַנִּיחָם: אִם עַל רַ' דָוְצִי טוֹב יִכָּתֵב לְאַבְּדָם וַעֲשֶׂרֶת אֲלָפִים פְרַנְקִים־כֶּסֶף נִשְׁקֹל מִידֵי הָרִידְבָז16 לְהָבִיא אֶל גִנְזֵי הַחֲלוּקָה17: וַיֹּאמֶר רַ' דָוְצִי לְיַדְלֶר הַכֶּסֶף נָתוּן לָנוּ וְהַיִּשּׁוּב לַעֲשׂוֹת בּוֹ כַּטּוֹב בְּעֵינֶיךָ: ס וּבַלַּיְלָה הַהוּא נָדְדָה שְׁנַת רַ' חַיִּים־בֶּרְלִין וַיֹּאמֶר לְהָבִיא אֶת הַחֻמָשׁ שֶׁל אַבָּא וַיְהִי נִקְרָא לִפְנֵי בֶרְלִין: וַיִּמָצֵא כָתוּב כִּי אֵין הֵיתָר לָעֲנָבִים שֶׁל הַיְהוּדִים אֲבָל יֵשׁ הֵיתָר לִבְנֵי בִיסְמַרְק וּשְׁטֶקֶר18 הַמְבַקְּשִׁים לִשְׁלֹחַ יַד בְּקִיּוּם הַיְהוּדִים: וַיֹּאמֶר בֶּרְלִין מִי בַחוּרְבָה וְיַדְלֶר בָּא לְבֶרְלִין לְבַקֵּשׁ גַם מִמֶּנּוּ לִתְלוֹת אֶת הַיִּשּׁוּב עַל הַקּוֹץ אַשֶׁר הֵכִינוּ בְּפִלְפּוּל: וַיֹּאמֶר בֶּרְלִין לְיַדְלֶר מַהֵר קַח אֶת הַלְבוּשׁ וְאֶת הַסְּמַג19 כַּאֲשֶׁר דִבַּרְתָּ וַעֲשֵׂה כֵן לַיִּשּוּב הַיְהוּדִי אַל תַּפֵּל דָּבָר מִכֹּל אֲשֶׁר דִבַּרְתָּ: וַיִּקָּרְאוּ סוֹפְרֵי רַ' דָוְצִי בְחֹדֶשׁ תַּמוּז בְּשִׁבְעָה יָמִים בּוֹ וַיִּכָּתֵב כְּכָל־אֲשֶׁר־צִוָּה יַדְלֶר אֶל בֶּן שָׁמֶן וְאֶל רִאשׁוֹן וְאֶל גְדֵרָה וּרְחוֹבוֹת מוֹשָׁבָה וּמוֹשָׁבָה כִּכְתָבָהּ אִישׁ וְאִישׁ כִּלְשׁוֹנוֹ לְהַזִּיק לְכָל הַיְהוּדִים שֶׁל הַיִּשּׁוּב הֶחָדָשׁ וּשְׁלָלָם לָבוֹז: פַּתְשֶׁגֶן הַכְּתָב לְהִנָּתֵן דָּת בְּכָל מוֹשָׁבָה וְמוֹשָׁבָה לִהְיוֹת עַתּוּדִים הַיּוֹם הַזֶּה20: ס וּמֹרִיָּה21 יָצְאָה מִלִּפְנֵי הַבַּדַּצְ בִּלְבוּשׁ אוֹרְגַן פְּסֹלֶת וָבוּר וְתַכְרִיךְ בּוֹץ וְזֻהֲמָן22 וּמַאֲמָרֶיהָ מְלֵאִים אֶרֶס לְיִּשּׁוּב הַיְּהוּדִים: וְהָרָצִים עִם הַפֵּאוֹת יָצְאוּ דְחוּפִים בִּדְבַר רַ' דָוְצִי וְהַדָּת נִתְּנָה בִּירוּשָׁלַיִם הַבִּירָה וְרַ' דָוְצִי וְיַדְלֶר אָסְרוּ לֶאֱכֹל וְלִשְׁתּוֹת וְהַמּוֹזְגִים הִתְקַשְּׁרוּ עִם הַגֶּרְמַנִּים וְהָעִיר יְרוּשָׁלַיִם נָבוֹכָה: ס
