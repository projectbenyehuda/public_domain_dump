נִשְׁמַת אֱמוּנֶיךָ קוֹרְאִים

נשמת


נִשְׁמַת / אֱמוּנֶיךָ קוֹרְאִים בְּגָרוֹן,

יַפְצִיחוּ לְךָ שִׁיר וְשֶׁבַח וְיַנְעִימוּ בְקוֹל רֹן.

גְּאַלְתָּם מִשִּׁעְבּוּד מִצְרַיִם בְּיַד מֹשֶׁה וְאַהֲרֹן –

וְהָיָה הַיּוֹם הַזֶּה לָכֶם לְזִכָּרוֹן!


נִשְׁמַת / בְּרוּרִים עוֹבְרֵי יַם סוּף, אִישׁ וְאִשָּׁה טַף וְיֶלֶד,

שׁוֹרֲרוּךָ: ׳זֶה אֵלִי וְאַנְוֵהוּ׳ בְּשִׁירָה וּבְסֶלֶד,

אָמַרְתָּ לָהֶם: כִּי תָבֹאוּ אֶל הָאָרֶץ טַבּוּר הַחֶלֶד –

שׁוֹר אוֹ כֶשֶׂב אוֹ עֵז כִּי יִוָּלֵד.


נִשְׁמַת / רְצוּצִים יָצְאוּ חָפְשִׁים עַל יַד יְקוּתִיאֵל

אֶל אֶרֶץ זָבַת חָלָב וּדְבַשׁ, מְקוֹם מִקְדַּשׁ אֲרִיאֵל,

הֶעָנָן בַּדֶּרֶךְ עֲלֵיהֶם וּשְׁכִינָתְךָ לִפְנֵיהֶם, אֵל,

וְנַמְתָּ לָהֶם: קַדֶּשׁ לִי כָל בְּכוֹר פֶּטֶר כָּל רֶחֶם בִּבְנֵי יִשְׂרָאֵל!


נִשְׁמַת / הַמְקֹרָאִים סְגֻלַּת עַמִּי,

וּמְיַחֲדִים תָּמִיד בְּכָל יוֹם לִשְׁמִי

לִפְרֹשׂ עֲלֵיהֶם סֻכַּת שְׁלוֹמִי –

אִם כֶּסֶף תַּלְוֶה אֶת עַמִּי.

נִשְׁמַת / מְהַלְּלִים בְּקוֹל ׳קָדוֹשׁ׳ וּ׳בָרוּךְ׳ עוֹנִים,

גּוֹי מִתַּחַת וּמִמַּעַל שַׁאֲנַנִּים,

לְעוֹזֵר רִאשׁוֹנִים וּמְעוֹרֵר אַחֲרוֹנִים –

וַיֹּאמֶר יְיָ אֶל מֹשֶׁה: פְּסָל לְךָ שְׁנֵי לֻחֹת אֲבָנִים כָּרִאשׁוֹנִים!


נִשְׁמַת / חֲיָלַי, קְהַל אֱמוּנָי,

רָאוּ הַקּוֹל יוֹצֵא מִפִּי הַגְּבוּרָה עַל הַר סִינַי

בִּבְרָקִים וַעֲנָנִים לִפְנֵי הֲמוֹנַי –

כְּנָם: וַיְדַבֵּר יְיָ אֶל מֹשֶׁה בְּמִדְבַּר סִינַי.


נִשְׁמַת / זַכִּים, טְהוֹרִים מִכָּל עַם,

יָשִׁירוּ לְךָ שִׁירָה בְטוּב טַעַם,

אֲשֶׁר הוֹצֵאתָם מֵאֶרֶץ מִצְרַיִם לְעֵינֵי כָּל עָם –

וַיְהִי בְּשַׁלַּח פַּרְעֹה אֶת הָעָם.


נִשְׁמַת / קְדוֹשִׁים קוֹרְאִים לְשׁוֹכֵן רֹם וַהֲדוֹם רַגְלָיו בֶּחָלֶד,

אֲשֶׁר צָר בְּחָכְמָה צוּרַת הַוָּלֶד,

צִוָּה לְהַקְדִּישׁ בְּקָרְךָ וְצֹאנְךָ אֲשֶׁר לְךָ יִוָּלֵד,

כְּנָם: כָּל הַבְּכוֹר אֲשֶׁר יִוָּלֵד.

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
