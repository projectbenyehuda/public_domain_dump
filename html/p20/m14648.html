יֶהֱמֶה לְבָבִי עַל נְדוֹד

יֶהֱמֶה לְבָבִי / עַל נְדוֹד כְּרוּבִי / וְגוֹאֲלִי.

דַּל מְאֹד בְּשָׂרִי, / שַׁח כְּאוֹב דְּבָרִי / לְמַחֲלִי.


אָרְכוּ שְׁנוֹתַי / בְּבֵית שְׁבִי,

קָצְרוּ יְמוֹתַי / בְּעִיר צְבִי,

נִתְּנוּ נְאוֹתַי / לְאוֹיְבִי.

אוֹיְבִי סְבִיבִי, / בָּעֲרוּ שְׁבִיבִי / וְגַחֲלִי,

מֵי בְּכִי נְהָרִי / לִי וְלִי יְאוֹרִי / וְנַחֲלִי.


בּוֹעֲרִים לְבוּלָם / יְהַלֲלוּ,

יִבְעֲרוּ בְּפִסְלָם / וְיִכְסְלוּ.

צוּר, גְּעַר בְּחֵילָם / וְיִכְשְׁלוּ!

שׁוֹאֲגִים כְּלָבִיא / לִי וְנִחֲרוּ בִי / בְּהַלֲלִי

אֶת שְׁמָךְ, וְצָרִי / קָם בְּמַאֲסָרִי / לְקַלֲלִי.


רִנֲנַת רְשָׁעִים / כְּעָב מְחֵה

וַעֲדַת מְרֵעִים / כְּקַשׁ דְּחֵה.

נוֹדְדִים וְנָעִים / לְאַט נְחֵה.

אֶעֱלֹז בְּשׁוּבִי / עוֹד לְחֵיק אֲהוּבִי / בְּאָהֳלִי,

אֶשְׁכְּחָה מְרוֹרִי / שָׁם בְּשִׁיר מְרָרִי / וְקוֹל כְּלִי.


מִשְׁכְּנוֹת תְּהִלָּה / כְּאָז בְּנֵה,

מַחֲנוֹת סְגֻלָּה / קְנוֹת שְׁנֵה,

מִשְׁפְּטֵי גְאֻלָּה / לְךָ – קְנֵה!

יוֹצְרִי וְאָבִי, / צִיר שְׁלַח וְנָבִיא / לְנַהֲלִי;

שׁוּב שְׁכֹן בְּחֶדְרִי, / קוּם הֱיֵה בְעֶזְרִי / לְגָאֳלִי!

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
