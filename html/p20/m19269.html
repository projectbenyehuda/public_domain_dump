אֵין אֱלוֹהַּ מִבַּלְעָדֶיךָ

מי כמוך


אֵין אֱלוֹהַּ מִבַּלְעָדֶיךָ / וּמַעֲשֵׂה יָדֶיךָ עֵדֶיךָ,

וּלְאֵין חֵקֶר חֲסָדֶיךָ / וְצִדְקָתְךָ כְּגַלֵּי הַיָּם.

בְּטֶרֶם עֲגֻלָּה וְחֶלְדָּהּ, / בַּחָכְמָה נוֹעַצְתָּ לְבַדָּהּ –

אֲרֻכָּה מֵאֶרֶץ מִדָּה / וּרְחָבָה מִנִּי יָם.

רַבּוֹת עָשִׂיתָ וַעֲצוּמוֹת, / נִלְאוּ לְחָקְרָם הַנְּשָׁמוֹת –

אִלּוּ מָלְאוּ חָכְמוֹת, / כַּמַּיִם יְכַסּוּ עַל יָם.

הַשָּׁמַיִם יְמִינְךָ נָטְתָה / וְהָאָרֶץ יָדְךָ עָשְׂתָה,

וְתֹהוּ וָבֹהוּ הָיְתָה / כְּדֻמָה בְּתוֹךְ הַיָּם.

מֵאוֹרְךָ אוֹר נִבְרָא / וּבִדְבָרְךָ יוֹם נִקְרָא,

רֹאשׁוֹ מִמִּזְרָח יֵרָא / וְאֶת עֲקֵבוֹ מִיָּם.

בֵּינוֹת הַמַּיִם, לְהוֹדִיעַ / פִּלְאֲךָ, הִבְדַּלְתָּ וַתַּרְגִּיעַ –

חֶצְיָם מִמַּעַל לָרָקִיעַ / וְחֶצְיָם אֶל הַיָּם.

רַמְתָּ וּדְבָרְךָ גָאָה / בִּגְבוּרָה מִכֹּל נִפְלָאָה,

יַבֶּשֶׁת הָאָרֶץ הֶרְאָה / וַיָּסֶךְ בִּדְלָתַיִם יָם.

בַּיַּבָּשָׁה צָמְחוּ דְּשָׁאִים / לָתֵת מִחְיַת נִבְרָאִים,

לְהַשְׁקוֹתָם יַעֲלֶה נְשִׂיאִים / הַקּוֹרֵא לְמֵי יָם.

יִשְׁתַּחֲווּ לוֹ עֲבָדִים / שֶׁמֶשׁ וְיָרֵחַ וּגְדוּדִים,

בְּגַלְגַּל עוֹלֶה מִקָּדִים / וְנָסַב לִפְאַת יָם.

מַחְשְׁבוֹת אֵל מָה אָמְצוּ / וְחֻקּוֹתָיו לַכֹּל חָרְצוּ,

וּמִמַּיִם בַּחֲמִישִׁי שָׁרְצוּ / צִפּוֹר שָׁמַיִם וּדְגֵי הַיָּם.

אֲדָמוֹת שָׂשׂוּ בְהוֹלִידָם / בְּהֵמוֹת, בֶּעָפָר יְסוֹדָם;

וַיֹּאמֶר: ׳נַעֲשֶׂה אָדָם, / וְיֵרְדְּ מִיָּם עַד יָם.׳

יָצָר מִצְּלָעָיו עַלְמָה / הַמַּאֲכִילָה עֵץ בְּעָרְמָה,

הַשָּׂמָה הַנּוֹשֶׁבֶת נְשַׁמָּה, / הַמַּחֲרֶבֶת יָם.

רִמָּם נָחָשׁ, וְנָגְעוּ / בָעֵץ – וְקוֹל שָׁמְעוּ,

דֹּק וְחוּג הִתְרוֹעֲעוּ / מִמִּזְרָח וּמִיָּם:

׳עַתָּה, הַנֶּחְבָּאִים, אִמְרוּ: / הֲלֹא כֹל מֵאִתִּי נוֹצְרוּ –

אֶרְאֵם, וְאִם יִסָּתְרוּ / מִנֶּגֶד עֵינַי בְּקַרְקַע הַיָּם.׳

זְדוֹנָם הוֹדוּ בָאַחְרוֹנָה: / חָנֵּנוּ, חָנֵּנוּ, שׁוֹכֵן מְעוֹנָה,

וְהַעֲבֵר חַטָּאת רִאשׁוֹנָה / וְתַשְׁלִיךְ בִּמְצֻלּוֹת יָם!׳

רוּמָם – לָאָרֶץ חִלֲלוֹ / וּבַגַּן הִשְׁכִּין חֵילוֹ

מִקֶּדֶם, וַיֹּאמֶר לוֹ: ׳נָא הַבֵּט דֶּרֶךְ יָם!׳

הוֹרֵג אָח בַּשָּׂדֶה / בִּקֵּשׁ בַּמֶּה יִפָּדֶה,

וַיֹּאמֶר: עֲוֹנִי אוֹדֶה, / כִּי גָדוֹל כַּיָּם.׳

בִּרְבוֹת הָאָדָם נִסְכְּלוּ, / דַּרְכָּם הִשְׁחִיתוּ וְנוֹאֲלוּ,

הַשְׁקֵט לֹא יוּכְלוּ – / וְהָרְשָׁעִים כַּיָּם.

מַבּוּל שְׁטָפָם וְגָוְעוּ, כִּי מֵימֵי זְבוּל נָגְעוּ,

וּמַעְיְנוֹת תְּהוֹם נִבְקְעוּ – / וַיְהִי סַעַר גָּדוֹל בַּיָּם.

דּוֹרְכֵי דֶרֶךְ תְּמִימָה / נִמְלְטוּ בִּסְפִינָה בְּחָכְמָה,

וּלְשָׁנָה יָבְשָׁה הָאֲדָמָה / וְנִשְּׁתוּ מַיִם מֵהַיָּם.

יָלְדוּ בָנִים כְּחֶפְצָם, / וּבִבְנוֹת מִגְדָּל לְרִבְצָם –

מִשָּׁם אֱלֹהִים הֱפִיצָם / וְנָעוּ מִיָּם עַד יָם.

נָגַהּ אָב, וּבְשִׂכְלוֹ / יָצָא עַל פִּי מְחוֹלְלוֹ

מֵאַרְצוֹ, וַיֵּט אָהֳלֹה / בֵּית אֵל מִיָּם.

תְּפָשׂוּהוּ לָאוּר זָרִים / וְנִמְלָט, וְהָיוּ אוֹמְרִים:

׳עַל כֵּן בָּאֻרִים / כַּבְּדוּ יְיָ בְּאִיֵּי הַיָּם!׳

רָעֲשׁוּ שְׁמֵי מְעוֹנוֹ / בִּשְׁלֹחַ יָד בִּבְנוֹ,

וּמְצוּקֵי אֶרֶץ לְמַעֲנוֹ / קוֹלָם כַּיָּם.

וַיִּשְׁמַע: ׳הֶרֶף יָדְךָ / מֵאֵת בִּנְךָ יְחִידְךָ!

עֵקֶב זֹאת – אַרְבֶּה הוֹדְךָ, / וְשַׂמְתִּי אֶת זַרְעֲךָ כְּחוֹל הַיָּם!׳

מָרוֹם, רַב עֲלִילִיָּה, / שָׁלַח נָזִיר לְמִחְיָה,

בְּשִׁבְעִים יָרְדוּ, וְהָיָה / מִסְפַּר בְּנֵי יִשְׂרָאֵל כְּחוֹל הַיָּם.

הֶעֱבִידָם תַּנִּין בְּמִרְדּוֹ – / וְלִפְדּוֹתָם שָׁלַח עַבְדּוֹ

נֹטֶה שָׁמַיִם לְבַדּוֹ / וְדוֹרֵךְ עַל בָּמֳתֵי יָם.

חֲנֵסִים לִשְׁמֹעַ מֵאֲנוּ / וּבְעֶשֶׂר מַכּוֹת נִבְחֲנוּ,

וִידִידִים נָסְעוּ וַיַּחֲנוּ / בֵּין מִגְדֹּל וּבֵין הַיָּם.

בֵּן נָשָׂא עֵינָיו – / וְהִנֵּה סְבִיבוֹתָיו עוֹנֲנָיו,

בַּיַּבָּשָׁה הֶעֱבִירוֹ עָנָיו / וּמַטֵּהוּ עַל הַיָּם.

רָדַף אַחְרֵיהֶם פַּרְעֹה, / וּבְסוּר אוֹפַנֵּי יִשְׁעוֹ –

חָשַׂף אֵל זְרוֹעוֹ / וְהָרַג אֶת הַתַּנִּין אֲשֶׁר בַּיָּם.

וְיִשְׂרָאֵל נִפְלָאִים רָאָה, / וַיָּשָׁר שִׁירָה נָאָה:

׳אְשִׁירָה לַייָ, כִּי גָאֹה גָּאָה, / סוּס וְרֹכְבוֹ רָמָה בַּיָּם!׳

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
