בָּטַחְתִּי בְאֵל נוֹרָא

מאורה


בָּטַחְתִּי בְאֵל נוֹרָא – / מִי זֶה יַרְשִׁיעֵנִי?

אֲדֹנָי לִי לֹא אִירָא – מִי אִישׁ יַכְנִיעֵנִי?

בְּאָהֳלָיו אָגוּרָה / וּבָהֶם אֵל יִטְּעֵנִי,

אֲנִי אֶל אֱלֹהִים אֶקְרָא – / וַאֲדֹנָי יוֹשִׁיעֵנִי.


אֵל מְיֻחָד בָּאִיִּים, / אַל תִּבְזֶה עֱנוּת עָנִי,

וְעֵת אֶקְרָא בְּעַד דְּחוּיִים – / אַל תַּעְלֵם מִמֶּנִּי

וְשִׂימֵנִי לְרֹאשׁ פְּדוּיִם / וּבֶטַח תּוֹשִׁיבֵנִי,

וּמַעֲנֶה רַךְ כְּשִׁיר לְוִיִּים / בִּינָה, אֵל, וְחָנֵּנִי

וְדֶרֶךְ אֹרַח חַיִּים / בְּלֶכְתִּי תּוֹדִיעֵנִי.


בָּרְחוּ כָּל טוֹבוֹתַי / כִּי אָרְכוּ יְגוֹנוֹתַי

וְסָפוּ בְהֶבֶל שְׁנוֹתַי / וּבַאֲנָחָה כָּל יְמוֹתַי,

וַיִּלְכְּדוּ עֲוֹנוֹתַי / אֶת רַגְלַי בְּחַטֹּאתַי.

יוֹדֵעַ עֲלֻמֹתַי / הַמֵּבִין מַחְשְׁבוֹתַי,

דְּלֵנִי מִמְּצוּקוֹתַי / וּמִתּוֹךְ תֹּשִׁיעֵנִי.


רַעְיוֹנַי יְעִירוּנִי / וְיֹאמְרוּ לִי: עָצֵל עוּרָה!

וּסְעִפַּי יוֹרוּנִי / מִיּוֹם מִשְׁפָּט נוֹרָא,

וְנוֹגְשֵׂי חֵטְא יוֹרוּנִי / כְּחִצִּים לַמַּטָּרָה

וְלַבֹּקֶר יְשַׁחֲרוּנִי / כִּי דִבַּרְתִּי סָרָה –

אוֹי לִי כִּי אֲסָרוּנִי / וְחַטֹּאתַי יְגִיעוּנִי.


הֵעַזְתִּי אֶת מִצְחִי / וְשָׁכַחְתִּי צוּר גּוֹחִי

וַיַּז דָּם עַל מִצְחִי / וְהִכָּה הַלֶּחִי.

לְזֹאת אֶשְׁפֹּךְ רֹן שִׂיחִי / לִפְנֵי עֹז מִבְטַחִי,

אוּלַי אֶמְצָא לְרוּחִי / מְנוּחָה בְּהִשְׁתּוֹחֲחִי,

בְּהָקִיצִי מִמְּנוּחִי / טוּבְךָ תַּשְׂבִּיעֵנִי.


מָה אָשִׁיב וּמָה אֶעֱנֶה / וּמַה לִּי, דְּבָרַי לָעוּ,

כִּי חִצֵּי שֹׁכְנִי סְנֶה / עִמָּדִי נוֹעָדוּ.

סְלַח, אֵל מַעֲלוֹת בּוֹנֶה, / לְאִישׁ מַעְגְּלוֹתָיו נָעוּ,

רַע אֵלָיו לֹא יְאֻנֶּה / וּבְאוֹרְךָ הָאִיר עֵינִי.

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
