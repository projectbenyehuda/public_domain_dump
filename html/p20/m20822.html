וּבְכֵן אֶשְׁפֹּךְ לִפְנֵי אֵל

רהיט


וּבְכֵן אֶשְׁפֹּךְ לִפְנֵי אֵל לְבָבִי / בְּזָכְרִי דִבְרֵי אֵלִיָּהוּ הַנָּבִיא.


אֵל, לְשַׁוְעִי הֱיֵה קָרוֹב / וּלְטוֹב עַבְדְּךָ עֲרֹב / בִּפְנוֹת הַיּוֹם לַעֲרֹב.

אָז נָסַע עַם אֵל – / וַיִּשְׁלַח מַלְאָכִים אֶל / כָּל גְּבוּל יִשְׂרָאֵל.

נִקְבְּצוּ נְבִיאֵי תֹהוּ / לַחֲקֹר אֱלוֹהַּ מִי הוּא; / וַיֹּאמֶר אֵלִיָּהוּ:

׳יַחֲדוּ אֵל, נֶאֱלָחִים, / וְעַל סְעִיפִים מִתְנַצְּחִים / עַד מָתַי אַתֶּם פֹּסְחִים?׳

אָז, בְּשָׁמְעָם דְּבַר נֹעַם, / נִלְאוּ לְהָשִׁיב טַעַם – / וְלֹא עָנוּ הָעָם.

׳בִּינוּ סוֹד מַחֲשָׁבִי, / כִּי לֵאלֹהֵי אָבִי / אֲנִי נוֹתַרְתִּי נָבִיא.׳

רַבּוּ, בְּאֶפֶס יְשָׁרִים, / נְבִיאֵי אֱלֹהִים אֲחֵרִים; / וַיִּתְנוּ לוֹ שְׁנַיִם פָּרִים:

׳הָאֶחָד יְהִי לָכֶם / וְאֵשׁ לֹא תְשִׂימוּן כְּדַתְכֶם / וּקְרָאתֶם בְּשֵׁם אֱלֹהֵיכֶם.

עֲשׂוּ רִאשׁוֹנָה יַחַד / וְקִרְאוּ בְשֵׁם הַנִּכְחָד – / בַּחֲרוּ לָכֶם הַפַּר הָאֶחָד.

זֹאת תַּעֲשׂוּ – וְכֵן אֶפְעַל!׳ / וּפַר עָשׂוּ מְתֵי מַעַל – / וַיִּקְרְאוּ בְשֵׁם הַבַּעַל:

׳רְאֵה, מִקְּרֹא גְרוֹנֵנוּ / נִחַר, בְּאֵשׁ חָנֵּנוּ! / הַבַּעַל, עֲנֵנוּ!׳

אָז הַנָּבִיא בְשִׂכְלוֹ / הֵתֵל בְּמִלּוּלוֹ: / ׳כִּי שִׂיחַ וְכִי שִׂיג לוֹ.׳

בְּקוֹל רָם כֻּלָם צוֹרְחִים, / מִתְגּוֹדְדִים נֶאֱנָחִים / בַּחֲרָבוֹת וּבָרְמָחִים.

רָאָה כִּי אֵין שׁוֹמְעָם / קְרָאָם מַשְׂכִּיל טַעַם – / וַיִּגְשׁוּ כָל הָעָם.

מִזְבֵּחַ הָרוּס לְפָנִים / אָז רִפֵּא בְמִנְיָנִים / שְׁתֵּים עֶשְׂרֵה אֲבָנִים.

אֱמוּנַת שַׁדַּי נַעֲלָה / בְּיֹשֶׁר פָּעֳלוֹ גִּלָּה – / וַיַּעַשׂ תְּעָלָה,

יַעֲרֹךְ וְנִתַּח וְשָׂח: ׳פְּנוּ, / וְאַרְבָּעָה כַדִּים תְּנוּ!׳ / וַיֹּאמֶר: ׳שְׁנוּ! וַיִּשְׁנוּ,

רָצוּ וְשִׁלְּשׁוּ כְאֵלֶּה / וּמַיִם – הָחֵל וְכַלֵּה, / וְגַם אֶת הַתְּעָלָה מִלֵּא.

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
