[זְמָן הִכָּה בְּחֵץ שָׁנוּן לְבָבִי] (תְּלֻנָּה עַל־הַזְמָּן)

Leone Ebreo או Jehuda Abravanel

[נולד סמוך לשנת 1460 בלִיסַבּוֹן, הלך בשנת 1492 לאיטליה וחי בנַפּוֹלִי גֶּ’נוֹבָה ובַּרְלַטָה בפלך אַפּוּלְיָה ושוב בנַפּוֹלִי. מת אחרי 1521]


תְּלֻנָּה עַל־הַזְמָּן

ں – – – ں – – – ں – –

זְמָן הִכָּה בְּחֵץ שָׁנוּן לְבָבִי / וְכִלְיוֹתַי יְפַלֵּחַ בְּקִרְבִּי

וְהִכַּנִי וּמַכָּתוֹ אֲנוּשָׁה / וְדִכַּנִי וְשָׂם נֶצַח כְּאֵבִי

פְּצָעַנִי וּבִלָּה אֶת־בְּשָׂרִי / וְאָכַל הַכְּאֵב דָּמִי וְחֶלְבִּי

וְגֵרָם כָּל־עֲצָמַי בַּחֲמָתוֹ / וְהִתְנַשֵּׂא וַיָּקָם־בִּי כְּלָבִיא

וְלֹא דַי כִּי צְנָפַנִי צְנֵפָה / וְשָׂמַנִי מְגֹרָשׁ נָד בְּאִבִּי

וְכַשִּׁכּוֹר הֱנִיעָנִי בְּתֵבֵל / וְהוּא עַל־פַּאֲתֵי אֶרֶץ מְסִבִּי

וְהֵן זֶה־לִּי כְּעֶשְׂרִים מִשְּׁנוֹתַי / אֲשֶׁר לֹא שָׁקְטוּ סוּסַי וְרִכְבִּי

וּמָדַד שָׁעֳלִי מַיִם וְעַפְרוֹת / אֲרָצוֹת כָּל וְהֵם כִּלּוּ אֲבִיבִי

וְהִבְרִיחַ יְדִיד מֶנִּי וְהֶגְלָה / בְּנֵי־גִילִי וְהִרְחִיק עַם קְרוֹבִי

וְלֹא אֶרְאֶה וְלֹא אֵדַע מְיֻדָּע / וְאֶת־אִמִּי וְאֶת־אַחַי וְאָבִי

וּפִזֵּר מַחֲמַדַּי זֶה לְצָפוֹן / וְזֶה קֵדְמָה וְזֶה אֶל־מַעֲרָבִי

לְבַל עוֹד יוּכְלוּ הַשְׁקֵט זְמָמַי / וְלֹא אֶמְצָא מְנוּחָה גַּם בְּחָשְׁבִי

וּבִהְיוֹת מֻעֲדוֹת פָּנַי לְמִזְרָח / פְּרִידָתָם תְּשׁוּפֵנִי עֲקֵבִי

וְרַגְלִי תֵט וְלִבִּי נָד וְנוֹדֵד / וְלֹא אֶבְחַן בְּלֶכְתִּי אוֹ בְשׁוּבִי

הֲלֹא כִלָּה זְמָן לִבִּי וְשִׁסַּע / וְנִתְּחָהוּ כְדֻבִּי אוֹ זְאֵבִי

וְהִכּוּהוּ אֲנָחוֹת עִם בְּהָלוֹת / בְּשֹׁד שׁוֹדֵד וּבִזַּת הוֹן וּבִשְׁבִי

בְּעֵינָיו זֶה מְעַט מִזְעָר וּבִקֵּשׁ / לְכַלֵּנִי לְכָבּוֹת אֶת־שְׁבִיבִי

וְהוֹלַדְתִּי שְׁנֵי בָנִים זְכָרִים / בְּנֵי־חֶמֶד בְּהוֹד וִיקָר וּבִצְבִי

וְהַקָּטָן קְרָאתִיהוּ שְׁמוּאֵל / וְגָבָהוּ זְמָן נוֹגְשִׂי וְאוֹרְבִי

וְהִכָּהוּ לְחָמֵשׁ מִשְּׁנוֹתָיו / וְגָדַל בַּעֲדוֹ רָגְזִי וְעָצְבִּי

וְהַגָּדוֹל אֲשֶׁר יִצְחָק קְרָאתִיו / אֲבַרְבַּנְאֵל כְּמוֹ צוּר מַחֲצָבִי

כְּשֵׁם גָּדוֹל בְּיִשְׂרָאֵל כְּהוֹרוֹ / כְּבֶן־יִשַׁי וְהוּא נֵר מַעֲרָבִי

בְּהִוָּלְדוֹ רְאִיתִיו טוֹב לְשַׁכֵּן / בְּלִבּוֹ בִּין וְטוּב הוֹרָיו וְטוּבִי

וּבֵן־שָׁנָה אֲהָהּ מִנִּי הֲדָפוֹ / וְהִדִּיחוֹ זְמָן שׂוֹנְאִי וְעוֹקְבִי

בְּעֵת גֹּרְשׁוּ בְּנֵי גָלוּת סְפָרַד / יְצַו מֶלֶךְ לְהַצִּיב מַאֲרָבִי

לְבַל אֵצֵא וְאֶעְבֹר בֵּין גְּזָרִים / וְלָקַחַת בְּנִי יוֹנֵק חֲלָבִי

לְהַכְנִיסוֹ בְדָתוֹ בַּעֲבוּרוֹ / וְגִלָּה אָזְנִי אִישׁ טוֹב אֲהוּבִי

שְׁלַחְתִּיו עִם מְנִיקָתוֹ בְּאִישׁוֹן / חֲצוֹת לַיִל כְּאִלּוּ הוּא גְנוּבִי

לְפוֹרְטוּגַל וְשָׁם מֶלֶךְ מְגֹאָל / אֲשֶׁר הָיָה לְפָנִים מַחֲרִיבִי

וְשָׁם גָּדַל כְּבוֹד אָבִי וְעָשְׁרוֹ / בְּחֵי אָבִיו וְהוּא מַלְכִּי חֲשׁוּבִי

וְזֶה קָם זֵד בְּלִיַּעַל וְאַכְזָר / וּבוֹצֵעַ וְאִישׁ חֶמְדָּן וְכַלְבִּי

וְעֵת קָשְׁרוּ סְגָנָיו בּוֹ וְאָחִיו / הֲלֹא אָבִי בְּקוֹשְׁרָיו שָׂם בְּכָזְבִי

וּבִקֵּשׁ בַּהֲרֹג אָחִיו לְהָרְגּוֹ / וְהִצִּיל נַפְשׁוֹ רוֹכֵב כְּרוּבִי

וְהוּא נִמְלַט לְקַשְׂטִילְיָה אֲשֶׁר שָׁם / מְקוֹם הוֹרַי וְשָׁמָּה בֵּית חֲצִיבִי

וְאַז שָׁלַל רְכוּשִׁי עַד־בְּלִי־דָי / וּבַז טוּבִי וְכַסְפִּי עִם זְהָבִי

וּבִרְאוֹתוֹ בְנִי כִּי בָא לְאַרְצוֹ / וּבֵית אָבִי בְּאִיטַלְיָה בְחָשְׁבִי

עֲצָרָהוּ לְבַל יָשׁוּב וְצִוָּה / לְבַל יִהְיֶה לְנִדָּחִי מְשִׁיבִי

בְּמוֹתוֹ אַחֲרָיו מֶלֶךְ כְּסִיל קָם / וְהוּא אָדוּק בְּדָתוֹ אִישׁ נְבוּבִי

וְאִנֵּס כָּל־עֲדַת יַעְקֹב וְהֶעְבִיר / עֲלֵי־דָת כָּל־בְּנֵי עַמִּי נְדִיבִי

וְרַבִּים הָרְגוּ עַצְמָם לְבִלְתִּי / עֲבֹר תּוֹרַת אֱלֹהִים מַרְחִיבִי

וְלֻקַּח מַחֲמַד נַפְשִׁי וְהוּמַר / שְׁמוֹ הַטּוֹב כְּפִי צוּרִי נְקוּבִי

כְּבָר הוּא בֶּן־שְׁתֵּים עֶשְׂרֵה עֲדֶן לֹא / רְאִיתִיו בַּעֲוֹן בִּצְעִי וְחוֹבִי

וְאָהִימָה וְעַל־רֹאשִׁי חֲמָסִי / וְעֶל־נַפְשִׁי תְלֻנָּתִי וְרִיבִי

הֲנִיסוֹתִיו אֱלֵי־פַח מִפְּחָדִים / וּמֵרֶשֶׁף שְׁלַחְתִּיו תּוֹךְ לְהָבִי

אֲקַוֵּהוּ וְתוֹחַלְתִּי מְשׁוּכָה / וְלָמָּה תֶּאֱחַר עֹפֶר אֲהוּבִי

וְלָמָּה זֶּה תְּדַכֵּא לֵב יְלָדְךָ / וְתִהְיֶה חֵץ בְּכִלְיוֹתַי כְמֵבִיא

וְתַחְשִׁיךְ בַּעֲרִיפַי אוֹר וְתָשִׂים / זְרִחָתוֹ לְמוּל עֵינַי כְּעַרְבִּי

וְסַהַר יֶחֱשַׁךְ תָּמִיד בְּפָנַי / וְכוֹכָבִי יְשַׁו נִסְתָּם בְּעָבִי

וְלֹא יָאִיר שְׁבִיב חַרְסִי מְעוֹנִי / וְלֹא יִזְרַח בְּאֶשְׁנַבִּי וְעֻבִּי

וְלֹא תִפְרַח חֲבַצֶלֶת שְׁרוֹנִי / וְלֹא יֵרֵד עֲלֵי עֶשְׂבִּי רְבִיבִי

וְתִגְזֹל אֶת־שְׁנַת עֵינַי בְּזִכְרָךְ / וְלֹא אֵדַע בְּעֵת קוּמִי וְשִׁכְבִי

וְלֹא אֶטְעַם בְּמַאְכָלִי וְנֹפֶת / לְחִכִּי מַר וְסַף־רַעַל עֲרֵבִי

בְּעֹנִי אֶלְחֲמָה עֻגַת רְצָפִים / וּבִבְכִי אֹכְלָה לַחְמִי חֲרֵבִי

בְּדִמְעָה אֶמְזְגָה מֵימַי וְאֶשְׁתֶּה / וְלֹא יָבוֹא בְּמוֹ פִי דַּם עֲנָבִי

וְהָיִיתִי שְׁתוּי מַיִם וְנָזִיר / עֲדֵי כִי חִשְּׁבוּנִי בֶּן־רְכָבִי

וְעֵת כִּי אֶחֱלֹם שׁוּבְךָ וְאֶרְאֶה / תְמוּנָתְךָ בְּעֵינֵי מַחֲשָׁבִי

חֲבָלִים נָפְלוּ־לִי בַנְּעִימִים / וּפָנַי לָבְשׁוּ מַרְאֵה צְהֻבִּי

וְיָשַׁנְתִּי וְעָרְבָה לִי שְׁנָתִי / הֱקִיצוֹתִי שְׂבַע רָצוֹן וְצָבִי

וְשָׁתִיתִי וּמֵימַי לִי עֲרֵבִים / וְנִמְתַּק לִי בְּחֶבְרָתְךָ רְגָבִי

הֲכִי עֵת אֶזְכְּרָה נוּדְךָ כְּמוֹ־אֵשׁ / לְבָבִי חַם וְיַכֵּנִי שְׁרָבִי

וְהָיִיתִי כְּאִישׁ נִדְהָם וְנִרְדָּם / שְׁפַל קוֹמָה קְצַר אָרְכִּי וְרָחְבִּי

בְּזִכְרֶךָ שְׂשׂוֹנִי עִם יְגוֹנִי / צֳרִי אַתָּה וְאַתְּ צָרִי וְאוֹיְבִי

וְצוּרָתְךָ עֲלֵי לִבִּי חֲרוּתָה / וּמֶרְחָקְךָ בְּתוֹךְ לִבִּי חֲטוּבִי

וְיוֹתֵר מִהְיוֹת לִי מַעֲלִיזִי / דְּמוּתֶךָ נְדוֹדֶךָ מְרִיבִי

פְּרִידָתְךָ תְּסַכֵּל אֶת־עֲצָתִי / גְּרוּשְׁךָ סָךְ וְעִוָּה אֶת־נְתִיבִי

וּבַעֳבוּרְךָ גְּאוֹן נַפְשִׁי מְאֹד שַׁח / וְשָׁפֵל בַּעֲבוּרֶךָ רְהָבִי

עֲדֵי שִׁקְמִים גְּבֹהִים מִבְּרוֹשִׁי / עֲדֵי רָם עַל־אֲרָזַי עֵץ אֲזוֹבִי

עֲדֵּי גָבַהּ עֲלֵי נִצִּי עֲטַלֵף / וְעַל־כַנְפֵי נְשָׁרַי עָף זְבוּבִי

וְכִנְעָרַי יְצוּרַי נֶחֱלָשִׁים / וְגָבַר על־אֲרָיוֹתַי כְּשָׂבִי

וְאֶמְאַס שִׁיר וְשָׁבַרְתִּי נְבָלַי / וְכִנּוֹרַי תְּלִיתִים עַל־עֲרָבִי

וְהָפַכְתִּי לְקוֹל בּוֹכִים זְמִירִי / וְעוּגָבִי בְּמַכְאוֹבִי כְּאוֹבִי

עֲגוּרִי נֶהֱפַךְ קִפּוֹד וְתוֹרִי / כְּתַנִּים שָׁב וְיוֹנָתִי כְּעוֹרְבִי

וְתִעַבְתִּי בְהֵיכְלֵי הַמְּלָכִים / וְתָאַבְתִּי שְׁכֹן מִדְבַּר עֲרָבִי

בְּנִי פֶּרֶץ גְּרוּשְׁךָ יִפְרְצֵנִי / יְשׁוּפֵנִי וְשָׂם פַּחִים סְבִיבִי

וְשָׂם מֹרֶךְ בְּתוֹךְ לִבִּי וְרֹגֶז / בְּמוֹרָשַׁי וּבַעְצָמַי רְקָבִי

וְאֶשְׁמַע אִמְּךָ כָּל־יוֹם מְבַכָּה / בְּקָרְאָהּ לָךְ "יְדִיד לִבִּי רְטֻבִּי

וּמִי זֶה מֵעֲלֵי חֵיקִי גְּנָבְךָ / וּמִי שָׂמְךָ פְּרִי בִטְנִי עֲזוּבִי"

וְכִי לֹא אֶעֱצֹר כֹּחַ לְסָבְלָה / וְלֹא לִטְמֹן דְּאָגָתִי בְחֻבִּי

עֲזַבְתִּיהָ וְהָלַכְתִּי לְשָׁרֵת / פְּנֵי מַלְכִּי נְתָנוֹ אֵל מְטִיבִי

וְכֵן נַדְתִּי וְלֹא נַחְתִּי וְנַעְתִּי / וְאֶשְׁכֹּן בֵּין אֱדוֹם תּוֹךְ עַם לְהָבִי

וְלֹא אֶמְצָא מְזוֹרִי אֶל־מְזוֹרִי / וּמִי יַיְשִׁיר זְמָן מַתְעִי וְעוֹלְבִי

וּמָאַסְתִּי יְמֵי עָמָל וְלֵילוֹת / וּמֵהֶם מָוְתָה אֶבְחַר וְאָבִי

וְחַיִּים כָּבְדוּ עָלַי וְיָמִים / כְּחוֹל יַמִּים עֲלֵי שִׁכְמִי וְגַבִּי

וּמַה בֶּצַע בְּחַיֵּי הַיְגוֹנִים / וְהַגִּיעַ אֱלֵי קִצִּי קְצוּבִי

לְמַר־נֶפֶשׁ הֲלֹא חַיִּים תְּמוּתָה / וְיֶשׁ־לִי רָב וּמִעוּטִי כְרֻבִּי

וּמָה אוֹחִיל לְרֹב יָמִים וְשָׁנִים / רְאוֹת טוֹבָה וְדֹב שַׁכּוּל בְּאוֹרְבִי

יְלִידֵי יוֹם בְּנֵי אַשְׁפָּה וְקַשְׁתָּם / בְּרוּם שַׁחַק זְמָן מוֹרִי וְרוֹבִי

וְהוּשַׂמְתִּי לְמַטָּרָה אֲלֵיהֶם / וְהֵמָּה סוֹבֲבִים עַל־חוּג קֳטָבִי

וְאָשׁוּבָה לְדַבֵּר אֶל־יְחִידִי / לְבִלְתִּי יִהְיֶה עוֹד מַאֲדִיבִי

בְּכוֹרִי שִׂים לְבָבְךָ דַּע הֲכִי בֶן / חֲכָמִים אַתְּ מְחֻכָּמִים כְּנָבִיא

וְהַחָכְמָה יְרֻשָּׁה לָךְ וְאַל־נָא / תְּאַבֵּד עוֹד יְמֵי יַלְדוּת חֲבִיבִי

רְאֵה עַתָּה בְּנִי חִמּוּד לְלִמּוּד / קְרֹא מִקְרָא וְהָבֵן מִכְּתָבִי

שְׁנוֹת מִשְׁנָה לְמֹד תַּלְמוּד בְּמִדּוֹת / שְׁלשׁ עֶשְׂרֵה וּפֵרוּשׁ עִם מְתִיבִי

וְאֵיךְ אוּכַל לְהִתְאַפֵּק בְּעָבְרוֹ / וְזֶה חָלְיִי וְחַרְחוּרִי וְחַרְבִּי

וְזֶה יִקְרַע בְּתַעַר הַשְּׂכִירָה / סְגוֹר לִבִּי וְלֹא תַעַר גְּלָבִי

וְזֶה יָבִיא בְלִבָּתִי אֲמוּלָה / בְּרַק חַרְבּוֹ בְּתוֹךְ בִּטְנִי וְקִרְבִּי

לְמִי זֶה אֶמְסְרָה רֻבֵּי תְבוּנוֹת / לְמִי אַשְׁקֶה עֲסִיס גִּתִּי וְיִקְבִי

וּמִי יִטְעַם וְיֹאכַל אַחֲרֵי עֵת / פְּרִידָתִי פְּרִי דָתִי וְכָתְבִי

וּמִי יָבִין סְפוּנֵי בִין בְּסִפְרֵי / תְעוּדוֹת מַעֲשֵׂה אָבִי נְצִיבִי

וּמִי יִשְׁאַב בְּצָמָא מֵי מְקוֹרוֹ / וּמִי יִשְׁתֶּה בְּצָיוֹנִי וְחָרְבִּי

וּמִי יִלְקֹט לְאַט פִּרְחֵי נְטָעַי / וְעֵץ־דַּעַת יְהִי אוֹרִי וְחוֹטְבִי

וּמִי יַעְצֹר וְיַשְׁלִים מַחְבְּרוֹתַי / וּמִי יֶאְרֹג שְׁתִי בִגְדִּי וְעֶרְבִּי

וּמִי יַעְדֶּה עֲדִי דָתִי בְּמוֹתִי / וּמִי יִרְכַּב עֲלֵי פִרְדִּי וְצָבִי

לְבַד אַתָּה יְדִיד נַפְשִׁי וְיוֹרְשִׁי / אֲנִי חַיָּב לְצוּרִי אַתְּ עֲרֵבִי

לְךָ צָמֵא לְךָ כָּמַהּ בְּשָׂרִי / בְּךָ אַשְׁקֶה צְמָאִי אַף רְעָבִי

לְךָ יָאְתָה תְבוּנָתִי הֲדוּרָה / וְהוֹד דֵּעִי וְחָכְמָה נוֹסֲסָה בִי

קְצָתָהּ לִי יְרֻשַּׁת־אָב מְאַלְּפִי / אֲבִי חָכְמוֹת וְהוּא מָרִי וְרַבִּי

וּמִקְצָתָהּ קְנוּיָה מִיְּגִיעִי / כְּבַשְׁתִּיהָ בְּמוֹ קַשְׁתִּי וְחַרְבִּי

וְעָמְקוּ מַחְשְבוֹתַי בָּהּ וְחַכְמֵי / אֱדוֹם הָיוּ בְעֵינַי כַּחֲגָבִי

וְהָלַכְתִּי בְבָתֵּי מִדְרְשֵׁיהֶם / וְאֵין מִי יַעֲרֹךְ אִתִּי קְרָבִי

אֲנַצֵּחַ אֱנוֹשׁ יָרוּם כְּנֶגְדִּי / וְאַכְנִיעַ וְאַרְשִׁיעַ מְרִיבִי

וּמִי זֶה יֶעֱרַב לִבּוֹ לְהַגִּיד / יְסוֹד רֵאשִׁית וְסוֹד מֶרְכָּב וְרוֹכְבִי

וְלִי נֶפֶשׁ יְתֵרָה נֶהֱדָרָה / עֲלֵי נַפְשׁוֹת בְּנֵי דוֹרִי עֲלוּבִי

וְלִי צוּרָה בְּעֹז צוּרָהּ בְּצוּרָה / עֲצוּרָה וַאֲסוּרָה תּוֹךְ כְּלוּבִי

וְכָלְתָה תַּעֲלֶה עַל־מַדְרְגָתִי / וְחָשַׁקְתִּי תְּקוֹמֵם עַל־שְׁלַבִּי

יְדִידִי מַה־לְּךָ בֵּין עַם טְמֵא־לֵב / כְּתַפּוּחַ בְּתוֹךְ יַעֲרֵי חֲרוּבִי

וְנַפְשָׁךְ הַטְּהוֹרָה בֵּין עֲמָמִים / כְּשׁוֹשַׁנָּה בְּבֵין חוֹחִי וְעֶשְׂבִּי

נְהַג וָלֵךְ וּבֹא עָדַי יְדִידִי / בְּרַח וּדְמֵה אֱלֵי־עֹפֶר וְלִצְבִי

וְהַלֶּךְ־נָא לְבֵית־אָב צוּר יְלָדְךָ / יְשַׂגְּבֶךָ אֱלֹהֵי מִשְׂגַּבִּי

וְהוּא אֵלִי יְיַשֵּׁר אֶת־נְתִיבְךָ / וּמִמֵּצַר יְבִיאָךְ אֶל־רְחוֹבִי

וְיִתֵּן בִּרְכוֹת הוֹרַי לְרֹאשְׁךָ / וְעוֹד לָךְ בִּרְכוֹת אָבִי וְשָׂבִי

וְיָאִיר מַחֲשַׁךְ רוּחִי וְיָשִׁיב / בְּרֹב חַסְדּוֹ לְמִישׁוֹר אֶת־עֲקֹבִי

אֲנִי אֶפְקֹד לְאֵל רוֹעִי זְרָעִי / וְאַשְׁלִיכָה אֱלֵי אָבִי יְהָבִי

וְיַרְאֵנִי צְבִי חִשְׁקִי לְפָנָיו / אֲנִי קוֹרֵא חֲמוּדִי מַקְשִׁיבִי

וְאְז אָשִׁיר לְיוֹצְרִי שִׁיר יְדִידֹת / אֲזַמֶּר־לוֹ בְּעוֹדִי שִׁיר עֲגָבִי

וְאַגִּישָׁה לְפָנָיו מִנְחָתִי / וְאֶעֶרְכָה תְשׁוּרָתִי לְהָבִיא

וּבָהּ נַפְשִׁי קְשׁוּרָה אֶל־קְדוֹשִׁי / וּבָהּ רֹאשִׁי וְשָׁם עֵינִי וְלִבִּי

וְיִיטַב לוֹ כְאֵילֵי מַהֲלָלִי / וּמִשּׁוֹר פָּר זְמִירָתִי וְנִיבִי

וְיַרְאֵנִי יְקָר צִיּוֹן וְיָפְיָהּ / בְּהוֹד מַלְכוּת וְשָׁם מִמְשַׁח כְּרוּבִי

וְיַחְדָּו יִהְיוּ תַמִּים בְּרֹאשָׁהּ / שְׁנֵי אוֹרִים בְּנוֹ דָוִד וְתִשְׁבִּי

וְלֹא אוֹיֵב יְחָרְפֶהָ לְדוֹרִים / וְלֹא יַהֵל לְעוֹלָם שָׁם עֲרָבִי.

(נַפּוֹלִי 1503)
