החרף

עִתּוֹתֵי עֹנֶג עָבָרוּ

וַיָבֹא קֹר תַּחְתֶּינָה

קָמֵל הַדֶּשֶׁא

הֶעָלִים נָבָלוּ

אַף רִבּוֹתָיִם

מֵעוֹף שָׁמָיִם

פָּנוּ עֹרֶף

לִפְנֵי חֹרֶף

כֹּה נָדְדָה מְנוּחָתִי

וַיָבֹא רֹגֶז תַּחְתֶּיהָ

יַעַן חָלְתָה יוֹנָתִי

נָבְלוּ שׁוֹשַׁנֵּי לְחָיֶיהָ


הַכְּפוֹר נִפְזַר כָּאֵפֶר

וּפְנֵי נָהָר יִתְלַכָּדוּ

הָהָר וָעֵמֶק

יַעַטְפוּ שֶׁלֶג

כֹּה יַבֶּשֶׁת

אַךְ לוֹבֶשֶׁת

תַּחַת פֶּרַח

מַעֲטֵה קֶרַח

כֵּן שׁוֹכֶבֶת יוֹנָתִי

אֵשׁ עֵינֶיהָ כָּבָתָה

הִלְבִּינוּ שִׂפְתֵי יָפָתִי

וְכָל הַדְרָתָהּ נִשְׁחָתָה


לַעֲרָבָה תֵּבֵל נֶהְפָּכָה

מֶרְחֲבֵי אֶרֶץ לַתֹּהוּ

חֶמְדַּת הַנֶּטַע

שָׁבָה לָאָיִן

וּמִן הַשַּׁיִת

עַד הַזַּיִת

עִם הַכֶּרֶם

הֻכּוּ חֵרֶם

כָּכָה אֹהֶל תַּמָּתִי

מָלֵא שֹׁאָה וָשָׁבֶר

כִּי עֲזָבַתּוּ יוֹנָתִי

וַתֵּרֶד אֱלֵי קָבֶר. –


יָמִים עוֹד יָבֹאוּ

תִּתְחַדֵּשׁ תֵּבֵל כְּקֶדֶם

אוּלָם אָנֹכִי

אֻמְלָל לָנֶּצַח

יָגוֹן וָעֶצֶב

מִבְּלִי קֶצֶב

הֶגֶה וָנֶהִי

חֶבְלִי יֶהִי

עֲדֵי תֵרֵד גְּוִיָּתִי

אֱלֵי בוֹר צַלְמָוֶת

וּבְתוֹךְ קֶבֶר יוֹנָתִי

עִמָּהּ אִישַׁן הַמָּוֶת. –


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
