תִּינוֹק שֶׁל בֵּית רַב...

תִּינוֹק שֶׁל בֵּית רַב בִּהְיוֹתִי, כַד הֲוִינָא טַלְיָא,

בְּ“אַלָּף-בֵּית” הַגַּן טִיַלְתִּי בִּמְסִבַּת פְּמַלְיָא

שֶׁל הַיְלָדִים הַקְטַנִּים, קְלַסְתֵּר פְּנֵיהֶם חִוְרִים,

הֲלֹא הֵמָה הַתִּינוֹקוֹת מִיַלְדֵי הָעִבְרִים!

וְהַפַּרְדֵס הוּא כַּךְ נִפְלָא, בָּעוֹלָם אֵין דֻגְמָתוֹ,

אֵין עַל עָפָר מָשְׁלוֹ, יָקָר הוּא וְאֵין זוּלָתוֹ!

עֲרוּגוֹתָיו בִּרְסִיסֵי טַל, בִּפְנִינִים מְשֻׁבָּצִים,

בְּפַלְגֵי מַיִם, אֶגְלֵי דֶמַע, יוֹנְקוֹתָיו מְרֻחָצִים!

פְּרָחָיו כְנִצָנֵי אָבִיב בְאִבֵּי הַתַּפּוּחִים,

מַרְאֶה לָבָן, חֲוַרְוָרִים, עוֹלְלֵי טִפּוּחִים!

וְכִי לֹא תִשְׁלוֹט בָּם עֵין רָעָה, חֹמָה לָהֶם וְגָדֵר,

מִימִינָם וּמִשְֹמֹאלָם כֹּתְלֵי בֵית הֶ“חָדֶר”!


כְּטָלֶה מֵחֲלֵב אִמּוֹ לֻקַּח, נִשָֹּא אֶל הָעֵדֶר,

כֵּן אָנֹכִי עַל הַכָּתֵף הוּבֵאתִי אֶל הַ“חֵדֶר”!

בּוֹ הַגַּן, הָ“אַלָּף-בֵּית” פָּרַח הֵצִיץ עָלִים,

אוֹתִיוֹת וּנְקוּדוֹת שְׁחוֹרוֹת (עָלִים אַךְ מְשָׁלִים!)

אַלָּף, בֵּית, גִימֶל, דֶלֶת; פַּתַּח, סֶגֹל, חוֹלָם

סְבָבוּנִי, הֲלָמוּנִי, הִרְעִישׁוּ כָל הָעוֹלָם!


עוֹלָם קָטָן הוּא כָל יֶלֶד, וְאָרוּךְ הוּא הַמַּקֵּל,

בּוֹ הַמְלַמֵּד הַכֵּה יַכֵּה, כִּכְפִיר עוֹלֵל יִשְֹכֵּל!

מֵהַבֹּקֶר וְעַד הָעֶרֶב, הוֹי! כָּלוּ כָּל הַכֹּחוֹת!

אֲסוּרֵי סַפְסַל הַיְלָדִים, הָאִמָּהוֹת בֹּכוֹת

מֵרֹב שִֹמְחָה כִי הִגִּיעוּ הַשָׁעָה וְהַסֵּדֶר

לְיַלְדֵיהֶן, עַטְרוֹת רֹאשָׁן, לְהִכָּנֵס אֶל "הַחָדֶר!…


חוּצָה יִשְׁפּוֹךְ שֶׁמֶשׁ אוֹרוֹ עַל הַיוֹם הָאָרוּךְ,

וּמִמְרוֹמָיו הָרָקִיעַ כֻּלּוֹ אוֹמֵר בָּרוּךְ!

בְּרוּכִים כָּל אוֹהֲבֵי חַיִים, חֹפֶשֹ בְּלִי מְצָרִים,

בְּמֵרְחֲבֵי חֵיק הַבְּרִיאָה, הַרְחֵק מֵחֲדָרִים!

לַמִּסְתַּתֵּר בַּחֲדָרִים, יָדוֹ עַל הֶחָזֶה,

רוּחַ מַחְנֵק יְצִיקֶנוֹ, מִיוֹם אֶל יוֹם יֵרָזֶה!

הֲיִפָּלֵא אֵיפֹא כִּי שָׁלְטָה הַמַּגֵּפָה

בַּיְלָדִים כִּבְבָתֵּי קְרָשִׁים הַשְֹרֵפָה? – –

מַר צוֹרַחַת אֵם עַל בָּנִים, קוֹרַעַת שָׁמַיִם הַצְעָקָה,

מִמָּר הַמָּוֶת לִפְדוֹתָם, מַרְבָּה תֵּת צְדָקָה!

אֵם פְּתַיָה! לוּלֵא נָשְֹאָה נַפְשָׁה אֶל הַ“חֵדֶר”,

בְּטֶרֶם מוֹעֵד אָז לֹא נִפְקַד שֵֹיָהּ מִן הָעֵדֶר!


זוֹכְרֵנִי כַד הֲוִינָא טַלְיָא, נִשָֹּא עַל כַּפָּיִם,

אֶל הַחֶדֶר, כִּפּרִי עֵץ הָדָר, כַּעֲדִי עֲדָיִם!

הָלֹךְ בְרֶגֶל לֹא הָלַכְתִּי, בַבֹּקֶר כָּשָׁל כֹּחִי!

וּמֵחֲלָמִישׁ חָזָק רוּחִי, עַז כַּבַּרְזֶל מֹחִי!

בְּרוּחִי בָרָאתִי “שַׁי” עוֹלָמוֹת בְּאַרְבַּע אַמּוֹת “חֵדֶר”,

מִיְאוֹר הֶעֱלֵיתִי לִוְיָתָן, שׁוֹר הַבַּר מֵעֵדֶר!

בְּאֶרֶץ מִדְבָּר נָטַעְתִּי כֶּרֶם, יֶרֶק, עֵשֶֹב, שִֹיחַ,

וָאֶהִי רוֹכֵב עַל גַב אָתוֹן, בִּזְרוֹעוֹת הַמָּשִׁיחַ!

עַד הַיוֹם הַזֶה חֶזְיוֹנַי לְנֶגֶד עֵינַי הִנָּם,

לְפָנַי נִגְלוּ מַטְמוֹנֵי אֶרֶץ, לִי אוֹצְרוֹתֶיהָ חִנָּם!

עוֹדֶנִי חוֹלֵם אַף בְּהָקִיץ כְּבִימֵי כְּבָר,

כְּמִקֶּדֶם וְעַד עָתָּה, מַהֲלָכִי עַל גֶשֶׁר נְיָר!

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
