בלב-ים

בְּלֶב-יָם / שמואל ליב גורדון

[מחזות ממגורשי ספרד]


א.

מִמְּרוֹם הָרָקִיעַ שֶׁמֶשׁ צָהֳרָיִם

תָּפִיץ לֶהָבוֹת עַל גַּלֵּי הַמָּיִם,

וּמֵי תְהוֹם רַבָּה הֶמְיָתָם שָׁכָכָה;

נִלְאוּ הַגַּלִּים, לֹא עוֹד יִתְהוֹלָלוּ,

מִשְׁבָּרִים גְּאֵיוֹנִים עָיְפוּ, הָשְׁפָּלוּ;

אַךְ דּוּמָם יִזָּלוּ, יֶהֱמוּ בַאֲנָחָה.

נִדְמוּ הָרוּחוֹת, גַּם הֵמָּה עָיָפוּ,

מִקַּרְנֵי הֶחָרֶס נִכְווּ כֻלָּהַם;

סָרוּ הַסְּעָרוֹת, יַחְדָּו הִתְעַלָּפוּ,

מֵרִשְׁפֵּי אֵשׁ לֹהֵט מִזִּקֵּי זָעַם.


וּשְׁחוֹרָה כָעוֹרֵב שָׁם בַּעֲצַלְתָּיִם,

כִּנְקֻדָּה קְטַנָּה עַל רַחֲבֵי הַמָּיִם,

מֵרָחוֹק, מֵרָחוֹק אַט תִּתְנוֹעֵעַ…

תָּחֹג, תָּנוּעַ וּכְמוֹ מִתְהַפֶּכֶת,

לִרְגָעִים תַּעֲמֹד אַף תָּשׁוּב לָלֶכֶת,

תָּרוּץ אַף תָּשׁוּב וָהִיא – כִּמְתַעְתֵּעַ,

תִּשְׁפַּל תֵּרֹמָם, תֵּחָבֵא מֵעַיִן,

וּכְרֶגַע תֵּרָאֶה שָׁמָּה עוֹבָרֶת.

מִי זֹאת הַנְּקֻדָּה? מֵאַיִן וּלְאַיִן?

הַגִּידוּ, הַגַּלִּים! סַפְּרָה, הֶחָרֶס!


וּלְנֹגַהּ הַשֶּׁמֶשׁ, שְׁחָקִים לוֹהֶטֶת,

הַנְּקֻדָּה הַשְּׁחוֹרָה הָלְאָה מְשׁוֹטֶטֶת,

עַל גַּבֵּי מִשְׁבָּרִים דּוּמָם יִנְהָרוּ;

אַט אַט תִּתְקָרֵב מֵאַפְסֵי שָׁמָיִם,

תִּקְרַב, תִּתְגַּדֵּל, תֵּרָא לָעֵינָיִם.

סְבִיבָהּ מִשְׁבָּרִים – וּכְמוֹ יִתְפּוֹרָרוֹ…

יִתְפָּרְדוּ, יִתְפַּצְּלוּ, הָלְאָה יַעֲבֹרוּ;

מִי הִיא זֹאת שָׁמָּה? וּמַה הִיא הַנְּקֻדָּה?

הַגִּידוּ נָא, גַּלִּים, אָנָּא סַפֵּרוּ!…

אֳנִי-שַׁיִט שָׁמָּה, בָּהּ גּוֹלֵי יְהוּדָה!


                                                                         ***

אֳנִי-שַׁיִט שָׁמָּה, בָּהּ נוֹדְדֵי נֶצַח,

עֲשׁוּקִים, נִדָּחִים מִשֹּׁד וָרֶצַח.

אֵין מָקוֹם לָמוֹ, אֵין מִשְׁכָּן, אֵין אֶרֶץ;

גַּם אֶרֶץ מְגוּרָם, סְפָרַד, הֱקִיאָתַם,

וּבְכַף הַקֶּלַע הָלְאָה הִשְׁלִיכָתַם,

וּבִלְבַב יַמִּים הֵם צְפוּיִים לַקָּרֶץ.

אֻמְלָלִים! אֻמְלָלִים! עֲשׁוּקֵי כֹחַ!

אָנָה תָּנוּסוּ? אֵי מִפְלָט תִּמְצָאוּ?

אָבַד כֹּל מָנוֹס, הָהּ, אֵין כֹּל מָנוֹחַ!

הֲטֶרֶם, אֻמְלָלִים, הֲטֶרֶם תֵּדָעוּ?…


אֵי כִּבְרַת אֶרֶץ, מְלֹא שַׁעַל אֲדָמָה,

בּוֹ רוּחַ תִּשְׁאָפוּ, תַּרְגִּיעוּ שָׁמָּה?

אָן אֵפוֹא, נִדָּחִים, אָנָה תְּשוֹטֵטוּ?

עַל מִצְחֲכֶם חָרוּת תַּו שַׁחַת, כִּלָּיוֹן,

תַּו דָּם, תַּו מָוֶת – לֹא תוּכְלוּ נִקָּיוֹן!

הַתְקַוּוּ, אֻמְלָלִים? הֲתִמָּלֵטוּ?

קַוֶּינָה, קַוֶּינָה, שֵׁיוֹת נִדָּחוֹת!

וּבַאֲשֶׁר תָּנֹסְנָה נֹסְנָה, שׁוֹטֵטְנָה!

עוֹד מִפְלָט תִּמְצֶאנָה, נְפָשׁוֹת נֶאֱנָחוֹת!

בֵּין שַׁעֲרֵי הַשַּׁחַת – שָׁמָּה תִּשְׁקֹטְנָה…


וּבָאֳנִי שָׁמָּה… פַּלָּצוּת וּזְוָעָה!

אַךְ מַחֲזֵה שַׁחַת, אַךְ צֶלֶם בַּלָּהָה!

שָׁם אֵם רַחֲמָנִיָּה דּוּמָם בּוֹכִיָּה,

זֹה יֶלֶד חוֹלֶה אֶל לִבָּה חוֹבֶקֶת,

וּמִנְּשִׁיקוֹת פִּיהָ פִּיהוּ נוֹשֶׁקֶת.

שָׁם עַלְמָה עֲנֻגָּה, רַכָּה, יְפֵה-פִיָּה.

עוֹרָהּ מַה-צָּפַד, מַה-קָּדְרוּ פָנֶיהָ,

שָׁמָּה בַּפִּנָּה תִּפֹּל מִתְעַלָּפֶת.

עַל יָדָהּ יֹלֶדֶת תִּסְפֹּק כַּפֶּיהָ

וּתְצַפְצֵף כַּצִּפּוֹר, הֹמָה, זוֹעָפֶת.


שָׁמָּה יֵרָאֶה אִישׁ זָקֵן סָרוּחַ,

עֵינָיו עֲצוּמוֹת, כָּל עוֹד בּוֹ הָרוּחַ,

שְׁעָרוֹ כַּכֶּסֶף, כַּדּוֹנַג פָּנֵיהוּ;

לִרְגָעִים יִתְפַּלֵּץ, יִקְמֹט הַמֵּצַח

וִיעַקֵּם שְׂפָתֵיהוּ… וּשְׁנַת הַנֵּצַח

כְּנָפֶיהָ הַשְּׁחוֹרוֹת תִּפְרֹשׂ עָלֵיהוּ…

וִילָדִים שָׁם שְׁנַיִם עַל מַצַּע תֶּבֶן,

בִּידֵיהֶם הַקְּטַנּוֹת יַחַד חֻבָּקוּ…

יָנוּחוּ, יִשְׁלָיוּ, יִדְּמוּ כָאֶבֶן,

וּשְׂפָתֵימוֹ סֻגְּרוּ, לֹא עוֹד יִנְאָקוּ.


וְּצִעירָה לְיָמִים, פָּנֶיהָ שָׁחָרוּ,

לְחָיֶיהָ שֻׁפּוּ, עֵינֶיהָ סֻגָּרוּ,

וּבְחֹרֵיהֶן עָמֹק, עָמֹק שָׁקָעוּ,

מִבְּלִי נוֹעַ שָׁמָּה דּוּמָם שׁוֹכֶבֶת…

עָלֶיהָ מֻטֶּה – הוֹי מַחֲזֵה עַצֶּבֶת! –

אִישׁ צָעִיר לְיָמִים, פָּנָיו לֹא יֵרָאוּ

מֻטֶּה, שְׁחוֹחַ אַף יוֹרִיד רֹאשֵׁהוּ

וּבְשִׂפְתֵי הַמֵתָה שְׂפָתָיו יִגָּעוּ. ­–

וּכְתֹמֶר מִקְשָׁה שָׁם יָצוּק גֵּוֵהוּ,

וּכְמוֹ שְׁנֵיהֶם יַחַד, יַחַד נִקְפָּאוּ.


וַאֲנָשִׁים שָׁמָּה מַה-שָּׁחַר מַרְאָמוֹ!

מַה נּוֹרָא מַבָּטָם יֵתַע סְבִיבָמוֹ,

עֲרָפֶל פָּנֵימוֹ, מִצְחָם עֲנָנָה…

רָאשֵׁיהֶם יוֹרִידוּ, וּכְמֵתֵי עוֹלָם

יַחַד נֶאֱלָמוּ, לֹא יַשְׁמִיעוּ קולָם,

וּמְשׁוֹטְטוֹת עֵינֵיהֶם אָנֶה וָאָנָה.

וַעֲיֵפִים וִיגֵעִים שָׁם הַמַּלָּחִים

מִשְׁבָּרִים הוֹמִים יֶחֱצוּ בַעֲצַלְתָּיִם –

אָנָה תִשְׁאָפוּ, מַלָּחִים נִדָּחִים?

אָנָה תַבִּיטוּ? אָן תִּשׂאוּ עֵינָיִם?


                                                                                       ***

הַשֶּׁמֶשׁ נֶעֱלָמָה… רוּחוֹת יִסְעָרוּ,

מֵרֶגַע לְרֶגַע חַיִל יִגְבָּרוּ,

וּמִשְׁבְּרֵי יַמִּים יִגְדָּלוּ, יִגְדָּלוּ –

וּכְהַרְרֵי אֹפֶל, זוּ יַחַד דֻּבָּקוּ,

יַעֲלוּ, יֵרֹמּוּ וּשְׁחָקִים יִשָּׁקוּ. –

וּמֵרֹם שָׁמָּה גַּלְגֵּל יִתְגַּלְגָּלוּ

וּבַהֲמֻלָּה גְדוֹלָה יִפְּלוּ בִמְצוֹלוֹת –

וּבִמְצוֹלוֹת שָׁמָּה, שָׁמָּה בְמַעֲמַקִּים,

יִשְׁטֹפוּ, יִזְרֹמוּ, יֵצְאוּ בִמְחוֹלוֹת…

לְקוֹל הֲמוֹן רוּחוֹת, חֲלִילֵי מַחֲשַׁכִּים.


וּרְעָמִים יִרְעָמוּ וּבְרָקִים בָּרֹם

לִרְגָעִים יִבְרֹקוּ, יָאִירוּ הַתְּהוֹם,

יִבְרֹקוּ, יֵעָלְמוּ וּכְמוֹ יֶחֱרָדוּ

מִשּׂוֹא הַגַּלִּים, מִקַּדְרוּת פָּנֵימוֹ,

מִשַּׂלְמַת הַקֶּצֶף תַּעַט גֵּוָמוֹ. –

עֲרָפֶל, שָׁמַיִם וּתְהוֹם הִתְלַכָּדוּ,

הִתְלַכְּדוּ, הִתְחַבְּרוּ וְלֹא יִתְפָּרָדוּ,

צִבְאוֹת הַתֹּפֶת שָׁמַיִם יֵעָלוּ

וּמַלְאֲכֵי מַעַל לִתְהוֹמוֹת יֵרֵדוּ –

וּבְעִמְקֵי שַׁחַת כִּפְתָנִים יִזְחָלוּ…


וְקוֹלוֹת אֲיֻמִּים כַּאֲנָחוֹת, כִּילָלוֹת,

כְּלַעַג שַׁאֲנַנִּים, כַּחֲרָפּוֹת, כִּקְלָלוֹת,

מֵעִמְקֵי תְהוֹמוֹת, מִמְּרוֹמֵי שָׁמַיִם,

מֵעָבֵי עֲרָפֶל יַחְדָּו יִתְבּוֹלָלוּ…

וּבְרַעַם וּבְזַעַם חוֹל יִתְחוֹלָלוּ –

יַחְדְּרוּ הַנֶּפֶשׁ, יַחֲרִישׁוּ אָזְנָיִם…



                                                                                         ***

חֲשֵׁכָה אֲיֻמָּה… מוֹרְאָה נִגְאָלָה,

קוֹל רַעַם בַּגַּלְגַּל, רִשְׁפֵּי הַבְּרָקִים…

קוֹל רַעַשׁ אֲנָחוֹת, קוֹל בְּכִי וִילָלָה…

קוֹֹל צְחוֹק, קוֹל לַעַג בַּמְּרוֹמִים, בַּשְּׁחָקִים…


וּמִפְּאַת קָדִּים שָׁם נִגְלֹה נִגְלָתָה

מַלְכַּת שָׁמַיִם בָּאֹפֶל כֻּסָּתָה,

וְגַלִּים אֲדֻמִּים פָּנֶיהָ קִדֵּמוּ,

וּמִשְׁבְּרֵי יַמִּים לִקְרָאתָהּ רָחָפוּ,

הִשְׁתַּחֲווּ מִנֶּגֶד, רֹאשָׁם כָּפָפוּ,

וּבְגָאוֹן וָגֹבַה הָלְאָה יִזְרֹמוּ. –

וּמְלֶכֶת שָׁמַיִם, מֹשֶׁלֶת הַיּוֹם,

תִּצְעַד מַעֲדַנּוֹת עַל מַעֲלוֹת רָקִיעַ;

וּלְעֻמָּתָהּ נִשְׁקָף כַּתּוֹלָע הַתְּהוֹם,

תְּהוֹם דָּם… תְּהוֹם נוֹרָא לִשְׁאוֹלָה יַגִּיעַ…


וָאֳנִי הַשַׁיִט מֵרָחוֹק נִשְׁקֶפֶת,

עַל גַבֵּי הַגַּלִים אַט, אַט מְרַחֶפֶת.

הַמְּשׁוֹטוֹת יִנֹּעוּ… וְּבתוֹךְ הָאֳנִיָּה –

בּוּקָה וּמְבוּקָה… אֵין אֵם רַחֲמָנִיָּה,

אֵין זָקֵן גּוֹסֵס, אֵין אִשָּׁה בוֹכִיָּה,

אֵין עַלְמָה רַכָּה, יֹלֶדֶת הוֹמִיָּה,

אֵין אֵשֶׁת נְעוּרִים, אֵין יוֹנְקֵי שָׁדָיִם,

אֵין לֵבָב סוֹעֵר, אֵין עַיִן נִגֶּרֶת…

אֵין תָּאֳמֵי צְבִיָּה, הַיְלָדִים הַשְּׁנָיִם –

כֻּלָּם בִּמְצוֹלוֹת טֻבְּעוּ כַּעוֹפֶרֶת.


אַךְ שָׁמָּה בַפִּנָּה כִּתְמוּנוֹת אֹפֶל

דּוּמָם יֵשֵׁבוּ, יָנוּעוּ בַשֵּׁפֶל –

עֲרָפֶל פָּנֵימוֹ, עֵינָם הַשַּׁחַת.

יָדָם בִּכְבֵדוּת בַּמָּשׁוֹט אוֹחֶזֶת,

יָנִיעוּ יֶחְדָּלוּ… נַפְשָׁם רוֹגֶזֶת…


                                                                                          ***

פִּתְאֹם קַו נֹגַהּ, קַו תִּקְוָה וָנַחַת,

הִתְרוֹצֵץ כַּבָּרָק בֵּין עַרְפְּלֵי לָיִל,

עַל חֶשְׁכַת הַפָּנִים מִשְּׁחוֹר שָׁחָרוּ –

וַיִּתְאוֹשָׁשׁוּ וַיַּגְבִּירוּ חָיִל,

וַיַּכּוּ בַמָּשׁוֹט… הָלְאָה עָבָרוּ.


וָהָלְאָה וָהָלְאָה אֳנִי הַשַּׁיִט

תְּעוֹפֵף כַּיּוֹנָה מִפַּרְסוֹת הָעַיִט.

עוֹד רֶגַע, עוֹד רֶגַע וָהִיא – נִצֶּלֶת…

יָגִיל כָּל לֵב, כָּל נֶפֶשׁ רוֹחֶשֶׁת:

'שָׁמָּה הָאָרֶץ, שָׁם, שָׁם הַיַּבֶּשֶׁת,

‘שָׁם תִּקְוַת הַחַיִּים, שָׁם הַתּוֹחֶלֶת…’

וּסְלָעִים וְכֵפִים בִּשְׂדֵרוֹת יֵרֹמּוּ,

וּכְשִׁנֵּי הַתְּהוֹם לָעַיִן יֵרָאוּ.

הַגַּלִּים בֵּינוֹתָם בְּשֶׁצֶף יֵהֹמוּ,

וּבִשְׂפַת הַחוֹל לִרְגָעִים יִגָּעוּ.


וּבֵין סַלְעֵי מָגוֹר סְבִיבָם נִשְׂעָרָה,

אֳנִיַּת הַשַּׁיִט בֶּטַח עָבָרָה,

וּלְיַד חוֹף הַחוֹל כְּבָר הָכֵן נִצֶּבֶת.

הַגּוֹלִים מִמֶּנָּה חִישׁ הִתְחַמָּקוּ

וְרִגְבֵי הֶעָפָר חִבְּקוּ וַיִּשָּׁקוּ…

וּבְדִמְעוֹת עֵינַיִם שְׂפָתָם דּוֹבֶבֶת:

'הֶאָח, אֶרֶץ, אֶרֶץ, אֶרֶץ הַחַיִּים!

'אַתְּ מָעוֹן לַתִּקְוָה, מִשְׁכַּן כָּל נָחַת,

'כָּל תְּלָאָה וְצוּקָה בָּךְ עֹנֶג שָׁמָיִם,

‘הַתֹּפֶת בָּךְ עֵדֶן, חַיִּים – הַשָּׁחַת!’


יָגִילוּ הַגּוֹלִים אֵין קֵץ לִשְׂשׂוֹנָם,

יִשְׂחֲקוּ לַגַּלִּים, מִשְׁבָּרִים עוֹד אֵינָם!

עַתָּה מְאֻשָּׁרִים הִנָּמוֹ בָּאָרֶץ!

וָהָלְאָה הַגּוֹלִים, הָלְאָה יֵלֵכוּ,

וּבְבִרְכַּיִם כּוֹשְׁלוֹת דַּרְכָּם יְשָׂרֵכוּ,

עַל צַלְעֵי הָהָר שָׁם אַף יִפְרְצוּ פָרֶץ

עֲלוֹת עַל רֹאשׁוֹ כָּל אֶחָד רִאשׁוֹנָה –

שָׁם תִּקְוַת הַחַיִּים קִנָּהּ כּוֹנֵנָה –

שָׁמָּה, יַחֲשֹׁבוּ, שָׁם נַפְשָׁם יַשְׂבִּיעוּ,

שָׁם יִמְצְאוּ מָנוֹחַ, שָׁמָּה יַרְגִּיעוּ.


וּמַעְלָה הַגּוֹלִים יִשְׂאוּ רַגְלָיִם

עַל יַרְכְּתֵי הָהָר וּבְזֵעַת אַפָּיִם…

וּמַעְלָה… וּמַעְלָה…….

………. פִּתְאֹם קוֹל נַהַם

מֵרָחוֹק יִשָּׁמַע… קוֹל מַרְגִּיז לְבָבוֹת…

קוֹל סֹעֵר כַּסַּעַר, חֹצֵב לֶהָבוֹת…


                                                                                                 ***

עַל רֹאשׁ הָהָר שָׁם אֲרִי מָלֵא זַעַם,

עֵינָיו אֵשׁ לֹהֵט, כַּתֹּפֶת פָּנֵיהוּ,

פִּיו פָּעוּר כַּקֶּבֶר וּלְשׁוֹנוֹ כַּדָּם…

וּזְנָבוֹ כָּאֶרֶז יֶאֱזֹר גֵּוֵהוּ ­–

וִישַׁחֵר לַטֶּרֶף אַף יִנְהֹם כַּיָּם.


וּמִמְּאוּרוֹתם מִסָּבִיב זִנֵּקוּ

גּוּרֵי אֲרָיוֹת, שִׁנַּיִם יַחֲרֹקוּ…

וּבְקוֹלוֹת אֲיֻמִּים יַחַד יִצְהָלוּ…


***


'הָהּ, חוּסָה… חֲמָל-נָא… שׁוֹכֵן שָׁמָיִם!…

                                                                                               ***

וּבְשֶׁצֶף קֶצֶף וּבַחֲמַת אַפָּיִם

עַל טַרְפָּם כֻּלָּמוֹ חִישׁ הִתְנַפָּלוּ.


                                                                                              ***

ב.

הַגַּלִּים, הַגַּלִּים עָלוּ שָׁמָיִם,

הָמוּ, רָעָשׁוּ, הֶחֱרִישׁוּ אָזְנָיִם,

נָדַמּוּ כֻלָּהַם, יַחַד שַׁאֲנָנוּ,

בִּזְרֹעוֹת הָעֶרֶב דּוּמָם יִישָׁנוּ…

וִירִיעַת הַתְּכֵלֶת בַּמְּרוֹמִים שָׁמָּה,

בַּעֲנָנִים כֵּהִים זְעֵיר שָׁם נִכְתָּמָה;

וּבְלִבְנַת מָוֶת בַּעֲדָם הַיָּרֵחַ

נִשְׁקָף שָׁם בָּדָד, עָיֵף וּרְפֵה כֹחַ.


וּדְמָמָה שׂוֹרֶרֶת, דִּמְמַת הַשַּׁחַת,

בִּמְרוֹמֵי אֵין קֵץ בִּתְהוֹמוֹת מִתַּחַת.


אַךְ שָׁמָּה, שָׁמָּה מֵרָחוֹק יִתְרוֹמֵם

עַל פְּנֵי הַמַּיִם אִי קָטָן, אִי שׁוֹמֵם,

עֲטוּף מַעֲטֵה כֵהָה כַּמֵּת מַרְאֵהוּ,

שָׁם קַו-צֵל שָׁחוֹר יִמָּשֵׁךְ אַחֲרֵיהוּ –

וּסְבִיבוֹ אַךְ מַיִם, מַיִם וָמָיִם,

אֵין גְּבוּל, אֵין קֵצֶה, עַד שׁוּלֵי שָׁמָיִם – –


שָׁם עַל שֶׁן-סֶלַע כִּצְלָלִים יֵרָאוּ,

יָנוּעוּ, יִגְבָּהוּ, רֹאשָׁם יִשָּׂאוּ –

מִי אַתֶּם שָׁמָּה? הַגִּידוּ, הַצְּלָלִים!

הַזְאֵבֵי עֶרֶב אִם לַהֲקַת שׁוּעָלִים?

אִם רוּחוֹת אֹפֶל, אִם שֵׁדֵי הַשַׁחַת?

אִם יוֹנִים קִנֵּנוּ בְּעֶבְרֵי פִי-פַחַת?

הַגִּידוּ, הַצְּלָלִים, מִי אַתֶּם שָׁמָּה?

הֲצִלְלֵי תֹפֶת אִם בְּרוּאֵי אֲדָמָה? –


וּדְמָמָה שׂוֹרֶרֶת, דִּמְמַת הַשַּׁחַת,

בְּמרוֹמֵי אֵין קֵץ, בִּתְהוֹמוֹת מִתַּחַת…


וַאֲנָחָה חֲרִישִׁית, אַנְחַת דּוּמִיָּה,

כְּצִפּוֹר גֹּוַעַת הוֹמִיָּה, הוֹמִיָּה –

וַאֲנָחָה חֲרִישִׁית וּכְקוֹל גֹּוֵעַ

כְּאַנְחַת הַהֵד שָׁם דּוּמָם תִּשָּׁמֵעַ…


                                                                                         ***

שָׁמָּה הַצְּלָלִים אֶל עָל יִנָּשֵׂאוּ,

יִשְׁפָּלוּ, יִשְׁפָּלוּ אַט, אַט יִנֹּעוּ.


וּבְרַחֲבֵי אֵין קֵץ פִּתְאֹם יִשָּׁמֵעַ,

וּלְרַחֲבֵי אֵין קֵץ סָבִיב יִשְׂתָּרֵעַ,

קוֹל גָּרוֹן נִּחָר, קוֹל הוֹמֶה כַּחֲלִילִים,

קוֹל מַשַּׁק שְׂפָתַיִם, קוֹל לַחַשׁ מִלִּים.


הַגִּידוּ, אֻמְלָלִים, מִי אַתֶּם שָׁמָּה?

הֲצִלְלֵי תֹפֶת אִם בְּרוּאֵי אֲדָמָה?…


                                                                                                 ***

'הָהּ, לִבִּי – הָהּ, נַפְשִׁי – אֵשׁ מִתְלַקַּחַת –

'אֵשׁ תֹּפֶת – בִּי תוּקַד – קִרְבִּי הַשַּׁחַת –

'הָהּ, בְּנִי – בַּר בִּטְנִי – פֹּרָת עֲלֵי עָיִן –

'כְּבָר נִקְפָּא–כַּקֶּרַח–כְּבָר מֵת–כְּבָר אָיִן –

'וּבִתִּי – הָהּ, בִּתִּי – גַּם הִיא – גָּוָעָה –

'הִיא קָרָה – כַּכְּפוֹר – הָהּ, נַפְשָׁהּ יָצָאָה –

'אָמוּתָה נָּא, אִישִׁי – אָמוּת – הַפָּעַם –

'הָהּ, בְּנִי – הָהּ, בִּתִּי – אֵל מָלֵא זָעַם! –

'תַּאֲלָתִי לָךְ – אֶרֶץ – אֶרֶץ סְפָרַד –

'אֲרוּרָה – אֲרוּרָה – תִּהְיִי עֲדֵי עַד! – '


וַאֲנָחָה אֲיֻמָּה כִּילֵל הַשְּׁאִיָּה –

כְּאַנְחַת גֹּוַעַת שָׁם דּוּמָם הוֹמִיָּה.

וַאֲנָחָה אֲיֻמָּה – – – – – – – –

– – – הַס, הַס וּדְמָמָה…

– אֵין מִלִּים – אֲנָחוֹת – שַׁמָּה וּשְׁמָמָה –


                                                                                                 ***

וּבָדָד שָׁם דּוּמָם, דּוּמָם כַּקֶּבֶר…

כַּקֶּבֶר, כַּמָּוֶת, כִּתְמוּנַת גֶּבֶר,

עַל פִּסְגַּת הַסֶּלַע שָׁמָּה נִצֶּבֶת,

כְּצֶלֶם בַּלָּהוֹת כִּתְמוּנַת צַלְמָוֶת.

הַגִּידָה, הַצֶּלֶם, מִי אַתָּה שָׁמָּה?

הֲשֵׁד מִשַּׁחַת אִם אֻמְלַל אֲדָמָה?


וּדְמָמָה שׂוֹרֶרֶת, דִּמְמַת הַשַּׁחַת

בִּמְרוֹמֵי אֵין קֵץ, בִּתְהוֹמוֹת מִתַּחַת.

עַל שֵׁן הַסֶּלַע עוֹדֶנָּה עוֹמֶדֶת

הַתְּמוּנָה הַשְּׁחוֹרָה וָהִיא – רוֹעֶדֶת…

רוֹעֶדֶת, נוֹדֶדֶת לִרְגָעִים כַּצֵּל –

וּבִתְהוֹם הַדּוּמִיָּה, בִּדְמִי הַלֵּיל,

קוֹל מִלִּים יִשָּׁמַע – וּלְפֶתַע פִּתְאֹם

יִשָּׁמַע בַּמְּרוֹמִים, יִשָּׁמַע בַּתְּהוֹם.


'וּבָדָד, הָהּ, בָּדָד… פֹּה עַל שֶׁן סֶלַע –

הָשְׁלַכְתִּי, טֻלְטַלְתִּי, וּבְכַף הַקֶּלַע –

אֵין אֵשֶׁת, אֵין בָּנִים……

……… אֵל בַּשָּׁמַיִם!

הָפִיצָה עֶבְרָתְךָ, עֶבְרוֹת אַפַּיִם,

וּשְׁלַח בִּי זַעְמֶךָ, אַל עֵינְךָ תָּחוּסָה,

יְהוּדִי הִנֵּנִי וּמַה כִּי אָמוּתָה? –

אָמוּתָה, אֶגְוָעָה, וּבְיַרְכְּתֵי שַׁחַת…

שָׁם אִשְׁתִּי, שָׁם בָּנַי, שָׁם אֶשְׂבַּע נַחַת –

שָׁם אַחַי בְּנֵי עַמִּי… שָׁמָּה כֻלָּהַם…

הָפִיצָה אַפֶּךָ, הַרְעֵם הָרָעַם…'


                                                                                              ***

שָׁמָּה הַתְּמוּנָה עוֹדֶנָּה נִשְׁקֶפֶת,

עַל שֵׁן הַסֶּלַע, עֲרָפֶל עוֹטֶפֶת,

וַאֲנָחָה אֲיֻמָּה סוֹעֶרֶת בְּזָעַם,

סוֹעֶרֶת, הוֹמִיָּה כַּהֲמוֹת הָרָעַם –

וּתְעוֹרֵר הַגַּלִּים וּתְעוֹרֵר בְּשָׁאוֹן.


                                                                                              ***

אַךְ שַׁמָּה מִסָּבִיב, שַׁמָּה קִפָּאוֹן –

הַגַּלִּים שָׁם יִישְׁנוּ בִּדְמִי עַצָּבֶת,

יִישָׁנוּ, יִרְדָּמוּ תַּרְדֵּמַת מָוֶת –

וּדְמָמָה מִתַּחַת וּשְׁאִיָּה בָרֹם,

שַׁמָּה בַמָּרוֹם, שִׁמָּמוֹן בַּתְּהוֹם – –


תר"ן

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
