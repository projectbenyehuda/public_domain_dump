הַזְרֹעַ וְהַלְּחָיַיִּם וְהַקֵּבָה

Efraim da Modena

]חי באמצע המאה הט"ו]


הַזְרֹעַ וְהַלְּחָיַיִּם וְהַקֵּבָה

אַחַי שִׁמְעוּ קוֹלִי / רֵעַי הַאְזִינוּ לִי / הַקְשִׁיבוּ מִלּוּלִי / בְּדַעַת בְּרוּרָה

פִּי אֶפְתַּח בְּמָשָׁל / שׁוֹֽמְעִי לֹא יִשַּׁל / וּבָשֵׁל יִתְבַּשֵּׁל / כְּבָשָׂר בַּסִּירָה

רַק וְלֹא יַאֲמִין / אַפִּיקוֹרוֹס אוֹ מִין / יִתְעֲבֵד הַדָּמִין / אַחֲרִיתוֹ מָרָה

יוֹסִיף יָמִים עשֶׁר / דֵּעָה בִּינָה כֹּשֶׁר / כָּל־מַעֲשָׂיו יֹשֶׁר / לוֹ תְהִי הַמִּשְׂרָה

מַאֲזִין אִמְרֵי־פִי / וְלֹא יִמְצָא דֹפִי / הוֹד הָדָר וָיֹפִי / לְנַפְשׁוֹ הַטְּהוֹרָה

בִּינוּ אֶת־שִׁירָתִי / חִרְשׁוּ בְעֶגְלָתִי / הַשְׂכִּילוּ אִמְרָתִי / אֲשֶר הִיא יְשָׁרָה

נוֹרָאוֹת אַשְׁמִיעְכֶם / נִפְלָאוֹת אַרְאֲכֶם / דֵּעָה אֲחַוְּכֶם / בִּדְרִישָׁה חֲקִירָה

יִתְבּוֹנְנוּ שְׁמוּעוֹת / בִּשְׁלֹשֶׁת הָרָעוֹת / רֵאשִׁיתָן זְוָעוֹת / וְסוֹפָן מְאֵרָה

וַאֲשֶׁר יְפֻתֶּה / עָרוּךְ אֵלָיו תָּפְתֶּה / בַּעֲצִיצוֹ יִשְׁתֶּה / כָּל־שְׁנוֹתָיו עֶבְרָה

אַף נֶאֶסְפָה שִׂמְחָה / וְגִילָה וּרְוָחָה / וְיוֹם הַמְּנוּחָה / הוּא אֵלָיו לְזָרָא.


בַּעַר יְסִיתֵהוּ / צְחוֹק וִיכַלֵּהוּ / נִסְתַּחֲפָה שָׂדֵהוּ / עֲצָתוֹ נִבְעָרָה

מַאְכָלוֹ לֹא לִשְׂבּוֹעַ / וְלֹא יִמְצָא מַרְגּוֹעַ / תָּמִיד יָנוּד יִנּוֹעַ / וְשׁוֹעַ הָהָרָה

מְנֻדָּח לַאְפֵלָה / הִתְקַצֵּף בִּקְלָלָה / וּפָנָה לְמָעְלָה / גַּם יְדַבֵּר סָרָה

וּמָעוֹת בְּכִיסוֹ / כְּחָשִׁים לְשׁוֹסוֹ / יִרְגַּז יַרְאֶה כַעְסוֹ / בְּמָדוֹן וְתִגְרָה

דַּלּוּת וְחֶסְרוֹן כִּיס / שַׁרְאֶצֶר רַב־סָרִיס / נֵרְגַל רַב־מָג עָכִיס / אַף זַעַם וְצָרָה

יָחִישָׁה מִפְלָט לוֹ / יִגְעֶה עַל־בְּלִי־לוֹ / יְחַנֵּן אֶת־קוֹלוֹ / כְּאִישׁ קָדוֹשׁ נוֹרָא

נָחוּץ בֵּית נְדִיבִים / בְּשַׁעַר בַּת־רַבִּים / לֶאֱסֹף זְהוּבִים / וְעֶרְוָתוֹ עֵרָה

אַךְ יָשׁוּב לְאֵיתָנוֹ / וְיֵשֵׁב עַל־כַּנוֹ / בְּכִיס עָרוֹ עֵינוֹ / לְפָנָיו בּוֹר כָּרָה

חוּשָׁהּ אַל־תַּעֲמֹד / הַדָּמִים אַל־תָּמֹד / מַהֵר יַתְחִיל לִלְמֹד / בְּסִפְרֵי וְסִפְרָא

זֹאת יַפְסִיק רֶגַע / יָד אַחֶרֶת תִּיגַּע / וְיִמְצָא כָּל־נֶגַע / לָמוּת נַפְשׁוֹ הֶעְרָה

קוֹרֵא בְּמַר נַפְשׁוֹ / צוֹעֵק אוֹי לוֹ רֹאשׁוֹ / וּמִצַּחְנַת בָּאְשׁוֹ / נִמְלְאָה הַבִּירָה

חָנוֹן לֹא יְחוֹנָן / כָּל־יָמָיו יִּתְאוֹנָן / וּבְנֵי בֵיתוֹ כּוֹנָן / בְּשֶׁבֶר וּסְעָרָה.


וְרָעָה עוֹד מִזֹּאת / אֲשֶׁר רָאוּי לִבְזוֹת / נְבָלָה גַּם לָזוּת / צָרָה כְּמַבְכִּירָה

אֲשֶׁר אֵלָיו יִנְעַם / לֶאֱכֹל רֹב מַטְעָם / וְגַם בְּתוֹךְ הָעָם / הוּא מַעֲלֶה גֵרָה

דָּבָר יִישֵׁר אֵלָיו / אַף כִּי אֵין בְּעָלָיו / יִצְעַק בַּחֲבָלָיו / יָחִיל כְּמוֹ הָרָה

יֹאכַל בְּלֹא עִתּוֹ / אִם נִתְמַלְּאָה סָאתוֹ / לֹא יוּכַל שְׂאֵתוֹ / לִבְנֵי בֵיתוֹ יִקְרָא

כִּי אֲחָזוֹ שָׁבָץ / לְרֹב אֹכֶל קָבָץ / תַּחַת מַשָּׂא רָבָץ / וְנַפְשׁוֹ בוֹעֵרָה

אֶת־הָאָרֶץ הִשְׁקָה / וְנַפְשׁוֹ רוֹקֵקָה / בּוּקָה וּמְבֻלָּקָה / כָּל־מַאְכָלוֹ יָרָה

וְאֶל־הַמֶּרְקָחוֹת / עֵינָיו הֵן פְּקוּחוֹת / וּשְׂפָתָיו פְּתוּחוֹת / פִּיהוּ כְּפִי פָרָה

וְאִם יֹאכַל בָּשָׂר / לְבַדּוֹ אַל יִישַׁר / וְדָג גָדוֹל יִכְשַׁר / עִמּוֹ לְחַבֵּרֶה

וְרֹב הַמַּטְעַמִּים / יְכַלּוּ הַדָּמִים / אֲחָזוּהוּ אֵימִים / מַכָּתוֹ מְרוֹרָה

יִמְכֹּר אֶת־לְבוּשׁוֹ / לְהַחֲיוֹת נַפְשׁוֹ / יִשְׁתֶּה יִשְׁכַּח רִישׁוֹ / לֹא תֵחַת בּוֹ גְעָרָה.


וּמַכַּת הַזָּנָב / אֲשֶר אֵלָיו יֶעְנַב / וְגָנֹב יִתְגַנַּב / יֶחְכַּם הִתְנַכֵּרָה

יֵצֵא בְּעוֹד לַיְלָה / בְּאִישׁוֹן וַאְפֵלָה / מִתְכַּסֶּה בַשִּׂמְלָה / וּבְיָדוֹ תְשׁוּרָה

וְיֵלֵךְ אֶל־בֵּיתָהּ / חֲבִיטָא מַבְרַכְתָּא / תְּשִׁיתֵהוּ בָתָה / מַפְרַקְתּוֹ שְׁבוּרָה

יִקְרַב אֶל־אֵשֶׁת אִישׁ / כָּל־מִשְׁפַּחְתּוֹ יַבְאִישׁ / בֵּן מַחְפִּיר וּמַבְאִישׁ / וְיֶעְרֶה מְקוֹרָהּ

יִקְרַב אֶל־הַגּוֹיָה / מַכָּתוֹ טְרִיָּה / אוֹי לַגְּוִיָּה / אֲשֶׁר בָּהּ הֶעֱרָה

יִקְרַב אֶל־הָעְרָיוֹת / וְאֶל־שְׁתֵי אֲחָיוֹת / לְאוֹחִים וּנְמִיּוֹת / יִֽהְיֶה לְבָרָה.


וְאֶל־כָּל־הַמַּכּוֹת / תִּמָּצֶאין אֲרוּכוֹת / וּרְפוּאוֹת עֲרוּכוֹת / לְכָל־גֵּו נִשְׁבָּרָה

כִּי עֵת אֲשֶר יִצְחֹק / וְלוֹ תְהִי לַחֹק / עָנְיוֹ אוֹתוֹ יִדְחֹק / וְיוּבֵל כַּפָּרָה

כִּי יִשָּׁאֵר עָרוּם / תַּחַת שִׁבְתּוֹ בְרוּם / וְתוֹלָעִים יָרוּם / יִשְׁתֶּה מֵי מְאֵרָה

יְמַלֵּט אִי רוֹקֶקֶת / כְּגֶפֶן בּוֹקֶקֶת / נַפְשׁוֹ לוֹ עוֹקֶקֶת / בְּרָעָב צְרוּרָה

גַּם אֶת־אֲשֶׁר יִצְרוֹ / וְאֵשׁ מְבַעֲרוֹ / וּפִיו מָרָה מָרֹה / לְאוֹמֶרֶת סוּרָה

אָז כִּי יִהְיֶה תָפֵל / נִשְּׁתוּ יַמֵּי עֹפֶל / עוֹמֵד אָבֵל אָפֵל / תָּבוֹא לוֹ מְגוּרָה

אַךְ אִישׁ בַּעַל קָנֶה / יִפְרֹץ וְלֹא יִבְנֶה / בֵּית הָאוֹפִים חוֹנֶה / בּוֹ בָחַר לְדִירָה

עַד פִּיו עָפָר יִקְרַב / עַד מַעְיָנוֹ יֶחְרַב / מֵעֵדֶן לוֹ יֶעְרֵב / נַפְשׁוֹ הִיא אֲרוּרָה

הֵן אֵל יְשׁוּעָתִי / יְחַזֵּק לִבָּתִי / לַהְגּוֹת בְּתוֹרָתִי / לְעָבְדָהּ וּלְשָׁמְרָהּ

אֶפְרַיִם בֶּן־יוֹאָב / בְּשָׂרוֹ לֹא יִכְאַב / וְנַפְשׁוֹ לֹא תִדְאַב / בְּיִרְאַת אֵל קְשׁוּרָה

אֶקְרָא אֶל־אַל עֶלְיוֹן / אֲשֶר לוֹ עֹז חֶבְיוֹן / בְּנֶחָמַת צִיּוֹן / אֶזְכֶּה לִרְאוֹת הֲדָרָה

יְקַבֵּץ פְּזוּרִים / יְכַנֵּס עֲדָרִים / יִשְׂרָאֵל חֲבֵרִים / יוּבַל שַׁי לַמּוֹרָא

בּוֹ אָשִׂים מָגִנִּי / וְהוּא יַעַזְרֵנִי / וְהוּא יְזַכֵּנִי / לְיוֹם הַבְּשׂרָה.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
