קְרָא שִׁיר זֶה לְנֶחָמָה

קְרָא שִׁיר זֶה לְנֶחָמָה / חכמת הראב״ע 76 (I)

שיר זה שלח להראב״ע ר׳ יוסף אבן צדיק כשפרסה נדה לכלתו בליל חופתה


> קְרָא שִׁיר זֶה לְנֶחָמָה > חֲתַן דָּמִים! אֲשֶׁר דָּמָה > לְאַיָּל עַל אֲפִיקֵי מַיִם > וְלֹא יוּכַל שְׁבוֹר צָמָא; > חֲדֵה עִם יַעֲלַת חֵשֶׁק, > יְפַת מַרְאֶה, יְפַת קוֹמָה. > וְאוּלָם שׁוּר וְאַל תִּגַּע, > כְּמוֹ תָשׁוּר פְּנֵי כִימָה; > וְאַל תַּסִּיג גְּבוּל, לָשׂוּם > דְּמֵי שָׁלוֹם בְּמִלְחָמָה, > וְהִתְעַצֵּב לְךָ, כִּי מִיץ > חֲלָבִים יַעֲלֶה חֵמָה, > וְכִי מִחוּץ לְךָ בִטְחָה, > וּמֵחֵדֶר לְךָ אֵימָה, > הֲלֹא עִתִּים אֲחָדִים הֵם > יְעוּפוּן כַּחֲלוֹם נוּמָה; > וְתַעַשׂ כֹּל אֲשֶׁר תּוּכַל, > לְעֵינֵי כֹל בְּיָד רָמָה, > וְנוֹדַע הַבְּמַחֲנִים אִם > בְּמִבְצָר עִירְךָ שׂוּמָה; > בְּיוֹם תָּבוֹא חֲמוּץ בֶּגֶד > וְשִׂמְלָתְךָ מְאָדָּמָה, > לְשֵׁמַע דָּאֲבָה נַפְשִׁי > וּמִתְאַנָּה כְּיָם הוֹמָה, > מְצֵרָה עַל קְדוֹשׁ אַחִים > הֱיוֹת בֵּין כֹּהֲנֵי בָמָה, > אֲשֶׁר נִשְׂגָּב וְאֵין בֵּינִי > וּבֵינוֹ שׁוּר וְלֹא חוֹמָה, > וְהָיִינוּ כְּמִתְגָּרִים > וְלֹא עַל מָה וְלֹא לָמָּה. > אֱלֹהִים יָחְנְךָ, אָחִי, > וְיַט לְךָ עֹז וְתַעֲצוּמָה, > וְעֵת יַעֲלֶה לְךָ מַעֲלָה, > תְּהִי עַל רֹאשׁ כְּסִיל רָמָה; > וְאִם תִּמְצָא אֲשֶׁר אֶמְצָא, > קְרָא שִׁיר זֶה לְנֶחָמָה. >
>
את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
