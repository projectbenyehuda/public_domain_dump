יוסף מתודע אל אחיו

וַיִגַש יְהוּדָה וַיאֹמֶר אֶל יוֹסֵף: יְדַבֶּר נָא עַבְדְךָ, וְאַל יְחַר בְּעֵינֵי אֲדוֹנִי. וַיאֹמָר לוֹ יוֹסֵף: דַבֵּר. וַיאֹמֶר יְהוּדָה: הֲלֹא בַתְּחִלָה כַּאֲשֶר בָּאנוּ אֶל אֲדוֹנֵנוּ לִשְבֹּר אֹכֶל וַתָּשֶׂם אוֹתָנוּ מְרַגְלִים אֶת הָאָרֶץ, לְמַעַן אֲשֶר נָבִיא לְךָ אֶת בִּנְיָמִין אָחִינוּ; וְכַאֲשֶר עָשִׂינוּ אֶת מִצְוָתְךָ עוֹדְךָ מְבַקֵש לְךָ תוֹאֲנוֹת לְהִתְעוֹלֵל עָלֵינוּ לְמַעַן אֲשֶר אָחִינוּ הַקָטָן יִשָאֵר בְּבֵיתְךָ וְלֹא יָשוּב אֶל אָבִינוּ הַזָקֵן. וְעַתָּה יִשְמַע נָא אֲדוֹנִי אֶת דְבָרָי: אֶשָאֵר נָא אָנֹכִי תַּחַת אָחִי, וְהַנַעַר יָשוּב עִם אֶחָיו אֶל אָבִיו. כִּי עַבְדְךָ עָרַב אֶת הַנַעַר לִפְנֵי עַבְדְךָ אָבִינוּ לֵאמֹר: אִם לֹא יָשוּב הַפַּעַם אֵלָיו וְחָטָאתִי לוֹ כָל הַיָמִים. וְאִם לֹא תַעַשׂ, אֲדוֹנִי, בַּקָשָתִי – חַי הָאֱלֹהִים, כִּי תְאַבֵּד אֶת נַפְשְךָ וְאֶת נֶפֶש בֵּיתְךָ הַיוֹם הַזֶה. הֲלֹא יָדַעְתָּ אִם לֹא שָמַעְתָּ אֶת אֲשֶר עָשׂוּ שְנֵי אַחַי שִמְעוֹן וְלֵוִי לָעִיר שְכֶם וְאֶת אֲשֶר הִכִּינוּ כֻלָנוּ לְפִי חָרֶב אֶת מַלְכֵי הָאֱמוֹרִי וְצִבְאוֹתֵיהֶם בַּעֲבוּר דִינָה אֲחוֹתֵינוּ, וְאֶת זֹאת נַעֲשֶׂה הַיוֹם גַם לְךָ וּלְבֵיתְךָ בַּעֲבוּר בִּנְיָמִין אָחִינוּ, אֲשֶר אַתָה אוֹמֵר לָקַחַת מֵאֵת אָבִינוּ הַזָקֵן וּלְהוֹרִיד אֶת שֵׂיבָתוֹ בְּיָגוֹן שְאוֹלָה. וְעַתָּה הָשִיבָה לָנוּ אֶת אָחִינוּ, וַאֲנִי אֵשֵב פֹּה תַּחַת הַנַעַר וְהוּא יָשוּב בְּשָלוֹם אֶל אָבִינוּ… וַיַעַן יוֹסֵף אֶת יְהוּדָה: לָמָה הִרְחַבְתָּ פִיךָ לְהִתְפָּאֵר בִּגְבוּרַתְכֶם? גַם אָנֹכִי גִבּוֹר כְּמוֹכֶם וּמִסָבִיב לִי גִבּוֹרֵי-חַיִל אֲשֶר הֶרְאוּ זֶה עַתָּה אֶת עוֹצֶם יָדָם בְּמִלְחַמְתֵּנוּ עִם בְּנֵי תַרְשִיש וְעָשִׂינוּ אוֹתָם מִרְמָס תַּחַת רַגְלֵינוּ. וְעַתָּה אַל תִּתְגָרוּ בְּרָעָה – לְכוּ לָכֶם אֶל אֲבִיכֶם, וְהַנַעַר אֲשֶר בְּאַמְתַּחְתּוֹ נִמְצָא הַגָבִיעַ הַגָנוּב יְהִי לִי לְעָבֶד, כַּחֹק בְּאַרְצֵנוּ! – וַיאֹמֶר יְהוּדָה: מַה לְךָ לְהַאֲשִים אוֹתָנוּ בִּגְנֵבָה! חָלִילָה לִבְנֵי אַבְרָהָם, יִצְחָק וְיַעֲקֹב מֵעֲשׂוֹת כַּדָבָר הַזֶה. אֵין זאֹת כִּי אַתָּה בְעַצְמְךָ שַׂמְתָּ אֶת הַגָבִיעַ בְּאַמְתַּחַת אָחִינוּ, כַּאֲשֶר עָשִׂיתָ בַכֶּסֶף בְּאַמְתְּחוֹת כֻּלָנוּ, לְהִתְגוֹלֵל עָלֵינוּ וּלְהִתְנַפֵּל עָלֵינוּ לְמַעַן אֲשֶר בִּנְיָמִין אָחִינוּ יִשָאֵר בְּבֵיתֶךָ… וַיִשְׁלְפוּ הָאַחִים אֶת חַרְבוֹתָם וַיֹאמְרוּ לְהִתְנַפֵּל עַל יוֹסֵף וְגִבּוֹרָיו וּלְהַצִיל מִידֵיהֶם אֶת בִּנְיָמִין. וַיִקְרָא אֲלֵיהֶם יוֹסֵף: אַל תִּשְפְּכוּ דָם! אוּלַי תִּמְכְּרוּ לִי אֶת אֲחִיכֶם הַקָטָן בִּמְחִיר אֲשֶר תָּשִיתוּ עָלָי; הֵן מָכוֹר מְכַרְתֶּם אֶת יוֹסֵף אֲחִיכֶם לְעֶבֶד לַיִשְמְעֵאלִים בְּעֶשְׂרִים כָּסֶף…

וַיוֹרִידוּ הָאַחִים אֶת חַרְבוֹתָם וְרָאשֵיהֶם אַרְצָה וַיַעַמְדוּ נִדְהָמִים לְמִשְמַע אָזְנֵיהֶם וְלֹא יָדְעוּ נַפְשָם… וְלֹא יָכוֹל יוֹסֵף לְהִהְאַפֵּק עוֹד וַיאֹמֶר לָהֶם: אֲנִי יוֹסֵף אֲחִיכֶם!… וְעַתָּה אַל יִחַר בְּעֵינֵיכֶם כּי מְכַרְתֶּם אוֹתִי הֵנָה, כִּי מֵאֵת ה' הָיְתָה זֹאת לְהַשְאִיר לָכֶם פְּלֵיטָה בָּאָרֶץ. – וַיִבָּהֲלוּ הָאַחִים מְאֹד וַיִבֹּכוּ בְּשָמְעָם אֶת דִבְרֵי יוֹסֵף, וּבִּנְיָמִין רָץ אֵלָיו וַיִפְּלוּ אִיש עַל צַוְארֵי אָחִיו וַיִבְכּוּ שְנֵיהֶם… וְאַחַר חִבֵּק יוֹסֵף גַם אֶת יֶתֶר אֶחָיו וַיְנַשְקוּ אִיש אָחִיו וַיִבְכּוּ כֻלָם בְּכִי גָדוֹל עַד מְאֹד.

וְהַקוֹל נִשְמַע בֵּית פַּרְעֹה לֵאמֹר: בָּאוּ אֲחֵי יוֹסֵף וְכִי הֵם בְּנֵי יַעֲקֹב נְשִׂיא הָעִבְרִים, וַיִשְלַח פַּרְעֹה אֶת שָׂרָיו לְבָרְכָם, וּלְיוֹסֵף שָלַח לְהַגִיד: אֱמוֹר לְאַחֶיךָ כִּי בְמֵיטַב אַרְצִי יוֹשִיבוּ נָא אֶת בְּנֵי מִשְפַּחְתָּם וְאֶת כָּל אֲשֶר לָהֶם. וַיָבֵא יוֹסֵף אֶת אֶחָיו לִפְנֵי פַרְעֹה, וַיַרְא הַמֶלֶךְ: וְהִנֵה כֻלָם גִבּוֹרִים וִיפֵי תֹאַר וַיאֹמֶר לְהַפְקִידָם לְשָׂרֵי צִבְאוֹתָיו. וַיֹאמְרוּ הָאַחִים לְמֶלֶךְ מִצְרַיִם: רַק אַנְשֵי מִקְנֶה עֲבָדֶיךָ וְלֹא עַל חַרְבֵּנוּ נִחְיֶה. וַיִתֵּן יוֹסֵף לְאֶחָיו עַשְתֵּי עֶשְׂרֵה מֶרְכָּבוֹת, וְגַם אֶת מִרְכֶּבֶת הַמִשְנֶה אֲשֶר רָכַב בָּהּ בְּיוֹם הִמָנוֹתוֹ לְמוֹשֵל מִצְרַיִם נָתַן לָהֶם לְהָבִיא בָּהּ אֶת יַעֲקֹב אָבִיו מֵאֶרֶץ כְּנַעַן מִצְרָיְמָה. וּמַתָּנוֹת רַבּוֹת נָתַן יוֹסֵף לְאֶחָיו וּלְכָל בְּנֵיהֶם וּבְנוֹתֵיהֶם, וַעֲשָׂרָה חֲמוֹרִים נוֹשְׂאִים מִכָּל טוּב אֶרֶץ מִצְרָיִם וַיִשַלְחֵם אַרְצָה כְּנַעַן לְהָבִיא אֶת כָּל בֵּיתָם וְצֹאנָם וּבְקָרָם וְכָל אֲשֶר לָהֶם אַרְצָה מִצְרַיִם לְהַצִילָם מִפְּנֵי הָרָעָב. וַיאֹמֶר לָהֶם: אַל תִּרְגְזוּ בַדֶרֶךְ וְאַל תָּרִיבוּ בֵּינֵיכֶם כִּי מְכַרְתֶּם אוֹתִי לְעָבֶד, כִּי לְמִחְיָה שְלָחַנִי ה' הֵנָה. וְעַתָּה בְשוּבְכֶם אֶל אָבִינוּ הַזָקֵן אַל תְּבַשְׂרוּ לוֹ פִתְאוֹם כִּי עוֹדֶנִי חָי, פֶּן יִבָּהֵל וִיבֻלַע לוֹ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
