מלחמת בני יעקב עם מלכי האמורי

וְעֵשָׂו וּבָנָיו וְכָל אֲשֶר לוֹ הָלְכוּ אַרְצָה שֵׂעִיר וַיֵאָחֲזוּ בָה וַיִפְרוּ וַיִרְבּוּ מְאֹד. וְאֵלֶה בְּנֵי עֵשָׂו אֲשֶר נוֹלְדוּ לוֹ בְּאֶרֶץ כְּנָעַן: וַתֵּלֶד עָדָה אֶת אֱלִיפַז בְּכוֹרוֹ וּבָשְׂמַת, הִיא מָחֲלַת, יָלְדָה לוֹ אֶת רְעוּאֵל, וָאָהֳלִיבָמָה יָלְדָה אֶת יְעוּש וְאֶת יַעֲלֶם וְאֶת קֹרַח. וְאֵלֶה בְּנֵי שְׂעִיר הַחוֹרִי יוֹשְבֵי אֶרֶץ שְׂעִיר: לוֹטֶן וְשוֹבָל וְצִבְעוֹן וַעֲנָה וְדִשוֹן וְאֵצֶר וְדִישָן. וְאֵלֶה בְּנֵי לוֹטָן: חוֹרִי וְהֵימָן, וַאֲחוֹת לוֹטָן תִּמְנָע, אֲשֶר הָיְתָָה פִּילֶגֶש לֶאֱלִיפַז בֶּן עֵשָׂו. וְאֵלֶה בְּנֵי צִבְעוֹן: אַיָה וַעֲנָה. הוּא עֲנָה אֲשֶר מָצָא אֶת הַיִמִים1 בַּמִדְבָּר בִּרְעוֹתוֹ אֶת הַחֲמוֹרִים לְצִבְעוֹן אָבִיו: וַיְהִי הַיוֹם, בִּרְעוֹתוֹ אֶת עֵדֶר חֲמוֹרָיו בַּמִדְבָּר כְּפַעַם בְּפַעַם וַיוֹלִיכֵם אֶל מִדְבַּר יָם־סוּף, וְהִנֵה רוּחַ סְעָרָה חֲזָקָה מְאֹד בָּאָה מֵעֵבֶר הַיָם וַתָּנַח עַל הַחֲמוֹרִים הָרוֹעִים וַיַעַמְדוּ כֻלָם בִּמְקוֹם מַרְעִיתָם. וּפִתְאֹם יָצְאוּ מֵעֵבֶר הַיָם כְּעֶשְׂרִים וּמֵאָה חַיוֹת גְדוֹלוֹת וְנוֹרָאוֹת עַד מְאֹד וַיָבוֹאוּ אֶל מְקוֹם הַחֲמוֹרִים וַיִתְיַצְבוּ אֶצְלָם. וְהַחַיוֹת הָהֵן מִמַחֲצִיתָן וּלְמַטָה הָיוּ כְּתֹאַר בְּנֵי־אָדָם וּמִמַחֲצִיתָן וּלְמַעְלָה – מֵהֶן כִּדְמוּת דֻבִּים, וּמֵהֶן כִּדְמוּת קוֹפִים וּזְנָבוֹת לָהֶן מֵאֲחוֹרֵיהֶן כִּזְנַב הַדוּכִיפַת. וַיָקוּמוּ הַחַיוֹת הָאֵלֶה וַיִרְכְּבוּ עַל הַחֲמוֹרִים וַיֵעָלְמוּ כֻלָם בַּמִדְבָּר; וְאַחַת מֵהַחַיוֹת נִגְשָה אֶל עֲנָה וַתַּכֵּהוּ בִּזְנָבָהּ וַתִּבְרַח גַם הִיא. וַיָרָץ עֲנָה לְבֵיתוֹ וַיְסַפֵּר אֶת כָּל הַמַעֲשֶׂה לְצִבְעוֹן אָבִיו, וַיֵלְכוּ אֲנָשִים רַבִּים לְבַקֵש אֶת הַחַיוֹת וְאֶת הַחֲמוֹרִים וְלֹא מָצָאוּ. מֵאָז לֹא יָסְפוּ עוֹד עֲנָה וְאֶחָיו לִרְעוֹת אֶת עֶדְרֵיהֶם בַּמָקוֹם הַהוּא כִּי יָרְאוּ מְאֹד לְנַפְשוֹתָם. –

וַיְהִי בַּשָנָה הַתְּשִיעִית לְשוּב יַעֲקֹב מִפַּדַן־אֲרָם אַרְצָה כְּנַעַן וַיִסַע מֵחֶבְרוֹן, הוּא וְכָל אֲשֶר לוֹ, וַיֵלְכוּ וַיָבוֹאוּ עַד סְבִיבוֹת שְׁכֶם וַיֵשְבוּ שָם, כִּי מָצְאוּ לָהֶם מִרְעֶה טוֹב וְשָמֵן לְמִקְנֵיהֶם בַּמָקוֹם הַהוּא. וְעִיר שְכֶם שָבָה וְנִבְנְתָה בָּעֵת הַהִיא וַיִהְיוּ בָהּ כִּשְלֹש מֵאוֹת אִיש וְאִשָה. וַיָבֹאוּ יַעֲקֹב וּבָנָיו וַעֲבָדָיו וַיִתְיַשְבוּ בְּחֶלְקַת הַשָׂדֶה אֲשֶר קָנָה מֵאֵת שְכֶם בֶּן חֲמוֹר. וַיִשְמְעוּ מַלְכֵי הַכְּנַעֲנִי וְהָאֱמוֹרִי, כִּי שָבוּ יַעֲקֹב וּבֵיתוֹ אֶל גְבוּל שְכֶם, וַיִזְכְּרוּ אֶת אֲשֶר עָשׂוּ בְּנֵי יַעֲקֹב לְיוֹשְבֵי הָעִיר, וַיִתְקַבְּצוּ שֵנִית עִם מַחֲנֵיהֶם הָרַבִּים לְהִלָחֵם בָּהֶם. וַיִשְלַח יָשוּב מֶלֶךְ תַּפּוּחַ אֶל כָּל הַמְלָכִים אֲשֶר סְבִיבוֹתָיו: אֶל אֵילוֹן מֶלֶךְ גַעַש וְאֶל אֵיהוֹרֵי מֶלֶךְ שִילֹה, וְאֶל פִּרְעָתוֹן מֶלֶךְ חֵפֶר וְאֶל סוּסִי מֶלֶךְ סַרְטָן וְאֶל לָבָן מֶלֶךְ בֵּית־חוֹרוֹן וְאֶל שָׂכָר מֶלֶךְ מַחֲנַיִם לֵאמֹר: עֲלוּ אֵלַי וְעִזְרוּנִי וְנַכֶּה אֶת יַעֲקֹב הָעִבְרִי וְאֶת בָּנָיו וְאֶת כָּל אֲשֶר לוֹ, כִּי בָאוּ שֵנִית שְכֶמָה לָרֶשֶת אוֹתָהּ וְלַהֲרוֹג אֶת יוֹשְבֶיהָ כְּבָרִאשוֹנָה. וַיִתְקַבְּצוּ יַחַד כָּל הַמְלָכִים הָאֵלֶה עִם מַחֲנֵיהֶם הָרַבִּים כַּחוֹל שֶעַל שְׂפַת הַיָם, וַיַחֲנוּ כֻלָם נֶגֶד תַּפּוּחַ בְּשֶבַע מַעֲרָכוֹת. וַיִשְלְחוּ סֵפֶר לְיַעֲקֹב וְאֶל בָּנָיו לֵאמֹר: צְאוּ אֵלֵינוּ וְנִתְרָאֶה עִמָכֶם פָּנִים בַּמִישוֹר וְנִנָקְמָה מִכֶּם אֶת נִקְמַת אַנְשֵי שְכֶם אֲשֶר הֲרַגְתֶּם. וַיִשְמְעוּ בְּנֵי יַעֲקֹב אֶת דִבְרֵי הַמְלָכִים וַיִחַר אַפָּם מְאֹד, וַיְמַהֲרוּ וַיָקוּמוּ וַיַחְגְרוּ אִיש־אִיש אֶת כְּלֵי מִלְחַמְתּוֹ, הֵם וְעַבְדֵיהֶם כְּמֵאָה וּשְנֵים עָשָׂר אִיש שוֹלֵף חָרֶב, וַיֵצְאוּ לִקְרַאת הַמְלָכִים. וְגַם יַעֲקֹב נִלְוָה הַפַּעַם אֲלֵיהֶם וַיִתְפַּלֵל לַה' וַיֹאמַר: “אֵל שַדָי! הַצֵל נָא בְּרַחֲמֶיךָ הָרַבִּים אֶת בָּנַי וַאֲנָשַי מִיַד אוֹיְבֵיהֶם הָרַבִּים וְהָעֲצוּמִים מֵהֶם, כִּי בְיָדְךָ הַכֹּחַ וְהַגְבוּרָה לְהוֹשִיעַ בְּרָב אוֹ בִמְעָט; וְנָתַתָּ לְבָנַי וְלַעֲבָדַי חֹזֶק־לֵב וָעֹז לְהִלָחֵם בַּקָמִים עֲלֵיהֶם וּלְהַכְנִיעָם. וְאִם טוֹב בְּעֵינֶיךָ לָקַחַת אֶת נַפְשוֹת בָּנַי וַעֲבָדַי – קָחֶם נָא בְּיַד מַלְאָכֶיךָ הַטוֹבִים בְּרַחֲמֶיךָ הָרַבִּים, אַךְ בְּיַד מַלְכֵי הָאֱמוֹרִי אַל נָא תַפִּילֵם הַיוֹם”. – וּבְנֵי יַעֲקֹב קָרְבוּ וַיִגְשוּ עַד מַצַב הַמְלָכִים וְצִבְאוֹתֵיהֶם – לְהִלָחֵם אִתָּם. וַיָרָץ יְהוּדָה רִאשוֹנָה לִפְנֵי אֶחָיו וַעֲשָׂרָה אֲנָשִים מֵעֲבָדָיו עִמוֹ וַיאֹמְרוּ לְהִתְנַפֵּל עַל יָשוּב מֶלֶךְ תַּפּוּחַ וְחֵילוֹ. וְיָשוּב רָכַב עַל סוּס אַבִּיר, וְשָׂרָיו וַעֲבָדָיו מִסָבִיב לוֹ, וְחוּא כֻלוֹ מְכֻסֶה מֵראֹשוֹ וְעַד רַגְלָיו בַּרְזֶל וּנְחֹשֶת, וְיָדַע לִירוֹת חִצִים בִּשְתֵּי יָדָיו פָּנִים וְאָחוֹר, כְּמִשְפָּטוֹ בְּכָל מִלְחֲמוֹתָיו – וְלֹא יַחֲטִיא. וַיוֹר יָשוּב חִצִים אֶל יְהוּדָה בְּמֶרְחַק שְלֹשִים אַמָה, וַיַחֲטֹף יְהוּדָה אֶת הַחִצִים בְּיָדָיו וַיַשְלִיכֵם עַל רָאשֵי אַנְשֵי יָשוּב. וַיְהִי כִּרְאוֹת יְהוּדָה כִּי יָשוּב הוֹלֵךְ וּמִתְקָרֵב אֵלָיו לְהַכּוֹתוֹ בְכִידוֹנוֹ וַיָרֶם אֶבֶן גְדוֹלָה מְאֹד מִן הָאָרֶץ וַיַשְלִיכָהּ עַל יָשוּב וַיִדָהֵם יָשוּב מֵעֹצֶם הַמַכָּה וַיִפּוֹל מֵעַל סוּסוֹ מְלֹא קוֹמָתוֹ אַרְצָה, וְהַמָגֵן אֲשֶר עָלָיו נִקְרַע לִשְנַיִם וַיָשְלַךְ מֵעָלָיו עַד מַצַב הַמַחֲנֶה הַשְנִיָה בְּמֶרְחַק חֲמֵש־עֶשְׂרֵה אַמָה. וְיֶתֶר הַמְלָכִים בִּרְאוֹתָם אֶת אֲשֶר עָשָׂה יְהוּדָה לְמֶלֶךְ תַּפּוּחַ מִהֲרוּ וַיָבֹאוּ עִם אַנְשֵיהֶם לְעֶזְרַת בַּעל־בְּרִיתָם, אַךְ יְהוּדָה הִתְנַפֵּל עֲלֵיהֶם בַּחֲמָתוֹ וַיַךְ בָּהֶם כְּעֶשְׂרִים אִיש, וְהַנִשְאָרִים נִבְהֲלוּ וַיִמָלְטוּ עַל נַפְשָם. וַיְהִי כִּרְאוֹת יָשוּב כִּי אֲנָשָיו עֲזָבוּהוּ לְנַפְשוֹ וַיְמַהֵר וַיָקָם עַל רַגְלָיו וַיִתְנַפֵּל עַל יְהוּדָה בְּשֶצֶף־קֶצֶף וַיָטֶל אֶת כִּידוֹנוֹ מוּל יְהוּדָה וַיאֹמֶר לְהַכּוֹתוֹ בְרֹאשוֹ. וַיָרֶם יְהוּדָה אֶת מָגִנוֹ מֵעַל לְרֹאשוֹ וַיְפַלַח הַכִּידוֹן אֶת מָגֵן יְהוּדָה לִשְנָיִם. וַיִתְחַזֵק יְהוּדָה וַיִשְלֹף חִיש אֶת חַרְבּוֹ וַיַךְ בָּהּ אֶת יָשוּב בְּקַרְסֻלָיו, וַיִכְרַע יָשוּב אַרְצָה וְכִידוֹנוֹ נָפַל מִיָדוֹ, וַיֶמַהֵר יְהוּדָה וַיִכְרֹת אֶת ראֹשוֹ מֵעָלָיו וַיַשְלִיכֶנוּ אֶל רַגְלָיו. וַיְהִי כִּרְאוֹת בְּנֵי יַעֲקֹב וְעַבְדֵיהֶם אֶת גְבוּרַת יְהוּדָה וַתִּתְעוֹדֵד בָּם רוּחָם וַיִלָחֲמוּ גַם הֵם בָּעֹז וַיַכּוּ בְּאַנְשֵי הַכְּנַעֲנִי וְהָאֱמוֹרִי בְּלִי חֶמְלָה וַיַפִּילוּ בָּהֶם כְּט"ו אֶלֶף אִיש, וְהַנִשְאָרִים נָפוֹצוּ לִכָל עִבֶר.

וְלֵוִי בֶּן יַעֲקֹב עָרַךְ מִלְחָמָה, הוּא וְי"ב מֵעֲבָדָיו, נֶגֶד אֵילוֹן מֶלֶךְ גַעַש וְשָׂרָיו – וַיַכּוּם לְפִי חָרֶב. וְאוּהִירִי מֶלֶךְ שִילֹה מָשַךְ בְּקַשְתּוֹ מוּל יַעֲקֹב, וַיִדְרֹךְ גַם יַעֲקֹב אֶת קַשְתּוֹ וַיוֹר בְּאוּהִירִי חֵץ וַיְמִיתֵהוּ. וַיִרְאוּ יֶתֶר הַמְלָכִים כִּי שְלֹשָה מֵאֲחֵיהֶם בַּעֲלֵי־בְרִיתָם נָפְלוּ בְּחֶרֶב בְּנֵי־יַעֲקֹב וַיִירְאוּ מְאֹד לְנַפְשוֹתָם וַיִבְרְחוּ וַיָנוּסוּ עַד עִיר חֵפֶר. וַיִרְדְפוּ בְּנֵי יַעֲקֹב וְאַנְשֵיהֶם עַד שַעַר הָעִיר וַיַכּוּ בְאוֹיְבֵיהֶם כְּאַרְבַּעַת אֲלָפִים אִיש, וּבֵינֵיהֶם נָפְלוּ חֲלָלִים גַם יֶתֶר הַמְלָכִים, וְהַנִשְאָרִים הִתְחַמְקוּ אֶל תּוֹךְ הָעִיר דֶרֶךְ הַשַעַר וַיִתְחַבְּאוּ בָהּ, כִּי חֵפֶר עִיר גְדוֹלָה לֵאלֹהִים וְרַחֲבַת־יָדַיִם מְאֹד. אַךְ בְּנֵי יַעֲקֹב רָדְפוּ אַחֲרֵיהֶם וַיֹאמְרוּ לָבוֹא הָעִירָה וְלַהֲשִׂיגָם. וַיֵצְאוּ מֵהָעִיר אַרְבָּעָה גִבּוֹרִים אַנְשֵי־מִדוֹת וַיַעַמְדוּ נֶגֶד פֶּתַח הַשַעַר וְחַרְבוֹתָם שְלוּפוֹת בִּידֵיהֶם וַיִתְיַצְבוּ נֶגֶד בְּנֵי יַעֲקֹב וְלֹא נְתָנוּם לָבוֹא הָעִירָה. וַיָרָץ אֲלֵיהֶם נַפְתָּלִי וַיִדָחֵק בֵּין אַרְבַּעַת הַגִבּוֹרִים וַיַךְ בְּחַרְבּוֹ אֶת שְנַיִם מֵהֶן וַיִכְרֹת אֶת שְנֵי רָאשֵיהֶם בְּפַעַם אֶחָת, וְהַשְנַיִם הָאֲחֵרִים בָּרְחוּ עַל נַפְשָם. וַיִרְדֹף נַפְתָּלִי אַחֲרֵיהֶם וַיַדְבִּיקֵם וַיָמֶת גַם אוֹתָם. אָז בָּאוּ בְּנֵי יַעֲקֹב הָעִירָה וַיִרְאוּ וְהִנֵה עוֹד חֵמָה אַחַת מִסָבִיב לָעִיר וַיְבַקְשוּ אֶת מְבוֹא הָעִיר, וַיַעֲלוּ אַחֲרָיו גַם שִמְעוֹן וְלֵוִי וַיֵרְדוּ שְלָשְתָּם מֵעַל הַחוֹמָה אֶל תּוֹךְ הָעִיר, וְאַחֲרֵיהֶם בָּאוּ יֶתֶר אֲחֵיהֶם וְעַבְדֵיהֶם וַיָבוֹזוּ אֶת שְלַל הָעִיר וַיֵלֵכוּ.

וַיְהִי בַּיוֹם הַשֵנִי וַיַעֲלוּ בְּנֵי יַעֲקֹב סַרְטָנָה, כִּי שָמְעוּ כִּי אֲנָשֶיהָ אֲשֶר נִשְאֲרוּ בָהּ נֶאֶסְפוּ לְהִלָחֵם אִתָּם שֵנִית. וְסַרְטָן עִיר גְבֹהָה וּבְצוּרָה עַד מְאֹד וְלָהּ חֵל עֹמֶק מִסָבִיב וְעֹמֶק הַחֵל חֲמִשִים אַמָה וְרָחְבּוֹ אַרְבָּעִים אַמָה. וַיְבַקְשוּ בְּנֵי יַעֲקֹב דֶרֶךְ לָבוֹא הָעִירָה וְלֹא מָצָאוּ, כִּי סָגַר עֲלֵיהֶם הַחֵל. וְיוֹשְבֵי סַרְטָן – כְּאַרְבַּע מֵאוֹת אִיש – עָלוּ עַל הַחוֹמָה וַיִרְאוּ וְהִנֵה בְּנֵי יַעֲקֹב מְבַקְשִים לַשָוְא אֶת מְבוֹא־הָעִיר, וַיְחָרְפוּ אוֹתָם וַיְקַלְלוּם וַיִלְעֲגוּ לָהֶם. וַתִּבְעַר חֲמַת בְּנֵי יַעֲקֹב עֲלֵיהֶם וַיְמַהֲרוּ וַיְדַלְגוּ מֵעַל הַחֵל וַיַעַבְרוּ בִּקְפִיצָה אֶחָת אֶת אַרְבָּעִים הָאַמָה וַיַעַמְדוּ תַּחַת חוֹמַת הָעִיר. וַיִרְאוּ וְהִנֵה כָּל שְעָרֶיהָ סְגוּרִים מִפְּנֵיהֶם בְּדַלְתֵי בַרְזֶל וּבְרִיחִים עָבִים עַד מְאֹד. וַיִגְשוּ בְּנֵי יַעֲקֹב לְשַבֵּר אֶת הַשְעָרִים וְלֹא נְתָנוּם יוֹשְבֵי הָעִיר, כִּי הִשְלִיכוּ עֲלֵיהֶם אֲבָנִים וְחִצִים מֵעַל הַחוֹמָה. וַיִקְפֹּץ יְהוּדָה קְפִיצָה נוֹרָאָה אַחַת מִזְרָחָה וַיַעַל עַל הַחוֹמָה, וְאַחֲרָיו קָפַץ גָד וַיַעַל יָמָה וְשִמְעוֹן וְלֵוִי צָפוֹנָה וְדָן וּרְאוּבֵן נֶגְבָּה. וְיוֹשְבֵי סַרְטָן בִּרְאוֹתָם אֶת גְבוּרַת בְּנֵי יַעֲקֹב הַמַפְלִיאָה וַיָנוּסוּ וַיָפוּצוּ כֻלָם מֵעַל הַחוֹמָה. וַיִגְשוּ יִשָׂכָר וְנַפְתָּלִי וַיַצִיתוּ אֵש בְּשַעֲרֵי הָעִיר וַיִמַס הַבַּרְזֶל וַיָבֹאוּ כָּל בְּנֵי יַעֲקֹב וְאַנְשֵיהֶם הָעִירָה. וַיַעַבְרוּ חוֹצֵץ בִּרְחוֹבוֹת הָעִיר וַיִרְאוּ וְהִנֵה מִגְדָל גָבֹהַ וְחָזָק מְאֹד בְּתוֹךְ הָעִיר וְהוּא מָלֵא אֲנָשִים, נָשִים וָטַף כִּשְלֹש מְאוֹת אִיש אֲשֶר הִתְבַּצְרוּ בוֹ. וַיִלָחֲמוּ בָּהֶם בְּנֵי יַעֲקֹב וַיַבְרִיחֵם מֵהַמִגְדָל וַיַכּוּם לְפִי חָרֶב. וַיֵצְאוּ אֲלֵיהֶם מֵהַמִגְדָל י"ב גִבּוֹרִים מִבְּנֵי הָעֲנָקִים וַיַעַרְכוּ מִלְחָמָה נֶגֶד שִמְעוֹן וְלֵוִי וַיִכְרְתוּ אֶת מָגִנֵיהֶם, וְאֶחָד מֵהֶם אָמַר לְהַכּוֹת בְּחַרְבּוֹ עַל רֹאש לֵוִי. וַיְמַהֵר לֵוִי וַיַחֲטֹף אֶת חֶרֶב הָעֲנָק מִיָדוֹ וַיִכְרָת בָּהּ אֶת רֹאשוֹ וַיָסֹבּוּ אַחַד עָשָׂר הָעֲנָקִים אֶת שִמְעוֹן וְלֵוִי וַיאֹמְרוּ לְהַכּוֹתָם נָפֶש. וַתִּכְבַּד הַמִלְחָמָה עַל שִמְעוֹן וְלֵוִי בַּפַּעַם הַזאֹת, וַיִצְעַק שִמְעוֹן צְעָקָה גְדוֹלָה. וַיַכֵּר יְהוּדָה מֵרָחוֹק אֶת קוֹל אֶחָיו וַיְמַהֵר לְעֶזְרָתָם, וַיִלָחֲמוּ שְלָשְתָּם נֶגֶד הַגִבּוֹרִים כָּל הַיוֹם וְלֹא יָכְלוּ לָהֶם. וַיְמַהֵר יִשָׂכָר וְנַפְתָּלִי וַיִתְנַפְּלוּ עֲלֵיהֶם מֵאֲחוֹרֵיהֶם וַיַהַרְגוּ שְלֹשָה מֵהֶם, וְהַנִשְאָרִים רָאוּ כִּי הָיְתָה לָהֶם הַמִלְחָמָה מִפָּנִים וּמֵאָחוֹר וַיִמָלְטוּ עַל נַפְשָם.

וַיְהִי בַּיוֹם הַשְלִישִי וַיֵצְאוּ לִקְרַאת בְּנֵי יַעֲקב אַנְשֵי תַפּוּחַ לְהִלָחֵם בָּהֶם, וּבַיוֹם הָרְבִיעִי חָבְרוּ אֲלֵיהֶם גַם יוֹשְבֵי אַרְבֵּל הֵם וּנְשֵיהֶם, אֲשֶר הָיוּ כֻלָן מְלֻמְדוֹת־מִלְחָמָה. וַתְּהִי הַמִלְחָמָה חֲזָקָה כָּל הַיוֹם הַהוּא וְרַק לְעֵת עֶרֶב הִכְרִיעוּם בְּנֵי יַעֲקֹב לְפִי חַרְבָּם, וַיָבֹזוּ אֶת שְלַל שְתֵּי הֶעָרִים הָאֵלֶה וַיֵלְכוּ. וַיְהִי בַּיוֹם הַחֲמִישִי וַיֵצְאוּ לִקְרָאתָם אַנְשֵי גַעַש לְהִלָחֵם בָּהֶם. וְגַעַש עִיר גְדוֹלָה וּבְצוּרָה עַד מְאֹד וְשָלֹש חוֹמוֹת מִסָבִיב לָהּ וְכָל אֲנָשֶיהָ גִבּוֹרִים מִבְּנֵי הָעֲנָקִים. וַיִגְשוּ בְּנֵי־יַעֲקֹב אֶל הַחוֹמָה וַיִרְאוּ וְהִנֵה שַעֲרֵי הָעִיר סְגוּרִים וּמְסֻגָרִים וְיוֹשְבֶיהָ, כַּחֲמֵש מְאוֹת אִיש, עָלוּ עַל הַחוֹמָה הַחִיצוֹנָה לְהִלָחֵם בָּהֶם. וְעַם רַב מֵהֶם אָרְבוּ לָהֶם מִחוּץ לָעִיר וַיַקִיפוּם מֵאֲחוֹרֵיהֶם, וַתְּהִי הַמִלְחָמָה לִבְנֵי יַעֲקֹב מִפָּנִים וּמֵאָחוֹר. וַיַרְא יְהוּדָה כִּי כָבְדָה הַמִלְחָמָה עָלָיו וְעַל אֶחָיו וַיִשְאַג בְּקוֹל אַרְיֵה, וַיִבָּהֲלוּ מִשַאֲגָתוֹ אַנְשֵי גַעַש וְרַבִּים מֵהֶם נָפְלוּ מֵעַל הַחוֹמָה מִפַּחַד וְעָצְמַת־הַקוֹל. וַיִגְשוּ בְּנֵי יַעֲקֹב לְשַבֵּר אֶת דַלְתֵי הַשַעַר וַיַשְלִיכוּ עֲלֵיהֶם יוֹשְבֵי הָעִיר אֲבָנִים וְחִצִים מֵעַל הַחוֹמָה וַיַבְרִיחוּם. וַיְחָרְפו בְּנֵי גַעַש אֶת בְּנֵי יַעֲקֹב וַיְגַדְפוּם, וַיְקַלְלוּם בֵּאלֹהִים. וַיִחַר אַף יְהוּדָה וַיְקַנֵא לֵאלֹהָיו וַיָרָץ מֵרָחוֹק בְּכָל כֹּחַ גְבוּרָתוֹ, וְחַרְבּוֹ שְלוּפָה בְּיָדוֹ, וַיְדַלֵג וַיַעַל בִּקְפִיצָה אַחַת עַל הַחוֹמָה, אַךְ מִכֹּחַ הַקְפִיצָה נָפְלָה הַחֶרֶב מִיָדוֹ אָרְצָה. וַיִרְאוּ אַנְשֵי גַעַש אֲשֶר עַל הַחוֹמָה, כִּי אֵין חֶרֶב בְּיַד יְהוּדָה וַיָסֹבּוּ אוֹתוֹ וַיאֹמְרוּ לְהַפִּילוֹ מֵעַל הַחוֹמָה. וְיַעֲקֹב וּבָנָיו מָשְׁכוּ בְּקַשְתוֹתָם מִתַּחַת לַחוֹמָה וַיַכּוּ שְלֹשָה מֵהָאֲנָשִים הַסוֹבְבִים אֶת יְהוּדָה. וַיִשְאַג יְהוּדָה שֵנִית בְּקוֹל־אַרְיֵה וַיָהָם אֶת הַנִשְאָרִים וַיָנוּסוּ אֶל הַחוֹמָה הַשְנִיָה. וְאֶחָד מִגִבּוֹרֵי הָעִיר גַעַש וּשְמוֹ עָרוּד זִנֵק פִּתְאֹם מוֹל יְהוּדָה וַיַךְ בְּחַרְבּוֹ עַל רֹאשוֹ. וַיִצְעַק יְהוּדָה מֵעָצְמַת הַכְּאֵב, וַיִדָהֵם הַגִבּוֹר מִקוֹל הַצְעָקָה וַיִמָעֲדוּ קַרְסֻלָיו וַיִפּוֹל אֶל תַּחַת הַחוֹמָה בֵּין בְּנֵי יַעֲקֹב, וַיַכּוּהוּ שָם וַיְמִיתוּהוּ. וַיִכְאַב מְאֹד ראש יְהוּדָה מֵעָצְמַת הַמַכָּה, וַיִצְעַק מָרָה בְּמַכְאוֹבוֹ. וַיִשְמַע דָן אֶת קוֹל יְהוּדָה וַיְמַהֵר וַיְדַלֵג מֵהָאָרֶץ קְפִיצָה אֶחָת וַיַעַל עַל הַחוֹמָה לְעֶזְרַת אָחִיו. וְאַחֲרָיו קָפְצוּ וְעָלוּ נַפְתָּלִי וְיִשָׂכָר וּזְבוּלֻן, וַיִקְפְּצוּ כֻלָם קְפִיצָה אֶחָת וַיַעַבְרוּ וַיַעֲלוּ עַל הַחוֹמָה הַשְנִיָה. וַיִרְאוּ אַנְשֵי גַעַש אֶת גְבוּרַת בְּנֵי יַעֲקֹב הַמַפְלִיאָה וַיָנוּסוּ מִפְּנֵיהֶם. וַיִרְדְפוּ אַחֲרֵיהֶם בְּנֵי יַעֲקֹב וַיַכּוּם לְפִי חָרֶב. וַיְהִי בַּיוֹם הַשִשִי וַיֵצְאוּ אַנְשֵי בֵית־חוֹרוֹן לְהִלָחֵם בִּבְנֵי יַעֲקֹב, כִּי שָמְעוּ אֲשֶר עָיְפוּ מְאֹד מֵחָמֵש הַמִלְחָמוֹת הַקוֹדְמוֹת, וַיִקְרְאוּ אַנְשֵי בֵּית־חוֹרוֹן לְעֶזְרָתָם גַם אֶת יוֹשְבֵי שִילֹה וּבְנוֹתֶיהָ. וַיָהָם ה' אֶת מַחֲנֵה בֵית חוֹרוֹן וְשִילֹה וַיִלָחֲמוּ כָּל הַלַיְלָה אִיש בְּאָחִיו. וַתְּהִי הַמְהוּמָה וְהַמְבוּסָה גְדוֹלָה מְאֹד בְּמַחֲנֵיהֶם וְקוֹל צַעֲקוֹתֵיהֶם נִשְמַע גַם בְּעֵבֶר הַיַרְדֵן מִזְרָחָה. וַיְהִי בַּבֹּקֶר וַיִרְאוּ בְּנֵי יַעֲקֹב כִּי כָל אַנְשֵי בֵית־חוֹרוֹן וְשִילֹה פְגָרִים מֵתִים, וַיָבִינוּ כִּי ה' הִפִּיל עֲלֵיהֶם מְהוּמָה וּמְבוּכָה וַיִלָחֲמוּ אִיש בְּאָחִיו וְאִיש בְּרֵעֵהוּ.

וַיָשוּבוּ בְּנֵי יַעֲקֹב בַּיוֹם הַשְבִיעִי שְכֶמָה. וַיְהִי כִּשְמֹעַ יֶתֶר מַלְכֵי כְנַעַן כִּי הִכּוּ בְּנֵי יַעֲקֹב אֶת כָּל אוֹיְבֵיהֶם מִסָבִיב וַיִשְלְחוּ אֲלֵיהֶם מַלְאֲכֵי־שָלוֹם, כִּי יִכְרְתוּ אִתָּם בְּרִית. וַיַשְלִימוּ בְּנֵי יַעֲקֹב לָהֶם וַיִשָבְעוּ אִיש לְאָחִיו. וַיְהִי שְלוֹם־אֱמֶת בֵּין בְּנֵי יַעֲקֹב וּבֵין מַלְכֵי אֶרֶץ כְּנָעַן.


  1. האימים; האגדה מתכונת, כנראה, לקנטברים או לעם פרא שישב אז במדבר ערב.  ↩
