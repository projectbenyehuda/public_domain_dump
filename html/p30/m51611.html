יצחק בבית שם ועבר

וַיְהִי כַּאֲשֶר עָבְרוּ יְמֵי אֵבֶל שָׂרָה וַיִשְלַח אַבְרָהָם אֶת יִצְחָק בְּנוֹ אֶל בֵּית שֵם וָעֵבֶר. וַיֵשֶב שָם יִצְחָק שָלֹש שָנִים וַיוֹסֶף לִלְמֹד מִפִּיהֶם אֶת תּוֹרַת ה' וְאֵת מוּסָרָהּ הַטוֹב. בָּעֵת הַהִיא מֵת אֲבִימֶלֶךְ מֶלֶךְ פְּלִשְתִּים וַיֵלֶךְ אַבְרָהָם עִם אֲנָשָיו אַרְצָה פְּלִשְתִּים לְנַחֵם אֶת בֵּית אֲבִימֶלֶךְ, וַיִפֶן מִשָם וַיָשָב אֶל בֵּיתוֹ אֲשֶר בִּבְאֵר־שָבַע. וַיְהִי אַחֲרֵי מוֹת אֲבִימֶלֶךְ וַיַמְלִיכוּ אַנְשֵי גְרָר אֶת נַמֶלֶךְ בְּנוֹ תַּחְתָּיו; וְהוּא אָז בֶּן שְתֵּים עֶשְׂרֵה שָנָה, וַיִקְרְאוּ הַפְּלִשְתִּים גַם אֶת שְמוֹ אֲבִימֶלֶךְ כְּשֵם אָבִיו, כִּי כֵן מִשְפָּטָם לִקְרֹא לְמַלְכֵיהֶם בְּשֵם “אֲבִימֶלֶךְ”. וְגַם לוֹט בֶּן הָרָן מֵת בַּיָמִים הָהֵם, בִּשְנַת הַשְלֹשִים וָתֵשַע לְחַיֵי יִצְחָק, וַיִהְיוּ כָּל יְמֵי לוֹט מֵאָה שָנָה וּשְתַּיִם וְאַרְבָּעִים שָנָה וַיָמוֹת. וְאֵלֶה בְּנֵי לוֹט, אֲשֶר יָלְדוּ בְנוֹתָיו: עַמוֹן וּמוֹאָב: וַיִוָלְדוּ גַם לָהֶם בָּנִים וּבָנוֹת, וַיִפְרוּ וַיִרְבּוּ מְאֹד וַיֵלְכוּ וַיִבְנוּ לָהֶם עָרִים בָּאָרֶץ אֲשֶר בְּעֵבֶר הַיַרְדֵן וַיִתְיַשְבוּ בָהֶן. וּבְשְנַת הָאַרְבָּעִים לְחַיֵי יִצְחָק מֵת נָחוֹר כְּבֶן מֵאָה וְשִבְעִים וּשְתַּיִם שָנָה, וַיִקָבֵר בְּחָרָן; וַיִתְאַבֵּל עָלָיו אַבְרָהָם יָמִים רַבִּים. בָּעֵת הַהִיא קָרָא אַבְרָהָם אֶל אֱלִיעֶזֶר עַבְדוֹ הַנֶאֱמָן וַיַפְקִידֵהוּ עַל כָּל בֵּיתוֹ, וַיֹאמֶר לוֹ אַבְרָהָם: הִנֵה נָא זָקַנְתִּי לֹא יָדַעְתִּי יוֹם מוֹתִי, וְעַתָּה קוּם לֵךְ אֶל בְּנֵי מִשְפַּחְתִּי אֲשֶר בְּחָרָן וְלָקַחְתָּ מִשָם אִשָה לִבְנִי לְיִצְחָק. וְאַשְבִּיעֲךָ בְּשֵם ה‘, אֲשֶר לֹא תִקַח לוֹ אִשָה מִבְּנוֹת־כְּנָעַן; וַה’ אֱלֹהֵי הַשָמַיִם וֵאלֹהֵי הָאָרֶץ, אֲשֶר לְקָחַנִי מִבֵּית אָבִי וַיְבִיאֵנִי אֶל הַמָקוֹם הַזֶה וַיאֹמֶר לִי: לְזַרְעֲךָ אֶתֵּן אֶת הָאָרֶץ הַזאֹת לְרִשְתָּהּ עַד עוֹלָם, – הוּא יִשְלַח מַלְאָכוֹ לְפָנֶיךָ וְיַצְלִיחַ דַרְכְּךָ וְלָקַחְתָּ אִשָה לִבְנִי מִבְּנוֹת מִשְפַּחְתִּי. וַיַעַשׂ אֱלִיעֶזֶר כַּאֲשֶר צִוָהוּ אַבְרָהָם אֲדוֹנָיו, וַיָקָם וַיִקַח עֲשָׂרָה גְמַלִים וַעֲשָׂרָה אֲנָשִים וַיֵלְכוּ חָרָנָה עִירָה נָחוֹר לְהָבִיא מִשָם אִשָה לְיִצְחָק. וַיְהִי בְּצֵאתָם לַדָרֶךְ וַיִשְלַח אַבְרָהָם בֵּיתָה שֵם וָעֵבֶר וַיָשִיבוּ מִשָם אֶת יִצְחָק בְּנוּ בְּאֵרָה־שָבַע.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
