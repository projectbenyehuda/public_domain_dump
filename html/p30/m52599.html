ראשית שעבוד מצרים

וַיְהִי אַחֲרֵי כֵן וַיִתְקַבְּצוּ כָּל יוֹעֲצֵי פַרְעֹה וּזְקֵנָיו וַחֲכָמָיו וַיָבֹאוּ לְפָנָיו וַיֹאמְרוּ לוֹ: הִנֵה עַם בְּנֵי יִשְׂרָאֵל רַב וְעָצוּם מִמֶנוּ וְהָיָה תִקְרֵנוּ מִלְחָמָה חֲדָשָה וְנוֹסַף גַם הוּא עַל שוֹנְאֵינוּ. וַיַעַן פַּרְעֹה אֶת חֲכָמָיו וְשָׂרָיו לֵאמֹר: זאֹת עֲצָתִי הַיְעוּצָה עַל אוֹדוֹת בְּנֵי יִשְׂרָאֵל: הִנֵה עָרֵי פִתוֹם וְרַעַמְסֵס בִּלְתִּי בְצוּרוֹת הֵן, וְעַתָּה לְכוּ וְהַעֲבִירוּ קוֹל בְּכָל הָאָרֶץ, כִּי כָל הַמִצְרִים וְהָעִבְרִים (־בְּנֵי יִשְׂרָאֵל) יָבוֹאוּ בְּמִצְוַת הַמֶלֶך לִבְנוֹת אֶת מִבְצְרֵי הֶעָרִים הָהֵן. וְהָיָה כַאֲשֶר יָבוֹאוּ גַם הָעִבְרִים לַעֲבוֹד אֶת עֲבוֹדָתָם וּנְתַתֶּם לָהֶם יָמִים אֲחָדִים אֶת שְׂכָרָם וְאַחַר תַּעַזְבוּ אוֹתָם לְבַדָם וְעָבֹד יַעֲבְדוּ חִנָם אֵין כָּסֶף; וְאִם יְמָאֲנוּ – וְהֱעֶבַדְתֶּם אוֹתָם בְּחָזְקָה. וְהָיָה אִם עָשׂה תַעֲשׂוּ, אֶת הַדָבָר הַזֶה וַהֲטִיבוֹנוּ לָנוּ, כִּי נְחַזֵק אֶת עָרֵי אַרְצֵנוּ, וַהֲרֵעוֹנוּ לָעִבְרִים כִּי יִמְעֲטוּ וְיֶחְלְשוּ מֵרֹב הָעֲבוֹדָה הַקָשָה. – וַתִּיטַב עֲצַת פַּרְעֹה בְּעֵינֵי שָׂרָיו וַיַעֲשׂוּ כֵן. וַיְהִי כִּשְמֹע כָּל הַמִצְרִים וְהָעִבְרִים אֶת דַת הַמֶלֶךְ וַיָבֹאוּ כֻלָם לִבְנוֹת וּלְבַצֵר אֶת פִּיתוֹם וְאֶת רַעַמְסֵס, מִלְבַד שֵבֶט לֵוִי אֲשֶר לֹא בָא לָעֲבוֹדָה עִם בְּנֵי יִשְׂרָאֵל אֲחֵיהֶם, כִּי הֵבִינוּ אֶת עָרְמַת הַמִצְרִים. וַיִבְנוּ עַבְדֵי פַרְעֹה עִם בְּנֵי יִשְׂרָאֵל כְּחֹדֶש יָמִים, וַיְשַלְמוּ לָהֶם יוֹם־יוֹם אֶת שְׂכַר עֲבוֹדָתָם. וַיְהִי מִקֵץ הַחֹדֶש וַיַעַזְבוּ הַמִצְרִים אֶת בְּנֵי יִשְׂרָאֵל לְבַדָם לַעֲשׂוֹת בַּמְלָאכָה, וַיֶחְדְלוּ גַם לְשַלֵם לָהֶם אֶת שְׂכָרָם. וְהָיָה כַאֲשֶר יְמָאֵן אִיש יִשְׂרָאֵל לַעֲשׂוֹת בַּמְלָאכָה וְנִגְשוּ אֵלָיו נוֹגְשֵי פַרְעֹה וְהִכּוּהוּ הַכָּה וּפָצֹעַ; כָּכָה יַעֲשׂוּ הַמִצְרִים לָעִבְרִים כָּל הַיָמִים. וַיִבְנוּ בְנֵי יִשְׂרָאֵל לְבַדָם אֶת פִּיתוֹם וְאֶת רַעַמְסֵס וַיְחַזְקוּ אֶת כָּל מִבְצְרֵי אֶרֶץ מִצְרַיִם וְאֶת חוֹמוֹתֶיהָ. וַתֵּלֶךְ יַד מִצְרַיִם הָלוֹךְ וְקָשָה עַל בְּנֵי יִשְׂרָאֵל וַיַעֲבִידוּ אוֹתָם בְּפָרֶךְ. וַיִקְרְאוּ הָעִבְרִים אֶת מַלוּל מֶלֶךְ מִצְרַיִם בְּשֵם מָרוֹר כִּי בְיָמָיו מֵרְרוּ הַמִצְרִים אֶת חַיֵיהֶם בַּעֲבוֹדָה קָשָה, בְּחֹמֶר וּבִלְבֵנִים וּבְכָל עֲבוֹדוֹת הַשָׂדֶה. אוּלָם כְּהוֹסִיף הַמִצְרִים לְעֲנוֹת אֶת בְּנֵי יִשְׂרָאֵל, כֵּן יִרְבּוּ וְכֵן יִפְרֹצוּ. – בָּעֵת הַהִיא מֵת הֲדַד בֶּן בְּדַד מֶלֶךְ אֱדוֹם וַיִמְלֹךְ תַּחְתָּיו שַׂמְלָה מִמַשְׂרֵקָה מֵאֶרֶץ בְּנֵי קֶדֶם. וַיְהִי בְּמָלְכוֹ וַיֹאמֶר לְהִלָחֵם עִם צְפוֹ מֶלֶךְ בְּנֵי כִתִּים עַל הַכּוֹתוֹ אֶת אוֹגִיאַש מֶלֶךְ קַרְתַּגָה אַךְ בְּנֵי עֵשָׂו מְנָעוּהוּ מִמִלְחָמָה עִם צְפוֹ, כִּי אֲחִיהֶם הוּא.

וַיְהִי כִּשְמֹע פַרְעֹה וְשָׂרָיו אֶת הַדָבָר הַזֶה, וַיַכְבִּידוּ אֶת עֻלָם עַל בְּנֵי יִשְׂרָאֵל וַיָאִיצוּ בָהֶם לֵאמֹר: כַּלוּ אֶת מַעֲשֵׂיכֶם בְּמִבְצְרֵי חוֹמוֹת פִּיתוֹם וְרַעֲמְסֵס, פֶּן יָבוֹאוּ בְּנֵי עֵשָׂו לְהִלָחֵם בָּנוּ; וְהֵן בְּשֶלָכֶם נִלְחֲמוּ אִתָּנוּ וְעוֹיְנִים אוֹתָנוּ כַּיוֹם הַזֶה. וַיוֹסִיפוּ הַמִצְרִים לְמָרֵר אֶת חַיֵי בְנֵי יִשְׂרָאֵל וּלְעַנוֹתָם בְּסִבְלוֹתָם, לְמַעַן הַמְעִיטָם וַהַכְחִידָם מֵאֶרֶץ מִצְרָיִם. וַיְהִי כִּרְאוֹת פַּרְעֹה כִּי הוּפְרָה מַחֲשַבְתּוֹ, וּבְנֵי יִשְׂרָאֵל פָּרִים וְרָבִים – וַיִקְרָא לְכָל זְקֵנָיו וַחֲכָמָיו וַיֹאמֶר לָהֶם עוּצוּ נָא אַתֶּם עֵצָה: מַה לַעֲשׂוֹת לָעִבְרִים לְמַעַן הַמְעִיטָם וּלְכַלוֹתַם. וַיַעַן אֶחָד מִשָׂרֵי מִצְרַיִם, וּשְמוֹ אִיוֹב מֵאֶרֶץ עוּץ, אֶת פַּרְעֹה לֵאמֹר: אִם עַל הַמֶלֶךְ טוֹב יְצַוֶה נָא לַמְיַלְדוֹת לְהָמִית כָּל הַזְכָרִים לִבְנֵי יִשְׂרָאֵל וְסָר מֵעָלֵינוּ פַּחַד הַמִלְחָמָה; וּבְאִם יוֹסִיפוּ לִפְרוֹת וְלִרְבּוֹת וְהָיוּ לְעַם עָצוּם וָרָב וְיוֹצִיאוּ אוֹתָנוּ מִן הָאָרֶץ, כַּאֲשֶר עָשוּ בְּנֵי עֵשָׂו לִבְנֵי שֵׂעִיר. וַתִּיטַב עֲצַת אִיוֹב בְּעֵינֵי פַרְעֹה וַיְצַו אֶת שְתֵּי הַמְיַלְדוֹת הָעִבְרִיוֹת, אֲשֵר שֵם הָאַחַת שִפְרָה וְשֵם הַשְנִיָה פּוּעָה, לֵאמֹר: בְּיַלֶדְכֶן אֶת הָעִבְרִיוֹת וּרְאִיתֶן: אִם בֵּן הוּא וַהֲמִתֶּן אוֹתוֹ וְאִם בַּת הִיא וְחָיָה. וַתִּירֶאנָה הַמְיַלְדוֹת אֶת הָאֱלֹהִים וְלֹא עָשׂוּ כִּדְבַר מֶלֶךְ מִצְרַיִם. וַיִוָדַע הַדָבָר לְפַרְעֹה וַיִקְרָא לַמְיַלְדוֹת וַיֹאמֶר לָהֶן: מַדוּעַ זֶה אֵינְכֶן עוֹשׂוֹת אֶת מִצְוָתִי? וַתַּעֲנֶינָה שִפְרָה וּפוּעָה: כִּי לֹא כַנָשִים הַמִצְרִיוֹת הָעִבְרִיוֹת: בְּטֶרֶם תָּבוֹא אֲלֵיהֶן הַמְיַלְדֶת וְיָלָדוּ. וַיֶיטֶב ה' לַמְיַלְדוֹת וַיִרְבּוּ בְּנֵי יִשְׂרָאֵל וַיַעַצְמוּ מְאֹד, וַיָקוּצוּ הַמִצְרִים מִפְּנֵיהֶם.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
