מלחמת אברהם במלכי בבל

וַיַעַל אַבְרָם מִמִצְרַיִם וַיָשָב אַרְצָה כְּנַעַן וַיְהִי לוֹ צֹאן וּבָקָר וַעֲבָדִים וּשְפָחוֹת וְכֶסֶף וְזָהָב הַרְבֵּה מְאֹד. וְגַם לְלוֹט בֶּן הָרָן הַהוֹלֵךְ עִם אַבְרָם הָיָה מִקְנֶה כָבֵד וְאֹהָלִים, כִּי הֵיטִיב לוֹ ה' בַּעֲבוּר אַבְרָם. וַיְהִי כְּשֶבֶת אַבְרָם בְּאֶרֶץ כְּנַעַן וַיָרִיבוּ רוֹעֵי לוֹט עִם רוֹעֵי אַבְרָם, כִּי הָיָה רְכוּשָם רָב וְלֹא יָכְלָה הָאָרֶץ לָשֵׂאת אוֹתָם יַחַד מִפְּנֵי מִקְנֵיהֶם. וַיֵלְכוּ רוֹעֵי לוֹט וַיִרְעוּ אֶת עֶדְרֵיהֶם בִּשְׂדוֹת אֲחֵרִים. וַיָבוֹאוּ אַנְשֵי הָאָרֶץ אֶל אַבְרָם וַיָרִיבוּ אִתּוֹ עַל אוֹדוֹת רוֹעֵי מִקְנֵה לוֹט. וַיאֹמֶר אַבְרָם אֶל לוֹט: מָה הַמַעֲשֶׂה הַזֶה אֲשֶר עָשִׂיתָ לִי לְהַבְאִישֵנִי בְּעֵינֵי יוֹשְבֵי הָאָרֶץ, אֲשֶר תְּצַוֶה אֶת רוֹעֶיךָ לִרְעוֹת בִּשְׂדוֹת אֲחֵרִים? הֲלאֹ יָדַעְתָּ כִּי גֵר אָנֹכִי בָאָרֶץ הַזאֹת וְלָמָה תַעֲשֶׂה כַּדָבָר הָרָע הַזֶה? – וַיָרֶב אַבְרָם בְּלוֹט יוֹם־יוֹם, וְלֹא שָמַע לוֹט בְּקוֹלוֹ וַיוֹסֶף לִרְעוֹת אֶת מִקְנֵהוּ בִּשְׂדוֹת יוֹשְבֵי הָאָרֶץ. וַיֹאמֶר אַבְרָם אֶל לוֹט: עַד מָתַי תִּהְיֶה אַתָּה לִי לְמוֹקֵש לְעֵינֵי עַם הָאָרֶץ? – עַתָּה אַל נָא תְהִי מְרִיבָה בֵּינֵינוּ כִּי אֲנָשִים אַחִים אֲנָחְנוּ: הִפָּרֵד נָא מֵעָלַי וּבְחַר לְךָ מָקוֹם אַחֵר לְשִבְתֶּךָ. וְאַתָּה אַל תִּירָא לְהִתְרַחֵק מֵעָלַי, כִּי אָנֹכִי אָגֵן עָלֶיךָ מִפְּנֵי יוֹשְבֵי הָאָרֶץ, אֲשֶר יאֹמְרוּ לַעֲשׂוֹת לְךָ רָעָה. – וַיִשְמַע הַפַּעַם לוֹט בְקוֹל אַבְרָם וַיָקָם וַיֵלֶךְ אֶל כִּכַּר הַיַרְדֵן, וַיַרְא כִּי כֻלָהּ מַשְקֶה, וַיֵט שָם אָהֳלוֹ וַיָגָר בִּסְדוֹם. וְאַבְרָם יָשַב בְּאֵלוֹנֵי מַמְרֵא אֲשֶר אֵצֶל חֶבְרוֹן, וַיֵשֶב בַּמָקוֹם הַהוּא יָמִים רַבִּים.

בָּעֵת הַהִיא שָלַח כְּדָרְלָעֹמֶר מֶלֶךְ עֵילָם אֶל אַמְרָפֶל (נִמְרֹד) מֶלֶךְ שִנְעָר, אֲשֶר נִכְנַע תַּחַת יָדוֹ, וְאֶל תִדְעַל מֶלֶךְ גוֹיִים וְאֶל אַרְיוֹךְ מֶלֶךְ אֶלָסָר, אֲשֶר כָּרַת אִתָּם בְּרִית, לֵאמֹר: עֲלוּ אֵלַי וְעִזְרוּנִי וְנַכֶּה אֶת כָּל עָרֵי סְדוֹם וְאֶת יוֹשְבֵיהֶן כִּי פָשְעוּ בִי. וַיַעֲלוּ אַרְבַּעַת הַמְלָכִים הָאֵלֶה עִם מַחֲנֵיהֶם הָרַבִּים, יַחַד כִּשְמוֹנֶה מֵאוֹת אֶלֶף אִיש, וַיַעַרְכוּ מִלְחָמָה אֶת בֶּרַע מֶלֶךְ סְדוֹם וְאֶת בִּרְשַע מֶלֶךְ עֲמוֹרָה וְשִנְאָב מֶלֶךְ אַדְמָה וְשֶמְאֵבֶר מֶלֶךְ צְבוֹיִים וּבֶלַע מֶלֶךְ צֹעַר – וַיַכּוּם מַכָּה רַבָּה בְּעֵמֶק הַשִׂדִים. וַיִנָגְפוּ מַלְכֵי סְדוֹם לִפְנֵי מַלְכֵי עֵילָם וַיָנוּסוּ מִפְּנֵיהֶם וַיִפְּלוּ אֶל בֶּאֱרוֹת הַחֵמָר אֲשֶר שָם, וְהַנִשְאָרִים הֶרָה נָסוּ. וַיָבֹזוּ מַלְכֵי עֵילָם אֶת עָרֵי סְדוֹם וַעֲמוֹרָה וְגַם אֶת לוֹט בֶּן הָרָן וְאֶת רְכוּשוּ לָקְחוּ וַיֵלְכוּ לָהֶם. וַיָבֹא עוּנִי עֶבֶד אַבְרָם, אֲשֶר הָיָה בַמִלְחָמָה, וַיַגֵד לְאַבְרָם אֵת אֲשֶר עָשוּ מַלְכֵי בָבֶל לְעָרֵי הַכִּכָּר וְגַם לְלוֹט אָחִיו. וַיֶרֶק אַבְרָם אֶת חֲנִיכָיו שְלֹש מֵאוֹת וּשְמוֹנָה עָשָׂר אִיש, וַיִרְדֹף אַחֲרֵי מַלְכֵי בָבֶל וַיִתְנַפֵּל עֲלֵיהֶם פִּתְאֹם בַּלַיְלָה וַיַכֵּם מַכָּה רַבָּה וַיָשֶב אֶת כָּל רְכוּש סְדוֹם, וְגַם אֶת לוֹט וְאֶת יֶתֶר הַשְבוּיִים וְאֶת נְשֵיהֶם וְטַפָּם הֵשִיב. וַיָשָב אַבְרָם לִמְקוֹמוֹ אֲשֶר בְּאֵלוֹנֵי מַמְרֵא, וְלוֹט וְיֶתֶר אַנְשֵי סְדוֹם שָבוּ אֶל עָרֵיהֶם וַיֵשְבוּ בָהֶן. –


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
