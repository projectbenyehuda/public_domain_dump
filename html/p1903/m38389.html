משל צג: יֹדֵעַ צַיִד אִישׁ שָׂדֶה

מַאֲמִינֵי דִבְרֵי תָפֵל / לִבָּם בָּאֹפֶל

יֹדֵעַ צַיִד אִישׁ שָׂדֶה / אֶת הַחַיּוֹת אֹרֵב וְצֹדֶה / וְהֵם הִכִּירוּהוּ בְלֶכְתּוֹ / כַּאֲשֶׁר נֹשֵׂא אִתּוֹ תֶּלְיוֹ וְקַשְׁתּוֹ / רוּחוֹ בַלַּיְלָה לָצוּד יָעַר / בּוֹ תִרְמֹשׂ כָּל חַיְתוֹ יָעַר / וְטֶרֶם בֹּאוֹ הֵמָּה נָסָעוּ / וְעִקְּבוֹתֵיהֶם לֹא נוֹדָעוּ / עַד כִּי רָאוּהוּ פָנִים אֶל פָּנִים / אַיָּלִים צְבָיִם וּשְׁפַנִּים / וְאַקּוֹ וְדִישֹׁן וּתְאוֹ וָזֶמֶר / וְכָל שַׂעֲרוֹת בְּשָׂרָם תְּסַמֵּר / מִפְּנֵי חֲמַת הַמֵּצִיק אֲשֶׁר הִכִּירוּ / כָּל אֵלֶּה לִבְרֹחַ חָבָרוּ / וְנָמֵר תַּחַת אַחַד הַשִּׂיחִים / רָאָה כִּי כֻלָּם בֹּרְחִים / וְכִי הוּבָא בְּלִבָּם מֹרֶךְ / וַיִּתְיַצֵּב עַל אֵם הַדֶּרֶךְ / לָדַעַת מַה זֶּה וְעַל מַה זֶּה / אִם בַּחֲלוֹם אוֹ בַמַּחֲזֶה / נָסוּ כֻלָּם יָחַד / וְהֵם אָמְרוּ לוֹ פֶה אֶחָד / הִנֵּה הָאָדָם רָאִינוּ / בַּמִּדְבָּר אֹרֵב לָנוּ / עִמּוֹ כְּלֵי מִלְחַמְתּוֹ / וַתֵּשֶׁב בְּאֵיתָן קַשְׁתּוֹ / וְלִירוֹת תָּמִיד מָתוּחַ / אַשְׁפָּתוֹ כְּקֶבֶר פָּתוּחַ / שָׁם הַחִצִּים הֵמָּה גְּדוּדָיו / וַיָּפֹזוּ זְרֹעֵי יָדָיו / וַיַּעַן הַנָּמֵר: אַל תָּמוּשׁוּ מִן הַמַּחֲנֶה / עִמְדוּ עֲמֹדוּ וְאֵין מַפְנֶה / וּכְהַפְנוֹתָם שִׁכְמָם לָלֶכֶת / נָמֵר לֹא מָשׁ מִן הַמַּעֲרֶכֶת / וַיֹּאמַר: לֹא אֶת הָאָדָם יָרֵאתִי / כִּי אִם יִקָּרֶה לִקְרָאתִי / אֶטְרֹף אֹתוֹ רֶגַע וְאַצְמִית / אַחַת דָּתוֹ לְהָמִית / שְׁבוּ עִמִּי אַל תִּירְאוּ חֵמָה וְאַכְזְרִיּוּת / וָאוֹלֵךְ אֶתְכֶם קוֹמֲמִיּוּת / וְהָאָדָם אֲשֶׁר שָׁם אֹרֵב / לִירוֹת אֵלָיו הֹלֵךְ וְקָרֵב / דָּרַךְ קַשְׁתּוֹ חִצָּיו יָרָה / לְשַׁלַּח לוֹ לְמַטָּרָה / וְיָצָא כַבָּרָק חִצּוֹ / וּבַצֶּלַע הַשֵּׁנִית מְחָצוֹ / וּבְלֵב הַנָּמֵר דְּאָגָה וְרָגְזָה / וַיִּתְחַזֵּק לָנוּס בְּחָפְזָה / וַיַּשֵּׂג אֶת הַחַיּוֹת אֲשֶׁר אֵין מָנוֹחַ לְכַף רַגְלֵיהֶם / וּבְטֶרֶם יִקְרַב אֲלֵיהֶם / קָרְאוּ לוֹ: הֲיִתָּכֵן אֲשֶׁר יָנוּס הַמֶּלֶךְ / וּכְמוּג לֵב בְּחָפְזָה וּבִמְנוּסָה יֵלֵךְ / וְהָאֹרֵב אִם לֹא טָרַפְתָּ וְלֹא רָצַחְתָּ / מָה הַבִּטָּחוֹן הַזֶּה אֲשֶׁר בָּטַחְתָּ / וַיַּעַן לָהֶם כַּאֲשֶׁר נִכְנַע בַּעֲמָלוֹ / וַיְהִי כְּאוֹב מֵאֶרֶץ קוֹלוֹ / הִנֵּה נִבְהַלְתִּי בְרַעְיוֹנַי / כִּי הָאָדָם לֹא רָאוּ עֵינָי / אַךְ זְבוּבֵי מָוֶת נְשָׁכוּנִי בְמִצְוָתוֹ / הֵם הֵם הַנִּקְרָאִים בְּנֵי אַשְׁפָּתוֹ / וְהָהּ לַיּוֹם אֲשֶׁר בָּטַחְתִּי בִּגְבוּרָתִי / אוֹי לִי עַל שִׁבְרִי נַחְלָה מַכָּתִי / כִּי לֹא יִתְעַלֶּה כָמוֹנִי גִּבּוֹר בְּרֹב כֹּחַ וְאוֹנִים / כַּאֲשֶׁרר אֶחָד מֵעֲבָדָיו הַקְּטַנִּים / שָׁלַח עַד תְּכוּנָתִי וְהָשְׁבַּרְתִּי / וַתִּדְבְּקַנִי הָרָעָה וָמַתִּי.

וְהַמָּשָׁל בִּרְאוֹתְךָ נִבְהָלִים וּבֹרְחִים / הַנֶּאֱנָקִים וְהַנֶּאֱנָחִים / בִּמְרִירוּת וּבְשִׁבְרוֹן מָתְנַיִם / וְנָמֵס כָּל לֵב וְרָפוּ כָּל יָדַיִם / מִפְּנֵי חֶרֶב לְטוּשָׁה / מִקּוֹל עֲנוֹת חֲלוּשָׁה / בִּמְקוֹם רָע אַל תַּעֲמֹד / כַּאֲשֶׁר חַיֶּיךָ תַּחְמֹד / וּגְבוּרָתְךָ אַל תְּסִיתֶךָ וְתַחְבֻּלוֹתֶיךָ / פֶּן יִתְנַגְּפוּ רַגְלֶיךָ בִּמְזִמּוֹתֶיךָ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
