משל נג: שֶׁמֶשׁ וְרוּחַ וְאִישׁ

הֶחָכָם יַעֲשֶׂה בְּחָכְמָתוֹ / יוֹתֵר מֵהַגִּבּוֹר בִּגְבוּרָתוֹ


הַשֶּׁמֶשׁ וְהָרוּחַ הִגִּישׁוּ עֲצֻמּוֹתָם / וַיֹּאמֶר הַשֶּׁמֶשׁ לָרוּחַ: נִקֵּיתִי וָאֵיתָם / וּבְרֹב יָפְיִי שָׂמוּנִי בַיּוֹם לְמֶמְשֶׁלֶת / וְאִם תִּתְגָּאֶה לְנֶגְדִּי תֵּחָשֵׁב לְךָ אִוֶּלֶת / בַּאֲשֶׁר לִי הַגְּבוּרָה וְהַחֵן וְהַזֹֹּהַר / וּכְעֶצֶם הַשָּׁמַיִם לָטֹהַר / וַיַּעַן הָרוּחַ: אִם כִּפְלַיִם לְךָ כְמַעֲלָלֶיךָ / רוּחַ הַמּוֹשֵׁל תַּעֲלֶה עָלֶיךָ / כִּי בַלַּיְלָה לֹא תוֹסִיף לְהֵרָאֹה / שֶׁמֶשׁ יָדַע מְבוֹאוֹ / וּמֶמֶשַׁלְתִּי בַיּוֹם וּבַלַּיְלָה / עַל כֵּן יֵאוֹת לִי הַמֶּמְשָׁלָה / וּמַעֲשַׂי בָאָרֶץ נוֹדָעִים / מְפָרֵק הָרִים וּמְשַׁבֵּר סְלָעִים / וַיֹּאמֶר הַשֶּׁמֶשׁ: נְנַסֶּה אֶת גְּבוּרָתֵנוּ / הִנֵּה הָאִישׁ הַיּוֹצֵא לִקְרָאתֵנוּ / בְּשִׂמְלָתוֹ וּבְאַדַּרְתּוֹ / צֵא נָא עַתָּה וְהִלָּחֵם אִתּוֹ / וּבְתַחְבֻּלּוֹת תַּעֲשֶׂה לְךָ מִלְחָמָה / עַד תַּפְשִׁיט אֵדֶר מִמּוּל שַׂלְמָה / וְאִם יַעֲזֹב בִּגְדוֹ אֶצְלֶךָ / הוֹד הַמְּלוּכָה אֶתֵּן עָלֶיךָ / וְתַשְׂגִּיא פָעָלְךָ וְתַפְשִׁיטֶנּוּ / שִׂמְלָה לְךָ קָצִין תִּהְיֶה לָנוּ / וַיֹּאמֶר בְּלִבּוֹ הָרוּחַ: אֲפַתֶּנּוּ בְעָרְמָתִי / אוֹ אַפְשִׁיטֶנּוּ בִגְבוּרָתִי / וַיֵּצֵא הָרוּחַ מֵאֵת הַשֶּׁמֶשׁ לְעֵינָיו / וְרוּחָ חָזָק מְרַחֶפֶת לְפָנָיו / וְהָאִישׁ הוֹלֵךְ דֶּרֶךְ יְשָׁרָה / וּבְבֹא מִמְּזָרִים קָרָה / יִגְלֹם אַדַּרְתּוֹ וְשִׂמְלָתוֹ וְיִתְעַטֵּף בָּהֶם / וְרוּחַ לֹא יָבֹא בֵינֵיהֶם / וְהָיָה כַאֲשֶׁר הִתְחַזֵּק הָרוּחַ בְּכָל מַאֲמַצֵּי כֹחוֹ / וְהָאִישׁ לֻלְאוֹת אַדַּרְתּוֹ יִקְשֹׁר סְבִיב מִצְחוֹ / וּבִימִינוֹ יַחֲזִיק כְּנָפָיו וְיִצְנֹף סְבִיב רֹאשׁוֹ / וְלֹא גִלָּה הָרוּחַ פְּנֵי לְבוּשׁוֹ / וְהָיָה כִּי נִרְאָה כִּי נִלְאָה / הָרוּחַ אֲשֶׁר גָּאֹה גָאָה / וְלִגְזֹל הָאַדֶּרֶת לֹא נִשְׂגָּבָה / וַתְּהִי תוֹחַלְתּוֹ נִכְזָבָה / וַתֹּאמֶר הַשֶּׁמֶשׁ: עַתָּה תִרְאֶה אֶת אֲשֶׁר אֶעֱשֶׂה לוֹ / וַתִּקְרַע סוּת מַאֲפֵל וְאֶת צִלֲלוֹ / וַיִּפְרֹץ בַּצְּלָלִים פֶּרֶץ / הַשֶּׁמֶשׁ יָצָא עַל הָאָרֶץ / וַתֵּתַצֵּב בְּכֹחָהּ לְעֻמָּתוֹ / כְּצֵאת הַשֶּׁמֶשׁ בִּגְבוּרָתוֹ / לְהַשְׁקִיף עַל הָאִישׁ מִמָּרוֹם / בְּהַשְׁקֵט אֶרֶץ מִדָּרוֹם / וְעֵין הַשֶּׁמֶשׁ בְּנַחַת שְׁזָפַתּוּ / וּמִכֹּבֶד בְּגָדָיו בִּעֲתַתּוּ / וּמֵאַדַּרְתּוֹ תַפְשִׁיטֵהוּ / עָרוֹם יָנוּס בַּיּוֹם הַהוּא / וְלֹא יוּכַל עוֹד לָשֵׂאת כֵּלָיו / אִם זָרְחָה הַשֶּׁמֶשׁ עָלָיו / וַתַּפְשִׁיטֵהוּ בְהַשְׁקֵט כַּאֲשֶׁר עָלָיו הִגִּיעָה / וְלֹא כֵן הָרוּחַ אֲשֶׁר בִּגְבוּרָה וּמִסַּעַר סֹעָה.

וְהַמָּשָׁל כִּי הֶחָכָם יַעֲשֶׂה בְחָכְמָתוֹ / אֲשֶׁר לֹא יוּכַל הַגִּבּוֹר בִּגְבוּרָתוֹ / כִּי דִבְרֵי חֲכָמִים כְּחִצִּים בְּיַד לוֹחֲמִים / וּכְלַבּוֹת אִשִּׁים בְּלַהַב / תַּחְתָּיו שָׁחֲחוּ עֹזְרֵי רָהַב / כַּאֲשֶׁר בְּקִירוֹת בִּין הֵמָּה מְפֻתָּחִים / לְהַמְשִׁיךְ לִבּוֹת וְאִם הֵם כְּרֶשֶׁת וּפַחִים / בַּאֲשֶׁר הֵם נִשְׁמָעִים בְּנַחַת / אַךְ מִקּוֹל כְּסִיל גֵּאֶה אִישׁ לֹא יֵחָת.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
