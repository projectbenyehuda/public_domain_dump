משל צט: שֻׁעָל וַעֲגָלָה וְדָגִים

מָלֵא תֹךְ וּמִרְמָה / יַרְבֶּה מַשְׂטֵמָה

שֻׁעָל לְדַרְכּוֹ הָלַךְ וּבָא / לָשׁוּט בָּאָרֶץ לְאָרְכָּהּ וּלְרָחְבָּהּ / וַיִּפֶן פָּנָיו כֹּה וָכֹה / וְהִנֵּה עֲגָלָה בָאָה לְדַרְכּוֹ / מְלֵאָה דָגִים מִן הַיָּם / וּלְתַאֲוַת רוּחוֹ בַּעְיָם / תַּחְבֻּלוֹת חָשַׁב בְּמַעַשׂ / נַפְשׁוֹ אִוְּתָה וַיָּעַשׂ / כְּמֵת בַּדֶּרֶךְ עָשָׂה עַצְמוֹ / עוֹדֶנּוּ בְעֶצֶם תֻּמּוֹ / וַיִּפְרֹשׂ כַּפָּיו לְמַעְלָה / נֶחְשַׁב כְּמֵת בַּמְסִלָּה / בָּעֲרָבָה בְּכֹשֶׁל בֶּרֶךְ / נָפַל עַל אֵם הַדֶּרֶךְ / וַיִּשָּׂא הָעַגָּל אֶת עֵינָיו / וַיַּרְא הַשֻּׁעָל לְפָנָיו / כָּרַע שָׁכַב שָׁם וְרָבַץ / כִּי אֲחָזָהוּ הַשָּׁבָץ / וַיִּקַּח אֹתוֹ וַיִּשָּׂאֶנּוּ / לֹא קָם וְלֹא זָע מִמֶּנּוּ / וַיִּשָּׂא אֹתוֹ עַל הָאַמְתַּחַת / וַיְכַסֵּהוּ הָעַגָּל כִּי יֵחַת / פֶּן יַכִּירוּ אֹתוֹ רֹדְפָיו / וַיַּסְתֵּר אֹתוֹ תַּחַת כְּנָפָיו / פֶּן יִרְאֶה אֹתוֹ הֲשֹׁאֲלוֹ / וַיְכַסֵּהוּ בַּבְּגָדִים וַיֵּחַם לוֹ / עִם הַדָּגִים אֲשֶׁר בָּעֲגָלָה / מֵהֶם עָשָׂה הַשֻּׁעָל כָּלָה / וַיְמַלֵּא כְרֵסוֹ מֵהֶם / לֹא חָסָה נַפְשׁוֹ עֲלֵיהֶם / מֵהֶם חֵלֶק אָכַל וַיְבַתֵּר / בְּלִבּוֹ אָמַר: אָכֹל וְהוֹתֵר / לֹא אַשְׁקִיט עַד בִּלְעִי / גַּם אֶת הַמָּלֵא תַּסִּיעִי / וַיֹּאכַל מִן הַדָּגִים וַיִּשְׂבַּע / גַּם בְּפִיו לָקַח עַד שֶׁבַע / וַיֹּאכַל וַיִּשְׂבַּע וַיּוֹתֵר / גַּם אֶת הָאַמְתַּחַת חָתַר / וַיִּצְנַח מִן הָעֲגָלָה וַיָּנָס / וּבִמְצוּדָה אֲשֶׁר מָצָא נִכְנָס / וַיִּתְחַבֵּא וְהַטֶּרֶף בֵּין שִׁנָּיו / וַיַּעֲבֹר הָעַגָּל עַל פָּנָיו / וְלֹא רָאָה כִּי הַשֻּׁעָל נִמְלָט / וַיֵּלֶךְ הַשֻּׁעָל בַּלָּט / וַיַּעֲמֹד בְּמִשְׁעוֹל הַכְּרָמִים / כִּי יָרֵא מִן הַקָּמִים / וְעִקְּבוֹתָיו לֹא נָעוּ / וּמַצְפֻּנָיו לֹא נוֹדָעוּ / וְכַאֲשֶׁר הָיָה נִסְתָּר שָׁמָּה / בְּפַחַד וְיִרְאָה וְאֵימָה / וַיִּפְגַּע בַּמָּקוֹם הַזְּאֵב / עָיֵף וְיָגֵעַ וְדֹאֵב / כִּי לַטֶּרֶף עָיְפָה נַפְשׁוֹ / וַיִּשָּׂא הַזְּאֵב אֶת רֹאשׁוֹ / וַיָּבֹא לִקְרָאתוֹ וַיְבָרֲכֵהוּ / וְהַשֻּׁעָל כַּמִּשְׁפָּט עָנָהוּ / אַךְ חֹשֵׁב מַחֲשָׁבוֹת לְהַרְחִיקוֹ מֵעַל גְּבוּלוֹ / כִּי יָרֵא אֹתוֹ פֶּן יָבֹא וְיִקַּח אֲשֶׁר לוֹ / וַיִּשְׁאַל לוֹ הַזְּאֵב: הֲשָׁלוֹם לוֹ / וּמַה בְּפִיו נֹשֵׂא לְמוּלוֹ / וַיַּעַן בְּקוֹל רָם הַשֻּׁעָל / בְּעָקְבָה בְּמֶרֶד וּבְמָעַל / שָׁלוֹם לִי וְשָׁלוֹם בּוֹאֶךָ / שַׂשְׂתִּי מְאֹד עַל אִמְרָתֶךָ / כִּי הִנֵּה יָצָאתִי הַמַּיְמָה / אֲשֶׁר נִמְצָא שָׁמָּה / וְקָפְאוּ הַמַּיִם מִקֶּרַח / הָלַכְתִּי כְּגִבּוֹר לָרוּץ אֹרַח / וּבַקֶּרַח חוֹר עָשִׂיתִי / וּבַחוֹר זְנָבִי כְּשָׁעָה חֲדָה שַׂמְתִּי / וּסְבִיב הַזָּנָב בָּאוּ דָגִים / וְכִתְּרוּנִי כְסִיר דּוּגִים / וַאֲחַזְתִּים כְּמוֹ בִמְצוּדָה / וּמֵהֶם עָשִׂיתִי צֵידָה / וּמִשָּׁם אָכַלְתִּי וְשָׂבַעְתִּי / תֵּלֶךְ וְתַעֲשֶׂה כַּאֲשֶׁר עָשִׂיתִי / כִּי יָדַעְתִּי כִּי לָרִיק יָגַעְתָּ / אַאֲמֵן כִּי לֹא מָצָאתָ / וּלְהָשִׁיב נַפְשְׁךָ הָעֲיֵפָה / תֹּאכַל אֶחָד וְיִהְיֶה לִתְרוּפָה / וַיִּתְאַו הַזְּאֵב לְמַטְעַמָּיו / וַיִפְרֹשׂ עָלָיו שְׁלוֹמָיו / וַיְבָרְכֵהוּ הַזְּאֵב וַיֵּלַךְ / וַיֹּאמֶר לוֹ: יְהִי לָךְ אֲשֶׁר לָךְ / וְלֹא אַאֲמִינְךָ לְהַכְנִיס זְנָבִי בְחוֹרִים / פֶּן יָבֹאוּ בִי הַמַּיִם הַמְאָרְרִים וּמָרִים / וַיָּרָץ הַזְּאֵב עַד הַתְּעָלָה / וְכֻלָּהּ נִקְפֵּית מִקֶּרַח לָיְלָה / וַיַּבֵּט הַחוֹר אֲשֶׁר נִקְּרוּ / הָרֹעִים אֲשֶׁר שָׁמְרוּ / צֹאנָם כְּחֹם הַיּוֹם לְהַשְׁקוֹתָם / וַיֹּאמֶר בְּלִבּוֹ: נִקֵּיתִי וְאֵיתָם / חָטָאתִי כִּי חָשַׁדְתִּי בַשֻּׁעָל / וְלֹא שָׁלַח יָדוֹ בַמָּעַל / אָמְנָם בְּפִיו נְכוֹנָה / וַיְהִי יָדָיו אֱמוּנָה / כִּי אֲנִי רֹאֶה הַתְּעָלָה / גַּם הַחוֹר אֲשֶׁר לְמַעְלָה / וַיָּשֶׂם בַּחוֹר זְנָבוֹ / וַיֵּשֶׁב שָׁם עַל גַּבּוֹ / לְהַמְתִּין עַד בֹּא עִתּוֹ / וַיֵּאָסְפוּ הַדָּגִים בְּמַלְכֻּדְתּוֹ

וְהַשֶּׁמֶשׁ הָלַךְ וְעָבַר / וְהַקֶּרַח עָלָה וְגָבַר / וְהַזָּנָב הִקְפִּיא בַתְּעָלָה / וְהַזְּאֵב מָלֵא גִילָה / כִּי חָשַׁב בְּמַחֲשֶׁבֶת לִבּוֹ / לֶאֱחֹז דָּגִים בִּזְנָבוֹ / וַיֹּאמַר: מְאֹד כָבְדָה זְנָבִי / יָדַעְתִּי כִּי אֵיתָן מוֹשָׁבִי / טוֹב זֶה מִסְּחַר כָּסֶף / אִם כָּל דְּגֵי הַתְּעָלָה יֵאָסֵף / עַתָּה אֶהְיֶה זָרִיז וְנִשְׂכָּר / אַךְ יָרֵא אֲנִי מִבְּנֵי נֵכָר / פֶּן יִרְאוּנִי וַהֲרָגוּנִי / וְרֵיקָם יְשִׁיבוּנִי / וַיִּתְחַזֵּק הַזְּאֵב לָקוּם מִמְּקוֹמוֹ / וְלֹא יָכֹל, כִּי הַקֶּרַח נָפַל לְפַעֲמוֹ / וַיֵּצֶר לוֹ מְאֹד מִזְּנָבוֹ אֲשֶׁר מָצָא צָרָה / וַיִּזְעַק זְעָקָה גְדוֹלָה וּמָרָה / כִּי לֹא יָכֹל לָקוּם עַל רַגְלָיו / כִּי מָעֲדוּ קַרְסֻלָּיו / וַיִּשְׁמְעוּ הָרֹעִים וַיִּתְמְהוּ לְקוֹלוֹ / וַיָּבֹאוּ לִרְאוֹת מֶה הָיָה לוֹ / וַיַּתִּירוּ הַכְּלָבִים לַהֲסִיתָם בּוֹ / כָּל אֶחָד הֵסִית אֶת כַּלְבּוֹ / וַיִּפְגְּעוּ בוֹ וַיְנַשְּׁכוּהוּ / וּמִכָּל צַד כִּתְּרוּהוּ / בְּעַל כָּרְחוֹ נִשְׁאַר שָׁמָּה / וַיַּכּוּהוּ וַיַּכְּתוּהוּ עַד הַחָרְמָה / וַיִּתְחַזֵּק בַּחֲמַת רוּחוֹ / אֲבָל תַּם לָרִיק כֹּחוֹ / כִּי נִלְחַץ זְנָבוֹ בַקֶּרַח / וְהַשֶּׁמֶשׁ עָלָיו זָרַח / וְהָרֹעִים לְהַכּוֹת הִרְבּוּ / וַיְמָרֲרֻהוּ וָרֹבּוּ / וַיִּשְׂטְמֻהוּ בַּעֲלֵי חִצִּים / וַיִּזְרְקוּ לוֹ כָּל כְּלֵי מַפָּצִים / כִּי שִׂנְאָתָם מִקֶּדֶם / וַיֹּאמְרוּ: לְכוּ וְנַכְחִידֵם / וְלֹא נִשְׁמַר דֶּרֶךְ נְתִיבוֹ / וְהָרֹעִים פָּשְׁעוּ בוֹ / וַיֶּהֱמוּ כֻלָּם עָלָיו / וְלֹא יָרְאוּ מִגֶּשֶׁת אֵלָיו / וַיִּתְמַרְמַר בַּחֲמַת כֹּחוֹ / כִּי כָהֲתָה עֵינוֹ וְנָס לֵחוֹ / עַד כִּי נִשְׁבַּר הַקֶּרַח / וְלַחֲבוּרָתָם לֹא אָרַח / וַיִּבְרַח עַד הַיַּעַר / בְּרוּחַ סֹעָה מִסָּעַר / וּבְכָל זֹאת לֹא מָצָא נֹפֶשׁ / כִּי הַכְּלָבִים עַזֵּי נָפֶשׁ / תָּמִיד אֹתוֹ הִדְבִּיקוּ / וּבְקוֹל רָם עָלָיו צָעָקוּ / רִדְּפוּ מַהֵר כִּי תַשִּׂיגוּ / וַיְנַבְּחוּ הַכְּלָבִים וַיִּצְעָקוּ / וַיִּמָּלֵט מִן הָרֹדְפִים / כִּי מִן הַיַּעַר נִרְפִּים / מִלִּרְדֹּף אַחֲרָיו שָׁבוּ / לָלֶכֶת בַּיַּעַר לֹא אָבוּ / וַיִּתְחַבֵּא הַזְּאֵב תַּחַת הַסְּנֶה / מֻכֶּה פָּצוּעַ וּמְעֻנֶּה / וַיִּתְאוֹנֵן הוּא לְבַדּוֹ / וְלֹא מָצָא עֵזֶר כְּנֶגְדּוֹ / וַיֹּאמַר: אָכֵן סָר מַר הַמָּוֶת / כִּי הַשֻּׁעָל דַּרְכִּי עִוֵּת / וְזָמַם לְהָמִית אֹתִי / לְהַאֲמִין אֹתוֹ נִסְכַּלְתִּי / נִרְאֶה כִּי בִי הִתְעוֹלָל / לְבַקֵּשׁ בַּתְּעָלָה שָׁלָל.

וְהַמָּשָׁל בְּאִישׁ מָלֵא תֹךְ וּמִרְמָה / יַחְשֹׁב עַל רֵעֵהוּ מַשְׂטֵמָה / וּבְקִרְבּוֹ יָשִׂים אָרְבּוֹ / רֶשֶׁת טָמַן עַל נְתִיבוֹ / וְשָׂמַח כִּי הִגְדִּיל אוֹנוֹ / אוֹי לְרָשָׁע אוֹי לִשְׁכֵנוֹ / כִּי בְכָל עֵת מְבַקֵּשׁ תֹּאֲנָה / וְאֵין בְּפִיו נְכוֹנָה / אָזְנָיו הַכְבֵּד וְעֵינָיו הָשַׁע / מֵרְשָׁעִים יֵצֵא רֶשַׁע / עַל שְׁכֵנָיו מְדַבֵּר סָרָה / שִׂנְאָה מְקַלְקֶלֶת הַשּׁוּרָה.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

אֲנִי הַמְדַבֵּר נְקִי תֹךְ וְחַף בַּר / יְהִי נָא בְּסֵבֶר נְשִׁיקוֹת בְּתֹךְ בָּר

וְאָוֶן מְחַבֵּר כְּמוֹ מֹץ בְּתֹךְ בָּר / שְׂפַת חֵן יְגַבֵּר פְּנֵי זָךְ וְגָבָר.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
