משל נו: כְּפִיר וְצַיָּד

רַבּוּ פוֹעֲלֵי אָוֶן וְדוֹבְרֵי הַוּוֹת / כִּי אֵין הַלְּבָבוֹת שָׁווֹת


כְּפִיר שֹׁאֵג מֵאַחַר הַמִּדְבָּר / וְיֹדֵעַ צַיִד אִישׁ שָׂדֶה עָלָיו יִתְגַּבָּר / וְיִתְיַצֵּב כְּגִבּוֹר לַעֲרֹךְ מִלְחָמָה בְּחִצִּים וּבְקֶשֶׁת יָבֹא שָׁמָּה / וְלֵב הַכְּפִיר יִתַּר מֵחֶרְדָּתוֹ / פֶּן יָבִיא בְִכִלְיוֹתָיו בְּנֵי אַשְׁפָּתוֹ / אָמְנָם הַצַּיָּד כִּפְלַיִם מִמֶּנּוּ פָחַד / עַד אֲשֶׁר הִמְתִּיקוּ סוֹד יַחַד / וְהָאֲרִי בְּעַיִן חֲפֵצָה אֵלָיו קָרָא / הִשָּׁמֵר וְהַשְׁקֵט אַל תִּירָא / הִנֵּה אֵימָתִי אַל תְּבַעֲתֶךָ / כִּי נִכְסְפָה נַפְשִׁי אֶל חֶבְרָתֶךָ / וּבְתֹם לֵבָב עָנָה לוֹ הַצַּיָּד / גַּם אֲנִי לֹא אֶשְׁלַח בְּךָ יָד / וַיֵּלְכוּ יַחַד שְׁנֵיהֶם / וּמֶתֶק הַחֶבְרָה בֵינֵיהֶם / וַיֹּאמֶר הָאִישׁ: כָּל הַחַי מִכָּל בָּשָׂר רָאִיתִי / וְלֹא מָצָאתִי הוֹד מַלְכוּת רַק לְנִבְרָא בְצַלְמִי וְכִדְמוּתִי / כִּי אָדָם הֹלֵךְ בְּצֶלֶם נָאֶה נֶחְמָד וְעָרֵב / עַל כֵּן גְּדֻלָּתוֹ עַל כָּל הָאָרֶץ יֶרֶב / וּמֶלֶךְ עַל כָּל נִבְרָא יִהְיֶה / גַּם עַל שַׁחַל וּכְפִיר וְאַרְיֵה / וַיַּעַן הַכְּפִיר: אַךְ לִי יֵאוֹת לִמְלֹךְ עַל כָּל יֹשְׁבֵי תֵבֵל וְאִיִּים / בַּאֲשֶׁר נָתַתִּי חִתִּיתִי בְאֶרֶץ הַחַיִּים / וְכִי אֵין עַל עָפָר מָשְׁלִי / וּמִי יִתְיַצֵּב לִפְנֵי שַׁאֲגַת קוֹלִי / כִּי אִם בְּעֵין אֵיבָה / אֲשׁוּרֶנּוּ תוֹחַלְתּוֹ נִכְזָבָה / בְּצַוָּארִי יָלִין עֹז וּלְפָנַי תָּרוּץ דְּאָבָה / וַיֵּלְכוּ כָּל הָעֲרָבָה / בִּדְבַר קִנְאָה דַּבֵּר וְהָלֹךְ / זֶה עַל זֶה מִתְנַשֵּׂא לֵאמֹר: אֲנִי אֶמְלֹךְ / עַד אֲשֶׁר רָאוּ גַל וּמַצֶּבֶת אֶבֶן מַשְׂכִּית עַל יָדָם / וְחָרוּת עַל הָאֶבֶן הָאַרְיֵה וְהָאָדָם / וְהָאִישׁ כְּמֶלֶךְ יוֹשֵׁב עַל כִּסֵּא מַלְכוּתוֹ / וְהָאֲרִי כוֹרֵעַ עַל כַּפּוֹת רַגְלָיו בְּצוּרָתוֹ / וַיֹּאמֶר הָאָדָם: אַיֵּה פִיךָ אֵפוֹ / הֲיִתְגַּדֵּל הַמַּשּׂוֹר עַל מְנִיפוֹ / עַתָּה תִרְאֶה הֲיִקְרְךָ דְבָרִי / כִּי הָאָדָם תִּמְצָא מֶלֶךְ עַל הָאֲרִי / מִיּוֹם אֲשֶׁר נִבְרְאוּ הַבְּהֵמוֹת וְהַחַיּוֹת / כִּימֵי עוֹלָם וּכְשָׁנִים קַדְמֹנִיּוֹת / עַד הַגַּל הַזֶּה וְעֵדָה הַמַּצֵּבָה / הִנֵּה קָמָה אֲלֻמָּתִי וְגַם נִצָּבָה / וַיַּעַן הָאֲרִי: הִנֵּה שֶׁקֶר הָעֵד / וְאִם נִסִּיתִי לְצַיֵּר כְּיוֹם מוֹעֵד / אֶעֱשֶׂה לוֹ כַּאֲשֶׁר זָמַם לַעֲשׂוֹת לָאֲרִי / וְלִי כַאֲשֶׁר אֶחְפֹּץ בִּיקָרִי / וְנַהֲפוֹךְ הָאֲרִי לְמֶלֶךְ וְהָאִישׁ לְרַגְלוֹ / כֹּרֵעַ וּמִשְׁתַּחֲוֶה לוֹ.

וְהַמָּשָׁל כִּי יֵשׁ אִישׁ לְמַלֵּא נַפְשׁוֹ יַחְמֹד / וַיַּרְא אֶת כָּל אֲשֶׁר עָשָׂה וְהִנֵּה טוֹב מְאֹד / כִּי הִשְׁלִיטוֹ לְחֶפְצוֹ וְיִצְרוֹ / וּבָא רֵעֵהוּ וַחֲקָרוֹ / לִסְתֹּר אֶת אֲשֶׁר בָּנָה / וְלִמְכֹּר אֶת אֲשֶׁר קָנָה / וְכִלְיוֹתָיו יִשְׁתּוֹנָן וְלִֵבּוֹ יִתְחַמֵּץ / לְהַשְׁפִּיל אֲשֶׁר הוֹסִיף זֶה אֹמֶץ / וּלְבַעֲבוּר כִּי לֹא כָּל הַלְּבָבוֹת שָׁווֹת / רַבּוּ פוֹעֲלֵי אָוֶן וְדוֹבְרֵי הַוּוֹת.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
