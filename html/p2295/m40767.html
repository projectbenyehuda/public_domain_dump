מִתּוֹךְ הַמַּחֲזֶה "יְסוֹד עוֹלָם": הַשִּׂיחָה בֵין תֶּרַח וְאַבְרָם

Mosè Zacuto

]נולד באַמְסְטֶרְדָּם בשנת 1625, התחנך בעיר מולדתו ובפוֹזְנָא, חי בוֵרוֹנָה, פָּדוֹבָה, וֵנֶצְיָה (1645) ומַנְטוֹבָה (1673). מת במַנְטוֹבָה בשנת 1697]


מִתּוֹךְ הַמַּחֲזֶה “יְסוֹד עוֹלָם”: הַשִּׂיחָה בֵין תֶּרַח וְאַבְרָם

– – ͝ – – – ͝ – – – –

תרח: מִי הוּא וְאֵיזֶה הוּא אֲשֶׁר בַּבַּעַל

שָׁלַח כְּצַר יָדוֹ בְּעַזּוּת־מֶצַח

מִי הָעֲוִיל וְצַר וּמוֹעֵל מַעַל

הִגְדִּיל בְּרִשְׁעָתוֹ כְּאֵבִי נֶצַח

מִי יַאֲמִין כִּי אֶת־אֱלוֹהַּ מַעַל

שִׁבַּר וְדָשׁ הֵדַק כְּהִדּוּשׁ קֶצַח

אוֹי כִּי בְנִי תִהְיֶה כְלִמָּתֵנוּ

לַעַג וְקֶלָס אֶל־סְבִיבוֹתֵינוּ.

– – ͝ – – – ͝ – – – –

אברם: הַסְכַּת וְהִשָּׁקֵט אֲדֹנִי נָא שְׁמַע

מִלֵּי, וּמַה־תִּצְעַק בְּלִי תּוֹעָלֶת

דַּע כִּי תְמוֹל בָּאָה לְחַדְרֵי הָאֱלִיל

אִשָּׁה כְּאַלְמָנָה וּמִתְאַבָּלֶת

לָהֶם לְמִנְחָתָהּ הֵבִיאָה צוּף־דְּבַשׁ

חַסְדָּם וְעֶזְרָם הָיְתָה שׁוֹאָלֶת

נֹכַח פְּנֵיהֶם הַזְּקֵנָה הָיְתָה

מִתְחַנְּנָה בּוֹכָה וּמִתְפַּלָּלֶת.

אַחַר תְּפִלָּתָהּ בְּרֹב חֶדְוָה וְגִיל

שָׁבָה לְאֹהָלָהּ בְּטוּב תּוֹחָלֶת

כִּי חָשְׁבָה תִּהְיֶה תְחִנָּתָהּ כְּבָר

עוֹלָה אֱלֵי רָצוֹן וּמִתְקַבָּלֶת.

וְאֶשְׁמְעָה אָז קוֹל שְׁאוֹן צַעַר כְּקוֹל

הָרֶה עֲלֵי מַשְׁבֵּר וְכִמְיַלָּלֵת

כִּי הָאֱלִיל גָּדוֹל בְּרוּחַ גַּאֲוָה

רוּחוֹ בְּרֵעָיו מוֹאֲסָה גּוֹעָלֶת

“לִי יָאֲתָה” אָמַר "נְדָבָה זֹאת לְבַד

כִּי לִי עֲלֵיכֶם הוֹד צְבִי מֶמְשָׁלֶת"

לֹא חָפְצוּ הֵמָּה שְׁמֹעַ לוֹ וְגַם

נַפְשָׁם בְּנַפְשׁוֹ הָֽיְתָה בּוֹחָלֶת

אָז הוּא בְאַפּוֹ קָם חֲמָתוֹ בָעֲרָה

לִמְאֹד כְּלַהַב אֵשׁ בְּתוֹךְ גֶּחָלֶת

חָגַר כְּלֵי־זַיִן כְּצֵאתוֹ לַקָּרָב

פַּטִּישׁ בְּיָד לָקַח וְגַם מַאְכָלֶת

אֶת־כָּל־אֱלִילֶיךָ כְּאַכְזָר שִׁבְּרָם

הִנָּם לְפָנֶיךָ כְּצִיץ נוֹבָלֶת

אָז נָס לְהֵחָבֵא וְנַפְשׁוֹ הָיְתָה

דָוָה וְשׁוֹמֵמָה וּמִתְבַּהָלֶת

וּלְהַאֲמִין מִלֵּי אֲדֹנִי נָא רְאֵה

אוֹתוֹ וְהִנּוֹ אַחֲרֵי הַדָּלֶת.


תרח: אוֹי לִי וְהָהּ עָלֵי אֱוִיל בַּעַר שְׁתֹק

לָמָּה תְּסַפֵּר מַעֲשֵׂה אִוָּלֶת

אֵיךְ יִפְעֲלוּ אִם הֵם כְּמוֹ־אֶבֶן אֲשֶׁר

עַל־הָאֲדָמָה נוֹֽפְלָה מוּטָלֶת


– – ͝ – – – ͝ – – – –

אברם: אִם־כֵּן יְדִיד נַפְשִׁי פְּקַח עֵינֶיךָ

הָבֵן גְּזֵרַת הָאֱמֶת גָּזַרְתָּ

מוֹצָא שְׂפָתֶיךָ אֲשֶׁר דִּבָּרְתָּ

תִּשְׁמַע וְהַשְׂכֵּל אֶל־שְׁתֵּי אָזְנֶיךָ.

לָמָּה יְפַתֶּךָ וּמַה יוֹנֶךָ

צֶלֶם אֲשֶׁר אַתָּה בְיָדְךָ צָרְתָּ

מֵעֵץ וּמֵאֶבֶן אֲשֶׁר בָּחָרְתָּ

וַתַּעֲזֹב צוּר עֹשְׂךָ קָנֶךָ.

לָמָּה לְשִׁבְרוֹן עֵץ וְשֶׁבֶר נֵבֶל

כָּל־מַחְשְׁבוֹתֶיךָ אֱנוֹשׁ יוֹדֵעַ

כֵּן תָּמְהוּ כֵּן נִבְהֲלוּ נֶחְפָּזוּ.

לֹא עוֹד אֲדֹנִי אַחֲרֵי הַהֶבֶל

תֵּלֵךְ אֲבָל אָבִי הֱיֵה שׁוֹמֵעַ:

חָכְמָה וּמוּסָר הָאֱוִילִים בָּזוּ.

הָאֵל אֲשֶׁר אֵלָיו אֲנִי כוֹרֵעַ

קַדְמוֹן מְצֻחְצָח צַח וְשֵׂכֶל זֹהַר

פָּשׁוּט בְּתַכְלִית הוּא וְעֶצֶם טֹהֵר

אֶחָד בְּמִדּוֹתָיו וְאֵין לוֹ רֵעַ.

קַיָּם וְלֹא יִשְׁנֶה וְיִתְנוֹעֵעַ

כִּי אֵין תְּמוּנָה לוֹ וְתַבְנִית־תֹּאֵר

עַל־כֵּן שְׁמוֹתָיו לוֹ לְשֵׁם הַתֹּאַר

כִּי הוּא לְבַדּוֹ אֶת־שְׁמוֹ יוֹדֵעַ.

עֻזּוֹ בְחֶבְיוֹן נֶעֱלָם מֵעָיִן

רַק קוֹל מְצִיאוּתוֹ לְבַד יִשְׁמָעוּ

אִשִּׁים וְחַשְׁמַלִּים וְאִישֵׁי־עָרֶץ.

הוֹצִיא פְעֻלּוֹתָיו לְיֵשׁ מֵאָיִן

אָז בַּאֲמָרָיו נוֹֽסְדוּ נִקְבָּעוּ

כָּל־יֽשְׁבֵי תֵבֵל וְשֹׁכְנֵי אָרֶץ.

אָמַר “יְהִי אוֹרָה” וְגַם הִצִּיעַ

אֶרֶץ בְּיוֹם רִאשׁוֹן וְהַשָּׁמָיִם

גָּזַר בְּיוֹם שֵׁנִי "בְּתוֹךְ הַמָּיִם

חוֹצֵץ לְהַבְדִּילָם יְהִי רָקִיעַ".

גַּם בַּשְּׁלִישִׁי הַתְּהוֹם הִצִּיעַ

אֶרֶץ וּמַיִם נֶחְלְקוּ לִשְׁנָיִם

אַךְ לַעֲשׂוֹת בּוֹ מַעֲשֵׂה יוֹמָיִם

צִוָּה “יְהִי עֵשֶׂב וְעֵץ מַזְרִיעַ”.

שֶׁמֶשׁ וְכוֹכָבִים וְהַיָּרֵחַ

צִפּוֹר בְּהֵמָה דָּג וְחַיְתוֹ־יָעַר

בִּזְמָן שְׁלשָׁה צָר לְתוֹעַלְתֵּנוּ.

וּלְסוֹף יְצִירוֹתָיו בְּלֵב שָׂמֵחַ

אָז “נַעֲשֶׂה” אָמַר לְשַׂרְפֵי־שָׁעַר

“אָדָם בְּצַלְמֵנוּ וְכִדְמוּתֵנוּ”.

הֵן מִמְּעוֹנוֹ יֶחֱזֶה יַשְׁגִיחַ

עַל־הַֽיְקוּם כֻּלּוֹ בְּעֵינֵי־יֹשֶׁר

כִּי מִלְּפָנָיו כָּל־יְקָר וָעשֶׁר

וּבְרֹב חֲסָדָיו אֶת־אֱנוֹשׁ מַצְלִיחַ.

מֵשִׁיב גְּמוּל שָׁלֵם אֲשֶׁר מַבְטִיחַ

לַהוֹֽלְכִים עִמּוֹ בְּדַרְכֵי־כֹשֶׁר

וּבְסוֹף יְמֵי חֶלְדָּם בְּעֵדֶן־אֹשֶׁר

נִשְׁמַת יְחִידָתָם הֲכִי יַנִּיחַ.

זֶה הוּא יְדִיד נַפְשִׁי אֲשֶׁר אָהָבְתִּי

הִנּוֹ מְפַלְּטִי לִי וְאוֹר בַּעְדֵנִי

בּוֹ אֶחֱסֶה בּוֹ אֶמְצְאָה מָנוֹחַ.

אִם בַּחֲרוֹנְךָ כַּאֲשֶׁר חָשָׁבְתִּי

תֹּאמַר בְּתוֹךְ לִבְּךָ לְהַשְׁמִידֵנִי

קֹוֵי יְיָ יַחֲלִיפוּ כֹחַ.

אָמְנָם עֲצַבֶּיךָ אֱלִילֵי־רוּחַ

אֹזֶן אֲלֵיהֶם יֵשׁ וְלֹא יַאְזִינוּ

עַיִן אֲלֵיהֶם יֵשׁ וְלֹא יָבִינוּ

כִּי גוּף לְבַד הֵמָּה וּבִלְתִּי רוּחַ.

צוּרָם וְצַלְמָם בַּדֳכִי יָשׁוּחַ

יוֹם יֹאכְלֵם תּוֹלָע וְעֵת יַזְקִינוּ

יִהְיוּ כְמוֹהֶם כָּל־אֲשֶׁר יַאְמִינוּ

בָּהֶם וְלִבָּם בָּם יְהִי בָטוּחַ.

עַתָּה הֲשִׁיבֵנִי: הֲאִם עָצַרְתָּ

כֹּחַ לְהִתְיַצֵּב בְּרֹב אוֹנֶךָ

הָבֵן אֱמוּנָתִי וְצוּף מִלֶּיהָ.

אִם בָּאֱלִילִים נַפְשְׁךָ קָשַׁרְתָּ

הַתֵּר כְּבֶן־חַיִל בְּרֹב אוֹנֶךָ

אֶת־כָּל־אֱסָרֶיהָ אֲשֶׁר עָלֶיהָ.


͝ – – – ͝ – – – ͝ – –

תרח: אֲהָהּ עָלֵי וּמִי יַאְמִין לְצַעְרִי

אֲשֶׁר נֶגְדִּי בְנִי נֶהְפֵּךְ לְצָרִי

אֲהָהּ כִּי יָֽצְאוּ מִנִּי מְהָרְסַי

לְשׁוֹחֵחַ כְּבוֹדִי וַהֲדָרִי

עֲוִיל שׁוֹטֶה וּבֵן סוֹרֵר וּמוֹרֶה

אֲשֶׁר קַמְתָּ לְהַשְׁפִּיל הוֹד יְקָרִי

אֱמֶת כִּי טָחֲנָה אִמְּךָ לְאַחֵר

וְאֵינְךָ לֹא קְרוֹבִי וְלֹא שְׁאֵרִי

וְאֵיךְ אֶפְשָׁר הֱיוֹתְךָ בֶּן־חֲלָצַי

אֲשֶׁר יָצָא בְשָׂרְךָ מִבְּשָׂרִי

וְאֵי בַּעַר כְּבוֹד הָאָב וּפַחְדּוֹ

וְאֵי מוֹרָאֲךָ שׂוֹנְאִי וְעוֹכְרִי

לְאָן אוֹלִיךְ כְּלִמָּתִי וּבָשְׁתִּי

וּמָה אֹמַר אֱלֵי מַלְכִּי וְשָׂרִי

וּמָה אָשִׁיב בְּעֵת יִרְצוּ שְׁכֵנַי

לְהִתְפַּלֵּל תְּפִלָּתָם בְּחַדְרִי

הֲלֹא אִם לִבְּךָ שַׂמְתָּ לְהַאְמִין

אֱמוּנָה זֹאת כְּמוֹ גֶבֶר הֲרָרִי

אֱמוּנָתְךָ לְךָ תִּהְיֶה לְבַדְּךָ

וְלָמָּה נֶהֱפַכְתָּ לִי לְעָרִי

וּמַדּוּעַ הֲדִקּוֹתָ אֱלֹהַי

אֲשֶׁר הָיוּ לְמִשְׂגָּב לִי וְעֶזְרִי

הֲלֹא חָרֵד לְבָבְךָ בָּם בְּקָרְבָךְ

לְכַבּוֹת אֶת־פְּאֵר עֵינַי וְאוֹרִי

אֲבָל כַּתֵּר זְעֵיר לִי וְאֲחַוֶּה

לְנִמְרוֹד אֶת־תְּלָאוֹתַי וְשִׁבְרִי

וְהוּא יִשְׁפֹּט עֲלִילָתְךָ בְּצֶדֶק

וְיִנְקֹם מִמְּךָ חֶבְלִי וְצִירִי

יְשִׂימוּךָ בְּנִבְכֵי בֵית אֲסוּרִים

וְיָמֶיךָ יְבַלֶּה מַאֲסָרִי

אֲזַי תַּשְׁפִּיל זְדוֹן לִבְּךָ וְתִהְיֶה

נְכֵה־רוּחַ וְחָרֵד עַל־דְּבָרִי.


(ויצא תרח בכדי חרון וקצף)

אברם: סַלְעִי וּמָגִנִּי בְּךָ חָסִיתִי

עֹשֵׂה גְדֹלוֹת אַתְּ עֲדֵי אֵין חֵקֶר

שׁוּר כִּי לְשִׁמְךָ צוּר אֲנִי עָשִׂיתִי

לִשְׁבֹּר וְלָדוּשׁ אֶת־אֱלִילֵי שֶׁקֶר

גַּם פִּלְאֲךָ שַׂחְתִּי וְלֹא כִסִּיתִי

לִהְיוֹת לְךָ יִשְׁעִי לְהוֹד וּלְיֶקֶר

לָכֵן לְעָזְרֵנִי בְּעֹז הוֹדֶךָ

קוּמָה יְיָ אֵל נְשָׂא יָדֶךָ.


(אברם בתפלתו)

נוֹרָא תְהִלּוֹת נָא וְעֹשֵׂה־פֶלֵא

מַה־תֶּחֱזֶה עָמָל וְתַבִּיט כָּעַס

הוֹדַע שְׁמָךְ לַבּוֹעֲרִים הָאֵלֶּה

אָנָּא אֱלֹהֵי עַל־כְּבוֹדְךָ תָּעַשׂ

שׁוּר צוּר אֲשֶׁר אָבִי בְּבֵית הַכֶּלֶא

אָמַר לְאָסְרֵנִי בְּאַף וּבְכָעַס

נָא בִי מְפַלְּטִי הַחֲזֵק אַל־תֶּרֶף

כִּי הַכְּפִירִים שֹׁאֲגִים לַטָּרֶף.


תָּפְתָּה עָרוּךְ (קֶטַע)

… ברגע הכנס הנפש אל תוך הקבר יחזרו למת חושיו אך בטירוף ועירוב ולכן יקדים הוא לדבר אל לבו כעין הדברים האלה:

[הַמֵּת]: אִם הַֽיְסוֹדוֹת נִרְקְבוּ נִפְרָצוּ

שָׁוְא עָמְלוּ בוֹנִים וְגוֹדְרֵי פָרֶץ

אִם יֶחֱרַץ חָרוּץ כְּרוּז הֶחָרֶץ

לֹא יִצְלְחוּ אִישִׁים וְאִם יֶחְרָצוּ

כִּי בַחֲרִיצוּת פָּעֳלָם יֶחְרָצוּ.


הָרוֹֽפְאִים כַּהוֹלֲלִים יָהֹלּוּ

וִימַשְּׁשׁוּ כַעִֽוְרִים בַּנָּשֶׁף

כִּי יֹאֽמְרוּ לַדְרִיךְ לְחוּמֵי־רָשֶׁף

לִמְחוֹז תְּרוּפַת מַחֲלָה יִסְבֹּלוּ

וַיַּחְתְּרוּ לָרִיק וְלֹא יָכֹלוּ.


יוֹם נִמְסְרוּ דִינֵי רְשָׁעִים לָנוּ

לוּ נוֹֽדְעוּ אֵלֶּה וְלוּ נִזְכָּרוּ

לֹא נִכְתְּבוּ מִיתוֹת אֲשֶׁר נִמְסָרוּ

כִּי עֵת לְמִיתָה פּוֹֽשְׁעִים יוּכָנוּ

יֹאמְרוּ “בְּיַד הָרוֹֽפְאִים יֻתָּנוּ”.


אִם נָֽתְנָה תּוֹרָה רְשׁוּת לָאֵלֶּה

הֵן הָֽיְתָה רַק לַחֲבֹשׁ כָּל־פָּצַע

אַךְ בַּפְּנִימִי אֵין בְּמַעֲשָׂם בָּצַע

רַק הוּא פְּעֻלַּת אֵל בְּדֶרֶךְ פֶּלֶא

עַל־כֵּן יְחִידָתִי בְּסִכְלָם תֵּלֶא.


עַתָּה בְנִכְלֵיהֶם חֲדָשִׁים בָּאוּ

לִנְתֹץ וְלִנְטֹשׁ לַהֲרֹג לִרְצֹחַ

הֵם רוֹפְאִים כִּשְׁמָם מְרַפִּים כֹּחַ

כִּי רַק לְהָרִיק אוֹ לְדָם יוּבָאוּ

גַּם אַחֲרֵי־כֵן הָרְכוּשׁ יִשָּׂאוּ.


זֶה־לִּי זְמָן אָרֹךְ אֲשֶׁר יָחִילוּ

גּוּפִי בְהַקָּזוֹת וּמִינֵי חָבֶל

זִיקוֹת וְהָרָקוֹת וְהַכֹּל הָבֶל.

אֶחְשֹׁב אֲשֶׁר יִכְלוּ וְאָז יָחִילוּ

וַיִּיגְעוּ לַשָּׁוְא וְלֹא יוֹעִילוּ.


אֶמֶשׁ הֲלָמוּנִי וּמַה־נֶּהְלָמְתִּי

בִּשְׁתִי וְשִׁקּוּי מַר אֲשֶׁר שָׁתִיתִי

טִפּוֹת כְּמוֹ מַר מִדְּלִי מָצִיתִי

עַד כִּי בְתַרְדֵּמָה כְּמֵת נִרְדָּמְתִּי

לֹא נִים וְלֹא נִים תִּיר וְלֹא תִיר נָמְתִּי.


אַךְ אַף־אֲשֶׁר עַד־פֹּה אֲנִי יָשָׁנְתִּי

לֹא סָר כָּאַב הָרֹאשׁ וְדַאְבוֹן נָפֶשׁ

אֶהְיֶה כְאִישׁ מֻשְׁלָךְ בְּטִיט וָרָפֶשׁ.

אִלּוּ כְפֶגֶר מֵת בְּבוֹר שָׁכָנְתִּי

שָׁם אֶשְׁקְטָה יוֹתֵר וּפֹה שַׁאְנָנְתִּי.


לֹא יֵשׁ זְמָן הַרְבֵּה אֲשֶׁר רָבָדוּ

מַצַּע כְּסָתוֹתַי וְכַר הַתָּבֶן

עַתָּה בְקָשְׁיָם יִדְּמוּ כָאָבֶן.

אִלּוּ עֲצָמַי אֶל־שְׁאוֹל יוּרָדוּ

שָׁם יִשְׁכְּבוּ יוֹתֵר וּפֹה נִפְרָדוּ.


אַף נֶעֱדַּר חֻמִּי וְהִתְפּוֹרָרְתִּי

וּכְסוּת בְּגָדִים לִי בְּלִי תּוֹעָלֶת

מַה־לִּי לְהוֹחִיל עוֹד וְאֵין תּוֹחָלֶת.

הֵן לוּ בְקֶבֶר הַתְּהוֹם נִסְתָּרְתִּי

יֵחַם בְּמִכְסֶה לִי וּפֹה נִקְרָרְתִּי.


גַּם כָּל־בְּשָׂמִים רֹאשׁ וּמִינֵי רֵיחַ

מוֹר וַאֲהָלוֹת עַל־יְצוּעִי נָפוּ

עַתָּה בְסִרְחוֹן מִסְּחִי נִדָּפוּ.

לוּ בֵין־פְּגָרִים אֵֽרְדָה בוֹרֵחַ

עוֹד יֶעֱרֵב רֵיחָם וּפֹה סוֹרֵחַ.


תִּקְצֵר גְּוִיָּתִי וְהִנָּהּ צָרָה

הֵן עַד־פְּאַת מִטָּה דְמֶשֶׁק עָרֶשׂ

תִּדְמֶה סְגוּרָה בֵּין־שְׁנֵי הַקָּרֶשׁ.

כִּי לוּ לְצִמְצוּם יַרְכְּתֵי־בוֹר סָרָה

יִרְחַב מְאֹד עוֹד לָהּ וּפֹה נֶאְסָרָה.


עוֹד עַל־יְגוֹנִי מַחֲלָה נִצַּחַת

כִּי אֶת־בְּשָׂרִי קָרְעוּ לֹא דָמוּ

כִּנָּה וְגַם פַּרְעֹשׁ וּפִשְׁפֵּשׁ קָמוּ.

לֹא כֵן בְבוֹר־שַחַת אֲשֶׁר מִתָּחַת

רִמָּה וְתוֹלָע עוֹֽקְצִים כַּמָחַט.


נֵרוֹת מֵאוֹרִי בַּעֲדִי חָשָׁכוּ

וּמְשָֽׁרְתַי אוֹתִי לְבַד עָזָבוּ

כִּי נִשְׁמְטוּ לִישֹׁן וְעוֹד לֹא שָׁבוּ.

אַף־מִלְּפָנַי אוֹהֲבַי נֶחְשָׂכוּ

קָטוֹן וְגָדוֹל נָדֲדוּ הָלָכוּ.


קַצְתִּי בְחַיָּי נָא וְנַפְשִׁי לָאָה

צָרוֹת תְּכוּפוֹת זוֹ לְזוֹ הֶרְאֵיתִי

מַשָּׂא כְבֵדוּתָם נְשׂא נִלְאֵיתִי.

אוֹי כִּי תְלָאָה עַל־תְּלָאָה בָאָה

הַמָּֽוְתָה הַמָּֽוְתָה אֶקְרָאָה.


מַהֵר צֳרִי צִירִי מְהֵרָה חוּשָׁה

בֹּא בֹּא וְלָמָּה תַעֲמֹד הַמָּוֵת

טוֹב לִי אֲשֶׁר אֵלֵךְ בְּגֵיא צַלְמָוֶת

כִּי סַעֲרַת מֵחוּשׁ אֲשֶׁר אָחוּשָׁה

הֵן שָׁבְרָה לִבִּי לְזֹאת אָנוּשָׁה.


הוֹאֵל אֱלֹהִים אֵל רְצֵה לָקַחַת

נַפְשִׁי וְרוּחִי לָךְ אֱסֹף הַפָּעַם

אָז יֶחְדְּלוּ צִירִים וְקוֹל הָרַעַם.

כִּי הֵן בְּרִדְתִּי אֶל־בְּאֵר הַשָּׁחַת

אֶשְׁקֹט וְאֶמְצָא שָׁם בְּעָפָר נָחַת.


אִם אַאֲרִיךְ נַפְשִׁי וּמָה אוֹעִילָה

מָה אֵֽשְׁבָה דוּמָם וְלֹא אֶצְעָקָה

כִּי מָה אֲיַחֵל עוֹד ואֵתְאַפָּקָה.

אִם בָּחֳלִי אַכְזָר וְזָר אוֹחִילָה

עַל־מָה אֲנִי יוֹשֵׁב וּמָה אוֹחִילָה.


לֹא יֵשׁ לְאֵל יָדִי לְהִתְנוֹעֵעַ

חוּשׁוּ עֲבָדַי מַהֲרוּ אֵתָיוּ

חוּשׁוּ לְעָזְרֵנִי רְצוּ וּבְעָיוּ

חוּסוּ עֲלֵי נֶהְפַּךְ כְּאִישׁ גֹּוֵעַ

אֵיךְ עַצְמְכֶם תַּעֲשׂוּ כְּלֹא שׁוֹמֵעַ.


אֵי לַהֲקַת בָּנַי וְכָל־הַחֶבֶל

אָן הָלְכוּ עֵבֶר וּבוּז וָחֵפֶר

אַיֵּה חֲצַר־מָוֶת וְאֵיפֹה עֵפֶר

לָמָּה תְאַחֵר בֹּא מְשָׁרְתִי הָבֶל

אֵיךְ נַאֲקִי לֹא תִּשְׁמְעִי אִיזָבֶל.


אוֹי כִּי בְקֶרֶב אָהֳלִי אֻמְלָלְתִּי

כִּי בִי כְבֶגֶד בּוֹֽגְדִים בָּגָדוּ

אִישׁ אִישׁ לְעַרְשׂוֹ הָלְכוּ נָדָדוּ

וַאֲנִי כְאִּישׁ נוֹפֵל לְבוֹר צָלָלְתִּי

לֹא עָלְתָה עַל־לֵב וְלֹא פִלָּלְתִּי.


אוֹי לִי אֲשֶׁר אֶסְבֹּל יְגוֹן הַצָּעַר

כִּי הֵם בְּהַשְׁקֵט שַׁאֲנָן נִשְׁלָמוּ

וּבְיָם שְׁנָתָם נִשְׁקְעוּ נִרְדָּמוּ.

אָמְנָם אֲנִי מַקְשִׁיב לְמוּל הַשָּׁעַר

קוֹל לֹא כְקוֹל אִישׁ אַךְ כְּקוֹל הַסָּעַר.


מַה־קּוֹל הֲמֻלָּה זֹאת אֲשֶׁר אֶשְׁמָעָה

קוֹל נַהֲמַת־יָם לֹא כְקוֹל הַנָּחַל

וּכְשַׁאֲגַת אַרְיֵה וְקוֹל הַשָּׁחֵל.

צַר־לִי מְאֹד עַתָּה וְלֹא אֵדָעָה

אִם קָרְבָה עִתִּי אֲשֶׁר אֶגְוָעָה.


יִרְעַד לְבָבִי אֶפְחֲדָה אָגוּרָה

הֵן מִשְּׁמוּעָה זֹאת מְאֹד שׁוֹמָמְתִּי

אַף מֵֽרְאוּת עֵינַי בְּחִיל נִפְעָמְתִּי

אֶתְפַּלְּצָה מִמַּחֲזֶה אָשׁוּרָה

אַרְחִיק נְדוֹד מִזֶּה וְחִישׁ אָסוּרָה.


אָמְנָם מְקוֹמִי הוּא קְצַר יָדַיִם

אוֹי לִי אֲשֶׁר לֹא אוּֽכְלָה לִבְרֹחַ

כִּי עָֽיְפָה נַפְשִׁי וְאֵין בִּי כֹחַ

גַּם חָֽשְׁלוּ רַגְלַי בְּפִיק בִּרְכָּיִם

רַעַד וְחַלְחָלָה בְּכָל־מָתְנָיִם.


מָתַי הֱבִיאוּנִי חֲבֵרַי הֵנָּה

אֵיכָה לְמָקוֹם צַר כְּזֶה הוּלָכְתִּי

אֵיךְ פֹּה כְּגוּף מוּבָס וּמֵת הָשְׁלָכְתִּי.

יִשֹּׁם לְבָבִי בִי וְגַם תִּכְלֶינָה

עֵינַי לְמַרְאָה זֶה אֲשֶׁר תִּרְאֶינָה.


הוא ענין העמקת הקבר עד גהינם

לדין חיבוט הקבר כמו שאמרו רז"ל.


אֶרְאֶה בְקָעוֹת כַּתְּהוֹם עָמָקוּ

אֶרְאֶה עֲמָקִים בַּשְּׁאוֹל מִתָּחַת

אֶרְאֶה מְחִלּוֹת מַהֲמֹרוֹת פָּחַת

אֶרְאֶה מְעָרוֹת לוֹהֲטוֹת דָּלָקוּ

אֶרְאֶה נְקִיקִים בַּשְּׁבִיב נֶחְקָקוּ.


אַבִּיט בְּקַרְקָעָם מְקוֹם יָצִיעַ

גָּפְרִית וּמֶלַח נֶעֱרָב בַּזָּפֶת

צֹאָה וְטִיט יָוֵן וְעַבְטִיט רָפֶת

יִטְבַּע מְהַלֵּךְ בּוֹ וְלֹא יַסִּיעַ

רַגְלָיו, אֲבָל עַד־קָדְקֳדוֹ יַגִּיעַ.


אַשְׂכִּיל בְּקִירוֹתָם סְבִיב כָּל־בָּיִת

כִּירוֹת וְתַנּוּרִים בְּדוּד נָפוּחַ

וּמְנַפְּחִים בָּהֶם בְּחֹזֶק רוּחַ

וּמְלַהֲטִים מִחוּץ וְגַם מִבָּיִת

עִטְרָן וְנֵפְטְ אָחוּז בְּקוֹץ וָשָׁיִת.


אָצִיץ תְּהוֹמוֹת יָם וְנִבְכֵי־מָיִם

קָפְאוּ כְמוֹ־נֵד נֹֽזְלִים בַּפָּלֶג

וּמְסוֹבֲבֵי חוֹמָה בְּהָרֵי־שָׁלֶג

קֶרַח וּבָרָד עִם כְּפוֹר שָׁמָיִם

לֹא נִהְיְתָה כָזֹאת בְּתוֹךְ מִצְרָיִם.


מֵרֹב רְעָדָה לִבְּבִי יִנּוֹעַ

כִּי אֶחֱזֶה בִּרְיוֹת בְּעֵינַי זָרוּ

לֹא הֹעֲלוּ עַל־לֵב וְלֹא שֹׁעָרוּ.

רוּחִי וְנַפְשִׁי תָּאֳרָם יָרֹעַ

לֹא־כֵן שְׂעִירִים יִֽדְמוּם לָרֹעַ.


הֵם כַּעֲנָקִים עַשְׁתְּרוֹת קַרְנָיִם

רוּמָם וְגָבְהָם יַאֲרִיךְ כַּתֹּרֶן

תַּעַר בְּרֹאשׁ אֶצְבַּע מְקוֹם צִפֹּרֶן

עָלוּ בְמִצְחֵיהֶם שְׁתֵּי קַרְנָיִם

חַדּוֹת וְכָל־גּוּפָם מְלֵא עֵינָיִם.


בִּפְנֵי לְהָבִים לוֹהֲטִים יִלְהָבוּ

וכְשׁוֹֽלְפֵי חֶרֶב וְרוֹמֵי־קָשֶׁת

עוֹטִים בְּקַשְׂקֶשֶׂת כְּמַעֲשֵׂה רָשֶׁת

בִּשְׁאָט בְּנֶפֶשׁ גַּאֲוָה יִרְהָבוּ

אִישׁ עַל־מְקוֹמוֹ עָמְדוּ נִצָּבוּ.


אָשׁוּר נְהַר דִּי־נוּר אֲשֶׁר יֵרֹמּוּ

גַּלָּיו וּמִשְׁבָּרָיו בְּזֶרֶם שֵׁטֶף

דּוֹמִים בְּמַרְאֵיהֶם לְגוּרֵי־חֶטֶף

כֵּן יִשְׂרְפוּ כֵּן יִשְׁטְפוּ יִזְרֹמוּ

כֵּן יִנְהֲמוּ כֵּן יֶהֱמוּ יֵהֹמוּ.


שָׁם אֶחֱזֶה נָחָשׁ וְשַׂרְפֵי־פָחַד

מִתּוֹךְ תְּהוֹמוֹת הַֽיְקוֹד יִשְׁרֹצוּ

מִזֶּה וּמִזֶּה בּוֹעֲרִים יִפְרֹצוּ

חַיּוֹת קְטַנּוֹת עִם־גְּדֹלוֹת יָחַד

שִׁנָּן וְצִפָּרְנָן כְּבַרְזֶל יָחַד.


מִשָּׁם לְעִמְקֵי הַתְּהוֹם יָרָדוּ

שָׁם יַעֲלוּ יִשְׂאוּ רְתִיחוֹת־קָצֶף

שָׁם בַּחֲרוֹן חֵמָה וְזַעַם קָצֶף

יַאָֽסְפוּ יַחְדָּו וְיִוָּעָדוּ

אָז יֵֽלְכוּ מִשָּׁם וְיִפָּרָדוּ.


שָׁם נֶחֶלְקוּ כָּל־נַהֲרֵי שַׁלְהָבֶת

לִפְאַת מְעָרוֹת אוֹ בְאֵרוֹת שֶׁבַע

לֹא יַאֲמִינוּ בַּעֲלֵי הַטֶּבַע

חֹזֶק שְׁבִיב הָאוֹר וְאֵשׁ צָרָבֶת

כִּי לֹא שִׁעָרוּם חוֹֽשְׁבֵי מַחְשָׁבֶת.


לִבִּי וְעֵינַי אַל־אֲשֶׁר יַשְׁגִּיחוּ

יִפָּֽלְאוּ יִתְמַהְמְהוּ יִתְמָהוּ

הֶמְיַת הֲמוֹנִים יִצְעֲקוּ יִנָּהוּ

כִּי לוֹחֲצִים אוֹתָם בְּאַף יַטְרִיחוּ

מִקֹּר לְחֹם מֵחֹם לְקֹר יַדִּיחוּ.


דֹּחוּ כְמוֹ־אֶבֶן בְּכַף הַקָּלַע

נָדִים דְּחוּפִים מִדְּחִי אֶל־דֶּחִי

יֻכּוּ בְשׁוֹט שׁוֹטֵף עֲלֵי הַלֶּחִי

וִירֻטְּשׁוּ בֵּין כָּל־נְקִיקֵי סָלַע

וּבְכָל־זְמָן יִתְכּוֹנֲנוּ אֶל־צָלַע.


אִם יִצְעֲקוּ כִּי צִיר כְּאֵבָם נֶצַח

אֵין קוֹל וְאֵין עוֹנֶה וְכֵן אֵין קֶשֶׁב

אָמְנָם סְחוּבִים יוּֽבְלוּ כַכֶּשֶׂב

אֶל־הַטְּבִיחָה וַהֲרֵגַת רֶצַח

יוּדַשׁ בְּשָׂרָם שָׁם כְּהִדּוּשׁ קֶצַח.


הַצּוֹרֲרִים אוֹתָם בְּדִין יַעֲמִידוּ

כָּל־עֵת וְעֵת לֹא יִתְּנוּם לָנוּחַ

לֹא יוּֽכְלוּ הַשְׁקֵט וְהָשֵׁב רוּחַ

זֶה בָא וְזֶה הוֹלֵךְ וְכֵן יַתְמִידוּ

בִּגְזַר נְקַם אַכְזָר עֲדֵי יַשְׁמִידוּ.


אֶל־זֶה בְּשֵׁן חַדָּה כְּצוּר יִנְשֹׁכוּ

לַלָּז בְּשָׂרוֹ יִסְרְקוּ יִשְׂרֹטוּ

וּלְזֶה בְּעַזּוּת שַׂעֲרוֹ יִמְרֹטוּ

עַל־זֶה בְּפַרְסָה כַּזְּאֵב יִדְרֹכוּ

וּבְיָד וְרֶגֶל יִכְתְּשׁוּ יַהְדֹּכוּ.


מִבַּהֲמוֹת שָׂדַי וְחַיְתוֹ־יַעַר

זָרִים וְאַכְזָרִים בְּלִי יַחְמֹלוּ

שָׁם הַמְּרוּדִים צַעֲרָם יִסְבֹּלוּ

בֵּין אִישׁ וּבֵין אִשָּׁה וְשָׂב וָנָעַר

עָשִׁיר וְרָשׁ חָכָם כְּסִיל וָבָעַר.


אֵלֶּה כְּמוֹ צִפּוֹר בְּפַח שֻׁדָּדוּ

אֵלֶּה בְּמוֹקֵשׁ נָפְלוּ נִשְׁבָּרוּ

אֵלֶּה בְּחַבְלֵי מַאֲסָר נֶאְסָרוּ

אֵלֶּה לְעִמְקֵי מַאֲפֵל הוּרָדוּ

אֵלֶּה בְּפַחִים נוֹֽקְשׁוּ נִלְכָּדוּ.


אֵלֶּה בְּחָצָץ יִגְרְסוּ שִׁנָּיִם

אֵלֶּה שְׂבֵעֵי מַר וְכוֹס קֻבָּעַת

אֵלֶּה בְּעַצְמָם תַּעֲלֶה תוֹלָעַת

אֵלֶּה תְּלוּאִים מִשְּׁתֵּי רַגְלָיִם

אֵלֶּה בְּהַלָּשׁוֹן וּבַשָּׁדָיִם.


אֵלֶּה מְרֵרָתָם כְּשֹׁד יָמֹצּוּ

אֵלֶּה דְּמֵיהֶם כַּשְּׁתִי יִבְלָעוּ

אֵלֶּה גְּלַל בִּטְנָם וְקִיא יִשְׂבָּעוּ

אֵלֶּה לְפִתִּים רֻצֲּצוּ נָרֹצוּ

אֵלֶּה לְדֹמֶן כַּסְּחִי נָפֹצוּ.


אֵלֶּה בְּשָׂרָם יִנְשְׁכוּ יֹאכֵלוּ

אֵלֶּה לְאָכְלָם גַּחֲלֵי הָרֹתֶם

אֵלֶּה תְּלוּיִם מִנְּחִיר הַחֹטֶם

אֵלֶּה בְּלַהַב אֻמְלְלוּ קָמֵלוּ

שִׁמְעוּ בְּנֵי אָדָם וְהִבָּהֵלוּ.


מִכָּל־אֲשֶׁר פֹּה אֶחֱזֶה רָגָזְתִּי

יִמַּס לְבָבִי בִי וְהָיָה מָיִם.

חוּלִי אֲדָמָה שֹׁמֲמוּ שָׁמָיִם

לֹא אֵֽדְעָה גַּם־כִּי עֲדֶן הֻגָּשְׁתִּי

הַמַּחֲנֶה הַזֶּה אֲשֶׁר פָּגָשְׁתִּי.


[הִתְיַצְּבוּ נֶגְדִּי]

הִתְיַצְּבוּ נֶגְדִּי מַזָּרוֹת הַזְּבוּל / הִתְמוֹטֲטָה מֶנִּי אֶרֶץ מִתָּחַת

אִם אֹֽמְרָה אָקוּם יַדִּיחוּנִי כְמֹץ / יִזְרוּ אֲנָשִׁים אֶל־רוּחַ בָּרָחַת

אֵשֵׁב וְכִהְיוֹתִי קֹרַח תַּחְתַּי הֲלֹא / פִיהָ אֲדָמָה אָז תִּהְיֶה פוֹתָחַת

אֶשְׁכַּב בְּחֶזְיוֹנוֹת אֶתְבָּהֵל בַּחֲלוֹם / אֶהְיֶה כְּנָס מִפַּח אֶל־תּוֹךְ הַפָּחַת

אֶתְנוֹדֲדָה תָמִיד בִּתְנוּעוֹת הַזְּמָן / תִּדְמֶה יְחִידָתִי עִיר הַנִדָּחַת

אִיגַע וְהַצְלָחָה מִמֶּנִּי רָחֲקָה / אֶחְשֹׁב וְלֹא אֵדַע עֵצָה לָקָחַת

לוּ אֶמְכְּרָה קֶמַח יָצוּמוּ יִשְׁאֲפוּ / רוּחַ וְהַחִטָּה תִּהְיֶה מוּנָחַת

מַיִם הֲאִם אַשְׁבִּיר מִמָּרָה יַחְשְׁבוּם / בִּינוּ הֲיֵשׁ כָּזֹאת רָעָה נִצָּחַת

אֶסְחַר בְּבִגְדֵי אִישׁ בִּימֵי הַקֹּר אֲזַי / יֵלְכוּ כְּכוּשִׁיִּים בִּלְתִּי מִטְפָּחַת

לוּ אֶקְבְּרָה אִישִׁים עֵץ חַיִּים יִמְצְאוּ / יִחְיוּ לְעוֹלָם לֹא יִרְאוּ הַשָּׁחַת

תִּלָאֶה יְחִידָתִי אַף־מִנִּי נָדֲדָה / כִּמְעַט בְּלִי תִקְוָה וּבְמַר צוֹרָחַת

“אָן תֵּלְכִי” אֹמַר וַתֹּאמַר “מִפְּנֵי / צָרָה גְבִרְתִּי אָנֹכִי בֹּרָחַת”.

בְּעוֹד אֶחְיֶה עֲרֵבִים לִי הֲלִיכָי / וּמָה אֶבְטַח בְּרֹב חֵילִי וְאֶחָי

וּמָה־יִצְדַּק אֱנוֹשׁ עִם־אֵל וְיִצְדַּק / וּמַה־יִּזְכֶּה יְלוּד אִשָּׁה וְאִישׁ־חָי

הֲלֹא הֵם תּוֹאֲמִים שָׁוִים בְּמִקְרֶה / תְּמִימֵי־לֵב וְנִמְהָרִים וְאִישׁ־חָי

וּמָה אִם כֵּן אֲיַחֵל עוֹד וְהִנֵּה / לְהֶבֶל אֶעֱמֹל תָּמִיד בְּשִׂיחָי

וְאֵיךְ אֶעְמֹד לְפָנֶיךָ אֱלֹהִים / וְאֵין כֹּחַ בְּרֹב מִלַּי וְשִׂיחָי

בְּיוֹם אֶשְׁכַּח לְרֶחֶם אַךְ לְרִמָּה / אֱהִי מָתוֹק כְּמוֹ־חָרוֹן כְּמוֹ־חָי

וְיִתְחַפֵּשׂ בְּרָב־כֹּחַ לְבוּשִׁי / וּמַהֵר אֵֽרְדָה מֵרוּם צְחִיחָי

אֲזַי אֶהְיֶה כְּעַרְעָר בַּעֲרָבָה / וְאֵין מַיִם אֱלֵי גַנִּי וְשִׂיחָי

אֲזַי אַתָּה בְּשַׁחֵת תִּטְבְּלֵנִי / וְאָדְמָתִי הֲכִי תִּסְּגֹּר בְּרִיחָי

בְּיַעַן אַתְּ בְּסוֹף יָמַי לְמָוֶת / תְּשִׁיבֵנִי וּבֵית מוֹעֵד לְכָל־חָי.


[מַלְאֲכֵי שָׁלוֹם]

– ͝ – – – ͝ – – – – –

מַלְאֲכֵי שָׁלוֹם הֲדַר צוּר הַשְׁמִיעוּ

מַהֲלַל מִזְמוֹר לְתוֹדָה הָרִיעוּ.


שֶׁכְּבָר עָבַר עֲנַן אֵיד הַפָּחַד

שַׁאֲפוּ זִרְחוּ בְּאוֹר אֵל הוֹפִיעוּ

הֵן בְּלֵיל אַתֶּם בְּמַר בּוֹכִים יָחַד

הֵד חֲצוֹת לַיְלָה בְּמָרָה תַדְמִיעוּ.


זַעֲוַת הַקּוֹל בְּמָרָה תַקְשִׁיבוּ

זַעֲקוֹת רָחֵל מְרוֹמִים הִבְקִיעוּ

כֹּה חֲסִידֵי־אֵל כְּמוֹכֶם יִדְאָבוּ

כִּי יְגוֹנֵיהֶם שְׁנָתָם יָנִיעוּ.


וַעֲדוּ עַתָּה קְהַלְכֶם וָשִׁירוּ

וַעֲלוּ עַל־כֵּס כְּבוֹד אֵל הַבִּיעוּ

תַּאֲוַת דּוֹדִים קְדוֹשִׁים הָעִירוּ

תַּחֲנוּן עַם־אֵל עֲדֵי־רוֹם הַגִּיעוּ.


הַאֲדִירוּ עֹז לְהַעְלוֹת קוֹלֵנוּ

הַמְּמוּ מַשְׂטִין וְכָל־צַר הַכְנִיעוּ

צַעֲדוּ־נָא עַד־הֲדֹם כֵּס צוּרֵנוּ

צַהֲלוּ וּכְנַף רְנָנִים הַרְקִיעוּ.


עַטְּרוּ כֶתֶר לְאֵל־חַי לָנֶצַח

עֹז לְכֶתֶר זֶה בְּסוֹד שֵׁם הַשְׁבִּיעוּ

יַעֲלֶה יִהְיֶה כְּצִיץ עַל־הַמֶּצַח

יַעֲנֶה אֶתְכֶם בְּעֵת בּוֹ תַפְגִּיעוּ.


רוֹמֲמוּ שַׁדַּי וְטוּבוֹ תַשְׁפִּיעוּ

רִנֲנַת מִזְמוֹר לְתוֹדָה הָרִיעוּ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
