יָהּ אֶת סֻכַּת

לכבוד ז׳ אושפיזין קדישין


יָהּ אֶת סֻכַּת דָּוִד תָּקִים

בִּזְכוּת שִׁבְעַת הַצַּדִּיקִים.


בִּזְכוּת אֶזְרָח תְּמִימֶךָ

מְשֹׁךְ חַסְדָּךְ לְעַמֶּךָ

וְנָא לְמַעַן שְׁמֶךָ

אַל נָא תָשֵׁב אוֹתָם רֵיקִים.

יָהּ אֶת סֻכַּת


בִּזְכוּת יִצְחָק עֲקֵדֶךָ

בָּרֵךְ דּוֹרְשֵׁי יִחוּדֶךָ

וְנָא גְבוּרוֹת יָדֶיךָ

בְּחַסְדָּךְ יִהְיוּ נִמְתָּקִים.

יָהּ אֶת סֻכַּת


בִּזְכוּת יַעֲקֹב אִישׁ תָּמִים

הוּא נֶגֶד קַו הָרַחֲמִים

תִּבְנֶה בֵיתְךָ בֵּית עוֹלָמִים

קַיָּם בִּשְׁמֵי שְׁחָקִים.

יָהּ אֶת סֻכַּת


בִּזְכוּת רָעְיָא מְהֵימָנָא

זָכָה לְפָנִים שֶׁל בִּינָה

חוּשָׁה תֵן לָנוּ חֲנִינָה

וּפְדֵה עַם לְךָ שׁוֹקְקִים.

יָהּ אֶת סֻכַּת


בִּזְכוּת הַכֹּהֵן אַהֲרֹן

בַּטֵּל כָּל הָאַף וְחָרוֹן

וְעַמָּךְ תָּמִיד רוֹן יָרוֹן

וּבְשִׁמְךָ יִהְיוּ דְבוּקִים.

יָהּ אֶת סֻכַּת


בִּזְכוּת יוֹסֵף צַדִּיקֶךָ

אֲשֶׁר שָׁמַר בְּרִיתֶךָ

גַּלֵּה כְבוֹד מַלְכוּתֶךָ

קַבֵּץ נִדָּחִים וּרְחוֹקִים.

יָהּ אֶת סֻכַּת


בִּזְכוּת דָּוִד יְדִידֶךָ

בְּנֵה מִשְׁכַּן כְּבוֹדֶךָ

וְאַמֵּץ עַם נְדוּדֶךָ

לַמְּדֵם תּוֹרָה וְחֻקִּים.

יָהּ אֶת סֻכַּת


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
