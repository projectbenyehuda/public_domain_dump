שַׁבְּחִי נַפְשִׁי

שַׁבְּחִי נַפְשִׁי לְצוּרֵךְ / כִּי לִכְבוֹד עֻזּוֹ יְצָרֵךְ

לוֹ תִכְרַע קוֹמָה וּבֶרֶךְ / גַּם נִשְׁמַת כָּל חַי תְּבָרֵךְ


שֵׁם נִכְבָּד נוֹרָא וְאָיֹם / שִׂימִי יִרְאָתוֹ בְלִבֵּךְ

יִחוּדוֹ זִכְרִי בְכָל יוֹם / צִפְנִי אַהְבָתוֹ בְקִרְבֵּךְ

וִיצַו לָךְ יֶשַׁע וּפִדְיוֹם / מִצַּר שֶׁעוֹמֵד לְרִיבֵךְ

כִּי הוּא חַיַּיִךְ וְאֹרֶךְ / יָמַיִךְ יִשְׁעֵךְ וְאוֹרֵךְ


לִמְדִי לַעֲשׂוֹת אֶת רְצוֹנוֹ / וּלְכִי אַחַר מַהֲלָכָיו

תִּנָּצְלִי מֵאֵשׁ חֲרוֹנוֹ / תִּוָּשְׁעִי מֵחֹם פְּלָכָיו

תָּבֹאִי לִרְעוֹת בְּגַנּוֹ / אַתְּ עִם כָּל קֹוָיו וְחוֹכָיו

עֵת תָּשׁוּבִי אֶל מְקוֹרֵךְ / אָז יִיטַב לִי בַעֲבוּרֵךְ


מֵהַבְלֵי עוֹלָם וְהֶתְלוֹ / הִמָּלְטִי נֶפֶשׁ יְחִידָה

מַאְסִי שִׂמְחָתוֹ וְגִילוֹ / זִכְרִי נָא יוֹם הַפְּרִידָה

עֵת תֶּחְזִי צוּרֵךְ בְּגָדְלוֹ / תָּנוּחִי מִן הָעֲבוֹדָה

הִתְרַצִּי אֵלָיו בְּשִׁירֵךְ / טֶרֶם שׁוּבֵךְ אֶל עֲפָרֵךְ


הֵא לָךְ מַחְשָׁבָה נְכוֹנָה / נִשְׁמַת אֵלִי הַטְּהוֹרָה

מִתַּאֲוַת גּוּף הַצְּפוּנָה / הִבָּדְלִי זַכָּה וּבָרָה

לֵאלֹהֵי קֶדֶם מְעוֹנָה / אַתְּ אָמָה בַּגֵּו עֲצוּרָה

כִּי הוּא תוֹחַלְתֵּךְ וְשִׂבְרֵךְ / הָרוֹפֵא חָלְיֵךְ וְשִׁבְרֵךְ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
