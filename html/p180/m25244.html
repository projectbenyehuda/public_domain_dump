שַׁחַר קַמְתִּי לְהוֹדוֹת

שַׁחַר קַמְתִּי לְהוֹדוֹת / לְךָ אֱלֹהֵי תְהִלָּתִי

וַאֲרַנֵּן לַבֹּקֶר / וְאוֹדִיעֲךָ חַטָּאתִי

יִנָּתֶן לִי בָּזֶה / שָׂכָר לִפְעֻלָּתִי

נַפְשִׁי בִּשְׁאֵלָתִי / וְעַמִּי בְּבַקָּשָׁתִי


לְפָנִים זֹאת בְּיִשְׂרָאֵל / בְּעֵת מַקְרִיבֵי קָרְבָּן

אִם חַטָּאתוֹ כְּשָׁנִים / כֻּלּוֹ הָפַךְ לָבָן

אֶדַּדֶּה כָל שְׁנוֹתַי / אֲנִי תָמִיד עַל חֻרְבָּן

וְאֵין לִי לְבַד מִלָּתִי / וְעוֹלָתִי תְפִלָּתִי


מִזְבֵּחַ בִּהְיוֹתוֹ / וְקָדָשִׁים לַזְּבִיחָה

וְהֵבִיא אִישׁ אֶת זִבְחוֹ / וְנִכַּפֵּר בִּסְלִיחָה

אָפֵס מִנִּי מְכַהֵן / וְאֵין זֶבַח וְאֵין מִנְחָה

תּוֹדָתִי זְבִיחָתִי / וְשִׂיחָתִי מִנְחָתִי


הֵן בִּהְיוֹת הָעֲבוֹדָה / וְכֹהֲנִים עַל מִשְׁמֶרֶת

הֲלֹא חַטָּאת מַכְשֶׁרֶת / וְהָעוֹלָה מְכַפֶּרֶת

וְאֵין חַטָּאת וְאֵין עוֹלָה / וְלֹא חֵלֶב וְיוֹתֶרֶת

לְבַד אֶשְׁפֹּךְ אֶת רִנָּתִי / וְאַפִּיל אֶת תְּחִנָּתִי


קֶדֶם בִּהְיוֹת מִקְדָּשׁ / בְּקוּם זְרִיזִים בְּאַשְׁמֹרֶת

טְהוֹרִים נִצְּבוּ לְהָפִיס / הַחֲדָשִׁים לִקְטֹרֶת

וְאֵין לְבוֹנָה וְאֵין קְטֹרֶת / וְנִשְׁאַרְתִּי שְׁחַרְחֹרֶת

בְּהִתְוַדּוֹת חַטָּאתִי / כְּמוֹ קָרְבָּן תּוֹדָתִי


חֲזֵה קָדוֹשׁ כִּי יָדִי / כָּבְדָה עַל אַנְחָתִי

וְאֵין מִי יָגֵן בַּעֲדִי / וְאֵין לִי בֵית מְנוּחָתִי

לְבַד בְּךָ אַזְכִּיר שְׁמֶךָ / מָעֹז צוּר יְשׁוּעָתִי

תַּעֲבִיר אֶת אַשְׁמָתִי / וְעָנְתָה בִּי צִדְקָתִי.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
