אֲנוּשִׁים וַעֲנוּשִׁים

וּבְכֵן מָה אָדָם וַתֵּדָעֵהוּ בֶּן אֱנוֹשׁ וַתְּחַשְּׁבֵהוּ

יְיָ מָה אָדָם


אֲנוּשִׁים וַעֲנוּשִׁים בְּקָשְׁיָם / אֲמֵלִים וַעֲמֵלִים בְּמִרְיָם

אַף רוֹאֵיהֶם יֹאמְרוּ אַיָּם / אָזְלוּ מַיִם מִנִּי יָם / אַךְ הֶבֶל בְּנֵי אָדָם


נְפָלִים שְׁפָלִים נִטְמוּ / נוֹדְדִים הֶבֶל יֶהֱמוּ

נִצְחָם וְכֹחָם יִתְּמוּ / נִמְשַׁל כַּבְּהֵמוֹת נִדְּמוּ / נִבְעָר כָּל אָדָם


יוֹם יוֹם זַדְנוּ וְאָשַׁמְנוּ / יַחְדָּו בֹּשְׁנוּ וְנִכְלַמְנוּ

יָעַצְנוּ רַע וְתֹךְ יָזַמְנוּ / יַחְפְּשׂוּ עוֹלוֹת תַּמְנוּ / יֵצֶר לֵב הָאָדָם


שׁוֹקֵד גָּדִישׁ וְרֶגֶב חוֹבֵשׁ / שׁוֹכֵן גּוּשׁ וְרִמָּה לוֹבֵשׁ

שָׁוֶה לַחֲצִיר גַּגּוֹת עָבֵשׁ / שֶׁקַּדְמַת שָׁלַף יָבֵשׁ / שֶׁלֹּא יִמְצָא הָאָדָם


לְקוּחִים לְדַבֵּר אִתָּם עַזּוֹת / לָתֵת לָהֶם בְּתֹפֶת אֲחֻזּוֹת

לְכוּדִים כַּצִּפֳּרִים הָאֲחוּזוֹת / לוּ חָכְמוּ יַשְׂכִּילוּ זֹאת / לִבּוֹת בְּנֵי אָדָם


מוּעָד וּמוּכָן לְקִיטוֹר גַּחַל / מְתֻקָּן וְנִסְכָּן לְתִגְרַת שַׁחַל

מָאוֹס יִמְאַס חַיָּיו בְּבַחַל / מָתְקוּ לוֹ רִגְבֵי נַחַל / מַה יִּתְרוֹן לָאָדָם


הֲמוֹנָם סָג וְיָשָׁר אֵין בָּם / הַוּוֹת בְּקִרְבָּם וְעָמָל בְּאָרְבָּם

הַמֵּבִין חֻבָּם וְחוֹפֵשׂ קִרְבָּם / הַיּוֹצֵר יַחַד לִבָּם / הִשְׁקִיף עַל בְּנֵי אָדָם


בְּבוֹא קִצּוֹ תְּסָעֲרֵהוּ / בְּאַמַּת בִּצְעוֹ תְּנַעֲרֵהוּ

בְּחֶרְמוֹ מָוֶת יְגֹרֵהוּ / בּוֹר כָּרָה וַיַּחְפְּרֵהוּ / בַּאֲשֶׁר הוּא סוֹף כָּל אָדָם


רָעוּץ וְרָצוּץ חֶלְיוֹ וְעֶדְיוֹ / רָאוּי לְשַׁלֵּם נִשְׁיוֹ בְקָשְׁיוֹ

רְגָבִים יְבַלּוּ שָׁרָשָׁיו וּפִרְיוֹ / רוֹאָיו יֹאמְרוּ אַיּוֹ / רַבָּה רָעַת אָדָם


בִּמְלֹאת סִפְקוֹ עֵצֶר יְנִיאֶנּוּ / בָּלַע חַיִל וַיְּקִיאֶנּוּ

בָּדָד יֵשֵׁב וְלֹא יִרְאֶנּוּ / בַּשָּׁלוֹם שׁוֹדֵד יְבוֹאֶנּוּ / בְּקוּם רְשָׁעִים יִסָּתֵר אָדָם


יְקָרוֹ נִשְׁחַת וְקַרְנוֹ נִגְדָּע / יְחִידָתוֹ מִשַּׁחַת בַּל יִפְדָּע

יוֹם אֲשֶׁר אֵלָיו יִתְוַדָּע / יְשַׁלֵּם אֵלָיו וְיֵדָע / יוֹדֵעַ מַחְשְׁבוֹת אָדָם


יֻדַּד כְּחֶזְיוֹן לַיְלָה בְּחֵמָה וְסָעַר / יְסֹעֵר כְּמוֹץ בְּסוּפַת יָעַר

יְגֹעַשׁ בְּגַעַשׁ כְּיָשִׁישׁ כְּנָעַר / יַחַד כְּסִיל וָבַעַר / יוּקְשִׁים בְּנֵי הָאָדָם


הִנֵּה אִם נֶחְפַּשׂ וְנִבְדַּק / הוּשַׁת כֻּלּוֹ סָג וְנֶחְדָּק

הָיֹה יִהְיֶה כְּאָבָק דַּק / הַאֱנוֹשׁ מֵאֱלוֹהַּ יִצְדָּק / הַמְלַמֵּד אָדָם


וְזָרִים בִּנְוֵהוּ יַאֲהִילוּן / וְנָכְרִים בְּאוֹצְרוֹתָיו יְגִילוּן

וְהוּא לִקְבָרוֹת יוֹבִילוּן / וּכְאוֹרֵחַ נָטָה לָלוּן / וַיֶּחֱלָשׁ וַיִּגְוַע אָדָם


דַּל לֹא יֶחְדַּל מֵעָווֹן וְאַשְׁמָה / דָּתוֹ אָוֶן עֲצָתוֹ תָרְמָה

דַּרְכּוֹ עָוֶל נְתִיבוֹ עָרְמָה / דִּבְרֵי פִיו אָוֶן וּמִרְמָה / דִּבְרַת בְּנֵי אָדָם


הוֹבֵר אֲשָׁמִים בִּצְרוֹר נֶחְתָּמִים / הָרֵי פְשָׁעִים עָלָיו נִכְתָּמִים

הִמֵּס יִמַּס בְּגַחֲלֵי רְתָמִים / הִנֵּה בִּהְיוֹתוֹ תָמִים הֶבֶל כָּל אָדָם


גָּזוּר מִטִּיט וְעַל טִיט נִסְמָךְ / גֵּווֹ פִתְאֹם יִתַּר וְיִמָּךְ

גּוּשׁ עָפָר עָנִי וָמָךְ / גֵּר אָנֹכִי עִמָּךְ / גַּם בְּנֵי אָדָם


בִּתְקֹף עָלָיו כִּידוֹר מֶלֶךְ / בְּכֵן יָשׁוּב לְאַרְצוֹ כְּהֵלֶךְ

בְּשׁוּבוֹ שָׁם עָרֹם כְּחֵלֶךְ / בַּהֶבֶל בָּא וּבַחֹשֶׁךְ יֵלֵךְ / בְּמוֹת אָדָם


יַעֲזֹב לַאֲחֵרִים רְכוּשׁוֹ / יְצוּעוֹ רִמָּה וְעָשׁ לְבוּשׁוֹ

יֵקַר פִּדְיוֹן נַפְשׁוֹ / יָשׁוּב עֲמָלוֹ בְרֹאשׁוֹ / יוֹם עַנּוֹת אָדָם


רוֹדֵף בְּחַיָּיו רוּחַ נָשׁוּב / רוֹדֶה בְמוֹתוֹ חֲמַת עַכְשׁוּב

רָאוּי כְּאֶפֶס וְאַיִן חָשׁוּב / רוּחַ הוֹלֵךְ וְלֹא יָשׁוּב / רוּחַ בְּנֵי הָאָדָם


וּמֵרֹב מְצוּקָיו בְּחַיָּיו יָרֵב / וּמוֹתוֹ עָרֵב וְיוֹמוֹ עוֹרֵב

וְקִצּוֹ בְקִרְבּוֹ לְנַפְשׁוֹ אוֹרֵב / וַיֵּלֶךְ הָלוֹךְ וְקָרֵב / וְרִחַק יְיָ אֶת הָאָדָם


לָרִיק יִיגַע וְיִדַּל מְאֹדוֹ / לַשָּׁוְא יַעֲמֹל וְיֶהְבַּל בְּעוֹדוֹ

לָכֵן לֹא יִשָּׂא מְאוּמָה בְּיָדוֹ / לֹא יֵרֵד אַחֲרָיו כְּבוֹדוֹ / לֹא יִכּוֹן אָדָם


חֳדָשָׁיו חִצֵּץ שׁוֹכֵן שְׁמֵי עָרֶץ / זָעוֹךְ נִזְעַךְ וְנִקְרַץ קָרֶץ

קָמוֹט נִקְמַט לְכָלָה וְחָרֶץ / זִכְרוֹ אָבַד מִנִּי אָרֶץ / זֶה חֵלֶק אָדָם.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
