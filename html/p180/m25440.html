שְׁעֵה לַחַשׁ קוֹרְאֶיךָ

שְׁעֵה לַחַשׁ / קוֹרְאֶיךָ / וֶעֱנוּתָם אַל תִּבְזֶה

לְשַׁוְעָתָם / הַט אֹזֶן / כְּבִימֵי אֲבִי כָל חוֹזֶה

מִמָּרוֹם / כְּהוֹדַעְתּוֹ / בַּמַּרְאֶה וּבַמַּחֲזֶה

וְהִשְׁמִיעַ / סְלַח נָא לַ־/ עֲווֹן הָעָם הַזֶּה


שְׁקֹד דֶּלֶת / וְתוֹחֶלֶת / אֲזַי תִּמְצָא אֱנוֹשׁ רָזֶה

לְמַד הֵיטֵב / בְּלֹא קֶטֶב / כְּפָץ צוּרָךְ בְּיַד חוֹזֶה

מְאַס הֶבֶל / עֲטֵה אֵבֶל / עֲלֵי גוּפָךְ אֲשֶׁר יִרְזֶה

הֲדֹף פִּשְׁעָךְ / עֲזֹב רִשְׁעָךְ / וְשִׂים דָּבְקָךְ בְּדַת כִּי זֶה

תְּהִי עִסְקָךְ / וְגַם חֻקָּךְ / בַּת מַזֶּה וּבֶן מַזֶּה

וְגַן עֵדֶן / תְּהֵא נֵדֶן / לְנִשְׁמָתָךְ וְאַתְּ תֶּחְזֶה

תְּהִי קַשּׁוּב / שְׁאָר יָשׁוּב / שְׁאָר יַעְקֹב לְאֵלִי זֶה


בְּאַוָּתִי / וְתַאְוָתִי / אֲנִי בָאתִי לְאֵל אֶדְרֹשׁ

רְחֹשׁ מוּסָר / לְהִתְיַסָּר / וְתוֹרָתוֹ אֲזַי אֶפְרֹשׁ

בְּמִשְׁפָּט בַּר / כְּמוֹ דֻבַּר / כְּשֶׁמֶן טוֹב עֲלֵי הָרֹאשׁ

יְהִי רֵיחוֹ / וּמַשְׁבִּיחוֹ / בְּרָב בֹּשֶׂם כְּמוֹ מֵרֹאשׁ

וְתוֹךְ קָדְשִׁי / יְהִי נַפְשִׁי / וְשָׁם אֶקְפָּא וְשָׁם אֶקְרֹשׁ

וְאָסִיר צַר / אֲשֶׁר הֵצַר / וְיִתָּלֶה כְפִיר בִּבְרוֹשׁ

וְאֵלִי חִישׁ / בְּעֵת תָּחִישׁ / חֲדַל אִישִׁים וְגַם נִבְזֶה


וְדַע פִּצְעִי / וְשׁוּר בִּצְעִי / וְגָלוּתִי בְּאִסְפַּמְיָא

הֲלֹם הוֹלְמִי / וּמַכְלִימִי / יְהֵא סָחוּב בְּסֵרַטְיָא

וְחִישׁ יוֹמוֹ / לְהֵתַמּוֹ / כְּמוֹ תַמּוּ בְאַנְטוֹכְיָא

זְקֹר וַעְקֹר / בְּאַבְנֵי קֹר / נֶשֶׁר שֶׁבַּעֲרַבְיָא

וְאֶתְפָּאֵר / בְּנִין וּשְׁאֵר / כְּהִבְטִיחִי יְלִיד בִּתְיָה

וְכָל אוֹיְבַי / וּמַכְאִיבַי / יִפְּלוּ בְאֶרֶץ נְשִׁיָּה

וְתוֹךְ מִקְדָּשׁ / הֶחָדָשׁ / דְּמֵי קָרְבָּן אֲזַי אַזֶּה


מְעוֹן תקר / כְּשָׁתָה קַר / וְגַם הֶחְרִיב יְאוֹר מָצוֹר

הֲבָלִים פָּץ / וּבָם חָפַץ / וְקָם כָּרוֹז בְּחוֹמַת צוֹר

דְּמֵי הַשֶּׂה / בְּיוֹם כֶּסֶא / חֲקָקוּהוּ וּבוֹ לִפְצֹר

וּבָא מִצַּד / לְכָל נָד צָד / וְכָל אִישּׁ גָּשׁ לְיַד מַעְצוֹר.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
