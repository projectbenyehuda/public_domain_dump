בְּטֶרֶם אֲגֻדַּת

וּבְכֵן בְּטֶרֶם הָרִים

וֵאלֹהִים מַלְכִּי מִקֶּדֶם


בְּטֶרֶם אֲגֻדַּת שְׁלֹשִׁים וּשְׁתַּיִם מְנוּיָה

וְעַד לֹא אָמוֹן שַׁעֲשׁוּעִים קְנוּיָה

אֱלֹהוּתְךָ אַחַת בְּלִי שְׁנוּיָה / מְעוֹנָה אֱלֹהֵי קֶדֶם


בְּטֶרֶם נוֹצַצְתָּ אוֹר צָפוּן וַתַּבְרִיר

וְעַד לֹא נֶחְבַּר נִיצוֹץ לַהַב עִם סַגְרִיר

נָאוֹר הָיִיתָ בְּאֶפֶס שַׁפְרִיר / הֲלֹא אַתָּה מִקֶּדֶם


בְּטֶרֶם יִקְרַת כִּסֵּא חֶבְיוֹן הַנֶּהְדָּר

וְעַד לֹא יִפְעַת פַּרְגּוֹד הַנִּגְדָּר

יָחִיד וְנִשְׂגָּב שִׁמְךָ הַנְֶאְדָּר / כִּי אֶזְכְּרָה מִקֶּדֶם


בְּטֶרֶם שִׁקַּעְתָּ מוּצָק בִּשְׁאוֹל וְתַחְתִּיָּה

וְעַד לֹא שִׁוִּּיתָ קַו יָרֹק עַל צִיָּה

שִׁפְרַרְתָּ חוּג גֻּלַּת עֲלִיָּה / וּמוֹצָאוֹתָיו מִקֶּדֶם


בְּטֶרֶם לִכַּדְתָּ זְרוּמֵי צוּלַת מַשָּׂאוֹן

וְעַד לֹא לֹהֲקוּ מִגַּיְא בְּכֹחַ וָאוֹן

לֶכֶת גַּלְגַּל לֶכֶת קִפָּאוֹן / וּמֵרֹאשׁ הַרֲרֵי קֶדֶם


בְּטֶרֶם מָסַר נֵרוֹת בְּגֻלַּת חוּג הַנִּשְׁקָף

וְעַד לֹא מָשַׁל בִּצְבָא רוֹם הַנִּזְקָף

מִלֵּא שֻׁלְחַן אֲדָמָה דֶּשֶׁן וַיַּקַּף / גַּן בְּעֵדֶן מִקֶּדֶם


בְּטֶרֶם הִתְאִיר דִּמְיוֹן צַלְמוֹ וַיְנוֹפֵף

וְעַד לֹא הֱבִיאוֹ תַחַת מְסוּכָה לְהִסְתּוֹפֵף

הִשְׁלִיחַ זִיז מִקְנֶה וְצִפּוֹר מְעוֹפֵף / אֲשֶׁר צִוָּה מִימֵי קֶדֶם


בְּטֶרֶם בְּכֹחוֹ סִבֵּב הָאוֹר בַּגַּלְגַּל

וְעַד לֹא בֵאֵר יָמִים לְהָאִיר כָּל מַעְגָּל

בֵּרַר יוֹם הַשְּׁבִיעִי לָעָם הַנִּסְגָּל / חָשַׁבְתִּי יָמִים מִקֶּדֶם


בְּטֶרֶם רִדַּד אֲפֻדַּת שִׁבְעָה שְׁחָקִים

וְעַד לֹא רִפַּד אֲגֻדַּת שִׁבְעָה אֲרָקִים

רָמַז מֵאָז שִׁבְעָה דְבָרִים לְהָקִים / אַבִּיעָה חִידוֹת מִנִּי קֶדֶם


בְּטֶרֶם יָעַד לִבְנוֹת בִּירָה וְאַרְמוֹן

וְעַד לֹא יָזַם לְגַלּוֹת כָּל מַטְמוֹן

יָעַץ לִקְנוֹת דָּת קְרוּאָה אָמוֹן / רֵאשִׁית דַּרְכּוֹ קֶדֶם


בְּטֶרֶם הִבְדִּיל וְהִגְדִּיל כָּל צָפָה

וְעַד לֹא הֵבַר לְשׁוֹנוֹת כָּל שָׂפָה

הִקְדִּים מִי זֹאת בָּרָה וְיָפָה / זְכֹר עֲדָתְךָ קָנִיתָ קֶדֶם


[בְּטֶרֶם ו

וְעַד לֹא ו

ו      קֶדֶם]


בְּטֶרֶם דִּגֵּל הַדְרַת הַזִּיו הַנִּמְתָּק

וְעַד לֹא דָלָה הָאוֹר הַנִּבְהָק

דִּלָּה כִסֵּא הַכָּבוֹד וְהוּא הַנֶּעְתָּק / עַל הָהָר אֲשֶׁר מִקֶּדֶם


בְּטֶרֶם הִתְאִיר גְּלִילוֹת אֲדָמָה בְמוֹעֵצוֹת

וְעַד לֹא הִצִּיב גְּבוּלוֹת עַמִּים נִקְבָּצוֹת

הֵכִין לְעַמּוֹ צְבִי לְכָל הָאֲרָצוֹת / מִימֵי קֶדֶם


בְּטֶרֶם גִּבֵּל וַיִּתְקַע אֶדֶן

וְעַד לֹא גִלְגֵּל אֹהֶל אַפֶּדֶן

גִּנֵּן לְחוֹסָיו טִירַת גַּן עֵדֶן / עַלִּיזָה מִימֵי קֶדֶם


בְּטֶרֶם בִּסֵּס כִּידוֹר וּמְדוּרַת הָעֶלֶק

וְעַד לֹא בִצֵּר נַחַל נִרְשַׁף בְּדֶלֶק

בִּעֵר לַנְּסוֹגִים כִּידוֹדֵי עֶלֶק / אֲשֶׁר הָיוּ מִימֵי קֶדֶם


בְּטֶרֶם יְהִירִים חַנְפֵי לֵב פָּרְצוּ פֶרֶץ

וְעַד לֹא יָעֲצוּ הַוּוֹת לְכָלָה וָחֶרֶץ

יָעַד תְּשׁוּבָה לְהַצִּילָם מִקֶּרֶץ / וְהָיוּ בָנָיו כְּקֶדֶם


בְּטֶרֶם רֶצֶף כָּל פְּעֻלָּה הָיָה נֶאְדָּר

וְעַד לֹא רְאוּת כָּל נִמְצָא הָיָה נֶהְדָּר

רוֹמְמוּת עֹז בְּכָל פֶּה נִסְדָּר / לָרוֹכֵב בִּשְׁמֵי קֶדֶם


בְּטֶרֶם וְלִפְנֵי קְרֹא עֲמוּסָיו זַעֲקָם הִקְשִׁיב

וְעַד לֹא וִדּוּי פִּשְׁעָם סְלִיחָה יָשִׁיב

וְנַחֲלַת חֶבְלוֹ כְּמֵאָז יוֹשִׁיב / תִּשְׁכֹּן כִּימֵי קֶדֶם


בְּטֶרֶם לִפְנוֹת עֶרֶב שִׂיחָם יֶעֱרַב

וְעַד לֹא לַיִל יִקְרַב יִשְׁעָם יִקְרַב

לְהָשִׁיב לְקַדְמוֹת שְׁאֵרִית עַם רָב / יִתְּנֵנִי כְיַרְחֵי קֶדֶם.
