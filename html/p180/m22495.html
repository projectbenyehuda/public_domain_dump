אחי מיֻדעי

עוד על נסיעתו מאת אוהביו


אַחַי מְיֻדָּעַי, בְּיוֹם נָסָעוּ / לִבִּי וּבִגְדוֹתַי הֲלֹא קָרָעוּ

רָאוּ דְמָעַי עַל לְחָיַי נוֹזְלִים / יֶחְצוּ כְבֵדִי בִי - וְלֹא יָדָעוּ!

לֹא כִי תְחַלִּי, עַד קְרָאתִים כִּמְעַט - / וַיִּתְנוּ עֹרֶף וְלֹא שָׁמָעוּ

אֶדְרֹשׁ וְאֶמְצָאֵם בְּתוֹךְ לִבִּי, אֲבָל / לֹא אֵדַע אֵי אָהֳלָם תָּקָעוּ

יוֹם עָקְרוּ לֵבָב אֲשֶׁר נִמְשַׁךְ בְּחוּט / הָאַהֲבָה - הָאַהֲבָה נָטָעוּ

יוֹם, רָכְבוּ בוֹ מַרְכְּבוֹת פֵּרוּד, עֲלֵי / סוּסָיו יְרִימוּן שׁוֹט וּבָם נָגָעוּ

לוּ תֶחֱזֵם, יִשְׂתָּרְגוּ עָלַי כְּמוֹ / אַנְשֵׁי נְקִיִּים אַחֲרֵי הָרְשָׁעוּ

אוֹ תֶחֱזוּ רֶדֶת מְטַר קַיִץ עֲלֵי / פִלְחֵי לְחָיֵיהֶם - וְלֹא נִבְצָעוּ!

וּבְעֵת רְאוֹתִי קוֹל בְּכִיתָם אַחֲרַי - / כָּבְדוּ עֲנָנַי, עַד אֲשֶׁר נִבְקָעוּ

כִּמְעַט יְהִי נִסְתָּר כְּבֵדִי מֵאֱנוֹשׁ / לוּלֵי שְׁתֵּי עֵינַי אֲשֶׁר נִדְמָעוּ

אָכֵן, כְּשׁוֹאֵל נִפְלְאוֹתָיו, נִסְתְּרוּ / מֵאוֹהֲבָיו, לוּלֵי אֲשֶׁר נִשְׁמָעוּ

לָמַד לְבָבִי לַהֲלֹךְ תָּמִיד עֲלֵי / הָרֵי מְזִמּוֹתָיו אֲשֶׁר נִרְקָעוּ

אֶזְכֹּר אֲחִי שֶׁמֶשׁ וְסַהַר, אֶזְכְּרָה / כָּל זָרְחוּ עָלַי וְכָל יָפָעוּ

יוֹנַת רְחוֹקִים! שַׁאֲלִי שָׁלוֹם לְאָח / כִּי מַחְשְׁבוֹתַי כָּל יָמַי בּוֹ תָעוּ

כָּלָה לְרֵעִי לֵב, וְנַפְשִׁי נִכְסְפָה / אֵלָיו מְאֹד, עַל כֵּן דְּבָרַי לָעוּ

יוֹם נִשְׁבְּעוּ יָמִים לְהִתְעוֹלֵל דָּמִי - / יָקוּם אֲמָרָם, כִּי בְךָ נִשְׁבָּעוּ

אוֹ גָרְעוּ יָמִים בְּאֹרֶךְ הַנְּדוֹד - / יָסְפוּ אֲהָבַי לָךְ וְלֹא נִגְרָעוּ

כַּשִּׁיר, אֲשֶׁר אִם לֹא יְדָעוֹ לַעֲלוֹת - / בָּתָיו אֱלֵי שַׁחַק אֲזַי נָגָעוּ

שִׁיר - נִמְשְׁלוּ בָתָּיו לְמַשְׂכִּיּוֹת חֲלִי / לוּלֵי אֲשֶׁר בִּכְנַף נְדוֹד נָסָעוּ

לָכֵן חֲקֹר נִיב סִפְרְךָ, פֶּן יִקְרְךָ / מִקְרֵה שְׁנֵי אוֹרִים אֲשֶׁר נִפְגָּעוּ

אָעִיד וְאֶתְמַהּ אִם מְשָׁכָם לִבְּךָ / מִנַּחֲלֵי דַעַת - וְלֹא נִמְנָעוּ!

שָׂרֵי תְבוּנוֹת מִמְּךָ חָלוּ, כְּמוֹ / שָׂרֵי סְפָרַד מִמְּךָ אַךְ זָעוּ

יִיטַב לְךָ, כִּי יִהְיוּ אַחְרֵי תְמוֹל, / לוּ כַעֲבָדִים - הֵן לְךָ נִרְצָעוּ!

שָׂמוּ תְהִלָּתָךְ אֲחוֹת דָּתָם וְעַל / אַפְסֵי אֲדָמָה חַסְדְּךָ זָרָעוּ

סָר זִכְרְךָ הַטּוֹב עֲלֵי תֵבֵל וְנָע / שֵׁם מִפְעֲלוֹתֶיךָ - וְהֵם לֹא נָעוּ

אוֹרוּ בְּנֹעַם שִׂכְלְךָ חָשְׁכֵי תְהוֹם / אַךְ שַׁחֲרֵי לֵילוֹת בְּךָ נִבְקָעוּ

עַל כָּל מְשַׂנְאֶיךָ יְמִינָךְ תַּעֲלֶה / אַף כָּל מְקַנְאֶיךָ לְךָ יִכְרָעוּ

כִּי נִפְלְאוֹת שִׁיר חֻדְּשׁוּ בָךְ אַחֲרֵי / כָלוּ, וְחָיוּ - אַחֲרֵי גָוָעוּ

תָּעוּ לְסַפֵּר מַחְשְׁבוֹתַי תַּאֲוַת / לִבִּי, וּבוֹשׁוּ מֵאֲשֶׁר יָגָעוּ

אֶשְׁאַל בְּנוֹת יָמִים לְבִלְתִּי הַאֲרִיךְ / יוֹם הַנְּדוֹד, אִם שְׁאִלְתִּים - שָׁעוּ?

כַּמָּה יְדִידִים, בַּזְּמַן הִתְפָּרְדוּ, / כֵּן צָעֲקוּ עַל זֹאת - וְלֹא נוֹשָׁעוּ!

אִם נוֹלְדָה תֵבֵל וְיָמִים לִמְּדָה / לַמְרוֹת, וּמוּסַר דַּרְכְּךָ פָרָעוּ -

הָאֵל יְשׁוֹבֵב בַּחֲסָדָיו, אַחֲלַי, / רֵעִים אֲשֶׁר מֵעִיר לְעִיר הֻסָּעוּ!

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
