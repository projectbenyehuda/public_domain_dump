שׁוּר יְדִיד

שׁוּר יְדִיד כְּאֵבִי / וַחֲמַס יְרִיבִי / וְלוֹחֲמִי

יִלְעֲגָה לְצִירִי / צַר וְיֵרְדְּ עָרִי / וְהוֹלְמִי


לוֹהֲטִים לְאֵידִי מְהַלְּלִים / אֶת כְּבוֹד זְקֵנַי מְחַלְּלִים / בּוֹזְזִים יְקָרִי וְשׁוֹלְלִים


שִׁלְּחוּ בְקִרְבִּי / גַּחֲלֵי שְׁבִיבִי / וְטַעֲמִי

מַר בְּמַאְסָרִי / גַּם דְּמַע נְהָרִי / בְּלִי דְמִי


מוֹעֲדַי כְּקֶדֶם תְּחַדְּשֵׁם / שׁוּב אֱלֵי הֲמוֹנָךְ וְקַדְּשֵׁם / אוֹיְבַי בְּאֵיבָה תְגָרְשֵׁם


לַהֲגִי וְנִיבִי / הַאֲזִין וְרִיבִי / מְרוֹמְמִי

רִיב וְגַם יְקָרִי / שָׂא בְיַד הֲדָרִי / וְנָעֳמִי


הַעֲלֵה מְיַחְלֵי פְדוּתְךָ / לַחֲצַר דְּבִירָךְ וּבֵיתְךָ / שָׁם לְךָ תְרַנֵּן עֲדָתְךָ


יַשְּׁרָה נְתִיבִי / יָהּ וּמַחֲרִיבִי / וְזוֹעֲמִי

אֹר וְאָז בְּהָרִי / אוֹדְךָ בְּשִׁירִי / וְנָאֳמִי.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
