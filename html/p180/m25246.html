שְׁכוּלָה בְּתוּלָה

שְׁכוּלָה בְּתוּלָה לְעִירֵךְ שׁוּבִי

לְאֶרֶץ קְרוּאָה נֵחֲלַת הַצֶּבִי

מְרוֹמְמִי אָנָה אָשׁוּב וְעָצְבִּי

הֵן גְָּדַל וְחֶרְפָּה שָׁבְרָה לִבִּי

וַהֲגִיגִי כָאֵשׁ יִבְעַר בְּקִרְבִּי

וְעֵינַי לְךָ אוֹמֵר לַצּוּלָה חֳרָבִי

אָדוֹן לְבַדְּךָ בְּמִרְעֶה טּוֹב רְעֵנוּ / יְיָ מַלְכֵּנוּ הוּא יוֹשִׁיעֵנוּ


לְמָתַי תְּרוֹמֵם שִׂמְחַת חָסוּר

אוֹיְבָיו עוֹנִים לוֹ כְּטָמֵא סוּר

מִיָּד אֶל יָד נִלְקָח וּמָסוּר

כְּעֶבֶד עָלָיו יִכְרוּן סַרְסוּר

טָהוֹר עֲנֵהוּ וְחֶבְלוֹ יָסוּר

מֵעַל שִׁכְמוֹ אֲשֶׁר בּוֹ אָסוּר / וּמִטּוּב יָדְךָ אַל תִּמְנָעֵנוּ


מִקּוֹל מְחָרֵף נֶאֱנַחְתִּי אָנוֹחַ

נִקְצָץ כְּיוֹנָה לֹא מָצְאָה מָנוֹחַ

אַרְבָּעָה הֵם שְׁמוֹנָה לִגְנֹחַ

צוּרִי זְכָר לִי צִדְקַת אֲבִי זָנוֹחַ

וְהָסֵר הָרָעָה וְדִמְעָה תִקְנֹחַ / וְאָז תַּרְבִּיץ בְּאַבְנֵי יִקְרָה יְצוּעֵנוּ


הֲתַסְגִּיר עֶבֶד בֵּן בְּכוֹר נְקַבְתּוֹ

וְלֹא תַסְגִּיר עֶבֶד אֶל אִישׁ [הִ]זְהַרְתּוֹ

וּמִבֵּין אֻמּוֹת שִׁבְעִים יְדַעְתּוֹ

וְאָז הִבְטַחְתּוֹ מִבְּלִי לְתִתּוֹ

וְאַתְּ בִּשַּׂרְתּוֹ יוֹם יָשׁוּב עֶזְרָתוֹ / קַבֵּץ תּוֹעֵינוּ וְהָשֵׁב רוֹעֵנוּ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
