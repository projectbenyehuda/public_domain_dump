אַתָּה אֵל קֶדֶם

אַתָּה אֵל קֶדֶם / הַנּוֹצֵר אָחוֹר וָקֶדֶם

עֲדָתְךָ קָנִיתָ קֶּדֶם / אֱלֹהִים מַלְכִּי מִקֶּדֶם


אַתָּה בִּימִינְךָ טִפַּחְתָּ שְׁמֵי עֶרֶץ / אַף יָדְךָ יָסְדָה אֶרֶץ

לְךָ שָׁמַיִם אַף לְךָ אֶרֶץ / בּוֹרֵא קְצוֹת הָאָרֶץ


אַתָּה גִּבַּהְתָּ מַיִם בָּעֲלִיָּה / וּמֵהֶם כִּנַּסְתָּה גִנַּזְתָּ בַתַּחְתִּיָּה

וּבִדְבָרְךָ הַכֹּל נִהְיָה / גְּדוֹל הָעֵצָה וְרַב הָעֲלִילִיָּה


אַתָּה דוֹרֵשׁ גְּלוּיִים וּסְתוּרִים / וְנִקְרֵאתָה מַשְׁפִּיל וּמֵרִים

בּוֹרֵא רוּחַ וְיוֹצֵר הָרִים / דּוֹבֵר צֶדֶק מַגִּיד מֵישָׁרִים


אַתָּה הַשָּׂם לָאָדָם עַיִן / [נֶעְלָם מִ]כָּל עַיִן

וּמְשׁוֹטֵט בְּכָל עַיִן / הַנּוֹתֵן רוֹזְנִים לְאָיִן


אַתָּה וְנוֹפֵחַ בָּאָדָם נִשְׁמַת רוּחוֹ / מֵכִין מִצְעֲדֵי אָרְחוֹ

[מִ]שְׁפָּט עוֹרֵךְ לְנִכְחוֹ / וּמַגִּיד לָאָדָם מַה שֵּׂחוֹ


אַתָּה זוֹכֵר הַבְּרִית וְהַחֶסֶד לְאֶלֶף דּוֹר / וּמֶמְשַׁלְתְּךָ בְּכָל דּוֹר וָדוֹר

כֵּן פַּצְתָּה לַאֲבִי גְדוֹר / זֶה שְׁמִי לְעוֹלָם וְזֶה זִכְרִי לְדוֹר דּוֹר


אַתָּה חַי וּמַחֲלִיף לְקוֹיֶיךָ כֹחַ / [וּ]מְשָׁרְתֶּיךָ גִּבּוֹרֵי כֹחַ

נוֹתֵן לַיָּעֵף כֹּחַ / חֲכַם לֵבָב וְאַמִּיץ כֹּחַ


אַתָּה טְהוֹר עֵינַיִם אֱלֹהִים חַיִּים / רַחֲמָן [וּמֶ]לֶךְ חָפֵץ בַּחַיִּים

סוֹדְךָ גִּלִּיתָ לְבָרִים וּנְקִיִּים / טוֹב חַסְדְּךָ מֵחַיִּים


אַתָּה יָחִיד וְשִׁמְךָ מְיֻחָד חוֹקֵר לֵ[ב] / רוֹאֶה כְלָיוֹת וָלֵב

הָרוֹפֵא לִשְׁבוּרֵי לֵב / יוֹדֵעַ תַּעֲלוּמוֹת לֵב


אַתָּה כִסְאֲךָ לַעֲדֵי עַד קַיָּם / [אַתָּה] מוֹשֵׁל בְּגֵאוּת הַיָּם

וַיָּסֶךְ בִּדְלָתַיִם יָם / כּוֹנֵס כַּנֵּד מֵי הַיָּם


אַתָּה לְעוֹלָם שֵׁם כְּבוֹדֶךָ וּמְרוֹמָם / וְעַל כֹּל תִּתְגַּדֵּל וְתִתְרוֹמָם

דּוֹר מַבּוּל מָחִיתָ יְקוּמָם / לוֹכֵד חֲכָמִים בְּעָרְמָם


אַתָּה מְכַסֶּה [הַ]שָּׁמַיִם בְּעָבִים / וּמַמְטִיר נִטְפֵי רְבִיבִים

מְכַלְכֵּל רָעִים וְטוֹבִים / מוֹנֶה מִסְפָּר לַכּוֹכָבִים


אַתָּה נוֹתֵן לַמְּלָכִים תְּשׁוּעָה / וּמַאֲזִין נְאָקָה וְשַׁוְעָה

רוֹכֵב שְׁמֵי שִׁבְעָה / נוֹטֶה שָׁמַיִם כַּיְרִיעָה


אַתָּה שַׂמְ[תָּ בָיָּ]ם דַּרְכּוֹ / וּבַנְּעָלִים נַמְתָּ [לְהַדְרִיכוֹ]

[ ]שְׁלֶיכָה שַׁמָּה בְתוֹכוֹ / סוּפָה וּסְעָרָה דַרְכּוֹ


אַתָּה עַל שָׁמַיִם יָדֶיךָ רוּמָה / נוֹתֵן עֹז וְעָצְמָה

גִּבּוֹר וְעוֹשֶׂה נְקָמָה / עוֹטֶה אוֹר כַּשַּׂלְמָה


אַתָּה פּוֹתֵחַ אֶת יָדֶיךָ / מַשְׂבִּיעַ לְכָל הַבְּרִיּוֹת מֵאוֹצְרוֹתֶיךָ

מְאֹד עָמְקוּ מַחְשְׁבוֹתֶיךָ / פְּלָאוֹת עֵדְווֹתֶיךָ


אַתָּה צוּר וּמַעֲמִיד עוֹלָם / וְדָבָר מִמְּךָ לֹא נֶעְלָם

יוֹדוּךָ עַמִּים כֻּלָּם / צִדְקָתְךָ צֶדֶק לְעוֹלָם


אַתָּה קָנִיתָ קִנְיַן קְדוּמָה לִדְרֹשׁ / נְסוּכָה מֵעוֹלָם מֵרֹאשׁ

לְךָ יְיָ הַמַּמְלָכָה וְהַמִּתְנַשֵּׂא לְכָל לְרֹאשׁ / קוֹרֵא הַדּוֹרוֹת מֵרֹאשׁ


אַתָּה רַחוּם מִשְׁעָן וּמִסְעָד / וַחֲסָדֶיךָ עַד עוֹלְמֵי עַד

הַמַּבִּיט לָאָרֶץ וַתִּרְעָד / רָם וְנִשָּׂא שׁוֹכֵן עַד


אַתָּה שִׁמַּעְתָּ לַאֲסוּרִים פְּקַח קוֹחַ / וּמְשָׁרְתֶיךָ אֵין שִׁכְחָה לִשְׁכֹּחַ

צוֹעֶה בְּרֹב כֹּחַ / שַׁדַּי לֹא מְצָאנוּהוּ שַׂגִּיא כֹחַ


אַתָּה תַקִּיף יָדְךָ רָמָה / יָהֵב דֵּעָ[ה] וְחָכְמָה

[ מָה / תּוֹלֶה אֶרֶץ עַל בְּלִימָה].


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
