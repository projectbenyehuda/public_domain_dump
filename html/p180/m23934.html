יְיָ אֵל אַדִּיר

אֲדֹנָי אֵל אַדִּיר עוֹטֶה אוֹר כָּאֲדֶּרֶת

מָעוֹן בְּאֶחָד וְאֵין שֵׁנִי לַמַּחְבֶּרֶת

אַתָּה גָדוֹל מֵאָז בְּאֵין תְּבוּנָה נֶחְקֶרֶת

הָיִיתָ דָּגוּל עַד לֹא פְעֻלָּה נוֹצֶרֶת

לָּנוּ הֲיִכּוֹן לְסַפֵּר גְּבוּרַת זְרוֹעַ גְּבֶרֶת

בְּדוֹר וָדוֹר אֱמוּנָתְךָ מֵאֵין נִגְמֶרֶת

בְּטֶרֶם זְרוּתִים נִמְתָּחוּ יְמִינְךָ נֶאְדֶּרֶת

הָרִים חִסַּנְתָּ וַתְּכוֹנֵן שְׁתִיָּה נִגְדֶּרֶת

יֻלָּדוּ טְפוּחֵי סְעָרָה וְסוּפָה נֶחֱרֶדֶת

וַתְּחוֹלֵל יְמִינְךָ שְׁעָלִים בְּמִדָּה נִגְזֶרֶת

אְֶרֶץ כֵּהָה הוֹפַעְתָּ וַתַּתְוִיךְ גֻּלָּה נִבְצֶרֶת

וְתֵבֵל לִכַּדְתָּ מִשְׁבְּרֵי צוּלוֹת בְּמוֹסֶרֶת

וּמֵעוֹלָם מִלֵּאתָ גֵיא דִשְׁנֵי צִיץ וְצַמֶּרֶת

עַד נְטוּיֵי זְבוּל עִטַּרְתָּ זְקוּפִים בְּכוֹתֶרֶת

עוֹלָם סָמַכְתָּ זִיז וְצִפּוֹר מֵרְקַק צַנְתֶּרֶת

אַתָּה עָרַכְתָּ שֻׁלְחָן וַתַּקְדֵּם כֵּרָה מְסֻדֶּרֶת

אֵל פֵּאַרְתָּ צָלוּם בְּצוּרָה מְהֻדֶּרֶת

צִוִּיתָ חֹק וַיְהִי כֵן בְּמִלָּה נֶאֱמֶרֶת

קְצֵה גְדֻלָּתְךָ נִפְלָאָה לְמַזְכֶּרֶת

רַבּוּ פְלָאֶיךָ מִסַּפֵּר בְּמִלָּה מְבֹאֶרֶת

שַׁרְתִּי אַחַת מִנִּי אֶלֶף וְיָדִי נִקְצֶרֶת

תָּקַפְתָּ מְאֹד תּוֹכֵן שְׁחָקִים בַּזֶּרֶת

לְךָ יְיָ הַגְּדֻלָּה וְהַגְּבוּרָה וְהַתִִּפְאֶרֶת


וְהַתִּפְאֶרֶת אֲפֻדָּתְךָ מֵאָז פּוֹדֶה וְגוֹאֵל

נֵצַח תְּקֻדַּשׁ יוֹשֵׁב תְּהִלּוֹת יִשְׂרָאֵל

יְקָרְךָ יְצַלְצְלוּ רִבְבוֹת מִיכָאֵל

שִׁירְךָ יְסַלְסְלוּ אַלְפֵי גַבְרִיאֵל

לְךָ עֹז יִתְּנוּ צִבְאוֹת רְפָאֵל

מַהֲלָלָךְ יְתַנּוּ מַחֲנוֹת נוּרִיאֵל

הֲמֻלּוֹת הֲמֻלּוֹת מְזַמְּרִים לָאֵל

חוֹקְרִים שֵׁם יָהּ נִקְרָא אַכְתְּרִיאֵל

זְקוּפִים כְּאֶחָד לָז לְלָז שׁוֹאֵל

הַלְלוּיָהּ הַלְלוּ אֵל


אֵל נִקְדָּשׁ לְמַעְלָה בְּאֵלֵי אוּרִים / וּלְמַטָּה בִּבְנֵי בְּכִירִים

לְמַעְלָה בְּגַעַשׁ גַּלְגַּלִים / וּלְמַטָּה בִּבְנֵי בְחִירִים

לְמַעְלָה בְּהִלּוֹת הַבְּזָקִים / וּלְמַטָּה בְּוַעַד וְתִיקִים

לְמַעְלָה בְּזִמְרַת זִקִּים / וּלְמַטָּה בְּחֹסֶן חֲשׁוּקִים

לְמַעְלָה בְּטֹהַר טָסִים / וּלְמַטָּה בְּיִקְרֵי יְחָסִים

לְמַעְלָה בְּכֶתֶר כְּרוּבִים / וּלְמַטָּה בְּלֶקַח לְבוּבִים

לְמַעְלָה בְּמַעֲנֵה מְלָכִים / וּלְמַטָּה בְּנֵצַח נְסִיכִים

לְמַעְלָה בְּשִׂיחַ שְׂרָפִים / וּלְמַטָּה בְּעֶתֶר עֲטוּפִים

לְמַעְלָה בְּפֶצַח פְּרוּפִים / וּלְמַטָּה בְּצֶרַח צְרוּפִים

לְמַעְלָה בִּקְהַל קְדוֹשִׁים / וּלְמַטָּה בְּרִנַּת רָאשִׁים

לְמַעְלָה בְשִׁירַת שְׁקוּפִים / וּלְמַטָּה בְּתֹכֶן תְּקוּפִים


תְּקוּפִים בְּשֶׁנֶן בְּשֶׁקֶד בְּשֶׁבַח שִׁירוֹת יַשְׁבִּיצוּ

בְּלַהַג בְּלֶמֶד בְּלֶקַח לְבָבוֹת יָלִיצוּ

בְּמַעַן בְּמֶלֶץ בְּמֶרֶץ מִלּוֹת יַמְרִיצוּ

בְּהֶגֶה בְּהוֹד בְּהָדָר הָגוּת יְהַרְבִּיצוּ

בְּצַהַל בְּצֶוַח בְּצֶרַח צְהוּרִים יָצִיצוּ

בְּעֶתֶר בְּעֶרֶץ בְּעֶלֶץ עֲנָוִים יַעֲלִיצוּ

בְּיֹשֶׁר בְּיֹתֶר בְּיֹקֶר יְשֵׁנִים יָקִיצוּ

בְּרַחַשׁ בְּרַעַשׁ בְּרֶגֶשׁ רַגְלָם יָרִיצוּ

וְהִקְדִּישׁוּ אֶת קְדוֹשׁ יַעֲקֹב וְאֶת אֱלֹהֵי יִשְׂרָאֵל יַעֲרִיצוּ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
