אֶבֶן בֹּחַן

אֶבֶן בֹּחַן פִּנָּה מְיֻקֶּרֶת

אֵיתָן מַשְׂכִּיל בְּחָכְמָה נֶחְקֶרֶת

בְּיָדוֹ עָלְתָה וַיַּעֲמֹד לְפִי קֶרֶת / וַיִּקְרָא בְשֵׁם יְיָ


בִּמְשׂוּכַת חֵדֶק פָּרַח כַּשּׁוֹשָׁן

גַּחַל הוּבָא עַל סֵמֶל כּוּשָׁן

גֵּאֶה חִלַּצְתּוֹ מֵאוּר הַכִּבְשָׁן / כִּי בָאֵשׁ יְיָ


דָּרַשׁ לְהוֹרוֹת מְתֵי סָרָה וְכַחַשׁ

דַּעַת עֶלְיוֹן לָהַג וְלָחַשׁ

הֲמוֹן חַנְפֵי לֵב קָרָא וְרָחַשׁ / דְּעוּ כִּי יְיָ


הַכֹּל יָצַר וְהַכֹּל הוּא סוֹבֵל

וְעָלָיו תְּלוּיִים שַׁחַק וְתֵבֵל

וּמַדּוּעַ תֵּלְכוּ אַחֲרֵי הַהֶבֶל / אַךְ יְראוּ אֶת יְיָ


זָנַח מוֹשָׁב וְנָטַשׁ מוֹלֶדֶת

זְבַדְתּוֹ בְגֵרוּת כַּנָּה מְיֻסֶּדֶת

חַשְׁתָּ וְרָחַשְׁתָּ מִלָּה נוֹעֶדֶת / זֹאת נַחֲלַת עַבְדֵי יְיָ


חֶמְדַּת נְחָלוֹת לְפָנָיו עָרַכְתָּ

טַפְסְרֵי מַמְלָכוֹת לְרַגְלָיו הִדְרַכְתָּ

טִכַּסְתָּ זַרְעוֹ וּבַכֹּל בֵּרַכְתָּ / הִנֵּה כִּי כֵן יְבֹרַךְ גָּבֶר יְרֵא יְיָ


יָאַב וְתָאַב עַל יְבֹשֶׁת מָקוֹר

יָזַם אוֹת לִדְרֹשׁ וְלַחְקֹר

כָּרֵַת לוֹ בְרִית בַּל יִשְׁקֹר / וְהֶאֱמִין בַּיָי


כַּאֲשֶׁר לֹא תוּכַל סְפֹר כּוֹכְבֵי עֶרֶץ

לֹא יִסָּפֵר זַרְעֲךָ בָּאָרֶץ

לָהַג בְּשָׁמְעוֹ וַיַּעַן בְּמֶרֶץ / אָמֵן כֵּן יַעֲשֶׂה יְיָ


מָרוֹם מַגִּיד מֵרֵאשִׁית אַחֲרִית

מָשָׁל הִמְרִיץ לִהְיוֹת לִבְרִית

נְצִיבֵי שָׁלִישִׁים בִּתַּר וְהִכְרִית / וַיֵּרָא כְבוֹד יְיָ


נִיבוֹ נָתַן עֵץ לֹא פוֹרֶה

סִנְסֵן בִּכְלוֹת הַמַּלְקוֹשׁ וְהַיּוֹרֶה

שָׂרִיג אֶזְרָח כְּאֶזְרָח מִתְעָרֶה / כִּי פָקַד יְיָ


עָמַס בְּתֻמּוֹ עֶשֶׂר מַסּוֹת

עָמַד בְּכֻלָּם בְּשִׁמְךָ לַחֲסוֹת

פֵּרַשׁ יְחִידוֹ עוֹלָה לַעֲשׂוֹת / רֵיחַ נִיחוֹחַ לַיְיָ


פְּלִיטֵי גִזְעוֹ מִשַּׁחַת חַלֵּץ

צִדְקוֹ בַעֲדָם יֹשֶׁר יָלֵץ

צַמֵּת גֵּאוּת עָרִיץ וָלֵץ / לִקְרֹא שְׁנַת רָצוֹן לַיְיָ


קוֹוֶיךָ תְּנַקֶּה מֵעָווֹן הַמּוּלָז

קוֹרְאִים בְּשִׁמְךָ לְבָבָם עָלָז

רוֹחֲשִׁים יַחְדָּו הוֹגִים לָז לְלָז / לְכוּ וְנָשׁוּבָה אֶל יְיָ


רַחֵם תִּזְכֹּר בְּעִתּוֹתֵי שֶׁצֶף

שֶׁבְיְךָ תְפַלֵּט מִמְדוּרַת רֶצֶף

שָׁפְטֵם בְּצֶדֶק וְלֹא בְקֶצֶף / כִּי צַדִּיק יְיָ


תְּשַׁלַּח אֲסִירֶיךָ מִבּוֹא הָעֶלֶט

תְּסוֹבֵב קוֹרְאֶיךָ בְּרָנֵּי פַלֵּט

שְׁמַע תְּחִנָּתָם וְאָוֶן אַל תַּשְׁלֵט / כִּי שׁוֹמֵעַ אֶל אֶבְיוֹנִים יְיָ


לְהַלְבִּין כַּשֶּׁלֶג אֲדֻמִּים כַּשָּׁנִים

מַעֲשֶׂיךָ לְחַיּוֹת בְּקֶרֶב שָׁנִים

הַעֲבֵר פְּשָׁעִים חֲדָשִׁים גַּם יְשָׁנִים / יַחְדָּו יָסוּפוּ נְאוּם יְיָ


קָדוֹשׁ בְּךָ שַׂמְתִּי שִׂבְרִי וּמִשְׁעָנִי

טָהוֹר אַל תִּבְזֶה עֱנוּת עָנִי

נָשָׂאתִי אֵלֶיךָ לְבָבִי וְעֵינִי / נְסָה עָלֵינוּ אוֹר פָּנֶיךָ יְיָ


בְּקָרְאָם וְיְדַבֵּר יְיָ אֶל מֹשֶׁה אַחֲרֵי מוֹת שְׁנֵי בְנֵי אַהֲרֹן בְּקִרְבָתָם לִפְנֵי יְיָ


שָׁמַע לַחַשׁ מָעֹז הַשַּׁלִּיט

בְּצֵל רַחֲמָיו יַסְתִּיר הֲמוֹנָיו וְיָלִיט

דְּהוּרֵי הֲמוֹנֵי גְדוּדָיו בְּיֵשַׁע רַב לְהַפְלִיט

גָּנוֹן וְהִצִּיל פָּסוֹחַ וְהִמְלִיט.
