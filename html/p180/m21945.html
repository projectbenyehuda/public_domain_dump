יְדִידַי שְׁעוּ

יְדִידַי, שְׁעוּ מִנִּי, אֲמָרֵר בְּבִכְיָתִי / וְאוּלַי שְׁבִיב לִבִּי אֲכַבֶּה בְדִמְעָתִי

בְּיַעַן יְקוֹד לִבִּי כְתָפְתֶּה, וְעַצְמוֹתַי / כְּמוֹ קַשׁ, וְהִנֵּה אֵשׁ וְעֵצִים מְדוּרָתִי

וּבִנְפֹל עֲלֵי עַיִן תְּנוּמָה וְתַרְדֵּמָה, / תְּעוֹפֵף שְׁנַת עֵינַי וְתִדַּד תְּנוּמָתִי

אֲשׁוֹטֵט וְאֶרְאֶה כוֹכְבֵי רוֹם וּמַזָּרָיו, / כְּאִלּוּ חֲקֹר סוֹדָם מְלֶאכֶת עֲבוֹדָתִי

וְיוֹנָה עֲלֵי אָמִיר תְּקוֹנֵן עֲלֵי פֵרוּד / אֲחוֹתָהּ אֲשֶׁר נָדְדָה וְתֶהְמֶה כְּהֶמְיָתִי

וְכָל עֵת אֲשֶׁר אֶדֹּם וְאוֹחִיל מְצֹא מַרְפֵּא / תְּעוֹרֵר יְגוֹן לִבִּי וְתַרְבֶּה מְהוּמָתִי

וְאָמוּת בְּמַכְאוֹבִי – וְנֶגְדִּי צְרִי גִלְעָד, / וְאֶצְמָא אֱלֵי מַיִם – וְגִיחוֹן לְעֻמָּתִי

בְּנַפְשִׁי, אֲשֶׁר אֹהַב חֲיוֹתוֹ וְשַׁלְוָתוֹ - / וְיִשְׂמַח לְיוֹם אֵידִי וְיֶאְהַב מְחִתָּתִי

אֲהַבְתִּיו בְּעוֹדִי בֶּאֱמֶת וּבְתָם לֵבָב - / וְהָיָה גְמוּלִי שָׁוְא, וְשֶׁקֶר תְּמוּרָתִי

וְלָאֵל מְיֻדָּעַי חֲשׁוּקַי אֲשֶׁר אָרְחוּ, / וְרָחַק, בְּעֵת רָחְקָם, מְשׂוֹשִי וְשִׂמְחָתִי

רְאִיתִים וְהֵם עַל הַגְּמַלִּים - וְסָר לִבִּי, / וְכָשַׁל מְאֹד כֹּחִי, וְכָלְתָה אֱיָלוּתִי

וְאֵיכָה אֱחִי אַחַר פְּנוֹתָם, וְאֶשָּׁאֵר / בְּנָסְעָם - וְהֵם רוּחִי וְנַפְשִׁי וְנִשְׁמָתִי

הֲיִחְיֶה אֱנוֹשׁ מֵאֵין נְשָׁמָה וְיִשָּׁאֵר - / הֲכִי נִשְׁאֲרָה מֵאֵין נְשָׁמָה גְּוִיָּתִי,

אֱלֹהִים יְהִי עִמָּם בְּנָסְעָם וּבַחְנוֹתָם, / הֱיוֹתָם לְמוּל עֵינַי בְּנָסְעִי וּבַחְנוֹתִי

וְיַשְׁקֵם בְּכָל מַחֲנֶה בְּמָטָר כְּמֵי עֵינַי, / וְיַמְטֵר עֲלֵיהֶם טַל וּמָטָר כְּדִמְעָתִי

וְיָמִית גְּמַלֵּיהֶם אֲשֶׁר נָשְׂאוּ אוֹתָם / בְּיַעַן שְׁלוֹמִי נָשְׂאוּ אָז וְשַׁלְוָתִי

וְאוּלָם אֲקַוֶּה שׁוּב לְבָבִי אֱלֵי גֵוִי / וְאוֹחִיל מְצֹא מַרְפֵּא וְשִׁקּוּי לְמַחְלָתִי

בְּשׁוּבִי אֱלֵי דוֹדַי וְשָׁמְרִי בְרִיתוֹתָם / וְשׁוּרִי פְנֵיהֶם הַמְּאִירוֹת אֲפֵלָתִי:

אֲחִיָּה, אֲשֶׁר נַפְשִׁי יְחַיֶּה בְשֶׂכֶל פִּיו, / וְאִם יַאֲרִיךְ חָלְיִי - בְּנִיבוֹ אֲרוּכָתִי

וְיִצְחָק, אֲשֶׁר תִּצְחַק אֲדָמָה בְלֶכְתּוֹ בָהּ, / וְתָעֹז וְתִתְגָּאֶה בְּעֻזּוֹ, כְּגַאְוָתִי

וְחַיּוּן, אֲשֶׁר חַיַּי תְּלוּיִים בְּשַׁלְוָתוֹ, / וְתָבוֹא, בְּעֵת יַחְלֹף שְׁלוֹמוֹ, חֲלִיפָתִי

שְׁלָשְׁתָּם אֱיָלוּתִי, וְחֵילִי עֲלֵי קָמַי, / וְחַרְבִּי, וְתוֹתָחִי, וְרָמְחִי, וְצִנָּתִי

חֲקַקְתִּים עֲלֵי לוּחַ לְבָבִי, וְיוֹתֶרֶת / כְּבֵדִי, וְאִישׁוֹן עֵין יְמִינִי, וּבָבָתִי.

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
