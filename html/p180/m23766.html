שְׁטָר עָלַי בְּעֵדִים

שְׁטָר עָלַי בְּעֵדִים וְקִנְיָן / לְשַׁלֵּם מָחֳרָת הַמִּנְיָן / וְהָיָה יְיָ לְדַיָּן

- - - - - - - - -

אֲזַי בְּחָשְׁבִי לְמֵרָחוֹק / בָּכִיתִי לִבְלִי חֹק / עֵת לִבְכּוֹת וְעֵת לִשְׂחֹק

נָפְלָה עָלַי חֲרָדָה / בְּזָכְרִי יוֹם הַקְּפָדָה / פַּחַד קְרָאָנִי וּרְעָדָה

יַעַן שְׁרִירוּת לְבָבִי / דָאַגְתִּי בְמַחְשָׁבִי / כִּי אֵיךְ אֶעֱלֶה אֶל אָבִי

שֹׁד לְמִי רַב זְדוֹנֵהוּ / יוֹם יְבֻקַּשׁ עֲווֹנֵהוּ / וְחִשַּׁב עִם קוֹנֵהוּ

לְיוֹם לֹא יִשָּׂא פְנֵי שָׂר / בְּהִתְנַגֵּשׁ לַמּוּסָר / הַנֶּפֶשׁ עִם הַבָּשָׂר

מִשְׁפְּטֵיהֶם וְדִינֵיהֶם / עַד הָאֵל אֲדוֹנֵיהֶם / יָבוֹא דְבַר שְׁנֵיהֶם

הַנֶּפֶשׁ תְּמַהֵר תְּשׁוּבָה / אֵין עָלַי חַטָּאת כְּתוּבָה / צִדְּקָה נַפְשָׁהּ מְשׁוּבָה

הַרְפֵּנִי וְעָזְבֵנִי / הֲלָרִיב תַּצִּיבֵנִי / הוֹדִיעֵנִי עַל מַה תְּרִיבֵנִי

קְדוֹשִׁי לֹא אֲהֵבָנִי / כִּי בְמַסְגֵּר הֱבִיאָנִי / בְּמַחֲשַׁכִּים הוֹשִׁיבַנִי

טֶנֶף גּוּף טוֹרַחַת / וּמִנְּבֵלָה סוֹרַחַת / אָנֹכִי בּוֹרַחַת

נִלְאֵיתִי מְאֹד כַּלְכֵּל / כִּי הַגּוּף הַנּוֹכֵל / הִשִּׂיאַנִי וָאֹכֵל

בְּכָל חֵטְא יִתְגַּלֵּעַ / וְאוֹתִי בְכַף קוֹלֵעַ / לֹא הֵשִׁיב יָדוֹ מִבַּלֵּעַ

רְעָצַנִי הֱנִיאָנִי / וְלַשַּׁחַת קְרָאָנִי / אַךְ עַתָּה הֶלְאָנִי

בְּכָל תַּאֲוָה הוּא נֶאֱסָר / וּמַאֲכָל וּמַשְׁקֶה לֹא סָר / וּבֶטֶן רְשָׁעִים תֶּחְסָר

יָגוּף לְטַמֵּא עַצְמוֹ / יַעַן כִּי הוּא כִּשְׁמוֹ / נָבָל שְׁמוֹ וּנְבָלָה עִמּוֹ

יְצוּדֵנִי פוֹתֵעַ / וְהוּא בְּתִתּוֹ מַלְתֵּעַ / בְּעֵינָיו כִּמְתַעְתֵּעַ

הֵן חֶטְאוֹ לֹא יְסֻפָּר / לָכֵן יָכְלַם וְְיָכְפָּר / כְּדֵי רִשְׁעָתוֹ בְּמִסְפָּר

וּמִמֶּנִּי אַל תִּפְרַע / וּמוּסַר גֵּו אַל תִּגְרַע / כִּי כֻלּוֹ חָנֵף וּמֵרָע

דִּבֶּר גּוּף לְהִוָּשַׁע / חֲדַל מִמֶּנִּי וְהָשַׁע / זַךְ אֲנִי בְּלִי פֶשַׁע

הֵן כְּאֶבֶן מַעֲמָסוֹת / מָלֵא בוּז וְקַלָּסוֹת / וּמַה יָּכֹלְתִּי עֲשׂוֹת

גַּם אֲנִי נִשְׁכַּל שָׁכוֹל / דּוֹמֶה לְעֵץ הָאֶשְׁכֹּל / אֲשֶׁר לֹא יִצְלַח לַכֹּל

בְּלִי נֶפֶשׁ אֵחָשֵׁב / כְּבַיִת מִבְּלִי יוֹשֵׁב / וְאֵין עוֹנֶה וְאֵין קָשֵׁב

יַעַן אַשְׁכִּים וְאַאֲמִישׁ / מֻשְׁלָךְ כְּמוֹ חַלָּמִישׁ / מִמְּקוֹמוֹ לֹא יָמִישׁ

רֹאשׁ הַפְּקֻדָּה כֻלָּהּ / עַל נֶפֶשׁ וּמַעֲלָלָהּ / שַׁלְּמוּ לָהּ כְּפָעֳלָהּ

וּשְׂאִי נַפְשִׁי מִרְיֵךְ / וְאִכְלִי נָא אֶת פִּרְיֵךְ / וְשַׁלְּמִי אֶת נִשְׁיֵךְ

לְכָל רַע לֹא צָמֵאתִי / אֲבָל אֲנִי בָךְ נִדְמֵיתִי / אֵיךְ תֹּאמְרִי לֹא נִטְמֵאתִי

מַחְשְׁבוֹתַיִךְ הֵבִינוּ / וְיָדַיִךְ הֵכִינוּ / וְנַחְנוּ מַה כִּי תַלִּינוּ

אֲמֵלִים אֲשֶׁר רָגָשׁוּ / וְצֶדֶק לֹא פָגָשׁוּ / אֶחָד בְּאֶחָד יִגָּשׁוּ

לֵב דּוֹרֵשׁ וּמַעְבָּדָיו / יַחְבִּיר שְׁנֵיהֶם עָדָיו / וּבָעֲרוּ שְׁנֵיהֶם יַחְדָּו

קִלְקוּל עֲווֹנָם יוּסָר / וְלֹא יִכְבַּד הַמּוּסָר / בְּשַׁגַּם הוּא בָשָׂר

יָהִיר שְׁרִירוּת יַחְמֹד / אִם כְּמִפְעָלוֹ תָמֹד / אֲדֹנָי מִי יַעֲמֹד

חֲשֹׂךְ זַעַם אֵימָתְךָ / וְאַל תְּיַסֵּר בַּחֲמָתְךָ / וְיִנָּפֵשׁ בֶּן אֲמָתְךָ

קְרַב נָא לְהַרְוִיחֵנִי / וְיוֹם לְרִיב תִּקָּחֵנִי / אַל בְּאַפְּךָ תוֹכִיחֵנִי.

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
