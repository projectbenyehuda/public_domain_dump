לְךָ שַׁדַּי

לְךָ שַׁדַּי / וְאָמַר דַּי / לְגֻלַּת רוֹם וְחֵילֶיהָ

לְךָ מָרוֹם / יָם וְדָרוֹם / וְכָל שֵׁשֶׁת גְּבוּלֶיהָ

הֲלֹא תָעִיד / וְגַם תַּגִּיד / גְּדֻלָּתְךָ בְּמִלֶּיהָ

הַמֵּאִיר / לָאָרֶץ / וְלַדָּרִים עָלֶיהָ


לְךָ עֶלְיוֹן / מְקוֹם חֶבְיוֹן / הֲדָרֶךָ וְהוֹדֶךָ

לְךָ הַתָּו / אֲשֶׁר תַּחְתָּיו / גַּלְגַּלֵּי יְסוֹדֶךָ

וְכָל גַּלְגַּל / וְכָל מַעְגָּל / יְהוֹדֶךָ יְיַחְדֶּךָ

הֲכִי הַכֹּל / כְּמוֹ אֶשְׁכֹּל / תְּלוּיִים הֵם בְּיָדֶךָ

וּמִי לֹא נָע / וְלֹא נִכְנָע / בְּזָכְרוֹ שֵׁם כְּבוֹדֶךָ

וּמִ יָכִיל / וּמִי יַשְׂכִּיל / וּמִי יַעֲמֹד בְּסוֹדֶךָ

גְּדוּלָתָךְ / פְּעֻלָּתָךְ / לְבַב אִישׁ לֹא יְכִילֶהָ


לְךָ כָּל יְסוֹד / וְתֵדַע סוֹד / הַגַּלְגַּל הָעֶלְיוֹן

אֲשֶׁר אָרַח / מִמִּזְרָח / וְיִתְגַּלְגֵּל כְּגִלָּיוֹן

וְסוֹד יַמְתִּיק / בְּעֵת יַעְתִּיק / גְּלִיל שֶׁמֶשׁ בְּלֹא רִפְיוֹן

עֲשׂוֹת רֶמֶשׁ / וְצֵאת שֶׁמֶשׁ / כְּמוֹ חָתָן בְּאַפִּרְיוֹן

וְהוּא הָעֵד / בְּכָל מוֹעֵד / בְּאֵין אֹמֶר וְלֹא הֶגְיוֹן

וְרֹאשׁ דְּבָרוֹ / כִּי יְצָרוֹ / אֵל אֶחָד בְּלִי דִמְיוֹן

עֲבֹר צִיָּה / וְתַחְתִּיָּה / לְהָאִיר מַאֲפֵלֶיהָ


לְךָ דַּיָּן / כָּל עִנְיָן / וְכָל קִנְיַן פְּלוּסֶיךָ

לְךָ אַדִּיר / לְךָ יַאְדִּיר / הֲמוֹן קַצְוֵי אֲפָסֶיךָ

מִי יַעְצֹר / וּמִי יִבְצֹר / פְּלָאֶיךָ וְנִסֶּיךָ

מִי חָכָם / וּמִי יֶחְכַּם / וּמִי יַעְשֶׂה כְמַעְשֶׂיךָ

וְאַתְּ נָאוֹר / וְתַעְטֶה אוֹר / אֶרֶץ לָךְ וְחֵילֶיהָ

לְךָ וֵילוֹן / וְהַחַלּוֹן / אֲשֶׁר לִפְאַת צְפוֹנֶךָ

לְךָ שִׁנְאָן / וְרִבּוֹ רִבְבָן / וְהֵם כְּאַיִן לְפָנֶיךָ

לָז אֶל לָז / בְּקוֹל מַעְלָז / הֲלֹא שׁוֹאֲלִים לִמְעוֹנֶךָ

וְיָעִידוּן / וְיַגִּידוּן / גְּבוּרָתְךָ וְחִנֶּךָ

וְהַנַּעְלָה / וְהַמְּעֻלָּה / תַּרְבֶּה מַהֲלָלֶיהָ


לְךָ קוֹשֵׁר / חִין יֹשֶׁר / סַנְדַּלְפוֹן לְכוֹתֶרֶת

לְךָ עָנָף / וְעֵיט כָּנָף / מְהֻדֶּרֶת וּמְעֻטֶּרֶת

לְךָ יָקָר / בְּלִי נֶחְקָר / הוֹד וְהָדָר וְתִפְאֶרֶת

וְנוֹרָאוֹת / וְנִפְלָאוֹת / עֲלִילוֹת לְמַזְכֶּרֶת

שִׁית רוֹם דּוֹר / לְדוֹר וָדוֹר / לְגִבְעָה וְכָל חֲיָלֶיהָ.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
