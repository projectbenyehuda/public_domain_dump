יחידה החטאה

תוכחה


יְחִידָה הַחַטָּאָה / בְּיוֹם דִּין הִנֵּה בָאָה

לְהִתְוַדּוֹת עַל מַעֲלָהּ – / וַאדֹנָי יִסְלַח לָהּ!


צֵידָה תֹם וַעֲתֶרֶת / קְחִי לָךְ, מִצְטַעֶרֶת,

וְחֶסֶד הָכִינִי עוֹד, / מֵרֹב הַדֶּרֶךְ מְאֹד.


חַלִּי נָא פְּנֵי יוֹצְרֵךְ / וְשִׁבְרִי אֶת גְּאוֹן יִצְרֵךְ,

בְּכִי עָלָיו וְהִתְעַנִּי, / קוּמִי נָא וְהִשְׁתַּנִּי!


קוּמִי, מַהֲרִי וְנַהֲרִי / אֶל אֲדֹנָי, וְהִטַּהֲרִי,

כִּי עַד כֹּה מִפִּשְׁעִי / לֹא רֻחַצְתְּ לְמִשְׁעִי.


בְּמֵי דִמְעָה הִתְרַחֲצִי / וְגַם לְקוֹנֵךְ הִתְרַצִּי,

וְחִיל תְּהִי סוֹבֶלֶת / כְּאִשָּׁה מִתְאַבֶּלֶת.


נֶחֱלֵית בִּשְׁחִין חָלְיֵךְ – / הִכָּה אוֹתָךְ מַעֲלֵךְ,

דְּבֶלֶת תְּפִלָּה מִרְחִי / עַל הַשְּׁחִין וְיֶחִי!


יָגַעַתְּ בְּרֹב הֶבֶל / בְּדָרְשֵׁךְ תַּעֲנוּגוֹת תֵּבֵל,

כִּי תִתְיַף בִּלְבוּשָׁהּ / וַתֵּיטֶב אֶת רֹאשָׁהּ.


הַקְרִיבִי מְקוֹם מִנְחָה – / לְבָבֵךְ, וְשִׁפְכִי שִׂיחָה,

וְהָאֵל כִּתְפִלָּתֵךְ / יִתֵּן אֶת שֵׁלָתֵךְ.


וְאִם דִּמְעֵךְ תַּזִּילִי – / בָּזֹאת נַפְשֵׁךְ תַּצִּילִי,

בִּכְנַף עֲצָתִי תֶחֱסִי, / דְּעִי וּרְאִי מַה תַּעֲשִׂי!


דָּרַשְׁתִּי רַעְיוֹנַי / לְהוֹכִיחָם בְּהֶגְיוֹנַי:

הַעְתִּירוּ אֶל מַלְכְּכֶם, / נֹכַח אֲדֹנָי דַּרְכְּכֶם!


הַרְבּוּ בְּבֵיתוֹ שִׁירָה, / וְנָבִיא לְפָנָיו תְּשׁוּרָה

מִטּוּב הֶגְיוֹנֵינוּ, / לְבָרֵךְ אֶת אֲדוֹנֵינוּ.


בּוֹאוּ שְׁעָרָיו בְּתוֹדָה, / תּוֹדָה וְזִמְרָה חֲמוּדָה,

וְתִשְׂמְחוּ בוֹ בְּפָעָלְכֶם, / וְיִשְׂמַח גַּם הוּא בָּכֶם!


נְגִידִים לְפָנָיו אֲמַלֵּל, / וְאִם אֶמָּנַע מֵהַלֵּל –

לָמָּה עַל אַדְמָתִי / הֶחֱיָה אֲדֹנָי אוֹתִי?


נַעֲלֶה עַל כָּל רָמִים / וְאֵין שֵׁנִי בַיְקוּמִים

כְּמוֹתוֹ אֵין בַּנְּשִׁיָּה / וְכָמוֹהוּ לֹא הָיָה.


תִּכֵּן אִישׁ עַל אֲדָמָה / וְנָפַח בְּאַפָּיו נְשָׁמָה –

וְשָׁת בְּכֹחַ הַנְּפָשִׁים / אַרְבָּעָה חָרָשִׁים.


נִבְרֵאתֶם, בְּנֵי אָדָם, / בְּנִפְלְאוֹת אֲשֶׁר קָדַם,

מֵאֶפֶס נִקְרָאוּ / וּמֵאַיִן יָבוֹאוּ.


אָזְנַיִם וְאַפַּיִם / שָׁת בָּכֶם, וְיָדַיִם,

וְעַיִן לִרְאוֹת פָּעֳלוֹ, / וְחֵךְ אֹכֶל יִטְעַם לוֹ.


לֵבָב בַּעַל חָכְמָה, / וְגוּפָה חֵן מְרֻקָּמָה –

וְצַר מַשְׁחִית אַחֲרָיו, / יוֹשֵׁב לוֹ בַּחֲדָרָיו!


הִשִּׁיא לִבּוֹ בְחֶטְאוֹ, / כִּי לַחֲטֹא מַשִּׁיאוֹ,

וְלֹא הֵבִין כְּבַעַר, / כִּי עוֹדֶנּוּ נַעַר.


אֵיךְ נִסְכַּל אֲחִי רִמָּה, / אִישׁ נוֹצַר מֵאֲדָמָה!

יִצְרוֹ בְּתֵבֵל עוֹבֵד – / וְאַחֲרִיתוֹ עֲדֵי אוֹבֵד.


יְאַחֵר עַל הַיַּיִן / וְיֶאֱהַב כָּל יְפַת עַיִן,

וְיַחְשֹׁק, נָעוּר וָרֵיק, / אִשָּׁה בְּנַחַל שׂוֹרֵק!


נִפְתֶּה! אֵיךְ לֹא תִזְכֹּר / עָווֹן, אוֹתְךָ יַעְכֹּר?

וְיוֹם דִּין כָּל מִפְעָלְךָ / לַמּוֹעֵד שָׁמוּר לְךָ!


אֵיךְ תִּשְׂמַח, הַגֶּבֶר? / מָחָר אַתָּה בַקֶּבֶר –

וְתִהְיֶה לִשְׁאוֹל סוֹחֵר, / וְנֶהְפַּכְתָּ לְאִישׁ אַחֵר!


הֲלֹא תִרְאֶה מוֹת גְּבָרִים / קְבָרוּם אֲנָשִׁים אֲחֵרִים,

הֲלֹא תַבִּיט בְּעֵין שִׂכְלָךְ: / הֵמָּה עוֹשִׂים גַּם לָךְ?


וְזֶה עוֹדֶנּוּ לֹא שָׁב, / עַד כַּמֵּתִים הוּא נֶחְשָׁב,

וְיִהְיוּ אוֹתוֹ נוֹשְׂאִים / בִּקְצֵה עֵמֶק רְפָאִים.


זָרְקָה בּוֹ שֵׂיבָתוֹ, / וְזָעוּ שׁוֹמְרֵי בֵיתוֹ,

וְיִקְרַב אֶל הַפְּחָתִים, / וְחַיָּתוֹ לַמְמִיתִים.


וְהִנֵּה בָא עַד סוֹפוֹ, / וַיַּשְׁחֵת אֶת גּוּפוֹ,

וְאֶת רוּחוֹ יַגְבִּיהוּ, / כִּי מַלְאַךְ יְיָ צְבָאוֹת הוּא.


מֵאִתּוֹ נִפְרָדוֹת / אַרְבַּע הַיְסוֹדוֹת,

וּמִמֶּנּוּ נִתָּכִים / חֲמֵשֶׁת הַמְּלָכִים.


לֹא יָבִין וְלֹא יַרְגִּישׁ / וּמִמְּקוֹמוֹ לֹא יָמִישׁ,

וּמִשְּׂפָתָיו הַבְּרוּרִים / אֵין אֹמֶר וְאֵין דְּבָרִים.


טֶרֶם יַדִּיחוּהוּ / בְּמַיִם יְדִיחוּהוּ,

וְגוּפָתוֹ הָאֲמוּלָה / הִיא לוּטָה בַשִּׂמְלָה.


כָּעֵת מָחָר בְּנָסְעוֹ, / יַעֲזֹב לְאַחֵר יְגִיעוֹ

וְיַנִּיחַ כָּל מְאֹדוֹ – / וּמְאוּמָה אֵין בְּיָדוֹ!


הָעֵת שָׁב לַהֶבֶל / יְנִדּוּהוּ מִתֵּבֵל,

וְיוֹדְעָיו, יוֹם שִׁבָּרוֹן, / נוֹשְׂאִים אֶת הָאָרוֹן.


לְבַלְּעוֹ פִּיהוּ פָצָה / אֲבַדּוֹן, וְהָאִישׁ יָצָא

מֵרָעָה אֶל רָעָה, / וְנָסַב הַקַּרְקָעָה.


אַחֲרֵי אוֹתוֹ קָבָרוּ / עָלָיו עָפָר צָבָרוּ,

וְאִישׁ אִישׁ יָרִים קוֹלוֹת: / גֹּלוּ אֲבָנִים גְּדוֹלוֹת!


רֵעִים אֲשֶׁר חָרָדוּ / מֵאִתּוֹ נִפְרָדוּ,

וְהוּא יִשָּׁאֵר נִגְנָז / אֶל הַסֶּלַע הַלָּז.


יֵצֵא יוֹם הַקְּפָדָה / מֵחֲרָדָה לַחֲרָדָה,

וְעָבִים בְּשַׁיִת יָמִיר / וְהוּא יוֹשֵׁב בְּשָׁמִיר.


הַגְּוִיָּה הַיְקָרָה / בְּתֵבַת חֹמֶר סְגוּרָה,

גַּל עָפָר שׁוֹכֶבֶת – / וְהָיָה הִיא רוֹכֶבֶת.


וְיִיקְצוּ הַמְזַעְזְעִים / הֵמָּה הַתּוֹלָעִים,

אָז יַעֲלוּ בְאִישׁוֹנָיו / וִינַקְּרוּ אֶת עֵינָיו.


לוּ יֵשׁ אָדָם דּוֹבֵר / בְּעֵת שָׁכְבוֹ בַקֶּבֶר –

יַעֲנֶה בְּלֵבָב נָמָס: / פָּגַעְתִּי אֶת סָס!


יֹאכַל עָשׁ וְלֹא יַחְמֹל / בְּשָׂרִי, וְהִנֵּה אֶתְמוֹל

קָם בְּגוּפָתִי בָאְשִׁי, / וְעַתָּה יָרוּם רֹאשִׁי.


לְמִשְׁלַח עָשׁ וּלְמִרְמָס, / עוֹרִי רָגַע וַיִּמַּס.

כָּל עוֹרְקַי יָבֵשׁוּ / וַעֲצָמַי עָשֵׁשׁוּ.


עַד שְׁנֵים עָשָׂר חֳדָשִׁים / נִשְׁכַּח מִלֵּב אֲנָשִׁים,

הוּא רָקַב וַיְבַל, / וְשֵׁם הָאִישׁ נָבָל.


אֶתְמוֹל בִּבְכִי זָעֲקוּ / רֵעָיו – וְעַתָּה יִשְׂחֲקוּ,

וַיִּתְּמוּ יְמֵי בְכִיתוֹ – / וְהִנֵּה מִשְׁתֶּה בְּבֵיתוֹ!


כָּכָה יִקְרֶה לְנִבְזֶה – / כָּל יְלוּד אִשָּׁה, וְכָזֶה

יְמוּתוּן כָּל בְּנֵי נְשִׁיָּה, / וְאֵין קֵצֶה לַגְּוִיָּה!


וְרַע עַל כָּל אֵלֶּה – / יוֹם לְדִין הָאֵל עוֹלֶה,

וּמַלְאֲכֵי מִשְׁפָּט יֶחֱזֶה / מִזֶּה וּמִזֶּה.


יִשְׂבַּע מִפְּרִי יָדָיו, / פְּרִי מְגָדָיו עַל חֲסָדָיו,

וּבוֹ לְעֵד יָקוּם כַּחֲשׁוֹ, / וְיָשׁוּב עֲמָלוֹ בְרֹאשׁוֹ.


לְזֹאת אוֹכִיחַ לִבִּי / כָּל יוֹם רוּחִי בְקִרְבִּי,

כִּי יָדַעְתִּי: לֹא / יִחְיֶה אַחֲרֵי נִפְלוֹ.


עַתָּה בִּינוּ, זְמָמַי, / וְעִזְבוּ עֶרְוַת עֲלוּמַי,

שִׁמְעוּ וְהַטּוּ אָזְנֵיכֶם, / וְיִשְׁמַע אֱלֹהִים אֲלֵיכֶם!


וְלָכֵן שְׁמַע, אִישׁ לֵבָב, / וְכָל אִישׁ בְּדַרְכּוֹ שׁוֹבָב,

לְשָׁרֵת פְּנֵי אֵל יַחַד, / וּלְעָבְדוֹ שְׁכֶם אֶחָד:


אִם נָא תִדְרְשׁוּ לְשִׁכְנוֹ, / תִּשְׂבְּעוּ מִזִּיו עֶדְנוֹ,

כְּחַיּוֹת מְשָׁרְתוֹת כֻּלָּם / הָעוֹמְדוֹת עַל תִּלָּם.


אֲלֵיכֶם אֶקְרָא, אִישִׁים, / יַחְדָּו כְּרָאשִׁים כְּרָשִׁים,

לַעֲבֹד פְּנֵי אֵל קָדַם, / כִּי זֶה כָּל הָאָדָם!


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
