קרית החול

קִרְיַת-הַחוֹל

א.

וַיְהִי בִּימֵי עַבְדֵּאל הַשֻּׁלְטָן, בִּשְׁנַת שָׁלשׁ לְמָלְכוֹ,

בַּחֹדֶשׁ הַתְּשִׁיעִי, בְּאֶחָד לַחֹדֶשׁ,

בֵּין עַרְבָּיִם:

וַיֵּט שֵׁבֶט-בֶּדְוִים עַל גְּמַלָּיו לָלוּן

עַל שְׂפַת-מִפְרָץ בְּאֶרֶץ-הַחוֹלוֹת.


כִּי אָמַר חָלִיל הַיָּשִׁישׁ, נְשִׂיא-הַשֵּׁבֶט:

הִנֵּה נָטָה רָקִיעַ לָלוּן בְּאָהֳלֵי-קֵדָר,

וּרְחוֹקָה עִיר-הַתְּמָרִים, מְחוֹז-חֶפְצֵנוּ,

מַהֲלַךְ יָמִים שְׁלשָׁה עֲדֶנָּה —

וְאָהַלְנוּ פֹּה הַלָּיְלָה.


וַיַּעֲנוּ זִקְנֵי-בֶּדְוִים וַיֹּאמְרוּ:

אָכֵן נִכְנְפָה הַחַמָּה אֶל תּוֹךְ נַרְתִּיקָהּ,

וְיָרֵחַ עַל אֹרַח לֹא יִגַּהּ,


וְיָם יְהַלֵּךְ אֵימִים —

נָלִינָה פֹּה עַד בֹּקֶר

וְלֹא יִדְבְּקֵנוּ סַעַר בַּדֶּרֶךְ, לָיְלָה.


וַיִּתְקַע אִישׁ יְתֵדוֹ וַיֶּאֱהַל

עַל גִּבְעוֹת-הַחוֹל הַמּוּפָזוֹת,

הַמִּתַּלְתְּלוֹת תִּלֵּי-תִּלִּים, גַּלֵּי-גַלִּים תֵּרַדְנָה

לִרְחַץ רַגְלֵיהֶן עִם שְׁקִיעָה

בַּיָּם הַגָּדוֹל, מְבוֹא-הַשָּׁמֶשׁ.

ב.

בַּחֲצִי-הַלַּיְלָה נָפְלָה אֵימָה חֲשֵׁכָה גְדוֹלָה

עַל בֶּדְוִי בָאֹהֶל

וְעַל גָּמָל כּוֹרֵעַ בַּחוֹל.


כִּי נִפְתְּחוּ אֲרֻבּוֹת

לְאַשְׁדוֹת-מְרוֹמִים

וַיַּעֲרוּ קִצְפָּם

וַיִּשְׁפְּכוּ זַעְפָּם

עַל בְּנֵי-חֲלוֹף;


כִּי נִפְעֲרוּ לוֹעוֹת-

נַחֲשׁוֹלֵי-כְפִירִים

אֲיֻמִּים-אַדִּירִים —

מֵרֵבֶץ-תְּהוֹמוֹת

זִנְּקוּ אֶל הַחוֹף!


נֵעוֹרוּ-הִתְעוֹרְרוּ

רֻכְסֵי-הֲרָרִים,

רָעֲשׁוּ-הִתְגָּעֲשׁוּ

הַרְרֵי-מִשְׁבָּרִים,

יַבֶּשֶׁת לִסְחֹף;


וַיִּנָּתְזוּ אַרְצָה

מִמַּרְכְּבוֹת-רַעַם,

הִתְמַלְּטוּ כִידוֹדֵי-

חֲזִיזֵי-זַעַם —

הַשָּׂרִיד לִשְׂרֹף!


אֵימִים הִשְׁתּוֹלְלוּ,

עֲנָקִים הִתְהוֹלְלוּ,

וַיֹּאחְזוּ בְרַעֲמוֹת-

בְּנֵי-שַׁחַץ מִתְרוֹצְצִים —

סַעַר לִרְדֹּף;


וַיִּלְפְּתוּ אֵיתָנִים

עַמּוּדֵי-הַשַּׁחַק —

אֲשִׁיּוֹת אַרְבָּע

עִם כִּסֵּא-אַלְלַהּ

בַּמַּבּוּל לִגְרֹף!


וְשָׂטָן מְרַקֵּד

בְּתוֹךְ הַתִּרְבֹּכֶת,

לְיִלְלַת-תַּנִּים

מְחַלֵּל בַּחֲלִילֵי-

אַגְמוֹן וָסוּף:


אֲהוֹ-הָהּ! אֲהוֹ-הָהּ!

אֵתָיוּ, הִתְגּוֹדְדוּ,

גּשׁוּ, הִתְגּוֹשְׁשׁוּ,

כְּל בְּנֵי-שַׁחַץ,

כָּל דּוֹרֵס בָּעוֹף:


עָרוּךְ הַשֻּׁלְחָן

לְכֵרָה רַבָּתִי:

כָּל יֵשׁ בַּיַּבֶּשֶׁת

וְכֹל יֵשׁ בַּיָּם

הַלַּיְלָה לִטְרֹף!


אֲהוֹ-הָהּ! אֲהוֹ-הָהּ!

הַקַּלַּחַת רוֹתַחַת!

וְכֹל אֲשֶׁר יָצַר

שׁוֹכֵן-מְרוֹמִים —

כֶּעָשָׁן יִנְדֹּף!


וַיִתְלַבְּטוּ גַפֵּי-אָהֳלֵי-בֶּדְוִים

כָּל הַלַּיְלָה,

סֹעֲרוּ כְכַנְפֵי נֶשֶׁר פָּצוּעַ

מְרֻדַּף-צָיִד.


וְיִּשְׁמֹט עוֹלָל דַּד-אִמּוֹ בָעֲלָטָה

וַיִשְׁכְּחוּ גִבּוֹרִים לַחַשׁ-אַהֲבָה בָעֲלָטָה,

וַיַּחְרְגוּ מִמִּסְגְּרוֹת-זְרוֹעוֹת-חֵשֶׁק.


וַיְהִי לָהֶם הַלַּיְלָה לֵיל-שִׁמּוּרִים:

אִישׁ וְאִשְׁתּוֹ וְהַטַּף בְּעֵיפַת-אֹהֶל

מַקְשִׁיבִים לְבִעוּתֵי-סוּפָה

בְּשׁוּב מַבּוּל לְשַׁחֵת הָאָרֶץ.

ג.

בַּלַּיְלָה הַהוּא

תָּעָה נָגִיב שְׁתֻם-הָעַיִן,

הוּא נָגִיב הַדֶּרְוִישׁ אִישׁ-הָרוּחַ,

סָבִיב-סָבִיב לְאָהֳלֵי-הַשֵּׁבֶט:


לְלֹא-דֶרֶךְ

יָמֵשׁ חשֶׁךְ, פָּרוּשׂ עַל הַחוֹלוֹת;

פַּלְגֵּי-מַיִם מִקְוֻצּוֹתָיו הַפְּרוּעוֹת יִשְׁטֹפוּ;

שֶׁבַע יִפֹּל נָגִיב וְקָם

וּמַחֲזֶה-לֹא-אֱנוֹשׁ יֶחֱזֶה.

אִישׁ-הָרוּחַ, מַה-מִּלָּיְלָה?

אִישׁ הָרוּחַ, מַה מִסַּעַר וּמִיָּם?


נְאֻם נָגִיב הַדֶּרְוִישׁ

וּנְאֻם הַגֶּבֶר שְׁתֻם-הֶעָיִן:

הוֹי אַתֶּם, בְּנֵי-אֹמֶן לְאַלְלַהּ!

יָרֵאתִי שִׂיחַ לִפְצוֹת:

שָׂרֵי-בַלָּהוֹת מְצָאוּנִי,

אֲפָפוּנִי מְצָרֵי-לָיְלָה:

כָּל רָע יַהֲרוּ עֲתִידוֹת —

כִּי נִתַּךְ חֲרוֹן-אַלְלַהּ עַל עַמּוֹ!


הֲלֹא תַאֲזִינוּ לְקוֹל-הָרַעַם,

מַרְכְּבוֹת-סוּפָה בַמְּרוֹמִים?

אוֹי לִשְׁתֻם-הָעַיִן, אֲשֶׁר רָאָה

חֲזִיז זֶה פִּלַּח לֵב-סַגְרִיר,

וַיַּבְעֵר אֲבוּקוֹת אֶלֶף

נְתִיבוֹת בּיָּם לְהָאִיר!


כִּי הֶרְאַנִי אַלְלַהּ אַדִּירֵי-צִים

בַּחֲזִיז וָרַעַם עַל רַעֲמוֹת-יָם

גּוֹי עָצוּם מֵרָחוֹק בָּא,

אוֹיֵב כַּבִּיר — מִקַּצְוֵי-אָרֶץ:

מֵאוֹת, אֲלָפִים, רְבָבוֹת — וְלֹא יִסָּפֵרוּ!

שְׂאֵת-מִשְׁבָּרִים לֹא תְבַעֲתֵם

וּמִסַּעַר לֹא יֵחַתּוּ:

תְּנוּמָה לֹא יִתְּנוּ לְעֵינָם —

לַחוֹפִים, לַחוֹפִים יַחְתֹּרוּ!


אָז יְרָגְנוּ זִקְנֵי-בֶדְוִים חֲרֵדִים,

אִישׁ בְּסֵתֶר-אָהֲֳלוֹ יֵרָגֵנוּ:

אָכֵן לָעוּ דִבְרֵי-נָגִיב, לָעוּ!


אֵין זֹאת, כִּי מֵעָצְמַת-בִּעוּתֵי-הַלֵּיל

חָכְמַת-אִישׁ-הָרוּחַ הִתְבַּלָּעָה:

אֵיכָה יָזוּעוּ כְלֵי-גֹמֶא הַלַּיְלָה

מֵעֹגֶן-מִבְטַחִים בְּסֵתֶר-נָמֵל?

אֵיכָה יַעֲמֹד בְּלֶב-יָם הַלַּיְלָה

מַלָּח נוֹעָז בְּרֹאשׁ-חִבֵּל?

רַק זַנְבוֹת-תַּנִּינִים רָאָה נָגִיב —

וַיִּהְיוּ בְּעֵינָיו כַּצִּים!


בְּסֵתֶר-אָהֳלִים זִקְנֵי-בֶדְוִים יֵרָגֵנוּ,

אָכֵן לֹא כֵן הַנָּשִׂיא חָלִיל עִמָּדוֹ.

עַל מַרְבַדָּיו הָעַתִּיקִים, מוֹרֶשֶׁת-אָבוֹת, יִרְגַּע,

מְגִלּוֹת-קֶדֶם גְּנוּזוֹת עַל לְבָבוֹ יַעֲלֶה,

בַּצָּמָא כָל דְּבַר הַדֶּרְוִישׁ הַתּוֹעֶה יִשְׁתֶּה:

אָכֵן לֹא תְּכַזֵּבְנָה מְגִלּוֹת-קֶדֶם —

וּבָא לְאֶרֶץ-הַחוֹלוֹת גּוֹאֵל!

נָגִיב, נָגִיב, מָה מִלֵּיל?


וְנָגִיב נָפַל עַל פָּנָיו וַיְקוֹנֵן:

אוֹי לִשְׁתֻם-הָעַיִן, אֲשֶׁר רָאָה

פִּלְאֵי-צִי-אוֹיֵב מִקַּצְוֵי-אָרֶץ:

מִגְדְּלֵי-קֶסֶם לוֹ,

יָשׁוּטוּ עַל פְּנֵי-הַמָּיִם;

אַרְמְנוֹת-מַשְׂכִּיּוֹת לוֹ,

יָצוּפוּ עַל פְּנֵי הַמָּיִם;

הֵיכְלֵי-מִקְדָּשִׁים לוֹ,

יָשׁוּטוּ עַל פְּנֵי-הַמָּיִם;

קִירוֹת, חוֹמוֹת, כְּפָרִים, עָרִים,

יָצוּפוּ עַל פְּנֵי הַמָּיִם!


חָזוּת קָשָׁה הֻגַּד לְנָגִיב:

אֲשֶׁר יָבֹזוּ צִיֵּי-נֵכָר אַרְצוֹ —

לַחוֹפִים, לַחוֹפִים יַחְתֹּרוּ!


וְהַנָּשִׂיא חָלִיל שׁוֹמֵעַ פֶּתַח-אָהֳלוֹ

וּפָחַד לִבּוֹ וְרָחַב כְּפִתְחֵי-הֵיכָלוֹת:

אָכֵן לֹא תְכַזֵּבְנָה מְגִלּוֹת-קֶדֶם!

אָכֵן פָּקַד אַלְלַהּ שִׁמְמַת-אֶרֶץ-הַחוֹלוֹת!

יְהִי שֵׁם-אַלְלַהּ מְבֹרָךְ עַד עוֹלָם.

ד.

בָּאַשְׁמֹרֶת הַשְּׁלִישִׁית נִסְכְּרוּ אֲרֻבּוֹת-הַשָּׁמַיִם,

וַיָּשֹׁכּוּ הַמָּיִם.

אָז יָשׁוּבוּ כְפִירֵי-גַלִּים לְרִבְצָם,

בִּמְאוּרוֹת-יָם לָבֶטַח יִרְבָּצוּ.


וּבֶדְוִים קָרְעוּ יְרִיעוֹת-אָהֳלֵיהֶם

לַסְכִּית לִבְשׂוֹרַת-מִזְרָח, כְּשֹׁךְ הַסַּעַר,

וַתָּצָד אָזְנָם קוֹל-הֲמִית-עֻגָּבִים

עוֹלֶה לִקְרַאת בֹּקֶר מִן הַיָּם.


וּבֹקֶר אָתָא

וַיְצַף אֶת גִּבְעוֹת-הַחוֹל בִּזְהָבוֹ.

וַיֵּתֵּךְ עַל עַרְבוֹת-הַיָּם אֶת כַּסְפּוֹ,

וַיַּבְרֵק בִּצְחוֹקוֹ הַנָּהִיר אֶל פְּנֵי-הַמַּחֲנֶה:

וַיַּצֵּת נְגֹהוֹת בְּעֵינֵי-בֶּדְוִים

נוֹהוֹת אַחַר אֵד מִתַּמֵּר מִן הַמָּיִם,

הֶאָח! מִתַּמֵּר וְעוֹלֶה

מֵעַל מַפְלִיאֵי-נַגֵּן בִּכְלֵי-שִׁיר.


וְהָרִנָּה עָבְרָה בַּמַּחֲנֶה:

אֵדִים הִתְעַלּוּ, צִים הִתְגַּלּוּ!

קוּמוּ אַתֶּם, בְּנֵי-אֹמֶן לְאַלְלַהּ!

צְאוּ וְחֲזוּ בְּתִפְאֶרֶת כְּלֵי-שָׁיִט!

הַסְכִּיתוּ וְשִׁמְעוּ עֻגָּבֵי-שָׁיִט!

הַסְכִּיתוּ וּשְׁמָעוּ:

מִקִּרְיוֹת-פְּלָאִים,

מֵאַרְצוֹת-מֶרְחַקִּים,

בִּרְכַת-אַלְלַהּ אֶל חוֹפֵינוּ יִשָׂאוּ!


אַךְ לָמָּה לֹא תוֹרֶן יֵרָאֶה, וְאֵי מִפְרְשֵׂי-הַבַּד?

אַךְ לָמָּה לֹא מָשׁוֹט יָזִיעַ יְרִיעַת-הַיָּם הָרוֹגַעַת?

אַךְ לָמָּה לֹא מַלָּחִים יָרִיעוּ, לֹא יַעֲנֶה רַב-חוֹבֵל!

אַךְ לָמָּה פְנֵיהֶם יַסְתִּירוּ הַפּוֹרְטִים עַל כְּלֵי-הַשִּׁיר!


הֲתַמּוּ לִגְוֹעַ חוֹבְלִים בִּנְתִיבוֹת-הַיָּם —

וְסַעַר בַּלַּיְלָה הִסִּיעַ הַצִּי לְלֹא-עָם?

אִם בְּיַרכְּתֵי-אֳנִי מַלְאַךְ-הַמָּוֶת נִצָּב

וּפָרַט בְּיָדוֹ הַקָּשָׁה עַל מֵיתְרֵי-עֻגָּב?

ה.

אָז יָשִׁיר קוֹל מִמַּעֲמַקֵּי טִירָה צָפָה,

נְהוּגָה עַל הַמַּיִם בְיַד נִסְתֶּרֶת,

אֶל הַחוֹף הַשּׁוֹמֵם בְּרֹאשׁ-הַצִּי חוֹתֶרֶת —

קוֹל מִטִּירָה:


הַאַתְּ זֹאת אִמִּי, מִנִּי-מִדְבָּר מִתַּמֶּרֶת:

הַאַתְּ זֹאת אִמִּי, בִּזְהַב-הַחוֹלוֹת נֶעְרָמָה?

הַאַתְּ זֹאת קְרָאתִנִי, אֶרֶץ תַּלְאוּבוֹת וּשְׁמָמָה,

וַתִּנְהֲגִי בִּשְׁבִי-חֲלוֹמוֹת נַפְשִׁי הַסּוֹעָה?


אִם לֹא אַתְּ בְּכֶתֶר-מַלְכוּת, אֵם שַׁכּוּלָה, לִי הוֹפַעַתְּ:

דְּמוּת קוֹרֶנֶת מִן הָאֹפֶל בְּמַחֲבוֹאֵי-מְחִלָּה —

שָׁם בְּנֵי-אַנְדַּלּוּסִיָּה גֵאִים קִדְּשוּ הַשֵּׁם בִּתְפִלָּה,

לְמַשַּׁק גַּלֵּי נְהַר-אַל-כַּבִּיר, יְפֵה-נוֹפָהּ יְלַחֵכוּ?


אִם לֹא אַתְּ נָתַתְּ בִּלְשׁונִי קִלְלַת בֵּן הַנֶּעְקָד

עֵת בִּשְּׂרוּ פַעֲמוֹנִים עֲלוֹת-כּוֹפְרִים לַמּוֹקֵד!

אִם לֹא אַתְּ, אֲשֶׁר חָרַתְּ עַל מִצְחִי אוֹת-הַקֵינִי

וַתָּשִׁיתִי אֵיבַת-עוֹלָם בֵּין מוֹלַדְתִּי וּבֵינִי!


בְּשַׁרְבִיט-עֶצֶר צִוִּיתִנִי, קְרָאתִנִי כָל יְמֵי-חֶלֶד

וַתִּכְלְאִי הַשָּׂשׂוֹן מִלֵּילוֹת-הִלּוּלִים בְּאַרְמוֹנִי!

גַּם כִּי נָחוּ גַלֵּי אַל-כַּבִּיר, כְּעֵדֶר-רְחֵלִים, מוּל חַלּוֹנִי —

אַתְּ צִוִּית סַעַר בְּלִבִּי, וַתַּתְעִינִי מִדָּרֶךְ.


קְרָאתִנִי — וָאָרוּצָה: צִוִּיתִנִי — וָאָבוֹאָה!

אַךְ לָמָּה רָעֲמוּ פָּנָיִךְ: לָמָּה עֵינֵךְ — נֵכָר?

חֲלוֹם חָלַמְתִּי, וַיִּפָּעֶם לִבִּי — וּמִי הַחֲלוֹם יִפְתֹּרָה?

הֲתָעִיתִי יְמֵי-דוֹרוֹת, וָאֶתְמַהְמַהּ, וָאֵחָר?


וְלָמָּה יָחִיל לִבִּי בְּשַׁעֲרֵי נוֹפֵךְ הַכָּסוּף,

וְלָמָּה תֶחֱרַד נַפְשִׁי לְמַרְאֵךְ, אִמִּי הַיְחִידָה!

וּמָה הָעֲבוֹתוֹת הַנִּמְתָחִים מֵאַנְדַּלּוּסִיָּה וְעַד הֵנָּה —

וּמַה לְאוֹר וָחשֶׁךְ בְּנַפְשִׁי חִלּוּפֵי-חִידָה?


הוֹי, אַנְדַּלּוּסִיָּה, מוֹלֶדֶת-חוֹרֶגֶת, חֲמוּדָה וּשְׂנוּאָה!

הוֹי גַּלֵּי-נְהַר-אַל-כַּבִּיר וּתְפִלַּת-מְחִלּוֹת רוֹעֶדֶת!

הוֹי צִלְצוּל-פַּעֲמוֹנִים נֶאְדָּרִי וְאֵשׁ-תָּפְתֶּה יוֹקֶדֶת —

הוֹי עֵרֶב-רָב אַכְזָרִי, בְּכַף-הַקֶּלַע הִטִּילַנִי!


הֲלֹא תְרַחֲמִינִי, אִמִּי, אֲשֶׁר דְמוּת-חוֹרֶגֶת צַרְתִּי

מֵעֵבֶר דְּמוּתֵךְ תַּאֲוַת-נֶפֶשׁ — כְּתָאֳמֵי-הַמַּטְבֵּעַ:

אֲשֶׁר לֹא אֶדְלֶה הָאֶחָת, וְאֵין צָרָתָהּ עִמָּהּ —

אֲשֶׁר בֵּין שְׁתֵּיכֶן יַחַד פְּתִיל-חַיַּי יִקָּרֵעַ!


— וּמִן הַטִּירָה עָבַר נֵהַם-עֻגָּבִים

אֶל אַרְמוֹנוֹת וְהֵיכָלוֹת וְאָהֳלִים לְלֹא-אָדָם,

וַתֵּהוֹם הַקִּרְיָה הַצָּפָה מֵעֲבָרִים:


הוֹי, אַתְּ תֵּימָן בַּת-עֲרָב יוֹקֶדֶת,

בְּחֵיקֵךְ הַקּוֹדֵחַ דְּמָעַי רֻתָּחוּ!

וְאַתְּ גַּם אַתְּ, תּוֹגַרְמָה-מוֹלֶדֶת,

בְּאֶלֶף לַיְלָה וְלֵילֵךְ יְמֵי-גֵר יִשְׁלָיוּ:

אֲלֵיכֶן נֶפֶשׁ תֵּצֵא!


הוֹי, גַּלֵּי-בּוֹסְפוֹרוּס מַמְתִּיקֵי-סוֹד:

רַחֲמִים!


הוֹי אַתְּ, אוּקְרַאִינָה בַּת-עַצֶּבֶת,

עַל קִבְרוֹת בָּנַיִךְ הַחוֹרְגִים מְלַבְלֶבֶת!

וְאַתְּ, גַּם אַתְּ, פּוֹלוֹנִיָּה מְחֻנֶּנֶת,

בִּדְמֵי-בָנַיִךְ הַשְּׂנוּאִים פּוֹרָחַת:

זִכְרְכֶן תַּאֲוַת-נֶפֶשׁ!

הוֹי, גַּלֵּי-הַוֵיכְסֶל הַמְּנַהֲמִים:

רַחֲמִים!


וּמִכָּל קְצוֹת-הַצִּי יַעֲנוּ עֻגָּבִים וְאָנוּ:

הוֹי, אַתָּה הַדְּנִיפֶּר, כַּסַּמְבַּטְיוֹן לֹא תַרְגִּיעַ!

הוֹי, אַתְּ סֵינָה, סֵינָה לוּטָה בָּעֲרָפֶל!

וְגַם אַתְּ דָּנֻבָּה, שָׂרָתִי בַּמְּדִינוֹת!

גַּם אַתָּה חִדֶּקֶל, בְּצֵל-דְקָלִים זוֹרֵם,

גַּם אַתֶּם, יְאוֹרִים עֲלוּפֵי-נְהָרָה!

אֲלֵיכֶם, אֲהוּבִים, נֶפֶשׁ תֵּצֵא:

רַחֵמוּ!


רַחֲמוּ עַל בָּנִים נִקְלָעִים בַּכָּף!

ו.

וַיְהִי כִּגְווֹעַ קוֹל עֲנוֹת-עֻגָּבִים,

וַיִּפְּלוּ זִקְנֵי-בֶּדְוִים עַל פְּנֵיהֶם

וַיִּתְפַּלְלוּ לְרַחֲמֵי-אַלְלַהּ.


וְנָגִיב שְׁתֻם-הָעַיִן

הוֹפִיעַ לָפֶתַע בִּקְצֵה הַמַּחֲנֶה הַכּוֹרֵעַ:

שָׂב, חֲגוּר-שַׂק, מִשְּׁבִי-הַלָּיְלָה.

וַיְהִי כִרְאוֹתוֹ אֶת אֶחָיו,

וַיעֲפֵּר לְעֻמָּתָם בְּחוֹל, וַיְקַלֵּל בְּקוֹל —

כִּי חָרָה לוֹ עַד מָוֶת, אֲשֶׁר כָּרָעוּ.

וַיִּקַּח עוֹד מְלֹא-הַחֹפֶן חוֹל-זָהָב,

וַיִּזֶּר לְרוּחַ-הַבֹּקֶר, לֵאמֹר:

כֵּן יִזְרֶה הָרוּחַ עַם-מַאֲמִינִים לִפְנֵי צִי-נֵכָר!


וְכִרְאוֹת נָגִיב אֶת אֶחָיו מִתְנַשְּׂאִים לָקוּם

וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וְעַתָּה, עַם עִזּוּז וְגִבּוֹר, מַה-תָּקוּמוּ מִכְּרֹעַ —

וּתְקוּמָה לֹא תְהִי לָכֶם עוֹד!

הָיוּ לְאַרְנְבוֹת-שָׂדֶה אֲרָיוֹת-הַמִּדְבָּר הָעַזִּים —

אֵיכָה יְשַׁחֲרוּ לַטָּרֶף?

אֲהָהּ, בְנֵי-אֹמֶן, אֵיכָה תֵצְאוּ לַדֶּרֶךְ —

וְאַלְלַהּ בְּקִרְבְּכֶם אָיִן!


וּכְצַחֵק הַבֹּקֶר בַּעֲרֵמוֹת-הַחוֹל, אֵלָיו עֲרֻמּוֹת הִתְאַוּוּ,

וּכְצַחֲקוֹ בְשׁוּלֵי-אֵפוֹדִים לִישִׁישִׁים, בִּסְבַךְ-קַרְנָיו נֶאֱחָזוּ,

כֵּן הִתְקַדְּרָה עֵין-נָגִיב, אַף כֵּן פָּנָיו רָעָמוּ,

וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

הוֹי, בֹּקֶר אַכְזָרִי, לָמָּה תִשַּׁק עֵינֵי-נָגִיב,

וְלָמָּה שֵׂיבַת-דֶּרְוִיש תִּשָּׁקָה?

בָּאתָ בַּעֲקֵב-לַיְלָה — וַאֲנִי לֹא קְרָאתִיךָ:

יָאִירוּ פָנֶיךָ אֶל פָּנַי — וּבְלִבִּי אֹפֶל יִשְׁכֹּן!

כֵּן תָּאִיר לְצָר בַּשַּׁעַר, וּקְסָמִים בְּיָדָיו —

וְאֲחַי בְּנֵי-בֶדְוִים נָמֹגוּ!


עָלְתָה חֲלוּדָה בְחַרְבָּם, קַשְׁתָּם לֹא-דְרוּכָה —

עַל-כֵּן בְּחִיל יִכְרָעוּ:

עַד יַעַל אוֹיֵב לַחוֹף וּמְצָאָם בְּבָשְׁתָּם —

עֵדֶר לַטֶּבַח יוּבָל!


וַיִּרְגַּז נָגִיב אֶל אֶחָיו וַיֵבְךּ,

וַיִשָּׂא עֲלֵיהֶם בְּמַר-נַפְשׁוֹ אֶת הַקִּינָה הַזֹּאת:

אֱלִי, קִרְיַת-הַתְּמָרִים,

אֱלוּ, יוֹשְׁבֵי נָיוֹת בַּמִּדְבָּר!

כִּי נִמְחָה חֶסֶד-הַנָּבִיא

מִדַּרְכֵּי אֶרֶץ-הַחוֹלוֹת,

כִּי סָר חֶסֶד-אַלְלַהּ

מֵאָרְחוֹת-מַאֲמִינָיו בַּמִּדְבָּר,

כִּי בָא הַצָּר בְּשַׁעֲרֵיהֶם —

וּמֵעֵינָם נֶעְלָמָה:

וְהָיוּ לַצָּר לַעֲבָדִים,

בְּנֵיהֶם חֲבוּשִׁים בְּרִכְבּוֹ,

בְּנוֹתָם מִשִּׁקְתוֹת-הַמַּיִם

יִסְחַב וְהִסְגִּיר הַהַרְמוֹנָה,

אַשְׁכְּלוֹת-גַּפְנָם יִקַּח,

תְּאֵנָתָם — וְאָרָה פִרְיָהּ,

תִּפְאֶרֶת-צֹאנָם יָגֹז

וְדַם-מְרִיאֵיהֶם יִסְבָּא,

גְּמַלֵּיהֶם יִטְעַן לַחֲמָם

וְשִׁלַּח עַל פְּנֵי-הַמַּיִם —

וְהוֹבִישׁ שֵׂיבַת נָגִיב

וְהוֹרִיד בְּיָגוֹן שְׁאוֹלָה!

ז.

וְזִקְנֵי-בֶדְוִים קָמוּ מֵהִשְׁתַּטֵּחַ,

וַיִּתְנַשֵּׂא גַם חָלִיל הַיָּשִׁישׁ אִתָּם, כַּלָּבִיא,

וַיִתֵּן לַהַט-עֵינָיו, כִּשְתֵי חֲנִיתוֹת-אוֹר, בְּנָגִיב,

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

אֲמֵלָל הַגֶּבֶר שָׂתַם אַלְלַהּ עֵינוֹ הָאֶחָת,

אָכֵן שִׁבְעָתַיִם אֻמְלַל הָאִישׁ,

עֵינוֹ הַשְּׁנִיָּה כִי תַכֶּה אֵיבָה בְעִוָּרוֹן!


הָמוּ מֵעַי לְךָ, אָחִי, אָחִי נָגִיב!

הַאַתָּה זֶה, אָמַרְתָּ חֶרֶב לִלְטשׁ?

הַאַתָּה אָמַרְתָּ קֶשֶׁת לִדְרֹךְ

אֶל פְּנֵי צִי הַנְּשָמוֹת יְסוּרוֹת-אַלְלַהּ?

הֲלֹא תְרַחֵם נִשְׁמוֹת-בְּנֵי-כַף-הַקֶּלַע,

אוֹזְלוֹת עַרְטִילָאוֹת בְּרַחֲבֵי-יַמִּים,

שׁוֹאֲפוֹת יְמוֹת-עוֹלָם אֶל כָּל חוֹפִים,

חוֹתְרוֹת אֱלֵי מְנוּחָה לַשָּׁוְא!

הַסְכֵּת וּשְׁמָע: כָּל הַבֹּקֶר לִפְדוּת תְּפַלֵּלְנָה —

וְהַיֶּשַׁע מִנְהֶן וָהָלְאָה!

רְאֵה גַם רְאֵה: כָּל הַבֹּקֶר לַחֹף תַּחְתֹּרְנָה —

וְהַחוֹף מִנְהֶן וָהָלְאָה!


אָכֵן קָרוֹב הָאוֹת וְלֹא יְבוֹשֵׁשׁ לָבוֹא!

אָכֵן נִבָּא נָגִיב וְלֹא יָדַע מָה נִבָּא!

הֲכִי עַל-כֵּן בְּחַצְרוֹת-אַלְלַהּ הִסְתּוֹפֵף,

וּמְגִּלּוֹת-קֶדֶם לְפָנָיו נִפְרְשׂוּ כַיְרִיעוֹת?

כִּי כֹה דְבַר מְגִלּוֹת-קֶדֶם אֶל אֶרֶץ-הַחוֹלוֹת:


לְהַכְרִית מִמֶּנָה אַלְלַהּ אָדָם וּבְהֵמָה,

לִהְיוֹתָהּ שְׁמָמָה כַמִּדְבָּר הַגָּדוֹל:

קִלְלַת-דּוֹרוֹת הִיא.

עֵץ לֹא תַצְמִיחַ, כִּי יִנָּפֵשׁ בֶּדְוִי עוֹבֵר בְּצִלּוֹ,

עֵין-סְתָרִים בְּצֵל-שִׂיחִים לֹא תָקֵר, כִּי יְרַוֶּה גָמָל צְמָאוֹ,

לֹא תֵחָרֵשׁ וְלֹא תִזָּרֵעַ!

וְלֹא יָקוּם לָהּ יֶשַׁע מִכָּל שִׁבְטֵי-בֶדְוִים

וּפְדוּת לֹא תְהִי לָהּ מִכָּל נָיוֹת בַּמִּדְבָּר —

אַדְמַת-קְלָלָה הִיא.

עַד יוֹלִיךְ סַעַר שֵׁבֶט-תּוֹעִים, בְּנֵי-כַף-הַקֶּלַע,

עַד תַּשְׁלִיךְ סוּפָה צִי נְשָׁמוֹת תּוֹעוֹת אֶל חוֹפֵי-הָאָרֶץ —

בָּהּ תִּקּוּנָם יִמְצְאוּ בְּנֵי-כַף-הַקָּלַע:

אָכֵן בְּהִפָּדוֹתָם גַּם אוֹתָהּ יִּגְאָלוּ!

ח.

דִּין-דָּן, דִּין-דָּן! צִלְצְלֵי-גִיל:

מְצִלּוֹת-גַּמֶּלֶת תָּרֹנָּה.

כָּל לֵבָב יָגִיל, כָּל לֵבָב יָחִיל:


קִרְיַּת-הַתְּמָרִים בָּאֹפֶק!

צַוָּארוֹ הַמְּדֻלְדָּל הֶאֱרִיךְ גָּמָל,

נָעַר בְּעֶרְגּוֹן-מוֹלֶדֶת;

קָצְרָה רוּחַ בֵּכֶר, עִזּוּז וְקַל,

וַיִּבְעַט — וַיְגַמֵּא אָרֶץ!


וְלָמָּה עוֹד יִתְמַהְמְהוּ בֶדְוִים,

לָמָּה לֹא יֵצְאוּ לַדָּרֶךְ?

הֲכִי עוֹד לֹא שָׁבָה נֶפֶשׁ-הָעָם

מִשְּׁבוּת-צִיֵּי-הַקָּסֶם!

לְצִי הַנְּשָׁמוֹת הַתּוֹעוֹת עַל יָם,

בְּחַבְלֵי-קֶסֶם רֻתָּקוּ:

עוֹד מִנִּי עֻגָּבֵי-הַסֵּתֶר הֵדִים

בְּלִבּוֹת-בֶדְוִים יֶאֱנָיוּ.


נִרְדְּמָה הָרוּחַ בֵּין אִבֵּי-הַסּוּף:

עַרְבוֹת-הַיָּם תִּשְׁלֶינָה…

וּפֶתַע מֶה חָרַד הָעָם מִקָּצֶה

וַיֵּצֵא לִבּוֹ הַיָּמָה?


אֲהָה, בְּנֵי-אֹמֶן, מַה-דְּמוּת תָּצוּף,

עוֹטָה אוֹר כַּשַּׂלְמָה, מִנִּבְכֵי-הַתְּכֵלֶת:

הֲבַרְבּוּר לְבָנָה מֵרַחֲצָה תַּעֲלֶה,

בְּמִכְמְרוֹת-כֶּסֶף נֶאֱחָזָה?


אַךְ לֹא מִתְּהוֹמוֹת הִיא בַרְבּוּר לְבָנָה,

אַף לֹא מִכְמְרוֹת-הַכָּסֶף!

רַק סִירָה לָהּ זֹךְ-בְּדֹלַח, עָלְתָה מִמְּצוּלָה,

וְדַיָּג יְשִׁיטֶנָּה — אָיִן.

אָנֶה וְאָנָה הִקְרִינָה בָאוֹר,

אַחַר מַעֲדַנּוֹת הִפְלִיגָה,

אֵין הֶגֶה וּמָשׁוֹט, שָׂחֹה וְחָתֹר,

אֶל צִי בְנֵי-כַף-הַקָּלַע.


וַיֵּדַע הָעָם, כִּי בָא קֵץ-הַפְּלָאוֹת,

אַךְ פִּשְׁרוֹ מִי יַגִּידֶנּוּ!

וַיִּשָּׂא הָעָם עֵינָיו, צוֹפֶה כֹה וָכֹה

לְנָגִיב שְׁתֻם-הָעַיִן:

זֶה יִמְצָא פֵשֶׁר וְהִגִּיד הָאוֹת —


אַךְ חָמַק נָגִיב וְאֵינֶנוּ:

בִּמְרִי לֵב נֶעְכָּר נָס דֶּרֶךְ-מִדְבָּר,

וַיְכַס הַחוֹל עִקְבוֹת הָאִישׁ וּגְמַלּוֹ.

ט.

וְהַסִּירָה בַת-הַמְּצוֹלוֹת הִשְׁתַּחְוְתָה אָנֶה וְאָנָה,

אַחַת לַשַּׁיָּרָה עַל הַחוֹף וְאַחַת לַצִּים יָמָּה,

אַחַר הֶחֱלִיקָה כְעָב קַלָּה, נִכְסֶפֶת בַּעֲקֵב-סַהַר,

נָגְעָה בַמַּיִם וְלֹא נָגְעָה, מִשַּׁלְוָתָם לֹא גָרָעָה.


אֵין הֶגֶה וְאֵין מָשׁוֹט, לְלֹא-הֶגֶה וְלֹא-אֹמֶר,

קָרְבָה אֶל פְּנֵי-הַצִּים, בְּשׁוּלֵי-הַטִּירָה נָשָׁקָה.

וַתָּקָם הַטִּירָה אֵלֶיהָ, סְגוּרָה וּמְסֻגֶּרֶת,

חַשְׁרַת-חִידוֹת כְּבֵדָה עַל כָּל סְבִיבֶיהָ שָׁתָה.


אָז יִרְאוּ בֶדְוִים מֵחוֹף, וִהִנֵּה עָלְתָה מִן הַסִּירָה

קְצָת-תִּקְוַת-חוּט מוּפֶזֶת, מֵחוֹל וְאוֹר נִשְׁזָרָה.

אֶל כָּתְלֵי-הַטִּירָה טִפְּסָה וַתָּנַח עַל הַמַּיִם:

אָז גַּם רֵאשִׁיתָהּ יִרְאוּ — בַּעֲקֵבָם הִיא נִכְרָכָה.


וְרָעֲדוּ יְדֵי-בֶדְוִים וּמָה הִרְעִידָן לֹא יָדָעוּ,

וְנִשְׁלְחוּ מֵאֲלֵיהֶן אֶל תִּקְוַת-חוּט-הַפֶּלֶא:

אַחֲרֶיהָ בַּעֲתֶרֶת-חֶמְדָּה לְבָבָם יַהֲלֹךְ תֶּמֶס,

אֶפֶס מֵעָפָר הֲרִימָהּ רוּחַ בָּהֶם לֹא קָמָה!


וְתִקְוַת-חוּט-הַזָּהָב נֶאֱרֶגֶת-מִשְׂתָּרֶגֶת

מֵחוֹף אֶל סִירַת-בְּדֹלַח, מִסִּירָה וּלְצִי-הַקֶּסֶם,

מִצִּי וּלְסִירַת-בְּדֹלַח — וְשָׁבָה לַיַּבֶּשֶׁת,

קַו לְקָו, כִּשְתִי-זָהָב בְּאֶרֶג-קַרְנֵי-שָׁמֶשׁ.


וַיְהִי כְנוּחַ הַמַּסֶּכֶת — גֶּשֶׁר-תִּקְווֹת-הַפֶּלֶא —

וַיְצַלְצְלוּ פַּעֲמוֹנִים בַּקִּרְיָה הַשָּׁטָה מִקָּצֶה:

וַיֶּחֶרְדוּ הַצִּים וַיָּנוּעוּ, כִי נָעָה סִירַת-הַבְּדֹלַח,

וַתַּפְלֵג עִם שׁוּלֵי-הַמַּסֶּכֶת, חוּטֶיהָ מִתּוֹכָהּ נֶאֱרָגוּ.


אָז תַּעַל טִירָה כוֹשֶׁלֶת בְּגֶשֶׁר-הַתִּקְווֹת רִאשׁוֹנָה.

אָז יַעֲלוּ מִגְדְּלֵי-קֶסֶם עִם אַרְמְנוֹת-מַשְׂכִּיּוֹת אַחֲרֶיהָ.

אָז יַעֲלוּ מִשְׁכְּנוֹת-אֵל, בָּתֵּי-חֵמָר בַּעֲקֵבָם,

אָהֳלֵי-בַד לְרַגְלָם, קִירוֹת וְחוֹמוֹת אַחֲרֵיהֶם,

יַחֲלִיקוּ בְגֶשֶׁר-הַתִּקְווֹת וְעָלוּ בַסָּךְ אֶל הַחוֹף —

סִירַת-הַבְּדֹלַח תְּלַוֵּם בִּבְרָכָה עִם שׁוּלֵי-הַגָּשֶׁר.


וַיְהִי כַאֲשֶׁר תַמּוּ כָּל הַצִּי לַעֲבוֹר,

וַיִּשְׂאוּ בֶדְוִים עֵינֵיהֶם וַיִּרְאוּ,

וְהִנֵּה בְעִיר גְּדוֹלָה לֵאלֹהִים רַגְלֵיהֶם עוֹמְדוֹת:

עֲטָרוּם צֶאֱלֵי הַתְּאֵנָה וְהַגֶּפֶן וְהַתֹּמֶר,

וַעֲנַן-הַקְּטֹרֶת עִם צְלִילֵי-זִמְרָה עוֹלֶה מִמִּקְדָּשִׁים,

וְסִירַת-הַבְּדֹלַח כָּאֵד נִדָּפָה,

וּמַחֲזוֹת-הַלַּיְלָה כַחֲלוֹם רָע נָמֹגוּ!


וַיַּעֲלוּ בֶדְוִים עַל גְּמַלֵּיהֶם וַיִּרְכְּבוּ

לְהַגִּיד בְּקִרְיַת-הַתְּמָרִים פֹּעַל-אַלְלַהּ,

לְשִׁבְטֵי נָיוֹת בַּמִּדְבָּר פִּלְאֵי-אַלְלַהּ,

אֲשֶׁר נָטַע עֵדֶן בַּצָּיּוֹן,

וְצִי נְשָׁמוֹת תּוֹעוֹת הֵקִים עִיר,

בָּנָה אֶת קַרְיַתּ-אַל-רָמֶל עַל שְׂפַת-הַיָּם,

הִיא קִרְיַת-הַחוֹל, עַד הַיּוֹם הַזֶּה.


תל-אביב, תרפ"ה.
