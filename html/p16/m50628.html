מַחֲזֶה מוּל מַחֲזֶה

Ι


עֵת בָּצִיר. צְלִיל כִּנּוֹרוֹת,

עוּגָבִים עִם מְצִלְתָּיִם:

חַג שֶׁל בַּכְּכוֹס זֶה הַיּוֹם,

חַג שֶׁל עֲגָבִים וָיָיִן.


הוֹד גְּלִימוֹת צִבְעוֹנִיּוֹת,

אֵשׁ־מַבָּט וִיקוֹד־שְׂפָתַיִם;

הֵד יָעִיר בָּהָר וָגַיְא

רַעַם הוֹלֵלוּת סוֹעֶרֶת.


הוֹד גְּלִימוֹת צִבְעוֹנִיּוֹת,

אֶצְעֲדוֹת נָשִׁים תַּבְרֵקְנָה

וַעֲסִיס עִנְבֵי־עַנְבָּר

יְשַׁכֵּר כָּל לֵב וָמֹחַ.


לֵב וּמֹחַ תְּשַׁכֵּר

תּוֹךְ כְּדֵי רִקּוּדֵי־פֶרֶא,

נְשִׁיקַת יְקוֹד שְׁפוּכָה

עֲלֵי־וֶרֶד לְחָיָיִם.


לַחַשׁ־חֲשָׁקִים נִשְמָע

עִם אַנְחוֹת כְּאֵב־הַמֶּתֶק.

לַיְלָה בָּא. גּוֹעֵשׁ מִשְׁתֵּה

הוֹלֵלוּת וּזְנוּת וָתֶבֶל…


–––––


וּבְמִקְדַּשׁ־הָאֱלֹהִים

אֱלֹהֵי נָקָם וָזַעַם,

יִשָׁמְעוּ אוֹתָהּ שָׁעָה

הִמְנוֹנוֹת סְדוּרִים בְּקֶצֶב.


מַבְרִיָקה אֵשׁ הַזְּבָחִים;

כֹּהֲנִים אֶל הַמִּזְבֵּחַ

מוֹבִילִים שִׁבְעִים פָּרִים,

עֲטוּרִים פִּרְחֵי־תִפְאֶרֶת.


וְרֹאשׁ־הַכֹּהֲנִים בְּקוֹל

תְּפִלָּה, עוֹטֵה מְעִיל־צַחַר

יַקְרִיבֵם אַחַד אֶחָד

לֵאלֹהֵי נָקָם וָזָעַם.


יִתְפַּלֵּל: "אֵל־אֱלֹהִים,

אֶת חֲרוֹן־אַפְּךָ הָשִׁיבָה

מֵעַמִּים, שֶׁלֹּא יָדְעוּ

אוֹר־אֱמֶת וְזֹהַר־צֶדֶק.


לְבָבוֹת שֶׁהֻקְשְׁחוּ

בִּנְשִׁימַת־שָׁלוֹם טַהֵרָה,

וּבְקַרְנֵי־אֱמֶת פַּזֵּר

צִלְלֵי תוֹעֵי־הָרוּחַ".


מִשְׁתַּפֵּךְ קוֹל־הַתְּפִלָּה

עַל רָאשֵׁי הָמוֹן כּוֹרֵעַ,

זַךְ כְּבֹקֶר הָאָבִיב

וְרַב־חֹם כְּשֶׁמֶשׁ־נֶגֶב.


ΙΙ

אוֹר קָלוּשׁ שֶׁל הַמְּנוֹרָה

דֹּם יָאִיר חֶשְׁכַת הַחֶדֶר

בְּמַחְתֶּרֶת אֲדָמָה.

בְּלֹא־הֵד, בְּלֹא־תוֹחֶלֶת


פְּתִילִים בְּתוֹךְ מְנוֹרָה

בַּת שִׁבְעַת קָנִים יִדְעָכוּ,

הֲפוּכָה בְּיַד זֵדִים

עַל־פְּנֵי בִימָה שׁוֹמֶמֶת.


עַל שֻׁלְחָן עָרוּךְ לִפְנֵי

הַבִּימָה מַצּוֹת תַּלְבֵּנָּה.

פֹּה וְשָׁם דַּפִּים בָּלִים

שֶׁל הַהַגָּדָה יַבְהִיקוּ.


עַל רִצְפָּה, בֵּין הַמְּכִתּוֹת,

עַל גְּוִיל־תּוֹרָה קָרוּעַ

וּמְגֹאַל־דָּמִים, גוּף־אִישׁ

מְנֻפַּץ־קָדְקֹד יָנוּחַ,


גּוּף “אָנוּס”, שֶׁנֶּהֶרַג

וְנִקְרַשׁ בְּתוֹךְ הָאֹפֶל,

רַק עִתִּים תָאִיר עָלָיו

הַמְּנוֹרָה זוֹ הַדּוֹעָכֶת.

––––––––

וְעַל־פְּנֵי כִכַּר מַדְרִיד,

שַׁם לִפְנֵי מקְדַּש־אֱלֹהַּ,

אֵל רַב־חֶסֶד וּסְלִיחָה,

מוּל מִקְדָּשׁ מוּצַף אוֹר־זֹהַר


יִתְנַשְׂאוּ שְׁנֵי עַמּוּדִים,

שְׁתַּיִם מְדוּרוֹת הֻתְקָנוּ,

שְׁנַיִם מִזְבְּחוֹת־עֲנָק,

שָׁם עוֹלוֹת כָּלִיל יָקְרָבוּ.


כֹּהֲנִים עוֹטֵי־שְחוֹרִיּם,

מִזְמוֹרֵי־קָדְשָׁם יָשִׁירוּ

וּכְדַרְכָּם אֶל הַמְּדוּרוֹת

שְׁנֵי זְקֵנִים חִוְּרִים יִמְשֹׁכוּ.


בַּמַּחְתֶּרֶת מְצָאוּם,

בְּאָכְלָם סְעֻדַּת־הַפֶּסַח…

לֹא תוּכַל זאֹת כְּנֵסִיָּה

סַלְחָנִית לָהֶם לִסְלֹחַ…


אוֹת נִתַּן, וְלֶהָבוֹת

הִתְפָּרְצוּ כְּגַלֵּי־זַעַם…

וְתִימְרוֹת עָשָׁן כִּסּוּ

פְּנֵי־חוֹטְאִים זְעוּמֵי־אֱלֹהַּ.


רֵיחַ שֶׁל קִיטוֹר וָדָם…

וּבְתִפְאֶרֶת־חַג וָקֶצֶב

“De profundis”יִשְׁתַּפֵּךְ

עַל רָאשֵי הָמוֹן כּוֹרֵעַ…


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
