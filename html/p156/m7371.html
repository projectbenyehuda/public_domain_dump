מחברת שישית


אָמַר עִמָּנוּאֵל הַמְּחַבֵּר:

הָיִיתִי עִם הַשָּׂר בִּימֵי חָרְפִּי בְּמוֹשַב אֲנָשִׁים נְבוֹנִים

בַּחוּרִים וּזְקֵנִים

מְדַבְּרִים מֵאֹדוֹת הַמִּדּוֹת הַחֲמוּדוֹת

וְהַמְּלִיצוֹת הַנִּכְבָּדוֹת

וּמְפָאֲרִים הַשִּׁיר וְחִין עֶרְכּוֹ

זֶה אוֹמֵר בְּכֹה וְזֶה אוֹמֵר בְּכֹה

מִי יְהַלֵּל שִׁירֵי חַכְמֵי סְפָרַד

וּמִי יֹאמַר כִּי סְפָרַד עָלָה שָׁמַיִם וְיָרַד

וּמִי יֹאמַר כִּי מְלִיצֵי רוֹמִי הֵם בְּמוֹרַד

וּמִי יֹאמַר כִּי בִּפְּרוֹבִינְצָ"ה תַּחֲנוֹת

הַשִּירִים הַחֲמוּדִים וְהַמְּלִיצוֹת הָעֲדִינוֹת

וּמִשִּירֵי חַכְמֵי רוֹמִי כֻּלָּם קוֹרְאִים בְּגָרוֹן

מְעֻוָּת לֹא יוּכַל לִתְקֹן וְחֶסְרוֹן

לֹא יִהְיֶה לָהִם זִכְרוֹן

וַאֲנִי הָיִיתִי בְּכָל זֶה מַחֲרִישׁ

וְלִבִּי כְּנָחָשׁ יִשָּך וּכְצִפְעֹנִי יַפְרִישׁ

וְשָׁם נִקְרָה אֱוִיל מַשְׁרִישׁ

מִבְּנֵי פְּרוֹבִינְצָ"ה בְּהֵמָה עַל צוּרַת אִישׁ

יַבִּיעַ שֶׁמֶן רָקְחֵנוּ בַּהֲבָלָיו וְיַבְאִישׁ

וְאַבְנֵי אֶלְגָּבִישׁ יִקְלַע

פִּלֵּג יְיָ לְשׁוֹנוֹ וּבִלַּע

יַעְתִּיר עָלֵינוּ מִלִּים

יִלְעַג וְאֵין מַכְלִים

יֹאמַר מַה תִּתִהַלְלִי רוֹמִי בָּעֲמָקִים

הִנֵּה בִּפְּרוֹבִינְצָ"ה צַנְתְּרוֹת הַשּׂכֶל הַחֵן מְרִיקִים

רְפָאִים יֵחָשְׁבוּ אַף הֵם כָּעֲנָקִים

שָׁמָּה יָשְׁבוּ כִסְאוֹת

לְלִבּוֹת הַנְּשֻׂאוֹת

וּבְכָל מַעַרְכוֹת הַמְשֹׁרְרִים וְהַמְּלִיצִים

נַחְנוּ נַעֲבֹר חֲלוּצִים

נַחְנוּ חֲגוּרֵי אֵפוֹד הַמּוֹעֵצָה

נַחְנוּ רֹאשֵׁי אַלְפֵי הַשִּׁיר וְהַמְּלִיצָה

בְּכָל הָאָרֶץ יָצָא קַו הוֹדֵנוּ

מִי זֶה אֲשֶׁר יַעֲמֹד בְּסוֹדֵנוּ

וּמִי כָמוֹנוּ וּמִי יוֹעִידֵנוּ

לָנוּ מִשְׁפַּט הַגְּאֻלָּה.

אֵין לָכֶם חֵלֶק בַּלָּשׁוֹן וְלֹא נַחֲלָה

וְאִם תָּבוֹזוּ לְשֵׂכֶל מִלָּי

בְּרוּ לָכֶם אִישׁ וְיֵרֵד אֵלָי

אָז רָאִיתִי אַנְשֵׁי גְּאֻלָּתִי כִּי דָּמָמוּ

בּוֹשׁוּ וְנִכְלָמוּ

יָשִׂימוּ יָד עַל פֶּה

וְלֵב כָּל הָעָם עָלָיו עֻלְפֶּה

כִּי גָּדְלָה צַעֲקַת הָאִישׁ עַד לְאֵין מַרְפֵּא

וְהָאֱוִיל בְּכָל זֹאת אוֹתָנוּ יְגַנֵּה

וְאֵין קוֹל וְאֵין עוֹנה

יֹאמַר עִמְדוּ עֲמֹדוּ וְאֵין מַפְנֶה

אָז לָבְשָׁה אוֹתִי רוּחַ קְנָאוֹת

וְנִשְׁבַּעְתִּי בֵּייָ צְבָאוֹת

לַעֲשׂוֹת לָהֶם אוֹת

וְלִרְאוֹת וּלְהֵרָאוֹת

אִם לָנוּ לֵבָב כְּמוֹהֵם בִּמְלִיצוֹת וְשִׁירִים

וָאֹמְרָה אֶל הַסְּגָנִים וְאֶל הַחוֹרִים

מִי הוּא זֶה וְאֵי זֶה הוּא אֲשֶׁר הוֹכִיחַ בַּדְּבָרִים

וַיַבְאִישׁ אֶת רֵיחֵנוּ בַּהֲבָלָיו

אַל יִפֹּל לֵב אָדָם עָלָיו

עַבְדְּכֶם יֵצֵא לְהִלָּחֶם-בּוֹ

וְאִם אֲנִי לֹא אָשִׁיב צוּר חַרְבּוֹ

וְלֹא אָשִׂים שִׁירָיו נִכְלָמִים

וְחָטָאתִי אֲלֵיכֶם כָּל הַיָּמִים

אָז אָמְרוּ אֵלַי אַנְשֵׁי עֲצָתִי בָּרוּךְ אַתָּה וּבָרוּךְ טַעְמָך

אֱלֹהִים יַעֲנֶה אֶת שְׁלוֹמָךְ

וֵייָ יִהְיֶה עִמָּךְ

וָאָקוּם מִמְּקוֹמִי וָאֵלֵך אֶל הָאִישׁ עַד מְקוֹמוֹ

וַאֶקְרָא אֶת הָאִישׁ בִּשְּמוֹ

וָאֹמְרָה הָעֶצֶב הַנִּבְזֶה הַנָּפוּץ הַקָּטָן הַמִּתְהַלֵּל בְּשִׁירִים וּבִמְלִיצוֹת וּבְמִשְׁפָּטָן

וַתֹּאמֶר עַל שִׁירוֹתֵינוּ כִּי הוֹבִישׁ מֶבָּטָן

הִנֵּה אָנֹכִי יָצָאתִי לְךָ לְשָׂטָן

וְאִם אַתָּה אִיּוֹב אֲנִי אֱלִיהוּא

מַה-יָדַעתָּ וְלֹא נֵדָע תָּבִין וְלֹא עִמָּנוּ הוּא

חֵי הַבְּרִית אִם אַתָּה יָם הַחָכְמוֹת אֲנִי בְּרִיחוֹ

יְיָ צְבָאוֹת שְׁלָחַנִי וְרוּחוֹ

אַחֲבֵּר בַּפֶּלֶס שִׁירִים וּמְלִיצוֹת בְּמֹאזְנַים אֶשְׁקֹל

אֶת מִי חֵרַפְתָּ וְגִדַּפְתָּ וְעַל מִי הֲרִימוֹתָ קוֹל

הִנֵּנִי הַקָּטֹן מְעַבְדֵי רַבּוֹתַי

וְלֹא הִשּׂגְתִּי בִּמְרֻצַת שִׂכְלִי אֲבַק מַעֲלַת אֲבוֹתַי

וְאַתָּה בְּעֵינֵיךָ הֵימָן וְכַלְכֹּל

רַב מְחוֹלֵל כֹּל

יַחְדָּו לְמַעֲרֶכֶת הַנִּסָּיוֹן נִקְרָבָה

הִנֵּה קָמָה אֲלֻמָּתִי וְגַם נִצָּבָה

אֶעֱנֶה אִם תִּשְׁאָלֵנִי

אוֹ אֲדַבֵּר וַהֲשִׁיבֵנִי

וַיַעַן הָאִישׁ וַיֹּאמַר כָּל יִשְׁעִי וְחֶפְצִי

לְנַסּוֹתְךָ כַּאֲשֶׁר רָאִיתִי בְאַרְצִי

יַעֲמֹד הַשּׁוֹאֵל וְיִשְׁאָל

וְיֹאמַר לְרֵעֵהוּ אִם תִּגְאַל גְּאַל

וְהוּא יַעֲנֶה הַשּׁוֹאֵל עַל שְׁאֵלָתוֹ

וְעַל פִּי הַפָּסוּק תִּהְיֶה תְּשׁוּבָתוּ

וְתִהְיֶה הֲפָקַת שְׁאֵלַת הַשּׁוֹאֵל וּמֵעִנְיָנָיו

דָּבָר דָּבֻר עַל אָפְנָיו

וְאַתָּה הִנְנִי אֶעֱרֹךְ מַעֲרֶכֶת שְׁאֵלוֹתַי וּמַחֲנוֹתָם כַאֲשֶׁר מִלִּבִּי אֲנִי בּוֹדֵא אוֹתָם

וְאִם עַל פִּי הַפָּסוּק תִּהְיֶה תְּשׁוּבָתָם

יָדַעְתִּי כִּי אַתֶּם אַנְשֵׁי הַשֵּׁם

וּבְבָתֵּי הַחָכְמָה וּבְחוֹמוֹתֶיהָ לָכֶם יָד וָשֵׁם

אָז אָמַרְתִּי לְעֵינֵי הַקְּהִלּוֹת

כִּמְהַתֵּל בּוֹ וּכְמוֹצֵא עֲלִילוֹת

אֲדֹנִי כַּמָּה יִהְיֶה מִסְפַּר הַשְּׁאֵלוֹת

וַיַעַן הָאִישׁ בְּקוֹל רָם

הַשְּׁאֵלוֹת חֲמִשִּים מִסְפָּרָם

וָאַעַן וָאֹמְרָה

אַל יִחַר לַאֲדֹנִי וַאֲדַבֵּרַה

אוּלַי יַחְסְרוּן חֲמִשִּים הַשְּאֵלוֹת חֲמִשּׁה

הֲתַּעֲנֵנִי בַּעֲבוּר זֶה קָשָׁה

וַיֹּאמֶר לֹא אַשְׁחִית אִם אֶמְצָא שָׁם אַרְבָּעִים וַחֲמִשּׁה

וָאֹסֶף עוֹד לְדַבֵּר אֵלָיו תַּעְתֻּעִים

וָאֹמְרָה אוּלַי יִמְצְאוּן שָׁם אַרְבָּעִים

הֲתִתְּנֵנִי בַּעֲבוּר זֶה לְגִדּוּפִים

וַיֹּאמֶר הָאִישׁ נִרְפִּים אַתֶּם נִרְפִּים

עַל כֵּן אַתֶּם הוֹלְכִים מִדְּחִי אֶל דֶּחִי

מִי נָשָׂא יָדוֹ אֵלֵינוּ וַיֶּחִי

לָנוּ מֵי הַחָכְמוֹת נוֹזְלִים וְלָכֶם הַמַּכְתֵּשׁ אֲשֶׁר בַּלֶּחִי

וָאַעַן וָאֹמַר הִכּוֹן וְהָכֵן לְךָ אַתָּה הַמִּתְלַהְלֵהַּ

הַעֲמֵק שְׁאָלָה אוֹ הַגְבֵּהַּ

וְרַבֶּה צְבָא שְׁאֵלוֹתֶיךָ וּמַחֲנוֹתָם

כְּכוֹכְבֵי הַשָּׁמַיִם וְאֶעֱרֹךְ תְּשׁוּבוֹתָם

אָז נִקְבְּצוּ עָלֵינוּ קְהִלָּה גְּדוֹלָה

לִרְאוֹת לְמִי תִּפְאֶרֶת הַמְּלִיצָה וּמַהֲלָלָהּ

וְהָאִישׁ שׁוֹאֵל שְׁאֵלוֹתָיו

וַאֲנִי עוֹנֶה עַל פִּי מִדּוֹתָיו

וַיְצַוּוּ עָלֵינוּ הַנִּכְבָּדִים וְהַחֲשׁוּבִים

לִהְיוֹת הַשְׁאֵלוֹת וְהַתְּשׁוּבוֹת בְּסֵפֶר נִכְתָּבִים

לְמַעַן יַעַמְדוּ יָמִים רַבִּים


וְאֵלֶּה הֵם שְׁאֵלוֹת הָאִישׁ אֵלָי: וְאֵלֶּה הֵם תְּשׁוּבוֹתָי:

מָה אוֹמֵר שַׁבְּתַּי           אִישׁ אִמּוֹ וְאָבִיו תִּירָאוּ וְאֶת שַׁבְּתוֹתָי

וְגַלְגַּלּוֹ מָה אוֹמֵר בְּאָזְנַי           אֶתְנַהֲלָה לְאִטִּי לְרֶגֶל הַמְּלָאכָה אֲשֶׁר לְפָנַי

וּמֶה עָשָׂה הָעֵת רִשְׁתּוֹ פָרָשׂ           שָׂם תֵּבֵל כַּמִּדְבָּר וְעָרָיו הָרָס

מָה עִנְיַן צֶדֶק           שׁוֹפֵט וְדוֹרֵש מִשְׁפָּט וּמְהִיר צֶדֶק

מָה אוֹמֵר מַאֲדִים כִּי יִתְאַדָּם           אָרוּר מֹנֵעַ חַרְבּוֹ מִדָּם

הַיָּדַעְתָּ הַדְּבָרִים אֲשֶׁר הַשּׁמֶש

תֹּאמְרֵם           כָּל מְאוֹרֵי אוֹר בַּשָּׁמַיִם אַקְדּירֵם

אֵיךְ יִקְרְאוּ נֹגַהּ הַיּוֹדְעִים עִנְיָנֶיהָ           הַמֹּכֶרֶת גֹּויִם בִּזְנוּנֶיה

וְכוֹכָב מָה עִנְיָנָיו           קֶסֶת הַסּוֹפֵר בְּמָתְנָיו

וְהַיָּרֵח מַה דַּרְכּוֹ           חָלָק מִשֶּׁמֶן חִכּוֹ

וּבְאֶחָד בּוֹ מַה טִיבוֹ           יְיָ בְּמִשְׁפָּט יָבוֹא

מָה אוֹמְרִים הַצַּיְקָנִים           בְּכָל עֵת יִהְיוּ בְּגָדֶיךָ לְבָנִים

מַה תֹּאמַר לַאֲשֶׁר זָרְחָה הַשֶּׁמֶש

עֲלֵיהֶם           סָר צִלָּם מֵעֲלֵיהֶם

וַאֲשֶׁר נָסוּ צְלָלָיו           זָרְחָה הַשֶּׁמֶש עָלָיו

מַה יִתְאַוּוּ הָעֲצֵלִים           שֶׁיָּפוּחַ הַיּוֹם וְנָסוּ הַצְּלָלִים

מָה אוֹמֵר הַחוֹלֶה           כְּסָלַי מָלְאוּ נִקְלֶה

מַדּוּעַ תֵּיטִיב רֹאשָׁה הַצְּבִיָּה           לְהַרְאוֹת הָעַמִּים וְהַשּׂרִים אֶת יָפְיָהּ

מַדּוּעַ נִבְרָא בּוֹר שָׁאוֹן וְטִיט הַיָּוֵן           לְהִסָּתֶר שָׁם פֹּעֲלֵי אָוֶן

מֶה חָשְׁבָה הַיּוֹשֶבֶת עַל מַשְׁבֵּר           חִשְּבָה לְהִשָּבֵר

מָה אוֹמֵר יִשְׂרָאֵל לְעֹשָׂיו           הַצִּילֵנִי נָא מִיַּד אָחִי מִיַּד עֵשָׂו

בַּמֶּה יִוָּדְעוּ הַנִּסְתָרִים           בְּמִשְׁפַּט הָאוּרִים

אָן לִבּוֹת הַחשְׁקִים וּמִי לְקָחָתַם           רָחֵל גְּנָבָתַם

מַה יֹאמְרוּ עֲשִׁירֵי אַדְמָתֵנוּ           לֹא יָדַעְנוּ מִי שָׂם כַּסְפֵּנוּ בְּאַמְתְּחֹתֵינוּ

מַה דֶּרֶךְ הֶעָצֵל           כְּעֶבֶד יִשְׁאַף צֵל

וַאֲשֶׁר עַל גּוּפוֹ יַחְמֹל           יִתְרוֹן לוֹ שֶיַּעֲמֹל

מַה דֶּרֶךְ הַמֹּאזְנַיִם לִגְרוֹם           לָשׂוּם שְׁפָלִים לְמָרוֹם

הֲתֹאמַר עַל הַחוֹבְלִים מִזְמוֹר           שְׁבוּ לָכֶם פֹּה עִם הַחֲמוֹר

וְאֵיךְ תִּקְרָא בַּעֲלֵי הַזָּקָן הַנִּבְזִים           שְׂעִירֵי עִזִּים

הֲיִמָּצֵא בּוּר בַּעַל מַעֲשִׂים           הַיְרֻצוּן בַּסֶּלַע סוּסִים

בַּמֶּה נַכִּיר חַרְבוֹת צֻרִים           בַּשֶׁבֶר אֲשֶׁר הֵם שֹׁבְרִים

וַאֲשֶׁר בֵּין טוֹב לָרַע לֹא יְבַקֵּר           עֵינֵי הָאֲנָשִׁים הָהֵם תְּנַקֵּר

וְרֹאשׁ מִי יִקָּרַע עֵרֶב וָשֶׁתִי           מֵאִישׁ שֹׁגֶה וּמִפֶּתִי

הֲתַּכִּיר גֵּרְשֹׁם           שׁוֹר נַגָּח הוּא מִתְּמֹל שִׁלְשֹׁם

וּמַה מִבְּנוֹ           הָפַךְ הַנֶּגַע אֶת עֵינוֹ

מִי בְּחַיָּיו נִקְבָּר           הַחֹלֶד וְהָעַכְבָּר

מִי שָׁמַר לְנַפְשִׁי הָעֳרְמָה           שִׁבְתְּךָ בְּתוֹך מִרְמָה

מַה סּוֹף כָּל בּוֹצֵע בָּצַע           פֶּצַע תַּחַת פֶּצַע

מִפְּנֵי מָה עָלָה מֹשֶׁה בְּהַר סִינָי           מִפְּנֵי אֲשֶׁר יָרַד עָלָיו אֲדֹנָי

אֶל מִי תוֹדֶה בִּרְאוֹתְךָ הָעָפְרָה           אֶל עֹשֶׂיהָ וְיֹצְרָהּ

מַה תֹּאמַר מִן הַדּוֹר שֶׁל עַכְשָׁו           זֶה דוֹר דּוֹר שָׁוְא

מָתַי נִשְׁחַת עַכּוֹ           בִּמְלֹאת שִׂפְקוֹ

וּמַדּוּעַ נִשְׁחַת עַכּוֹ           כִּי הִשְׁחִית כָּל בָּשָׂר אֶת דַּרְכּוֹ

וּמָה אוֹמֵר עַכּוֹ           רְאֵה יְיָ וְהַבִּיטָה לְמִי עוֹלַלְתָּ כֹּה

מַה תֹּאמַר מִבַּעַל הָעֲגוּמָה           יוֹסִיף עֶצֶב עִמָּה

מַה תֹּאמַר מִן הַיָּתוֹם           מִכַּף רֶגֶל וְעַד רֹאשׁ אֵין בּוֹ מְתוֹם

מָה אוֹמֶרֶת הַגֶּפֶן אֶל הַנְּדִיבִים           נְשָׂא יָדֶךָ אַל תִּשְׁכַּח עֲנָבִים

מָה אוֹמֵר הַכּוֹס בֶּאֱמֹר הַשִּׁכּוֹר           לֹא אִירָא מֵרִבְבוֹת עָם אֲשֶׁר סָבִיב

          מִי לַיְיָ אֵלָי           שָׁתוּ עָלָי


וַיְהִי כִּכְלוֹת הָאִישׁ לִשְׁאֹל שְׁאֵלוֹתָיו

וַיַּרְא כִּי עֲנִיתִיו עַל פִּי מִדּוֹתָיו

וּפִתַּחְתִּי מוֹסְרוֹת חִידוֹתָיו

בִּדְבָרִים דְּבוּרִים עַל אָפְנֵיהֶם

אֵין עָיֵף וְאֵין כּוֹשֵׁל בָּהֶם

וְלֹא יָכֹל אוֹתִי לִכְבּשׁ

הוֹדָה וְלֹא בוֹשׁ

וַיֹּאמֶר אֵלָי חֵי אָנִי כִּי אַתָּה פֶלֶא הַזְּמָן

וְיָתֵד תְּקוּעָה למְּלִיצוֹת בְּמָקוֹם נֶאֱמָן

רָאִינוּ בְךָ אֲשֶׁר לֹא רָאִינוּ בְאִישׁ בִּלְעָדֶיךָ

הָאֻמְנָם נוּכַל כַּבְּדֶךָ

וְהַשָּרִים וְהָעָם כָּרְעוּ עַל בִּרְכֵּיהֶם

וַיִפְּלוּ עַל פְּנֵיהֶם

וַיֹּאמְרוּ מִי הֶאֱמִין לִשְׁמֻעָתֵנוּ וּזְרוֹעַ יְיָ עַל מִי נִגְלָתָה

הַיּוֹם הַזֶּה רָאִינוּ בְךָ אֲשֶׁר עַיִן לֹא רָאָתָה

אַתַּה תִּשָּׂא הוֹד הַמְּלִיצָה וַעֲטֶרֶת הַשּׁיר כִּי לְךָ יָאָתָה

וַיִהְיוּ אַנְשֵׁי עֲדָתִי בִּי מִתְפָּאֲרִים

וּמְשׁוֹרְרֵי הַזְּמָן בּוֹרְחִים מִפָּנַי וּמִסְתַּתְּרִים

וְיַחְדָּו הָלְכוּ בַכְּלִמָּה חָרָשֵׁי שִׁירִים


וָאֶשָׂר מְשָׁלִי וָאֹמַר:


אִישׁ רִיב לְבבִי עִם בְּנֵי הַשִּׁיר

נִוָּכְחָה יֹאמַר וְאֵין עֹנֶה

עִמְדוּ עֲמֹדוּ תֹּפְשֵׂי כִנּוֹר

אֶקְרָא בְּיוֹם מִבְחָן וְאֵין מַפְנֶה.
