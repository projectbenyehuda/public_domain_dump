מחברת שמינית

אָמַר עִמָּנוּאֵל הַמְחַבֵּר: הָיִיתִי עִם הַשָּׂר הַמְאֻשָּׁר

בַּעֲבוֹתוֹת אַהֲבָה נִקְשָׁר

נִתְעַלְּסָה בָּאֳהָבִים

וַנְּדַבֵּר רַק מֵהַשִּׁירִים הָעֲרֵבִים

וַיְהִי הַיּוֹם וַיֹּאמֶר אֵלָי שָׁמַעְתִּי אוֹמְרִים

כִּי רַבִּי שְלֹמֹה בֶּן-גַּבִּירוֹל חִבֵּר שִׁיר מִמָּאתַיִם טוּרִים

סַפִּיר גִּזְרָתָם

הֲשָׁמַעְתָּ עֲלֵיהֶם אוֹ רָאִיתָ אוֹתָם

וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר רָאִיתִי וְעָשִׂיתִי כְתַבְנִיתָם וּכְמַתְכֻּנְתָּם

לֹא יֻתַּן סְגוֹר תַּחְתָּם

וַיָּפָג לֵב הַשָּׂר כִּי לֹא הֶאֱמִין

וָאֹמַר חֵי אֲנִי אִם אִשׁ לְהַשְׂמִיל וּלְהֵמִין

וַיֹּאמֶר אֵלַי הַשָּׂר אֲדֹנִי: הַגֶּד-נָא לִי אֵיךְ הָיָה הַדָּבָר בְּנִי

וָאַעַן וָאֹמַר יֵדַע אֲדֹנִי

כִּי הָיָה לִי רֵעַ נֶאֱמָן

חוֹמֵל עָלַי וְרַחֲמָן

עֲנָק עַל גַּרְגְּרוֹתַי עֲנַדְתִּיו

וְעַל לוּחַ לִבִּי חֲקַקְתִּיו

וְאַהֲבַת עוֹלָם אֲהַבְתִּיו

לֹא אֶקְּבֶנּוּ בְשֵׁם פֶּן עָלָיו יִכָּבֵד

וְלֹא אַזְכִּיר שֵׁם עִירוֹ פֶּן אַהֲבָתוֹ מִמֶּנִי תֹאבֵד

גָּבַר בְּשִׂכְלוֹ וּבְצַחוּת הֶגְיוֹנָיו

עַל כֹּל אֲשֶׁר עָבְרוּ לְפָנָיו

הָיִינוּ בִימֵי חָרְפֵּנוּ בִדְמֵי הַחֵשֶׁק מִתְבּוֹסְסִים

וְאַבְנִי נִזְרֵי הָאַהֲבָה עַל אַדְמָתֵנוּ מִתְנוֹסְסִים

גָּבַרְנוּ בְשִׁירוֹתֵינוּ עַל כֹּל אֲשֶׁר קְדָמוּנוּ

אֲרָזִים בְּגַן אֱלֹהִים לֹא עֲמָמוּנוּ

וּבַעֲבוּר הֱיוֹתוֹ אָז בִּימֵי עֲלוּמִים

וַאֲנִי כַּבִּיר מֵאָבִיו יָמִים

וְיוֹמִי פָנָה לַעֲרוֹב

וְהוּא הָיָה אֶל זְמַן הָעֶדְנָה קָרוֹב

הִתְחִיל לָלֶכֶת אַחֲרֵי הַבַּחוּרִים

וְלִרְדֹּף אַחַר הַבְלֵי הַנְּעוּרִים

פַּעַם יָשִׁיר וּפַעַם יְנַגֵּן

וּפַעַם יַחֲזִיק צִנָּה וּמָגֵן

וּפַעַם יַעֲמֹד עִם שָׂרִים זָהָב לָהֶם

וּפַעַם חַרְבּוֹ יְעוֹפֵף עַל פְּנֵיהֶם

עָשָׂה בְחָכְמָתוֹ מַפָּה הֲדוּרָה

לֹא יִשָּׁקֵל כֶּסֶף מְחִירָהּ

הִרְבָּה בָהּ אֹתוֹתָיו

לְהַרְאוֹת הָעַמִּים וְהַשָּׂרִים אֶת אֹדוֹתָיו

וְחָקַק בָּהּ שִׁירֵי רְנָנָיו

הִפְלִיא לַעֲשׂוֹת מִכֹּל אֲשֶׁר הָיוּ לְפָנָיו

וּבְהַכִּירִי מַעֲלָתוֹ

וְאֶת יְקָר תִּפְאֶרֶת גְּדֻלָּתוֹ

וְחָפַצְתִּי לְהַמְצִיא צֳרִי גִיל-עַד אֶל מַחֲלָתוֹ

הוֹכַחְתִּי לְנַפְשִׁי בַתְּחִלָּה

וְהֶעֱמַדְתִּיו עַל מַעלָתוֹ הַמְּהֻלָּלָה

וְעוֹרַרְתִּי אֶת נַפְשׁוֹתֵינוּ מִמַּעֲלָהּ

עַד אֲשֶׁר אַחֲרֵי שׁוּבוֹ נִחַם וְהתְאַפָּק

וְאַחֲרֵי הִוָּדְעוֹ עַל יָרֵךְ סָפָק

וְהִנּוֹ הַיּוֹם רֹאשׁ לַנְּזִירִים

וְעֹזֵר לְכָל הַנֶּעֱזָרִים

וּבִלְעָדָיו בַּחֲסִידוּת אִישׁ אֶת יָדוֹ וְאֶת רַגְלוֹ לֹא יָרִים

וְזֹאת הִיא שִׁירָתִי

הַשְּׁלוּחָה אֵלָיו מֵאִתִּי:

אֲבִי חָכְמָה וּמוֹעֵצָה וְנִזְרָהּ

נְזִיר אַחִים פְּאֵר תּוֹרָה וְכִתְרָהּ

נְזִיר אַחִים לְךָ אֶקֹּד וְאֶכְרַע

בְּרֹב אֵימָה בְּנֶפֶשׁ בָּךְ קְשׁוּרָה

וְאֶת חֶלְקִי בְעוֹלָמִי וְחַבְלִי

וְאֵין לִי בַזְּמָן מִמְּךָ תְמוּרָה

וְהֵא לִבִּי כְּקֹטֶב הַצְּפוֹנִי

לְךָ קַיָּם וְנַפְשִׁי לָךְ מְכוּרָה

יְדַעְתִּיךָ אֲבִי חָכְמָה וּמוּסָר

וְלִנְדִיבוּת הֲלֹא רֹאשׁ וַעֲטָרָה

יְלָדַתְךָ תְעוּדָת אֵל וְטֶרֶם

יְלָדַתְךָ אֱמֶת הָיְתָה עֲקָרָה

וְחָכְמָתָךְ בְּדֶרֶךְ אוֹת וּמוֹפֵת

וְחָכְמַת בִּלְתְּךָ דֶּרֶךְ סְבָרָה

וּמַהְלָלָךְ אֲמִתִּיִּי וּמַהְלַל

נְדִיבֵי עָם בְּדֶרֶךְ הַעֲבָרָה

וְאַתְּ קוֹלְךָ כְּקוֹל מַלְאַךְ אֱלֹהִים

וְקוֹל זוּלָתְךָ לִי קוֹל הֲבָרָה

וְנָטַעְתָּ נְטַע אַהְבָה בְּתֵבֵל

וְכָל-אִישׁ בִּלְתְּךָ עָקֹר עֲקָרָהּ

וְאִישׁ זוּלָתְךָ בִּשְׂדֵי תְבוּנוֹת

כְּעָנִי הַמְהַפֵּךְ בַּחֲרָרָה

לְאוֹר חִצְּךָ יְהַלֵּךְ כָּל מְשׁוֹרֵר

וְרֶשֶׁת בִּלְתְּךָ חִנָּם מְזֹרָה

לְךָ יִשְׁתַּחֲווּ הֵם מִמְּקוֹמָם

וְיָשִׂימוּ בְאַפֶּךָ קְטוֹרָה

יְחַיֵּבוּ נְבוֹנֵי עָם וְתִמְצָא

בְּחָכְמָתָךְ בְחִיּוּבָם סְתִירָה

וּמִי יַעֲמֹד פְּנֵי אֵימָתְךָ דּוֹד

בְּיוֹם צֵאת אֵשׁ וְעָשָׁן מִנְּחִירָה

וְלוּ אַתָּה בְּעֶת-בֹּא אָז עֲמָלֵק

זְעוּם הָאֵל לְצַנֵּן הַקְּדֵרָה

לְךָ אָמְרוּ בְּחַר לָנוּ אֲנָשִׁים

הֱיֵה רֹעֶה עֲדַת אֵל חַי וְשָׂרָהּ

וְלוּ אַתָּה בְּמִלְחֶמֶת מְדָנִים

יְצַוֶּה אֵל לְךָ רֵד אַתְּ וּפֻרָה

חֲדַל גִּדְעוֹן חֲדַל יִפְתָּח וּבִן נוּן

וְחִדְלוּ כָּל אֱנוֹשׁ חֹגֵר חֲגֹרָה

כְּבָר קָם מִבְּנֵי אֵל שַׂר וְנָגִיד

צְבִי עֵדָה וְגַחַלְתָּהּ וְנֵרָהּ

גְּאוֹן הַדּוֹר סְגֻלַּת כָּל סְגֻלָּה

מְשׂוֹשׂ אֶרֶץ הֲמוֹנָהּ וֶהֲדָרָהּ

וּמִי אָנִי וּמִי בֵיתִי צְבִי חֵן

לְחַבֵּר שִׁיר וְיַד שִׂכְלִי קְצֵרָה

וְאָמְנָה דּוֹד בְּמֶתֶק וַעֲנָוָה

בְּלֵב נִשְׁבָּר בְּמוֹ תוֹדָה וְזִמְרָה

זְמִרוֹתַי יְבֹאוּךָ וְאַךְ לֹא

לְהוֹכִיחַ אֲבָל דֶּרֶךְ זְכִירָה

יְבֹאוּן לָךְ כְּפוּפֵי רֹאשׁ כְּאַגְמֹן

בְּמֶתֶק פֶּה וְלֹא דֶרֶךְ שְׂרָרָה

יְבַקְשׁוּ אֶת מְחִילָתְךָ בְּרֵאשִׁית

וְאַחַר יִפְתְּחוּ שָׂפָה בְרוּרָה

וְיוֹכִיחוּ לְנַפְשִׁי בַּתְּחִלָּה

וְשִׂכְלִי בַּאֲשֶׁר שָׁגַג וְהִמְרָה

בְּחַיֶּיךָ צְבִי חֵן אֵין לְנַפְשִׁי

חֲסִידוּת כַּחֲצִי גַּרְגַּר שְׂעֹרָה

וְהַחֵשֶׁק תְּמוֹל הָיָה כְעַלְמָה

וְהָיִיתִי אֲנִי בַּעַל נְעוּרָהּ

וְדִמִּיתִי הֱיוֹת חֵשֶׁק לְחֹמֶר

וְאָנִי אֶהְיֶה אֵלַיו לְצוּרָה

וְעֵרֶךְ חִשְׁקְךָ הָיָה בְעֵינַי

לְמוּל חִשְׁקִי כְּאֶחָד לַעֲשָׂרָה

וְהָיְתָה לִי עֲטֶרֶת הָענָוָה

וְחֵשֶׁק מֵעֲלֵי רֹאשִׁי הֱסִירָהּ

וְלוּלֵי כִּי אֲנִי בוֹשׁ מֵאֱלֹהַי

וּפֶן אֶהְיֶה לְעַם קֹדֶשׁ לְזָרָא

אֲזַי אֶבְנֶה לְיַעְלוֹת מִזְבְּחוֹת חֵטְא

וְנָטָעְתִּי לְכָל יַעְלָה אֲשֵׁרָה

וְכָל יַעְלָה עֲפַר רַגְלָהּ אֲלַחֵךְ

וּבַמָּוֶת אֲחוֹנֵן אֶת עֲפָרָהּ

אֲבָל אֶתְחַסְּדָה בִּפְנֵי אֲנָשִׁים

שְׁלֵמִים בַּעֲלֵי מוּסָר וְתוֹרָה

וְאֶתְנַכֵּר וְלֹא אֶהְיֶה כְחֹשֵׁק

וְאֶתְחַסֵּד וְלֹא אֶשְׁגֶּה בְזָרָה

וְאֶתְמַרְמַר בְּלֵב נִשְׁבָּר בְּיַעַן

מְגִלַּת חֹשְׁקִים שַׂמְתִּי לְשִׁירָה

וּמִי יִתֵּן וְיָכֹלְתִּי לְכַבּוֹת

אֲנִי נֵרָהּ וְגַחַלְתָּהּ וְזִכְרָהּ

וְנֶחְרַטְתִּי חֲרָטַת הַבְּתוּלָה

אֲשֶׁר הֶעְרָה בְמִקְרֶה אִישׁ מְקֹרָהּ

אֲבָל מָה אוֹמְרָה וּבְנֵי עֲדָתִי

אֱמֶת לֹא יִשְׁקְלוּ כֶסֶף מְחִירָהּ

וְשִׁגְגָתָךְ צְבִי הַחֵן שְׁגָגָה

בְּעֵינֵי הַחֲכָמִים לֹא זְעִירָה

אֲבָל יֹאמְרוּ בְנֵי מוּסָר והַשְׂכֵּל

שְׁגָגָה בַּעֲצַבִּים הִיא חֲבוּרָה

כְּמַפָּה מַעֲשֵׂה רִקְמָה עֲשׂוּיִם

אֲבִידָע וַחֲנֹךְ וּבְנֵי קְטוּרָה

בְּגַן עֶדְנָם דְּשֵׁנִים רַעֲנַנִּים

בְּצֵל הָדָס גְּבֶרֶת עִם גְּבִירָהּ

וְשָׁמָּה נַהֲרוֹת מֹשְׁכוֹת גְּבִירוֹת

וְכָל עָפְרָה נְעִימָה וַהֲדוּרָה

וְסָבִיב לַיְאוֹר כָּל שַׂר וְנָגִיד

בְּרִשְׁתָּם יִמְשְׁכוּ אִישׁ אִישׁ נְעָרָה.

וְגַם אַתְּ שָׁם צְבִי הַחֵן כְּמוֹהֶם

בְּרִשְׁתָּךְ תַּעֲלֶה לָּךְ הַגְּבִירָה

רְאִיתִיהָ וְשָׂחַקְתִּי בְּמוֹ-פִי

וְהָיוּ כּוֹכְבֵי נִשְׁפָּהּ אֲפֵלִים

וְאַתָּה דּוֹד לְבַד אוֹרָהּ וְשַׁחְרָהּ

וְהָיוּ כָּל בְּנֵי תֵבֵל שְׁכֻרִים

בְּיֵין תּוֹעָה וְאַתְּ קָדְקֹד נְזִירָהּ

קְרַב אַתָּה נְשִׂיא הַדּוֹר מְשִׁיחַ

אֱלֹהֵי יַעֲקֹב וּשְׁכֹן חֲצֵרָהּ

וְהִתְעַנַּג בְּצוּף עֶדְנָהּ וְדִשְׁנָהּ

סְחַר סַחְרָהּ וְתִקְצֹר אֶת קְצִירָהּ

יְפוּצוּן חוּץ עֲיָנוֹת שִׂכְלְךָ דּוֹד

רְאֵה עַם אֵל אֲשֶׁר יִשְׁכֹּן בְּמָרָה

לְךָ יֵאוֹת הֱיוֹתְךָ אוֹת לְרִבּוֹת

וְלִהְיוֹת אַתְּ פְּדוּת חָכְמָה וְכָפְרָהּ

וְאַתָּה וַאֲשֶׁר הֵמָּה כְגִילְךָ

בְּעוֹלָם זֶה כְּנִרוֹת הַמְּנוֹרָה

נְשִׂיא הַדּוֹר רְאֵה כָּל הַיְצוּרִים

לְמוּל עֶרְכָּךְ כְּצֹאן קֵדָר וּבָצְרָה

וְלוּ יִשְׁתַּתְּפוּ עִמְּךָ בְּשֶׂכֶל

כְּמוֹ הִשְׁתַּתְּפוּ לָךְ בַּיְצִירָה

רְאֵה אַנְשֵׁי זְמַנְּךָ לָךְ נְתֻנִים

וְכָל נֶפֶשׁ לְמִשְמַעְתָּךְ זְהִירָה

רְאֵה כִּי שָׁחֲחוּ תַחְתָּךְ יְצוּרִים

וְהָיִיתָ לְעֵדַת אַל כְּפִירָהּ

וְאַתָּה לַזְּמָן נֵר מַעֲרָבִי

וְלָעֵדָה הֲלֹא שִׁמְשָׁהּ וְסַהְרָהּ

רְעֵה עֵדָה פְּאֵר הוֹדָהּ וְשִׂים אֶת

כְּסִילֶיהָ בְּמוֹ מַעְלַת בְּקָרָהּ

שְׁלֹף חֶרֶב גְּבוּרָה בֶּאֱוִילוּת

וּבִכְסִילוּת וְהַחְרֵם כָּל זְכוּרָהּ

וְשַׁלַּח חֵץ בְּאִישׁ לֹחֵץ וְתוֹרֶה

לְמַטָּרָה בְּלֵב נֶפֶשׁ יְהִירָה

בְּרוֹק בָּרָק וְכוֹנֵן חֵץ גְּעָרָה

וְשַׁלַּח חֵץ וּבָרָק מִמְּרֹרָה

וְכָל נֶפֶשׁ בְּמַעְלָתָהּ תּכוֹנֵן

וְתַבְדִּיל הַשְּׁחוֹרָה מִצְּחוֹרָה

וְשׁוּר יִפְעָתָךְ וּזְכֹר בְּאֵרָהּ

וְצוּר בּוֹרְאָהּ וּמַטְרִיפָהּ וּבֹרָהּ

בְּטֶרֶם יֶחֱשַׁךְ שִׁמְשָׁהּ וְאוֹרָהּ

וְלֶדֶת חֹק וּבִיאַת הַצְּפִירָה

רְאֵה עוֹלָם כְּמוֹ אוּלָם וְתֵבֵל

דְּמוּת בִּקְעָה וְנחְנוּ כַּחֲצִירָהּ

בְּבָקְרֵנוּ הֲכִי נָצִיץ וְנַחְלֹף

וְלָעֶרֶב הֲכִי נִשְׁכֹּן חֲרֵרָה

נְזִיר אַחִים וּמִי רָאוּי לְעֵדֶן

וְעֵדֶן גַּן לְמִי רָאוּי לְדִירָה

וּמִי יִבְנֶה בְּעוֹלַם הַנְּשָׁמוֹת

לְשִׁבְתּוֹ בֵּית זְבֻל אַרְמוֹן וְטִירָה

וְתוֹרַת אֵל לְמִי תִכּוֹן וּמִי זֶה

אֲשֶׁר יֵאוֹת לְעָבְדָהּ אוֹ לְשָׁמְרָהּ

וּמִי יַחְקֹר פְּעֻלַּת אֵל בְּשִׂכְלוֹ

וּמִי הָאִישׁ אֲשֶׁר יֵאוֹת לְחָקְרָהּ

לְבַד אַתָּה אֲבִי חָכְמָה וְנִזְרָהּ

צְבִי עֶדְיָהּ וּמִבְחוֹר רֹאשׁ אֲמִירָהּ

הֲנַנִּיחַ לְעוֹלַם הַנְּשָׁמוֹת

לְכָל גִּבֵּן וְאִישׁ בַּעַל חֲטֹרָה

וְיַרְוִיחוֹ בְסִכְלוּתָם לְעֵדֶן

וְנַרְוִיחַ בְּשִׂכְלֵנוּ מְעָרָה

הֲטוֹב כִּי תַעֲדֹפְנָה עַל שְׁנֵינוּ

בְּטוּב הַמַּעשִׂים פּוּעָה וְשִׁפְרָה

הֲבִסְגֻלּוֹת וְכֹחוֹת בֶּהֱמִיּוֹת

לְנֶפֶשׁ אִישׁ תְּהִי מַעְלָה וְיִתְרָה

רְאֵה הַתַּרְנְגוֹל וּשְׁכַח נְדִיבוּת

נְדִיבֵי עָם רְאֵה קַלּוּת לְבִכְרָה

רְאֵה גָּמָל אֲשֶׁר יוּלַד לְעָמָל

רְאֵה חַמּוֹר לְכָל מַשָּׂא וְתִגְרָה

וְעַכָּבִישׁ רְאֵה קַלּוּת פְּעָלָיו

וְרִשְׁתּוֹ לַזְּבוּב פָּרַשׂ וְדִקְרָה

וְעֵת אָרְגוֹ עֲגֻלָּה תוֹךְ עֲגֻלָּה

כְּאִלּוּ בַמְּחוּגָה הוּא תְאָרָהּ

וְלִבְנוֹת הֵיכְלֵי עֹנֶג וּבָתִּים

מְלֵאִים טוּב רְאֵה בִּנְיַן דְּבֹרָה

וְלַחְרִיצוּת רְאֵה דּוֹד הַנְּמָלָה

חֲרוּצַת יָד וְאַף אִם הִיא צְעִירָה

ְולַשַּׁחַל גְּבוּרַת לֵב וְאֹמֶץ

וְיָדוֹ הִיא כְּיַד עֵשָׂו שְׂעִירָה

וְלַמִּשְׁגַּל וְאֶל מַאְכָל וּמִשְׁתֶּה

רְאֵה חַזִּיר רְאֵה עֵגֶל וּפָרָה

וְהִתְיַחֵד אֱנוֹשׁ בִּלְבָד בְּנֶפֶשׁ

מְדַבֶּרֶת מְטֹהָרָה וּבָרָה

לְפָאֵר לֶאֱלֹהֶיהָ בְּכָל לֵב

וְלִבְחֹן סוֹד פְּעֻלַּת אֵל יְצָרָהּ

וְנֶפֶשׁ זֹאת מְזוֹנָהּ הַחֲסִידוּת

בְּיַעַן הִיא לְבַד רִפְאוּת לְשָׁדָּהּ

לְפִי שִׂכְלָהּ תְּהֻלַּל אוֹ תְחֻלַּל

וּמַעְלָתָהּ לְפִי צִדְקָהּ וְיָשְׁרָהּ

וְלֹא תוֹחִיל לְבַד אֶל הָאֱלֹהִים

לְרַפֹּאת מַחֲצָהּ מֶנָּהּ וְשִׁבְרָהּ

וְאִם הִיא בַּזְּמָן תִּגְעַל וְתִתְעַל

עֲלֵי אֶרֶץ הֲתִירָא בַהֲמִירָהּ

וְאֵיךְ תִּרְעַב אֲשֶׁר עַל עָב רְכוּבָהּ

וְתַשְׁלִיךְ יָהֳבָהּ עַל אֵל וְשִׂבְרָהּ

וְנֶפֶשׁ הַמְקַצֶּרֶת וְחֹטֵאת

בְּהַגִּיעַ זְמָן קִצָּהּ וְתֹרָהּ

בְּנֵי אֵל יַענוּ הֵם יֹאמְרוּ לָהּ

הֲזֹאת הִיא נָעֳמִי אֵל הֶאֱמִירָהּ

הֲזֹאת הִיא נָעֳמִי יָצְאָה לְלַקֵּט

בְּכַרְמֵי בִין וְאֵיךְ שָׁבָה נְעוּרָה

לְלַקֵּט יָצְאָה זֹאת מַה בְּיָדָהּ

וּמִלֶּדֶת תְּעוּדָה הִיא עֲצֻרָה

תְּנוּ לָהּ מִפְּרִי יַד מַעֲשֶׂיהָ

בְּמִשְׁקָלָם בְּמִדָּה בַּמְּשׂוֹרָה

רְאוּ מַה בֵּין הֱיוֹתָהּ בַּעֲרָבוֹת

וּבֵין לִהְיוֹת בְּגוּשׁ עָפָר מְדוֹרָהּ

וְאֵיכָה דּוֹד בְּדָבָר נעֲלֵינוּ

תְּהִי נֶפֶשׁ דְּבַר יֹצְרָהּ מְמִירָה

וְנָסֹב עַל דְּבַר חֵשֶׁק כְּעָצֵל

בְּמִטָּתוֹ וְהַדֶּלֶת בְּצִירָהּ

וְהַחֵשֶׁק וְהַשֵּׂכֶל הֲפָכִים

וּמִי יוּכַל לְאָוֶן וַעֲצָרָה

וְלַשֵּׂכֶל נְעִימוּת וַחֲשִׁיבוּת

וְהוֹד מַלְכוּת וּמִשְׁפַּט הַבְּכוֹרָה

וְאֵין תִּקְוָה וְאַחְרִית רַק לְהַכְרִית

לְנֶפֶשׁ תַּעֲבִיד שִׂכְלָהּ לְיִצְרָהּ

הֲיִמְצָא חֵן אֱנוֹשׁ אָנוּשׁ בְּנֶפֶשׁ

אֲשֶׁר יַלְדוּת וְיֵין חֵשֶׁק עֲבָרָהּ

וְהַנֶּפֶשׁ וְהַחֵשֶׁק כְּצָרוֹת

וְאֵין בֵּינָם לְעוֹלָמִים פְּשָׁרָה

וְאֵין טַעְנָה לְמִתְעַלֵּם וְרֶוַח

לְמִתְרַשֵּׁל וְעָצֵל בַּחֲקִירָה

אֲבָל חֻיַּב אֱנוֹשׁ לַחְקֹר תְּבוּנָה

וְלָבֹא עַד תְּכוּנָתָהּ וְחִקְרָהּ

וְנִתְבַּע אִישׁ כְּפִי כֹחוֹ וְשִׂכְלוֹ

וְהַכִּירוֹ לְמַעְלָתָהּ וְיָקְרָהּ

וְאַתְּ שָׁקוּל לְפִי שִׂכְלָךְ כְּרִבּוֹא

וְלָךְ חָכְמָה וְאֵיךְ תִּגְרַע שְׁאֵרָהּ

קְחָה שִׁירִי אֲסִיר חִשְׁקָךְ וְנַפְשׁוֹ

נְתוּנָה לָךְ וּבָהּ תִּמְשֹׁל לְמָכְרָהּ

וּבָאָה בַבְּרִית אִתְּךָ וְאָכֵן

בְּכָל דּוֹד בִּלְתְּךָ בָּגְדָה וְקָשְׁרָה

וּמִיּוֹם הַנְּדֹד הִפְרִיד שְׁנֵינוּ

לְחִשְׁקָךְ הַצְּבִי גֹּלָה וְסוּרָה

וְחָשַׁךְ לַנְּדֹד אוֹרָהּ וְזָהְרָהּ

וּמִיּוֹם הַנְּדֹד אוֹר לֹא זְכָרָהּ

מְחַכֶּה אֶל רְאוֹת יִפְעָתְךָ דּוֹד

וְלִבָּהּ בָּהּ כְלֵב אִשָּׁה מְצֵרָה

רְאֵה שִׁירִי וְזָכְרֵנִי לְטוֹבָה

נְזִיר אַחִים פְּאֵר תּוֹרָה וְכִתְרָהּ.

וַיִּתְמַהּ הַשָּׂר עַל יְפִי הַשִּׁירָה הַמְהֻלָּלָה

וַיִּשָּׁבַע כִּי אֵין עַל עָפָר מָשְׁלָהּ

וַיֹּאמַר מַה נִכְבָּד וּמְאֻשָּׁר הָאִישׁ הַשְּׁלוּחִים אלָיו אֵלּוּ הַחֲרוּזוֹת

וְתָמֵהַּ אֲנִי אִם יֵשׁ אִישׁ בַּדּוֹר רָאוּי אֶל שִׁירָה כָזֹאת

וְיָרֵא אֲנִי פֶּן בִּהְיוֹתְךָ אֲהוּבוֹ

רָחַקְתָּ מִן הָאֱמֶת וְנָטיתָ מִנְּתִיבוֹ

לְקַיֵּם מַאֲמַר הֶחָכָם מֵיטַב הַשִּׁיר כְּזָבוֹ

וַאֲשֶׁר חֻיַּבְנוּ אֲנַחְנוּ קְהַל הַמְשֹׁרֲרִים

לְהַפְלִיג בְּשֶׁבַח וּבִגְנוּת בַּשִּׁירִים

אֵין לְהַרְחִיק לְגַמְרֵי הָאֱמֶת

עַד שֶׁנֹּאמַר מִן הַמֵּת חַי וּמִן הַחַי מֵת

אָמְנָם חֻיַּב הַמְשׁוֹרֵר לָשִׂים כָּל אִישׁ בְּמַעֲלָתוֹ

וּלְהַפְלִיג קְצָת עַל הָאֱמֶת בִּמְלִיצָתוֹ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

צְבִי חֵן שִׂים מְלִיצָתָךְ בְּנוּיָה

לְפִי חָכְמָה ולֹא עַל פִּי רְגִילוּת

וְשִׂים כָּל אִישׁ בְּמַעְלָתוֹ בְּשִׁירָךְ

כְּרוּב בִּיקָר וְלַשָּׂטָן בְּזֻלּוּת

תְּנָה צִיץ הוֹד אֱלֵי נָדִיב וְשׁוּעַ

וְתָאֵר זֵד וְאִישׁ נָבָל בְּנִבְלוּת

וְהַשַּׁחַל תְּתָאֵר בַּגְּבוּרָה

וְהָעוֹרֵב וְהַכֶּלֶב בְּכִילוּת

וְדַע כִּי אִישׁ יְתָאֵר הַגְּבוּרָה

בְּרִפְיוֹן יָד וְהַחָכְמָה בְּסִכְלוּת

לְדִמְיוֹן אִישׁ יְתָאֵר רוּם שְׁחָקִים

בְּמוֹ שִׁפְלוּת וְסִבּוּבָם בְּעַצְלוּת.

וָאַעַן וָאֹמַר יֵדַע הַשָּׂר הַנִּכְבָּד

כִּי הָאִישׁ אֲשֶׁר פֵּאַרְתִּי בְשִׁירִי מֶלֶךְ לְמַעֲלָתוֹ נֶעֱבָד

וְלֹא עָבְַרתִּי הַגְּבוּל

וְלוּ הֶעֱלִיתִיו בְּשִׁירִי עַל זְבוּל

וְלוּ רָאִיתָ שִׁירוֹתַי אֲשֶׁר שָׁלַחְתִּי לוֹ בִימֵי חָרְפִּי

אָנֹכִי הַדַּל בְּאַלְפִּי

נִפְלֵאתָ עַל עֹצֶם חִשְׁקֵנוּ

וְיָדַעְתָּ כִּי לֹא שֶׁקֶר נִסְכֵּנוּ

וּבִפְרָט הַשִּׁירָה הַהֲדוּרָה

אֲשֶׁר לֹא יִשָּׁקֵל כֶּסֶף מְחִירָהּ

יִתְּנוּהָ מְלִיצֵי הַזְּמָן עַל רֹאשָׁם עֲטָרָה

אֲשֶׁר שָׁלַחְתִּי לוֹ הָעֵת הִרְחִיק נְדוֹדוֹ

וְהִגִּיעַנִי עַד שַׁעֲרֵי מָוֶת בְּפֵרוּדוֹ

וּבְחַבְלֵי הַשִּׁירָה הַהִיא מְשַׁכְתִּיו

עַד אֲשֶׁר אֶל בֵּית אִמִּי וְאֶל חֲדַר הוֹרָתִי הֲבֵאתִיו

וַיֹּאמֶר הַשָּׂר בְּחַסְדֶךָ

הַרְאֵנִי נָא אֶת כְּבוֹדֶךָ.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

רָאִיתִי הַשֶּׁמֶשׁ יִזְרַח

בִּתְקוּפָתוֹ וּבְטָהְרָתוֹ

עַל חוּג שָׁמַיִם יִתְהַלָּךְ

יָרוּץ אֹרַח בִּמְסִבָּתוֹ

וּתְקוּפָתוֹ עַד לִקְצוֹת אֶ–

רֶץ אֵין נִסְתָּר מֵחַמָּתוֹ

וּמְאוֹרֵי אוֹר בַּשָּׁמַיִם

קֹדְרִים מוּל זֹהַר יִפְעָתוֹ

יִפרֹשׂ כַּנְפֵי אוֹר יִקָּחֵם

יִשָּׂאֵמוֹ עַל אֶבְרָתוֹ

רָאִיתִי עָמַדְתִּי מַרְעִיד

לִבִּי יִתַּר מֵחֶרְדָּתוֹ

כִי נֶעְכַּרְתִּי עַד זָכַרְתִּי

חֶמְדַּת לִבִּי וּסְגֻלָּתוֹ

נֶטַע נַעְמָן צִנְצֶנֶת מָן

לָקַח מִמֶּנִּי עֶמְדָּתוֹ

כִּי הוּא שֶׁמֶשׁ הַחָכְמָה הוּא

הַזּוֹרֵחַ בִּגְבֻרָתוֹ

הַמֵּסִיר שָׂפָה לִמְלִיצִים

שִׂכֵּל שִׁירוֹ מִדֵּי צֵאתוֹ

כָּל לָשׁוֹן תָּקוּם לַמִּשְׁפָּט

אִתּוֹ תַּרְשִׁיעַ צִדְקָתוֹ

הוּא אוֹר לִבִּי הוּא שִׂמְחָתוֹ

הוּא חַגּוֹ חָדְשׁוֹ שַׁבַּתּוֹ

הוּא יַרְדֵּן הַשֵּׂכֶל כִּי בוֹ

יִטְהַר כָּל לֵב מִצָּרַעְתּוֹ

הוּא עַצְמוּת הַשֵּׂכֶל עֶלְיוֹן

הוּא אֵיכוּתוֹ הוּא מָהוּתוֹ

הוּא חֹמֶר לוֹ הוּא פוֹעֵל לוֹ

הוּא צוּרָתוֹ הוּא תַכְלִיתוֹ

מִי זֶה יֶחְשֹׂף סוֹד הֶגְיוֹנוֹ

מִי זֶה יִמְצָא אֶת חִידָתוֹ

אִלּוּ נוֹדַע בִּבְנֵי אֵלִים

אָז יֶחְפָּצוּן אֶת קִרְבָתוֹ

אִלּוּ הִטָּה יַד שִׂכְלוֹ עַל

יָם הַבִּין נָס מִגַּעְרָתוֹ

אִלּוּ פָגַע בִּסְנֵה מֹשֶׁה

אֻכַּל חֻבַּל מִלַּבָּתוֹ

אִלּוּ פָגְעוּ בוֹ נִשְׂרָפוּ

שַׁדְרַךְ מֵישַׁךְ מִגַּחַלְתּוֹ

הָהּ לִבְרָקָיו כִּי הֵאִירוּ

עַד כִּי גִלּוּ לִי חֶמְדָּתוֹ

כִּי הִשְׁאִירוּ אֹתִי אַחְרֵי

נָסְעוֹ כּוֹאֵב לִנְסִיעָתוֹ

רָאִיתִי הַחֵשֶׁק כֵּלַי

כֵּלָיו רָעִים מִן נִדְבָתוֹ

רָאִיתִי עֵת פָּתַח שָׁמָיו

מַרְאוֹת שֵׂכֶל וּתְמוּנָתוֹ

רָאִיתִי כָּל שַׂר בּוֹ נִקְשָׁר

שָׂרִים סָרִים אֶל מִשְׁמַעְתּוֹ

רָאִיתִי כִּי כָל שָׂרִים יִקְ–

חוּ בִכְלֵיהֶם מִזִּמְרָתוֹ

רָאִיתִי הֵימָן עִם כַּלְכֹּל

נִלְאוּ כַלְכֵּל סוֹד דִּבְרָתוֹ

רָאִיתִי אָסָף לֹא יָסֵף

דַּבֵּר עֵת שָׁמַע אִמְרָתוֹ

רָאִיתִי כָּל לָשׁוֹן דָּבֵק

אֶל חֵךְ לֹחֵךְ צוּף יַערָתוֹ

רָאִיתִי דָּוִד וּשְׁלֹמֹה

כִּי יִתמָהוּ עַל חָכְמָתוֹ

רָאיתָ חִירָם וּבְצַלְאֵל

עֵירֻמִּם הֵם מֵעָרְמָתוֹ

רָאִיתִי כִּי אַלְמֻדְנַבִּי

לֹא הִשִּׂיג אַבְקַת שִׁירָתוֹ

לוּ חֻבַּר מִכָּל הַלִּבּוֹת

כִּי כֻלָּם בָּאִים בִּבְרִיתוֹ

חִשְׁקוֹ הָאֵשׁ וּמְשָׁלָיו הָ–

עֵצִים לִבִּי שֵׂה עֹלָתוֹ

חָפֵץ הַשֵּׂכֶל הָעֶלְיוֹן

הַרְאוֹת לָנוּ אֶת עַצְמוּתוֹ

יָרַד מִכִּסְאוֹ בָּא אֶל בֵּית

אִמּוֹ אֶל חֶדֶר הוֹרָתוֹ

וַיָּבֹא אֵלֶיהָ וַתַּ–

הַר לוֹ וַיּוֹלֶד בִּדְמוּתוֹ

עַד כִּי הַשֵּׂכֶל הַפֹּעֵל

עַל יָדוֹ נָתַן טַבַּעְתּוֹ

לָתֵת תַּבְנִית לַחְתֹּם תָּכְנִית

לָתֵת אֶל כָּל חַי צוּרָתוֹ

קוּמִי שִׁירָתִי יָפָתִי

יוֹנַתִי וּלְכִי וּמְצָאתוֹ

הִלְכִי זִמרָתִי סֹבִּי עִיר

אוּלַי תִּקָּרִי לִקְרָאתוֹ

סֹבִּי צָפוֹן בּוֹאִי תֵימָן

בּוֹאִי סוּכּוֹ וֹמְעוֹנָתוֹ

גַּלִּי שׁוּק וּנְהָרוֹת עִבְרִי

עַד בּוֹאֵךְ חַדְרֵי משְׂכִּיתוֹ

דִּרְשִׁי שִׁירָתִי בִּמְקוֹם הַ–

חֵשֶׁק וּסְבִיבוֹת גִּבְעָתוֹ

אוּלַי נֶחְבָּא אֶל הַכֵּלִים

וּבְלֵב חֹשְׁקִים שָׂם תַּחְנֹתוֹ

אִם שָׂם בֵּין כּוֹכָבִים קנּוֹ

תָּצוּרִי עָלָיו עַד רִדְתּוֹ

הֵיטִיבִי נַגֵּן הַרְבִּי שִׁיר

אוּלַי יַעַן עֵת שָׁמְעָתוֹ

הָרִימִי בַכּוֹחַ קוֹלֵךְ

אוּלַי יֵעוֹר מִשִּׁכְּחָתוֹ

אלַ תִּדֹּמִּי אוּלַי יָשֵׁן

הוּא יִקַּץ מִתַּרְדֵּמָתוֹ

שִׁירָתִי מַה תּגִּידִי לוֹ

שֶׁחָלִיתִי עַל אַהְבָתוֹ

לָמָּה לָקַח לִבִּי מִתּוֹךְ

קִרְבִּי אִם נִכְסַף אֶל בֵּיתוֹ

אַחְלַי יִזְכָּר-לוֹ עֵת הָיָה

קִיר לִבִּי רֵאשִׁית מַמְלַכְתּוֹ

יָשִׂים דִּמְעָתִי עַל נֹדוֹ

אֶל תּוֹךְ נֹאדוֹ וּבְסִפְרָתוֹ

אִם יִמְצָא אַהְבָתִי חוֹמָה

יִבְנֶה עָלֶיהָ טִירָתוֹ

אִם דֶּלֶת הִיא יָצוּר עָלֶי–

הָ לוּחַ אֶרֶז בַּחְנֹתוֹ

אִם יִחְפֹּץ לַעְלוֹת שָׁמַיְמָה

יַשְׁלִיךְ עָלַי אֶת אַדַּרְתּוֹ

אִם לָנוּחַ תּוֹךְ לֵב חֹשֵׁק

קִיר לִבִּי הִנֵּה מִטָּתוֹ

אִם יִתְמַהְמַהּ מַה לִּי עוֹד אִם

לֹא יִשְׁלַח לִי אֶת בִּרְכָתוֹ

כָּזֹאת אִמְרִי אֵלָיו עַד אָן

דּוֹדְךָ יִזְעַק מֵאַנְחָתוֹ

עֶבֶד נִמְכַּר לָךְ לִצְמִתֻת

הַגִּידָה לִּי מַה מַּשְׂכֻּרְתּוֹ

עַד מָתַי תִּתְאַכְזַר עָלָיו

עַד אָן תִּבְזֶה אֶת עַנּוֹתוֹ

עַד מָתַי הָרוּג חִשְׁקֶךָ

עַל הָעֵץ תָּלִין נִבְלָתוֹ

לָמָּה הִכִּיתָ עַבְדֶּךָ

עִם אַבְנֵי פֵרוּד וּסְקַלְתּוֹ

לָמָּה שָׁלַחְתָּ אֵשׁ חִשְׁקָךְ

עַל לֵב מֹדַעְתָּךְ וּשְׂרַפְתּוֹ

לָמָּה הֵבֵאתָ אִישׁ סוֹדְךָ

תּוֹךְ יָם הַחֵשֶׁק וּטְבַעתּוֹ

כָּזֹאת אִמְרִי אֵלָיו דּוֹדְךָ

תָּמִיד נִצָּב עַל מִשְׁמַרְתּוֹ

נֹשֵׂא עַל כָּתֵף מַהְלָלְךָ

וּנְדִיבוּתְךָ עַל דָּבֶּשְׁתּוֹ

מִתְיַצֵּב וּמְצַפֶּה לִרְאוֹת

מַה זֶּה תָשִׁיב עַל תּוֹכַחְתּוֹ

אִם חָטָא לָךְ לָקַח מִיָּדְ–

ךָ כִפְלַיִם מֵחַטָּאתוֹ

אִם לֹא יָשׁוּב אֵלָיו חִנְּךָ

אָנָה יוֹלִיךְ אֶת חֶרְפָּתוֹ

בַּמֶּה יִתְרַצֶּה לָשׁוּב אֶל

חִנְךָ הַשָּׂר לִפְנֵי מוֹתוֹ

אִם בִּבְכִי וּדְמָעוֹת הִנֵּה

כִּי הוֹבִישׁ עַיִשׁ בִּבְכִיתוֹ

גַּם שָׂכַר עָלֶיךָ גִיחוֹן

וּפְרָת לִהְיוֹת עִם דִּמְעָתוֹ

כָּל מֵרֵעָיו לֹא-יָכֹלוּ

הָקִים אוֹתוֹ וּלְהַבְרוֹתוֹ

וַיְמָאֵן הִתְנַחֵם עַד כִּי

תֵּפֶן הַשָּׂר אֶל מִנְחָתוֹ

שִׁירָתִי אִם יִפְגָּשֵׁךְ אִישׁ

מַה תָּשִׁיבִי לִשְׁאֵלָתוֹ

הִנֵּה מִנְחַת עִמָּנוּאֵל

לִשְׁלֹמֹה הַשָּׂר מֹדַעְתּוֹ.

כִּשְׁמֹעַ הַשָּׂר יְפִי הַשִּׁירָה

נִפְלָא עַל חַשְׁמַלֵּי זָהֳרָהּ

וַיֹּאמַר שִׁירָתְךָ זֹאת גּבָרְהָ

עַל כָּל הַשִּׁירוֹת בְּיָפְיָהּ

וְחִשְׁקְכֶם זֶה כָמוֹהוּ מִן הָעוֹלָם לֹא נִהְיָה

וְעַתָּה הַגֶּד-נָא לִי אִם חֲשׁוּקְךָ זֶה חָטָא לְךָ מֵעוֹדוֹ

וְאִם הוּא הָאִישׁ אֲשֶׁר שָׁמַעְתִּי בַעֲדוֹ

כִּי עָלֶיךָ יְנוֹפֵף יָדוֹ

וּפָרַע חֹק הַמּוּסָר

בְּקָרְאוֹ אוֹתְךָ בִּכְתָבוֹ אֲרִי נוֹהֵם מִתּוֹךְ קֻפָּה שֶׁל בָּשָׂר

וָאַעַן וָאֹמַר אֲדֹנִי לֹא זֶה הַדֶרֶךְ ולֹא זֹה הָעִיר

כִּי הָאישׁ אֲשֶׁר הֵרִים יָד בִּי מִמֶּנִי לְיָמִים צָעִיר

וּשְׁמוֹ יוֹאָב שָׂר הַצָּבָא אֲשֶׁר להִלָּחֵם עִמִּי בָא

וּבְעֵת יָצָאתִי לִקְרָאתוֹ עִם חֵיל מְלִיצוֹתַי נֶחְבָא

וְעָמַד בִּרְאוֹתוֹ אוֹתָם מִשְׁתָּאֶה

וּבְעֵת אָמַרְתִּי לוֹ לְכָה פָנִים נִתְרָאֶה

וּרְאֵה וַעֲשֵׂה בְּתַבְנִיתָם אֲשֶׁר אַתָּה מָרְאֶה

נִסְתַּר אוֹ נֶחְבָּא אֵין רוֹאֶה

וַיֹּאמֶר הַשָּׂר אֵיךְ הָיָה הַדָּבָר

וָאַעַן וָאֹמַר יֵדַע אֲדֹנִי הַנָּבָר

כִּי יוֹאָב שַׂר הַצָּבָא אֲשֶׁר שָׁמַעְתָּ מֵעִנְיָנָיו

הָיָה אִישׁ חָכָם בְּעֵינָיו

גַּאֲוָה אֲפִיקֵי מָגִנָּיו

וְכָל חַיּוֹת לֹא יַעַמְדוּ לְפָנָיו

וְהוּא לְכָל מִבְצָר יִשְׂחָק

וּמִכָּל חֲכַם לֵב יִרְחָק

וְגָעַר בּוֹ וְנָס מִמֶּרְחָק

וַיְהִי הַיּוֹם וּמֵיְיָ הָיְתָה נְסִבָּה

וְשָׁמַע כִּי הוֹצֵאתִי עָלָיו דִּבָּה

וְסִפַּרתִּי מִמֶּנּוּ הֶבֶל וְרָעָה רַבָּה

בְּפִי וּבְמִכְתָּבִי

אֲשֶׁר לֹא צִוִּיתִי וְלֹא דִבַּרְתִּי ולֹא עָלְתָה עַל לִבִּי

וַיִּתְנַדֵּב לְנוֹפֵף עָלַי צוּר חַרבּוֹ

וַיִּכְתֹּב אֵלַי אִגֶּרֶת כַּאֲשֶׁר עִם לְבָבוֹ

כָּתַב בָּהּ הֲתֻלִים וְחִידוֹת

וּפְנֵיהֶם וְכַנְפֵיהֶם פְּרֻדוֹת

פַּעַם יִתְנַשֵּׂא וּפַעַם יִכָּנַע

פַּעַם יְחָרֵף וּפַעַם יִמָּנַע

הָלַךְ בְּשִׁירוֹ נָד וָנָע

הִזְכִּיר בּוֹ שִׁיר הַיִּחוּד אֲשֶׁר לוֹ

לְהַרְאוֹת הָעַמִים וְהַשָּׂרִים שִׂכְלוֹ

וְכַאֲשֶׁר קָרָאתִי אִגַּרְתּוֹ לִבִּי הִתְאַנָּף

כִּי לֹא הָיָה לִדְבָרָיו שֹׁרֶשׁ וְעָנָף

וָאֲכוֹנֵן מֵיתְרֵי מִכְתָּבִי

וָאָשִׁיב לוֹ דָבָר כַּאֲשֶׁר עִם לְבָבִי

וְאֵלֶּה דִבְרֵי אִגַּרְתִּי

הַשְּׁלוּחָה אֵלָיו מֵאתִּי:

מִי זֶה מַחְשִׁיךְ עֵצָה בְמִלִּים

לָחוּד חִידוֹת וְלָשֵׂאת מְשָׁלִים

בְּרֹב חֲלוֹמוֹת וַהֲבָלִים

יִלְעַג וְאֵין מַכְלִים

הֲקוֹלְךָ זֶה בְּנִי יוֹאָב שַׂר הָעֲשָׂרוֹת

יְחוֹלֵל אַיָּלוֹת וַיֶחֱשֹׂף יְעָרוֹת

תִּקַּח מֵחֵלֶם וּמִטּוֹבִיָּה עֲטָרוֹת

כִּי תִכְתֹּב עָלַי מְרֹרוֹת

לֹא יָדַעְתִּי סְפֹרוֹת

חַיֶּיךָ אִם חֲלוֹם חָלַמְתָּ אֲנִי אֶפְתֹּר

וְיִרְעֲשׁוּ הַסִּפִּים אִם אַךְ הַכַּפְתֹּר

וּבִמְקוֹם שַׁאֲגַת אַרְיֵה וְקוֹל שַׁחַל מַה לְּקוֹל הַתּוֹר

זֶה כַמֶּה שָׁמַעְתִּי כִּי תּוֹר

אַךְ לֹא הֶאֱמַנְתִּי כִּי תַעֲשֶׂה וְתִסְתּוֹר

וְלָכֵן קַבֶּל-לְךָ אַבְנֵי קֶלַע תִּקְלַע מְלִיצָתִי כִּי הִגִּיעַ הַתּוֹר

וְתִקְצוֹר קְצִירְךָ אֲשֶׁר זָרַעְתָּ

וְתִשְׁמַע מֵאַשֶׁר הִגַּדְתָּ וְהוֹדַעְתָּ וְהִשְׁמַעְתָּ

חַיֶּיךָ אֲדֹנִי עֲדַיִן לֹא הִגַּעְתָּ לְרוֹעֵי צֹאן

וּדְבָרֶיךָ לֹא לְרָצוֹן

כִּי תִכְתֹּב עָלֵינוּ לָצוֹן

וְאִם עָסַקְתָּ בְּפֵרוּשׁ וּבִגְמָרָא

לֹא זָכיתָ לִהְיוֹת תַּנָּא אוֹ אֲמוֹרָא

וְיָדַעְתִּי כִּי רַעְיוֹנֶיךָ עָבְרוּ עַל דִּבְרֵי סְדוֹם וַעֲמוֹרָה

וַיִּכְרְתוּ מִשָּׁם זְמוֹרָה

וְלָכֵן שׁוֹלְחִים לְאַפָּם אֶת הַזְּמוֹרָה

וְשָׁמַעְתִּי כִּי בֵין אַחִים תַּפְרִיא

וְלָכֵן כָּעֵת בַּמָּרוֹם תַּמְרִיא

שָׁלַחְתָּ אִגֶּרֶת הֲתֻלִים

וָאֶקְרַב אֵלֶיהָ וְלֹא מָצָאתִי לָהּ בְּתוּלִים

וְעַתָּה הַכֶּר-נָא לְמִי הַחוֹתֶמֶת וְהַפְּתִילִים

הֶרְאֵיתָנוּ תְפִלַּת הַיִּחוּד אֲשֶׁר לָךְ

אֲשֶׁר הִיא לְעֵדָה עַל שִׂכְלָךְ

אָחִי יְהִי לְךָ אֵת אֲשֶׁר לָךְ

הַגֶּד-נָא הַשָּׂר מִמִּי נָעָמְתָּ

מָה הַחֲלוֹם הַזֶּה אֲשֶׁר חָלָמְתָּ

לְהִתְפָּאֵר בְּחָכְמוֹת נַעֲלָמוֹת

וַתֹּאמַר עֶשֶׂר יָדוֹת לִי בְּכָל חָכְמוֹת

הֲשָׁמַעְתָּ מִיָּמֶיךָ עַל הַהִגָּיוֹן

הֲרָאִיתָ לְעוֹלָם סֵפֶר הַחִזָּיוֹן

וְאֵיךְ לֹא תֶחֱרַד וְלֹא תִלָּפֵת

וְלֹא רָאִיתָ לְעוֹלָם סֵפֶר הַמּוֹפֵת

הֲנִגְלוּ לְךָ תַּעֲלֻמוֹת הַהֶקֵּשׁ

כִּי לִדְרֹשׁ בְּכָל חָכְמָה תְּבַקֵּשׁ

הֲלָמַדְתָּ סֵפֶר הַנִּצּוּחַ וְהַשִּׁירוֹת

וְסֵפֶר הַמְּלִיצָה וְהַמַּאֲמָרוֹת

וְאִם לִבְּךָ דָבָר טוֹב רָחָשׁ

הֲיָדַעְתָּ סֵפֶר הַחוּשׁ וְהַמֻּחָשׁ

וְאִם אַתָּה הָאֹמֵר לַצּוּלָה חֳרָבִי

הֲלָמַדְתָּ הַזַּהֲרָאבִי

וְאִם לָמַדְתָּ הַפָּרָשָׁה עִבְרִי

הֲרָאִיתָ סֵפֶר הָאַוִּירִי

הַגֶּד-נָא מָה אָמוּר

בְּסֵפֶר הַטּוֹב הַגָּמוּר

וּמַה דְּרָכָיו וּמַה מִּשְׁפָּטוֹ

וּמַה בְּסֵפֶר הַנֶּפֶשׁ לַאֲרִיסְטוֹ

מַה תִּתְהַלֵּל בָעֲמָקִים וְעִמְקְךָ זָב

וּבָתֵּי חָכְמוֹתֶיךָ לְאַכְזָב

וְאֵיךְ שִׁיר יִחוּדְךָ לֹא פֵאַרְתָּ

וּבִמְחוּגַת הַשֵּכֶל אוֹתוֹ לֹא תֵאַרְתָּ

וְאִגֶּרֶת טוֹבָתְךָ לֹא עִטַּרְתָּ

וְעַל כַּמָּה שְׁגָגוֹת עָבַרְתָּ

וְלוּ יָדַעְתָּ כַּמָּה שָׁגִיתָ בְּמִלּוֹת אֲמָרֶיךָ

אָז שָׁמַרְתָּ מַחֲסוֹם לְפִיךָ וּגְבוּל לִדְבָרֶיךָ

בְּשׂוּמְךָ בְּמִלַּת וּבֶן אָדָם הַשּׁוּרֶק בִמְקוֹם מֶלֶךְ

לֹא יָדַעְתָּ הַדֶּרֶךְ אֲשֶׁר בָּהּ תֵּלֶךְ

וּבַעשׂוֹתְךָ צָמִיד וְתָמִיד עַל מִשְׁקַל עוֹלָמִית

לֹא מָצָאתִי לְךָ רֵעַ וְעָמִית

אַחַת דָּתְךָ לְהָמִית

והוּא מַה שֶּׁיּוֹרֶה עַל תִּגְבֹּרֶת הנֶּפֶשׁ הַבֶּהָמִית

וּבְאָמְרְךָ וּמַה לַּתֶּבֶן אֶת הַבָּר

וְשַׂמְתָּ הַשּׁוּרֶק בִּמְקוֹם מֶלֶךְ אֵי זֶה דֶרֶךְ עָבָר

רוּחַ הַשִּׁירִים אִתְּךָ לוֹא לִזרוֹת וְלוֹא לְהָבֵר

וּבְאָמְרְךָ שׁוֹד שׁוֹטֵף בִּמְקוֹם שׁוֹט

יוֹרֶה עַל הֱיוֹתְךָ בְּיַם הַפָּסוּק בְּאֵין מָשׁוֹט

וּבְאָמְרְךָ לֹא תְבַזֶּה בִּמְקוֹם יְבַזֶה

עֱנוּת הַשִּׁיר תִּבְזֶה

לֹא יָדַעְתִּי מַה זֶּה וְעַל מַה זֶּה

וּבְאָמְרְךָ חֲכַם בִּינָה שָׁבָה בִינָתְךָ לְחָכְמָה

אוֹי לָהּ לְאוֹתָהּ כְּלִמָּה

וּבְאָמְרְךָ תַּלְמִידְךָ אֲנִי וּמֵימֶיךָ אֲנִי שׁוֹתֶה

לֹא כֵן אָחִי כִּי גֶחָלִים עַל הָרֹאשׁ אַתָּה חוֹתֶה

פֶּן יֹאמְרוּ הָרַב כַּתַּלְמִיד

וְהַכְּלִי כַּצָּמִיד

הַכֹּל הוֹלֵךְ אַחַר הַמַּעֲמִיד

וּבְאָמְרְךָ הֶעָלֶה נִדָּף תַּעֲרֹץ וְאֶת קַשׁ יָבֵשׁ תִּרְדֹּף

לֹא כֵן אַתָּה בְעֵינֶיךָ כִּי בְצַד וּבְכָתֵף תֶּהְדֹּף

וְעַל הוֹדָאָתְךָ בִגְדֻלָתִי תִּשְׁמַע חֲלוֹם לִפְתֹּר

בָּנִיתָ אִי אַתָּה יָכוֹל לִסְתֹּר

וְהִשְׁתּוֹקֶקְךָ לְהִתְאַבֵּק בַּעֲפַר רַגְלָי

נוֹדָע כִּי זֶה לִשְׁלֵמוּת פְּעָלָי

וְלִשְמֹעַ נֹעַם מְשָׁלָי

וּמִי לִמֶּדְךָ לוֹמַר מִן זֶבֶד זַבְדוּת

וּמִן עַנְפֵי וְהֵם זְכָרִים פְּרוּדוֹת

וְאֵיךְ תֹּאמַר הֲרֵינִי כְבֶן עַזַּאי בְּשׁוּקֵי טְבֶרְיָה

וּבְאָמְרְךָ גְוִיֵּנוּ מִן גְּוִיָּה

לֹא הִתְפָּאַרְתִּי בְּמַה שֶׁאֵין בִּי

וְלֹא הֲנִיפוֹתִי עַל אָדָם חַרְבִּי

לֹא יָדַעְתִּי פִרְקֵי הַמּוֹרֶה עַל פֶּה

וְכַאֲשֶׁר יַחְקְרוּנִי בְעַד סוֹדָם אֶעֱמֹד עֻלְפֶּה

אָכֵן אֵדַע הַדָּבָר עַל שָׁרְשׁוֹ

וְלֹא בִקַּשְׁתִּי מִן הַפֶּרֶק כִּי אִם פֵּרוּשׁוֹ

וְשָׂרַפְתִּי מִחוּץ לַמַּחֲנֶה עֹרוֹ וּפִרְשׁוֹ

אֱמָר-נָא הַשָּׂר מַה בֶּצַע בַּאֲשֶׁר יֹאמְרוּ הַשְּׂפָתָיִם

וְהַלְּבָבוֹת תִּשְׁכַּבְנָה בֵין שְׁפַתָּיִם

וּמַה בֶּצַע בַּאֲשֶׁר תִּרְאֶינָה הָרֹאוֹת בָּאֲרֻבּוֹת

וְחָשְׁכוּ עֵינֵי הַלְּבָבוֹת

לֹא כֵן אָחִי הַגָּבֹהַּ הַגְבֵּהַּ וְהַשָּׁפָלָה הַשְׁפֵּל

הַלֵּב עִקָּר וְהַפֶּה טָפֵל

וְאִם אַבְרָהָם מַה צֹּרֶךְ לְאַמְרָפֶל

וּבְאוֹר שִׁבְעָתַיִם מַה בֶּצַע בָּעֲרָפֶל

וְעַתָּה הֲיַעֲמֹד לִבְּךָ אִם תֶּחֱזַקְנָה יָדֶיךָ

אִם יֹאמְרוּ לְךָ רִבִּי פָּרֵשׁ סוֹדֶךָ

מַה לְךָ לְהִתְפָּאֵר וּלְהִתְהַדֵּר בְּמַקְהֵלוֹת

כַּלַּךְ מִדַּבְּרוֹתֶיךָ אֵצֶל נְגָעִים וְאָהֳלוֹת

וְשָׁם אִם תִּרְצֶה לָחוּד חוּד

וְאַל תַּשְׁמַע קוֹלְךָ עִמָּנוּ עוֹד עַל דְּבַר יִחוּד

הַנַּח הַמּוֹרֶה לַמּוֹרֶה וְהַמַּדָּע

לְיוֹדְעֵי דַּעַת וּמְבִינֵי מַדָּע

וְהַשֵּׂכֶל לְכַלְכֹּל וְדַרְדַּע

וְהַמָּן לְאָסָף וּלְהֵימָן

וּבִמְקוֹם מָרְדְּכַי מַה צֹּרֶךְ לְהָמָן

הַנַּח הַסְּתָרִים לִמְבִינֵיהֶם

וְהַכְּתָרִים לְבַעֲלֵיהֶם

וְהַסּוֹדוֹת לִבְנֵי עֲלִיָּה

וְהָעֲטָרוֹת תִּהְיֶינָה לְחֵלֶם וּלְטוֹבִיָּה

וְהֶעָלִים וְהַקְּלִפּוֹת לַאֲשֶׁר אֵין מֹחַ בְּקָדְקֳדָם

שֹׁכְנֵי בָתֵּי חֹמֶר אֲשֶׁר בֶּעָפָר יְסוֹדָם

כָּעוֹף יִתְעוֹפֵף כְּבוֹדָם

אַל תָּבֹא נַפְשִׁי בְסֹדָם

וְעַתָּה לָמָּה פָנֶיךָ תַּסְתִּיר

וְצַדִּיק תָּמִים תַּכְתִּיר

וּבִבְלִי דַעַת מִלִּין תַּעְתִּיר

וְחִצֶּיךָ שְׁנוּנִים תִּיר

נִים וְלָא נִים תִּיר וְלָא תִיר

וְאֵיךְ לֹא תִירָא אַתָּה הַיָּשֵׁן הַשֹּׁכֵב בְּאַמְבַּטִּי

מֵחִצֵּי מְשָלַי וּמִן הַחֶרֶב אֲשֶׁר בְּפִי עֵטִי

אֲשֶׁר אוֹרֶה אֶל הַשַּׂעֲרָה וְלֹא אַחֲטִיא

וְלֹא יָרֵאתָ פֶּן יִפְגְּעוּ בְךָ אֲנָשִׁים מוֹרִים בַּקָּשֶׁת

מִשֵּׂתָם יָגוּרוּ אֵלִים וְחִיל אָחַז יֹשְׁבֵי פְּלָשֶׁת

וְהָיוּ לְדָם בַּיַּבָּשֶׁת

וְלוּ רָאִיתָ חֲמוּדוֹתַי

אַפִּרְיוֹנֵי שִׂכְלִי וּרפִידוֹתַי

וּמֹר וַאֲהָלוֹת קְצִיעוֹת כָּל בִּגְדוֹתַי

וְשֶׁמֶן הָעֲנָוָה עַל הָרֹאשׁ שֶׁיֹּרֵד עַל פִּי מִדּוֹתַי

לֹא תַאֲמִין כִּי יְסֻפָּר

לָהֵן מַלְכָּא מִלְכִּי עֲלָךְ יִשְׁפָּר

וְתִיטַב לְךָ מִשּׁוֹר פָּר

חֲדַל לְךָ מֵהְתִפָּאֵר בְּמַה שֶּׁאֵין בְּךָ מִדִּבְרֵי חָכְמָה

וּבַמָּקוֹם שֶׁיֵּשׁ אֲנָשִׁים הִשְׁתַּדֵּל לִהְיוֹת בְּהֵמָה

אַךְ לִקְטַנֵּי אֲמָנָה

וְלַנָּשִׁים הַנּוֹשְׂאוֹת וְנוֹתְנוֹת וּמוֹזְרוֹת בַּלְּבָנָה

זֶה יִתָּכֵן

תְּפַתֶּה וְגַם תּוּכָל צֵא וַעֲשֵׂה כֵן

וְהִנְנִי נִשְׁבָּע בְּחֵי הַשִּיר וּבִגְאוֹנוֹ

אֲשֶׁר נָטַע בְּלִבִּי אָהֳלֵי אַפַּדְנוֹ

וְנָתַן בִּימִינִי דִגְלוֹ וּבִשְׂמֹאלִי מָגִנּוֹ

וּמְשָׁחַנִי מֵחֲבֵרַי שֶׁמֶן שְׂשׂוֹנוֹ

כִּי אִם מֵהיּוֹם הַזֶּה וָמַעְלָה תִּתְפָּאֵר

לְדַבֵּר בְּחִידוֹת וְסוֹדוֹת לְבָאֵר

לֹא תִשְׁתֶּה מֵי בְאֵר

וְאִם לְעוֹלָם תִּתְהַלֵּל

לִשְׁאֹל אִסּוּר מוֹסִיף וְאִסּוּר כּוֹלֵל

וְאִם לֹא תַנִּיחַ לְחֵלֶם וּלְטוֹבִיָּה הָעֲטָרוֹת

וְאִם יַשִּׂיאֲךָ זְדוֹן לִבְּךָ לִפְסֹל לְעוֹלָם שְׁטָרוֹת

וְאִם לֹא תֵלֵךְ שְׁחוֹחַ לִפְנֵי גְדוֹלֵי הַמַּעֲלָה

וְאִם תְּפָרֵשׁ עוֹד סוֹדוֹת הַתָּהֳלָה

וְאִם תְּבַזֶּה עוֹד הַשִּׁירִים אֲשֶׁר אָהָבְתִּי

אֶמְחֶה אוֹתְךָ מִסֵּפֶר הַחַבְרוּתָה אֲשֶׁר כָּתָבְתִּי

וְתַחַת אֲשֶׁר הִרְאִיתַנִי פְעָלֶיךָ

וְחִבּוּרְךָ עַל הַקָּמִים וְהַטּוֹעֲנִים עָלֶיךָ

קְצֹר פְּרִי הֲתֻלֶיךָ

וּרְאֵה נֹעַם פְּעָלָי

וְחֶמְדַּת מְשָׁלָי

אֲשֶׁר חִבַּרְתִּי לַמִּתְוַכְּחִים וְהַטּוֹעֲנִים עָלָי:

אַתְּ בֵּין עֲצֵי יַעַר לְבָנוֹן חוֹחַ

בֵּין רֹזְנֵי אֶרֶץ וְשָׂרִים תַּעֲלוּל

בֵּין לִשְׁכוֹת קֹדֶשׁ כְּמוֹ רֶפֶת וְלוּל

בֵּין הַנְּבִיאִים נִבְּאֵי מָנוֹחַ.

בֵּין מַמְלְכוֹת סִיחוֹן וְעוֹג זָנוֹחַ

כָּל הָאֱנוֹשׁ עִלָּה וְאַתָּה כַעֲלוּל

בֵּין זֹחְלֵי עָפָר וְרֶמֶשׂ שַׁבְּלוּל

בֵּין הַנְּהָרִים שֹׁטְפִים שִׁלוֹחַ.

גַּנָּב אֲשֶׁר תִּגְנֹב חֲמֻדוֹת הַזְּמָן

מִשַּׁבְּתַי עַצְלוּת וּמִן נֹגַהּ זְנוּת

מִמַּאֲדִים חָרוֹן וְזֶרֶש מֵהֲמָן

מֵהַזְּבוּבִים שִׁיר וּמֵעָשׁ מִסְכְּנוּת

מִגֵּחֲזִי כָזָב וְנֶגַע נַעמָן

הַגַּאֲוָה מִן גִּיד וּמִן עֶרְוָה גְּנוּת.

רְאֵה אָחִי כַּמָּה כָבוֹד כִּבַּדְתִּיךָ

כַּאֲשֶׁר חָמַלתִּי עַל אַהֲבָתְךָ וַעֲנִיתִיךָ

וּמַה גָּדְלָה תְשׁוּקָתִי בְךָ

כַּאֲשֶׁר לֹא הִנַּחְתִּיךָ עַל רָהְבְּךָ

טוֹב לְךָ לִשְׁמֹעַ גַּעֲרָתִי

וְלַחֲסוֹת בְּצֵל אֶבְרָתִי

מֵאֲשֶׁר יְפָאֲרוּךָ הַכְּסִילִים

וְתִתְהַלַּל בְּפִי יוֹנְקִים וְעוֹלְלִים

וַאֲשֶׁר מָדַד בְּשָעֳלוֹ מַיִם

וְתִכֵּן בַּזֶּרֶת שָׁמַיִם

יְחַיֵּנוּ וִיחַיֶּךָ מִיֹּמָיִם

וְיָחוֹן אוֹתָנוּ וְאוֹתְךָ דֵעָה וְהַשְׂכֵּל

לְכִי שִׁירָתִי אֶל הַשָּׂר וְהָאֲמַרְכֵּל

מְקֻטֶּרֶת מֹר וּלְבוֹנָה מִכֹּל אַבְקַת רוֹכֵל

וְהָיָה כִּי יִפְגָּשְׁךָ אִישׁ וּשְׁאָלְךָ שֹׁאֵל

וְלָדַעַת שֵׁם מְחַבֶּרְךָ יוֹאֵל

וְקָרָאתָ שְׁמוֹ עִמָּנוּאֵל.

כּשְׁמֹעַ הַשָּׂר דִּבְרֵי אִגַּרְתִּי חִבְּקַנִי וּנְשָׁקַנִי

כִּי הוֹכַחְתּוֹ תוֹכֵחָה נִמְרֶצֶת

וְהוֹבַשְׁתָּ מְלִיצָתוֹ הַמְנֹאֶצֶת

וְהֲסִירוֹתָ אוֹתהּ מִגְּבִירָה תַּחַת אֲשֶׁר עָשְׂתָה לָהּ מִפְלֶצֶת

וְעַתָּה כְּתֹב אוֹתהּ אֵלַי לְזִכָּרוֹן

וְאָשִימָה אוֹתָהּ לִי לְמִשְׁמֶרֶת בְּצַד הָאָרוֹן

לִרְאוֹתָהּ לְעִתִּים מְזֻמָּנִים

לְחַדֵּשׁ נִצָּנֵי הַשְּשׂוֹנִים

וּלְהַבְרִיחַ מִכָּל צַד הַיְגוֹנִים

וָאַעַן וָאֹמַר דַּע אֲדֹנִי הַמְהֻלָּל בְּכָל פֶּה וְגָרוֹן

כִּי לוּ רָאִיתָ הַמְּלִיצָה אֲשֶׁר שָׁלַחְתִּי לְרִבִּי אַהֲרֹן

רָאִיתָ בָהּ עַל כָּל הַמְּלִיצוֹת יִתְרוֹן

וְאָז יִמָּלֵא שְׂחוֹק פִּיךָ וּלְשׁוֹנְךָ רִנָּה

הִיא עָלְתָה עַל כָּל הַמְּלִיצוֹת בֶּאֱמוּנָה

לְשַׂמֵּחַ נֶפֶשׁ נַעֲנָה

וּלְחַדֵּשׁ שְשׂוֹנָהּ

וּלְהַעֲבִיר יְגוֹנָהּ

וַיֹּאמֶר הַשָּׂר הַגֶּד-נָא לִי עִנְיָנָהּ

וָאֹמַר דַּע הַשָּׂר כִּי הָיִיתִי בְּפֵירוּשָׁ"א הַמְּדִינָה

בַּעַל חָכְמָה וְכִשְׁרוֹן

וְעִמּוֹ סְפָרִים חֲמוּדִים

מַשָּׂא צֶמֶד פְּרָדִים

רְאִינוּהוּ כִרְאוֹת פְּנֵי אֱלֹהִים

שֶׁעָשָׂה לָנוּ נִסִּים

עוֹד אָמַרְתָּ בְאָזְנָי

אַתָּה עַתָּה בְּרוּךְ אֲדֹנָי

כִּי הָיָה בְדַעְתְּךָ לִקְבֹּל עָלֵינוּ

עַל הַהַעְתָקוֹת אֲשֶׁר מִמְּךָ הִצַּלְנוּ

וְלֵאמֹר כִּי עֶדְיְךָ נִצַּלְנוּ

אָחִי הֲסָבוּר אַתָּה כִּי אֲנַחְנוּ הִתְעַצַּלְנוּ

הֲלֹא רָאִיתָ הַבְּרָכָה אֲשֶׁר לְךָ אָצַלְנוּ

וְאֶת הַמַּקְלוֹת אֲשֶׁר פִּצַּלְנוּ

הָעֹמְדִים חֶדֶר בְּחָדֶר

לְתַקֵּן הַנְּתָחִים אֶת הָרֹאשׁ וְאֶת הַפָּדֶר

וְלִקְרֹעַ הָרֹאשׁ עֵרֶב וָשֶׁתִי

מֵאִישׁ שׁוֹגֶה וּמִפֶּתִי

הֲלֹא הֻגַּד לְךָ אָחִי הַשְּׁבוּעָה אֲשֶׁר נִשְׁבַּעְנוּ בְמַקְהֵלוֹת

נֶפֶשׁ מְכַבְּדֵינוּ לְכַבֵּד וּלְהַעֲלוֹת

וְנֶפֶשׁ בּוֹזֵינוּ וְאִישׁ יָשִׂים עֲלִילוֹת

לְיַחְמֵנָּה בְמַקְלוֹת

אֱמָר-נָא הַשָּׂר מֶה חָזִיתָ

כַּאֲשֶׁר דַּרְכֵי הַנְּדִיבוּת בָּזִיתָ

שִׁבַּרְנוּ חָבִיּוֹת סְפָרֶיךָ וְהִתְעַבַּרְתָּ

וּמֹשֶׁה רַבֵּנוּ שִׁבֵּר הַלּוּחוֹת וְנַעֲנָה יִישַׁר כֹּחֲךָ שֶׁשִּׁבַּרְתָּ

וּכְבָר לִמַּדְתָּנוּ שׁוֹבֵר אָדָם בַּשַּׁבָּתוֹת הַחֲמוּדוֹת וְהַיְקָרוֹת

חָבִית מְלֵאָה גְרוֹגָרוֹת

וְכָל שֶׁכֵן שֶׁיֶּשׁ לְךָ לִמְחוֹל

אִם שִׁבַּרְנוּ חָבִיּוֹת מְלֵאוֹת סְפָרִים כַּחוֹל

וּכְבָר נִכְתַּב מִידֵי אַבִּיר

וּבְרָכָה לְרֹאשׁ מַשְׁבִּיר

וַאֲשֶׁר אָמַרְתָּ כִּי אֲבוֹתֵינוּ הַקְּדוֹשִׁים לֹא הָיוּ עוֹשִׂים כֵּן

וְעַל כֵּן דַּרְכֵּנוּ לֹא יִתָּכֵן

חַי יוֹצַר רוּחַ אָדָם בְּקִרְבּוֹ

מָקוֹם הִנִיחוּ לָנוּ אֲבוֹתֵינוּ לְהִתְגַּדֶּר בּוֹ

עוֹד אָמַרְתָּ בְאָזְנִי

מַה לָּכֶם לִרְאוֹת עִזְּבוֹנִי

אוּלַי יֵשׁ דְּבַר סֵתֶר בְּמַצְפּוּנִי

אָבִי הֲלֹא הִרְאִיתָנוּ בְּקֶלֶף הַזִּכָּרוֹן

שְׁמוֹת יְשֵׁנֵי חֶבְרוֹן

הֵם הֵם שְׁמוֹת הַסְּפָרִים מִן הָרִאשׁוֹן עַד הָאַחֲרוֹן

וְהֶחֲזַקְנוּךָ בְּחֶזְקַת נֶאֱמַן רוּחַ וְנָבָר

כּי לֹא הֶעֱלַמְּתָּ מִמֶּנּוּ דָּבָר

וְלָכֵן הַפַּח נִשְבָּר

כִּי בּהַרְאוֹתְךָ שְׁמוֹת הַסְּפָרִים לְכֻלָּנוּ

אָמַרְנוּ מַרְאֶה מָקוֹם הוּא לָנוּ

וּבְהַכִּירֵנוּ עֶרְכָּם וְטוּבָם

וְהַתְּעוּדוֹת הַצְּפוּנוֹת בְּקִרְבָּם

נִכְסַפְנוּ לְהַתִּירָם מִמַּאֲסָרָם

וּלְהַרְאוֹת הָעַמִּים יָפְיָם וַהֲדָרָם

וְאִם תֹּאמַר אוּלַי יֶשׁ בָּהֶם דְּבַר נִסְתָּר יָדַעְנוּ כִּי לֹא הָיָה בְתוֹכָם עַלְמָה

כִּי שָׁהִינוּ מֵעֵת נְסִיעָתְךָ כְּדֵי יְצִיאַת נְשָׁמָה

וּבַאֲשֶׁר אָמַרְתָּ הֲלֹא כָתוּב בְּסִפְרְכֶם נֶאֱמָנַי

וְאָהַבְתָּ לְרֵעֲךָ כָּמוֹךָ אֲנִי אֲדֹנָי

תְּשׁוּבַת שְׁאֵלָתְךָ מֹר עוֹבֵר נוֹטְפָה

כְּבָר פֵּרְשׁוּהָ רַבּוֹתֵינוּ בְּרוֹר לוֹ מִיתָה יָפָה

וְעַתָּה אָחִי בָזִיתָ לִגְנֵבָתֵנוּ וְנַפְשְׁךָ לָנוּ תִלְעָב

הֲשָׁכַחְתָּ לֹא יָבוּזוּ לַגַּנָּב

כִּי יִגְנוֹב לְמַלֵּא נַפְשׁוֹ כִּי יִרְעָב

וְעַתָּה נַפְשֵׁנוּ הַיְבֵשָׁה

נִכְסְפָה לְקוֹל הַתְּבוּנָה הַחֲדָשָׁה

וְרַעֲיוֹנֵינוּ אוֹמְרִים מִי יָבִיא לָנוּ שֶׁבֶר רְעָבוֹן

מִשִּׁפְרֵי הַעְתָּקַת רִבִּי מֹשֶׁה אִבְּן תִּבּוֹן

הוּא יְנַהֲגֵנוּ עַל מוּת

שָׁמָה כָל נְעִימוּת

מִבְחַר הַיְדִיעוֹת וְזִמְרַת הַשְּׁלֵמוּת

רְדוּ שִׁבְרוּ לָנוּ מִשָּׁם וְנִחְיֶה וְלֹא נָמוּת

וַאֲשֶׁר שָׁאַלְתָּ לְהַגִּיד לְךָ הָאֱמֶת

חַי מֵמִית הַחַי וּמְחַיֵּה הַמֵּת

לְהַעְתִּיק סְפָרֶיךָ עַיִן לֹא הֶעֱלַמְנוּ

וְלֹא נִדְאַג רַק עַל אֲשֶׁר לֹא הִשְׁלַמְנוּ

וְלֹא נְבַקֵּשׁ מְחִילָה רַק אִם הָיָה שָׁם עַצְלוּת

וְלֹא נָבִיא קָרְבַּן כַּפָּרָה רַק עַל הַהִתְרַשְּׁלוּת

וְכִי מִבִּלְעֲדֵי עִמָּנוּאֵל שִׁבְּרוּ הַנְּעָרִים הֶחָבִיּוֹת

חֵי מִגְדַּל דָּוִד בָּנוּי לְתַלְפִּיּוֹת

וְחֵי אוֹר פְּנֵי הָעֳפָרִים וְיפְעַת פְּנֵי הַצְּבִיּוֹת

כִּי זֶה כַּמָּה אָמַרְתִּי אֶל הַנְּעָרִים

עָרֹם יַעֲרִים

וְכִי הֵמָּה מִשְׂתַּכְּרִים

וּכְזֹהַר הָרָקִיעַ מַזְהִירִים

בַּשֶּׁבֶר אֲשֶׁר הֵם שׁוֹבְרִים

וָאֹמַר אֲלֵיהֶם גַּם הַיּוֹם גַּם מִתְּמוֹל

אַל תִּפְנוּ אֶל הַיָּמִין וְאֶל הַשְּׂמֹאל

שִׁבְּרוּם כְּשֵׁבֶר נֵבֶל יוֹצְרִים כָּתוּת לֹא יַחְמֹל

וְאִם הֶעְתַּקְתִּי סֵפֶר הַשֵּׁמַע הַטִּבְעִי

זֶה טִבְעִי

וּכְבָר יָצָא בְכָל הָאָרֶץ טִבְעִי

וְלָכֵן אֹמַר לְכָל נֶפֶשׁ אִוֶּלֶת טִבְעִי

וְהִנְנִי נִשְׁבָּע בַּאֲשֶר זֵרָה אָרְחִי וְרִבְעִי

וּבְיַרְדֵּן הַתְּעוּדוֹת רְחָצַנִי לְמִשְׁעִי

כִּי לֹא הִכַּרְתָּ מִמֶּנִּי רִבְעִי

זֶה כַּמָּה תְעוּדוֹת שָׁלַלְתִּי

וּבְעִמְקֵי הַתְּבוּנוֹת צָלַלְתִּי

וּמְקוֹם שַׁחֲרֵי הַתְּעוּדָה יָדַעְתִּי

וּבְרִיחֵי הַתְּבוּנָה גָּדַעְתִּי

וּבָעַמִּים עֻזָּם הוֹדַעְתִּי

וְאֶת הַנְּעָרִים יוֹדַעְתִּי אֶל מְקוֹם פְּלֹנִי אַלְמֹנִי תַּחֲנוֹת

הַתְּבוּנוֹת הָעֲדִינוֹת

וּסְגֻלַּת מְלָכֵים וְהַמְּדִינוֹת

כִּי בְמַקְלִי עָבַרְתִּי יַרְדֵּן הַתְּבוּנוֹת

וְעַתָּה הָיִיתִי לִשְׁנֵי מַחֲנוֹת

וַאֲשֶׁר אָמַרְתָּ

כִּי הִתְעַבַּרְתָּ

לְהָסִיר מֵנַּפְשֵׁנוּ חֶלאָתָהּ

וּלְהָעִיר אוֹתָהּ מִפְּתַיּוּתָהּ

וּלְזָרֵז אוֹתָהּ עַל שִׁגְגָתָהּ

וְכִי אַתָּה הוּא הָרוֹפֵא אֶת מַחֲלָתָהּ

אוֹ הַכֹּהֵן הַמְּחַטֵּא אוֹתָהּ

עוֹד רָצִיתָ לְהַשְׁבִּיעֵנוּ אֲשֶׁר כָּזֹאת לֹא נוֹסִיף

חֵי אֲנִי כִּי אֲשֶׁר יֹאמַר לְךָ עָוֶל פָּעַלְתִּי לֹא אוֹסִיף

וּדלָא מוֹסִיף יְסִיף

וּמִי לֹא יְטַהֵר נַפְשׁוֹ מִמַּעֲלָהּ

וְלֹא יוֹסִיף בּכָל עֹז מַעלָה עַל מַעֲלָה

חֲדַל אָחִינוּ מֵהִתְקַצֵּף וְקַלֵּל

וְאֶת שֵׂיבָתְךָ עַל נְקַלָּה לֹא תְחַלֵּל

כִּי הָעֹשֶׁק הַחֲכָמִים יְהוֹלֵל

וְקוֹל הַמְּרִיבָה אַיָּלוֹת יְחוֹלֵל

טוֹב לךָ לְהַחֲרִישׁ וּלְדַבֵּר דִּבְרֵי עֲרֵבוּת

וְלוֹמַר כִּי הִשְׁאַלְתָּ לָנוּ הַסְּפָרִים דֶּרֶךְ נְדִיבוּת

וַאֲשֶׁר יָדוֹ מֵהוֹשִׁיעַ לֹא תִקְצָר

יַרְחִיב לְךָ בַּצָּר

וְיַהֲפֹךְ מִדּוֹתֶיךָ לְטוֹב

וְהָיִיתָ כְּגַן רָטוֹב

כְּנֶפֶשׁ אֲשֶׁר חֶבְרָתְךָ יוֹאֵל

וְאַהֲבָתְךָ דֹרֵשׁ וְשֹׁאֵל

נוֹטֶה אֵלֶיךָ שָׁלוֹם מְלֹא רֹחַב אַרְצְךָ עִמָּנוּאֵל.

וַיְהִי כִּשְׁמֹעַ הַשָּׂר הַמְּלִיצָה בְּמַשְׂמְרוֹת הַיֳפִי נְטוּעָה

אָז נִמְלָא שְׂחוֹק פִּיו וּשְׂפָתָיו תְּרוּעָה

וַיֹּאמַר חֵי אֲנִי כִּי גִבְעוֹת עוֹלָם לְפָנֶיהָ שַׁחוּ

וּמִדַּבְּרוֹתֶיהָ בְלַהֲבוֹת אֵשׁ הַשֵּׂכֶל הִתְלַקָּחוּ

וַיִּקֹּד אַרְצָה וַיִּשְׁתָּחוּ

וַיֶּעֱרַב עָלָיו שִׂיחִי

וַיֹּאמַר יֶשׁ לִי רָב אָחִי

זֶה קָרְבָּנִי לַחְמִי לְאִשַּׁי רֵיחַ נִיחֹחִי

וָאֵשֵׁב עִם הַשָּׂר מִתְעַנֵּג בְּדֹדָיו

אֹרֶה בְשָׂמָיו וּנְרָדָיו

וְאֹכֵל פְּרִי מְגָדָיו.
