עִם הָאֵקָלִפְּטוּס

1

שָׁלוֹם לְךָ, עֵצִי רְחַב-הָעֲנָפִים!

גִּדַּלְתָּ לְמַכְבִּיר גְּזָעִים

וְתִהְיֶה לִי לְמָגֵן

מֵעֵין אוֹיֵב אוֹרֶבֶת,

מִיָּד שְׁלוּחָה.

גַּם הַלַּיְלָה…


גַּם הַלַּיְלָה אֶל גִּזְעֲךָ אֶצָּמֵד,

אֶת אָזְנַי וְעֵינַי אֲחַדֵּד

וּלְתוֹךְ הָאֲפֵלָה אוֹתָן אֶדְקוֹר

וְאֶשְׁמוֹר.


וְאֶשְׁמוֹר עַל שְׁנַת הַטַּף הַתְּמִימָה

עַל שְׁנַת הָאֲנָשִׁים הַכְּבֵדָה –

הַחֲבֵרִים לְעָמָל וְלִיצִירָה –

וְעַל פְּרִי עֲמָלָם.


הַלַּיְלָה, הָעֵץ, הֱיוּ בְּעֶזְרָתִי,

תְּסוֹכְכוּ, תָּגֵנּוּ עַל תִּקְווֹת אֲנָשִׁים,

עַל תִּקְווֹת שֶׁלֹּא תְּאֻכְזַבְנָה,

שֶׁלֹא תְּחֻלַּלְנָה

תִּקְווֹת.

2

מִלְּמַטָּה עַל הָאֲדָמָה, צְלָלִים רוֹחֲשִׁים.

מִלְמַעְלָה, מִבְּעַד הָעֲנָפִים, לוֹחֲשִׁים

כּוֹכָבִים.

רַחַשׁ.

לַחַשׁ.

נְשִׁיבַת רוּחַ – עֲנָפֶיךָ יֶהְגּוּ נְכָאִים,

נֶאֱנָחִים, חוֹרְקִים הַגְּזָעִים;

מִמְּרוֹמֵי הָרָקִיעַ מִתְגַּלְגְּלִים מַזָּלוֹת

וְנוֹפְלִים כְּזִרְמֵי דְּמָעוֹת –

אֲדֻמּוֹת.

בַּחֲשֵׁכָה מִתְגַּנֵּב תָּחַשׁ.

3

בָּאֹפֶק

הֵנִיף מְנִיפָה זוֹרֵק-אוֹר;

שֵׁנִי.

וַיִּדְהַר, וַיְגַמֵּא אֶרֶץ סוּס-זָהָב,

רַעְמָתוֹ עַרְפִלֵּי-כֶּסֶף מִתְאַבְּכִים,

וַיִּזְחַל הֶרָה – נָחָשׁ מִתְפַּתֵּל,

מַבְרִיק כַּפָּז, עֵינָיו רְשָׁפִים,

וַיָּנוּסוּ צְלָלִים מִפָּנָיו,

מְבֹהָלִים.

עֵדָה שְֹטָנִית.

דָּרֵי תֹּפֶת.


וּבַמֶּרְחַקִּים,

בִּשְׁחוֹר בִּצַּת הַלַּיִל,

כְּשַׁלְפּוּחִיוֹת

מִתְבַּקְעוֹת יְרִיּוֹת,

וְלֶהָבוֹת פּוֹרְחוֹת,

וְתַנִּים יְלֵל.

4

זוֹרְקֵי-אוֹר, לֶהָבוֹת, לְאוֹרְכֶם הַמֻּזְהָב,

  • מִי אֲנִי כָּאן, הַצָּמוּד לְגִזְעֵי הָאִילָן, -

זְבוּב מְפַרְפֵּר בֵּין קוּרֵי עַכָּבִישׁ?

עַכָּבִישׁ יוֹשֵב-מַאֲרָב?

עִם אוֹרְכֶם הַנִּקְלָע מִצָּפוֹן לְדָרוֹם

תִּתְנַדְנֵד גַּם נַפְשִׁי, תִּקָּלַע אֱלֵי תְּהוֹם,

תְּרַחֵף, תִּפְרוֹשֹ אֵבֶר אֶל אַחִים בּוֹדְדִים,

הָעוֹמְדִים בַּלֵּילוֹת, לִגְזָעִים צְמוּדִים…


פֶּרֶא, אִישׁ הַמִּדְבָּר!

הֲתֵחַת יוֹנָה מִתַּחְמָס

בַּהֲגִנָּהּ עַל חַיֵּי גּוֹזָלֶיהָ?

הַאִם תְּחַשֵּׁב תַּרְנְגֹלֶת כֹּחָהּ,

בִּפְרֹשֹ הַדּוֹרֵס אֶת צִלּוֹ

עַל אֶפְרֹחֶיהָ?

כָּל אֵם – לְבִיאָה מוּל הַשְּׁכוֹל,

כָּל אָדָם – גּוּר-אַרְיֵה מוּל הַיָּד הַגּוֹדַעַת

אֶת פְּתִיל הֶעָתִיד.

זְכוֹר זֹאת וְדַע – לֹא תַּדְבִּירֵנוּ!

הַלַּיְלָה, הָעֵץ הַלָּזֶה, הַמַּקְשִׁיב לְלִבִּי, -

סָהֲדִי, -

לֹא תַּכְנִיעֵנִי!

לֹא תַּכְנִיעַ אוֹתִי!

5

רָאִיתִי, בְּשִׁלְהֵי הַלַּיְלָה, בְּשִׁפּוּלֵי רָקִיעַ,

חֲצִי-סַהַר – לִטְרַת כָּבֵד רָקוּב;

עֲרָפֶל קַל מִדָּם עָקוּב,

עַל הַצֵּל;

וַעֲדַת-עוֹרְבִים שְׁחוֹרָה,

מִסָּבִיב לוֹ,

יוֹצֵאת בְּמָחוֹל,

חָגָה חוּגִים

וְעוֹנֶדֶת

זֵר לוֹ

שָׁחוֹר;

וְהִנֵּה רוֹקֶדֶת

כִּמְטֹרֶפֶת –

“מִסְפָּרַיִם” –

וּפִתְאֹם

נִפְרֶדֶת

לִמְחוֹל מַחֲנָיִם.

וּבְדוּמִיָּה

לוֹבֵשׁ וּפוֹשֵׁט צוּרָה

הָרִקּוּד.

זְבוּל מְכַשֵׁפָה,

סִיּוּט.

דִּמְמַת שְׁאוֹל -.

בַּכֹּל.

בַּשָּׁמַיִם –

לִטְרַת כָּבֵד רָקוּב,

מַשַּק כְּנָפַיִם,

מָחוֹל שָׁחוֹר

וַעֲרָפֶל מִדָּם עָקוּב.

6

רָאִיתִי מֵאַפְסַיִם חוֹפֵן גַּרְגְּרִים הִגִּיהַ,

אַחֲרָיו שֵׁנִי, שְׁלִישִׁי – מְלוֹא חָפְנַיִם אוֹר;

נָבַל הָאֹדֶם, הָעֲרָפֶל הָפַךְ אָפוֹר,

מֵהַזֵּר נָשְׁרוּ אֶחָד אֶחָד פִּרְחֵי הַשְּׁחוֹר

וַיְּהִי – אוֹר!

7

שָׁלוֹם לְךָ, עֵצִי, רְחַב-הָעֲנָפִים!

גִּדַּלְתָּ לְמַכְבִּיר גְּזָעִים

וַתִּהְיֶה לִי לְמָגֵן

מֵעֵין-אוֹיֵב אוֹרֶבֶת,

מִיָּד שְׁלוּחָה

גַּם הַלָּיְלָה.

אוגוסט 1936
