המעשה בבחור ספיר, בשפיר אהובו וברשע הזקן האשמאי

נְאֻם לְמוּאֵל בֶּן־אִיתִיאֵל: הָיָה אִישׁ בְּאֶרֶץ לְבָאִים / מְצוֹדֵד בְּתוֹכָם צְבָאִים / וְשֵׁם הָאִישׁ סַפִּיר מִגִּבְעַת יְסוֹד תַּרְשִׁישׁ / וְשֵׁם אִמּוֹ תַּרְשִׁישׁ / וַיְהִי כָּל־הַיָּמִים צִפֳּרִים צוֹדֶה / אִישׁ יֹדֵעַ צַיִד אִישׁ שָׂדֶה. / וַיִּבְחַר לוֹ הַצְּבָאִים וְהַצְּבָאוֹת / מִבֵּין פְּרִיצֵי הַחַיּוֹת / כֹּה יַעֲשֶׂה סַפִּיר וְכֵן מִשְׁפָּטוֹ, וַיְהִי לוֹ קֶשֶׁת וְכִידוֹן וְכָל־כְּלֵי־נֶשֶׁק / וְהָיָה גָּדוֹל מִכָּל־בְּנֵי־חֵשֶׁק / וַיֵּלֵךְ הוּא וְכָל־גְּדוּדֵי חִשְׁקוֹ וּפָשַׁט עַל הַצְּבִי / וַיִּשְׁבְּ מִמֶּנּוּ שֶׁבִי / וַיָּבֵא צְבָאִים וּצְבָאוֹת צְרוּרִים / מִמְּעוֹנוֹת אֲרָיוֹת מֵהַרְרֵי נְמֵרִים.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֲנִי צוֹדֶה וְלִי תֵאוֹת צְבִיָּה / אֲשֶׁר אַצִּיל צְבִי מִפְּנֵי לְבִיָּא

וְאֵי צַיָּד אֲשֶׁר יָצוּד כְּצֵידִי / בְּמִרְבַּץ הָאֲרָיוֹת הַצְּבִיָּה

אֲבָל עֵינֵי צְבָאִים יַעַזְרוּנִי / לְהַכּוֹת בָּם וְאֵין קֵץ לַגְּוִיָּה

אֲנִי אַפִּיל וְאָמִית אַחֲרֵי עֵין / צְבִי כִּי הִיא כְּמוֹ קֶשֶׁת רְמִיָּה

וְאִם אַחַת עֵין צְבִי וְאָז / יְשִׁיבֵהוּ אֲרִי הוֹסֵף שְׁנִיָּה.


וַיַּעַבְרוּ הַצְּבָאִים וְהַצְּבָאוֹת לְפָנָיו / וְהִנֵּה אֶחָד מֵהֶם נָשָׂא חֵן בְּעֵינָיו / וּשְׁמוֹ שַׁפִּיר וְשֵׁם אֲחוֹתוֹ צְבִיָּה. / וַיֶּאֱהָבֵהוּ מְאֹד וַיְהִי סְבִיבוֹ / כִּי אַהֲבַת נַפְשׁוֹ אֲהֵבוֹ / כִּי לֹא הָיָה כְתִפְאַרְתּוֹ / בַּצְּבָאִים אֲשֶׁר גָּדְלוּ אִתּוֹ / וַיְשִׂימֵהוּ עַל־מִשְׁמַעְתּוֹ / וַיַּשְׁכִּיבֵהוּ בְחֵיקוֹ / וַיַּאֲכִילֵהוּ מִפִּתּוֹ / וַיִּקַּח הַצְּבִי בְּחֶרְמוֹ וְיַאַסְפֵהוּ בְּמִכְמַרְתּוֹ / וַיְהִי חִשְׁקוֹ הֹלֵךְ וְגָדֵל, וְהָיָה אִם יָצָא מֵאִתּוֹ אָנֶה וָאָנָה יָצְאָה נַפְשׁוֹ / כִּי קְשׁוּרָה בְנַפְשׁוֹ / וּבְלֶחְיוֹ זוֹרֵחַ שִׁמְשׁוֹ / וְלֵילוֹ שְׂעַר רֹאשׁוֹ / וְהָיָה כִּי הֱצִיקַתְהוּ אַהֲבָתוֹ / וַתָּחֶל לְעַנּוֹתוֹ / וַיֹּאמַר ״אֵיךְ חִשְׁקִי עַתָּה הֶלְאָנִי / הֲלֹא־מְשַׂחֵק אָנִי״. / וַיַּעֲנוּהוּ יִתְדוֹת הַחֵשֶׁק אֲשֶׁר בְּלִבּוֹ נִתְקָעוּ / וּמַסְמְרוֹת הָאַהַב אֲשֶׁר בְּנַפְשׁוֹ נִטָּעוּ / וַיֹּאמְרוּ ״הֲלֹא בַתְּחִלָּה הִתְרַפִּיתָ / עַתָּה צַר כֹּחֲךָ וְהִתְעַנִּיתָ / אִם שִׂחַקְתָּ פַּעַם בָּאַהֲבָה / וּתְשַׂחֵק בְּךָ גַם־הִיא רְבָבָה״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

נְקַלָּה אַהֲבָה מֵרֹאשׁ / וְתוֹסִיף עַד אֲשֶׁר תִּכְבַּד

וְאִם תִּכְבַּד בְּלֵב חוֹשֵׁק / תְּעַנֵּהוּ עֲדֵי יֹאבַד

וְיֵשׁ דָּבָר בְּרֹאשׁ נִבְזֶה / וְהוּא בָאַחֲרִית נִכְבָּד

הֲלֹא לֹוֶה אֲסִיר מַלְוֶה / וְחוֹשֵׁק לַצְּבִי נֶעְבָּד.


וְאִם בָּא לְשַׁעַשְׁעוֹ בְּזֹהַר פָּנָיו / נִחֲתוּ בְלִבּוֹ עֵינָיו / וַיְבַהֲלוּהוּ רַעְיוֹנָיו / וַיִּתְקַבְּצוּ לְהִלָּחֵם בּוֹ גְּדוּדֵי חִשְׁקוֹ וַחֲיָלָיו / וַיִּירְאוּ מִגֶּשֶׁת אֵלָיו / וַיַּרְא כִּי לֹא יוֹעִילוּ כְלֵי נִשְׁקוֹ / וַיִּקְרָא לְכָל כְּלֵי חִשְׁקוֹ / וַיִּוָּעֲצוּ אַנְשֵׁי הַמְּלִיצָה / וְלֹא מָצְאוּ עֵצָה / וַיְהִי מַשְׁגִּיחַ מֵחַלּוֹנוֹת מֵצִיץ מֵהַחֲרַכִּים / וְהִנֵּה בְּפָנָיו מֵימֵי הַחֵן מְפַכִּים.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בִּלְחִי צְבִי מֵי־חֵן וּמֵי / יֹפִי בְּפָנָיו נִגְּרוּ

יִשָּׂא מְאוֹרֵי רוֹם וְכָל / אוֹרֵי שְׁחָקִים יֶחְפְּרוּ

עֵינָיו אֲרָיוֹת יִמְחֲצוּ / כָּל־יוֹם חֲיָלִים יִגְבְּרוּ

עַל־לֶחְיוֹ שֶׁמֶשׁ וְכָל / שׁוֹשָׁן בְּלֶחְיוֹ יֶאֱרוּ.


וַיְהִי עַד־כֹּה וַיַּעֲמֹד [לְפָנָיו אִישׁ] מְמֻשָּׁךְ וּמְמֹרָט / גָּר בְּיַרְכְּתֵי אֲרָרָט / בָּרְבִיעִי בְשַׁבָּת / וּשְׁלֹשָׁה עָשָׂר לְחֹדֶשׁ שְׁבָט / וַיְהִי כַאֲשֶׁר הַצְּבִי נִבָּט / מָט וְהִתְמוֹטָט / קָט וְהִתְקוֹטָט שָׁם וְהִתְשׁוֹטָט / וְלֹא נִשְׁקָט / וַיְּהַלֵּךְ אַט / וְשֵׁם הָאִישׁ בִּרְשַׁע / בֶּן־מֵישַׁע / וַיִּשְׁתַּחוּ וְעַל בִּרְכָּיו כָּרַע / וְלִבּוֹ נִפְרַע / וְנַפְשׁוֹ קָצְרָה מֵהִשְׂתָּרַע / וְיָבֵשׁ חִכּוֹ וְרֻקּוֹ לֹא בָלַע / וּמָרַט רֹאשׁוֹ / וְקָרַע לְבוּשׁוֹ / וַיִּבָּהֵל הַנַּעַר / וַיֹּאמֶר לוֹ ״וְלָמָּה זֶּה הַסַּעַר״ / וַיְהִי בְּשָׁמְעוֹ מִלָּיו / וַתָּשָׁב רוּחוֹ אֵלָיו / וַיֹּאמֶר לוֹ ״בֵּן הָיָה לִי / וַיִּבְרַח מֵאִתִּי וַיֵּלֵךְ / כְּתֹאַר בֶּן־מֶלֶךְ / עַל כֵּן בָּכִיתִי לְנוּדוֹ מִזְּבוּלוֹ״ / כִּחֵשׁ לוֹ / אַחַר־כֵּן שָׁאַל בִּרְשַׁע בִּשְׁלוֹמוֹ / וַיְבָרְכֵהוּ וַיְפַתֵּהוּ / אוּלַי אֲחָרָיו יִמְשְׁכֵהוּ / וַיִּשָּׂא הַנַּעַר אֶת עֵינוֹ / וַיַּרְא אֶת זְקָנוֹ / וַיִּתְמַהּ בַּהֲדָרָהּ / כִּי הָפַךְ לְלָבָן שְׂעָרָהּ. / לְבָנָה כְשַׁחַר / גִּזַּת־צֶמֶר צַחַר / אֲרֻכָּה רְחָבָה עֲמֻקָּה / בּוּקָה וּמְבוּקָה וּמְבֻלָּקָה / נְכוֹנָה לַעֲלוּקָה / מְכֻבֶּסֶת בַּנֶּתֶר וּבֹרִית / מְדוּרָתָהּ אֵשׁ וְגָפְרִית / בִּרְאוֹתוֹ דְבַר הַזָּקָן / חָשַׁב בּוֹ הַכֹּל מְתֻקָּן / וַיֹּאמַר: אֵין זֶה כִּי אִם שׁוֹפֵט הָרָאוּי לְדַיָּנוּת / וְכֶרֶם וְזָקָן חֲצִי הַרַבָּנוּת / וַיַּחֲשֹׁב כִּי הוּא דַיָּן וּמְהִיר צֶדֶק / וְהוּא כִּמְשֻׁכַת חֶדֶק / וַיֹּאמַר: כַּבֵּד אֲכַבְּדֶנוּ / וּמִשִּׁירִי אֲהוֹדֶנוּ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

זָקֵן זְקָנוֹ זָקְנָה בַתֹּם / מֵימֵי תְבוּנָה הֶאֱרִיכוּהָ

שִׁלְּחוּ קְצִירֶיהָ כְּבוֹד חָכְמָה / עַל־נַהֲרוֹת צֶדֶק מְשָׁכוּהָ

אִם־תַּחֲשֹׁב יוֹם לַעֲשׂוֹת אָוֶן / רָחְבָּהּ וְאָרְכָּהּ הֵם חֲשָׂכוּהָ.


וַיֹּאמֶר לוֹ הַנַּעַר:״הִנֵּה נַפְשִׁי דָבְקָה בְךָ וְתִרְצֶךָ / הַגִּידָה נָא לִי אֵי־מִזֶּה עַם אַתָּה אַיֵּה אַרְצֶךָ״? / וַיַּעֲנֵהוּ לֵאמֹר: ״גֵּר הָיִיתִי בְאֶרֶץ אֲרָרָט / וּתְשׁוּבָתִי לִסְדֹם לְמִשְׁפָּט / וּמֵעִיר בִּרְשַׁע בֶּן־מֵישַׁע / הַשּׁוֹפֵט סְדֹם עַד־לֶשַׁע / וְשָׁם בַּשָּׁלִישׁ / דֶּרֶךְ לֹא עָבַר בָּהּ אִישׁ״ / וַיֹּאמֶר לוֹ:״אִם אַתָּה מִסְּדֹם / מִי הֵבִיאֲךָ הֲלֹם״? / וַיִּשְׁאַג שְׁאָגָה / בְּרָגְזָה וּבִדְאָגָה / וַיֹּאמַר: ״הֲלֹא נְתִיבוֹתַי דָּרַכְתִּי / וָאֳרָחוֹת עֲקַלְקַלּוֹת הָלַכְתִּי / וּבָאתִי יְעָרוֹת / וְעָבַרְתִּי נְהָרוֹת / וְיָרַדְתִּי בְקָעוֹת / וְשָׁפַכְתִּי רֹב דְּמָעוֹת / וְיָרַדְתִּי יַמִּים עַל יַמִּים וּבָאתִי צִיּוֹת / שַׁמּוֹת וּשְׁאִיּוֹת שָׁמָּה נִתְקַבְּצוּ דַיּוֹת / וְלֹא מָצָאתִי מְנוּחָה / בְּאֶרֶץ עֲרָבָה וְשׁוּחָה / וְהָלַכְתִּי בְמִדְבַּר שְׁמָמָה / בְּרָעָב וּבְצָמָא / וְאֵין כָּל בַּיְּעָרִים / כִּי אִם הַר הַפֻּרִים / וּבָאתִי עִיִּים וְאִיִּים / וְאִיֵּי כִתִּיִּים / וְכִמְעַט נִגְזַרְתִּי מֵאֶרֶץ הַחַיִּים / מֵעַם אֶל עַם / נֶחֱרָף וְנִזְעָם / וּבָאתִי אֶל אֶרֶץ פְּלִשְׁתִּים וְתַרְשִׁישׁ וְכִתִּים / אֲשֶׁר שָׁם אֲרָם נְחִתִּים / חֲוִילָה וְדִקְלָה / וְתַרְשִׁישׁ וְרִבְלָה / וְאַיָּלוֹן וְיִתְלָה / מָדַי וְיָוָן / וְדִישׁוֹן וְעַלְוַָן / וְעַכְבּוֹר וְזַעֲוָן / וְצִבְעוֹן וְדִישָׁן / וְחֶמְדָּן וְאֶשְׁבָּן וַעֲנָה וַעֲקָן / וְשָׁם רִבִּיתִי זֶה הַזָּקָן / הֲדוֹרָם וְיָקְטָן / וְשׁוֹבָל וְלוֹטָן / וְהוּא אָחִי הַקָּטָן / וְדוֹדָנִים וְצִידוֹנִים / וְכַפְתּוֹרִים וּמִצְרִים / וְהָעַרְקִי וְהַסִּינִי וְאַרְוָדִי וְעַרְבִים / וְאַלְּמוֹדָד וְאַרְפַּכְשַׁד אֲבִימֶלֶךְ וְכִתִּיִּים / וּבָאתִי עַד גְּרָר וְעַד חוּל תֹּגַרְמָה / וּפְלִשְׁתִּים סְדוֹמָהּ וַעֲמֹרָה עַד גְּבוּל אַדְמָה / וְלִצְבֹאִים דָּרַכְתִּי כָל־אֵלֶּה הַדְּרָכִים / וְהַגּוֹיִים וְהַמְּלָכִים / וְהָלַכְתִּי חֲשֵׁכִים / וְרַעְיוֹנַי נְבוּכִים / עַד אֲשֶׁר זָקַנְתִּי וְשַׂבְתִּי / וּמִדַּרְכִּי לֹא שַׁבְתִּי / וְשָׁאַלְתִּי מֵעִיר אֶל עִיר / גָּדוֹל וְצָעִיר / וְכֹה אָמַרְתִּי בְלֶכְתִּי: הַרְאִיתֶם צְבִי־תֹאַר יְפֵה־עַיִן / וַיֹּאמְרוּ: אָיִן! / וְעַתָּה אֲנִי הוֹלֵךְ בְּמַר נַפְשִׁי וּבְחֹם לְבָבִי / וְאַבִּיט וְהִנְּךָ סְבִיבִי״ / וַיַּעַן צְבִי וַיֹּאמַר ״לָמָּה תְדַבֵּר עוֹד דְּבָרֶיךָ / הַשְׁמִיעֵנִי מִטּוּב שִׁירֶיךָ.״


וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בְּחֵי אַהַב רְאִיתֶם דּוֹד / בְּלִבּוֹת אוֹהֲבִים יֶאְרַח

מְאוֹר חַמָּה בְּפָנָיו שָׂם / וְשָׁם יֶעְרַב וְשָׁם יִזְרָח

וְחִשְׁקוֹ שָׂם בְּלֵב כָּל אִישׁ / בְּיוֹמוֹ צָץ וְגַם פָּרַח

וְרַבּוּ אֹהֲבָיו מֵחוֹל / וְחִשְׁקִי יֶעֱרַב יִבְרַח

וְהוּא נָקֵל עֲלֵי לִבִּי / וְנַפְשִׁי הָיְתָה טֹרַח

בְּכָל נָתִיב אֲבַקֵּשׁ דּוֹד / בְּכָל־דֶּרֶךְ בְּכָל־אֹרַח

נְתִיבוֹת לֹא עֲבָרָם אִישׁ / לָחֹרֶב יוֹם וְלַקֶּרַח

אֲסוֹבֵב יָם וְלַדָּרוֹם / וְלַצָּפוֹן וְלַמִּזְרָח

אֲנִי חוֹשֵׁק עֲבָרוֹ יַיִן / יְדִידוּת ״דּוֹד״ בְּמַר אֶצְרַח

וְאָז פָּנָה צְבִי הַחֵן / וְאָז הָלַךְ וְאָז אֶזְרַח.


וַיֹּאמֶר לוֹ: ״הֲטִיבוֹתָ / מִלִּים נְכוֹנוֹת

הֲשִׁיבוֹתָ״ / וַיְהִי אַחֲרֵי כֵן וַיִּיקַץ סַפִּיר וַיֵּרָדַם /

וַיִּקְרָא לָעִיר אֵין שָׁם אִישׁ וְקוֹל אָדָם / וַיֶּחֱרַד חֲרָדָה /

וַיָּקָם בִּסְעָרָה וּבִרְעָדָה / וַיְחַפֵּשׂ וְלֹא מָצָא / וַיִּפֹּל

מְלֹא־קוֹמָתוֹ אַרְצָה / וַיָּקָם לִגְרֹד הָעֹפֶר / וַיֵּשֵׁב עַל הָאֵפֶר / וַיִּתְגֹּלֵל וַיִּתְעַטֵּף / וַיָּהָם לִבּוֹ

וַיִּתְעַלֵּף.


וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

חֻמִּי חֻמִּי / אִשִּׁי אִשִּׁי / חֵילִי חֵילִי / וּמְאוֹר שִׁמְשִׁי

דּוֹדִי דּוֹדִי / הוֹדִי הוֹדִי / בָּשְׂמִי צוּפִי / יַעְרִי דִּבְשִׁי

הוּא שַׁעְשׁוּעִי / מִכָּל רֵעִי / דְּבָרוֹ יֵינִי / פִּיו תִּירוֹשִׁי

אַךְ מֵאוֹכֵל / יָצָא מַאְכָל / כִּי לֻקַּח / מֵחַדְרֵי נַפְשִׁי.


וַיְהִי לָאָרֶץ דּוֹבֵק / וְהָאֲבָנִים חוֹבֵק / וְקוֹרֵא וְצוֹעֵק / כְּעֵגֶל מַרְבֵּק / וְלִבּוֹ בּוֹקֵק / נָעוּר וָרֵק / וְשִׁנָּיו חוֹרֵק / וְעֵינָיו בּוֹרֵק / וּשְׂפָתָיו פּוֹשֵׂק / וְהוּא יוֹרֵק / וּבִלְשׁוֹנוֹ לוֹקֵק / וַיֹּאמְרוּ: שָׂרֵי הָעֳפָרִים / אֲשֶׁר בְּנַפְשׁוֹ קְשׁוּרִים / וַיְדַבְּרוּ עַל לִבּוֹ / וַיְקִימוּהוּ מִן הֶעָפָר אֲשֶׁר שָׁם מְסִבּוֹ / וַיְהִי כַּאֲשֶׁר הֱצִיקוּהוּ בְּדִבְרֵיהֶם / וַיְפַתּוּהוּ בְנִכְלֵיהֶם / וַיָּנָס מִפְּנֵיהֶם / וַיִּרְחַק מֵעֲלֵיהֶם / וַיִּבְרַח אֶל הֶהָרִים וְאֶל הַגֵּאָיוֹת / וַיֵּלֵךְ אֶל הַחַיּוֹת / וַתֵּצֵא אִמּוֹ וְאַחְיוֹתָיו / וּצְבִיָּה אֲחוֹת שָׂעִיר אֶל מִשְׁפְּחוֹתָיו / וְכָל הַצְּבָאִים אֲשֶׁר גָּדְלוּ אִתּוֹ וְכָל־מְתֵי מַחְשְׁבוֹתָיו / וַיְבַקְשׁוּהוּ בְּכָל הֶהָרִים / וּבַפְּתָחִים וּבֵין הַגְּדֵרִים / וּבְגֵיא הַצְּבֹעִים / וּבֵין נְקִיקֵי הַסְּלָעִים / וַיִּמְצְאוּהוּ עַל־שְׂפַת נַחֲלֵי צְבָאֵי הַיְדִידוֹת / וְהוּא בוֹכֶה עַל הַפְּרִידוֹת. / וַיִּשָּׂא עֵינָיו וַיִּרְאֵם / וְעֵינָיו כְּעֵינֵי רְאֵם / וְלֹא נִשְׁאַר בּוֹ כִּי אִם יְפֵה עֵינַיִם / וְזֹהַר הַלְּחָיַיִם / וַיְהִי כִּרְאוֹתָם בָּרַח לְרֹאשׁ הַגִּבְעָה / מִתְנַקֵּשׁ בְּדִמְעָה / וַיִּקְרְאוּ בַלָּט בְּטוּב מַעֲנֶה / וְאֵין קוֹל וְאֵין עֹנֶה. / וַיְּחַלּוּ פָנָיו / וַיַּסֵּב מֵהֶם עֵינָיו / וַיַּעַן אֶחָד מֵהַנְּעָרִים / וַיַּשְׁבִּיעֵהוּ בְחֵי מִבְחַר הָעֳפָרִים / וַיִּשְׁמַע וַיִּנְאַק / וְלֹא הִתְאַפָּק.


וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אָץ כַּצְּבִי כָב לַהֲבִי / כִּי אֵשׁ בְּלִבִּי נִשְּׂקָה

לֹא מֵי נְחָלִים יִשְׁטְפוּ / אִשִּׁי לְבַד דּוֹד יֶחְשְׁקָה

כִּי מִי יְרַפֵּא מַחֲלָה / רִפְאוּת מְזוֹרָהּ רָחֲקָה

רִגְבֵי נְחָלִים שׁוֹטְפִים / תַּחַת חֲבוֹק דּוֹד אֶחְבְּקָה.


וַיִּפֶן אָנֶה וָאָנָה / וַיְדַבְּרוּ אֵלָיו וְלֹא עָנָה / וַיַּרְא עֲדַת צְבָאִים / וְלַהֲקַת פְּרָאִים / וַיָּרָץ אֲלֵיהֶם / וַיֵּלֵךְ עִמָּהֶם, / וַיָּשׁוּבוּ גַּם אֵלֶּה לְאָהֳלֵיהֶם / וַיִּהְיוּ כָל יוֹם מְבַקְּשִׁים לְשַׁחֲרוֹ וּלְבַקְשׁוֹ / אוּלַי יְחַיּוּ אֶת נַפְשׁוֹ / וְהָיָה אִם רָאָם / יָנוּס מִפְּנֵיהֶם כִּי שְׂנֵאָם. / וַתְּהִי לוֹ אָחוֹת קְטַנָּה / יָפָה כַלְּבָנָה / אֲחוֹת צְבִיָּה וּשְׁמָה אֶקְדָּח / וּבְיָפְיָהּ אֵשׁ בַּלְּבָבוֹת תִּקְדַּח / הִיא תִקְרָא בִשְׁמוֹ / וַתָּשֶׂם בַּסַּלִּי בְּסֶלַע לַחְמוֹ / וַיִּקְרַב הוּא וְהַצְּבָאִים בַּלָּט / וַיֹּאכְלוּ וַיֹּאכַל גַּם הוּא מְעַט / וְהָיָה בִּקְרֹב אִישׁ יִרְאֵהוּ כִּי בְרִירוֹ יִשְׂחַק / וְגָעַר בּוֹ וְנָס מִמֶּרְחָק / וְשָׁאַג וְדָאַג וְזָעַק כְּגוּר אֲרָיוֹת נָעַר / וּכְעֵגֶל מַרְבֵּק גָּעָה וְגָעַר / וּפִיהוּ לָרוּחַ פָּעַר / וַיֵּרֵד אֶל הַנַּחַל / וַיִּקְרָא כְקוֹל שַׁחַל. / וַיְהִי הַיּוֹם וַתֵּצֵא אֲחוֹתוֹ צְבִיָּה וַתִּקְרָא בַחֲנוּנֶיהָ / וּבְטוּב שִׁירֶיהָ וּרְנָנֶיהָ / וַיִּפֶן אֵלֶיהָ / וְלֹא הֵשִׁיב לְמִלֶּיהָ / וַתַּשְׁבִּיעֵהוּ בְחֵי הַצְּבִי שַׁפִּיר, וַיִּפֹּל עַל הַאֲדָמָה / וַיַחְשְׁבוּ כִּי לֹא נוֹתְרָה בּוֹ נְשָׁמָה / וַיִּקְרְבוּ אֵלָיו בּוֹכִים וְסוֹפְקִים כַּפַּיִם / וַיַּזּוּ עַל פָּנָיו הַמַּיִם / וַיְהִי כְמֵת עַד עֲלוֹת הַיּוֹם עָלָיו / וַתָּשָׁב רוּחוֹ אֵלָיו / וַיָּנָס מִפְּנֵיהֶם וַיִּתְחַלְחַל / וַיָּבֹא אֶל תּוֹךְ הַנַּחַל.


וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

הָהּ עַל־נְדוֹד דּוֹד הָהּ עֲלֵי / נַפְשׁוֹ לְנוּדוֹ זוֹעֲפָה

כָּל־יוֹם תְּצַפֶּה אֶל־מְקוֹם / דּוֹדָהּ וְרוּחַ שָׁאֲפָה

לֹא יָדְעָה כִּי חָשְׁקָה עַד־נוֹד יְדִיד הִתְנַגְּפָה

הִנָּהּ בְּתוֹךְ נָהָר וְהִיא / בִצְְבָא צְבִי הִתְעַטְּפָה.


וַיָּבוֹא אִישׁ אֶחָד מִקְּרוֹבָיו / וַיֹּאמַר ״אֲנִי אֶרְפָּא אֹהֲבָיו״ / וַיְמַהֲרוּ וַיַּבְהִילוּהוּ / וְלִשְׂפַת הַנַּחַל הוֹבִילוּהוּ / וַיִּמְצְאוּהוּ כּוֹתֵב בְּרִירוֹ / וְזוֹכֵר הַצְּבִי חֲבֵרוֹ / וַיִּרְאֵם וַיָּקָם לָנוּס וַיְבַקֵּשׁ וַיִּקְרָא לוֹ הָאִישׁ ״לֵךְ [וְאַרְאֶךָ] אֶת הָאִישׁ אֲשֶׁר אַתָּה מְבַקֵּשׁ״ / וַיְהִי כְּשָׁמְעוֹ מִלָּיו / וַיְחִי וַיָּקָם עַל רַגְלָיו / וַיֵּלֵךְ הוּא וְהָאִישׁ הַמְרַגֵּל / וְהִנֵּה הַצְּבִי וּבִרְשַׁע עַל עֵין רֹגֵל / וַיְהִי כִּרְאוֹתוֹ הַצְּבִי / וַיִּטְרְפֵהוּ כְלָבִיא / וַיְחַבְּקֵהוּ וַיִּפֹּל עַל צַוָּארָיו וַיִּשָּׁקֵהוּ וַיִּבְכּוּ / וּלְחָיֵי בִרְשַׁע הִכּוּ / וַיְעַנּוּהוּ וַיְשִׂימוּהוּ בַמִּשְׁמָר / עַל כִּי נַפְשׁוֹ הֵמַר / וַיַּעַן הַצְּבִי שַׁפִּיר מִתְאוֹנֵן / מִתְנַפֵּל וּמִתְחַנֵּן / וַיֹּאמַר: ״הִנְנִי נַעַר קָטֹן / אַל יַחְשָׁב לִי אֲדֹנִי עָווֹן / כִּי רָאִיתִי זָקָן וְשֵׂיבָה / וּמַחֲלִיק אֲמָרָיו כִּנְקֵבָה / וַיִּקָּחֵנִי בְחֶלְקַת אֲמָרָיו / וּבְתַחֲנוּן דְּבָרָיו / הֲלֹא יָדַעְתִּי כִּי זְקָנוֹ חֲרָפוֹת / וַיְצִיבֵנִי לְמַדְחֵפוֹת / וַיִּתֶּן לִי מִזִּמְרַת אַרְצוֹ וְכִבְּדַנִי / וְאַחֲרָיו מְשָׁכַנִי / וָאָבֹא הַיּוֹם אֶל הָעַיִן / וְאָדָם אַיִן / עַד כִּי עָלֵינוּ נַחְתָּ / לְמוֹשִׁיעַ לִי שֻׁלַּחְתָּ״ / וַיֹּאמַר: ״לֹא אַזְכִּיר עוֹד אֶת עֲווֹנֶךָ / אַחֲרֵי רְאוֹתִי אֶת פָּנֶיךָ.״ / וַיִּשְׂמַח לִבּוֹ בְדוֹדוֹ / וַיָּגֶל כְּבוֹדוֹ / וַיַּרְְא אֶת הָעַיִן / וְהִיא רַחֲבַת יָדַיִם / וּמֵימֶיהָ מְתוּקִים / יַשְׁקוּ בְקָעוֹת וַעֲמָקִים / וְעַל שְׂפָתֶיהָ עֲצֵי פְרִי־טַעַם נְטוּעִים / יַשְׁקוּהָ לִרְגָעִים / וְשִׁלְּחוּ וְעָשוּ עֲנָפִים / וְכָל עוֹפוֹת בָּם מְצַפְצְפִים / וְהָעַיִן נְקִיָּה, כִּפְנֵי צְבִיָּה / חֲצָצֶיהָ לְבָנִים / אֲדֻמִּים גְּדוֹלִים וּקְטַנִּים / אָדְמוּ עֶצֶם מִפְּנִינִים.


וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

מַיִם עֲלֵי חָצָץ כְּתַרְשִׁישׁ נִגְּרוּ / עָרְבוּ לְשׁוֹתֵיהֶם וְלִמְאֹד מָתְקוּ

יִדְמוּ לְמֵי הַחֵן אֲשֶׁר בִּפְנֵי צְבִי / תִּרְאוּ עֲלֵי זֹהַר לְחָיָיו יִשְׂחֲקוּ

עַיִן סְבִיבֶהָ עֲצֵי פֶּרִי אֲשֶׁר / שָׁרְשָׁם בְּמַיִם גָּבְרוּ גַּם עָתְקוּ

יִדְמוּ אֱלֵי קוֹמוֹת צְבָאוֹת מִשְּׁדֵי / אִמּוֹת עֲדִינוֹת וַעֲלָמוֹת יָנְקוּ

קוֹל תּוֹר וּמִינֵי עוֹף וְעַיִט צִפְצְפוּ / עַל רֹאשׁ יְנִיקוֹת זֶה לָזֶה בָּם נָשְׁקוּ.

יִדְּמוּ אֱלֵי שָׁרִים וְשָׁרוֹת נִגְּנוּ / עַל רֹאשׁ עֳפָרִים חֵיק צְבָאוֹת חִבְּקוּ.


וַיָּבֹא בִרְשַׁע הַמְּעֻנֶּה / מִדִּבְרֵי סַפִּיר עִם שַׁפִּיר וַיְקַנֵּא / וַיִּמְשֹׁךְ לְפָנָיו וְאַחֲרָיו / וַיִּמַּסּוּ אֲסוּרָיו / וַיָּקָם עַל סַפִּיר הַצּוֹדֶה / וַיָּנֵצוּ שְׁנֵיהֶם בַּשָּׂדֶה / וַיֹּאמֶר בִּרְשַׁע: ״אַל נָא תְהִי בֵינֵינוּ מְרִיבָה / הִנֵּה־נָא הָעִיר הַזֹּאת קְרֹבָה / וְשָׁם שׁוֹפֵט אֱמֶת / יָעִיד עַל הַחַי כִּי הֲרָגוֹ הַמֵּת / הוּא יוֹכִיחַ בֵּינֵינוּ / יָשֵׁת יָדוֹ עַל שְׁנֵינוּ / וּשְׁמוֹ אַרְיוֹךְ / אֲשֶׁר דָּם נָקִי יִשְׁפֹּךְ״ / וַיָּבוֹאוּ עַד הָעִיר וּשְׁמָהּ פְּתָאִים / אֲשֶׁר אֵין שָׁם מְתִים / וַיָּבוֹאוּ עַד הַשּׁוֹפֵט מַטֶּה מִשְׁפַּט גֵּר־יָתוֹם / וְיְסַפְּרוּ לְפָנָיו דִבְרֵיהֶם וְקוֹרוֹתָם / וְכָל הַמּוֹצְאוֹת אוֹתָם / וַיִּחַר אַפּוֹ / וַיִּסְפֹּק כַּפּוֹ / וַיִּגְעַר אֶל בִּרְשַׁע לֵאמֹר: ״סוּר מִלְפָנַי נִקְלֶה וְנִבְזֶה / מִשְׁפַּט מָוֶת לָאִישׁ הַזֶּה״ / וַיִּקְרָא בְּקוֹל גָּדוֹל: ״הֲלֹא עַל זֹאת הָאֲדָמָה תֶּאְשַׁם / אָשָׁם הוּא אָשֹׁם אָשָׁם / עַל־כֵּן הַנַּעַר לֹא יֵלֵךְ עִמָּכֶם / וְאַתֶּם תֵּלְכוּ לְדַרְכְּכֶם״. / כִּשְׁמֹעַ בִּרְשַׁע בְּאָמְרוֹ: ״שִׁמְטוּ הַנַּעַר שָׁמֹט״ / וַיֹּאמַר: ״הָאֶחָד בָּא לָגוּר וַיִּשְׁפֹּט שָׁפֹט / הִנֵּה הַנַּעַר בְּנִי וַאֲנִי יְלִדְתִּיו וְהוּא בְנִי יְחִידִי / וְאַתָּה תַצִּילֶנּוּ מִיָּדִי״ / וַיִּשְׁמַע סַפִּיר כִּי הוּא בַנַּעַר מִסְתּוֹלֵל / וַתֵּצֵא רוּחוֹ וַיִּשְׁתּוֹלֵל / וַיְּהוֹלֵל וַיִּתְמוֹלֵל / וּבְלִי מַדָּע יְמַלֵּל / וּכְאִשָּׁה מְצֵרָה מְגַדֵּף וּמְקַלֵּל / וַיִּפֹּל אַרְצָה וַיִּתְהַפָּךְ / וְכַמַּיִם נִשְׁפַּךְ / וַיִּקְרָא לוֹ הַצְּבִי / ״אוֹהֲבִי אוֹהֲבִי״ / וְאֵין קוֹל וְאֵין – עֹנֶה / וְאֵין מַבִּיט וְאֵין פּוֹנֶה. / וַיִּקַּח הַצְּבִי שַׁפִּיר מְלֹא כַף מֵי אַהַב / וַיַז עַל פָּנָיו וַיִּמְעַט הַלַּהַב / וְעֵינָיו פָּקַח / וּכְנַף הַצְּבִי סַפִּיר לָקַח / וַיִּצְעַק מֵרֹב חֻמּוֹ / וַיֵבְךְ הַצְּבִי עִמּוֹ.


וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בְּנֵי־חֵשֶׁק בְּנֵי מַחְלָה / וְהֵם בִּלְבַד בְּנֵי שֶׁבֶר

בְּלִבּוֹת בַּעֲלֵי כָנָף / בְּכָל יוֹם יַעֲלוּ אֵבֶר

אֲשֶׁר לִבִּי כְּעוֹף יִדְאֶה / וְיִתְעוֹפֵף לְכָל עֵבֶר

חֳלִי־אַהַב חֳלִי כָבֵד /וְזוּלָתוֹ כְּצֵל עוֹבֵר.


עַד אֲשֶׁר יֵדַע הַסַּעַר / וּמַה־יִּהְיֶה לַנַּעַר / וַיֹּאמֶר הַצְּבִי שַׁפִּיר: / ״הֲרָאִיתָ הַזָּקֵן בִּרְשַׁע / כִּי [זְקָנוֹ] זְקַן רֶשַׁע / וְאָרְכָּהּ וְרָחְבָּהּ / הֲלֹא עָמָל וָאָוֶן בְּרָהְבָּהּ / וּמְרוֹרַת פְּתָנִים בְּקִרְבָּהּ / זֶה אֲדֹנִי הִקְנַנִי / וּבַעַל הַזָּקָן גְּנָבַנִי / וְהוּא רִמַּנִי״ / וַתַּחְמֹל עֵין הַשּׁוֹפֵט עַל־סַפִּיר וְעַל־אוֹהֲבוֹ / וַיְצַו לַאֲשֶׁר סְבִיבוֹ / וַיַּכּוּ לְבִרְשַׁע וַיִּמְחֲצוּהוּ / וּבֵין הַדְּלָתוֹת לְחָצוּהוּ / וַיָּמָת וַיִּסְחֲבוּהוּ / וּבְאַשְׁפַּתּוֹת הִשְׁלִיכוּהוּ.


וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וּמֵת בִּרְשַׁע בְּנוֹ מֵישַׁע / בְּחֵטְא אַדְמָה וְחֵטְא לָשַׁע

בְּרִשְׁעָתוֹ אֲשֶׁר רָשַׁע / וּבִפְשָׁעָיו אֲשֶׁר פָּשַׁע.


וְסַפִּיר עוֹדֶנּוּ שׁוֹכֵב בְּלֶב־יָם אַהַב / וּבִמְדוּרַת רַהַב / וְאֵינוֹ מַקְשִׁיב וְשׁוֹמֵעַ / וְלֹא מֵקִיץ וְיוֹדֵעַ / וַיִּקְרְבוּ אֵלָיו הַשּׁוֹפֵט וְהָעֹפֶר / וַיִּקְרְאוּהוּ עַד קָם מֵעַל הָאֵפֶר.


וַיִשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וְיָשַׁנְתִּי תְבוֹאֵנִי דְמוּת דּוֹד / וְרָאִיתִי תְּמוּנָתוֹ בְנוּמִי

הֲקִיצוֹתִי יְרִיבֵַנִי לְמַעַן / דְּמוּתוֹ אֶחֱזֶה־לִּי בַחֲלוֹמִי

עֲנִיתִיהוּ וְאִם־אֶחְטָא אֲקָרֵב / שְׂעִיר חַטָּאת לְפָנָיו עַל־אֲשָׁמִי

הֲכִינוֹתִי וְהִקְרַבְתִּי לְחַטָּאת / לְפָנֶיךָ סְגוֹר לִבִּי וְדָמִי.


וַיֹּאמֶר לוֹ שַׁפִּיר: / אֲדֹנִי הִתְבַשֵּׂר / וְכַעַס מִלִּבְּךָ הָסֵר״ / וַיַּגֵּד דִּבְרֵי הַשּׁוֹפֵט וְחֶמְלָתוֹ / וַיִּפֹּל מְלֹא־קוֹמָתוֹ / וַיֹּאמֶר לוֹ שַׁפִּיר: ״שָׁמַעְתִּי עָלֶיךָ כִּי מִשְׁפַּט גֵר תַּטֶּה / וְהַדַּל עַל־יַד רֶשַׁע תִּרְטֶה / וְלֹא עָשִׂיתָ עִמִּי כְּרִשְׁעֶךָ / וְתַסִּיג בְּרֹעֶךָ / וְאוֹתִי הֶחֱיִיתָ / מִטֶּרֶף בְּנִי עָלִיתָ״.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
