תחרות בין משוררים. משוררנו זוכה בתחרות זאת.

נְאֻם לְמוּאֵל בֶּן־אִיתִיאֵל: הָיִינוּ בְּמוֹשַׁב חֲכָמִים / וּבְסוֹד עֲרוּמִים / וְהָיִינוּ מְקַלְקְלִים פְּנֵי הַתְּבוּנָה / מְחַקְּרִים עֹמֶק הַתְּבוּנָה / וַיַעַן אֶחָד מֵהֶם אִישׁ מְשׁוֹרֵר / אִישׁ יְסוֹדוֹת הַשִּׁיר מְצָרֵף וּמְבָרֵר / וּשְׁמוֹ יְחִיאֵל בֶּן יְרַחְמִיאֵל / וַיֹּאמַר: הַרְאִיתֶם שִׁיר יְמִימָה אֲהוּבַת יָשְׁפֶה / וּמַה נָּעִים וּמַה יָּפֶה / עַל כֵּן עַל כָּל אַנְשֵׁי הַשִּׁיר הִיא מִשְׂתָּרֶרֶת / כִּי חֲנִית שִׁירָה עַל חֲמִישָׁה דִמְיוֹנוֹת בְּפַעַם אַחַת מְעוֹרֶרֶת. / וַיַּעַן הָאֶחָד אִם תְּעוֹרֵר חֲנִיתָהּ. / עַל חֲמִשָּׁה דִמְיוֹנוֹת בְּפַעַם אֶחָת / אָכֵן מִמֶּנָּה דָבָר לֹא נִכְחָד. / וָאֶעֱנֵם, אֲנִי נִשְׁעָן עַל שִׂכְלִי / סַפְּרוּ נָא לִי / וַיֹּאמְרוּ זֶה נֶאֱמַר / הַשִּׁיר אִם אֶחָד אִם שְׁנַיִם / וַיֹּאמְרוּ לִי בְהַלֵּל נָא נִקְרָאֵם שְׁנַיִם / וְהֵא לְךָ הַשִּׁיר / וְאִם תּוּכַל דִּמְיוֹנָם לְהַשִּׁיר / וַחֲקוֹר מִלָּיו וְדִמְיוֹנֵיהֶן וְאִם תּוּכַל [לָתֵת] רוֹכְבִים עֲלֵיהֶם.


וַיִּשְׂאוּ מְשָׁלָם וַיֹּאמְרוּ:

כְּחָלָב אוֹ כְעֵין קֶרַח / כְּשֶׁלֶג כַּכְּפוֹר בָּרָד.


וַיִּתְמְהוּ כָל בַּעֲלֵי הַשִּׁיר עַל טוּב שִׂיחָיו / וַיְּקַנְּאוּ בוֹ אֶחָיו / כִּי אָמְרוּ הִנֵּה אֲנַחְנוּ מְמַשְּׁלִים בַּחֲמִשָּׁה שְׁקָלִים / וְהִיא שְׁקָלָתָם בְּמִשְׁקָל אֶחָד וּמִדָּה אַחַת נְכוֹנָה / שֶׁקֶּצֶב אֶחָד לְכֻלָּהְנָה / וְזֶה מְדַבֵּר בְּקִנְאָה וְזֶה מְהַלֵּל בְּיִרְאָה / וְזֶה מַחֲרִישׁ בְּשִׂנְאָה / וְזֶה יָבֵשׁ גְּרוֹנוֹ בַצִּמְאָה / כִּי תָמְהוּ כֻלָּם עַל טוּב הַשִּׁיר וּמַעֲנָיו / כִּי נִבְהֲלוּ מִפָּנָיו. / וָאַעַן וָאֹמַר: לָמָּה נִבְהַלְתֶּם / וְעַל הַשִּׁיר תְּמַהְתֶּם / וְהִנֵּה אִם הַשִּׁיר נֶחְמָד מִזָּהָב וּמִפָּז / עַבְדְכֶם יֵלֵךְ וְהִכָּה אֶת הַמְּשׁוֹרֵר הַלָּז. / וַיֹּאמְרוּ לִי כְּמִתְלַהְלִים / וְלָמָּה אֲנַחְנוּ מַתְמִיהִים / אַתָּה תִהְיֶה לָנוּ לֵאלֹהִים, וַיִּסְגְּדוּ לִי כִסְגוֹד לַמֹּלֶךָ / וַיֹּאמְרוּ לִי אֵי חֲנִית הַמֶּלֶךְ / וְהָאֶחָד יְהַתֵּל בִּי / וְהַשֵּׁנִי יְשָׁרְתֵּנִי וַיֵּלֵךְ סְבִיבִי / וְהָאֶחָד אוֹמֵר בַּמֶּה תַכֶּה אַנְשֵׁי הַשִּׁירוֹת / וּבַעֲלֵי אֲמָרוֹת טְהוֹרוֹת / הֲתְנִיעֵמוֹ בְחֵילְךָ / הֲתוֹרִידֵמוֹ בְּמַקְלֶךָ / הֲתַבֵּט אַחַת אוֹ שְׁתַּיִם / הֲתַשְׁכִּיבֵנוּ בֵּין שְׁפָתַיִם. / וָאֶעֱנֵם בְּטוּב מַעֲנַי / כִּי לֹא בְחֶרֶב וּבַחֲנִית יוֹשִׁיעַ יְיָ / וַיַּעֲנוּנִי לֵאמֹר הֲלֹא זֶה עָשָׂה כָּמוֹהָ וְלָמָּה תִתְרָאֶה / וּרְאֵה וַעֲשֵׂה כְתַבְנִיתוֹ אֲשֶׁר אַתָּה מַרְאֶה / וּשְׁמֹר הָעִנְיָן וְהַדִּמְיוֹן / וְאַל תִּגְרַע מִן הַנִּקָּיוֹן:


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

כְּצֶמֶר אוֹ כְּמוֹ שַׁחַר / כְּדָר צִיץ אוֹ כְמוֹ וָרָד.


וַיְהִי כְשָׁמְעָם שִׁירִי הֶחֱשׁוּ / וּבְאָמְרָם כִּחֲשׁוּ / וְהִתְבּוֹשְׁשׁוּ וְהִתְקוֹשְׁשׁוּ / וּכְמִתְגַנְּבִים הִתְלַחֲשׁוּ / וַיִּכָּלְמוּ וְלֹא נִשְׁמַע קוֹלָם / וַיִּפְּלוּ חֲלָלִים כֻּלָּם. / וָאֹמַר אֲלֵיהֶם / וַהֲלֹא אֲמַרְתֶּם לִי / בְּשָׁמְעֲכֶם מְשָׁלִי / לֹא תוּכַל לוֹ / וַאֲמַרְתֶּם בִּרְאוֹתְכֶם הָעִנְיָן וְהַדִּמְיוֹן / עַד מָתַי תוּכַל נִקָּיוֹן / הִנֵּה יְכָלְתִּיו וְהִכִּיתִיו / וַתַּחֲנֶה מִלָּה נֶגֶד מִלָּה וְדִמְיוֹן נֶגֶד דִּמְיוֹן / כְּפַלְגֵי מַיִם בְּצָיוֹן:


מְשׁוֹרֵר אַךְ בְּכַף שִׁירַי / בְּיַד מִלַּי אֲאַבֵּד שָׁר

בְּיוֹם נִבְחַן נְגִיד שִׁירִים / הֲלֹא אָז מַעֲנִי יִכְשַׁר

וְאִלּוּ שָׁר כְּשִׁירִי שָׁר / אֲזַי בִּמְשׁוֹרְרִים אֻשַּׁר

וְאִישׁ יֹאמַר אֲשֶׁר נִמְצָא / כְּטוֹב שִׁירִי אֲזַי קָשַׁר

מְשׁוֹרֵר יִשְׁמְעָה שִׁירִי / כְּאוֹן מָצָא בְּאִישׁ עָשַׁר

עֲרֵב מִלָּה יְפֵה עִנְיָן / וְשָׁר מִשְׁקָל בְּלֵב נִקְשַׁר

לְמַעַן יֶחֱרַד שׁוֹמְעוֹ / הֲלֹא זֶה שִׁיר אֲשֶׁר יוּשָׁר.


וַיּוֹדוּ כֻלָּם וַיֹּאמְרוּ מֻשְׁלָם:

לְךָ הַשִּׁיר וְלָךְ יֵאוֹת / וְהִנֵּה זֶה לְךָ הָאוֹת

אֲשֶׁר הִתְחַבְּרוּ צִבְאוֹת / נְגִידֵי שִׁיר לְךָ הַלְאוֹת

נְפַצְתָּם בַּהֲדָר שִׁירָךְ / וְהִפַּלְתָּם לְמַשֻּׁאוֹת

וּמִטּוּבֵי רְנָנֶיךָ / כְּהַיּוֹם נָשְׂאוּ מַשְּׂאוֹת

עֲדֵי אָמְרוּ הֲבַמַּרְאוֹת / רְנָנֶיךָ הַנִּבָּאוֹת.


בִּרְאוֹתִי כִּי בְּפִיהֶם מְהַלְּלִים / וּבְקִרְבָּם מְקַלְּלִים / וּמִקֶּדֶם הָיוּ לְאוֹהֲבִים / וּבְשָׁמְעָם שִׁירִי נֶהְפְּכוּ לִי לְאוֹיְבִים / וָאֹמַר הִנְנִי מְשׁוֹרֵר בְּדִמְיוֹן שִׁשָּׁה / וְתֵרֶא אוֹיַבְתִּי תְכַסֶּהָ בוּשָׁה.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

לְשִׁשָּׁה אוֹהֲבִי דָמָה / וּבָהֶם כָּל יְצוּר כָּבַשׁ

כְּשֶׁמֶשׁ כַּכְּרוּב כִּצְבִי / כַּאֹדֶם כַּכְּפוֹר כִּדְבַשׁ.


וַיֹּאמְרוּ כֻּלָּם לֹא שָׁמַעְנוּ מְשׁוֹרֵר יָשִׁיר שִׁשָּׁה דִּמְיוֹנִים בְּמִשְׁקְלֵי מִשְׁקָלָיו / אַף כִּי זֶה הַמִּשְׁקָל [קַל] בְּרַגְלָיו / וְהָאֶחָד אוֹמֵר מִקֶּדֶם שְׁמַעְתּוֹ / וְהָאֶחָד אוֹמֵר גָּנוּב הוּא אִתּוֹ / לָמָּה תִתְהַלֵּל בְּשִׁיר אֲשֶׁר לֹא חִבַּרְתּוֹ / אֲשֶׁר לֹא עָמַלְתָּ בוֹ וְלֹא גִּדַּלְתּוֹ. / וַיַּעַן יְחִיאֵל הַמְשׁוֹרֵר וַיֹּאמַר: יָשִׂים שִׁיר אַחֵר לְצִדּוֹ / וְיַעֲשֶׂה לוֹ עֵזֶר כְּנֶגְדּוֹ / וּבָזֶה יִוָּדַע כִּי הוּא שִׁירוֹ / וְהַשִּׁיר הַשֵּׁנִי יָקִים אֶת חֲבֵרוֹ.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

כְּעָשׁ כָּעָב כְּמוֹר כַּאֲרִי / כְּהֵימָן כַּצֳּרִי יֶחְבָּשׁ.


וַיַּעֲנוּ כָּל הָעָם יַחְדָּו וַיֹּאמְרוּ: לְךָ הַשִּׁיר וְאוֹתוֹ חִבַּרְתָּ / וְטוֹב הַדָּבָר אֲשֶׁר דִּבַּרְתָּ / אֲבָל בָּאֵר תְּבָאֵר הַשִּׁיר הָאַחֲרוֹן / בְּחָכְמָה וָדַעַת וְכִשְׁרוֹן.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

כְּעָשׁ נַעְלֶה כְּעָב נָדִיב / כְּרֵיחַ מוֹר כְּחֵיל אַרְיֵה

כְּמוֹ הֵימָן בְּחָכְמָתוֹ / בְּמַרְאֵהוּ שְׁחִין יִחְיֶה.


וַיְהִי כְּכַלּוֹתִי מֵהַשִּׁיר וָאַקְשִׁיב וְהִנָּם תְּהִלָּתִי יְחַוּוּ / וַיִּקְדוּ וַיִּשְׁתַּחֲווּ / וָאֹמַר אֲלֵיהֶם / הִנְנִי מוֹסִיף עֲלֵיהֶם / מָשָׁל אֶחָד וְהָיוּ שִׁבְעָה / וַיָּרִיעוּ עָלַי תְּרוּעָה / וַיֹּאמְרוּ לְךָ הַחָכְמָה / וְאִתְּךָ הָעָרְמָה / הָבָה נִתְחַכְּמָה.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


בְּשֶׁבַע אוֹהֲבַי נַעְלֶה / וְכִמְעַט בָּם יְהִי נָבִיא

כְּעָשׁ כָּעָב כְּמוֹר כַּאֲרִי / כְּהֵימָן כַּצֳּרִי כִצְבִי.


הֲלֹא אִישׁ אַתָּה מִי כָמוֹךָ רַב פְּעָלִים / הִנֵּה בְּשִׁשָּׁה דִמְיוֹנִים עַל כָּל יְצוּר תִּשְׂתָּרַע / וּבְשֶׁבַע לֹא יִגַּע בְּךָ רָע / אָמְנָם כָּל לֵב יִקַּח מֵאִמְרוֹתֶיךָ / יִשָּׂא מִדַּבְּרוֹתֶיךָ. / וַיְהִי כַּאֲשֶׁר הֵכִינוּ לִבָּם / אַחַר אֲשֶׁר קֵרְבוּ לִבָּם בְּאָרְבָּם / וַיַּעַן יְחִיאֵל הַמְשׁוֹרֵר וַיֹּאמַר: לָךְ יֵאוֹת עַל בַּעֲלֵי הַשִּׁיר לְהִשְׂתָּרֵר / וְהוּא מַכִּיר בְּסוֹד הַשִּׁיר וּבַדִּמְיוֹנִים / וִיחִיאֵל הָיָה אִישׁ גָּדוֹל לִבְנֵי הַשִּׁיר וּנְשׂוּא פָנִים. / וָאֹמַר אֲהַלֲלֶנּוּ בְשִׁשָּׁה דִמְיוֹנִים.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

קַל כַּצְּבִי עַז כַּאֲרִי מָתוֹק כְּצוּף / רֵיחוֹ כְּמוֹר יָדָיו כְּעָב מִלָּיו צֳרִי

מִנּוֹזְלֵי יָדָיו סְלָעִים דָּשְׁאוּ / כַּפּוֹ כְּעָב תַּמְטִיר וְתַטְרִיחַ בְּרִי

כַּפָּיו כְּשֹׁרֶשׁ וַעֲנָפָיו אֶצְבְּעוֹ/תָיו עַל אֲגַם מַיִם וְלֹא יָמִישׁ פְּרִי

דָּמָה בְעַנּוֹתוֹ כְּגֶבֶר אֵין אֱיָל / וּבְיוֹם קְרָב עִם לוֹחֲמָיו נִמְצָא אֲרִי

כָּל אִישׁ אֲשֶׁר מַמְרֶה אֲמָרַי בוֹ הֲלֹא / הוּא אִישׁ גְּאֵיוֹן אִישׁ מְדָנִים אִישׁ מֶרִי.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
