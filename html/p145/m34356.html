סדר עבודת יום הכפורים

רשות.


בָּרוּךְ אַתָּה לָנֶצַח,

הָאֵל הַבּוֹחֵר בָּנוּ,

חֻקֶּיךָ חוֹנַנְתָּנוּ,

בָּם נִמָּלֵט מֵרֶצַח.

לָנֶצַח נָרִים מֶצַח

נִצְחַק אֶל שׁוֹד וּלְפָרֶץ.


אֵין חֵקֶר אֶל־חַסְדֶּךָ:

הִנֵּה פָּתַחְתָּ שַׁעַר

אֶל־כָּל־חוֹטֵא וָבַעַר

לָשׁוּב בִּבְכִי נֶגְדֶּךָ.

לַדָּךְ תִּתֵּן יָדֶךָ,

כִּי־אֵין צַדִּיק בָּאָרֶץ.


וּבְעֹצֶם צִדְקָתֶךָ

הוֹאַלְתָּ צַוֹּתֵנוּ

עַנּוֹת כַּיּוֹם נַפְשֵׁנוּ,

וּבְצוֹם לָשׁוּב עָדֶיךָ:

אָז תֵּכַהּ מִסִּפְרֶךָ

מַזְכֶּרֶת חֵטְא וָרֶשַׁע.


אַף לִידִידְךָ אַהֲרֹן,

וּלְכָל־כֹּהֵן הָרַב,

צִוִּיתָ כִּי יִקְרַב

כַּיּוֹם אֶל־הָאָרוֹן;

יָרֹן, יַשְׁבִּית חָרוֹן;

וַתִּתֶּן־לָנוּ יֵשַׁע.


עַתָּה אֵין עוֹד כָּל־אֵלּוּ,

אֵין עוֹד שֵׁרוּת מִקְדָּשׁ;

הַר אֵל שָׂדֶה הוּדַשֹ,

כִּי חֵטְא וּמְרִי גָדֵלוּ;

חָדֵלוּ, הָהּ! חָדֵלוּ

עוֹלוֹת אַף־חַטָּאוֹת.


בַּמֶּה יִתְרַצֶּה־נָא

אֶל־הָאָדוֹן הָעֶבֶד

מַה־יַּעַשׂ נָא עַם כֶּבֶד

עָוֹן, אִם־רַב זָנָה?

אָנָה יִמְצָא אָמְנָה

לַמָּוֶת תּוֹצָאוֹת?


אִם־לֹא יִדְכֶּה יָשׁוֹחַ,

אֶת־חַטֹּאתָיו יַכִּיר,

סֵדֶר הַיּוֹם יַזְכִּיר,

יַזְכִּיר רֵיחַ נִיחֹחַ;

וּבְכֹחַ יָסִיר חֹחַ,

יָסִיר מַדְוֵי הַנָּפֶשׁ.


אוּלַי אִם־יַעַשׂ כֹּה

אֵל חַי לוֹ יִמָּצֵא,

אוּלַי לִבּוֹ יִרְצֶה,

אוּלַי יָאִיר חָשְׁכּוֹ;

מַלְכּוֹ יִפְרוֹס סֻכּוֹ,

תַּחְתָּיו יִשְׁקוֹט יִנָּפֵשׁ.


הִנֵּה כִי־כֵן אַף־אָנוּ

כָּכָה נַעַשׂ הַיּוֹם;

אוּלַי נִמְצָא פִדְיוֹם.

אוּלַי תִמָּצֵא לָנוּ;

כֻּלָּנוּ הִנֵּה בָאנוּ,

הַבִּיטָה כִּי נֶאֱנָחְנוּ.


הַבִּיטָה כִּי נֶאֱנָחָה

נַפְשֵׁנוּ, צוּר שָׁמָיִם!

מֵעַתָּה בִּשְׂפָתָיִם

קָרְבַּן אִשֶּׁה נִזְבָּחָה;

וּסְלָחָה אִם נִדָּחָה,

כִּי טִיט וָרֶפֶשׁ נָחְנוּ.


עבודה


לִפְנֵי יוֹם הֶעָשׂוֹר כֹּהֵן יַפְרִישׁוּ (יומא ב')

שָׁבוּעַ מִבֵּיתוֹ, וּלְתָא יִתֵּנוּ;

וּלְכַהֵן בִּמְקוֹמוֹ אַחֵר יַקְדִּישׁוּ,

מִפַּחַד בַּלֵּילוֹת מִקְרֶה יִקְרֶנּוּ.

כָּזֹאת מִימֵי עוֹלָם הֵן לֹא קָרָהוּ, (כ"א)

אַךְ לֹא לַנִּפְלָאוֹת נַפְשָׁם יִשָּׂאוּ.


שָׁמָּה אֶל־אֵל אֵלִים יִהְיֶה פֹלֵחַ, (י"ד)

אֶת־הַדָּם הוּא יִזְרֹק, וּמְנֹרָה יָעַל,

וּקְטֹרֶת הוּא יַקְטִיר, גַּם לַמִּזְבֵּחַ

(עִתִּים יַחְפֹּץ) רֶגֶל וָרֹאשׁ הוּא יָעַל;

וּשְׁלִישִׁי וּשְׁבִיעִי, כַּאֲשֶׁר בַּסֵּפֶר,(ח')

מִן־הַפָּרָה יַזּוּ עָלָיו הָאֵפֶר.


וּלְהוֹרוֹתוֹ כָל חֹק לוֹ יִמָּסֵרוּ (י"ח)

מִזִּקְנֵי הָעֵדָה, קוּמָה וּקְרָאָה,

אִישִׁי כֹּהֵן גָּדוֹל, אֵלָיו יֹאמֵרוּ.

פֶּן־יֵשׁ חֻקָּה דַקָּה מִמְךָ נִפְלָאָה;

פֶּן־יִמָּצֵא דָבָר לֹא הִתְבּוֹנָנְתָּ,

אוֹ שָׁכַחְתָּ אוֹתוֹ בַּאֲשֶׁר זָקָנְתָּ.


עֶרֶב יוֹם הֶעָשׂוֹר כֹּהֵן יִנְהָגוּ (י"ח)

שַׁעַר קֵדְמָה מִזְרָחָה, וּלְפָנָיו

אַיִל, כֶּבֶשׂ, בָּקָר שָׁמָּה יֻצָּגוּ,

עַד עַל־כֻּלָּם יִתֵּן אֶת־מַעְיָנָיו;

וּלְמַעַן כִּרְאוֹתוֹ יִתְבּוֹנֵן בָּמוֹ

אֵיכֹה לִפְנֵי מַלְכּוֹ יַקְרִיב אוֹתָמוֹ.


כָּל־שִׁבְעַת הַיָּמִים לֶאֱכֹל לִשְׂבֹּעַ (י"ח)

מֶנּוּ לֹא יִמָּנַע, חֶפְצוֹ יָפִיקוּ;

רַק עֶרֶב הֶעָשׂוֹר, הָאוֹר כִּנְסֹעַ,

מֵרוֹב אָכוֹל שָׁתֹה יָדָיו יַרְחִיקוּ;

כִּי דֶרֶךְ הָאָדָם כַּאֲשֶׁר שָׂבֵעַ

אֶל־יַם הַתַּרְדֵּמָה לִהְיוֹת טֹבֵעַ.


וּלְמַעַן עוֹד יִלְמַד לַחְפֹן חָפְנַיִם, (י"ח)

אֵלֶּה לִבְנֵי אַהֲרֹן אוֹתוֹ יִמְסֹרוּ;

הֵם אֶל־בֵּית אַבְטִינָס (כִּי מִיּוֹמַיִם

לִמְלֶאכֶת הַקִּטֵּר בָּנָיו יַעֲצֹרוּ)

אוֹתוֹ יוֹלִיכוּ: הֵם הֵיטֵב יוֹרוּהוּ,

אַחַר כִּדְבָרִים אֵלֶּה יַשְׁבִּיעוּהוּ:

כֹּהֵן! עַבְדי בֵית דִּין וּשְׁלִיחָיו אָנוּ,

וְאַתְּ שָׁלִיחַ לָנוּ אַף לָהֵמָּה;

הֵם הִשְׁבַּעְנוּ אוֹתְךָ בְאֵל עָשָׂנוּ,

וַאֲשֶׁר רַק יֵכָל זֹאת אֵלָיו נְעֵמָה –

אִם־תָּסוּר מִמִּלֵּינוּ אֶל־הַהֶבֶל,

אִם בִּקְהַל צָדוֹקִים תַּשְׁלִיךְ הַחֶבֶל.


יַעַן הֶחָמָס קָם אֶל־מַטֵּה־רֶשַׁע (י"ט)

בִּימֵי בַיִת שֵׁנִי, כָּזֹאת יִפְעָלוּ;

כִּי־אָז נִפְרַץ גָּדֵר, וַיִּגְדַּל פֶּשַׁע,

וּדְבַר הָאֵל נִרְמַס, וַיִּתְנַכָּלוּ

לָתוּר אַחַר לִבָּם אַנְשֵׁי אֵין־דָּעַת,

מִתַּלְמִידֵי צָדוֹק עֵדָה מִרְשָׁעַת.


וּבְאִוַּלְתָם רַבָּה הֵמָּה הֶאֱמִינוּ,

כִּי אֶת־קִטֵּר יוֹם זֶה, יוֹם הַסֹּלֵחַ,

בָּאֵשׁ נַקְטִיר מִחוּץ, כִּי כֹה הֵבִינוּ;

וּכְרוּם עָשָׁן לָרוּם, וּכְצֵאת הָרֵיחַ,

הַכֹּהֵן הַגָּדוֹל אוֹתוֹ יָשִׂימָה

אֶל־לִפְנֵי הָאָרוֹן לִפְנֵי לִפְנִימָה.


לֹא־כֵן אַנְשֵׁי הַשְׂכֵּל: אָמְנָם יַקְטִירוּ

לִפְנֵי אָרוֹן כַּיוֹם כִּי־כֵן שָׁמֵעוּ –

לָכֵן אֶת־הַכֹּהֵן עַתָּה יַזְכִּירוּ,

כִּי אִם־מַעֲשָׂיו לָהֶם לֹא יִוָּדֵעוּ,

לִפְנֵי שֹׁכֵן בִּסְנֶה נִגְלוּ כֻלָּמוֹ;

לָכֵן, לִבְלִי יַתְעֵם, יִשָּׁבַע לָמוֹ.


מִיָּד יִפְרוֹשׁ לִבְכּוֹת בִּלְתִּי מַרְגּוֹעַ, (י"ח)

הַקְשֵׁב כִּי יֵחָשֵׁב גֶּבֶר פּשֵׁעַ;

גַּם־הֵם עֵינָם בִּבְכִי תִדְמַע דָּמוֹעַ,

פֶּן־הִכְלִימוּ חִנָּם אֹהֵב וָרֵעַ;

כִּי הַחוֹשֵׁד אִישׁ תָּם לֹא נִקָּה נֶצַח,

גַּם־הַצָּרַעַת תִּזְרַח־לוֹ בַּמֶּצַח.


בַּלַּיְלָה מֵעֵינָיו שֵׁנָה יַרְחִיקוּ, (י"ט)

עַפְעפָיו יִשְׁמֹרוּן לִבְלִי יָנוּמוּ,

וּבְעֵסֶק זֶה אוֹ זֶה אוֹתוֹ יַעֲסִיקוּ;

אַךְ אִם רֶגַע יָנוּם, פִרְחָח יָקוּמוּ,

וּבְעֶצְבָּעָם יַכּוּ אָנֶה וָאָנָה,

לֵאמֹר: עוּרָה אָדוֹן, מַה־זֶּה תִּישָׁנָה?


אִם אֶת־בִּטְנוֹ מֵאָז בָּשָׂר וָיַיֵן (י"ח)

מֵחֻקֵּי אֵל מִלֵּא, בָּהֶם יָשִׂיחַ;

רָגִיל לִקְרוֹא יִקְרָא; אָכֵן אִם אַיִן,

אֶל־דִּבְרֵיהֶם הֵיטֵב הֵיטֵב יַשְׁגִּיחַ;

אַף־הֵם בַּאֲשֶׁר יוֹאֵל אֵלָיו יִקְרָאוּ,

אִיוֹב, אוֹ דָנִיאֵל, אוֹ בִּנְחֶמְיָהוּ.


אֶת־הַמִּזְבֵּחַ בֶּחָצוֹת יִתְרוֹמוּ, (כ)

אֶת־דֶּשֶׁן הָאִשִּׁים מֶנּוּ יָמִישׁוּ;

לֹא עַד יִקְרָא שֶׂכְוִי הַיּוֹם יִדֹּמּוּ

כִּשְׁאַר כָּל־הַיָּמִים; אוּלָם יָחִישׁוּ;

כִּי הַמְּלָאכוֹת רַבּוּ, לָכֵן יִפְחָדוּ,

לָכֵן מֵאָז בַּלַּיְלָה יִתְעַתָּדוּ.


בָּרִאשׁוֹנָה יִתְרוֹם אֶת־הַמִּזְבֵּחַ (כ"ב)

כָּל־הֶחָפֵץ, כָּל־הַקֹּדֵם רֵעֵהוּ;

אַךְ פֶּן יִסָּכֶן אִישׁ, אוֹ יִפָּסֵחַ,

כִּי יַכֶּנּוּ אָחִיו, אוֹ יַפִּילֵהוּ

(כִּי כֻלָּם לַעֲבוֹד אֵל גִּיל רַב יָגִילוּ),

אַף לִתְרוּמַת הַדֶּשֶׁן פּוּר יַפִּילוּ.


הַפּוּר אֵיכָה יַפִּילוּ? יִקָּהֵלוּ (כ"ה)

כִּדְמוּת גֹּרֶן עָגֹל; אָמְנָם יָרִימוּ

כֹבַע מֵרֹאשׁ אֶחָד, כִּי בוֹ יָחֵלוּ

לִמְנוֹת מִסְפָּר שַׂגִּיא, עַד־אִם יַשְׁלִימוּ.

וּלְבִלְתִּי שְׂאֵת רֹאשָׁם אֶצְבָּע יַגְבִּיהוּ (כ"ב)

אֶפֶס כִּי אֶת־הַכֹּהֵן לֹא יוֹצִיאוּ.


סָבִיב לִקְרֹב הַיּוֹם, עֵת יֶחְשַׁךְ נֹגַהּ, (כ"ח)

קוּמוּ אַחַי, יֹאמַר הַפָקִּיד לָמוֹ,

וּרְאוּ אִם יָאִיר שֶׁמֶשׁ אוֹר וָנֹגַּהּ,

וּשְׁאַר הַכֹּכָבִים חָשַׁךְ אוֹרָמוֹ;

אִם אֶת־כָּל־הַמִּזְרָח חַמָּה הֵאִירָה,

וּבְיִפְעָתָהּ עַד אֶל־חֶבְרוֹן הִזְהִירָה.


הֵן הֵאִיר הַמִּזְרָח,אִם־גֶּבֶר יַעַן, (כ"ח)

אֶל־שַׁעַר הַמִּרְוָה כֹּהֵן יוֹבִילוּ

לִרְחוֹץ אֶת־כָּל־גּוּפוֹ; יַעַן וּבְיַעַן

לֹא בִלְתִּי־זֹאת בּוֹא לֶחָצֵר יַעְפִּילוּ;

אַף־הַמֵּיסֵךְ יִרְחַץ, אַךְ מֵטִיל מַיִם

טָעוּן קִדּוּשׁ יָדַיִם אַף רַגְלָיִם.


לָכֵן גִּבּוֹר כָּאֲרִי אַף־קַל כַּנֶּשֶׁר (ל')

חָמֵשׁ יִטְבּוֹל בַּיּוֹם הַנֹּתֵן כֹּפֶר

אֶל־אִישׁ בּוֹ לֹא כֹפֵר, וּפְעָמִים עֶשֶׂר

יִרְחַץ יָדָיו רַגְלָיו מָהִיר כַּעֹפֶר;

וּטְבִילוֹת אֵלֶּה בַּקֹּדֶשׁ תִּהְיֶינָה,

וּלְבַד הָרִאשׁוֹנָה בַּחוֹל יַעֲשֶׂנָּה.


סָדִין בֵּינוֹ לָעָם אָז יִשָּׁטֵחַ, (ל')

פֶּן לטִבְוּלֹ אֶל־עֵין כֹּל יַחְפִּיר, יוֹבִישָׁה;

פָּשַׁט, יָרַד, טָבַל, וַיִּקָּנֵּחַ; (ל"א)

אֶת־בִּגְדֵי הַזָּהָב לִלְבּשׁ יָחִישָׁה:

עוֹד יִתְקַדַּשׁ; אַךְ אִם־בָּא בַיָּמִים

הֵטִילוּ בַמִּקְוֶה מַיִם חַמִּים.


לִשְׁחֹט אֶת־הַתָּמִיד אַחַר יֵצֵאָה, (ל"א)

וִימָרֵק עַל־יָדוֹ אַחֵר הַטֶּבַח;

כִּי לוּ רַק־הֶרֶף עַיִן יִתְמַהְמֵהַּ,

אֵיכָה יֶאֱסֹף אֵפוֹא אֶת־דַּם הַזֶּבַח?

כִּי מִי מִבַּלְעָדָיו יֶאֱסֹף הַדָּם,

אִם רֻחַק זוּלָתוֹ כָּל־הָאָדָם?


קִבֵּל הַדָּם כֻּלּוֹ, וַיִּזְרָקֵהוּ; (ל"א)

אָז הַנֵּרוֹת יֵיטִיב, וּקְטֹרֶת יָעַל;

אַחַר אֶת־רֹאשׁ הַזֶּבַח, וּנְתָחֵהוּ,

אַף־מִנְחָתוֹ יַקְרִיב אֶל־אֵל מִמָּעַל,

וִינַּסֵּךְ בַּשִׁיתִין נֵסֶךְ הַיָּיִן:

אֵלֵימוֹ קֵץ וָחֵקֶר כִּמְעַט אָיִן.


כִּי הָיָה סֵפֶל כֶּסֶף בַּמִּזְבֵּחַ

נָבוּב כֻּלּוֹ, כִּי בוֹ יוֹם יוֹם יִשְׁפֹּכוּ

אֶת־כָּל־נִסְכֵּי הַיַּיִן הַמְּשַׂמֵחַ,

וּלְעֹמֶק הַשִּׁיתִין מִשָּׁם יַהֲלֹכוּ;

אַף־אַחֵר הָיָה שָׁם, רַק־דַּק מִמֶּנּוּ;

בֶּחָג נִסְכֵּי הַמַּיִם בּוֹ יִתֵּנוּ.


יֻתַּן אֶל־הַכֹּהֵן כּוֹס מָלֵא מֶסֶךְ;

יִתְקָעוּ, גַּם יָרִיעוּ, אַף יִתְקָעוּ;

וּסְגַן אֶצְלוֹ יִהְיֶה, וּכְנָסְכּוֹ נֵסֶךְ

יִפְרֹשׂ הַמִּטְפָּחוֹת; וּבְכֵן יָדָעוּ

כִּי הַכֹּהֵן נוֹסֵךְ, וּבְכֵן יַכִּירוּ,

וּבְכֵן גַּם־הַלְּוִיִּם בַּשִּׁיר יָשִׁירוּ.


שְׁנֵי כִבְשֵׂי שָׁנָה אַחַר כָּל־אֵלֶּה,

אִם יוֹם שַׁבָּת הַיּוֹם, לַמּוּסָף יָעַשׂ;

וּכְפִקּוּדֵי הָאֵל הָעֹשֶׂה פֶלֶא

בֵּרַךְ אֶת־יִשְׂרָאֵל, וַיָּפֶר כָּעַשׂ;

כִּי אֵיךְ אֵפוֹא רֵיקָם מִלָּיו יָשׁוּבוּ,

אִם שִׂפְתוֹתָיו שֵׁם הַיּוֹצֵר יָנוּבוּ?


עֶשְׂרוֹנֵי סֹלֶת בַּשֶּׁמֶן נִבְלָלוּ

יַקְטִיר, אָז לֶחֶם הַקֹּדֶשׁ יַסִּיעוּ

מִשֻּׁלְחַן הַזָּהָב; רַק לֹא יוּכָלוּ

קַחַת אוֹתוֹ טֶרֶם חָדָשׁ יַצִּיעוּ;

לָכֵן עוֹד זֶה לֶחֶם יָשָׁן יַעְתִּיק,

חָדָשׁ יָשִׁית אַחֵר תַּחַת עַתִּיק.


אֶת־כַּפּוֹת הַלְבֹנָה הִקְטִיר וַיָּעַל;

אָז לַמּוּסָף יַסֵּךְ נֶסֶךְ הֶחָמֶד,

כַּדָּבָר הָאָמוּר מֵאָז מִמָּעַל.

הַיּוֹם יוֹם צוֹם, לָכֵן כָּל אִישׁ יִשָּׁמֶר,

וּלְחַלֵּק אֶת־הַפַּת לֹא יִקָּהֵלוּ

כִּשְׁאַר יָמִים, וּלְעֶרֶב יִיָּחֵלוּ.


כַּאֲשֶׁר עָשָׂה כָּל־אֵלֶּה עַד אִם־תָּמּוּ,

מוּסַף הַיּוֹם יַקְרִיב עַל־הַמִּזְבֵּחַ;

וּכְדִבְרֵי־אֵל, כִּי מִכָּל־הוֹן יִנְעָמוּ,

פָּר וּכְבָשִׂים שִׁבְעָה יִהְיֶה זֹבֵחַ,

מִנְחָתָם, נִסְכֵּיהֶם; אָמְנָם הָאָיִל

וּשְׂעִיר חַטָּאת יַקְרִיב עִם קָרְבַּן־לָיִל.


אָז יֵתֶא לַפַּרְוָה (וּשְׁמָהּ מַזְכֶּרֶת (ל"ה בתוספות)

אֶל־גֶּבֶר קוֹסֵם קֶסֶם, חֹבֵר חָבֶר:

הוּא בִּכְשָׁפָיו בַּתָּא חָתַר מַחְתֶּרֶת

לִרְאוֹת כֹּהֵן, אָמְנָם חָפַר לוֹ קָבֶר:

רַבִּים קָמוּ עָלָיו וַיַּהַרְגּוּהוּ):

אֶל־הַלִשְׁכָּה הַזֹּאת אָז יוֹבִילוּהוּ.


סָדִין בֵּינוֹ לָעָם אָז יִשָּׁטֵחַ. (ל"ד)

פֶּן לִטְבּוֹל אֶל־עֵין כֹּל יַחְפִּיר, יוֹבִישָׁה;

קִדֵּשׁ יָדָיו, רַגְלָיו, וַיִּקָנֵחַ;

טָבַל, אָז בִּגְדֵי־בַד לִלְבּשׁ יָחִישָׁה,

וּשְׁתֵּי יָדָיו קִדַּשׁ, וּשְׁתֵּי רַגְלָיִם,

לֹא בַכִּיוֹר, כִּי־אִם בִּכְלִי פַּרְוָיִם.


בַּשַּׁחַר יִלְבַּשׁ בּוּץ מִנִּי מִצְרָיִם, (ל"ד)

נִכְבָּד מִכָּל־מִין בָּד, כַּשֶּׁלֶג צָחַר;

אַךְ בַּד־הֹדוּ יִלְבַּשׁ בֵּין הָעַרְבַּיִם,

שָׁלִישׁ יִמְעַט עֶרְכּוֹ מִשֶׁלַּשָּׁחַר:

כִּשְׁנֵים עָשָׂר מָנֶה בִּגְדֵי הָעֶרֶב,

וּלְהַרְבּוֹת אִם־יַחְפֹץ מִבֵּיתוֹ יֶרֶב.


בָּא־לוֹ אֵצֶל פָּרוֹ, הוּא פַּר־חַטָּאת, (ל"ה)

וּשְׁתֵּי יָדָיו יִסְמוֹךְ עָלָיו. וּבְכֹחַ;

אַף־יִתְוַדֶּה עָלָיו מִכָּל־חַטָּאת,

וִיהִי שׁוֹאֵל מֵאֵל יוֹאֵל לִסְלֹחַ;

וּפְעָמִים שָׁלשׁ אֶת־הַשֵּׁם יַשְׁמִיעַ,

כִּכְתָבוֹ, כִּלְשֹׁנוֹ, וְכֹה יַבִּיעַ:


אָנָּא יהוה! חָטָאתִי, אַף־עָוִיתִי,

וּלְפָנֶיךָ הַרְבֵּה פָּשַׁעְתִּי אָנִי,

וּבְנֵי בֵיתִי; אָמְנָם הִנֵּה רָאִיתִי

כִּי רַק לֵיטִיב אֵלֵינוּ צִוִּיתָנִי

לַעֲבוֹד תַּחַת אַחַי בִּדְבִיר הַקֹּדֶשׁ,

לָבֹא בֵיתֶךְ בֶעָשׂוֹר לַחֹדֶשׁ.


אָנָּא יהוה! כַּפֵּר וּמְחַל וּסְלָחָה.

חַנּוּן, רַחוּם, עַל־כָּל־עָוֹן וָפֶשַע,

עַל־כָּל־חַטָּאת נַפְשֵׁנוּ בוֹ נִדָּחָה;

כָּל־מַעַל כַּפֶּר־נָא, וּשְׁכַח כָּל־רֶשַׁע;

עַל־רֹאשׁ גֶּבֶר צָעִיר הַגְדֵּל חַסְדֶּךָ,

כַּכָּתוּב עַל־תּוֹרַת משֶׁה עַבְדֶּךָ:


כִּי בַיּוֹם הַזֶּה יְכַפֵּר עֲלֵיכֶם לְטַהֵר אֶתְכֶם

מִכֹּל חַטּאֹתֵיכֶם לִפְנֵי יהוה תִּטְהָרוּ.


הָעָם הַקָּמִים שָׁם, כַּאֲשֶׁר יַקְשִׁיבוּ (ס"ו)

אֶת־הַשֵּׁם הַמְּפֹרָשׁ, יַשְׁפִּילוּ מֶצַח,

יִפֹּלוּ עַל פָּנֵימוֹ, אַף יָשִׁיבוּ:

בָּרוּךְ שֵׁם מַלְכוּתוֹ עוֹלָם וָנֶצַח.

אַף־הוּא יִגְמֹר הַשֵּׁם רֶגַע גָּמָרוּ,

וִיחַזֵּק תִּקְוָתָם, לֵאמֹר: תִּטְהָרוּ.


אֶל־מִזְרָח הֶחָצֵר אָז יָבֹא לוֹ, (ל"ז)

שָׁם לִצְפוֹן הַמִּזְבֵּחַ אָז יָבֹאָה.

לִימִינוֹ הַשָּׁלִישׁ, אַךְ לִשְׂמֹאלוֹ

רֹאשׁ בֵּית אָב הָעֹבֵד הַיּוֹם לֵאלֹהַּ;

וּשְׂעִירֵי הָעִזִּים גַּם יִמְצָא שָׁמָּה,

וְקַלְפִי גוֹרָלוֹת שָׁמָּה הוּשָׂמָה.


טָרַף בִּשְׁתֵּי יָדָיו, הוֹצִיא פוּרֶיהָ: (ל"ט)

אִם בִּימִינוֹ חַטָּאת עָלָה מִמַּעַל,

יֹאמַר הַשָּׁלִישׁ לוֹ: יָמִין הַגְבֵּהַּ;

אָמְנָם אִם־בִּשְׂמֹאלוֹ יָצָא וַיַּעַל,

יֹאמַר רֹאשׁ בֵּית הָאָב: הָרֵם יָדֶךָ,

אִישִׁי כֹהֵן גָּדוֹל: בָּא בִשְׂמֹאלֶךָ.


אָז עַל־רֹאשׁ כָּל־שָׂעִיר גּוֹרָל יָשֵׂם:

אֶחָד יִתֵּן עַל־רֹאשׁ הַמִּשְׁתַּלֵּחַ,

אֶחָד עַל־הַחַטָּאת, אוּלָם בַּשֵּׁם

אָז לַיהוה חַטָּאת יִהְיֶה צוֵֹחַ;

אַף כָּל־מַקְשִׁיב יָשִׁיב, אַךְ יַשְׁפִּיל מֶצַח:

בָּרוּךְ שֵׁם מַלְכוּתוֹ עוֹלָם וָנֶצַח.


וּלְשׁוֹן תֹּלָע שָׁנִי עַל־הַשָּׁלוּחַ (מ"א)

עַל קָדְקֳדוֹ יִקְשׁוֹר, אַף יַצִּיגֵהוּ

נֹכַח בֵּית שִׁלּוּחוֹ; אַךְ לַטָּבוּחַ

יִקְשׁוֹר אֶת הַתֹּלָע עַל צַוָּארֵהוּ:

יַעַן מִבִּלְתִּי־זֹאת אֵיכָה יֻכָּרוּ?

וּבְמַרְאֶה וּבְקוֹמָה שָׁוִים נִבְחָרוּ. (ס"ב)


שָׁב לוֹ אֵצֶל פָּרוֹ, הוּא פַר־חַטָּאת,

וּשְׁתֵּי יָדָיו יִסְמֹךְ עָלָיו, וּבְכֹחַ;

אַף־יִתְוַדֶּה עָלָיו מִכָּל־חַטָּאת,

וִיהִי שׁוֹאֵל מֵאֵל יוֹאֵל לִסְלֹחַ

וּפְעָמִים שָׁלשׁ אֶת־הַשֵּׁם יַשְׁמִיעַ

כִּכְתָבוֹ, כִּלְשׁוֹנוֹ, וְכֹה יַבִּיעַ:


אָנָּא יהוה! חָטָאתִי, אַף עָוִיתִי,

וּלְפָנֶיךָ הַרְבֵּה פָּשַׁעְתִּי אָנִי,

וּבְנֵי אַהֲרֹן; אָמְנָם הִנֵּה רָאִיתִי

בִּי רַק לֵיטִיב אֵלֵינוּ צִוִּיתָנִי

לַעֲבֹד תַּחַת אַחַי בִּדְבִיר הַקֹּדֶשׁ,

לָבֹא בֵיתֶךָ בֶּעָשׂוֹר לַחֹדֶשׁ.


אָנָּא יהוה! כַּפֵּר וּמְחַל וּסְלָחָה,

חַנּוּן, רַחוּם, עַל־כָּל־עָוֹן וָפֶשַׁע,

עַל־כָּל־חַטָּאת נַפְשֵׁנוּ בוֹ נִדָּחָה;

כָּל־מַעַל כַּפֶּר־נָא, וּשְׁכַח כָּל־רֶשַׁע.

עַל־רֹאשׁ שֵׁבֶט צָעִיר הַגְדֵּל חַסְדֶּךָ,

כַּכָּתוּב עַל־תּוֹרַת משֶׁה עַבְדֶּךָ:


כִּי בַיּוֹם הַזֶּה יְכַפֵּר עֲלֵיכֶם לְטַהֵר אֶתְכֶם

מִכֹּל חַטֹּאתֵיכֶם לִפְנֵי יהוה תִּטְהָרוּ.


הָעָם הַקָּמִים שָׁם, כַּאֲשֶׁר יַקְשִׁיבוּ (ס"ו)

אֶת־הַשֵּׁם הַמְּפֹרָשׁ, יַשְׁפִּילוּ מֶצַח,

יִפֹּלוּ עַל־פָּנֵימוֹ, אַף יָשִׁיבוּ:

בָּרוּךְ שֵׁם מַלְכוּתוֹ עוֹלָם וָנֶצַח;

אַף־הוּא יִגְמוֹר הַשֵּׁם רֶגַע גָּמָרוּ,

וִיחַזֵּק תִּקְוָתָם, לֵאמֹר: תִּטְהָרוּ.


כַּמִּשְׁפָּט הַיָּדוּעַ יִזְבָּחֶנּוּ, (מ"ג)

וּלְקַבֵּץ אֶת־דָּמוֹ לֹא יִתְמַהְמֵהַּ,

וּלְמַעַן לֹא יִקְפָּא, אָז יַפְקִידֶנּוּ

בִּידֵי כֹּהֵן אַחֵר, עַד־אִם יֵצֵאָה;

וּלְמָרֵס בּוֹ תָמִיד לֹא יִבָּצֵר

בָּרֹבֶד הָרְבִיעִי שֶׁבֶּחָצֵר.


יִקַּח מַחְתָּה, יָבא אֶל־הַמִּזְבֵּחַ,

וִיפַנֶּה גֶחָלָיו הֵנָּה וָהֵנָּה,

וַאֲשֶׁר אֻכָּלוּ בַּמַּחְתָּה לֹקֵחַ,

בָּרֹבֶד הָרְבִיעִי אָז יַנִּיחֶנָּה;

וַאֲחֵרִים אֶת־הַכַּף אֵלָיו יוֹצִיאוּ, (מ"ו)

וּכְלִי מָלֵא קִטֵּר אֵלָיו יָבִיאוּ.


בֶּחָפְנַיִם יַחְפּוֹן כֹּהֵן מִנֵּהוּ, (מ"ז)

וּלְתוֹךְ הַכַּף יִתֵּן, וּבְלֹא כָל־כֶּלִי:

כָּכָה צִוָּה אֵל חַי, וַאֲנִי אַנְוֵהוּ;

קוֹלוֹ לָעַד אֶשְׁמַע, אַף אִם־הוּא פֶלִאי.

אָז אֶת־מַחְתַּת הָאֵשׁ יִתֵּן בַּכַּף

הַיְּמָנִית: בִּשְׂמֹאלוֹ יִשָּׂא הַכָּף.


בְהֵיכָל יִתְהַלֵּךְ, עַד־כִּי יַגִּיעַ

לִשְׁתֵּי הַפָּרֹכוֹת הַמַּבְדִּילוֹת

בֵּין יֶתֶר הַהֵיכָל אֶל־הַיָּצִיעַ

הַמְּקֻדֶּשֶׁת מִכָּל הַהֵיכָלוֹת;

וּפְרוּפָה הָאַחַת נֶגְבָּה תֵימָנָה, (נ"ב)

וּפִניִמָּיה אָמְנָם לִשְׂמֹאל הוּכָנָה.


הָלַךְ מִבֵּינֵיהֶן, עַד־אִם יַגִּיעַ

אֶל־הָאָרֹן; אָז יָשֶׁת בֵּין בַּדָּיו

מַחְתַּת הָאֵשׁ, וּבְרֹב בִּינָה וִיגִיעַ

עֵרָה אֶת־הַקִּטֵּר אֶל־תּוֹךְ יָדָיו.

מֵאָז נִטַּל אָרֹן אֶבֶן הָיָתָה (נ"ג)

לָתֵת עָלֶיהָ, וּשְׁתִיָּה נִקְרָאתָה.


וּקְטֹרֶת אָז יִתֵּן עַל גֶּחָלֶיהָ, (נ"ב)

יוֹחִיל עַד יִמָּלֵא עָשָׁן וָרֵיחַ

כָּל־הַמָּקוֹם הַהוּא, וְאָז יֵצֵאָה;

וּקְדֹרַנִּית אָמְנָם יִהְיֶה פֹסֵחַ.

עוֹדֶנּוּ בָהֵיכָל בָּאֵל יַפְגִּיעַ,

רַק לֹא יַכְבִּיר מִלִּין, וְכֹה יַבִּיעַ:


אָנָּא אֵלִי! / הֵעָתֶר לִי: (נ"ג)

וּשְׁעֵה אֵל רָב / אֶל־אִישׁ קָטוֹן.

אוֹצַר מָטָר / אַל־נָא תֶּאְטָר,

אַף אִם־יֶשׁ־אִישׁ / מָטָר יִשְׂטוֹן.

אָנָּא עֶלְיוֹן! / יֶחְדַּל אֶבְיוֹן,

רַק לֹא יֶחְדַּל / אִישׁ הַשִּׁלְטוֹן.


מִיַּד מֵנִיס הַדָּם אָז יִקָּחֶנּוּ,

וּכְפַעַם רִאשׁוֹנָה יָבֹא הַבָּיְתָה;

וּשְׁמֹנֶה הַזָּיוֹת יַזֶּה מִמֶּנּוּ

בֵּין בַּדֵּי הָאָרֹן שָׁמָּה מִבָּיְתָה;

רַק לֹא־יָרִים, אַף לֹא־יַשְׁפִּיל אוֹתָנָה,

אַךְ מִן־הָרִאשׁוֹנָה יַבְדִּיל כֻּלָנָה.


אַף בִּמְנוֹתוֹ אוֹתָן יַבְדִּיל בֵּינְתָּיִם,

יַבְדִּיל הָרִאשׁוֹנָה מֵרֵעוֹתֶיהָ,

לֹא כַמִּשְׁפָּט יִמְנֶה: אַחַת, וּשְׁתָּיִם;

לִפְנֵי כָל־הַזָּיָה יָשׁוּב יִמְנֶהָ;

וּשְׁאַר הַדָּם, כַּאֲשֶׁר הִשְׁלִים וַיְּכַל,

יִתֵּן עַל־כַּן זָהָב שֶׁבָּהֵיכָל.


וּשְׂעִיר חַטָּאת יוּבָא: הוּא יִזְבָּחֶנּוּ,

וּלְקַבֵּץ אֶת־דָּמוֹ יָחִישׁ וִימַהֵר;

וּשְׁמֹנֶה הַזָּיוֹת יַזֶּה מִמֶּנּוּ

בֵּין בַּדֵּי הָאָרֹן; אָכֵן יִזָּהֵר

עֵת כִּי יַזֶּה, כַּאֲשֶׁר מֵאָז אָמָרְנוּ,

אַף בִּמְנוֹתוֹ אֹתָן, כַּאֲשֶׁר דִּבָּרְנוּ.


וּדְמֵי הַפָּר לָקַח, וַיֵּז מֵהֵמָּה

בַפָּרֹכֶת מִחוּץ, כַּאֲשֶׁר בַּפַּעַם

הָרִאשׁוֹנָה בִּפְנִים; וּמְאֹד נָעֵמָה

שֵׁרוּתוֹ לֵאלֹהִים, אַף אִם־הַטַּעַם

מִמֶּנּוּ יִפָּלֵא הַפְלֵא וָפֶלֶא;

אַף לִדְמֵי הַשָּׂעִיר יַעַשׂ כָּאֵלֶּה.


עֵרָה אֶת הַדָּמִים יַחְדָּו פַּעֲמַיִם,

וּבְמִזְבַּח הַזָּהָב יַזֶּה אוֹתָמוֹ:

לִמְקֻצְעוֹת הַהַרְאֵל בַּיַּרְכָתַיִם (נ"ח)

אַרְבַּעַת הַזָּיוֹת אָז יַזֶּה בָּמוֹ;

אַךְ שֶׁבַע הַזָּיוֹת אַחַר כָּל־זֶה

עַל־טֹהַר הַמִּזְבֵּחַ גַּם יַזֶּה.


בָּא לוֹ אֶל־הַשָּׂעִיר הַמִּשְׁתַּלֵּחַ, (ס"ו)

וּשְׁתֵּי יָדָיו יִסְמוֹךְ עָלָיו, וּבְכֹחַ;

אַף יִתְוַדֶּה עָלָיו אֶל־הַסֹּלֵחַ,

וִיהִי שׁוֹאֵל מֵאֵל יוֹאֵל לִסְלֹחַ;

וּפְעָמִים שָׁלשׁ אֶת־הַשֵּׁם יַשְׁמִיעַ,

כִּכְתָבוֹ, כִּלְשֹׁנוֹ, וְכֹה יַבִּיעַ:


אָנָּא יהוה! עַמֶּךְ חֵטְא חָטָאוּ,

עָווּ עָוֹה, וַיִּפְשָׁעוּן פָּשׁוֹעַ;

הַיּוֹם בִּבְכִי וָצוֹם עָדֶיךָ בָאוּ,

כִּי כֵן אָמַרְתָּ יִתְכַּפֵּר כָּל־רֹעַ;

וּכְפִקּוּדֶיךָ, בֶּעָשׂוֹר לַחֹדֶשׁ

גַּם־אָנֹכִי אֶעֱבֹד בִּדְבִיר הַקֹּדֶשׁ.


אָנָּא יהוה! כַּפֵּר וּמְחַל וּסְלָחָה,

חַנּוּן, רַחוּם, עַל־כָּל־עָוֹן וָפֶשַׁע,

עַל־כָּל־חַטָּאת נַפְשֵׁנוּ בוֹ נִדָּחָה;

כָּל־מַעַל כַּפֶּר־נָא, וּשְׁכַח כָּל־רֶשַׁע.

עַל־רֹאשׁ שֵׁבֶט צָעִיר הַגְדֵּל חַסְדֶּךָ,

כַּכָּתוּב עַל־תּוֹרַת משֶׁה עַבְדֶּךָ:


כִּי בַיּוֹם הַזֶּה יְכַפֵּר עֲלֵיכֶם לְטַהֵר אֶתְכֶם

מִכֹּל חַטֹּאתֵיכֶם לִפְנֵי יהוה תִּטְהָרוּ.


הָעָם הַקָּמִים שָׁם, כַּאֲשֶׁר יַקְשִׁיבוּ (ס"ו)

אֶת־הַשֵּׁם הַמְּפֹרָשׁ, יַשְׁפִּילוּ מֶצַח,

יִפֹּלוּ עַל־פָּנֵימוֹ, אַף יָשִׁיבוּ:

בָּרוּךְ שֵׁם מַלְכוּתוֹ עוֹלָם וָנֶצַח;

אַף־הוּא יִגְמוֹר הַשֵּׁם רֶגַע גָּמָרוּ,

וִיחַזֵּק תִּקְוָתָם, לֵאמֹר: תִּטְהָרוּ.


הַשָּׂעִיר עִם־עִתִּי אָז יִשְׁלָחֶנּוּ, (ס"ו)

וּלְהוֹלִיכוֹ אַף־הַזָּרִים יַשׂפִּיקוּ,

אַךְ זוּלָתִי כֹהֵן לֹא יוֹלִיכֶנּוּ,

וּבְנֵי אַהֲרֹן אֶת־הַזָּרִים הִרְחִיקוּ.

הָאִישׁ מַה־יַּעַשׂ בּוֹ כַּאֲשֶׁר יַגִּיעַ

אֶל רֹאשׁ הָהָר אַחַר עָמָל וִיגִיעַ?


תֹּלַעַת הַשָּׁנִי יַחְלֹק לִשְׁנָיִם: (ס"ז)

אֶת־חֶצְיוֹ הָאֶחָד יִקְשֹׁר בַּסָּלַע,

וּשְׁאֵרִיתוֹ יִקְשֹׁר בֵּין הַקַּרְנָיִם;

וּכְגֶבֶר כִּי יִקְלַע אֶבֶן בַּקָּלַע

הַשָּׂעִיר מֵאָחוֹר מַטָּה יַדִּיחַ,

וּבְכִמְעַט־רֶגַע אֶת־נַפְשׁוֹ יַפְרִיחַ.


אֶל־פָּר וּשְׂעִיר חַטָּאת הַנִּשְׂרָפים (ס"ח)

בָּא־לוֹ כֹּהֵן גָּדוֹל, וַיְּחַץ בִּטְנָמוֹ:

לֹא יִרְאֶה בַכָּבֵד כָּאַשָּׁפִים,

רַק חֶלְבֵיהֶם יוצִיא מִמּוֹשָׁבָמוֹ;

וּשְׁנֵיהֶם, גַּם־הַפָּר, גַּם־הַשָּׂעִיר,

יִשְׁלַח בֵּית הַמּוֹקֵד מִחוּץ לָעִיר.


הוֹדַע אֵלָיו כִּי לַמִּדְבָּר הִגִּיעַ, (ס"ח)

עִם־הָאִישׁ הָעִתִּי, הַמִּשְׁתַּלֵּחַ,

יָבֹא כֹהֵן גָּדוֹל אֶל־הַיָּצִיעַ,

לִקְרֹא דִבְרֵי צוּרוֹ עָלֵז, שָׂמֵחַ;

בַּבֶּגֶד שֶׁעָלָיו, אִם־כֹּה רָצָה,

אוֹ מִבֵּיתוֹ בֶּגֶד לָבָן יִמְצָא.


אָז אַחֲרֵי־מוֹת יָקוּם לִקְרֹא בַסֵּפֶר, (ס"ח)

אַחַר בֶּאֱמוֹר יִקְרָא אַךְ בֶּעָשׂוֹר;

אוֹתוֹ אַחַר יִגְלוֹל, מִכְלוֹל כָּל־שֶׁפֶר,

אַף כַּמִּשְׁפָּט אוֹתוֹ הֵיטֵב יִקְשׁוֹר;

וּלְכָל־הָעָם יֹאמַר: אַחַי! עַד־הֵנָּה.

יוֹתֵר מִשֶּׁקָּרָאתִי כָתוּב הֵנָּה.


אַךְ בֶּעָשׂוֹר שֵׁנִית יִקְרָא מפִּיהוּ, (ס"ח)

פֶּן כִּי־יַחְפֹּשׂ אֹתָהּ עֵדָה יַטְרִיחַ;

וִיבָרֵךְ אֵל שַׁדַּי עַל־תּוֹרַת פִּיהוּ,

אַף עַל־שֵׁרוּת אֶחָיו, אַף יִשְׁפָּךְ־שִׂיחַ

עַל־כָּל־חֵטְא, עַל־בֵּית אֵל, עַל־יִשְׂרָאֵל,

עַל־הָעִיר, עַל־אֶחָיו, כַּאֲשֶׁר יוֹאֵל.


סָדִין בֵּינוֹ לָעָם אָז יִשָּׁטֵחַ. (ע')

פֶּן לִטְבּוֹל אֶל־עֵין כֹּל יַחְפִּיר, יוֹבִישָׁה;

קִדֵּשׁ יָדָיו, רַגְלָיו, וַיִּקָנֵחַ;

טָבַל; בִּגְדֵי זָהָב לִלְבּשׁ יָחִישָׁה,

וִיקַדֵּשׁ אֶת־רַגְלָיו, אַף אֶת־יָדָיו,

כִּי כֵן יַעַשׂ כָּל־עֵת יָמִיר מַדָּיו.


אֵיל הָעוֹלָה שֶׁלּוֹ, אַף אֶת הָאַיִל (ע')

שֶׁלַּמּוּסָף, אַף אֶת־חֶלְבֵי חַטָּאת,

יַקְרִיב אֶל־אֵל הַבּוֹרֵא יוֹם וָלַיִל,

סוֹלֵחַ אֶל־עַמּוֹ אֶת־כָּל־חַטָּאת,

נֹתֵן אֶת־צָרֵיהֶם כֶּעָפָר חָרֶב;

אָז כַּמִּשְׁפָּט יַקְרִיב קָרְבַּן הָעָרֶב.


סָדִין בֵּינוֹ לָעָם אָז יִשָּׁטֵחַ. (ע')

פֶּן לִטְבּוֹל אֶל־עֵין כֹּל יַחְפִּיר, יוֹבִישָׁה;

קִדֵּשׁ יָדָיו, רַגְלָיו, וַיִּקָנֵחַ;

טָבַל; אָז בִּגְדֵי־בָד לִלְבּשׁ יָחִישָׁה,

וּשְׁתֵּי יָדָיו קִדֵּשׁ, וּשְׁתֵּי רַגְלָיִם,

לֹא בַכִּיּוֹר, כִּי־אִם בִּכְלִי פַרְוָיִם.


מִבֵּית קֹדֶשׁ כָּל קֹדֶשׁ אָז לוֹקֵחַ (ע')

הַכַּף עִם־הַמַּחְתָּה, כִּי שָׁם נִשְׁאָרוּ;

קִדֵּשׁ יָדָיו, רַגְלָיו, וַיִקָּנֵחַ;

טָבַל; יִלְבַּשׁ בִּגְדֵי זָהָב יָקָרוּ;

וִיקַדֵּשׁ אֶת־רַגְלָיו, אַף אֶת־יָדָיו,

כִּי כֵן יַעַשׂ כָּל־עֵת יָמִיר מַדָּיו.


מִיָּד יַקְטֵר קִטֵּר בֵּין הָעַרְבָּיִם,

גַּם שֶׁבַע הַנֵּרוֹת כַּמִשְׁפָּט יָעַל;

וִיקַדֶּשׁ־עוֹד יָדָיו עִם־הָרַגְלָיִם,

וּבְכֹה הִשְׁלִים חֻקּוֹ, וַיְּכֻפַּר־מָעַל;

פָּשַׁט מַדָּיו, וַיַּעַט אֶת־מַעֲטֵהוּ,

וִילַוּוּהוּ הָעָם עַד אֶל־נָוֵהוּ.


יוֹם טוֹב אֶל־כָּל־אִישׁ טוֹב מֵעַתָּה יָעַשׂ,

יָשִׂישׂ אֶת־אִישׁ יָשִׁישׁ זָקֵן וָיָלֶד;

גִּיל לֹא פֹרַשׁ יִירָשׁ, גֹּרַשׁ כָּל־כָּעַש,

אַחַר כִּי הוּא נִבְחַר מֵאַנְשֵׁי־חָלֶד:

יָצָא גַם לֹא מָצָא מַצָּה וָפֶגַע,

כִּי אֵין קִרְבּוֹ אָרְבּוֹ, לִבּוֹ בַּל־נֶגַע.


אֵל רָב! הִנֵּה מָה רָב אָז לָנוּ אשֶׁר!

חַסְדֶּךָ לַכֹּהֵן מָאֵן לֹא יֶהִי;

אֶבְיוֹן וָדַל יֶחְדַּל, יִגְדַּל הָעשֶׁר;

לֹא הוֹמִיָּה קִרְיָה בִּכְיָה וָנֶהִי:

קִטֵּר יַקְטֵר, שׁוֹטֵר יִהְיֶה בַדָּבֶר,

יִמְצָא פִדְיוֹם יוֹם יוֹם לִפְשָׁעָיו גָּבֶר.


הֵן בְּךָ נִפְגַּע אִם־רָע יִגַּע עַד־נָפֶשׁ;

לֹא מִצָּרָה נִירָא, נִצְחַק אֶל־פָּרֶץ:

שִׁמְךָ מִבְצָר עֶת־צָר, וָצָר יִתָּפָשׁ;

שִׁמְךָ אַבִּיר, הוּא הַמַּשְׁבִּיר הָאָרֶץ,

אַתָּה הָאֵל, ובְיִשְׂרָאֵל תִּשְׁכֹּנָה,

וּלְקוֹל נַעֲנָה תִּפְנֶה מִן הַהַרְמוֹנָה.


הֵן כִישִׁימוֹן אַרְמוֹן, אַלְמוֹן גּוֹי שׁוֹעַ;

אָבַד עֶדְיוֹ, יָפְיוֹ, וַיֹּאבַד כֹּפֶר.

תֹּעֶה טַהֵר מַהֵר, מַהֵר מַרְגּוֹעַ;

רוּץ, רִיב רִיבִי, אָבִי! כִּצְבִי וָעֹפֶר,

שׁוּר עָם נִדְכָּה הֻכָּה, סֻכָּה נֹפָלֶת;


תִּקְוָה, חֶדְוָה, נָאוָה אֶל־דַּל כָּמוֹנִי;

וּבְכָל־חָזֶה תֶּחֱזֶה רַק־זֶה, רַק־סֵבֶל.

נִבְעֶה, נִגְעֶה, נִפְעֶה, נִכְנָעֵי עֹנִי;

קַבֵּל אָבל נֹבֵל, סֹבֵל כָּל־חֵבֶל.

אֶל־אִישׁ יִזְחַל וִיחַל תִּמְחַל כָּל־פֶּשַׁע;

סַלָּח! תִּסְלַח נֶאֱלח, וּשְׁלַח הַיֶּשַׁע.
