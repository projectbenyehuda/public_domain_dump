[את מלחמות אדוני בהינקמו לי אחזה]

מגן

אֶת מִלְחֲמוֹת אֲדֹנָי בְּהִנָּקְמוֹ לִי אֶחֱזֶה

בְּקָרְאוֹ הֵידָד עַל צוֹרֲרָי וְנִצְחָם עַל בְּגָדַי אַזֶּה

גַּת דֶּרֶךְ לְבֶן עֲמָלֵק בֶּן הַמְּרַצֵּחַ הַזֶּה

יוֹם בְּשׂוֹרָה הַיּוֹם הַזֶּה

דֹב שַׁכּוּל נָהַם וְעֵצוֹת קָרָא מִמֶּרְחָק

הִפִּיל פּוּר וְהֻפָר חָקַק אָוֶן וְנִמְחָק

וּבְהִמָּלֵךְ בְּבֵית הַמֶּלֶךְ הָבָה נַשְׁחִית אֶת בֵּית יִצְחָק

יוֹשֵׁב בַּשָּמַיִם יִשְׂחָק

זְאֵבֵי עֶרֶב הִתְעָרְבוּ לְאַבֵּד שֵׂה הַשִּׁבְיָה

חָשְׁבוּ יָמִים לְאַבְּדָם וַתֵּהֹם עָלַי הַקִּרְיָה

טָמְנוּ פַח לְרַגְלַי וַיֹּאמְרוּ לֹא יִרְאֶה יָהּ

וְיְיָ שָׁם הָיָה

יִסְעֲדוּ לַהֲפִיצֵנִי בְּשֵׁטֶף אַף אַכְזָרִים

כַּאֲשֶׁר כּוֹנֲנוּ לְהַשְׁחִית וּלְהַשְׁפִּיל מְרוֹם הָרִים

לְחָצוּנִי וַאֲנִי תְּפִלָּה וְהָנְחַלְתִּי מִשְׁפָּטִים יְשָרִים

כִּי אֱלֹהִים שֹׁפֵט זֶה יַשְׁפִּיל וְזֶה יָרִים.

פזמון

יָעַץ הַצַּר תַּחְבֻּלּוֹת אַךְ טִיחַ עֲצָתוֹ תָּפֵל

וַיִּקְרָא צוּר אֶל עֲבָדָיו חַכּוּ לִי יֹשְׁבֵי שָׁפֶל

וִידַעְתֶּם כִּי אֲנִי אֲדֹנָי בּוֹקֵע אוֹר מֵאֹפֶל

הִשְׁפָּלְתִּי עֵץ גָּבֹהַּ הִגְבַּהְתִּי עֵץ שָׁפֵל

הֹפֵךְ לְבֹקֶר צַלְמָוֶת בְּרַחֲמִים הִשְׁקִיף מִזְּבֻלוֹ

וַיַּרְא צֹאנוֹ בְּיַד צֹרֵר וַיֹּאמֶר לְחַבֵּל עֻלּוֹ

צֵל רַחֲמִים עָלָיו פָּרַשׂ וּמֵעַל הַצָּר סָר צִלּוֹ

גַּם קְבוּרָה לֹא הָיְתָה לוֹ טוֹב מִמֶּנּוּ הַנָּפֶל

וְאַחַר נוֹאַשְׁתִּי מִטּוֹב סְבָבַנִי בְּרָנֵי פַלֵט

כִּי הָשְׁלַכְתִּי עַל חֲסָדָיו וָאֶצְלַח וָאִמָּלֵט

טִבַּע רַגְלַי צָר בַּבֹּץ וּפָנַי טָמַן בַּמֶּלֶט

וְנֶהְפַּךְ הָעֶבֶד לְשֹׁלֵט וְאוּלָם יִבֹּל הַר נוֹפֵל

דִּבֵּר עַל לִבִּי בְּשׂוֹרָה אַחַר הֹוָה נִפְלְאַתָה

וַיֵּבוֹשׁ פֶּסֶל וְעֹבְדָיו כִּי הָאֲשֵׁרָה כֹרָתָה

וְחֶרֶב יְשׁוּעָה הוּחַדָּה וְיָצְאָה כִּי כֵן צֻוָּתָה

תֵּעוֹר חֶרֶב שְׁלִישְׁתָהּ חֶרֶב חֲלָלִים וְתִכָּפֵל

הִדָּרֵשׁ כֵּן לְעֶדְרֶךָ וּרְעֵה אֶת צֹאן הַהֲרֵגָה

וְעִמָּם לִשְׁמֹר שְׁאֵרִיתָם צִבְאוֹת רַחֲמִים הַצִּיגָה

וְיוֹנָה מֵחַגְוֵי סֶלַע וּמִסֵּתֶר הַמַּדְרֵגָה

הוֹצֵא וְשִׂימָהּ מְעֻנָּגָה בְאַרְמוֹן וּבַחַן וָעֹפֶל.

כרוג

יָרִים הוֹד הֲמוֹנָיו הַסֹּדְרִים לְמָגֵן

וְיוֹפִיעַ יֶשַׁע בְּעַד חוֹסָיו לְהָגֵן

שְׂמֵחִים בִּנְפֹל צָר כְּנַגֵּן הַמְּנַגֵּן

קוּמוּ הַשָּׂרִים מִשְׁכוּ מָגֵן.

מחיה

מִכְּנַף הָאָרֶץ זְמִרוֹת שָׁמְעוּ אָזְנֵינוּ

נְאֻם מְשׁוֹרֵר לָאֵל הַרְנִינוּ לֵאלֹהִים עֻזֵּנוּ

שָם אֵשׁ וְגָפְרִית מְנַת כּוֹס מַרְגִּיזֵנוּ

זֶה חֵלֶק שׁוֹסֵינוּ וְגוֹרָל לְבֹזְזֵינוּ

עֵת נִכְּלוּ לִי שֹׂטְנַי עָלַי פָּקַחְתָּ עֵינֶיךָ

פָּנִיתָ אֶל תְּפִלַּת הָעַרְעָר בְּהִטְעַטְּפוֹ לְפָנֶיךָ

צֶדֶק בִּשּׂרְתּוֹ וַתְּאַזְּרֵהוּ שִׂמְחָה בְּמַעֲנֶיךָ

לֵךְ וּפִתַּחְתָּ הַשַּׂק מֵעַל מָתְנֶיךָ

קַמְתִּי מִתַּעֲנִיתִי וָאֶקַּח גִּבּוֹר מַלְקוֹחַ

רֹאשׁ מָחַצְתָּ מִבֵּית רָשָׁע וָאוֹרֶד בָּנָיו לִטְבוֹחַ

שֹאֲפַי נֶאֱלַם לְשׁוֹנָם וְלֹא עָצְרוּ כֹחַ

כֻּלָּם כְּלָבִים אִלְּמִים לא יוּכְלּו לִנְבֹּחַ

תְּעָלָה הִקְדַּמְתָּ לְמַכָּתִי וְלִפְנֵי דְּרֹךְ קֶשֶׁת צִנָּה

וְאֶבֶן מָאֲסוּ הַבּוֹנִים הָיְתָה לְרֹאשׁ פִּנָּה

דִּבַּרְתָּ עַל לֵב מִתְאַבֶּלֶת וַתַּעֲנֶה שֹׁאֶלֶת מִתְעַנָּה

בָּעֶרֶב יָלִין בֶּכִי וְלַבֹּקֶר רִנָּה.

כרוג

לְפָנָיו יָלִין בֶּכִי בְּלֵב חָרֵד לַהֲמוֹנָיו

שְׁלֹשֶׁת יָמִים צָם וְהִגְדִּיל תַּחְנוּנָיו

בְּיוֹם הַשְּלִישִׁי יְקַו רַחֲמֵי אֲדֹנָיו

יְקֻמֵנוּ וְנִחְיֶה לְפָנָיו.

פזמון

יֹאמַר נָא יִשְׂרָאֵל יוֹם צוּר מִיָּד צָר פָּדָם

יְיָ לִי לֹא אִירָא מַה יַעֲשֶׂה לִי אָדָם

הֵידָד בִּקְרֹא עָלַי צָר

יָדַעְתִּי כִּי יָדְךָ לֹא תִקְצָר

וָאֶקְרָא יָהּ מִן הַמֵצָר

הָבָה לָנוּ עֶזְרָת מִצָּר וְשָׁוְא תְּשׁוּעַת אָדָם

וְכַיּוֹם גְּמוּלָיו נַזְכִּיר כֻּלָּנוּ

כְּעַל כֹּל אֲשֶׁר גְּמָלָנוּ

יוֹם צֹרֵר כִּמְעַט כִּלָּנוּ

לוּלֵא יְיָ שֶׁהָיָה לָנוּ בְּקוּם עָלֵינוּ אָדָם

דָּגוּל מֵרְבָבָה הִרְעִים

וַיּוֹצֵא מִנְּקִיקֵי הַסְּלָעִים

הַנֶּחְבָּאִים מִתִּגְרַת רָעִים

כִּי בְקוּם רְשָׁעִים יִסָּתֵר אָדָם

הֵעִיר רוּם אֶבְיוֹנָי

וַיִּתְגָּאוּ עַל גְאֵיוֹנָי

וְשִׁירָה חֲדָשָׁה בַּהֲמוֹנָי

טוֹב לַחֲסוֹת בַּאדֹנָי מִבְּטֹחַ בָּאָדָם.

פזמון

יוֹשֵׁב תְּהִלּוֹת מֵאֵין תְּחִילוֹת

עוּרָה וְהִגָּלֵה וּבְכֵן לְךָ תַעְלֶה

הָרֵם יְמִינְךָ כִּי אֵיךְ לְעֵינְךָ

בִּנְךָ יְהִי גֹּלֶה וּבְכֵן לְךָ תַעְלֶה

וְנֶאֱסְפוּ יוֹדְעַי וְסָפוּ

מִנִּי וְאֵין גֹּלֶה וּבְכֵן לְךָ תַעְלֶה

דַּבֵּר בְּאָזְנַי מָתַי יְמִינִי

עַל אוֹיְבַי תַּעְלֶה וּבְכֵן לְךָ תַעְלֶה

הָעִיר לְרֹדֶם חַדֵּשׁ כְּקֶדֶם

הוֹד עָם אֲשֶׁר יִשְׁלֶה וּבְכֵן לְךָ תַעְלֶה.

הָטְבַּע בְּחֶטְאוֹ עַד כִּי כְלָאוֹ

פִּשְׁעוֹ בְּבוֹר כֶּלֶא וּבְכֵן לְךָ תַעְלֶה

לָכֵן בְּשִׁבְיוֹ יוֹדֶה בְּמִרְיוֹ

עַיִן לְךָ יִדְלֶה וּבְכֵן לְךָ תַעְלֶה

וּבְשִׁיר יְהֹודְךָ כִּי מִכְּבוֹדְךָ

עוֹלָמְךָ מָלֵא וּבְכֵן לְךָ תַעְלֶה

יֵטִיב בְּרַחְשׁוֹ קָדוֹשׁ בְּשַׁלְּשׁוֹ

לְךָ צוּר עֲשׂה פֶלֶא וּבְכֵן לְךָ תַעְלֶה.

סלוק

אָמַר אוֹיֵב אֶרְדֹּף כְּאַרְיֵה וְאֶחְטֹף

וְאַשִּׂיג כְּשִׁבֹּלֶת וְאֶשְׁטֹף

וְאֵשׁ קְרָב אַדְלֵק

לְבֶן אֲגָג וַעֲמָלֵק אֲחַלֵק שָׁלָל

וַיֹאמֶר יְיָ בִּי נִשְׁבַּעְתִּי

אִם אֶעֱקֹר אֲשֶׁר נָטַעְתִּי

פָּנוּ דֶרֶך וְסֹלּוּ

כִּי שׂוֹלֲלֵיכֶם תָּשֹׁלּוּ וְעַתָּה לַשּׁלָל

אָמַר אוֹיֵב גָזַרְתִּי בַּחֹדֶשׁ אֲדָר

לִכְרוֹת פְּרִי עֵץ הָדָר

כִּי מִי יַעֲרֹךְ אֵלַי בְּכֹחַ

וְאֵיךְ מַלְקוֹחַ יֻקַּח מִגִּבּוֹר

וַיֹאמֶר יְיָ דָּרַכְתִּי קֶשֶׁת

לְעֹבְדֵי הַבֹּשֶׁת

וְעֵינַי עַל צֹאנִי אֶפְקַח

גַּם שְׁבִי גִּבּוֹר יֻקַּח

מַה תִּתְהַלֵּל בְּרָעָה הַגִּבּוֹר

אָמַר אוֹיֵב הִנֵּה עַם מְפֹרָד וּמְפֻזָּר

וּמִכִּסֵא מְלוּכָה נִגְזָר

בָּשֵׁל קָצִיר

וְיָבַשׁ חָצִיר לְכוּ וְנַכְחִידֵם מִגּוֹי

וַיֹאמֶר יְיָ וְאִם לֹא בְּרִיתִי יוֹמָם וָלַיְלָה

וְעַמּוּדֵי אֶרֶץ תִּכַּנְתִּי סֶלָה

כֵּן בְּרִיתִי עִם בָּנַי

לְבַל יִכָּרְתוּ מִלְפָנַי מִהְיוֹת גוֹי

אָמַר אוֹיֵב זְעוּמַת בַּעְלָהּ

גְּרוּשַׁת אָהֳלָהּ

הָכִינוּ שְׁלָחִים

מִרְקוּ הָרְמָחִים

כִּי אֵין גֹּאֵל

לִבְתוּלַת יִשְׂרָאֵל נָפְלָה לֹא תוֹסִיף קוּם

וַיֹּאמֶר יְיָ חַי אֲנִי נִשְׁבַּעְתִּי

שְׁאוֹנְכֶם שָׁמַעְתִּי

וּמִמַּעַל אֶקְרָא עַל שֹׁכְנֵי עָפָר

עֻצוּ עֵצָה וְתֻפָר דַּבְּרוּ דָּבָר וְלֹא יָקוּם

אָמַר אוֹיֵב טֶבַח אָכִין וְאַקְדִּישׁ קְרֻאַי

לְאַלְפֵי יְהוּדָה שְׂנֻאָי

כִּי הִנֵּה פִסְלִי

יַעֲזָר לִי וְאַיֵּה אֱלֹהֵיהֶם

וַיֹּאמֶר יְיָ יָדַעְתִּי צְפוּן חֻבָּם

וּטְמוּן לִבָּם

כִּי לֹא מַחְשְׁבוֹתַי מַחְשְׁבוֹתֵיהֶם

וְלֹא דְרָכַי דַּרְכֵיהֶם יִפְּלוּ מִמֹּעֲצוֹתיהֶם

אָמַר אוֹיֵב כַּמָּה תִבְטְחוּ עַל שֶׁקֶר

וְשׁוֹט שׁוֹטֵף בּטֶרֶם בֹּקֶר

אַיֵּה אֱלֹהֵיכֶם

יָקוּמוּ וְיַעְזְרֻכֶם

יְהִי עֲלֵיכֶם סִתְרָה

עַל מִי תָּנוּסוּ לְעֶזְרָה וְאָנָה תַעַזְבוּ כְבוֹדְכֶם

וַיֹאמֶר יְיָ לָכֵן חַכּוּ לִי

לְיוֹם אַרְבֶּה חֵילִי

וּבֹגְדִים יִבְלוּ כְּבֶגֶד

וְאַתֶּם תַּעַמְדוּ מִנֶּגֶד יְיָ יִלָּחֵם לָכֶם

אָמַר אוֹיֵב מַה יִפְעַל אֵל

לְגוֹלַת אֲרִיֵאל

כִּי כְפֶשַׂע בֵּינָם וּבֵין הַמָּוֶת

וְיֹאבְדוּ בְּאֶרֶץ צַלְמָוֶת וְאַיֵּה כָל נִפְלְאוֹתָיו

וַיֹּאמֶר יְיָ נָתַתִּי רוֹזְנִים לְאַיִן

וּתְשׁוּעָתִי כְּהֶרֶף עָיִן

וְחֵלֶק רַע אֲחַלֵּק

לְבֶן אֲגָג וַעֲמָלֵק

הַצִּיבוּ לְפָנָיו מִזְבֵּחַ

הָכִינוּ לְבָנָיו מַטְבֵּחַ בַּעֲוֹן אֲבוֹתָיו

אָמַר אוֹיֵב סְפָרַי קִדַּמְתִּי

וּנְדָרַי שִׁלַּמְתִּי

בָּךְ שְׁאֵרִית יַעֲקֹב לְכַלּוֹתֵךְ

אֲנוּשָׁה נַחְלָה מַכָּתֵךְ רְפֻאוֹת תְּעָלָה אֵין לָךְ

וַיֹּאמֶר יְיָ עַל אֵלֶּה לֹא אֶתְאַפַּק

וְעַל מַרְכְּבוֹת יְשׁוּעָה אֶתְרַפָּק

אַל תִּירָא יַעֲקֹב עַבְדִּי

מֵאִתְּךָ לֹא יָמוּשׁ חַסְדִּי אָנֹכִי מָגֵן לָךְ.

צוֹרֲרִי פָּקַד עָלַי מַלְאֲכֵי חֲמָתוֹ וְקִצְפּוֹ

וְצוּרִי צִוָּה מַלְאֲכֵי רַחֲמִים לְבַל יַשְׁחִית וְהִרְבָּה

לְהָשִׁיב אַפּוֹ

צוֹרֲרִי קָרָא עָלַי מוֹעֵד וְעִתֵּד מַלְאֲכֵי מָוֶת בִּכְלֵי מִלְחַמְתּוֹ

וְצוּרִי רָכַב עַל כְּרוּב יֶשַׁע וְשָׂם בַּעֲדִי צִנָּה

וְסוֹחֵרָה אֲמִתּוֹ

צוֹרֲרִי שָׁלַח מָלְאָכָיו אֶפְעֶה וְשָׂרָף מְעוֹפֵף

וְצוּרִי תָּמַךְ יְמִינִי בְּרָכְבוֹ עַל כְּנַף שָׂרָף

בִּשְׂתַּיִם יְעוֹפֵף.

יְעוֹפֵף בִּשְׁתַּיִם וּבִשְׁתַּיִם יְכַסֶּה פָנָיו וּבִשְׁתַּיִם יְכַסֶּה רַגְלָיו

הֲמוֹנִים הֲמוֹנִים מַכְנִיסֵי תַחֲנוּנִים פְּנֵי אֲדֹנֵי הָאֲדֹנִים

וְכָל צְבָא זְבֻלָיו עֹמְדִים עָלָיו

וַיֹּאמְרוּ חוּסָה עַל אֹם עֲמוּסָה נְתַתָּהּ לִמְשִׁיסָה

וּגְעַר בְסוֹרֵר וְהָשֵׁב לְצוֹרֵר כִּפְרִי מַעֲלָלָיו

דַהֲרוֹת אֶרְאֶלִּים חֲיָלִים חֲיָלִים זֶה אֶל זֶה שֹׁאֲלִים

וְלִדְבַר אֵל יָחוּשׁוּ מִפִּיהֶם לֹא יָמוּשׁוּ מַהֲלָלָיו

הַיּוֹם זֶרַע קֹדֶשׁ יַקְדִּישׁוּהוּ בְּסוֹד שַׂרְפֵי קֹדֶשׁ וְשֹׁאֲלִים

זֶה אֶל זֶה כְּאֶרְאֶלָּיו וּכְרוּבָיו אֵל נַעֲרָץ בְּסוֹד קְדוֹשִׁים

רַבָּה וְנֹורָא עַל כָּל סְבִיבָיו.
