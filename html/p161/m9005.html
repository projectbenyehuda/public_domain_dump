אֶל חַכְמֵי נַרְבּוֹנָא

לִגְבִירֵי תוֹרָה, וְאַלּוּפֵי מִשְׂרָה, מְיַשְּׁרֵי עֲקַלְקַלּוֹת, וּבְלִבָּם מְסִלּוֹת, קָדְשֵׁי קָדָשִׁים, אֶל ה' נִגָּשִׁים, גֹּאֲלִי הַגֹּלָה וּמַחְלִימֵי הַחוֹלָה, וְלָהֶם מִשְׁפָּט הַגְּאֻלָּה, עַם נְדָבוֹת, וּבָרֵי לְבָבוֹת, מִכִּסֵּא כָבוֹד חֻצְּבוּ, וְעָלָיו יִתְיַצְּבוּ, נְתֻנִים לְהָבִין וּלְהֹרוֹת, וְהָיוּ לִמְאֹרוֹת, כָּל מְבַקֵּשׁ לַעֲמֹד בְּסִינַי יַעֲמֹד עַל נְוֵיהֶם, וְדֹרֵשׁ כְּבוֹד ה' יִדְרֹשׁ אֶת פְּנֵיהֶם, בַּאֲשֶׁר דָּרְכוּ, תַּחְתָּם סִינַי, וּבַאֲשֶׁר הָלְכוּ, עֲלֵיהֶם כְּבוֹד ה', כֵּן יִהְיֶה תָמִיד הוֹדָם, בַּל יוּסָר, וּכְבוֹדָם, בַּל יֶחְסָר, וּפְאֵרָם יִרְבֶּה, וְנֵרָם לֹא יִכְבֶּה, הֵם הֵם מָרֵי גָלֻיּוֹת, וְחֶבְרַת מַרְגָּלִיּוֹת, יְדִידַי וְנִכְבָּדַי, אֲהוּבַי וְנֶחְמָדַי, רַב פְּלוֹנְִי וְרַב פְּלוֹנִי, הַנַּעֲלָמִים מֵעֵינַי, הַנִּקְשָׁרִים בְּרַעְיוֹנַי, הַקְרוֹבִים אֵלַי, הָרְחוֹקִים מֵעָלַי, וְאֵיךְ לֹא קְרוֹבִים, וְהֵם בַּקְּרָבִים, וְאֵיךְ לֹא רְחוֹקִים, וְהֵם בַּשְּׁחָקִים. כֵּן יְהִי רָצוֹן לִשְׁמֹעַ תְּפִלָּתָם, וּלְהַצִּיב שְׁמוּעָתָם, וִיגַלֶּה לָהֶם עֲתֶרֶת שָׁלוֹם, אָמֵן.

מִמֶּנִּי אֲנִי יוֹסֵף הַלֵּוִי בר' מֵאִיר נ"ע דֹּרֵשׁ שְׁלוֹמְכֶם, וְשֹׁגֶה בְּאַהֲבַתְכֶם, הַמִּתְבָּרֵךְ בְּשִׁמְכֶם, הַמִתְבַּשֵּׂר בִּשְׁלוֹמְכֶם, יַד דֶּרֶךְ שֹׁמֵר, וּבִגְלַלְכֶם אֹמֵר, מִי יִתֶּן לִי אֵבֶר כַּיּוֹנָה, אָעוּפָה וְאֶשְׁכֹּנָה, לְהַרְחִיק נְדֻדִּים, וְלִרְאוֹת יְדִידִים, וְלִקְשֹׁר פֵּרוּדִים, וְלַצְמִיד אֲחָדִים, כֵּן הָיְתָה תְפִלָּתִי, מִתְּחִלָּתִי, וְזֹאת בְּרִיתִי, עַד אַחֲרִיתִי, וּמֵאָז הֶעִירַנִי פְלוֹנִי, הַבָּא מֵאִתְּכֶם, וּמֵבִיא אִגַּרְתְּכֶם, מַה יָּקָר הַבָּא, וּמַה נֶּחְמָד הַמּוּבָא, מִכְתָּב שְׁלַחְתֶּם, וְלֵבָב לְקַחְתֶּם, הַחוּר בִּתְכֵלֶת שִׁבַּצְתֶּם, וְהַשֹּׁהַם בַּפּוּךְ הִרְבַּצְתֶּם, הוּא הַסֵּפֶר, הַנֹּתֵן אִמְרֵי שֶׁפֶר, טוּרָיו בְּתַרְשִׁישׁ מְמֻלָּאִים, וּדְבָרָיו בְּפָז מְסֻלָּאִים, בְּעִנְיְנֵי הַחָכְמָה, הִתְחַבְּרוּ גְשָׁמָיו, וּבְעִנְיְנֵי מְזִמָּה, הִתְקַדְּרוּ שָׁמָיו, צָפוֹן הֵנִיפָה גְשָׁמָיו, וְתֵימָן הִרְעִיפָה בְשָׂמָיו, קִדְּמוּהוּ בַדֶּרֶךְ, וַיִּקְרְאוּ לְפָנָיו אַבְרֵךְ, הִתְעוֹרֵר עֲלֵיהֶם מַחְשָׁבִי, וָאֹמְרָה בִּלְבָבִי, מִי זֹאת הַנִּשְׁקָפָה כַלְבָנָה מְקֻטֶּרֶת מֹר וּלְבוֹנָה, אֵין זֶה כִּי אִם לִידִידַי, וּמִכְתָּבָם הַבָּא לְיָדַי, כִּי לָהֶם לְבַדָּם יִתָּכֵן, וַיְהִי כֵן. לְקַחְתִּיו, וַיַּנְעִימוּנִי לְקָחָיו, פְּתַחְתִּיו, וַיְשַׁעֲשְׁעוּנִי פְתָחָיו, וְגַם הַיָּשִׁישׁ הַנִּזְכָּר, בְּטוּב מַעֲלָלָיו נִכָּר, נִכְמְרוּ לוֹ נִחוּמַי, וְהָמוּ עָלָיו רַחֲמַי, וְהַנֶּפֶשׁ לְעָנְיוֹ מִתְמוֹגֶגֶת, וְאֵין הַיָּד מַשֶּׂגֶת, וְאוּלָם עֲזַרְתִּיו כְּפִי כֹחִי, וַהֲנִידוֹתִי בְשִׂיחִי, וְלֹא אֵחֲרָה תְשׁוּבָתִי, עַצְלוּת אַהֲבָתִי, וְהַיּוֹדֵעַ וָעֵד יוֹדֵעַ, כִּי הַנֶּפֶשׁ בָּכֶם מִתְאַמֶּצֶת, וְלָרוּץ עָדֵיכֶם מִתְרוֹצֶצֶת, אַךְ רָחוֹק (הַדֶּרֶךְ) בֵּינֵינוּ, וְאֵין הַיָּמִים בִּרְצוֹנֵנוּ. וְגַם יִוָּדַע רֹב שִׂמְחָתִי בָּאִגֶּרֶת, וְתוֹחַלְתִּי לְאַחֶרֶת, וְעַתָּה בְחַסְדְּכֶם, אַל יִרְפּוּ יְדֵיכֶם, וְכִתְבוּ וְחַדְּשׁוּ, וּשְׁנוּ וְשַׁלְּשׁוּ, וְיֶשׁ לִי צֹרֶךְ, לוּלֵי אֹרֶךְ הַדֶּרֶךְ, לִבְחֹר לִי מִכָּל הַפֵּרוּשִׁים, פֵּרוּשׁ סֵדֶר קָדָשִׁים, לוּ יֵשׁ נָדִיב יַעֲנִיקֶנּוּ, וְסוֹפֵר יַעְתִּיקֶנּוּ, וַאֲנִי אֶתֵּן אֶת שְׂכָרוֹ, כִּי יָבוֹא דְבָרוֹ, וְעַתָּה אִם יֵשׁ אֶת נַפְשְׁכֶם, לִדָּרֵשׁ לְדֹרֶשְׁכֶם, מַלְאוּ כַיּוֹם יֶדְכֶם, וּמַלְּאוּ חֵפֶץ דּוֹדְכֶם, וְאִם לֹא תְיַאֲשׁוּנִי מִמֶּנוּ, הוֹדִיעוּנִי בַּמֶּה אֲקַדְמֶנּוּ, וְהִנֵּה פָנַי מֻעָדוֹת, עַל אִגְּרוֹתֵיכֶם הַנִּכְבָּדוֹת, סַמְּכוּנִי בְתַפּוּחֵיהֶן, וְרַפְּדוּנִי בִרְקֻחֵיהֶן, וְהָאֱלֹהִים יַגְדִּיל שִׁמְכֶם, וְיַעֲנֶה אֶת שְׁלוֹמְכֶם, אָמֵן.


האגרת כתובה בשמו של ר' יוסף הלוי אבן מיגש.
