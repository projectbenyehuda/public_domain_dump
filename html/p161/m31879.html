עֲלֵי כָזֹאת תִּבְכֶּינָה

קינה על משה אבן עזרא


עֲלֵי כָזֹאת תִּבְכֶּינָה / שַׁאֲנָנוֹת וְתֶהֱמֶינָה

וּמִסְפֵּד מַר תִּשֶּׂאנָה / וְתִקְרַעְנָה הַכֻּתֳּנוֹת.

עֲלֵי כָזֹאת הַתּוּגָה / אֵין לַדְּמָעוֹת פּוּגָה

וּבְכָל נֶפֶשׁ דְּאָגָה / וּבְכָל לֵבָב מְגִִנּוֹת –

עַל אֲבֵדַת אֶדֶר יְקָר, / נֶעְקַר וְכָל לֵב עָקַר

וּבַת כָּל עַיִן נָקַר / וַיַּחְשִׁיךְ אִישׁוֹנוֹת.

עֱנוּ, בַּעֲלֵי יְדִידוֹתָיו, / מְלִיצוֹתָיו וְחִידוֹתָיו,

וְכִתְבוּ אֶת אוֹדוֹתָיו / עַל סֵפֶר הַקִּינוֹת.

גְּבִיר אֶזְרָח, אֲשֶׁר כַּגֵּר / שָׁכַן בְּמַעְצָר וּמַסְגֵּר

וְתַחַת רֵעִים אוֹגֵר / הָאֲנָחוֹת וְהַיְגוֹנוֹת.

הֲגוּ ״עַל כָּל רֹאשׁ קָרְחָה / וּבְכָל חוּצוֹת צְוָחָה

וַהֲנָחָה תְהִי אֲנָחָה / וְעֵינַיִם – מַעֶיָנוֹת״.

וְ״הוֹי אָדוֹן וְהוֹי הוֹדוֹ״ / סִפְדוּ עַל מִפְקָדוֹ

וּלְתַנּוֹת עַל מְרוּדוֹ / הִתְחַבְּרוּ לַיְעֵנוֹת.

עֲטֶרֶת גֵּאוּת סְפָרַד – / אֵיךְ מַרְאֲשׁוֹתָיו יָרַד

וַיִּשָּׁאֵר מְפֹרָד / מִבָּנִים וּמִבָּנוֹת?

כַּת אַחִים אַרְבָּעָה / כְּמוֹ כִימָה נִקְבָּעָה –

רוּחַ אֵיךְ נָסָעָה / וְנָגְעָה בְּאַרְבַּע פִּנּוֹת?

וְתַשְׁלִיכֵם אֶל אֶרֶץ / אַחֶרֶת בְּיוֹם חֶרֶץ

וּפָרַץ בָּהֶם פֶּרֶץ / אֲשֶׁר אֵין לִגְדֹּר וְלִבְנוֹת.

הָרַס מִבְנֵה עִירָם / וְלֶעָרִים הֶעֱבִירָם

וְנָטַשׁ אֶת אַדִּירָם / בְּבֵית גֵּרוּת לֵעָנוֹת.

אֲסָרוּהוּ בְמַחֲבוֹאִים / כַּשּׁוֹכֵן בֵּית כְּלָאִים

צוֹעֵק מִתַּחֲלוּאִים / וּמִמַּכּוֹת נֶאֱמָנוֹת.

אַיֵּה אֶשְׁכּוֹל כֹּפֶר? / אַיֵּה חָכָם וְסוֹפֵר

וְנוֹתֵן אִמְרֵי שֶׁפֶר / וְשִׂיחוֹתָיו בִּנְגִינוֹת?

פְּאֵר זְקֵנִים וִישִׁישִׁים / מִתְעָרֵב בִּקְדוֹשִׁים,

עִם אֲדֹנָי וְעִם אֲנָשִׁים / נֶאֱמָן שׁוֹמֵר אֱמוּנוֹת.

נְעִים זְמִירוֹת יִשְׂרָאֵל / וְהוֹד הָאֲרָצוֹת הָאֵל

וּמַחֲזִיק בְּתוֹרַת הָאֵל / וּמַחְמַדֵּי הַתְּבוּנוֹת.

סְתוּמוֹתָיו מִי יְבָאֵר / וְדוֹר עָרוֹם יִשָּׁאֵר,

אַחֲרָיו בְּמִי יִתְפָּאֵר / בַּגּוֹיִם וּבַלְּשׁוֹנוֹת?

גְּבִיר הָעָם וּבְחִירוֹ – / אָכֵן בַּעֲווֹן דּוֹרוֹ

סָתַם אֵל אֶת אוֹרוֹ / וְהִשְׁכִּין עָלָיו עֲנָנוֹת.

מֹשֶׁה מֹשֶׁה אָחִי, / מְאוֹר סַהֲרִי וּמִזְרָחִי

וְנֵרִי וְנֹגַהּ זַרְחִי / מִשָּׁנִים קַדְמוֹנוֹת.

כְּאֵבִי נוֹסַף בְּיוֹסֵף / לִדְמוּתוֹ אֶכָּסֵף,

אַלְלַי כִּי אֵאָסֵף / וְלֹא אַבִּיט לַתְּמוּנוֹת.

בִּתְשׁוּקָתָם נֶאֱסַף / וּבְנֶפֶשׁ מָרָה סָף,

בְּיוֹם אֲשֶׁר נִגְדַּע וָסַף / אַל תָּבֹאנָה רְנָנוֹת!

יוֹדֵעַ יְגוֹנוֹתָיו / וְהַמְקַצֵּר אֶת שְׁנוֹתָיו

יְכַפֵּר בָּם עֲווֹנוֹתָיו / הַגְּלוּיוֹת וְהַצְּפוּנוֹת.

וִינַחֵם אֶת אֲבֵלָיו / וְיִפְרֹשׂ כְּנָפוֹ עָלָיו

וְיִפְנֶה בְרַחֲמִים אֵלָיו / וְאַחְיוֹתָיו הַבְּחוּנוֹת.

וְיִתֵּן אֶת מַהֲלָכָיו / בֵּין כְּרוּבָיו וּמַלְאָכָיו

בְּצֵל שַׁדַּי וּמַעֲרָכָיו / וּבִמְנוּחוֹת שַׁאֲנָנוֹת.


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
