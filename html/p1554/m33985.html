וְהַשְּׁמוּעָה בָאָה

מתוך: צלצל כנפים או אוהב נשים. אבן־שבתי יורד ארצה מגן עדן.



וְהַשְּׁמוּעָה בָאָה עַד יְהוּדָה בֶן־שַׁבְּתַי בֵּין צִבְאוֹת נְגֹהִים / בְּעֵדֶן גַּן אֱלֹהִים – / הוּא יְהוּדָה הַיָּדוּעַ בֵּין הָאֲנָשִׁים / שׂוֹנֵא הַנָּשִׁים. / וַיִּתְאַפַּק הָאִישׁ בְּשָׁמְעוֹ מִלַּי / וַיָּעָף אֵלַי, / וַיֶּחֱרַד וְשַׂעֲרַת רֹאשׁוֹ תְּסַמֵּר, / וַיִּטֹּשׁ סֻכָּתוֹ בַּגַּן עַל יְדֵי שׁוֹמֵר / וַיַּפְקֵד שַׂר עַל מַנְעוּלָיו וּבְרִיחָיו, / וַיֵּרֶד יְהוּדָה מֵאֵת אֶחָיו. / וַיִּתְיַצֵּב עַל אֵם הַדֶּרֶךְ לְדַבֵּר אֵלַי – / וַאֲנִי הִתְחַזַּקְתִּי כְּיַד אֱלֹהַי הַטּוֹבָה עָלַי. / וַיִּתְכַּסֶּה בָאֲפֵר עַל עֵינָיו / וְקֶסֶת הַסּוֹפֵר בְּמָתְנָיו, / שׁוֹקָיו אֳנִי תַרְשִׁישׁ, גְּוִיָּתוֹ כְתַרְשִׁישׁ, / וְהוּא חָגוּר חֶרֶב כְּמַרְאֵה חֶרֶב בְּנֵי עַמּוֹן / וְקוֹל דְּבָרָיו כְּקוֹל הָמוֹן. / פָּתַח בִּבְדִיחוּתָא / לִשָּׁנָא דְצַחוּתָא, / מְתוּקוֹת מִדְּבַשׁ מִדַּבְּרוֹתָיו, / נֹפֶת תִּטֹּפְנָה שִׂפְתּוֹתָיו. / וָאֶתְבּוֹנֵן אֵלָיו וָאַכִּירֵהוּ / כִּי מִבְּנֵי הַכְּסִילִים הוּא, / וָאֹמַר ״הַשְׁמִיעֵנִי, אֲדוֹנִי, סְפוּן סוֹדֶךָ, כִּי שׁוֹמֵעַ עַבְדֶּךָ״.

וַיֹּאמֶר יְהוּדָה בֶן־שַׁבְּתַי ״בַּמֶּה סָבַרְתָּ לְהַחֲרִיד יְחִידָתִי? / לָמָּה הִרְגַּזְתַּנִי לְהַעֲלוֹת אוֹתִי? / אֲשֶׁר חָפַצְתִּי עָזַבְתָּ / אֲשֶׁר שָׂנֵאתִי אֲַהַבְתָּ – / וְאוּלָם עַתָּה הָרִיב נְטֹשׁ וּבַקֵּשׁ שָׁלוֹם, / כִּי הֲבִיאוֹתַנִי עַד הֲלֹם!״ / וָאַעַן וָאֹמַר ״בִּי אֲדוֹנִי, אִם כַּיּוֹם הֵפַרְתִּי כְבוֹדֶךָ – / אַל יִחַר אַפְּךָ בְּעַבְדֶּךָ, / כִּי לֹא דִמִּיתִי לְהָשִׁיב אֶל אֲדוֹנִי עַזּוֹת, / בְּתֹם לְבָבִי וּבְנִקְיוֹן כַּפַּי עָשִׂיתִי זֹאת! / כִּי הַנָּשִׁים בְּשִׂכְלָם – / יְסוֹד עוֹלָם, / וַהֲמוֹן הַיַּעֲלוֹת הַיְּפֵיפִיּוֹת – / מִגְדַּל עֹז בָּנוּי לְתַלְפִּיּוֹת. / וַתֹּאֹמֶר, אֲדוֹנִי, לַמּוֹלִידִים בְּצַלְמָם ׳לֹא תַרְבּוּ אָשָׁם!׳ / כִּמְעַט שָׁכַב אַחַד הָעָם אֶת אִשְׁתּוֹ וְהֵבֵאתָ עָלֵינוּ אָשָׁם. / וַתֹּאמֶר לְשׁוֹמֵר הַתּוֹלָדוֹת ׳נַפְשְׁךָ טָבַחְתָּ! / הִנְּךָ מֵת עַל הָאִשָּׁה אֲשֶׁר לָקַחְתָּ׳. / חָלִילָה לַאדוֹנִי אִם לְשֹׁד הָאַיָּלוֹת תִּשְׂחַק – / מִדְּבַר שֶׁקֶר תִּרְחַק! / חָלִילָה לְּךָ מִכְּתֹב לַהֲמוֹן הַיְּעָלוֹת גֵּט – / וְהָיָה בְךָ חֵטְא! / וְאוּלָם עַד כֹּה רַעְיוֹנַי יוֹעֲצִים: / נַחְנוּ נַעֲבֹר חֲלוּצִים / לִפְנֵי שְׁנֵי בְנֵי הַפַּרְתְּמִים / הַנֶּאֱהָבִים וְהַנְּעִימִים, / נוֹשְׂאֵי צִנָּה, / יוֹדְעֵי בִינָה, / אֲנָשִׁים אַחִים, / טוֹבִים וּנְכוֹחִים, / תִּפְאֶרֶת הַסְּגֻלּוֹת, / עֲטֶרֶת הַתְּהִלּוֹת, / אֲשֶׁר נַפְשָׁם בִּקְהַל הַמַּעֲלָה וְהַמַּדָּע תֵחַד – / תּוֹרָה וּגְדֻלָּה בְּמָקוֹם אֶחָד. / הַתְּהִלָּה נִמְשֶׁכֶת עִמָּם / וַתְּהִי הַמִּשְׂרָה עַל שִׁכְמָם, / כָּל אֶחָד דֵּעָה יוֹרֶה / הַשְּׂרִידִים אֲשֶׁר יְיָ קוֹרֵא, / אֲשֶׁר בִּמְזוֹר חֶמְדָּתָן יְרַפְּאוּ כָל נֶגַע וָפֶשַׁע, / אַנְשֵׁי אֱמֶת שׂוֹנְאֵי בֶצַע, / רָאשֵׁי הַפִּנּוֹת, / שָׂרֵי הַמְּדִינוֹת, / כָּל אֶחָד יוֹדֵעַ צְפוּנוֹת / נָגִיד וּמְצַוֶּה וּמַחֲזֵה שַׁדַּי יֶחֱזֶה, / הַיּוֹשְׁבִים בְּאַרְלֵדִי בִגְלִילוֹת פְּרוֹוֶינְצָא בַּמִּקְצוֹעַ הָרִאשׁוֹן מִקָּצֶה מִזֶּה. / לִפְנֵיהֶם יָרֹן רַחֲשִׁי / בְּסוֹדָם אַל תָּבוֹא נַפְשִׁי./ עֲצָתָם תַּנְחֵנוּ כִּרְצוֹן שׁוֹכְנֵי סְנֶה וְכוֹכְבֵי נְגֹהִים – / כַּאֲשֶׁר יִשְׁאַל אִישׁ בִּדְבַר אֱלֹהִים, / אֲשֶׁר לְבָבִי לַחֲזוֹת בְּנָעֳמָם נֶחְפָּז – / שְׁנֵי טוּרֵי שֹׁהַם הַמְמֻלָּאִים בַּפָּז.

הַטּוּר הָרִאשׁוֹן – / מֵאִיר בֶּן־שְׁלוֹמוֹן, / הָעוֹמֵד בַּפָּרֶץ / הַמֵּאִיר לָאָרֶץ, / מֹר וַאֲהָלִים, / יוֹשֵׁב אֹהָלִים, / עוֹטֶה אֵפוֹד בַּד, / וְהָאִישׁ נִכְבָּד / רַעֲנָן כְּגַן רָטֹב / הוֹלֵךְ וְגָדֵל וָטוֹב. / נִשְׁאַל כַּיּוֹם אֶת פִּיו לְהַיְשִׁירֵנוּ / וְהִגִּיד לָנוּ אֶת דְּרָכֵינוּ, / בַּעֲצָתוֹ הַטּוֹבָה יֵדַע צְפוּן הַנּוֹלָד וְחֻבּוֹ – / כֹּל אֲשֶׁר יְדַבֵּר בּוֹא יָבוֹא.

הַטּוּר הַשֵּׁנִי – יְהוּדָה אָחִיו בֶּן־שְׁלֹמֹה אֲשֶׁר מִן הַחֲמוּדוֹת אֵלָיו יִתְלַקְּטוּ סְבִיב מַחֲנֵהוּ / וְהָאִישׁ גָּדוֹל וּמַעֲשֵׂהוּ. / אִישׁ תֹּאַר וַייָ עִמּוֹ, וְלוֹ בַחֲמוּדוֹת – / עֶשֶׂר יָדוֹת, / מְסִלַּת לְבָבוֹ לַיְדִידִים רְחָבָה / תּוֹכוֹ רָצוּף אַהֲבָה, / בְּחַצְרוֹת לְבָבוֹ וְטִירָתוֹ יַנְחֵם, / לֹא יְשַׁקֵּר וְלֹא יִנָּחֵם. / וְכָמוֹהוּ מַנְעוּל סֵתֶר תְּשׁוּקָתִי וְהוֹדִי / קַמְתִּי אֲנִי לִפְתֹּחַ לְדוֹדִי, / זִכְרוֹ אָדוֹן לְרַעְיוֹנַי שַׂמְתִּיו / וְנָגִיד לִי נְתַתִּיו. / הוּא אֲשֶׁר מְצָאתִיו יַעְרִי וְדִבְשִׁי – / אֵת שֶׁאַהֲבָה נַפְשִׁי, / אִישׁ אֱמֶת אֲשֶׁר הֶחָמָס זָעַם, / כִּי אֵין כָּמוֹהוּ בְּכָל הָעָם. / אֲשֶׁר לִבּוֹתֵינוּ לֹא יִמְצְאוּ בְרַעְיוֹנֵיהֶם / יִשְׁפְּטוּ הֵם, / כִּי הֵמָּה הַמַּחֲזִיקִים כָּל בֶּדֶק / וְשָׁפְטוּ אֶת הָעָם מִשְׁפַּט צֶדֶק״.

וַיֹּאמֶר יְהוּדָה בֶן־שַׁבְּתַי ״טוֹב דִּבַּרְתָּ! הַשּׁוֹפְטִים יָאִירוּ רַעְיוֹנֵינוּ הַבָּאִים / וְהֵבֵאתָ אַתָּה אֶת הַדְּבָרִים אֶל הָאֱלֹהִים״. / וַנָּבוֹא יַחְדָּו אֶל בֵּיתָם / בְּחַצְרֵיהֶם וּבְטִירוֹתָם / וַנִּמְצָאֵם בַּעֲלִיַּת הַמְּקֵרָה פַּת הַתְּבוּנוֹת סוֹעֲדִים, / יוֹשְׁבִים אִישׁ עַל כִּסְאוֹ מְלֻבָּשִׁים בְּגָדִים. / וַיָּרוּצוּ לְעֻמָּתֵנוּ / וַיִּשְׂמְחוּ לִקְרָאתֵנוּ, / וָאֹמַר אֶל יְהוּדָה בֶן־שַׁבְּתַי ״דַּבֵּר אַתָּה לִפְנֵי כָל הֲמוֹנַי, / דְּרָשׁ נָא כַיּוֹם אֶת דְּבַר יְיָ!״

וַיְדַבֵּר יְהוּדָה בֶן־שַׁבְּתַי לִפְנֵיהֶם וַיֹּאמַר ״אָמַרְתִּי: לְפָנַי אַל יִדְרְכוּ אֲנָשִׁים / דֶּרֶךְ נָשִׁים! / וָאֹמַר לְכָל נוֹשֵׂא אִשָּׁה בְּנֹעַם מְשָׁלַי: / הִפָּרֶד נָא מֵעָלַי! / וּלְכָל עוֹזְבָהּ וּלְכָל הֲמוֹנַי: / בּוֹא, בְּרוּךְ יְיָ! / כִּי הֵנָּה נְמִבְזוֹת וְנִקְלוֹת, וְאִם טוֹבוֹת מַרְאֵיהֶן – / נָשִׁים דַּעְתָּן קַלָּה עֲלֵיהֶן. / וְהָאִישׁ הַזֶּה טוֹבִיָּה הֶחֱלִישׁ חָסְנִי / הֵשִׁיב אָחוֹר יְמִינִי – / יַגִּיד לָכֶם כַּיּוֹם מַחְבַּרְתּוֹ אֲשֶׁר הִצִּיב לְהַאֲדִיבִי / וְסֵפֶר כָּתַב אִישׁ רִיבִי״. / וַיַּעֲנוּ שְׁנֵי הָאֲצִילִים, / שְׁנֵי הַמְּאוֹרוֹת הַגְּדוֹלִים, / וַיֹּאמְרוּ ״שְׁמַע נָא, יְהוּדָה הַמֵּלִיץ הַגָּדוֹל! אַל תִּמְעֹל בָּאַיָּלוֹת מַעַל, / אַל תִּתֵּן אֶת הָאִשָּׁה לִפְנֵי בַּת בְּלִיַּעַל – / וְהִנֵּה בִימֵי נֹחַ יָרְדוּ בְנֵי הָאֱלֹהִים לְהַעֲמִידָם בְּסוֹדָם, / אַף כִּי שׁוֹכְנֵי בָתֵי חֹמֶר אֲשֶׁר בֶּעָפָר יְסוֹדָם, / כִּי הִיא מִכְּבוֹד הָאֱלֹהִים נֶאֱצֶלֶת / וּמֵיְיָ אִשָּׁה מַשְׂכֶּלֶת, / הוֹד וְהָדָר נְסוּכָה, / חָלָק מִשֶּׁמֶן חִכָּהּ. / וְלַיִל נִבְצַר מִן הָאִישׁ יַעֲלָה – נִבְצַר מִמֶּנּוּ מִגְדַּל עֹפֶל, / הַלַּיְלָה הַהוּא יִקָּחֵהוּ אֹפֶל! / עַל כֵּן חֲדַל לְךָ וַחֲתֹם דְּבָרֶיךָ, / אַל תִּתֵּן אֶת פִּיךָ לַחֲטִיא אֶת בְּשָׂרֶךָ! / וְטוֹבִיָּה הִפְלִיא לַעֲשׂוֹת בְּנָעֳמוֹ / וְהַדִּין עִמּוֹ. / וְאוּלָם לִכְבוֹדְךָ יֵחָלֵק הַמִּשְׁפָּט לַחֵצִי, וּלְבַעֲבוּר סִפְרְךָ חָיֹה יִחְיֶה – / גַּם לוֹ גַם לְךָ לֹא יִהְיֶה! / וְשִׁמְעוּ הַדִּין אֲשֶׁר אֱמֶת יְסוֹבֲבֵהוּ, / כִּי הַמִּשְׁפָּט לֵאלֹהִים הוּא: / יָבִיא כָל אִישׁ עָפְרָה אֶל בֵּיתוֹ / לַעֲמֹד לְפָנָיו לְשָׁרְתוֹ / וְהֶעֱמִיד מִמֶּנָּה סְגָנִים וּסְגָנוֹת / וְיָלְדָה לוֹ בָנִים אוֹ בָנוֹת, / כִּי לֹא כָל רַע יְגוּרֵהוּ / אִם בַּעַל אִשָּׁה הוּא. / כָּל יְמֵי חֲיוֹתָהּ יִתְּנֶנָּה אֶצְעָדָה וְצָמִיד / בְּאַהֲבָתָהּ יִשְׁגֶּה תָמִיד, / וּבְיוֹם מוֹתָהּ יִסְפֹּד חֶמְדָּתָהּ – / וְיִשָּׁמֵר מִקַּחַת תְּמוּרָתָהּ / וּמִתֵּת מַלְכוּתָהּ לִרְעוּתָהּ. / וְאִם הַמָּוֶת הֶעֱבִירָהּ / וַיְסִירֶהָ מִגְּבִירָה, / יִשְׁמֹר אֶת יְלָדֶיהָ חַסְדּוֹ וַאֲמִתּוֹ / כַּאֲשֶׁר הָיְתָה בְּאָמְנָה אִתּוֹ, / כִּי כָל בְּרִיתָהּ יָפֵר בְּמַעֲלוֹ, / אִם אַחֶרֶת יִקַּח לוֹ. / וְלֹא יִתָּכֵן שֶׁיִּשְׁכַּח רִאשׁוֹנָה וְדֹדֶיהָ / וְיָצְאוּ יְלָדֶיהָ / וִינַשֵּׂא בֶן הַשֵּׁנִית וְרֶגֶל הָרִאשׁוֹן יִתֵּן לָמוֹט – / הָאֶחָד בָּא לָגוּר וַיִּשְׁפֹּט שָׁפוֹט! / וְכֹה דִבְּרָה לָנוּ בְּדַעַת וְחָכְמָה / תּוֹרַת יְיָ תְּמִימָה – / אֲשֶׁר בְּנַחֲלָתוֹ בְּנוֹ הַגָּדוֹל יִזְכֹּר / וְאֶת בֶּן הָשְּׂנוּאָה יַכִּיר כִּי הוּא הַבְּכוֹר: / כִּי אֵיכָכָה בְּעֵין לְבָבוֹ יִרְצֶה לִטֹּל מִזֶּה / וְלָתֵת לַזֶּה, / אַף כִּי הַגָּדוֹל עוֹמֵד בַּפָּרֶץ / לוֹ שֵׁם כְּשֵׁם הַגְּדוֹלִים אֲשֶׁר בָּאָרֶץ! / וְהַקָּטֹן לֹא נִסָּה בִּתְהִלּוֹת רוּחוֹ לְהָעִיר / אֲשֶׁר לֹא יָדַע לָלֶכֶת אֶל עִיר. / עַל כֵּן אָמַרְנוּ: תַּעֲמֹד עָפְרָה בִגְבוּלוֹ / אַחַת, לֹא יִשְׁנֶה לוֹ / וּלְבַד עֵינוֹ בְנוֹ הַגָּדוֹל בְּנַחֲלָתוֹ יִפְקַח / גַּם אֶת הַכֹּל יִקַּח״. / וַיַּעַן לָהֶם יְהוּדָה בֶן־שַׁבְּתַי כְּאִישׁ נָבָר / וַיֹּאמַר ״טוֹב הַדָּבָר!״ / וַיָּשָׁב לִקְרָאתוֹ וַיִּפְרֹשׂ כְּנָפַיִם / עַל פְּנֵי רְקִיעַ הַשָּׁמַיִם / וּמַלְאֲכֵי מַחֲנַיִם אוֹתוֹ נוֹשְׂאִים / וְטוֹבִיָּה וּמֵאִיר וִיהוּדָה רוֹאִים / וַיֹּאמְרוּ ״נְהַלֵּל הַבּוֹרֵא שׁוֹכֵן עֶרֶץ / כִּי רָאִינוּ אֱלֹהִים עוֹלִים מִן הָאָרֶץ״.



מקור: שירמן – השירה העברית בספרד ובפרובאנס; הוצאת מוסד ביאליק ודביר תשי״ז תשכ״א
