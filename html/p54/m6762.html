כִּי יָבוֹא הַמְּשׁוֹרֵר אֲלֵיכֶם

הַדּוֹר, כִּי יָבוֹא הַמְּשׁוֹרֵר אֲלֵיכֶם וְיִשָּׂא נִיב-נְכֹחוֹת,

כְּבַד רֶטֶט רְָגזָה וּכְעָסִים וְחֶמְלַת כָּל מָחָר,

עַל רָז הָאָדָם הַנִּפְגָּם וּמַפְלִיג עַל כָּל תֹּרֶן נֵכָר

בִּשְאוֹן תַּחְרֻיוֹת מְחַצְצְרוֹת אֶל הֶפְקֵר מַעֲרָכוֹת.

בִּזְדוֹן לַעְגְכֶם בּוֹ תִּרְהָבוּ: הֲלֹא יְמַשֵּל מְשָלִים –

שוּב דֶּרֶך גְּאֻלָּה יִבְדֶּה לָנּוּ וּשְׁלוֹם מִקְלָטִים;

שִׁלְפֵי תַנְחוּמִים לִרְזוֹן נֶפֶש יָתוּר וְיִתְבַּע גְּמוּל לְבָטִים,

כָּל שִׂיא מַדָּע לֹא סִלֵּק לוֹ עֲדֶנָה – חֶשְׁבּוֹן נִכְשָלִים.

לֹא הוּא בַרְזִלֵּנוּ לִבֵּן, לֹא זֵעַת נִצָּחוֹן וּפְחָדָיו

הִרְסִיסָה מִצְחוֹ בְּהַתְרִיס הָאָדָם עִם גְּלָמִים אֲטוּמִים:

לֹא הוּא לַגוֹרָל יַעֲמֹד לְכַהֵן בִּוְתִיקוּת, כִּבְפֻלְחָן חַרְטֻמִּים.

לְחֹסֶן פֶּחָם וְחַשְׁמַל – – – הוּא מֵאָז חֲזוֹן-שֶׁפֶר רָדָף,

וְרוּח יֶהְבַּל גַּם עַכְשָו… הָהּ, הַגֶּבֶר, אֵחַרְתָּ כִמְעָט:

מַה לָּנּוּ תִּסְכֹּן חָזוּת פְּדוּת תַּשִּׂישָׁה לָאָבוֹת,

וַאֲנַחְנוּ גֻבַּרְנוּ עַד כֹּה, וְלֹא לָנּוּ יְהִי לְכָבוֹד

לַחְלֹק עִמָּהֵם בְּעָנְיָם בְּשוֹרַת יֶשַׁע אֶחָת.

וּרְאֵה, גַּם צַלְמֵנוּ הָפַך תְּבַלוּל מֵאלֹהִים וְרַבַּת פְּסִילִים

סָגַד, הִדַּמָּה אֲלֵיהֶם, וּלְכּל חֹמֶר וָסֵמֶל זָנָה

עַתָּה, בְּנֵי דְרוֹר וּכְפִירָה, בְּאַחֲרוֹן עִוָּרוֹן וְקִנְאָה

נִשְבַּעְנוּ אֵמוּן לַמְּכוֹנָה… הָהּ, לַשָׁוְא תְּשַפֵּר נִיב פְּלִילִים –

וַאֲנַחְנוּ טָעַמְנוּ שִׂלְטוֹן וְנַעְפִּיל בִּרִתְמַת נִצְחוֹנוֹ,

מַה לָּנוּ, כּוֹבְשֵי רְקִיעִים, כָּל מִגְדָּל אוֹ סֻלָּם דּוֹר קַדְמוֹן?…

חוֹזֶה, לֵךְ בְּרַח לְךָ וַחֲצֹב פֶּלֶא-הֶמְנוֹן

לְבֹרֶג וּסְלִיל וְגַלְגַּל וּלְמַלְכוּת הָאַחֲרִית לַמְּכוֹנוֹת – – –

– הָהּ, נְאוּם עֲבָדִים שֶמָּשְׁלוּ!… שַׁאֲלוּ לִנְתִיבוֹת הָאָדָם,

מִי הֵם שֶפָּקְדוּ רִאשוֹנִים אֶת חֲזִית הַמְּרָדִים?

אִם לֹא חוֹזֵיכֶם נֶחְלְצוּ בּוֹדְדִים מִמַּחֲנוֹת גַּמָּדִים,

מִתַּחַת כְּנַף אָב רִאשוֹן, בְּבַגְרוּת וּבְרֵאשִית מְרִי-דָם.

עוֹד עַל כִּתְפֵיכֶם הֶעֱמִיקוּ צַלְקוֹת הַלְּבֵנִים וְהַטִּיט,

בְּפַחַד יְלָדִים צְבַרְתֶּם לְמִשְׁכַּן אֵל צְבָאוֹת;

עוֹד חֹנֶף קָרְבָּן וּתְפֶלָּה לוֹ עָשַׁן מֶמִּזְבַּח קְנָאוֹת

וְנֵר הַתַֹמִיד – מִי גָּנַב לְהָאִיר מֶשַּׁמְנוֹ הַכָּתֶית

אֶת דֶּרֶך לוּצֶיפֶר, אֲבִי כָּל מַדָּע, הִתְחָרוּת וִיגוֹנִים,

אֶת זֶה שֶׁגָּרָה דְמֵיכֶם הַקְּלוֹשִׁים וַיַעֲמֹד בִּבְרִיתְכֶם לְהָקִים

יַחְדָּו אֶת מִבְצַר הַגְּאוֹנִים וּלְהַחְצִיף עַל כָּל קְצוֹת מֶרְחַקִּים,

שֶׁשָׂם עֲלֵיכֶם בְּגִנְזֵי תַגְלִיוֹת, הִתְחַבֵּט בְּפִקּוּל פִּתְרוֹנִים?…

כִּי לָמָּה הַיּוֹם כָּל פִּרְחָח שֶׁדָּלַג עַל מִפְתַּן מַעְבָּדָה

בְּדֶגֶל הַדְּרוֹר יִתְרַבְרֵב, יִתְיַחֵשׂ לִבְרוּרֵי חֲלוּצָיו;

וּבֶטַח יִפְסֹק מִשְפָּטוֹ מִלּוּחוֹת מִסְפָּרִים לוֹ חָצָב –

כִּי כֶסֶל כָּל רוֹח חָזוֹן, כִּי בְדוּיוֹת, שַׂעֲשׁוּעֵי אַגָּדָה

כָּל עֶצֶב הֶגְיוֹן הַמְּשוֹרֵר הַמְסַלֵּד מֵעֲרֹך נַפְתּוּלִים

וְאֹמֶץ קְרָבוֹת יְרַשֵּל בְּמִטְעַן חִנּוּטִים וְטַפֵל – – –

אַתֶּם, בְּנֵי עַבְדוּת לְכָל אֵל וּמַנְגְנוֹן: בְּקוּם יוֹם עֲרָפֵל

וְרֶגַע יַחְמֹק דַּרְכְּכֶם לַמֵּצָר מִפִּרְזוֹן אֵין גְּבוּלִים;

וְאֵי שָׁם בַּלוּחוֹת יִשְׁתַּבֵּשׁ חֶשְׁבּוֹנְכֶם: שִׁבְעָתַיִם נִסְתָּר

יָגִיחַ הָרָז שֶׁפֻּלַּש וְגִמֵּד וּבְבְרֶית זְנוּנִים

בָּאתֶם כְּבָר עִמּוֹ – וַעֲמַדְתֶּם כֹּה שְׁפוּפִים וּכְפוּפִים

מוּל בַּעַל דְּבָבְכֶם מֵעוֹלָם עַד עוֹלָם, שֶׁכָּל פֶּתַח אָטָר,

טִמֵּר כָּל מָבוֹא בִּפְנֵיכֶם, וְלַשָּוְא תְּפַלְפְּלוּ צֵרוּפִים,

תַּעֲרִימוּ חִקְרֵי נֻסְחָאוֹת, כִּי הוּא מְסֻגַּר מַעְגָּלָיו

בְּאֵלֶם נִצְחוֹ יִתְכַּרְבֵּל מִמַשְׁתִּית עַד נְתִיב הֶחָלָב –

אָז יֵשׁ אֲשֶׁר אִישׁ הֶחָזוֹן עוֹד יָשוּב עַל תּוֹעַת דַּרְכֵיכֶם הַטְּרוּפִים,

יֵדַע כָּל גְּאוֹן עֱזוּזְכֶם וֶעֱנוּתְכֶם וִירַפֵּא כָּל אַכְזָב

בְּנֶפֶש אָדָם… הֲלֹא הוּא הָרִאשוֹן נְתִיב קִרְיָה לָהּ סָלָל,

וְהוּא יְכַפֵּר עַל הַמֶּרֶד – יִצְבֹּר חֲמוּדוֹת הַשָּׁלָל

עַל רֶכֶב לוּצִיפֶר: מַתַּת דוֹר גִּבּוֹר שֶׂנִּכְסָף לַחֲזוֹן אֱלֹהָיו.

את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
