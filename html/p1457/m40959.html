יַחַד עַד לֹא יְסוֹדִים קְדוּמָה הִיא

יְסוֹדָהּ יְכֻנֶּה קִינִים וָהֶגֶה וָהִי

יָחִיד הֵבִין דַּרְכָּהּ וְאַחֲרִיתָהּ מַה הִיא

תְּהוֹם אָמַר לֹא בִי הִיא


רִנַּת שִׁירָתָהּ מִי יוּכַל לֵידַע

רְשׁוֹם בָּהּ חָזוֹן וְכָל דְּבַר מַדָּע

רִנָּה פָּצְחוּ הָרִים לָזֹאת מוּדָע

רִמְזָהּ עֲמֻקָּה מִשְׁאוֹל מַה תֵּדַע


בָּהּ נִשְׁתַּעְשַׁע וְיָסַד וְהֵכִין רוּמָה

בְּמַאֲמָרָהּ הִרְקִיעַ עַמּוּדֵי בְלִימָה

בָּה נִמְלַךְ חֲצוֹב יְצִיר לַהֶבֶל דָּמָה

בְּרִיתוֹ זָכַר לְקַיֵּם חֶרֶס אֶת חַרְסֵי אֲדָמָה


יְמִינוֹ טִפְּחָה אָחוֹר וָקֶדֶם

יַחַד כְּמוֹ מָגְלָם אָחוֹר וָקֶדֶם

פֵּרַשְׁתָּ פִּלַּשְׁתָּ מִקַּדְמֵי קֶדֶם

יָעַץ וְהֵבִין אַחֲרִיתָהּ אֱלֹהֵי קֶדֶם


קְצִיבַת דִּבְרַת אֲרֻכָּה מֵאֶרֶץ

קְדוּמָה טֶרֶם תְּחִלַּת שָׁמַיִם וָאָרֶץ

קְנוּאָה לְגֵזַע לֶךְ לְךָ אֶל הָאָרֶץ

קְבוּעָה מִנִּי שִׂים אָדָם עֲלֵי אָרֶץ


לְבוּבָה בְחָכְמָה דַּעַת וְעֵצוֹת

לֶקַח כְּתָב שָׁלִישִׁים וּכְתָב מוֹעֵצוֹת

לָקַח עַם יָצָא מִצֹּעַן לוֹעֵט מַצּוֹת

לְפָאֵר לְרוֹמֵם לגדלו הָעֵיצוֹת


רְצוּצִים יֵחָפְשׁוּ מִתְרַפִּים בְּעֵת אָבִיב לְקַדְּשִׁי

רוֹמְמַנִי דִּלַּנִי חֲמִשִּׁים מִמַּכְאוֹת לְחָבְשִׁי

רְסִיסֵי טַל הִטְלִיל וְהִתְלִיל מ[..]יכת אנושי

רֶקַח עָסִיס הִשְׁקַנִי בַּחֹדֶשׁ הַשְּׁלִישִׁי


ובכן ולך תעלה


ח

אוֹתוֹת אֲשֶׁר שִׁלַּשְׁתָּ / בַּאֲשֶׁר בֵּאַרְתָ וּפִלַּסְתָּ / גַּם גְדוֹלוֹת חִלַּשְׁתָּ

דּוֹק וָחֶלֶד הִשְׁלַשְׁתָּ / הֵכַנְתָּ וְהֵיטַבְתָּ וְהֶחְקַרְתָּ / וְרֹאשׁ אוֹתִיּוֹת שִׁלַּשְׁתָּ

זָכוֹר וְשָׁמוֹר אִחַלְתָּ / חוֹק מְשֻׁלָּשׁ […] / טִיעַת שַׁעֲשׁוּעִים פֵּרַשְׁתָּ

יַחַד בְּרָכָה הִשְׁלַשְׁתָּ / כְּמוֹ שְׁלֹשָׁה הִקְדַּשְׁתָּ / לִשְׁלֹשָׁה אוֹתוֹ הִלְמַדְתָּ

מֵאוֹתִיּוֹת מָנְעָם פֵּרַשְׁתָּ / [נ..] נמר הִפְלַשְׁתָּ / סוֹד זֶה הִשְׁלַשְׁתָּ

עֲבוֹדָה וּתְעוּדָה פִּלַּשְׁתָּ / פְּאֵר שלוש [- - - - - - - - - - - - - -]
