וַיֵּרֶד אוֹמֵן

דבירין


וַיֵּרֶד אוֹמֵן מְחוֹלַת מַחֲנַיִם / וַיּוֹרֶד בָּרָה מְאִירַת עֵינַיִם

וַיֵּרֶד גִּזְבָּר הַמֻּנְעָם / וַיּוֹרֶד דְּרָכֶיהָ דַרְכֵי נֹעַם

וַיֵּרֶד הַמְלַמֵּד מַתְּנַת נַחֲלִיאֵל / וַיּוֹרֶד וַיְלַמְּדָהּ אֶת בְּנֵי יִשְׂרָאֵל

וַיֵּרֶד זַךְ בְּיִשּׁוּבִים / וַיּוֹרֶד חֻקִּים וּמִצְווֹת טוֹבִים

וַיֵּרֶד טוֹבִיָּה מִמְּעוֹנִים / וַיּוֹרֶד יְקָרָה מִפְּנִינִים

וַיֵּרֶד כָּלוּל בְּרַב חֵן / וַיּוֹרֶד לִוְיַת חֵן


לִוְיַת חֵן / הָשְׁמָעוּ כְמֵהִים

סוֹד הַדִּבֵּר כְּיָצָא רָעֲדוּ נְגוֹהִים

וַיְדַבֵּר אֱלֹהִים


וַיְדַבֵּר מֶלֶךְ חַיּוֹת וּשְׂרִָפִים / וַיָּנוּעוּ אַמּוֹת הַסִּפִּים

וַיְדַבֵּר סִדְרֵי סוֹד לְבַעֲלֵי חוֹק / וַיַּעֲמִידֶהָ לְיַעֲקֹב לְחוֹק

וַיְדַבֵּר פִּקּוּדֵי אֵל / וַיְצַוֵּם אֶל בְּנֵי יִשְׂרָאֵל

וַיְדַבֵּר קוֹלוֹ עַל הַמָּיִם / וַיֵּרָאוּ אֲפִיקֵי מָיִם

וַיְדַבֵּר שַׁדַּי אֶל עַם מָזְהָר / וַיִּתְיַצְּבוּ בְּתַחְתִּית הָהָר


הָהָר / בָּעַר כְּנִגְלֵיתִי לְהַגְבִּיהֶךָ

לַהֲאִירְךָ וּלְהַגִּיהֶךָ

אָנֹכִי יְיָ אֱלֹהֶיךָ


אָנֹכִי אָרַגְתִּי שְׁמֵי עֶרֶץ / אָנֹכִי בּוֹרֵא קְצוֹת הָאָרֶץ

אָנֹכִי גִלִּיתִי כְבוֹדִי לֶעָנָיו / אָנֹכִי דִּבַּרְתִּי דָּבָר דָּבוּר עַל אָפְנָיו

אָנֹכִי הוּא עוֹלָם מָלֵא כְבוֹדִי / אָנֹכִי וְאֵין אֱלֹהִים עִמָּדִי

אָנֹכִי זָקַפְתִּי עַמּוּדֵי עוֹלָם / אָנֹכִי חַי וְקַיָּם לְעוֹלָם

אָנֹכִי טִפַּחְתִּי דוֹק לְטַכְּסָה / אָנֹכִי יוֹצֵר הָאָרֶץ וְעוֹשָׂהּ

אָנֹכִי כִּלַּלְתִּי כֵּס מָרוֹם מֵרִאשׁוֹן / אָנֹכִי לִי תִּכְרַע כָּל בֶּרֶךְ תִּשָּׁבַע כָּל לָשׁוֹן


לָשׁוֹן / מְמוֹגֵג הוֹדַעְתִּיךָ בִדְבָרִים

וְהִזְהַרְתִּיךָ שֶׁלֹּא לַעֲבֹד זָרִים

לֹא יִהְיֶה לְךָ אֱלֹהִים אַחֵרִים


לֹא יִהְיֶה לְךָ מַעֲשֵׂה אֱלִילִים / נִשָּׂאִים עַל כָּתֵף וְנִסְבָּלִים

לֹא יִהְיֶה לְךָ סֵמֶל בְּרַהַב / עֲצַבֵּיהֶם כֶסֶף וְזָהָב

לֹא יִהְיֶה לְךָ פֶסֶל מָכְלָם / צוֹפָיו עִוְרִים כֻּלָּם

לֹא יִהְיֶה לְךָ קמלות תֹּרֶף / רְתוּקוֹת כֶּסֶף צוֹרֵף

לֹא יִהְיֶה לְךָ שְׁאֵרִית תּוֹעֵבָה / תַבְנִית זָכָר אוֹ נְקֵבָה


זָכָר וּנְקֵבָה / בָּרָא אֵל רָם וְנִשָּׂא

שְׁמוֹר פִּקּוּדָיו וּבַל תְּהִי לִמְשִׁסָּה

וּשְׁמוֹ לַשָּׁוְא לֹא תִשָּׂא


לֹא תִשָּׂא לַשָּׁוְא שֵׁם אַדִּיר הַיּוֹדֵעַ וָעֵד / הַנִּזְכָּר עַל הַבִּנְיָן וְהוּא רוֹעֵד

לֹא תִשָּׂא לַשָּׁוְא שֵׁם גִּבּוֹר הַמֻּפְלָא / הַנִּזְכָּר עַל הַדֶּבֶר וְהוּא כָלֶה

לֹא תִשָּׂא לַשָּׁוְא שֵׁם הַדִּבּוּר הַמְרֻצֶּה / הַנִּזְכָּר עַל הַוָּלָד וְהוּא יוֹצֵא

לֹא תִשָּׂא לַשָּׁוְא שֵׁם זֹכֵר חֹק לֶדֶת / הַנִּזְכָּר עַל הֶחָיָה וְהִיא יוֹלֶדֶת

לֹא תִשָּׂא לַשָּׁוְא שֵׁם גְּמוּל נִפְרָע / הַנִּזְכָּר עַל הַיָּם וְהוּא נִבְקָע

לֹא תִשָּׂא לַשָּׁוְא שֵׁם כְּלִיל נֵזֶר / הַנִּזְכָּר עַל הַלַּיִשׁ וְהוּא חוֹזֵר


חוֹזֵר / וְדוֹבֵר וְגוֹעֶה בְרִגּוּשׁוֹ

לְהַשְׁמִיעָךְ בְּפֵירוּשׁוֹ

זָכוֹר אֶת יוֹם הַשַּׁבָּת לְקַדְּשׁוֹ


זָכוֹר מְקֻדָּשׁ יוֹם מִיָּמִים בְּעֶלֶץ / נָחָה שָׁקְטָה כָּל הָאָרֶץ

זָכוֹר סְעוּדוֹת מְשֻׁלָּשׁוֹת בַּשַּׁבָּת / עַל כֵּן בֵּרַךְ יְיָ אֶת יוֹם הַשַּׁבָּת

זָכוֹר פִּנּוּק טוֹבוֹת רַבּוֹת / צְפוּנֵי בֵין הֲדַסֵּי יְשִׁיבוֹת

זָכוֹר קִדּוּשׁ יוֹם הַשַּׁבָּת / רְאוּ כִּי יְיָ נָתַן לָכֶם הַשַּׁבָּת

זָכוֹר שִׁמּוּר שַׁבָּת בְּחִבּוּבֶיךָ / וְהִתְעַנֵּג עַל יְיָ וְיִתֵּן לְךָ מִשְׁאֲלוֹת לִבֶּךָ


לִבֶּךָ / שִׁית לִמְקִימֶיךָ

לְמַעַן יַאֲרִיכוּן יָמֶיךָ

כַּבֵּד אֶת אָבִיךָ וְאֶת אִמֶּךָ


כַּבֵּד אוֹמְנֶיךָ מִלֵּידָה / בְּתוֹךְ קָהָל וְעֵדָה

כַּבֵּד גְּדוֹלֶיךָ לִסְמֹךְ / דִּבְרֵיהֶם בְּלִבְּךָ תִתְמֹךְ

כַּבֵּד הוֹרֶיךָ מִמַּטְעַמֶּיךָ / וְאַל תִּטֹּשׁ תּוֹרַת אִמֶּיךָ

כַּבֵּד זְקֵנֶיךָ וְתֵרָפֵא / חַיִּים הֵם לְמוֹצְאֵיהֶם וּלְכָל בְּשָׂרוֹ מַרְפֵּא

כַּבֵּד טוֹעֲנֶיךָ בַחַיִּים / יִרְבּוּ יָמֶיךָ וְיוֹסִיפוּ לְךָ שְׁנוֹת חַיִּים

כַּבֵּד כָּרָאוּי אָבִיךָ וְאִמֶּךָ / לְמַעַן יַאֲרִיכוּן יָמֶיךָ


יָמֶיךָ / יַאֲרִיךְ וּתְצֻחְצַח

אִם תִּשְׁמוֹר דִּבְרֵי צַח

הִזָּהֵר בְּנִי לֹא תִרְצַח


לֹא תִרְצַח מְרֻקָּם בִּדְמוּת נֶפֶשׁ / נֶפֶשׁ תַּחַת נֶפֶשׁ

לֹא תִרְצַח סְבוּאֵי יַיִן / עַיִן תַּחַת עַיִן

לֹא תִרְצַח פִּרְצַת בָּשָׂר וָדָם / צֶלֶם אֱלֹהִים עָשָׂה אֶת הָאָדָם

לֹא תִרְצַח קָדְקֹד קוֹמַת / רוֹצֵחַ מוֹת יוּמַת

לֹא תִרְצַח שְׁפִיכוּת דָּם / תֵּדַע כִּי דַם הָאָדָם בָּאָדָם


דַּם הָאָדָם בָּאָדָם / יִשָּׁפֵךְ בְּאַף

חֲדַל לְךָ בְנִי מִזַּעַף

וְהִזָּהֵר לֹא תִנְאַף


לֹא תִנְאַף אוֹרֶבֶת בְּכָל מַעְגָּלֶיהָ / בְּבֵיתָהּ לֹא יִשְׁכְּנוּ רַגְלֶיהָ

לֹא תִנְאַף גִּלָּת עֶרְוָתָהּ / דַּרְכֵי שְׁאוֹל בֵּיתָהּ

לֹא תִנְאַף הַהִפִּילָה חֲלָלִים בְּחוֹבָהּ / וְחוֹטֵא יִלָּכֶד בָּהּ

לֹא תִנְאַף זוֹנָה וּמְחַכָּה / חָלָק מִשֶּׁמֶן חִכָּהּ

לֹא תִנְאַף טוֹעָה וּמְזַנָּה / יָדֶיהָ אֲסוּרוֹת טוֹב יִמָּלֵט מִמֶּנָּה

לֹא תִנְאַף כְּכַלְבָּה נַשְׁכָנִית / לְבַל תִּדָּקֵר בַּחֲנִית


בַּחֲנִית / נָא אַל תִזְנֹב

וּמִטּוּב אַל תִּשְׂבַּע וְתָנוּב

אִם תִּזָּהֵר לֹא תִגְנוֹב


לֹא תִגְנֹב מְאוּמָה וְתֵלֵךְ עִם גַּנָּבִים / נְקִי כַפַּיִם אוֹרְבִים

לֹא תִגְנֹב [ס…….] לרחשו / עִם גַּנָּב חוֹלֵק שׂוֹנֵא נַפְשׁוֹ

לֹא תִגְנֹב פֶּן תִּנָּקֵשׁ / צַעַד הַיָּשָׁר לֹא תְעַקֵּשׁ

לֹא תִגְנֹב קִנְיָן וְתִקָּרֵא גַנָּב / רְחַק מִבָּאָה אֶל בֵּית הַגַּנָּב

לֹא תִגְנֹב שָׁוֶה חֲדוֹלָה / תִּרְחַק מִגַּנָּבִים בַּלַּיְלָה


לַיְלָה / וְיוֹם מְעִידִים שֶׁקֶר

רְחַק מֵהֶם עֶרֶב וּבֹקֶר

לֹא תַעֲנֶה בְרֵעֲךָ עֵד שֶׁקֶר


לֹא תַעֲנֶה אַתָּה עֵדוּת שֶׁקֶר / בִּלְשׁוֹנְךָ לֹא תִפְצֶה דֶקֶר

לֹא תַעֲנֶה גְּנוּת נָכוֹן / דּוֹבֵר שְׁקָרִים לֹא יִכּוֹן

לֹא תַעֲנֶה הוֹדָיַת שֶׁקֶר / וְלֹא תִשָּׁבְעוּ בִשְׁמִי לַשֶּׁקֶר

לֹא תַעֲנֶה זְכִירַת עֶרְיָה / חֶלְקַת לָשׁוֹן נָכְרִיָּה

לֹא תַעֲנֶה טָעוּת שֶׁקֶר / יָפִיחַ כְּזָבִים עֵד שָׁקֶר

לֹא תַעֲנֶה כַּאֲשֶׁר פָּקְרוּ / לֹא תְכַחֲשׁוּ וְלֹא תְשַׁקְּרוּ


לֹא תְשַׁקְּרוּ / שָׁוְא לִלְמֹד

וְעַל דַּם רֵעַ לֹא תַעֲמֹד

וְהִזָּהֵר לֹא תַחְמֹד


לֹא תַחְמֹד מַחְמַד עַיִן / נִבְהָל לַהוֹן אִישׁ רַע עָיִן

לֹא תַחְמֹד שְׂדֵה אִישׁ עָצֵל / עֲשֵׂה זֹאת אֵפוֹא בְנִי וְהִנָּצֵל

לֹא תַחְמֹד פּוֹעֵל כְּבוֹדוֹ / צְרוֹר הַכֶּסֶף לָקַח בְּיָדוֹ

לֹא תַחְמֹד קִנְיַן קְרוֹבֶיךָ / רוּץ וְאַל תַּחְמֹד יָפְיָהּ בִּלְבָבֶךָ

לֹא תַחְמֹד שֶׁלַּחֲבֵירֶיךָ / תֹּאכַל יְגִיעַ כַּפֶּיךָ אַשְׁרֶיךָ


אַשְׁרֶיךָ / גּוֹי מְשֻׁבָּח מִכָּל הַנִּבְרָאִים

וּמְקַבֵּל הַנִּשְׁמָעִים וְהַנִּקְרָאִים

וְכָל הָעָם רוֹאִים


אֵל קָדוֹשׁ יוֹשֵׁב תְּהִלּוֹת

בְּדַבְּרוֹ אָנֹכִי רָעֲשׁוּ חֲיָלוֹת


גִּבּוֹר וְכַסּוֹפֵר מְלַמֵּד תַּחְבּוּלוֹת

דִּבֵּר לֹא יִהְיֶה לְךָ אֱלִילוֹת


וְכָל הָעָם רוֹאִים אֶת הַקּוֹלוֹת


הוֹד קוֹלוֹ הִשְׁמִיעַ מִמַּעֲלוֹת

וְהוֹדִיעַ שֵׁם לֹא תִשָּׂא בְּעַוְּלוֹת


זִמֵּן בְּנַחַת קוֹל הֲמֻלּוֹת

חִכֵּם זָכוֹר יוֹם גִּילוֹת

וְכָל הָעָם רוֹאִים אֶת הַקּוֹלוֹת


טֶרֶם יֻלָּדוּ מִפְעֲלוֹת

יִדַּע כַּבֵּד מְיַחֲלֶיךָ בְחִילוֹת

וְכָל הָעָם רוֹאִים אֶת הַקּוֹלוֹת


כּוֹבֵשׁ כַּעַס בְּחֶמְלוֹת

לִמֵּד לֹא תִרְצַח לְכַלּוֹת

וְכָל הָעָם רוֹאִים אֶת הַקּוֹלוֹת


מְשׁוֹטְטוֹת עֵינָיו בְּכָל מְסִלּוֹת

נָם הִזָּהֵר לֹא תִנְאַף עֵינֶיהָ תְלוּלוֹת

וְכָל הָעָם רוֹאִים אֶת הַקּוֹלוֹת


סוֹכֵךְ עַל נְפָשׁוֹת לֹא נִגְנָבוֹת וְכָלוֹת

עָן לֹא תִגְנֹב בְּאָלוֹת

וְכָל הָעָם רוֹאִים אֶת הַקּוֹלוֹת


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
