וַיֵּרֶד אֲבִי

וַיֵּרֶד / אֲבִי סוֹכֹה מֻקְרָן עוֹר פָּנִים

מֵרוּם חַגְוֵי מְעוֹנֵי שִׁנְאַנִּים

בְּחֶקֶק לְהוֹרוֹת חִידוֹתֶיהָ לַשּׁוֹשַׁנִּים

יְקָרָה הִיא מִפְּנִינִים


וְיְדַבֵּר / גֵּאֶה בְּקוֹל נַעֲמָה

וּמִקּוֹלוֹ חָלוּ אָדָם וּבְהֵמָה

דְּבָרוֹ יְפוֹצֵץ יְסוֹדֵי אֲדָמָה

תּוֹרַת יְיָ תְּמִימָה


אָנֹכִי / הִשְׁמִיעַ עֲדָרָיו לְבַל יֻמְעֲדוּן

וְעֶשְׂרִים וְאַרְבַּע מִיל לְאָחוֹר הָרְתָּעוּ וְהָאָחְדוּן

וְעוֹל מַלְכוּתוֹ קִבְּלוּ וְיָשְׁרוֹ הֱעִידוּן

מַלְכֵי צְבָאוֹת יִדּוֹדוֹן יִדּוֹדוּן


לֹא יִהְיֶה לְךָ / זָנוּחַ לְשֵׁם שְׁעִיָּה

דְּבוֹק בְּמַטְרִיף וּמַסְפִּיק לַכֹּל מִחְיָה

חֶדְוַת מְאִירַת עֵינַיִם לְנֶפֶשׁ תְּחִיָּה

לִי עֵצָה וְתוּשִׁיָּה


לֹא תִשָּׂא / טָעוּת פֶּן תֵּרָשֵׁף בְּאֵשׁ בּוֹעֵרָה

נְצוֹר מִצְוָה נְתוּנָה מִשְּׁמֵי שִׁפְרָה

יֻצַּג גִּזְעֲךָ אִם תַּתְמִיד בְּאוֹרַח יְשָׁרָה

אֲנִי בִינָה לִי גְבוּרָה


זָכוֹר / כִּבּוּד מָבְחָר מִשִּׁבְעָה לְנוֹפֶשׁ

וְתֻרְגַּע בְּמַחְסֶה וְתֻצַּל מִטִּיט וָרֶפֶשׁ

לְהַשִּׂיג לְנוֹצְרֶיהָ מֵעֹבֶשׁ חֹפֶשׁ

תּוֹרַת יְיָ תְּמִימָה מְשִׁיבַת נֶפֶשׁ


כַּבֵּד / מִתְחַנְּנִים בַּעַדְךָ מְצוֹא תְקוּמָה

וְהִתְכַּבֵּד נֶצַח פְּנֵי חוֹקֵר תַּעֲלוּמָה

נוֹצָה וּמָשׁוּךְ וְרָבוּץ אֲמִיצְךָ בְנֶחָמָה

לָתֵת לִפְתָאִים עָרְמָה


לֹא תִרְצַח / סָאוּר בְּצֶלֶם רִקְמָתִי

פֶּן תִּזְעַק מֵרֶצַח וְאָעִיר חֲמָתִי

עַד לֹא תֶחֱנַף וַאֲשַׁקֵּר בֶּאֱמוּנָתִי

עֵדוּת יְיָ נֶאֱמָנָה מַחְכִּימַת פֶּתִי


לֹא תִנְאַף / פְּרוּצָה בְּכָל נִבּוּל מַעַד

הַמּוֹבֶלֶת כַּשּׁוֹר לַטֶּבַח שָׂטֵי עִמָּהּ בְּצַעַד

צֶדֶק וַעֲנָוָה תְּבַע וְתִוָּשַׁע בְּסַעַד

יִרְאַת יְיָ טְהוֹרָה עוֹמֶדֶת לָעַד


לֹא תִגְנֹב / קַרְקָעִים וּמִטַּלְטְלִים בְּחַטָּאִים לְהִתְעָרֵב

מְדוֹר חֶמְדָּתָךְ מִמְּכוֹנוֹ פֶּן יִתְחָרֵב

רְחָבָה מִנִּי יָם חֲשׁוֹק וְלַשְּׁכִינָה תִתְקָרֵב

הַנֶּחֱמָדִים מִזָּהָב וּמִפַּז רָב


לֹא תַעֲנֶה / שָׁוְא אָלוֹת אֲשֶׁר יְסוֹד מַרְשִׁיפִים

אֱמוּנָה אֹמֶן […. ….] ברכבות מְטֻנָּפִים

שַׁחַר וָנֶשֶׁף בַּקֵּר אִפְנוּנֵי אִלּוּפִים

מְתוּקִים מִדְּבַשׁ וְנֹפֶת צוּפִים


לֹא תַחְמֹד / תְּלָאוֹת לְהִתְעוֹלֵל עֲלִילוֹת

פֶּן יְחַסֶּדְךָ שׁוֹמֵעַ בְּמַהֲלֻמֵּי קְלָלוֹת

תִּשְׂמַח בְּחֶלְקָךְ מַתַּן מְנַהֵל עָלוֹת

וְנָבוֹן יִקְנֶה תַחְבּוּלוֹת


את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
