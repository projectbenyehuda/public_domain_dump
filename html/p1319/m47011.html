עת הזמיר

הַבִּיטוּ רְאוּ אֶת־הָאֲדָמָה מִנֶּגֶד

מִתְחַדְּשָׁהּ מְשַׁנָּהּ לְבושָׁה מַעֲדָהּ בֶגֶד

כִּבְתוּלָה תִּתְיַפֶּה וְתַעֲדֶה כֵּלֶיהָ

תַּחַת שֶׁלֶג וְקִפָּאוֹן כְּפוֹר גַּם קֶרַח

עֵשֶׂב וְיָרַק דֶּשֶׁא צִיצָה גַּם פֶּרַח

לָבְשָׁה רִקְמָה וַתֵּרֶב רִקּוּחֶיהָ:


תַּחַת מֶגְרְפָתוֹ זֶרַע יִפָּתַח

יוֹצִיא אֶת־רֹאשׁוֹ הִנֵּהוּ צוֹמֵחַ

וְכֵן כָּל־שׁוֹכְנֵי־עָפָר יְעֹרְרוּ יָקִיצוּ

יִחְיוּ עֲצֵי יַעַר יָגִיל כֶּרֶם זָיִת

נִטְעֵי שַׁעֲשׁוּעַ עַד שָׁמִיר וָשָׁיִת

כֻּלָּם יִלְבְּשׁוּ הָדָר עֹנֶג יָצִיצוּ:


שָׁם יִפְרְחוּ דּוּדָאִים בְּעַנְוָה בַּסֵּתֶר

נֶגְדָּם הַחֲבַצֶּלֶת תָּרִים הַכֶּתֶר

אֶל־כָּל־לֵב נֹעַם וְתַאֲוָה לָעֵינַיִם

שָׁם הַשּׁוֹשַׁנָּה בְּפִטּוּרֵי עָלֶיהָ

תִּרְוֶה שִׁכְבַת הַטַּל יָלִין בְּחֵיקָהָּ

וְרוּחַ צַח בְּקִנְאָתוֹ יִשַּׁק לָהּ שְׂפָתָיִם:


מִי זֶה בָּא מִקֶּדֶם כְּמַרְאֵה הַזֹּהַר

לְפָנָיו יָנוּס צֵל וְחֹשֶׁךְ שָׁת לָטֹהַר

זֶה אָדֹם לְבוּשׁוֹ כְּאַרְגָּמָן אַדֶּרֶת

יִפְרֹשׂ כַּנְפֵי זָהָב עַל־גִּבְעוֹת נֶצַח

עַל הָהָר הֶחָלָק עַל גּוֹבַהּ מֵצַח

וְנוֹגַהּ אֵשׁ לֶהָבָה וְרַבָּהּ תִּפְאֶרֶת:


הַבֹּקֶר הֵאִיר זוֹרֵחַ הַשֶּׁמֶשׁ

וּמִפָּנָיו חָלְפוּ מְשׁוֹאָה וָאֶמֶשׁ

יָשִׂישׁ כְּגִבּוֹר לְהָפִיץ רִשְׁפֵי קַשְּׁתֵּהוּ

אֲבָל לֹא לְהָמִית חִצָּיו לְהַחֲיוֹת יִפְעָלוּ

אִם מִקֶּרַח לַיְלָה שְׁדֵמוֹת אֻמְלָלוּ

יִשְׁלַח תְּעָלָה לָהֶם בִּבְרַק חִיצֵּיהוּ:


הִנֵּה עֹבֵד אֲדָמָה יַשְׁכִּים לְפָעֳלֵהוּ

לְהוֹצִיא תְּבוּאַת אֶרֶץ בְּכֹחַ שׁוֹרֵהוּ

בְּקָרָיו בְּמוֹסְרוֹת הָעוֹל יֶאֶסְרֵם יָחַד

בְּצֶקֶת עָפָר לְמוּצָק בְּאֵתוֹ יְפַלֵּחַ

וּרְגָבִים יְרַכֵּךְ יְשַׁדֵּר וִיפַתֵּחַ

יִשְׂמַח בַּעֲבֹדָתוֹ יִשְׂחַק לַפָּחַד:


וּמֹשֵׁךְ הַזֶּרַע יִפְרֹשׂ מִטְפַּחַת

יַבְדִּילוֹ לְמִינוֹ יְשִׂימוֹ בַּצַּלַּחַת

יְפִיצוֹ עֲלֵי אֲדָמָה וִישַׁוֶּה פָּנֶיהָ

חָנְנֵהוּ נָא אָדוֹן וְשִׂים עָלָיו עַיִן

בַּל יְהִי אֶפֶס עֲמָלוֹ יְגִיעוֹ לָאָיִן

תְּעַטֵּר שְׁנַת טֹבָתְךָ תְּבָרֵךְ צְמָחֶיהָ:


רוּחַ יֶשַׁע הָפִיחִי עַל־הַחֵלֶק

בַּל יְהִי לְמַאֲכַל אַרְבֶּה חָסִיל וָיֶלֶק

הָגִינִי גַּם אַתְּ עָלָיו שֶׁמֶשׁ צֶדֶק

וְגֶשֶׁם נְדָבוֹת עָלָיו נְשִֹאִים תָּנִיפוּ

וְאֶת־טַלְּכֶם שָׁמַיִם עָלָיו תַּרְעִיפוּ

בַּל תַּעֲלוּ עָלָיו קִמְשׁוֹנִים וָחֶדֶק:


מֶה קוֹל הַזֶּה בְּאָזְנָי שְׁרִיקוֹת הָעֵדֶר

יְרַקְּדוּ יִקְפְּצוּ עַל־מְסוּכָהּ יְדַלְּגוּ גֶּדֶר

פָּרִים וּכְבָשִׂים לַחָפְשִׁי יָצָאוּ

כָּל־הַלַּיְלָה תַּחַת דֶּלֶת וּבְרִיחַ

אֻסְּרוּ עַל־מַסְגֵּר עֻלְּפוּ שָׁכְנוּ צְחִיחַ

עַתָּה יִשְׁאֲפוּ רוּחַ מֶרְחָב מָצָאוּ:


שָׁם לִבְלִי חֹק הַשּׁוֹר אֶת־פִּיו פּוֹתֵחַ

יִגְעֶה בְקוֹל בּוֹ יַגִּיד כִּי הוּא שָׂמֵחַ

יַחְפֹּר בָּעֵמֶק שָׂשׂ וְרוֹקֵעַ בְּרֶגֶל

אַחֲרָיו פָּרָתוֹ יִמְשֹׁךְ תִּצְעַד בְּנַחַת

מִכֹּבֶד שָׁדֶיהָ כְּגֶפֶן סֹרַחַת

וַתַּבֵּט מֵאַחֲרֶיהָ אֵי הָעֵגֶל:


הִנֵּה הוּא נִשְׁאָר בֵּין כֶּשֶׂב וָאָיִל

שָׁם יַעֲרֹךְ מִלְחֶמֶת גַּם יַעֲשֶׂה חָיִל

נֶגֶד כֶּלֶב עַז נֶפֶשׁ וְאַמִּיץ כֹּחַ

סוֹבֵב סוֹבֵב יַקִּיף בְּדֹמֶן וָרֶפֶשׁ

אַחֲרָיו יָרוּץ הַכֶּלֶב בִּשְׁאָט נֶפֶשׁ

יֶחְרַץ לְשֹׁנוֹ לֹא יַחְשֹׂךְ מִנְּכוֹחַ:


לְזֹאת יִקְצֹף הָרֹעֶה יֶחְרֶה הַנַּעַר

וַיִּקַּח אֶת־מַקְלוֹ חֲצָבוֹ מִיַּעַר

וּכְשֵׁבֶט מֹשֵׁל יְנִיפוֹ יְצַו מַרְגּוֹעַ

וְאֶבֶן מִיַּלְקוּטוֹ יְמַהֵר לְקַלֵּעַ

אַחֲרֵי הַכֶּלֶב אֵינֶנּוּ שֹׁמֵעַ

וְהִנֵּה שֶׁקֶט אֵין פֹּצֶה פֶה לָנוֹעַ:


וּלְאַט לְאַט בְּנֵי צֹאן מֵרִבְצָם יִרְחָקוּ

יִתְגּוֹדְדוּ יַחַד אִישׁ אָחִיו יִדְחָקוּ

לְכַר נִרְחָב מֵרָחוֹק מְגַמַּת פְּנֵימוֹ

וְקוֹל חָלִיל יַנְעִים זֶמֶר יֵיטִיב דַּעַת

וּכְמוֹהוּ מְאֹד יֶעֱרַב לָאֹזֶן שֹׁמַעַת

מַשַּׁק פַּעֲמֵיהֶם וְהֶמְיַת פַּעֲמֹנֵימוֹ:


וְאֶל־מִרְעֶה טוֹב וְשָׁמֵן הַצֹּאן נָפוֹצוּ

וְעַתּוּדִים מֵהַר אֶל־גִּבְעָה יִקְפֹּצוּ

יִדְרְשׁוּ אַחֲרֵי יֶרֶק יָתוּרוּ לְאֹכֶל

רָאשֵׁיהֶם לָאָרֶץ שְׁחוֹחַ יְהַלֵּכוּ

לָמֹּץ רֵיחַ שָׂדֶהָ דֶּשֶׁא יְלַחֵכוּ

אֲבָל יִזְּלוּ בְשָׂמָיו מִכֹּל אַבְקַת רֹכֵל:


שְׁאוֹן נָחַל נוֹבֵעַ כְּהָקֵיר מֵיְּמֵהוּ

דְּמָמָה וְקוֹל עָרֵב כִּי יֶהֱמוּ גַלֵּיהוּ

אֵיךְ יִשְׂמַח לֵב יֶעֱרַב לְנֶפֶשׁ בֹּדֶדֶת

יִשְּׂאוּ בְתֹף וְכִנּוֹר בְּכָל־מִרְזַח סָרוּחַ

אֹהֲבֵי הֲמֻלָּה וָרַעַשׁ וְרֹדְפֵי רוּחַ

וְרִנְנַת נַחַל שֹׁטֵף נַפְשִׁי חֹמֶדֶת:


שָׁם מִבֵּין הֶהָרִים מִפַּכִּים הַמַּיִם

עַל־סְעִיף סֶלַע יִפְּלוּ אַמּוֹת מָאתָיִם

לִתְהוֹם רַבָּה מִמֶּנּוּ יִתְגּוֹלָלוּ

הֵם כְּסִיר מַרְתִּיחַ יִתְגָּעֲשׁוּ וְיִשְׁאָיוּ

לִרְסִיסֵי פְּנִינִים לְמִיץ חָלָב יִדְמָיוּ

יִשְׁתַּקְשְׁקוּ וּכְמוֹ רֶכֶב יִתְהֹלָלוּ:


וְתַחַת לְרַגְלֵי הָרֵי עַד שָׁם יִגָּרוּ

בְּאַגַּן הַסַּהַר הַמַּיִם יִנְהָרוּ

מֶנּוּ תֹּצְאוֹת נַחַל יָזוּב וְיָגִיחַ

מֵי עֲקַלָּתוֹן יִלָּפְתוּ אֹרְחוֹתָמוֹ

יְמוֹגְגוּ עֲמָקִים וְצִמְחֵיהֶם כֻּלָמוֹ

יְרַווּ נַחַת כָּל־עֵץ יַשְׁקוּ כָּל־שִׂיחַ:


עוֹדֶנָּה מִתֵּימָן רוּחַ נוֹשֶׁפֶת

וְהִנֵּה עָב קְטַנָּה עֲלֵי חֶרֶס חֹפֶפֶת

עַד כֹּה וְשָׁמַיִם עָבִים הִתְקַדָּרוּ

וְהִנֵּה דְּמָמָה מֵרוּחַ וּמִסַּעַר

אֵין קוֹל עָלֶהָ נִדָּף יִשְׁתְּקוּ עֲצֵי יַעַר

שֶׁקֶט עַל־פְּנֵי הַנַּחַל גַּלִּים סָרוּ:


וְעֵינֵיהֶם שָׁמַיְמָה שׁוֹר עֵז וְכֶשֶׂב

וְהַכֶּלֶב כַּבָּקָר הוּא יֹאכַל עֵשֶׂב

וְעוֹף הַמַּיִם אֶבְרָה וְנוֹצָה יִטְבָּעוּ

יְיַחֲלוּ לַמָּטָר הַצְּמָחִים כֻּלָמוֹ

גַּם הֶהָרִים וּגְבָעוֹת קִוּוּ לָמוֹ

יִשְׁאֲלוּ לַמַּיִם כִּי יִשְׁתּוּ וְיִשְׂבָּעוּ:


פִּתְאֹם הַנְּשִׂיאִים אֶת־אוֹצָרָם יָרִיקוּ

וּמַרְפֵּא לְכָל־בָּשָׂר חַיִּים יָפִיקוּ

עוֹד הַגֶּשֶׁם דַּק כִּכְפוֹר עַל־הָאָרֶץ

אָכֵן עַתָּה אֲגָלָיו יְרְבְּיו וְיִגְדָּלוּ

וַיִּרְבּוּ הַמַּיִם רְבִיבִים נָפָלוּ

יִפְרְצוּ בָאָרֶץ פֶּרֶץ עַלפְּנֵי פָרֶץ:


מָה־זֹאת כְּמַרְאֵה הַחַשְׁמַל מַרְבֶּה צֶבַע

יַרְקִיעַ בַּשְּׁחָקִּים מַרְאֵהוּ שֶׁבַע

הֲלֹּא תֵּדָעֵהוּ הַקֶּשֶׁת הִנֶּנּוּ

נִטְפֵי מַיִם בֶּעָנָן כִּי נִשְׁאָרוּ

אִם קַרְנֵי הַשֶּׁמֶשׁ עֲלֵיהֶם עָבָרוּ

יֵרָאֶה הַקֶּשֶׁת יָעוּף וְאֵינֶנּוּ:


שָׁם מִבֵּין עֳפָאִים אֶת־קוֹלָם יִתֵּנוּ

צִפֳּרִים מָצְאוּ בַיִת וְשָׁם יְקַנֵּנוּ

דְּרוֹר נֹדֵד כָּנָף עֲלֵי עָנָף יְנִיפֶנּוּ

יְצַפְצֵף וְיִקְרָא אֶל־רְעוּתוֹ בְּצֵל רֹבֶצֶת

יָבִיא אֹכֶל לָהּ וְהִיא מִפִּיו מֹצֶצֶת

וּלְהָבִיא שֵׁנִית חִישׁ יָעוּף וְאֵינֶנּוּ:


וְהַסְּיס בֹּנֶה מַעֲלוֹתָיו עַל־רֹאשׁ תֹּרֶן

יָבִיא חֵמֶר וְטָח עִם פֶּה וּבְצִפֹּרֶן

בְּרִירוֹ יְיַפֵּהוּ כְּבִמְחוּגָה יְתָאֲרֵהוּ

יָבִיא מֵעַל־פְּנֵי הַשָּׂדֶה טִיט וְדֹמֶן

יְחַזֵּק תִּפְאֶרֶת בֵּיתוֹ אֱמוּנָה אֹמֶן

לַעֲמֹד בִּסְעָרָה וְרוּחַ לֹא יְנִידֵהוּ:


וְעַל־שְׂפַת הַנַּחַל יְרַקֵּד הַצְּפַרְדֵּעַ

בֵּין חָרוּל יִנְהַק יִזְעַק וִישַׁוֵּעַ

בְּקוֹל הַסִּירִים תַּחַת הַסִּיר נָפוּחַ

יִלְטֹשׁ עֵינָיו יַטֶּה שְׁכֶם יָרִים עֹרֶף

וְלָכוֹף כְּאַגְמֹן רֹאשׁוֹ וּכְמִפְּנֵי חֹרֶף

יִתְעַוֵת כִּצְנֵפָה יִדְכֶּה יָשׁוּחַ:


כָּל־רֶמֶשׁ וְשֶׁרֶץ וְכָל־מַרְבֵּה רַגְלָיִם

כָּל־הֹלֵךְ עַל־גָּחוֹן אֲשֶׁר לוֹ כְנָפָיִם

יַחְתְּרוּ בְחוֹרֵי עָפָר יְנַתְּרוּ וְיָעוּפוּ

שְׁמָהּ יִדְאֶה הַזְּבוּב כִּמְטַחֲוֵי קֶשֶׁת

שְׁמָמִית תְתַפְּשׂוֹ בְּהַכִינָהּ לוֹ הָרֶשֶׁת

בְּבֹא הָעֲטַלֵּף עֲלֵיהֶם יַחְדָּו יָסוּפוּ:


וְהַנְּמָלָה תִּרְמֹשׂ מִבֹּקֶר עַד עָרֶב

לֶאֱגֹר מַאֲכָלָהּ לְהָכִין לְבֵיתָהּ טָרֶף

וּבֵצִים עֲזוּבוֹת בְּפִיהָ תִּשָׂאֵמוֹ

תְּבִיאֵמוֹ וְתַצִּילֵם מִשֶּׁטֶף פֶּלֶג

בֵּין כֵּפִים גָּרְנָהּ לֹא תִּירָא מִשֶּׁלֶג

בָּנְתָה בֵיתָהּ בְּדַעַת חֲדָרִים תְּמַלְּאֵמוֹ:


פֹּה נִצָּב הָעֶלֶם וְחֹשֵׁב מַחֲשֶׁבֶת

וּמִן עַרְבֵי נַחַל יָדוֹ קֹצֶבֶת

בְּסַכִּין מְלֻטָּשׁ חֲלִילִים מַעֲשֵׂה שַׁעֲשׁוּעַ

וִיפַצֵל אוֹתָם וְיִּקֹּב חֹר לִנְפּוֹחַ

יְרַנֵּן וְיָגִיל הָלֹךְ יֵלֵךְ וְשָׂמֹחַ

וַיָּנַח וַיֵּשֶׁב עַל הַמַּבּוּעַ:


וְיַבֵּט יֶלֶק צֶבַע רִקְמָתָיִם

וְחִישׁ יִדְלֹק אַחֲרֵיהוּ לְתָפְשׂוֹ בְּיָדָיִם

וְהוּא פָשַׁט וַיָּעֹף יְהַתֵּל בָּעֶלֶם

וַיִּזְרֹק אַחֲרָיו צְרוֹר עָפָר וָאֶבֶן

וּבְיָדוֹ לְהַכֹּתוֹ קָנֶה גַּם תֶּבֶן

אֲבָל בְּלִי רְאוֹת נִדַּח בַּחֲרִיצֵי הַתֶּלֶם:


וְתַחַת שׂוֹבֶךְ הָאֵלָה בְּצֵל עָלֶיהָ

שָׁם נַעֲרָה יְפַת תֹּאַר בְּנֹעַם מַרְאֶהָ

יֹשֶׁבֶת אֵצֶל דּוֹדָה וְסוֹד יַמְתִּיקוּ

מָחֲתָה מֵרַקָּתוֹ זֵעַת אַפָּיִם

וְהוּא עַל בִּרְכָּהּ יָנַח מֵעֲמַל כַּפָּיִם

צוּף תַּחַת לְשׁוֹנָם אֲמָרֵיהֶם הֶחֱלִיקוּ:


כָּכָה כֹּל יָגִיל בְּעֵת הַזָּמִיר הִגִּיעַ

תֵּבֵל וּמְלֹאָהּ שׂבַע שְׂמָחוֹת יַשְׂבִּיעַ

עוּרָה כִנּוֹרִי בַּל תִּכְלֶה לְזַמְּרֶנּוּ

אֲבָל מָלֵאתִי רוּחַ מִדַּבֵּר נִפְעַמְתִּי

מַשְּׂכִּיּוֹת לִבִּי רַבּוּ לָכֵן נֶאֱלַמְתִּי

יִדְמוּ בְנוֹת שִׁיר אִם רַעֲיוֹנִי יְהוֹדֶנּוּ:



את הטקסט[ים] לעיל הפיקו מתנדבי פרויקט בן־יהודה באינטרנט.
