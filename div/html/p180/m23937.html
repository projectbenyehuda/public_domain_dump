וַיְדַבֵּר א

וַיְדַבֵּר א א א א א תוּ

בְּקִרְבָתָם לִפְנֵי יְיָ וַיָּמוּתוּ


וַיֹּאמֶר נ נ נ נ נ רֶת

כִּי בֶּעָנָן אֵרָאֶה עַל הַכַּפֹּרֶת


בְּזֹאת י י י י י לָה

בְּפַר בֶּן בָּקָר לְחַטָּאת וְאַיִל לְעוֹלָה


כְּתֹנֶת שׁ שׁ שׁ שׁ שׁ שָׁם

וְרָחַץ בַּמַּיִם אֶת בְּשָׂרוֹ וּלְבֵשָׁם


וּמֵאֵת ל ל ל ל ל לָה

שְׁנֵי שְׂעִירִים עִזִּים לְחַטָּאת וְאַיִל אֶחָד לְעוֹלָה


וְהִקְרִיב מ מ מ מ מ תוֹ

וְכִפֶּר בַּעֲדוֹ וּבְעַד בֵּיתוֹ


וְלָקַח ה ה ה ה ה עֵד

וְהֶעֱמִיד אוֹתָם לִפְנֵי יְיָ פֶּתַח אֹהֶל מוֹעֵד


וְנָתַן ה ה ה ה ה זֵל

גּוֹרָל אֶחָד לַיְיָ וְגוֹרָל אֶחָד לַעֲזָאזֵל


וְהִקְרִיב ק ק ק ק ק ת

אֲשֶׁר עָלָה עָלָיו הַגּוֹרָל לַיְיָ וְעָשָׂהוּ חַטָּאת


וְהַשָּׂעִיר ט ט ט ט ט רָה

לְשַׁלַּח אוֹתוֹ לַעֲזָאזֵל הַמִּדְבָּרָה


וְהִקְרִיב נ נ נ נ נ לוֹ

וְשָׁחַט אֶת פַּר הַחַטָּאת אֲשֶׁר לוֹ


וְלָקַח בּוֹעֶרֶת בְּיָמִין בְּדַקָּה בִּשְׂמֹאל נֶעֱרֶכֶת

וְהֵבִיא מִבֵּית לַפָּרֹכֶת

וְנָתַן רָצוּף רָפוּד רְצָפִים [ר ] בִּנְעִימוּת

וְכִסָּה עֲנַן הַקְּטֹרֶת אֶת הַכַּפֹּרֶת אֲשֶׁר עַל הָעֵדוּת וְלֹא יָמוּת


וְלָקַח בָּלוּל בָּחוּן בְּתֶלֶף בְּצֶלֶף לְהַצְבִּיעוֹ

יַזֶּה שֶֶׁבַע פְּעָמִים מִן הַדָּם בְּאֶצְבָּעוֹ


וְשָׁחַט יָעוּד יַעַן יָכִין יָצוּק בְּמִשְׁמֶרֶת

וְהִזָּה אוֹתוֹ עַל הַכַּפֹּרֶת וְלִפְנֵי הַכַּפֹּרֶת


וְכִפֶּר יִקְרַת יִתְרַת יֹקֶשׁ יְחִידִים וְחַטָּאתָם

וְכֵן יַעֲשֶׂה לְאֹהֶל מוֹעֵד הַשּׁוֹכֵן אִתָּם בְּתוֹךְ טֻמְאוֹתָם


וְכָל אָדָם הַָגֵז הַיְצֵא הַחוּץ הָעֵת יוֹאֵל

וְכִפֶּר בַּעֲדוֹ וּבְעַד בֵּיתוֹ וּבְעַד כָּל קְהַל יִשְׂרָאֵל


וְיָצָא וְכִפֶּר וְעֵרַב וְלָקַח וְנָתַן הֶחָבִיב

עַל קַרְנוֹת הַמִּזְבֵּחַ סָבִיב


וְהִזָּה דֶּרֶךְ דָּתוֹ דָּם דָּלוּף בַּאֲרִיאֵל

וְטִהֲרוֹ וְקִדְּשׁוֹ מִטֻּמְאוֹת בְּנֵי יִשְׂרָאֵל


וְכִלָּה הֲלִיכוֹת הַשְּׁלֹשָׁה הַכֹּהֵן הַגָּדוֹל בְּאֶזְרָחַי

וְהִקְרִיב אֶת הַשָּׂעִיר הֶחָי


וְסָמַךְ גְּלִילָיו גְּמֹר גֹּעַל גְּאוּלִים בְּתִפְאָרָה

וְשִׁלַּח בְּיַד אִישׁ עִתִּי הַמִּדְבָּרָה


וְנָשָׂא בְרֹאשׁוֹ בָּשְׁנַת בְּחִירִים בְּקֶשֶׁר מְחֻבָּר

וְשִׁלַּח אֶת הַשָּׂעִיר בַּמִּדְבָּר


וּבָא יָדִיד יַקִּיר יַעַן יִפְשֹׁט יְקָרִים לְבֵשָׁם

בְּבֹאוֹ אֶל הַקֹּדֶשׁ וְהִנִּיחָם שָׁם


וְרָחַץ רָגִיל רִחוּשׁ רֶשֶׁם רְצוּיָה מְנֻתָּחָה

וְאֵת חֵלֶב הַחַטָּאת יַקְטִיר הַמִּזְבֵּחָה


וְהַמְּשַׁלֵּחַ וֻעַד וְכִבֵּס וְרָחַץ וְטֹהַר בַּמִּפְנֶה

וְאַחֲרֵי כֵן יָבוֹא אֶל הַמַּחֲנֶה


לְקוּחִים לְהָבִיא [ל ל ] כַנִּרְשָׁם

יוֹצִיא אֶל מִחוּץ לַמַּחֲנֶה וְשָׂרְפוּ בָאֵשׁ אֶת עוֹרוֹתָם וְאֶת בְּשָׂרָם וְאֶת פִּרְשָׁם.
