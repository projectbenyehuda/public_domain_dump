אֲפַכֶּה דְמָעַי

אֲפַכֶּה דְמָעַי / עַל רֹב פְּשָׁעַי / חֲמַרְמָרוּ מֵעַי / בְּיוֹם קָרְאִי בְּקוֹל

בְּחַסְדְּךָ אֶבְטָח / כִּי אַתָּה מִבְטָח / וְכַפַּי לְךָ אֶשְׁטַח / בְּעֵת אָרִים קוֹל

גֹּדֶל פִּשְׁעִי / הַעֲבֵר צוּר יִשְׁעִי / וּסְלַח רִשְׁעִי / בְּצָרְחִי בְּקוֹל

דַּלְתְּךָ פְּתַח נָא / לְבֵן בָּא לְךָ בְרִנָּה / וּבַשְּׂרֵהוּ מְעוֹנָה / חֲנַנְתִּיךָ בְּזַעַקְךָ בְקוֹל

הַקְשֵׁב תְּפִלָּה / שׁוֹכֵן מַעְלָה / וְהָסֵר תָּהֳלָה / מִמְּבַקְשֶׁיךָ בְקוֹל

וּכְתֹב לְחַיִּים / עַם נִבְחָר מֵאִיִּים / וְיֵדְעוּ גוֹיִים / כִּי חָשַׁקְתָּ בְקוֹל

זַכֵּה בְדִינֶךָ / מְחַלֵּה פָנֶיךָ / וְחֹן עַל בָּנֶיךָ / מַקְשִׁיבֵי קוֹל

חוֹקֵר עֲלוּמִים / זְכֹר בְּרִית עוֹלָמִים / לְעַם נִכְלָמִים / מִלְּהָרִים קוֹל

טַהֵר נְקִיָּה / בְּיַד צַר שְׁבוּיָה / וְהִיא הוֹמִיָּה / שׁוֹאֶגֶת לְךָ בְקוֹל

יֶעֱרַב שִׂיחָהּ / וְיִכּוֹן שִׁבְחָהּ / לְחוֹקֵר טוּחָהּ / בְּעֵת תִּתֶּן קוֹל

כְּבוֹדְךָ גַלֵּה / עַל עַם אֵלֶּה / וְהַפְלֵא פֶלֶא / יוֹם יִצְרַח בְּקוֹל

לְאָנָה אֶבְרַח / וְעִם מִי אֶאֱרָח / וּלְמִי אֶצְרָח / עַל כֵּן לְךָ אֶתֶּן קוֹל

מוֹחֵץ וְרוֹפֵא / חֲבָשׁ נָא לְנִסְפֶּה / מָחוּץ וְנִכְפֶּה / בְּלִי הָרִים קוֹל

נַהֵל בְּחַסְדָּךְ / לְעַם דַּל וָדָךְ / מָלֵא פַחְדָּךְ / וְהוּא מִתְחַנֵּן בְּקוֹל

שַׂגֵּב רְחוּמִים / וְהָעֵר רַחֲמִים / וּפְדֵה עֲגוּמִים / כְּעָגוּר יְצַפְצְפוּ בְקוֹל

עֲווֹנָם תִּכְבֹּשׁ / וְצִירָם תַּחְבֹּשׁ / וְעַמְּךָ אַל יֵבוֹשׁ / בְּעֶתֶר בְּקוֹל

פְּדֵה נָא עַמָּךְ / וְהָקֵם נָאֳמָךְ / וְהָפֵר זַעֲמָךְ / מִמַּנְעִימֵי קוֹל

צְדָקָה תַעֲשֶׂה / לְעַם עֹל נוֹשֵׂא / וּפִשְׁעָם תְּכַסֶּה / וּבַשְּׂרֵם בְּקוֹל

קַבֵּץ פְּזוּרָה / אָיֹם וְנוֹרָא / וְקַנֵּא לְתוֹרָה / נִתְּנָה בְקוֹל

רְפָא נָא שִׁבְרָהּ / וַחֲבֹשׁ מְזוֹרָהּ / וְיִזְרַח אוֹרָהּ / וְרוֹמְמָהּ בְּקוֹל

שְׁעֵה נָא לְעַם דַּל / וְעוֹד אַל יִדַּל / וְכֹחֲךָ יִגְדַּל / וְחַזְּקֵהוּ בְקוֹל

תְּחַדֵּשׁ עֵדֶיךָ / לְבָאִים עָדֶיךָ / אֲזַי אֲנִי אוֹדֶךָ / בִּתְרוּעַת קוֹל.
