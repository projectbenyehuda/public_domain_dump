בְּעוֹדִי מִבְּלִי שֵׂכֶל

כתובת השיר: וזה שיר ידידות, מדובר בו נכבדות. אהבה רוחנית.


בְּעוֹדִי מִבְּלִי שֵׂכֶל וְחָכְמָה / לְבָבִי רָץ לְכָל יָפָה וְתַמָּה,

וּבִזְמַן נַעֲרוּת אָהַב עֲלָמוֹת / וְחָשַׁק יַעֲלוֹת כַּמָּה וְכַמָּה,

וְלָפִיק תַּאֲוָה הָיָה רְצוֹנוֹ / לְבָד, מִכָּל יְפַת מַרְאֶה וְקוֹמָה

וְלֹא הִבְדִּיל בְּזוּיָה מִיְּקָרָה / וּבַת נָדִיב כְּנִדְגָּלוֹת אֲיֻמָּה.

וְעֵת פָּנָה זְמַן נֹעַר, וְהָיָה / כְּנוֹסֵעַ, וּמֶרְכַּבְתּוֹ רְתוּמָה –

בְּרֹעַ מַעֲשַׂי חָשַׁב לְבָבִי / וְכִמְעַט כִּסְּתָה פָנַי כְּלִמָּה

וְשַׁבְתִּי מֵחֲשֹׁק עַלְמָה, לְבַד אִם / כְּבֻדָּה הִיא וּבַת מֶלֶךְ פְּנִימָה.

וְהִנֵּה נִקְשְׁרָה נַפְשִׁי בְנֶפֶשׁ / צְבִיָּה יְקָרָה מֵעָשׁ וְכִימָה:

צְבִיָּה הִיא, וְדִמּוּהָ לְשֶׁמֶשׁ, / וְאִם מִכָּל דְּמוּת יָקְרָה וְרָמָה,

וְשֵׁם יָפְיָהּ כְּבָר פָּשַׁט בְּתֵבֵל / וְאָחַז פַּאֲתֵי מִזְרָח וְיָמָּה.

אֲהַבְתִּיהָ, וְשַׂמְתִּי אַהֲבָתָהּ / בְּבַת עֵינִי בְּעֵט בַּרְזֶל רְשׁוּמָה,

וְאַהְבָתָהּ כְּטַבַּעַת בְּחוֹתָם / נְתוּנָה תּוֹךְ סְגוֹר לִבִּי וְשׂוּמָה,

וְהִנֵּה הִיא בְּמוֹרָשֵׁי לְבָבִי / כְּזֵר זָהָב בְּטוּרֵי אַחֲלָמָה,

אֲבָל יוֹם יוֹם תְּעַנֵּנִי בְנוּדָהּ / וְלֹא תִתֵּן לְעַפְעַפַּי תְּנוּמָה.

וְלֹא אֶחְשֹׁב נְגֹעַ בָּהּ לְעוֹלָם – / וּמַה חַיַּי אֲנִי כִּי אֶגְּעָה, מָה?

וְיָדַעְתִּי הֲכִי יַם־צוּף וְנֹפֶת / בְּפִיהָ, וַאֲנִי אָמוּת בְּצָמָא!

וְדַי לִי בָהּ שְׁמֹעַ מַאֲמָרִים / עֲרֵבִים וַחֲזוֹת צוּרָה נְעִימָה,

אֲבָל דַּי לִי בְּעֵת הַמַּחֲשָׁבָה / תְּמוּנָתָהּ בְּלִבִּי לִי מְדֻמָּה.

הֲכִי לוּ אֶשְׁמְעָהּ אוֹ אֶחֱזֶנָּה – / אֲזַי לֹא נִשְׁאֲרָה לִי כָל מְהוּמָה,

הֲכִי הַמֵּת בְּמַרְאֶהָ תְחַיֶּה / וְהַכּוֹשֵׁל בְּמִלֶּיהָ מְקִימָה,

וְאֵין חֶפְצִי בְתַאְוָתָהּ לְתַעְנוּג / גְּוִיָּה, רַק לְתַעְנוּג הַנְּשָׁמָה.

וְהֵן אִם נֶעְלְמָה מִנִּי, הֲתוּכַל / הֱיוֹת מֵעֵין לְבָבִי נַעֲלָמָה?

רְחוֹקָה הִיא, וּמִקָּרוֹב בְּכָל עֵת / אֲשׁוּרֶנָּה בְּעַיִן לֹא עֲצוּמָה

בְּמַחְשָׁבָה מְאֹד זַכָּה וּבָרָה / וּבִתְשׁוּקָה גְדוֹלָה וַעֲצוּמָה.

וְיֵשׁ אוֹמְרִים ״הֲלִפְלוֹנִית תְּחַכֶּה? / וּבִזְבוּל הִיא, אֱמֶת, סַהַר וְחַמָּה!

וְזֹהַר אוֹר תְּקַו לִתְפֹּשׂ בְּיָדְךָ – / וְלֹא יִדְבַּק בְּיָדֶיךָ מְאוּמָה.

וְלָמָּה תַאֲדִיב נַפְשָׁךְ בְּחִשְׁקָהּ? / הֲיַחְשֹׁק עָשׁ וְכִימָה עָשׁ וְרִמָּה?

הֲתוּכַל לַעֲלוֹת לָהּ אַתְּ לְשַׁחַק? / וְאִם תֵּרֵד לְךָ הִיא לָאֲדָמָה?

עֲזֹב חִשְׁקָהּ וְסוּרָה מִגְּבִירוֹת / וּבִנְעָרוֹת בְּחַר יוֹנָה תְמִימָה

אֲשֶׁר לָפִיק זְמָמֶיךָ תְמַהֵר / וְתָרוּץ לַעֲשׂוֹת חֶפְצָךְ, יְהִי מָה,

וְלָעַד לֹא תְרוֹצֵץ רַק לְעָפְרָה / יְהִי לָךְ הוֹן לְהַשִּׂיג לָהּ וְעְָצְמָה״.

עֲנִיתִים ״בּוֹעֲרִים, אַל תֶּהְבְּלוּ נָא / וְדֹמוּ נָא, תְּהִי לָכֶם לְחָכְמָה!

הֲטוֹב כִּי אֶחְשֹׁק עַלְמָה מְאַוָּה / לְמַלּאוֹת תַּאֲוַת נֹעַר, וְהוֹמָה?

הֲלֹא כַאְשֶׁר תְּמַלֵּא תַאֲוָתִי / תְּמַלֵּא תַאֲוַת כָּל בָּא בְּמִרְמָה,

וְאֶתְנַחֵם בְּהָפִיק מַאֲוַיַּי – / וְיִקְרֵנִי כְמִקְרֵה הַבְּהֵמָה!

וְלָעַד אַהֲבָה לֹא תַעֲמֹד, רַק / אֲשֶׁר תִּהְיֶה תְלוּיָה עַל בְּלִימָה.

וּמַה נָּעִים וְטוֹב חֵשֶׁק אֲשֶׁר לֹא / יְהִי נוֹסָד עֲלֵי עָווֹן וְאַשְׁמָה!

וְחֵשֶׁק בּוֹעֲרִים – חֵשֶׁק בְּתֶבֶל / וְחֵשֶׁק רוֹזְנֵי תֵבֵל – מְזִמָּה״.
