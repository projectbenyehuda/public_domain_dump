[השערוריה באורווה]

וַיֹּאמֶר אֵלַי “עַד מָתַי תָעֵז פָּנֶיךָ / לְחָרְפֵנִי וּלְהַכֵּנִי בְשׁוֹט לְשׁוֹנֶךָ?/ אָכֵן אֶסְבֹּל חֶרְפָּתֶךָ / לְמַעַן שִׂכְלְךָ וּתְבוּנָתֶךָ”. / וַיָּכֹף רֹאשׁוֹ וַיֹּאמֶר בְּבֹשֶׁת פָּנִים / “כָּזֶה וְכָזֶה יִשָּׂא הָאֱנוֹש מֵאוֹהֲבָיו הַנֶּאֱמָנִים. / וְעִם כָּל זֶה אֶתְמַהּ עַל חָכְמָתֶךָ – / וְיוֹתֵר עַל אֲכִילָתֶךָ: / אֲשֶׁר מְאוּמָה לֹא הִשְׁאַרְתָּ / וְשֻׁלְחָנִי מִכָּל טוּב חִסַּרְתָּ, / וּדְבַר שְׁלֹמֹה אֶל לִבְּךָ לֹא שַׁתָּ / “וְשַׂמְתָּ שַׂכִּין בְּלֹעֶךָ אִם בַּעַל נֶפֶשׁ אָתָּה”. / הִכְזַבְתָּ דִּבְרֵי הֶחָכָם אֲשֶׁר אָמַר, כִּי חָכְמָה וְדֵעָה / לֹא יִשְׁכְּנוּ בְּבֶטֶן מְלֵאָה וְנֶפֶשׁ שְׂבֵעָה”. / וָאֹמַר אֵלָיו “עַד אָן תְּמַלֶּל אֵלֶּה / וְאֵי זֶה מַאֲכָל בִּטְנִי מִלֵּא? / וַהֲלֹא כָל אֲשֶׁר אָכַלְתִּי הַלַּיְלָה בְרַעֲבוֹנִי / לֹא מִלֵּא חֶדֶר אֶחָד מֵחַדְרֵי בִטְנִי”.

וַיֹּאמַר “הַמָּקוֹם יַצִּילֵנִי מִכָּל צָרָה וְצוּקָה, / לֹא אַמְשִׁיל בִּטְנְךָ כִי אִם לַעֲלוּקָה! / וְאֶתְמַהּ עַל שְׁלֹמֹה אֲשֶׁר אָמַר: שָׁלוֹשׁ הֵנָּה לֹא תִשְׂבַּעְנָה אַרְבַּע לֹא אָמְרוּ הוֹן. אֵיךְ לֹא אָמַר: חֲמִשָּׁה – וְתִהְיֶה בִטְנְךָ הַחֲמִישִׁית אֲשֶׁר לֹא תִשְׂבַּע לְעוֹלָם, / וְהִיא שְׁקוּלָה כְּנֶגֶד כֻּלָּם”.

וּבְעוֹד הָיָה מְדַבֵּר עִמִּי בִלְשׁוֹן צַחוֹת / אָמַרְתִּי “בָּרוּךְ זוֹכֵר כָּל הַנִּשְׁכָּחוֹת! / אֲדוֹנִי בַּעַל הַבַּיִת, יָדַעְתָּ אִם אָכַל הַלַּיְלָה חֲמוֹרִי מִסְפּוֹא / אוֹ נָפַל עָלָיו הַקִּיר וּמָלַק אֶת רֹאשׁוֹ מִמּוּל עָרְפּוֹ / אֲשֶׁר לֹא אֶשְׁמַע נַהֲקָתוֹ / וְלֹא אַאֲזִין נַאֲקָתוֹ?”

וַיֹּאמַר “אֵין זֶה כִּי אִם מֵרֹב שְׂבִיעָתוֹ”. / וָאֹמַר בְּלִבִּי: מִי יִתֵּן וְיִצְדְּקוּ מִלָּיו / וְיִשְׂבַּע הַחֲמוֹר אַחֲרֵי אֲשֶׁר לֹא נִשְׂבַּע בְּעָלָיו / וַיִּקְרָא לָעֶבֶד וַיֹּאמֶר אֵלָיו / “נָתַתָּ הַלַּיְלָה לַחֲמוֹרוֹ אֶת בְּלִילוֹ / מַדּועַ לֹא יִשָּׁמַע קוֹלוֹ?” / וַיֹּאמֶר הָעֶבֶד "אֲדוֹנִי, שַׂמְתִּי לְפָנָיו תֶּבֶן וּמִסְפּוֹא מְלוֹא הָאֵבוּס / וְאָכַל וְשָׂבַע עַד אֲשֶׁר אוֹתָם נַפְשׁוֹ תָבוּס ". / וַיֹּאמַר “וּמִי יִתֵּן וְהָיְתָה כְקֵבַת הַחֲמוֹר קֵבַת אֲדוֹנָיו / וְאוּלַי הָיִינוּ אוֹכְלִים מִן הַנִּשְׁאָר לְפָנָיו”. / וְחָשַׁבְתִּי בְלִבִּי כִּי הָעֶבֶד הַמַּמֵּר, אֱמֶת הָיָה אוֹמֵר. / וָאֹמַר “אֲדוֹנִי, בַּעֲבוּר אַהֲבָתְךָ וּנְדִיבָתְך ָ חָשַׁבְתִּי כָל מַאֲכָל כְּמָן / וְהִנֵּה שֻׁלְחָנְךָ לְפָנֶיךָ סִימָן, / וְעַל כֵּן לֹא הוֹתַרְתִּי מִמֶּנּוּ עַד בֹּקֶר”. / וַיִּזְעַק וַיֹּאמַר “אֵין כָּל דְּבָרֶיךָ בִלְתִּי כָזָב וָשֶׁקֶר! / עַד אָן תְּהוֹתֵת עָלַי / וְתָבוּז לְשֵׂכֶל מִלַּי? / דֹּם וְהֵאָלֵם, / כִּי לֹא תֵבוֹשׁ וְלֹא תִכָּלֵם. / טֵאטֵאתָ שֻׁלְחָנִי בְּמַטְאֲטֵא הַשְׁמֵד – / וַעֲדֶן כָּל אֲשֶׁר בְּבֵיתִי לֶאֱכֹל אַתָּה חוֹמֵד!” / וָאֹמַר "אֲדוֹנִי, כַּפְּךָ מֵעָלַי הַרְחַק / וְאַל תָּשֶׁת לִבְּךָ לְדִבְרֵי לַעַג וּמִשְׂחָק ". / וַיֹּאמַר “וְאֵיךְ אָסִיר אֶת אַפִּי? / וְאָכַלְתָּ אֶת כָּל יְגִיעַ כַּפִּי / וְכִמְעַט אוֹתִי בְמַכְתֵּשׁ לְחָיֶיךָ תִשְׁחַק – / וְתֹאמַר כִּי תִלְעַג וְכִי תִשְׂחַק!” / וָאֹמַר אֵלָיו "אֲדוֹנִי, מִיּוֹם הֱיוֹתִי / כֵּן הָיְתָה אֲכִילָתִי, / וְעַל כָּל הַנִּשְׁאָר לִבִּי יִכְאַב, / כִּי מִנְּעוּרַי גְדֵלַנִּי כְאָב / וּמְבֶּטֶן אִמִּי אַנְחֶנָּה / לְרַוּוֹת בִּטְנִי וּלְהַשְׂבִּיעֶנָּה “. / וַיִּצְחַק צְחוֹק הַדּוֹמֶה לִבְכִי וַיַּעֲנֵנִי עַזּוֹת / “אָרוּר כָּל בֵּן אֲשֶׁר יִלְמַד מֵאָבִיו הַמְּלָאכָה הַנִּבְזָה הַזֹּאת!” / וַיֹּאמַר “וּמִי הָיָה אָבִיךָ?” אָמַרְתִּי לוֹ “אָבִי ז”ל הָיָה אִישׁ תָּם וְנָבָר, / רוֹפֵא חָכָם וּמַשְׂכִּיל בְּכָל דָּבָר”. / וַיֹּאמַר “מִדְּבָרֶיךָ אֶתְבּוֹנֵן כִּי אֱמֶת הַדָּבָר וְנָכוֹן, / כִּי הָיָה חָכָם בְּכָל דָּבָר וְנָבוֹן, / וְעֵדוּת חָכְמָתוֹ הַגְּדוֹלָה – / אֲשֶׁר לִמַּדְך ָ חָכְמַת הָאֲכִילָה!”

וּבְעוֹד הָיִינוּ מְדַבְּרִים נִדְעַךְ הַנֵּר אֲשֶׁר הָיָה לְפָנֵינוּ / וַיְעַצֵּם הָאֹפֶל אֶת עֵינֵינוּ. / וַיִּקְרָא אֶל עַבְדּוֹ וַיֹּאמֶר לוֹ “הַדְלֵק מְהֵרָה אֶת הַנֵּר!” / וְאֵין קוֹל וְאֵין עוֹנֶה, כִּי הֲרָגוֹ אֲחִי עָנֵר. / וָאֹמַר “מַה קָּרָה לָעֶבֶד הַזֶּה?” / וַיֹּאמַר “אוּלַי הוּא שׁוֹכֵב הוֹזֶה / וַיּוֹשִיבֵנִי בְמַחֲשַׁכִּים / עַד אֲשֶׁר אוֹר הַבֹּקֶר מֵצִיץ מִן הַחֲרַכִּים”.

וָאֹמַר אֵלָיו “אֲדוֹנִי, הַבֹּקֶר אוֹר / הָקֵץ אֶת הָעֶבֶד וְיָלוּשׁ וְיָשֶׂם בַּבָּצֵק מְעַט שְׂאוֹר”. / וַיֹּאמֶר אֵלַי “הַבָּשָׂר עוֹדֶנּוּ בֵין שִׁנֶּיך ָ – / וְכָל זֶה מְהֵרָה תַחְפֹּץ לְמַלֵּא בִטְנֶךָ? / הֲלֹא שָׁמַעְתָּ דְבַר אַבּוּקְרַט בְּאָמְרוֹ, כִּי הֲבָאַת מַאֲכָל עַל מַאֲכָל / רֹב בְּנֵי אָדָם הֵמִית וְשָׁכַל”. / וָאֹמַר “כֵּן דִּבַּרְתָּ / וְצָדַקְתָּ בְכָל אֲשֶׁר אָמַרְתָּ, / אֲבָל כָּל אֲשֶׁר אָכַלְתִּי נֶעְכַּל בְּמֵעַי וְנִסְפֶּה / כֶּאֱכֹל קַשׁ לְשׁוֹן אֵשׁ וַחֲשַׁשׁ לֶהָבָה יִרְפֶּה”. / וַיֹּאמַר “הַמָּקוֹם יַכֶּה בְשִׁלְשׁוּל קֵבָתֶך ָ / עַד אֲשֶׁר לֹא תוּכַל לַעְצֹר אֶת צוֹאָתֶךָ!” / וָאֹמַר “יָדַעְתִּי כִּי רָעָה עָלַי חָרַשְׁתָּ / אָכֵן הִתְפַּלַּלְתָּ וְלֹא פֵרַשְׁתָּ: / אִם אֲחָזַנִי שִׁלְשׁוּל, אוּלַי יִהְיֶה לִי מַרְפֵּא וַאֲרוּכָה, / וְיַהֲפֹך ְ לִי הַשֵּׁם אֶת הַקְּלָלָה לִבְרָכָה, / כִּי אָמַר גַּאלֵינוּס : כַּאֲשֶׁר יִקְרֶה לְאָדָם שִׁלְשׁוּל מִן הַלֵּחוֹת אֲשֶׁר הַטֶּבַע צָרִיךְ לְהוֹצִיאָם מִן הַגּוּף וּלְפַנּוֹתָם / יֶחֶזְקוּ בוֹ כָל הָאֲבָרִים וְיִהְיֶה סִבַּת בְּרִיאוּתָם”. / וַיֹּאמַר "אִם טָעִיתִי בִתְפִלָּתִי, / הַמָּקוֹם יוֹדֵעַ כַּוָּנָתִי ". / אָמַרְתִּי “יָדַעְתִּי כַּוָּנַת לִבְּךָ וּמַחֲשָׁבוֹ / אָכֵן קִלְלַת חִנָּם לֹא תָבוֹא”.

וָאֵלֵךְ לִרְאוֹת אֶת חֲמוֹרִי וָאָבוֹא אֶל הָאֵבוּס וָאֶמְצָא אוֹתוֹ מִבְּלִי תֶבֶן וּמִסְפּוֹא / וְלֹא יָכוֹל לַהְגּוֹת עַל טַרְפּוֹ / וְכִמְעַט מְצָאָהוּ מַחֲנָק, / כִּי הָעֶבֶד הַבְּלִיַּעַל חָסַם אֶת פִּיו לְבַל יִנְהַק. / וָאֶגַּשׁ אֵלָיו לְהָסִיר אֶת הַמַּחְסוֹם, וְכַאֲשֶׁר הֲסִירוֹתִיו נְשָׁכַנִי / וְכִמְעַט בֵּין רַגְלָיו הִדְרִיכַנִי. / וָאֶקַּח אֶת הַמַּקֵּל וָאַךְ אוֹתוֹ – / וְעַד לַשָּׁמַיִם עָלְתָה שַׁוְעָתו ֹ. / וַיִּפְתַּח יְיָ אֶת פִּי הַחֲמוֹר וַיֹּאמֶר בְּעֶבְרָה וָזַעַם / “לָמָּה תַכֵּנִי וַאֲנִי מִמִּשְׁפַּחַת אֲתוֹן בִּלְעָם?” / וָאֶתְמַהּ מִדְּבָרוֹ וְחָרַדְתִּי מֵחֲרוֹנו ֹ / וְאָמַרְתִּי: אוּלַי הַשָּׂטָן מְדַבֵּר מִגְּרוֹנוֹ. / וָאֹמַר אֵלָיו “וְאֵיךְ לֹא אַכֶּךָּ – וְאַתָּה נְשַׁכְתַּנִי / וְכִמְעַט בֵּין רַגְלֶיךָ רְמַסְתַּנִי?” / וַיֹּאמֶר הַחֲמוֹר “הִנֵּה יָשַׁבְתָּ עִם הַנָּבָל לְהִתְעַנֵּג וְלֶאֱכֹל – / וַאֲנִי בְרָעָב וּבְצָמָא וּבְחֹסֶר כֹּל / וְכָל הַלַּיְלָה אֶשְׁמְרָה לְפִי מַחְסוֹם”. / אָמַרְתִּי “הֲבִלְעָם אָנֹכִי אֲשֶׁר יָדַע לִקְסֹם? / מֵאַיִן אֵדַע כִּי צָמֵאתָ וְרָעַבְתָּ / וְהָעֶבֶד הָרָע אָמַר אֵלַי, כִּי מֵאֲשֶׁר אָכַלְתָּ וְשָׂבַעְתָּ / בָּאֵבוּס רָבַצְתָּ וְכָרַעְתָּ”. / וָאֶזְעַק אֶל הָעֶבֶד וָאֹמַר אֵלָיו “לָמָּה הֵמַתָּ אֶת חֲמוֹרִי בְרָעָב וּבְצָמָא / וְהִרְעַבְתָּ אוֹתוֹ בְאַכְזְרִיּוּת חֵמָה?” / וַיַּעַן אוֹתִי וַיִּשָּׁבַע, כִּי שָׂם לְפָנָיו תֶּבֶן וּמִסְפּוֹא וְאָכַל וְשָׂבַע – / אָכֵן קֵבַת הַחֲמוֹר כְּקֵבַת אֲדוֹנָיו / אֲשֶׁר לֹא הִשְׁאִיר מְאוּמָה לְפָנָיו. / אָמַרְתִּי “וְאִם נָתַתָּ לוֹ מִסְפּוֹא לְשָׂבְעָתו ֹ, / מַדּוּעַ חָסַמְתָּ אוֹתוֹ / לְבַל אֶשְׁמַע נַהֲקָתוֹ?” / אָמַר "בַּעֲבוּר כִּי הוּא חֲמוֹר שֶׁעֲסָקָיו רָעִים / וְכָל הַלַּיְלָה הָיָה צוֹעֵק וּמַרְעִים, / וְעוֹד כִּי פָחַדְתִּי פֶּן יִשְׁכֵנִי בְשִׁנָּיו ".

אָמַרְתִּי “בֶּאֱמֶת כַּעֶבֶד כַּאדוֹנָיו! / וְאָמַר הָעֲרָב, כִּי הָעֶבֶד מֵחֹמֶר אֲדוֹנָיו קֹרַץ”. / וַיִּשְׁמָעֵנִי אֲדוֹנָיו וּלְשׁוֹנוֹ עָלַי חָרַץ, / וַיֹאמֶר עֵינָן “לָמָּה תַכְבִּיר עָלַי בְמִלִּים / וּתְבַזֵּנִי זֶה שָׁלֹשׁ פְּעָמִים?” / וָאֹמַר אֵלָיו “עַל זֶה הוֹצֵאתַנִי מֵעִירִי / לְהָמִית בָּרָעָב אוֹתִי וַחֲמוֹרִי?” וַיֹּאמַר “אָרוּר הַיּוֹם אֲשֶׁר הוֹצֵאתִיךָ בוֹ, / בְּמִסְפַּר יְרָחִים אַל יָבוֹא! / כִּי לֹא הֲבֵאתִיךָ אֶל בֵּיתִי כִּי אִם לְבָשְׁתִּי וּכְלִמָּתִי / וּלְהוֹכִיחַ עָלַי פִשְׁעִי וְאַשְׁמָתִי. / אֲבָל חֵי נַפְשִׁי וְנֶפֶשׁ זְקֵנִי אַשְׁמְדַאי, / אִם לֹא תֹּאמַר לְסִכְלוּתְךָ דַי, / וּלְקַחְתִּיךָ בְאֶצְבָּעִי הַקְּטַנָּה / וְהֶרְאִיתִיךָ כַנָּה / וְקִרְיַת חָנָה!” / וָאַבִּיט אֵלָיו – וְהִנֵּה פְּנֵי לְהָבִים פָּנָיו / וְכִידוֹדֵי אֵשׁ עֵינָיו / וּמִנְּחִירָיו יֵצֵא עָשָׁן / כְּקִיטוֹר הַכִּבְשָׁן. / וָאֹמַר בְּלָשׁוֹן רַכָּה / וְנַפְשִׁי עָלַי מְבַכָּה / “אֲדוֹנִי, אֶשְׁאָלְךָ וְאַל יִחַר בְּעֵינֶיךָ: / אֵיזֶה אַשְׁמְדַאי הָיָה זְקֵנֶךָ?” / וַיֹּאמַר “אַשְׁמְדַאי הַשַּׂר הַגָּדוֹל אֲשֶׁר בְּאֶבְרָתוֹ / הִשִּׂיא הַמֶּלֶךְ שְׁלֹמֹה מִמַּלְכוּתוֹ / וְהִרְחִיקָהוּ מֵאַדְמָתוֹ”. / וָאֹמַר “אוֹי לְנַפְשִׁי כִּי חֲשַׁבְתִּיך ָ מִן הַיְדִידִים – / וְאַתָּה מִבְּנֵי הַשֵּׁדִים / אֲשֶׁר רוּחוֹת בְּנֵי אָדָם מְצוֹדֲדִים! / זֶה חַסְדְּךָ אֶת רֵעֶךָ? / לָמָּה לֹא הִגַּדְתָּ לִי זַרְעֶךָ / וּמַדּוּעַ כָּכָה רִמִּיתַנִי / וּמֵאֶרֶץ מוֹלַדְתִּי הוֹצֵאתַנִי?” / וַיֹּאמֶר אֵלַי “וְאַיֵּה אֵפוֹא שִׂכְלְךָ וּתְבוּנָתֶךָ / וּמַדָּעֲךָ וְחָכְמָתֶךָ? / וַהֲלֹא אָמַרְתִּי לְךָ כִּי שְׁמִי עֵינָן הַנָּטָשׁ בֶּן אַרְנָן הַדָּשׁ! הֲפֹךְ שְׁמִי וְשֵׁם אָבִי וְתִמְצָא: עֵינָן הַשָּׂטָן בֶּן אָרְנָן הַשֵּׁד – בֶּן חֲצַר-מָוֶת / בֶּן עַזְמָוֶת / בֶּן צַלְמָוֶת / בֶּן חֲרָדָה / בֶּן רְעָדָה / בֶּן אֵימָה / בֶּן חֵמָה / בֶּן מְשַׁכֵּלָה / בֶּן חַלְחָלָה / בֶּן דֶּבֶר / בֶּן שֶׁבֶר / בֶּן יִגְרַע / בֶּן שֵׁם רַע / בֶּן עַלְוָן / בֶּן יָוָן / בֶּן לָצוֹן / בֶּן אָסוֹן / בֶּן מַגֵּפָה / בֶּן אֲנָפָה / בֶּן רָם / בֶּן שָׁרָם / בֶּן אַשְׁמַאי / בֶּן אַרְמַאי / בֶּן סַרְחַדּוֹן / בֶּן אֲבַדּוֹן / בֶּן קַלְקָלָה / בֶּן קְלָלָה / בֶּן אַבְדַאי / בֶּן אַשְׁמְדַאי / מֶלֶךְ הַשֵּׁדִים / אֲשֶׁר כֻּלָּם הָיוּ לוֹ לַעֲבָדִים”.

וּכְשָׁמְעִי אֶת כָּל אֵלֶּה הַשֵּׁמוֹת / אֲחָזוּנִי שָׁאוֹן וְשַמּוֹת / וְנָפְלוּ עָלַי פַּחַד וְאֵימוֹת / וְשָׁכַח לִבִּי כָּל גִּיל וַעֲלִיצוּת / וְאָחַז בְּשָׂרִי פַּלָּצוּת.

וָאֹמַר “אוֹי לִי עַל שִׁבְרִי / כִּי פֹה יִהְיֶה קִבְרִי! / אֱמֶת אָמְרוּ חֲכָמִים: רַגְלוֹהִי דְּבַר אֱנָש ׁ אִינּוּן עֲרֵבִין בֵּהּ / לַאֲתַר דְּמִתְבְּעִי תַמָּן מוֹבְלִין יָתֵה ּ”.

וַיֹּאמֶר אֵלַי “אַל תִּירָא וְאַל תֵּחַת / וְכָל דְּבָרֶיךָ יִהְיוּ בְנַחַת: / כִּי חֵי כָל הַנְּשָׁמוֹת אֲשֶׁר זָכַרְתִּי – / לֹא אֶעֱזָבְךָ עַד אֲשֶׁר עָשִׂיתִי אֶת אֲשֶׁר דִּבַּרְתִּי! / קוּם הִתְהַלֵּךְ בָּאָרֶץ לְאָרְכָּהּ וּלְרָחְבָּהּ / וּרְאֵה יָפְיָהּ וְטוּבָהּ, / אֶפֶס כִּי עַז הָעָם הַיּוֹשֵׁב בָּהּ”.
