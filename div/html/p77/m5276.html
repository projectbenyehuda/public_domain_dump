לו הדיו

כתובת: אין קול ואין עונה על כן שלחתי אליו את דברי השירה הזאת ואומר

לוּ הַדְּיוֹ בִקְצֵה יְאוֹר מִצְרַיִם / אוֹ הַמְּגִלָּה בַאֲרָם נַהְרַיִם

אוֹ יַעֲשֶׂה לוֹ עֵט כְּנָפַיִם כְּעֵיט / הָרִים לְעוֹפֵף עַל פְּנֵי שָׁמַיִם

יוֹמָם וְלַיְלָה אַחֲרֵיהֶם אֶרְדְּפָה / עַד אֶתְפְּשֵׂם עִם אֶצְבְּעוֹת יָדַיִם

אוֹ אִם שְׁלֹשׁ אֵלֶּה זְמָן לֹֹא כֹוֹנְנָם / תַּחַת דְּיוֹ אֶקְחָה שְׁחוֹר עֵינַיִם

וּמְקוֹם מְגִלָּה עוֹר לְבָבִי אֶפְשְׁטָה / וַאֶתְּנָה טוּחַי כְלוּחוֹתַיִם

וּתְהִי לְשׁוֹנִי עֵט רְצוֹנִי יִמְשְׁכוּ / תָמִיד לְבַל יִדְבַּק לְמַלְקוֹחַיִם

אֶשָּׂא שְׁלוֹמִי אֶל מְיוּדָעַי וּמֵ- / רָחוֹק לְדוֹדַי אֶפְרְשָׂה כַּפַּיִם

וּבְנִיב שְׂפָתַי אֶדְרְשָׁה לִמְאַהֲבַי / הַיּוֹשְׁבִים בַּחְמַת וּבִסְפַרְוַיִם

טוּרֵי אֲהָבִים אֶכְתְּבָה וַאְסוֹבְבָה / לַאְשֶׁר בְּהָם וּבְעַשְׁתְּרוֹת קַרְנַיִם

וּבְנָאֳמִי אֶפְקֹד יְדוּעֵי הַנְּדוֹד / וּלְיוֹדְעַי אֶשְׁתַּחֲוֶה אַפַּיִם

תִּשְׁכַּח יְמִינִי אִם יְקָרָם אֶשְׁכְּחָה / בֹּקֶר וְצָהְרַיִם וּבֵין עַרְבַּיִם

כִּי אִם לְאַפְסֵי יָם רְחוֹקִים חָלְפוּ / עַל קִיר לְבָבִי לָקְחוּ מַחְנַיִם

אֶזְכּוֹר יְדִידוּתָם וּמִדֵּי דֵבְּרִי / בָהֶם כְּתַנּוּר יֶהֱמוּ מֵעַיִם

לָצֵאת אֲלֵיהֶם לוּ שְׂעִפַּי יוּכְלוּן / מִן הַצְּבָאִים יִשְׁאֲלוּן רַגְלַיִם

יָחֵף בְּרַגְלִי אֶעְבְרָה אוֹ אֶמְכְּרָה / צַדִּיק וְאֶבְיוֹן בַּעֲבוּר נַעְלַיִם

בִּרְצוֹן לְבָבִי לַחֲזוֹתָם אֹחֲזָה / כַנְפֵי אֲדָמָה רַחֲבַת יָדַיִם

כִּי קָרְבוּ יַחְדָּו בְּעֵינַי פַאֲתֵי / תֵבֵל כְּאַמָּה אוֹ כְאַמָּתַיִם

אַךְ הַזְּמָן גָּדַר נְתִיבוֹתַי וְשָׂם / בַּסַּד שְׁתֵי רַגְלַי וּבִנְחוּשְׁתַּיִם

גָּדַר בְּגָזִית אֶת דְּרָכַי וַתְּהִי / נַפְשִׁי כְמַבְכִּירָה עֲלֵי אָבְנַיִם

וָאֶסְבְּלָה וָאמְרָה אַךְ זֶה חֳלִי / אֶשָּׂא עֲלֵי שִׁכְמִי וּבַכְּתֵפַיִם

אֶשָּׂא חֲרוֹן אַף הַנְּדוֹד וַאֶשְׁכְּבָה / בֵינוֹת שְׁפָתַיִם וּבֵין כִּירַיִם

וּבְמַעֲנֶה פִיךָ שְׁאֹנִי אֶשְׁכְּחָה / בוֹ אֶמְצְאָה נַחַת מְלֹא חָפְנַיִם

אַנְשֶׁה יְגוֹנִי עִם לְשׁוֹן זָהָב אֲשֶׁר / לֹא יַעֲרִיכֵנוּ זְהַב פַּרְוַיִם

וַיְּהִי דְבָרָךְ לִי לְעֹז וּלְמַחֲזִיק / וּבְחִין נְאוּמָךְ לָקְחָה קַרְנַיִם

וַהְגוּתְךָ מִשְׁעַן לְרוּחַ נַעֲנָה / כַּאְשֶׁר אֱלֵי נֶפֶשׁ עֲיֵפָה מַיִם

אַךְ מַה לְּעֵת כָּזֹאת הֲלֹא אֶלְאֶה סְבֹל / מִשֵּׂאת פְּרִידָה כָבְדָה כִפְלַיִם

כִּי שָׂם זְמָן עַל דַּל שְׂפָתֶיךָ זְמָם / וּלְשׁוֹנְךָ סָגַר לְבֵין שִׁנַיִם

נִלְאָה נְשֹׂא לִבִּי מְצוּקַת נוּדְךָ / עוֹל סִבְלוֹ הָיָה לְשִׁבְעָתַיִם

כִּי אֵיךְ בְּלִי מִשְׁעַנְתְּךָ רוּחִי יְחִי / אוֹ בִלְתְּךָ אֵיךְ יַעֲמֹד יוֹמַיִם

אֵיךְ יַעֲמֹד רֶגַע וְאֵין לוֹ מַעֲמָד / כִּי חֵץ נְדוּדָךְ יֶחֱצוֹ לִשְׁנַיִם

אַחְלַי שְׁעֵה לֵב נֶחֱלֶה סוֹבֵב עֲלֵי / קֹטֶב יְדִידוּתָךְ כְּמוֹ רֵחַיִם

מִפַּאֲתֵי מִזְרָח לְךָ אֶל מַעֲרָב / יוֹדוּ בְכִנּוֹרוֹת וּבִמְצִלְתַּיִם


סימן

נִטְעֵי אֲהָבֵינוּ כְצִיץ שָׂדַי / צָמְחוּ וְהָיָה הַנְּדוֹד מַגָּל

אַךְ נִשְׁאֲרוּ שָׁרְשֵׁי יְדִידוּתָךְ / תּוֹךְ מַחֲנוֹת רוּחִי וְעֵד הַגַּל
