אדברה וירוח לי

          לעיני בני ישראל, לשם ולתהלת דון יצחק אברבנאל.

הַפֹּרְטִים עַל פִּי כִנּוֹר וָנֶבֶל

כִּמְעַט הֲלֹא כֻלָּמוֹ

שִׂיחָם הֲכִי הִגְדִּילּו;

וַיֵּצְאוּ לִלְקֹט בִּנְגִינָתָמוֹ

צוּף מָן[וְלֹא הוֹעִילוּ],

עַל גִּיל וְעַל הָאֵבֶל;

וַיַחְשְׁבוּ מַשָׂא כָבֵד וָסֵבֶל,

גַּלּוֹת עֲדַת חָנֵף, שַׁבֵּחַ גֶּבֶר

חָכָם וּמַשְׁבִּיר שֶׁבֶר;

וַיִתְּנוּ לַתֹּם כָּתֵף סֹרָרֶת,

רַק לַעֲשׂוֹת לָהֶם הוֹן, שֵׁם תִּפְאָרֶת.


הַקֹּפְאִים עַל שִׁמְרִיהֶם יַכְתִּירוּ,

וִיהַלְלוּ דַרְכָּמוּ;

וּבְמָר-דְּרוֹר וָבֶשֶׂם

הֵם יִרְקְחוּ בִגְדָם, גַּם כִּי בוּז לָמוֹ;

הוֹדָם כְּאוֹר הַלֶּשֶׁם

יִתְרֹעֲעוּ, יָשִׁירוּ,

גַּם כֹכְבֵי נִשְׁפָּם לָהֶם יָאִירוּ;

עַל יוֹם כְּלוּלוֹתָם אֹמֶר יַמְתִּיקוּ,

מִצוּף דְּבַשׁ יָצִיקוּ;

כָּזֹאת בְּיוֹם מוֹתָם פֶלֶא יַשְׁמִיעוּ,

עַל הָאֲבָנִים רַב טוּבָם יוֹדִיעוּ.


הֵם יַחְנוּ סָבִיב הֵנָּה וָהֵנָּה

לִדְרשׁ לְשׁוֹן הָאָרֶץ;

"דֹּבֵר שְׂפַת בֶּן-עֵבֶר

[הֵם אֹמְרִים] פֹרֵץ עָלָיו כָּל-פָּרֶץ;

כִּי חֵן בְּפִי הַגֶּבֶר

אֵין לָהּ, הֲלֹא עוֹדֶנָּה

זָרָה, מְשׁוֹרֵר הִנֵּה יִמְאָסֶנָּה;

שָׂפָה יְשָׁנָה הִיא כִימֵי שָׁמָיִם,

עַתָּה כְזֶרֶם מָיִם

הֵן עָבְרָה פִתְאֹם, כִּי יֵשׁ לָהּ חֶלֶף;

שָׂפָה עֲנִיָּה אֵיךְ תִּהְיֶה לָאֶלֶף?"

- - - - - - -

בִּמְשֹׁרְרֵי עַמִּים יִנְעַם טוּב טָעַם

אִם הוֹד, אֱמֶת וָישֶׁר

מַעְגַּל זְמִירוֹתָמוֹ,

וִירוֹמְמוּ חָכְמָה, לֹא הוֹן וָעשֶׁר;

אָז אוֹר יְקָרוֹת לָמוֹ,

אוֹר צַח בְּסֵתֶר רָעַם,

אָז צָלְחָה דַרְכָּם גַם זֹאת הַפָּעַם.

וּבְשֵׁם אֲדֹנַי, צוּר תָּמִים, עָשָׂנִי.

כֵּן אֶעֱשֶׂה גַם אָנִי

לִכְבוֹד חֲכַם לֵבָב, אִישׁ עֹשֵׂה פֶלֶא

יִצְחָק שְׁמוֹ, בֵּיתוֹ אַבַּרְבָּנֵילִי.


בִּשְׁנַת צְחֹק יֻלַּד יִצְחָק בָּאָרֶץ,

יֻלַּד בָּעִיר לִסְבֹּנָה,

נָאוָה וְקִרְיַת מֶלֶך;

חָכְמוֹת בְּתוֹכָהּ כָּל-הַיּוֹם תָּרֹנָה,

הֵן הִיא לְחֵפֶץ הֵלֶךְ,

מָחוֹז בְּבֹא הַקָּרֶץ,

מִשֹּׁד לְהַצִּילוֹ, מִנִּי כָל-פָּרֶץ,

וּשְׁאוֹן הֲמוֹן גַּלִּים לֹא יַפְחִידֵהוּ;

יִשְׂמַח לְקוֹל רֹאֵהוּ

מֵהָאֳנִי יוֹרֵד אֶל הַיַּבֶּשֶׁת

בַּל יֹאכְלֵהוּ עוֹד לָבוּשׁ קַשְׂקֶשֶׂת.


יִצְחָק בְּלֵב תָּמִים, עֹדֶנּוּ נָעַר

אָהַב מְלַמֵּד ישֶׁר,

חֹקֵר אֱמוּנָה אֹמֶן,

דֹּבֵר אֱמֶת, רֹדֵף צֶדֶק וָאשֶׁר.

בִּסְחִי, בְּאֵשׁ וָדֹמֶן

בָּזָה כְסִיל וָבָעַר,

וּכְבַהֲמוֹת שָׂדָי, וּכְחַיְתוֹ-יָעַר.

נָתַן בְּאוֹצָרוֹ חָכְמָה וָדָעַת.

לֹא חָח וְלֹא טַבָּעַת,

עַד כִּי בְרֹב שִׂכְלוֹ הִפְלִיא, וַיֵּלֶךְ

בֵּין הַפְקִידִים עַל גִּנְזֵי הַמֶּלֶךְ.


עָמַד בְּבֵית אַלְפוֹנס שָׁלֵו, וַיֵּרֶב

עָשְׁרוֹ וְשֵׁם תִּפְאֶרֶת

עַל מַעֲשָׂיו לַמֶּלֶךְ;

יִקְרַת תְּבוּנָתוֹ הִנֵּה נֶהְדֶּרֶת

בָּעִיר וְגַם בַּפֶּלֶךְ

אָז בֵּין זְאֵבֵי עֶרֶב

הִצִּיל בְּחָכְמָתוֹ נַפְשׁוֹ מֵחֶרֶב,

כִּי אֹרְבִים חִצָם לִירוֹת בַּסֵּתֶר

הֵם כּוֹנְנוּ עַל יֶתֶר,

עֹשִׂים בְּתַחְבּוּלוֹת לַחְצֹב לוֹ קֶבֶר;

מִי יַעֲמֹד לִפְנֵי קִנְאַת הַגֶּבֶר?


אָבַד אֱנוֹשׁ חָסִיד, אוֹי! מִן הָאָרֶץ

אַלְפוֹנס בְּפָּרְטוֹגָאלוּ

מֶלֶךְ וְשֹׁפֵט צֶדֶק!

כָּלוּ יְמֵי חַיָּיו, כֶּעָשָׁן כָּלוּ,

אֵין עוֹד מְחַזֵּק בֶּדֶק,

בָּא לַיְּהוּדִים קָרֶץ,

כִּי אֹהֲבָם אֵין עוֹד, אֵין גֹּדֵר פָּרֶץ;

עָפָר עֲלֵי רֹאשָׁם שָׂמוּ כֻּלָּמוֹ.

הָהּ! אֵין תְּעָלָה לָמוֹ.

כִּי קָם בְּנוֹ יוֹאָן לָהֶם מַרְזֵחַ,

יָרֵא אֲזַי יִצְחָק, מִשָׁם בֹּרֵחַ


חָמֵשׁ וְאַרְבָּעִים מִיּוֹם יִוָּלֶד

שָׁנָיו אֲשֶׁר עָבָרוּ;

אַךְ אַחֲרֵי הַשֶּׁקֶט

הִנֵּה כְמוֹקֵד עַצְמוֹתָיו נִחָרוּ;

אֵין עוֹד פְּרִי, אֵין לֶקֶט

נָע נָד עֲלֵי הֶחָלֶד,

אָבַד צְרוֹר כַּסְפּוֹ, עָרוֹם כַּיָּלֶד

הָלַךְ לְקַסְטִלְיָיא לִמְצֹא מָנוֹחַ,

כִּי אֵין אֱיָל וָכֹחַ

לוֹ, מִמְּגִנַּת לֵב, מֵחִיל וָפָחַד,

אַף כִּי בְּנֵי בֵיתוֹ לֹא הָיוּ יָחַד.


הָאֵל אֲשֶׁר נֹתֵן לַיָּעֵף כֹּחַ,

יֶתֶר שְׂאֵת וָיֶשַׁע,

בַּיּוֹם וְגַם בַּלָּיִל

נַפְשׁוֹת חֲסִידָיו הַשֹׁמֵר מֵרֶשַׁע

נָתַן לְיִצְחָק חָיִל

מֶנְהוּ לְבַעֵר חוֹחַ

בֶּטַח שְׁלֵיו לִשְׁכֹּן, אָז אָז לָקֹחַ

סֵפֶר בְּרִית הֵחֵל, דַּת אֵל שָׁמָיִם

אָז חִישׁ וּבִשְׁנָתָיִם

סֵפֶר דְּבָרִים וּנְבִיאֵי הַצֶּדֶק,

הוֹאִיל בְּפִיו בֵּאֵר, לָסִיר כָּל-חֶדֶק.

לִבּוֹ בְקִרְבּוֹ חַם, וּכְאֵשׁ שַׁלְהֶבֶת

לֹהֵט בְּתוֹךְ הַיָּעַר,

אִוָּה לְמוֹשַׁב מֶלֶךְ,

לֹא שָׁקְטָה רוּחוֹ עַד צֵאת מִשָּׁעַר

קִרְיַת צְבִי, וַיֵּלֶךְ

אֶל עִיר אֲשֶׁר נִשְׂגֶּבֶת

מַדְרִיד, וְשָׁם יִצְחָק חָמַד לָשֶׁבֶת;

אָז בַּחֲצַר פֶרְנָאנדוֹ בִּלְתִּי רָעַד

הֵבִין וְהִרְחִיב צָעַד,

עַד כִּי בְעֵינָיו חֵן מָצָא בַל נֵכֶל,

נָבוֹן וְחָכָם, אִישׁ יֹדֵעַ שֵׂכֶל.


הוּא שַׁלְאֲנָן ישֵׁב, יוֹם יוֹם מַצְלִיחַ,

מֵסִיר בְּעַמּוֹ גֹעַל,

מֵטִיב לְאֹם דָּל, חֵלֶךְ,

כִּי בֶאֱמוּנָה הוּא עֹשֶׂה כָל-פֹּעַל

תָּמִיד בְּבֵית הַמֶּלֶךְ;

מַגִּיד לְמַלְכּוֹ שִׂיחַ,

עַל אֹיְבָיו יָרִיעַ אַף יַצְרִיחַ;

שָׁלֵו בְּמַתְכֻּנְתּוֹ, רָחוֹק מִכָּעַשׂ,

מִקּוֹל סְאוֹן, וַיָּעַשׂ

בַּיֹעֲצִים לוֹ שֵׁם, גָּדַר כָּל-פָּרֶץ,

שָׁמְעוֹ וְאוֹרוֹ עַל כַּנְפוֹת הָאָרֶץ.


אַחַר שְׁמֹנֶה שָׁנִים כִּי בֹטֵחַ

יִצְחָק בְּצֵל מַחְסֵהוּ,

חָשַׁךְ בְּצֵאתוֹ שֶׁמֶשׁ,

וַיֶחֱרַד יִצְחָק, רָפוּ יָדֵיהוּ;

לֹא זָרְחָה כָּאֶמֶשׁ

כִּימָה וְגַם יָרֵחַ;

הָיָה עֲלָטָה, עָשׁ עוֹד לֹא זֹרֵחַ;

אָבַד כְּסִיל כִּסְלוֹ כִכְסִיל וָבָעַר.

נָעוּ עֲצֵי הַיָּעַר;

בָּרָד וְאַבְנֵי אֶלְגָבִישׁ נָפָלוּ,

חוֹמָה וְחֵל עִבְרִים יַחְדָּו אֻמְלָלוּ.


הוֹי אָח וְהוֹי אָחוֹת, אִשָּׁה עִם בָּעַל,

בָּנִים וְהוֹרֵי ישֶׁר!

שָׁבַת מְשׂוֹשׂ לִבָּמוֹ,

נֶהְפַּךְ לְאֵבֶל גִּיל, לָעֹנִי עשֶׁר,

צָרָה וְיָגוֹן לָמוֹ;

שֹׁתֶה בְסַף הָרָעַל

כָּל אִישׁ יְהוּדִי, גַּם כִּי בוֹ אֵין מָעַל,

מִקְצֵה סְפָרָד חִישׁ לָצֵאת כַּהֵלֶךְ

כִּי כֵן דְּבַר הַמֶּלֶךְ:

"שָׁלשׁ אֲנִי נוֹטֵל, עִבְרִי, עָלֶיךָ:

מֻת, אוֹ בְרַח לָךְ, אוֹ הָמֵר דָּתֶךָ"


יִצְחָק שְׂבַע קָלוֹן סֹפֵק עַל יֶרֶךְ,

כִּי שִׁמְמָה וָקָרֶץ,

חֶרְפָּה, שְׂחֹק וְקֶלֶס

תָּמִיד לְיִשְׂרָאֵל יִהְיֶה בָאָרֶץ.

שָׁקַל אֲזַי בַּפֶּלֶס:

אִם תֵּת דֳּמִי לַפֶּרֶךְ,

כִי טוֹב מְאֹד לָלֶכֶת

לִבְכּוֹת, לְהִתְחַנֵּן, לִשְׁאֹל לַמֶּלֶךְ,

מִמַחֲשַׁבְתּוֹ יָשׁוּב עַל עָם חֵלֶךְ.


אֵזוֹר לְמָתְנָיו שָׂם וִיקָר וָישֶׁר,

אֵמוּן וְעַנְוָה צֶדֶק;

לַחְגֹּר חֲלָצָיו כֹּחַ,

חַיִל לְחַזֵּק בִּיהוּדָה כָּל בֶּדֶק

בִּינָה לְהָכִין אשֶׁר,

וּמְעִיל צְדָקָה תַּחַת בִּגְדֵי עשֶׁר.

הֵן רָץ כַּנֶּשֶׁר חָשׁ לָעוּף, וַיֵּלֶךְ

לָשִׁיב חֲמַת הַמֶּלֶךְ;

עַל, חֵלְכָה אוּלַי יָחֹס, יוֹשִׁיעַ

דַּכָּא, וְכֹה יִצְחָק אֹמֶר הִבִּיעַ:


"מַלְכִּי, אֲדֹנִי רָם, רַחֵם עַבְדֶּךָ

בֹּכֶה בְמַר הַנֶּפֶשׁ,

וּבְצוֹם וְשַׂק וָאֵפֶר,

עַל עַמְּךָ, עַמִּי, מֻשְׁלָךְ בָּרֶפֶשׁ!

אוֹתִי מְחֵה מִסֵּפֶר

חַיִּים, וְעַל עַמֶּךָ

חוּסָה, וְאַל תִּשְׁפֹךְ עָלָיו זַעְמֶךָ;

הַבֵּט עֳנִי עַמִּי עַל כָּל-הָאָרֶץ,

נֶחְשָׁב כְּמוֹ הַשָּׁרֶץ;

בָּזוּי וְשָׁסוּי, חַכְלִילִי עֵינָיִם,

מִיֵין חֲמַת שַׁדַּי שֹׁתֶה כַמָּיִם".


מַחְסִי וּמִשְׂגַבִּי, כִּי לָךְ הַכֹּחַ,

עַל עַמְּךָ הָעֶבֶד

מַה לָךְ לְשַׁנּוֹת טָעַם?

מֵחַסְדְּךָ רָב לֹא אֶשְׁאַל לְזֶכֶר

כִי אִם לְכַלֹּת זָעַם;

אֶת-יָדְךָ פָּתֹחַ

תִּפְתַּח לְעַם עָנִי, תֹּאבֶה לִסְלֹחַ

מַלְכִּי, לְתַעְרָהּ נָא הָשֵׁב הַחֶרֶב,

בִּינָה הֲגִיגִי, הֶרֶב

לִי הַסְּלִיחָה; אַל יֶעְשַׁן אַפֶּךָ

חִנָּם בְיִשְׂרָאֵל צֹאן מַרְעִיתֶךָ.


שׁוּר אַחֲרֵי מִי, מֶלֶךְ רָב, יָצָאתָ,

אַחַר זְבוּב וָשָׁרֶץ;

פַּרְעשׁ הֲלֹא רָדַפְתָּ

נֶחְשָׁב כְּאַיִן עַל כַּנְפוֹת הָאָרֶץ.

מַה זֶּה אֲשֶׁר הִשְׁקַפְתָּ

עַל עַמְּךָ, כִּי בָאתָ

נֶגְדּוֹ בְּשֶׁטֶף אַף? מַה זֶּה מָצָאתָ

לִדְחוֹת מְפֻזָר וּמְפֹרָר בֶּחָלֶד

מִשָׂב וְעַד הַיָּלֶד?

מַה זֹּאת, אֲדֹנִי רָם, עָשִׂיתָ לָנוּ,

עַל מֶה חֳרִי, הָאַף הַגָּדוֹל בָּנוּ?"


עַל מֶה, אֲדוֹנִי, פֹה חָצַבְתָּ קֶבֶר

לָנוּ? אֲהָהּ הַקְשִׁיבָה

קוֹל צֹעֲקִים אֵלֶיךָ!

בִּרְצוֹנְךָ נַעַר, זָקֵן הֵטִיבָה

בֹּכִים לְכַף רַגְלֶךָ!

אִם יֶחֱטָא רַק גֶּבֶר

הוּא בַעֲוֹן בִּצְעוֹ יוּמַת בַּל שֶׁבֶר;

אַךְ עַל עֲדַת עִבְרִים הָסֵר כָּל-כָּעַשׂ

מִלִּבְּךָ; הִיא תָעַשׂ

אָז, כָּל-אֲשֶׁר תֹּאמַר, אָדוֹן, אֵלֶיהָ,

נָקִי וּבַר תִּרְאֶה פֹעַל כַּפֶּיהָ"


"מַלְכִּי וְצוּר יִשְׁעִי הֶרֶב לִסְלֹחַ

עַל כָּל-אֲשֶׁר דִּבַּרְתִּי;

עָנִי וְכֹאֵב אָנִי.

אִם הַגְּבוּל, וּכְעַז פָּנִים עָבַרְתִּי.

עֹז אַפְךָ הִכָּנִי,

אֵין בִּי עֱזוּז וָכֹחַ,

לֹא מָצְאָה נַפְשִׁי מֵאָז מָנוֹחַ.

שִׁמְעָה תְפִלָּתִי אַךְ זֹאת הַפַּעַם,

קַצֵּץ כְּלֵי הַזָּעַם!

אָנָּא לְגוֹי נִבְזֶה הַגְדִּל חַסְדֶּךָ,

כַּפֵּר לְיִשְׂרָאֵל, גַּם לִי עַבְדֶּךָ."


אוֹי! כַּאֲשֶׁר כִּלָּה יִצְחָק כָּל-אֹמֶר,

הִשְׂכִּיל אֲזַי וַיָּבֶן,

כִּי אָבְדָה תוֹחֶלֶת

עִבְרִים אֲמֵלָלִים, וּכְמוֹ הָאָבֶן

קָשֶׁה דְבַר מֶמְשֶׁלֶת;

רָעָה מְלֹא הָעֹמֶר

נֶגְדָּם, וְהִנָּם, הָהּ! מִרְמָס כַּחֹמֶר.

הִנֵּה שְׁמֹעַ לֹא אָבָה הַמֶּלֶךְ,

וַיֶחֱרַד וַיֵּלֶךְ

יִצְחָק חֲפוּי רֹאשׁוֹ, אָבֵל, כִּי שָׁמָּה.

לֹא שָׁקְטָה רוּחוֹ, גַּם לֹא נֻחָמָה.


אוֹי וַאֲבוֹי קִינִים בַּחוּץ נִשְׁמָעוּ

אֵין לַיְּהוּדִים הֶגֶּה;

בָּאָה שְׁנַת הַזֶּרֶם

לָהֶם, בְּקִינִים הָהּ! וָהִי וָהֶגֶה,

הֵם נִתְפְּשׁוּ בַחֶרֶם,

מִפֹּה וּמִפֹּה נָעוּ,

נָצוּ, וּבִישִׁימוֹן הַדֶּרֶךְ תָּעוּ;

אָבַד יְקָר הוֹנָם בִּקְצוֹת הָאָרֶץ

[פָּרַץ אֲדֹנָי פָּרֶץ!]

אִישִׁים וְנָשִׁים, שָׂב, יָשִׁישׁ וָיָלֶד,

מִטַּף וְעַד זָקֵן נָדוּ בֶחָלֶד.


יִצְחָק וּבֵיתוֹ מִמַּדְרִיד, בַּיָּמָה

דֶּרֶךְ אֳנִי רַבָּתִי,

בִּרְתֵת וְחִיל וָפָחַד,

בָּאוּ לְקִרְיַת נָאפּוֹלִי שָׂרָתִי;

וַיִּמְצְאוּ גַם יָחַד

שִׁבְתָּם וְקוּמָם שָׁמָּה,

עַל כִּי עֲדַת עִבְרִים עוֹד בָּהּ רֻחָמָה.

מֵאֵין הֲפוּגוֹת אָז יִצְחָק עַל אֵלֶּה

הִרְבָּה לְהַפְלִיא פֶלֶא;

בָּטַח בְּחֶסֶד אֵל נוֹרָא, וַיֵּלֶךְ

שָׁלֵו לְפֶרְדִּינָאנדוֹ שָׁמָּה מֶלֶךְ.


דִּבֶּר וּמָצָא חֵן, בִּינָה יֹדֵעַ;

אֶת כָּל-אֲשֶׁר קָרָהוּ

סִפֵּר בְּעֹז וָכֹחַ;

פָּקִיד בְּאוֹצָרוֹ פִתְאֹם שָׂמָהוּ

מַלְכּוֹ, וּבוֹ לִבְטֹחַ

הֵחֵל כְּאָח וּכְרֵעַ,

עַל כִּי בְּיִצְחָק הוֹד הִכִיר וָרֵעַ.

הִנֵּה מְשַׂנְאָיו מִפָּנָיו יָפוּצוּ,

כָּל-אֹהֲבָיו יָרוּצוּ

לוֹ, כִּי בְנֵר שִׂכְלוֹ שִׂכְלָם יָאִירוּ,

וּבְאוֹר בְּרַק לִקְחוֹ, לִקְחָם יַזְהִרוּ.


וַיְהִי בְיוֹם מוֹת פֶרְנָאנְדוֹ הַמֶּלֶךְ

עַמָּיו בְּכִי נָשָׂאוּ,

מִסְפֵּד, נְהִי וָאֵבֶל,

כִּי הוּא בְמָלְכוֹ טוֹב, כֻּלָּם מָצָאוּ,

מַרְחִיק בְּעַמּוֹ הֶבֶל.

תּוֹמֵךְ יְמִין הַחֵלֶךְ,

אִם גֵּר וְאִם אֶזְרָח, תּוֹשָׁב וָהֵלֶךְ.

אַלְפוֹנס בְּנוֹ תַחְתָּיו קָם, גַּם כִּי נָעַר,

שֹׁפֵט אֱמֶת בַּשָּׁעַר;

וּבְכֵן יְכַבֵּד אָב אָהַב בַּחֵשֶׁק

יִצְחָק, וְעַל בֵּיתוֹ שָׂמוֹ בֶן-מֶשֶׁק.

הֵרִים כְּבוֹד הוֹדוֹ עַד לַשָּׁמָיִם.

כִּי בַחֲצַר הַמֶּלֶךְ

יִצְחָק בְּפִי כָּל-גֶּבֶר:

רָצוּי וְדֹרֵשׁ טוֹב בָּעִיר וָפֶלֶךְ.

יוֹעֵץ וּמַשְׁבִּיר שֶׁבֶר,

אִמְרוֹ כְמוֹצָא מָיִם,

עֻזוֹ כְיָם גָּדוֹל וּרְחַב יָדָיִם.

אֶל מוּל פְנֵי אַלְפוֹנס בֹּקֶר וָלָיִל,

עָמַד לְהַרְאוֹת חָיִל;

וּבְכָל-מָקוֹם מוֹשַׁב מַלְכּוֹ בָּאָרֶץ,

שֹׁקֵד לְהַצִּילוֹ מִשֹּׁד וָקָרֶץ.


אוֹיָה! מְשׂוֹשׂ מַלְכּוֹ שָׁבַת לָנֶצַח

עַל כִּי בְלִי כָל-פֶשַׁע

קַרְלוֹ שְׁמִינִי מֶלֶךְ

צָרְפַת קְרָב עָרַךְ נֶגְדּוֹ מֵרֶשַׁע;

בָּרַח אֲזַי וַיֵּלֶךְ

אַלְפוֹנס בְּשִׁפְלוּת מֵצַח

אֶל עִיר וְחֹף מֵיסִינָה; שָׁם מֵרֶצַח

נַפְשׁוֹ לְהַצִּיל [עִם יִצְחָק בֹּרֵחַ

אִתּוֹ] הֲלֹא בֹטֵחַ:

אַךְ קָדְחָה חִישׁ בּוֹ אֵשׁ הַקַּדָּחַת,

בִּדְמִי יְמֵי חַיָּיו רָאָה הַשָּׁחַת.


יִצְחָק אֲזַי דָּוֶה בִּבְלִי כָל-סָעַד,

וּמְלֵא כְאֵב בַּשָׁעַר;

[כִּי אַהֲבָה כַמָּוֶת

עַזָּה בְקִרְבּוֹ עַל מַלְכּוֹ, עוֹד נָעַר

הֹלֵךְ בְּגֵיא צַלְמָוֶת;]

וַיֹּאחֲזֵהוּ רָעַד,

יִרְאָה וּפַלָּצוּת מֵהַרְחִיב צָעַד,

אָמַר בְּלִבּוֹ אָז, כִּי טוֹב לִנְסֹעַ

מֵעִר אֲשֶׁר בָּה רֹעַ.

וּבְצִי בְלֵב יַמִּים, בִּשְׁנַת הַמֶּרִי

הָלַךְ לְחֹף קָרְפוּ לָשֵׂאת לוֹ פֶּרִי.


שָׁמָּה חֲכָמִים גַּם מֵרָחוֹק בָּאוּ

דִּבְרֵי אֱנוֹשׁ לִשְׁמֹעַ

כֻּלּוֹ יְקָר וָאֶדֶר;

יִצְחָק בְּשַׁלְוָה אָז הֵחֵל לִנְטֹעַ

מַטָּע לְשֵׁם וָהֶדֶר;

הִתְמַהְמְהוּ, תָּמָהוּ,

אֵיךְ בּוֹ פְּרִי דַת אֵל פִּתְאֹם מָצָאוּ;

נֶחְמָד לְמַרְאֶה, טוֹב לַחֵךְ עוֹדֶנּוּ,

וַיִּלְקְטוּ מִמֶּנּוּ

אִישׁ אִישׁ לְפִי אָכְלוֹ, חֹמֶשׁ אוֹ רֹבַע,

כִּי אֹהֲבָם הוֹצִיא לָהֶם לָשׂבַע.


יִצְחָק בְּשָׂשׂוֹן פֹּה מָצָא הַסֵּפֶר

בָּזוּז [וְאֵין יוֹדֵעַ

עַד מָה] בְּעִיר לִסְבֹּנָה

מֶנְהוּ, וְעַל מִשְׁנֵה תּוֹרָה נוֹבֵעַ

מֵי דָת, וּבוֹ תָרֹנָּה

חָכְמוֹת בְּאִמְרֵי שֶׁפֶר,

אֵין תַּעֲלוּלִים בּוֹ, אֵין מִשְׁלֵי אֵפֶר,

יִצְחָק בְּבֶן-אָמוֹץ שָׂם אָז מַחְסֵהוּ,

הִפְלִיא בְּרוֹם יָדֵיהוּ,

צַחוֹת יְשַׁעְיָה נֶאְדָּרִי בַכֹּחַ,

בֵּין חֹקְרֵי לִבּוֹת לִרְמֹס הַחוֹחַ.


יָצָא בְיָד רָמָה מִשָּׁם, וַיֵּלֶךְ

יִצְחָק לְמוֹנוֹפּוֹלִי

(עִיר עַל שְׂפַת הַיָּמָּה)

בָּרִיא, שְׁלֵיו, בִּלְתִּי יָגוֹן וָחֹלִי

פֹה שָׁקְטָה, נֻחָמָה

נַפְשׁוֹ, אֲשֶׁר כַּהֵלֶךְ

מִפֹּה וּמִפֹּה נָע מֵעִיר אֶל פֶּלֶךְ.

חֹסֶן יְשׁוּעוֹת אֶל עַמּוֹ הִשְׁמִיעַ,

אוֹרוֹ הֲלוֹם הוֹפִיעַ;

הִשְׁקָה פְּנֵי אַרְצוֹ מִמַּעְיַן יֶשַׁע,

יֵשַׁע מְשִׁיחוֹ הוּא, מֹחֶה כָּל-פֶּשַׁע.


שָׁם מַחֲזֶה שַׁדַּי, חֶזְיוֹן אֵל נֵצַח

הִפְלִיא בְיִצְחָק שָׁמָּה;

בִּשְׂדֵה אֱמֶת וָצֶדֶק

הִנֵּה אֲלֻמָתוֹ נִצָּבָה, קָמָה;

אֵין קוֹץ וְאֵין בָּה חֵדֶק,

רַק לָהּ מְלִילוֹת קֶצַח,

אֵמוּן וּפִסַּת בַּר-לֵבָב לָנֶצַח.

רַב מִפְעֲלוֹת הָאֵל בִּבְרִיאַת חֶלֶד,

סִפֵּר לְלָבוּשׁ גֶּלֶד;

הִזְכִּיר יְמוֹת עוֹלָם, גָּלֻת בֶּן-עֵבֶר,

הֵבִין שְׁנוֹת דֹּר דֹּר לָתֵת-לוֹ שֶׁבֶר.


מֵרֹאשׁ אֲמָנָה הוּא הִזִּיל מֵי יֶשַׁע

מֵעֵין יְקָר וָלֶקַח;

הִשְׁקָה בְנֵי עַם עֵבֶר

מִנַּחֲלַת אָבוֹת, מִיֵּין הָרֶקַח

מָתוֹק לְחֵךְ כָּל-גֶּבֶר,

נָעִים לְכַלֵּא פֶשַׁע,

מַה טּוֹב לְהַשְׁבִּית חִישׁ שֵׁבֶט הָרֶשַׁע.

חִדֵּשׁ בְּמִכְתָבָיו חֻקֵּי שָׁמָיִם,

הָיָה כְעֵץ עַל מָיִם

שָׁתוּל, אֲשֶׁר פִּרְיוֹ לִמְשׂוֹשׂ קֹנֵהוּ

יִתֵּן בְּעִתּוֹ, לֹא יִבּוֹל עָלֵהוּ.


חִזֵּק בְּמָעֻזוֹ עַל אֶדֶן טֹהַר

עַמּוּד אֱמֶת לָנֶצַח;

הִתְוָה בְּקַו הַיּשֶׁר

צֶדֶק לְעוֹלָמִים לִהְיוֹת עַל מֶצַּח;

הֶרְאָה זְבוּל הָאשֶׁר,

הֵפִיץ עֲטֶרֶת זֹהַר

עַל רֹאשׁ זְקֵנִים בָּרֵי לֵב וָתֹאַר;

עָשָׂה בְעִתּוֹ כֹּל, וּבְתַכְלִית יֹפִי

כִּבָּה יְקוֹד אֵשׁ דֹּפִי;

עַל מִזְבְּחוֹת כָּבוֹד הִקְרִיב כָּל-חָיִל,

עַל בֵּית מְרִי פָסַח כִּצְבִי וָאָיִל.


יִצְחָק בְעֵט סֹפֵר גָּדַר כָּל-פָרֶץ

שֶׁבַע שְׁנֵי הַשֶּׁקֶט

מִדֵּי הֱיוֹתוֹ שָׁמָּה,

דַּי לוֹ בְחָכְמָתוֹ לִלְקֹט בָּהּ לֶקֶט;

מִמַּעֲמַקֵּי יָמָּה

אוֹ מִתְּהֹמוֹת אֶרֶץ

הֵן הֶעֱמִיק, הִרְחִיב שָׁמְעוֹ. מִקָּרֶץ

הִצִּיל לְאֹם גֹּלֶה, נִדָּח, נָגוּעַ,

בַּבֹּץ וְטִיט טָבוּעַ;

כִּי בַעֲדוֹ הִזְהִיר אוֹר אִמְרֵי שֶׁפֶר,

נֹגַהּ שְׁבִיב דַּרְכּוֹ בִמְגִלַּת סֵפֶר.


אָרְכוּ לוֹ שָׁם הַיָּמִים, רֵעַ

בָּא לוֹ בְחֹזֶק מֵצַח

אִתּוֹ לְמַלֶּל-אֵלֶּה:

"שָׁלוֹם לְךָ יִצְחָק, שָׁלוֹם לָנֶצַח!

הַצְלַח וְהַפְלֵה פֶלֶא;

כִּי בָךְ עֱזוּז וָדֵעַ;

מַלְכִּי בְּפָּרְטוֹגָאל הִנֵּה יֹדֵעַ

אַל תַּעֲרֹץ, הַסְכֵּת וּשְׁמַע בַּל רָעַד

כִּי הִנְנִי לָךְ סָעַד,

אוּלַי בְּעֵינָיו חֵן יִמְצָא עַמֶּךָ,

אִם תַּאֲזִין אֵלַי, וּבְטוּב טַעְמֶךָ".


דַּע אֹהֲבִי כִּי צִיר לַמֶּלֶךְ אָנִי,

צִיר נֶאֱמָן שָׁלוּחַ

לִשְׁאֹל בְּבֹא אֵלֶיךָ

אִם תַּעֲשֶׂה חֶפְצוֹ? מַלְכִּי בָּטוּחַ

יִשְׁמַע לְעֹז קוֹלֶךָ

(כֵּן כַּאֲשֶׁר צִוָּנִי)

אַלּוּף וְשָׂרֵי עַם הַוֵּינֶצְיָּאְנִי;

כִּי בָם גְּדֹלוֹת, נִפְלָאוֹת אֵין חֵקֶר

לַחְקֹר בְּדִבְרֵי שֶׁקֶר,

לִשְׁפֹּט אֱמֶת בַּתֹּם וּלְהָבִין טָעַם

גַּם לַעֲזֹב עֶבְרָה אוֹ לִשְׁפָּךְ-זָעַם"


אַחַר אֲשֶׁר כִּלָּה צִיר פָּרְטוֹגָאלוּ

כָּל-סוֹד, וְשָׂם בַּסֵּתֶר

אֶל חֵיק יְדִיד פִּקֵּחַ,

יִצְחָק אֲזַי הֵשִׁיב לוֹ בִּשְׂפַת יֶתֶר:

"הִנֵּה אֲנִי שֹׁכֵחַ

אֶת-כָּל אֲשֶׁר נֹאָלוּ

נֶגְדִּי אֲנָשִׁים, כִּי עַתָּה נָפָלוּ

לִשְׁאֹל עֲצָתִי; אִישׁ נֹקֵם אֵינֶנִי,

לָכֵן הֲלֹם הִנֵּנִי

חָשׁ לַעֲבֹד מַלְכוּת, וּכְיַרְחֵי קֶדֶם;

אֵין לִי לְהוֹסִיף עוֹד עַל זֶה, וָאֶדֶם".


חִישׁ אֹהֲבוֹ נָסַע שָׁלֵו מִשָּׁעַר,

כִּי מִשְׁאֲלוֹת הַמֶּלֶךְ

יִצְחָק בְּבֹא קִרְיָתָה

הֵן יַעֲשֶׂה. אָז עֹז אָזַר, וַיֵּלֶךְ

יִצְחָק בְּלִי אֵימָתָה,

בִּלְתִּי רְתֵת וָשָׂעַר

[עִמּוֹ בְנוֹ יוֹסֵף עוֹד רַךְ וָנָעַר]

אֶל חֹף וְנֶצְיָה, עִיר בִּשְׂפַת הַיָּמָה,

מִבְחַר גְּבָרִים שָׁמָּה,

שָׂרִים וְאַנְשֵׁי שֵׁם, נִכְבַּדֵּי אָרֶץ,

הָעֹמְדִים יוֹם יוֹם לִגְדֹּר כָּל-פָרֶץ.


יִצְחָק בְּוֵינֶצְיָה אָז הֵרִים תֹּרֶן

כִּי כָל אֲשֶׁר הֵכִינוּ

צִיר נֶאֱמָן וָמֶלֶךְ

עָשָׂה. תְּבוּנָתוֹ שָׁמָּה הֵבִינוּ

אֶזְרָח וְגֵר וְהֶלֶךְ;

עֶטּוֹ כְמוֹ צִפֹּרֶן

שָׁמִיר וּמִכְתָּבוֹ כִּתְבוּאַת גֹּרֶן

גַּם ישְׁבֵי הָאִי רָאוּ, תָּמָהוּ.

אֶת כָּל אֲשֶׁר קָרָהוּ

כָּתַב בְּחֹסֶן רָב אֶל רֹאשׁ לִסְבֹּנָה

יִצְחָק, אֲשֶׁר חָכְמוֹת אֵלָיו תָּרֹנָּה.


צִוָּה לְמִשְׁפַּחְתּוֹ לָלֶכֶת שָׁמָּה,

כִּי הוּא וּבֵיתוֹ יָחַד

שָׁם יִמְצְאוּ מַרְגּוֹעַ

עַד תֹּם יְמֵי חַיָּיו בִּבְלִי כָּל-פָּחַד,

חֶרְדַת מְרִי וָרֹעַ.

נַפְשׁוֹ הֲלֹם נֻחָמָה

עִם אֹהֲבָיו רַבִּים מִקֵּדְמָה, יָמָּה.

תַּכְלִית בְּרִית הָאֵל וּנְבִיאֵי צֶדֶק

בֵּאֵר לְבַעֶר-חֵדֶק

מִמַּחְשְׁבוֹת סָכָל וּרְפֵה יָדָיִם,

מִקּוֹל מְחַרֵף מַעַרְכוֹת שָׁמָיִם.


אֶל רַב תְּמִים דֵּעִים, מֹרֶה הָאשֶׁר,

שָׁאוּל שְׁמוֹ בָאָרֶץ

כָּתַב בְּמַר הַנֶּפֶשׁ,

אֵיךְ בָּא לְכָל-בֵּיתוֹ פֶּרֶץ עַל פָּרֶץ

לִהְיוֹת כְּטִיט וָרֶפֶשׁ,

גַּם כִּי בְדַרְכֵי ישֶׁר

אֵמוּן וְזֵעַת אַף קָנָה לוֹ עשֶׁר

גָּדוֹל, וְלֻקַּח מֶנְהוֹ כִמְעַט רֶגַע

עַל צוֹק, חֲלוֹף וָפֶגַע;

אַךְ אַחֲרֵי בֹאוֹ אֶל עִיר שָׂרָתִי,

שָׁבָה לְמִשְׁפַּחְתּוֹ טוֹבָה רַבָּתִי.


יִצְחָק בְּוֵינֶצְיָה בָחַר לָשֶׁבֶת.

כִּי שָׁם מְקוֹם הַיֶּשַׁע;

קִרְיָה אֲשֶׁר בָּהּ צֶדֶק

יָלִין. וְלֹא יָקוּם שֵׁבֶט הָרֶשַׁע;

לֹא יִמְצְאוּ בָהּ בֶּדֶק

רֹאִים בְּעֹז נִשְׂגֶּבֶת

לָרִיק מְחִי נֶגְדָּהּ, לַשָׁוְא מַקֶּבֶת;

תִּשְׂחַק לְדָיֵק עִיר בָּאִיִים רָמָה,

מַלְכָּה בְתוֹךְ הַיָּמָה;

רַק בָּה בְּעַד רָעָב צָרִים יָבֹאוּ,

מִשְׁטַח חֲרָמִים שָׁוְא, צִנָּה לַתֹּהוּ.


אִוָה לְמוֹשָׁב לוֹ מַצְפּוּן הַדָּעַת

וּמְעוֹן נְכֹנֵי אָרֶץ,

כִּי שָׁם פְּרִי וָלֶקֶט

אֶל אֹהֲבֵי מוּסָר וּרְחֵקֵי קָרֶץ.

שָׁם מָצְאָה כָל-שֶׁקֶט

רוּחוֹ, וְשָׁם לָטָעַת

הֵחֵל בְּתוֹךְ גִּנַּת אֹזֶן שֹׁמָעַת

כָּל-מִין עֲצֵי בִינָה נֶגֶד הַיֵּצֶר

לָרָשׁ וְיֹרֵשׁ עֵצֶר.

הֶרְאָה לְאִישׁ נָבוּב דֶּרֶךְ הַמֶּלֶךְ

מֹרֶה נְבֻכִים הוּא, אִם גֵּר אִם הֵלֶךְ.


אוֹיָה! וְרַק חָמֵשׁ שָׁנִים עָבָרוּ

מִיּוֹם אֲשֶׁר הִגִּיעַ

לִמְצֹא הֲלֹם מַרְגּוֹעַ,

בקהל אדני את קלו לשמיע;

הֵן בָּא זְמַן לִנְסֹעַ,

שָׁנָיו, אֲבוֹי! גָּמָדוּ,

כָּל-אֹהֲבָיו בָּכוּ וַיִּתְמַרְמָרוּ,

כִּי כָהֲתָה עֵינוֹ, מֶנְהוּ נָס לֵחַ,

כַּדְרוֹר, כְּעוֹף פּוֹרֵחַ;

נֶהְפַךְ לְמָר מָתוֹק, עֵנָב לַבֹּסֶר.

הָהּ! נֶאֱסַף צַדִּיק בִּשְׁנַת הַ חֹסֶר.


שִׁבְעִים וְעוֹד שָׁנָה רוּחוֹ נִשְׁאָרָה

בּוֹ לַעֲמֹד לָאֶדֶן

אֶל ישְׁבֵי הֶחָלֶד;

עָף אַחֲרֵי כֵן, עָף יִצְחָק לָעֶדֶן

נָקִי וְחַף כַּיָּלֶד.

אוֹרָה וְשִׂמְחָה סָרָה

כִּי הַשְּׁמוּעָה כֹּה וָכֹה נִמְהָרָה.

אָז אֹהֲבָיו רַבִּים הַפָּאדוֹבָנִי

בָּאוּ לְוֵינֶצְיָאנִי,

לִשְׁאֹל לְהִתְחַנֵּן מִטּוּב לִבָּמוֹ,

לִקְבֹּר גְּוִיָּתוֹ אֵצֶל עִירָמוֹ,


בָּכוּ בְקוֹל גָּדוֹל וַיִּתְחַלְחָלוּ

בָּנָיו, וּפָאדוֹבָנִי

הָיוּ מְאֹד לָטֹרַח

וַיִפְרְצוּ בִּלְבַב הַוֵּינֶצְיָאנִי

לַשְׁבִּית קְרָב מֵאֹרַח.

הֵם לַאֲשֶׁר שָׁאָלוּ

לֹא מֵאֲנוּ; וּכְאִישׁ אֶחָד נִקְהָלוּ

עִבְרִים לְהָבִיא גֵו יִצְחָק אֶל קֶבֶר,

לֹא עַל שְׁכֶם הַגֶּבֶר;

אַךְ בַּעֲגָלָה לִכְבוֹדוֹ שָׂמוּהוּ,

וּלְשַׁעֲרֵי פָּאדוֹבָה הוֹלִיכוּהוּ.

קִרְיַת מְשׂוֹשִׂי, נֶאֱמָנָה, נִשְׂגֶבֶת,

רָמָה, וּמִימֵי קֶדֶם

מוֹשַׁב יְקָר וָדָעַת;

לִשְׂמֹאל, לְצַד יָמִין, אָחוֹר וָקֶדֶם

מִמֵּךְ הֲלֹא נוֹבָעַת

עֵין אַחֲוָה; בָּךְ שֶׁבֶת

וּמְכוֹן בְּרִית שָׁלוֹם: כָּל יָד כֹּתֶבֶת

עֻזֵּךְ וּמָעֻזֵךְ, עִיר הַתִּפְאֶרֶת;

אַשְׁרִיךְ אֲשֶׁר נִקְבֶּרֶת

תַּחְתֵּךְ גְּוִיַּת אִישׁ, מַחְמַד כָּל-גֶּבֶר,

יִצְחָק שְׁמוֹ. זֶה לוֹ צִיּוּן עַל קֶבֶר.

חָכָם הֲלֹם קֻבַּר, שֵׂכֶל יֹדֵעַ,

נֶחְמָד כְּיֵין הָרֶקַח

יוֹם יוֹם בְּבֵית הַמֶּלֶךְ;

תִּקֶּן סְפָרִים בִּתְבוּנָה וָלֶקַח;

גָּלָה שְׁלֵיו, וַיֵּלֶךְ

זָקֵן וְאַף שָׂבֵעַ

מֵעִיר לְעִיר, גַּם כִּי עָיֵף, יָגֵעַ.

חֹקֵר וְלֹא נֹטֶה עַל קַו הַתֹּהוּ,

אַיֵּה אֱנוֹשׁ כָּמֹהוּ?

נוֹדָע בְּכָל-גּוֹיִם, אִישׁ עוֹשֵׂה אֵלֶּה,

לֹא קָם כְּדוֹן יִצְחָק אַבַּרְבָּנֵילֵי.
