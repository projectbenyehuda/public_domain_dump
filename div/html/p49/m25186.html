[יחידתי לבבך יפת]

ולה ענד אנתקאלה פי אלצבא מן קרטבה ופיהא דכר אלאנתראב

ולו (שיר) בצאתו בנעוריו מקרטבה ובו הזכרת ההליכה לנכר


U – – – U – – – U //– – U – – – U – – – U – –

יְחִידָתִי, לְבָבֵךְ יַפְתְּ חִידָה // תְּהִי בִדְבַר נְדֹד אַחִים יְחִידָה,

בְּאוֹתִיּוֹת כְּמַשְׂכִּיּוֹת כְּתוּבָה // וּבִנְקֻדּוֹת מְחֻשָּׁקוֹת נְקוּדָה,

וְהַלָּמֵד כְּקֹהֶלֶת תְּעוּדָה // שְׁבָחָיו לַמְּדִי קָהָל וְעֵדָה

וּבֹאִי וַהֲגִי לַמּוֹעַדִים אֶת // חֲסָדָיו כַּאֲשֶׁר תָּבוֹא חֲסִידָה

וְאַל תִּנְשִׁי פְּרִידָתֵךְ יְדִידֵךְ // וְתָשׁוּבִי וְתִבְכִּי עַל פְּרִידָה,

לְמַעַן אַחֲרֵי נִשְכַּח זְכַרְתִּיו // וְשִׁכְחָתֵךְ אֲהוּבֵךְ עֵין מְרִידָה.

וְשׁוֹאֵל עַל מְקוֹם אֵשׁ הַנְּדוּדִים, // הֲשִׁיבוֹתִיו: בְּלִבִּי הִיא יְקוּדָה,

בְּהַשִּׂיאָה רְתִיחַת דַּם לְבָבִי // אֱלֵי רֹאשִׁי – עֲלֵי עֵינִי יְרֵדָה.

וְאֵיךְ יוּכַל אֱנוֹשׁ רִיבִי לְהַכְחִישׁ // אֲשֶׁר אֹמַר – וְדִמְעָתִי מְעִידָה,

כְּעֵדוּת עָשׁ לְבַת־עֵינִי, אֲשֶׁר הִיא // לְבָנֶיהָ עֲדֵי בֹקֶר יְעוּדָה!

בְּנוּדִי עָשְׁשׁוּ עֵינַי וְעָבְשׁוּ // עֲצָמַי כַּאֲשֶׁר עָבְשָׁה פְרוּדָה,

וּמָדַדְתִּי פְּנֵי אֶרֶץ בְּרַגְלַי, // כְּאִלּוּ הֵם שְׁנֵי חַבְלֵי מְדִידָה,

וְהָיוּ לִי דְּמֵי עֵינַי בְּבֵיתִי // לְלֶחֶם, וַעֲלֵי דַרְכִּי לְצֵידָה,

וְלוּלֵי כִי כְּתָב שָׁלַח יְדִידִי // בְּדָבָר טוֹב לְנַפְשִׁי הַחֲרֵדָה

וּבִשְׁלוֹמוֹ – אֲזַי בָּאָה לְנַפְשִׁי, // בְּלֹא עִתָּהּ, תְּמוּתָה הָעֲתִידָה!

דְּבָרִים נִמְצְאוּ כַצֵּל לְעֶבֶד, // וְכַגּוֹאֵל לְאַנְשֵׁי עִיר לְכוּדָה,

וְכַמָטָר עֲלֵי אֶרֶץ צְמֵאָה, // וְכַשָּׁלוֹם עֲלֵי נֶפֶשׁ פְּחוּדָה.

קְרָאתִימוֹ וְקָרָאתִי בְּקוֹל רָם: // דְּרוֹר נִקְרָא לְנַפְשִׁי הַשְּׁדוּדָה,

וְשָׁלֵם קֵץ יְגוֹנִי וַעֲווֹנִי, // וּבָא שָׁלוֹם, וְסָרָה הַקְּפָדָה!

וְאִם בִּכְתָב פְּקָדַנִי יְדִידִי – // יְחִידָתִי בְּנוֹשָׁעִים פְּקוּדָה,

וּבָא נוֹחַ לְיָדִי הָעֲמֵלָה, // וּמִשְׁלוֹחַ לְרַגְלִי הָעֲקוּדָה!

אֲנִי נִתְפָּשׂ בְּיַד זָרִים, כְּאִלּוּ // אֲנִי שֵׁנִי לְאִישׁ צִלָּה וְעָדָה,

וְאָסוּר תּוֹךְ מְצוּדַת נוֹד – חֲבֵרַי, // הֲתִתְרַפּוּ לְהַתִּיר הַמְּצוּדָה?

וּמִי יוּכַל לְהַצִּיל אֶת יְדִידוֹ // וְיִתְעַלֵּם – כְּזוֹבֵחַ לְשֵׁדָה!

אֲהוּבִי, אִם לְעֵקֶר נֶעֱבַדְתִי – // שְׁאֵרִי אַתְּ, פְּדֵנִי מֵעֲבוֹדָה!

וְאִם נִגְזָל אֲנִי מִבֵּין בְּנֵי אָב, // כְּבִנְיָמִין – הֱיֵה אַתָּה יְהוּדָה!

וְאִם אוֹבֵד אֲנִי – הָשֵׁב לְנִדְחִי, // וְלָךְ עַל אֵל שְׂכַר הָשֵׁב אֲבֵדָה!

וְאִם אַיִן – לְאֵל בִּי סוֹד, וְעַתָּה // תְּבוֹאֵנִי בְּנוֹרָאוֹת פְּקִידָה!
