יְמַהֵר יוֹם

ורתאה הדה אלמרתיה והי אטול מא רתאה בה ודכר פיה אלתלתה מן אהלה מאתו בעדה

וקונן עליו את הקינה הזאת, והיא הארכה מכל הקינות שבהן קונן עליו, והזכיר בה שלשה ממשפחתו שמתו אחריו

יְמַהֵר יוֹם / פְּקֻדָּתִי / וְיֹאבַד יוֹם / יְלִידָתִי

וְחִישׁ אוּבַל / אֱלֵי עָפָר / וְאַל אֶרְאֶה / בְּרָעָתִי.

הֲיֵשׁ מַכְאוֹב / כְּמַכְאוֹבִי, / הֲיֵשׁ דּוֹמָה / לְמַכָּתִי?

לְמִי אַגִּיד / יְגוֹן לִבִּי / וּמִי יִרְגַּז / לְקוֹרָתִי?

וּמִי יִשְׁמַע / מְרִי שִׂיחִי / וְיִתְעַשֵּׁת / לְהַוָּתִי?

יְחִידָתִי, / הֲיִי כֹפֶר / לְאָח יֻלָּד / לְצָרָתִי,

וּמִי יִתֵּן / וְתִלָּקַח / תְּמוּרָתוֹ / יְחִידָתִי!

אֲהָהּ עַל אָח / אֲשֶׁר אָחַז / כְּאָב חוֹמֵל / בְּאַהְבָתִי,

אֲשֶׁר הָיָה – / וְהָיִיתִי, / וְאֵינֶנּוּ – / וְחָיִיתִי!

וְלוּ אֵיטִיב / וְאֶשְׁמֹר אֶת / בְּרִיתֹו שֶׁ- / הֲפִירוֹתִי –

אֲזַי בָּאָה, / בְּיוֹם בָּאָה / חֲלִיפָתוֹ, / חֲלִיפָתִי!

זְכַרְתִּיךָ, / בְּכוֹר אָבִי, / בְּעֶרֶשׂ לֵיל / הֲנָחָתִי–

וְגָזַל זִכ ְ- / רְךָ מִתּוֹךְ / שְׁתֵּי עֵינַי / תְּנוּמָתִי;

וְאֶזְכֹּר אֶת / מְלוֹנָךְ, צָם / בְּבֵית שַׁחַת, / בְּבָרוֹתִי –

וְתִתְהַפֵּךְ / בְּתוֹךְ מֵעַי, / לְרֹאשׁ פֶּתֶן, / אֲכִילָתִי;

וְאֶזְכֹּר שִׁכ ְ- / נְך ָ צָמֵא / בְּקֶבֶר עֵת / שְׁתִיָתִי –

וְיִמָּסְכוּן / דְּמֵי עֵינַי / וְשִׁקּוּיַי / וְדִמְעָתִי.

אֲהָהּ, אָחִי, / הֲסִבּוֹתָ / לְקִינִים אֶת / עֲלִיצָתִי,

וְהָיִיתָ / נְגִינָתִי – / וְאַתְּ הַיּוֹם / מְגִנָּתִי;

וְהָיָה זִכ ְ- / רְךָ עָרֵב / עֲלֵי נַפְשִׁי / וְחַיָּתִי–

וְהַיּוֹם זִכ ְ- / רְךָ לַבַּת / יְקוֹד יִשְׁלַח / בְּלִבָּתִי.

וְעֵת אֶשְׁמַע / אֲשֶׁר יִקְרָא / בְּשֵׁם ‘יִצְחָק’ / בְּקִרְבָתִי –

אֲזַי תֵּחַם / יְחִידָתִי / וְתִשְׁתַּפֵּךְ / מְרֵרָתִי,

וְנֶגְדִּי אֶ- / חֱזֶה קוֹמוֹת / זְקוּפוֹת מִ- / נְּפִילָתִי,

וּפִיּוֹת מָ- / לְאוּ בִשְׂחוֹק / וְקוֹל רִנָּה / לְשׁוֹאָתִי,

וְאֵיךְ לֹא אֶ- / עֱצֹר בִּכְיִי / בְּבֹא שָׂשִׂים / לְבִכְיָתִי?

אֲהָהּ עַל אָח, / טְמַנְתִּיהו ּ / בְּיָדִי תּוֹךְ / מְחִלָּתִי,

וְהוֹרַדְתִּיו / בְּבוֹר עָמֹק, / וְהִשְׁכַּבְתִּיו – / וְיָצָאתִי,

כְּאִלּוּ לֹא / יְמֵי נֹעַר / אֱלֵי צַלְעִי / בְּמִטָּתִי,

וְלֹא יָשַׁב / לְבֵית סֵפֶר / וְדָת עִמִּי / בְּשַׁחְרוּתִי.

אֲהָהּ עַל אָח, / מְשַׁכְתִּיהו ּ / אֱלֵי שַׁחַת / בְּאַוָּתִי,

וְצַלְמָוֶת / וְגוּשׁ עָפָר / בְּצֵל חַדְרוֹ / הֲמִירוֹתִי,

כְּאִלּוּ לֹא / נְשַׁקְתִּיהו ּ / וְעָרְבָה לוֹ / נְשִׁיקָתִי,

וְלֹא שָׁכֵן / וְלֹא יָשֵׁן / בְּצֵל בֵּיתִי / וְקוֹרָתִי,

וְלֹא שָׂמַח / כְּאוֹכֵל עַד / בְּבוֹאִי מִ- / נְּתִיבָתִי,

וְלֹא שַׂשְׂתִּי / בְּתוֹךְ בֵּיתוֹ, / וְלֹא שָׂשׂ הוּא / בְּתוֹךְ בֵּיתִי,

וְלֹא חָלָה / לְרָעָתִי, / וְלֹא חָדָה / בְּטוֹבָתִי,

וְלֹא הָיָה / מְנוּסָתִי, / וּמָעֻזִּי / בְּצָרָתִי,

אֲשֶׁר סוֹדִי / אֲגַלֶּה לוֹ / וְאוֹדִיעו ֹ / תְּלָאָתִי.

מְאוֹר עֵינַי / אֲשֶׁר דָּעַךְ / וְהִרְחִיב אֶת / כְּוִיָּתִי,

הֲכִי יָחִיד / עֲזַבְנוּךָ / בְּקֶבֶר, יוֹם / מְהוּמָתִי?

הֲנִכְסַפְתָּ / לְרֵעִים מִן / שְׁאֵר אָבִי / וְהוֹרָתִי,

וְלָקַחְתָּ / שְׁלֹשָׁה מִן / בְּנֵי בֵיתִי / וְתַרְבּוּתִי

וְהִשְׁכַּבְתָּם / סְבִיבֶיך ָ / וְנָדוּ מִן / מְסִיבָּתִי,

וְלֹא הוֹעִיל / לְךָ מוֹתָם – / וְהוּא עָלַי / כְּמִיתָתִי.

מְעַט תִּשְׁכַּב / וְתִנֶּתֵן / בְּצַד קִבְרָךְ / קְבוּרָתִי,

וְיָמִים שֶׁ- / גְּזָלוּך ָ / מְהֵרָה יִג ְ- / זְלו ּ אוֹתִי.

יְנִיחָך ְ אֵל / אֲשֶׁר הֵמִית / בְּיוֹם מוֹתָךְ / מְנוּחָתִי,

וְיִגְמָלְך ָ, / כְּצִדְקָךְ, צוּר / גְּמָלַנִי / כְּרִשְׁעָתִי.

אֲנִי מוֹאֵס / שְׁנַת עֵינַי / וְקֵץ לָעַד / בְּנוּמָתִי,

וּמֵבִין כִּי / בְּיוֹם מָחָר / תְּבוֹאֵנִי / מְחִתָּתִי,

וְכֶעָשׂוּי / לְגוּפָתָךְ / יְהִי עָשׂוּי / לְגוּפָתִי.

וּמִיּוֹם בּוֹא / פְּקֻדָּתָך ְ / אֲנִי חַרֵד / לְחַטָּאתִי,

וְתוֹכַחַת / בְּמִיתָתָך ְ / לְכָל נָבוֹן / וְכָל פֶּתִי!
