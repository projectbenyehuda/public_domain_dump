אמת, נעלה מלכי

וַיֵּלְכוּ הַצָּרִים בִּרְצוֹנָם אֶל אִשְׁבִּילִיָּה וַיִּהְיוּ שָׁם עִם בַּעַל בְּרִיתָם הָאִישׁ עַבַּאד שַׂר הָעִיר. וַיִּמְרֹד וַיִּבְגֹּד בָּם וַיַּחֲזִיק בָּהֶם וַיַּאַסְרֵם בִּנְחֻשְׁתַּיִם וּמִלֵּא בָתֵּי הַכֶּלֶא מֵהֶם, וְזֶה מֵהַשָּׁנָה הַנִּזְכֶּרֶת. וְכַאֲשֶׁר מָסַךְ ה' בְּקִרְבָּם רוּחַ עִוְעִים וְקָמוּ קְצָתָם עַל קְצָתָם, הֻתְּרָה הַסְכָּמָתָם וְנִתְקַלְקְלָה עֲצָתָם וּבָטְלָה מַחֲשַׁבְתָּם וְהִכְזִיב ה' מֵהֶם וּמֵרֹע פְּעֻלָּתָם. וְאָמַר בָּזֶה, יְכַבְּדֵהוּ הָאֵל, וּקְרָאָהוּ:


זִמְרָה

אֱמֶת, נַעֲלָה מַלְכִּי בְּדַרְכּוֹ וּבִמְלַאכְתּוֹ, / וּמַה נִּבְדְּלוּ כָל חִשְּׁבוֹנָיו וּמַחְשַׁבְתּוֹ,

וּמַה נִּפְלְאוּ מִכָּל פְּעָלִים פְּעֻלּוֹתָיו / וְסוֹדוֹ אֲשֶׁר יָעַץ וְיַעַל עֲלֵי דַעְתּוֹ!

מְגַלֶּה עֲמֻקּוֹת לַאֲנָשִׁים – וְלֹא עָלוּ / עֲלֵי לֵב, אֲבָל הָיוּ חֲתוּמִים בְּטַבַּעְתּוֹ

יְדַמֶּה אֱנוֹשׁ כִּי יֵשׁ בְּיָדוֹ לְהָרַע אוֹ / לְהֵיטִיב וְלִקְשֹׁר אוֹ לְהַתִּיר עֲבוֹת קַשְׁתּו

וְכִי נֶעֱזַר, בִּקְרָב עֲלֵי צַר, בְּתַחְבּוּלוֹת, / וְנוֹשַׁע בְּעַם רַב אוֹ בְּכֹחַ בְּמִלְחַמְתּוֹ –

וְיֵשׁ מִי יְחֻבַּר אֶת כְּפִירִים לְהִוָּשַׁע / בְּחֶבְרָם – וְיִתְפָּרְדוּ עֲצָמָיו בְּמַחְבַּרְתּוֹ,

וְאוֹכֵל – וְאָכְלוֹ מַאֲכֶלֶת בְּכִלְיוֹתָיו, / וְחוֹתֵר – וְאֵין לוֹ דָם, כְּגַנָּב בְּמַחְתַּרְתּוֹ,

וְרָעֵב אֲשֶׁר אָחַז בְּכִכָּר לְהָסִיר אֶת / רְעָבוֹ – וּמִיתָתוֹ טְמוּנָה בְּתוֹךְ פִּתּוֹ,

וְגִבּוֹר אֲשֶׁר נוֹקַש בְּסוּסוֹ בְּמַעְרֶכֶת, / וּבָחוּר אֲשֶׁר כָּשַׁל בְּמַסְלוּל בְּאַדַּרְתּוֹ

וְכַמָּה יְהַלֵּךְ עַל אֲדָמָה רְחָבָה אִישׁ / מְשֻׁלָּח וְלֹא בָן כִּי כְּבָר בָּא בְּפַח כַּתּוֹא,

וּמִי עִוְּרוֹ צוּרוֹ – יְהַלֵּךְ לְרֶשֶׁת צַר / בְּרַגְלָיו עֲדֵי יִפֹּל בְּחֶרְמוֹ וּמִכְמַרְתּוֹ,

וּמִי בִהֲלוֹ הָאֵל – יְרוֹצֵץ לְהַוָּתוֹ, / בְּתֻמּוֹ וְטוּב לִבּוֹ, וְיָמוּת בְּלֹא עִתּוֹ,

וְעִוֵּת בְּמִקְנָתוֹ וּמִכְרוֹ – יְהִי טוֹעֵם / פְּעָלָיו בְּקִנְיָנָיו, וְדַרְכּוֹ בְּמִמְכַּרְתּוֹ,

וְקוֹצֵר כְּפִי רִשְׁעוֹ אֲשֶׁר יַחֲרֹשׁ רֶשַׁע, / וּמוֹצֵא פְּרִי כַחְשׁוֹ בְּתַלְמוֹ וּמַחְרַשְׁתּוֹ.

וְכַמָּה אֲשֶׁר תִּרְאֶה קְרָסִים בְּקַרְסֻלָּיו / בְּהָלְכוֹ לְשַׁלֵּחַ נְקִיִּים בְּיַד רִשְׁתּוֹ,

וְשַׁלִּיט אֲשֶׁר הִשְׁכִּים בְּאַרְמוֹן חֲבֵרוֹ שָׂשׂ / וְצָהֵל וּמִתְרוֹנֵן – וְיָלִין בְּמַהְפַּכְתּוֹ,

כְּצָרַי, אֲשֶׁר תַּתָּה, אֱלֹהַי, בְּפַח יָקוּשׁ / מְשַׂנְאָם, אֲשֶׁר נִרְאָה כְּאוֹהֵב, וּמַלְכֻּדְתּוֹ,

בְּשָׁלְחָךְ נְשִׂיא עֶפְרוֹן מְעַקֵּל נְתִיבוֹתָיו / בְּדֶרֶךְ נְתִיבָה אֶל אֲבַדּוֹן, וְאִבַּדְתּוֹ,

וְעִמּוֹ נְשִׂיא חִתִּים וּמֶלֶךְ יְבוּס חַיִּים / כְּמֵתִים, וְכֻלָּהַם בְּרִכְבּוֹ וּמֶרְכַּבְתּוֹ,

יְחִישׁוּן, בְּכָל חוֹבֵשׁ פְּאֵרִים וְלוֹבֵשׁ אֶת / כְּתֹנֶת, לְבֵית הֶפְשֵׁט פְּאֵרוֹ וְכֻתָּנְתּוֹ,

וְכָל רָץ כְּגִבּוֹר אֶל מְקוֹם הוֹן וּמַלְקוֹחַ, / וְשָׁם נִלְקְחָה נַפְשׁוֹ וְנֶפֶשׁ אֲשֶׁר אִתּוֹ,

וְכָל רָץ כְּמוֹ אַיָּל בְּרַגְלָיו אֱלֵי קִצּוֹ, / וּמָהִיר כְּמוֹ צִפּוֹר אֱלֵי פַח לְמַפַּלְתּוֹ,

וְכָל שַׂר הֲכִינוֹתָ, בְּבוֹא יוֹם פְּקֻדָּתוֹ / וְאֵידוֹ, לְנַפְשׁוֹ בֵּית פְּקֻדּוֹת, וְהִפְקַדְתּוֹ,

וְכָל נֶחֱלַף מִקּוֹל מְחוֹלוֹ וְצִיץ נִזְרוֹ – / צְוָחָה בְּפִי בִתּוֹ וְחֶבֶל בְּרֹאשׁ אִשׁתּוֹ,

וּמֻכֶּה – וְלֹא נָקַם בְּמַכֶּה, וְלֹא נֶעְנַשׁ / מְחִירֵי רְפוּאָתוֹ, וְלֹא נִתְּנָה שִׁבְתּוֹ,

וְיָקָר – וְלֹא הוֹקִיר פְּעָמָיו, וְלֹא רַגְלֵי / חֲיָלָיו, בְּלֶכְתּוֹ אֶל מְשַׂנְאוֹ, וְחִלַּלְתּוֹ,

וְהָלַךְ לְהָבִיא הוֹן וְהָמוֹן בְּתוֹךְ כִּיסוֹ – / וְנִשְׁלַל אֲשֶׁר הוֹלִיךְ בְּכִיסוֹ וְאַמְתַּחְתּוֹ

וְסָבַב כְּמוֹ רוֹכֵל בְּקִרְיָה לְהִשְׂתַּכֵּר – / וְעָזַב יְחִידָתוֹ בְּתוֹכָהּ וּמַרְכֻּלְתּוֹ,

וּמֶלֶךְ אֲשֶׁר גָּאָה רְמִיָּה – וְרִמּוּהוּ / שְֹרָפִים, וְהֶעְמִיקוּ שְׁחוּתוֹ, וְהִשְׁקַעְתּוֹ,

וְלֹא הוּא לְבַדּוֹ סָף, וְנִמְלַט הֲמוֹן עֻזּוֹ – / וְלֹא רַע מְעַט בָּא עַל הֲמוֹנוֹ וּמִשְׁפַּחְתּוֹ

בְּהָלְכָם לְעִיר אוֹיֵב, וְהֵם לֹא יְדַמּוּ כֵן, / לְשַׁלֵּח בְּשׂוֹנְאָם מַהֲלֻמּוֹ וּמַקַּבְתּוֹ,

וְסָרוּ לְהֵעָזֵר בְּשִׁלְחוֹ עֲלֵי אוֹיֵב – / וְנֶהְפַּךְ לְאוֹיֵב וַהֲפָכָם בְּמִשְׁלַחְתּוֹ.

וְחָתַךְ יְמִינוֹ מִכְּסִילוּת, בְּכַף יָדוֹ, / וְשָׁבְרָה בְּנֵי קַשְׁתּוֹ זְרוֹעוֹ, בְּאִוַּלְתּוֹ,

וְהָיוּ בְּיָדוֹ מַאֲכָלוֹת לְזַבֵּחַ / מְלָכִים, וְהַרְבּוֹת מַאֲכָלְתּוֹ וּמַכָּלְתּוֹ,

וְהָיָה מְפָרֵק בָּם סְלָעִים – וְקִצֵּץ בָּם / זְרוֹעָיו, וּפֵרַק אֶת צְלָעָיו וּמַפְרַקְתּוֹ,

וּמַקֵּל, בְּעַמִּים פֵּאֲרוּהוּ, וּמַטֵּה עֹז – / חֲלָשָׁם, וְנִשְׁמַע קוֹל חֲלוּשָׁה בְּמֶמְשַׁלְתּוֹ,

אֱלֹהִים הֱסִיתָם וַהֲמָמָם וְשִׁלַּח בָּם / מְאֵרָה, וְגָעַר בָּם – וְחָרוּ בְּמִגְעַרְתּוֹ,

וְכָכָה יְסַכֵּל אֵל אֲנָשִׁים לְנַשֵּׂא אִישׁ / אֲשֶׁר סָר לְמִשְׁמַעְתּוֹ וְנוֹכַח בְּתוֹכַחְתּוֹ

וְחָרֵד בְּאִמְרָתוֹ, אֲשֶׁר יַחֲרִיד בִּשְׁמוֹ / נְסִיכִים, וְיַפִּיל אֶת קְצִינִים בְּמַהְמָרְתּוֹ,

לְךָ יַעֲשֶׂה זִמְרָה בְּשִׂיחַ אֲשֶׁר נָשָׂא / בְּמֹאזְנָיו, וְשִׁיר נָעִים הֱבִיאוֹ בְּמִשְׁקָלְתּוֹ,

לְמַעַן הֲנִיאוֹתָ יְרִיבוֹ וְהִכְבַּדְתָּ / כְּבָלָיו – וְהִנְוֵיתָ יְדִידָךְ וְכִבַּדְתּוֹ.

אֲנִי אֶחֱטָא לָאֵל בְּאָמְרִי: כְּבָר הֵבִיא / פְּעָמַי בְּסוּגַר צַר, אֱלֹהַי, וּבִנְחֻשְׁתּוֹ,

וְסָגַר בְּעַד עֶזְרִי דְּלָתוֹת – וְהוּא פָתַח / בְּסוֹדוֹ לְעֶזְרָתִי, וְלֹא אֵדְעָה, דַּלְתּוֹ,

וְשָׂחַק מְשַׂנְאִי בִי בְּקָהָל – וְאַתָּה, אֵל, / בְּסֵתֶר וְלֹא נֵדַע, בְּפִי כֹל שְׂחוֹק שַׂמְתּוֹ,

אֲזַי אוֹהֲבִי עָטָה שְׂמָחוֹת כְּמִצְנֶפֶת / וְלָט צוֹרְרִי פָנָיו בְּבוּשָׁה בְּאַדַּרְתּוֹ

וְחָגַר יְדִידִי גִיל, וּמִלֵּא שְׂחוֹק פִּיהוּ, / וְשָׁב הַמְשַׂנְאִי בַכְּלִמּוֹת בְּמַחְגָּרְתּוֹ.

וְכָכָה תְּהִי גֵאוּת אֱלֹהִים וְעֹז מֶלֶךְ, / אֲשֶׁר נִכְנְעוּ גֵאֵי מְלָכִים לְמַמְלַכְתּוֹ.

וְיָמִים לְאִישׁ יַרְאוּ אֲשֶׁר לֹא יְדַמֵּהוּ / בְּחוֹלֵל זְמַן סוֹחֵר בְּעוֹלָם וְתוֹלַדְתּוֹ,

וְכַמָּה פְּלָאִים יֵשׁ לְאֵל חַי בְּלֵב יָמִים, / יְגַלֵּם – וְיִגָּלוּ הֲדָרוֹ וְתִפְאַרְתּוֹ.

יְבֹרַךְ אֲשֶׁר בָּרָא זְמַן, יַעֲשֶׂה עִנְיָן / בְּתֹכֶן, וְאַחֵר – לֹא בְּיָשְׁרוֹ וּמַתְכֻּנְתּוֹ,

וּפַעַם יְחַלֵּל חֹק וּמִשְׁמָר, וְאַחֶרֶת / יְקַיֵּם דְּבַר עוֹלָם בְּחֻקּוֹ וּמִשְׁמַרְתּוֹ!
