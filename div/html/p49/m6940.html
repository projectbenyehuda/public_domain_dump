אֲהַלֵּל אֲשֶׁר

[ופי א]לעאם אלמורך ͘ סאר ידיר עלי ענד אלשיך בן עבד אללה פי קרמונה ותט ͘ אפרא עלי אלקיאם ואלות ͘ יב בעד כאן אלתזמהא בן עבד אללה פנקצהא ונפד ͘ אלכ ͘ רוג ͘ אלי בלדה פוקע אלתגלב עלי אכת ͘ רה ומות אלשיך אלמד ͘ כור פקאל פי ד ͘ לך והי ארג ͘ וזה

ובשנה בתאריך הנ"ל (תת"ג, 1043) הלך ידיר אל הזקן בן עבד אללה בקרמונה ושניהם כרתו ברית להתקוממות ולהתנפלות ואחרי שקבל עליו בן עבד אללה הפר את התחיבותו והחליט לצאת אל ארצו והיה הנצחון על רובו (של צבאו) ומת הזקן הנזכר, ואמר (הנגיד) על ענין זה והוא "‘שיר חרוזי’

נְגִינָה

אֲהַלֵּל אֲשֶׁר אֵין לוֹ דְמוּת וּתְמוּנָה, / לְמַעַן פְּעֻלָּתוֹ אֲשֶׁר נֶאְמָנָה,

וְיַעַן בְּכָל עוֹנָה, אֲשֶׁר אָכִינָה / לְבָבִי לְדָרְשֵׁהוּ, בְּצַר לִי – עָנָה,

וְאֵימִים אֲשֶׁר אֶלְאֶה נְשׂא אָפוּנָה / וְאֵלֵךְ עֲלֵי דֶרֶךְ וְלֹא אָבִינָה

בְּעֵת אַשְׂמְאִילָה בוֹ וְעֵת אַיְמִינָה. / בְּאָמְרִי: תְּנָה לִּי, אֵל, כְּנַף הַיּוֹנָה,

5 וְאָעוּף אֱלֵי מִדְבָּר וְשָׁם אֶשְׁכֹּנָה – / וְאוֹמֵר: הֲיָדֶיךָ בְּאֵיד תִּרְפֶּינָה

וְלִבָּךְ בְּיוֹם פַּחַד וְעֶבְרָה פָּנָה / וְנָמַס, וְעוֹשֶׂה כַּאֲשֶׁר תַּעְשֶׂינָה

בְּנוֹת תּוֹר, אֲשֶׁר קוֹרֵא וְנֵץ תִּרְאֶינָה, / לְזֹאת תַּעֲרֹךְ שׁוּעַ לְאֵל כָּל עוֹנָה

וְתָקוּם, בְּכָל עֵבֶר וְעַל כָּל פִּנָּה, / וְתִצְעַק, בְּעֵינַיִם לְאֵל תִּשְׁעֶינָה,

וְתִלְעַג לְבוֹטְחִים עַל חֲנִית אוֹ צִנָּה! / עֲנִיתִיו: שְׁמַע מִלַּי וְלִי הַאְזִינָה:

10 בְּשָׁלשׁ, בְּאֵיד, אֶבְטַח וְאֶשָּׁעֵנָה: / תְּפִלָּה אֲקַדֵּם יוֹם קְרָב – רִאשׁוֹנָה,

וְשִׁיר לַאֲשֶׁר לִי יַעֲשֶׂה – אַחְרוֹנָה, / וְלֵבָב אֲשֶׁר יִשְׂחַק לְקוֹל הָרִנָּה,

כְּאִלּוּ אֲרִי רָעֵב בְּסֻכּוֹ עֻנָּה, / וְעֶרֶב עֲלֵי גוּרָיו מְצוּדָה לָנָה,

וְהִשְׁכִּים, וּמַרְאֵהוּ בְּאַף נִשְׁתַּנָּה, / וְיָצָא — וְטֶרֶף בֵּין צְלָעָיו חָנָה.



אֲסַפֵּר פְּעֻלַּת אֵל אֲשֶׁר עָשָׂה נָא / בְּצָרָה אֲשֶׁר רֹאשִׁי בְּיוֹם הִלְבִּינָה

15 וְקֶרֶב עֲצָמַי מַחֲלָה הִשְׁכִּינָה, / וּבָאתָה בְּמוֹרָשַׁי אֱלֵי קִרְבֶּנָה,

וְאָמְרוּ מְשַׂנְאִים: אֵין לְזֹאת תַּקָּנָה! / בְּעֵת קוּם אֲשֶׁר הֵפֵר בְּרִית נִתָּנָה

וְקָשַׁר, וְהִקִּיף רֹאשׁ, וְעָלַי בָּנָה, / וּבָזָה שְׁבוּעָתוֹ, וְאָלָה שִׁנָּה,

וְעָשָׂה בְּלֹא דַעַת וּמֵאֵין בִּינָה, / וְהֵבִיא קְרוֹב מַלְכִּי לְרִיב וּלְשִׂטְנָה

וְאָרְבוּ לְהָרַע כַּאֲרִי בִמְעוֹנָה, / אֲשֶׁר יָעֲצוּ אֶשְׁמַע וְלֹא אַאְמִינָה

20 עֲלֵי שָׂב אֲשֶׁר יָצָא שְׁמוֹ בֶאְמוּנָה, / בְּעָזְבוֹ אֱמוּנָתוֹ – וְלֹא בִתְלֻנָּה –

וְהֵפִיץ גְּדוּדָיו וַחֲיָלָיו מָנָה, / וְאַחַר לְבָבוֹ כַּעֲוִילִים זָנָה –

לְהִלָּחֲמוֹ בָאנוּ בְּחֶרֶב יוֹנָה, / בְּחֵלֶב נְסִיכֵי עַם וְדָם דֻּשָּׁנָה,

וְצַרְנוּ וְלָכַדְנוּ כְּרָךְ וּמְדִינָה / בְּעַם רַב וּמֶרְכָּבָה לְרִיב נָכוֹנָה,

וְקִנְאַת אֱיָלוּתִי בְּצָרִי עָשְׁנָה, / וְנָדוֹשׁ כְּמוֹ מַתְבֵּן בְּמוֹ מַדְמֵנָה,

25 וְנָפַל לְפִי חֶרֶב לְעֵת הַזִּקְנָה, / בְּיַד אִישׁ אֲשֶׁר הִכָּה וּבוֹ לֹא שָׁנָה.

לְנָפְלוֹ – בְּנוֹת עַמּוֹ בְּמַר תִּבְכֶּינָה / וְסוֹפְדוֹת כְּמוֹ תַנִּים וְכִבְנוֹת יַעְנָה

וּמִתְאַבְּלוֹת יַחַד וְנוֹשְׂאוֹת קִינָה, / וְקִרְיַת מְשׂוֹשׂוֹ נִשְׂאֲרָה כִמְלוּנָה,

וְהָיְתָה בְּמוֹת מַלְכָּהּ כְּמוֹ אַלְמָנָה, / וּבִזַּת הֲמוֹנָהּ אוֹהֲבַי אָכְלוּ נָא.

וְקָרוֹב אֲשֶׁר מָרַד וּמַלְכּוֹ הוֹנָה, / אֲשֶׁר מִכְּלוּא נִמְלַט וְלָרַע שָׁנָה,

30 שְׁפָטוֹ כְּמִפְעָלָיו אֲשֶׁר לוֹ קָנָה, / וְשַׁבְנוּ, כְּיוֹם אַרְנוֹן וְיוֹם מַתָּנָה,

בְּפָנִים מְאִירִים, יִדְמְיוּן לִלְבָנָה. / לְפוֹדֶה יְחִידָתִי בְּאֵין מַשְׁעֵנָה,

וְנוֹתֵן מְדִינוֹת צוֹרֲרִים מַתָּנָה / לְחַבָּל, וְנַפְשׁוֹתָם לְיָדוֹ אִנָּה,

אֲהוֹדֶה יְמֵי חַיַּי בְּשִׁיר וּנְגִינָה, / תְּחֻדַּש בְּפִי כָל אִישׁ – וְהִיא נוֹשָׁנָה,

וְאַחַר בְּלוֹתָהּ תִּהְיֶה לָּה עֶדְנָה, / תְּשׁוֹטֵט בְּמִזְרָח, תַּעֲבוֹר תֵּימָנָה

35 וְתֵרֵד אֱלֵי יָם, תַּעֲלֶה צָפוֹנָה, / כְּעֹפֶר עֲלֵי הָרֵי שְׂנִיר וַאְמָנָה.

אֱלֹהַי, בְּבוֹא צָרִי לְרִיב וּרְגִינָה – / עֲשֵׂה לִי כְּפָעְלֶךָ בְּזֹאת הַשָּׁנָה,

וְאוֹדֶה לְךָ לָעַד בְּגִיל וּרְנָנָה!
