רוֹעִי מֵרִבְעִי אֶשְׁקֹטָה

רוֹעִי מֵרִבְעִי אֶשְׁקֹטָה / וּגְבוּרוֹתֶיךָ אַבִּיטָה,

פִּי הוֹגֶה אֶת נוֹרְאוֹתֶיךָ / וּלְשׁוֹנִי נִסֶּיךָ בּוֹטָה.

עַל הָאָרֶץ וִיסוֹדֶיהָ / וּמְכוֹנָהּ יָדָךְ שָׁלְטָה,

תַּשְׁחִית זַיִת פָּרַח, וִיבוּל / גֶּפֶן, וּתְאֵנָה חָנָטָה.

אֵיכָה יֶחְטָא גֶבֶר יִרְאֶה / כָּל גֶּבֶר דַרְכּוֹ יָרָטָה –

הִסְגִּירוֹ אֵל אֶל עַוָּלִים, / עַל יַד רָשָׁע אוֹתוֹ רָטָה?

תִּנְדֹּף, כִּנְבֹל עָלֶה, נֶפֶשׁ / עַל פִּי הַנֵּבֶל פָּרָטָה.

דּוֹרֵךְ עַל גַּב אֶרֶץ וּכְעוֹ/טֶה אֶת בִּגְדּוֹ אוֹתָהּ עָטָה,

מוֹצִיא שֶׁמֶשׁ וּבְעֵת יִרְצֶה / מוֹסְדוֹת הֶהָרִים לִהֵטָה.

מֵנִיף הָרִים הָנֵף רוּחַ / שִׁבֹּלֶת עַל אֶרֶץ חִטָּה,

הַמַּרְקִיד אֶרֶץ כָּעֵגֶל / וּכְבֶן רֵימִים כֵּן שָׁאָטָה,

וּבְזַעְמוֹ נָעָה מַתְכֻּנְתָּהּ / וּכְהוֹלֵךְ עַל עָקֹב שָׁטָה.

שָׁמַעְתִּי – וַתִּרְגַּז בִּטְנִי / וּלְקוֹל רַעַשׁ אֶתְקוֹטָטָה,

מִדֵּי זָכְרִי מַה שֶּׁהָיָה / יִדֹּד לִבִּי וָאָנוּטָה.

קַמְתָּ לַעְרֹץ אֶרֶץ נֶשֶׁף / וּבְאִישׁוֹן לַיְלָה וַעְלָטָה,

הָפַכְתָּ תֵבֵל כִּקְעָרָה, / שַׂמְתָּ רוּם אֶרֶץ מִלְּמַטָּה.

זוֹכֵר אֵימַת הַלַּיְלָה הוּא / נַפְשׁוֹ בַּחַיִּים נָקָטָה.

אֵימָה רַבָּה עָצְמָה, עַד כִּי / גַּבְהוּת הָרָמִים הִמְעִיטָה.

פַּעְמַיִם שָׁלֹשׁ נָגַע בָּהּ / אֵל וּכְשִׁכּוֹר הִתְמוֹטָטָה.

חֶרְדַּת רַעַשׁ, כִּגְדוּד פּוֹשֵׁט / עַל עִיר, בִּלְבָבוֹת פָּשָׁטָה.

הִלְבִּינָה אֶת רֹאשׁ כָּל עַלְמָה / מֵרֹב יָמִים לֹא נִמְרָטָה.

עַָמּוּדֵי שָׁמַיִם רָפוּ / וִיסוּדַת אֶרֶץ נִשְׁמָטָה.

קִירוֹת נָשְׁקוּ זֶה לָזֶה, שׁוּר / גָּבוֹהַּ עַל אֶרֶץ נָטָה,

עֵת הִתְפַּלֵּץ מוֹסְדוֹתֶיהָ / רָעְדָה, כִּי מֵאֵל נִבָּטָה.

בִּנְיָה שׂוּמָה עַל שָׁת אַמִּיץ / מֵהִנָּתֵץ לֹא נִקְֶלָטָה.

חוֹמָה רָמָה עַל גַּב קַרְקַע / כַּחְבֹּט זֵיתִים כֵּן חֻבָּטָה.

יוֹנְקִים נָפְלוּ מֵחֵיק אִמָּם / מֵחִיל אוֹתָם לֹא לָקָטָה.

אֵם עַל בָּנִים חַתִּים, מֵרִפְ/יוֹן יָדַיִם, לֹא הִבִּיטָה,

נִשְׁתָּוָה בָהּ אֵשֶׁת חַיִל / עִם אִשָּׁה סוֹרֵרָה שׂוֹטָה.

עַל כָּל גַּגּוֹת שֶׁבֶר, כַּהְמוֹת / אֵם, עַל יָחִיד גּוּף שָׂרָטָה.

עַל שׁוֹכְנֵיהֶם בָּתִים נָפְלוּ, / עַל שָׁרְשֵׁיהֶם – אֶרֶז, שִׁטָּה.

כָּל יוֹשֵׁב הִתְגוֹלֵל עַל הָ/אָרֶץ כִּגְלֹל חֶרֶב לוּטָה;

כָּל עוֹמֵד קָרַס, כָּל יָצַע / מִטָּה נָפַל מֵעַל מִטָּה;

כָּל בַּת מֶלֶךְ בִּקְצֵה חֶדְרָה / פָּנִים מֵאֵימָה לָאָטָה;

כָּל לֵב הוֹגָה בִּיגוֹן חוֹלָה / מֵחֵבֶל זָכָר הִמְלִיטָה;

כָּל נֶפֶשׁ אִישׁ בֵּין מוֹרָשָׁיו / כִּסְפִינָה בֵין גַּלִּים שָׁטָה;

כָּל אִישׁ וָאִישׁ עָמַד נִבְהָל / וַיִּתְוַדֶּה אֵת שֶׁחָטָא,

עָנָה: "כֹּה יִשְׁפֹּט אֵל עֵדָה / עָנִי וָדָל לֹא שָׁפָטָה,

רָוְתָה, כָּשְׂתָה, שָׁמְנָה, עָבְתָה / וּבְאֵל הִשְׁמִינָהּ בָּעָטָה״.

אָמַרְנוּ בָּא קִצֵּנוּ, כִּי / יַד הַשְׁמֵד אֶרֶץ טֵאָטָה.

הָלְכָה מִמַּעֲרָב אֶל מִזְרָח, / אוֹרְחוֹתֶיהָ לֹא עִבֵּטָה.

הִשְׁקַתְנוּ מֵי אָבְדָן, וּנְזִיד / מָוֶת אוֹתָנוּ הִלְעִיטָה.

אַחַר כָּל זֹאת הִשְׁקִיט הָאֵל / אֶרֶץ, אַחַר כִּי קֻמָּטָה,

לָשׂוּם לָנוּ שֵׁם וּשְׁאֵרִית / וּלְהַחְיוֹת לָנוּ לִפְלֵיטָה.

חָגְרָה עֵדָה אַבְנֵט שִׂמְחָה / וּלְבוּשׁ חֶרְדָתָהּ פָּשָׁטָה,

כָּל הַלַּיְלָה הִרְבָּת רִנָּה / וּתְחִנָּה עַל כִּי נִמְלָטָה,

עַל כִּי יֶשַׁע וּמְעִיל צֶדֶק / יַד הָאֵל אוֹתָהּ יָעָטָה.

צוּר, יָד רָמָה כָזֹאת תָּמִיד / לִפְדּוֹת וּלְרַחֵם הוֹשִׁיטָה!

אֵל תָּלָה אֶרֶץ עַל מַיִם / וַעְלֵי תֹהוּ צָפוֹן נָטָה,

מִשָּׁמַיִם הִשְׁמַעְתָּ דִּין – / אֶרֶץ יָרְאָה גַּם שָׁקָטָה,

הִרְעַשְׁתָּה אֶרֶץ וּפְצַמְתָּהּ – / רַפֵּא שְׁבָרֶיהָ כִּי מָטָה!
