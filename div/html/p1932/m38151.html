הייתי וזמן הנעורים

בְּנֵי מִקְרָא בְּמִצְרַיִם


– – – הָיִיתִי וּזְמַן הַנְּעוּרִים מִכְּלֵי הַחֶבְרָה

אֶל כְּלֵי הַנְּדוֹד לֹא הוּרַק / וִימֵי הָעַלְמוּת לִשְׁמִי הַטּוֹב שֶׁמֶן תּוּרַק / וְעִתּוֹת הֶעָמָל רְחוֹקִים מִמֶּנִּי בַּל תִּקְרָב / כִּרְחוֹק מִזְרָח מִמַּעֲרָב / וְשַׁד יֵין רִקְחֵי הַשְּׂמָחוֹת מוֹצֶה וְיוֹנֵק / וַעֲסִיס רִמּוֹנֵי הַמְּנוּחָה לְהָרֵי הַבְּשָׂמִים סוֹחֵר וּמְזַנֵּק / וְכָל תַּעֲנוּגֵי הָעוֹלָם מִנַּפְשִׁי הַדְּשֵׁנָה / לֹא יֵלְכוּ אָנֶה וְאָנָה / וַאֲנִי עַל בִּרְכֵּי הַיַּלְדוּת וְהַשַּׁחֲרוּת מִשְׁתַּעְשֵׁעַ וְאָמוֹן / וְשׁוֹכֵב בְּחֵיק שַׁלְוַת הַזְּמָן בִּמְדִינַת נֹא אָמוֹן / הַמְּדִינָה הַמְּלֵאָה בִיקַר הַחֵן כָּרִמּוֹן / וְשֵׁמַע יָפְיָהּ נִשְׁמַע בְּקַצְוֵי הָעוֹלָם כַּפַּעֲמוֹן / הָעִיר הַמְּלֵאָה הַמְאֻשָּׁרָה בְּכָל חֵן בַּבִּנְיָן הַטּוֹב מִזְּמָן קַדְמוֹן / וְקֶרֶן כָּל הַמְּדִינוֹת לְפָנֶיהָ כָּפוּף כְּאַגְמוֹן / בַּעֲלַת הַמִּגְדָּל וְהָאַרְמוֹן / כְּגֹבַהּ הַר בָּשָׁן וְחֶרְמוֹן / וְהַדִּירוֹת הָרְחָבוֹת / וְקוֹרוֹתָם אֲרָזִים בַּזָּהָב וּבַשָּׁשַׁר מְשׁוּחוֹת / וּבְכָל מִינֵי אֲבָנִים מְפֻתָּחוֹת / וְאַבְנֵי בִנְיָנָהּ כְּאִלּוּ בְזֹהַר הָאוֹר טוּחוֹת / וִיסוֹדוֹתֶיהָ הַחֲמוּדוֹת כְּאִלּוּ הֵם עַל אַדְנֵי שֵׁשׁ מְיֻסָּדוֹת / וּמִכָּל חַצְרוֹתֵיהֶם לֹא נֶעְדָּר / בַּהַט וָשֵׁשׁ וְדָר / וְהַגַּנּוֹת וְהַפַּרְדֵּסִים / כַּמַּיִם לַיָּם מְכַסִּים / וּבִדְרָכֶיהָ הַגֶּשֶׁם יוֹם יוֹם יִגְבָּר / זֶרֶם מַיִם עָבָר / וּבְעֵת תְּבַקְשֶׁנּוּ / וְלֹא תִמְצָאֶנּוּ / וַיַעֲבֹר וְהִנֵּה אֵינֶנּוּ / כִּי תְעָלוֹת יְסוֹדוֹתֶיהָ בהדלוֹת בָּעִיר / וַיּוֹצִיאוּהוּ וַיַּנִּיחוּהוּ מִחוּץ לָעִיר / וְכָל צְמָחֶיהָ מָצְאוּ חֵן בְּעֵינַי / כַּאֲהָלִים נָטַע ה׳ / עַד הָיָה שְׁמָהּ בְּפִי שׁוֹכְנֶיהָ וְיוֹשְׁבֶיהָ / יִקָּרֵא חֶפְצִי בָהּ / וְסוֹחֲרֵי כָל הָעוֹלָם בָּהּ בָּחֲרוּ וְחָשְׁקוּ כִּי שֶׁפַע יַמִּים יִינָקוּ / וְכַמָּה אֳנִיּוֹת סוֹחֵר בְּלֵילָהּ וְיוֹמָהּ / מִמֶּרְחָק תָּבִיא לַחְמָהּ / וּבְכָל עֵת יְבוֹאוּהָ סְפִינוֹת מֵאֲרָם וּפְלִשְׁתִּים / וְצִים מִיַּד כִּתִּים / כִּי הַמֶּלֶךְ הַגָּדוֹל אֲלֶכְּסַנְדֵּר בְּחֶזְקָתוֹ יָרָה אַבְנֵי יְסוֹדוֹתֶיהָ / וּבְעָצְמָתוֹ הִצִּיב דְּלָתֶיהָ / חוֹמָתָהּ גְּבֹהָה וּבְצוּרָה / לְעִתּוֹת בַּצָּרָה / וְלָהּ הַחֵן וְהַתִּפְאָרָה / עַל צוֹר הַמַּעֲטִירָה / כִּי לָהּ וּבַעֲדָהּ אָמַר הָאֵל בְּחֶזְיוֹנוֹ: / ׳הֲתֵיטְבִי מִנּוֹא׳ / וְכַמָּה לְשָׁמְרָהּ בָּנוּ פִּסְגָה וּצְרִיחַ / חוֹמָה גְּבוֹהָה דְּלָתַיִם וּבְרִיחַ / וּבְכָל לַיְלָה לִהְיוֹתָהּ לְשׁוֹכְנֶיהָ שׁוֹמֶרֶת / סֹגֶרֶת וּמְסֻגֶּרֶת / וְזֶה הַשִּׁיר עֲשִׂיתִיו לְעֵד וּלְאוֹת עַל גָּדְלָהּ / מְסַפֵּר אֶחָד אָחוּז מֵחֲמֵשׁ הַמֵּאוֹת מִמַּהֲלָלָהּ.


לְנוֹא אָמוֹן פְּאֵר הוֹד וַעֲטָרָה / עֲלֵי כָל עִיר יְפֵפִיָּה יְקָרָה

וְלָהּ כָּל הַמְּדִינוֹת כַּשְּׁפָחוֹת / בְּטוּב מוֹשָׁב וְהִיא לָהֶן גְּבִירָה

וּמִי רָאָה בְּכָל עוֹלָם כְּמוֹתָהּ / בְּחוֹמָתָהּ אֲשֶׁר בָּהּ הִיא בְּצוּרָה

בְּבִנְיָנָהּ יְקָר לָקְחָה עֲלֵי כָל / מְדִינוֹת חֵן וְצוֹר הַמַּעֲטִירָה.


וּבְעוֹדִי בָהּ בֵּין חֲבֵרַי וְרֵעַי / וְדוֹדַי וּמְיֻדָּעַי / בְּגִילַי וְשַׁעֲשׁוּעַי / תָּמִיד לִרְגָעַי / לָקְחָה אָזְנִי שֶׁמֶץ דָּבָר / מֵאִישׁ הָיָה לִי נֶחְבָּר / כִּי בְמִצְרַיִם הָעִיר הַגְּדוֹלָה / גּוֹי טָהוֹר וְעַם סְגֻלָּה / דֶּגֶל יְקָר מַעֲלָתוֹ רָם וְנַעֲלָה / עַל כָּל קְהִלּוֹת הַגּוֹלָה / חָכָם וְנָבוֹן בְּכָל דִּבְרֵי נְבוּאָה וְחֶזְיוֹן / וְיוֹדֵעַ דַּעַת עֶלְיוֹן / וְנַפְשׁוֹ בִּלְבָד חוֹמְדָה וּמִתְאַוָּה / לִשְׁמֹר הַתּוֹרָה וְהַמִּצְוָה / וּשְׁמוֹ נוֹדָע בֵּין הָעַמִּים וְנִקְרָא: / עַם בְּנֵי מִקְרָא / הַמַּאֲמִין בַּה׳ כִּי הוּא אֶחָד וְאֵין עוֹד מִלְבַדּוֹ / וּבְמֹשֶׁה עַבְדּוֹ / וּבִשְׁאָר הַנְּבִיאִים הַנִּבְחָרִים / הַנִּזְכָּרִים בְּאַרְבָּעָה וְעֶשְׂרִים / וּדְבַר ה׳ אֲשֶׁר הוֹרִיד בְּתוֹרָתוֹ / לֹא יַחֲלִיפֶנּוּ וְלֹא יָמִיר אוֹתוֹ / הִיא הַתּוֹרָה אֲשֶׁר עַל יַד מֹשֶׁה נְתוּנָה / וְאֵין זוּלָתָהּ לֹא תַלְמוּד וְלֹא מִשְׁנָה / וּבְדַעְתּוֹ נְכוֹנָה / כִּי כָל מִצְוָה זוּלַת מִצְווֹת הַתּוֹרָה שֶׁבִּכְתָב הַחֲמוּדָה / מִצְוַת אֲנָשִׁים מְלֻמָּדָה.


וּכְשָׁמְעִי זֶה הַדָּבָר מִמְּקוֹמִי קַמְתִּי / וְלִבִּי לַנְּסִיעָה שַׂמְתִּי / וְלָשׁוּר אֲשֶׁר שָׁמַעְתִּי / חַשְׁתִּי וְלֹא הִתְמַהְמַהְתִּי / כִּי אָמַרְתִּי לְנַפְשִׁי אוּלַי בְלֶכְתִּי אֶחֱזֶה / אֶת הַמַּרְאֶה הַגָּדוֹל הַזֶּה / וְהָעָם אֲשֶׁר בָּחַר בְּכִשְׁרוֹן וּתְמִימוּת / אֵלְכָה וְאֶרְאֶנּוּ בְטֶרֶם אָמוּת / וְנָסַעְתִּי וְעָזַבְתִּי אֶת אַרְצִי / וְרָכַבְתִּי עַל צִי / לְהַגִּיעַ לְחֶפְצִי / בְּמָשׂוֹשׂ וְעֶלְצִי / וָאֶמְצְאָה עִיר עַל נְהַר גַּן עֵדֶן מוֹשָׁבָהּ / וְכָל חֵן וְיֹפִי וִיקָר נִמְצָא בָהּ / וַיְהִי כַאֲשֶׁר שָׁמְעוּ עָלֶיהָ אָזְנָי / בִּנְטָעֶיהָ וּמְגָדֶיהָ כְּגַן ה׳ / וַאֲנָשֶׁיהָ כְּכוֹכְבֵי הַשָּׁמַיִם / וְהָעִיר רַחֲבַת יָדָיִם / וּבָתֶּיהָ עֲרֵבִים / גְּבוֹהִים וּרְחָבִים / וּבָהּ שִׁירֵי עֲגָבִים / לְבַב שׁוֹמְעֵיהֶם מְשִׁיבִים / וְגַנּוֹת טוֹבִים / נְפָשׁוֹת מְשִׁיבִים / עַל הַיָּם קְצוּבִים / וּבוֹרוֹת חֲצוּבִים / וּבְשׁוּרִי גָדְלָהּ וְרוֹב הֶהָמוֹן אֲשֶׁר בָּהּ מִתְחַבֵּר / נִפְעַמְתִּי וְלֹא אֲדַבֵּר / וְדִבַּרְתִּי לְלִבִּי הַנִּפְעָם: / זֹאת הָעִיר רַבָּתִי עָם / עַל זֹאת יֵאוֹת תֹּאמַרְנָה בַעֲדָהּ כָּל הַלְּשׁוֹנוֹת / רַבָּתִי בַגּוֹיִם שָׂרָתִי בַמְּדִינוֹת / וָאֶפְתְּחָה פִי בְטוּב דִּבּוּר וָזֶמֶר / וָאֶשָּׂא מְשָׁלִי וַאֹמַר:


לְמִצְרַיִם עֲלֵי כָל עִיר גְּדֻלָּה

בְּהוֹד מַלְכָּהּ אֲשֶׁר בָּהּ רָם וְנַעְלָה

וְחֵילָהּ כַּעֲפַר אֶרֶץ בְּמִסְפָּר

וְרוֹב עַמָּהּ כְּכוֹכָבִים בְּמַעְלָה

וְאִם הִיא אָז לְקֵדָר בֵּית מְלוּכָה

לְפַרְעֹה בָּהּ יְקָר הָיְתָה תְּחִלָּה

בְּכָל עֵת בָּהּ כְּבוֹד מַלְכוּת יְחֻדָּשׁ

אֲשֶׁר נִשָּׂא וְרָם עַל כָּל תְּהִלָּה.


וְאַחַר הִשְׂתַּכַּלְתִּי לְעֹצֶם הָעִיר / רוּחִי וְלִבִּי הָאֵל הֵעִיר / לַחֲקֹר אֶת הַדָּבָר אֲשֶׁר בְּעִירִי שָׁמַעְתִּי / וְלַעֲמֹד עַל סוֹד הָעִנְיָן אֲשֶׁר בַּעֲבוּרוֹ מִנָּוִי נָסַעְתִּי / וָאֶשְׁאַל בְּעַד גְּבוּל הָעָם הַנִּזְכָּר / אֲשֶׁר רֵיחַ שְׁמָנָיו שָׁלַח לִי הַזְּמָן לְאֶשְׁכָּר / וַיּוֹרוּנִי אֲנָשִׁים עָלָיו / וַיַּגִּיעוּנִי אֵלָיו / וָאֶמְצָא קָהָל בְּפָנָיו שֹׂבַע שְׂמָחוֹת / יָגוֹן מִלְּבַב כָּל אִישׁ דּוֹחוֹת / וּבָתִים פְּתוּחִים לַגֵּר וְלַתּוֹשָׁב / וְהוּא לְבָדָד יִשְׁכֹּן וּבַגּוֹיִם לֹא יִתְחַשָּׁב / וְאִישׁוֹן עֵין דְּרִישָׁתוֹ לַכָּתוּב בְּפוּךְ הָאֱמֶת נִקְרָע / תָּם וְיָשָׁר וְסָר מֵרָע / וּבְמִלְחֶמֶת הַתּוֹרָה אֲשֶׁר לֵב וְנֶפֶשׁ מְשִׁיבָה / אֶחָד יִרְדֹּף אֶלֶף וּשְׁנַיִם יָנִיסוּ רְבָבָה / וּמֵרֹב תֹּקֶף מִלּוּלָיו / יִירְאוּ מִגֶּשֶׁת אֵלָיו / וּבִקְרָב מְלַאכְתָּהּ הָעֲצוּמָה / כֻּלָּם אֲחוּזֵי חֶרֶב מְלֻמְּדֵי מִלְחָמָה / וּמִפַּחְדָּם לִתְעוֹת בְּדֶרֶךְ הָעִיּוּן יִתְהַלְּכוּ / אִישׁ חַרְבּוֹ עַל יְרֵכוֹ / רוּמְמוֹת אֵל בִּגְרוֹנָם / וּלְהָשִׁיב דָּבָר לְשׂוֹנְאֵיהֶם / בְּדִבְרֵי תוֹרָה שָׁנְנוּ לְשׁוֹנָם.


וּבִרְאוֹתִי זֶה הָעָם הַנִּכְבָּד יָדַעְתִּי / כִּי לְעֵץ חַיִּים נָגַעְתִּי / וּבְלִבִּי אֲנִי אָמַרְתִּי / כִּי לֹא לָרִיק יָגַעְתִּי / וְעַתָּה יָדַעְתִּי / כִּי כָל אֲשֶׁר שָׁמַעְתִּי / עַל הָעָם הַזֶּה בְאַרְצִי / וְהִנֵּה לֹא הֻגַּד לִי הַחֵצִי / וָאֶקֹּד וָאֶשְׁתַּחֲוֶה עַל פָּנָי / וָאֲבָרֵךְ אֶת ה׳ / אֲשֶׁר מִתּוֹךְ נָוִי וּמְעוֹנִי / בְּחַסְדּוֹ וּבְטוּבוֹ הִנְחַנִי / לְמַלֹּאוֹת חֶפְצִי וּרְצוֹנִי / בְּעַם אֲשֶׁר אֵין לוֹ בַזְּמָן שֵׁנִי / וַיִּשְׁאָלוּנִי הָעָם וַיֹּאמְרוּ לִי: ׳ה׳ אֱלֹהֶיךָ יִרְצֶךָ / מַה מְּלַאכְתְּךָ וּמֵאַיִן תָּבֹא וּמָה אַרְצֶךָ / וַאֲנִי נִכְלַמְתִּי מֵהֶם כִּי טַעַם הַגֵּרוּת בְּפִי מַר. וָאֶשָּׂא מְשָׁלִי וַאֹמַר:


בְּנוֹא אָמוֹן יְלָדוּנִי וְשָׁרְשִׁי / בְּאֶרֶץ מַעֲרָב מֵעִיר סְפָרַד

וְהַשִּׁיר לִי מְלֶאכֶת מִנְּעוּרָי / וְלִבִּי לוֹ מְאֹד יֶהְמֶה וְיֶחְרָד

שְׁקַלְתִּיו מִבְּלִי חֶסְרוֹן וְיִתְרוֹן / וּמִלֵּאתִיו מְחֻבָּר גַּם מְפֹרָד

וְרָצָה לַעֲבֹד אוֹתִי לְבַדִּי / וְגַם פָּשַׁע בְּזוּלָתִי וּמָרָד

וְלֹא שָׁקַט עֲדֵי עָלָה בְסֻלַּם / חֲמוּדוֹתָיו וְדִבֶּר בִּי וְיָרָד.


וְכִשְׁמֹעַ הָעָם עֵרֶךְ שִׁירִי אֲשֶׁר עָרַכְתִּי / יָדְעוּ אַרְצִי וּמְלַאכְתִּי / וְגָדְלָה בְּעֵינָם מַעֲלָתִי / וְהִסְכִּימוּ עַל אַהֲבָתִי. – – –
