שְׁנֵי הַחֲתָנִים

בְִּעִיר אַחַת הָיָה פַּרְנַס קָהָל, אִישׁ נִכְבָּד, גָּדוֹל בְּתוֹרָה וּבַעֲשִׁירוּת, כְּמוֹ שֶׁנֶּאֱמַר: תּוֹרָה וּגְדֻלָּה בְּמָקוֹם אֶחָד. וְהוּא וְאִשְׁתּוֹ אֲנָשִׁים זְקֵנִים. וְהָיְתָה לָהֶם בַּת יְחִידָה, שֶׁנּוֹתְרָה לָהֶם מִכַּמָּה בָּנִים שֶׁמֵּתוּ, נַעֲרָה יָפָה וַחֲכָמָה וּמֻשְׁלֶמֶת בְּכָל הַמַּעֲלוֹת.

פַּעַם לְאַחַר שֵׁנָה שֶׁל שַׁבָּת, בְּשֶׁבֶת הַפַּרְנָס וְהַפַּרְנָסִית לְבַדָּם בַּחֶדֶר, כִּי בִּתָּם הָלְכָה אֵצֶל חֲבֶרְתָּהּ בַּת שְׁכֵנָם, אָמַר הַפַּרְנָס לְאִשְׁתּוֹ: הִנֵּה זָקַנּוּ, לֹא נֵדַע יוֹם מוֹתֵנוּ – מַה נַּעֲשֶׂה לְבִתֵּנוּ שֶׁהִגִּיעָה לְפִרְקָהּ? לְמִי נַשִּׂיא אוֹתָהּ וְהִיא רְאוּיָה לְחָתָן לְפִי מַעֲלוֹתֶיהָ?

אָמְרָה הַפַּרְנָסִית: בֶּאֱמֶת הִגִיעָה הַשָּׁעָה וְיֵשׁ לְעַיֵּן בַּדָּבָר. הֵשִׁיב: עִיַּנְתִּי בַּדָּבָר עַיֵּן הֵיטֵב וּבָאתִי לִכְלַל דֵעָה, שֶׁנַּשִּׂיא אוֹתָהּ לְבֶן אֲחוֹתִי, שֶׁהוּא עָשִׁיר וּמִדּוֹתָיו נָאוֹת. אַָּמְרָה הַפַּרְנָסִית: אִם נִגְזַר שֶׁנַּשִּׂיא אוֹתָהּ לְאִישׁ קָרוֹב לָנוּ, נִתֶּן לָהּ אֶת בֶּן אָחִי, שֶׁאָמְנָם עָנִי הוּא, אֲבָל חֲכַם־לֵב וְשׁוֹקֵד עַל לִמּוּדוֹ.

וְהַפַּרְנָס עָמַד עַל דַּעְתּוֹ: בֶּן אֲחוֹתִי טוֹב מִמֶּנּוּ, שֶׁהוּא גַם בֶּן תּוֹרָה וְגַם עָשִׁיר, וּבִתֵּנוּ תֵשֵב אֶצְלוֹ בְּצֵל הַחָכְמָה וּבְצֵל הַכֶּסֶף. הוּא בְּשֶׁלּוֹ וְהִיא בְּשֶׁלָּהּ, וְלֹא הִכְרִיעוּ בַּדָּבָר.

אַחֲרֵי יָמִים אֲחָדִים עָשָׂה הַפַּרְנָס סְעֻדָּה גְדוֹלָה וְקָרָא אֵלֶיהָ אֶת יְדִידָיו טוֹבֵי הָעִיר, אֲנָשִׁים חֲכָמִים וְיִרְאֵי הַשֵּׁם, וְגַם לִשְׁנֵי הַבַּחוּרִים קְרוֹבֵיהֶם קָרָא לִשְׂמֹחַ עִמָּהֶם. הָיוּ הַקְּרוּאִים מַרְבִּים שִׂיחָה עִם הַבַּחוּרִים, שׁוֹאֲלִים וּמְשִׁיבִים וְנֶהֱנִים מִטּוּב טַעְמָם וְדַעְתָּם.

כְּתֹם הַסְּעֻדָּה בִּקֵּשׁ הַפַּרְנָס מִן הַבַּחוּרִים לָשׁוּב לְבָתֵּיהֶם, כִּי יֶשׁ לוֹ לְהִוָּעֵץ בְּעִנְיָן חָשׁוּב עִם אַנְשֵׁי סוֹדוֹ. הַבַּחוּרִים אָמְרוּ שָׁלוֹם לַכֹּל וְהָלְכוּ לָהֶם. אָז שִׁלַּח הַפַּרְנָס גַּם אֶת בִּתּוֹ אֶל חֶדְרָהּ, אַחַר פָּנָה אֶל אוֹרְחָיו בַּדְּבָרִים הָאֵלֶּה: הִנֵּה זָקַנּוּ, אֲנִי וְאִשְׁתִּי, בָּאנוּ בַּיָּמִים, וְאַדִּיר חֶפְצֵנוּ לְהַשִּׂיא אֶת בִּתֵּנוּ לְאִישׁ. אֲבָל אֵין אָנוּ, אֲנִי וְאִשְׁתִּי, יְכוֹלִים לָבוֹא לִידֵי הִשְׁתַּוּוּת: אֲנִי רוֹצֶה לִתְּנָהּ לְבֶן אֲחוֹתִי, וְהִיא רוֹצָה בְּבֶן אָחִיהָ. הִנֵּה יְדַעְתֶּם אֶת שְׁנֵי הַבַּחוּרִים, שְׁמַעְתֶּם שִׁיחָם וּרְאִיתֶם מִנְהָגָם – שִׁפְטוּ אַתֶּם בֵּינֵינוּ, כִּי אֲנָשִׁים חֲכָמִים אַתֶּם וּבַעֲלֵי נִסָּיוֹן, וּמוֹצָא פִּיכֶם נִשְׁמֹר.

נוֹעֲצוּ הָאֲנָשִׁים שָׁעָה אֲרֻכָּה וְלַסּוֹף הוֹדִיעַ הַגָּדוֹל שֶׁבָּהֶם אֶת דַּעְתָּם הַמֻּסְכֶּמֶת: שְׁנֵי הַבַּחוּרִים כְּאֶחָד טוֹבִים הֵמָּה, וְכָל אֶחָד מִן הַשְּׁנַיִם בְּלֹא סָפֵק רָאוּי שֶׁתִּנָּתֵן לוֹ הַנַּעֲרָה הַמֻּשְׁלֶמֶת, וְאֵין בְּיָדֵנוּ לְהַכְרִיעַ לֹא כְּלַפֵּי זֶה וְלֹא כְּלַפֵּי זֶה. עַל כֵּן שָׁקַלְנוּ יָפֶה בְּדַעְתֵּנוּ, וּרְאֵה זֶה מָצָאנוּ: יֻתַּן לְכָל אֶחָד מִן הַבַּחוּרִים מָאתַיִם אֲדֻמִּים וְיֵלְכוּ שְׁנֵיהֶם לְמֶרְחַקִּים לִקְנוֹת בַּכֶּסֶף סְחוֹרָה, כָּל אֶחָד לְפִי אֹמֶד דַעְתּוֹ, וּמִי שֶׁיֵּיטִיב לִקְנוֹת, לוֹ הוֹכִיחַ הַשֵּׁם אֶת הַנַּעֲרָה. בְּזֹאת יִבָּחֵנוּ.

מָצְאָה הָעֵצָה חֵן בְּעֵינֵי הַפַּרְנָס וְהַפַּרְנָסִית וְהוֹדוּ לְבַעֲלֵי הָעֵצָה. לְמָחָר קָרָא הַפַּרְנָס לַבַּחוּרִים וְאָמַר לָהֶם: הִנֵּה אֲנִי נוֹתֵן לְכָל אֶחָד מִכֶּם מֵהוֹנִי מָאתַיִם אֲדֻמִּים, וְאַתֶּם הָכִינוּ עַצְמְכֶם לְדֶרֶךְ רְחוֹקָה לִקְנוֹת בַּכֶּסֶף סְחוֹרָה, וּמִי שֶׁיֵּיטִיב לִקְנוֹת לוֹ תִּהְיֶה בִּתִּי לְאִשָּׁה. לְכוּ לְשָׁלוֹם וְהַשֵּׁם יַצְלִיחַ דַרְכְּכֶם.

יָצְאוּ שְׁנֵיהֶם לַדֶּרֶךְ וְכָל אֶחָד חוֹשֵׁב בְּלִבּוֹ: בְּחֶלְקִי תִּפֹּל הַנַּעֲרָה הַחֲשׁוּבָה.

הָלְכוּ הַבַּחוּרִים יָמִים רַבִּים עַד שֶׁהִגִּיעוּ לְעִיר גְדוֹלָה מְשֻׁפַּעַת בִּסְחוֹרָה מִכָּל הַמִּינִים. עָבְרוּ בַּשּׁוּק בֵּין הַחֲנֻיּוֹת וְהִסְתַּכְּלוּ יָפֶה בְּכָל הַסְּחוֹרוֹת, וְלַסּוֹף קָנָה בֶּן אֲחוֹת הַפַּרְנָס בַּדִּים טוֹבִים, אֵטוּן וְצֶמֶר וָמֶשִׁי, שֶׁמִּלְאוּ שַׂק גָּדוֹל, וּבֶן אֲחִי הַפַּרְנָסִית לֹא קָנָה אֶלָּא אֲבָנִים טוֹבוֹת, שֶׁדַּי הָיָה לָהֶן בְּכִיס קָטָן.

בִּהְיוֹתָם בַּדֶּרֶךְ לָשׁוּב סָרוּ לְבֵית מָלוֹן

לָלוּן שָׁם הַלָּיְלָה. וְהָיוּ שָׁם אֲנָשִׁים פּוֹחֲזִים וְרֵיקִים, לֵצָנִים וְגַנָּבִים, וּבָאוּ אֶל הַבַּחוּרִים בְּכָל מִינֵי תַּחְבּוּלוֹת וְאוֹנָאוֹת וְלֹא הִנִּיחוּ לָהֶם עַד שֶׁנּוֹדַע לָהֶם מָה הַסְּחוֹרָה שֶׁהֵם מוֹבִילִים עִמָּהֶם. וּבַחֲצִי הַלַּיְלָה, כַּאֲשֶׁר נִרְדְמוּ הַבַּחוּרִים, בִּהְיוֹתָם עֲיֵפִים וִיגֵעִים מִן הַדֶּרֶךְ, בָּאוּ הַגַּנָּבִים אֶל חֶדְרָם וְלָקְחוּ אֶת כִּיס הָאֲבָנִים הַטּוֹבוֹת, וּמִן הַשַּׂק הַגָּדוֹל לֹא לָקְחוּ אֶלָּא מְעַט מִן הַסְּחוֹרָה אֲשֶׁר לְמַעְלָה בְּפִי הַשָּׂק. כְּשֶׁהֵקִיצוּ הַבַּחוּרִים מִשְּׁנָתָם יָדְעוּ אֵת אֲשֶׁר נַעֲשָׂה לָהֶם. וּבַעַל הָאֲבָנִים הַטּוֹבוֹת נִצְטַעֵר צַעַר רַב, כִּי מְאוּמָה אֵין עוֹד בְּיָדוֹ שֶׁיּוֹלִיךְ אֶל הַפַּרְנָס. וְאוּלָם הַבָּחוּר הַשֵּׁנִי שָׂמַח בְּחֶלְקוֹ, כִּי רֹב הַסְּחוֹרָה נִשְׁאֲרָה בְּשַׂקּוֹ. אָמַר בַּעַל הַסְּחוֹרָה אֶל חֲבֵרוֹ, וְלֹא הִסְתִּיר שִׂמְחָתוֹ: בּוֹא, נְמַהֵר לָשׁוּב אֶל עִירֵנוּ וְאֶל בֵּיתֵנוּ, כִּי טוֹב לָנוּ שָׁם מִפֹּה.

הִסְתַּכֵּל בּוֹ הָאַחֵר בְּעַיִן בּוֹחֶנֶת וְאָמַר: לֹא אֵלֵךְ עִמְּךָ שֶׁאֵין בֵּינֵינוּ כָּל צַד שֶׁל שִׁוְיוֹן.

חָזַר הַבָּחוּר עִם שְׁאֵרִית הַסְּחוֹרָה, וּכְשֶׁבָּא לְעִירוֹ שְׁאֵלוּהוּ הָאֲנָשִׁים: הֵיכָן חֲבֵרְךָ?

אָמַר: לֹא יָדַעְתִּי, הֲשׁוֹמֵר חֲבֵרִי אָנֹכִי? וְלֹא סִּפֵּר מֶה עָלְתָה לוֹ וְלָמָּה נִפְרַד מִמֶּנּוּ.

אָמְרוּ לוֹ: אִם הַבָּחוּר הַשֵּׁנִי לֹא יָבוֹא לְמוֹעֵד חֲצִי שָׁנָה תִּנָּתֵן לְךָ הַנַּעֲרָה לְאִשָּׁה.

שָׂמַח הַבָּחוּר וְשָׂמַח דוֹדוֹ, וְהַפַּרְנָסִית הִצְטַעֲרָה עַל בֶּן אָחִיהָ, אֲבָל בְּלִבָּהּ חָשְׁבָה: עוֹד לָנוּ מוֹעֵד חֲצִי שָׁנָה, אוּלַי יִרְצֶה הַשֵּׁם וְיַחֲזִירֵהוּ הַבָּיְתְָה. לְיוֹם זֶה אֲקַוֶּה.

וְהַבָּחוּר הַהוּא הָלַךְ לְדַרְכּוֹ וּבָא לְיַעַר גָדוֹל. הָלַךְ בַּיַּעַר שְׁלשָׁה יָמִים וּשְׁלשָׁה לֵילוֹת. צֵידָתוֹ כָּלְתָה וְהָרָעָב הֱצִיקוֹ. וְהִנֵּה רָאָה אִילָן עוֹשֶׂה תַּפּוּחִים נֶחְמָדִים לְמַרְאֶה. קָטַף אֶחָד מֵהֶם וַאֲכָלוֹ. אַךְ כִּלָּה לֶאֱכֹל אֶת הַתַּפּוּחַ וְהִנֵּה הוּא מְצֹרָע כַּשֶּׁלֶג. נִבְהַל מְאֹד וּבְמַר נַפְשוֹ הָלַךְ הָלְאָה. וְהִנֵּה עוֹד אִילָן עוֹשֶׂה תַּפּוּחִים כָּרִאשׁוֹן. אָמַר: אֹכַל גַּם מִזֶּה וְאָמוּת וְיָבֹא הַקֵּץ לְיִסּוּרָי. קָטַף תַּפּוּחַ וַאֲכָלוֹ. מִיָּד נִתְרַפֵּא מִצָּרַעְתּוֹ, וְגַם רַעֲבוֹנוֹ עָבַר כְּלֹא הָיָה. הוֹדָה הַבָּחוּר לְהַשֵּׁם חַסְדוֹ. חָזַר וְלָקַח מִתַּפּוּחֵי הַצָּרַעַת וְנָתַן בְּתַרְמִילוֹ, אַחַר הָלַךְ אֶל עֵץ הַמַּרְפֵּא וְלָקַח מִפִּרְיוֹ. אָמַר: דָּבָר נִפְלָא לָמַדְתִּי בַּיַּעַר הַזֶּה. אוּלַי יִהְיוּ לִי הַפֵּרוֹת הָאֵלֶּה לְתוֹעֶלֶת.

יָמִים רַבִּים נָדַד מִמָּקוֹם לְמָקוֹם, עַד כִּי בָּא אֶל עִיר גְדֹולָה, רָאָה אֶת אֲנָשֶׁיהָ מִתְהַלְּכִים כַּאֲבֵלִים כֻּלָּם. שְׁאֵלָם לְפֵשֶׁר עִצְּבוֹנָם. אָמַר לוֹ אֶחָד מֵהֶם: יֵשׁ לָנוּ מֶלֶךְ חָסִיד וְנִצְטָרַע, יְרַחֲמֵהוּ הַשֵּׁם. כְּבָר בָּאוּ כָּל רוֹפְאֵי מִזְרָח וּמַעֲרָב וְכֻלָּם נוֹאֲשׁוּ מֵהוֹשִׁיעוֹ. עַל כֵּן אָנוּ מִצְטַעֲרִים וּבוֹכִים כָּל הַיָּמִים.

אָמַר הַבָּחוּר: הֲבִיאֵנִי אֶל הַמֶּלֶךְ. אוּלַי בְּעֶזְרַת הַשֵּׁם אוּכַל לְהָבִיא לוֹ מַרְפֵּא.

הֵשִׁיב הָאִישׁ: אֵינִי מַאֲמִין שֶׁרְפוּאָתוֹ בִּידֵי אָדָם. אֲבָל אִם רְצוֹנְךָ לֵילֵךְ אֵלָיו, אֲבִיאֲךָ שָׁמָּה.

בָּא הַבָּחוּר וְנָפַל לְרַגְלֵי הַמֶּלֶךְ הַחוֹלֶה וְאָמַר: יִשְׂגֶּה שְׁלוֹמְךָ, אֲדוֹנִי הַמֶּלֶךְ.

דִּבֶּר אֵלָיו הַמֶּלֶךְ מִתַּחַת לַמַּסְוֶה אֲשֶׁר עַל פָּנָיו הַמְּצֹרָעִים: הֲיֵּשׁ לְאֵל יָדְךָ לְהוֹשִׁיעֵנִי, אַחֲרֵי אֲשֶׁר נִלְאוּ כָּל הָרוֹפְאִים לְהוֹשִׁיעַ?

עָנָה הַבָּחוּר וְאָמַר: בְּעֶזְרַת הַשֵּׁם, אֲנַסֶּה.

הָלַךְ לְבֵית־מִרְקַחַת סָמוּךְ. חָתַךְ חֲתִיכָה מִתַּפּוּחַ הַצָּרַעַת וּמְשָׁחָהּ בִּדְבַשׁ וּנְתָנָהּ לַמֶּלֶךְ. וַיְהִי אַךְ אֲכָלָהּ הַמֶּלֶךְ גָבְרָה צָרַעְתּוֹ וְגָדַל כְּאֵבוֹ.

אַךְ הַבָּחוּר אָמַר לוֹ: אַחֲרֵי אֲשֶׁר הוּרַע לְךָ יָשׁוּב וְיִיטַב לָךְ.

הָלַךְ שֵׁנִית לְבֵית־הַמִּרְקַחַת, חָתַךְ חֲתִיכָה מִתַּפּוּחַ הַמַּרְפֵּא וּמְשָׁחָהּ בִּדְבַשׁ וֶהֱבִיאָהּ לַמֶּלֶךְ. וַיְהִי אַךְ אֲכָלָהּ אוֹרוּ עַינָיו וְצָרַעְתּוֹ עָבְרָה וּבְשָׂרוֹ כִּבְשַׂר נַעַר קָטֹן.

שָׂמַח הַמֶּלֶךְ וְכָל אַנְשֵׁי הָעִיר אֶבְלָם נֶהְפַּךְ לְשָׂשׂוֹן.

וְהַמֶּלֶךְ קָרָא לַבָּחוּר בַּסֵּתֶר וּשְׁאֵלוֹ: מַה חֶפְצְךָ, רוֹפֵא מַפְלִיא לַעֲשׂוֹת? עַד חֲצִי הַמַּלְכוּת וְיִנָּתֶן לָךְ.

אָמַר לוֹ הַבָּחוּר: בִּמְדִינַת מַלְכוּתְךָ יֶשְׁנָהּ עִיר פְּלוֹנִית וּבָהּ יָגוּרוּ אָבִי וְאִמִּי הָעֲנִיּים. אִם עָלֶיךָ טוֹב, שַׁלְּחֵנִי אֶל עִירִי וְתֵן לִי דֵי כַּלְכֵּל שֵׂיבָתָם.

אָמַר לוֹ הַמֶּלֶךְ: רְאֵה נְתַתִּיךָ שַׂר הָעִיר וּמוֹשְׁלָהּ.

וְהַמֶּלֶךְ נָתַן לוֹ אֶת טַבַּעְתּוֹ וְשָׂם רְבִיד־זָהָב עַל צַוָּארוֹ וְאַרְבַּע מֵאוֹת פָּרָשִׁים נָתַן לוֹ לְלַוּוֹתוֹ בַּדֶּרֶךְ.

כְּשֶׁשָּׁמְעוּ אַנְשֵׁי הָעִיר, כִּי שַׂר צָעִיר עוֹמֵד לָבוֹא הָעִירָה, יָצְאוּ לִקְרָאתוֹ בְּכָבוֹד גָדוֹל.

וְהַיְּהוּדִים שָׁלְחוּ אֶת הַפַּרְנָס לָלֶכֶת לִפְנֵיהֶם וּלְהָבִיא אֶת מִנְחָתָם לַשָּׂר: כְּלֵי־זָהָב נָאִים שֶׁהֵכִינוּ בִּשְׁבִילוֹ אֵצֶל צוֹרֵף אָמָּן.

וְכֵן דִּבֵּר הַפַּרְנָס בְּאָזְנֵי הַשַּׂר: אֲדוֹנִי הַשַּׂר רַב הַחֶסֶד! יְהוּדֵי עִירֵנוּ מְקַדְּמִים פָּנֶיךָ בִּבְרָכָה וּמְחַלִּים פָּנֶיךָ כִּי תּוֹאִיל לָקַחַת מֵהֶם אֶת הַמַּתָּת הַמְעַטָּה הַזֹּאת שֶׁהֵכִינוּ לִכְבוֹדֶךָ.

לָקַח מֵהֶם הַשַּׂר אֶת הַכֵּלִים וְהִבְטִיחָם לִמְשֹׁל בָּהֶם בְּחֶסֶד וּבְרַחֲמִים. וַיְהִי בְּהַבִּיטוֹ כֹּה וָכֹה רָאָה בְּתוֹךְ הַקָּהָל אֶת אָבִיו לָבוּשׁ בְּגָדִים קְרוּעִים. פָּנָה אֶל מְקַדְּמֵי פָּנָיו וְאָמַר: קְחוּ אֶת הַמַּתָּנָה אֲשֶׁר הֲבֵאתֶם לִי וּתְנוּהָ לָאִישׁ הַלָּזֶה, כִּי רוֹאֶה אֲנִי אֶת מַחְסוֹרוֹ. וְעוֹד דָבָר אֶחָד אֲבַקֵּשׁ מִכֶּם: כִּי תָחֹגּוּ חַג נִשּׂוּאִים בַּעֲדַתְכֶם תּוֹדִיעוּנִי וְאָבֹא לַחֻפָּה. אָמַר הַפַּרְנָס: הַשָּׁבוּע הַזֶּה אֲנִי מַשִּׂיא אֶת בִּתִּי.

אָמַר הַשָּׂר: אִם כֵּן תִּקְרָאֵנִי וְאָבֹא.

וְהַפַּרְנָס הוֹדָהוּ עַל חַסְדּוֹ וּבֵרְכוֹ בְּחַיִּים אֲרֻכִּים.

וּבְיוֹם הַשִּׂמְחָה בָּא הַשַּׂר לָבוּשׁ בִּגְדֵי־הָדָר וּרְבִיד־זָהָב עַל צַוָּארוֹ, וְיָשַׁב בִּמְקוֹם הַכָּבוֹד שֶׁהֵכִינוּ לוֹ לְיַד הַחֻפָּה.

וּבְהַגִּיעַ הַשָּׁעָה לְסַדֵּר הַקִּדּוּשִׁין עָמַד הַשָּׂר עַל רַגְלָיו וְאָמַר: חַכּוּ כִּמְעַט רֶגַע, דָבָר לִי אֲלֵיכֶם.

עָמַד כָּל הַקָּהָל מַחֲרִישׁ לְמוֹצָא פִּיו.

וְהַשָּׂר אָמַר בְּקוֹל: הַכַּלָּה הַיּוֹשֶׁבֶת עַל הַכִּסֵּא יְעוּדָה לִי. אֲנִי בֶּן דּוֹדָהּ, אֲשֶׁר נָסַעְתִּי עִם מָאתַיִם הָאֲדֻמִּים לִקְנוֹת סְחוֹרָה. וְעַתָּה עֵינֵיכֶם הָרוֹאוֹת מִי מִשְּׁנֵינוּ הֵיטִיב לִקְנוֹת.

וְסִפֵּר לָהֶם אֶת דְּבַר הַתַּפּוּחִים וְכָל הַמּוֹצְאוֹת אוֹתוֹ.

וְהֶחָתָן שֶׁעָמַד מוּכָן לְהִכָּנֵס לַחֻפָּה, הִסְתַּכֵּל רֶגַע בְּפָנָיו, אַחַר הוֹרִיד אֶת עֵינָיו וְיָצָא. וְהַשָּׂר עָמַד בִּמְקוֹמוֹ תַּחַת הַחֻפָּה וְקִדֵּשׁ אֶת הַכַּלָּה בְּשִׂמְחָה. הֵם שְׂמֵחִים וְכָל הַקָּהָל שְׂמֵחִים עִמָּהֶם, אַךְ יוֹתֵר מִכֻּלָּם שָׂמְחוּ אָבִיו וְאִמּוֹ הָעֲנִיִּים. וְאֵם הַכַּלָּה, הַפַּרְנָסִית, שְׂמֵחָה וְרוֹקְדָה וְשָׁרָה:

זֶה הַיּוֹם שֶׁקִּוִּיתִי לוֹ!

זֶה הַיּוֹם שֶׁקִּוִּיתִי לוֹ!
