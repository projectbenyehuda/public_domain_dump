לְשׁוֹנִי אֶשְׁתְּקַד הָיְתָה אֲמוּצָה

לְשׁוֹנִי אֶשְׁתְּקַד הָיְתָה אֲמוּצָה / אֲבָל הַיּוֹם מְצָאתִיהָ רְצוּצָה

וְרַעְיוֹנַי תְּמוֹל הָיוּ חֲזָקִים / וְהַיּוֹם נִמְשְׁלוּ אֶל עִיר פְּרוּצָה

וְהַשִּׁיר יִהֲיֶה נֶאְחָז בְּיָדִי / וְנָס מֶנִּי וְאֵין לִי עוֹד מְרוּצָה

וּמָה אַעַשׂ לְמוֹרָשֵׁי לְבָבִי / עֲזָבוּנִי וְנִשְׁאַרְתִּי לְשִׁמְצָה

הֲלֹא אַתֶּם תְּמוֹל רָצִים לְיָדִי / אֲבַקֶּשְׁכֶם לְשׁוֹחֵר טוֹב וְאֶמְצָא

וְאִם אֶשְׁלָחֲכֶם אֶל מוּל מְקַנְאַי / לְהַכְחִיד אֶת שְׁמָם בַּמַּעֲרָצָה

הֱיִיתֶם לִי לְסוֹחֵרָה וְצִנָּה / הֱיִיתֶם לִי לְעֵת מִלָּה נְחוּצָה

בְּנוֹת הַשִּׁיר וְלָמָּה תִפְשְׁעוּ בִי / וְלִקְרַאתְכֶם אֱמֶת נַפְשִׁי חֲפֵצָה

לְחַלּוֹת אֶת פְּנֵי שָׂרִים יְדוּעִים / וְכָל חָכְמָה בְּנִיבָם נֶחֱרָצָה

וְלָהֶם מִּשְׁפְּטֵי תוֹרָה גְּלוּיִים / בְּקִדּוּשִׁין וְיִבּוּם וַחֲלִיצָה

וְהַמִּשְׁנָה סְדוּרָה הִיא בְלִבָּם / וְכָל חִידָה וּמָשָׁל גַּם מְלִיצָה

יְדוּעִים הֵם בְּדִין טוּמְאָה וְטַהְרָה / וְלָהֶם שֵׁם בְּחָכְמַת הַקְּמִיצָה

וּמַשְׂכִּילִים בְּעֵרוּבִין וְשַׁבָּת / וּבִמְנָחוֹת וְהוֹרָיוֹת וּבֵיצָה

וּמֵהֶם תִּשְׁאֲלָה מִקְרָא בְּדִקְדּוּק / וּמֵהֶם תִּלְמְדָה מוּסָר וְעֵצָה

נְשִׁיכָתָם בְּגֵו עָצֵל כְּשׁוּעָל / מְשֻׁלַּחַת וְכָעַקְרָב עֲקִיצָה

הֱשִׁיבוּנִי שְׂעִפֵּי שִׁיר וְאֵיכָה / תְּבַקְשֵׁנוּ וּמִי יִתֵּן וְנִרְצָה

וְהִנֶּנּוּ עֲבָדֶיךָ כְּמֵאָז / וְאִם תִּשְׁלַח לְמִלְחָמָה לְיָהְצָה

אֲבָל אָנוּ יְרֵאִים פֶּן תְּבַלַּע / חֲרוּזֶיךָ וְתִדְאַג עַל קְפִיצָה

וְכַאְשֶׁר תַּעֲשֶׂינָה הַדְּבוֹרִים / לְעָרֶיךָ וְתִגְבַּר הַמְּחִיצָה

הֱשִׁיבוֹתִים וְעַל מַה זֶּה וּמַה זֶּה / וְאֵין כֹּל רַק קְבוּצַת שִׁיר קְבוּצָה

מְשַׁחֶרֶת פְּנֵי אַחִים וְרֵעִים / וְדוֹרֶשֶׁת שְׁלוֹמָם לִי בְּדִיצָה

לְמֵקִים כּוֹשְׁלַי תָמִיד בְּמִלָּיו / וְלִמְאַמֵּץ לְבִרְכִּי הָרְבוּצָה

חֲסָדָיו הֵם יְבוֹאוּנִי לְעֵת צָר / יְהוּ מִשְׁעָן לְנַפְשִׁי הַחֲלוּצָה

וְכֻלָּנוּ בְצֶדֶק נֶחֱזֶה אֶל / תְּמוּנַת אֵל וְנִשְׂבַּע בַּהֲקִיצָה

נְקַוֶּה לַחֲזוֹת מִשְׁכַּן דְּבִירוֹ / וּמֵאַרְבַּע יְקַבֵּץ הַנְּפוּצָה.
