יְדִידִי אִם תִּבְחַר

יְדִידִי אִם תִּבְחַר / לְהָבִין סוֹד זֹהַר

עֲמֹד נָא בַשַּׁחַר / לְשַׁעַר הָאוֹרָה


פְּתָחָיו הֵם רַבִּים / עֲשָׂרָה נִכְתָּבִים

רְכוּבֵיהֶם עָבִים / בְּמַלְכוּת הַנּוֹרָא


שְׁמוֹ אַלֶ״ף דֶּלֶ״ת / וְנוּ״ן יוֹ״ד מוֹעֶלֶת

וְהַדִּין פּוֹעֶלֶת / בְּעוֹבְרֵי הַתּוֹרָה


וּפֶתַח הַשֵּׁנִי / יְסוֹדוֹ בִּמְעוֹנִי

וְאֵל חַי יַעֲנֵינִי / בְּעֵת צַדִּיק יִקְרָא


וְנֵצַח עִם הַהוֹד / יְרֵכַיִם בַּסּוֹד

וּמַשְׁפִּיעִים לִיסוֹד / וּמֵהֶם לָעֲטָרָה


בְּשֵׁמוֹת נִפְלָאוֹת / אֱלֹהִים וּצְבָאוֹ

נְבִיאִים וּנְבִיאוֹת / מְקוֹמָם הַכִּירָה


וּמַחְשַׁבְתָּךְ רֵעִי / לְקָו הָאֶמְצָעִי

אֲחַוֶּה לָךְ דֵּעִי / וְלוֹ הַתִּפְאָרָה


וְיוֹ״ד הֵ״א רֵאשִׁיתוֹ / וְוָא״ו הֵ״א אַחֲרִיתוֹ

וְהַלֵּב מוֹדַעְתּוֹ / בְּנֶפֶשׁ הַבָּרָה


וּמִמָּךְ לֹא נֶעְלָם / זְרוֹעוֹת הָעוֹלָם

וְעֹצֶם מִפְעָלָם / גְּדֻלָּה וּגְבוּרָה


וּמִיָּמִין אֵלִי / אֱלֹהִים לִשְׂמָאלִי

וְהַסּוֹד הוּא פֶלִי / לְבַעַל הַמִּקְרָא


וְעַד פֹּה מִשְׁמַרְתָּךְ / וְטוּב כַּוָּנָתָךְ

וְתֵיטִיב אַחֲרִיתָךְ / בְּשֵׂכֶל וּבְצוּרָה


וְשָׁלֹשׁ אַחֲרוֹנוֹת / וְהֵמָּה רִאשׁוֹנוֹת

בְּעֹמֶק עִנְיָנוֹת / לְמַרְבֵּה הַמִּשְׂרָה


וְהֵן כֶּתֶר חָכְמָה / וּבִינָה נַעֲלָמָה

וְהִנֵּה נִשְׁלָמָה / עֲשָׂרָה בִסְפִירָה


שְׁמוֹתֵיהֶן אֶהְיֶה / וְיָהּ הֹוֶה יִהְיֶה

וּבָם חָיֹה תִחְיֶה / נְשָׁמָה נִזְהָרָה


וּבַבִּינָה סוֹדוֹת / אֱלֹהִים בִּנְקֻדּוֹת

בְּאַרְבַּע מוּעָדוֹת / וּבָהֶם נִכְתָּרָה


וְאֵין סוֹף לִתְהִלּוֹת / לְעִלַּת הָעִלּוֹת

מְמַלֵּא מִשְׁאָלוֹת / לְכָל נֶפֶשׁ נִבְרָא.
