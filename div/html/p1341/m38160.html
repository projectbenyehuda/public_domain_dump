וָאֶסַּע וָאֵלְכָה : מקאמה

הַכִּילַי וְהַנָּדִיב


וָאֶסַּע וָאֵלְכָה מִן הַכְּפָרִים / לִרְאוֹת בָּאָרֶץ וִיפִי הֶעָרִים / וַיְהִי הַיּוֹם וָאָבֹא בִקְהִלָּה מְפֹאָרָה / וּבְתוֹכָה בַּטְלָנִים יוֹתֵר מֵעֲשָׂרָה / וְרָאִיתִי כָּל הָעִיר שׁוֹמֵמָה / וְעַל פְּנֵי הַחוּצוֹת דְּמָמָה / אֵין קוֹל וְאֵין קָשֶׁב / אֵין גֵּר וְאֵין תּוֹשָׁב / רַק נַעַר אֶחָד נוֹשֵׂא יַיִן וְלֶחֶם עֲשָׂרָה חֲמֹרִים / וָאֶשְׁאָלֵהוּ וַיֹּאמַר: ׳הַיּוֹם עֶרֶב יוֹם כִּפּוּרִים׳.


וָאָרוּצָה הֵנָּה וָהֵנָּה לְבַקֵּשׁ מָלוֹן / וָאֶשָּׂא עֵינַי וְרָאִיתִי אִישׁ נִשְׁקָף בְּעַד הַחַלּוֹן / וָאֹמַר: ׳הוֹשִׁיעָה אֲדֹנִי / הוֹדִיעֵנִי בְּחַסְדְּךָ בֵּית מְלוֹנִי / כִּי מְאֹד חֻלֵּיתִי וְאָפֵס כֹּחִי / הֲלֹא עִבְרִי אָנֹכִי / וּמָחָר יוֹם צוֹם וּזְעָקָה / וְאֵיךְ אָצוּם וְנַפְשִׁי רֵיקָה?׳.


וְהִנֵּה הָאִישׁ עוֹמֵד וְכוֹס יַיִן עַל כַּפּוֹ / וְתִמְרוֹת עָשָׁן עוֹלוֹת מֵאַפּוֹ / וַיְהִי כַּאֲשֶׁר שָׁאַלְתִּי בִשְׁלוֹמוֹ / וַיַּסֵּב אֶת פָּנָיו וַיִּשְׁאַג בִּשְׁאוֹנוֹ: ׳גֶּשׁ הָלְאָה הַרְחֵק מִבֵּיתִי / פֶּן אֶפְגַּע בְּךָ מֵרֹב כַּעְסִי / הֲלֹא מִנְּעוּרַי מְתָעֵב אָנֹכִי כָּל גֵּר וְאוֹרֵחַ / אֲשֶׁר עַל הָעֵדָה לָטוֹרַח / וּמַה לָּנוּ לַהֲמוֹן הָאוֹרְחִים / אֲשֶׁר עָלֵינוּ כְּיוֹנִים בָּאֲרֻבָּה פוֹרְחִים / נִרְפִּים הֵמָּה וְשׂוֹנְאֵי מְלָאכָה / וִיבַקְשׁוּ מַעֲדַנִּים וּמְזוֹן אֲרוּחָה / וְיָקוּמוּ לְשַׂחֶק בָּנוּ / וְלִלְחֹךְ כִּלְחֹךְ הַשּׁוֹר אֶת כָּל אֲשֶׁר לָנוּ / וְאֶקַּח הַיּוֹם אֶת פִּתִּי וְיֵינִי / וְאֶתְּנָה לְזֶה הַנָּכְרִי הֶעָנִי? / וְהַתּוֹרָה צִוְּתָה לָתֵת הַנְּבֵלָה לַגֵּר אֲשֶׁר בִּשְׁעָרֶיךָ / וְלֹא לְאוֹרֵחַ אֲשֶׁר יָבֹא עָלֶיךָ / וְכַאֲשֶׁר אָמַר אִיּוֹב: דְּלָתַי לָאוֹרֵחַ אֶפְתָּח / שְׁחִין וְצָרַעַת בּוֹ פָרָח / כִּי רוֹב הָעוֹבְרִים וְשָׁבִים / מְצֹרָעִים הֵם וְזָבִים׳ / וַאֲנִי אָמַרְתִּי פֵּירוּשׁ אֱמֶת עַל זֶה הַפָּסוּק / דָּבָר נֶחְמָד וּמָתוֹק / אַךְ אֵין כָּאן מְקוֹמוֹ לְהַאֲרִיךְ / כִּי כְבָר שֻׁלְחָנִי עָרוּךְ.


וַיְכַל לְדַבֵּר וַיִּסְגֹּר הַדֶּלֶת בַּעֲדוֹ / וָאָרוּץ אַחֲרָיו בְּתוֹךְ בֵּיתוֹ / וַתָּרֶב כַּעְסוֹ / וַיֹּאמַר: ׳עַז פָּנִים מֵאַנְתָּ הִכָּלֵם / מִי הֱבִיאֲךָ הֲלוֹם?׳ / וָאֹמַר: ׳לֹא בָאתִי עַל זְבָחֶיךָ / וְלֹא אוֹכִיחֲךָ עַל דְּבָרֶיךָ / לֵךְ אֱכוֹל בְּשִׂמְחָה לַחְמֶךָ / וּמְכוֹר נְבֵלָתְךָ לַגֵּר אֲשֶׁר בִּשְׁעָרֶיךָ / אַךְ תְּנָה לִי פַּת לֶחֶם וּמֶלַח / וּמָחָר יוֹם הַדִּין לְעָשִׁיר וְהֵלֶךְ / וְגַם אֲנִי קָרוּא לַמֶּלֶךְ׳ / וַיֹּאמַר: ׳הַבֶּט נָא הַבַּיִת אֲשֶׁר מִמּוּלִי / שָׁם שׁוֹכֵן חֲסַר דַּעַת וֶאֱוִילִי / הוּא הָאִישׁ הַמְּאַסֵּף לְכָל הָאוֹרְחִים / וְיֵין נְסִיכוֹ דּוֹרְכִים / וְלֹא רָאִיתִי אִישׁ מְפַזֵּר כְּמוֹתוֹ / וְלֹא יָנִיחַ מְאוּמָה אַחֲרֵי מוֹתוֹ / וְעוֹכֵר שְׁאֵרוֹ וּבְנֵי בֵיתוֹ / וְגַם הוּא גֵּר בְּתוֹכֵנוּ / וְלֹא יִתְחַבֵּר אֶת בְּנֵי עֲדָתֵנוּ / וּבִקְהָלוֹ לֹא תֵחַד כְּבוֹדֵנוּ / וְעַתָּה שְׂטֵה וַעֲבֹר מֵעַל פָּנָי / וְאָכִין לִבִּי לִתְּפִלָּה וְתַחֲנוּנָי / אֲשֶׁר אֶתְפַּלֵּל מָחָר לִפְנֵי אֲדֹנָי׳ / וָאֹמְרָה: ׳יִשְׁמַע ה׳ אֶת קוֹל תַּחֲנוּנֶיךָ / וִיחַזֵּק וִיאַמֵּץ לְבָבֶךָ / לִקְפֹּץ אֶת יָדְךָ מֵעֲנִיֵּי עַמֶּךָ / וְכָל זָר לֹא יִקְרַב בְּאָהֳלֶךָ / וְיִהְיוּ עֲנִיִּים בְּנֵי בֵיתֶךָ׳.


וָאָבֹא אֶל בֵּית פְּלוֹנִי / וַיֵּצֵא לִקְרָאתִי נַעַר אַדְמוֹנִי / וַיֹּאמַר: ׳מַהֵר סוּרָה אֲדֹנִי / וַעֲלֵה הָעֲלִיָּה בַחֲדַר הַמִּשְׁתֶּה / וְגַם אַתָּה לְשָׂבְעָה תֹאכַל וְתִשְׁתֶּה׳ / וָאֹמַר: ׳מַה זֶּה מִהַרְתָּ בְּנִי / וַאֲדֹנֶיךָ לֹא יָדַע אֶת שְׁמִי / כִּי מֵאֶרֶץ מֶרְחָק אֲנִי׳ / וַיַּעַן: ׳הֲלֹא בֵית אֲדֹנִי פָתוּחַ / לְכָל גֵּר וְאוֹרֵחַ וְשָׁם יָנוּחַ׳ / וּבֹשְׁתִּי וְנִכְלַמְתִּי לְהָרִים פְּעָמָי / כִּי לֹא אָכַלְתִּי לֶחֶם עֹנִי כָל יָמָי / אַךְ חִצֵּי הַכָּפָן דּוֹלְקִים בִלְבָבִי / וְכַנּוֹשֶׁה אֲשֶׁר בָּא לָקַחַת אֶת יְהָבִי / כֵּן יְרִיבוּן רִיבִי / וָאָסִיר אֶת מַסְוֵה הַבּוּשָׁה / וְלַחֲדַר הָאֹכֶל אָחִישָׁה / וְהִנֵּה עֲדַת אֲנָשִׁים / שָׁם אוֹכְלִים חֲמוּשִׁים / וְלֹא מִתְבּוֹשְׁשִׁים.


וָאֵשְׁבָה לִי בִקְצֵה הַשֻּׁלְחָן / וּפָנַי חָפוּ וְהַנְּעָרִים יָרוּצוּ וְיָעוּפוּ / וַיַּצִּיגוּ לְפָנַי כָל נֵתַח טוֹבָה / נֶחְמָד לְמַרְאֶה וּבְשַׂר תַּאֲוָה / וְאֶת הַכֹּל אָכַלְתִּי בְחִפָּזוֹן / כִּי קָפַץ עָלַי חֲמַת הָרָזוֹן / וַנֹּאכַל דַּי וְהוֹתֵר / עַד כִּי תַם כָּל הַבָּשָׂר / אַחֲרֵי כֵן הֵבִיאוּ כַּדִּים וְכוֹסוֹת / מְלֵאִים יַיִן אֲשֶׁר שָׂם בָּנוּ הָאוֹתוֹת / כִּי הַיַּיִן בּוֹעֵר וְיָפִיחַ לְהָבִים / הֵנִיס כָּל הַדְּאָגוֹת וְהֶעָבִים / וַנְּמַלֵּא פִינוּ שְׂחוֹק אֲנִי וְרֵעַי הַקְּרֹבִים / וַנִּשָּׂא כוֹס הַיַּיִן וַנָּרִיעַ / וְכָל אִישׁ גְּבוּרָתוֹ יוֹדִיעַ / וּשְׂחוֹק וַהֲתֻלִּים יַבִּיעַ.


וַיָּקָם הָאָדוֹן מֵעַל כִּסְאוֹ וַיֹּאמַר: ׳אֵיכָה כַיּוֹם הַזֶּה שְׂמָחוֹת נַחְבִּירָה / וּמָחָר יוֹם הַקָּדוֹשׁ וְהַנּוֹרָא / וּמִי יַעֲמֹד אִם לֹא נַפְשׁוֹ מְטֹהָרָה? / אֲבָל אֲשָׁמִים רָאִיתִי בִּבְנֵי אָדָם / אֲשֶׁר חָמָס וָשֹׁד בְּיָדָם / וְיִתְעוֹלְלוּ וִיהוֹלְלוּ כָל הַשָּׁנָה / אֵין אִישׁ אֶל יוֹצְרוֹ פָנָה / ובְהַגִּיעַ הַסְּלִיחָה וְהַתְּשׁוּבָה / אָז יַחְשְׁבוּ תְּפִלָּתָם לְחוֹבָה / וְיִנָּחֲמוּ עַל רָעָתָם / וְיִתְוַדּוּ אֶת חַטָּאתָם / וְכָל הַיּוֹם יִתְקַדְּשׁוּ בְצוֹם וּתְפִלָּה / וְיִנְדְּרוּ נֶדֶר לְשֵׁם וּתְהִלָּה / אַךְ כִּמְעַט פָּנָה הַיּוֹם וְיָשׁוּבוּ לְבֵיתָם / וְיָשׁוּבוּ לְחַטָּאתָם / וְלֹא נוֹדַע כִּי בָא הַתְּשׁוּבָה בְּקִרְבּוֹ / וְהַיֵּצֶר יָשִׂים עָלָיו שֵׁנִית אֶת אָרְבּוֹ / וּכְכֶלֶב שָׁב אֶל קִיאוֹ וְצַחֲנָתוֹ / כֵּן הָרָשָׁע אֶל רִשְׁעָתוֹ׳.


וָאֹמַר: ׳מַה צָּדְקוּ דִּבְרֵי אֲדֹנִי / כָּזֹאת וְכָזֹאת זֶה הַיּוֹם קָרַנִי / אִישׁ אֶחָד אֲשֶׁר יִתְקַדֵּשׁ לְיוֹם מָחָר / מֵאֵן תֵּת פַּת לֶחֶם לְנֵכָר / וְאֵיךְ יַאֲמִין כִּי יִשְׁמַע ה׳ תְּפִלָּתוֹ / וְעוֹד לֹא רֹחַץ מִצּוֹאָתוֹ / וְאֵיךְ יַחֲשֹׁב לְטַהֵר בִּטְבִילָה אַחַת אֶת בְּשָׂרוֹ / וְהַנֶּגַע פָּשָׁה בְעוֹרוֹ / וְאִם זֶה בְטֻמְאַת הַגּוּף מֵרֶפֶשׁ / אַף כִּי בְטֻמְאַת הַנֶּפֶשׁ?. – –
