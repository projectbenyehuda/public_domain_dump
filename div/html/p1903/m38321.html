משל עב: אַיִל וּרְחֵלִים עֲשָׂרָה

אִֵין אֱמוּנָה בְּפִי הַמְקַנֵּא / כִּי הַשֶּׁקֶר לוֹ לְמִקְנֶה


לְאַיִל הָיוּ רְחֵלִים עֲשָׂרָה / וְחֶבְרָתוֹ לָהֶם שָׁפָרָה / וַיֶּאֱהַב אֶת הַשְּׁתַיִם אֲשֶׁר אֵין בַּמַּחֲנֶה כְמוֹהֶן / וְשַׁכּוּלָה אֵין בָּהֶן / אַךְ אִם יִקְרַב אֶל אַחַת מֵהֵנָּה / יָבִיא בְלֵב הַשֵּׁנִית קִנְאָה וּמְגִנָּה / וַתֵּרֶא הָאַחַת אֲשֶׁר אָהַב יוֹתֵר מִמֶּנָּה אֶת רְעוּתָהּ / וַתְּקַנֵּא רָחֵל בַּאֲחֹתָהּ / וּבְחֵץ שָׁחוּט לְשׁוֹנָהּ / תְּדַבֵּר עָלֶיהָ שִׂטְנָה / וְאֶת קַשְׁתָּהּ שֶׁקֶר דָּרְכָה / כִּי חֲנִית הַקִּנְאָה בְלִבָּהּ מְעוּכָה / לֵאמֹר: רַעְיָתְךָ חֲדָשִׁים לַבְּקָרִים / תַּחַת אִישָׁהּ תִּקַּח אֶת זָרִים / מִבִּלְעֲדֵי אִישָׁהּ נָתַן בָּהּ הָאַיִל שְׁכָבְתּוֹ / לְהַסִּיתָהּ מִמְּךָ וּלְשׂוּמָהּ אֶל מִשְׁמַעְתּוֹ / וּמִסְתּוֹפֶפֶת בְּצִלּוֹ בְּרוּחַ נְדִיבָה / בָּעֶרֶב הִיא בָאָה וּבַבֹּקֶר הִיא שָׁבָה / וְאֵיךְ תַּחְמֹל עָלֶיהָ / טֻמְאָתָהּ בְּשׁוּלֶיהָ / מִשְׁפָּטָהּ לִהְיוֹת עֲרוּפָה / בְּאַוַּת נַפְשָׁהּ שָׁאֲפָה / רוּחַ, לְהִפָּרֵד מֵעָלֶיךָ בִּזְנוּתָהּ / בְּתוֹכָהּ טֻמְאָתָה / תִּתֹּם חֶלְאָתָהּ / וַתְּדַבֵּר כָּאֵלֶּה דִּבְרֵי תֹךְ וּמִרְמָה / וְעוֹד נוֹסַף עֲלֵיהֶם דְּבָרִים רַבִּים כָּהֵמָּה / וַיַּעַן הָאַיִל: לִדְבָרֵךְ לֹא הֶאֱמַנְתִּי / כִּי עָתָק וָבוּז תַּחַת לְשׁוֹנֵךְ יָדַעְתִּי / אֲשֶׁר בָּדִית מִלִּבֵּךְ דְּבָרִים לֹא טוֹבִים / וַתְּדַבְּרִי עָלֶיהָ כְזָבִים / בְּשִׂנְאָתֵךְ אוֹתָהּ וּבְקִנְאָה / אֲשֶׁר דַּרְכֵי יֹשֶׁר מְנִיאָה / וְאִם הָיְתָה נַפְשֵׁךְ בְּנַפְשָׁהּ קְשׁוּרָה / בְּאַהֲבַת עוֹלָם עֲרוּכָה בַכֹּל וּשְׁמֻרָה / אָז לִדְבָרַיִךְ אָשִׁית לִבִּי / אַךְ כָּל עוֹד נַפְשִׁי בִי / אַרְחִיק מוֹצִיא דִבָּה / אִם מִשִּׂנְאָה וְקִנְאָה תִּהְיֶה נְסִבָּה / וְלֹא אֶתֵּן מַשְׂכֻּרְתּוֹ מָעוֹג / בְּחַנְפֵי לַעֲגֵי מָעוֹג.


וְהַמָּשָׁל אִם עַל אִישׁ כְּרִשְׁעוֹ / יַגִּיד עָלָיו רֵעוֹ / תַּשְׂכִּיל אִם הַדְּבָרִים בְּרוּרִים / עֵינֶיךָ תֶּחֱזֶינָה מֵישָׁרִים / כִּי אִם רֵעוֹ נֶהְפַּךְ לְעָרוֹ / תָּבִין כִּי בְזָדוֹן דְּבָרוֹ / וְאִם הוּא כְנַפְשׁוֹ אֹהֲבוֹ / וּמִיָּמָיו לֹא עֲצָבוֹ / חֲשֹׁב כִּי אֱמֶת הַדָּבָר / וְלֵב הַשֵּׁנִי מִן הָרִאשׁוֹן תַּם וְגָבָר / חִצֵּי רָכִיל שְׁנוּנִים עִם גַּחֲלֵי רְתָמִים / דִּבְרֵי נִרְגָּן כְּמִתְלַהֲמִים / הַהֹפְכִים אַהֲבָה לְחִלּוּף / וְנִרְגָּן מַפְרִיד אַלּוּף.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


רְחַק נִרְגָּן בְּהַפְרִידוֹ לְחֹשְׁקִים / וְחֵץ שִׂטְנָה יְשַׁלַּח כִּבְרָקִים

וְלִבּוֹ מַר וְחִכּוֹ מַמְתַּקִּים / אֲשֶׁר חֹנֶף שְׂפָתָיו הֵם מְרִיקִים

מְאַס זֵדִים בְּפִיהֶם לָךְ מְנַשְּׁקִים / וְתוֹכָם נַחֲלֵי אֵיבָה עֲמֻקִּים

בְּיוֹם גִּיל לָךְ קְרוֹבִים גַּם דְּבֵקִים / וְיוֹם אֵיד תִּמְצָאֵם חִישׁ רְחוֹקִים.
