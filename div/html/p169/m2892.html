חַלְלֵי חֹרֶף

א

מְנֻדֶּה וְשָׁחוּר נִתְעַרְפֵּל הַּיַּעַר

וַיָּנָס בִּמְצוּלוֹת הִרְהוּרוֹ הַקַּר.


עַל שֵׂיבַת קַרְקַפְתּוֹ יָנוּחַ הַבְּדִיל

וּלְרַגְלָיו הַשְּׁלָגִים מְלֹא רִבְבוֹת מִיל.


וּנְשִׁיַּת עוֹלָמִים בְּחִיקוֹ, וּכְמוֹ

עוֹד תְּקוּפַת-הַקֶּרַח מְנַמְנְמָה בוֹ.


ב

וּלְבָדָד נִצֶּבֶת, נִדַּחַת מִכְּפָר

הַ“טַּחֲנָה שֶׁל רוּחַ” בְּטַבּוּר הָהָר.


אֶת גַּנָּה הַמְחֻדָּד נָעֲצָה בַּשְּׁחוֹר

וּסְבִיבָה מִשְׂתָּרְעוֹת פַּרְסָאוֹת הַצְּחוֹר.


שׁוֹ­בֶתֶת הַטַּחֲנָה וּשְׁקוּעָה בַדְּמִי,

וּנְטוּיוֹת גַּפֶּיהָ בְּעֵרֶב וּשְׁתִי:


הָאַחַת לַבִּקְעָה, הַשְּׁנִיָּה לָעָב,

הַשְּׁתַּיִם לְרוּחוֹת הָעוֹלָם הַשָּׂב;


כִּנְכוֹנָה לְהַכְרִיז:"הַבּיִטוּ וּרְאוּ;

יְחִידָה אָנֹכִי בְּתֵבֵל הַזּוּ!"


ג

וְיֵשׁ אֲשֶׁר יָבֹא בֶּן-עוֹרֵב בִּיעָף

וְאִוָּה לוֹ מוֹשָׁב עַל-גַבֵּי הַגָּף


וְהִשְׁקִיף מִשָּׁמָּה עַל יָמִין, עַל שְׂמֹאל–:

רַק עַרְבוֹת הַשְּׁלָגִים, זוּלָתָן אֵין כֹּל –


וְסִמֵּר נוֹצָתוֹ וְצָרַח:“קְרָע-קְרָעָה!”

לֵאמֹר: פֻּרְעָנוּת מִתְרַגְּשָׁה וּבָאָה!


ד

וְקָדְרוּ שָׁמַיִם בְּיֶתֶר שְׂאֵת

כִּמְכִינִים אֶת עַצְמָם לְדַבֵּר רְתֵת.


וְרוּחַ קַל-אֵבָר וְשָׁנוּן וְקַר

אָז יֵעוֹר בַּבִּקְעָה וְטָס אֶל הַיָּעַר.


מַפְסִיקוֹ בִזְרִיזוּת בְּאֶמְצַע הָרָז:

“הִתְאוֹשֵׁשׁ, הַסָּבָא!” הוּא סוֹטְרוֹ – וְגָז


הָלְאָה, אֶל יִשּׁוּב: אֶל כְּפָר אוֹ אֶל עִיר –

וְנִתְקָל בְּדַרְכּוֹ בְּקַרְפִּיף אוֹ דִיר;


אוֹ נֶחְנָק בְּמֵזִיד וְנֵעוֹר בִּיָעף

וּמֵטִיל כּוֹבָעוֹ שֶׁל עוֹבֵר-וָשָׁב;


אוֹ מֵשִׁיב מִגְּדֵרוֹת לַבְנוּנִית הַמּוּךְ

וּפָגַע וְסָטַר בְּחֹטֶם אוֹ תְנוּךְ.


ה

וּפִתְאֹם – וְסוּפָה לֹא נוֹדָע מֵאֵי

תִּשְׂתָּרֵר לְבָנָה עַל גֶּבַע וָגֵי;


וְהָיְתָה מְהוּמָה בְּמַעַרְכוֹת הַיְקוּם

וּכִלְכּוּל הַתְּהוֹמוֹת שֶׁל שֵׁפֶל וָרוּם;


וּשְׁרִיקוֹת וַחֲרִיקוֹת מִתְעָרְבוֹת בָּרֹן

וּמְחוֹלוֹת שֶׁל הֶפְקֵר בִּגְנִיחוֹת אֵין-אוֹן;


וְעָרִים וּכְפָרִים וּבְקָעוֹת וָיַעַר

יִסָּחֲפוּ בַדַּיְסָה שֶׁל פְּקָעוֹת וָסַעַר.


ו

אַשְׁרֵי הַסְּפוּנִים בְּאוֹתוֹ לֵיל-זַעַם

בְּסֵתֶר אָהֳלָם הַמּוּנָף, הַחַם;


הַיּוֹשְׁבִים עַל שְׁרַפְרַף לְנֹכַח הָאָח

וְנֶהֱנִים מֵאוֹרוֹ וְחֻמּוֹ הָרָךְ;


וּשְׁכוּרִים וִיגֵעִים מֵהֶבֶל הַחֹם

מְפַהֲקִים, תּוֹךְ שִׂיחָה עַל מַעֲשֵׂי יוֹם-יוֹם;


מְהַפְּכִים בֵּינְתַיִם בֵּין גַּחֲלֵי-הַיְקוֹד

פְּנֵיהֶם – לְהָבִים חַכְלִילִים מְאֹד;


וְעַצְלָה בַּמֹחַ וְשַׁלְוָה בַּלֵּב

וְחֻלְשָׁה מְתוּקָה מְפַעְפְּעָה בַּגֵּו.


ז

וְאַשְׁרֵי הַיּוֹשְׁבִים בַּחֲבוּרָה בְּבַית

מְלֵא-אוֹרָה וְחַיִּים וּמְבַלִּים הָעֵת

בְּחַיֵּי הָעוֹלָם אוֹ חַיֵּי הַשָּׁעָה:

בִּפְלֻגְּתּוֹת נִלְהָבוֹת אוֹ לְשׁוֹן-הָרָע;


מִתְחָרִים בִּשְׁטוּת אוֹ לוֹקְקִים הָרִיר

עַל אוֹתָהּ הָרִיבָה הַיְדוּעָה בָעִיר.


הַמֵּחַם יְבַעְבֵּעַ וְקִיטוֹר וּסְאוֹן

מִתְמַזְּגִים בֵּשָׁאוֹן אוֹ צָהֳלַת-הָאוֹן.


ח

יֵשׂ אֹשֶׁר עוֹד אֶחָד בְּתֵבִל הַזּוּ…

לֹא רַבִּים הַזּוֹכִים לָאֹשֶׁר הַהוּא:


יְחִידִים עַל דַּרְגָּשׁ מְצֻפֵּה וָרַךְ

בְּמִקְדַּשׁ אַהֲבָה שֶׁנּוֹצַר לְכַךְ,


וְצוֹלְלִים בִּקְסָמִים שֶׁל אֹפֶל וָזִיו–

שְׁנֵי לִבּוֹת נִתְמַזְגוּ בְּלִי אֹמֶר וָנִיב.


עֲשָׁשִׁית יְרֻקָּה דוֹלֶקֶת בַּדְּמִי;

צָנוּעַ מִסְתַּתֵּר בָּזָּוִית הָרְאִי;


הַפְּסַנְתֵּר מְהַרְהֵר וּמוֹנֶה בַחֲלוֹם

הַנְּגִינוֹת שֶׁנִגֵּן מֵאָז עַד הַיּוֹם;


וְלוּלְבֵי עֲצִיצִים וּרְהִיטֵי הַפְּאֵר,

שֶׁקְּצָתָם בָּאֹפֶל וּקְצָתָם בָּאוֹר;


וִירִיעוֹת הָרִקְמָה וּתְמוּנוֹת עַל קִיר,

וּמְנוֹרוֹת-הַבְּדֹלַח וּפְסִילֵי-הַגִּיר –


כָּל אֵלֶּה נִשְׁתַּקְעוּ בִדְמָמָה שֶׁל סוֹד

הַמְּסִיתָה בַחֲשָׁאִי וּמְגָרָה מְאֹד…


וְאַשְׁרֵי שָׁבָּחְרוּ הַמִּשְׁכָּן הַזֶּה

וְאוֹתוֹ הַלַּיְלָה לִנְשִׁיקוֹת הַפֶּה.


ט

לְחִנָּם הֵם אוֹזְרִים שְׁאֵרִית הַחַיּל

הָאוֹבְדִים בַּדֶּרֶךְ בְּאוֹתוֹ הַלֵּיל;


לְחִנָּם יְשַׁוְּעוּ עֶזְרָת מִצָּר:

אֶנְקָתָם תְּסֹעַר בַּמָּחוֹל הַזָּר.


וְעֵינָם שֶׁעָשְׁשָׁה מֵעָקַת הַקֹּר

תְּבַקֵּשׁ לְחִנָּם בְּיֵאוּש שְׁבִיב-אוֹר.


אֵין זֵכֶר לִיִשּׁוּב, אֵין סִימָן לִשְׁבִיל,

רַק אֶמֶשׁ וִילָלָה וְעֶבְרָה וָחִיל.


כִּמְדֻמֶּה שֶׁעָלָה הַשָּׂטָן מִתְּהוֹם

וַיְקַדֵּשׁ לַשֵּׁדִים עֲצָרָה וָצוֹם;


וַיַּבְהֵל מֵרִבְצָם כָּל מוֹרְדֵי הָאוֹר:

מַטְאַטְאִים וּמְכַשְּׁפוֹת, וְשׁוּנְרוֹת הַשְּׁחוֹר,


לְעָבְרָם בְּאָלָה לַעֲשׂוֹת כְּכֹל

שֶׁצִוּוּ עֲלֵיהֶם אֲבַדּוֹן וּשְׁאוֹל.

                       -  -  -  -  -  -  -  -  -  -  -  -

י

כְּבָר עָמְדָה הַסּוּסָה שֶׁשֵׂרְכָה עַד כָּאן

כְּאִלּוּ שׁוֹאֶלֶת:“וְעַתָּה לְאָן?”


וּבְאוֹתָהּ עֲמִידָה הַחְלָטָה כָּזֹה:

וְלוּ גַם אֵהָרֵג לֹא אָמוּשׁ מִפֹּה!


וּלְחִנָּם מַצְלִיפִים עַל גַּבָּה:לְכִי!

וּכְפוּפַת בִּרְכַּיִם הִיא עוֹמְדָה בִקְשִׁי.


מַרְגִיזָה קְצָת עוֹרָהּ עַל יָרֵךְ אוֹ גָב

וְלוֹקְקָה לִפְרָקִים אֶת כִּיחָהּ הַזָּב!


אוֹ תָגַע בְּעַצְלוּת בְּדַל אָזְנָהּ, כְּמוֹ

לְהַקְשִׁיב הַהוֹלֵם עוֹד דָּפְקָהּ אִם לֹא?


יא

לֹא חָשׁוּ הָאוֹבְדִים בְּאוֹתוֹ הַסַּעַר

אֵיךְ אֻבַּן לְבָבָם בְּחֵיקָם הַקָּר:


אֵיךְ קָפָא בַקָּדְקֹד הַמֹּחַ הַלַּח,

אֵיךְ נִדְלַח וְנִפְסַק רַעְיוֹנָם הַזַּךְ;


אֵיךְ כָּבְתָה בַת-עֵינָם, אֵיךְ נִטַּל הַקּוֹל,

וּבְעוֹרְקָם אֵיךְ פָּסְקוּ דְמֵיהֶם מִנְּזֹל.


כִּי חָדְרָה הַסְּעָרָה אֶל בִּטְנָם כַּחֵץ

וַתַּגְלֵד אֶבְרֵיהֶם, וַתְּשִׂימֵם כָּעֵץ;


וַתִּסְחַט נִשְׁמוֹתָם וַתִּזְרֵן בִּצְחוֹק

לְאַרְבַּע הָרוּחוֹת וּלְגֹבַהּ אֵין-חֹק.


רַק רַחַשׁ לֹא בָרוּר הִזְדַּעֲזַע בּלַבֵּ

וְזֶרֶם חֲמִימוּת כְּמו-נִתַּךְ בַּגֵּו;


וַיַּחֲלֹף צֵל-רַעְיוֹ כְּעֵין הַחַשְׁמַל

וּבוּלְמוֹס-תַּרְדֵּמָה אֲפָפָם – וַחֲסַל.

                               -  -  -  -  -  -  -  -  -  -  -

יב

לְמָחָר יִמְצָאוּם בַּבִּקְעָה אוּ הָר

נְעוּצִים אוֹ טְמוּנִים בַּשֶׁלֶג הַקָּר.


פַּרְצוּפָם נָפוּחַ,וּכְמַרְאֵה הַטְּחוֹל;

יְעִירוּם וִיעוּרְרוּם אֵין-רוּחַ, אֵין-קוֹל.


וּצְפוּפִים נִצְמָדִים כְּאִלּוּ בִּצְבַת

יְחַבְּקוּ אִישׁ אָחִיו בְהֶסַּח הַדַּעַת.


וְשָׂפָה לִשָׂפָה בִנְשִׁיקַת אֵין-סוֹף…

כֹּה חִבְּקוּ וַיִּשְׁנוּ בְנֵי-הַחֲלוֹף.


יג

בַּבִּקְעָה לַבְנוּנִית אַדִּירָה מְאֹד

הַמְסַמְּאָה הָעַיִן בְּחֻצְפַּת הַהוֹד.


וְשַׁלְוָה עֲנָקִית וְגָאוֹן וָאוֹר

בֶּחָלָל הַמַתִּיז נִיצוֹצוֹת שֶׁל כְּפוֹר.


וְעוֹרְבִים מַבְהִיקִים וּכֶזֶפת לִשְׁחֹר

מְדַדִּים, מִשְׁתַּגְּעִים מִשִּׁפְעַת הַצְּחוֹר.


מְחֻסְפָּס וְלָבָן כָּל אִילָן בַּיַעַר,

יוֹם-גְּנוּסְיָה לָעֵצִים, יוֹם טָהוֹר וָקַר.


וְחֶדְוָה שֶׁל זָהָב וָחֹרֶף עַל פְּנֵי

הַשֶֹּמֶשׁ הַמּוֹשֵׁל הַנַעֲרָץ, הַגֵּא.


שִׂישִׂי, אַתְּ תֵּבֵל, בִּרְעָדָה וָדֹם,

כִּי מוֹשְׁלֵךְ הַשֶּׁמֶשׁ שְׂבַע רָצוֹן הַיּוֹם.


יד

רַק אַחַת הַטַּחֲנָה נִצֶּבֶת עַל הָר

כְּכֶתֶם דְּיוֹ שְׁחוֹרָה עַל לִבְנַת הַנְּיָר.


וּמְשֻנֶּה מַרְאֶהָ כִּי גֻדְּעָה הַגָּף

שֶׁאָמְרָה לְהַעְפִּיל אֶל בָּמָתֵי-הָעָב,


כְּנָפִיל שֶׁנֶּעֱרַף בַּקַרְדֹם הַחַד

וְהוּא נִשְׁאַר עַל רַגְלָיו מִשְׁתּוֹמֵם לָעַד…


וּכְאִלּוּ אוֹמֶרֶת: "הַבִּיטוּ וּרְאוּ:

דֵּרָאוֹן אָנֹכִי בְּתֵבֵל הַזּוּ!"


תרס"ט
