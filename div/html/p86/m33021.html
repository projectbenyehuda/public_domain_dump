ספר תחכמוני - שַׂעַר עֶשְׂרִים וְשִשָּׁה

בְּסִפּוּר עִנְיְנֵי הַדְּרָכִים. וְכָל אֲשֶׁר בָּם נִמְשָׁכִים.


נְאֻם הֵימָן הָאֶזְרָחִי. נָסַעְתִּי בִסְחוֹרָה לְאֶרֶץ בָּצְרָה. וַיְהִי הַיּוֹם וַאֲנִי מִתְהַלֵּךְ בִּרְחוֹב הַמְּדִינָה לִרְאוֹת בְּנֵי הָעִיר. הָרַב וְהַצָּעִיר. וָאֵרֶא בַּחוּרֵי חֶמֶד לְדַבֵּר דִּבְרֵי חָכְמוֹת נִקְבָּצִים. וּבְפִיהֶם עִנְיְנֵי מִלִּים נִמְרָצִים. וָאֶקְרַב אֲלֵיהֶם. לְהָרִיחַ רִקְחֵי בִשְׂמֵיהֶם. וְלִרְווֹת מִנּוֹזְלֵי גִּשְׁמֵיהֶם. וּבְעוֹדִי מִתְבּוֹנֵן אֲלֵיהֶם. וְשׁוֹמֵעַ אִמְרֵיהֶם. וְהִנֵּה נִצָּב אִישׁ זָקֵן לִפְנֵיהֶם. וַיֹּאמֶר לָהֶם: שִׂימוּ לַשָּׁמַיִם עֵינֵיכֶם. וְזִכְרוּ אֱלֹהֵיכֶם. וְאַל תִּבְטְחוּ בְעָשְׁרְכֶם. כִּי הָעוֹלָם יִכְלֶה. וּכְרָקָב יִבְלֶה. וְאֵין לְאִישׁ רַק בְּמַעֲשֵׂה הַצְּדָקָה נָחַת. וְהִיא תַצִּילֵהוּ מֵרֶדֶת שָׁחַת. וּמֵאֵשׁ מִתְלַקָּחַת.

אֲנִי הַגֶּבֶר. רָאִיתִי בְאֶרֶץ נוֹד כָּל יָגוֹן וְכָל שֶׁבֶר. וְכִמְעַט יָרַדְתִּי אֶל הַקֶּבֶר. וְשָתִיתִי בַהֲלִיכַת הַדְּרָכִים. גְּבִיעִים בְּלַעֲנָה מְסוּכִים. וְרָאִיתִי כַמָּה צָרוֹת וּמְאֹרָעוֹת. וְקוֹרוֹת רָעוֹת. כִּי אֵין יְגִיעָה רָעָה כַּהֲלִיכַת הַדְּרָכִים. אֲשֶׁר בָּם הָרַעְיוֹנִים נְבוּכִים. וְעַל כֵּן אָמְרוּ הַמּוֹשְׁלִים: הַהֲלִיכָה. מְבוּכָה. וְהַנְּסִיעָה. מְאֹרָעָה. וְהַפְּרִידָה. חֲרָדָה. וְכָל הַנּוֹסֵעַ מֵאַרְצוֹ. וְיֵצֵא מִמְּחוֹז חֶפְצוֹ. תִּשְׂבַּע בָּרָעוֹת נַפְשׁוֹ. וְלֹא יִשָּׂא רֹאשׁוֹ. הַשֶּמֶשׁ תַּכֶּנּוּ. וְהַסְּעָרָה תְּדַכֶּנּוּ. וְהַחֹרֶב יִשְׂרְפֶנּוּ. וְהַגֶּשֶׁם יִשְׁטְפֶנּוּ. וְהָרוּחַ תִּדְּפֶנּוּ. וְיֵלֵךְ מִן הַשַּׁחַר וְעַד הָאֲפֵלָה. לַחֹרֶב בַּיּוֹם וְלַקֶּרַח בַּלַּיְלָה. פַּעַם יַעַל גְּבָעוֹת. וּפַעַם יֵרֵד בְּקָעוֹת. וְיִקְרַע נְהָרִים. וְיִרְקַע הָרִים. וְיִדְרֹךְ יְעִרים וּמִדְבָּרִים. וְיַעֲבֹר גַּלִּים וּמִשְׁבָּרִים. וְיֵלֵךְ בַּעֲרָבוֹת וְצִיּוֹת. בֵּין נְחָשִׁים וַאֲרָיוֹת. וּפְרִיצֵי חַיּוֹת. וְאִם יֵרְדוּ עָלָיו גְּשָׁמִים. יְבַעֳתוּהוּ אֵימִים. וְיִזְּלוּ עֵינָיו כְּפִי הַגְּשָׁמִים דָּמִים. וְיִיגַע בְּלֶכְתּוֹ. וְיִתְעַנֶּה בַעֲלוֹתוֹ וְרִדְתּוֹ. וְאִם יִפֹּל מֵעַל מֶרְכַּבְתּוֹ. יִטְבֹּל בְּגָדָיו בְּטִיט וָמַיִם. וְסוּסוֹ יָשׁוּט בְּמֵי אַפְסַיִם. וְהַקֶּרֶב וְהַכְּרָעַיִם. יִרְחַץ בַּמַּיִם. וְלִבּוֹ בֵּין תַּנּוּר וְכִירַיִם. וּמִלְּבַד כָּל זֶה סַכָּנַת הַדֶּרֶךְ. וְרֹב הַפַּחַד וְהַמֹּרֶךְ. כִּי אִם לִסְטִים יִפְגָּעוּהוּ, לְשַׁעֲרֵי מָוֶת יַגִּיעוּהוּ. וְאִם יִמָּלֵט עָרֹם יַעַזְבוּהוּ. כַּאֲשֶׁר יָצָא מִבֶּטֶן אִמּוֹ יַנִּיחוּהוּ. וְאִם יֵלֵךְ רַגְלִי. יִמְצָא נֵזֶק וָחֳלִי. כִּי הָאֲבָנִים יַכְאִיבוּהוּ. וְהַקּוֹצִים יְצִיקוּהוּ. וְהַשּׁוּחוֹת עַל פָּנָיו יַפִּילוּהוּ. וְהַמַּעְבָּרוֹת בְּמֵימָם יִטְבְּלוּהוּ. וְאִם יֵרֵד שֶׁלֶג עָלָיו. אָז יִכָּפְלוּ חִילָיו. וְיִהְיוּ בְגָדָיו לְבָנִים. וְלִבּוֹ שָׁחוֹר בְּרֹב הַיְגוֹנִים. כְּשׁוֹבַךְ הַיּוֹנִים. לָבָן מִבֶּחוּץ וְשָׁחוֹר מִבִּפְנִים. וְאִם יָבֹא בַיָּם. וְנִשְּׂאוּ גַּלָּיו וְדָכְיָם. וְיוֹצִיאוּ עָלָיו כְּלֵי זַעְמוֹ. וְהָיְתָה רוּחַ אַחֶרֶת עִמּוֹ. אוֹ יְקַלֵּל יוֹמוֹ. וְיִמְאַס עוֹלָמוֹ. בִּרְאוֹתוֹ הָאֳנִיָּה עוֹלָה לִמְרוֹמוֹת. וְיוֹרְדָה לִתְהוֹמוֹת. וּבְנֵי מֵעָיו בְּקִרְבּוֹ. עוֹלִים וְיוֹרְדִים בּוֹ. עַד יָטִיל אֶל תּוֹךְ הַיָּם סְחוֹרָתוֹ. לְהַצִּיל מִמָּוֶת נִשְׁמָתוֹ. וְאִם יִמָּלֵט מִן הַיָּם גִּיל יָגִיל לֹא יִמַּד. וְיַחְשֹׁב כִּי בַיּוֹם הַהוּא נוֹלַד. וְאַף עַל פִּי כִי הוֹנוֹ בַיָּם עֲזָבָהוּ. וְעָרֹם יָשׁוּב בַּיּוֹם הַהוּא. יִשְׂמַח וְהוּא עָנִי וְאֻמְלָל. שֶׁהָיְתָה לוֹ נַפְשׁוֹ לְשָׁלָל. וְאַחֲרֵי כָל אֵלֶּה הַצָּרוֹת. וְהַתְּלָאוֹת הַמָּרוֹת. אִם לֹא יַשִּׂיג בַּהֲלִיכָתוֹ שְׁאֵלָתוֹ. תְּהִי לָרִיק יְגִיעָתוֹ. הֵן כָּל זֶה יִמְצָא לְכָל נוֹסֵעַ מִמְּעוֹנוֹ. כְּמוֹ הָעוֹף הַנּוֹדֵד מִקִּנּוֹ. עַד יַכֶּה לְחָיָיו. וְיִבְחַר מוֹתוֹ מֵחַיָּיו.

וַיִּשָּׂא מְשָׁלוֹ:

אֲשֶׁר יָמָיו יְבַלֶּה בַנְּסִיעָה

יְהִי עָמֵל וְלֹא יֶחְסַר יְגִיעָה.

וְנַפְשׁוֹ בֶּאֱזוֹר יָגוֹן אֲזוּרָה

וּבַת עֵינוֹ בְּפוּךְ לַיִל קְרוּעָה.

יְרַוֶּה הַחֲרֵרִים מִדְּמָעָיו

עֲדֵי יִצְמַח בְּאֶרֶץ לֹא זְרוּעָה.

וְעֵת יַחְשֹׁב מְצֹא נוֹחַ וְלִשְׁקֹט

זְמַן יֵעוֹר וְיָרִיעַ תְּרוּעָה.

וְרַגְלָיו יוֹם וְלֵיל דּוֹרְכִים כְאִלּוּ

עֲלֵיהֶם מִמְּצָא הַשְׁקֵט שְׁבוּעָה.

וְנַפְשׁוֹ יוֹם לְיוֹם תִּצְפֹּר כְּצִפֹּר

כְּאִלּוּ הִיא בְכַף קֶלַע קְלוּעָה.

וּבֵיתָהּ תַּחֲשֹׁב מַאְסָר וְנוּדָהּ

מְבַשֵּׂר טוֹב וּמַשְׁמִיעַ יְשׁוּעָה.

וְכַאֲשֶׁר כִּלָּה דְבָרוֹ. וְשִׁירוֹ. אָמַר לוֹ אִישׁ מִן הַחֲבוּרָה. הִנֵּה יָדַעְנוּ כְּי לְךָ שָׂפָה בְרוּרָה. אֲבָל אַתָּה זָכַרְתָּ קוֹרוֹת הַנְּסִיעָה. וְעָזַבְתָּ דִבָּתָהּ לְדוֹר אַחֲרוֹן קְבוּעָה. וְעַתָּה סַפֵּר מִדּוֹתֶיהָ הַטּוֹבוֹת. כַּאֲשֶׂר סִפַּרְתָּ מִדּוֹתֶיהָ הָרָעוֹת. וּזְכֹר נְעִימוֹתֶיהֶ כַּאֲשֶׁר זָכַרְתָּ הַמְאֹרָעוֹת.

וַיַּעַן וַיֹּאמַר: וְאִם הַנְּסִיעָה רָעָה וְהַהֲלִיכָה מְבוּכָה. וּבַלַּעֲנָה מְסוּכָה. הַלֹא בִנְסִיעָה. יַשִּׂיג הַנִּכְשָׁל תְּשׁוּעָה. וּבַהֲלִיכָתוֹ. תְּעָלָה לְמַכָּתוֹ. כִּי בָהּ יַשִּׂיג הָאִישׁ חֲפָצָיו. וְיִגְבְּרוּ חֲלוּצָיו. וְיֶאֱזֹר כְּגִבּוֹר חֲלָצָיו. וְיִדְרֹךְ לְלֵב אוֹיְבָיו חִצָּיו. וְלוּלֵי הַנְּסִיעָה לֹא עָלָה כְרֶגַע מַזָּלוֹ. וְלֹא יֶעְשַׁר וְלֹא יָקוּם חֵילוֹ. וְהַנּוֹסֵעַ בִּדְרָכִים בִּמְעַט יָמִים יִרְבֶּה טוּבֹו. וְיִכָּפֵל זְהָבוֹ. אִם תִּרְאֵהוּ הַיּוֹם עָרֹם וָהֵלֶךְ. מָחָר יִדְמֶה לְבֶן מֶלֶךְ. וְהַשּׁוֹכֵן בְּבֵיתוֹ. וְלֹא יֵצֵא מִסְּבִיב מִטָּתוֹ. לֹא תָסוּר מֵעָלָיו חֶרְפָּתוֹ. כִּי לֹא יֵצֵא מִצָּרָתוֹ. עַד יֵצֵא מֵאַדְמָתוֹ. וְלֹא יִסַּע מִלִּבּוֹ יְגוֹנוֹ. עַד יִסַּע מִמְּעוֹנוֹ. וּבְעוֹדוֹ יִשְׁכֹּן בְּאַרְצוֹ. לֹא יִמְצָא חֶפְצוֹ. עַל כֵּן יֹאמְרוּ הַמּוֹשְׁלִים: הַנְּסִיעָה. תְּשׁוּעָה. וְהַהֲלִיכָה. בְּרָכָה. וְכָל הַנּוֹסֵעַ. הוּא לְחֶפְצוֹ נוֹגֵעַ. וְכָל הַהוֹלֵךְ. הוּא עַל חֶפְצוֹ מוֹלֵךְ. וְהַמְשׁוֹטֵט. פְּרִי חֲפָצָיו יְלַקֵּט. וְהַיּוֹשֵׁב. כְּאֶבֶן דּוּמָם יֵחָשֵׁב. וּבַהֲלִיכָתוֹ יִרְאֶה פָנִים חֲדָשׁוֹת. וִירַפֵּא מִלְּבָבוֹ מַכּוֹת אֲנוּשׁוֹת. וְיִתְחַדֵּשׁ גִּילוֹ. וְיָאִיר הִלּוֹ. וְיִשְׁקֹט שְׁאוֹנוֹ. וְיָסִיר יְגוֹן רַעְיוֹנוֹ. וְיָשׁוּב לְאַרְצוֹ בְכִיס מָלֵא. וּמַזָּלוֹ עוֹלֶה. וְיִמְצָא הַנְּפָשׁוֹת אֵלָיו נִכְסָפוֹת. וְהָעֵינַיִם לְאֵם הַדֶּרֶךְ נִשְׁקָפוֹת.

וַיִּשָּׂא מְשָׁלוֹ:

אֲשֶר יִפְחַד לְבָבוֹ לַנְּסִיעָה

בְּאֵיד יִשְׂבַּע וְלֹא יֶחְסַר יְגִיעָה.

וְהֶעָצֵל סְבִיב בֵּיתוֹ יְסוֹבֵב

וְהוּא דוֹמֶה לְיָתֵד הַתְּקוּעָה.

וְלֹא יָסוּר יְגוֹנוֹ מִלְּבָבוֹ

וּמֵרָעָה יְהִי יוֹצֵא לְרָעָה.

וְהַמַּשְׂכִּיל יְשׁוֹטֵט יָם וּמִזְרָח

עֲדֵי יִמְצָא לְצָרָתוֹ תְשׁוּעָה.

וְהֶחָרוּץ יְהִי מוֹאֵס בְּשִׁבְתּוֹ

עֲדֵי יַשִּׂיג חֲפָצָיו בִּתְנוּעָה.

וְיִרְאֶה נִפְלְאוֹת עוֹלָם בְּלֶכְתּוֹ

וְיִצְבֹּר הוֹן וְיִלְמַד בִּין וְדֵעָה.

וְעֵת שׁוּבוֹ יְהִי לַמְיַחֲלִים לוֹ

מְבַשֵּׂר טוֹב וּמַשְׁמִיעַ יְשׁוּעָה.


וּבְכַלּוֹתוֹ אֶת שִׁירָיו. וּמְלוֹאֵי דְבָרָיו. כָּל הָעָם מִבִּרְכוֹתָם הֶעֶנִיקוּהוּ. וּמִשְׁדֵי נִדְבוֹתָם הֵינִיקוּהוּ. וְעַל שְׁמוֹ שְׁאֵלוּהוּ.

וַיַּעַן וַיֹּאמַר:

אֲנִי חֶבֶר מְחַבֵּר הַפְּנִינִים

וּמַאְמָרַי בְּלֵב צָרַי צְנִינִים.

בְּמַהְלָלַי אֲרוֹמֵם הַשְּׁפָלִים

וּבְלָשׁוֹנִי אֲגַדַּע רֹאש גְּאוֹנִים.

בְּטוּב נִיבִי אֲרוֹמֵם כָּל בְּנֵי בוּז

וְאַשְׂבִּיעַ מְתֵי גַאֲוָה קְלוֹנִים.

וְהַנָּכְבָּד אֲבִיאוֹ תוֹךְ שְׁאוֹלָה

וְהַשָּׁפָל אֲרִימוֹ עַד עֲנָנִים.
