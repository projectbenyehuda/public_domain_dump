ספר תחכמוני - שַׁעַר אַרְבָּעִים וַחֲמִשָׁה

מִמִּשְׁלֵי הַחָכְמָה. וְתַעֲלוּמוֹת סְתוּמָה


נְאֻם הֵימָן הָאֶזְרָחִי: נָסַעְתִּי מֵאֲלֶכְּסַנְדְרִיָה. לִטְבֶרְיָּה. וּמָצָאתִי בָהּ חָבֵר וְכַאֲשֶׁר רָאִיתִיהוּ צָהֲלו רַעְיוֹנַי. וַיְּאוֹרו עֵינַי. וְהִתְחֲזַּקְתִּי בוֹ וְאָמַרְתִּי לוֹ: לְכָה נָא אִתִּי לְבֵיתִי. וּנְוֵה מְנוּחָתִי. וְאָשִׁים לְפָנֶיךָ מִנְחָתִי. וְנִתְעַלְּסָה בַאֲהָבִים. וְנִרְוֶה דְמֵי עֲנָבִים. וַיָּבוֹא עִמִּי. לַהֲדוֹמִי. וְשַׂמְתִּי לפָנָיו שֻׁלְחָן עָרוּךְ. וְיַיִן מָסוּךְ. וְכַאֲשֶׁר אָכַל וְשָׂבַע וְדָשֵׁן . וְשָׁקַט הַלָּשׁוֹן וְהַשֵּׁן. אָמַר לִי: חוֹבָה עָלַי לְהָשִׁיב גְּמוּלְךָ. וּלְשַׁלֵם פָּעָלְךָ. מֵאֲשֶׁר נָתַן הַבּוֹרֵא אֵלַי. כִּי חַסְדְּךָ גָּדוֹל עָלַי. וְעַתָּה אַגִּידָה לְךָ מִבְחַר הַמְּשָׁלִים הַנְּעִימִים. אֲשֶׁר שָׁמַעְתִּי מִפִּי חֲכָמִים. וְאָמַרְתִּי לוֹ: דַבֵּר כִּי שׁוֹמֵעַ אָנֹכִי. אָמַר: שָׁמוֹעַ שָׁמַעְנוּ. וַאֲבוֹתֵינוּ סִפְּרוּ לָנוּ. כִּי אִישׁ מִגְּדוֹלֵי הַחֲכָמִים. הָרָמִים. הָיָה אוֹהֵב מִשְׁלֵי הַמּוּסָר. וּבְכָל עֵת הָיָה אֲלֵיהֶם סָר. וּמֵהֶם לֹא לֹא סָר. וְיוֹם אֶחָד נִתְקַבְּצוּ לְפָנָיו אַרְבָּעָה וְעֶשְׂרִים אִישׁ מִגִּבּוֹרֵי הַחֲכָמִים אֲשֶׁר נִקְּבוּ בְשֵׁמוֹת. וַיֹּאמֶר לָהֶם: יִזְכֹּר כָּל אֶחָד מִכֶּם. מָשָׁל מִמִּשְׁלֵי הַחָכְמָה. לָתֵת לִפְתָאִים עָרְמָה. לְנַעַר דַּעַת וּמְזִמָּה. לְמַעַן יְהִי לְאַנְשֵׁי לֵב מוֹרֶה. וּבְדֶרֶךְ יָשָׁר חַטָּאִים יוֹרֶה. אָמְרוּ: טוֹב הַדָּבָר אֲשֶׁר דִּבַּרְתָּ. וַיַּעַן הַגָּדוֹל מֵהֶם וַיֹּאמַר: הִגִּידוּ לָנוּ כִּי עָבַר אִישׁ מִן הַחֲסִידִים עַל קִבְרוֹת הַמֵּתִים וַיֹּאמַר: שָׁלוֹם עֲלֵיכֶם שוֹכְנֵי אֶרֶץ צַלְמָוֶת, הַשׁוֹכְנִים בַּחֲצַר מָוֶת, הוֹדִיעוּנִי מַה מְּצָאתֶם. וְטֹרַח הַדִּין אֵיךְ נְשָׂאתָם.

אֲכַלְתָּם פְּרִי מִפְעֲלֵיכֶם. וּקְצַרְתֶּם זֶרַע מַעַלְלֵיכֶם. נְסַעְתֶּם אַתֶּם בָּרִאשׁוֹנָה. וַאֲנַחְנוּ נוֹסְעִים בָּאַחֲרוֹנָה. הָאֵל יִסְלַח לָנוּ וְלָכֶם. וְיַחְמֹל עָלֵינוּ וַעֲלֵיכֶם. אָמַר אַחֵר: אִישׁ מִן הַחֲסִידִים הָיוּ חֲבֵרָיו לְפָנָיו. מְשַׁבְּחִים מִפְעָלָיו וְעִנְיָנָיו. וְיִשָּׂא לַשָּׁמַיִם עֵינָיו. וְאָמַר: רִבּוֹן הָעוֹלָמִים אַתָּה יָדַעְתָּ מִמֶּנִּי. מַה שֶׁלֹא אֵדַע אֲנִי מִנַּפְשִׁי. וַאֲנִי יוֹדֵעַ מִנַּפְשִׁי. מַה שֶׁלֹּא יֵדְעוּ אֵלֶּה. אָנָּא אֶלֹהַי שִׂימֵנִי כַאֲשֶׁר יַחְשְׁבוּנִי. וְאַל תַּעַנְשֵׁנִי בְּמַה שֶׁהֵם מְשַׁקְּרִים עָלַי. אָמַר אַחֵר: הָיָה מִנְהַג אִישׁ מִן הַחֲסִידִים בְּבוֹא עָלָיו צָרָה שֶׁיֹּאמַר בִּתְפִלָּתוֹ: יְהִי רָצוֹן מִלְפָנֶיךָ שֶׁתְּהֵא זֹאת הַתְּלָאָה לְמוּסָר וּלְזִכָּרוֹן. וְלֹא לְענֶשׁ וְחָרוֹן. אָמַר אַחֵר: אִישׁ מִן הַחֲסִידִים מֵת אֶחָד מֵאוֹהֲבָיו. וּבִקְצָת הַלֵּילוֹת רָאָהוּ בַּחֲלוֹם עַל מִשְׁכָּבוֹ. וְאָמַר לוֹ: הוֹדִיעֵנִי מַה מָּצָאתָ בְעוֹלָמְךָ. וְאַיֵּה מְקוֹמְךָ. אָמַר לוֹ: אַל תִּשְׁאַל עַל אֲשֶׁר מָצָאנוּ. מָצָאנוּ הַחֶשְׁבּוֹן עָרוּךְ. וְהַסֵּפֶר פָּתוּחַ. וְהַדַּיָּן לֹא יִקַּח שֹׁחַד. וְשָׁכַחְנוּ מַעֲמַד הַמָּוֶת לְמַעֲמַד הַדִּין אֲשֶׁר הוּא מַר מִמָּוֶת.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אָמְרוּ לְדוֹדֵנוּ וְלַאְהוּבֵנוּ

גֹּדֶל יְגוֹנֵנוּ וְרֹב עָצְבֵּנוּ.

בָּאנוּ לְמַעֲמַד דִין אֲשֶׁר הִבְעִיר יְקוֹד

לַהַב בְּנַפְשֵׁנוּ וּבִלְבָבֵנוּ.

חשֶׁךְ וְאֹפֶל וַעֲלָטָה הֵם סְבִי־

בֵנוּ בְּעַרְשֵׂנוּ וּמִשְׁכָּבֵנוּ.

מִי יִתְּנֵנוּ נֵדְעָה כָל־זֶה בְּחַ־

יֵּינוּ וְנָשׁוּב מִנְּתִיב חוֹבֵנוּ.

לָכֵן יְדִידֵינוּ מְאֹד הִשָּׁמְרוּ

פֶּן יִקְרְכֶם מַכְאוֹב כְּמַכְאוֹבֵנוּ.

אָמַר אַחֵר: אִשָּׁה מִבְּנוֹת עֲרָב מֵת בְּנָה. וְכַאֲשֶׁר קְבָרַתְהוּ. אָמְרָה: בְּנִי בְּנִי הָיִיתָ הַלְוָאַת מַלְוֶה . וּמַתְּנַת נוֹתֵן. וּפִקְדוֹן מַפְקִיד. וְהַיּוֹם נִפְרַע הַמַּלְוֶה מֵהַלְוָאָתוֹ. וְלָקַח הַמַפְקִיד עֶרֻבָּתוֹ. וְקִבֵּל הַנּוֹתֵן מַתְּנָתוֹ. הָאֵל יִתֶּן־לִי תְמוּרָתְךָ סֶבֶל טוֹב. וְיִגְמְלֵנִי עַל אֲשֶׁר לָקַח מִמֶּנִּי שָׂכָר טוֹב. אָמַר אַחֵר: אִשָּׁה מִבְּנוֹת עֲרָב מֵת בְּנָהּ וְהִתְנַחֲמָה מְהֵרָה עָלָיו. אָמְרוּ לָהּ הַשְּׁכֵנוֹת: אֵיךְ תּוּכְלִי לְהִתְנַחֵם מְהֵרָה. אָמְרָה לָהֶן: יֵשׁ בַּנִּחוּמִים שְׁתֵּי מִדוֹת טוֹבוֹת. שִׂמְחָה בָעוֹלָם הַזֶּה וְשָׂכָר לָעוֹלָם הַבָּא. וְיֵשׁ בְּרֹב הָאֶבֶל שְׁתֵּי מִדוֹת רָעוֹת. אִבּוּד שְׂכַר הַתְּלָאָה. וְעֹנֶשׁ עַל רֹב הַדְּאָגָה. אָמַר אַחֵר: אִשָּׁה מִבְּנוֹת עֲרָב הָיָה מִנְהָגָהּ בִּתְפִלָּתָה שֶׁתֹּאמַר: רִבּוֹן הָעוֹלָמִים. אִם אָשַׁמְתִּי לָךְ. אַשְׁמָתִי לֹא תַזִּיק לָךְ. וְאִם תִּמְחֹל לִי מְחִילָתְךָ לא תֶחְסַר לָךְ. אָנָּא אֶלֹהַי מְחַל לִי מַה שֶׁלֹא יִזִּיק לָךְ. וְתֵן לי מַה שֶׁלֹא יֶחְסַר לָךְ. אָמַר אַחֵר: אִישׂ מִן הַחֲסִידִים פָּגַע בְּאַחַד הַשָׂרִים וְהוּא מִתְהַלֵּךְ בְּגַאֲוָה. אָמַר לוֹ הֶחָסִיד: דַּע כִּי הַבּוֹרֵא מוֹאֵס בְּזֹאת הַהֲלִיכָה. אָמַר לוֹ הַשָׂר: אִלוּ יָדַעְתָּ מִי אָנִי. כַּדָּבָר הַזֶּה לא הִשְׁמַעְתָּנִי. אָמַר הֶחָסִיד: יְדַעְתִּיךָ כִּי רֵאשִׁיתְךָ טִפָּה סְרוּחָה. וְאַחֲרִיתְךָ חֶלְאָה זְנוּחָה. וּבְעוֹדְךָ בַחַיִּים נִבְלָתְךָ כַּסּוּחָה. אָמַר אַחֵר: חָטָא אִישׁ אֶחָד לְמֶלֶךְ מִן הַמְּלָכִים וַיְבִיאוּהוּ לְפָנָיו. וּבְבוֹאוֹ אָמַר לַמֶּלֶךְ: אַשְׁבִּיעֲךָ בָּאֵל לְבַל תְּשְׂמַּח עָלַי אוֹיֵב הֶעֶצַבְתּוֹ. וְאַל תַּעֲצִיב אוֹהֵב שִׂמַחְתּוֹ. וְאַל תַּעֲקֹר מַטַּע חֶסֶד נְטַעְתּוֹ. וְאַל תַּהֲרֹס בִּנְיָן צֶדֶק יְסַדְתּוֹ. וַיַּחְמֹל עָלָיו הַמֶּלֶךְ וַיְשַׁלָחֵהוּ. אָמַר אַחֵר: שָׁאַל אִישׁ מִן הַחֲכָמִים לַחֲבֵרוֹ וַיֹּאמֶר לוֹ: כַּמָה תִּשְׁתֶּה מִן הַיַּיִן עַל אֲכִילָתֶךָ. אָמַר: שָׁלשׁ כּוֹסוֹת, אָמַר לוֹ: הֵיטַבְתָּ כִּי לֹא שְׁתִיתוֹ שְׁתִיַּת הַשִּׁכּוֹרִים. וְלֹא אֲסַרְתּוֹ אִסוּר הַנְּזִירים. אָמַר אַחֵר: אִישׁ מִן הַחֲכָמִים חָלָה אֶת חָלְיוֹ אֲשֶׁר יָמוּת בּוֹ. וְהָיָה מִתְפַּלֵּל לְאֵל לִמְחֹל חוֹבוֹ. וַחֲבֵרָיו הָיוּ מְשַׁבְּחִים הַמִּדּוֹת אֲשֶׁר בּוֹ. וּבוֹכִים סְבִיבוֹ. אָמַר לָהֶם: עִזְבוּנִי מִן הַתְּהִלָּה. וְעִזְרוּנִי בַתְּפִלָּה. אָמַר אַחֵר: אִישׁ מִן הַשָּׂרִים מֵת בְּנוֹ וּבָא אֶחָד מִן הַחֲכָמִים לְנַחֲמֵהוּ וְאָמַר לוֹ: כְּשֶׁתִּהְיֶה אַתָּה הַשְּׁאֵרִית הַנִּמְצָאָה. לֹא נִדְאַג עַל הַתְּלָאָה. וְהַמָּוֶת הִגְדִּיל עָלֵינוּ חַסְדּוֹ. בְּעָבְרוֹ עָלֶיךָ וְלא שָׁלַח בְּךָ יָדוֹ. אָמַר אַחֵר: אִישׁ חָכָם בָּא לְנַחֵם אַחַד הַשָּׂרִים עַל בְּנוֹ אָמַר לוֹ: יִתֶּן־לְךָ הָאֵל נִחוּם אָרוּךְ. וְשָׂכָר עָרוּךְ. וְזֶרַע בָּרוּךְ. וְיֵצֶר סָמוּךְ. אָמַר אַחֵר: אִישׁ מִן הַמְשׁוֹרְרִים שִׁבַּח לְאֶחָד מִן הַמְּלָכִים בִּימֵי חֲיוֹתוֹ. וְקוֹנֵן עָלָיו בְּמוֹתוֹ. אָמְרוּ לוֹ חֲבֵרָיו: מַדּוּעַ תְהִלוֹתֶיךָ טוֹבִים מִקִּינוֹתֶיךָ. אָמַר: כִּי תְהִלּוֹתַי הָיוּ חוֹבָה. וְקִינוֹתַי נְדָבָה. אָמַר אַחֵר: שָׁאֲלוּ לְאִישׁ לְסַפֵּר יְפִי חֲשׁוּקָתוֹ. אָמַר: בִּטְנָהּ כְּמוֹ אַגָּן. וּפָנֶיהָ גַן. וּשְׂפָתֶיהָ מָן. וּלְחָיֶיהָ שׁוֹשָׁן. וְקוֹמָתָהּ עֵץ רַעֲנָן. רְסִיסָיו בְּדָלְחֵי רְבִידֶיהָ. וְנִצָּנָיו בְּגָדֶיהָ. וּפִרְיוֹ שְׁנֵי שָׁדֶיהָ. אָמַר אַחֵר: שָׁאֲלוּ לְאִישׁ לְסַפֵּר יְפִי חֲשׁוּקָתוֹ. וְאָמַר: אֵין כָּמוֹהָ בְּכָל הַמְּדִינוֹת. יְחִידָה בַבָּנוֹת. וְיָפָה בָעֲדִינוֹת. אִם תַּעֲמֹד הִיא קְנֵה זָהָב. וְאִם תֵּשֵׁב הִיא אַיֶלֶת אַהַב. וְאִם תֵּלֵךְ הִיא יוֹנַת רַהַב. וְאִם תְּדַבֵּר תַבְעִיר בַּלֵּב לַהַב. אָמַר אָחֵר: אִישׁ מִן הַמְשׁוֹרְרִים עָשָׂה שִׁירָה לְאַחַד הַשָּׂרִים וְהַשָּׂר כִּבְּדָהוּ. וּבְבֵיתוֹ הִשְׁכִּינָהוּ. וְהָיָה לְבַעַל הַבַּיִת בַּת יָפָה וְהַמְשׁוֹרֵר עָשָׂה שִׁירֵי חֵשֶׁק וַיְשׁוֹרֵר אוֹתָם. וְהִיא שׁוֹמַעַת פֶּתַח הָאֹהֶל.

וַיֹּאמַר:

צְבִיַּת חֵן כְּנִדְגָּלוֹת אֲיֻמָּה

כְּבוּדָּה אַתְּ כְּבַת מֶלֶךְ פְּנִימָה.

חֲשַׁקְתִּיךְ יַעֲלָה בָרָה וְתַמָּה

וְאַתְּ שׁוֹמְעָה וְאָזְנֵךְ לֹא סְתוּמָה.

פְּנִי אֵלַי בְּעֵין רָצוֹן וְחֶמְלָה

וּמַשְׂכֻּרְתֵּךְ תְּהִי מֵאֵל שְׁלֵמָה.


וְכַאֲשֶׁר שָׁמְעָה קוֹלוֹ. וַתַּעַן וַיֹּאמֶר לוֹ:

דְּבָרֶיךָ בְלִי דַעַת וְחָכְמָה

וְשִׁירְךָ שִׁיר מְתֵי חָנֵף וְזִמָּה.

וְתָעִיד כִּי לְךָ יַחַשׂ מְטֻנָּף

וְאִשָּׁה לֹא יְלָדַתְךָ תְמִימָה.

וְעַל כֵּן הַחֲרֵשׁ תֵּצֵא בְשָׁלוֹם

וְאִם אַיִן תְּכַסֵּךְ כְּלִמָּה.


אָמַר אַחֵר: עָבַר אִישׁ מִן הַחֲסִידִים בַּדֶּרֶךְ וְרָאָה אֶחָד מִן הַפָּרִיצִים תָּלוּי עַל הָעֵץ. אָמַר הֶחָסִיד לַחֲבֵרוֹ: כָּל מִי שֶׁיַּעֲזֹב הַנְּכוֹחָה בְרִשְׁעָתוֹ. הָעֵץ יְתַקֵּן אֲשֶׁר עִוְּתוֹ. אָמַר אַחֵר: הֵבִיאוּ לִפְנֵי הַמֶּלֶךְ אִישׁ חַיָּב וְצִוָּה לְהַכּוֹתוֹ בְּשׁוֹטִים. וְהָיָה הָאִישׁ הַמֻּכֶּה אָרֹךְ מְאֹד. וְהַמַּכֶּה אוֹתוֹ קְצַר קוֹמָה. וַיֹּאמֶר הַמַּכֶּה אֶל הַמֻּכֶּה: הַשְׁפֵּל גּוּפְךָ כְּדֵי שֶׁאוּכַל לְהַכּוֹתֶךָ. אָמַר לוֹ הַמֻּכָּה: אוֹי לְךָ בֶּן הַזּוֹנָה הֲלִסְעֻדָּה תִּקְרָא אוֹתִי. חַי ה' הָיִיתִי רוֹצֶה לִהְיוֹת אָרֹךְ כְּמוֹ עוֹג. ואַתָּה קָצָר כְּמוֹ גוֹג. וַיִשְׂחַק הַמֶּלֶךְ וַיְשַׁלָחֵהוּ. אָמַר אַחֵר: רָאָה אִישׁ חָכָם לִבְנוֹ כִּי הוּא מִסְתַּכֵּל בְּאִשָׁה יָפָה, וַיֹּאמֶר לוֹ: בְּנִי בְּנִי מַה תֶּאֱהַב בַּנָּשִׁים וּבִטְנֵיהֶם בּוֹרוֹת. וּשְׁדֵיהֶם נֹאדֹות. וְיָפְיָם עֶרְוָה. וְהֵם קִבְרוֹת הַתַּאֲוָה. אָמַר אַחֵר: שָׁאֲלוּ לְאִישׁ כָּפְרִי וְאָמְרוּ לוֹ: כְּשֶׁתֵּצֵא מִבֵּיתֶךָ. מִי יִשְׁמֹר אִשְׁתֶּךָ. אָמַר: הִפְקַדְתִּי עָלֶיהָ שְׁנֵי שׁוֹמְרִים. עֲזַבְתִּיהָ עֲרֻמָּה לְבַל תֵּצֵא עִם הַנָּשִׁים. וּרְעֵבָה לְבַל תְּשַׂחֵק עִם הָאֲנָשִׁים. אָמַר אַחֵר: שָׁאֲלוּ לְאִישׁ כָּפְרִי וְאָמְרוּ לוֹ: אֵי זֶה מֵהַחֲתָנִים תִּרְצֶה לִבְנוֹתֶיךָ. אָמַר: הַחֲתָנִים שְׁלשָׁה: חֲתַן מָמוֹן וַחֲתַן יַחַס וַחֲתַן קֶבֶר. וְטוֹב מִכֻּלָּם חֲתַן הַקֶּבֶר. אָמַר אַחֵר: שָׁאֲלוּ אִישׁ זָקֵן בָּא בַּיָּמִים עַל שְׁלוֹמוֹ וְעִנְיָנָיו. אָמַר: אֲנִי חוֹלֶה מִבִּטְנִי וּמִיִּצְרִי. מַחֲלַת בִּטְנִי: כְּשֶׁאֶרְעַב אֶגְוַע. וּכְשֶׁאֶשְׁבַּע אִיגַע. וּמַחֲלַת יִצְרִי: כְּשֶׁאָקוּם יָנוּם. וּכְשֶׁאָנוּם יָקוּם. אָמַר אַחֵר: אִישׁ אֶחָד מִן הַמְשׁוֹרְרִים בָּא לְבֵית אַחַד הַשָּׂרִים. וְדָפַק וְלֹא פָתְחוּ הַשְׁעָרִים. וַיֹּאמֶר אֶל הָעוֹמְדִים: מִי שֶׁתִּהְיֶה יְמִינוֹ עֲסוּקָה בְאִשָּׁה. וּשְׂמֹאלוֹ בַּאֲשִׁישָׁה. וּמִצְחוֹ נָחוּשָׁה. וְאִמוֹ קְדֵשָׁה. וּמִשְׁפַּחְתּוֹ מִן גֵּו גְּרוּשָׁה. מֵחֲמַס אָדָם לֹא תְכַסֵּהוּ בוּשָׁה. אָמַר אַחֵר: שָׁאֲלוּ לְאִישׁ מִבְּנֵי עֲרָב עַל אִישׁ כִּילַי, וְאָמַר: נַבְלוּתוֹ לְאֵין חֵקֶר. וּמוֹעֲדָיו שֶׁקֶר. וְאַהֲבָתוֹ כַּעֲנָן בֹּקֶר, וַעֲנָנָיו בְּלִי מְטָרִים. וּתְלָמָיו חֲרֵרִים. וּשְׁדֵי אִמּוֹ צוֹמְקִים. וְגַפְנֵי טוֹבָתוֹ בוֹקְקִים. הוּא כִבְהֵמָה רוֹבְצָה. אוֹכְלָה כֹּל אֲשֶׁר קִבְּצָה. וּבוֹלַעַת כֹּל אֲשֶׁר תִּמְצָא. אָמַר אַחֵר: שְׁנֵי גִבּוֹרִים הִתְפָּאֲרוּ זֶה עַל זֶה. וַיֹּאמֶר הָאֶחָד לַשֵּׁנִי: אֵין תַּעֲרֹךְ אֵלַי וַאֲנִי בֶן סוֹתֵם הַפְּרָצִים. אָמַר לוֹ: צָדַקְתָּ בַאֲשֶׁר דִּבַּרְתָּ. אֲבָל אָבִיךְ סְתָמָם בִּמְנוּסָתוֹ. וְאָבִי סִתְּמָם בַּחֲנִיתוֹ.

אָמַר הַמַּגִּיד: וּכְשָׁמְעִי מִשְׁלֵי חֶבֶר הַקֵּינִי. צָפַנְתִּים בְּלִבִּי וּבְרַעְיוֹנִי. וְהֶחְזַקְתִּיו יָמִים רַבִּים עִמִּי. מִתְעַדֵּן בְּבֵיתִי וְאוּלַמִּי. אַחֲרֵי כֵן נָסַע וְהִכְאִיב לִבִּי בְּפֵרוּדוֹ. וְעָזַב בְּקִרְבִּי גַחֲלֵי יְקוֹדוֹ.
