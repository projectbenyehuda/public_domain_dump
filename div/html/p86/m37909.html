ידי כבדה על אנחתי

יָדִי כָבְדָה עַל אַנְחָתִי וְלִבִּי נוֹאֵק בְּצִירוֹ

לְיוֹם יָבֹא שְׁאוֹן מָוֶת כְּרוֹעֶה יְבַקֵּר עֶדְרוֹ:

יִשְׁאַג וְיַעֲבֹר כְּאַרְיֵה, וְיַפְרִיד בְּעָבְרוֹ

בֵּין אִישׁ וּבֵין אָחִיו וּבֵין גֵּרוֹ.


הִתְעַתְּדוּ, בְּנֵי חֲלוֹף, בְּקוּמוֹ לַעֲרוֹץ הָאָרֶץ

כִּי הַמָּוֶת יִקְרוֹץ הֲמוֹנֵינוּ קָרֶץ

וּלְהָכִין בִּשְׁאוֹל מְעוֹנֵינוּ בְּיוֹם כָּלָה וָחָרֶץ

נִשְׁלְחָה אֲנָשִׁים לְפָנֵינוּ וְיַחְפְּרוּ לָנוּ אֶת הָאָרֶץ.


וְעַד מָה בְּנֵי אִישׁ הִתְעוֹלְלוּ עֲלִילוֹת

וּנְפָשׁוֹת עֲלוּבוֹת תְּשִׁיבוּן בְּמַעֲשֵׂי שְׁפָלוֹת?

וְכַמָּה יִקְרָא בָם לְהַמַּעֲלוֹת הַגְּדוֹלוֹת –

וְלֹא אֲבִיתֶם לַעֲלוֹת.


דַּלִּים – לְמָחָר יְגָרְשֵׁם הַזְּמָן מִמּוֹרָשָׁם

וּשְׁאוֹן הַמָּוֶת יַעֲקֹר פִּרְחָם וְשָׁרְשָׁם

וּבִשְׁאוֹל יִשְׁכְּנוּ הַגִִּבּוֹרִים אֲשֶׁר בַּשַּׁחַק רֹאשָׁם

וְגַם בְּנֵי עֲנָקִים רָאִינוּ שָׁם.


הָקִיצוּ, יְשֵׁנִים בְּחֵיק תַּאֲוַתְכֶם

וְהִתְעַתְּדוּ לְדֶרֶךְ רְחוֹקָה כִּי קְרוֹבָה נְסִיעַתְכֶם!

בְּנוּ לָכֶם בָּתִים בְּבֵית מוֹעֵד טֶרֶם מוֹתְכֶם

לָתוּר לָכֶם מָקוֹם לַחֲנֹתְכֶם.


בְּכוּ נָא לַיּוֹם בּוֹ הַנְּשָׁמָה נְשַׁמָּה

וְהַגְּוִיָּה – תַּחַת הָאָרֶץ הוּעַמָּה

וְאַתֶּם וּבָתֵּיכֶם בִּשְׁאוֹל, גְּוִיַּתְכֶם נָדַמָּה.

וּבְנֵיכֶם אֲשֶׁר לֹא יָדְעוּ הַיּוֹם טוֹב וָרָע – הֵמָּה יָבֹאוּ שָׁמָּה.


נִיב הַקְשִׁיבוּ וְהַטּוּ אֹזֶן נִסְגָּרָה

וְעִזְבוּ תֵבֵל אֲשֶׁר בָּהּ נַפְשְׁכֶם נִקְשָׁרָה

אֲחֵרִים בָּאִים לָשֶׁבֶת תַּחְתֵּיכֶם. כִּי כֵן גְּזֵרָה נִגְזָרָה

וְאַתֶּם פְּנוּ לָכֶם וּסְעוּ הַמִּדְבָּרָה!


שׁוֹכְבִים בֶּטַח – פִּתְאֹם הַזְּמָן הֲדִימָּם

וּבְבוֹא קִצָּם בְּכוֹר מָוֶת הֲשִׁימָם

וְגַם יַד יְיָ הָיְתָה בָם לְהֻמָּם

מִקֶּרֶב הַמַּחֲנֶה עַד תֻּמָּם.


לִשְׁאוֹל יִסְחֲרוּ הָעֲנָקִים וְהַגִּבּוֹרִים

וְלֹא יִנָּצְלוּ מִמֶּנּוּ הַנִּבְצָרִים בִּמְצָרִים

וְהַמָּוֶת יוֹצִיא מִבֵּיתָם הָרְשָׁעִים הַשּׁוֹכְבִים בַּחֲדָרִים

וְהָעַוִּים הַיּוֹשְׁבִים בַּחֲצֵרִים.


מַה לְּךָ, בֶּן־חֲלוֹף, תִּשְׁגֶּה בְּתַעֲנוּג אֲדָמָה?

עוּרָה וְהִנָּצֵל מִפְּנֵי הָאַף וְהַחֵמָה

הַכְנִיעַ יִצְרְךָ בְּלֵב אַמִּיץ וּבְיָד רָמָה

וְהִתְגָּר בּוֹ מִלְחָמָה!


הוֹי הַהוֹלֵךְ שׁוֹבָב כָּל יְמֵי נִצְחוֹ!

הֲלֹא יִזְכֹּר יוֹם יֶחְשַׁךְ שִׁמְשׁוֹ וִירֵחוֹ

וְיִרְאֶה חֲכָמִים יָמוּתוּ – וְהוּא יָעֵז מִצְחוֹ

כִּי הִקְשָׁה יְיָ אֱלֹהֶיךָ אֶת רוּחוֹ.


בְּכֵן עַל חֲטָאֵינוּ נָהִים בְּשִׂיחֵנוּ וְנָרִיד

כִּי לֹא זָכַרְנוּ הַמָּוֶת אֲשֶׁר בֵּין אַחִים יַפְרִיד

אָבַדְנוּ שְׂכָרֵנוּ וְטוּבֵינוּ וְשָׁכַחְנוּ יוֹם דִּין הַמַּחֲרִיד

לֹא הִשְׁאַרְנוּ שָׂרִיד.


נָטִינוּ מִדַּרְכֵי אֵל וְאָטַמְנוּ אָזְנֵינוּ

עַל כֵּן גְּמָלָנוּ יוֹצְרֵנוּ כְּפִי זְדוֹנֵינוּ

וְשִׁלַּם לְחֵיקֵינוּ רוֹעַ מַעֲשֵׂינוּ וַעֲווֹנֵינוּ

אֶת הַכֹּל נָתַן יְיָ אֱלֹהֵינוּ לְפָנֵינוּ.


חֶרְדַּת יוֹם הַדִּין תּוֹצִיאֲכֶם מֵהֵיכְלֵיכֶם

וּבְלִי לָשׁוֹן יִקְרָא הַמָּוֶת אֲלֵיכֶם:

דְּעוּ, כִּי מָחָר תִּסְעוּן לִשְׁאוֹל מֵאָהֳלֵיכֶם

וְשַׁבְתֶּם אִישׁ לִירֻשָּׁתוֹ אֲשֶׁר נָתַתִּי לָכֶם.


רֹאשְׁךָ הַשְׁפִּיל, בֶּן־אִישׁ, וּזְכוֹר יוֹם אֵל

אוּלַי תִּזְכֶּה נַפְשְׁךָ לַחֲזוֹת בְּנֹעַם אֵל

וּבִרְצוֹתְךָ לְהִתְוַכֵּחַ מִפַּחַד אֵל

עֵינֶיךָ הָרוֹאוֹת אֶת כָּל אֲשֶׁר עָשָׂה יְיָ אֵל.


יוֹם יָבֹא הַקֵּץ בְּחֵמוֹת שְׁפוּכוֹת

וְיִלָּחֵם עִם בְּנֵי אִישׁ וְקַשְּׁתוֹתָיו דְּרוּכוֹת

וְכַאֲשֶׁר הֶאֱבִיד הָרִאשׁוֹנִים מַעֲרָכוֹת מוּל מַעֲרָכוֹת

יַעֲשֶׂה יְיָ לְכָל הַמַּמְלָכוֹת.


זְרוֹעַ רָמָה גַּלּוּ וְהַיְשִׁירוּ מִפְעֲלֵיכֶם

וְהִלָּחֲמוּ עִם הַיֵּצֶר הַמִּתְגּוֹלֵל עֲלֵיכֶם

וְאַל תַּעַרְצוּ מִפָּנָיו, כִּי רַבּוּ מֵחֲיָלָיו חֲיָלֵיכֶם

כִּי יְיָ אֱלֹהֵיכֶם הוּא הַנִּלְחָם לָכֶם.
