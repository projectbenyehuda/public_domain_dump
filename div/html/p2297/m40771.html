מָוֶת

עַל־מָה בְּנֵי תֵּבֵל / בּוֹכִים בְּרָב־אֵבֶל / כִּבְנַהֲרוֹת בָּבֶל / נִשְׁמַע קוֹל יְלָלָה.

הַקּוֹל לְמֵרָחוֹק / נִשְׁמַע, וְלִבְלִי־חֹק / פִּי נִמְלְאָה בִשְׂחֹק / בִּשְׁמֹעַ אֲמֻלָה.

לֹא יַחֲטִיא דַּעְתִּי / לַחְשֹׁב תְּבוּנָתִי / מֶנִּי וּמֵאִתִּי / נֶעֶֽשְׂתָה נְבָלָה.

חַרְבִּי מְשׂוֹשׂ לִבִּי / שָׁבָה צְבִי שֶׁבִי / חָכָם וְאִישׁ נָבִיא / שָׁתָה כוֹס רְעָלָה.


חַיִּים

מַה־לְךָ פְּאֵר אַכְזָר / אִם אֵל לְךָ גָזַר / נֶעֱזָר כְּעֶבֶד זָר / אֵיךְ עַצְמְךָ תְּפַנֵּק

לִשְׁמֹר דְּבַר שַׁדַּי / כָּל־חַֽיְתוֹ שָׂדַי / יָעוּף כְּזִיז שָׂדַי / וּכְאַרְיֵה יְזַנֵּק.

אַךְ רַע לְרַע יֻשְׁלָח / לֹא זֶה וְזֶה יִצְלָח / כַּצִי אֲשֶׁר מַלָּח / בּוֹ עוֹלֵל וְיוֹנֵק.

קוּם בּוֹשׁ לְבשׁ חֶרְפָּה / מוּכָן לְמַדְחֵפָה / לִסְקֹל וְלִשְׂרֵפָה / וּלְהֶרֶג וְחָנֶק.


מָוֶת

מַה־לָּךְ לְהַרְחִיב פֶּה / נִסְפֶּה וְלֹא תִצְפֶּה / כִּי אֵין לְךָ מַרְפֵּא / מִצִּיר מַחֲלָתִי.

אֶקְצֹר אֲשֶׁר תִּזְרַע / תַּרְבֶּה אֲנִי אֶגְרַע / דָּת יוֹעֲצִים אֶפְרַע / תִּתְקַיֵּם עֲצָתִי.

אֵין דַּל וְאֵין עָשִׁיר / עֵת בֹּא עֲדַי יָשִׁיר / אַף מַעֲשָׂיו יַכְשִׁיר / יִלָּכֵד בְּרִשְׁתִּי.

דֹּם כַּהֲדֹם רַגְלַי / מִדְרָס לְמַעְגָּלַי / מַהֵר וּבוֹא אֵלַי / תִּשְׁתֶּה כּוֹס חֲמָתִי.


הַיּוֹם שְׁבָה בָּדָד / כִּי אָהֳלָךְ שֻׁדָּד / רֹאשׁ דָּת כְּעוֹף נֹדָד / צֻדַּד תּוֹךְ מְצוֹדִי.

משֶׁה זְכוּתוֹ רַב / לֻמַּד כְרַב לַקְרָב / בִּי רָב וְלֹא עָרַב / אֵלָיו חֵץ כְּבֵדִי.

נִלְחַם בְּעֹז כֹּחוֹ / שַׁח שַׁךְ נְתִיב אָרְחוֹ / סוֹף נֶאֱסַף רוּחוֹ / וּמְסָרוֹ בְיָדִי.

הוּא עָף מְקוֹם עַפְעַף / רָץ רָץ וְלֹא יִיעָף / עַל־זֹאת לְבַד נוֹעָף / כִּי נִבְהַל בְּפַחְדִּי.


חַיִּים (בינו לבין עצמו)

אוֹי אוֹי לְשֶׁבֶר זֶה / אֵיךְ עַל־אֲבִי חוֹזֶה / מַזֶּה וּבֶן־מַזֶּה / שָׁלַט בֶּן־זְדוֹנִים

גָּדוֹל וְרָם יַשְׁפִּיל / הַר לַעֲלוֹת יַעְפִּיל / נָזִיר כְּזֵיר יַפִּיל / יִבְנֶה הַלְּבֵנִים

בַּר בַּר יְשַׁלַח בַּר / יִשְׁבַּר לְבַב גָבָר / שׁוֹסֶה גְרָנוֹת בָּר / לִטַּע קִמְּשׁוֹנִים

אֵיךְ רֹאשׁ גְּדוּד שָׁדוּד / נָפַל בְּלִי פְדוּת / נִשְׁבָּה אֲרוֹן עֵדוּת / מִשַּׁעַר זְקֵנִים.


(אל פני המות)

אוֹיְבִי וְאִישׁ רִיבִי / לָבִיא עֲלֵי לִבִּי / שָׂשׂ דָּץ בְּמַכְאוֹבִי / עַל־חוֹבִי מְשׂוֹשָׂךְ.

אֵיךְ רִיב לְמוּל אָדוֹן / תִּדּוֹן בְּלֵב זָדוֹן / תִּשְׁלַח בְּפִקָּדוֹן / הָאֵל יָד מְבִישָׁךְ.

אֵיךְ תַחֲבשׁ לִכְבּשׁ / עִיר צוּר, וְלֹא תַבשׁ / גַּסּוּת כְּסוּת תִּלְבּשׁ / וּלְבוּשָׁךְ לְבוּשָׁךְ.

לֹא לֹא כְגוֹרָלָךְ / מַלְאָךְ כְּרוּב הָלָךְ / דֹּם מִזְּדוֹן מִלָּךְ / כִּי הוּא יַחֲרִישָׁךְ.


מָוֶת

עַל־מָה דְבָרִים עוֹד / הֶבֶל וְרִיק לִסְעֹד / לַעַג וּבוּז תִּסְעֹד / עַצְמֶךָ תְּפַתֶּה.

כִּי כָל־בְּשַׂר הָחַי / יִפֹּל בְּתוֹךְ שׁוּחַי / הוֹלֵךְ לְעֵדֶן חַי / לִשְׁאוֹלָה וְתָפְתֶּה.

אֵין עִיר וְאֵין מִקְלָט / אֵין נָס וְאֵין נִמְלָט / נִגְזַר וְכֵן נֶחְלַט / עַל־עָרוּם וּפוֹתֶה.

יִפֹּל בְּמִכְמַרְתִּי / בֶּן־צוּף וְאֶפְרָתִי / אֶת־כּוֹס רְעָלָתִי / כֹּל יִרְוֶה וְיִשְׁתֶּה.


חַיִּים

דֹּם דֹּם שְׁתֹק שׁוֹטֶה / חוֹטֵא בְּעֵת בּוֹטֵא / עוֹטֶה וּמִתְחַטֵּא / עַל־פֶּשַׁע וְזָדוֹן.

אִם מוֹֽסְדֵי תֵבֵל / מָטוּ כְּצִיץ נוֹבֵל / יִבְכֶּה וְיִתְאַבֵּל / עֶבֶד שַׂר וְאָדוֹן.

אִם בַּאֲרָזִים עָל / לַהַב מְלֹא שֹׁעַל / מַה־יַעֲשֶׂה יִפְעַל / אֵזוֹב קִיר כְּנָדוֹן.

דָּרֵי זְבוּל יִדְכּוּ / כִּי לֹא בְדִּין יִזְכּוּ / אִם כֵּן בְּמַר יִבְכּוּ / מָוֶת וַאֲבָדוֹן.


מָוֶת

אוֹדֶה עֲלֵי פִשְׁעִי / כִּי פָֽשְׂתָה נִגְעִי / עַל־שַׂר וְאִישׁ שׁוֹעִי / אוֹר עַמִּים וְצִלָּם.

מַעְלָה וּמַטָּה הֵם / בּוֹכִים לְמִינֵיהֶם / עָלַי בְחֶבְלֵיהֶם / מַשְׁמִיעִים בְּקוֹלָם.

אוֹי לִי לְאָן אֶבְרַח / אִם מַעֲרָב מִזְרָח / אֶזְרַח בְּתוֹךְ זֶרַח / אוֹ אֶעְלַם בְּעֵילָם.

כָּפוּף בְּלֵב מַסְעִיר / אֵלֵךְ לְאָח צָעִיר / עִמּוֹ נְהִי אָעִיר / נֶאְלָם בּוֹשׁ וְנִכְלָם.


מָוֶת חַיִּים

קִרְאוּ בְקוֹל עַנּוֹת / תַּנּוֹת בְּנוֹת בִּינוֹת / קִינוֹת כְּמוֹ תַנּוֹת / מִסְפֵּד מַר כְּתַנִּים

שִׁבְרוּ בְנֵי־תֵבֵל / נֵבֶל כְּמוֹ־נֶבֶל / חֵבֶל וְרָב־אֵבֶל / לֹא עוּגָב וּמִנִּים.

שׁוֹלָל מְעִיל מַעְלָה / עַל־מִוֹעֲלֵי מַעְלָה / אֶדֶר יְקָר גָּלָה / וּלְבוּשׁ הַסְּדִינִים.

הַיּוֹם בְּחַטַּאתְכֶם / לֻקַּח וְסָר מִכֶּם / מַעְדֶּה וּמַלְבִּישְׁכֶם / שָׁנִי עִם־עֲדָנִים.


נֵר מַעֲרָב כָּבָה / רֹאשׁ דָּת וְשַׂר צָבָא / שָׁב מִצְּבֹא צָבָא / גַּם לֹא יַעֲבֹד עוֹד.

כִּי לַעֲבֹד עֶלְיוֹן / הָלַךְ לְרוּם חֶבְיוֹן / שָׁמָּה בְאַפִּרְיוֹן / לֹא יָמוּשׁ וְיִמְעֹד.

אַךְ לִי כְאֵב חֹלִי / סָר שַׂר וְרֹאשׁ חֵילִי / מַלְכִּי מְנַת חֶבְלִי / לִי מַשְׁעֵן וּמִסְעוֹד.

שָׁקַע וּבָא שִׁמְשִׁי / שֶׁאָהֲבָה נַפְשִׁי / הוּמַר בְּמַר דִּבְשִׁי /רוֹשׁ אֹכַל וְאֶסְעֹד.


רַב טוֹב לְיִשְׂרָאֵל / מֵשִׁיב לְכָל־שׁוֹאֵל / בָּקִי בְרָזֵי אֵל / וּבְסִתְרֵי בְרֵאשִׁית.

יָדַע בְּכָל־מֻפְלֶא / נִגְלֶה וְהוּא פֶלֶא / עוֹלֶה וּמִתְעַלֶּה / עַל־דַּעַת אֱנוּשִׁית.

וּדְבַר אֱמֶת מוֹרֶה / תֶּבֶן וְקַשׁ זוֹרֶה / מוֹרֶה לְכָל־מוֹרֶה / אֶל־דֹּפֶן חֲמִישִׁית.

תּוֹעֶה בְגֶלֲלוֹ / נִמְכַּר לְעֶבֶד לוֹ / הוֹצִיא בְרֹב חֵילוֹ / לַחֵרוּת וְחָפְשִׁית.


מֵלִיץ וּמִן־אֶלֶף / יָחִיד וְאֵין חֵלֶף / מִן־תָּיו וְעַד־אָלֶף / בָּנָה נַחֲלַת־הוֹן.

לֹא יֵשׁ דְּיוֹ דַיּוֹ / לִכְתֹּב פְּאֵר עֶדְיוֹ / גָּבַהּ צְבִי עֶדְיוֹ / שָׂמָהוּ לְגָאוֹן.

הֵשִׁיב בְּדָרוּשָׁיו / לֵב עַם לְצוּר עוֹשָׂיו / דָּתָן בְּדָת עַכְשָׁיו / בֶּן־פֶּלֶת וְהוּא אוֹן.

הִשְׁפִּיל זְדוֹן קָמִים / הַלּוֹחֲמִים תַּמִּים / הִצְלִיל בְּלֵב יַמִּים / כָּל־סוֹאֵן בְּשָׁאוֹן.


רוֹעֶה אֲשֶׁר צוֹעֶה / אֵיךְ צֹאן בְּלִי מִרְעֶה / תִּטֹּשׁ וְלֹא תִשְׁעֶה / תּוֹעֶה עַל־גְּבָעוֹת.

שֶׂה שׁוּר בְּנִקְרַת־צוּר / עָצוּר בְּאֵין עָצוּר / נָצוּר וְאֵין לָהּ צוּר / גַּם פּוֹעֵל יְשׁוּעוֹת.

תִּשְׁמַע וְלֹא תָחוּשׁ / תִּפְנֶה וְלֹא תָחוּשׁ / תִּרְאֶה וְלֹא תָחוּס / אֶל־בֶּכִי וְשַׁוְעוֹת.

רַב עַל־פְּרִידָתְךָ / גַּם עַל־פְּרִי דָתְךָ / שׁוֹתָה פְרִידָתְךָ / בַּשָּׁלִישׁ דְּמָעוֹת.


מִי זֶה יְנַחֵם עַם / נִפְעָם בְּיוֹם זַעַם / לִבּוֹ כְּמוֹ־פָעַם / הוֹלֵם מַהֲלֻמּוֹת.

בּוֹכֶה לְדוֹר יָתוֹם / בִּלְתִּי מְתֹם נִתֹּם / חָזוֹן חֲתֹם וּסְתֹם / חָכְמוֹת נֶעֱלָמוֹת.

כִּי נָֽפְלוּ רָאשִׁים / עִירִין וְקַדִּישִׁין / שָׂרִים וְשָׁלִישִׁים / כָּל־דּוֹרֵשׁ רְשׁוּמוֹת.

נִשְׁאַר אֱנוֹשׁ חֻלִּין / מַכְבִּיר לְבַד מִלִּין / בִּיקָר וְלֹא יָלִין / נִמְשַׁל כַּבְּהֵמוֹת.


אַךְ זֹאת יְנַחֵם כֹּל / אֵשׁ זֶה לְכָל־אֶשְכֹּל / חָכָם וְרֹב יָכֹל / עִם־תֻּמִּים וְאֻרִים.

צִיר נֶאֱמָן נִפְטָר / דּוֹד בִּכְלִיל עֻטָּר / כִּי כֵן חָתוּם בִּשְׁטָר / עַל־כָּל הַֽיְצוּרִים.

הַכֹּל זְמָן יִכְלֶה / עָשִׁיר וְרָשׁ נִקְלֶה / שׁוֹקֵט וְגַם גּוֹלֶה / יוֹשֵׁב עַל־שְׁמָרִים.

אָכֵן לְאִישׁ יָשָׁר / יֶחְסַר לְבַד בָּשָׂר / רוּחוֹ אֲזַי יִכְשַׁר / לָצֵאת לַדְּרוֹרִים.


לִרְאוֹת אֲשֶׁר פָּעַל / מַעְלָה וְרוֹם יַעַל / שַׁעַר בְּלִי נִגְעָל / וּדְרָכִים גְּבֹהִים.

יִרְאֶה וְיִתְפַּלֵּא / טוֹב אוֹֽצְרוֹת מָלֵא / צָפוּן בְּאוֹר נִגְלֶה / וּדְבָרִים תְּמֵהִים.

שָׁמָּה לְקַבֵּל לוֹ / בָּאִים בְּנֵי גִילוֹ / אוֹתוֹ לְהֵיכָלוֹ / יִלְווּהוּ כְמֵהִים.

כִּי שָׁם בְּהוֹד נִגְלוּ / יִרְאוּ וְיִבָּהְלוּ / עֵת עוֹד לְרוּם יַעְלוּ / לָהֶם הָאֱלֹהִים.


שָׁמָּה בְּרוּם חֶבְיוֹן / יֶחְזוּ לְאֵל עֶלְיוֹן / יִרְאוּ בְּלִי דִמְיוֹן / אוֹר זִיו הַשְּׁכִינָה.

אָז יִכְרְעוּ נֶגְדּוֹ / גַּם יִשְׁאֲלוּ חַסְדּוֹ / אֶל־עַם וְצֹאן יָדוֹ / חֶמְלָה וַחֲנִינָה.

יַחְמֹל לְנִשְׁאָרִים / יַכְרִית פְּנֵי זָרִים / יִקְבֹּץ מְפֻזָּרִים / מִכָּל־עִיר וּפִנָּה.

“דַּי דַּי” לְיוֹם רָעָה / יֹאמַר “וְאַל שַׁוְעָה” / הַזּוֹרְעִים דִּמְעָה / יִקְצֹרוּ בְרִנָּה.


(1697)
