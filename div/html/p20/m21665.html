שִׁמְעוּ נָא אֶל דִּבְרֵי הָרוֹפֵא

הנהגת הבריאות


שִׁמְעוּ נָא אֶל דִּבְרֵי הָרוֹפֵא

עַם נָבוֹן, הֶחָזָק וְהָרָפֶה,

כִּי בֵאֵר אֶל כָּל מַחֲלָה מַרְפֵּא,

מֵחֹדֶשׁ אֶל חֹדֶשׁ אֵיךְ יִצְפֶּה,

אִישׁ אִישׁ אֶת גּוּפוֹ, וְאַף אֶת הַפֶּה,

בַּעֲבוּר יִהְיֶה בָרִיא גַם יָפֶה.


יָפֶה הָאִישׁ יִהְיֶה,

רַבּוֹת שָׁנִים יִחְיֶה,

אַךְ יִשְׁמֹר מִצְוֹתָיו,

חֻקּוֹתָיו וְתוֹרוֹתָיו.


תִּשְׁרֵי הוּא הָרִאשׁוֹן לֶחֳדָשִׁים,

בּוֹ יְנַשְׁבוּ הָרוּחוֹת הַקָּשִׁים,

וִיקָרְרוּ אֶת גּוּפֵי הָאֲנָשִׁים,

רָאוּי לֶאֱכוֹל בּוֹ אֶת הַכְּרֵשִׁים,

כָּל חָרִיף זֶרַע גַּם שָׁרָשִׁים,

וּלְחַזֵּק הַכְּלָיוֹת הָרְפוּשִׁים.


אֶת הַכְּלָיוֹת טַהֵר

מִן הַלֵּחָה מַהֵר,

וּלְהַמְתִּיק אֶת הַדָּם

חָלָב יִשְׁתֶּה אָדָם.


מַרְחֶשְׁוָן חֹדֶשׁ שֵׁנִי נִקְרָא,

בּוֹ יַעְדִּיף לֶעָנִי הַצָּרָה,

מִבְּלִי אֵשׁ וּמַלְבּוּשׁ בַּקָּרָה,

בּוֹ תִּגְבַּר הַמָּרָה הַשְּׁחוֹרָה,

כָּל מֶרְחָץ תִּקְרָא לוֹ מֵי מָרָה,

פֶּן תִּיגַע פֶּן תִּיעַף הִוָּסְרָה!


בֹּקֶר תִּשְׁתֶּה תִירוֹשׁ,

שֵׁכָר מִדְּבַשׁ תִּדְרֹשׁ,

הַשּׁוּמִים וְהַבְּצָלִים

תֹּאכַל, גַּם הַמְּשַׁלְשְׁלִים.


כִּסְלֵו הוּא הַחֹדֶשׁ הַשְּׁלִישִׁי,

יִרְבֶּה בּוֹ הַמִּנְהָג בֶּן נִמְשִׁי,

וּשְׁחִין רַע שֶׁיֵּרָפֵא בִקְשִׁי,

בְּאִישׁ שָׁחוּר, אַךְ הַלָּבָן חָפְשִׁי,

תַּרְנְגוֹל וּבְשַׂר כֶּבֶשׂ אַל תֶּשִׁי,

בָּקָר וּבְשַׂר עֵז שָׂנְאָה נַפְשִׁי.


רַב קִנָּמוֹן תֹּאכַל,

חֹם מֶרְחָץ לֹא תוּכַל,

תַּקִּיז דָּם בַּצְּפָרִים,

וְגַם תִּרְוֶה מַמְרוֹרִים.


טֵבֵת הוּא הָרְבִיעִי לַמּוֹנִים,

הוּא מוֹעֵד הַזָּמִיר לַגְּפַנִּים,

תֹּאכַל בּוֹ בָשָׂר מִמַּשְׁמַנִּים,

תִּשְׁתֶּה הַמְרֻקָּחִים מִיֵּינִים,

גַּם תִּרְחַץ בַּמֶּרְחָץ רַב מוֹנִים,

תָּקֹץ מִשְׁכַּב נָשִׁים מֵעֲדָנִים.


מַרְפֵּא בוֹ לְכָל מַחֲלָה,

וְאַף יֶרֶק לַאֲכִילָה,

אַךְ הִשָּׁמֵר לָרֹב,

שֶׁלֹּא תֹאכַל הַכְּרוּב.


שְׁבָט אַל תֹּאכַל בּוֹ אֶת הַתְּרָדִים,

וּצְנוֹנוֹת, גַּם דָּגִים נֶאֱחָזִים,

כִּי בוֹ קֹר עִם לֵחָה נִצְמָדִים,

וְהַמַּיִם בָּאָרֶץ נֶאֱחָדִים,

גַּם בַּגּוּף בֶּחָזֶה נוֹעָדִים,

לֶאֱכָל בּוֹ חֲרִיפִים הוּא דִין וְחַדִּים.


פִּלְפֵּל שׁוּם כָּל חַמִּים,

וְאַל תּוֹצִיאוּ דָּמִים,

וְאַל תִּשְׁתּוּ הַמַּיִם,

פֶּן תַּהֲרֹס מֵעַיִם.


אֲדָר בּוֹ אֶת מֵעֶיךָ בַּמַּשְׁקֶה

הַמְטַהֵר אֶת הַכֶּרֶם תְּנַקֶּה;

אֶת הַדָּם הַנּוֹסָף חִישׁ נַקֵּה,

מֵאֶצְבַּע הַגּוּדָל יִנָּקֶה,

הַקָּרֵב אֶל הָאִשָּׁה יִלְקֶה,

וְאַל תֹּאכַל סִלְקִי גַּם כָּל יַרְקִי.


וּתְשַׁלְשֵׁל הַשַּׁחַת,

גַּם רַחֵק תִּגְלַחַת,

זֶרַע כַּרְפַּס תִּטְעָם,

עִם קֻנְדִּיטוּן יִנְעָם.


נִיסָן הוּא מוֹעֵד הַצִּמְחוֹנִים

בָּאָרֶץ, גַּם מוֹעֵד שׁוֹשַׁנִּים;

בּוֹ יִרְבּוּ הַסְּגוֹרוֹת וְהַשְּׁחִינִים

בַּבָּשָׂר, וּנְגָעִים בַּגְּרוֹנִים,

מֵרֹב לֵחָה עִם דַּם הַשּׁוֹכְנִים

בַּבַּחוּרִים, אַךְ לֹא בַזְּקֵנִים.


מֶרְחָץ יָבֵשׁ תִּבְחָר,

בָּשָׂר צְלִי תִּשְׁחָר,

תַּקִּיז דָּם וּתְשַׁלְּשֵׁל

אַךְ הַרְבּוֹת הוּאר כֹשֵׁל.


אִיָּר בּוֹ יַתְחִילוּ הַקְּצִירִים,

תּוֹךְ אֶרֶץ הַצְּבִי מִן הַשְּׂעוֹרִים;

בּוֹ הַדָּם עִם לֵחָה מִתְעוֹרְרִים,

כֹּבֶד רֹאשׁ עִם דִּמְעַת סַנְוֵרִים;

כָּל רֹאשׁ לֹא תֹאכַל גַּם הַמְּקָרְרִים

וּכְרָעַיִם רָעִים לַגְּבָרִים.


תַקִּיז דָּם מִן הַשְּׂמֹאל,

וְעַל בִּטְנְךָ לֹא תַחְמֹל,

גַּם כָּל דָּבָר נֶחְמָץ

לֹא תֹאכַל, אָז תֶּאֱמָץ.


סִיוָן בּוֹ גַּם תּוֹלְדוֹת הָעֲנָבִים,

גַּם מַדְוִים חַדִּים הַמַּכְאִיבִים,

לָכֵן אֵלֶּה יִהְיוּ לְךָ לְאֹיְבִים,

מַאֲכָל חַם מֶרְחָץ גַּם מִשְׁכָּבִים

עִם נָשִׁים, כִּי כֻלָּם מַעֲצִיבִים,

כִּי דָם עִם הָאֲדֻמָּה מִתְעָרְבִים.


תַּקִּיז דָּם מִכָּבֵד,

וּתְשַׁלְשֵׁל פֶּן תֹּאבֵד,

זֶרַע שׁוֹמֵר לַעֲנָה,

עִם יַיִן תִּשְׁתֶּה נָא.


תַּמּוּז בּוֹ יִרְבֶּה הַחֹם בְּגוּפִים,

עִם יֹבֶשׁ שֶׁיִבְעַר כִּרְשָׁפִים,

הִשָּׁמֵר פֶּן תֹּאכַל הַחֲרִיפִים,

וְתֹאכַל מַאֲכָל קַר גַּם כָּל רָפִים

הַתְּמָרִים, וּפְרִי רַךְ מֵעֲנָפִים,

קָרִים תִּשְׁתֶה בֹקֶר כָּעֲיֵפִים.


תִּשְׁתֶּה שֵׁכָר שְׂעוֹרִים,

גַּם תִּרְחַץ בַּקָּרִים,

אִם אֶל אִשָּׁה תִּקְרָב,

עִם שׁוֹנִים תַּתְעָרָב.


אָב הוּא חֹדֶשׁ חֶזְקַת הַחַמָּה,

יַעְדִּיף בּוֹ הַמָּרָה הָאֲדֻמָּה,

וְשׁוּם וּצְנוֹן וּבְצָלִים מֵהֵמָּה,

אַל תֹּאכַל פֶּן תִּגְבַּר הַחֵמָה,

וְגַם תַּחֲרֵב אֶת לֵחַת הַנְּשִׁימָה,

עֵת חַרְבוֹנֵי קַיִץ עִם צָמָא.


מִצּוּי כַּרְפַּס תַּשְׁקֶה

בִטְנְךָ, דֹּמֶן תְּנַקֶּה,

וּלְצָיוֹן יֵאָמֵר,

מִן אִשָּׁה הִשָּׁמֵר.


אֱלוּל הִנֵּה הוּא אַחֲרוֹן לַיְרָחִים,

בּוֹ יִפְּלוּ וְגַם יִבְלוּ הַצְּמָחִים,

הָרוּחוֹת הֶעָבִים מַטְרִיחִים,

וְיִתְחַלְּפוּ הָעִתִּים הַצַּחִים.

גַּם גּוּפֵי בַחוּרִים נִקְדָּחִים,

הִזָּהֲרוּ פֶּן תֹּאכְלוּ אֲבַטִּיחִים.


כָּל מַרְבִּים הַמָּרָה,

טָמֵא לָמוֹ תִּקְרָא,

אַךְ תֹּאכַל הַחַמִּים,

הַבְּצָלִים וְהַשּׁוּמִים.
