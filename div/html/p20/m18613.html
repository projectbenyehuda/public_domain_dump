יְיָ שִׁמְךָ לְעוֹלָם

מוסתג׳אב

יְיָ שִׁמְךָ לְעוֹלָם.

יְיָ אֶחָד שֶׁלֹּא יִשְׁנֶה, / שִׁמְךָ אוֹר מִחֹשֶׁךְ קוֹנֶה,

אָמַרְתִּי: עוֹלָם חֶסֶד יִבָּנֶה / בְּשֵׁם יְיָ אֵל עוֹלָם.

יְיָ, בְּךָ נִתְכְּנוּ עִנְיָנָי, / שִׁמְךָ בְּזָכְרִי – אוֹרוּ עֵינָי;

בָּרֲכִי נַפְשִׁי אֶת יְיָ / וּבָרוּךְ שֵׁם כְּבוֹדוֹ לְעוֹלָם.

יְיָ, רְצוֹנִי כִּי תִקָּחֵנִי; / שִׁמְךָ רַחוּם אַט יוֹכִיחֵנִי,

רוּחֲךָ טוֹבָה תַּנְחֵנִי / וַאֲבָרְכָה שִׁמְךָ לְעוֹלָם.

יְיָ, הוֹכַחְתָּ מִי תֶאֱהֲבֶנּוּ, / שִׁמְךָ הוּא בְאֶחָד – וּמִי יְשִׁיבֶנּוּ?

הָאֹמְרִים לָאֵל: סוּר מִמֶּנּוּ! – / שְׁמָם מָחִיתָ לְעוֹלָם.

יְיָ, מִמְּךָ אֶשְׁאַל לְעוֹדְדֵנִי, / שִׁמְךָ מָסַרְתָּ לַנְּבִיאִים לְהַגִּידֵנִי:

מִי כָמוֹנִי וּמִי יֹעִידֶנִּי? / זֶה שְּׁמִי לְעוֹלָם!

יְיָ, חֲנִינוֹתֶיךָ תָּמִיד בְּאֵין קָרֶץ, / שִׁמְךָ חוֹשֵׁב וְעוֹשֶׂה מַה יִּרֶץ,

חַסְדְּךָ, יְיָ, מָלְאָה הָאָרֶץ – / יְהִי שֵׁם יְיָ מְבֹרָךְ לְעוֹלָם!

יְיָ, זְמִירוֹתֶיךָ בְּאֵין מִסְפַָּר, / שִׁמְךָ – זוֹכְרוֹ בֶאֱמֶת בַּל יֶחְפָּר, זֶרַע יַעַבְדֶנּוּ יְסֻפָּר – / יְהִי שְׁמוֹ לְעוֹלָם.

יְיָ קוֹרֵא מֵרֹאשׁ הֲמֻלּוֹת, / שִׁמְךָ קֵרַבְתָּ לִנְפָשׁוֹת בְּמִלּוֹת:

קָדוֹשׁ יוֹשֵׁב תְּהִלּוֹת, / וְיִגְדַּל שִׁמְךָ לְעוֹלָם.
