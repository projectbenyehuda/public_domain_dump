אָקוּם בַּחֲצוֹת לַיְלָה

חתימה: אלשבזי משתא שמו


אָקוּם בַּחֲצוֹת לַיְלָה / אָשִׁיר שִׁיר לַאדוֹנִי

הַמְרוֹמָם עַל תְּהִלָּה / וְהוּא אֶחָד וְאֵין שֵׁנִי

צוּר שׁוֹמֵעַ תְּפִלָּה / וְנִיב שְׂפָתַי וּלְשׁוֹנִי

יָחִישׁ לְקֵץ הַגְּאֻלָּה / לְקַבֵּץ שִׁבְטֵי הֲמוֹנִי

בִּזְכוּת עוֹקֵד וְנֶעְקָד / אֱלֹהַי הוֹשִׁיעֵנִי:


לִבְנוֹ יְחִידוֹ יִצְחָק / עָקַד לִרְצוֹן קוֹנוֹ

לְהַעֲלוֹתוֹ נִתְחַזָּק / וּבָדַק אֶת סַכִּינוֹ

וְהִנֵּה אַיִל נִדְחָק / נֶאֱחַז בַּסְּבָךְ בְּקַרְנוֹ

קְרָאָהוּ יָהּ מִמֶּרְחָק / לְהַקְרִיבוֹ תַּחַת בְּנוֹ

וְכָרַת לוֹ בְּהַר הַמּוֹר / בְּרִית עוֹלָם לְזָכְרֵנִי:


שְׁנֵיהֶם עָמְדוּ בַּנֵּס / בְּיוֹם שֶׁהָלְכוּ יַחְדָּיו

יְדִיד יְוֹיָ מִשְׁתַּנֵּס / וְחַרְבּוֹ תַּחַת לְמַדָּיו

וְיִצְחָק מִבְּלִי אוֹנֵס / עִמּוֹ לְהַחֲזִיק יָדָיו

וַעֲצֵי עוֹלָה תוֹפֵס / וְשׁוֹמֵר דָּת תַּפְקִידָיו

וּשְׁנֵי נְעָרָיו עִמּוֹ / הִנְהִיגָם אַב הֲמוֹנִי:


בַּיּוֹם הַשְּׁלִישִׁי לְלֶכְתּוֹ / לְאֶרֶץ הַמּוֹרִיָּה

נִתְכַּוֵּן בְּמַחְשַׁבְתּוֹ / בְּחוֹשֶׁק לָבִיא וְלִבְיָה

לְמֵרָחוֹק שָׂם דַּעְתּוֹ / וַיְוֹיָ שָׁם הָיָה

נָם לִנְעָרָיו: שְׁבוּ פֹּה / וְיִצְחָק בְּנִי יְלַוֵּנִי:


זָכוּ שְׁנֵיהֶם לַעֲלוֹת / בְּמַעְלָה הָעֶלְיוֹנָה

וַיְבָרְכֵם נוֹרָא תְּהִלּוֹת / וּבָאוּ עַד חֶבְרוֹנָה

וְהִעְמִיד מֵהֶם קְהִלּוֹת / שְׁבָטִים בְּנֵי צִיּוֹנָה

זֶרַע יַעֲקֹב אִישׁ תָּם / סְגֻלַּת אֵל חַי קוֹנִי:


יִזְכּוֹר לָנוּ צוּרִי / זְכוּת חֵן שְׁלוֹשָׁה אָבוֹת

בְּיוֹם דִּין יָאִיר נֵרִי / וְיָסִיר כָּל מַכְאוֹבוֹת

וִיקַנֵּא בַּעֲבוּרִי / וִישַׂמֵּחַ לְבָבוֹת

וְיוֹצִיא לָאוֹר דִּינִי / וְיִמְחֶה אֶת כָּל עֲווֹנִי:


מִבּוֹר גָּלוּת יִדְלֶה / לְעַם קָדְשׁוֹ שֶׁטָּבַע

בִּזְכוּת אָבוֹת יִתְלֶה / אֲשֶׁר הֵם בְּקִרְיַת אַרְבַּע

וְעֵת רָצוֹן יְגַלֶּה / וְיוֹסִיף חֵן בַּת שֶׁבַע

וִיקַבֵּל הַשָּׁבִים / וִימִינוֹ תִּסְעָדֵנִי:


שַׁאֲלִי בַת הַיְּדִידָה / בְּיוֹם זֶה מֵאֱלֹהַיִךְ

סְלִיחָה, וּתְנִי תּוֹדָה / וְרַצִּי עַל בָּנַיִךְ

תָּגֵלְנָה בְּנוֹת יְהוּדָה / בְּקִבּוּץ מְפֻזָּרַיִךְ

וְאֶשְׁמַע כִּי בָא מַלְכִּי / יְפֵה עַיִן אַדְמוֹנִי:


תָּכִין לְךָ הַדֶּרֶךְ / בֶּן יְשׁוּרוּן לְהִתְכּוֹנֵן

לַעֲלוֹת מִבֵּית פֶּרֶךְ / קוּם לֵךְ וְאַל תִּתְאוֹנֵן

פְּשׁוֹט מִלִּבָּךְ מוֹרֶךְ / וְשׁוּב מֵחֵטְא וְהִתְבּוֹנֵן

הַיּוֹם בְּסֵפֶר חַיִּים / אֵל שַׁדַּי יִכְתְּבֵנִי:


אַהְבַת כְּלִילַת יוֹפִי / יוֹם זֶה בְּלִבִּי עָלָה

וַאֲנִי גוֹלֶה בְּשׁוֹפִי / וְלִבִּי בְּאַהֲבָה חָלָה

וְאַזְכִּיר נְבוּאַת צוֹפִי / בְּיֶשַׁע וּבְהַצָּלָה

וְנַפְשִׁי הַנִּבְהָלָה / בַּהֲקִיצִי תְּשִׂיחֵנִי:


מִתֵּימָן שָׁמַעְתִי / דְּבַר חֶזְיוֹן תַּנְחוּמִים

וְכָאַרְבֶּה נִעַרְתִּי / לְהַזְכִּיר זְכוּת אִישׁ תָּמִים

וּבֵאלֹהַי בָּטַחְתִּי / יַצִּיב גְּבוּלוֹת עַמִּים

לְמִסְפַּר בְּנֵי יִשְׂרָאֵל / בְּחוֹתַם אֵל נֶחְתָּמִים

מִגְדַּל עוֹז שֵׁם יְוֹיָ / מְנַת חֶלְקִי וּבִטְחוֹנִי:


שְׁלֹשָׁה אָבוֹת בַּקְּשׁוּ / יוֹם זֶה רַחֲמִים עָלַי

לְמָלֵא רַחֲמִים דִּרְשׁוּ / לִקְשׁוֹר בְּאוֹתוֹת דְּגָלַי

וּמְשַׂנְאַי יִנָּתְשׁוּ / וְאָשׁוּב אֶל אָהֳלַי

בְּיוֹם הַדִּין יֵעָנְשׁוּ / וּתְלַהֲטֵם אֵשׁ גְּאוֹנִי:


מִקּוֹל שׁוֹפָר יֶהֱמוּ / סַנְטוּרֵי מַשְׂטִינַי

כֻּלָּם כְּאֶבֶן יִדְּמוּ / וְלֹא יִזָּכְרוּ עֲווֹנַי

וְעַם קֹדֶשׁ לוּ חָכְמוּ / לְהַשְׂכִּיל זֹאת בְּמִשְׁכָּנַי

לֹא בָאתִי עַד הֵנָּה / וְלֹא יַד צָר תְּנִידַנִי:


וּרְצוֹן צוּרִי יְשׁוֹבֵב / לְבָנִים הַנֶּאֱהָבִים

חַסְדּוֹ וְטוּבוֹ יְסוֹבֵב / שְׁבָטִים הַשּׁוֹבָבִים

עַמּוֹ אֲשֶׁר הוּא חוֹבֵב / וּלְעָבְדוֹ נִצָּבִים

וּמַצְדִּיקֵי הָרַבִּים / בִּזְכוּתָם יִרְצֵנִי:
