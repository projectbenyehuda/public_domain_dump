עֲצַת מַשְׂכִּיל

אל משה בן יעקב

עֲצַת מַשְׂכִּיל בְּיִרְאַת אֵל תְּרִיחוֹ / וְאוֹתִיּוֹת בְּלִי לָשׁוֹן תְּשִׂיחוֹ,

וְיוֹם יִתְקַדְּרוּ עָבֵי סְעִפָּיו / תְּפִיצֵמוֹ כְּמוֹ שֶׁמֶשׁ בְּזָרְחוֹ,

וְהַמָּדַע שְׁמוֹ נֵר הָאֱלֹהִים / וְהוּא טֶרֶם הֱיוֹת גַּלְגַּל וְנָגְחוֹ,

וְהוּא הַיְסוֹד לְנִשְׁמַת כֹּל וְרוּחַ / בְּאַפֵּי אֹהֲבָיו הָאֵל נְפָחוֹ.

וְלֹא יִתְרַע לְרֵעַ תַּהֲפֻכוֹת / וְלֹא יָכִין בְּלֵב בֹּגֵד מְנוּחוֹ.

וּמַה הוֹעִיל בְּאִישׁ אַחַר לְבָבוֹ / לְבַל יָתוּר וְתַאְוָתוֹ נְכֹחוֹ,

וְהַשַּׁלִּיט – אֲשֶׁר יַשְׁלִיט לְדֵעוֹ, / וְהַמֹּשֵׁל – אֲשֶׁר יִמְשֹׁל בְּרוּחוֹ,

וְהַגִּבּוֹר – אֲשֶׁר יֵלֵךְ כְּמוֹ אֵשׁ / וְלֹא יִירָא יְקוֹד כִּבְשָׁן וּפִיחוֹ,

וְהַמַּשְׂכִּיל – אֲשֶׁר יִתְעַל בְּשִׂכְלוֹ, / וְהַחַלָּשׁ – אֲשֶׁר יָעֹז בְּכֹחוֹ,

וְהֶעָנִי – אֲשֶׁר יִשְׁמֹר לְשֹׁמְרוֹ, / וְהַנָּדִיב – אֲשֶׁר יַאֲמִין בְּבָרְחוֹ.

חֲמָסִי עַל אֲמִתִּי כִּי לְרַגְלָהּ / לְבָבִי זָע וְאֵין רֵע יְנִיחוֹ,

וְלֹא אָמַר לְשֵׂכֶל הוֹן בְּדָרְשׁוֹ / וְהוֹן עוֹלָם כְּמֵת מִלֵּב שְׁכֵחוֹ,

וְלֹא רָץ לֶאֱסֹף חַיִל וְלִכְנוֹס / וְאִלּוּ בָא בְיָדוֹ לֹא לְקָחוֹ,

וְלֹא תָעָה בְמַסְלוּל הַיְדִידוּת – / וְאֵיךְ גַּלְגַּל יְהִי תֹעֶה בְאָרְחוֹ?

וְסוֹד דּוֹדָיו חֲדַר לִבּוֹ הֱבִיאוֹ / וְעַל פִּתְחוֹ לְשׁוֹנוֹ שָׂם בְּרִיחוֹ,

עֲדֵי עָלָה אֱלֵי הַר הָאֱמוּנָה / וְיָם הַתֹּם בְּכַף רֶגֶל צְלָחוֹ,

וְכָל חֶפְצוֹ לְסַתֵּר מַעֲלוֹתָיו – / וְאֵיךְ בֹּשֶׂם יְהִי מַכְחִיד לְרֵיחוֹ?

וְלֹא יַכִּיר תְּהִלּוֹתָיו לְבַד אִישׁ / אֲשֶׁר אִישׁוֹן זְמָמָיו צוּר פְּקָחוֹ.

וּבִלְתּוֹ כָּל אֱנוֹשׁ נִמְצָא לְשׁוֹנוֹ / כְּחֵץ שָׁחוּט בְּשֶׁמֶן זַךְ מְשָׁחוֹ,

פְּעָמִים יַעֲנֶה עַזּוֹת וְיַעְתִּיר / בְּמִדְבָּרָיו וְאֵין מַעְצָר לְרוּחוֹ,

וּפַעַם יַעֲנֶה רַכּוֹת וְיִדַּל, / כְּאִלּוּ אֵל שְׁלָחָהוּ וְרוּחוֹ,

וְהוֹן תֵּבֵל יְהִי מֹאֵס וְגִיחוֹן / צְמָאוֹ לֹא יְהִי שֹׁבֵר בְּגִיחוֹ,

יְפַתֵּנִי בְּהַסְתֵּר מַחֲלָתוֹ / וְצָרַעְתּוֹ כְבָר זָרְחָה בְמִצְחוֹ.

אֲנִי אוֹת כִּי אֲשֶׁר נִגְזַר עֲלֵי אִישׁ – / בְּשִׂכְלוֹ לֹא יְשִׁיבֵהוּ וְשִׁלְחוֹ.

צְנָפַנִי זְמָן מִיּוֹם הֱיוֹתִי / וְעוֹד לֹא נָס הֲדַר אַפִּי וְלֵחוֹ,

וְהִזְקַנְתִּי, וְהֵן שַׂבְתִּי, וְיָדוֹ / נְטוּיָה עוֹד רְדֹף אֹתִי וְדָחֹה.

יְשִׂיחוּן בַּעֲדִי אִישִׁים: תְּהִלּוֹת / אֲהֵבוּהוּ וְיֶלֶד יוֹם זְנָחוֹ,

וְהָיָה שַׁאֲנָן מִימֵי נְעוּרָיו / וְעֵת זֹקֶן לְאֶרֶץ נוֹד נְסָחוֹ,

וְשָׁקַט אֶל שְׁמָרָיו אָז וְאֹהֶל / בְּרִית שָׁלוֹם עֲלֵי בֵיתוֹ מְתָחוֹ,

וְסָגַר בַּעֲדוֹ דַלְתֵי חֲסָדָיו / וְחֶרֶב הַנְּדֹד עָלָיו פְּתָחוֹ,

וְשָׂם קֶדֶם עֲלֵי סֶלַע פְּעָמָיו / אֲבָל הֵמַס בְּרוּחַ פִּיו צְחִיחוֹ,

וְקִנּוֹ עַל צְרִיחַ שָׁת וְאֶל שֹׁד / זְמָן קָרָא וְצָרַח עַל צְרִיחוֹ,

וְהַכֶּרֶם אֲשֶׁר שֹרֵק נְטָעוֹ / בְּאֵשׁ בֶּגֶד בְּלִי עָווֹן כְּסָחוֹ,

עֲנָבִים לַעֲשׂוֹת קִוָּה, וְאָכֵן / בְּאֻשִׁים נֶהְפְּכוּ טַרְפּוֹ וְצִמְחוֹ,

וּמַעְיַן אַהֲבָה הֹלֵךְ לְאִטּוֹ / בְּרֶגֶל גַּאֲוָה מֹשֶׁה דְלָחוֹ,

וְהִרְוַנִי בְמֵי מָרָה וְדוֹדַי / יְרַוֻּהוּ בְמֵי יַרְדֵּן יְרֵחוֹ.

שְׁמַע, מֹשֶׁה בְנוֹ יַעְקֹב, תְּהִלּוֹת / נְאֻם מֹשֶׁה בְנוֹ יַעְקֹב וְשִׂיחוֹ.

חֲשֹׁב אַהַב כְּסוּס דֹּהֵר וְאַךְ לֹא / תְּרִיצֵהוּ בְמִתְגּוֹ אוֹ בְחַחוֹ,

וְלֹא יֵאוֹת הֱיוֹת מֶרְכָּב לְבֹגֵד / עֲלֵי פִיו לוּ מְכָרוֹ אוֹ טְבָחוֹ,

וְאֵין הוֹעִיל בְּאַהַב פֶּה עֲדֵי כִי / יְהִי חָרוּת עֲלֵי לֶב־אִישׁ וְלוּחוֹ.

קְחָה שִׁירֵי מְרִיבוֹת וַאֲהָבוֹת / פְּנִינֵי מִדְבְּרֵי חִכִּי וְלִקְחוֹ,

כְּרֵיחַ קִנְּמָן־בֶּשֶׂם לְכָל אַף / פְּנֵי לַיִל וּמֹר בַּיּוֹם לְרוּחוֹ,

לְזָרִים צַחֲנַת מִלָּיו וְרוֹשׁוֹ / וְלָךְ צוּפֵי תְּהִלּוֹתָיו וְרִקְחוֹ.

שְׁלַחְתִּיו לֹא לְיַשֵּׁר אֶת לְבָבוֹ / וְלֹא לָשִׁיב אֲבֵדָתוֹ וְנִדְחוֹ,

לְבַל יֹאמַר: כְּבָר גָּנַב לְבָבִי / וְאֵשׁ חָרוֹן בְּאַפּוֹ לֹא קְדָחוֹ.

וּמַצִּית הַקְּרָב – וַאֲנִי לְשָׁלוֹם – / לְעוֹלָם לֹא אֲרַפֶּה אֶת מְזִיחוֹ,

וְלוּ גַּלְגַּל יְהַתֵּל בִי, בְּשִׁמְשׁוֹ / אֱהִי בֹחֵל וְאֶמְאַס אֶת יְרֵחוֹ.
