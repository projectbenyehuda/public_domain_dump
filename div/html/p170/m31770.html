הֲרָאִיתָ בְרֵכוֹת

אל שלמה בן ראש


הֲרָאִיתָ בְרֵכוֹת אוֹ רְהָטִים / וְעַפְעַפַּי בְתוֹכָם כַּמְּשׁוֹטִים

וְנִטְפֵיהֶם כְּמוֹ גַּרְגַּר בְּדֹלַח / וְהֵם לַחִים, לְזֹאת אֵינָם לְקוּטִים,

וְלֹא אוּכַל מְצוֹת אוֹתָם כְּאִלּוּ / בְּזָדוֹן עִנְבֵי רוֹשׁ בָּם שְׂחוּטִים.

זְמַנִּי שָׂם סְבִיבוֹת תַּאֲוָתִי / חֲרָבוֹת בָּרְקוּ מֵהֶם לְהָטִים,

וְהֵם מִתְהַפְּכִים לַרְחִיק אֲהָבַי / וּבִצְדִיָּה לְהַקְרִיבָם מְרוּטִים.

כְּלָלוֹת יִשְׁלְחוּ אֵלַי יְלָדָיו / יְגוֹנֵיהֶם וְאֶל כָּל עַם פְּרָטִים,

וְיַשְׁלִימוּ מְתֵי אָוֶן וְעָמָל / וּבִי יַעֲשׂוּ – וְאֵין חוֹמֵל – שְׁפוּטִים.

הֱקִימוּנִי לְמַטָּרָה וּמוֹרִים / סְגוֹר לִבִּי וְאֵין מוֹרִים וְחוֹטִאים,

וְצַלְעוֹתַי הֱכִינוֹתִי לְמָגֵן / לְחִצֵּיהֶם וּמֵי עֵינַי שְׁלָטִים.

חֲמָסִי עַל שְׂעִפַּי כִּי בְעַד דּוֹד / לְהַמְתִּיק סוֹד לְבַד תּוֹעִים וְשָׁטִים,

וְלֹא מָצְאוּ לְבַד חָנֵף וּמֵרָע, / וְיוֹם לָחֶם – שְׂרוּעִים אוֹ קְלוּטִים,

וְחַיִּים בַּעֲפַר בֶּגֶד קְבוּרִים / בְּתַכְרִיכֵי כְסִילוּתָם מְעֻטִּים,

יְדַמּוּ כִּי כְסוּיִים הֵם וּמִסְוֵה / תְּהִלּוֹתָם יְהַלֵּכוּ פְשׁוּטִים,

וְדוֹרֵשׁ מִלְּשׁוֹנוֹתָם אֱמוּנִים / כְּדוֹרֵשׁ מִנְּבִיאֵי אֵל אוֹב וְאִטִּים.

וְאִם שֶׁמֶשׁ אֱמֶת יִזְרַח – צְלָלָיו / מְהֵרָה לַעֲרֹב פּוֹנִים וְנוֹטִים,

כְּמוֹ אוֹרֵי שְׁלֹמֹה בִּי פְנֵיהֶם / נְסוּכִים בַּעֲנַן פֵּרוּד וְלוּטִים,

וּמִלָּיו מָשְׁכוּ גִזְרֵי לְבָבוֹת / הֲכִי סוּת הָאֱמֶת לוֹבְשִׁים וְעוֹטִים,

אֲשֶׁר מָתְקוּ [בְּלִי] לָשׁוֹן יְצַפְצֵף / בְּגַן סֵפֶר כְּמוֹ תוֹרִים וְעֵטִים.

בְּשִׁנָּיו צוּף אֱמוּנָה לַיְשָׁרִים / וּבָהֶם לַעֲגַת זֵדִים וְשֵׂטִים,

כְּקֹר מַיִם אֳהָבָיו לַאֲהוּבָיו / וְלִמְשַׂנְאָיו כְּעַקְרַבִּים וְשׁוֹטִים.

שְׁמוֹ בֶּן רֹאשׁ וְהוּא נֹפֶת וְתִירוֹשׁ / וְאִם הוּא רֹאשׁ לְרָאשֵׁי הַשְּׁבָטִים,

וְכָל יוֹצֵא לְמִלְחַמְתּוֹ בְחֶרֶב / ספר אותה(?) הוֹדֵף בְּמַטִּים.

לְךָ שָׁלוֹם, יְדִיד הָלַךְ לְדַרְכּוֹ / וְעָבַט הַנְּפָשׁוֹת בַּעֲבוֹטִים

וּבִלְתּוֹ נוֹתְרוּ כָל הַפְּגָרִים / בְּיַד סֶלַע תְּשׁוּקָתוֹ שְׁמוּטִים,

וְשָׁלוֹם עַל פְּלִיטֵי הַחֲסָדִים / – וְאַנְשֵׁי הָאֱמֶת הַיּוֹם פְּלֵיטִים –

וְכוֹכָבִים לְהָאִיר אֶת־סְפָרַד / לְמִצְרַיִם יְפִיצוּן אוֹר וּפוּטִים,

גְּבִירִים אָחֲזוּ מַדָּע וְחָכְמָה / וְצַחוֹת עוֹבְרִים אוֹתָם וְעֵטִים (?)

וְעַל אָמִיר בְּגַן הַשְׂכֵּל וּמוּסָר / אֲרָזִים לֹא עֲמָמוּהוּ וְשִׁטִּים,

אֲשֶׁר עָצְּמוּ תְהִלּוֹתָיו וְרַבּוּ / לְאֵין חֵקֶר, וְאִם שָׁנָיו מְעַטִּים,

מְאוֹרִים יִתְאֲבוּ סַפֵּר חֲסָדָיו / וְאִלּוּ יִהְיוּ דוֹבְרִים וּבוֹטִים.

יְפַלֵּט אֵל הֲמוֹנָם מִזְּמַנָּם, / הֲכִי דִינָיו לְגֵו בְּרִים שְׁפָטִים,

וְיַעְמִיד עַל סְלָעִים פַּעֲמֵיהֶם / וּמוּעָדִים לְבַל יִהְיוּ וּמָטִים.
