מסע לארץ ישראל

מָבוֹא

אֶּת הַמַּסָּע לְאֶרֶץ-יִשְׂרָאֵל הַזֶּה חִבֵּר אִישׁ תָּם וְיָשָׁר וְחָכָם גָּדוֹל וַּבַעַל כִּשְׁרוֹנוֹת מְצֻיָּנִים הַד“ר אֱלִיעֶזֶר הַלֵּוִי, מַזְכִּירוֹ שֶׁל רַבִּי מֹשֶׁה מוֹנְטִיפְיוֹרִי, בּוֹ הוּא מְתָאֵר אֶת מַסָּעוֹ לְאֶרֵץ-יִשְׂרָאֵל בִּשְׁנַת תקצ”ח.

ד“ר הַלֵּוִי נוֹלַד בְּי' בְּתַמּוּז תקס”ט בָּעֲיָרָה צִילְץ בִּשְׁלֶזְיָה לְאָבִיו, ר' מָרְדְּכַי, יְהוּדִי תַּלְמִיד-חָכָם. עוֹד בִּימֵי יַלְדוּתוֹ הִצְטַיֵּן בְּכִשְׁרוֹנוֹתָיו וּבְיִחוּד בְּכֹחַ הַזִּכָּרוֹן. בַּזְמָן הַהוּא חִנְּכוּ אֶת הַנְּעָרִים בִּשְׁלֶזְיָה רַק עַל בִּרְכֵּי הַתּוֹרָה וְכָל לִמּוּד לָשוֹן זָרָה וּמַדָּע זָר נֶחְשַׁב לְעָוֹן. אַף עַל פִּי כֵן, לָמַד הַנַּעַר אֱלִיעֶזֶר גַּם גֶּרְמָנִית וְצָרְפַתִּית וְגַם הִסְטוֹרְיָה וְגֵאוֹגְרָפְיָה. בִּהְיוֹתוֹ בֶּן שְׁלֹש-עֶשְׂרֵה מֵת עָלָיו אָבִיו. מֵאָז נָדַד מֵעִיר לְעִיר לִקְנוֹת תּוֹרָה וְחָכְמָה. דּוֹדוֹ שְׁלָחוֹ לְלִיסָא וְשָׁם לָמַד מִלְּבַד תַּלְמוּד גַּם לְשׁוֹנוֹת רוֹמָא וְיָוָן וְחָכְמַת הַמַּתֵּמָטִיקָה וְהַהִסְטוֹרְיָה. אַחֲרֵי כֵן עָבַר לִישִׁיבַת נִיקֶלְסְבּוּרְג. כָּאן לָמַד גַּם חָכְמַת הַקַּבָּלָה. הוּא זָכָה לִהְיוֹת גַּם מִתַּלְמִידֵי הַגָּאוֹן הַמְּפֻרְסָם בְּחָכְמָתוֹ וַחֲסִידוּתוֹ ר' מֹשֶׁה סוֹפֵר, הַנִּקְרָא “חֲתַם סוֹפֵר”. בְּלָמְדוֹ בִּישִׁיבַת ר' מֹשֶׁה סוֹפֵר בִּפְרֶסְבּוּרְג הִשְׁתַּלֵּם גַּם בִּלְשׁוֹן צָרְפַת וְלָמַד אִיטַלְקִית וּסְפַרְדִית.

עוֹד בַּיָּמִים הָהֵם הִקְדִּישׁ אֶת עַצְמוֹ בְּיִחוּד לְלִמּוּד חָכְמַת הַמִּזְרָח. יְדִיעוֹתָיו הָרַבּוֹת, אֲשֶׁר מֵהֶן יִמְצָא לַחְמוֹ בַּיָּמִים הָהֵם, הֱבִיאוּהוּ בְּקֶשֶר עִם כַּמָּה חֲכָמִים וְהֵם עָזְרוּ לוֹ לְהוֹסִיף דָּעַת. אַחֲרֵי שָׁלֹשׁ שָׁנִים הָלַךְ לָאוּנִיבֶרְסִיטָה לְוִינָה וְאַחַר כָּךְ לְבֶרְלִין וְשָׁמַע תּוֹרָה מִפִּי גְדוֹלֵי הַמְלֻמָּדִים בַּיָּמִים הָהֵם וְשָׁם גָּמַר אֶת חֹק לִמּוּדָיו. יְדִידָיו הִצִּיעוּ לְפָנָיו לְקַבֵּל מִשְׂרַת רַבָּנוּת בְּאַחַת הַקְּהִלּוֹת הַחֲשׁוּבוֹת, אַךְ הוּא מֵאֵן; הָלַךְ לְהַמְבּוּרְג וְאַחַר כָּךְ לְלָנְדּוֹן כְּדֵי לְהִשְׁתַּלֵּם בְּמִקְצוֹעַ חֲקִירַת הַמִּזְרָח בְּעָבְדוֹ בַּמּוּזֵיאוּמִים הַשּׁוֹנִים. בְּלָנְדּוֹן הִשְׁתַּלֵּם גַּם בִּלְשׁוֹן הַחַרְטֻמִּים הַמִּצְרִים. בֵּינָתַים נָסַע לְשֵׁם לִמּוּד לְפָרִיס. וְאֲחַרֵי שׁוּבוֹ לְלָנְדּוֹן נִשְׁלַח בִּשְׁנַת 1837–39, עַל יְדֵי הָחֲכָמִים וַאֲנָשִׁים מֵרָמֵי הַמַּעֲלָה אֲשֶׂר הִכִּירוּ אֶת עֶרְכּוֹ, לְמַסָּע לְמִצְרַיִם, לוּב, כּוּשׁ, סוּרִיָּה, אֶרֶץ-יִשְׂרָאֵל, תֻּרְכִּיָּה, אַסְיָה הַקְּטַנָּה וְיָוָן. בַּאֲלֶכְּסַנְדְּרִיָּה שֶׁל מִצְרַיִם הֻצַּג לִפְנֵי מֻחַמַד עַלִּי פֶּחָה מִשְׁנֵה הַמֶּלֶךְ, וּמָצָא חֵן בְּעֵינָיו בְּפָתְרוֹ לוֹ אַחַת הַכְּתָבוֹת בִּכְתָב הַחַרְטֻמִּים.

מֻחַמַּד עַלִּי נָתַן לוֹ מִכְתַּב הַמְלָצָה. בְּאַלְקָאהִרָה שָׁהָה זְמַן מָה וְלָמַד עַרְבִית מִפִּי שֵׁיךְ עַרְבִי מְפֻרְסָם. אַחַר כָּךְ נָסַע לְלוּב וְעָשָׂה זְמָן רַב בְּבִירָתָהּ דִּר וְחָקַר בַּלָּשׁוֹן הַלּוּבִית. מִכָּאן נָסַע לְכוּשׁ, וּבַדֶּרֶךְ מָצָא לוֹ חֹמֶר רַב לַעֲבוֹדָה מַדָּעִית. בַּמִּנְזָרִים בְּכוּשׁ לָמַד אֶת הַלָּשׁוֹן הַקּוֹפְּטִית. מִמִּצְרַיִם הָלַךְ לְאֶרֶץ-יִשְׂרָאֵל.

בַּחוֹבֶרֶת שֶׁלְּפָנֵינוּ יִמְצָא הַקּוֹרֵא אֶת מִכְתְּבֵי מַסָּעוֹ זֶה שֶׁבָּהֶם הוּא מְתָאֵר אֶת אֲשֶׁר קָרָהוּ בְּאֶרֶץ-יִשְׂרָאֵל וְאֵת אֲשֶׁר רָאָה שָׁם. לְזֵכֶר הַנֵּס שֶׁקָּרָהוּ בְּהִנָּצְלוֹ מִידֵי הַדְּרוּזִים הָיוּ יְלָדָיו מִתְחַפְּשִׂים בְּפוּרִים כְּבֶדְוִים. אֲחַרֵי עָבְרוֹ בְּכָל אַרְצוֹת הַמִּזְרָח וְאַחֲרֵי חָקְרוֹ בַּלְּשׁוֹנוֹת הַמִּתְהַלְּכוֹת שָׁם, שָׁב לְאֵרוֹפָּה. בְּעָבְרוֹ בְּרוֹמָא דָרַשׁ בְּבֵית הַכְּנֶסֶת, וְהִנֵּה נִמְצָא שָׁם בְּדֶרֶךְ מִקְרֶה גַם סֵיר מֹשֶׁה מוֹנְטִיפְיוֹרִי וְאִשְׁתּוֹ יְהוּדִית. עַד מְהֵרָה נִקְשְׁרוּ נַפְשׁוֹת מוֹנְטִיפְיוֹרִי וְהַלֵּוִי בְּקֶשֶׁר אַהֲבָה וִידִידוּת וְלֹא נִפְרְדוּ עוֹד. הַלֵּוִי הָיָה לוֹ לְמַזְכִּיר וַיְלַוֶּה אוֹתוֹ בְּכָל מַסְּעוֹתָיו וְאֶת כָּל כֹּחוֹ הִקְדִּישׁ מֵאָז לַעֲבוֹדַת עַמֵּנוּ. כְּשֶׁשָּׁאֲלָה אוֹתוֹ בִּתּוֹ, אֵיךְ זֶה נֵאוֹת לַעֲזֹב אֶת חֲקִירוֹתָיו בַּשָּׂפוֹת הַשֵּׁמִיּוֹת, אֲשֶׁר לְפִי דַעַת כָּל גְּדוֹלֵי הַדּוֹר הַהוּא הָיָה יָכֹל לַעֲשׂוֹת בָּהֶן גְּדוֹלוֹת, עָנָה הַלֵּוִי: “נוֹכַחְתִּי לָדַעַת כִּי עַל-יְדֵי עֲבוֹדָתִי לְיַד מוֹנְטִיפְיוֹרִי אוּכַל לְהַקְדִּישׁ אֶת כִּשְׁרוֹנוֹתַי, חַיַּי וְכֹחוֹתַי לָאָרֶץ הַקְּדוֹשָׁה וּלְעַמִּי הַנִּדְכָּא. רַעְיוֹן זֶה מָשַךְ אוֹתִי בְּעֹז וְלֹא מָצָאתִי אֶת לִבִּי לִדְחוֹת אֶת הַצָּעַת מוֹנְטִיפְיוֹרִי. לְבַדִּי לֹא יָכֹלְתִּי לִפְעֹל הַרְבֵּה, וְאוּלָם עַל-יְדֵי הִתְלַהֲבוּת מוֹנְטִיפְיוֹרִי לָרַעְיוֹן, עַל-יְדֵי הַשְׁפָּעָתוֹ וְכַסְפּוֹ, הָיְתָה לִי הַתִּקְוָה לִפְעֹל נְצוּרוֹת.”

בְּעֵת עֲלִילַת הַדָּם בְּדַמֶּשֶׂק בִּשְׁנַת 1840 לִוָּה הַלֵּוִי אֶת מוֹנְטִיפְיוֹרִי לְדַמֶּשֶׂק וְעָזַר הַרְבֵּה לָגֹל אֶת הָאַשְׁמָה מֵעַל הַיְּהוּדִים. בְּשׁוּבָם מִדַּמֶּשֶׂק בִּקְרוּ אֶת הַשֻׂלְטָן הַתֻּרְכִּי בְּקוּשְׁטָא, וְהִשִּׂיגוּ זְכֻיּוֹת אֶזְרָח לִיהוּדֵי תֻּרְכִּיָּה. אַחַר כָּךְ נָסְעוּ לְרוֹמָא, וְהִשְׁתַּדְּלוּ לְהָסִיר אֶת הַכְּתֹבֶת מֵעַל קֶבֶר הַנָּזִיר תּוֹמַס, אֲשֶׁר רְצִיחָתוֹ הָיְתָה הַסִּבָּה לַעֲלִילַת הַדָּם. עַל הַמַּצֵבָה הָיָה כָּתוּב: “כָּאן תָּנַחְנָה הָעֲצָמוֹת הַקְּדוֹשׁוֹת שֶׁל הָאָב תּוֹמַס שֶׁנִּרְצַח בִּידֵי הַיְּהוּדִים”. מֵרוֹמָא נָסְעוּ לְצָרְפַת וְהִשְׁתַּדְּלוּ לִפְנֵי הַמֶּלֶךְ לוּאִי פִילִיףּ לְטוֹבַת הַיְּהוּדִים. אָמְנָם גַּם בְּמֶשֶׁךְ הַזְּמָן הַהוּא הִקְדִּישׁ הַלֵּוִי אֶת יֶתֶר זְמַנּוֹ לַחֲקִירוֹת, כְּכֹל אֲשֶׁר נִזְדַּמֵּן לְיָדוֹ בִּמְקוֹמוֹת מַסָּעוֹ.

בְּשׁוּבָם לְלָנְדּוֹן כִּבְּדוּ אוֹתָם כָּבוֹד גָּדוֹל. מוֹנְטִיפְיוֹרִי הִצִּיג לִפְנֵי אֵשֶת חַסְדּוֹ הַמַּלְכָּה וִיקְטוֹרִיָּה אֶת יְדִידוֹ הַלֵּוִי, וְגַם הִיא הִרְבְּתָה לְהַלְלוֹ עַל פְּעֻלּוֹתָיו הָעֲצוּמוֹת. בִּכְלָל כִּבֵּד מֹשֶׁה מוֹנְטִיפְיוֹרִי מְאֹד אֶת מַזְכִּירוֹ, וְכַאֲשֶׁר הָיוּ מַרְבִּים לְשַׁבֵּחַ אֶת מוֹנְטִיפְיוֹרִי עַל מַעֲשָׂיו, הָיָה רָגִיל לֵאמֹר: “לֹא לִי עֲלֵיכֶם לְהוֹדוֹת כִּי-אִם לִידִידִי הַלֵּוִי שֶׁעוֹרְרַנִי וְהִלְהִיבַנִי לַעֲשׂוֹת אֵת כָּל הַמַּעֲשִׂים הָאֵלֶּה, וְגַם אֲנִי צָרִיךְ לְהוֹדוֹת לוֹ עַל הַצְלָחָתִי”. בִּשְׁנַת 1846 לִוָּה אֶת מוֹנְטִיפְיוֹרִי בְּמַסָּעוֹ לְרוּסְיָה לָחוּשׁ לְעֶזְרַת הַיְּהוּדִים הַנִּדְכָּאִים. אֲחֵרי רְאָיוֹן עִם צַר רוּסְיָה עָבְרוּ בְּכָל הַקְּהִלּוֹת, וְהַלֵּוִי, כְּדַרְכּוֹ גַם בַּנְּסִיעוֹת הַקּוֹדְמוֹת דָּרַש בְּכָל מָקוֹם בְּבָתֵּי-הַכְּנֶסֶת וְהִפְלִיא אֶת לֵב שׁוֹמְעָיו. בְּיִחוּד נוֹדְעָה לְשֵׁם דְּרָשָׁתוֹ בְּבֵית-הַכְּנֶסֶת

הַגָּדוֹל בְּוִילְנָה בְּאָזְנֵי אָלְפֵי שׁוֹמְעִים. וְכָכָה לִוָּה הַלֵּוִי אֶת מוֹנְטִיפְיוֹרִי בִּשְלֹשׁ-עֶשְׂרֵה מַסְּעוֹתָיו: בְּאֶרֶץ-יִשְׂרָאֵל, בְּרוּמֶנְיָה, בְּרוּסְיָה וּבְרוֹמָא. בשנת 1866 יִסֵּד מוֹנְטִיפְיוֹרִי בְּרַמְסְגוּט, מְקוֹם מְגוּרָיו, בֵּית-מִדְרָש לְרַבָּנִים. הַלֵּוִי סִדֵּר אֶת בֵּית-הַמִּדְרָש וְנִמְנָה לִהְיוֹת מְנַהֲלוֹ. בָּאוּלָם הַגָּדוֹל שֶׁל בֵּית-הַמִּדְרָש לְרַבָּנִים, בְּמָקוֹם שֶׁשָּׁם תְּלוּיִם יוֹתֵר מִמֵּאָה מִכְתְּבֵי-בְּרָכָה הַמְתַנִּים אֶת שִׁבְחֵי מוֹנְטִיפְיוֹרִי, צִוָּה מוֹנְטִיפְיוֹרִי לִתְלוֹת אֶת תְּמוּנַת הַלֵּוִי בְּגָדְלוֹ הַטִּבְעִי.

הָאַהֲבָה בֵּין מוֹנְטִיפְיוֹרִי וְהַלֵּוִי לֹא נֶעְכְּרָה מֵעוֹלָם. ר' מֹשֶׁה מוֹנְטִיפְיוֹרִי מִנָּה אֶת הַלֵּוִי גַם בֵּין אֶפִּיטְרוֹפְּסָיו אֲשֶׁר יוֹצִיאוּ אֶת צַוָּאָתוֹ לַפֹּעַל. כְּשֶׁחָלָה מוֹנְטִיפְיוֹרִי אֶת מַחֲלָתוֹ אֲשֶׁר מֵת בָּהּ, לֹא נָתַן לְהַלֵּוִי לָזוּז מִמִּטָּתוֹ.

עֲבוֹדָה רַבָּה הִשְׁקִיעַ הַלֵּוִי עַל פִּי בַּקָּשַת מוֹנְטִיפְיוֹרִי בְּסִדּוּר “יוֹמָנֵי סֵיר מֹשֶׁה מוֹנְטִיפְיוֹרִי וְהַגְּבֶרֶת מוֹנְטִיפְיוֹרִי”, שֶׁבָּהֶם הִשְׁתַּמֵּשׁ בַּחֲמִשָּׁה יוֹמָנֶיהָ שֶׁל יְהוּדִית, אֵשֶׁת מוֹנְטִיפְיוֹרִי הַנִּפְלָאָה, חֲמִשָּה וּשְּׁמוֹנִים יוֹמָנָיו שֶׁל מֹשֶׁה מוֹנְטִיפְיוֹרִי מִשְּׁנַת 1814–1883 וּמְקוֹרוֹת שׁוֹנִים שֶׁנִּמְצְאוּ בִּידֵי הַלֵּוִי.

אַחֲרֵי מוֹת מוֹנְטִיפְיוֹרִי עָבַר הַלֵּוִי לְלָנְדּוֹן, וְחָשַׂב לְהוֹצִיא אֶת כְּתָבָיו הוּא, אַחֲרֵי גָמְרוֹ לְהוֹצִיא אֶת יוֹמָנֵי מוֹנְטִיפְיוֹרִי, וְאוּלָם הַמָּוֶת קִדְּמָהוּ בְּטֶרֶם הִסְפִּיק לְקַיֵּם אֶת מַחֲשַׁבְתּוֹ. ר' אֱלִיעֶזֶר הַלֵּוִי נִפְטַר בְּרֹאשׁ חֹדֶשׁ כִּסְלֵו שְׁנַת תר"ן.

אֶת מִכְתְּבֵי הַמַּסָּע לְאֶרֶץ-יִשְׂרָאֵל פִּרְסֵם הַלֵּוִי לָרִאשׁוֹנָה בָּעִתּוֹן הַגֶּרְמַנִּי Alegmeine Zeitung des Judentums 1839 בְּגִלְיוֹנוֹת 18–78.

מִכְתָּבִים אֵלּוּ תֻּרְגְּמוּ עַל יְדֵי אַחַד גְּדוֹלֵי הַסּוֹפְרִים בַּזְּמָן הַהוּא, ר' מָרְדְּכַי אַהֲרֹן גִינְזְבּוּרְג, וְנִדְפְּסוּ בְּקֹבֶץ מַאֲמָרָיו הַנִּקְרָא בְּשֵׁם “דְּבִיר”. בְּחוֹבֶרֶת זוֹ אֲנִי מוֹסֵר אֶת תַּרְגּוּמוֹ שֶל הָרמ“א גִּינְזְבּוּרְג, אַחֲרֵי שִׁהִשְׁוֵיתִיו אֶל הַמָּקוֹר הַגֶּרְמָנִי. הִתַּרְתִּי לְעַצְמִי לַעֲשׂוֹת שִׁנוּיִּים קַלִּים בַּתַּרְגּוּם בְּלִי לִפְגֹעַ בְּעֶצֶם סִגְנוֹנוֹ שֶׁל הַמְתַרְגֵּם, בְּמָקוֹם שֶׁהַדְּבָרִים לֹא נִמְצְאוּ מְכֻוָּנִים לַמָּקוֹר, אוֹ בְּמָקוֹם שֶׁהַמְתַרְגֵּם הִשְׁתַּמֵּשׁ בִּמְלִיצָה יְתֵרָה. בִּמְקוֹמוֹת אַחֲדִים הוֹסַפְתִּי מִן הַמָּקוֹר מַה שֶׁהֶחֱסִיר הָרמא”ג. גַּם הִשְׁמַטְתִּי מְקוֹמוֹת בּוֹדְדִים שֶׁאֵינָם שַׁיָּכִים לְגוּף עִנְיָנֵנוּ. הוֹסַפְתִּי בְּשׁוּלֵי הַתַּרְגּוּם הֶעָרוֹת שׁוֹנוֹת לְבֵאוּר הַדְּבָרִים. בְּהֶעָרוֹת אֵלּוּ הָיָה לִי לְעֵינַיִם מַר אֱלִיעֶזֶר רִיבְלִין, וְתוֹדָתִי נְתּונָה לוֹ בָּזֶה. אֶת הַחֹמֶר לְתוֹלְדוֹת הַלֵּוִי לָקַחְתִּי מִמַּאֲמָרָהּ שֶׁל בַּת ד"ר הַלֵּוִי עַל אֹדוֹת אָבִיהָ

מִכְתָּב רִאשׁוֹן

ג’וּנְיַא בְּקִרְבַת צִידוֹן כ“ו סיון תקצ”ח

יָדַעְתִּי אֶל נָכוֹן כִּי שְׁאֵלָתֵךְ הָרִאשׁוֹנָה תִּהְיֶה, מָה הֵבִיאָנִי לְסוּרִיָּה, וְהִנְנִי בִּתְשׁוּבָה עַל שְׁאֵלָתֵךְ זֹאת. אֲנִי הָיִיתִי נָכוֹן לָצֵאת מִמִּצְרַיִם, וְהִנֵּה הֱצִיקַתְנִי רוּחִי לִרְאוֹת גַּם אֵת אֶרֶץ אֲבוֹתֵינוּ, כִּי מַדּוּעַ זֶה אַרְבֶּה יְגִיעָה לַחְקֹר אֶת לְשׁוֹן הַמִּצְרִים הַקַּדְמוֹנִים, אֶת רוּחָם וְאֶת מִדּוֹתֵיהֶם וְלֹא אִיגַע לְכָל הַפָּחוֹת בְּאוֹתָהּ מִדָּה לַחְקֹר אֶת הָאָרֶץ אֲשֶׁר הוֹפִיעָה עָלֶיהָ אֱמוּנָתֵנוּ הַקְּדוֹשָׁה בְּיָפְיָהּ וַהֲדָרָהּ, אֶרֶץ אֲשֶׁר בְּכָל נָהָר מִנַּהֲרוֹתֶיהָ וּבְכָל עֵץ מֵעֵצֶיהָ צְפוּנִים זִכְרוֹנוֹת קְדוֹשִׁים?

הִתְכּוֹנַנְתִּי לְדֶרֶךְ שְׁנֵים-עָשָׂר יוֹם בַּמִּדְבָּר, וְהָיִיתִי נָכוֹן לָשֶׁבֶת עַל הַגְּמָלִים אֲשֶׁר שָׂכַרְתִּי, וְהִנֵּה הָרוּחַ הַנִּקְרָא בְּפִי יוֹשְׁבֵי הָאָרֶץ חַמְסִין יָצָא לְשָׂטָן לִי, כִּי רַב כֹּחַ הָרוּחַ הַזֶּה לְהַפִּיל עַל עֶרֶשׂ דְּוָי גַּם אֶת הָאֲנָשִׁים הַיּוֹתֵר בְּרִיאִים וּמוּצָקִים בְּטִבְעָם, וְרַק מְעַטִּים הֵמָּה גַּם בִּבְנֵי הָאָרֶץ אֲשֶׁר יִנָּקוּ מִפְּעֻלָּה רָעָה אֲשֶׁר יִפְעַל הַסַּעַר הַהוּא הַמִּתְחוֹלֵל עַל רָאשֵׁיהֶם חֲמִשִּׁים יוֹם מִפֶּסַח עַד שָׁבוּעוֹת1, כִּי הוּא רוּחַ דָּרוֹם חַם מְאֹד הַמְּנַשֵּב לִפְעָמִים מֵהַבֹּקֶר עַד הָעֶרֶב וְחוֹדֵר עַד לַחֲדָרִים הַפְּנִימִיִּים.

הָעֵד הֵעִיד בִּי הָרוֹפֵא לְבַל אֹחַז דַּרְכִּי דֶּרֶךְ הַמִּדְבָּר, וְיָעַץ לְמַהֵר לָצֵאת מֵאַלְקָאהִרָה בַּהֶקְדֵּם הָאֶפְשָׁרִי, וְעַל-פִּי הָעֵצָה הַזֹּאת יָרַדְתִּי בָּאֳנִיָּה בְּשִׁבְעָה לְחֹדֶשׁ אִיָּר, וָאָבוֹא בְּי"א לַחֹדֶשׁ הַהוּא לְאֲלֶכְּסַנְדְּרִיָּה וּמֵי הַנִּילוּס פָּעֲלוּ עַל בְּרִיאוּתִי פְּעֻלָּה טוֹבָה רַבָּה יוֹתֵר מֵאֲשֶׁר קִוִּיתִי. כִּמְעַט מָלְאָה לִי חֲצִי שָׁעָה עַל הַיְּאוֹר וְהִנֵּה פָּסוּ כָּל צָרוֹת נַפְשִׁי.

יָצָאתִי מֵאֲלֶכְּסַנְדְּרִיָּה ט"ו לְחֹדֶשׁ אִיָּר וָאֵרֵד בָּאֳנִיַּת קִיטוֹר אַנְגְּלִית לָבוֹא לְבַיְרוּת. שָׁם הֻכְרַחְתִּי לָשֶבֶת בְּקָרָנְטִינָה מִחוּץ לָעִיר אֶחָד וְעֶשְׂרִים יוֹם לְהוֹכִיחַ לָהֶם כִּי לֹא דָּבְקָה בִּי מַחֲלַת הָחֳלִי-רָע, – אֲבָל אַךְ מָלְאוּ לִי יְמֵי הַקָּרָנְטִינָה וָאֱהִי נָכוֹן לְהַצִּיג כַּף רַגְלִי עַל מִפְתַּן הָעִיר, וְהִנֵּה נוֹדַע לִי כִּי פָּרְצָה בְּבַיְרוּת הַמַּגֵּפָה אֲשֶׁר הֵמָּה יְרֵאִים מִמֶּנָּה, וַאֲנִי מֵאֲשֶׁר חָשַׁשְׁתִּי פֶּן יַטִּילוּ יוֹשְׁבֵי שְׁאָר הֶעָרִים עָלַי יְמֵי קָרָנְטִינָה חֲדָשִׁים, נִמְנַעְתִּי מִבּוֹא הָעִירָה, וָאָסוּר מְהֵרָה מִשָּם וָאֵלֵךְ לְדַמֶּשֶׂק. נָמוֹג לִבִּי בַּדֶּרֶךְ מִפַּחַד הַשְּׁמוּעוֹת הָרָעוֹת אֲשֶׁר הִגִּיעוּ לְאָזְנַי עַל דְּבַר רֶצַח וָגֵזֶל הַצְּפוּיִים לְעוֹבְרֵי הַדֶּרֶךְ הַהוּא – לוּלֵא הִכְחִישׁ הָאִישׁ הַמּוֹלִיךְ אוֹתִי אֵת כָּל דִּבְרֵי הַשְּׁמוּעוֹת הָאֵלֶּה בְּוַדָּאוּת, וְדָאַג לְהִתְכֹּונֵן לַדֶּרֶךְ בְּכָל אֶמְצְעֵי הַבִּטָּחוֹן אֲשֶׁר מָצְאָה יָדוֹ. אוּלָם בִּכְפָר אַלְזִבְּדָאנִי – דֶּרֶךְ יוֹם אֶחָד מִדַּמֶּשֶׂק שָׁמַעְתִּי כִּי נֶכֶד הָאֱמִיר בַשִׁיר מוֹשֵׁל הָרֵי סוּרִיָּה אֲשֶׁר יוֹשְׁבֶיהָ נוֹצְרִים מִכַּת הַמָּרוֹנִיטִים2 יָצָא בְּרֹאשׁ חֵילוֹ לְבַעֵר אֶת הַשּׁוֹדְדִים מִן הָאָרֶץ. אָז רָוַח לִי וָאֶמְצָא אֶת לְבָבִי לִנְסֹעַ לְבַדִּי בְּחֶבְרַת מוֹלִיכֵי פִּרְדִּי לְדַמֶּשֶׂק (אֶת מְשָׁרְתִי עָזַבְתִּי בְּבַעַל בֶּךְּ).

בְּשִׁבְתּוֹ לִקְרֹא בַּזֹּהַר לְעִלּוּי נִשְׁמַת הַמָּנוֹחַ הַנִּזְכָּר – הִנְנִי מִשְׁתַּמֵּש בַּמִּבְטָא “לִקְרֹא”, כִּי הַעִקָּר הִיא הַקְּרִיאָה וְלֹא הָהֲבָנָה. – אַרְבָּעָה חֲכָמִים נֶאֶסְפוּ לְהַחֲלִיף מִשְׁמְרוֹתֵיהֶם בַּקְּרִיאָה. וְהָאֲנָשִׁים הָאֵלֶּה בִּרְאוֹתָם אִישׁ זָר לָבוּשׁ הָדָר, אִישׁ עָשִׁיר לְמַרְאִית עַיִן, הִשְׁתָּאוּ עַל בַּקְּרִי אוֹתָם. שׂוֹחַחְתִּי הַרְבֵּה שָׁעוֹת אִתָּם, וְאַחַר הָלַכְתִּי בְּחֶבְרַת אַחַד הַחֲכָמִים אֶל בֵּית הַמִּדְרָש אֲשֶׁר לִרְפָאֵל פַרְחִי. הָלַכְנוּ בְּדֶרֶךְ שְׁתֵּי חֲצֵרוֹת מְרֻצָּפוֹת בְּאַבְנֵי שַׁיִשׁ לְבָנוֹת וּמְרֻבָּעוֹת, בְּאֵרוֹת אַרְטֵזִיּוֹת, וְשׁוֹשַׁנִּים נוֹתְנוֹת רֵיחַ טוֹב כָּלְלוּ אֶת הֲדָרָן. וְהַשֶּׁמֶשׁ הִבְרִיקָה בְּכָל יָפְיָהּ בְּתוֹךְ כָּתְלֵי הַבַּיִת הַמֻּזְהָבִים. בֵּית-הַמִּדְרָשׁ קָטֹן הוּא, אֲבָל תּוֹכוֹ אוֹצָר מָלֵא סְפָרִים רַבִּים, אֲשֶׁר אָמְנָם בִּהְיוֹתָם כֻּלָּם סְפָרִים נִדְפָּסִים לֹא יָקְרוּ בְּעֵינַי, כִּי אִתָּנוּ בְּנֵי אֵרוֹפָּה נִמְצָאִים כָּל הַסְּפָרִים הַיּוֹצְאִים מִבָּתֵּי הַדְּפוּס. וְאַךְ כְּתָב-יָד אֶחָד מָצָאתִי בּוֹ, וְהוּא גוּף סֵפֶר מוֹרֶה-נְבוּכִים בִּלְשׁוֹן עַרְבִי חָסֵר הַפְּרָקִים הָרִאשׁוֹנִים.

שׁוֹמֵר הַסְּפָרִים לֹא הָיָה מוֹכֵר אוֹתוֹ לִי בְּעַד כָּל מְחִיר, אַף כִּי בְּאַהֲבַת הָאֱמֶת לֹא בּוֹשׁ לְהוֹדוֹת כִּי אֵינֶנּוּ מֵבִין דָּבָר מִכָּל הַכָּתוּב בַּסֵּפֶר הָהוּא. גַּם תנ“ך כְּתָב-יָד עִם פֵּרוּשׁ רד”ק וְעִם כָּל הַמּוֹלָדוֹת3
מָצָאתִי שָמָּה, וְהוּא כָּלוּל בַּהֲדָרוֹ. לֹא רָאִיתִי כָּמוֹהוּ לְיֹפִי בְּכָל כְּתָבֵי-הַיָּד אֲשֶׁר רָאִיתִי בִּימֵי חַיַּי, אַף לֹא בִּכְתָבֵי הַקְּרָאִים אֲשֶׁר בְּאַלְקָאהִרָה; כָּל דַּף וְדַף מְיֻפֶּה בְּקִשּׁוּטֵי זָהָב.

אַחַר-כָּךְ בִּקַּרְתִּי אֶת חָכָם יַעֲקֹב. הָלַכְנוּ אֶל בֵּית הַכְּנֶסֶת וַנִּמְצָא שָׁם נֶאֱסָפִים4 רַבִּים וּבְתוֹכָם גַּם עֲשִׁירֵי עָם. בִּכְלָל יֶשְׁנָם הַרְבֵּה בָּתִּים נִכְבָּדִים וְאַחַד הַשֻּׁלְחָנִים שֶׁל אִבְּרָהִים פֶּחָה יְהוּדִי הוּא וּשְׁמוֹ בְּכַר, כִּי יְקָרָה מַעֲלַת הַבְּכוֹרָה בְּעֵינֵיהֶם עַד כִּי יִקְרְאוּ אֶת יַלְדָּם הָרִאשׁוֹן, אִם זָכָר וְאִם נְקֵבָה, בְּשֵׁם הַכָּבוֹד הַזֶּה כָּל יְמֵי חַיָּיו.

מִמָּחֳרַת הַיּוֹם הָלַכְתִּי לָשׂוּחַ אֶל הַמִּגְרָש הַמִּשְׂתָּרֵעַ לִפְנֵי שַׁעַר הָעִיר בַּאב-אֶל-שַׁרְק5.

שָׁם יָגוּרוּ עוֹד מִימֵי קֶדֶם מְצֹרָעִים מֻחְלָטִים6, שָׁם נוֹלְדוּ, שָׁם יִקְחוּ נָשִׁים בֵּינָם לְבֵין עַצְמָם וְדָבָר אֵין לָהֶם עִם אָדָם זוּלָתָם. לֹא יַעַסְקוּ בְּשׁוּם עֵסֶק, כִּי-אִם יֹאכְלוּ אֵת לֶחֶם הַחֶסֶד אֲשֶׁר יוֹשִׁיט לָהֶם הַפֶּחָה וְאַנְשֵׁי חֶסֶד זוּלָתוֹ אֲשֶׁר נָגְעָה אַהֲבַת הַבְּרִיוֹת בְּלִבָּם, וְכִמְעַט יִצְטָרֵעַ אֶחָד מִבְּנֵי הָעִיר לֹא יָחוּשׁוּ לְרַפְּאוֹתוֹ וְלֹא יִדְרְשׁוּ בְּרוֹפְאִים, כִּי-אִם יַהַדְפוּהוּ אֶל מִקְלַט הַמְצֹרָעִים הַזֶּה לְהִסָּגֵּר שָׁם עַד יוֹם מוֹתוֹ, אִם נִכְנַס אָדָם פַּעַם אֶל חֲצַר הַמְצֹרָעִים לֹא יָשׁוּב עוֹד מִמֶּנָּה עַד עוֹלָם. כַּמִּשְׁפָּט הַזֶּה יָעֲשׂוּ בְּנֵי הַמִּזְרָח גַּם לְאֲחֵיהֶם גַּם לְבָתֵּיהֶם וְאַף לְבִנְיְנֵי חֶמְדָּתָם. אִם הִמָּצֵא יִמָּצֵא בָּהֶם פֶּרֶץ קָטֹן לֹא יָחוּשׁוּ לְתַקְּנוֹ כִּי-אִם יְחַבְּקוּ אֶת יְדֵיהֶם בְּחֵיקָם עַד אֲשֶׁר יִרְחַב הַפֶּרֶץ, וְהָיָה הֵיכַל הַחֶמְדָּה בִּרְבוֹת הַיָּמִים לִמְעִי מַפֵּלָה, אֲשֶׁר יָשֹׁמּוּ וְיִשְׁרְקוּ עָלָיו אוֹרְחֵי אֵרוֹפָּה הַבָּאִים שָׁמָּה.

יָצָאתִי מֵאֵת הָעִיר וָאֵפֶן אֶל הַכְּפָר ג’וֹבָּר אֲשֶׁר בּוֹ נִמְצְאָה מְעָרַת אֵלִיָּהוּ הַמְּרֻמֶּזֶת בִּמְלָכִים א' י"ט בַּפָּסוּק “לֵךְ שׁוּב מִדְּבָּרָה דַמֶּשֶׂק”, וְהִיא מְעָרָה קְטַנָּה בְּמַעֲמַקֵּי הַצּוּר וְעָלֶיהָ בָּנוּ בֵּית-כְּנֶסֶת יָפֶה. מִשְׁפָּחוֹת רַבּוֹת יָגוּרוּ שָׁם בִּקְבִיעוּת, וְיוֹתֵר מֵהֶם הֲמוֹן הָעָם הָבָּאִים מֵאֵת הָעִיר לְהִתְפַּלֵּל שָׁמָּה בְּשַׁבָּתוֹת וּבִימֵי מוֹעֵד, אוֹ לָלֶכֶת לָשׂוּחַ בַּמָּקוֹם הַנָּאֶה הַהוּא. כִּי כָל יְהוּדִי יַרְגִּישׁ אֶת נַפְשׁוֹ מְרוֹמָמָה וּנְכוֹנָה אֶל קִרְבַת אֱלֹהִים בִּמְקוֹם קֹדֶשׁ זֶה אֲשֶׁר הִתְהַלֵּךְ אֵלִיָהוּ בּוֹ – גַּם הָעַרְבִים יַאֲמִינוּ בְּמָסֹרֶת זוֹ כְּמוֹ גַם בְּמָסֹרוֹת אֲחֵרוֹת, אֲשֶׁר יַחְשְׁבוּן הַיְּהוּדִים לֶאֱמֶת בְּאָמְרָם, מָסֹרֶת הִיא בְּיָדָם מִפִּי נְבִיאֵיהֶם וּמְלֻמְּדֵּיהֶם הַקַּדְמוֹנִים7. בְּבֵית הַקְּבָרוֹת אֲשֶׁר בַּמָּקוֹם הַזֶּה נִמְצָא קֶבֶר הַמְּקֻבָּל הַמְפֻרְסָם ר' חַיִּים וִיטַאל8.

אֵלֶּה כָּל הַדְּבָרִים אֲשֶׁר בְּפִי עַל דְּבַר הָעִיר הַקַּדְמוֹנִיָּה הַזֹּאת. תּוֹשָׁבֵי הָעִיר הָעַרְבִים הֵמָּה אַנְשֵׁי לֵבָב אֲשֶׁר רוּחַ גַּאֲוַת אֶרֶץ הַמִּזְרָח נוֹסְסָה בָּם, אֲבָל בְּכָל אֵלֶּה אֵין מַעֲצוֹר לִבְנֵי אֵרוֹפָּה לָלֶכֶת בְּבִגְדֵי אַרְצָם בְּכָל מָקוֹם וְלִזְכּוֹת בְּמֶשֶׁךְ זְמָן קָצָר בְּכָל זְכֻיּוֹת אֶזְרְחֵי הָאָרֶץ, וּבְכָל זֹאת נִשְׁקֶפֶת אֵיבָה מְסֻתֶּרֶת פְּעָמִים רַבּוֹת דֶּרֶךְ אֲרֻבּוֹת לְבָבָם. הֵם רַק מְצַפִּים לְהִזְדַּמְּנוּת לְמַעַן הַרְאוֹת עַד כַּמָּה רוֹפְפוּ קִשְׁרֵי הַיְּדִידוּת אֲשֶׁר הִתְקַשְּׁרוּ בָּהֶם עִם בְּנֵי אֵרוֹפָּה “הַפְרַנְקִים”.

עַל אֹדוֹת מִסְפַּר הַיְּהוּדִים אֲשֶׁר בָּהּ, עַל דְּבַר מוֹסְדוֹתֵיהֶם, בָּתֵּי-מִדְרְשֵׁיהֶם וּבָתֵּי-חוֹלֵיהֶם אוֹדִיעַךָ בִּמְּרוּצַת הַזְּמָן, כַּאֲשֶׁר אֲדַבֵּר עַל אֹדוֹתָם בְּדֶרֶךְ כְּלָל, כִּי לֹא מָלְאוּ לִי בְּדַמֶּשֶׂק יֹותֵר מִשְּׁלֹשָׁה יָמִים. וְאַחֲרֵי אֲשֶׁר רָאִיתִי אֶת הַדְּבָרִים הָעִקָּרִיִּים הַמְּצֻיָּנִים, רָכַבְתִּי עַל סוּס בְּחֶבְרַת הַמְשָׁרְתִים אֲשֶׁר שָׂכַרְתִּי שָׁם וָאָבוֹא בְּיוֹם אֶחָד אֶל הַמָּקוֹם אֲשֶׁר אֲנִי יֹושֵׁב בּוֹ הַיּוֹם. וְאַתָּה שָׁלוֹם.


מִכְתָּב שֵׁנִי

צְפָת י“ג תַּמּוּז תקצ”ח

מִכָּאן הָלַכְתִּי אֶל צַיְדָּה (צִידוֹן), מָקוֹם אֲשֶר נִמְצְאוּ בּוֹ מֵאַרְבָּעִים עַד חֲמִשִּׁים מִשְפָּחוֹת יְהוּדִיּוֹת מִתְפַּרְנְסוֹת בְּכָבוֹד, מִקְצָתָם בְּמִסְחָר וּמִקְצָתָם בַּעֲבוֹדַת שְׁכִירֵי יוֹם. וּבִלְעֲדֵיהֶם יָגוּרוּ יְהוּדִים מְכֻבָּדִים בְּדִיר-אַל-קַמַר עִיר מוֹשָׁב נוֹצְרִים וּדְרוּזִים רַבִּים. גַּם חֲכַם הָעֵדָה הַזֹּאת אִישׁ מְלֻמָּד וְסוֹחֵר נִכְבָּד. לַיְּהוּדִים בֵּית-כְּנֶסֶת יָפֶה וּמְלַמְּדִים הַמּוֹרִים לַנֹּעַר אֶת הַלִּמּוּדִים הַנְּחוּצִים וְאַף גַּם כְּתֹב עַרְבִית. קָרוֹב לָעִיר הַהִיא הַמָּקוֹם אֲשֶׁר עַל-פִּי מָסֹרֶת קַדְמוֹנִיָּה עָמַד עָלָיו שִׁקּוּץ צִידוֹנִים בִּימֵי קֶדֶם. עוֹד יַרְאוּ אֶת קֶבֶר זְבֻלוּן בְּגַן גָּדוֹל.

יְעָצוּנִי לְבַל אֹחַז דַּרְכִּי דֶרֶךְ הַיַּבָּשָׁה, כִּי לְפִי הַשְּׁמוּעָה הָיְתָה הַדֶּרֶךְ בְּחֶזְקַת סַכָּנָה. עַל כֵּן הִפְלַגְתִּי בָּאֳנִיָּה וָאָבוֹא בְּמֶשֶׁךְ שְׁנֵי יָמִים עַד קְצֵה גְבוּל אֶרֶץ-יִשְׂרָאֵל, כִּי גַם עַכּוֹ לְאֶרֶץ-יִשְׂרָאֵל תֵּחָשֵב. הַדֶּגֶל הַבְּרִיטִי הַמּוּרָם עַל בֵּית הַצִּיר אֲשֶׁר לָהּ בָּעִיר הַזֹּאת, הָאָדוֹן הַנִּכְבָּד פִינְצִי, מָשַׁךְ אֶת לִבִּי לָסוּר אֶל בֵּיתוֹ. אֶפֶס כִּי לֹא מְצָאתִיו בְּבֵיתוֹ, וּמְתֻרְגְּמָנוֹ הַנִּכְבָּד הִגִּיד לִי, כִּי אֶמְצָאֵהוּ בִּטְבֶרְיָה. עֲדַת יִשְרָאֵל בָּעִיר הַזֹּאת מִצְעָרָה הִיא, כִּי לֹא נִמְצְאוּ בָּהּ יוֹתֵר מִשְּׁתֵּים-עֶשְׂרֵה עַד חֲמֵשׁ-עֶשְׂרֵה מִשְׁפָּחוֹת, וּבְכָל זֹאת נִמְצָא בָּהּ בֵּית-כְּנֶסֶת יָפֶה וּבֵית-סֵפֶר. בִּקַּרְתִּי אֶת בֵּית-הַסֵּפֶר וָאֶבְחַן בִּרְשׁוּת הַמּוֹרֶה אֶת יְדִיעַת הַתַּלְמִידִים בִּלְשׁוֹן עַרְבִית וַיֵּיטִיבוּ לַעֲנוֹתֵנִי עַל כָּל שְׁאֵלוֹתַי. הַמּוֹרֶה מְקַבֵּל שְׂכָרוֹ בְּעַד

כָּל תַּלְמִיד שְׁנֵי גְרוּשׁ אוֹ שִׁשִּׁים פַּארַאס (פְּרוּטוֹת) לְשָׁבוּעַ. זְמַן הַלִּמּוּד מֵחֵל אַחַר תְּפִלַּת הַשַּׁחַר וְנִמְשָׁךְ עַד הָעֶרֶב, וְרַק בְּעֵת הַצָּהֳרַיִם הִיא שְׁעַת מְנוּחָה אֲשֶׁר בָּהּ יֹאכְלוּ הַתַּלְמִידִים לֶחֶם וְקִשּׁוּאִים, וְשָׁעָה אַחַת לִפְנֵי שְקִיעַת הַחַמָּה יֵאָסְפוּ הַתַּלְמִידִים לַחֲצַר בֵּית-הַכְּנֶסֶת לִקְרֹא זֹהַר, תַּלְמוּד וְתוֹרָה. מָצָאתִי כָּאן גַּם רוֹפְאִים יְהוּדִים. מוֹשֵׁל הָעִיר בִּקְּשָנִי בְּסֵבֶר פָּנִים יָפוֹת לְהִתְאַכְסֵן בְּבֵיתוֹ. אֲבָל אֲנִי לֹא מָצָאתִי טוֹב לִי, כִּי-אִם לָסוּר אֶל בַּיִת אֶחָד מֵאַחַי בְּנֵי-יִשְׂרָאֵל, בְּמָקוֹם אֲשֶׁר יֵקַל לִי לַחְקֹר אֵת כָּל דָּבָר הַנּוֹגֵעַ לְאַחֵינוּ הַגָּרִים בַּסְּבִיבָה הַזֹּאת.

הַמְשָׁרְתִים אֲשֶׁר הֵבֵאתִי אִתִּי מִדַּמֶּשֶׂק עֲזָבוּנִי מִפְּנֵי הַסַּכָּנָה הָהוֹלֶכֶת וְרַבָּה מִיּוֹם לְיוֹם. וּמַה יָכֹלְתִּי עֲשׂוֹת? אִם אָמַרְתִּי אָשׁוּב חֲזָרָה עַד בַּיְרוּת, הִנֵּה הַדֶּרֶךְ הַזֶּה הוּא מְסֻכָּן כַּדֶּרֶךְ הַמּוֹלִיךְ לִצְפָת. עַל כֵּן נֶאֱנַסְתִּי לָשֶׁבֶת תַּחְתַּי בְּעַכּוֹ יָמִים אֲחָדִים עַד עֲבֹר סַכָּנַת הַדֶּרֶךְ, כִּי אֵין אִישׁ עָרֵב לְבָבוֹ לְלַוּוֹתֵנִי בַּדֶּרֶךְ עַד צְפָת (דֶּרֶךְ יוֹם וָחֵצִי) אַף כִּי חָפַצְתִּי לְהַרְבּוֹת שְׂכַר לְוָיָה עַד מֵאָה גְרוּשׁ. חָמוּשׁ בְּנֶשֶׁק אָחַזְתִּי בְּדַרְכִּי אַחֲרֵי אֲשֶׁר קָצָה נַפְשִׁי לְחַכּוֹת עוֹד, בְּלִוְיַת שְׁנֵי מוֹלִיכֵי פְּרָדִים אֲשֶׁר חִזַּקְתִּי אֶת לִבָּם בְּאָמְרִי: כִ’צְלִי אַלְלַהּ (בֵּאלֹהִים מִבְטַחִי), וְהֵמָּה הֵשִׁיבוּ: אִן-שַׁאאַ-אַלְלַהּ9 dir=“rtl”>מַאפִ’ישׁ חַוְף (אִם יִרְצֶה הַשֵּׁם, אֵין פַּחַד). אֲבָל הָאֱמֶת אַגִּיד, כִּי לִבִּי הָיָה חָרֵד מְאֹד, וְּביִחוּד בִּרְאוֹתִי אֶת בַּעַל פִּרְדָּתִי מוֹסֵר לִידֵי עַבְדּוֹ קְנֵה-רֹבֶה וַיֹּאמֶר אֵלָיו: “פְּקַח עֵינֶיךָ”, וְאַחֲרֵי דַּבְּרוֹ כַּזֹאת שָׁב חֲזָרָה.

אָחַזְנוּ בַּדֶּרֶךְ וַנִרְכַּב אָט, וְעֵינֵינוּ צוֹפוֹת פַּעַם לְיָמִין וּפַעַם לִשְׂמֹאל. כַּחֲצִי שָׁעָה לִפְנֵי בּוֹא הַשֶּמֶשׁ סַרְנוּ אֶל הַמָּלוֹן אֲשֶׁר עַל הַדֶּרֶךְ, וְהַדְּבָרִים אֲשֶׁר שָׁמַעְנוּ שָׁם גַּם הֵמָּה לֹא הָיוּ מְסֻגָּלִים לְהָסִיר פַּחַד מִלְּבָבֵנוּ. כְּצֵאת הַשֶּׁמֶשׁ עַל הָאָרֶץ שָׁבְנוּ אָחַזְנוּ בַּדֶּרֶךְ, עַד אֲשֶׁר בַּצָּהֳרַיִם בָּאנוּ בִּסְבִיבוֹת צְפָת הָעִיר הַמְּהֻלָּלָה וְהָאֻמְלָלָה, וְהִיא יוֹשֶׁבֶת עַל מְרוֹמֵי הָרִים, וְאִם גַּם הֲרוּסָה עַכְשָו הִנֵּה הִיא כְּמִבְצָר שָׁמֵם וּבְכֹחוֹת מְעַטִּים יְכֹלָה הָיְתָה לִהְיוֹת לִמְקוֹם מִבְצָר לְכָל הַסְּבִיבָה.

כְּדֶרֶךְ שְׁתֵּי שָׁעוֹת לְפָנֶיהָ הָאָרֶץ שְׁמָמָה מֵחֲמַת הָרַעַשׁ10 אֲשֶׁר הֶעְתִּיק צוּרִים מִמְּקוֹמָם וַיְפוֹצְצֵם בְּכֹחוֹ. לֹא רָאִיתִי מְאוּמָה בִּלְתִּי אִם אַבְנֵי גִיר שְׂרוּפוֹת בְּכָל מָקוֹם אֲשֶׁר עֵינִי צוֹפִיָּה, עַד כִּי הָמוּ מֵעַי וָאֶקְרָא: הֲזֹאת נַחֲלַת אֲבוֹתַי? זֹאת הָאָרֶץ שֶׁיֹּאמְרוּ זָבַת חָלָב וּדְבַשׁ הִיא? וְכָל אִישׁ אֲשֶׁר דְּמֵי יִשְׂרָאֵל נוֹזְלִים בְּעוֹרְקָיו יָבִין לְרִגְשׁוֹתַי. תָּפוּשׂ בְּרִגְשׁוֹתַי אֵלֶּה עָזַבְתִּי אֶת סוּסִי לָלֶכֶת אֶל אֲשֶׁר הָיָה חֶפְצוֹ לָלֶכֶת. הֱקִיצוֹתִי מִתַּרְדֵּמַת לְבָבִי וְהִנֵּה אֲנִי לְבַדִּי בִּמְקוֹם תֹּהוּ נוֹרָא. אָז אָחַזְתִּי בְּרִסְנוֹ וַאֶתֵּן אֶת מִתְגִּי בְּפִיו לְהַטּוֹתוֹ הַדֶּרֶךְ, עַד כִּי אַחֲרֵי עֲבֹר שָׁעָה אַחַת נִרְאֲתָה כָּל הָעִיר לְעֵינַי. לִימִינִי הָיָה בֵּית-הַקְּבָרוֹת הַגָּדוֹל אֲשֶׁר עָלָיו אָמַר הָר"י הַקָּדוֹשׁ11, כִּי פִּי שִׁשָּׁה בְּיוֹצְאֵי מִצְרַיִם קְבוּרִים בּוֹ, וּלְפָנַי הַר נִשָּׂא נֶחְפֶּה בֶּחָרְבוֹת הָעִיר וְגַם הוּא מְקוֹם-קְבָרוֹת לַהֲמוֹן מֵאוֹת מֵתֵי הָרַעַש אֲשֶׁר יָרְדוּ חַיִּים שְׁאוֹלָה. הִנֵּה זָכִיתִי עַתָּה לִהְיוֹת בָּאָרֶץ הַנִּבְחֶרֶת אֲשֶׁר עֵינֵי כָּל בְּנֵי-עַמֵּנוּ כָּלוֹת אֵלֶיהָ. וְהִנֵּה אִשָּׁה אַחַת לוֹבֶשֶת בִּגְדֵי אַשְׁכְּנַז נִצֶּבֶת לִקְרָאתִי וַתְּבָרְכֵנִי בְּשָׁלוֹם. וְכִמְעַט הֲשִׁיבוֹתִי לָהּ בְּרָכָה, מִהֲרָה מְלֵאָה שִׂמְחָה לְבַשֵּׂר לְאִישָׁהּ כִּי אוֹרֵחַ אַשְׁכְּנָזִי בָּא, וְגַם הוּא וְכָל בְּנֵי בֵּיתוֹ בָּאִים שְׂבֵעֵי נַחַת וַיּוֹרִידוּנִי מֵעַל הַסּוּס, וְכָל יְגִיעָה אֲשֶׁר הָיְתָה בְּכֹחָם לֹא חָשְׂכוּ מִמֶּנִּי לְהוֹכִיחַ לִי כִּי יָקָר אֲנִי בְּעֵינֵיהֶם.

מִקְצָת הָאֻמְלָלִים הָאֵלֶּה עָזְבוּ אֶת בְּנֵיהֶם וְאֶת קְרוֹבֵיהֶם, אֶת בָּתֵיהֶם וְאֶת כָּל רְכוּשָׁם בְּחוּץ-לָאָרֶץ וַיִּתְנַדְּבוּ לֵאלֹהִים וּלְתוֹרָתוֹ הַקְּדוֹשָׁה. אֲבָל הָהּ! מָה נִכְזְבָה תּוֹחַלְתָּם! בַּדֶּרֶךְ עָבַר עֲלֵיהֶם בִּזָּה וּמַחֲלָה, עֹנִי וְכָל צָרָה עַד בּוֹאָם אֶל הַמָּקוֹם הַנִּבְחָר וְשָׁם מְצָאוּם רָעוֹת רַבּוֹת וְצָרוֹת חֲדָשׁוֹת אֲשֶׁר הִכְרִיחוּ אוֹתָם לְבַטֵּל לִפְעָמִים גַּם אֶת סֵדֶר הַתְּפִלּוֹת הַקְּבוּעוֹת, תַּחַת אֲשֶר דִּמּוּ לָעֲבֹד אֶת אֱלֹהִים כָּל יְמֵיהֶם. וְכָל אִישׁ לֵבָב יַרְגִּישׁ בְּנַפְשׁוֹ אֶת הָרְגָשׁוֹת אֲשֶׁר יִתְרוֹצְצוּ בִּלְבַב הָאֻמְלָלִים הָאֵלֶּה מִדֵּי בָּרְכָם אָת הָאוֹרֵחַ הַבָּא אֲלֵיהֶם מְאֶרֶץ מוֹלַדְתָּם. מַה נִמְשְׁכוּ עֵינֵיהֶם אַחֲרֵי סְפָרַי וּכְלֵי-זֵינִי וְאַחֲרֵי בִּגְדֵי תֻּרְכִּיָּה הַיָּפִים אֲשֶׁר הֵבִיא לִי מוֹלִיךְ פִּרְדָּתִי, בְּעוֹד אֲשֶר אֲנִי יוֹשֵׁב עַל אַדְמַת הַקֹּדֶשׁ תָּפוּשׂ בְּרַעְיוֹנֵי לְבָבִי וּמְחַשֵּׁב אֶת עֶרֶךְ הַתַּעֲנוּג אֲשֶׁר אֶתְעַנֵּג לְסַפֵּר בִּפְרָטִיּוּת לְאַחַי אֲשֶׁר בְּאֵרוֹפָּה אֶת כָּל אֲשֶׁר אֲנִי רוֹאֶה הַיּוֹם, אִם יִזַכֵּנִי אֱלֹהֵי חַסְדִּי לָשׁוּב בְּשָׁלוֹם.

לְעֵת תְּפִלַּת הַמִּנְחָה שִחַרְתִּי אֶת בֵּית-הַכְּנֶסֶת שֶׁל הָר"י הַקָּדוֹשׁ, אֲשֶׁר אַךְ זֶה זְמָן קָצָר שָׁב עַל כַּנּוֹ וְעַל מִשְפָּטוֹ אַחֲרֵי הַחֻרְבָּן אֲשֶׁר הֶחֱרִיבוֹ הָרָעַשׁ. רָאִיתִי כִּי עֵינֵי כָּל הָעָם עָלַי, וְכִמְעַט כִּלִּינוּ לְהִתְפַּלֵּל וְהִנֵּה נִדְבְּרוּ בִּי לָתֵת לִי מִבְּנוֹתֵיהֶם אִשָּׁה, כִּי לֹא הָיָה סָפֵק בְּעֵינֵיהֶם כִּי לְהִשְׁתַּקֵּעַ בָּאתִי לָאָרֶץ הַקְּדוֹשָׁה.

הָלַכְתִּי לָשׂוּחַ בְּחֶבְרָתָם עַל עֲרֵמוֹת אֲבָנִים גְּדוֹלות אֲשֶׁר נִסְתַּכַּנְתִּי עַל כָּל צַעַד וְשַׁעַל לְהִנָּגֵף בָּם אוֹ לִפֹּל וְּלִהִתְגּוֹלֵל מֵרֹאשֵׁי הַצּוּרִים אַרְצָה, וַיְסַפְּרוּ לִי אֶת כָּל הָרָעָה אֲשֶׁר הִדִּיחָה עֲלֵיהֶם מִדַּת הַדִּין הַקָּשָׁה, כִּי הָעִיר הָיְתָה בְּנוּיָה בְּשִׁפּוּעַ הָהָר וּרְחוֹבוֹתֶיהָ מְסֻדָּרִים בְּסֵדֶר נָאֶה, עַד כִּי נִרְאֲתָה בָּעֶרֶב מֵרָחוֹק כְּתַבְנִית הַר בּוֹעֵר בָּאֵשׁ, וּפִתְאֹם נָפְלָה בְּכֹחַ הָרַעַש וַתִּקְבֹּר אֶת יוֹשְבֶיהָ תַּחְתֶּיהָ. בִּשְׁנַת תקצ"ד הִשְׁקוּ אוֹתָם הַמֻּסְלָמִים אֶת כּוֹס הָחֵמָה, כִּי נָפְלוּ עֲלֵיהֶם וַיָּבֹזּוּ אֶת בָּתֵּיהֶם וַיַּהַרְגוּ אֵת כָּל הַבָּא בְּיָדָם, וַיַצּוּרוּ אַרְבָּעָה שָׁבוּעוֹת עַל שְׁאֵרִיתָם אֲשֶׁר מִלְּטוּ חַיֵּיהֶם אֶל אַרְמוֹן יוֹסֵף בֶּן גּוּרְיוֹן12 הַנִּשְׂגָּב בְּקָרְאָם בְּאָזְנֵיהֶם כָּל הַיּוֹם: הַאתּ פְלוּס יַהוּד (יְהוּדִים, הָבוּ כֶּסֶף) וַיַּעֲשׂוּ כָּל תּוֹעֵבָה.

לְפָנַי הֹר-הָהָר אֲשֶׁר עָלָיו נִמְצָא בִּכְפָר מִירוֹם (לְפִי דַעַת אֲחָדִים מוֹרוֹן) קִבְרוֹ שֶׁל ר' שִׁמְעוֹן בֶּן יוֹחַאִי וּבֵית הַקְּבָרוֹת הַגָּדוֹל עִם קֶבֶר הוֹשֵעַ בֶּן בְּאֵרִי.

אַרְבָּעָה יָמִים רְצוּפִים בָּאוּ מִבְּנֵי הָעִיר לְבַקְּרֵנִי בְּכָבוֹד. בְּיוֹם הַשַּׁבָּת נִקְרֵאתִי מֵאֵת רַבִּי אַבְרָהָם דֹּב13 לֶאֱכֹל עַל שֻׁלְחָנוֹ. הָאִישׁ הַזֶּה הוּא אַחַד הָאֲנָשִׁים הַמְּלֻמָּדִים וְהַיְקָרִים אֲשֶׁר רָאִיתִי מֵעוֹדִי, וַיְהִי הֲנָקֵל לוֹ לַעֲבֹד אֶת עֲבוֹדַת הָעֵדָה בְּלִי כָּל קַבָּלַת שָׂכָר מִקֻּפַּת הַקָּהָל, וְהִנֵּה הוּא מְחַלֵּק עוֹד אֵת כָּל אֲשֶׁר לוֹ עִם עֲנִיֵּי עַמּוֹ; כַּעֲשָׂרָה עַד חַמִשָּׁה-עָשָׂר אֲנָשִׁים אוֹכְלִים לֶחֶם חֻקָּם עַל שֻׁלְחָנוֹ תָּמִיד, וְהַהִתְלַהֲבוּת אֲשֶׁר הִתְלַהֲבוּ הָאֲנָשִׁים הָאֵלֶּה בְּנַגְּנָם אֶת זְמִירוֹת הַשַּׁבָּת נָתְנָה עֵדֶיהָ כִּי מִמַּעֲמַקֵּי לְבָבָם הִיא יוֹצֵאת בְּכֹח אֱמוּנַת הָאֱמֶת אֲשֶׁר הֵמָּה מַרְגִּישִׁים. רַבּוֹת יִסְבְּלוּ הַיְּהוּדִים יוֹשְׁבֵי הָאָרֶץ הַזֹּאת עַל דְּבַר הָאֱמֶת אֲשֶׁר הֵמָּה מִתְחַזְּקִים עָלֶיהָ וּבְכָל-זֹאת יִשְׂמְחוּ בְּחֶלְקָם וְיִתְנַחֲמוּ בְּעִנְיָם.

בָּתֵּינוּ וַנְּבַקֵּש מִפְלָט לָנוּ בְּבֵית-הַקְּבָרוֹת, וְהַדְּרוּזִים רָצִים אַחֲרֵינוּ, בְּקוֹל קוֹרֵא לַא תַּ’כַאף, לַא תַּכַ’אף (אַל תִּירָא, אַל תִּירָא). בִּקְּשׁוּ לְהָשִׁיב אוֹתָנוּ בִּדְּבָרִים טוֹבִים, כִּי דִמּוּ לִמְצֹא בְּחֵיקֵנוּ עוֹד דִּבְרֵי חֵפֶץ. אֲבָל אַחֲרֵי אֲשֶׁר לֹא הָיָה שׁוֹמֵעַ לְדִבְרֵיהֶם הַטּוֹבִים מֵאִתָּנוּ, לָבְשׁוּ חֵמות וַיִפְרְצוּ עָלֵינוּ מִכָּל פֵּאָה, וְהַמֻּסְלָמִים יוֹשְׁבֵי הָעִיר שָׂמוּ אִתָּם יָד אַחַת, עַד כִּי אָבְדָה לָנוּ כָּל תִּקְוָה. אוּלָם אֲנִי עָמַדְתִּי, הִבַּטְתִּי אַחֲרַי בִּמְּנוּחָה, כִּי לֹא הָיָה לִי לִדְאֹג עַל דָּבָר בִּלְתִּי אִם עַל חַיַּי, וּמַה בֶּצַע בַּחַיִּים לִרְאוֹת בְּאָבְדַן כַמָּה אַלְפֵי נְפָשוֹת נְקִיּוֹת מִבְּנֵי עַמִּי. עַל-כֵּן לֹא שַׂמְתִּי אֶת לְבָבִי לְהִמָּלֵט, וּמַה גַם אַחֲרֵי אֲשֶׁר לֹא יָדַעְתִּי מְקוֹם מָנוֹס לִי. וְאַךְ לְתֻמִּי נְשָׂאוּנִי רַגְלַי אֶל אַחַד הַכְּפָרִים הַנִּקְרָא עֵין זֵיתּוּן, לְבַל יִרְאוּ עֵינַי לְכָל הַפָּחוֹת בָּרָעָה הַמּוֹצֵאת אֶת אַלְפֵי בְּנֵי-עַמִּי. אֲבָל הָאוֹיְבִים הִמְטִירוּ עָלַי זֶרֶם אֲבָנִים וְכַדּוּרֵי רֹבֶה. וְגַם שָׁמָּה כִּתְּרוּנִי זְקֵנִים, נָשִׁים וִילָדִים הַמְיַלְלִים עַל שִׁבְרָם, וְיוֹשְׁבֵי הַכְּפָר הַנֶּחְשָׁבִים גַּם הֵמָּה לְמִשְׁפַּחַת הַדְּרוּזִים הֵעִיזוּ אוֹתָנוּ אֶל מָקוֹם אֶחָד וַיִּסְגְּרוּנוּ בְּבֵית-כְּנֶסֶת קָטֹן.

וּלְמַלְּאוֹת סְאַת רִשְׁעָתָם וְשׁוֹבְבוּתָם הֵחֵלּוּ לְצַחֶק בָּנוּ צְחוֹק אַכְזָרִי נוֹרָא מְאֹד, כִּי בָאוּ עָלֵינוּ חֲמִשָּׁה דְרוּזִים חֲמוּשִׁים בִּקְנֵי-רֹבֶה, בְּסַכִּינִים, בְּקַרְדֻּמּוֹת וּבְמַקְלוֹת וַיַּכּוּ אוֹתָנוּ בְּחֶלְקֵי גּוּפֵנוּ הַיּוֹתֵר רַכִּים וַיִּפְצְעוּ אוֹתָנוּ פְּצָעִים קָשִׁים עַד שְׁפָךְ דָּם, וְאַחֲרֵי-כֵן גֵּרְשׁוּ אוֹתָנוּ הָחוּצָה, וְעַד אֶלֶף פְּעָמִים כּוֹנְנוּ חַרְבָּם עַל לֵב כָּל אֶחָד מִמֶּנּוּ, לָכֹף אוֹתָנוּ לְהוֹדוֹת עַל הַמָּקוֹם אֲשֶׁר בּוֹ אוֹצְרוֹתֵינוּ טְמוּנִים, עַד כִּי נָפְלוּ מַרְבִּיתֵנוּ אַרְצָה וַיִּתְעַלָּפוּ. אָז נָפְלוּ עָלֵינוּ וַיִּפְשְׁטוּ אֶת בְּגָדֵינוּ וַיַּנִּיחוּ אוֹתָנוּ מִתְגּוֹלְלִים עֲרֻמִּים עַל פְּנֵי הַשָּׂדֶה. אֲבָל אֵלֶּה הָלְכוּ מִמֶּנּוּ וְאֵלֶּה בָּאוּ לְחַדֵּשׁ אַכְזְרִיּוּתָם אִתָּנוּ. בְּעַיִן מְלֵאָה מְנוּחָה הִבַּטְתִּי אֶל הַמָּוֶת הַמְּרַחֵף עַל פָּנָי. וְעַד כַּמָּה פְּעָמִים הִטֵּיתִי רֹאשִׁי אֶל הָחֶרֶב וַאֲבַקֵּשׁ מֵהֶם לְהָתִּיז אוֹתוֹ פַּעַם אֶחָת. וְלֹא אָבוּ לִי הָאַכְזָרִים הָאֵלֶּה וַיּוֹסִיפוּ לְהִתְעוֹלֵל בָּנוּ בְּיִסּוּרִים אֲשֶׁר אֵין קֵץ לָהֶם, עַד אֲשֶׁר כִּלָּה הַלַּיְלָה אֶת צָרַת נַפְשֵׁנוּ וּלְמָחֳרָתוֹ רָוַח לָנוּ לָשׁוּב לִצְפָת כָּל אֲשֶׁר הָיָה עוֹד כֹּחַ בּוֹ לְהָנִיעַ אֶת רַגְלָיו, כִּי עָזְבוּ הַדְּרוּזִים הַזֵּידוֹנִים אֶת הָעִיר בַּשָּׁעָה הַשְּׁבִיעִית בַּבֹּקֶר. שַׁבְתִּי הָעִירָה וַתְּהִי דַרְכִּי הָרִאשׁוֹנָה אֶל חַדְרִי לִרְאוֹת אִם הִשְׁאִירוּ הַשׁוֹדְדִים דָּבָר. וְהִנֵּה אֵין שָׂרִיד לִבְגָדַי וּלַחֲפָצָי.

אֲבָל לֹא עֲלֵיהֶם הָיָה לִבִּי חָרֵד, כִּי-אִם עַל אַחַד-עָשָׂר הַקֻּנְטְרְסִים אֲשֶׁר קִבַּצְתִּי וְהֶעְתַּקְתִּי הַרְבֵּה, עַד כִּי הָיוּ נְכוֹנִים לָבוֹא אֶל בֵּית-הַדְּפוּס, וְעַתָּה מִקְצָתָם נִשְׂרְפוּ וּמִקְצָתָם נִקְרְעוּ לַחֲתִיכוֹת קְטַנּוֹת עַד כִּי לֹא יָכֹלְתִּי לְחַבֵּר אוֹתָם לְאֲחָדִים עוֹד. בַּעַל-בֵּיתִי הָיָה בְּמִסְפַּר הַנִּמְלָטִים וּבְבֵיתוֹ לֹא מָצָאתִי בִּלְתִּי אִם כֵּלִים שְׁבוּרִים, בְּגָדִים קְרוּעִים, נוֹצוֹת נְפוֹצוֹת וּסְפָרִים קְרוּעִים. מִכָּאן הָלַכְתִּי אֶל בֵּית-הָרַב וָאֶמְצָא אוֹתוֹ שׁוֹפֵךְ אֶת לְבָבוֹ לִפְנֵי אֱלֹהִים בְּחֶבְרַת אִשְתּוֹ וַאֲחָדִים מֵאַנְשֵׁי בֵּיתוֹ. שָׁם הִרְבֵּינוּ לִבְכּוֹת עַל צָרַת נַפְשֵׁנוּ עַד אֲשֶׁר הִתְעוֹדַדְתִּי אֲנִי לְהֵיטִיב פָּנַי וַאֲבָרֵךְ אֶת אֱלֹהִים בְּפָנִים שׂוֹחֲקוֹת עַל אֲשֶׁר הִצִּיל אֶת נַפְשׁוֹתֵינוּ מֵחֶרֶב הַפְּרָאִים הָאֵלֶּה. וְאַחַר סִפֵּר לִי הָרַב דֹּב, כִּי כַּאֲשֶׁר נִפְרַדְתִּי מִמֶּנּוּ הִקִּיפוּ הַדְּרוּזִים אוֹתָם גַּם הֵמָּה וַיְעַנּוּם וַיְצָרוּם בְּלִי כָּל חֶמְלָה, וְאֶת כָּל אֲשֶׁר מָצְאוּ בָּזְזוּ לָהֶם. וְגַם אַחֲרֵי אֲשֶׁר נָתְנוּ אֵת כָּל הַמְזֻמָּן אֲשֶׁר נִמְצָא בְּיָדָם, סַךְ מֵאָה וַחֲמִשִּׁים כִּיסִים (הַכִּיס הוּא סַךְ חָמֵשׁ לִירוֹת סְטֶרְלִינְג) הִקְּשׁוּ לִשְׁאֹל מֵאִתָּם עוֹד אֶלֶף כִּיסִים, וַיֶּאֶסְרוּ אֶת הָרַב הַנִּכְבָּד אֶת ר' אַבְרָהָם דֹּב, וַיָּכִינוּ אֶת נַפְשׁוֹ לָמוּת מָוֶת אַכְזָרִי מְאֹד, וְהָרַב הַזֶּה עָמַד בִּמְּנוּחָה, וַיִּשְׁאַל מְעַט מַיִם לִרְחֹץ כַּפָּיו וַיַּצְדִּיק עָלָיו אֶת הַדִּין בַּדְּבָרִים הָאֵלֶּה לֵאמֹר: בָּנַי, הַנִּיחוּ לִי וְאֶקְרָא בְּשֵׁם אֱלֹהִים וַאֲבָרְכֵהוּ עַל מִשְפָּטוֹ אֲשֶׁר שְׁפָטָנוּ הַיּוֹם! וְכַאֲשֶׁר הֵחֵלּוּ מִקְצָת הָעֵדָה בָּעֵת הַהִיא לִבְרֹחַ עַל נַפְשָׁם תָּפְסוּ הַדְּרוּזִים בַּחֶרֶב וַיַּכּוּ בָּעָם עַד כִּי יָגְעָה יָדָם, וְאַחַר יָשְׁבוּ בִּמְּנוּחָה לֶאֱכֹל לֶחֶם, וְאֶת הָרַב הִתִּירוּ מִמַּאֲסָרוֹ אַחֲרֵי אֲשֶׁר עִנּוּהוּ בְּיִסּוּרִים קָשִׁים. וְהִנֵּה הוּא חַי וְעוֹמֵד לְנֶגְדִּי כְּאִלּוּ לֹא נַעֲשָׂה אִתּוֹ דָּבָר.

הַדְּרוּזִים עָזְבוּ אֶת הָעִיר וְהַמֻּסְלָמִים שָׁבוּ אֵלֶיהָ, וַיְהִי הַמַּעֲשֶׂה הָרִאשׁוֹן אֲשֶׁר עָשׂוּ אַחֲרֵי שׁוּבָם לָכֹף אֶת הַיְּהוּדִים לָתֵת לָהֶם כְּתָב תְּעוּדָה כִּי הֵגֵנּוּ עֲלֵיהֶם בֶּאֱמוּנָה וּבְכָל כֹּחָם מֵחֲמַת הַדְּרוּזִים, וְאִם יְמָאֲנוּ הַיְּהוּדִים לָתֵת לָהֶם תְּעוּדַת שֶׁקֶר כָּזֹאת, כִּי עַתָּה אִיְּמוּ לְגָרוֹת בָּהֶם אֶת הָעַרְבִים הַיּוֹשְׁבִים בִּסְבִיבוֹת הָעִיר. וְאָמְנָם לֹא אִיּוּמִים רֵיקִים הָיוּ דִבְרֵיהֶם אֵלֶּה, כִּי כְּבָר נִרְאוּ עִקְּבוֹת הַמַּשְׁחִיתִים הַחֲדָשִׁים הָאֵלֶּה בִּמְּקוֹמוֹת יְדוּעִים, עַל כֵּן נֶאֶנְסוּ הַיְּהוּדִים לָתֵת לָהֶם בִּכְתָב אֶת אֲשֶׁר הֵמָּה מְבַקְּשִׁים.

עָמְדָה הַשְּאֵלָה לְפָנֵינוּ מָה לַעֲשׂוֹת עַתָּה? צַעֲקַת הָעֵדָה הָיְתָה גְּדוֹלָה אֵלַי לְפַתּוֹת אֶת הָרַב לָלֶכֶת אִתִּי עַכּוֹ וּמִשָּׁם אֶל חֵיפָה הַקְּרוֹבָה אֶל הַיָּם, עִיר מוֹשַׁב רַבִּים מִקּוֹנְסוּלֵי אֵרוֹפָּה, כִּי הָעִיר הַזֹּאת בְּטוּחָה מִפַּחַד אוֹיֵב וְנָקֵל לְהִמָּלֵט מִמֶּנָּה לְעִתּוֹת סַכָּנָה חֲדָשָׁה. אֲבָל הָרַב בְּצִדְקָתוֹ אָמַר, כִּי אָמְנָם אִם יַסְכִּימוֹּ רֹב הָעָם אֶל הָעֵצָה הַזֹּאת, כִּי עַתָּה הַחוֹבָה עָלָיו לָלֶכֶת אַחֲרֵי רַבִּים וְלַעֲשׂוֹת אֶת רְצוֹן הָעֵדָה. אוּלָם לְדַעְתּוֹ מְחֻיָּבִים הֵמָּה לָשֶׁבֶת תַּחְתָּם בִּמְקוֹמָם הַזֶּה וְיַעֲבֹר עֲלֵיהֶם מָה, כְּדֵי לְהַחֲזִיק אֶת יִשּׁוּב הָעֵדָה הַקְּדוּמָה הַזֹּאת עַל כַּנָּהּ וּכְמוֹ כֵן לְהַצִּיל אֶת בָּתֵּי-הַכְּנֵסִיּוֹת עַל מְכוֹנָם, אֲשֶׁר יִהְיוּ לֶחֳרָבָה אִם תָּמוּשׁ הָעֵדָה מֵאֵת הָעִיר. בִּדְבָרִים אֵלֶּה הֶחֱלִישׁ הָרַב אֶת רַאֲיָתָם אֲשֶׁר הֵבִיאוּ לְדַעְתָּם מִמַּאֲמַר הַתַּלְמוּד “וְהַגָּלִיל יֶחֱרַב”15, וַיֵּט אֶת לְבָב הָעֵדָה לַעֲמֹד בִּמְקוֹמָם. רַק אֲנִי לְבַדִּי לֹא שָׁמַעְתִּי לְעֲצַת הָרַב אֲשֶׁר יְעָצַנִי בְּטוּב לְבָבוֹ לָשֶׁבֶת אִתּוֹ עַד בּוֹא עֶזְרִי מֵרָחוֹק לַעֲשׂוֹת בְּגָדִים לְכַסּוֹת מַעֲרֻמַּי, כִּי-אִם לָקַחְתִּי בְּשִׂמְחָה מִיָּדָם לֶחֶם כְּדֵי מִחְיַת שְׁנֵי יָמִים אֲשֶׁר נָתְנוּ לִי בְּדִמְעָה וַאֶעֱזֹב אֶת הָעִיר עוֹד בַּיּוֹם הַהוּא.

עוֹד אֲנִי מְדַבֵּר אִתָּם וְהִנֵּה עַרְבִי וּשְׁמוֹ מֻצְטַפָה מֻחַמֵּד תּוֹשַׁב צְפָת בָּא וַיָּבֵא לְהָרַב אֵת כָּל הַכֶּסֶף הַמְזֻמָּן הַנִּמְצָא בְּחֵיקוֹ קָרְבָּן מִנְחָה לְטוֹבַת הָעֵדָה. כִּי הָיָה אִישׁ חֶסֶד מֵעוֹדוֹ לַיְּהוּדִים. וְגַם בַּפַּעַם הַזֹּאת נִסָּה לְכַפֵּר אֶת פְּנֵי הַדְּרוּזִים בְּכַמָּה אֲלָפִים גְּרוּש לְהָשִׁיב אֶת יָדָם מֵאֵת הָרַב, וְלֹא הִצְלִיחַ. וַיְּהִי הַמְּעַט בְּעֵינֵיהֶם אֲשֶׁר לֹא שָׁמְעוּ לִדְּבָרָיו, כִּי בִּקְּשׁוּ לְהָמִית גַּם אוֹתוֹ עַל אֲשֶׁר דִּבֶּר טוֹב עַל הַיְּהוּדִים, וְהָאִישׁ הַזֶּה כִּרְאוֹתוֹ אוֹתִי מִתְכַּסֶּה בְּבֶגֶד קָרוּעַ, נָתַן לִי בְּגָדִים שְׁלֵמִים וּשְׁנֵים-עָשָׂר גְּרוּשׁ בִּמְזֻמָּן.

וְהִנְנִי בְּרֶגַע הַזֶּה עוֹד מֻכֶּה פָּצוּעַ מִכַּף רַגְלִי עַד קָדְקֳדִי. אֲבָל הוֹדוֹת לֵאלֹהִים חַסְדּוֹ עֹודֶנִּי חָי. הַעְתָּקוֹתַי אָבְדוּ, אֲבָל גּוּפֵי הַכְּתָבִים מֻנָּחִים בִּידֵי הַגֵּנֵרַל קוֹנְסוּל אֲשֶׁר לְמֶלֶךְ פְּרוּסְיָה בַּאֲלֶכְּסַנְדְּרִיָּה. אָמְנָם כִּי יַעַבְרוּ עָלַי כַּמָּה שְׁנוֹת עֲבוֹדָה טֶרֶם אַשְׁלִים אֶת נִזְקִי. אֲבָל אֶת עֲמָלִי לֹא אֶכְלָא וְעַל יְגִיעַת נַפְשִׁי לֹא אָחוּס. וּבְדֶרֶךְ כְּלָל לֹא רָפָה רוּחִי, וַאֲקַוֶּה לַעֲשׂוֹת אֶת דַּרְכִּי עַד יְרוּשָׁלַיִם.


מִכְתָּב שְׁלִישִׁי

שְׁכֶם, אוֹר לְיוֹם ה׳ מְנַחֵם אָב תקצ״ח לפ״ק

אַחֲרֵי הִפָּרְדִי בִּקְצָרָה אֲבָל בְּאַהֲבָה מֵאֵת אַחַי, הוֹשִׁיבוּנִי עַל הַסּוּס, כִּי לֹא יָכֹלְתִּי לָלֶכֶת בְּרַגְלָי, וָאֵצֵא מֵאֵת הָעִיר בְּלִוְיַת קָהָל גָּדוֹל יוֹתֵר מֵחֲמִשִּׁים אִישׁ. רָאִיתִי אֵת כָּל כִּכַּר הָאָרֶץ הַהִיא, וַיְהִי בְּעֵינַי כִּשְׂדֵה קְבָרוֹת. רָאִיתִי אֶת בְּנֵי-לִוְיָתִי וַיִּהְיוּ גַם הֵמָּה כְּמֵתִים בְּעֵינָי. כֻּלָּנוּ מַחֲשִׁים, אֵין דּוֹבֵר דָּבָר, כִּי-אִם אִישׁ אִישׁ תָּפוּשׂ בְּרַעְיוֹנֵי לְבָבוֹ עַל דְּבַר מְהוּמַת יוֹם הֶעָבָר. רֹב בְּנֵי-לִוְיָתִי הִתְחַפְּשׂוּ בְּבִגְדֵי בֵּדוּאִים לְמַעַן הִתְכַּחֵשׁ לִפְנֵי הָאוֹיְבִים הַחֲדָשִׁים אֲשֶׁר אֲנַחְנוּ יְרֵאִים מֵהֶם בַּדָּרֶךְ.

כַּחֲצוֹת הַלַּיְלָה בָּאנוּ אֶל הַכְּפָר אַלְרָאמָה16, וַנִּוָּעֵץ לִשְׁבֹּת בּוֹ אֶת יוֹם הַמָּחָר, הוּא יוֹם הַשַּׁבָּת. קָבַעְנוּ לָנוּ מָקוֹם לַחֲנוֹתֵנוּ מִחוּץ לַכְּפָר בְּצֵל עֲצֵי זֵיתִים אֲשֶׁר הֵמָּה רַבִּים מְאֹד בַּכְּפָר הַזֶּה, וַנִּסְעַד אֶת לְבָבֵנוּ בִּפְתוֹתֵי לֶחֶם יְבֵשִׁים הַנִּמְצָאִים בְּיַלְקוּטֵנוּ. רַבַּת שָׂבְעָה נַפְשֵׁנוּ הַחֹרֶב בַּיּוֹם וְהַקֶּרַח בַּלָּיְלָה. כַּחֲצִי שָׁעָה אַחֲרֵי שְׁקִיעַת הַחַמָּה הָיִינוּ נְכוֹנִים לָצֵאת לַדֶּרֶךְ, וְהִנֵּה הָעַרְבִים נוֹהֲגֵי הַפְּרָדִים הַמְלַוִּים אוֹתָנוּ יָעֲצוּ לָשֶׁבֶת תַּחְתֵּינוּ כָּל הַלַּיְלָה מִפַּחַד הָאוֹיְבִים הַקְּרוֹבִים לָנוּ עוֹד לְפִי הַשְׁעָרָתָם. וּבֶאֱמֶת שָׁמַעְנוּ בָּרֶגַע הַהוּא קוֹל בְּנֵי-אָדָם, מִדְרַךְ פַּרְסוֹת גְּמַלִּים וְצִלְצוּל כְּלֵי-זַיִן, עַד כִּי נָמֵס כָּל לֵב מִפָּחַד. וְרַק אֲנִי אִמַּצְתִּי אֶת לִבִּי וָאֲצַו אֶת מְשָׁרְתִי לְהָבִיא אֶת סוּסִי אֵלַי, וּבְדַעְתִּי הָיָה לַעֲשׂוֹת דַּרְכִּי עַד עַכּוֹ יְחִידִי אִם יְמָאֵן מְשָׁרְתִי לְלַוּוֹתֵנִי. עוֹד אָנוּ מִתְוַכְּחִים עַל אֹדוֹת הַדָּבָר הַזֶּה, וְהִנֵּה גְמַלִּים טְעוּנִים אֲבַק שְׂרֵפָה בָּאִים וְאַנְשֵׁי חַיִל מִצְּבָא הַפֶּחָה אִתָּם לְשָׁמְרָם, מִהַרְתִּי אֶל בַּעַל הַמִּשְׁמָר הָרִאשׁוֹן וָאֶשְׁאָלֵהוּ אָנָה פְּנֵיהֶם מוּעָדוֹת? וַיַּעֲנֵנִי כִּי מָחָר יֵלְכוּ לְעַכּוֹ, וְכִי אַנְשֵׁי חַיִל מְסֻדָּרִים חוֹנִים מֵעֵבֶר הַכְּפָר מִמּוּלֵנוּ. הַבְּשׂוֹרָה יָצְקָה בָּנוּ רוּחַ חָדָשׁ עַד כִּי נֵאוֹתוּ גַם הָעַרְבִים לָלֶכֶת אִתָּנוּ.

כְּשָׁעוֹת אֲחָדוֹת לִפְנֵי צֵאת הַשֶּׁמֶשׁ בָּאנוּ אֶל שַׁעַר עַכּוֹ, וְאַחֲרֵי אֲשֶׁר הִרְבּוּ אַנְשֵׁי הַמִּשְׁמָר לַחְקֹר וְלִבְחֹן אֶת דְּבָרֵנוּ נָתְנוּ לָנוּ רִשְׁיוֹן לַחֲנוֹת בְּבֵית-הַקְּבָרוֹת אֲשֶׁר לִפְנֵי הָעִיר עַד נְכוֹן הַיּוֹם. הַשָּׁעוֹת הָאֵלֶּה הָיוּ לִי שְׁעוֹת עָמָל, כִּי רֵיחַ הַמֵּתִים פָּעַל פְּעֻלָּה רָעָה עַל פְּצָעַי אֲשֶׁר לֹא נִרְפְּאוּ עוֹד. כִּמְעַט נִפְתְּחוּ שַׁעֲרֵי הָעִיר, וְהִנֵּה כַּמָּה מֵאוֹת פְּלִיטֵי צְפָת אֲשֶׁר נִמְלְטוּ אֶל עַכּוֹ לְפָנֵינוּ יוֹצְאִים לִקְרָאתֵנוּ וַיְבָרְכוּנִי.

יָחֵף, עָטוּף בִּמְעִיל וּבְלוֹאֵי בֶּגֶד עַל רֹאשִׁי הָלַכְתִּי אֶל הַבַּיִת אֲשֶׁר הָיָה מִשְׁכָּנִי בַּתְּחִלָּה. רָאוּנִי בְּנֵי-הַבַּיִת וַיִּבְכּוּ מָרָה. לֹא הִתְנַחֲמוּ עַד אֲשֶׁר הִבְטַחְתִּי לָהֶם, כִּי הִנְנִי בָּרִיא בְּכָל אֲבָרַי. אַחַר הָלַכְתִּי אֶל סִינְיוֹר פִינְצִי, סְגָן קוֹנְסוּל אַנְגְּלִיָּה בָּעִיר הַהִיא, וְהוּא הֵיטִיב פָּנָיו אֵלַי וַיַּאַסְפֵנִי אֶל בֵּיתוֹ, כִּי הָיָה הַבַּיִת מָלֵא אֲנָשִׁים אֲשֶׁר סָבְלוּ מֵהָאָסוֹן כָּמוֹנִי. דַּעַת כֻּלָּנוּ הָיְתָה כִּי אֵלֵךְ בִּבְגָדַי הַמְטֻלָּאִים אֶל מֻחַמַּד עַבְּד-אִלְהָאדִי אֲשֶׁר הָיִיתִי מְקֹרָב אֵלָיו וּמְכֻבָּד בְּעֵינָיו מְאֹד בִּימֵי עָשְׁרִי. אֲבָל לִי אָבַד עָשְׁרִי, וְהָאִישׁ הַזֶּה אָבַד אֶת כֹּחַ זִכְרוֹנוֹ – וְלֹא יָדַע אֹדוֹתַי דָּבָר, אַף כִּי הֵעִידוּ עָלַי עֲבָדָיו, כִּי אֲנִי הָאִישׁ אֲשֶׁר פִּזֵּר לִי לְפָנִים כָּבוֹד וִיקָר מְלֹא חָפְנַיִם, עַד אֲשֶר הוֹצֵאתִי מֵחֵיקִי אֶת הַפִרְמָאן בְּחוֹתַם מֻחַמַּד עַלִּי הַנִּשְׁאָר לִי לִפְלֵטָה, וָאַרְאֵהוּ אֶת הַחוֹב אֲשֶׁר הֵטִיל עָלָיו אֲדוֹנָיו לָחוּשׁ לְטוֹבָתִי. אָז הֵחֵל כֹּחַ זִכְרוֹנוֹ לְהִתְעוֹרֵר וַיִּתְפַּלֵּא עַל הַשִּׁכְחָה אֲשֶׁר גָּבְרָה עָלָיו, וַיָּשָׁב לַהֲדֹר אֶת פָּנַי וּלְהַבְטִיחֵנִי אֶת עֶזְרָתוֹ. אַף נָתַן בַּיּוֹם הַהוּא אוֹת עַל אַהֲבָתוֹ אֲשֶׁר הִתְחַדְּשָׁה, כִּי אָמַר לָכֹף אֶת בַּעַל הַסּוּס אֲשֶׁר שָׂכַרְתִּי לָקַחַת בִּשְׂכָרוֹ אַךְ שְׁלֹשָׁה-עָשָׂר גְּרוּשׁ תַּחַת מֵאָה אֲשֶׁר הִתְנֵיתִי אִתּוֹ. אוּלָם אֲנִי לֹא קִבַּלְתִּי אֶת הַטּוֹבָה הַזֹּאת מִיָּדוֹ, וָאֲבַקְשֵׁהוּ לִנְטֹשׁ אֶת דְּבַר הַסּוּס וְהַמִּקָּח, וּלְהַלְווֹת לִי כֶּסֶף לִקְנוֹת אֶת הַבְּגָדִים אֲשֶׁר אֲנִי חָסֵר, וְהִנֵּה עַד מְהֵרָה שִׁנָּה טַעַם אַהֲבָתוֹ וַיַּעֲנֵנִי כִּי בִּדְבָרִים הַנּוֹגְעִים אֶל הַכֶּסֶף צָרִיךְ הוּא לִשְׁאֹל בָּרִאשׁוֹנָה‎ אֶת מֻחַמַּד עַלִּי. הֲסִבּוֹתִי אֶת גַּבִּי אֵלָיו דֶּרֶךְ בּוּז וָאֵצֵא מֵאֵת פָּנָיו בְּלִוְיַת פִינְצִי וָאֶקַּח מֵאֵת פִינְצִי אֵת כָּל אֲשֶׁר אֲנִי חָסֵר, אַף כִּי כְּבָר פִּזֵּר כַּסְפּוֹ אֶל כַּמָּה מֵאוֹת עֲנִיִּים לְפָנָי.

נִמְלַכְתִּי וָאֶכְתֹּב מִכְתָּב אֶחָד אֶל הָאָדוֹן נ. מוֹר, הַקּוֹנְסוּל הָאַנְגְּלִי בְּבַיְרוּת, וּמִכְתָּב שֵׁנִי אֶל אוֹהֲבִי אֶל קְלוֹט בַּי רוֹפֵא מֻחַמַּד עַלִּי, לְהוֹדִיעָם אֶת הַתְּלָאָה‎ אֲשֶׁר עָבְרָה עָלַי, וְעַד בּוֹא תְּשׁוּבָתָם שׁוֹטַטְתִּי בְּעַכּוֹ וּבִסְבִיבוֹתֶיהָ. רִאשׁוֹנָה הָלַכְתִּי אֶל חֵיפָה דֶרֶךְ שָׁלֹשׁ שָׁעוֹת מֵעַכּוֹ הַיּוֹשֶׁבֶת עַל מְבוֹא הַיָּם, וּבִגְלַל הַדָּבָר הַזֶּה יִשְׁכְּנוּ בְּתוֹכָהּ קוֹנְסוּלֵי אַנְגְּלִיָּה וְצָרְפַת וְקוֹנְסוּלֵי מַמְלָכוֹת אֲחֵרוֹת רַבִּים. אֵלֶיהָ הִתְמַלְּטוּ בְּעִתּוֹת הַחֵרֶם יַקִּירֵי מִשְׁפִּחוֹת טְבֶרְיָה מִפַּחַד‎ הַדְּרוּזִים, אֲשֶׁר דִּמּוּ לַעֲשׂוֹת לָהּ כַּאֲשֶׁר עָשׂוּ לִצְפָת. וְהֵמָּה מִתְגּוֹלְלִים בַּחֲדָרִים מְנֻוָּלִים וְצָרִים מְאֹד, אַרְבַּע עַד חָמֵשׁ מִשְׁפָּחוֹת בְּחֶדֶר אֶחָד. מָקוֹם לַעֲשׂוֹת צָרְכֵיהֶם לֹא נִתַּן לָהֶם.

בָּעֶרֶב בִּקַּרְתִּי אֶת הַקּוֹנְסוּל הָאַנְגְּלִי מָלָגַמְבּוּ (Malagambu) אֲשֶׁר אֲסָפַנִי אֶל בֵּיתוֹ וַיִּתֶּן לִי חֶדֶר מְיֻחָד וּמִטָּה טוֹבָה לְמִשְׁכָּב. דֶּרֶךְ חֲצִי שָׁעָה מִן הַמָּקוֹם הַזֶּה יִתְנוֹסֵס הַר הַכַּרְמֶל הַמְפֻרְסָם בְּיָפְיוֹ, וּלְרַגְלָיו הַמְּעָרָה הַיָּפָה אֲשֶׁר הִסְתַּתֵּר בָּהּ אֵלִיָּהוּ. עַל-יַד הַמְּעָרָה בָּנוּי בֵּית תְּפִלָּה לְנוֹצְרִים, וְעַל רֹאשׁ הָהָר מִנְזָר. גַּם אַבְנֵי הַמִּזְבֵּחַ אֲשֶׁר הֵקִים אֵלִיָּהוּ בְּקַנְאוֹ אֶת קִנְאַת אֱלֹהִים אֱמֶת בְּתוֹךְ עוֹבְדֵי הַבַּעַל הֵמָּה מַרְאִים שָׁם. הָהָר הַזֶּה פּוֹרִי מְאֹד. גַּנִּים רַבִּים מִכָּל קְצוֹתָיו מִסָּבִיב, וְהָעוֹמֵד עַל רֹאשׁוֹ רוֹאֶה אֶת פְּנֵי הַיָּם. בְּשׁוּבִי לְעֵת הַצָּהֳרַיִם רָאִיתִי תֻּרְכִּי לוֹקֵחַ בְּחָזְקָה מִידֵי יְהוּדִי עָנִי אֶת חֲמוֹרוֹ, שְׁאֵרִית רְכוּשׁוֹ שֶׁנִּשְׁאַר לוֹ, בְּאָמְרוֹ כִּי בְּמִצְוַת אֲדוֹנָיו, קוֹנְסוּל מֶמְשֶׁלֶת צָרְפַת, הוּא עוֹשֶׂה זֹאת. אִי שָׁמָיִם! מַה שׁוֹנִים מִשְׁפְּטֵי אֵרוֹפָּה מִמִּשְׁפְּטֵי אַסְיָה. בְּאֵרוֹפָּה אָמַר מֶלֶךְ צָרְפַת לְהִלָּחֵם בִּשְׁוֵיץ עַל אֲשֶׁר מֵאֲנָה לָתֵת לְאִישׁ יְהוּדִי זְכוּת הַיְשִׁיבָה בְּאַרְצָהּ, וּבְאַסְיָה לוֹקֵחַ קוֹנְסוּל הַמֶּלֶךְ הַהוּא מִידֵי יְהוּדִי אֵת כָּל רְכוּשׁוֹ‎ בְּחָזְקָה. וְנַנִּיחַ כִּי מִזְּדוֹן נַפְשׁוֹ עָשָׂה הַמְשָׁרֵת אֶת מַעֲשֶׂה הֶחָמָס הַזֶּה, בְּכָל-זֹאת נִקְרָא שֵׁם אֲדוֹנָיו עָלָיו. וְאִם גַּם בַּיָּמִים הָאֵלֶּה תַּחַת מֶמְשֶׁלֶת מֻחַמַּד עַלִּי אֲשֶׁר הִנְהִיג תַּרְבּוּת נַעֲלָה יָעֵזוּ אַנְשֵׁי חָמָס לַעֲשֹׁק לְעֵינֵי הַשֶּׁמֶשׁ, מַה סָבְלוּ הַיְּהוּדִים לִפְנֵי זֶה! זֶה לֹא כְּבָר שָׁמַעְתִּי עַל עֹשֶׁק כָּזֶה אֲשֶׁר עָבַר עַל אִישׁ יְהוּדִי. וְיוֹתֵר מִזֶּה, כִּי הֻכָּה הֶעָשׁוּק מִידֵי עוֹשְׁקָיו בְּמֵאָה מַלְקוֹת. בְּכָל זֹאת אָמַר אִבְּרַהִים פֶּחָה לֵאמֹר: ״הֵיטִיבוּ אֲשֶׁר עָשׂוּ״.

בְּשׁוּבִי בִּקֵּשׁ הַקּוֹנְסוּל מִמֶּנִּי לָשֶׁבֶת בְּבֵיתוֹ עַד אֲשֶׁר תַּגִּיעַ תְּשׁוּבַת אֶחָד מִידִידַי אֲשֶׁר כָּתַבְתִּי אֲלֵיהֶם, וְלֹא נֶעְתַּרְתִּי אֵלָיו מִטַּעַם הַנִּמּוּס, כִּי לֹא לְכָבוֹד יִהְיֶה הַדָּבָר לָאָדוֹן פִינְצִי אַשֶׁר פָּתַח דַּלְתוֹת בֵּיתוֹ לִי רִאשׁוֹנָה אִם אֶעֶזְבֶנּוּ עָתָּה. אוּלָם אִם יֵשׁ אֶת לְבַב הַקּוֹנְסוּל לְשָׁרְתֵנִי, כִּי עַתָּה יַעֲשֶׂה אֶת הַטּוֹב וְיַלְוֶה לִי אֵיזֶה מֵאוֹת גְּרוּשׁ. וַיַּעֲנֵנִי הַקּוֹנְסוּל כִּי יֵשׁ וָיֵשׁ אֶת לְבָבוֹ לַעֲשׂוֹת לִי אֶת הַטּוֹבָה הַזֹּאת אַחֲרֵי אֵיזֶה יָמִים, אִם אֶשְׁלַח שָׁלִיחַ מְיֻחָד אֵלָיו אַחֲרֵיהֶם.

בֵּרַכְתִּיו עַל טוּב לְבָבוֹ וְשַׁבְתִּי לְעַכּוֹ. אֲבָל הַשָּׁלִיחַ אֲשֶׁר שָׁלַחְתִּי אֵלָיו אַחֲרֵי יָמִים אֲחָדִים שָׁב רֵיקָם אֵלָי.

אַחֲרֵי יָמִים אֲחָדִים בָּא אֵלַי שָׁלִיחַ מְיֻחָד מֵאוֹהֲבַי אֲשֶׁר בְּבַירוּת וּבְיָדוֹ חֲמֵשׁ מֵאוֹת גְּרוּשׁ בַּעֲדִי וַיְעוֹרְרֵנִי לָלֶכֶת אִתּוֹ אֶל בַּיְרוּת, כִּי שָׁם נָכוֹן לִי עֵזֶר יוֹתֵר גָּדוֹל מִזֶּה. אֲבָל חֶסְרוֹן הַכֶּסֶף לֹא הֱנִיעַנִי לָצֵאת מֵאֶרֶץ-יִשְׂרָאֵל בִּגְלָלוֹ, כִּי-אִם יָשַׁבְתִּי עַל הַחֲמוֹר לָלֶכֶת לִירוּשָׁלַיִם, הָעִיר אֲשֶׁר נִכְסְפָה נַפְשִׁי אֵלֶיהָ, אַחֲרֵי אֲשֶׁר קָנִיתִי לִי בִּמְחִיר 24 גְּרוּשׁ אֶת הַבְּגָדִים אֲשֶׁר אֲנִי חָסֵר וּנְיָר, דְּיוֹ וְעֵטִים. וַיְהִי בְּמַעֲלֵה הֶהָרִים לֹא עָצַר הַחֲמוֹר כֹּחַ לָשֵׂאת אוֹתִי, וּבְמוֹרְדוֹתֵיהֶם לֹא עָצַרְתִּי אֲנִי כֹּחַ לָשֵׂאת אֶת תְּנוּעַת הַמֶּרְכָּב, עַל כֵּן הָלַכְתִּי רַגְלִי כָּל הַדֶּרֶךְ הַהִיא לְרַגְלֵי הַחֲמוֹר. בַּמְּלוֹנִים אֲשֶׁר עָמַדְתִּי לְהִנָּפֵשׁ בָּם בִּקַּשְׁתִּי אֹכֶל וְלֹא מָצָאתִי בִּלְתִּי-אִם לֶחֶם יָבֵשׁ וְחָלָב חָמוּץ פַּעַם אַחַת בַּיּוֹם, וּבְכָל-זֹאת הָיִיתִי שָׂמֵחַ בְּלִבִּי עַל אֲשֶׁר זָכִיתִי לָבוֹא אֶל אֲחֻזַּת אֲבוֹתַי וְסֵפֶר הַתנ״ך בְּיָדִי. אַחֲרֵי מַהֲלַךְ אַרְבָּעָה יָמִים בָּאתִי עַד שְׁכֶם. רָאִיתִי אֵת כָּל הָאָרֶץ בְּיִחוּד מִבּוֹאֲכָה מֵעַכּוֹ וְהִנֵּה הִיא כְּגַן אֱלֹהִים כֻּלָּה מַשְׁקֶה וְיוֹשְׁבֶיהָ בְּרִיאִים וְגִבּוֹרֵי כֹּחַ. לִימִינָהּ הַר-גְּרִזִּים אֲשֶׁר הַבְּרָכָה שׁוֹרָה עָלָיו עַד הַיּוֹם, כִּי הוּא מְכֻסֶּה עֲצֵי פְּרִי וּמַעְיָנוֹת טוֹבִים יוֹצְאִים מִמֶּנּוּ. לִשְׂמֹאלוֹ הַר עֵיבָל אֲשֶׁר הַקְּלָלָה רוֹבֶצֶת בּוֹ עַד הַיּוֹם. תֶּאֱבַל נֶפֶשׁ כָּל הָרוֹאֶה אוֹתוֹ, עַד כִּי לֹא שָׁוְא אֶקְרָא לוֹ הַר אֵבֶל תַּחַת עֵיבָל, כִּי הוּא צַר, עָרֹם וְקוֹדֵר אֲשֶׁר אֵין בּוֹ כָּל עֵץ וְעֵין מָיִם. וְרַק בְּמוֹרַד הָהָר נִמְצָא אֵיזֶה פִּרִי אֲשֶׁר יִקְרָאֶנּוּ הָעַרְבִי בְּשֵׁם (אַלְצַבַּר) סַבְלָנוּת, כִּי קְלִפָּתוֹ וְעָלָיו מְלֵאִים עֳקָצִים עוֹקְצִים שְׁנוּנִים, הַדּוֹקְרִים כִּמְחָטִים אֵת כָּל הַיָּד הַנּוֹגַעַת בָּם נְגִיעָה קַלָּה, עַד כִּי צָרִיךְ הָאָדָם סַבְלָנוּת רַבָּה טֶרֶם יַעֲלֶה בְּיָדוֹ לֶאֱכֹל אֶת הַפְּרִי הַהוּא.

וְהַפְּרִי אָמְנָם מָתוֹק לַחֵךְ, אֲבָל קָשֶׁה לָגּוּף, וּבְנֵי אֵרוֹפָּה יִקְרְאוּ אוֹתוֹ בְּשֵׁם תְּאֵנַת אָדָם, כִּי לְדַעְתָּם הִתְכֵּסּוּ אָדָם וְחַוָּה בַּעֲלֵי הָעֵץ הַזֶּה. מִדֶּרֶךְ בְּנֵי הָאָרֶץ לַעֲשׂוֹת גִּדְרֵי גַנֵּיהֶם מִמִּין הָעֵץ הַזֶּה, כִּי לֹא יוּכַל אִישׁ לַעֲבֹר עֲלֵיהֶם מִבְּלִי שֶׁיִּתְנַזֵּק. גֶּזַע הָעֵץ הַזֶּה הוּא מְחֻבָּר מֵעָלָיו17, כִּי דֶרֶךְ גִּדּוּלוֹ הוּא כָּזֶה: בָּרִאשׁוֹנָה יִצְמַח עָלֶה מִן הָאָרֶץ וְעַל רֹאשׁוֹ עָלֶה שֵׁנִי וְעַל רֹאשׁ הַשֵּׁנִי שְׁלִישִׁי. וּבִרְבוֹת הַיָּמִים יִתְקַשּׁוּ הֶעָלִים וְהָיוּ לְעֵץ וּלְגֶזַע. בָּעֵמֶק אֲשֶׁר בְּתוֹךְ הֶהָרִים הָאֵלֶּה יוֹשֶׁבֶת הָעִיר אֲשֶׁר הִיא רַבָּתִי עָם וּבַעֲלָת רְכוּשׁ. שְׁתַּיִם עַד שָׁלֹשׁ מִשְׁפָּחוֹת מִן הַיְהוּדִים יָגוּרוּ בָּהּ עַל חֶלְקָה אַחַת קְטַנָּה, וּבְכָל זֹאת יִשְׂמְחוּ בְּחֶלְקָם וְיִחְיוּ חַיִּים נְעִימִים, בְּתוֹכָם גַּם סוֹחֲרִים עֲשִׁירִים גַּם אֲנָשִׁים מְלֻמָּדִים. כַּחֲצוֹת הַלַּיְלָה יִתְאַסְּפוּ שְׁלֹשָׁה עַד אַרְבָּעָה אֲנָשִׁים אֶל בֵּית-הַכְּנֶסֶת וְשָׁפְכוּ נַפְשָׁם בִּתְפִלָּה לִפְנֵי אֱלֹהִים, וְאַחַר יִלְמְדוּ תּוֹרָה, תְּהִלִּים עִם פֵּרוּשִׁים וּמִשְׁנָיוֹת וְכַדּוֹמֶה עַד כַּחֲצִי שָׁעָה אַחֲרֵי צֵאת הַשֶּׁמֶשׁ, וְאָז יֵאָסְפוּ כָּל הָעֵדָה לְהִתְפַּלֵּל.‎ אַחֲרֵי הַתְּפִלָּה יִקְרָא אַחַד הַנִּמְצָאִים פֶּרֶק מִן הַתּוֹרָה עִם פֵּרוּשׁ אוֹ פֶּרֶק מִן הַמִּשְׁנָה, וְאַחַר יָשׁוּבוּ אֶל בָּתֵיהֶם לֶאֱכֹל פַּת שַׁחֲרִית וְיִשְׁקְדוּ עַל עֲבוֹדָתָם בְּטוּב לֵבָב. לִפְנֵי שְׁקִיעַת הַשֶּׁמֶשׁ יָשׁוּבוּ אֶל בָּתֵיהֶם, יִתְפַּלְלוּ וְיֹאכְלוּ אֶת הַסְּעֻדָּה הָעִקָּרִית. בְּתוֹכָם מָצָאתִי לַמְדָן אֶחָד אֲשֶׁר תּוֹרָתוֹ אֻמָּנוּתוֹ וּפַרְנָסָתוֹ מֻטֶּלֶת עַל אִשְׁתּוֹ, כַּאֲשֶׁר מִדֶּרֶךְ נְשֵׁי הָאָרֶץ הַזֹּאת, בִּכְלָל, לְהָקֵל אֶת עֹל הַפַּרְנָסָה מֵעַל בַּעֲלֵיהֶן עַל יְדֵי יְגִיעַ כַּפֵּיהֶן. גַּם יְהוּדִי פַּרְסִי מָצָאתִי שָׁם הַמִּתְפַּרְנֵס מִמְּלֶאכֶת הַחַיָּטִים.

עִם הַשֹּׁמְרוֹנִים (הֵמָּה הַכּוּתִים אוֹ גֵרֵי הָאֲרָיוֹת). אֲשֶׁר בָּעִיר הַזֹּאת אֵין לַיְּהוּדִים דָּבָר בִּלְתִּי אִם בִּהְיוֹת לָהֶם אֵיזֶה עֵסֶק אִתָּם.

בעה״ק יְרוּשָׁלַיִם תוב״ב18 י״ב אָב תקצ״ח לפ״ק19

וְעַתָּה, בָּרוּךְ הַשֵּׁם, הִנְנִי בָּעִיר אֲשֶׁר נִכְסַפְתִּי אֵלֶיהָ יָמִים רַבִּים, וְהַמַּחֲשָׁבָה הַזֹּאת הִיא תְּנַחֲמֵנִי עַל כָּל הַצָּרוֹת אֲשֶׁר עָבְרוּ עָלַי בַּדֶּרֶךְ, כִּי סוֹף סוֹף הִנְנִי בִּירוּשָׁלָיִם. עָלִיתִי וְיָרַדְתִּי בְּכָל הַדֶּרֶךְ הַזֶּה בְּמַעֲלוֹת וּבְמוֹרָדוֹת, כִּי יְרוּשָׁלַיִם הָרִים סָבִיב לָהּ. הָעָם הַיּוֹשֵׁב עָלֶיהָ עַזֵּי נֶפֶשׁ הֵמָּה, וְלֹא הָיוּ בְּבָתֵּיהֶם בָּעֵת הַהִיא, כִּי גַם הֵמָּה גַם בַּעֲלֵי שְׁכֶם הָלְכוּ עִם אִבְּרַהִים פֶּחָה לְהִלָּחֵם בַּדְּרוּזִים. אֲנִי בַּדֶּרֶךְ סָמוּךְ לִכְפָר אֶחָד וָאֶפְגַּע בַּהֲמוֹן בַּעֲלֵי מִקְנֶה וְעֶדְרֵיהֶם הָרוֹבְצִים עַל עֵין מַיִם, גָּלְלוּ אֶבֶן מֵעַל פִּי הַמַּעְיָן וַיִּשְׁאֲבוּ מַיִם בְּנֹאד שֶׁל עוֹר לְהַשְׁקוֹת אֶת הָעֲדָרִים. וְאִלּוּ בָּאָה בָּעֵת הַהִיא עַלְמָה אַחַת אֶל הָעֲדָרִים, כִּי עַתָּה אָמַרְתִּי רָחֵל הִיא. לִפְנֵי מְשֹׁל אִבְּרַהִים פֶּחָה בְּסוּרִיָּה וּבְאֶרֶץ-יִשְׂרָאֵל הָיָה הַדֶּרֶךְ בְּחֶזְקַת סַכָּנָה רַבָּה, וְגַם הַיּוֹם חָרְדוּ בְּנֵי-לִוְיָתִי, וַיַּרְאוּנִי אֶת הַמְּקוֹמוֹת אֲשֶׁר הָיוּ לְטֶרֶף, לְרֶצַח וָשֹׁד פְּעָמִים רַבּוֹת. הַדֶּרֶךְ שָׁמְמָה כְּאִלּוּ לֹא עָבְרָה עָלֶיהָ רֶגֶל אֱנוֹשׁ שָׁנִים רַבּוֹת, עַד כִּי לֹא יָכֹלְתִּי לְשַׁעֵר בְּנַפְשִׁי, כִּי אֶל עִיר רַבָּתִי עָם אֲנִי הוֹלֵךְ, כִּי-אִם אֶל מָקוֹם אֲשֶׁר בּוֹ כָּלָה הַיִּשּׁוּב. אֲהָה! דַּרְכֵי צִיּוֹן אֲבֵלוֹת. וּפִתְאֹם נִרְאֲתָה הָעִיר לְעֵינַי, וְהִנֵּה הִיא גְדוֹלָה בֶּאֱמֶת, וְרָאשֵׁי הַמִּגְדָּלִים הַבּוֹלְטִים מִבָּתֵּי תְּפִלַּת הַמֻּסְלָמִים מְיַפִּים אוֹתָהּ מְאֹד. אֲבָל מֵהֲדָרָהּ אֲשֶׁר הָיָה לָהּ בִּימֵי קֶדֶם אֵין שָׂרִיד וְאֵין רֹשֶׁם. וּבְמִדָּה אֲשֶׁר קָרַבְתִּי אֶל הָעִיר בָּהּ בַּמִּדָּה רָבְתָה הַשְּׁמָמָה לְעֵינָי. וְרַק מְעָרוֹת נוֹרָאוֹת נִגְלוּ לִי בְּכָל מָקוֹם אֲשֶׁר עֵינִי צוֹפָה, וַאֲבָנִים מְנֻפָּצוֹת אֲשֶׁר אוּלַי הָיוּ בִּימֵי קֶדֶם פְּאֵר הֵיכְלֵי חֶמְדָּה. הַשְּׁוָקִים אֲשֶׁר עָבַרְתִּי בָּהֶם לָבוֹא אֶל רְחוֹב הַיְּהוּדִים הָיוּ צָרִים וּמְלֻכְלָכִים כְּמוֹ בְּעָרֵי מִצְרָיִם. הָלַכְתִּי אֶל הָרַב ר׳ יִשְׂרָאֵל20, רֹאשׁ עֲדַת הַפְּרוּשִׁים וָאֲבַקֵּשׁ מִמֶּנּוּ בַּיִת לְדִירָה בִּמְחִיר. וְהִנֵּה לֹא קַל הָיָה לִי לְהוֹכִיחַ לוֹ, כִּי יְהוּדִי אֲנִי, כִּי כְּרוֹעֶה עַרְבִי נֶחְשַׁבְתִּי בְּעֵינָיו עַל-פִּי הַבְּגָדִים אֲשֶׁר עָלַי, עַד אֲשֶׁר הִתַּרְתִּי לְעַצְמִי אֶת עֲצַת חז״ל לְהַגִּיד בְּמָקוֹם אֲשֶׁר אֵין מַכִּירִים אוֹתִי, כִּי תַּלְמִיד-חָכָם אָנִי. אָז הִכִּירַנִי לָתֵת מָעוֹן לִי בְּבֵית ר׳ יִצְחָק גְּרַוְוֶר, וּבְיוֹם הַמָּחָר בָּאוּ כָּל רָאשֵׁי הָעֵדָה לְקַבֵּל פָּנַי וַיַּזְמִינוּנִי אִישׁ אִישׁ לְבֵיתוֹ. אֲבָל פְּצָעַי אֲשֶׁר לֹא נִרְפְּאוּ עֲדֶנָה לֹא נְתָנוּנִי לָמוּשׁ מִבֵּיתִי בַּיָּמִים הָרִאשׁוֹנִים.

בְּתִשְׁעָה בְּאָב הָלַכְתִּי אֶל בֵּית-הַכְּנֶסֶת אֲשֶׁר לְהָרַב ר׳ יִשְׂרָאֵל בְּבֵיתוֹ21, וְהוּא אַחַד הַבִּנְיָנִים הַיָּפִים אֲשֶׁר לְאַחֵינוּ בִּירוּשָׁלָיִם. בְּהַשְׁפָּעַת ר׳ צְבִי לֶעהרן, אֲשֶׁר הִרְבָּה לְהֵיטִיב לְאֶרֶץ-יִשְׂרָאֵל וּלְיוֹשְׁבֶיהָ, קָנָה אָחִיו ר׳ עֲקִיבָא לֶעהרן אֶת הֶחָצֵר הָזֹּאת בִּמְחִיר רַב וְסִדֵּר בָּהּ גַּם דִּירוֹת רַבּוֹת הָעוֹמְדוֹת לְהִשָּׂכֵר, גַּם בֵּית-מִדְרָשׁ יָפֶה. אֲבָל עַתָּה נֶאֱסַר מִטַּעַם הַמֶּמְשָׁלָה לְכָל בְּנֵי אֵירוֹפָּה לִקְנוֹת בָּיִת.

אָמְנָם לֹא דָבָר רֵיק הוּא לְהִתְעַנּוֹת אֶת תַּעֲנִית תִּשְׁעָה בְּאָב בִּירוּשָׁלָיִם. הַמַּרְאֶה בִּלְבַדּוֹ דֵי אוֹן בְּיָדוֹ לִתְפֹּשׂ אֶת לְבַב הָאוֹרֵחַ בִּרְאוֹתוֹ דֶרֶךְ חַלּוֹן בֵּית-הַכְּנֶסֶת אֶת הַר הַזֵּיתִים עִם קֶבֶר חֻלְדָּה הַנְּבִיאָה הָעוֹמֵד עָלָיו, וְעַל הַר הַבַּיִת בֵּית-תְּפִלָּה לְמֻסְלָמִים וּבַיִת אַחֵר גָּדוֹל הַנִּקְרָא בְּשֵׁם מִדְרָשׁ שְׁלֹמה22. הָמֵס יִמַּס לֵב הַיְּהוּדִי מִדַּאֲבוֹן נֶפֶשׁ, בְּשָׁמְעוֹ אֶת אָחִיו קוֹרֵא הַקִּינוֹת שָׁם, כִּי כָל הַקִּינוֹת הָרְשׁוּמוֹת בִּמְגִלַּת אֵיכָה אֲשֶׁר הוּא שׁוֹמֵעַ בְּאָזְנָיו, יִרְאֶה בְּעֵינָיו אֶת רִשּׁוּמָם בִּירוּשָׁלַיִם עוֹד הַיּוֹם. וּמַה גַם בְּיוֹם הַמָּחָר כַּאֲשֶׁר הִשְׁמִיעַ בְּעַל-פֶּה קִינוֹת שֶׁנִתְחַבְּרוּ בְּאוֹתָהּ שָׁעָה בִּלְשׁוֹן אַשְׁכְּנַז, וַיַּזְכִּירוּ אֶת נִשְׁמַת אַלְפַּיִם נַפְשׁוֹת בְּנֵי צְפָת אֲשֶׁר יָרְדוּ חַיִּים שְׁאוֹלָה בְּרַעַשׁ הַשָּׁנָה שֶׁעָבְרָה; אָז כָּלְתָה כָּל עַיִן בִּדְמָעוֹת, אֲבָל רַב לָנוּ בְּדִבְרֵי עֶצֶב אֵלֶּה.

רָכַבְתִּי בִּסְבִיבוֹת יְרוּשָׁלַיִם וּבִקַּרְתִּי רִאשׁוֹנָה אֶת מְעָרַת כַּלְבָּא שָׂבוֹעַ23. בָּרִאשׁוֹנָה יָבוֹאוּ אֶל חָצֵר גְּדוֹלָה וְעַל פִּתְחָהּ נָטוּעַ עֵץ גָּבוֹהַּ. מִצַּד שְׂמֹאל נִמְצָא מְרֻבָּע גָּדוֹל חָצוּב בְּצוּר וּמִמֶּנּוּ נִכְנָסִים דֶּרֶךְ פֶּתַח שֵׁנִי וְהוֹלְכִים מֵחֶדֶר לְחֶדֶר, כֻּלָּם חֲצוּבִים בָּאֳמָנוּת וּבְטַעַם טוֹב, עַד הַמָּקוֹם אֲשֶׁר בּוֹ הָאָרוֹן, וְהוּא מְיֻפֶּה בְּפִתּוּחִים יָפִים, תַּבְנִית צְרוֹרוֹת פְּרָחִים, עֲלֵי גֶפֶן וְאַשְׁכּוֹלוֹת. גַּם בַּחֲצַר הַמְּעָרָה נִמְצְאוּ בְּכָל עֵבֶר פִּתּוּחִים יָפִים חֲצוּבִים בָּאֳמָנוּת. קָרוֹב אֶל הַקֶּבֶר הַזֶּה מְעָרַת הַסַּנְהֶדְרִין וּבָהּ שְׁלֹשָׁה וְשִׁבְעִים אֲרוֹנוֹת. וְגַם הַמְּעָרָה הַהִיא מְלֵאָה חֲדָרִים חֲדָרִים קְטַנִּים, וּבְכָל קִיר מִקִּירוֹתֶיהָ מִשְּׁלֹשָׁה עַד אַרְבָּעָה כּוּכִין. סַבּוֹתִי מִשָּׁם וָאֵלֵךְ אֶל מְעָרַת הַצַּדִיק24 וְשָׁם מִקְוֵה מַיִם יְבֵשָׁה, אֲשֶׁר זֶה לֹא מִכְּבָר הֵסֵבּוּ אֶת מְרוּצַת מֵימֶיהָ לְמָקוֹם אַחֵר. אֵין קֵץ לַמְּעָרוֹת הָרַבּוֹת הָאֵלֶּה, וּבִנְיְנֵיהֶן שׁוֹנִים מְאֹד גַּם מִבִּנְיְנֵי הָעַרְבִים גַּם מִבִּנְיְנֵי הַיְּוָנִים.

פָּנִיתִי מִשָּׁם לָלֶכֶת אֶל קֶבֶר חֻלְדָּה אֲשֶׁר בְּהַר הַזֵּיתִים תַּחַת בִּנְיָן גָּדוֹל. הָלַכְתִּי בְּצַד חוֹמַת הָעִיר מִחוּצָה לָהּ וָאֶעֱבֹר אֶת שְׁנֵי הַשְּׁעָרִים הַסְּתוּמִים הַנִּקְרָאִים שַׁעֲרֵי רַחֲמִים עַד בּוֹאִי אֶל הַמָּקוֹם אֲשֶׁר הָיָה בּוֹ עֵין גִּיחוֹן לְפָנִים25 שֶׁרִשּׁוּמוֹ נִכָּר עַד הַיּוֹם כְּאִלּוּ אַךְ עַתָּה יָבְשׁוּ מֵימָיו. רָכַבְתִּי בְּמַעְלֵה הַר הַזֵּיתִים עַד בּוֹאִי אֶל‎ קֶבֶר חֻלְדָּה. שָׁם אוֹרוּ עֵינַי, כִּי רָאִיתִי אֶת הַמָּקוֹם אֲשֶׁר עָמַד עָלָיו בֵּית מִקְדָּשֵׁנוּ. עַתָּה עוֹמֵד עָלָיו בֵּית-תְּפִלָּה לְמֻסְלָמִים וּבֵית מִדְרַשׁ שְׁלֹמֹה, הַמָּקוֹם אֲשֶׁר מִגְדַּל בֵּית-הַתְּפִלָּה הָיָה לְפִי מַה שֶׁאוֹמְרִים מְקוֹם קֹדֶשׁ הַקֳּדָשִׁים, וּבְתוֹכוֹ מַרְאִים עוֹד שְׂרִידֵי אֶבֶן-יְסוֹד גְּדוֹלָה הִיא אֶבֶן-הַשְּׁתִיָּה הַמְפֻרְסֶמֶת, אֲשֶׁר לְמַעֲלַת קְדֻשָּׁתָהּ הִקִּיפוּ אוֹתָהּ בְּגָדֵר. גַּם מִדְרָשׁ שְׁלֹמֹה הוּא בִּנְיָן יָפֶה, וּבוֹ סִפְרִיַּת כְּתָבֵי-יָד רַבִּים בְּלָשׁוֹן עַרְבִי, וּמְלֻמָּדִים רַבִּים שׁוֹקְדִים בּוֹ עַל לִמּוּדָם וּמְבָאֲרִים אֶת סֵפֶר הַקֻּרְאָן (סֵפֶר מִקְרָא, סֵפֶר תּוֹרַת מֻחַמַּד). אִבְּרַהִים פֶּחָה הִתִּיר לַיְּהוּדִים וּלְכָל אוֹרֵחַ לְבַקֵּר אֶת מְקוֹם הַקֹּדֶשׁ הַזֶּה, וּבְכָל זֹאת מְבַקְרָיו מְעַטִּים, הַיְּהוּדִים חוֹשְׁשִׁים לָבוֹא אֶל

הַמָּקוֹם אֲשֶׁר לֹא בָּא בּוֹ גַם הַכֹּהֵן הַגָּדוֹל יוֹתֵר מִפַּעַם אַחַת בְּשָׁנָה26, וּלְאוֹרְחֵי אֵרוֹפָּה גֵם הֵמָּה לֹא יַעֲלֶה הַדָּבָר לָבוֹא שָׁמָּה בִּלְתִּי-אִם אַחֲרֵי יְגִיעָה רַבָּה. רָאִיתִי מֵרֹאשׁ הָהָר הַזֶּה אֵת כָּל חֶבֶל הָאָרֶץ הַהִיא עַד יָם הַמֶּלַח. לִימִינִי הָיְתָה רָמָה מְקוֹם קֶבֶר שְׁמוּאֵל.

גַּם עַל הַקֶּבֶר הַהוּא הֶעֱמִידוּ הַמֻּסְלָמִים בֵּית-תְּפִלָּה יָפֶה. אֲבָל הַיְּהוּדִים גַּם הֵמָּה יְבַקְּרוּ אוֹתוֹ. לִשְׂמֹאלִי רָאִיתִי אֶת הָעִיר כֻּלָּה עַד יַד-אַבְשָׁלוֹם הַמְפֻרְסָם, עֵמֶק יְהוֹשָׁפָט וְקֶבֶר זְכַרְיָה, נִכְנַסְתִּי בְּקֶבֶר חֻלְדָּה וָאֵרֵד בַּמַּדְרֵגוֹת מַטָּה מָטָּה עָמֹק מְאֹד עַד הַחֶדֶר הַתַּחְתּוֹן, וָאֶמְצָא אֶת הַקֶּבֶר בַּתַּבְנִית הַזֹּאת

מְכֻסֶּה בְּמִטְפָּחוֹת רַבּוֹת וְעָלָיו מְנוֹרוֹת בִּלְתִּי דְלוּקוֹת בָּעֵת הַהִיא. הַקַּרְקַע רְצוּפָה בְּמַחְצְלוֹת קַשׁ וַעֲלֵיהֶן יִכְרְעוּ הַמֻּסְלָמִים וְיִתְפַּלְלוּ מִדֵּי בּוֹאָם שָׁמָּה. אָמְנָם לֹא נִמְצְאָה שָׁם כָּל מַצֵּבָה וּכְתֹבֶת. אֲבָל כְּבָר הִגַּדְתִּי לְמַעְלָה כִּי הַמֻּסְלָמִים מַאֲמִינִים וּמוֹדִים בְּכָל דָּבָר אֲשֶׁר בָּאָה עָלָיו הַמָּסֹרֶת הַיְּהוּדִית. הַקֶּבֶר הַזֶּה, לְפִי עֵדוּת הַתַּלְמוּד, הָיָה לְפָנִים בְּתוֹךְ הָעִיר27. אֲחֵרִים אוֹמְרִים שֶׁהָעֲצָמוֹת הַנִּמְצָאוֹת בַּקְּבָרִים הָאֵלֶּה הוּבְאוּ לְנַחַל קִדְרוֹן. אֵיךְ שֶׁיִּהְיֶה עַתָּה הֵמָּה כִּבְרַת אֶרֶץ רְחוֹקָה מִמֶּנָּה וָחוּצָה – וְהַחוֹמָה אֵינֶנָּה מַקֶפֶת אֵפוֹא בִּלְתִּי אִם חֵלֶק אֶחָד מֵאֵת הָעִיר הַקַּדְמוֹנִיָּה. קָרוֹב לָהּ נִמְצָאִים קִבְרֵי חַגַּי וּמַלְאָכִי אֲשֶׁר גַּם הֵמָּה בִּנְיָנִים יָפִים. הָלַכְתִּי מַטָּה לְעֻמַּת יַד אַבְשָׁלוֹם אֲשֶׁר הוּא בַּיִת יָפֶה בָּנוּי עַל-פִּי חֻקֵּי הַטַּעַם הַטּוֹב. אֲבָל אֵינֶנּוּ גָדוֹל בְּכַמּוּתוֹ. עוֹד לִפְנֵי אֵיזֶה שָׁנִים הָיְתָה מִתְרוֹמֶמֶת יָד שֶׁל אֶבֶן מֵרֹאשׁ הַבִּנְיָן הַזֶּה. אֲבָל עַתָּה גָבַר עָלֶיהָ הָרוּחַ וַיְפוֹצְצָהּ28, וְאֶצְלוֹ מְעָרָה גְדוֹלָה מְיֻפֶּה בְּעַמוּדִים וּפִתּוּחִים וּבְתוֹכוֹ קֶבֶר זְכַרְיָה. גַּם הוּא סָתוּם מִכָּל צַד. אֲבָל מְשֻׁנֶּה בְּבִנְיָנוֹ מִקֶּבֶר אַבְשָׁלוֹם.

שָׁמַעְתִּי מְסַפְּרִים, כִּי בְּקִירוֹת הַקֶּבֶר הַזֶּה דְבוּקִים עוֹד כַּמָּה רְסִיסֵי דָם. אֲבָל אֲנִי לֹא יָכֹלְתִּי לִרְאוֹתָם, אַף כִּי הֶעֱמַקְתִּי לְהַבִּיט דֶּרֶךְ בֶּקַע אֶחָד אֲשֶׁר הָיָה פָּתוּחַ לְפָנַי בָּקִּיר לָרְוָחָה.

כָּל הָהָר הַהוּא וְכָל הָעֵמֶק מְלֵאִים קִבְרֵי מֵתִים וּמְעָרוֹת. כָּל קֶבֶר מְכֻסֶּה בְּאֶבֶן גְּדוֹלָה, וְהַרְבֵּה מֵהַכְּתָבוֹת אֲשֶׁר עֲלֵיהֶן אֶפְשָׁר עוֹד לִקְרֹא בָּרוּר. מִשָּׁם הוֹלִיכוּנִי אֶל עֵין מַיִם חָצוּב בְּצוּר עָמֹק מְאֹד. מַדְרֵגוֹת רַבּוֹת מוֹלִיכוֹת אֶל קַרְקָעִיתוֹ, וְהוּא רָחָב מְאֹד עַד כִּי יוּכְלוּ אֲנָשִׁים רַבִּים לִטְבֹּל בּוֹ יַחַד, וַיֹּאמְרוּ לִי כִּי הוּא מִקְוֵה הַטְּבִילָה שֶׁל הַכֹּהֵן הַגָּדוֹל29, וְדִבְרֵיהֶם הֵמָּה קְרוֹבִים אֶל הָאֱמֶת לְפִי הַשְׁעָרָתִי, כִּי פִּתְחָהּ הָרָחָב וּמַדְרְגוֹתֶיהָ הַיָּפוֹת יוֹכִיחוּ עָלֶיהָ, כִּי לְתַשְׁמִישׁ אִישׁ גָּדוֹל נוֹעָדָה. אַף מְקוֹמָהּ בְּהַר-הַבָּיִת. וְאוּלַי הָיָה דֶרֶךְ נִסְתָּר מֵאֵת הָעֲזָרָה אֵלֶיהָ. הָלַכְתִּי מִמֶּנָּה אֶל עֵין מַיִם אַחֵר אֲשֶׁר הָעַרְבִים יִקְרְאוּ לוֹ בְּשֵׁם שׁוֹבִּי, וְהַיְּהוּדִים אוֹמְרִים כִּי הוּא מֵי הַשִּׁילוֹחַ אֲשֶׁר יַעֲלוּ בַּחוֹל וְיֹאבְדוּ, וְיָשׁוּבוּ וְיֵרָאוּ בַּמָּקוֹם הַזֶּה, וּבָזֶה גָמַרְתִּי אֶת הַ״זִּיאָרָאתּ״30 בַּיּוֹם הָרִאשׁוֹן.

מִמָּחֳרַת הַיּוֹם שַׁבְתִּי לְבַקֵּר אֶת קַדְמוֹנִיּוֹת הָעִיר בְּחֶבְרַת הָרַב הַנִּכְבָּד ר׳ יִשְׂרָאֵל וִיהוּדִים רַבִּים אֲחֵרִים, וַנָּשֶׂם גַּם הַפַּעַם אֶל הַר הַזֵּיתִים פָּנֵינוּ.

כְּדֶרֶךְ שָׁעָה מִחוּץ לָעִיר הֶרְאַנִי הָרַב אַחַד הַבִּנְיָנִים הַגְּדוֹלִים אֲשֶׁר רָאִיתִי מֵעוֹדִי. עָמַדְתִּי מִחוּץ לַבַּיִת עַל יַד הַחַלּוֹנוֹת הָעֶלְיוֹנִים, וַנַּבֵּט אֶל מְבוֹא הַבַּיִת דֶּלֶת גְּדוֹלָה מִתַּחְתֵּינוּ בַּמַּעֲמַקִּים. שִׁעַרְתִּי בְּנַפְשִׁי כַּמָּה הִתְמוֹטְטָה אַדְמַת יְרוּשָׁלַיִם אִם בִּנְיָן מִגֹּבַהּ כָּזֶה שָׁקַע כָּל-כָּךְ לְמָטָּה. הַבִּנְיָן הַזֶּה נִקְרָא בְּפִיהֶם בֵּית-הַכְּנֶסֶת שֶׁל ראב״ע(?) לְסִבָּה לֹא יָדַעְתִּי שַׁחֲרָהּ. אֹפֶן הַבִּנְיָן דּוֹמֶה לָזֶה שֶׁל בִּנְיָן ביהכ״נ שֶׁל רשב״י בְּמִירוֹן, הָלַכְנוּ מִמְּעָרָה לִמְעָרָה. אַחַר יָשַׁבְנוּ וַנֹּאכַל אֲרֻחַת קִשּׁוּאִים פְּשׁוּטָה, וָאָשׁוּב לְבֵיתִי שָׂמֵחַ וְטוֹב לֵב עַל אֲשֶׁר זָכִיתִי לִהְיוֹת בְּחֶבְרַת יְהוּדִים מְצֻיָּנִים כָּאֵלֶּה.

וּבְשׁוּבֵנוּ בִּקַּרְנוּ עוֹד אֶת קִבְרֵי דָוִד וּשְׁלֹמֹה31, וְהוּא בִּנְיָן יָפֶה מִבַּחוּץ, אֲבָל בִּפְנִימִיּוּתוֹ לֹא נִמְצָא דָבָר מְצֻיָּן, וְאַחֲרֵי אֲשֶׁר לֹא מָצְאָה יָדִי לָתֵת לְהַשּׁוֹמֵר כֶּסֶף דֵּי שְׁאֵלָתוֹ לֹא נִכְנַסְתִּי לְתוֹכוֹ.

בַּיּוֹם הַשְּׁלִישִׁי בִּקַּרְתִּי אֶת הַמָּקוֹם הַמְפֻרְסָם בְּכָל הַגּוֹלָה, אֶת הַכֹּתֶל הַמַּעֲרָבִי – זִכָּרוֹן לִימֵי קֶדֶם. שָׁם רָאִיתִי עוֹד תִּשְׁעָה טוּרֵי אֶבֶן מֵחוֹמַת הַיְסוֹד. הָאֲבָנִים גְּדוֹלוֹת מְאֹד. מִקְצָתָן בְּנוֹת שְׁתֵּים-עֶשְׂרֵה אַמָּה בְּאָרְכָּן, וְרָחְבָּן כְּקוֹמַת אִישׁ.

וְהַפְּחוּתוֹת בָּהֶן בְּנוֹת שְׁתֵּי אַמּוֹת בָּאֹרֶךְ וְרֶבַע אַמָּה בָּרֹחַב, וְכֻלָּן מוּסַתּוֹת עַל אַרְבַּע קְצוֹת שִׁטְחָן, עַד כִּי הַחֵלֶק הָאֶמְצָעִי בּוֹלֵט וְיוֹצֵא. אַף הֵן אֵינָן מְחֻבָּרוֹת זוֹ לְזוֹ בְּסִיד וּבְחֹמֶר, וּבְכָל-זֹאת תִּתְלַכֵּדְנָה אִשָּׁה בַּאֲחוֹתָהּ עַד כִּי אֵין לְהַכְנִיס מַחַט בֵּינֵיהֶן.

אֶל מְקוֹם הַקֹּדֶשׁ הַזֶּה יֵאָסְפוּ הַיְּהוּדִים בְּעַרְבֵי שַׁבָּתוֹת לִשְׁפֹּךְ שִׂיחָם לִפְנֵי ה׳ בְּעַד שְׁלוֹם אֲחֵיהֶם הַיּוֹשְׁבִים עַל אַדְמַת נֵכָר, וְאֶת אַנְשֵׁי חַסְדָּם יַזְכִּירוּ אִישׁ אִישׁ בִּשְׁמוֹ וְיִתְפַּלְלוּ בַּעֲדוֹ בְּיִחוּד. לִימִין הַחוֹמָה הַקְּדוֹשָׁה הַזֹּאת עוֹמֵד בַּיִת קָטֹן, וַיְסַפֵּר לִי בֶּן-לִוְיָתִי כִּי בְּתוֹךְ הַבַּיִת הַקָּטֹן הַזֶּה יֶשְׁנָן מַדְרֵגוֹת גְּבוֹהוֹת וּמֵהֶן יִרְאוּ אֶת פְּנִים הַר הַבַּיִת וְאַחַד הַבִּנְיָנִים הַסְּמוּכִים לוֹ. אֶפֶס כִּי לֹא יָעֵז לְלַוּוֹתֵנִי אֶל הַבַּיִת הַזֶּה מִפְּנֵי בִּגְדֵי הַיְּהוּדִים הַפּוֹלָנִים אֲשֶׁר הוּא לוֹבֵשׁ וְגַם מֵחֶסְרוֹן יְדִיעָתוֹ בְּלָשׁוֹן עַרְבִי. עַל-כֵּן הִרְהַבְתִּי בְּנַפְשִׁי עֹז וָאֵלֵךְ לְבַדִּי. גַּם מָצָאתִי אֶת הַמַּדְרֵגָה הַנִּשְׁקָפָה עַל-יְדֵי חַלּוֹן אֶל פְּנִים הֶחָצֵר. רָאִיתִי חֲדָרִים רַבִּים שֶׁבָּהֶם פְּרוּשׂוֹת מַחֲצָלוֹת שֶׁל קַשׁ. אֲבָל זוּלָתָם לֹא רָאִיתִי בּוֹ כָּל דָּבָר יָקָר. וְהַחֲדָרִים הָאֵלֶּה לֹא יִפָּתְחוּ בִּלְתִּי אִם לְעִתִּים קְבוּעוֹת כְּמוֹ בְּחֹדֶשׁ רַמַצאן, וְכַדּוֹמֶה. אַבְנֵיהֶם עוֹד גְּדוֹלוֹת כִּמְעַט מֵהָאֲבָנִים אֲשֶׁר רָאִיתִי בָּרִאשׁוֹנָה.

הוֹסַפְתִּי לָלֶכֶת עַד פֶּתַח הַר הַבַּיִת וְהִנֵּה מֻסְלָמִי אֶחָד הִזְהִיר אוֹתִי לְבַל אַרְחִיק לָלֶכֶת הָלְאָה פֶּן יִקְרָאֵנִי אָסוֹן. שָׁמַעְתִּי לַעֲצָתוֹ, כִּי יָדַעְתִּי אֶת הַתְּנָאִים אֲשֶׁר עַל-פִּיהֶם יוּכַל בֶּן אֵרוֹפָּה לְהִכָּנֵס כָּאן, וָאֵלֵךְ אַחֲרֵי הַמֻּסְלָמִי אֶל אַחַד הַחֲדָרִים הַנִּשְׁקָפִים אֶל הַר הַבַּיִת, בְּמָקוֹם אֲשֶׁר לֹא הָיָה מַעֲצוֹר לְפָנַי לִרְאוֹת אֶת הַכֹּל בָּרוּר. עוֹד הַיּוֹם נִרְאוּ בּוֹ שְׁנֵים-עָשָׂר שְׁעָרִים.

אָרְכּוֹ חֲמֵשׁ מֵאוֹת אַמָּה וְרָחְבּוֹ שְׁלֹשׁ מֵאוֹת, כַּאֲשֶׁר הָיָה לְפָנִים וְעָלָיו נְטוּעִים עֲצֵי בְּרוֹשִׁים (Cypress) רַבִּים הַנִּרְאִים בְּיִחוּד מִפְּאַת מַעֲרָב. צִיַּרְתִּי צִיּוּר מִכָּל אֲשֶׁר אֲנִי רוֹאֶה, לְהַרְאוֹת אֶת תַּבְנִית הַמָּקוֹם לְאוֹהֲבַי אֲשֶׁר בְּאֵרוֹפָּה אִם יְשִׁיבֵנִי אֱלֹהֵי חַסְדִּי אֲלֵיהֶם.

מִחוּץ לַחוֹמָה הֶרְאוּנִי אֶת הַמָּקוֹם אֲשֶׁר הָיָה בּוֹ שַׁעַר הַמֵּתִים לְפָנִים, וּבוֹ בּוֹר מַיִם גָּדוֹל מְאֹד וַאֲרוֹן-מֵתִים יָפֶה עַל יָדוֹ, אֲשֶׁר תַּבְנִיתוֹ מִבִּפְנִים שׁוֹנָה מִתַּבְנִית הָאֲרוֹנוֹת הָאֲחֵרִים, וּבוֹ נְקָבִים רַבִּים אֲשֶׁר נֻקְּבוּ בְּכַוָּנָה מְיֻחֶדֶת לְהוֹצִיא מִמֶּנּוּ אֶת הַמָּיִם. בְּנֵי-לִוְיָתִי אָמְרוּ, כִּי בּוֹ הָיוּ רוֹחֲצִים אֶת הַמֵּתִים לְפָנִים, וְנִרְאִים דִּבְרֵיהֶם בְּעֵינַי, אַחֲרֵי אֲשֶׁר מְקוֹם הָאָרוֹן מוֹכִיחַ עָלָיו כִּי לֹא עַל פִּי מִקְרֶה בָּא שָׁמָּה.

בְּעֶרֶב שַׁבָּת בִּקַּרְתִּי אֶת בֵּית-הַתְּפִלָּה שֶׁל הַסְּפַרְדִים32, וָאֵלֵךְ מִבֵּית-כְּנֶסֶת אֶל בֵּית-כְּנֶסֶת אֲשֶׁר כֻּלָּם סְמוּכִים זֶה לָזֶה. לִפְנֵי מֶמְשֶׁלֶת אִבְּרַהִים פֶּחָה לֹא הָיָה לָהֶם כִּמְעַט מָקוֹם לַחֲסוֹת בּוֹ מִפְּנֵי הַגְּשָׁמִים, וְגַם תִּקּוּן מְקוֹמוֹת כָּאֵלֶּה לֹא הֻתְּרָה לָהֶם כִּי-אִם אַחֲרֵי הִשְׁתַּדְּלוּת מְרֻבָּה וְעַתָּה הִנֵּה בָּתֵּי-הַכְּנֶסֶת שֶׁלָּהֶם הַמְפֹאָרִים בְּכָל בָּתֵּי-הַכְּנֶסֶת אֲשֶׁר רָאִיתִי מֵעוֹדִי. בְּאֶחָד מֵהֶם נִמִצָא מְקוֹם מִשְׁקָע מְקֻדָּשׁ בִּקְדֻשָּׁה יְתֵרָה. יֵשׁ אוֹמְרִים, כִּי הוּא מְקוֹם תְּפִלַּת אֵלִיָּהוּ, וְיֵשׁ אוֹמְרִים, כִּי הַקְּדֻשָּׁה הַמְיֻחֶסֶת לַמָּקוֹם הַזֶּה מְיֻסֶּדֶת עַל מַאֲמַר חז״ל: ״מֵעוֹלָם לֹא זָזָה הַשְּׁכִינָה מִמָּקוֹם זֶה״. בְּבֵית-תְּפִלָּה אֶחָד הֶרְאוּנִי פַּךְ שֶׁמֶן שֶׁל נְחֹשֶׁת, יָשָׁן מְאֹד, אֲשֶׁר לֹא יָזִיזוּ אוֹתוֹ מִמְּקוֹמוֹ בִּלְתִּי אִם עַל פִּי הַדְּחַק וְגַם אָז עַל יְדֵי גוֹי. הֵם סִדְּרוּ גַם בְּאֵר מַזְרֵק מַיִם (אַרְטֵזִי) אֲשֶׁר יִזְרֹק מֵימָיו בְּסֻכּוֹת, כִּי הֵמָּה שְׂמֵחִים מְאֹד בְּשִׂמְחַת בֵּית הַשּׁוֹאֵבָה, וְחַג גָּדוֹל הוּא לְכָל הָעִיר עֵת נְסֹךְ הַמָּיִם. גַּם רָאִיתִי בְּאַחַד בָּתֵּי-הַכְּנֶסֶת אֶת אֲרוֹן-הַקֹּדֶשׁ שֶׁנַּעֲשָׂה בְּמִצְוַת ר׳ צְבִי לֶעהרן33, יָפֶה וּמְהֻדָּר בְּכָל פְּאֵר. פָּנִיתִי וָאֵלֵךְ אֶל ״בֵּית-יְשִׁיבָה״ קָטֹן אֲשֶׁר בּוֹ יִתְפַּלְלוּ כָּל הַשָּׁנָה עַל-פִּי כַּוָּנַת הָאר״י34, מֵחֲצוֹת הַלַּיְלָה עַד הַצָּהֳרַיִם יִתְפַּלְלוּ בּוֹ מִשְּׁלֹשִׁים עַד אַרְבָּעִים נָפֶשׁ. בָּעֶרֶב הָלַכְתִּי אֶל בֵּית-הַכְּנֶסֶת שֶׁל הָרַב ר׳ יִשְׂרָאֵל, וָאֶמְצָא אֶת הָעֵדָה מוֹדִים לַה׳ חַסְדּוֹ עַל הַמַּגֵּפָה אֲשֶׁר נֶעֱצְרָה אַךְ זֶה יָמִים אֲחָדִים לִפְנֵי בּוֹאִי יְרוּשָׁלַיְמָה, הִיא הַמַּגֵּפָה אֲשֶׁר טָרְפָה כַּמָּה נְפָשׁוֹת מֵהָעֵדָה וַאֲשֶׁר אָכְלָה גַם אֶת כַּסְפָּם, כִּי רַק אַחֲרֵי אֲשֶׁר הִרְבּוּ מְחִיר נִתַּן לְחוֹלֵיהֶם מָקוֹם מְיֻחָד בְּבֵית-הַחוֹלִים הַכְּלָלִי.

שָׁפְרוּ לִי הַחַיִּים בִּירוּשָׁלַיִם כַּאֲשֶׁר עָבְרוּ יְמֵי הַזָּעַם. שׁוֹכִחִים כָּאן אֵת כָּל הַדְּאָגוֹת אֲשֶׁר מְלֵאִים אוֹתָן בְּאֵרוֹפָּה לִחְיוֹת חַיֵּי כָּבוֹד כְּמוֹ שֶׁרְגִילִים לַחְשֹׁב שָׁם. כִּי פֹּה בִּירוּשָׁלַיִם אֵין דְּאָגָה וְאֵין תַּכְלִית אַחֶרֶת מֵאֲשֶׁר לִשְׁקֹד עַל הַתּוֹרָה וְעַל סִפְרוּת יִשְׂרָאֵל. וּבְכָל-זֹאת לֹא יַחְסְרוּ בָּהּ גַּם אַנְשֵׁי רוּחַ כַּבִּיר וַאֲנָשִׁים מַשְׂכִּילִים אֲשֶׁר תִּנְעַם חֶבְרָתָם.

מִמָּחֳרַת יוֹם הַשַּׁבָּת שָׁבָה הָעֵדָה לְהוֹדוֹת בִּתְפִלָּה חַמָּה לֵאלֹהִים. אַחֲרֵי-זֶה הִתְפַּלְלוּ עַל אַנְשֵׁי-חַסְדָּם אֲשֶׁר נְקָבוּם בְּשֵׁמוֹת. בָּרִאשׁוֹנָה זָכְרוּ לְטוֹבָה אֶת ר׳ צְבִי לֶעהרן וְאֶת ר׳ עֲקִיבָא אָחִיו, וְאַחֲרֵיהֶם אֶת הַשֶּׁרִיף הַנִּכְבָּד ר׳ מֹשֶׁה מוֹנְטִיפְיוֹרִי וְאֶת אִשְׁתּוֹ הַגְּבִירָה, אֶת ר׳ אָשֵׁר שִׁמְשׁוֹן וְעוֹד אֲחֵרִים.

וְאַחֲרֵי הַתְּפִלָּה לֹא הִסְפַּקְתִּי כִּמְעַט לִטְעֹם פַּת שַׁחֲרִית, וַתָּבוֹא כָּל הָעֵדָה לְבַקְּרֵנִי עַד גַּם הָרַב ר׳ יִשְׂרָאֵל לֹא נֶעְדַּר, וְהוּא אִישׁ אֲשֶׁר אֵינוֹ מְדַבֵּר בְּיוֹם הַשַּׁבָּת בִּלְתִּי-אִם בִּלְשׁוֹן-הַקֹּדֶשׁ לְבַדָּה. עַל כֵּן הָיְתָה לִי שִׂיחָתוֹ לְשַׁעֲשׁוּעַ וּלְתוֹרָה. הַמִּנְהָג בֵּינֵיהֶם לִשְׁלֹחַ מָנוֹת לְהָאוֹרֵחַ הַבָּא, וְאַחֲרֵי אֲשֶׁר הַמָּנוֹת הָאֵלֶּה הֵן עַל-פִּי הָרֹב בַּקְבּוּק יַיִן נִבְחָר, הָיָה יֵינִי רַב בַּיּוֹם הַהוּא, עַד כִּי לֹא קַמְנוּ מֵעַל הַשֻּׁלְחָן עַד עֵת תְּפִלַּת הַמִּנְחָה. אַחַד הַתְּפִלָּה שַׁבְתִּי וְהָזְמַנְתִּי אֶל ר׳ יִשְׂרָאֵל לַסְּעֻדָּה הַשְּׁלִישִׁית. הִרְבֵּינוּ לִדְרֹשׁ בְּדִבְרֵי תּוֹרָה. וְאַחַר נָתְנוּ הַמְּסֻבִּים בִּזִמִירוֹת קוֹלָם וַיַּמְשִׁיכוּ הַפִּזְמוֹנִים עַד שְׁקִיעַת הַשָּׁמֶשׁ.


מִכְתָּב חֲמִישִׁי

חֶבְרוֹן, חֲמִשָּׁה-עָשָׂר מְנַחֵם אָב תקצ״ח

הַדֶּרֶךְ אֲשֶׁר מִירוּשָׁלַיִם לְחֶבְרוֹן הוּא בְּחֶזְקַת סַכָּנָה רַבָּה. וַאֲנִי הָלַכְתִּי עָלָיו בְּלִי כָּל לְוָיָה, וּבְכָל-זֹאת, בָּרוּךְ הַשֵּׁם, לֹא קָרַנִי רָע. רָאִיתִי אֶת קֶבֶר רָחֵל בְּמָקוֹם אֲשֶׁר הַדֶּרֶךְ נוֹטֶה אֶל בֵּית-לֶחֶם וְהוּא בִּנְיָן מְרֻבָּע וְעָלָיו גַּג עָגֹל מְהֻדָּר. הַקֶּבֶר אֲשֶׁר בְּתוֹכוֹ מְסֻיָּד סִיד לָבָן, וְעָלָיו רִבְבוֹת אַלְפֵי שְׁמוֹת אוֹרְחִים כְּתוּבִים בִּכְתָב עִבְרִי. עָזַבְתִּי אֶת בֵּית-לֶחֶם לִשְׂמֹאלִי וָאֵלֵךְ בְּדֶרֶךְ יְשָׁרָה חֶבְרוֹנָה אֶל הַמָּקוֹם הַמְהֻלָּל לְפָנִים, אֲשֶׁר בּוֹ גָר אַבְרָהָם אֲבִי אֻמָּתֵנוּ, וַיְהִי מַהֲלָכִי הָרִאשׁוֹן אֶל הָרַב הַמַּגִּיד אֲשֶׁר הָיָה בְּיָדִי מִכְתָּב אֵלָיו מֵאֵת הָרַב ר׳ יִשְׂרָאֵל. וְהֵמָּה קִבְּלוּנִי בְּכָבוֹד גָּדוֹל בְּדִירָה נָאָה וּבְמַאֲכָלִים מֻבְחָרִים. רָאִיתִי אֶת הָעֵדָה הַזֹּאת, וַיִּהְיוּ בְּעֵינַי כִּבְחִירֵי יִשְׂרָאֵל. כֻּלָם מַכְנִיסֵי אוֹרְחִים אֲשֶׁר צָרוּ עֵינֵיהֶם לָתֵת לָאוֹרֵחַ לְהוֹצִיא אֵיזוֹ הוֹצָאָה מִכִּיסוֹ. וְרָאשֵׁי הָעֵדָה יִתְעַסְּקוּ לִפְעָמִים יָמִים תְּמִימִים עִם הָאוֹרֵחַ לְהָקֵל טֹרֵחַ הַדֶּרֶךְ מֵעָלָיו וּלְעַנֵּג אֶת יְשִׁיבָתוֹ בְּקִרְבָּם. בָּעֶרֶב הָרִאשׁוֹן בִּקְרַנִי אִישׁ יָשִׁישׁ ר׳ מֹשֶׁה מֵייזִיל מִוִּילְנָה, אִישׁ יוֹדֵעַ הַרְבֵּה אֲשֶׁר חִבֵּר כַּמָּה סְפָרִים קְטַנִּים. הָאִישׁ הַזֶּה הֻצַּג בִּימֵי נְעוּרָיו לִפְנֵי נַפּוֹלְיוֹן לַחֲרִיפוּת שִׂכְלוֹ, וְיָכֹל הָיָה לַעֲלוֹת מַעְלָה מָעְלָה, וְהִנֵּה הִתְעוֹרְרָה בּוֹ הַתְּשׁוּקָה לָלֶכֶת לְאֶרֶץ-יִשְׂרָאֵל, וַתִּתְפַּעֵל נַפְשִׁי הִתְפַּעֲלוּת אֵין גְּבוּל בְּדַבְּרִי אִתּוֹ עַל אֹדוֹת גְּדוֹלֵי מְשׁוֹרְרֵי גֶרְמַנְיָה, עַל אֹדוֹת שִׁילְלֶר, לֶסִינְג, וּבְיִחוּד עַל אֹדוֹת הֶרְדֶּר.

בַּיּוֹם הַשֵּׁנִי בִּקַּרְתִּי שׁוּב אֶת הַקְּבָרִים הַמְהֻלָּלִים, וּבָרִאשׁוֹנָה אֶת קֶבֶר אַבְנֵר בֶּן נֵר אֲשֶׁר הוּא בְּתוֹךְ הָעִיר בְּבֵית עַרְבִי, וְאַף כִּי אָסַר אִבְּרַהִים פֶּחָה לָקַחַת מַתָּת מֵאֵת הַבָּאִים שָׁמָּה לְבַקֵּר קַדְמוֹנִיּוֹת, בְּכָל-זֹאת דַּלְתוֹת הַקֶּבֶר נְעוּלוֹת, לֹא יָבוֹא כָּל יְהוּדִי אֵלָיו טֶרֶם יִתֵּן כֶּסֶף מְחִיר זֶה. יָרַדְנוּ מַדְרֵגָה אַחַת לְמַטָּה, וַנָּבוֹא אֶל חֶדֶר רְחַב יָדַיִם אֲשֶׁר הַקֶּבֶר בְּזָוִיתוֹ הַקִּיצוֹנִית, וְהוּא בְּתַבְנִית קֶבֶר רָחֵל, אֲבָל גָּבוֹהַּ מְעַט מִמֶּנּוּ, וְגַם הוּא מְכֻסֶּה בִּשְׁמוֹת אוֹרְחִים. פָּנִיתִי אֶל שְׂדֵה-קִבְרוֹת הַיְּהוּדִים וָאֶתְמַהּ מְאֹד עַל אֲשֶׁר לֹא מָצָאתִי שָׁם תַּבְנִית כָּל קֶבֶר כִּי-אִם אֲבָנִים שְׁלֵמוֹת צְבוּרוֹת זוֹ עַל יַד זוֹ אֲשֶׁר הֵמָּה לְפִי דִבְרֵי בֶּן-לִוְיָתִי אַבְנֵי הַמַּצֵּבוֹת.

וְאַף אֶת שֵׁם הַמֵּת אֵינָם נוֹהֲגִים לִכְתֹּב עַל מַצַּבְתּוֹ לְסִבָּה אֲשֶׁר לֹא יָדַע בֶּן-לִוְיָתִי, עַד כִּי לֹא נִמְצָא כָּל כְּתֹבֶת בְּחֶבְרוֹן. רָאִיתִי שָׁם מְעָרָה יָפָה בַּעֲלַת כּוּכִין רַבִּים אֲשֶׁר בְּאֶחָד מֵהֶם מְקוֹם מְנוּחַת עָתְנִיאֵל בֶּן קְנַז, וְעוֹד מְעָרָה אַחַת בַּעֲלַת כּוּכִין אֲשֶׁר לְפִי הַמָּסֹרֶת נִמְצָא בְּתוֹכָהּ קֶבֶר יִשַׁי אֲבִי דָוִד. אַחֲרֵי כֵן הָלַכְתִּי אֶל מְעָרַת הַמַּכְפֵּלָה הַמְהֻלָּלָה. לְפָנֶיהָ בֵּית-תְּפִלָּה גָדוֹל לְמֻסְלָמִים בַּקָּצֶה הָאַחֲרוֹן שֶׁל הַשָּׂדֶה. בְּמַדְרֵגוֹת רַבּוֹת עוֹלִים אָל פִּתְחוֹ. וְעַל הַמַּדְרֵגָה הָרִאשׁוֹנָה מִשְּׂמֹאל לַכְּנִיסָה הָרִאשׁוֹנָה נִמְצָא קֶבֶר הָרמבּ״ן35. לְשׁוּם אֵרוֹפִּי לֹא יִתְּנוּ לָבֹא אֶל בֵּית-הַתְּפִלָּה בְּעַד כָּל מְחִיר, בִּלְתִּי אִם עַד הַפֶּתַח הַשֵּׁנִי לְבַדּוֹ. שָׁם נִמְצָא חוֹר כְּגֹדֶל כַּף אִישׁ, וְדַרְכּוֹ שָׁב וְיָצָא חָתוּל אֶחָד אֲשֶׁר הִשְׁלִיכוּ לְתוֹךְ קֶבֶר רוּת (אִם אֵינֶנִּי שׁוֹגֶה), הַנִּמְצָא מִחוּץ לָעִיר בַּמֶּרְחָק. לְפִי זֶה לֹא שָׁוְא הָיְתָה הַהַשְׁעָרָה כִּי גְדוֹלָה הַמְּעָרָה הַהִיא מְאֹד. הַיְּהוּדִים כְּמוֹ גַם הַמֻּסְלָמִים יְסַפְּרוּ, כִּי לִפְעָמִים יִתְגַּלֶּה אַבְרָהָם לְהַצִּילָם מִצָּרָה הַמִּתְרַגֶּשֶׁת לָבוֹא עֲלֵיהֶם. וְהִנְנִי לִמְסֹר אַחַד הַסִּפּוּרִים הַיּוֹתֵר חֲשׁוּבִים, אֲשֶׁר מָסַר לִי אַחַד הָאֲנָשִׁים הַיּוֹתֵר נֶאֱמָנִים, אַחֲרֵי אֲשֶׁר הֱבִיאַנִי אֶל פֶּתַח אֶחָד וּבוֹ חוֹר עָגֹל אֲשֶׁר תִּכָּנֵס וְתֵצֵא בּוֹ יַד אָדָם בִּרְוָחָה, פָּתַח הַמְסַפֵּר וְאָמַר: ״לִפְנֵי שָׁנִים רַבּוֹת קָרָא הַשֵּׁיךְ אֶת רֹאשׁ הָעֵדָה הַיְּהוּדִית וַיִּדְרֹשׁ מִמֶּנּוּ סַךְ כֶּסֶף מְסֻיָּם, וְעוֹד יוֹתֵר מִזֶּה הִטִּיל עָלָיו לָתֵת לוֹ אֵת כָּל הַסַּךְ הַהוּא בְּמִין מַטְבֵּעַ יְדוּעָה אֲשֶׁר לֹא נִמְצְאָה אִתָּם בְּשׁוּם פָּנִים, כִּי אַךְ תּוֹאֲנָה בִּקֵּשׁ הַשֵּׁיךְ בְּחֶפְצוֹ לְהַשְׁמִיד אֵת כָּל הַיְּהוּדִים, וְהִנֵּה אִישׁ בִּלְתִּי יָדוּעַ לָהֶם הֵבִיא אֶת סְכוּם הַכֶּסֶף בְּמִין הַמַּטְבֵּעַ אֲשֶׁר הַשֵּׁיךְ מְבַקֵּשׁ וַיּוֹשֶׁט אוֹתוֹ לָהֶם דֶּרֶךְ הַחוֹר הַהוּא״. וְסִפּוּרִים כָּאֵלֶּה מְצוּיִם אֶצְלָם לַאֲלָפִים. עֲדַת הַיְּהוּדִים בְּחֶבְרוֹן קְטַנָּה, וְרַק שְׁנֵי בָּתֵּי-כְּנֵסִיּוֹת בָּהּ, הָאֶחָד לָאַשְׁכְּנַזִּים וְהָאֶחָד לַסְּפַרְדִים, הֵם חַיִּים בְּאַחֲוָה וְרֵעוּת וְכִמְעַט כֻּלָּם יָגוּרוּ בְּחָצֵר אֶחָת. הַדִּירוֹת קְטַנּוֹת מְאֹד אֲבָל נְקִיּוֹת, מְרֻוָּחוֹת וְנָאוֹת. אַף כִּי מֻתָּר לָהֶם לָגוּר גַּם בִּרְחוֹבוֹת אֲחֵרִים, בְּכָל-זֹאת לֹא רַבִּים יִשְׁתַּמְּשׁוּ בַּהֶתֵּר הַזֶּה.

כַּאֲשֶׁר כָּבַשׁ אִבְּרַהִים פֶּחָה אֶת סוּרִיָּה, נָתַן רְשׁוּת לְאַנְשֵׁי-צְבָאוֹ לִמְשֹׁל וְלָבֹז אֶת הָעִיר. וַיְהִי הַמְּעַט אֲשֶׁר עָשָׂה הַצָּבָא רָעָה רַבָּה לַיְּהוּדִים, כִּי הוֹסִיפוּ עוֹד הָעַרְבִים לְהָעִיד עֵדוּת שֶׁקֶר, כִּי בְּיַד הַיְּהוּדִים הִפְקִידוּ הַמֻּסְלָמִים אֲשֶׁר נִמְלְטוּ מִן הָעִיר אֶת אוֹצְרוֹתֵיהֶם. לַשָּׁוְא הוֹכִיחוּ לִפְנֵי אִבְּרַהִים כִּי הַיְּהוּדִים הָאֵלֶּה כֻּלָּם בְּנֵי אֵרוֹפָּה הֵם אֲשֶׁר כָּל דָּבָר אֵין לָהֶם בְּעֶסְקֵי הַמְּדִינָה וּבַמְּרִידָה אֲשֶׁר מָרְדוּ אֶזְרְחֵי הָאָרֶץ בּוֹ, כִּי הֵשִׁיב תְּשׁוּבָה נִצַּחַת לֵאמֹר, כִּי דָבָר אֲשֶׁר יָצָא מִפִּיו לֹא יוּכַל לְהָשִׁיב אָחוֹר, וְאַנְשֵׁי-צְבָאוֹ יוּכְלוּ לְהַמְשִׁיךְ אֶת מַעֲשֵׂיהֶם כְּכֹל הָעוֹלֶה עַל רוּחָם. וְהַיְּהוּדִים הֵמָּה מְלֻמָּדִים בְּמַכּוֹת כָּאֵלֶּה, עַד כִּי אַחֲרֵי כָּל אֵלֶּה מְעַטִּים מֵהֶם יִהְיֶה אֶת לִבָּם לַעֲזֹב אֶת הָאָרֶץ אִם גַּם תִּהְיֶה הָאֶפְשָׁרוּת בְּיָדָם לְעָזְבָהּ, וְרֻבָּם יִבְחֲרוּ יוֹתֵר לְסַכֵּן אֶת נַפְשָׁם בְּאֶלֶף סַכָּנוֹת יוֹם יוֹם מֵעֲזֹב אֶת הָאָרֶץ הַקְּדושָׁה.

שֵׁיךְ הָעִיר הָלַךְ בָּעֵת הַהִיא לִירוּשָׁלַיִם לְהִוָּדַע יַד מִי עַל הָעֶלְיוֹנָה, אִם יַד אִבְּרַהִים פֶּחָה אוֹ יַד הַדְּרוּזִים. וּבְדַעְתּוֹ הָיָה לַעֲשׂוֹת הוּא וְכָל יוֹשְׁבֵי חֶבְרוֹן יָד אַחַת עִם הַמּוֹרְדִים יוֹשְׁבֵי יְרוּשָׁלַיִם אִם תְּהִי יָדָם תַּקִיפָה. וְאָז הָיָה חַס-וְשָׁלוֹם כִּלָּיוֹן גָּמוּר נִשְׁקָף לַיְּהוּדִים. אֲבָל אִבְּרַהִים פֶּחָה יוֹדֵעַ אֶת לְבַב אֲנָשָׁיו, וַיְפַרְסֵם אֶת נִצְחוֹנוֹ מֵרָאשֵׁי כָּל הַחוֹמוֹת בִּתְרוּעָה רַבָּה, עַד כִּי נָפַל פַּחְדּוֹ עַל הַתֻּרְכִּים וְעַל הָעַרְבִים.

הַיּוֹם עָשׂוּ כָּל בְּנֵי הָעִיר לִכְבוֹדִי מִשְׁתֶּה גָדוֹל בְּחַדְרִי אֲשֶׁר אֲנִי גָר בְּתוֹכוֹ. אַחֲרֵי אֲשֶׁר פִּקְפְּקוּ לְהַטְרִיחֵנִי לָבוֹא אֶל בֵּית אֶחָד מֵהֶם.

מֵעֵת אֲשֶר הֵחֵלּוּ יְמֵי הָרֹגֶז בִּצְפָת גָּדַל מִסְפַּר עֲדַת חֶבְרוֹן בְּמָאתַיִם נָפֶשׁ. לְעֵת תְּפִלַּת הַמִּנְחָה הֶרְאוּנִי מְנוֹרַת כֶּסֶף יָפָה תְּלוּיָה לִפְנֵי אֲרוֹן הַקֹּדֶשׁ. קָרָאתִי אֶת הַכְּתֹבֶת אֲשֶׁר עָלֶיהָ וְהִנֵּה הִיא נִדְבַת ר׳ אָשֵׁר שִׁמְשׁוֹן מִלָּנְדּוֹן אֲשֶׁר הִרְבָּה לִנְטוֹת חֶסֶד לְיוֹשְׁבֵי אֶרֶץ-יִשְׂרָאֵל מֵעוֹדוֹ, וְעַד הַיּוֹם, סִפֵּר לִי תַּלְמִיד חָכָם נִכְבָּד בִּצְפָת, מַחֲזִיק הוּא בְּיָדוֹ וּבְיַד רַבִּים אֲחֵרִים בִּתְמִיכָה הֲגוּנָה.

אָמְנָם עַל שָׁוְא נוֹסְדוּ מַחְשְׁבוֹת בְּנֵי אֵרוֹפָּה אֲשֶׁר יַחְשְׁבוּ כִּי יוֹשְׁבֵי יְרוּשָׁלַיִם, צְפָת וְחֶבְרוֹן וְכָל אֶרֶץ-יִשְׂרָאֵל כֻּלָּם בַּטְלָנִים הֵם, אֲשֶׁר אֵין לָהֶם כָּל רְכוּשׁ וְאֻמָּנוּת, וַאֲשֶׁר רַק לִחְיוֹת חַיֵּי מְנוּחָה וּבַטָּלָה הוֹלְכִים הֵמָּה שָׁמָּה. בֶּאֱמֶת רַבִּים הֵבִיאוּ רְכוּשׁ רַב אִתָּם, וְרַבִּים הִתְפַּרְנְסוּ בְּאַרְצָם בְּשֶׁפַע וּבְכָבוֹד. אֲבָל הָרֶגֶשׁ הַדָּתִי הֱבִיאָם לָאָרֶץ הַקְּדוֹשָׁה. וְהָיָה לָנוּ ר׳ מָרְדְּכַי שְׁנִיצֶר36 בִּירוּשָׁלַיִם לְמָשָׁל אֶחָד מִנִּי עֲשָׂרָה. הוּא הָיָה לְפָנִים פְּאֵר אֲקַדֶּמְיָה אַחַת בְּוִילְנָה. וּפִסּוּלָיו הָיוּ יְכֹלִים לְכַלְכֵּל אוֹתוֹ גַם הַיּוֹם בְּאֶרֶץ גֶּרְמַנְיָה, אַךְ הוּא עָזַב אֶת כְּבוֹדוֹ וְאֶת שֶׁפַע פַּרְנָסָתוֹ לֶאֱכֹל לֶחֶם עֶצֶב בְּזֵעַת אַפַּיִם בִּירוּשָׁלַיִם. וְעוֹד זוּלָתוֹ נִמְצְאוּ כָּאן כַּמָּה חָרָשֵׁי אֶבֶן מְהִירִים בִּמְלַאכְתָּם אֲשֶׁר גַּם חָכְמַת אֶרֶץ גֶּרְמַנְיָה וְסִפְרוּתָהּ לֹא זָרִים הֵמָּה לָהֶם. אֲבָל רוּחַ הָאֱמוּנָה חָזְקָה עֲלֵיהֶם וַתּוֹלִיכֵם אֶל הָאָרֶץ הַקְּדוֹשָׁה.

בִּקַּרְתִּי אַחֲרֵי יָמִים אֲחָדִים אֶת הַמָּקוֹם הַנִּקְרָא חַלְחוּל אֲשֶׁר בּוֹ קִבְרֵי הַנְּבִיאִים נָתָן וְגָד. לֹא רָחוֹק מִמֶּנּוּ בִּנְיָן יָפֶה מְאֹד בָּנוּי מֵאֲבָנִים גְּדוֹלוֹת וַאֲרֻכּוֹת וּבְאֶמְצָעוֹ בְּאֵר עָגֹל וְיָפֶה, וְקָרוֹב אֵלָיו בַּיִת גָּדוֹל כָּמוֹהוּ בָּנוּי מֵאֲבָנִים קְטַנּוֹת מֵאַבְנֵי הַבִּנְיָן הָרִאשׁוֹן, וְהַבָּתִּים הָאֵלֶּה נִקְרָאִים אָהֳלֵי אַבְרָהָם וְשָׂרָה. סִגְנוֹן בִּנְיְנֵיהֶם כְּסִגְנוֹן הַבִּנְיָנִים הַגְּדוֹלִים אֲשֶׁר רָאִיתִי בְּמִירוֹן וּבִמְקוֹמוֹת אֲחֵרִים וַאֲשֶׁר הָעַרְבִים יִקְרְאוּ לָהֶם בִּנְיְנֵי הַיְּהוּדִים.

גָּמַרְתִּי עַתָּה אֶת תֵּאוּרֵי חֶבְרוֹן וּמָחָר אָשׁוּב יְרוּשָׁלָיְמָה. מִשָּׁם אֶסַּע לְיָפוֹ, לְעַכּוֹ וּלְבַיְרוּת, וְאַחַר אֵלֵךְ לְדַמֶּשֶׂק וַאֲסַפֵּר לְאִבְּרַהִים פֶּחָה אֶת תְּלָאוֹת דַּרְכִּי. כִּי הַדְּרוּזִים אֲשֶׁר בָּזְזוּ אֶת צְפָת נָפְלוּ שְׁבִי בְּיָדוֹ, וּמַנְהִיגָם יָדוּעַ. כָּל כֶּסֶף הַבִּזָּה הוּא כָּעֵת בִּידֵי הַפֶּחָה. וְהִנְנִי לְנַסּוֹת אוּלַי תַּצְלִיחַ בְּיָדִי לַעֲשׂוֹת דָּבָר לְטוֹבַת אַחַי הָעֲנִיִּים, אַף כִּי לֹא הֵשִׁיב הַפֶּחָה עַד הַיּוֹם אֶת הַגָּזֵל אֲשֶׁר גָּזְלוּ עֲבָדָיו לִפְנֵי אַרְבַּע שָׁנִים אֶת יוֹשְׁבֵי צְפָת.

  13. ר‘ אַבְרָהָם דֹּב מֵאוּרִיטְשׁ הָיָה מַנְהִיג “הַחֲסִידִים” בִּצְפָת. הוּא הָיָה רַב בְּזִ’יטוֹמִיר וְתַלְמִידוֹ הַגָּדוֹל שֶׁל רַבִּי מָרְדְּכַי מִטְּשַארנָאבָּאל. חִבֵּר סֵפֶר “בַּת–עַיִן”, וְנִפְטַר בִּצְפָת בְּי“ב כִּסְלֵו שְׁנַת תר”א (תּוֹלְדוֹת חַכְמֵי–יְרוּשָלֵם, פְרוּמְקִין–רִיבְלִין ח“ג פ”ה, י"א) (הַמְּסַדֵּר).  ↩

  14. הֶחָכָם שׁוּשָן הַזֶּה סָמַךְ כַּנִּרְאֶה עַל נְתִינָיו בִּשְׁעַת צָרָה, כִּי הוּא הִנֵּהוּ כִּמְעַט בַּעַל נִכְסֵי הַכְּפָר דַּלַּתָּא כֻּלּוֹ וְכֻלָּם אֲהֵבוּהוּ. כִּי רַק הַיְהוּדִים הַסְּפַרְדִים נֶחְשְׁבוּ לְאֶזְרְחֵי הָאָרֶץ אֲשֶׁר רִשְׁיוֹן לָהֶם לִקְנוֹת קַרְקָעוֹת.  ↩

  15. הַיְנוּ לְפִי דִּבְרֵי הַתַּלְמוּד יֶחֱרַב הַגָּלִיל לְגַמְרֵי לִפְנֵי בּוֹא הַמָּשִׁיחַ (הַמְּסַדֵּר).  ↩

  16. אַלְרָאמָה לַכְּפָר הַזֶּה עֲצֵי זַיִת מְרֻבִּים מְאֹד וְלָכֵן יֵשׁ בּוֹ מִסְפָּר רַב שֶׁל תּוֹשָׁבִים.  ↩

  17. לְפִי הָאֱמֶת אֵין הֶעָלִים הָאֵלֶּה כִּי אִם חֵלֶק מִן הָעֵץ וְרַק הָעֳקָצִים הֵם עָלָיו (הַמְסַדֵּר).  ↩

  18. בְּעִיר הַקֹּדֶשׁ יְרוּשָׁלַיִם תִּבָּנֶה ְתִכּוֹנֵן בִּמְהֵרָה בְיָמֵינוּ (הַמְסַדֵּר).  ↩

  19. לִפְרָט קָטֹן (הַמְסַדֵּר).  ↩

  20. הוּא ר׳ יִשְׂרָאֵל מִשְּׁקְלוֹב תַּלְמִיד הגר״א (הַגָּאוֹן ר׳ אֵלִיָּהוּ מִוִּילְנָא) מְחַבֵּר סֵפֶר ״פְּאַת–הַשֻּׁלְחָן״ (הִלְכוֹת א״י), רֹאשׁ הָעוֹלִים ״הַפְּרוּשִׁים״. עָלָה לְא״י בִּשְׁנַת תק״ע וְנִפְטַר בִּטְבֶרְיָה ט׳ סִיוָן שְׁנַת תקצ״ט (תּוֹלְדוֹת חַכְמֵי–יְרוּשָׁלֵם פְרוּמְקִין–רִיבְלִין ח״ג פ״ה, סִמָּן ג׳).  ↩

  21. חֲצַר ר׳ יִשְׂרָאֵל נִקְרָא כַּיּוֹם ״חֲצַר ר׳ יְשַׁעְיָה״ עַל שֵׁם ר׳ יְשַׁעְיָה בַּרְדָּקִי חֲתָנוֹ וּמְמַלֵּא מְקוֹמוֹ שֶׁל ר׳ יִשְׂרָאֵל, וּבוֹ בֵּית–הַכְּנֶסֶת הַנִּקְרָא ״סֻכַּת–שָׁלוֹם״ (הַמְסַדֵּר).  ↩

  22. בֵּית–הַתְּפִלָּה הוּא מִסְגַּד עֹמָר וּ״מִדְרַשׁ שְׁלֹמֹה״ הוּא מִסְגַּד אַלְאַקְצָא (הַמְסַדֵּר).  ↩

  23. הִיא מְעָרַת קִבְרֵי מַלְכָּה הֵלֶנָה וּמִשְׁפַּחְתָּהּ (הַמְסַדֵּר).  ↩

  24. מְעָרַת שִׁמְעוֹן הַצַּדִיק מִשְּׁיָרֵי אַנְשֵׁי כְּנֶסֶת הַגְּדוֹלָה; שָׁם יִתְאַסֵּף הָמוֹן רַב שֶׁל יְהוּדִים בְּלֹ״ג בָּעֹמֶר כְּדֶרֶךְ שֶׁמִתְאַסְּפִים בְּמִירוֹן לְיַד קֶבֶר ר' שִׁמְעוֹן בֶּן יוֹחַאי (הַמְסַדֵּר).  ↩

  25. שֵׁם הַמָּקוֹם אֲשֶׁר יָצָא בּוֹ עֵין גִּיחוֹן לְפָנִים הוּא מֵי הַשִּׁילוֹחַ, וְהָעַרְבִים יִקְרְאוּ לַמָּקוֹם בִּיר אַיוּבּ (הַמְסַדֵּר).  ↩

  26. דָּבָר זֶה אֵינוֹ מְדֻיָּק, כִּי הַכֹּהֵןהַגָּדוֹל לֹא הָיָה בָּא אֶלָּא פַּעַם אַחַת בְּשָׁנָה רַק בְּקֹדֶשׁ–הַקֳּדָשִׁים, וְאוּלָם עַל–פִּי דִין יִשְׂרָאֵל אָסוּר לְהִכָּנֵס לְהַר–הַבַּיִת כֻּלּוֹ כָּל עוֹד לֹא נִטַּהֲרוּ עַל–יְדֵי מֵי אֵפֶר פָּרָה אֲדֻמָּה שֶׁאֵינָם בַּזְּמָן הַזֶּה (הַמְסַדֵּר).  ↩

  27. כְּבָר רַבּוּ הַחוֹקְרִים לִסְתֹּר מָסֹרֶת זוֹ, כִּי מֵעוֹלָם לֹא הִקִּיפָה חוֹמַת יְרוּשָׁלַיִם גַּם אֶת הַר הַזֵּיתִים (הַמְסַדֵּר).  ↩

  28. לְפִי אַגָּדָה יְרוּשַׁלְמִית נִקְטְעָה בִּפְקֻדַּת נַפּוֹלְיוֹן הָרִאשׁוֹן שֶׁאָמַר: יָד שֶׁהוּרְמָה נֶגֶד אָבִיהָ תִּקָּטַע. וְאוּלָם נַפּוֹלְיוֹן לֹא הָיָה מֵעוֹלָם בִּירוּשָׁלָיִם (הַמְסַדֵּר).  ↩

  29. הִיא הַנִּקְרֵאת מִקְוַת ר׳ יִשְׁמָעֵאל כֹּהֵן גָּדוֹל, אֶחָד מֵעֲשָׂרָה הֲרוּגֵי מַלְכוּת (הַמְסַדֵּר).  ↩

  30. זִיאָרָה פֵּרוּשָׁהּ הִשְׁתַּטְּחוּת עַל קְבָרִים וּבִקּוּר בִּמְקוֹמוֹת קְדוֹשִׁים (הַמְסַדֵּר).  ↩

  31. הֵם הַנִּקְרָאִים בְּשֵׁם קִבְרֵי מַלְכוּת בֵּית דָּוִד וְהֵם בְּצִיּוֹן (הַמְסַדֵּר).  ↩

  32. הוּא בֵּית–הַכְּנֶסֶת שֶׁל ר׳ יוֹחָנָן בֶּן זַכַּאי, וְשָׁם בֵּית–כְּנֶסֶת סְטַמְבּוּלִי וּבֵית–כְּנֶסֶת תַּלְמוּד–תּוֹרָה וְעוֹד בָּתֵּי–תְּפִלָּה שׁוֹנִים סְמוּכִים זֶה לָזֶה (הַמְסַדֵּר).  ↩

  33. ר׳ צְבִי לֶעהרן רֹאשׁ פְּקִידֵי וַאֲמַרְכְּלֵי ארה״ק בְּאַמְשְׂטֶרְדַּם, שֶׁפָּעַל הַרְבֵּה לְיִשּׁוּב יְרוּשָׁלַיִם (הַמְסַדֵּר).  ↩

  34. הוּא בֵּית–יְשִׁיבָה לִמְקֻבָּלִים הַיָּדוּעַ בְּשֵׁם ״בֵּית–אֵל״ (הַמְסַדֵּר).  ↩

  35. לְפִי הָאַגָּדָה נִקְבַּר הָרמב״ן בְּחֶבְרוֹן. מְקוֹרָהּ הוּא, שֶׁהָרמב״ן קָנָה לוֹ קֶבֶר בְּחֶבְרוֹן בְּחַיָּיו וְאוּלָם הוּא נִפְטַר וְנִקְבַּר בְּעַכּוֹ (הַמְסַדֵּר).  ↩

  36. הוּא ר׳ מָרְדְּכַי פִּתֵּחַ שְׁנִיצֶר אַחַדעֲשֶׂרֶת הָרִאשׁוֹנִים שֶׁיִּסְּדוּ אֶת הַיִּשּׁוּב הָאַשְׁכְּנַזִּי בִּירוּשָׁלַיִם בִּשְׁנַת תק״ע (תּוֹלְדוֹת חַכְמֵי יְרוּשָׁלַיִם פְרוּמְקִין–רִיבְלִין, ח׳ ג׳, 260) (הַמְסַדֵּר).  ↩
