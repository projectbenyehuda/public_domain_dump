


	
	[אֶזְעַק אֵלֶיךָ מַלְכִּי וּקְרוֹבִי (סְלִיחָה)]
	



[נוֹלד ברוֹמָא לערך 1030. אחרי שנת 1070 הלך לווֹרְמַיְשָׂא וכנראה נהרג בימי הפורענות של 1096]





סליחה




אֶזְעַק אֵלֶיךָ מַלְכִּי וּקְרוֹבִי



אֹדוֹת מַכָּתִי אֲנוּשָׁה בְחוֹבִי



אָנָא יְיָ חַתֵּל מַכְאוֹבִי



כִּי-חִצֶּיךָ נִחֲתוּ-בִי.



כִּי נוֹחֲלָה נַפְשִׁי וְנִבְהָלָה



בִּפְקֻדּוֹת סְמוּכוֹת בָּאוֹת לְחַלְּלָהּ



בְּגֹדֶל חַסְדְּךָ מְרֹמָם וְנַעֲלָה



קָרְבָה אֶל-נַפְשִׁי גְאָלָהּ.



גְּאָלָהּ מִמְרֵעִים עָלֶיהָ חָבָרוּ



גְּבֹהָה וְעָתָק יַבִּיעוּ יְדַבֵּרוּ



גַּאֲוָה עֲנָקַתְמוֹ עָצְמוּ וְגָבָרוּ



חֶלְבָּמוֹ סָגְרוּ פִּימוֹ דִּבֵּרוּ.



דִּבְּרוּ בְגֵאוּת אוֹתִי לְשָׁרֵשׁ



דִּמּוּ לִמְחוֹת שְׁמִי בְּהִתְפָּרֵשׁ



דּוֹדִי לָמָּה תַבִּיט וְתַחֲרֵשׁ



הֲכֹל הַיּוֹם יַחֲרשׁ הַחֹרֵש.



הַחֹרֵשׁ אָוֶן וְזוֹרֵעַ עַוְלָה



הוֹמֶה וּמִתְאַנֵּה עָלַי לְהִתְאוֹנֲנָה



הָרֹאוֹת חָשְׁכוּ מִבְּכִי וִילָלָה



דִּמְעָתִי לֶחֶם יוֹמָם וָלָיְלָה.



וָלַיְלָה יָדִי נִגְּרָה בִנְשִׁילוּת



וָאֱהִי נָגוּעַ בַּיּוֹם מִסִּבְלוּת



וּלְבַלְּעִי חִנָּם בְּלִי עַצְלוּת



רְשָׁעִים יִתְהַלָּכוּן כְּרֻם זֻלֻּת.



זֻלֻּת גְּדוּדֵי זְנוּת וְתָרְפָּה



זָלֲלוּ יְקָרִי וְחִבְּקוּנִי אַשְׁפָּה



זֶה אֵלִי מַכָּתִי תִרְפָּא



כִּי עָלֶיךָ נָשָׂאתִי חֶרְפָּה.



חֶרְפָּה שָׁבְרָה לִבִּי וָאֶשָּׂרֵף



חֲזוֹתִי שַׁלְוַת שׁוֹחֲחֵי תָרֶף



חַלְתִּי מִקּוֹל מְגַדֵּף וּמְחָרֵף



כַּאֲרִי שׁוֹאֵג טֹרֵף טָרֶף.



טֶרֶף נֻתַּתִּי לְשִנֵּי מַחֲרִיבָי



טוּבִי וְחֶלְבִּי שַׁמְנִי וְשִׁקֻּוַי



טָמוּן חֲבשׁ פְּנֵי מַחֲרִיבָי



רִיבָה יְיָ אֶת יְרִיבַי.



יְרִיבַי יַכְבִּידוּ עֹל לְהַטְרִיחִי



יָעֲצוּ סוֹד מִיִּרְאָתְךָ לְהַדִּיחִי



יְחִידָתִי נִכְסְפָה וְכָלְתָה רוּחִי



לִבִּי סְחַרְחַר עֲזָבַנִי כֹחִי.



כֹּחִי אָכְלוּ מְתֵי רֶצַח



כִּי-רְמָסוּנִי הֲמָמוּנִי בְּעַזּוּת מֶצַח



כְּבוֹדִי תָרִים אָדוֹם וְצַח



הָקִיצָה אַל-תִּזְנַח לָנֶצַח.



לָנֶצַח עֻזְּךָ בִּשְׁמֵי אֲוֵרָי



לֶכֶת בְּחֻקֶּיךָ תְּמֹךְ אֲשׁוּרָי



לָעַד הֱיֵּה לִי בְעוֹזְרָי



יְיָ אֱלֹהִים מִבְטַחִי מִנְּעוּרָי.



מִנְּעוּרַי סִגַּלְתַּנִי לְךָ לְמָנָה



מִלְּפָנֶיךָ אַל-תַּשְׁלִיכֵנִי לְעֵת זִקְנָה



מִקְדָּשׁ מְעַט מוּלְךָ כְּאֶתְחַנְּנָה



נִדְבוֹת-פִּי רְצֵה-נָא.



נָא אַדִּיר נוֹרָא עֲלִילָה



נַטְּלֵנִי נַשְּׂאֵנִי בְּאַהַב וְגִילָה



נֶצַח סוֹבֲבֵנִי בְאֶבְרָתְךָ בְּחֶמְלָה



רֳנֵּי פַלֵּט תְּסוֹבֲבֵנִי סֶלָה.



סֶלָה עֲנִיתָנִי מִקַּרְנֵי רֵמִים



סוֹבַבְתָּ וּמָחַצְתָּ מְלָכִים רָמִים



סַלֵּה חַיַּת קָנֶה מִתְּחוּמִים



עֲדַת אַבִּירִים בְּעֶגְלֵי עַמִּים.



עַמִּים בָּזוּ מָאֲסוּ פְנִינֶיךָ



עָצוּ וְהֶעֱרִימוּ סוֹד צְפוּנֶיךָ



עָלַי לְטוֹבָה שִׂימָה עֵינֶיךָ



יְמִינְךָ וּזְרוֹעֲךָ וְאוֹר פָּנֶיךָ.



פָּנֶיךָ הָאֵר לְאֹם אִוִּיתָ



פּוֹעֵל לָשֶׁבֶת מָכוֹן שִׁוִּיתָ



פְּקֹד בְרַחֲמִים תִּפְאַרְתְּךָ כְּחִוִּיתָ



וְעוּרָה אֵלַי מִשְׁפָּט צִוִּיתָ.



צִוִּיתָ צֶדֶק לְיַעֲקֹב קְרוֹבֶךָ



צִיץ וְנֵזֶר שַׁעְשׁוּעַ רוֹבֶיךָ



צְלַח רְכַב לְיֶשַׁע כְּרוּבֶיךָ



אַל-יֵבשׁוּ בִי קוֶֹיךָ.



קוֶֹיךָ יָגְעוּ רָצוּ וַיִּעָפוּ



קָדְרוּ בְלַחַץ אוֹיְבִים וְזֻלְעָפוּ



קָמֵיהֶם קַלַּע וַהֲדֹף וְיֻסְחָפוּ



כִּי-אַתָּה אֲשֶׁר-הִכִּיתָ רָדָפוּ.



רָדְפוּ הֲנִיעֵם כַּשִּׁכּוֹר בְּזַעְמֶךָ



רְדֹף וְהַשְׁמִידֵם מִתַּחַת שָׁמֶיךָ



רַעַל הַשְׁקֵם וְתַכְחִידֵם מִמְּרוֹמֶיךָ



וְיַעְלְצוּ בְךָ אֹהֲבֵי שְׁמֶךָ.



שִׁמְךָ נִקְרָא צַּדִיק וּמוֹשִׁיעַ



שַׁוְעַת הַכֹּל מְקַבֵּל וְשִׁיעַ



שִׁבְתְּךָ בַּדִּין רָמִים תַּרְשִׁיעַ



וְאֶת-עַם עָנִי תּוֹשִׁיעַ.



תּוֹשִׁיעַ אֶבְיוֹן מֵרִגְשַׁת בּוֹלָשֶׁת



תָּגֵן תַּצִּיל אֵלָיו מִגָּשֶׁת



רְאֵה כִּי אָזְלָה מְמַשְׁמָשֶׁת



כִּי הִנֵּה הָרְשָׁעִים יִדְרְכוּן קָשֶׁת.



קֶשֶׁת נְחוּשָׁה תְשַׁבֵּר לְהַקְדִּישִׁי



לְכָה לִיְשׁוּעָתִי מַלְכִּי וּקְדוֹשִׁי



וְאַל תֵּפֶן אֶל קשִׁי



אָנָּא יְיָ מַלְּטָה נַפְשִׁי.



נַפְשִׁי תְבָרֶכְךָ כִּי תְעוֹדֲדֵנִי



יְשַׁבְּחוּנְךָ שְׂפָתַי עֵת תִּפְדֵּנִי



מִמִּזְרָח וּמִמַּעֲרָב יִירָאוּךָ אֲדֹנִי



כִּי-עָשִׂיתָ מִשְׁפָּטִי וְדִינִי.



וְדִינִי לְךָ לַעֲשׂוֹת וְלָדָעַת



סִתְרִי אַתָּה זֹאת מוּדָעַת



בַּעַל הַגְּבוּרָה וְהַגְּדֻלָּה וְהַדָּעַת



נִפְלָאִים מַעֲשֶׂיךָ וְנַפְשִׁי יֹדָעַת.



יוֹדַעַת נוֹרָאוֹת עָשִׂיתָ לְקַדְמוֹנִי



רָאֹה תַרְאֵנִי נִקְמַת מוֹנִי



בְּכֵן בִּקְהַל שׁוֹמֵר אֱמוּנָי



תָּמִיד יִגְדַּל יְיָ.



יְיָ קְרָאתַנִי בֵן בְּכוֹר



שׁוּב בְּרִיתַנִי בְּלֶתֶךְ זְכֹר



בֶּעָוֹן אוֹתִי אַל תִּמְכֹּר



בְּרֹגֶז רֶחֶם תִּזְכֹּר.



תִּזְכֹּר קִיּוּם אֵיתַנֵי קֶדֶם



ישֶׁר בְּרִית וְחֶסֶד מִקֶּדֶם



חֹב תִּמְחֹק וְתַלְבִּין אֹדֶם



זְכֹר עֲדָתְךָ קָנִיתָ קֶּדֶם.



קֶדֶם יוֹשֵׁב בְּעֹז חֶבְיוֹן



בְּתוֹר הַמַּעֲלָה אֲחָזַנִי בְצִבְיוֹן



תִּשְׁבִּי תְּנַחֵץ לְבַשְׂרִי בְרִצְיוֹן



כִּי-בָנָה יְיָ צִיּוֹן.



צִיּוֹן וִירוּשָׁלִַםִ תִּרְאֶינָה עֵינָי



אֹמֶץ קְדֻשָּׁתְךָ לְהַעֲרִיץ בְּשִׁנּוּנִי



חֲזוֹתְךָ עַיִן בְעַיִן כְּלִימְדּוֹנָי



אָמֵן כֵּן יֹאמַר יְיָ.
