


	
	חֲגִיגַת הַקַּרְנֵוָל אֵצֶל הַנּוֹצְרִים (מִתּוֹךְ הַסֵּפֶר “פִּלְפּוּל זְמָן זְמַנִּים וּזְמַנֵּיהֶם”)
	



Jona Coen Rappa

[חי בקָזָלַי מוֹנְפֵרָטוֹ לערך 1680]






וּבַיָּמִים הָהֵם וּבָעֵת הַהִיא / לֹא נִשְׁמַע קוֹל וַי
אוֹי הֹוָה וּנְהִי / וְנָס יָגוֹן וְהִי / וְאֵין דּוֹרֵשׁ / וְאֵין
מְבַקֵּשׁ / כִּי אִם־רִבּוּי וְשִׁנּוּי יֵינוֹת וּמַאֲכָלִים / וְאֵין
קוֹל וְאֵין עֹנֶה כִּי אִם מִנִּים וְעוּגָב תֻּפִּים וְכִנּוֹרוֹת
וּנְבָלִים / וְקָל קַרְנָא מַשְׁרוֹקִיתָא קַיתְרֹס וַחֲלִלִים /
וְשַׂבְּכָא וּפְסַנְתֵּרִין וְכֹל זְנֵי זְמָרָא נְבוּבִין וַחֲלוּלִים /
וְשָׁרִים כְּחֹלְלִים / עוּגוֹת מַצּוֹת וְחֹמֶץ וְהָאִסְקְרִיטִין
וְהַסּוּפְגָּנִין / וְהַחַלּוֹת וְהָאִצְטַגְנִין / זָבוֹת וְזָבִים חָלָב
וּדְבַשׁ וְתַבְלִין / וּרְקִיקִין וּרְבוּכָה אֲשֶׁר עַל־מַחֲבַת
בַּשֶּׁמֶן תֵּעָשֶׂינָה / בְּכָל־סְטָר וַאֲתָר מְצֵינָא / וְיַיִן
מְרֻקָּח וּמְעֻטָּר / בְּכָל־מָקוֹם מֻגָּשׁ מֻקְטָר / וְכָל־צִפּוֹר
כָּל־כָּנָף / אֲשֶׁר הוּרַם וַאֲשֶׁר הוּנַף / וּשְׁאָר כָּל־הָעוֹפוֹת
וּבַרְבֻּרִים אֲבוּסִים / הֵמָּה נִקְרְאוּ הֵמָּה נִבְחֲרוּ וְהֵמָּה
הַנִּגָּשִׁים / וְשׁוֹר פָּר / בְּרִיתוֹ לֹא תוּפָר / וְעֵגֶל / נוֹשֵׂא
הַדֶּגֶל / וְכִמְעַט כָּלְתָה רֶגֶל / וְאַיִל / עוֹשֶׂה חַיִל / וְשֶׂה /
אֲשֶׁר כֹּל בּוֹ חוֹסֶה / וּבְכָל־עִיר וָעִיר / קוֹל קוֹרֵא מִשֵּׂעִיר /
וּגְדִי / גָּד גְּדִי / וְטָלֶה / מְקַצֵּר וְעוֹלֶה / וְהָאַרְנֶבֶת /
מִיָּד לְיָד נִגְנֶבֶת / וְשָׁפָן / בָּא אַחֲרִיתָן וְסוֹפָן / וּבָשָׂר
וְשֹׁמֶן הַחֲזִיר / הוּא הַסּוֹבֵב וְהוּא הַמַּחְזִיר / וְאֵין מִי
שֶׁמִּמֶּנּוּ יַזִּיר / וּמַאֲכָלֵי גְבִינָה וְחָלָב אֲשֶׁר לֹא יִמָּנוּ
וּמֵרֹב לֹא יִסָּפֵרוּ / תִּשְׁבָּחוֹת יַשְׁמִיעוּ וּזְמִירוֹת יֹאמֵרוּ
/ וְכוּתָח הַבַּבְלִי וְזִיתוֹס הַמִּצְרִי / אֵינוֹ הוֹלֵךְ עֲרִירִי /
וְיִטְבְּלוּ בָם בְּשַׂר שׁוֹר וְעֵגֶל וּמְרִיא / וְכָל־מִינֵי לִפְתָּן
וּמַטְעַמִּים מֻרְכָּבִים / וּמַאֲכָלִים חֲשׁוּבִים / וּמְעֻלִּים
וַעֲרֵבִים / כֻּלָּם בְּרוּרִים כֻּלָּם אֲהוּבִים.

סוֹף סוֹף כָּל־הַדָּגִים / אָחוֹר נְהוּגִים וְנוּגִים /
וְיָרָק / בְּפָנָיו יוּרַק / וְיֵצֵא יָצֹא כְּבָרָק / וּמִינֵי קִטְנִית
/ אֵין תָּכְנִית וְתַבְנִית / וְכָל־הָרוֹאֶה אוֹתָן בַּחֲלוֹם יוֹשֵׁב
עָלָיו בְּתַעֲנִית / וּבְשַׂר הָרָזָה / נִמְאֶסֶת וּנְמִבְזָה /
וְכָל־הַתַּבְשִׁילִין / צְלוּיִם שְׁפוּדִים שְׁלוּקִין וּכְבוּשִׁין
מִכָּל־מִין וּמִין בְּטֵלִין וּמְבֻטָּלִין / לָא שְׁרִירִין וְלָא
קַיָּמִין / וּבְמִסְפַּר וּבְמִפְקַד יֶתֶר הָעִדּוּנִין / אֲשֶׁר עַל־זֶה
אֲנַחְנוּ דָנִין / נַעַר וְזָקֵן טַף וְנָשִׁים בְּשַׁנּוֹתָם אֶת־טַעֲמָם
אֶת־טִבְעָם יִתְחַפְּשׂוּ וּשְׁנוֹתֵיהֶם יַסְתִּירוּ וְיַעְלִימוּ /
וּבְלֶכְתָּם בִּשְׁוָקִים וּבִרְחוֹבוֹת הִכָּלֵם לֹא יִכָּלְמוּ וְלֹא
יַאְדִּימוּ / וְסֵתֶר פָּנִים יָשִׂימוּ.

הֶן־אֵלֶּה קְצוֹת דַּרְכֵיהֶם / וְתַעֲנוּגֵיהֶם / וְטוּב
מַאֲכָלֵיהֶם וּמִשְׁתֵּיהֶם / וַעֲרֵבוּת מַשְׁקֵיהֶם / יֶחֱרֶה אַף יְיָ
בָּהֶם / זֶה הַכְּלָל בַּיָּמִים הָהֵם / וּבַזְּמָן הַזֶּה יְבֻקַּשׁ
הֶחָכָם וְאֵינֶנּוּ / הַמַּשְׂכִּיל וְלֹא תִמְצָאֶנּוּ / הַנָּבוֹן
וּבַעַל דַּעַת מִקַּצְוֵי אֶרֶץ וּמִיָּם עַד־יָם / וְלֹא נוֹדַע מְקוֹמָם
אַיָּם / אִם בַּעַל הַמּוּסָר / צִלּוֹ סָר / וְאִם הַמְּלִיצִים
וְהֶחָרוּצִים / כֻּלָּם עָבְרוּ חֲלוּצִים / וְחָלְפוּ לָמוֹ דְחוּפִים
וְרָצִים / וּמָשְׁכוּ יְדֵיהֶם אֶת־לֹצֲצִים / בַּעֲלֵי עֲנָוָה /
שְׁטוּפִים בְּמִיץ הָעֲנָבָה / כְּאִלּוּ הֻקְבַּע הַדָּבָר חוֹבָה /
הַחֲסִידִים / חֲסֵרִים חֲסוּדִים וַחֲשׁוּדִים / הַפִּקְחִים / נֶהְפָכִים
וְהוֹפְכִים / וּבְכוֹסוֹת וּבִצְלוֹחִיּוֹת נוֹפְחִים / הַסּוֹפְרִים /
כּוֹפְרִים וְנֶחְפָּרִים / הַשּׁוֹפְטִים / נִשְׁפָּטִים וּמֻפְתִּים /
וְאֶחָד אֶחָד נִשְׁמָטִים / אַנְשֵׁי אֱמֶת וְהַיּשֶׁר / אָכְלוּ בֹסֶר /
וְלֹא יָדְעוּ כִּי יְבוֹאֵם חֹסֶר / הַנְּגִידִים / נֶאֱגָדִים /
וּבָגְדוּ בֶּגֶד בּוֹגְדִים / הַשּׁוֹטְרִים נִפְטָרִים / וְנִסְתָּרִים
בִּסְתָרִים / הַמּוֹשְׁלִים / לִמְשָׁלִים / הַיּוֹעֲצִים / כְּקוֹצִים
וְעֵצִים / צַחֵי הַלָּשׁוֹן / לְלַעַג וּלְקֶלֶס וּלְלָצוֹן / וְהַשָּׂבָע
לְעָשִׁיר אֵינֶנּוּ מַנִּיחַ לוֹ לִישׁוֹן / הַמְשׁוֹרֲרִים / סוֹרֲרִים /
וְהֵמָּה צוֹרֲרִים וּצְרוּרִים / הַטִּבְעִיִּים וְהַתְּכוּנָיִים /
בַּדַּאִים וּפְנוּיִים / וְהַפִּילוֹסוֹפִים וְהָאַשָּׁפִים
וְהַמְכַשְּׁפִים / נִכְסָפִים וְנֶהְדָּפִים / וְלֹא נֶאֱסָפִים /
הַחַרְטֻמִּים / אֲטֻמִים וּסְתֻמִים / וְהַקּוֹסְמִים / נֶחְשָׁבִים
כְּקֵיסָמִים / הַהוֹבְרִים / כִּשְׁבוּרִים וּקְבוּרִים / וְאֵין אֹמֶר
וְאֵין דְּבָרִים / וּבַעֲלֵי מִבְטָא / בִּבְלִי הַבָּטָה / בָּאָה לָהֶם
הַחֲבָטָה / וְכֻלָּם מְלֵאֵי חֲרָטָה.



נִמְצָא הַדִּקְדּוּק / בְּבַקְבּוּק / הַהִגָּיוֹן /
בְּשִׁגָּיוֹן / חָכְמַת הַשִׁיר / בְּדוּד וְסִיר / הַפְּעֻלָּה /
בִּבְתוּלָה וּבְעוּלָה / הַצֵּרוּף / לְגִדּוּף וְחֵרוּף / חָכְמַת
הַמֶּחְקָר / אַחֲרֵי הַצֹּאן וְהַבָּקָר / הַתַּחְבּוּלָה / נִתְבָּהֲלָה
/ הַהַנְדָּסָה / בְּדַיְסָא / וְהִיא נְתוּשָׁה וּדְרוּסָה / הַהֲלָצָה /
לְכָלָה וְנֶחֱרָצָה / הַחֶשְׁבּוֹן / לְעִזָּבוֹן / וּלְזָבוֹן /
הַתִּשְׁבֹּרֶת / נִרְאֲתָה וְהִיא שְׁחַרְחֹרֶת / מְלֶאכֶת הַיָּד /
מְגֻיָּד / הָרְפוּאָה / כְּאִיסְטֵיס וּפוּאָה / חָכְמַת הַתְּכוּנָה /
מְגֻנָּה / כִּי אֵין בְּפִיהָ נְכוֹנָה / הַחַיָּטִים / בּוֹעֲטִים /
הַסַּפָּרִים / מְסַפְּרִים סִפּוּרִים שַׁפִּירִים / הַכּוֹבְסִים /
מִיִּצְרָם נִכְבָּשִׁים / וְהָרַצְעָנִים / נִשְׁעָנִים וּנְזוֹנִים /
וְאֵינָם מִתְעַנִּים / הַסַפָּן / שָׂמֵחַ בְּכָל־אוֹפָן / הַחַמָּר /
אָזֵל וּמְזַמֵּר / הַגַּמָּל / הָרָה לוֹ עָמָל / הָרוֹעֶה / מְטַיֵּל
וְתוֹעֶה / וְכָל־הָאוֹמֵר שֶׁאֵין לוֹ חֵלֶק בְּהִתּוּלִים הַלָּלוּ
אֵינוֹ אֶלָּא טוֹעֶה / הַנַּגָּר / דּוֹמֶה לְתַגָּר / וְאֵינוֹ רָץ
לִבְאֵר הַמַּיִם כְּהָגָר / הַבַּנַּאי / בּוֹנֶה דֶרֶךְ גְּנַאי /
וְכִמְעַט שֶׁאֵין לוֹ פְנַאי / וּבַלָּן / רֹאשׁ לְכֻלָּן / כְּשֶׁהוּא
רוֹאֶה הַצּוֹחֲקוֹת וְשׁוֹמֵעַ לְקוֹלָן / אוֹמֵר: “בְּרִיךְ רַחְמָנָא
דִּיְהַבָךְ לָן” / חָרָשׁ / יֶחֱרַשׁ / וְחוֹשֵׁב חוֹשֵׁב מַחֲשָׁבוֹת /
אֲשֶׁר לֹא טוֹבוֹת / וְרוֹקֵם עוֹקֵם מְלַאכְתּוֹ / וְלַלֵּצִים הוּא
יָלִיץ בְּקִימָתוֹ וְשִׁבְתּוֹ / וְגַם הַמֶּלֶךְ מוֹחֵל עַל־כְּבוֹדוֹ /
וְאֵינוֹ חוֹשֵׁשׁ לְאֵימָתוֹ וַהֲדָרוֹ וְהוֹדוֹ / וְהַמַּלְכָּה
מִלְכָּהּ עֲלַהּ יִשְׁפַּר / וִיפֵה־תֹאַר וּמַרְאֶה בְּעֵינֶיהָ תֶחְפַּר
/ הַדּוּכוֹס / עַל־יְקָרוֹ לֹא יָחוֹס / וְהַשָּׂר / מִפְּאֵרוֹ יֶחְסָר /
וְהַפַּחוֹת פּוֹחֲתִים וְהוֹלְכִים אַחַר נוֹגְנִים / וְהַשִּׁלְטוֹנִים
וְהָאֲחַשְׁדַּרְפְּנִים וְהַסְּגָנִים / נִכְשָׁלִים בִּבְנֵי אָדָם
שֶׁאֵינָם מְהֻגָּנִים / וְשָׂרֵי הַצָּבָא / בִּטְנָם צָבָה / וְשָׂרֵי
אֲלָפִים וּמֵאוֹת וַחֲמִשִּׁים וַעֲשָׂרוֹת / מַנִּיחִים
אֶת־חֲיָלוֹתֵיהֶם וְאֶת־הַשְּׁמִירוֹת / מִשְׁמַרְתָּן וּמִשְׁמֶרֶת
הָעֲיָרוֹת / וְהוֹלְכִין אַחַר הַבְּתוּלוֹת הַיָּפוֹת וְהַנָּשִׁים
הַיְקָרוֹת / הָאַפִּיפְיוֹר / שׁוֹתֶה יַיִן בְּמִזְרְקֵי־כֶסֶף
וּבְכִיּוֹר / וְהוּא עוֹמֵד וּמְטַיֵּל עַל־שְׂפַת הַיְאוֹר /
הַחַשְׁמַנִּים / אוֹתָן דִּרְבוּ יַתִּירָא הָיָה נוֹתֵן עֲלֵיהֶם
סִימָנִים / בָּאַשְׁמַנִּים / שְׁמֵנִים וּדְשֵׁנִים וְרַעֲנַנִּים /
הַהֶגְמוֹן / אֲשֶׁר כְּאַגְמוֹן / רֹאשׁוֹ יַכְנִיעַ וְיָכֹף / גַּם הוּא
בְמַעֲלָלָיו דּוֹמֶה לְקוֹף.

וְהַכֹּהֲנִים וְהָעָם הָעוֹמְדִים בַּעֲבוֹדָה זָרָה /
נֶהֱנִין וְנִזּוֹנִין מִזִּיו הַשְּׁכוּנָה וְהַחֲבוּרָה / וְחָטְאוּ
וְעָווּ וּפָשְׁעוּ וְעָבְרוּ הָעֲבֵרָה / אֵין יוֹם שֶׁאֵין סְעוּדָתוֹ
מְרֻבָּה מֵחֲבֶרְתָּהּ / וְלָא אִיבְּרִי סִיהֲרָא לְשֵׁינְתָא / אֶלָּא
לִזְנוּתָא וְהֶפְקְרוּתָא.

שֶׁבִּן־לַיְלָה הָיָה וּבִן־לַיְלָה אָבְדָה שִׂמְחָתָם /
וְסָרָה מֵעֲלֵיהֶם אִוַּלְתָּם / וְיָקוּמוּ אִישׁ אִישׁ מִמְּסִבָּתָם /
וְרִקּוּדָם וּצְחִיקוּתָם / בְּשָׁמְעָם אֶת־רִנָּתָם / וְאֶת־קוֹל
תְּרוּעָתָם / בְּבֵית אוֹן אֱלֹהֵיהֶם / וְיִפָּרְדוּ אִישׁ מֵעַל
אֲחֵיהֶם / רֵעֵיהֶם וְחַבְרֵיהֶם / וְיֵרָגְנוּ בְאָהֳלֵיהֶם.

וַיְהִי מִמָּחֳרָת זֶה נָגוּעַ וְזֶה מֻכֶּה / זֶה
נִשְׁבָּר וְזֶה נִדְכֶּה / זֶה כּוֹאֵב וְזֶה דּוֹאֵג / זֶה מִתְאַנֵּחַ
וְזֶה שׁוֹאֵג / זֶה חוֹשֵׁשׁ בִּגְרוֹנוֹ / וְזֶה בִּשְׁרִירֵי בִטְנוֹ /
זֶה מִמִּלּוּי אִצְטָמְכָתוֹ צוֹעֵק כְּאִלּוּ נִשְׁתַּטָּה / וְזֶה
מִזְּרֹעוֹתָיו וּמַרְגְלוֹתָיו דּוֹאֵב כְּאִלּוּ נָפְלָה עֲלֵיהֶם מוֹטָה
/ וְחוֹלֶה וְנוֹפֵל עַל־הַמִּטָּה / זֶה חָשׁ בְּרֹאשׁוֹ / וְזֶה אָבַד
תַּאֲוָתוֹ וְאֶפְשׁוֹ / וְלֹא יָדַע כִּי הוּא בְנַפְשׁוֹ / כֻּלָּם
אֲחֻזֵי חֹרֶב וְקֶרַח בָּאוּ בֵית שֶׁמֶץ אֱלֹהֵיהֶם / וְהֵמָּה כָּרְעוּ
עַל־בִּרְכֵיהֶם / וַיִּזְרְקוּ עָפָר עַל־רָאשֵׁיהֶם.
