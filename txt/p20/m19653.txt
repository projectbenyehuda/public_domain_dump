


	
	אֶתְעוֹרֵר וַאְשׁוֹרֵר יוֹם גִּילִי
	



גאולה




אֶתְעוֹרֵר / וַאְשׁוֹרֵר יוֹם גִּילִי,



גִּילִי רַב / יוֹם יֶעְרַב / הוֹד קוֹלִי,



קוֹלִי רָם, / כִּי הוּרַם / נֵס דִּגְלִי,



דִּגְלִי עַל / כֹּל פָּעַל / לִי אֵלִי,



אֵלִי זֶה, / שֶׁאֶחְזֶה / מִמּוּלִי;



מִמּוּלִי / מִשְׁאָלִי, / הָיָה לִי,



לִי – טוּבוֹ / הִצִּיבוֹ, / הִצִּיבוֹ / לָשׂוּשׂ בּוֹ.



בּוֹ שָׂמַח, / בּוֹ צָמַח / צִיץ נִזְרִי,



נִזְרִי גֵא, / כִּי אֶהֱגֶה / דַת צוּרִי;



צוּרִי בָּא, / כִּי נִכְבָּה / נֵר צָרִי,



צָרִי סָר / בַּמּוּסָר / מֵעִירִי –



עִירִי שָׁב / עִיר נוֹשָׁב, / וּדְבִירִי;



וּדְבִירִי / כִּדְבָרִי, / כִּדְבָרִי / סָר צִירִי,



צִירִי רַד – / כֹּל חָרַד, / כָּל חָרַד, / יוֹמוֹ רָד.



רָד גֵּאֶה, / אַךְ נִכְאֶה / רֹאשׁ הֵרִים



הֵרִים [……….]



[……….]



[……….]



כִּי עָלֶיךָ הָשְׁלַכְתִּי מֵרֶחֶם.



רֹאשׁ כֹּחִי תַּהֲפֹךְ לְרַע, נִגְלָה מְכַסֵּהוּ,



וְנִלְקַח מִצְּבָאוֹת – חֵלֶק שָׁוְא יְכַסֵּהוּ;



לְךָ, יְיָ, חֶסֶד – בְּיִרְאָתְךָ מַחֲסֵהוּ,



כִּי אַתָּה תְשַׁלֵּם לְאִישׁ כְּמַעֲשֵׂהוּ.



הוּבֵאתִי הֵנָּה לְהִתְבּוֹנֵן עִנְיָנַי,



נִלְחָם בִּי יִצְרִי, הֱמִיתַנִי בַאֲזֵנָי,



בּוֹשְׁתִּי מִפְּנֵי אֵל – וָאֶשְׁפֹּךְ תַּחְנוּנַי,
כִּי שֹׁמֵעַ אֶל אֶבְיוֹנִים יְיָ.
