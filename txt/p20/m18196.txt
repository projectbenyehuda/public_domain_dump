


	
	חרדו רעיוני (אופן לראש השנה)
	




חָרְדוּ רַעְיוֹנַי, / כִּי אֱלֹהֵי מִשְׁפָּט / שׁוֹכֵן עַד וְקָדוֹשׁ, /
מְדַבֵּר בִּצְדָקָה.



בְּאָהֳבוֹ עַם הֲמוֹנַי, / שָׂם לוֹ חֹק וּמִשְׁפָּט – / וְדָת יוֹם זֶה
קָדוֹשׁ / תִּהְיֶה לוֹ צְדָקָה,



וְיִגְבַּהּ יְיָ / צְבָאוֹת בַּמִּשְׁפָּט / וְהָאֵל הַקָּדוֹשׁ /
נִקְדָּשׁ בִּצְדָקָה.



אַבִּירֵי בְרִיאוֹת / אֶרֶץ, הָחִישׁוּ / לְרַצּוֹת בּוֹרַאֲכֶם /
וְחַסְדּוֹ יַאֲמִצְכֶם,



וּבִנְפָשׁוֹת צְמֵאוֹת / לוֹ שִׁיר הַגִּישׁוּ, / הוּא יְרַוֶּה צְמָאֲכֶם
/ וְדִשְׁנוֹ יַחֲלִיצְכֶם –



אֶת יְיָ צְבָאוֹת – / אֹתוֹ תַקְדִּישׁוּ, / וְהוּא מוֹרַאֲכֶם / וְהוּא
מַעֲרִצְכֶם.



בְּמִבְטָא שְׂפָתַיִם / לְךָ אֶעֱרֹךְ רְנָנָי / וְעִנְיְנֵי תְהִלָּתְךָ
/ עֲלֵי לִבִּי חֲרוּשִׁים;



גָּדוֹל בְּמַחֲנַיִם / שִׁמְךָ, מְעוֹן מְעוֹנָי: / בְּרוֹם –
בַּהֲמֻלָּתְךָ, / לְמַטָּה – בַּנְּפָשִׁים,



וְיוֹדוּ שָׁמַיִם / פִּלְאֲךָ, יְיָ, / אַף אֱמוּנָתְךָ / בִּקְהַל
קְדֹשִׁים.



רָאֲךָ מִבְּשָׂרִי / לֵב, שַׂמְתּוֹ לְצוֹפֶה, / שְׂעִפַּי – הוֹשִׁיבָם /
בִּמְקוֹם הַקֹּדֶשׁ.



מְאֹד אֶשְׂמַח, / בְּזָכְרִי / יוֹם תֵּת חֹק יָפֶה / לְבָנִים, בְּבוֹא
מְחוֹבְבָם / מֵרִבְבוֹת קֹדֶשׁ –

רֶכֶב אֱלֹהִים רִ-/ בֹּתַיִם אַלְפֵי / שִׁנְאָן, יְיָ בָם / סִינַי
בַּקֹּדֶשׁ.



הַכֹּל מַגִּידִים / גְּבוּרָתוֹ וְגָדְלוֹ / בְּעוֹלָמִים שְׁנַיִם /
יְהַלֲלוּהוּ יָחַד:



מִשְׁתַּחֲוִים וּמוֹדִים / מְתִים חוֹסִים בְּצִלּוֹ, / וּצְבָא
הַשָּׁמַיִם / יְשָׁרְתוּהוּ בְּפָחַד –



שְׂרָפִים עֹמְדִים / מִמַּעַל לוֹ, / שֵׁשׁ כְּנָפַיִם / שֵׁשׁ כְּנָפַיִם
לְאֶחָד.



מְבַקְּשִׁים אַיֵּה זֶה / מַחְנֶךָ, קָדוֹשׁ, / בְּנֵי אֱלֹהִים, כִּי
אוֹת / כְּבוֹדְךָ בָּם וְהוֹדוֹ;



רָאָהוּ בְמַחֲזֶה / אִישׁ אֱלֹהִים קָדוֹשׁ / עַל כִּסֵּא פְלָאוֹת /
וּסְבִיבָיו גְּדוּדוֹ –



וְקָרָא זֶה אֶל זֶה / וְאָמַר: קָדוֹשׁ, קָדוֹשׁ, קָדוֹשׁ / יְיָ
צְבָאוֹת, / מְלֹא כָל הָאָרֶץ כְּבוֹדוֹ!
