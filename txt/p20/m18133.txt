


	
	אעירה שחר על דברתך (סליחה)
	






אָעִירָה שַׁחַר עַל דִּבְּרָתֶךָ,



אָקִיצָה מִשְּׁנָתִי לְסַפֵּר גְּבוּרוֹתֶיךָ,



אַשְׁכִּימָה לִבְקָרִים לְרוֹמֵם הַדְרָתֶךָ –



אַשְׁמוּרוֹת קִדַּמְתִּי לָשִׂיחַ בְּאִמְרָתֶךָ.



בְּסַרְעַפַּי בְּקִרְבִּי אַגִּידָה נִצְחֶךָ,



בְּעֶשְׁתּוֹנוֹת לִבִּי גֹּדֶל שִׁבְחֶךָ,



בְּעוֹדִי אַזְכִּירָה פִּלְאֵי כֹחֶךָ,



בְּאֵימָה וּבִרְעָדָה סֶלָה אֲנַצְּחֶךָ.



גְּבוּרוֹתֶיךָ שְׁחָקִים מְסַפְּרִים בְּאַהֲבָה,



גְּדֻלָּתֶךָ יַגִּידוּ תֵּבֵל וְיוֹשְׁבֵי בָהּ,



גְּאוֹנְךָ מַרְגִּיז הַיָּם וְהֶחָרָבָה,



גַּעֲרָתְךָ מְיַבֶּשֶׁת תְּהוֹם רַבָּה.



דֵּעָה אַיֵּה וּלְשׁוֹן לִמּוּדִים,



דָּרְבוֹנֵי חָכְמָה, אֲנָשִׁים מְלֻמָּדִים?



דִּקְדּוּק מַעֲשִׂים נוֹרָאִים וְנֶחֱמָדִים –



דְּרֹשׁ פָּעֳלָם, וְכֻלּוֹ מַחֲמַדִּים.



הֵן מִקְצָתָם אַגִּידָה וַאֲסַפֵּר –



הוֹרוּנִי יְשִׁישִׁים יוֹדְעֵי סֵפֶר,



הֲבִינוֹתִי מִדַּעְתָּם לַחְשֹׁב וּלְסַפֵּר



הֶגְיוֹן לְשׁוֹנָם וְאִמְרֵי שֶׁפֶר.



וְטוֹב וְנָעִים בֹּקֶר לְקַדֵּם



וּלְחַלּוֹת פָּנֶיךָ, אֱלֹהֵי קֶדֶם.



וְאָנֹכִי תוֹלַעַת תְּפִלָּתִי אֲקַדֵּם



וְרַחֲמֶיךָ אֲבַקֵּשׁ וְלֹא אֵרָדֵם.



זָכוֹר תִּזְכּוֹר יְמוֹת עוֹלָם,



זְכוֹר רַחֲמֶיךָ וַחֲסָדֶיךָ מֵעוֹלָם:



זָן בְּרִיּוֹתֶיךָ בְחֶסֶד כֻּלָּם,



זֵדִים וַחֲסִידִים – וְלֹא כְפָעֳלָם.



חַנּוּן וְרַחוּם, עוֹשֶׂה גְּדוֹלוֹת,



חוֹקֵר כְּלָיוֹת וְרַב עֲלִילִיּוֹת,



חוֹפֵשׂ טְמוּנוֹת וְסִתְרֵי מִפְעָלוֹת,



חוֹנֵן וּמַפְלִיא, נוֹרָא תְהִלּוֹת.



טְבוּלֵי רִגְיוֹן יְנַצְּחוּךָ נִגּוּנִים,



טַפְסָרֵי עֲרָבוֹת יְשׁוֹרְרוּךָ רְנָנִים,



טְכוּסֵי מֶרְכָּבָה, רִבְבוֹת שִׁנְאָנִים



טַעַם יַמְתִּיקוּ וּבְרַעַד מִתְחַנְּנִים.



יוֹרְדֵי הַיָּם, יוֹדְעֵי נִסֶּיךָ,



יַגִּידוּ בָאִיִּים גֹּדֶל מַעֲשֶׂיךָ,



יוֹשְׁבֵי אֹהָלִים, מִבֶּטֶן עֲמוּסֶיךָ,



יִחוּדְךָ מְיַחֲדִים בְּקֶרֶב מַכְעִיסֶיךָ.



כֹּחַ גְּבוּרוֹתֶיךָ מִי זֶה יְמַלֵּל,



כַּעֲלִילוֹתֶיךָ לְהַשְׁמִיעַ תְּהִלּוֹת וּלְפַלֵּל?



כִּלְיוֹתַי יִסְּרוּנִי יוֹם וָלֵיל



כְּבוֹד שִׁמְךָ לְנַצֵּחַ וּלְהַלֵּל.



לַיְלָה אָקוּם, אֵלֶיךָ אֶתְחַנֵּן,



לַחַשׁ צָקוּן בְּלִבִּי אֶתְבּוֹנֵן –



לִשְׁקֹד עַל דַּלְתֹתֶיךָ וּלְהִשְׁתּוֹנֵן,



לַחֲסוֹת בְּצֵל כְּנָפֶיךָ וּלְהִתְלוֹנֵן.



מִי אָנֹכִי וּמִי עַבְדֶּךָ?



מִבְטָא שְׂפָתַי אֶעֱרֹךְ נֶגְדֶּךָ!



מֻבְטָח אֲנִי בְּרַחֲמֶיךָ וַחֲסָדֶיךָ,



מִדָּתְךָ – לִשְׁמֹעַ בָּאֵי עָדֶיךָ.



נַפְשִׁי אִוִּיתִיךָ, שׁוֹמֵעַ עֲתִירוֹת!



נוֹצֵר חֶסֶד לְאַלְפֵי דּוֹרוֹת,



נוֹשֵׂא עָוֹן וּפֶשַׁע עֲבֵרוֹת,



נָא לְמַעַנְךָ נַקֵּנִי מִנִּסְתָּרוֹת!



שִׂימָה נָּא עִמְּךָ עָרְבֵנִי,



סַעֲדֵנִי בִימִינְךָ וְעַנְוָתְךָ תַרְבֵּנִי,



שֹׂבַע שְׂמָחוֹת פָּנֶיךָ תְּסוֹבְבֵנִי,



סוֹכְכֵנִי בְּאֶבְרָתְךָ וְלָבֶטַח תּוֹשִׁיבֵנִי.



עָלֶיךָ מֵעוֹדִי הִשְׁלַכְתִּי יְהָבִי,



עֶזְרָתִי הָיִיתָ, אֱלֹהֵי אָבִי,



עֻזִּי וּמָעֻזִּי, רִיבָה רִיבִי!



עֲנֵנִי בְקָרְאִי, בַּצַּר תַּרְחִיבִי!



פְּשָׁעַי תִּמְחֶה וְתַעֲבִיר אֲשָׁמִים,



פַּלְּטֵנִי מֵעָוֹן וּמֵאֹדֶם חֲטָאִים!



פְּדָעֵנִי מִשַּׁחַת וּמִגַּחֲלֵי רְתָמִים!



פִּקּוּדֶיךָ הֲבִינֵנִי – וְאֶהֱיֶה תָּמִים.



צָמְאָה לְךָ רוּחִי בְּקִרְבִּי.



צוּר, צָרְפָה כִלְיוֹתַי וְלִבִּי!



צִדְקָתְךָ תְּמַגְּנֵנִי, צוּרִי וּמִשְׂגַבִּי,



צַדְּקֵנִי בַּמִּשְׁפָּט וְאַל נָא תַּעַזְבִי!



קָרוֹב אַתָּה לְכָל קוֹרְאֶיךָ וְדוֹרְשֶׁיךָ,



קוֹוֶיךָ לֹא יֵבוֹשׁוּ וּמְבַקְשֶׁיךָ.



קַבְּלֵנִי בְּרָצוֹן, כִּי אֲבַקְשֶׁךָ,



קְשֹׁב תְּחִנָּתִי מִמְּעוֹן קָדְשֶׁךָ!



רַחֲמֶיךָ, יְיָ, וַחֲסָדֶיךָ מֵעוֹלָם.



רִשְׁעִי כַפֵּר, יָדוּעַ וְנֶעְלָם.



רְאֵה עָנִי חָפוּי וְנִכְלָם,



רַחֵם וְשָׂא חַטֹּאתַי כֻּלָּם!



שַׁעֲרֵי תְפִלָּה פְּנִימִי וְחִיצוֹן –



שַׁדַּי, תִּפְתַּח דַּלְתוֹתָם בְּעִלָּצוֹן!



שַׁי, כְּתָמִים מִן הַצֹּאן,



שָׁם אֲמָרַי יַעֲלוּ לְרָצוֹן.



תּוֹדוֹת, שְׁלָמִים וְזִבְחֵי עֵדָה,



תִּקְרֹבֶת עוֹלָה כְּעַל מוֹקְדָה –



תְּמוּרָתָם תְּפִלָּתִי וּתְפִלַּת הָעֲבוֹדָה



תִּכּוֹן כִּקְטֹרֶת-בֹּשֶׂם-וְקִדָּה.



אֲנִי בְרַחֲמֶיךָ שָׁכַבְתִּי וָאִישָׁנָה,



הֱקִיצוֹתִי מִשְּׁנָתִי נִפְלְאוֹתֶיךָ לְשַׁנְּנָה.



חַזְּקֵנִי וְאַמְּצֵנִי עֶזְרָה הַיְשָׁנָה,



עַצְמוֹתַי לְהַחְלִיף וּלְהַפְרִיחַ כַּשּׁוֹשַׁנָּה!
