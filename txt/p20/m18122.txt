


	
	אבי עבר על רשעי (תוכחה)
	




אָבִי, עֲבֹר עַל רִשְׁעִי; / הֲתִדְרוֹשׁ לַעֲוֹן בִּצְעִי? / וּמֶה לֹא תִשָּׂא פִשְׁעִי?



אָעִיר סְעִפַּי לְהוֹדָם / וְאַגִּיד לָהֶם יְסוֹדָם – / מַה יִּתְרוֹן לָאָדָם?



בְּתֵבֵל שָׂשׂ אֵיךְ אֶהִי / וְאַחֲרִית גִּילָה נֶהִי? / וּמִדַּת יָמַי מַה הִיא?



רוּצִי, נַפְשִׁי, וַחֲסִי / בְּצֵל תֻּמֵּךְ, וְהִתְכַּסִּי; / דְּעִי
וּרְאִי מַה תַּעֲשִׂי.



הֲכִי דְרָכַי עִוִּיתִי / וְהוֹד יָסוּף אִוִּיתִי – / וְעַתָּה מַה קִּוִּיתִי?



מִזֵּכֶר יוֹם הַמַּר / וּפַחְדּוֹ בְּשָׂרִי סָמַר; / בִּי, יְיָ, מָה
אֹמַר?



בְּרוּאֶיךָ אִם תְּנַסֶּה / וְעַל פֶּשַׁע לֹא תְכַסֶּה – / לֹא נֵדַע מַה נַּעֲשֶׂה.



רַחוּם, לְחַסְדְּךָ אֲחַכֶּה / וְאִם כְּדֵי רֶשַׁע תַּכֶּה – / מָה
אֱנוֹשׁ כִּי יִזְכֶּה?



בַּעֲבוּר יֵצֶר בָּעַל / בְּדָתְךָ לִבִּי מָעַל; / חָטָאתִי, מָה
אֶפְעַל?



יִפְּלוּ בְּבֹשֶׁת פָּנַי, / כִּי כְּחוֹל הַיָּם אוֹנָי; / מָה אָשִׁיב
לַייָ?



מְתֵי תֵבֵל, אֵלֶיהָ / תְּשׁוּבוּן בַּהֲבָלֶיהָ – / וּמַה כִּשְׁרוֹן
לִבְעָלֶיהָ?



אַתֶּם יְסוֹד תַּכְלִיתָהּ / וְאִם תִּשְׂמְחוּ בְּרֵאשִׁיתָהּ – / מַה
תַּעֲשׂוּ בְּאַחֲרִיתָהּ?



יוֹם יֵעוֹר מִתְנַשֵּׂא, / עֲרוֹם צֶדֶק מִי יְכַסֶּה? / וּמִי יֹאמַר
לוֹ: מַה תַּעֲשֶׂה?



רוּחִי אִם יִתְגָּאֵל / בְּגֵוִי – לֹא יִגָּאֵל; / וּמָה אֶעֱשֶׂה כִּי
יָקוּם אֵל?



חַשְׁתִּי לְבַקֵּשׁ שָׁרְשִׁי / בְּעוֹדִי לָצֵאת חָפְשִׁי – / וּמַה
קִּצִּי כִּי אַאֲרִיךְ נַפְשִׁי?



זְכֹר, כְּבוֹדִי, הוֹד תֻּמְּךָ, / קְנֵה חַיִּים לְעַצְמְךָ – / וּמָה
יְיָ דּוֹרֵשׁ מִמְּךָ?
