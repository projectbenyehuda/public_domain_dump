


	
	אֶת חֶפְצִי מַלֵּא
	




אֶת חֶפְצִי מַלֵּא / הַצָּפוּן בְּקִרְבִּי,

כִּי לֹא יִפָּלֵא / מִמְּךָ כָּל חֻבִּי,

לְךָ סוֹדִי נִגְלֶה / כָּל שֶׁאֲדַבֶּר בִּי,

אֲנִי טֶרֶם אֲכַלֶּה / לְדַבֵּר אֶל לִבִּי.



בִּקַּשְׁתִּי יְסוֹדִי / לְדֵעָה מַחֲבוֹאוֹ,

כִּי בְגוּפִי כְּבוֹדִי / כְּמֶלֶךְ עַל כִּסְאוֹ –

וְלֹא אֵדַע סוֹדִי / וְאֵיךְ צוּרִי בְּרָאוֹ

וְאֵיךְ בּוֹ מַעֲמָדִי / וְלָמָּה בִּי כְלָאוֹ?

נְתָנוֹ עִמָּדִי – / וּמָנִּי יוֹצִיאוֹ

וְעַל מַעֲשֵׂה יָדַי / לַמִּשְׁפָּט יְבִיאוֹ.

הָהּ לִי יוֹם אֵידִי / בִּפְקֹד אֵל חֶטְאוֹ!

עַל כֵּן בְּכָל מְאֹדִי / אָשׁוּב לְמִקְרָאוֹ.

הֵא בְדָתוֹ עֵדִי / כִּי בְקִרְבָּה פִּלְאוֹ

וְחָלִילָה בְּעוֹדִי / לְהַמְרוֹת פִּי נְבִיאוֹ,

לַעֲשׂוֹת טוֹבָה אוֹ רָעָה מִלִּבִּי.



רַבּוּ חֲסָדֶיךָ / בָּהֶם יְצַרְתָּנִי,

אֲנִי אֲנִי מְעִידֶךָ/ כִּי אַתְּ יְלִדְתָּנִי

וּלְבִלְעָדֶיךָ / לֹא הֶעֱבַדְתָּנִי.

עַל כֵּן אוֹדֶךָ / לְטוֹבָה זְבַדְתָּנִי.

מְעַט מִכְּבוֹדֶךָ / אֱמֶת כִּבַּדְתָּנִי –

אָכֵן בְּסוֹדֶךָ / לֹא הֶעֱמַדְתָּנִי.

לִדְבֹּק בְּהוֹדֶךָ / עֲטָרוֹת עֲנַדְתָּנִי

וּבִרְצוֹן עֲבָדֶיךָ / עַד כֹּה פְּקַדְתָּנִי.

בְּמַעֲשֵׂה יָדֶיךָ / מְאֹד הִפְחַדְתָּנִי –

וּבְפִקּוּדֶיךָ / הֲכִי עוֹדַדְתָּנִי,

אֲשֶׁר לִמַּדְתָּנִי: לְךָ אָמַר לִבִּי.



מֵאִתְּךָ חָכְמָה / לָמְדוּ רַעְיוֹנָי,

כִּי בְדַעַת וְעָרְמָה / תַּעֲנוּג מַעֲדַנָּי

וְלָמָּה עוֹד אָצְמָא / כִּי בָךְ מַעְיָנָי!

בְּךָ יָדִי רָמָה, / בְּךָ אָשָּׂא פָנָי.

אִם הַשָּׁמַיְמָה / נָשָׂאתִי עֵינָי –

שָׂהֲדֶיךָ שָׁמָּה; / גַּם הֵם בַּאֲדָנָי,

כִּי בְכָל הָאֲדָמָה / עֵדַי נֶאֱמָנָי.

אַתָּה אֵל אֱמֶת, וּמָה / אֲנִי אוֹ עִנְיָנָי?

הֵן אֵין לִי עָצְמָה / לָדַעַת צְפוּנָי;

עַל מַה זֶּה וְלָמָּה / וּמֶה הָיָה לְפָנָי –

לְחַפֶּשׂ זֹאת, אֲדֹנָי, לֹא גָבַהּ לִבִּי.
