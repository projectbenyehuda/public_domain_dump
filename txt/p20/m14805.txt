


	
	אֵל לְמוֹשָׁעוֹת פְּדֵנִי
	




אֵל לְמוֹשָׁעוֹת פְּדֵנִי, / הַעֲלֵנִי מִמְּצֻלוֹת־

יָם! חֲטָאַי רִדְּפוּנִי, / הֶעֱווּ יֹשֶׁר מְסִלּוֹת;

אִם עֲוֹנַי קִדְּמוּנִי – / אֵיךְ אֲקַדֶּמְךָ בְּעוֹלוֹת?

אַךְ לְבָבִי אַקְרִיב – וְתִרְצֵנִי,

וַאֲיַחֵל עַד שֶׁתְּחָנֵּנִי.

יוֹצְרִי, רְצֵה שִׂיחִי,

יֶעֱרַב לְךָ לִקְחִי,

נָא אַל תְּדִינֵנִי –

עַל מַה תְּרִיבֵנִי?

חוּשָׁה רְפָאֵנִי

וּסְלַח לְחַטָּאתִי!



בּוֹרְאִי, בִּנְךָ בְּכוֹרְךָ – / מָשְׁלָה בוֹ יַד מְעַנָּיו,

גֵּרְשׁוּהוּ מִדְּבִירָךְ, / נָעֲלוּ דֶלֶת בְּפָנָיו:

קוּם, פְדֵה וּגְאַל אֲסִירָךְ / מִיְּדֵי שׂוֹטְנָיו וּמוֹנָיו,

כִּי לְיִשְׁעָךְ תָּמִיד לְבַד יוֹחִיל,

טוֹב וְיָחִיל דּוּמָם אֲחָזוֹ חִיל.

מַעֲנָיו לְךָ קָצְרוּ,

צַעֲדָיו מְאֹד יֵצְרוּ.

מַהֵר שְׁבֹר כַּבְלוֹ,

פַּנֵּה נְתִיב רַגְלוֹ,

הָרֵם כְּנֵס דִּגְלוֹ

תּוֹךְ עִיר קְהִלָּתִי!



רָקְבוּ עַצְמוֹת מְקַוֶּה / רַחֲמֶיךָ הַנְּצוּרוֹת,

בֵּן רְאוֹת אָבִיו יְאַוֶּה, / לַחֲזוֹת נֹעַם
גְּבוּרוֹת.

קוּם וְתֵן לוֹ עֹז, וְצַוֵּה / יִשְׁעֲךָ, עֶזְרָה בְצָרוֹת.

הַעֲלֵהוּ אֶרֶץ יְרֻשָּׁתוֹ,

שַׁלְּמָה לוֹ בִשְׁנַת גְּאֻלָּתוֹ.

יַחֲנֶה בְמִשְׁכָּנוֹ,

יַעֲלֶה לְדוּכָנוֹ;

יֵשֵׁב בְּאַרְמוֹנוֹ,

עָרָיו וּבִנְיָנוֹ

אַרְצוֹ וְקִנְיָנוֹ,

עָרֵי אֲחֻזָּתִי.



מַהֲלַל אֵל – נִפְלְאוֹתָיו; / צוּר – בְּתוֹךְ לִבִּי מְצָאתִיו!

עַל מְרוֹמִים מְשְׁכְּנוֹתָיו – / רַק בְּעֵין שִׂכְלִי חֲזִיתִיו!

וּבְגָבְהֵי מַחֲנוֹתָיו – / אַךְ בְּבֵית קָדְשִׁי רְאִיתִיו!

הֵן לְבָבִי שַׂמְתִּיו כְּנִיחוֹח –

אַךְ יְצוּרִים לְךְ יַעֲלוּ רֵיחַ

נִשְׁחֲטוּ בְּבֵית עוֹ[לָם]

בֵּין דְּבִיר וּבֵין אוּלָם –

לִבִּי לְךָ מִקְדָּשׁ,

יַעְנֶה בְשִׁיר חָדָשׁ,

יִגְדַּל וְיִתְקַדָּשׁ

שִׁמְךָ בְּשִׁירָתִי.
