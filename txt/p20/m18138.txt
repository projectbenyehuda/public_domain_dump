


	
	אם זכרתי עוני(רהוטה)
	






אִם זָכַרְתִּי עֲוֹנִי



אֵאֵָלֵם וְיִפְּלוּ פָנַי.



אַךְ יְשִׁיבוּנִי רַעְיוֹנַי:



אֵל רַחוּם וְחַנּוּן אֲדֹנָי!



בְּשַׁנּוֹתִי טַעְמִי לִמְכּוֹר



בְּיַד הַיֵּצֶר הַשִּׁכּוֹר –



בֶּאֱמֶת, אוֹתִי תַעְכּוֹר.



בְּרֹגֶז רַחֵם תִּזְכּוֹר!



רָעָתִי עָנְתָה בְּפָנָי,



רֹאשִׁי עָבְרוּ עֲוֹנָי,



רַבּוּ מֵחוֹל זְדוֹנָי –



רַחֲמֶיךָ רַבִּים, אֲדֹנָי!



הַשְׁכֵּם אֲקַדְּמֶךָ:



הִצַּלְתַּנִי בְּעִתּוֹת זַעְמֶךָ –



הוֹשִׁיעֵנִי לְמַעַן שְׁמֶךָ,



הֲמוֹן מֵעֶיךָ וְרַחֲמֶיךָ.



מָה אַעְלִים מִמֶּךָ?



מָלֵא כָבוֹד עוֹלָמֶךָ!



מִי אֶשָּׂא אֵימֶיךָ?



מַהֵר יְקַדְּמוּנוּ רַחֲמֶיךָ!



בְּיָדְךָ מַתְכֹּנֶת מְתוֹם –



בְּיָדָם מַעֲשֵׂיהֶם לַחְתֹּם.



בָּחַרְתָּ בְּהוֹלְכֵי תֹם,



בְּךָ יְרֻחַם יָתוֹם.



רְדָפוּנִי פְשָׁעַי כְּשׂוֹטְנָי;



רָצִיתִי לַהֲסִירָם מִפָּנָי –



רַחֲמֶיךָ שַׁתִּי אֲזֵנָי,



רַחוּם וְחַנּוּן אֲדֹנָי!



מִפַּחְדְּךָ לִבִּי יֵחָם,



מַעֲווֹת נְעוּרִים הִנִּיחָם



מַעֲלִי; וּמָעַל וְנִחָם –



וּמוֹדֶה וְעוֹזֵב יְרֻחָם!



אֱמוּנָתְךָ מַשְׂכִּילִים יְבִינוּן,



אֲשֶׁר תּוֹרָתְךָ יַעֲנוּן;



אִמְרָתְךָ לָשִׁיר יְשַׁנֲנוּן,



אֵל רַחוּם וְחַנּוּן.



יוֹסִיף שֵׁנִית לִקְנוֹתֵנוּ,



יֹאמַר דַּי לִמְגִנָּתֵנוּ!



יָדַע יִצְרֵנוּ וֶעֱנוּתֵנוּ –



יְרַחֲמֵנוּ, יִכְבֹּשׁ עֲוֹנֹתֵינוּ.



רָם וְנִשָּׂא הוֹדֶךָ,



רְצֵה בָעֳנִי עֲבָדֶיךָ



רָצִים לְבַקֵּשׁ עָדֶיךָ



רַחֲמֶיךָ, יְיָ, וַחֲסָדֶיךָ.



שִׁלְּמוּ שְׂפָתֵינוּ אֲשָׁמִים,



לְפָנֶיךָ זִבְחֵי שְׁלָמִים.



שְׁכֶם אֶחָד נוֹאֲמִים,



לַייָ אֱלֹהֵינוּ הָרַחֲמִים.



אָנָּא, מֶלֶךְ רַחוּם וְחַנּוּן, / הַרְבֵּה עִמּוֹ פְדוּת –



פְּדֵה עַם נֶהֱרַג עָלֶיךָ / בְּלַחַץ אוֹיֵב וְעַבְדוּת!



כְּלָאוֹ פִשְׁעוֹ בְּמוֹ כֶלֶא / וַיְנַהֲגֵהוּ בִּכְבֵדוּת.



הָשֵׁב שְׁבוּתוֹ, וְיָשׁוּב / לִימֵי הַשַּׁחֲרוּת וְהַיַּלְדוּת!



אֵל מֶלֶךְ יוֹשֵׁב עַל כִּסֵּא רַחֲמִים, מִתְנַהֵג בַּחֲסִידוּת.
