


	
	כִּי שֵׁם יְיָ אֶקְרָא
	



מוסתג׳אב


כִּי שֵׁם יְיָ אֶקְרָא.



אֶקְרָא אֶל לִבִּי לִרְצוֹנֶךָ: / מַה לְּךָ נִרְדָּם בְּאִישׁוֹנֶךָ?



קוּם קְרָא שֵׁם מְעוֹנֶךָ! / וְאָמַר מָה אֶקְרָא?



אֶקְרָא בְּשֵׁם הַנּוֹדָע בִּנְבוּאוֹת, / הוּחַל מִנִּפְלְאוֹתֶיךָ
לִרְאוֹת,



וְדָרְשׁוּ אוֹתְךָ לִצְבָאוֹת – / וְהָיוּ לְךָ לְמִקְרָא.



אֶקְרָא רֹאשׁ בֵּין שְׁנַיִם מִתְנַשֵּׂא / בְּרֹאשׁ, דְּמוּת כִּדְמוּת
עוֹשֶׂה –



וְאֵלָיו הוּא נוֹשֵׂא / אֶת נַפְשׁוֹ וְקָרָא:



אֶקְרָא הַמְּחוֹקֵק צוּרוֹת וְנִמְצָאוּ, / בְּהַמְשֵׁל כּוֹכָבִים
נִפְלָאוּ,



אֲלֵיהֶם הַנּוֹאָלִִים יִקְרָאוּ – / אֲנִי אֵלֶיךָ אֶקְרָא.



אֶקְרָא מִבִּינַת נְשָׁמָה הֶאֱמָרְתָּ, / יְלִידֵי אִשָּׁה בָּהּ
הִפְאָרְתָּ;



גַּם מֵהֶם סְגֻלָּה בָחָרְתָּ – / וְשִׁמְךָ עָלֵינוּ נִקְרָא.



אֶקְרָא קְרִיאָה בִּיחִידָה מְפֻנָּה / לַשֵּׁם, אֲשֶׁר בּוֹ הִיא
מְכֻנָּה,



וְכָל קְרָבַי יִפְצְחוּ רִנָּה / כַּהֲרִימִי קוֹלִי וָאֶקְרָא;



אֶקְרָא: טוֹב טוּבוֹ לֹא יִגָּרָע, / אֶשְׁמֹר חֹק אוֹ אֶפְרָע,



עַל כֵּן בַּטּוֹב וּבָרָע / קוֹלִי אֶל יְיָ אֶקְרָא.



אֶקְרָא: נֶצַח שִׁמְךָ הַנַּעֲלֶה, / שֵׁם מִכָּל שֵׁם יִפָּלֵא.



בְּזִכְרוֹ מִשְּׁאוֹל נַעֲלֶה – / תְּחַיֵּנוּ וּבְשִׁמְךָ נִקְרָא.
