


	
	נִשְׁמַת אֵל מִמְּקוֹר חַיִּים
	



נשמת






נִשְׁמַת / אֵל, מִמְּקוֹר חַיִּים הִיא נִלְקָחָה, / אֱלֹהֶיהָ מַאֲמֶרֶת;



אַחֲרֵי בְלוֹת הֵיכָלָהּ תִּמְצָא מְנוּחָה –



אוֹר זָרֻעַ לַצַּדִּיק וּלְיִשְׁרֵי לֵב שִׂמְחָה.



נִשְׁמַת / בּוֹרְאִי הֱבִינַתְנִי – וַיָּאוֹרוּ עֵינַי, / בְּכָל יוֹם
מְבָרֶכֶת;



בְּעוֹדָהּ בְּאַפִּי בְּדָתְךָ אָשִׂים הֶגְיוֹנָי –



בְּיָדְךָ אַפְקִיד רוּחִי, פָּדִיתָה אוֹתִי, יְיָ.



נִשְׁמַת / רוּחֲךָ הַטּוֹבָה לִמְּדַתְנִי חֻקֶּיךָ, / רָנֵּי תֵת
מְרַנֶּנֶת;



רָאֲתָה תֹּם קְרֵבֶיךָ וְחֹסֶר רְחֵקֶיךָ –



רֹאשׁ דְּבָרְךָ אֱמֶת וּלְעוֹלָם כָּל מִשְׁפַּט צִדְקֶךָ.



נִשְׁמַת / הוֹדְךָ הֶחֱיַתְנִי, בְּמַלֹּאת צוּרִי שָׁרְשִׁי, / הֲדָרְךָ
מְמַלֶּלֶת;



השכל עֲלֵי זִמְרָה תַחְתִּית ישיא –



הֲלֹא מִצְעָר הִיא וּתְחִי נַפְשִׁי.



נִשְׁמַת / מַתְכֹּנֶת גֵּוִי, בְּמָלְכָהּ עָלַי – / מִמַּלְכוּת יָהּ
מָמְלֶכֶת;



מְיַסְּרַתְנִי יוֹם לְיוֹם לְיַשֵּׁר מַעֲגָלָי –



מָה אָשִׁיב לַייָ כָּל תַּגְמוּלוֹהִי עָלָי?



נִשְׁמַת / קָשְׁטְךָ לְכַבֵּד – בָּנֶיךָ לִמַּדְתָּם, / קְדֻשָּׁתְךָ
מְקַדֶּשֶׁת;



קְרָאתַנִי לַעֲמֹד בְּדָתָם הֶעֱמַדְתָּם



קֶדֶם יָדַעְתִּי מֵעֵדֹתֶיךָ, כִּי לְעוֹלָם יְסַדְּתָם.



נִשְׁמַת / טֹהַר כְּבוֹדֶךָ – אוֹתוֹתֶיךָ בְקִרְבָּהּ, / טֻמְאַת הַגֵּו
מְטַהֶרֶת;



טֹרַח תַּאֲוַת תֵּבֵל תָּסִיר מִקִּרְבָּהּ –



טוֹב לִפְנֵי הָאֱלֹהִים יִמָּלֵט מִמֶּנָּה וְחוֹטֵא יִלָּכֶד בָּהּ.



נִשְׁמַת / נָעָמְךָ הִתְאַוְּּתָה דְּרָכֶיךָ לִלְמֹד, / נֶצַח
מְנַצַּחַת;



נוֹדַע כִּי הַכֹּל לִכְבוֹדְךָ יַעֲמֹד –



נִפְלָאִים מַעֲשֶׂיךָ וְנַפְשִׁי יֹדַעַת מְאֹד.



נִשְׁמַת / חַיִּים, מִדֵּי הֱיוֹתָהּ בְּאַפִּי – / זִמְרַת עָזָּהּ
מְזַמֶּרֶת;



קוֹלִי הֲרִימוֹתִי אֶל אֵל וָאֶפְרֹשׂ כַּפִּי –



וְשִׂפְתֵי רְנָנוֹת יְהַלֶּל פִּי.
