


	
	אַיֶּלֶת חֵן מָצְאָה
	



אהבה






אַיֶּלֶת חֵן מָצְאָה / [דִּ]מְ[יוֹ]נָהּ בִּזְמָן רְחוֹבוֹת / לֵיל
סָבְבָה,



בִּצְבִי נַפְשָׁהּ נִקְשְׁרָה / וַתִּלְאֶה לִזְכֹּר בְּמִלָּה / מִי
עִ[צְּ]בָה.



׳בִּי, עָפְרִי, בָּךְ אֶחְשְׁקָה, / חִנָּךְ הִתְעָרֵב בְּלִבִּי,



אִם מֵעֵינָךְ אֶרְחֲקָה – / קָרוֹב תִּמָּצֵא בְקִרְבִּי,



לִדְבַר אוֹיְבָה עָתְקָה / הִסְתַּרְתִּי דּוֹדָךְ בְּחֻבִּי.



בִּשְׁמָךְ לִשְׁמָךְ אֶקְרְאָה, / עָמַדְתִּי עַל אֵם נְתִיבוֹת /
וָאֶחְשְׁבָה –



לִי חֶמְדָּה לֹא נִשְׁאֲרָה / וּבְיָדִי אֵין רַק תְּפִלָּה / כִּי
עָרְבָה.׳



׳רָפוּ מִלַּיִךְ, עֲדֵי / הִנְעַמְתְּ לִבִּי לְאִטֵּךְ,



הָמוּ מֵעַי אֶל יְדֵי / עִיר שָׁלֵם לַיְלָהּ בְּ[שׁוּטֵ]ךְ



וּמְשַׁכְתִּינִי שָׁם בְּדֵי / חִנֵּךְ וָאָרוּץ לְשִׁלְטֵךְ



לַחְיוֹת דּוֹדִים, אֶצְמְאָה / לִנְשִׁיקוֹת פִּיךְ הָעֲרֵבוֹת, / לוּ
קָרְבָה



עֵת דּוֹדִים – אֶתְעוֹרֲרָה, / כִּי כִתְאוֹב חָתָן לְכַלָּה / לָךְ
אֶתְאֲבָה.׳





׳הַיּוֹם קוֹמָתִי כְּבַד / שַׁח, כִּי שָׂבַעְתִּי נְדוּדִים,



כָּבְדָה יָדִי, כִּי כְבַד / אֹזֶן אַתָּה, עַד לְלוֹמְדִים



שָׁאַלְתִּי, אִם לִי לְבַד / זֹאת, אוֹ כִּי חֹק הַיְדִידִים?



אִמְרוּ לִי, מַה נִּבְּאָה / רוּחַ בָּרַת מַחֲשָׁבוֹת? / אִם נִכְתְּבָה



עִתִּי – עַל מָה אֵחֲרָה? / נָא הוֹרוּנִי הַמְּסִלָּה, / אֶתְיַצְּבָה.׳



׳מוֹעֲדֵי יֶשַׁע אֶחֱרוּ / עֵקֶב שֶׁהֶעְוִית נְתִיבֵךְ,



אַל בּוּז אוֹיְבִים, פָּעֲרוּ / פֶּה עָלַיִךְ, יַעֲצִיבֵךְ!



חוֹזֵי קֶדֶם אָמְרוּ: / שׁוּבִי אֵלַי [וַ]אֲשִׁיבֵךְ,



אַהְבָתִי לָךְ נִפְלָאָה / וּבְאֵר תָּקִיר מֵי אֲהָבוֹת / לֹא נִכְזְבָה!



אִם ייסירך [אל] קְרָא / טוֹבָה תוֹכַחַת מְגֻלָּה / מֵאַהֲבָה.׳
