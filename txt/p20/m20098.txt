


	
	אֵל חַי יִקָּרֵא
	






אֵל חַי יִקָּרֵא / הַמְּקָרֶה / בַמַּיִם עֲלִיּוֹת



וּבְיִשְׁעוֹ פוֹדֶה / עַם יוֹרְדֵי / הַיָּם בָּאֳנִיּוֹת.



בַּיָּם שָׂם דֶּרֶךְ / אֶל צֹרֶךְ / עוֹבְרֵי אָרְחוֹת יַמִּים,



בָּאוּ בִּסְפִינוֹת / כִּמְדִינוֹת / בֵּינוֹת גַּלִּים רָמִים,



עַל מַיִם עָמְדוּ / כִּי לָמְדוּ / מִתֵּבַת אִישׁ תָּמִים,



תֵּבוֹת בִּמְכוֹנוֹת / מִשְׁתַּנּוֹת / תַּחְתִּיוֹת וּשְׁנִיּוֹת –



וַאֲנִי מִצָּרָה לִי / אֶקְרָא אֵלִי / מִבּוֹר תַּחְתִּיּוֹת.



רוּחַ אֵל תְּנַחֵם / וּתְנִיחֵם / צִדְקוֹתָיו יַגִּידוּ,



בִּשְׁלֹשָׁה נִסִּים / אוֹת עוֹשִׂים, / יַעֲלוּ גַם יוֹרִידוּ,



וּמַלָּחֵיהֶם / בִּכְלֵיהֶם / הַמַּיִם יָנִידוּ.



בָּרוּךְ מַחְכִּימָם / וּמְשִׂימָם / לִשְׁכֶם אֶחָד, לִהְיוֹת



רַבֵּי הַחוֹבְלִים / בָּהּ מוֹשְׁלִים / כִּנְפָשִׁים בִּגְוִיּוֹת.



הֵן בָּאֳנִי כֹפֶר / הוּא כוֹפֵר / הַחוֹנִים עַל יָדוֹ,



וָאֳנִי קָטָן לוֹ / עַל דִּגְלוֹ – / הַהוּא בַעַל סוֹדוֹ,



וּכְלִי בַרְזֶל בּוֹ, / וּסְבִיבוֹ / יֵלֵךְ עַד יַעֲמִידוֹ;



וַאֲנָשִׁים קָשִׁים / וְשָׁלִשִׁים / עוֹמְדִים בַּשְּׁלִישִׁיּוֹת,



גַּם חִצֵּי רַבִּים / בָּם רַבִּים / עִם חֶרֶב פִּיפִיּוֹת.



מִי יִסְפֹּר מַעֲשָׂיו / אוֹ נִסָּיו? / הוּא רַב עֲלִילִיּוֹת,



הוּא מַפְלִיא סִבּוֹת / בִּמְסִבּוֹת, / הוּא מַגְדִּיל תּוּשִׁיּוֹת,



אֵין חָכְמָה נֶגְדּוֹ; / הוּא לְבַדּוֹ / יֵדַע הָאוֹתִיּוֹת.



אֶל נַחַל עֶדְנוֹ / צוֹפִיּוֹת / כָּל נְפָשוֹת הַחַיּוֹת



וּבְיָדוֹ גֻלּוֹת / עִלִּיּוֹת / גַּם גֻלּוֹת תַּחְתִּיּוֹת.
