


	
	רָזֵי לָיְלָה
	




חֲצוֹת לָיִל. בְּעַד אֶשְׁנַבִּי הַפָּתוּחַ

בַּלָּט אֶל-חֶדְרִי שָׁטְפוּ גַּלֵּי רוּחַ

אִישׁ אַחֲרֵי אָחִיו. חֶרֶשׁ אַט נָשָׁבוּ,

כְּמוֹ מִסּוֹד קְדוֹשִׁים עַתָּה שָׁבוּ.

וּבְעָבְרָם רִפְרְפוּ עָלַי בִּדְמָמָה,

כְּרֶמֶז קַל מֵאֶרֶץ נַעֲלָמָה.

רַק-לַהַב נֵרִי יֶחֱרַד – הֲשֹׁמֵעַ

הוּא לְשׁוֹן כָּל-גַּל, שֶׁכָּכָה יִזְדַּעְזֵעַ?

גַּם-צִלִּי עַל-הַקִּיר, הַצֵּל הַשָׁחֹר,

לִרְגָעִים מְלֹא קוֹמָתוֹ זָע לְאָחוֹר.

מֵאַיִן הֵם הָרוּחוֹת הַפַּלְמוֹנִים,

הַבָּאִים בְּלִי רְשׁוּת, כְּמוֹ אֵין אֲדֹנִים

לְחֶדְרִי? מִי הֵם? הֶחָרָשֵׁי צִירִים

וּמְרַגְּלִים חֶרֶשׁ רֹאִים וְאֵינָם נִרְאִים?

וּמָה, בִּכְלָל, הָרוּחוֹת? מָה חַיֵּיהֶן, טוּבָן?

הַשְּׁאֵלוֹת – שְׁאֵלוֹת הֶבֶל הֵן, כְּמוּבָן –

אַךְ מִי זֶה יַרְשִׁיעֵנִי, מְשׁוֹרֵר עָנִי,

בְּשִׁבְתִּי לַיְלָה בָּדָד אֶל-שֻׁלְחָנִי,

וּלְנֶגְדִּי צֵל מְשֻׁנֶּה מְכַרְכֵּר עַל-הַקִּיר –

אִם-נַפְשִׁי הִתְעוֹרָרָה, דִּבְּרָה שִׁיר?

וּפִתְאֹם צַר הַמָּקוֹם לְנִשְׁמָתִי

מֵהִשְׁתַּטֵּחַ תַּחַת צֵל קֹרָתִי.

וּכְעֶבֶד יִשְׁאַף צֵל וּכְנֶשֶׁר – עָבִים,

כֵּן כָּלְתָה נַפְשִׁי לִסְפֹּר הַכּוֹכָבִים.

הַצֹּרֶךְ גָּדַל גָּדַל: בִּי הִתְעוֹרֵר –

וּבְכָל-עֱזוּז חֻצְפָּתוֹ – חוּשׁ הַמְשׁוֹרֵר.

וָאוֹצִיא בְּעַד אֶשְׁנַבִּי הַפָּתוּחַ

אֶת-רֹאשִׁי חוּצָה לִשְׁאֹף רוּחַ, רוּחַ.

וּלְעֵינַי נִגְלָה שַׂר שֶׁל-לַיְלָה – כְּלוֹמַר:

מַחֲשַׁכִּים, שְׁחוֹר, דּוּמִיָּה, צֵל וְגוֹמֵר.

רַק אֶצְלִי עַל-הָאָרֶץ כְּאַמָּתַיִם

עוֹד רֹעֵד אוֹר גֹּוֵעַ לָעֵינַיִם,

גַּם-כּוֹכָב אֶחָד שָׁם עוֹד יִקְרֹץ עֵינוֹ,

וּשְׁאָר הָעוֹלָם שׁוֹתֵק כְּאִלּוּ אֵינוֹ –

מִפֹּה וָהָלְאָה וּמִפֹּה וָמָעְלָה

רַק לַיְלָה, לַיְלָה, לָיְלָה –

וּכְצֵל עַל-צֵל וּכְצֵל מִקֶּרֶב צֵל

יָשׁוּטוּ צְלָמִים אִלְּמִים בִּתְהוֹם הַלֵּיל.

וּכְמוֹ שְׁתֵּי עֵינַי נֶחְטְפוּ, וּכְאִלּוּ

אֶת-רֹאשִׁי פִּתְאֹם מֵעָלַי הִבְדִּילוּ

בְּקַרְדֹּם חַד וְכָבֵד – כֹּה חִתֵּלָה

כֹּל סְבִיבִי בַּחֲתֻלָּתָהּ הָאֲפֵלָה

וַתְּכַס מִפָּנַי כָּל-הַיְקוּם, אֲפִלּוּ

דְּמוּת בֵּית-הַכְּנֶסֶת עֵינַי לֹא הִצִּילוּ,

זֶה הָעֲנָק הַסֹּכֵךְ בְּקוֹמָתוֹ

הַכְּפוּפָה כָּל-הַבָּתִּים בִּשְׁכֻנָּתוֹ.

גַּם אֶת-עֲצֵי הַסְּרָק מִבֵּית-הָעוֹלָם

הִכַּרְתִּי בַּמַּחֲשַׁכִּים רַק בְּקוֹלָם.

וַהֲבָרוֹת סְתוּמוֹת, רָזֵי לֵיל נִשָּׂאִים,

מִשּׁוּט בָּאַוֵּר, קְלוּטוֹת לִמְקֻטָּעִים,

תְּמַלֵּט, תִּגְּנֹב אָזְנִי הַקַּשָּׁבֶת

מִנִּבְכֵי דְמָמָה וּמִיָּם צַלְמָוֶת,

כִּמְגַנֵּב דְּבָרִים וּמְמַלֵּט נִיב

מִשִּׂפְתֵי יָשֵׁן, חֲלוֹמוֹ דּוֹבֵב פִּיו –

מָה-רַב הָרָז, עֲמֻקָּה הַתַּעֲלוּמָה

הַלּוּטָה בַּאֲפֵלָה זֹה הַסְּתוּמָה!

עַל-יָדִי אַלְפֵי אֶלֶף נַעֲלָמִים

יַעֲרִימוּ סוֹד – וּבְפִיּוֹת נֶאֱלָמִים.

וְרִבְבוֹת עֵינֵי חֹשֶךְ לִי יִלְטֹשׁוּ

וּמְזִמּוֹת שְׁחֹרוֹת כְּמוֹ עָלַי יַחֲרֹשוּ.

מִמּוּלִי שָׁם עֹמֶדֶת חֻרְבָּה מָטָה –

מִי יֵדַע מַה-יִּתְלוֹנֵן קִרְבָּהּ עָתָּה?

וְשָׁם מִבֵּית-הַכְּנֶסֶת תַּבִּיט שְׁמָמָה –

וּמִי יוֹדֵעַ מַה-יֵּעָשֶׂה שָׁמָּה?

וּמַה-יִּתְחָרֵשׁ שָׁם וּמַה-יְּרֻקָּם

בֵּין חֲרֻלֵּי גָדֵר דְּחוּיָה, קִיר מְעֻקָּם?

מַה-יִּרְזְמוּן הַקּוֹלוֹת הַנִּפְלָאִים

הַצֹּלְלִים בַּמֶּרְחָק וְנִבְלָעִים?

הֵד מָה וּמִי הֵם? לָמָּה כֹּה יֶהֱמָיוּ

וּבְעֹצֶב עוֹלָם דּוּמָם אַט יִכְלָיוּ?

אֶת-אַנְחַת מִי יוֹלִיכוּ כַּתַּעֲלוּמָה?

אֶת-תַּאֲוַת מִי, אֶת-תִּקְוַת מִי יוֹרִידוּ דוּמָה?

הַתְּלוּנָה חֲרִישִׁית הִיא אִם תְּחִנָּה זַכָּה

תִּתְמוֹגֵג שָׁם? אִם-כִּלְיוֹן נֶפֶשׁ דַּכָּא

אֲסִירַת חֹשֶךְ, מְלֵאֲתִי מַאֲוַיִּים

תִּתְעַטֵּף, תַּעֲרֹג, תְּפַלֵּל: חַיִּים, חַיִּים!

אוֹ קִינַת סֵתֶר הִיא וּשְׂפוּנֵי דְמָעוֹת

עַל חֻרְבַּן אֵיזֹה אֶרֶץ רַבַּת פְּלָאוֹת,

עֲלֵי מַשּׁוּאוֹת חַיֵּי רִיק וְתָפֵל

הַנֹּבְלִים שָׁם כַּצִּיץ בְּלַיְלָה אָפֵל?

וְדוּמָם יַחְדְּרוּ הַקּוֹלוֹת הַנֶּחְבָּאִים

לַלֵּב – וּמִלְאוּ אוֹתוֹ הֶמְיַת פְּלָאִים.

וּכְמוֹ מִקַּצְוֵי עַד, מֵאַפְסֵי תֹהוּ

הִרְהוּרִים סְתוּמִים אֶל-הַלֵּב יָבֹאוּ,

וּמֹשְׁכִים אוֹתוֹ שָׁמָּה, שָׁמָּה, שָׁמָּה –

לִקְצוֹת מֶרְחַקֵּי עַד, אֶל אַפְסֵי דְמָמָה,

הַרְחֵק מִפֹּה, לִמְקוֹם מִשָּׁם תֵּצֶאנָה

בְּנוֹת קוֹל, הֲבָרוֹת פְּלִיאוֹת – וּתְמוּגֶינָה.

וּכְמוֹהֶן גַּם-בִּלְבָבִי עָמֹק עָמֹק

חֲלוֹם פֶּלִאי נוֹשָׁן נֵעוֹר דּוּמָם, נָמוֹג.

אָז אָבִין אֲנִי אָנָה לִבִּי כֹּה קְרָאָנִי

מֵעוֹדִי עַד-הַיּוֹם – וּמָה עֲרִירִי אָנִי!

וְשַׂר-הַלַּיְלָה בְּקוֹל דְּמָמָה דַקָּה

עֹמֵד עָלַי וְלֹחֵשׁ עַל-הַמַּכָּה.

וּבְכָל-הַיְקוּם מִלְמַעְלָה וּמִתַּחַת

לֹא-תֹאבַד לְבַטָּלָה נְשִׁימָה אַחַת.

עוֹד רָז בֹּלֵעַ רָז בִּמְצוּלַת דְּמָמָה דַקָּה –

וְאָזְנִי כְּבָר הִשְׁלִיכָה שָׁמָּה חַכָּה,

וְתָּצָד דְּמִי וְקוֹל, וּמִמַּעֲמַקִּים

כְּבָר עֵינִי חָתְרָה שִׁפְעַת הַמַּחֲשַׁכִּים,

וַתִּשְׁמֹר רַגְלֵי נֶשֶׁף אַט יִתְהַלָּךְ,

כִּיעָף כְּרוּב קַל וּכְמֻטּוֹת כַּנְפֵי מַלְאָךְ.

וּמְעַט מְעַט נָגֹלּוּ פְנֵי הַלּוֹט

וַיָּקֵא צֵל אֶת-צֵל, סוֹד גִּלָּה סוֹד.

וְקוֹל לַקּוֹל, וְקַו לָקָו, וְצֵל לַצֵּל –

וָאַכִּיר אוֹתְךָ בְּכָל-צְפוּנֶיךָ, שַׂר שֶׁל-לֵיל!

עַל-מֶרְחַב אֶרֶץ אֶשְׁמַע נִיד כְּנָפֶיךָ,

נִשְׁמָתְךָ אָחוּשׁ, אָבִין הִרְהוּרֶיךָ.

מִדְּמִי רְחוֹבוֹת רֵיקִים חִדְלֵי אִישִׁים,

מִשֶּׁקֶט בָּתִּים עוֹמְדִים וּמַחֲרִישִׁים

כְּאִלְּמִים מִקַּרְקָעָם עַד-גַּגּוֹתָם

וְשֹׁמְרִים, סְתוּמֵי חַלּוֹן, תַּעֲלוּמוֹתָם;

מֵאֶשְׁנָב בּוֹדֵד מֵאִיר שָׁם מִפִּנַּת גַּג

בִּקְצֵה הָרְחוֹב וּמְסַנֵּן אוֹר בְּעַד מָסָך דַּק,

מֵאֵד הָאוֹר הַקָּלוּשׁ, אוֹר הָרְמִיָּה,

הַפָּרוּשׂ הַרְחֵק בָּעֲרָפֶל כַּיְרִיעָה

עַל-גַּפֵּי מְרוֹמֵי קֶרֶת שָׁם מִלְמָעְלָה;

מִבַּת-קוֹל תְּרוּעַת חֲצֹצְרָה, גְּנֻבְתִי לָיְלָה,

אֲשֶׁר מִמְקוֹם עַלִּיזִים נָסָה, הִתְפּוֹצָצָה,

וּכְמוֹ מֵרֹאשׁ הַגָּג עָלַי קָפָצָה;

מִתְּנוּמַת הָעֲשָׂבִים, מִמְשַׁק הַקִּמְּשׁוֹנִים,

מִפְּלֵיטֵי קוֹלוֹת אֹבְדִים בַּמֶּרְחַקִּים –

מִכֻּלָּם אָחוּשׁ, אֶשְׁמַע רְמָזִים דַּקִּים

לַחֲלוֹמוֹת פְּלָאִים, חֲלוֹמוֹת בְּלִי פִתְרוֹנִים.


  

תרנ"ט.
