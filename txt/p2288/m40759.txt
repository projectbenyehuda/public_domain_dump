


	
	[אֶרֶץ אֶרֶץ אָרֶץ מַה־תֶּהֱמִי עָלַי] (תּוֹכֵחָה)
	



Josef Carmi

[חי במוֹדָנָה במחצית הראשוֹנה של המאה הי"ז]





תוֹכֵחָה (קֶטַע)


אֶרֶץ אֶרֶץ אָרֶץ מַה־תֶּהֱמִי עָלַי / וְתֹאחֲזִי
בְשׁוּלַי / לְהַסְתִּירֵנִי



בְּסֵתֶר אָהֳלֵךְ / וְלִשְׁכַּב עוֹד אֶצְלֵךְ / אִם תְּרוֹפַפְנָה
עַמּוּדַיִךְ /



וְתִמּוֹטְנָה יְסוֹדוֹתַיִךְ / הֵן בָּךְ יַד הַהֶעְדֵּר הוֹיָה /
וְכוֹסֵךְ רֹאשׁ



רְוָיָה / לֹא יִנָּקֶה כָל בָּךְ נוֹגֵעַ / אִם־חָכָם אִם־מִשְׁתַּגֵּע /
אֲדַמֵּךְ



לְאֵשֶׁת זְנוּנִים / יְקָרָה הִיא מִפְּנִינִים /הַצְּמִידִים עַל־יָדֶיָה



וּצְרוֹר הַמֹּר בֵּין שָׁדֶיהָ / קְוצֻּוֹתֶיהָ בְּזָהָב תַּעְצֹר /
וְשֵׁשׁ



וְאַרְגָּמָן תִּקְשֹׁר / וְהִנֵּה תַּחְתֶּיהָ תַעֲמֹד הַבֶּהָרֶת / נֶגַע
צָרַעַת מַמְאָרֶת.



אֶרֶץ כָּכָה רִמִּיתִינִי / וְהַכְרֵעַ הִכְרַעְתִּינִי / כִּי אַךְ
לַאֲוֵיר הָעוֹלָם



הוּבֵאתִי / מֵאִתָּךְ נִקְרֵאתִי / וַתַּעַבְרִי לְפָנַי / כְּעֶרֶךְ
שָׁנַי /



בְּכֵלִים מִכֵּלִים שׁוֹנִים / וְרוּחַ הַחַיָּה בָּאוֹפַנִּים / הֵן
לִמַּדְתִּינִי



מִנְּעוּרַי / לְבִלְתִּי שְׁמֹעַ לְקוֹל מוֹרַי / עַד מִפִּרְיֵךְ
יַשְׂבִּיעוּנִי /



וּבִנְאוֹת דֶּשֶׁא יִרְעוּנִי /וְכִי נַפְתּוּלֵי אֱלֹהִים נִפְתַּלְתִּי
/ כִּי



בְּבֵית הַמִּדְרָשׁ שֻׁתַּלְתִּי / תִּתְרָאִי לִי בְעָרְמָה /
בְּבִגְדֵי־חֹפֶשׁ



וְרִקְמָה / אָז אֶתְאַו לְאֵפוֹד בַּד וּמְעִיל / וְעַד אֵינֶנּוּ לְפִי
צָמִיד



פָּתִיל / אַחֲרֵי־כֵן בְּאִסְתִּירָא בִלְגִינָא קִישׁ קִישׁ קַרְיָא /
מָעָה



וְדִינָר וּפוּנְדְּיוֹן / סֶלַע דַּרְכּוֹן וְדַרְכְּמוֹן / תָּשׁוּבִי
אֵלַי שְׁלִישִׁיָּה /



וּבִלְעָדָם נֶאֱלַמְתִּי דוּמִיָּה / וְכִי שָׁרְשִׁי פָתוּחַ אֱלֵי־מָיִם
/ יַתְחִיל



לְהָצִיץ מִיּוֹמָיִם / אוֹתוֹ תַשְׁחִיתִי וְתַכְרִיתִי בְהַמְצָאוֹתַיִךְ
/ כִּי



תִקְחִינִי בַּחֲלַקְלַקּוֹתַיִךְ / לָשׁוּט בָּאָרֶץ וּלְהִתְהַלֵּךְ
בָּהּ / לְאָרְכָּהּ



וּלְרָחְבָּהּ / עַל־הֶהָרִים הָרָמִים / וְלִרְכֹּב אֳנִיּוֹת בְּלֵב
יַמִּים /



בַּמִּדְבָּר וּבָעֲרָבָה / מִדַּמֶּשֶׂק עַד־חוֹבָה / וּדְבַר יְיָ /
כְּסֵפֶר



הֶחָתוּם לְעֵינָי / כִּי תֹאמְרִי "מַה־בֶּצַע בְּדָמֶךָ / כִּי
יִמְלְאוּ



יָמֶיךָ / וְשָׁכַבְתָּ עִם־אֲבוֹתֶיךָ / וְכָל־יְקָר לֹא רָאֲתָה עֵינֶךָ
/ הֲלֹא



טוֹב לָךְ / עוֹדְךָ בָּחוּר וָרַךְ / לֶאֱחֹז בָּזֶה וְגַם מִזֶּה /כְּבֹא
מוֹעֵד



צֵאתֶךָ / תִּכָּבֵד וְתֵשֵׁב בְּבֵיתֶךָ / וְתִקַּח מִפִּרְיוֹ וְתֹאכַל"
/ וְכִי



בְאֵלֶּה כַּף רַגְלַי / לֹא נִסּוּ וּמַעְגָּלַי / הֵן תַּרְהִיבֵנִי
בְנַפְשִׁי עֹז /



לֵירֵד אֶל־גִּנַּת אֱגוֹז / רַחַם וְרַחֲמָתַיִם / עֶרֶב וָבֹקֶר
וְצָהֳרַיִם /



וְשָׁמָּה אֶרְוֶה דֹדִים / אֶתְעַלֵּס בָּאֳהָבִים / אוֹ לָלֶכֶת
אֶל־מִשְׁתֵּה



הַיַּיִן / עַד מוֹתָרִי מִן־הַבְּהֵמָה אָיִן / אוֹ לַחֲגֹר חֶרֶב עַל־



חֲלָצַיִם / קַסְדָּא וְשִׁרְיוֹן וּמְגָפַיִּם / אוֹ לִלְבּשׁ כְּתֹנֶת
פַּסִּים /



בְּגֵאָה וְגָאוֹן וַחֲמָסִים / אוֹ בְכִנּוֹר שִׁיר עֲגָבִים הַחְבִּיר /
וְאֵשׁ



זָרָה בִּלְבָבוֹת הַבְעִיר / וְנוֹסַף עַל־אֵלֶּה כַּמֶּה וְכַמָּה /
אֶרֶץ



רַבַּת הַמְּהוּמָה /אֲשֶׁר בָּם גְּדוּד תִּגְדֹדִי / וְנֶפֶשׁ יְקָרָה
תָצוּדִי.



נַפְשִׁי נַפְשִׁי אִם הָאֲרוּרָה הַזֹּאת / לַאֲלָפִים וְלִרְבָבוֹת /
חֲלָלִים



הִפִּילָה / הֵן לֹא נִתְבּוֹנֲנָה וְנַשְׂכִּילָה / הֲנֹאמְרָה שָׁלוֹם /
וְהִנֵּה



חֲלוֹם / הֲלֵב נַקְשֶׁה / וְהַנּוֹשֶׁה יַשֶּׁה / הֲנִיגַע לְהַעֲשִׁיר /
וְהִנֵּה



מָוֶת בַּשִּׁיר / הֲנַעֲלֶה אֵבֶר / וּמָחָר בַּקֶּבֶר / כִּי גַם־בָּנוּ
הִפִּילָה



הַגּוֹרָל / וּשְׁתֵה גַם־אַתָּה וְהֵעָרֵל / הֵן אָמְרָה בִלְבָבָהּ /
וַתֶּאְסֹר



אֶת־רִכְבָּהּ עִם אַף וְחֵמָה / לַעֲרֹךְ מִלְחָמָה / כִּי רָשָׁע רַע



כְּבֹא יוֹם נָפְלוֹ / אֶרֶץ מִתְקוֹמָמָה לוֹ.
