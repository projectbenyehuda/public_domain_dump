


	
	רָנֵּי פַלֵּט: לַַמְּנַצֵּחַ לְהוֹדוֹת מִזְמוֹר שִיר
	



1.


הֵידָד! מַלְכֵּנוּ חַי! וִיחִי עֹוֹד לָנֶצַח!

וַיְפַלְּטֵהוּ יְיָ מִשּוֹד וָרֶצַח,

וַיִּשְׁלַח מַלְאַךְ פָּנָיו וַיַּצִּילֵהוּ.

צַהֲלִי וָרֹנִּי יֹשֶבֶת רַסִּיָּה,

עוֹד תִּפְרְחִי כַזַּיִת וּכְגֶפֶן פּוֹרִיָּה,

בְּצֵל כַּנְפֵי אֲלֶכְּסַנְדֶּר וּבְאוֹר פָּנֵיהוּ;

עוֹד יָנוּב בְּשֵׂיבָה עַל כִּסֵּא מַמְלַכְתּוֹ,

עוֹד תָּגֵן עָלָיו אֵל רַחוּם וּבֵרַכְתּוֹ.




2.


וּבֵרַכְתּוֹ אֵל רַחוּם, כִּי רַבּוּ רַחֲמֵיהוּ,

וּבֵרַכְתּוֹ, אֵל חַנּון, כִּי חַנוּן הִנֵּהוּ,

וּבֵרַכְתּוֹ רַב חֶסֶד, וַחֲסָדָיו מָה רָמוּ!

הֵן מֵאָז שֶׁמֶשׁ מֶמְשַׁלְתּוֹ הוֹפִיעַ,

יוֹם לְיוֹם צִדְקָתו וְרַב טוּבוֹ יַבִּיעַ,

וּבְּגֶשֶם נְדָבוֹת חֲסָדָיו זֹרָמוּ,

חֶסֶד וּצְדָקָה עַל גּוֹי עַל אָדָם יָחַד,

וַיֵּשֶב הָעָם בֶּטַח שָׁלֵו מִפָּחַד.




3.


וַיִּחַר אַף הַשָּׂטָן בִּשְׁאוֹל מִתַּחַת,

שׂנֵא שַׁלְוַת עַמִּים שִׂנְאָה נִצַּחַת,

וַיֹּאמֶר לַשְּׁאוֹל וַתָּקִיא אַנְשֵי רֶשַע,

מְהַרְסֵי שְׁלוֹם הָאָרֶץ בַּחֲמַס יָדָיִם,

אַנְשֵׁי אָוֶן שׁוֹתֵי עַוְלָה כַמָּיִם,

תַּרְבּוּת אֲנָשִׁים חַטָּאִים יַלְדֵי פֶשַׁע,

מִגֵּיא בֶן הִנֹּם מִבֶּטֶן שְׁאוֹל יָצָאוּ,

וּבְאֶרֶץ מְבֹרֶכֶת רוֹש וְלַעֲנָה זָרָעוּ.




4.


אוֹי לָכֶם סָרֵי סוֹרְרִים בְּנֵי בְלִיָּעַל!

לְשׁוֹנְכֶם חֵץ שָׁחוּט אִמְרֵי פִיכֶם רָעַל,

וּבְתוֹעֲבוֹתֵיכֶם הָאָרֶץ תַּחֲנִיפוּ,

בַּחוּרִים וּצְעִירִים שׁוֹלָל תּוֹלִיכוּ,

כִּי לְהָרִים מִכְשׁוֹל לָשׁוֹן תַּאֲרִיכוּ,

וְאַתֶּם הַמַּכְשִׁילִים וְעוֹד סָרָה תּוֹסִיפֹוּ,

שְׁבִילֵי עוֹלָם בְּדַרְכֵיכֶם תַּכְשִׁילוּ,

וּלְתֹהוּ וָבֹהוּ לְהָפְכָם תּוֹאִילוּ.




5.


שָׁוְא לָכֶם טָחֵי תָפֵל וּבֹנֵי חַיִץ,

כִּי הִנֵּה שְׁאוֹל, כְּבִכּוּרָה בְּטֶרֶם קַיִץ,

תִּבְלַעֲכֶם וְאֶת קוֹרֵי עַכָּבִישׁ תֶּאֱרֹגוּ,

שִׁמְכֶם וְזִכְרְכֶם יְכֻסֶּה בִּמְשׁוֹאָה וָאָמֶש,

וְשֵם אֲלֶכְּסַנְדֶּר עוֹד יִגּוֹן עִם שֶׁמֶשׁ,

וְיוֹם הִצִּילוֹ יְיָ־עַמִּים יָחֹגּוּ,

יַעַן לְעַמָּיו אָב רַחוּם הִנֵּהוּ,

וּמֵחַיֵּיהֶם לָהֶם יָקְרוּ חַיֵּיהוּ.




6.


וְאָנֹכִי לְמַלְכִּי אָנֹכִי אָשִׁירָה,

חַסְדֵי יְיָ לְדוֹר אַחֲרוֹן אַזְכִּירָה,

בְּקוּם עַל מַלְכֵּנוּ אִישׁ רַע וְבְלִיַּעַל,

אִישׁ דָּמִים וּמִרְמָה אִישׁ רֹעֶה רוּחַ,

וַיּוֹר חֵץ פִּתְאֹם בְּלֶכְתּוֹ לָשׂוּחַ,

וַיַּרְא אֱלֹהִים וַיַּשֵׁף מִמַּעַל,

וַיִקְרָא: אַל תִּירָא! מָגֵן לְךָ מָעְלָה

וּרְאֵה כִּי הַחֵצִי מִמְּךָ הוּא וָהָלְאָה!




7.


שֵׁן חָרַק הַשָּטָן בִּשְׁאוֹל מִתָּחַת

מִפָּנָיו חָרְדוּ כָּל שֵׁדֵי הַשָּׁחַת,

בַּעַל צְפוֹן בַּעַל זְבוּב וְעַשְׁתְּרוֹת קַרְנָיִם,

מִקִּצְפּוֹ וַחֲרוֹנוֹ אָחוֹר נָסוֹגוּ,

אַמּוֹת סִפֵּי הַשְּׁאוֹל נָעוּ וַיָחוֹגוּ,

וְקוֹל רַעַם אַפּוֹ עָלָה עַד שָׁמָיִם;

וּמַלְאֲכֵי מַעַל שָׂחֲקוּ וַיִּשְׁמָעוּ

“יְחִי אֲלֶכְּסַנְדֶּר!” קָרְאוּ, וְכַף מָחָאוּ.




8.


אֶל עִמְקֵי שְׁאוֹל מַלְאֲכֵי זַעַם הָשְׁלָכוּ,

בּוֹשׁוּ גַּם חַתּוּ צֵידָם לֹא חָרָכוּ,

וּלְחוּצוֹת רַסִּיָה אוֹרָה הָיָתָה;

רִבְבוֹת עָם אֶל בָּתֵּי מִקְדָּש נָהָרוּ,

דִּמְעוֹת גִּיל וְשָׂשׂוֹן כַּמַּיִם נִגָּרוּ;

וְתוֹדַת הָעָם הַשָׁמַיְמָה עָלָתָה.

וַיָרַח יְיָ אֶת רֵיחַ הַנִּיחוֹחַ,

וַיֹּאמַר: לְכוּ קוַֹי הַחֲלִיפוּ כֹחַ!




9.


לָכֵן קֹוֵי יְיָ, שִׁמְעוּ שָׁמֹעַ,

הַחֲלִיפוּ כֹח וִיהְיוּ לְמַלְכְּכֶם זְרוֹעַ!

וּבִעַרְתֶּם הָרָע בַּאֲשֶר תִּמְצְאוּ,

אַל נָא יֵרַךְ לְבַבְכֶם וְנַפְשְׁכֶם לֹא תֵלֶא

מִזַּנְבוֹת הָאוּדִים הָעֲשֵׁנִים הָאֵלֶּה,

יְיָ עִמָכֶם, אַל נָא תִירָאוּ,

וְהִצַלְתֶּם אֶת נַפְשְׁכֶם מִשּׁוֹד וָשֶבֶר,

וְשָלוֹם לָכֶם מִכָל פִּנָּה וָעֵבֶר.




10.


וְאַתָּה יִשְׂרָאֵל, גּוֹי קָו וּבֹטֵחַ,

גּוֹי מַטֶּה לְעֹל שֶׁכֶם טוֹב לֵב וְשָׂמֵחַ,

כָּל יִשְעוֹ כָּל חֶפְצוֹ רַק בֶּטַח שֶׁבֶת;

גּוֹי לֹא יָדַע מֵאָז מֶרֶד וָמֶרִי,

גַּם עִם אוֹיְבָיו לֹא הָלַךְ בַּחֲמַת קֶרִי,

גּוֹי אִיש חַסְדּוֹ כָּל נַפְשׁוֹ אֹהֶבֶת,

גּוֹי אֱהֹב מַלְכּוֹ אֱלֹהָיו יוֹרֶנּוּ,

וַעֲצַת פֹּשְׁעִים וּמֹרְדִים רָחֲקָה מִמֶּנּוּ;




11.


לְךָ עַמִּי יִשְׂרָאֵל מִלַּי אֶעֱרְכָה,

וּבְלֵב נִשְׁבָּר וְנִדְכֶּה מְרִי שִׂיחִי אֶשְׁפֹּכָה,

נִבְהַלְתִּי מֵרְאוֹת נַעֲוֵיתִי מִשְּׁמֹעַ,

כִּי גַם בִּצְעִירֶיךָ נָפְלָה שַׁלְהָבֶת,

עַל זֹאת נַפְשִי תֶאֱבַל תִּלְבַּש עֲצָבֶת,

הַנֶּגַע הַזֶּה, לֹא אֵדַע יָדֹעַ,

אֵיך נִרְאָה בְּבֵיתְךָ, גוֹי אֹהֵב מַלְכֵּהוּ?

וּמֵחֲלָצֶיךּ אֵיךְ פֹּשְעִים יֵצֵאוּ?




12.


אִם גַּם הַפֹּשְׁעִם הָאֵלֶּה מָעָטוּ

וְהֵם גַּם בְּתוֹרַת אֱלֹהֶיךָ בָּעָטוּ,

גַּם אוֹתְךּ יִשְׂנְאוּ וַיִּהְיוּ עָרֶיךָ,

בָּנֶיךָ הֵמָּה עַל שִׁמְךָ נִקְרָאוּ,

לִבְנֵי יִשְׂרָאֵל בַּשַׁעַר נוֹדָעוּ,

אִם גַּם לֹא הִקְשִׁיבוּ תּוֹרַת מוֹרֶיךָ,

וַיָשִׂימוּ עָרְפָּם בַּרְזֶל וּנְחֹשֶׁת,

עָזְבוּ מוּסָרְךָ וַיִּנָּזְרוּ לַבֹּשֶת.




13.


עַל אֵלֶּה הַבָּנִים חֶלְאַת עַמֵּינוּ,

הֵן נִשָׂא חֶרְפָּה וּכְלִמָּה תְכַסֵּנוּ;

אֲבָל לֹא בְכָל זֹאת אֲשֵמִים אֲנַחְנוּ,

אָכֵן רוּחַ הוּא בַזְמָן, רוּחַ שֶׁקֶר,

עֹשֶׂה לָנוּ רָעוֹת גְּדֹלוֹת וְאֵין חֵקֶר,

מֵאָז יִסְעַר לֹא שָלַוְנוּ לֹא נַחְנוּ,

לָאֵל נְקַוֶּה כִּי יִגְאָלֵנוּ מִמֶּנּוּ,

יָמִים יָבֹאוּ וְהָרוּחַ יִשָׂאֶנּוּ.




14.


נוֹדֶה לַיָי חַסְדּוֹ זֹאת הַפָּעַם,

וְאֵלָיו נִתְפַּלֵל כִּי יִכְלֶה הַזָּעַם,

וְשָׁבוּ בָנִים לַאֲבִיהֶם אֲבִי הָאָרֶץ;

גָּדְלוּ תִּפְאַרְתּוֹ יַכִּירוּ יֵדָעוּ,

לְפָנָיו יִשְׁתַּחֲווּ וּבֶרֶךְ יִכְרָעוּ,

יָשׁוּבוּ, וְלֹא עוֹד יְזִידוּן לִפְרֹץ פָּרֶץ;

אַשְרֵי הָאִיש לֹא חָטָא חַף מִפֶּשַׁע

הֹלֵךְ מֵישָרִים וְעוֹזֵב דֶּרֶךְ רֶשַׁע.




15.


וְאַתָּה יְיָ, בָּרֵךְ נָא מַלְכֵּנוּ,

אֲלֶכְּסַנְדֶּר הַשֵּׁנִי הוּא אֲדוֹנֵינוּ,

מִגֶּזַע מַלְכֵי רוּסְיָא חֹטֶר וָנֶצֶר;

בָּרֵךּ מַמְלַכְתּוֹ בַּיָּם וּבַיַּבֶּשֶת,

בָּרְכֵהוּ בּבְּרָכָה בַּתּוֹרָה מְשֻׁלֶּשֶׁת,

עִם אִשְׁתּוֹ הַקִּיסְרִית וּבְנוֹ יֹרֵש עֶצֶר,

לָעַד יָרוּם הוֹדָם וְאוֹרָם זֹרֵחַ,

וְשַלְוַת הַשְׁקֵט סְבִיבָם עַד בְּלִי יָרֵחַ.




16.


כָּל אוֹיְבֵי מַלְכֵּנוּ בֹּשֶׁת יִלְבָּשׁוּ,

הֵם תֻּכּוּ לְרַגְלָיו בְּעָפָר יִתְפַּלָּשׁוּ,

עַל עֵקֶב בָּשְׁתָּם יָשׁוּבוּ יֵבוֹשׁוּ;

עֵינֵי הָעִוְרִים הָאֵלֶּה תָּאוֹרְנָה,

וְנֶאֶסְפוּ יַחַד כְּעָמִיר גֹּרְנָה,

יִכְרְעוּ בֶרֶךְ וְחַסְדֵי הַקֵּיסַר יִדְרֹשׁוּ,

הָרֹה עָמָל וְאָוֶן עוֹד לֹא יֹוסִיפוּ,

יֵדְעוּ יָבִינוּ כִּי טוֹב בְּרָע הֶחֱלִיפוּ,




17.


כָל אֲשֶׁר שִׂפְתוֹתָם תִּטֹּפְנָה נֹפֶת,

וּבְקִרְבָּם עָשׁוּ חֹזֶה עִם שְׁאוֹל וְתֹפֶת,

כָּל בִּרְכַּיִם אֲשֶר כָּרְעוּ לַבָּעַל,

אֶת תּוֹעֲבוֹתֵיהֶם מִלִּבָּם יָסִירוּ,

אֶת יְקָר תִּפְאֶרֶת מַלְכֵּנוּ יַכִּירוּ,

כִּי יְיָ אֱלֹהָיו עִמּוֹ וְיָעַל,

וּבְאַרְצוֹ וּבְעַמּוֹ צְדָקָה יַצְמִיחַ,

וּבְכָל אֲשֶר יִפְנֶה יַשְׂכִּיל יַצְלִיחַ. אמן!
