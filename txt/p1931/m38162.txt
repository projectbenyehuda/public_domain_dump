


	
	וַיָּקָם הַחוֹשֵׁק הַשְּׁלִישִׁי : מקאמה
	







הַחוֹשֵׁק שֶׁהִתְחַפֵּשׂ לְנָזִיר. תוכן המעשה: מעשה בבת מלך יפיפיה, שאביה כולא אותה בארמון כדי למנוע את חושקיה להגיע אליה. חמישה בני מלכים מנסים בתחבולות שונות להיכנס לארמון. הראשון הופיע בדמות מנגן, השני כרוכל בשמים, אך שניהם נכשלו. השלישי התחפש כנזיר, ולהלן הסיפור עליו:






וַיָּקָם הַחוֹשֵׁק הַשְּׁלִישִׁי וַיִלְבַּשׁ בְּגָדָיו הַנֶּחֱמָדִים / וְעַל גְּרוֹנוֹ רְבִידִים / וַיֵּלֶךְ תְּחִלָּה כְּבֶן מֶלֶךְ רוֹכֵב עַל סוּס לָבָן וְהוּא עוֹמֵד בֵּין הַהֲדַסִּים / עִם כְּתֹנֶת הַפַּסִּים / וְתִפְאֶרֶת הָעֲכָסִים / וַיִּתְיַצֵּב מֵרָחוֹק לִרְאוֹת אֶת יְפִי הַבִּנְיָן וְהָעִיר וְחוֹמוֹתֶיהָ / בְּנוּיָה לְתַלְפִּיָּה / בְּאַבְנֵי שַׁיְשָׁה / כּוּחְלָא וּמַרְמְרָא / וַיַּרְא
וְהִנֵּה גַּמָּדִים בְּמִגְדְּלוֹתֶיהָ / וַיִּסְתַּכֵּל בְּאַנְשֵׁי
הַמִּגְדָּל וְהִנֵּה הֵם מְעַט מִזְעָר לֹא כַבִּיר / וַיַּחֲשֹׁב הַחוֹשֵׁק הֱיוֹתוֹ עַל כֹּל אָדוֹן וּגְבִיר / עַל הַהֵיכָל וְהַדְּבִיר / וַיָּשָׁב לְבֵית זְבוּלוֹ / וּלְמִשְׁכַּן אָהֳלוֹ / וַיַּעֲרִים תַּחְבּוּלוֹת וּמוֹעֵצוֹת / עַד כָּלָה וְנֶחֱרָצוֹת / לִהְיוֹת בְּאַחֲרִית הַזַּעַם / נוֹתֵן טַעַם בַּר נוֹתֵן טַעַם / וַיֹּאמַר:
׳עַתָּה אַפְשִׁיט בִּגְדֵי הַחֵשֶׁק וְאֶלְבַּשׁ בִּגְדֵי נְזִירִים אֲרוּכִים וִישָׁנִים / וְכֵלִים מִכֵּלִים שׁוֹנִים / וְאַכְרִיז סִפְרֵי נְזִירוּת / וּבָזֶה אֵצֵא לְחֵרוּת / כִּי אֶהְיֶה מוּחְזָק בְּכַשְׁרוּת / וַיִּשָּׂא מְשָׁלוֹ:







לִבִּי וְעֵינַי קְהַל נֶאֱמָנַי / אָסְפוּ עֲדָתָם וְגָמְרוּ וְנִמְנוּ

סוּתָם לְהַפְשִׁיט וְהֶדֶר וְתַכְשִׁיט / וּלְבוּשׁ הֲדָרָם בְּנָכְרִי יְשַׁנּוּ

לַעְבֹר בְּעָרְמָה בְּסִפְרֵי נְזִירִים / וּבְשׁוּק צְבִיָּה בְּקוֹל עֹז יְתַנּוּ

אוּלַי בְחָכְמָה בְּדַעַת מְזִמָּה / יַעְלַת תְּהִלָּה יְפַתּוּ וְיוֹנוּ

יֵשְׁבוּ בְמָרָה וְיַחְנוּ בְמִתְקָה / וּלְפִי כְּבוֹדָם חֲדָרִים יְמַנּוּ.







וַיִּתְלַבֵּשׁ הַחוֹשֵׁק בִּגְדֵי נָזִיר אֲרוּכִים / שְׁמוּרִים וַעֲרוּכִים / וַיִּתְחַכֵּם לָדַעַת יְדִיעַת הַהֲפָכִים / וַיְגַדֵּל שְׂעָרוֹ וּשְׂעַר הַזָּקָן / וַיִּקַּח סִפְרוֹ כְּדֵי שֶׁיִּסְתַכְּלוּ מַה שֶּׁיֵשׁ בַּקַּנְקַן / וְלֹא יִמְצְאוּהוּ מִכֹּל וְכֹל רֵיקָן / וַיָּשֶׂם הַצָּנִיף הַטָּהוֹר עַל רֹאשׁוֹ / וַיִּתֵּן מִטְפַּחַת סְבִיב גְּרוֹנוֹ לְהַעֲלִים כַּחֲשׁוֹ / וַיַּתְחִיל לָלֶכֶת וּנְעָרָיו עִמּוֹ / וְהוּא נוֹשֵׂא סֵפֶר הַנְּזִירוּת עַל שִׁכְמוֹ /
בְּשַׁנּוֹתוֹ אֶת טַעְמוֹ / וַיֵּלֶךְ תְּחִלָּה בָּעִיר וּגְבוּלוֹתֶיהָ / וַיָּסָב כָּל חוֹמוֹתֶיהָ / פַּעַם וּפַעֲמָיִם / וַיֵּלֶךְ כָּל הַיּוֹם בְּשָׁוֵה קִרְיָתָיִם / כְּדֶרֶךְ יוֹם כֹּה וּכְדֶרֶךְ יוֹם כֹּה וּכְאַמָּתַיִם / לְהוֹדִיעַ לְכָל רוֹאָיו חָכְמַת נְזִירוּתוֹ / וְרֹב
פְּרִישׁוּתוֹ / וְלֹא פְרִיצוּתוֹ / עַד אֲשֶׁר הוּחְזַק לְנָזִיר / בִּהְיוֹתוֹ מַרְאֶה טְלָפָיו כַחֲזִיר / וְכָל זֶה עָשָׂה כָל הַיָּמִים אוּלַי יוּכַל אֲבֵדַת חֲבֵרָיו לְהַחְזִיר.





וַיֵּצֵא שְׁמוֹ בְּכָל גְּבוּלוֹת הָעִיר / כַּגָּדוֹל כַּצָּעִיר / וַיִּשָּׁמַע שָׁמְעוֹ בְּבֵית הַגְּבִירָה / אֲשֶׁר עִם לְבוֹנַת הַנְּזִירוּת קְטוּרָה / וַיַּגִּידוּ לָהּ אֶת כְּבוֹדוֹ וֶעֱזוּזוֹ וְנִפְלְאוֹתָיו / וְשֵׁם סִפְרוֹ הַמְעוֹרֵר לֵב יְשֵׁנִים בְּדִבְרֵי גְבוּרוֹתָיו / וַתְּצַו לְקָרְאוֹ לִרְאוֹת סִפְרֵי חֶמְדָּתוֹ / וְאֶת יְקָר תִּפְאֶרֶת גְּדֻלָּתוֹ / וַיָּבֹא לְבֵיתָהּ וּבַדֶּרֶךְ הֶאֱרִיךְ שׁוּלֵי בְגָדָיו / וַיִּסְחַב עַל הָאָרֶץ מַדָּיו / וַיִּסְרֹק בַּמַּסְרֵק זְקָנוֹ לְהַאֲרִיךְ הַשֵּׂעָר / וַיֵּלֶךְ מִשַּׁעַר לָשַׁעַר / וּמִצַּעַר לָסַעַר / וַיְהִי בְהַגִּיעוֹ עַד הַסֻּלָּם וּבַהֵיכָל נִטְפָּל / מֵאֲרִיכוּת בְּגָדָיו בֵּין רַגְלָיו כָּרַע נָפָל / וַיַּתְחִיל לַעֲלוֹת בְּמַדְרֵגוֹת הַסֻּלָּם / וַיֵּצֵא מִשָּׁם לְקוֹל דְּבָרִים בְּלִי נִשְׁמַע קוֹלָם / כִּי אִם לַנָּזִיר הַמְזֻיָּף בְּמַעֲשֵׂה תַעְתּוּעָיו / וְהַבְלֵי שַׁעֲשׁוּעָיו / וְהַקּוֹל שָׁמַע בְּאָזְנוֹ / בַּעֲלוֹתוֹ בְּמַדְרֵגוֹת גְּאוֹנוֹ / וַיִּשְׁמַע קוֹל רַעַם גָּדוֹל מִמָּקוֹם הַמִּשְׁתַּמֵּר / וְכָזֶה אוֹמֵר:







מִי זֶה אֲשֶׁר עָלָה בְרוּם דַּרְגִּי / יָדוֹ בְּעֵץ סֻלָּם בְּעֹז תָּמְכָה

לָבוֹא בְגַן עֶדְנִי לְלַקֵּט אוֹר / מָרוֹר וְעוֹלָשִׁין וְגַם תָּמְכָה

כִּי בָא דְמוּת שׁוּעָל בְּסוּת לָבִיא / שָׂחַק לְבָבוֹ בוֹ וְהוּא בָכָה

קָרַב לְהִתְעַלֵּף בְּשׁוֹשַׁנַּי / וַיַּעֲבֹר עֵמֶק וְגֵיא בָּכָא

וּבְכֵן לְהַבִּיט זִיו יְפֵפִיָּה / נַפְשׁוֹ בְשֶׁמֶן גִּיל וְשִׁיר סוֹכָה

מַה לּוֹ לְטַמֵּא מִשְׁכְּנוֹת הֵיכָל / יַעְלָה בְרוּחַ חֵן הֲלֹא סוֹכָה

לָכֵן אֲשַׁבֵּר מַדְרְגוֹתַי כִי / לֹא יַעֲלֶה פֶסֶל וּמַסֵּכָה

אֶלְבַּשׁ מְעִיל קִנְאָה לְהַפִּיל אִישׁ / כָּסוּי בְּבִגְדֵי תֹם וּמַסֵּכָה

בַּל יַעֲלֶה אֶל בֵּית חֲצַר מַלְכוּת / רִצְפַּת הֲדַר כָּל גּוֹי וּמַמְלָכָה

כֹּכְבֵי נְבוּכָה לוֹ נְבוֹכִים כִּי / לֹא יַעְזְרוּ גַלְגָּל וּמַעְרָכָה.







וַיִּשְׁמַע הַנָּזִיר אֶת הַדָּבָר הָרָע הַזֶּה וַיִּתְאַבָּל / וַיִּשְׁתּוֹמֵם כְּשָׁעָה אַחַת וַיַּעֲמֹד מְבֻלְבָּל /
כִּי נִסְתַּפֵּק אִם הַשְּׁפָחוֹת הִפְחִידוּהוּ לְבִלְתִּי יַעֲלֶה / אוֹ אִם הַקּוֹל הָיָה מִמַּזַּל טָלֶה / וַיַּעֲמֹד בַּסֻּלָּם תָּלוּי וְעוֹמֵד / וְלֹא עוֹלֶה וְלֹא יוֹרֵד / וְכִמְעַט חוֹם שִׁמְשָׁא אַרֵישׁ דִּיקְלֵי / וַיְהִי כַאֲשֶׁר רָאָה שִׁמְשׁוֹ שׁוֹקֵעַ הָיָה צֹוֵחַ אוֹ
גְּנוֹחֵי אוֹ יְלוֹלֵי / עַד אֲשֶׁר בָּאָת שִׁפְחַת הַגְּבִירָה וַתֹּאמֶר לוֹ: ׳עֲלֵה נָא הַנָּזִיר הַגָּדוֹל / וְאַל תֶּחְדַּל חָדוֹל / בּוֹא מִבְחַר הַפְּרוּשִׁים / וְזֶרַע הַקְּדוֹשִׁים׳ / וְכִשְׁמֹעַ אֶת קוֹל הַשִּׁפְחָה יָדַע כִּי הַקּוֹל אֲשֶׁר שָׁמַע בַּתְּחִלָּה / לֹא הָיָה מִבֵּית גְּבֶרֶת הַתְּהִלָּה / וְעִם כָּל זֶה נִתְבַּהֵל כְּאִישׁ אֲשֶׁר יָשֵׁן / וְאַרְכֻבָּתֵיהּ דָּא לְדָא נָקְשָׁן / וַיַּעַל בַּסֻּלָּם וַיִּתְנוֹעֵעַ / וּבִגְדוֹ הָאָרוֹךְ נֶאֱחַז בְּמַסְמֵר הַמַּדְרֵגָה וַיִּקָרֵעַ / וַיִּתְבַּהֵל מְאֹד מֵהַסִּימָן הָרָע הַזֶּה / וּמִשְׁמַן בְּשָׂרוֹ יֵרָזֶה / וַיִּכָּנֵס בְּנֵס דִּגְלוֹ וְסִפְרוֹ בְּפֶתַח הַיָּצִיעַ / וְהוּא לִפְנֵי כָל רוֹאָיו שַׂק וָאֵפֶר יַצִּיעַ / וַיַּעֲלֶה עוֹד עַד הַהֵיכָל הָעֶלְיוֹן / וְטֶרֶם הִכָּנְסוֹ רָחַץ פָּנָיו בְּנִקָּיוֹן / וַיַּרְא שָׁם מַרְאוֹת אֱלֹהִים בְּכֹחַ הַדִּמְיוֹן / וַיָּבֹא עַד חֲדַר הַצְּבִיָּה / וְהוּא בָנוּי לְתַלְפִּיָּה / וַיִּשְׁתַּחוּ וַיִּקֹּד וּלְנֶגְדָּהּ כַָּרַע / וּלְבָבוֹ כֻּלּוֹ חָנֵף וּמֵרָע / וַתֵּרֶא הַצְּבִיָּה אֶת מַרְאֵהוּ וְאֶת גְּבוֹהַּ קוֹמָתוֹ / וְהַצָּנִיף אֲשֶׁר בְּרֹאשׁוֹ וְהִדּוּר תַּבְנִיתוֹ / וַתֹּאמַר: ׳וַדַּאי זֶה הוּא אֲשֶׁר הַמִּשְׂרָה עַל שִׁכְמוֹ / וְהֵבִיא סֵפֶר הַנְּזִירוּת עִמּוֹ׳ / וַתְּמַהֵר לַהֲקִימוֹ מֵהָאָרֶץ וְהָרִצְפָּה / וַתּוֹשִׁיבֵהוּ בְּכִסֵּא הָאוּלָם וְהַכִּפָּה
/ וַתֹּאמֶר לוֹ: ׳הַאַתָּה הָאָדָם הַגָּדוֹל בָּעֲנָקִים / אֲשֶׁר שָׁמַעְתִּי שִׁמְעֲךָ מִפִּי עוֹלְלִים וְיוֹנְקִים?׳ / וַיְמַהֵר הַמְזֻיָּף לְהָשִׁיב פִּתְגָם / בְּנוֹתֵן טַעַם לִפְגָם / וַיֹּאמֶר לָהּ: ׳אֲנִי עֶבֶד הַנְּזִירִים וְאֶתְאַבֵּק בַּעֲפַר רַגְלֵיהֶם / וְאֶשְׁתֶּה בַצָּמָא אֶת דִּבְרֵיהֶם / וְאִם גְּבִירָתִי עַל רֹאשָׁם
שָׁמַתְנִי / הֵן מִדְרַךְ לְכַף רַגְלֵיהֶם אָנִי׳. – – –
