


	
	ח אֵלֶּה הַחוּקִים
	







סלוק לעצרת לאלעזר






אֵלֶּה הַחֻקִּים / אֲשֶׁר הֵם נִתְּנוּ מִשְּׁחָקִים / בְּקוֹלוֹת וּבְרָקִים

דְּבָרִים עַתִּיקִים / מִדְּבַשׁ מְתוּקִים / בְּשִׁבְעִים לָשׁוֹן נֶחְקָקִים

עַל לוּחוֹת חֲקוּקִים / חַיִּים מְחֻשָּׁקִים / מְלֵאִים וְלֹא רֵיקִים

לִוְיַת חֵן וַעֲנָקִים / מִזַּן אֶל זַן מְפִיקִים / חֵאט וְאַשְׁמָה מְנַקִּים

חֶמְאָה וְחָלָב מֵינִיקִים / זִיו פָּנִים מַבְהִיקִים / יְשָׁרִים וְטוֹבִים וְצַדִּיקִים

קְרוֹבִים וְלֹא רְחוֹקִים / חָלָב לִפְנֵי יוֹנְקִים / דְּבַשׁ לְחֵיךְ דְּבֵקִים

נִשְׁמָעִים מֵחֵךְ מַמְתַקִּים / עֵץ חַיִּים הִיא לַמַּחֲזִיקִים





וּכְדַבְּרוֹ בְפַרְהֶסְיָה מֵרֹאשׁ סִינַי / בְּלָשׁוֹן מִצְרִית אָנֹכִי יְיָ

הָרִים נָזְלוּ מִפְּנֵי יְיָ / כַּרְמֶל וְתָבוֹר וְחֶרְמוֹן נַעֲשׂוּ גְנַיי / כִּי נִתְגָּאוּ לִפְנֵי יְיָ

וְעָלָה וְנִתְעַלָּה הַר סִינַי / וּמֵרָחוֹק סְקָרוּהוּ הֲמוֹנַי / וּבָאֶצְבַּע נָמוּ זֶה סִינַי

וּבְשִׂמְחָה וַיֵּרֶד יְיָ עַל הַר סִינַי / לְגַלּוֹת לֶעָנָו צְפוּנוֹת לִפְנַיי

שִׁשָּׁה בִּכְתָב בְּהֶגֶא עִנְיָנַיי / וְשִׁשָּׁה בְּהִלְכוֹת שִׁנּוּנַיי

זֶה סֵפֶר תּוֹלְדוֹת אָדָם וְכָל אֵיתָנַיי / וִיצִיאַת מִצְרַיִם גְּאֻלַּת בָּנַיי

וְקָרְבָּנוֹת תוֹרַת כֹּהֲנַיי / וְחֹמֶשׁ פִּקּוּדֵי מִנְיָנַיי

וּמִשְׁנֵה תוֹרֵַת יְיָ / וְסֵפֶר עוּצִי יְרֵא יְיָ

לְהוֹדִיעַ סֶכֶם לִוְיָתָן וּבְהֵמוֹת סְעוּדַת מְזֻמָּנַי

עֵרֶךְ זְרָעִים לְלַמֵּד נְבוֹנַיי / וְסֵדֶר מוֹעֵד לְבוֹנֵן תַּחְכְּמוֹנַי

וְסִפּוּק נָשִׁים אֵיךְ לְטַהֵר בְּמַחֲנַי / וּמִשְׁפַּט נְזִיקִים לְהוֹדִיעַ דִּינַי

וְאַזְהָרַת קָדָשִׁים לַשְׁמִיעַ נְבוֹנַיי / וְתוֹרַת טְהָרוֹת וְעִסְקֵי עִנְיָנַי

וְאָז הֶאֱמִינוּ אֱמוּנַי / כִּי הֲלָכָה לְמֹשֶׁה מִסִּינַי

וְנָהֲגוּ לִשְׁנוֹת גְּדוֹלַי וּקְטַנַּי / וּמֹשֶׁה קִבֵּל תּוֹרָה מִסִּינַי

וְאָז קָרַן מַרְאִיתוֹ / כְּמַלְאֲכֵי שָׁרֵת דְּמוּתוֹ / כְּגִבּוֹרֵי כֹחַ גְּבוּרָתוֹ

כְּאֶרֶז גָּבוֹהַּ קוֹמָתוֹ / בַּעֲלוֹתוֹ וִירִידָתוֹ

וְכָל חֲכָמִים לָמְדוּ מֵחָכְמָתוֹ / וְכָל הַנְּבִיאִים מִנְּבוּאָתוֹ

כִּי עַל שְׁלֹשָׁה נָתַן יְחִידָתוֹ / עַל עַמּוֹ וַעֲדָתוֹ

וְעַל מִשְׁפַּט צִדְקָתוֹ / וְעַל חֻקֵּי תוֹרָתוֹ

לָכֵן מְסָרָהּ לִיהוֹשׁוּעַ בְּמוֹתוֹ





כִּי אַלְפַּיִם הָיְתָה בְמַצְפּוּנִים / מִתְעַלֶּסֶת בִּכְנַף רְנָנִים

מְשַׂחֶקֶת בְּחֶבְיוֹן עוֹז מְעוֹנִים / בְּסֵתֶר שַׁדַּי וּבְצֵל רַעֲנַנִּים

וּכְנִתְחַתְּנָה נָזְלוּ גַבְנֻנִּים / וְעַל פִּיהָ הָיוּ נוֹסְעִים וְחוֹנִים

כְּבֻדָּה בַּת מֶלֶךְ יְקָרָה מִפְּנִינִים / כִּבְתוּלָה מֻצְנַעַת לִפְנִים

וּמִפָּנֶיהָ נִקְרַע יַרְדֵּן וְעָבְרוּ בָנִים / וְעוֹשֵׂה קְרָב מְשָׁרֵת בִּמְעוֹנִים

אַף הוּא בְמוֹתוֹ מְסָרָהּ לִזְקֵנִים





כִּי כְאַגַּן הַסַּהַר / מֶזֶג בַּל יֶחְסַר / נִדְמָה בָהּ כָּל נָגִיד וָשָׂר

לָדַעַת חָכְמָה וּמוּסָר / וְאִם הִשְׁכִּים וְהֶעֱרִיב כּוֹחַ יווסר

אֲמָרֶיהָ מַרְפֵּה לְכָל בָּשָׂר / וְהַתָּרָה לְכָל שְׁבוּעַת אִסָּר

וּבְרָכָה לבד לֹא תֶחְסַר / וְכָל חוֹלִי וּמַכְאוֹב מֶנּוּ יוּסַר

אַף זְקֵנִים מְסָרוּהָ לְשׁוֹפְטִים אַרְבָּעָה עָשָׂר





וְעַד לֹא נִמְתָּחוּ יְרִיעוֹת בְּרוּאִים / וּבְטֶרֶם נִתְכּוֹנָנוּ עַמּוּדֵי נִבְרָאִים

חֻקֶּיהָ הָיוּ צְפוּנִים וְנֶחְבָּאִים / אַלְפַּיִם שָׁנָה צְנוּעִים וְנִפְלָאִים

גַּם רֵיקֶיהָ כְּפֶלַח הָרִמּוֹן מִצְווֹת מְלֵאִים / שְׁלֵמִים וּבַכֹּל נָאִים

לַעֲבוֹד עֲבוֹדָתָהּ בְּיִרְאָה בָאִים / וְשׁוֹפְטִים מְסָרוּהָ לִנְבִיאִים




וּבַעֲבוּרָהּ בְּכָבוֹד נִגְלָה / יְיָ בורה לִקְדוֹשִׁים וגלה


בִּיקָר וּבִגְדֻלָּה / בְּרִנָּה וּבְשִׂמְחָה וּבְגִילָה

וְנָם לְזִקְנֵי אוֹם נִדְגָּלָה / וִהְיִיתֶם לִי סְגֻלָּה

אַף הַנְּבִיאִים מְסָרוּהָ לְאַנְשֵׁי כְנֶסֶת הַגְּדוֹלָה





וּלְמַעְלָה גָּהַר כְּאַרְיֵה מִסֻּבְּכוֹ / וַיִּקְרָא כֹּה וָכֹה

אַרְבָּעִים וְאַרְבָּעִים בֶּעָנָן סִכְּכוֹ / וַיַּפְקִידֵהוּ לְצֹאנוֹ הֱיוֹת נְסִיכוֹ

וּמֵרִשְׁפֵּי חֲצוּבֵי אֵשׁ חֲסָכוֹ / אֵשׁ וְיַיִן וְחָלָב וּדְבַשׁ הִמְסִיכוֹ

וַיֵּט עַל חֶמֶד סֻכּוֹ / וּמֵאַנְשֵׁי הַכְּנֶסֶת נִתְּנָה לְאִישׁ סוֹכוֹ





מִמִּדְבָּר נִתְּנָה מַתָּנָה / לִמְשׁוּלֵי כְבִכּוּרֵי תְאֵנָה

קְרוּאֵי עֲדַת מִי מָנָה / חֵלֶק וְנַחֲלָה גּוֹרָל וּמָנָה

שָׁרָה מֵרֹאשׁ אֲמָנָה / עֵדוּת נֶאֱמָנָה / לָמוֹ בְּאַהַב נִתְחַתְּנָה

וּבַעֲבֻרָהּ עָנָיו נִתְעַנָּה / וּפָץ בְּתַחֲנוּן סְלַח נָא / וְאִם אַיִן מְחֵנִי נָא

אִישׁ סוֹכוֹ כְּפָנָה / מְסָרָהּ לְאִישׁ יְרוּשָׁלַיִם וְחָסִיד בִּכְהֻנָּה





וְכָל הַנִּרְאוֹת הָאֵלֶּה בִּשְׁבִילִי / סְבִיבוֹת סִינַי כְּהִגְבִּילִי

בְּקֶצֶף בְּלִי לְחַבְּלִי / לִרְצוֹתִי וְלִשְׁעוֹתִי וּלְקַבְּלִי

בְּשָׁלוֹם וּבְשִׂמְחָה לְהוֹבִילִי / עַל כִּנְפֵי נְשָׁרִים לְסָבְלִי

וְהִיא חָשְׁקָה נַפְשִׁי מִשַּׁחַת בְּלִי / בְּעֵת נִמְסְרָה לְבֶן פְּרַחְיָה מַתַּיי הָאַרְבְּלִי





וּמֵרֹאשׁ הָיְתָה עוֹז מִבְטָח / לֵב בַּעְלָהּ בָּהּ בָּטַח

הִנֵּה אֵל יְשׁוּעָתִי אֶבְטַח / וּבַעֲבוּרָהּ וַיִּשְׁכֹּן יִשְׂרָאֵל בֶּטַח

כְּנִמְסְרָה לְבֶן טָבַיי וּלְשִׁמְעוֹן בֶּן שָׁטַח





כִּי כְּכַלָּה מְסֻתֶּרֶת בְּעוֹז חֶבְיוֹן / יוֹשֶׁבֶת בְּסֵתֶר עֶלְיוֹן



צְפוּנָה מֵרֵאָיוֹן / נִתְּנָה לֶעָנָיו בְּצִוְיוֹן / בשורש וּבְחֵפֶץ וּבְרִצְיוֹן



שָׁפָל מִכֹּל נִתַּן עֶלְיוֹן / וְנִתְעַלְּפוּ גַבְנֻנִּים בְּקֵהָיוֹן

וְנִתְּנוּ לִזְוָעָה וּלְבִזָּיוֹן / כְּנֶחְקָקָה דָת בְּגִלָּיוֹן

לְהִמָּסֵר לִשְׁמַעְיָה וְאַבְטַלְיוֹן





יַעֲלוֹזוּ רֵעִים ואיחים / כְּנִשְׁתַּלְּחוּ בְּפַרְדֵּס שְׁלָחִים

כְּעֵץ מְבַכֵּר לִירָחִים / רֵיחַ טוֹב מְפִיחִים / בְּלֶקַח טוֹב סוֹחֲחִים

שָׂשִׂים וּשְׂמֵחִים / כַּשּׁוֹשָׁן מַפְרִיחִים

וּכְמוֹ קִבְּלוּהָ מֵרֹאשׁ שְׁנֵי אַחִים / כֵּן נִמְסְרָה בַסּוֹף לִשְׁנֵי אַחִים





וְקָרַב יוֹם שִׂמְחָתָהּ לְהִנָּתֵן / לְכַלַּת לְבָנוֹן לְהִתְחַתֵּן

רָעֲשׁוּ דּוֹק וְגֵא וְהַיָּם קוֹל נוֹתֵן / וְנֶאֶסְפוּ עַם אֶל קוֹסֵם בְּלִי לְהִתְמַתֵּן

וְנָמוּ מַה זֶּה בָּעוֹלָם לְהִנָּתֵן / וֶהֱשִׁיבָם יְיָ עוֹז לְעַמּוֹ יִתֵּן





וֶהֱשִׁיבוּהוּ חוֹשְׁבֵי בְלִבָּם וּבְפִיהֶם שָׁלוֹם / כָּל זֹאת עֲבוּר נְתִיבוֹתֶיהָ שָׁלוֹם

מֵאֵת אֲדוֹן הַשָּׁלוֹם / וְעָל בְּלִבּוֹ לְהוֹשִׁיב עַמּוֹ בִּנְוֵה שָׁלוֹם

בְּמֶתֶג וָרֶסֶן עֶדְיֵנוּ לָמָּה יִבְלוֹם / יְיָ יְבָרֵךְ אֶת עַמּוֹ בַשָּׁלוֹם





וְרִבּוֹתַיִם אַלְפֵי שִׁנְאַנֵּי קֹדֶשׁ / יָרְדוּ הַיּוֹם לְסִינַי בַּקֹּדֶשׁ

לְשִׂמְחַת תּוֹרָה הַכְּתוּבָה בִלְשׁוֹן קֹדֶשׁ / בְּמַאֲמַר מַלְכִּי בַקֹּדֶשׁ

וּשְׁנֵים עָשָׂר מִיל נִתְאַחֲרוּ אַנְשֵׁי קֹדֶשׁ





וּכְדַבְּרוֹ אָנֹכִי הֵעִיפוּ נֶפֶשׁ / וְהִטִּיף מִטַּל קְווּצוֹתָיו וְחָזְרָה בָם נֶפֶשׁ

וְדִבְּרוּ לְרוֹעֶה צֹאן קֹדֶשׁ / דַּבֵּר אַתָּה עִמָּנוּ דִּבְּרוֹת קֹדֶשׁ

וְאַל יְדַבֵּר עִמָּנוּ נֶאְדָּר בַּקֹּדֶשׁ / וְנָתְנוּ זֶמֶר וְשִׁלְּשׁוּ קֹדֶשׁ



לַנֶּאְדָּר בַּקֹּדֶשׁ / מַלְכִּי בַקֹּדֶשׁ / נַעֲרָץ בַּקֹּדֶשׁ / כְּסוֹד אֵילֵי קֹדֶשׁ
