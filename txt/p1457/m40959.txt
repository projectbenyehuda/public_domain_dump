


	
	אֲאַדְרָה אֲרוֹמְמָה
	







בשמך רחמן נתחיל קדושת חג השבועות





א


אֲאַדְרָה אֲרוֹמְמָה אֲסַלְּדָה בְחִיל מַנְפִּישִׁי

בְּחָרַנִי בְּרָרַנִי בִּקְּרַנִי וּפִתַּח שָׁרְשִׁי

גְּנָנַנִי גְּאָלַנִי גְּהָצַנִי וְדִלּוּחַ חִבְּשִׁי

גִּלַּנִי דִּדַּנִי דְּרָשַׁנִי וּפָדָה נַפְשִׁי

הֵחִישׁ הִקְרִיב הֶאֱמִין דְּלוֹתִי מֵרִפְשִׁי

וְהִנְחַנִי וְחִזְּקַנִי וְחִבְּקַנִי שָׁבַת וְהִנְפִּישִׁי

זְמַן זֶה זָעְזְעוּ בַּיּוֹם הַשִּׁישִׁי

חֻלְּאוּ חֻתְּתוּ חֻלְחֲלוּ מֵרוּם מַנְפִּישִׁי

טָפוּ טויעו טֻפְטְפוּ גַּם לְהַשְׁלִישִׁי

יוֹרְעוּ יֻקְּעוּ יֻגְּעוּ בְּעֵת קִדְּשִׁי

כְּרָצָה כְּאָמַר כְּבָט דִּינוֹ לְשַׁלְּשִׁי

לְתַכֵּן לְהָכֵן לְהַשְׁכֵּן לְוַעֵד לְהַאֲסֵף לְקַבֵּץ לְצָרֵף לְנַחֵל לִמְשַׁלְּשִׁי / בַּחֹדֶשׁ הַשְּׁלִישִׁי




א1


לֵאלֹהִים / אֹזֶר אֱמוּנָה לַחַשׁ לְקַנְּנָה

לְךָ בִכְתָב שַׁוְעָתָם הַאֲזֵן נָא

עוֹשֵׂי מוֹעֲדֶיךָ מִגּוֹלָה קַבֵּץ נָא

זְכוֹר הֲרוּסָה לְהַגְבִּיהָהּ מוּל מְעוֹנָה

רַחוּם רְצֵה נָא וְגוֹנְנָהּ

בְּצֶדֶק נָם יֻקַּח נָא



ברוך מגן אברהם




ב


מוֹגְגוּ מוֹעֲדוֹ מְאֹד גַּבְנֻנִּים רוֹעֲדִים

מֻצּוּ מֻלְּאוּ מֻדְּדוּ מֻתְּקוּ לְפוּדִים

נִקְּפוּ נִקְּבוּ נִתְיַצְּבוּ מִתּוֹךְ פְּחָדִים

נִתְרַצְּדוּ נָטְפוּ נָמֹשׁוּ מָטוּ רְפוּדִים

סְגוּלִים סָעוּ שָׂמְחוּ נֵזֶר מְאֻפָּדִים

סְדוּרִים סוֹד שֵׂכֶל בְּיוֹפִי נוֹעָדִים

עֲלוּבִים עֲמָמִים עֲבָדִים הָיוּ מְעֻבָּדִים

עֻזָּרוּ עֻוָּדוּ עוֹז מִפַּז נֶחֱמָדִים

פֻּתְּתוּ פֻּלְּשׁוּ פֻּלְּצוּ מִקֶּדֶם מָעֳמָדִים

פּוֹרְחוּ פּוֹרְרוּ פּוֹרְעוּ וְהָיוּ רְעוּדִים



צופתו צָעָצְעוּ צ[תת]ל[…] צֻקְצְקוּ כֵּס מַעְדִּים



צְנוּעִים צְהוּלִים צְעוּקִים הָעֳשׂוּ צְמוּדִים

צָגוּ צֻמְּדוּ צוֹהֲלוּ צֻנְּעוּ צֻלְצְלוּ צֻמְּתוּ לְנָסְעוּ מֵרְפִידִים




ב1


צֶדֶק / בְּרוּכִים הִנְחַלְתָּה דָת קְנוּיָה

יָעַצְתָּה יִדַּעְתָּה יְקָר בְּעַמְּךָ חֲנוּיָה

רָצִיתָה רפיתה רִבִּיתָה אוֹם עֲנוּיָה

בָּעַלְתָּה בֵּרַכְתָּה בָּחַנְתָּה יַחַד זְרוּיָה

יִמַּנְתָּה זַלְתָּה לְהַחֲיוֹתָם בְּצֶדֶק נֶעְקַד בְּנֶטֶף טִלְלֵי תְּחִיָּה



ברוך מחיה המתים




ג


קָנִיתָה קֶדֶם קִנְיַן קְנוּיָה לְהִתְעַלָּה

קִיוּוּךָ קְנוּיִים קוֹרְבוּ קָנִיתָה לְהַעֲלָה

קֻדְּשׁוּ קְבוּעִים קִנְיַן קֶרֶץ בְּמַעֲלָה

קֵירַב קָצִין קְנוּיֵי קוֹבֵץ בְּחֶמְלָה

רָעֲשׁוּ רָעֲדוּ רָגְזוּ רֻגְּזוּ מַעֲלָה

רֻתְּחוּ רוֹפְפוּ רוֹכְבִים רְכוּבִים בַּהֲמֻלָּה

רובלו רועֲלוּ רָעֲדוּ רָחֲפוּ געלה

רֻדְּדוּ רֻצְּצוּ רֻפְּדוּ רָגְשׁוּ מַעֲלָה

שֻׁתְּתוּ שְׁחָקִים שִׁכְשְׁכוּ שִׁפְעַת שְׁפָלָה

שוגלו שֻׁפְּלוּ שָׁמַיִם שִׁבֳּלֵי שָׁעְלָה

שֻׁקְּדוּ שֻׁוּוּ שָׁאוּ שֻׁכְּנוּ עֲרָפֶלָּה

שילות שֹׁרֶשׁ שְׁעִיעַת שָׁקוּד הָעֳלָה

תָּכַף תְּמוּר תּוֹלְדוֹת תִּפְקֹד לְהַעֲלָה

תרס תְּאִירַת תִּנְחֶלֶת תְּעוּדַת תְּעָלָה

תִּרְגַּלְתָּ תְמוּכִים תָּמִים תְּשׁוּרַת רְעוּלָה

תְּרַפְּאֵהוּ תֻּשְׁקְדַת תַּבְלִיתוֹ תֻּרְצְאַת תְּמִידַת תורגל

תֻּחְסְרַת תֻּצְמְדַת תֻּפְקְדַת תֻּפְרְחַת בֶּן עַמְרָם וְאֵלָיו עָלָה




ד


קְבִיעַת קְבִיצַת קְהִילַת קְלִיחַת מַיִם

קוֹפְפוּ קוֹתְתוּ קָפוּ קָפְאוּ מַיִם

יֻזְּפוּ יָגְעֲשׁוּ יָעֲבוּ יועעו מַיִם

יָלְקוּ יָרְעֲשׁוּ יָרְקְעוּ יָבְקְעוּ אֲפִיקֵי מַיִם





לְךָ לְזַמֵַּר לְרַנֵּן לְהַשְׁמִיעַ וּלְגַל

לְהוֹד לְהַבִּיעַ לְהַשְׁפִּיעַ לָזֶה רִגַּל

רוֹעֶה רוּחַ רָצוֹן רָחַשׁ וְסִגַּל

רֻפְּקוּ רוֹעָם רֻתְּקוּ רַעַם נִשְׁמַע בַּגַּלְגַּל





חֲסוּנֶיךָ חֲבוּבֶיךָ חֲסוּיִים חַסֵּן כְּאֵילִים

חֲשַׁקְתָּם חֲפַצְתָּם חוֹפַפְתָּם חִבַּבְתָּם כְּלוּלִים

חֲמַדְתָּם חִסַּנְתָּם חִתַּלְתָּם חֲמוּקֵי מָכְלָלִים

חִזַּקְתָּה חִבַּרְתָּה חִסַּנְתָּה חִסְּנוּךָ כִבְנֵי אֵלִים





זִמְּרוּ זְרוּיִים זְנוּחִים זִכְרֵי מַיִם

זְרִיזִים זִכִּיתָה זֹאת זֻבְּדוּ מַיִם

זָע זוֹעֵם זוֹעֵף זְמִימַת מַיִם

זֶה זֵר זִמֵּן זְרִימַת זוֹרְמוּ מַיִם





קֶדֶם קָנִיתָה קְנוּיַת קֶדֶם בַּכֹּחַ

קַדֵּשׁ קְנֵה קָרֵב קוֹל כֹּחַ

קיוֶיךָ קִיוִים קְנוּיִים קְצוּבַת כֹּחַ

קְנוּיִים קְצוּבִים קֻצְּרוּ קוֹל בְּתִתְּךָ בַכֹּחַ / קוֹל יְיָ בַּכֹּחַ




ה


אֲזַמְּרָה אֶדֶר אָדוֹן אֱלוֹהַּ חֲזִיזִים

לְרוֹמֵם לְסַלֵּד לְנַצֵּחַ וְלָשִׁיר […]יזים

עֶלֶס עֶרֶץ עֶלֶץ ערסות הָרָזִים

עתמה עֻצְּמָה עֹצֶם עֱזוּזוֹ שִׁבֵּר אֲרָזִים




פזמון


אַאְדִּיר לֵאלֹהֵי יַעֲקֹב

לְבִקְעָה שָׂם רְכָסִים וּלְמִישׁוֹר עָקוֹב

עוֹז לְעַמּוֹ יִתֵּן עדן יַעֲקֹב

זֹאת דָּתוֹ הִנְחִיל לְזֶרַע יַעֲקֹב

רָשׁוּם חֵלֶק עַמּוֹ יַעֲקֹב

תּוֹרָה צִוָּה לָנוּ מֹשֶׁה מוֹרָשָׁה קְהִלַּת עַעֲקֹב

עַם קָדוֹשׁ

רְצוּיֶיךָ רחוקיך […]שיך רִשְׁפֵּי אֵשׁ


רֻתַּח רָעַד רָקַד רַדְתָּה בָאֵשׁ

יַחַד יועם יָעֳקַד יה מִלַּהֲבוֹת אֵשׁ




פזמון אאדיר






יָעַצְתָּה יִחַדְתָּה יִמַּנְתָּה יוֹרְדֵי בָר

הִדְרַרְתָּה הֶחְבַּרְתָּה הֶחְלַמְתָּה הִזְמַנְתָּה חיבר

וִעַדְתָּה וְתִכַּנְתָּה וּבָנִיתָה וְעָלָה מִמִּדְבָּר

דָּפַקְתָּה דָּרַשְׁתָּה הִנְחִיתָה הוֹדֵיתָה יָחִיל מִמִּדְבָּר

ו


אוֹצְרוֹת אָמֹן אֲשֶׁר הֵכִין לְבָרְכוֹ

אֲשֶׁר אֹפֶל הִגְבִּיהַּ לְהַעֲרִיכוֹ

אַלְפַּיִם שָׁנָה נִמְתְּקָה בְחִכּוֹ

יְיָ קָנַנִי רֵאשִׁית דַּרְכּוֹ





לְרוֹמֵם לְהַעֲרִיץ לַמִּתְנַשֵּׂא לְכָל רֹאשׁ

לְסַפֵּר גְּדוֹלוֹת מוֹרָא […]לות מֵרֹאשׁ

לִזְכֹּר מְשׁוּלַת נְסוּכָה מֵרֹאשׁ

מֵעוֹלָם נִסַּכְתִּי מֵרֹאשׁ




עצם אֵשׁ בע[…]תו גְּדוֹל הַעֵצוֹת


עָץ וְאִמֵּן וְזִמֵּן שַׁעֲשׁוּעִים לִרְצוֹת

עַמֻּדִים טֶרֶם פָּעַל בְּלִימָה הֵרָתוֹת

עַד לֹא עָשָׂה אֶרֶץ וְחוּצוֹת





זֹאת תּוֹרָה הוּקְרָה זְמַנָּה הָחְקְרָה

זְהוּרָה זִיו זָהֳרָתוֹ הָכְתְּרָה

זִמְּרָה זָרְקָה מוֹרָה וּקְטוֹרָה

זְבוּדָה חֲבוּבָה מִפְּנִינִים יְקָרָה





רוֹצָהּ רוֹמְמָהּ מָדַד מַיִם בְּשׁוֹעַל

רוּם וְתַחַת יָסַד וְעָשׂ וּפָעַל

רְחָבָה נִמְלַךְ בָּהּ וּמְאוּם לֹא הִמְעַל

בְּאַמְּצוֹ שְׁחָקִים מִמַּעַל





בְּכֵן עַל לוּחוֹת שְׁנַיִם הָתְלַלְתִּי

בְּאוֹר פָּנָיו הִלַּכְתִּי וְהָכְלַלְתִּי



בְּבֵיאוּר ב[….]רישה וַחֲקִירָה הֻלַּלְתִּי



בְּאֵין תְּהוֹמוֹת חוֹלַלְתִּי





יַחַד עַד לֹא יְסוֹדִים קְדוּמָה הִיא

יְסוֹדָהּ יְכֻנֶּה קִינִים וָהֶגֶה וָהִי

יָחִיד הֵבִין דַּרְכָּהּ וְאַחֲרִיתָהּ מַה הִיא

תְּהוֹם אָמַר לֹא בִי הִיא





רִנַּת שִׁירָתָהּ מִי יוּכַל לֵידַע

רְשׁוֹם בָּהּ חָזוֹן וְכָל דְּבַר מַדָּע

רִנָּה פָּצְחוּ הָרִים לָזֹאת מוּדָע

רִמְזָהּ עֲמֻקָּה מִשְׁאוֹל מַה תֵּדַע





בָּהּ נִשְׁתַּעְשַׁע וְיָסַד וְהֵכִין רוּמָה

בְּמַאֲמָרָהּ הִרְקִיעַ עַמּוּדֵי בְלִימָה

בָּה נִמְלַךְ חֲצוֹב יְצִיר לַהֶבֶל דָּמָה

בְּרִיתוֹ זָכַר לְקַיֵּם חֶרֶס אֶת חַרְסֵי אֲדָמָה





יְמִינוֹ טִפְּחָה אָחוֹר וָקֶדֶם

יַחַד כְּמוֹ מָגְלָם אָחוֹר וָקֶדֶם

פֵּרַשְׁתָּ פִּלַּשְׁתָּ מִקַּדְמֵי קֶדֶם

יָעַץ וְהֵבִין אַחֲרִיתָהּ אֱלֹהֵי קֶדֶם





קְצִיבַת דִּבְרַת אֲרֻכָּה מֵאֶרֶץ

קְדוּמָה טֶרֶם תְּחִלַּת שָׁמַיִם וָאָרֶץ

קְנוּאָה לְגֵזַע לֶךְ לְךָ אֶל הָאָרֶץ

קְבוּעָה מִנִּי שִׂים אָדָם עֲלֵי אָרֶץ





לְבוּבָה בְחָכְמָה דַּעַת וְעֵצוֹת

לֶקַח כְּתָב שָׁלִישִׁים וּכְתָב מוֹעֵצוֹת

לָקַח עַם יָצָא מִצֹּעַן לוֹעֵט מַצּוֹת

לְפָאֵר לְרוֹמֵם לגדלו הָעֵיצוֹת





רְצוּצִים יֵחָפְשׁוּ מִתְרַפִּים בְּעֵת אָבִיב לְקַדְּשִׁי

רוֹמְמַנִי דִּלַּנִי חֲמִשִּׁים מִמַּכְאוֹת לְחָבְשִׁי



רְסִיסֵי טַל הִטְלִיל וְהִתְלִיל מ[..]יכת אנושי



רֶקַח עָסִיס הִשְׁקַנִי בַּחֹדֶשׁ הַשְּׁלִישִׁי




ובכן ולך תעלה





ח


אוֹתוֹת אֲשֶׁר שִׁלַּשְׁתָּ / בַּאֲשֶׁר בֵּאַרְתָ וּפִלַּסְתָּ / גַּם גְדוֹלוֹת חִלַּשְׁתָּ

דּוֹק וָחֶלֶד הִשְׁלַשְׁתָּ / הֵכַנְתָּ וְהֵיטַבְתָּ וְהֶחְקַרְתָּ / וְרֹאשׁ אוֹתִיּוֹת שִׁלַּשְׁתָּ

זָכוֹר וְשָׁמוֹר אִחַלְתָּ / חוֹק מְשֻׁלָּשׁ […] / טִיעַת שַׁעֲשׁוּעִים פֵּרַשְׁתָּ

יַחַד בְּרָכָה הִשְׁלַשְׁתָּ / כְּמוֹ שְׁלֹשָׁה הִקְדַּשְׁתָּ / לִשְׁלֹשָׁה אוֹתוֹ הִלְמַדְתָּ



מֵאוֹתִיּוֹת מָנְעָם פֵּרַשְׁתָּ / [נ..] נמר הִפְלַשְׁתָּ / סוֹד זֶה הִשְׁלַשְׁתָּ


עֲבוֹדָה וּתְעוּדָה פִּלַּשְׁתָּ / פְּאֵר שלוש [- - - - - - - - - - - - - -]
