


	
	[סדר עולם]
	







ובכן יי קנני ראשית דרכו






אַָגְמֵי קָרִים עַד לֹא הוּרָקוּ

אוֹצְרוֹת שֶׁלֶג עַד לֹא הִבְרִיקוּ

אָז מִזֵּיעַת כִּפָּה זָרִים עַד לֹא נִתְרָקוּ

אֲחִיַּת רִבּוּעַ עַד לֹא נִבְרָקוּ

אֶמֶשׁ שׁוֹאָה וּמְשׁוֹאָה עַד לֹא הֶעֱרִיקוּ

אֶרֶץ עֵיפָתָה עַד לֹא הִבְהִיקוּ

אֶרְאֶלֵּי אוֹפַן עַד לֹא נִבְצָקוּ

אֲפוּדֵי זֶרֶת עַד לֹא נִצָּקוּ

אִלּוּף חֲכָמִים עַד לֹא דֻקְדָּקוּ

אֲחֻזַּת עֹשֶׁר וּנְכָסִים עַד לֹא נֶעְתָּקוּ

אֲמָרוֹת טְהוֹרוֹת יוֹם יוֹם שִׂחֲקוּ

אֲצוּלַת אֲמָנָה בְּשִׁבְעִים לְהַנְחֵל

אֲחוּזַת פִּיפִיּוֹת מִיָּמִין לְהַבְחֵן

אָרְכָּהּ אַרְבַּע גְּבוּלוֹת לְהַנְחֵל

איסוס חִשּׂוּךְ צְפוּנָה לְהָחֵן

אֶתְנַן מִסְּחַר לְהַפְרִישׁ בָּהּ כָּל שׁוֹאֵף וְגוֹחֵן

אֲשֶׁר בָּהּ עָלוּ זָכָר וּנְקוּבָיו לְהִטָּחֵן

אֱמוּנָה אַלְפַּיִם מֵרוּם לְהַשְׁחֵל

אַחַר תְּשַׁע מֵאוֹת וְשִׁבְעִים וְאַרְבָּעָה פִּקּוּד לְהַתְחֵל

אַרְבָּעִים וּשְׁמֹנֶה וּמָאתַיִם עֲשֵׂה וּמְצָא חֵן

אַיֶּלֶת אֲהָבִים וְיַעֲלַת חֵן





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





בְּהַרְרֵי קַדְמוֹנִים בָּט מִשַּׁחֲקוֹ

בְּשִׁלּוּשׁ רָאשֵׁי שׁוֹר תְּהוֹם בְּהַעֲמִיקוֹ

בּוֹכִים מִקֶּרֶת לְתַחַת בְּהַצִּיקוֹ

בְּפָנִים שֹׂחֲקוֹת אַגָּדוֹ בַּהֲפִיקוֹ

בְּתִשְׁעִים וּשְׁמוֹנֶה פָנִים כְּלָל וּפְרָט בְּהַמְתִּיקוֹ

בְּחֶלֶשׁ שְׁמוֹנִים מִצְּרוּפָה בְנַתְּקוֹ

בֵּין טָמֵא לְטָהוֹר בְּאַזְהָרוֹת בְּהַתְנִיקוֹ

בְּהֶגְיוֹן צֶמֶר וּפִשְׁתִּים רִיב בְּשַׁתְּקוֹ

בְּחֻקּוֹ חוּג קִדַּמְתִּי לְשַׂחֲקוֹ

בְּשׂוּמוֹ לַיָּם חֻקּוֹ





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





גָּזַר וְעָרַךְ מֵחַדְרֵי עֲלִיָּה

גָּד דִּקְדּוּקֵי טַעְמֵי הוֹרָיָה

גֹּדֶל תִּפְאַרְתָּךְ בְּפִי כֹל ענייה

גִּיל תָּגִיל בָּהּ מֵרֹאשׁ צְפוּיָה

גְּלוּיָה וַחֲבוּיָה אַלְפַּיִם קְנוּיָה

גְּאוֹן תְּהִלָּתִי עַד אָנָה שְׁאוּיָה

גָּד לִי מַה יִּתְרוֹן וְאֵין צָבָא בִצְבִיָּה

גַּם בַּעֲבוּרָה בָּעוֹלָם מְרִיָּה

גַּדַּע פָּקֵי פְלִילִיָּה

גְּדוֹל הָעֵצָה וְרַב הָעֲלִילִיָּה





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





דַּהֲרַת אֲבָנַיִךְ וַעֲתִידוֹתָם

דּוֹרֵי חֲמֵשׁ מֵאוֹת וְתוֹלְדוֹתָם

דְּרִישַׁת קַל וָחֹמֶר וְגִידָתָם

דִּגּוּל תּוֹרוֹתַיִךְ וְעֵידוֹתָם

דִּבְרֵי נִגּוּנַיִךְ וַאֲגֻדָּתָם

דִּמְיוֹן יְצוּרִים וּפְקֻדָּתָם

דְּבַשׁ וְחָלָב יָמִיצוּ עֲדָתָם

דַּכָּא וְשָׁפָל אֱחוֹז בְּמִדָּתָם

דְפוֹק אַרְבַּע מִדּוֹת בִּלְמִידָתָם

דִּבְרֵי חֲכָמִים וְחִידוֹתָם





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





הַזֹּאת כְּבֻדָּה בַחֶדֶר עַד אָנָה נְהוּגָה

הִיא לִפְנַי לִפְנִים כְּנוּסָה בְּלִי תוּגָה

הֵן כַּעֲרֻמָּה מִבְּלִי לְבוּשׁ צָגָה

הֲלֹא לָהּ סִבַּרְתִּי בְּלִי תַעְגִּין וּתְפוּגָה

הֱיוֹתָהּ צְפוּנָה תוּשִׁיָּה לְמוֹגְגָה

הַמְתִּיק טַעַם לְפֶה לְהַנְשִׁיק עֲרִיגָה

הַמְשֻׁלֶּשֶׁת בְּמִצְוַת חֲגִיגָה

הַבְטָחַת בְּעָלֶיהָ מַסִּיגָה

הִיא יָפָה מַעֲתֶקֶת מִדְּאָגָה



הוֹגֶיהָ בניסיון שיי עוגה



הַנָּוָה וְהַמְעֻנָּגָה





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





וְאִזּוּן וְחִקּוּר אלפים מִתְיַשְּׁבִים

וּבִשְׁתֵּי סַנְהֶדְרָיוֹת מְאַלְּפֶיהָ יוֹשְׁבִים

וּבְעֶשְׂרִים וּשְׁלֹשָׁה בַּקְּטַנָּה יוֹשְׁבִים

וְשִׁבְעִים וְאֶחָד בַּגְּדוֹלָה מִתְחַשְּׁבִים

וּבִשְׁתֵּים עֶשְׂרֵה מִדּוֹת מִשְׁתּוֹבְבִים

וּבִכְרֵיתִי וּפְלֵיתִי מִסְתּוֹבְבִים

וּבְשַׁעַר הַתָּוֶךְ לְחַתֵּךְ הֲלָכָה מִתְנוֹבְבִים

וְקִימָתָהּ וַעֲלִיָּתָהּ נוֹחֲלֶיהָ מְצֻוִּים

וְעָנְתָה אַיֶּלֶת לִסְבוּב שְׁבִיבִים

וְכָל דָּאיְרֵי אַרְעָא כְּלָא חֲשִׁיבִין





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





זָמָה לָאָדוֹן לְפָנָיו לְהַשְׁמִיעַ

זִכְרוֹן כָּל מַעֲשִׂים לְפָנָיו יָדוּעַ

זֶה אֲשֶׁר תִּגְלוֹם זְמַן לוֹ קָבוּעַ

זְבוּד איולתי וּבְאוֹרַח רָבוּעַ

זְנוּחַ הוֹרָיָה וּבִמְצוּדָה תָבוּעַ

זֵכֶר פּוֹעַל מֶנּוּ לִתְבּוֹעַ

זַעַם אִם תַּאֲרִיךְ אוֹתוֹ מִלְּבַלֵּעַ

זִמּוּן יְחִידָה בּוֹ מִלְּקַלֵּעַ

זְדוֹנוֹ חָקוּק בְּעֹנֶשׁ לִפְרוֹעַ

זְבוּבֵי מָוֶת יַבְאִישׁ יַבִּיעַ





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





חָצַב גְּבוּלוֹת בְּעֹנֶשׁ לְהַרְוִיחַ

חֲנוֹת עֵץ חַיִּים בְּזֶמֶר וְשִׁבּוּחַ

חֵטְא אִם גָּרַם לְהַתִּיז לַזּוֹבֵחַ

חוֹזְרִים וְשָׁבִים לְהִנָּצֵל מִטִּבּוּחַ

חִלּוּל לִקְדֻשָּׁתִי בְּלִי תַאֲבוּחַ

חֲמוּדָה מִפַּז בְּתִשְׁעִים וּשְׁמוֹנֶה לְשִׁבּוּחַ

חֲיוֹת בְּעֵץ חַיִּים כִּי הוּא הָשְׁבַּח לְהָפִיחַ

חָשַׁק טָהֳרָה לְטוּמְאָה לְהַגְבִּיחַ

חִסַּד כְּתוּרָה וְחָפַץ מִזַּבֵּחַ

חֲכָם לֵבָב וְאַמִּיץ כֹּחַ





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





טַעְמֵי כָבוֹד בִּישִׁיבַת שֵׁפֶל

טַעְמֵי טָהוֹר בְּבַחַן וָעֹפֶל

טַעַם חָקוּק כִּי יִפֹּל הַנּוֹפֵל

טָרוּד לְהַפִּיל טָחֵי תָפֵל

טָעַנְתִּי תוֹכַחַת לְשַׁמֵּעַ בְּכֵפֶל

טוֹב נִרְאָה בֵּין פָּארָן וָתֹפֶל

טוֹעֶה מִזֹּאת יָשְׁחַל לַשֶּׁפֶל

טוֹב מִמֶּנּוּ הַנָּפֶל





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





יָהּ כְּהִכְרִיז נַעֲשֶׂה אָדָם

יָגְנוּ פְנֵיהֶם פְּנֵי אָדָם

יַחַד כְּהִתְכַנֵּס עָפָר מִמָּקוֹם הַמְקֻדָּם

יִקְּרוֹ וּבִשְׁתִיַּת שָׁלוֹם קָדַם

יָזַם לְהַנְשִׁימוֹ יוֹצֵר אָדָם

יְחָצוֹ חֶצְיוֹ מַיִם וְחֶצְיוֹ דָם

יָצַק בּוֹ תֹאַר עַד לֹא יֵרָדַם

יִדְּעוֹ עַל סֵדֶר זֹאת תּוֹרַת הָאָדָם

יִמֵּן קְדוּמָה מִקַּדְמֵי קֶדֶם קִדֵּם

יוֹדֵעַ מַחְשְׁבוֹת אָדָם





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





כִּבְּדוֹ וְכִלְּלוֹ בִּשְׁלֹשׁ עֶשְׂרֵה לְחַפּוֹתוֹ

כְּהוֹד גַּלְגַּל חַמָּה מִתַּפּוּחָתוֹ

כִּדְמוּת בּוֹרְאוֹ הֱיוֹת דְּמוּתוֹ

כְּשִׁיעוּר קוֹמָה קוֹמָתוֹ

כְּאֵימַת יוֹצְרוֹ עַל כָּל אֵימָתוֹ

כָּמַהּ וְסָדַר לְפָנָיו תּוֹרָתוֹ

כִּי הוּא רֵאשִׁית קְנִיָּתוֹ

וְהוּא תְחִלַּת בְּרִיָּתוֹ

כְּעָרְבָה לָאֻמָּן צְפִיָּתוֹ

כְּחָתָן יוֹצֵא מֵַחֻפָּתוֹ





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





לָקוּט וְקָרוּץ מֵחֹמֶר אֲדָמָה

לְבוּשֵׁי כֹהֵן מַדְווֹ יְהִי נִדְמָה

לְלָבָן וְשָׁחוֹר וְאָדוֹם וְיָרוֹק דָּמָה

לְהַתְחִיל מִצְוָה נִתְקַדְּמָה

לֹא עָמַד בָּהּ וְלַהֶבֶל דָּמָה

לַמִּיתָה חַיָּתוֹ נִרְדְּמָה

לְפַתּוֹתוֹ פֶּתֶן וּפֹתָה וּפְתַיּוּת גָּרְמָה

לָכֵן קוֹמָה גְבוֹהָה בּוֹ הָגְמְמָה

לָזֹאת שִׁמַּע יוֹשֵׁב רוּמָה

לָתֵת לִפְתָאִים עָרְמָה

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





מֵרֹאשׁ צָפִיתִי כֹּחַ לַחֲזוֹת

מְנַחֵם מִמַּעַשׂ מֵעוֹצֵב זֹאת

מִלּוּל מַאֲמָרִי בְּלִי לְהַבְזוֹת

מַטָּע אֲרָזִים נָטַע לְהָעִיד בִּנְפָשׁוֹת פּוֹחֲזוֹת

מַעֲצִיר יְצוּרִים שָׁנָה שְׁלֵמָה בָּזֹאת

מַעְפִּיל לְשׁוֹבֵב מִלַּעַג הַלְּעִיזוֹת

מִתּוֹךְ עֲשָׂרָה נְצוֹר תּוֹרָה זֹאת

מְיֻחָס לְמַפְרֵעַ חֵן בַּעֲלִיזוֹת

מְפַתֶּה הָיָה זֶה לָזֹאת

מֵימֵי נֹחַ זֹאת





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





נוֹאֵם דָּבָר אָמוֹן פָּצָה

נָא בְתַחְבּוּלוֹת חָכְמוֹת יַעֲצָה

נָמָה בַלֵּב וּבְדַעַת עָלְצָה

נֹעַם שִׁירִים בְּשַׁעֲשׁוּעַ נִצְּצָה

נוֹרָא עֲלִילָה לְהַקְדִּישׁ וּלְהַעֲרִיצָה

נוֹרָאוֹת בְּצֶדֶק עָנְתָה בְעֵצָה

נְאוֹת נִינָיו בָּם יָד נִפְרָצָה

נֶאֱרַר זַרְעוֹ וּקְלָלָה נֶחֱרָצָה

נְטֵה אֹזֶן לְדִבְרֵי חֲכָמִים בְּדִיצָה

נִכְנַס יַיִן וְסוֹד יָצָא





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





סַב לְסוֹף עֶשְׂרִים צָץ אִישׁ עֲצָתִי

סֶלָה לְמוּלוֹ שַשְׂתִּי וְעָלַצְתִּי

שָׂרַף פְּסִילִים וְעָלָיו הֵצַצְתִּי

סוֹבְלָיו עָזַב לְהִכָּנֵס בִּמְחִצָּתִי

שְׂרִיפַת כִּבְשָׁן מֶנָּהּ אוֹתוֹ הִצַּלְתִּי

סֻגַּב בַּעֲשָׂרָה אֲשֶׁר אוֹתוֹ צָרַפְתִּי

סֻיַּם בַּבְּרִית אֲשֶׁר אִתּוֹ כָרַתִּי וְנִרְצֵיתִי

שָׂמַחְתִּי בוֹ וְשַׂשְׂתִּי וְדַצְתִּי

סֵפֶר תּוֹרָה מָצָאתִי





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





עָנְתָה אָמוֹן לָאָדוֹן לְעַלְּסוֹ וּלְנַצְּחוֹ

עֶלֶם אֲשֶׁר חֲנַנְתּוֹ בִּכְלוֹת כֹּחוֹ

עֲקָדוֹ עַל עֲצֵי מִזְבְּחוֹ

עָצוּר שְׁלֹשָׁה יָמִים עָשׂ אֶפְרוֹחוֹ

עָרַב וְנִרְצָה נִיחוֹחוֹ

עָצַם וּבְכָל אֶרֶץ הֵפִיחַ רֵיחוֹ

עִנְיַן כְּרַחֵם אָב עַל בָּנִים בְּשָׁכְחוֹ

עֲטִיפַת תְּחִנָּה הָיָה לוֹ לַעֲרוֹךְ בְּשִׂיחוֹ

עַתָּה יָדַעְתִּי שִׁמַּעְתּוֹ לְשַׁבְּחוֹ

עוֹשֵׂה אֶרֶץ בְּכֹחוֹ





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





פְּאֵר נִתְעַטֵּף אֱלֹהִים חַיִּים

פְּרִי יַקִּיר לְמַטָּה לְהוֹסִיף בְּאַרְצוֹת חַיִּים

פָּנָה לִשְׁתּוֹת מִמַּיִם חַיִּים

פּוֹגְשִׁים חֲרוּטִים עִם הַכָּתוּב לְחַיִּים

פֵּתַח דָּבָר לְהָבִין פְּתָיִים

פָּתַח בְּלֵב שָׁלֵם כְּאַחַד הַצְּבָיִים

פֶּצַח צָעֲקוּ לוֹ כִּיתֵי רִבּוֹתַיִים

פְּקוּדָה יְחִידָה קְנוּיָה אַלְפַּיִם

פָּקַד לְחַבֵּר בְּעֵץ הַחַיִּים

פְּרִי צַדִּיק עֵץ חַיִּים





יוֹשֵׁב הָיָה וּמְחַשֵּׁב

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





צוֹפִיָּה הֲלִיכוֹת הֲלָכוֹת פָּרְטָה לְהַמְלִיכוֹ

צָפִיתָ בְּצַעַד נְתִיב הִלּוּכוֹ

צְרוּפָה לִצְרֹף בֶּצַע בְּתָמְכוֹ

צָנַח מִדֶּרֶךְ בְּשַׁעֲשׁוּעַ רַכּוֹ

צָרַח לְקָרֵב רְחֵיקֵי מַלְכּוֹ

צַר שׂוֹנֵא נוֹקֵם וְנוֹטֵר אוֹתוֹ בְּבָרְכוֹ

צָמוּד וְנָקוּב בְּדֶרֶךְ עֶרְכּוֹ

צִדְקַת תָּמִים תְּיַשֵּׁר דַּרְכּוֹ

חָכְמָה וְדֵעָה לְיַשֵּׁב

יְצוּרִים לִבְרֹאת וְעוֹלָם יַשֵּׁב





קוֹשְׁבָה טַעַם הִתְעוֹרְרִי

קְדוֹשָׁה לְקָדוֹשׁ הִתְחַבְּרִי

קוֹל יַעֲקֹב בּוֹ הִתְאַשְּׁרִי

קוֹלוֹ בָּאֹהָלִים כְּמֵלִיץ תָּשׁוּרִי

קָבוּעַ תָּאֳרוֹ בְּכִסֵּא הֲדָרִי

קָראוּי יַעֲקֹב וּמְכֻנֶּה יִשְׂרָאֵל לְאַדְּרִי

קֵץ קָרַב שִׁבְעִים לָשׁוֹן בְּבָאֲרִי

קְדוּמָה מֵאָז קִשְׁבִי דְבָרִי

קוּמִי אוֹרִי

רְמִיָּה לְנֶגְדָּךְ אֵינָהּ שׁוֹלֶמֶת

רֵאשִׁית חָכְמָה יִרְאָתָךְ קוֹדֶמֶת

רַחַשׁ דָּבָר מִמְּךָ לֹא נֶעֱלֶמֶת

רָם בְּמִדָּה בַּכֹּל מַקְדֶּמֶת

רְמִיַּת גְּדָיִים הָיְתָה מַתְאֶמֶת

רָץ שָׂעִיר בֵּן עַז בְּבוּשָׁה נִכְלֶמֶת

רָשׁוּם חֲמִישִׁי חֲנִיטָיו הִשְׁבִּיתוּ בְעֶמֶת

רֵאשִׁית כִּזּוּב הַכּוֹפֵר בָּאֱמֶת

רָם נִקְרֵאתָ אֱלֹהִים אֱמֶת

רֹאשׁ דְּבָרְךָ אֱמֶת

תּוֹעֲפוֹת רְאֵם הִרְבָּה לְעַלְּסֶיהָ

תּוֹחֶלֶת תִּקְוָה שִׁמַּע לנוסיה

תֹּכֶן קוֹל חוֹצֵב אִזֵּן לְעוֹשֶׂיהָ

תָּלָה בִסְעָרָה הֲלָכוֹת לְפַלְּסֶיהָ

תַּלְמוּד לְהַרְבּוֹת עֲשָׂרָה כְנוּסֶיהָ

תַּחֲרוּת וּמָדוֹן תִּכֵּן למעשיה

תִּפְאֶרֶת עֲטֶרֶת לְהַטִּיף עֲסִיסֶיהָ

תָּר לְקֵץ חַיִּים לְיַשֵּׁר רְכָסֶיהָ

תַּחְכְּמוֹנֵי שִׁבְתָּהּ כְּעֵקֶץ עֲכָסֶיהָ

תֹּקֶף יְשׁוּעוֹת בְּרִבּוּעַ כּוֹסֶיהָ

תַעֲצוּם קוֹל בְּאַלְפֵי שִׁנְאָן שָׁת כִּנּוּסֶיהָ

תֻּכָּנוּ רִבּוֹתַיִם בְּחֹסֶן גְּיָסֶיהָ

תֹּאַר פָּמִלְיָה שֶׁלּוֹ בְּעֹשֶׁר נְכָסֶיהָ

תָּפְסוּ כְלֵי הוֹד לְהִסְתּוֹפֵף בְּצֵל חוֹסֶיהָ

תַּקִּיף וְאַדִּיר חָד בִּמְשׂוֹשֶׂיהָ

תְּרִיזַת זָרִים מִתּוֹךְ פַּנִּיקְטֵיהֶן בִּקְלָסֶיהָ

תְּמוּר כִּי קוֹל נֶדֶד בְּהֵמָה וְחַיָּה לֹא נִשְׁמַע בְּבִסּוּסֶיהָ

תָּמְהוּ אֶרְאֶלִּים בִּזְכוּת מְפַרְנְסֶיהָ

תַּנּוֹתוֹ לְצִיר לְהַזְהִיר עוֹמְסֶיהָ

תֹּקֶף חֲרָדָה וּרְעָדָה נִשְׁמַע בְּכָל גֵּיא וַאֲפָסֶיהָ

תָּמִים תִּכֵּן טוֹב כַּמַּטְמוֹנִים מְחַפְּסֶיהָ

תְּנוּ לָהּ מִפְּרִי יָדֶיהָ וִיהַלְלוּהָ בַשְְּׁעָרִים מַעֲשֶׂיהָ





חָדָה אָדוֹן בְּשִׂמְחַת נֹעַם

זָעַק לֶעָנָיו בְּטַעַם

קוּם רֵד הָעֵד בָּעָם

וַיֵּרֶד מֹשֶׁה אֶל הָעָם
