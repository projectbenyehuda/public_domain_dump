


	
	ליום הפורים
	








שְׂאוּ שִׁירִים / בְּיוֹם פּוּרִים,



וּבַרְבּוּרִים / לְאִישׁ אֶבְיוֹן.

שְׂאוּ רִנָּה / בְּכָל־פִּנָּה,

שְׁתוּ, שִׁכְרוּ, / בְּמוֹת הָמָן.



שְׂאוּ זִמְרָה / לְאֵל נוֹרָא;



קְחוּ כִנּוֹר, / קְחוּ עוּגָב.

בְּמַקְהֵלוֹת / תְּנוּ קוֹלוֹת,

תְּנוּ תוֹדָה / לְאֵל נִשְׂגָּב;

אֲשֶׁר פָּדָה / לְעַם־חֶמְדָּה,

וְהוּא מָעֹז, / וְהוּא מִשְׂגָּב,

צְאוּ שָׁרִים, / שְׂאוּ שִׁירִים,

כְּמוֹ דָוִד / כְּמוֹ הֵימָן.





הֲטוֹב מַרְפֶּה, / וְלַחְסָם־פֶּה?



וּמַה־נַּסְתִּיר / גְּבוּרָתוֹ?

הֲלֹא נָגִיל, / בְּרֹן וָגִיל;

הֲלֹא רַבָּה / אֱמוּנָתוֹ.

וְלַחֻפְּשָׁה / כְּבַר נָשָׂא

לְיִשְׂרָאֵל / בְּאֶבְרָתוֹ.

וְגַם נִלְחָם / בְּזֶרַע חָם;

שְׂלָו הֵבִיא, / וְהִמְטִיר מָן.







בְּכָל הַיְּקוּם / הֲלֹא נָקוּם

וְצִדְקוֹתָיו / נְתַנֶּה־נָא.

וְרוֹב חַסְדּוֹ, / וְעֹז יָדוֹ,

הֲלֹא נַגִּיד / בְּכָל פִּנָּה;

הֲלֹא נָרֹן / לְזִכָּרוֹן

בְּמוֹעֵד זֶה / בְּכָל־שָׁנָה.

וּמָה אֶחְסוֹם / לְפִי מַחְסוֹם?

בְּטַח רוּחִי / בְּצוּר רַחְמָן.





וְהוּא יִסְלַח / לְאִישׁ נֶאֱלָח,



וְכָל־חַטָּאת / כְּבֹר יַלְבֵּן;

וְהוּא יוֹרֶה / שְׂפַת יָרֵא,

וְהוּא יָחֹן / פְּתַיוּת בֵּן.

הֲלֹא אֶבְטַח / (וְאַף אִם־טַח

מְאֹד עֵינִי / וְלֹא יָבֵן)

בְּאֵל מַיְשִׁיר / לְדֹבֵר שִׁיר,

וְשָׂם שָׂפָה / לְכָל־נֶאֱמָן.





וּמִשְׁכָּנוֹ / וְאַפַּדְנוֹ

בְּרוּם עוֹלָם / שְׁמֵי שָׁמָיו;

וְשָׁם יַשְׁקִיף / גְּבָר תַּקִיף,

וְחַטָּאתוֹ / וְכָל־מוּמָיו.

וְהוּא יִנְעַל / אֱנוֹשׁ יִפְעַל

אֲשֶׁר לֹא־טוֹב / בְּתוֹךְ עַמָּיו;

וְיַפִּיל צָר / בְּאֵין מַעְצָר

בְּמִכְמַרְתּוֹ אֲשֶׁר טָמָן.





תְּשׁוּאוֹת חֵן/ לְךָ בחֵן



עֲצַת לֵבָב / הֲלֹא נִשָּׂא.

בְּמוֹ גָרוֹן / הֲלֹא נָרֹן

לְבֹרֵא יָם / וְיַבָּשָׁה.

לְךָ נוֹדֶה, / לְךָ פוֹדֶה,

מְאַבֵּד צָר / אֲשֶׁר הִקְשָׁה;

וְלֹא נִשְׁעֶה / לְעָם תֹּעֶה,



וְאֶל־פִּסְלָם, / וְאֶל־צַלְמָן.





אֱלֹהִים רַב, / וּבוֹ מִי־רָב?



וְרַב כֹּחוֹ / וְאֶזְרֹעוֹ.

וְהוּא יִרְדּוֹף, / וְהוּא יֶהְדּוֹף

לְזֵד יָהִיר, / כְּמוֹ פַרְעֹה.

אֲבָל יֶאֱהָב / לְעָם נִכְאָב,

וְיִתְהַלֵּךְ / לְהַרְגִּיעוֹ.

הֲלֹא כָל־אֵל / פְּעֻלּות אֵל,

וְלֹא מַעֲשֵׂה / יְדֵי אָמָן.





אֲחַשְׁוֵרוֹשׁ / לְשָׂר וָרֹאשׁ

עֲמָלֵקִי / כְבָר גֻּדָּל.

וְהוּא שׁוֹבָב / בְּרוּם לֵבָב

מְאֹד קָצַף / בְּעָם נִבְדָּל.

וְרוֹב כַּסְפּוֹ / בְּחֹם אַפּוֹ

אֲזַי נָדַב / לְהַשְׁמִיד דָּל.

וּבוֹכִיָּה / קְהִלַּת־יָהּ;

וְהוֹמִיָּה. / כְּשׁוֹד שַׁלְמָן.





וְכָל־הָעָם / כְּאִישׁ נִפְעָם,



וּכָאָבֵל / חֲפוּי קָדְקֹד.

וְקוֹל קִינָה / בְּכָל פִּנָּה;

וְעֵת לִבְכּוֹת, / וְלֹא לִרְקוֹד.

וְעִיר מַלְכוּת / מְלוֹן בָכוּת,

וְכָל־אָדָם / בְּצוֹם יִקֹּד.

וְיוֹם וָלֵיל / בְּכִי וִילֵל,

וְיִשְׂרָאֵל / כְּאִישׁ אַלְמָן.





וְיִתְעַטֵּף / וְשָׂב וָטָף,

וְהֶחָתָן, / וְהַכַּלָּה –

מְאֹד יִזְחַל, / וְיִתְחַלְחַל,

בּלֵב נָמֵס / וּבֶהָלָה.

וְהַמַלְכָּה, / כְּאִישׁ נִדְכָּא,

בְּמר תִּבְכֶּה / בְהֵיכָלָהּ.

וְקָם שָׁאוֹן, / וְתִמָּהוֹן,

וְהַשַּׁמָּה / מְקוֹם מִשְׁמָן.





אֱלֹהִים חַי, / חֲזוּ אחַי,



אֲשֶׁר עָשָׂה / לְיִשְׂרָאֵל.

גְּבוּרוֹתָיו / וְאוֹתוֹתָיו,

אֲשֶׁר הִפְלָא / בְּאוֹן וָאֵל.

יְמִין תָּקְפוֹ / רְאוּ אֵפוֹא;

זְכוֹר עַמִּי / וְהִבָּהֵל.

זְכוֹר צִדְקוֹ / בְּעַם־חֶלְקוֹ

בְּכָל־שָׁעָה / וְעֵת וּזְמָן.





וּמַה־פָּעַל / אֱלֹהֵי עָל



וְאֵיךְ מִלֵּט / יְדִיד נַפְשׁוֹ;

וְאֵיךְ הִבְדִּיל, / וְאֵיךְ הִגְדִּיל,

וְאֵיךְ רוֹמֵם / בְּחִיר קָדְשוֹ;

וְאֵיךְ הֵחֵל / בְּבֶן־רָחֵל,

בְּהָעִיר שָׂר / עֲלֵי עַרְשׂוֹ;

וְאֵיךְ כִּבֵּד / לְאִישׁ אֹבֵד,

וְאֵיךְ נֶהְפַּךְ / לְאִישׁ נִסְמָן.





אֱנוֹשׁ צוֹרֵר, / וּמִשְתָּרֵר,



וּבַר־לֵבָב / בְּלֵב יִשְׂטוֹן,

לְהַכְרִיתוֹ / וְלִתְלוֹתוֹ.

לְדַבֵּר בָּא / לְאִישׁ שִׁלְטוֹן;

וְהַסֻּלָּם / כְּבַר נִשְׁלָם,

וְעֵץ נִשְׂגָּב / כְּהָר קָטוֹן;

וְהַתָּלוּי / לְאֵל גָּלוּי,

וְלֹא הוּכַן, / וְלֹא זֻמָּן.







כְּצֵאת חַמָּה, / מְלֵא חֵמָה



מְמַהֵר בָּא / לְבֵית הַשָּׂר;

וְתוֹךְ קִרְבּוֹ, / בְּרוּם לִבּוֹ,

אֲנִי(אָמַר) / מְאֹד אֻשָּׁר;

וְאָנֹכִי / לְמוֹ מַלְכִּי

מְאֹד אָהוּב, / מְאֹד נִקְשָׁר;



וּמִי־טָמֵא / אֲשֶׁר־יִדְמֶה



לְשַׂר־שָׂרִים, / לְאִישׁ חַשְׁמָן?





אֱוִיל נִמְהָר, / וְלֹא נִזְהָר,



בְּרֹאשׁ נִשָּׂא / וְעֵין גַּבְהוּת,

הֲלֹא כִי־כֵן / כְּאִישׁ מִסְכֵּן

בְּפִיו נוֹקַשׁ / בְּרוֹב גֵּאוּת.

אֲזַי שָׁפָל / אֲזַי נָפָל;

וּבָא שׁוֹדוֹ / וְאֵין רִפְאוּת.

כְּצַלְמוּנָע / אֲזַי נִכְנָע;

כְּמוֹ פַרְעֹה / כְּמוֹ נַעֲמָן.





אֲזַי צֻוָּה / (רְאוּ הוָֹה!)



לְהַלְבִּישׁ אִישׁ / אֲשֶׁר שָׂנֵא;

בְּרַכָּבוֹ / לְהַרְכִּיבוֹ,

בְּסוּס נִכְבָּד / כְּשָׂר מִשְׁנֶה;



וְעַל־כָּרְחוֹ / בְּכָל־רוּחוֹ



קְרֹא אַבְרֵךְ / לְאִישׁ נַעֲנֶה.

אֱנוֹשׁ נָבָל / אֲזַי נֶאֱבָל,

וְשָׁת לִבּוֹ / אֱלֵי סִימָן.





אֲבָל אֶבְלוֹ / אֱמֶת כֻּלּוֹ,



וְלֹא אַחֵר / אֲשֶׁר־יָגוֹר.

הֲלֹא לָאֵל / וְאוֹן וְאֵל,

הֲלֹא אֵלָיו / חֲנִית וּסְגוֹר.

וְהוּא אִוָהּ, / וְהוּא צִוָּהּ;

וּמִי יִפְתַּח / וְהוּא יִסְגּוֹר?

וּמִי יַחְמס, / וּמִי יִרְמֹס

יְרֵא אֵל חַי / בְּלֵב נֶאֱמָן?





בְּעֵת־מִשְׁתֶּה, / וְהוּא שֹׁתֶה



עֲלֵי שֻׁלְחָן / אֲחַשְׁוֵרושׁ;

וּמִתְפַּנֵּק / בְּכוֹס חָנֵק,

וּמִתְעַנֵּג / בְּמַשְׁקֵה־רוֹשׁ;

וְלֹא חָכְמָה / וְרַק אַשְׁמָה,

וְרַק זִמָּה / לְבַד יִדְרוֹש;

וּבֹזֶה יָהּ / וְתוּשִׁיָּה;

כְּדֶרֶךְ אִישׁ / אֲשֶׁר שָׁמָן.





וְלֹא יָדָע / אֲשֶׁר נִגְדָּע,



וְלֹא הֵבִין / אֲשֶׁר־הֻכָּה.

אֲזַי נִגְרָז / וְנִגְלָה רָז;

וְאֵין מַחְסֶה, / וְאֵין סֻכָּה.

וְהַנָּקָם / אֲזַי הוּקָם,

וְלֹא רֵיקָם / דְּבַר מַלְכָּה.

וּבָא קִצּוֹ / עֲלֵי עֵצוֹ,

אֲשֶׁר־הֵכִין / תְּמוֹל וַיְּמָן.





וְרָשׁ וָדָל / אֲזַי גָּדָל,



כְּמוֹ הָמָן / וְעוֹד יוֹתֵר;



וְכָל־חֵילוֹ / וְכָל־גֻּדְלוֹ,



הֲלֹא כֻלּוֹ / לְאַב אֶסְתֵּר.

וְהוּא פָלִיט, / וְהוּא שַׁלִּיט,

וְהוּא מוֹשֵׁל, / וְהוּא שֹׁטֵר.

בְּאֵין מַצָּה / אֲזַי יָצָא

בְּתַכְרִיךְ בּוּץ / וְאַרְגָּמָן.





לְאֵל אֵלִים, / בְּמַקְהֵלִים,



וּכִנְבָלִים / הֲלֹא נִפְצָח.

וְנִתְכּוֹנֵן, / וְנִתְחַנֵּן,

אֲשֶׁר יָבֹא / בְּרוּחַ צָח;

וְיֵרֶא רָשׁ / אֲשֶׁר־גֹּרָשׁ,



וְגַם־נִגְזַל, / וְגַם־נִרְצָח.



וְאֶל־מִזְרָח / תְּנִי יִצְרָח,



וְאֶל־צָפוֹן, / וְאֶל־תֵּימָן.



אֱסוֹף אַחַי / וְנִדָּחָי,



וְיִשְׂרָאֵל / קְבוֹץ עַל־יָד.

תְּנָה רִשְׁיוֹן / לְעַם־צִיּוֹן,

וְיִבֶן עִיר, / וְיַצִּיב יָד.

וְיִשְׁבָּר־מוֹט, / וְלֹא יִמּוֹט,

וְלֹא יִירָא / פְנֵי צַיָּד.

וּמֵאַרְבֶּה / מְאֹד יִרְבֶּה,

וְלֹא יִמָּד, / וְלֹא יִמָּן.





וְעַד־עֵילוֹם / יְהִי שָׁלוֹם,



וְרַק שִׁירִים / וְקוֹל זִמְרָה.

וְגָר נָמֵר / בְּאֵין שֹׁמֵר

בְּבֵית הַשּׁוֹר / וְהַפָּרָה.

וְהַצָּפִיר / מְאֹד יַחְפִּיר,

וְרַב יַעֲבֹר / לְעָם נִבְרָא.

וּמָלַךְ שָׂר / בְּכָל־בָּשָׂר,

וּמֶמְשַׁלְתּוֹ / בְּכָל־מִכְמָן.





יוּבַל שַׁי / לְבֶן־יִשַׁי,



אֲשֶׁר יִמְלֹךְ / בְּעִיר צִיּוֹן.

וְלֹא יִמְעָד, / וְלֹא יִרְעָד;

לְעוֹלָמִים / יְהִי עֶלְיוֹן.

שְׂאוּ שִׁירִים / בְיוֹם פּוּרִים,

וּבַרְבּוּרִים / לְאִישׁ אֶבְיוֹן.

שְׂאוּ רִנָּה / בְּכָל־פִּנָּה,

שְׁתוּ,שִׁכְרוּ, / בְּמוֹת הָמָן.
