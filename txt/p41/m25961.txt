כוס זהב בידי ד'.

(תלים)





הגביע

אנחש בגביע (בראשית)

א


אֶרְחָמְךָ כּוֹס אֶרְחָמְךָ גָבִיעַ

בָּךְ אֶחֱזָה אוֹרִי וְיִשְׁעִי

עֵת פִּיךָ לְפִּי אַסִיעַ

אָקִיא מִבִּטְנִי חֶטְאִי וּפִּשְׁעִי


ב


אָקִיא מִבִּטְנִי חַטאוֹת נְעוּרִים

תַּאֲוָה נִשְׁכָּחָה וְאַהֲבָה נִסְתֶּרֶת

בִּתִהוֹמךָ גָבִיעַ בַּכּוֹס תַּמְרוּרִים

יַחַד יִצְלָלוּ שָׁם כָּעוֹפֶרֶת


ג


אֲנַחֵשׁ בַּגָבִיעַ וְאָבוֹז לִגְאוֹנִים

הַדוֹרְכִים עֲלי אֶרֶץ בְּרֶגֶל גַאֲוָה

חֲיִים אֶמְצָא גַם בַּכּוֹס יְגוֹנִים

אֶשְׁתֶּה – וְנַפְשִׁי תִּגְרוֹס לְתַּאֲוָה


ד


יִשְׁתּוּ יֵין נְסִיכָם מָסָכוּ

תִּתְהוֹלֵל לְשִׁכָּרוֹן נַפְשָׁם הַזוֹלֵלָה

יֹאכְלוּ לְמַעֲדַנִים בַּשֻלְחָן עָרָכוּ

וְאֲנִי אֶשְׁתֶּה מִכּוֹס הַתַּרְעֵלָה


ה


יֶהֱמוּ הַמְלָכִים הָרוֹזְנִים וְחַשְׁמַנִים

יַחַד יִכְלָיוּן יִהְיוּ לְבָעֵר

בַּצֵל צַלְמָוֶת יִשְׁכְּנוּ בָּאַשְׁמַנִים

וְאַנִי וְהַתִּקְוָה לָנֶצַח נִשָׁאֵר




לֵ – אֹמָר





א


אֶשְׁתֶּה – לֹא אֵדַע מַדוּעַ?

לְהָסִיר יֵיני מִבִּלְתִּי יְכֹלֶת

אֶשְׁתֶּה – וְשִׁירִי אָז אַשְׁמִיעַ

כָּל הִגָיוֹן עִם רְסִיס בַּמִשְׁקֹלֶת.


ב


אֶשְׁתֶּה – וְרוּחִי עָל תַּרְקִיע

לִמְקוֹם הַנֵצח מִבְּלִי חֶלֶף

מַה זֶה הַקֶסֶם בַּגָבִיעַ

רוּחַ חָדָשׁ עִם כָּל דֶלֶף?


ג


אוּלַי זֶה הַכּוֹס הַגָבִיעַ

מִמֶנוֹ הַיַּיִן אֶשְׁתֶּה לִרְוָיָה

קֶדֶם מְשׁוֹרֵר לְרוּחֲךָ יְצוּעַ

רֹאשְׁךָ – אָז לֶדֶם הָיִה




ד1


לֶעָפָר שָבָה אָז גִלְגָלְתֶּךָ

בִּיְדֵי הַיוֹצֵר עַל הָאַבְנָיִם

לָאֵפֶר הָפַךְ אֵדֶר גְויָתֶךָ

אַךְ לֹא רוּחֲךָ רוּחַ שָׁמָיִם.


ה


לָכֵן עֵת אֲחַבֵּק הַגָבִיעַ

הַכּוֹס אֶמְצֶה בִּנְשִקוּת שְׂפָתַיִם

רוּחֲךָ אָז עָלַי תוֹפִיעַ

הֲגיגָךְ אָבין כְּמוֹ בַּחַיִּים.


ו


אוּלַי עֵת אשׁוּב לָעֲפָרִי



יַד הַיוֹצֵר תַּעֲשֶׂה2 עַל הָאַבְנָיִם



כּוֹס וְגָּבִיעַ מֵראשׁ בְּשָׂרִי

מֶנוֹ יַּיִן יִשְׁתּוּ הַחַיִּים.


ז


אַנִי תְּפִלָה; כִּי מְשׁוֹרֵר כָּמוֹנִי

יִשְׁתֶּה מִמֶנוֹ ויאֹמַר “הִנֵנִי”

יִשְׁתֶּה וְיָשִׁיר אַךְ לֹא מֵעוֹנִי

וּכְמוֹ הִזְכַּרְתִּךָ לְטוֹב יִזְכָּרֵנִי






למלאות הגליון

הגפן בוכיה

מתורגם עברית משירי מלכת רומאניא





א


מַדוּעַ הַגֶפֶן הַפּוֹרִיָה

מַדוּעַ עֵינָהּ בּוֹכִיָה?

תִּשְׁפּוֹךְ דְמָעוֹת טְהוֹרוֹת

לִבָּהּ מִי יוֹדֵעַ

לָךְ אֲסַפֵּר רֵעַ

קָרוּ לָה הַקוֹרוֹת:


ב


תִּבְכֶּה – כִּי דַם עֲנָבֶיהָ

לֵחַ עֲסִיס פֹּארוֹתֶיהָ

הַמְשַׂמֵחַ דַלִים וְחוֹרִים

הַדַם הַזֶה – הַנֶפֶש

יִתְבּוֹלֵל בַּטִיט וְרֶפֶשׁ

שָׁם בַּחֵיךְ שִׁכּוֹרִים:


ערב שבעות.





למבקרים

בדרכי רבי מאיר הלכתי, רמון מצאתי תוכו אכלתי, וקליפתו זרקתי.

המשורר.





א


עוּרָה – כִּי שֶׁמֶשׁ גִבּוֹר חַיִל

הֵנִיס מִפָּנָיו צִבְאוֹת הַלָּיִל

חשֶׁךְ עַרָפֶל נָסוּ מִפְּנֵי אֲדִירוֹ

וְעַל הֵיכַל הַשּׁוּלְטַן נָטָה שַׁפְרִירוֹ.


ב


לִרְאוֹת הַמַּחֲזֶה נָשָׂאתִי עָיִן

וְהִנֵה קוֹל שָׁמַעֲתִּי מִבֵּית הַיָּיִן

"מִקְדָשְךָ פְּנִימָה נִפְתְּחוּ חֲדָרָיו

“לָמָה עָצֵל בַּל תִּדְרוֹךְ שְׁעָרָיו?”


ג


הַגֵּבֶר קָרָא שָׁם לַנַּעַר

"לָאֵלֶּה הָאוֹרְחִים פְּתַח שַׁעַר

"לֹא נֵדַע כַּמָה יִתְמַהֲמָהוּ

“וְעֵת יִפָּרֵדוּ – עִקְבוֹתָם לֹא נוֹדָעוּ.”


ד


רֹאשּׁ הַשָׁנָה הִנֵּה הִגִיעָה

עֵת נֶפֶש מַשְׂכֶּלֶת בָּדָד תַּרְגִיעָה

לִתְשׁוּקוֹת נוֹשָׁנוֹת שְׁחָקִים תַּרְקִיעַ

לַצַדיקֵי עוֹלָם עָל תַּגְבִּיהַ.


ה


גַּם נַפְשִׁי עַתָּה מָרוֹם תִּתְנַשָׂאָה

כָּעוֹטְיָה בָּדָד בַּחֲלוֹמוֹת תִּרְגָעָה

לִתְּשׁוּקוֹת חָלְפוּ – אֵיךְ חוֹמֶדֶת

וּלְתַאֲוָה הַנֶּפֶשׁ תְּבַקֵשׁ נִפְרֶדֶת.


ו


הַקַּיִץ עָבַר עִמוֹ הַשּׁוֹשַׁנִים

אַךְ הַחֲבַצֶלֶת עוֹד בַּגַּנִים

נֶהֶדֶרֶת בִּשְׁלַל צִבְעֵי רִקְמָתַיִם

בַּגַּנִים שְׁתוּלִים עַל יִבְלֵי-מָיִם


ז


גַּם שִׁפְתֵי דָוִד נֵצַח סֻגָּרוּ

אַךְ שִׁירוֹ מְשׁוֹרְרִים עוֹד שׁוֹרָרוּ

וּכְנַף רְנָנִים תָּשִׁיר לְשׁוֹשָׁן

מִשִׁירֵי דוֹדִים וּמִיַּיִן נוֹשָׁן.


ח


מַלֵּא הַכּוֹס – וּבְאֲבִיב חֶלְדֶיךָ

הָסֵר אֵדֶר חוֹרֶף חֶרְפָּתֶךָ

כִּי צִפֺּר הַזְּמַן מְשָׂרֶכֶת דְּרָכֶיהָ

הַצִפֹּר תָּמִיד תְּפָרֵשׁ כְּנָפֶיהָ.


ט


אִם הַיַּיִן מִבָּבֶל אוֹ פָּרָס

אַחַת לִי אִם מִמֶשֶׁךְ וְתִירָס

מִכּוֹס הַחַיִּים – רְסיסָו אַט יִזוֹלוּ

וּמֵעֵץ הַחַיִּים – עַלָיו אַחַת אַחַת יִפּוֹלוּ.


י


חַדָשִׁים לַבְּקָרִים שׁוֹשַׁנִים יִצְּמָחוּ

אַךְ אַיֵה הַשּׁוֹשַׁנִים תְּמוֹל פָּרָחוּ?

וּבְחֹדֶש הָאָבִיב תִּפְרַח הַשּׁוֹשַׁנָה

הַחַבַצֶלֶת אָז תִּבּוֹל בַּגַּנָה.


יא


יִבּוֹלוּ הַפְּרָחִים יִקְמְטוּ הַשּׁוֹשַׁנִים

יֶהֱמוּ הַמְּלָכִים הָרוֹזְנִים וְחַשְׁמַנִים

מַה לִי וְלָהֶם? – בִּדְרָכַי אֵלֵכָה

אֲיֵה הָאוֹר? אֲבַקֵּשׁ בַּחֲשֵׁכָה


יב


לִי מָנוֹחַ בִמְקוֹם גֶּפֶן פּוֹרִיָה

מַבְדֶּלֶת אֶרֶץ נוֹשָבֶת מִצִיָה

בִּמְקוֹם לֹא נִכַּר שוֹעַ לִפְנֵי דָל

וְאֵל בַּמָרוֹם – יוֹשֵׁב עָל.


יג


כִּנּוֹר בְּיָדִי שִׁיר הַשְׁמִיעַ

וְכּוֹס יַיִּן לְפי אַסִּיעַ

וְאַתְּ צְבִיָה יְמִינִי תִתְמוֹכִי

לָעֵדֶן – אָז הַמִּדְבָּר תַהֲפוֹכִי.


יד


אַחַר תַּעֲנוֹגֵי חֶלֶד רַבִּים נוֹהִים

וְרַבִּים יְצַפּוּ לִשְׂכַר בִּגְבהֹיִם

בַּל תִּבְטַח – קַח יֵשׁ לָךְ הַלוֹם

אַל תִּשְׁאַף לְמֶרְחַקִּים – לַחֲלוֹם.


יה


הַבֵּט הַשׁוֹשַׁנָה לָךְ שׁוֹחֶקֶת

לְטוֹבָתְךָ בַּחַיִּים אַךְ מִשְׁתּוֹקֶקֶת

הָסֵר צְנִיף רָאשֵׁי הַזֵּר

וּבְּגַן תְּפַזֵּר הָאוֹצָר לְאַחֵר


יו


אֵלֶה הָאוֹצרִים בָּר כַּלֶשָׁם

וְאֵלֶה הַמְפַזְרִים לָרוּחַ כַּגֶּשֶׁם

לֶעָפָר יָשׁוּבוּ אֶחָד גּוֹרָלָם

וּמֵעָפָר יֶחְפָּרוּ שֵׁנִית עֲמָלָם.


יז


תִּקְוַת תֵּבֵל לֶעָפָּר תָּשׁוּבָה

מִבְּלִי עֲשׁוֹת פְּרִי תְּנוּבָה

כַּשֶּׁלֶג עֲלֵי אֶרֶץ צִיָּה

תֶּמֶס תֵּלֵךְ וְנַפְשֶׁךָ לֹא חִיָּה.


יח


הַבֵּט הֵיכַל הַשּׁוּלְטַן וַחֳדָרָיו

יוֹמָם וָלַיְלָה לֹא יִסָּגֵרוּ שְׁעָרָיו

שׁוּלְטַן אַחַר שׁוּלְטַן בָּאוּ בַּהֲדָרָם

וֹבְּלִי יוֹמָם שָׁבוּ לָעַפָרָם


יט


הָאוֹחַ וְהָאְרִי בְהֵיכַל בָּתָה

בִּמְקוֹם בֵּלְשַׁטְצַר לִרְוָיָה שָׁתָה

וְעַל ראשׁ נִמְרֹד יִרְקַע בְּפַּרְסָתוֹ

עַיִר פֶּרֶא וְלֹא יְעִירוֹ מִשְׁנָתוֹ.


כ


וְשׁוֹשַׁנָה אֲדוּמָה בְּצִבְעֵי שָׁלָל

מִדַּם גִּבּוֹר מְאֳדָמָה – נָפַל חָלָל

ובְחֵין הַחֲבַצֶלֶת בֶּעָפָר עֻפְּרָה

גְּוִיָה טְהוֹרָה לָנוּ מַה יְקָרָה!


כא


מַלֵּא הַכּוֹס – וְהַיַיִּן נִמְסוֹכָה

בַּשֶּׁבֶר תִּקְוָתֵינוּ עָלָיו נִסְּמוֹכָה

מַלֵּא הַכּוֹס – כִּי מִי יוֹדעַ

אִם הוּא לֹא מִשִׁיקוּ רַעֲיָה וָרֵעַ?


כב


מַלֵּא הַכּוֹס יָפָתִי יַיִּן וָשֵׁכָר

יָפִיץ דַאֲגַת תְמוֹל וְיִרְאַת מָחָר

מָחָר! – אוּלַי עָמוֹק אִקָבֵרָה

עִם אַלְפֵי שָׁנִים תְּמוֹל חָלְפוּ אִסָפֵרָה?


כג


כִּי אֵלֶּה לָנוּ נְעִימִים וְנֶאֱהָבִים

שָׁתוּ מָצוּ מִדַּם הָעֲנָבִים

אָז דָּרַךְ הַזְּמַן בְּגִתּוֹ

אַחַר כָּל אֶחָד שֶׁפִי עָזַב לְאִטוֹ.


כד


נַחֲנוּ נָרִיעַ עַתָּה בָּאַרְמּוֹנָם

אַחַר לֶעָפָר יָרַד גְּאוֹנָם

גַּם נַחֲנוּ שׁאוֹל נַצִיעַ

לְאַחֵר לְרַפֵּד בִּגְּוִיָתֵנוּ יְצוּעַ.


כה


רְאֵה בַּטוֹב בַּחַיִים חַייָתְךָ

טֶרֶם לֶעָפר תָּשׁוב גְּוִיָתְךָ

עָפָר לֶעָפָר – וּבֵּן־חֲלוֹף

כָּלָה יָיִן תַּם מְשׁוֹרֵר וַגֲם הַסֺּף


כו


גוֹרָל אֶחָד לָאֵלֶה הַיוֹם יִדְאָגוּ

וּלְאֵלֶה לְיוֹם מָחָר יִשְׂאָגוּ

מִמִגְדָל עוֹפֶל קוֹל קוֹרֵא בָּרָמָה

“פְּתָאִים שְׂכַרְכֶם לֹא פֹּה וְשָׁמָה”


כז


הַּבֵּט חַכְמֵי־קֶדֶם רָשְׁמוּ רְשׁוּמִים

מִסוֹדוֹת וְרָזִים נָאֲמוּ נְאוּמִים

שְׂחוֹק לְרֵעֵיהֶם אַנְשֵׁי הָרוּחַ

בֶּעָפַר פִּימוֹ מָלְאוּ טּוּחַ


כח


בִּנְעוּרַי בֲּקֵשׁ חָכְמָה חָפַּצְתִּי

חֹזֶה שָׁאַלְתִּי עִם חָכָם נוֹעָצְתִּי

וּבְדֶלֶת זֶה נַפְשִׁי בָּאָה

בַּדֶלֶת זֶּה רֵיקָם יָצָאָה.


כט


זִרְעֵי הַחָכְמָה זָרַעֲתִּי עִמָם

לִנְטוֹעַ עָמַלְתִּי בַּשׂוּרוֹת לְתּוּמָם

וּמַה קָצַרְתִּי? – בְּחָפְנִי רוּחַ

כַּמַיִם בָאתִי הָלַכְתִּי כָּרוּחַ.


ל


פֺּה הִנֵנִי – לָמָה לֹא אֵדָעָה

מַדוּעַ זֶרֶם מַיִם נָד וְנָע?

וּמַדוּעַ יָפוּחַ צָפוֹנָה הָרוּחַ

הָגִידוּ לִי אַנְשֵׁי הָרוּחַ?


לא


מַדוּעַ? לָמָה? מֵאַיִן? וּלְאָיִן?

שְׁתֵה לִרְוָיָה מִכּוֹס הַיָיִן

שְׁתֵה! – וְשַׂר הַשִׁכְּחָה פּוּרָה

יִרְמוֹס לֶעָפָר הַחֲלִיפָּה וְהַתְּמוּרָה.


לב


מִטַבּוּר הָאָרֶץ מָרוֹם נַעֲלֵיתִי

עַל כַּנְפֵי שַׁחַר נִשֵׂאתִי

לַחִידּוֹת רַבּוֹת פִּתְרוֹנִים מָצָאתִי

אַךְ לֹא לְחִידַת חֲיָיתִי.


לג


דֶלֶת מָצָאתִי בְּלִי מַפְתֵּחַ

מַסֵכָה נְסוּכָה וְאֵין פּוֹתֵחַ

מְעַט אֲמָרִים – בֵּנִי וּבֵּינֶיךָ

אַחַר – אַיֵה אַנִי? וְאֵפֺה הִנֶךָ?


לד


אֶרֶץ לֹא עָנְתָה וְיָם דוֹמֵם

בַּגַלֵי תְּכֵלֶת אַךְ קוֹל וְרוֹמֵם

גַם קַצְוֵי רָקִיעַ עָמְדוּ דוֹם

הַמְכוּסִים בִּצְנִיפֵי לֵיל וְיוֹם.


לה


בֵּינִי וּבֵּינֶךָ הַגֵד חֵלֵכָה

כָּעִוֵר נְגַשֵׁשׁ פֺּה בַּחֲשֵׁכָה

מִי הוּא הַפּוֹעֵל בֵּיּנֶךָ וּבֵינִי?

וְקוֹל שָׁמַעֲתִּי “עִוֵר בָּךְ הִנֵנִי”:


לו


אָז לִכְלִי חֵרֶשׁ לְשִׂפְתֵי הַכּוֹס

דָבַּקְתִּי שְׂפָתַי לִשְׁאַל הָאוֹת

וְשָּׂפָה לְשָׂפָה נָתְנָה אֺמֶר

“שְׁתֵי, כִּי בְּמוֹתךָ הִנְךָ חֺמֶר”:


לז


הַכּוֹס מֵחֶרֶשׁ בְּקוֹל דְמָמָה

נָאֲמָה! – אָז חָיְתָה עֲלֵי אֲדָמָה

שָׁתִיתִי וּבִנְישִׁיקוֹת דָבַקְתִּי הַשָּׂפָה

וְכַּמָה נְשִׁיקוֹת אָז נָתְנָה הַיָפָה?


לח


אֶזְכּוֹר עָמַדְתִּי עַל אֵם דְרָכָיִים

לִרְאוֹת הַיוֹצֵר עוֹשֵׂה בָּאַבְנָיִם

מֵחֺמֶר לַח – כֶּלִי לְמַעֲשֵׂהוּ

וּלְאַט אָחִי שָׁמַעֲתִּי קוֹלֵיהוּ.


לט


הֲאִם לֹא סִפְּרוּ לָנוּ אֲבוֹתֵנוּ

מִיְמֵי קֶדֶם מִזֶה יוֹצְרֵנוּ?

אֵיךְ מֵחֹמֶר אֵל מֵרָמָה

יָצַר הָאָדָם חֶרֶשׁ מֵאֲדָמָה?


מ


וּשְׁפוֹךְ לָאֶרֶץ מִכּוֹסְךָ הַמְלֵאָה

לֹא לְרַוֹתִ אֲדָמָה צְמֵאָה

אךְ לְכַּבּוֹת בָעָיִן אֵשׁ־בּוֹעֶרֶת

בָּאָרֶץ קֶדֶם עָמוֹק נִקְבֶּרֶת


מא


כְּמוֹ הָעֵשֶׂב רֹאשוֹ יִשָׂא עָל

בַּבֺּקֶר לִשְׂתּוֹת מִכּוֹס שָּׁמָיִם – טָל

כֶּן אַתָּה הָרֵם רֹאשֶׁךָ

עַד כִּכְלִי רֵיק יִמַּגְרֵךְ יוֹצְרֶךָ:


מב


בַּל תִשְׁמַע לְטָחֵי תָּפֵל טוּחַ

דַּאֲגוֹת מָחָר פַּזֵר לָרוּחַ

וּשְׁלַח בְּיָדְךָ לְאַפְּךָ הַזְמוֹרָה

מִגֶפֶן פּוֹרִיָה לְאַפְּךָ קְטוֹרָה:


מג


הַיַיִן תִשְּׁתֶּה וּתְחַבֵּק הַשָׂפָה

וְעֵת תַּכְלִית כֹּל גַם לָךְ נִשְׁקָפָּה

זְכוֹר – הַיוֹם הִנְךָ כַּיוֹם אֶתְמוֹל

וּמָחָר תְהִי גַם כִּתְּמוֹל


מד


הַמַלְאָךְ הַגוֹאֵל בַּכּוֹס הַיְגוֹנִים

עֵת יִמְצָאֲךָ בְּלִי אוֹנִים

מִכּוֹסוֹ שְׁתֵה לְפִּיךָ הַצִיעַ

בַּל תִּירָא שְאוֹלָה תַצִיעַ:


מה


אִם אֲבַק חָמְרָה הַנֶפֶשׂ נוֹעֶרֶת

עֲרוּמָה בִּשְׁמֵי מָרוֹם עוֹבֶרֶת

אֵיךְ לֹא תֵבוֹשׁ בַּחַיִים חַיָיתָהּ

בְּנִבְלַת חֹמֶר שִׂים מְעוֹנָתָהּ?


מו


הַחַיִים כָּאֹהֶל בְּעַד יוֹם מַרְגוֹעַ

נָכוֹן בְּעַד מוֹשֵׁל אוֹ שׁוֹעַ

אַחַר הַכּוּשִׁי יַסִיעַ יְתֵידוֹתָיו

וּלְמַעַן אַחֵר יְקוֹמֵם יְרִיעוֹתָיו:


מז


בַּל תִּירָא מִיוֹם תְּמוּרָה וָחֶלֶף

מִכּוֹס הַחַיִים דָלפוּ דֶלֶף

כָּמוֹנוּ רְסִיסִים – וְעוֹד יִדְלוֹפוּ

כָּמוֹנוּ יִזְלוּ וְגַם יַחֲלוֹפוּ:


מח


עֵת נָבוֹא תַּחַת הַלוֹט

תֵּבֵל כְּקֶדֶם תְּהִי תֵּבַל עוֹד

צֵאתֵנוּ וּבוֹאֵנוּ בַּל תַּרְגִשָׁה

כִּצְרוֹר תַּשְׁלִיךְ הַיָם לְהַרְעִישָׁה


מט


רֶגַע עֲמוֹר ונַפְּשֶׁךָ חִיהָ

מַיִם חֲיִים לִשְׁתּוֹת בַּצִיָה

הָאוֹרְחָה נוֹסַעַת – הָרֵם עָיִן

מַהֵר אֵחֲרֶנָהּ לִגְבוּל הָאָיִן:


נ


הֲתֹאֱבֶה לְגַלוֹת פְּנֵי הַלוֹט

דַעַת תַּכְלִית כֹּל הַיְסוֹד

עֲשֵׂה מַהֵר הָפֵר חָק

כִּי אֶמֶת מִשֶׁקֶר – שֵׁעָר דָק.


נא


כַּשֵׂעָר הָאֶמֶת מִשֶׁקֶר נִבְדֶלֶת



אוּלַי אוֹת א מַפְתֵּחַ לַדֶלֶת



מָצָאתָ מַפְתֵּחַ לְבֵּית הָאוֹצַר הָבִיאָה



גַם לַאֲדוֹן כֹּל הַבְּרִיאָה:


נב


הֳאָדוֹן סוֹד חֲיָיתוֹ בְּחֲיָיתֵנוּ

וּבְפָנָיו נִרְאֶה גַם פָּנֵיּנוּ

אַךְ נַחֲנוּ נַחֲלוֹף נִהְיֶה לְבָעֵר

והוּא לְבַדוֹ לָנֶצַח יִשָׁאֵר:


נג


אַרְגִיעָה בִּין – וְשׁוּב לָעֵדֶר

בַחשֶׁךְ אוֹר מִשְׁטָר וָסֵדֶר

מִדוֹרוֹת עוֹלָם פָּעַל לְמַעֲנֵהוּ

פָּעֲל וְעֵינוֹ רָאֳתָה הִנֵהוּ


נד


ואִם לָרִיק – אָז אָרְצָה שׁוֹחֵחַ

אוֹ הַבֵּט מָרוֹם מְקוֹם לֹא יִפָּתֵחַ

הַבֵּט הַיוֹם כִּי הִנֶךָ

וּבְיוֹם מָחָר אָז אֲיֶכָּה?


נה


בַּל תַחֲקוֹר בְדָתוֹת וְחוּקִים

רַדֵף אֲמָרִים מֵחוֹזִים עֲתִּיקִים

טוֹב בַּכּוֹס לָשִׂים עָיִן

מֵאֶכֺל בּוֹסֶר וְלִרְדוֹף הָאָיִן:


נו


הַתֵּדַע יְדִידִי –? בַבֵּיתִי נְהָרָה

אִשְׁתִּי גָרַשְׁתִּי, הָיְתָה עֲקָרָה

אֵשֶׁת מְזִימָה פְּתַיָה וְהוֹמִיָה

וּלְאִשָה אֵרַשְתּי הַגֶפֶן הַפּוֹרִיָה:


נז


כִּי הַיֵש וְהָאָיִן אֶתְאַו בַּתָּיו

וּמַעַל וּמָטָה אֲתָאֵר בַקָו

מִכֹּל הַסוֹדוֹת נַפְשִּׁי יוֹדָעַת

סוֹד הַכָּמוֹס בַכּוֹס הַקֻבָּעָת:


נח


עָלָי יִתְלַחֲשׁוּ אוֹהַבָיִ הַנִרְגָנִים

כִּי טָעִיתִי בַחֶשְבּוֹן הַשָנִים

אַךְ נִמָחוּ מִלוּחַ – הָעֵת

בַּל נוֹלָד מָחָר וּתְמוֹל מֵת:


נט


בָּעֶרֶב עָמַדְתּי בַפֶּתַח עֵינָיִם

וָאֶרְאֶה מַלְאָךְ לְפָנָי מִשָמָיִם

וּבְיָדוֹ כּוֹס מֶנוֹ שָׁתִיתִי

הֶאָח אָמַרְתִּי, הַיַיִן רָוִיתי:


ס


הַיַיִן בְכֺּחוֹ גְדוֹלוֹת יְחוֹלֵל

עַמִים וְגוֹיִם בְהֶגְיּוֹנוֹ יְהוֹלֵל

הַקוֹסֵם בִּקִסָמָיו – וְלִזְּהַב פַּרְוָיִם

יַהֲפֹךְ מְטִיל עוֹפֶרֶת הַחַיִים:


סא


הַנְבִיאִים, הַחוֹזִים, הָרוֹאִים בֶּעָבִים

אֲשָׁפִים וְחַרְטוּמִים יֶחֱזוּ בַּכּוֹכָבִים

פַחַד וּדְאָגָה מֶנָם נֶפֶשׂ תָּשׁוּחַ

בְּחַרְבּוֹ הַקָשָׁה יְפַזְּרֵם לָרוּחַ:


סב


נְטַע יי הַגֶפֶן הַפּוֹרִיָה

מְחַלְלֶיָה נַפְשוֹ לֹא חִיָה

יָדוֹ עָשָׂתָה נָטְּעָה זְרוּעֶיהָ

לְרַוֹת אָדָם מִכּוֹס תַּנְחוּמֶיהָ:


סג


הֲאִם לֹא אֶשְׁתֶּה מִכּוֹס תַּנְחוּמִים

מִפַּחַד עֲתִידוֹת יוֹם שְׁלוֹמִים?

אוֹ בַתּוֹחֶלֶת מְמוּשָׁכָה לָעִדָן וְעֲדָנִים

לַמַלֵא כּוֹסִי מִיֵין שְׁמָנִים?


סד


מִפַּחַד תָּפְתֶּה וּמִתּוֹחֶלֶת שָׁמָיִם

אַחַת אֵדַע – יָעוּפוּ הַחַיִים

אַחַת אֵדַע – כֹּל הָאָדָם כּוֹזֵב

לֹא יָשׁוּב לְבֵּיתוֹ עֵת מְקוֹמוֹ עוֹזֵב:


סה


לַשַׁעֵר צַלְמָוֶת רִבֹּאוֹת נָהָּרוּ

בַּל אֶחָד שָב מֵאֵלֶה עָבָרוּ

לְהוֹרוֹת לָנוּ הַדֶרֶךְ נֵלֵכָה

וּכְעִוְרִים בַּדֶרֶךְ נְגַשֵׁשׁ בַחֲשֵׁכָה:


סו


חֶזְיוֹנוֹת חָזוּ מִקֶדֶם הָרוּזִים

קְדוֹשֵׁי יָה חָזוּ בַּגְבֹהִים

מִשְׁנָתָם נֵעוֹרוּ – חֲלֹמוֹתָם נָאֳמוּ

אַחַר בִשְּׁנָתָם – שֵׁנִית נִרְדָמוּ


סז


בַּחֵקֶר שַׁדַי נַפְשִׁי הָלָכָה

עִם גְוִיָתִי עוֹז דָרָכָה

אֲמָרֶיהָ לִי הֵשִיבָה כַּנוֹפֶת

אֲנִי הָעֶדֶן וְאֲנִי הַתוֹפֶּת:


סח


אֲנִי הָעֵדֶן – תַּאֲוָתְךָ נִמְלָאָה

אֲנִי הַתּוֹפֶת תַּאֲוָתְךָ נִתָּעָה

מֵחשֶׁךְ בָּאִנּו וּלְחשֶׁךְ נֵלֵכָה

לִרְוָיָה שׁתֵה אֶנוֹשּׁ תֵלֵכָה:


סט


נַחֲנוּ כַּגוֹלִָמִים מֵחֵמֶר וּרְגָבִים

לַמַראוֹת עָיִן רָצִים וְשָׁבִים

לְאוֹר הַנֵר הַקוֹסֵם יַרְאֵמוֹ

בַחֲצִי לָיִל בִּקְסָמָיו יְנַהֲלֵמוֹ:


ע


יַצִיג לְפָּנֵינוּ אַךְ גוֹלָמִים

יִשְׁחֲקוּ עַל עֶבְרֵי לֵילוֹת וְיָמִים

יִשְחֲקוּ כְּמוֹ חַי כְּמוֹ חָרוֹן

אַחַר הַקוֹסֵם יְשוּמֵם בֶּאָרוֹן:


עא


כַּכַּדיר מִשְׁחָק בִּיְדֵי הַנְעָרִים

לַמִשְׁחָק הַכַּדוּר מִשְׁטָר וּסְדָרִים

אִם בָּךְ הַכַּדוּר נוֹגֵעַ

לָמָה? – חָק הַמְשַחֵק יּוֹדֵעַ:


עב


הַיָד הַנָעָה בִּכְּתָב אֶמֶת

הָלְאָה הָלְאָה לָךְ רוֹשֶׁמֶת

וּבְכָל חָכְמָתֶךָ אָחֹר לֹא תְשׁוּבֶנָה

וּבְכָל דִמְעוֹתֶךָ אוֹת אַחַת לֹא תִמָחֶנָה


עג


לְטַבּוּר הָאָרֶץ יִקְרְאוּ “רָקִיעַ”

לָעֵזֶר יָדְךָ בַּל תַּגֶבִּיהַ

כִּי כָּמוֹנוּ נוֹדֵר הוּא וְנוֹסֵעַ

וּמְאַיִן וּלְאַיִן? מִי יוֹדֵעַ?


עד


מֵעַפַר אָרֶץ לָשׁוּ בְּיָדָם

בְּרֵאשִׁית הַבְּרִיאָה – אַחֲרוֹן הָאָדָם

וּבְיוֹם הָרִאשׁוֹן כָּתְבוּ בָרָמָיִם

יוֹם שְׁלוּמִים בְּאַחֲרִית הַיָמִים


עה


תְּמוֹל – אֲסוֹן הַיוֹם הֵבִיאָה

וּמָחָר – דְאֳגָה אוֹ בִשְׂפָתֶךָ תְּרוּעָה

שְׁתֵה – וּבַכּוֹס שִׂים עָיִן

כִּי לֹא תֵדַע אָנָה? מִי? וְאָיִן?


עו


הַגֵו מֵעָפָר טִיט וָרֶפֶשׁ

אָרְשּׁוּ לְבַת מְרוֹמִים הַנֶפֶשׁ

אֵיךְ יִתְאַחֲדוּ יַחַד שְׁלוֹמִים

תְּהוֹם מִתַּחַת וְרָקִיעַ מְרוֹמִים:


עז


הַיַיִן כַּשֶׁמֶן בָּא בָּעַצְמוֹתַי

וּבּוֹ עַתָּה אָשִׁים אוֹתוֹתַי

אֲנַחֵשׁ בַּגָבִיעַ וָאֶמְצָא הַמְפַתֵּתַ

עֵת הַכֹּהֵן בַּחוּץ מַר צוֹרֵחַ:


עח


אֵיךְ מֵאַיִן מֵאֶפֶס וּמְאוּמָה

נוּכַל לְהוֹצִיא רֶגֶשׂ וּמְזִימָה

לִהְיוֹת לְדֵרָאוֹן בַּגֵיא הַתּוֹפֶת

בְּעַד תַּעֲנוּגִים מָתְקוּ כַּנוֹפֶת?


עט


שַׁלֵם נִשְׁיִי – לֹא לָוִיתִי

וְגַם הַנוֹשֶׁה לֹא רָאִיתִי

הָשֵׁב זָהָב בְּעַד בְרַק סִיר

לִתְמוּרָה כָּזֹאת אוֹ לִמְחִיר:


פ


הֵן אַתָּה סָלַלְתָּ לִי הַדֶרֶךְ

הַלִפּוֹל שָׁמָה וְלִשְבּוֹר יֶרֶךְ

הַרְצוֹנְךָ כּי אֲשׁוּרַי יִמְעָדוּ

וּבַטִיט הֶעָוֹן יַחַד יִתְלַכָּדוּ:


פא


אָדָם יַצַרְתָּ, מֵעָפָר קָבַצתָּ

גַם בָּעֵדֶן נָחָשׁ הִּרְבַּצְתָּ

אִם מִשְּׂחוֹר הָעָוֹן חָשַׁךְ תָּאֳרוֹ

כֶּאָדָם סְלַח מֵחֶטְאוֹ לְטָהָרוֹ:


פב


בַּעֲרוֹב הַיוֹם בִּנְטוֹת הַצְלָלִים

דוֹמָם עָמַדְתִּי בֵּין צֶאֱלִים

בַּבֵּית הַיוֹצֵר שָם בּדְמָמָה

בֵּין כְּלֵי־חֶרֶשׁ מֵחַרְשֵׂי הָאֲדָמָה


פג


כֵּלִים מִכֵּלִים שּׁוֹנִים סֻדָרוּ

רוּבָּם גַם בַּיוֹפִי נֶהֱדָרוּ

בַּקִיר וְעַל הָאָרֶץ עָמָדוּ

אִישׁ בֵאָחִיו יַחַד הִתְלַכָּדוּ


פד


וְאֶחָד מֵהֶם נָתַן אוֹמֶר

לֹא לָרִיק יוֹצְרִי עֶשָׂנוּ מֵחֹמֶר

לַכֶּלִי עָשָׂנִי אוֹ יְּשַׁבְרֵני

וּלְעָפָר כְּקֶדֶם הַיוֹצֵר יְשׁוּבֵני


פה


וְקוֹל עָנָה מִבֵּין הַקִיר

אוֹמֶר נָתַן אֶחָד סִיר

"מִכֵּלִי הַיוֹצָר שָׁתָה בְּשִׂמְחָתוֹ

וְיָדוֹ עָשְׁתָה – לֹא יְשַׁבֵּר בַחֲמָתוֹ"


פו


וְסִיר אַחֵר מִבֵּין הַסִירִים

עָנָה בִּדְמוּיַת קוֹל אֲמָרִים

קָטוֹן אֲנִי – עָלַי הַיוֹצֵר יָדוֹ הָדָה

וּבְעֲשׂוֹתִי אָז יָדוֹ מַדוּעַ רָעָדָה?


פז


אָז סִיר בְּרוֹגֶז קוֹל הִשְׁמִיִעַ

מִיוֹצר וְסִיר – כֵּן הִבִּיעַ

אֶחָי כּוּלְכֶם הַגִידוּ לִי

הַיוֹצֵר מִי? וְהַסִיר מִי?


פח


אוּלַי קֶדֶם יוֹצְרִים הָיִינוּ

טֶּרֶם לַסִירִים נִכְחַד קִימָנוּ

אוּלַי זֶה הַיוֹצֵר יִהְיֶה לַסִיר

לַחֵמֶר טוּחַ בּוֹ הַקִיר:


פט


וְאֶחָד סִיר נָתַן אוֹמֶר

יֵש יוֹצֵר עוֹשֵׂה בַּחֹמֶר

הַמַאֲמִין יַאֲמִין יְהִי בָּרוּךְ

וְדוֹבֵר שְקָרִים לוֹ תַָּפְתֶה עָרוּךְ:


צ


וְסִיר אַחֵר יָלַל מִזוֹקֶן חֻלֵיתִי

עַצְמוֹתַי נִחָרוּ לַיַיִן צָמֵאתִי

מַלְאֵנִי מִמִיץ עֵנָב נַפְשִׁי יְבֵשָׁה

אוּלַי כְּקֶדֶם נְעוּרַי אֲחַדֵשָׁה:


צא


עֵת הַסִירִים יַחַד הִתְלֶחָשׁוּ

יָרֵח בְּהִלוֹ קַוָיּוּ רָעָשׁוּ

הִבִּיט בִּפְנֵיהֶם – וּפָּנָיו נָהָרוּ

עֵת הַסִירִים לִרְסִיסִים נִשְׁבָרוּ


צב


בְּדַם עֵנָב תִרְחַץ גְוִיָתִי

עֵת יוֹמִי יָבוֹא תִתַּם חֲיָיתִי

בַתַּכְרֵיךְ הַחַיִים לוּטָה עוֹר נִשְׁמָתִי

וּבַפִּנַת הַכֶּרֶם תַחֲצוֹב קִבְרָתִי:


צג


גַם עֵת אֲחַבֵּק רְגָבִים

אֶמְצֶה אִינָק מִלֵחַ עֲנָבִים

הַהֵלֶךְ כִּי יָבוֹא לְאִטוֹ

גֶפֶן פּוֹרִיָה תְהִי אִתּוֹ:


צד


אֱלִילִים הְלַלְתִּי בְּעוֹכְרָי הָיוּ

עֲלֵי גוֹרָלִי רַבִּים יִבְכָּיוּ

בַּכּוֹס רֵיק טָבְעוּ חֲיָיתִי

וּבְעַד נְזִיד עֲדַשִׁים מָכְרוּ בְּכוֹרָתִי


צה


לִרְגָעִים נִחַמְתִּי עַל תַהֲלוּכוֹתַי

גַם לֵילוֹת יִסְרוּנִי כִּלְיוֹתַי

אַךְ הָאָבִיב בִמְטַאַטֵאַ הַשׁוֹשַׁנִים

נֶחָמוֹתַי לָרוּחַ פִּזֵר בַּגַנִים:


צו


הַיַיִן בְּעוֹכְרִי וּבּוֹ תּוֹחַלְתִּי

וְאִם אֵדֶר תִּפְאַרְתִּי בֶּעָפָר עוֹלַלְתִּי

עַתָּה אֶשְּׁתּוֹמֵם אֵיךְ הַיוֹגְבִים יִסְחָרוּ

בַחֲצִי הָעוֹדֵף מֵאֲשֶׁר יִמְכָּרוּ?


צז


הָאָבִיב יַחֲלוֹף עִם הַשׁוֹשַׁנִים

עֲלוּמַי יִבּוֹלוּ כֶעֳלִים בַּגַנִים

וּכְנַף רְנָנִים תָּשִׁיר זְמִרֶיהָ

מֵאַיִן? וּלְאַיִן? תִּפָרֵשׁ כְּנָפֶיהָ?


צח


לֹא אוּכַל הַלוֹט לְגַלֵה

בַּמַבָּט עָיִן תַּאֲוָתִי לְמַלֵא

בַמִדְבָּר הַחַיִים רַוֵה צִמְאוֹנִי

לִרְאוֹת בַקֹדֶש מַלְכִּי וְאֲדוֹנִי:


צט


לוּא מַלְאָךְ שָׂרָף מְעוֹפֵף

מִיַד הַמַזְּכִּיר עָלַי יְאֹפֵף

מַזְכִּיר יְמוֹת עוֹלָם בְסִפְרָתוֹ

אוֹ הֶסִירוֹ מִמְשְׁמַרְתּוֹ מְאַחֵר יִקַח עֶמְדָתוֹ:


ק


וְאַתְּ אַהֲבָה בַּכֹּחַ נֶאֱדָרִי

בַּקוֹשְּׁרִים עִמִי וְעִמוֹ תִקְשָׁרִי

אָז נְחַדֵשּׂ תֵבֵל וְצֶאֱצָאֶהָ

כְחֵפֶץ לִבֵּנוּ נַעֲרוֹךְ מַעַרְכוֹתֶיהָ:


קא


וְאַתָּה יָרֵחַ בְהִלְךָ תִרְאֵנִי

יָמִים יָבוֹאוּ – וּלְשָוא תְבַקְשֵׁנִי

בֵּין מֵתֵי עוֹלָם – עָפָר זְּרוּעֶיהָ

וּשְׁנָתָם יִרְדָמוּ אֶרֶץ וְצֶאֱצָאֶהָ:


קב


וְאַתָּה יְדִידִי בְּאַחֲרִית הַיָמִים

תָבוֹא בִּמְקוֹם נִטְעֵי הַכְּרָמִים

בִמְקוֹם אָז פָּרַצְתִּי פֶרֶץ

כּוֹס רֵיק מַגֵר לָאָרֶץ:






תם






“ג” נוסף במקור, צ"ל ד', וכן הלאה מיספור הבתים  המתוקן – הערת פרויקט בן־יהודה.  ↩
