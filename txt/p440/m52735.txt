


	
	יְרֹחָם וְתַפּוּחֵי הָאֲדָמָה (סִפּוּר)
	




יְרֹחָם הִנֵהוּ כְּבָר “בָּחוּר” בֶּן שֶבַע – שְמוֹנֶה
שָנִים, הוּא יוֹדֵעַ הַרְבֵּה. שְנֵי סְפָרִים הִסְפִּיק כְּבָר לִקְרֹע.
אָמְנָם גַם אָחִיו הַקָטָן חֶזְקִי, זֶה שֶהִתְחִיל לִפְנֵי חֹדֶש יָמִים
לָלֶכֶת, יוֹדֵעַ מְלָאכָה זוּ שֶל קְרִיעַת־סְפָרִים. אֶלָא שֶיְרֹחָם
קוֹרֵעַ אֶת סְפָרָיו מִתּוֹךְ לִמוּד, וְזֶה הַקָטָן חֶזְקִי,
כְּשֶמִזְדַמֵן תַּחַת יָדוֹ סֵפֶר, יִהְיֶה אֵיזֶה שֶיִהְיֶה: סִדוּרָהּ
שֶל אִמָא, פִּנְקָסוֹ שֶל אַבָּא אוֹ גְמָרָתוֹ שֶל סָבָא – הוּא אֵינוֹ
מַבְחִין בָּהֶם וְקוֹרֵעַ אוֹתָם. כַּךְ הוּא טִבְעוֹ שֶל חֶזְקִי!

יְרֹחָם, זֶה הַ“בָּחוּר” תַּלְמִיד בֵּית הַסֵפֶר, אוֹהֵב
אֶת אָחִיו הַקָטָן בְּכָל לִבּוֹ וְנוֹשֵק לוֹ בְכָל פַּעַם שְתֵי
לְחָיָיו הַקְטַנוֹת וְהַמְלֵאוֹת. וְאֵין כָּל פֶּלֶא: כְּלוּם יֵש לְךָ
אָדָם שֶרָאָה אֶת חֶזְקִי וְלֹא יֶאֱהָבֵהוּ?
אֱמֶת שֶהַקָטָן אֵינוֹ סוֹבֵל בְּיוֹתֵר אֶת נְשִיקוֹתָיו
שֶל אָחִיו, וְלוּלֵא הַכַּפְתּוֹרִים הַנוֹצְצִים שֶיְרֹחָם נוֹתֵן לוֹ
חִלוּף הַנְשִיקוֹת הָיָה מַרְעִיש הַקָטָן עוֹלָמוֹת.
וְיָכֹל הוּא חֶזְקִי לְהַרְעִיש עוֹלָמוֹת. אִמָא שֶלוֹ
אוֹמֶרֶת כַּךְ: קָטָן זֶה פַּעֲמוֹן יֵש לוֹ בִּגְרוֹנוֹ, פַּעֲמוֹן
קָטָן, וּכְשֶהוּא רוֹצֶה הוּא מְצַלְצֵל בּוֹ עַד שֶתְּצִלֶנָה
הָאָזְנָיִם.

וְאוּלָם יְרֹחָם יוֹדֵעַ כֵּיצַד לָגֶשֶת אֶל הַתַּכְשִיט
הַקָטָן הַזֶה. בְּיָדַיִם רֵיקוֹת לֹא יִקְרַב אֵלָיו, וּבְטֶרֶם גִשְתּוֹ
הוּא מְקַשְקֵש בְּכִיס בִּגְדוֹ בְכַפְתּוֹרִים אֲחָדִים. וְיֵש שֶהוּא
קוֹנֶה אֶת לִבּוֹ בְּפִסַת נְיָר פְּשוּטָה.

וְלָכֵן מִשֶרוֹאֶה חֶזְקִי אֶת יְרֹחָם אָחִיו הוֹלֵךְ
וְקָרֵב אֵלָיו, פָּנָיו שֶל הַפָּעוּט צוֹחֲקִים – צוֹחֲקִים לוֹ,
מְנַעְנֵעַ לוֹ אֶת יָדָיו הַקְטַנוֹת וְהֶעָבוֹת, וְקוֹרֵא מִתּוֹךְ
רִגְשֵי שִׂמְחָה מִינֵי הֲבָרוֹת שֶקָשֶה לַהֲבִינָן.
רַק פַּעַם אַחַת נִבְהַל הַקָטָן מִפְּנֵי יְרֹחָם
אָחִיו, וְזֶה הַדָבָר: אִכָּר אֶחָד הֵבִיא לְבֵיתָם שַׂק מָלֵא תַפּוּחֵי
אֲדָמָה יָפִים מְאֹד. הוֹצִיא יְרֹחָם אֲחָדִים מִתּוֹךְ הַשַׂק, נִכְנַס
אֶל חֲדַר הַבִּשוּל וּזְרָקָם אֶחָד אֶחָד לְתוֹךְ הָרֶמֶץ שֶבַּתַּנוּר
מִלְמָטָה. כִּי אָהַב יְרֹחָם מְאֹד לֶאֱפוֹת תַּפּוּחֵי אֲדָמָה
בְתוֹךְ1 הָרֶמֶץ וּלְאָכְלָם אַחֲרֵי כַךְ.



בְּאֵפֶר הַתַּנוּר הָיוּ עוֹד נִיצוֹצִים וּשְבִיבִים
לוֹחֲשִים רַבִּים, וְתַפּוּחֵי הָאֲדָמָה נִתְבַּשְלוּ וְהִתְחִילוּ
מִתְכַּוְצִים, וִיְרֹחָם הִפְשִיל אֶת שַרְוֻלוֹ וְהוֹצִיא אֶת תַּפּוּחֵי
הָאֲדָמָה, זֶה אַחַר זֶה, וּבַחֲדַר הַבִּשוּל גוּפָא עָשָׂה אֶת
הַסְעוּדָה, וְהָיוּ תַפּוּחֵי הָאֲדָמָה יוֹצְאִים מִן הַתַּנוּר אֶל
הַפֶּה, מִן הַתַּנוּר אֶל הַפֶּה…
הוֹי, לוּ הִסְתַּכֵּל יְרֹחָם בְּאוֹתָה שָעָה בָּרְאִי – מֻבְטָחַנִי שֶלֹא הָיָה מַכִּיר אֶת עַצְמוֹ.&gt;
– אַךְ אֵלֶה הָיוּ תַפּוּחֵי אֲדָמָה! קָרָא יְרֹחָם וְהָלַךְ לְאַחַד הַחֲדָרִים לְבַקֵש אֶת חֶזְקִי אָחִיו הַקָטָן.
וְהוּא אָמְנָם מְצָאָהוּ.

– חֶזְקֵלִי! קָרָא יְרֹחָם וְהוֹשִיט אֶת יָדָיו
לַפָּעוּט. אֶלָא שֶהַפָּעוּט פָּקַח עָלָיו שְתֵּי עֵינַיִם גְדוֹלוֹת
וּמְלֵאוֹת פַּחַד, וְהִתְחִיל בּוֹכֶה, בּוֹכֶה וּמְצַלְצֵל בַּפַּעֲמוֹן
שֶבִּגְרוֹנוֹ!

יְרֹחָם לֹא יָדַע, מַה זֶה הָיָה לוֹ לְחֶזְקִי פִתְאֹם,
כִּי נִבְהַל כָּכָה. וְלָכֵן מִהֵר אֵלָיו לְהַשְתִּיקוֹ, בְּמַשְמְשוֹ
בְכִיס בִּגְדוֹ בְכַפְתּוֹרִים אֲחָדִים. אַךְ מִשֶנָגַע יְרֹחָם
בְּאָחִיו, הִתְחִיל הַלָז מְפַחֵד עוֹד יוֹתֵר, וְקוֹל בִּכְיוֹ עוֹד גָדָל.

כָּל בְּנֵי הַבַּיִת הִתְחִילוּ רָצִים לְקוֹלוֹ שֶל חֶזְקִי, כִּי הָיָה הַפָּעוּט חָבִיב וְרָצוּי לַכֹּל.&gt;
כְּשֶנִכְנְסוּ כָל בְּנֵי הַבַּיִת אֶל הַחֶדֶר, פָּנָה
אֲלֵיהֶם יְרֹחָם כְּדֵי לְסַפֵּר לָהֶם אֶת כָּל הַקוֹרוֹת אוֹתוֹ,
וְהִנֵה צְחוֹק פָּרַץ מִגְרוֹן כֻּלָם. מַבִּיטִים הֵם בְּפָנָיו
וְצוֹחֲקִים…

– לָמָה אַתֶּם צוֹחֲקִים? שָאַל יְרֹחָם בְּתִמָהוֹן.
יָצְאָה מִתּוֹכָם שִפְרָה אֲחוֹתוֹ, וְאָחֲזָה אוֹתוֹ
בִזְרוֹעוֹ, נְהָגַתְהוּ וְהֶעֱמִידַתְהוּ לִפְנֵי הָרְאִי – אוֹי
לָעֵינַיִם שֶרָאוּהוּ בְכַךְ!

כָּל פִּיחוֹ שֶל הַתַּנוּר הָיָה עַל פְּנֵי יְרֹחָם!
וּמַה בּוֹש “בָּחוּר” זֶה בְּהִוָדַע לִבְנֵי הַבַּיִת כָּל הַדָבָר
מֵרֵאשִיתוֹ עַד קָצֵהוּ…

מִן הַיוֹם הַהוּא לֹא הוֹסִיף עוֹד יְרֹחָם לְטַפֵּל
בְּתַפּוּחֵי אֲדָמָה, וְכַאֲשֶר רָאָהוּ חֶזְקִי, צָחַק אֵלָיו וְצָהַל
לִקְרָאתוֹ כִּתְמוֹל שִלְשוֹם. לֹא יָדַע חֶזְקִי שֶזֶהוּ הַשֵד הַשָחוֹר
שֶל אֶתְמוֹל שֶרָצָה לְחָטְפוֹ.
