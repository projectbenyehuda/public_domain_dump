


	
	בַּחֲלוֹם לַיְלָה חֲלַמְתִּי לַעֲדַת קֹדֶשׁ בְּנֵי תָם
	



חתימה: שבזי




בַּחֲלוֹם לַיְלָה חֲלַמְתִּי / לַעֲדַת קֹדֶשׁ בְּנֵי תָם



בֵּין מְשַׂנֱאַי נֶאֱחַזְתִּי / אָסָרוּנִי תּוֹךְ נְוָתָם



אָז יְמִין הָאֵל זְכַרְתִּי / וַאֲנַתֵּץ אַרְמְנוֹתָם



גַּם וְחֵילֵיהֶם נְתַשְׁתִּי / נִפְזְרוּ מִמִּשְׁכְּנוֹתָם



נָפְלָה אֵימָה עֲלֵיהֶם



קוֹל בְּפַחַד פִּזְּרֵיהֶם



לֹא הֱרִימוּן אֶת יְדֵיהֶם



וַאֲנִי חֵילָם שְׁלַלְתִּי / וַאֲחַפֵּשׂ מִצְפְּנוֹתָם



בַּהֲדַר דּוֹדִי צְלַחְתִּי / בָּא בְּאַרְבַּעַת קְצוֹתָם:



שָׁמְמוּ אוֹיְבִים וְקָמִים / וַחֲבַצֶּלֶת שְׂמֵחָה



בָּא יְדִידִי בֵּין כְּרָמִים / הֶחֱרִיב בָּצְרָה וּמָחָה



וַחֲסִידִים וַחֲכָמִים / מִנְּדוֹד קִבַּץ וְנָחָה



וַאֲנִי אֶצְלָם שְׂמַחְתִּי / אֶרְדְּפָה עַל אַהֲבָתָם



זֶה שְׁנַת הַתְּכ"וּ לְמִנְיָן



אֶקְבְּצָה מִקְנֶה וְקִנְיָן



עָמְדָה צִיּוֹן בְּבִנְיָן



אָז שְׂדֵה גוֹלָה מְכַרְתִּי / עַל בְּנֵי עַוְלָה כְּדַעְתָּם



יוֹם דְּבַר נָתָן שְׁמַעְתִּי / נָפְלוּ אוֹיְבַי בְּרִשְׁתָּם:



בַּת מְלָכִים דַּבְּרִי שִׁיר / עוֹרְרִי דוֹד רַב פְּעָלִים



קַבְּצִי עָנִי וְעָשִׁיר / וּקְשְׁרִים אַרְבַּע דְּגָלִים



סוֹד חֲלוֹם אָבִין לְהַפְשִׁיר / לַעֲלוֹת שָׁלֹשׁ רְגָלִים



אִם חֲלוֹמִי זֶה אֲמִתִּי / צַח בְּתוֹךְ פַּרְגּוֹד וְנֶחְתָּם



בּוֹ אֲמִתַּת הַבְּשׂוֹרָה



הוּא כְגָבִיעֵי מְנוֹרָה



מִגְבְּעֵי צַחוּת וְאוֹרָה



בּוֹ יְדִיעָתִי רְשַׁמְתִּי / סוֹד אֱמֶת גָּלוּי וְנִסְתָּם



וַאֲנִי דוֹדִי חֲמַדְתִּי / יִגְּהָה נֵר הָל לְנֶעְתָּם:



זָרְזָה גוֹלֶה לְחַרְצָךְ / קוּם וְנַקְבִּיל הַיְּדִידִים



וַעֲלֵה חַזְמַ"ק לְאַרְצָךְ / נִקְבְּצָה כָּל הַשְּׂרִידִים



גַּם בְּנָךְ יוֹסֵף יְעִיצְָךְ / אָז וְתִהְיוּן לִי מְעִידִים



כִּי בְשִׁירִי זֶה סְגַלְתִּי / אֶת יְסוֹדוֹתַי בְּגִשְׁתָּם



עִם שְׁסַ“ה וּרְמַ”ח בְּתוֹךְ גּוּף



גּוֹרְמֵי מַחְלָה וְסִגּוּף



הוֹן וּמַחְשָׁבָה וְרִגּוּף



גַּם שְׂכָלַי הֶאֱצַלְתִּי / יָדְעוּ אֶת דּוֹר אֲבוֹתָם



כִּי בְשֵׁם שַׁדַּי בְּטַחְתִּי / רָם עֲלֵי כָל מַחְשְׁבוֹתָם:



יִסְגְּלָה […]
