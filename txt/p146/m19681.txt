


	
	מַסַאֻן פִי רַצַׄא רַבִּי עַלַי שִׁעבֹּה תְגַׄלַּא
	



החתימה: משתא אלשבזי (84 – 85: סין אלף לאם ומים; ולד יוסף)




מַסַאֻן פִי רַצַׄא רַבִּי / עַלַי שִׁעבֹּה תְגַלַּא



אַנַאר שַׁרקִי וְגַרבִּי / בְּאִחסַאנֻן וּפַצׄלַא



בִּגֵׄאהֹה אִלתַטַף בִּי / וַאַועַדנִי בִּקַוְלַא



וְנוּרֹה חַאטִ קֻרבִּי / עֲלַי מַא כַּאן קַבּלַא



שַׁרִיף אלדִּין אַרסַל לִי



דַׄכִּי אַלנַּפסִ וַאלעַקִלי



וַיַנפִי פַאעִלַ אלגַׄהלִי



חֲמוּד נַפְשִׁי וְגַם לִבִּי / לְסוֹדוֹתַי מְגַלֶּה



בְּכָל שַׁחְרִי וְעַרְבִּי / שְׁאָלוֹתַי יְמַלֵּא:



שְׁזַפְתִּיו וַאֲנִי יוֹשֵׁב / בְּתֵימַָן נִבְהֲלִי



וְלִבִּי מַעֲשָׂיו חוֹשֵׁב / וְדוֹרֵשׁ פּוֹעֲלִי



וְסוֹד תּוֹרָה בְּחֵן קוֹשֵׁב / וְשִׂכְלִי צוֹהֲלִי



וְשָׂמַחְתִּי וְסָר עָצְבִּי / בִּכַׄירֻן שַׁאמִלַא



בַּלַג שַׁוקִי אִלַי אלְקֻדסִי



וַרַיתַ אלמֻלךּ פִי חִסִּי



תַהַיַּא וַאזהִיַ אלכֻּרסִי



שְׂמַח דּוֹדִי וְרוּץ כִּצְבִי / וְעוֹדֵד דָּל מְחַלֶּה



שְׂמֹאלֶךָ יְדִיד חַרְבִּי / תְּרִימֵנִי וְתִדְלֶה:



תְּמָכַנִי יְדִיד נַפְשִׁי / בְּפוּרִים לַעֲלוֹת



וְלִוַּנִי לְהַר קָדְשִׁי / בְּרוּמֵי מַעֲלוֹת



וְשַָׂם חֶלְקִי בְּקָו שִׂשִּׁי / וְרִפֵּא מַחֲלוֹת



וְדִשַּׁנְתִּי בְּטוּב חֶלְבִּי / וְנָפַל צָר כְּעָלֶה



וְרַעְיָה שָׂמְחָה עִם דּוֹד



וְשַׁחוּ מַמְזְרֵי אַשְׁדּוֹד



וְשָׁב צִפּוֹר שְׁהוּא בוֹדֵד



וְסַרְעַפַּי בְּתוֹךְ קִרְבִּי / שַׁכַּר לִלַּה תְעַלַּה



וַתַוְחִיד אֻסמַהוּ חַסְבִּי / אַכִׄירֻן תֻׄם קַבּלַא:



אֲהוּבִי בָךְ אֲנִי מוֹדֶה / וְאָרוּץ עִמְּךָ



וְאֶעְזֹב הוֹן וְגַם שָׂדֶה / וְאָבוֹא אֶצְלְךָ



וְעַל פִּשְׁעִי אֲנִי מוֹדֶה / רְצֵה עַל עַבְדְּךָ



תֻעַוִּצׄנִי תֻזִיל גַלְבִּי / וּבַּעדַ אלצִּׄיקִ אַסלַא



וַאַנצֻׄר הַיאֲתַ אלקֻרבַּאן



וַיַסלִי כַַׄאטִרִי אלתַּעבַּאן



בִּאַהלַ אלְעִלם וַאלרֻּבַּאן



לַנַא יַפתוּן מַא יַגבִּי / אֲלַאחְבַּארַ אלְאֲגִׄלַּא



וּמַאהֻם מַנבַּע אלעַדׄבִּי / שַׁבִּיהַ אלנַּחל וַאַלחלַא:



אֲהוּבִי הַחֲמוּד נָתָן / עֲלֵי הַדַּל תְּבַקֵּשׁ



וְעַל בֵּיתִי וְהַמִּפְתָּן / שְׁלַח יָד עַל מְעַקֵּשׁ



וְנִכְנָס בָּהּ יְהֵא נִרְתָּן / וְתִהְיֶה לוֹ לְמוֹקֵשֹ



וְעָלָיו יִגְזְרָה תִשְׁבִּי / יְהֶא שׁוֹטֶה וְחוֹלֶה



וְלִי יָהּ יִתְּנָה כֵפֶל



וְיֶָרִים כָּל בְּנֵי שֵׁפֶל



וְאוֹרָה תִנְצְחָה אֹפֶל



וַזַאל אלכַׄוף וַאלרַּעבִּי / וַמֻלךּ אללַּה תַוַלַּא



וַחַיַּא כֻּלִּ מֻתַנַבִּי / וַטַאבּ מַן צַאם וַצַלַּא:



לְבוּשׁ סַיֵּע וְגַם לֶחֶם / לְמִשְׁתּוֹקֵק וְגַם דָּל



אֱלֹהַי עַבְדְּךָ רַחֵם / וְקַבֵּל נֶחֱדָל



וְאֶת כָּל לוֹחֲמַי תִּלְחַם / וְאַל נָא יַגְּדָל



יְמִין הָאֵל הֲלֹא תַצְבִּיא / לְשַׂגֵּב דֵָּל מְחַלֶּה



הֲדַר צִיּוֹן חֲזַת עֵינִי



וְאֵלִיָּה וְיִנּוֹנִי



וְאָז אָשִׁיר לְצוּר קוֹנִי



בְּשִׁיר חָדָשׁ נְאֻם נִיבִי / יֻנַבִּהּ לְאלגַפוּלַא



יֻגַׄלִּי אלהַם וַאלכֻּרבִּי / לַקַלבַּ אלמֻסתַדִׄלַּא:



שְׁבָא שִׁבְתַּי צְבִי וּצְלַח / עֲלֵי מַלְכוּתְךָ



נְבַזְבַּז לַיְּדִיד תִּשְׁלַח / לְבָבִי עַמְּךָ



וְאֵשׁ עַל אוֹיְבַי תִּדְלַח / וְיָרוּם הוֹדְךָ



גְּאוֹנֵיהֶם יְהוּ שֶׁבִי / יֻצַאלוּן אלבַּלַא



וַאַדעוּ אללַּהּ יֻעַוִּצׄנִי



מַעַא אלאַלאסבַּאט יַרחַמנִי



יֻדִׄלּ מַן כַּאן יַהמִלנִי



וַאַנסִי שַׁרּ חִטַאם שַׁגבִּי / אֲנַאל פַיץׄ אלכַּמַאלַא



יְהֶא שִׁבְתִּי וּמַצָּבִי / בְּצֵל נוֹרָא תְהִלָּה:



בְּהַר זֵיתִים גְּבוּל בָּקַע / יְדִידִי יוֹם עֲלוֹת



וּבֶן אָמָה אֱוִיל פָּקַע / יְרֵכָיו נוֹפְלוֹת



גְּאוֹן שָׂרוֹ תְּהוֹם שָׁקַע / וְרַגְלָיו צוֹלְלוֹת



שְׁלוֹמוֹת רַעֲיָה כִּתְבִי / וַכֻׄצִּי לִאלמֻהַלַּא



גַׄמִיע סִלסִלתִ יַעקוּבּ



תַקִי לִלַּהִּ מַחבּוּבּ



וּפִי אלתַּורַאתִ מַנסוּבּ



וּבִשְׂדֵה אַהֲבָה רִגְבִי / לְכִלְאַיִם אֲפַלֶּא



וְעָלָה מִשְּׁאוֹל קָטְבִי / בְּתִפְאֶרֶת מְסֻלֶּא:



זְרִיזִים סַנְדְּרֵי לֵידַע / הֲלָכוֹת רֹוְחוֹת



וְיוֹרוּן כָּל בְּנֵי דַּרְדַּע / בְּסוֹד הַמַּפְתְּחוֹת



וְקֶרֶן כּוֹזְבִים נִגְדַּע / וְעֵינַי זוֹרְחוֹת



וְאֶשְׁלוֹךְ עַל יְדִיד יָהְבִּי / אֲשֶׁר אוֹיְבַי מְכַלֶּה



וְהוּא חֶשְׁקִי וְכָל חֶפְצִי



וְעוֹז חַיַּי בְּתוֹךְ אַרְצִי



אֲשֶׁר הִזְרִיחַ לְבֶן פִּרְצִי



גְּשִׁי נַפְשִׁי וְהִתְרַבִּי / קְנִי חָכְמָה לְנַחְלָה



וְהַאְרִיכִי בְּמֶרְחַבִּי / לְרֹאשִׁי עָב יְטַלֶּא:



יַקוּלַ אלַעבּד אֲנַא אלחַלַּאם / בִּפַכּרִי פִי אללַּיַאלִי



לִעַארִף סַבּעַתַ אלאַקלַאם / אֻרִיד מַטלַבּ סוּאַלִי



וַאֻסמִי סִין אַלִף לַאם / וּמִים פִי אלנַּצׄם סַאלִי



וַלַד יוּסִף פַקַד אִגׄבִּי / לְמַטלַבּהוּ בִּסַהלַא



אֲקַו צוּרִי וְאֵלִי



בְּחַסְדּוֹ יִמְחֲלָה מַעְלִי



וְעָלָיו מַחֲמַל חֶדְלִי



יַטִיב עַיְשִׁי וַשֻׁרבִּי / בִּסַמחַ אלמֻסתַצִׄלַּא



מְחוֹל צוּרִי וְאָבִי / לִעַבּדֻן צַאם וַצַלַּא:
