


	
	שַׁחְתִּי לְעָפָר
	



חתימה: שלם בירב יוסף אלשבזי משתא






שַׁחְתִּי לְעָפָר מִיגוֹנָי

וְשֵׁנָה נָדְדָה מֵעֵינָי

הוֹי מֵחֲטָאַי וּזְדוֹנָי

הִרְהוּר לִבִּי וַעֲווֹנָי

וָאֶתְבּוֹנֵן בְּהֶגְיוֹנָי

אַשְׁרֵי אִישׁ יָרֵא אֶת יְוֹיָ:





לְדַלְתֵי צוּרִי דָּפַקְתִּי

וְשִׂמְחָה וְצַהְלָה שָׁכַחְתִּי

אוֹיָה לִי כִי עָבַרְתִּי

וְלֹא שָׁלַוְתִּי וְלֹא נַחְתִּי

מִצְווֹת אֱלֹהַי זָנַחְתִּי

זְכוֹר רַחֲמֶיךָ יְוֹיָ:





מִי יִתֵּן וְהָיָה לְבָבִי

דָּבֵק בְּסוֹד צוּרִי וְאָבִי

אוּלַי יִמְחֲלָה אֶת חוֹבִי

כִּי מֵעֲווֹנִי רַב כְּאֵבִי

שִׂים נְדִיבִים מוֹשָׁבִי

אוֹדְךָ יְוֹיָ:





בְּאַרְצוֹת אוֹיְבַי דִּכּוּנִי

שׂוֹנְאַי חִנָּם פְּצָעוּנִי

רָעָה בְטוֹבָה גְּמָלוּנִי

וּמִדְּבַר תּוֹרָה הִרְפּוּנִי

לָקְחוּ רְדִידִי וְהִכּוּנִי

רְפָאֵנִי יְוֹיָ:





יְחִידָתִי בְּלַיִל הוֹמָה

גּוֹלֶה וְנוֹדֵד בְּתוֹךְ תֵּימָא

פְּעָמִים נוֹפֵל בְּתַרְדֵּמָה

כִּי רְדָפַנִי אַף וְחֵמָה

וְעִתִּים נַפְשִׁי שׁוֹמְמָה

וָאֶתְחַנַּן אֶל יְוֹיָ:





רוֹצֶה תְּשׁוּבָה יִרְצֵנִי

וּמִבֵּין מְשַׂנְאַי יְקַבְּצֵנִי

לַחֲזוֹת בְּטוּבוֹ יְחַלְּצֵנִי

וּמִכִּשְׁלוֹנִי יְאַמְּצֵנִי

וְאָשִׁיב לֵץ הַמְּאִיצֵנִי

עָזִּי וְזִמְרַת יָהּ יְוֹיָ:





בָּטַחְתִּי בְּרָם עַל גֵּאִים

שׁוּלָיו הֵיכְלֵי רוּם מְלֵאִים

רוֹפֵא לְבָבוֹת נִדְכָּאִים

וְהַכֹּל בְּחַסְדּוֹ נֶהֱנָאִים

וַאֲנִי בְּתוֹךְ הַבָּאִים

אָשִׁירָה לַיְוֹיָ:





יִרְצֶה צוּרִי יְרַחֵמָה

לִבִּי וְנַפְשִׁי יְנַחֵמָה

צִיּוֹן כְּקֶדֶם יְקוֹמֵמָה

יַשְׁפִּיל בְּנֵי רָע וּמַשְׂטֵמָה

יַשְׁלִיךְ כְּאֶבֶן בְּמַרְגֵּמָה

יִכְלוּ עוֹזְבֵי יְוֹיָ:





וִדּוּי דְּבָרֵינוּ יְקַבֵּל

יְרַחֵם עָם עָנִי וְסוֹבֵל

עָלָיו צָרִים לְשׁוֹנוֹ מְנַבֵּל

וְהוּא מִתְאוֹנֵן וּמִתְאַבֵּל

וּבְהִתְגָּאוֹת רָע מְחַבֵּל

אַזְכִּיר תְּהִלּוֹת יְוֹיָ:





סוֹמֵךְ נוֹפְלִים יִסְמְכֵנוּ

וְיָגֵן בְּרַחֲמָיו עָלֵינוּ

לָגוּר בְּאֶרֶץ אוֹיְבֵינוּ

עַד בּוֹא דְּבָרוֹ לְקַבְּצֵנוּ

תָּגִיל וְתִשְׂמַח נַפְשֵׁנוּ

אָז תִּתְעַנֵּג עַל יְוֹיָ:





פִּתְחֵי רַחֲמִים וְהַצָּלָה

יִפְתַּח לְעַמּוֹ אֲשֶׁר גָּלָה

מִכּוֹבֶד עַבְטִיט חָלָה

תּוֹהֶה וְכוֹאֵב בְּחַלְחָלָה

מִמֵּי חֲסָדִים בְּחֵן דָּלָה

חַסְדֵי יְוֹיָ:





אֹרַח לְבָבִי כְּבוֹרֵחַ

דּוֹרוֹן לְרָשָׁע מְשַׁלֵּחַ

וְקוֹרֵא לְמוֹחֵל וְסוֹלֵחַ

עַמּוֹ כְקֶדֶם יְשַׂמֵּחַ

הַמַּזְכִּירִים אֶת יְוֹיָ:





לִבְשִׁי יְחִידָה לְשַׂק שָׁחוֹר

וּלְכִי בְּתֵימָן כְּמוֹ סוֹחֵר

לַעֲשׂוֹת רְצוֹן יוֹצְרִי בּוֹחֵר

כִּי בוֹא יָבוֹא לֹא יְאַחֵר

צוּר אֱלֹהֵי קֶדֶם וְאָחוֹר

כֹּל פָּעַל יְוֹיָ:





שׁוּלַמִּית שַׁלְּמִי עַל בֵּן

יוֹם זֶה חֶטְאוֹ יִתְלַבֵּן

עֵינַיִךְ בְּרֵכוֹת בְּחֶשְׁבּוֹן

שָׁמָּה גְּבוּלוֹת עַד דִּיבוֹן

שִׁפְעַת אֲדוֹנֵךְ הָרִבּוֹן

אֲשֶׁר גְּמָלָנוּ יְוֹיָ:





בַּחֲרִי בַּת צִיּוֹן תְּשׁוּבָה

יוֹם זֶה תְּפִלָּתֵךְ קְשׁוּבָה

עוֹרְרִי לִבְנֵי הָאֲהוּבָה

אֲשֶׁר מִבְּנֵי שִׁפְחָה עֲלוּבָה

תִּשְׂמַח וְרַבָּה הָעֲזוּבָה

כִּי כֹה אָמַר יְוֹיָ:





זֶה הַיּוֹם אֲשֶׁר קִוִּינוּ

וְלוֹ מֵאֶתְמוֹל אִוִּינוּ

סְגֻלַּת יַעֲקֹב הָבִינוּ

עִרְכוּ תְּהִלּוֹת וְהַרְנִינוּ

טַהֲרוּ לִבְּכֶם וְהָכִינוּ

יִשְׂרָאֵל נוֹשַׁע בַּיוֹיָ:





יַעְזְרֵנוּ שׁוֹכֵן מְרוֹמִים

וְיָגֵן עַל חֶבְרַת שְׁלֵמִים

יִמְחוֹל עֲווֹן בָּנִים אֲשֵׁמִים

הוֹמִים בְּקוֹלָם כַּיְּתוֹמִים

יַשְׁמִיעַ לְדָל נִחוּמִים

הַנִלְוָה אֶל יְוֹיָ:





מֵרוֹב עֲווֹנִי נִכְלַמְתִּי

וּבְמַעְצֵבָה שָׁכַבְתִּי

וְלַעְשׂוֹת רְצוֹן יָהּ תָּאַבְתִּי

וּמֵרוֹב הֲבָלַי שָׁכַחְתִּי

וְעֵת אֶזְכּוֹר כִּי עָזַבְתִּי

וָאֶצְעַק אֶל יְוֹיָ:





שִׁפְעַת אֱלֹהִים חָי יְזַלֵּף

עַל בֵּן מַכְחִישׁ וּמִתְחַלֵּף

מֵרוֹב דְּאָגָה מִתְעַלֵּף

נוֹפֵל בְּיַד אוֹיֵב מְסַלֵּף

חֲבָל עַל דָּל דְּמָעָיו דּוֹלֵף

אוּלַי יֶחֱנַן יְוֹיָ:





תֶּהֱמֶה בְּלַיִל חֲבַצֶּלֶת

עַל בֶּן מְלָכִים מְבוֹהֶלֶת

תָּרוּץ לְמוּל דּוֹד כְּאַיֶּלֶת

שֶׁפַע רַחֲמִים מְקַבֶּלֶת

וּבְעֵת רָצוֹן שׁוֹאֶלֶת

לַחֲזוֹת בְּנוֹעַם יְוֹיָ:





אֶקְרָא יַעֲנֵנִי בְּשׁוּבִי

צוּר יוֹדֵעַ צְפוּן חֻבִּי

וִישׁוֹבְבֵנִי לְמוֹשָׁבִי

כְּיוֹמֵי שַׁלְוָתִי וְטוּבִי

בְּעֵת סַרְעַפַּי בְּקִרְבִּי

אֶתְהַלֵּךְ לִפְנֵי יְוֹיָ:
