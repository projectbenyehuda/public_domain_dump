


	
	אֲהוּב לִבִּי בְּהַר סִינָי
	



חתימה: אבו שמעון אלשבזי






אֲהוּב לִבִּי בְּהַר סִינָי

הֱעִירַנִי בְהֶגְיוֹנָי

וְקוֹלוֹ שָׁמְעוּ אָזְנָי

בְּיָפְיוֹ צָהֲלוּ פָנָי:





בְּיוֹם מֹשֶׁה בְּחֵן נִגַּשׁ

וְהִגְבִּיל עָם אֲשֶׁר קִדַּשׁ

בְּיוֹם שִׁשִּׁי לְרֹאשׁ חֹדֶשׁ

בְּסִיוָן נִפְקְחוּ עֵינָי:





וְקִבַּלְנוּ שְׁנֵי לוּחוֹת

כְּאוֹר בָּהִיר מְצֻחְצָחוֹת

וְאוֹתִיּוֹת מְפֻתָּחוֹת

וְהֵן כֶּתֶר לְנֶאְמָנָי:





שְׁלוּמֵי לֵב יְפַרְסֵמוּ

לְדָת סִינַי יְקַדֵּמוּ

וְאֵל שַׁדַּי יְרוֹמֵמוּ

וּמַשְׂכִּילִים בְּכָל דִּינָי:





מְלֻמָּדִים בְּלֵב נִהְנָה

בְּתוֹרַת אֵל וְהַמִּשְׁנָה

שְׂרִידִים בַּעֲלֵי אָמְנָה

וּמַזְהִירִים לְרַעְיוֹנָי:





עֲדַת קֹדֶשׁ שְׁחַרְחֹרָה

וּמַנְהֶרֶת בְּאוֹר תּוֹרָה

בְּצֵל שַׁדַּי שְׁהוּא נוֹרָא

מְקוֹר חַיַּי וּמַעְיָנָי:





וּמַה נָּעִים הֲדָרֵנוּ

בְּיוֹם נָשׁוּב לְעִירֵנוּ

וְרוּחַ יָהּ תְּעִירֵנוּ

בְּהֵיכָלַי וְאַרְמוֹנָי:





נְחֵנִי יָהּ וְזַכֵּנִי

וְאֶרְאֶה מַחֲמַד עֵינִי

יְפַת עַיִן וְאַדְמוֹנִי

גְּאוֹן עֻזִּי וְחֶזְיוֹנָי:





אֲנִי לִבִּי תְּמוֹל צָפָה

הֲדַר כַּלָה מְאֹד יָפָה

וְחָתָן זֶה בְּרֹאשׁ חֻפָּה

וְאָז שַׁרְתִּי בְּנִגּוּנָי:





לְבוּשׁ דַּעַת וְטוֹב שֵׂכֶל

וְכַלָּתוֹ כְּמוֹ מִיכַל

וְאִישׁ צַדִּיק וְאַדְרִיכַל

בְּרֹאשׁ שׁוּרָה לְהַזְמִינָי:





שְׁתוּ דוֹדִים לְיֵין, שִׁכְרוּ,

וְדָת תּוֹרַת אֱמֶת שִׁמְרוּ

יְרוּשְׁלַם עִמְכֶם זִכְרוּ

בְּשִׂמְחָתִי וּבִרְנָנָי:





בְּעֵת שֶׁתֹּאמְרוּ הַזָּן

וְיִשְׂבַּע גּוּף אֱוִיל רַגְזָן

וְיֵצֶר טוֹב כְּמוֹ חַזָּן

לְאֵיבָרַי וּבִנְיָנָי:





זְכוּת אָבוֹת תְּחַיֵּנוּ

וְנָשׁוּבָה לְאַרְצֵנוּ

שְׁנֵים עָשָׂר שְׁבָטֵינוּ

וְכֹהֵן שָׁם וְגַם סִגְנָי:





יְלוּד מַשְׁתָּא יְהֶא נִזְמָן

בְּבוֹא מֶלֶךְ גְּבִיר נֶאְמָן

וְנִטְעַם מִצְּלוֹחִית מָן

וְאֶתְבּוֹנֵן בְּמַצְפּוּנָי:





לְךָ שָׁלוֹם כְּמוֹ נָהָר

יְלוּד פֶּרֶץ בְּאוֹר נִזְהָר

כְּיוֹם זָרַח בְּרֹאשׁ הָהָר

לְבֶן עַמְרָם וְלִזְקֵנָי:
