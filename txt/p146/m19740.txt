


	
	שְׁכוּרָה מֵחֲמַר חֹמֶר הֲפִיצִי
	



החתימה: שלם שבזי שץ




שְׁכוּרָה מֵחֲמַר חֹמֶר הֲפִיצִי / הֲבָלַיִךְ וְהִכּוֹנִי בְשִׂכְלִי:



לְכַמָּה הַזְּמָן יַתְעֵךְ וְיַפְרִיד / לְכוֹחוֹתַי בְּעֵצוֹתָיו
וְיַפְלִיא:



מְנוּחָתֵךְ בְּעֵדֶן גַּן אֱלֹהִים / עֲשִׂי מִצְוֹת אֱלֹהַיִךְ
וְסִבְלִי:



שְׁאוֹן יִצְרֵךְ בְּעָרְמָה אַל יְסִיתֵךְ / וְיַכְאִיבֵךְ בְּרַע סוּרוֹ
וְיַחְלִיא:



בְּשׁוּבֵךְ תִּשְׁכְּנִי לָךְ בָּאֳהָלִים / וְשָׁמָּה הִיא מְדֻשָׁתִי
וְצַהְלִי:



זְרִיזִים יַעֲנוּ שִׁירָה נְעִימָה / יְעִירוּנִי לְהִתְקוֹמֵם
בְּנָפְלִי:



יְחִידָתִי, בְּמִשְׁמַרְתֵּךְ עֲמֹדִי / וְצַפִּי יוֹם כְּלוּלוֹתַי
וְגִילִי:



שְׁכִינָה עַל מְקוֹמֵיהֶם תְּרַחֵף / גְּמוּלֵיהֶם בְּשָׁלוֹם יָעֳרַב
לִי:



צְבָאִי לֹא לְרֹב יָמִים יְכוֹנֵן / וְלֹא יִשְׁפּוֹת לְמַכְלוּלִי
וְתִלִּי:
