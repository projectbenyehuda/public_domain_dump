


	
	מעמד ה: לְבָבִי מֵחֲרָדָה
	



החתימה: לשלם הקטן






לְבָבִי מֵחֲרָדָה / בְּגוּפִי נִתְעַלָּם

עַל צִיר הַתְּעוּדָה / הַגְּבִיר הַנִּשְׁלָם

הוֹי עַל כְּלִי חֶמְדָּה / מְסַלְסֵל סוֹד נֶעְלָם

יְהִי שֵׁם יְוֹיָ מְבוֹרָךְ / מֵעַתָּה וְעַד עוֹלָם:





שָׁקַט הַמִּדְרָשׁ / וְגַם בֵּית הַכְּנֶסֶת

עַל זֶה שְֶׁנִּגְרָשׁ / שְׂשׂוֹנָהּ מוֹאֶסֶת

כִּי מָוֶת בָּלַשׁ / וּבָא לוֹ בְּהֶסֶּת

וְאֵיכָה מִמֶּנּוּ / נוּכַל נִתְעַלָּם:





לַכֹּל הוּא שׂוֹטֵם / וְשׁוֹלֵט כַּמֶּלֶךְ

לָרוּחַ אוֹטֵם / וּבַגּוּף לֹא תֵלֵךְ

בְּאַף הַחַי חוֹטֵם / בְּזַעְפּוֹ עַד יֻשְׁלַךְ

עוֹכֵר הַבִּרְיוֹת / וּמֵסִיר אֶת צִלָּם:





מַה-יַּעֲשֶׂה אָדָם / וְאֵין בְּיָדוֹ יְכוֹלֶת

וְזֶה שִׂטְנוֹ קָדַם / שְׁהוּא אֵשׁ אוֹכֶלֶת

וְכָל הַיצוּר יְסוֹדָם / מִגּוּשׁ נִגְבֶּלֶת

וְחוֹזְרִים לִיסוֹדָם / וְיִתְחַדֵּשׁ אֶבְלָם:





הֵיךְ תְּהֶא לוֹ תְקוּמָה / אִישׁ מִן הֶעָפָר

כִּי נַפְשׁוֹ עֲמוּמָה / מֵעֵת שֶׁנּוּצָּר

וּבְעָווֹן וְאַשְׁמָה / יִצְרוֹ בוֹ יִפְצָר

וּמִתִּגְרַת מַעֲשָׂיו / יָשׁוּב דָּךְ נִכְלָם:





קוֹדֶם יְצִירָתוֹ / מִטִּפָּה סְרוּחָה

וְאַחַר בְּרִיאָתוֹ / צוּרָה נִמְתָּחָה

וְיוֹם יָבוֹא עִתּוֹ / וְנַפְשׁוֹ נִלְקָחָה

שָׁב אֶל הֶעָפָר / מִתְוַדֶּה וְנֶאֱלָם:





טוֹב יָחִיל וְדוּמָם / וְאִם לָאו מַה-יָּשִׁיב

לְאַחַר שֶׁנֶּעְמָם / בַּהֲבָלִים יַחְשִׁיב

וּמָוֶת פִּיו זָמָם / וְאָזְנוֹ מִלְּהַקְשִׁיב

וְאֵידוֹ אָרַב לוֹ / וְאֶת רֹאשׁוֹ הָלָם:





נַחֵם אֲבֵלֵינוּ / אֱלֹהִים מִמְּרוֹמָךְ

וְהַשְׁקֵף לְבָרְכֵנוּ / מִתּוֹךְ שְׁמֵי מְרוֹמָךְ

וַעֲנֵה קָרְאֵנוּ / וַעֲשֵׂה לְמַעַן שְׁמָךְ

וּבְיוֹם שַׁוְעֵנוּ / יָהּ אַל תִּתְעַלָּם:
