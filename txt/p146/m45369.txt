


	
	שׁוּבוּ שׁוֹבָבִים
	



חתימה: שלם בירב יוסף משתא






שׁוּבוּ שׁוֹבָבִים / וְחַכּוּ לְצוּר יִשְׁעֲכֶם

לְשַׂמֵּחַ נֶעְצָבִים / וּלְהַחֲיוֹת רוּחֲכֶם

בְּנֶחָמַת נְדִיבִים / אֲשֶׁר נִבְּאוּ בְגִינְכֶם

וַיִּכְתְּבוּ כְּתָבִים / לְהַשְׁמִיעַ אָזְנְכֶם

אַתֶּם נִצָּבִים / הַיּוֹם כֻּלְּכֶם:





לְיוֹם הַדִּין הֲכָנָה / לְבַת גַּלִּים שָׁבָה

קְהַל עֲדַת מִי מָנָה / בַּמָּרוֹם נִצָּבָה

מֵרֹאשׁ הַשָּׁנָה / וְעַד עֶרֶב עֲרָבָה

וּבַת מְלָכִים חוֹנָה / וְהִנֵּה דוֹדִי בָא

וְשָׁם רִבּוֹא רְבָבָה / וּרְאִיתֶם וְשָׂשׂ לִבְּכֶם:





מְחִילָה וּסְלִיחָה / חֲבַצֶּלֶת תְּצַפֶּה

לְצוּרָהּ תָּשִׂיחָה / וְלוֹ יְשַׁבַּח כָּל פֶּה

עֲדַת אֵל תַּרְוִיחָה / בַּחֲצַר דּוֹד יָפֶה

וְקוֹדֵם בּוֹא צוֹפֶה / רִיבוּ בְּאִמְּכֶם:





בֶּהָלָה אַחֲזָה / בְּנֵי שֵׂעִיר וְאָמָה

וּמִדַּת דִּין נָתְזָה / לְמִגְדָּלָם דָּמָה

חֲיָלֵיהֶם בָּזְזָה / עֲדַת אֵל בְּיָד רָמָה

וּבְשׁוּבָהּ עָלְזָה / לְדוֹדָהּ הַשְׁלָמָה

וְעָלֶיהָ שָׂמָה / טָרְחֲכֶם וּמַשַּׂאֲכֶם:





יֶשַׁע מִתְבַּשְּׂרִים / בְּנֵי יַעֲקֹב כֻּלָּם

וְלִבּוֹתָם מְשַׂבְּרִים / לְצוּר קוֹנָם גּוֹאֲלָם

וְחֵן רַעְיָה מְעוֹרְרִים / בְּדִבְרֵי מַהֲלָלָם

מְשַׁבְּחִים וּמְזַמְּרִים / וּמַצְלִיחִים עֲמָלָם

וְאוֹמְרִים בְּהִתְפַּלְלָם / שִׁיתוּ לִבְּכֶם:





רַחֲמִים וִישׁוּעָה / לְעָם גּוֹלִים בְּעָנְיָם

גַּנָּה הַנְּטוּעָה / בִּנְהָרוֹת דָּכְיָם

בִּצְדָקָה זְרוּעָה / כְּבוֹד הָאֵל מְחַיָּם

וּמַעְתִּיקֵי שְׁמוּעָה / יְקָר שְׁכִינָה עֶדְיָם

וְעוֹנִים בְּנִיב צִבְיָם / יְהִי כֵן יְוֹיָ עִמָּכֶם:





בְּעֵת רָצוֹן פָּצָה / פִּי שְׁבָח לְצוּר קוֹנִי

וְחֵן וְשֵׂכֶל מָצָא / בְּנוֹעַם הֶגְיוֹנִי

וְנַפְשִׁי נוֹעֲצָה / תְּשׁוּבָה תְּלַמְּדֵנִי

בְּהֶבְלֵי שְׂמֹאל קָצָה / וּתְעוֹרֵר הֲמוֹנִי

וְכַלָּה וְנֶחֱרָצָה / הָשְׁלַח בְּשׂוֹטְמֵנִי

יָהּ יַשְׁמִיעֵנִי / וְהִתְהַלַּכְתִּי בְּתוֹכְכֶם:





יִכַָּתְבוּן יְדִידִים / בְּסֵפֶר זִכָּרוֹן

לְחַיִּים וְעוּדִים / יָשׁוּבוּ לְבִצָּרוֹן

כְּלַל בְּנֵי הַיְּהוּדִים / סְגֻלַּת הַיִּתְרוֹן

שְׁבָח וְהַלֵּל תַּקְדִּים / חֲבַצֶּלֶת שָׁרוֹן

עֲבוּר מֹשֶׁה וְאַהֲרֹן / חַיִּים כֻּלְּכֶם:





וַתִּשְׁמַע אָזְנִי / שֶׁמֶץ מִנְּבִיאִים

כִּי אָשׁוּב לְכַנִּי / בְּיוֹם קִבּוּץ טְלָאִים

לְאֶרֶץ מַעֲדַנִּי / מֵאֶרֶץ רְפָאִים

וְטַל אוֹרוֹת גַּנִּי / יְחַיֶּה נִדְכָּאִים

צְאוּ אֲסִירֵי כְּלָאִים / וּתְחִי נַפְשְׁכֶם:





סַבְלָה גְּוִיָּתִי / יִסּוּרֵי הַזְּמָן

כִּי בְּרוֹב אַשְׁמָתִי / חָשׁ לִבִּי וְנִטְמָן

וְגָרַם לְרַעְיָתִי / לַעֲזוֹב דּוֹד נֶאֱמָן

וְשָׁלַט בֶּן אֲמָתִי / עַל הֲמוֹנַי בְּתֵימָן

פְּזוּרִים תְּנוּ סִימָן / וִישַׁבְתֶּם לָבֶטַח בְּאַרְצְכֶם:





פְּדוּת לִבְנֵי גְּבֶרֶת / בִּימִינוֹ יָחִישׁ

וְיִזְכּוֹר בְּרִית כָּרֵת / רָם וְנִשָּׂא, לְהַלְבִּישׁ

עֲטְֶרֶת תִּפְאֶרֶת / לְיִנּוֹן, וּלְהַכְתִּישׁ

לְעֵשָׂו אֲשֶׁר חָרַת / מַשְׂטֵמָה וְהִבְאִישׁ

וּפֶרֶא יִשְׂבַּע רִישׁ / לְכוּ בְּאוּר אֶשְׁכֶם:





מֵאַרְבַּע פֵּאוֹת / יְקַבֵּץ לַעֲדָתוֹ

אֱלֹהֵי הַצְּבָאוֹת / מְשַׁלְּשִׁים קְדֻשָּׁתוֹ

וּלְרוּחוֹת נְכֵאוֹת / מְחַיֶּה בְּאַהֲבָתוֹ

וְכֻלָּן נִקְרָאוֹת / לְעֵת בּוֹא יְשׁוּעָתוֹ

וְנָם לִסְגֻלָּתוֹ / לְכוּ אֶל אַרְצְכֶם:





שַׁפְרִיר מָעֻזּוֹ / יַלְבִּישׁ גְּבֶרֶת עוֹז

וְלוֹ נִפְלְאוֹת רָזוֹ / בְּחוֹתָמוֹ גּוֹנֵז

וְעַל סְגֻלַּת מַחֲזֵי / נְבִיאִים הוֹד יְפַרְשֵׁז

וְשֶׁמֶץ בִּרְמָזוֹ / קוֹל דְמָמָה אוֹחֵז

אִם יִהְיֶה נְבִיאֲכֶם:





תַּחְתּוֹר לִי חֲתִירָה / אֱלֹהַי לְקַבְּלֵנִי

כִּי לְבָבִי יִירָא / בְּהִפָּקֵד עֲווֹנִי

בְּרַחֲמִים הָאִירָה / נְתִיבַי לְמָשְׁכֵנִי

וְעָלַי הָעִירָה / זְכוּת אַהֲרֹן אֲדוֹנִי

וֶאֱמוֹר לְסָמְכֵנִי / אָנֹכִי אָנֹכִי הוּא מְנַחֶמְכֶם:





אוֹרְחוֹתַי יְכוֹנֵן / צוּר, מַחְסֶה לְדַלִּים

וְאָשׁוּב וְאֶתְבּוֹנֵן / בְּמַעֲשַׂי הַנִּקְלִים

כְּשִׁכּוֹר מִתְרוֹנֵן / פְּשָׁעָיו מִתְגַּלִּים

אֱלוּלֵי יָהּ חוֹנֵן / כַּיָּם סוֹבֵל גַּלִּים

הָבוּ לַיוֹיָ בְּנֵי אֵלִים / לְהַשְׁמִיעַ בַּמָּרוֹם קוֹלְכֶם:
