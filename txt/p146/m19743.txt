


	
	אֲנִי הַיּוֹם בְּגָלוּתִי מְסַבֵּר
	



חתימה: אלשבזי משתא מכונה




אֲנִי הַיּוֹם בְּגָלוּתִי מְסַבֵּר / לְמַולַא אַלכֻּל חְוַא כֻּל
אַלוְגׄוּדִּי:



לְעוֹלָמוֹת שְׁנַיִם הוּא מְקַשֵּׁר / וְכֻּל מַוְגׄוּדּ גְׄעַלהוּ פִי
חְדּוּדִּי:



שְׁבָח לִשְׁמֵיהּ תְּלָת דַּרְגִּין תְּסַדֵּר / בְּאַעלַאהַא בְּפַרדּוּס
אַלוְדּוּדִּי:



בְּחֶסֶד עַד גְּבוּל רַעְיָה מְיַשֵּׁר / בְּפַיץׄ חַאצִל מֻזַאוִגׄ
לִאלפְרוּדִּי:



זֹהַר זִיו שְׁכִינָה שַׁו וְאַנְהֵר / בְּאַגׄיַאשֻׁן מִן אַמלַאךּ
אַלגׄוּדִּי:



יְקוֹד גַּלְגַּל לְרָע דָּלִיק וְנָהֵר / גְׄעַל פִיהַא כּוַאכִּבּ
כַּאלְפנוּדִּי:



מְקַטְרְגִין דְּחָא לְגוּשׁ עַפְרָא וְאַקְטֵר / בְּאַגׄסַאמֻן
תְּהַאנִיַה לִאללְּחוּדִּי:



שׁוּפְרָא דְּנַעֲמָה וְלֵילִין שַׁו וְאַזְעֵר / תְּדַׄרוּן מִנְּהֻם
כַׄמס אֻמַם צְדּוּדִי:



תְּעֵרַבְרְבִין עֲלֵיהוֹן אִתְגְּזֵר / מַאתּוּן פִי אַלבַּר קַוְמֻן
חְסוּדִּי:



אַתְּעַר אְלָהִין לְנֹחַ דְּאִתְכְּשֵׁר / מִן נַסל טַוְפַאן דַאךּ
אַלגְׄחוּדִּי:



מְדוֹר זֵד דְּנִמְרוֹד בְּמִגְדְּלָא דְּאַסְדֵּר / מִן לִבּן מַחבּוּז
צַנו אַלזְּהוּדִּי



כָּל חַד אֶתְפְּלִיג בְּלִישָׁן מְדַבֵּר / סַבּעִין אֻמַּהּ דַאתּ
אַלכְּיוּדִּי:



וְאִתְנְהַר אַבְרָם כְּנָהִיר דְּגָבֵר / כַׄאלַף לְנִמרוֹד עַן
אַלסְגׄוּדִּי



נַסְבּוּהּ לְאַתּוּן דְּנוּרָא אֶתְקְדֵר / נַגַׄאהּ רַבִּי עַן
אַלכְּמוּדִי:



הוּא גְּזָר עָרְלְתָא אָת קְיָמָא נָטֵר / וְאַעטַאהּ יִצחַק נַאל
אַלסְּעוּדִּי:



אֲבוּהִי דְיַעֲקֹב דְשִׁבְטַיָּא נָהֵר / קַוְמֻן זְכִּיֻּן שִׁבּהּ
אַלאְסוּדִּי:



שְׁלַח לוֹן מֹשֶׁה וַאֲמַר לֵיהּ: אִזְדְּהֵר / עַרִּף לְקַוְמִי גַאיַת
עְהוּדִּי:



שְׁלָמָא לְעַמָּא סְחוֹר סְחוֹר מִתְּעֵר / מַא לַאח בַּארִק גַיְתׄ
אַלרְעוּדִי:
