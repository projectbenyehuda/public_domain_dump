


	
	הַזְּמָן עָבַר עֵת הַקֵּץ הִגִּיעַ
	





































הַזְּמָן עָבַר / עֵת הַקֵּץ הִגִּיעַ / נְעִים קוֹל הַמְּבַשֵּׂר יַשְׁמִיעַ:


נֵצֵא הַשָּׂדֶה! בַּכְּפָרִים נָלִינָה / גַּם גִּבְעוֹת אַרְצֵנוּ עָלוֹת נָהִינָה:

מַה לָּנוּ, אַחַי, שֶׁבֶת בַּגָּלוּת / מְקוֹם אַךְ נְהִי מוֹשַׁב כָּל יְלָלוּת.

נֵצֵא הַשָּׂדֶה, בַּכְּפָרִים נָלִינָה / וְשִׁיר חָדָשׁ נָשִׁיר אֶת שִׁיר צִיּוֹנָה:

הֵן דָּשְׁאוּ נְאוֹת שָׂדֶה כַּרְמֶל וְיַעַר / שִׂמְחָה עַל כָּל פָּנִים מִזָּקֵן עַד נַעַר:

יִשְׂמְחוּ וְיָגִילוּ עַתָּה בְּבוֹא הָאָבִיב / אַף גִּילַת וְרַנֵּן יַחְדָּיו מִסָּבִיב:

מַה לָּנוּ אַחַי דַּבֵּר בִּשְׂפַת הַגְרִית / הֲלֹא הִיא כְשִׁפְחָה לִשְׂפָתֵנוּ הָעִבְרִית:

יִשְׂמְחוּ וְיָגִילוּ נַעֲרֵי הָעִבְרִים / הַשּׁוֹקְדִים לִלְמֹד אֶת שְׂפַת הָעִבְרִים:

וְאֶת חֲתָנַיִךְ קוֹרְאִים יַחַד עִם הַבֵּאוּרִים / דֵּעַ בְּשִׂים לֵב לְבַל הֱיוֹת בּוֹעֲרִים:

הַרְבֵּה שִׁירִים נָשִׁיר מִשִּׁיר צִיּוֹנָה / בִּשְׂפַת עֵבֶר הָעוֹלָה עַל כֻּלָּנָה:



דְּבַר ה׳ נִלְמַד, נַעֲשֶׂה וּנְקַיֵּמָה / יִתֵּן אֶת רוּחוֹ עָלֵינוּ וְנֶחְכָּמָה:



חֲסָדָיו נַזְכִּירָה, וּשְׁבָח לוֹ נַקְדִּימָה / יוֹצֵר כֹּל תְּהַלֵּל לוֹ כֹּל נְשָׁמָה:
