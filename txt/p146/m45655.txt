


	
	לִבִּי בְּתוֹךְ קִרְבִּי נִמְסָה
	



חתימה: לאלשבזי






לִבִּי בְּתוֹךְ קִרְבִּי נִמְסָה / יוֹם גָּלִיתִי בְּיוֹם עֶבְרָה

רַצִּי נָא עַל הֲדַסָּה / לִפְנִי צוּרֵךְ יְהֶא לְעֶזְרָה

עַל פְּשָׁעֵינוּ תְּהִי מְכַסָּה / וְנָשׁוּב לְאַרְצֵנוּ בְּהַדְרָה

כּוֹס יְשׁוּעוֹת אֶשָּׂא / וּבְשֵׁם יְהֹוָה אֶקְרָא:





אַהֲבַת יֵחוּד אֱלֹהֵינוּ / תְּשַׂגֵּב דַּלִּים נִכְלָמִים

וְיִתֵּן מִשְׁאֲלוֹת לִבֵּנוּ / בְּחֵן וּבְחֶסֶד וּבְרַחֲמִים

כִּי בַיוֹיָ מַחְסֵנוּ / יִמְחֹל וְיִסְלַח לָאֲשָׁמִים

יַצִּילֵנוּ מִן הַקָּמִים / וְנַעְלֶה לְהַר צִיּוֹן חֶבְרָה:





לִבִּי יְעִירֵנִי בְּחֶשְׁקִי / אָקוּם וְאֶקְרָא יַעֲנֵנִי

אֶשְׂמַח וְאָשִׁירָה בְּעָסְקִי / לְדוֹבֵב שְׂפָתַי וּלְשׁוֹנִי

תָּמִיד אֶשְׂמַח בְּחֶלְקִי / וְאֶשְׁאַל צוּרִי יַעְזְרֵנִי

וְרוּחַ אֱלֹהִים תְּעִירֵנִי / בָּרוּךְ מַמְצִיא לְכָל נִבְרָא:





שָׁכַבְתִּי וָאִישָׁנָה / וּבִשְׁבָח לְצוּרִי הֲקִיצוֹתִי

קָמְתִּי בְּיוֹם רֹאשׁ הַשָּׁנָה / לִשְׁאַל סִפְקִי אֲנִי וּבֵיתִי

אֵלֶךְ לִי אָנֶה וְאָנָה / וְחֵן אֵל שַׁדַּי מָצָאתִי

וְאָמְנָם כִּי חָטָאתִי / וְחֶטְאִי גָּרַם עוֹל וְעֶבְרָה:





בָּאתִי לְחַלּוֹת פְּנֵי צוּרִי / כִּי הוּא רוֹפֵא לְכָל בָּשָׂר

יַאֲזִין אֲמָרַי וְדִבּוּרִי / כִּי קָנְתָה נַפְשִׁי מוּסָר

יָקוּם יְקַנֵּא בַעֲבוּרִי / בִּזְכוּת שְׁבָטַי שְׁנֵים עָשָׂר

יְהֹוָה רוֹעִי לֹא אֶחְסָר / הוּא עֶזְרָתִי בְּיוֹם צָרָה:





זְכוֹר דּוֹדִי רוֹב רַחֲמֶיךָ / וּסְלַח נָא לַעֲווֹנֵינוּ

הוֹשִׁיעָה אֶת עַמֶּךָ / לְמַעֲנָךְ לֹא לְמַעֲנֵנוּ

אָז יֵחָלְצוּן יְדִידֶיךָ / לְתוֹךְ אֶרֶץ נַחֲלָתֵנוּ

בִּזְכוּת אַהֲבַת אֲבוֹתֵינוּ / יִרְצֶה צְבִי לַחְשׁוֹק בְּעַפְרָה:





יוֹם כִּפּוּר עִנּוּי נְפָשׁוֹת / וְשׁוֹאֵל חַסְדּוֹ לֹא יֶחְדָּל

אִסּוּר אֲכִילָה וּשְׁתִיָּה / רְחִיצָה סִיכָה וּנְעִילַת סַנְדָּל

וְאִסּוּר תַּשְׁמִישׁ הַמִּטָּה / עַל עַם קֹדֶשׁ הַנִּבְדָּל

וְאַשְׁרֵי הַמַּשְׂכִּיל אֶל דָּל / בְּיוֹם רָעָה יִמָּלֵט בְּטַהְרָה:
