


	
	נֶפֶשׁ יְחִידָה אֵיךְ לְסוֹד אֵל תִּצְפְּנִי
	



החתימה: שלם בן יוסף




נֶפֶשׁ יְחִידָה אֵיךְ לְסוֹד אֵל תִּצְפְּנִי / עוּרִי יְחִידָתִי
וְלָמָּה תִישְׁנִי:



שַׁאְלִי נְוַת שָׁלוֹם וְחִבְרִי עִם בְּנֵי / חָכְמָה וְתוּשִׁיָּה וְאָז
תִּבָּחֲנִי:



לִקְרַאת פְּנֵי צוּרֵךְ בְּתוֹם הִתְפָּאֲרִי / תַּחְסִי בְּצֵל הָאֵל
וְגַם תִּתְלוֹנְנִי:



מֵרוּם שְׁחָקִים עַד שְׁפָלִים תִּמְשְׁכִי / שֶׁפַע אֱלֹהַיִךְ וּבוֹ
תִּתְעַדְּנִי:



בִּחְרִי לְדֶרֶךְ טוֹב וְאַל תִּשְׂטִי אֱלֵי / דֶּרֶךְ בְּנֵי עַוְלָה
וּבוֹ תִּתְיַגְּנִי:



נַחַת וּמָנוֹחַ לְעַצְמֵךְ בַּקְּשִׁי / בִּחְרִי לְעֵץ חַיִּים וּבוֹ
הִשָּׁעֲנִי:



יוֹשֶׁר פְּעָלַיִךְ לְמוּל אֵל תַּעְרְכִי / חָכְמַת בְּנֵי דַעַת
יְחִידָתִי קְנִי:



וּנְוַת הֲדַר מֶלֶךְ וְחַשְׁמַלֵּי מְרוֹם / יִהְיֶה מְכוֹן שִׁבְתֵּךְ
מְקוֹם הוֹד תֻּזְמְנִי:



סָבִיב יְרוּצוּן מַלְאֲכֵי רוֹם אַהֲבָה / שָׁם מַחֲנוֹת אַרְבַּע
וְשָׁמָּה אוֹפְנִי:



פָּצִים בְּדִבְרֵי שִׁיר לְנוֹכַח יוֹצְרָם / יִגְדַּל וְיִשְׁתַּבַּח
שְׁמוֹ צוּר כּוֹנְנִי:
