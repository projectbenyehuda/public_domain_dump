


	
	אֱלֹהֵי יִשְׂרָאֵל
	



חתימה: אבו שמעון אלשבזי






אֱלֹהֵי יִשְׂרָאֵל / יְצַוֶּה רוּמִיאֵל / לִסְגּוֹר פִּי מְחַבְּלִים

בֵּין לְבָנָה / לְוִילוֹן חָנָה / וּבוֹ דִּין וְרַחֲמִים נִכְלָלִים



וּמֵחַשְׁמַל / שְׁעַ״ח רְקִיעִים תִּגְמַל / חֶסֶד וְדִין פּוֹעֲלִים



וּמְנַהֲגָם מַלְכָּם / עֵת רוּצָם בְּדַרְכָּם / וּמִשִּׁפְעוֹ מְקַבְּלִים

וּמַרְאֵה הַצּוּרוֹת / עַל כֵּס יָהּ קְשׁוּרוֹת / בְּאוֹפָן וּסְגַלְגַּלִּים

כָּל בְּרוּאֵי מַעְלָה וּמַטָּה / לַיוֹיָ מְהַלְלִים:





בְּחַסְדּוֹ קָשַׁר / רַחֲמִים וְדִין יוֹשֶׁר / וְקָו הַמַּכְרִיעַ

וְיָרַד סִלּוֹן / עֲדֵי הַוִּילוֹן / מִזֶּה וְזֶה יַשְׁפִּיעַ

וּמֵהֶן תַּחְתּוֹנִים / בְּיַחַד נִהְנִים / יוֹרְדִים בְּהַצְנֵעַ

וּנְפָשׁוֹת חַיּוֹת / קְשׁוּרוֹת בַּגְְּוִיּוֹת / בְּעוֹלָם זֶה חוֹנִים

וְנַפְשׁוֹת בְּנֵי אָדָם / בְּכִסֵּא יְסוֹדָם / הָאוֹר תַּזְרִיעַ

וְנַפְשׁוֹת שְׁאָר חַיִּים / מִן הַגַּלְגַּלִּים:





וְאָדָם הָרִאשׁוֹן / נַעַר מִפִּישׁוֹן / הִצְמִיחַ שִׁבְעִים לָשׁוֹן

וְסוֹד תּוֹרָה גִּלָּה / לְיִשְׂרָאֵל סְגֻלָּה / כְּעֵץ לַח מִתְדַּשֵּׁן

וְשִׁבְעִים הָאֻמּוֹת / אֲשֶׁר נִקְּבוּ בְּשֵׁמוֹת / לְשָׂרֵי הַחֶלְשׁוֹן

וּסְגֻלַּת הֲמוֹנָי / קוֹדֶשׁ לַיוֹיָ / יִנְצְרֶנּוּ כְּאִישׁוֹן

מְיַחֲדִים שֵׁם כְּבוֹדוֹ / וְטוּבוֹ וְחַסְדּוֹ מְיַחֲלִים:





שִׁבְטֵי יָהּ עוּרוּ / וְאַל נָא תָסוּרוּ / [……שַׁר]

הִכּוֹנוּ דַּרְכְּכֶם / לַעֲשׂוֹת רְצוֹן מַלְכְּכֶם / אֲשֶׁר דִּגְלְכֶם קָשַׁר

סוּרוּ מֵהַוּוֹת / וְעִסְקוּ בַּמִּצְווֹת / יְהֶא לָכֶם כּוֹשֶׁר

וְאַשְׁרֵי נְשׂוּי פֶּשַׁע / בְּשׁוּבוֹ יִוָּשַׁע / וּמַה יּוֹעִיל עוֹשֶׁר



מְעַשֵּׂר מָמוֹנוֹ / וְהַשָּׁב מֵעֲווֹנוֹ / תְּשׁוּבָתוֹ מְקַבְּלִים:





מַה תֶּהֱמִי נַפְשִׁי / מִתְּמוֹל הִתְקַדְּשִׁי / בֵּין בַּעֲלֵי תְּשׁוּבָה

בָּעֲשִׂירִי בְּתִשְׁרֵי / תְּשׁוּבָתֵךְ בִּחְרִי / הַבַּת הָאֲהוּבָה

יוֹם מֹשֶׁה הֻרְגַל / וְנִמְחָל עֲווֹן עֵגֶל / לַעֲדָתִי הַחֲשׁוּבָה

וְהֻקְבָּע יוֹם מְחִילָה / בְּקוֹל שִׂמְחָה וְגִילָה / לְרַפֹּאת כָּל מְשׁוּבָה

וּמִדָּתוֹ בִּינָה / אֲשֶׁר בָּהּ מִתְנַהֲלִים:





עֲקֶרֶת בָּנִים דְּלִי / בֵּן בְּגָלוּת צוֹלְלִי / וְשַׂגְּבִי לְאֶבְיוֹן

וְהָעִירִי יְשֵׁנִים / בְּחֵן אֵם הַבָּנִים / כִּי דָלְלוּ בְּשִׁבְיוֹן

בִּזְכוּת אֵיתָנִים / אָבוֹת רִאשׁוֹנִים / וּשְׂרִידֵי הֶגְיוֹן

בִּמְחִילָה בַּשְּׂרִי / וְקוּמִי הִתְעוֹרְרִי / בְּסֵפֶר גִּלָּיוֹן

עֲלֵי זָדוֹן וְשִׁגְיוֹן / מִתְוַדִּים וּמִתְפַּלְּלִים:





וּבְשׂוֹרָה טוֹבָה / לְבַעֲלֵי הַתְּשׁוּבָה / כִּי נִמְחָל עֲווֹנָם

וְזוֹ תַּקָּנָה / לַעֲדַת מִי מָנָה / יִשְׂמְחוּ מִיגוֹנָם

וְנַפְשׁוֹתָם תָּרוֹן / בְּסֵפֶר זִכָּרוֹן / יִכָּתְבוּן הֲמוֹנָם

וְצוּרִי עֹז מִגְדָּל /הַמִּשְׂגָּב לַדָּל / יִרְצֵנִי בְּגִינָם

בְּיוֹם מֹאזְנֵי צֶדֶק / זְכוּת וְעָווֹן שׁוֹקְלִים:





נוֹגַהּ תִּפְאֶרֶת / בְּמַלְכוּת מַזְהֶרֶת / לְהָאִיר שְׁבָטֵינוּ

וְאַבְרָהָם וְיִצְחָק / וְיַעֲקֹב לֹא יִרְחָק / לְרַצּוֹת עָלֵינוּ

וּמֹשֶׁה וְאַהֲרֹן / וַיוֹיָ רִאשׁוֹן וְאַחֲרוֹן / וְדָוִד מַלְכֵּנוּ

יְעִירוּן נֶעְלָבִים / בְּיוֹם שׁוּב שׁוֹבְבִים / וְיִמְחֶה פְשָׁעֵינוּ

בָּרוּךְ אֵל עֶלְיוֹן / הַסּוֹמֵךְ נוֹפְלִים:
