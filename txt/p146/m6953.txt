


	
	שחר אהלל צורי ואעיר כל נרדם
	



[שַׁחַר אֲהַלֵּל צוּרִי וְאָעִיר כָּל נִרְדָּם] / ר' שלום שבזי

בקשה

  


שַׁחַר אֲהַלֵּל צוּרִי וְאָעִיר כָּל נִרְדָּם,



לְיַחֵד אֶת שְׁמוֹ לֹא יְשַׁנּוּ אֶת תַּפְקִידָם,



וְהוּא סִבַּת כָּל סִבָּה וְעִילַּת יְסוֹדָם,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת אָדָם.


  


לְחַסְדּוֹ מְיַחֲלִים הַגּוּפוֹת וְהַנְּפָשׁוֹת,



וּבָחַר בְּיִשְׂרָאֵל אֲשֶׁר נַפְשׁוֹתָם מְקוּדָּשׁוֹת,



וּמֵאוֹר שְׁכִינָתוֹ הֵמָּה מְלוּבָּשׁוֹת,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת שֵׁת.


  


מְרוֹמָם עַל כָּל בְּרָכָה וּתְהִלָּה נֶאְדָּרִי בַּכּוֹחַ,



בָּרָא הַכֹּל בְּשִׁשָּׁה יָמִים וְיוֹם הַשְּׁבִיעִי יוֹם מָנוֹחַ,



בּוֹ תַּכְלִית הַכֹּל וְתָצִיץ הַנְּפָשׁוֹת בְּסוֹד שַׁדַּי לִזְרוֹחַ,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת נֹחַ.


  


בְּרֵאשִׁית בָּרָא מַלְאָכִים וְגַלְגַּלִּים וְנִגְהָם,



וְכִיוַּן אֶת סֶדֶר הַכּוֹכָבִים וְגָבְהָם,



וְגָלַל אוֹר מִפְּנֵי חֹשֶׁךְ לְהַגִּיהָם,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת אַבְרָהָם.


 


יוֹדוּהוּ גְּדוּדָיו וְאִיֵּי הַיָּם מִמֶּרְחָק,



וְעַמּוֹ אֲשֶׁר בָּחַר, לוֹמְדֵי תּוֹרָתוֹ אֲשֶׁר חָק,



יָחוֹן בְּרַחֲמָיו עַל עַם עָנִי הַנִּדְחָק,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת יִצְחָק.


  


רוֹצֶה עֲבוֹדָה יִשְׁמוֹר עַם נַחֲלָתוֹ בְּלֶכְתָּם,



יִשְׁמַע וְיַקְשִׁיב בְּטֵה רְצוֹן תְּפִלָּתָם,



וּבְחַסְדּוֹ וְטוּבוֹ יְעוֹרֵר אֶת אַהְבָתָם,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת יַעֲקֹב אִישׁ תָּם.


  


בְּשֶׁקֶט וְשַׁאֲנַן תִּשְׂמַח יְחִידָתִי וְתָרוֹן,



וְתַחְמוֹד לֵישַׁב בְּצִלּוֹ חֲבַצֶּלֶת הַשָּׁרוֹן,



אָז יִכְתְּבֵינוּ בְּסֵפֶר חַיִּים וְזִכָּרוֹן,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת אַהֲרֹן.


  


יְקַבֵּל תְּפִלָּתֵינוּ כְּקָרְבַּן אִשֶּׁה,



יִזְכּוֹר זְכוּת אָבוֹתֵינוּ וּבְרִיתוֹ יִנְשֶׁה,



וְיַצִּיל אוֹתָנוּ מִיַּד אֲדוֹנִים קָשֶׁה,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת מֹשֶׁה.


 


וִדּוּיֵינוּ יַאֲזִין וְיַצִּילֵנוּ מִגּוֹי אַכְזָר,



וְנִשְׂמַח וּבְשִׂמְחָתֵינוּ אַל יִתְעָרַב זָר,



לָבֵשׁ ה' עוֹז הִתְאַזָּר,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת אֶלְעָזָר.


  


סוֹמֵךְ נוֹפְלִים אֲשֶׁר עַל עַמּוֹ חָס,



וְהִצִּילוֹ עַל יַד מֹשֶׁה רֹאשׁ הַנְּבִיאִים הַמְּיוּחָס,



וְעֵשָׂו וְחֵילוֹ יְהֶא בְּהַר שֵׂעִיר נִפְחָס,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת פִּינְחָס.


 


פְּדוּת יָחִישׁ צוּרֵינוּ לְעָם בְּדַרְכּוֹ צָנוּעַ,



וְיַשְׁפִּיעַ בְּרָכָה עַל כָּל זֶרַע זֵירוּעַ,



וְעָנִי וְאֶבְיוֹן – יְהֶא שְׁטָר חוֹבוֹ קָרוּעַ,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת יְהוֹשֻׁעַ.


  


אָז יָגִיל יַעֲקֹב, יִשְׂמַח יִשְׂרָאֵל,



וְנִשְׁמַע בְּשׂוֹרָה כִּי בָּא לְצִיּוֹן גּוֹאֵל,



תָּגִיל יְחִידָתִי יוֹם בִּנְיַן אֲרִיאֵל,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת עָתְנִיאֵל.


  


לִתְפִלָּתֵינוּ יַאֲזִין וֶעֱנוּת עָנִי אַל יִבְזֶה,



וְעָלֵינוּ מֵי טוֹהַר מֵחֲסָדָיו יַזֶּה,



אָז נָגִיל בִּמְנוּחָתֵינוּ כְּהַיּוֹם הַזֶּה,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת שְׁמוּאֵל הַחוֹזֶה.


  


שַׁדַּי אֲשֶׁר שׁוּלָיו אֶת הַהֵיכָל מְלֵאִים,



וְכָל זֶרַע קְדוֹשָׁיו – בְּצִלּוֹ הֵם נִהְנָאִים,



נוֹרָא וְאָיוֹם וְרָם עַל כָּל גֵּאִים,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת נְבִיאִים.


  


בְּטוּבוֹ יְשַׂגְּבֵינוּ אֱלֹהֵי עוֹלָמִים,



וִינַקֵּינוּ מִכָּל עָוֹן וְחֵטְא וַאֲשָׁמִים,



וְיַצִּילֵנוּ מֵאִישׁ מִרְמָה וְדָמִים,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת חֲכָמִים.


  


זְמָן יְשׁוּעָתֵינוּ יְקָרֵב וִיבָרְכֵנוּ בַּבְּרָכוֹת,



אֲשֶׁר אֶל תּוֹךְ יָם הָאַחֲרוֹן מְשׁוּכוֹת,



וִיכוֹנֵן בֵּית קָדְשֵׁינוּ וְהַלְּשָׁכוֹת,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת שׁוֹנֵי הֲלָכוֹת.


  


יְזַמֵּן פַּרְנָסָתֵינוּ וְיָסִיר כָּל מְגִנָּה,



וְיִשְׁפּוֹת שָׁלוֹם לָנוּ מִכָּל צַד וּפִנָה,



וְיִשְׁלַח לָנוּ מַלְאַךְ חֶסֶד הַמְּמוּנָּה,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת אַנְשֵׁי מִשְׁנָה.


  


יִשְׁמַע תְּפִלָּתֵינוּ בְּיָחִיד וּבְצִבּוּר,



יִתְבָּרַךְ שְׁמוֹ, בּוֹרֵא נִיב שְׂפָתַיִם לְדִבּוּר,



אֲשֶׁר גִּילָּה לָנוּ אֶת סוֹד הָעִבּוּר,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת סִפְרֵי הַחִבּוּר.


  


סוּכַּת שְׁלוֹמוֹ יִפְרוֹשׂ עַל גּוֹלִים הַנִּלְבָּטִים,



מְקַיְּימֵי מִצְווֹתָיו בִּכְלָלִים וּפְרָטִים,



אֲשֶׁר נָתַן לָהֶם תּוֹרָה, חֻקִּים וּמִשְׁפָּטִים,



בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת שְׁנֵים עָשָׂר שְׁבָטִים.
