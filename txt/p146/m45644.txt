


	
	אֲקַו לָאֵל
	



חתימה: אבו יהודה שבזי






אֲקַו לָאֵל / הַגּוֹאֵל / רָם וְנִשָּׂא בְּהֵיכָלוֹ

יְרַחֵם רָשׁ / אֲשֶׁר נִגְרָשׁ / מִבֵּיתוֹ וְאוֹהָלוֹ

וִיפַרְנֵס / בְּדִבְרֵי נֵס / לְדָל וְאֶבְיוֹן אֲשֶׁר אֵין לוֹ

יְאַמֵּץ עָם / אֲשֶׁר נִזְעָם / וְיָאִירָה לְעֵין שִׂכְלוֹ

וְיַצְלִיחַ / וְיַרְוִיחַ / בְּעוֹלָמוֹ בְּיוֹם חֶדְלוֹ

הוּא עָזִּי / וּמָעֻזִּי / דּוֹדִי לִי וַאֲנִי לוֹ



הַלְלוּהוּ / בִּגְבוּרוֹתָיו / הַלְלוּהוּ כְּרוֹב גֻדְלוֹ:





בְּתוֹךְ תֵּימָן / אֲנִי נִטְמָן / וּבֶן אָמָה הֵצַר לִי

וְלִי חָמַס / וְהִרְבָּה מַס / וְדִכַּנִי בְּיוֹם חֶשְׁלִי

קְרָב חָפֵץ / וּמִתְנַפֵּץ / יְרִיבֵנִי בְּבֵית אֶבְלִי

מְבַקֵּשׁ רִיב / וּלְהַחְרִיב / כְּנָסִיּוֹת לְהָרָע לִי



וְאִם שִׁקֵּר / הֲלוֹא בוֹקֵר / וְיֹדַע יְוֹיָ אֶת אֲשֶׁר לוֹ:





וִדּוּי פִּי / בְּלֵיל נִשְׁפִּי / יִרְצֶה יָהּ וְיַעְנֵנִי

וְיוֹפִיעַ / לְהוֹשִׁיעַ / בְּעֵת רָצוֹן לַהֲמוֹנִי

וְיַצִּיל בַּת / נֶעְלֶבֶת / בְּתוֹךְ צִלָּהּ יַנְחֵנִי

וְיִרְצֶה עַל / אֲסִיר נִגְעַל / וּמִשֵּׁפֶל יְרִימֵנִי

יְצַו יִנּוֹן / לְדָל אוֹנֵן / יְפֵה עַיִן וְאַדְמוֹנִי



אֵל נוֹרָא / דְּרוֹר יִקְרָא / הָעָם בָּחַר לְנַחֲלָה לוֹ:





יְדִיד הִכּוֹן / בְּעֵת תִּשְׁכּוֹן / וְהִתְחַנֵּן בִּתְפִלָּה

עֲלוֹת שַׁחַר / אַל תְּאַחֵר / עֲדֵי הַקֵּץ יְהֶא נִגְלָה

וְחִישׁ מַהֵר / וְהִטַּהֵר / וְלָךְ אַל יִתְּנָה חֶמְלָה

מְאַס קֵדָר / אֲשֶׁר נֶהְדָּר / וְלֵךְ לָךְ מִבְּנֵי עַוְלָה

וְשׁוּב וּחְדָל / וְחוֹן עַל דָּל / וְתִמְצָא חֵן וְרוֹב צַהְלָה



וְאִם צַר לָךְ / נְגַהּ שִׂכְלָךְ / דּוֹם לַיוֹיָ וְהִתְחוֹלֵל לוֹ:





הֱיוּ נְכוֹנִים / מְזֻמָּנִים / וְהַשְׂכִּילוּ לְהִתְבּוֹנֵן

בְּאוֹת נִפְלָא / מִלְמַעְלָה / יְבַשֵּׂרָה בְּבוֹא יִנּוֹן

וִישַׁכְלֵל / לְבֵית מִכְלוֹל / מְקוֹם קֹדֶשׁ אֲשֶׁר כּוֹנֵן

וְאָז יָבוֹא / לְמוֹשָׁבוֹ / לְאֶבְיוֹנִים יְהֶא חוֹנֵן

וְשִׁיר חָדָשׁ / בְּתוֹךְ מִקְדָּשׁ / לְוִיִּים בּוֹ תִּתְרוֹנֵן



לְאֵל הוֹדוּ / וְהַתְמִידוּ / שִׁירוּ לוֹ זַמְּרוּ לוֹ:

וְאִם חָבְרוּ / בְּיוֹם גָּבְרוּ / מְשַׂנְאִים בַּהֲמוֹנֵיהֶם

כִּמְעַט קָט / וְהֵן נִשְׁקָט / הֲדַר גָּבְהָם וְשָׂרֵיהֶם

וְאֵשׁ תִּבְעַר / חֲזִיר יַעַר / וְהַהַגְרִים וְסִגְנֵיהֶם

וְאֵשׁ יִצְחָק / מִמֶּרְחָק / תְּכַלֶּה אַרְמְנוֹתֵיהֶם

הֲפוֹךְ חַרְבָּם / לְתוֹךְ לִבָּם / אֲדוֹן הַכֹּל וְכַלֵּיהֶם

וְהַצִּיל דָּל / אֲשֶׁר נִבְדָּל / וַיַּעֲבֵר אֶת אֲשֶׁר לוֹ:





דְּרוֹר צִפּוֹר / חֲשִׁי לִסְפּוֹר / מְתֵי מִסְפַּר שְׁבָטֵינוּ

וְכוֹפֶר תְּנִי / וְהִתְבּוֹנְנִי / וְחַלִּי פְּנֵי אֱלֹהֵינוּ

מְצוֹא כוֹפֶר / צְבִי עוֹפֶר / וְהָגֵן עַל יְדִידֵינוּ

וְתֵן מַרְפֵּא / לְגוֹלִים פֹּה / עֲשֵׂה בִּזְכוּת אֲבוֹתֵינוּ

פְּדֵה מִסְכֵּן / מִסְתַּכֵּן / יְהֶא שָׁלוֹם בְּעִירֵנוּ



אֲדוֹן עוֹלָם / הַנֶּעְלָם / וְתוֹעֲפוֹת הָרִים לוֹ:





הֲדַר קָמִים / מְזוֹהָמִים / יְהֶא הוֹלֵךְ בְּאֵשׁ עֶבְרָה

יְזוֹרָבוּ / וְיֶחְרָבוּ / יְלוּד הָגָר וְשֵׂעִירָה

לְפִי חֶרֶב / תִּתְקָרֵב / בְּנֵי מֵישָׁא וְגַם בָּצְרָה

וְתִנָּצֵל / כְּלִילַת צֵל / חֲצַר קָדְשִׁי מְקוֹם זִמְרָה

עֲלִי נַפְשִׁי / לְהַר קָדְשִׁי / וְלָךְ צוּרִי דְּרוֹר יִקְרָא



וְאִישׁ נֶאְהָב / בְּחֵן נִרְהָב / וְשָׁב וְרָפָא לוֹ:





שְׁאוֹן גַּלֵּי / זְמָן מִלֵּא / לְבָבֵינוּ דְּבַר שֶׁקֶר

וְאוֹיְבֵינוּ / סְבִיבֵינוּ / בְּאַפָּם זוֹלְלוּ יוֹקֶר

וְהִקִּיפוּ / וְהוֹסִיפוּ / דְּבָרֵיהֶם בְּלִי חֵקֶר

וּמֶלֶךְ זָר / נִתְאַכְזָר / לְקָדְשֵׁנוּ בְּאַף הִפְקַר

וְזֶה נֶחְלָשׁ / וְזֶה נִבְלָשׁ / וְזֶה הִכָּה וְזֶה נִזְקַר



וְהַנִּמְלָט / לְעִיר מִקְלָט / לֹא יָדַעְנוּ מֶה הָיָה לוֹ:





בְּךָ אָרוּץ / גְּדוּד חָלוּץ / צוּר עָזִּי וְזִמְרָתִי

פְּדֵה נִרְדָּף / כְּקַשׁ נִדָּף / נְפוֹץ חֲפֵצֵי רָעָתִי

וְשִׂים תִּקְוָה / לְבֶן אִוָּה / לְהָבִין סוֹד בְּתוֹרָתִי

וְהַךְ מַכְחִישׁ / בְּרַע מַרְחִישׁ / וּמִתְלַעֵג עֲלֵי דָּתִי

וְזַלְזַלִּים / בְּאַף תַּכְלִים / וְרַחֵם עַל אֲיֻמָּתִי

שְׁלַח צֶמַח / בְּךָ יִשְׂמַח / תַּאֲוַת לִבּוֹ נָתַתָּה לוֹ:





זָן וּמֵזִין / רְצֵה וְהַאֲזִין / עֲנִיֵּי עָם וְהַשְׁכִּינָם

הָגֵן יְתוֹמִים / שְׁהֶן הוֹמִים / וּמַלֵּא יָהּ לְחֶסְרוֹנָם

וּבֶן הָגָר / בְּאַף סָגַר / עֲלֵי כַסְפָּם וּמָמוֹנָם

וּבָא רָבַץ / עֲדֵי קִבַּץ / לְתִירוֹשָׁם וְכָל הוֹנָם

וּמִתְנָאֶה / וּמִתְגָּאֶה / בְּרֵעַי יַעֲבוֹד חִנָּם



וּבְפֶרֶךְ / יְשַׁעְבֵּד רַךְ / וּפָעֳלוֹ לֹא יִתֵּן לוֹ:





יָבוֹא יוֹם / הַפִּדְיוֹם / וְאַקְבִּיל אֶת פְּנֵי דּוֹדִי

וְאֶמְשׁוֹל עַל / חֲזִיר מָעַל / וְאֶתְנַעֵר בְּיוֹם חֶלְדִּי

וְהַפֶּרֶא / יִתְעָרֶה / וְיִתְבָּהֵל מִנֶּגְדִּי

רְצֵה אֵל חַי / דְּבַר שִׂיחִי / בְּיָדָךְ חַזְּקָה יָדִי



שְׁפוֹט אוֹיְבַי / מַכְאִיבַי / אֱמָר נָא: קוּם, עֲמוֹד עַבְדִּי!



בְּעִיר צִיּוֹן / וְאַפִּרְיוֹן / שְׁבָה וּצְלַח בְּרוֹב חַסְדִּי!



כְּמוֹ נַמְתָּ / וְאָמַרְתָּ: / וּבְרִיתִי נֶאֱמָנָה לוֹ:
