


	
	אֶל אֵל בָּרוּך וּמְבוֹרָך אֶצְעַק מֵעוֹנִי וְדַלּוּת
	




אֶל אֵל בָּרוּך וּמְבוֹרָך



אֶצְעַק מֵעוֹנִי וְדַלּוּת.



מִי אָנִי גּוֹלֶה וּמֻשְׁלָך



בִּגְבוּלֵי תֵּימָן בְּשַׁפְלוּת.



רַק לִבִּי נֵזְכָּר וְנִמְלָך



הֵסִיר תַּחְלוּאִים וְעַצְלוּת.



יִלְמַד סוֹד תּוֹרָה וְיַהְלָך,



יַנְשֵׁינִי צַעַר וְאַבְלוּת.





יִלְמַד עִם בַּעְלֵי תְּעוּדָה,



יָסִיר מִמֶּנִּי חֲלוּדָה.



אַזְכִּירָה גֶּזַע יְהוּדָה



יוֹם כּוֹכָב יִזְרָח וְיִדְרָך



וִינַהֵג דִּגְלִי בְּחִמְלוּת.



שׁוּב שׁוֹבֵב לָאֵל בְּחַדְלָך,



אַל תַּנְהִיגֵנִי בְּקַלּוּת.





לָך צוּרִי נַפְשִׁי תְּקַוֶּה,



אַתָּה מִשְׂגַּבִּי וְעָזִי.



לִבִּי לָך בִּשְׁבָח יְחַוֶּוה,



בִּגְבוּל עָרֵי הַפְּרָזִי.



אַל תַכְלִים עָנִי מְאַוֶּוה



לָבִּינָה, סוֹדִי וְרָזִי.



חַזֵּק דַּל נִכְאֶה וְנֶחְלָך,



מֵעוֹל זֵידוֹנִים וְסַבְלוּת.


  


וּרְצֵה עַל יוֹנָה עֲלוּבָה,



קַבֵּל 1מִמֶּנָּהּ תְּשׁוּבָה.



צוֹפָה לָ תִּרְפָּא מְשׁוּבָה,



בָּנֶיהָ הֵם קוֹרְאִים לָך.



תִּשְׁמוֹר נַפְשׁוֹתָם בְּגָלוּת



כִּי גָּבַר אֵיבָם וְנִמְשָׁך,



יַטְעֶה נִינֵיהֶם בְּסִכְלוּת.





שִׁמְ עוֹז יָהּ לַהֲמוֹנִים,



תָּשׁוּרָה עַמָּך בְּרָצוֹן.



תִּרְצֶה אָז תִּמְחוֹל עֲוֹנָם



כִּעְשִׁיתַת הַנּוֹהֲגֵי (עֲלֵי) צֹאן.



תָּסִיר כָּל תִּגְרַת יְגוֹנָם,



רֶוַוח תַּמְצִיאֵם וְעֶלְצוֹן.



אָז נִמְצָא שָׁלוֹם בְּעִירָך,



תָּאִירָה חוֹשֶׁך וְאַפְלוּת.


  


בַּת גַּלִּים תִּשְׂמַח וְתָגִיל



יוֹם תָּקִים דִּגְלָהּ וְתַדְגִּיל.



תַּפְצִיחַ בִּשְׁבָחָך וְתַרְגִּיל,



אָז תֶּחְסֶה תָּמִיד בְּצִלָּך,



תִּנְשֶׁה תַּחְלוּאִים וְגַלּוּת.



יִשְׂמַח עַם קוֹדֶשׁ בְּחִירָך,



יִשְׁכַּח תַּמְרוּרִים וְחַשְׁלוּת.


  


בָּך בָּטַח לִבִּי וְנַפְשִׁי,



שׁוֹמֵיַע שַׁוְעַת עֲנִיִּים.



כִּי זִכְרָך מָעוֹז לְרֹאשִׁי,



וּמְנַהֵג גּוּפִי בְּחַיִּים.



וְיוֹם הַפְרֵד רוּחִי לְחָפְשִׁי



זַכֵּינִי בֵּין הַנְּקִיִּים.



כִּי לָקָה גּוּפִי וְנִמְרָך,



נִתְיַיסֵּר הַרְבֵּה בְּכַשְׁלוּת.


  


אִם עַל זֶה חָטַאת לְשׁוֹנִי,



תִּמְחוֹל חַטָּאתִי אֲדוֹנִי,



כִּרְצוֹנָך אֵינוּ רְצוֹנִי.



תַּשְׁפִּיעַ לִי מֵהוֹד יְקָרָך,



בִּטְעֹת חַיָּיתִי בְּשָׁלוּת.



כִּי גּוּפִי חָבַר וְנֶחְרָך



בִּיסוֹדוֹת אַרְבַּע בְּדַגְלוּת.


  


זָכְרָה לִּי חֶסֶד נְעוּרִים,



בַּעֲלוֹת מֹשֶׁה וַאֲבוֹתָי.



בָּך חוֹסִים יַחַד קְשׁוּרִים,



הֵכַנְתָּ כָּל אֹרְחוֹתַי.



עָלוּ אֶל עָרִים בְּצוּרִים,



וַתִּלְחוֹם אֶת מִלְחֲמוֹתַי.



בֶּן עַמְרָם נִיהַג לְעֶדְרָך,



וַיֵּדַע סוֹד הָאֲצִילוּת.


  


הִשִּׂיג מַלְאָכִים וְגַלְגַּל,



עִם צוּרַת רַגְלֵי סְגַלְגַּל,



וִיסוֹדוֹת אַרְבַּע בְּמַעְגַּל.



אָז קִבַּל תּוֹרַת יְמִינָך,



בִּימִינָהּ עוֹשֶׁר וְגַדְלוּת.



הוּא שָׁמַע קוֹל מַאֲמָרָךָ,



מִכְתָּב מִתְבָּאֵר וְחָרוּת.





יָעִיר דּוֹד חַסְדּוֹ לְרַעְיָה,



וִיקַבֵּץ בָּנִים נְדוּדִים,



כִּי מֶנְהוּ חַיִּים וּמֵחְיָה



לִכְלָל זֶרַע הַיְּהוּדִים.



מַשְׁגִּיחַ כָּל בִּרְיָה וּבִרְיָה,



צוֹפִים לִרְצוֹנוֹ גְּדוּדִים.



תַּזְכִּיר שִׁיר וּשְׁבַח לְצוּרָך,



הַמַּסְפִּיק לַכֹּל בְּנַהְלוּת.


  


בָּרוּך הוּא עוֹשֵׂה בְּרֵאשִׁית,



יִפְתַּח אֶת לִבִּי וְיָשִׁית



לִימִינִי נֶפֶשׁ אֱנוּשִׁית.



עוּר לִבִּי, הָבֵן בְּיָשְׁרָך,



אַל תָּרִימֵינִי כְּזֻלּוּת.



שָׁלוֹם לָך חַזְמַק וְדוֹרָך,



בַּעְלֵי הַאְמָנָה וְיַשְׁרוּת.







נוסח מתחת לכתוב: וּרְצֵה.  ↩
