


	
	אֵל מְרוֹמָם
	



ו. מילה

החתימה: א״ב שלם בירב יהוסף […] בן משתא






אֵל מְרוֹמָם עַל תְּהִלָּה / אָבִינוּ מַלְכֵּנוּ

צִוָּה לְאַבְרָהָם תְּחִלָּה / רֵאשִׁית אֲבוֹתֵינוּ

לָמוּל בְּשַׂר הָעָרְלָה / לְטַהֵר אֶת נַפְשֵׁנוּ

לְהָסִיר זֻהְמָא מְגֻלָּה / לְיַשֵּׁר אָרְחוֹתֵינוּ:





בַּזָּכָר זוֹ מִצְוָה / כִּי הִיא כְּנֶגֶד זָכוֹר

לְהָסִיר זֻהְמָא נִדְוָה / מִצַּד עֵמֶק עָכוֹר

זָרַק נָחָשׁ בְּחַוָּה / וְהֻצְרְכָה אֶל עֵין סוֹכֵר

כִּי בְחֶשְׁקוֹ נִתְאַוָּה / לְעוֹרֵר חֵטְא עָלֵינוּ:





גְּבוּלוֹת עַמִּים הִשִּׂיג / חוּץ מִגְּבוּל צַמֶּרֶת

וְחַיֵּי אָדָם הִפְסִיג / אֲבָל נַפְשׁוֹ נִשְׁמֶרֶת

צְרוּפָה כְּכֶסֶף מִסִּיג / לְגַן עֵדֶן חוֹזֶרֶת

וְאִישׁ שָׂדֶה הַמַּלְעִיג / לֹא יָבוֹא בְּעִירֵנוּ:





דִּבֶּר בְּזֶה אֵל עֶלְיוֹן / לְאַבְרָהָם חֹק דָּתוֹ

הוֹפִיעַ מִצִּיּוֹן / וְהִלְבִּישׁוֹ נְבוּאָתוֹ

וְאָמַר לוֹ בְּסוֹד רַעְיוֹן / לָמוּל בְּשַׂר עָרְלָתוֹ

וּבָנָיו עָשִׁיר וְאֶבְיוֹן / מַחְזִיקִים בְּדָתֵנוּ:





הֲלֹא הִיא אוֹת בְּרִית קֹדֶשׁ / בְּכִסֵּא מַעְלָה רְשׁוּמָה

וְכָל הָעוֹבֵר נֶעְנָשׁ / בְּכָרֵת נַפְשׁוֹ עֲגוּמָה

וְאִם בַּשְּׁמִינִי חָלָשׁ / יַבְרִיא בְּעָרְלָה אֲטוּמָה

וְאַחַר יָמִים תּוּמַשׁ / כְּדָת בְּנֵי עַמֵּנוּ:





וּשְׁלוֹשׁ עֶשְׂרֵה בְּרִיתוֹת / עַל הַמִּילָה נֶאְמְרוּ

בִּשְׁלוֹשׁ עֶשְׂרֵה מִדּוֹת / שֶׁלְּרַחֲמִים נִתְבָּרְרוּ

לְזֶרַע יַעֲקֹב לְמוּדוֹת / שִׁבְטֵי יָהּ נִבְחֲרוּ

הָאוֹמְרִים טוֹב לְהוֹדוֹת / לַיוֹיָ אֱלֹהֵינוּ:





תּוֹרַת מֹשֶׁה / זֶרַע קְדוֹשִׁים כֻּלְּכֶם

אֲשֶׁר מֵאֲדוֹנִים קָשֶׁה / הֻשְׁלָח וְהוֹצִיא דִּגְלְכֶם

וְלַעֲלוֹת לְסִינַי הֻרְשָׁה / וְתוֹרָה וְדָת הִנְחִילְכֶם

וְסִלַּק שְׂמֹאל הַנּוֹשֶׁה / בְּיוֹם בָּא בַּהֲמוֹנֵינוּ:





חֶסֶד לְאַבְרָהָם חָן / וְשִׁבְטֵי יָהּ הַקְּדוֹשִׁים

וּפַחַד יִצְחָק נִצְחָן / יִקְרָא אֲלֵיכֶם אִישִׁים

וְיַעֲקֹב אִישׁ תָּם נִבְחָן / הוּא רֹאשׁ הַשָּׁלִישִׁים

וְצַדִּיקִים טוֹב רֵיחָן / כִּי בוֹ יִשְׂמַח לִבֵּנוּ:





טַלְלֵי רָצוֹן וּבְרָכָה / שׁוֹרָה בָּעֶלְיוֹנִים

וּלְבַת כֹּל בְּהַמְשָׁכָה / יוֹרְדִים פָּנִים בְּפָנִים

וּמַלְאַך וְגַלְגַּל כָּכָה / מְקַבְּלִים וְהַתַּחְתּוֹנִים

וּקְהַל יַעֲקֹב זָכָה / בְּשֶׁפַע טוֹב יְחַיֵּנוּ:





יְחַיֵּנוּ מִיּוֹמָיִם / וִיקִימֵנוּ בֵַּשְּׁלִישִׁי

וּבְפֶתַח עֵינָיִם / עַתָּה יָרוּם רֹאשִׁי

זְכוּת מִילָה בִּנְתַּיִם / תְּזַכֶּה שִׂכְלִי וְנַפְשִׁי

וּבִזְכוּת לוּחוֹת שְׁנַיִם / נִמְצָא מְחוֹז חֶפְצֵנוּ:





כָּךְ דָּת מִילָה נוֹהֵג / בַּיּוֹם הַשְּׁמִינִי תָּמִיד

וְדוֹחֶה הוּא אֶת הַשַּׁבָּת / מַלְכוּת וִיסוֹד תַּצְמִיד

וְסוֹף שָׁעָה אַחֲרוֹנָה / אִם נוּלָּד שְׁבוֹת הַעְמִיד

וּמוּל אוֹתוֹ בְּיוֹם רִאשׁוֹן / כְּפִי דָּת רִבּוֹתֵינוּ:





לְכָךְ הַנּוּלָּד בְּעֶרֶב / שַׁבָּת בֵּין הַשְּׁמָשׁוֹת

יוֹם הָעֲשִׂירִי תֶּעֱרַב / מִילָתוֹ וְאַל תָּשֵׁת

סָפֵק בְּמִנְיָן שְׁמוֹנָה / יָמִים הַמְפוֹרָשׁוֹת

וְאִם רִאשׁוֹן עֶרֶב יוֹם טוֹב / בְּאַחַת עֶשְׂרֵה יִמּוֹל יְדִידֵנוּ:





מִלְּבַד רֹאשׁ הַשָּׁנָה / אִם יוֹם רִאשׁוֹן עַרְבּוֹ

תְּהֶא מִילַת זֶה הֲגוּנָה / בִּשְׁנֵים עָשָׂר לָבוֹא

בִּבְרִית קֹדֶשׁ אֱמוּנָה / לְאִישׁ אֲשֶׁר יָשִׁית לִבּוֹ

וְחַפֵּשׂ בְּהִלְכוֹת מִילָה / וְשָׁם תִּמְצָא דְּבָרֵנוּ:





נוֹרָא וְאָיוֹם שׁוּתָף / לִתֵּן לָנוּ שָׂכָר

וְזֶרַע יַעֲקֹב נֻסָּף / קְדֻשָּׁה עַל בֶּן נֵכָר

בַּעֲנַן שְׁכִינָה יֻקָּף / בְּהִמּוֹל לוֹ כָּל זָכָר

וְנִרְגָּן וְחֵילוֹ נִדְחָף / וַיוֹיָ בְּרָאשֵֵׁנוּ:





סְגֻלַּת כָּל הָעַמִּים / מַה נָּאווּ מַעֲשֵׂיכֶם

בִּזְכוּת שַׁבָּת וּמִילָה / רָע לֹא יִגַּע אֲלֵיכֶם

נְקִיִּים מִכָּל אֲשָׁמִים / בְּנֵי אֵל חַי אַשְׁרֵיכֶם

בְּסוֹד מִצְוַת אִישׁ תָּמִים / יָקוּם בְּעֶזְרָתֵנוּ:





עַל זְקֵנֵינוּ יַחְמוֹל / וּבַחוּרִים וְטַף וְנָשִׁים

וְחֶסֶד לָנוּ יִגְמוֹל / נוֹרָא בְצִבְאוֹת קְדוֹשִׁים

יְבָרֵךְ זֶה הַנִּמּוֹל / וְעָם זֶה הַנִּגָּשִׁים

מְזֻמָּנִים מֵאֶתְמוֹל / לַעֲשׂוֹת רְצוֹן יוֹצְרֵנוּ:





פְּדוּת וּרְפוּאָה יִשְׁלַח / אֱלֹהֵי קֶדֶם מְעוֹנָה

וְהַנִּמּוֹל זֶה יִצְלַח / בְּסוֹד דַּעַת עֶלְיוֹנָה

וּבְחֵן זְכוּת בְּרִית מֶלַח / נַעְלֶה אֶל צִיּוֹנָה

וְאֵלִיָּהוּ יִשְׁלַח / בַּגְּאֻלָּה יְבַשְּׂרֵנוּ:





צוֹפֶה עֲתִידוֹת יְרַחֵם / עַל כָּל בְּנֵי הַגּוֹלָה

אֶת בַּת צִיּוֹן יְנַחֵם / יְרַפֵּא נֶפֶשׁ חוֹלָה

וְיַשְׁמִיד אוֹיֵב לוֹחֵם / וְיִרְצֶה שַׁוְעִי כְּעוֹלָה

פְּזוּרֵי עַמּוֹ יְנַחֵם / וְיָחִישׁ לְהוֹשִׁיעֵנוּ:





קוֹרֵא כָּל הַדּוֹרוֹת / מֵרֹאשׁ סִדְרֵי בְּרֵאשִׁית

בַּעֲשָׂרָה מַאֲמָרוֹת / כָּל הַמְּצוּאִים הֵשִׁית

וּנְפָשׁוֹת הַטְּהוֹרוֹת / מְדַבֶּרֶת וֶאֱנוּשִׁית

בְּכִסֵּא כְבוֹדוֹ מְאִירוֹת / גִּנַּת אֱגוֹז חֶלְקֵנוּ:





רוֹעֵד לִבִּי וְכוֹסֵף / לְהַחְזִיק בְּעוֹז הַמִּגְדָּל

יַשְׁפִּיעַ מֵאֵין סוֹף / צוּרִי עַל עֶבֶד דָּל

שָׁלֵם בֵּירַב יְהוֹסֵף / הַמְחַבֵּר הַנֶּחְדָּל

בֶּן מַשְׁתָּא הַמְאַסֵּף / טַעְמֵי תּוֹרָתֵנוּ:





שָׁלוֹם לָכֶם בְּנֵי בְּרִית / כְּמוֹ נָהָר שׁוֹפֵעַ

בִּלְשׁוֹן קוֹדֶשׁ וְעִבְרִית / וּמִכָּל חָבֵר וְרֵעַ

תִּהְיוּ לְשֵׁם וְלִשְׁאֵרִית / בְּרָכָה בְּסוֹד הַצְנֵעַ

מֵרֵאשִׁית וְעַד אַחֲרִית / אֱלֹהִים יְחָנֵּנוּ וִיבָרְכֵנוּ:





תּוֹמֵךְ כָּל דַּל וְאֶבְיוֹן / יְקַבֵּץ פְּזוּרֵי עַמּוֹ

מִי יִתֵּן מִצִּיּוֹן / תְּשׁוּעָתִי לְמַעַן שְׁמוֹ

וְנִרְאֶה גֵּי חִזָּיוֹן / לַחֲזוֹת אֶת מַנְעַמּוֹ

וְיִנּוֹן עַל אַפִּרְיוֹן / יְדִידִי בָּרוּךְ טַעְמוֹ

וְסַנְדְּרֵי בַּעְלֵי הֶגְיוֹן / לְהָאִיר עֵין שִׂכְלֵנוּ:
