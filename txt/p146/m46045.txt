


	
	עוּרִי גְּבֶרֶת
	



חתימה: לאלשבזי

על שפשטו בני ישמעאל בגדי ספר תורה והשליכוהו ערום אצל מלכם וגזר שמד עלינו או נלך ארצנו קודם הקץ.






עוּרִי גְּבֶרֶת שְׁחַרְחוֹרָה

חַלִּי פְּנֵי אָיוֹם וְנוֹרָא

כִּי פֶרֶא גָּזַר גְּזֵרָה

וּפָשַׁט יָד בְּסֵפֶר תּוֹרָה

תְּנוּ עוֹז לֵאלֹהִים וּתְנוּ כָבוֹד לַתּוֹרָה:





לַתּוֹרָה בְּכוּ מַלְאָכִים

בְּסוֹף כָּל פָּרָשָׁה סְמוּכִים

זִלְזַל בָּהּ אִישׁ תַּהֲפוּכִים

וְשָׁחוּ שְׂרִידֵינוּ נְבוּכִים

וַהֲמוֹן שְׁבָטֵחנוּ בּוֹכִים

אוֹי לָנוּ מִבַּעֲלֵי גְּמֵירָה:





גְּמֵירָה בֶּן אָמָה גָּמַר

לְהַחְלִיף דָּת קֹדֶשׁ אָמַר

וּבְהַמְּרוֹתָיו חָשׁ וְהָמַר

וּבְרִית אַחִים לֹא שָׁמַר

יוֹם לְיוֹם חֲמַס דָּתוֹ קָמַר

חֲמָתוֹ עָלַי יְעוֹרֵרָה:





יְעוֹרֵרָה יָהּ לְאַהְבָתִי

וְיָחִישׁ בְּרָצוֹן גְּאֻלָּתִי

יָסִיר לְבִכְיִי וְאַנְחָתִי

וְאָשׁוּב לְאֶרֶץ נַחֲלָתִי

יַאֲזִינָה קוֹל תְּפִלָּתִי

וְיוֹשִׁיעֵנוּ מְהֵרָה:





מְהֵרָה חוּשִׁי חֲבַצֶּלֶת

וּדְלִי לְנֶפֶשׁ צוֹלֶלֶת

שַׁאֲלִי לְדוֹדֵךְ מִשְׁאֶלֶת

מְרוֹמָם וְלוֹ הַמֶּמְשֶׁלֶת

יִזְכּוֹר צְבִי חֵן אַיֶּלֶת

יְהֶא לָהּ צִנָּה וְסוֹחֵרָה:





וְסוֹחֵרָה אֲמִתּוֹ לְהַרְאוֹת

לְעַמּוֹ בְּקוֹלוֹ נִפְלָאוֹת

וְהָיוּ דְּבָרָיו לְעֵד וּלְאוֹת

לְהַפִּיל מַמְלָכוֹת גּוֹאוֹת

עֲלֵיהֶן רוֹב צָרוֹת בָּאוֹת

וְעַל אַפָּם יְשַׁלַּח זְמוֹרָה:





זְמוֹרָה הִטְנִיפוּ וְהָמוּ

הַגְרִים אֲשֶׁר בְּרֶשַׁע קָמוּ

יַחַד בְּשֶׁקֶר נֶאְשָׁמוּ

וְכִמְעַט מִזְעָר יִכְלָמוּ

יִשְׁכְּבוּ בְּבָשְׁתָּם וְיִרְדָּמוּ

יְמוּתוּן בְּאַשְׁמָה וַעֲבֵרָה:





וַעֲבֵרָה בְּאַף תְּסוֹבְבֵהֶם

יַחְדָּו לֹא יֶחֱצוּ יְמֵיהֶם

זַעַם וְחֵמָה יִרְדְּפֵהֶם

וְגָפְרִית תְּזוּרָה עַל נְוֵיהֶם

וְשִׁבְטֵי יָהּ בָּא גּוֹאֲלֵהֶם

לַעֲלוֹת אֶל אֶרֶץ טְהוֹרָה:





טְהוֹרָה לַעֲדַת מִי מָנָה

נְתוּנָה לָהֶם בְּמַתָּנָה

מִיּוֹם אֲשֶׁר יַעֲקֹב חָנָה

וְזִכָּה לָנוּ בְּהַזְמָנָה

וְנָדַדְנוּ אָנֶה וְאָנָה

וּמְחַכִּים יוֹם בּוֹא בְּשׂוֹרָה:





בְּשׂוֹרָה תְּרַפֵּא כָל חוֹלִים

וּתְנַחֵם אֶת הָאֲבֵלִים

תָּאִיר שְׂכָלִים הָאֲפֵלִים

וְנָקוּם וְנַעֲלֶה בְּרֹאשׁ גּוֹלִים

בִּגְזֵרַת עִירִין וְגַלְגַּלִּים

תִּהְיֶה שִׂמְחָה וְאוֹרָה:





וְאוֹרָה מֵאוֹר הָרִאשׁוֹן

וְשָׁלוֹם רָב כְּמֵי נְהַר פִּישׁוֹן

אָז נֵשֵׁב עַל מִטּוֹת שֵׁן

לְמַעְלָה מִשִּׁבְעִים לָשׁוֹן

וְנַפְשׁוֹתֵינוּ תִּתְדַּשֵּׁן

לְהָאִיר גּוּפוֹת כַּמְּנוֹרָה:





בַּעֲלוֹת אַרְבָּעָה דְּגָלִים

וּבְיוֹם עֲלוֹת שָׁלוֹשׁ רְגָלִים

וְחָלִיל מַכֶּה וּמַהֲלָלִים

בְּהַר צִיּוֹן רֹאשׁ מִגְדָּלִים

כִּי מַעֲלָתָהּ יְתֵרָה:





יְתֵרָה מִן הַכֹּל כְּלוּלָה

לַעֲדַת קֹדֶשׁ הָאֲצוּלָה

קָו הָאֶמְצָעִי שְׁתוּלָה

לְהַעֲלוֹת בָּהּ מִנְחָה בְּלוּלָה

בְּיוֹם הַבִּכּוּרִים כְּפוּלָה

כְּעִנְיָן חֵלֶק בְּכוֹרָה:





בְּכוֹרָה [לְיַעֲקֹב נִמְכֶּרֶת]

מֵאִישׁ שָׂעִיר כְּאַדֶּרֶת

וְלוֹ הַמַּלְכוּת נִבְחֶרֶת

וְרוֹב [שִׁפְעָה] מִתִּפְאֶרֶת

בְּשֵׁם הָעֶצֶם נִקְשֶׁרֶת

אֶת הַכְּרָמִים נוֹטֵרָה:





נוֹטֵרָה נִטְרִי כַּרְמֵךְ

קוּמִי קַבְּצִי אֶת עַמֵּךְ

כִּי הֵם נִקְרָאִים עַל שְׁמֵךְ

בְּרוּכָה אַתְּ וּבָרוּךְ טַעְמֵךְ

שׁוּבִי אֶל מִגְדַּל מְקוֹמֵךְ

הִפָּרְדִי מִשַּׁדְמוֹת עֲמוֹרָה:





עֲמוֹרָה וּסְדוֹם הַהֲפוּכָה

מִיַּיִן נֶסֶךְ נְסוּכָה

לְכוֹחוֹת חִיצוֹנוֹת מְשׁוּכָה

וַחֲמַת מֶלֶךְ בָּהּ שְׁפוּכָה

וּבַת נָדִיב תָּמִיד בְּרוּכָה

הִיא הַקְּטַנָּה הַצְּפִירָה:





הַצְּפִירָה יַשְּׁרִי דַרְכֵּךְ

חַלִּי הַיּוֹם פְּנֵי מַלְכֵּךְ

יָבוֹא וְיַגִּיהַּ חָשְׁכֵּךְ

וְאָז תָּשׁוּבִי לְשׁוֹבַכֵּךְ

וְנַפְשִׁי בְּצִלֵּךְ תִּזְדַּכֵּךְ

תָּשׁוּב כְּקֶדֶם כְּשֵׁרָה:





כְּשֵׁרָה מִכָּל חֵטְא וּפֶשַׁע

דְּעִי כִּי יִשְׂרָאֵל נוֹשַׁע

וְנִבְדָּל מִבַּעְלֵי רֶשַׁע

וְרוּחַ יָהּ בָּם תִּשְׁתַּעְשַׁע

וּבָהּ נִגָּאֵל וְנִוָּשַׁע

וְשָׁם מַחֲנֵה אֱלֹהִים שׁוֹרָה:





שׁוֹרָה בְּדִגְלוֹתַי אַרְבַּע

בְּאוֹתוֹת אֲשֶׁר דּוֹדִי קָבַע

וְהִיא נִקְרֶאֶת בַּת שֶׁבַע

בְּזַעְפָּה חֵיל פַּרְעֹה טָבַע

טְנוּפֵי דַּעַת הַחֲסֵרָה:





חֲסֵרָה דַּעְתָּם וּטְנוּפִים

בַּעֲלֵי קֶסֶם וּכְשָׁפִים

יַחַד לְהָרַע נֶאְסָפִים

לְעֵמֶק עָכוֹר נִדְחָפִים

עוֹבְדֵי אֱלִילִים וּתְרָפִים

נְבִיאֵי בַּעַל וַאֲשֵׁרָה:





וַאֲשֵׁרָה לֹא תִּתְיַשֵּׁר

בְּדֶרֶךְ סַרְטוּ ובְמִישׁוֹר

כִּי אֲיֻמָּתִי בַּהֶכְשֵׁר

עָלַי חוּט חֶסֶד תִּקְשׁוֹר

וְעָנִי וְאֶבְיוֹן מִתְעַשֵּׁר

וְיִקַּח לוֹ מִטּוּב זְמוֹרָה:





זְמוֹרָה בִּמְקוֹם מַחֲנָיִם

בְּצִיּוֹן וִירוּשָׁלַיִם

יִשְׂמַח יְהוּדָה וְאֶפְרַיִם

וַאֲנִי אֶשְׂמַח בִּנְתָּיִם

הַאֲזִינוּ הַשָּׁמַיִם וַאֲדַבֵּרָה:





וַאֲדַבֵּרָה בְּפִי לְהָשִׁיר

חִידוֹת וּמָשָׁל בְּאוֹת אַקְשִׁיר

וְרוּחַ שְׂפָתַי סוֹד תַּפְשִׁיר

לְשַׂמֵּחַ עָנִי וְעָשִׁיר
