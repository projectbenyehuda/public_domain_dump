


	
	אֲיֻמָּה בְּהַר הַמּוֹר [...] וְצֶמַח אֲרִי גִבּוֹר
	



חתימה: אלשב[ז]י




אֲיֻמָּה בְּהַר הַמּוֹר / עֲלִי לָךְ וּבַשְּׂרִי



וְנָשִׁיר בְּשִׁיר מִזְמוֹר / בְּנִגּוּן לְיוֹצְרִי



וְצֶמַח אֲרִי גִבּוֹר / מְהֻדָּר בְּשַׁעֲרִי



וְזָמִיר וְקוֹל הַתּוֹר / בְּאַרְצִי מְעוֹרְרִי



שְׂרָיָה חֲגוֹר חַרְבָּךְ



וְהַשְׁחֵת גְּדוּד אוֹיְבָךְ



וְחָזְקָה וְאַל תִֹנְבָּךְ



לְהוּגְנֵי אֱדוֹם תִּגְמוֹר / וְתַשְׁקִיף בְּשַׁחֲרִי



כְּרָמָיו בְּאַף תִּזְמוֹר / וְתַשְׁחִית לְעֵץ פְּרִי:



לְקַיְת אַלְפְתַא לאצִּׄרגַׄאם / בֵבַסְטֹהּ לְחִין יְנוּב



עֲלַיְהּ תַנְצִׄם אַלאַנְגַאם / וְתַתְצֵאפֵף אלְחִרוּב



וְהוּ יַקְתֻל אלדִּרְגַׄאם / וְתֻהְדַם לֹהּ אלדְּרוּב



וְנַאר אַכִּלַה תַדְׄעַר / גִׄיוּשֹׁהּ וְתַעְתְרִי



וִתַמְחִי בְנִי אלְעִמְלַאק



בִאלַאסְכַׄאט וַאלְאַחְרַאק



וְיַנְפִדְׄהֻם אלְכַׄלַּאק



וּבַיְת מֻלְכְּהֹם יֻדְבִר / וּפִיהּ נַחְס גַׄאיִרִי



וְיַצְהֻר מְלִךּ טַאהִר / אֲמִיר אלְעֲשַׁאיִרִי:



שְׁכִינָה בְּצֵל כַּנְפֵי / כְּרוּבִים תְּסוֹכְכִי



לְיִנּוֹן וְחֵיל תָּקְפוֹ / וְצוּרֵךְ תְּבָרְכִי



וְיִשְׁפּוֹךְ חֲרוֹן אַפּוֹ / בְּפֶרֶא הֲפַכְפְכִי



וְעוֹבֵר וּמִתְעַבֵּר / וְדָתוֹ מְכוֹעֲרִי



מְהַתֵּל עֲלֵי דָתִי



וּבוֹזֶה לְתוֹרָתִי



וְהֵסִיר עֲטָרָתִי



וְיָמָיו כְּצֵל עוֹבֵר / וְאֵידוֹ מְמַהֲרִי



וְעַמִּי בְּיַד שׁוֹמֵר / מְשֻׁבָּח מְפוֹאֲרִי:



בִּגׄוּדׄה עֲצִׄים דַאיִם / וִעַאלִי בְמֻלְכְּהוּ



וְהוּ יַחרִס אלנַּאיִם / בִתַוְחִיד אִסְמְהוּ



וְאַגְׄעַל סִרַאט קַאיִם / לִמַן כַׄץ דִׄכְרְהוּ



וְנוּר אלְפְלַךּ דַאִיר / בִשַׁאנֹהּ וִסַאיִרִי



לְהוּ אנְוַאר סַטַּאעַה



וְתַכְׄדֻם לְהוּ טַאעַה



וִדַוְר לַם יְקִף סַאעַה



וְמִן גַׄוְהֲרֹהּ נַאשִׁר / עֲלַי אלְבַחר וַאלְבַרִּי



בִלַיְלֻן קְמַר סַאמִר / וְשַׁמס אַוַּל אלְפַגְׄרִי:



יְחִידָה בְּשִׁיר הוֹדִי / לְאֵל חַי וְרַנְּנִי



וְעִמְדִי וְיַחֵדִי / וְלִבֵּךְ תְּכַוְּנִי



וְחֶשְׁקִי וְכָל סוֹדִי / בְּמַלְכִּי וְכוֹנְנִי



וְאֵלָיו אֲנִי נוֹהֵר / בְּעַרְבִּי וְשַׁחֲרִי



מְחַיֶּה לְנִשְׁמָתִי



וְעָזִּי וְזִמְרָתִי



וְאֵלָיו תְּשׁוּקָתִי



וְחַסְדּוֹ אֲנִי זוֹכֵר / וְטוּבוֹ מְעוֹרְרִי



וְלִבִּי כְּאֵשׁ בּוֹעֵר / בְּאַהְבָה מְזוֹרְרִי:



תִשַׁוַּקְת יַא וִדִּי / בִנַצְׄמִי וִלַחְנְתִי



בִדִׄכְר אלְוִדוּד סִידִי / אֲבַלִּג בִנִיְּתִי



וְפִי גַנַּת אלְכֻׄלְדִי / יִסַכִּן לְחַיְוְתִי



כְׄטַא עַבְדְהוּ יַגְפִר / וְיַסְמַח לִאַוְזַארִי



יִסַלִּם עֲלַי נַפְסִי



בִיַוְם תַנְצֻׄר אלְכֻּרְסִי



וְפִי צִׄלְּהוּ תַמסִי



וּבִאלגַׄוְף תְעוּד תַסְכֻּן / לְהוּ אלְחַמְד וַאלשֻּׁכְּרִי



וְגִׄסמִי בְהַא יַבְשִׁר / מְעַא אַוַּל אלסַּחרִי:
