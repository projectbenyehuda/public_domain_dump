


	
	ספר תחכמוני - שַׁעַר עֶשְׂרִים וּשְׁמוֹנָה
	



בְּשִׁבְחֵי צִיּוֹן וִירוּשָׁלַיִם. וּבַקָּשָׁה רוּמָהּ תַּעֲלֶה לַשָּׁמַיִם.






נְאֻם הֵימָן הָאֶזְרָחִי: נָסַעְתִּי מֵאֶרֶץ נוֹף. אֶל
יְפֵה נוֹף. וְהַזְּמַן יְנִידֵנִי כַדּוּר וְיִצְנֹף. וְאָמַרְתִּי
לְרַעְיוֹנַי רַב לָכֶם לִשְׁכַּב בֵּין שְׁפַתָּיִם. וְלָשֶׁבֶת בֵּין
תַּנּוּר וְכִירַיִם. אַעֲלֶה אֶתְכֵם מֵעֳנִי מִצְרַיִם. לִרְאוֹת בְּטוּב
יְרוּשָׁלַיִם. וְשַׂמְתִּי מְגַמָּתִי אֶל הַר יְיָ. וְהָעִיר אֲשֶׁר
בָּחַר יְיָ. אֲשֶׁר מִקֶּדֶם שָׁכְנוּ בָהּ מְתֵי ישֶׁר וּזְכוּת. רוֹאֵי
פְנֵי הַמֶּלֶךְ הַיּוֹשְׁבִים רִאשׁוֹנָה בַמַּלְכוּת. וְכַאֲשֶׁר בָּאתִי
גְבוּלוֹתֶיהָ. וּפִסַּגְתִּי אַרְמְנוֹתֶיהָ. וְנָשַׁקְתִּי חָרְבוֹתֶיהָ.
וְחוֹנַנְתִּי עַפְרוֹתֶיהָ. אָמַרְתִּי: מַה טֹּבוּ אֹהָלֶיךָ. הַר
צִיּוֹן וּמִשְׁכָּנֶיךָ. וּמַה נָּעֲמוּ אַרְצוֹתֶיךָ וּמְעוֹנֶיךָ.
אָמוּתָה הַפַּעַם אַחֲרֵי רְאוֹתִי אֶת פָּנֶיךָ.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

נַפְשִׁי לְצִיוֹן מִסְּפָרַד גָּלְתָה

מִמַּעֲמַקִים עַד שְׁחָקִים עָלְתָה.

גִיל הִרְבְּתָה יוֹם רָאֲתָה הַר אֵל וְהוּא

הַיּוֹם אֲשֶׁר מִיּוֹם הֱיוֹתָהּ אִוְּתָה.

כַּמָּה חֲסִידִים נִכְסְפוּ לִרְאוֹת וְלֹא

זָכוּ וְהִיא חוֹטְאָה וּבַמֶּה זָכְתָה.

צוּר הֶחֱלִיץ גּוּפָהּ בְּצִיּוֹן אַחֲרֵי

אָבְדָה וְנָס לֵחָהּ וְעֵינָהּ כָּהֲתָה.

וּבְיַרְכְּתֵי מַעְרָב כְּמֵתָה נֶחְשְבָה

עַד כִּי בְצִיּוֹן עִיר יְיָ חָיְתָה.

צִיּוֹן אֲשֶׁר כַּמָּה מְדִינוֹת נֶהְדְּרוּ

אַךְ כַּהֲדָרָהּ עֵין אֱנוֹשׁ לֹא רָאֲתָה.

לֹא אֵדְעָה אִם הַמְּרוֹמִים שָׁחֳחוּ

נֶגְדָּהּ וְאִם הִיא עַל מְרוֹמִים עָלְתָה.

הָעִיר אֲשֶׁר בָּהּ הַשְּׁכִינָה שָׁכְנָה

וּלְעֵין בְּנֵי אִישִׁים יְקָרָהּ גִּלְּתָהּ.

עַל הַיְלָדִים צָלְחָה רוּחָהּ וְטֶ-

רֶם יִקְרְאוּ אֶל הַנְּבוּאָה עָנְתָה.

עַיִן אֲשֶׁר לֹא רָאֲתָה זִיו תָּאֳרָהּ

לֹא חָסְרָה חשֶׁךְ וְאוֹר לֹא רָאֲתָה.

יָפְתָה. וְאִם נוֹתְרָה עֲרֻמָּה מִיְּקָר

וּבְלִי חֲלָאִים יַעֲלַת חֵן יָפְתָה.

עֵינִי בְחֹם לִבִּי [קְרָבַי] נָחֲתָה

מֵהַצְּלָעוֹת מֵי דְמָעוֹת דָלְתָה.

אַיֵּה שְׁכִינַת אֵל וְאַיֵּה זָהֳרָהּ

שֶׁמֶשׁ יְקָר אֵיךְ בַּחֲצוֹת יוֹם פָּנְתָה.

גִּבְעַת לְבוֹנָה אֵיךְ בְּתַחְתִּית שָׁפְלָה

אַחַר אֲשֶׁר עַל בָּמֳתֵי רוּם גָּאֲתָה.

גָּלְתָה עֲדָתָהּ מִמְּעוֹנוֹת הַיְקָר

כִּי מַאֲמַר צוּר יוֹצְרָהּ לֹא עָשְׂתָה.

הַעוֹזְבָה אַלּוּף נְעוּרֶיהָ אֲבָל

חָשְׁקָה בְזוּלָתוֹ וְתַחְתָּיו שָׂטְתָה.

רוּחַ קְנָאוֹת עָבְרָה עָלָיו וּמֵי

מָרִים מְאֹרָרִים כְּשׂוֹטָה הָשְׁקְתָה.

וַתִּהְיֶה אָלָה בְתוֹךְ עַמָּהּ עֲדֵי

נָפְלָה יְרֵכָהּ חִישׁ וּבִטְנָהּ צָבְתָה

תּוֹחִיל לְהִטַּהֵר בְּמֵי גָלוּת וְעוֹד

לֹא טָהֲרָה עַד כֹּה וְעוֹד לֹא נִקְּתָה.

זֶה לָהּ שְׁנוֹת אֶלֶף וְעוֹד מֵאָה וְאַר-

בָּעִים וּשְׁמוֹנֶה מִנְּוָתָהּ גָּלְתָה.

הִיא גֹרְשָׁה מִתּוֹךְ נְוֵה צִיּוֹן וְצִּ-

יּוֹן שַׁבְּתוֹתֶיהָ עֲדֵי כֹה רָצְתָה.

תּוֹחִיל לְעֵת רָצוֹן וְעֵינָהּ מֵחֲכוֹת

כֵּהָה וְנַפְשָׁהּ כָּלְתָה עַד כָּלְתָה.

תִּבְכֶּה בְּעֵת תִּרְאֶה עֲדַת זָרִים אֲשֶׁר

יַעְלוּ לְבֵית הָאֵל וְהִיא לֹא עָלְתָה.

לֹא קָצְרָה יַד אֵל לְהוֹשִׁיעָהּ וְעַ-

מּוֹ לֶאֱסֹףְ מוֹשִׁיב יְחִידִים בַּיְתָה.

עוֹד צוּר יְשִׁיבֶנָּה לְאַפִּרְיוֹן וְעוֹד

תָּשׁוּב לְעֶדְנָה אַחֲרֵי כִי בָלְתָה.





וּכְכַלּוֹתִי לַעֲרֹךְ נִיב חִכִּי. וְחִין עֶרְכִּי.
לִפְנֵי מַלְכִּי. קַמְתִּי מִכְּרֹעַ עַל בִּרְכִּי. לְשׁוֹטֵט בָּעִיר.
וְאֵשׁ הָאֲנָחוֹת צַלְעִי תַּבְעִיר. וִיקוֹד הַיְגוֹנוֹת תְּלַהֵט
הַצְּלָעוֹת. וְעֵינַי מְגוֹלָלוֹת בִּדְמָעוֹת. וָאֶשָּׂא עֵינַי וָאֵרָא
מְקוֹם הַמִּקְדָּשׁ וְהָעֲזָרָה. אֲשֶׁר הֵסִירוּ עֲוֹנֵינוּ מִשָּׁם
הַמְּנוֹרָה הַטְּהוֹרָה. וְהִבְעִירוּ תַחְתֶּיהָ אֵשׁ זָרָה. וּבְעוֹדִי
מִתְהַלֵּךְ בְּנֶפֶשׁ שׁוֹאֶפֶת. וְדִמְעָה נוֹטֶפֶת. כְּנַחְלָה
שׁוֹטֶפֶת. רָאִיתִי לְעֻמָּתִי אִישׁ מִבְּנֵי הָעִיר. וַיֹאמֶר לִי:
אֶחְשֹׁב כִּי אַתָּה מִקְצֵה גָּלֻיּוֹת. וּמֵאֲרָצוֹת נָכְרִיּוֹת.
אָמַרְתִּי: כֵּן הַדָּבָר אֲשֶׁר דִּבַּרְתָּ. וְעַתָּה רְצוֹנִי לִשְׁאֹל
דָּבָר מִמְךָ. אָמַר לִי: הִנְנִי עִמְךָ. שְׁאַל וְאַגִּידָה לְךָ.
אָמַרְתִּי לוֹ: מָתַי בָּאוּ הַיְּהוּדִים לָעִיר הַזֹּאת. אָמַר לִי:
מִיּוֹם לְכָדוּהָ יִשְׁמְעֵאלִים. שְׁכָנוּהָ יִשְרְאֵלִים. אָמַרְתִּי
לוֹ: וּמַדּוּעַ לֹא שְׁכָנוּהָ בִּהְיוֹתָהּ בְּיַד הָעֲרֵלִים. אָמַר
לִי: מִפְּנֵי שֶׁאָמְרוּ כִּי הָרַגְנוּ אֱלֹהֵיהֶם. וְעָשִׂינוּ חֶרְפָּה
לָהֶם. וְאִלּוּ בְתוֹכָהּ מְצָאוּנוּ. אֲזַי חַיִּים בְּלָעוּנוּ. הֵן
נִזְבַּח אֶת תּוֹעֲבוֹתֵיהֶם לְעֵינֵיהֶם וְלֹא יִסְקְלוּנוּ. אָמַרְתִּי:
וְאֵיךְ הָיְתָה סִבַּת בִּיאַתְכֶם אֶל הַמָּקוֹם הַזֶּה. אָמַר: כִּי
הָאֵל קִנֵּא לִשְׁמוֹ. וַיַּחְמֹל עַל עַמּוֹ. וַיֹּאמַר: לֹא טוֹב הֱיוֹת
בְּנֵי עֵשָׂו הֵיכַל קָדְשִׁי יוֹרְשִׁים. וּבְנֵי יַעֲקֹב מִשָּׁם
גְּרוּשִׁים. פֶּן יֹאמְרוּ הַגּוֹיִים בְּקִנְאָה. זָנַח הָאֵל בְּנוֹ
בְּכוֹרוֹ בְּשִׂנְאָה. וְהֵרִים בֶּן הַגְּרוּשָׁה לְמַעֲלַת
הַנְּשׂוּאָה. וְלֹא יוּכָל לְבַכֵּר אֶת בֶּן הָאֲהוּבָה עְל פְּנֵי בֶן
הַשְּׂנוּאָה. וַיָּעַר אֱלֹהִים אֶת רוּחַ מֶלֶךְ יִשְׁמְעֵאלִים
בִּשְׁנַת אַרְבַּעַת אֲלָפִים וּתְשַׁע מֵאוֹת וַחֲמִשִּׁים לַיְצִירָה.
וְנָחָה עָלָיו רוּחַ עֵצָה וּגְבוּרָה. וַיַּעַל הוּא וְכָל חֵילוֹ
מִמִּצְרַיִם. וַיָּצַר עַל יְרוּשָׁלַיִם. וַיִּתְּנֶהָ ה' בְּיָדוֹ
וַיְצַו לְהַעֲבִיר קוֹל בְּכָל הָעִיר. אֶל כָּל רַב וְצָעִיר. לֵאמֹר:
דַּבְּרוּ עַל לֵב יְרוּשָׁלַיִם. לָבוֹא אֵלֶיהָ כָּל הָרוֹצֶה מִזֶּרַע
אֶפְרַיִם. אֲשֶׁר יִשָּׁאֵר מֵאַשּׁוּר וּמִמִּצְרַיִם. וְהַנִּדָּחִים
בִּקְצֵה הַשָּׁמַיִם. וַיִּתְקַבְּצוּ מִכָּל פֵּאָה אֵלֶיהָ. וְחָנוּ
בִגְבוּלֶיהָ. וַאֲנַחְנוּ עַתָּה יוֹשְׁבִים בְּצֵל הַשַּׁלְוָה
הַנִּמְתֶּקֶת. אִם לֹא תִהְיֶה נֶעְתֶּקֶת וְנִתֶּקֶת. כִּי אֲנַחְנוּ
יְרֵאִים לְרֹעַ הַמַּעֲשִׂים הַנַּעֲשִׂים בָּהּ. וּמִן הֶחָמָס
וְהָרֶשַׁע אֲשֶׁר בְּקִרְבָּה. וְאֵשׁ הַשִּׂנְאָה וְהַמַּחֲלֹקֶת. אֲשֶׁר
בָּהּ דּוֹלֶקֶת. וְלִבּוֹת שׁוֹכְנֶיהָ חוֹלֶקֶת. כִּי כֻלָּם הֵם
רָאשִׁים. וְאַכְזָרִים קָשִׁים. וְכָל אִישׁ יְבַקֵּשׁ רָעַת חֲבֵרוֹ.
הָאָב שׂוֹנֵא לִבְנוֹ בְכוֹרוֹ. וְהַבֵּן לְהוֹרוֹ. הַבָּנִים מְלַקְּטִים
עֵצִים. וְהָאָבוֹת לְהַבְעִיר בָּם הָאֵשׁ רָצִים. וְכָל הַמַּעֲשִׂים
בִּלְבְבוֹת שְׁנַיִם. וְאֵין בָּהֶם עוֹשִׂים לְשֵׁם שָׁמַיִם. וּבְכָל
יוֹם יַרְבּוּ מְרִיבוֹת וּמַחֲלֹקוֹת. וְכָל הַלְּבָבוֹת נֶחֱלָקוֹת. עַד
קָרָאתִי שֶׁם-הָעִיר סֶלַע הַמַּחְלָקוֹת. וְהָרָאשִׁים הַמְמוּנִים
עֲלֵיהֶם. נוֹשְׁכִים בְּשִׁנֵּיהֶם. הָרֹאשׁ אֶחָד יִפְנֶה לְהַזִּיק
לָאַחֵר וְהָאַחֵר כְּנֶגְדּוֹ. וְכָל אֶחָד לַחֲבֵרוֹ כִּצְנִינִים
בְּצִדּוֹ. וְיֵשׁ עוֹד בֵּינֵיהֶם מַסִּיתִים וּמַדִּיחִים. וּמַפְרִידִים
בֵּין אַחִים. זֶה יִפְרֹשׁ רִשְׁתּוֹ. וַתֵּשֶׁב בְּאֵיתָן קַשְׁתּוֹ.
וְזֶה מַגְדִּיל הַמְּדוּרָה. וּמַבְעִיר הַבְּעֵרָה. וְעַל כֵּן נַעֲרֹךְ
לָאֵל תְּפִלָּתֵנוּ. לְכַלּוֹת הַקּוֹצִים מִבֵּינוֹתֵינוּ. וּתְשׁוּעָתוֹ
עַיִן בְּעַיִן נֶחֱזֶה. וְיָסֵר מֵעָלֵינוּ רַק אֶת הַמָּוֶת
הַזֶּה.





וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֵשׁ הַמְּרִיבָה יוֹם וְלֵיל דּוֹלֶקֶת

בָּנוּ וְלִבּוֹת אוֹהֲבִים חוֹלֶקֶת.

אִם לֹא יְכַבּוּהָ מְתֵי ישֶׁר וְתֹם

בָּנוּ חֲמַת אֵל תִּהְיֶה נִשֶּׂקֶת.

אֶל-כָּל-חֳלִי יִמְצָא אֱנוֹשׁ מַרְפֵּא אֲבָל

קִנְאָה תְּשַׁבֵּר עַד שְׁאוֹל מַפְרֶקֶת.

כָּל-הָעֲוֹנוֹת יִסְבְּלֵם הָאֵל וְאַף

יַאְרִיךְ וְלֹא יוּכַל שְׂאֵת מַחְלֹקֶת.





אָמַר הַמַּגִּיד: וּבְהַשְׁלִימוֹ נְאֻמוֹ. שָׁאַלְתִּי
עַל שְׁמוֹ. וַיִּצְחַק עַל יְגוֹנִי. וַיֹּאמַר: הֲלֹא תַכִּיר חֲבֵרְךָ
חֶבֶר הַקֵּינִי. וַיִּשְׂמַח לִבִּי עַל מְצִיאוּתוֹ. וְנִקְשֵׁרְתִּי
בַּעֲבוֹת חֶבְרָתוֹ. וְהִתְעַלַּסְתִּי בְּקִרְבָתוֹ. כָּל יְמֵי
שִׁבְתִּי בְּאַדְמָתוֹ.
