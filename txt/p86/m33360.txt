


	
	ספר תחכמוני - שַׁעַר אַרְבָּעִים וְאַרְבָּעָה
	



בְּרַב הַשוֹאֵל לְתַלְמִידָיו שְׁאֵלוֹת. לְהַבְחִין תַּלְמִידָיו בַּתְּשׁוּבוֹת הָעוֹלוֹת




נְאֻם הֵימָן הָאֶזְרָחִי: נָסַעְתִּי עם חֲבֵרִים
תַּאֲוָה לָעֵינַיִם. מִלְּבוֹא חֲמָת עַד נַחַל מִצְרַיִם. וּבְעוֹדִי
מְסוֹבֵב בִּשְׁעָרֶיהָ. וּבְאַחַת עָרֶיהָ. רָאִיתִי אַרְמוֹן מְפֻתָּחִים
שְׁעָרָיו. בְּנוּיִים בְּאַבְנֵי שַׁיִשׁ חֲצֵרָיו וַחֲדָרָיו.
וּבְתוֹכוֹ עַם רָב. וַאֲסַפְסֻף נֶעֶרָב. וּבֵינָם אִישׁ זָקֵן הָעֲנָוָה
מְעִילָיו. וְהַצְּנִיעוּת שׁוּלָיו. וְהַחֲסִידוּת כְּמַסְוֶה עַל
פָּנָיו. וְיִרְאַת שַׁדַּי כְּעַמּוּד אֵשׁ בֵּין עֵינָיו. וּלְפָנָיו
חֲמִשִׁים תַּלְמִידִים. מַשְׂכִּילִים וַחֲמוּדִים. וְהֵם יוֹשְׁבִים
לְרַגְלָיו. וּמְנַשְּׁקִים שׁוּלָיו. וַיֹּאמֶר לָהֶם: דְּעוּ כִּי
רְצוֹנִי לִבְחֹן תְּבוּנַתְכֶם. וְלִשְׁקֹל בַּפֶלֶס שֵׂכֶל חָכְמַתְכֶם.
וְעַתָּה יֹאמַר כָּל אִיש מִכֶּם מָשָׁל נֶחְמָד. אֲשֶׁר כָּל הַבָּא
עָלָיו וְעָמָד. וַיֹּאמְרוּ: כֹּל אֲשֶׁר תְּצַוֶּה נַעֲשֶׂה. וּבְכֹל
אֲשֶׁר תִּרְצֶה אֶת עֲבָדֶיךָ נַסֵּה. וַיִּגַּשׁ הָרִאשׁוֹן מִן
הַתַּלְמִידִים וַיֹּאמַר: בְּעֵת הַמִּבְחָן וְהַמֶּחְקָר. יֵקַל אָדָם
אוֹ יֵקַר. אָמַר אַחֵר: הָעֶבֶד הוּא חָפְשִׁי כְּשֶׁיִּסְתַּפֵּק
בְּחֶלְקוֹ. וְהַחָפְשִׁי הוּא עֶבֶד כְּשֶׁיְּבַקֵּשׁ יוֹתֵר מִלֶחֶם
חֻקוֹ. אָמַר אַחֵר: בֶּן אָדָם רוֹדֵף וְנִרְדָּף. רוֹדֵף מַה שֶׁלֹא
יַשִּׂיגֶנּוּ. וְנִרְדָּף מִמִּי שֶׁלֹּא יִמָּלֵט מִמֶּנּוּ. אָמַר
אַחֵר: אֶתְמַהּ עַל מִי שֶׁיִּקְנֶה עֲבָדִים בְּהוֹנוֹ. אֵיךְ לֹא
יִקְנֶה הַחָפְשִׁים בְּמֶתֶק לְשׁוֹנוֹ. אָמַר אַחֵר: יְמֵי הָאָדָם
שְׁלֹשָׁה. אֶתְמוֹל עָבַר זְמַנּוֹ. וְהַיּוֹם הוּא חוֹלֵף וְאֵינוֹ.
וּמָחָר סָתוּם וְאֵין מִי יְבִינוֹ. אָמַר אַחֵר: לִמְדוּ הַחָכְמָה לֹא
לְהִתְגַּדֵּל עַל המַשְׂכִּילִים. וְלֹא לְרַמּוֹת בָּהּ הַכְּסִילִים.
אָמַר אַחֵר: שְׁתִיַּת אָדָם וּמַאֲכָלו. יְעַוְּרוּ עֵינֵי שִׂכְלוֹ.
אָמַר אַחֵר: אִם תְּבַקֵּש מִתֵּבֵל דֵּי סִפְקְךָ. מִעוּטָהּ יַסְפִּיק
לְךָ. וְאִם תְּבַקֵשׁ יוֹתֵר מִסִּפְקְךָ. כֻּלָּהּ לֹא תַסְפִּיק לְךָ.
אָמַר אַחֵר: כָּל הַבָּא בַּעֲבוֹדַת מֶלֶךְ בְּלֹא אֱמוּנָה. הֵבִיא
נַפְשׁוֹ בְּסַכָּנָה. אָמַר אַחֵר: אַל תְּהִי מִן הָאֲנָשִׁים אֲשֶׁר
בְּעֵת צָרָתָם. יַחְדְּלוּ מֵרָעָתָם. וּבְעֵת חֲלוֹתָם. יִנָּחֲמוּ עַל
חֲטָאתָם. וּבְעֵת עֲנִיוּתָם. יַכְנִיעוּ גַאֲוָתָם. אָמַר אַחֵר:
מְלַמֵּד בָּנָיו חָכְמָה בְּנַעֲרוּתָם. יִמְצָא טוֹבָתָם בְּבַחֲרוּתָם.
כְּעֵץ הַשָּׂדֶה אִם בְּרֵאשִׁית הַשָּׁנָה תַּרְבֶּה עֲבוֹדָתוֹ.
בְּאַחֲרִיתָהּ יַרְבֶּה לְךָ תְבוּאָתוֹ. אָמַר אַחֵר: כָּל הָרוֹצֶה
לַעֲלוֹת בְּמַעֲלוֹת גְדוֹלוֹת. יִשָׁמֵר מִן הַנְּבָלוֹת. אָמַר אַחֵר:
מִי שֶׁבָּא אֶל הָעוֹלָם הַזֶּה יִשְׂבַּע מְרוֹרוֹת. וּמִי שֶׁיָּצָא
מִמֶּנּוּ נִמְלַט מִן הַצָּרוֹת. אָמַר אַחֵר: בֶּן אָדָם כִּבְהֵמָה
אֲבוּסָה. מֵעֲוֹנוֹת עֲמוּסָה. וְרוֹבֶצֶת תַּחַת הַמַּשָׂא, אָמַר
אַחֵר: כְּשֶׁיִּקְצֹף הָאֵל עַל הַקְּהִלָּה יִתֵּן הָעשֶׁר לְסִכְלֵיהֶם.
וּלְקוֹפְצֵי יְדֵיהֶם. וּכְשֶׁיֶאֱהַב אוֹתָם יִתֵּן הָעֹשֶר
לִנְדִיבֵיהֶם. וְלַטּוֹבִים אֲשֶׁר בָּהֶם. אָמַר אַחֵר: אִם תִּכָּנַע
לְמִי שֶׁתִּרְצֶה תִּהְיֶה אֲסִירוֹ. וְאִם לא תִצְטָרֵך לְמִי
שֶׁתִּרְצֶה תִּהְיֶה חֲבֵרוֹ. וְאִם תַּעֲשֶׂה חֶסֶד לְמִי שֶׁתִּרְצֶה
תִּהְיֶה גְבִירוֹ. אָמַר אַחֵר: הַחָכְמָה תִקְרָא מַהֲרוּ וְלִמְּדוּנִי
. טֶרֶם תְּבַקְשׁוּנִי וְלֹא תִמְצָאוּנִי. אָמַר אַחֵר: הַהַלְוָאָה
רֵאשִׁיתָהּ אַהֲבָה. וְסוֹפָה מְרִיבָה וְאֵיבָה. אָמַר אַחֵר: כָּל
הַמִּתְאַפֶּק. הוּא עַל שַׁעֲרֵי הַיֶּשַׁע דּוֹפֵק. אָמַר אַחֵר:
שְׁלשָׁה לֹא יִשְׂבְּעוּ מִשְּׁלשָׁה. אֹזֶן מִדְּבָרִים. וְאֶרֶץ
מִמְּטָרים. וּנְקֵבָה מִזְּכָרִים. אָמַר אַחֵר: מִי שֶׁיֶּשׁ־לו יַחַס
נִכְבָּד. רָאוּי לְשָׁמְרוֹ בְטוּב הַמַּעֲשִׂים פֶּן יֹאבַד. אָמַר
אַחֵר: שְׁמֹר פִּיךָ ונִיבֶךָ. פֶן תִּהְיֶה חֲלַל חַרְבֶּךָ.

אָמַר אַחֵר: כָּל הָרוֹאֶה נְסִיעָתוֹ מִן הַנֹּעַר אֶל
הָעֶדְנָה. יֵדַע כִּי כֵן יִסַּע מִן הָעֶדְנָה אֶל הַזִּקְנָה. אָמַר
אַחֵר: הַתְּשׁוּבָה לַחֲטָאִים. כְּמוֹ הָרְפוּאָה לַתַּחְלוּאִים.
אָמַר אַחֵר: מִבְחַר הַמִּדּוֹת שְׁתִיקָה תּוֹלַדְתָּהּ מֵעָנָוָה.
וְיקַר־רוּחַ […] דוֹמֶה לְגַאֲוָה. אָמַר אַחֵר: חָלפוּ יְמֵי
הַנַּעֲרוּת, וְעָיְפָה מִרְכֶּבֶת הַבַּחֲרוּת. בְּעִיֵּי הַתַּאֲוָה
וְהַשְּׁרִירוּת. אָמַר אַחֵר: מִי שֶׁיִּהְיֶה זַךְ בְּמַעֲלָלָיו. לא
יִמְצְאוּ בְיוֹם הַדִּין טַעֲנָה עָלָיו. אָמַר אַחֵר: מַרְבֶּה דְבָרִים
יִלָּבֵט. וְכָל מַחֲרִישׁ יִמָּלֵט. אָמַר אַחֵר: צֹרֶךְ הַנֶּפֶשׁ אֶל
הַמּוּסָר. כְּצֹרֶךְ הָאָרֶץ אֶל הַמָּטָר. אָמַר אַחֵר: הַנֶפֶשׁ
כְּצֶלֶם וְהַחָכְמָה יָפְיוֹ. אוֹ כְעֵץ וְהַחָכְמָה פִרְיוֹ. אָמַר
אַחֵר: הַנָּדִיב יִתְנַדֵּב לְנַפְשׁוֹ וּלְאוֹהֲבָיו. וְהַכִּילַי
יִצְפֹּן הוֹנוֹ לְאוֹיְבָיו. אָמַר אַחֵר: כְּשֶׁיֶּחֱטָא הָאִישׁ אַל
יַבִּיט לְמִעוּט מַעֲלוֹ. רַק לִגְדֻלַּת מִי שֶׁחָטָא לוֹ. אָמַר אַחֵר:
כָּל הָלּוֹמֵד חָכְמָה. אִם יִוָּרֵשׁ תְּהִי לוֹ לְעשֶׁר. וְאִם
יַעֲשִׁיר תַּדְרִיכֵהוּ בִנְתִיב יֹשֶׁר. אָמַר אַחֵר: הַמָּוֶת קָשֶׁה
מִכָּל מַה שֶׁעָבַר לְפָנָיו. וְּנָקֵל מִן הַבָּא אַחֲרָיו. אָמַר
אַחֵר: כְּמוֹ שֶׁיִּתְיַשְּׁרוּ הַמִּדּוֹת הָרָעוֹת בְּחֶבְרַת
הַיְשָׁרִים. כֵּן יִשָׁחֲתוּ הַמִּדּוֹת הַטּוֹבוֹת בְּחֶבְרַת
הָאַכְזָרִים. אָמַר אַחֵר: הַיּשֶׁר וְהַזַּכּוּת. עַמּוּדֵי
הַמַּלְכוּת. אָמַר אַחֵר: הָכֵן צֵדָה לְיוֹם הַפְּרִידָה. וּמִבְחַר
הַצֵּדָה הַיּשֶׁר וְהָעֲבוֹדָה. אָמַר אַחֵר;: כָּל מִי שֶׁלֹא יִסְבּל
חֵטְא אוֹהֵבו. יְשִׁיבֵהוּ אוֹיְבוֹ. אָמַר אַחֵר: טוֹב הֱיוֹת הָאָדָם
אִלֶּם וּמַשְׂכִּיל. מִהְיוֹת רַב דְּבָרִים וּכְסִיל. אָמַר אַחֵר:
הַרְבֵּה לַעֲשׂוֹת חֲסָדִים. כִּי הַזְּמַן מִתְהַפֵּךְ פַּעַם יַשְׁפִּיל
הַנִּכְבָּדִים. וּפַעַם יָרִים הָעֲבָדִים. אָמַר אַחֵר: הַמּוּסָר טוֹב
לְךָ מִן הַיַּחַס. כִּי הַיַּחַס צָרִיךְ לַמּוּסָר וְאֵין הַמּוּסָר
צָרִיךְ לַיַּחַס. אָמַר אַחֵר: מִי שֶׁיִּרְאֶה הַפַּח בְּעֵין לִבּוֹ.
לֹא יִפֹּל בּוֹ. אָמַר אַחֵר: הָאָדָם כְּצִפּוֹר כָּל יְמֵי הֶבְלוֹ.
וְתַעֲנוּגֵי עוֹלָם כְּרֶשֶׁת לְרַגְלוֹ. אָמַר אַחֵר: כָּל מַשְׁלִיט עַל
שִׂכְלוֹ תַּאֲוָתוֹ. לא יַצְלִיחַ בִּימֵי חֲיוֹתוֹ. וְלֹא יָנוּחַ
אַחֲרֵי מוֹתוֹ. אָמַר אַחֵר: מוּם הַנְּדָבָה הָעֲנָיּוּת וְהַחֶסְרוֹן.
וּמוּם הַדִּבּוּר הַיִּתְרוֹן. אָמַר אַחֵר: דּוֹרֵשׁ מֵאוֹהֲבָיו
תְהִלָּה בְלִי נְדָבָה. יִמְצָא בּוּז וְאֵיבָה. אָמַר אַחֵר: הֶחָכָם
נִכְבָּד וְאִם מִשְׁפַּחְתּוֹ בְזוּיָה. וַחֲבֵרָיו רַבִּים וְאִם הוּא
בְּאֶרֶץ נָכְרִיָה. אָמַר אַחֵר: הַשְׁקִיטוּ הַצּוּקוֹת הַגְּדוֹלוֹת.
בְּכֹחַ הַצְּדָקוֹת וְהַתְּפִלוֹת. אָמַר אַחֵר: הַפְּרִישׁוּת מִן
הָעוֹלָם רְפוּאָה לְכָל יָגוֹן. וְעשֶׁר מִכָּל חֶסְרוֹן. אָמַר אַחֵר:
מִבְחַר הַמַּעֲשִׂים תְּשׁוּבָה מִמְּשׁוּבָה, וּנְקִיוּת מֵאַכְזָריוּת.
וְטָהֲרָה מֵעֲבֵרָה.

אָמַר הַמַּגִּיד: וּבְהַשְׁלִימָם חָמוּדוֹת מִלֵּיהֶם.
וּמְלִיצוֹת מִשְׁלֵיהֶם. אָמֵר לָהֶם הַזָּקֵן: הִשְׂכַּלְתָּם.
וַחֲכָמִים סִכַּלְתֶּם. וְעַתָּה בְּרוּכִים אַתֶּם. וָאֶשְׁאַל לְאֶחָד
מִן הַתַּלְמִידִים העוֹמְדִים לְפָנָיו עַל עִנְיָנוֹ וּשְׁמוֹ. וְאַיֵּה
מְקוֹמוֹ. וְאָרֶץ בְּנֵי עַמּוֹ. וְהוּא הֵבִין שְׁאֵלָתִי. וַיַּעַן
וַיֹּאמֶר וּפָנָיו מְגַמָּתִי:

הַשׁוֹאֲלִים עָלַי לְהוֹדִיעָם

סוֹדִי וּמָה אַרְצִי וּמִי אָנִי.

לוּ אַחֲרִישׁ מַעֲשַׂי יְעִידוּן לִי

כִּי הַיְקָר בִּשְׁמוֹ קְרָאָנִי.

אֵלוֹן בְּצַעְנַנִּים מְגוּרוֹתַי

אוּלָם שְׁמִי חֶבֶר יְלִיד קֵינִי.



וּכְשָׁמְעִי שִׁירוֹתָיו. אָמַרְתִּי לְכָל הָעוֹמְדִים
סְבִיבוֹתָיו: זֶה כֶּתֶר הַתְּעוּדוֹת. וְאֹסֶף כָּל הַחֲמוּדוֹת.
אֲשֶׁר לוֹ בְּכָל חָכְמָה עֶשֶׂר יָדוֹת. אֲבָל הָיִיתִי נֶחְפָּז לָלֶכֶת
וְקָרָאתִי שָׁלוֹם אֵלָיו. וְנָטִיתִי מֵעָלָיו.
