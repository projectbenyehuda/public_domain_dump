


	
	ספר תחכמוני -שַׁעַר שְׁלשָׁה עָשָׂר
	



וִכּוּחַ הַנֶּפֶשׁ עִם הַגּוּף וְהַשֵּׂכֶל. מִי מִשְּׁלָשְׁתָּם הַנּוֹכֵל






נְאֻם הֵימָן הָאֶזְרָחִי: וַיְהִי כִּי אָרְכוּ לִי יְמֵי
הַמְּשׁוּבָה. נִכְסְפָה נַפְשִׁי לָשׁוּב לְדֶרֶך הַתְּשׁוּבָה. אוּלַי
אֲבָרֵר לִבִּי. מִסִּיג מַחְשָׁבִי. וַאֲטַהֵר נַפְשִׁי. מִדְּמֵי
כַחְשִׁי. וְאֶרְחַץ רַעְיוֹנִי. מִטֻּמְאַת זְדוֹנִי. וַיְהִי הַיּוֹם
וַאֲנִי עִם חֶבֶר הַקֵּינִי יוֹשֵׁב. וְהָיִיתִי בְּעִנְיְנֵי הָעוֹלָם
חוֹשֵׁב. וַיֹּאמֶר לִי: עָלָה בְלִבְּי לַעֲזֹב אֵלּוּ הָאֲרָצוֹת
וְלָשׁוּב לְאַרְצִי. וְאֶל מְחוֹז חֶפְצִי. וְעַתָּה שְׁאַל מָה אֶתֵּן
לָךְ בְּטֶרֶם אֶלָּקַח מֵעִמָּךְ. אָמַרְתִּי לוֹ: שְׁאֵלָה אַחַת אֲנִי
שׁוֹאֵל מֵעִמָּךְ. וְדוֹרֵשׁ מִמָּךְ. שֶׁתַּעֲזֹב לִי מֵחָכְמָתְךָ
מַזְכֶּרֶת. אֲשִׂימָנָּה לְרֹאשִׁי עֲטֶרֶת. וְתִהְיֶה לִי לְכָבוֹד
וּלְתִפְאֶרֶת. וּתְלַמְּדֵנִי מָשָׁל מִמִּשְׁלֵי עֲבוֹדַת אֱלֹהִים.
לָצֵאת בּוֹ מִמַּחְשַׁכִּים לִנְגֹהִים. וְתוֹדִיעֵנִי סוֹד הַגּוּף
וְהַנְּשָׁמָה וְעִנְיְנֵיהֶם. וּמִי יְקַבֵּל הָעֹנֶשׁ מֵהֶם. אִם
יְקַבְּלוּהוּ שְׁנֵיהֶם. אִם יַחְלְקוּהוּ בֵינֵיהֶם. אָמַר לִי:
הֵיטַבְתָּ לִשְׁאֹל. וְאֵדַע כִּי אֵין זוּלָתִי לִגְאֹל. וְהִנְנִי
מַגִּיד לְךָ דִּבְרֵי חָכְמָה. מֵעִנְיְנֵי הַגּוּף וְהַנְּשָׁמָה. אֲשֶׁר
נוֹדַע לְכֻלָּנוּ. וַאֲבוֹתֵינוּ סִפְּרוּ לָנוּ. אָמַרְתִּי לוֹ:
סַפְּרָה נָא לִי אִמְרָתְךָ הַנְּעִימָה. וַעֲצָתְךָ הַתְּמִימָה. וּתְהִי
מַשְׂכֻּרְתְּךָ שְׁלֵמָה. אָמַר: יָדוֹעַ תֵּדַע כִּי בְיוֹם מַעֲמַד
הַדִּין כְּשֶיִּפָּתְחוּ הַסְּפָרִים. לַחְקֹר מַעֲשֵׂי הָרְשָׁעִים
וְהַיְשָׁרִים. וְיַחְקְרוּ עַל הַצְּדָקוֹת וְהַשְּׁגִיאוֹת. וְיִשְׁקְלוּ
הָעֲבֵרוֹת וְהַזְּכֻיּוֹת. תִּפֹּל חֲרָדָה עַל הַחַטָּאִים וְיִדְרשׁ
כָּל אֶחָד מֵהֶם טְעָנוֹת וְתַחְבּוּלוֹת. וְיִתְעוֹלֵל עֲלִילוֹת. אָז
יִוָּעֲדוּ הַגּוּף וְהַנְּשָׁמָה לְמִשְׁפַּט אֱלֹהֵיהֶם. וְעַד
הָאֱלֹהִים יָבֹא דְבַר שְׁנֵיהֶם. וְכָל אֶחָד מֵהֶם יַגִּיד טַעֲנָתוֹ
וְהַשּׁוֹפֵט יִשְׁפֹּט בֵּינֵיהֶם.

וַתַּעַן הַנֶּפֶשׁ וַתֹּאמֶר לַגּוּף:

אוֹי לְךָ הַגּוּף הַנָּגוּף כִּי אַתָּה מַחֲרִיב עוֹלָם.
אֵינְךָ מִן הָאֵל נִכְלָם. אַתָּה עוֹלַלְתָּ לִי הַיָּגוֹן הַזֶּה. כִּי
בְשֶׁלְךָ הַצַּעַר הַגָּדוֹל הַזֶּה. כִּי בְכָל יוֹם הָיִיתָ מַרְבֶּה
חֵטְא וַעֲבֵרָה. וּמַגְדִּיל הַמְּדוּרָה. בְּרֹעַ מַחְשְׁבוֹתֶיךָ. וְרֹב
תּוֹעֲבוֹתֶיךָ. וְלֹא הָיָה לִבְּךָ וְרַעְיוֹנֶיךָ. כִּי אִם לָתוּר
אַחֲרֵי עֵינֶיךָ. וּלְהַמְתִּיק לְשׁוֹנֶךָ. וּלְמַלֹּאת בִּטְנֶךָ.
וּלְהִתְעַסֵּק בְּעִנְיָנֶיךָ. וְתַזְנוּנֶיךָ. וְכָל מַחְשְׁבוֹתֶיךָ
לְהַטְבִּיעֵנִי בְּטִיט וָרֶפֶשׁ. וּלְמָכְרֵנִי לַמָּוֶת בְּלִי
חֹפֶשׁ.

וַיַּעַן הַגּוּף וַיֹּאמֶר לַנֶּפֶשׁ:

הוֹי הַנֶּפֶשׁ הַחוֹטֵאת הֲלֹא אַתְּ בְּמַחְשְׁבוֹתַיִךְ
הוֹגַעְתִּנִי. וּבַעֲצוֹתַיִךְ מְשַׁכְתִּיִני. אֲהָהּ בִּתִּי הַכְרֵעַ
הִכְרַעְתִּנִי. כִּי מַה בְּיָדִי רָעָה וַפֶשַׁע. וְאֵיךְ אוּכַל
לַעֲשׂוֹת חֵטְא וָרֶשַׁע. לוּלֵי אֲשֶׁר עוֹרַרְתִּנִי. וּבְחַבְלֵי
שָׁוְא מְשַׁכְתִּנִי. כִּי אַתְּ הָיִית קוֹשֶׁרֶת לִבִּי בַּחֲבָרַיִךְ.
וּמוֹשֶׁכֶת גּוּפִי בַּעֲבוֹת שְׁקָרַיִךְ. וּמְעוֹרֶרֶת אוֹתִי בְּמֶתֶק
אֲמָרַיִךְ. וַאֲנִי אָבוֹא אַחֲרַיִךְ. וּמִלֵּאתִי אֶת
דְּבָרַיִךְ.

אָמְרָה הַנֶּפֶשׁ:

אֲנִי הָיִיתִי עֶלְיוֹנָה. יוֹשֶׁבֶת בַּמַּלְכוּת
רִאשׁוֹנָה. וְהָיִיתִי כְּיוֹנָה בְחֵיק אֱלֹהַּ מְקַנֶּנֶת. וְעַל
שֻׁלְחָנוֹ מְרַנֶּנֶת. וְיָרַדְתִּי מִמַּעֲלוֹת גְּבוֹהִים.
וְנִפְרַדְתִּי מִבְּנֵי הָאֱלֹהִים. וּבַאתִי אֵלֶיךָ לְהַחֲיוֹתְךָ.
לְהֵיטִיבְךָ בְּאַחֲרִיתְךָ. וְאַתָּה הֶעֱבַדְתַּנִי בְחַטֹּאתֶיךָ.
וְהוֹגַעְתַּנִי בַעֲוֹנוֹתֶיךָ. וּמְשַׁכְתַּנִי בְּחַבְלֵי
תַעְתּוּעֶיךָ. לְהַבְלֵי שַׁעֲשׁוּעֶיךָ. לְהִתְעַדֵּן בַּאֲכִילָה
וּשְׁתִיָה. וּלְהִתְעַנֵּג עִם כָּל צוּרָה יְפֵה-פִיָּה. וְלֶאֲכֹל
מִינֵי מַעֲדַנִּים. וְלִלְבּשׁ בִּגְדֵי עֲדָנִים. וְלִהְיוֹת בְּגָדֶיךָ
לְבָנִים. וְגוּפְךָ מְעֻדָּן בִּשְׁמָנִים. וּמַה לִי אֲנִי וְלַבְּגָדִים
הַלְּבָנִים אוֹ לַשְּׁמָנִים. וּמֵאַיִן לְנֶפֶשׁ זַכָּה וּנְקִיָּה.
אֲכִילָה וּשְׁתִיָּה. אוֹ לְחַבֵּק צוּרָה יְפֵה-פִיָּה. הֲלֹא כָל אֵלֶּה
הַתַּעֲנוּגִים בְּךָ יִדְבָּקוּ וְיִתְרָעוּ. וּבִי לוֹ יִפְגָעוּ.



אָמַר הַגּוּף: לֹא כֵן הַדָּבָר רַק אֲנִי הָיִיתִי
כְאֶבֶן דּוּמָה. מֻשְׁלָךְ עַל הָאֲדָמָה. עַד יָרַדְתְּ מִן הַשָּׁמַיִם.
וְהֶעֱמַדְתִּנִי עַל רַגְלַיִם. וּפָקַחַתְּ לִי לִרְאוֹת עֵינַיִם.
וְחָנַנְתְּ לִי לְמַשֵּׁשׁ יָדַיִם. וְלֶאֱכל לָשׁוֹן וְשִׁנַּיִם.
וּלְדַבֵּר שְׁפָתַיִם. וּמַה לִי וּלְאֵלֶּה הַתַּעֲנוּגִים בִּהְיוֹתִי.
לוּלֵי מִהַרְתְּ וּבָאת לִקְרָאתִי. וַתֶּחֱלַשׁ הַנֶּפֶשׁ לִדְבָרָיו.
וְלֹא מָצְאָה מַעֲנֶה עַל אֲמָרָיו. וְכִשְׁמֹעַ הַשֵּׂכֶל כִּי הַנֶּפֶשׁ
נִבְהֲלָה. וּמַכָּתָהּ מִיַּד הַיֵּצֶר נַחְלָה. וְגָבַר עָלֶיהָ הַגּוּף
בְּרֹב דְּבָרִים. וְעַל הָאֱמֶת הִגְבִּיר הַשְּׁקָרִים. אָז הִתְעוֹרֵר
הַשֵּׂכֶל וַיֹּאמֶר לַנֶּפֶשׁ: עוּרִי נֶפֶשׁ בָּרָה. מִכְּבוֹד אֵל
גְּזוּרָה. וּבְמַאֲסַר הַגּוּף עֲצוּרָה. וְהִשָּׁמְרִי מִן הַיֵּצֶר
הָרַע. אֲשֶׁר בָּךְ פְּרָעוֹת פָּרַע. וְכַאֲרִי שָׁכַב כָּרַע. לָכֵן
הִטָּהֲרִי מִטֻּמְאָתֵךְ. וְהִנָּקִי מִטֶּנֶף גְּוִיָּתֵךְ. וְשִׁיתִי
לִבֵּךְ לְמַעְלָה. לִנְוֵה הַכָּבוֹד וְהַתְּהִלָּה. אֲשֶׁר שָׁם הָיָה
אָהֳלֵךְ בַּתְּחִלָּה.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

לִבְשִׁי יְחִידָה עֹז וְאֶל-אֵל רַנְנִי

עוּפִי כְיוֹנָה אַךְ בְּצִלּוֹ קַנְנִי.

אִם אַתְּ עֲרֻמָּה מִכְּסוּת יֹשֶׁר וְתֹם

צֵל רַחֲמָיו דִּרְשִׁי וּבָם הִתְלוֹנְנִי.

עוִּרי קְחִי עִמָּךְ תְּשׁוּרַת מַאֲמָר

וּדְעִי בְיוֹם הַדִּין לְצוּר מַה תַּעֲנִי.

לִבֵּךְ בְּבֹרִית כַּבְּסִי תוֹךְ נַחֲלֵי

ישֶׁר וְהִזַּכִּי עֲדֵי תִתְלַבְּנִי.

מִכְרִי יְקָר הַיּוֹם וְטוּב מָחָר קְנִי

וּקְחִי רְצוֹן קוֹנֵךְ וְכָל הוֹנֵךְ תְּנִי.

סִפְדִי בְּמָרָה עַל יְמוֹתַיִךְ אֲשֶׁר

חָלְפוּ בְהֶבֶל וַעֲלֵיהֶם קוֹנְנִי.

דַּרְכֵּךְ רְחוֹקָה הִיא קְחִי צֵדָה וּמִ-

מָּרָה סְעִי אוּלַי בְּמִתְקָה תַחֲנִי.

הוֹי לָךְ עֲצֵלָה כִּי חֲבֵרַיִךְ לְאֵל

קָמוּ חֲצוֹת לַיִל אֲבָל אַתְּ תִּישְׁנִי.

טָבְעָה אֳנִי סִכְלֵךְ בְּיָם הַתַּאֲוָה

עוּרִי בְּחֹזֶק יָד וְהַעְלִי הָאֳנִי.

בָּתִּים וְעָרֵי מִסְכְּנוֹת לַתַּאֲוֹת

עָשִׂית וְלִבְנוֹתָם לְבֵנִים תִּלְבְּנִי.

מָחָר בְּיוֹם דִּין יִדְרְשׁוּ חֶשְׁבּוֹן לְמַעְ-

שַׂיִךְ וְתֹכֶן הַלְּבֵנִים תִּתְּנִי.

וּדְעִי הֲכִי מַצְרֵף בְּיַד הָאֵל לְכָל

נֶפֶשׁ וּבֹו תִצָּרְפִי תִבָּחֲנִי.

אִם תֵּצְאִי זַכָּה וּבָרָה שׁוֹרֲרִי

וּבְתֹף וְכִנּוֹר צַהֲלִי קוֹלֵךְ תְּנִי.

אַךְ אִם מְלֵאָה סִיג וְאָוֶן תֵּצְאִי

בֹּאִי בְנִקְרַת צוּר וְשָׁם הִטָּמְנִי.

וּבְרַק יְקֹד שִׁלְחִי וְהַמְטִירִי מְטַר

דִּמְעַה וְעַל לִבֵּךְ עֲנַן אֵד עַנְנִי.

הִתְנַעֲרִי דַלָּה עֲנִיָה סֹעֲרָה

מָתַי שְׁבִיָּה תֵצְאִי מִכּוּר עֳנִי.

עַד אָן בְּחַדְרֵי הַמְּרִי תִסָּתְרִי

וּבְאָהֳלֵי הַגַּאֲוָה תֵרָגְנִי.

וַתִּבְטְחִי עַל טוּב זְמַן בּוֹגֵד וְהוּא

קָנֶה רָצוּץ וּבוֹ תִשָּׁעֲנִי.

גֵּרַשְׁתְּ צֶדֶק אֲשֶׁר הוּא בוֹעֲלֵךְ

אַךְ עִם בְּנֵי הַתַּאֲוָה תִתְחַתְּנִי.

שׁוּבִי אֲסִירָה לַאדֹנַיִךְ וְאַל

תִּתְאַחֲרִי לָשׁוּב וְאַל תֵּעָגְנִי

אוּלַי בְּתוֹךְ מַחְנֵה אֱלֹהִים תַחֲנִי

עַמִּי וְהָיָה זַךְ וְקָדוֹשׁ מַחֲנִי.

צוּרִי חֲקַרְתַּנִי וְאֵיכָה מִמְּךָ

אַסְתִּיר עֲוֹנוֹתַי וְאַתָּה בוֹחֲנִי.

אֶפְצַר לְהִמָּלֵט וְיִצְרִי רוֹדְפִי

תָּמִיד כְּשָׂטָן עַל יְמִינִי שׂוֹטְנִי.

שַׁדַּי בְּזָכְרִי רֹב חֲסָדִים שֶׁגְּמַל-

תַּנִי מְאֹד אֵבוֹשׁ וְאֹמַר מָה אֲנִי.

אֶשְׁכַּח אֲנִי עַצְמִי וְאַּתָּה שׁוֹמְרִי

אֶשְׁכֹּן נְוֵה חַסְדָּךְ וְאַתָּה שׁוֹכְנִי.

בַּשֵּׂר אֲדוֹן עוֹלָם לְנֶפֶשׁ נַעֲנָה

עוּרִי, בְּנַהְרֵי נָעֳמִי הִתְעַדְּנִי.

כִּי לֹא לְמַעְנֵךְ אֶגְמְלֵךְ חֶסֶד וְחֵן

רַק בַּעֲבוּרִי אֶעֱשֶׂה וּלְמַעֲנִי.

עֵת תִּקְרְאִי אוֹתִי בְצָרָה אֶעֱנֶה

לָךְ כִּי אֲנִי הוּא הַמְדַבֵּר הִנְנִי.

אָמְרָה הַנֶּפֶשׁ לַשֵּׂכֶל:

וְאֵיךְ יוּכַל לְהִמָּלֵט מִכַּבְלוֹ. וְלִשְׁבֹּר
עֻלּוֹ. מִי שֶׁנָּפַל בְּיַד אֲדוֹנִים קָשֶׁה. וְטָבַע בִּמְצוּלוֹת יָם
וְאֵין לוֹ מוֹשֶׁה. הַיֵּצֶר לִימִינִי כַּנָּחָשׁ יִשָּׁכֵנִי. וְהַגּוּף
בַּעֲבוֹתוֹ יִמְשְׁכֵנִי. וַאֲנִי בְתוֹכָם יְחִידָה וְגַלְמוּדָה.
וּבְמַאֲסָרָם לְכוּדָה. זֶה נוֹשֵׁךְ וְזֶה מוֹשֵׁךְ. אוֹיָה לִי כִּי
גַרְתִּי מֶשֶׁךְ.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:



מֵהֵיכְלֵי גַלְגַּל יְקָר גָּלִיתִי

וּבְבוֹר חֲשֵׁכִים אָהֳלִי נָטִיתִי.

וַיְשַׁלְּחוּנִי מִגְּלִילֵי מַעְלָה

לַגּוּף וְתוֹכוֹ בֵית זְבוּל בָּנִיתִי.

אֶתְמוֹל בְּחֵיק צוּרִי כְעַלְמָה צוֹחֲקָה

מִתְרַפְּקָה מִשְׁתַּקְשְׁקָה הָיִיתִי.

אַחַר הֱיוֹתִי כַּבְּתוּלָה אֵרְשׂוּ

אוֹתִי וְגוּף טָמֵא לְאִישׁ קָנִיתִי.

וַיִּבְעֲלָה אוֹתִי בְטֻמְאָתוֹ וּמִ-

יַּלְדֵי זְנוּנָיו יוֹם וְלַיִל הָרִיתִי.

אֵלֵד עֲלֵי בִרְכָּיו בְּכָל יוֹם רִבֲבוֹת

בָּנִים וְטִפַּחְתִּי וְגַם רִבִּיתִי.

אוֹחִיל לְפֵרוּד אֹהֲבִי הַזַּךְ וְהָ-

רִאשֻוןֹ וְעָלָיו יוֹם וְלֵיל בָּכִיתִי.

לִבִּי לְפֵרוּדוֹ כְּחָלִיל יֶהֱמֶה

עַד מִסְּבֹל הָאַהֲבָה חָלִיתִי.

לֹא יָכְלוּ הָרִים סְבֹל חִשְׁקוֹ וְנִלְ-

אוּ מִנְּשׂוֹא אוֹתוֹ וְלֹא לָאִיתִי

לָמָה זְנָחַנִי כְאִשָׁה זָנְתָה

תַּחַת אֲדֹנֶיהָ וְלֹא זָנִיתִי.

לָמָּה פְרָעַנִי וְהִשְׁקַנִי בְּמֵי

מָרִים מְאֹרָרִים וְלֹא שָׂטִיתִי.

לוּ יַעְזְרוּנִי עַנֲנֵי עֵינַי אֲזַי

תֵּבֵל בְּמַבּוּל הַבְּכִי כִסִּיתִי.

אֶחְפֹּר בְּבַת עֵינִי בְּאֵרוֹת עָמְקוּ

גַּם מִכְּבֵדִי מֵי בְכִי דָלִיתִי.

אֶבְכֶּה בְּשׁוּרִי הַנְּשָׁמוֹת הַקְּדוֹ-

שׁוֹת יַעֲלוּ אֶל אֵל וְלֹא עָלִיתִי.

כָּל-עֵת בְּגוּף אֶשְׁכֹּן אֲנִי בוֹ נֶחְשְׁבָה

מֵתָה וְאִלּוּ אֵצְאָה חָיִּיתִי.

הָיָה בְּרוֹם קִנִּי כְּמוֹ יוֹנָה וּמִ-

שַּׁחַק אֱלֵי אֶרֶץ כְּעוֹף דָּאִיתִי.

עַל הַרֲרֵי בֹשֶׂם בְּגַלְגַּלֵּי זְבוּל

שׁוֹשַׁן בְּגַנֵּי אַהֲבָה רָעִיתִי.

וָאֶשְׁאֲפָה רֶקַח בְּזָכְרִי יוֹצְרִי

גַּם מִכְּבוֹדוֹ מָר-דְּרוֹר אָרִיתִי.

הֵן בַּעֲדוֹ עוֹבְרֵי דְרָכִים אֶשְׁאֲלָה

וַעְלֵי נְתִיבוֹת בַּעֲדוֹ צִפִּיתִי.

נָא סַפְּרוּ אֵלָיו תְּשׁוּקָתִי הֲכִי

לִרְוֹת בְּמֵימֵי נָעֳמוֹ צָמִיתִי.

נָא עוֹבְרֵי דֶרֶךְ שְׂאוּ שָׁלוֹם לְדוֹד

שֶׁאָהֲבָה נַפְשִׁי וְלֹו כָלִיתִי.

מִיּוֹם גְּלוֹתִי מֵאֲכֹל עַל שֻׁלְחֲנוֹת

טוּבֹו הֲלֹא בְצוֹם עָנִיתִי.

מִיוֹם אֲשֶׁר כּוֹסוֹת אֲהָבָיו נִמְנְעוּ

מִפִּי בְּכוֹסוֹת לַעֲנָה רָוִיתִי.

יִמַר לְחִכִּי צוּף דְּבַשׁ בִּלְתּוֹ וְכָל

כָּבוֹד לְבַד הוֹד אֹהֲבִי בָּזִיתִי.

הִכָּה שְׁתֵּי עֵינִי בְּסַנְוֵר הַנְּדוֹד

עד זהרם חָשַׁךְ וְלֹא רָאִיתִי.

אַחֲרֵי הֶיוֹתִי אָז בְּהֵיכַל אֵל כְּנֵר

דוֹלֵק בְּגוּף אָפֵל הֲלֹא כָהִיתִי.

עֵת אֶזְכְּרָה יוֹם דִּין לְאֶרֶץ יִפְּלוּ

פָנַי וְאֵימָה כַמְּעִיל עָטִיתִי.

הִנֵּה כְעֵינֵי הָעֲבָדִים אֶל יְדֵי

אָדוֹן שְׁתֵּי עֵינַי לְאֵל תָּלִיתִי

עַד כִּי יְשִׁיבֵנִי לְאַפִּרְיוֹן יְקָר

לַחְסוֹת בְּצִלּוֹ כַּאֲשֶׁר חָסִיתִי.



וַיַעַן הַיֵּצֶר וַיֹּאמֶר לַנֶּפֶשׁ:

שִׁמְעִי נָא זֹאת עֲנִיָּה. אֲשֶׁר מִיֵּין אֲנָחוֹת
כּוֹסָהּ רְוָיָה. דְּעִי כִּי חֶלְדֵּךְ כְּאַיִן. וְהוֹדֵךְ יָסוּף
כְּהֶרֶף עָיִן. וּמִשְׁמַן בְּשָׂרֵךְ יֵרָזֶה. וּבֵית אֲבָנִים יֶחֱזֶה.
וּכְרֶגַע יֹאבַד הוֹדֵךְ. וְיִשָּׁבֵר יְגוֹנֵךְ וְאֵידֵךְ. לָכֵן אַל
תָּמִירִי חֶדְוָתֵךְ. בְּאַנִחָתֵךְ. וְאַל תַּעַזְבִי
בָּעוֹלָם הַזֶּה שִׂמְחָתֵךְ. כִּי אַחֲרֵי הַמָּוֶת אֵין
שִׂמְחָה וְאֵין מַרְגֵּעָה. וְאֵין יְשִׁיבָה וְאֵין נְסִיעָה. ובְעוֹדֵךְ
בַּחַיִּים אִם תִּשְׂמְחִי תוֹעִילִי. וְלַעֲשׂוֹת תַּשְׂכִּילִי. כִּי
אַחֲרֵי מוֹתֵךְ לֹא תוּכְלִי. לְהָשִׁיב אֲשֶׁר אִבַּדְתְּ. וּלְהֵיטִיב
אֲשֶׁר הֵרַעַתְּ. וְלֹא לִתְפֹּר אֲשֶׁר קָרַעַתְּ. וְעַתָּה עִזְבִי
הַשֵּׂכֶל וּשְׁקָרָיו. כִּי לֹא יוּכַל לְקַיֵּים לָךְ אֶת
דְּבָרָיו.

וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר:

נֶפֶשׁ יְקָרָה אֵיךְ בְּשִׂכְלֵךְ תִּבְטְחִי

גִּילִי וְחֶלְקֵךְ מִזְּמַן אַל תִּשְׁכְּחִי.

כִּי אַחֲרֵי מוֹתֵךְ כְּאֶפֶס תִּהְיִי

לֹא תִשְׂמְחִי יוֹתֵר וְלֹא תֵאָנְחִי.

אֶתְמוֹל הֲיָשׁוּב אַחֲרֵי לֶכְתּוֹ. וְכֵן

אִישׁ אַחֲרֵי מוֹתוֹ לְעוֹלָם לֹא יְחִי.

אָכֵן בְּעוֹדֵךְ בַּגְּוִיָה שׁוֹכְנָה

אָז תּוּכְלִי לַעְלֹז וּבָהּ אָז תִּשְמְחִי.

אָז תּוּכְלִי טוֹבָה לְעַצְמֵךְ לַעֲשׂוֹת

לָכֵן בְּעוֹסְקֵי רִיק וְשָׁוְא אַל תִּבְטְחִי.

לָמָּה נְוֵה מָשׂוֹשׂ וְשַׁלְוָה תִּסְגְּרִי

אַךְ שַׁעֲרֵי יָגוֹן לְנַפְשֵׁךְ תִּפְתְּחִי.

הָרוֹצְחִים הֵמָּה בְּבַרְזֶל הוֹרְגִים

וּבְלִי כְלִי נַפְשֵׁךְ עֲנִיָּה תִּרְצְחִי.

תֵּבֵל נְוֵה תַעְנוּג וְלָמָּה מִנְּוֵה

הַתַּעֲנוּג אֶל הַמְּצוּקִים תִּבְרְחִי.

שִׂכְלֵךְ נְתָנֵךְ תּוֹךְ כְּלוּא הַמַּאֲסָר

לָכֵן עֲנִיָּה סֹעֲרָה הִתְפַּתְּחִי.

שִׂמְחִי בְטֶרֶם מִנְּוָתֵךְ תֵּצְאִי

גִּילִי מְאֹד טֶרֶם אֲשֶׁר תִּלָּקְחִי.

הָאֵל עֲלֵי כָל הַבְּרוּאִים יַחֲלֹק

טוּבוֹ וְלָכֵן מַהֲרִי חֶלְקֵךְ קְחִי.

מִנְעִי יְחִידָה מִבְּכִי קוֹלֵךְ וְנַפְ-

שֵׁךְ מִנְּשׂא עָמָל וְרַגְלֵךְ מִדְּחִי.

אִם תַּעֲלִי מִבּוֹר עֱנוּתֵךְ תִּגְבְּהִי

אִם תֵּצְאִי מִתּוֹךְ חֲשֵׁכִים תִּזְרְחִי.

כִּי אַתְּ בְּתוֹךְ גּוּפֵךְ כְּמוֹ כוֹכָב. וְעֵת

מוֹתֵךְ תְּהִי בִשְׁאוֹל לְכָל רֶגֶל סְחִי.

אַךְ אַחֲרֵי צֵאתֵךְ דְּעִי כִי לֹא לְעַד

לָךְ יָאמְרוּ בּוֹאִי וְתוֹךְ גּוּפֵךְ פְּחִי.

אָמַר הַשֵּׂכֶל לַנֶּפֶשׁ:

נֶפֶשׁ קְדוֹשָׁה. בְּעֹל הַיֵּצֶר כְּבוּשָׁה. אַל
יְסִיתוּךְ שְׁקָרָיו. וְאַל יַכְחִידוּךְ דְבָרָיו. וּדְעִי כִּי מֶלֶךְ
בָּשָׂר וָדָם אִם אִישׁ יְשָׁרְתֵהוּ. לֹא יִשְׁקֹט עַד יִגְמְלֵהוּ.
וְיָשִׁיב לוֹ כְמַעֲשֵׁהוּ. וְאִם אָדָם הוּא גּוּשׁ וְרִמָּה. וְעָפָר
מִן הָאֲדָמָה. יוּכַל לְהָשִׁיב גְּמוּל לְעַבְדּוֹ. כָּל שֶׁכֵּן בּוֹרֵא
הַכֹּל כִּי לֹא תִקְצַר יָדוֹ. לְהָשִׁיב גְּמוּלָתֵךְ. אַחֲרֵי צֵאתֵךְ
מִגְּוִיָּתֵךְ וּלְהַחֲיוֹתֵךְ. עַד תְּקַבְּלִי שָׂכָר טוֹב
לִפְעֻלָּתֵךְ. וְיֵשׁ תִּקְוָה לְאַחֲרִיתֵךְ. אָמְרָה הַנֶּפֶשׁ: אֶמְצָא
חֵן בְּעֵינֶיךָ אֲדוֹנִי כִּי נִחַמְתָּנִי.. וּבְדִבְרֵי פִיךָ
חִזַּקְתָּנִי. וּמִן הַיֵּצֶר הָרַע הִצַּלְתַּנִי וְחִלַּצְתָּנִי.
וְעַתָּה יָדַעְתִּי כִּי יֵשׁ לִי תִקְוָה בְּרַחֲמֵי הָאֵל לַהֲשִׁיבֵנִי
לִנְוֵה הָאוֹר וְהַתְּהִלָּה. וְתָמִיד יֵשׁ לִבְטֹחַ. עַל אֱלֹהֵינוּ
כִּי יַרְבֶּה לִסְלֹחַ.

וַיִּשָּׂא מְשָׁלוֹ וָיֹּאמַר:

אֲנִי חֶבֶר אֲנִי חֶמְדַּת נְבוֹנִים

וּמוֹלַדְתִּי בְּאֵלוֹן צַעֲנַנִּים.

אֲעוֹרֵר הַנְּפָשׁוֹת הַיְשָׁנוֹת

בְּתוֹכָחוֹת כְּמוֹ חִצִּים שְׁנוּנִים.

וְתָמִיד אֶפְתְּחָה שַׁעְרֵי תְּשׁוּבָה

לְגוֹי צַדִּיק וְגַם שׁוֹמֵר אֱמוּנִים.

אֲחַלֵּץ הַגְּוִיָּה מִיקוֹד אֵשׁ

וְאַצִּיל הַנְּשָׁמָה מֵחֲרוֹנִים.

וְהַיֵּצֶר וְהַתַּאֲוָה יְסוּפוּן

כְּמוֹ קַשׁ אוֹ כְּמוֹ אוּדִים עֲשֵׁנִים.

לְזֹאת אַשְׁרֵי אֱנוֹשׁ יָפִיק רְצוֹן אֵל

וְיִתְעוֹרֵר בְּעֵת אִישִׁים יְשֵׁנִים.

וְיַעְקֹר מִלְבַבוֹ חֵטְא וְאָוֶן

וְיִטַּע בּוֹ נְטָעִים נַעֲמָנִים.

וְאָז יֵאוֹר בְּאוֹר חַיִּים וְיִזְכֶּה

רְאוֹת אוֹר יוֹצְרוֹ פָּנִים בְּפָנִים.
