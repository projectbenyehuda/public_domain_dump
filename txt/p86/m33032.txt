


	
	ספר תחכמוני - שַׁעַר שְׁלשִׁים וְשִׁבְעָה
	



מִן הַנָּשׁוּךְ מֵהַנָּחָשׁ. אֲשֶׁר אֵין לוֹ לָחַשׁ




נְאֻם הֵימָן הָאֶזְרָחִי: נָסְעִּתי מִצּוֹר. וְעַד
חָצוֹר. עִם חֲבוּרָה. מְתֵי זְרוֹעַ עִם גְּבוּרָה. וּבְעוֹדֵנוּ
הוֹלְכִים עַל הַנָּתִיב בַּמִּדְבָּר בָּעֲרָבָה מָצָאנוּ כַּר נִרְחָב.
וְעֵמֶק נֶחְמָד וְנֶאֱהָב. לָבְשׁוּ צְמָחָיו מִשְׁבְּצוֹת זָהָב.
וְעָלָיו נִצָּנָיו כְּנִיצוֹצֵי לַהַב. וְנָשְבוּ רוּחוֹתָיו. מֵאַרְבַּע
רוּחוֹתָיו. וְנָזְלוּ טְלָלָיו. וְהִסְתַּבְּכוּ צְלָלָיו. וְעָלוּ
צְמָחָיו עַד לֵב הַשָּׁמַיִם. וְהָאֲדָמָה אֶרֶץ נַחֲלֵי מָיִם.
וְכַאֲשֶׁר בּוֹ פָגַעְנוּ. שִׂמְחָה גְדוֹלָה שָׂמַחְנוּ. וּבְתוֹכָהּ
נַחְנוּ. וּבְחַרְנוּהוּ מִרְבָּץ לַבְּהֵמוֹת. וּמְנוּחָה לַנְּשָׁמוֹת.
וְיָשַׁבְנוּ שָׁם לִמְצֹא מַרְגֵּעָה. מֵרֹב הַיְגִיעָה. וְנָפְלָה
עָלֵינוּ תַּרְדֵּמָה. וְנִלְכַּדְנוּ בְּחַבְלֵי הַתְּנוּמָה.
וּבְעוֹדֵנוּ יְשֵׁנִים שָׁמַעְנוּ קוֹל חֲרָדָה. וּפַחַד קְרָאָנוּ
וּרְעָדָה. וְכַאֲשֶׁר קַמְנוּ רָאִינוּ וְהִנֵּה אִישׁ מִן הַחֲבוּרָה.
צוֹעֵק צְעָקָה גְדוֹלָה וּמָרָה. יְבַקֵּעַ צוּר הַחַלָּמִישׁ
בְּצַעֲקוֹתָיו. וִיפוֹצֵץ הַלְּבָבוֹת בְּשַׁאֲגוֹתָיו. וְכַאֲשֶׁר
שְׁאִלְנוּהוּ. וְעַל סוֹדוֹ חֲקַרְנוּהוּ. אָמַר לָנוּ כִּי נְשָׁכָהוּ
נָחָשׁ. אֲשֶר אֵין לוֹ לָחַשׁ. וְלֹא מָצָאנוּ לוֹ עֵצָה. כַּאֲשֶׁר
הַנֶּפֶשׁ חֲפֵצָה. כִּי הָיִינוּ בְאֶרֶץ צִיָּה. לֹא מְקוֹם רְפוּאָה
וּרְטִיָּה. וְלֹא מָזוֹר לְמַכָּה טְרִיָה. וּבְעוֹדֵנוּ נֶעֱצָבִים
לְעָצְבּוֹ. וּבוֹכִים לְמַכְאוֹבוֹ. עָמַד עָלָיו אִישׁ. לְשׁוֹנוֹ
כְצִפְעוֹנִי יַפְרִישׁ. וַיֹּאמֶר לָנוּ: לָמָה תִתְרָאוּ. וַתִּרְאוּ
חֲתַת וַתִּירָאוּ. אִם תִּתְּנוּ שְׂכָרִי אֲנִי אֶרְפָּא מַחֲלָתוֹ.
וּמַחַץ מַכָּתוֹ. וְתִרְאוּ עַיִן בְּעַיִן רְפוּאָתוֹ. וּבְטֶרֶם נָסְעוֹ
מִזֶּה יָשׁוּב לְקַדְמָתוֹ. אָמַרְנוּ לוֹ: אִם תָּקִים דְּבָרֶיךָ.
וּתְשַׁלֵּם נְדָרֶיךָ. נַרְבֶּה שְׂכָרֶךָ. וְאָז יִבָּקַע כַּשַּׁחַר
אוֹרֶךָ. וַיְמַהֵר וַיַּעַשׂ עַל הָאָרֶץ עִגּוּל מֵאֲבָנִים. וַיָּשֶׂם
עֲלֵיהֶם עֵצִים קְטַנִּים. וְכִסָּה עַל הָעֵצִים עָפָר וַיַּעֲשֵׂהוּ
כְּמִין מְעָרָה. וְעָשָׂה צוּרַת פֶּתַח עַל הַמְּעָרָה. וְעָמַד
כְּצוֹפֶה עַל מִשְמֶרֶת. וְהֵחֵל לְהַקְטִיר קְטֹרֶת.

וַיַּעַן וַיֹּאמַר:

שִׁמְעוּ הָרִים רִיבִי. וְהַאֲזִינוּ נִיבִי. וְהַטּוּ
אָזְנְכֶם בְּהַטִּיפִי. לְאִמְרֵי פִי. אֳנִי מַשְׁבְּיעַ הַמִּדְבָּר
וְהָעֲרָבָה. וְהַיָּם וְהֶחֳרָבָה. וְהַשּׂוּחוֹת וְהַגֵּאָיוֹת.
וְהַתַּלְאוּבוֹת וְהַצִּיּוֹת. וּמְעוֹנוֹת אֲרָיוֹת. וּפְרִיצֵי חַיּוֹת.
בְּחַרְבִּי הַשְּׁלוּפָה. וּבִזְרוֹעִי הַחֲשׂוּפָה. וּבַסְּעָרָה
הַמְרַחֶפֶת. וּבָאֵשׁ הַשּׂוֹרֶפֶת. וּבַיִּפְעָה הַשּׁוֹאֶפֶת.
וּבַשִּׁפְעָה הַשּׁוֹטֶפֶת. גָּזַרְתִּי עַל כֻּלְכֶם לְעוֹרֵר
הַנְּחָשִׁים מִמַּחֲבוֹאֵיהֶם. וּלְגָרֵשׁ הַצִּפְעוֹנִים מֵחוֹרֵיהֶם.
וּלְהוֹצִיא הָעַקְרַבִּים וְהַפְּתָנִים מִמְּבוֹאֵיהֶם. וְכָל אֲשֶׁר
הוּא מֵהֶם בַּעַל כָּנָף מְעוֹפֵף. אֲשֶׁר עַמּוּדֵי שָׁמַיִם מְרוֹפֵף.
יָבוֹאוּ כֻלָּם. מֵאַפְסֵי עוֹלָם. וְיִתְקַבְּצוּ עַל כַּנְפֵי רוּחוֹת.
מֵאַרְבַּע רוּחוֹת. אֵין לָהֶם מָנוֹס וְלֹא מִפְלָט. וְלֹא מָעוֹז וְלֹא
עִיר מִקְלָט. וְאִישׁ מֵהֶם אַל יְהִי נִמְלָט. עַד יָבוֹאוּ כֻלָּם
בְּכֶפֶל רִסְנִי וְיִלָּכְדוּ בְּחַבְלִי. וּבְמַאֲסַר עֻלִּי.
הִתְקַדְּשׁוּ וְהִתְנַגְּשׁוּ. הִתְאוֹשְׁשׁוּ וְהִתְקוֹשְׁשׁוּ.
וְהִתְרוֹצְצוּ וְהִתְפּוֹצְצוּ. וּמִכַּל צַד הִתְקַבְּצוּ. יֵצֵא כָל
צִפְעוֹנִי מֵחֶדְרוֹ. וְכָל נָחָשׁ מֵחוֹרוֹ. וְכָל אֶפְעֶה מִנִּקְרַת
צוּרוֹ. וְכָל עַקְרָב מִקִּירוֹ. וְכָל שָׂרָף מִמְּאוּרָתוֹ. וְכָל
פֶּתֶן מִמְּעוֹנָתוֹ. בְּעֹז גְּבוּרוֹת מַאֲמִירוֹ וּמַמְאִירוֹ.
וּבִגְזֵרַת מַשִּׁירוֹ וּמְסִירוֹ. וּבְגַעֲרַת מְעִירוֹ וּמַבְעִירוֹ
מֵחֶדְרוֹ. וּמַהֲרוּ וְנַהֲרוּ. וְאַל תְּאַחֲרוּ. וּפָנַי שַׁחֲרוּ.
הִתְקָרְבוּ. וְאַל תִּתְעַכְּבוּ. וְהִתְקוֹשְׁשוּ. וְאַל תְּבוֹשְׁשׁוּ.
וְהִתְמַהֲרוּ. וְאַל תִּתְמַהְמְהוּ. בְּטֶרֶם אֲשַׁלַּח בָּכֶם שְׁפָטַי
הָרָעִים. וְצִירֵי נְגָעִים לִרְגָעִים. וּמִשְׁלַחַת מַלְאֲכֵי רָעִים.
וּמַכָּה מְשַׁכֶּלֶת. וְאֵשׁ אוֹכֶלֶת. בְּעֹז אֲשֶׁר לוֹ הַמֶּמְשֶׁלֶת.
וְהַגְּבוּרָה וְהַיְכֹלֶת.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אַשְׁבִּיעֲכֶם כָּל-זוֹחֲלֵי עָפָר

הַשּׁוֹכְנִים צִיָּה וְתַלְאוּבוֹת.

גַּם בַּעֲלֵי כָנָף וְכָל-שָׂרָף

שׁוֹכֵן בְּמִדְבָּרוֹת וּבָחְרָבוֹת.

בָּאֵל וּבִכְבוֹדוֹ וְעֹז יָדוֹ

וּבְעַנְנֵי רוּחוֹת וְהֶעָבוֹת.

לָבוֹא לְמִצְוָתִי וְאֶל חֶפְצִי

תִּהְיוּ מְזֻמָּנוֹת וְנִצָּבוֹת.

טֶרֶם בְּחֵילְכֶם אֶשְׁלְחָה שַׂרְפֵי

זַעַם חֲצוּבֵי אֵשׁ וְלֶהָבוֹת.

וַיַּחֲרִישׁ כְּפִי שָׁעָה וְהוּא עַל הָאָרֶץ יוֹשֵׁב.



וְהִקְשִׁיב קֶשֶׁב רַב קָשֶׁב. וְהִנֵּה קוֹל שָׁאוֹן מֵעִיר. הַיְשֵׁנִים
מֵעִיר. וְהַלְּבָבוֹת מַבְעִיר. וּסְלָעִים יְפוֹרֵר. וּמֵתִים יְעוֹרֵר.
וְאָבָק עוֹלֶה עַד לֵב הַשָּׁמָיִם. וּבָא הַשֶּׁמֶשׁ בַּצָּהֳרָיִם.
וְחָשְׁכוּ הָעֵינָיִם. וְכַאֲשֶׁר נָשָׂאנוּ עֵינֵינוּ. רָאִינוּ.
וְהִנֵּה מִכָּל צַד חַיּוֹת מְלֵאוֹת עֵינַיִם. וּשְׂרָפִים בַּעֲלֵי
כְנָפַיִם. מֵהֶם עוֹמְדִים בְּלִי רַגְלַיִם. וּמֵהֶם בְּרֹאשָׁם
קַרְנַיִם. וְאַחֲרֵיהֶם נְחָשִׁים. וְצִפְעוֹנִים מְשֻׁנִּים וְקָשִׁים.
מֵהֶם עוֹמְדִים בְּלִי רַגְלַיִם כְּמוֹ קְרָשִׁים. וּמֵהֶם בִּשְׁנֵי
רָאשִׁים. יַרְעִישׁוּ הָאָרֶץ בְּרוּצָם. וְיַחֲרִידוּ הָעוֹלָם
בְּקִבּוּצָם. וְכֻלָּם בָּאִים לְעֻמָּתֵנוּ. וְשָׂמוּ מְגַמַּת פְּנֵיהֶם
אֵלֵינוּ. וּבְחִצֵּי עֵינֵיהֶם יוֹרוּנוּ. וְכִמְעַט חַיִּים בְּלָעוּנוּ.
וְכַאֲשֶׁר רָאִינוּ הַמַּרְאֶה הַגָּדוֹל הַזֶּה קִדַּמְנוּ לִבְרֹחַ.
וְלֹא עָצַרְנוּ כֹחַ. וַיִּצְחַק הָאִיׁש עָלֵינוּ. וַיֹּאמֶר אֵלֵינוּ:
יֵשֵׁב כָּל אֶחָד מִכֶּם בַּהֲדוֹמוֹ. וְאַל יֵצֵא אִישׁ מִמְּקוֹמוֹ.
וְחִזְקוּ אַל תִּירָאוּ. וְהִתְיַצְּבוּ וּרְאוּ. נִפְלְאוֹת הַבּוֹרֵא
אֲשֶׁר לִמַּד לַעֲבָדָיו. וְגִלָּה לָהֶם סוֹדָיו. וַיָּקָם מְהֵרָה.
וּפָתַח פִּי הַמְּעָרָה. וְנִכְנְסוּ שָׁם. הַנְּחָשִׁים כֻּלָּם. וְסָגַר
פִּי הַמְּעָרָה עֲלֵיהֶם אַחֲרֵי כֵן דִּבֶּר בְּלַחַשׁ. וְהִשְׁבִּיעַ
כָּל נָחָשׁ. וַיִּפְתַּח פִּי הַמְּעָרָה וַיֵּצְאוּ כֻלָּם אֶחָד אֶחָד.
וְהָלְכוּ יַחַד. וְלֹא נִשְׁאַר בַּמְּעָרָה כִּי אִם אֶחָד. וְכֹל אֲשֶׁר
יִרְצֶה לִבְרֹחַ. לֹא יַעֲצָר-כֹּחַ. וּפַעַם יִתְחַזֵּק וְלָנוּס
יַחְשֹׁב. וַיֵּצֵא יָצוֹא וָשׁוֹב. וְלֹא יָכֹל לִבְרֹחַ וְלָסוּר.
כְּאִלּוּ הוּא בְכַבְלֵי בַרְזֶל אָסוּר. וְכַאֲשֶׁר רְאִינוּהוּ
בִמְקוֹמוֹ מַחֲזִיק. יָדַעְניּ כִּי הוּא הַמַּזִיק. וַיִּקְרָא הָאִישׁ
וַיֹּאמַר: הַגִּישׁוּ אֵלַי הַחוֹלֶה. וְתִרְאוּ מַעֲשֵׂה פֶלֶא.
וַיָּבִיאוּ הַחוֹלֶה אֵלָיו וְהֵסֵב מְקוֹם הַמַּכָּה עַל פִּי
הַמְּעָרָה. וְצָעַק עַל הַנָּחָשׁ בְּקוֹל גְּעָרָה. וְהַנָּחָשׁ הָיָה
עָרוֹם וְגַעֲרָתוֹ הֵבִין. תַּחַת גְּעָרָה בְּמֵבִין. וְהוֹצִיא רֹאשׁוֹ
מִן הַחוֹר. וְהוּא כְמוֹ פֶחָם שָׁחוֹר. וְשָׂם פִּיהוּ עַל מְקוֹם
הַנְּשֵׁיכַה. וַיָּמָץ כָּל הַסַּם וְכָל הַחֵמָה הַשְּׁפוּכָה. וּמִיָּד
עָלְתָה לַחוֹלֶה אֲרוּכָה. וְהַנָּחָשׁ נָפַל מֵת בַּמַּעֲרָכָה.

אָמַר הַמַּגִּיד: וּבִרְאוֹתֵנוּ נוֹרְאוֹת פְּעֻלוֹתָיו
וּמַעֲשֵׂה פְּלָאָיו. אָמַרְנוּ: בָּרוּךְ הַנּוֹתֵן מֵחָכְמָתוֹ
לִירֵאָיו. וְנָתַן לוֹ כָל אִישׁ מֵהוֹנוֹ. כְּפִי אֲשֶׁר תַּשִּׂיג
יְמִינוֹ. אַחֲרֵי כֵן שְׁאִלְנוּהוּ עַל מוֹשְׁבוֹתָיו. וְתוֹלְדוֹתָיו
לְמִשְׁפְּחוֹתָיו.

וַיַּעַן וַיֹּאמַר:

בְּעֵת מַאְמָר אֲנִי בַעַל דְּבָרִים

וּבִקְרֹב לָקְּרָב אֶחְלשׁ גְּבָרִים.

אֲנִי חֶבֶר וְחָבֵר לַחֲבֵרִים

וְלִנְחָשִׁים אֲנִי חוֹבֵר חֲבָרִים.

אָמַר הַמַּגִּיד: וּכְשָׁמְעִי שִׁירָתוֹ. הִכַּרְתִּי
אוֹתוֹ. וְהוֹדַעְתִּי לְכָל בְּנֵי הַחֲבוּרָה מַעֲלָתוֹ. וְאֶת יְקָר
תִּפְאֶרֶת גְּדֻלָּתוֹ. וְשָׂמַחְנוּ כָל הַדֶּרֶךְ בְּחֶבְרַתוֹ. אַחֲרֵי
כֵן נִפְרַדְנוּ וְהָלַךְ כָּל אִישׁ לִנְתִיבָתוֹ. וְגַם הוּא שָׂם
מְגַמָּתוֹ לְאַדְמָתוֹ.
