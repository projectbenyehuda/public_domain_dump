


	
	ספר תחכמוני - שַׁעַר עֶשְׂרִים ושִׁבְעָה
	



בְּשֶׁבַח הַיַיִן וְתוֹעֲלוֹתָיו. וְהַמִּקְרִים אֲשֶׁר יִקְרוּ לְשׁוֹתָיו






נְאֻם הֵימן הָאֶזרָחִי: רָצִיתִי בִימֵי הַנְּעוּרִים.
לְהִבָּדֵל מִדַּרְכֵי הַסּוֹבְאִים וְהַשִּׁכּוֹרִים. וְנִכְסַפְתִּי
לָלֶכֶת בְּדַרְכֵי הַנְּזִירִים. וְאָסַרְתִּי עַל נַפְשִׁי כָּל מַחְמַד
עַיִן. וְכָל אֲשֶׁר יֵצֵא מִגֶּפֶן הַיַּיִן. וְקִבַּלְתִּי עָלַי
הַנְּזִירוּת בְּתַאֲוָה שְׁלֵמָה. עַד מְלֹאת לִי שָׁנָה תְמִימָה.
וְכַאֲשֶׁר קִיַּמְתִּי נִדְרִי. וְהִשְׁלַמְתִּי יְמֵי נִזְרִי.
נִכְסַפְתִּי לְבֵית הַכְּרָמִים. כִּי אָרְכוּ לִי שָׁם הַיָּמִים.
וְאָמַרְתִּי: לְמָתַי אֶמְנָע נַפְשִׁי מִכָּל שִׂמְחָה. עַד יִכְלוּ
בְיָגוֹן חַיַּי וּשְׁנוֹתַי בַּאֲנָחָה. אָשׁוּבָה לְקַדְמָתִי כִּי יַד
גִּילִי חַתָּה. וְכִי טוֹב לִי אָז מֵעַתָּה. אָז מִהַרְתִּי לִפְשׂט
בִּגְדֵי הַנְּזִירוּת. וְלָבַשְׁתִּי מַחְלְצוֹת הַשִּׁכְּרוּת.
וּמִהַרְתִּי וְלֹא הִתְאַפַּקְתִּי. וְעַל דַּלְתֵי הַסּוֹבְאִים
דָּפַקְתִּי. וְרָאִיתִי הַשִּׂמְחָה רְאוֹת עַיִן בְּעַיִן. וּבָאתִי
אֶל בֵּית מִשְׁתֵּה הַיַּיִן. וְהִנֵּה שָׁם חֶבְרַת בַּחוּרִים.
וְהַגְּבִיעִים כִּמְאוֹרִים. בִּגְלִילֵי יְדֵיהֶם מְאִירִים. וְכָל
אֶחָד מֵהֶם מִתְרוֹנֵן עַל גְבִיעוֹ וְאַגָּנוֹ. וּמְשַׁבֵּחַ יֵינוֹ.
זֶה אוֹמֵר: תֵּבֵל כְּמוֹ חֲשׁוּקָה יְפַת עַיִן. וְרֹק פִּיהָ הַיַּיִן
וְהַגַּנִים לְחָיֶיהָ וּפָנֶיהָ. וְהַמּעְיָּנִים עֵינֶיהָ. וְזֶה אוֹמֵר:
הַיַּיִן שְׁליחַ הַשְּׂמָחוֹת. וּמַבְרִיחַ הָאֲנָחוֹת. וְזֶה אוֹמֵר:
לוּלֵי בַת הַגְּפָנִים. כִּסָה עַל הַלֵּב עֲשַׁן הַיְגוֹנִים. וּפָנוּ
אֵלֵינוּ הַשְּׂשׂוֹנִים. עֹרֶף וְלֹא פָנִים. וְזֶה אוֹמֵר: הַיָגוֹן
כְּצָרַעַת פּוֹרַחַת. וְהַיַּיִן רְפוּאָה לוֹ וּמִרְקַחַת. לְהַצִּיל
מֵרָדֶת שחַת. וְזָה אוֹמֵר: הַיַּיִן רְפוּאַת אֲנוּשִׁים. וּגְבוּרַת
חֲלוּשִׁים. וּמְשַׂמֵּחַ אֱלֹהִים וַאֲנָשִׁים. וְהָיָה בְתוֹכָם אִישׁ
בַּעַל שֵׂיבָה. עוֹמֵד עַל מַצֵּבָה. וְתָמִיד יִתֵּן בַּכּוֹס עֵינוֹ.
וַיִמָּצֵא הַגָּבִיעַ סוֹבֵב עַל יַד יְמִינוֹ. וּכְשָׁמְעוֹ דִבְרֵי
חֲבֵרָיו וּמִלִּיהֶם. צָחַק וְלָעַג עֲלֵיהֶם. וְאָמַר: אוֹי לַמָּלִיצוֹת
אֲשֶׁר שֻׁפְּכו פְנִינֵיהֶם. וְחֻלְּלוּ עִנְיְנֵיהֶם. הֲכָזֶה יִהְיֶה
שֶׁבַח הַיַּיִן וּמַהֲלָלָיו. וְאִם אֵלֶה מַעֲלָלָיו. הָלֹא אִם
שְׁכַחְתֶּם מָתְקוֹ וְטַעְמוֹ. שְׁכַחְתֶּם יָפְיוֹ וְנָעֳמוֹ.
וְנִרְדּוֹ וּבָשְׂמוֹ. וְזִכְרוֹ וּשְׁמוֹ. וְלֹא זְכַרְתֶּם לֵחוֹ.
וְרִקְחוֹ וְרֵיחוֹ. וְצִירוֹ וְצָרְיוֹ. וּמִזְגוֹ וְחֶלְיוֹ. וְהוֹדוֹ
וּמַרְאֵהוּ. וּמַעֲשֵׂהוּ וּמִכְסֵהוּ. וְצוּפוֹ וְדִבְשׁוֹ. וְיוֹרוֹ
וּמַלְקוֹשׁוֹ. וְנַחֲלֵי רְבִיבוֹ. וְגַחֲלֵי שְׁבִיבוֹ. כָּל זֶה
שְׁכַחְתֶּם. וְדֶרֶךְ אַחֶרֶת לְקַחְתֶּם. וַיֹאמְרוּ לוֹ כָל הָעָם:
אִם תָּגִיד לָנוּ שֶׁבַח חֶמְדוֹתָיו. שֶׁיוֹרֵד עַל פִּי מִדּוֹתָיו.
יִהְיֶה לְךָ חֵלֶק בְּהוֹנֵנוּ. וּבַאֲשֶׁר תִּשְׁלַט יְמִינֵנוּ.



וַיַּעַן וַיֹּאמַר:

דְעוּ כִּי הַיַּיִן שִׂמְחַת נְשָׁמוֹת חֲלוּשׁוֹת.
וְחֶדְוָת נְפָשׁוֹת אֲנוּשׁוֹת. וּמַחְמַד לְבָבוֹת וּבָבוֹת. וְרֵיחַ
גַּן אֱלֹהִים רֵיחוֹ. וְרֶקַח הָרוֹכְלִים רִקָחוֹ. וּפְרִי עֵדֶן
מָתְקוֹ. וּצְרִי גִלְעָד רֻקּוֹ. וְהַשִּׂמְחָה נָפְלָה בְחֶלְקוֹ.
וְהַחֶדְוָה אֵשֶׁת חֵיקוֹ. פְנֵי לְהָבִים פָּנָיו. וּבְרַק חֶרֶב
מִתְהַפֶּכֶת בֵּין עֵינָיו. בְּיָד רָמָה יְנִיפְנָּה. וְעַל צְבָאוֹת
הַיְגוֹנִים יִשְׁלְפֶנָּה. לְכַבּוֹת בְּאִשּׁוֹ הָאוּדִים הָעֲשֵׁנִים.
וּלְהָעִיר בְּטַלּוֹ יְשֵׁנֵי הַשְּׂשׂוֹנִים. גְּבִיעָיו כִּכְפוֹרֵי
שֶׁלֶג לְבָנִים. וַעֲסִיסָיו אָדְמוּ עֶצֶם מִפְּנִינִים. לְשַׂמֵּחַ
נִקְשִׁים. וּלְהָשִׁיב רָשִׁים כְּרָאשִׁים. וּלְרַכֵּן קָשִׁים.
מַרְאֵהוּ תֹפֶת. וְטַעְמוֹ נֹפֶת. וְאֶל עֵדֶן צְלָלָיו. וְטַל אוֹרוֹת
טְלָלָיו. וּכְלֵי ה' כֵּלָיו. בְּהַר הַיֳּפִי הֵרִים רַגְלָיו.
וּבְחַדְרֵי הַלְּבָבוֹת תָּקַע אֹהָלָיו. וְעַל הַיְגוֹנִים יַצְבִּיא
חֵילָיו. הִבְעִיר גֶּחָלָיו בִּנְחָלָיו. וְהִסִּיק שְׁבִיבָיו. תּוֹךְ
רְבִיבָיו. וְהִתִּיךְ גַּחֶלֶת קוֹדַחַת. כְּמוֹ בְּבָרָד אֵשׁ
מִתְלַקַּחַת. וְהֶעֱלָה לַהַב. בְּמֵי זָהָב. וְהַטְבִּיע בִּבְדֹלַח
בָּרֶקֶת. וְאֹדֶם פִּטְדָה נִסֶּקֶת. זְרָמָיו מִצּוּף מְתוּקִים
וּבְמַעֲבֵה הַלֵּב יְצוּקִים. וְרֵיחוֹ שׁוֹשַׁנַּת הָעֲמָקִים. רָפֶה
וְיַפִּיל חֲזָקִים. וְחַלָּשׁ וְיָמִית עֲנָקִים. לֶחְיוֹ לְחִי חָשׁוּק.
וּפִיהוּ בְכָל פֶּה נָשׁוּק. וְאוֹרוֹ בָאֲפֵלָה זוֹרֶם. וְעֵינוֹ
עֵינֵי עִוְרִים פּוֹקֵחַ.





וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בַּת־הַגְּפָנִים יָקְרוּ דָמֶיהָ

מְדַם כְּבֵדִי יֶחְמְרוּ מֵימֶיהָ.

חַיִּים בְּטוּב טַעְמָה וְאוֹר שֶׁמֶשׁ בְּזִיו

לֶחְיָהּ וְהַמּוֹר מֶרְקָחוֹת סַמֶּיהָ.

תִּתְכַּס בְּכוֹס שׂהַם וְאָז תִּתְגֵּל וְאֵין

דָּבָר יְכַסֶנָּה וְיַעֲלִימֶהָ.

אָדְמוּ לְחָיֶיהָ כְּאִלּוּ זִיו פְּנֵי

רוֹאִים יְבִישֶׁנָּה וְיַכְלִימֶהָ.

הַכּוֹס כְּלִי שֹׁהַם וְיֵינָהּ יַהֲפֹךְ

אוֹתָהּ לְעֵין אֹדֶם וְיַאְדִּימֶהָ.

תּוֹךְ הַכְּלִי תִדְמֶה לְמֵתָה נִטְמְנָה

רַק תִּחְיֶה עֵת יִשְׁפְּכוּ דָמֶיהָ.

אַחַר בְּלוֹת תָּשׁוּב לְעֶדְנָתָהּ וְאִם

תַּזְקִין בְּכִלְחָהּ חִדְּשָׁה יָמֶיהָ.

אָמְנָם כְּשֶׁמֶשׁ הִיא בְגַלְגַּל הַיֳּפִי

רַק הַגְבִיעִים נֶחְשְׁבוּ שָׁמֶיהָ.





וּבְהַשְׁלִימוֹ שִׁירָיו. יָשְׁרוּ בְעֵינֵי כָל הָעָם
דְּבָרָיו. וַיֹּאמְרוּ לוֹ: הֵטַבְתָּ בְשִׁירְךָ. וְנָעַמְתָּ
בִדְבָרֶךָ. וְשִׁבַּחְתָּ הַיַּיִן בְּכָל מִדּוֹתָיו. וְגִלִּיתָ
חֲמוּדוֹתָיו. אֲבָל אִם תְּסַפֵּר רֹעַ מִדּוֹתָיו. כַּאֲשֶׁר סִפַּרְתָּ
חֲמוּדוֹתָיו. תּוֹדִיעַ מַעֲלָתְךָ וְתִפְרשׁ. וְהָיִיתָ לָנוּ
לְרֹאשׁ.

וַיַּעַן וַיֹּאמֶר לָהֶם: הוֹי גִּבּוֹרִים לִשְׁתּוֹת
יַיִן. הֵן אַתֶּם מֵאַיִן. הֲלֹא יְדַעְתֶּם כִּי אֵין כָּמוֹהוּ.
וְהַמִּתְהַלֵּל בּוֹ בּוֹטֵחַ עַל תֹּהוּ. וּבַת הַגְּפָנִים כַּמָּה
אֲנָשִׁים הִכְשִׁילָה. כִּי רַבִּים חֲלָלִים הִפִּילָה. וּבְעֵת אִישׁ
אוֹתָהּ יִשְׁתֶּה. אַךְ אֵשׁ בְּתוֹךְ לִבּוֹ תַחְתֶּה. וּתְבַלְבֵּל כָּל
נַפְשׁוֹ. וְתַכְאִיב רֹאשׁוֹ. וְתוֹצִיאֵהוּ מִדַּעְתּוֹ וּתְבוּנָתוֹ.
וְתָבוֹא אֵלָיו בַּלָּאט וְתִתְקַע הַיָּתֵד בְּרַקָּתוֹ. וְהַיַּיִן
יְסוֹד כָּל עָוֹן וְאַשְׁמָה. וְכָל עֲבֵרָה וְזִמָּה. וְהוּא עִקָּר
לְכָל רֶשַׁע. וּמְחֻבָּר לְכָל פֶּשַׁע. וּבְּבֵית מִשְׁתֵּה הַיַּיִן
יִוָּעֲדוּ הָרְשָׁעִים וְהֶעָרִיצִים. וְהָאַכְזָרִים וְהַפָּרִיצִים.
וְלַעֲשׂוֹת זִמָּה יִתְחַבָּרוּ. וְעַל דָּגָן וְתִירוֹשׁ יִתְגּוֹרָרוּ.
וְשָׁם יִצָּמְדוּ הַנּוֹאֲפִים עִם הַקְּדֵשִׁים. וְיָבוֹאוּ הָאֲנָשִׁים
עַל הַנָּשִׁים. וְיִתְחַבְּרוּ הַמְנַגְּנִים וְהַמְרַנְּנִים. וּמִן
הַנִּגּוּנִים. יִמָּשֵׁךְ אָדָם לְתַזְנוּנִים. בְּאִישׁוֹן לַיְלָה
וּבָאֲפֵלָה. וְקָרַב זֶה אֶל זֶה כָּל הַלַּיְלָה. לְהַרְבּוֹת נְבָלוֹת.
וּלְהִתְעוֹלֵל עֳלִילוֹת. וְהוּא מֵסִיר הַבּשֶׁת וְהָעֲנָוָה. וְכָל
מִדָּה נָאוָה. וּמְאַבֵּד הַדַּעַת. וּמֵשִׁיב נֶפֶשׁ הַמַּשְׂכִּיל
מִשְתַּגַּעַת. לְהָקֵל הַמִּצְוֹת הַחֲמוּרוֹת. וְלִרְדֹּף הַתַּאֲווֹת
וְהָעֲבֵרוֹת. וְהַיַּיִן יַחְבִּיר בְּבַיִת אֶחָד אֲצִילִים. עִם
נְבָלִים. וּגְאוֹנִים עִם שְׁפָלִים. וְהַנַּעֲלֶה. עִם הַנִּקְלֶה.
וְהַזָּנָב עִם הָרֹאשׁ. וְהַצּוֹאָה עִם בְּשָׂמִים רֹאשׁ. וְכָל
הַשּׁוֹגֶה בוֹ יְאַבֵּד כָּל הוֹנוֹ. וְיוֹצִיא כָּל מָמוֹנוֹ.
וְיִתְבַּטֵּל מִן הָאֻמָּנוּת אֲשֶׁר אִתּוֹ. כִּי בוֹ שָׁבַת מִכָּל
מְלַאכְתּוֹ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בַּת הַגְּפָנִים יֶחְשְׁכוּ שָׁמֶיהָ

אֶל יוֹדְעֵי מוּמָהּ וְתַעְלוּמֶיהָ.

זַכָּה לְעֵין הַבּוֹעֲרִים אָכֵן לְעֵין

חַכְמֵי לְבָבוֹת יֶחְמְרוּ מֵימֶיהָ.

תָּזוּב בְּדָם לַבּוֹעֲלִים אוֹתָהּ וְלֹא

תִּטְהַר לְעוֹלָם מִמְּקוֹר דָּמֶיהָ.

טוֹעֵם מְרוֹרֶיהָ כְּרוֹשׁ יִמְצָא לְחִ-

כּוֹ צוּף דְּבַשׁ פִּיהָ וּמַטְעַמֶּיהָ.

מַבִּיט בְּעֵין שִׂכְלוֹ לְחֶלְאָתָהּ הֲלֹא

יִמְאַס בְּטוּב רִקְחָהּ וּבִבְשָׂמֶיהָ.

מִיּוֹם הֱיוֹתָהּ בַּעֲבֵרוֹת נוֹלְדָה

גַּם בַּעֲבֵרוֹת יִמְלְאוּ יָמֶיהָ.

כָּל-הַחֲטָאִים מִבְּנֵי אִישׁ נִכְרְתוּ

לוּ נִכְרְתָה מִתּוֹךְ בְּנֵי עַמֶּיהָ.

אָמַר הַמַּגִּיד: וְכִשְׁמֹעַ הָעָם פְּלִיאוֹת אֲמָרָיו.
נִמְשָׁךְ לִבָּם לִדְבָרָיו. וְרַבִּים מֵהֶם עָזְבוּ הַיַּיִן
וַיִּמְאָסוּהוּ. וְנָדְרוּ נֶדֶר כָּל יְמוֹתָם לְבַל יִשְׁתּוּהוּ.
וָאֹמַר אֵלָיו: בְּחֵי הַמַּזְרִיחַ כּוֹכְבֵי מְרוֹמֶיךָ. הוֹדִיעֵנִי
מַה שְּׁמֶךָ. אָמַר: אֵלוֹן בְּצַעֲנַנִּים מְעוֹנִי. וְהַמַּשְׂכִּיל
יֵדָעֵנִי. וּכְשָׁמְעִי נִיבוֹ. הֶחֱזַקְתִּי בוֹ. וְאָמַרְתִּי לוֹ:
הַאַתָּה גְבִירֵנוּ. חֶבֶר הַקֵּינִי חֲבֵרֵנוּ. אָמַר: אֲנִי אִישׁ
שְׁלוֹמְךָ וֶאֱמוּנֶיךָ. וּמִבְחַר נֶאֱמָנֶיךָ. וְעָמַדְתִּי כְּפִי
שָׁעָה אִתּוֹ. וְאַחֲרֵי כֵן הָלַךְ לִמְעוֹן שִׁבְתּוֹ. וְעָזַב
בְּלִבִּי אֵשׁ כִּידוֹדוֹ. וְנַפְשִׁי עֲשׁוּקָה בְיָדוֹ.
