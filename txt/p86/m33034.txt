


	
	ספר תחכמוני - שַׁעַר שְׁלשִׁים וְתִשְׁעָה
	



וִכּוּחַ הַלַּיְלָה וְהַיּוֹם. מִי הוּא יוֹתֵר נוֹרָא וְאָיֹם






נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי בְרִמּוֹן פָּרֶץ. עִם
אֲצִילִים גּוֹדְרֵי גָדֵר וְעוֹמְדִים בַּפָּרֶץ. אֲדוֹנֵי קְהִלּוֹת.
וְאַדְנֵי תְהִלּוֹת. יְרִיקוּן מִלְּשׁוֹנָם חֶרֶב פִּיפִיּוֹת.
וִיפוֹרְרוּ בְשִׂכְלָם סַלְעֵי תוּשִׁיּוֹת. וַיַּעַן אִישׁ מֵהֶם
וַיֹּאמַר: אֲסַפְּרָה לָכֶם מִשְׁלֵי קַדְמוֹנִים וְחִידוֹתָם. אֲשֶׁר
חֲכָמִים יַגִּידוּ וְלֹא כִחֲדוּ מֵאֲבוֹתָם. אָמְרוּ לוֹ: זְכֹר
מְלִיצָתְךָ וְנֵדָעֶנָּה. וְסַפֵּר לָנוּ אִמְרָתְךָ וְנִשְׁמָעֶנָּה.
וַיַּעַן וַיֹּאמַר: דְּעוּ כִּי בְעֵת חָנְתָה הַשֶּׁמֶשׁ בְּמַזַּל
טָלֶה. וַיִּתְחַזֵּק אוֹרָהּ וַיִּגָּלֶה. וְהֵאִירוּ חֲשֵׁכִים.
וְהֵחֵלּוּ לָצֵאת הַמְּלָכִים. וְהַלֵּילוֹת יָרְדוּ לְמִצְעָר וּמְקוֹמָם
הִשְפִּילוּ. וְהַיָּמִים הִגְדִּילוּ. וְלַעֲלוֹת הֶעְפִּילוּ. לָבַשׁ
אוֹר הַיּוֹם גַּאֳוָה גְדוֹלָה. וַיְדַבֵּר עָתָק עַל הַלַּיְלָה.
וַיַּעַן וַיֹּאמַר: אָנֹכִי נְגִיד הַנְּגִידִים. וְהַלֵּילוֹת לִי
לַעֲבָדִים. וְהָאֵל הִפְקִידַנִי עֲלֵיהֶם לִהְיוֹת קְצִינָם וּפֶחָם.
כִּי אֲנִי לָבָן וְהֵם שְׁחוֹרִים כְּפֶחָם. וַאֲנִי מִבְּנֵי שֵׁם וְהֵם
מִבְּנֵי חָם. וְעֶל כֵּן הֵם עֲבָדַי כָּל יְמֵי נִצְחָם. בִּרְצוֹנָם אוֹ
עַל כָּרְחָם. וְכִשְׁמֹעַ הַלַּיְלָה דִבְרֵי הַיּוֹם. אָמַר הַלַּיְלָה:
הָהּ לַיּוֹם. הַחוֹשֵׁב בְּנַפְשׁוֹ כִּי הוּא נוֹרָא וְאָיֹם. וְהוּא
נִמְכַּר בְּלִי פִדְיוֹם. וּבַמֶּה יִתְגָּאֶה עָלַי כִּי אִם
בְּשִׁמְשׁוֹ הַשּׁוֹזֶפֶת. וְאִשּׁוֹ הַשּׂוֹרֶפֶת. וְלֹא יֵדַע כִּי
הַיּוֹם לְעָמָל יוּלָּד. וְלִיגִיעָה הוּא מְעֻתָּד. כִּי בַיּוֹם אֵין
לְאִיׁש מְנוּחָה. וּבַלַּיְלָה תִּהְיֶה הָרְוָחָה. בּוֹ יָנוּחוּ
הָעֲמֵלִים. וְיִשְׁכְּחוּ רִישָׁם הַדַּלִּים. וּבֹו הָעֲיֵפִים יִמְצְאוּ
מָנוֹחַ. וְשָׁם יָנוּחוּ יְגִיעֵי כֹחַ. וּבוֹ יְנַשֶּׁה כָּל דּוֹאֵג
יְגוֹנָיו. וְעֶבֶד חָפְשִׁי מֵאֲדוֹנָיו. וְלוּלֵא הַלַּיְלָה בְּנֵי
אִישׁ מְנוּחָה לֹא מָצָאוּ. וּמֵעַבְדוּת לַחֹפֶשׁ לֹא יָצָאוּ. וְאִם
בְּקַדְרוּתִי חֵרַפְתַּנִי כַּמָּה רָאוּ עֵינֶיךָ דָּבָר לָבָן וְהוּא
נִרְמָס וְנִמְאָס. וְכַמָּה שָׁחוֹר יָקָר וְנִכְמָס. הִנֵּה לֹבֶן
הָעַיִן. נֶחְשָׁב לְאַיִן. וְשַׁחֲרוּתוֹ הוּא הַנִּכְבָּד וְהַנָּאוֹר.
וְהַדֶּרֶךְ יִשְׁכֹּן אוֹר.

וַיַּעַן הַיּוֹם וַיֹּאמַר: צְחוֹק עָשָׂה לִי אֱלֹהִים
בִּרְאוֹתִי הַגְּבוֹהִים נָפָלוּ. וְהַשְּׁפָלִים לַשַּׁחַק עָלוּ. כָּל
גֶּיא יִנָּשֵׂא וְכָל הַר וְגִבְעָה יִשְׁפָּלוּ. וּבַמֶּה יִתְפָּאֵר
הַלַּיְלָה וְאֵין בּוֹ דָבָר חָמוּד. הִנֵּה הַלַּיְלָה הַהוּא יְהִי
גַלְמוּד. כִּי בוֹ תֶּחֱשַׁכְנָה עֵינֵי הָרוֹאִים. וְיֵלְכוּ נְבוּכִים
כָּל הַבְּרוּאִים. וּבַעֲלוֹת הַשַּׁחַר עֵינֵיהֶם יָאִירוּ. וּכְזֹהַר
הָרָקִיעַ יַזְהִירוּ. כִּי בְאוֹר הַיּוֹם יַצְלִיחַ אָדָם בְּכָל
דְרָכָיו. וְיוּכַל לַעֲשׂוֹת כָּל צְרָכָיו. לִמְכֹּר אוֹ לִקְנוֹת.
וְלַהֲרֹס אוֹ לִבְנוֹת. וּבַלַּיְלָה לֹא יוּכַל לַעֲשׂוֹת כִּי אִם
לִגְנֹב אוֹ לִזְנוֹת. וְלַחְתֹּר הַקִּירוֹת וְלָבוֹא בְּעַד
הַחַלּוֹנוֹת.

וַיַּעַן הַלַּיְלָה וַיֹּאמַר: הַלֹא יָדַעְתָּ. אִם לֹא
שָׁמַעְתָּ. כִּי אֲנִי יְסוֹד הַחֶבְרָה. וְהָאַהֲבָה הַגְּמוּרָה. כִּי
בִי יִתְחַבְּרוּ הַחוֹשְׁקִים וְהַיְדִידִים. וּבִי יִרְווּ דוֹדִים.
וּבְךָ יִשְׂבְּעוּ נְדוּדִים. וַאֲנִי אֲחַבֵּר הַחוֹשְׁקִים וְאַתָּה
תַפְרִידֵם. וְאַפְרִיד הַיְגוֹנִים וְאַתָּה תַצְמִידֵם. וְהַיְדִידִים
בִּי יִתְחַבָּקוּ. אִישׁ בְּאָחִיהוּ יְדֻבָּקוּ. וּבְךָ זֶה מִזֶּה
יִרְחָקוּ. וּבִרְאוֹתָם אוֹר שְׁחָרֶיךָ. יֹאמְרוּ קוּמוּ וְנִבְרַח
כֻּלָּנוּ. כִּי בָא הָאִישׁ אֲשֶׁר כִּלָּנוּ. וַאֲשֶׁר דִּמָּה
לָנוּ.

וַיַּעַן הַיּוֹם וַיֹּאמַר: יֹאבַד יוֹם אִוָּלֶד בּוֹ.
וְאַל יִגַּהּ שְׁבִיבוֹ. אַחֲרֵי אֲשֶׁר הֲקִלּוּנִי רֵקִים וּנְעוּרִים.
וְצָחֲקוּ עָלַי צְעִירִים. וְעָלְתָה עָשׁ וְרִמָּה. עַל עָשׁ וְכִימָה.
וְגָאוּ תְהוֹמוֹת. עַל הַמְּקוֹמוֹת. הֲלֹא יָדַעְתָּ כִּי בִי חַיֵּי
הָעוֹלָם. וְשִׁמְשִׁי גְּבֶרֶת הַבְּרוּאִים כֻּלָּם. וּבְאוֹרִי
יִפְרְחוּ הַפְּרָחִים. וְיִצְמְחוּ הַצְּמָחִים. מִמֶּגֶד תְּבוּאֹות
שֶׁמֶשׁ וּמִמֶּגֶד גֶּרֶשׁ יְרָחִים. וְאַתָּה אֵין בְּךָ כֹּחַ
לְהַצִּיל. לֹא לְעֵזֶר וְלֹא לְהוֹעִיל.

אָמַר הַלַּיְלָה: הֲתִתְגָּאֶה עָלַי בְּשִׁמְשְׁךָ
וְהִיא נְקֵבָה. וּמִמַּקֶּבֶת הַנָּשִים חֲצוּבָה. הֲלֹא רָעוֹתֶיךָ
רַבּוֹת. כִּי לְמִלְחֶמֶת הַזְּכָרִים תּוֹצִיא הַנְּקֵבוֹת. וְאֵיךְ
תַּעֲצֹר הַשֶּׁמֶשׁ כֹּחַ. לִפְנֵי הַיָּרֵחַ. מַלְכִּי. וּמַגִּיהַּ
חָשְׁכִּי. הֲלֹא בְּעֵת יָאִיר סַהֲרִי תִסָּתֵר בָּשְׁתֶּךָ. וְתַחְבִּיא
בַחֲדַר הַנָּשִׁים גְּבִרְתֶּךָ. וְשִׁמְשְׁךָ יְחִידָה. בְּלִי קָהָל
וְעֵדָה. וְהַיָּרֵחַ גְּדוּדָיו רַבִּים וְהוּא בְתוֹכָם מִתְהַלֵּךְ.
כִּי בְרָב עָם הַדְרַת מֶלֶךְ. וּבַעֲלוֹת סַהֲרִי שִׁמְשְׁךָ נוֹטָה
וְאוֹרַחַת. כְּסַפַּחַת בּוֹרַחַת. וְתִקְרַב לַשַּׁחַת. וְתִדְמֶה
לַיּוֹנָה תְקַנֵּן בְּעֶבְרֵי פִי פַחַת. וְתִתְגַּנֵּב לִבְרֹחַ
בְּאֵימָה. כַּאֲשֶׁר יִתְגַּנֵּב הָעָם הַנִּכְלָמִים בְּנוּסָם
בַּמִּלְחָמָה.

אָמַר הַיּוֹם: רַב לְךָ רוֹעֶה רוּחַ. כְּסִיר נָפוּחַ.
וּמַעֲלַת הַיָּרֵחַ. מִזִּיו הַשֶּׁמֶשׁ הוּא גּוֹנֵב וְלוֹקֵחַ. וְלוּלֵי
הַשֶּׁמֶשׁ הוּעַם אוֹרוֹ וְשָׁנָה. וְהָיָה עַל פְּנֵי שַׁחַק כְּבַהֶרֶת
לְבָנָה. וְאִם בְּכָל הַלַּיְלָה יִגְאֶה הַיָּרֵחַ בִּגְדוּדָיו
וְחֵילוֹ. בֹּקֶר וְיוֹדַע ה' אֶת אֲשֶׁר לוֹ. כִּי בַעֲלוֹת אוֹרִי
יִסָּתֵר הַיָּרֵחַ הוּא וְכָל חֵילוֹ. כְּגַנָּב בַּמַּחְתֶּרֶת אִם
זָרְחָה הַשֶּמֶשׁ עָלָיו דָּמִים לוֹ. וְהַסַּהַר תְּכַסֵּהוּ בוּשָׁה
בְּעֵת שִׁמְשִׁי תֵצֵא. כְּבשֶׁת גַּנָּב כִּי יִמָּצֵא.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר הִרְבּוּ לְדַבֵּר דְּבָרִים
הִכִּיר הַלַּיְלָה חֶסְרוֹנוֹ. וְנִפְקַח עֵינוֹ. וַיֵּדַע מַעֲלַת
הַיּוֹם וְיִתְרוֹנוֹ. אָמַר: הִנֵּה יָדַעְתִּי כִּי מַעֲלָתְךָ נְסוּכָה
וְאַתָּה מֶלֶךְ וַאִני מִשְנֶה. וְכָל גְּדוּדַי לְךָ לְמַחֲנֶה. וְעַתָּה
לְכָה וְנָשׁוּבָה לִנְוֵה הַחֶבְרָה הַבְּרוּכָה. וּנְחַדֵּשׁ שָׁם
הַמְּלוּכָה.

אָמַר הַמַּגִּיד: וּכְכַלּוֹת הֶחָכָם דְּבָרָיו.
נִמְשַׁךְ לִבִּי לְנֹעַם אֲמָרָיו. וְכַאֲשֶׁר רָאַנִי מִתְבּוֹנֵן
אֵלָיו. וְתָמֵהַּ לְמִלָּיו.

וַיַּעַן וַיֹּאמַר:

אֲנִי חָבֵר לְכָל מִי יִדְרְשֵׁנִי

אֲנִי יַם-בִּין לְכָל מִי יַחְקְרֵנִי.

אֲשׁוֹטֵט בֵּין בְּנֵי אִישִׁים בְּשִׂכְלִי

וְאֶתְהַפֵּךְ לְכָל-צַד עִם זְמַנִּי.
