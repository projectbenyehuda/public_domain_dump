


	
	ספר תחכמוני - שַׁעַר תִּשְׁעָה עָשָׂר
	



מִשְׁבְעָה בַחוּרֵי חֶמֶד טוֹבָה. מִתְוַכְּחִים אֵיזוֹ מִדָּה טוֹבָה






נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי בְאֶרֶץ פְּתוֹר. עיר
בִּלְעָם בְּנוֹ בְעוֹר. וּבְעוֹדִי עַל יַד נְהַר פְּרָת מִתְהַלֵךְ
תַּחַת צִלְלֵי הַצְּמָחִים. וְסִבְכֵי הַפְּרָחִים. רָאִיתִי שִׁבְעָה
בַחוּרֵי חֶמֶד. מִמִּבְחַר הַצֶּמֶד. וְהֵם עַל שְׂפַת הַנָּהָר
יוֹשְׁבִים. וּבְדִבְרֵי מְלִיצוֹת אֶת לִבָּם מֵטִיבִים. וַיַּעַן אִישׁ
מֵהֶם וַיֹּאמַר: אֵיזוֹ הִיא הַמִּדָּה הַטוֹבָה. אֲשֶׁר עַל כָּל
הַמִּדּוֹת אֲהוּבָה. וּבְעֵינֵי אֱלֹהִים וְאָדָם חֲשׁוּבָה. אָמַר אֶחָד
מֵהֶם: יָדַעְתִּי כִּי כָל הַמִּדּוֹת. הֵם חֲמוּדוֹת. אֲבָל אֵין
בְּכֻלָם מִדָּה נָאוָה. כְּמוֹ הָעֲנָוָה. כִּי הִיא תְּכֵסֶּה כָל דֹּפִי
. וּתְגַלֶה כָל יֹפִי. וְתַעֲבִיר עַל כָּל הַפְּשָׁעִים. וְתַחְבִּיר
בְּעָלֶיהָ לַצְּנוּעִים. וְתַרְבֶּה לוֹ דוֹדִים וְרֵעִים.
וְתַנְחִילֵהוּ שֵׁם יָקָר וְנָעִים.



וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בְּמִדּוֹת הָאֱנוֹשׁ אֵין כָּעֲנָוָה

וְהִיא יָפָה לְכָל מַשְׂכִּיל וְנָאוָה.

תְּעוֹרֵר אַהֲבָה מִלֵּב מְשַׂנְאִים

וְתַסְתִּיר מֵאֱנוֹשׁ כָּל חֵטְא וּמַעְוָה.



אָמַר לוֹ חֲבֵרוֹ: מִדֶּרֶךְ נְכוֹחָה תָּעִיתָ. וְרוּחַ
רָאִיתָ. וּכְאַיִן נֶחְשָׁבָה הָעֲנָוָה אוֹ הָעַזּוּת. לְנֶגֶד
הַזְרִיזוּת. כִּי בָהּ יִכְבּשׁ אָדָם הַנְפָשִׁים. וְיִמְצָא חֵן
בְּעֵינֵי אֱלֹהִים וַאֲנָשִׁים. וְיִנְחַל כָּבוֹד וּגְדֻלָּה רַבָּה.
בָּעוֹלָם הַזֶּה וּבַבָּא. וּבָהּ יִצְבֹּר חַיִל יָנוּב. וּבָתִּים
מְלֵאִים כָּל טוּב.



וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֱמֶת כִּי הַזְּרִיזוּת אֵין כְּעֶרְכָּה

וְאַשְׁרֵי אִישׁ אֲשֶׁר יֵלֵךְ בְּדַרְכָּהּ.

וּמִדּוֹת הַיְקָר כֻּלָן שְׁפָחוֹת

וְלָהֶן הַזְּרִיזוּת הִיא כְמַלְכָּה.



אָמַר חֲבֵרוֹ הַשְׁלִישִׁי; נִסְכַּלְתָּ. בַּאֲשֶׁר
מִלַּלְתָּ. כִּי אֵין מִדָּה טוֹבָה וִיקָרָה. כְּאֹמֶץ לֵב וּגְבוּרָה.
כִּי בָה יַכְנִיעַ אָדָם כָּל אוֹיְבָיו. וְיֵיטִיב לְאוֹהֲבָיו.
וְיִהְיֶה לִגְדוֹלִים נֶחְבָּר. וְיָרִיע אַף יַצְרִיחַ עַל אוֹיְבָיו
יִתְגַבָּר. וְיַעֲלֶה לְכִסֵּא הַגְּדֻלָּה. עֲדֵי יִקְרְאוּ לְפָנָיו.
סֹלּוּ סֹלּוּ הַמְסִלָּה.



וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֱמֶת אֵין לָאֱנוֹשׂ מִדָּה יְקָרָה

כְּאֹמֶץ לֵב מְעֹרָב עם גְבוּרָה.

הָכִי בָהּ הָאֱנוֹשׁ יִכְבּשׁ מְשַׂנְאָיו.

וְיוֹרִידֵם בְּיָגוֹן לַקְבוּרָה.





אָמַר הָרְבִיעִי: תָּעִיתָ מִן הַדֶּרֶךְ. וְנֶעֶבַדְתָּ
אֶל הַסִכְלוּת בְּפֶרֶךְ. כִּי אֵין בְּכָל הַמִּדּוֹת מִדָּה הָגוּנָה.
כְּמוֹ הָאֱמוּנָה. כִּי בָהּ יָרִים אָדָם רֹאשׁוֹ. וְתִיקַר נַפְשׁוֹ.
וְיִהְיֶה נִכְבָּד בְּעֵינֵי כָל בָּשָׂר וָדָם. וְיִמְצָא חֵן וְשֵׂכֶל
טוֹב בְּעֵינֵי אֱלֹהִים וְאָדָם.





וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

דְּעֶה כִּי אֵין לְךָ מִדָה הָגוּנָה

בְּעֵין הָאֵל כְּמִדַּת הָאֱמוּנָה.

וְאִלוּ לַנְּבוּאָה יִהְיֶה גוּף

תְּהִי אָז הָאֱמוּנָה לָהּ תְּמוּנָה.



אָמַר הַחֲמִישִׁי: אֵין בְּפִיךָ נְכוֹנָה. וְאִמְרָתְךָ
בְּלִי תְּבוּנָה. כִּי הַמִּדָּה הָרָמָה. הִיא הַחָכְמָה. תָּרִים
שְׁפָלֶיהָ. וְתִשָׂא עַל רֹאשָׁם רַגְלֶיהָ. וְהַחָכְמָה תְחַיֶּה
בְעָלֶיהָ. וְלוּלֵי הַחָכְמָה. לֹא הָיָה יִתְרוֹן לְאָדָם מִן
הַבְּהֵמָה.



וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֵין בַּזְמַן מִדָּה מְאֹד רָמָה

וּפְאֵר לְראשׁ אָדָם כְּמוֹ חָכְמָה.

כִּי בָהּ אֱנוֹשׁ יַעַל לְכֵס מַלְכוּת

גַם בָּהּ עֲיֵפִים יִמְצְאוּ עָצְמָה.



אָמַר הַשִּׁשִׁי: מַחֲזֶה שָׁוְא חָזִיתָ. וְשֶׁקֶר
עָנִיתָ. כִּי אֵין מִדָּה טוֹבָה לְכָל בָּשָׂר. כְּמוֹ הַמּוּסָר.
כִּי בַּמּוּסָר יְכַבְּדוּ לָאָדָם יוֹדְעָיו. וְיֶאֱהָבוּהוּ שׁוֹמְעָיו
. וִיכַסּוּ עַל כָּל פְּשָׁעָיו. וְזִכְרוֹ מָתוֹק לְכָל פִּיוֹת.
וּמַהֲלָלוֹ כְּמִגְדָּל בָּנוּי לְתַלְפִּיוֹת. וְהוּא מַחְמָד
לַלְּבָבוֹת. וּתְהִלוֹתָיו לְדוֹר דּוֹרִים נִצָבוֹת.





וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֵין לֶאֱנוֹשׁ מִדָּה כְּמוֹ מוּסָר

כִּי הִיא פְאֵר וָהוֹד לְכָל בָּשָׂר.

אִם כָּל־יְקָר יִהְיֶה לְאִישׁ מֵאֵין

מוּסָר. דְעֶה כִּי כָּל יְקָר יֶחְסָר.



אָמַר הַשְׁבִיעי: אַתָּה עַל רִיק בָּטוּחַ. רוֹדֵף
קָדִים וְרוֹעֶה רוּחַ. כִּי אֵין בְּכָל הַמִדּוֹת דָּבָר טוֹב. כְּמוֹ
הַלֵּב הַטּוֹב. כִּי בוֹ יִהְיֶה הָאָדָם אָהוּב לַבְּרוּאִים.
וְיַעֲלוּהוּ לְרֹאשׁ הַקְּרוּאִים. וְהוּא נִכְבָּד בְּעֵינֵי
הַשּׁוֹמְעִים וְהָרוֹאִים כָּל רוֹאָיו יֶאֱהָבוּהוּ. וְשׂוֹנְאָיו
יְשַׁבְּחוּהוּ. וּבְנֵי אָדָם יְהַלְלוּהוּ. וּמַלְאֲכֵי שַׁחַק לְטוֹבָה
יִזְכְּרוּהוּ.



וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֵין בֶּאֱמֶת מִדָּה כְמוֹ לֵב טוֹב

בָּהּ הָאֱנוֹשׁ יִפְרַח כְּגַן רָטֹב.

כִּי בוֹ יְהִי אָהוּב לְצוּרוֹ גַם.

יִמְצָא בְעֵינָיו חֵן וְשֵׂכֶל טוֹב.



וְכִשְׁמֹעַ הַזְקֵן דִּבְרֵיהֶם. אָמַר לָהֶם: כֻּלְכֶם
נְבוּכִים. וְהוֹלְכִי חֲשֵׁכִים. וְהַנְּכוֹחָה מִכֶּם נֶעְלֶמֶת.
וְלֹא יְדַעְתֶּם לבְחֹר הָאֱמֶת. כִּי כָל הַמִּדּוֹת לרַגְלֶיהָ
תִכְרַעֲנָה. וְהִיא עָלְתָה עַל כֻּלָּנָה. בָּהּ יִתְכַּפְּרוּ כָּל
הָאֲשָׁמוֹת. יָסוּרוּ מִן הַלֵּב הַמַּשְׂטְמוֹת. וּבָהּ יַשִׂיג אִישׁ
חֲפָנָיו הָרְחוֹקִים. וְאִלּוּ הָיוּ בַשְׁחָקִים. וּבָהּ יִמָּנֶה עִם
הַחֲסִידִים. כִּי בָהּ יִגְמֹל צְדָקוֹת וַחֲסָדִים. וּבָהּ יִקְנֶה שֵׁם
טוֹב. וְיִהְיֶה זִכְרוֹ כַשֶּׁמֶן הַטּוֹב. וּמִי שֶׁאֵין בּוֹ נְדָבָה.
תָּשׁוּב צִדְקָתוֹ חוֹבָה. וַחֲסָדָיו שְׁגִיאוֹת. וְטוֹבוֹתָיו
חֲטָאוֹת. וְכָל חֲבֵרָיו יִמְאָסוּהוּ. וְיוֹדְעָיו יִשְׂנָאוּהוּ.
וְרֵעָיו לָרַע יִזְכְּרוּהוּ. וְגָרֵי בֵיתוֹ וְאַמְהוֹתָיו לְזָר
יַחְשְׁבוּהוּ. וּבַעל הַנְדָבָה יִשָּׂא פָנָיו. כִּי הִיא תְכַסֶּה כָּל
עֲוֹנָיו. וְתִמְחֶה זְדוֹנָיו. וְאוֹיְבָיו יֶאֱהָבוּהוּ. וְשׂוֹנְאָיו
יְשַׁבְּחוּהוּ. וּמְקַנְאָיו יוֹדוּהוּ. וּמְקַלְליו יְבָרְכוּהוּ. כִּי
בְנִדְבָתוֹ יִכְבּשׁ לִבּוֹתָם. וְיִמְשֹׁךְ אַהֲבָתָם. וּבָהּ תָּשׁוּב
הַגַּאֲוָה. נָאוָה. וְהַחֶסְרוֹן. יִתְרוֹן. וּבָה |הַכְּסִיל.
נֶחְשָׁב לְמַשְׂכִּיל. וְהַנִּקְלֶה. עַל שְׁחָקִים יַעֲלֶה. וּבְכֵן
רָאִיתִי רְשָׁעִים. וּבַעֲלֵי מַעֲשִׂים רָעִים. וְאִם יֵשׁ בָּהֶם
נְדָבָה. תַּסְתִּיר מֵהֶם כָּל רֶשַׁע וְכָל חוֹבָה. וְהַמִּדָּה הָרָעָה
תָּשׁוּב טוֹבָה. וְעַל כָּל פְּשָׁעִים תְּכַסֶה אַהֲבָה. וְכֵן
רָאִיתִי אַנְשֵׁי אֱמוּנָה. וּבַעֲלֵי שֵׂכֶל וּתְבוּנָה. וּבָהֶם כָּל
מִדָּה הֲגוּנָה. וְאִם אֵין בָּם נְדָבָה תָּשׁוּב הַיִרְאָה. חַטָּאָה
. וְהָעֲנָוָה. גַאֲוָה. וְהַזְרִיזות. עַזּוּת. וְהַשֶׂכֶל סִכְלוּת.
וְהַיִּתְרוֹן מִגְרַעַת. וְהַתְּבוּנָה חֶסְרוֹן דַּעַת. כִּי לְרַגְלֵי
הַנְּדָבָה. כָּל הַמִּדּוֹת הַטּוֹבוֹת תִּכְרַעְנָה. וְעָלֶיהָ נֶאֱמַר:
רַבּוֹת בָּנוֹת עָשׂוּ חָיִל וְאַתְּ עָלִית עַל כֻּלָּנָה.



וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֱמֶת בִּבְנֵי זְמַן מִדּוֹת חֲמוּדוֹת

וְעַל כֻּלָן מְעֻלָּה הַנְּדָבָה.

וְכָל מִדָּה וְאִם תִּהְיֶה חֲמוּדָה

הֲלֹא מֵהַנְדָבָה הִיא גְנוּבָה.



אָמַר הַמַּגִיד: וּכְשָׁמְעִי שִׁיחוֹ. וְנֹעַם לִקְחוֹ
. רָצִיתִי לַחְקֹר אִם חָכְמוֹתָיו. כְּפִי מְלִיצוֹתָיו. וְאָמַרְתִּי
לוֹ: בְּחֵי הַנּוֹתֵן לְךָ שָׂפָה בְרוּרָה. וְרוּחַ עֵצָה וּגְבוּרָה.
הוֹדִיעֵנִי נֵצֶר מַטְעֶךָ. וּנְוֶה רִבְעֶךָ.



וַיַּעַן וַיֹּאמַר:

אֲנִי חֶבֶר מְחַבֵּר שִׁיר חַמֻדוֹת

וּמֵפִיץ מִלְּשׁוֹנִי אֵשׁ פְּלָדוֹת.

וְאֶסְעַד הַלְבָבוֹת בַּאֲמָרוֹת

מְמֻלָּאוֹת בְּאַבְנֵי נִיב וְחִידוֹת.

אֲכַסֶּה אוֹהֲבַי מַעֲטֵה תְהִלָּה

וְאַלְבִּישׁ אוֹיְבַי בִּגְדֵי חֲרָדוֹת.





בְּשָׁמְעִי שִׁירָיו. וְתֹקֶף אֲמָרָיו. יָדַעְתִּי
כִּי הוּא מְאַלְפֵנוּ וּמוֹרֵנוּ. חֶבֶר הַקֵּינִי חֲבֵרֵנוּ.
וְעָמַדְתִּי כְפִי רֶגַע עִמּוֹ. לְהָרִיחַ בָּשְׁמוֹ. וְלִרְווֹת
בְּמֶתֶק נְאֻמוֹ. אַחֲרֵי כֵן קָרָאתִי שְׁלוֹמִי עָלָיו. וּפָנָה כָּל
אִישׁ מִמֶּנּוּ לְאֹהָלָיו.
