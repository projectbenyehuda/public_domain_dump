


	
	מחברת הנדיבים
	








נְאוּם יְהוּדָה בֶּן חֲרִיזִי הַסְּפָרַדִּי: בְּעוֹד



שַׁדַּי עִמָּדִי / וּבִימֵי חָרְפִּי וּכְבוֹדִי / עַל נַהֲרֵי שַׁלְוָה



פָּרַחְתִּי / וּבְגַלְגַּל הַשֵּׂכֶל זָרַחְתִּי / וְעַנְנֵי גִילָה רָווּ



תְלָמַי / וְהֵפִיחוּ מִתּוֹךְ גִּנְזֵי אֲסָמַי בְּשָׂמַי / וְנָגַהּ אוֹר



לְשָׁמַי / עַד בָּגַד בִּי הַזְּמַן וְהִתְמַכֵּר לְהָרַע לִי /



וַיַּעֲזוֹב בִּגְדּוֹ אֶצְלִי / וְשָׁלַח יָדוֹ בִּשְׁלוֹמָיו /



וַיְדַבֵּר לְסֶלַע עֵינַי וְנָתַן מֵימָיו / וְגָדַר דְּרָכַי וְעִוָּה



נְתִיבוֹתַי / וְשָׂם אֶל יָם הַדְּמָעוֹת תּוֹצְאוֹתַי / וַיָּסַר אֶת



אוֹפַן מַרְכְּבוֹתַי / וַיְנַהֲגֵנִי בִּכְבֵדוּת / וְקָשַׁר בֵּינִי



וּבֵין הַפְּרֵידָה יְדִידוּת / וְכַאֲשֶׁר הִכָּה בְּמַטֵּה הַיָּגוֹן



סַלְעֵי צְלָעַי / וְזוֹרְמוּ דְמָעַי לִרְגָעַי וָאַעַן וָאוֹמַר

אִלּו כְפִי אֵידִי דְמָעַי יִזְּלוּ / לֹא דָרְכָה רֶגֶל אֱנוֹשׁ יַבֶּשֶׁת

בֶּאְמֶת דְּעוּ כִי לֹא לְמֵי נֹח לְבַד / כִּי אִם לְדִמְעִי נִרְאֲתָה הַקֶּשֶׁת.





וְכַאֲשֶׁר נָסְעוּ אֲהוּבַי מִבֵּין זְרוֹעַי /



וַיַּחֲנוּ עֲצָבַי בֵּין צְלָעַי / עַל כַּנְפֵי פֵירוּד רִפַּדְתִּי



יְצוּעַי / וְנֶחְפַּזְתִּי לְמַסָּעַי / אַשְׁקֶה בְדִמְעָתִי אֶרֶץ



צִיָּה / כְּאִלּוּ דִּמְעָתִי יָם וַאֲנִי אֳנִיָּה / וָאַעַן



וָאוֹמַר

כָּל הַדְּמָעוֹת יִזְּלוּ גַם יֶחְדְּלוּ / עִתִּים לְבַד דִּמְעוֹת אֲסִירֵי אַהַב

עֵת מִכְּבֵדָם אֵיד אֲנָחוֹת יַעֲלֶה / יִהְיֶה לְמַיִם בַּעֲלוֹת הַלַּהַב.





וַיִּמָּלֵךְ עָלַי לִבִּי לִקְשׁוֹר עַפְעַפַּי



בְּעַפְעַפֵּי שְׁחָרִים / וְהִנְנִי בְּכַנְפֵי נְשָׁרִים / לִדְרוֹךְ



צִיּוֹת / וְלִרְכּוֹב אֳנִיּוֹת / עַד תַּגִּיעַ הַנֶּפֶשׁ לִרְצוֹנָהּ /



וְתָשׁוּב לְתַכְלִית גְּאוֹנָהּ / וּבְצֵאתִי מֵאַרְצִי / וּמִמְּכוֹן



רִבְצִי / הוֹכִיחַתְנִי לְשׁוֹן שִׂכְלִי עַל הַרְחִיקִי נְדוֹד / מִכָּל



רֵעַ וָדוֹד / וָאַעַן וָאוֹמַר

אַל נָא עָצֵל לְבַקֵּשׁ לָךְ / מִחְיָה וְשִׂים קִנָּךְ מְלוֹן אוֹרְחִים

כִּי לֹא תְהִי מוֹצֵא מְנוּחָה בִמְ/נוּחָה וּבַעְלִיּוֹת מְרֻוָּחִים

הַבֵּט לְשׁוֹן קֹדֶשׁ וְשׁוּר כִּי לֹא / יִתְקַבְּצוּ תוֹכוֹ שְׁנֵי נָחִים.





וְשַׂמְתִּי מְגַמָּתִי אֶל אָמוֹן מִנֹּא / וְשָׂכַרְתִּי



לִי מִבְּנוֹת הַיָּם כּוּשִׁית וְלֹא מִבְּנוֹת חַוָּה / שְׁחוֹרָה



וְנָאוָה / כְּבָרָק מְמַהֶרֶת / וְעַל כַּנְפֵי רוּחַ דּוֹהֶרֶת



וְנוֹהֶרֶת / נוֹסַעַת בְּלִי רַגְלַיִם / מְעוֹפֶפֶת בְּלִי כְּנָפַיִם /



עוֹלָה וְיוֹרֶדֶת / הָרָה וְיוֹלֶדֶת / וּבְמַשְׁבֵּר אֵינֶנָּה עוֹמֶדֶת



/ תָּמִיד יִתְרוֹצְצוּ הַבָּנִים בְּקִרְבָּהּ / וְיֵשׁ נְשָׁמָה לָעָם



עָלֶיהָ וְרוּחַ לַהוֹלְכִים בָּהּ / בְּהִתְנוֹסֵס בְּיַד הָרוּחַ



דְּגָלֶיהָ / תִּתְעַטֵּף בִּמְעִילֶיהָ / וְחַמָּתָהּ בְּשׁוּלֶיהָ /



וְאִם רָפְתָה רוּחָהּ מֵעָלֶיהָ / תַּחְשׂוֹף עַל פָּנֶיהָ שׁוּלֶיהָ /



וַתָּסַר צְעִיפָהּ מֵעָלֶיהָ / וַיְהִי כִּי עָלִינוּ בְגַבָּהּ /



וַנְּהִי כְּסוֹד סָתוּם בְּלִבָּהּ / וַנִּסַּע מֵחוֹרֶב וְאֶרֶץ צִיָּה /



אֶל הָאֳנִיָּה / וַתּוֹצִיאֵנוּ לִרְוָיָה / וְנָשִׁינוּ נְשִׁיָּה /



כְּאִלּוּ הַיַּבֶּשֶׁת / לָבְשָׁה מִפָּנֵינוּ בֹּשֶׁת / וְנֶעֶלְמָה



לְמַרְאֵה עֵינַיִם / וְאֵין כִּי אִם מַיִם וְשָׁמַיִם דּוֹמִים לְיַמִּים



שְׁנַיִם / וְאֵין מַבְדִּיל בֵּין מַיִם לָמָיִם / וַנָּבוֹא בְּצִי



אַדִּיר וַיִּסְגֹּר יְיָ בַּעֲדוֹ / וְהַסְּפִינָה צְפוּנָה בְּלֵב הַיָּם



וּכְבֵדוֹ / וְהָרוּחַ כְּרָכִיל יְגַלֶּה סוֹדוֹ / וְהַמְשׁוֹטִים



חֲלוּצִים נְחוּצִים / וְדוֹמִים לִשְׁנֵי חִצִּים / כַּבְּרָקִים רָצִים /



וְהָאֹרֶן וְהַתֹּרֶן / כְּמוֹץ יְסוֹעֵר מִגֹּרֶן / וְהַקְּלָעִים /



נִקְרָעִים / וִידֵי הַמַּלָּחִים / כִּבְלוֹאֵי מְלָחִים / וְהַגַּלִּים /



יִתְגַּלְגְּלוּ כְגַלְגַּלִּים / עַד קָרַבְנוּ לִנְוֵה מְנוּחֵנוּ /



וְשׁוֹבְבָה רוּחַ הַשָּׁלוֹם רוּחֵנוּ / וַיֵּרָאוּ אֲפִיקֵי יָם לְרַב



הַחוֹבֵל / וַיִּגָּלוּ מוֹסְדוֹת תֵּבֵל / וְהַמִּגְדָּל אֲשֶׁר לְחוֹף



הַיָּם בְּצִפִּיָּתֵינוּ צִפִּינוּ / וּבֵין מִגְדּוֹל וּבֵין הַיָּם



חָנִינוּ / וְיָצָאנוּ מֵעֲבוֹדַת הַיָּם הַקָּשָׁה / לִמְנוּחַת



הַיַּבָּשָׁה / וְשׁוֹרַרְנוּ לָאֵל שִׁירָה חֲדָשָׁה / עַל גְּאֻלָּתֵינוּ



/ וְעַל פְּדוּת נַפְשֵׁינוּ / וּבָאנוּ לְעִיר נֹא אָמוֹן לִלְטֹשׁ



עֵינֵינוּ בְּצִיצֵי צְמָחֶיהָ / וְגֶרֶשׁ יְרָחֶיהָ / וְכַפְתּוֹרֶיהָ



וּפְרָחֶיהָ / וְהָעִיר כְּכַלָּה כְּלִילַת יוֹפִי / בְּלִי דּוֹפִי /



וְשָׁם קְהִלָּה נִכְבֶּדֶת אַךְ מָצָאתִי בָּהֶם אִישׁ מַחְפִּיר



וּמַבְאִישׁ הָיָה לִי מִשְׁעֶנֶת קָנֶה רָצוּץ / חֲשַׁבְתִּיו הֲדַס



מְצָאתִיהוּ נַעֲצוּץ / וְהוּא מִכָּל חָכְמָה נָעוּר וָרֵק / כְּגֶפֶן



בּוֹקֵק / וְהַחָכְמָה מֵאַיִן תָּבוֹא לִבְנֵי עוֹנְנָה / אוֹ מִי נָתַן



לַשֶּׂכְוִי בִּינָה / תֵּדַע עֵינוֹ בְּהוֹנוֹ / לָתֵת שָׁלוֹם



לִשְׁכֵנוֹ.



לְשׁוֹן כּוֹסוֹ דָּבֵק / לְחִכּוֹ בַּצָּמָא / וְשִׂפְתֵי שֻׁלְחָנוֹ / עֲטוּפִים בָּרָעָב



לְהָסֵךְ אֶת רַגְלָיו / יְפַחֵד פֶּן יִצְמָא / וְכֵן יִירָא לִבְדּוֹק / נְקָבָיו פֶּן יִרְעַב





וְרוֹעַ צוּרָתוֹ / תְּגַלֶּה לְךָ נַבְלוּתוֹ / וְכָל



מוּם אֲשֶׁר יַעֲלִימֶנּוּ / מִמַּרְאֵה פָּנָיו תְּשֻׁרֶנּוּ / מִבַּיִת



וּמִחוּץ תְּצַפֶּנּוּ

וְיֵשׁ אִישִׁים מְתוּקֵי נִיב וְשָׂפָה / וּבִלְשׁוֹנָם יְפַתּוּנִי וְנִיבָם

יְגַלּוּ לִי פְּנֵי רָצוֹן וְאַהְבָה / וְיָדַעְתִּי הֲכִי אָרְבָּם בְּקִרְבָּם

וְאֵיךְ יוּכְלוּ צְפוֹן רוּחַ לְבָבָם / וְהַכָּרַת פְּנֵיהֶם עָנְתָה בָם.





וְאִם אַאֲרִיךְ בּוֹ אִמְרָתִי / אֲחַלֵּל כְּבוֹד



מְלִיצָתִי / וְאָשִׂים לְאַל מִלָּתִי / וְכַאֲשֶׁר רָתַמְתִּי מִנֹּא



אָמוֹן מִרְכֶּבֶת הַפְּרִידָה / לָקַחְתִּי טוּב זִכְרָהּ לְצֵידָה /



וּבָאתִי בָּאֳנִיּוֹת שִׁיחוֹר / נוֹסְעוֹת עַל יַד הַיְּאוֹר / בֵּין



גֻּלּוֹת הַגַּלִּים כָּעָב תְּעוּפֶינָה / וַתָּבֹאנָה אֶל קִרְבֶּנָה /



וְלָהֶם מִן הַמְּשׁוֹטִים כְּנָפַיִם כְּכַנְפֵי הַחֲסִידָה וְלֹא



שָׁקַטְנוּ לִדְאוֹת / בְּכַנְפֵי הָאֳנִיּוֹת / עַד קָרְצָה לָנוּ



מִצְרַיִם בְּעֵינֶיהָ / וְגִלְּתָה לָנוּ צְעִיף פָּנֶיהָ / וְיָצְאוּ



לִקְרָאתֵנוּ בְּשָׂמֶיהָ / בְּמִרְקְחוֹת סַמֶּיהָ / וְכַאֲשֶׁר קָרַבְנוּ



לִנְוֵה הֲדוֹמָהּ / שָׁאַלְנוּ בִּלְשׁוֹן הַתְּשׁוּקָה לִשְׁלוֹמָהּ /



אַף הִיא תָּשִׁיב אֲמָרֶיהָ / בִּלְשׁוֹן רוּחוֹת שְׁחָרֶיהָ /



הַנּוֹשְׁבוֹת מֵעֲבָרֶיהָ / וַיְהִי בְּבוֹאֵנוּ הָעִירָה תָּמַהְנוּ



בְּנוֹעַם בִּנְיָנֶיהָ / וְחֹסֶן אַרְמוֹנֶיהָ / וַעֲרוּגוֹת גַּנֶּיהָ /



וְרֵיחַ שְׁמָנֶיהָ / וַנַּעֲבוֹר אֶל עֵבֶר הַיְּאוֹר / לִנְוֵה הַזּוֹהַר



וְהָאוֹר / אֲשֶׁר שָׁם הָיָה מַעֲמַד מֹשֶׁה אִישׁ הָאֱלֹהִים /



וְהִשְׁתַּחֲוִינוּ שָׁם לִפְנֵי הָאֱלֹהִים / וְנַחְנוּ שָׁם יָמִים



שְׁלֹשָׁה וּבַיּוֹם הַשְּׁלִישִׁי שַׂמְנוּ אֶל הָעִיר מְגַמַּת פָּנֵינוּ



/ וְקָרַעְנוּ בְּפוּךְ יָפְיָהּ עֵינֵינוּ / וְהִנֵּה כֻּלָּהּ זִיו



וָזוֹהַר / כְּעֶצֶם הַשָּׁמַיִם לָטוֹהַר / כְּאִלּוּ הָעוֹלָם חֹשֶׁךְ



וְהִיא מְנוֹרַת הַמָּאוֹר / כֻּלָּהּ מִקְשָׁה אַחַת זָהָב טָהוֹר /



וְהִיא נֶחְלֶקֶת לִשְׁתֵּי מְדִינוֹת / עֲדִינוֹת / הָרִאשׁוֹנָה



מִצְרַיִם הַנְּסוּכָה / וְהַשֵּׁנִית עִיר הַמְּלוּכָה / וּבִשְׁתֵּיהֶן



מִבְּנֵי הַגּוֹלָה קְהִלּוֹת / לָאֵל בְּלִבָּם מְסִילוֹת / וּבְתוֹכָם



הַכֶּתֶם הַטּוֹב / הַנִּכְתָּר בְּשֵׁם טוֹב / רַבֵּינוּ מְנַחֵם יָרוּם



הוֹדוֹ אֲשֶׁר בְּמִדּוֹתָיו הַלְּשׁוֹנוֹת תִּקְצַרְנָה / וְכָל



תּוֹדוֹתַי [וְאִם] תֶּאֱרַכְנָה / בְּמַהֲלָלוֹ תִּמְעַטְנָה /



וּתְהִלּוֹתַי וְאִם מִיָּם עַד יָם תֵּלַכְנָה /



רַק בַּיְּאוֹר תִּשָּׁאַרְנָה / כִּי אוֹתוֹ רָאִיתִי נָדִיב בְּדוֹרוֹ /



וְאֵין לְאֵל יָדוֹ / וְיִתְחַנֵּן כְּשׁוֹאֵל / לְכָל שׁוֹאֵל / וְכָל



אֲשֶׁר יִתְּנֵהוּ / מִדַּם הַלֵּב יִקָּחֵהוּ / וּמֵחֵלֶב כְּלָיוֹת



יַתִּיכֵהוּ / וּמִבֵּין שִׁנָּיו יוֹצִיאֵהוּ / וְיִשְׂמַח לִבּוֹ



בְּתִתּוֹ / כְּשִׂמְחַת הַשּׁוֹאֵל בְּקַחְתּוֹ / וְנִדְבָתוֹ לֹא



לִשְׂכֹּר תְּהִלָּה / וְלֹא לְבַקֵּשׁ גְּדֻלָּה / רַק לְנֶפֶשׁ יְקָרָה



אֲשֶׁר בּוֹ / וְרוּחַ קְדֻשָּׁה אֲשֶׁר נָתַן אֱלֹהִים בְּלִבּוֹ /



וּבֶאֱמֶת רָאִיתִי פָּנָיו כִּרְאוֹת פְּנֵי אֱלֹהִים.

יוֹם לָבְשׁוּ שָׁמַי עֲנַן אֵיד גַּם / מַבּוּל דְּמָעַי בָּא לְהַשְׁחִיתוֹ

הָיָה מְאוֹר פָּנָיו כְּמוֹ קֶשֶׁת / בַּעְנָן זְמָן נִרְאָה לְהַחְיוֹתוֹ.





וּבְעוֹד אֲנִי תָּמֵיהַּ לְטוֹבוֹתָיו אֲשֶׁר קִדְּמוּ



אוֹתִי / וְהִנֵּה כְּפִיר אֲרָיוֹת שׁוֹאֵג לִקְרָאתִי / מִקּוֹלוֹ



יֵחָתוּ כְפִירִים / וּמִשָּׁאֲגָתוֹ יִמַּס לֵב הַגִּבּוֹרִים / הוּא



גּוּר הַחָכְמוֹת / שַׁחַל הַמְזִמּוֹת / רַבֵּינוּ אַבְרָהָם בֶּן



אֲדוֹנֵינוּ נֵזֶר גָּלֻיּוֹת כָּל יִשְׂרָאֵל רבי מֹשֶׁה ז״ל גַּם הוּא



גָּבַר עָלַי חַסְדּוֹ / וַיָּאצֶל עָלַי מִכְּבוֹדוֹ / וּבַחֲדַר אַהֲבָיו



הֱבִיאַנִי / וְלִוְיַת חֵן לְרֹאשׁוֹ שָׂמַנִי / וּמַה יּוּכַל לְשׁוֹן



הָעֵט לְחַבֵּר / וַחֲיָלָיו לְגַבֵּר / לִפְנֵי אֲדוֹן הַמִּדּוֹת



הַנִּכְבָּדוֹת / אֲשֶׁר לוֹ עַל הַחֲכָמִים עֶשֶׂר יָדוֹת / וְאִם



תִּשְׁאַל הַיְּקָר הַאַתָּה אָדוֹן / יֹאמַר לֹא כִּי עֶבֶד אַבְרָהָם



אָנֹכִי / וְלֹא קָנַנִי וּסְגֻלּוֹתָיו / אֲבָל יָרַשְׁתִּי



מֵאֲבוֹתָיו.





אִם הַיְּקָר אָדוֹן לְכָל אָדוֹן / הוּא בֶּאֱמֶת עֶבֶד לְאַבְרָהָם

מִיּוֹם כְּבוֹדוֹ נֶעֱלַם מִנִּי / אֵלֵךְ מְשׁוֹטֵט בַּעֲדוֹ נִדְהָם

כָּמְהוּ סְעִפַּי לוֹ בְּעָבְרִי אֶ/רֶץ נוֹד וְאִתִּי יַעֲבוֹר כִּמְהָם.





אַחַר הַדְּבָרִים הָאֵלֶּה יָצָאנוּ מִמִּצְרַיִם /



לִרְאוֹת בְּטוּב יְרוּשָׁלַיִם / וַנֵּפֶן וַנַּעֲבוֹר דֶּרֶךְ



הַמִּדְבָּרָה / בְּאֶרֶץ צִיָּה וְצָרָה / וּבְחַרְבוֹנֵי קַיִץ עַצְמִי



חָרָה / וְקָרָאתִי שֵׁם הַמָּקוֹם הַהוּא תַּבְעֵרָה / כִּי שָׁם בְּנֵי



יָמִים מַקְרִיבִים אֵשׁ זָרָה / וַתְּהִי נַפְשִׁי בְּרִשְׁפֵיהֶם



שְׂרוּפָה / וְתַנִּים וּבְנוֹת יַעֲנָה יִבְכּוּ אֶת הַשְּׂרֵפָה /



וַנֵּלֶךְ בַּמִּדְבָּר הַגָּדוֹל וְהַנּוֹרָא הַהוּא מְקוֹם דַּאֲבוֹן



נֶפֶשׁ וְכִלְיוֹן עֵינַיִם / נָחָשׁ שָרָף וְעַקְרָב וְצִמָּאוֹן אֲשֶׁר



אֵין מַיִם / אֲבָל שַׂמְנוּהוּ בְּזֵעַת אַפֵּינוּ / וְדִמְעַת



עַפְעַפֵּינוּ מְקוֹם נַהֲרַיִם / יְאוֹרִים רַחֲבֵי יָדַיִם / וַיְהִי



בְּבוֹאֵנוּ עַד עַזָּה / אַחֲרֵי יְגִיעָה קָשָׁה וְעַזָּה / עָרַכְנוּ



לָאֵל תְּהִלָּה בְּתוֹף וּמָחוֹל / עַל אֲשֶׁר עָבַרְנוּ מַעֲבַר הַחוֹל /



וְיָצָאנוּ מִן הַקֹּדֶשׁ אֶל הַחוֹל / וְלֹא חָדַלְנוּ לָרוּץ כְּנַחַל



שׁוֹטֵף / וּכְבָרָק חוֹטֵף / יַסְתִּירֵנוּ כְּנַף הָאֲפֵלָה /



וַיּוֹרֵנוּ הַנְּתִיבוֹת אִישׁוֹן לַיְלָה / עַד נִפְתְּחוּ לָנוּ



שַׁעֲרֵי שָׁמַיִם / הִיא עִיר יְיָ יְרוּשָׁלַיִם / וַנִּקּוֹד



וַנִּשְׁתַּחֲוֶה לְאֵל עֵילוֹם / אֲשֶׁר פְּקָדַנוּ בִּפְקֻדַּת שָׁלוֹם /



וְנַחְנוּ שָׁם חֹדֶשׁ יָמִים / וּבַיָּמִים הָהֵם חֲבָלִים נָפְלוּ לִי



בַּנְּעִימִים / יָמִים כְּאִלּוּ מִפְּנִינִים נֶחְצְבוּ וּמֵעֲצֵי



חַיִּים נִקְצְבוּ / וּמִכֹּכְבֵי שַׁחַק גֻּנְּבוּ / וְהָיִינוּ



מִתְהַלְּכִים בְּכָל יוֹם לִבְכּוֹת עַל צִיּוֹן הַמִּתְאַבֶּלֶת עַל



בָּנֶיהָ / עַד כָּרוּ יִבְלֵי הַדְּמָעוֹת שִׁיחוֹת בְּפָנֶיהָ /



וּלְקוֹנֵן עַל הֲרִיסוֹת אַרְמוֹנֶיהָ / וּשְׁאֵרִית בִּנְיָנֶיהָ /



וּלְסוֹבֵב עַל צִיּוּנֵי הַצַּדִּיקִים אֲשֶׁר סְבִיבוֹתֶיהָ /



וְלִסְפּוֹד עַל הָאֵם וְעַל בְּנוֹתֶיהָ / וּבִקְצָת הַיָּמִים



תִּמְשְׁכֵנוּ יַד הַתְּשׁוּקָה בַּעֲבוֹתִים / לַעֲלוֹת אֶל הַר



הַזֵּיתִים / לַעֲרוֹךְ תְּפִלּוֹת לְעוֹשֵׂה נִפְלָאוֹת /



לְהִשְׁתַּחֲווֹת לְמֶלֶךְ יְיָ צְבָאוֹת / וְהָיִינוּ בְּעַיִן דּוֹמַעַת



/ וְנֶפֶשׁ נִכְנַעַת / מַשְׁקִיפִים עַל מְקוֹם הָעֲזָרָה / אֲשֶׁר



נֶחְלַף לַעֲבוֹדָה זָרָה / וְהָיִינוּ מַסְתִּירִים פָּנֵינוּ מֵהַבִּיט



לִנְוֵה הַכָּבוֹד וְהַגְּדֻלָּה / אֲשֶׁר הָיָה שָׁם אָהֳלוֹ בַּתְּחִלָּה



/ אֲשֶׁר מִקֶּדֶם שָׁכְנָה בוֹ שְׁכִינָה / וְעַתָּה שָׁכְנָה עָלָיו



עֲנָנָה / וְלֹא נִשְׁאַר מִן הַבִּנְיָנִים הָהֵם / כִּי אִם אוֹתוֹת



יַכְאִיבוּ לֵב זוֹכְרֵיהֶם / וְיַדְאִיבוּ בִּבְכִי עֵין רוֹאֵיהֶם /



וּבִמְלֹאת הַיָּמִים הָאֵלֶּה שַׂמְנוּ מְגַמַּת פָּנֵינוּ מִדְבָּרָה



דַּמֶּשֶׂק / וְקָרָאנוּ שָׁם הַזְּמָן עֵשֶׂק / כִּי הִתְעַשַּׂקְנוּ



עִמּוֹ / וּבְבוֹאֵנוּ עָדֶיהָ רָאִינוּ עִיר יְפֵהפִיָּה / בְּנוּיָה



לְתַלְפִּיָּה / דְּרָכֶיהָ דַּרְכֵי נוֹעַם / וְכֻלָּה יָפָה ושָׂרַת



טַעַם / דְשֵׁנָה כְּגֶפֶן פּוֹרִיָּה / בִּלְחִי שׁוֹשַׁנֶּיהָ / וְעֵינֵי



מַעְיָנֶיהָ / שׂוֹחֶקֶת וּבוֹכִיָּה / וְכָל נְאוֹתֶיהָ עֲרוּגוֹת



הֲדַסִּים / וְכָל יְעָרֶיהָ גַּנּוֹת וּפַרְדֵּסִים / כַּמַּיִם לַיָּם



מְכַסִּים / עַל דְּרָכֶיהָ נְחָלִים נִטָּיוּ / וּבִלְחִי הֲרָרֶיהָ אֵשׁ



וָשֶׁלֶג יִשְׁלָיוּ / וּמִצּוּף שְׂפָתֶיהָ צְמֵאִים יִרְוָיוּ / וּבְכָל



הַמְּסִלּוֹת / תְּעָלוֹת / וְעַל הַדְּרָכִים / לַמַּיִם מַהְלְכִים /



וְעַל הַנְּתִיבוֹת / מַרְאוֹת הַשּׁוֹאֲבוֹת / וְרֵיחָהּ כְּרֵיחַ



לְבָנוֹן / וְסָבִיב לָהּ תָּבוֹר וְחֶרְמוֹן / וּבַקַּיִץ וּבַחֹרֶף



תַּשְׁלֵג בְּצַלְמוֹן / וּבְתוֹךְ הָעִיר מִבְּנֵי הַגּוֹלָה / קְהִלָּה



גְּדוֹלָה / וּבְרֹאשָׁם יְחִיד הַדּוֹר וּפִלְאוֹ / מִמִּזְרַח שֶׁמֶשׁ



עַד מְבוֹאוֹ / נְגִיד הַזְּמָן וּנְשִׂיאוֹ / יָדָיו יָם נְדָבוֹת /



וּפָנָיו בֹּקֶר לֹא עָבוֹת / הַשַּׂר הַגָּדוֹל הָרוֹפֵא ר׳ מֹשֶׁה יָרוּם



הוֹדוֹ.

שׁוֹאֵל בְּעַד מֹשֶׁה פְּאֵר שָׂרֵי זְמָן / צוּרֵי סְלָעִים דָּשְׁאוּ מִנּוֹזְלָיו

מוֹשֵׁל אֲשֶׁר מָשַׁל בְּרֹאשׁ שָׂרִים אֲבָל / נִכְנַע לְכָל דּוֹרְשָׁיו עֲדֵי הֵם מוֹשְׁלָיו

נַפְשׁוֹ לְאֵל מָכַר וְעֵינָיו אֶל יְדֵי / שׁוֹאֵל כְּעֵין עֶבֶד אֱלֵי יַד גּוֹאֲלָיו



מַה נּוֹאֲלוּ דּוֹרְשָׁיו אֲשֶׁר חָשְׁבוּ הֲכִי / הוּא יִגְמְלֵם [חֶסֶד] אֲבָל הֵם גּוֹמְלָיו



יִתֵּן וְיִתְנַדֵּב כְּנִדְבַת עַנְנֵי / שַׁחַק וְיִתְחַנֵּן תְּחִנַּת שׁוֹאֲלָיו.





הוּא לְבַדּוֹ מִבְּנֵי עִירוֹ כִבְּדַנִי / וְזֶבֶד טוֹב



זְבָדַנִי / אֵין בָּם אֱנוֹשׁ יַעֲטֶה סוּת הַמַּהְלָלִים כְּמַדּוֹ /



יִרְאוּ קְרוֹב לַנְּדָבָה וְנִגַּשׁ מֹשֶׁה לְבַדּוֹ / וְכִמְעַט אָבְדָה



נַפְשִׁי בְּעָנְיָהּ / לוּלֵי מָצָאתִי בָּם אִישׁ שֵׂכֶל בִּפְעֻלּוֹתָיו



/ וְנֶחְמָד בְּמִדּוֹתָיו / הָאָח הַנֶּאֱמָן / צִנְצֶנֶת הַמָּן / נֶטַע



נַעֲמָן / וְאֵינוֹ מִבְּנֵי הַמְּדִינָה / רַק מִבְּנוֹת עֲדִינָה / הוּא



עֶזְרָא עָלָה מִבָּבֶל / לְהָאִיר פְּנֵי תֵּבֵל / וּמְצָאתִיו בְּיָם



יְגוֹנִי רַב הַחוֹבֵל / וְהִשְׁכִּיחַנִי בְּמֶתֶק חֶבְרָתוֹ הַחֲמוּדָה /



לַעֲנַת הַפְּרִידָה / וְהִגְדִּיל כְּבוֹדִי וְהוֹדִי / לָאִיִּים



הָרְחוֹקִים אֲשֶׁר לֹא שָׁמְעוּ אֶת שִׁמְעִי וְלֹא רָאוּ אֶת כְּבוֹדִי /



צוּר בְּרַחֲמָיו יְשַׁלֵּם גְּמוּלוֹ / וְיִרְצֶה פָּעֳלוֹ.





כּוֹכָב מִשְּׁחָקִים יָרַד / הֶחְרִיד רוֹם וְהִרְעִישׁ

מִצִּבְאוֹת מְרוֹמִים שָׂרַד / לִשְׁכּוֹן בֵּין בְּנֵי אִישׁ

לִבּוֹ אֵשׁ בְּעֹצְֶם בָּרָד / וּגְוִיָּה כְתַרְשִׁישׁ

בּוֹ מִדּוֹת וּבָם עִם אֵל רָד / מִדּוֹת עִיר וְקַדִּישׁ

מַסְוֵה חֵן וְתוֹם עַל עֵינָיו / וּכְנַף הוֹד יְנוֹפֵף

בִּשְׁתַּיִם יְכַסֶּה פָּנָיו / בִּשְׁתַּיִם יְעוֹפֵף.





עֶזְרָא בַּזְּמָן הוּא אֶחָד / אַךְ נֶחְשָׁב קְהִלָּה

כִּי הִתְחַבְּרוּ בוֹ [יַחַד] / שֵׂכֶל עִם תְּהִלָּה

וּבְסוֹד אֵל כְּבוֹדוֹ תֵּחַד / עֵת יַעְרוֹךְ תְּפִלָּה

וּבְמַעְלוֹת יְקָר הִתְאַחַד / וַיַּעְלֶה תְּחִלָּה

מִגְדַּל עוֹז יְחַוֶּה גָּדְלוֹ / עֵת עָלָיו יְחוֹפֵף

יָרִיעַ וּמֵהוֹד קוֹלוֹ / הַגַּלְגַּל יְרוֹפֵף.





לִלְחוֹם מִלְחֲמוֹת הַתּוֹרָה / פִּיו חֶרֶב לְטוּשָׁה

וַתִּהְיֶה בְּעֵט תִּפְאָרָה / עַל לִבּוֹ חֲרוּשָׁה

וּבְשָׂפָה בְּרוּרָה יִקְרָא / וּנְעִימָה קְדוֹשָׁה

קוֹל עֲנוֹת אֲבָל בִּגְבוּרָה / אַךְ לֹא קוֹל חֲלוּשָׁה

כָּל הוֹמֶה בְּיָגוֹן נֶאְנָח / אוֹ עַל לֵב יְתוֹפֵף

בּוֹ יִשְׂמַח וְזַעְמוֹ יִזְנַח / כָּל נוֹגֵן וְתוֹפֵף.





עֵת יַעְטוֹף לְאֵל כְּשׁוֹאֵל / אִמְרֵי פִיו לְרָצוֹן

וּנְקִיָּה תְּפִלָּתוֹ אֶל / אֵל מִמּוּם וְלָצוֹן

הִיא עוֹלָה כְּעוֹלָה לָאֵל / טוֹב מִקָּרְבְּנוֹת צֹאן

לִבּוֹ זַךְ בְּלִי מִתְגָּאֵל / כִּפְנִימִי כְּחִיצוֹן

עַל לֵב דּוֹד בְּחִילָה יִפְעֶה / טַל אוֹרוֹת יְנוֹפֵף

וּבְלֵב אוֹיְבָיו הוּא אֶפְעֶה / עִם שָׂרָף מְעוֹפֵף.





וְעַתָּה קָצְרָה נַפְשֵׁינוּ וּבַחַיִּים זָנַחְנוּ /



כִּי בַּאֲשֶׁר נַחְנוּ לֹא נַחְנוּ / וְנוֹסְעִים אֲנַחְנוּ / לִנְוֵה



הַחֲמוּדוֹת / וּמְקוֹם הַמִּדּוֹת הַנִּכְבָּדוֹת / הִיא אֲרַם צוֹבָה



אֲשֶׁר שָׁם גַּלְגַּל הַגְּדֻלָּה / וּנְקֻדַּת הָעֲגֻלָּה / הוּא הַשַּׂר



הַגָּדוֹל מוֹפֵת הַזְּמַנִּים / וּרְאָיָה לָאַחֲרוֹנִים עַל



הָרִאשׁוֹנִים / מַנְהִיג הָעֵדֶר / וְרֹאשׁ הַסֵּדֶר / יוֹסֵף הַשַּׁלִּיט



עַל הָאָרֶץ / הַמַּשְׁבִּיר לְכָל עַם הָאָרֶץ.



שְׂאוּ שָׁלוֹם לִגְבִיר / חֲמוּדוֹת בִּין אוֹסֵף / תְּהִלָּתוֹ מִפָּז / יְקָרָה וָכֶסֶף



זְמָן לִסְפּוֹר מִקְצָת / חֲמוּדוֹתָיו כּוֹסֵף / וּבִקֵּשׁ לוֹ מֵלִיץ / הֱיוֹת סוֹדוֹ חוֹשֵׂף.





וְאֶת יְהוּדָה שָׁלַח לְפָנָיו אֶל יוֹסֵף שֶׁמֶשׁ



תְּהִלּוֹת עָלָה מִמַּעֲרָב / וַיִּחַן בַּמִּזְרָח וַיְהִי שָׁם לְגוֹי



גָּדוֹל וְעָצוּם וָרָב / בָּנָה אוֹצָרוֹת הַחֲמוּדוֹת / וַיְמַלֵּא



אוֹתָם בַּר הַתְּעוּדוֹת / וַיִּצְבֹּר יוֹסֵף בָּר / וּבָא לַעֲרוֹךְ



שֻׁלְחָן בַּמִּדְבָּר / לָדַעַת לָעוּת אֶת יָעֵף דָּבָר / הִרְוָה



צְמֵאִים / וְרִפֵּא נְכָאִים / וְהֵשִׁיב לֵב אָבוֹת עַל בָּנִים /



וּמָחָה דִּמְעָה מֵעַל כָּל פָּנִים.

בְּךָ הוֹבִישׁ אֱלֹהִים יָם דְּמָעוֹת / וּבָךְ אָמַר לְצוּלָתָם חֳרָבִי

בְּלִבָּךְ בֵּית זְבוּל בָּנָה לְשִׁבְתּוֹ / לְשׁוֹן קֹדֶשׁ וְיַהֵל שָׁם עֲרָבִי

וְהָיִיתָ גְּבִיר וָרָב בְּמַעְרַב / וּבַמִּזְרָח מְשָׁחָךְ אֵל לְנָבִיא

וְאָמְנָם כִּי אֲרַם צוֹבָה כְּמִקְדָּשׁ / וְאַתָּה בּוֹ כְּמוֹ נֵר מַעֲרָבִי.





בְּךָ חִדֵּשׁ הַזְּמָן אַחֲרֵי בְּלוֹתוֹ נְעוּרִים /



וְרָחַץ בְּמַיִם טְהוֹרִים / וּפָשַׁט אֶת בְּגָדָיו וְלָבַשׁ בְּגָדִים



אֲחֵרִים / וּבְיוֹם טָהֳרָתוֹ לָקַח לוֹ שְׁנֵי תּוֹרִים / תּוֹר



הַתּוֹרָה וְתוֹר הַמִּשְׂרָה אֲשֶׁר בְּךָ נֶחְבָּרִים / וּבִשְׁנֵיהֶם



נָאווּ לְחָיָיו בַּתּוֹרִים / צַוָּארוֹ בַּחֲרוּזֵי הַשִּׁירִים.

הַשַּׂר בְּתוֹךְ פִּיהוּ לְשׁוֹן זָהָב וְאַ/דֶּרֶת תְּהִלָּתוֹ מְאֹד נֶאְדֶּרֶת

בָּזַז חֲמוּדוֹת בִּין וְגָנַב לֵב מְתֵי / שֵׂכֶל וְלֹא נִמְצָא בְּתוֹךְ מַחְתֶּרֶת

וּבְעֵת פְּלִילִים חִפְּשׂוּ תּוֹךְ אָהֳלוֹ / מָצְאוּ לְשׁוֹן זָהָב וְהָאַדֶּרֶת.





אַשְׁרֵי הָאִישׁ אֲשֶׁר יְלָדָהוּ / וּצְרוֹר הַמּוֹר



אֲשֶׁר צְנָפָהוּ / גָּבַר עַל בְּנֵי דוֹרוֹ / וְכַבֵּן כְּהוֹרוֹ /



וְכַשֹּׁרֶשׁ אֲמִירוֹ / וְאֵין לְמִי אַמְשִׁיל אֶחָד מֵהֶם וְאַשְׁוֶנּוּ



/ כִּי מִמִּדּוֹת בְּנֵי הַדּוֹר יוֹסֵף אֵינֶנּוּ / וְשִׁמְעוֹן



אֵינֶנּוּ / הַרְאִיתֶם אֲשֶׁר בָּחַר בּוֹ יְיָ / זֶה סִינַי / אֶבֶן



יְקָרָה לֹא הֵבִין אֱנוֹשׁ עֶרְכָּהּ / אֱלֹהִים הֵבִין



דַּרְכָּהּ.

הָיָה בְּאֶרֶץ מַעֲרָב אוֹדֶם בְּתוֹךְ / הַבּוֹץ וְאֵין אִישׁ סוֹד יְקָרוֹ חוֹסֵף

עַד כִּי יְדֵי הַיְקָר לְדָרְשׁוֹ מִהֲרוּ / וַיִּמְשְׁכוּ וַיַּעֲלוּ אֶת יוֹסֵף.





הוּא כּוֹלֵל הַמִּדּוֹת הַחֲמוּדוֹת / וּבְסוֹדוֹת



הַתְּעוּדוֹת / לוֹ עֶשֶׂר יָדוֹת / וּבְסוֹד הָרְפוּאוֹת / יְגַלֶּה



נִפְלָאוֹת / וּבְחָכְמַת הַמִּדּוֹת / בָּנָה בֵּית הַמִּדּוֹת /



וּבִשְׁאֵלוֹת הַמִּנְיָן / מֵשִׁיב כַּעִנְיָן / וּבִנְתִיבוֹת



הַגַּלְגַּל / יְפַלֵּס כָּל מַעְגָּל / וּבִתְעוּדוֹת הַהִגָּיוֹן /



יַסְכֵּל כָּל רַעְיוֹן / וְרָזֵי הַתּוֹרָה / יוֹצִיאֵם לְאוֹרָה / וְסוֹד



הַדִּקְדּוּק / עַל לִבּוֹ חָקוּק / וְיִפְתַּח בִּגְמָרָא / כָּל דֶּלֶת



סְגוּרָה / וּבְסוֹד הַמִּשְׁנָה / מִפִּיו דַּעַת וּתְבוּנָה /



וְחָכְמָתוֹ כְּבֻדָּה בַת מֶלֶךְ / וּבְבוֹאָהּ לִפְנֵי הַמֶּלֶךְ /



תְּשׁוּאוֹת חֵן חֵן לָהּ / וַיְבַהֵל אֶת תַּמְרוּקֶיהָ וְאֶת מָנוֹתֶיהָ



לָתֶת לָהּ / וַאֲרוּחָתוֹ אֲרוּחַת תָּמִיד / וּמְנוֹרַת שִׂכְלוֹ



בְּאֹהֶל מוֹעֵד הֶעֱמִיד / לְהַעֲלוֹת נֵר תָּמִיד.

וְאַחֲרָיו הֶחְזִיקוּ בְּנֵי אֱלֹהִים וּקְדוֹשֵׁי



עֶלְיוֹן / תּוֹמְכֵי הַדֶּגֶל הַנַּעֲלֶה / וְהֵם בְּנֵי הַמְּעֻלֶּה /



אֲשֶׁר אֶל גְּוִיּוֹת הַמִּשְׂרָה הֵם נְשָׁמוֹת / וּלְכֹכְבֵי



הַתְּהִלָּה מְרוֹמוֹת / וּבְרֹאשָׁם הָרַב הַנִּכְבָּד הַמְּעֻלֶּה רַבִּי



מֹשֶׁה / אֲשֶׁר בּוֹ יְנַגֵּן הַמְּנַגֵּן וְלֹא יֶחֱשֶׁה.





יַד כָּל רוֹדְפֵי הַמַּעְלָה / אָזְלָה / מֵהַשִּׂיג וּמֹשֶׁה עָלָה





מַעְבָּדָיו / עֲמוּסֵי טוֹבוֹת

הֵם עֵדָיו / לְיַחַס אָבוֹת

גַּם יָדָיו / כְּיָם לִנְדָבוֹת

כָּל הַבָּא וּפָנָיו חִלָּה / הֶעְלָה / מֵי זָהָב וְדָלֹה דָּלָה.









קִרְבָתוֹ / לְדוֹדָיו רָצוֹן

אַהְבָתוֹ / לְמַר לֵב שָׂשׂוֹן

נִדְבָתוֹ / כְּלִיל הַיָּגוֹן

כִּמְנוֹרָה וְזָהָב כֻּלָּהּ / גֻּלָּה / עַל רֹאשָׁהּ לְשֵׁם וּתְהִלָּה.





רוֹב דֵּעוֹ / כְּרוֹב עַנְוָתוֹ

כַּצִּנְעוֹ / יְשַׁו גַּאְוָתוֹ

הִכָּנְעוֹ / כְּפִי מַעְלָתוֹ

אִם נַפְשׁוֹ לְעָפָר שָׁפְלָה / גָּדְלָה / הִיא יוֹרְדָה וְהִיא הָעוֹלָה.





זַךְ מַעְלָל / נְתִיב טוֹב סִקַּל

עֵת מִכְלָל / הֲדָרוֹ נִשְׁקַל

כָּל מַהְלָל / לְעֶרְכּוֹ יֵקַל

וּלְנֶפֶשׁ בְּרוֹם לָהּ מַעְלָה / נִקְלָה / מַה נַּעְשָׂה יְקָר וּגְדֻלָּה.





מוֹשֵׁךְ עוֹל / אֱמוּנָה וּבְרִית

תּוֹם יִסְבּוֹל / וְעוֹל שָׁוְא יַכְרִית

הוּא אֶל כָּל / תְּחִלָּה אַחְרִית

אַךְ סוֹד מַהֲלָלוֹ נִפְלָא / נַעְלָה / כִּי לוֹ רֹאשׁ וְאֵין לוֹ תִּכְלָה.





הֶאְדִּיר אֵל / בְּמֹשֶׁה מִשְׂרָה

דָּנִיאֵל / כְּפִיר הַתּוֹרָה

יוֹסֵף אֶל / חֲשֵׁכִים אוֹרָה

לוֹ אוֹר זָהֳרָם בַּאֲפֵלָה / נִגְלָה / חֹשֶׁךְ סָר וְשַׁחַר עָלָה.





תמת בעון אללה תעאלא



בָּרוּךְ נוֹתֵן לַיָּעֵף כֹּחַ וּלְאֵין אוֹנִים עָצְמָה יַרְבֶּה

מִשְׁקָל

שִׁמְעוּ זֹאת וְתָבִינוּ / וְתַנּוּ לָאֵל שִׁירָה

שִׁירוּ לוֹ בְּתֻשְׁבָּחוֹת / וּבַכִּנּוֹר וְקוֹל זִמְרָה

תְּיַשֵּׁר אִישׁ לֵאלֹהָיו / יְזַמְּרוּ לָאֵל הַנּוֹרָא.

תַּם.
