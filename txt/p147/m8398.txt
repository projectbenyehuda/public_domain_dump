


	
	מַלְכוּת שַדַּי
	




כָּל-הַטּוֹבִים אוֹתָךְ יְקַוּוּ

בְּכָל-לִבָּם לְתוֹרֵךְ יִתְאַוּוּ

זֶה-כַּמָּה אֶחֱרוּ פְעָמַיִךְ

וְרֶגַע לֹא-שְׁכַחוּךְ שֹׁאֲפָיִךְ!

אַךְ-אַתְּ תַּגִּיהִי אֲפֵלָתָם

וְתָשִׂימִי קֵץ לְתוּגָתָם;

מוּמֵי הַגְּוִיּוֹת תָּסִירִי

וּמַחֲלַת הַנְּפָשׁוֹת תַּעֲבִירִי!

עָרִיץ לֹא-יָעֵז לְמוּלֵךְ

וְשֹׁד לֹא-יִשָׁמַע בִּגְבוּלֵךְ

עָשׁוּק לֹא-יִבְכֶּה עַל-עשְׁקוֹ

וְשֹׁפֵט לֹא-יֶהְדַּר אֶת-נִשְׁקוֹ!

יֵעָלְמוּ שָׂרִים וְעָבָדִים,

יִכְלוּ עֲנָקִים וְגַמָּדִים;

יִנָּשְׂאוּ עֲמָקִים וּבְקָעוֹת,

יִשְׁפְּלוּ הָרִים וּגְבָעוֹת!

מַלְכֵּךְ לֹא-יִהְיֶה בֵּל,

כִּי-תְרִיחֵהוּ יִרְאַת אֵל;

רוּחַ חָכְמָה תְּפַעְמֵהוּ

וְאַהֲבַת יֹשֶׁר תַּנְחֵהוּ!

חֶסֶד וֶאֱמוּנָה דְּבָרוֹ,

כִּי-הַתּוֹרָה עֲנָק לְצַוָּארוֹ;

זָהָב וְצָבָא לֹא-יְרַבֶּה

וּפִשְׁתָּה כֵהָה לֹא-יְכַבֶּה!

אָדָם לָאָדָם לֹא-יִתְנַכֵּר,

בֵּין אֶזְרָח וָגֵר לֹא-יְבַקֵּר;

כָּל הַגְּבוּלִים וְהַחוֹמוֹת נֶהֱרָסוּ,

כָּל-הַמִּבְצָרִים וְהַקְּרִיּוֹת נִרְמָסוּ!

לֹא יִשְׁפְּטוּ-עוֹד חֶרֶב וָדָם

וְלֹא יִלָּחֵם-עוֹד עָם בָּעָם;

כָּל-הַגּוֹיִם שְׁקֵטִים וּשְׁלֵוִים

וְלִדְבַר אֲדֹנָי תְּאֵבִים!

לֹא-יִהְיֶה עוֹד נוֹשֶׁה צָר

וְנַחֲלַת-אִישׁ לֹא-יִירַשׁ זָר;

כִּי-בִּסְדָרַיִךְ שְׁמִטָּה וְיוֹבֵל

וְאָפֵס כָּל-נוֹגֵשׂ וְחוֹבֵל.

בְּכָל-הָאָרֶץ אוֹרֵך יוֹפִיעַ,

לְכָל-הָעַמִּים שָׁמְעֵךְ יַגִּיעַ;

כִּסְאוֹת עָרִיצֵיהֶם יְמַגֵּרוּ

וַעֲצַבֵּי יְדֵיהֶם יְשַׁבֵּרוּ!

עַל-דַּלְתוֹת יְרוּשָׁלַיִם יִשְׁקֹדוּ,

שְׁכֶם-אֶחָד אֶת-אֲדֹנָי יַעֲבֹדוּ;

בָּךְ יִלְמְדוּ תּוֹרַת-הַצֶּדֶק

לְבָעֵר גַּם-הֵם כָּל-חֶדֶק!

בַּכֹּל חָכְמָתֵךְ נִרְאָה

וּמָלְאָה תֵבֵל יִרְאָה;

בְּכָל-פִּנָּה מִשְׁפָּט וְתוֹרָה

בְּכָל-צַד אֱמוּנָה טְהוֹרָה!

הַצּוֹלֵעָה וְהַנִּדָּחָה תֵּאָסֵף,

כִּי-זְרוֹעַ הַקֹּדֶשׁ תֵּחָשֵׂף;

יֶחְדַּל מֵהָאָרֶץ אֶבְיוֹן,

כִּי-הַהַשְׁגָּחָה יָצְאָה מֵחֶבְיוֹן!

אַל-תִּסְתַּתְּרִי עוֹד תִּקְוָתֵנוּ,

אַל-תִּתְמַהְמְהִי עוֹד נֶחָמָתֵנוּ;

בֹאִי הוֹפִיעִי וַּדַּי,

קִרְבִי מַלְכוּת שַׁדַּי!


  

נדפס בשבועון “בת קול” גליון ל' לבוב תרע"ג.
