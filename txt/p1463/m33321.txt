


	
	אָנוּסָה לְעֶזְרָה
	








אָנוּסָה לְעֶזְרָה / אֶמְצָא נֶגְדִּי / אֵל קָרוֹב לִי / בְּעֵת קָרְאִי בְקוֹל

אֲשֶׁר בַּעֲדַת אֵל / בְּקִרְבִּי נִצָּב / וּפֹה בְּמִקְדַּשׁ מְעַט / אֲצַפְצֵף לוֹ בְקוֹל.





בַּקְרֵנִי דָּרְשֵׁינִי / שֶׂה פְזוּרָה אָנִי / נִגְזַזְתִּי וְנֶאֱלַמְתִּי / בְּלִי לְהָרִים קוֹל

בֶּאֱמוֹר גּוֹזְזַי: / נִדָּחָה הִיא / שׁוֹמְרָהּ וְצִלָּהּ / לֹא יִשְׁאַג קוֹל.





גַּלְתִּי שִׂיחַ בְּחֻקָּיו / וְחִכִּי עָרֵב / הִטָּה אֹזֶן וְשָׂח: / הַשְׁמִיעִינִי קוֹל.

גָּז וּבָרַח מֶנִּי / כְּעֹפֶר עַל הָרֵי בָתֶר / בְּבַקְשׁוֹ דָּת וָאוֹת / בְּמִשְׁכְּנוֹתַי וְאֵין קוֹל.





דִּלֵּג מִבֶּתֶר לְבֶתֶר / הֲשִׁיבֵהוּ אֵלַי / אוּלַי יִשָּׂא פָנֶיךָ / עֵקֶב שָׁמְעֲךָ בְּקוֹל

דְּרוֹשׁ טוֹבָה לָמוֹ / וּרְאֵה שֵׂה מוֹרִיָּה / אֵלֶם פִּיהוּ יְהִי צֶדֶק / לְלֹא שָׁמְעָה בְּקוֹל.





הָסֵר מֵחָלָק / יָדַיִם שְׂעִירוֹת / הוֹגָה בְתַחֲנוּנִים / כִּי לְךָ הַקּוֹל

הוֹשַׁע כִּי לֹא / תִשָּׁכַח עֵדוּת / מִפִּי זַרְעוֹ / לֹא יָסוּף קוֹל.





וְחוֹזַי וּמְלִיצַי / הֵם בְּנֵי אִמִּי / וְאָז נִיחֲרוּ בִי / לְמַעַן אֶשְׁמַע קוֹל

וְעַל מִשְׁמַרְתָּם / יַעַמְדוּ וְיִזְעָקוּ / וְגַל לָהֶם סוֹד / וְהוּא יַעֲנֵם בְּקוֹל.





זֶה חָמַק מֶנִּי / אֲסוֹבֵב וַאֲבַקְשֶׁנּוּ / בְּכָל-מָקוֹם הוּא / אָנָה אֶשָּׂא קוֹל

זֵכֶר דּוֹדִי לְמַטָּה / אַדִּיר בַּמָּרוֹם / מְלֹא כָל-הָאָרֶץ כְּבוֹדוֹ / לְמַעְלָה קוֹרְאִים קוֹל.





חֲיָלֵי מַיִם / אֲשֶׁר בָּם שְׁבִילוֹ / דִבְּרוּ: תְּמוּנָה לֹא רָאִינוּ / זוּלָתִי קוֹל

חִפַּשְׂתִּי יְשִׁימוֹן / הֲיֵשׁ, וְאָמַר: אָיִן, / קֶדֶם תִּתּוֹ עוֹז / בִּעֲתַתְנִי בְּקוֹל.





טָהוֹר דִּלֵּג / הָרִים וְעָבַר / וּמִמְּעוֹ הַר מוֹר / אָז נָתַן קוֹל

טִמֵּאתִי יְדִידוּת / שִׁכְנוֹ וְעָלָה / לְיוֹם כֶּסֶא יָבוֹא / בְּאָזְנַי קוֹל.





יָקַרְתִּי בְּעֵינָיו / וְנִלְוָה לִי בַּשֶּׁבִי / עִמּוֹ אָנֹכִי / אָז הִבְטִיחַנִי בְּקוֹל

יָרַד בְּשִׁנְעָר / וְשָׂם כֵּס בְּעֵילָם / וְהִשְׁמִיעַ כַּאֲרִי / וּכְנָחָשׁ קוֹל.





כִּלָּה מֶנִּי דוֹב / כְּהִתְרַפֵּא מִמְּלֶאכֶת / חַק בְּמִכְתָּב / וַיַּעֲבֶר קוֹל

כִּבֶּשׁ לִי / אַרְבָּעָה רָאשֵׁי נָמֵר / וְגַם אֲנִי בְּהוֹדָיוֹת סֶלָה / אַשְׁמִיעַ לוֹ בְּקוֹל.





לְחַיַּת קָנֶה / אָז מָכַר אָרֶץ / וּמִי לִי בַשָּׁמַיִם / אָז הֵרִימָה קוֹל

לֵאלֹהֵי יִשְׁעִי / מִשִּׁנֵּי בַרְזֶל שִׁוַּעְתִּי / וּמֵרֶגֶל עַב טִיט / הִצְרַחְתִּי קוֹל.





מִדַּת קִצִּי / לֹא הוֹדִיעַנִי / מָתַי בְּאַרְצִי / תּוֹר יַשְׁמִיעַ קוֹל

מִיּוֹדְעֵי סֵפֶר / סָתַם קִצִּי / לְבַל דַּעַת צוֹפַי / עֵת יִשְּׂאוּ קוֹל.





נָא הַבֵּט וּרְאֵה / עָנְיִי וּמְרוּדִי / אֵין לִי מַכִּיר / לְמִי אֶשָּׂא קוֹל

נֶצַח אֲקַוֶּה / כִּי לֹא יִפּוֹל דָּבָר / מִמַּקְשִׁיבֵי / דְּמָמָה וָקוֹל.





שׂוֹשׂ יָשִׂישׂ / לִבִּי בְּקִרְבִּי / בְּשָׁמְעִי דּוֹדִי דוֹפֵק / עַל פְּתָחַי קוֹל

סֶלָה יְשִׂימֵנִי / כַּחוֹתָם עַל לֵב / כְּאָז תַּחַת הַתַּפּוּחַ / עוֹרְרַנִי בְּקוֹל.





עִלִּיתַנִי אֵל / עַל כָּל-בָּנוֹת / כִּי בַעֲבוּרִי / בְּחוֹרֵב תַּתָּה קוֹל

עַל כָּל-אֱלֹהִים / מְאֹד נַעֲלֵיתָה / נֶצַח תִּתְעַלֶּה / בִּתְרוּעַת קוֹל.





פָּעָה מִמִּדְבָּר / צִפּוֹר מִמִּצְרַיִם / וְיוֹנָה הִשְׁמִיעָה / מֵאַשּׁוּר קוֹל

פְּקוֹד צִפּוֹר בַּיִת / דְּרוֹשׁ יוֹנַת אֵלֶם / תְּקַע לָמוֹ בְּשׁוֹפָר / וּשְׁרוֹק לָמוֹ בְּקוֹל.





צוֹר חֻקִּים, מֶנִּי / לְבַל יָעוּפוּ כַנֶּשֶׁר / וּבַל יִכָּנְפוּ / מַשְׁמִיעֵי קוֹל

צְרוּפָה אֶלְמַד / וְעֵינַי לְמוֹרַי / כְּאָז צִיר מְדַבֵּר / וְאֵל מְשִׁיבוֹ בְּקוֹל.





קָרַב קֵץ / בָּא עֵת מִשְׁפָּט / קָם מֵלִיץ יוֹשֶׁר / לְהִתְחַנֵּן בְּקוֹל

קֻדַּשׁ חֹדֶשׁ / וְהוּכַן מוֹעֵד / אֶתְקַע בְּשׁוֹפָר / וְיַעֲנֶה לִי בְּקוֹל.





רֶגֶשׁ מִקֶּבֶר / צְוָחָה מִסֶּלַע / בְּתֵת יְבֵישֵׁי עֶצֶם / מֵעָפָר קוֹל

רְאוּ נֵס בֶּהָרִים / וְקוֹל שׁוֹפָר בָּאָרֶץ / לְהַשְׁמִיעַ רֶנֶן / מִדְּמוּמֵי קוֹל.





שֶׁנֶג לֵב הוּתַל / עוֹד בַּל יַטֶּנוּ / בְּלִי לְהַכְבִּיד אֹזֶן / מִשְּׁמוֹעַ בְּקוֹל

שׁוֹבֵב לִי כְּקֶדֶם / דָּת מוֹרָשָׁה / אֲשֶׁר בָּהּ עִלְּפוּנִי / בְּלַפִּידִים וָקוֹל.





תְּבוּנָה הֵפִיק / אִישׁ נְבוֹן דָּבָר / וְחַק נֹעַם זְמִירוֹת / בִּנְעִימַת קוֹל

תְּהִלָּה יְתַנּוּ / אָז לַכֹּל הִשְׁמִיעוּ / לְאֵל מוֹשֵׁל בַּכֹּל / יַמְתִּיקוּ בְקוֹל.





תַּחַת בְּנֵי צִיּוֹן / בְּנֵי יָוָן שַׁחוּ / הִבְרַקְתָּ חִצִּים / וַתְּהֻמֵּם בְּקוֹל

תַּרְעֵם לְבוֹזְזַי / תִּתְקַע בַּשּׁוֹפָר / בְּסַעֲרוֹת תֵּימָן / אָז יֵלֶךְ קוֹל.
