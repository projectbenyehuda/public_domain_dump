


	
	אֶפְחַד בְּמַעֲשַׂי
	








אֶפְחַד בְּמַעֲשַׂי / אֶדְאַג בְּכָל-עֵת / אִירָא בְּיוֹם דִּין / בְּבוֹאִי לְזִכָּרוֹן.

אֶדְרוֹשׁ לְחַנּוּן / אֲחַלֶּה לְרַחוּם / אֲחַנֵּן לְחַק לִי / יוֹם לְזִכָּרוֹן.





בְּבוֹאִי לַמִּשְׁפָּט / בְּמִי אֶשְָׁעֵן? / וּמִי יְחַפֶּשׂ-לִי / צֶדֶק לְזִכָּרוֹן?

בָּאָבוֹת בָּטַחְתִּי / וּפָעֳלָם אָכַלְתִּי, / הֵם הָיוּ לִי / קֶדֶם לְזִכָּרוֹן.





גָבְרָה זְרוֹעִי / כְּשָׂח: מְחֵינִי נָא, / לְבַל יִמַּח מֶנִּי / שֵׁם וְזִכָּרוֹן.

גֶּבֶר אִם יַעֲמוֹד / לְפָנָיו הֲיוֹעִיל / בְּעֵת יְבֻקַּשׁ מֶנִּי / זְכוּת לְזִכָּרוֹן.





דַצְתִּי בְּלוֹבֵשׁ אֵפוֹד / וְחוֹשֶׁן הַמִּשְׁפָּט / אֲשֶׁר בָּם הוּחַק / שְׁמִי לְזִכָּרוֹן.

דִּלֵּג בַּמַּחְתָּה / עַד יֻעֲצַר נֶגֶף / בְּגֶשֶׁת זָר מְכַהֵן / יָקוּד לְזִכָּרוֹן.





הַבִּיטָה אֵל / בְּעָמְדִי לְפָנֶיךָ / אֵין בְּקִרְבִּי אֱנוֹשׁ / תָּוִי לְזִכָּרוֹן.

הֲיֵשׁ מִי יְפַלֵּל / וְיָשִׁיב חֵימָה / וְיֻחַק לַדּוֹרוֹת / שְׁמוֹ לְזִכָּרוֹן.





וּמִי שׂוֹנֵא בֶצַע / יְדַבֵּר: עֲנוּ בִי, / וְיַעַן וְיֹאמַר: עֵד / אָדוֹן לְזִכָּרוֹן.

וּבִטְלֵה חָלָב / יְכַפֵּר בַּעֲדֵינוּ / וּכְנֶגֶד שְׁנֵי עֳפָרִים / יַעַל לְזִכָּרוֹן.





זַעַף וְיֵרֶא, וְיֹאמַר: / תְּהִי יָדְךָ בִּי, / אֲשֶׁר כְּלֵב אֱלֹהָיו / חָשׁוּב לְזִכָּרוֹן.

זָעַק וְהֵשִׁיב / חֶרֶב אֶל נְדָנָהּ / וְהוּשַׁת לוֹ כַּשֶּׁמֶשׁ / כֵּס לְזִכָּרוֹן.





חִכִּיתִי בְּעֵת עָצֶר / לְפוֹתֵחַ וּמַמְטִיר / כְּמֵשִׁיב רוּחַ לַיֶּלֶד / אָבוּד מִזִּכָּרוֹן

חַי לִרְאוֹת בְּרִית / בְּדַבְּרוֹ: קִנֵּאתִי, / כִּי עָזְבוּ עָם / בְּרִית לְזִכָּרוֹן.





טֶרֶם הָיָה לִי / מְכַפֵּר פָּנִים / מִנְחָה הוֹלֶכֶת / כַּשּׁוֹחַד לְזִכָּרוֹן.

טֶרֶף נֵרְדְּ וְסַמִּים / לְמֵיטַב חֲדָרָיו / דָּם וָחֵלֶב לְנִיחֹחַ / וְלֶחֶם לְזִכָּרוֹן.





יֻצַּגְתִּי עַל / גֶּחָלַי רֵיקָה / כִּי לֹא אַלְמָן / שַׁתִּי לְזִכָּרוֹן.

יָהּ אֶבְטַח בָּךְ / וְלֹא בִנְדִיבִים / כִּי הֵם בַּקֶּבֶר / וְלָנֶצַח שִׁמְךָ לְזִכָּרוֹן.





כָּל אֵלֶּה סְמָכוּנִי / וְרַחֲמֶיךָ בִּקַּשְׁתִּי / לוּלֵי הֵם תַּמְתִּי / וְאֵינִי לְזִכָּרוֹן.

כִּי הֵם בִּזְרוֹעַ / עָדֶיךָ לֹא בָאוּ / רוֹמְמֹתֶיךָ בְּפִיהֶם / שָׂמוּ לְזִכָּרוֹן.





לְיוֹם זֶה נִכְמָס / סֶכֶם חֶשְׁבּוֹנוֹת / תְּחִלָּה לְיָמִים / וְרֹאשׁ לְזִכָּרוֹן.

לְהִקָּרֵא בוֹ / כְּתָב עֵט וְשָׁמִיר / גָּלוּי וּבָאֵר / וְיָדוּעַ לְזִכָּרוֹן.





מָוֶת וְחַיִּים / שָׁלוֹם וּמִלְחֶמֶת / צַחְצָחוֹת וָשׂוֹבַע / בָּאוּ לְזִכָּרוֹן.

מַעַלְלֵי גֶבֶר / וּמִסְפַּר צְעָדָיו / נִשְׁכְּחוּ מֵאֱנוֹשׁ / וְלָאֵל לְזִכָּרוֹן.





נִסְתְּרָה דַרְכִּי, / מִי יוּכַל שִׂיחַ, / לַשָּׁוְא נִכְתַּב לִי / חֵטְא לְזִכָּרוֹן.

נֶגֶד פְּנֵי גֶבֶר / מַעֲשָׂיו יוֹכִיחוּ / וְיַעֲנֶה כַחֲשׁוֹ / עֵד לְזִכָּרוֹן.





שִׂיחוּ מְזִימוֹת אֵל / יַחַד כָּל בְּנֵי אִישׁ / עוֹבְרֵי תַחַת שֵׁבֶט / כַּצֹּאן לְזִכָּרוֹן.

סוֹגֵר דֶּלֶת / בְּעַד תְּמִימִים בְּזַעַם / עַד בּוֹא קִצָּם / צֵאת לְזִכָּרוֹן.





עֲשֵׂה פֶלֶא לַחַיִּים / לְבַל יְהוּ כַמֵּתִים / הֲיֵשׁ אֲמִתֶּיךָ / בַּקֶּבֶר לְזִכָּרוֹן.

עוֹרַרְתָּה אָז בְּפוֹעַל / עוֹרְרֵנוּ בְּלֹא פוֹעַל / הֲלֹא לְנִפְלְאוֹתֶיךָ / תַּעַשׂ לְזִכָּרוֹן.





פְּנֵה אֱלֹהִים / בְּיוֹשְׁבֵי גַנִּים / מַקְשִׁיב לְנִדְבְּרֵי / בְּדָת לְזִכָּרוֹן.

פָּעֳלָם לְפָנֶיךָ / וּשְׂכָרָם אִתָּךְ / אֹכְלֵי לֶחֶם הָעֲצָבִים / בְּסֵפֶר זִכָּרוֹן.





צָצוּ שׁוּעָלִים / מְחַבְּלִים כְּרָמִים / לְהַכְרִית מִגֶּפֶן / שׁוֹרֶשׁ וְזִכָּרוֹן.

צְרָרוּם בְּפֶרֶךְ / נָאֲקוּ וְנוֹשָׁעוּ / בְּכוֹשֶׁר הַרְרֵי קֶדֶם / הוּחַק לְזִכָּרוֹן.





קֶדֶם בַּנְתָּה לַדּוֹרוֹת / וְאֵין בָּהֶם חֵפֶץ / חִלַּפְתָּם וְאִבַּדְתָּם / מִהְיוֹת לְזִכָּרוֹן.

קַחְתָּה דּוֹר מֵאֶלֶף / אֲמָרֶיךָ הִנְחַלְתָּם / לְמַעֲנָם בְּכָל דּוֹר / חַקְתָּה לְזִכָּרוֹן.





רָם חָשַׁק מְאֹד / בְּכַלַּת נְעוּרִים / הֲרוּגֶיהָ וַעֲנִיֶּיהָ / בָּאוּ לְזִכָּרוֹן.

רָצָה אַחֲרֶיךָ / בְּגֵיא צִיָּה וְצַלְמָוֶת / אַהֲבַת כְּלוּלוֹתֶיהָ / תַּעַשׂ לְזִכָּרוֹן.





שִׁיחֲתוּ עַם אֵל / שְׂאֵת שִׁמְצָה בְּקָמֵיהֶם / שֵׁם פֶּסֶל וְלֹא אֵל / שָׁתוּ לְזִכָּרוֹן.

שֻׁכָּלוּ כִּמְעַט רֶגַע / לוּלֵי קָם בַּפֶּרֶץ / מְעוֹרֵר שְׁבוּעוֹת / בְּרִית זִכָּרוֹן.





תִּיעַבְתָּה מֵאָז / עֲדַת כָּל אֻמִּים / חֲשַׁקְתָּנוּ מֵהֶם / עֵדוּת לְזִכָּרוֹן.



תְּמוּר כֶּסֶף נִמְאָס / דּוֹר נִשְׁכָּח קַחְתָּה / קְנֵינוּ שֵׁנִית / כִּי שֻׁכַּחְנוּ מִזִּכָּרוֹן.





תָּר אִישׁ תָּם / בְּמִי זַרְעוֹ יְכֻנֶּה / שִׂכֵּל יָדָיו / לְאוֹת לְזִכָּרוֹן.

תְּרַפֵּק מְשׁוּל אֶפְרַיִם / בְּשַׁעֲשׁוּעַ יֶלֶד / הֲלֹא בְּכוֹרִי הוּא / חַקְתָּה לְזִכָּרוֹן.
