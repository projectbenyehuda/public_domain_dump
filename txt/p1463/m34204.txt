


	
	אֶכְרְעָה וְאֶבְרְכָה
	








אֶכְרְעָה וְאֶבְרְכָה / אַעַן לָשׁוֹן רַכָּה / אָקוּמָה ל…. / אֶתְיַצֵּב כְּשָׁפָל וְדַכָּה.

בְּפִיפִיּוֹת מֵישָׁרִים / ב…… / בִּשְׂפָתוֹת מְיֻשָּׁרִים / בְּחֵן עֲשָׂרָה הַסְּדָרִים.

ג….. אֲפַלֵּל / גֵּאוּת לֹא אֲמַלֵּל / גְּרוֹנִי קוֹל יְהוֹלֵל / ג… וְעוֹלֵל.

דַּעַת יָקָר / דְּבַר גְּדוֹלוֹת יְקָר / דּוֹק…. / [אָז?] בְּדַעַת חָקַר.

הִצִּיב רָקִיעַ / הִכְפִּיף וְהִרְקִיעַ / הִבְהִיק וְהִבְקִיעַ / הוּא יָסַד וְהִשְׁקִיעַ

וַאֲבָרֵךְ בְּמַקְהֵלוֹת / וְאַבִּיעַ תְהִלּוֹת / וּלְתוֹכֵן עֲלִילוֹת / אֲסַפֵּר גְּדוֹלוֹת.

זֵכֶר נִפְלָאוֹת / זֶבֶד לְהַנְאוֹת / חָשׁוֹת קֵץ הַפְּלָאוֹת / חַי זָכְרָה לִי לְטוֹבָה אוֹת.

טוֹב אֲיָּלוֹת / טַע פְּעָמַי כְּאַיָּלוֹת / טַעֲמֵינִי בְּשִׁבְעָה קוֹלוֹת / אֹמֶץ גְּדוֹלוֹת.

יְקָר בְּצָרְחִי / יְחִי סֶלָה רוּחִי / יִגֶל בַּמּוּסָר רוּחִי / יֶעֱרַב עָלָיו שִׂיחִי.

כְּהַבִּיט אֵל / כְּהִתְוַדּוֹתִי יִשַׁע אֵל / כְּשָׁפָל אֲאַשֵּׁר לָאֵל / אֲרוֹמֵם לָאֵל.

לִבִּי אֲפַנֶּה / לְרַחֲמֵנִי יִפְנֶה / לַעֲרֹךְ שֶׁוַע בְּמַעֲנֶה / טֶרֶם אֶקְרָא וְיַעֲנֶה

מִלּוּלִי הוֹרוֹת / מִפִּי בְמַהֲמוּרוֹת / מְבַשֵּׂר טוֹב בְּתוֹךְ חֲבוּרוֹת / אַזְכִּיר גְּבוּרוֹת.

נֹעַם אֶקְרַב / נְגִינוֹתַי יֶעֱרַב / נוֹרְאוֹת מֶלֶךְ רָב / אֲבַשֵּׂר בְּקָהָל רָב.

שִׁיר אֲסַלְסְלָה / שִׂמְחָה וְגִילָה / סוֹד רַב עֲלִילָה / אֶזְכֹּר סֶלָה.

עוֹד אֲלַבְלֵב / עֲרוֹךְ בְּפֶה [וּבְלֵב / ע]… נוֹתֵן לֵב / לְאָדָם מַעַרְכֵי לֵב.

פָּתַחְתִּי בִּינִיתָה / פְּנִינוֹת גּוּף הֵכַנְ[תָּה / פ…] קְרָבַי בָּחַנְתָּה / אַתָּה כּוֹנַנְתָּה.

צָרַפְתָּה לִבִּי / צוּר בָּחַנְתָּה קִרְבִּי / צָפַנְתָּה לָךְ אִבִּי / רוּחַ יְיָ דִּבֶּר בִּי.

קוֹלוֹת יְחוֹנֵינוּ / קֹוֶה אֵל יְכוֹנֵנוּ / קִרְבַת אֱלֹהִים בְּתוֹכֵינוּ / אַדִּיר אֲדוֹנֵינוּ.

רְצֵה קוֹלִי / רְאוֹתָךְ פִּלּוּלִי / רָם יְיָ חֵילִי / אֲדַבְּרָה וְיִרְוַח לִי.

שְׁעֵה בְּדַבְּרִי מֵישָׁרִים / שִׁלּוּם שְׂפָתַי פָּרִים / שׁוֹפְכִים חֲדָשִׁים לַבְּקָרִים / אַדִּיר בָּאַדִּירִים. תִּכַּנְתָּ כֹל / תּוֹמֵךְ וּמוֹשֵׁל בַּכֹּל / תַּחְבּוּלוֹתֶיךָ לְרַחֵם כֹּל / תּוֹכֵן כֹּל.

תּוֹכֵן וְגַם יָכוֹל / תִּכַּנְתָּה בְּמִדָּה כֹל / תּוֹשִׁיעַ עוֹלָמִים לִשְׂכֹּל / אָז בְּאֵין כֹּל.

יַחַד תְּהִלּוֹת אֲדַבֵּר / עֲרוֹךְ….. בר / [ק]וֹל לֹא אַרְחִישׁ מִלְּדַבֵּר / בְּעָמְדִי … בר.
