


	
	שִׁיר אִכָּרִים
	



עַל אַדְמַת הַקֹּדֶשׁ

שִׁיר בֹקֶר


אחד



נַשְׁכִּימָה אִכָּרִים,

לַעֲבוֹדָה בֶאֱמוּנָה;

נַעֲלֶה עַל הֶהָרִים,

שָׁם רַבָּה הַתְּכוּנָה.

מַחֲרֵשָׁה, מַזְמֵרָה,

תֶּלְיֵנוּ קַשְׁתֵּנוּ;

אֵת, מַגָּל וּמְגֵרָה

נֵשֶׁק גַּאֲוָתֵנוּ!



מקהלה



בְּזֵעַת אַפָּיִם,

נַעֲשֶׂה הַמְּלָאכָה;

וּמְטַר שָׁמָיִם

יָבִיא הַבְּרָכָה;

כְּבָנִים נֶאֱמָנִים,

עַל שְׁדֵי אִמֵּנוּ,

לֹא בְּבֹשֶׁת פָּנִים,

נֹאכֵל לַחֲמֵנוּ!



אחד



אֲדָמָה פוֹרִיָּה

אֵם טוֹבָה נֶאֱמֶנֶת;

לְעוֹבְדֶיהָ בְלֹא רְמִיָּה,

אֵת יְבוּלָהּ נוֹתֶנֶת;

לְמִתְרַפִּים עֲצֵלִים,

לֹא תַחְלוֹץ שָׁדֶיהָ;

לַחֲרוּצִים עֲמֵלִים,

הִיא תָאִיר פָּנֶיהָ.



מקהלה



בְּחֵפֶץ כַּפָּיִם,

נַעֲבוֹד בַּכְּרָמִים,

וְיַיִן כַּמַּיִם,

נִשְׁתֶּה כָל־יָמִים;

כְּבָנִים נִכְבָּדִים,

לְאֵם רַחֲמָנִיָּה,

לֹא כְגֵרִים נָדִים

בְּאֶרֶץ נָכְרִיָּה!



אחד



בִּתְפִלָּה וַעֲבוֹדָה

מַה־נָּעִים חֶלְקֵנוּ,

וּבְרִנָּה וְקוֹל תּוֹדָה,

נַנְעִים אֵת פָּעֳלֵנוּ;

לֹא קִנְאָה לֹא אֵיבָה,

בְּקִרְבֵנוּ שׂוֹרֶרֶת;

לֹא גַאֲוָה מַדְהֵבָה,

רוּחֵנוּ עוֹכֶרֶת!



אחד



פֹּה הַדְרַת הַטֶּבַע,

תַּאֲוָה לָעֵינָיִם;

עַל כָּל־הַר וָגֶבַע,

אַךְ בִּרְכוֹת שָׁמַיִם;

רֵיחַ שָׂדֶה נָעִים,

מַבְרִיא הָאֵבָרִים,

לֹא אֲוִירִים רָעִים,

מַשְׂחִיתִים יְקַר כְּרִים!



מקהלה



שָׂמַיִם זַכִּים,

מְאִירִים עֵינָיִם,

וּמַיִם מְפַכִּים,

מָנָה אַפָּיִם,

לָנוּ הֵם לְמָנָה,

בְּיָד נְדִיבָה מְלֵאָה,

כָּל־יְמֵי הַשָּׁנָה,

בְּלֹא רוּחַ נְכֵאָה!



אחד



הַחַיִּים הַמָּרִים,

בְּבַטָּלָה וּרְמִיָּה;

בִּמְעוֹנוֹת הֶעָרִים,

שָׁם בְּאֶרֶץ נָכְרִיָּה;

לָנוּ הֵם זָרִים,

בְּחַיֵּינוּ הַתְּמִימִים;

בְּאָהֳלֵינוּ בַכְּפָרִים,

בַּעֲבוֹדָה בַנְעִימִים.



מקהלה



פֹּה חַיֵּי יְשָׁרִים,

נְבַלֶּה כָל־יָמִים;

בְּמִשְׁפָּט וּסְדָרִים,

חַיִּים וְקַיָּמִים!

יְגִיעָה וְהַנָּאָה,

לָנוּ חֲבָלִים;

דְּאָגָה וּתְלָאָה,

מְנַת גּוֹרַל נְבָלִים.



אחד



מָה חַיֵּי עֲצֵלִים,

אֲצִילִים גֵּאוֹנִים;

מִתְכַּבְּדִים וּבְטֵלִים,

אֲדוֹנִים אֵין אוֹנִים.

הָאוֹכְלִים מַעֲדַנִּים,

פְּרִי עֲמַל אֲחֵרִים,

וּמְשַׂחֲקִים מִשְׁמַנִּים,

בִּמְסֻבַּת חֲבֵרִים.



מקהלה



חֲיֵּי בַּטָּלָה,

מַחֲלָה מְחַבֶּלֶת

רֶשַׁע וּנְבָלָה

זִמָּה מְנַוֶּלֶת;

תַּאֲוָה מַמְאֶרֶת,

חוֹלַת הַמְּדִינָה,

פֹּה לֹא שׂוֹרֶרֶת,

בְּטֶבַע עֲדִינָה.



אחד



הוֹצָאָה מְפֻזֶּרֶת,

עַל צְרָכִים לֹא צְרִיכִים,

בִּתְשׁוּקָה עוֹכֶרֶת,

לְתַפְנוּקֵי הַנְּסִיכִים.

חֲדָרִים מְרֻוָּחִים,

מְלֵאִים כֵּלִים שׁוֹנִים;

לִפְרָקִים נִפְתָּחִים,

לְקַבֵּל פְּנֵי אֲדוֹנִים.



מקהלה



פֹּה אֵין מוֹתָרוֹת,

מְקַצְּרִים הַשָּׁנִים;

מַלְבִּינִים שְׂעָרוֹת,

מַשְׁחִיתִים פָּנִים.

פֹּה צְרָכִים דְּרוּשִׁים,

טוֹבִים מַבְרִיאִים,

לְהֵחָלֵץ חוּשִׁים,

אֲנַחֲנוּ מְבִיאִים.



אחד



מְפֻנָּקִים מִתְיַפִּים

עַל חֶשְׁבּוֹן חֵיל זָרִים

עֲצֵלִים מִתְרַפִּים

כַּנָּשִׁים כַּגְּבָרִים;

אֶבְיוֹנִים גֵּאיוֹנִים

בְּהַכְנָסָה מְצֻמְצָמָה,

מִתְגָּאִים כַּאֲדוֹנִים,

כַּעֲשִׁירִים בְּיָד רָמָה.



מקהלה



עָרְמָה וּרְמִיָּה,

זִמָּה וּנְבָלָה,

עוֹשִׂים בִּצְדִיָּה

שָׁם לַבֶּהָלָה;

לֹא יַחְרְכוּ צֵידָם,

גְּזֵלַת חֵלְכָּאִים,

פִּתְאֹם יוֹם אֵידָם,

וַיֵּרְדוּ פְלָאִים.



אחד



שָׁם חֶבְרוֹת מַקְהֵלוֹת,

מְלֻמְּדֵי מִשְׂחָק יְדוּעִים

בַּיָּמִים בַּלֵילוֹת,

עַל מִשְׂחַק תַּעְתּוּעִים;

מִלְחָמָה דוּמִיָּה,

כְּלֵי חָמָס בִּדְמָמָה;

גָּנֹב גֵּזֶל, רְמִיָּה,

הַמָּה כְלֵי מִלְחָמָה.



מקהלה



יְגִיעַ כַּפָּיִם,

מְנַת תְּמִימֵי דֵעִים,

אֹשֶׁר אַפָּיִם,

לֹא יֵדְעוּ מְרֵעִים.

פֹּה יַד אֱמוּנָה,

הִיא תַעֲשֶׁה חָיִל,

בְּתוֹרָה וּתְבוּנָה,

יוֹמָם גַּם לָיִל.



אחד



שָׁם בָּנִים כֶּחָשִׁים,

עִוְּרִים מִבְּרַק אוֹרָה;

וּבְנְזִיד עֲדָשִׁים,

מוֹכְרִים אֵת הַבְּכוֹרָה;

לַעֲמוֹד בֵּין אַנְשֵׁי שֵׁם,

מְחַקִּים חֻקִּים זָרִים,

וּבוֹזִים לִיקְהַת אֵם.

לֹא שֵׁם לָהּ בַּשְּׁעָרִים.



מקהלה



אֲנַחְנוּ כֵנִים,

עוֹסְקִים בַּתּוֹרָה,

וּבְטַעַם זְקֵנִים,

לָנוּ רַב אוֹרָה.

שְׂפָתֵנוּ בְרוּרָה,

עִבְרִית אִמֵּנוּ;

עֲרוּכָה, שְׁמוּרָה,

בְּפִינוּ, בְּלִבֵּנוּ.



אחד



שָׁם חֵלֶק שְׂפָתָיִם,

וּפָנִים כֶּחָשִׁים;

תָּם וּנְקִי כַפָּיִם,

הוּא חֶרְפַּת אֲנָשִׁים.

עַל תַּחֲרוּת וְהוֹנָאָה

הַחֶבְרָה קַיֶּמֶת,

הַקָּפָה וְהַלְוָאָה,

סְחַר כֻּלָּם מְצַמְצֶמֶת.



מקהלה



פֹּה פְרִי הָאֲדָמָה,

יְגִיעַ כַּפָּיִם;

יְחַיֶּה כָל־נְשָׁמָה,

בִּרְכַת שָׁמָיִם.

בְּפָנִים נֶאֱמָנִים,

וְלֵב אֱמֶת וְתָמִים;

פָּנִים אֶל פְּנִים

נְדַבֵּר כָּל־יָמִים!



כלם



כֵּן זֶרַע זָרוּעַ

עַל אֶרֶץ נָכְרִיָּה

הוּא מָשְׁחָת פָּרוּעַ

לֹא יַעֲשֶׂה תוּשִׁיָה.

אַשְׁרֵינוּ אִכָּרִים,

עַל אֶרֶץ נָעֵמָה,

פֹּה עַל יְקַר הַכָּרִים,

יָדֵנוּ רוֹמֵמָה!
