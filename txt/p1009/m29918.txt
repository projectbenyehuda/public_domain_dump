


	
	פְּתַח פִּי בְנִהְיָה
	








פְּתַח פִּי בְנִהְיָה / וְקִינָה וּבְכִיָּה / לְהוֹלְכִים בַּשִּׁבְיָה / בְּמַלְכוּת הַהַגְרִים

וְאֶבְחַר לִי קִינוֹת / בְּעִנְיָנִים שׁוֹנוֹת / וְאֶעֱזֹב לִנְגִינוֹת / וּמַה לִּי עוֹד שִׁירִים

וְאַגִּידָה פְלָאִים / לְדוֹרוֹת הַבָּאִים / וּמוּסָר לִפְתָאִים / וּמְזִמּוֹת לִנְעָרִים

יְגוֹנִי וְהוּא נוֹטֶה / שְׁנָתוֹ כְשַׁ״טֶה / וְעַל שָׂפָם אֶעְטֶה / כְּאֵבֶל לִבְכוֹרִים

יְמוֹתַי לָעֲבוֹדָה / וְלֵילוֹתַי נִידָה / שְׁנָתִי נָדָדָה / וְחַיַּי נֵאָרִים

דִּמְעוֹתַי אֶשְׁאָב / וְגוּפִי הוּא נִכְאָב / בְּעֵת כִּי נוֹלַד אָב / לְכָל הַנִּזְכָּרִים

לְכָל צָרָה הוּא רֹאשׁ / וּמִקְדָּשׁוֹת לַחֲרֹשׁ / וְהִשְׁקַנִי מֵי רוֹשׁ / וְלַעֲנָה וּמְרוֹרִים

וְצָר קִוָּה לְהַשְׁמִיד / לְזָקֵן עִם תַּלְמִיד / וּבוֹ עֵינַי תָּמִיד / בְּדִמְעָה נִגָּרִים

לְבָבִי אַרְגִּיעָה / בְּקִינוֹת אַבִּיעָה / לְזֹאת לֹא נִשְׁמָעָה / בְּכָל הַמַּחְזוּרִים

עֲלֵי עָם וּקְהִלָּה / שְׁבוּיִים בַגּוֹלָה / בְּהַסְכָּמַת מַעְלָה / וְשִׁבְעִים הַשָּׂרִים

מְדֻכִּים וּמְעֻנִּים / וְהוֹגִים כַּיּוֹנִים / בְּמִסְפֵּד כַּתַּנִּים / בְּלִבּוֹת נִשְׁבָּרִים

וְנוֹשְׂאִים לַסֵּבֶל / מְעֻנִּים בַּכֶּבֶל / וְגוֹדֶל הָאֵבֶל / בְּחָרְבַּן הֶעָרִים

וְלֶחֶם לֹא נִמְצָא / בְּבַיִת וּבְחוּצָה / וְקָרְאִי לֹא נִרְצָה / דְּמָעַי נִגָּרִים

קְהִלּוֹת נֶעְרָכִים / בְּכָל רֶגַע בּוֹכִים / בְּלִבּוֹת נִשְׁפָּכִים / וְשׁוֹתִים מֵי מָרִים

פְּזוּרִים בַּמַּעְרָב / לְקֶרַח וּלְשָׁרָב / וְכָל נֶגַע יִקְרָב / בְּרָאשֵׁי מִבְצָרִים

וְכַמָּה יָקָר סַף / לְפַת לֶחֶם נִכְסַף / בְּגֵאָיוֹת נֶאֱסַף / וְלֹא בָא לִקְבָרִים

וּמֵהֶם נוֹדָעוּ / וְכַעִוְרִים נָעוּ / בְּמִבְצָר גָּוָעוּ / לְעֵינֵי הַצָּרִים

וְשׁוֹכְנֵי מַשְׂכִּיּוֹת / כְּיוֹנֵי גֵאָיוֹת / לְקֵדָר וּנְבָיוֹת / בְּחִנָּם נִמְכָּרִים

וּמֵצַח עוֹפֶפֶת / לְבַקֵּשׁ טוֹטֶפֶת / הֲלֹא הִיא נְהְפֶּכֶת / לְכָל הַמִּשְׁבָּרִים

וְעַיִן דּוֹלֶקֶת / לְאַשְׁפָּה חוֹבֶקֶת / בְּכָל יוֹם צוֹעֶקֶת / תְּרוּעָה וּשְׁבָרִים

וְלָשׁוֹן לוֹ יִשְׁפָּר / בְּחָכְמָה אֵין מִסְפָּר / יְלַחֵךְ הֶעָפָר / עֲרָבִים וּשְׁחָרִים

וּפָנִים מַבְהִיקִים / לְעֵדוֹת וּלְחֻקִּים / מְכֻסִּים בַּשַּׂקִּים / בְּאַשְׁמָה מֻשְׁחָרִים

וְאָזְנַיִם נִגְעָם / בְּעֵת אוֹיֵב יִזְעָם / וְאֵיךְ זָהָב יוּעָם / וּפָנָיו נִכָּרִים

וְיָדַיִם עוֹרְכוֹת / לְהַפֵּךְ בִּמְלָאכוֹת / לְרֵיחַיִם מוֹשְׁכוֹת / וּבָהֶם נִקְשָׁרִים

וְרַגְלַיִם תֵּלֶךְ / לְמִצְוַת הַמֶּלֶךְ / לְעָשִׁיר וּלְחֵלֶךְ / בְּכֶבֶל נֶעְצָרִים

וְנֶפֶשׁ נוֹסֶפֶת / יְקָרָה נִכְסֶפֶת / וְעַתָּה נֶחְרֶפֶת / בְּאַחַד הַצּוּרִים

לְבָבוֹת נִקְרָעוּ / וּמָתְנַיִם זָעוּ / נְגִידִים הָטְבָּעוּ / בְּיַד הָאַכְזָרִים

וְאֵיךְ אֶמְצָא נֹפֶשׁ / אֲנַחֵם הַנֶּפֶשׁ / וּמִטִּיט וָרֶפֶשׁ / לְמַיִם אַדִּירִים

וְהַיָּגוֹן כָּתוֹת / וְנֶהֶרְסוּ שָׁתוֹת / בְּלֵילֵי שַׁבָּתוֹת / מְכֻבָּד בָּאוּרִים

נְטוּעָה הַכַּנָּה / שְׂשׂוֹנָהּ נִשְׁתַּנָה / בְּרֵאשִׁית הַשָּׁנָה / וְיוֹם הַכִּפּוּרִים

אֲנוּשׁוֹת הַמַּכּוֹת / לְדֵעוֹת הַמְחַכּוֹת / וְיוֹם חַג הַסֻּכּוֹת / בְּעֹצֶם יִסּוּרִים

וּמָתְנִי הוּא מוֹעֵד / וְגוּפִי הוּא רוֹעֵד / לְבִיאַת הַמּוֹעֵד / וְיוֹם הַבִּכּוּרִים

וְהַגֶּשֶׁם חָלַף / וְגָבַר הַסֶּלֶף / וְהַצִּיץ לֹא שָׁלַף / בְּשִׂמְחַת הַפּוּרִים

וְאֵין חוֹמֵל עָלַי / לְחַזֵּק נֶחְשָׁלַי / וְנִשְׁאֲרוּ רַגְלַי / סְבוּכִים בַּסִּירִים

וְאֵין מַקְשִׁיב קֶשֶׁב / וְאֵין אוֹמֵר הָשֵׁב / כְּקָרְבָּן לֹא נֶחְשַׁב / כְּאִישִׁים נִבְעָרִים

סְעִפַּי אַל תִּנְשׁוּ / וְרַעְיוֹנַי חוּשׁוּ / פְּרוּדוֹת עָבָשׁוּ / וְחַתּוּ גִבּוֹרִים

אֲטַהֵר הַנֶּפֶשׁ / וְאֶכְרַע בַּחֹמֶשׁ / לְמַזְרִיחַ שֶׁמֶשׁ / וּמַעְרִיב בִּדְבָרִים

וְאָנִיעַ שָׂפָה / בְּקוֹמָה נִכְפָּפָה / לְרוּחַ נִשְׁטָפָה / וְחַיִּים נֶעְכָּרִים

לְעַצְמוֹתַי אֲמַגֵּר / בְּמַלְכוּת הַחִגֵּר / לְאֶזְרָח עִם הַגֵּר / חֲבָלִים עִם צִירִים

לְשִׁבְרִי הוּא אִוָּה / נְתִיבוֹתַי עִוָּה / וְדִמְעִי הוּא רִוָּה / גְּבָעוֹת עִם הָרִים

לְעֵדָה נִזְנָחָה / אֲסוּרָה הַשִּׂמְחָה / וְתָמִיד הָאֲנָחָה / חֲדָשִׁים לַבְּקָרִים

וְגָבַר בֶּן אָמָה / בְּקֶצֶף וּבְחֵמָה / עֲלֵי יוֹנָה תַּמָּה / כְּחַיּוֹת וּנְמֵרִים

וְיֵשׁ מֵהֶם נִדְכֶּה / בְּכָל רֶגַע יִבְכֶּה / וְיֵשׁ בִּשְׁחִין מֻכֶּה / וְגַם בַּסַּנְוֵרִים

וְלֹא נָשָׂא פָנִים / וְלֹא חָן לִזְקֵנִים / וְאָבוֹת עַל בָּנִים / בְּכֶבֶל נֶאֱסָרִים

וּמִי יָזוּן לַטַּף / בְּבוֹא יוֹם הַשֶּׁטֶף / וְאָב מֵהֶם נֶחְטַף / וְשָׂבְעוּ תַּמְרוּרִים

וּמִי יָאִיר אוֹרָה / לְלוֹמְדֵי הַתּוֹרָה / יְנַהֵג כַּשּׁוּרָה / לְבָנִים נִמְהָרִים

וְנָשִׁים שַׁאֲנַנּוֹת / עֲזוּבוֹת אַלְמָנוֹת / בְּקִינוֹתֵיהֶן שׁוֹנוֹת / מְתַנּוֹת לִגְבָרִים

אֲחַזֵּק הַבֶּדֶק / בְּתוֹךְ עָם כַּחֶדֶק / וְהוּא שׁוֹפֵט צֶדֶק / בְּמִבְחַר מֵישָׁרִים

וְאֵיךְ לֹא יִרְאֶה יָהּ / לְבָנָיו בַּשִּׁבְיָה / וְעַד כַּמָּה הוֹיָה / נְעִילָה לִשְׁעָרִים

וְעוֹד שַׁעַר אַחֵר / לְדִמְעָה לֹא תְאַחֵר / אֲנִי תָמִיד שׁוֹחֵר / וּמִי לִי לִסְגוֹרִים

אֱלֹהַי הַקְשִׁיבָה / בְּכִי עַיִן זָבָה / וְתִהְיֶה נִכְתָּבָה / לְחַיִּים בִּסְפָרִים

וְדָתָךְ לֹא נָמֵר / וְאַחְוָה לֹא נָפֵר / וְאִם דִּינָךְ גּוֹמֵר / אֲנַחְנוּ נִמְסָרִים

וְעֵת בָּנִים תִּמְכֹּר / בְּרִית אָבוֹת תִּזְכֹּר / וְאַל נָא אֵל תַּעְכֹּר / לְמַעַמַד הַיְקָרִים

וְאִם חֶטְאִי גָּבָר / וְלִבִּי לֹא נִשְׁבָּר / רְאֵה נָא לְנָבָר / וְהַצֵּל נוֹתָרִים

הֱיֵה מָעוֹז וּמַחְסֶה / לְעָם נֶאֱלָם כַּשֶּׂה / בְּצִלָּךְ יִתְכַּסֶּה / וְתַעֲזֹב לַעֲדָרִים

אֱלֹהִים חָנֵּנוּ / וּבוּז רַב שָׂבַעְנוּ / גְּאַל נָא אוֹתָנוּ / כְּמִיַּד הַמִּצְרִים

אֱמֹר נָא אֵל שַׁדַּי / לְצָרוֹתֵינוּ דַּי / וְעַנֵּה מַעֲבִידַי / מְלָכִים עִם שָׂרִים

וְלִבָּם תַּקְשִׁיחַ / וְדָעְתָּם תַּשִּׁיחַ / וְיָבוֹא מָשִׁיחַ / וְקַרְנֵנוּ תָרִים

בְּבוֹא בֶּן־עַמִּיאֵל / לְנַחֵם יִשְׂרָאֵל / וְזֶרַע יִשְׁמָעֵאל / לְהַכְרִית וּלְהַחְרִים

פְּנֵה אֶל חַטָּאתָם / וְעַרְבֵּב שִׂמְחָתָם / הֲפֹךְ אֶת טִירָתָם / כְּמַהְפֵּכַת זָרִים

אֲבַשֵּׂר גָּלִיּוֹת / בְּרִדְתָּם תַּחְתִּיּוֹת / וְנַעְלֶה לַעֲלִיּוֹת / זְקֵנִים וּנְעָרִים

וְתִרְבֶּה הַדַּעַת / כְּשָׂדֶה נִזְרַעַת / וְחָכְמָה נוֹבַעַת / כְּיַמִּים וּנְהָרִים

וְהַצֵּל נָא גוֹחִי / לְרַגְלִי מִדֶּחִי / וְנַפְשִׁי כִּי תֶחִי / תְּהַלֵּל בִּשְׁעָרִים

וְגָלוּתִי תָשִׁיב / וְרִנָּתִי תַּקְשִׁיב / הֲלֹא אַתָּה מֵשִׁיב / נְשָׁמוֹת לִפְגָרִים

וּבִמְלֹאת הַיָּמִים / לְגַן עֵדֶן תָּמִים / וְאוֹדֶה בָעַמִּים / בְּפָנִים מַזְהִירִים.
