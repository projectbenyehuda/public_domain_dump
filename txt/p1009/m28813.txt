


	
	אֲשֶׁר שָׁת מַלְכוּתוֹ
	








אֲשֶׁר שָׁת מַלְכוּתוֹ / עֲלֵי כָל מַמְלֶכֶת / לְפָנָיו בּוֹרֶכֶת / וּבוֹ מִתְבָּרֶכֶת



יְסוֹדָהּ הוּא אֵל חַי / בְּנֵצַח עִם הַהוֹד / ולוֹ הַתִּפְאֶרֶת / בְּקָו מִתְהַלֶּכֶת



זְרוֹעוֹת הָעוֹלָם / גְּדֻלָּה וּגְבוּרָה / לְנֶגֶד כָּל סוֹרֵר / כְּקֶשֶׁת דּוֹרֶכֶת



פְּעֻלָּתָם תָּמִיד / בְּבִינָה וּבְחָכְמָה / אֲצִילוּתָם תִּמְצָא / לְכֶתֶר נִמְשֶׁכֶת



וְאֵין סוֹף עִלָּתָם / וְסַפִּיר גִּזְרָתָם / דְּרֹשׁ וַחְקֹר הֵיטֵב / בְּסֵפֶר מַעֲרֶכֶת



וְתִמְצָא שָׁם בִּנְיָן / מְיֻסָּד תַּלְפִּיּוֹת / לְמַעְלָה וּלְמַטָּה / בְּעֵץ מִסְתַּבֶּכֶת



וְהוּא עֵץ הַחַיִּים / לְכָל מֵבִין סוֹדוֹ / בְּצִלּוֹ חִמַּדְתִּי / נְתִיבוֹת לָלֶכֶת



וְלוֹ שָׁם עַמּוּדִים / יְהוּדָה גּוּר אַרְיֵה / וּמִנְחָתוֹ סֹלֶת / בְּיֵינָהּ מוֹסֶכֶת



שְׁתוּ דוֹדִים שִׁכְרוּ / וְכָל רֵעִים אִכְלוּ / רְאוּ כִּי דְּבַשׁ תַּחַת / לְשׁוֹנִי נִמְשֶׁכֶת



וְהָרִימוּ מִכְשׁוֹל / וְכָל דֵּעָה זָרָה / אֲשֶׁר אוֹר הַשֶּׁמֶשׁ / לְלַיְלָה הוֹפֶכֶת



וְהוּא עָמוֹק עָמוֹק / וּמִי יִמְצָאֶנּוּ / וְעַל זֹאת שַׁאֲגָתִי / כְּמַיִם נִתֶּכֶת



וְסוֹף הַמַּחְשָׁבָה / לְאֵל אֶשָּׂא עֵינַי / אֲשֶׁר לוֹ כָּל לָשׁוֹן / תְּהִלָּה עוֹרֶכֶת.
