


	
	לְאִישׁ נָדִיב עֲרָבִי תּוֹךְ דְּבִירוֹ
	






לְאִישׁ נָדִיב עֲרָבִי תּוֹךְ דְּבִירוֹ / וְהוּא יַהֵל עֲרָבִי עַל מְאוֹרוֹ

וְכָל עוֹבֵר וְשָׁב רָצִים לְפִתְחוֹ / לְמַתַּנְתּוֹ בְּצֹאנוֹ עִם בְּקָרוֹ

וּמִצָּפוֹן וּמִמִּזְרָח וּמִיָּם / וּמִנֶּגֶב יְבוֹאוּן לַחֲצֵרוֹ

יְמַלֵּא מִשְׁאֲלוֹת כָּל דַּל וְאֶבְיוֹן / בְּלִי עֶצֶב וּפָנָיו בַהֲדָרוֹ



עֲדֵי שָׁמְעוֹ בְּאַפְסֵי־יָם רְחוֹקִים / מְתַנִּים אֶת שְׁבָחוֹ עִם יְצוּרוֹ 



הֲכִי גָבַר עֲלֵי נָדִיב וְשׁוֹעַ / וְהִפְלִיג עַל בְּנֵי עַמּוֹ בְּשׁוּרוֹ

בְּבוֹא חֶבֶל עֲרָבִיִּים לְדוּמָה / מְצָאוּהוּ מְעֻתָּד עַל קְצִירוֹ

שְׁאָלוּהוּ לְבֵית נָדִיב וְאַיֵּה / מְעוֹן רִבְצוֹ וְאֵיפֹה עִיר מְגוּרוֹ

וְהוּא עָנָה אֲנִי עַבְדּוֹ וְעַל מַה / תְּבַקֵּשׁוּ יְמַלֵּא מַחֲסוֹרוֹ

אֲזַי אָמְרוּ אֱמֶת אָנוּ לְחוּצִים / בְּחֵץ רָעָב הֲדָפָנוּ בְּשִׁבְרוֹ

וּבָנֵינוּ נְטוּשִׁים בַּיְּעָרִים / בְּמֶרְחַקִּים לְכֹבֶד מַאֲסָרוֹ

וְאִם יֵלֵךְ אֱנוֹשׁ מֶנּוּ בְּרֵיקָם / לְעֵינֵי כֹל יְהֶא עוֹכֵר שְׁאֵרוֹ

לְמֵי חַסְדּוֹ וְאֶת פָּנָיו נְחַלֶּה / נְקַוֶּה לוֹ עֲדֵי עֵת בּוֹא דְבָרוֹ

בְּחִישׁ הָלַךְ לְחַפֵּשׂ אוֹצְרוֹתָיו / וְלֹא מָצָא מְאוּמָה בַּחֲדָרוֹ

וְלֹא מִקְנֵה עֲדָרִים בַּגְּדֵרוֹת / וְלֹא טָלוּא וְנָקוֹד עַל שְׂכָרוֹ

וְאָז הָיָה כְמֵשִׁיב עַל בְּעָלָיו / וְלֹא יַמְרֶה עֲלֵי מִצְוַת גְּבִירוֹ

נְתָנַנִי אֲדוֹנִי לַהֲמוֹנְכֶם / וְאִם אָנִי זְקַן בֵּיתוֹ בְּחִירוֹ

לְכוּ מִכְרוּ לְשֶׁבֶר רַעֲבוֹנְכֶם / וְאִם תִּרְצוּ לְשָׁרֵת אִישׁ בְּעִירוֹ

אֲנִי אוֹמֵן לְכָל עוֹלָל וְיוֹנֵק / וּבַבַּיִת וּבַשָּׂדֶה בְּנִירוֹ

אֲנִי אֶשְׁמֹר פְּרִי גִנָּה וּפַרְדֵּס / וְעֵץ עוֹשֶׂה פְּרִי אַשְׁקֶה בְּתוֹרוֹ

וְהֵם לָקְחוּ נְתָנוּהוּ בְּמִשְׁמָר / עֲדֵי בוֹאָם לְאַרְצָם דֹּב וְגוּרוֹ

וּמִנּוּהוּ לְנוֹטֵר בַּכְּרָמִים / יְהֶא פָקִיד עֲלֵיהֶם עַד בְּצִירוֹ

וּמִן הַיּוֹם אֲשֶׁר עָשׂוּ פְרִיהֶן / וְלֹא שִׁנָּה כְּמִנְהָגוֹ בְּדוֹרוֹ

לְכָל שׁוֹאֵל עֲנָבִים אוֹ תְאֵנִים / וְהוּא נוֹתֵן בְּשֶׁפַע עַל יְקָרוֹ

וְהוּא יוֹשֵׁב בְּסֻכָּתוֹ כְּחָתָן / נְגִינוֹתָיו עֲלֵי אָפְנָיו בְּשִׁירוֹ

עֲדֵי נִכָּר בְּמַעְנֵה פִיו וְשִׂיחוֹ / מְאֻשָּׁר הוּא לְמוֹלַדְתּוֹ וְצוּרוֹ

שְׁהוּא תַּחְכָּם לְנָגִיד עַל נְדִיבִים / בְּחָנוּהוּ כְּמוֹ זָהָב בְּכוּרוֹ

וְאָז רָצוּ אֱלֵי מַשָּׂא וְדוּמָה / וְהֶעְבִירוּ בְּקוֹל נִשָּׂא בְּצִירוֹ

עֲדֵי בָאוּ אֲלָפִים עִם רְבָבוֹת / וְלֹא אֶחָד אֲשֶׁר הִתִּיר חֲגוֹרוֹ

בְּשׁוּרָם אִישׁ כְּפִי שִׂכְלוֹ יְהֻלָּל / כְּמוֹ גֶבֶר אֲשֶׁר יַיִן עֲבָרוֹ

קְצִינֵי יִשְׁמְעֵאלִים נֶאֱסָפוּ / עֲשׂוֹת מִשְׁפָּט וְלֹא תִמְעַד אֲשׁוּרוֹ

בְּנֵי יָקְטָן אֲזַי נִלְאוּ לְחַפֵּשׂ / בְּסִפְרֵיהֶם לְאִישׁ כָּזֶה פְְאֵרוֹ

וְלֹא מָצְאוּ לְמַשְׂכֻּרְתּוֹ חֲלִיפוֹת / וְאִם יַקְרִיב אֱנוֹשׁ לִבְנוֹ בְּכוֹרוֹ

אֲזַי קָמוּ לְאִישׁ נֶזֶם שְׁלָלוֹ / וְאִישׁ עֶדְרוֹ וְשׁוֹרוֹ וַחֲמוֹרוֹ

וְאִישׁ מֵבִיא גְמַלִּים הַמְּנִיקוֹת / וְאִישׁ מֵבִיא לְפָרָתוֹ וּפָרוֹ

וּבָאוּ מַחֲנֵה שָׁרִים וְשָׁרוֹת / וְכָל רוֹאָיו יְלַחֵךְ אֶת עֲפָרוֹ

וְלִוּוּהוּ בְּעֹצֶם מַתְנוֹתָיו / אֱלֵי עִירוֹ וְהוּשַׁב אֶל מְקוֹרוֹ

נְדִיבֵי עָם לְדַעְתָּם נֶאֱסָפוּ / לְקַלֵּס לוֹ אֲשֶׁר הָלְכוּ לְאוֹרוֹ 

וְהוּא תַחְכָּם וְהִתְחַכַּם בְּשׁוּבוֹ / לְכָל נָגִיד עֲזָרוֹ הוּא עֲצָרוֹ

עֲדֵי נָתַן לְכָל אֶחָד חֲלִיפוֹת / וְשָׁב נָדִיב עֲרָבִי תּוֹךְ דְּבִירוֹ.
