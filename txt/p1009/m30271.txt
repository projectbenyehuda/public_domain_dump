


	
	תְּנִי נֶפֶשׁ יְקָרָה בַתְּחִלָּה
	




תְּנִי נֶפֶשׁ יְקָרָה בַתְּחִלָּה / תְּנִי כָבוֹד לְאֵל נוֹרָא עֲלִילָה

וְהִשְׁתַּחֲוִי לְמוּל בֵּיתוֹ וְכִרְעִי / בְּתַחֲנוּנִים בְּאִישׁוֹן הָאֲפֵלָה

וְזִכְרִי מַעֲשִׂים בִּימֵי נְעוּרֵךְ / בְּעָבְרֵךְ עַל חֲמוּרָה גַּם נְקַלָּה

זְדוֹן לִבֵּךְ הֲלֹא נִכְתַּב בְּסֵפֶר / וְלֹא הֵכִיל וְעוֹד כָּתְבוּ מְגִלָּה

יְמוֹתַיִךְ הֲלֹא חָלְפוּ וְעָבְרוּ / בְּיַם סִכְלוּת טְבוּעָה בַּמְּצוּלָה

וְתַחַת הֵיכְלֵי עֹנֶג וּפַנַּג / בְּרֹאשׁ צוּרִים אֱמֶת סוּרָה וְגוֹלָה

וּמַלְבּוּשֵׁךְ לְמֶשִׁי עִם רְקָמוֹת / וְאַרְגָּמָן וּבַתַּחַשׁ נְעוּלָה

וְאֵיךְ שָׁפַל עֲדֵי אֶרֶץ יְקָרֵךְ / וְסָר נִזְרֵךְ כְּשִׁמְשׁוֹן עִם דְּלִילָה

וְתָמִיד אֶשְׁמְרָה מַחְסוֹם לְמוֹ פִי / וְדוּמִיָּה אֱלֵי נַפְשִׁי תְּהִלָּה

אֲבָל נַפְשִׁי צֳרִי אֶמְצָא לְצִירֵךְ / וְאֶשְׁלָחֵךְ וְעִבְרִי הַמְסִלָּה

וְהַצָּפוֹן לְמוּל עֵינֵךְ בְּלֶכְתֵּךְ / עֲלֵי הָהָר וּמִישׁוֹר הַשְּׁפֵלָה

וּמָנוֹחַ לְעוֹלָם אַל יְהִי לָךְ / עֲדֵי תִרְאִי לְצִקְלַג עִם קְעִילָה

בְּעִיר צִיּוֹן יְהֶא שָׁמָּה חֲנוֹתֵךְ / וְעִיר שָׁלֵם וְהָעִיר הָאֲבֵלָה

וְיוֹם בּוֹאֵךְ בְּמִלּוּי הַלְּבָנָה / וְשָׁעָתֵךְ תְּהֶא שָׁעָה מְעֻלָּה

שְׁקִי בַלָּאט לְאַדְמַת הָעֲזָרוֹת / וְשִׁפְכִי לֵב וְהַרְבִּי הַתְּפִלָּה

וּמִשָּׁם תַּעֲלֶה לָנוּ רְפוּאָה / וְגַם יֶתֶר שְאֵת עִם עֹז וְחֶמְלָה

וְלַשָּׁבִים שְׁעָרֶיהָ פְּתוּחִים / וְלֶחָנֵף בְּפָנָיו הִיא נְעוּלָה

אֱלֹהִים שָׁם שְׁמוֹ נִקְרָא וְשַׁדַּי / גְּבוּרָה לוֹ וְנֵצַח עִם גְּדֻלָּה

וּמַתַּנְתּוֹ לְכָל נוּצָּר וְנִבְרָא / וְלַגָּזֵל אֱמֶת שׂוֹנֵא בְעוֹלָה

וְהַגִּידִי שְׁלוֹמִי לוֹ בְּרֵאשִׁית / וְזִכְרִי אֶת שְׁבָחוֹ הַמְּעֻלָּה

וְסַפֵּרִי בְּדַלּוּת הָאֲסוּרִים / וְסָר שִׂכְלָם וְאֵין דַּעַת צְלוּלָה

וְאֵין לִי בַזְּמָן הָרָע מְתִיקָה / אֲשֶׁר לֹא תִהְיֶה בָּרֹאשׁ בְּלוּלָה

וּמֵעַי רֻתְּחוּ הָמוּ בְקִרְבִּי / וְכָל יָמֵַי שְׁנַת עֵינַי גְּזוּלָה

וְעֵינִי עָשְׁשָׁה מֵרֹב בְּכִיָּה / וּמַחֲלָתִי לְבַת יַעֲנָה מְשׁוּלָה

וְלוּ אֶהְיֶה אֲנִי עִם עַם נְמָלִים / לְרֹב דַּלּוּת נְשָׂאַתְנִי נְמָלָה

אֲנִי אֶדֶן וְהַצָּרוֹת קְרָשִׁים / וְאֵין רֶגַע בְּלִי צָרָה כְּפוּלָה

אֱמֶת כָּל מִשְׁבְּרֵי תֵּבֵל מְצָאתִים / וְרָעוֹת רַבּוֹת תּוֹכְפוֹת לְמַעְלָה

וְרִישׁ עִם מַחְסוֹר גָּבְרוּ שְׁנֵיהֶם / וּמַה כּוֹחִי לְיָדָם הַגְּדוֹלָה

וְעָרוֹם מִבְּלִי לָבוּשׁ לְקָרָה / לְכַסּוֹת אֶת בְּשָׂרִי הַמְּגֻלָּה

וּפַת לַחְמִי בְּמֵי דִמְעִי מְסוּכָה / וְהֵם חֶלְקִי וְכוֹסִי לַאֲכִילָה

וְיִסַּר הַזְּמָן צָרָיו בְּטֶפַח / וְלִי יִסַּר בְּאַמָּה הָאֲצִילָה

נְגִינָתִי חֲשַׁבְתִּיהָ לְקִינָה / וְקוֹל שִׁירִי אֲכַנֶּה קוֹל יְלָלָה

וְאִם לֹא אֵצְאָה מִמַּאֲסָרִי / אֲזַי אֵרֵד בְּיָגוֹן אֶל שְׁאוֹלָה

וְלֹא אֶרְאֶה לְבָנַי עִם בְּנוֹתַי / וְאַהֲבָתָם בְּתוֹךְ לִבִּי שְׁתוּלָה

וְנָשַׁי הַשְּׁתַיִם הָעֲזוּבוֹת / וְלֹא אָזוּן לְאִמִּי הַשְּׁכוּלָה

וּמִי יָאִיר בְּיוֹם צָרָה לְעֵינַי / וּמִי יִגְלֶה לְאָזְנִי הָעֲרֵלָה

הֲלֹא נַפְשִׁי אֲשֶׁר נִבְרֵאת יְקָרָה /יְקָרָה הִיא וּמִכִּסֵּא אֲצוּלָה

וְיוֹם יִפְקֹד אֱלֹהִים אֶת עֲבָדָיו / וְדַרְכִּי הִיא לְעֻמָּתוֹ סְלוּלָה

אֲדוֹן עוֹלָם וּמַתִּיר הָאֲסוּרִים / גְּאָלֵנִי אֲסַלֵּד לָךְ בְּחִילָה

וְיֵאוּת אֶל יְצִיר חֹמֶר מְעִילָה / וְיֵאוּת לָךְ אֱלֹהִים הַמְּחִילָה

לְךָ אֶכְרַע שְׁחָרִים וַעֲרָבִים / וְאֶפְרֹשׂ כַּף בְּמִנְחָה עִם נְעִילָה

וְאֶתְחַזֵּק כְּפִי כֹחִי לְעָבְדָךְ / אֲחוֹנֵן דַּל וְאֵיטִיב הַפְּעֻלָּה

וְעוֹנֶה צוֹעֲקִים נִקְרָא שְׁמֶךָ / מְרוֹמֵם דַּל וְסוֹמֵךְ הַנְּפִילָה

יְחוֹנֵן צוּר בְּחֶמְלָה עַל עֲבָדָיו / יְקָרֵב קֵץ וְיָחִישׁ הַגְּאֻלָּה

וְיוֹם יָקִיץ יְשֵׁנֵי הָאֲדָמָה / תְּנִי נֶפֶשׁ יְקָרָה בַתְּחִלָּה.
