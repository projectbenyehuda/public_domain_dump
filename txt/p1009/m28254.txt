


	
	אֶל אֵל אֶקְרָא מֵרֹאשׁ צוּרִים
	






אֶל אֵל אֶקְרָא / מֵרֹאשׁ צוּרִים / מֵקִים לַדַּל / מַשְׁפִּיל וּמֵרִים



בִּינוֹתִי עִם / שִׂכְלִי כִּי אַתְּ / מֵאִיר תֵּבֵל / גַּם לַדָּרִים



גַּל מֵעָלַי / חֶרְפָּה וָבוּז / אֶהְגֶּה תָּמִיד / עוֹד בִּסְפָרִים



דִּבְרֵי הָרָב / יוֹסֵף אַלְבּוּ / נִרְשָׁם בַּעַל / הָעִיקָרִים



הוּא לִי חֶמְדָּה / גַּם לִסְגֻלָּה / תּוֹךְ אַרְבַּעַת / הַמַּאֲמָרִים



וַיָּשֶׁת לִי / חָכְמָה רָמָה / הִיא כַמַּיִם / הָאַדִּירִים



זַכֵּנִי אֵל / לֵידַע סוֹדָהּ / הַפְשֵׁט מֶנִּי / לֵב נִמְהָרִים



חַטָּאתִי אֵל / אַל תִּזְכָּר עוֹד / הֵן בִּזְקוּנִים / הֵן בִּנְעוּרִים



טֶרֶם תָּבוֹא / עַל עַבְדֶּךָ / בַּמִשְׁפָּטִים / הַנֶּעְכָּרִים



יַעַן אוּכַל / לִקְנוֹת עוֹלָם / אוֹסִיף לֶקַח / טוֹב בִּסְגוֹרִים



כַּוָּנָתִי / לִרְדֹּף צֶדֶק / אֶדְרֹךְ מִבְחַר / הַמֵּישָׁרִים



לָכֵן יִצְרִי / הוּא שֶׁעִכַּב / הִטְבִּיל עִיסָתִי / בִּשְׂאוֹרִים



מָה אוֹחִיל עוֹד / מִן הַיֵּצֶר / תִּבְכֶּה נַפְשִׁי / בַּמִּסְתָּרִים



נִין הָאָמָה / עַל כָּל אֵלֶּה / הוּא הִשְׁקַנִי / כּוֹס עִיקָרִים



סִלָּה מֶנִּי / כָּל אַבִּירַי / הוֹשִׁיבַנִי / בַּמַּאֲסָרִים



עָמַד נֶגְדִּי / הִדְאִיבַנִי / וַאֲנִי מֻכֶּה / בַּסַּנְוֵרִים



פָּרַץ בִּנְיַן / גֶּדֶר אַחְוָה / הָפַךְ לִבּוֹ / לָאַכְזָרִים



צָמְתוּ בַבּוֹר / חַיֵּי שָׂרִים / וּפְנֵיהֶם עוֹד / לֹא נֶהְדָּרִים



קַנֹּא קַנֵּא / וּנְקֹם נָקָם / קַדֵּשׁ שִׁמְךָ / מוּל הַצָּרִים



רַחֵם עַמָּךְ / וּגְאַל אוֹתָם / מִיַּד אוֹיְבִים / מִן הַהַגְרִים



שִׁמְעָה וּפְנֵה / אֶל הָעַרְעָר / וּזְכֹר אָבוֹת / הֵם הַיְקָרִים



תַּכְלִית מִלַּי / עִם רֹב שִׂיחִי / אֶשָּׂא עֵינַי / אֶל הֶהָרִים.
