


	
	מִבֵּין עֳפָאִים קוֹל זְמִירוֹת יַעֲנוּ
	








מִבֵּין עֳפָאִים קוֹל זְמִירוֹת יַעֲנוּ / עָגוּר וְתוֹר וְסִיס בְּשִׁיר יָרֹנוּ

וָאֶשְׁאֲלָה סִבַּת זְמִירָתָם וְאֵיךְ / עָמְדוּ לְמִשְׁמַרְתָּם וְלֹא יִסְכֹּנוּ

וַיֹּאמְרוּ כִּי שָׁמְעוּ זִמְרָה וְשִׁיר / בָּא מִשְּׁמֵי שַׁחַק וְהִכּוֹנָנוּ

לַעְנוֹת בְּשִׂמְחָתָם וְצַהְלָתָם הֲכִי / שִׂיחִים בְּיַחַד הָרִים יְרַנֵּנוּ

יַעַן אֲשֶׁר בָּא אִישׁ חֲכַם רָזִים וְגַם / מַשְׂכִּיל בְּכָל חָכְמָה וְצָץ פִּרְחֵנוּ

רָב אַשְׁכְּנָזִי הוּא וְנִקְרָא אַב הֲמוֹן / לוֹ יָד וְשֵׁם כִּדְבַשׁ בְּמַלְקוֹחֵנוּ

הֵאִיר פְּנֵי תֵימָן וְהוֹסִיף עַל מְאוֹר / שֶׁמֶשׁ וְיָרֵחַ וְכוֹכָבֵינוּ

וָאֶעֱנֶה אַף אֲנִי חֶלְקִי בְּטוּב / לִקְחִי וְגַם שִׂיחִי לְהַכְתִּירֶנּוּ

שָׁלוֹם לְבוֹא הַשָּׂר אֲבִי מִשְׂרָה וְכָל / חָכְמָה מְפֹאָרָה אֱמֶת צִבְיֵנוּ

תָּמְכוּ לְבָבֵנוּ תְּשׁוּבוֹתָיו בְּעֵת / נִשְׁמַע לְזוּלָתוֹ וְהִשְׂבִּיעָנוּ

מִקְרָא וְגַם מִשְׁנָה וְתַלְמוּד חִבְּרוֹ / פַּרְדֵּס נְטָעָהוּ וְהֶאְכִילָנוּ

עַל רֹב סְפָרָיו כָּל לְבָבוֹת עָלְצוּ / הִתְחַלְּצוּ לִלְמֹד וְלֶאְהָבֶנּוּ

עַד כִּי רְאוֹת פָּנָיו בְּהֶגְיוֹנוֹ הֲלֹא / תָרוֹן לְשׁוֹן אִלֵּם וְתַרְהִיבֶנּוּ

וָאֶטְעֲמָה מִצּוּף דְּבַשׁ יַעְרוֹ וְהֵן / אֹרוּ שְׁתֵּי עֵינַי וְרַב כֹּחֵנוּ

אֵלֶּה זְמִירוֹת יֹאמְרוּ לִי נֵלְכָה / נִכְרַע לְמוֹרֵנוּ וְגַם מַלְכֵּנוּ

אוּלַי בְעֵינָךְ יִמְצְאוּ חֵן כִּי לְךָ / יִתְחַנְּנוּ לֵאמֹר נְשָׂא טָרְחֵנוּ

אֵלֶּה פְרָחֵינוּ וְלֹא גָמְלוּ עֲדֶן / אוּלַי יְמַהֵר יוֹם גְּמוֹל פִּרְחֵנוּ

חוּט זִכְרְךָ הַטּוֹב מְשָׁכָנוּ וְעַד / נִפְגַּע בְּךָ לֹא נַעֲצֹר רוּחֵנוּ

גִּילוּ בְנֵי תֵימָן אֱמֶת לֵכוּ חֲזוּ / פָּנִים אֲשֶׁר הֵאִיר לְמַחְשָׁכֵנוּ

וּפְנוּ לְשִׂיחוֹתָיו וְאִמְרוֹתָיו הֲלֹא / זֶהוּ אֲשֶׁר דָּלָה וְגַם הִשְׁקָנוּ

מִמַּעֲיַן חָכְמָה מְקוֹר בִּינָה אֱמֶת / שָׁאַב בְּלֵב רָחֵב וְיָצַק לָנוּ

הִתְאַזְּרוּ הֵחָלְצוּ הִקָּבְצוּ / לֹא בַעֲצַלְתַּיִם וְסוֹף אָשְׁרֵנוּ

אִם נֵדְעָה הַשְׂכֵּל וְאוֹתוֹ נִרְדְּפָה / נִפְנֶה לְעֵץ חַיִּים וְהוּא חַיֵּינוּ

יוֹמָם וְגַם לַיִל לְזֹאת מִי יִתְּנָה / עֵדוֹת וְהַחֻקִּים בְּשִׂיחָתֵנוּ

אוּלַי יְכֻפַּר בָּם חֲטָאֵינוּ וְאִם / יַעְזֹר אֱלֹהִים יִקָּבְצוּ נִדְחֵינוּ

יִרְפָּא לְכָל מָזוֹר בְּגָלוּת רָחֲבָה / כַּיּוֹם וְהֵם מַרְפֵּא לְמַכְאוֹבֵינוּ

עַל יַד אֲבִיר רוֹעִים גְּבִיר דּוֹרוֹ וְהוּא / כִּמְעַט יְעוֹרֵר בִּיאַת מְשִׁיחֵנוּ

יִבְנֶה לְעִיר קֹדֶשׁ וְהַהַרְאֵל וְגַם / יִבְנֶה לְעִיר צִיּוֹן וּמִקְדָשֵׁנוּ.
