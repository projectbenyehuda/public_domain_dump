


	
	רְאוּ סֵפֶר חֲקַרְתִּיהוּ בְּחָכְמָה
	








רְאוּ סֵפֶר חֲקַרְתִּיהוּ בְּחָכְמָה / אֲרַגְתִּיהוּ כְּמֶשִׁי הַמְזִמָּה

יְדֵי רַעְיוֹן מְשָׁכוּהוּ בְּהַשְׂכֵּל / יְדֵי מַחְשָׁב רְקַמְתִּיהוּ בְּרִקְמָה

וְעֵין הַלֵּב וּמוֹרָשָׁיו יְצָקוּם / וְסַרְעַפָּיו בְּנִיב יֹשֶׁר חֲתוּמָה

בְּאֹזֶן שָׁמְעָה כָל מֶחְקְרֵי בִין / וְעַיִן רָאֲתָה הֵיכָל פְּנִימָה

בְּסִימַנֵי בְנֵי אָדָם בְּנוּיִים / מְיֻסָּדִים בְּרֹב אוֹנִים וְעָצְמָה

בְּאִישׁ צַדִּיק בְּמִדּוֹתָיו וְחָסִיד / וְאִישׁ עָנָו וְאִישׁ עָרוּם בְּעָרְמָה

וְהַנָּדִיב וְהַשּׁוֹעַ וְכִילַי / וְאִישׁ רָשָׁע שְׁפַל בֶּרֶךְ וְקוֹמָה

קְצַר רוּחַ וְאֶרֶךְ אַף וְרַגְזָן / וְאִישׁ נוֹאֵף בְּמַעְשֵׂהוּ כְלִמָּה

וְהֶעָצֵל וְהָרָכִיל וְהַלֵּץ / וְהַנִּמְהָר וּפֶתִי כַבְּהֵמָה

וְהַנֶּאְמָן וּבַעַל סוֹד וּמָהִיר / חֲכַם לֵבָב בְּדַעְתּוֹ הַתְּמִימָה

וְקֵרֵחַ וְהָרַצְחָן וְגַנָּב / וְעַיִן רָע וְנוֹקֵם הַנְּקָמָה

וּמִתְרוֹשֵׁשׁ וּמִתְעַשֵּׁר וְאֵין כֹּל / וְהֶחָכָם בְּעֵינָיו אֵין מְאוּמָה



[בְּגֻלְגֻּלְתּוֹ בְּאָזְנָיו גַם וּמִצְחוֹ / וּבִזְקָנוֹ וְעֵינוֹ הָרְשׁוּמָה]



וְאִישׁ חֹטֶם וְהֶחָרוּם וְגִבֵּן / וְהַחִגֵּר בְּרַגְלוֹ הָעֲקוּמָה

וְשָׂרוּעַ וְהַקָּלוּט בְּגוּפוֹ / וְאִישׁ קָצַר וְאָרוּךְ בָּאֲדָמָה

וְהַיּוֹעֵץ וְאִישׁ נָבוּב בְּלִבּוֹ / וְהַגַּזְלָן וְאִישׁ עִקֵּשׁ בְּמִרְמָה

וְעוֹד נוֹסַף עַל אֵלֶּה בְּרַבִּים / בְּתוֹךְ סִפְרִי הֲרִימוֹתִי תְּרוּמָה

לְאַפְלַאטוּן וְגַאלִינוּס וּבּוּקְרַאט / וּמִתְפַּלְסֵף בְּחָכְמָתוֹ שְׁלֵמָה

אֲסַפְתִּימוֹ בְּרַעְיוֹנַי לְהַשְׂכִּיל / לְמַעַן יִהֲיוּ בַעְדִי לְחוֹמָה

לְקֵץ יָמַי נְשָׂאַנִי זְמַנִּי / לְאֶרֶץ כּוּשׁ וְהֹדוּ הַזְּעוּמָה

לְעִיר פַּתְרוֹס וְכַסְלוּחִים וְאֵין אִישׁ / מְאַסֵּף לִי לְנַפְשִׁי הָעֲגוּמָה

וְהִנֵּה אִישׁ גְּוִיָּתוֹ מְלֵאָה / בְּכָל מוּמִין וְקוֹמָתוֹ צְנוּמָה

וְאָז שָׁלַח לְיָדוֹ לַהֲבִיאָהּ / לְפַס יָדִי הֱקִימַנִי תְקוּמָה

מְשָׁכַנִי לְבַיִת טוֹב וְרָחָב / לְפַת לַחְמִי וְעֵשֶׂב לַבְּהֵמָה

וּבֶן בָּקָר זְבָחוֹ לַאֲרִיחוּת / וְיֵין רֶקַח לְהָשִׁיב הַנְּשָׁמָה

וְכָל דָּבָר בְּכִיסוֹ שָׁם מְצָאתִִיו / וְאָז שַׁבְתִּי כְּרָחֵל נֶאֱלָמָה

וְחָשַׁבְתִּי לְחָכְמָתִי לְסִכְלוּת / יְגִיעָתִי לְרִיק תּוֹרִישׁ זְמָמָהּ

וְגָמַרְתִּי בְּלִבִּי עַל סְפָרַי / לְהַשְׁלִיכָם מְצוּלוֹת יָם בְּחֵימָה

אֱנוֹשׁ כָּזֶה בְּסִימָנָיו מְגֻנֶּה / וְאֵיךְ תִּהְיֶה פְּעֻלָּתוֹ תְּמִימָה

וְיֵחָשֵׁב אֱוִיל מַחְרִישׁ כְּחָכָם / וְאִוַּלְתּוֹ בְּשַׁעַר נַעֲלָמָה

לְקֵץ יָמִים עֲשָׂרָה אָז קְרָאתִיו / וְאָמַרְתִּי: תְּבֹרַךְ מִמְּרוֹמָה

תְּבֹרַךְ מֵאֲנָשִׁים הַקְּדוּמִים / לְמַעַן כִּי פְעֻלָּתָךְ נְעִימָה

וְעָמַדְתִּי לְהַעְמִיס עַל חֲמוֹרִי / כְּמִתְעַסֵּק וְהֵן נַפְשִׁי כְּהוֹמָה

וְאָז אָמַר: וְאָנָה תַּהֲלֹךְ חִישׁ / וְכָל מַחְסוֹרְךָ כָּתוּב בְּשׁוּמָה

וּפַת לַחְמְךָ וְיֵינְךָ עִם בְּהֶמְתָּךְ / כְּתוּבִים הֵם וְלֹא נִפְקַד מְאוּמָה

עֲמֹד שַׁלֵּם פְּרַע עַתָּה לְחוֹבָךְ / וְלֹא תִטְבַּע בְּתוֹךְ יַם הַתְּנוּמָה

כְּשָׁמְעִי אָז מְשׂוֹשִׂי רָב וְנַפְשִׁי / הֱשִׁיבוֹתִיו אֲנִי אוֹכֵל תְּרוּמָה

אֲשַׁלֵּם לָךְ וְאוֹסִיף הַחֲמִישִׁית / וְאַהֲבָתִי לְךָ נִפְלֵאת עֲצוּמָה

לְמַעַן אֵין יְכֹלֶת בְּיָדִי לְגָמְלָךְ / עֲלֵי אַחַת וְאֵיךְ אוּכַל וְכַמָּה

וְעַתָּה כִי הֲקִימוֹתָ דְּבָרַי / אֲנִי עַבְדָּךְ וְנַפְשִׁי לָךְ לְאָמָה

וְשִׁלַּמְתִּי נְקֻדִּים עִם בְּרֻדִּים / וְלֹא יָדַע וּמַחְשַׁבְתִּי סְתוּמָה

לְזֵד יָהִיר וְלֹא רָשָׁע וְנִבְזֶה / וְאִישׁ מָדוֹן וּבַעַל חֵטְא וְאַשְׁמָה

וְהָלַכְתִּי לְתַנּוֹת לַחֲכָמִים / רְאוּ סֵפֶר חֲקַרְתִּיהוּ לְחָכְמָה.
