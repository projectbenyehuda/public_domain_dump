


	
	אֱמוּנָה יְצָרָהּ יְסוֹד הַיְּסוֹדוֹת
	




אֱמוּנָה יְצָרָהּ / יְסוֹד הַיְּסוֹדוֹת / תְּרַנֵּן לְפָנָיו / בְּשִׁירֵי יְדִידוֹת

בְּטֶרֶם שְׁחָקִים / וְאֶרֶץ מְתוּחִים / וְעַד לֹא אֲרָקִים / וְתֵבֶל יְסוּדוֹת

גְּנָזָהּ לְפָנָיו / אֲלָפִים שְׁנַיִם / וְדוֹרוֹת עֲשָׂרִים / וְשִׁשָּׁה סְדוּרוֹת

דְּבַשׁ מִלְּשׁוֹנָהּ / לְיוֹדְעֵי פְשָׁטִים / וְכָבוֹד וְחַיִּים / לְמֵבִין בְּסוֹדוֹת

הֲדוּרִים תְּיַשֵּׁר / בְּמֵיטַב אֲמָרִים / וְתָשִׁיב לְנֶפֶשׁ / בְּמָשָׁל וְחִידוֹת

וְתַחְכִּים לְפֶתִי / וְסָכָל וְנִמְהָר / וְאִלֵּם יְרַנֵּן / וְחִגֵּר בְקִידוֹת

זְמַנִּים מְשֻׁנִּים / וְהִיא לֹא תְּשֻׁנֶּה / לְנֵצַח נְצָחִים / בְּמִצְווֹת תְּמִידוֹת

חֲקוּקָה בְּחֻקִּים / וּמִצְווֹת יְשָׁרִים / וְעֵדוּת בְּכָל עֵת / בְּחָכְמָה מְעִידוֹת

טְהוֹרָה וּבָרָה / מְאִירָה לְעֵין כֹּל / מְסִלּוֹת נְתִיבוֹת / בְּכָל לֵב אֲפוּדוֹת

יְמִין עֹז וְעִזּוּז / נְתָנָהּ לְמֹשֶׁה / בְּסִינַי מְסָרָהּ / לְיַד אִישׁ חֲמוּדוֹת

כְּנוֹחַ נְבוּאָה / לְבִלְתּוֹ בְּרוּחַ / אֱלֹהִים צְבָאוֹת / בְּלָבְשָׁם חֲרָדוֹת

לְבַד בֵּן לְעַמְרָם / מְדַבֵּר בְּעָמְדוֹ / בְּכֹחוֹ וְשִׂיחוֹ / כְּמוֹ אֵשׁ פְּלָדוֹת

מְנוּחָה וְשָׁלוֹם / וְחַיִּים אֲרוּכִים / לְרָב לַנְּבִיאִים / צְדָקוֹת רְבוּדוֹת

נְתָנָהּ בְּלֵב טוֹב / וְצִוָּה לְשָׁמְרָהּ / הֲלֹא בַּעֲבוּרָהּ / נְפָשׁוֹת פְּקוּדוֹת

שְׂמֹאלָהּ יְעַשֵּׁר / לְרֵיקִים וְכָבוֹד / וְאֹרֶךְ לְיָמִים / בְּיָמִין אֲגוּדוֹת

עֲרֵבָה בְּכָל פֶּה / וּמַרְפֵּא חֳלָיִים / לְדַבֵּר בְּסוֹדָהּ / יְדֵי עֵט כְּבֵדוֹת

פְּתַח לִי אֱלֹהִים / לְבָבוֹת סְתוּמוֹת / וְלֹא יִהֲיוּ עוֹד / בְּהֶבֶל שְׁדוּדוֹת

צְדָקָה עֲשֵׂה נָא / לְמַעַן שְׁמֶךָ / וְהַתֵּר לְרַגְלַי / בְּתוֹךְ פֶּה לְכוּדוֹת

קְדֻשָּׁה וּמַלְכוּת / תְּשׁוֹבֵב לְצִיּוֹן / וְנִזְכֶּה לְשָׁרֵת / בְּאַרְבַּע עֲבוֹדוֹת

רְאֵה זִכְרְךָ יָהּ / אֱמֶת תּוֹךְ לְבָבִי / וְאִם רַב עֲווֹנִי / וְרַבּוּ בְּגִידוֹת

שְׁלַח מִמְּרוֹמִים / וְקַבֵּל לְשִׁירִי / כְּחַטָּאת כְּאָשָׁם / כְּעוֹלוֹת וְתוֹדוֹת

תְּקַבֵּל לְעַבְדָּךְ / וְאֶכְרַע וְאוֹדֶה / וְתָמִיד אֲזַמֵּר / לְךָ טוֹב לְהוֹדוֹת.
