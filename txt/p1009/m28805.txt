


	
	אֲשֶׁר בָּרָא יְצִירוֹתָיו אֲלֻמּוֹת
	








אֲשֶׁר בָּרָא יְצִירוֹתָיו אֲלֻמּוֹת / גְּוָנֵיהֶם לְבָנוֹת וַאֲדֻמּוֹת

וְעֵץ יַעַר בְּפֵרוֹתָיו מְשֻׁנִּים / וְטַעְמֵהוּ מְרוֹרוֹת עִם בְּשׂוּמוֹת

וְאֵד יַעַל לְהַשְׁקוֹת הָאֲדָמָה / וְטַל אוֹרוֹת וְכֶעָבִים גְּשׁוּמוֹת

וְנִבָּא הַנְּבִיאִים מֵהֲדָרוֹ / בְּתַרְדֵּמָה וְהֶעֱבִיר קוֹל דְּמָמוֹת

וְחִיָּה הַפְּגָרִים הַמְדֻכִּים / אֲשֶׁר הָיוּ בְּמַכּוֹתָן הֲלוּמוֹת

וְלֶעָצֵל אֱמֶת הִלְבִּישׁ קְרָעִים / וּבָחַר לוֹ מְעַט שֵׁנוֹת וְנוּמוֹת

וְלִי נָתַן לְשׁוֹן לִמּוּד לְמַעַן / אֲהֶא מֵפִיק לְכָל שׁוֹאֵל זְמָמוֹת

וְהוּא פָּתַח לְשׁוֹן אִלֵּם בְּרִנָּה / וּפָתַח לֵב לְפֵיוֹת הַחֲסוּמוֹת

וְרַעְיוֹנַי כְּבָר חָנוּ בְמָרָה / וְעַתָּה סוֹף בְּמִתְקָה הֵם טְעוּמוֹת

וְהִנְחִיל דָּת וְתוּשִׁיָּה לְעַמּוֹ / בְּמִדְבַּר סִין וּמִבֵּית הַיְשִׁימוֹת

וְהוֹפִיעַ בְּהַר שֵׂעִיר וּפָארָן / וּמַחְפִּירוֹת פְּנֵיהֶם מִכְּלִימוֹת

וְלַחְמָם הוּא אֱמֶת לֶחֶם שְׁקָרִים / מְזֵי רָעָב בְּרִשְׁפֵּיהֶם לְחוּמוֹת

וְדָתוֹת הָאֱמֶת יִשְׂאוּ בְּרָכָה / בְּמַעַשׂ טוֹב וְדָתֵיהֶם מְדֻמּוֹת

לְבָבוֹת יִשְׂמְחוּ יוֹם קוּם אֱלֹהִים / בְּעֵת יִבְנֶה לְעָרִים הַנְּשַׁמּוֹת

וְאָז יִגְלֶה לְאֹזֶן הָאֲנָשִׁים / בְּלִבּוֹתָם הֲלֹא נִגְלוּ סְתוּמוֹת

וְעַל צִיּוֹן וּבָנֶיהָ יְרַחֵם / בְּיוֹם יִנְקֹם לַנְּפָשׁוֹת הָעֲגוּמוֹת

וּמִקְדָּשׁוֹ אֱמֶת יִבְנֶה מְהֵרָה / עֲזָרוֹתָיו וְהֵיכָלוֹת פְּנִימוֹת

וְאָז יִהְיוּ בְדוּכָנָם לְוִיִּים / לְשׁוֹרֵר שִׁיר בְּדַקּוֹת עִם צְנוּמוֹת

וּמֵי גִיחוֹן אֲשֶׁר סָתְמוּ מְלָכִים / יְפֹרָצוּ וּמֵימֵיהֶם קְדוּמוֹת

וְכָל אֵלֶּה בְּדִבְרֵי הַנְּבִיאִים / וּמִקֶּדֶם בְּדָת הַחֵן רְשׁוּמוֹת

וְאַרְצֵנוּ אֱמֶת תִּתֵּן יְבוּלָהּ / בְּכָל פֶּלֶךְ וְיִבָּנוּ שְׁמָמוֹת

וְאַשְׁרֵי אִישׁ אֲשֶׁר יִרְאֶה פְלָאוֹת / בְּעֵת יִזְכּוּ נְפָשׁוֹת הַתְּמִימוֹת

יְזַכֵּנִי אֱלֹקִים חַי וְקַיָּם / אֲקַדֵּם שִׁיר לְרוֹקַע הָאֲדָמוֹת.
