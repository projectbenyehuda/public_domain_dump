


	
	אֱלֹהַי פָּתַח פִּי
	






אֱלֹהַי פָּתַח פִּי / וְגַם הֵאִיר נִשְׁפִּי / וְהֵשִׁיב יוֹם חָרְפִּי / נְגִינָתִי עָרְבָה

אֲקַדֵּם בִּתְהִלּוֹת / לְעוֹשֶׂה לִגְדוֹלוֹת / וְעִלַּת כָּל עִלּוֹת / וְסִבַּת כָּל סִבָּה

בְּשׁוּרִי נִפְלָאוֹת / עֲתִידוֹת כִּי בָאוֹת / לְעוֹלָם הָיוּ אוֹת / נְבוּאָה נִכְתָּבָה

בְּמַשָּׂא הָאָמָה / הֱיוֹתָהּ לִשְׁמָמָה / בְּאַשְׁמָה נִכְתָּמָה / וְטִירָתָהּ חָרְבָה

גְּאוֹנָהּ הוּא נִשְׁבַּר / בְּשֶׁבֶר עַל שֶׁבֶר / וְהוּבַל לַקֶּבֶר / שְׁאוֹל לוֹ הִרְחִיבָה

גְּמוּלָהּ שֻׁלַּם לָהּ / כְּרֹעַ מִפְעָלָהּ / וְנָפַל מַזָּלָהּ / וְגַם נֵרָהּ כָּבָה

דְּבָרָהּ הַנָּחוּץ / כְּבָר נִרְמַס בַּחוּץ / וְאִישָׁהּ הַשָּׁחוּץ / בְּיַד דּוּמָה נֶחְבָּא

דְּעוּ כִתּוֹת יְוָנִים / בְּלַעַג שַׁאֲנָנִים / וּבוּז לִגְאֵיוֹנִים / בְּאֶרֶץ לוֹ צָבָא

הֲמוֹן קוֹל הָרוֹעֵץ / כְּבָר יָבֵשׁ כָּעֵץ / וְשָׁבָת הַנּוֹגֵשׂ / וְשָׁבְתָה מַדְהֵבָה

הֲלֹא פִתְחוֹ נִסְגָּר / וּבְשַׂקִּים נֶחְגָּר / וְעֵדֶר בֶּן הָגָר / בְּיַד מָוֶת נִשְׁבָּה

וְשָׁבְתוּ הָאֵיבוֹת / וְגַם דִּבְרֵי רִיבוֹת / וְנִפְרָעוּ חוֹבוֹת / וְאֵין עוֹד דִּין הָבָה

וּמִצְחוֹ הָרָחֵב / כְּבָר נָס בַטַּחַב / וְעַל פָּנָיו נִסְחַב / בְּתוֹךְ אֵשׁ לֶהָבָה

זְמָן נָתַן עֶדְיוֹ / וְשִׁלַּם אֶת נִשְׁיוֹ / וְהִתְבּוֹנֵן אַיּוֹ / בְּתִקְוָה נִכְזָבָה

זְכֹר דִּבְרֵי נָבָל / וְלִשְׁאוֹלָה הוּבָל / וְקִלְלַת הַר עֵיבָל / בְּגוּפוֹ נִצָּבָה

חֲדִי יוֹנָה תַמָּה / בְּשֶׁבֶר הָאָמָה / וְקוּמִי עַד כַּמָּה / בְּרוּחֵךְ נֶעֱצָבָה

חֲבִי כִמְעַט רֶגַע / לְזֶרַע הַמְשֻׁגָּע / וְלוֹ עֵת וָפֶגַע / וּמַכָּתוֹ רַבָּה

טְמֵא לֵב עִם שָׂפָה / הֲלֹא שָׂבַע חֶרְפָּה / מְגִלָּתוֹ צָפָה / וְתוֹכָהּ תּוֹעֵבָה

טְפוּלָה בַשֶּׁקֶר / לְרֹב עַד אֵין חֵקֶר / וְהִיא כַעֲנַן בֹּקֶר / לְתֹהוּ נִגְנָבָה

יְלוּדֵי יִשְׁמָעֵאל / בְּסִפְרָם מִתְגָּעֵל / וְדַלְתָּם תִּנָּעֵל / וְשִׁמְשָׁם עָרָבָה

יְדֵיהֶם הֵם רָפוּ / וּבִפְנֵיהֶם חָפוּ / בְּשָׁרְשָׁם נִקְטָפוּ / כְּמוֹצִיאֵי דִבָּה

כְּפִירָם הַנִּזְכָּר / הֲלֹא הוּא שָׁב נֶעְכָּר / וְהָיָה כַּזָּכָר / וְנֶחְשָׁב כִּנְקֵבָה

כְּבוֹדוֹ בִּזָּיוֹן / וְתַחַת אַפִּרְיוֹן / לְמַחְלוֹן וּלְכִלְיוֹן / וְנַפְשׁוֹ שׁוֹבֵבָה

לְקוּחִים הַהַגְרִים / לְהַשְׁמִיד וּלְהַחְרִים / כְּמַהְפֵּכַת זָרִים / לְמֶמְשַׁלְתָּם קִצְבָּה

לְדַכֵּא בִּנְגָעִים / וּמַכּוֹת וּפְצָעִים / הֲמוֹן זֵדִים רָעִים / לְנַעַר עִם שֵׂיבָה

מְנוּחָתָם כַּמָּה / בְּתַתְקַ״ף נִשְׁלָמָה / שְׁנוֹת מַלְכֵי אָמָה / כְּרֶגַע נֶחְשָׁבָה

מְנוּחַת הַחִגֵּר / בְּתֹפֶת עַל מַסְגֵּר / וְכִסְאוֹ מִתְמַגֵּר / וְרַגְלוֹ נֶעְקָבָה

נְשִׂיאִים קוֹל שָׁאוֹן / לְכָל חַי דֵּרָאוֹן / כְּלַהַב צִמָּאוֹן / וְנַפְשׁוֹ דָאָבָה

נְשִׂיא רֹאשׁ כָּל קֵדָר / בְּגֵיהִנָּם קָדָר / בְּפָנָיו אֵין הָדָר / וְעֵינוֹ הִרְקִיבָה

סְגֻלָּה הוּא לָכֶם / וְכַבּוּ אוֹר אֶשְׁכֶם / וּמֶלֶךְ הוּא לָכֶם / בְּנוּ לָכֶם מַצֵּבָה

סְבִיבָתוֹ פַּחַת / בְּאֵשׁ מִתְלַקַּחַת / וְנַפְשׁוֹ בַּשַּׁחַת / בְּכָל יוֹם הִיא שָׁבָה

עֲדֵי אֵשׁ עַל גּוּפוֹ / וְדוּמָה יָנִיפוֹ / בְּקֶלַע וּדְחָפוֹ / וְנִבְהַל: מִי זֶה בָא

עֲדַת חָנֵף גַּלְמוּד / כְּמִפְעָלָם לָמוּד / בְּלִיַּעַל לָמוּד / בְּשִׂנְאָה גַּם אֵיבָה

פְּנֵי אֵל חַי תִּרְאוּ / בְּנֵי יַעֲקֹב וּגְאוּ / וְהַבִּיטוּ וּרְאוּ / תְּחִלַּת הַטּוֹבָה

פְּנוּ אֶל אֵל צוּרְכֶם / וּמִיכָאֵל שַׂרְכֶם / וְהִנֵּה יִשַּׁרְכֶם / בְּנֶפֶשׁ חֻצָּבָה

צְדָקוֹתָיו רַבּוֹת / וּמִדּוֹתָיו טוֹבוֹת / וּבִזְכוּת הָאָבוֹת / גְּאֻלַּתְכֶם קָרְבָה

צְדָקוֹתַי קַלּוּ / מְזִמּוֹתַי יַעֲלוּ / וְעֵינַי לָךְ דַּלּוּ / וְעָשְׁקָה לִי עָרְבָה

קְחָה נָא מִנְחָתִי / וְעַל רֹב אַשְׁמָתִי / קְבוּעָה חַטָּאתִי / שְׂעִירָה אוֹ כִּשְׂבָּה

קְדוֹשִׁים אַחֲרֵי מוֹת / תְּפִלּוֹת נֶעֱלָמוֹת / לְתוֹכֵן עוֹלָמוֹת / וְרוֹכֵב בִּמְסִבָּה

רְצֵה אֵל חַי עַמָּךְ / וְשַׁכְלֵל אוּלַמָּךְ / וְעַם עָנִי וָמָךְ / בְּצִיּוֹן הֵיטִיבָה

רְצוּעֵי סִפֶּךָ / וְסוֹבְלֵי קִצְפֶּךָ / לְמַעַן צוֹפֶךָ / בְּשִׁמְךָ נִתְנַבָּא

שְׁעֵה אֶת שַׁוְעָתָם / מְחֵה אֶת דִּמְעָתָם / וְתִקְרַב רִנָּתָם / וְתִרְצֶה בִּתְשׁוּבָה

שְׁנָתִי בָּהּ אוֹרָה / בְּהַשׁלַ״ג לִיצִירָה / תְּחִלָּה לִבְשׂוֹרָה / לְאֻמָּה נֶאֱהָבָה

תְּקַבֵּל מִלּוּלִי / וְתִשְׁמַע פִּלּוּלִי / וְתַאֲזִין הִלּוּלִי / בְּרָצוֹן וּבְחִבָּה

תְּמַהֵר בִּמְשִׁיחִי / וְיֶעֱרַב לָךְ שִׂיחִי / כְּרֵיחַ נִיחוֹחִי / כְּעוֹלָה וּנְדָבָה

חֲקֹר אֹרֶךְ מַעֲנִית / לְחוֹתֵם הַתָּכְנִית / וְאַל תּוֹסִיף שֵׁנִית / תְּדִינוֹ תּוֹךְ לַבָּה

זְכֹר נָא כִּי כָרַע / וְחַטָּאתוֹ פָּרַע / בְּחַסְדֶּךָ תִּקְרַע / שְׁטָרוֹ כִּי נִגְבָּה

קְדוֹרַנִּית רָזֶה / בְּמִשְׁמָנָיו אֶחֱזֶה / בְּעוֹדִי חַי בָּזֶה / וְלָעוֹלָם הַבָּא

וְאִם לָךְ עֹז חֶבְיוֹן / כְּבוֹדָךְ תּוֹךְ צִיּוֹן / וְאַתָּה הוּא עֶלְיוֹן / עֲלֵי כָל מֶרְכָּבָה.
