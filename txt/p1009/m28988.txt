


	
	יַקְדִּישׁוּן בַּיּוֹם שָׁלוֹשׁ
	








יַקְדִּישׁוּן בַּיּוֹם שָׁלוֹשׁ / מָרֵי נֶפֶשׁ עֲשׁוּקָה

אֶת שֵׁם אֵל חַי לִדְרֹשׁ / בָּעֵצָה הָעֲמֻקָּה





זֹוֹכֵר חַסְדּוֹ לִתְשׁוּאוֹת / הֵמָּה תָמִיד עֲבָדָיו

חוֹכִים לְפָנָיו לֵירָאוֹת / שׁוֹמֵר רַגְלֵי חֲסִידָיו

מָתַי הַקֵּץ לַפְּלָאוֹת / כָּמוּס עִם מַגִּידָיו

זוֹקֵף לִכְפוּפֵי רֹאשׁ / מַצִּיל מִכָּל צוּקָה

כַּמָּה אוֹחִיל עַד בּוֹשׁ / לַדֶּרֶךְ הָרְחוֹקָה.



כָּשַׁל כֹּחִי עַד עֶרֶב / נֹפֶשׁ לֹא מָצָאתִי

חָרָה לִבִּי מֵחֹרֶב / מִסֵּבֶל יָגַעְתִּי

מָתַי רִנָּתִי תִּקְרַב / אֶל חֶדֶר הוֹרָתִי

אֶת עַם אָמָה לִנְתֹּשׁ / בּוּקָה וּמְבֻלָּקָה

כִּי עַל גַבִּי תַחֲרֹשׁ / מַעֲנִית לֹא נִתָּקָה.



רָקַד לוֹחֵם כִּי יָקַר / מִנִּי מַלְאָךְ חוֹנֶה

נֶאֱלַמְתִּי עַד אֵין חֵקֶר / לֹא מָצָאתִי מַעֲנֶה

מָתַי נַפְשִׁי בָּךְ תִּיקַר / הַנֶּהֱרָסוֹת תִּבְנֶה

נֶחְשָׁב לַעֲנָוָה רֹאשׁ / הַשָּׂם לַשֵּׁם חֻקָּה

נֶהְפַּךְ לַעֲנָה וָרֹאשׁ / גֶּפֶן הַשּׂוֹרֵקָה.



יָמַי כָלוּ מִסֵּבֶר / לִנְפֹּל יַלְדֵי רֶשַׁע

וּכְפֶתַע אֶשְׁכֹּן קֶבֶר / הַרְפֵּה מֶנִּי הָשַׁע

מָתַי יִצְלַח גֶּבֶר / נָד אַחַת וָתֵשַׁע

צַמֶּרֶת תִּדְהָר וּבְרוֹשׁ / יִבְגֹּד מַעְיָן הִשְׁקָה

מֵעַתָּה לֹא תֵבוֹשׁ / הַבַּת הַשּׁוֹקֵקָה.



הָקֵם סֻכָּה נוֹפֶלֶת / מֵרֹאשׁ צוּרִים תּוֹפִיעַ

אָז תִּמָּלֵא מִשְׁאֶלֶת / יוֹם עָם עָנִי תוֹשִׁיעַ

מָתַי אָמָה נוֹבֶלֶת / וּבְנֵי עֵשָׂו תַּרְשִׁיעַ

תִּלְבֹּשֶׁת נָקָם תִּלְבֹּשׁ / לָעֵדָה הַחֲשׁוּקָה

כָּל מְעֻוָּתִי תִּכְבֹּשׁ / לֹא תִהְיֶה עוֹד חֲקוּקָה.
