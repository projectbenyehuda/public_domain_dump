


	
	אֲקַדֵּם שִׁיר לְאֵל שׁוֹכֵן מְרוֹמוֹת
	



מקור: זכריה אלצ׳אהרי: ספר המוסר






אֲקַדֵּם שִׁיר לְאֵל שׁוֹכֵן מְרוֹמוֹת / בְּשִׁגָּיוֹן יְנֻגַּן עַל עֲלָמוֹת

בְּרֹב חַסְדּוֹ אֲשֶׁר עָשָׂה לְעַבְדּוֹ / בְּנִפְלָאוֹת גְּדוֹלוֹת וַעֲצוּמוֹת

גְּמָלַנִי וְרִפָּא אֶת בְּשָׂרִי / וְקָרַם עוֹר וְגִידִים וַעֲצָמוֹת

דְּאָבָה רָחֲקָה מִנִּי וְנַפְשִׁי / וְרַעְיוֹנַי קְדוֹשִׁים אַחֲרֵי מוֹת

הֲלֹא אָזְנִי אֲשֶׁר שָׁמְעָה דְּבָרוֹ / בְּתַרְדֵּמָה וְחְֶזְיוֹן בַּחֲלוֹמוֹת

וְלִי גִּלָּה בְּיָמִים הַקְּדוּמִים / וְשָׁכַחְתִּי בְּעֹצֶם הַתְּנוּמוֹת

זְמַנִּים חָלְפוּ כַמָּה וְעָבְרוּ / וְאֵין דּוֹרֵשׁ לְמֵבִין תַּעֲלוּמוֹת

חֲדָשִׁים לַבְּקָרִים אֶזְכְּרֶנּוּ / וּבִשְׂמֹאלוֹ וּבִימִינוֹ נְעִימוֹת

טְמוּנִים רֹב חֲסָדָיו תּוֹךְ לְבָבִי / בְּיוֹם פָּדָה לְנַפְשִׁי מִתְּהוֹמוֹת

יְחַוּוּ מַעֲשָׂיו נִבְרָא וְנוּצָּר / וְתַנִּינִים וְאָדָם עִם בְּהֵמוֹת

כְּמוֹ רִפָּא לְעַצְמוֹתַי וְהֶעֱמִיד / גְּוִיָּה שָׂבְעָה מִמַּהֲלוּמוֹת

לְדוֹרְשָׁיו בֶּאֱמֶת רַחוּם וְחַנּוּן / וְלָאוֹיְבִים אֱמֶת הוּא אֵל נְקָמוֹת

מְחַיֶּה הַנְּפָשׁוֹת בַּגְּוִיּוֹת / וְהַמֵּתִים יְחַיֶּה מֵעֲרֵימוֹת

נְפוֹצִים הֵם בְּאַפְסֵי הַקְּצָווֹת / וְנִדָּחִים שְׂבֵעִים בַּמְּהוּמוֹת

סְלַח נָא לַעֲווֹן בָּנִים מְקַוִּים / יְשׁוּעָתָךְ תְּמוּנַת הָרְקָמוֹת

עֲמוּסִים הַתְּלָאוֹת מִנְּעוּרִים / וְעַד זִקְנָה בְּרֹב פַּחַד וְאֵימוֹת

פְּדֵה עַמָּךְ וּמִיַּד לוֹחֲצֵיהֶם / הֱיֵה לָהֶם לְבִצָּרוֹן וְחוֹמוֹת

צְדָקָה תַּעֲשֶׂה כַיּוֹם לְבָנִים / וְעַל אוֹיְבִים הֲלֹא תַחְגֹּר לְחֵמוֹת

קְחָה שִׁירִי מְקוֹם זֶבַח וּמִנְחָה / מְקוֹם עוֹלָה וְחַטָּאת וַאֲשָׁמוֹת

רְצֵה שִׂיחִי וְעָרְבֵנִי לְעוֹלָם / בְּיוֹם תָּבִיא בְּמִשְׁפָּט עַל עֲלֻמּוֹת

שְׁעֵה שַׁוְעִי וְזָכְרֵנִי לְחַיִּים / וְכָתְבֵנִי בְּטוֹבוֹת הַחֲתוּמוֹת

תְּזַכֵּנִי לְהִתְעַדֵּן בְּעֶדְנָךְ / מְחַיֶּה הַפְּגָרִים בַּנְּשָׁמוֹת.
