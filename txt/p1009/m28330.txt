


	
	אֵל מִסְתַּתֵּר לִמְאֹד נֶעְלָם
	




אקרוסטיקון: א״ב אני זכריה בן שעדיה



אֵל מִסְתַּתֵּר / לִמְאֹד נֶעְלָם / יוֹדוּךָ אֵל / עַמִּים כֻּלָּם

בָּרָאתָ רֹאשׁ / שֵׂכֶל נִבְדָּל / מִן הָעֶשֶׂר / יַשְׂכִּיל שִׂכְלָם

גַּלְגַּל סוֹבֵב / לִנְקֻדָּתוֹ / בּוֹ כּוֹכָבִים / מֵאִיר הִלָּם

דַּרְכּוֹ סוֹבֵב / עַל הַיְסוֹדוֹת / תַּחְבּוּלוֹתָיו / יָצָא פָעֳלָם

הֵן יִתְהַוּוּ / הַמִּתְהַוִּים / צוּרוֹת יוֹצְאִים / מֵהֵיכָלָם

וַתִּבְחַר לָךְ / מִכָּל אֵלֶּה / אָדָם מֵבִין / מַשְׂכִּיל מָשְׁלָם

זָהָב מוּפָז / מִן הַבִּרְיוֹת / מֻבְדָּל מֵהֶם / בָּרִיא אוּלָם

חָטְאוּ בַּגָּן / הוּא עִם אִשְׁתּוֹ / עֵץ הַדַּעַת / סִבַּת אָכְלָם

טָבְעוּ בִּיוֵן / יַם סִכְלוּתָם / עֶלְיוֹן צוֹפֶה / כָּל מַעֲלָלָם

יָחִיד עָנָה / צֵא מִן הַגָּן / רַב לָךְ שֶׁבֶת / עוֹד בִּגְבוּלָם

כַּבִּיר כֹּחַ / יָסַד אֶרֶץ / עַל מַתְכֻּנְתָּה / עַד בּוֹא מַעֲלָם

לַנִּשְׁכָּחִים / מִנִּי רָגֶל / הַנִּבְטָחִים / עַל רֹב חֵילָם

מָחָה אוֹתָם / מִן הָאָרֶץ / הֵם נִשְׁקָעוּ / תּוֹךְ מַבּוּלָם

נֹחַ צַדִּיק / תּוֹךְ תֵּיבָתוֹ / חַי בִּמְצוֹלוֹת / נָשְׂאוּ קוֹלָם

סָרְחוּ דּוֹרוֹת / הַפַּלָּגָה / עֵת נוֹעָצוּ / עַל מִגְדָּלָם

עַיִן תִּלְעַג / מֵרוֹם לָהֶם / גָּמַר אֹמֶר / עַל בִּלְבּוּלָם

פִּזַּר אוֹתָם / אֶל כָּל פֵּאוֹת / לִלְשׁוֹנוֹתָם / כִּי סָר צִלָּם

צָמַח אֵיתָן / הָאֶזְרָחִי / הֵאִיר תֵּבֵל / עַל רֹאשׁ תִּלָּם

קָדוֹשׁ נֶעְקָד / עַל מִזְבֵּחַ / צָעֲקוּ חוּצָה / הֵן אֶרְאֶלָּם

רוֹעֶה הַצֹּאן / עֶשְׂרִים שָׁנָה / בֵּין אֹהָלִים / חָזָה סֻלָּם

שִׁבְטֵי הַשֵּׁם / עֵת נִטָּעוּ / גַּם שֹׁרָשׁוֹ / הָאֵל עִלָּם

תָּמִיד הָיוּ / עוֹבְדֵי אֵל חַי / עַד הַחֲלוֹמוֹת / יוֹסֵף גִּלָּם

אַחֲרֵי נִמְכַּר / אֶל מִצְרַיִם / אֶל יַד פַּרְעֹה / צוֹפֶה הָשְׁלָם

נִתַּן מֶלֶךְ / עַל כָּל אַרְצוֹ / בִּשְׁנוֹת שׂוֹבַע / עֵת הַחִלָּם

זָכַר הַחֲלוֹם / גַּם פִּתְרוֹנוֹ / שִׁלַּם לָהֶם / לֹא כִגְמוּלָם

כָּלוּ יָמָיו / גַּם דּוֹרוֹתָיו / וַיָּשִׂימוּ / חֹמֶר סִבְלָם

רָאָה אֵל חַי / זֶרַע יַעֲקֹב / יָרְדוּ עַד כִּי / עָפָר מָשְׁלָם

יָחִיד בָּחַר / מֹשֶׁה עַבְדּוֹ / לִגְאֹל עַמּוֹ / מִיַּד נִכְלָם

הֵבִיא עָלָיו / עֶשֶׂר מַכּוֹת / עַד שָׁב פַּרְעֹה / בּוֹשׁ גַּם נֶאֱלָם

בָּקַע יַמִּים / וַיַּשִׁירוּ / הֵן לִקְטַנָּם / הֵן לִגְדוֹלָם

נָתַן תּוֹרָה / בִּשְׁנֵי לוּחוֹת / תַּרְיַ״ג מִצְווֹת / דָּת הִנְחִילָם

שָׁמְעוּ כֻלָּם / עַל הַר סִינַי / גַּם שָׁמַיִם / נָתְנוּ טַלָּם

עָנוּ כֻלָּם / נַעַשׂ נִשְׁמָע / מִן הַתּוֹעִים / אֵל הִבְדִּילָם

דֶּרֶךְ בַּיָּם / הוּא הַנּוֹתֵן / בַּמִּדְבָּרוֹת / מָן הֶאֱכִילָם

יִסַּד הָעֹז / מִפִּי עוֹלְלִים / מָלֵא רַחֲמִים / בּוֹרֵא עוֹלָם

הַשָּׂם נַפְשִׁי / מִן הַחַיִּים / אֵל מִסְתַּתֵּר / לִמְאֹד נֶעְלָם.
