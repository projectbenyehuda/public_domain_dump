


	
	יוֹנָה רְאִיתִיהָ בְּתַכְלִית הַיֳּפִי
	








יוֹנָה רְאִיתִיהָ בְּתַכְלִית הַיֳּפִי / שִׂכְלִי עֲזַבְתִּיו לַהֲמוֹן מִלֶּיהָ

קָרְעָה סְגוֹר לִבִּי בְּיַם חֶשְׁקָהּ עֲדֵי / כִּמְעַט וְאֶגְוַע מִשְּׁאוֹן גַלֶּיהָ

לוּ יֶחֱזֶה אוֹתָהּ חֲסִיד הַדּוֹר יְהִי / נוֹפֵל בְּרִשְׁתָּהּ וַחֲבָלֶיהָ

לוּ שָׁמְעוּ שִׁמְעָהּ בְּנֵי אֵלִים אֲזַי / נִלְאוּ לְהַשִּׂיג לַאֲבַק רַגְלֶיהָ

תַּחְפִּיר פְּנֵי סַהַר בְּמִלּוּאוֹ וְעוֹד / תַּזְהִיר בְּלָבְשָׁהּ אֶת מְעִילֶיהָ

גָּנְבָה בְּחִכָּהּ אֶת לְבַב הַחוֹשְׁקִים / נִכְווּ וְנִשְׂרָפוּ מֵהֲבָלֶיהָ

מֵאוֹר פְּנֵי הִלָּהּ מְאוֹרוֹת חָפְרוּ / לָקְחָה מְאוֹר שֶׁמֶשׁ מִזְּבוּלֶיהָ

לָהּ צוֹבְאִים שָׂרִים בְּשֵׁשׁ מֵאוֹת וְעוֹד / נוֹסָף שְׁלֹשׁ עֶשְׂרֵה עַל חֲיָלֶיהָ

רָאשֵׁי גְּיָסוֹת הֵם וְעוֹרְכִים הַקְּרָב / נֶגֶד רְחוֹקִים מִגְּבוּלֶיהָ

מִשְׁתַּחֲוִים נֶגְדָּהּ בְּאַפִּרְיוֹן וְגַם / תָּמִיד יְסַפֵּרוּ מַעֲלָלֶיהָ



עָפְרָה צְבִיַּת חֵן וְשָׁנֶיהָ כְּמוֹ / הַשְׁלַ״ח וְיוֹמַיִם לְתַגְמוּלֶיהָ



לֹא זָקְנָה לֹא כָהֲתָה כִּי כָל מְאוֹר / יָפִיץ לְכָל מֵאִיר מִכְּלִילֶיהָ

שֹׂבַע לְכָל רָעֵב בְּגוֹ עֶדְנָהּ וְהִיא / תַרְוֶה צְמֵאִים מִשְּׁבִילֶיהָ

הָאֵל אֲשֶׁר לוֹ עֲלִילוֹת נִתְכְּנוּ / יִפְסֹק מְנָתִי בֵּין אֲצִילֶיהָ

אֶזְכֶּה לְהִסְתּוֹפֵף בְּהוֹד צִלָּהּ וְאִם / יַעֲזֹר וְאֶהְיֶה מִבְּעָלֶיהָ

אִינַק שְׁדֵי חֶמְדָּה וְאֶשְׁתַּעְשַׁע בְּעֵת / אוּכַל עֲלוֹת עִמָּהּ לְמִגְדָּלֶיהָ

נַמְתִּיק לְסוֹד יַחְדָּו וְאֶתְהַלֵּךְ לְאַט / נִרְצָע כְּבֶן בַּיִת וְאַמְשִׁילֶהָ

זֶה הוּא אֲשֶׁר אֶתְאָו בְּכָל רֶגַע וְעֵת / זֶה מַחֲמַד נַפְשִׁי וּמִשְׁאָלֶיהָ

תֵּלֵךְ נְתִיב יֹשֶׁר וְתִתְעַדֵּן בְּגַן / חָכְמָה תְּחַיֶּה אֶת בְּעָלֶיהָ.
