


	
	כְּמִתּוֹךְ הַחֲלוֹם אֶרְאֶה תְּמוּנָה
	








כְּמִתּוֹךְ הַחֲלוֹם אֶרְאֶה תְּמוּנָה / בְּשׁוּרִי מִפְּאַת צָפוֹן עֲנָנָה

כְּתַבְנִית אִישׁ וְהוּא עוֹטֶה עֲנָוָה / מְעִיל צֶדֶק וְעַל פָּנָיו שְׁכִינָה

כְּלִיל שֵׂכֶל אֲחִי יֹפִי צְבִי חֵן / וְתִפְאָרָה וְאֵין קֵץ לַתְּכוּנָה

וְשִׁבְעַת הַחֲבֵרִים הַחֲשׁוּבִים / סְבָבוּהוּ וְהָיוּ בּוֹ שְׁמוֹנָה

חֲלוֹמִי הֶעֱבִירַנִי לְמוּלָם / וְלִבִּי עֵר גְּוִיָּה הַיְּשֵׁנָה

וְשָׁאַלְתִּי לְמִי אֵלֶּה בְּתֵבֵל / הֱשִׁיבוּנִי לְאַבְרָהָם לְמִקְנָה

צְבִי הָפַךְ נְאוֹת מִדְבָּר לְעֵדֶן / וְהֵמִיר אֶת יְמֵי זֹקֶן בְּעֶדְנָה

צְמָחִים יָבְשׁוּ לוּלֵא מְטָרָיו / יְחַיּוּ כָּל פְּרִי נֵטַע וְכַנָּה

וְכָל אִישׁ שָׂשׂ וְכָל עֵץ צָץ לְמֵימָיו / מְשָׁכָם הוּא כְּפַרְפַּר וַאֲמָנָה

צְבִי מָאַס בְּהַבְלֵי הַזְּמַנִּים / וּמִלֵּא אוֹצְרוֹת בִּין כָּל פְּנִינָה

צְבִי מָאַס בְּחֵן יַלְדוּת וְשַׁחְרוּת / וְהִרְבִּיץ דָּת וְתוּשִׁיָּה מְכֻנָּה

יְדֵי לִבִּי קְרָאתִים וְלֹא מְצָאתִים / עֲזָבוּנִי בְּלֹא שֵׂכֶל וּבִינָה

חֲרוּזֵי שִׁיר וְלָמָּה תִפְשְׁעוּ בִּי / לְכוּ הִקָּבְצוּ אֵלַי גְּשׁוּ נָא

קְשַׁרְתֶּם כֻּלְּכֶם עָלַי וְנַסְתֶּם / וְלִבִּי יַהֲלֹךְ אָנֶה וְאָנָה

יְדַעְתִּיכֶם תְּמוֹל רָצִים לְיָדִי / הֱיִיתֶם לִי כְּסוֹחֵרָה וְצִנָּה

נְפוֹצֹתֶם בְּבַקְשִׁי לֶאֱסָפְכֶם / מְלִיצַי דַּבְּרוּ אֵלַי נְכוֹנָה

וְעַתָּה אֵלְכָה לִי אֶל מְקוֹמִי / עֲלֵיכֶם אֶמְצָא תָּמִיד תְּלֻנָּה

שְׂאוּ עָבִים שְׁלוֹמִי אֶל אֲדוֹנַי / וְלָהֶם חַדְּשׁוּ מַלְכוּת יְשָׁנָה

וְסַפֵּרוּ קְצַר רוּחִי וְנַפְשִׁי / וְאוֹיְבַי נָתְנוּ עַל לֵב מְגִנָּה

וְלוּלֵי חַסְדֵי אֵל נִהֲגוּנִי / וְתוֹרַת אֵל בְּכָל עֵת לִי לְמָנָה

אֲזַי כָּלָה שְׁאֵרִי עִם לְבָבִי / וְנַפְשִׁי יַעֲמֹד כַּיָּם שְׁאוֹנָהּ

אֱלֹהִים חַי גְּאָלֵנוּ בְּרָצוֹן / וְאֵיטִיב שִׁיר בְּרֵאשִׁית כָּל תְּחִנָּה

וְזָכְרָה אֶת עֲבָדֶיךָ בְּעֵת צָר / לְמַעַן אָב יְדָעָךְ בֶּאֱמוּנָה

וְשָׁלוֹם רָב כְּחוֹל יַמִּים לְיֶתֶר / בְּסוֹד אַהֲבָה בְּלִבִּי הִיא צְפוּנָה

וְעֵת תָּבוֹא פְּקִידָה לַיְּהוּדִים / וּמִצְעָדַי לְדִקְלָה הַמְּדִינָה

וְשׁוֹקַיִם לְשׁוֹקַיִם יְרוּצוּן / בְּשֶׁבֶר מִבְצְרֵי אָמָה דְּשֵׁנָה

וְנֵרָאֶה פְּנֵי אֵל לַחֲזוֹתוֹ / וְנָבוֹא שַׁעֲרֵי צִיּוֹן בְּרִנָּה.
