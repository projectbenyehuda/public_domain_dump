


	
	אֲשֶׁר שָׁת מַרְכְּבוֹתָיו עַל עֲנָנִים
	








אֲשֶׁר שָׁת מַרְכְּבוֹתָיו עַל עֲנָנִים / וְאֶרֶץ הֶעֱמִיד בִּבְלִי אֲדָנִים

בְּחָרוּךָ אֲבוֹת עוֹלָם לְאָדוֹן / וּבַעֲבוּרָם פְּתַח מַסְגֵּר בְּחוּנִים

גְּאוֹנָם שַׁח וְשָׁפְלוּ עַד לְעָפָר / לְךָ צָעֲקוּ עֲדֵי חָרוּ גְּרוֹנִים

דְּאָגוֹתָם בְּכָל עוֹנָה וְרֶגַע / לְמִיּוֹם עָבְרוּ אֹרְחוֹת דְּדָנִים

הֲמוֹן קֵדָר אֲשֶׁר הִקְדִּיר פְּנֵיהֶם / וְלֹא נָשָׂא פְּנֵי בָנִים הֲגוּנִים

וְעוֹד הָאָח אֲשֶׁר נוּלָּד לְצָרָה / וְעֶבְרָתוֹ לְרֹב יָמִים וְשָׁנִים

זְדוֹן לִבּוֹ בְּתוֹלֶדֶת נְזִידוֹ / וְאָרַח הוּא לְחֶבְרַת הַזְּדוֹנִים

חֲמוּשִׁים עַל בְּנֵי תוֹרָה וְדָתָם / בְּקִנְאָתָם הֲלֹא מָלְאוּ חֲרוֹנִים

טְמֵאֵי לֵב וְנֶפֶשׁ עִם גְּוִיָּה / וּמַרְבִּיתָם בְּמִרְמוֹתָם טְעוּנִים

יְמִינָם הִיא יְמִין שֶׁקֶר בְּפִיהֶם / וְעָרְמוֹתָם חֲדָשִׁים גַּם יְשָׁנִים

כְּבוֹדָם בּוּז וְרֹב זָדוֹן וְקָלוֹן / וְנַפְשׁוֹתָם בְּקִרְבּוֹתָם צְפוּנִים

לְשׁוֹנָם יַחֲלִיקוּן מַחֲמָאוֹת / וְנֶהְדָּרִים בְּבִגְדֵיהֶם לְבָנִים

מְשֹׁךְ חַסְדָּךְ לְיוֹדְעֶיךָ בְּחֶמְלָה / וְהַשְׁבֵּת צָר מְשַׁלֵּחַ מְדָנִים

נְתָנַנִי בְרֹאשׁ צוּרִים וְאִבַּד / לְמַחְמַדֵּי חֲכָמַי עִם נְבוֹנִים

סְגוֹר פִּיהוּ וְלֹא יָעִיר חֲמָתוֹ / עֲלֵי בִנְךָ וְלֹא יָבוֹא סְגָנִים

עֲמֹד לִימִין סְגֻלָּתָךְ כְּקֶדֶם / וְחַטָּאתָם כְּבֹשׁ וּמְחֵה עֲווֹנִים

פְּדֵנִי נָא כְּאָז מִבֵּית עֲבָדִים / וְאֶהְגֶּה עוֹד יְקָרָה מִפְּנִינִים

צְמֵאִים לָךְ לְעֶזְרַת עָם כְּמֵהִים / וְלֹא יִהְיוּ בְּצִדֵּינוּ צְנִינִים

קְרָא לָהֶם לְאַרְבַּע הַקְּצָווֹת / וְהֵם יַעְנוּ גְּדוֹלִים עִם קְטַנִּים

רְפָא שִׁבְרָם בְּנִטְעֵי שַׁעֲשׁוּעִים / שְׁתוּלִים בָּם וְיִהְיוּ רַעֲנַנִּים

שְׁנוֹתֵינוּ קְצָרוֹת נֶעֱצָרוֹת / וְהֵם שִׁבְעִים וּבִגְבוּרוֹת שְׁמֹנִים

תְּנִיחֵנוּ וְתַנְחֵנוּ בְצֶדֶק / וְרוּחַ חֵן לְדַבֵּר תַּחֲנוּנִים

אֲדוֹן עוֹלָם וּמֵבִין תַּעֲלוּמוֹת / רְצֵה שִׂיחַי וְאִם הֵם אַחֲרוֹנִים

בְּחַסְדֶּךָ פְּנֵה אֶל מַעֲנֵה פִי / וְנִשְׁמָתִי תְּצַו לִרְעוֹת בְּגַנִּים

נְגִינָתִי לְךָ תֶּעְרַב וְתִזְכֹּר / בְּיוֹם שֵׂאתִי חֳלָיִים נֶאֱמָנִים

רְאֵה יוֹם רָחֲקוּ דּוֹדַי וְרֵעַי / וְנֶעְדָּרוּ אֲהוּבִים רַחֲמָנִים

בְּנָשְׂאִי אֶל דְּבִיר קָדְשָׁךְ לְיָדִי / וְהֶעְלִיתָ אֲרוּכָתִי בְאוֹנִים

נְפוּגוֹתִי וְנִדְכֵּיתִי כְהַיּוֹם / וְהָאוֹהֵב פְּצָעָיו נֶאֱמָנִים

חֲמֹל עַל בֵּן בְּכוֹר שָׁפָל וְנִבְזֶה / בְּגָלוּתוֹ וּמַכְעִיסָיו חֲתָנִים

לְךָ יִכְרַע שְׁחָרִים וַעֲרָבִים / מְיַחֵל רַחֲמֵי הָאָב לְבָנִים

קְרָא נָא טוֹב לְדַל שׁוֹאֵל בְּפִתְחָךְ / וּפָנָיו בוֹשׁ וְעָלָה קִמְּשׁוֹנִים

עֲנֵה שִׂיחַ מְנַצֵּחַ מְזַמֵּר / אֲשֶׁר שָׁת מַרְכְּבוֹתָיו עַל עֲנָנִים.
