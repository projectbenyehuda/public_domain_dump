


	
	מָתוֹק לְמִצּוּף וּפַנַּג
	








מָתוֹק לְמִצּוּף וּפַנַּג / מוּל צוּר וְלִבּוֹ כְדוֹנַג

מִצְוַת חֲכָמִים לְעַנַּג / שַׁבָּת וְלֵידַע בְּסוֹדָהּ





זֹהַר אֱמוּנָה תְּסַיֵּם / מֵתִים בְּקָרוֹב תְּחַיֵּם

זָכוֹר וְשָׁמוֹר לְקַיֵּם / בַּכּוֹס אֱמֶת הִיא כְּבוּדָּה





כַּבֵּד אֱלֹהִים וְתוּחָן / תַּשְׁלִים נְשָׁמוֹת בְּרוּחָן

כִּסֵּא וּמִטָּה וְשֻׁלְחָן / עָרוּךְ וּמָאוֹר לְהוֹדָהּ





רוּמָה לְמַחְשָׁב לְהוֹרִים / אַחַר שְׁלֹשָׁה כְּתָרִים

רִקְמָה תְּמוּרַת שְׁחוֹרִים / תֶּאֱרֹג לְכַלָּה בְּבִגְדָהּ





יָקָר בְּעֵינֵי אֱלֹהֵי / עוֹלָם צְבָאוֹת בְּגָבְהֵי



יוֹ״ד הֵ״א תְּחַבֵּר לְוָי״ו הֵ״א / תַּשְׁלִים לְנַפְשָׁךְ בְּעוֹדָהּ





הוֹדַע לְמַשְׁחִיר וּמַלְבִּין / נַפְשׁוֹ חֲצוּבָה מְקוֹר בִּין

הַסּוֹד לְמַשְׂכִּיל וּמֵבִין / רָמוּז בְּסוֹף הָעֲבוֹדָה





בּוֹחֵר אֱמוּנִים בְּרוּחָם / בּוֹחֵל כְּסִילִים בְּלִקְחָם

בָּשָׂר וְיַיִן וְלֶחֶם / מִשְׁנֶה בְּשַׁבָּת לְעָנְדָהּ





נָכוֹן לְחֹמֶר וְצוּרָה / עוֹמֵד בְּחוֹמָה בְּצוּרָה

נוֹפֶשׁ לְנֶפֶשׁ יְתֵרָה / תִּמְצָא וְאַתָּה נְגִידָהּ





סוֹף תַּהֲלֹךְ חִישׁ לְאוֹרָהּ / עַד יָם תְּשַׁלַּח קְצִירָהּ

סוֹד הַשְּׁבִיעִית מְקוֹרָהּ / גַּם הָעֲשִׂירִית יְסוֹדָהּ





עוֹטֶה מְעִיל הוֹד לְאַהֲבַת / אֵל חַי וּמֵכִין בְּנִדְבַת

עֹנֶג מְעֻלֶּה לְשַׁבַּת / לִקְדוֹשׁ יי חֲמוּדָה





דּוֹרֵךְ בְּדַרְכֵי יְשִׁישִׁים / מַשְׂכִּיל וְטוֹב עִם אֲנָשִׁים

דִּבֶּר בְּחֵפֶץ קְדוֹשִׁים / נֶפֶשׁ דְּחֵה מִיְּקוֹדָהּ





יָבוֹא בְּשִׂמְחָה לְגַנּוֹ / יֹאכַל מְגָדִים בְּעֶדְנוֹ

יָשִׁיב אֱלֹהִים לְמַעֲנוֹ / נֶפֶשׁ חֲסִידוֹ לְעָמְדָה





הוּא הַמְקָרֶה בְמַיִם / גָבְהֵי שְׁחָקִים בְּמַיִם

הוּא גוֹאֲלָהּ פַּעֲמַיִם / שָׁלֹשׁ וְתָשׁוּב לְנֹאדָה





יַא כַאלק אַלכּל כּן לִי / ואן לַם תַּכּוּן לִי פַמַן לִי



(בורא הכל, עמוד לימיני! שאם אין אתה לי, מי לי?)
