


	
	עַל מִשְׁפְּחוֹת הַמּוֹשְׁלִים
	








עַל מִשְׁפְּחוֹת הַמּוֹשְׁלִים עָבַרְתִּי / אֶתְמוֹל לְאֶרֶץ נוֹגְנִים שִׁחַרְתִּי,

כִּנּוֹר לְנַגֵּן טִפְּחָה יָדִי, וְעַל / יָמִין יְתָרַי בַּכְּלִי קָשַׁרְתִּי.

וָאְנַגְּנָה – אוּלַי בְּיוֹם אֶתְעַדְּנָה / יִרְוַח לְמָרֵי לֵב – וּבַשִּׁיר שַׁרְתִּי.

אִם לֹא בְיָמַי – לוּ בְמוֹתִי יִמְצְאוּ / שִׁיר אַחֲרֵי עַל הַר נְבוֹ קֻבַּרְתִּי

״אִלּוּ כְגָרְנִי קָם לְשִׁיר נָבִיא – יְהִי / קֹדֶשׁ וְאַל בּוֹ תִגְעוּ״ בֻּשַּׂרְתִּי.

מַה מָּצְאוּ בִי חֵטְא וְעָווֹן שוֹטְנַי? / הַאֶעֱבֹר תּוֹרוֹת, בְּרִית הֵפַרְתִּי?

יֵחָרְבוּ צָרַי וְלֹא יִכָּתְבוּ / בִכְתָב בְּנֵי עַמִּי! לְיוֹם אָמַרְתִּי.

צַר לִי עֲמָלֵקִי לְדָלְקִי יֶאֱרֹב – / אֶקֹּב כְּשִׁמְעִי, אַחֲרָיו עִפַּרְתִּי.

לוּ קָם צְבָא שַׁחַק וְדָלַק אַחֲרַי – / הַאֶעֱרֹץ כַּיּוֹם וְהִשְׂתָּעַרְתִּי?

הִתְפַּזְּרוּ עִתּוֹת אֲנָחָה, עָבְרוּ, / וּשְׁנוֹת הֲנָחָה כַּעֲדִי עִטַּרְתִּי,

אֶהְיֶה לְעוֹלָם עוֹד גְּבִיר, אִם מִדְּבִיר / נִכְבָּד כְּאַרְבֶּה אוֹ כְגוֹב נִגְעַרְתִּי!

חֶלֶד! לְאַט לִי לַאֲצִילֵי אַט – וְאַט / לָהֶם שְׁלוֹמֵי אַהֲבָה זָכַרְתִּי.

הֵיכַל יְיָ, הָעֲבוֹדָה, לִפְנַי, / אִם אֶבְגְּדָה בָהֶם וְאִם שָׁקַּרְתִּי!

רוּחַ זְמִירַי עַל פְּנֵיהֶם יַחֲלֹף, / שַׂעְרַת בְּשָׂרָם מֵרְתֵת שָׁמַרְתִּי.

הֵם לִקְרָאתִי חָרְדוּ לֹא עָמְדוּ / רָאוּ וְהִבִּיטוּ אֲשֶׁר בִּקַּרְתִּי,

כַּמָּה יְבִיאוּן מַעֲדַנֵּיהֶם לְפִי / וָאֵשְׁתְּ מִיֵּינָם וְהִשְׁתַּכַּרְתִּי,

וָאֶשְׁכְּנָה תוֹכָם כְּמֶלֶךְ בַּגְּדוּד / נִשְׁעָן עֲלֵי יָדָם וְהִתְפָּאַרְתִּי.

הֵמָה אֲהֵבוּנִי נְדָבָה – אֹהֲבֵם / נֶצַח, וּבָחְרוּ בַּאֲשֶׁר בָּחַרְתִּי.
