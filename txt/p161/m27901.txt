


	
	[אש הגדוד נפלה ונפלאת]
	



א






אֵשׁ הַנְּדוֹד נָפְלָה וְנִפְלֵאת הַמִּכְוָה בַּאֲסִירֵי הַתִּקְוָה





       לַבָּה אֲשֶׁר שֻׁלְּחָה בְלִבָּהּ לֹא שָׁקְעָה

       שָׂבְעָה נְדֻדִים וְעוֹד בְּדוֹדִים לֹא שָׂבְעָה

       תִּיבַשׁ בְּאֵשׁ אַהֲבָה וְתִרְוֶה מֵי דִמְעָה





דִּמְעָה וְאֵשׁ אוֹכְלָה וְהַנֶּפֶשׁ דָּוָה בֵּין צְמֵאָה בֵּין רָוָה





       לָמֵד לְבָבִי סְבֹל פְּרִידָה מִיָּמָיו

       נִשְׁבַּע לְבַל יֵשְׁתְּ מִזְּמָן כּוֹס תַּנְחוּמָיו

       יוֹם יוֹם נְבִיאָיו יְבַשְּׂרוּנִי בִשְׁלוֹמָיו





וַאֲנִי כְּבֶן יִמְלָה אֲחַו בָּאָה הֹוָה אֹמְרָה כִּי אֵין שַׁלְוָה





       רוּחַ בְּפִי נָשְׁבָה וְנָטְפָה מֹר עֹבֵר

       לִביִּ לדוֹדִי שְׂאִי וְשִׂימִי לוֹ אֵבֶר

       יַחְשׁוֹב בְּיוֹם יֶחֱזֵם וּבָהֶם יִתְחַבֵּר





כִּי עַד זְבוּל נַעֲלָה וְלִבְנֵי עָשׁ נִלְוָה רַק וְלֹא לִבְנֵי חַוָּה





       עָנָן רְבַדּוֹ כְרוּב מְעוֹנוֹ כַשַּׁחַק

       רָחוֹק מְצָאתִיו בְּתוֹךְ קְרָבַי יוֹם רָחַק

       גַּם אוֹת לְעוֹלָם תְּהִי בְּרִיתִי אֶת יִצְחָק





אֶחְזֶה בְּרִית סֶלָה וְאִם תֻּפַר מִצְוָה אֵין לְהָפֵר הָאַחְוָה





       הָאֵל נְתָנוֹ לְאוֹר בְּיָם וּבְחָרָבָה

       וַיֶּאֱסֹר לַהֲלֹךְ לְחוֹף יָם מֶרְכָּבָה

       וּבְלֵיל נְדוֹד שָׁר לְאוֹהֲבָיו שִׁירַת אַהְבָה





זודני קבלה פאני מן גדוה כארז אלי אלעדוה.




ב






הִנֵּה אֲנִי מַשָּׂא אֱלֵי רוּחִי

אֵיךְ יוּכְלוּ זָרִים שְׂאֵת טָרְחִי

נָסְעוּ חֲמוּדוֹתַי בְּעֵת רָאוּ

כִּי כָהֲתָה עֵינִי וְנָס לֵחִי

מִּמַּחֲשַׁכַּי בָּרְחוּ דוֹדַי

אַחַר הֲלֹךְ גּוֹיִם לְאוֹר זַרְחִי

הֵם הֵם אֲהָבַי לַאֲהוּבַי מִי

הִבְאִישׁ בְּעֵינֵי אֹהֲבַי רֵיחִי

כִּי בַגְדוּ רֵעַי וְגַם אַחַי

הֵם הֶאֱבִידוּ בַּנְּדֹד נִצְחִי

אִם אֶחְיֶה אַחֲרֵי פְרִידָתָם

לֹא עַל רְצוֹנִי כִּי בְּעַל כָּרְחִי

אֵשֵׁב מְפֹרָד עִם בְּנֵי יָמִים

הַשֹּׂחֲקִים לִי עַל מְרִי שִׂיחִי

אֶבְכֶּה וְכוֹכָבִים אֲבַכֶּה עֵת

אֶצְעַק אֲהָהּ אָחִי אֲהָהּ אָחִי

משֶׁה כְּחוֹתָם הוּא עֲלֵי לִבִּי

יוֹסֵף כְּטוֹטֶפֶת עֲלֵי מִצְחִי

אֵשֵׁב וְיֵשֵׁב צַלְמְךָ עִמִּי

אֵלֵךְ וְיֵלֵךְ תָּאָרְךָ נִכְחִי

רֵעַי וְשַׁעְשֻׁעַי סְפָרֶיךָ

הֵמָּה הֲגִיג לִבִּי וְהֵם לִקְחִי

אַל תַּחְשְׂכֵם פֶּן תַּחֲשׂךְ חַיָּי

אִם תַּעְצְרֵם לֹא אֶעֱצֹר כֹּחִי

אֹמַר לְרוּחַ עָבְרָה עָלָי

שָׁלוֹם שְׂאִי אֶל מַחְמַד רוּחִי

וּקְחִי תְּשׁוּרַת מַעֲנוֹ וּפְחִי

בַּהֲרוּג פְּרִידָתוֹ עֲדֵי יֶחִי.




ג






אֲחִי מוּסָר גְּבִיר וָשָׂר

חֶמְדַּת עַרָב וָעֵבֶר

אֲהָהּ נַדְתָּ וְנִפְרַדְתָּ

וְרָעֲךָ שֹׁד וָשָׁבֶר

אֲהָהּ כַּגֵּר בְּתוֹךְ מַסְגֵּר

נִשְׁאַרְתָּ בְּלִי חָבֵר

וְאָח וְאָחוֹת וּמִשְׁפָּחוֹת

מְפֻזָּרִים בְּכָל עֵבֶר

עֲדֵי אַרְצָךְ בְּבוֹא קִצָּךְ

תִּקְרָא בְּפִי כָל גָּבֶר

מַה לְּך פֹה וּמִי לְךָ פֹה

כִּי חָצַבְתָּ לְּךָ פֹּה קָבֶר.

מַה אֹמַר וְרַע וָמָר

גְּזֵרָה מִפִּי שַׁדָּי

גְּלוֹת כְּבוֹדִי וְסוּר הוֹדִי

וְלָקַחַת מַחֲמַדָּי

זֶה חֶלְקִי וְסוֹף חֻקִּי

וְגוֹרָלִי מְנָת מִדָּי

לֻקַּח צִיץ אֲשֶׁר הֵצִיץ

כִּרְבִיד פָּז עַל מַדָּי

רֹאשׁ חֲכָמַי וּשְׁלֵמַי

וּפְאֵר כָּל־נִכְבַּדָּי

בָּא שִׁמְשִׁי וְסָף שָׁרְשִׁי

וְאֵיךְ לֹא יִיבְשׁוּ בַדָּי

בְּמָה אֶגְאֶה וְלֹא אֶרְאֶה

כִּי אִם צְנִינִים לְצִדָּי

וְאֶצְעַק מָר בְּתוֹךְ מִכְמָר

מִי יִתֶּן לִי אֵבֶר

מַה לְּךָ פֹה וּמִי לְךָ פֹה

כִּי חָצַבְתָּ לְּךָ פֹּה קָבֶר.

בְּמָה אָחֵל וְאֶתְחַלְחֵל

בְּתֻמּוֹ אִם בְּחָכְמָתוֹ

וְאִם בְּהוֹדוֹ וּבִכְבוֹדוֹ

וּגְדֻלַּת מַעֲלָתוֹ

וְאִם בְּצָרְיוֹ וְטוּב פִּרְיוֹ

וְשַׁעֲשֻׁעֵי חִידָתוֹ

וְאִם עַל יוֹם בְּלִי פִדְיוֹם

לֻקְּחָה בוֹ נִשְׁמָתוֹ

וְאֵין מֹדָע אֲשֶׁר יֵדַע

יְגוֹן לִבּוֹ וְצָרָתוֹ

וְאֵין אָחוֹת תְּהִי לִמְחוֹת

מִלֶּחְיוֹ דִּמְעָתוֹ

בְּעֵת בִּכְיוֹ בְּבֵית עָנְיוֹ

עַל אֶחָיו וְחֶבְרָתוֹ

אֲהָהּ אַחַי אֲחִיכֶם חַי

וְלֹא תִבְכּוּ פְּרִידָתוֹ

וְהִנֵּה מֵת וְיִצָּמֵת

וְלֹא תִשְׂאוּ מִטָּתוֹ

וּלְנָכְרִים קוֹל יָרִים

תְּנוּ לִי אֲחֻוזַּת קָבֶר

מַה לְּךָ פֹה וּמִי לְּךָ פֹּה

כִּי חָצַבְתָּ לְּךָ פֹּה קָבֵר.

הָאַחִים מְשֻלָּחִים

נֹדְדִים בְּאַרְבַּע רוּחוֹת

וְהָאֲחָיוֹת הֹמִיוֹת

נֶאֱנָקוֹת נֶאֱנָחוֹת

בְּכָל מְדִינָה נְהִי קִיָנה

מִשְׁפָּחוֹת מִשְׁפָּחוֹת

כָּל יֹדְעָיו וְכָל שֹׁמְעָיו

הִשְמִיעוּ קוֹל צְוָחוֹת

גְּבִיר כָּלָה וְהוֹד גָּלָה

וְנָגֹלּוּ הַשּׂמָחוֹת

וְאֵיד פָּרֹח וְגִיל בָּרַח

וְיִנָּחוּ הָאֲנָחוֹת

לְמוֹת הַשָּׂר הַטִּפְסָר

עֹטֶה מַעְיָן שְלֻחוֹת

אֲבִי עֵצָה וּמְלִיצָה

וְלָשׁוֹן תְּדַבֵּר צַחוֹת

אֵין מִשְׁעָן בַּל יִצְעָן

נִגְדַּע וַיִּשָּׁבֵר

מַה לְּךָ פֹה וּמִי לְךָ פֹה

כִּי חָצַבְתָּ לְּךָ פֹּה קָבֶר.

לַנֶּגַע בָּא פֶּגַע

וְהֵפֵר הַמַּחֲשָׁבוֹת

יְקָר נִשְׁבָּה וְנֵר כָּבָה

וְחָצַב מִלֵּב לְהָבוֹת

וְעֵינַיִם כַּמַּיִם

כַּמַּעְיָנוֹת זָבוֹת

בִּנְפֹל בְּרוֹשׁ וְנִכְרַת רֹאשׁ

מֵרָאשֵׁי הָאָבוֹת

נְדִיב הָעָם וְשַׁעֲשֻׁעָם

יֹעֵץ וְקָם עַל נְדִיבוֹת

בְּכוֹר מָוֶת וְצַלְמָוֶת

קִדְּמוֹ כִּזְאֵב עֲרָבוֹת

וּגְזֵרָה מִנּוֹרָא

קָרְבָה מֵאֵין קְרָבוֹת

וַתִּקַּח הַלֶּקַח

וְהַנְּפָשׁוֹת וְהַלְּבָבוֹת

וַתַּכְרִית הָאַחֲרִית

וְהַתִּקְוָה וְהַשָּׁבֶר

מַה לְּךָ פֹה וּמִי לְךָ פֹה

כִּי חָצַבְתָּ לְּךָ פֹּה קָבֶר.
