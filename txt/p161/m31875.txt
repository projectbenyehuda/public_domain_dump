


	
	זֹאת הַתְּלָאָה
	



קינה על שלמה אבן פֶרוּצִיאֵל






זֹאת הַתְּלָאָה, לְבַל אֶשְׁכַּח מְצָרֶיהָ – / קִירוֹת לְבָבִי אֲשַׁוֶּה אֶת סְפָרֶיהָ.

עָבְרָה כְּמוֹקֵד בְּתוֹךְ בִּתְרֵי לְבָבוֹת, מְצָ/אָתַם כְּאַשְׁפָּה לְחֵץ קֶטֶב מְרִירֶיהָ.

קָדְחָה, וְלוּלֵי דְמָעוֹת הַמְכַבּוֹת קְצֵה / אִשָּׁה – כְּקָט נִשְׂרְפָה עַצְמִי בְּאוּרֶיהָ.



יֹאמְרוּ: ״הֲתִירָא מְגוֹר תֵּבֵל?״ עֲנִיתִים: ״וְאֵיךְ / אִירָא – וְיַחְדָּו כְּבָר בָּאוּ מְגוּרֶיהָ?״



עוֹד לֹא אֲפַחֵד לְכָל יָגוֹן וְלֹא אֶשְׂמְחָה / אֶל גִּיל, וְלוּ יִרְקְדוּ נֶגְדִּי הֲרָרֶיהָ.

רַב לַזְּמָן מִבְּגֹד בִּרְבִיד גְּרוֹנוֹ וְדַ/יָּהּ הָאֲדָמָה לְהַשְׁלִיךְ אֶת כְּתָרֶיהָ.

עִמִּי אֲנָחָה תְּפוֹצֵץ הַסְּלָעִים בְּעֵת / יַעַל עֲשָׁנָהּ וְתֹפֶת מִנְּחִירֶיהָ,

לָאִישׁ אֲשֶׁר קִנְּאוּ צִבְאוֹת שְׁחָקִים לְאוֹר / פָּנָיו, וְתֵבֵל – יְרֵאוּהוּ כְפִירֶיהָ.

יַחַד בְּנֵי אִישׁ וְהָאָרֶץ דְּבָרִים, אֲבָל / הָיָה שְׁלֹמֹה גְּבִיר עִנְיַן דְּבָרֶיהָ.

אוֹ הָאֲדָמָה בְתוּלָה תַּעֲבִיד לוֹ סְגָ/נֶיהָ – וְהוּא בִלְבָד בַּעַל נְעוּרֶיהָ.

רַבּוּ אֱמוּנוֹת אֲשֶׁר נָתַן בְּתוֹכָהּ – וְהִיא / לִזְנוֹת בְּבִלְתּוֹ מְאֹד רַבּוּ קְשָׁרֶיהָ.

זָנְתָה, וְלוּ חָכְמָה – אָז נִכְבְּשָׁה לוֹ כְּמוֹ / אָמָה וְנָתְנָה בְכָפְרוֹ כָּל יְקָרֶיהָ.

וּתְמוֹל בְּעָמְדוֹ עֲלֵי אֶרֶץ – בְּלִי אֶדְרְשָׁה / שֶׁמֶשׁ; וְהַיּוֹם, אֲהָהּ, אֶדְרֹשׁ מְאוֹרֶיהָ.

הִנֵּה וְאִם אֶחֱזֶה צַלְמוֹ בְּמַרְאֵה חֲלוֹם – / עוֹד אֶבְגְּדָה בָהּ וְלֹא אָעִיר שְׁחָרֶיהָ.

יִגְמֹל אֱלֹהִים לְתֵבֵל כַּאֲשֶׁר עָשְׂתָה, / טָמְנָה כְרוּבָהּ בְּעַד תַּחְתִּית עֲפָרֶיהָ.

לוּ נָדְבוּ הַדְּמָעוֹת כַּאֲשֶׁר נָדְבָה / יָדוֹ לְפָנִים בְּדֹלַח מִמְּטָרֶיהָ –

לֹא נֶעְצְרוּ, עַד אֲשֶׁר רִוּוּ אֲדָמָה וְהִצְ/מִיחוּ עֲרָבִים בְּתוֹךְ צִיּוֹת חֲרֵרֶיהָ.

אַל טַל, עֲלֵיכֶם, שְׂדֵי שֵׂעִיר, לְעוֹלָם וְרַב / לָכֶם בְּדִמְעָה כְּמֵי דִימוֹן שְׂעִירֶיהָ.

שַׁחַת הֱיִיתֶם לְמִשְׂרַת בֵּית שְׁלֹמֹה וְהִשְׁ/לַכְתֶּם צְפִירָה וְחִלַּלְתֶּם נְזָרֶיהָ.

רֶכֶב גְּאוֹן יַעֲקֹב פִּתְאֹם הֲפַכְתֶּם לְמִרְ/כֶּבֶת אֲבַדּוֹן וְצַלְמָוֶת סְדָרֶיהָ.

קִינִים וְהֶגֶה וְהִי הוֹ הוֹ וְאוֹי וַאֲבוֹי / מִבַּת יְהוּדָה מְמָרֶרֶת אֲמָרֶיהָ,

וּפְשֹׁט עֲדָיִים וְלָבוֹשׁ כַּלְּבָבוֹת אֱשׁוּן / חֹשֶׁךְ, וְשׂוּם מִבְּלוֹאֵי שַׂק חֲגוֹרֶיהָ.

הָהּ לִי, גְּבִירִי! לְמִי זֶה אַחֲרֶיךָ תְּהִי / מִשְׂרַת הֲדָרְךָ וּמִצְנֶפֶת פְּאֵרֶיהָ?

לָמָּה אֱהִי מִבְּנֵי חַיִּים – וְאַתָּה בְּעֵ/פָתָה מְסֻגָּר בְּאַשְׁמַנֵּי קְבָרֶיהָ?

אֶל מִי יְנוּסוּן מְעַט הַצֹּאן וְגָלוּת סְפָ/רַד? בִּלְתְּךָ מִי יְהִי רוֹעֶה עֲדָרֶיהָ?

וּלְמִי לְנַהֵל טְלָאֶיהָ בְּחֹצֶן? וּמִי / מוֹטוֹת וּמוֹסֵר יְפַתַּח מֵאֲסִירֶיהָ?

רוֹשׁ הֶאֱכַלְתָּ עֲזוּבֶיךָ וְאַחַר שְׁתוֹ/תָךְ כּוֹס שְׁאוֹל הָפְקְדוּ לִמְצוֹת שְׁמָרֶיהָ.

שִׁבְרוּ כְלֵי שִׁיר וְהַאְזִינוּ לְיוֹנַת רְחוֹ/קִים, עֵת תְּקוֹנֵן עֲלֵי פֵרוּד חֲבֵרֶיהָ,

חַנֵּן תְּחַנֵּן בְּמַר קוֹלָהּ – וְלִבִּי מְאֹד / יֵרַךְ בְּדַעְתּוֹ אֲשֶׁר צִירַי כְּצִירֶיהָ.

רָנִּי, יְפַת הָעֲנָק, אֶל הַמְּרֻוִּים בְּתַרְ/עֵלָה וְאוּלַי תְּעִירִין אֶת שְׁכוּרֶיהָ,

וִהְיִי לְעֵדָה עֲלֵי דִמְעָה אֲשֶׁר תַּעֲרֹף / עִם דַּם כְּבֵדִי עֲדֵי אָדְמוּ נְהָרֶיהָ.

הָהּ לַמְקַנֵּן תְּמוֹל עַל עָשׁ – וְהַיּוֹם בְּעַד / דֶּלֶת עֲפָרוֹת בְּלִי תִסֹּב בְּצִירֶיהָ.

אֶכְרֹת לְעֵינַי בְּרִית בַּל אֶחֱזֶה עוֹד יְקָר / אַחַר יְקָרוֹ וְלָעַד לֹא אֲפִירֶהָ.

לֹא אֶזְכְּרָה עוֹד כְּאֵב אַחַר כְּאֵבוֹ, וְזֹאת / הַמַּחֲלָה לֹא צֳרִי גִלְעָד לְצִירֶיהָ.

יִשֹּׁם לְבָבִי בְּאַנְחוֹתַי וְנַפְשִׁי תְמָ/רֵר בַּבְּכִי, כִּי מְרוֹרוֹת רוֹשׁ מְרוֹרֶיהָ,

לֹא יִתְּנֵנִי יְקוֹד רוּחִי הֲשִׁיבָהּ – וּמָה / אָשִׁיב לְרוּחַ אֲשֶׁר תִּשְׂרֹף פְּגָרֶיהָ?

אִם הַתְּלָאָה מְאֹד קָשָׁה כְּשָׁמִיר, / הֲלֹא מָצְאָה לְבָבוֹת חֲזָקִים מִשְּׁמִירֶיהָ.

מִי לִי בְּאֵדֶר יְקָר אַחַר נְסֹעַ גְּוִ/יַּת הַגְּבִיר אֶל שְׁאוֹל עִם כָּל הֲדָרֶיהָ?

תִּרְאֶה מְהוּמָה בְּיוֹם נָסְעוֹ אֱלֵי מַחֲנֶה / בִּמְעֵי אֲדָמָה, וְהוּא בָא בַּחֲדָרֶיהָ.

לָכֵן שְׁעוּ עַד אֲמָרֵר בַּבְּכִי, פֶּן אֲבַ/קַּע אֶת צְחִיחָהּ וְאָבוֹא תּוֹךְ סְגוֹרֶיהָ.

אֶמֶשׁ רְקִיעִי עֲלֵי רֹאשִׁי רְאִיתִיו – אֲהָהּ, / אַרְצִי, כְּבָר סָתְמָה עָלָיו שְׁעָרֶיהָ.

יֹאמְרוּ: ״כְּבָר נֶעְלְמָה מִמְּךָ תְּמוּנַת גְּבִיר!״ – / לֹא יֵדְעוּ כִּי בְעֵין הַלֵּב אֲשׁוּרֶהָ.

דַּי לָאֲדָמָה נְתֹן בִּשְׂמֹאל נְעִימוֹת וּבִי/מִינָהּ תְּכוֹנֵן לְחֵץ מָוֶת יְתָרֶיהָ.

עֶשֶׂר פְּעָמִים תְּסוֹבֵב עַל אֱנוֹשׁ לַעֲרֹךְ, / תִּקְצֹר וְתִלְקֹט וְתֵפֶן לַאֲחוֹרֶיהָ.

הֶבֶל בְּעֵינָהּ מְלוּכַת הַמְּלָכִים וְלֹא / תִשָּׂא פְּנֵי שַׂר וְלֹא תִשְׁעֶה עֲשִׁירֶיהָ.

נוֹצְרָה כְנֹפֶת בְּרֵאשִׁיתָהּ – וְסוֹפָהּ כְּרוֹשׁ, / מִי יִתְּנֵנִי וְלֹא אֶטְעַם יְעָרֶיהָ!

פַּחַד וּפַחַת וּפַח תִּקְרָא לְיָבֵשׁ וְלַח, / עַד תֶּאֱסֹף עִם עֲטוּפֶיהָ קְשׁוּרֶיהָ.

שָׁפְכָה דְּמֵי בַעֲלָהּ בִּידֵי רְשָׁעִים לְגַ/לּוֹת מוּם זְנוּתָהּ, לְהַבִּיט אֶל מְעוֹרֶיהָ.

יוֹם נִשְׁבְּעָה לֶאֱסֹף שֶׁמֶשׁ בְּחֵיקָהּ בְּעֵת / נִקְבַּר שְׁלֹמֹה, הֲלֹא קָמוּ נְדָרֶיהָ.

בָּכוּ גְבִירִים לְאוֹסֵף אֶת פְּאֵרָהּ, וְכֵן / בָּכוּ עֲנִיִּים לְזוֹרֶה אֶת סְגוֹרֶיהָ.

טוּב הַשְּׁמָנִים, אֲהָהּ, נִשְׁפַּךְ וְנִלְקַח אֲרוֹן / תּוֹרָה וְצִיּוֹן הֱצִיקוּהָ מְצֵרֶיהָ.

רוּם אָהֳלֵי אָהֳלָה שָׁפֵל כְּהַיּוֹם וְאִי / כָבוֹד וְחָדַל דְּרוֹר מֹר מִצְּרוֹרֶיהָ.

מֵאָהֳלִיבָה יְקָר נָסַע וּבָשְׂמָהּ לְמַק / נֶהְפַּךְ, וְנַעְלָה כְרוּב מִתּוֹךְ דְּבִירֶיהָ.

סֹבּוּ יְדִידִים לְמִרְכֶּבֶת, תְנַהֵג אֲדוֹ/נֶיהָ בְּעֹצֶר וּבִכְבֵדוּת אֲשׁוּרֶיהָ!

פֶּגֶר שְׁלֹמֹה סְבָבוּהוּ שְׁלוֹמוֹת בְּצִדְ/קָתוֹ אֲשֶׁר שִׁלְּחָה עַד יָם קְצִירֶיהָ.

וּרְאוּ תְּמוּנָה נְשׂוּאַת חֵן, רְדוּמָה – וְלִ/בָּהּ עֵר, יְשֵׁנָה – וְטוּב זִכְרָהּ יְעִירֶיהָ.

תִּישַׁן וְתִבְלֶה בֶּעָפָר, אַךְ בְּעוֹדָהּ שְׁמָהּ / חָדָשׁ בְּכָל־יוֹם וְהוּא הֶגְיוֹן גְּבִירֶיהָ.

וּבְעֵת עֲטֹף בֵּן בְּחֵיק אִמּוֹ לְלֶחֶם, תְּהַזְ/כִּיר שֵׁם שְׁלֹמֹה וְתִסְעַד לֵב צְעִירֶיהָ.

כָּבְדָה מְאֹד מַחֲלַת אֶבְלוֹ, עֲדֵי בִתְּרָה / יִתְרַת כְּבֵדִים וְעָבְרָה בֵּין בְּתָרֶיהָ.

יִשְׁפֹּךְ אֱלֹהִים מְטַר זַעַם עֲלֵי בַת אֱדוֹם, / שָׁרְשָׁהּ יְרַצֵּץ וְקִצֵּץ אֶת אֲמִירֶיהָ!

יָשִׁיב לְחֵיקָהּ גְּמוּל בִּשְׁכוֹל וְאַלְמוֹן וְיַשְׁ/כִּיב כָּל הֲמוֹנָהּ כְּכָל צַלְמֵי אֲשֵׁרֶיהָ!

מַגָּל יְשַׁלַּח וְיִקְצֹר אֶת קְצִירָהּ וְגַת / יִדְרֹךְ לְעַמָּהּ עֲדֵי תֹם כָּל בְּצִירֶיהָ

וּדְבַר נְבִיאוֹ יְמַהֵר לָשׂוּם לְזֶפֶת נְחָ/לֶיהָ וְאַרְצָהּ, כְּמוֹ גָפְרִית – זְמִירֶיהָ!

תַּנְחוּם יְדַבֵּר לְנֶפֶשׁ הָעֲגוּמִים לְאַ/מֵּץ אֶת לְבָבָהּ וְלִבְנוֹת אֶת גְּדֵרֶיהָ,

וּפְנֵי חֲסָדָיו יְגַלֶּה עַל שְׁלֹמֹה וְעַל / נֶפֶשׁ אֲבֵלָיו לְרַפּוֹת אֶת שְׁבָרֶיהָ.

נָוֶה יְשַׁו לוֹ בְּרֹאשׁ יִשְׁרֵי אֲדָמָה וּבַ/יִת נֶאֱמָן יַעֲשֶׂה לּוֹ עִם נְבָרֶיהָ.

עַל כָּל פְּשָׁעִים יְכַפֶּר לוֹ וְיִמְחֶה גְּלוּי / חַטַּאת שְׁלֹמֹה יְיַעְבֹר עַל סְתָרֶיהָ.

שֶׁמֶשׁ צְדָקָה תְהִי זוֹרְחָה בְּפָנָיו לְעוֹ/לָמִים וְיִשְׁכֹּן עֲדֵי עַד בַּחֲצֵרֶיהָ,

וּשְׂכַר חֲסִידִים יְהָרִיעַ בְּאָזְנָיו וְיִשְׁ/מַע מֵאֱמוּנָה נְעִים שִׁירוֹת זְמִירֶיהָ.

הָסֵר, אֱלֹהִים, כְּאֵב מִיַּעֲקֹב וּשְׁלוֹם / נֶפֶשׁ שְׁלֹמֹה שְׁלַח מִכָּל עֲבָרֶיהָ.



יוֹם נֶהֱרַג הַגְּבִיר – עֶשְׂרִים בְּאִיָּר שְׁנַת / תַּתְסַ״ח – סְפוּרִים עֲלֵי סְפוּרֶיהָ.



לֹא נִשְׁלְמוּ לוֹ לְאַרְבָּעִים שְׁנוֹתָיו וּבָא / יוֹם קֵץ לְנַפְשׁוֹ וְשָׁבָה אֶל מְגוּרֶיהָ.

לִבִּי וְעֵינַי נְתַתִּימוֹ פְקִידִים עֲלֵי / זֹאת הַתְּלָאָה לְבַל אֶשְׁכַּח מְצָרֶיהָ.
