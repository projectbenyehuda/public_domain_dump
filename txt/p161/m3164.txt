


	
	אגגי הנותן רוזנים משחק
	



אֲגָגִי הַנּוֹתֵן רוֹזְנִים מִשְׂחָק / ר' יהודה הלוי

  

ו


אֲגָגִי הַנּוֹתֵן רוֹזְנִים מִשְׂחָק



בְּחָשְׁבוֹ מַחְשְבוֹת אָוֶן עַל בֵּית יִשְׂחָק



יוֹשֵׁב בַּשּׁמַיִם יִשְׂחָק


 


גָּזָר אֹמֶר אֲשֶׁר דְּבָרוֹ לֹא יָשׁוּב רֵיקָם



דַּרְכָּּם פָּקַד וַיָּשֶׁב גְּמוּלָם אֶל חֵיקָם



יִשְׂמַח צַדִּיק כִּי חָזָה נָקָם


  


הֶחָסֹן לִנְעֹרֶת כְּרֶגַע וַיָּרוֹץ



וַאֲשֶׁר נוֹאַשְׁתִּי חִזְּקַנִי לְבַל אֶעֱרֹץ



כִּי לֹא לַקַּלִּים הַּמרוֹץ





יָדַעְתִּי הַיוֹם וַיֵּדְעוּ דַּלֵּי הֲמוֹנָי



הַלְּקֻחִים לַמָּוֶת וּמְיַחֲלִים יְשׁוּעַת אֲדֹנָי



כִּי לֹא בְּחֶרֶב וּבַחֲנִית יְהוֹשִׁיעַ אֲדֹנָי


 


זֶה יָדַעְתִּי כִּי אֱלֹהִים הֶעֳלַנִי מִמָוֶת



חֲסָדָיו קִדְּמוּנִי בְּאֶרֶץ צַלְמָוֶת



כִּי כְפֶשַע בֵּינִי וּבֵין הַמָּוֶת


  


טֹרְפִים רַבּוּ עַל שֶׂה פְּזוּרָה דְחוּיָה



יָעַץ הָמָן וְיַד הַמֶּלֶךְ עִמּוֹ הָיָה



וַאדֹנָי שָׁם הָיָה


  


כָּתַב גְּזֵרָתָם וַיֵּשְׁבוּ לֶאֶכֹל וְשָׁתוֹ



לְהַשְׁמִיד עַם קֹדֶש וּלְהַשְׁחִית שְׁאֵרִיתוֹ



וַיִּתְנַכְּלוּ אֹתוֹ לַהֲמִיתוֹ


  


עֻצוּ עֵצָה וְתֻפָר כִּי עִמָּנוּ אֵל אָיוֹם



אַל תִּתְהַלֵּל בְּיוֹם מָחָר מִבְּעוֹד יוֹם



כִּי לֹא תֵדַע מַה יֵלֶד יוֹם


 


מִתְעָּנה בְּצוֹם וְחֹגֶרֶת שַׂק מִפְּנֵי הַיִּרְאָה



נִחוּמָה כִּי הִנֵּה אֲדֹנָי מֵעִיר רוּחַ קִנְאָה



מִנְעִי רַגְלֵךְ מִיָּחֵף וּגְרוֹנֵךְ מִצִּמְאָה


 


סְגֻּלָתוֹ וְעַמּוֹ בְּטַח בַּאדֹנָי מְנָת חֶלְקֶךָ



עֶזְרְךָ יִשְׁלַח מִקֹּדֶש וּבְיוֹם דָּחֳקֶךָ



חָנוֹן יָחְנְךָ לְקוֹל זַעֲקֶךָ


 


פָּנָה לִתְפִלַּת עֲבָדָיו וְזַעֲקָתָם יָחִילוּ



צַדִּיקִים יָגִילוּ וַאֲשֶר פּוּר הִפִּילוּ



זַעֲקַת שֶׁבֶר יְיֵלִילוּ


 


יְיֵלִילוּ אַכְזָרִים וְיָגִילוּ יְשָׁרִים



וְיֵדְעוּ רְחוֹקִים מְפֻזָּרִים



כִּי אֱלֹהִים שֹׁפֵט זֶה יַשְׁפִּיל וְזֶה יָרִים.


  


קִּצַץ כְּנַף הַנֶּשֶׁר אֶרֶךְ הָאֵבֶר



רָמָתוֹ הִשְׁפִּיל וְקָרָא לוֹ שֶּבֶר עַל שֶׁבֶר



קְבוּרַת חֲמוֹר יִקָּבֵר


  


שָׁרְשוֹ הִכְרִית וִילָדָיו שָׁמַט שָׁמוֹט



תָּלָהוּ נֶגֶד הַשֶׁמֶשׁ עַל הַמּוֹט



וַיְחַזְּקֵהוּ בְּמַסְמְרִים לֹא יִמוֹט


  


חַסְדְּךָ לְעוֹלָם לֹא חָשַׂכְתָּ מֵאֹהֲבֶיךָ



זֶרַע עֲמָלֵק אִבַּדְתָּ וְהִרְחַקְתָּ מִסְּבִיבֶיךָ



כֵּן יֹאבְדוּ כָל אוֹיְבֶיךָ.
