


	
	[אתה כוננת מישרים ועיתדת יום פקודתך]
	



מגן


אַתָּה כּוֹנַנְתָּ מֵישָׁרִים וְעִתַּדְתָּ יוֹם פְּקֻדָּתֶךָ



בְּטֶרֶם תַּרְקִיעַ שְׁחָקִים וַתָּכֶן בָּהֶם אֱמוּנָתֶךָ



גָּזַרְתָּ עַל בְּנֵי חֲלוֹף לַהֲבִיאָם בְּפֶלֶס צִדְקָתֶךָ



שֵׁבֶט מִישֹׁר שֵׁבֶט מַלְכוּתֶךָ

דֶּרֶךְ חַיִּים הוֹדַעְתַּנִי וַחֲנַנְתַּנִי שֵׂכֶל כֹּבֵשׁ



הִזְהַרְתַּנִי לְיוֹם פְּקֻדָּה בְּפָקְדְּךָ עַל לַח וְיָבֵשׁ



וְנִגְעֵי שִׁגֲגוֹתַי וּזְדוֹנַי מֵאָז הָיִיתָ חֹבֵשׁ



אֲדֹנָי מָלָךְ גֵּאוּת לָבֵשׁ

זֶה הַיּוֹם זָעַמְתָּ זֵדִים שְׁמָם מָחִיתָ לְעוֹלָם



חָלַקְתָּ לַצַּדִּיקִים בְּרָכָה חַיִּים עַד הָעוֹלָם



טִכַּסְתָּ אֵלֶּה לְחַיֵּי עוֹלָם וְאֵלֶּה לַחֲרָפוֹת לְדִרְאוֹן עוֹלָם



וַיֵשֶׁב אֲדֹנָי מֶלֶךְ לְעֹולָם

יִדַּעְתָּ לְבֵינוֹנִים תְּשׁוּבָה לְהַצִּיל מִמָּוֶת נַפְשָׁם



כִּי הֶאֱרַכְתָּם עַד עָשׂוֹר אוּלַי יֵצְאוּ מִמּוֹקְשָׁם



לְהִתְהַלֵּךְ בְּאֶרֶץ הַחַיִּים וְלַעֲבֹר בַּסָּךְ עַד מִקְדָּשָׁם



מַלְכָּם לִפְנֵיהֶם וַאדֹנָי בְּרֹאשָׁם.



כרוג למגן


יֶחֱזוּ פָנֵימוֹ הַיּוֹם קְהִלַּתְכֶם



וְיָרִיע לְסַקֵּל דֶּרֶךְ לִקְרַאתְכֶם



הוֹד יֵשַׁע יְגַלֶּה כְּמֵאָז לִפְנֵיכֶם



אֲדֹנָי צְבָאוֹת יָגֵן עֲלֵיכֶם.



מחיה


מִגְדַּל עֹז שֵׁם אֲדֹנָי לְהִנָּצֵל מִמְּדוּרָה עֲרוּכָה



נִשְׁמַת אַפּוֹ כְּנַחַל גָּפְרִית מַסֵּכָה עַל שֹׂנְאָיו נְסוּכָה



סֵפֶר זִכְרֹנוֹת פֹּתֵחַ וְכֹתֵב כָּל אֲשֶׁר זָכָה



זֵכֶר צַדִּיק לִבְרָכָה

עֲבָדָיו כֻּלָּם הַיּוֹם בְּשִׁבְתּוֹ לִשְׁפֹּט אֱמוּנוֹת



פֵּרַשׂ סִפְרוֹ וּפֵרְשׁוֹ וּפָקַד גּוֹיִם וּלְשׁוֹנוֹת



צַדִּיק הַצְדֵּק וְהַרְשִׁיעַ רָשָׁע כָּל אִישׁ בִּדְרָכָיו לְהִמָּנוֹת



וַיֹאמֶר לְהָבִיא אֶת סֵפֶר הַזִּכְרוֹנוֹת

קָבַע כִּסֵּא מִשְׁפָּט וְחָתַם לְחַיִּים נֶאֱמָנָי



רִשְׁעֵי אֶרֶץ הִדִּיחַ מִהְיוֹתָם יֹרְשֵׁי עֲדָנָי



שֹׁמֵר לְבֵינוֹנִים תְּשׁוּבָה אוּלַי בַּעֲשֶׂרֶת יְמֵי תַחֲנוּנָי



יִזְכְּרוּ וְיָשֻׁבוּ אֶל אֲדֹנָי

תָּבוֹא לְפָנָיו הַיּוֹם שַׁוְעַת צִבְאוֹתָיו



תֹּעֵי צֹאנוֹ יְשׁוֹבֵב לְמִרְבַּץ נָוֵהוּ וּמִכְלְאוֹתָיו



תִּזָכַרְנָה רִאשׁוֹנוֹת פְּלָאָיו בְּאַחֲרוֹנוֹת נוֹרְאוֹתָיו



זֵכֶר עָשָׂה לְנִפְלְאוֹתָיו.



כרוג למחיה


יִשְׂרָאֵל לְהָשִׁית חֶלְקָם בַּחַיִּים



הַיּוֹם תִּכְתְּבֵמוֹ עִם הַכְּתוּבִים לְחַיִּים



וְאֶת דִּינָם הוֹצֵא לֵאוֹר בְּאוֹר הַחַיִּים



זֹכֵר יְצוּרָיו בְּרַחֲמִים לְחַיִּים.



סלוק


יִרְאֵי אֲדֹנָי הַלְלוּהוּ כָּל זֶרַע יַעֲקֹב כַּבְּדוּהוּ

הַיּוֹם בְּהַזְכַּרְכֶם יוֹם פְּקֻדַּת יֹצֶרְכֶם



אָיֹם וְנוֹרָא הוּא

וּבַהֲכִינוֹ כִּסֵּא פְלָאָיו וְהַקְדִּישׁ לַמִּשְׁפָּט קְרֻאָיו



לְשַׁלֵּם לְאִישׁ כְּמַעֲשֵׂהוּ

דִּינוֹ בְּשָׁלֹשׁ כִּתּוֹת פְּעֻלָּתָם עַל סֵפֶר חֲרוּתוֹת



וְשׁוֹמֵר תּוֹרָה אַשְׁרֵהוּ

הֲמוֹנִים הֲמוֹנִים לְיוֹם דִּין נְכוֹנִים



כְּאֶחָד לָרוּץ בְּעֵמֶק הֶחָרוּץ



עֵמֶק הַמֶּלֶךְ הוּא

הַצַּדִּיקִים לְחַיִּים וּבֵינוֹנִים תְּלוּיִים



וּרְשָׁעִים לְחֶרְפַּת הֲמוֹנָי כִּי כִחֲשׁוּ בַּאדֹנָי



וַיֹאמְרוּ לוֹא הוּא

לְהַצְדִּיק צַדִּיק כְּצִדְקָתוֹ וּלְהַרְשִׁיעַ רָשָׁע כְּרִשְׁעָתוֹ



וּלְהַאֲרִיךְ לְבֵינוֹנִי כִּי לֹא בָזָה וְלֹא שִׁקַּץ עֱנוּת עָנִי



אוּלַי מִשְׁגֶּה הוּא

וְאוּלַי יִתֵּן אֶל לִבּוֹ לְבַעֵר טְמוּנֵי חֻבּוֹ



כִּי יִרְאֶה בְּמוֹת רָשָׁע וְשָׁב מֵעָוֹן וָפָשַׁע



פֶּן יָמוּת גַּם הוּא

יָחִיד נִקְרָא רַב חֶסֶד מַטֶּה כְּלַפֵּי חָסֶד



וְרוֹצֶה וִדּוּי אֱמוּנַי לְמַעַן שִׁמְךָ אֲדֹנָי



וְסָלַחְתָּ לַעֲוֹנִי כִּי רַב הוּא

בֹּחֵר בִּתְשׁוּבָה וְעַל פֶּשַׁע עֹבֵר פַּעֲמַיִם שָׁלֹשׁ עִם גָּבֶר



אֲבָל אִם הַחֵטְא גָּבַר הָרִאשׁוֹן אֵלָיו יְחֻבָּר



וְנוֹסַף גַּם הוּא

רִאשׁוֹן רִאשׁוֹן מַעֲבִיר לַחוֹטֵא אִם חֲמֻדוֹת הִגְבִּיר



וְיֵחָשֶׁב לוֹ חֶטְאוֹ כְּאִילוּ לֹא עֲשָׂאוֹ



כִּי בַמֶה נֶחְשָׁב הוּא

שֶׁעֲוֹנוֹתֵיהֶם כִּזְכוּיוֹתֵיהֶם שְׁקוּלִים יֹרְדִים וּמְצַפְצְפִים וְעֹלִים



כִּי אֲדֹנָי יִפְקְדֵם מִבֶּטֶן שְׁאוֹל יְעֹודֲדֵם



כִּי חַנּוּן וְרַחוּם הוּא

מוּכָן כִּסְאוֹ לַמִּשְׁפָּט בְּעֵמֶק הֶחָרוּץ הוּא עֵמֶק יְהוֹשָׁפָט



לְהַרְאוֹת צִדְקוֹתָיו לַאֲשֶׁר לֹא הֵבִינוּ פְּעֻלוֹתָיו



כִּי לֹא יָדְעוּ מַה הוּא

וּמִי מְכַלְכֵּל אֶת יוֹם בּוֹאוֹ כִּי מִמְּזְרַח שֶׁמֶשׁ עַד מְבוֹאוֹ



יִקְרָא חֲיָלָיו לְהִשְׁתַּחֲוֹת לַהֲדֹם רַגְלָיו



קָדוֹשׁ הוּא

אִם יִטָּמֵן אָדָם בַּצוּר לֹא יִתְעַלֵּם מִפְּנֵי הַצּוּר



כִּי צֶדֶק מַהֲלָכָיו וּמִשְׁפָּט כָּל דְּרָכָיו



צַדִּיק וְיָשָׁר הוּא

לְבָבוֹת חֹקֵר וּצְפוּנוֹת סֹקֵר וּמֹאֵס בַּשּׁקֶר וְטֹפְלוֹ מְעַקֵּר



וּכְנַהֲמַת יָם וְגָלָּיו יִנְהֹם עָלָיו



בַּיּוֹם הַהוּא

וְנִשְׂגַּב אֲדֹנָי לְבַדוֹ בַּיּוֹם הַהוּא.

הַהוּא יִקָּרֵא עֵמֶק הֶחָרוּץ כִּי שָׁם יֶחֱרַץ עֹמֶק הַדִּין וְעַד



צֶדֶק יָשׁוּב מִשְׁפָּט

לַיּוֹם אֲשֶׁר הוּא עֹשֶׂה אֵין מָנוֹס וְאֵין מִכְסֶה



בְּשִׁבְתּוֹ לַכִּסֵּא מִתְנַשֵּׂא וְאֵשׁ עֲלוּקָה בְּאַפּוֹ נִשְּׂקָה



כִּי בָאֵשׁ אֲדֹנָי נִשְׁפָּט

וַיִּקְרָא שְׁחָקִים וַיַּרְגֵז אֲרָקִים וְעָדָיו יְבֹאוּן עֵדָיו



לְהַגִּיד עַל עָשִׁיר וּמִסְכֵּן עַל כֵּן



לֹא יָקֻמוּ רְשָׁעִים בַּמִּשְׁפָּט

יַחַד כָּעָשִׁיר כָּאֶבְיוֹן כַּשָׁפָל כָּעֶלְיוֹן כַּמֶּלֶך כַּחֵלֶךְ



כָּאֶזְרָח כַּהֵלֶךְ כָּאָדוֹן כָּעֶבֶד כַּנִּמְצָא כָּאֹבֵד



שָׁם שָׂם לוֹ חֹק וּמִשְׁפָּט

חַיִּים כְּנֶגֶד מָוֶת וְאוֹר כְּנֶגֶד צַלְמָוֶת שִׁוָּה וְצִוָּה



לִדְבֹּק בַּטוֹב וְלִרְחֹק מִקְּטֹב יַדְרֵךְ עֲנָוִים בַּמִשְׁפָּט

זִכְרוּ זֶה הַיּוֹם נוֹרָא וְאָיוֹם לָכֶם לְפִדְיוֹן הוּכַן



לְשׁוֹבַבְכֶם עֲלֵי כֵן וּבְיוֹם נִקְרָא עוֹלָם וְנִבְרָא



וְנִפְקְדָה שָׂרָה וַתְּהִי עֲקָרָה מְלֵאֲתִי מִשְפָּט

קְרֻאִים לַדִּין יִפְחֲדוּ מִיּוֹם דִּין וְהַיָּרֵא דְבָרוֹ



נִזְהָר בְּמוּסָרוֹ וְיִסְּרוֹ לַמִּשְׁפָּט

וַיִגְבָּה אֲדֹנָי צְבָאוֹת בַּמִּשְׁפָּט.






בַּמִּשְׁפָּט עַמְּךָ אַל תַּרְשִׁיעַ וְהַיּוֹם בִּתְקוֹעַ שׁוֹפָר וְהָרִיעַ



תֵּחָשֵׁב לוֹ לִצְדָקָה

אֵילֵי שִׂפְתוֹתָם תְּקַבֵּל בִּמְשֹׁךְ בְּקֶרֶן הַיּוֹבֵל



לַעֲשׂוֹת צְדָקָה

מְלוֹךְ עֲלֵיהֶם יוֹם יַמְלִיכוּךָ וְזָכְרֵם יוֹם יִזְכְּרוּךָ



וּפְדֵה צִיּוֹן בְּמִשְׁפָּט וְשָׁבֶיהָ בִּצְדָקָה

אֶת זְכִירָתָם תַּחֲשֹׁב עֲקֵדָה לְהָכִין אוֹתָהּ וּלְסַעֲדָהּ



בְּמִשְׁפָּט וּבִצְדָקָה

דִּמְעָתָם תְּקַבֵּל כְּנִסְכֵּיהֶם וְהָרִים יִשְׂאוּ שָׁלוֹם לָהֶם



וּגְבָעוֹת בִּצְדָקָה

וְשָׂטָן תְּעַרְבֵּב וּפִי עַוְלָה תִּבְלוֹם וְהָיָה מַעֲשֶׂה הַצְּדָקָה שָׁלוֹם



וַעֲבֹדַת הַצְּדָקָה

מֶלֶךְ מַשְׁקִיט וְאֵין מַרְשִׁיעַ זֹכֵר נִשְׁכָּחוֹת רַב לְהוֹשִׁיעַ



מְדַבֵּר בִּצְדָקָה

וְהָאֵל הַקָּדוֹשׁ נִקְדַּשׁ בִּצְדָקָה.

בִּצְדָקָה יְיַחֲדוּהוּ יְצוּרָיו וּבְמִשְׁפָּט יְהַדְּרוּהוּ הֲדוּרָיו



בִּצְדָקָה יְוַעֲדוּהוּ וְתִיקָיו וּבְמִשְׁפָּט יְדַגְּלוּהוּ דְּבֵקָיו



בִּצְדָקָה יְהַלֵלוּהוּ כְרוּבָיו וּבְמִשְׁפָּט יַעֲרִיצוּ



לְאֵל נַעֲרָץ בְּסוֹד קְדֹשִׁים רַבָּה וְנוֹרָא עַל כָּל סְבִיבָיו.
