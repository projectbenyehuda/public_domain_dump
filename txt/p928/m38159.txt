


	
	ובעת ההיא העיר ה'
	








הַיִּשׁוּב וְהַמִּדְבָּר





– – – וּבָעֵת הַהִיא הֵעִיר ה׳ אֶת רוּחַ חָכָם אֶחָד אֲמִתִּי / וּשְׁמוֹ אִתַּי הַגִּתִּי / וְלוֹ רֵעֶה נֶאֱמָן אִישׁ צַדִּיקִי / וּשְׁמוֹ חוּשַׁי הָעַרְכִּי / וַיֵּשְׁבוּ יַחְדָּו בְּבֵית לֶחֶם יְהוּדָה / עַל הַתּוֹרָה וְהָעֲבוֹדָה / וּמִדֵּי שָׁנָה בְשָׁנָה הָלְכוּ בְּגִלְגּוּל לִסְבֹּב אֶת בֵּית אֵל / לָדַעַת מַה יַּעֲשֶׂה
יִשְׂרָאֵל / וַיִּרְאוּ כִי רַבּוּ רָעוֹת הָאָדָם בָּאֲדָמָה / אֵין תּוֹרָה וְאֵין חָכְמָה / אֵין יִרְאָה וְאֵין אַהֲבָה כִּי אִם תַּחְבּוּלָה וּמִרְמָה / וַיֹּאמֶר אִתַּי הַגִּתִּי: ׳שְׁמַע נָא חוּשַׁי רֵעִי וַעֲמִיתִי / הֲשַׂמְתָּ לִבְּךָ עַל בְּנֵי דּוֹרֵנוּ / כַּמָּה הֵרֵעוּ וְהִשְׁחִיתוּ מֵאֲבוֹתֵינוּ / כָּל חֲכַם לֵב יָבֹא לִסְפֹּד לְצָרָה וְלִבְכּוֹת עַל הַמְּצוּקָה / וְכָל חוֹנֵף, וּלְבָבוֹת גּוֹנֵב, כְּאֶזְרָח בְּיִשְׂרָאֵל יֵשֵׁב בְּסֻכָּה / עֲשׂוּיָה לְצֵל וּלְעֹנֶג / וְכָל חֲכַם לֵב נָמֵס כַּדּוֹנַג / כִּי כָל אִישׁ מִרְמָה וְדָמִים / בְּעֵינֵי הַדּוֹר חֲכָמִים / וְכָל אַנְשֵׁי צְחוֹק וְלָצוֹן / לִפְנֵיהֶם לְרָצוֹן / פְּנִימִי וְחִיצוֹן / וְיֹאמְרוּ ׳רְדָה!׳ / לְכָל מְמַשֵּׁל
וּבַעַל מִדָּה / וּלְכָל בַּעַל מוּסָר יֹאמְרוּ סָר / וְכֹה אוֹמְרִים הַשָּׂרִים וְהֶהָמוֹן: / ׳אֵין חָכְמָה וְאֵין תְּבוּנָה וְדַעַת זוּלַת אֵל הַמָּמוֹן׳ / לָכֵן אַתָּה חוּשַׁי, רֵעִי וַעֲמִיתִי / וְנֶאֱמַן בֵּיתִי, נָנוּסָה נָא מִפְּנֵי הַמְּצֵרִים / אֶל אֶחָד הֶהָרִים / אִם בְּמִדְבָּרִים אוֹ בִיעָרִים / בִּמְקוֹם שֶׁאֵין בְּנֵי אָדָם דָּרִים / וּמִפְּנֵי צָרָה גוֹבֶרֶת אָנוּ בוֹרְחִים / וּבַמִּדְבָּר מִי יִתֵּן לָנוּ מְלוֹן אוֹרְחִים.





וַיַּעַן חוּשַׁי הָעַרְכִּי: / ׳אַתָּה יָדַעְתָּ אֶת דַּרְכִּי / מִנְּעוּרַי עַד עָתָּה / אֵין לִי אֶלָּא אָתָּה / וּבַזְּמָן אֵין לִי מִמְּךָ תְּמוּרָה / וְלֹא יַפְרִידֵנִי מִמְּךָ שׁוּם מִקְרֶה / זוּלַת הַקְּבוּרָה / וְעוֹד כָּל יְמֵי חַיַּי / אַתָּה חֶלְקִי וּמַאֲוַיַּי / אִם בַּמִּדְבָּר תָּשִׂים קִנֶּךָ / שָׁם אֶעֱמֹד לַעֲבֹד
וּלְשָׁרֵת לְפָנֶיךָ׳.





וַיָּקוּמוּ הָאֲנָשִׁים בַּבֹּקֶר טֶרֶם יַכִּיר אִישׁ אֶת רֵעֵהוּ / וְאֶל הַמִּדְבָּר שָׂמוּ אֶת פְּנֵיהֶם לְדָעֵהוּ / וַיִּסְעוּ מֵעֵמֶק הָרֹפְאִים / וַיָּבֹאוּ בְמִדְבַּר פְּרָאִים / וּמִשָּׁם נָסְעוּ לְמִדְבַּר זִיּוּף, אֲשֶׁר בְּתַבְעֵרָה / וַיַּחֲנוּ בְמָרָה / וּמִשָּׁם נָסְעוּ דֶּרֶךְ מִדְבַּר שׁוּר / קִדְמַת אַשּׁוּר / וַיִּמְצְאוּ שָׁם אִילָנוֹת / נְטוּעוֹת עַל מֵי מַעְיָנוֹת / וּלְפָנִים יָשְׁבוּ שָׁם עַם בִּינוֹת / וּכְשֶׁרַבָּה הָאֲפִיקוֹרֶסֶת וְהַמִּינוּת / נֶחֶרְבוּ הַמְּדִינוֹת הָעֲדִינוֹת / וְנִשְׁאֲרוּ הָאִילָנוֹת / אֶל הָאֲרָיוֹת לִמְעוֹנוֹת / וְשָׁם קִנְּנוּ תַנִּים וּבְנוֹת הַיְּעֵנוֹת.





וַיֹּאמֶר אִתַּי הַגִּתִּי: / ׳פֹּה בָחַרְתִּי / לְהִסְתּוֹפֵף עִם הָאֲרָיוֹת וְהַנְּמֵרִים / וְלֹא עִם בְּנֵי עַמִּי הָאַכְזָרִים / כִּי חַיּוֹת הַיְעָרִים / אֵינָם בַּעֲלֵי בְחִירִים / לַעֲשׂוֹת לָנוּ רָעָה כִּבְנֵי הָאָדָם אֲרוּרִים / וְנָקֵל לַה׳ לְשַׁנּוֹת הַטֶּבַע הַבְּרוּרָה / מִלְּשַׁדֵּד אֶת הַבְּחִירָה / וְרַק אוֹתָהּ לְבַדָּהּ נָתַן אֱלֹהִים לִבְנֵי הָאָדָם לַעֲנוֹת / וְלֹא כֵן עָשָׂה אֶל הַתַּנִּים וּבְנוֹת הַיְעֵנוֹת / וְאִם נַעֲשֶׂה אֶת רְצוֹן אֱלֹהֵינוּ / הָאֲרָיוֹת וְהַגּוּרִים יָגוּרוּ לְפָנֵינוּ / וְאִם ה׳ עִמָּנוּ / כָּל חַיְתוֹ יַעַר יַשְׁלִימוּ אִתָּנוּ׳.





וַיַּכְרִיתוּ מִשָּׁם אִילָנוֹת / וַיַּעֲשׂוּ לָהֶם מְעוֹנוֹת / וְשָׁם יָשְׁבוּ וְעָסְקוּ בְסִפְרֵי תְּבוּנוֹת / וַיֹּאמְרוּ אַךְ טָב / לְתוֹרָה שֶׁבִּכְתָב / וְיָפֶה / לְתוֹרָה שֶׁבְּעַל פֶּה / וְלִפְעָמִים הָלְכוּ לְטַיֵּל בַּפַּרְדֵּס / לֵידַע אֶת אֲמִתִּית הַדָּת / כְּפִי הַחָכְמָה הַמְּסוּרָה / מִבַּעֲלֵי כֹחַ וּגְבוּרָה / אַנְשֵׁי רוּחַ הַקֹּדֶשׁ אֲשֶׁר קִבְּלוּ בְשָׂפָה בְרוּרָה / אֶת הַקַּבָּלָה כְּפִי הַתּוֹרָה / וּבְשָׁעָה שֶׁאֵין לֹא יוֹם וְלֹא לַיְלָה / שֶׁלֹּא לֵילֵךְ לָרִיק וְלַבֶּהָלָה / לָמְדוּ מְלִיצוֹת וּמוּסָרִים יָפִים /
מֵחֲסִידֵי הַפִּילוֹסוֹפִים / כְּדֵי שֶׁלֹּא יִהְיוּ סְכָלִים כְּקוֹפִים בְּהַשָּׁמַיִם וְהָעוֹלָם / וּבְנֶפֶשׁ וְצוּרָה וְגוֹלֶם / וּבְדֵעוֹת וּמִדּוֹת / קָנוּ עֶשֶׂר יָדוֹת / בְּמַרְאֶה וְלֹא בְחִידוֹת /
וְהִתְבּוֹנְנוּ בְּמַעֲשֵׂי ה׳ הַנּוֹרָא / בְּכָל מַה שֶּׁבְּעוֹלָמוֹ בָרָא / וְעַל הַכֹּל שָׂם חוּשַׁי אֶת עִיּוּנוֹ / וְאִתַּי הַגִּתִּי שָׂם עֲלֵיהֶם אֶת הֶגְיוֹנוֹ / וְנָתְנוּ שֶׁבַח לְיוֹצֵר בְּרֵאשִׁית / אֲשֶׁר בָּרָא אֶת כֹּל בְּיָמִים שֵׁשֶׁת. – – –
