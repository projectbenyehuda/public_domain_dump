


	
	מֵעֵבֶר לַיָּמִים (א)
	



א



עַל סַף הָאַרְבָּעִים


וּבְיוֹם אֶחָד מִמִּסְפַר יָמֶיךָ,



בְּהַצְהִיב צַמֶּרֶת בַּגְרוּתְךָ בָּרִאשוֹנָה –



מְעֻטָּר שְׁנוֹת הַבִּינָה –



וְנִפְתַּעְתָּ פִּתְאֹם לָדַעַת,



שֶׁכָּל הַהֵשֵׂגִים שֶׁסִּגַּלְתָּ לְךָ



בִּשְׁלַל הַפִּרְזוֹנִים וּבְמִשְעוֹלֵי שְׁלָפִים,



בִּנְכָסִים וּבְמַדָּעִים,



עַד אֵלֶּה כִבּוּשֶׁיךָ הַטְּמִירִים,



הַוַּדָאִיִים כִּמְעָט,



שֶׁרְכָשָׁתַם נַפְשְׁךָ לְלֹא הַפְגֵּן דְּגָלִים,



בְּכִסּוּפֵי הִתְבּוֹדְדוּת



וּבְהִזְדַּוְּגוּת חֲרֵדָה לְעַרְעֲרֵי מִין הָאָדָם –



כִּי כָּל אֵלֶּה – אֵלֶּה – קִנְיָנְךָ כֻּלּוֹ –



אֵינוֹ שָׁוֶה


  


בַּחַיִּים, שֶׁהָפְקְדוּ אֶצְלְךָ וְאָבְדוּ אֵי-שָׁם,



שֶׁאִי-אֶפְשָׁר לִרְאוֹתָם עוֹד וּלְתָפְשָׂם



בְּעַד קְנֵה צוֹפֵי-רְקִיעִים מִמּוֹנְטְ-וִילְסוֹן,



וְשֶׁלֹּא, שֶׁלֹּא יָשוּבוּ - - -



וְסֵרַבְתָּ לְתַכְלִית חֶשְׁבּוֹן,



שֶׁדִּמִּיתָ לְהִתְחַסֵּן וּלְהִתְרַבְרֵב בּוֹ



בִּפְנֵי הָעֲלוּמִים הַמִּתְחַדְּשִׁים מֵעֵבֶר לְיָמֶיךָ –



וְסָלַדְתָּ מִזִּקְנָה זוֹ, הַמִּתְהַדֶּרֶת



בְּלֶקֶט נִסְיוֹנוֹת חַיֵּי אָדָם,



וּמִתְקַשֶּטֶת בְּקִמְטֵי שֵׁיבָה הַמִּתַּלְמִים



כְּאִלּוּ נָהִיר לָהֶם כִּוּוּן דַּרְכֵי פִתְרוֹן,



וּמְפַצֶּלֶת בְּתַעַר כֶּסֶף שְׁחוֹר רַקּוֹתֶיךָ,



וְשֶעָמְדָה לְךָ בְּיֹקֶר שְׁנוֹת חַיֶּיךָ –



וְנִפְתַּעְתָּ קוֹבֵל, מִתְמַרְמֵר, מַתְרִיס,



כִּי אֵין לְפָנֶיהָ רַק אוֹתָה חִידַת הָעֲלוּמִים,



שֶׁנִּתַּשְּׁשָׁה, נִתְדַּהֲתָה בְשִׁלְהֵי צָהֳרֶיךָ



וְרוֹחֶשֶת מֵרִמַּת סְפֵקוֹת וְהִרְהוּרִים –



וּבְכָל מִטְעָנָהּ הָרַב אֵינָה שָׁוָה



בְּיוֹם אֶחָד שֶׁל נְעוּרִים,



קַל-רֶגֶל, בַּזְבְּזָן וּגְלוּם חֶשְׁבּוֹן,



הַפּוֹרֵק אֶת סִבְלוֹתָיו לְחוֹף-מָחָר



מְנַקֵּד אֵי-שָׁם מִשֶּׁפַע הַזְּמַנִּים,



וּלְמֶרְחַקִּים מִתְרוֹנְנִים,



מִתְרוֹנְנִים מִנַּצַח נְעוּרִים שֶׁלֹּא יָסוּף –



בִּכְחֹל, בְּזֶמֶר וּבְבִזְבּוּז



לֹא יָסוּף - - -




ב



אוֹרֵחַ


לָלֶכֶת בִּרְחוֹבוֹת תֵּבֵל אֶחָד אוֹרַח,



שֶׁיֵּשׁ לוֹ כָּאן מַכָּר וְשָם אִילָן רַב צֵל –



וּבְמֶחֱזַת דְּרָכִים לָשִׂישׂ עַל כָּל הֵד, עַל כָּל רֵיחַ –



לָחוּד: אֵי אֶתְאַכְסֵן הַיּוֹם, אֵי לֵב אַשְׁכִּיב עִם לֵיל?


 


לִשְׁמֹעַ רֹן תְּמוֹלִים פּוֹזֵם עַל סַף כָּל דֶּלֶת,



לִלְכֹּד כָּל רֶטֶט שֶׁנִּקְלָט בְּזִיז חַלּוֹן –



כְּהֵלֶך שֶׁגָּלַף אָזְנוֹ אֶל כָּל הֲמִית בַּחֶלֶד,



אֶל כָּל חֶדְוָה וּמְרִי, קוֹבְלִים בֵּמְרַאֲשוֹתָי מָלוֹן.


 


כֹּה נֵטֶל נְדוּדִים אַפְשִיל כְּבֵד עִצָּבוֹן וָזֶמֶר



וְהוֹד כָּל נוֹף וּפֶלֶא הָאִשָּה וְגוֹרַל הָאִיש –



וְאִם אֵי שָׁמָּה גֶּבֶר יְבֻקַּש לְאַהֲבָה וּמֶמֶר,



וְאִם אָח לְסֵבֶל רָב – אֲנִי, קַל טִלְטוּלִים, אָחִיש


 


עַד בַּיִת, בֵּיתִי-לִי, אָקִים בְּיוֹם שְׁפֹל הַבֶּרֶךְ,



עֲיֵף מַעְגַּל בַּגְרוּת וְחֶמֶד וִיגוֹנִים –



מֵעֵבֶר וּמִמּוּל יִסְגֹּר אָז אַחֲרוֹן כָּל דֶּרֶך



עַל הוֹד רְחוֹבוֹת תֵּבֵל וּשְאוֹן תְּמוֹלֵי מְלוֹנִים.


 

ג



מַרְגּוֹעַ


לָשֶׁבֶת עַל סַף, עַל אֶדֶן חַלּוֹן, עַל סְדָן אִילָן:



בַּתְּחִלָּה – לִסְקֹר בִרְחֹש תְּלוּלִית נְמָלִים וּבִקְרֹן פְּרָחִים,



וּלְשַׂבֵּעַ מַבָּט מִמְּלוֹא לִבְלוּב עַרְמוֹן וּלְהַרְעִיבוֹ לְהֶמְשֵׁך דְּרָכִים –



לָשֶׁבֶת מוּל אַרְבַּע קְצָווֹת עַל אוֹרָן וְצִלָּן.



וְלִצְפּוֹת – פִּתְאֹם לִצְפּוֹת – אֶל שֶׁקֶף הַקְּצָווֹת:



הוֹי, מַלְכוּת כָּל הוֹד – בְּרֹן כַּרְמִיל וּתְכֵלֶת, כָּל זִיו וִיקָר!…



וְזָכוֹר, זָכוֹר, בֶּן-אָדָם – אִיקוֹנִין הַבּוֹרֵא – שֶׁהַמֶּלְכוּת עִקָּר,



וְטָפֵל שְׁאוֹן כָּל הִתְנַצְחוּת, פְּגוּמִים כָּל שִׁקּוּל וְכִבּוּש
וְכָבוֹד.


  


לְגַמּאֵ, לִמְתֹּחַ הֱיוֹתְךָ – לְגַוְּנָהּ לָעֲבָרִים:



לָקֳבָל כָּל בָּרִי, שֶׁנִשְׁתַּפֵּר שוּב בְּהֵרְיוֹן חִידָה,



לְקֳבָל כָּל עוֹלָמוֹת-בְּרֵאשִית שְׁקוּפִים, אַלִּימֵי מִדָּה –



לְאַרְבַּעַת הָאֳפָקִים הַמְּפֻלָּשִׁים לְלֹא תַּרְמִית מְצָרִים.


 

ד



הַיּוֹם גּוֹסֵס


הֲלוּצִינַצִיָּה.



הִתְלַכְּדוּ כֹה אֳפָקִים, תְּהוֹם-שַׁלְהֶבֶת הִתְקַמְּרָה



וַתְּשַלְשֵל דָּלְיֵי-בְרֵאשִית לִמְשׁוֹת מֵהֶחָלָל



כַּדּוּר-אֶרֶץ מִתְעַרְסֵל בְּמִקְלַע קַרְנֵי שְׁקִיעָה.


  


בִּמְרוֹמֵי-בְלִימָה פָּתַח הָאֵלֶם שְׁעָרָיו הַמְתֻלָּעִים



וַיַּזְמִין כָּל מִי שְׁנִתְעַיֵּף לְהִתְאַחֵז



בְּמֵצַר שָׁרָשָׁיו – וּלְקַצְוֵי-אָן תְּלוּשִים יַעֲרֹג…


 


מַה סָּלְדָה נַפְשִׁי מֵאֵין-חוֹף: עֲסִיסֵי הָאֲדָמָה



אֲפָפוּהָ עוֹד וַחֲמִימוּת רֵיחָה – וַתִּתְכַּנֵּף נַפְשִׁי



בְּסֻבְּכֵי שְׁלֹש צַמָּרוֹת מְלַבְלְבוֹת מוּל חַלּוֹנִי - - -


  


הוֹי, כִּי פֶּתַע הַשְּׁקִיעָה תָקְעָה צִפָּרְנֶיהָ הַמְּכֻרְכָּמוֹת



בִּבְלוֹרִית שְׁלֹשֶׁת אִילָנַי וַתְכַרְסֵם עֹז מַצַּבְתָּם,



וַתְּפַרְקֵד צִלָּם, וַתְּנגַבְּחֵם וַתַּפְשִׁילֵם אֶל עָל:


 


וְהִנֵּה נִתְּקוּ, נִשְׁתַּלְפוּ לְשֶׁקֶט הָאַפְלוּל



וַיְתַמְּרוּ גִזְעָם לְמָטָּה… מִסָּבִיב פָּשְׁתָה



עַל דַּרְכֵי הָעֶרֶב עֲיֵפוּת: לֹא לָתוּר וְלֹא לִפְתֹר…


 


וַיְהִי אָז כְּאִלּוּ לֹא הֶאֱמִינוּ עוֹד בְּאַדְמָתָם,



רַק צֶלֶם הִבְרִיכוּ אֶל חֵיקָה בְּקִדַּת פְּרִידָה



וַיַאֲבִירוּ לִמְחוֹזָם וְעַנְפֵיהֶם סְמוּרִים בִּיקוֹד.


 


וְקַצָּץ-פֶּלִאי הַמְּגַרְזֵן בְּיַעֲרוֹת עַנְנֵי-חֲלוֹף



שָׁט לִקְרָאתָם – וּבַעֲקֵבוֹ עַכָּבִיש-לֵיל



כְּבֻלְמוּס-תֹּהוּ יַעֲלֹק כָּל זִיו, כָּל הֵד לַיְקוּם – – –


 


הָהּ, אֵלִי! עַד שֶׁיָפוּחַ יוֹם חָדָשׁ, מִי יִתְחַזֵּק



נֶגֶד אֳפָקִים מְאֻכָּלִים וּמַחֲשַכִּים רְעֵבִים, חוֹסְלִים



כָּל קַיָּם, הַשָׂשׂ לְשׂבַע הֱיוֹתוֹ וְהִשָּׁתְלוֹ?…


  


כִּי אִם אֵלֶּה אִילָנֵי הָאֵיתָנִים, מָעוֹז כָּל נוֹף,



לֹא הָיוּ גַם הֵמָּה כָּל עַצְמָם, כִּי אִם מִשְׂחָק בְּדוּי-יוֹם



לֹא הָיוּ רַק אֶרֶג-אֶפַע, רַק עִנְקֵי-מִקְסָם:


 


אָז אֵיפֹה, מָתַי וְאֵיך הָיְתָה-חָיְתָה נַפְשִׁי,



מִמִּטְוֵה כָּל רִקְמָתַיִם עֲדִינָה, קְלִילָה,



קְלוּעַת זִיו מִסְתוֹרִין, פִּרְפּוּרֵי הֵדִים וְגַוָּנִים?…


 


גָּסַס הָאֹפֶק… דֻּבְלְלוּ צְלָלִים… קַר וּמְעֻרְטָל



הַיְקוּם הִרְכִּין עָרְפּוֹ, הִתְחַבֵּא וְאֶל נַפְשׁוֹ חָמָק –



וּבְחֶרְדַּת הַחִדָּלוֹן הִתְפָּלַּלְנוּ לְמוֹפֵת וּפְדוּת:


 


אֱלֹהִים! יְהִי אוֹר וְחֹסֶן אַלּוֹנִים וְנַפְתּוּלֵי אָדָם,



וְהֵרוֹן וָלֶדֶת וְסִבְלוֹת וְשׂבַע אֲדָמוֹת –



פֶּן יִשְׂכַּל הַנֵּצַח אֶת זְמַנָּיו וְיִגְהַר לָמוּת – – –
