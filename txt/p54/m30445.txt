


	
	גדיש
	







א. קוּרֵי קַיִץ






צָרָה וּתְלוּלָה הַמְּסִלָּה… נִשְׁתַּמְּטָה מִדֶּרֶךְ הַמֶּלֶךְ:

מְטַפְּסָה עַל בֵּין שְׂדוֹת דָּגָן, מִתְגַּהֲרָה גְמִישָׁה לְמִשְׁעוֹל,

מִתְלַכְסְנָה לְעֵמֶק כִּמְטַחֲוֵי שַׁעֲטַת סוּס, וְסַקְרָנִית

שׁוּב תַּעְפִּיל בֵּין גַּלֵּי הַחִטָּה, וְשׁוּב יוֹרְטָה בֵין תְּלָמִים מִסְתַּמְּקִים

שֶׁל סֶלֶק וְלֶפֶת וּכְרוּב… רֶגַע תִּגְחֹן וְצָנְחָה־לָפְתָה

מַרְגְּלוֹת צַלְעֵי גַבְנוּנִים הַשּׁוֹמְרִים לְמֶרְחַב מִישׁוֹרִים,

עַד שׁוּב תַּדְרִיג־תַּפְשִׁיל שְׁבִיל־פְּתַלְתֹּל וּזְרִיזָה תְּשַׁרְבְּבוֹ מָטָּה…

וּבְחָפְזָהּ מִגִּבְעָה אַחֲרוֹנָה — וְהִיא טֶרֶם תַּעֲצֹר בְּנוּסָהּ —

וּלְפֶתַע תְּפֻשַּׂק לִימִין וְלִשְׂמֹאל, וּמְרֻדֶּדֶת תִּזְדַּלֵּג,

וְיֹאבַד מַסְלוּלָהּ בַּדֶּרֶךְ הָרְחָבָה שֶׁאָרְבָה לַפְּלִיטָה

מֵעֵבֶר כָּל גִּבְעוֹת מִפְלָטָהּ, שָׁם פִּלְּשָׁה בַסֵּתֶר לָהּ נָתִיב,

בַּעֲזֹז מֶרֶד פִּלְאִי בְחִשְׁקָהּ לְעֵבֶר מֶרְחַקִּים לֹא־סְלוּלִים

וַתְּשָׂרֵךְ מִדֶֶּרֶךְ הַמֶּלֶךְ. — כִּשְׂרִיד תְּבוּסַת־מִרְדָהּ מִתְחַמֵּק

עוֹד פְּתִיל־מִשְׁעוֹל גַּלְמוּד בַּחוֹל, רֶגַע יִזְחַל וַעֲרוּף רֹאשׁ יִכָּנֵף

בְּשַׁעַר חֲצַר הָאֲחֻזָּה…





בַּמּוֹרָד הָרִאשׁוֹן לַמְּסִלָּה

כְּבָר הִרְגִּישׁ הַסּוּס כִּי יַד רוֹכְבוֹ נִתְרַשְּׁלָה — הָרֶסֶן הַמְדֻלְדָּל

אַט הִקִּישׁ עַל כְּרָעָיו בְּרוּצוֹ וְהִתְחַלְחֵל הַמֶּתֶג בֵּין שִׁנָּיו;

תָּמֵהַּ הוּא פִגֵּר לְאִטּוֹ, וַיָּתָר בּוֹ מִחְיָה אֲרָעִית

בִּפְאַת שְׂדוֹת קָצִיר קֵרְחִים, בְּלֶקֶט שִׁבֳּלִים בּוֹדֵדוֹת,

שֶׁגָּלְשׁוּ מִשְּׂפַת הָעֲגָלָה תְמוֹל־שִׁלְשׁוֹם בַּדֶּרֶךְ לַגֹּרֶן…

הַיָּמִים יְמֵי כְּלוֹת הַקָּצִיר, וְשִׁיר שִׁלְהֵי קַיִץ מִסְתַּנֵּן

בְּמִשְׁנֵה־תְכֵלֶת לַמְּרוֹמִים, וַעֲמֻקָּה כַדְּמָמָה הַשְּׁקוּפָה

תְּרַטֵּט פֹּה חֶמְלַת־כִּפְלַיִם עַל כַּנְפֵי הַזֹּהַר הַמְזֻקָּק,

וְיַד־מִי מְבַזְבְּזָה אוֹתָהּ נַחְשׁוֹלִים נַחְשׁוֹלִים בַּבְּלִימָה,

לִצְבֹּט עַל לֹא־דָבָר כָּל נֶפֶשׁ…בְּכָל אֵלֶּה מֶרְחֲבֵי שׂבַע,

שֶׁזִּמְרַת מַגָּלִים גְּרָפָתַם עוֹד שִׁלְשׁוֹם וּבִטְחַת אֲלֻמִּים,

מְמָרְטְטָה תוּגָה חֲרִישִׁית וְעַד סַף הָאֲסָמִים תְּקַנֵּן:

לְקִצְבָּהּ מִתְנַהֲלָה מַחֲרֵשָׁה וּמְשַׁסְּפָה שְׁחוֹר־רִגְבֵי שָׂדַי,

וּלְבָנָה־צְנוּעָה הִיא פוֹשָׂה־מִתְפַּקְּרָה מֵאַבְנֵי הַטַּחֲנָה

הַגּוֹרְסוֹת גַּרְעִינֵי הַשַׁתָּא — הֲלֹא הִיא הַתּוּגָה הָרַבָּה

שֶׁעָזְבוּ הָאָבִיב, הַקַּיִץ, שֶׁעוֹזֵב כָּל תְּמוֹל שֶׁנִּסְתַּחֵף,

וְכָל מַה שֶּׁהָיָה־לֹא־הָיָה בְּחֶתֶף עִדָּנִים לַחַיִּים.





הַזָּקֵן הֵנִיעַ בְּרֹאשׁוֹ לְגָרֵשׁ מַחֲשָׁבָה טַרְדָּנִית,

שֶׁנִּקְבָה לְאַט וּבִדְמִי וְנִתְאַכְסְנָה בִפְנִים אֵי שָׁם עָמֹק,

וּבְזִמְזוּם מְגַמְגֵּם־סַרְבָּנִי הִיא חִטְחֲטָה־חָתְרָה בְּמֹחוֹ,

וְהוּא עִמְעֵם עָלֶיהָ לַשָּׁוְא, עַד שֶׁנִּרְאֲתָה לוֹ בְדִמְיוֹנוֹ

כְיַתּוּשׁ שֶׁפָּלַט בַּשָּׁרָב אֶל תַּחְתִּית הַכֶּרֶס לַבְּהֵמָה,

וְנִתְקַע בְּרַגְלָיו וּבְעֻקְצוֹ, וּמוֹצֵץ שָׁם אֵגֶל דָּם כֵּהֶה,

לְאַט וּלְתֵאָבוֹן, וּמְחֻצָּף הוּא לוֹעֵג לִתְנוּפַת זְנָבָהּ…

הִתְאַמֵּץ הָרוֹכֵב לְטַלְטֵל מִמֶּנָה אֶת דַּעְתּוֹ וּלְיַשְּׁבָהּ

בְּדַאֲגוֹת מִשְׁקוֹ וּקְטִיפוֹ, הִשְׁתָּעָה וַיְפַזֵּל בְּעֵינָיו,

עַד נִתְקַל מַבָּטוֹ בְמִכְסֵה עֲרֵמַת הַחִטָּה בַשָּׂדֶה,

שֶׁפִּקְפֵּק הָרוּחַ בְּתִבְנוֹ הַלַּיְלָה וְצָרִיךְ לְרַדְּדוֹ

בַּתֶּבֶן הֶחָדָשׁ שֶׁלְּאַחַר הַדַּיִשׁ; הִכְרִיחַ אֶת מֹחוֹ

לְטַפֵּל קָט־רֶגַע בְּמִקְוֵה־הַחֹרֶף לַדָּגִים וּלְשַׁקֵּל





הֲיֵשׁ לְהַעֲמִיקוֹ טִפְחַיִם: בַּחֹרֶף הַקַּר דְּאֶשְׁתַּקַּד

הִגְלִידוֹ הַכְּפוֹר עַד תַּחְתִּיתוֹ וַתָּמָת הַדָּגָה בַקֶּרַח,

וְרַבּוּ סִימָנִים הַמְבַשְּׂרִים הִשָּׁנוֹת הַחֹרֶף בְּתָקְפּוֹ —

וּבְעוֹד שָׁבוּעוֹת מִסְפָּר, בִּשְׁבֹת הָעֲסִיסִים בָּעֵצִים, אַל יִשְׁכַּח

לִנְקֹף אֶשְׁכְּרוֹעַ הַמְּשׂוּכָה, שֶׁהִפְרִיא וְהֶאֱפִיל עַל מִדְרוֹן

הַשָּׂדוֹת, הַנִּשְׁקָף אֶל פְּנֵי גְזֻזְטְרַת בֵּיתוֹ וּמְלַבֵּב כָּל רוֹאָיו

בַּהֲדַר צִבְעֵי שְׁטִיחָיו — עַד אָבִיב הַבָּא הָעֵץ יַקְרִים פְּצָעָיו — — —

אַךְ חָלְפוּ כָל אֵלֶּה הַדְּאָגוֹת כְּפַרְפְּרֵי קַיִץ פְּזִיזִים

מִשַּׁרְווּל מְטֻלָּא לִמְכַשֵּׁף — בַּחֲשַׁאי לוֹ יְנַקֵּר הַיַּתּוּשׁ,

עַד נִכְנַע הָאִישׁ וּנְטוּל רָצוֹן הִתְמַכֵּר לְשֶׁטֶף זִכְרוֹנוֹת:





הָאָבִיב כֹּה פִּגֵּר הָשַׁתָּא, נִתְעַרְבְּלוּ בָרָד וּגְשָׁמִים,

וּבְסִיוָן עוֹד קִרְצְפוּ לַשָּׁוְא קַרְנֵי שַׁחַר בַּחֲלֻדָּה הָרְטֻבָּה

לְמַפְּלֵי עֲנָנִים בָּאֹפֶק, שָׁוְא נָגְחוּ נְגֹהוֹתָיו הַקְּלוּשִׁים

אֶל חַשְׁרַת הַכְּרֵסִים הַנְּפוּחִים — עִם מִטְרוֹת הַבֹּקֶר עֻמָּמוּ…

וַיְהִי כָל דֶֶּרֶךְ מְעֻרְפָּל וּבָחוּשׁ, וּלְלֹא תִּקְוַת מָחוֹז

בִּתְּרוּהוּ אוֹפַנִּים לַמֶּרְחָק — — — וּבְאֶמְצַע הַחֹדֶשׁ הִתְגַּבֵּר

הַקַּיִץ, וַיַּשְׁכֵּם וַיַּפְשֵׁל אֶת אֵדֵי קִפּוּלָיו מֵעֵבֶר

לַיְּעָרִים, וַיַּצֶּת לַפִּידֵי־רְסִיסִים בִּירַק צַמְרוֹתֵיהֶם,

הִזְדָּרֵז וַיִּפְטֹר כָּל פֶּקַע שֶׁסֻּרְבַּל בָּרֹטֶב, וּבִחֵל

וְגִיֵּד בְּחַרְטוֹ הֶעָדִין כָּל עָלֶה מְכֻרְבָּל, וַיְעַנְבֵּר

הַשְּׂרָף בְּכָל פִּצְלַת אִילָן… אָז גָּלַשׁ לְיִוְנֵי הַדְּרָכִים,

וַיְגַבֵּל וַיָּטַח עַבְטִיטָם וַיְנַגְּבֵם בְּחָצְנוֹ שַׁלְהָבוֹת —

וְעוֹד יוֹם וְיוֹמַיִם וַיְלַבְלֶב הֶחָצִיר מְסֻלְסָל וְעָנֹג,

וְיַרְקוֹ הַכֵּהֶה מְפֻטָּם — — — זֶה הָיָה עִם קְצִיר חֲצִיר סִיוָן

בָּאָחוּ הַמְדֻשָּׁן נִיחוֹחַ וָרֹן, בִּתּוֹ שָׁמְטָה מַגָּלָהּ,

הֵצֵלָּה בְיָדָָהּ עַל מִצְחָהּ הֶחָבוּשׁ מִטְפַּחַת בַּד־לָבָן,

וַתִּצְהַל לִקְרָאתוֹ: הָהּ, אַבָּא, הֲלֹא אוֹרְחִים בָּאִים וּקְרֵבִים

בַּמְּסִלָּה… הַבִּיטָה, שָׁם, שָׁם… וְעוֹד עֵינוֹ הַחַדָּה לֹא עָמְדָה,

כְּאָז לִפְנֵי שָׁנִים עֶשְׂרִים, וַיְמַצַּע מַבָּטוֹ לַמֶּרְחָב

הַצָּלוּל וּמְעַוֵּר בְּזָהֳרוֹ, וַיַּרְא שְׁנֵי אֲנָשִׁים עֲמוּסִים

תַּרְמִילֵי תַיָּרִים מִתְקָרְבִים לְאִטָּם מִגִּבְעַת הַמְּסִלָּה,

וְיוֹרְדִים אֲלֵיהֶם לָאָחוּ… לֹא אָרַךְ הַמַּשָּׂא וּמַתָּן:

שְׁנֵי רֵעִים, מְשׁוֹרֵר וּפַסָּל, לְרֶגֶל אֻמְנוּתָם הַדּוֹרְשָׁה

מְנוּחָה וְרִכּוּז הַנֶּפֶשׁ, הֲלוּמֵי שְׂאוֹן כְּרָךְ הֵנָּה עָרְקוּ,

וַיַּחְמְדוּ נָווֹ לְשִׁבְתָּם, כִּי פֻתּוּ לְבֵיתוֹ הַמַּשְׁקִיף

מִתֵּל עֲטוּף יֶרֶק לְמֶרְחַק הַשָּׂדוֹת, וּמְעֻטָּר גַּן נֶהְדָּר,

הַמַּשְׁחִיל אֶת שָׁבְלוֹ הָרָקוּם בְּזִיו חַמָּנִיּוֹת וּפְרָגִים

אֶל רַחֲבַת הַקָּמָה בַשְּׁפֵלָה — אִם יִכְשַׁר הַדָּבָר לְפָנָיו,

וְיִשְׂכְּרוּ חֶדֶר בְּבֵיתוֹ לְחָדְשֵׁי הַקַּיִץ, וְתֵכֶף

יוֹרִידוּ לְשָׁם תַּרְמִילֵיהֶם — אִם תֵּאוֹת הָעַלְמָה הַכְּבוּדָה — —





הִתְבָּרֵךְ בִּלְבָבוֹ, כִּי הִקְרָה אֱלֹהִים לִקְרָאתוֹ קַיָּטִים

יְשָׁרִים וַאֲנִינִים — כֹּה תָפַשׂ אֶת טִיבָם בָּרֶגַע הָרִאשׁוֹן —

וְחָלְפָה דְאָגָה מִלִּבּו: בַּת תַּמָּה וּבָרָה בְבֵיתוֹ,

וְגִילַת עֲלוּמִים בְּדָמָהּ וְאֵין אֵם צוֹפִיָּה הֲלִיכוֹת…

הוּא זָכוּר: אֶשְׁתַּקַּד בַּקַּיִץ הִתְאָרְחוּ בְּבֵיתוֹ צְעִירִים,

וְשִׁירִים וּמְחוֹלוֹת לֹא שָׁבְתוּ, וְלֹא הִקְפִּיד כְּנֶגְדָּם, כִּי אָמָר:

תִּתְעַלֵּס הַנַּעֲרָה וְתָפִיג אֶת שִׁמְמוֹן לֵילוֹת הַחֹרֶף

שֶׁבִּלְּתָה בַכְּפָר עִם אָבִיהָ מִיּוֹם גָמְרָהּ חֹק לִמּוּדֶיהָ

בָּעִיר; כִּי כֹה קָשֶׁה עָלֶיהָ לְעָזְבוֹ, אַף כִּי כָלְתָה נַפְשָׁהּ

לֹא פַּעַם בַּסֵּתֶר, בַּעֲנֹב עֲרְפִלֵּי סְתָו אֶת הַגַּן וְהַבַּיִת,

לְהֶמְיַת הָעִיר… וְעַל כֵּן שָׂמַח כָּכָה כִּי יָכֹל לְאַכְסֵן

בְּנֵי כְּרָךְ, וְהִסְתַּלֵּק לִמְלַאכְתּוֹ וְלֹא כִהָה בָמוֹ — עַד פַּעַם

נִזְדַּמְּנוּ אֶל בֵּיתוֹ זְקַן כְּפָרוֹ עִם אִשְׁתּוֹ, וְזוֹ גִלְּתָה אֶת אָזְנוֹ,

כִּי נוֹדְפִים הַצְּעִירִים לְמִינוּת וְכוֹפְרִים בָּעִקָּר, וְהַרְבֵּה

רְאָיוֹת לָאִשָּׁה שֶׁעָקְבָה אַחֲרֵיהֶם בַּלָּאט, כִּי יָדְעָה:

יְתוֹמָה הַנַּעֲרָה לְלֹא אֵם, וְהָאָב — לוֹ עֵין גֶּבֶר וְלֹא יַבְחִין,

וְאֵיךְ תַּחְמִיץ מִצְוָה גְדוֹלָה כָזוֹ וְתִתְיַצֵּב מֵרָחוֹק…

מֵאָז שֻׁבְּבָה דַעְתּוֹ עָלָיו וְנִתְמַעֲטָה יָדוֹ וְקֻפְּחָה

עֲבוֹדַת הַקָּצִיר בִּמְלוֹאוֹ — אַךְ הוֹדוֹת לָאֵל, הַכֹּל נִגְמַר

בְּשָׁלוֹם, כִּי קֵהָה טוּב־טַעְמָהּ שֶׁל בִּתּוֹ בְּכָל גַּסּוּת וְתִפְלוּת —

וּמְאֹד נָחָה דַעְתּוֹ מֵאֵלֶּה הַשְּׁנַיִם, כִּי בָטַח בִּטְבִיעַת

תְּפִישָׂתוֹ, וְנֶאֶמְנוּ עָלָיו חֲבִילוֹת סִפְרֵיהֶם, וַיִּגְזֹר:

אֶל נָכוֹן הֵם כֵּנִים וַהֲגוּנִים, עֲרֵבָה בָזֶה הַשְׂכָּלָתָם,

וְחוּט שֶׁל רְצִינוּת שֶׁהֻשְׁחַל מִמִּצְחָם וְנָגַע בְּדַבְּרָם

בְּזָוִיּוֹת שְׂחוֹק־דַּק לִשְׂפָתָיִם. — אָכֵן עַל שְׂחוֹק זֶה וְחִנּוֹ

עָמְדָה הַבַּת מִלְּכַתְּחִלָּה: זֶה דַרְכָּהּ לְהַבְחִין בִּרְמָזִים,

לִקְלֹט בְּנוֹת־גָּוֶן וּרְחָשִׁים וְלַחֲשׂוֹף אֲרֶשֶׁת פַּרְצוּפִים,

מֵרִפְרוּף שֶׁנִּכְנַף בֵּין קִמְטֵי הַמֵּצַח, מֵחִיּוּךְ שֶׁגָּלַשׁ

כִּמְעַט וְהִתְחַמֵּק… סִימָנִים כָּהֵמָּה לָרֹב לָהּ נִתְבַּקְּשׁוּ

בִּפְנֵי שְׁנֵי הָרֵעִים, וַתָּדָן אֶת כֻּלָּם לִזְכוּת. — אָמְנָם יָמִים

מֻעָטִים לֹא עָבְרוּ, וְשׁוּב סָרָה הַזְּקֵנָה עִם עֶרֶב אֶל בֵּיתָם,

וַתְּפַטֵּט עַל דָּא וְעַל הָא, וּכְמוֹ אַגַּב אוֹרְחָה הִפְלִיטָה:

שָׁמַעְתִּי, יַקִּירַי, מְשַׁמֵּשׁ דַיַּרְכֶם הָאֶחָד בְּאֻמְנוּת הַפִּסּוּל,

מְלָאכָה חֲשׁוּבָה מְאֹד, וּמִצְטַיֵּן כְּפָרֵנוּ בַּסְּבִיבָה

בִּפְסִילֵי הַקְּדוֹשִׁים, וְאֵין כְּבֵית־עָלְמִין שֶׁלָּנוּ לִיפִי

מַצֵּבוֹת, וַאֲנִי רַבַּת אֵדַע פַּסָּלִים, וְכֻלָּם מְיַשְּׁרִים

בְּיִרְאָה וְהֵם אַנְשֵׁי חֲמוּדוֹת, וְלֹא אַחַת חַשְׁתִּי עֲתִידוֹת:

הַנַּעֲרָה תִּגְדַּל וּנְבַקֵּשׁ לָהּ שָׁם בְּחִיר נְעוּרִים…כֹּה פָתְחָה

וְנֶאֶנְחָה מְרוֹרוֹת — אַף יָדְעוּ הַזָּקֵן וּבִתּוֹ אֶת שְׁכֶנְתָּם:

לֹא דַרְכָּהּ לְהַמְתִּין בִּשְׁבָחִים — וּבְעוֹד רֶגַע פִּעְפְּעָה הַשִּׂטְנָה:

הֲלֹא כִי הִיא נִכְזְבָה מְעַט מֵאֻמְנוּתוֹ — דַּיָּרָם לֹא חָקַק

מַצֵּבוֹת, לֹא שָׁקַד עַל קְבָרִים… וְאָמְנָם הִיא יוֹדְעָה כִּי דָבָר

אֵין לָהּ עִם אֲנָשִׁים הַלָּלוּ; לֹא הִיא אַחֲרָאִית לִמְלַאכְתָּם

וּלְאָרְחָם, אַךְ קָשְׁתָה עָלֶיהָ הַחוֹבָה לַיְתוֹמָה — וּפַעַם

הֵצִיצָה לְחֶדְרָם, וַתִּרְאֶה בְגַלֵּף דַּיָּרָם אֶת גְּלָמָיו,

וְנִתְרְזָה עֵינָהּ מִפְּרִיצוּת מַעֲרֻמִּים, לְלֹא מַעֲטֵה צְנִיעוּת

שֶׁקָּשַׁר בְּחֹמֶר וּבְאָבֶן… וְנִתְּנָה לְהֵאָמֵר הָאֱמֶת:

מִכֵּיוָן שֶׁנִּתְפַּשׂ כָּל כָּךְ — מִטָּרֵף בְּלִבְטֵי בְשָׂרִים —

לְעֶרְיָה מְשַׁטָּה לָאִשָּׁה, לִבְלִיטַת כְּתֵפַיִם וּזְרוֹעוֹת,

וַאֲבָרִים עוֹד פְּחוּתֵי מַדְרֵגָה — מֵחֹמֶר הֵן קֹרַץ הָאָדָם,

רַק בָּשָׂר וָדָם וְגָרוּעַ מִזֶּה וְכַיּוֹצֵא בָאֵלֶּה;

מִכֵּיוָן שֶׁשָּׁקַע בְּהַבְלֵי הַטֻּמְאָה לְגוּפִים וְצוּרוֹת,

הִתְמַכֵּר לְתַבְנִית וּדְמוּת, וְהִתְחַיֵּב בְּנַפְשׁוֹ שֶׁנְּטָשָׁהּ

פְּרוּצָה וּלְלֹא דַאֲגַת סְיָגִים, נִתְרַחֵק מִיִּרְאַת שָׁמַיִם,

כַּיָּאוּת לְפַסָּל אֲמִתִּי הָעוֹסֵק בִּדְבָרִים שֶׁל קְדֻשָּׁה,

וּמַצִּיב צִיּוּנִים לִנְפָשׁוֹת… וְהִיא אֵינָהּ חֲסֵרָה הוֹכָחוֹת,

כִּי הֵיטֵב הִתְחַקְּתָה עַל טִיבוֹ — כִּי יָדְעָה יְתוֹמָה הַנַּעֲרָה,

וְרַק גֶּבֶר הָאָב, וְעָלֶיהָ לַעֲמֹד בְּתַפְקִידָהּ — — — הָשַׁתָּא

רַק חִיֵּךְ הַזָּקֵן לַחֲשָׁשׁוֹת שֶׁל שְׁכֶנְתּוֹ וַיְצַדֵּק דַּיָּרוֹ:

אֲשֶׁר אֵינוֹ פָחוֹת צָנוּעַ מֵאֵלֶּה רֵעֶיהָ הַחוֹצְבִים

מַצֵּבוֹת לַמֵּתִים, אַךְ פָּחוֹת מֵהֵמָּה הוּא בָהוּל עַל כֶּסֶף,

וּלְשִׁבְחוֹ יֵאָמֵר הַדָּבָר — כִּי אֵלֶּה שֶׁהִיא מַאֲמִירָתָם

לִירֵאִים וּשְׁלֵמִים, וְאַף פַּעַם לֹא חִלְּלָה יָדָם מַגְלֵפָם

לַחֲשׂוֹף מַעֲרֻמֵּי הַגּוּף וְתוֹעֵבָה לָהֶם עֶרְיַת בְּשָׂרִים, —

הֵם כְּפוּנִים וּמְהַמִּים כָּל כָּךְ לִצְבֹּר הוֹן וְלֹא עָפָר, כְּאִלּוּ

לֹא זֶה רָמַז לָמוֹ מִגֶּשֶׁר תָּחוּחַ לַחַיִּים עַל דֶּרֶךְ

כָּל בָּשָׂר, שֶׁהֵם תְּדִירִים לוֹ כָל יְמֵיהֶם, וּמְקַשְּׁטִים בְּמַעֲשֵׂה

מַחֲשֶׁבֶת אֶת תִּלֵּי מִשְׁעוֹלָיו… וַהֲלֹא זָכוּר עוֹד סִכְסוּךְ שֶׁאֵרַע

לוֹ עִם קְרוֹבֶיהָ, בְּמוֹת עָלָיו אִשְׁתּוֹ, וּבְוַדַּאי לֹא שָׁכְחָה

גַּם הִיא… אָז הִתְחַמְּקָה הַשְּׁכֵנָה נְזוּפָה, כִּי פָגְעוּ בְאֻמְנוּת

מִשְׁפַּחְתָּהּ; בַּת חוֹצֵב מַצֵּבוֹת הִיא הָיְתָה וְאַחֶיהָ בִּמְלָאכָה

זוֹ תָפְשׂוּ וַעֲלֵיהֶם כְּבוֹדָהּ…





הַזָּקֵן הִתְחָרֵט, כִּי עָלַב

אֶת שְׁכֶנְתּוֹ, אַךְ שָׁמַר בְּנַפְשׁוֹ זֶה יָמִים שִׂנְאָתוֹ הָרַבָּה

לָעוֹסְקִים בְּבָתֵּי הָעָלְמִין, וַאֲדִישִׁים־מִתְחַסְּדִים מִסְתַּכְּלִים

בִּפְנֵי זְוֲעַת מָוֶת, כְּאִלּוּ הֵם פָּתְרוּ הַסּוֹד… לְעֵת זִקְנָה

עוֹד גָּדְלָה הַטִּינָה בְלִבּוֹ לְכָל סַפְסְרֵי מָוֶת הַלָּלוּ,

הַלּוֹטְשִׁים עֵינֵיהֶם אֶל כָּל חֶלְקָה פּוֹרָה, לְזָרְעָהּ מַצֵּבוֹת,

וּלְסַפְּחָהּ אֶל נַחֲלַת אֲדוֹנָם לְרֶמֶז אֶצְבָּעוֹ הַגְּרוּמָה,

כְּמוֹ חִנָּם יַעְפִּיל הָאָדָם — הַפִּתְרוֹן לְאִתֵּי קַבְּרָנִים —

וְכָל שִׁירַת חַיִּים וּמִרְדָּם הַקָּשָׁה בַחֲזוֹן הַחִדָּלוֹן,

כָּל תִּכְלָה וְתַכְלִית לָאָדָם, כָּל קַשְׁיוּת וְשִׁכְרוֹן נַפְתּוּלָיו,

וּתְבִיעַת סַנְוֵרִים עַקְצָנִית לְקַיֵּם מֶרְכָּזוֹ הַפָּעוּט

בְּעִגּוּל הֲוָיָה חֲלוּלָה — הֵם צוֹרְרִים בְּאֵלֶם אֲבָנִים,

בְּאִוְשַׁת הַוִּתּוּר הָאַחֲרוֹן הַמְרַחֵף עַל כְּתָב הַמַּצֵּבוֹת — —

אַךְ רָחוֹק מֵהֶם, כַּמָּה רָחוֹק מִכָּל סֵמֶל מָוֶת וּפְחָדָיו,

דַּיָּרוֹ הַפַּסָּל… "לְקַעְקֵעַ וּלְצַלֵּק כָּל קְלִפָּה וְלַחְתֹּר,

כְּנַהֲקַת אָח חַי בְּיַתְמוּתוֹ, אֶל מִקְשַׁת אָח אִלֵּם וּמְרֻחָק,

לְנַקֵּר עַד עָמְקֵי מַעְיָנָיו וּלְפַלְגָּם לַאֲפִיקֵי הַמְרֻשָׁשׁ,

לְשַׁחְרֵר תַּמְצִיתוֹ הַטְּמִירָה, לְגַבֵּר בָּהּ דֳּמִי שֶׁפֻּכַּח

לְכַל מִלְמוּל, לָשׁוּב וּלְהִזְדַּוֵּג לַדּוֹמֵם — זֹאת יַחְפֹּץ כֵּילַפִּי…"





הֲלֹא אֵלֶּה דְבָרָיו בְּאָזְנֵי הַנַּעֲרָה בִתּוֹ, שֶׁאָמַר

בְּאֶחָד מֵעַרְבֵי הַקַּיִץ הַצְּלוּלִים, כִּי רָצָה לְחַוֵּר

לָהּ סוֹד כְּמִיהַת אָמָּן, וּמְבֻיָּשׁ אָז הִבִּיט לְעֵבֶר רֵעֵהוּ,

כְּשׁוֹאֵל: הַאֻמְנָם הֵיטַבְתִּי הַיּוֹם לְהִתְבַּטֵּא, הַמְשׁוֹרֵר,

וּמְדֻיָּק וּמְנֻסָּח מִשְׁפָּטִי, אוֹ עִלֵּג וּמְשֻׁבָּשׁ, כְּרָגִיל —

וַיְהִי כְמוֹ בִקֵּשׁ סְלִיחָה מֵרֵעוֹ, כִּי הִסִּיג אֶת גְּבוּלוֹ,

כִּי הוּא, נְזִיר דְּמָמָה, כְּבַד־פֶּה, לְמוּד אֶבֶן וְגַרְזֶן, יְנַסֶּה

לְעוֹרֵר גַּם הוּא הֲמִית עִנְבְּלֵי כֶסֶף לְמִלִּים וְהֵדִים,

כְּרֵעוֹ הַחוֹצֵב בִּצְלִילִים…





וְצָרִיךְ לְהוֹדוֹת הָאֱמֶת:

לוּ חָפְצָה הַשְּׁכֵנָה לְהַבְאִישׁ דְּבָרִים כְּנֶגֶד הַמְשׁוֹרֵר —

אוֹ אָז הָיָה מוֹסֵר לְבִתּוֹ לְלַמֵּד עָלָיו סַנֵּגוֹרִיָּה…

כִּי לוֹ מִלְּכַתְּחִלָּה לֹא הָיְתָה אֻמָּנוּתוֹ שֶׁל זֶה מְבֹרֶרֶת

לְמִשְׁעִי, וְדַעְתּוֹ לֹא סָבְלָה בֶן־אָדָם הַמְטַפֵּל כָּל יָמָיו

בְּשִׁירָה גְרֵידָא, כְּפַרְנֵס אִישׁ שָׂדוֹ וּבְהֶמְתּוֹ וַעֲסָקָיו. —

גְּדוֹלָה הַשִּׁירָה!… הַהוּא לֹא יְדָעָהּ?… מִכָּל מְצָרֵי חַיָּיו

הֲלֹא נִהֲלַתּוּ אֶל שְׁבִילֵי פִרְזוֹנָהּ לְשַׁכֵּךְ מְרוֹרוֹת;

אַךְ שָׁר לוֹ הָאָדָם מִמֵּילָא: בִּנְעוּרָיו תִּזְדַּמֵּן לוֹ שִׁירָה

בְּכָל זְמָן וּבְכָל מָקוֹם…הוֹי, הוֹי! עוֹד הוּא זוֹכֵר, אֵיךְ מֻפְשַׁל מִכְנָסָיו

וּמְלֻכְלָךְ עַד בִּרְכָּיו בְּלִשְׁלוּשׁ בְּהֵמוֹת הוּא זִבֵּל שַׁדְמוֹתָיו,

וְקָלְחוּ הַצְּלִילִים מִגְּרוֹנוֹ עֲדִינִים וּמְסֻלְסְלֵי עֶצֶב

גַּעְגּוּעֵי כָל יִחוּל וְעִצְּבוֹן הַחֶמְדָּה וְחֶדְוַת כִּסּוּפִים —

נִתְמַזְּגוּ בְאשׁ זֶבֶל עִם בֶּשֶׂם הַשִּׁירָה וּפִלְּשָׁה הַנֶּפֶשׁ,

לְלֹא דַעַת עַד־מָה, לְרִאשׁוֹתֵי תוֹלָדוֹת לָאָדָם וַאֲדָמָה,

וַתִּסְפֹּג מִמַּחְתַּת מִסְתּוֹרִין לִקּוּטֵי נִחוּמִים וִיגוֹנִים —

בְּעוֹד רֶגַע וְעָרְבָה הַשִּׁירָה בְסִבְלוֹת וְדַאֲגֹות מִשְׁקוֹ,

רַק תָּסַס בְּנַפְשׁוֹ בְּצִנְעָא בְּשׂוֹרַת פְּדוּת מִתְמַהְמֵהָה,

לְתַבֵּל כָּל צַעַר בְּדָמוֹ וּלְשַׁפֵּר כָּל עֶלְבּוֹן הַשְׁלָיוֹת…

כִּי נִתַּן הַזֶּמֶר לָאָדָם כְּרֵיחַ לַפֶּרִי, הַשִּׁירָה

הִיא קְטֹרֶת הָאֵשׁ בְּהֵאָכֵל הַנֶּפֶשׁ עַל עֶשֶׁן גּוֹרָלָהּ.

אַךְ יָפָה הַפְּדוּת לַמִּתְגּוֹשֵׁשׁ, הַתַּבְלִין לְפִתּוֹ שֶׁל עָמֵל,

וְאֵי עֲמַל חַיֵּי הַמְשׁוֹרֵר, מַה־יִּתְרוֹן כִּי זָכָה בַשִּׁירָה?…

לֹא נִטְעַן כָּל סֵבֶל עַל שִׁכְמוֹ, לֹא נִצַּב כְּשׁוֹמֵר אַחֲרָאִי

לְטֶפַח הַחַיִּים הַזָּעִיר, שָׁם יִנְבֹּט בַּפִּנָּה עֲתִידוֹ;

לוֹ נָקֵל לְפָרֵק אֶת אָהֳלוֹ, לֹא אָשְׁרוֹ מִסְתּוֹפֵף בְּזָוִית

דּוֹגֶרֶת וַחֲרֵדָה עַל גִּנְזָהּ: כֹּה רַבִּים מְלוֹנָיו בַּדְּרָכִים —

וְשָׂדוֹ לֹא יִירָא מִבָּרָד; לוֹ יִרְמֹז יְבוּל שַׁדְמוֹת נְדוּדִים…

כֹּה תָמִיד הִתְהַלֵּךְ בַּגַּן וּבַשָּׂדֶה שָׂמֵחַ וְשִׁכּוֹר

מֵעשֶׁר שֶׁטִּפְּחוּ אֲחֵרִים בְּזֵעָה, עִצָּבוֹן וְתִקְוָה,

וְהוּא אֶצְבַּע קְטַנָּה לֹא נָקָף… עוֹד נִזְכָּר: בְּלֵיל בְּהֵאָסֵף

לִפְעָמִים הַגָּדִישׁ הַגּוֹרְנָה, מִדַּאֲגַת גְּשָׁמִים לְמָחָר,

וְהִתְלַקְּחָה הַשִּׂמְחָה בְעֵינֵי הַמְשׁוֹרֵר לְנִצְנוּץ מַלְגֵּזִים

בָּאֹפֶל, לְרִשְׁרוּשׁ אֲלֻמִּים מוּנָפִים בְּדִמְמַת שָׂדוֹת,

כְּאִלּוּ לוֹ הָיָה הָרְכוּשׁ, לוֹ הַשִּׁלֵּם בְּעַד יְמֵי עָמָל,

לוֹ בִטְחַת הַגֹּרֶן הַמְּלֵאָה — וַהֲלֹא רֵיקָם יָשׁוּב לְעִירוֹ

וַחֲסַר־כֹּל כִּלְעֻמַּת שֶׁבָּא, וּמַה־לּוֹ וּלְשִׂמְחַת הָאֹסֶף?…

לֹא, לֹא!… קָשֶׁה הָיָה לָאִכָּר לִקְלֹט נִימוּס חַיִּים מִתְפַּקְּרִים

לְלֹא דַאֲגַת קִנְיָן, אִם רֶגֶב אֲדָמָה אוֹ גָלְמֵי פְּסָלִים,

לְטַפְּחוֹ וּלְסַגְּלוֹ וּלְזַוְּגוֹ אֶל שִׁמְמוֹן יָמִים — — — אַף יָדַע,

בְּזֹאת שָׁוִים הָיוּ הָרֵעִים: גֵּאִים וּצְנוּעִים כְּאֶחָד,

וְעֵינָם לֹא צָרָה בְעשֶׁר זוּלָתָם — בִּשְׁאָר תְּכוּנוֹת נַפְשָׁם

נִבְדָּלִים כֹּה הָיוּ הַשְׁנָיִם…





הַסּוּס לָחַךְ סְפִיחוֹ וַיַּעְצֹר

זְמָן מֻעָט בְּלֶכְתּוֹ, כְּרוֹצֶה לַחֲקֹר פֵּשֶׁר דָּבָר מִבְּעָלָיו;

אַךְ רָפָה כֹה הָיְתָה הַנְּקִישָׁה בִּירֵכוֹ: לֹא בִכְדֵי זֵרוּז,

וְסָפֵק נִעְנוּעַ הָרֶגֶל בְּעָלְמָא, וְהָיָה עוֹד נִמְלָךְ

בְּדַעְתּוֹ זְמָן רַב לוּלֵא רָאָה אֶת פַּאֲתֵי הַדֶּרֶךְ שֶׁשָּׁכְלוּ

אֶת כָּל פְּלֵטַת מִחְיָה, וַתַּעַל בְּאָזְנָיו הֲמִית שִׁקְתוֹת עֶרֶב,

וַיִּצְהַל מִגִּיל וַיִּזְדָּרֵז… נִתְּקוּ רַגְלֵי הָאִישׁ מֵאַרְכּוֹפָיו,

סָמְרוּ גַבּוֹתָיו וְרָטְטָה שֵׂיבַת שְׂפָמוֹ, וְצַעֲרוֹ

הַכָּבוּשׁ שׁוּב טִגֵּן אֶת לִבּוֹ: שְׁנֵיהֶם נִתְאַהֲבוּ בְּבִתּוֹ —

כֹּה רָוְחָה הַשְּׁמוּעָה בַשָּׂדוֹת, וְשׁוּב הוּא לֹא הִבְחִין כִּדְבָעֵי

אַהֲבָה זוֹ מַה מַּשְׁמָעָהּ?… כִּי הָיָה הָאֶחָד, הַמְשׁוֹרֵר,

מִתְרַגֵּשׁ וּמְחַפְּשָׂהּ בְּצֵאתָהּ לַכְּפָר בְּיוֹם שַׁבָּת, וְנָבוֹךְ

הִתְהַלֵּךְ אָז אָנֶה וָאָנָה בַמְּסִלָּה, מְחַכֶּה לְשׁוּבָהּ;

וּבְבוֹאָהּ — וַיַּהֲפֹךְ לְדַרְכּוֹ, בְּטֶרֶם עוֹד תַּסְפִּיק לִרְאוֹתוֹ,

כְּאִלּוּ לֹא עָגַם לָהּ כְּלָל: מַה לּוֹ עִנְיָן בְּצֵאתָהּ וּבְשׁוּבָהּ?…

וְכֹה הָיָה מִסְתַּתֵּר מֵאֲחוֹרֵי עֲרֵמוֹת לִרְאוֹת בָּהּ בְּעָבְדָהּ

עִם בְּנֵי הָאִכָּרִים וְכֻלּוֹ דְּבֵקוּת וְעֶרְגָּה וּדְאָגָה

טְהוֹרָה־עֲדִינָה — וּפִתְאֹם, כְּאִישׁ הַמְמַלֵּט חַיָּתוֹ,

הוּא נָס מֵהַשָּׂדֶה מַר־רוּחַ, בְּשָׁמְעוֹ אֶת צְחוֹקָהּ הַצָּלוּל

עוֹנֶה לְקוֹלוֹת בַּחוּרִים עַלִּיזִים… וְדָבָר לֹא יָכֹל

לְהַצְהִיל שׁוּב פָּנָיו הַנּוֹפְלִים — מִתְמַרְמֵר וְזָעֵף הִתְחַמֵּק

מִשִּׂיחָה, וְנִפְקַד הַמָּקוֹם הָרְבִיעִי בְּשִׁבְתָּם לַשֻּׁלְחָן — — —

אַךְ יֵשׁ אֲשֶׁר פְּגָשָׁהּ בַּגַּן אוֹ בַדֶּרֶךְ, לִפְרָקִים רְחוֹקִים,

בְּנוֹחַ עָלֶיהָ לְרֶגַע הָעֶצֶב הֶעָנֹג־מְזֻכָּךְ,

שֶׁטֻּפַּח בְּעַרְבֵי יַתְמוּתָהּ, אָז נִטְפַּל וְהָלַךְ לְצִדָּהּ,

וְעֶצֶב נִשְׁתַּלֵּב בְּעֶצֶב, עַד שֶׁלֹּא יָכֹל הִתְאַפֵּק

וְהִתְחִיל מִתְחַנֵּן וּמַפְצִיר: לִי צַחֲקִי, נַעֲרָה, אֶת צְחוֹקֵךְ

הַנֶּהְדָּר וּמֵנִיס כָּל צֵל, וָאֱהִי לָךְ כְּהֵלֶךְ שֶׁעָבַר

עַל בֵּיתֵךְ וַיַּעֲטֹף בְּדַרְכּוֹ, כִּי כָפַר בִּמְחוֹזוֹ הַכָּחֹל —

וְהִיא צָחֲקָה לִדְבָרָיו, פָּג עָצְבָּהּ וַעֲלִיצוּת תְּזִיזָה תְקָפַתָּה;

אָז פִּשֵּׂק אֶת כָּל קִמְטֵי נַפְשׁוֹ לִקְרַאת אשֶׁר רַעֲנָן וּמַבְרִיא

שֶׁהֵקִיר לוֹ צְחוֹקָהּ, וַיְהִי כְמוֹ נִגְרַף מֵחֵשֶׁק סְחַרְחָרִים

לִשְׁכַּב לְרַגְלֶיהָ, לִרְווֹת מֶתֶק גִּילָהּ עַד יְצִיאַת נְשָׁמָה,

עַד אַפְסֵי שִׁכָּרוֹן־חִדָּלוֹן… אַךְ פִּתְאֹם הַנַּעֲרָה עָרְקָה

בְּמוֹרַד הַפִּסְגָּה — רַק פִּעְפַּע בַּגַּן רֵיחַ צְחוֹקָהּ, וְהוּא עָמַד

וְהִקְשִׁיב רַב קֶשֶׁב לַהֵד, כְּמוֹ רֻסַּק אֵי־שָׁם פַּעֲמוֹן גָּבִישׁ

בְּדָמוֹ וְעוֹד צְלִילִים מִדַּרְדְּרִים, וּמְטֹרָף מִתְרוֹצֵץ הָעִנְבָּל — —





הֵרִיחָה הַנַּעֲרָה שִׁלְטוֹן: בְּאֵי־אֵיזֶה חֶבְיוֹן סְתָרִים

מִקַּדְמַת יֵשׁוּתָהּ, בְּהִפָּלֵג הַמִּינִים לְתַחֲרוּת וּמִמְשָׁל,

לִבְלְבוּ פִּקְעֵי מְרִי וְאַכְזְרִיּוּת וְרַחֲמִים… בִּן־לַיְלָה

בָּגְרָה לְאַהֲבָה וּשְׂרָרָה, וַתַּעֲמֹד עַל כֹּחָהּ, שֶׁחָשְׂפָה

מִמְּקוֹר רִאשׁוֹתֶיהָ, כֹּה קַדְמוֹן וְעַלִּיז לִקְרַאת אוֹיֵב־גָּבֶר.





וּבְכָל אֲשֶׁר עָקַב אַחֲרֵיהֶם הָאָב, עֻמְּמָה עֵינוֹ מֵהַבְחִין

עַל בֻּרְיוֹ דְּבַר־מָה בְּעַרְבֹּלֶת נַפְתּוּלִים שֶׁסֵּעֲרָה פִתְאֹם

אֶת שַׁלְווֹת בֵּיתוֹ וּשְׂדוֹתָיו… הַאֵין זֹאת כִּי שִׁנְּתָה הָאַהֲבָה

מִשְׁפָּטָהּ מִנְּעוּרָיו עַד הֵנָּה, אוֹ כָהֲתָה נַפְשׁוֹ לְתָפְשָׂהּ?

אַךְ אַחַת הוּא רָאָה וְהִבְחִין: חָל שִׁנּוּי בַּיַּלְדָּה — זֶה עָצְבָּהּ

לֹא רִפְרֵף עוֹד שָׁקוּף וְזַךְ, כִּי נִשְׁתַּקַּע מִתְנַכֵּר בְּעֵינָהּ

וְנֹפֶךְ שֶׁל הֶפְקֵר מַר־רוּחַ הִשְׁתַּחֵץ בְּרֹן עֲלִיצוּתָהּ,

וְלִבּוֹ לֹא נִחֵשׁ דְּאָגָה: נִקְשְׁרָה בִתּוֹ בַמְשׁוֹרֵר…

אַךְ לָמָּה נִתְכַּוְּנָה לְקַנְתְּרוֹ תָמִיד, וּלְמָרֵר תִּקְוָתוֹ,

וְלִצְחוֹק לְרוּחוֹ הַסָּרָה — עַד יֵשׁ אֲשֶׁר נִדְמֶה לִפְעָמִים

כִּי רַק מִתְעוֹלֶלֶת הִיא בוֹ, וּבִידִידוֹ הַפַּסָּל, הַמָּתוּן

וְנוֹחַ לַבְּרִיּוֹת מִמֶּנּוּ, דָּבֵקָה… וְהָיָה כִּי נִקְרָה

הַלָּז לָהּ בַּגַּן אוֹ בַחֹרְשָׁה, וְהוּא יוֹשֵׁב וְהוֹזֶה כְדַרְכּוֹ,

וַתִּקְרַב בַּלָּט לְהַפְתִּיעוֹ, עַד שֶׁנִּתְחַלְחַל וְנִרְתַּע,

מַסְמִיק וּמְסַנֵּן חִיּוּכוֹ הַמְבֻיָּשׁ… אָז יָשְׁבָה לְצִדּוֹ,

וְהָיְתָה גַם הִיא מְחַיֶּכֶת־מִסְתַּכְּלָה בַּמַּעְיָן שֶׁנִּבְקַע

בְּעֵינוֹ, וַיִּפְטֹר אֶת אָשְׁרוֹ לְאַט וּבִצְנִיעוּת וְחָרֵד

מְאֹד עַל כָּל אֵגֶל וָאֵגֶל… אָז יֵשׁ אֲשֶׁר פַּעַם בְּעֶרֶב

יוֹם חָג, כָּבֵד רֶנֶן הָאֵלֶם בְּדָמוֹ וְנִסְתַּחֲפָה עֵינוֹ

מִמַּרְאוֹת הַפְּלָדוֹת לַשְּׁקִיעָה, בִּגְבוֹר בּוֹ גַעְגּוּעָיו לְבַזְבֵּז

אֶת נֵטֶל הָאשֶׁר הַגָּמֵל, אֶת שְׂרָף הַמִּסְתּוֹרִין הַסּוֹטֵף —

וַיֵּדַע לְפָרֵשׁ כָּל רָז בְּשִׂרְטוּטֵי הַחַכְלִיל לֶעָבִים,

לְפַעֲנֵחַ כָּל תָּו וְכָל תָּג בְּדִמְדּוּמִים מְפַלְּטִים דְּמֻיּוֹת

וְחִידוֹת־סְמָלִים; אָז הִשְׂכִּיל לְפַתּוֹת כָּל לֵב וְכָל עַיִן

לַעֲקֹב אַחֲרֵי טִיסַת דִּמְיוֹנוֹ וּלְסַלֵּק מָסַכֵּי אַרְגָּמָן

מֵחֶבְיוֹן מֶרְחַקִּים טְמִירִים, לִקְסֹם בְּשִׁפּוּלֵי אֳפָקִים

בִּיתָנִים וּפִלְאֵי עֲרָבוֹת, לְטַכֵּס בֶּחָלָל הַדּוֹעֵךְ

שְׂדוֹת קְרָב לְהִתְגּוֹשְׁשׁוּת אֵלִים. עַד תִּקְלֹט הָאֹזֶן בַּדְּמָמָה

הֵד־תְּרוּעָה לְתַחֲרוּת עֲנָקִים מִתְגַּלְגֵּל בְּאַפְסֵי עוֹלָמוֹת…

יֵשׁ נִדְמֶה לַזָּקֵן: הָאִישׁ חֲסַר דֵּעָה… אַךְ יֵשׁ שֶׁנִּתְפַּתָּה

לְפֶלֶא הַמַּרְאוֹת שֶׁאָרְבוּ מִתְּכֹל הַמֶּרְחַבְיָה וְדִמָּה:

לֹא בָדָה הָאִישׁ אֶת כָּל אֵלֶּה, וְכָל מַה שֶּׁעוֹרֵר לְעֵינֵיהֶם

הֵן הָיָה הַכֹּל בְּשֶׁכְּבָר אַךְ בְּטִרְדוֹת הַיָּמִים שְׁכָחוּהוּ…

וְעוֹד רָשׁוּם בְּתַחְתִּית זִכְרוֹנָם בִּכְתָב־רֵאשִׁית דֵּהֶה וְקָלוּשׁ

דְּבַר־מָה מֵאַגָּדוֹת וַעֲלִילוֹת, אַךְ חָדְלוּ לְהַבְחִין זֶה כַּמָּה

אִם הָיוּ הַדְּבָרִים מֵעוֹלָם… וְאוּלַי כָל אֵלֶּה הַחַיִּים

הַחוֹלְפִים־קַיָּמִים וְחוֹזְרִים חֲלִילָה בְחֶתֶף הַזְּמַנִּים,

כְּאֶפַע שַׁלְחוּפִיּוֹת, כְּצֵל לְלֹא מַמָּשׁ הֵם כֻּלָּם לְעֻמַּת

הַדָּבָר הַגָּלוּף בְּמַאֲפֵל זִכְרוֹנָם הַתָּשׁ, שֶׁבִּנְגֹעַ

בּוֹ חֶרֶט הָאֳמָן וְחָיָה, וַינַתֵּק בְּשַׁרְבִיט מַלְכוּתוֹ

אֶת רַתּוֹק תַּהְפּוּכוֹת לַצְּלָלִים, שֶׁחֻשַּׁק מִשִּׁבְבֵי סִפּוּקִים,

מִפֵּרוּרֵי בַלָּהוֹת וְתִקְוָה, מִבַּבּוּאוֹת שָׁוְא לְנִצְחוֹנוֹת, — — —





כֹּה קָסְמוּ בִּדְמִי שְׁקִיעוֹת קַיִץ חֶזְיוֹנוֹת חֲצוּבִים לַפַּסָּל;

בַּבֹּקֶר נִסְתַּלְּקוּ לְפִנּוֹת, שְׁקֵטִים כִּבְחַדְרֵי מַשְׂכִּיּוֹת;

קְשֵׁי שַׁיִשׁ וְגֵאֵי הוֹד־מַלְכוּת יָבוּזוּ לְדַאֲגוֹת יָמִים,

אֲסַפְסוּף כֹּה סוֹאֵן וְרָעֵב, הַצּוֹבֵא עַל כָּל צֵל שֶׁל גֹּרֶן…

אַךְ יֵשׁ דְּאָגוֹת חֲצוּפוֹת, שֶׁבְּתוֹךְ כְּדֵי פַחֲזוּת וְטִרְחָה, נִתְקָלוֹת

בְּמִקְשַׁת הָאֵלֶם בַּפִּנּוֹת, בְּתֹקֶף חֶזְיוֹנוֹת לָאָדָם,

וְנִרְתְּעוּ בִילֵל אֵין־אֱיָל וְהִסְתִּירוּ אֶת חֶרְפַּת חִדְלוֹנָן

בִּשְׁאוֹן טְרָדוֹת טְפֵלוֹת, כִּשְׁפָחוֹת שֶׁסָּרְחוּ בְמַעֲמַד הַגְּבִירָה…

כֹּה הָיָה הָאִישׁ וְזֶה שִׂיחוֹ, וְדַעְתּוֹ לַחֲזוֹת וּלְתַעְתֵּעַ,

כְּבַד־פֶּה וּמְלִיצָתוֹ פְגוּמָה, אַךְ יָדַע לַחְצֹב כְּרַב־אַשָּׁף

נִשְׁכָּחוֹת מִמַּחְצַב פְּלָאִים, מִיַּלְדוּת רְחוֹקָה לָאָדָם. — — —





כֹּה הִרְהֵר הַזָּקֵן, וְלֹא עָמְדָה דְמוּת הַקַּיָּטִים הַשְּׁנַיִם

לְעֵינָיו כֹּה בְרוּרָה כָל חָדְשֵׁי הַקַּיִץ הַגָּז, בְּהִתְהַלְּכוֹ

עִמָּהֶם בְּבֵיתוֹ וּשְׂדוֹתָיו, כַּעֲלוֹתָהּ חֲרִיפַת הַדְּיוֹקָן,

וַחֲשׂוּפַת אֲרֶשֶׁת מִתְעַלְּמָה, זֶה עַכְשָׁו בְּשׁוּבוֹ מִלַּוּוֹת

הָרֵעִים הַשָּׁבִים הָעִירָה… אַף הִרְגִּישׁ כִּי אֵלֶּה הִרְהוּרָיו

מְחַסְפְּסִים עֹקֶץ מַחֲשַׁבְתּוֹ, מְגַלִּים קָצֵהוּ הַגָּמוּם,

וּמְחַפִּים־מַעֲרִימִים עַל קָדְקֹד הַדְּאָגָה שֶׁזָּחֲלָה־אָרְבָה

וְטֶרֶם תְּזַנֵּק — אַךְ רָבְצָה כֹה כְבֵדָה עַל מִפְתַּן כָּל יָמָיו.





וּפִתְאֹם הִתְנַעֵר הַסּוּס בְּהַגִּיעוֹ לַגַּבְנוּן הָאַחֲרוֹן,

הִתְעַכֵּב לְרֶגַע לִשְׁאֹף רֵיחַ אֻרְוָה הַנּוֹהֵר בָּאֲוִיר,

כֹּה חַם וְשָׂבֵעַ לִנְחִירָיו, שֶׁנֻּפְּחוּ לִבְלִיל הָאֲבוּסִים,

וַיְמַהֵר כְּחֵץ אֶל הֶחָצֵר… אוֹ אָז מֵאֵי־שָׁם־רָחוֹק גָּחוּ,

מְטַרְטְרִים לְקֶצֶב הַדְּהָרוֹת, דִּבְרֵי הַפְּרִידָה לַמְּשׁוֹרֵר,

וְהָלְמוּ עַל אָזְנֵי הָאִישׁ: "שָׁלוֹם, סָבָא, קַח תּוֹדָה וּבְרָכָה

בְּעַד כָּל רִפְרוּף בֹּסֶר לָאשֶׁר שֶׁרָמַז פֹּה לָנוּ וַיְפַתֶּה

לְפַלְדוֹת בִּכּוּרָיו — וְלֹא קָם וְלֹא גָמֵל… אַךְ אַתָּה שָׂא תוֹדָה,

וּדְרוֹשׁ בִּשְׁלוֹם בִּתְּךָ הַחֲמוּדָה בִשְׁמִי וּבְשֵׁם זֶה הַבַּיְּשָׁן,

שֶׁנִּדְחַף לַקָּרוֹן, וּבְוַדַּאי יְטַמֵּר שָׁם דִּמְעָה עַקְשָׁנִית

וּמְיַחֵד שֵׁם בִּתְּךָ בְרַחַשׁ וּבְטָהֳרָה — הוֹי, הוֹי, שׁוֹטֶה רָגִישׁ!

הֲתַמּוּ נְעָרוֹת בַּכְּרַכִּים לִכְרוּז הַפַּנָּסִים בַּחֹרֶף —

וֶאֱמֹר לָהּ…" וְנָבוֹךְ אָז הִשְׁפִּיל אֶת עֵינָיו וְהִפְסִיק לְרֶגַע —

"וֶאֱמֹר לָהּ… כֹּה יָפָה הַמְּסִלָּה הַתְּלוּלָה בִשְׁקֹעַ הַחַמָּה,

כִּי יַאֲפִיל הַגַּן נִכְחָהּ צְלָלָיו, וְנִדְמִים מִשְׁעוֹלָיו לַהֵלֶךְ

כִּתְלוּיִים בֵּין שְׂבָכוֹת יְרֻקּוֹת, כִּגְשָׁרִים מִתְקַשְּׁתִים בַּסַּהַר;

וּכְקוֹסְמִים נִצָּבִים עַל פִּתְחוֹ שְׁלשֶׁת עַרְמוֹנִים לְלֹא נִיעַ,

חֲבוּשִׁים מִצְנָפוֹת עֲבֻתּוֹת וְעוֹסְקִים בְּרָזֵי אֲפֵלָה…

בַּשָּׂדוֹת שִׁבֹּלֶת לִשְׁכֶנְתָּהּ מְחַוָּה קִדָּתָהּ בְּלַחַשׁ,

וְאִוְשַׁת פִּטְפּוּטִים הִתְהַלְּכָה… אָז יֵשׁ כִּי בִשְׁלִיחוּת דְּחוּפָה

מִגְּבֶרֶת הַַחִטָּה, הַצְּהֻבָּה־חֲמוּקָה, לִבְחִירָהּ הָאָדוֹן

הַדָּגָן הַיָּהִיר־יְרַקְרַק, אַרְנֶבֶת, צִיר מָהִיר וּמְבָהָל,

תְּדַלֵּג בִּקְפִיצָה נֶאֱמָנָה עַל פְּנֵי כָל הַמְּסִלָּה לְרָחְבָּהּ,

וּסְמוּרַת אָזְנַיִם עוֹד תַּפְלִיג בַּדָּגָן מִפַּחֲדֵי חִנָּם…

וְהַרְחֵק בָּאֹפֶק נִיר אַחֲרוֹן טוֹבֵל בְּאַגְּנֵי שַׁלְהָבוֹת —

לֹא נִשְׁכַּח כָּל אֵלֶּה, הַזָּקֵן… וּלְשָׁלוֹם פְּקֹד בִּתְּךָ הַיְקָרָה…"





וְתוֹךְ כְּדֵי מֵרוּץ הַסּוּס הִבְחִין עַכְשָׁו בִּמְבוּכַת הַמְשׁוֹרֵר:

הִתְקַלֵּף לְאַט חִיּוּךְ־זַיְּפָן, מִתְחַפֶּה בְעָקְבַת דִּבּוּרִים

לִטְרֹף מִלַּת צַעַר יְחִידָה, שֶׁחִלְחְלָה צוֹרְבָה־מַחֲנִיקָה

בְדָמוֹ, וּלְבָלְמָהּ… כָּמוֹהוּ, כְּרֵעוֹ הַשַּׁתְקָן, הֶעְלִימוּ

דְבַר־מָה שֶׁלֹּא נָסַע עִמָּהֶם, שֶׁנִּשְׁאַר מְרַחֵף בַּמְּסִלָּה,

בֶּחָצֵר, בִּשְׁבִילֵי שַׁלֶּכֶת, בַּגָּן וְעוֹד יוֹתֵר מֵהֵמָּה,





בְּעֵינֵי הַתּוּגָה לְבִתּוֹ — — — וּטְרוּד מַחְשְׁבוֹתָיו לֹא קָפַץ

בֶּחָצֵר מִסּוּסוֹ, כְּדַרְכּוֹ, וּבְרֶגַע הָאַחֲרוֹן רַק הִסְפִּיק

לְהַרְכִּין אֶת רֹאשׁוֹ בִמְעוּף וְלִגְחֹן עַל פְּנֵי עֹרֶף סוּסוֹ,

שֶׁדִּלֵּג עַל מִפְתַּן הָאֻרְוָה וְשִׁנָּיו מְכַרְסְמוֹת מִתְגּוֹ,

לוּ אֵחֵר עוֹד רֶגַע — וְנֻפְּצָה גֻלְגָּלְתּוֹ אֶל קוֹרַת הַמַּשְׁקוֹף.




ב. אָדָם וְצַפְצָפָה






עַל פֶּתַח הַשַּׁעַר צַפְצָפָה גַלְמוּדָה תְשַׁטַּח עֳפָאִים

זְקוּפִים וּקְלוּשֵׁי הַיֶּרֶק, וּדְבֵקָה בִתְפִלַּת עֲרִירִים

לַחֲלִיפוֹת הַזְּמָן חֶרֶשׁ תַּקְשִׁיב… בְּאָבִיב תְּפַטֵּם נִיחוֹחַ

הַצֳּרִי בַּפְּקָעִים, תְּסַמֵּר אֶת סֻבְּכָהּ הַיָּרֹק־אֲפַרְפַּר,

בַּסְּתָו דֵּהֶה־כָתֹם תַּשִּׁירוֹ, כִּמְעַט לְלֹא חֶבְלֵי שַׁלֶּכֶת:

אֲצִילָה תִּתְבּוֹדֵד פֹּה שָׁנִים אֶל מוּל שַׁדְמוֹת בָּר, לֹא תִּתְרַבְרֵב

בְּשִׁפְעַת קִשּׁוּטִים עִם קַיִץ, וְלֹא תִרְגַּז מִסַּעַר כִּי יִפְרֹץ

בְּיִבְבַת עָרִיצִים לִבְחשׁ אֶת כָּל מַחֲבֵא בַּסְּבָכִים, כִּמְחַפֵּשׂ

מַטְמוֹנוֹ הָאָבוּד; כִּי יַחְפֹּן מְלוֹא כְנָפָיו מִכְּבֻדַּת עַרְמוֹנִים,

וְיִצְנֹף שַׁרְבִיטֵי עֲרָבוֹת וְיִכֹּף צַמַּרְתָּן, וִיפַלֵּשׁ

כָּל חֹרְשָׁה וְכָל כִּפָּה יְרֻקָּה, עַד תָּחוּשׁ הַשֶּׁמֶשׁ וְתִתְרֹם

מִגֶּרֶשׁ בִּכּוּרִים אֶת מִכְסַת הַמַּס לְאָחִיהָ הַשַּׁלִיט. —

אָז סְגוּרָה תִּסְתַּכֵּל בְּתוֹכָהּ וּתְהַרְהֵר אֶת חֶשְׁבּוֹן עוֹלָמָהּ,

וְחִידַת הֲוָיָה תְפַקֵּל לְחַטְחֵט עַד גַּרְעִין הַפִּתְרוֹן:

מַה תַּחְשׁשׁ מִסַּעַר עוֹד עָתָּה?… וְהִיא נִשְׁאֲרָה שְׂרִידָה לִפְלֵיטָה

מִשְּׂדֵרַת צַפְצָפוֹת גְּמִישׁוֹת שֶׁצָּמְחוּ לְעֶבְרֵי הַדֶּרֶךְ

לִפְנֵי הַמִּלְחָמָה הַגְּדוֹלָה… וְלֹא כְמוֹת כָּל הָעֵצִים הֵן מֵתוּ:

לֹא עָשְׁשׁוּ מִזֹּקֶן וְרָקָב, לֹא כִרְסְמָה רִמָּה אֶת שָׁרְשָׁן,

לֹא פִלְּחָן בַּשָּׁרָב הַבָּרָק וְלֹא הוּנַף הַגַּרְזֶן עַל גִּזְעָן —

כִּי נֶעֶרְפָה הָאַחַת מִכַּדּוּר תּוֹתָחִים וְשִׁלְדָּהּ הֶחָרוּךְ

עוֹד בָּלַט שְׁנָתַיִם פֹּה וּמְפֻחָם וְנֶעֶקְרָה הַשְּׁנִיָּה

בְּרַעַשׁ, עַל מֶגְרְפוֹת שָׁרָשֶׁיהָ, וַתְּזַנֵּק בִּגְלוֹמֵי לֶהָבוֹת

אֶל בֵּין הַמִּתְגּוֹשְׁשִׁים; מִמּוּלָהּ שְׁתֵּי רֵעוֹת זְקוּפוֹת וּשְׁקֵטוֹת,

בְּסוּפַת יָם־אֵשׁ וַאֲבַדּוֹן זְרוֹעוֹת־לַפִּידִים שִׁלֵּבוּ…

כַּעֲבֹר הַהַתְקָפָה וַתֵּרֶא וְהִיא יְחִידָה בַּעֲדַת הַצַּפְצָפוֹת,

שֶׁפָּסַח הַמָּוֶת עָלֶיהָ. — מֵאָז בָּדָד תְּבַלֶּה יָמֶיהָ,

כְּשׁוֹמֵר עַל שַׁעַר הֶחָצֵר, וְקוֹדְרָה תְקַדֵּם פְּנֵי בָאָיו,

כְּנָזִיר עַל מִפְתַּן בֵּית יָיִן… אַךְ יֵשׁ מִטִּיּוּלָיו בַּחַגִּים,

כִּי יָשׁוּב בְּעָלֶיהָ הַזָּקֵן מֵעֵבֶר הַכְּפָר וְיִתְעַכֵּב

עָלֶיהָ זְעֵיר כְּבַד זִכְרוֹנוֹת, וּטְרוּד גַּעְגּוּעִים יִתְרַפֵּק

עַל עֵד זְמַנִּים טוֹבִים שֶׁחָלְפוּ — אָז יַקְשִׁיב בְּרוּחַ עֲרָבִים

כְּהֵד אִוְשַׁת חֶמְלָה מִתְאַפְּקָה לוֹ עוֹנֶה מִבֵּין הָעֲנָפִים. —

בַּלַּיְלָה, בַּשַּׁחַר, תְּצַפֶּה לְדַרְכֵי הַסַּהַר הַתְּלוּלִים:

הֲיֵּשׁ אֲשֶׁר יָבוֹא מִשָּׁם גַּם הַפִּתְרוֹן, כְּמוֹ לִפְנֵי שָׁנִים

בְּמַרְכְּבוֹת פַּלְדוֹת־שִׁגָּעוֹן הַחִידָה הֵגִיחָה אֲיֻמָּה. — — —





בֶּאֱמֶת לֹא הֶעֱרִים הַמִּקְרֶה הַפַּעַם, לֹא חָרַשׁ תַּכְסִיסִים

לְאַנּוֹת לִשְׁלָשְׁתָּם עַל דַּרְכָּם לָאשֶׁר אֶת עָצְבָּם הַגָּדוֹל —

תְּמִימִים וּבְטוּחֵי מַזָּלָם, רְעוּלֵי הָעַיִן הַבְּהִירָה,

וּגְלוּיֵי הָאֵמוּן הֵם יָצְאוּ — הַיְנַכֵּל לָהֶם בְּתַחְבּוּלוֹת?…





רַק פָּרַשׂ אֶת רִשְׁתּוֹ בַּשֶּׁמֶשׁ, וַיְהִי לָהּ מֵרָחוֹק, בְּנֹגַהּ

עֲלוּמִים וְקַיִץ, כְּמַרְאֵה קַשְׂקַשֵֹּׂי הַכֶּסֶף, כְּמַעֲשֵׂה

גַּבְלוּת לְאוֹר — וַיִּלָּכְדוּ, וְטֶרֶם יַבְחִינוּ בַמּוֹקֵשׁ,

וְעוֹד הֵם מְפַרְפְּרִים תְּמֵהִים, וְעוֹד צְחוֹקָם מִתְעָרֵב בִּיגוֹנָם,

וְהוּא סָגַר עֲלֵיהֶם הַצֶּמֶת, וַיִּמְסֹר מַפְתֵּחַ הַקְּסָמִים

לִבְעָלָיו הַפִּלְאִי, וַיֵּלְכוּ — וְלֹא זָכְרוּ וְלֹא שָׁתוּ אֶת לִבָּם…





וְכֹה אֵרַע הַדָּבָר, כִּי הִפְלָא הַפַּעַם הַפֶּלִאי לוֹ אוֹתָהּ

צַפְצָפָה לְחֶפְצוֹ: בָּעֶרֶב הַשֵּׁנִי לְבוֹא שְׁנֵי הָרֵעִים,

בְּטֶרֶם כְּלוֹת הָעֲבוֹדָה בַּשָּׂדוֹת, הֵם נִצְּבוּ תַחְתֶּיהָ,

וַיִּרְאוּ מֵרָחוֹק מִפִּסְגַּת הַמְּסִלָּה הַנַּעֲרָה יוֹרְדָה

וְקוֹלְעָה צְרוֹר פִּרְחֵי שָׂדֶה… אַךְ קָרְבָה, נִזְדָּרְזוּ הַשְּׁנַיִם

לְעֻמָּתָה בְּבִרְכַּת הַקּוֹצְרִים, הִרְכִּינָה אֶת רֹאשָׁהּ בְּקִדָּה,

וַתַּסְמֵק בְּצֵל הַדִּמְדּוּמִים, כִּי נִחֲשָׁה מֵרֹאשׁ דְּבַר הַפְּגִישָׁה;

וְעוֹד הִיא מִתְבַּיֶּשֶׁת — וְחִיּוּךְ כֹּה תָזִיז הִתְבַּשֵּׂר בְּעֵינָהּ,

וְעוֹד הִיא טֶרֶם תָּבִין עַד מָה, וַיִּתְפָּעֶם בָּהּ לִבָּהּ לַבְּשׂוֹרָה:

הוֹי נָפַל פֹּה דָבָר עִם בּוֹאָהּ — כְּמוֹ נִפְרְדוּ פִּתְאֹם הָרֵעִים,

וְנִצְּבוּ כֹה זָרִים־מִתְפַּגְּנִים כְּנֶגְדָּהּ, שְׁנֵי גְבָרִים דְּבוּבִים,

הַמְבַטְּלִים אִישׁ אָחִיו מֵעִקָּר, וּנְכוֹנִים לְרֶמֶז אֶצְבָּעָהּ,

שְׁחוּצִים וְעַזִּים, לִקְשׁר קְרָב דָּמִים אֶל כָּל רֹאשׁ דְּרָכִים

לְרֹן אשֶׁר טָמִיר וְעַתִּיק בְּדָמָם… וְעוֹד הִיא טֶרֶם תֵּדַע

כִּי נִתְפְּשָׂה בַּמּוֹקֵשׁ גַּם הִיא, וַתִּצְחַק לִרְאוֹתָם מִתְכַּתְּשִׁים

בַּחֲשַׁאי, וַתִּתְחַמֵּק לֶחָצֵר וְשִׁירָהּ הִסְתַּלְסֵל כְּסוֹחֵט

אַשְׁכֹּלוֹת מִתְבַּשְּׁלִים בִּדְמִי, עַד שֶׁדִּמּוּ הָרֵעִים לְהַאֲזִין

בְּקוֹלָהּ כְּנֹפֶךְ שֶׁל לַעַג־נִצָּחוֹן בְּתַחֲרוּת אִלֶּמֶת,

וּנְזוּפִים הִשְׁלִימוּ בֵינֵיהֶם, אַף סָנַט הַמְשׁוֹרֵר בִּידִידוֹ:

"תֵּן דַּעֲתְּךָ, הַבַּטְלָן, וּלְאוֹתוֹ הַדְּבָשׁ אַל תְּשַׁלְשֵׁל לְשׁוֹנְךָ





לְגֹדֶל תַּאֲבוֹנְךָ הַמְגֻשָּׁם, פֶּן תִּסְלֹד מִבָּסְרוֹ הֶחָרִיף,

וְעוֹד תַּעַל עַל שְׂפַת כָּל הַכְּפָר בִּגְלַל פְּתַיָּה גֵאָה, גַּחְכָּנִית"…

וְשׁוּב לֹא נִתְקָרְרָה דַעְתּוֹ עַד שׁוּבוֹ לְחֶדְרָם, וַיִּמְצָא

עַל שֻׁלְחַן אַלּוֹנִים צְרוֹר פְּרָחִים רַעֲנָן עַל טִלְלֵי עַרְבָּם,

הַמְשַׁקֵּף קְלָחָיו מֵעָצִיץ הַקּוֹרֵן עַל גְּבִישׁוֹ וּמֵימָיו —

אָז מָצָא הַפַּסָּל אֶת יָדָיו גַּם הוּא לְקַפֵּחַ בִּירִיבוֹ:

"רַע לֵב וּמִתְאַנֶּה, וְתוֹלֶה עֲלִילוֹת שֶׁל תִּפְלוּת — שִׂים תּוֹדָה,

כִּי פָחֲזָה שׁוּב רַתְחָנוּתְךָ עָלֶיךָ וַתִּבְסֹר בַּנַּעֲרָה

עַל חִנָּם, וַתְּנַבֵּל אֶת פִּיךָ — הֲלֹא תִתְפֹּשׂ בּוּשָׁה בְּלִבְּךָ…"

וְאָמְנָם מִיָּד קָפַץ פִּיו הַקַּטֵגוֹר, הִתְפּוֹגֵג לְמַרְאֵה

יִרְקֵי גִבְעוֹלִים מִצְטוֹפְפִים, מֵינִיקִים פִּקְעֵיהֶם הַכְּמֵהִים

לְרֵיחַ לֵילָם הָאַחֲרוֹן, שֶׁדָּגַר עַל שָׂדוֹת וְהִצְמִיד

בִּמְרוֹמָיו רְסִיסִים־עוֹלָמוֹת לְרִבְבוֹת רַתּוֹקֵי יַהֲלוֹמִים…





וּבְאוֹתָהּ הַשָּׁעָה מִמָּחֳרָת, בְּצֵל הַצַּפְצָפָה שֶׁטָּבְלָה,

וָתִיקָה־אִלֶּמֶת, צַמַּרְתָּהּ לְדִמְמַת אַשְׁדּוֹת הַשְּׁקִיעָה —



שׁוּב חִכּוּ הָרֵעִים… (וְלֹא נוֹעֲדוּ — פָּשׁוּט: "הֲיֵשׁ לְךָ מִפְלָט



הַנּוֹחַ בִּשְׁעַת עֶרֶב זוֹ מִצַּפְצָפָה עוֹמֶדֶת בְּגַפָּהּ,

נִשְׁקֶפֶת לַשָֹׂדוֹת הַכָּבִים… וּבְוַדַּאי לֹא יַעֲבֹר הָעֶרֶב

פֹּה אָדָם לְהַרְגִּיז שַׁלְוָתָם; כִּי גַם זֹאת הַנַּעֲרָה — אוֹתָהּ

שֶׁל תְּמוֹל, הֲשָׁכָחְתָּ? — אֶל נָכוֹן כְּבָר שָׁבָה וְעוֹשָׂה בָרֶפֶת,

וְלֹא לָהּ נִתְכַּוַּנּוּ — הַאֵין זֹאת?"… וְהֵיטֵב יָדְעוּ הַמְשׁוֹרֵר

וְרֵעוֹ, כִּי בוֹשְׁשָׁה הַנַּעֲרָה הָעֶרֶב לָשׁוּב מֵהַשָֹּׂדֶה,

וְרוּחָם כֹּה קָצְרָה…) וַיְהִי כְבָר הַלַּיְלָה לָבוֹא וַתֵּרָאֶה

יוֹרֶדֶת בַּמְּסִלָּה, וְכַלְבָּהּ הַשָּׂעִיר לְפָנֶיהָ מְשׁוֹטֵט,

וּמְצַדֵּד לְעֶבְרֵי הַשָּׂדוֹת, וְרֶגַע זְעֵיר יִקְפָּא תַחְתָּיו,

מִזְדַּקֵּף וּמְסַמֵּר אָזְנַיִם לְכָל זִיזָה קַלָּה בַקָּמָה,

וְתַאֲוַת טֶרֶף מַהְבִּילָה מִנְּחִירָיו… וְהִיא דַעְתָּהּ נְתוּנָה

לַכֶּלֶב, וְגַם נִפְנוּף עַפְעַף לֹא כִוְּנָה לְעֵבֶר הַשָּׁעַר;

וְלוּלֵא הַכֶּלֶב שֶׁרָטַן בְּנַהֲמָה, סָפֵק נְבִיחָה

לִקְרַאת פָּנִים זָרִים וְסָפֵק יְדִידוּת לְמוֹדָע מִקָּרוֹב,

לֹא הָיְתָה מַרְגִּישָׁה בָּהֶם… הוֹי, שׁוּב זָעָה הִתְמַרְמְרוּת הַמְשׁוֹרֵר

עָלָיו וְשׁוּב הִבְהֵב עֶלְבּוֹנוֹ שֶׁשָּׁכַךְ — אַךְ רֵעוֹ כְּבָר גִּיֵּס

כָּל דִּבְרֵי הַתּוֹדָה שֶׁצִּחְצַח בַּחֲשַׁאי וּבְזֵעַת אַפַּיִם

כָּל אוֹתוֹ הַיּוֹם, לְהַפְתִּיעַ חֲבֵרוֹ, וּלְקַדֵּם הַנַּעֲרָה

הָרִאשׁוֹן בִּבְרָכָה, וַיִּפְתַּח: "בִּשְׁמִי וּבְשֵׁם רֵעִי קְחִי תוֹדָה,

הָעַלְמָה הַכְּבוּדָה, בְּעַד צְרוֹר פְּרָחַיִךְ, שֶׁרֵיחָם רַק יִדֹּף,

אַךְ יַעֲמֹד זִכְרָם…" נִתְבַּלְבֵּל וְלֹא סִיֵּם, וְעוֹד הָיְתָה רַבָּה

הַמְּלִיצָה בְפִיו, אַךְ אָז נִתְקְלוּ עֵינָיו בְּרֵעוֹ שֶׁהִשְׁחִיל

מַבָּטוֹ לַמֶּרְחָק הַמַּעֲרִיב לְלַקֵּשׁ נְגֹהוֹת אַחֲרוֹנִים,

וּשְׂפָתָיו כְּאִלּוּ מְצַיְּצוֹת, וְהוּא הִסְבִּיר לְעַצְמוֹ צִיּוּצָן:

הָהּ, רַמָּאי, כְּבַד פֶּה וּמְעוּט שֵׂכֶל, הָהּ, גַּמְלָן הַבָּא בְמַחְתֶּרֶת!

מִי שָׂמְךָ לְמֵלִיץ? מְנֻוָּל וְטִפֵּשׁ, הֲתִסְכֹּר אֶת פִּיךָ?…

וְלַנַּעֲרָה הִבִּיט, וְאוֹתוֹ הַחִיּוּךְ שֶׁל תְּמוֹל שׁוּב נִזְדַּקֵּר בְּעֵינָהּ:

הִיא בָאָה לִרְאוֹת בַּמִּתְגּוֹשְׁשִׁים וַיֶּעֱרַב הַמַּחֲזֶה לְנַפְשָׁהּ…





וְהוּא הִצְדִּיק עַל עַצְמוֹ הַדִּין — עַתָּה הֵבִין כִּי מָעַל בְּרֵעוֹ,

שֶׁנִּלְחַם גֵּא־שְׁתִיקָה עַל כְּבוֹדָם, כְּבוֹד גֶּבֶר מְזֻלְזָל, וַיִּרְהַב

בַּנַּעֲרָה הַזְחוּחָה, וְהוּא נָטַשׁ מִלְחָמָה בְּחֻמָּהּ וְנִבְהַל

לְשַׁלֵּם מַס־תְּבוּסָה… וְעוֹד הִיא מִתְיַהֶרֶת, מְסָרְבָה לְקַבְּלוֹ,

מִתְכַּוְּנָה לְמַצּוֹת כִּשְׁלוֹנוֹ, כִּי יָדְעָה בְּשֶׁל מִי כָּל הַחִרְחוּר

וְשׁוּב לֹא הִתְאַפְּקָה: כִּתְמוֹל שִׁסַּע צְחוֹקָהּ אֶת דִּמְמַת הַשָּׂדוֹת

כְּפַלַּח מַחֲרֵשָׁה לְטוּשָׁה אֶת אַדְמַת הַבּוּר רְגָבִים רְגָבִים,

הִתְפּוֹרֵר בָּאֹפֶל לִכְרוּזִים, בְּנוֹת־הֵדִים — וְנִתַּק… כִּי פִתְאֹם

הִרְגִּישָׁה: עוֹד אֶחָד פֹּה נִשְׁעַן לָעֵץ — כַּשַּׁף לַיִל קְשֵׁה מְזִמּוֹת,

הַכּוֹבֵשׁ אֶת רִשְׁפֵּי חֲרוֹנוֹ: וְאִם תּוּכַל עַיִן לְהָרַע —

אֵין טוֹב לָהּ מִפַּיְּסוֹ וּלְהָסֵב אֶת קְסָמָיו; כִּי עַתָּה לָהּ נִרְאֶה

הָאִישׁ בַּדִּמְדּוּמִים הָאוֹרְבִים כֹּה פֶלִאי, וַתִּירָא מִפָּנָיו…

הִתְכּוֹפְפָה וַתְּגַפֵּף אֶת כַּלְבָּהּ כִּמְבַקְּשָׁה בוֹ עֵזֶר, וְהוּא פָזַל

לְעֵבֶר הָרֵעִים וַיָּלֹק בִּיבָבָה חֲרִישִׁית יָדֶיהָ;

נִרְגַּעַת הִיא סִלְסְלָה שְׂעָרוֹ לְאַט וַתִּתְאוֹנֵן חֲנִינוֹת:

הָהּ, נִירוֹן, מַה חֵטְא חָטְאָה גְבִרְתֵּךְ, כִּי תְמוֹל בְּהֵחָפְזָהּ לָרֶפֶת,

לַחֲלִיבַת הָעֶרֶב, וַתַּעֲזֹב עֲצִיץ פְּרָחֶיהָ בְּמִקְרֶה

בַּחֲדַר הָאֲנָשִׁים הָאֵלֶּה?… כִּי נִדְבְּרוּ שְׁנֵיהֶם הָעֶרֶב,

וַיִּגְדְּרוּ דַרְכָּהּ חָצֵרָה, וְתוֹאֲנוֹת יְבַקְּשׁוּ עַל חִנָּם —

וְנִזְקְפָה פִתְאֹם נִזְדַּקֵּר הַכֶּלֶב אַף הוּא: אַל יְדַמּוּ

לְצַחֵק בְּכָפְרִית תְּמִימָה — הֵטִיחָה לְעֵבֶר הַמְשׁוֹרֵר

שֶׁכְּאִלּוּ אֵי־שָׁם הִפְלִיג הַרְחֵק — הִיא תֵדַע לְכַלְכֵּל עִנְיָנָהּ,

וְגֵאוּת בֶּן־כְּרָךְ לֹא יְאַיֵּם עָלֶיהָ… יִשְׂטוּ מִמֶּנָּה:

הִיא תוֹדָה לֹא בִקְּשָׁה מֵהֶם, לֹא נִתְכַּוְּנָה לְנַשְּׂאָם בִּתְשׁוּרָה,

וְלוּלֵא שֶׁשָּׁכְחָה אֶתְמוֹל, הָיְתָה לוֹקַחַת הַפְּרָחִים

בַּחֲזָרָה, וְאוּלַי גַּם הָיְתָה מְמוֹלְלָה טַרְפֵּיהֶם לְאַבְקָה…

מַתָּנָה?… לֹא, לֹא! דָּבָר זֶה כְלָל לֹא עָלָה עַל דַּעְתָּהּ וּלמַעֲנָהּ

יָכְלוּ הַפְּרָחִים לַעֲמֹד בַּאֲשֶׁר יְהִי — הַחֲסַר מָקוֹם בֵּיתָם?…

וְהַאֻמְנָם יִתְפָּאֲרוּ בְלִבָּם כִּי זֹאת לָהּ הַפַּעַם הַיְחִידָה,

לְרֶגֶל הִתְאַכְסְנָם אֶצְלָהּ, לְהָבִיא הַבַּיְתָה צְרוֹר פְּרָחִים…

הֵן בָּרוּר הַדָּבָר!… וְעַל חִנָּם יִתְקַשֶּׁה הָאָדוֹן הַפַּסָּל

בְּתוֹדָתוֹ, לַשָּׁוְא נִטְרַד רֵעוֹ לְכָאן: טָעוּת הָיְתָה בְיָדָם…





וּפִתְאֹם נִתְחַלְּפוּ הַמַּחֲנוֹת וְשִׁנְּתָה פָנֶיהָ הַחֲזִית:

נִתְפַּשְּׁרוּ הַגְּבָרִים בְּלִי אֹמֶר — הָהּ, לֹא עַל נְקַלָּה תְמֻגַּר

מְצוּדַת מַשְׂטֵמַת הַמִּינִים, וְאִם גַּם הִיא לֹא פִסְּגָה מִגְדָּלִים

בָּרוֹם, כְּהָאַהֲבָה אֲחוֹתָהּ — — — הִרְגִּישָׁה הַנַעֲרָה כִּי עַתָּה

לֹא תוּכַל הִסְתַּכֵּל מֵרָחוֹק בְּהִתְחָרוֹת הַגְּבָרִים לִכְבוֹדָהּ,

הֵרִיחָה כָל תֹּקֶף הִסְתָּעֲרוּת מִצִּדָּם וְלֹא נָפַל לְבָבָהּ,

הֲמִית דָּמָה שָׁקְקָה, הָלְמָה: הַקְּרָב יִהְיֶה עַלִּיז וְאַכְזָר,

וְרִנְּנוּ לוֹ בְשָׂרָהּ וְרוּחָהּ, שֶׁנִּבְדְּקוּ עַל יְדֵי מַצְבִּיא פֶלִאי

הַלַּיְלָה, בֶּעֱזוּז עֲלוּמֶיהָ — וְנִמְצְאוּ זַכִּים וְעַזִּים…

לְאַט הִפְנָה רֹאשׁוֹ הַמְשׁוֹרֵר, וַיַּרְכֵּן אֶת קוֹלוֹ, וְקָשֶׁה

לָהּ הָיָה לְהַבְחִין בִּדְבָרָיו הַמְתוּנִים בֵּין לַעַג לִרְצִינוּת,

וְכֹה עָנָה: "אֶל חִנָּם נִטְרַדְתִּי?… אַךְ לָצוֹן חָמַדְתְּ לָךְ, הָעַלְמָה,

וְזֶה חֶפְצֵךְ לְהָתֵל, כִּי תֹאמְרִי לָדַעַת אֶת כָּל אֲשֶׁר אֶלְאַט

בְּחֶדְרִי עִם רֵעִי, — הֲלֹא כֹה תִתְיַמְּרִי לָדַעַת דְּבַר עָמְדִי

בְּצֵל הַצַּפְצָפָה, לִגְדּוֹר לָךְ אֶת דַּרְכֵּךְ וּלְהַפִּיל תּוֹדָתִי

בְּעַד צְרוֹר פִּרְחֵי שָׂדֶה, שֶׁנָּפַל בְּמִקְרֶה, בִּשְׁגָגָה, אוֹ יִהְיֶה

בְּאֵיזֶה דֶּרֶךְ שֶׁיִּהְיֶה — סוֹף דָּבָר: שֶׁנָּפַל מֵעַצְמוֹ

אֶל חֶדְרִי בֵּין סְפָרַי לְקַשְּׁטָם, וְרֵיחָם לֹא עָמַד עַד בֹּקֶר,

וְנָשְׁרוּ טַרְפֵּיהֶם בִּן־לַיְלָה וְטִאטְאָם רֵעִי הַחוּצָה".





הוֹי, הוֹי! עַתָּה גָדַשׁ הַסְּאָה בְפַחֲזוֹ, עוֹד רֶגַע וְטָפַח

רֵעֵהוּ כְזָבוֹ עַל פָּנָיו — אַךְ זֶה טֶרֶם יַסְפִּיק לְאַזֵּן

כָּל עָקְבַת הַמְשׁוֹרֵר וּלְהַחְלִיט מִי הֶעְדִּיף זָדוֹן בַּיְרִיבִים —

וְקָפְצָה הַנַּעֲרָה בִרְתֵת, וְנִתְלַקַּח הַבּוּז בַּאֲפֵלַת

עֵינֶיהָ הַלְטוּשׁוֹת: "הַאֻמְנָם, אֲדוֹנַי, רַק זֹאת וְלֹא יוֹתֵר?…

רַק רֵיחָם שֶׁפָּג וְטַרְפֵּיהֶם שֶׁנָּשְׁרוּ — וְלֹא עֻפְּשׁוּ הַמַּיִם,

וַעֲצִיצִי גַּם הוּא, עָצִיץ פָּשׁוּט לְפִרְחֵי שָׂדֶה צְנוּעִים —

מְחֻצָּף כָּזֶה! — לֹא נִתְרַפֵּס מִקִּנְאָה וְלֹא לָקַח בּוּשָׁה

בְּחֶבְרַת כָּל סִפְרֵי הַחָכְמָה… כִּי כָכָה חַנּוֹתָ אֲמָתְךָ,

אֲדוֹנִי…"





הָהּ, כָּפְרִית כָּזוֹ… וְלֹא דַי שֶׁלֹּא נִרְתְּעָה שְׁפוּפָה

וּנְבוֹכָה מִפְּנֵי רַהַב גֶּבֶר וְיִתְרוֹן בֶּן כְּרַךְ — עוֹד הֵעֵזָּה

הִתְקַלֵּס בָּהֶם יוֹתֵר: "אָכֵן נִסְכַּלְתִּי לַחְשֹׁב, כִּי יִתְיַצְּבוּ

אֲנָשִׁים בְּפֶתַח הַשַּׁעַר בָּעֶרֶב, בִּשְׁבֹת הָעֲבוֹדָה,

וְחִכּוּ לְמִי שֶׁהוּא שָׁם, אוֹ יֵשׁ וְהֵם קָשְׁרוּ תוֹדָה

בַּלֵּב לְפַלְמוֹנִי בְעַד מַתַּת־חִבָּה, וּתְהִי גַם כֹּה מְעוּטָה

וְקַלַּת הָעֵרֶךְ כְּצֶדֶף מְמֹרָק, כַּעֲנַף לִילָךְ מַכְחִיל,



כְּסֵפֶל בִּכּוּרֵי תוּת־יַעַר — וְסִלְחוּ, אֲדוֹנַי הַטּוֹבִים:



בֵּין יוֹשְׁבֵי הַכְּפָר נִתְחַנַּכְתִּי, פְּחוּתֵי הַשְׂכָּלָה וְנִימוּס,

וָאֶהְבַּל לְסַגֵּל בַּעֲרוּתָם וְטִיבָם לְאַנְשֵׁי הַמַּעֲלָה;

אֶצְלֵנוּ, לְמָשָׁל, מְגַנִּים אֶת מוּגֵי הַלֵּב — אַךְ הֲנָעֵז

לָדוּן אִישׁ הַתַּרְבּוּת, שֶׁמָּנַע מֵאֵי־אֵיזוֹ סִבָּה שֶׁתִּהְיֶה —

מִדְּאָגָה לִפְחִיתַת כְּבוֹדוֹ, אוֹ קִנֵּא בָזֶה לִכְבוֹד גְּבָרִים —

תּוֹדָתוֹ בְעַד צְרוֹר פְּרָחִים, וְהֵם רַק פִּרְחֵי שָׂדֶה עֲנָוִים,

מִנַּעֲרָה פְּשׁוּטָה, מְעַט גֵּאָה, סַרְבָּנִית, וַיְזַלְזֵל בַּמַּתָּת,

בְּטַאטְאוֹ אוֹתוֹ הַחוּצָה — הֲנוּכַל גַּם לוֹ לִקְרוֹא פַחְדָּן?…"

עַתָּה לֹא הִסֵּס הַפַּסָּל: פַּחְדָּנוּת!… יְהִירָה שֶׁכְּמוֹתָהּ

מִתְלוֹצְצָה בָמוֹ, וְלִבּוֹ לֹא יַעֲמֹד לַעֲזוֹר לְרֵעֵהוּ?…

אַךְ רְאֵה! נָטַשׁ רֵעוֹ כָל טִינָא וּמְרִירוּת, כְּאִלּוּ לֹא אֵלָיו

נִתְכַּוְּנָה הַנַּעֲרָה, וַיְשַׁקֵּל בִּדְבָרָיו: "וַהֲיֵשׁ, עַלְמָה כְבוּדָה,

אֲשֶׁר אִם יַנִּיחַ אִישׁ דָּבָר לְתֻמּוֹ וּמְצָאוֹ הַשֵּׁנִי

בְמִקְרֶה, וְהוּא יָבוֹא לְהוֹדוֹת לַחֲבֵרוֹ בְעַד הַמַּתָּנָה,

וְלֹא יִהְיֶה לְלַעַג בַּכְּפָר אוֹ בָעִיר?… וַהֲלֹא אַתְּ הַפְּרָחִים,



בְּחָפְזֵךְ לָרֶפֶת, בְּמִקְרֶה הִנַּחְתְּ1 עַל שֻׁלְחָנִי, וְאַחַת



לָךְ הָיְתָה עֵין מִי הֵם יְשַׂמְּחוּ, יַד מִי תַשְׁקֵם מָחָר עִם בֹּקֶר —

הֲלֹא זֶהוּ דְבָרֵךְ — וְלָמָה תְבַדִּי עָלֵינוּ עֲלִילוֹת,

וּמְרוֹרוֹת תְּרַגְּלִי בִגְבָרִים, שֶׁמּוּגֵי הַלֵּב הֵם מִסְתַּפְּנִים

לִכְפּוֹת לָךְ תּוֹדָתָם מֵחֲשָׁשֵׁי שָׁוְא?… אוֹ אֵין זֹאת כִּי בִשְׁרִירוּת

לְבָבֵךְ עִמְעַמְתְּ עַל הָאֱמֶת וְלָנוּ זִמַּנְתְּ צְרוֹר פְּרָחָיִךְ?"





רַק עַכְשָׁו נִסְתַּבְּרָה לַפַּסָּל כָּל עָרְמַת הַתִּגְרָה, וַיַּעֲרֵץ

אֶת רֵעוֹ, שֶׁיָּדַע לְאַחֵד כֹּה יָפֶה גְּאוֹן גֶּבֶר וְתָקְפּוֹ

עִם סְלִיחָה וּמִדַּת סַבְלָנוּת לַחַלָּשׁ… הוּא רָאָה כִּי נִקְטַם

כָּל חֵשֶׁק נַפְתּוּלִים לַנַּעֲרָה פִתְאֹם, פָּג מִרְיָהּ וְשַׁחֲצָהּ,

וְלוּ נִמְשַׁךְ הָרִיב עוֹד זְמַן מֻעָט בֵּינֵיהֶם, וְלֹא הָיְתָה תִפְאַרְתָּם

בְּנִצְחוֹן שְׁנֵי גְבָרִים עַל נַעֲרָה רָפָה מַתְרִיסָה כְנֶגְדָם…

וְעוֹד פַּעַם הִיא טִכְּסָה אֶת זֵינָהּ, אַךְ נִגְמַם כְּבָר עֻקְצוֹ הַשָּׁנוּן:

"וְאַתָּה, אֲדוֹנִי, הַאִם לֹא שִׁקַּרְתָּ?… וַאֲנִי הֵן יָדַעְתִּי,

כִּי לִי פֹּה חִכִּיתָ, וּלְהוֹדוֹת לִי בָאתָ, וַתִּתְפֹּשׂ אֶת פִּיךָ

בָּרֶגַע הָאַחֲרוֹן, כִּי חָרָה לְךָ דְּבַר רִדְתִּי לְאִטִּי

מִפִּסְגַּת הַמְּסִלָּה, וְעֵינִי לֹא רִפְרְפָה אֲלֵיכֶם, וְדַעְתִּי

טְרוּדָה בַכֶּלֶב, כְּמוֹ לֹא יָדַעְתִּי כְבָר מֵאָז בֹּקֶר

כִּי פֹה לִי תְחַכּוּ עִם עֶרֶב — הַאִם לֹא כִחַשְׁתָּ גַּם אָתָּה?…"





הָרִיב דָּעַךְ לְאַט… טָפְפָה שַׁלְוַת הַלַּיְלָה בְּפַעֲמֵי חֶסֶד,

לְלֹא נִיעַ, עַל רֻכְסֵי צַמָּרוֹת וַתְּרַדֵּד מְשִׁי קִמְטֵי שָׁבְלָהּ

עַל חֶלְקַת אֲגַמִּים נִרְדָּמִים; רַחִימָה־עֲנֻגָּה הִיא לָטְפָה

בִּכּוּרֵי מְלִילוֹת מִסְתַּבְּלוֹת, וַתּוֹשֶׁט שָׁדֶיהָ הַכְּחֻלִּים

לְשִׂפְתֵי אֲדָמָה מְרֻפָּטוֹת, וַתְּשַׁכֵּךְ אֶל חָזָהּ הַמְטֻנָּן

כָּל עִצְּבוֹן הֲוָיָה לְחוֹלֵל מִקַּדְמוּת קַדְמָתָהּ, לְלֹא מִפְלָט,

בִּדְוַי לֵדַת נֶצַח כָּל פִּטְרָה, לְחַסְפֵּס בְּשָׁבָץ כָּל גָּבִישׁ,

לִדְגֹּר רַבַּת דְּאָגָה כָּל גֹּלֶם וּלְפַקְּלוֹ לְפַרְפָּר בֶּן־יוֹמוֹ;

לְכַמֵּר בְּכָל תַּלְמֵי חַיִּים סוֹד חֶבְלֵי בְתוּלֵי־נְצָחִים,

וּלְהַבְכִּיר־לְהַפְקִיר יְבוּלָהּ לְגֹרֶן טְמִירָה, הַפְּרוּזָה

לְרִבְבוֹת לוֹעִים פְּעוּרִים וּכְפוּנִים, הַמְשַׁכְּלִים גְּדִישָׁהּ

לְלֹא שׂבַע — רַק תָּמִיד עוֹד נִפְלַט הַגַּרְעִין הַלָּקִישׁ, הַפֵּלְאי,

הַמְעַבֵּר אֶת רַחְמָהּ בְּסֵבֶל הָרִבְיָה עַד אַפְסֵי הַדּוֹרוֹת…

וַתַּהַר הַדְּמָמָה הָרַבָּה כָּל אֶנְקַת הַיְקוּם בְּהִתְלַבְּטוֹ,

עֲיֵף עֶרְגַּת מָחוֹז וְתַכְלִית, וְתִכְלָה לִפְדוּת וּלְנִחוּמִים;

וַתֵּדַע כִּי טֶרֶם יַגִּיעַ הַמּוֹעֵד, וְעוֹד חִשְׁקוֹ לֹא רִוָּה

הַ“יְהִי” מִבְּרֵאשִׁית הַכַּבִּיר, וּלְהֵדוֹ תְּבַקֵּשׁ אֲדָמָה,

כְּפַרְפָּר מִסָּבִיב לַמְּדוּרָה, בַּחֲלַל הַמַּזָּלוֹת זִוּוּגָהּ…





הָרִיב שָׁכַךְ לְאַט… חֶרֶשׁ הוֹסִיף הַמְשׁוֹרֵר לְנַפְשׁוֹ: וְאוּלַי

מִמֶּנִּי צָדַקְתְּ — בָּאתִי הֵנָה כִּי גָמְלָה הַתּוֹדָה מֵאֶתְמוֹל

כְּגֶרֶשׁ־סְמָדַר בְּנַפְשִׁי, וּפִתְאֹם כֹּה כָבֵד לִי מָהֳלָהּ

(מְשׁוֹרֵר אָנֹכִי וְשׁוֹקֵד לְגַמֵּר כָּל בָּסְרִי ­— וְרֵעַ

עֲרִירִי כָמֹנִי אֲבַקֵּשׁ לְחַלֵּק עִמּוֹ טַנְאֵי אָסְפִּי),

וָאֵצֵא הֲלוֹם לְהַשִּׁירָהּ בְּצִלְלֵי עֵץ גַּלְמוּד עִם עֶרֶב:

הֲיֵשׁ רֵעַ יָפֶה מִמֶּנּוּ, כִּי תְפֻתֶּה עִם שְׁקִיעָה לַבְּדִידוּת,

כִּי יַטִּיף פַּךְ אשֶׁר לַדָּם אֶגְלֵי וְרָדִים, וּמְפַלְּחָה אֶת לִבְּךָ

בְּעֹז נַהֲמַת תּוֹדָה־לֹא־תוֹדָה לְחֵיק תֵּבֵל נוּגָה וְהָרָה

וַחֲסֻנָּה־חֲסֻנָּה… וְהָיָה כִי תַעֲצֹם עֵינֶיךָ וְתַבִּיט

אֶל תּוֹכְךָ־תּוֹכָה, בְּכֶפֶל הַפֶּלֶא, וְתִבְלֹם כָּל אֹמֶר:

אַל דַּבֵּר, אַל הוֹדוֹת, פֶּן יֶחְסַר הַפַּךְ… רְאֵה, מִתְעַטְּפָה צַפְצָפָה

בְּטַלִּית שֶׁל תְּכֵלֶת וְחִידָה, וּמְבַדְּרָה צִיצִיּוֹת־צְלָלִים,

שֶׁגָּחוּ מִכָּל הַמַּחֲבוֹאִים, וְלוֹיוֹת מִצְנַפְתָּהּ אַרְגָּמָן —

שְׁעַת מִשְׁנֵה־שְׁתִיקָה גַּם לָהּ… הַקְשֵׁב: הֵקִיר אֶת גַּלָּיו הַמַּעְיָן

חֲפָרוּהוּ גַלְמוּדִים מֵעוֹלָם וְכָל הֵלֶךְ מִלֵּא אֶת כַּדּוֹ,

וַיָּנַח בְּצֵל הַצַּפְצָפָה הַמְנַסְּחָה לוֹ אֵלֶם הַוִּתּוּר

עַל כָּל מָחוֹז קוֹסֵם בְּכִלְיוֹן נְדוּדִים — — — כִּי עַתָּה לֹא אֵדַע,

הַאֻמְנָם חִכִּיתִי לָךְ, נַעֲרָה, אוֹ קָרְאָה מִתְּמוֹל הַצַּפְצָפָה

וָאָשׁוּב אֵלֶיהָ לְהִתְבּוֹדֵד עִם לַחֲשֵׁי שְׁקִיעָה.





הַנַּעֲרָה

הִשְׁפִּילָה עֵינֶיהָ בְּדַבְּרוֹ, וְדִבְלְלָה יָדָהּ וְלָטְפָה

בְּגַב כַּלְבָּהּ הַמְסֻלְסָל וְחָטְמוֹ הַקָּרִיר וְלַח — פִּתְאֹם זָקְפָה

הָרֹאשׁ וְעוֹד פְּלֵטַת הַמְּרִי בְּעֵינֶיהָ, אַךְ תְּלוּיָה בוֹגֵדָה

הַדִּמְעָה בְעַנְוַת רִיסֶיהָ: הֲיוּכַל הֱיוֹת אֲשֶׁר הֵלֶךְ

בֶּן־עִיר וְאוֹרֵחַ בַּכְּפָר, שֶׁתְּמוֹל בּוֹאוֹ, כְּבָר יָדַע לְהַפְלוֹת

מִכָּל עֲצֵי גַן לוֹ צַפְצָפָה גַלְמוּדָה, לְאָהֳבָהּ כָמוֹנִי

וּכְאָבִי — וְהוּא קוֹרוֹתֶיהָ לֹא יֵדָע, לֹא רָאָה כָמוֹנִי

עֲדַת הַצַּפְצָפוֹת בַּהֲדָרָן, כִּי שִׂדְּרוּ הַמְּסִלָּה לֶחָצֵר,

וְלֹא הָיָה הַשַּׁעַר אָז פָּרוּז לַשְּׁדֵמוֹת, לְלֹא סֵמֶל מִקְלָט:

כְּאוּלָם עֲמוּם הָעַמּוּדִים, עֲבֻתָּה הֶאֱפִילָה הַשְּׂדֵרָה

עָלָיו, וּבַעֲבוֹר הַמֶּרְכָּבָה בְטִיסָה בַּמְּסִלָּה וְדָצוּ

כְבִיעָף צַפְצָפוֹת לִקְרָאתָהּ, כְּצִירִים דְּחוּפֵי בְשׂוֹרָה;

וְרִשְׁרְשׁוּ פַרְסוֹת הַסּוּסִים בַּמַּרְבַד הַיָּרֹק־אֲדַמְדָּם,

שֶׁפִּעְפַּע בְּכָל יְמוֹת שָׁנָה נִיחֹוחַ שַׁלֶּכֶת לַח־חָרִיף,

וְטוֹב הָיָה וְיָפֶה לְטַיֵּל בַּשְּׂדֵרָה… וְהִיא עוֹד פְּעוּטָה,

וְשֶׁקֶט וּבִטְחָה בָאוּהָ לִרְאוֹת הַצַּפְצָּפוֹת נִצָּבוֹת

עָלֶיהָ כִּבְנוֹת־הָעֲנָק — — — וְהִיא סִפְּרָה אֵיךְ חַפִּים מֵאַשְׁמָה

הֵן נָטְלוּ גְזַר־דִּינָן, לְלֹא טַעַם וּמִשְׁפָּט, בִּזְדוֹן הָעִרְבּוּבְיָה

לְקִנְאַת הָאָדָם וּלְרִשְׁעוּת שֶׁשָּׁכְרָה מִתַּרְעֵלַת אֲבַדּוֹן…

הָהּ, הִיא נַעֲרָה פְּשׁוּטָה שֶׁטָּעֲמָה זְעֵיר רַק מִסּוֹדוֹת הַמַּדָּע,

וַאֲשֶׁר עִם לְבָבָהּ תְּדַבֵּר, אַף תֵּדַע כִּי נִבְצַר מִמֶּנָּה

לְתַבֵּל כָּל גֹּעַל לַקֶּטֶל בְּחֵקֶר כָּל סִבָּה וְתַכְלִית —

הִיא יוֹדְעָה: פֹּה עָמְדוּ צַפְצָפוֹת אֲצִילוֹת וְצָפְנוּ בְסֻבְּכָן

אַגָּדָה שֶׁל דּוֹרוֹת, וְתִקְוָה נוֹשָׁנָה, וְרִפְרוּף שֶׁל פִּתְרוֹן

לְסִדְרֵי כָל עֶדְנָה וּכְמִישָׁה, לַחֲלִיפוֹת שֶׁל צַעַר וְחֶדְוָה,

לְמַחֲזוֹרֵי חַיִּים וָמָוֶת — וְעַתָּה נִסְתַּלְּקוּ לָנֶצַח — — —

"וְיֵשׁ נִתְקְלָה רַגְלוֹ שֶׁל אַבָּא בְּגִיד שֹׁרֶשׁ־שָׂרִיד שֶׁעִקֵּל

אֶת גַּבּוֹ הַמְאֻבָּן לְמַעְלָה, כִּמְגַשֵּׁשׁ לְעֶבְרֵי הַשְּׂדֵרָה —

וַיְיַבְּלוֹ עַל רְקָבוֹ וּזְרָקוֹ, וַיַּעֲמֹד בִּדְמִי עַל הֶחָרִיץ,

וְנַפְשׁוֹ הָעוֹגְמָה תֵרָגֵן כִּי טֻשְׁטְשׁוּ גְבוּלוֹת יַצִּיבִים,

וְחָנְטָה שֵׂיבָתוֹ בַלָּט גַּרְעִין־חֶשֶׁד לִשְׂאוֹן הַנִּצְחוֹנוֹת,

הָעוֹלֶה מִתְרַבְרֵב בְּלַהֲטֵי כִבּוּשִׁים חֲדָשִׁים לַבְּקָרִים;

לְעֹדֶף שְׁלַל כֹּחֵי־כֹּחוֹת לָאָדָם, כְּמוֹ חֻדְּשׁוּ לוֹ אוֹנֵי

נְעוּרָיו, אַדְמָתוֹ וְשָׁמָיו — כֹּה יַחֲשׁשׁ אַבָּא מִתַּרְמִית,

מִזִּיּוּף מִתְגַּוֵּן, מֵרֹקֶן אֲסָמִים…" פִּתְאֹם הִסְמִיקָה

וַתְּהִי כְמִתְנַצֶּלֶת: "אַל תִּצְחֲקוּ לִי, וַאֲנִי כִמְעַט בּוּרָה

מִדַּעַת, וְנוֹחַ לְנַפְשִׁי לִקְלוֹט לֶקַח אָב מִלְּהִתָּפֵשׂ

לְפִלְפּוּל כָּל תּוֹרוֹת עֲמֻקּוֹת שֶׁחִדְּדוּ אָזְנִי בְחָרְפֵי

שָׁהוּתִי בִכְּרַכִּים, וְגֻנְּבוּ לִכְפָרִי בִּסְפָרִים נְזִירִים,

הַיּוֹדְעִים לַחְשׂוֹף סִבּוֹת סְתָרִים וּלְבַסֵּס כָּל סֵדֶר עַל נְצָחָיו,

לִמְצֹוא הֶכְרֵחַ מְחֻכָּם בַּכֹּל, וְלִשְׁקוֹל כָּל תּוֹעֶלֶת אֲפִילָה,

הַמְשַׁפְּרָה כָל דִּרְאוֹן דָּם שָׁפוּךְ בְּחֶזְיוֹנוֹת שַׂגִּיאִים לֶעָתִיד,

שֶׁאֵין קָרְבָּן יָקָר לְפִרְיוֹ… כִּי יֵשׁ וְכָל אֵלֶּה לֹא יוּכְלוּ

לְפַזֵּר אַף שֶׁמֶץ מֵאוֹתוֹ הָעֶצֶב הַגָּדוֹל שֶׁקִּנֵּן

כְּסֵמֶל פַּלָּצוּת בְּצַפְצָפָה יְתוֹמָה, לֹא יוּכְלוּ לְהָפִיס

בְּנַפְשִׁי אֶת מֶמֶר הַקִּטְרוּג שֶׁל אַבָּא, שֶׁאוּלַי תְדַמּוּ

כִּי טַעַם זְקוּנִים לוֹ תָפֵל, שֶׁקָּהָה לַעֲסִיס בֹּסֶר מְרוֹרִים,

הַמְפַרְנֵס אֶת צַמְלוֹ וּמְעַתְּדוֹ לְדוֹרוֹת חֲזָקִים הָעוֹלִים

מֵאֵדֵי הַקֶּטֶל הַמְצָרֵף — — —





הָרֵעִים הִקְשִׁיבוּ לַדְּבָרִים

וְהָיָה הַכֹּל מוּבָן כָּכָה, וְאֵין מַה לְהִשְׁתָּאוֹת מַדּוּעַ

הַנַּעֲרָה דִבְּרָה כָל זֹאת… יָרְדָה דִמְמַת־פִּתְרוֹן לָעוֹלָם,

וְנוֹפְפוּ צְעִיפֵי מִסְתּוֹרִין, כְּאוֹמְרִים לְפַלֵּשׁ רָז עוֹלָם

עַל אַפְסֵי מַשְׁתִּיּוֹת: הַכֹּל גָּלוּי־מוּבָן, סְתָם לְלֹא פֵשֶׁר,

כָּל סֵדֶר מְשֻׁלָּב בְּהֶפְכּוֹ… נִצְטוֹפְפוּ עֲנָפִים וּצְלָלִים

בְּגַן לֵיל הַקַּיִץ, וַיִּלְכְּדוּ בְרִשְׁתּוֹת הַיֶּרֶק הַמְּבֻשָּׂם

דַּג־סַהַר מִנִּבְכֵי כְּחֹל־כָּסֶף; הִתְנוֹעֲעָה מְנוּחוֹת הַבְּרֵכָה

וַתְּיַשֵּׁן בַּרְוַזֵי הַבָּר בְּבַבּוּאוֹת הַסְּבָךְ לָעֲרָבִים:

נִתְעַרְבְּבוּ מַעְלָה וּמַטָּה, גִּשְׁשׁוּ צַמָּרוֹת בַּקַּרְקַע,

וּגְזָעִים הִתְהַלְּכוּ תְלוּשִׁים — כְּמוֹ עָיַף קִמְעָא הַיּוֹצֵר,

אוֹ חָמַד שַׁעֲשׁוּעִים וָנֹפֶשׁ, וַיְעַמְעֵם בְּקַרְנֵי אֶצְבָּעוֹת

לְרֶגַע עַל עֵינָיו הַצּוֹפוֹת הֲלִיכוֹת הָעוֹלָם וְחֻקָּיו…

וְיָפֶה וְחָפְשִׁי לְהַפְלִיא לוֹ נִרְאֶה הָעוֹלָם הֶהָפוּךְ,

הַמְשֻׁחְרָר מִזְּדוֹן תְּבוּנַת בְּעָלָיו וּמֻפְקָר לְנוֹיוֹ וּלְחַגּוֹ,

הִתְחַפֵּשׂ כְּעוֹצֵם עַפְעַפָּיו לְהַשְׁהוֹת אֶת מַרְאֵה הַקְּסָמִים —

וְאוּלַי הִתְעַשֵּׁת בִּדְמִי הָאֱלֹהִים — — — פִּתְאֹם גָּעוּ הַפָּרוֹת

בָּרֶפֶת בְּאֶנְקָה עֲמוּמָה וְכִבְדַת מַחֲשַׁכֵּי כְּמִיהָה,

כַּהֲמִית חֶבְלֵי תֹהוּ מִתְקַשֶּׁה בְּלִבְטֵי תִקּוּנוֹ, כִּי הִרְגִּישׁ

שֶׁשָּׁבְתָה יַד אֵל עַל נַחְשׁוֹלָיו מִשַּׁכְלֵל אֶת פֶּלֶא הַיְצִירָה

עַד תֻּמּוֹ, עַד אָפֵס כָּל אֹפֶל וְטִשְׁטוּשׁ: כֹּה חִלְחְלָה פִרְאָה

וּצְרוּדָה הַגְּעִיָּה בְחֶבְיוֹן שַׁבָּתוֹן וְהֶפְקֵר לַיּוֹצֵר,

וַתְּשַׁסַּע פָּרֹכֶת הַתַּשְׁבֵּץ שֶׁל רִקְמַת בֵּין שְׁמָשׁוֹת לְעֵינָיו,

וַתְּגַל לוֹ כָל תֻּרְפָּה וְגַלְמוּת — כְּבַד־רַעֲמָה הִתְנָעֵר הַבּוֹרֵא

בְּנִקְרַת הַפְּלָדוֹת לָאֵין־סוֹף, עַל אָבְנֵי סַפִּיר וּבְרָקִים,

וַיְרַתֵּק חִשּׁוּקֵי שַׁלְהָבוֹת וַיִּפְקֹד מַעַרְכוֹת הַבְּרִיאָה…





כֹּה חָלְפוּ דִמְדּוּמֵי הַשְּׁקִיעָה וַיָּשָׁב הַכֹּל כְּשֶׁהָיָה,

וְשׁוּב כָּבֵד סֵבֶל הַפְּלִיאָה וְחִידַת הָאַחֲרִית בָּעוֹלָם —

הַנַּעֲרָה פָנְתָה לָלֶכֶת וַתַּעֲצֹר רֶגַע בְּדַרְכָּהּ:

"וּרְאוּ זֹאת שָׁכַחְתִּי, וַאֲנִי כְבָר הוֹאַלְתִּי לְסַפֵּר אֶת מוֹצְאוֹת

הַשְּׂדֵרָה: עִם שִׁלְהֵי הַקַּיִץ, בַּבֹּקֶר, בַּבֹּקֶר, מִצַּמְרוֹת

הַפָּז לַצַּפְצָפוֹת יֵשׁ נִסָּה תִיּוּרוֹ הָרִאשׁוֹן לַזֹּהַר

כָּל גּוֹזָל מֵעֲדַת הַחֲסִידוֹת, שֶׁרִפְּדוּ לָהֶן תַּחֲנַת קֵיצָן

זֶה דוֹרוֹת בְּסִבְכֵי הַשְּׂדֵרָה… שִׁקְשְׁקוּ כְנָפַיִם צְעִירוֹת;

וַיָּחָג הָאֶחָד חוּג קָטָן, וַהֲלוּם פְּחָדִים הָפַךְ אֶל קִנּוֹ,

וְאָחִיו הָעַז יִשֵּׁר מַסְלוּל, עַד שִׁכּוֹר מִתְּכֵלֶת וְחֶדְוָה

שָׁב עָיֵף אֶל רִבְצוֹ וַיִּלְטֹף אֶבְרָתוֹ — זוֹ תְּשׁוּרַת הַפְּלָאִים

שֶׁבִּשְּׂרָה לוֹ אשֶׁר עֲרָבוֹת וְיַמִּים בְּבַגְרוּת מֻטָּתָהּ…

בָּעֶרֶב בָּעֶרֶב עַל רֹאשׁ הַצַּפְצָפוֹת נִתְכַּנְּסוּ הַחֲסִידוֹת

מִטִּיּוּל־הַצַּיִד עַל פְּנֵי שְׂדוֹת קָצִיר וָאָחוּ, וּבְטֶרֶם

תֵּרַדְנָה לְקִנָּן עוֹד צָפוּ זְמָן רַב, אַחַת אַחַת בַּלַּיְלָה,

מַקּוֹרָן מְשֻׁרְבָּט וְלֹא עָיְפָה רַגְלָן — עַד פַּעַם בָּעֶרֶב,

כִּי עִפְּרָה רוּחַ צְפוֹנִית וַתִּדְלַח גַּוָּנֵי הָאֹפֶק,

וַיִּדְהֶה אַרְגְּמָנוֹ וְהֶאֱפִיר, וְרִדְּדָה עוֹפֶרֶת שִׁפּוּלָיו,

וְצֻנַּן הַטַּל עַל פַּגַּת הַשְּׁזִיפִים — הֵן טִכְּסוּ בְלַחַשׁ

סוֹד עָצוּב וּמְבַשֵּׂר סְתָו וּמַקּוֹרָן נִזְדַּקֵּר אֶל חָזָן…

וְלֹא תָמִיד כִּוַּנְתִּי לְהַשְׁכִּים מִמָּחֳרָת בַּבֹּקֶר וְלִרְאוֹת

בְּהַפְלִיג הַסִּיעָה דָרוֹמָה… וְעַכְשָׁו לַשָּׁוְא רַק תְּרַפֵּד

עִם כָּל אָבִיב חָדָשׁ אֶת סֻבְּכָהּ הָרַךְ הַצַּפְצָפָה: כָּל חֲסִידָה

תְשַׁכַּח אֶת רַגְלָהּ וְהָלְאָה תְעַקֵּל נְתִיבָה מִמֶּנָּה,

כְּאִלּוּ מִיַּרְקָהּ הָרַעֲנָן גּוֹלְשָׁה כָּל אֵימַת זִכְרוֹנוֹת

לְמָרֵר שִׁיר שַׁלְווֹת כָּל קֵן… אֵלֶּה תוֹלְדוֹת צַפְצָפָה פְּלִיטָה,

שֶׁשָּׁבְקוּ אֲחָיוֹת לָהּ תָּרְנֵי שְׁחָרִים וּמִפְרְשֵׂי שְׁקִיעוֹת,

אַךְ חִנָּם תְּגַשְׁגֵּשׁ בְּעֹגֶן הָעֶצֶב אֶל חוֹף מַלְכוּת מְנוּחוֹת…

וְסִלְחוּ לְפִטְפוּט שִׁפְחַתְכֶם הָעֶרֶב, וְשָׁלוֹם…" וּפִתְאֹם

וְלֹא נוֹתַר כָּל זֵכֶר לְעָגְמָה בְּקוֹלָהּ, הוּא צָהַל שׁוּב עַלִּיז

וְגוֹדֵשׁ עֲסִיסָיו בְּפִזְמוֹן מְרֻסָּק מֵאַפְלוּל הֶחָצֵר —

עַד נִדְמֶה לָרֵעִים הַדְּהוּמִים, כִּי תִעְתְּעָה בָמוֹ, הִתְחַפְּשָׂה,

וַתִּצְפֹּן הֵד לַעֲגָהּ וּמִרְיָהּ… וּבְעוֹד רֶגַע עָרְגוּ בְסֵתֶר

לִצְחוֹקָהּ הֶעָנֹג וְחָצוּף, שֶׁתְּמוֹל הֶעֱלִיבָם וְעַתָּה

הוּא חִלְחֵל בְּדָמָם כְּצֶלְצְלֵי קַיִץ בְּמֶרְחַב עַרְבָּיִם…





כְּבָר לָכַד הַלַּיְלָה מֶמְשַׁלְתּוֹ וַיָּנַח בִּכְחֹל אַפִּרְיוֹנוֹ,

וִירִיעוֹת יְצוּעוֹ הִתְפַּשְּׂקוּ לְרֵיחַ־חֲצוֹת לָאַגָּדוֹת

שֶׁיָּצְאוּ לָשׂוּחַ, עֲטוּפוֹת רְדִידִים, עַל פְּתִילֵי מִשְׁעוֹלִים —

עַל דַּרְגוֹת הַמַּעֲקֶה יָשְׁבָה הַנַּעֲרָה וַתִּקְבֹּל בְּאָזְנֵי

הַכֶּלֶב הַנֶּאֱמָן: "הַגִּידָה, מֶה עָוְתָה גְבִרְתֵּךְ, כִּי חָלְפָה

הַשַּׁלְוָה מִנָּוִי, וְחֶדְוַת הַמָּחָר שֶׁתָּמִיד בְּהִירַת

הָרֹן יִשְּׁנָה אוֹתִי וַתְּאַלֵּם זְהַב חֲלוֹמוֹתַי לַבְּקָרִים,

הַלַּיְלָה הִתְנַכְּרָה רְעוּלַת הַפָּנִים, וְעֶצֶב גִּבֵּחַ

מְעֻנַּן הַמֵּצַח וּגְרוּם הָאֶצְבָּעוֹת הִתְרַפֵּק עַל עַרְשִׂי

בְּחִיּוּךְ שֶׁקָּפָא עַל רַחֲמָיו — וָאִירָא וָאֵצֵא אֵלֶיךָ…

וְהֵן הָיְתָה וְגֻנְּבָה הַתּוּגָה גַּם אֵלַי מִמַּחֲבֵא חֹרְשׁוֹת,

אַךְ דָּמִי לֹא עִכְּלָהּ וּבְטֶרֶם תִּסְתַּכֵּל בְּמָהֳלוֹ הוּא פְלָטָהּ

עִם רִנַּת הַסְּמָדַר לִנְעוּרָיו וּלְקֶצֶב מְחוֹלוֹ הַפָּזִיז — — —

הֲתִזְכֹּר, לֹא כְבָר זֶה, וְאשֶׁר רְוֵה בְשׂוֹרוֹת נִסֵּר בְּדָמִי,

שֶׁכָּפַן מִתְפָּעֵם לְפִשְׁרוֹ, וָאֶשְׁאַל לְתֻמִּי אֶת אַבָּא:

מַה יַּעֲשֶׂה הָאָדָם הֶעָצוּב בְּיָדָיו וּבְרַגְלָיו?… וְכִהָה

אָז אַבָּא בְפַחֲזוּת בִּתּוֹ, אַךְ תַּחַת גַּבּוֹתָיו הַקְּפוּדוֹת,

עִם דְּאָגָה מֵעַיִן קַנְתָּנִית, הִסְתַּכְּלָה בְרָכָה חֲשָׁאִית

לְרוּחַ הַגִּילָה שֶׁנָּשַׁב בְּאַלְמְנוּת בֵּיתוֹ לְרַגְלִי…

וְאֵיךְ אֶשָּׂא פָנַי אֶל אָבִי וְלֹא אָבִיא לוֹ מָחָר עִם בֹּקֶר,

לְהָשִׁיב אֶת נֶפֶשׁ הַגַּלְמוּד, תְּשׁוּרַת עֲלוּמַי וּשְׂשׂוֹנָם?…

כִּי לֹא בִימֵי עָנָן וּסְתָו, בִּשְׁבוֹת פִּלְחֵי עֶרְיַת הֶחָרִישׁ,

וְזוֹחֲלִים אֹרְחוֹת עַרְפִלִּים טְעוּנִים עַל שִׁכְמָם הַתָּלוּל

גַּבְנוּנֵי מִפְלָשִׁים אֲפוּרִים־מְטֻלָּאִים — לֹא אָז בָּא לִי עָצְבִּי,

כְּעֶרְגָּה טְמִירָה וְרַבָּה לְצִדְפֵי עֲנָנִים בְּנִיסָן,

לְמִטְרוֹת הַכֶּסֶף בַּקַּיִץ, כְּגַעְגּוּעִים לְחִטָּה מִתְפַּטְּמָה,

לְפִטְרֵי הַתִּרְזָה הַגְּלוּמִים־מְתוּקִים, לְחַג יַעַר רַעֲנָן

מְדֻשָּׁן הַשְּׂרָף וּמִתְרוֹנֵן מִצִּיּוּץ מַקְהֵלוֹת עִם שֶׁמֶשׁ —

כִּי עַתָּה, בִּמְלוֹא שֶׁגֶר קַיִץ, בַּעֲמֹד דְּגַן־יָהּ בְּכָל חֻמּוֹ,

וְאֶגְלֵי הַתּוּת אָדְמוּ, קָרְשׁוּ, וּמוֹזְגָה הַשֶּׁמֶשׁ בְּסֵפֶל

כָּל פְּרִי עֲסִיס רִקְחָהּ, שֶׁאָסִיף רַב תַּבְלִין יְטַגְּנוֹ בַמִּטְבָּח

בְּעַרְבֵי דְלָפָיו — גָח עָצְבִּי לְהַבְהִיל מַס־פִּרְטוֹ מִסְּמָדְרִי;

מִתַּמֵּם הִתְנַכֵּר מֵאָרְבּוֹ, פִּתַּנִי בַלָּאט לְהִתְגָּרוֹת

הֲתוּלִים בִּשְׁנֵי הַקַּיָּטִים, וַיְנַקֵּשׁ לְחֻפְשִׁי וַיְסַרְסֵר

בִּצְרוֹר פִּרְחֵי שָׂדֶה… סִבְּכַנִי בְבוֹגְדוֹת וַיַּחְצִיף

אֶת לִבִּי בְרַהַב וּבְעָקְבָה… וַאֲנִי עוֹד בְּאֶמְצַע הַמִּשְׂחָק,

וַיַּחְשׂוֹף הַמַּסְוֶה מִפָּנָיו וּבְעֶצֶם הַקְּרָב צָבַת דָּמַי —

עוֹד עָמַד בִּי לִבִּי לְכַסּוֹת תְּבוּסָתִי, לִפְלוֹט בְּשִׁיר עַלִּיז

חָצֵרָה… בְּבֵיתִי דְּלָקַנִי הָעֶצֶב וַיָּשֶׁת עַל עַרְשִׂי — — —

וְקָצִיר כִּי יָבוֹא בִמְהֵרָה, אֵיךְ אֶתְחַר בְּצַהֲלַת שִׁירִי

וּבְמִסְפַּר אֲלֻמַּי עִם נַעֲרֵי הַכְּפָר, וְאֵיךְ אוּכַל אֲפַגֵּר

בִּמְחוֹל הַהִלּוּלִים בָּעֶרֶב הָאַחֲרוֹן, בִּכְלוֹא שְׂדוֹת הַקָּצִיר

אֶת מִכְסַת זְהָבָם לַגְּרָנוֹת — וַאֲנִי, הֲלֹא תֵדַע, הִסְכַּנְתִּי

לַעֲנֹד זֵר הַחִטָּה לְרֹאשִׁי, לָצֵאת הָרִאשׁוֹנָה בַּמְּחוֹלוֹת —

הָהּ, נִירוֹן…" אַךְ כִּרְכֵּר הַכֶּלֶב וְנִרְתַּע מִגְּבִרְתּוֹ, וַיְזוֹרֵר

כִּי סָלְדָה לְשׁוֹנוֹ מֵאֵגֶל מָלוּחַ שֶׁגָּלַשׁ מִלֶּחְיָהּ…





בִּמְרוֹם שַׁדְמוֹת־תְּכֵלֶת יַד מִי תָלְשָׁה אֶחָד לְאֶחָד כּוֹכָבִים,

וַתִּקְלַע שַׁרְבִיטֵי קִלְחֵיהֶם לְלַפֵּד צְרוֹר־פִּרְחֵי סַפִּיר;

וְלִקְרַאת פִּתְרוֹן־חֲצוֹת מִתְמַהְמֵהַּ צַפְצָפָה הִפְצִיעָה צַמַּרְתָּהּ.




ג. מֵהַמַּטָּרָה וְהָלְאָה






מְנֻפִּים בְּשִׁבְעַת רְקִיעִים וּשְׁקוּפִים עַד אַפְסֵי כָל פְּרָזוֹן

הִכְחִילוּ צָהֳרַיִם בַּשָּׁרָב; נִסְתַּחֲפוּ שָׂדוֹת וּדְרָכִים

כְּתַלְמֵי קֶצֶף מְגֻוָּן בְּאוֹקְיָנוֹס מְפֻלַּשׁ הַזֹּהַר —

בַּמְּרוֹמִים הַשֶּׁקֶט הַגָּדוֹל, הַתְּכֵלֶת הַזַּכָּה: שַׁבָּתוֹן…

רַק מַטָּה, בִּמְסִלּוֹת יְרוּטוֹת לַטֶּרֶף, עוֹד רָחֲשׁוּ, זָעוּ

נְקֻדּוֹת עֲמוּמוֹת: נְמָלָה וְאָדָם וְעֵדֶר שֶׁל פָּרוֹת,

שֶׁשֵּׂרְכוּ נְתִיבוֹת מִחְיָתָם בְּעָמָל וּבְמֶמֶר הַשּׂוֹגְשִׂים

וּבוֹלְשִׁים בְּכָל סֶדֶק וָפֶרֶץ… כֹּה חֹל וְשַׁבָּתוֹן נִתְלַכְּדוּ,

הִתְכַּתֵּשׁ שֶׁקֶט אַפְסָיִם עִם אוֹיֵב קְשֵׁה רָצוֹן וְסַרְבָּן,

שֶׁהִתְקִיף אַפֶּדֶן שַׁבְרִירָיו וַיִּמְרוֹד בַּמַּלְכוּת הַקּוֹפְאָה

עַל זָהֳרֵי נִצְחָהּ, תָּקַע צִפָּרְנָיו בַהֲדַר בְּלִימַת אֵלִים,

כִּרְסֵם בְּסַפִּיר עֶרְיָתָהּ וְחִטְחֵט וְקִרְקֵר בַּבְּדָקִים,

פִּתְפֵּת שַׁלְמוּתָהּ בִּמְרִי, וְהִשְׁרִיצָהּ פְּרוּדוֹת וּקְמָצִים;

וַיַּעֲמֵק לַחְשׂוֹף וּלְזַעֲזֵעַ עַד נִקֵּר מִטַּבּוּר מַשְׁתִּיתָהּ

אֶת שְׁלִילָהּ הַטָּמִיר, הַנִּרְדָּם בְּאַחֲרוֹן חֶבְיוֹן בְּתוּלֶיהָ,



הַגָּלוּם בְּשִׁבְעַת צְעִיפֵי הַתֹּהוּ: הַחַיִּים — וְלַד־זְקוּנִים



לַהֲוָיָה שֶׁפֻּתְּתָה פַעַם לַפְּרִיָּה, וַעֲיֵפָה תְנַסֶּה

לַשָּׁוְא לִסְכֹּר רַחְמָהּ — — — אַךְ יֵשׁ אֲשֶׁר פִּתְאֹם יִתְפַּכְּחוּ הַחַיִּים —

בְּאֶחָד מֵרִבְבוֹת אָדָם בְּאֶחָד מִשִּׁשִּׁים לְרֶגַע —

יִשְׁתַּחְרְרוּ־יִתְהוּ בְּאֵיבַת תִּמָּהוֹן עַל חֶתֶף דּוֹרוֹתָם,

יַכִּירוּ כָל מִרְמַת הָאַכְזָב הַמֵּסִית, וּבְכִלְיוֹן מְחוֹזִים

יַעַרְגוּ לְשַׁלְוַת קַדְמַת־מְקוֹרָם, שֶׁצַּר כְּבָר לְקַבְּלָם

וּלְשַׁכֵּךְ אֶת שִׁטְפָּם בֵּין גְּדוֹתָיו… וְכֹה שׁוֹקְעָה מִתְפּוֹרְרָה מַלְכוּת

שַׁבָּתוֹן עַל הוֹד עַקְרוּתָהּ־שַׁלְמוּתָהּ, כִּי נִפְתְּתָה פַעַם

לַפֵּרוּד — וּבְמֶרֶד נִדָּחִים נִקְלָעִים הַחַיִּים לְלֹא מִקְלָט.





קָרְנוּ מִפְרְשֵׂי זָהָב בַּמֶּרְחָב שֶׁשָּׁקַט עַל תְּכֶלְתּוֹ,

וְלֹא יָדַע כִּי גָמֵל בְּבִטְחַת מְרוֹמָיו הַבָּגֶד — כִּי פִתְאֹם

מִשַּׁחֲקֵי־דְמִי, בְּהִשְׁתַּתֵּק הָעוֹלָם עַל תּוֹעֲפוֹת צָהֳרָיו,

מִיַּרְכְּתֵי תְכֵלֶת, מִתָּרְנֵי הַחַג, נִתַּק נֵץ, מַהֵר הִשְׁחִיר

לְעֵבֶר חֲצַר הָאֲחֻזָּה, הִגִּיעַ עַד גְּבוּל הָעֲרָבוֹת,

הַחוֹפְפוֹת בִּבְרֵכַת־נְזִירוֹת שְׂעָרָן הַיָּרֹק לְלֵיל כְּלוּלוֹת,

וַיְרַדֵּד אֶת מֻטַּת כְּנָפָיו וְנִתְקַע בֵּין גַּלֵּי הַזָּהָב…

כִּמְעַט רֶגַע הִקְשִׁיב לְפִטְפּוּט הַיּוֹנִים עַל גַּגּוֹת הֶחָצֵר,

וַיָּחָג בַּכְּחֹל מַטָּה מָטָּה, וַיִּצְלֹל כַּאֲנָךְ כְּבַד עֹפַרְתּוֹ

אֶל בֵּין הַשַּׁאֲנַנּוֹת — עוֹד רֶגַע וְהִמְרִיא וּמַשַּׁק כְּנָפָיו

בְּעֹז שִׁסַּע נָתִיב בַּשֶּׁמֶשׁ; וַהֲדוּקָה בֵּין קַרְסֵי צִפָּרְנָיו,

מִתְעַלְּפָה לְרֵיחַ מְרוֹמִים, לְפַחֲדֵי בַרְזֶל בִּבְשָׂרָהּ,

אַט פִּרְפְּרָה יוֹנָה וַתִּגְוַע… שָׁם הַרְחֵק בַּחֻפָּה הַכְּחֻלָּה,

הַבָּרָה מִבֹּר וּמְסֻנָּנָה בְּכוּר שִׁבְעַת רְקִיעֵי בְּרֵאשִׁית,

חֲרוּזֵי אֳדָמִים נִגָּרוּ, וְחֻלְּלָה טַהֲרַת שַׁבָּתוֹן

מִשַּׁוְעָה גוֹסֶסֶת לַחַיִּים, מְנַּהֲקַת רֶצַח שְׂבֵעָה.

לֹא פַעַם כְּבָר תָּפַשׂ הַזָּקֵן אֶת רוֹבוֹ לִנְקֹם דְּמֵי יוֹנָיו

בַּנֵּץ, אַךְ הַנַּעֲרָה עָצְרָה בְיָדוֹ: "אַל, אַבָּא!… מְרַחֲמָה

אֲנִי אֶת צִפָּרַי הַתַּמּוֹת וּלְבָבִי מִתְחַמֵּץ לָרֶצַח,

אַךְ חָסָה אֲנִי עַל הַדּוֹרֵס, הַבָּא עַל אַכְזְרִיּוּת מִשְׁפָּטוֹ,

שֶׁחָקְקוּ לוֹ בֵּין מִפְלָשָׁיו, וְלֹא יַעֲרִים לְכַסּוֹת דַּם טַרְפּוֹ;

וְכֹה יָפֶה וְאַמִּיץ יַדְבִּיקוֹ, כֹּה הָגוּן וְגֵאוּת תַּפְקִידוֹ". —

הִשְׁתָּאֶה הַזָּקֵן לְבִתּוֹ וְלֹא יָרָה… בְּחָדְשֵׁי הַקַּיִץ

לֹא בִקְּרָם הַנֵּץ כֹּה תְכוּפוֹת, כִּי תָר מִחְיָה גַסָּה לְקִנּוֹ

עַל מְאוּרוֹת אַרְנָבוֹת וּשְׁפַנִּים, שֶׁחִמְּמוּ וַלְדֵיהֶם בַּשֵּׁמֶשׁ,

בַּסְּבָךְ וּבַחִישׁוֹת, שָׁם חִטְּטוּ הַקּוֹרְאִים סָפִיחַ וּשְׁרָצִים:

רַב הָיָה הַטֶּרֶף בְּהֶפְקֵר הַשְּׁפֵלוֹת, בַּסּוּף וּבֶהָרִים,

וּבִרְכַּת כָּל צַיִד בְּכֵרָה יְרֻקַּת מַזְנוֹנִים, שֶׁנִּמְנֶה

גַם הוּא בֵין חֲשׁוּבֵי קְרוּאֶיהָ, וְלֹא יָרַד לָצוּד בַּחֲצֵרוֹת

זְמַן רָב, וַיִּמָּנַע מִנְּווֹת הָאֲנָשִׁים עַד גְּמַר מִשְׁתֵּה קַיִץ…

אַךְ יִקְרֶה בְעוֹנַת הַקָּצִיר, בִּדְפוֹק הָעֲבוֹדָה לַשָּׂדוֹת

אֶת כָּל תּוֹפֵשׂ מַגָּל וּמוֹשֵׁךְ בְּתֶלֶם, וּבְדִמְמַת חָצֵר

מְנַסְּרָה בְגִיל עֲדַת יוֹנִים, כְּעָב רְקוּמַת תּוֹלָע וְחָרוּץ,

מִסְתַּדְּרוֹת בְּטוּרִים לְבָנִים, וּבְתַחֲרוּת־טִיסָה מִתְבַּדְּרוֹת

כְּבַהֲרוֹת כֶּסֶף שֶׁשֻּׁבְּצוּ בִזְהַב אַבְקוֹת זִיו — וְּבעֹוד רֶגַע

תָּחוֹגְנָה שׁוּב נִכְחָן; וְגָדְלָה הֲמֻלַּת הַפְּתַיּוֹת לְגָרוֹת

הַנֵּץ מִמִּקְלָטוֹ, וְעָט לָאֲחֻזָּה לָבֹר לְקִנּוּחַ

סְעוּדָתוֹ הַגַּסָּה עוֹד לִפְתָּן מְפֻטָּם בַּעֲדַת הַמִּתְעַלְּסוֹת. — —

אֲבֵלוֹת נִדְחָפוֹת הַיּוֹנִים לִדְגֹּר עַל קִנֵּיהֶן בְּרֶטֶט,

וְלֹא יָדְעוּ כִּי הָיְתָה יַד גְּבִרְתָּן הָרַכָּה וּמְלַטְּפָה בַמָּעַל,

כִּי צִדְּדָה הָרוֹבֶה מֵאוֹיְבָם וַתָּחָס עַל רָהְבּוֹ הַפִּרְאִי —

הֲלֹא הִיא שֶׁתֵּדַע לְנַשְּׁקָן וּלְרַפֵּד הַקֵּן לְגוֹזְלֵיהֶן,

וּבְשַׁאֲגַת קֶסֶם בְּעֵינָהּ תְּלַוֶּה אֶת רוֹצְחָן, תִּסְתַּכֵּל

אֵיךְ גּוֹלְשִׁים בַּתְּכֵלֶת הַצְּלוּלָה אֲגָלִים אֲדֻמִּים וְחַמִּים…





מִמּוּל אַשְׁדּוֹת תּוֹלָע בְּנִמְלֵי הַשְּׁקִיעָה, עַל שְׂדוֹת דָּגָן גְּמוּמִים,

הִתְפַּרְקְדוּ צְלָלִים בֵּין אָהֳלֵי זָהָב מְתֻלָּמִים שֶׁתָּקַע

הַקַּיִץ בְּיוֹם קָצִיר רִאשׁוֹן; הַפּוֹעֲלִים שָׁבוּ מִמְּלַאכְתָּם…

הִתְנַהֵל הַזָּקֵן לְאַט, דּוּמָם קָטַף מְלִילוֹת וּפְרָכָן

בֵּין כַּפָּיו וַיָּבָר לְרוּחַ הָעֶרֶב גַּרְעִינֵי הַדָּגָן,

שֶׁשְּׁזוּפִים וּצְנוּמִים הִתְגּוֹלְלוּ עַל כַּף מְיֻבֶּלֶת וְנֻפּוּ

אֶל בֵּין אֶצְבְּעוֹתֶיהָ, וְכִסְכְּסֵם אַחַת אַחַת מֵרְגִילוּת

בֵּין שִׁנָּיו, וְלֹא עָמַד עַל טִיבָם וְטַעְמָם… עַד שֶׁלֹּא יָכֹל

הִתְאַפֵּק, וַיִּפֶן אֶל בִּתּוֹ, וַיְהִי דְבָרוֹ שָׁקוּל וּמְיֻשָּׁב,

כַּאֲגַב שִׂיחָה גְרֵדָא: "הֲלֹא תֵדְעִי, בִּתִּי, אֵין נַפְשִׁי לְהַשְׁהוֹת

אֶת שְׁנֵי הַדַּיָּרִים בְּבֵיתִי…" הַנַּעֲרָה לֹא עָנְתָה… וַיּוֹסֶף:

"רָאִיתִי כִּי רָפְתָה הַמְּלָאכָה בַשָּׂדֶה, גַּם אַתְּ בִתִּי אֵינֵךְ

כְּדַרְכֵּךְ מֵאָז — נָפַל דָּבָר וַאֲנִי טֶרֶם אַבְחִין — וָאֶשְׂכֹּר

לִי פּוֹעֲלִים שְׁנַיִם חֲרוּצִים… הַאֵין טוֹב לַחְשׂוֹךְ אֶת הַחֶדֶר

בִּשְׁבִילָם וְיוּכְלוּ לְהַשְׁכִּים לַשָּׂדוֹת…" וַתַּעַן הַנַּעֲרָה,

וַיְהִי כְמוֹ מָשְׁכָה בְּרֶסֶן הַמִּלִּים לְנַהֲגָן בְּנַחַת:

"בָּאָסָם רַק שְׁאֵרִית בְּלִיל סֻבֵּי אֶשְׁתַּקַד בַּפִּנּוֹת, וְעוֹד פְּנוּיוֹת

הַגְּרָנוֹת כִּמְעַט, כִּי מָאתַיִם הַסְּאִים לֶפֶת־שֶׁמֶן שֶׁנֶּעֶרְמוּ

בָהֶן, בְּיוֹם גְּשָׁמִים רִאשׁוֹן, בִּשְׁבוֹת קְצִיר הַשָּׂדוֹת, תְּדוּשֵׁן —

כָּהֵם וְכָהֵמָּה אִם פּוֹעֲלֵי שָׂדֶה לְךָ תִּשְׂכֹּר, לֹא יֶחְסַר

לָהֶם מְקוֹם לִינָה — וְאֵין צֹרֶךְ בַּחֲדַר הַדַּיָרִים…" וַיֵּלְכוּ

תְּפוּשֵׂי מַחֲשָׁבוֹת וְשׁוֹתְקִים, אַךְ הֻדַּק עוֹד יוֹתֵר הַקֶּשֶׁר

הַמְאַחֶה בַלָּאט רַעְיוֹן סְתָרִים, וְכֹה כָבְדָה הַשְּׁתִיקָה הָאוֹרְבָה

פְּעוּרַת הַלּוֹעַ הָרֵיק — עַד שׁוּב פָּתְחָה הַנַּעֲרָה בִּמְתִינוּת,

וַתְּהִי כְמַמְשִׁיכָה לְאַט שִׂיחָה קַלָּה, וּכְאִלּוּ לֹא הָיְתָה

מַפְלִיגָה בְדֹחַק דְּבָרֶיהָ לִתְחוּם רָחוֹק־זָר, כְּמִתְכַּוְּנָה

לְיַשֵּׁב הַלֵּב וּלְהַפְתִּיעוֹ וּלְפַיְּסוֹ עַל כָּרְחוֹ — וַתֹּאמַר:

"הִתְקַדֵּם הָעוֹלָם וַתֶּרֶב הַדַּעַת וְאָנוּ לֹא נֵדָע,

כִּי סָגַר עָלֵינוּ הַכְּפָר עַל אַדְמָתוֹ וְשִׁפְעוֹ וְשִׁקְטוֹ…

אַךְ יֵשׁ אֲשֶׁר יָבוֹא הַדַּוָּר וְקֻנְטְרֵס בְּיָדוֹ, שֶׁשָּׁלְחוּ

אֵלֵינוּ מִבָּתֵּי חֲרשֶׁת רְחוֹקִים, וְאָז רַק נִשְׁתּוֹמֵם

לִרְאוֹת בַּצִּיּוּרִים צֵרוּפֵי מְכוֹנוֹת שֶׁשֻּׁכְלְלוּ, חֻדְּשׁוּ,

אֵי שָׁם הַרְחֵק מְאֹד מִכְּפָרֵינוּ, לְכָל שִׁמּוּשׁ בַּיִת וְשָׂדֶה,

וְדוֹחֲקוֹת רַגְלֵי כָל חַי וְכָל כֶּלִי, שֶׁעָמְדוּ תְּדִירִים

לְאֵמוּן תַּפְקִידָם מֵאָז דְּאֹג אָדָם לְמִשְׁקוֹ עַד הֵנָּה.

וְרֶגֶשׁ תִּמָּהוֹן וְגָאוֹן בִּפְנֵי חָכְמַת אָדָם יְבוֹאֵנוּ…

אַךְ יֵשׁ וְיִלָּוֶה אֵלָיו, מִמַּאֲפֵל נֶפֶשׁ יְגֻנָּב,

כְּרִפְרוּף מַשְׂטֵמָה עִוֶּרֶת־אִלֶּמֶת לָאוֹיֵב הַיָּהִיר,

הַמְבַיֵּשׁ כֵּלֵינוּ הַיְשָׁנִים שֶׁטִּפְּלוּ בָּם דּוֹרוֹת בִּשְׁלֵמוּת,

שֶׁלְּטָשׁוּם בְּעַרְבֵי הַקָּצִיר בְּרִנַּת הַשּׂבַע לְמָחָר…

אָז יֵשׁ כִּי נִתְגַּנֵּב לָאֻרְוָה בַלָּאט — נְוֵה־מִשְׁנֶה לָאָדָם

מֵאָז וּמֵעוֹלָם — וְאַהֲבַת חֶמְלָה וּבשֶׁת בְּעָלִים,

שֶׁזִּלְזֵל בְּעַבְדּוֹ הַיָּשָׁר, תּוֹקְפוֹת לִבֵּנוּ לְמַרְאֵה

הַסּוּסִים הַמְנוֹפְפִים זָנָב וְנוֹחֲרִים מְנוּחוֹת עַל בְּלִילָם,

וְעוֹד זֵעַת עֲמָלָם מַלְבִּינָה אֶת קִצְפָּהּ עַל מֶתֶג וָרֶסֶן.

בְּחִבָּה נְלַטֵּף צַוָּארָם, נִתְחַפֵּשׂ כְּגוֹמְלֵי אֱמוּנִים,

אַךְ שִׁקְשׁוּק הַמְּכוֹנָה כְבָר אוֹרֵב מִדַּפֵּי הַקֻּנְטְרֵס הַמְצֻיָּר,

וּמְבַשֵּׂר אֶת בּוֹאָהּ וְלַעֲגָהּ לְכָל מִדּוֹת גְּמוּדוֹת־עֲנָווֹת,

שֶׁסִּפְּקוּ עַד דַּי כָּל צָרְכֵינוּ בִסְאָה וּבְקַב — הִיא תְהַתֵּל

בְּאוֹן צֶמֶד סוּסִים, בִּשְׁרִירֵי צַלְעוֹתָם, וּבְמַעְגַּל חֶשְׁבּוֹנוֹת

תְּחַשֵּׁק אוֹפַנִּים וּגְלִילִים לְהַבְהִיל אֶת נִדְלֵי בַרְזִלָּה;

וַיְהִי לָהּ הַשָּׂדֶה לְקַו וְהַגֹּרֶן לְמִדָּה שַׁחְצָנִית,

תַּקִּיפַת נִצָּחוֹן כְּבָר נָטְעָה בַלֵּב אִי־אֱמוּנָה וְחֶשֶׁד

לִמְרוּצַת הַדָּם בְּכָל חַי, וּלְכָל רָצוֹן מְגַשֵּׁשׁ, מִזְדָּרֵז,

נִכְנַע מִפַּחֲדֵי אֲדוֹנִים, אוֹ פוֹעֵם מֵחֶדְוַת יְצִירָה…

כֹּה נִפְסַח עַל מִפְנֵה דְרָכַיִם: רִוַּחְנוּ מְעַט מֵהַיָּשָׁן,

וַנְּפַגֵּר לִפְנוֹת אֶל הֶחָדָשׁ, הַכּוֹבֵשׁ. — הֲלֹא תַסְכִּים, אַבָּא,

כִּי שָׁם פִּסְּגוּ מִגְדָּל עֵת אָנוּ בַשֵּׁפֶל יָשַׁבְנוּ, וּבְמִקְרֶה

יִזְדַּמֵּן גַּם לָנוּ לְהַכִּיר דְּבַר־מָה מִכָּל גְּדוֹלוֹת הַמַּדָּע,

בְּבוֹא קֻנְטְרֵס דֹּאַר, אוֹ קַיְטָן כִּי יָסוּר לְבַלּוֹת פֹּה חֻפְשׁוֹ…"





מַקִּישָׁה בְקַרְסֵי הַבַּרְזֶל וּמְקַשְּׁשָׁה פְלִיטֵי שִׁבֳּלִים,

הִתְנוֹדְדָה מַלְקֵשָׁה עַל כִּבְרַת הַקָּצִיר לְדַהֲרוֹת הַסּוּסִים,

שֶׁשִּׁפּוּ אָזְנֵיהֶם לַהֶמְיָה בְּשִׁקְתוֹת עֶרֶב וְסִמְּרוּ

שְׂעַר גַבָּם הָעוֹקֵץ לְזֵכֶר הַמַּתְבֵּן הַמְרֻדָּד בֶּחָצֵר,

שֶׁנָּעִים הִתְגָּרֵד בֵּין שְׁבָבָיו הַחַדִּים לְלֹא רֶסֶן וְאֻכָּף…

זֵרְזָם הַזָּקֵן, בְּעָבְרָם עַל פָּנָיו, בִּשְׁרִיקָה, כְּדַרְכּוֹ,

אַךְ דַּעְתּוֹ לֹא נָחָה וַתַּפְלֵג מִשְּׂדוֹתָיו וְסוּסָיו — וְיָפֶה

הוּא כִוֵּן כָּעֵת, אֵיזֶה דֶרֶךְ עָבְרָה הָרוּחַ עַל בִּתּוֹ

לְגַנּוֹת הַחַיִּים בַּכְּפָר וּלְהִתָּפֵשׂ לְלַהֲטֵי הַמַּדָּע,

שֶׁאָחַז בְּעֹרֶף הָאָדָם, כְּגוֹרָל אַכְזָרִי הַמְחַדֵּשׁ

לוֹ פְלָאָיו לַבְּקָרִים, וְלֹא יֶאֱצַל לוֹ שָׁהוּת לְסַגְּלָם וּלְחַמְּדָם —

וּמַה יִּסְכֹּן דָּבָר לָאִישׁ, אִם לְאָהֳבוֹ וּלְכַבְּדוֹ לֹא יִלְמָד?…

עַד נִדְמֶה לִפְעָמִים: מִתְרַבְרֵב רַק תּוֹפֵשׂ הָאָדָם בְּשֹׁבֶל

מַדָּעוֹ, אַךְ חָרֵד הוּא יֶחְרַד מִפָּנָיו, פֶּן יִפְרֹץ הַגְּבוּלִים

וְיִשְׂגֶה מֵעַל כֹּחַ אָדָם… וְכֹה שִׂכְלוֹ הַפָּשׁוּט לוֹ יַגִּיד:

יֵשׁ עֹדֶף שֶׁל דַּעַת בְּכָל אֵלֶּה, וְאֵין יִתְרוֹן שֶׁל חָכְמָה וָאשֶׁר.





וְלֹא חִנָּם הוּא חָשַׁשׁ מֵאוֹתָן הַשִּׂיחוֹת שֶׁקָּבְעָה לָהּ בִּתּוֹ

בַּלֵּיל עַד חֲצוֹת עַל הַדַּרְגֵּשׁ עִם שְׁנֵי הַדַּיָּרִים, וַיֵּדַע:

מִשָּׁם תָּזְזָה רוּחַ עָלֶיהָ לְזַלְזֵל בְּבִרְכַּת הַשַּׁלְוָה,

מִשָּׁם נִכֵּל שָׂטָן סְבוּא־דַעַת לְקַרְקֵר כָּל בִּטְחַת מוֹרֶשֶׁת…

כְּבָר נִים וְלֹא נִים, עֲיֵף יְמֵי קָצִיר, יֵשׁ גָּרַע אֶל אָזְנוֹ

בְּדִמְמַת לַיִל זְעֵיר מִשִּׂיחָתָם, וּטְרָדוֹ מִשְּׁנָתוֹ;

אָז יֵשׁ יַרְדֶּה חוּשָׁיו הַיְגֵעִים לִקְנוֹת דְּבַר חָכְמָה לְנַפְשׁוֹ —

כֹּה הוֹבִיר אֶת מֹחוֹ זֶה שָׁנִים, וַיִּשְׁכַּח הַמְּעַט שֶׁבִּנְעוּרָיו

אָצַר אֶל תּוֹכוֹ — וְעַכְשָׁו הוּא קוֹלֵט בְּקשִׁי כָּל דַּעַת:

וּרְאֵה עוֹלָם טָרוּף!… בֶּאֱמֶת הוּא יָבִין רַק רָאשֵׁי הַדְּבָרִים,

וְלֹא יֵרֵד לְסוֹף דִּקְדּוּקֵיהֶם, כִּי תֶאֱנַס הַשֵּנָה עַפְעַפָּיו;

אַךְ כָּל מַה שֶׁתָּפַשׂ הֶגְיוֹנוֹ, שֶׁגָּרַס וּפָשַׁט לוֹ שִׂכְלוֹ,

הוּא מַמָּשׁ הַהֶפֶךְ הַגָּמוּר, חִלּוּפֵי תוֹלָדוֹת וּסְדָרִים:

כֹּה יִפְסְקוּ מַסְקָנוֹת עַל דְּבָרִים גְּלוּמִים וּשְׂפוּנִים, וּנְהִירִים

לָהֶם כָּל מַסְלוּלֵי רָקִיעַ, כְּאִלּוּ הֵם סִתְּתוּ מַרְצַפְתָּם;

אַךְ סְתוּמָה וּנְעוּלָה בְאַפְלוּל רָז נֶצַח לְתוֹפְשֵׂי הַמַּדָּע

כָּל אַבְקָה שֶׁאָדָם מְיֻשָּׁב לֹא יַעֲמוֹד לְבַקְּרָהּ וּלְאַזְּנָהּ;

יִתְאַמְּרוּ בִקְפִיצַת דְּרָכִים וּמְבַדִּים כָּל מֶרְחָק לְאֶפֶס,

אַךְ כָּל גַּרְגַּר חוֹל — זֶה הַמּוֹדָע הֶעָנָו הַנִּדָּשׁ מִקַּדְמוּת

הָעוֹלָם מִתַּחַת כָּל עָקֵב — הוּא לָמוֹ הַחִידָה הָרַבָּה,

לוֹ קוֹשְׁרִים פִּלְפּוּלִים וּמְצַחְצְחִים סְבָרוֹת כְּעַל עוֹלָם וּמְלוֹאוֹ,

וּמְעַרְבְּבִים אֶפֶס בְּאֵין־סוֹף — — — מַה פֶּלֶא כִּי כָל בּוּקֵי סְרִיקֵי

הַלָּלוּ הֶאֱפִילוּ עַל מֹחָהּ הַבָּהִיר וְזָרִיז שֶׁל בִּתּוֹ,

לְעַמֵּם אֶת חֻפְשַׁת חֶדְוָתָהּ וּלְעַבֵּט כָּל מִשְׁעוֹל מְנוּחוֹת,

וּלְזַהֵם אֶת כָּל יִפְעַת קֵיצָם הָעוֹמֵד בְּזִיווֹ וּבְשִׁפְעוֹ…

כִּי עַתָּה הִיא שׁוּב לֹא תְפַזֵּז בַּקָּמָה, לִצְבּוֹט קִמְצֵי דָגָן מְלוֹא זְרוֹעוֹת

וּלְשַׁנְּסָם בְּאֵזוֹר שִׁבֳּלִים — כְּפוּפָה וְנָאוָה וּגְמִישָׁה;

וְלֹא תִלְגֹז עֲמִירִים לַקָּרוֹן, זְקוּפָה וּסְמוּרַת שָׁדַיִם,

וּדְרוּכַת שְׁרִירִים מְרַנְּנִים־מָקְשִׁים מִטַּל וּמִשֶּׁמֶשׁ,

וְרוֹמְזִים־מַעֲרִימִים עַל סוֹד שֶׁהִצְנִיעוּ בְּבֹחַל חַמּוּקִים;

וְלֹא עוֹד קוֹל הַנַּעֲרָה יְהַדֵּד בְּתַחֲרוּת קְרוֹנוֹת לַגֹּרֶן,

לְעוֹדֵד הַסּוּסִים וּלְהַצְלִיף עַל גַּבָּם בִּנְזִיפָה שֶׁל חִבָּה — —

הַמַּגָל, הַמַּלְגֵז, הַסּוּס: סִמְלֵי בִטְחָה וְחֶדְוָה לָאֹשֶר,

שֶׁפֵּאֵר כָּל פִּנָּה בַּחַיִּים וְתִחֵם בְּמִדָּה כָּל חֶלְקָה,

שֶׁמָּזַג כָּל כּוֹס בִּכְדֵי צָמָא, וַיְפַלֵּשׁ לָאָדָם מְחוֹזָיו



בִכְדֵי מֶתֶק עֲיֵפוּת — לֹא הָלְאָה! — כִּי זִמֵּן לוֹ צִפּוּי־דְּפִיקָה



בְּשַׁעֲרֵי שְׁלוֹמִים וּפִתְרוֹן עַל חוֹפֵי הַסּוֹד לַמֵּעֵבֶר —

הָהּ, קָהֲתָה נַפְשׁוֹ שֶׁל אָדָם בָּאֵלֶּה, הִשְׁתַּחֵץ וַיִּבְעָט

בְּכָל קַו וּבְכָל צַו — וּבְשַׁרְווּלִים מָפְשָׁלִים וּבְזֵעַת נִדּוֹנִים

הוּא יִרְתֹּם הַמְּכוֹנָה לְאַפְסֵי אָפְסָיִם…

הִרְגִּישָׁה הַנַּעֲרָה

בְּקִטְרוּג שֶׁנִּמְתַּח כְּנֶגֶד רֵעֶיהָ, וְכָל נִדְנוּד הִרְהוּרִים

שֶׁגָּרְמוּ דְבָרֶיהָ לַזָּקֵן, כִּי רַבַּת חָשְׂכָה גַעְגּוּעִים

בְּנַפְשָׁהּ גַּם הִיא לָעוֹלָם הַהוֹלֵךְ וְנִדְחַק וְנִפְסַל

אֶל בֵּין כְּלֵי בַרְזִּלּוֹ הֶחָלוּד, בְּפִנּוֹת אֲפֵלוֹת לְכָל חָצֵר:

כְּאִלּוּ הוּא שִׁכֵּל מֶגְרְפוֹתָיו, וְלֹא זָן דּוֹרוֹת רַבִּים וּשְׁלֵמִים

מִלְּשַׁדּוֹ וְגִרְשׁוֹ — רַק עִרְבֵּל שִׁדְפוֹן מֹץ רֵיק בַּאֲסָמָיו;

וּכְמוֹ לֹא הִסְתַּכֵּל בִּדְוַי רַב־נַפְתּוּלִים, לֹא נָהַק קְשֵׁה־לְבָטִים,

כֹּה חָרֵד עַל כָּל אֲפִיל אשֶׁר לְקַשְּׁרוֹ, לְכַמְּרוֹ וּלְבַכְּרוֹ

עַל תַּלְמֵי הַדּוֹרוֹת הַבָּאִים; כְּאִלּוּ לֹא חִטֵּא כָל רֶגַע

שֶׁל שׂבַע וּבִטְחָה בְּדַם בְּרוּרֵי בָנָיו לְכָל אֵל וְשָׂטָן,

מִתּוֹךְ עֶרְגַּת גְאֻלָּה טְמִירָה — — — וְהִיא זָכְרָה מְחָאַת יְדִידָהּ,

שֶׁאֶתְמוֹל בַּלַּיְלָה, בְּטֶרֶם יִפָּרְדוּ, בָּהּ סִיֵּם הַשִּׂיחָה;

וְגֻנְּבוּ לִדְבָרָיו הִשְׁתַּפְּכוּת מִתְאַפְּקָה, הֵד נֶפֶשׁ רְעוּלָה,

וְרֶטֶט עִצָּבוֹן שֶׁנִּכְנַף בְּצִנְעַת חֲצוֹת לְלֵיל קַיִץ,

שֶׁנָּשַׁב מֵעֹולָם כֹּה זָר וּמִתְנַכֵּר — הֵד־רֶטֶט שֶׁגִּשְׁגֵּשׁ

מִתְבַּיֵּשׁ עַל דַּרְכֵי הַצְּלָלִים לֶעָבַר, הִתְגָּרֵד לְשָׁבְלָהּ

שֶׁל גֵּאוּת יַחְסָנִית־אֲנִינָה לְדוֹרוֹת שֶׁסָּלְדוּ וְחָשְׁשׁוּ

מִמֹּךְ אֶת גְּבוּרָתָם בְּקִיטוֹר וּבְחַשְׁמַל לִשְׁרִיקַת נִצְחוֹנוֹת,

וּלְשַׁלְשֵׁל לְכִבְשַׁן הַמַּדָע “הֲזָיוֹת מוֹרָשָׁה”… וְחִלְחֲלָה

בְּקוֹלוֹ כְּמִיהָה לְדוֹר אֲצִיל דְּיוֹקָן, שֶׁטִּפַּח וְעָצַר

אֶת כֹּחוֹ הַגָּדוֹל לַחֲזוֹת חוֹפֵי פִתְרוֹן לְלֹא שַׁחַץ כִּבּוּשׁ,

שֶׁסִּלֵּק תְּכוּפוֹת אֶת עֵינוֹ מִנִּצְחוֹן הַפֵּרוּד לַאֲבִיקָיו,

שְׁקַפְדַּקָּיו וְאִזְמֵלָיו, וַיִּרְקֹם חֲלוֹם טְמִיר זִוּוּג מִסְתּוֹרִין

בִּדְמִי חֶשְׁבּוֹן־וִדּוּי לַנֶּפֶשׁ, בְּחֶשְׁכַּת צַעֲקָתָהּ לְתַכְלִית;

וַיְהִי תָמִיד מוּכָן וּמְאֻשָּׁר לְוַתֵּר עַל קִנְיְנֵי שִׂכְלוֹ,

בְּעַד אוֹת הָרַחֲמִים, בְּעַד רֶמֶז מִצְעָר מֵאֶצְבַּע הַשְּׁכִינָה:

אֶת זֹאת שַׁוְעַת לְבָטָיו לַפֶּלֶא, כִּסּוּפָיו לְמוֹפֵת אֵל שַׂגִּיא,

אֶת זֹאת דַּעְתּוֹ עָגֹב בַּפִּנּּוֹת עַל פְּלֵיטַת מוֹרֶשֶׁת חֲנוּטָה,

לְפַטְּמָהּ, לְהַחְיוֹתָהּ בְּדָמוֹ, לְהִתְאַכְזֵר לִפְעָמִים לְמַעֲנָהּ

לְשִׂכְלוֹ הַמְבֻגָּר, לְעַפֵּר אַחֲרָיו רֶמֶץ מִזְבְּחוֹת שֶׁקָּרַע —

אֶת כָּל זֹאת לֹא יִסְלַח לוֹ יוֹרְשׁוֹ, דּוֹר קַנָּא לַעִנְקֵי הַבַּרְזֶל,

הַמְמֻנֶּה עַל כְּבוֹד הוֹד הַשֵּׂכֶל, עַל מַלְכוּת אָדָם וּמְכוֹנָה,

הַמַּאֲמִין לַחְתֹּר עַד הַיְסוֹד, וְאִם גַּם יִסַּק עַד גְּבוּלוֹת הַבְּלִימָה,



וּלְפַקֵּל הַכֹּל עַד הַגַּרְעִין: אֵל־רָז הַמִּטַּמֵּר בִּקְלִפּוֹת — —



וְלֹא בְסִגּוּפִים וּתְפִלּוֹת, וְלֹא בְתַעְתּוּעֵי כָל חָזוֹן,

וּבְעַנְוַת חַלָּשִׁים, כְּדוֹר אָבוֹת נִרְצַע לְאַפְלוּל מִקְדָּשִׁים

לְאֵל זֶה יִתְגַּבָּר: בִּקְרָב וּבַזַּיִן שֶׁהֶעֱרִים לְסַגְּלוֹ

וּלְחַדְּדוֹ בִמְרִי, בּוֹ יִסְתָּעֵר עָלָיו לַחֲשׂוֹף אַחֲרוֹן סִתְרוֹ,

מִגֵּאוּת צַמַּרְתּוֹ עַד תַּחְתִּית סֻלְיָתוֹ, עַד רֶחֶם הַמַּשְׁתִּית;

וְאִם לֹא יִמְצָאֶנּוּ — וְיִשְׁתַּחְרֵר הָאָדָם מִבְּדוּיוֹת רִאשׁוֹתָיו,

וִיבַשֵּׂר הַדְּרוֹר לְדוֹרוֹתָיו: אֵין גַּרְעִין, אֵין רָז עוֹד — נִשָּׁבַע

מִלְחֶמֶת נִצָּחוֹן לָאָדָם עַל הֶפְקֵר כָּל עַפְרוֹת כַּדּוּרוֹ,

לְמַגֵּר גְּבוּלוֹת אַחֲרוֹנִים וּלְבַסֵּס אֶת מַלְכוּת הַמְּכוֹנָה,

וּלְמַהֵר לְקַצְוֵי מֶרְחַקְיָה בְּגֵאוּת חוֹלְשֵׁי הָאֶפֶס;

וְלֹא נִשְׁקֹט עַד נַפְלִיג בַּחֲלַל עוֹלָמֵנוּ הַפָּחוּת, וְנָטוּס

בְּהֶלְעֵם אֶחָד מִמֶּנּוּ וָָהָלְאָה — — —

יְדִידָהּ אָז צָחַק,

כִּי הִגְזִים זְעֵיר בְּקִנְאָתוֹ, — לָהּ נִדְמֶה שֶׁרֻסַּק בְּתַחְתִּית

צְחוֹקוֹ הֵד עָמוּם, דָּלוּחַ, לֹא חָפְשִׁי, הַמְקֻלָּע לְלֹא מוֹצָא

בֵּין גְּבוּלִים מִתְנַשְּׁכִים בְּטִשְׁטוּשׁ שִׂרְטוּטִים, בֵּין דִּמְיוֹן וּמַמָּשׁ,

בֵּין צִדּוּק שְׁאָר־רוּחַ אָדָם, כְּבוֹד־גְּבוּרָה לְמַרְדּוּת גְּאוֹנוֹ,

וּבֵין עֶצֶב סוֹלֵד מִתְפַּלֵּץ מִקְּצִיר הָאַכְזָבָה הָאָיֹם — — —





כֹּה עָמְדוּ הֵם תְּמוֹל מוּל הַלַּיְלָה, בְּאֶמְצַע הַיְקוּם שָׁלשׁ נְקֻדּוֹת,

שְׁלשָׁה גַּרְגְּרִים מְסֻבְּלֵי עֲסִיסִים עַל תַּלְמֵי נִיר פֶּלִאי,

וְסִמֵּר טַל־תְּסִיסָה כָּל עֹרֵק וְדִרְדֵּר נַחְשׁוֹלִים בְּדָמָם —

וּבְדָלְיֵי גָּבִישׁ מְצֻנָּן הַלֵּיל טָבַל סְפוֹגֵי־חִתּוּלִים,

וְכִרְבֵּל שַׁלְהֲבוֹת בְּשָׂרָם לְשַׁבֵּב אֶת חִשְׁקוֹ הַמְלֻפָּד;

וּנְשִׁימָה בְשֵׁלָה־עֲצוּרָה הִזְדַּקְּרָה בְכָל אֵגֶל דָּמָם,

וּדְמָמָה בִדְמָמָה הִתְרוֹצְצָה וְנִבְלְמָה שְׁאָגָה בִשְׁאָגָה…

נִשְׁתַּתֵּק הַמְשׁוֹרֵר: הַס, הָס! אֵי שָׁם חִידָה וּפִתְרוֹן נִזְדַּמְּנוּ

עַל אֵם דַּרְכֵי לֵיל, וַיַּחֲלִיפוּ לְאוֹת אַחֲוַת נֶצַח אֶת סִמְלֵי

שִׁלְטוֹנָם בַּיְקוּם… וּמַה צֹרֶךְ עוֹד עַתָּה לָרִיב וּלְקַנֵּא

לָרָז שֶׁהִתְעָרָה כֹה תַקִּיף בַּלֵּיל, בַּמֶּרְחַקִּים, בִּבְשָׂרָם,

וַיִּלְכֹּד כָּל מְסִלָּה, וַיְפַשֵּׂק אֶת שׁוּלָיו הַפְּרוּמִים.

וּקְפוּדִים־מַרְתִּיעִים נִתְכַּנְּסוּ לְמִפְקַד הַשֵּׂכֶל־הַמַּצְבִּיא

כָּל רִכְבוֹת הַמַּדָּע לָאָדָם מְדֻבְלְלוֹת רִתְמָה, וּשְׁמוּטִים

כְּלֵי נִשְׁקָם נִצְטוֹפְפוּ רָצָיו אֶל תַּחֲנוֹת הַמָּגֵן לַמֹּחַ.

הוֹי, הוֹי! עוֹד לֹא נִכְנָע, לֹא קָרַס מַצְבִּיאָם מִתֹּקֶף רָז־לַיְלָה;

הֲלֹא זֶה תַפְקִידוֹ לָפוּחַ בָּאֹפֶל וְלִנְעֹץ בַּדְּרָכִים

לַפִּידֵי דְגָלָיו… אַךְ חָדַר מִמַּאֲפֵל הַמֶּרְחָב צְלִיל עָגוּם —



לֹא נַהַם, לֹא אִיּוּם: צְלִיל רָפֶה, כְּהֵד־מִצְמוּץ מִתְּהוֹמוֹת כּוֹכָבִים —



וַיִּכְפֹּן לִקְרָאתוֹ בְרֶטֶט כְּבַד עֶרְגָּה הַדָּם זֶה הַפִּרְאִי,

שֶׁהוּא אַלְפֵי שָׁנִים וְדוֹרוֹת הֵתְחָרֵשׁ־הִתְחַכֵּם לְרַסְּנוֹ

לְדַהֲרוֹת שִׁחְרוּר, לְסַנְּנוֹ מֵהֶבְלֵי בְּנוֹת־קוֹלוֹת עֲמוּמוֹת,

וְזֶה עוֹמֵד בְּעֵינוֹ עֲדֶנָה: לִכְרוּז רִאשׁוֹן טָמִיר הִתְקַשֵּׁר,

וַיִּטּשׁ דַּבָּרוֹ, וַיִּנְהַר לְמַחְשַׁךְ רִאשׁוֹנֵי מַבּוּעָיו,

מִתְרַפֵּס־מִתְרַפֵּק וְעוֹגֵב עַל צִלְלֵי אַגָּדוֹת וְשִׁעְבּוּד — — —

קְשֵׁה־חָזוֹן הַדָּם!… וְלֹא פַעַם רִאשׁוֹנָה הִתְפָּרֵץ מִפָּנָיו

בְּכָל עָצְמַת עַגְמָתוֹ, אַךְ הוּא עֹוד יְיַשְּׁבוֹ וִיפַלֵּג אֲפִיקָיו…

וְנֶאֱמָן שַׁחְרוֹ הַגָּדוֹל שֶׁל חֹפֶשׁ וְשִׁלְטוֹן לָאָדָם,

שֶׁיַּעֲזֹב אָז מִדְלַח חֵזְיוֹנוֹת וִירַפֵּא סְדַן שִׂכְלוֹ הַמְצֻנָּן,

לְנַגֵּד כָּל מִקְשַׁת מִכְשׁוֹלִים וּלְשַׁלְהֵב תְּשׁוּקָתוֹ אֲלֵיהֶם,

וּלְהַפְלוֹת כָּל מַמָּשׁ מִבְּדָיוֹת, כָּל כִּבּוּשׁ מִכְּמִיהַת הִזְדַּוְגוּת…





אַךְ תְּמוֹל בַּחֲצוֹת עוֹד הִתְחַמְּקוּ מִתִּלֵּי חֲזִית הַכִּבּוּשִׁים,

בְּרַעֲדַת מִצְהָלָה, שְׁלשָׁה חַיָּלִים מִפְּלֻגַּת חֲלוּצָיו;

וּשְׂמֵחִים לִתְבוּסַת מַצְבִּיאָם הַלֵּיל, בַּחֲשָׁאי שִׁרְבְּבוּ זֵינָם,

וַיִּבְלְמוּ רִנַּת בְּגִידָתָם — לָשׁוּב וּלְהִסְתַּפֵּק כִּילָדִים

בְּשַׁעֲשׁוּעֵי צִפִּיָּה, וּלְקַשֵּׁט זָוִיּוֹת קְטַנּוֹת בָּאֵין־סוֹף;

וּלְהַטִּיל הַדְּאָגָה הַגְּדוֹלָה לְכִבּוּשׁ, לְשִׁכְלוּל, לְשִׁחְרוּר,

לִגְמוּל סִבְלוֹת חַיִּים, עַל שֶׁכֶם אָב כַּבִּיר, רַב טִרְחוֹת וָעֹצֶם…

וְכָל חַד מֵהַשְּׁלשָׁה אָז גָּשַׁשׁ לִשְׁבִילוֹ, לְאָשְׁרוֹ, לְרִשְׁתּוֹ,

לְרַוּוֹת אֶת רִגְעוֹ בַנֶּצַח, הָלְאָה מִתְּשׁוּאוֹת הַכּוֹבְשִׁים. —

וּכְיֵשׁוּת עִקָּרִית וְנַעֲלָה עַל כָּל אָסְפֵי שָׁלָל לָאָדָם,

עַל כָּל נִצְחוֹן מִרְדּוֹ, כְּאַחֲרוֹן פֶּשֶׁר, כְּחוֹתַם יַצִּיבוּת

יְחִידָה־אֵיתָנָה בַּבְּרִיאָה, שׁוֹטְטוּ גִלְשֵׁי עֲנָנִים.

עַל כִּפַּת זְבוּל כֶּסֶף לַסַּהַר, וְזָרְמָה לְעֻמָּתָם מִתַּחַת

הַקָּמָּה בְרִנְנַת אִוְשָׁה — כְּאִלּוּ בָם צוֹרְרָה וַדָּאוּת

עֶלְיוֹנָה עַל כָּל כִּבּוּשׁ שֵׂכֶל… וְנוֹעֲדוּ לְטִכּוּס חֲשָׁאִין

עַרְמוֹנֵי הֶחָצֵר, מְחַפִּים עַל סוֹדָם בְּקַפְּלֶטִים עֲבֻתִּים,

שֶׁנָּשַׁר שְׂרִיד פּוּכָם הַלְּבַנְבַן כְּשֶׁלֶג מְפֻטַּם בְּשַׂם קַיִץ — — —

לְלֹא נִיד טִרְטְנוּ מֹאזְנֵי הַבְּרִיאָה, נְעוּצִים בְּקִמְרוֹן דְּמִי־תֵבֵל:

הַכֹּל שָׁקוּל זֶה לְעֻמַּת זֶה — פִּתְאֹם זִמְזֵם זְבוּב נִדָּח שְׁכוּל אוֹרָה,

וְנִתְלַכְסֵן בִּרְתֵת מוֹט־עֲנָק לַמֹּאזְנָיִם, הִתְחַלְחַל, הִזְדַּקֵּר

בָּאֵין־סוֹף, כְּאִלּוּ הִתְמוֹטְטוּ עָלָיו כָּל סַלְעֵי הֶהָרִים;

וְעוֹד רֶגַע וַיְפַסַּח כִּמְקֻטַּע שׁוֹקַיִם, כְּמוֹ הָשְׁלַךְ נֵטֶל

סְעָרִים עַל פֶּלֶס אֳפָקָיו… מִי קִלְקֵל פֹּה חֶשְׁבּוֹן מְלוֹא־תֵבֵל

שֶׁנָּח שְׁוֵה־מִשְׁקָל עַל שַׁלווֹ?… וַהֲלֹא פֶקַע מִצְעָר לְאַהֲבָה

רִאשׁוֹנָה רַק נָבַט וְלִבְלֵב רַק־נִצָּה בְלֵב נַעֲרָה — לֹא יוֹתֵר

הִתְחוֹלֵל בַּיְקוּם — — — וְעוֹד סֵרֵב וְהִתְרִיס הַלֵּב, וַיִּלָּחֵם

לִדְרוֹר גִּיל יַלְדוּתוֹ, הַמְּקֻפָּח בְּלֹא־יוֹמוֹ, הִתְחַבֵּט, הִתְחַנֵּן

מִדַּם תַּחְתִּיוֹתָיו לַגּוֹרָל לִדְחוֹת יוֹם נִסָּיוֹן — אַךְ טִפְּלָה

הָאַהֲבָה בְּמוֹט מֹאזְנֵי אֵלִים, וַתָּחָשׁ לְמַלֵּא צַו טָמִיר…





הַזָּקֵן וּבִתּוֹ הִתְקָרְבוּ לֶחָצֵר… בִּכְחֹל אֵדֵי שָׂדוֹת,

מִסְתַּמֵּךְ עַל בִּרְכּוֹ הָאַחַת, הָעֶרֶב אַט פָּרַם צַוָּארוֹן

מְסֻלְסַל הַתּוֹלָע מֵרְדִיד מַלְכוּת לַיְלָה, וַיְפַרְשֵׁז אֶת קַלְעֵי

עַרְפִלָּיו עַל מִשְׁתֵּה הָאוּרִים לַשְּׁקִיעָה — וּבְקִדָּה הִסְתַּלֵּק.





וְשׁוּב זָמְמָה בוֹגְדוֹת הַשְּׁתִיקָה וַתִּכְבַּד עַל רוּחַ הַנַּעֲרָה

שֶׁחָשָׁה בְרֹגֶז אָבִיהָ, וַתֵּדַע כָּל מְרוֹרוֹת שֶׁסָבַל

עַל רֵעָהּ הַמְשׁוֹרֵר, כִּי פִּרְנֵס אֶת רוּחָהּ בְּפִטּוּם מַדּוּחִים

כֹּה חָרִיף וּמַדְהִים, וּמְגָרֶה נְעוּרִים מִסַּף חַיֵי שַׁלְוָה,

מִגַּג וּמִגֹּרֶן, לְמֶרְחַק פְּרָזוֹת תְּלוּשׁ מָחוֹז וְאשֶׁר…

וּבְרַחַשׁ נִחוּמִים הִיא לִטְּפָה אֶת יָדוֹ: "אַךְ שָׁוְא, אַבָּא, תַּחְשׁשׁ

כִּי יַטּוּ יְדִידַי אֶת לִבִּי לִכְפֹּר בְּכָל יָשָׁן — גַּם נַפְשָׁם

לֹא נִתְּקָה עוֹד אַחֲרוֹן שָׁרְשָׁהּ מִמֶּנוּ, וּלְזִמְרַת מַעְיָנוֹ

עוֹד תִּכְפֹּן מִמִּסְכְּנוּת בְּאֵרוֹת לָהּ חָפְרוּ אֶל כָּל מִפְנֵה דְרָכִים

חֲדָשִׁים, וְאֵלֶּה הֵם דִּבְרֵי יְדִידִי: "אַל, נַעֲרָה, תְּדַמִּי,

כִּי נָקֵל לָאָדָם לְמַהֵר בִּנְתִיבוֹת שֶׁסָּלְלָה הַמְּכוֹנָה,

וְכִי דוֹר חָדָשׁ זֶה, הַמְשַּׁלֵחַ חֲלוּצָיו לְמֶרְחֲבֵי כִבּוּשׁ,

רַק רַהֲבֵי רוּחוֹ שֶׁטָּפַח, רַק הֵם דּוֹפְקִים פְּעָמָיו לְקַפֵּץ

כָּל מֶרְחָק — וְלֹא כְאֵב דּוֹרוֹת שֶׁחֻסְפַּס, צָרֶבֶת צִפִּיָּה

מִקַּדְמַת־כָּל תִּכְלָה לַנֶּפֶשׁ וְעַד עַכְשָׁו — וְאֶל חִנָּם, אֶל חִנָּם…

נִתְקַצְּרָה הַשָּׁעָה לָאָדָם מִפַּחַד תַּעֲלוּלֵי תַעֲלוּמָה,

הַמְתַעְתְּעִים בּוֹ עַל תְּהוֹמוֹת זְמַנִּים וּמְבוֹכֵי תוֹלָדוֹת —

מִיֵּאוּשׁ בִּעוּתִים הוּא גִיֵּס כָּל רוּחוֹ וַיִּמְתַּח כָּל שָׁרִיר,

לִסְבּוֹא אֶת כָּל כֹּחוֹ כְּשִׁמְשׁוֹן הָעִוֵּר בְּשִׁכְרוֹן נַפְתּוּלִים;

לְהַרְאוֹת לַיָּרִיב הַנִּסְתָּר כָּל מִרְצוֹ וּמִרְדּוֹ וּמִמְרוֹ,

לְהַרְאוֹת אֵיךְ אָמַן הַנִּיצוֹץ הַפָּגוּם שֶׁנִּפַּח בְּאַפּוֹ,

אֵיךְ טִפְּחוֹ וְחִשְּׁלוֹ לְיוֹם קְרָבוֹת, אֵיךְ גִּמְּשׁוֹ וּפִלְּדוֹ, כִּי יוּכַל

לִבְלשׁ בְּכָל בִּדְקֵי גִזְנַכֵּי הָאוֹיֵב, וּלְהָסִית כְּנֶגְדּוֹ

כָּל גּוּשֵׁי מַחֲנוֹתָיו הַכְּבוּלִים, לְפַקְפְּקָם, לְפָרְקָם, לְפַרְפְּרָם

בְּעֶזְרַת הַמְּכוֹנָה, לָהּ כָּתַב אֶת יָדוֹ, לָהּ הִפְרִין אֱמוּנִים,

אִם תִּזְקֹף מִלְחַמְתּוֹ וּתְשַׁגֵּם כָּל שָׁרִיר וּמְזִמָּה לְמִפְעַל

נִצְחוֹנוֹ, אִם תָּסִיר אֶת חֶרְפַּת סַנְוֵרָיו!… וְאִם תְּרוּעַת הַצְּעִירִים

תִּצְרֹם אֶת אָזְנֵךְ — הֲלֹא כֵן לֹא יֵדְעוּ חֲלוּצֵי כָל חֲזִית

עֱנוּת דַּאֲגַת שָׂרֵי צִבְאוֹתָם וּפַחֲדֵי כִשָּׁלוֹן, וְאַחֲרָיוּת

לְגוֹרַל מִלְחַמְתָּם עִם שׂוֹנֵא, שֶׁפְּעָמִים כֹּה תְכוּפוֹת הִבְקִיעוּ

מִבְצָרָיו, וַיְחַצְצְרוּ לְאַחֲרוֹן מַאֲמָץ — וּבְרֶמֶץ מַפֹּלֶת











הֵם מָצְאוּ אֶת צִלּוֹ, רַק צִלּוֹ, וְזֶה גַם לֹא זָע לְתוֹתְחֵיהֶם…

וְעוֹד מִבְצָר מִתַּלֵּל עַל מִבְצָר, וְצַר אֶרֶךְ־צֵל מְפַסֵּחַ

עַל צִבְאוֹת נַנָּסִים מַתְקִיפִים… וְאַל תִּשְׁעִי, נַעֲרָה, לְצַהֲלַת

חֲלוּצֵי צִי עַלִּיז הִסְתַּעֲרוּת — לְלֵב קַבַּרְנִיטָיו אִם תִּכְרִי,

וְתִשְְׁמְעִי הַלְמוּת חֲרָדוֹת עַד רָאשֵׁי הַתְּרָנִים… וְלֹא פַעַם

בְּסַעַר הַתְקָפָה יְנַסֵּר כְּרוּז צָרוּד וְעָגוּם: לֹא נוּכַל!…

הַדֶּרֶךְ מַעְפִּילָה מִמֶּנּוּ וָהָלְאָה — מָרוֹמָה!… אָז יַפְשִׁיל

הָאֶחָד עֲקֵבָיו לַמִּשְׁעוֹל הַנּוֹשָׁן — וְהוּא מִן הַבְּרוּרִים

בַּמַּחֲנֶה — וּמְהַלֵּךְ הַכְּרוּז וְרוֹוֵחַ — — —" וּרְאֵה לָמְדָה בִתְּךָ



אֶת כָּל אֵלה2 הַדְּבָרִים מֵרֵעָהּ… וּפַעַם, קְשֵׁה־רוּחַ, כֹּה
נִסְּחָם:

"יֵשׁ רוּחַ עִוֶּרֶת לַמְּכוֹנָה הַמְגַמְּאָה לְלֹא־תַאֲוָה אֶל לוֹעָהּ

אֶת כָּל שִׁטְחֵי בּוּר לְכָל עָתִיד, וּפוֹלְטָה לְלוֹא תַגְמוּל וַעֲיֵפוּת

לְשַׁעֲרֵי כָל אָסָם עֲמִירֵי גְדִישִׁים בְּבֶהָלַת תַּבְעֵרוֹת,

בְּטֶרֶם עוֹד תַּמְתִּין הָעַיִן בְּשִׂמְחַת הַנְּבִיטָה לָאָבִיב,

לְלֹא רַחַשׁ יִחוּל וּלְלֹא אִוְשַׁת דְּאָגָה… מִמְּזִמּוֹת בַרְזִלָּהּ

יִסְתָּעֵר עָלֵינוּ בִתְכִיפָה כָל צָפוּי לְאַחֲרִית יָמִים;

וּבְדִיּוּק מְסֻרְגָּל, הַמְגַשֵּׁשׁ לְלֹא רָאֹה וְתָעֹה, הִיא סוֹכְרָה

אֶת כָּל תְּהוֹם־בְּתוּלָה בַנֶּפֶשׁ, וּמְפַטְּמָה עַד קִיא רַעֲבוֹן אָדָם…"

וְאָז נִעֲנַע לוֹ רֵעוֹ בְרֹאשׁוֹ, וּבְחָפְצוֹ לְסַיְּעוֹ בִדְבָרָיו,

הוּא סִתֵּת בְּאֵלֶם רִגְשָׁתוֹ לְצָבְתָהּ בְּמִבְטָא כְּבַד־לָשׁוֹן,

כְּדַרְכּוֹ, עַד מָצָא מְשָׁלוֹ וַיַּעַן: "כִּוַּנְתָּ, יְדִידִי…

מַלְעִיטִים אוֹתָנוּ עַד קִיא… מַמָּשׁ נִדְמֶה כְאִלּוּ הִתְפַּקְּעוּ

כָּל אַלְפֵי הַפְּטָמוֹת לְאֵלַת הַיְקוּם בְּפָגוֹדוֹת עַתִּיקוֹת,

וּמְצַנְתְּרוֹת מֹהַל כֹּה פוֹשֵׁר וְתָפֵל בְּאַשְׁדוֹת מִפְלָצוֹת — — —"…

גַּם הֵם נִמְנִים, אַבָּא, עַל פְּלִיטֵי הַכּוֹבְשִׁים…

הִיא פָתְחָה הַפִּשְׁפָּשׁ

לֶחָצֵר וַתַּבֵּט בְּחִיּוּךְ שֶׁל פִּיּוּס בְּעֵינֵי הַזָּקֵן…

הִתְפּוֹגְגָה מֹרַת רוּחוֹ כִמְעַט, וּבְדַאֲגַת חִבָּה

הִסְתַּכֵּל בְּבִתּוֹ וַיֹּאמַר: "וְעַתָּה לְכִי אֶל הָרֶפֶת,

כִּי צָרִיךְ לִדְאֹג עוֹד לַפָּרוֹת הַטּוֹבוֹת בְּטֶרֶם יִתְרוֹקֵן

מְקוֹמָן לִמְכוֹנָה מְתַּרְמְלָה עֲטִינִים שֶׁל חָלָב…"

הוּא עָמַד

לְבַדּוֹ וְהִקְשִׁיב לְפִזְמוֹן עַרְבִּית הַבָּא מִן הָרֶפֶת

סְפוּג עָגְמָה וּמוֹשֵׁךְ סִלּוּדָיו, כְּשִׁירַת־פְּרִידָה מֵרֵאשִׁית

אֲלֻמּוֹת לְחֵיק הָאֲדָמָה, כְּבִרְכַּת הַגֹּרֶן הַמְקַדְּמָה

פְנֵי קְרוּאֵי בִּכּוּרִים… עוֹד בִּשְּׂרָה הַשִּׁירָה שְׁקִיעַת שְׁלָווֹת

וְצִדּוּק חֶשְׁבּוֹנוֹת לְחַיִּים מִתְעַרְטְלִים־חוֹלְפִים, וּזְרִיחַת

יוֹם חָדָשׁ מִתְפַּקֵּל חַד־קֶרֶן עַל אָפְקֵי דוֹר יָבוֹא, שָׁם רַכָּה־

עֲנֻגָּה כְגִבְעוֹל בָּאָבִיב מְלַבְלְבָה בִתּוֹ… וְלִבּוֹ

שֶׁשָּׁכַב מִתְרַצֶּה לְחִיּוּךְ־חֲנִינוֹת בְּעֵינָהּ, שׁוּב רָטַן

מִתְעַקֵּשׁ: הֲלֹא טוֹב הָיָה עוֹשֶׂה לִפְטֹר הַדַּיָּרִים מִבֵּיתוֹ,

וְטוֹב הָיָה לַהֲרוֹג אֶת הַנֵּץ הַדּוֹרֵס אֶת יוֹנָיו הַתַּמּוֹת…
