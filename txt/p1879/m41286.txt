


	
	אֲנִי לָךְ
	




אֲנִי לָךְ אֵל אֲהַלֵּל וַאֲמִירָךְ

וְתָמִיד אֹחֲזָה סַנְסַן אֲמִירָךְ

וְאַתְּ אֵלִי וּמַלְכִּי לֹא אֲמִירָךְ

וְלֹא אַחְלִיף בְּנַפְשִׁי הַחֲשׁוּבָה:





מְהֵרָה יַעַבְרוּ אוֹיְבַי בְּשִׁלְחִי

אֲשֶׁר הֵם דִּבְּקוּ עַצְמִי בְשִׁלְחִי

רְצֵנִי אֵל וְתִפְנֶה לִי בְשָׁלְחִי

תְּשׁוּרַת מַהֲלָלַי הָעֲרֵבָה:





יְחִידָה, תִּשְׁמְעִי קוֹלִי עֲדִינָה

לְגֹבַהּ וּלְגֵאָה גָאוֹן עֲדִי נָא

וּמִשִּׁפְלוּת קְחִי קָצֶה עֲדִי נָא

נְמִיכוּת לֵב וְאָז תִּזְכִּי לְטוֹבָה:





וְלִי יֶעְרַב בְּעֵת אֶקְרַב לְשִׁירָךְ

וְאֶתְאַו לָךְ לְנִיב קוֹלָךְ וְשִׁירָךְ

וְכָל רוֹאִים לְעֶדְיָךְ אִם לְשֵׁירָךְ

לְךָ תָּהִים וּמִשְׁתָּאִים בְּאַהְבָה:





רְאוּ יִדְמֶה דְמוּת דּוֹד אֶל כְּתָמִים

טְהוֹרוֹת פָּז וְאֵין בָּהֶם כְּתָמִים

וְעִנְיָנָיו הֲלֹא הֵמָּה כְתָמִים

אֲשֶׁר עָבַד לְאֵל יוֹצְרוֹ בְּאַהְבָה:





אֲהַלֶּלְךָ אֱלֹהַי אַתְּ וּמַלְכִּי

וְלָךְ כָּל מִשְׁאֲלוֹת לִבִּי וּמַלְכִּי

וּמִי יִתֵּן לְאוֹיְבִי צַר וּמָל כִּי

לְעִירִי בָא וְשָׂמָּה לִי חֲרֵבָה:





אֲהַלֶּלְךָ בְּתֹף כִּנּוֹר וּמָחוֹל

וְאוּלַי תַּעֲבִיר חֶטְאִי וְתִמְחֹל

לְחַטָּאתִי וְאִם הוּא רַב כְּמוֹ חוֹל

וְלֹא אוֹסִיף אֲנִי עַבְדָּךְ לְדַאְבָה:





בְּחֶמְלָתָךְ חֲמֹל עָלַי וְעִמִּי

לְטוֹבָה אוֹת עֲשֵׂה אֵלִי, וְעַמִּי

תְּקַבֵּץ תּוֹךְ נַחֲלָתוֹ וְעָמִּי

תְּגַלֶּה לִי בְרֹב חֶדְוָה וְחִבָּה:





לְךָ אֲנִי וְכַסְפִּי גַם בְּצִירִי

שְׁמַע קוֹלִי בְקָרְאִי לָךְ בְּצִירִי

חֲשֹׁב דִּמְעִי כְּמוֹ עִנְבֵי בְצִירִי

וְקַבֵּל אֶת תְּפִלָּתִי בְחִבָּה:





מְהֵרָה מִקְדָּשׁוֹ יִבְנֶה כְּרָמִים

וְלִשְׁכוֹתָיו מְכוֹנִים לַכְּרָמִים

וְעֶרְכָּם לֹא נַעַרְכֶנּוּ כְּרָמִים

וְכָל אַבְנֵי יְקָר עֵרֶךְ וְטוֹבָה:





שְׂרֹף נָא אַרְמְנוֹת רַבָּה בְלַבָּה

אֲשֶׁר הִיא הִקְשְׁתָה עָרְפָּהּ וְלִבָּהּ

וְאָז שָׁלוֹם לְהַיּוֹצֵא וְלַבָּא

וְחִישׁ קַיֵּם נֶחָמָה הַכְּתוּבָה:





פְּדֵה עַמָּךְ כִּי שָׂבְעוּ בוּז וְגַם רוֹשׁ

וְחִישׁ קַבֵּץ וּבֶן דָוִד יְהִי רֹאשׁ

בְּיָדָם כָּל בְּשָׂמִים טוֹב וְגַם רֹאשׁ

וְכָל אֶבֶן יְקָרָה הִיא וְטוֹבָה:





וּמִלֵּא אוֹיְבִי נִקְלֶה לְכִסְלִי

וְשׁוּב יֹאמַר סְלַח וּמְחֹל לְכִסְלִי

וּבָךְ שַׂמְתִּי אֱלֹהִים חַי לְכִסְלִי

וּמָחַלְתִּי וְלֹא אֶחְפֹּץ בְּאֵיבָה:





מְהֵרָה תִשְׁלְחָה מַלְכִּי וְאַלְפִּי

וְלִי יִגְאַל וְאֶת צֹאנִי וְאַלְפִּי

וְחִישׁ אַקְרִיב לְךָ עוֹלוֹת, וְאַלְפִּי

בְּיוֹם אֶחָד לְךָ אַעְלֶה נְדָבָה:





וְקִצֵּי הַזְּמַן עָבְרוּ וְצָבָא

וּמָתַי תִּקְבְּצָה עַמִּי וְצָבָא

בְּקָרְאִי נֶחֱלָה רֹאשִׁי וְצָבָה

וְחִישׁ קַבֵּץ לְבִתְּךָ הָאֲהוּבָה:





וְחִישׁ אַשְׁלִים לְךָ נִדְרִי בְּחֶבְרוֹן

וְאַקְרִיב קָרְבְּנוֹתַי לָךְ בְּחוֹב רוֹן

וְכָל אַחַי הָעוֹמְדִים בָּרְחוֹב רוֹן

יְרַנְּנוּ בְּרֹב חֶדְוָה וְחִבָּה:





אֲיֻמָּה, יִתְּנָה לָךְ אֵל חֲנִינָה

בְּעֵינָיו, חִישׁ בְּעִירֵךְ תַּחֲנִי נָא

וּבִטְחִי בוֹ וּבְזֶה תִבָּחֲנִי נַ-

אֲמִי שִׁירָה הֲגוּנָה וַחֲבִיבָה:





אֲנִי אָשׁוּב אֲטַהֵר אֶת שְׁבִילִי

וְחִישׁ הַחְזֵר לְעִירִי אֶת שְׁבִי לִי

וְלִי נַמְתָּ בְּעִירֵךְ תֵּשְׁבִי לִ-

מְנוּחוֹת שַׁאֲנַנּוֹת גַּם בְּטוֹבָה:





וּבָגַד בִּי יְדִיד נַפְשִׁי וְרֵעִי

וְשָׂם עַצְמוֹ כְמוֹ צֹאָה וְרֵעִי

וְדַעְתּוֹ הִיא כְּדַעַת צֹאן וְרֵעִי

וְלֹא זָכַר עֲשׂוֹת חֶסֶד וְטוֹבָה:





בְּגָדַנִי וְהֶלְאַנִי זְמַנִּי

וְהָפַךְ אֶת יְמֵי חַגֵּי זְמַנִּי

לְתוּגָה וַאֲנִינָה, בַזְּמַן נִ-

חְיֶה נַהֵר וְנִתְעַדֵּן בְּטוֹבָה:





לְךָ נַפְשִׁי מְאֹד תַּעְרֹג וְתִתְאַב

סְלִיחָתְךָ וְלֹא תַחְלֶה וְתִדְאַב

וְחִישׁ קַבֵּל שְׁאֵלָתִי וְתֵת אָב

סְלִיחָה לָהּ וְקַבֵּל הַתְּשׁוּבָה:





אֲצַפֶּה לֶאֱלֹהֵי הַצְּבָאוֹת

יְמַהֵר לִי כְאַיָּלוֹת צְבָאוֹת

וְיָחִישׁ לִי זְמַנִּים גַּם צְבָאוֹת

וְאָז נִתְעַלְּסָה יַחַד בְּחִבָּה:
