


	
	בְּלִבִּי חֹם
	




בְּלִבִּי חֹם לְמִפְקַד הַנְּעוּרִים,



וְעַל הָגְלַת יְהוּדָה – אוּר וְאוּרִים



וְעַל שִׁכְנִי טְמֵא מֵת חוּץ לְצִיּוֹן –



כְּמָוֶת אוֹ כְּמוֹ חִבּוּט קְבַרִים.



הֲיִתָּכֵן הֱיוֹת שׁוּבָה לְסוּרָה



וְיוֹם רֹגֶז לְמַרְגִּיזִים וְסוּרִים?



וְהַאִם יֵשׁ לְמָסֹרֶת, בְּפִי עַם



אֲסוּרִים, מִמְּפַתֵּחַ אֲסוּרִים,



לְהוֹרִיד בַּעֲלֵי מַשְׁחִית לְשַׁחַת



וְעַם צַדִּיק עֲלֵי צֶדֶק לְהָרִים?



וְאִם יִיטַב בְּפִי, אוֹ פִי בְנִי, עַל



מְרוֹם דּוּכָן דְּבַר זִמְרָה וְשִׁירִים?



וְאִם אֶרְאֶה בְּנֵי אֵל חַי כְּעָבִים



וְכַיּוֹנִים אֱלֵי צִיוֹן מְהִירִים?



בְּחַיֵּי גוֹאֲלִי חַי! עוֹד אֲקַו, עַד



אֲשֶׁר לֹא-חַי אֱהִי, קִבּוּץ פְּזוּרִים



וְלֹא אֹמַר: אֲנִי אַדִּיר וְהָדוּר,



מְקוֹמִי בֵּין מְלָכִים וְהֲדוּרִים,



וְיֵצֶר לֹא יְפַתֵּנִי בְּאָמְרוֹ:



אֱלֹהִים אַתְּ וְנִכְבָּד מִגְּבָרִים,



וּמַה תּוֹסִיף בְּבוֹא יֶשַׁע לְיוֹסֵף,



וּמַה תִּקְבֹּץ בְּהִקָּבֵץ פְּזוּרִים?



עֲמִידָה בַעֲזָרָה טוֹב לְנַפְשִׁי



וְיָפֶה מִמְּלֹךְ עַל כָּל יְצוּרִים,



וּמִשְׁתֵּה מַשְׁמַנִּים בַּאֲדָמָה



טְמֵאָה – לִי כְּמוֹ מִשְׁתֵּה שְׁמָרִים.



בְּנֵה בִיקָר מְרוֹם צִיּוֹן וְהָרֵם



בְּנֵי צִיּוֹן וְצִיּוֹן, הַיְּקָרִים,



וְחַדֵּשׁ לַאֲהוּבַת הַנְּעוּרִים



כְּמוֹ קֶדֶם וְכַנֶּשֶׁר, נְעוּרִים,



וְהַשְׂבֵּעַ שְׂיֵךָ, הָאֲכוּלָה



בְּפִי לֵישִׁים, בְּחֶלְבֵי הַכְּפִירִים,



וְאָז אָבִיא לְךָ כָלִיל וְאַעַל



עֲלֵי מִזְבַּחֲךָ אֵילִים וּפָרִים.



הֱקִיצַנִי לְבָבִי מִשְּׁנָתִי



בְּעוֹד לַיִל וְלִי דִבֵּר דְּבָרִים:



הֲלַהֶבֶל וְלָרִיק שָׁת אֱלֹהִים



עֲלֵי פִיךָ בְּדֹלַח הָאֲמָרִים,



וְהָיָה לִּי לְמֹוׁשָׁעוֹת, לְמַעַן



אֱהִי יוֹשֵׁב וּפִתְחֵי פִי סְגוּרִים?



פְּתַח פִּיךָ וּמַלֵּא אֶת נְדִיבֵי



עֲמָמֶיךָ בְּנִדְבַת מַאֲמָרִים,



וְשִׁיתָה יַהֲלוֹם פִּיךָ וּמַשְׂכִּית



זְהַב חִכָּךְ בְּחוּט הַשִּׁיר תְּפוּרִים,



וְיַשֵּׁר בַּהֲדַר קֹדֶשׁ זְמִירוֹת



לְאֵל יִשֵּׁר לְךָ בַצַּר הֲדוּירִם,



וְחָבַש יוֹם קְרָב עַל נִזְרְךָ מִן



יְשׁוּעָתוֹ וְעֶזְרָתוֹ נְזָרִים,



בְּקוּמָךְ אֶל קְרָב רַבִּים וְרוֹמִים,



וְעַמִּים מִמְּךָ רַבִּים, וְשָׂרִים



וּבַעְדֶּךָ לְפָנֶיךָ גְּדוּדִים



כְּלַיִל וַחֲיָלִים כַּהֲרָרִים,



וְחָשְׁבוּ כִי פְּעָמֶיךָ בְּאַרְצָם



בְּחַבְלֵי מַלְאֲכֵי רָעִים קְשׁוּרִים



אֲזַי פָּגְשׁוּ רְמָחִים אֶת רְמָחִים



וְרָמוּ הַגְּבִירִים בַּגְּבִירִים



וְשָׁלְחוּ אֶת דְּבוֹרֵי חֵץ, וְעָשׂוּ



אֲשֶׁר לֹא תַעֲשֶׂינָה הַדְּבוֹרִים,



וְהֵמָּה עַל גְּדוֹלֵיהֶם יְהִירִים



וְהֵמָּה עִם קְטַנֵּיהֶם צְעִירִים



וְכָל בָּא אֶל קְרָב חָלוּץ, וְאוּלָם



עֲלֵי רֹאשׁוֹ לְהִתְפָּאֵר פְּאֵרִים,



בְּחֹם לֵבָב כְּאֵשׁ תִּבְעַר, וְרוּם לֵב



כְּשָׁמַיִם, וְקוֹמָה כַתְּמָרִים,



וּבִנְטִירָה כְּמוֹ גָמָל, וְאֵיבָה



כְּמוֹ נָחָשׁ, וְקֶטֶב כַּמְּרִירִים,



אֲנָשִׁים לֹא אֲנָשִׁים הֵם, אֲבָל הֵם



מְלָכִים וַאֲצִילִים מַעֲטִירִים,



וּמַעְמִידִים בְּחַדְרֵי לֵב נְפָשׁוֹת



בְּהוֹצִיאָם נְפָשׁוֹת מֵחֲדָרִים,



כְּאִלוּ עֵץ חֲנִיתָם עַל יְמִינָם



מְנוֹר אוֹרֵג, וּפִיהוּ חַד כְּסִירִים,



וְעֵינֵיהֶם, לְאַבְקַת מַרְכְּבוֹתָם,



מְמֻלָּאוֹת בְּחֹשֶׁךְ לֹא סְדָרִים,



וְגַבּוֹתָם כְּאַרְמוֹנִים סְרוּחִים



בְּפִי חֶרֶב לְאוֹר הַיּוֹם חֲתוּרִים,



וְכֻלָּם יִצְעֲדוּ לָמוּת כְּהוֹלֵךְ



בְּחֶדְוָה לֶאֱרוֹת צוּף מִיְּעָרִים,



וְיָרוּצוּ לְשַׁחֵת אֶת בְּשָׂרָם



וְיָקוּצוּ בְּחַיי הַבְּשָׂרִים.



עֲלֵיהֶם מִמְּדֻכָּאִים כְּוִיּוֹת



וְחַבּוּרוֹת וְרִקְמָה מֵעֲשִׁירִים,



וְגִזְרָה בַחֲצֵרוֹת כַּמְּלָכִים



וּמֵחִצִּים עֲלֵי עוֹרָם גְּזָרִים.



אֲזַי קָם אֵל אֲשֶׁר נִסָּה עֲשׂה לִי



בְּנִסָּיו, אֶת חֲדָשִׁים לַבְּקָרִים,



וְצָפַן מִגְּבוּרָה אֶת לְבָבָם



וְשָׁתָם בַּקְּרָבוֹת כַּשְּׁבוּרִים



וְשָׁלַח אֶת בְּכוֹר מָוֶת וּמָשַׁךְ



בְּנֵי חַיִל בְּצִיצִית אֶל קְבָרִים,



וּבָא בֵינוֹת גְּבִירִים הַצְּמוּדִים



בְּמַעְרָכָה וְהִנִּיחָם פְּגָרִים,



וְאָז הָיוּ בְּמִלְחַמְתָּם פְּרוּדִים



אֲשֶׁר הָיוּ בְּמִלְחַמְתָּם חֲבֵרִים,



וְיַחַד נִסְחֲפוּ, כִּי אֵל הֲדָפָם



וְנָסוּ מִפְּנֵי חֶרֶב גְעוּרִים,



וְרֻבָּם אֻסְּרוּ מֻכִּים, בְּחֶבֶל



מְמֻדָּדִים, וּמִקְצָתָם דְּקוּרִים.



לְעוֹלָם יִשְׁמְרוּ שָׁדוֹת וְחוּצוֹת –



וְהֵם הָיוּ בְּחַצְרוֹתָם שְׁמוּרִים,



וְעוֹד לֹא יִכְסְמוּ רֹאשָׁם לְכָבוֹד



וְתִפְאֶרֶת, וְאֵינֵימוֹ נְזִירִים,



וְצִוָּה אֵל זְמַן לָבֹז יְגִיעַ



בְּנֵי אָדָם וְלָתֵת לַאֲחֵרִים,



וְלָקַחַת בְּשָׂר כָּל שַׂר שְׁמַן לֵב,



קְשֵׁה עֹרֶף, וְלָתֵת לַנְּמֵרִים.



אֲשֶׁר הָיוּ בְּעָרֵיהֶם נְזוֹרִים –



וְהֵם הַיּוֹם בְּתַלְאוּבוֹת חֳמָרִים,



וְגַבֵּיהֶם לְבָנִים כַּלְּבָנָה –



וְעַתָּה הֵם כְּמוֹ כוּשִׁים שְׁחֹרִים.



עֲלֵיהֶם נִקְבְּצוּ דַיּוֹת וְאַיּוֹת,



וְנֶגְדָּם יִמְחֲאוּ כַפָּם שְׂעִירִים,



וְכָל שׁוֹאֵג בְּיוֹם אֵיד כַּכְּפִירִים –



עֲלֵיהֶם יִשְׁאֲגוּ הַיּוֹם כְּפִירִים.



לְצוּרִי אֶעֱשֶׂה שִׁיר, יַעֲלֶה עַל



בְּנֵי הַשִּׁיר, וְיִהְיֶה שִׁיר לְשִׁירִים,



וְיֵלְכוּ יוֹנְקוֹתָיו כַּלְּבָנוֹן,



וְעַד שִׁנְעָר יְשַׁלֵּח קְצִירִים,



דְּבָרָיו יִבְחֲרוּ לָנוּב בְּלֵב טוֹב



בְּמַקְהֵלוֹת וּבַחְתֻנּוֹת בְּחוּרִים,



וְטוּרִים, יִשְׁמְרוּ שָׂרִים בְּלִבָּם



וְיֶהְגּוּ בָם בְּשִׁבְתָּם בַּשְׁעָרִים,



וְיִרְאוּ כִי אֱלֹהִים יֵשׁ וְשֹוֹפֵט



אֲשֶׁר יִשֵּׁר הֲדוּרִים לַיְשַׁרִים,



אֲשֶׁר יֵטִיב לְעוֹשֶׂה טוֹב, וְיֵרַע



לְכָל עוֹשֶׂה וְכָל דּוֹבֵר שְׁקָרִים.
