


	
	תֵּדַע בְּנִי
	




וְלוֹ שִׁיר נִקְרָא בְעַרְבִי אַרְגוּזָה וְזֶה שֵׁם שֶל כָּל שִׁיר
שֶׁחֶרֶז חֶצְיוֹ מֵחֶרֶז סוֹפוֹ (וְהוּא הַנִּקְרָא בִלְשׁוֹנֵנוּ שִׁיר
חֲרוּזִי) וּבָנָה אוֹתוֹ עַל צַוָּאָה אֲשֶׁר כָּתַב וְיִסְּרַנִו בָהּ
וְהוּא הַשִּׁיר הַנִּכְתָּב בְּסוֹף אִגֶּרֶת הַסְּמִיכוּת וּבְעַרְבִי אל
אצאפה

תֵּדַע, בְּנִי, כִּי צוּר יְצָרָךְ נוֹרָא, לָכֵן יְרָא אוֹתוֹ וְחֻקָּיו
שָׁמְרָה,



וּתְהִי בְכָל מִפְעָל וְכָל מַאְמָרָה מִלִּבְּךָ עַל לִבְּךָ מִשְׁמָרָה,



חַרְצֹב זְדוֹנָךְ מִמְּךָ הַתִּירָה וּנְטַע מְקוֹמוֹ תֹם, וְאָוֶן
עָקְרָה.



הַוַּת רְמִיָּה מִמְּעוֹנָךְ בַּעְרָה, כִּי מִלְּפָנִים אֶת אֲנָשִׁים
דָּקְרָה.



אַל בַּחֲדָרִים חֶטְאֲךָ תַּסְתִּירָה, כִּי אֵין לְאִישׁ מֵעֵין אֱלֹהִים
סִתְרָה.



הוֹדֵה וְשׁוּב מֵחֵטְא לְאֵל וּלְנִבְרָא, כִּי שָׁב וּמוֹדֶה נָעֲמוּ
לַמּוֹרָא.



בַּקֵּש אֱמֶת וּבְעַד צְדָקָה חָתְרָה, כִּי הָאֱמֶת – חַיִּים, וְצֶדֶק –
אוֹרָה.



אַוַּת לְבָבָךְ וַחֲמָתָךְ גָּעְרָה, וֶהְיֵה לְדַל– מַשְׂכִּיל,
וְלַמָּךְ – עֶזְרָה.



נַחֵם אֲבֵלִים, וֶאֱמוּנִים נָצְרָה, וּגְמֹל חֲסָדִים, וַעֲנָוָה
אָזְרָה,



מֵתִים לְוֵה וּבְכֵה, וְחוֹלִים בָּקְרָה, וַעְגֹם לְאֶבְיוֹנָךְ
בְּנֶפֶשׁ מָרָה.



וּקְרָא יְמוֹתֶיךָ בְּדִבְרֵי תוֹרָה וּדְרֹש בְּתַלְמוּדָה וּבִין
בַּמִּקְרָא.



אִם לָחֳלִי רַע נַפְשְׁךָ נִמְסָרָה – קַבֵּל, וְתוֹכְחַת אֵל, וְאִם
נֶעְכָּרָה,



וּבְבוֹא מְהוּמָה – רוּץ וְלוֹ הַעְתִּירָה, וּשְׁמוֹ בְּטוֹב וָרָע
לְטוֹבָה זָכְרָה.



וּבְשׁוּרְךָ נִגַּשׂ בְּחוֹב – הוֹן אָגְרָה, וּבְזָר, אֲשֶׁר נִתְפַּשׂ
בְּחֵטְא, הִוָּסְרָה.



תֵּן לַאֲשֶׁר יָדוֹ מְזוֹן פִּיו חָסְרָה, כִּי יַד אֱלֹהֶיךָ לְךָ
הֶעְשִׁירָה,



אַךְ אֵת אֲשֶׁר תִּתֵּן – בְּמִשְׁפָּט פָּזְרָה וּבְדָת, וְרֹב חֵילָךְ
לְךָ הַשְׁאִירָה,



כִּי אִם תְּפַזֵּר כֹּל וְלֹא תוֹתִירָה – תִּדַּל, וְתִשָּׂכֵר לְזָר
בַּאְגוֹרָה,



אִם תֶּאֳהַב אַהְבַת אֲנָשִׁים – שָׁבְרָה רוּחָךְ לְעַם אֶרֶץ,
וְאִישִׁים פַּאְרָה



וּגְמֹל מְהֵרָה טוֹב, וְעַל רַע אַחְרָה תַּגְמוּל, וְכָל הַיּוֹם
שְׁלוֹמִים שַׁחְרָה.



וּזְנַח רְעוּת רוּחַ, וְהָסֵר עֶבְרָה מִלִּבְּךָ, וּבְרַח לְךָ
מִתִּגְרָה.



מֵרִיב לְשׁוֹנוֹת עוּף כְּבַעַל אֶבְרָה, וּבְבֵית כְּלוּא פִיךָ
לְשׁוֹנָךְ אָסְרָה.



וּשְׂנָא שׁלשָׁה: פֶּה מְדַבֵּר סָרָה, וּדְבַר לְשׁוֹן פּוֹתֶה,
וְדִבְרֵי זָרָה.



מַהְלַל עֲמִיתָךְ בֵּין אֲנָשִׁים בַּשְׂרָה וּבְלִבְּךָ מוּמוֹ וְדָפְיוֹ
קָבְרָה.



כַּפֵּר לְרֵעֶךָ מְרִי וַעְשָׂרָה, כִּי אִישׁ אֲשֶׁר לֹא יֶחֳטָא לֹא
נִבְרָא.



אַל תַּעְלִים מֵאָח לְעִתּוֹת צָרָה וַעְנֵה מְדֻכָּא יוֹם בְּשִׁמְךָ
יִקְרָא.



טוֹבָה לְאִישׁ שׂוֹנֵא בְּפֶה דַבֵּרָה וּבְקִרְבְּךָ מִמַּעֲשָׂיו
הִשְׁתַּמְרָה,



וַאְשֶׁר לְךָ יַעַשׂ בְּלִבָּךְ נָטְרָה, וְּבְעֵת אֲשֶׁר תִּמְצָא
לְהָרְגוֹ–מַהְרָה.



מַמְרֵה גְבִיר, יָדוֹ בְעַמּוֹ עָצְרָה – חָטָא, וְאֶת נַפְשׁוֹ לְמָוֶת
הֶעְרָה.



לִפְנֵי מְלָכִים חֶלְקְךָ הַכִּירָה וּשְׁבָה בְשֵׁפֶל – תַּעֲלֶה אֶל
מִשְׂרָה.



בִּיקָר אֱכֹל, וּשְׁתֵה– וְלֹא אֶל שָׁכְרָה, וּלְבַשׁ לְשֵׁם, וּקְנֵה
לְהָדָר דִירָה.



לִמְלַאכְתְּךָ, בִּשְׁכַב עֲצֵלִים, עוּרָה שַׁחַר, וּמָתְנַיִם לְעָמָל
חָגְרָה



וּבְסַחֲרָךְ – מִסְחַר נְקִיִּים סַחְרָה וּבְעֵת אֲשֶׁר תָּחֵל בְּדָבָר
– גָּמְרָה.



בִּפְנֵי אֱנוֹשׁ וּבְמִפְעֲלוֹתָיו תּוּרָה סוֹדָיו, וְאִם נַפְשׁוֹ
עֲלֵיהֶם סָגְרָה.



אַל תַּאֲמֵן כָּל אִישׁ וְלֹא כָּל אִמְרָה, וּדְבַר אֲשֶׁר
תִּשְׁמַע–בְּלִבָּךְ חָקְרָה.



הַחְרֵשׁ בְּהַצּוֹת סוֹרֲרִים עַל דִּבְרָה יֹאמְרוּ לְךָ הוֹכַח עֲלֵי
אִישׁ הַמְרָה



הַמְעֵט אֲמָרֶיךָ וְאֶת כָּל סִפְרָה פַּלֵּס, וְעֵת תֹּאמַר –
בְּמִשְׁקָל אָמְרָה



הָקֵץ, בְּנִי, וּשְׁתֵּי שְׁכוּנוֹת שׁוּרָה בָעִיר: בְּזֹאת מִסְפֵּד
וּבָזֹאת זִמְרָה.



רוֹאֶה בְקִרְיָה צָהֳלָה וַחְבוּרָה, יָגוֹן וּפֵרוּד אַחֲרֵיהֶם גּוּרָה.



רוֹאֶה אֲשֶׁר יָגַע וְיָדוֹ עָשְׁרָה, אִם לֵב לְךָ – עָלָיו תְּהִי בָךְ
יִתְרָה.



רוֹאֶה בְדוֹרוֹ תַם וּבַעַל יִשְׁרָה, קַנֵּא וְלַעְשׂוֹתָךְ כְּמוֹתוֹ
נָהְרָה.



רוֹאֶה בְעַפְעַפֵּי חֲבֵרוֹ שַׂעְרָה, אֵיךְ תַּעֲלִים אִישׁוֹנְךָ עַל
קוֹרָה?



כּתְפֹר קְרָעִים – אֶת זְמַנָּךְ תָּפְרָה, כִּצְרֹר זְהוּבִים– עַל
חֲבֵרָךְ צוּרָה,



כִּקְשֹׁר עֲדָיִים – אֶת עֲצָתָךְ קָשְׁרָה, כֶּאְסֹר מְלָכִים – סוֹד
לְבָבָךְ אָסְרָה,



כִּשְׁבֹר חֲרָשִׂים – יִצְרְךָ שַׁבֵּרָה, כִּנְעֹר שְׂמָלוֹת – מִמְּרִי
הִתְנַעְרָה,



כַּעְבֹר שְׂעָרוֹת – צוֹק בְּחָכְמָה עָבְרָה, כִּפְתֹר חֲלוֹם פִּלְאֵי
עֲלוּמוֹת פָּתְרָה,



כִּבְחֹר בְּדֹלַח – אִישׁ לְחֶבְרָה בָּחְרָה, כִּגְדֹר עֲלֵי צֹאן – עַל
נְוָתָךְ גָּדְרָה,



כֶּאְצֹר פְּרִי קַיִץ – מְלִיצוֹת אָצְרָה, כִּצְבֹר עֲפַר אֶרֶץ –
מְזִמּוֹת צָבְרָה,



כִּגְזֹר סְפִינָה יָם – גְּדוֹלוֹת גָּזְרָה, כִּבְתֹר אֲרִי שֶׂה – לֵב
מְשַׂנְאִים בָּתְרָה



כִּקְצֹר יְדֵי מֵת – יַד מְרֵעִים קָצְרָה וּרְדֹף רְצוֹן הָאֵל כְּצִיר
עַל בִּכְרָה



וּכְמוֹ מְתֵי תִקְוָה לְאֵל שַׂבֵּרָה וּכְמוֹ בְנֵי עֶלְיוֹן אֱמֶת
סַפֵּרָה.



שָׁלְמוּ מְאַת בַּיִת בְּחִבּוּר שִׁירָה, בָּהֶם מְאַת מִצְוָה,
כְּשֶׁמֶשׁ זָהְרָהּ.



אִם נַפְשַׁךָ עַל נַפְשְׁךָ יָקָרָה שִׂימֵם לְטוֹטָפוֹת וְלָאַזְכָּרָה



וּרְחַשׁ בְּטוּרֵיהֶם לְרֹאשׁ אַשְׁמוּרָה, יִהְיוּ לְלִוְיַת חֵן לְךָ,
וַעְטָרָה!





תַּמָּה הָאַרְגוּזָה וְיֵשׁ בָּה נג בָּתִּים
