


	
	[דבריך בתוך]
	







פראנעה ען דלך ובעת אליה מא הצר

והשיב לו (הנגיד) על זה ושלח לו מה שנזדמן (בכיסו)




U – – – U – – – U //– – U – – – U – – – U – –



דְּבָרֶיךָ בְּתוֹךְ לִבּוֹת נְטוּעִים // וּמַדֶּיךָ בְּתַחְבּוּלוֹת שְׂרוּעִים,

וְשִׁירֶיךָ שְׁקוּלִים וַחֲזָקִים // וּמִשֶּׁבֶר וְעִנְיָן רַע מְנוּעִים,

וְאַתְּ מוֹשֵׁל עֲלֵי מִלִּים טְהוֹרִים // כְּסַפִּירִים, וְעִנְיָנִים יְגֵעִים,

וְאַתְּ מוֹלֵךְ עֲלֵי לִבּוֹת חֲכָמִים // מְחֻכָּמִים, וּמִדַּעַת שְׂבֵעִים,

וְעַל פִּיךָ, לְדוֹדֶיךָ, נְשִׁיקוֹת, // וּבִימִינָךְ, לְצָרֶיךָ, פְּצָעִים,

וְגִדּוּפִים לְךָ – נָגְעוּ אֱלֵי לֵב // זְמַן בּוֹגֵד כְּמוֹ חִצִּים תְּקוּעִים,

בְּגָמְלוֹ רַע לְכָל מֵטִיב כְּמוֹתָךְ, // וְגָמְלוֹ טוֹב לְרֵעָיו הַמְּרֵעִים,

וְרָעָתוֹ אֲרֻכָּה עַד שְׁמִטּוֹת // וְעַד יוֹבְלוֹת, וְטוֹבָתוֹ – רְגָעִים,

וְשַׂמְתָּ ‘חֵטְא וְעָווֹן’ שֵׁם זְמַנָּךְ, // וְשֵׁם יָמִים הֲסִבּוֹתָ ‘פְּשָעִים’

וְיֵשׁ רַבִּים אֲשֶׁר זַכּוּ וְצַחוּ – // וּמִטּוֹבָה, כְּמוֹתֶךָ, גְּרוּעִים,

זְמַן הֵרַע וְלֹא אֵחַר לְךָ, כִּי // זְמַן יִפְרַע בְּנִשָּׂאִים פְּרָעִים.

וְאִם חֶטְאָךְ יְבַקֶּשְׁךָ – חֲמָסָךְ // עֲלֵי יֵצֶר וְרַעְיוֹנִים נְגוּעִים!

וְעוֹד הַיּוֹם וְיִפָּתְּחוּן סְפָרִים // וְיִבָּדְלוּ חֲסִידִים מִן רְשָׁעִים

וְעַד לֹא בָא הֲיָרַע הַזְּמַן לָךְ // וְיֶשׁ לִי הוֹן וְיַד נָדִיב וְשׁוֹעִים!

אֲנִי זָר מִקְּהָת, אִם לֹא אֲשַׁו לָךְ // תְּנוּכֵי הַזְּמַן הָרַע רְצוּעִים!

וְנָכְרִי מִמְּרָרִי, אִם יְהִי לִי // רְכוּשׁ – וּזְמַן יְהִי שַׁלִּיט בְּרֵעִים!

לְקַח נָא מִשְּׂרוּקָיו לָךְ סְמָדַר // וְעוֹד תֹּאכַל עֲנָבִים מִנְּטִיעִים,

וְעוֹד תִּשְׁתֶּה עֲסִיס רִמּוֹן, וְתִשְכַּב // בְּמִטַּת שֵׁן וּמִקְלַעַת פְּקָעִים,

וְעוֹד תַּכֶּה, כְּמַכָּה הִכְּךָ, אֶת // זְמַנָּךְ, אוֹ תְּנַגֵּעַ נְגָעִים!

וְשַׂחְתָּ: 'אֵיךְ תְּהִי נִרְאֵה כְּאִישִׁים // אֲשֶׁר הֵמָּה בְּבוֹר יָגוֹן שְׁקוּעִים?

וּמַה תִּירָא וּמַה יֵּרַךְ לְבָבָךְ – // וְהַיָּמִים עֲבָדִים לָךְ יְדוּעִים!' –

הֲתָבִין כִּי יְמֵי מֵבִין־לְאַחְרִית, // וְאִם טֹבוּ, מְעַט הֵמָּה וְרָעִים!

וְאֵין טוֹב אִם אֱהִי שׁוֹכֵב, כְּפוֹתִים // בְּלִי לֵבָב, בְּבֵית מִדּוֹת רְבוּעִים!

אֲנִי שׁוֹתֵק – וְשֵׁמַע מַעֲשַׂי עַד // לְמֵרָחוֹק כְּשׁוֹפָרוֹת תְּקוּעִים;

וְאֶתְחַפֵּשׂ – וְלִי דָבָר וּמִפְעָל // אֲשֶׁר יִהְיוּ בְּדוֹר אַחְרוֹן שְׁמוּעִים;

וְאֶתְהַלֵּךְ בְּבִקְעָה עִם בְּנֵי אִישׁ – // וְיֶשׁ־לִי לֵב עֲלֵי רָאשֵׁי סְלָעִים!

וְעַנְוָתִי כְּמוֹ תֵדַע, וְאוּלָם // לְאַט אֶקְרַע בְּלֵב צָרַי קְרָעִים.

וְאַתְּ רֵעִי, וְאַהְבָתְךָ זְרוּעָה // בְּתוֹךְ לִבִּי כְּמוֹ אוֹרִים זְרוּעִים,

וְשִׁבְתִּי עִמְּךָ אֶתְאַו וְאֶבְחַר // עֲלֵי הַשְׁקֵט וּמִשְׁכָּב עַל יְצוּעִים,

וְשֶׁבֶת עִם חֲכַם לֵבָב וְנָבוֹן // לְנַפְשִׁי, מֵחֲיוֹתָהּ, טוֹב וְנָעִים!
