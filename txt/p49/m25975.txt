


	
	הֲתֵדַע אֶת
	








הֲתֵדַע אֶת פְּעָלַי לַחֲכָמִים, / וְלִזְקֵנִים בְּאָרְחוֹתָם תְּמִימִים,

וּבַחוּרִים מְתֵי יִרְאָה, וְטוֹבִים / בְּמַרְאֵיהֶם, וְלִצְעִירִים לְיָמִים? –

הֲכִינוֹתִי מְקוֹם מִשְׁתֶּה, וְשַׂחְתִּי: / ״עֲלוּ בַלּוּ יְמֵיכֶם בַּנְּעִימִים!

עֲלוּ, נָשִׂישׂ בְּטֶרֶם בּוֹא פְּרִידָה / וְאֵיד נָכוֹן לְצַלְעֵי כָל יְקוּמִים.

וְהַחֶבְרָה וְהַמִּחְיָה – אֲחָיוֹת, / וְהַפֵּרוּד וְהַמָּוֶת – תְּאוֹמִים!״

אֲזַי בָּאוּ אֱלֵי מַצָּע מְשֻׁבָּץ / בְּאַרְגָּמָן וּמַכְלוּלִים רְקוּמִים,

וְאֶל כֵּלִים, וְאֶל מִינֵי מְגָדִים, / בְּיַד שׁוֹעִים כְּמוֹתֵנוּ עֲרוּמִים,

עֲלֵי מַטַּע אֲהָלִים, בֵּין עֲנָפָיו / בְּנֵי תוֹר שׂוֹחֲקִים יַחַד וְהוֹמִים,

בְּיוֹם רָווּ בְּיֵין עָבִים וְשָׁכְרוּ / וְהוֹצִיאוּ צְפוּנֵיהֶם תְּלָמִים,

וְיָשַׁבְנוּ בְּמוֹשָׁב טוֹב, וְאֵין בּוֹ / כְּבַד לֵב, וַהֲפִיקוֹנוּ זְמָמִים.

וְשָׁלַחְתִּי לְבֶן יוּבַל, וּמִהַר / בְּכִנּוֹרוֹ וּמִנָּיו הַשְּׁלֵמִים,

וְתָמַכְתִּי בְּאֵם אֶקְדָּח פְּתוּחָה, / מְשֻׁקֶּדֶת, מְעֻלֶּפֶת לְשָׁמִים,

וְקָרָאתִי בְּנֵי בֵיתִי וְקָמוּ / לְבֶן שׂוֹרֵק וְנֶכֶד הַכְּרָמִים,

אֱלֵי עָשׁוּק מְאוֹר עוֹלָם בְּכַדּוֹ, / כְּתָמָר הָעֲשׁוּקָה בֵּין יְבָמִים,

וְהֵרִיקוּ מְעֻבֶּרֶת מְלֵאָה – / וְלֹא הָיְתָה מְלֵאָה בַעֲצָמִים,

וְהִתִּירוּ אֲסִיר־בָּצִיר, וְעָצוּר / שְׁנָתַיִם בְּלֵב כַּדִּים אֲטוּמִים,

וְהִנֵּה שָׂב, זְמַן שִׁפָּה בְּשָׂרוֹ, / אֲבָל עֵינוֹ מְאָדָּם כַּעֲלָמִים,

מְקֻטָּר – לֹא בְּאַבְקַת הַלְּבוֹנָה, / מְרֻקָּח – לֹא בְּרָאשֵׁי הַבְּשָׂמִים,

כְּמַרְאֵה אֵשׁ – וּמִשְׁתַּפֵּךְ כְּמַיִם, / דְּמוּת פֶּתִי – וּמֵסִית הָעֲרוּמִים,

וְכַבָּרָק וְיֵשׁ עִמּוֹ גְשָׁמִים – / וְיֵשׁ בָּרָק וְאֵין עִמּוֹ גְשָׁמִים.

שְׁפָכוּהוּ עֲלֵי גּוּפָם בְּלַיִל / וְיָצְאוּ מִלְּבוּשׁ חָשְׁכָּם עֲרֻמִּים,

וְלֹא יָכְלוּ, בְּעֵת נוֹסַף מְאוֹרוֹ / עֲלֵי אוֹר יוֹם, לְהַבִּיט אֶל מְרוֹמִים.

וְתָפְשׂוּ בוֹ – אֲבָל הִטָּם בְּחֵלֶק / שְׂפָתָיו, לֹא בְּחֹזֶק יָד וְאֵימִים,

וְדִבֶּר־בָּם – וְהוּא דוֹמֵם – וְהִרְקִיד / יְקָרִים כַּצְּבִי וּכְבֶן רְאֵמִים,

וְלִרְצוֹנָם בְּפֶתַע סָר תְּבוּנָם, / וְדַעְתָּם הָגְלְתָה הָגְלָת שְׁלוֹמִים.

וְרָאוּ דַל – וְלוֹ שִׁלְטוֹן עֲלֵי שַׂר, / וּפֶתִי – לוֹ מְלוּכָה עַל חֲכָמִים,

בְּמַרְאֵהוּ, וְטוּב טַעְמוֹ, וְרֵיחוֹ – / וְהָחְבָּאוּ בְּכוֹסוֹת הַשְּׁהָמִים,

וְאָדְמוּ מִפְּנִינִים עַל־יְדֵי מָךְ, / אֲסִיר הַכַּד, כְּמוֹ פָנִים אֲדֻמִּים,

כְּאִלּוּ אוֹר אֲשֶׁר לֻבַּשׁ, וְנֹגַהּ, / בְּרַקּוֹתָם מְדֻבָּקִים וְשׂוּמִים.

שְׁלָחוּהוּ עֲלֵי פִיהֶם בְּמוֹרָד – / וְעָלָה הוּא לְרֹאשׁ שָׂרִים וְרָמִים,

וְהִשְׁכִּיבָם לְעֵת עֶרֶב, כְּאִלּוּ / בְּמַקֶּבֶת בְּיַד יָעֵל הֲלוּמִים.

הֲקִימוֹתִים לְעֵת בֹּקֶר – וְאֵין בָּם / אֲשֶׁר יָקוּם וְרֹאשָׁם לֹא מְרִימִים,

כְּאִלּוּ אֵין לְרַגְלֵיהֶם בְּהוֹנוֹת / וְאֵין לָהֶם בְּצַוָּארָם גְּרָמִים.

וְאָמְרוּ לִי: ״אֱמֹר מִי אַתְּ וּמַה לָּךְ?״ / וְנִנְעָרוּ כְּנֵעוֹר מֵחֲלוֹמִים.

עֲנִיתִימוֹ: ״תְּנוּ אֹזֶן לְמִלִּים, יְעִידוּן בָּם מְיֻדָּעִים וְקָמִים,

וְישָּׁמְעוּ בְּכָל חוּצוֹת וְעַל גַּב / מְרוֹם קֶרֶת וְאֵין מַכְלִים וּמַדְמִים:

אֲנִי נֶכֶד קְהָת וּשְׁאֵר מְרָרִי, / מְתֵי הַשֵּׁם, מְטִיבֵי הַטְּעָמִים,

וּמֵאָבִי וּמִבֶּן־צוּף שְׁמוּאֵל / לְפָנִים הָעֳרוּ דָמִים בְּדָמִים,

וּבֵין אָבִי וּבֵין מֹשֶׁה נְבִיא אֵל / שְׁאֵר בָּשָׂר וְקִרְבַת הַלְּחוּמִים,



אֲנִי אֶקְרָא שְׁמוֹ ׳אָבִי׳, וְהוּא לִי / ׳בְּנִי׳ יִקְרָא בְּהִקָּבֵץ עֲמָמִים,



וּמִי יִמְצָא בְּיַחְסִי מוּם לְשִׁמְצָה – / חֲקֹר, תִּמְצָא אֱנוֹשׁ מָלֵא בְּמוּמִים!

וְלִי עֹשֶׁר וְכָבוֹד רַב, וְלָאֵל / לְבַדּוֹ עֹז וְתַעְצוּמוֹת עֲצוּמִים,

וּפִתְחוֹן פֶּה עֲלֵי פִיּוֹת עֲרֵבוֹת, / וּמַשָּׂא רֹאשׁ עֲלֵי רָאשִׁים כְּסוּמִים,

וּמַמְלָכָה עֲלֵי גִנְזֵי זְהָבִים, וּמֶמְשֶׁלֶת עֲלֵי גִנְזֵי בְּרוֹמִים,

בְּמוֹשָׁבִי חֲכָמִים מַחֲרִישִׁים, / וּמַשְׂכִּילִים, כְּאִלּוּ הֵם אֲשֵׁמִים,

וְיָקוּמוּ בְּפִיפִיּוֹת חֲסוּמוֹת / בְּעֵת עָבְרִי, וְאִישׁוֹנִים עֲצוּמִים,

וְיִוָּתְרוּ לְנֶגְדִּי נֶאֱלָמִים / חֲכָמִים לַעֲשׂוֹת חַיִל אֲלוּמִים,

וְאֶחְשֹׂף הַדְּבָרִים הַנְּגִידִים / וְהָיוּ עִנְיְנֵיהֶם נַעֲלָמִים,

וְאַטִּיף הַחֲרָזִים הַיְּקָרִים / וְאֵין לָהֶם בְּתֵבֵל אָח וְדוֹמִים,

וְעַל כָּל זֹאת, וְטוֹב מִזֹּאת וּמִזֹּאת, / בְּכָל יָכְלִי פְּעָלַי לֹא זְעוּמִים,

וְאִם אֶפְשַׁע בְּאֹנֶס אוֹ בְּנֶעְלָם – / יְכַפֵּר אֵל אֲנָסִים וַעֲלוּמִים!״
