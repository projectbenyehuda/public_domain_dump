


	
	שעה מני
	



שְׁעֵה מִנִּי / שמואל הנגיד

העברתי את המספרים שמאל נראה לי יותר ברור

וּפָשַׁט יַדַּיִר הַשַּׂר בַּמָּקוֹם נִקְרָא אַרְגּוֹנָה בַּקַּיִץ שֶׁל שְׁנַת תת"א [1041; 4801], וְהָיָה עִמּוֹ וַאצִל וּמֻוַּפַק, וּשְׁנֵיהֶם שָׂרֵי צְבָאוֹת יְדוּעִים מֵרָאשֵׁי הָאַנְדַּאלוּס, וְגָבְרוּ עַל אַרְגּוֹנָה וְהָרְגוּ שַׂר הָעִיר. אַחַר כֵּן נָסְעוּ מִשָּׁם אֶל מָקוֹם נִקְרָא שָׂמַנְתִּאן וְגָבְרוּ עַל רֹב הַמִּגְדָּלִים אֲשֶׁר שָׁם. וְיָצְאוּ אַחֲרֵי כֵן הַמַּחֲנוֹת לִקְרָאתָם וְהָיְתָה בָּם יַד ה' וְנֶהֶרְגוּ וַאצִל וּמֻוַּפַק. וּבָרַח הַשַּׂר יַדַּיִר עַד אֲשֶׁר סֻגַּר בְּקָרְטָבָה וּנְשָׂאוּהוּ מִשָּׁם עַד אֲשֶׁר שָׂמוּהוּ אָסוּר בְּמִגְדַּל מֻנֶּקַר. וְאָמַר אֲדוֹנִי אָבִי, מְסַפֵּר בְּמַה שֶּׁקָרָה לוֹ וּמַה שֶׁנִּתְנַהֵג הוּא בוֹ, מְשַׁבֵּחַ לָאֵל עַל מַה שֶׁזִּמֵּן לוֹ מִזֶּה בַּנִּצוּחַ הֶחָזָק.



שְׁעֵה מִנִּי, עֲמִיתִי וַחֲבֵרִי,    וְהַט אֹזֶן וְלֵב אֶל מַאֲמָרִי.



הֲלִי תַפְחִיד בְּמִלִּים לֹא נְכוֹנִים?  וְאֵיךְ אֶפְחַד – וְצוּר יִשְׁעִי וְאוֹרִי!



עֲמִיתִי אַתְּ, וְאִם תַּרְבֶּה עֲלִילוֹת – תְּהִי אֶצְלִי כְּאַכְזָרִי וְנָכְרִי,



וְאֵין טוֹבָה לְמִסְתּוֹלֵל בְּבֵיתִי,   וְאֵין שָׁלוֹם לְמִתְעוֹלֵל בְּעִירִי.



וְאִם לֹא תַעֲזֹב רָעָה בְּפִיךָ    וְתִתְאַמֵּץ כְּבֵית מֶרִי בְּמֶרִי – 5



פְּנוּ אֵלַי, מְיֻדָּעַי, וְשִׁמְעוּ,    וּבֹרוּ אֶת דְּבָרוֹ מִדְּבָרִי:



יְרִיבֵנִי עֲלֵי חֶבְרַת מְלָכִים –    עֲנִיתִיו: זֹאת מְנָת חֶלְקִי וְתוֹרִי.



וְהִפְחִיד מֵחֲרוֹן אַפָּם וְקִצְפָּם –  הֲשִׁיבוֹתִיו: בְּאֵל מַחְסִי וְשִׂבְרִי.



וְשָׂח: מַה לָּךְ וְלִקְרָבוֹת? עֲנִיתִיו:  כְּבָר נִגְזַר מְקוֹם מוֹתִי וְקִבְרִי,



וְהָאֵל, הַמְבַשֵּׂר לִי עֲלֵי יַד    שְׂרָפָיו בַּחֲלוֹם, יִהְיֶה בְּעֶזְרִי. 10



וְאִם אֶמַּק, בְּחַטָּאתִי – הֲלִי יָד?  יְבוֹאֵנִי אֲשֶׁר נִכְתַּב בְּסִפְרִי!



וְיֵשׁ אָדָם אֲשֶׁר לֹא שַׂר בְּחַיָּיו –  כְּתִבְנִי מֵת בְּלֹא עִתּוֹ, וְזִמְרִי,



וְיֵשׁ אַחֵר אֲשֶׁר לֹא בָא בְּקֶרֶב   מְרִיבָה אוֹ קְרָב – נִטְמַן כְּמִצְרִי,



וְיֵשׁ נִסְפֶּה בְּלֹא מִשְׁפָּט, וְהָיָה   בְּעוֹדוֹ מָךְ וְרָעֵב וַעֲרִירִי.



חֲדַל מִנִּי מְעַט, אוּלַי אֲבַלֶּה    בְּטוֹב יָמַי וְעַל רֹאשִׁי פְּאֵרִי, 15



וְאֶשְׁתֶּה מֵי מְנוּחוֹת מִבְּאֵרִי,   בְּכוֹס יֶשַׁע, וְנוֹזְלִים מִן יְאוֹרִי!



הֲטוֹב לִי כִּי אֱהִי מוֹאֵס בְּכוֹסִי   וְלֹא אֶשְׂמַח בְּגוֹרָלִי וּפוּרִי?



וְאִם תָּשִׁית לְאַחְרִיתִי לְבָבָךְ,   וְלֹא יִשְׁוֶה לְךָ הוֹדִי וְעָשְׁרִי,



לְמַעַן כִּי בְּכָל שָׁנָה בְּמַיִם    וְאֵשׁ אָבוֹא, וּמֵהֶם רֹב יְקָרִי –



הֲלֹא תִירָא כְּמוֹתִי אַחֲרִיתָךְ,   וְאַתְּ שָׁלֵו בְּתוֹךְ בֵּיתָךְ וּבָרִיא! 20



אֲנִי אֶבְטַח בְּאֵל הִפִּיל אֲשׁוּרַי   בְּתוֹךְ פַּחִים טְמָנוּם לַאֲשׁוּרִי.


  


בְּיוֹם בָּא צַר לְעִיר מבְצָר, וְטָבַח  בְּתוֹכוֹ שַׂר כְּמוֹ עֵגֶל וְכִמְרִיא,



וְצַר זֶה מִשְּׁאֵר מַלְכִּי – וְרָעַת   רְחוֹקִי תֶחֱסַר מֵרַע שְׁאֵרִי.



וְהָיוּ שָׁם שְׁנַיִם מִנְּסִיכֵי     סְפָרַד עִם גְּדוּדֵי הַצְּמָרִי,



וְלָכְדוּ עִיר, וְהָלְכוּ אֶל מְצָדוֹת   לְשַׁדּוֹתַם כְּמוֹ קֶטֶב מְרִירִי, 25



וְיָצָאנוּ לְהַדִּיחוֹ – וְיָצָא     אֱלֹהִים וֶהֱנִיאוֹ בַעֲבוּרִי,



וְרֹאשׁ אֵלוּל עֲלֵי רָאשֵׁי סְלָעִים   מְצָאנוּם, לֹא מְקוֹם עֵשֶׂב וּפֶרִי,



וְהוֹרַדְנוּם אֱלֵי אֶרֶץ כְּמוֹ תוֹא   אֲשֶׁר נָגַע אֱלֵי מָרוֹם וְהִמְרִיא,



וְנָפוֹצוּ לְאַרְבָּעָה חֲמִשָּׁה     כְּמוֹ זַיִת אֲשֶׁר נָקַף שְׂכִירִי,



וְחִלַּלְנוּם שְׁנַיִם מוּל שְׁלשָׁה    כְּמוֹ נוֹקֵד כְּתָב בִּסְגוֹל וְצֵרִי, 30



וְהִכִּינוּם בְּמַכָּה כִלְּתָה אֶת    סְגֻלַּת עוֹג וְסִיחוֹן הָאֱמוֹרִי,



וְהֻכּוּ הַנְּסִיכִים הַשְּׁנַיִם –     וְזֶה תַגְמוּל אֲשֶׁר הוֹלֵךְ בְּקֶרִי,



וְתַמּוּ כֹל – לְבַד תִּשְׁעָה אֲנָשִׁים  אֲשֶׁר נָסוּ וְהוּא עִמָּם עֲשִׂירִי,



וּבֻקַּשׁ עַד אֲשֶׁר הוּבַל כְּמוֹ שַׁי   וְכִתְשׁוּרָה אֱלֵי מַלְכִּי בְּתִשְׁרִי.


  


יְדִידִי, בַּעֲדִי קָם צוּר בְּצָרוֹת,   וְעַל כָּכָה תְּהִלּוֹתַי לְצוּרִי, 35



וְשִׁירוֹתַי לְאֵל, חָזָה מְגוֹר צַר   בְּתוֹךְ לִבִּי, וְהֵפִיג אֶת מְגוֹרִי,



וְזִמְרוֹתַי לְרוֹפֵא, שָׂם צֳרִי עַל   מְקוֹם הַצִּיר, וְהִדִּיחַ לְצִירִי.



וְאוֹמֵר לִי: הֲתוּכַל אַתְּ לְהַלֵּל?   הֲשִׁיבוֹתִיו: אֲנִי דָוִד בְּדוֹרִי!



וְעָנָנִי: הֲשָׁאוּל בַּנְּבִיאִים?    עֲנִיתִיהוּ: יְרֻשָּׁה מִמְּרָרִי



וּמֵאַסִּיר וְאֶלְקָנָה וְאָסָף     וּמִישָׁאֵל וְאֶלְצָפָן וְסִתְרִי! 40



וְאֵיכָה לֹא תְּהִי שִׁירָה סְדוּרָה   בְּפִי לָאֵל אֲשֶׁר גָּהָה מְזוֹרִי?



וּמֵהוֹרַי יְדוּתוּן הַמְהַלֵּל     לְנַצֵּחַ, וְיָצָאתִי לְהוֹרִי.



לְאֵל עָלַי לְהַמְתִּיק סוֹד בְּשִׁירִי –  וְלִי עָלָיו לְמָרֵר לֵב מְצֵרִי,



וְהוּא עָרֵב לְחַלֵּל שׁוֹרְרַי – וַ-   אֲנִי עָרֵב לְחַלּוֹתוֹ בְּשִׁירִי,



וְלִפְעֹל לוֹ כְּמוֹ שָׂכִיר בְּכָל יוֹם –  וְהוּא יִתֵּן בְּיוֹמִי אֶת שְׂכָרִי! 45
