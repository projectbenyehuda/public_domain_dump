


	
	זמן לאנשים
	



וגזא אעזא אללה פי שתי אלמורך אלי אלגרב תם אלי ריה פפתח אללה פיהא פתוחא גלת אלנעמה בהא ואתפק אן תופי אבנא כאלתי חיניד ולמא עלם בדלך כאטבני מעזיא וואצפא למא כאן פי תלך אלגזאה מן אלנצר ואלצהור בפצל אללה תעאלי פקאל



ויצא למלחמה, יוקירהו האל, בחרף באותו תאריך (תתי"ו 1056) אל המערב ואחרי כן אל ריה, וכבש האל בה כבושים, רבו הנעימות לרגלם. וקרה שמתו שני בני דודתי בזמן ההוא וכאשר נודע לו הדבר כתב לי (אגרת), מנחם ומתאר מה שהיה במלחמה ההיא מן הנצחון והתהלה בחסדי האל יתעלה, ואמר




זְמַן לָאֲנָשִׁים יַשְׁקְ אֶת כּוֹס וְקֻבַּעַת / מְלֵאוֹת בְּתַרְעֵלָה מְרַוָּה וּמַשְׂבַּעַת,



וְיָמִיר בְּזָדוֹן רַב וְעוֹדֵף בְּטוֹבוֹתָיו, / אֲשֶׁר יַעֲשֶׂה מִשְׁגֶּה, בְּחֹסֶר וּמִגְרַעַת,



כְּאִלּוּ בְּנוּם יָחֹן בְּרָכוֹת, וְיִתְנֶחָם / בְּהָקִיץ, וְיוֹרֵם אֶל עֲלוּקָה בְּמִקְלַעַת



זְמַן דַּק וְחָרוּם בַּנְּדִיבוּת, וְאִם יַעַשׂ / נְבָלָה – יְבִיאֶהָ מְלֵאָה וְשׂוֹרָעַת.



חֲדַל מִזְּמַן, דּוֹמֵם בְּבוֹא טוֹב, וּמַשְׁמִיעַ, / בְּבוֹאָם יְמֵי רַע, קוֹל תְּרוּעָה וּמֵרַעַת,



זְמַן הַמְחַשֵּׁב עִם נְדִיב לֵב חֲצִי גֵרָה / לְדַעַת, וְנוֹתֵן רַב לְנָבָל בְּלִי דַעַת,



וּבוֹחֵר אֲהוּב נֶפֶשׁ לְהָמִית, וְחָבֵר טוֹב / לְהָנִיד, וְנֶטַע לַעֲקֹר אַחֲרֵי טַעַת,



וְקֹב אֶת שְׁאוֹל עַל כִּי תְּבַלַּע כְּבַלַּע אֶת / יְרִיעָה, וְתָגֹל כַּסְּפָרִים, וּבוֹלַעַת



יְדִיד אֵם, תְּמוֹל הָיְתָה שְׂמֵחָה בְּיִפְעָתוֹ / וְהַיּוֹם לְרָעָתוֹ כְּשֶׁטֶף מְדַמַּעַת,



וְדָוָה בְּלֵב נִשְׁבָּר לְשִׁבְרוֹ וּמוֹרָשִׁים / אֲשֶׁר אֵין מְתֹם בָּהֵם, וְנֶפֶשׁ מְנֻגָּעַת.



בְּנִי, אִם זְמַן תָּקַף בְּנֵי אַחֲיוֹת אִמָּךְ / בְּיָד, לֵב דְּבֵקֵי־לֵב מְסַלָּה וְקוֹרַעַת



וְחָבַשׁ יְחִידָתָךְ בְּחִתּוּל מְעַט אֶתְמוֹל, / וְהִיא נָא בְּרֹב מָזוֹר כְּתוּתָה וּמוּדַעַת



כְּמוֹ כֵן זְמַן מַכְאִיב יְחַבֵּק לְאִישׁ בִּשְׂמֹאל / טְהוֹרָה וְיַרְחִיקוֹ בְּיָמִין מְצֹרַעַת



וְעַל כֵּן אֲשַׁבַּח אֶת חֲכַם לֵב אֲשֶׁר יִשָּׂא / שְׂאֵתוֹ עֲלֵי נֶפֶשׁ מְבִינָה וְיוֹדַעַת,



וְעַל כֵּן תְּנַחֵם נַפְשְׁךָ, בַּעֲבוּר תִּהְיֶה / כְּבוּדָה, וּבְרַע לֹא נְמִבְזָה וְנִכְנָעַת.



תְּמַהּ מִזְּמַן, לוֹ פֶה, בְּחֵךְ זֶה תְּמָרֵר פִּי / וְעֵין צוּף בְּמַכְתֵּשׁ פִּי בְּחֵךְ זֶה מְבַקַּעַת



וְאִם יַךְ אֱנוֹשׁ כַּפּוֹ לְכַפּוֹ בְּצָרוֹתָיו / יְמַהֵר וְיַעְדֶּה יַד יְמִינוֹ בְּטַבָּעַת.



בְּעֵת יַחֲבֹשׁ אוֹמֵר: פְּעָלִים מְחֻכָּמִים! וְעֵת יַחֲלִיא אוֹמֵר: פְּעֻלַּת מְשֻׁגָּעַת!



וְלּא כֵן אֲדַבֵּר, כִּי זְמַנִּי מְשֻׁלָּח מִן / אֱלֹהִים לְכָל נֶפֶשׁ נְגוּפָה וְנוֹשָׁעַת.



כְּמוֹ הֶאֱבַלְתִּי יוֹם יְגוֹנִי לְמוֹשִׁיעִי – / לְשׁוֹנִי מְרַנֶּנֶת בְּיִשְׁעוֹ וּמַבַּעַת,



אֲשֶׁר הֶעֱמִיד רַגְלִי בְּמִישׁוֹר וְאוֹיַבְתִּי, / בְּבֹץ מִבְּלִי מוֹצָא, נְתָנָהּ מְטֻבַּעַת,



וּבָנְתָה מְצָדוֹת בַּסְּלָעִים, לְהִמָּלֵט / בְּתוֹכָם בְּבוֹא מֵפִיץ, וּבָצְרָה וְנִשְׁבַּעַת



אֲשֶׁר לֹא תְבוֹאֵם יַד מְבַעֵת, אֲשֶׁר בָּקַע / זְבוּלָהּ בְּחֹזֶק־יָד פְּעָמִים וְלֹא נִבְעַת,



וְאוּלָם אֱלֹהִים לֹא יְסַגֵּר בְּעַד מִבְצָר / לְצַדִּיק, וְלֹא יִשְׁלַח תְּשׁוּעָה לְמַרְשַׁעַת



פְּתַחְנוּם בְּיָד כָּבְדָה מְאֹד לַעֲדַת חָנֵף / בְּבוֹא מֵאֲשֶׁר רוֹאָה בְּאַרְצָהּ וְשׁוֹמַעַת



וּבָאנוּ בְּמִבְצָרִים גְּבוֹהִים בְּשָׁמַיִם, / אֲשֶׁר מֵאֲנוּ מִבּוֹא אֲלֵיהֶם וּמִגַּעַת,



וְשַׂמְנוּ סְבִיבוֹת לָהּ נְצִיביִם וּמַצָּבָה, / לְכָל בָּא וְיוֹצֵא בָהּ מְנִיעָה וּמוֹנַעַת



וְשַׁבְנוּ בְּמֶלֶךְ רַב וְאִישִׁים נְקוּבֵי שֵׁם / וְשָׂרִים מְתֵי שֵׁמַע וְסָרִים לְמִשְׁמַעַת



לְקִרְיָה מְלֵאָה מֵרְנָנוֹת וְקוֹל זִמְרָה / וּמִצִּלְצְלֵי שָׁמַע, וְשָׁרָה וּמַשְׁמַעַת,



שְׂמֵחָה בְּבוֹאֶיהָ וְאוֹתָם מְשַׂמַּחַת / וּבָם שַׁעֲשׁוּעֶיהָ וְאוֹתָם מְשַׁעְשָׁעַת.
