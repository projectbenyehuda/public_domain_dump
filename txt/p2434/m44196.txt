


	
	הֲלָנֶצַח רַעְיָה
	




הֲלָנֶצַח רַעְיָה / צְבִיָּה נְקִיָּה / לְשִׁפְחָה וְגוֹיָה / חֶמְדַּת מַאֲוַיָּהּ

זְרוּיָה דְחוּיָה / בְּאֶרֶץ מַאְפֵּלְיָה / סוֹעֲרָה עֲנִיָּה / הוֹמִיָּה כָאֳנִיָּה

צוֹעֲקָה בּוכִיָּה / דִּמְעָתָהּ עַל לֶחְיָהּ / אֵיךְ צָר בְּיָד הוֹיָה / וּבִזְרוֹעַ נְטוּיָה

הִכָּה אוֹתָהּ בִּצְדִיָּה / וּבְחַיִל מַכָּה טְרִיָּה / עָשָׂה בָהּ פְּלִילִיָּה / עֲלֵי דָת וּפְלִילִיָּה

הִצִּיגָה גְּלוּיָה / בְּעֵרוֹם וְעֶרְיָה / שָׁת גַּפְנָהּ לִשְׁאִיָּה / שַׁמָּה שַׁעְרוּרִיָּה

וְעֶדְנָה גַּן רְוָיָה / לְמִדְבַּר אֶרֶץ צִיָּה / מְקוֹם תַּנִּין וְחַיָּה / מְעוֹן דַּיָּה וְאַיָּה

אֵל רַב עֲלִילִיָּה / דָּר בִּשְׁמֵי עֲלִיָּה / אֵיךְ עֵדָה הַמְּנוּיָה / בְּמִפְקַד עֲשִׂירִיָּה

סְגֻלָּה שְׁלִישִׁיָּה / שׁוֹמְרָה דָת שְׁלִישִׁיָּה / הָיְתָה כְעוֹטְיָה / עַל עֶדְרֵי נָכְרִיָּה

הֲתָמִיד יְפֵהפִיָּה / כְּאָמָה בְזוּיָה / שִׂים עָנְיָהּ תַּחַת מֶרְיָהּ / יוֹצֵר עוֹלָמִים בְּיָהּ

בַּשְּׂרָה בְּאֵלִיָּה / הֲשִׁיבָהּ לִנְוֵה צִבְיָהּ / הַשְׂבַּע בַּטּוֹב עֶדְיָהּ / הָשֵׁב עֶדְיָהּ וְחֶלְיָהּ

חִישׁ תִּהְיֶה בְּנוּיָה / בְּזִיוָהּ וּבְנוֹיָהּ / וּכְהֻנָּה וּלְוִיָּה / אֶל הַר הַמּוֹרִיָּה

בְּשִׁירִים תְּהַלֵּל יָהּ / בְּתוֹךְ רְחוֹבוֹת קִרְיָה / אָנוּ לְיָהּ, וּלְיָהּ / עֵינֵינוּ צוֹפִיָּה.
