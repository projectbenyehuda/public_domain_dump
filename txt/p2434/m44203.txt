


	
	אֵיךְ יִמְצָא
	



שיר המדבר על פרידת הנשמה / לאדם אשר להבל דמה.

סימן: אהה אבוי אהה לי. לחן: אשמחה כי אפתחה.






אֵיךְ יִמְצָא חֵן / בְּעֵין בּוֹחֵן / לֵב וְטוּחוֹת

אִישׁ כָּל חוּשָׁיו / בְּמַשְׂאוֹת שָׁוְא / וּמַדּוּחוֹת

וְסוֹף שִׁבְרוֹ / בְּתוֹךְ קִבְרוֹ / אֵין מְנוּחוֹת

רְדוּף רִמָּה / עֲדֵי חָרְמָה / לֶאְכֹל לַחְמוֹ.





הוֹן רַב יִכְסֹף / יָקָר יֶאְסֹף / בְּיָד רָמָה

אָוֶן יַגְבִּיר / וּבְלֵב יַחְבִּיר / תּוֹךְ וּמִרְמָה

יִטְמֹן רֶשֶׁת / יִדְרֹךְ קֶשֶׁת / יַחְפֹּר גֻמָּה

לְכָל אִישׁ דַּל / וְלֹא יֶחְדַּל / לִשְׁפֹּךְ דָּמוֹ.





הוּא לֹא יֵדַע / קַרְנוֹ נִגְדַּע / בְּאַף חֵמָה

וְזִיו הוֹדוֹ / גַּם כְּבוֹדוֹ / שָׂם לִכְלִמָּה

וְלֹא יִמְצָא / לוֹ כָל עֵצָה / גַּם מְזִמָּה

עֵת יֵרָתֵק / אָז יִנָּתֵק / כָּל זְמָמוֹ.





אֵיךְ הוּא הוֹלֵל / וְהִשְׁתּוֹלֵל / בְּלִי חָכְמָה

אֲשֶׁר חֻלָּל / וְהִתְגּוֹלָל / בְּמֵי אַשְׁמָה

וְאַחְרִיתוֹ / לְשַׁחֲתוֹ / בְּיַד דּוּמָה

בְּרֹב קֶצֶף / עֵת יֵאָסֵף / אֱלֵי עַמּוֹ.





בְּעֵת יִגְדַּל / חָלְיוֹ יִדַּל / עַל מִשְׁכָּבוֹ

עַד אֵין רוֹפֵא / וְאֵין מַרְפֵּא / לְמַכְאוֹבוֹ

הֲלֹא יִזְכֹּר / זְמַן יַעְכֹּר / אֶת מַצָּבוֹ

כְּאִישׁ שׂוֹנֵא / לוֹ מִתְאַנֶּה / בְּאַף זַעְמוֹ.





וְיִתְגַּבֵּר / בְּלִי אֵבֶר / הוּא לִפְרֹחַ

אוּלַי יִצֹּר / בְּעִיר מָצוֹר / שָׁם לִבְרֹחַ

אַךְ לֹא יַפְלִיט / כִּי אֵין שַׁלִּיט / הוּא בָרוּחַ

לִכְלֹא אוֹתוֹ / מִשַּׁחֲתוֹ / כִּי בָא יוֹמוֹ.





יִפְתַּח עֵינָיו / נֶגֶד פָּנָיו / אִישׁ מַשְׁחִיתוֹ

וּבְיַד הָדָה / חֶרֶב חַדָּה / לְעֻמָּתוֹ

לִבּוֹ יִתַּר / וְאֵין נִסְתָּר / מֵחֲמָתוֹ

עוֹדוֹ נֶגְדּוֹ / לְהַשְׁמִידוֹ / עֲדֵי תֻמּוֹ.





אֵיךְ לֹא יָשׁוּב / חֲמַת עַכְשׁוּב / מֵרָעָתוֹ

טֶרֶם בָּא יוֹם / אֵין לוֹ פִדְיוֹם / לְמַחְלָתוֹ

יוֹם בָּא יָמֵשׁ / מְאוֹר שֶׁמֶשׁ / לְחֶשְׁכָּתוֹ

וְיָרֵחַ / אֵין זוֹרֵחַ / עֵת הִדְמִימוֹ.





הֶבֶל דָּמָה / אִישׁ אֲדָמָה / רַב אֲנָחוֹת

מָלֵא עַוְלוֹת / תּוֹךְ מְגִלּוֹת / מְפֻתָּחוֹת

וְיִזָּכֵר / לִפְנֵי זוֹכֵר / כָּל נִשְׁכָּחוֹת

וִיהִי נוֹכַח / עֵת יִתְוַכַּח / יָשָׁר עִמּוֹ.





הָהּ מַה יַעַן / עַל מִי נִשְׁעָן / אִישׁ גּוּשׁ עָפָר

כִּי בָז לִדְבַר / אֵל, וְעָבָר / וּבְרִית הֵפַר

הֲלֹא יִדְאַג / וּמַר יִשְׁאַג / בְּקוֹל שׁוֹפָר

אֵיךְ הוּא נִגְרָשׁ / וְלֹא דָרַשׁ / טוֹב לְעַצְמוֹ.





לָזֹאת יָעִיר / נַפְשׁוֹ, יָאִיר / אֶת נְתִיבוֹ

יָסוּר מֵרָע / וְגַם יִקְרַע / סְגוֹר לִבּוֹ

יִשְׁאַל מֵאֵל / כְּדַל שׁוֹאֵל / לַהֲשִׁיבוֹ

יַעְטֶה בֹשֶׁת / לֵב אֵל יָשֵׁת / עֵת הִכְלִימוֹ.





יְהִי עוֹפֵף / לְהִסְתּוֹפֵף / בֵּית אֵל אָיוֹם

לַעְרֹךְ תָּמִיד / שִׁיר גַּם יַצְמִיד / לַיְלָה עַד יוֹם

בְּקוֹל תּוֹרָה / אָז מְהֵרָה / יִמְצָא פִדְיוֹם

טוֹב וְנָעִים / יִכָּתֵב עִם / חוֹשְׁבֵי שְׁמוֹ.
