מחברות עמנואל – מחברת עשרים וארבע / עמנואל הרומי

מחברת עשרים וארבע

אָמַר הַמְּחַבֵּר: בִּהְיוֹתִי בְּבֵית הַשָּׂר נֶטַע נַעֲמָן / הָיָה לִי רֵע גּוֹאֵל וְרַחְמָן / רָד עִם אֵל וְעִם קְדוֹשִׁים נֶאֱמָן / וּשְׁמֹו צִדְקִיָה / נִקְבְּצוּ בוֹ כָּל הַמַּעֲלוֹת לַחַיָּה / וְכָמוֹהוּ לֹא הָיָה לְפָנָיו / אִישׁ חֲמֻדוֹת וּבַיְשָׁן וְעָנָיו / נָקִי בְּכָל שִׁשָּׁה סְדָרִים / וְלוֹ יָד וָשֵׁם בְּכָל אַרְבָּעָה וְעֶשְׂרִים סְפָרִים / חָכָם בְּדִקְדּוּק הַלָּשׁוֹן / וּבְסוֹד הָעִבּוּר הָיָה הָרִאשׁוֹן / מַה נוֹסִיף לְדַבֵּר עוֹד מֵאֹדוֹתָיו / וְהוּא הִפְלִיא לַעֲשׂוֹת מִכָּל אֲשֶׁר הָיוּ בְדוֹרוֹתָיו / וַאֲהִבַנִי אַהֲבָה עַזָּה / וְאַהֲבַת כָּל אִישׁ זוּלַת אַהֲבָתוֹ הָיְתָה בְעֵינַי נְמִבְזָה / וְהָיוּ לוֹ שְׁנֵי בָנִים / בְּחֵיק הַיֳפִי אֱמוּנִים / הָיוּ נֶחְמָדִים לְהַשְׂכִּיל וְלָעֵינַיִם תַּאֲוָה / וְיָרְשוּ מֵאֲבִיהֶם הַחָכְמָה וְהָעֲנָוָה / הָיְתָה רוּח אֱלֹהִים בְּקִרְבָּם / וַאֲבִיהֶם אַהֲבַת נַפְשׁוֹ אֲהֵבָם / וַיְהִי הַיּוֹם וַאֲנִי הָיִיתִי עִם הַשָּׂר בַחֲצַר גִנַּת הַבִּיתָן / לִרְאוֹת הֲפָרְחָה הַגֶּפֶן וְאִם הַנֵּרְדְּ רֵיחוֹ נָתַָן / וּלְפָנֵינוּ סְעֻדַּת לִוְיָתָן / וְהִנֵּה אִישׁ עִתִּי וּבְיָדוֹ רִצְפָּה / בְּמֶלְקָחַיִם לָקַח לִבִּי וּמְזִיחַ שִׂמְחָתִי רִפָּה / בְּאָמְרוֹ הִנְנִי שָׁלוּחַ אֵלֶיךָ קָשָׁה / בִּבְגֹד הַזְּמָן בְּבַעַל נֶפֶשׁ קְדוֹשָׁה / וְהָרַס לְּעֶבְרָתוֹ הָאֶבֶן הָרֹאשָׁה / הַנּוֹדַעַת לַחֲסִידוּת וְלִקְדֻשָּׁה / קָבַר צִדְקִיָּה הֶחָסִיד וְהֶעָנָיו / בְּחֹדֶשׁ אֶחָד שְׁנֵי בָנָיו / וְתַחַת הֱיוֹתוֹ מֵעוֹדוֹ נוֹתֵן אִמְרֵי שֶׁפֶר / וְנִלְחָם אֵת קִרְיַת סֵפֶר / שָׂם שַׂק בְּמָתְנָיו וְהוּא יוֹשֵׁב בְּתוֹךְ הָאֵפֶר / וְעֶשְׁתֹּנוֹתָיו אֹבְדוֹת / וּפְנֵיהֶם וְכַנְפֵיהֶם פְּרוּדוֹת / וְלָכֵן שַׂמְתִּי פָנַי מוּעָדוֹת / לְהַגִּיד אֵלֶיךָ אֶת כָּל הַתְּלָאָה אֲשֶׁר מְצָאַתְהוּ /

וְהָרָעָה אֲשֶׁר הִדְבִּיקַתְהוּ / לְבַעֲבוּר תְּחָבֵּר אֵלָיו דִּבְרֵי קִינוֹת / כִּי אֵין שָׁר וּמֵלִיץ בְּכָל הַמְּדִינוֹת / אֲשֶׁר לֹא שָׁלַח אֵלָיו מִכְתְּבֵי יְגוֹנִים / וְלָמָּה יִהְיוֹּ מִכְתָּבֶיךָ אַחֲרוֹנִים / כְּשָׁמְעִי הַדָּבָר עָמַדְתִּי מַשְׁמִים / וְנָפְלוּ עָלָי אֵימִים וְקָרָאתִי שֵׁם הַמָּקוֹם הַהוּא עֵמֶק עָכוֹר / וְסָפַדְתִּי עֲלֵיהֶם כְּמִסְפֵּד עַל הַיָּחִיד וְהָמֵר עֲלֵיהֶם כְּהָמֵר עַל הַבְּכוֹר / וָאֶתְּנָה אֶת לִבִּי לְחַבֵּר מְלִיצוֹת / וְקִינוֹת נִמְרָצוֹת / יַבְכּוֹּ עֵינֵי הַלְּבָבוֹת / וְיַזִּילוּ דִמְעוֹת הָרוֹאוֹת בָּאֲרֻבּוֹת / וּלְדַבֵּר עִם הַזְּמָן דִּבְרֵי רִיבוֹת / וְכִרְאוֹת הַשָׂר מַחֲשֶׁבֶת לִבִּי / וִיֹּאמֶר אֵלָי לֹא כֵן אָבִי / אַחֲלַי לְפָנֶיךָ עֲשֶׂה כְאִישׁ נָבוֹן וְלֹא תְחַבֵּר דָּבָר מִדִּבְרֵי עִצָבוֹן/ רַק שִׂים לְנֶגְדְּךָ דִּבְרֵי הַחֲכָמִים / וְלֶקֶט מִסִּפְרֵי חָכְמוֹתָם לְשָׁמִים / וּמִגִּנְזֵי בְרוֹמֵיהֶם אֲמָרִים נְעִימִים / וְשַלֵּם לוֹ וְלַאֲבֵלָיו נִחֻמִּים בְּעִנְיָנִים חֲדָשִׁים / לְקוּחִים מִדִּבְרֵי הַקְּדוֹשִׁים / וּבִמְלִיצוֹת בְּתוּלוֹת / אֲשֶׁר לֹא הָיוּ מִמְּשׁוֹרְרֵי הַזְּמָן בְּעוּלוֹת / וְאָז יִהְיוּ חַיִּים לְמוֹצְאֵיהֶם / וְיִמְצְאוּ חֵן וְשֶׂכַל טוֹב בְּעֵינֵי רוֹאֵיהֶם / וְאָז תִּרְאֶה קִינוֹת זוּלָתֶךָ / וּמַה יִהְיֶה בֵין מְלִיצָתָם לִמְלִיצָתֶךָ / וּבֵין עֲצָתָם לַעֲצַתֶךָ / הֵם יְחַדְּשׁוּ הַיָּגוֹן בְּלֵב אֲבֵלָיו וִיעוֹרְרוּהוּ / וְאַתָּה אִם יִהְיֶה בִּלְבָבָם יָגוֹן אֶל הַשּׁוֹקֶת יְעָרוּהוּ / כְּשָמְעִי דִּבְרֵי הַשָּׂר יָשְׁרוּ בְעֵיָני / וּלְמַלֹּאת מִצְוָתוֹ שַׂמְתִּי פָנָי / וָאָקוּם וָאֲחַבֵּר מְלִיצָה נִמְרֶצֶת / מִתּוֹכְחוֹת מוּסָר מְקֻבֶּצֶת / תְּנַחֵם בְּטוּב לִקְּחָהּ כָּל לֵב אָמֵל / וְתִתֵּן חַיִּים לְמַרֵי נֶפֶשׁ וְאוֹר לְעָמֵל / וָאֶשְׁלָחֶנָּהּ אֵלָיו בְּיַד אִישׁ מֵאַנְשֵׁי הַחָכְמָה / וַיִתְבּוֹנֵן עָלֶיהָ וַתְּהִי אֵלָיו לְנֶחָמָה / וְאֵלֶּה דִּבְרֵי מְלִיצָתִי / הַשְּלוּחָה אֶל הַשָּׂר צִדְקִיָּה אִישׁ עֲצָתִי:

בְּיַד הַזְּמָן חֶרֶב פִּיפִיּוֹת / יְנַהֵג בָּנָיו וּבְנוֹתָיו כִּשְׁבוּיוֹת / הַפְּגָעִים יָחוּלוּ עָבְרוּ מַשְׂכִּיּוֹת / הָרוּחוֹת יִשְּׁבוּ וְלֹא עַל תַּאֲוַת בַּעֲלֵי הָאֳנִיּוֹת / אֵיךְ לֶב-אִישׁ לִשְׁקֹט עַל שְׁמָרָיו יַחְשֹׁב / וְהַחַיּוֹת רָצוֹא וָשׁוֹב / לֹא יִשְׁקְטוּ בַזְּמָן כִּי אִם רְשָעָיו / וְלֹא יַעַמְדוּ עַל הַחֹמֶץ כִּי אִם תּוֹלָעָיו / אָבִי אַל תִּדְאַג אִם נִהְיֵיתָ וְנָחֳלֵיתָ נוֹרָאוֹת / כִּי כַמָּה טוֹבוֹת צְפוּנוֹת בִּמְעֵי הַתְּלָאוֹת / וְכַמָּה אֲנַחְנוּ רוּאִים כִּי יִתְרַפְּאוּ הִגּוּפוֹת בַּחֳלָאִים / וְנָפַל בַּעֲצוּמָיו חַלְכָּאִים / וְאַל תֹּאמַר כִּי אֵין דָּן דִּינְךָ לְמָזוֹר / כִּי הִנֵּה הַכּוֹכָב יֵיְשִׁיר אַחַר הֱיוֹתוֹ נָזוֹר / וְהַשֶּׁמֶשׁ אַחַר אֲשֶׁר יֶעֱרַב יִזְרַח וְיַחֲזוֹר / וְהַלֵּילוֹת הָרוֹת מֵרָצוֹן אֵל אָיוֹם / וְלֹא נֵדַע מַה יֵּלֶד יוֹם / וְיֵש בָּאֵל תְּמוּרָה מִּכָּל אֲבֵדָה / וְאַחַר כָּל רוּחַ נוֹשֶׁבֶת מְנוּחָה וַעֲמִידָה / הֲיִתָּכֵן בִּתְנוּעַת הַגַּלְגַּל מִבְּלִי הַרְכַּבַת הַטְּבָעִים / וּבַעֲלִיָתוֹ וִירִידָתוֹ מִבִלְתִּי בִיאַת הַפְּגָעִים / עֵת לְהַשְׁלִיך יְגוֹנִים וְעֵת לִכְנוֹס / זֹאת הַכּוֹס אֵין מִשְּׁתִיָתָהּ מָנוֹס / הֵן לֹא הִשְׁקְךָ הָאֵל מִכּוֹס הַתַּרְעֵלָה קֻבַּעַת / רַק לְהַרְאוֹת לְכָל יוֹדְעֵי דַּעַת / עַד הֵיכָן כֹּחַ הַסֵּבֶל מַגַּעַת / לֹא יִבְחֵן הָאֵל כִּי אִם הָאֲצִילִים / וְכֵן בַּשָּמַיִם כּוֹכָבִים אֵין מִסְפָּר וַחֲיָלִים / וְאֵין הַלָּקוּת רַק לִשְׁנֵי הַמְּאוֹרוֹת הַגְּדוֹלִים / לֹא יִבְחַן הָאֵל הַסְּכָלִים קִצְרֵי יָד / וְכֵן אִלּוּ הָיָה הָעֹרֵב טוֹב לֹא נִמְלַט מִיַּד הַצַּיָּד / אָמְנָם הַשָּׁלֵם יוֹסִיף עַל שְׁלֵמוּתוֹ כְשֶׁיִצָּרֵף / וְכֵן יוּבָא הַזָּהָב בָּאֵשׁ וְלֹא יִשָּׂרֵף /אָבִי רְאֵה גַּם רְאֵה מַדּוּעַ יִבְכּוּ אָבוֹת עַל בָּנִים / אֲשֶׁר בְּחֵיק יוֹצְרָם אֱמוּנִים / אֵיךְ תֹּאמַר הַשָּׂר אָבַד נִצְחִי / וּבָנֶיךָ בְּתַעֲנוּג עוֹלָם וּבְשָׂשׂוֹן נִצְחִי / לָמָּה

הָיִיתָ נֵזֶר זְמַנֵּנוּ כְּאִישׁ נִדְהָם / וְכוֹכְבֵי שַׁחַרְךָ אָסְפוּ נָגְהָם / אַחֲרֵי אֲשֶׁר בָּנֶיךָ בְּחֵיק אַבְרָהָם / הַאִם חַיָּה אֲכָלָתַם / אוֹ הָיוּ כַּקַּשׁ אֵשׁ שְׂרָפָתָם / הַאִם כִּנְפֹל לִפְנֵי בְנֵי עַוְלָה נָפְלוּ בֶחָרֶב / אוֹ נָפְלוּ בְיַד חַיָּה לָמוֹ אָרֶב / הַאִם הָשְׁלְכוּ מִקִּבְרָם כְּנֵצֶר נִתְעָב / אוֹ הָיוּ לְחוּמֵי רֶשֶׁף וּמְזֵי רָעָב / מִי לְקָחָם כִּי אִם זֹאת הִיא הַמְּלִיצָה / אֲשֶׁר שָׁלַחְתִּי לְבַעַל הָעֲלִיצָה / עוּרָה כְבוֹדִי כִּי הִנֶּה הַסְּתָיו עָבַר וְהַגֶּשֶׁם אָזָל / וְהֵשִׁיב הַזְּמָן אֶת הַגְּזֵלָה אֲשֶׁר גָּזָל / שָׁמַעְתִּי וּבְכֵן יָבִיאוּ צַוָּארָם בַּעֲבוֹדַת הַקָּהָל וְהָיָה זֶה שָׁלוֹם / וְהָאֵל עִלַּת הָעִלּוֹת / הַנֶּאְדָּר בַּקֹּדֶשׁ וְנוֹרָא תְהִלּוֹת / וְלֹו נִתְכְּנוּ עֲלִילוֹת / הַקַּדְמוֹן אֲשֶׁר לֹא סָר מִהְיוֹת / הַבּוֹחֵן לִבּוֹת וּכְלַיּוֹת / יִשְׁמַע קוֹלֵנוּ מִבּוֹר תַּחְתִּיוֹת / וִיזַכֵּנוּ אֶל מֵי גֻּלּוֹת עִלִּיּוֹת / שֶׁפַע תּוּשִׁיּוֹת / כִּימֵּי עוֹלָם וּכְשָׁנִים קַדְמוֹנִיּוֹת / וְיַחֲזִיר עֲטֶרֶת קְהָלֵנוּ לְיָשְׁנָהּ / וּגְדוּדֵי הַשְּׂמָחוֹת יָבֹאוּ בָּאָרֶץ בָּא שָׁנָה / וּפְדוּיֵי אֲדֹנָי יְשׁוּבוּן וּבָאוּ צִיּוֹן בְּרִנָּה / בְּחַיֵיכוֹן וּבְיוֹמֵיכוֹן יֵאָמֵן / וּבְחַיֵּי דְּכָל בֵּית יִשְׂרָאֵל בַּעֲגָלָא וּבִזְמָן קָרִיב וְאִמְרוּ אָמֵן /


כִּשְׁמוֹעַ הַשָּׂר הַמְּלִיצָה הַהֲדוּרָה אֲשֶׁר יָפְיָהּ לֹא נֶחְקָר / הָיְתָה לוֹ אוֹרָה וְשִׂמְחָה וְשָׂשׂוֹן וִיקָר / וַיֹאמַר זֹאת הַמְּלִיצָה הִיא פֶרַח כָּל הַמְּלִיצוֹת אֲשֶׁר הִשְׁמִעְתָּנִי / אֶמְצָא חֵן בְּעֵינֶיךָ כִּי נִחַמְתָּנִי / בְּךָ יִמְצָא אִישׁ כָּל מַאֲוַיָיו / הַפּוֹרֵשׁ מִמְּךָ כְּפוֹרֵש מֵחַיַָּיו / יָשַׁבְתִּי עִם הַשָּׂר יָמִים וְשָׁנִים וְהוּא מִתְעַלֵּס בְּאַהֲבָתִי / וַיִּהְיוּ בְעֵינָיו כְּיָמִים אֲחָדִים בְּאַהֲבָתוֹ אוֹתִי / אַחֲרֵי כֵן גָּזַר הָאֵל בִּפְרִידָה / וְלָקַחְתִּי הַנְּסִיעָה לִי לְצֵידָה / וְהִפְקַדְתִּי הַשָּׂר בְּיַד אֵל עֵילוֹם / וָאֵלֵךְ מֵאִתּוֹ בְשָׁלוֹם.
