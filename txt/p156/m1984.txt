


	
	מחברות עמנואל
	




אָמַר עִמָּנוּאֵל הַמְחַבֵּר: בְּשׁוּבֵנוּ מֵהַמַּעֲרָכָה עִם הַשָּׂר הַמְאֻשָּׁר • הָיִיתִי בַעֲבוֹתוֹת אַהֲבַת הַגְּבִירָה נִקְשָׁר • וּלְדַבֵּר רַק מֵאוֹדוֹתֶיהָ בְּעֵינַי יִכְשָׁר • הָיָה חִשְׁקִי בָהּ אַמִּיץ וְאָיוֹם • אַךְ בָּהּ יָשֻׁב יַהֲפֹךְ יָדוֹ כָּל הַיּוֹם • וְכִרְאוֹת הַשָּׂר כִּי נִלְכַּדְתִּי בִמְצוּדָתָהּ • וְנַפְשִׁי נִקְשְׁרָה בְאַהֲבָתָהּ • אָמַר אֵלַי: הַיּוֹם הַזֶּה אָחַל חִשְׁקֶךָ • שְׁמַע מִמֶּנִּי דָבָר וְשָׂאֵהוּ בְחֵיקֶךָ • דַּע כִּי דְבַר הַגְּבִירָה אֲשֶׁר רָאִיתָ מָךְ מֵעֶרְכֶּךָ • אָמְנָם לָהּ אָחוֹת יָפָה כְפִי מַה שֶּׁנֶּאֱמַר • עָמַד טַעְמָהּ בָּהּ וְרֵיחָהּ לֹא נָמָר • טוֹבַת שֵׂכֶל וּצְנוּעָה • וְהַנַּעֲרָה טוֹבַת מַרְאֶה מְאֹד בְּתוּלָה וְאִישׁ לֹא יְדָעָהּ • נֶחְמֶדֶת לְהַשְׂכִּיל וְלָעֵינַיִם תַּאֲוָה • וְגָבְרָה עַל כָּל הַגְּבִירוֹת בְּמוּסָר וּבַעֲנָוָה • יָפָה כַלְּבָנָה וְכַשֶּׁמֶשׁ בְּהִירָה • וּבִמְלֶאכֶת הַשִּׁיר וְהַמְּלִיצָה זְרִיזָה וּמְהִירָה • לֹא יֻתַּן סְגוֹר תַּחְתֶּיהָ וְלֹא יִשָּׁקֵל כֶּסֶף מְחִירָהּ • הִיא מִתַּעֲנוּגֵי הַחֵשֶׁק וְהַיְדִידֹת אָחוֹר נָסוֹגָה • וּבְשׁוֹשַׁנֵּי הַנְּזִירוּת וְהַבְּדִידוּת סוּגָה • קָצָה בַחוֹשְׁקִים וּבִבְנֵי‑בְלִיַּעַל • וְלָהּ הַבִּרְכַּיִם אֲשֶׁר לֹא כָרְעוּ לַבָּעַל • וְהַפֶּה אֲשֶׁר לֹא נָשַׁק לוֹ • וְתִשְכֹּן בְאָהֳלָהּ מִבְּלִי לוֹ • לָהּ בִּרְכוֹת שָׁדַיִם מֵעָל • וְלֹא לָקְחָה מִן הַזְּמָן מִחוּט וְעַד שְׁרוֹךְ נָעַל • רַק לֶחֶם לֶאֱכֹל וּבֶגֶד לִלבֹּשׁ • וַתָּשֶׂם בְּלִבָּהּ אֶת רֶסֶן תַּאֲוָתָהּ לִכְבֹּשׁ • וְעָלְתָה בְחָכְמָתָהּ לִתְהִלָּה וּלְשֵׁם וּלְתִפְאֶרֶת • הִיא עַל עַמּוּדֵי הַשִּׁיר וְהַמְּלִיצָה גֻלָּה וְכֹתֶרֶת • וּבְפִיהָ לְשׁוֹן הַזָּהָב וְהָאַדֶּרֶת • וְלָצֵאת מיָּם הַתַּאֲוֹת אֶל יַבֶּשֶׁת הַנְּזִירוּת חוֹתֶרֶת • וְכָל שָׁר וּמֵלִיץ בְּעֶרְכָּהּ כְּגַנָּב נִמְצָא בַמַּחְתֶּרֶת • תַּחְשֹׁב צַחוֹת לְשׁוֹנָם לַעֲגֵי שָׂפָה וְלָשׁוֹן אַחֶרֶת • הִיא צִנְצֶנֶת הַמָּן • אֵין עֲרֹךְ אֵלֶיהָ בַזְּמָן • נִלְחֲמוּ בָהּ הַזָּלִים זָהָב מִכִּיס וְכֶסֶף בַּקָּנֶה יִשְׁקֹלוּ • אָבוּ נְשִׂיאֵי הָאָרֶץ לְקַחְתָּהּ לְאִשָּׁה וְלֹא יָכֹלוּ • וְלוּ הִטְּתָה אָזְנָהּ, כָּל יוֹצְאֵי שַׁעַר עִירָהּ יִמֹּלוּ • וְלוּלֵי תִשְׁכֹּן בְּסֵתֶר אָהֳלָהּ נָגֹלּוּ • כַּסֵּפֶר הַשָּׁמַיִם וְכָל צְבָאָם יִבֹּלוּ • וְלוּ רָאִיתָ הוֹדָהּ עָמַדְתָּ מִשְׁתָּאֶה • וְהָיִיתָ מְשֻׁגָּע מִמַּרְאֶה עֵינֶיךָ אֲשֶׁר תִּרְאֶה • וַיְהִי כְשָׁמְעִי נִבְהֲלוּ רַעְיוֹנַי • וַתִּדַּד שְׁנָתִי מֵעֵינַי • וְנִבְהַלְתִּי מֵחָשְׁקָהּ לְמִשְׁמַע אָזְנַי • וַיְהִי חִשְקָהּ כְּאֵשׁ בּוֹעֶרֶת עָצוּר בְּעַצְמוֹתַי • וַיִּתְּכוּ כַמַּיִם שַׁאֲגוֹתַי • לֹא תֶעֱרַב לִי כָל שִׂמְחָה • וְנֶהְפַּךְ שְׂשׂוֹנִי לַאֲנָחָה • וְנִלְאֵיתִי כַּלְכֵּל וְלֹא אוּכָל • עַד הָיִיתִי בְעֵינֵי הַשָּׂר סָכָל • וַיֹּאמֶר אֵלַי הַשָּׂר: הַהִגִּיעַ מִסִכְלוּתְךָ לַחֲשֹׁק בְּמַלְכַּת הַשָּׁמַיִם אֲשֶׁר שִׁמְעָהּ הִבְהִילָנוּ • מִי יַעֲלֶה לָנוּ הַשָּׁמַיְמָה וְיִקָּחֶהָ לָנוּ • חֲדַל לְךָ אָחִי פֶּן נִתְבַּזֶּה • וְאַל תּוֹסֵף דַּבֵּר אֵלַי עוֹד בַּדָּבָר הַזֶּה • וָאֹמַר אֶל הַשָּׂר בְּנֶפֶשׁ חֲרֵדָה • וְעֵינִי עֵינִי מַיִם יוֹרְדָה • אָנָּא הַשָּׂר אַבִּיר הַתְּעוּדָה • הַרְאֵנִי נָא אֶת כְבוֹדָהּ • כִּי הִנְנִי נִשְׁבַּע בִּגְאוֹנִי • וְחֵי הַחֵשֶׁק אֲשֶׁר בֵּינְךָ וּבֵינִי • כִּי אִם יַעֲלֶה לַשָּׁמַיִם שִׂיאָהּ • וּמִמַּעַל לְכוֹכְבֵי אֵל תָּשִׂים כִּסְאָהּ • שָׁם אֲצַוֶּה דִמְעוֹתַי • וּמִשָּׁם יוֹרִידוּהָ תַּחֲנוּנוּתַי • כִּי לִי שִׁירִים כַּבְּרָקִים יְרוֹצֵצוּ • וּכְעֵין נְחֹשֶׁת קָלָל יְנוֹצֵצוּ • וּמְלִיצוֹת נִמְרָצוֹת תַּעֲשֶׂינָה בְלֵב הַיְּעָלוֹת פְּרָצוֹת • וְתִקְרַעְנָה סְגוֹר לִבָּם יְקָרְבוּ כַתַּנּוּר לִבָּם בְּאָרְבָּם •


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:



לִי יִקְּדוּ שָׁרִים וְלִי יִכרָעוּ



עַל פִּי מְלִיצִים יַחֲנוּ יִסָּעוּ



אִתִּי בְּסוֹד שִׁירִים חֲדָשׁוֹת נִפְלְאוּ



לֹא נִבְרְאוּ מֵאָז וְלֹא‑נִשְׁמָעוּ



לִי שִׁיר יְנוֹבֵב הַבְּתֻלוֹת לִי זְמִיר



חֵשֶׁק דְּבָרָיו כָּל לְבַב יִקְרָעוּ



אִלּוּ שְׁמָעוּהוּ בְנוֹת חֵן גָּוְעוּ



אוֹ שָׁכְבוּ עִם כֹּל אֲשֶׁר פָּגָעוּ



לוּ הַמְּלָכוֹת שָׁמְעוּ מִפִּי זְמִיר



אָז נָפְלוּ תַחתַּי וְלֹא יִבְצָעוּ



אִלּוּ בְּצַר רוּחִי אֲמָרֵר בַּבְּכִי



אָז בַּבְּכִי כָּל הַיְקוּם גָּוָעוּ,



יִבָּקְעוּ כָל‑מַעְיְנוֹת דִּמְעָה עֲדֵי



אֶרֶץ וְשָׁמַיִם הֲכִי יִדְמָעוּ



לוּ לַנְּבָלִים אֶשְׁאֲלָה מַתָּן אֲזַי



לֹא קָפְצוּ יָדָם וְלֹא מָנָעוּ



עַל‑פִּי תְּנוּעוֹת מַאֲמָרַי נֹסְעִים



כָּל‑שֹׁמְעֵי שִׁירַי וְלֹא יִפְשָׂעוּ



כֵּן יִשְׂמְחוּ כֵּן יִדְאֲגוּ כֵּן יִשְׂחֲקוּ



כֵּן יִקְצְרוּ חֶסֶד וְכֵן יִזְרָעוּ



יִתְעַנְּגוּ מִזִּיו אֲמָרַי שֹׁחֲרַי



מִשֹּׁד זְמִירַי יִינְקוּ יִשְׂבָּעוּ



אֶדְלֶה פְנִינֵי מַאֲמָרִים יָקְרוּ



לוּ עָמְקוּ אוֹ בַתְּהוֹם טָבָעוּ



אִלּוּ בְנֵי‑קֹרַח כְּשִׁירַי שֹׁרֲרוּ



לֹא נִבְלְעוּ אָז כַּאֲשֶׁר נִבְלָעוּ



אָסָף וְהֵימָן אִם‑בְּדוֹרִי יִהְיוּ



לֹא נוֹדְעוּ בַשִּׁיר כְּמוֹ נוֹדָעוּ



לֹא נִכְּרוּ בַשִּׁיר וְאַף בַּל‑נִטְּעוּ



בַּל שֹׁרְשׁוּ כַיּוֹם וּבַל זֹרָעוּ



מַה לָּךְ יְדוּתוּן שִׁיר וְהֵימָן מַה לְּךָ



כִנּוֹר פְּנוּ אֵלַי וְהִוָּשֵׁעוּ



הָיוּ מְלִיצֵי הַזְּמָנִּים עִלְּגֵי



שָׂפָה בְּדֶרֶךְ שִׁיר כְּעִוְרִים נָעוּ



אֶדְרֹךְ אֲנִי עַל בָּמֳתֵי הַשִּׁיר וּבִי



יִתְהַלְלוּ שָׁרִים וּבִי נִשְׁבָּעוּ



הַדֹּרְכִים מִפְתַּן חֲסָדַי לַעֲלוֹת



עַד מַעֲלוֹתַי יִעֲפוּ יִיגָעוּ



עַל כֵּן לְכָל הֹרֵס וּמַעְפִּיל לַעֲלוֹת



אֹמַר: רְדוּ דוֹדִים וְהִכָּנֵעוּ



אַל‑תִּקְרְבוּ סוּרוּ וְאַל תִּגָּעוּ



שִׁירַי רְאוּ רָאוֹ וְאַל תֵּדָעוּ •


  


כִּשְׁמוֹעַ הַשָּׂר הַשִּׁירָה נִפְלָא עַל נְעִימוּתָהּ • וַיֹּאמֶר לֹא רָאִיתִי כְתַבְנִיתָהּ • אִם תַּחְפֹּץ לִרְאוֹת הַיַּעֲלָה וְתִפְאַרְתָּהּ • הִנֵּה הִיא יוֹצֵאת יוֹם יוֹם וְעוֹלָה אֶל בֵּית תְּפִלָּתָהּ • הַשׁכֵּם בַּבֹּקֶר וְשָׁם תִּראֶה זֹהַר יִפְעָתָהּ • כְּשָׁמְעִי זֹאת הִתְנַחַמְתִּי • וְאַשְׁמֻרוֹת קִדַּמְתִּי • וְיָשַׁבְתִּי פֶתַח הַשַּׁעַר • וְרַעְיוֹנַי אָחֲזוּ שָׂעַר • וְלִבִּי כְּאֵשׁ תִּבְעַר יָעַר • עַד אֲשֶׁר תַּעֲבֹר הַיָּפָה וְאֶרְאֶנָּה • אִם הִיא כְּלִילַת יֹפִי כַּאֲשֶׁר יֹאמְרוּ מִמֶּנָּה • עוֹד מְעַט וְהִנֵּה • קוֹל הֲמֻלָּה כְּקוֹל מַחֲנֶה • כָּל הַחוֹשְׁקִים עֵת בּוֹאָהּ יְקַוּוּ • וְקָם כָּל הָעָם וְהִשְׁתַּחֲווּ • וָאֶשֶּׂא עֵינַי וָאֶרְאֶה אֶת הַמַּרְאָה הַיְקָרָה • וְהִנֵּה חֶבְרַת גְּבִירוֹת וּבְתוֹכָם נַעֲרָה • יָפָה כַלְּבָנָה וְכַחַמָּה בָרָה • וַיִּפְגְּעוּ בָנוּ מַלְאֲכֵי אוֹרָהּ • וַיְבוֹאוּנוּ חַשְׁמַלֵּי זָהֳרָהּ • וְאֵשׁ אַהֲבָתָהּ בַּלְבָבוֹת בָּעֲרָה • וָאֶקְרָא שֵׁם הַמָּקוֹם הַהוּא תַבְעֵרָה • כִּי בָעֲרָה בָּנוּ אֵשׁ חִשְׁקָהּ הַחֵךְ וְנֹפֶת הַמֶּלְקָחַיִם • וּבְכוֹר הַחֵשֶׁק בַּשֶּׁבִי אַחַר הָרֵחַיִם • וּמַה לְּנֹפֶת תִּטֹּפְנָה שִׂפְתוֹתַיִךְ • וְאֵין נוֹשֵׁק • וּמָה חֵפֶץ בַּחַיִּים אִם אֵין חוֹשֵׁק • וּמַה בֶּצַע בִּשְׁנֵי שָׁדַיִךְ • אִם לֹא יָלִינוּ בֵינֵהֶם דּוֹדַיִךְ • וּמַה בֶּצַע בְּרֵיחַ אַפֵּךְ כַּתַּפּוּחִים • וְהַחוֹשְׁקִים תַּחַת אַחַד הַשִּׂיחִים • וּמַה לְּחַמּוּקֵי יְרֵכַיִךְ כְּמוֹ חֲלָאִים • אִם לֹא תִתְּנִי שָׁם מַרְבֵּץ לִצְבָאִים • וּמַה בֶּצַע בְּבִטְנֵךְ עֲרֵמַת חִטִּים • וְאַתְּ בַּגִּלְגָּל וְחוֹשְׁקַיִךְ בַּשִּׁטִּים • לְקוּחִים לַמָּוֶת וְלַהֶרֶג מָטִים • הֲיִרְְוֻין דּוֹדַיִךְ מַלְאֲכֵי רוּם • אוֹ צִבְאוֹת שְׁחָקִים יְשַחֲרוּם • אֵין חֵפֶץ בְּדוֹדַיִךְ לִבְנֵי אֵלִים • וְלֹא אֶל מֵי יְדִידוֹתַיִךְ מְיַחֲלִים • מַלְאֲכֵי שָׁמַיִם • לָנוּ הַמָּיִם • רַעְיָה מָה אוֹסִיף לְדַבֵּר וְחִשְׁקֵךְ לְחָצַנִּי • הֵעִיק תַּחְתַּי וְעָדַיִךְ הֱרִיצַנִי • וּלְגַלּוֹת אֵלַיִךְ אוֹדוֹתַי וְאֶת אֲשֶׁר הִתְעַלֵּל בִּי חִשְׁקֵךְ אִלְּצַנִי • כָּלוּ בְיָגוֹן חַיַי וְיָמַי • כִּי הֶחֱרַשְׁתִּי בָּלוּ עצָמַי • רִוִּיתִי הָאֲרָצוֹת בְּדִמְעוֹתָי • וְלֹא שָׁאֲלוּ מָטָר אַחֲרַי גִּבְעוֹתָי • פָּנִיתִי אֶל הַיַּמִּים • לִמְצוֹא לִי מְנַחֲמִים • יָגַעְתִּי וְלֹא מָצָאתִי • וְלָכֵן עָדַיִךְ בְּלֵב בּוֹטֵחַ וְנֶאֱמָן בָּאתִי • בְּרוּחַ נְכֵאָה וְנֶפֶשׁ שְׁפֵלָה • אַחַר בַּקָּשַׁת הַמְּחִילָה • לְבַקֵּשׁ מִדּוּדָאֵי חִשְׁקֵךְ • וּמֵרִמּוֹנֵי חֵיקֵךְ • וְחָיְתָה נַפְשִׁי בִגְלָלֵךְ • יְהִי שָׁלוֹם בְּחֵילֵךְ • כְּנֶפֶשׁ הַחוֹלֶה בְּאַהֲבָתוֹ • הָאִגֶּרֶת הַזֹּאת שְׁלוּחָה מֵאִתּוֹ • לֹא יָסוּר שֵׁבֶט הַחֵשֶׁק מִקִּרבּוֹ • וְלֹא יָנִיחַ רָגְזוֹ וְעָצְבּוֹ • עַד עֲשׂוֹתוֹ וְעַד הֲקִימוֹ מְזִמּוֹת לִבּוֹ •


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


יֹאמְרוּ שְׁחָקִים לָךְ אֲיֻמָּה נָאְוָה



רַב לָךְ, צְבִיָּה, רַב שְׁלוֹם בָּנָיִךְ



בָּנוּ צְבָא מָרוֹם וּבָךְ עֵינָיִךְ



כִּי‑נִהְיֶה נֶגְדָּם כְּהַיּוֹם רַאְוָה

וּלְכוֹכְבֵי שַׁחַק כְּהַיּוֹם תַּאְוָה



הַיַּעֲלָה לִמְשׁוֹל בְּמִכְמַנָּיִךְ



לֹא‑יֹאמְרוּ שָׁוְא עָמְלוּ בֹנָיִךְ



כָּל שֹׁמְעֵי שִׁמְעֵךְ אֱמֻנַת גַּאְוָה


  


אֶשְׁמַע כְּסִיל וָעָשׁ וְכִימָה יֹאמְרוּ



מִי יִתְּנָה כַיּוֹם וְנִהְיֶה שֹׁקְדִים



בִּפְנֵי צְבִיַּת‑חֵן אֲזַי טוֹב לָנוּ


  


מָה אֶל שְׁחָקִים כִּי בְעֹז יִתְפָּאְרוּ



בּוֹ יַעֲלוּ יוֹמָם וְלַיְלָה יֹרְדִים



עֹז הַשְּׁחָקִים יוֹם לְיוֹם הִגְלָנוּ •





וַיֵּלֶךְ הַנַּעַר אֶל אַרְמוֹן הַגְּבִירָה • וַיִּמְצָאֶהָ סֹגֶרֶת וּמְסֻגֶּרֶת בְּתוֹךְ הַבִּירָה • וּבִגְדֵי נְזִירִים לוֹבֶשֶׁת • וְהִיא מִטֻּמְאַת הַזְּמָן מִתְקַדֶּשֶׁת, וַיִּתֵּן הָאִגֶּרֶת בְּיָדָהּ • וַתִּקְרָאֶנָּה לָדַעַת סוֹדָהּ • וַיְהִי כְקָרְאָהּ אֶת הַמְּגִלָּה עָלָה עָשָׁן בְּאַפָּהּ • וַתֹּאמֶר הַשּׁוֹלֵחַ זֹאת הָאִגֶּרֶת יִשְׂבַּע בְּחֶרְפָּה • וַתִּתְמַהְמַהּ כְּרֶגַע • וַתֹּאמֶר עוֹד נֶגַע • אָבִיא עַל הַשֹּׁלֵחַ אֶת הַמְּגִלָּה הַזֹּאת • וְלֹא יוֹעִילוּהוּ הַמְּלִיצָה וְהַחֲרוּזוֹת • וַתִּקַּח גִּלָּיוֹן גָּדוֹל • וַתִּכְתֹּב עָלָיו תְּשׁוּבַת אִגַּרְתִּי • וַתֹּאמֶר אֶל הַנַּעַר הוֹלֵךְ הָאִגֶּרֶת הַזֹּאת אֶל הַנָּבָל מֵאִתִּי • וְאֵלֶּה דִבְרֵי אִגֶּרֶת הַגְּבִירָה • אֲשֶׁר הָיוּ בִמְלִיצָה וְשִׁירָה:



אֲנִי בְצִלְלֵי הֵיכָלִי מִסְתּוֹפֶפֶת • בְּחַלּוֹן בֵּיתִי בְעַד אֶשְׁנַבִּי נִשְׁקֶפֶת • וְרוּחַ אֱלֹהִים עַל פָּנַי מְרַחֶפֶת • וְלִבִּי קְטֹרֶת תְּבוּנָה יִרְקַח • שְׁתֻם עֵין הַתַּאֲוָה וְעֵין הַשֵּׂכֶל יִפְקַח • מֵעֹצֶר הַחֵשֶׁק וּמִמִּשְׁפָּטוֹ לֻקַּח • וְכוֹס פֵּרוּד הָאֳהֳבִים רָוִיתִי • וּתְמוּנַת הַשֵּׂכֶל מִימִינִי וּלְנֶגְדִּי שִׁוִּיתִי • קַצְתִּי בְמַחֲמַדֵי הַזְּמָן וְלִישׁוּעַת הַנֶּפֶשׁ קִוִּיתִי • וְלֹא עָלָה בְלִבִּי חֵשֶׁק וְלֹא צִוִּיתִי • וְהִנֵּה יָד שׁלוּחָה אֵלַי מֵאִתְּךָ הַשָּׁר תִּלְחָצֵנִי • וְשִׁירִים אָרְגוּ אֶצְבְּעוֹתֶיהָ יִסְעֲרוּ לַהֲפִיצֵנִי • יְסִיתוּנִי לָלֶכֶת אַחֲרֵי הַצְּבָאִים • וְלִלְבֹּשׁ בְּגָדִים צוֹאִים • כּוֹרְעִים וּמִתְנַפְּלִים • וְלַעֲלוֹת הָהָרָה מַעְפִּילִים • יֹאמְרוּ הִנְנוּ שְׁלוּחֵי אֲבִי הַחוֹשְׁקִים • וּשְׁלוּחֵי מִצְוָה אֵינָם נִזּוֹקִים • קְרָאתִים וַיִּמְתְּקוּ כַלַּעֲנָה בְּפִי • וְקָדְחוּ אֵשׁ עֶבְרָה בְאַפִּי • וְעַתָּה הַגֶּד נָא לִי שַׂר צְבָא הַחוֹשְׁקִים • הֲרִיסוֹת הַחֵשֶׁק מֵקִים • מַה יַּעֲשׂוּ לִי שִׁירֶיךָ יָצוּרוּ עָלַי בְּחֵילָם • יִתְּנוּ עָלַי בְּקוֹלָם • וְלִבִּי כְּהַר צִיּוֹן לֹא יָמוּט לְעוֹלָם • נִשְׁעַנְתָּ עַל אִילָן יַעֲשֶׂה פְּרִי כָחַשׁ • וּפָגַעְתָּ בְלַחְשְׁךָ נָחָש בְּלִי לָחַשׁ • כִּי רַגְלִי עַל מִרְמָה לֹא תַחַשׁ • פְּנֵה לְךָ עַל יָמִין אוֹ עַל שְׂמֹאל • וּמַה יִתְרוֹן לְךָ לָרוּחַ תַּעֲמֹל • כִּי לִבִּי לְמַסַּת חֹשְׁקִים יִלְעַג וְלֹא יַחְמֹל • נְתַתִּיו כְּשָׁמִיר חָזָק מִצֹּר • לָבוֹא מִפָּנֶיךָ בַמָּצוֹר • וְרוּחַ נְגִידִים לִבְצֹר • לֹא גָדַל עַל אֲהָבִים • וְלֹא פָנָה אֶל רְהָבִים • הִנֵּה בַשָּׁמַיִם עֵדִי וְשָׂהֲדִי • כִּי לֹא תֹאכַל פְרִי מְגָדַי • ולֹא אֶתֵּן לְךָ אֶת דּוֹדַי • אַל תּוֹסִיף עוֹד הָתֵל בְּיַעֲלָה • אֲשֶׁר הִקְשִׁיחָה תַאֲוַת הָעוֹלָם לְלֹא לָהּ • וְהִזְנִיחָה תֵבֵל וַעֲמָלָהּ • וְקָשְׁרָה עַל הַזְּמָן קָשֶׁר • וְלָבְשָׁה סוּת הָעֲנָוָה וְהַיֹּשֶׁר • מַה לְּךָ פֹה וּמִי לְךָ פֹה • כִּי תֵלֵךְ הָלֹךְ וְצָפֹה • נָגֹף וְרָפוֹא • וַאֲנִי טִהַרְתִּי רַעְיוֹנַי וָאֶצְרְפֵם • לֹא יַבְרִיחֵם בֶּן קֶשֶׁת וְקֶשֶׁת נְחוּשָׁה לֹא יַחְלְפֵם • רָחַצְתִּי אֶת רַגְלַי מִטֻּמְאַת הַזְּמָן אֵיכָכָה אֲטַנְּפֵם • חַי אֲנִי לְקוֹל הַחֹשְׁקִים תִּצַּלְנָה אָזְנַי • וְטִמֵּאתִי הַחֵשֶׁק בְּעֵינַי • כְּטֻמְאַת הַנִּדָּה הָיְתָה דַרְכָּם לְפָנַי • וָאֹמַר בְּעַד לִבִּי זֶה סִינַי • מִפְּנֵי אֲשֶׁר יָרַד עָלָיו אֲדֹנָי • וְעַתָּה מַה זֶּה רוּחֲךָ סָרָה • וּמַחֲשַׁבְתְּךָ לִזְנוּנִים הָרָה • לַחֲשֹׁק בּעָפְרָה וּלְדַבֵּר עַל לִבָּהּ • אֲשֶׁר לֹא בְּשַׁחְתָּם נִתְפָּשׂוּ וּבְרַגְלֵי הַחֹשְׁקִים נִרְמָסוּ • גַּם אִם גָּבְהוּ בְּנוֹת צִיּוֹן וִירוּשָׁלָיִם • וּבָנוֹת לָעִיר סְפַרְוָיִם • הָלְכוּ נְטוּיוֹת גָּרוֹן וּמֻשַׂקְּרוֹת עֵינָיִם • לֹא כִּי יָבוֹאוּ הָאֲנָשִׁים בִּמְעוֹנוֹת אֲרָיוֹת • וְיָצוּדוּ מִשָּׁם הַלְּבִיּוֹת • וְיוֹרִידוּ יוֹנֵי הַגֵּאָיוֹת • מִמִּגְדַּל דָּוִד בָּנוּי לְתַלְפִּיּוֹת • וְיָצוּדוּ הַדָּגִים ממְּצוּלוֹת יָם • וְיָנִיפוּ יְדֵיהֶם עַל הַנָּהָר בַּעְיָם • וְיוֹרִידוּ הַחֲסִידָה מִן הַשָּׁמַיִם בְּשִׂכְלָם • וכָל אֲשֶׁר יְמָאֲנוּ לִשְמוֹעַ בּקוֹלָם • בְּרֶשֶׁת זוֹ טָמָנוּ נִלְכְּדָה רַגְלָם • חֲבַצֶּלֶת הַשָּׁרוֹן שׁוֹשַׁנַּת הָעֲמָקִים • אַל תִּהְיִי כִבְנוֹת הָרֵיקִים • תְּמָאֲנֶנָּה לְהַפְקִיד חִשְׁקָם בְּמִבְחַר הַמְּקוֹמוֹת • וּבִיפִי הַנְעִימוֹת • הַמִּתְקַדְּשׁוֹת וְהַמִּטַּהֲרוֹת אֶל הַמְּקוֹמוֹת הַמְּפֻרְסָמוֹת • וְאַחַר אַחַת בַּתָּוֶךְ תִּבְחַרְנָה הָעִוְרִים וְהַפִּסְחִים לַסְתִּיר עֲצָתָם • תְּכַבֶּינָה בָהֶם אֵשׁ תַּאֲוָתָם • לֹא כֵן אֲחוֹתִי בַּחֲרֵי לָךְ מִן הַצְּבָאִים עֹפֶר • וּמִן הַכְּפָרִים אֶשְׁכּוֹל הַכֹּפֶר • וּקְחִי דוֹדָיו לָךְ לְצֵידָה וְהִתְעַלְּסִי בַאֲהָבָיו •יִכַבֶּה לַהַב חִשְקֵךְ וְכַבִּי אֵשׁ לְהָבָיו • וְיָסֹבּוּ עָלַיִךְ רַבָּיו • וְהִנֵּה בָאתִי רִאשׁוֹן לְכָל סֵבֶל בֵּית הַחֹשְׁקִים • הֲרִיסוֹת הַחֵשֶׁק להָקִים • וּפֶן תֹּאמְרִי שַׂמְתִּי לִבִּי מִגְדָּל • וְגָעַרְתִּי בַחֹשֵׁק וְיֶחְדָּל • הִנְנִי נִשְׁבַּע בְּאַהֲבָתֵךְ הַיקָרָה בְּעֵינַי • כִּי בָזִית מִכְתָּבִי וְלֹא נִבְהֲלוּ רַעְיוֹנַי • וֹמֵאַנְתְּ חִשְׁקִי וְלֹא נָפְלוּ פָנַי • אַךְ אָמַרְתִּי יֵשׁ לִי תִקְוָה • כִּי כֵן חֹק כָל יעֲלָה וְיַעֲלָה • בְּפִיהֶן תּוֹכַחַת מְגֻלָּה • וְתוֹכָן רָצוּף אַהֲבָה וְחֶמְלָה • מַה מְּאֹד נָעַם אֵלַי זֶה הַמֵּאוּן וּמַתְכֻּנְתּוֹ • וּמַה מְּאֹד הָיָה דָבָר בְּעִתּוֹ • לֹא יִפּוֹל לֵב חשֵׁק עַל זֶה בֶאֱמוּנָה • כִּי זֶה מֵחֹק הַמּוּסָר לְמָאֵן בָּרִאשׁוֹנָה • חֵי הַבְּרִית לוּ נָתַתְּ לִי חִשְׁקֵךְ בְּפַעַם רִאשׁוֹנָה לֹא עָלֵז לִבִּי כַאֲשֶׁר יַעֲלֹז • וְהָיִיתִי בְמעְגַּל אַהֲבָתֵךְ נָלֹז • וּבָזֹאת יָדַעְתִּי כִּי רוּחַ נְדִיבָה זרְקָה בָךְ הָאֲהָבוֹת • וּנְדִיבָה אַתְּ וְתָקוּמִי עַל נְדִיבוֹת • וּבְלֵב בֹּטֵחַ וְנֶאֱמָן אֶעֱמוֹד בְּאַהֲבָתֵךְ אֶת מַחַץ חִשְׁקִי לַחְבֹּשׁ • וְאֵדַע כִּי לֹא אֵבוֹשׁ • וְלָכֵן לָקַחְתִּי מִפְּנִינֵי הַמַּאֲמָרִים אֲמָרוֹת טְהוֹרוֹת וְעָשִׂיתִי אוֹתָם עַל רֹאשׁ תְּהִלָּתֵךְ עֲטָרוֹת • הוֹאִילִי נָא הַצְבִיָּה בְּהוֹדֵךְ • וְהוֹשִׁיטִי אֲלֵיהֶם אֶת שַׁרְבִיט חַסְדֵּךְ • וְאִם אֵינָם לְפִי כְבוֹדֵךְ • כִּי אֵין נְבָלָה שֶׁלֹּא תְכַלְכְּלָהּ נְדִיבוּתֵךְ • וְאֵין עַזּוֹת שֶׁלֹּא תְקַדְּמֶנּוּ עַנְוָתֵךְ • וְאֵין עָוֹן שֶׁלֹּא תְכַפְּרֵהוּ מְחִילָתֵךְ • וְהִנְנִי מְיַחֵל לְמַעְנָהּ בִּרְכָתֵךְ • אוֹ לִשְׁתִיקַת הוֹדָאָתֵךְ • וְלָךְ הַצְּבִיָּה רַב שָׁלוֹם עַד בְּלִי יָרֵחַ • מֵאֵת הַצְּבִי דּוּדָאֵי חִשְׁקוֹ נָתְנוּ רֵיחַ • וְאֶל מְקוֹם אַהֲבָתֵךְ שֹׁאֵף וְזֹרֵחַ • יַחֲזִיק בְּכַנְפֵי חִשְׁקֵךְ בְּאַהֲבָתוֹ • לֹא יִשְלָחֵךְ כִּי אִם בֵּרַכְתּוֹ •


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


  


הַגֵּד כְּרוּב מִמְשַׁח בְּרֹב חַסְדֶּךָ



מִכָּל‑עֲשָׂקֶיךָ הֲאִם פָּנִיתָ



עַד לַתְּעוּדָה בֵּית זְבֻל בָּנִיתָ



מִקְדַּשׁ אֲדֹנָי כּוֹנְנוּ יָדֶיךָ


  


הִיא הַצְּבִיָּה בַּעֲלַת סוֹדֶךָ



קִנְיָנְךָ הַשָּׂר אֲשֶׁר קָנִיתָ



בִּימִינְךָ צַרְתָּהּ וְלֹא שָׁנִיתָ



עֵת נֶחֱזֶהָ יוֹם לְיוֹם נוֹדֶךָ


  


מֶה חָכְמוּ צִבְאוֹת גְּדוּדֵי מַעַל



אִם עוֹרְרוּךָ הֵם עֲשׂוֹת תִּפְאֶרֶת



תַּבְנִית כְּבוֹדָם לַעֲמֹד בֵּינֵינוּ


  


כִּי‑בָהּ לְבַד נַשִּׂיג פְּאֵרָם נַעַל



לִרְאוֹת פְּאֵר עֶלְיוֹן וְהַכֹּתֶרֶת



יוֹם הַצְּבִיָּה תֶּחֱזֶה עֵינֵנוּ •


  


וַיְהִי כִרְאוֹת הַגְּבִירָה אֶת קְשִׁי מִצְחִי • וַתֹּאמֶר גַּם הַיּוֹם מְרִי שִׂיחִי • עַד מָתַי לֹא יִתְּנֵנִי זֶה הַנָּבָל הָשֵׁב רוּחִי • חַי אֲנִי כִּי בַּפַּעַם הַזֹּאת אֲשִׂימֶנּוּ נִכְלָם • וְלֹא יוֹסִיף לִרְאוֹתִי עוֹד עַד עוֹלָם • וַתִּכְתֹּב מְגִלַּת סֵפֶר וַתִּשְׁפֹּךְ בָּהּ אֶת כָּל נַפְשָׁהּ • וַתִּשְׁלָחֶנָּה אֵלַי וַתַּעֲנֵנִי קָשָׁה • וְאֵלֶּה דִבְרֵי אִגַּרְתָּהּ • הַשְּׁלוּחָה אֵלַי מֵאִתָּהּ:

לִבְזֹה נֶפֶשׁ לִמְתָעֵב גּוֹי לְעֶבֶד מֹשְׁלִים • אֲשֶׁר הוֹצֵאתַ מִפִּיךָ מִלִּים • לָחוּד חִידוֹת וְלָשֵׂאת מְשָׁלִים • בְּרֹב חֲלוֹמוֹת וַהֲבָלִים • וַתִּלְעַג וְאֵין מַכְלִים • לָמָּה הִרְגַּזְתַּנִי לַעֲנוֹת אֶל מַאֲמָרֶיךָ • וְחִשְׁקִי סָר מֵעָלֶיךָ וַיְהִי עָרֶךָ • מַה לִּי וּלְךַ כִי בָאתָ אֵלַי בִּי לְהִלָּחֵם • וְלִבִּי נִשְׁבַּע וְלֹא יִנָּחֵם • אַכְזָרִי הוּא וְלֹא יְרַחֵם • נְתַתִּיו כְּעַמּוּד בַּרְזֶל מִמְּקוֹמוֹ לֹא יָמִישׁ • עַד תּוֹצִיא לְךָ מַיִם מִצּוּר הַחַלָּמִישׁ • לְמַעַן תֵּדַע כִּי לֹא כַּלְנוֹ כְּכַרְכְּמִישׁ • וּמֵאֲשֶׁר חַנַּנִי הַזְּמָן לֹא אַעֲנִיֶקךָ • וְיֵינִי עִם חֲלָבִי לֹא אַשְׁקֶךָ • וְאִם תִּהְיֶה חוֹתָם עַל יַד יְמִינִי מִשָּׁם אֲנַתְּקֶךָ • וְכָל אֲשֶׁר תּוֹסִיף לְהִכָּנַע אַקְשֶׁה עָרְפִּי • כַּאֲשֶׁר הָיִיתִי בִימֵי חָרְפִּי • וּבְצִדְקָתִי הֶחֱזַקְתִּי וְלֹא אַרְפֶּה • וְחַרְבִּי הַקָּשָׁה עַל הַחֹשְׁקִים אֲנִיפֶיהָ • וּמְשׁוֹקְקַי יְנִידוּךָ רַעְיוֹנַי מֵעַכְשָׁו • לֹא תוֹסִיף הָבִיא מִנְחַת שָׁוְא • כִּי לִבִּי צוּר מָעֹז וּקְשֵׁה מֶצַח • לְהָרִים קוֹל בִּתְרוּעָה וְלִפְתֹחַ פֶּה בְּרֶצַח • בַּל יִסַּע יְתֵדוֹתָיו לָנֶצַח • לֹא יִמַּסּוּ אֲסוּרָיו הָעֵת עִם הַחוֹשְׁקִים יֵאָבֵקוּ • וְכָל חֲבָלָיו בַּל יִנַּתֵקוּ וּפְנֵי הַצְּבָאִים לֹא נֶהְדָּרוּ • וְהִנֵּה עֹפֶר הַצְּבָאִים יַעֲרֹג כְּאַיָּל עַל אֲפִיקֵי אַהֲבָתִי • הוֹמֶה וְצָמֵא אֶל מֵי יְדִידוּתִי • וְהִכְזַבְתִּי תוֹחלַתְוֹ בְגַאֲוָתִי • קָרָא וְלֹא וְאֵין שׁוֹאֵל דּוֹדַי לְצֵידָה • פָּרְחוּ גְפָנַי וְאֵין לוֹקֵט • הִבְשִׁילוּ אַשְׁכְּלוֹת עֲדָנַי וְאֵין שֹׂחֵט • הֵנַצוּ רִמּוֹנַי • וְאֵין אֹחֵז בְּסַנְסִנַּי • יַחְשְׁבוּנִי בֵינֵיהֶם לְמוֹקֵשׁ • אֵין דּוֹרֵשׁ וְאֵין מְבַקֵּשׁ • וָאָשִׁיב אֶל לִבִּי וַאֲחַשֵּׁב • כִּי אֶל עַקְרַבִּים אֲנִי יוֹשֵׁב • אַסְתִּירָה פָנַי מֵהֶם • לֹא מֵהֶם ולֹא מֵהֲמוֹנָם וְלֹא מֵהֱמֵיהֶם • חוּשׁוּ הַחֹשְׁקִים הַנְּעִימִים • כִּי אַתֶּם נֵזֶר הַיָּמִים • צְבִי עֶדְיָם וּגְאוֹן תִּפְאַרְתָּם • נִזְמָם וְחֶלְיָתָם • אוֹכְלֵי הַמָּן • יְתֵדוֹת הַתְּקוּעוֹת בִּמְקוֹם נֶאֱמָן • שִׁמְעוּ אֵלַי וְאִכְלוּ טוֹב וְתִתְעַנֵּג בַּדֶּשֶׁן נַפְשְׁכֶם • כִּי מִי יֹאכַל וּמִי יָחוּשׁ חוּץ מִכֶּם • חֵי הַבְּרִית לוּ יְדַעְתֶּם סוֹדוֹתֵינוּ • וְאֶת אֲשֶׁר לַהֶבֶת הַחֹשְׁקִים תְּלַהֵט לִבּוֹתֵינוּ • וְלוּ תִהְיוּ בְחִשְׁקְכֶם נִמְהָרִים • אֲזַי נֵדְעָה וְנִרְדְּפָה אַחֲרֵי הָעֳפָרִים • בִּמְעוֹנוֹת אֲרָיוֹת וּבְהַרְרֵי נְמֵרִים • כַּאֲשֶׁר יִרְדֹּף הַקֹּרֵא בֶהָרִים • דּוֹדִי צָפַנְתִּי לְךָ חֵשֶק נֶאֱמָן • וְלֵב חֹמֵל וְרַחֲמָן • טְמֹן חִשְׁקְךָ בְּחֻבּוֹ • וְלַהֶבֶת אַהֲבָתְךָ בְּקִרְבּוֹ • וְאֵלֶיךָ תְשׁוּקָתוֹ וְאַתָּה תִּמְשָׁל בּוֹ • יָבוֹא דוֹדִי לְגַנּוֹ וְיֹאכַל פְּרִי מְגָדַי • שָׁם אֶתֵּן לְךָ אֶת דּוֹדַי • לֹא אֶשְאַל וְלֹא אֲנַסֶּה • כַּטּוֹב וְכַיָּשָׁר בְּעֵינֶיךָ לַעֲשׂוֹת בִּי עֲשֵׂה • וְחָיְתָה נַפְשִׁי בִגְלָלְךָ • וַתִּגְזַר אֹמֶר וַיָּקָם לְךָ • ועַל דְּרָכֶיךָ יִגַּה אוֹר כְנֶפֶשׁ חוֹלַת אַהֲבָתֶךָ • צָמְאָה אֶל מֵי יְדִידוּתֶךָ • תִּקֹּד וְתִשְׁאַף וְתֶהֱמֶה וְתִסְעַר • עַד יֵצֵא כַנֹּגַהּ צִדְקָהּ וִישׁוּעָתָהּ כְּלַפִּיד יִבְעַר •


  


וַתִּשָּׂא אֶת קוֹלָהּ וַתֹּאמַר:


  


הָסֵב עֵינֵי חִשְׁקָךְ דּוֹדִי



עַד אָן בִּי תִיר חִצִּים דּוֹד צַח



אִם‑לֹא תֵדַע כִּי אִם‑חִצֵּי



חִשְׁקָךְ לֹא תִיר צַח לֹא תִרְצָח •


  


וַיְהִי כַאֲשֶׁר דִּבְרֵי הַנַּעַר שָׁמְעוּ אָזְנָי • וּמְלִיצַת הַגְּבֶרֶת רָאוּ עֵינָי • וַתִּצְלַח עָלַי רוּחַ אֲדֹנָי • וָאֵלֵךְ קַל חִישׁ אֶל הַשָּׂר שָׂמֵחַ בְּכָל מְאֹדִי • וְאִגֶּרֶת הַגְּבִירָה פְּתוּחָה בְיָדִי • וָאֹמַר הִגַּדְתִּי הַיּוֹם אֵלֶיךָ הַשָּׂר כִּי לֹא סָכַלְתִּי • כִּי נַפְתּוּלֵי אֱלֹהִים נִפְתַּלְתִּי • עִם גְּבֶרֶת הַמְּהַלְלִים גַּם יָכֹלְתִּי • וְאַחֲרֶי אֲשֶׁר מָרְתָה בִי רִפִּיתִי • מְזִיחַ אֲפִיקֶיהָ וְקָשְׁיָהּ • וְשָׁבָה לִהְיוֹת בְּאַהֲבָתִי לוֹיָה • וְגַם זָבַת חָלָב וּדְבַשׁ הִיא וְזֶה פִּרְיָהּ •


  


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

מָה רַב טוּבִי כִּי עָפְרַת‑חֵן



קָרְאָה אֹתִי אֶל יוֹם נוֹעָד



הִבְטִיחַתְנִי מִצּוּף יַעְרָהּ



אֵיכָה לִבִּי כַיּוֹם יִרְעָד



אִם‑רַגְלִי עַל‑רִצְפַת חִשְׁקָהּ



יַעְמֹד אֵיכָה רַגְלִי יִמְעָד



אִם לִי נֹעַם צוּף אַהְבָתָהּ



מַה לִּי וָלָךְ זַעַם תִּצְעָד



הִיא גוֹרָלִי וּמְנָת חֶבְלִי



וּזְמַנִּי לִי אֹתָהּ יָעָד



הִיא חֶמְדָּתִי רֹאשׁ שִׂמחָתִי



וּלְכָל שַׁעְשׁוּעַי בֵּית וָעַד



אֶשְׂחַק אֶל שֹׁד עַד אִינַךְ שֹׁד



תַּנְחֻמֶיהָ לִבִּי יִסְעָד



אֶשְׂחַק אֶל שֹׁד עַד אִינַק שֹׁד



יִקְחוּ רֵעַי מִצְפֵּה גִלְעָד



לִי מֵעָפְרָה מִיץ‑פֶּה גִּיל‑עַד •


  


כִּשְׁמֹעַ הַשָּׂר עָלָה עָשָׁן בְּאַפּוֹ • יֶהֱמֶה עָלַי וְיִסְעַר כַּאֲשֶׁר יֶהְגֶּה הָאַרְיֵה וְהַכְּפִיר עַל טַרְפּוֹ • וְאֵשׁ מִפִּיו תֹּאכֵל • נִלְאֶה כַּלְכֵּל • וַיֹּאמֶר אָרוּר יִהְיֶה לִבִּי הַנּוֹכֵל • אֲשֶׁר עַד כֹּה הֶחֱרַשְתִּי • וְעֶצֶם מֵעֲצָמַי הוֹבַשְׁתִּי • וּלְהַכְלִים בָּשָׂר מִבְּשָׂרִי נִדְרַשְׁתִּי • הַצְבִיָּה אֲשֶׁר נָפְלָה בְרִשְתֶּךָ • וְנִלְכְּדָה בִמְצוּדָתֶךָ • הִיא מִבְּנוֹת עַמִּי • אֲחוֹתִי בַת אָבִי הִיא אַךְ לֹא בַת אִמִּי • וְחָשַׁבְתִּי בְלִבִּי כִי לֹא תִלָּכֵד בְּרִשְׁתֶּךָ • וְאַתָּה תָשׁוּב עַל עֵקֶב בָּשְתֶּךָ • אַחֲרֵי אֲשֶׁר חָשַׁבְתָּ לַעֲלוֹת בַּאשֶׁר לֹא תַגִּיעַ • וְאָמַרְתִּי בְלִבִּי זֶה הַסָּכָל אֲשֶׁר לֹא יַרְגִּיעַ • חֵטְא רַבִּים יִשָּׂא וְלַפֹּשְׁעִים יַפְגִּיעַ • אָמְנָם אַחֲרֵי אֲשֶׁר הִגִּיעַ מִסִּכְלוּתָהּ • לְחַלֵּל הַדְרַת יִפְעָתָהּ • מַהֲרָה חוּשָׁה אַל תַּעֲמֹד • בַּאֲשֶׁר כְּבוֹדִי תַּחְמֹד • וְשָׂא מְשָׁלֶיךָ וְשִׁירֶיךָ • וֶאֱמֹר לַגְּבִירָה כִּי לְבַעֲבוּר נְסוֹתָהּ בָּאוּ זְמִירֶיךָ • וְחַזֵּק אֶת לִבָּהּ בִּדְבַר הַנְּזִירוּת אֲשֶׁר בָּאָה בִבְרִיתוֹ • כַּאֲשֶׁר הָיְתָה בְאָמְנָה אִתּוֹ • וְאִם אֶת הַדָּבָר הַזֶּה תַּעֲשֶׂה אַשְׁרֶיךָ • וְאִם אַיִן דַּע כִּי חִשְׁקִי סָר מֵעָלֶיךָ וַיְהִי עָרֶךָ • כְּשָׁמְעִי דִבְרֵי הַשָּׂר עַל אֹדוֹת הַגְּבֶרֶת • בֹּשׁתִּי וְגַם נִכְלַמְתִּי כְגַנָּב נִמְצָא בַמַּחְתֶּרֶת • וָאֹמְרָה אָנָּא הַשָּׂר שָׂא נָא חַטָּאתִי אַךְ הַפָּעַם • וְהַעֲבֵר מֵעָלַי עֶבְרָה וָזָעַם • כִּי הִנְנִי נִשְׁבָּע בְּנוֹרָא וְאָיום • כִּי אָנֹכִי לֹא יָדַעְתִּי בִּלְתִּי הַיּוֹם • וְהִנְנִי לְמַלֹּאת מִצְוָתְךָ אֶהְיֶה נֶחְפָּז • כִּי יָקְרָה מִצְוָתְךָ בְּעֵינַי מִזָּהָב וּמִפָּז • וָאָקוּם קַל חִישׁ וָאֲחַבֵּר מְלִיצָה נִמְרֶצֶת • מִתּוֹכְחוֹת מוּסָר מְקֻבֶּצֶת • לְהַגִּיד לַגְּבִירָה כִּי לְשָׂטָן לָהּ יָצָאתִי • וּלְבַעֲבוּר נַסּוֹת אוֹתָהּ בָּאתִי • וּלְהוֹכִיחָהּ בִּדְבָרִים • בַּעֲבוּר הִתְנַדְּבָהּ לַעֲזֹב תּוֹרַת הַנְּזִירִים • לָלֶכֶת אַחֲרֵי הַבַּחוּרִים • וְאֵלֶּה דִבְרֵי אִגַּרְתִּי • הַשְּׁלוּחָה אֶל הַגְּבִירָה מֵאִתִּי:



זֶה הַיּוֹם עָשָׂה הַשֵּׁם מָה רַב טוּבוֹ • נָגִילָה וְנִשְׂמְחָה בוֹ • זֶה כַּמָּה לַחְשֹׂף סוֹד הַצְּבִיּוֹת לָאִינוּ • זֶה הַיּוֹם שֶׁקִּוִּינוּהוּ מָצָאנוּ רָאִינוּ • הַיּוֹם הִבְדַּלְנוּ בֵּין קֹדֶשׁ לְחוֹל • הַיּוֹם נִגְלוּ שְׂפֻנֵי טְמוּנֵי חוֹל • הַיּוֹם קוֹל הַנִּסָּיוֹן יְחוֹלֵל אַיָּלוֹת וְיַחְשֹׂף יְעָרוֹת • הַיּוֹם נִגְלוּ סוֹדוֹת הַמִּתְקַדְּשׁוֹת וְהַמִּטַּהֲרוֹת • הַיּוֹם גַּלּוֹתִי אֶת חֶרְפַּת הַצְּבִיּוֹת • הַיּוֹם חָשַׂפְתִּי סוֹדוֹת הַלְּבִיּוֹת • נַפְתּוּלֵי אֱלֹהִים נִפְתַּלְתִּי • וּבְעִמְקֵי הַמְּזִמּוֹת צָלָלְתִּי • וְהֶרְאָה אוֹתִי הַזְּמָן אֲשֶׁר רָאֹה לֹא פִלַּלְתִּי • עַד אֲשֶׁר בֵּין הַבְּהֵמָה הַטְּהוֹרָה לַטְּמֵאָה הִבְדַּלְתִּי • וְקֶרֶן הַצְּבִיּוֹת בֶּעָפָר עוֹלַלְתִּי • הִשְׁבַּתִּי מִטָּהֳרָם וְנִזְרָם לָאָרֶץ חִלַּלְתִּי • מִי הֶאֱמִין כִי הַמִּטַּהֶרֶת וְהַמִּתְקַדֶּשֶׁת • וּבִגְדֵי הַנְּזִירוּת בְּגַאֲוָה לוֹבֶשֶׁת • יִמְצָאוּהָ הַמּוֹרִים בַּקֶּשֶׁת • וְיֶעֱרַב לִבָּהּ לָגֶשֶׁת • וְכַצִּפֳּרִים הָאֲחֻזּוֹת בַּפָּח • תִּפֹּל אֶל תּוֹךְ רֶשֶׁת • חַי מוֹשֵׁל בִּגְבוּרָתוֹ עוֹלָם וְעֵינָיו בַּגּוֹיִם תִּצְפֶּינָה • כִּי כַמָּה צְבִיּוֹת אֲשֶׁר אֶל חֵשֶׁק תִפְנֶינָה • וּלְקוֹל הַחֹשְׁקִים לֹא תִשְׁמַעָנה וְלֹא תֹאבֶינָה • לִבּוֹת אַנְשׁי קֹדֶשׁ תִּשְׁבֶּינָה • וּבַמִּסְתָּרִים בְּהֵיכְלֵי הַחֹשְׁקִים תְּסֻבֶּינָה • וַיָּבוֹאוּ אֶל קִרְבֶּנָה • וְלֹא נוֹדַע כִּי בָאוּ אֶל קִרְבֶּנָה • מִי הֶאֱמִין כִּי הַיַּעֲלָה הַמְהֻלָּלָה • בְּרֶשֶׁת זוּ טָמַנְתִּי נִלְכְּדָה רַגְלָהּ • אַיֵּה הֲמוֹן חָכְמָתָהּ וְאֵיךְ לֹא נֶעֱזָרָה • וּתְמוֹל עַל בְּכִי הַחֹשְקִים הִתְאַכְזְרָה • וְעַתָּה אֵיךְ רוּחָהּ סָרָה וְזָרָה • וְעַל פְּרִי הֲדָרָהּ עָלְתָה חֲזָזִית • אִמְרִי נָא הַצְּביָּה מָה הַמַּחֲזֶה הַזֶּה אֲשֶׁר חָזִית • כַּאֲשֶׁר קָדְשֵׁי הַנְּזִירוּת בָּזִית • וְלָלֶכֶת אַחֲרֵי הַחֹשְׁקִים הִתְבַּזִּית • אָהַבְתְּ מִשְׁכָּבָם יָד חָזִית • מִי הֶאֱמִין כִּי מִשְּׁמֵי הַשֵּׂכֶל תִּפְּלִי • כִּי מֵאִתּוֹ גִלִּית וַתַּעֲלִי • אִי עַמּוּד בַּרְזֶל וְחוֹמַת נַחֹשֶׁת • אֵיךְ לֹא הָיִית לְעֵזֶר וּלְהוֹעִיל כִּי אִם לְבֹשֶׁת • אִי יָם הַתְּבוּנָה אֵיךְ חָסְרוּ מֵימַיִךְ עַד יְבֹשֶׁת • אֵיךְ לֹא עָמַדְתְּ בְּנִסְיוֹנֵךְ הָעֵת לְשָׂטָן יָצָאתִי • אֵיךְ לֹא הֶעֱלֵית עַל לְבָבֵךְ כִּי לְבַעֲבוּר נַסּוֹת אוֹתָךְ בָּאתִי • אִמְרִי נָא הַצְּבִיָּה מִמִּי נָעַמְתְּ • מָה הַחֲלוֹם הַזֶּה אֲשֶׁר חָלַמְתְּ • כַּאֲשֶׁר הוֹבַשְׁתְּ פְּנֵי הַצְבִיּוֹת וְיִפְעַת הַלְּבִיּוֹת הִכְלַמְתְּ • הִשְׁמַעְתִּיךְ לִשְׁמֹעַ אֹזֶן תַּעְתּוּעֵי הַזִּמָּה • וַתִּמְכְּרִי בָהֶם סַחֲרֵי הַמְּזִמָּה • טְמֵאַת הַשֵּׁם רַבַּת הַמְּהוּמָה • אוֹי לָהּ לְאוֹתָהּ כְּלִמָּה • אָמֹר אָמַרְתִּי מִלַּיִךְ יְקִימוּן כּוֹשֵׁל • וְהִנֵּה עָלַיִךְ יִמְשֹׁל כָּל מוֹשֵׁל • וְעַל גַּבֵּך כָּל שָׁר יַחֲרֹשׁ • יַפְטִירוּ בְשָׂפָה יָנִיעוּ רֹאשׁ • תְּמוֹל יָצָא לָךְ שֵׁם בַּגּוֹיִים • וְהַיּוֹם תִּהְיִי לַעַג וָקֶלֶס בָּאִיִּים • בְּזָכְרָם כִּי עִם אֱלֹהִים וְעִם אֲנָשִׁים שָׂרִית • וּבַחֹשְׁקִים לְהִלָּחֵם הִתְגָּרִית • וַתִסָּתְרִי בְנַחַל כְּרִית • וְהַיּוֹם בָּזִית אָלָה לְהָפֵר בְּרִית • הָיָה בְשֶׁכְּבָר אֱמוּנַת עִתַּיִךְ חֹסֶן יְשׁוּעוֹת • וְאַתְּ אָלִית וְגַם אָמַרְתְּ בְאָזְנַי שְׁבוּעֵי שְבוּעוֹת • לְהִדָּבֵק בְּדַרְכֵי הַנְּזִירִים • וּלְבִלְתִּי לֶכֶת אַחֲרֵי הַבַּחוּרִים • אַיֵּה הֲמוֹן שְׁבוּעוֹתַיִךְ אֲשֶׁר נִשְבַּעַתְּ • וְקָלוֹן מִכָּבוֹד שָׂבַעַתְּ • כַּאֲשֶׁר בִּיוֵן מְצוּלַת הַחֵשֶׁק טָבַעַתְּ • וַתִּזְנִים וְגַם לֹא שָׂבַעַתְּ • תְּמוֹל קָרְאוּ לְמַחֲנוֹת שִׂכְלֵךְ מַחֲנָיִם • וְהַיּוֹם גֹּמֶר בַּת דִּבְלָיִם • יֹאמְרוּ אַיֵּה הַקְּדֵשָׁה הִיא בָעֵינַיִם • הָיִית פְּאֵר הָעֲדִינוֹת • רַבָּתִי בַגּוֹיִם שָׂרָתִי בַּמְּדִינוֹת • יָקֹשְׁתִּי לָךְ הַצְּבִיָּה וְנִלְכַּדְתְּ • הֲשֵׂכֶל אֶין בָּך אִם יוֹעֲצֵךְ אִבַּדְתְּ • אֵיךְ פָּשַׁטְתְּ מְעִיל הָרוּחָנִית וְלָבַשְׁתְּ בִּגְדֵי חֵשֶׁק • וּלְהִלָּחֵם בִּנְזִירוּת יָדֵךְ אַחַת מַחֲזֶקֶת בַּשֶּׁלַח וְהַשֵּׁנִית בַּנֵּשֶׁק • אַיֵּה עֲטֶרֶת הַפְּרִישׁוּת הָרַבָּנִי • אַיֵּה זִמְרַת הַיֵּשׁוּת הָעֶלְיוֹנִי • אֵיךְ מָכָרְתְּ הַקַּיָּם וְהַנַּעֲלֶה בַכָּלֶה • וּפְרִי עֵץ הַחַיִּים בְּעָלֶה • הִנֵּה אָנֹכִי יָצָאתִי לְשָׂטָן • לְגַנּוֹת הַצְּבִיּוֹת וּלְהוֹבִישׁ מֶבָּטָן • כִּי מִי יָחוּס עֲלֵיהֶן וּמִי יַחְמֹל • מְקָרְבוֹת בְּיָמִין וְדוֹחוֹת בִּשְׂמֹאל • בְּלֵב עָרֵל וְלֹא נִמּוֹל • וַאֲשֶׁר תִּרְאֶה קְדוֹשָׁה וּטְהוֹרָה • הִיא לִזְנוּנִים הָרָה • וַאֲשֶׁר תִּרְאֶה לֵב מַשְׂכִּיל וּמַזְהִיר • הוּא לֵץ יָהִיר • בְּשַׁחֲקֵי הַשֵּׂכֶל בָּהִיר • וּבֵין שׁוּרוֹת הַחֵשֶׁק יַצְהִיר • אִמְרִי נָא, הַצְּבִיָּה, עַל מִי נִבְטַח • אַחַרֵי אֲשֶׁר לִבֵּךְ מֵהַשְׂכִּיל טָח • וְלִמְצוּדִים וְלַחֲרָמִים מִשְׁטָח • וְאַתְּ עָלִית לִמְרוֹם הַשֵּׂכֶל וְשָׁבִית שֶׁבִי • וַתִּגְדְּלִי יֶתֶר אֶל הַנֶּגֶב וְאֶל הַצֶּבִי • וְאֵיךְ הִנַּחְתְּ כְּבוֹד הַצְּבִיּוֹת עַל קֶרֶן הַצְּבִי • וְנָתַתְּ אוֹתָם פְּלֵיטוֹת וּבְנוֹתֵיהֶם בַּשְּׁבִי • וְלָךְ אָמְרוּ הַצְּבִיּוֹת לְעֵינֵינוּ • זֹאת אֶדֶר הַיְקָר וַהֲדַר מַחְמַדֵּינוּ • זֹאת תְּנַחֲמֵנוּ מִמַּעֲשֵׂנוּ וּמֵעִצְּבוֹן יָדֵינוּ • זֹאת יָתֵד תְּקוּעָה בְּמָקוֹם נֶאֱמָן תָּלִינוּ עָלֶיהָ כָל כְּבוֹדֵנוּ • וְהִשְׁלַכְנוּ עָלֶיהָ אֶת יְהָבֵנוּ • הִיא פֶּרַח הַזְּמָן וְצִיץ נִזְרוֹ • צִבְיוֹ תֵלֵךְ לִקְרַאת נְחָשִׁים • עָשְׂתָה לָהּ אַפִּרְיוֹן בְּנַפְשָׁהּ הַחֲשׁוּבָה • רְפִידָתָהּ זָהָב סוֹדוֹת הַמֶּרְכָּבָה • תּוֹכָהּ רָצוּף אַהֲבַת כָּל שְׁלֵמוּת • הִיא יַחַס וְצוּרָה לַנְּעִימוּת • וְהִיא תְּנַהֲגֵנוּ עַל‑מוּת • מַה יָּפְיָהּ וּמַה טּוּבָהּ • הַשֵּׂכֶל מְסֻבָּב וְהִיא סִבָּה • הִיא לָעֲנָוָה וְלַנְּזִירוּת צוּרָה • צְאֶינָה וּרְאֶינָה בְּנוֹת צִיּוֹן בָּעֲטָרָה • שֶׁעִטְּרָה לָהּ הַנָּוָה וְהַמְעֻנָּגָה • בְּשׁוֹשַׁנֵּי הַנְּזִירוּת סוּגָה • כָּזֹאת וְכָזֹאת אָמְרוּ עָלַיִךְ הַצְּבִיּוֹת פַּעַם וּפַעֲמַיִם • לֹא יוֹם אֶחָד וְלֹא יוֹמַיִם • וְעַתָּה מַה תֹּאמְרִי הָעֵת כַּאֲשֶׁר הִכְזַבְתְּ תוֹחַלְתָּם • וּמִשְׁאֲרוֹת אֲשָׁמַיִךְ צְרוּרוֹת בְּשִׂמְלוֹתָם • שׁוּבִי שׁוּבִי הַצְּבִיָּה וְתַחַת יְדֵי הַנְּזִירוֹת הִתְעַנִּי • וְלֶאֱלִילֵי הַחֵשֶׁק אַל תִּפְנִי • וְלֹא תִהְיִי לְאִישׁ וְלֹא תִּזְנִי • וְכִי תִמְצְאִי אִישׁ לֹא תְבָרְכֶנּוּ • וְאִם יִקְרָאֵךְ חֵשֶׁק לְשָׁלוֹם אַל תַּעֲנֶנּוּ • וְתֵשְׁבִי בְסֵתֶר עֶלְיוֹן וּבְצֵל שַׁדַּי תִּתְלוֹנְנִי • וּבִצְדָקָה תְּכוֹנְנִי • רַחֲקִי מֵעֹשֶׁק כִּי לֹא תִירָאִי • אִמְרִי לָעֲנָוָה תְנִי וְלַנְּזִירוֹת אַל תִּכְלָאִי • כִּי רָם יְיָ וְשָׁפָל יִרְאֶה • וְעָלַיִךְ יִזְרַח יְיָ וּכְבוֹדוֹ עָלַיִךְ יֵרָאֶה •



וַיְהִי כַּהֲתִמִּי לְחַבֵּר הָאִגֶּרֶת • אֲשֶׁר מִתּוֹכְחוֹת מוּסָר מְחֻבֶּרֶת • שְׁלַחְתִּיהָ בְיַד הַנַּעַר אֶל אַרְמוֹן הַבִּירָה • אֲשֶׁר שָם הַגְּבִירָה • וַיְהִי כִּרְאוֹת הַגְּבִירָה אֶת הָאִגֶּרֶת • וּבָהּ תּוֹכַחַת מְגֻלָּה מֵאַהֲבָה מְסֻתֶּרֶת • וַתֹּאמַר זֹאת חֶרֶב חָלָל הַגָּדוֹל הַחוֹדֶרֶת • וְלֹא קָמָה עוֹד רוּחַ בָּהּ • וַתִּתְעַצֵּב אֶל לִבָּהּ • וַתֵּבְךְּ וַתִּסְפֹּד בְּקוֹל מַר • וַתִּשָּׂא אֶת קוֹלָהּ וַתֹּאמַר • אִי עֲצוּבַת רוּחַ וְהַשְּׁדוּדָה • הַסְּכָלָה וְהַבּוֹגֵדָה • הָיִית עַל כָּל הַבְּתוּלוֹת נִכְבָּדָה • עֲשָׂאֵךְ הַזְּמָן בְּעוֹלָמֵךְ יְחִידָה • בְּתִתּוֹ אוֹתָךְ רִאשׁוֹנָה לַחֲסִידוּת וְלִקְדֻשָּׁה • וְעַתָּה יִשְׁאֲלוּ בַעֲדֵךְ אַיֵּה הַקְּדֵשָׁה • אוֹי לָהּ לְאוֹתָהּ בּוּשָׁה • לֹא הֶאֱמַנְתִּי כִּי הַחֵשֶׁק הָאַכְזָר בִּיצוּרַי יִלָּחֵם • וּמֵעֲבֹד אֵל חַי יַדִּיחֵם • וְלָלֶכֶת אַחֲרֵי תַאֲוַת הַזְּמָן יַנְחֵם • וַיָּקוּמוּ כָל אַחֶיהָ וְכָל אַחְיוֹתֶיהָ לְנַחֲמָהּ וַתְּמָאֵן לְהִתְנַחֵם • וַתֹּאמֶר כִּי עוֹד מְעַט וְלֹא יַעֲבֹר • אֶל שְׁאוֹל תּוּרַד אֶל יַרְכְּתֵי בוֹר • וַיְהִי כַּאֲשֶׁר דִּמְּתָה הַגְּבִירָה כֵּן הָיָתָה • לֶחֶםֶ לֹא אָכְלָה וּמַיִם לֹא שָׁתָתָה • עַד אֲשֶׁר רוּחָה סָפָתָה • אָרוּר אַפָּהּ כִּי עָז וְעֶבְרָתָהּ כִּי קָשָׁתָה • וַתָּמָת הַגְּבֶרֶת • אֲשֶׁר הָיְתָה בְרֹאשׁ הַזְּמָן לַעֲטֶרֶת • וְעָלְתָה עַל כָּל הַגְבִירוֹת לִתְהִלָּה וּלְשֵׁם וּלְתִפְאֶרֶת • וְכִרְאוֹתִי כִי מוֹתָהּ סִבַּבְתִּי • אֶל בֵּית הַשָּׂר עוֹד לֹא שַׁבְתִּי • וָאֶלְבַּשׁ כְּלֵי גוֹלָה • וָאָקוּם וָאֵלֵךְ אֶל הַמָּקוֹם אֲשֶׁר הָיָה שָׁם אָהֳלִי בַתְּחִלָּה • וָאֲקוֹנֵן עַל הַגְּבִירָה • בְּלֵב נִשְׁבָּר וְרוּחַ מְצֵרָה:


  


נָפְלָה עֲטֶרֶת הַזְּמָן כִּי נָפְלָה



עָשׁ מֵעֲלֵי כִסֵּא יְקָרָהּ מִזְּבוּל



יָבְשׁוּ פְרָחֵינוּ וְאֻמְלַל כָּל‑יְבוּל



צִיצַת אֲדָמָה אָבְלָה אַף נָבְלָה


  


קָדְרוּ פְנֵי שַׁחַק וְתֵבֵל אֻמְלְלָה



אוֹיָה לְיֶרַח זִו אֲשֶׁר נֶהְפַּךְ לְבוּל



רָבְתָה אֲנָחָה בּוֹ וְעָבְרָה הַגְּבוּל



עַל‑עָשׁ וּמַעְלָתָהּ אֲשֶׁר בּוֹ שָׁפְלָה


  


עָשׁ בַּעֲבוּרֵךְ אֹהֲבָה הַמָּוְתָה



מַה מָּתְקָה מִיּוֹם אֲשֶׁר בָּךְ דָּבְקָה



מַה נִּפְלְאָה מָוֶת בְּעָשׁ הִתְחַבְּרָה


  


לֹא הַתְּמוּתָה מַעֲלָתֵךְ מֹתְתָה



אַךְ הַתְּמוּתָה הִיא בְמִתְקֵךְ מָתְקָה



עַל‑צוּף וְיַעְרַת הַדְּבָשׁ הִתְגַּבְּרָה •
