


	
	מחברות עמנואל
	



מחברות עמנואל – הַמַּחְבֶּרֶת הָעֶשְׂרִים וְשֶׁבַע / עמנואל הרומי


אַָמַר הַמְּחַבֵּר: הָיִיתִי עִם הַשָּׂר בִּיְמֵי הַנְּעוּרִים בְּעִיר וִיאֶנָּ”ה / וּבִהְיוֹתֵנוּ מְשׁוֹטְטִים בִּרְחוֹבוֹתֶיהָ אַחַת הֵנָּה וְאַחַת הֵנָּה / בַּחַגִּים וּבַמּוֹעֲדִים / רָאִינוּ חֶבְרַת אֲנָשִׁים נוֹעָדִים / חוֹשְׁבִים שֶׁהֵם מְדַבְּרִים מִדִּבְרֵי חָכְמָה / וְאֵין אִתָּם יוֹדְעִים עַד מָה / זֶה יִשְׁאַל וְזֶה יַעֲנֶה / זֶה יְחָרֵף וְזֶה יְגַנֶּה / זֶה יִצְעַק וְזֶה אֵלָיו יִתְאַנֶּה / קוֹל מִלְחָמָה בַּמַּחֲנֶה / אֵין בֵּינֵיהֶם הַסְכָּמָה מִדָּבָר / קְצָתָם מְסַפְּרִים מַעֲשִׂים שֶׁהָיוּ כְבָר / וּקְצָתָם אוֹמְרִים מֶלֶךְ פְּלֹונִי גָבָר / וּבָא בָאֲרָצוֹת וְשָׁטַף וְעָבָר / וּקְצָתָם אוֹמְרִים אֲנִי הָיִיתִי בִנְהַר כְּבָר / וּקְצָת נְדִיבֵי עַמִּים נֶאֱסָפוּ / וּלְדַבֵּר חֲמֻדּוֹת נִכְסָפוּ / וַיִּתְנַבְּאוּ וְלֹא יָסָפוּ / וּקְצָתָם לֹא בְדַעַת יְדַבֵּרוּ וְדִבְרֵיהֶם לֹא בְהַשְׂכֵּל/ וַיְהִי בְלִבִּי כְאֵשׁ עָצוּר בְּעַצְמוֹתַי נִלְאֵיתִי כַלְכֵּל /



וָאֹמַר אֶל הַשָּׂר הֲרָאִיתָ מָה הֵם עוֹשִׂים / כְּאִלּוּ הַגְּרָנוֹת שׁוֹסִים / אוֹ כְגִבּוֹרִים בַּמִּלְחָמָה בוֹסִים / אֵין בָּהֵם אִישׁ דְּבָרָיו בְּמִשְׁפָּט יְכַלְכֵּל / וְכָל אֶחָד מֵהֵם עֲצַת חֲבֵרוֹ יְסַכֵּל / עַד מָה אֲנַחְנוּ מַחְשִׁים / וְלֹא נִהְיֶה אֶל הַמַּעֲרָכָה נִגָּשִׁים / וְנוֹצִיא מִפִּינוּ הַתַּרְשִׁישִׁים / הָבָה נֵרְדָה וְנַבְלָה שָׁם שְׂפָתָם / וְנַרְאֶה אֲלֵיהֶם כִּי נִבְעֲרָה עֲצָתָם /וְנֵדְעָה מָה אַחֲרִיתָם / וְנַשְׁמִיעֵם פְּנִינֵי אִמְרוֹתֵינוּ/ וּנְשַׂחֶק בָּם כַּצִּפּוֹר וְנִקְשְׁרֵם לִנְעֲרוֹתֵינוּ /



וַיֹּאמֵר אֵלַי הַשָּׂר אֱלֹהִים יַעֲנֶה אֶת שְׁלוֹמֶךָ / וְעַל טוּבְךָ יִזָּכֵר שְׁמֶךָ / כָּמוֹנִי כָמוֹךָ כְּעַמִּי כְעַמֶּךָ / אֶל כָּל אֲשֶׁר תַּטֶּנִי / תִּקְרָא וְאֹמַר הִנֵּנִי /



וָאֹמַר אֶל הַשָּׂר וּמַה הִיא הַחָכְמָה אֲשֶׁר בָּהּ


נְדַבֵּר / וַחֲיָלִים עָלֶיהָ נְגַבֵּר /


וַיֹאמֶר הַשָּׂר אֶמְצָא חֵן בְּעֵינֶיךָ וּנְדַבֵּר מֵחָכְמַת הַמַּזָּלוֹת / וְאָז תִּרְאֶה אִם אָשִׁיב הַצֵּל אֲחוֹרָנִּית עֶשֶׂר מַעֲלוֹת / וְאִם אֵדַּע שַׁעֲרֵי מָכוֹן וּוִילוֹן / וְאִם אֹמַר שֶׁמֶשׁ בְּגִבְעוֹן דּוֹם וְיָרֵחַ בְּעֵמֶק אַיָּלוֹן / כִּי אֲנִי הוּא הַיֹּושֵׁב בְּחָכְמַת הַמַּזָּלוֹת רִאשׁוֹנָה / וְיַד כָּל הָעָם בָּאַחֲרוֹנָה / וְאַל נָא אֶשָּׂא פְּנֵי אִישׁ וְאֶל אָדָם לֹא אֲכַנֶּה / כִּי אֲנִי אֶהֱרֹס וְלֹא יִבָּנֶה / אֲשֶׁר אִם יוּכַל אִישׁ לִמְנוֹת אֶת עֲפַר הָאָרֶץ גַּם חָכְמָתִי כֹה תִּמָּנֶה / וְלָכֵן קְרָא וְאָנֹכִי אֶעֱנֶה / וְאֶל חָכְמָה אַחֶרֶת לֹא נִפְנֶה / וְאָז תִּרְאֶה כִּי הָאֲנָשִׁים הָאֵלֶּה לֹא יִמְצְאוּ מַעֲנֶה / כִּי הַמָּקוֹם מְקוֹם מִקְנֶה / וְיִשְׁמְרוּ לְפִיהֶם מַחֲסוֹם/ כִּי לַיְלָה מֵחָזוֹן וַחֲשֵׁכָה לָהֶם מִקְּסוֹם /



וָאֹמַר אֶל הַשָּׂר מָשְׁכֵנִי אַחֲרֶיךָ / הֵן לוּ יְהִי כִדְבָרֶיךָ /



וַנִּקְרַב אֶל הַזְּקֵנִים וְהַשָׂרִים / וָאֹמַר אֲלֵיהֶם שִׁמְעוּ נָא הַמּוֹרִים / הַקֵּץ לְדִבְרֵי רוּחַ/ אוֹ יִהְיֶה הָאָדָם אוֹ יִהְיֶה יַבְרוּחַ / הַיּוֹם הַזֶּה הוּא לַאדֹנָי קֹדֶשׁ הִלּוּלִים / לֹא עֵת לְדַבֵּר דִבְרֵי הֲתֻלִּים / וּלְסַפֵּר חֲלוֹמוֹת וַהֲבָלִים / אוֹ נְדַבֵּר מִדְבַר חָכְמָה רָמָה אוֹ נַחֲרִיש / הֲלֹא יְדַעְתֶּם כִּי מְרַדֵּף רֵיקִים יִשְׂבַּע רִישׁ / הִנֵּה מַה טּוֹב וּמַה נָּעִים כִּי מַה שֶּׁיָּרַשְׁנוּ מִן הַחָכְמָה אוֹתוֹ נוֹרִישׁ /



וַיַּחֲרִישׁוּ הָעָם וְלֹא עָנוּ אוֹתִי דָבָר / וְהַזָּדוֹן בְּלֵב קְצָתָם גָּבָר / וְחָלַף בָּהֶם וְשָׁטַף וְעָבָר /



וַיֹּאמֶר הַשָּׂר תַּחַת דַּבֵּר הַיּוֹם דִּבְרֵי מַהֲתַלּוֹת / אַחֲרֵי אֲשֶׁר חָדְלוּ הַקּוֹלוֹת / נְדַּבֵּר מִן הַחָכְמוֹת הַמְּהֻלָּלוֹת / וְנִזְכּוֹר טֶבַע שְׁנֵים עָשָׂר מַזָּלּוֹת / וְתַחַת לִשְמֹעַ דִבְרֵי הָאֵלָה וְהָאַלּוֹן אֲשֶׁר בַּשַּׁלֶּכֶת / וַעֲתַלְיָהוּ הַמּוֹלֶכֶת / נְדַבֵּר בְּטֶבַע שִׁבְעַת כּוֹכְבֵי לֶכֶת / וּבְסוֹדוֹת הָאֵל אֲשֶׁר גְּאָלָנוּ / וּמִבּוֹר שָׁאוֹן וְטִיט הַיָּוֵן הֶעֳלָנוּ / אֲשֶׁר שָׁמַעְנוּ וַנֵּדָעֵם וַאֲבוֹתֵינוּ סִפְּרוּ לָנוּ /



וָאֹמַר אֶל הָשָּׂר מָשְׁכֵנִי אַחֲרֶיךָ / פְּתַח פִּיךָ וְיָאִירוּ דְבָרֶיךָ /



וַיֹּאמֶר הַשָּׂר: מִי כְשַׁבְּתַי אֶל הַחָכְמָה הָאֱלֹהִיָּה / וּמִי כַצֶּדֶק מְפַתֵּחַ שְׁלָחִיָּה / וּמִי כְמַאְדִּים עַל הַתְּנוּעָה וְהַנְּטִיָּה / וּמִי כַשֶּׁמֶש עַל הַמַּלְכוּת וְהָעֲלִיָּה / וּמִי כַנֹּגַהּ עַל הַיֹּפִי וְהָרְוָיָּה / וּמִי כַכּוֹכָב עַל הַכְּתָב וְהַמִּכְתָּב יָדוֹ הוֹיָה / וּמִי כַלְּבָנָה זָבַת חָלָב וּדְבַשׁ הִיא וְזֶה פִּרְיָהּ /



וָאַעַן וָאֹמַר: מִי כְשַׁבְּתַי בַּיָּם דַּרְכּוֹ / וּמִי כַצֶּדֶק יָצִיץ מֵחֲרַכּוֹ/ וּמִי כְמַאְדִּים דָּם בְּשָׁפְכוֹ / וּמִי כַשֶּׁמֶש עַל כּוֹכְבֵי אֵל בְּמָלְכוֹ / וּמִי כַנֹּגַהּ בַּחֵשֶׁק לַעֲרֹךְ עֶרְכּוֹ / וּמִי כַכּוֹכָב שֵׁבֶט סוֹפֵר בְּתוֹמְכוֹ / וּמִי כַיָּרֵחַ חָלַק מִשֶּׁמֶן חִכּוֹ /



וַיֹּאמֶר הַשָּׂר: מִי כְשַׁבְּתַי מַכֶּה עַמִּים בְּעֶבְרָה / וּמִי כַצֶּדֶק יַדְרִיכֵם בְּדֶרֶךְ יְשָׁרָה/ וּמִי כְמַאְדִּים יוֹצִיא חַרְבּוֹ מִתַּעֲרָהּ / וּמִי כַשֶׁמֶש תּוֹפִיעַ עָלֵינּו נְהָרָה / וּמִי כַנֹּגַהּ בְּאֵשׁ הַחֵשֶׁק כְּמוֹ תַנּוּר בּוֹעֵרָה / וּמִי כַכּוֹכָב אֶצְבַּע אֱלֹהִים לַחֲרוֹת דִּבְּרוֹת עֲשָׂרָה/ וּמִי כַלְּבָנָה תוֹצִיא מִמְזָרִים קָרָה /



וָאַעַן וָאֹמַר: מִי כְשַׁבְּתַי עוֹצֵר בְּעַד הַטִּפָּה / וּמִי כַצֶּדֶק צִפָּה רֹאשָׁהּ וְחִפָּה / וּמִי כְמַאְדִּים מְזִיחַ אֲפִיקָהּ רִפָּה / וּמִי כַשֶּׁמֶשׁ רוּחַ חַיִים בִּכְנָפֶיהָ שָׁאֲפָה / וּמִי כַנֹּגַהּ צוּרָתָהּ רָקַם וְיִפָּה / וּמִי כַכּוֹכָב צִיֵּר רֹאשׁ וְזָנָב אַגְמוֹן וְכִפָּה / וּמִי כַלְּבָנָה יוֹצִיא חָתָן מֵחֻפָּה /



וַיֹאמֶר הַשָּׂר: מִי כְשַׁבְּתַּי עַל הַחֲמוֹרִים / ּוּמִי כַצֶּדֶק עַל הַגְּמָלִים וְהַכָּרִים / וּמִי כְמַאְדִּים עַל הָעִזִּים וְהַשְּׂעִירִים / וּמִי כַשֶּׁמֶשׁ עַל הַסּוּסִים הָאַבִּירִים / וּמִי כַנֹּגַהּ עַל הַפְּרָדִים הָאֲרוּרִים / וּמִי כַכּוֹכָב עַל הַצֹּאן וְהָעֲדָרִים / וּמִי כַלְּבָנָה עַל הַשְּׁוָרִים/



וָאַעַן וָאֹמַר: מִי כְשַׁבְּתַי מֵעִיד עַל שַׁבְּתוֹתַי / וּמִי כַצֶּדֶק / לְחַזֵּק כָּל בֶּדֶק / וּמִי כְמַאְדִּים / לִשְׁפֹּךְ דָּם יַקְדִּים / וּמִי כַחַמָּה / לְחַדֵּשׁ פְּנֵי אֲדָמָה / וּמִי כַנֹּגַהּ / סָבִיב לוֹ נֹגַהּ / וּמִי כַכּוֹכָב / בֵּין רַגְלֵי הַשֶּׁמֶשׁ כָּרַע וְשָׁכַב / וּמִי כַיָּרֵחַ / נָטָה לָלוּן כְּאוֹרֵחַ/



וַיֹאמֶר הַשָּׂר: אֵין כְּשַׁבְּתַי עוֹקֵר מַלְכֻיּוֹת וּמַחֲרִיב / וְאֵין כַּצֶּדֶק צַדִּיק וְצֶדֶק יַקְרִיב / וְלֹא כְמַאְדִּים מְחַרְחֵר רִיב / וְלֹא כַשֶּׁמֶשׁ מַזְרִיחַ וּמַעְרִיב / וְאֵין כַּנֹּגַהּ בַּחֵשֶׁק וּבִזְנוּנָיו / וְלֹא כַכּוֹכָב קֶסֶת הַסּוֹפֵר בְּמָתְנָיו / וְאֵין כַּלְּבָנָה דוֹבֵר דָּבָר דָּבוּר עַל אָפְנָיו /



וָאַעַן וָאֹמַר: אֵין כְּיוֹם שַׁבְּתַי בִּימוֹת הַשָּׁנָה / וְלֹא כַצֶּדֶק יִהְיֶה עִם הַצֶּדֶק בֶּאֱמוּנָה / וְלֹא כְמַאְדִּים מַחֲזִיק מָגֵן וְצִנָּה / וְלֹא כַשֶּׁמֶשׁ עֲדִינָה / וְאֵין כַּנֹּגַהּ זוֹנָה/ וְלֹא צוּרָה בָרָקִיעַ כְּצוּרַת כּוֹכָב קְטַנָּה / וְלֹא מָהִיר כַּלְּבָנָה /



וַיֹאמֶר הַשָּׂר: אֵין כְּשַׁבְּתַי מַשְׂכִּיל וְנָהִיר / וְלֹא כַצֶּדֶק זָהִיר / וְלֹא כְמַאְדִּים זֵד יָהִיר / וְלֹא כַשֶּׁמֶשׁ זוֹרֵחַ וּמַזְהִיר / וְלֹא כַנֹּגַהּ בַּשְּׁחָקִים מָהִיר / וְלֹא כַכּוֹכָב סוֹפֵר מָהִיר / וְלֹא כַלְּבָנָה בֵּין שׁוּרוֹתַיִם יַזְהִיר /



וָאַעַן וָאֹמַר: אֵין כְשַׁבְּתַי עָצֵל וּפִסֵּחַ / וְלֹא כַצֶּדֶק נוֹצֵחַ / וְלֹא כְמַאְדִּים רוֹצֵחַ / וְלֹא כַשֶּׁמֶשׁ שׁוֹאֵף וְזוֹרֵחַ / וְלֹא כַנֹּגַהּ חֹשֵׁק וְשָׂמֵחַ / וְלֹא כַכּוֹכָב סוֹפֵר וּפִקֵּחַ / וְלֹא אִישׁ דְּבָרִים וּמֵלִיץ כַּיָּרֵחַ /



וָיֹאמֶר הַשָּׂר: שְׁנוֹת שַׁבְּתַי לִקְבוּרוֹת / וּשְׁנוֹת צֶדֶק צֶדֶק לִרְדֹּף וּלְהוֹרוֹת / וּשְׁנוֹת מַאְדִּים לָנוּעַ אַחַר הַסְּחוֹרוֹת / וּשְׁנוֹת חַמָּה לִרְדֹּף הַשְּׂרָרוֹת / וּשְׁנוֹת נֹגַהּ לִשְׁגּוֹת בְּזָרוֹת / וּשְׁנוֹת כּוֹכָב לִלְמוֹד הַתּוֹרוֹת / וּשְׁנוֹת הַלְּבָנָה בְּאֵשׁ הַיַּלְדּוּת כְּמַרְאֶה הַלַּפִּידִים בּוֹעֲרוֹת /



וָאַעַן וָאֹמַר: הַשֶּׁמֶשׁ כְּחָתָן בְּתוֹר הַמַּעֲלָה / וְהַלְּבָנָה תִתְכַּס בְּצָעִיף וְתִתְעַלֵּף כְּכַלָּה / וַחֲמֵשֶׁת כּוֹכְבֵי אֵל כְּשׁוֹשְׁבִינִים מְשׁוֹרְרִים בְּקוֹל הַמֻלָּה / וּשְׁנֵים עָשָׂר נְשִׂיאִים לְעֻמָּתָם יָחוּלוּ בַמְּחוֹלוֹת וְחוֹזְרִים חֲלִילָה / וּצְבָא הַשָּׁמַיִם מִימִינָם וּמִשְׂמֹאלָם מְרִיעִים תְּרוּעָה גְדוֹלָה / וְהָעִיר שׁוּשָׁן צָהֲלָה /



וַיֹּאמֶר הַשָּׂר: הַשֶּׁמֶשׁ כַּמֶּלֶךְ וְהַלְּבָנָה גְבִירָתוֹ / נֹגַהּ פִּילַגְשׁוֹ וְאוֹמַנְתּוֹ / וְכוֹכָב סוֹפֵר מָהִיר בִּמְלַאכְתוֹ / וְשַׁבְּתַי יוֹדֵעַ בִּינָה לָעִתִּים / וְצֶדֶק דָּן דִּין אֱמֶת לַאֲמִתּוֹ / וּמַאְדִּים שַׂר הַטַּבָּחִים וְהַמַּזָּלוֹת שָׂרֵי צִבְאוֹת מַלְכוּתוֹ / וְכָל צְבָא הַשָּׁמַיִם עַמּוֹ וְנַחֲלָתוֹ / יוֹשְׁבִים תַּחַת גַּפְנוֹ וְתַחַת תְּאֵנָתוֹ /



וָאַעַן וָאֹמַר: מִי כַטָּלֶה מוֹצִיא רוּחַ מֵאוֹצְרוֹתָיו / וְכַשּׁוֹר יָשֵׁת חֹשֶׁךְ סִתְרוֹ וְסֻכָּתוֹ סְבִיבוֹתָיו / וְכַתְּאוֹמִים עָשָׂה לָרוּחַ מִשְׂקָל וּמַיִם תִּכֵּן בְּמִדּוֹתָיו / וְכַסַּרְטָן מַשְׁקֶה הָרִים מֵעֲלִיּוֹתָיו / וְכָאַרְיֵה חָזָק וְגִבּוֹר בְּמַמְשְׁלוֹתָיו / וְכַבְּתוּלָה יֹרְדָה הָעַיְנָה לִשְׁאֹב גֶּשֶׁם מִטְרוֹתָיו / וּכְמֹאזְנַיִם יִשְׁקֹל הָרֵי עוֹלָם וְגִבְעוֹתָיו / וְכָעַקְרָב בַּסּוּפָה וּבִשְׂעָרָה הֲלִיכוֹתָיו / וְכַקֶּשֶת יוֹרֶה יוֹרֶה וּמַלְקוֹשׁ בְּקַשְׁתּוֹתָיו / וְכַגְּדִי תֶאֱרַכְנָה לֵילוֹתָיו / וְכַדְּלִי קִיֵּם וְלֹא יַחֵל דְּבָרוֹ וּמוֹצָא שְׂפָתָיו / וְכַדָּגִים מְאַסֵּף לְכָל מַחֲנוֹתָיו/



וַיֹּאמֶר הַשָּׂר: טָלֶה עַל רֹאשׁוֹ מְשָׁלוֹ / וְשׁוֹר עַל צַוָּארוֹ עֹל סֻבֳּלוֹ / וּתְאוֹמִים יָדָיו רַב לוֹ / וְסַרְטָן הֶחָזֶה גְבוּלוֹ / וְאַרְיֵה הָאִסְטוֹמְכָא בְגוֹרָלוֹ / וְהַבְּתוּלָה הַטַּבּוּר


כֻּלּוֹ / וְלַמֹּאזְנַיִם הַבֶּטֶן וְחֵילוֹ / וְלָעַקְרָב הָאֲשָׁכִים וּמְחִי קֳבָלוֹ / וְלַקֶּשֶׁת יַרְכְּתֵי אָהֳלוֹ / גְדִי הַבִּרְכַּיִם יֵשְׁבוּ בְצִלּוֹ / וּדְלִי הַשֹּוֹקַיִם לְחֶבְלוֹ / וְדָגִים הָרַגְלַיִם יַחֲנוּ בְאֹתוֹת עַל דִּגְלוֹ /


וָאַעַן וָאֹמַר: יֹאמַר טָלֶה עָלִיתִי לְמַעְלָה מִכּוֹכְבֵי אֵל בְּעוֹנָתִי / וְלֹא עֲזָרַנִי אֶל זֶה קַלּוֹתִי /



וַיֹּאמֶר הַשָּׂר: יֹאמַר שׁוֹר עָלִיתִי אֶל הַשְּׁחָקִים בְּסִבַּת עֹל סֻבֳּלִי / וְלֹא עֲזָרַנִי אֶל זֶה גְבוּרָה הָיְתָה לִי /



וָאַעַן וָאֹמַר: יֹאמַר תְּאוֹמִים הָיִיתִי בַּשַּׁחַק בְּסִבַּת הָאֲחָוָה / וּבִלְעָדֶיהָ הָיִיתִי לִפְנֵי מְלָכִים לְרַאֲוָה /



וַיֹּאמֶר הַשָּׂר: יֹאמַר סַרְטָן הָיִיתִי בַשָּמַיִם לִהְיוֹת לָרְשָׁעִים תַּבְנִית / לְהָסֵב אֶת לִבָּם אֲחוֹרַנִּית /



וָאַעַן וָאֹמַר: יֹאמַר אַרְיֵה לִמְדוּ מִמֶּנִּי צְנִיעוּת וּגְבוּרָה כִּי אֶצְלִי בְּתוּלָה וְלֹא גָעַרְתִּי בָהּ לְעֹולָם גְעָרָה /



וַיֹּאמֶר הַשָּׂר: תֹּאמַר בְּתוּלָה אִתִּי אַהֲבַת חֶסֶד וְהַצְנַע לֶכֶת / כִּי לֹא פִּתּוּנִי שִׁבְעָה כוֹכְבֵי לֶכֶת /



וָאַעַן וָאֹמַר: יֹאמַר מֹאזְנַיִם פַּלְּסוּ מַעְגַּל רַגְלֵיכֶם בָּנַי / כִּי פֶלֶס וּמֹאזְנַיִם מִשְׁפָּט לַאדֹנָי /



וַיֹּאמֶר הַשָּׂר: יֹאמַר עַקְרָב אִם תִּהְיוּ לְרָעוֹת אַחֲרֵי רַבִּים / אֲנִי אֲיַסֵּר אֶתְכֶם בָּעַקְרַבִּים /



וָאַעַן וָאֹמַר: יֹאמַר קֶשֶׁת הוּשַׂמְתִּי בַשָּׁמַיִם לִרְאוֹת זִיקִים חִצִּים וָמָוֶת / לַעוֹזְבִים אָרְחוֹת יֹשֶׁר לָלֶכֶת בְּדֶרֶךְ צַלְמָוֶת /



וַיֹּאמֶר הַשָּׂר: יֹאמַר גְּדִי אֶרְבַּץ עִם אַרְיֵה וָקֶשֶׁת וְתִתְעַנַּג בַּדֶּשֶׁן נַפְשִׁי / לֹא יָרֵעוּ וְלֹא יַשְׁחִיתוּ בְכָל הַר קָדְשִׁי /



וָאַעַן וָאֹמַר: יֹאמַר דְּלִי בּלְבַבְךָ עֵצָה וּלְךָ דְלִי וְאֵיךְ לֹא תִדְלֶה / וַאֲנִי נְשִׂיאִים וָרוּחַ גֶּשֶׁם מִקְּצֵה הָאָרֶץ אַעֲלֶה /



וַיֹּאמֶר הַשָּׂר: יֹאמַר דָגִים לֹא הֶעֱלַנִי אֶל הַשָּׁמַיִם רַק הַשְּׁתִיקָה / וְעַתָּה בּוֹא כָתְבָה עַל לוּחַ לִבֶּךָ וְעַל סֵפֶר חֻקָּהּ /



וָאַעַן וָאֹמַר: כַּמָּה גְדוֹלָה מַעֲלַת הָעֲנָוָה / שֶׁהֲרֵי קָדַם טָלֶה בַּמַּזָּלוֹת לְכָל בַּעֲלֵי הָעֹצֶם וְהַגַּאֲוָה /



וַיֹּאמֶר הַשָּׂר: מַה גָּדְלָה מְאֹד מַעֲלַת בַּעַל הַיָּפָה וְאַף אִם לוֹ קְרָנַיִם / שֶׁהֲרֵי הוּשָׂם שׁוֹר מַקְרִין מַפְרִיס מַזָּל בַּשָּׁמַיִם /



וָאַעַן וָאֹמַר: מַה גָּדַל חֵשֶׁק הַחֹשְׁקִים הַנִּתְאָמִים / שֶׁהֲרֵי הוּשָׂם בַּשָּמַיִם מַזַּל תְּאוֹמִים /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַ אִם נְדִיבֵי עַמִּים יֵלְכוּ קְדוֹרַנִּית / אַחֲרֵי הֱיוֹת סַרְטָן בַּשָּׁמַיִם וְחוֹזֵר אֲחוֹרַנִּית /



וָאַעַן וָאֹמַר: מַה גָּדְלָה מַעֲלַת הָאִישׁ אֶל תַּכְלִית הַגְּבוּרָה יַגִּיע / שֶּהֲרֵי הוּשָׂם אַרְיֵה מַזָּל בָּרָקִיעַ /



וַיֹּאמֶר הַשָּׂר: מַה גָּדְלָה מַעֲלַת הַבְּתוּלָה / שֶׁהֲרֵי בֵּין מַזָּלוֹת הַשָּׁמַיִם תַּחֲנֶה בְאוֹתוֹת עַל דִּגְלָהּ /



וָאַעַן וָאֹמַר: אֵיךְ לֹא יִרְאוּ וְלֹא יְירְאוּ הָרְשָׁעִים וְהֵם בַּעֲלֵי הָעֵינַיִם / כִּי פֶלֶס וּמֹאזְנַיִם לַאדֹנָי בַּשָּׁמַיִם /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ אִם יַעֲלֶה לַשָּׁמַיִם שִׂיא הָרְשָׁעִים / אַחֲרֵי הֱיוֹת עַקְרָב לְמַעְלָה מִשִּׁבְעָה רְקִיעִים /



וָאַעַן וָאֹמַר: אֵין לִתְמֹהַּ עַל הַגְּזֵרוֹת הֶחָלוֹת בַּיַּבָּשֶׁת / אַחֲרֵי הֱיוֹת בַּשָּׁמַיִם הַמּוֹרִים בַּקָּשֶׁת /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ עַל בְּשַר הַגְּדִי אִם גָּדְלָה מַעֲלָתוֹ / אֲשֶׁר לְמַעְלָה מִכּוֹכְבֵי אֵל תַּחֲנוֹתוֹ /



וָאַעַן וָאֹמַר: לֹא יָדַעְתִּי אֵיךְ הֶעָבִים יִמְלְאוּ גֶשֶׁם וְיָרִיקוּ מַיִם / עַד רָאִיתִי מַזָּל דְּלִי בַשָּׁמַיִם /



וַיֹּאמֶר הַשָּׂר: לֹא יָדְעָה נַפְשִׁי אֵיךְ עָלוּ הַדָּגִים בַּשָּׁמַיִם / עַד רָאִיתִי כוֹכָבִים בַּדְּלִי יוֹצְאוֹת לִשְׁאֹב מָיִם /



וָאַעַן וָאֹמַר: אֵיךְ אִישׁ לֹא יְרַחֵף עַל גוֹזָלָיו וִירַחֵם / בִּרְאוֹתוֹ שָׁמַיִם עַיִשׁ עַל בָּנֶיהָ תַנְחֵם /



וַיֹּאמֶר הַשָּׂר: רָאִיתִי בַשַּׁחַק דְּלִי וְאֵין בּוֹר וְאֵין מַיִם / וְאוּלַי כִּי הוּא לִדְלוֹת וּלְהַעֲלוֹת נֶפֶשׁ הַצַּדִּיק בַּשָּׁמַיִם /



וָאַעַן וָאֹמַר: נִפְלֵאתִי בִרְאוֹתִי כִי אֵין צוּרַת עוֹף בַּמַּזָּלוֹת וְלֹא יָדַעְתִּי לַדָּבָר פֵּשֶׁר/ עַד אֲשֶׁר הִבְחַנְתִּי כִּי זֶה מִיִּרְאַת כּוֹכַב הַנֶּשֶׁר /



וַיֹּאמֶר הַשָּׂר: לֹא נִבְרָא מֹאזְנַיִם בָּרָקִיעַ / רַק לִהְיוֹתוֹ מֹרֶךְ בְּלֵב הָרְשָׁעִים מַגִּיעַ/



וָאַעַן וָאֹמַר: אֶתְמַהּ אַחֲרֵי הֱיוֹת שַׁבְּתַי לַבְּרוּאִים מִשְׁטַח חֲרָמִים / אֵיךְ עָלָה בְגוֹרָלוֹ מִבְחַר הַיָּמִים /



וַיֹּאמֶר הַשָּׂר: לֹא תָמַהְתִּי אֵיךְ הֻזְהֲרוּ מֵעֲשׂוֹת מְלָאכָה בַשַּׁבָּת אֲבוֹתַי / אַחֲרֵי אֲשֶׁר הִבְחַנְתִּי בְשִׂכְלִי עַצְלוּת שַׁבְּתַי /



וָאַעַן וָאֹמַר: אֵין לִתְמֹהַ אִם הַיָּמִים פָּרְשׂוּ רֶשֶׁת לְיַד מַעְגָּל / אַחֲרֵי הֱיוֹת שַׁבְּתַי וּמַאְדִּים בַּגַּלְגָּל /



וַיֹּאמֶר הַשָּׂר: פְּלוֹנִי יִתְחַבֵּר פַּעַם בַּחֹדֶשׁ עִם הָעֲדִינָה / כְּהִתְחַבְּרוּת שֶׁמֶשׁ פַּעַם בְּחֹדֶשׁ עִם הַלְּבָנָה /



וָאַעַן וָאֹמַר: פְּלוֹנִי צָנוּעַ כַּשֶּׁמֶשׁ הַחוֹנָה בְכָל שָׁנָה חֹדֶשׁ בַּבְּתוּלָה / וְלֹא שָׁת לִבּוֹ לַעֲשׂוֹת עִמָּהּ נְבָלָה /



וַיֹּאמֶר הַשָּׂר: יֹאמְרוּ בְקַדְרוּת הַשֶּׁמֶשׁ דִּבְרֵי חֲלוֹמוֹת וַהֲבָלִים / וְלֹא יֵדְעוּ כִי פְנֵי הַצְּבִיּוֹת אוֹרוֹ גוֹזְלִים /



וָאַעַן וָאֹמַר: לֹא עָלָה בְלִבִּי לְבְַזּוֹת וּלְשַׁקֵּץ אִישׁ קְצַר קוֹמָה וְעַנּוֹתוֹ / מִן הַיּוֹם אֲשֶׁר יָדַעְתִּי עֵרֶךְ גוּף כּוֹכַב חַמָּה וּמַעֲלָתוֹ /



וַיֹּאמֶר הַשָּׂר: רָאִיתִי פַעַם וּפַעֲמַיִם אוֹר הַלְּבָנָה גֻּנַּב / וְלֹא יָדַעְתִּי הַסִּבָּה כִּי תַשְׁקִיף מִן הָאֶשְׁנָב /



וָאַעַן וָאֹמַר: לְכוֹכְבֵי גַלְגַּל הַשְּׁמִינִי נֶפֶשׁ צְנוּעָה / אֲשֶׁר עָמְדָה בְתוּלָה בֵינֵיהֶם חֲמֶשֶׁת אֲלָפִים וּשְׁמֹנָה וּשְׁמֹנִים שָׁנָה וְאִישׁ לֹא יְדָעָהּ /



וַיֹּאמֶר הַשָּׂר: אֶחְשֹׁב הֱיוֹת כּוֹכְבֵי הַשָּׁמַיִם זְקֵנִים אוֹ סָרִיסִים / אִם שֹׁכְנָה בְתוּלָה עִמָּהֶם וְלֹא עָשׂוּ לָהּ כְּתֹנֶת פַּסִּים /



וָאַעַן וָאֹמַר: אֵין לִתְמֹהַּ אִם יֵשׁ בָּאָרֶץ כְּסִילִים לְמִינֵיהֶם / אַחֲרֵי רְאוֹתֵנוּ כוֹכְבֵי שָׁמַיִם וּכְסִילֵיהֶם /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ אִם מָעֲטוּ גְדוֹלֵי הַמַּעֲלָה וְרַבּוּ הַשּׁוֹבָבִים / יֵשׂ בָּרָקִיעַ שֶׁמֶשׁ אֶחָד וּשְׁנֵי כְסִילִים וּשְׁנֵי כְלָבִים /



וָאַעַן וָאֹמַר: אֶחְשֹׁב כִּי הַכֶּלֶב בַּשָּמַיִם יִהְיֶה / לִשְׁמֹר טָלֶה וּגְדִי מִפִּי אַרְיֵה /



וַיֹּאמֶר הַשָּׂר: אֶחְשֹׁב כִּי לֹא נִבְרָא אַרְיֵה בַשְּׁחָקִים / רַק לִשְׁמֹר בְּתוּלָה מִן הַחֹשְׁקִים /



וָאַעַן וָאֹמַר: לֹא הָיָה אַרְיֵה אֵצֶל בְּתוּלָה / רַק מִדְּאָגָה מִדָּבָר פֶּן תַּעֲשֶׂה נְבָלָה/



וַיֹּאמֶר הַשָּׂר: אִלּוּ אִישׁ עַל הַנָּשִׁים נֶאֱמָן יִהְיֶה / הָיָה אֵצֶל בְּתוּלָה תְּאוֹמִים וְלֹא אַרְיֵה /



וָאַעַן וָאֹמַר: מַה גָּדְלָה מַעֲלַת דָּגִים עַל כָּל הַמְּזוֹנוֹת / שֶׁהֲרֵי הוּשַׂם בַּמַּזָּלוֹת מְאַסֵּף לְכָל הַמַחֲנוֹת /



וַיֹּאמֶר הַשָּׂר: תָּמֵהַּ אֲנִי אֵיךְ לֹא יִשְׁמְרוּ הָאֻמּוֹת שַׁבְּתּוֹתַי / בִּרְאוֹתָם כִּי הָעֶלְיוֹן מִכּוֹכְבֵי לֶכֶת הוּא כּוֹכָב שַׁבְּתַי /



וָאַעַן וָאֹמַר: תַּמֵהַ אֲנִי אֵיךְ יִהְיֶה דֶרֶךְ הָרָשָׁע כִּמְשֻׂכַּת חֶדֶק / בִּרְאוֹתִי בִּשְׁמֵי הַשָּמַיִם כּוֹכַב צֶדֶק /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ אִם הַצַּדִּיקִים אֶת הָרְשָׁעִים מְנַדִּים / שֶׁהֲרֵי הוּשַׂם בַּשָּׁמַיִם צֶדֶק עַל מַאְדִּים /



וָאַעַן וָאֹמַר: לֹא יִתְגָּאֶה הַמֶּלֶךְ לָחֹן בְּנֵי תְמוּתָה לְעִתּוֹת / כִּי הִנֵּה הַשֶּׁמֶשׁ נִצּוֹצֵי חַשְׁמַלָּיו חִבְּקוּ אַשְׁפַּתוֹת /



וַיֹּאמֶר הַשָּׂר: אֵיךְ לֹא יִהְיֶה לַחֵשֶׁק אַחֲרֵי נִמְכַּר גְּאֻלָּה וּפִדְיוֹם / וְאוֹר נֹגַהּ הוֹלֵךְ וָאוֹר עַד נְכוֹן הַיּוֹם /



וָאַעַן וָאֹמַר: לֹא יֵבוֹשׁ שְׁפַל רוּחַ לְהִתְחַבֵּר עִם בַּעַל מַעֲלָה רָמָה / כִּי הִנֵּה כּוֹכָב הַקָּטֹן שֶׁבַּכּוֹכָבִים לֹא יִפָּרֵד מִן הַחַמָּה /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ אִם הַיְּעֵלָה הַלְּבָנָה תִהְיֶה עִם הַחֹשְׁקִים בֶּאֱמוּנָה / כִּי הִנֵּה לֹא יִתְחַבֵּר הַשֶּׁמֶשׁ פַּעַם בַּחֹדֶשׁ רַק עִם הַלְּבָנָה /



וָאַעַן וָאֹמַר: אֵין לִתְמֹהַּ אִם הָעַקְרָב לִשְׁפֹּךְ דָּם יַקְדִּים / אַחֲרֵי הֱיוֹת עַקְרָב בֵּית מַאְדִּים /



וַיֹּאמֶר הַשָּׂר: נִקְרְאוּ בַעֲלֵי קְרָנַיִם אַנְשֵׁי בּוֹגְדוֹת אֲשֶׁר לֹא עָמְדָה רַגְלָם בַּמִּישׁוֹר / מִפְּנֵי הֱיוֹת בֵּית נֹגַהּ הַמּוֹכֶרֶת גּוֹיִם בִּזְנוּנֶיהָ שׁוֹר /



וָאַעַן וָאֹמַר: אֵין לִתְמֹהַּ עַל מַאְדִּים אִם יָדָיו מְלֻמָּדוֹת לַקְּרָב / אַחֲרֵי הֱיוֹת בְּבֵיתוֹ עַקְרָב /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ אִם הַיָּפָה כַלְּבָנָה תַּחֲזֹר אֲחוֹרַנִּית לְעֵת זִקְנָה וְתִקְטַן / אַחֲרֵי הֱיוֹת בֵּית לְבָנָה סַרְטָן /



וָאַעַן וָאֹמַר: הַשֹּׁאֲלִים מַדּוּעַ הַחֹשְׁקִים זָלִים זָהָב מִכִּיס וְלֹא יַחֲמֹלוּ / כִּי בֵית נֹגַהּ מֹאזְנַיִם אֲשֶׁר כֶּסֶף בְּקָנֶה יִשְׁקֹלוּ /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ אִם הַדָּג הֵקִיא אֶת יוֹנָה עַל הַיַּבָּשָׁה / כִּי אַחֲרֵי הֱיוֹת דָּגִים בֵּית צֶדֶק הָיָה לָהֶם הַצֶּדֶק לְמוֹרָשָׁה /



וָאַעַן וָאֹמַר: אֵין לִתְמֹהַּ מִן הַיָּרֵחַ אִם הוּא לִפְעָמִים בְּמַהֲלָכוֹ נָזוֹר / שֶׁהֲרֵי בֵיתוֹ סַרְטָן וְהוּא אֲחוֹרָנִית נָזוֹר /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ אִם הַשֶּׁמֶשׁ כְּמֶלֶך עַל תֵּבֵל יִהְיֶה / אַחֲרֵי הֱיוֹת בֵּיתוֹ אַרְיֶה /



וָאַעַן וָאֹמַר: אֵין לִתְמֹהַּ עַל הָאַרְיֵה אִם הוּא מֶלֶךְ עַל חַיּוֹת הָאֲדָמָה / אַחֲרֵי הֱיוֹת אַרְיֵה בֵית חַמָּה /



וַיֹּאמֶר הַשָּׂר: אֵין לִתְמֹהַּ אִם הַבְּתוּלָה תִתְיַפֶּה כְכַלָּה / אַחֲרֵי הֱיוֹת בֵּית כּוֹכַב חַמָּה בְּתוּלָה /



וָאַעַן וָאֹמַר: תִּדְמֶה הַשֶּׁמֶשׁ לְחָתָן יְכַהֵן פְּאֵר בְּגַלְגַּלֶּיהָ / וְהַלְּבָנָה לְכַלָּה תַעְדֶּה כֵלֶיהָ /



וַיֹּאמֶר הַשָּׂר: אֶרְאֶה חֲמֶשֶׁת כּוֹכְבֵי הַנְּבוּכָה / כְּשׁוֹשְׁבִינִים הָעוֹמְדִים עַל הַבְּרָכָה/



וָאַעַן וָאֹמַר: אֶרְאֶה שְׁתֵּים עֵשְׂרֶה צוּרוֹת אֲשֶׁר בַּאֲפֻדַּת גַּלְגַּל הַמַּזָּלוֹת / לְדִמְיוֹן שָׁרִים וְחוֹלְלִים יָחוּלוּ בַמְּחוֹלוֹת /



וַיֹּאמֶר הַשָּׂר: אֶרְאֶה צְבַא הַשָּׁמַיִם מִימִינָם וּמִשְׂמֹאלָם כַּחוֹל / לְדִמְיוֹן הָעָם הַנִּקְבָּצִים לִרְאוֹת הַמָּחוֹל /



וַיְהִי כִשְׁמֹעַ הָעָם אֶת מְלִיצוֹתֵינוּ הַחַמֻדוֹת / וְהַחָכְמוֹת הָרָמוֹת וְהַנִּכְבָּדוֹת / עָמְדוּ מַרְעִידִים וּתְמֵהִים / כְּאִלּוּ שָׁם נִגְלוּ עֲלֵיהֶם הָאֱלֹהִים / וַיַּעֲמֹד אֶחָד מֵרָאשֵׁיהֶם בְּמִצְוַת אֲחֵיהֶם לְבָרְכֵנוּ / וּלְהוֹדוֹת אֶל מַעֲלָתֵנוּ / וַיֹּאמַר שָׁלוֹם לָכֶם שָׁלוֹם הַרְרֵי אֵל / רָאשֵׁי אַלְפֵי יִשְׂרָאֵל / בָּרוּךְ אֲשֶׁר חָלַק לָכֶם מִכְּבוֹדוֹ / וְיַאצֶל עֲלֵיכֶם מֵהוֹדוֹ/ לֹא יָדַעְנוּ בִלְתִּי הַיּוֹם כִּי שׁוֹכְנֵי בָתֵי חֹמֶר אֲשֶׁר כֶּעָפָר יְסוֹדָם / יֵדְעוּ מֵהַלְכוֹת צִבְאוֹת הַשָּׁמַיִם וְסוֹדָם / חָשַׁבְנוּ


כִּי הַשָּׁמַיִם שָׁמַיִם לַאדֹנָי וְהָאָרֶץ נָתַן לִבְנֵי אָדָם / אָמְנָם הַיּוֹם הַזֶּה רָאִינוּ נִפְלָאוֹת / וְשָׁמַעְנוּ מִכּוֹכְבֵי שַׁחַק נוֹרָאוֹת / לֹא נִשְׁאַל עַל מַעֲלָתְכֶם מוֹפֵת אוֹ אוֹת / אֵין לִתְמֹהַּ אִם הָיִינוּ בְעֵינֵיכֶם לְחֶרְפָּה וּלְקֶלֶס / שְׁקַלְתֶּם גִּבְעוֹת עוֹלָם בְּמֹאזְנַיִם וְהַרְרֵי עַד בַּפֶּלֶס / לְשֵׁמַעַ אֹזֶן שָׁמַעְנוּ שִׁמְעֲכֶם וְעַתָּה עֵינֵינוּ רָאָתָה / מִי לֹא יִירָא מִמַּעֲלַתְכֶם כִּי לָכֶם יָאָתָה / דִּבַּרְתֶּם הַיּוֹם נְגִידִים / הִנְנוּ לַאֲדֹנֵינוּ לַעֲבָדִים / וְעָלֵינוּ וְעַל פֹּעַל יָדֵינוּ תְצַוּוּ / כִּי לָכֶם הַלְּבָבוֹת יְקַוּוּ / וּמִדֶּשֶׁן חָכְמַתְכֶם יִרְוּוּ /


אַחֲרֵי כֵן קָם הָעָם וְהִשְׁתַחֲוּוּ / וַאֲנַחְנוּ הִפְקַדְנוּם לֵאלֹהִים /


וְהֵם נִשְׁאֲרוּ עַל מַעֲלָתֵנוּ תְמֵהִים / וַיְהִי כִי בָאנוּ אֶל הַמָּלוֹן וַיֹּאמֶר הַשָּׂר הֲרָאִיתָ אֵיךְ חָשַׂפְתִּי הַיּוֹם סוֹדִי / וְאֵיךְ בְּחָכְמוֹת הַמַּזָּלוֹת גָּבְרָה יָדִי / מִי יִתֵּן וְתִגְבַּר יָדִי בַּשִּׁירוֹת הַמְּהֻלָּלוֹת / כַּאֲשֶׁר גָּבְרָה בְּחָכְמַת הַמַּזָּלוֹת / אָמְנָם אִם אִישׁ מֵהֶם לְדַבֵּר בְּעִנְיַן הַשִּׁיר יֹאבֶה / אֶצְטָרֵךְ לָבֹא חֶדֶר בְּחֶדֶר לְהֵחָבֵא /


וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר אִם אִישׁ מֵהֶם יָבֹא וּמֵעִנְיַן הַשִּׁיר יִשְׁאַל / אָשִׂים מִלָּתוֹ לְאַל / וְאֶת גְאֻלָּתְךָ אָנֹכִי אֶגְאָל /



עוֹדֵנוּ בַּדְּבָרִים הָאֵלֶּה מְדַבְּרִים / וְהִנֵּה עֶשְׂרִים אֲנָשִׁים כְּאִישׁ אֶחָד חֲבֵרִים / בָּאוּ עָדֵינוּ / לַחֲשֹׂף סוֹדֵנוּ / וַיֹּאמְרוּ אֵלֵינוּ אֲדוֹנֵינוּ / הַיּוֹם רָאִינוּ מִכֶּם נוֹרָאוֹת / אֲשֶׁר לֹא יִשְׂבַּע אִישׁ לְדַבֵּר וְלֹא תִשְׂבַּע עַיִן לִרְאוֹת / וְעַתָּה יִיטַב בְּעֵינֵיכֶם לְהוֹדִיעֵנוּ עַל הַנְּכוֹנָה / אִם בִּמְלֶאכֶת הַשִּׁיר יְדֵיכֶם אֱמוּנָה /



וַיֹּאמֶר הַשָּׂר יֵשׁ וָיֵשׁ / הִנֵּה חֲבֵרִי לֹא יִתְבַּיֵּשׁ / לִמַּדְתִּיו מִן הַשִּׁירִים כְּבָר / מַה שֶּׁיֵּדַע לְהָשִׁיב שֹׁאֲלוֹ דָבָר / וְעַתָּה יַחֲשֹׂף כָּל אֶחָד מִכֶּם סוֹדוֹ / וְאָז תִּרְאוּ כִּי רַב חֵילוֹ וְכִי כַבִּיר מָצְאָה יָדוֹ /



וַיִּגַּשׁ הָרִאשׁוֹן וַיֹּאמַר אֲדֹנִי אֲנִי מֵעִיר אַחַת וּשְׁמָהּ בֵּיתְתַּר / וּבִשְׁכוּנָתִי עַלְמָה וּשְׁמָהּ אֶסְתֵר / דַּיָּהּ לְכָל הַמְּלָאכָה לַעֲשׂוֹת אוֹתָהּ וְהוֹתֵר / מָצְאָה חֵן בְּעֵינַי מִכָּל הַבְּתוּלוֹת / קוֹמָתָהּ דָמְתָה לְתָמָר וְשָׁדֶיהָ לְאַשְׁכּוֹלוֹת / הִיא יַעֲלַת חֵן וְיַעֲלַת אֲהָבִים / מַעֲשֶׂה חֹשֵׁב עָשָׂה אוֹתָהּ כְּרוּבִים / וְשָׁדֶיהָ כְּדָתָן וַאֲבִירָם יָצְאוּ נִצָּבִים / קָרְעוּ מֵעֲלֵיהֶם אֶת עֶדְיָם / לֹא יָדַעְתִּי אִם הֵם אֲבֵלִים לְעוֹרֵר בִּכְיָם / אוֹ לְהַרְאוֹת הָעַמִּים וְהַשָּׂרִים אֶת יָפְיָם / כִּי לְיָפְיָהּ בַּתֵּבֵל פָּרְצָה פָרֶץ / אֵין כָּמוֹהָ בְכָל הָאָרֶץ/ וְלוּ תֹאמַר שֶׁמֶשׁ בְּגִבְעוֹן דּוֹם וְיָרֵחַ בְּעֵמֶק אַיָּלוֹן / לֹא תִּמְצָא נֶגַע וְקָלוֹן / וְכָל אֶחָד מֵהֶם מַאֲמַר אֶסְתֵּר קִיֵּם / וְכָמוֹהַ לֹא נִרְאֵית מִיָּם עַד יָם / וְעַתָּה הוֹאֵל וְחַבֵּר עָלֶיהָ שִׁיר נִפְלָא / אֲשֶׁר בְּכֶתֶם אוֹפִיר לֹא יְסֻלָּא /



וָאָקוּם וָאֲחַבֵּר עָלָיו זֹאת הַשִּׁירָה / וַיִּשָׁבַע כִּי לֹא יֻתַּן סְגוֹר תַּחְתֶּיהָ וְלֹא יִשָּׁקֵל כֶּסֶף מְחִירָהּ /

  

וָאֶשָׂא מְשָלִי וָאֹמַר:

  

שְׁאֵלוּנִי עַל מָה שְׁדֵי זֹאת הָעַלְמָה



לְבוּשָׁם קָרָעוּ לְעוֹרֵר אֶת בִּכְיָם



הֲאִם קוֹדְרִים עַל אֵם יְהוּ כַּאְשֶׁר נִרְאֵם



קְרוּעֵי בִגְדֵיהֶם לְנַצֵּל אֶת עֶדְיָם



עֲנִיתִים קָרָעוּ וְרֹאשָׁם פָּרָעוּ



לְהַרְאוֹת הָעַמִּים וְהַשָּׂרִים יָפְיָם



וְאָמְרוּ מַה טִּבָּם עֲנִיתִים מַה טּוּבָם



דְּבָש וַחְלֵב זָבוֹת וְהִנֵּה לָךְ פִּרְיָם



הֲלֹא זֹאת הִיא אֶסְתֵּר לְחֹשְׁקִים תֵּעָתֵר



וְכָמֹהָ לֹא תֶחֱזֶה מִיָּם עַד יָם



וְאִלּוּ דּוֹם שֶׁמֶשׁ וְאִישׁ חֹשֶׁךְ יָמֵשׁ



תְּצַו הַיָּפָה מַאֲמַר אֶסְתֵּר קִיַּם .

  

וַיִּגַּשׁ הַשֵּׁנִי / וַיֹּאמַר אֲדֹנִי / יֵשׁ לִי שָׁכֵן וּשְׁמֹו שְׁלֹמֹה / וּמִדֵּי דַבְּרִי בוֹ וְאֶזְכֹּר מוּמוֹ / יֶחֱרַד לְבָבִי וְיִתַּר מִמְּקוֹמוֹ / וְלוּ הָיָה בַדּוֹר אִישׁ אֲשֶׁר רִשְׁעֵי הָאָרֶץ יַצְמִית / שְׁכֵנִי הָיְתָה אַחַת דָּתוֹ לְהַמִית / וְשֵׁם אִשְׁתּוֹ שְׁלוֹמִית / וּשְׁנֵיהֶם מֵאֲנוּ לָשׁוּב הֶחֱזִיקוּ בְתַרְמִית / כִּי הָאִישׁ הַהוּא לְחֵרוּפִים וּלְגִדּוּפִים / וְהַאִשָּׁה כֻלָּהּ נִאוּפִים / בָּהֶם כָּל דֹּפִי וְכָל גְּנָאי / רָעִים וְחַטָּאִים לַאֲדֹנָי / וְחָפַצְתִּי לְחַבֵּר שִׁיר אֶחֱשׂף סוֹדָם / לְהַגִּיד חַסְדָּם וְנִפְלְאוֹתָם לִבְנֵי אָדָם / וְדֶרֶךְ הַשִּׁיר אֵלַי צָר אֵין דֶּרֶךְ לִנְטוֹת יָמִין וּשְׂמֹאל / וְיָדַעְתִּי כִּי אֵין זוּלָתְךָ לִגְאֹל / וְלָכֵן חַבֵּר עֲלֵיהֶם שִׁיר כַּאֲשֶׁר עִם לְבָבֶךָ / וְאֵל שַׁדַּי יְבָרֵך אוֹתְךָ וְיַפְרְךָ וְיַרְבֶּךָ /



אָז נַפְשׁוֹ מִדֶּשֶׁן שִׁירִי רִוֵּיתִי / וְהִנַּבֵּאתִי כַאֲשֶׁר צֻוֵּיתִי /

  

וָאֶשָׂא מְשָלִי וָאֹמַר:

  

בְּעֵת תַּחְפֹּץ רְאוֹת שׁוּעָל מְרַמֶּה



וְאֵשֶׁת רֵעַ תְּיֻחַד לַנְּבָלָה



לְכָה אֶל בֵּית שְׁלֹמֹה אִישׁ שְׁלֹומִית



וְשָׁם תִּרְאֶה חֲצַר שׁוּעָל וּבָלָה .

  

וַיִּגַּשׁ הַשְּלִישִׁי וַיֹּאמַר אֲדֹנִי יֵש-לִי שָׁכֵן אֶחָד גִּבֵּן שׁוֹבָב / רוּם עֵינַיִם וּרְחַב לֵבָב / כְּאִלּוּ הָיָה מִבְּנֵי חֶבֶר הַקֵּינִי מִבְּנֵי חוֹבָב / יִגְאֶה כַשַּׁחַל וְשָכַח מוּמוֹ / בַּעֲבוּר הֱיוֹת הַמִּשְׂרָה עַל שִׁכְמוֹ / חָשַׁב כִּי זֶה הָאֶבְיוֹן לְבָשְׁתּוֹ / כִּי תֵשֵׁב בְּאֵיתָן קַשְׁתּוֹ / וּכְתֵפוֹ אֲשֶׁר מִשִׁכְמָהּ תִּפֹּל וְדַבָּשְׁתּוֹ / יִהְיֶה הָהָר חָמַד אֱלֹהִים לְשִׁבְתּוֹ / וְחָפַצְתִּי לְחַבֵּר לוֹ שִׁיר אֲגַלֶּה בוֹ מוּמוֹ / וּמֵאָז לֹא אֶזְכְּרֶנּוּ וְלֹא אֲדַבֵּר עוֹד בִּשְׁמוֹ / וְיָדַעְתִּי כִּי כֹל תוּכַל / כִּי שׁוּלֵי שִׁירוֹתֶיךָ מְלֵאִים אֶת הַהֵיכָל / וְכָל הַנֶּחֱרִים בְּךָ תַּחַת רַגְלֶיךָ שַׁתָּה / וְכָל לָשׁוֹן תָּקוּם אִתְּךָ לַמִּשְׁפָּט חַתָּה / וְעַתָּה אַל תְּנַקֵּהוּ כִּי אִישׁ חָכָם אָתָּה /



וָאָקוּם וָאֲחַבֵּר-לוֹ זֶה הַשִּׁיר / וַיֹּאמַר בָּרוּךְ אֲדֹנָי וַאעְשִׁיר /

  

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

  

מִשְׂרָה עֲלֵי שִׁכְמָךְ נְגִידִי הָיְתָה



מֵאָז וּמַה יּוֹעִיל לְךָ בָרְחֶךָ



בָּרוּךְ אֱלֹהִים חַי בְּחֶמְלָתוֹ לְךָ



נָתַן שְׁכֶם אֶחָד עֲלֵי אַחֶיךָ.

  

וַיִּגַּשׁ הָרְבִיעִי וַיֹּאמַר אֲדֹנִי יֵשׁ-לִי אִשָּׁה יָפָה / כְּמוֹ שַׁחַר נִשְׁקָפָה / פּוֹרִיָּה וַעֲנֵפָה / פִּרְיָהּ לְמַאֲכָל וְעָלֶיהָ לִתְרוּפָה / אֲהַבְתִיהָ וָאֹהֲבֶנָּה / וּמִיִּרְאָתִי פֶּן תִּבְגֹּד בִּי לַיְלָה וָיוֹם אֶצְּרֶנָּה / וְקָרוֹב אֶל בֵּיתִי אִישׁ וּשְׁמוֹ יִצְחָק הַבַּבְלִי / וְהוּא יֹשֵׁב מִמּוּלִי / וְחָלַק מִשֶּׁמֶן חִכּוֹ / וּמִדֵּי יוֹם וְיוֹם יָצִיץ מֵחֲרַכּוֹ / וְיַבִּיט כֹּה וָכֹה / יֶאֱרֹב בַּמִּסְתָּר כְּאַרְיֵה בְסֻכֹּה / וּמוּל אִשְׁתִּי עַפְעָפָּיו יֵיְשִׁיר / פַּעַם יְנַגֵּן וּפַעַם יָשִׁיר / וְהַיּוֹם אֲשֶׁר פַּעֲמַיִם וְשָׁלֹשׁ לֹא יִרְאֶנָּה / יַחְשֹׁב כִּי מִמָּרוֹם שָׁלַח אֵשׁ בְּעַצְמוֹתָיו וַיִּרְדֶּנָּה /


וְחָפַצְתִּי לְחַבֵּר שִׁיר בּוֹ אוֹדִיעֶנּוּ / כִּי כָבֵד עָלַי הַמַּעֲשֶׂה אֲשֶׁר יַעֲשֶׂנּוּ / וְאִם עָלָה עֶצֶם בְּגוֹרָלִי יְגָרְרֶנּוּ / וְיַנִּיחַ לִי הַצְּבִיָּה אֲשֶׁר עָלְתָה בְּגוֹרָלִי / כִּי הִיא לְבַדָּהּ חֶלְקִי מִכָּל עֲמָלִי / וְלָכֵן אֲדֹנִי אִם תַּחְפֹּץ בְּאַהֲבָתִי / גְּאַל אַתָּה אֶת גְּאֻלָּתִי / וְתָשׁוּב תְּנַחֲמֵנִי וְתֶרֶב גְּדֻלָּתִי /


וָאָקוּם וָאֲחַבֵּר לוֹ זֶה הַשִּׁיר בִּלְשׁוֹן אֲרָמִי / וַיִּקָּחֵהוּ וַיֵּלֵךְ מֵעִמִּי /

  

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

  

מָה לֵיהּ לְיִצְחָק רְחַמְתִּי לְמֶחְזֵי



הָא לָא חֲמִיתֵי לְעֶלֶם דְּבִיתְהוּ



יֶחְדֵּי בְחוּלָק דְּסָלִיק לְעַדְוֵיהּ



אַנָּן בְּדִידָן וְאִינְהוּ בְדִידִהוּ .

  

וַיִּגַּש הַחֲמִישִׁי וַיֹּאמַר אֲדֹנִי הָיוּ לִי שְׁנֵי חֲשׁוּקִים / בְּמַעֲבֶה הַחֵן יְצוּקִים / וְכִרְאִי מוּצָק חֲזָקִים / וּמִנֹּפֶת צוּף מְתוּקִים / מַה נָּעֲמוּ וּמַה נָּאווּ / בְּלֶכְתָּם קוֹמְמִיוּת וְלֹא כָרְעוּ לִי וְלֹא הִשְׁתַּחֲווּ / שָׁגִיתִי בְאַהֲבָתָם וּבְכָל עֵת אוֹתִי רִוּוּ / וְכַאֲשֶׁר הוֹרִידוּ לָאָרֶץ רֹאשָׁם וְלִי הִשְׁתַּחֲווּ / מָאַסְתִּי חֶבְרָתָם / וּמְאַסְתִּים וְגַם גְּעַלְתִּים לְכַלּוֹתָם / לְהָפֵר בְּרִיתִי אִתָּם / וְרָצִיתִי שֶׁתְּחַבֵּר עַל זֶה שִׁיר נִמְרָץ / אֲשֶׁר כָּל אֹזֶן לְשָׁמְעוֹ תֶּחֱרַץ /



וָאַעַן וָאֹמַר חַיֶיךָ הַחִידָה אֲשֶׁר תֹּאמְרֶנָּה / הִגִּיעָה לְיַד אִישׁ יִפְתְּרֶנָּה /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

שְׁדֵי עָפְרָה בְּלֶכְתָּם קוֹמְמִיוּת



פְּדוּת לִבִּי וְנַפְשִׁי בַּחֲלָבָם



וּמִן הַיּוֹם אֲשֶׁר הִשְׁתַּחֲווּ לִי



מְאַסְתִּימוֹ וְנַפְשִׁי בָחֲלָה בָם .

  

וַיִּגַּשׁ הַשִּׁשִׁי וַיֹּאמַר אֲדֹנִי יֵשׁ לִי רֵעַ מַשְׂכִּיל וְזֶרַע קְדוֹשִׁים / וְהוּא הֶחָרוּץ וְהַפִּקֵּחַ שֶׁבָּאֲנָשִׁים / וּשְׁנוֹתָיו הִגִּיעּו לַחֲמִשִּׁים / הֶעֱבִיד שִׂכְלוֹ לְתַאֲוָתוֹ / וְיִרְעֲשׁוּ הָרִים מִגַּעֲרָתוֹ / וּמִנְּעוּרָיו רָדַף אַחֲרֵי הַהֵבֶל / וַיֹּאַמר טוֹב לָלֶכֶת אֶל בֵּית מִשְׁתֶּה מִלֶּכֶת אֶל בֵּית אֵבֶל / וְהָיָה כִנּוֹר וָנֶבֶל / תֹּף וְחָלִיל וְיַיִּן מִשְׁתָּיו / כִּלָּה בַתֹּהוּ וּבַבֶּהָלָה שְׁנוֹתָיו / וְהוּא נַחֵשׁ יְנַחֵשׁ / אָלֹה וְכַחֵשׁ / וְרָצֹחַ וְגָנֹב וְנָאֹף / שַׁמּוֹת וְשָאֹף / וְתָמַהְתִּי וְאֶתְמַהּ אֵיךְ מִן הַמָּוֶת לֹא יֶחֱרַד / בִּרְאוֹתוֹ כִּי יוֹמוֹ מְאֹד רַד / וּבְקָרוֹב יִהְיֶה מִן הָעוֹלָם הַזֶּה נִפְרָד / וְחָפַצְתִּי לִשְׁלֹחַ לוֹ שִׁיר וּלְהַגִּיד לוֹ / עַד מָתַי יְנַבֵּל פְּאֵר שִׂכְלוֹ / וְשָׁכַח כִּי בְּפֶתַע יִנָּתֵק מִבְטַחוֹ מֵאָהֳלוֹ / וְתִשְׁכֹּן בְּאָהֳלוֹ מִבְּלִי לוֹ / וּמְאוּמָה לֹא יִשָּׂא בַעֲמָלוֹ / וְעַתָּה אֲדֹנִי יָרֹם הוֹדֶךָ / הוֹאֵל וּפְנֵה אֵלַי בְּגֹדֶל חַסְדֶּךָ/ וְחַבֵּר עַל הָעִנְיָן הַמְּתֹאָר


/ שִׁיר מְפֹאָר / כִּי יָדַעְתִּי אֶת אֲשֶׁר תְּבָרֵךְ מְבֹרָךְ וַאֲשֶׁר תָּאֹר יוּאָר /


וָאָקוּם וַאֲחַבֵּר אֵלָיו זֹאת הַשִּׁירָה / וַיַעַנְדֶנָּה לְרֹאשֹוֹ עֲטָרָה / וַיִּשְׁלָחֶהָ לְרֵעֵהוּ אֲשֶׁר לְתִתּוֹ לוֹ בְשׂוֹרָה /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

לְעַד מַה תְּנַבֵּל פְּאֵר שִׂכְלְךָ דוֹד



וְאֵיךְ לִבְּךָ לֹא יְרוֹפֵף וְיִרְעוֹד



אֲשֶׁר תֶּחֱרַץ תֶּהֱמֶה עַל דְּבַר מוּם



וְאָלֹה וְכַחֵשׁ וְנָאֹף וְסָעוֹד



וְאַתְּ בֶּן חֲמִשִּׁים וְזֶרַע קְדוֹשִׁים



וְהֵחֵל אֲשׁוּרְךָ לְהִמּוֹט וְלִמְעוֹד



נְטֵה מֵעֲשׂוֹת רָע צְבִי חֵן וְשׁוּב מִ-



צְבָא הָעֲבוֹדָה וְלֹא תַעֲבֹד עוֹד .

  

וַיִּגַּשׁ הַשְּׁבִיעִי וַיֹּאמַר אֲדֹנִי יֵש-לִי שָׁכֵן שׁוֹתֶה שִׁכּוֹר / יִתֵּן הַיֶּלֶד בַּזּוֹנָה וְהַיַּלְדָּה בַיַּיִן יִמְכֹּר / חָפֵץ שֶׁיִּטְּפוּ הֶהָרִים עָסִיס / וּמִשִּׂנְאָתוֹ הַמַּיִם יִשְׂנָא הַטַּל וְהָרְסִיס / וְאֵין לוֹ בְדָבָר מִדִּבְרֵי הַחָכְמָה טַכְסִיס / וּנְשֵׁי בֵיתוֹ הֵם לְרוֹאֵיהֶם צָעַר / תִּדְמֶינָה לְחַיּוֹת הַיָּעַר / רָעֲמוּ פָנִים סָעֲרוּ סָעַר / בְּבֵיתֹו הִרְגִיעָה לִילִית וְקִנְּנָה קִפּוֹד / וְיֵש עֵת לִבְכּוֹת וְעֵת לִסְפּוֹד / שָּם נִקְבְּצוּ דַיּוֹת / וְלוּלֵי שֶׁחָסֵר הָאוֹפַנִּים הָיוּ לוֹ אַרְבַּע חַיּוֹת / וְעַתָּה חַבֵּר אֵלַי שִׁיר אֶשְׁלָחֶנוּ אֵלָיו / לָתֶת לוֹ כִדְרָכָיו וּכְמַעֲלָלָיו /



וָאָקוּם וָאֲחַבֵּר לוֹ שִׁיר זֶה מוּפָז עֶרְכּוֹ / וַיִּפֶן וַיֵּלֶךְ לְדַרְכּוֹ /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

בֵּית הַצְּבִי שָׁם נִקְבְּצוּ דַיּוֹת



לֹיוֹת וְשָׁמָּה קִנְּנָה קִפּוֹז



לוֹ יִטְּפוּ הָרִים עֲסִיס יִשְׁתֵּם



הֵן יַעֲשֹׁק נָהָר וְלֹא יַחְפּוֹז .

  

וַיִּגַּשׁ הַשְּמִינִי וַיֹּאמַר אֲדֹנִי יֵש לִי רֵעַ וְעָמִית / אַחַת דָּתוֹ לְהַמִית / וְאֶתְמַהּ אֵיךְ הָאָרֶץ אוֹתוֹ תִסְבֹּל / וְאֵיךְ כִּנְבֹּל עָלֶה מִגֶּפֶן לֹא יִבֹּל / כִּי בְאֶחָיו חִנָּם יַחְבֹּל / וְיַעֲבֹר תּוֹרוֹת וְיַחֲלֹף חוֹק / וְהָיוּ בְעֵינָיו לִשְׂחוֹק / וְשָׁכַח כִּי מֵרְשָׁעִים אֲדֹנָי רָחוֹק / וְאִם עַד כֹּה הֶעְלִים הָאֵל מִמֶּנּוּ אֶת עֵינָיו / כְּאִלּוּ לֹא רָאָה עֶרְוַת זְנוּנָיו / אֵין שִׁכְחָה לִפְנֵי כִסֵּא כְבוֹדוֹ / וּפִתְאֹם יָקוּם אֵידוֹ / וְיִרְאוּ עֵינָיו כִּידוֹ / וְאֵיךְ שָׁכַח בֶּן הַזּוֹנָה / כִּי מָרָה תִהְיֶה בָאַחֲרוֹנָה / כִּי לַאדֹנָי הַשָּׁמַיִם וּשְׁמֵי הַשָּמַיִם / וְעָבַר עַל פִּשְׁעוֹ פַּעַם וּפַעֲמַיִם / וְאִם לֹא יָשׁוּב חַרְבּוֹ יִלְטֹשׁ וְעַל הָאָרֶץ יִשְׁפְּכֶנוּ כַמָּיִם / וְחָפַצְתִּי אֲשֶׁר תְּחַבֵּר בּוֹ שִׁיר אֲעוֹרְרֵהוּ / אוּלַי מִדַּרְכּוֹ הָרָעָה אֲסִירֵהוּ /



וָאֹמַר אֵלָיו הִנֵּה הַשִּׁיר אֲשֶׁר תְּבַקֵּשׁ / לְהַסִיר מֵעֲמִיתְךָ לֵב עִקֵּשׁ /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

אִם בַּעֲבֹר תּוֹרוֹת וְחֹק לַחְלֹף



צוּר הֶעֱלִים עַיִן וְחֵטְא נָשָׂא



שׁוּב מִזְּדוֹן לִבָּךְ אֲרַי לָאו כָּל



שַׁעְתָּא וְשַׁעְתָּא מִתְרְחִישׁ נִסָּא.

  

וַיִּגַּשׁ הַתְּשִׁיעִי וַיֹּאמַר אֲדֹנִי חָפַצְתִּי שֶׁתְּחַבֵּר אֵלַי תְּפִלָּה קְצָרָה / אֶתְפַּלֵּל אוֹתָהּ בְּעֵת צָרָה / אֶתְקָרֵב בָּהּ אֶל הָאֱלֹהִים / הַמֵּרִים שְׁפָלִים וּמַשְׁפִּיל גְּבֹהִים / תִּהְיֶה מִמִּבְחַר הַמְּלִיצוֹת חֲצוּבָה / וּמִזִּמְרַת הָעִנְיָנִים קְצוּבָה / וְאִם תִּשָּׂא מַשְׂאַת הַתְּפִלָּה הַזֹּאת אֵלַי מֵאֵת פָּנֶיךָ / יָדַעְתִּי כִּי מָצָאתִי חֵן בְּעֵינֶיךָ /



וָאַעַן וָאֹמַר הִנֵּה הַשִּׁיר כְּחֶפְצֶךָ / וַאדֹנָי אֱלֹהֶיךָ יִרְצֶךָ /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

מִמִּי אֱלֹהַי אֶשְׁאֲלָה שָׁאוֹל



חֶפְצִי וְאֵין זוּלָתְךָ לִגְאוֹל



לוּ חַסְדְּךָ לֹא תַחֲיֶה רוּחִי



בַּדֵּי עֲצָמַי יָרְדוּ לִשְׁאוֹל



לוּלֵי אֲסַפְתַּנִי בְרֹב חַסְדָּךְ



מַתִּי בְּבֵין צָרְעָה וְאֶשְׁתָּאוֹל



צִלַּךְ לְבַד אֶשְׁאַף וְלַיָּמִין



אָסוּר בְּסוּר הַבּוֹעֲרִים לִשְׂמֹאל



אֶרְחַץ בְּבוֹר כָּפַּי בְּיוֹם עַרְלֵי



לֵב בַּזְּמָן יִתְגָּאֲלוּ גָאוֹל



אִם כּוֹפְרִים רָשׁוּ וְרָעֵבוּ



נָא עַבְדְּךָ אַל תַּעֲזֹב לִשְׁאוֹל .

  

וַיִּגַּשׁ הָעֲשִׂירִי וַיֹּאמַר אֲדֹנִי יָדַעְתִּי גַּם יָדַעְתִּי כִּי הַנֶּפֶשׁ יִפְעָתָהּ בָּרָה וּטְהוֹרָה / עֲטֶרֶת צְבִי צְפִירַת תִּפְאָרָה / עֲרוּכָה בַכֹּל וּשְׁמוּרָה / מִצּוּר הַשֵּׂכֶל חֻצָּבָה / וּמִלַּהֲבוֹת חַשְׁמַלָּיו נִקְצָבָה / וּבִמְרוֹם הַמַּעֲלוֹת נִצָּבָה / אָצַל אוֹתָהּ הָאֵל מֵהוֹדוֹ הָעֶלְיוֹן / וְנֹגַהּ כָּאוֹר תִּהְיֶה וְקַרְנַיִם מִיָּדוֹ לָהּ וְשָׁם חֶבְיוֹן / וּבְחֶמְלָתוֹ עָלֶיהָ עָשָׂה לָהּ בְּעוֹלָם הַשֵּׂכֶל אַפִּרְיוֹן / לְהַשִּׂיג הַמֻּרְגָּשׁוֹת בְּאֶמְצָעוּת הַדִּמְיוֹן / וְכָתַב לָהּ גִּלָּיוֹן / וְכַאֲשֶׁר יִשְׁלָחֶנָּה בְזֶה הָעוֹלָם הָאֶבְיוֹן / הַשָּׁפָל וְאָפֵל וְאֵין נֹגַהּ לוֹ רַק קֶצֶף וּכְדֵי בִּזָּיוֹן / תִּשְׂמַח בּוֹ כְּשִׂמְחַת יוֹנָה עַל הַקִּיקָיוֹן / וְתִשְׁכַּח הָעֲנִיָה וְהַדַּלָּה / הַמָּקוֹם אֲשֶׁר הָיָה שָׁם אָהֳלָהּ בַּתְּחִלָּה / וְתַחְשֹׁב כִּי בָאָה אֶל הַמְּנוּחָה וְאֶל הַנַּחֲלָה / וְכַאֲשֶׁר יְעוֹרֵר הַשֵּׂכֶל הַנְּפָשׁוֹת הָאֲנוּשׁוֹת לֵאמֹר / הֲלֹא תִּזְכְּרוּ הַר הַמּוֹר / בֵּית אֱלֹהִים אֲשֶׁר שָׁם הָיִינוּ / תִּתְנַכֵּרְנָה וְתֹאמַרְנָה הַעוֹד לָנוּ חֵלֶק וְנַחֲלָה בְּבֵית אָבִינוּ / וְלָכֵן בָּאתִי אֵלֶיךָ הַשָּׂר הַנַּעֲלֶה / לְבַקֵּשׁ אַחַר בַּקָּשַׁת הַמְּחִילָה / לְחַבֵּר שִׁירָה יִהְיֶה זֶה עִנְיָנָהּ / אַשְׁפִּיל בָּהּ הַנֶּפֶשׁ הַנַּעֲנָה / וְאַזְכִּירֶנָּה אֶת עֲוֹנָהּ / וְיָרָד הֲדָרָהּ וַהֲמוֹנָהּ וּשְׁאוֹנָהּ / וְאִם תְּחַבְּרֶנָּה כְפִי שִׂכְלְךָ / הַנֶּפֶשׁ וְהָרְכוּשׁ קַח לְךָ /



וָאַעַן וָאֹמַר לֹא חָפַצְתִּי מִמְּךָ כֶסֶף וּרְכוּשׁ / וָאָקוּם וָאֲחַבֵּר לוֹ שִׁיר לֹא יַעַרְכֶנּוּ יְגִיעַ מִצְרַיִם וּסְחַר כּוּשׁ /



וַיִּשְׂמַח עָלָיו הָאִישׁ וְאֵלָיו עָרָב / כְּמוֹצֵא שָׁלָל רָב /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

נַפְשׁוֹת בְּנֵי אָדָם אֲשֶׁר לֹא בָנוּ



סוֹד בֵּית אֱלֹהִים כַּאֲשֶׁר הָיִינוּ



מִתְנַכְּרוֹת אוֹמְרוֹת הֲעוֹד יֵשׁ לָנּוּ



חֵלֶק וְאִם נַחְלָה בְּבֵית אָבִינוּ .

  

וַיִּגַּשׁ הָאַחַד עָשָׂר וַיֹּאמַר יֵשׁ אַיֶּלֶת אֲהָבִים וְיַעֲלַת חֵן בִּשְׁכוּנָתִי / הָיִיתִי נוֹתֵן בְּאַהֲבָתָהּ שְׂדֵה מִקְנָתִי / כִּי לָהּ שְׁתֵּי עֵינַיִם / קוֹרְעוֹת הַלְּבָבוֹת לִשְׁנַיִם / וְחָלִיתִי בְאַהֲבָתָהּ וְהֶחֳדַלְתִּי אֶת דִּשְׁנִי / כִּי בִרְאוֹתִי בְשִׂפְתוֹתֶיהָ אֶת תִּקְוַת חוּט הַשָּׁנִי / וְאֶרְאֶה בְרַד הַשִּׁנַּיִם אָמוּת לַשָׁחַת / כִּי אֶרְאֶה בָּרָד וְאֵשׁ מִתְלַקַּחַת / וְלָכֵן אַחֲרֵי אֲשֶׁר אֲנִי בְיָם חִשְׁקָהּ נִטְבָּע / הִנְנִי בִגְאוֹנִי נִשְׁבָּע / כִּי אֶת קוֹבְעַי נֶפֶשׁ אֶקְבָּע / וְלוּ תִהְיֶה אֵשֶׁת אִישׁ הַבֵּינַיִם / אַצְמִיחַ לוֹ קְרָנַיִם / וְלָכֵן חוּשָׁה הַשָּׂר וְהָיִיתָ לִי לְעֵינָיִם / וְחַבֵּר שִׁיר נֶחְמָד לְרוֹאָיו / תִּכְסֹפְנָה הַלְּבָבוֹת לִשְׁמֹעַ פְּלָאָיו /



כְּשָׁמְעִי נָתַן הַחֵשֶׁק בְּפִי חָזוֹן נִפְרָץ / וָאֲחַבֵּר שִׁיר נִמְרָץ / וְאָז הָאִישׁ לְקַחְתּוֹ חָרָץ / וַיֹּאמֶר אֵלַי לוּ תַחְפֹּץ כִּכַּר כֶּסֶף אֶשְׁקֹל / וָאַעַן וָאֹמַר חַנַּנִי אֱלֹהִים וְיֵש-לִי כֹל /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

אֶל יַעֲלַת הַחֵן שְׁתֵּי עֵינַיִם



הַקּוֹרְעִים לֵב חֹשְׁקִים לִשְׁנַיִם



בָּרָד וְאֵש הִתְלַקְּחוּ בָהּ הֵם שְׂפַת



שָׁנִי אֲשֶׁר לָהּ עַל בְּרַד שִׁנַּיִם



אַחַר אֲשֶׁר נַפְשִׁי לְחִשְׁקָהּ תֶּהֱמֶה



צָרָה כְמַבְכִּירָה עֲלֵי אָבְנַיִם



אַעַשׂ וְלוּ אֵשֶׁת רְפָאִים תִּהְיֶה



עַל הָרְפָאִים עַשְׁתְּרוֹת קַרְנַיִם .

  

וַיִּגַּשׁ הַשְּנֵים עָשָׂר וַיֹּאמַר בִּי אֲדֹנִי צַר לִי מְאֹד וּלְנַפְשִׁי רִיבוֹתַי / וַאֲשֶׁר קָרָה לִי מִן הָרַע אָנֹכִי סַבּוֹתִי / בְּיָדְעִי מַעֲלַת הַנֶּפֶשׁ וְלֹא בִינוֹתִי / וִהְיוֹת הַגּוּף טִיט וָרֶפֶשׁ וּמִסֵּבֶל שִׁכְמוֹ הֲסִירוֹתִי / מָכַרְתִּי הַקַּיָּם בַּכָּלֶה וְהַטּוֹב בָּרַע הֲמִירוֹתִי / וּמֵרֶחֶם כָּרְשָעִים זֹרְתִּי / וְאָחוֹר נָזֹרְתִּי / קְרָאוּנִי וְלֹא נֵעֹרְתִּי / וְלֹא חָרַדְתִּי מֵחֲרוֹן אַף אֲדֹנָי וְלֹא יָגֹרְתִּי / מִיּוֹם הֱיוֹתִי עַל בִּרְכֵּי הַזְּמָן יָשַׁנְתִּי / וּבְשַׁמְנוֹ רֹאשִׁי דִשַׁנְתִּי / הָיָה בִי שָׁפָל וְלֹא הִבְחַנְתִּי / וְתוֹרַת אֱמֶת הָיְתָה בְּפִי וּבִלְבָבִי וְלֹא הֶאֱמַנְתִּי / קְרָאוּנִי וְלֹא הֶאֱזַנְתִּי / וְלָכֵן בּדוּמִיָּה בָדָד יָשַׁבְתִּי / כִּי אֲנִי זָקַנְתִּי וָשַׂבְתִּי / וְיָמַי כָּלוּ בְעָשָׁן / אִירָא מֵחֲרוֹן אַף אֲדֹנָי פֶּן יֶעֱשָׁן / וְלוּ הֶאֱזַנְתִּי דִבְרֵי חֲכָמִים / חֲבָלִים נָפְלוּ לִי בַנְעִימִים / וְלָכֵן מִלְחָמָה תִהְיֶה בֵינִי וּבֵין נַפְשִׁי כָּל הַיָּמִים / וְעַתָּה הַשָּׂר חַבֵּר-לִי שִׁיר וְכוֹנְנֵהוּ / וּמַעֲשֵׂה יָדַי כּוֹנְנָה עָלָיו וּמַעֲשֵׂה יָדַי כּוֹנְנֵהוּ /



וָאָקוּם וָאֲחַבֵּר זֶה הַשִּׁיר וָאֹמַר אֵלָיו קָחֵהוּ / אַשְׁרֶיךָ אִם לֹא תִּשְׁכָּחֵהוּ /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

צַר לִי בְּיָשְעִי מַעֲלַת הַנֶּפֶשׁ



עַל מַעֲלַת הַגּוּף וְלֹא יָגֹרְתִּי



כִּי מִנְּעוּרַי כָּרְשָׁעִים זֹרְתִּי



וָאֶטְבְּעָה כַיּוֹם בְּטִיט וָרֶפֶשׁ

  

אֵיךְ אֶעֱדֶה רָמָה וּבִגְדֵי חֹפֶשׁ



כַמָּה קְרָאוּנִי וְלֹא נֵעֹרְתִּי



אָז עִם אֱלֹהִים בֶּאֱמֶת יָשֹׁרְתִּי



וָאֶהֱיָה קוֹנֶה לְנַפְשִׁי נֹפֶשׁ

  

מָה אֹמְרָה עוֹד וַאֲנִי זָקַנְתִּי



כָּלוּ יְמֵי חַיַי בְּעָשָׁן כָּלוּ



עַל כֵּן לְבוּשִׁי שַׂק וּמַעְטֵה אֵבֶל

  

רַבַּת קְרָאוּנִי וְלֹא הֶאְזַנְתִּי



לוּ חָכְמוּ רֵעַי אֲשֶׁר סָכָלוּ



אָז בַּנְּעִימִים יִהְיֶה לִי חֶבֶל .

  

וַיִּגַּשׁ הַשְּׁלֹשָׂה עָשָׂר וַיֹּאמַר אֲדֹנִי יֵשׁ לִי אָחוֹת יָפָה וּשְׁמָהּ תָּמָר / וְעֵינֶיהָ מִבְּכִי חֲמַרְמָר / וּבְשָׂרָהּ סָמָר / לֹא עָמַד טַעְמָהּ בָּהּ וְרֵיחָהּ נָמָר / כִּי שַׁדַּי לָהּ הֵמַר / בְּסִבַּת הַחֹשְׁקִים אֲשֶׁר בָּהּ יִלָּחֵמוּ / אַכְזָרִים הֵמָּה וְלֹא יְרַחֵמוּ / לֹא יִתְּנוּ שֵׁנָה לְעֵינֵיהָ וּלְעַפְעַפֶּיהָ תְּנוּמָה / הִיא לְשָׁלוֹם וְהֵמָּה לַמִּלְחָמָה / זֶה כַמָּה הוֹדִיעָה אוֹתָם / כִּי קָצָה בְאַהֲבָתָם / וְלוּ הָיוּ עַל יַד יְמִינָהּ חוֹתָם / מִשָּׁם תְּנַתְּקֵם / וּכְטִיט חוּצוֹת תְּדִקֵּם / וְלֹא תִמְצָא עִמָּהֶם מַרְגֹּעַ / וְלוּ יִתַּמּוּ כָּל הַחֹשְׁקִים לִגְוֹעַ / לֹא יוּכְלוּ עֲלוֹת בְּהַר הַיְּעֵלָה וּבְקָצֵהוּ לִנְגֹעַ / וְלָכֵן הַשָּׂר חַבֵּר-לָהּ שִׁיר חָדָשׁ וּבוֹ תַחֲרוּת / כִּי הִיא פְסִילֵי הַחֵשֶׁק תְּגַדֵּעַ וַאֲשֵׁרָיו תִּכְרֹת / וְהִיא תִשְׁלָחֶנּוּ אֶל הַחֹשְׁקִים / אֲשֶׁר דְּלָתֶיהָ דוֹפְקִים / אוֹלַי מִתַּרְמִית לִבָּם יֶחְדָּלוּ / וְלֹא יֵלְכוּ עוֹד אֶחֲרֵי הַהֵבֶל וְיֶהְבָּלוּ /



וָאָקוּם וָאֲחַבֵּר לוֹ זֶה הַשִּׁיר הֶעָרֵב / וַיִּיטַב בְּעֵינָיו כְּאִלּוּ נִתַּן בְּחוֹרֵב /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

הַחֹשְׁקִים מַה תִּיגְעוּ יָגֹעַ



נַפְשִׁי בְחֵשֶׁק מֵאֲנָה לִנְגֹּעַ



אֵיךְ תִּרְדְּפוּ יַעְלָה אֲשֶׁר לֹא מָצְאָה



נַפְשָׁהּ בְּחֶבְרַת חֹשְׁקִים מַרְגֹּעַ



אֵיךְ אָזְרָה חַיִל בְּעֹז הִתְאַכְזְרָה



לִכְרֹת אֲשֵׁרֵיהֶם וּבָם לִפְגֹעַ



לֹא שׁוֹקְטָה אַף שׁוֹפְטָה הָעֹגְבִים



עַד יִתְּמוּ הַחֹשְׁקִים לִגְוֹעַ .

  

וַיִּגַּשׁ הָאַרְבָּעָה עָשָׂר וַיֹּאמַר אֲדֹנִי יֵשׁ בְּעִירִי אֵשֶׁת חֵן חֲשַׁקְתִּיהָ / וּמִן הַיֹּום אֲשֶׁר רְאִיתִיהָ / אִוִּיתִיהָ חֲמַדְתִּיהָ / וְאַהֲבַת עוֹלָם אֲהַבְתִּיהָ / וְעַל לוּחַ לִבִּי חֲקַקְתִּיהָ / וּלְסַפֵּר הוֹדָהּ נִלְאָה כָּל חֹשֵׁק וְלָאִיתִי / כִּי הִיא פֶּרַח כָּל הַצְּבִיּוֹת אֲשֶׁר בְּיָמַי רָאִיתִי / חִלִּיתִי פְנֵי בוֹרְאִי לְחַסֵּר מִמִּסְפַּר שְׁנוֹתַי וּלְהוֹסִיף עַל שְׁנוֹתֶיהָ / כִּי הִיא מְלֵאָה הִדּוּרִים / עַל כָּל גְּדוֹתֶיהָ / עַד כִּי כוֹכְבֵי הַשָּׁמַיִם וּכְסִילֵיהֶם / פָּשְׁטוּ אֶת מְעִילֵיהֶם / וְאֶדֶר הַיְּקָר אֲשֶׁר עֲלֵיהֶם / וְנָתְנוּ בְעִזְּבוֹנֶיהָ / וְצֶדֶק וְנֹגַהּ יָרְדוּ מִכִּסְאוֹתָם וְהָיוּ שְׁתֵּי עֵינֶיהָ / וְהָרָקִיעַ אֲשֶׁר עַל רָאשֵׁי הַחַיּוֹת הָיָה בְרַד שִׁנֶּיהָ / וְהָאֵשׁ תִּזְרַח וְֱתֶעֱרַב בְּפָנֶיהָ / וְהַמַּלְאָךְ הַדּוֹבֵר הָיָה לְשׁוֹנָהּ / וְהִיא הַיּוֹשֶׁבֶת בְּמַלְכוּת הַדַּעַת רִאשׁוֹנָה / וְיַד כָּל חַכְמֵי הַזְּמָן בָּאַחֲרוֹנָה / וְהַכְּרוּבִים הָיוּ שְׁנֵי יָדֶיהָ / וְהָרֵי בְשָׂמִים הָיוּ שְׁנֵי שָׁדֶיהָ / אִם אֶשְׁכָּחֵם כָּל יְמֵי הֱיוֹתִי תִּשְׁכַּח יְמִינִי / וְאִם לֹא אַעֲלֵם עַל רֹאש שִׂמְחתִי תִּדְבַּק לְחִכִּי לְשׁוֹנִי / וְזֶה כַמָּה תָמַהְתִּי וָאֶתְמַהּ / אֵיךְ בְּשָׂרִי לֹא תֹאכְלֶנָּה רִמָּה / כִּי הִיא תִמְחַץ בְּחֵץ עֵינֶיהָ / וְלוּלֵא תְרַפֵּא אוֹתִי בְּאוֹר פָּנֶיהָ / וּבְטוֹב רֵיחַ שְׁמָנֶיהָ / לֹא אוּכַל לָקוּם מִפָּנֶיהָ / אָמְנָם הִיא תִמְחַץ בְּחֵץ עֵינֶיהָ / וּתְרַפֵּא בְחַשְׁמַלֵּי זָהֳרָהּ / וְעַל כֵּן יֵלְכוּ הַחֹשְׁקִים לְאוֹרָהּ / וְלוּ תִרְצַח הַיָּפָה אֶת כָּל הַיְּקוּם / לֹא יֹאמְרוּ נְתָנַנּוּ אֲדֹנָי בִּידֵי לֹא אוּכַל קוּם / וְעַתָּה הַשָּׂר אַשְׁבִּיעֲךָ בָאיָּלוֹת וּבַצְּבָאוֹת / לְחַבֵּר שִׁיר כַּאֲשֶׁר יֵאוֹת / וְאָשׁלָחֶנּוּ אֶל הַצְּבִיָּה חֲשׁוּקָתִי / כִּי הִיא לְבַדָּהּ בַּזְּמָן תְּשׁוּקָתִי /



וָאֹמַר אֵלָיו הִנֵּה הַשִּׁיר אֲשֶׁר לְפִי הָעִנְיָן יֵאוֹת / וַיִּקָּחֵהוּ וַיְּבָרְכֵנִי בְשֵׁם אֲדֹנָי צְבָאוֹת /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

לֹא יֹאמְרוּ שָׁוא עָמְלוּ בוֹנַיִךְ



הַחֹשְׁקִים עֵת יֶחֱזוּ פָנַיִךְ



בָּרוּךְ אֲשֶׁר כָּכָה בְעוֹלָמוֹ אֲשֶׁר



אָצַר סְגֻלּוֹתָיו בְּמִכְמַנַּיִךְ



כָּל כּוֹכְבֵי שַׁחַק וְחֵיָלם נָתְנוּ



הַחֵן וְהַיֹּּפִי בְעִזְּבוֹנַיִךְ



נֹגַהּ וְכוֹכָב אֵת הֲדָרֵךְ נָתְנוּ



שָׁנִי בְקָשְׁרָם עַל בְּרַד שִׁנַּיִךְ



אַשְׁרֵי אֱנוֹשׁ יַבִּיט לְהוֹדֵךְ יַעֲלָה



אַשְׁרֵי אֱנוֹשׁ יֹאחַז בְּסַנְסִנַּיִךְ



יוֹאֵל אֱלֹהִים חַי וְיוֹסִיף מַחֲצִית



מִסְפַּר שְׁנוֹת חַיֵי עֲלֵי שָׁנָיִךְ



מִי לִמְּדֵךְ קֶשֶׁת יְפֵהפִיָה אֲשֶׁר



מָחַצְתְּ סְגוֹר לִבִּי בְּחֵץ עֵינַיִךְ



אִם תִּקְטְלִינִי לָךְ אֲיַחֵל יַעֲלָה



אֶצְעַק וְלוּ אַתְּ אֹטְמָה אָזְנַיִךְ



אִם אֶשְׁכְּחָה חִשְׁקֵךְ צְבִיַת חֵן וְאִם



אֶתְאַו רְאוֹת רָזוֹן בְּמִשְׁמַנַּיִךְ



אוֹ אֶזְנְחָה יִפְעַת תְּמוּנָתֵךְ וְאִם



לֹא אֶבְזְרָה עַצְמוֹת בְּנֵי חֹנַיִךְ



תִּשְׁכַּח יְמִינִי לַחֲבֹק עוֹד יַעֲלָה



אוֹ תַּחְשְׁבִינִי מִכְּלַל מוֹנַיִךְ



עָפְרָה וְאִלּוּ תִרְצְחִי כָּל הַיְּקוּם



לֹא יֹאמְרוּ שָׁוְא עָמְלוּ בוֹנָיִךְ .

  

וַיִּגַּשׁ הַחֲמִשָּׁה עָשָׂר וַיֹּאמַר אֲדֹנִי חַנַּנִי אֱלֹהִים בֵּן וְהוּא בְשִׂכְלוֹ אָגוּר / וּלְשׁוֹנוֹ זָהָב סָגוּר / הוֹבִילוּהוּ רַגְלָיו מֵרָחוֹק לָגוּר / בְּשִׁבְתוֹ תַחַת צִלִּי / הָיָה שָׁלֵו בְּבֵיתִי וְרַעֲנָן בְּהֵיכָלִי / כְּאִישׁוֹן עֵינִי נְצַרְתִּיו / וְלַעֲשׂוֹת הַטּוֹב וְהַיָּשָׁר הִזְהַרְתִּיו / וְאַחֲרֵי אֲשֶׁר גָּזַר הָאֵל בַּפְּרִידָה / וְנָהַג בַּשְׁבִי נַפְשִׁי הַשְׁדוּדָה / חָשַׁבְתִּי לִשְׁלֹחַ אֵלָיו שִׁירָה / תִּהְיֶה לְנֶגֶד עֵינָיו לְאַזְכָּרָה / וְהוּא לְהַזְכִּירוֹ שֶׁיִּגְלֶה לִמְקוֹם תּוֹרָה / וְיַעֲשֶׂה מִן הַמִּדּוֹת הַטּוֹבוֹת לְרֹאשׁוֹ עֲטָרָה / וְאַחֲרֵי אֲשֶׁר נָתַן הָאֵל מַיִם עֲמֻקִּים עֵצָה בְלִבּוֹ / יִדְלֶנּוּ בְעוֹדֶנוּ בְאִבּוֹ / וְרוּחַ אֲדֹנָי נוֹסְסָה בּוֹ / וְיָשִׂים כָּאָרֶץ גֵּווֹ וְכַחוּץ לַעוֹבְרִים / וְיִקַּח מוּסַר הַשְׂכֵּל צֶדֶק וּמִשְׁפָּט וּמֵישָׁרִים / וְאֵת אֲשֶׁר לְרֵעֵהוּ לֹא יַחְמֹד / וּבִמְקוֹם הַגְּדוֹלִים לֹא יַעֲמֹד / וְחֶרֶב שִׂכְלוֹ עַל הַזְּמָן יִשְׁלֹף / וְיִפְתַּח פִּיו אֶל דִּין כָּל בְּנֵי חֲלֹף / וְיִזְכֹּר כִּי סוֹף כָּל אָדָם לָרֶדֶת אֶל אַבְנֵי בוֹר / וְהַשְׁאוֹל אוֹתָנוּ חֳמָרִים חֳמָרִים יִצְבֹּר / אַל יָנוּס הַקַּל וְאַל יִמָּלֵט הַגִּבּוֹר / וְעַתָּה אֲדֹנִי נֵר הַמַּעֲרָבִי / עָלֶיךָ הִשְׁלַכְתִּי אֶת יְהָבִי / לְחַבֵּר אֵלָיו שִׁיר יִגְזֹר הַלֵּב לִגְזָרִים / כִּי שָׁמַעְתִּי כִּי אָחַז דֶּרֶךְ הַמְּשׁוֹרְרִים /



וָאַעַן וָאֹמַר חַיֶּיךָ אִם הוּא מְשׁוֹרֵר / עָלֵינּו בְשִׁירָיו לֹא יִשְׂתָּרֵר / כִּי בִרְאוֹתוֹ שִׁירִי אִם הוּא נִרְדָּם יִתְעוֹרֵר / וְטֶרֶם יִרְקַּח כָּמוֹהוּ שֶׁבַע פְּעָמִים יְזוֹרֵר /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

צְבִי מָאַס עֲוֹן וְאוֹן



וְאֶל מְקוֹם אֱמוּן גְּלֵה



רְאֵה בְלִבְּךָ בְאֵר



וְלָךְ דְּלִי וְקוּם דְּלֵה



שְׂפַת אֱמֶת וְדִין בְּנֵי



חֲלוֹף אֱהַב וְשָׁוְא אֲלֵה



זְכֹר זְמָן יְצַו לְךָ



רְדָה בְּבוֹר בְּלִי בְלֵה



אֱמֹר לְנַפְשְׁךָ שְׁחִי



וְטוֹב אֲמָר-לְךָ עֲלֵה.

  

וַיִּגַּשׁ הַשִּׁשָׁה עָשָׂר וַיֹּאמַר אֲדֹנִי יֵשׁ בִּשְׁכוּנָתִי אָדָם רָשָׁע / אָזְנָיו הַכְבֵּד וְעֵינָיו הָשַׁע / וְעַל פַּת לֶחֶם יִפְשַׁע / וְאִם יִרְאֶה עָם בַּאדֹנָי נוֹשָׁע / פְּנֵיהֶם יִטְחָן / וְֹיֹאמַר הֲיוּכַל אֵל לַעֲרֹךְ שֻׁלְחָן / וְלֹא יֵדַע כִּי אֲדֹנָי צַדִּיק יִבְחָן / וְרָשָׁע וְאֹהֵב חָמָס שָׂנְאָה נַפְשׁוֹ / וְיִשְׁפֹּך עֲלֵיהֶם חֲמַת אַפּוֹ וְאִשּׁוֹ / וְתָמַהְתִּי אֵיךְ יוּחַן רָשָׁע בַּל לָמַד צֶדֶק / וְכָל דְּרָכָיו כִּמְשֻׂכַּת חֶדֶק / וְעַתָּה אֲדֹנִי חַבֵּר-לִי שִׁיר עַל הָעִנְיָן הַזֶּה / וְאֶשְׁלָחֶנּוּ אֶל הָאִישׁ אֲשֶׁר דְּרָכָיו בּוֹזֶה /



וָאַעַן וָאֹמַר הִנֵּה שִׁיר עַל זֶה הָעִנְיָן יְדַבֵּר / וְדַלְתוֹת נְחוּשָׁה יְשַׁבֵּר /

  

אֵלִי אֵלִי אֵיכָה יוּחַן



רָשָׁע וּפְנֵי אֶבְיוֹן יִטְחַן



אַף יִתְמַרְמֵר זֵד וַיֹּאמַר



אִם יוּכַל אֵל לַעֲרֹך שֻׁלְחָן



שׁוּב אֵל עֶלְיוֹן וּבְעִיר צִיּוֹן



תִּיסוֹד אֶבֶן אֶבֶן בֹּחַן



יֵדַע רָשָׁע עֵינָיו הָשַׁע



כִּי אַתָּה אֵל צַדִּיק תִּבְחָן.

  

וַיִּגַּשׁ הַשִּׁבְעָה עָשָׂר וַיֹּאמַר אֲדֹנִי יֵשׁ בְּעִירִי שְֹתֵּי אֲחָיוֹת יְפֵהפִיוֹת / תִּנְהַגְנָה לִבּוֹת רוֹאֵיהֶם כִּשְׁבוּיּוֹת / הָיוּ כִימָה וּכְסִיל לִפְנֵיהֶן עֹטְיוֹת / וּבְנוֹת עַיִשׁ כְּדַיּוֹת אוֹ עָזְנִיּוֹת / וְשִׁבְעָה כוֹכְבֵי לֶכֶת לְמִינֵיהֶם / שֶׁבַע הַנְּעָרוֹת הָרְאוּיוֹת לָתֵת לָהֶם / וּבְנוֹת הַחֵן בְּעֶרְכָּם כּוּשִׁיּוֹת / צַוָּארָן כְּמִגְדַּל דָּוִד בָּנוּי לְתַלְפִּיּוֹת / וּלְשׁוֹנָן חֶרֶב פִּיפִיּוֹת / עָבְרוּ מַשְׂכִּיּוֹת / וּבִשְׂפַת שָׁנִי וּבְרַד הַשִּנַּיִם / קוֹרְעוֹת הַלְּבָבוֹת לִשְׁנַיִם / הוֹלְכוֹת נְטוּיּוֹת גָּרוֹן וּמְשַׂקְּרוֹת עֵינַיִם / וְכָל אִישׁ אֲשֶׁר הוֹדָם יֶחֱזֵה / יֹאמַר מַחֲנֶה אֱלֹהִים זֶה / וְיִקְרָא שֵׁם הַמָּקוֹם הַהוּא מַחֲנַיִם / וְאַחַר אֲשֶׁר נַפְשֵׁנוּ צָרָה כְמַבְכִּירָה עַל הָאָבְנָיִם / אֵין לִתְמֹהַּ אִם לְבַעֲלֵיהֶן עַשְׁתְּרוֹת קַרְנַיִם / עַתָּה אוּרְיָן בַּר אוּרְיָן / עֲדֵה גָאוֹן וְגֹבַהּ וּלְבַש צְדָקָה כַּשִּׁרְיָן / וְחַבֵּר עֲלֵיהֶן שִׁיר כְּפִי יָפְיָן /



וָאַעַן וָאֹמַר חַיֶּיךָ הַשִׁיר יִהְיֶה כָלִיל בַּהֲדָרוֹ / וְיֵלְכוּ הַחֹשְׁקִים לְאוֹרוֹ /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:



הַהוֹלְכוֹת וּמְשַׂקְרוֹת עֵינַיִם



הַקּוֹרְעֹות לֵב חֹשְׁקִים לִשְׁנַיִם



הַנּוֹתְנוֹת נֶפֶשׁ צְבָאִים לִהְיוֹת



צָרָה כְּמַבְכִּירָה עֲלֵי אָבְנַיִם



לֹא אֵדְעָה בַמֶּה תְצוּדוּן אֵת לְבַב



חֹשֵׁק הֲאִם הוּא עַל בְּרַד שִׁנַּיִם



אוֹ עַל שְׁדֵי חֶמֶד אֲשֶׁר מוֹר נוֹטְפוֹת



כַּיּוֹם וְאִם הוּא בַעֲגִיל אָזְנַיִם



מִי זֶה אֲשֶׁר יִתְמַהּ בְּעֵת זֹאת נֶחֱזֶה



אִם לָרְפָאִים עַשְׁתְּרוֹת קַרְנַיִם



תִּהְיוּ לְרָצוֹן לִי וְכָל מָקוֹם אֲשֶׁר



אֶרְאֶה פְנֵיכֶם אֶקְרְאָה מַחְנָיִם .

  

וַיִּגַּשׁ הַשְּמֹנָה עָשָׂר וַיֹּאמַר אֲדֹנִי רָאִיתִי מְתֵי רִשְׁעָה / שְׁכוּרִים בְּיֵין תּוֹעָה / וְלֵצִים חָמְדוּ לָצוֹן / וְתַחַת הֱיוֹתָם לִפְנֵי אֱלֹהִים לְרָצוֹן / וְהִנֵּה שָׂשׂוֹן וְשִׂמְחָה הָרֹג בָּקָר וְשָׁחֹט צֹאן / וְתָמַהְתִּי וְאֶתְמַהּ אֵיךְ יָמִירוּ טוֹבָה נִצְחִית וְעוֹמֶדֶת / בְּטוֹבָה כָלָה וְאוֹבֶדֶת / וּמְשׂוֹש עוֹלָם בְּאֵשׁ יוֹקֶדֶת / וּמַה טּוֹב לָאִישׁ לִהְיוֹת צִדְקָתוֹ עוֹמֶדֶת לָעַד / לָעֶרֶב יְחַלֵּק שָׁלָל וְלַבֹּקֶר יֹאכַל עַד / מֵאֲשֶׁר רַגְלוֹ תִמּוֹט וְתִמְעַד / וּבְזַעַם אֶל יַרְכְּתֵי בוֹר יִצְעַד / וְעַתָּה מַה בֵּין הֱיוֹת אִישׁ בְּגַלְגַּל עֲרָבוֹת/ אוֹ הֱיוֹתוֹ מְעֻתָּד בְּפִי הֶחֳרָבוֹת / אַחֲלַי אֲדֹנִי לְפָנֶיךָ / לְחַבֵּר עַל זֶה הַשִּׁיר כְּפִי רְאוֹת פָּנֶיךָ /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

עַד אָן מְתֵי רִשְׁעָה בְּיֵין תּוֹעָה



שָׁכְרוּ וְלֵצִים חָמְדוּ לָצוֹן



שָׁכְחוּ מְשׂוֹשׂ עוֹלָם וְגִיל נִצְחִי



לֵאמֹר הֲרֹג בָּקָר וְשָׁחֹט צֹאן



לוּ חָכְמוּ אָז טוֹב וְיָפִיקוּ



מֵאֵת אֱלֹהִים גּוֹאֲלָם רָצוֹן



מַה טּוֹב וְנָעִים לַחֲסוֹת בִּכְנַף



רוֹכֵב עֲרָבוֹת מִהְיוֹת חִיצוֹן.

  

וַיִַּשׁ הַתִּשְׁעָה עָשָׂר וַיֹּאמַר אֲדֹנִי מָצָאתָי כָתוּב לְאֶחָד מִן הַחֲכָמִים / בְּתוֹךְ מוּסָרָיו הַנְּעִימִים / כִּי מִי שֶׁיִּתֵּן מֵהוֹנוֹ לְאַחֵר יִהְיֶה אֲדֹנָיו / וְאִם יִצְטָרֵךְ לוֹ יֵקַל בְּעֵינָיו / וְאִם יְיַאֵשׁ לִבּוֹ מִמֶּנּוּ / כָּמוֹהוּ בְמַדְרֵגָתוֹ יַחְשְׁבֶנּוּ / וְאִם יְכַבְּדֶנּוּ יְכֻבַּד מִמֶּנּוּ / וְחָפַצְתִּי שֶׁהַדְּבָרִים הָאֵלֶּה בְּשִׁיר תְּחַבְּרֵם / וְכָבוֹר וְהָדָר תְּעַטְּרֵם /



וָאַעַן וָאֹמַר הִנֵּה הַשִּׁיר קָחֶנּוּ / וּלְפִי הָעִנְיָן אֵין לְמַעְלָה מִמֶּנּוּ /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

תְּנָה מֵהוֹנְךָ אֶל אִישׁ וְתִהְיֶה



בְּעֵינֵיהוּ נְגִידוֹ אוֹ אֲדֹנָיו



וְיַאֵשׁ לִבְּךָ מֵאִישׁ זְמַנָּךְ



וְכָמוֹהוּ תְּהִי נֶחְשָׁב בְפָנָיו



וְכַבֵּד אִישׁ וּמִמֶּנּוּ תְכֻבַּד



וְהִצְטָרֵךְ לְאִישׁ תֵּקַל בְּעֵינָיו .

  

וַיִּגַּשׁ הָעֶשְׂרִים וַיֹּאמַר אֲדֹנִי יֵשׁ-לִי רֵעַ גֵּאֶה וָרָם / וְחַטֹּאתָיו לֹא יוּכַל אִישׁ לְסָפְרָם / וַאֲנִי שְׂנֵאתִיו בַּעֲבוּרָם / לֹא יֵבוֹשׁ הַסָּכָל וְלֹא יִכָּלֵם / וְלֹא יַעֲלֶה עַל לְבָבוֹ כִּי לֹא יוּכַל לֱהִתְעַלֵּם / וְאֵל גְּמוּלוֹת אֲדֹנָי כְּעַל גְּמֻלוֹת כְּעַל יְשַׁלֵּם / וְאֵיךְ רֶסֶן תַּאֲוָתוֹ לֹא יִהְיֶה כוֹבֵשׁ / וְלֹא יִזְכֹּר הֶבֶל נִדָּף וְקַשׁ יָבֵשׁ / כִּי אֲדֹנָי מֶלֶךְ גֵאוּת לָבַשׁ / חָפַצְתִּי בְשִׁיר נִמְרָץ לְהוֹכִיחוֹ / אוּלַי אָשִׁיב אֶת נִדָּחוֹ / וְאֵלֶיךָ הַשָּׂר יָדַי פְּרוּשׁוּת / קוּם כִּי עָלֶיךָ הַדָּבָר לַעֲשׂוֹת /



וָאָקוּם וַאֲחַבֵּר שִׁיר כְּפִי הַעִנְיָן אֲשֶׁר יִבְחָרֶנּוּ / וָאֹמַר הִנֵּה הַשִּׁיר אֲשֶׁר יַכְלִימֶנּוּ / וְעַל טָעוּתוֹ יַעֲמִידֶנּוּ / לֹא יַעַרְכֶנּוּ כַסְפּוֹ וּזְהָבוֹ / שְׁלַח יָדְךָ וֶאֱחֹז בִּזְנָבוֹ /

  

וָאֶשָׂא מְשָׁלִי וָאֹמַר:

  

הֲיִגְאֶה וְיִשְׁלַם אֱנוֹשׁ דַּך וְנִכְלָם



וְעַל טוֹב וְעַל רָע גְּמוּל לוֹ יְשֻׁלָּם



וְיַרְבֶּה אֲשָׁמָיו וְחַטֹּאת עֲלוּמָיו



כְּחוֹל יָם וְסֵפֶר אֱמֶת לֹא יְכִילָם



וְיִשְׁכַּח מְעוֹדֵד עֲנָוִים וְדֶרֶךְ



רְשָׁעִים יְעַוֵּת וְיִמְלֹךְ לְעוֹלָם .

  

וַיִּשְׂמַח הָאִישׁ וַיַעֲמֹד מִזַּעְפּוֹ / וַיִּשְׁלַח יָדוֹ וַיַחֲזֶק בּוֹ וַיְהִי לְמַטֶּה בְּכַפּוֹ /



וַיְהִי כַּאֲשֶׁר תַּמּוּ כָל הָאֲנָשִׁים לִשְׁאֹל אִישׁ אִישׁ אֶת שְׁאֵלָתוֹ / וּמָצָא כָּל אֶחָד מֵהֶם צֳרִי אֶל מַחֲלָתוֹ / הוֹדוּ וּבֵרְכוּ לְאֵל אֶחָד וְאֵין שֵׁנִי / וַיֹּאמְרוּ אֵלַי הֶחֱיִיתָנוּ נִמְצָא חֵן בְּעֵינֵי אֲדֹנָי / וְהַחֶסֶד אֲשֶׁר עָשִׂיתָ עִמָּנוּ אִישׁ לֹא שִׁעֵר וְלֹא יְדָעוֹ / וְנִהְיֶה אֲנַחְנוּ וְאַדְמוֹתֵינוּ עֲבָדִים לְפַרְעֹה / אֲשֶׁר הַשָּׂר שֶׁבְּחָרְךָ לְנַחֲלָתוֹ וְחֶבְלוֹ / וְאַשְׁרֵי הָעָם שֶׁכָּכָה לוֹ /



וַיֵּלְכוּ שְׂמֵחִים וְטוֹבֵי לֵב אִישׁ לְאָהֳלוֹ / וַיֹּאמֶר הַשָּׂר בֶּאֱמֶת הִשְׂבַּעֱתָּ נֶפֶשׁ הָאֲנָשִׁים בְּצַחְצָחוֹת / כַּאֲשֶׁר בְּעִנְיָן הַשִּׁירִים דִּבַּרְתָּ צַחוֹת / וְאוֹתְךָ לְאֹרֶךְ יָמִים וְשֹׂבַע שְׂמָחוֹת / יִפְקֹד אֲדֹנָי אֲלֹהֵי הָרוּחוֹת.
