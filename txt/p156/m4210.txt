


	
	מחברות עמנואל
	



מחברות עמנואל – מחברת עשרים ושלוש / עמנואל הרומי


אָמַר הַמְּחַבֵּר: בִּהְיוֹתִי עִם הַשָּׂר יָרוּם הוֹדוֹ / חָשְׁבוּ גּוֹאֲלָיו וְרֵעָיו הֱיוֹתִי אִישׁ סוֹדוֹ / אֲהַבְתִּים וַאֲהֵבוּנִי / וּלְרוֹפֵא מֻמְחֶה חֲשָׁבוּנִי / וְאֵלַי יָבֹא כָּל אִישׁ אֲשֶׁר לוֹ הַנֶּגַע / וּמִקְרֶה רָע אוֹ פֶגַע / וְהָיִיתִי מְרַפֵּא אוֹתָם בְּרֶגַע / וַיְהִי הַיּוֹם וַיָּבֹא אֵלַי אִישׁ מִגֹּאֲלֵי הַשָּׂר וּמִזֶּרַע הַמְּלוּכָה / לְבַקֵּשׁ עַל מַחֲלָתוֹ אֲרוּכָה / וַיַּגֶּד לִי כָּל נִגְעֵי לְבָבוֹ / וַיִּּפְצַר בִּי לָשִׂים עֵינִי עָלָיו וּלְהָסִיר עָצְבּוֹ / כִּי הָיָה רֹאשׁוֹ וְרֻבּוֹ בְצוּקָה / וְאִסְטוּמָכְתוֹ בּוּקָה וּמְבוּקָה וּמְבֻלָּקָה / וָאֹמַר אֶל הָאִישׁ חַיֶּיךָ מֵעֲשׂוֹת רְצוֹנְךָ לֹא אֶתְרַפֶּה / הִנְנִי מַעֲלֶה לְךָ אֲרוּכָה וּמַרְפֵּא / וָאָקוּם קַל חִישׁ לְרֶגֶל הַמְּלָאכָה אֲשֶׁר לְפָנַי / וְעוֹרַרְתִּי רַעְיוֹנַי / וְשִׁנַּסְתִּי מָתְנַי / לְהָכִין צֳרִי אֶל מַחֲלָתוֹ / לִהְיוֹת נָקִי מֵאָלָתוֹ / וָאָכִין לוֹ רְפוּאָה נִמְרֶצֶת / מִזִּמְרַת הָרְפוּאוֹת מְקֻבֶּצֶת / הָיְתָה בְּסִפְרֵי אִפּוֹקְרָט רוֹבֶצֶת / עֲשִׂיתִיהָ עַל יַד אִישׁ פִּקֵּחַ / רֹקַח מֶרְקַחַת מַעֲשֶׂה רֹקֵחַ / וָאֵתֵּן לוֹ הָרְפוּאָה תּוֹךְ הַצַּלָחַת / וָאֶבְחַר לוֹ מִן הַשָּׁעוֹת הַמְּשֻבָּחַת / בִּהְיוֹת הַלְּבָנָה בָּעַקְרָב בְּמַעֲלָה צוֹמָחַת / וְחָשַׁבְתִּי כִּי בְאוֹתוֹ הַמַּשְׁקֶה תִּהְיֶה רְפוּאֲתוֹ / וַהֲסַרַת מֻרְאָתוֹ / וְתִתֻּם חֶלְאָתוֹ / כִּי הַמַּשְׁקֶה הַהוּא לְפִי דִבְרֵי הַחֲכָמִים / הָיָה בוֹ לְשַׁלְשְׁלוֹ עֶשְׂרִים פְּעָמִים / וָאֹמַר אֶל הָאִישׁ הִתְבּוֹדֵד אַחֲרֵי שְׁתוֹתְךָ זֶה הַמַּשְׁקֶה / פְּנֵה מִכָּל עֲסָקֶיךָ עַד אֲשֶׁר גּוּפְךָ יִנָּקֶה / וְלֹא תַטְרִיד רַעְיוֹנְךָ בְּדָבָר מֵהַדְּבָרִים / וּסְגוֹר דַּלְתֵי בֵיתְךָ מִכָּל עֲבָרִים / כִּי הָרוּחַ יִגְרֹם לַעֲשׂוֹת תְּקִיעָה וּתְרוּעָה וְּשְׁלֹשָׁה שְׁבָרִים / וְזֵרַזְתִּיו מִכָּל הַצָּרִיך / וּבָזֶה אֵין צֹרֶךְ לְהַאֲרִיךְ / וְהוּא לֹא כֵן יְדַמֶּה וּלְבָבוֹ לֹא כֵן יַחְשֹׁוב / כִּי אַחֲרֵי שְׁתוֹתוֹ הָרְפוּאָה יָצָא יָצוֹא וָשׁוֹב / וַיִּשָּׂא בְנַפְשׁוֹ עֵצוֹת / לְחַבֵּר בַּלַּיְלָה הַהוּא שִׁירִים וּמְלִיצוֹת / וְלֹא סָר הָאִישׁ הַמִּסְכֵּן / כָּל הַלַּיְלָה מֵעֲשׂוֹת כֵּן / עַד אֲשֶׁר לְזֹאת הַסִּבָּה / כֹּחַ הָרְפוּאָה נִכְבָּה / וְלֹא פָעֲלָה בּוֹ דְבַר תּוֹעֶלֶת / וְשָׁכַח מַאֲמָר אַנְשֵׁי הַחָכְמָה כִּי מַחֲשָׁבָה פוֹעֶלֶת / וְאִוֶּלֶת כְּסִילִים אִוֶּלֶת / וַתְּהִי הָרְפוּאָה הַהִיא כְאֵלָה נוֹבֶלֶת / וַיְהִי בְּאַשְׁמֹרֶת הַבֹּקֶר הָלַכְתִּי אֶל בֵּית הָאִישׁ / בֵּן מַחְפִּיר וּמַבְאִישׁ / וָאֶשְׁאַלֵהוּ עַל אֹדוֹתָיו / וְאֵיךְ פָּעַל בּוֹ הַמַּשְׁקֶה וְקוֹרוֹתָיו / וַיִּשָּׁבַע הָאִישׁ בְּחַיֵּי אֲבוֹתָיו / וַיֹּאמַר הַמַּשְׁקֶה וְהַדִּיאָה-פִיְניוֹן אֲשֶׁר חֲשַׁבְתּוֹ נִכְבָּד וְיָקָר / לֹא שִׁלְשְׁלַנִי כְּלָל כָּל עִקָּר / כְּשָׁמְעִי עָמַדְתִּי מַשְׁמִים / וְנָפְלוּ עָלַי אֵימִים / וְלִבִּי עַל הָרְפוּאָה רַע יַחֲרֹשׁ / וְלֹא הָיָה לִי פֶּה לְהָשִׁיב וְלֹא מֵצַח לְהָרִים רֹאשׁ / וּבְעָמְדִי מַרְעִיד וּמִשְׁתָּאֶה / אָמַר אֵלַי הָאִישׁ הַנִּכְאֶה / חַיֶּיךָ הָרוֹפֵא חִבַּרְתִּי הַלַּיְלָה שִׁיר מְפֹאָר וְנֶחְמָד / אֲשֶׁר כָּל הַבָּא אֵלָיו וְעָמָד / וְהִנְנִי שָׂמֵחַ עָלָיו כְּעַל כָּל הוֹן / וְהוּא לֹא יָדַע הֱיוֹתוֹ מֻכֶּה בְשִׁגָּעוֹן וּבְעִוָּרוֹן וּבְתִמָּהוֹן / וַיּוֹצֵא מֵחֵיקוֹ גִלָּיוֹן / כָּתוּב עָלָיו שִׁיר חִבְּרוֹ הָאֶבְיוֹן / חָשַׁב הֱיוֹתוֹ עַל כָּל הַשִּׁירִים עֶלְיוֹן / וְכַאֲשֶׁר הִתְבּוֹנַנְתִּי עָלָיו רְאִיתִיו מָלֵא חֶרְפָּה וּכְדֵי בִּזָּיוֹן / וְאֹמַר אֶל לִבִּי אֵין לִתְמֹהַּ אִם שָׂמַח יוֹנָה עַל הַקִּיקָיוֹן / וּבְוַדַּאי זֶה הוּא מִפֹּעַל הַנִּקָּיוֹן / כִּי מַה שֶּׁהָיָה לוֹ לְהוֹצִיא דֶרֶךְ בֵּית חֹרוֹן תַּחְתּוֹן / הוֹצִיא דֶרֶךְ בֵּית חֹרוֹן עֶלְיוֹן / כִּי הַשִּׁיר אֲשֶׁר חָשַׁב הֱיוֹתוֹ בְמֹאזְנֵי צֶדֶק מְאֻזָּן / כְּדֶרֶךְ שִׁיר כָּל חַרְזָן / הָיוּ מִזִּיו שְׁגִיאוֹתָיו מְלֵאִים מְפִיקִים מִזַּן אֶל זָן / הַכְּתִיבָה הָיְתָה רָעָה / וְהַמְּלִיצָה הָיְתָה נַחֲלָאָה וְצוֹלֵעָה / וְהַמִּשְׁקָל הָיָה מְעֻקָּל / וְהוּא לֹא-פָּנִים קִלְקַל / אָסַף הַשִּׁיר הַהוּא כָּל חֶרְפָּה וּגְנוּת / וּשְׁגָגוֹת לֹא בְמִסְכֵּנוֹת / עַד אֲשֶׁר הָיוּ עָלַי לְטֹרַח נִלְאֵיתִי נְשֹׂא / וַיִּשְׁאָלֵנִי הָאִישׁ וַיֹּאמַר הַהֵטַבְתִּי לִרְאוֹת הַהִשְׂכַּלְתִּי עֲשֹׂה /



וָאַעַן וָאֹמַר חַיֶּיךָ בַּעַל הַמְּלִיצָה שָׁכַחְתָּ שֵׁם הַמֶּרְקַחַת כִּי קָרָאתָ לְדִיאָה-פִינְדְיוֹן דִּיאָה-פִינְיוֹן / עַל דֶּרֶךְ הַצִּידוֹנִים שֶׁיִּקְרְאוּ לַחֶרְמוֹן שִׁרְיוֹן / וְלֹא תֹּאמַר כִּי לֹא פָעֲלָה הָרְפוּאָה / כִּי אַדְּרַבָּא פָּעֲלָה פְּעֻלָּה נִפְלָאָה / וּמִי שָׁמַע כָּזֹאת וּמִי רָאָה / כִּי הוֹצִיאָה מִגּוּפְךָ כָּל טִנּוּף וְטֻמְאָה / וְכָל בָּאְשָׁה וְצַחֲנָה וְחֶלְאָה / בְּלֹא צַעַר וּבְלֹא מַכְאוֹב / כְּאִלּוּ קסְמָה לְךָ בָּאוֹב / וְעָשְׂתָה אִתְּךָ נוֹרָאוֹת / וּגְבוּרוֹת וְנִפְלָאוֹת / וְזֶה לְךָ הָאוֹת / כִּי מַה שֶּׁהָיָה לָהּ לְהוֹצִיא מִלְמַטָּה הוֹצִיאָה דֶרֶךְ הַלָּשׁוֹן / וְכָזֶה לֹא הָיָה מִימוֹת אָדָם הָרִאשׁוֹן / שָׂא נָא עֵינֶיךָ וּרְאֵה הַשִּׁיר אֲשֶׁר הֲקֵאֹתוֹ / אֵיךְ יַעֲלֶה בָאְשׁוֹ וְתַעֲלֶה צַחֲנָתוֹ / אֵין לִתְמֹהַּ אִם הָיָה לְךָ / עִם הֶחֳלָאִים מִלְחָמָה עֲרוּכָה / וְכֹבֵד בָּאִסְטֻמְכָא / וּדְקִירַת הָעֵינַיִם / וּמַעַד הַמָּתְנַיִם / וְרִפְיוֹן הַשִּׁנַּיִם / וְצִלְצוּל הָאָזְנַיִם / וְכַיּוֹצֵא בָהֶם / מִן הֶחֳלָאִים לְמִינֵיהֶם / אָמְנָם יֵשׁ לִתְמֹהַּ אֵיךְ הָיִיתָ עַד כֹּה בַחַיִּים / אַשְׁרֶיךָ שֶׁנִּשְׁאֲרוּ מֵעַתָּה אֲבָרֶיךָ נְקִיִּים /



כִּשְׁמֹעַ הָאִישׁ בּוֹשׁ וְנִכְלָם / וְקָפַץ פִּיו וְנֶאֱלָם / וְהָיְתָה לוֹ חֶרְפַּת עוֹלָם / וְנִסְתַּר בְּיַרְכְּתֵי בֵיתוֹ / וַאֲנִי הָיִיתִי נֶחְפָּז לִבְרֹחַ מֵאֵימָתוֹ / פֶּן יָעִיר עָלַי כָּל חֲמָתוֹ /



וּבְשׁוּבֵנִי בֵית הַשָּׂר וּרְאִיתִיו מֵרָחוֹק / אָז יִמָּלֵא פִי שְׂחוֹק / וַיִּתְמַהּ הַשָּׂר עַל רֹב שַׂחֲקִי / אֲשֶׁר לֹא מֵחֻקִּי / וַיֹּאמַר שְׂמְתָּנוּ עַל רֹב שְׂחָקְךָ תְּמֵהִים /



וָאַעַן וָאֹמַר צְחוֹק עָשָׂה לִי אֱלֹהִים / וְכִרְאוֹתִי כִּי הַשָּׂר לָדַעַת הַמַּעֲשֶׂה יִכְסוֹף/ סִפַּרְתִּי לוֹ כָּל הַמְּאֹרָע מֵרֹאשׁ וְעַד סוֹף /



כִּשְׁמֹעַ הַשָּׂר נִמְלָא פִּיו שְׂחוֹק / וְהַקּוֹל נִשְׁמַע לְמֵרָחוֹק
/ וַיִּשָּׁבַע הַשָּׂר בְּחַיֵּי רֹאשׁוֹ / כִּי זֶה הַמַּעֲשֶׂה הִשְׂבִּיעַ בְּצַחְצָחוֹת נַפְשׁוֹ / וַיֹּאמֶר אֵלַי הִנְנִי נִשְׁבָּע בְּנֶפֶשׁ אֲבוֹתַי / כִּי לֹא נָעַם לִי סִפּוּר כַּזֶּה כָּל יְמוֹתַי / וּלְעוֹלָם לֹא עָשִׂיתָ גְבוּרוֹת נִפְלָאוֹת / רַק הַפַּעַם הַזֹּאת / אָמְנָם קָשֶׁה בְעֵינַי שֶׁהִפְסַדְתָּ הַהוֹצָאוֹת / אֲשֶׁר הוֹצֵאתָ בַמִּרְקַחַת / אַף עַל פִּי שֶׁלֹּא הוֹצִיא הַסָּכָל דָּבָר מִתָּחַת /



וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר אִם יַגִּיעַ מִסִּכְלוּתוֹ שֶׁלֹּא לִפְרֹעַ / אֲחַבֵּר עָלָיו שִׁיר שֶׁלֹּא רָאִיתָ כָמוֹהוּ לָרֹעַ / וּבִרְאוֹתוֹ אוֹתוֹ יֹאמַר עֵת לִקְרֹעַ / וְרָעָה עָלָיו זֹאת מִכָּל הָרָעָה אֲשֶׁר מְצָאַתְהוּ מִיָּמָיו / כִּי אֲפַרְסֵם סִכְלוּתוֹ לְעֵינֵי בְנֵי עַמָּיו / וְיִהְיֶה לְמָשָׁל וְלִשְׁנִינָה / וְלֹא יֶחְסַר כָּל יָמָיו מְגִנָּה /



וָאָקוּם בְּמִצְוַת הַשָּׂר מְשִׁיחִי / וָאֶשְׁלַח מַלְאָךְ אֶל הָאִישׁ לִשְׁלֹחַ אֵלַי שְׂכַר טָרְחִי / וּשְׂכַר הַמִּרְקַחַת הַמְּשֻׁבַּחַת / אֲשֶׁר פְּדָעַתְהוּ מֵרֶדֶת שַׁחַת / וַיְמָאֵן הָאִישׁ וַיִּשָּׁבַע בְּחֵי נַפְשׁוֹ / כִּי לֹא יִשְׁלַח אֵלַי מְאוּמָה מֵרְכוּשׁוֹ / אָז נָתַתִּי לַחֲטֹא חִכִּי וְלִשְׁאֹל בְּאָלָה נַפְשׁוֹ.




וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

קוּם הַשָּׂר קַח לְךָ גִלָּיוֹן



וּכְתֹב בּוֹ שִׁיר מִשִּׁיר צִיּוֹן



שִׁיר חִבַּרְתִּי עֵת נָתַתִּי



אֶל יוֹסֵף מַשְׁקֶה אֵזוֹבִיוֹן



חָשַׁבְתִּי כִּי יָרִיק מֵעָיו



יִפְדֶּנּוּ מִיַּד כִּלָּיוֹן



אָמַרְתִּי לוֹ דוֹד הֵחָבֵא



וּסְגֹר דֶּלֶת הָאַפִּרְיוֹן



הִשָּׁמֵר מִצִּנִּים פַחִים



כִּי זֶה לְךָ מִפִּיד יוֹם פִּדְיוֹן



הַבֹּקֶר אוֹר שִׁחַרְתִּיהוּ



לִרְאוֹת אִם יָרַד לַטַּמְיוֹן



וּמְצָאתִיהוּ וּבְיָדוֹ שִׁיר



מָלֵא סִיגִים עִם שִׁגָּיוֹן



שִׁיר מָלֵא חֶרְפָּה וּכְלִמָּה



אֵין שִׁעוּר אֶל הַבִּזָּיוֹן



כַּאֲשֶׁר אֵין שִׁעוּר לַפֵּאָה



לַבִּכּוּרִים לָרֵאָיוֹן



וּלְחֶרְפַּת סִכְלוּתוֹ קָרָא



בִּשְׁמוֹ מַשָּׂא גֵיא חִזָּיוֹן



יַעַן חָשְׁבוֹ כִּי עַל תֵּבֵל



אֵין עֵרֶךְ אֵלָיו אוֹ דִמְיוֹן



אָמַרְתִּי שָׂמַחְתָּ עַל שִׁיר



תַּחְרִיד צוּרָתוֹ כָּל רַעְיוֹן



כִּשְׂמֹחַ יוֹנָה הַנָּבִיא



עֵת שָׂמַח עַל הַקִּיקָיוֹן



וָאֶשְׁאַל הָאִישׁ וָאֹמַר



מַה פָּעַל הַדִּיָאה פִינְדְּיוֹן



וַיֹּאמֶר לִי חַיֶּיךָ לֹא



פָּעַל דָּבָר הַדִּיאָה פִינְיוֹן



הֶחֱלִיף אֶת שֵׁם הַמִּרְקָחוֹת



כִּקְרֹא אֶל הַר חֶרְמוֹן שִׁרְיוֹן



וָאַעַן אַל תֹּאמַר כִּי לֹא



פָּעַל דְּבַר הַנִּקָּיוֹן



כִּי בִּלְתּוֹ הָיִיתָ אַח אוֹ



רֵעַ אֶל מַחְלוֹן אוֹ כִלְיוֹן



שׁוּר מֵרְאֹתְךּ גַּם חֶלְאָתְךָ



שִׁיר הִקֵאתָ אִישׁ הָאֶבְיוֹן



כִּי תַחַת צֶאתוֹ מִתַּחַת



יָצָא מִבֵּית חֹרוֹן עֶלְיוֹן.




וַיֹּאמֶר הַשָּׂר חִבַּרְתָּ שִׁיר עַל חֶרְפַּת שִׁירָתוֹ / הֲלֹא תְחַבֵּר שִׁיר עַל נַבְלוּתוֹ /



וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר לוּ יָדַעְתִּי הֱיוֹתוֹ נָבָל בַּטֶּבַע / הָיִיתִי מוֹסִיף לְיַסֵּר אוֹתוֹ כְּחָטֹאתָיו שֶׁבַע / אָמְנָם בְּמַה שֶּׁחָטָא לִי אֵין לְהָבִיא רְאָיָה / מִשּׁוּם מַעֲשֶׂה שֶׁהָיָה /



וַיֹּאמֶר הַשָּׂר דַּע אֶל נָכוֹן כִּי הוּא אֲבִי אֲבוֹת הַכִּילוּת / וְלֹא הִשִּׂיג אָדָם אֲבָקוֹ בְּעִנְיָן הַנֶּבְלוּת /



וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר עָלָיו שִׁיר אֲחַבֵּר / וּלְהַגִיד נַבְלוּתוֹ חֵילִים אֲגַבֵּר.




וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

הוֹאִיל לִבִּי בֵּאֵר זֻלּוּת



שַׂר יוֹסֵף כִּי לִמְאֹד גָּדַל



הֵחֵל לִסְפֹּר אַךְ בִּרְאוֹתוֹ



כִּי אֵין מִסְפָּר לִסְפֹּר חָדַל



יוֹם כִּלַי לְכִילוּת עָשָׂה



סֻכָּה הוּא בָנָה לוֹ מִגְדָּל



כַּצַּדִּיק כָּרָשָׁע הִשְׁוָה



לֹא נִכַּר שׁוֹעַ לִפְנֵי דָּל



לָמַד לִקְפֹּץ יָד כִּי לֹא יִ-



פֹּל מִיָּדוֹ גַרְגֵּר חַרְדָּל.




וַיֹּאמֶר הַשָּׂר חַיֶּיךָ מֵעוֹלָם לֹא הוֹדַעְתָּנִי מַעֲשֶׂה שִׂמְחָה כָזֶה / וּבְטוֹב עָשִׂיתָ נְקָמָה מֶזֶּה הַנִּבְזֶה /



וָאַעַן וָאֹמַר חַיֶּיךָ לוּ שָׁמַעְתָּ מִפִּי / אֵת אֲשֶׁר קָרָה לִי בִימֵי חָרְפִּי / עִם חֲשׁוּקָתִי הַנְּעִימָה / אֲשֶׁר הָיְתָה בְנָיוֹת בָּרָמָה / וְאֵיךְ עֲזַרְתָּנִי חָכְמָתִי עִמָּהּ / אָז נִפְלֵאתָ הַפְלֵא וָפֶלֶא עַל הַמְּאֹרָע / וְיָדַעְתָּ כִּי לְדַעְתִּי בָחוֹר בַּטּוֹב וּמָאוֹס בָּרָע / הוֹשִׁיעָה לִי יְמִינִי וּסְגוֹר לִבָּהּ הַקָּשֶׁה קָרָע /



וַיֹּאמֶר הַשָּׂר הוֹדִיעֵנִי אֵיךְ הָיָה הַמַּעֲשֶׂה / וְלֹא תַעֲלִים מִמֶּנִּי דָבָר וְלֹא תְכַסֶּה /



וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר חָשַׁקְתִּי בִימֵי חָרְפִּי אִשָּׁה נְעִימָה / יָפָה כַלְּבָנָה בָּרָה כַּחַמָּה / חֲשַׁקְתִּיהָ חֵשֶׁק עָצוּם וְאַכְזָר / בְּחָשְׁבִי לִהְיוֹת מִמֶּנָּה נֶעֱזָר / וְהִיא חֲשָׁבַתְנִי לְזָר / אֲנִי רוֹדֵף אַהֲבָתָהּ וְהִיא תִבְרָח / אֶתְחַנֵּן אֵלֶיהָ בְלֵב נִשְׁבָּר וְהִיא מָרָה תִצְרָח / אֲנִי שׁוֹכֵן בַּמַּעֲרָב וָהִיא בַּמִּזְרָח / וְאִם אֶשְׁכֹּן אֲנִי בַּצָּפוֹן הִיא בַדָּרוֹם / וְאִם אֲנִי בִשְׁאוֹל תַּחְתִַּית הִיא תַמְרִיא בַמָּרוֹם / וְאִם אֲנִי בַמָּרוֹם הִיא תֵרֵד לִבְאֵר שַׁחַת / וְלֹא מָצָאתִי לְעוֹלָם עִמָּהּ נַחַת / וַיְהִי הַיּוֹם הֵשִׁיב הַזְּמָן אֶת הַגְּזֵלָה אֲשֶׁר גָּזָל / וֶהֱבִיאַנִי הַמַּזָּל / לִרְעוֹת בַּגַּנִּים / וְלִלְקֹט שׁוֹשַנִּים / וְלִרְאוֹת הֲפָרְחָה הַגֶּפֶן הֵנֵצוּ הָרִמּוֹנִים / וּבִהְיוֹתִי הוֹלֵךְ בַּגַּן לְרוּחַ הַיּוֹם עָמַדְתִּי מִשְׁתָּאֶה / וָאֶשָּׂא עֵינַי וָאֶרְאֶה / וְהִנֵּה הַיָּפָה עֵינֵיהָ כְּיוֹנִים עַל אֲפִיקֵי מַיִם / זוֹרַחַת כַּשֶּׁמֶשׁ בַּחֲצִי הַשָּׁמַיִם / מָלְאָה הָאָרֶץ מִזָּהֳרָה זֹהַר / וְהִיא כְּמַעֲשֶׂה לִבְנַת הַסַּפִּיר וּכְעֶצֶם הַשָּׁמַיִם לָטֹהַר / אָז אִם עָלַז לִבִּי אֵין לִתְמֹהַּ / וּבֵרַכְתִּי לַגָּבֹהַּ מֵעַל גָּבֹהַּ / וָאָקוּם וָאֵלֵךְ בְּלֵב עָלֵז אֶל הַיָּפָה / אֲשֶׁר כְּמוֹ שַׁחַר נִשְׁקָפָה /



וָאֹמְרָה אֵלֶיהָ אַחַר בַּקָּשַת מְחִילַת הַשְּׁגָגָה / שָׁלוֹם עָלַיִךְ הַנָּאוָה וְהַמְּעֻנָּגָה / אֲשֶׁר בְּשׁוֹשַנֵּי הַיְּדִידֹת סוּגָה / אִמְרִי נָא עֲדִינָה גְבֶרֶת מַמְלָכוֹת עַד מָתָי תִּהְיֶה נַפְשִׁי הַנּוּגָה / עַל אֲפִיקֵי מֵי יְדִידוּתֵךְ כְּאַיָל עֲרוּגָה /וְאוֹר יִפְעָתֵךְ אוֹתִי יוֹם יוֹם עֹשֵׁק / וַאֲנִי עָשִׂיתִי בַעֲבוּרֵךְ מַה שֶּׁלֹּא עָשָׂה לְעוֹלָם עַל חֲשׁוּקָתוֹ חֹשֵׁק / וְלוּ הוֹאַלְתְּ הָיִיתִי עֲפַר רַגְלַיִךְ לוֹחֵךְ וְנוֹשֵׁק / אַחֲלַי גְּבִרְתִּי בַּפַּעַם הַזֹּאת גַלִּי אֵלַי מַצְפּוּן לִבֵּךְ הַטָּמוּן בְּחֻבֵּךְ / וְלֹא תָלִינִי מַחְשָׁבוֹת אוֹנֵךְ עוֹד בְּקִרְבֵּךְ / אִם תַּחְפְּצִי בְחִשְׁקִי הִנֵּנִי / וְאִם תְּמָאַנִּי / חִשְׁקִי חֵשֶׁק עָפְרָה אַחֶרֶת לְעוֹלָם לָא יַעֲנֵנִי/



כִּרְאוֹת הַיָּפָה אוֹתִי קוֹרֵא מִן הַמֵּצַר / וְהִיא הָיְתָה בְמִשְׁעוֹל צָר / אֲשֶׁר אֵין דֶּרֶךְ לִנְטוֹת יָמִין וּשְׁמֹאל / וְזָכְרָה אֲשֶׁר עָשִׂיתִי בַעֲבוּר אַהֲבָתָהּ גַּם מִשִׁלְשׁוֹם גַּם מִתְּמוֹל / וַתֹּאמֶר בְּלִבָּהּ יֶשׁ-לִי עַל הָאִישׁ הַזֶּה לַחֲמֹל / וְלֹא כִגְמוּל יָדָיו עָלָיו אֶגְמֹל / טוֹב לִי לְהָסִיר מִלִּבּוֹ אַהֲבָתִי לִצְמִיתוּת / וְכָתַב לָהּ סֵפֶר כְּרִיתוּת / וְלֹא יִרְדֹּף אוֹתִי בַשְּׁוָקִים וּבַרְחוֹבוֹת וּבְכָל עֲבָרִים /
כַּאֲשֶׁר יִרְדֹּף הַקֹּרֵא בֶּהָרִים / אֲבַקֵּשׁ מִזֶּה הַסָּכָל /אֲשֶׁר לַעֲשׂוֹת לְעוֹלָם לֹא יוּכָל / אֲנַסֶּה אוֹתוֹ בְחִידוֹת אוּלַי תִּהְיֶיןָ יְדֵי שִׂכְלוֹ מֵהוֹשִׁיעַ כְּבֵדוֹת / אִם אַשְׁמִיעֵהוּ חִידָתִי בִמְשָׁלַי / וְהוּא לֹא יוּכַל לְהַגִּידָהּ אֵלַי / וּתְכַל תְּלוּנָתוֹ מֵעָלַי /



וַתֵּפֵן אֵלַי וַתִּשָּׂא אֶת קוֹלָהּ וַתֹּאמַר / הוֹי הַחֹשֵׁק הַנִּמְהָר וְהַמַּר / אֲשֶׁר אֵלַי תִּתְמַרְמָר / וּבְשָׂרְךָ עָלַי סָמָר / הִנְנִי נִשְׁבַּעַת בְּחַיֵּי הַנּוֹרָא וְהָאָיוֹם / כִּי בִשְׁתַּיִם תִּתְחַתֵּן בִּי הַיּוֹם / אֲנִי אָחוּד לְךָ חִידָתִי / אֲשֶׁר לָכַדְתִּי בִמְצוּדָתִי / וְאִם יִגְזֹר הַמַּזָּל וְתַעֲמֹד בְּסוֹדָהּ / וְתוּכַל הַחִידָה לְהַגִּידָהּ / הִנְנִי נִשְׁבַּעַת בְּאֵל עֵילוֹם / כִּי הִנְנִי נוֹתֶנֶת לְךָ אֶת בְּרִיתִי שָׁלוֹם / וְלֹא אוֹסִיף הַחֹשְׁקִים עוֹד לָקוֹב / רַק אָשִׂים לְמִישׁוֹר לְבָבִי הֶעָקוֹב / אַרְכִּיב אֶפְרַיִם יַחֲרוֹשׁ יְהוּדָה יְשַׂדֵּד לִי יַעֲקֹב / וְאִם לֹא תִמְצָא חִידָתִי בִשְׁלֵמוּת / לֹא תְקַוֶּה מִמֶּנִּי לְעוֹלָם נְעִימוּת / וּבְיוֹם רְאוֹתְךָ פָנַי תָּמוּת / הִנְנִי בְּצֵל אִילָן הַתַּפּוּחַ הַזֶּה אֶתְלוֹנָן / אֲשֶׁר פִּרְיוֹ לִתְרוּפָה וְעָלֵהוּ רַעֲנָן / וְכֻלּוּ שָׁלֵו וְשַׁאֲנָן / בְּזֹאת תִּבָּחֵן / אִם בְּעֵינַי תִּמְצָא חֵן / אִם תֹּאמַר לִי דְבָרִים אֲמִתִּים / כְּמִסְפַּר הַתּפּוּחִים אֲשֶׁר בּוֹ נְתַתִּים / וּשְׁקָרִים וַהֲתֻלִים כְּמִסְפַּר הֶעָלִים / אָז מַאֲוַיֵּי לִבְּךָ אַשְׁלִים / וְאִם אֵין זֹאת עֵינַי מֵרְאוֹתְךָ אַעֲלִים / וְאַתָּה תִהְיֶה כְּאַחַד מִן הַנְּבָלִים /



כְּשָׁמְעִי בֵּרַכְתִּי בָּרוּךְ מְנַחֵם אֲבֵלִים / הַמַּחֲזִיר אֲבֵדָה לַבְּעָלִים / וָאַעַן וָאֹמַר שִׁמְעִי צוּרָה הַהֲדוּרָה / אֶת מִסְפַּר הַחִידָה וְאֶת שִׁבְרָהּ / הִנְנִי מַגִּיד לָךְ כְּמִסְפַר הַתַּפּוּחִים דְּבָרִים אֲמִתִּיִּים / לְכָל הַתַּפּוּחִים עֳקָצִים בָּם תְּלוּיִּים / וּשְׁקָרִים כְּמִסְפַּר הֶעָלִים לְהַגִּיד אֵהְיֶה נֶחְפָּז / כָּל עָלֶה הוּא מִזָּהָב וּמִפָּז.




וָאֶשָּׂא מְשָׁלִי וָאֹמַר:




צְבִיַּת חֵן בְּתַפּוּחַ תְּעוֹרֵר



חֲשׁוּקָה לַחֲסוֹת תַּחַת צְלָלָיו



קְרַב לָךְ בַּל קְרַב לִי עַד אֲמָרְךָ



בְּתַפּוּחָיו אֱמֶת אַף שָׁוְא בְּעָלָיו



רְאִי יָדוֹת כְּתַפּוּחַ בְּחֶשְבּוֹן



וְאֶל עָלֶה וַעֲלֶה פָּז בְּשׁוּלָיו



בְּנִיב קָצָר אֲגַל סוֹדֵךְ יְחִידָה



תְּנִי נָא לִי לְמָנָה שׁוֹק וְעָלָיו .




וַיְהִי כִּשְׁמֹעַ הַצּוּרָה הַנְּעִימָה / אֶת מִסְפַּר הַחִידָה הָעֲלוּמָה / הָיְתָה רוּחַ אֲחֶרֶת עִמָּהּ / וַתֹּאמֶר אֵלַי בְּלֵב נִשְׁבָּר / מֵיְיָ יָצָא הַדָּבָר / יָקֹשְׁתִּי לִי וְנִלְכַּדְתִּי / מָצָאתָ הַחִידָה אֲשֶׁר חַדְתִּי / קָנִיתָ אוֹתִי הַיּוֹם בְּשִׂכְלָךְ / וְאוֹר יִפְעָתִי אֲשֶׁר עַל כָּל אוֹר מָלָךְ / נְתַתּוֹ זוֹרֵחַ לְמוּלָךְ / וְהִנֵּה אַהֲבָתִי לְעֻמַּתָךְ וְאֶצְלָךְ / לְעֵינֵי בְנֵי עַמִּי נְתַתִּיהָ לָךְ /



וָאַעַן וָאֹמַר גְּבִירְתִּי וּמַה יִּהְיֶה הַעֵרָבוֹן / אֲשֶׁר תִּתְּנִי אֵלַי לִשְׁבֹּר רְעָבוֹן /



וַתַּעַן וַתֹּאמַר אַחֲרֵי אֲשֶׁר הוֹדִיעַ הָאֵל אוֹתְךָ אֶת כָּל זֹאת אֵין כָּמוֹךָ חָכָם וְנָבוֹן / אֲרוּרוֹת תִּהְיֵיןָ הַצְּבִיּוֹת /
וְיוֹנֵי הַגֵּאָיוֹת / הַמַּעֲלִימוֹת עַיִן מֵהַחֲכָמִים וְהַפִּקְּחִים / וְתַעֲגֹבְנָה לַסְתִּיר עֲצָתָם עִם הָעִוְּרִים וְהַפִּסְּחִים / הִנְךָ דוֹדִי אֶשְׁכּוֹל הַכֹּפֶר בַּעַל הַשִּׁירִים הַצַּחִים / וַאֲנִי כְּשׁוֹשָׁנָה בֵּין הַחוֹחִים / וּכְבָר אָמְרוּ הַחֲכָמִים שְׂפָתַיִם יִשָּׁק מֵשִׁיב דְּבָרִים נְכֹחִים /



וַתְּמַהֵר וַתִּפְשֹׁט אֶת מְעִילָהּ וַתַּשְׁלֵךְ תַּחַת אַחַד הַשִּׂיחִים / וַתֹּאמַר אֵלַי אַתָּה תִּהְיֶה עַל בֵּיתִי וְעַל פִּיךָ אֶשָּׁק / וְלֹא תִהְיֶה מֵחִשְׁקֶי לְעוֹלָם נֶעֱשָׁק / וַתִּקְרַב אֵלַי וַתִּשָּׁקֵנִי / שְׂמֹאלָהּ תַּחַת לְרֹאשִׁי וִימִינָהּ תְּחַבְּקֵנִי / וַיִּהְיוּ דֹדֶיהָ לִי עֲרֵבִים / וְעַד נְטוֹת הַיּוֹם הִתְעַלַּסְנוּ בָאֳהָבִים / וַתְּהִי אַהֲבָתֵנוּ יוֹם יֹום נוֹסֶפֶת / הוֹלֶכֶת וְעוֹדֶפֶת / עַד אֲשֶׁר בָּאַחֲרִית הַמָּוֶת חֲטָפָהּ / וּמְאַסֵּף הַנְּפָשׁוֹת הַחֲסִידוֹת אֲסָפָהּ / עַד פֹּה הִגִּיעַ מַעֲשֶׂה הַיָּפָה.



כִּשְׁמֹעַ הַשָּׂר אֶת אֲשֶׁר חָכְמָתִי עֲזָרַתְנִי / וְעַל בָּמֳתֵי הַיְּעֵלָה הִרְכִּיבַתְנִי / וַתְּנִיקֵנִי דְּבַשׁ מִסַּלְעָהּ / וְשֶׁמֶן מֵחַלְמִישׁ צוּר לְשָׂבְעָה / אָמַר הִנְנִי נִשִׁבָּע בְּנוֹטֵה שָׁמַיִם כַּיְרִיעָה / כִּי אַתָּה רִאשׁוֹן לְכָל סֵבֶל בֵּית הַחֹשְׁקִים / וְאֵין כָּמוֹךָ נוֹטֶה אָהֳלֵי הַחָכְמָה וִירִיעוֹתֶיהָ מֵקִים / אָמְנָם הַזֹּקֶן הִכְבִּיד עָלֶיךָ /וּבְסִבָּתוֹ אֶחֱרוּ פַּעֲמֵי מַרְכְּבוֹת מְשָׁלֶיךָ / וּמִיּוֹם אֲשֶׁר חָלִיתָ אֶת רַגְלֶיךָ / בֵּין מְשׁוֹרְרֵי הַזְּמָן לֹא נִשְמַע קוֹלֶךָ / וְרַעַשׁ רִכְבְּךָ וַהֲמוֹן גַּלְגִּלֶיךָ / כַּאֲשֶׁר שְׁמַעֲנוּם בְּסוֹד אֱלוֹהַּ עֲלֵי אָהֳלֶיךָ /



וָאַעַן וָאֹמַר אֲדֹנִי לוּ לִדְבָרַי הִקְשַׁבְתָּ / וּבְסוֹד נְבוֹנִים יָשַׁבְתָּ / אָז יָדַעְתָּ כִּי אֵין הַדָּבָר כַּאֲשֶׁר חָשַׁבְתָּ אָתָּה / כִּי בְעִנְיַן הַשִּׁיר וְהַמְּלִיצָה כְּכֹחִי אָז וּכְכֹחִי עָתָּה / וּמָתַי חִבַּרְתִּי שִׁירִים מְתוּקִים / בְּמַעֲנֵה הַשֵׂכֶל יְצוּקִים / רַק בִּהְיוֹת יְצוּרַי נֶאֱנָחִים וְנֶאֱנָקִים / בַּחֲלוֹתִי אֶת רַגְלַי / וְאָז נָשָׂאתִי מְשָׁלַי / וְהִנָּם הַיּוֹם עַל כַּפַּיִם חֲקוּקִים / וְעַל גַּרְגְּרוֹת אַנְשֵׁי הַחָכְמָה כַעֲנָקִים / וּזְהַב הַשֵּׂכֶל מֵעֲלֵיהֶם מְרִיקִים /



וַיֹּאמֶר הַשָּׂר הֲזֶה חַסְדְּךָ אֶת רֵעָךָ / וּמַדּוּעַ לֹא הֶרְאִיתַנִי מַעֲשֶׂה צַעֲצוּעֶיךָ /



וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר עָשִׂיתִי בְיָמַי שִׁירִים נִכְבָּדִים / מַשָּׂא צֶמֶד פְּרָדִים / הֲתַחְשֹׁב כִּי רָאִיתָ כֻּלָּם / חַיֶּיךָ לוּ רָאִיתָ הָעֲשִׂירִית מֵהֶם עָמַדְתָּ נֶאֱלָם / וְלוּ יִסְפְּרֵם אִישׁ תִּבָּקֵעַ הָאָרֶץ לְקוֹלָם / וּמִי שֶׁיַּחְשֹׁב כִּי הִגִּיעַ לְתַכְלִיתָם חֲלוֹם חָלָם / וּלְבַעֲבוּר תֵּדַע כִּי לֹא שָׁכַחְתִּי הַשִּׁירִים וְלֹא אֶשְׁכָּחֵם / אַזְכִּיר הַשִּׁירִים אֲשֶׁר חִבַּרְתִּי בַּחֲלוֹתִי אֶת רָגְלַי מָהֵר וְקָחֵם / יֵדַע אֲדֹנִי הַשָּׂר כִּי בִּהְיוֹתִי בְעִיר אַנְקוֹנָ"ה / חָלִיתִי אֶת רַגְלַי חֳלִי כָבֵד וְעָבַר עַצְמֹנָה / עָמַדְתִּי כָּל יְמֵי הָעֹמֶר כְתֹמֶר מִקְשָׁה / לֹא נִגַּשְׁתִּי אֶל אִשָּׁה / כִּי הָיָה כְאֵבִי נֶצַח וּמַכָּתִי אֲנוּשָׁה / וְיַד הַמַּחֲלָה הוֹלֶכֶת הָלֹךְ וְקָשָׁה / רָעַבְתִּי וְצָמִיתִי / לֹא אָכַלְתִּי דָג וְיַיִן לֹא שָׁתִיתִי / צָעַקְתִּי מִכְּאֵב לֵב וּמִשֶּׁבֶר רוּחַ אֲיֵלִיל / וְשָׁכַחְתִּי שִׂמְחַת לֵבָב הַהוֹלֵךְ בֶּחָלִיל / כִּי הָיָה כְאֵבִי כָלִיל / וְנִרְאֶה בַעֲלִיל / וְהָרוֹפְאִים הָיוּ רוֹפְאֵי אֱלִיל / וּבִהְיוֹתִי בְּחֹזֶק הַמַּכְאוֹב / קָסְמָה נַפְשִׁי כְאִלּוּ בָאוֹב / וַיַּעַן אֲדֹנָי מִן הַסְּעָרָה / וְחִבַּרְתִּי זֹאת הַשִּׁירָה הַיְּקָרָה.



וָאֶשָׂא מְשָׁלִי וָאֹמַר:




מִי הֶאֱמִין כִּי כָל יְמֵי הָעֹמֶר



מִקְשָׁה אָהִי יֹשֵׁב לְדִמְיוֹן תֹּמֶר



כִּי דָמְמוּ חוּשַׁי וְנִפְעַמְתִּי וְאֵין



מִן הָאֱנוּשׁוּת בִּי לְבַד הַחֹמֶר



לִבִּי אֲשֶׁר הָגָה בְּדַת אֵל יוֹם וְלֵיל



מֵאֵן כְּהַיּוֹם לֶאֱמֹר וַיֹּאמֶר



צָדוּ צְעָדַי וַעֲצָמַי נִפְזְרוּ



לוּ בָּעֲשָׁא אָבִי וְאִמִּי גֹמֶר



אֶזְכֹּר יְמֵי שִׁבְתִּי כְּמֶלֶךְ בַּגְּדוּד



לִי יִחֲלוּ עַמִּי וְאֶגְזֹר אֹמֶר



צַר לִי צֳרִי גִלְעָד וְאֵין אִישׁ יְתְּנָהּ



מִמָּךְ בְּצִנְצַנְתִּי מְלֹא הָעֹמֶר .




כִּשְׁמֹעַ הַשָּׂר הַשִּׁיר הַנִּמְרָץ / אָמַר בְּחַיָּי בְּיָמֶיךְ הָיָה חָזוֹן נִפְרָץ / וְעַתָּה הוֹדִיעֵנִי בְחַיֶּיךָ / אִם חִבַּרְתָּ עוֹד שִׁיר עַל חָלְיֶךָ /



וָאַעַן וָאֹמַר חֳבַּרְתִּי עַל הָרוֹפְאִים הַסּוֹבְאִים / אֲשֶׁר לִדְרֹשׁ אוֹתִי הָיוּ בָאִים / וּבִרְאוֹתִי אוֹתָם מִתְנַבְּאִים / אֶהְגֶּה אַךְ נְכָאִים / וְאָמַרְתִּי נַפְשִׁי בְתוֹךְ לְבָאִים / וְהִנֵּה הַשִּׁיר אֲשֶׁר חִבַּרְתִּי עֲלֵיהֶם / וַאֲקַלְלֵם בְּשֵׁם אֱלֹהֵיהֶם.




חוּשָׁה לְבָבִי קַח לְךָ צִנְצֶנֶת



אַחַת וְתֵן שָׁם מָן מְלֹא הָעֹמֶר



אוֹ מִן צֳרִי גִלְעָד קְחָה לְךָ חֹמֶר



לִהְיוֹת צֳרִי אֶל מַחֲלָה נוֹשֶׁנֶת

אִם אֵלְכָה בַחוּץ בְּלֹא מִשְׁעֶנֶת



אָז אֶחְשׁבֹ לִהְיוֹת כְּדָרְלָעֹמֶר



אֵשֵׁב בְּרֹאשׁ עַמִּים וְאֶגְזֹר אֹמֶר



וּלְגֹאֲלֵי אֶשְׁפּוֹת שְׁלָם וּכְעֶנֶת

אֶל מַעֲצֵבָה תִּשְׁכְּבוּן הָרוֹפְאִים



כִּי נִסְכְּכֶם שֶׁקֶר וְאַתֶּם אָיִן



לִי תִּשְׂחָקוּ עֵת כִּי יְגוֹנַי צוֹבְאִים

תִּרְאוּ חֲתַת וְתַעֲלִימוּ עָיִן



אֵיךְ תֶּחֱשׁוּ הַזּוֹלְלִים הַסּוֹבְאִים



הִנְנִי שְׁכוּר חֵמָה וְלֹא מִיָּיִן .




וַיֹּאמֶר הַשָּׂר חַיֶּיךָ הַמְּשׁוֹרֵר נִחַמְתָּנִי / אַחֲרֵי אֲשֶׁר שִׁירִים מְפֹאָרִים כָּאֵלֶּה הִשְׁמִעְתָּנִי / וְעַתָּה הוֹדִיעֵנִי בְּחַסְדֶּךָ / אִם רָאִיתָ הַשָּׂר הַנִּכְבָּד רִבִּי קָלוֹנִימוֹס מֵעוֹדֶךָ / כִּי שָׁמַעְתִּי כִּי הוּא לְבַדּוֹ בַזְּמָן הָשׁוּק וְנֶאֱהָב וְלוֹ לְשׁוֹן הַזָּהָב וְהָאַדֶּרֶת / וְכָל שָׂר וּמֵלִיץ זוּלָתוֹ כְּגַנָּב נִמְצָא בַמַּחְתֶּרֶת /

כָלִיל הוּא בַהֲדָרוֹ / מַה נּוֹסִיף עוֹד לְדַבֵּר חוֹתַם תָּכְנִית / וְחֵשֶׁק אַהֲבַת מַעֲלָתְכֶם עַל גַבֵּנוּ הֶאֱרִיךְ מַעֲנִית / וּלְתָאֵר מַעֲלָתְכֶם יְדֵי הָעָם כְּבֵדִים / כִּי בָכֶם הַחָכְמָה וְהָעַנֲוָה וְהַנְּשִׂיאוּת רוֹכְבִים צְמָדִים /



וּשְׁאֵלָתְךָ הַשָּׂר הַנִּכְבָּד מוֹרֵנוּ רִבִּי שְׁמוּאֵל מֵאִתָּנוּ / לְיָעֵץ אֶל הַנִּכְבָּד אָחִיךָ פֶּלֶא זְמַנֵּנוּ / לָשׁוּב מַהֵר אֶל מְחוֹז חֶפְצוֹ / אֶל נָוֵהוּ וְאֶל אַרְצוֹ / אַף עַל פִּי שֶׁהַשְּׁאֵלָה הִיא לָנוּ נֶגְדִית / לְמַעֲלַת הַנְּזָקִין נָשׂוּם לָהֶם בִּעִידֵי עִידִית / לֹֹא נִתֵּן דֳּמִי לוֹ / עַד אֲשֶׁר יָבֹא לַחֲזוֹת בְּנֹעַם בְּנֵי בֵיתוֹ וּלְבַקֵּר בְּהֵיכָלוֹ / אַף עַל פִּי שֶׁאֵין צֹרֶךְ כִּי תָמִיד נִרְאֵהוּ מֵצֵר וְדוֹאֵג עַל עַכָּבָתוֹ / וּפַלְגֵי מַיִם יוֹרְדוֹת עֵינָיו עַל גָלוּתוֹ / וְלִבּוֹ בְאֵשׁ הַנְּדֹד בּוֹעֵר / הוֹמֶה וְסוֹעֵר / וּמִשְׁתַּדֵּל בְּכָל עֹז לְהַשְׁלִים מְלֶאכֶת אֲדֹנֵנוּ הַמֶּלֶךְ שֶׁיִּחְיֶה וְאֵת אֲשֶׁר נִגְזַר עָלָיו כְּדֵי לָשׁוּב אֶל מְחוֹז חֶפְצוֹ / וְאֶל מְעוֹן רִבְצוֹ / לְרַחֵף עַל גּוֹזָלָיו וּלְהַפְלָגַת חִשְׁקוֹ לָשׁוּב אֲלֵיהֶם וְלִהְיוֹת בְּבֵיתוֹ כְּבָרִאשׁוֹנָה / יוֹם אֶחָד בַּשָּׁנָה חָשׁוּב שָׁנָה / אָמְנָם מְלֶאכֶת אֲדֹנֵנוּ הַמֶּלֶךְ שֶׁיִּחְיֶה אֲשֶׁר בְּיָדוֹ אֲרֻכָּה מֵאֶרֶץ מִדָּה וּרְחָבָה מִנִּי יָם / וּמַאֲמָר הַמֶּלֶךְ קִיַּם / וּלְהַנִיף יָדוֹ אֶל מְלֶאכֶת הַהַעְתָּקָה בַּעְיָם / וְלֹא יַרְפֶּנָּה עַד תֻּמָּהּ / וּלְמַלֹּאת מִצְוָתוֹ חוֹבָה עֲצוּמָה / וְהָאֵל הַמְּשַׁפֵּר שְׁחָקִים בְּחָכְמָה / הַתּוֹלֶה אֶרֶץ עַל בְּלִימָה / וְהַנּוֹתֵן לָעָם עָלֶיהָ נְשָׁמָה / יַגְדִּיל מַעֲלָתְךָ וְיַעֲשֶׂה שָׁלוֹם לָךְ / שָׁלוֹם יַעֲשֶׂה לָךְ / כְּנֶפֶשׁ אֱדֹנֵנוּ וְנֶפֶשׁ דוֹרְשְׁךָ שְׁלוֹמְךָ וְטוֹבָתְךָ כָל הַיָּמִים / בְּנֵי קְהַל רוֹמָ"א הַחֲתוּמִים / הַנּוֹטִים אֶל מַעֲלָתְךָ כְּנָהָר שְׁלוֹמִים / וּבְרָכוֹת כְּחוֹל יַמִּים / וְכָבוֹד כְּחֶפְצֶךָ / מְלֹא רֹחַב אַרְצֶךָ / עִמָּנוּאֵל.



כִּשְׁמֹעַ הַשָּׂר הַמִּכְתָּב הַנִּבְחָר / אָז קָפַץ וְנִשְׁבַּע וְלֹא אֵחַר / וַיֹּאמֶר הִנְנִי נִשְׁבָּע בַּאֲשֶׁר נָתַן טַל יַלְדּוּתְךָ מֵרֶחֶם מִשְּׁחָר / כִּי אַתָּה בַעַל הַלָּשׁוֹן / וְלַשִּׁיר וְלַמְּלִיצָה גוֹאֵל וְרִאשׁוֹן / וְכָל מְשׁוֹרְרֵי הַזְּמָן נַחַל קִישׁוֹן גְּרָפָם נַחַל קְדוּמִים נַחַל קִישׁוֹן / בָּרוּךְ אֲשֶׁר הֶעֱמִידְךָ לְנֵס עַמִּים / וְלִהְיוֹת נִיר לְבָנָיו כָּל הַיָּמִים /

וָאֵשֵׁב עִם הַשָּׂר בְּשַׁלְוָה וּבְמִשְׁמַנֵּי הַמְּדִינָה / כַּאֲשֶׁר הָיִיתִי אִתּוֹ בְאָמְנָה.
