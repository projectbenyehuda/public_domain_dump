


	
	מחברות עמנואל
	



מחברות עמנואל – מחברת שמונה-עשרה / עמנואל הרומי


אָמַר הַמְּחַבֵּר: בִּהְיוֹתִי עִם הַשָּׂר מַעְיָן הַמִּתְגַּבֵּר / מְפָרֵק הָרֵי הַתְּבוּנָה וּסְלָעֶיהָ מְשַׁבֵּר / הָיִינוּ פַעַם בְּדִבְרֵי חָכְמוֹת נְדַבֵּר / וַחֲיָּלִים נְגַבֵּר / וּפַעַם עַל רִיב לֹא לָנוּ נִתְעַבֵּר / נְדַבֵּר עַל כָּל דּוֹר וָדוֹר מִשְׁפָּחָה וּמִשְׁפָּחָה כְּפִי הָרָצוֹן / פַּעַם בִּדְבָרִים עֲרֵבִים וּפַעַם בְּדִבְרֵי לָצוֹן / וַנַּחֲשֹׁב גְּדוֹלִים מִמֶּנּוּ לְרוֹעֵי צֹאן /



וַיְהִי הַיּוֹם וַיּאֹמֶר אֵלַי הַשָּׂר אֱמֹר נָא בְחַסְדֶּךָ / אֵיךְ הָיָה הָעִנְיָן עִם פְּלוֹנִי אִישׁ סוֹדֶךָ / כַּאֲשֶׁר שִׁבַּח זְכִירָתוֹ עַל זְכִירָתֶךָ / וּשְׁמִירָתוֹ עַל שְׁמִירָתֶךָ / וּבַעֲבוּר אֲשֶׁר גֵּאוּת עָשָׁן הִתְאַבָּךְ / וּמַטָּרָה לְחִצָּיו הִצִּיבָךְ / נֶאֱחַז בְּקַרְנָיו בַּסְּבָךְ / וְאַתָּה הֲשִׁיבוֹתָ לוֹ כְּדַרְכֵי טוּבָךְ / אֱמֹר נָא אֵיךְ הָיָה הַמַּעֲשֶׂה / וְלֹא תַחֲמֹל עָלָיו וְלֹא תְכַסֶּה /



וָאַעַן וָאֹמַר הָיָה פְּלוֹנִי אִישׁ סוֹדִי / הַיּוֹתֵר זָכְרָן אֲשֶׁר רָאִיתִי מֵעוֹדִי / לֹא שָׁכַח לְעוֹלָם דָּבָר רָאָהוּ / אוֹ מִפִּי אַחֵר שְׁמָעָהוּ / וְהָיָה מְסַפֵּר לְאִשְׁתּוֹ וּלְאוֹהֲבָיו אֵת כָּל אֲשֶׁר קָרָהוּ / וְהָיָה מִתְפָּאֵר בְּטוֹב זְכִירָתוֹ / וְשָׂשׂ עַל נֹעַם אִמְרָתוֹ / כִּי הָאִישׁ הָיָה מְדַבֵּר צַחוֹת / וְלוֹ מִבְחַר הַסִּפּוּרִים וְהַשִּׂיחוֹת / וְאִם הָיָה שׁוֹמֵעַ אִישׁ דּוֹרֵשׁ / אוֹ בַתּוֹרָה קוֹרֵא וּמְפָרֵש / הָיָה זוֹכֵר כָּל דִּבְרֵיהֶם / עִקָּרֵיהֶם וְעַנְפֵיהֶם / וּפְנֵיהֶם וְכַנְפֵיהֶם / הַתְחָלוֹתֵיהֶם וְהַנָּחוֹתֵיהֶם וְכַוָּנוֹתֵיהֶם / בְּאוֹתִיּוֹתֵיהֶם / וּבְמלּוֹתֵיהֶם / אִישׁ אִישׁ לִלְשׁוֹנוֹ בְּאַרְצוֹתָם בְּגוֹיֵיהֶם / גַּם הָיָה הַמְּדַבֵּר נוֹצְרִי אוֹ יִשְׂרְאֵלִי / עֵרִי וַאֲרוֹדִי וְאֵרְאֵלִי / כָּמוֹהוּ יְדַבֵּר וְהוּא לֹא פָנִים קִלְקֵל


וַחֲיָּלִים יְגַבֵּר / סוֹף דָּבָר כָּל אִישׁ לְפָנָיו שָׁב נֶאֱלָם / וּמִלְּדַבֵּר בִּמְקוֹמוֹ בּוֹשׁ וְנִכְלָם / וְלֹא נִרְאֵית זְכִירָה כִזְכִירָתוֹ מִן הָעוֹלָם וְעַד הָעוֹלָם/ וַיְהִי הַיּוֹם דִּבֵּר רוּחוֹ אוֹתוֹ / לְפָאֵר בְּמַקְהֵלוֹת זְכִירָתוֹ / וְלוּלֵי הֶאֱרִיךְ לָשׁוֹן עָלַי הָיִיתִי מְקַיֵּם דִּבְרָתוֹ / אָמְנָם רָצָה לְהַגְדִּיל זְכִירָתוֹ בְהַשְׁפָּלַת זְכִירָתִי / וּלְהַעֲלוֹת אִמְרָתוֹ עַל אִמְרָתִי / אָז בְּחָמְלִי עַל מַעֲלָתִי/ אָמַרְתִּי לִלְשׁוֹנִי עִזְרִינִי וְאֶתְנַשָּׂא / כִּי לֹא מְשַׂנְאִי יְחָרְפֵנִי וְאֶשָּׂא/ אָמְנָם הוּא אֱנוֹשׁ כְּעֶרְכִּי / וְעֵת הַרְאֵהוּ כִּי שֶׁקֶר נִסְכּוֹ וְלֹא נִסְכִּי/ אָז דִּבְּרָה לְשׁוֹנִי בְחִכִּי / וְנִרְאֵיתִי לַכֹּל כִּי לֹא מַחְשְׁבוֹתָיו מַחְשְׁבוֹתָי / וְלֹא דְּרָכָיו דְּרָכָי / אָז נִשְׁאַר מִזְּכִירָתוֹ בּוֹשׁ וְנִכְלָם /וּזְכִירָתִי עָלְתָה בַּמַּדְרֵגָה הָעֶלְיוֹנָה שֶׁבַּסֻּלָּם / וְהָיָה לִי שֵׁם עוֹלָם /


וַיֹּאֶמר הַשָּׂר רָצִיתִי לִשְׁמֹעַ בִּפְרָט אֵיךְ הָיוּ מַאֲמָרָיו / וּתְשׁוּבוֹתֶיךָ אֶל דְּבָרָיו /


וָאַעַן וָאֹמַר יֵדַע אֲדֹנִי נֵזֶר הַמְּבִינִים / כִּי יוֹם אֶחָד עָלִיתִי הַשַּׁעֲרָה אֶל הַזְּקֵנִים / וְשָׁם הָיוּ אַנְשֵׁי חֶסֶד אַנְשֵׁי אֱמוּנִים / בְּחֵיק הַמַּעֲלָה אֱמוּנִים / וְאָז הִתְחִיל לְסַפֵּר מִן הַגְּדוֹלוֹת כָּל אִישׁ חָכָם / וַיִּתְנַבֵּא גַּם פְּלוֹנִי בְתוֹכָם / וְהִתְחִיל לְפָאַר זְכִירָתוֹ עַל כָּל הַזְּכִירוֹת / וְלָשִׂים אוֹתָהּ לְמַעְלָה מִגַּלְגַּלֵי הַמְּאוֹרוֹת / וַיֹּאמַר שִׁמְעוּ שָׁמַיִם וְהַאֲזִינִי אָרֶץ / כְּמוֹ שֶׁזְּכִירָתִי פָּרְצָה בִגְבוּל הַזְּכִירוֹת פָּרֶץ / וְעִמָּנוּאֵל דּוֹדִי בְעֵינַי לֹא יוּחַן / כִּי הוּא בַעֲוֹנוֹת שַׁכְחָן / לֹא יִזְכֹּר מַה שֶֹאָכַל הַיּוֹם עַל הַשֻּׁלְחָן / וְהִנְנִי נִשְׁבָּע בְּחַיֵּי מִי אֲשֶׁר אֶאֱהַב וַאֲהַבְתִּיו / כִּי אֵין אִישׁ אֲשֶׁר פַּעַם אַחַת בְּיָמַי רְאִיתִיו / אֲשֶׁר כָּל יְמֵי חֶלְדִּי לֹא הִכַּרְתִּיו / וְבִרְאוֹתִי אוֹתוֹ מֵרָחוֹק בִּשְׁמוֹ קְרָאתִּיו / וְכָל קוֹרוֹתָיו אֲשֶׁר קְרָאוּנִי עִמּוֹ הִזְכַּרְתִּיו / וָאֵדַע שֵׁם כָּל עִיר וּכְפָר וּכְרָךְ / עַד דַּמֶּשֶׂק וְעַד חַדְרָךְ / וְאֶזְכֹּר כָּל הַקּוֹרוֹת וְהַחִדּוּשִׁים / וּשְׁמוֹת הָאֲנָשִׁים וְהַנָּשִׁים / וְהַמְּלָכוֹת וְהַשָּׂרוֹת וְהַפִּילַגְשִׁים / וְעִמָּנוּאֵל יַעֲמֹד כְּאִישׁ מִשְׁתָּאֶה / לֹא יִזְכֹּר דָּבָר מִכָּל אֲשֶׁר יִרְאֶה / יִרְאוּ עֵינָיו וְלֹא יַחְקֹר / וְיִשְׁמְעוּ אָזְנָיו וְלֹא יִזְכֹּר / וְרַעֲיוֹנוֹתָיו יָחֹגּוּ וְיָנוּעוּ כַּשִּׁכּוֹר / וְהִתְחִיל לְסַפֵּר קוֹרוֹת מְלָכִים / וְשָׂרִים וּקְצִינִים וּנְסִיכִים / וְהֶאֱרִיךְ לְדַבֵּר דִּבְרֵי הֲבָלִים וַהֲתֻלִּים / עַד שֶׁשַּׁבְנוּ לְדַבֵּר מִן הַחַתּוּלִים / עַד אֲשֶׁר קַצְנוּ מִשְּׁמֹעַ / כִּמְעַט שֶׁמֵּרֹב הַצַּעַר הָעֵינַיִם הֵחֵלּוּ לִדְמֹעַ / וְכָל עוֹד שֶׁהָיָה מְחַזֵּר לִזְכֹּר הוֹדוֹ / וּלְהַאֲרִיךְ בִּכְבוֹדוֹ / אַךְ בִּי יָשׁוּב יַהֲפֹך יָדוֹ / וּכְשָׁמְעִי דְּבָרִים לֹא כֵנִים / וּקְצָת הָעָם הָיוּ כְמִתְאוֹנְנִים / וְאַחַר זְכִירָתִי מְרַנְנִים / אָמַרְתִּי עֵת לְהַשְׁלִיךְ אֲבָנִים / וְעַפֵּר בֶּעָפָר / כִּי פְּלוֹנִי בְרִית הָאַחֲוָה הֵפַר/

וָאֹמְרָה אַתָּה הָאִישׁ הַמִּתְפָּאֵר בִּזְכִירָה / וַתִּתֵּן אוֹתָהּ עַל רֹאשְׁךָ עֲטָרָה / אֵיךְ לֹא תַּעֲלֶה עַל לְבָבְךָ מַה שֶּׁיֵּש לִי מִן הַזְּכִירָה / בַּחֲמִשָּׁה חֻמְּשֵׁי תוֹרָה / וּמִיהוֹשֻעַ וְשׁוֹפְטִים וּשְׁמוּאֵל / וּמִמְּלָכִים וְיִשַׁעְיָה וְיִרְמִיָה וִיחֶזְקֵאֵל / וּמֵהוֹשֵעַ וְיוֹאֵל /

וּמֵעָמוֹס וְעוֹבַדְיָה / וּמִיּוֹנָה וּמִיכָה וְנַחוּם וַחֲבַקּוּק וּצְפָנְיָה / וּמֵחַגַּי וּמַלְאָכִי וּזְכרְיָה / וּמִדִּבְרֵי הַיָּמִים וּמִדָּנִיאֵל וְעֶזְרָא וְחָמֵשׁ מְגִלּוֹת / וּמֵאִיוֹב וּמִשְׁלֵי וְסֵפֶר תְּהִלּוֹת / וָאֵדַע בְּכָל מִלָּה וּמִלָּה / הַנִּקּוּד וְהַדִּקְדּוּק וְהַפְּעֻלָּה וְהַפְּשָׁט וְהַדְּרָש וְהַסּוֹד / מִקִּירוֹת הַסִּפּוּן וְעַד הַיְּסוֹד / וְסִפְרֵי הַדִּקְדּוּק כֻּלָּם / לְמִקְּטַנָּם וְעַד גְּדוֹלָם / וְסֵפֶר הַשָּׁמַיִם וְהָעוֹלָם / וְסִפְרֵי אִבְּן עֶזְרָא בְּדַרְכֵי הַחֶשְׁבּוֹן / וּפֵרוּשׁ קֹהֶלֶת וּמַאֲמַר יִקָּווּ הַמַּיִם אֶל הֶחָכָם רַבִּי שְׁמוּאֵל אִבְּן תִּבּוֹן / וְסֵפֶר מוֹרֵה הַנְּבוּכִים / וְסֵפֶר תַּלְמַי בְּחָכְמַת הַכּוֹכָבִים וְהַמַּהְלָכִים / וְסֵפֶר הַמִּדּוֹת וְהָעֲרָכִים וּפֵרוּשֵׁי הַמִּשְׁנָה וְהַחִבּוּר / וְסֵפֶר רַבִּי אַבְרָהָם הַנָּשִׂיא בְּסוֹד הָעִבּוּר / וְחֵלֶק הַדִּקְדּוּק וְהַשָׁרָשִׁים לְרַבִּי דָוִד קִמְחִי / הָאוֹמְרִים לְנַפְשִׁי רָנִּי וְשִׂמְחִי / וְסֵפֶר צַחוֹת וּמֹאזְנַיִם / הַנֶּחְמָדִים לְהַשְׂכִּיל וְתַאֲוָה הֵם לָעֵינָיִם / וְסֵפֶר הַחִיּוּג' וְהָרִקְמָה / וְסֵפֶר הַהַשָּׂגָה וְהַהַכְלָמָה / וְסֵפֶר הַחִזָּיוֹן / וְסֵפֶר כְּלִי הַנְּחֹשֶׁת וְהַהִגָּיוֹן / אַתָּה שָׂמְתָּ מְגַמַּת פָּנֶיךָ בִּרְאוֹתְךָ קָצִין אוֹ שָׂר / אוֹ מֶלֶך אוֹ נָגִיד אוֹ טִפְסָר / לָדַעַת מִדּוֹתָיו וְטַכְסִיסָיו / וַחֲיָלָיו וְאוֹכְלוֹסָיו / וְתַעֲנוּגָיו וּמְשׂוֹשָׂיו / וַאֲפוּנָיו וְעָשׂוּי כְּרוּבִים וְתִמּוֹרִים / וְהַלִּשְׁכָּה אֲשֶׁר נֶגֶד הַגִּזְרָה / וּפְנֵי הַכְּרוּב לַתִּמּוֹרָה / וְתִמּוֹרִים אֶל אֵילָו / וּשְׁמוֹנֶה מַעֲלוֹת מֵעָלָיו / וְהָאִישׁ הָעוֹמֵד בֵּין הַהֲדַסִּים אֲשֶׁר בַּמְּצֻלָה / וּמְנוֹרַת זָהָב כֻּלָּהּ / וְעַל רֹאשָׁהּ גֻּלָּה / וּשְׁנֵי הַזֵּיתִים מִימִינָהּ וּמִשְּׂמֹאלָהּ / וְהַמְּגִלָּה עָפָה / וְהָאִשָּׁה הַיּוֹשֶׁבֶת בְּתוֹךְ הָאֵיפָה / וְהָעֲטָרוֹת אֲשֶׁר לְחֵלֶם וּלְטוּבְיָּה / וְלִידַעְיָה וּלְחֵן בֶּן צְפַנְיָה / וְהָאַרְבַּע מֵרְכָּבוֹת הַיּוֹצְאוֹת מִבֵּין שְׁנֵי הָרִים / וְהַסּוּסִים בְּרֻדִּים אֲמֻצִּים לְבָנִים וּשְחוֹרִים / וְאִם בָּאתִי לִזְכֹּר כָּל הַדְּבָרִים / אֲשֶׁר עַל לְבָבִי יַעֲלוּ / יִכְלֶה הַזְּמָן וְהֵם לֹא יִכְלוּ / מִלְבַד אֲשֶׁר אֵדַע הַכַּוָּנוֹת / וְהַסּוֹדוֹת הַנְּכוֹנוֹת / אֲשֶׁר בְּכָל הַדְּבָרִים הַנִּזְכָּרִים צְפוּנוֹת / וְלָכֵן חֲדַל לְךָ / מִזְכֹר זְכִירָתְךָ בַּאֲשֶׁר כְּבוֹדְךָ תַחְמֹד / וּבִמְקוֹם גְּדוֹלִים בְּמַעֲלָה אַל תַעֲמֹד / לֵךְ וְסַפֵּר הֲתֻלֶּיךָ לִקְטַנֵּי אֲמָנָה / וְלַנָּשִׁים הַמּוֹזְרוֹת בַּלְּבָנָה / כִּי שִׂכְלְךָ בַדְּבָרִים הָאֵלֶּה קָטֹן מֵהָכִיל / וְאָז תַּצְלִיחַ אֶת דְּרָכֶיךָ וְאָז תַּשְׂכִּיל /


וַיְהִי כִּשְׁמֹעַ פְּלוֹנִי סְאוֹנִי סוֹאֵן בְּרָעַשׁ / חָמַד לַעֲמֹד מִצְּפוֹן לְהַר גָּעַשׁ / וְהַדְּבָרִים אֲשֶׁר אָמַר לֹא אֲמָרָם / כִּי קִבֵּל נֵזֶק וְצַעַר וּבֹשֶׁת בַּעֲבוּרָם / וַיֹּאמַר הַיּוֹם חָשְׁכוּ כוֹכְבֵי נִשְׁפִּי / הֵן קַלּוֹתִי מָה אֲשִׁיבֶךָּ יָדִי שַׂמְתִּי לְמוֹ-פִי / וְקָפַץ פִּיו וְנֶאֱלָם / וְעָמַד בּוֹש וְנִכְלָם / וְלוֹ חֶרְפַּת עוֹלָם / וְכָל הָעָם קָרְאוּ אַחֲרָיו הֵידָד / וּבִרְאוֹתוֹ כִי שִׁמְשִׁי זָרְחָה נוֹדָד / וְעָמַד בְּפָנִים נִזְעָמִים / וְלֹא רְאִינוּהוּ חֹדֶשׁ יָמִים /



וַיִּשְׂחַק הַשָּׂר וַיֹּאמַר מַה מְּאֹד הֵיטַבְתָּ לְדַבֵּר / אַחֲרֵי הֱיוֹתוֹ עַל רִיב לֹא לוֹ מִתְעַבֵּר / כָּכָה יֵעָשֶׂה לְכָל אִישׁ / אֲשֶׁר יִתְהָלֵּל וְרֵיחַ רֵעָיו בְּעֵינֵי הָאֲנָשִׁים יַבְאִישׁ /



וָאֹמַר אֶל הַשָּׂר יֵדַע אֲדוֹנִי נֵזֶר הוֹדִי / כִּי לוּלֵי עָנְיִּי וּמְרוּדִי / וַאֲשֶׁר חָמַלְתִּי עַל כְּבוֹדִי / הָיִיתִי מַכֶּה אוֹתוֹ בְמַטֶּה אֲשֶׁר בְּיָדִי / כַּאֲשֶׁר עָלַי הֶאֱרִיךְ לָשׁוֹן / אָמְנָם אוֹצְרוֹתַי גְּרָפָם נַחַל קִישׁוֹן / וְהַחֶסְרוֹן אֵינֵנוּ מַנִּיחַ לִי לִישׁוֹן / עַד יַשְׁקִיף וְיֵרֶא אֲדֹנָי מִשָּׁמַיִם / וְצָרוֹת לְבָבִי יִמָּאֲסוּ כְמוֹ מַיִם / וְלֹא תָקוּם צָרָה פַעֲמַיִם / וְתַחַת מַחֲגֹרֶת שַׂק מְעִיל הוֹד אֶעֱדֶּה / וְהַקּוֹל אֲשֶׁר אֹמֵר רְדָה יֹאמַר רְדֵה /

  

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

  

אֵלִי פְּדָעֵנִי בְּעֻזָּךְ מִידֵי



עֹנִי וְתַחַת שַׂק מְעִיל הוֹד אֶעֱדֶה



צַו לַזְּמָן אֵל נֶאֱמָן לִי יַעֲשֶׂה



שָׁלוֹם וְקוֹל אֹמֵר רְדָה יֹאמַר רְדֵה.

  

וַיֹּאמֶר הַשָּׂר מַה מְּאֹד חָטָא הַזְּמָן / כַּאֲשֶׁר לֹא בָנָה לְךָ בַּיִת נֶאֱמָן / וַיַּחֲלֹק כְּבוֹדוֹ לִפְנֵי אֱלִיפַז וְתֵימָן / וַיַּלְבִּישֵׁם תְּכֵלֶת וְאַרְגָּמָן / וְלֹא יִתֵּן לָהֵם צָרַעַת נַעֲמָן / וְכַאֲשֶׁר יִשְׁאָלוּהוּ בָנָיו לְמִי נַמְטִיר הַמָּן / יֹאמַר לַמְהוּמָן / וְלֹא יִתֵּן מָרְדְּכָי עַל בֵּית הָמָן / וְאֵיךְ לֹא יַחֲלֹק כְּבוֹדוֹ לַאֲחִימָן / וּלְכַלְכֹּל וּלְדַרְדַּע וְהֵימָן / וּמַדּוּעַ לֹא חָלַק לְךָ מִן הַהוֹן לְשָׂבְעָה / כַּאֲשֶׁר חָלַק לְךָ מֵזִמְרַת הַיְּדִיעָה / וְלִבְנֵי הַפִּילַגְשִׁים נָתַן מַתָּנוֹת וְלִבְנֵי בְרִיעָה / וְלָאִישׁ אֲשֶׁר יִקְרַב אֶל כָּל בְּהֵמָה לְרִבְעָהּ / וְלֹא יִתֵּן לָהֶם כָּל רֹאש מֻקְרָח וְכָל זָקָן גְּרוּעָה / וּמִן הַקְּלָלוֹת שֶׁבְּיִרְמִיָה אֹרֶךְ הַיְרִיעָה / וְאָמַר הֶחָכָם מִמּוּמֵי זֶה הָעוֹלָם וּמֵהֶבְלוֹ / שֶׁהוּא לֹא יָחֹן אִישׁ וָאִיש


כְּפִי שִׂכְלוֹ / רַק יָחֹן אֲנָשִׁים לְבָבָם לְמֵרָע / וּמִצַּדִּיק עֵינָיו יִגְרָע / יַהֲרֹג יֹאשִׁיָהוּ וְיִמְלֹך חָפְרַע / וְיָמִיר רַע בְּטוֹב וְטוֹב בְּרָע /


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

  

אָמַר הֶחָכָם מִמּוּמֵי



זֶה הָעוֹלָם אוֹ מִן הֶבְלוֹ



לֹא יָחֹן אִישׁ מַה שֶּׁרָאוּי



רַק יוֹסִיף לוֹ אוֹ יִגְרַע לוֹ.

  

וָאֹמַר אֶל הַשָּׂר חַנַּנִי אֱלֹהִים וְיֶש לִי כֹל / כַּאֲשֶׁר עָשָׂה אוֹתְךָ עַל הַהוֹן יָכוֹל / וְלוּ הָיָה כָל אִישׁ חָכָם כָּמוֹךָ וְגָבָר / וּבַנְּדִיבוּת יָדוֹ כְיָדְךָ תִּגְבָּר / לֹא חָסַר אִישׁ בְּעוֹדוֹ דָבָר / אָמְנָם הַנְּדִיבוּת לְךָ לְמִקְנָה / יָדְךָ תִהְיֶה בוֹ בָרִאשׁוֹנָה / וְיַד כָּל הָעָם בָּאַחֲרוֹנָה / וּבְכָל אֲשֶׁר הִתְהַלַּכְתִּי מֵעִיר אֶל עִיר / לֹא מָצָאתִי אִישׁ מִשְּׁנַת הַכִּילוֹת יֵעוֹר וְיָעִיר / כִּי כֻלָּם יֹאמְרוּ שַׁלֵּם יְשַלֵּם הַמַּבְעִיר / וָאֹמְרָה בַעֲדָם אֵלֶּה בְנֵי שֵׂעִיר / וַאֲשֶׁר יִקָּרְאוּ נְדִיבֵיהֶם / וְרָאשֵׁי אַלּוּפֵיהֶם / הָפַךְ לִבָּם לִשְׂנוֹא / וְלֹא מָצָאתִי בָם רַק נוֹקֵם וְקַנּוֹא / וְכָל אֶחָד וְאֶחָד יַעֲלִים עֵינוֹ / וּכְמוֹ פֶתֶן חֵרֵש יַאְטֵם אָזְנוֹ / אִם אֶקְרָאֵהוּ בְדָבָר יוֹעִילֵהוּ קַל חִישׁ יַעֲנֶה / אִם אֶקְרָאֵהוּ בְצָרָתָה לִי עֹרֶף יִפְנֶה / וְהַזְּמָן יוֹסִיף כָּל אֲשֶׁר לוֹ לְמִשְׁנֶה / וְרָעָה אֵלָיו לֹא תְאֻנֶּה / יִדְמֶה לְכֶלֶב אִישׁ חָרַשׁ בַּרְזֵל יִישַׁן לְקוֹל פַּטִּיש וְהֹלֵם פָּעַם / וְלֹא יַרְגִּיש שַׁאֲגַת אַרְיֵה וְקוֹל רָעַם / וְיִיקַץ לְקוֹל הַשִּׁנַיִם וְיִּנָּעֵר כְּפַעַם בְּפַעַם / לֹא אֵדְעָה מָה אַחֲרִית הַזָּעַם /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

דוֹד יוֹם קְרָאתִיהוּ לְעֶזְרָתוֹ



גָּז חִישׁ כְּצִפּוֹר מִמְּרוֹם קִנּוֹ



אַךְ יוֹם לְעֶזְרָתִי קְרָאתִיהוּ



הֶחְרִישׁ כְּפֶתֶן יֶאֱטֹם אָזְנוֹ



נִדְמֶה לְכֶלֶב אִישׁ חֲרַשׁ בַּרְזֶל



בֵּית הוֹלְמִים שִׁבְתּוֹ וּמִשְׁכָּנוֹ



יִישַׁן לְקוֹל פַּטִּישׁ וְיֵרָדֵם



יִיקַץ לְעֵת אֹכֶל לְקוֹל שִׁנּוֹ.

  

וְיֵדַע אֲדֹנִי הַדָּבָר / כִּי מָצָאתִי הַנְּדִיבִים זוּלָתְךָ כַּיְעֵנִים בַּמִּדְבָּר / וּמִנְחָתָם וְנִסְכֵּיהֶם כִּמְדֻבַּר / וּלְעוֹלָם


לֹא קִבַּלְתִּי בוּשָׁה / רַק בְּאֶרֶץ מָרֵשָׁה / מִנְּדִיבִים שְׁלֹשָׁה / מְצָאתִים פֶּתַח הַשַׁעַר / וְרַעְיוֹנַי אַחֲזוּ שָׂעַר / כִּי הָיִיתִי רָעֵב וְצָמֵא / סוֹעֵר וְהוֹמֶה / וָאֶקְרַב אֲלֵיהֶם וָאֹמַר הַיּוֹם נָפְלוּ לִי בַנְּעִימִים חֲבָלִים / כִּי הִקְרָה לְפָנַי נְדִיבִים אֵל אֵלִים / וְעַתָּה רַחֲמוּ עָלַי הָאֲצִילִים / מַחֲלִימֵי הַחוֹלִים / וְגוֹאֲלֵי הַגּוֹלִים / כִּי אֵין חֵפֶץ בַּכְּסִילִים / כִּי הַפִּיל לֹא יְקִימֵהוּ כִּי אִם הַפִּילִים / וְהָעָנָק לֹא יְקִימֵהוּ כִּי אִם הַנְּפִילִים/ וְעַתָּה חוֹסֶה עַל נְדִיבִי אֲשֶׁר הִתְגַּלְגַָל / עָלָיו אוֹפָן הַגַּלְגָּל / נִטְרָד בְּטִרְדּוֹתָיו / וְלָבֵשׁ חֶרְדוֹתָיו / וְאַל תַּחְמְלוּ עַל אִישׁ יְשַנֶּה דָּתָיו / שֶׁיּוֹרֵד עַל פִּי מִדּוֹתָיו /


וָאֶשָׂא מְשָלִי וָאֹמַר:

  

חוּסוּ עַל נָדִיב הִתְגַּלְגָּל



עָלָיו גַּלְגַּל עִם טִרְדּוֹתָיו



אַךְ לֹא תַחְמֹלוּ עַל עָשִׁיר



שֶׁיּוֹרֵד עַל פִּי מִדּוֹתָיו.

  

וַיַּעַן הָאֶחָד וַיֹּאמַר אֱמֹר נָא עִקֵּשׁ / עַד מָתַי תִהְיֶה לָנּוּ לְמוֹקֵשׁ / לְנָדִיב לֹא גְמַלְתּוֹ אֵיךְ תְּבַקֵּש / מַדּוּעַ חֹק הַמּוּסָר פָּרַעְתָּ / וּמְעִיל הַבֹּשֶׁת מֵעַל פָּנֶיךָ קָרַעְתָּ / וְקָלוֹן מִכָּבוֹד שָׂבַעְתָּ / הֲלֹא יָדַעְתָּ אִם לֹא שָׁמַעְתָּ / כִּי אַדְמָתְךָ אִם לֹא תַעַבְדֶנָּה לֹא תוֹסֵף כֹּחָהּ לְךָ / וְהִנֵּה הִיא צוֹעֶקֶת לְמוּלְךָ/ אִם לֹא תַשְׁקֵנִי אֲנִי לֹא אַאֲכִילְךָ /

  

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

  

לְנָדִיב לֹא גְמַלְתּוֹ אַל תְּבַקֵּשׁ



תְּקַדֵּם לוֹ וְאַז יָשׁוּב לְגָמְלָךְ



זְכֹר אַדְמָתְךָ לָךְ אֹמְרָה קוּם



וְהַשְׁקֵנִי וְאַחַר אַאֲכִילָךְ.

  

וַיַעַן הַשֵּׁנִי וַיֹאמַר מַה לִּי וְלָךְ אִישׁ מַהֲתַלּוֹת / כִּי תְבַקֵּש לְךָ גְדוֹלוֹת / הַיּוֹם רַבּוּ עֲבָדִים הַמִּתְפָּרְצִים / אֲשֶׁר אֹמֶר הַנְּדִיבוּת לוֹחֲצִים / הֲלֹא יָדַעְתָּ הָעֲדָרִים אֲשֶׁר עָלַי רוֹבְצִים / וְהָאֵם רוֹבֶצֶת עַל הָאֶפְרוֹחִים / אוֹ עַל הַבֵּיצִים / וְהַנּוֹגְשִׂים אֲשֶׁר אָצִים וְרָצִים / כִּי לִי בָנִים שְׂלֹשָׁה וְחָמֵשׁ בָּנוֹת גְּדוֹלוֹת / וּמַשְׂאַת הַחֵשֶׁק הֵחֵלָּה לַעֲלוֹת / וּמִיִּרְאָתִי פֶּן אַחַת מֵהֶם תִּמְצָא עֲלִילוֹת / לָלֶכֶת אֶל גִּבְעַת הָעֲרָלוֹת / וּבְאַוַּת נַפְשָׁהּ תִמְשֹׁךְ יַד הַחֹשְׁקִים וְתִנְאָף / קָנִיתִי הַטַּבָּעוֹת וְנִזְמֵי הָאָף / וְהַנְּטִיפוֹת וְהַשֵּׁרוֹת וְהָרְעָלוֹת / וְכֶסֶף אֶשְׁקֹל כְּמֹהַר הַבְּתוּלוֹת / וּבְטֹרַח גָּדוֹל אוֹתָם אֲפַרְנֵּס / וְנִדְחֵי צָרְכֵיהֶם אֲכַנֵּס / לוּלֵי כִי מִן הַשָּׁמַיִם עוֹשִׂים לִי נֵס / וְאֵיךְ אֶהֱיֶה עַל בָּנַי אַכְזָרִי / וְאֶשְׁכַּח עֶצֶם מֵעֲצָמַי וּבָשָׂר מִבְּשָׂרִי / וְיִשְׂבְּעוּ זָרִים כֹּחי וְעֲצָבַי בְּבֵית נָכְרִי / וְאָמַר הֶחָכָם הָאִישׁ אֲשֶׁר יִתֵּן הוֹנוֹ לְזָרִים / יֵחָשֵׁב מִן הָאַכְזָרִים / כִּי הַקְּרוֹבִים בִּקְרוֹבִים נֶעֱזָרִים / וְדוֹמֶה בְּמַעֲשֵׂהוּ אֶל הָעַיִן הָרוֹאָה הַשְּׁחָקִים / אֲשֶׁר בְּתַכְלִית הָרֹחַק רְחוֹקִים / וְלֹא יִרְאֶה הַקְּרוֹבִים/ אֲשֶׁר סְבִיבוֹ הֵם יוֹשְׁבִים /

  

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

  

אֲדַמֶּה אִישׁ אֲשֶׁר יִתֵּן רְכוּשׁוֹ



אֱלֵי זָרִים וְיִמְנַע מִקְּרוֹבוֹ



אֱלֵי עַיִן אֲשֶׁר יִרְאֶה רְחוֹקָיו



וְלֹא יִרְאֶה אֲשֶׁר יִשְׁכֹּן סְבִיבוֹ.

  

וַיַעַן הַשְּׁלִישִׁי וַיֹּאמַר שָׁמַעְתִּי אוֹמְרִים / כִּי כָתוּב בְּאֶחָד מִן הַסְּפָרִים / אֲשֶׁר בְּאַרְבָּעָה וְעֶשְׂרִים / לֹא יָדַעְתִּי אִם בְּשִׁיר הַשִּׁירִים / אוֹ בְקֹהֶלֶת אוֹ בְמִשְׁלֵי / מִסְּמַרְטוּטֵי כַּלְמֵי וּמִמַּהֲדוּרֵי מִלֵּי / חָשַׁבְתָּ לְהָרִיק הַכֶּסֶף הַמּוּשָׁב בְּפִי אַמְתְּחוֹתֵינוּ / וְהַהוֹן אֲשֶׁר לָכַדְנוּ בְרִשְׁתֵּנוּ / וְלָקַחְנוּ מִיַּד הַזְּמָן בְּחַרְבֵּנוּ וּבְקַשְׁתֵּנוּ / בִּדְבָרִים עֲרֵבִים / וּבְקָרְאֲךָ אוֹתָנוּ אֲצִילִים וּנְדִיבִים / וְיָצָאתָ אֵלֵינוּ לְשָׂטָן / וְלָכֵן דַּע כִּי יָדֵינוּ אֲחוֹת לוֹטָן / לֹא חִנָּם שִׁנָּה הַזְּמָן דַּתֶךָ / רַק עַל פִּי מִדּוֹתֶיךָ / וְלָכֵן נָשִׂים הַיֵּאוּשׁ וְהַמְּנִיעוּת פְּרִי מִדַּבְּרוֹתֶיךָ /

  

וַיִּשָּׂא מְשָלוֹ וַיֹּאמַר:

  

אֲשֶׁר קִוָּה לְהַרְבּוֹת אֶת נְדִיבוּ-



תְךָ בִּרְצוֹת לְךָ אֵת מִשְׁאֲלוֹתָיו



קְפֹץ יָדָךָ וְשִׂים אֶת הַמְּנִיעוּת



וְהַיֵּאוּש פְּרִי מִדַּבְּרוֹתָיו.

  

כְּשָׁמְעִי כִּסְתָּה כְלִמָּה פָּנַי / וּפַלְגֵי מַיִם יָרְדוּ עֵינַי / וָאֹמְרָה בְקוֹל נֶעֱצָב / מִי הֶאֱמִין כִּי אֶתְיַצָּב / לִפְנֵי הַחֹלֶד וְהָעַכְבָּר וְהַצָּב / אֶקְרָא אֲלֵיהֶם וְלֹא יַעֲנוּנִי / וְתַחַת אַהֲבָתִי יִשְׂטְנוּנִי / וְהַזְּמָן הוֹלֵך וְסוֹעֵר וְאוֹתִי יַעֲשֶׂה כָלָה / וְרוֹאֶה אֲנִי אֶת פְּנֵי נְדִיבָיו וְהִנֵּה אֵינָם כַּאֲשֶׁר בַּתְּחִלָּה / וָאֶתְּנָה אֶת לִבִּי לִרְאוֹת סִבַּת אֵלֶּה הַתְּלָאוֹת / וְאִם בְּסִכְלוּתִי אוֹ בְרִשְׁעִי עָלַי בָּאוֹת / וְכָל אֲשֶׁר חָקַרְתִּי וְאֶחְקֹר אֶמְצָא / בְּנַפְשִׁי דֵעָה וּמוֹעֵצָה / וְעַל יְמֵי הִתְעַלְמוּתָהּ סוֹפֶדֶת / וְעַל חֵיל תַּאֲוֹתֶיהָ מוֹרֶדֶת וּבוֹגֶדֶת /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

הָהּ כִּי יְלִידֵי הַזְּמָן בִּי נִחֲרוּ



אֵינָם אֱלֵי נַפְשִׂי כְּמוֹ הָיוּ תְמוֹל



כִּי אֶחְשְׁבָה יוֹם וְלַיְלָה אֶעֱמוֹל



אוֹחִיל לְבִרְכוֹת אֵל וְהֵמָּה אֵחֲרוּ

  

אֵיכָה אֲקַוֵּמוֹ וְהֵמָּה סָחֲרוּ



אֶרֶץ אֲשֶׁר לֹא אֵדְעָה אִם הַשְּׂמֹאל



אֵלֵךְ לְיָמִין הוֹלְכוֹת הֵמָּה לְמוּל



וּתְמוֹל עֲבָדוּנִי וְפָנַי שִׁחֲרוּ

  

וְאֶתְּנָה לֵב אִם לְרֹעַ מַעֲלִי



צָרוֹת זְמָן עָלַי כְּהַיּוֹם שֹׁקְדִים



וָאֶחֱזֶה צִדְקִי וְיֹשֶׁר פָּעֳלִי

  

אָבְדַן יְמֵי שֵׂכֶל בְּכָל יוֹם סוֹפְדִים



אָרִיב בְּמוֹ שַׁחֲרוּת וְאַשְׁלִיךְ נַעֲלִי



עַל חֵיל בְּחוּרוֹתַי בְּשִׂכְלִי מוֹרְדִים.

  

וַיְהִי כִשְׁמֹעַ הָאֲנָשִׁים אֶת דְּבָרַי / לֹא חָרְדוּ אֶל מַאֲמָרַי / וְלֹא נִכְנְעוּ לִבּוֹתָם / הֵרֵעוּ מֵאֲבוֹתָם / וַיֹּאמְרוּ לִי אִי הַנִּכְאָב אֲשֶׁר רֶשַע חָרַשְׁתָּ / וְלָכֵן מַעֲטֵה הָעֳנִי לָבַשְׁתָּ / אֵיךְ פְּנֵי אֲבוֹתֶיךָ הוֹבַשְׁתָּ / וּמַדּוּע אֹהָלֶיךָ נָטַשְׁתָּ / הַאִם אָכַלְתָּ כָּל אֲשֶׁר יָרָשְׁתָּ /



וָאַעַן וָאֹמַר נָטַשְׁתִּי אָהֳלִי כִּי כֵן צֻוֵּיתִי / וְנַפְשִׁי מַמְּרוּרִים רִוֵּיתִי / וְקַצְתִּי בַזְּמָן וּבוֹ נִבְאַשְׁתִּי / כִּי מִמְּצֹא חֶפְצִי בוֹ נוֹאַשְׁתִּי / וְלָכֵן בְּבֹשֶׁת פָּנִים נִגַּשְׁתִּי / אֶל הַמַּחֲנֶה הַזֶּה אֲשֶׁר פָּגַשְׁתִּי / וְלֹא אֶשְׂנָא הָעֲנִיּוּת / רַק בַּעֲבוּר הֱיוֹתִי נִכְנָע אֶל הַפְּתַיּוּת / וְצָעַקְתִּי וְלֹא נַעֲנֵיתִי / וְאוּלַי כִּי לְטוֹב לִִי עֻנֵּתִי /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

הַשֹּׁאֲלִים אֵיךְ אָהֳלִי נָטַשְׁתִּי



לֹא תֵדְעוּ כִּי כֵן זְמָן צִוַּנִי



הוּא מַמְּרוֹר עִם לַעֲנָה הִרְוַנִי



עַד מִמְּצוֹא חֶפְצִי אֱמֶת נוֹאַשְׁתִּי

  

קַצְתִּי בְמַחְמַדָּיו וּבוֹ נִבְאַשְׁתִּי



מָה אֹמְרָה עוֹד הַזְּמָן הִלְוַנִי



יָמִים יְמֵי עֶצֶב עֲדֵי הִשְׁוַנִי



אֶל מַחֲנוֹת הָאֵל אֲשֶׁר פָּגַשְׁתִּי

  

אֶשְׂנָא זְמָן תַּחַת אֲשֶׁר קִנֵּאתִי



כִּי אֶחֱזֶה נַפְשִׁי אֲשֶׁר כּוֹרַעַת



אֶל הַר פְּתַיּוּת עַד אֲשֶׁר נַעְנֵיתִי

  

יָדַע זְמָן כִּי אָהֳלָיו שָׂנֵּאתִי



עַל כֵּן לְקִיקְיוֹנִי יְמָן תּוֹלַעַת



אוּלַי יְדִידַי כִּי לְטוֹב עֻנֵּיתִי.



וַיְהִי כִרְאוֹת הָאֲנָשִׁים חֶסְרוֹנִי חָשְׁבוּ מִלָּתִי לְאֵל /



וַיֹּאמֶר אֶחָד מֵהֶם נֵדְעָה מַה בְּפִיו יִשְׁאַל /



וַיֹּאמְרוּ רֵעָיו מַה לָּנוּ וְלַצָּרָה אִם תִּגְאַל גְּאַל /



וַיֹּאמֶר הָאִישׁ וּלְשׁוֹנוֹ כָּאֵשׁ תִּהֲלַךְ / שְׁאַל מָה אֶתֵּן לָךְ /



וָאֹמְרָה בְּנֶפֶשׁ שְׁפָלָה / אַחַר בַּקָּשַׁת הַמְּחִילָה / יָדֶיךָ הַשָּׂר לְמוּדוֹת בְּחֶמְלָה / בְּכֶפֶל מִלְּשׁוֹנִי בִשְׁאֵלָה /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

נְדִיבֵי עָם הֲטוֹב כִּי תֹאמְרוּ לִי



שְׁאַל מֶנּוּ וְנָפִיק לָךְ בְחֶמְלָה



יְדֵיכֶם הֵם לְמוּדוֹת בַּנְּתִינָה



בְּכֶפֶל מִלְּשׁוֹנִי בַשְּׁאֵלָה.

  

וַיֹּאמֶר הָאִישׁ בֹּאָה אִתִּי הַבַּיִת וּסְעָד /



וָאֵלֵךְ אַחֲרָיו אֶל בֵּיתוֹ וְלִבִּי בְצָרַת רַעֲבוֹנִי יִרְעָד /



וַיֹּאמֶר הָאִישׁ הֲתֹאכַל מִן הַיְּרָקוֹת הַחֲשׁוּבִים /



וָאַעַן וָאֹמַר אִלּוּ הַיְּרָקוֹת טוֹבִים אֲכָלוּם הַכְּלָבִים /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

שְׁאֵלוּנִי בְמַה תִּסְעַד לְבָבָךְ



הֲתִבְחַר בַּיְּרָקוֹת הַחֲשׁוּבִים



עֲנִיתִים הֵן יְהִי טוֹב בַּיְּרָקוֹת



כְּלָל עִקָּר אֲכָלוּם הַכְּלָבִים.

  

וַיֹּאמֶר הָאִישׁ הִנְנִי נִשְׁבָּע בַּאֲשֶׁר אֶל יֵשׁ הוֹצִיאַנִי מֵאַיִן / כִּי בְּבֵיתִי לֹא תֹאכַל בָּשָׂר וְלֹא תִשְׁתֶּה יַיִן /



אָז שָׂתַקְתִּי וְהֶעֱלַמְתִּי עַיִן / וְאָמַרְתִּי בְלִבִּי זֶה נִשְׁאַר מִבְּנֵי קַיִן / וַיּוּשָׂם לְפָנַי מִן הַיְּרָקוֹת קְעָרָה / וְגַרְגְּרֵי זֵיתִים עֲשָׂרָה / וְצַפַּחַת מִמֵּי הַמְּעָרָה / בְּמִדָּה וּבְמִשְׁקָל וּבִמְשׂוּרָה / וּפְתוֹתֵי לֶחֶם שְׂעוֹרִים / וַיִּהְיוּ פִי וְיָדַי לֶאֱכֹל נִמְהָרִים / וַיָּבֹאוּ בִי הַמַּיִם הַמְּאָרְרִים לְמָרִים / וְהָיוּ סְבִיבַי שְׁלֹשָׁה נְעָרִים/ יִקְחוּ מִלְּפָנַי הַמְּעַט אֲשֶׁר בַּעֲדִי הֵכִינוּ / וַיָּשׁוּבוּ וְכָל הַמִּשְׁפָּחָה עָלַי יַלִּינוּ / וַיֹּאמְרוּ לָקַח זֶה הָרְעַבְתָּן אֶת כָּל אֲשֶׁר לְאָבִינוּ / עוֹדָם מְדַבְּרִים וְהִנֵּה אִשְׁתּוֹ כָאֵשׁ יַעַר תִּבְעַר / תֶּהֱמֶה וְתִסְעַר / וּבְבַעְלָהּ תָּרִיב וְתִגְעַר / וָתֹאמַר זֶה הַהֵלֶך בְּעֵינַי לֹא יוּחָן / כִּי יֹאכַל כְּאִלּוּ יִטְחָן / לֹא יָקוּם לְעוֹלָם מִן הְשֻּלְחָן /



וַיֹּאמֶר הָאִישׁ חַי אֲנִי אִם לֹא יַנִּיחַ הָאֲכִילָה זֶה הַסָּכָל / לֹא יִפָּרֵד מִמֶּנִי עַד יָקִיא אֶת אֲשֶׁר אָכָל /



וַיְּהִי כְּשָׁמְעִי קַמְתִּי קַל חִישׁ וּפַת לֶחֶם בְּיָדִי /



וַיֹּאמֶר הָאִישׁ הִנֵּה בַשָּׁמַיִם עֵדִי / כִּי אִם לֹא תַנִּיחַ הַלֶּחֶם אֲשַׁסֵּעֲךָ כְּשַׁסַּע הַגְּדִי /



וָאַשְׁלִיךְ הַלֶּחֶם מִיָּדִי / וָאֶבְרַח /


וְהָאִישׁ אַחֲרַי מַר יִצְרַח / וְהַנְּעָרִים יֶהְדְּפוּנִי / וְהַכְּלָבִים יִרְדְּפוּנִי / וְהַצָּרוֹת מִכָּל עֲבָרַי הִקִּיפוּנִי / וְהַמַּיִם עַד נֶפֶשׁ אֲפָפוּנִי / וְעַד פֶּתַח שַׁעַר הָעִיר רְדָפוּנִי / וְלוּ הִשִּׂיגוּנִי טְרָפוּנִי / וּרְאֹה פָנֶיךָ וּפְנֵי אִישׁ עוֹד לֹא פִלָּלְתִּי / וְעַתָּה רְאֵה הַשָּׂר אֵיךְ נָפָלְתִּי /


וַיֹּאמֶר הַשָּׂר בֶּאֱמֶת הָיְתָה זֹאת חֶרְפָּה עֲצוּמָה וּכְלִמָּה / וְכִמְעַט שֶׁאֵין אַחֲרֶיהָ נֶחָמָה / וְעַתָּה הֲלֹא עָשִׂיתָ בָּאֲנָשִׁים הָהֵם נְקָמָה /



וָאַעַן וָאֹמַר נַפְתוּלֵי אֱלֹהִים נִפְתַּלְתִּי / וְקַרְנָם בֶּעָפָר עוֹלַלְתִּי / וְהִשְׁבַּתִּי מִטָּהֳרָם וְנִזְרָם לָאָרֶץ חִלַּלְתִּי / וּבְעַמִּים חֶרְפָּתָם הוֹדַעְתִּי / הִגַּדְתִּי וְהוֹדַעְתִּי וְהִשְׁמַעְתִּי / כִּי לֹא דַי הֱיוֹתָם אוֹתִי בוֹזִים / רַק שֶׁהִקְהִילוּ עָלַי אֲנָשִׁים מְחַבְּרֵי חֲרוּזִים / וַיִּתְקַבְּצוּ עֲלֵיהֶם אֲנָשִׁים רֵקִים וּפוֹחֲזִים / מִן הַיְּהוּדִים הַפְּרָזִים / אֲשֶׁר לֹא עָמְדוּ אֲבוֹתֵיהֶם עַל הַר גְּרִיזִים / הִתְפָּאֲרוּ עָלַי בְּשִׁירֵיהֶם / וְהַדִּבֵּר אֵין בָּהֶם / יִשְׁלְחוּ לִי שִׁירִים מְטֻלָּאִים / סְבֻכִים וּכְסָבְאָם סְבוּאִים / יְסַפְּרוּ בָהֶם כִּי לָהֶם בְּכָל הַתְּעוּדוֹת / בְּכֶפֶל מִמֶּנִּי עֶשֶׂר יָדוֹת / וְהָיוּ בָהֶם אֲנָשִׁים יוּלְדוּ עַל בִּרְכַּי וּגְמוּלֵי חֲסָדַי / לָמְדוּ מִמֶּנִּי דֶרֶךְ הַשִּׂיר וְהֵם מַעֲשֶׂה יָדַי / וְלֹא הִשִּׂיגוּ בִמְרוּצָתָם אֲבַק שִׂירַי / וְאֵין אֶחָד מֵהֵם יְזַנֵּב הַנֶּחְשָׁלִים אַחֲרַי / וּבִרְאוֹתִי כִּי יָנִיפוּ עָלַי יַד שִׁלְחָם / בִּשְׁאָט בְּנֶפֶשׁ בִּגְאוֹן רוּחָם / בָּנִים גִּדַּלְתִּי וְרוֹמַמְתִּי וְנָשָׂאתִי טָרְחָם / וְקָרַבְתִּי אֲלֵיהֶם בְּבָרְחָם / וּבְעֵת קָדַר עֲלֵיהֶם הַיּוֹם הָיִיתִי שִׂמְשָׂם וְיַרְחָם / וְלֹא יִזְכְּרוּ מִי שָׂם צִיץ נֵזֶר הַשִּׁיר עַל מִצְחָם / אָמַרְתִּי חַי אָנִי אַבְאִישׁ וְאַבִּיעַ שֶׁמֶן רָקְחָם / וְתָשׁוּב לְשׁוֹנָם מֻדְבָּק מַלְקוֹחָם / וְעַל כָּרְחָם אֶהְיֶה מַלְכָּם וּמְשִׁיחָם / וָאָקוּם וָאֲחַבֵּר עֲלֵיהֶם שִׁיר נִמְרָץ / בְּפִי חָזוֹן נִפְרָץ / הוֹדַעְתִּי בָעַמִּים עֻזָּם / וְכִי הֵם יֶתֶר הַגָּזָם / וְהַנִיפוֹתִי עֲלֵיהֶם שִׁיר חַרְבִּי / כַּאֲשֶׁר הָיָה עִם לְבָבִי / וְהִנֵּה הַשִּׁיר אֲשֶׁר עֲלֵיהֶם חִבַּרְתִּי / כַּאֲשֶׁר הִתְעַבַּרְתִּי / וְלוּחוֹת בְּרִיתִי אִתָּם שִׁבַּרְתִּי / וְהוֹדַעְתִּי אוֹתָם כִּי עֲלֵיהֶם גָּבַרְתִּי / וְהַעַלְתִּי עַד לַשָּׁמַיִם שִׂיאִי וְאוֹתָם בְּעִמְקֵי שְׁאוֹלָה קָבָרְתִּי /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

אֶשְׁמַע דִּבַּת רַבִּים רָבִים



עָלַי יָנִיפוּ יַד שִׁלְחָם



בִּי יָשִׂיחוּ לִי יָפִיחוּ



בִּשְׁאָט נֶפֶשׁ בִּגְאוֹן רוּחָם



בָּנִים גִּדַּלְתִּי רוֹמַמְתִּי



זֶה כַמָּה נָשָׂאתִי טָרְחָם



וּכְנֶשֶׁר יָעִיר אֶת קִנּוֹ



כֵּן רָחַפְתִּי עַל אֶפְרוֹחָם



זֶה מַשְׂכֻּרְתִּי חֵלֶף כִּי כֵן



קָרַבְתִּי לָהֶם יוֹם בָּרְחָם



וּבְיוֹם קָדְרוּ כּוֹכְבֵי שִׂכְלָם



הָיִיתִי שִׁמְשָׁם וִירֵחָם



וּבְעֵת תָּעוּ בִנְתִיבוֹת בִּין



שִׂכְלִי לִמְחוֹז חֶפְצָם הִנְחָם



לֹא יָדַעְתִּי מִי יוֹלַדְתָּם



אִם הַקָּאָת אוֹ הָרָחָם



לֹא יַכִּירוּ לֹא יִזְכּוֹרוּ



חַסְדִּי עִמָּם יוֹם נָס לֵחָם



אֵין בָּם יָצִיץ מִי זֶה שָׂם צִיץ



נֵזֶר הַקֹּדֶש עַל מִצְחָם



מִי רוּחַ שֵׂכֶל בָּם מָסַךְ



אוֹ מִי נָסַךְ אֶת יֵין רִקְחָם



אֵיכָה בִי יָרִיבוּ וּזְמָן



מָשַׁח אוֹתִי לִהְיוֹת פֶּחָם



לִי חָלַק גֻּלּוֹת עִלִּיּוֹת



מִן שָׁמַיִם יוֹם יוֹם אֶלְחַם



אַךְ לָהֶם גֻּלּוֹת תַּחְתִּיּוֹת



בִּרְכוֹת שָׁדַיִם וָרָחַם



וּזְמַנִּי יוֹם שָׂם חִטָּתִי



שׁוֹרָה אָז הֵפִיץ אֶת קִצְחָם



וּבְעֵת קִטַּרְתִּי אֶת מוֹרִי



הִבְאִישׁ הִבִּיעַ אֶת רֵיחָם



וּבְיָדִי מַטֵּה הָאוֹתוֹת



אֵיכָה אֶחְרַד לִבְרַק רִמְחָם



אִם מֵימֵי חָכְמוֹתָם כַּיָּם



לִבִּי הוּא נַחַשׁ וּבְרִיחָם



בִּנְשׂוֹא גַלֵּי מֵי יַמֵּיהֶם



אָז מַטֵּה שִׂכְלִי הִשְׁבִּיחָם



וּבְיוֹם הִצִּיבוּ חֵילֵיהֶם



מַצַּב חַיל שִׂכְלִי הִבְרִיחָם



עָלִיתִי אֶל עִיר גִבּוֹרָם



אַף הוֹרַדְתִּי עֹז בָּרִיחָם



אָנֹכִי הִשְׁוֵיתִי כַיוֹם



אֶת שָׂרֵיהֶם אֶל פִּרְחָחָם



הֶחָרָש הַמַּסְגֵּר בַּגֵּר



בַּתּוֹשָׁבָם וּבְאֶזְרָחָם



חָנִיתִי כַדּוּר עַל חֵילָם



יָדַעְתִּי נוּעַם גַּם נוּחָם



הָיָה כַיּוֹם נִצְחִי נֵצַח



אַךְ חֵילֵיהֶם אָבַד נִצְחָם



אֶזְעַק חָמָס אֶל הַיָּמִים



וּלְחֹק גַּלְגַּל כִּי הִשְׁכִּיחָם



מַדּוּעַ דַּרְכָּם צָלֵחָה



אוֹ מָה רָאָה כִּי הִרְוִיחָם



אִלּוּ הִכִּיר מַעְבָּדֵיהֶם



אוֹ אִלּוּ יָדַע אֶת שִׂיחָם



לֹא הִשְׁכִּיבָם עַל מִטּוֹת שֵׁן



וּסְרוּחִים לֹא אָסַף נִדְחָם



לֹא שָׁעָה אֶל מִנְחַת הֶבְלָם



אַף לֹא רָצָה כַיּוֹם זִבְחָם



לֹא נָתַן בַּר הַלֶּחִי אוֹ



מָמוֹן נִכְנָס בָּא עַל נִכְחָם



עַל כֵּן הֶעְלֶה עַל גוֹרָלָם



מָמוֹן יוֹצֵא אוֹ הַנִּלְחָם



עַל זֹאת לִבִּי הוּא יָחִיל בִּי



וּבְתוֹך קִרְבִּי יוֹם יוֹם יֵחָם



כִּי הִסְכַּנְתִּי כָּל אָרְחוֹתָם



זֵרֵיתִי רִבְעָם גַּם אָרְחָם



אֵיךְ מוּל פִּרְיִי פָּרַח עַל שָׁרְ-



שִׁי הַבִּין יַשְׁווּ אֶת קִלְחָם



אִם יִרְאוּ חִדֶּקֶל שִׂכְלִי



אָז יִמְאָסוּ מֵי שִׁלֹחָם



לוּ יָרִיחוּ קִטֵּר בָּשְׂמִי



אָז יִבְזוּ רֵיחַ נִיחוֹחָם



יָשִׂימוּ יָד אֶל פֶּה שָׂרִים



וּלְשׁוֹנָם מֻדְבָּק מַלְקוֹחָם



אַקְדִּיר שִׁמְשָׁם אֶתֵּן שָׁרְשָׁם



כַּמָּק וּכְאָבָק דַּק פִּרְחָם



הִנְנִי לַדּוֹר פִּרְצָם אֶגְדֹּר



אֶהְיֶה לָהֶם רֹאש וּמְשִׁיחָם



יֵלְכוּ גוֹיִם אֶל אוֹר


שִׁירַי


וּמְלָכִים אֶל נֹגַהּ זָרְחָם



עֶשֶׂר יָדוֹת לִי בַחָכְמוֹת



מִי זֶה אִתִּי יִהְיֶה נִלְחָם



אֶקְרָא חֹם קֹר גַּם לַקֹּר אֶקְ-



רָא חֹם וּלְהַבָּרָד פֶּחָם



לִי חָכְמָה לִי בִינָה אֶקְרָא



בָּהּ עַל כָּרְחָם בַּעַל קֹר חָם



לוּלֵי חָסְדִּי אָז הָפַכְתִּי



אֶל מַעְרָבָם אֶת מִזְרָחָם



אֵיכָה יִבְזוּ עֶרְכִּי עַם זוּ



וַיֹּאמְרוּ כִי מִמֵּי לִקְחָם



אֶשְׁאַב שֵׂכֶל אַבְקַת רוֹכֵל



אָשִּׂיג נַחַש לִי בַּרִיחָם



אֵיךְ תַּחַת לֶחֶם אַבִּירִים



וּדְגַן רוּם אֶקְטֹף מַלּוּחָם



אוֹ תַחַת עֶדְיֵי הַשֵּׂכֶל



אֶחְמֹד עֶדְיָם אוֹ מַלְקוֹחָם



אֵיכָה יָרִימוּ קֶרֶן אוֹ



עָלַי יָנִיפוּ יַד שִׁלְחָם.

  

וַיֹּאמֶר הַשָּׂר בֶּאֱמֶת פָּרַעְתָּ בָּם פְּרָעוֹת / וְתַחַת אֲשֶׁר הִשְׁקוּךָ בְּשָׁלִישׁ דְּמָעוֹת / שָׂבְעָה נָפְשָׁם בְּרָעוֹת / וּבְאֱמֶת הָיוּ כְסִילִים / וּבְתַכְלִית הַנַּבְלוּת נְבָלִים / כִּי הַנְּדִיבִים יִתְנַדְּבוּ בַגּוּפִים וּבַנְּשָׁמוֹת / וְאַף אִם יִהְיוּ חַסְרֵי כָל נְעִימוֹת / לְדִמְיוֹן הַחֶרֶב הַחוֹתֶכֶת הָעֲצָמוֹת / וְאַף אִם תִּהְיֶה בַעֲלַת פְּגִימוֹת /

  

וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר:

  

נְדִיבֵי עָם הֲכִי יִתְנַדְּבוּ לָךְ



וְאַף כִּי הֵם חֲסְרֵי כָל נְעִימוֹת



כְּמוֹ חֶרֶב בְּצֵאתָהּ מִתְּעָרָהּ



אֲשֶׁר תַּחְתֹּךְ וְיֵש בָּהּ רֹב פְּנִימוֹת.

  

וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר הָיָה בָהֶם אִישׁ עָשִׁיר לָרֹב / כִּי יָרַשׁ נַחֲלָה מִקָּרוֹב / וְהוּא אָמַר וַיָּבֹא עָרֹב / וְהָיָה אוֹהֲבִי בַּזְּמָן שֶׁעָבַר / יוֹתֵר מִרֹחַק נְהַר כְּבָר /



וַיֹּאמֶר הַשָּׂר הֲלֹא שָׁמַעְתָּ מְשַל הַקַּדְמוֹנִי / הַכָּתוּב בְּסֵפֶר תַּחְכְּמוֹנִי / שֶׁאֹמַר דַּי לְךָ מֵעֲמִיתְךָ הַנֶּאֱמָן / כַּאֲשֶׁר יַנְחִילֵהוּ הַזְּמָן / יְרֻשַּׁת עֹשֶׁר וּמַעֲלָה רָמָה / אִם יַרְאֶה לְךָ עֲשִׂירִית אַהֲבָתוֹ הַקְּדוּמָה /

  

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

  

יְהִי דַי לָךְ יְדִידִי מֵעֲמִיתָךְ



בְּיָרְשׁוֹ מַעֲלָה נִשֵּׂאת וְרָמָה



בְּיוֹם תִּתּוֹ וְהַרְאוֹתוֹ לְךָ אֵת



עֲשִׂירִית אַהֲבָתוֹ הַקְּדוּמָה.

  

וָאַעַן וָאֹמַר חַי אָנִי כִּי לוּלֵי חֲסָדַי אֲשֶׁר קִדְּמוּם / לֹא חָשַׁבְתִּי עֲלֵיהֶם דְּבַר מוּם / אָמְנָם אֲנִי מֵעוֹלָם בְּהוֹנִי פְדִיתִים / וּבְעֵת צָרָתָם עֲנִיתִים / וְהֵם לֹא דַי שֶׁלֹּא הָיוּ בְצָרָתִי קְרוֹבִים / אֶלָּא שֶׁדִּבְּרוּ עָלַי כְּזָבִים / וְלָכֵן צַר לִי עַל הַחֲסָדִים אֲשֶׁר גְּמַלְתִּים / כִּי נִחַמְתִּי כִּי עֲשִׂיתִים /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

בְּנֵי נָתָן בְּנֵי נָבָל בְּנֵי שֵׁם



קְדַשְׁתִּימוֹ חֲשַׁבְתִּימוֹ כְזָבִים



אֲנִי תָמִיד פְּדִיתִימוֹ וָאֶפְדֵּם



וְהֵמָּה דִּבְּרוּ עָלַי כְזָבִים.

  

וְדַע הַשָּׂר אֲדֹנִי / צִיץ מִצְחִי וְנֵזֶר זְמַנִּי / כִּי כַאֲשֶׁר הֱבִיאַנִי / לַחֲשֹׁק בַּתְּבוּנָה / וְלִהְיוֹת יָדִי בַחָכְמוֹת אֱמוּנָה / הוּא רְאוֹתִי הֱיוֹתִי נִמְאָס בְּעֵינֵי הַסְּכָלִים / וְנִגַּשׂ וְנַעֲנֶה אֶל הַשְּׁפָלִים / וְכָל אֲשֶׁר יַחְשְׁבוּנִי לְנִבְזֶה שׂוֹטְנַי/ תִּגְדַּל מַעֲלָתִי בְעֵינַי / וּבָזֶּה נוֹדָע לִי חִין עֶרְכִּי / כִּי לוּ יַחְפְּצוּ הַכְּסִילִים דַּרְכִּי / אָז יִוָּדַע לִי כִּי שֶׁקֶר נִסְכִּי /

  

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

  

אֲשֶׁר גָּרַם לְנַפְשִׁי לַחֲשֹׁק אֶת



תְּבוּנָתִי וְרָדְפִי טוּב פְּעָלִים



רְאוֹתִי כִּי אֲנִי נִמְאָס לְכָל בָּז



וְנִגַּשׂ נַעֲנֶה אֶל הַשְּׁפָלִים.

  

וַאֲשֶׁר גָּרַם אֶל הַסְּכָלִים הָהֵם / לִהְיוֹת מַעֲלָתִי פְּחוּתָה בְעֵינֵיהֶם / הוּא רְאוֹתָם צְרוֹרוֹת כַּסְפָּם בְּאַמְתְּחוֹתֵיהֶם / וַיִּרְאוּ אַמְתַּחְתִּי רֵיקָה / בּוּקָה וּמְבֻלָקָה / וְנַפְשִׁי שׁוֹקֵקָה / וְאָמַר הֶחָכָם חֶסְרוֹן הַהוֹן יַעֲלִים הַיֹּשֶׁר / וְכֵן הַסִּכְלוּת יַעֲלִימֵהוּ הָעֹשֶׁר /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

אָמַר הֶחָכָם יֵשׁ חֶסְרוֹן



עֹשֶׁר יַעְלִים רֹב הַיֹּשֶׁר



כֵּן יֵשׁ סִכְלוּת עִם הוֹלֵלוּת



יַעְלִים אוֹתָם רֹב הָעֹשֶׁר.

  

וְשִבְּחוּ הַכְּסִילִים כִּי לֹא קָנוּ הָעֹשֶׁר בְּשִׂכְלָם / רַק גַּלְגַּל הוּא שֶׁחוֹזֵר בָּעוֹלָם / וְאִם יֶחֱרַץ חָרוּץ / וְלִקְנוֹת הוֹן יֶחֱרַד וְיָרוּץ / לֹא יִצְלַח בְּמַעֲשֵׂהוּ / וְדֶרֶך תְּבוּנוֹת לֹא תוֹעִילֵהוּ / וְיַגִּיעַ הַהוֹן בִּנְקַלָּה לְיַד אִישׁ שׁוֹבָב / בְּשִׁגָּעוֹן וּבְעִוָּרוֹן וּבְתִמְהוֹן לֵבָב /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

מַה


תֶּחֱרַץ חָרוּץ וְאֶרְאֶה הַזְּמָן


יָאֵר פְּנֵי חַסְדּוֹ לַגֵּר וּלְעֵקֶר



אִם הַגְּזֵרָה יוֹרְדָה מִמַּעַל



אִם כֵּן חֲרִיצוּת הָאֲנָשִׁים שֶׁקֶר.

  

וְדִמּוּ הַחֲכָמִים בְּשִׂכְלָם / הָעוֹלָם הַזֶּה לְסֻלָּם / זֶה יוֹרֵד בּוֹ וְזֶה עוֹלֶה / זֶה שׁוֹקֵט וְזֶה גוֹלֶה / זֶה נִכְבָּד וְזֶה נִקְלֶה / וּמִקְרָא בְּצַד וּבְכָתֵף יֵהְדֹּפוּ / שָׁנָה עַל שָׁנָה חַגִּים יִנְקֹפוּ / וּמַחֲסֵה כָזָב וְסֵתֶר מַיִם יִשְׁטֹפוּ / וְאִישָׁיו לְמַאֲמָרוֹ כְסוּפָה יִרְדֹּפוּ / כַּלְּבוּש יַחֲלִיפֵם וְיַחֲלֹפוּ /

  

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

  

רְאֵה מִקְרֵי זְמָן בּוֹגֵד



אֱנוֹשׁ אָנוּשׁ וְהִפָּלֵא



וְדָבָר מִמְּךָ נֶעְלָם



אֲנִי אֶת אָזְנְךָ גוֹלֶה



פְּרָטָיו לֹא אֲבָאֵר כִּי



זְמַנִּי מִסְּפֹר יִכְלֶה



זְמָן יָרִיק לְךָ מָצוֹק



וְהַטּוֹב מִמְּךָ יִכְלֶה



וְאִישָׁיו חֹלְפִים יוֹם יוֹם



וְאָדָם כַּלְּבוּש יִבְלֶה



וּמִי יוֹרִיש וּמִי יַעְשִׁיר



וּמִי יֵרֵד וּמִי יַעְלֶה



וּמִי מֶלֶךְ וּמִי הֵלֶךְ



וּמִי נִכְבָּד וּמִי נִקְלֶה



וּמִי עֶלְיוֹן וּמִי אֶבְיוֹן



וּמִי שׁוֹקֵט וּמִי גוֹלֶה



וּמִי נִטְרָף וּמִי נִשְׂרָף



וּמִי נִסְקָל וּמִי נִתְלֶה



מְתֵי מִסְפָּר כְּתַבְתִּים לָךְ



וְעוֹד נוֹסָף עֲלֵי אֵלֶּה.

  

וְאָמַר שְׁלֹמֹה בְחָכְמָתוֹ פַּעַם וּפַעֲמַיִם / לַכֹּל זְמָן וְעֵת לְכָל חֵפֶץ תַּחַת הַשָׁמָיִם / וְלָכֵן הֵיָּרֵא אֶת דְּבַר אֲדֹנָי מֵעֲבָדָיו / יְזַכֶּה אָרְחוֹ וְיַיְשִׁיר מַעְבָּדָיו / וְעַל דְּבַר אֱמֶת וְעַנְוָה צֶדֶק יִשְׁקֹד / וּמַעֲשָׂיו יִדְרֹש וְיִפְקֹד / כִּי הַזְּמָן יִתֵּן פְּנֵי הַצֹּאן אֶל עָקֹד / וְתַחַת כְּבוֹדוֹ יֵקַד יְקֹד / וְטָרֵף זְרוֹעַ אַף קָדְקֹד / כִּי עֵת סְפוֹד וְעֵת רְקוֹד /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

רְאֵה צְבִי זְמָן וְעֵת



וְעַל דְּבַר אֱמֶת שְׁקֹד



זְמָן וְעֵת לְכָל יְצוּר



וְעֵת סְפוֹד וְעֵת רְקֹד.

  

וַיְּהִי כְּכַלּוֹתִי לְדַבֵּר / וְשִׁיר עְלִיצֻתִי לְחַבֵּר / וַיֹּאמֶר הַשָּׂר הוֹדַעְתַּנִי אֵיךְ רַבַּת שָׂבְעָה לָהּ נַפְשְׁךָ יְגוֹנִים / וְלִשְׁמֹעַ לַעַג הַשַּׁאֲנַנִּים / וְהַבּוּז לִגְאֵיוֹנִים אֲשֶׁר לֹא נָשְׂאוּ לְמַעֲלָתְךָ פָנִים / וְעַתָּה הוֹדִיעֵנִי אִם מָצָאתָ נָדִיב בַּדּוֹר / אֲשֶׁר פֶּרֶץ הַזְּמָן יִגְדֹּר /



וָאַעַן וָאֹמַר פָּגַשְׁתִּי שָׂשׂ עוֹשֵׂה צֶדֶק / אִישׁ נָדִיב מְחַזֵּק הַבֶּדֶק / תָּמִים בְּדוֹרוֹתָיו / יַעֲרֹפוּ כַמָּטָר נִדְבוֹת מִטְרוֹתָיו / וְהִזִּיל עָלַי מִטַּל נִדְבוֹתָיו / וְאָכַלְתִּי בְּבֵיתוֹ דָּגִים חֲשׁוּבִים / וּגְדַיֵּי עִזִּים טוֹבִים / וְחִבַּרְתִּי עֲלֵיהֶם שִׁירִים עֲרֵבִים /



וַיֹּאמֶר הַשָּׂר אֲדֹנִי / אֵיךְ הָיָה הַדָּבָר בְּנִי /



וָאַעַן וָאֹמַר הָיָה בְּעִיר אַנְקוֹנָ"א אָדָם נִכְבָּד / וּלְהִתְנַדֵּב עָלָיו לֹא יִכְבָּד / מֶלֶךְ לִנְדִיבוּת נֶעֱבָד / וְחָשַׁבְתִּי כִּי כָל הַנְּדִיבוּת בְּמוֹתוֹ יֹאבַד / וַיְהִי הַיּוֹם וְהוּא מְצָאַנִי בְּאֵשׁ הַיְּגוֹנִים בּוֹעֵר / וּבְכָל מַחֲמַדֵּי הַזְּמָן גּוֹעֵר / כִּי יָם הַתְּלָאוֹת הוֹלֵך וְסוֹעֵר /



וַיֹּאמֶר אֵלַי הָאִישׁ מָה אַתָּה רוֹאֶה / כִּי עָמַדְתָּ הַיּוֹם מִשְׁתָּאֶה /



וָאַעַן וָאֹמַר חַיֶּיךָ אֲנִי מַרְעִיד עָמַדְתִּי / כִּי הַיּוֹם לֹא סָעַדְתִּי / וּבָכִיתִי וְסָפַדְתִּי / כִּי רָאִיתִי דָגִים גְּדוֹלִים וְלָשֶׁבֶת בְּצִלָּם חִמַּדְתִּי / וְאִם לֹא אֹכַל מֵהֶם הֵן גָּוַעְתִּי אָבַדְתִּי /



וַיֹּאֶמר הָאִישׁ בְּחַיַי אִם תִּדֹּר לְחַבֵּר עֲלֵיהֶם שִׁירִים עַלִּיזִים / יִהְיוּ לְךָ הַדָּגִים וּשְׁנֵי גְדָיֵי עִזִּים / אָז אָמַרְתִּי עֵת לַעֲשׂוֹת חֲרוּזִים / אָז הָלַכְתִּי עִם הָאִישׁ אֶל חֲצַר גִּנַּת הַבִּיתָן / וְהֶרְאָנִי דָג גָּדוֹל רָאוּי לְשָׁאוּל וִיהוֹנָתָן / וּבִרְאוֹתִי אוֹתוֹ חָשַׁבְתִּי הֱיוֹתוֹ לִוְיָתָן / וָאֹמַר אֶל לִבִּי אֶחְשֹׁב הֱיוֹתִי הַיּוֹם חָתָן /



וָאֹמַר אֶל הַנִּכְבָּד אִם תַּעֲשֶׂה לִי מַטְעַמִּים עֲרֵבִים / אֲחַבֵּר הַיּוֹם שִׁירִים מִצּוּר הַשֵׂכֶל חֲצוּבִים /



וַיֹאמֶר הָאִישׁ הִכּוֹן וְהָכֵן לְךָ הַשִּׁירִים / כִּי הַמַּטְעַמִּים יִהְיוּ חֲמוּדִים וִיקָרִים / וַיְהִי כַאֲשֶׁר תַּגִּיעַ עֵת הַסְּעֻדָּה / יָשַׁבְנוּ אֲנִי וְהַנִּכְבָּד שַׂר הַתְּעוּדָה /



וַיֹּאמֶר הַשָּׂר יָבִיאוּ הַטְּלָאִים הַנְּעִימִים / וְהַגִּישׁוּ הַמַּטְעַמִּים / וְנַחֲלִיץ בָּהֶם הָעֲצָמִים כִּי הַטָּלֶה רָאוּי לִמְרוֹם הַתְּהִלּוֹת / וְלָכֵן שָׂם אוֹתוֹ הָאֵל רִאשׁוֹן לַמַּזָּלוֹת /



וָאַעַן וָאֹמַר חָיֵיךָ אֲדֹנִי דָלְפָה נַפְשִׁי מִתּוּגָה / וְאֵין רְפוּאָה כַדָּג לַדְּאָגָה / כִּי לָשׁוֹן נוֹפֵל עַל הַלָּשׁוֹן / וְסָכְלוּ הָאוֹמְרִים טָלֶה רִאשׁוֹן /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

שִׂים דוֹד לְעֻמַּת הַדְאָג דָּג



כִּי הוּא לְבַד רִפְאוּת אֱלֵי נוּגִים



מָה בָּעֲרוּ הָאֹמְרִים טָלֶה



רִאשׁוֹן לְמַזָּלוֹת וְלֹא דָגִים.

  

וָיֹּאמֶר הָאִישׁ חַיֶּיךָ כְּבָר הִסְכִּימוּ חַכְמֵי הַתְּכוּנָה / וּבְכָל שִׁבְטֵי יִשְׁרָאֵל מוּדַעַת נֶאֱמָנָה / כִּי דֶגֶל טָלֶה נוֹסֵעַ בָּרִאשׁוֹנָה / וְדֶגֶל דָּגִים בָּאַחֲרוֹנָה / וְהִנְנִי נִשְׁבָּע כִּי דָג לֹא יָבֹא לְפָנֵינוּ / עַד אֲשֶׁר בְּחָכְמַתְךָ תַרְאֵנוּ / מַעֲלַת דָּגִים עַל טָלֶה / וְאִם הַסּוֹד הַזֶּה תְגַלֶּה / לְהַנְחִיל אֹהֲבַי יֵשׁ וְאֹצְרוֹתֵיהֶם אֲמַלֵּא /



וָאַעַן וָאֹמַר דַּע אֲדֹנִי כִּי אֵין לַטָּלֶה יִתְרוֹן / בַּעֲבוּר הֱיוֹתוֹ לַמַּזָּלוֹת רִאשׁוֹן וְדָגִים אַחֲרוֹן / כִּי לְמַעֲלוֹת הַדָּגִים וְכוֹכָבָיו הָעֲדִינוֹת / הוּשַׂם בָּאֲחֲרוֹנָה מְאַסֵּף לְכָל הַמַּחֲנוֹת / מְצֹרָף אֶל זֶה הֱיוֹת הַכּוֹכָבִים וְאִם רַבּוּ כַחוֹל / הַכּוֹכָבִים אֲשֶׁר בִּשְׁנֵים עָשָׂר מַזָּלוֹת יְהַלְּלוּ שֵׁם אֵל בְּמָחוֹל / כִּי סוֹבְבִים סוֹבְבִים הוֹלְכִים הַמַּזָּלוֹת / בַּיָּמִים וּבַלֵּילוֹת / לְדִמְיוֹן אֲנָשִׂים יָחֹלּוּ בִּמְחוֹלוֹת / וּמִדֶּרֶך אַנְשֵׁי הַמָּחוֹל לְהִתְגּוֹלֵל כַּמַּזָּלוֹת סָבִיב / וּבַמָּחוֹל הָאַחֲרוֹן חָבִיב /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

אֵל עוֹלָם שָׂם הַמַּזָּלוֹת



כִּדְמוּת מָחוֹל סוֹבֵב סָבִיב



עַל כֵּן הוּשָׂם דָּגִים אַחְרוֹן



כִּי בַמָּחוֹל אַחְרוֹן חָבִיב.

  

וַיֹּאמֶר הָאִישׁ הֵטַבְתָּ אֶת אֲשֶׁר דִּבַּרְתָּ / וְהִנְנִי מֵקִים הָאֹמֶר אֲשֶׁר גָּזַרְתָּ / וַיְצָו לְהָבִיא בָרִאשׁוֹנָה הַדָּגִים / וְנִהְיֶה אוֹכְלִים וְשׁוֹתִים וְחוֹגְגִים / וְאָחָר הַדָּגִים הֵבִיאוּ הַטְּלָאִים הַצְּלוּיִים מוֹר עוֹבֵר נוֹטְפִים / כְּאִלּוּ מִן הַשּׁוֹשַׁנִּים קְטוּפִים / אוֹ מִשִּׂפְתוֹת הַיְּעֵלוֹת חֲטוּפִים / וּמְתוּקִים מִדְּבָש וְנֹפֶת צוּפִים / וְהַמִּרְקָחוֹת וּזְנִים הָיוּ עַל הַשֻּלְחָן כַּאֲבָנִים / וְהַשְׁקוֹת בִּכְלֵי זָהָב וְכֵלִים מִכֵּלִים שׁוֹנִים / וְיֵין מַלְכוּת רָב / מָתוֹק לַנֶּפֶשׁ וְלַחֵךְ יֶעֱרָב / וְאָכַלְנוּ וְשָׁתִינוּ עַד אֲשֶׁר לֹא הִכַּרְנוּ מִזְרָח מִמַּעֲרָב /



וָאֹמַר יֵשׁ לִי רָב / בָּרוּךְ אֲדֹנָי צוּרִי הַמְּלַמֵּד יָדַי לַקְּרָב /



וַיֹּאמֶר הָאִישׁ אָרוּר הָאִישׁ אֲשֶׁר לֹא יִהְיֶה הַיּוֹם שִׁכּוֹר / עַד אֲשֶׁר לֹא יַבְחִין בֵּין הַלֶּתֶך וְהַכֹּור / וַיֹּאמֶר בְּדִבּוּר אֶחָד שָׁמוֹר וְזָכוֹר / וְלֹא יַבְחִין בֵּין פֶּתַח תִּקְוָה לְעֵמֶק עָכוֹר / לְמַעַן זֶה הַיּוֹם כָּל יְמֵי חַיֵּינוּ נִזְכּוֹר /



וָאֹמַר אֶל הָאִישׁ אִם תִּפְטֹר אוֹתִי הַיּוֹם מִלּוֹמַר עוֹד שִׁיר / אֹמַר בָּרוּךְ אֲדֹנָי וַאעְשִׁיר / כִּי אַחֲרֵי הֱיוֹתִי שִׁכּוֹר / וְחַבֵּר שִׁיר עוֹד לֹא אֶזְכּוֹר / וַיֹּאמֶר הָאִישׁ חִבַּרְתָּ שְׁנֵי שִׁירִים וְאֵיךְ מִשִּׁיר תִּהְיֶה שׁוֹתֵק / וּכְבָר לִמַּדְתָּנוּ וְהַחוּט הַמְּשֻׁלָּשׁ לֹא בִמְהֵרָה יִנָּתֵק / וְלָכֵן חַבֵּר שִׁיר שְׁלִישִׁי / וְתִשְׁקְלֶהוּ בְּמֹאזְנֶיךָ / וְאַחֲרֵי זֹאת לְהִשְׁתַּכֵּר שִׂים פָּנֶיךָ / עַד אֲשֶׁר רֹאשְׁךָ עֲלֶיךָ יִכְבָּד /



וָאָקוּם וָאֶעֱשֶׂה מִצְוַת הַנִּכְבָּד /

  

וָאֶשָּׂא מְשָלִי וָאַֹמר:

  

מָתְקוּ לְחִכִּי טְלָאִים וְדָגִים



עָרְכוּ לְפָנַי יְדֵי הַצְבָאוֹת



הֵם בַּזְמַן מָן לְטַעֲמָם וְהִנְנִי



אוֹדֶה לְקָדוֹשׁ אֲדֹנָי צְבָאוֹת



טָלֶה בְּשׂוּמוֹ לְרֹאשׁ כָּל גְּדוּדָיו



וְדָגִים מְאַסֵּף לְכָל הַצְּבָאוֹת.

  

וַיְהִי כְּהַשְׁלִימִי הַשִּׁירָה / הָלַכְנוּ וְיָשַׁבְנוּ בַעֲלִיַּת הַמְּקֵרָה / וַיּוּשָׂם לְפָנֵינוּ שֻׁלְחָן מַעֲשֶׂה צַעֲצוּעִים / וְכוֹסוֹת מְלֵאִים וּגְבִיעִים / וָאֶהְיֶה שַׁעֲשׁוּעִים / וְעָשִׂינו מִן הַיַּיִן מִשְׁפָּטִים / וְכַאֲשֶׁר בָּאתִי לָקוּם מָצָאתִי בִטְנִי עֲרֵמַת חִטִּים / וְלֹא יָכֹלְתִּי לָקוּם כְּאִלּוּ הָיִיתִי אָסוּר בָּרְהָטִים /



וַיּוֹאֵל הָאִישׁ גַּם הוּא לָקוּם / וּבִרְאוֹתוֹ כִּי לֹא יוּכַל וְלֹא יְקִימוּהוּ כָל הַיְּקוּם / אָמַר נְתָּנַנִי אֲדֹנָי בִּידֵי לֹא אוּכָל קוּם / כִּי רֹאשִׁי הוֹלֵךְ הֵנָּה וָהֵנָּה / וְהַיַּיִן שָׁלַח אֵשׁ בְּעַצְמוֹתַי וַיִּרְדֶּנָּה /



וָאַעַן וָאֹמַר הִנְנִי נִשְׁבָּע בְּחַיֵּי נָחוֹר / כִּי תָאַבְתִּי לִשְׁתוֹת מֵי שִׁיחוֹר / כִּי הַיַּיִן שָׁלַח יָדוֹ מִן הַחוֹר / וּפָרַשׂ רֶשֶׁת לְרָגְלַי הֶשִׁיבַנִי אָחוֹר / וְלֹא אַכִּיר הַלָּבָן מִן הַשָּׁחוֹר /



וְכִרְאוֹת אַנְשֵׁי הַבַּיִת יַד הַשִּׁכְּרוּת הַחֲזָקָה / עָשׂוּ בַבַּיִת קוֹל דְּמָמָה דַקָּה / וַתִּפֹּל עָלֵינוּ תַרְדֵּמָה וְיָשַׁנּוּ שֵׁנָה עֲמֻקָּה / עַד אֲשֶׁר פָּנָה הַשֶּׁמֶשׁ לַעֲרֹב דֶּרֶךְ גְּבוּלוֹ / וְהַשִּׁכְּרוּת חָלַף הָלַךְ לוֹ / וְעָמַדְתִּי יָמִים שְׁלֹשָׁה עִם הַנִּכְבָּד שִׂמְחַת גִּילִי / וְזֶה הָיָה חֶלְקִי מֶכָּל עֲמָלִי /



וַיֹּאמֶר הַשָּׂר נָדִיב הָיָה נִכְבָּד / וְזֶבֶד טוֹב אוֹתְךָ זָבָד / הֲדָמוּ הַגְּבִירוֹת אֵלָיו בִּנְדִיבוּת /



וָאַעַן וָאֹמַר בְּחַיַי אָסְפוּ עִם הַנְּדִיבוּת / כָּל עֲרֵבוּת וְכָל חֲשִׁיבוּת / כִּי עָמְדוּ לְשָׁרְתֵּנִי בְפָנִים יָפוֹת / וּבִזְרוֹעוֹת חֲשׂוּפוֹת / כִּשְׁפָחוֹת חֲרוּפוֹת / וְלֹא סָרוּ מֵעֲשׁוֹת הַבִּרְיָּה וְלַבֵּב לְבִיבוֹת / וְהַמִּטּוֹת הָיוּ בְאֵטוּן מִצְרַיִם חֲטוּבוֹת / וּבָכוּ לְפֵרוּדִי פַעַם וּפַעֲמַיִם / כְּאִלּוּ נִפְתְּחוּ אֲרֻבּוֹת הַשָּׁמַיִם / אָמְרוּ עַד מֶה הַשָּׂר תִּשְׁבֵּנוּ / הִנֵּה נְדוֹדְךָ יִקְרַע סְגוֹר לִבֵּנוּ /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

אַז תַּחֲנוֹת לִבָּךְ יְדִידִים יִשְׁאֲלוּ



וָאֶעֱנֶה אוֹתָם בְּקָמֶרִינוֹ



שָּׁם תַּאֲוַת נֶפֶשׁ וְשָם אָרִינוּ



מֹר אָהֳלוֹת עִם כָּל בְּשָׂמִים נַעֲלוּ

  

שָׁמָּה אֲסִירִים מִתְּמוּתָה נִגְאֲלוּ



שָׁם עִיר אֱלֹהִים בֶּאֱמֶת שָׁרִינוּ



בִּשְׁתֵּי לְבִיבוֹת מָן אֲשֶׁר בָּרִינוּ



עִם יַעֲלוֹת בִּגְאוֹן אֲדֹנָי צַהֲלוּ

  

נַפְשִׁי פְדוּת דֶּמַע רְאִיתִיו עַל לְחִי



יַעְלוֹת וְדוֹדִים לִי כְּחָלִיל יֶהֱמוּ



הָאֹמְרִים עַד מָה צְּבִי תִּשְׁבֵּנוּ

  

מַה תֶּהֱמִי נַפְשִׁי וּמַה תִּשְׁתּוֹחֲחִי



פִּקְּדִי לְשָׁלוֹם הַצְּבִיּוֹת נָאֲמוּ



אוֹי לַנְּדוֹד יִקְרַע סְגוֹר לִבֵּנוּ.

  

וַיֹּאמֶר הַשָּׂר הֲלֹא חָסַרְתָּ דָבָר בְּבֵית הַנִּכְבָּד /



וָאַעַן וָאֹמַר בְּחַיָי חָסַרְתִּי דָבָר אֲשֶׁר עָלַי מְאֹד כָּבֵד / כִּי רָאִיתִי בְבֵית הַשָּׂר שׁוּמִים / כְּאִלּוּ נִגְזְרוּ מִן הָאוּרִים וּמִן הַתֻּמִּים / וּבְצָלִים מְהֻלָּלִים / יַחְפִּירוּ בְיָפְיָם הַגַּלְגַּלִּים / וְגִלְדֵיהֶם עָבִים/ וְהַמָּן בְּתוֹכָם נִצְפָּן / וּמַרְאֵיהֶם וּמַעֲשֵׂיהֶם כַּאֲשֶׁר יִהְיֶה הָאוֹפַן בְּתוֹךְ הָאוֹפָן / וְהֶחָצִיר עָב וְיָקָר / כְּמַטֶּה אֱלִישָׁע וְיָפְיוֹ לֹא נֶחְקָר /



וָאֹמַר אֶל הַנִּכְבָּד וְאֶל אֶדֶר הַיְקָר / הַדָּג וְהַבָּשָׂר טָפֵל וְהַשּׁוּם וְהַבָּצָל וְהֶחָצִיר עִקָּר / הָבִיאוּ לִי מַטֵּה הֶחָצִיר כִּי רְצוֹנִי לַעֲשׂוֹת בֹּו הַמּוֹפְתִים / וְשָׁמַעְתִּי אוֹמְרִים כִּי הוּא טוֹב לְמַעֲשֶׂה חֲבִתִּים /



וַיֹּאמֶר הָאִישׁ חַיֶּיךָ בְבֵיתִי הַיּוֹם שׁוּם וְחָצִיר לֹא תֹאכַל / כִּי רֵיחָם נִלְאֵיתִי כַלְכֵּל וְלֹא אוּכָל / וְזֶה הַשֻּׁלְחָן לִפְנֵי אֲדֹנָי / וְלֹא יִקְרַב בּוֹ דָבָר שֶׁל גְּנָאי /



כְּשָׁמְעִי הָיִיתִי אֶל הַיְּגוֹנִים לְמַאֲכָל / מִלְּבַד שֶׁהַנִּכְבָּד הֶחֱזִיק אוֹתִי סָכָל / וָאֹמְרָה הַיּוֹם לֻקַּח מִמֶּנִּי מַחְמָד עֵינָי / כִּי הַשּׁוּם וְהַבָּצָל וְהֶחָצִיר מִבְחַר עֲדָנָי / וְעַתָּה אֶפְּלָה נָא בְיַד אֲדֹנָי /

  

וָאֶשָּׂא מְשָלִי וָאֹמַר:

  

אֶרְאֶה אֲנִי הַמָּן וְאֶרְאֶה הַשְּׂלָיו



אַךְ אֹמְרִים לִי סוּר וְאַל תִּקְרַב הֲלוֹם



כָּל מַעֲדַנִּים הֵם בְּעֵינַי כַּחֲלוֹם



פָּגַע גְאוֹן לִבִּי כְּהַיּוֹם שׁוֹלְלָיו

  

כִּי הַבְּצָלִים הֵם לְלִבִּי גוֹאֲלָיו



הַשּוּם וְהֶחָצִיר וְהַבְּצָלִים שְׁלוֹם



בֵּיתִי וְעַל רֹאשִׁי עֲטֶרֶת יַהֲלֹם



כִּי בָם לְבָבִי הוּא יְכַלְכֵּל מַחֲלָיו

  

הַשּוּם צְבִי אֶרֶץ וּפֶרַח כָּל יְבוּל



הַחֵן יְלָדָהוּ וְהַהוֹד רִקְּמוֹ



עַיִש וּבָנֶיהָ בְּרִית לוֹ כוֹרְתִים

  

גִלְדֵי בְּצָלִים הֵם כְּגַלְגַּלֵי זְבוּל



אוֹפָן בְּתוֹך אוֹפָן וְהֶחָצִיר כְּמוֹ



מַטֶּה אֱלִישָׁע לַעֲשׂוֹת הַמּוֹפְתִים.

  

וַיֹּאמֶר הַשָּׂר הֵיטַבְתָּ לְדַבֵּר / וְשִׁירִים לְחַבֵּר / בָּרוּךְ אֲדֹנָי אֱלֹהֵי יִשְׂרָאֵל / אֲשֶׁר לֹא הִשְׁבִּית לַשִּׁירִים וְלַמְּלִיצוֹת גּוֹאֵל.
