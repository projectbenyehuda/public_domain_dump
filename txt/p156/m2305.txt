


	
	מחברות עמנואל: הַמַּחְבֶּרֶת הָעֶשְׂרִים וְחָמֵש
	




אָמַר הַמְּחַבֵּר: יָצָאתִי עִם הַשָּׂר בִּימֵי הַנְּעוּרִים / אַחַר סְעֻדַּת פּוּרִים / לִרְאוֹת בִּשְׂחוֹק הַשִּׁכּוֹרִים / כִּי שָׁמַעְנוּ אוֹמְרִים בָּרְחוֹב יֵאַסְפוּ כָּל הָעֳדָרִים/ כְּאִיש אֶחָד חֲבֵרִים / וְשָׁמָּה יִצְבְּרוּ הַשִּׁכּוֹרִים חֳמָרִים חֳמָרִים / וְכַּאֲשֶׁר קָרַבְנוּ אֶל הָרְחוֹב / וַאֲנַחְנוּ שׁוּלֵי הַגַּאֲוָה נִסְחוֹב / רָאִינוּ קְהִלָּה גְדוֹלָה / וְהִנֵּה שָׁם קוֹל הַמֻלָּה / כְּקוֹל מַחֲנֶה / וְעַם רָב אֲשֶׁר לֹא יִסָּפֵר מֵרֹב וְלֹא יִמָּנֶה / יִתְלַכְּדוּ וְלֹא יִתְפָּרָדוּ / וְכַאֲשֶׁר שִׁמְשֵׁנוּ זָרְחָה נֹדָדוּ / רָאוּנוּ נְעָרִים וְנֶחְבָּאוּ וִישִׁישִׁים קָמוּ עָמָדוּ / וַיֹּאמְרוּ לָנוּ בְּרוּכִים הַבָּאִים / וַיִּתְּנוּ לָנוּ מָקוֹם בְּרֹאשׁ הַקְּרוּאִים / יָשַׁבְנוּ עִמָּם וְעֵינֵינוּ רוֹאִים / חֶבֶל שִׁכּוֹרִים כְּסָבְאָם סְבוּאִים / לְפָנֵינוּ מִתְנַבְּאִים / אָז אָמַרְתִּי נַפְשִׁי בְּתוֹך לְבָאִים / קְצָתָם הָיָה קוֹרֵעַ כְּסוּתוֹ בַחֲמָתוֹ / וְאוֹמֵר כִּי הוּא שְׁלֹמֹה בְחָכְמָתוֹ / וּקְצָתָם דּוֹבֵר הַוַּת נַפְשׁוֹ / וּקְצָתָם יָקִיא פִרְשׁוֹ / יַעֲלֶה בָאְשׁוֹ/ וּקְצָתָם שׁוֹתֵת מַיִם בֵּין יְרֵכָיו / וּלְעֵינֵי הַכֹּל יַעֲשֶׂה צְרָכָיו / וּקְצָתָם מִן הָעֵדָה יָרֹם / וַיִּפְשֹׁט בְּגָדָיו וְיֵלֵך עָרֹם / וַיֹּאמֵר כִּי הִיא מִצְוַת עֲשֵׂה שֶׁהַזְּמָן גְּרָמָהּ / וְהַשִׁכְּרוּת וְהַפְּרִיצוּת בְּיוֹם זֶה עָרְמָה / וּקְצָתָם יָחוּלוּ בִמְחֹלוֹת / בַּחוּרִים וְגַם בְּתוּלוֹת / וּתְדַבֵּרְנָה כְּאַחַת הַנְּבָלוֹת / אַף בְּלֵב יִפְעָלוּן עֹלוֹת / וָאֶקְרָא שֵׁם הַמָּקוֹם הַהוּא עֵמֶק הַבָּכָא / בִּרְאוֹתִי שָׁם מְעוּף צוּקָה וַחֲשֵׁכָה / וּבִרְאוֹתִי אֵלֶּה הַנִּבְזִים / אֲשֶׁר בְּמַעְגְּלוֹתָם נְלֹזִים / וּבְיֵין תּוֹעֶה שֹׁכְבִים וְהֹזִים / אָמַרְתִּי מִי נָתַן הַיַּיִן לְבֹזְזִים / וְהַשֵּׂכֶל לִמְשִׁסָּה / לִשְׂחוֹק אָמַרְתִּי מְהוֹלָל וּלְשִׂמְחָה מַה זֹּה עֹשָׂה / רָאִינוּ שָׁם מַה שֶּׁלֹּא יוּכַל אִישׁ לְדַבֵּר וְאֹזֶן לִשְׁמֹעַ / עַד אֲשֶׁר הֵחֵלּוּ הָעֵינַיִם לִדְמֹעַ / לֹא רָאִיתִי כָהֵנָּה בְכָל אֶרֶץ מִצְרַיִם לָרֹעַ / אָז אָמַרְתִּי עֵת לִקְרֹעַ / וּבִרְאוֹתִי סִכְלוּת הָאֲנָשִׁים גּוֹבֵר / וַאֲנִי הָיִיתִי בְחוֹמַת שִׁכְּרוּתָם שׁוֹבֵר / וְהִנֵּה אִישׁ שִׁכּוֹר לְפָנֵינוּ עוֹבֵר / וְעַל רִיב לֹא לוֹ מִתְעַבֵּר / פַּעַם יָשִׁיר וִירוֹנֵן / וּפַעַם יִסְפֹּד וִיקוֹנֵן / וּפַעַם לְרוֹאָיו מִתְחַנֵּן/ וּפַעַם כְּגִבּוֹר מִיַּיִן מִתְרוֹנֵן / פַּעַם יִתְעַבָּר / וּפַעַם יִתְגַּבָּר / וְחֹק הַמּוּסָר אִבֵּד וְשִׁבַּר / וּדְבָרָיו לֹא לִזְרוֹת וְלֹא לְהָבַר / כִּי הַשִּׁכְּרוּת עַל כָּל אֲפִיקָיו גָּבַר / וְחָלַף בּוֹ וְשָׁטַף וְעָבָר / וְאַחֲרֵי אֲשֶׁר הִתְעוֹלֵל לְעֵינֵי הָעַמִּים פָּעַר פִּיו לִבְלִי חֹק/ וְסָפַק יָדָיו בְּקִיאוֹ וְהָיָה לִשְׂחוֹק /

וַיַּרְא הָעָם וַיָּנוּעוּ וַיַּעַמְדוּ מֵרָחוֹק / וַיִּהְיוּ הַסְּגַנִּים וְהַשָּׂרִים / מֵאֹדוֹת הָאִישׁ הַלָּזֶה מְדַבְּרִים / וּבִהיוֹת הָאִישׁ הַהוּא שֹכֵב וְהֹזֶה / אָמַרְתִּי אֵין לִתְמֹהַּ עַל הָאִישׁ הַלָּזֶה / אֲשֶׁר הוּא דְרָכָיו בֹּזֶה / וְאִם הַמִּנְהָג כְּמִנְהַג יֵהוּא / לֹא יוֹם הַכִּפּוּרִים הוּא / אַךְ יוֹם הַקִּיא פּוּרִים הוּא /

כִּשְׁמֹעַ הַשָּׂר אֶת דְּבָרַי / נִפְלָא הַפְלֵא וָפֶלֶא עַל מַאֲמָרַי / וַיֹּאמֶר אֵלָי חַיֶּיךָ הִשְׁלַכְתָּ מִפִּיךָ סַפִּיר / אוֹר הַשֶּׁמֶש בְּיָפְיוֹ מַחְפִּיר / לֹא יְסֻלֶּה בְכֶתֶם אוֹפִיר / אֶחְשֹׁב כִּי בָא בְפִיךָ דֶרֶךְ מִקְרֶה / וְאִם אֵין זֶה אֵין כָּמוֹךָ בְכָל הָאָרֶץ מוֹרֶה /

וָאַעַן וָאֹמַר חַיֶּיךָ הַשָּׂר לוּלֵי הֱיוֹתִי יָרֵא / פֶּן אַשְׁלִיךְ הַיּוֹם פְּנִינַי / לִפְנֵי אֲנָשִׁים לֹא יָבִינוּ עִנְיָנַי / וְחִלְלוּ אֶת סְפוּנַי / הָיִיתִי מַזְכִּיר מֵעִנְיָנוֹ שֶׁל יוֹם דְּבָרִים רְצוּיִים / אֲשֶׁר לֹא נִרְאוּ בְכָל הָאָרֶץ וּבְכָל הַגּוֹיִים / אָמְנָם כִּי יַפְסִיד הַיּוֹם דְּבָרָיו / וְאִם אֵין אִישׁ בְּכָל הֶהָמוֹן הַזֶּה הָרָב / אֲשֶׁר לֹא מִלָּא הַיּוֹם מִיַּיִן חֲדָרָיו / וְהֵצַר לוֹ בְכָל שְׁעָרָיו / עַד רֶדֶת חוֹמוֹתָיו / כַּאֲשֶׁר תִּרְאֶה כָּל אִישׁ מֵהֶם יְגַלֶּה חָכְמֹתָיו / וְהָאִישׁ הַזֶּה אֲשֶׁר תִּרְאֵהוּ מִשְׁתַּגֵּעַ / הוּא חוֹבֵר חֲבָרִים מְחֻכָּם וְדַרְדֵּעַ / בְּעֶרֶך אֲשֶׁר לֹא יַגִּיעַ אֶל הַקָּצֶה הָאַחֲרוֹן מִן הַשִּׁכְּרוּת / כַּאֲשֶׁר הֱיִיתֶם רוֹאִים עַל לוּחוֹת סִפְרִי חָרוּת / אוּלַי כָּתַבְתִּי דִבְרֵי הַשִּׁכְּרוּת עַל סֵפֶר / כַּאֲשֶׁר עִם לְבָבִי נוֹתֵן אִמְרֵי שֶׁפֶר / וְלוּ שְׁמַעְתֶּם שִׁכְּרוּת חֲבֵרִי אֲמַרְתֶּם רָב / וְכָמוֹהוּ לֹא נִמְצָא מִן הַמִּזְרָח עַד הַמַּעֲרָב /

אָז נִקְהֲלוּ עָלַי קָהָל מֵאֲנָשִׁים / שָׂרֵי עֲשָׂרוֹת וְשָׂרֵי חֲמִשִּׁים / זְקֵנִים וִישִׁישִׁים / לְבַד מִן הַיְֹהוּדִים הַפְּרָזִים הַיּוֹשְׁבִים בְּעָרֵי הַפְּרָזוֹת / וַיֹּאמְרוּ חַי אֲשֶׁר עָשָׂה לָנוּ אֶת הַנֶּפֶשׁ הַזֹּאת / כִּי דְבָרֶיךָ דִבְרֵי נְבוּאוֹת / וְלֹא נָסוּר מִזֶּה עַד אֲשֶׁר אֹדוֹת חַבֵרְךָ תַשְׁמִיעֵנוּ / וְשִׁכְּרוּתוֹ וְתַעְתּוּעָיו תּוֹדִיעֵנוּ/

וָאַעַן וָאֹמַר אַחֲרֵי אֲשֶׁר חִשְׁקְכֶם גּוֹזֵר לְהַגִּידוֹ מִי לֹא יִנָּבֵא / הִנְנִי מֵעֲשׂוֹת מִצְוַתְכֶם לֹא אֵחָבֵא / חַיֵיכֶם הָיָה לִי לְחָבֵר אִישׁ זוֹלֵל וְסוֹבֵא / וּלְקוֹל מוֹרָיו לֹא יִשְׁמַע וְלֹא יֹאבֶה / וּמִדֵּי יוֹם יוֹם יִשְׁתַּכֵּר / עַד אֲשֶׁר בֵּין טוֹב לְרַע לֹא יְבַקֵּר / וּבֶאֱמוּנָתוֹ יְשַקֵּר / וּבְעֵת שִׁכְּרוּתוֹ הוֹרֵג כָּל הַבָּא לְפָנָיו וְעוֹקֵר / כָּל הַנִּמְצָא יִדָּקֵר / יָשִׂים פִּיו אֶל הֶחָבִית כְּשֶׁיִּצְמָא / וְכָל אֲשֶׁר בְּתוֹכוֹ יִטְמָא / כֵּן יַעֲשֶׂה יוֹם יוֹם אֵין נָקִי / וּבְיוֹם אֲשֶׁר לֹא יָקִיא / יִפְחַד פֶּן הָאָרֶץ אוֹתוֹ תָקִיא / יַעֲמֹד עַל מִשְׁמֶרֶת הַיַּיִן קָבוּעַ / וּבְשִׂנְאָתוֹ הַמַּיִם יִשְׁבֹּר הַכַּד עַל הַמַּבּוּעַ / וְיַעֲשֶׂה פּוּרִים בְּכָל שָֹבוּעַ / וְכַאֲשֶׁר יִשְׁאָלוּהוּ עַל אֹדֹתָיו / יֹאמַר כִּי כָךְ קִבֵּל מֵרַבּוֹתָיו / וְכֵן הָיוּ נוֹהֲגִים אֲבוֹתָיו וַאֲבוֹת אֲבוֹתָיו / יֹאמַר מַה מְּאֹד חָטָא הַזַּגָּג הָאָרוּר / צְפוּנָה חַטָּאתוֹ וַעֲוֹנוֹ צָרוּר / אֲשֶׁר לֹא עָשָׂה הַכּוֹסוֹת לְדִמְיוֹן הַקְּדֵרָה וְהַפָּרוּר / כִּי אָז יָצָאנוּ מֵעַבְדוּת לַשִּׁחְרוּר / כִּי בַּעֲשׂוֹתוֹ הַכּוֹסוֹת קְטַנִּים הֱשִׁמָּנוּ וְהֶלְאָנוּ / כַּאֲשֶׁר כָּזֹּאת הֶרְאָנוּ / הֶחֱלָנוּ וְהִבְהִילָנוּ / וְלוּ עֲשָׂאָם כִּכְלֵי הַשֶּׁתֶן אָז טוֹב לָנוּ / וְעַל דִּבְרֵי הַצּוֹמוֹת / יַחְגֹּר חֵמוֹת עֲצוּמוֹת / יֹאמַר כִּי בַּעֲבוּרָם לֹא תִחְיֶינָה הָעֲצָמוֹת / וּבִרְאוֹתוֹ בֶאֱלוּל נָשִׁים מִתְעַנּוֹת וּמַשְׁלִימוֹת / וּנְעָרִים וְּזֵקנִים / בִּבְכִי יָבֹאוּ וּבְתַחֲנוּנִים / יַעֲשֶׂה מִסְפֵּד כַּתַּנִּים / וְאֵבֶל כִּבְנוֹת יַעֲנָה / וַיֹּאמֶר כִּי בַּעֲבוּר אֲשֶׁר בְּיָמָיו הִתְעַנָּה / אַחֲרִיתוֹ מָרָה כַלַּעֲנָה / וְחָלָה חֳלִי כָבֵד / בַּטְּחוֹל וּבַכָּבֵד / וְאִם עוֹד אַחֲרִיתוֹ עֲדֵי אֹבֵד / וְאֶל צוֹם גְּדַלְיָה / אָמַר כִּי דָמוֹ בְרֹאשׁוֹ הָיָה / כִּי נַפְשׁוֹ לֹא חִיָּה / וּלְגַלּוֹת סִתְרֵי מַצְפוּנָיו / וְעֶרְוַת זְנוּנָיו / וְהַשָּׂטָן מְרַקֵּד בֵּין קְרָנָיו / וְיִתְנַבֵּא בְתוֹךְ הַבַּיִת וְיָחוּד חִידוֹת / וְיִתְכַּוֵּן לְדַבֵּר בְּסוֹדוֹת / וְיַחְשֹׁב כִּי הַכּוֹסוֹת צָרִים וּקְצָרִים

אִם הָיוּ כִּכְלֵי הַשֶּׁתֶן

אָז הָיוּ נָאִים וַהְדוּרִים

אַף יִשָּׁבַע כִּי כָל שׁוֹתֵי

מַיִם הֻכּוּ בַסַּנְוֵרִים

יֹאמַר הִנֵּה עִם מֵי מַבּוּל

הִשְׁחִית הָאֵל כָּל הַיְצוּרִים

הִנֵּה כָּל הַבֵּן הַיִּלּוֹד

הִשְׁלִיכוּ בַיְאֹר הַמִּצְרִים

הִנֵה פַרְעֹה עִם כָּל חֵילֹו

צָלְלוּ תוֹךְ מַיִם אַדִּירִים

הִנֵּה מֹשֶׁה עַל הַמַּיִם

אָמַר שִׁמְעוּ נָא הַמּוֹרִים

הִנֵּה עֵת הָאִשָּׁה תִשְׂטֶה

תַּחַת אִישָׁהּ תּקִחַּ זָרִים

לַצְבּוֹת בָּטֶן לַנְפִּיל יָרֶך

צִוָּה אֵל לִשְׁתּוֹת מֵי מָרִים

לוּ עָשָׂה אֵל אֶת הַמַּבּוּל

מִזֶּרֶם יֵינוֹת כַבִּירִים

לֹא עָשָׂה נֹחַ תֵּבַת עֵץ

גֹּפֶר לִהְיוֹת בָּהּ נִסְגָּרִים

אַף לֹא נִנִעָרוּ הַחַיֵּים

אָכֵן הָיוּ אָז נִנְעָרִים

יֵעוֹר בַּבֹּקֶר בַּבֹּקֶר

יָקוּם אֶל קוֹל הַצִּפֳּרִים

יֹאמַר נָפְלָה אֵשׁ בִּגְרוֹנִי

חַמּוֹתִי רָאִיתִי אוֹרִים

יָבֵש חִכִּי אַף מָךְ עֶרְכִּי

כִּי יָבְשׁוּ כָּל הַצִּנּוֹרִים

נִלְאֵיתִי כַלְכֵּל לֹא אוּכָל

נָפַלְתִּי בִידֵי אַכְזָרִים

גְּשׁוּ רֵעִים אֵלַי כִּי עִם

הַצָּמָא לִי דִין וּדְבָרִים

מַהְרוּ חוּשׁוּ נָא כִּי רָעְשׁוּ

סִפַּי הֻכּוּ הַכַּפְתּוֹרִים

הַגְמִיאוּנִי מִיֵּין חֶלְבּוֹן

כִּי הוּא רִפְאוּת צִיר וּמְזוֹרִים

אִם לִי יַיִן יִתֵּן בַּכּוֹס

עֵינוֹ יִתְהַלֵּך מֵישָׁרִים

אֶשְׂמַח כִּי רַב חֵילִי וּכְאִישׁ

יָצָא לִמְלֹךְ מִבֵּית סוּרִים

אָז יַגִּישׁוּ לוֹ מֵרֵעָיו

יַיִן מַרְפֵּא לֵב וּבְשָׂרִים

יִשְׁתֶּה יִשְׁכָּר אַף יִתְגַּבָּר

יַעַשׂ אֵלָיו חַרְבוֹת צוּרִים

וּכְאַרְיֵה כֵן יִכְסֹף לִטְרֹף

כִּכְפִיר יוֹשֵׁב בַּמִּסְתָּרִים

יִשְׁכַּח רִישׁוֹ עֵת תִּירוֹשׁוֹ

עַל רוּם רֹאשׁוֹ קִנּוֹ יָרִים

יָשִׂישׂ יִשְׂמַח גִּיל לוֹ יִצְמַח

חֵלֶף בִּכְיוֹ שָׁר בַּשִׁירִים

וּבְעֵת יַחְשֹׁב כִּי הוּא יֹאמַר

אָז אִמְרֵי נֹעַם וּטְהוֹרִים

אִם אַתָּה אִישׁ בִּינָה תִשְׁמַע

תַּחַת הַשִּׁיר קוֹל הַסִּירִים

עֵת יִתְנוֹדָד עֵת יִתגּוֹדָד

יִקְרָא הֵידָד לֹא הֵד הָרִים

וּכְחֶזְיוֹן לַיְלָה כֵּן יוֹרֵד

יֵשֵׁב בָּדָד מָאַס עָרִים

וּכְאִישׁ שׁוֹכֵב עַל רֹאש חִבֵּל

כֵּן יִתְאַבֵּל יוֹמוֹ יַחְרִים

וּכְפֶתַע פִּתְאֹם הוּא יִנְאֹם

כִּי הוּא יִלְכֹּד עִיר מִבְצָרִים

וּלְלֹא דָבָר הוּא יִתְגּבָּר

יֹאמַר אֶגְזֹר יָם לִגְזָרִים

וּרְכוּש אֶבְזֹר אֶנְהֶה מָזֹר

אִישִׁים אֶעְזֹר הֵם נֶעְזָרִים

עִוְרִים אוֹלִיךְ יַחַד אַשְׁלִיךְ

בַּכְּלִמָּה חָרָשֵׁי צִירִים

עִם כָּל שׁוֹאֵל הוּא יִתְגָּאֵל

יֹאמַר בִּינֹתִי בַּסְּפָרִים

יֹאמַר לוּ הָיוּ הֶעָבִים

יַיִן אוֹ שֵׁכָר מַמְטִירִים

אוֹ הָיוּ הַיַּמִּים יֵינוֹת

לֹא יַרְאֻנִי הַמִּשְׁבָּרִים

אַף לֹא קִוִּינוּ גַן עֵדֶן

רַק לִהְיוֹת בַּיָּם נִקְבָּרִים

מַדּוּעַ לֹא הָיִיתִי דָג

עִם קַשְׂקֶשֶׂת וּסְנַפִּירִים

וִימֵי חַיַּי הָיוּ כִימֵי

הַיָּם נֵצַח וּלְדֹור דּוֹרִים

וּמְלֹא הַיָּם הָיָה יַיִן

מִן הַיֵּינוֹת הַנִּבְחָרִים

וּלְבִלְתִּי רֶדֶת עוֹד גֶּשֶׁם

הָיוּ שָׁמַיִם נֶעְצָרִים

אַף מַעְיָנוֹת נִכְבַדֵּי מַ-

יִם וּתְהוֹם הָיוּ נִסְכָּרִים

פֶּן יָבֹאוּ בִגְבוּלֵנוּ

פֶּן יִהְיוּ לַיָּם נִגָּרִים

אָז לֹא קִנֵּאתִי אֶל שָׂרִים

וּלְמִנְזָרִים אוֹ טַפְסָרִים

אַף לֹא קִנֵּאתִי לִבְנוֹת עָשׂ

לִצְבָא שַׁחַק הַמַּזְהִירִים

לֹא קִנֵּאתִי כִימָה וּכְסִיל

אַף לֹא קִנֵּאתִי לַמְּאוֹרִים

אִם בַיָּם דַרְכִּי לֹא אִירָא

מִן הַגַּלִּים הַנִּסְעָרִים

וּבְעֵת יִשְׁתֶּה זֶה הַסּוֹבֵא

עֵינָיו זַכּוּ מִסַפִּירִים

לֹא יָסֵך אֶת רַגְלָיו יֹאמַר

כִּי אָסוּר מִדִּבְרֵי סוֹפְרִים

וּלְהַבְדִּיל בֵּין הָאוֹר לַחֹשֶׁךְ

בֵּין יַיִן צָלוּל לִשְׁמָרִים

מַה שֶּׁלֹא רָאוּ עֵינָיו עֵת

חוּל עָלָיו אֵלֶּה הַמִּקְרִים

עֵת תָּחֵל הַמַּשְׂאֵת לַעְלוֹת

וּבְעֵת יַעְלוּ הַקִּטּוּרִים

עֵת יִתְנַבֵּא תּוֹך הַבַּיִת

עֵת יַסְפִּיק וּבְיַלְדֵי נָכְרִים

עֵת יִתְכַּוֵּן לָחוּד חִידוֹת

וּלְבָאֵר סוֹדוֹת נִסְתָּרִים

עֵת הַמּוּמִים הַנַּעְלָמִים

תִּהְיֶיןָ עַל פִּיו נִזְכָּרִים

עֵת יַחְשֹׂף סוֹדוֹת יוֹלַדְתּוֹ

וּבְעֵת יַחְשֹׂף סוֹדוֹת הוֹרִים

עֵת עוֹלָלִים וּמְהֹלָלִים

יָסֹבּוּ עָלָיו כִּדְבוֹרִים

עֵת יָשׁוּב נִלְעָג בִּלְשׁוֹנוֹ

וּבְעֵת לִבּוֹ לֵב נִמְהָרִים

עֵת יֵלֵךְ חַתְחַתִּים בַּדֶּ-

רֶךְ עֵת יִפֹּל תּוֹך מִכְמוֹרִים

עֵת יִשָּׂאוּהוּ עַל כָּתֵף

דִּמְיוֹן מֵת יוּבַל לִקְבָרִים

עֵת נָשָׁיו תֶּלְיוֹ וּמְבוּשָׁיו

תִּרְחַצְנָה עִם מַיִם קָרִים

עֵת יִשְׁכַּב אַרְצָה עַל עָרְפּוֹ

לִדְמוּת מֵתִים וּמְדֻקָּרִים

לֹא רָאָה שִׂמְחָה מִיָּמָיו

לֹא לָקַח מֶנּוּ מוּסָרִים

שָׁכַח הַלֵּץ לֵץ הַיַּיִן

זָנַח כִּי כָל צוֹפָיו עִוְרִים

כָּל שֹׁגִים בּוֹ לֹא יֶחְכָּמוּ

סוּרָיו הֵמָּה הַגִּבּוֹרִים

אֶתְמַהּ עַל רוֹאֵי מִקְרֵי הַ-

שִּׁכְּרוּת מִיַּלְדֵי הָעִבְרִים

אֵיךְ לֹא יָשׁוּבוּ בִרְאוֹתָם

מִקְרָיו פָּרוּשִׁים וּנְזִירִים

אוֹ אֵיךְ לֹא יָשׁוּבוּ אַל דָּת

אֻמַּת יִשְׁמָעאֵל אוֹ הַגְרִים

זִכְרוּ נֹחַ מֶה הָיָה לוֹ

עֵת נָטַע כֶּרֶם כִּבְעָרִים

שָׁתָה וַיִּשְׁכָּר וַיִּתְגָּל

נִדְמָה אֶל חַיּוֹת הַיְּעָרִים

מֶה עָשׂוּ לוֹ חָם וּכְנַעַן

אוֹיְבֵי הָאֵל בִּיקַר כָּרִים

זִכְרוּ אֶת לוֹט הוֹלֵךְ עִם אַבְרָם

וּבְנוֹתָיו בִּמְחִלּוֹת צוּרִים

עֵת הִשְׁקוּהוּ יַיִן עֵת מִ-

מֶּנּוּ יָלָדוּ מַמְזֵרִים

אַשְׁרָיו אִם לֹא שָׁתָה יַיִן

אַשְׁרָיו אִם שָׁתָה מֵי נִמְרִים

זִכְרוּ יוֹנָדָב בֶּן רֵכָב

וּבְנֵי בָנָיו הַמֻּבְחָרִים

זִכְרוּ דָנִיאֵל רָד עִם אֵל

וּבְכָל חֶזְיוֹן רֹאשׁ לַפּוֹתְרִים

מָה רַב טוּבָם כִּי נִזְהָרוּ

מַה נֶהְדָּרִים מַה נִּפְאָרִים

זֹאת בִּרְאוֹתִי אָז הִכַּרְתִּי

אֶת מַעְשֵׂי אֵל חַי הַיְשָׁרִים

אָז בַּעְשׂוֹתוֹ בִתְבוּנָתוֹ

מִמַּיִם יַמִּים וּנְהָרִים

כִּי לֹא הָיוּ מִיֵּיִן חֶלְמוֹן

מֵי בוֹרוֹת שִׁיחִין וִיאוֹרִים.




כִּשְׁמֹעַ הַשָּׂר הַשִּׁיר הַנִּמְרָץ / קָם מִמְּקוֹמוֹ וְלִקְרָאתִי רָץ / וְחִבְּקַנִי וּנְשָׁקַנִי / וְכַאֲשֶׁר יִדְבַּק הָאֵזוֹר אֶל מָתְנֵי אִישׁ אֵלָיו הִדְבִּיקָנִי / וַיֹּאמֶר אֵלָי הֵיטַבְתָּ לִרְאוֹת / הַיּוֹם עָשִׂיתָ גְבוּרוֹת נִפְלָאוֹת /

וַיְהִי כַאֲשֶׁר שָׁמְעוּ הַסְּגָנִים וְהַשָּׂרִים / הַשִׁירִים הַיְּקָרִים / כָּרְעוּ עַל בִּרְכֵּיהֶם בְּיָדְעָם כִּי שֶׁקֶר נִסְכֵּיהֶם / וּבְהַכִּירָם עֵרֶךְ מַעֲלָתִי עַל עֶרְכֵּיהֶם / וַיִּהְיוּ בְתוֹך הָעָם אֲנָשִׁים מַשְׂכִּילִים וּבְקַנֹּאתָם עַל מַעֲלָתִי עָמְדוּ כְרֶגַע נִבְהָלִים / הֶעְתִיקוּ מֵהֵם מִלִּים / אַחֲרֵי כֵן חָשְׁבוּ לְגַבֵּר חֲיָלִים / וְהֵחֵלוּ לָחוּד חִידוֹת וְלָשֵׂאת מְשָׁלִים / וְעַל הַיַּיִן אָמְרוּ הֲתֻלִים / הֲבֵל הֲבָלִים / וּבְחָשְׁבָם כִּי אֵלַי בְּעִנְיַן הַשִּׁיר יִדְמוּ / מִבְּלִי הַדַּעַת נִדְמֹה נִדְמוּ /

אָז לָבְשָׁה אוֹתִי רוּחַ קְנָאוֹת / וְרָצִיתִי לְהַרְאוֹת עַל מַעֲלָתִי מוֹפֵת וָאוֹת / וּלְהַשְׁמִיעָם נִפְלָאוֹת וְדִבְרֵי נְבוּאוֹת / וָאֹמַר אֲלֵיהֶם אַחֲרֵי אֲשֶׁר מִן הַשִּׁיר הַהוּא שְׂבַעְתֶּם / וּלְחַבֵּר עוֹד שִׁירִים יְגַעְתֶּם וְהוֹגַעְתֶּם / אֲנִי אַשְׂבִּיעַ הַיּוֹם רַעְבוֹן שִׂכְלְכֶם אִם תֹּאבוּ וּשְׁמַעְתֶּם / וְעַל הַיַּיִן אָחוּד חִידוֹת וְדִבְרֵי מְשָׁלִים / וְאוֹצִיא מִצּוּר הַחַלָּמִישׁ מַיִם קָרִים זָרִים נוֹזְלִים/

וַיַּעֲנוּ כָל הָעָם יַחְדָּו וַיֹּאמְרוּ רָאֹה רָאִינוּ כִּי אַתָּה לְבַדְּךָ תִּשָּׂא הוֹד הַשִּׁיר וְהַמְּלִיצָה בִשְׁלֵמוּת / דַּבֵּר אַתָּה עִמָּנוּ דִבְרֵי נְעִימוּת / וְאַל יְדַבֵּר עִמָּנוּ אָחֵר פֶּן נָמוּת /

וַיְהִי כִרְאוֹתִי כִּי הָעָם לְקוֹלִי מַאֲזִינִים / אָמַרְתִּי שִׁמְעוּ חֲכָמִים אֵלַי וְרוֹזְנִים / אֲבוֹתֵינוּ סִפְּרוּ לָנוּ זֶה יָמִים וְשָׁנִים / כִּי הָיוּ בְעִיר חֶשְׁבּוֹן בְּיָמִים קַדְמוֹנִים / שְמֵנִים זְקֵנִים / הָיוּ חֲכָמִים מְחֻכָּמִים / וּבָנוּ מִקְדַּש הַחָכְמָה כְּמוֹ רָמִים הָיוּ בַעֲלֵי מְלִיצָה חֲזָקָה וַעֲמֻּקָה / וּמִדְּבָש וְנֹפֶת צוּפִים מְתוּקָה / וָיְהִי הַיּוֹם וְהִנֵּה הִגִּיעַ יוֹם שִׂמְחַת תּוֹרָה / וּשְׁמוֹנִים אִישׁ הַזְּקֵנִים נֶאֶסְפוּ בְּבֵית נְשִׂיא אַרְצָם לְחֶבְרָה / וְשָׁם יָנְקוּ שְׁדֵי הַזְּמוֹרָה / וַיְהִי הֵמָּה מֵיטִיבִים אֶת לִבָּם / רוּחַ הַמְּלִיצָה נוֹסְסָה בָם / וַיֹּאמְרוּ אִישׁ אֶל רֵעֵהוּ נְדַבֵּר הַיּוֹם מֵאֹדוֹת הַיָּיִן / הַנּוֹתֵן רוֹזְנִים לְאָיִן / וְכָל אֶחָד מִמֶּנּוּ יֹאמַר כַּטּוֹב וְכַיָּשָׁר בְּעֵינָיו / עַל הַיַּיִן דָּבָר דָּבוּר עַל אָפְנָיו/ וַיֵּאוֹתוּ הַזְּקֵנִים לְדַבֵּר עַל הַיַּיִן צַחוֹת / לְהַשְׂבִּיעַ הַנְּפָשִׁים בְּצַחְצָחוֹת / וּלְחַדֵּש עַל נְשִׂיאָם שׂבַע שְׂמָחוֹת /

וַיִּגַּש הָרִאשׁוֹן וַיֹּאמַר אוֹמֵר הַיָּיִן בְּאָשְׁרִי / כִּי הִנֵּה שְׁלֹמֹה שֶׁחָשַׁב לִמְחוֹת זִכְרִי / אָמַר בְּסִפְרוֹ תַּרְתִּי בְלִבִּי לִמְשׁוֹךְ בַּיַּיִן אֶת בְּשָׂרִי /

וַיִּגַּש הַשֵּׁנִי וַיֹּאמַר אוֹמֵר הַיָּיִן אֲנִי בְיַלְדוּתִי יֶלֶד שַׁעֲשׁוּעִים / וּבְבַחֲרוּתִי מַעֲשֶׂה צַעֲצוּעִים / וּבְזִקְנוּתִי לְמַעְלָה מִשִּׁבְעָה רְקִיעִים /

וַיִּגַּשׁ הַשְּׁלִישִׁי וַיֹּאמַר אוֹמֵר הַיָּיִן לוּלֵי הֱיוֹתִי בֵּן מְנַבֵּל אָב / כִּי נֹחַ אָב בִּי הֻכְאָב / אֶתְפָּאֵר כִּי אֲנִי הוֹלַדְתִּי עַמּוֹן וּמוֹאָב /

וַיִּגַּשׁ הָרְבִיעִי וַיֹּאמַר אוֹמֵר הַיָּיִן אֵין לִתְמֹהַּ אִם יְקִיאֵנִי יְלוּד אִשָּׁה / בַּעַל נֶפֶשׁ קְדוֹשָׁה / אַחֲרֵי שֶׁהַדָּג הֵקִיא אֶת יוֹנָה עַל הַיַּבָּשָׁה /

וַיִּגַּשׁ הַחֲמִישִׁי וַיֹּאמַר אוֹמֵר הַיָּיִן כְּשֶׁיִּשָּׂאֵנִי אִישׁ בִּמְרוֹם רֹאשׁוֹ אֶתְבַּצֵּר / וְשָׁם אֶבְנֶה בַיִת אוֹ חָצֵר / וְיִהְיֶה בְּיָדַי כַּחֹמֶר בְּיַד הַיּוֹצֵר /

וַיִּגַּשׁ הַשִּׁשִׁי וַיֹּאמַר אוֹמֵר הַיָּיִן אֲנִי שׁוֹתֶה בְכֵפֶל רִסְנִי אֶכְבּוֹש / וּלְיָפְיִי עָרוֹם יִרְאֵנִי וְלֹא אֵבוֹשׁ / וְלָזֶה לֹא אֶשְׁאַל בֶּגֶד לִלְבּוֹשׁ /

וַיִּגַּשׁ הַשְּבִיעִי וַיֹּאמַר אוֹמֵר הַיָּיִן מָלַכְתִּי עַל תֵּבֶל מִימוֹת נֹחַ / אֶתֶּן לַיָּעֵף וּלְאֵין אוֹנִי פְּקַח-כֹּחַ / וְכָל אֲשֶׁר אַזְקִין אוֹסִיף כֹּחַ /

וַיִּגַּשׁ הַשְּמִינִי וַיֹּאמַר אוֹמֵר הַיָּיִן חָדְלוּ פְרָזוֹן בַּתֵּבֵל עַד שַׁקַּמְתִּי / וְאִם בְּתַחְתִּיוֹת הָאָרֶץ רֻקַּמְתִּי / לֹא הֶלְאַנִי אִישׁ כִּי אִם בּוֹ נִקַּמְתִּי /

וַיִּגַּשׁ הַתְּשִׁיעִי וַיֹּאמַר אוֹמֵר הַיָּיִן בָּרָאתִי מַשְׁחִית לְחַבֵּל / כִּי אִישׁ לֹא יִשְׁתֵּנִי יִתְאַבֵּל / וַאֲשֶׁר יִשְׁתֵּנִי יִהְיֶה כְּשֹׁכֵב בְּרֹאשׁ חִבֵּל /

וַיִּגַּשׁ הָעֲשִׂירִי וַיֹּאמַר אוֹמֵר הַיָּיִן חָכָם אֲנִי מִדָּנִיאֵל / אֲשֶׁר לֹא חָפֵץ בִּי לְהִתְגָּאֵל / וּלְגַלּוֹת כָּל סוֹד אֲנִי קָרוֹב מִמֶּנּוּ וְגוֹאֵל /

וַיִּגַּשׁ הָאַחַד עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן אֲנִי אֲגוֹבֵב הַבְּתוּלוֹת / וְאֶתְּנֵם בְּיַד הַחֹשְׁקִים כְּעֵדֶר הָרְחֵלוֹת / וְאָשִׂים כָּל הַנִּסְתָּרוֹת מְגֻלּוֹת /

וַיִּגַּשׁ הַשְּׁנֵים עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן אֲנִי תִפְאֶרֶת בַּחוּרִים / וּבִי תִטְחַן לְאַחֵר אֵשֶׁת נְעוּרִים / וְעָלֶיהָ יִכְרְעוּן אֲחֵרִים /

וַיִּגַּשׁ הַשְּׁלֹשָׁה עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן אֲנִי נוֹרָא וְאָיוֹם / וְאַמְתִּיק לַעֲנַת אוֹרְרֵי יוֹם / וְאֶהְיֶה פִּדְיוֹם לְפִיד יוֹם /

וַיִּגַּשׁ הָאַרְבָּעָה עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן חָכְמָה וְכָבוֹד יָסַפְתִּי / וְכֶאֱסֹף בֵיצִים עֲזֻבוֹת כָּל הָאָרֶץ אֲנִי אָסָפְתִּי / וְלֹא הָיְתָה תַעֲלֻמָה אֲשֶׁר לֹא חָשָׂפְתִּי /

וַיִּגַּשׁ הַחֲמִּשָׁה עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן אֲנִי אַשְׁקֶה הַעַמִּים סַף רַעַל / וּבִלְעָדַי לֹא כָרְעוּ הַצְּבִיּוֹת לַבַּעַל / וְלֹא אָמְרוּ יְהִי אֱלֹהָיו עִמּוֹ וְיָעַל /

וַיִּגַּשׁ הַשִּׂשָׁה עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן עֹז אָזַרְתִּי / וּבְעֵת לְבָדָד אֶשְׁכֹּן בֶּטַח לִבִּי וְנֶעֱזָרְתִּי / צָפוּ מַיִם עַל רֹאשִׁי וְאָמַרְתִּי נִגְזָרְתִּי /

וַיִּגַּשׁ הַשִּׁבְָעה עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן בָּרוּךְ הָאִישׁ אֲשֶׁר לֵידָתִי בִּשָּׂר / כְּדִין אֶהְיֶה לַתֵּבֵל לְרֹאשׁ וּלְשָׂר / כִּי הַמַּיִם הִשְׁחִיתוּ כָל בָּשָׂר /

וַיִּגַּשׁ הַשְּׁמוֹנָה עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן הִנְנִי בִגְאוֹנִי נִשְׁבָּע / כִּי מִי שֶׁיָּצִיף מַיִם עַל רֹאשִׁי נֶפֶשׁ יִקְבָּע / וְדָמִי בְרֹאשׁוֹ יִהְיֶה אִם אֶטְבָּע /

וַיִּגַּשׁ הַתִּשְׁעָה עָשָׂר וַיֹּאמַר אוֹמֵר הַיָּיִן הַמַּיִם בְּסוּפָה וּבִשְׂעָרָה דַרְכָּם וּמִפַּחְדָּם אִישִׁים יִרְהוּ / וְהַמִּנְהָג כְּמִנְהַג יֵהוּא / בֶּאֱלָהָא דְסָנֵינָא לְהוֹא /

וַיִּגַּשׁ הָעֶשְׂרִים וַיֹּאמַר אוֹמֵר הַיָּיִן לַמַּיִם לוּ הִמְטִיר הָאֵל אוֹתִי בְאוֹתָם הָעִתִּים / עַל אַנְשֵׁי הַמַּבּוּל כָּמוֹךָ לֹא הֵמַתִּים / אִמַֹצְתִּים אַף עֲזָרְתִּים /

וַיִּגַּשׁ הָאֶחָד ועֶשְׂרִים וַיֹּאמַר אוֹמֵר הַיָּיִן אֲנִי אַרְנִין לְשׁוֹן אִלֵּם / וְאִם יִשְׁתּוּנִי הַמֵּתִים אֲצִילֵם / מִיַּד שְׁאוֹל אֶפְדֵּם מִמָּוֶת אֶגְאָלֵם /

וַיִּגַּשׁ הַשְּׁמֹנָה וְשִׁבְעִים וַיֹּאמַר אוֹמֵר הַיָּיִן אִם יִתָּלוּ הָאֲנָשִׁים כָּאַשְׁכֹּלוֹת / אֲנִי שַׂמְתִּי אִמּוֹתָם שַׁכּוּלוֹת / וְעוֹלְלֵיהַם כַּגֶּפֶן הָשֵׁב יָדִי כְּבוֹצֵר עַל סַלְסִלוֹת /

וַיִּגַּשׁ הַתִּשְׁעָה וְשִׁבְעִים וַיֹּאמַר אוֹמֵר הַיָּיִן בִּפָנַי שֹׂבַע שְׂמָחוֹת / וּמִפָּנַי יָנוּסוּ יָגוֹן וַאֲנָחוֹת / וְכַאֲשֶׁר יִשְׁתּוּנִי הָעוֹפוֹת יִלְמְדוּ לְדַבֵּר צַחוֹת /

וַיִּגַּשׁ הַשְׁמֹנִים וַיֹּאמַר אוֹמֵר הַיָּיֵן אִם אַעֲמִיד יוֹם אוֹ יוֹמַיִם / וְלֹא אֲתַקֵּן אֶת אֲשֶׁר עִוְּתוּ הַמַּיִם / יִגְוַע כָּל בָּשָׂר אֲשֶׁר תַּחַת הַשָּׁמַיִם /

וַיְּהִי כַּהֲתִמִּי הַמְּשָׁלִים הַנִּזְכָּרִים / אָמַרְתִּי אֶל הַסְּגָנִים וְהַחוֹרִים / הִנֵּה שְׁמַעְתֶּם דִּבְרֵי שְׁמֹנִים זְקֵנִים / כְּפִי מַה שֶּׁסִּפְּרוּ לָנוּ אֲבוֹתֵינוּ הַקַּדְמוֹנִים/ וְכַאֲשֶׁר הִגִּיעוּ לְאָזְנַי דִּבְרֵיהֶם הָעֲרֵביִם / פְּנֵיהֶם פְּנֵי לֶהָבִים / שַׂמְתִּי עַל לוּחַ לִבִּי כְתוּבִים / וְלָעַד בְּצוּר שִׂכְלִי חֲצוּבִים / לְמַעַן יַעַמְדוּ יָמִים רַבִּים/ וְכָלַלְתִּי עִנְיָנֵיהֶם בְּשִׁיר נֶחְמָד / אֲשֶׁר כָּל הַבָּא אֵלָיו וְעָמָד / קִבַּצְתִּיו מִפְּנִינֵי הַמַּאֲמָרִים / וְיִסַּדְתִּיו בְּסַפִּירִים / וְהִנֵּה הַשִּׁיר אֲשֶׁר עַל מְלִיצַת הַזְּקֵנִים חִבַּרְתִּי / וּבִמְחוּגַת שִׂכְלִי אוֹתוֹ תֵאַרְתִּי / כַּאֲשֶׁר עַל טוּב צַוָּארָם עָבַרְתִּי /

יֹאמַר יַיִן עֹז אָזַרְתִּי

לַפְּגָרִים נֶפֶשׁ הֶחְזַרְתִּי

אֵין לִתְמֹהַּ אִם לִי תִכְרַע

כָל בֶּרֶךְ עַם זוּ יָצַרְתִּי

כִּי רַבִּים מִמָּוֶת אֶגְאַל

וִימֵי רַבִּים כֵּן הִקְצַרְתִּי

זֶה חַסְדִּי עִם רֵעַי כִּי כָל

מוֹצְאִי יֹאמַר אַךְ עָשַׁרְתִּי

וּבְעֵת מִמֶּנִּי יִשְׂבַּע אִישׁ

יֹאמַר דָּבָר לֹא חָסַרְתִּי

מִלֵּב אָסִיר תּוּגָה אָסִיר

שִׂמְחָה וָגִיל לֹא עוֹרַרְתִּי

וּלְהַגִּיעוֹ לִמְחוֹז חֶפְצוֹ

בִּמְשׁוֹטוֹת שִׂכְלִי חָתַרְתִּי

הָיוּ שׁוֹכְנֵי עָפָר יוֹם עַל

לִבָּם טַל אוֹרוֹת הִמְטַרְתִּי

אִישׁ לֹא שָׁתָה יַיִן יִיעָף

יֹאמַר כֹּחַ לֹא עָצַרְתִּי

הָיִיתִי שִׂמְחַת הַיְּצוּרִים

וּשְׁנַת טוֹבָתָם עִטַּרְתִּי

מִן לֵדוּתִי טַל יַלְדּותִי

אַזְקִין אָז תִּגְדַּל תִּפְאָרְתִּי

מִיּוֹם נֹחַ נָטַע כֶּרֶם

לִמְלֹך עַל תֶּבֶל נִכְתַּרְתִּי

בִּי יִתְהַלֵּל כָּל מִתְהַלֵּל

עַל כָּל תַּעְנוּגִים אֻשַּרְתִּי

וּלְהַרְאוֹת הָעַמִּים יָפְיִי

עַל גִּלְדִּי שַׂק לֹא תָפַרְתִּי

עָרוֹם יִרְאוּנִי לּא אֵבוֹשׁ

לֹא חָפַרְתִּי כִּי נֶהְדַּרְתִּי

כָּל אִישׁ יַחְשֹׁק אֶת פִּי לִנְשׁוֹק

וּבְעֵינֵי רוֹאַי יָקַרְתִּי

שָׂר עִם סוֹחֵר פָּנַי שׁוֹחֵר

וּפְנֵי אַחֵר לֹא שִׁחַרְתִּי

אָסִיר שָׂפָה אֶל נֶאְמָנִים

וּלְשׁוֹן נִלְעָגִים הִגְבַּרְתִּי

אַפְרִיד לִבּוֹת הַנִּצְמָדִים

חֹשֵׁק עִם עָפְרָה הֶחְבַּרְתִּי

וּבְתוּלוֹת נוֹבַבְתִּי וּבְיַד

הַחֹשְׁקִים אוֹתָם הִסְגַּרְתִּי

כָּל הַחֹשְׁקִים לִי זִבֵּחוּ

אַף קִטֵּרוּ עַל מִכְמַרְתִּי

כִּי בָהֵמָּה שָׁמֵן חֶלְקָם

וּלְהַמְצִיא חֻקָּם מִהַרְתִּי

כָּל דֶּרֶךְ אֶדְרֹךְ נָגַהּ אוֹר

יָקוּם-לִי אֹמֶר גָּזַרְתִּי

אִישִׁים חָשְׁבוּ לִקְבֹּר אוֹתִי

זֶה כַּמָּה אוֹתָם קָבַרְתִּי

אוֹתִי דִמּוּ לַהְרֹג וּבְעֹז

יָדַי עַל כֻּלָּם גָּבַרְתִּי

אֶת שֶׁהִתִּירוּ אָסַרְתִּי

גַּם אֶת שֶׁאָסְרוּ הִתַּרְתִּי

וּבְעֵת שָׂחְקּו אָז בָּכִיתִי

בְעֵת בָּכוּ אָז שׁוֹרַרְתִּי

וּבְעֵת פָּרְצוּ אָז בָּנִיתִי

וּבְעֵת בָּנוּ אָז סָתַרְתִּי

וּבְעֵת עָקְרוּ אָז נָטַעְתִּי

וּבְעֵת נַטְעוּ אָז עָקַרְתִּי

וּבְעֵת דִּבְּרוּ הֶחְשֵׁיתִי

וּבְעֵת הֶחְשׁוּ אָז דִּבַּרְתִּי

וּבְעֵת שָׁמְרוּ הִשְׁלַכְתִּי

וּבְעֵת הִשְׁלִיכוּ שָׁמַרְתִּי

קֶשֶׁר אַמִּיץ אַתִּיר וּבְרִית

כָּרוּת בֵּין רֵעִים הִפַרְתִּי

שׁוֹשֵׁיתִי עִיר אַף מַמְלָכוֹת

וּגְבוּלוֹת עַמִּים הֵסַרְתִּי

וִיסוֹדֵי עִיר גִּלִּיתִי אַף

אַבְנֵי עִיר לַגַּיְא הִגַּרְתִּי

כִּצְבָא רוֹם בִּזְבוּל נִשְׁאַרְתִּי

וּלְבַדִּי נָבִיא נוֹתַרְתִּי

קָדְרוּ שֶׁמֶש וּלְבָנָה אַךְ

אוֹרִי יָהֵל לֹא קָדַרְתִּי

בֵּין אִם יִהְיֶה בֵּין לֹא יִהְיֶה

בִּקְהַל רָב צֶדֶק בִּשַּׂרְתִּי

בָּטְחוּ אִישִׁים בִּנְדָרַי אַךְ

לֹא קִיַּמְתִּי שֶׁנָּדַרְתִּי

כֶּרֶם לֹא נָטַעְתִּי אֶבְצֹר

שָׂדֶה לֹא אֶזְרַע קָצַרְתִּי

עָלַי פָּצוּ פִיהֶם חַכְמֵי

עוֹלָם אָמְרוּ כִּי סָרַרְתִּי

לְעַוֵּר אֶת עֵינַי חָשְׁבוּ

אָכֵן עֵינֵיהֶם עִוַּרְתִּי

וּבְחָשְׁבָם כִּי אוֹתִי שַּחוּ

עַל חוּג רֹאשָׁם נִתְבַּצַּרְתִּי

חָשַׂפְתִּי מַצְפּוּן לִבּותָם

וּבְקִרְבָּם סוֹד לֹא הִשְׁאַרְתִּי

בִּזּוּנִי יַעַן נִמְכַּרְתִּי

שָׁכְחוּ כִּי אוֹתָם מָכַרְתִּי

אַשְׁכִּיר יוֹם יוֹם אִישִׁים אֵין קֵץ

אַךְ מִיָּמַי לֹא שָׁכַרְתִּי

וּבְרַגְלֵיהֶם אִם נִרְמַסְתִּי

אֶזְכֹּר כִּי רַבִּיםִ יִסַּרְתִּי

אִישִׁים הוֹצִיאוּ דִבַּתִי

נָפְלוּ תוֹך שַׁחַת חָפַרְתִּי

וּבְעֵת כִּי הֵם עָלַי קָשְׁרוּ

תּוֹך לִבּוֹתָם אָז נִקְשַׁרְתִּי

לֹא הָיוּ לָהֶם יָדַיִם

יוֹם פָּנַי מֵהֶם הִסְתַּרְתִּי

שִׁבַּרְתִּי רָאשֵׁי תַנִּינִים

וּבְעֻזִּי הַיָּם פּוֹרַרְתִּי

לֹא נִמְלְטוּ מִיַּד רִשְׁתִּי

הֵם חָלְמוּ אַךְ פָּתַרְתִּי

אֵימִים הִשְׁפַּלְתִּי עַד אָרֶץ

שָׂרִים בֶּעָפָר עִפַּרְתִּי

וּבְשִׁבְתִּי בָדָד דּוּמִיָּה

בָּטַח לִבִּי אַף נֶעְזַרְתִּי

אָמְנָם עֵת צָפוּ מַיִם עַל

רֹאשִׁי אָמַרְתִּי נִגְזָרְתִּי.

וַיְהִי כִּשְׁמֹעַ הַשָּׂר וְהַקְּהִלּוֹת / קוֹל הַשִּׁיר יְחוֹלֵל אַיָּלוֹת / בֵּרְכוּ לְנֶאְדָּר בַּקֹּדֶשׁ נוֹרָא תְהִלּוֹת / אֲשֶׁר כָּכָה לוֹ בְעוֹלָמוֹ / וְכָל הָעָם שָׁב לִמְקוֹמוֹ.
