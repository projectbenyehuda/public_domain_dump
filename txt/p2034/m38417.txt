


	
	[אֵלִיָּהוּ הַנָּבִיא עַד־מָתַי אַתָּה תְאַחֵר (זֶמֶר)]
	



 [מספּוֹנְטוֹ באיטליה הדרומית, חי במחצית השניה של המאה הי"א]




זֶמֶר לְמוֹצָאֵי שַׁבָּת



אֵלִיָּהוּ הַנָּבִיא עַד-מָתַי אַתָּה תְאַחֵר



וּבֵן יַקִּיר עֵינָיו תְּלוּיוֹת וּלְבִיאָתְךָ הוּא מְשַׁחֵר



בֹּא בִמְהֵרָה בֹּא בִמְהֵרָה בִּמְהֵרָה וְאַל-תְּאַחֵר



בֹּא יָבֹא בֹּא יָבֹא בֹּא יָבֹא לֹא יְאַחֵר.



עֵת לַחֲשׁוֹת וְעֵת לְדַבֵּר וְעַד-מָתַי אַתָּה



תַחֲרִישׁ



וְהָאֶלֶף כְּבָר עָבְרוּ וְעַצְמְךָ תַּעֲשֶׂה כְמַחֲרִישׁ



וְהַנָּחָשׁ הוּא נוֹשֵׁךְ וּכְצִפְעֹנִי יַפְרִישׁ.



נִכְבָּדוֹת מְדֻבָּר בְּךָ בְּמוֹצָאֵי יוֹם נָפְשָׁם



וְגַם בַּבְּרִית לְךָ מְתַקְּנִים מָקוֹם לָשֶׁבֶת שָׁם



וּפָנֶיךָ לָמָּה תַסְתִּיר וְלֹא תִרְאֶה צָרַת נַפְשָׁם.



נִין עַמְרָמִי בְּעֵת הָלַךְ וַיְמָאֵן וְלֹא רָצָה



“גּוֹאֵל אַחֲרוֹן אִם אֶהְיֶה אָז אָרִיץ בִּמְרוּצָה”



וְאַף עַל-פִּי שֶׁלֹּא זָכָה לֹא מָנַע לִגְדֹּר פִּרְצָה.



הֲלֹא תִזְכֹּר בְּיוֹם עֲמָדְךָ עַל-קְבוּרַת בְּשַׂר



חֲמוֹר



וּבָא שׁוֹאֵל וּשְׁאֵלְךָ “אֵימַת קָא אָתֵי מוֹר”



תְּשׁוּבָתְךָ הָיְתָה-לוֹ "הַיּוֹם אִם תִּשְׁמְעוּ בְּקוֹל צְרוֹר



הַמּוֹר".



כְּבֶן-חַיִל אֵזוֹר חֲלֹץ וְתֵלֵךְ אֶל-בֶּן-יִשָׁי



וְסַפֵּר לוֹ מִלּוֹקִים וְקוֹבְלִים לָאֵל בַּחֲשָׁי



אוּלַי יְמַהֵר בִּיאָתוֹ וְיַצִּילִי מִיַּד נוֹשָׁי.



הֲלֹא לְצָרְפִית כְּשֻׁלַּחְתָּ הָיִיתָ לָהּ



לְמַשְׁעֵנָה



כַּד וְצַפַּחַת לֹא חָסְרוּ בַּעֲבוּר עוּגָה קְטַנָּה



וְעַל-דִּבּוּר “בָּאתָ אֵלַי” הֲשִׁיבוֹתָ רוּחַ בְּנָהּ.



נֵץ עָמְרִי בְּעֵת שֻׁלַּחְתָּ לְבַקֵּר אֶת-בֵּית



חִיאֵל



עַל שֶׁבָּזָה וְגַם הִלְעִיג אֶת-רוֹעֵה יִשְׂרָאֵל



טַל וּמָטָר מָנַעְתָּ לוֹ בִּגְזֵרַת הָאֵל.



חֲכָמִים וְתַלְמִידֵיהֶם הַמְבִינִים בְּדַת סוֹדָם



זֹאת הַכֹּל הֵם יוֹדְעִים וְגַם מַסֹּרֶת בְּיָדָם



כִּי הַקֵּץ לֹא יָבֹא עַל-דִּבּוּר בָּשָׂר וָדָם.



קוֹוֶיךָ בְּךָ שְׁעֵנִים וּבְפִיהֶם סְדוּרָה מִלָּה



"לָמָּה הוּא מִתְרַשֵּׁל מִלִּזְעֹק בַּתְּפִלָּה



אֶפְשַׁר עַל-פְּגִיעָתוֹ הָיְתָה מַגַּעַת גְּאֻלָּה".



בֹּא בִמְהֵרָה בֹּא בִמְהֵרָה בִּמְהֵרָה וְאַל-תְּאַחֵר

בֹּא יָבֹא בֹּא יָבֹא בֹּא יָבֹא לֹא יְאַחֵר.
