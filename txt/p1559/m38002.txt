


	
	ויהי בימים הרבים ההם
	








גְזֵרוֹת קַסְטִילְיָה

וַיְהִי בַיָּמִים הָרַבִּים הָהֵם / בִּהְיוֹתֵנוּ נִלְחָצִים מֵהַגּוֹיִם וּמַלְכֵיהֶם / בְּגָלוּת הַחֵל הַזֶּה הָרַע / אֲשֶׁר עֲטֶרֶת רֹאשֵׁנוּ פָרַע / כִּי אֱדוֹם הִכְבִּיד מְאֹד בְּעֻלֵּנוּ / חֲזַק וְנִתְחַזֵּק בְּעַד עַמֵּנוּ / עַד כִּי הָיִינוּ נִרְפִּים / כָּל הַיּוֹם נִגָּפִים / כִּי כָל שַׂר וּמוֹשֵׁל בְּכִלָּיוֹן / יְנוֹפֵף
יָדוֹ עַל הַר צִיּוֹן / עַד שֶׁהָיוּ לִשְׁרֵקָה / מֵרֹב צָרָה וְצוּקָה / וְאֹרֶךְ הַגָּלוּת אֲשֶׁר בָּם אָרַךְ / וְהָאוֹיֵב עַל שִׁכְמָם דָּרַךְ / וּמַצִּיק בְּכָל עֵת וּבְכָל שָׁעָה / עַד שֶׁקֶּרֶן עֻזּוֹ נִגְדָּעָה / וּבְרֹב הַמְּצוּקוֹת וְהַיְגוֹנוֹת / נִשְׁכְּחוּ הַצָּרוֹת הָרִאשׁוֹנוֹת / וְקוֹרוֹת הַזְּמָן אֲשֶׁר דְּחָקוּנוּ / עַל הֶהָרִים דְּלָקוּנוּ / כִּי חָשַׁב לְהַשְׁחִית אֲנָשִׁים / עֲצוּמִים, עַם קְדוֹשִׁים / וּבְהָלְכוֹ בַדֶּרֶךְ הַזֶּה רָגִיל / נֶאֱסַף שִׂמְחָה וְגִיל / אֵיךְ אֻמָּה חֲכָמָה וְקַלָּסָה / נָפְלָה מִשָּׁמַיִם אַרְצָה /
אֲשֶׁר הִכְאִיב אוֹיֵב אֶת נַפְשֵׁנוּ / וַיָּסַר עֲטֶרֶת רֹאשֵׁנוּ / עַד כִּי הָיִינוּ לְחֶרְפָּה וּלְבִזָּה / כָּל הַיּוֹם לְרָגְזָה / כִּי לְכָל הַדֶּרֶךְ אֲשֶׁר הַיְהוּדִי יִפְנֶה לְבָבוֹ / בְּחִצִּים וּבְקֶשֶׁת יָבוֹא / וְכָל הַיּוֹם הוֹלֵךְ נִרְדָּף / כְּקַשׁ נִדָּף / וְלֹא מָצְאוּ מָנוֹחַ / וַיֵּלְכוּ בְלֹא כֹחַ / עַד כִּי בְרֹב
הַקּוֹרוֹת / וַיַּחֲנוּ עַל עֵי״ן חֲרֹ״ד / וּבְרֹב עֲמָלָם וְעָנְיֵיהֶם / קָצוּ בְחַיֵּיהֶם / כִּי אֵיךְ יִחְיֶה הָאָדָם בֵּין גּוֹי עַז פָּנִים / אֲשֶׁר חִצָּיו שְׁנוּנִים / לְאַבֵּד וּלְהַשְׁמִיד / לַיְּהוּדִים
תָּמִיד / עַד שֶׁנָּדְדוּ מִקּוֹל הָמוֹן / מִנַּחַל אַרְנוֹ״ן עַל הַר חֶרְמוֹן / וְתִפְאַרְתָּם נִשְׁפָּלָה / וְדַרְכָּם נִבְהָלָה / וְכָל צְבִי מֻדָּח מְעוּף צוּקָה וָאֲפֵלָה מְנֻדָּח / וְאָמְנָם בִּהְיוֹת אֲבוֹתֵינוּ בְּתוֹכָם / עָבְרוּ זְמַנָּם עִמָּם וְעִם מַלְכָּם / וְלֹא
דְחָקוּם בְּרֹב הַתְּלָאוֹת / כִּי י״י צְבָאוֹת הָיָה צוֹפֶה עֲלֵיהֶם / וּמַבִּיט בְּעָנְיֵיהֶם / וּמַצִּילָם מִן הַצָּרוֹת / וּמִן הַמִּקְרִים וְהַסְּעָרוֹת / וְאִם בַּזְּמָן הַהוּא אֹפֶל כִּסָּמוֹ / מֵאֶרֶץ כִּתִּים נִגְלָה לָמוֹ / וְרָאָה הָאֵל אָפְלָם / וְהָיָה לָהֶם לְאוֹר
עוֹלָם / אוֹ פִרְחֵי שְׁמַד עֲלֵיהֶם נִתְחַדְּשׁוּ / וְגַם נָשַׁף בָּהֶם וַיִּבָשׁוּ / וְיָדָם לֹא קָצְרָה וּזְכוּתָם לֹא נֶאֱסַר / כִּי י״י סָגַר פֶּה כָל שַׂר / לְבַל יְחַדְּשׁוּ גְּזֵרוֹת עֲלֵיהֶם / הַגּוֹיִם הָהֵם / אָמְנָם בַּזְּמָן הַזֶּה הָעֹל גָּדַל / עַד כִּי לִסְפֹּר חָדַל / מִכָּל צָרוֹת מְשֻׁנּוֹת / בְּכָל יוֹם עָלֵינוּ נְתוּנוֹת / הַנֶּחְמָדִים מִזָּהָב וּמִפָּז / לִשְׁלֹל שָׁלָל וְלָבֹז בַּז / כִּי גָבְרוּ מְאֹד הַמִּסִּים / וְהַנֹּגְשִׂים אָצִים / וְהַמַּס הָיָה לָאֲלָפִים וּלְמֵאוֹת / וַיִּדַּל יִשְׂרָאֵל מְאֹד / וְכָל מוֹשֵׁל יָדוֹ הָדָה / וַיִּצְעֲקוּ בְנֵי יִשְׂרָאֵל מִן הָעֲבוֹדָ״ה / וּמַלְכָּם בְּעֻלָּם מְאֹד הִרְבָּה / וַיִּרְדְּפֵם עַד חוֹבָ״ה / וְעוֹשִׂים שְׁמַד בְּעִנְיָנִים אֲחֵרִים / אֲנָשִׁים מְדָנִי״ם סוֹחֲרִי״ם / אֲשֶׁר אוֹיֵב
תְּפִלָּתָם רָכַל / וּבָא אֶל מִקְדָּשׁוֹ לְהִתְפַּלֵּל וְלֹא יוּכָל / וְכָל הַיּוֹם מְכַלּוֹת / הַמַּסּוֹת הַגְּדוֹלוֹת / כִּי רָעַת הַמַּס רַב עַל כָּל מִדָּה / הוּא יִשְׁסֶה אוֹצַר כָּל כְּלִי חֶמְדָּה / וְהוּא
יַפְשִׁיט כָּל הוֹד אִישׁ מֵעָלָיו / הָמֵק בְּשָׂרוֹ וְהוּא עוֹמֵד עַל רַגְלָיו / וְעַל כָּל הַלַּחַץ וְהַתֶּלֶא / נָתוּן בְּמַאֲסָר וּבְכֶלֶא / עַד כִּי מִשְׁחַת מֵאִישׁ מַרְאֵהוּ / וְחִשֵּׁב עִם קוֹנֵהוּ /
וּבִרְאוֹתִי כָזֶה בְּכָל עֵת וְעוֹנָה / נָשָׂאתִי עֵינַי לְאֵל שׁוֹכֵן מְעוֹנָה / כִּי קַצְתִּי בַקּוֹרוֹת הָאֵל / וַתִּקְצַר נַפְשִׁי בַעֲמַל יִשְׂרָאֵל / וְאָמַרְתִּי: ׳אָנָּא, אֱלֹהֵינוּ, אֲשֶׁר מִמִּצְרַיִם הוֹצֵאתָנוּ / וּמִבֵּית עֲבָדִים נְשָׂאתָנוּ / וַהֲבֵאתָנוּ עַל כַּנְפֵי נְשָׁרִים / לְצִיּוֹן וְקִרְיַת יְעָרִים / אֵיךְ נְתַתָּנוּ לְבִזָּה וּלְשַׁמָּה / בֵּין מִזֶּה וְשַׁמָּה / וּגְאוּלָה מִמֶּנּוּ רָחֲקָה /
אַף כִּי קָרְבָה צָרָה וְצוּקָה / וְאֵיךְ לֹא לְצָרוֹתֵינוּ תְצַפֶּה וּלְמַכְאוֹבֵינוּ תְרַפֵּא / וְאֵיךְ תִּרְצֶה בִּכְלוֹתֵנוּ / בִּיגוֹן יָמֵינוּ וּשְׁנוֹתֵינוּ / וְלֹא תְרַחֵם בְּעַמְּךָ הַפְּזוּרִים / בְּכָל
צַעַר נִמְסָרִים / וְרַבָּה עֲלֵיהֶם הַמְּצוּקָה / וְכָל מִכְשׁוֹל וּפוּקָה / וְאִם מֵאֻמָּה יַחְדְּלוּ חָדֹל / וּמַה תַּעֲשֶׂה לְשִׁמְךָ הַגָּדוֹל / וְלָמָּה שַׂמְתָּנוּ כְעֶבֶד נִבְזֶה / וּזְכֹר כִּי עַבְדְּךָ הַגּוֹי הַזֶּה / וְאַל תַּכְרִית שְׁמֵנוּ מֵעוֹלָם / יוֹדֵעַ כָּל נֶעְלָם. – – –
