


	
	קריעת ים־סוף, מתן תורה, העגל, המרגלים, המלחמות בעבר הירדן
	




וְהַמִצְרִים הָיו מְקַבְּרִים שְלֹשָה יָמִים אֶת
בְּכוֹרֵיהֶם וַיְהִי בַּיוֹם הַשְלִישִי לְצֵאת בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ
מִצְרַיִם וַיָקוּמוּ שָׂרֵי פַרְעֹה וְרַבִּים מֵעַם הָאָרֶץ וַיֹאמְרוּ
אִיש אֶל רֵעֵהוּ: הֲלֹא מֹשֶה וְאַהֲרֹן אָמְרוּ לְמַלְכֵּנוּ: “דֶרֶךְ
שְלֹשֶת יָמִים נֵלֵךְ וְזָבַחְנוּ לַיְיָ אֱלֹהֵינוּ,” ­– וְהִנֵה הַיוֹם
עָבַר הַמוֹעֵד הַזֶה וַעֲלֵיהֶם לָשוּב לַעֲבוֹדָתָם. וַיַשְכִּימוּ
בַבֹּקֶר בַּיוֹם הָרְבִיעִי כָּל שָׂרֵי פַרְעֹה וְעִמָם עַם רַב כִּשְבַע
מֵאוֹת אֶלֶף אִיש וַיֵצְאוּ חֲמוּשִים וַיִרְדְפוּ אַחֲרֵי בְנֵי
יִשְׂרַאֵל וַיַשִׂיגוּם בַּיוֹם הַחֲמִישִי עַל פִּי הַחִירוֹת.
וַיֹאמְרוּ הַמִצְּרִים אֶל בְּנֵי יִשְׂרָאֵל: הֲלֹא לִשְלֹשָה יָמִים
בִּקַשְתֶּם לָלֶכֶת הַמִדְבָּרָה לִזְבּוֹחַ לַיְיָ אֱלֹהֵיכֶם, וּמַדוּעַ
אֵינְכֶם שָבִים אֶל אֲדוֹנֵיכֶם? ­– וַיַעֲנוּ מֹשֶה וְאַהֲרֹן: יְיָ
אֱלֹהֵינוּ פָקַד עָלֵינוּ כִי לֹא נוֹסִיף עוֹד לָשוּב מִצְרָיְמָה, כִּי
אִם אֶל אֶרֶץ מְגוּרֵי אֲבוֹתֵינוּ נֵלֵךְ, אֲשֶר נִשְבַּע ה' לָתֵת
לָנוּ. וַיִרְאוּ שָׂרֵי פַרְעֹה כִּי כֵן ­– וַיִלָחֲמוּ בִבְנֵי
יִשְׂרָאֵל לַהֲשִיבָם מִצְרָיְמָה. וַתִּגְבַּר יַד יִשְׂרָאֵל עַל
הַמִצְרִים וַיַכּוּם מַכָּה רַבָּה, וְהַנִשְאָרִים נִמְלְטוּ עַל נַפְשָם
וַיָשוּבוּ מִצְרַיְמָה וַיַגִידוּ לְפַרְעהֹ, כִּי בָרַח עַם הָעִבְרִים
וְלֹא יוֹסִיפוּ לַשוּב עוֹד מִצְרָיְמָה. וַיִשְמַע פַּרְעֹה אֶת הַדָבָר
הַזֶה וַיִרְדֹף עִם כָּל חֵילוֹ אַחֲרֵי בְנֵי יִשְרָאֵל וַיַשִׂיגוּם
בַּיוֹם הַשְבִיעִי לְצֵאתָם מִמִצְרַיִם חוֹנִים עַל יָם־סוּף. וַיִשְׂאוּ
בְנֵי יִשְׂרָאֵל אֶת עֵינֵיהֶם וַיִרְאוּ וְהִנֵה פַרְעֹה וְחֵילוֹ
רוֹדְפִים אַחֲרֵיהֶם וַיִירְאוּ מְאֹד וַיִצְעֲקוּ אֶל ה‘. וַיִתְחַלְקוּ
כָל שִבְטֵי יִשְׂרָאֵל בְּדִבְרֵיהֶם לְאַרְבָּעָה רָאשִים: בְּנֵי
רְאוּבֵן, שִמְעוֹן וִישָׂכָר אָמְרוּ לְהַפִּיל אֶת נַפְשָם הַיָמָה
מִפַּחַד וּבֶהָלָה. וַיֹאמֶר אֲלֵיהֶם מֹשֶה: “אַל תִּירָאוּ!” ­– בְּנֵי
דָן, גָד וְאָשֵר אָמְרוּ לְהִתְנַפֵּל עַל הַמִצְרִים וּלְהֻמֵם,­ –
וַיאֹמֶר לָהֶם מֹשֶה: הִתְיַצְבוּ וּרְאוּ אֶת יְשוּעַת יְיָ; בְּנֵי
זְבוּלֻן, בִּנְיָמִין וְנַפְתָּלִי אָמְרוּ לָשוּב מִצְרַיְמָה – וַיֹאמֶר
אֲלֵיהֶם מֹשֶה: “אֲשֶר רְאִיתֶם אֶת מִצְרַיִם הַיוֹם, לֹא תוֹסִיפוּ
לִרְאוֹתָם עַד־עוֹלָם”; בְּנֵי יְהוּדָה, אֶפְרַיִם וּמְנַשֶה אָמְרוּ
לְהִלָחֵם בַּמִצְרִים ­– וַיֹאמֶר אֲלֵיהֶם מֹשֶה: "ה’ יִלָחֵם לָכֶם
וְאַתֶּם תַּחֲרִישוּן."

וַיֵט מֹשֶה אֶת מַטֵהוּ וַיַךְ אֶת הַיָם, עַל פִּי יְיָ,
וַיִבְקָעֵהוּ וַיִקָרַע הַיָם לִשְנֵים עָשָׂר קְרָעִים וַיַעַבְרוּ בְנֵי
יִשְׂרָאֵל בּוֹ בֶּחֱרָבָה, כַּאֲשֶר יַעֲבוֹר הָאָדָם בְּדֶרֶךְ
סְלוּלָה. וַיְהִי בְּבוֹא בְנֵי יִשְׂרָאֵל הַיָמָה וַיָבוֹאוּ גַם
פַּרְעֹה וְחֵילוֹ אַחֲרֵיהֶם ­– וַיָשוּבוּ מֵי הַיָם עֲלֵיהֶם
וַיִטָבְעוּ כֻלָם, הֵם וְרִכְבָּם וְפָרָשֵיהֶם, וְלֹא נוֹתַר מֵהֶם אִיש
זוּלָתִי פַרְעֹה אֲשֶר נִחַם אָז עַל רִשְעָתוֹ וַיַאֲמֵן בַּה‘. וַיְצַו
ה’ אֶת מַלְאָכוֹ וַיוֹצִיאֵהוּ מִבֵּין גַלֵי הַיָם וַיַשְלִיכֵהוּ
אַרְצָה נִינְוֵה וַיִמְלֹךְ עָלֶיהָ יָמִים רַבִּים.

וַיָבֹאוּ בְנֵי יִשְׂרָאֵל אֶל מִדְבַּר סִין בַּחֲמִשהָ
עָשָר יוֹם לַחדֶש הַשֵנִי לְצֵאתָם מֵאֶרֶץ מִצְרַיִם. וַיִתֵּן לָהֶם ה'
אֶת הַמָן יוֹם־יוֹם, וְאוֹתוֹ אָכְלוּ אַרְבָּעִים שָנָה כָּל יְמֵי
לֶכְתָּם בַּמִדְבָּר, עַד בּוֹאָם אֶל גְבוּל אֶרֶץ כְּנָעַן. וַיִסְעוּ
בְנֵי יִשְׂרָאֵל מִמִדְבַּר־סִין וַיַחֲנוּ בְאָלוּש, וַיִסְעוּ מֵאָלוּש
וַיַחֲנוּ בִרְפִידִים. וַיְהִי בִּהְיוֹת בְּנֵי יִשְׂרָאֵל בִּרְפִידִים
וַיִתְנַפֵּל עֲלֵיהֶם מֵאֲחוֹרֵיהֶם עֲמָלֵק בֶּן אֱלִיפַז בֶּן עֵשָׂו,
אֲחִי צְפוֹ, וַיִלָחֵם אִתָּם וְעִמוֹ עַם רַב מְאֹד, וְקוֹסְמִים
וּמְכַשְפִים וּבַעֲלֵי אוֹב. וַתֶּחֱזַק הַמִלְחָמָה עִם עֲמָלֵק עַל
בְּנֵי יִשְׂרָאֵל, וַיַכֵּהוּ יְהוֹשֻעַ, מְשָרֵת מֹשֶה מִשֵבֶט
אֶפְרַיִם, לְפִי חָרֶב. וַיִסְעוּ בְנֵי יִשְׂרָאֵל מֵרְפִידִים וַיַחֲנוּ
בְמִדְבָּר סִינָי. וּבָעֵת הַהִיא בָּא שָמָה רְעוּאֵל הַמִדְיָנִי חוֹתֵן
מֹשֶה עִם צִפּוֹרָה בִּתּוֹ וּשְנֵי בָנֶיהָ גֵרְשׂם וֶאֱלִיעֶזֶר, כִּי
שָמַע אֶת כָּל הַנִפְלָאוֹת אֲשֶר עָשָה ה' לִבְנֵי יִשְׂרָאֵל. וַיֵשֶב
יִתְרוֹ (־רְעוּאֵל) וּבְנֵי בֵיתוֹ מֵאָז בְּקֶרֶב בְּנֵי יִשְׂרָאֵל
יָמִים רַבִּים וַיֵדְעוּ אֶת יְיָ אֵל עֶלְיוֹן מֵהַיוֹם הַהוּא
וָהָלְאָה.

וַיְהִי בַּחֹדֶש הַשְלִישִי לְצֵאת בְּנֵי יִשְׂרָאֵל
מֵאֶרֶץ מִצְרַיִם, בְּשֵשֶת יָמִים בּוֹ, וַיִתֵּן ה' לִבְנֵי יִשְׂרָאֵל
עַל הַר סִינַי בְּקוֹלוֹת וּבְרָקִים אֶת עֲשֶׂרֶת־הַדִבְּרוֹת עַל יְדֵי
מֹשֶה. ­– וַיִשְכֹּן כְּבוֹד ה' עַל הַר סִינַי,
וַיְהִי שָם מֹשֶה אַרְבָּעִים יוֹם וְאַרְבָּעִים לָיְלָה. לֶחֶם לֹא
אָכַל וּמַיִם לֹא שָתָה, וַיַשְמִיעֵהוּ ה' חֻקִים וּמִשְפָּטִים
צַדִיקִים לְהוֹרוֹת אֶת בְּנֵי יִשְׂרָאֵל. וַיִכְתֹּב אֶת עֲשֶׂרֶת
הַדִבְּרוֹת עַל שְנֵי לוּחוֹת אָבֶן. וּבְנֵי יִשְׂרָאֵל וְהָאֲסַפְסוּף
אֲשֶר בְּקִרְבָּם רָאוּ כִּי בוֹשֵש מֹשֶה לָרֶדֶת מִן הָהָר וַיִקָהֲלוּ
עַל אַהֲרֹן וַיֹאמְרוּ לוֹ: זֶה מֹשֶה הָאִיש לֹא יָדַעְנוּ מֶה הָיָה
לוֹ. עַתָּה קוּם וַעֲשֵׂה לָנוּ אֱלֹהִים אֲשֶר יֵלְכוּ לְפָנֵינוּ, וָלֹא
מוֹת תָּמוּת. וָיִירָא אַהֲרֹן מְאֹד וַיְצַו וַיָבִיאוּ לוֹ אִיש־אִיש
אֶת נִזְמֵי זְהָבוֹ וַיַעַשׂ עֵגֶל מַסֵכָה, וַיִשְׁתַּחֲווּ לוֹ בְנֵי
יִשְׂרָאֵל וַיַעַבְדוּהוּ. וַיֹאמֶר ה' אֶל מֹשֶה: לֵך רֵד, כִּי שִחֵת
עַמְךָ אֲשֶר הוֹצֵאתָ מֵאֶרֶץ מִצְרַיִם, וְעַתָּה הַנִיחָה לִי
­– וַאֲכַלֶה אוֹתָם וְאֶעֱשֶׂה
אוֹתְךָ לְגוֹי גָדוֹל. וַיְחַל מֹשֶה אֶת פְּנֵי ה' בְּעַד עַמוֹ
לְבִלְתִּי הַשְחִיתוֹ. וַיְהִי כַּאֲשֶר יָרַד מֹשֶה מֵעַל הָהָר, וַיַרְא
אֶת הָעֵגֶל וַיִחַר אַפּוֹ בִּבְנֵי יִשְׂרָאֵל וַיְשַבֵּר אֶת הַלוּחוֹת
תַּחַת הָהָר. וַיָבֹא אֶל הַמַחֲנֶה, הוּא וּמְשָרְתוֹ יְהוֹשֻעַ אִתּוֹ,
וַיִתֹּץ אֶת הָעֵגֶל וַיִשְׂרְפוֹ בָאֵש. וַיְהִי מִמָחֳרָת וַיַעַל מֹשֶה
אֶל הָהָר וַיִפְסֹל שָם שְנֵי לוּחוֹת אֲבָנִים אֲחֵרִים כָּרִאשוֹנִים
וַיִכְתֹּב עֲלֵיהֶם אֶת עֲשֶׂרֶת הַדִבְּרוֹת. וַיְהִי מִקֵץ אַרְבָּעִים
יוֹם וַיֵרֶד מֹשֶה מִן הָהָר וּשְנֵי לוּחוֹת אֶבֶן כָּרִאשוֹנִים
בְּיָדוֹ. וַיָשָב מֹשֶה אֶל מַחֲנֵה יִשְׂרָאֵל וַיְלַמְדֵם אֶת חֻקֵי
תוֹרַת ה' וּמִשְפָּטָיו. וַיְצַו מֹשֶה אֶת בְּנֵי יִשְׂרָאֵל, כִּי אִיש
אִיש יָבִיא אֶת תְּרוּמָתוֹ, כְּפִי אֲשֶר יִדְבֶנוּ לִבּוֹ, וְהָיְתָה
לְבִנְיַן מִשְכָּן לַיְיָ. וַיִשְׂמַח הָעָם לַדָבָר הַזֶה וַיָבֵא אִיש
אִיש אֶת תְּרוּמָתוֹ לִמְלֶאכֶת הַמִשְׁכָּן בַּזָהָב וּבַכֶסֶף
וּבַנְחֹשֶת, בַּעֲצֵי־שִטִים, בְּעוֹרוֹת וּבְאַרְגָמָן וּבְשֵש
וּבְתוֹלַעַת שָנִי. וַיָבֹאוּ חֲכָמִים וְחָרָשִים
וַיִבְנוּ אֶת הַמִשְכָּן, וַתֵּכֶל הַמְלָאכָה מִקֵץ חֲמִשָה חֳדָשִים.
וַיַעֲשוּ אֶת אֲרוֹן הָעֵדוּת וַיָשִׂימוּ בוֹ אֶת לוּחוֹת הַבְּרִית,
וַיַעֲשׂוּ אֶת הַשֻלְחָן וְאֶת הַמְנוֹרָה וְאֶת הַמִזְבֵּחַ, וְאֶת
בִּגְדֵי אַהֲרֹן הַכֹהֵן וּבָנָיו לַעֲבוֹדָתָם וְאֶת בִּגְדֵי
הַלְוִיִים1 לַעֲבוֹדָתָם עֲבוֹדַת הַקדֶש, כְּכָל אֲשֶר צִוָה יְיָ אֶת מֹשֶה.
וַיַחֲנוּ בְנֵי יִשְׂרָאֵל בְּמִדְבַּר פָּארָן. בָּעֵת
הַהִיא דִבֶּר ה' אֶל מֹשֶה כִּי יִשְלַח שְנֵים עָשָׂר אִיש לָתוּר אֶת
אֶרֶץ כְּנָעַן. וַיִשְלַח מֹשֶה שְנֵים עָשָׂר אִיש, אִיש אֶחָד אִיש
אֶחָד לַמַטֶה ­– לַחֲקֹר אֶת אֶרֶץ כְּנָעַן. וַיֵלְכוּ וַיָתוּרוּ
אוֹתָהּ מִמִדְבַּר סִין עַד רְחוֹב לְבוֹא חֲמָת. וַיָסוּרוּ אֶל מֹשֶה
מִקֵץ אַרְבָּעִים יוֹם. וַיוֹצִיאוּ עֲשָׂרָה מֵהַתָּרִים אֶת דִבַּת
הָאָרֶץ רָעָה, כִּי אֶרֶץ אוֹכֶלֶת יוֹשְבֶיהָ הִיא וּבְנֵי עֲנָק
יוֹשְבִים בָּהּ. רַק יְהוֹשֻעַ בִּן נוּן, מְשָרֵת משֶׁה, וְכָלֵב בֶּן
יְפֻנה נְשִׂיא שֵבֶט יְהוּדָה, נִסוּ לְהַרְגִיעַ אֶת הָעָם, אַךְ בְּנֵי
יִשְׂרָאֵל לֹא שָמְעוּ בְקוֹלָם וַיֹאמְרוּ לָשִׂים רֹאשׁ וְלָשׁוּב
מִצְרָיְמָה. וַיִקְצֹף ה' עַל בְּנֵי־יִשְׂרָאֵל וַיִשָׁבַע, אִם יִרְאֶה
אִיש מִבֶּן עֶשְׂרִים וָמַעְלָה מֵהַדוֹר הָרָע הַזֶה אֶת אֶרֶץ כְּנַעַן,
כִּי אִם כָּלֵב בֶּן יְפֻנֶה וִיהוֹשֻעַ בִּן נוּן, וְרַק בְּנֵי יוֹצְאֵי
מִצְרַיִם יָבוֹאוּ אֶל הָאָרֶץ. וַיָנוּעַ בְּנֵי יִשְׂרָאֵל בַּמִדְבָּר אַרְבָּעִים שָנָה עַד תֹּם כָּל הַדוֹר אֲשֶר יָצָא
מִמִצְרָיִם.

וַיְהִי בִּשְנַת שְלֹשִים וָשֵש לְצֵאת בְּנֵי יִשְׂרָאֵל
מֵאֶרֶץ מִמִצְרַיִם, וַיֵצֵא סִיחוֹן מֶלֶךְ הָאֱמוֹרִי אֲשֶר בְּעֵבֶר
הַיַרְדֵן מִזְרָחָה לְהִלָחֵם עִם בְּנֵי מוֹאָב. וַיִשְלַח סִיחוֹן
מַלְאָכִים אֶל בְּעוֹר בֶּן יַנוּס בֶּן בִּלְעָם, יוֹעֵץ מֶלֶךְ
מִצְרַיִם, וְאֶל בִּלְעָם בֶּן־בְּעוֹר מִפְּתוֹר אֲשֶר בַאֲרַם־נַהֲרַיִם
לְקַלֵל אֶת מוֹאָב לְמַעַן תִּנָתֵן בְּיָדוֹ. וַיָבֹא בְעוֹר בֶּן יַנוּס
וּבִלְעָם בְּנוֹ עִירָה סִיחוֹן וַיְקַלְלוּ אֶת בְּנֵי מוֹאָב וְאֶת
מַלְכָּם. וְסִיחוֹן עָלָה לַמִלְחָמָה עֲלֵיהֶם וַיַכֵּם מַכָּה רַבָּה
וַיַכְנִיעֵם תַּחְתָּיו, וְאֶת הָעִיר הַגְדוֹלָה חֶשְבּוֹן לָקַח מֵהֶם.
וַיָשֶׂם סִיחוֹן נְצִיבִים בְּחֶשְבּוֹן, עַל כֵּן אָמְרוּ אָז
הַמוֹשְלִים בְּעוֹר וּבִלְעָם בְּנוֹ:

"בֹּאוּ חֶשְבּוֹן, תִּבָּנֶה וְתִכּוֹנֵן עִיר מוֹאָב;

אוֹי לְךָ, מוֹאָב, אָבַדְתָּ עַם כְּמוֹש!" ­ –

וַיִתֵּן סִיחוֹן מַתָּנוֹת רַבּוֹת לִבְעוֹר וּלְבִלְעָם
בְנוֹ וַיְשַלְחֵם בְּכָבוֹד אֶל אַרְצָם. בָּעֵת הַהִיא עָבְרוּ בְּנֵי
יִשְׂרָאֵל בְּמַסְעֵיהֶם מִדֶרֶךְ מִדְבַּר מוֹאָב וַיָסֵבּוּ אֶת
מִדְבַּר אֱדוֹם וַיָבֹאוּ אֶל מִדְבַּר צִין הִיא קָדֵש, בַּחֹדֶש
הָרִאשוֹן בִּשְנַת מ' לְצֵאתָם מִמִצְרָיִם. וַתָּמָת שָם מִרְיָם
הַנְבִיאָה אֲחוֹת אַהֲרֹן וּמֹשֶה וַתִּקָבֵר שָם. וַיִסְעוּ בְנֵי
יִשְׂרָאֵל מִקָדֵש־בַּרְנֵעַ וַיָבֹאוּ אֶל הֹר־הָהָר. וַיָמָת שָם
אַהֲרֹן בֶּן מֵאה וְעֶשְׂרִים וְשָלֹש שָנִים וַיִקָבֵר שָם. וַיִסְעוּ
בְנֵי יִשְׂרָאֵל מֵהֹר הָהָר וַיָבֹאוּ בְּעִיֵי הָעֲבָרִים אֲשֶר
בִּגְבוּל אֶרֶץ מוֹאָב. וְלֹא אָבוּ בְנֵי מוֹאָב תֵּת לִבְנֵי יִשְׂרָאֵל
לַעֲבוֹר בְּאַרְצָם, וַיֵטוּ מֵעֲלֵיהֶם וַיָבֹאוּ אֶל אַרְנוֹן אֲשֶר
בֵּין גְבוּל אֱדוֹם וּמוֹאָב. וַיִשְלְחוּ בְנֵי יִשְׂרָאֵל מַלְאָכִים
אֶל סִיחוֹן מֶלֶךְ הָאֱמוֹרִי וַיְבַקְשוּהוּ כִּי יִתֵּן לָהֶם לַעֲבוֹר
דֶרֶךְ אַרְצוֹ. וַיְמָאֵן סִיחוֹן וַיִקְבֹּץ אֶת כָּל חֵילוֹ וַיִלָחֵם
בִּבְנֵי יִשְׂרָאֵל. וַיִתֵּן ה' אֶת סִיחוֹן וְאֶת עַמוֹ בְיַד בְּנֵי
יִשְׂרָאֵל וַיִירְשוּ אֶת אַרְצוֹ. וַיִפְנוּ בְנֵי יִשְׂרָאֵל וַיַעֲלוּ
דֶרֶךְ הַבָּשָׁן, וַיֵצֵא עוֹג מֶלֶךְ הַבָּשָן לִקְרַאת בְּנֵי
יִשְׂרָאֵל לְהִלָחֵם אִתָּם, וְעִמוֹ עַם רַב וְגִבּוֹרִים מְלֻמְדֵי
מִלְחָמָה מֵאֵין כְּמוֹהֶם. וְעוֹג מֶלֶךְ הַבָּשָן הָיָה עֲנָק גִבּוֹר
חַיִל וְנַעֲרוֹן בְּנוֹ עוֹד חָזָק מִמֶנוּ. וַיאֹמֶר עוֹג אֶל לִבּוֹ:
הִנֵה מַחֲנֵה יִשְׂרָאֵל מַהֲלַךְ שָלֹש פַּרְסָאוֹת ­– עַתָּה אֲמִיתֵם
בְּלֹא חֶרֶב וּבְלִי חֲנִית. וַיַעַל עוֹג עַל הָהָר וַיִקַח מִשָם אֶבֶן
אַחַת גְדוֹלָה וַיָשֶׂם אוֹתָהּ עַל רֹאשוֹ וַיֹאמֶר לְהַשְלִיכָהּ עַל
מַחֲנֵה יִשְׂרָאֵל. וַיָבֹא מַלְאַךְ ה' וַיִקוֹב חוֹר בָּאָבֶן הַהִיא
וַתֵּרֶד הָאֶבֶן עַד צַוַאר עוֹג וַתַּפִּילֵהוּ בְּכֹבֶד מַשָׂאָהּ
אָרְצָה. וַיִגַש אֵלָיו מֹשֶה וַיַכֵּהוּ בְקַרְסֻלֵי רַגְלָיו
וַיְמִיתֵהוּ. וַיִרְדְפוּ בְּנֵי יִשְׂרָאֵל אַחֲרֵי צִבְאוֹת עוֹג מֶלֶךְ
הַבָּשָן וַיַכּוּם לְפִי חָרֶב וַיִלְכְּדוּ בְנֵי יִשְׂרָאֵל אֶת
אַרְצוֹת שְנֵי מַלְכֵי הָאֱמוֹרִי אֲשֶר בְּעֵבֶר הַיַרְדֵן מִזְרָחָה,
שִשִים עִיר מִנַחַל אַרְנוֹן עַד הַר חֶרְמוֹן. הַצִידוֹנִים קָרְאוּ
לָהָר הַזֶה שִׂרְיוֹן וְהָאֱמוֹרִי קָרְאוּ לוֹ שְׂנִיר אוֹ
טוּר־תַּלְגָא, לֵאמֹר: הַר הַשֶלֶג. כִּי רֹאשוֹ מְכֻסֶה גַם בִּימוֹת
הַקַיִץ גַם בִּימוֹת הַסְתָו שֶלֶג־עוֹלַמִים.









בּני שבט לוי, שלא מבּית אהרן.  ↩
