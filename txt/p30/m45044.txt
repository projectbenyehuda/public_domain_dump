


	
	נמרד, תרח, אברם
	




וְכוּש בֶּן חָם בֶּן נֹחַ לָקַח לוֹ אִשָה בַּיָמִים
הָהֵם וַתֵּלֶד בֵּן וַיִקְרָא אֶת שְמוֹ נִמְרֹד לֵאמֹר: הֵחֵלוּ בְּנֵי
הָאָדָם לִמְרֹד בַּה‘. וַיִגְדַל הַיֶלֶד וְאָבִיו אֲהֵבוֹ מְאֹד,
כִּי בֶּן זְקוּנִים הוּא לוֹ, וַיִתֵּן לוֹ כּוּש אֶת כָּתְנוֹת הָעוֹר
אֲשֶר עָשָׂה אֱלֹהִים לָאָדָם וּלְאִשְתּוֹ. וַיְהִי אַחֲרֵי מוֹת אָדָם
וַיֻתְּנוּ הַכֻּתָּנוֹת לְחֲנוֹךְ בֶּן יֶרֶד, וַחֲנוֹךְ הִנְחִילָן
לִמְתוּשֶלַח בְּנוֹ, וּבְמוֹת מְתוּשֶלַח לְקָחָן נֹחַ וַיְבִיאֵן אֶל
הַתֵּבָה. וַיְהִי בְּצֵאתָם מִן הַתֵּבָה וַיִגְנֹב חָם אֶת הַכֻּתָּנוֹת
הָהֵן מֵאֵת נֹחַ אָבִיו וַיִקָחֵן וַיַסְתִּירֵן מֵאֵת אֶחָיו; וּבְלֶדֶת
כּוּש אֶת נִמְרֹד וַיִתֵן לוֹ אֶת הַבְּגָדִים
הָהֵם בְּאַהֲבָתוֹ אוֹתוֹ. וְכַאֲשֶר גָדַל נִמְרֹד וַיְהִי לְבֶן
עֶשְׂרִים וַיִלְבַּש אֶת הַכֻּתָּנוֹת וַיִתְחַזֵק
מְאֹד מֵהַיוֹם הַהוּא וַיְהִי לְגִבּוֹר צַיִד בָּאָרֶץ. וַיִלָחֵם
נִמְרֹד אֶת מִלְחֲמוֹת אֶחָיו בְּכָּל אוֹיְבֵיהֶם מִסָבִיב, וַיִתֵּן ה’
אוֹתָם בְּיָדוֹ וַיַכְנִיעֵם תַּחְתָּיו. וַיְהִי בִּהְיוֹת נִמְרֹד בֶּן אַרְבָּעִים שָׁנָה וַתִּפְרֹץ מִלְחָמָה בֵּין
אֶחָיו וּבֵין בְּנֵי יֶפֶת. וַיִתְחַזֵק נִמְרֹד וַיֵלֵך וַיִקְבֹּץ
אֶת כָּל בְּנֵי כוּש וַיִשְׂכֹּר גַם מֵאוֹהֲבָיו
וְיוֹדְעָיו וַיֵצֵא אִתָּם לַמִלְחָמָה נֶגֶד אוֹיְבָיו. וַיֵלְכוּ
הָאֲנָשִים הָאֵלֶה, בְּמִסְפַר חָמֵש מֵאוֹת וְאַרְבָּעִים אִיש, וַיִלָחֲמוּ בִּבְנֵי יֶפֶת וַיַכּוּם
וַיַכְנִיעוּם תַּחַת יָדָם. וַיָשֶׂם נִמְרֹד עַל בְּנֵי יֶפֶת נְצִיבִים
וַיִקַח מִבְּנֵיהֶם לְעֵרָבוֹן וַיִהְיוּ לוֹ מֵאָז לְמַס עוֹבֵד. וַיְהִי
כַּאֲשֶׁר שָׁב נִמְרֹד מֵהַמִלְחָמָה וַיִוָעֲדוּ אֲנָשָיו יַחַד,
וַיַמְלִיכוּהוּ עֲלֵיהֶם וַיָשִׂימוּ כֶּתֶר מַלְכוּת בְּרֹאשוֹ וַיָשֶׂם
נִמְרֹד שוֹפְטִים וּמַנְהִיגִים בָּאָרֶץ, וַיָשֶׂם לְשַׂר־צְבָאוֹ אֶת תֶּרַח
בֶּן־נָחוֹר וַיְגַדְלֵהוּ עַל כָּל שָׂרָיו. וַיְהִי כַּאֲשֶר נִצַח
נִמְרֹד אֶת כָּל אוֹיְבָיו
מִסָבִיב וַיִתְיָעֵץ עִם יוֹעֲצָיו לִבְנוֹת עִיר לְבִירַת־מַלְכוּתוֹ
וַיַעֲשׂוּ כֵן. וַיִמְצְאוּ בִקְעָה גְדוֹלָה וּרְחָבָה מְאֹד וַיִקְרָא
נִמְרֹד אֶת שֵם הָעִיר אֲשֶר בָּנָה לְשִבְתּוֹ – שִנְעָר, כִּי נָעַר
ה' אֶת כָּל אוֹיְבָיו מִפָּנָיו. וַתִּגְדַל מַלְכוּת נִמְרֹד עַל כָּל
הַגוֹיִם וְהַלְשוֹנוֹת. וְלֹא הָלַךְ נִמְרֹד בְּדַרְכֵי ה' וַיַרְשִיעַ
מִכָּל בְּנֵי הָאָדָם אֲשֶר הָיוּ לְפָנָיו אַחַר הַמַבּוּל, וַיַעַשׂ לוֹ
אֱלֹהֵי עֵץ וָאָבֶן וַיִשְתַּחוּ לָהֶם וַיִמְרֹד בַּה' וַיְלַמֵד אֶת
עֲבָדָיו וְכָל בְּנֵי עַמוֹ אֶת דְרָכָיו הָרָעִים. וְגַם מִרְדוֹן בְּנוֹ
הִרְשִיעַ עוֹד מֵאָבִיו, וְהָיָה כָּל אֲשֶר יִשְמַע אֶת מַעֲשֵׂי מִרְדוֹן וְאָמָר: מֵרְשָעִים יֵצֵא רֶשַע.

וְתֶרַח בֶּן נָחוֹר שַׂר־צְבָא נִמְרֹד הָיָה גָדוֹל
מְאֹד בַּיָמִים הָהֵם בְּעֵינֵי הַמֶלֶךְ וּבְעֵינֵי כָל עֲבָדָיו.
וַיִקַח תֶּרַח לוֹ אִשָה וּשְמָהּ אֲמַתְלָאָה בַּת כַּרְנְבוֹ וַתֵּלֶד
לוֹ בֵּן בִּשְנַת הַשִבְעִים לְחַיָיו, וַיִקְרָא אֶת שְמוֹ אַבְרָם, כִּי הֱרִימָהוּ הַמֶלֶךְ בַּיָמִים הָהֵם וַיְנַשְׂאֵהוּ מֵעַל כָּל
שָׂרָיו. וַיְהִי בַּלַיְלָה הַהוּא, עֵת הוּלֶדֶת אֶת
אַבְרָם, וַיָבֹאוּ כָּל עַבְדֵי תֶרַח וְכָל חַכְמֵי נִמְרֹד
וְחַרְטוּמָיו וַיֹאכְלוּ וַיִשְתּוּ בְּבֵיתוֹ וַיִשְׂמְחוּ אִתּוֹ
בְּשִׂמְחָתוֹ. וַיְהִי כַּחֲצות הַלַיְלָה אַחֲרֵי הַמִשְתֶּה, בְּצֵאת
חַרְטֻמֵי נִמְרֹד מִבֵּית תֶּרַח, וִַיִשְׂאוּ עֵינֵיהֶם הַשָמַיְמָה אֶל
הַכּוֹכָבִים וַיִרְאוּ וְהִנֵה כּוֹכָב אֶחָד גָדוֹל בָּא מִמִזְרַח שֶמֶש
וַיָרָץ בָּרָקִיעַ וַיִבְלַע אַרְבָּעָה
כּוֹכָבִים מֵאַרְבַּע רוּחוֹת הַשָמָיִם. וַיִתְמְהוּ חַכְמֵי הַמֶלֶךְ
לַמַרְאֶה הַהוּא וַיֹאמְרוּ אִיש אֶל רֵעֵהוּ: אֵין זֶה כִּי אִם הַיֶלֶד
הַנוֹלָד לְתֶרַח יִגְדַל וְיִגְבַּר וְיִירַש אֶת כָּל הָאָרֶץ. וַיְהִי בַּבֹּקֶר
וַיַשְכִּימוּ חַכְמֵי נִמְרֹד וְחַרְטוּמָיו וַיִוָעֲדוּ בְּבֵית־מוֹעֲדָם
וַיֹאמְרוּ אִיש אֶל רֵעֵהוּ: “הִנֵה הַמַרְאֶה אֲשֶר רָאִינוּ אֶמֶש
נֶעְלַם מֵאֵת הַמֶלֶךְ, וְהָיָה אִם יִוָדַע לוֹ הַדָבָר
בְּאַחֲרִית הַיָמִים וְיִתְעַבֵּר בָּנוּ
וֶהֱמִיתָנוּ. וְעַתָּה לְכוּ וְנַגִידָה לַמֶלֶךְ אֶת הַמַרְאֶה אֲשֶר
רָאִינוּ וְאֶת פִּתְרוֹן הַדָבָר”. וַיַעֲשׂוּ כֵן וַיֵלְכוּ וַיָבֹאוּ
אֶל נִמְרֹד וַיַגִידוּ לוֹ אֶת דבַר הַמַרְאֶה
אֲשֶר רָאוּ בְּצֵאתָם מֵאֵת תֶרַח. וַיִיעָצוּהוּ לִקְנוֹת מֵאֵת
תֶּרַח אֶת בְּנוֹ הָרַךְ הַנוֹלָד וְלַהֲמִיתֵהוּ טֶרֶם יִגְדַל וְיִירַש אֶת כָּל הָאָרֶץ.
וַיִשְמַע הַמֶלֶךְ אֶת דִבְרֵיהֶם וַתִּיטַב עֲצָתָם בְּעֵינָיו,
וַיִשְלַח וַיִקְרָא אֶת תֶרַח וַיָבֹא וַיַעֲמֹד
לְפָנָיו. וַיֹאמֶר אֵלָיו נִמְרֹד: הֻגֹד הֻגַד לִי אֲשֶר נוֹלַד לְךָ
בֵּן, וּמַרְאֶה כָּזֶה נִרְאָה אֶמֶש בַּשָמָיִם; וְעַתָּה תְּנָה לִי אֶת
בִּנְךָ וּנְמִיתֵהוּ טֶרֶם תִּצְמַח רָעָתוֹ, וְאֶתֵּן לְךָ מְחִירוֹ –
מְלֹא בֵיתְךָ כֶּסֶף וְזָהָב." – וַיַעַן תֶרַח
אֶת הַמֶּלֶךְ וַיֹאמֶר אֵלָיו: הִנֵה שָמַעְתִּי, אֲדוֹנִי הַמֶלֶךְ, אֶת
דְבָרֶיךָ. אוּלָם הַרְשֵנִי נָא וַאֲסַפֵּר לְךָ אֶת אֲשֶר קָרַנִי אֶתְמוֹל: אֲעִיוֹן בֶּן מוּרַד בָּא אֵלָי אֶמֶש
וַיְבַקְשֵנִי כִּי אֶמְכּר לוֹ אֶת הַסוּס אֲשֶר נָתַן לִי הַמֶלֶךְ
לְמַתָּנָה; – וָאֶעֱנֵהוּ: כַּאֲשֶר תִּיעָצֵנִי, הַמֶלֶךְ, אֶעֱשֶׂה. וַיִחַר אַף נִמְרֹד וַיֹאמַר:
הַחֲסַר כֶּסֶף וְזָהָב אַתָּה, תֶּרַח, כִּי תִבְעַט בְּמִנְחָתִי? –
וַיֹאמֵר תֶּרַח אֶל נִמְרֹד: “וְאֵיךְ אֶתֵּן בִּמְחִיר אֶת בְּנִי
אֲשֶר קָנִיתִי מֵאֵת ה'?” וַיְהִי כִּשְמֹעַ נִמְרֹד אֶת דִבְרֵי תֶּרַח
וְאֶת הַמָשָל אֲשֶר מָשַל לוֹ – וַתִּבְעַר בּוֹ
חֲמָתוֹ עַד לְהַשְחִית. וַיַרְא תֶּרַח כִּי כָלְתָה אֵלָיו הָרָעָה
וַיֹאמַר: יֻתַּן לִי זְמַן שְלֹשֶת יָמִים
וְדִבַּרְתִּי עַל לֵב אִשְתִּי וְאַנְשֵי בֵיתִי, כִּי יַעֲשׂוּ אֶת דְבַר
הַמֶלֶךְ. וַיֵאוֹת נִמְרֹד לַדָבָר הַזֶה. וַיְהִי כְּשוּב תֶרַח לְבֵיתוֹ
וַיְסַפֵּר לְאִשְתּוֹ וּלְאֶחָיו אֶת רְצוֹן הַמֶלֶךְ וַיִירְאוּ
הָאֲנָשִים יִרְאָה גְדוֹלָה. בַּיוֹם הַשְלִישִי שָלַח נִמְרֹד אֶל תֶּרַח
לֵאמֹר: תְּנָה לִי אֶת בִּנְךָ הַנוֹלָד, כַּאֲשֶר דִבַּרְתִּי לָךְ;
וְהָיָה אִם לֹא תְמַלֵא מִצְוָתִי זֹאת – וְהִשְמַדְתִּי הַיוֹם
אֶת כָּל בְּנֵי בֵיתְךָ וְלֹא אַשְאִיר לְךָ
זֵכֶר. וַתִּקַח אֲמַתְלָאָה אֶת בֶּן שִפְחָתָהּ, אֲשֶר יָלְדָה בְּלֵיל
הֻלֶדֶת אֶת אַבְרָם, וַיָבֵא תֶּרַח אוֹתוֹ אֶל נִמְרֹד. וַיִקַח נִמְרֹד
מִידֵי תֶּרַח אֶת הַיֶלֶד הַמוּבָא וַיַפִּילֵהוּ אַרְצָה וַיְנַפֵּץ אֶת
מֹחוֹ; וַיִשָכַח הַדָבָר מִלִבּוֹ, כִּי מֵאֵת ה' הָיְתָה זֹאת לְבִלְתִּי
הָמִית אֶת אַבְרָם. וְתֶרַח לָקַח אֶת אַבְרָם בְּנוֹ בַּסֵתֶר וְאֶת אִמוֹ וְאֶת מֵינִקְתוֹ וַיַחְבִּיאֵם בִּמְעָרָה, וַיִהְיוּ שָם כְּעֶשֶׂר שָנִים.

וְהָרָן בֶּן תֶּרַח אֲחִי אַבְרָם הַגָדוֹל לָקַח לוֹ
אִשָה בַּיָמִים הָהֵם בִּהְיוֹתוֹ בֶּן שְלֹשִים וָתֵּשַע שָנָה, וַתֵּלֶד
בֵּן וַיִקְרָא אֶת שְמוֹ לוֹט, וַתֵּלֶד עוֹד בַּת וַיִקְרָא אֶת
שְמָהּ מִלְכָּה. וּבִהְיוֹת הָרָן בֶּן אַרְבָּעִים
וּשְתַּיִם שָנָה יָלְדָה אִשְתּוֹ אֶת שָׂרַי, הִיא הַשָנָה הָעֲשִירִית
לְחַיֵי אַבְרָם. בַּיָמִים הָהֵם יָצָא אַבְרָם הוּא וְאִמוֹ
וּמֵינִקְתּוֹ מִן הַמְעָרָה, וַיֵלְכוּ אֶל נֹחַ וְשֵם בְּנוֹ וַיֵשְבוּ בְּבֵיתָם וַיִלְמַד שָם אַבְרָם מוּסַר
ה' וּדְרָכָיו הַטוֹבִים, וַיְהִי בְּבֵית נֹחַ שְלֹשִים וָתֵשַע
שָנָה.

וְכָל בְּנֵי הָאָדָם שָבוּ וַיִפְשְעוּ בַּה' בַּיָמִים
הָהֵם, וַיַעֲשׂוּ לָהֶם אֱלֹהֵי עֵץ וָאָבֶן וַיַעַבְדוּם, וְגַם הַמֶלֶךְ
וְשָׂרָיו וַעֲבָדָיו וְגַם תֶּרַח וּבְנֵי בֵּיתוֹ הִשְתַּחֲווּ לָהֶם.
וַיִהְיוּ לְתֶרַח י“בּ אֱלִילֵי עֵץ וָאָבֶן גְדוֹלִים לְי”בּ חָדְשֵי
הַשָנָה, וַיַעֲבֹד אֶת כָּל אֶחָד בְּחָדְשוֹ. וְאַבְרָם בֶּן תֶּרַח
הוֹלֵךְ וְגָדֵל אָז בְּבֵית נֹחַ, וַה' עִמוֹ. וַיִתֵּן לוֹ ה' לֵב
שוֹמֵעַ וּתְבוּנָה וַיֵדַע כִּי הָאֱלִילִים מַעֲשֵׂי יְדֵי הָאָדָם–הֶבֶל
הֵמָה. כִּי עוֹד בְּיַלְדוּתוֹ, בִּרְאוֹת אַבְרָם אֶת הַשֶמֶש וַיֹאמֶר
אֶל לִבּוֹ: הוּא הָאֱלֹהִים וְאֹתוֹ אֶעֱבֹד, וַיַעֲבֹד בַּיוֹם הַהוּא
אֶת הַשֶמֶש וַיִתְפַּלֵל אֵלָיו. וַיְהִי
בָּעֶרֶב וַתָּבוֹא הַשֶמֶש וַיֶחֱשַךְ הַיוֹם–וַיֹאמֶר
אַבְרָם בְּלִבּוֹ: אֵין זֶה אֵיפוֹא אֱלֹהַ. וַיְהִי בַּלַיְלָה וַיַרְא
אַבְרָם אֶת הַיָרֵחַ וַיֹאמַר: אַךְ זֶה הוּא הָאֱלֹהִים אֲשֶר בָּרָא אֶת
הַשָמַיִם וְאֶת הָאָרֶץ וְאֶת כָּל צְבָאָם, וַיַעֲבֹד אוֹתוֹ כָּל
הַלַיְלָה הַהוּא וַיִתְפַּלֵל אֵלָיו. וַיְהִי בַּבֹּקֶר וַיֵעָלֵם
הַיָרֵחַ – וַתִּזְרַח שֵנִית הַשֶמֶש, וַיֹאמֶר אַבְרָם: אֵין שְנֵיהֶם
אֱלֹהִים כִּי אִם מְשָרְתָיו וְעוֹשֵׂי רְצוֹנוֹ הָעוֹמְדִים עַל
מִשְמַרְתָּם בְּמוֹעֲדָם. וַיָבֶן אָז אַבְרָם, כִּי בּוֹרֵא הָעוֹלָם
וּמַנְהִיגוֹ אֶחָד הוּא וְאֵין שֵנִי לוֹ.
