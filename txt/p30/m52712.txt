


	
	מות יעקב
	




וַיְחִי יַעֲקֹב בְּאֶרֶץ מִצְרַיִם שְבַע עֶשְׂרֵה שָנָה,
וַיִהְיוּ כָל יְמֵי שְנֵי חַיָיו שֶבַע וְאַרְבָּעִים שָנָה וּמְאַת
שָנָה. בָּעֵת הַהִיא חָלָה יַעֲקֹב אֶת חָלְיוֹ אֲשֶר יָמוּת בּוֹ.
וַיִשְלַח וַיִקְרָא אֶל יוֹסֵף בְּנוֹ מִמִצְרַיִם, וַיָבֹא אֵלָיו.
וַיאֹמֶר יַעֲקֹב אֶל יוֹסֵף וְאֶל בָּנָיו: הִנֵה אָנֹכִי מֵת וֵאלֹהֵי
אֲבוֹתֵיכֶם פָּקוֹד יִפְקוֹד אֶתְכֶם וַהֲשִיבְכֶם אֶל הָאָרֶץ אֲשֶר
נִשְבַּע לָתֵת לָכֶם וְלִבְנֵיכֶם אַחֲרֵיכֶם. וְעַתָּה בְמוֹתִי קִבְרוּ
אוֹתִי בִּמְעָרַת הַמַכְפֵּלָה אֲשֶר בְּחֶבְרוֹן בְּקִבְרוֹת אֲבוֹתָי.
וַיִשָבְעו לוֹ בָנָיו עַל הַדָבָר הַזֶה. וַיְצַו יַעֲקֹב לְבָנָיו
לֵאמֹר: עִבְדוּ אֶת ה' אֱלֹהֵיכֶם כִּי הוּא יַצִיל אֶתְכֶם מִכָּל צָרָה,
כַּאֲשֶר הִצִיל אֶת אֲבוֹתֵיכֶם; וְעַתָּה קִרְאוּ לִי לְכָל בְּנֵיכֶם
וַאֲבָרְכֵם לִפְנֵי מוֹתִי. וַיָבֹאוּ כָּל יַלְדֵי בְּנֵי יַעֲקֹב
וַיַעַמְדוּ לְפָנָיו וַיְבָרֶךְ אֶת כָּל אֶחָד וַיאֹמֶר לָהֶם: ה'
אֱלֹהֵי אֲבוֹתֵיכֶם יְבָרֵךְ אֶת כֻּלְכֶם וְיוֹסִיף עֲלֵיכֶם אֶלֶף
אַלְפֵי פְעָמִים וְיִרְבְּכֶם וְיַפְרְכֶם וִהְיִיתֶם לְגוֹי גָדוֹל
כַּאֲשֶר נִשְבַּע ה' לְאַבְרָהָם אָבִינוּ. וַיְהִי מִמָחֳרָת וַיִקְרָא
יַעֲקֹב שֵנִית לְבָנָיו וַיִקָבְצוּ כֻלָם וַיָבֹאוּ וַיַעַמְדוּ
לְפָנָיו. וַיְבָרֶךְ יַעֲקֹב גַם אֶת בָּנָיו אִיש-אִיש כְּבִרְכָתוֹ
בֵרַךְ אוֹתָם, הֲלֹא הִיא כְתוּבָה בְּסֵפֶר תּוֹרַת ה'. וַיאֹמֶר יַעֲקֹב
לִיהוּדָה: יְהוּדָה בְנִי, יָדַעְתִּי כִּי גְבִיר לְאַחֶיךָ תִּהְיֶה
וּמָלֹךְ יִמְלְכוּ בָנֶיךָ עֲלֵיהֶם עַד עוֹלָם. וַיִפֶן אֶל כָּל בָּנָיו
וַיאֹמַר: הִנֵה אֲנִי נֶאֱסָף הַיוֹם אֶל עַמִי, וְאַתֶּם שָׂאוּנִי
מִמִצְרַיִם וּקְבַרְתֻּנִי בִּקְבוּרַת אֲבוֹתַי אֲשֶר בִּמְעָרַת
הַמַכְפֵּלָה. וְכָכָה תִּשְׂאוּ אֶת גְוִיָתִי מֵאֶרֶץ מִצְרַיִם:
יְהוּדָה, יִשָׂכָר וּזְבוּלֻן יִשְׂאוּ אֶת מִטָתִי מִזְרָחָה; רְאוּבֵן,
שִמְעוֹן, גָד – תֵּימָנָה; אֶפְרַיִם מְנַשֶה וּבִנְיָמִין –
יָמָה;

דָן, אָשֵר וְנַפְתָּלִי – צָפוֹנָה; וְלֵוִי בְּנִי אַל
יִשָׂא עִמְכֶם, כִּי הוּא וּבָנָיו אַחֲרָיו יִשְׂאוּ אֶת אֲרוֹן בְּרִית
ה' כָּל יְמֵיהֵם. וְגַם יוֹסֵף בְּנִי אַל יִשָׂא אֶת מִטָתִי, כִּי
כַּמֶלֶך – כְּבוֹדוֹ, אַךְ אֶפְרַיִם וּמְנַשֶה בָנָיו יִהְיוּ תַחְתָּיו.
וְאַתֶּם בְּנִי, כַּבְּדוּ וְהוֹקִירוּ אִיש אֶת אָחִיו וְאִיש אֶת
קְרוֹבוֹ וְצַווּ גַם אַתֶּם אֶת בְּנֵיכֶם וּבְנֵי בְנֵיכֶם אַחֲרֵיכֶם
לַעֲבוֹד לַה' אֱלֹהֵי אֲבוֹתֵינוּ, לְמַעַן אֲשֶר תַּאֲרִיכוּ יָמִים עַל
הָאֲדָמָה אֲשֶר נִשְבַּע ה' לַאֲבוֹתֵיכֶם לָתֵת לָכֶם, כִּי תַעֲשׂוּ אֶת
הַטוֹב וְהַיָשָר בְּעֵינָיו. וְאַתָּה יוֹסֵף בְּנִי, שָׂא נָא לְפֶשַע
אַחֶיךָ אֲשֶר רָעָה גְמָלוּךָ, כִּי הִנֵה ה' חֲשָבָהּ לְטוֹבָה. וְאַל
נָא תַעֲזֹב, בְּנִי, אֶת אַחֶיךָ וְאֶת כָּל בְּנֵי מִשְפַּחְתָּם
מִפְּנֵי יוֹשְבֵי הָאָרֶץ וְהָגֵן תָּגֵן עֲלֵיהֶם מִפְּנֵי לוֹחֲצֵיהֶם.
וַיַעֲנוּ בְּנֵי יַעֲקֹב פֶּה אֶחָד לֵאמֹר: כְּכָל אֲשֶר צִוִיתָנו כֵּן
נַעֲשֶׂה. וַיוֹסֶף יַעֲקֹב לְצַווֹת לְבָנָיו וַיאֹמַר: בָּנַי!
יָדַעְתִּי אֶת אֲשֶר יִקְרֶה אֶתְכֶם בְּאַחֲרִית הַיָמִים בְּאֶרֶץ
מִצְרָיִם: צָרוֹת רַבּוֹת וְרָעוֹת יִמְצְאוּ בָאָרֶץ הַזאֹת אֶת
בְּנֵיכֶם וְאֶת בְּנֵי בְנֵיכֶם. וְהָיָה כִּי יֵלְכוּ בְנֵיכֶם בְּדֶרֶךְ
ה' וְשָלַח לָהֶם מוֹשִיעַ וְהִצִילָם מִכָּל צָרוֹתֵיהֶם וֶהֱשִיבָם אֶל
אֶרֶץ אֲבוֹתֵיהֶם לְרִשְתָּהּ
