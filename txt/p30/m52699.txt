


	
	יוסף וזליכה
	




וְהַיִשְמְעֵאלִים אֲשֶר קָנוּ אֶת יוֹסֵף מֵאֵת
הַמִדְיָנִים הָלְכוּ עִמוֹ לָרֶדֶת מִצְרָיְמָה. וַיְהִי כַּאֲשֶר
הִקְרִיבוּ לָבוֹא עַד גְבוּל אֶרֶץ מִצְרָיִם, וַיִפְגְשוּם אַרְבָּעָה
אֲנָשִים מִבְּנֵי מְדָן בֶּן אַבְרָהָם. וַיֹאמְרוּ אֲלֵיהֶם
הַיִשְמְעֵאלִים: הֲתַחְפְּצוּ לִקְנוֹת מִיָדֵנוּ אֶת הָעֶבֶד הַזֶה? –
וַיִרְאוּ הַמְדָנִים אֶת יוֹסֵף וְהִנֵה נַעַר יָפֶה הוּא עַד לְהַלֵל
וַיִקְנוּהוּ מִידֵי הַיִשְמְעֵאלִים. וַיָבֹאוּ הַיִשְמְעֵאלִים
מִצְרַיְמָה וְגַם הַמְדָנִים בָּאוּ שָמָה עִם יוֹסֵף בַּיוֹם הַהוּא.
וַיֹאמְרוּ הַמְדָנִים אִיש אֶל רֵעֵהוּ: הִנֵה שָמַעְנוּ כִּי פּוֹטִיפַר
סְרִיס פַּרְעֹה, שַר הַטַבָּחִים, מְבַקֵש לוֹ עֶבֶד טוֹב אֲשֶר יַעֲמוֹד
לְפָנָיו וּלְהַפְקִידֵהוּ עַל בֵּיתוֹ וְעַל כָּל אֲשֶר לוֹ, הָבָה
נִמְכֹּר נָא לוֹ אֶת הַנַעַר הַזֶה אִם יִתֵּן לָנוּ אֶת מְחִירוֹ
כַּאֲשֶר נָשִית עָלָיו. וַיָבֹאוּ הַמְדָנִים בֵּיתָה פּוֹטִיפַר
וַיַצִיגוּ לְפָנָיו אֶת יוֹסֵף וַיִישַר בְּעֵינָיו מְאֹד מְאֹד. וַיאֹמֶר
פּוֹטִיפַר אֶל הַמְדָנִים: אֲנִי אֶקְנֶה אֶת הַנַעַר הַזֶה מִיֶדְכֶם,
אַךְ הַגִידוּ לִי: אוּלַי גָנוּב הוּא אִתְכֶם? כִּי לֹא עֶבֶד וְלֹא
בֶּן-עֶבֶד הוּא לְפִי תָאֳרוֹ וּמַרְאֵה פָנָיו. וַיֵלְכוּ הַמְדָנִים
וַיָבִיאוּ אֶת הַיִשְמְעֵאלִים ויַצַיִגוּם לִפְנֵי פּוֹטִיפַר,
וַיָעִידוּ הַיִשְמְעֵאלִים כִּי גַם הֵם קָנוּ אֶת הָעֶבֶד הַזֶה מִידֵי
הָעִבְרִים וַיִמְכְּרוּהוּ לְהַמְדָנִים. וַיִשְמַע פּוֹטִיפַר אֶת
דִבְרֵי הַיִשְמְעֵאלִים וַיֵקֶן אֶת יוֹסֵף מִידֵי הַמְדָנִים
וַיְבִיאֵהוּ אֶל בֵּיתוֹ כִּי יְשָרְתֵהוּ. וַיִמְצָא יוֹסֵף חֵן
בְּעֵינֵי פּוֹטִיפַר וַיַאֲמֵן בּוֹ וַיַפְקִידֵהוּ עַל כָּל בֵּיתוֹ
וּנְכָסָיו. וַיְבָרֵךְ ה' אֶת יוֹסֵף וַיְהִי אִיש מַצְלִיחַ. וַיַרְא
פּוֹטִיפַר כִּי בֵרְכָהוּ ה' בִּגְלַל יוֹסֵף, וַיִתֵּן בְּיָדוֹ אֶת כָּל
אֲשֶר לוֹ, וַיְהִי יוֹסֵף הַמוֹצִיא וְהַמֵבִיא וְעַל פִּיו יִשַק כָּל
בֵּית פּוֹטִיפַר.

וְיוֹסֵף אָז בֶּן שְמוֹנֶה עֶשְׂרֵה שָנָה, בָּחוּר
יְפֵה-עֵינַיִם וְטוֹב מַרְאֶה, וְכָמוֹהוּ לֹא הָיָה עוֹד לְיפִי בְּכָל
אֶרֶץ מִצְרָיִם. וַתִּשָׂא זְלִיכָה אֵשֶת פּוֹטִיפַר אֶת עֵינֶיהָ אֶל
יוֹסֵף וַתַּחְמְדֵהוּ בְלִבָּהּ, וַתִּדְבַּק נַפְשָהּ בּוֹ וַתָּחֶל
לַהֲסִיתֵהוּ כִּי יִקָחֶהָ לְאִשָה. וְלֹא שָמַע אֵלֶיהָ יוֹסֵף וְלֹא
נִפְתָּה אַחֲרֵי דְבָרֶיהָ וּמַחֲמָאוֹת פִּיהָ וְלֹא אָבָה לִבְגֹד
בַּאֲדוֹנוֹ. וַיְהִי כַּאֲשֶר לֹא יָכְלָה זְלִיכָה לְפַתּוֹת אֶת יוֹסֵף
וַתָּחַל מֵרֹב אַהֲבָתָה אֵלָיו וּמִקֶצֶף וַתִּפּוֹל לְמִשְכָּב.
וַתָבאֹנָה כָּל שָׂרוֹת מִצְרָיִם לְבַקְרָהּ וַיִשְאָלוּהָ: מַדוּעַ
פָּנַיִךְ כֹּה דַלִים וְרָזִים וְאַתְּ אִשָה לְשָׂר גָדוֹל בַּמַלְכוּת
וְאֵינָךְ חֲסֵרָה דָבָר. וַתַּעַן זְלִיכָה אֶת שָׂרוֹתֶיהָ וַתּאֹמֶר
בְּמַר נַפְשָהּ: הַיוֹם יִוָדַע לָכֶן בְּשֶלְמָה בָּא עָלַי הֶחֳלִי
הַזֶה. וַתְּצַו אֶת נַעֲרוֹתֶיהָ לַעֲשׂוֹת מִשְתֶּה לְכָל הַנָשִים,
אֲשֶר בָּאוּ לְבַקְרָהּ, וּלְצַווֹת עַל יוֹסֵף כִּי יְשָרֵת לִפְנֵיהֶן.
וַיְהִי כַּאֲשֶר רָאוּ הַשָׂרוֹת אוֹתוֹ וְאֶת יְפִי תָאֳרוֹ וְלֹא
יָכְלוּ לְהָסֵב אֶת עֵינֵיהֶן מִמֶנוּ וַתַּחְתֹּכְנָה אֶת אֶצְבְּעוֹת
יְדֵיהֶן בַּשַׂכִּינִים, אֲשֶר קָלְפוּ בָהֶן אֶתְרוֹגִים, – עַד
שְפָךְ-דָם. וְלֹא שָׂמוּ לֵב לַדָם הַשוֹטֵף מִידֵיהֶן עַל
שִׂמְלוֹתֵיהֶן. כִּי לֹא יָכְלוּ לִגְרֹע עַיִן מִיוֹסֵף וּמִיפִי תָאֳרוֹ
הַמַפְלִיא, וְלֹא הִרְגִישוּ אֶת הַנַעֲשֶׂה עִמָהֶן. וַתֵּרֶא זְלִיכָה
אֶת הַדָם הַשוֹטֵף מִידֵי שָׂרוֹתֶיהָ וַתּאֹמֶר לָהֶן: הַבֵּטְנָה נָא
אֶל יְדֵיכֶן וְאֶל אֶתְרוֹגֵיכֶן. וַתַּבֵּטְנָה הַשָׂרוֹת וַתִּרְאֶינָה
וְהִנֵה מִידֵיהֶן וְאֶתְרוֹגֵיהֶן שוֹטֵף דָם רַב עַל שִׂמְלוֹתֵיהֶן.
וַתּאֹמֶר לָהֶן זְלִיכָה: אַתֶּן רַק רֶגַע רְאִיתֶן אוֹתוֹ וְקָרְכֶן
כַּדָבָר הַזֶה, – וְאֵיכָכָה לֹא אִדַל אָנֹכִי וְלֹא יִכְחַש בְּשָׂרִי
בִּרְאוֹתִי אוֹתוֹ יוֹם-יוֹם? –

וַתּוֹסֶף זְלִיכָה לַחֲלוֹת מֵאַהֲבָתָהּ אֶת יוֹסֵף.
וַיְהִי הַיוֹם וְיוֹסֵף עוֹשֶׂה אֶת מְלַאכְתּוֹ בַּבַּיִת, וַתָּבֹא
אֵלָיו זְלִיכָה בַּלָט וַתִּפּוֹל לְרַגְלָיו וַתֵּבְךְ וַתִּתְחַנֵן
אֵלָיו כִּי לֹא יִבְעַט בְּאַהֲבָתָהּ. וַיַעַן יוֹסֵף וַיאֹמֶר לָהּ:
“הֵן אֲדוֹנִי מַאֲמִין בִּי, וְאֵיךְ אֶעֱשֶׂה אֶת הָרָעָה הַזֹאת
וְחָטָאתִי לֵאלֹהִים? – אַךְ הִיא לֹא חָדְלָה מִמֶנוּ וַתּוֹסֶף
לְפַתּוֹתוֹ כְּפַעַם-בְּפָעַם. וַיְהִי אַחֲרֵי-כֵן וַיִמָלֵא נַחַל
שִיחוֹר עַל כָּל גְדוֹתָיו, וַיֵצְאוּ כָּל יוֹשְבֵי מִצְרָיִם בְּתֻפִּים
וּבִמְחוֹלוֹת אֶל שְׂפַת הַנָהָר לָחוֹג אֶת חַג-הָעָם הַזֶה,
כְּמִשְפָּטָם מֵאָז לַעֲשׂוֹת יוֹם-טוֹב וּמִשְתֶּה כַּאֲשֶר מִתְמַלֵא
הַיְאוֹר אֶת מֵימָיו וּמַשְקֶה אֶת כָּל אֶרֶץ מִצְרָיִם. וַיֵצְאוּ גַם
כָּל בְּנֵי בֵּית פּוֹטִיפַר וַעֲבָדָיו וְשִפְחוֹתָיו וַיֵלְכוּ אֶל
שְׂפַת הַנָהָר, וְרַק זְלִיכָה נִשְאֲרָה בַבַּיִת כִּי אָמְרָה: חוֹלָה
אָנֹכִי. וַתִּלְבַּש אֶת בִּגְדֵי חֲמוּדוֹתֶיהָ וַתִּתְקַשֵט בִּבְשָמִים
וַתֵּשֶב בְּפֶתַח הֵיכָלָהּ וַתְּחַכֶּה לְבוֹא יוֹסֵף. וַיְהִי כַעֲבוֹר
יוֹסֵף עַל פָּנֶיהָ וַתִּתְפְּשֵׂהוּ בְּבִגְדוֹ וַתֹּאמֶר לוֹ: חֵי
פַרְעֹה אִם אֵינְךָ עוֹשֶׂה אֶת דְבָרִי הַיוֹם – מוֹת תָּמוּת!”
וַתְּמַהֵר וַתִּשְלֹף חַרְבָּהּ מִתַּחַת בְּגָדֶיהָ וַתִּתְּנֶהָ עַל
צַוַאר יוֹסֵף. וַיֶחֱזַק יוֹסֵף מִמֶנָה וַיַעֲזֹב אֶת בִּגְדוֹ בְיָדָהּ
וַיָנָס מִפָּנֶיהָ הַחוּצָה. וַתִּפְחַד זְלִיכָה לְנַפְשָהּ, פֶּן
יִוָדַע קְלוֹנָהּ בָּרַבִּים, וַתַּעַשׂ בְעָרְמָה: וַתֵּלֶךְ וַתִּשְכַּב
בְּמִטָתָהּ וַתָּנַח אֶצְלָהּ אֶת בֶּגֶד יוֹסֵף וַתִּצְעַק: רְאוּ אֶת
הָעֶבֶד הָעִבְרִי, כִּי בָא לְצַחֵק בִּי. וַיְהִי בְּשוּב פּוֹטִיפַר
לְבֵיתוֹ וַיִשְמַע אֶת דִבְרֵי אִשְתּוֹ, וַיְצַו לְיַסֵר בְּשֵבֶט אֶת
יוֹסֵף, וַיִקְרָא לַכֹּהֲנִים כִּי יִשְפְּטוּהוּ מִשְפַּט מָוֶת כַּחֹק
לָאִיש הַבָּא לִגְזֹל אֶת אֵשֶת רֵעֵהוּ. אוּלָם כִּרְאוֹת בֶּן
פּוֹטִיפַר הַקָטָן, שֶגַם הוּא נִשְאַר בַּבַּיִת וְלֹא יָצָא אֶל שְׂפַת
הַנָהָר, כִּי עַבְדֵי אָבִיו מַכִּים אֶת יוֹסֵף, וַיָבֹא אֶל הַשוֹפְטִים
וַיאֹמַר: שֶקֶר אִמִי דוֹבֶרֶת, כָּזֶה וְכָזֶה הָיָה הַדָבָר. וַיִכָּלֵם
פּוֹטִיפַר לְדִבְרֵי בְנוֹ הַקָטָן וְהַשוֹפְטִים נָבוֹכוּ, וַיָבִינוּ
כִּי חַף יוֹסֵף מִפֶּשַע, וַיִשְפְּטוּ אוֹתוֹ רַק לְמַאֲסָר בַּעֲוֹן
שֵם-רָע שֶיָצָא בִגְלָלוֹ עַל אַחַת מִשָׂרוֹת מִצְרָיִם
הַכְּבוּדוֹת.

וַיְהִי בָּעֵת הַהִיא, הִיא שְנַת רֶדֶת יוֹסֵף
מִצְרַיְמָה, וַיֵלֶךְ רְאוּבֵן בֶּן יַעֲקֹב תִּמְנָתָה וַיִקַח אֶת
אֵלִיוֹרָם בַּת חִוִי הַכְּנַעֲנִי לוֹ לְאִשָה, וַתֵּלֶד לוֹ אֶת חֲנוֹךְ
וְאֶת פַּלוּא וְאֶת חֶצְרוֹן וְאֶת כַּרְמִי. וְשִמְעוֹן אָחִיו לָקַח לוֹ
אֶת דִינָה אֲחוֹתוֹ1 וַתֵּלֶד לוֹ אֶת נְמוּאֵל
וְיָמִין וְאֹהַד וְיָכִין וָצֹחַר. וַיִקַח לוֹ שִמְעוֹן עוֹד אִשָה אַחַת
וּשְמָהּ בּוּנָה, אֲשֶר שָבָה בְּעוֹדֶנָה יַלְדָה קְטַנָה בְּהַכּוֹתוֹ
אֶת הָעִיר שְכֶם וַתְּהִי לְשִפְחָה לְדִינָה. וַתֵּלֶד בּוּנָה
לְשִמְעוֹן בֵּן וַיִקְרָא אֶת שְמוֹ שָאוּל. וִיהוּדָה לָקַח לוֹ אִשָה
מֵעִיר עֲדוּלָם וּשְמָהּ עִלִית בַּת שוּעַ וַתֵּלֶד לוֹ אֶת עֵר וְאֶת
אוֹנָן וְאֶת שֵלָה. וַיְהִי כַּאֲשֶר גָדַל עֵר וַיִקַח לוֹ יְהוּדָה
אָבִיו אִשָה וּשְמָהּ תָּמָר בַּת עֵילָם בֶּן שֵם. וַיָמָת עֵר וַיִקַח
יְהוּדָה אֶת תָּמָר כַּלָתוֹ2 לוֹ לְאִשָה
וַתֵּלֶד לוֹ אֶת פֶּרֶץ וְאֶת זָרַח. וַיֵלְכוּ לֵוִי וְיִשָׂכָר אַרְצָה
הַקֶדֶם וַיִקְחוּ לָהֶם שָם אֶת בְּנוֹת יוֹבָב בֶּן יָקְטָן בֶּן עֵבֶר
לָהֶם לְנָשִים. וּלְיוֹבָב שְתֵּי בָנוֹת, שֵם הַבְּכִירָה עֲדִינָה וְשֵם
הַצְעִירָה אֲרִידָה, וַיִקַח לֵוִי לְאִשָה אֶת עֲדִינָה וְיִשָׂכָר לָקַח
לוֹ אֶת אֲרִידָה וַיָשוּבוּ אַרְצָה כְּנַעַן אֶל בֵּית אֲבִיהֶם.
וַתֵּלֶד עֲדִינָה לְלֵוִי אֶת גֵרְשוֹן וְאֶת קְהָת וְאֶת מְרָרִי,
וְאֲרִידָה יָלְדָה לְיִשָׂכָר אֶת תּוֹלָע וּפֻוָה וְיוֹב וְשִמְרוֹן.
וְדָן הָלַךְ אֶל אֶרֶץ מוֹאָב וַיִקַח אֶת אַפְלֶלֶת בַּת חֲמִידַן
הַמוֹאָבית לְאִשָה, וַתֵּלֶד לוֹ בֵּן וַיִקְרָא אֶת שְמוֹ חוּשִים.
וַיֵלְכוּ גָד וְנַפְתָּלִי חָרָנָה וַיִקְחוּ מִשָם אֶת בְּנוֹת אֲמוֹרָם:
שֵם הַגְדוֹלָה מְרֵימַת וְשֵם הַקְטָנָה עוּצִית. וַיִקַח נַפְתָּלִי אֶת
מְרִימַת וְגָד לָקַח לוֹ אֶת עוּצִית וַיְבִיאוּן אַרְצָה כְּנַעַן לְבֵית
אֲבִיהֶם. וַתֵּלֶד מְרִימַת לְנַפְתָּלִי אֶת יַחְצְאֵל וְגוּנִי וְיֵצֶר
וְשִלֵם, וְעוּצִית יָלְדָה לְגָד אָת צִפְיוֹן וְחַגִי וְשוּנִי
וְאֶצְבּוֹן עֵרִי וַאֲרוֹדִי וְאַרְאֵלִי. וְאָשֵר לָקַח לוֹ לְאִשָה אֶת
עֶדוֹן בַּת אֶפְלָל בֶּן חֲדַד בֶּן יִשְמָעֵאל. וַתָּמָת עֶדוֹן עֲקָרָה
וְאֵין לָהּ וָלָד, וַיֵלֶךְ אָשֶר אֶל עֵבֶר-הַנָהָר וַיִקַח לוֹ אֶת
הֲדוֹרָה בַּת אֲבִימָאֵל בֶּן עֵבֶר בֶּן שֵם לְאִשָה. וְהַנַעֲרָה טוֹבַת
מַרְאֶה וּבַעֲלַת שֵׂכֶל וַתְּהִי לְאִשָה לְמַלְכִּיאֵל בֶּן עֵילָם בֶּן
שֵם. וַתֵּלֶד הֲדוֹרָה לְמַלְכִּיאֵל בַּת וּשְמָהּ שֶׂרַח. וַיְהִי
אַחֲרֵי מוֹת אֵשֶת אָשֵר הָרִאשוֹנָה וַתְּהִי הֲדוֹרָה לוֹ לְאִשָה
וַתָּבֵא אִתָּהּ גַם אֶת בִּתָּהּ אַרְצָה כְּנַעַן, וְשֶׂרַח בַּת שָלֹש
שָנִים בָּעֵת הַהִיא. וַתִּגְדַל הַיַלְדָה בְּבֵית יַעֲקֹב וַתְּהִי
סֶרַח טוֹבַת-מַרְאֶה וַיִתֵּן לָהּ אֱלֹהִים גַם שֵׂכֶל רַב וְלֵב טוֹב.
וַתֵּלֶד הֲדוֹרָה לְאָשֵר אֶת יִמְנָה וְיִשְוָה וְיִשְוִי וּבְרִיעָה.
וּזְבוּלֻן הָלַךְ מִדְיָנָה וַיִקַח לוֹ לְאִשָה אֶת מָרוּשָׂה בַּת
מוֹּרַד בֶּן אֲבִידַע בֶּן מִדְיָן וַיְבִיאֶהָ אַרְצָה כְּנַעַן,
וַתֵּלֶד לוֹ אֶת סֶרֶד וְאֶת אֵילוֹן וְיַחְצְאֵל. וַיִשְלַח יַעֲקֹב אֶל
אֲרָם בֶּן צוֹבָא בֶּן תֶּרַח וַיִקַח אֶת מַחְלִיָה בַּת אֲרָם לְאִשָה
לְבִנְיָמִין בְּנוֹ הַקָטָן, וַתֵּלֶד לוֹ אֶת בֶּלַע וָבֶכֶר, אַשְבְּאֵל
גֵרָא וְנֲעֲמָן. וְאַחֲרֵי כֵן הָלַךְ בִּנְיָמִין וַיִקַח לוֹ לְאִשָה
גַם אֶת עָרְבַת בַּת שִמְרן בֶּן אַבְרָהָם עַל אִשְתּוֹ הָרִאשוֹנָה,
וַתֵּלֶד לוֹ אֶת אֵחִי וָרֹאש וּמֻפִּים וְחֻפִּים וָאָרְדְ. –.
