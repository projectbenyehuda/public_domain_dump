


	
	חטאות סדום
	




וַיְהִי מִיָמִים וַיַשְחִיתוּ אַנְשֵי סְדוֹם וַעֲמוֹרָה
אֶת דַרְכָּם לִפְנֵי ה'. וַתְּהִי לָהֶם בְּאַרְצָם בִּקְעָה רְחָבָה
מְאֹד, כְּמַהֲלַךְ חֲצִי יוֹם, וּבָהּ מַעְיְנוֹת מַיִם וּדְשָאִים
רַבִּים. וְהָלְכוּ שָמָה יוֹשְבֵי סְדוֹם וְיֶתֶר עָרֵי הַכִּכָּר
אַרְבַּעַת יָמִים בַּשָנָה, הֵם וּנְשֵיהֶם וּבְנֵיהֶם וּבְנוֹתֵיהֶם,
וְעָרְכוּ שָם מִשְתִּים וּשְׂמָחוֹת בְּתֻפִּים וּבִמְחוֹלוֹת. וּכְטוֹב
לִבָּם בַּיַיִן וְעָמְדוּ וְגָזְלוּ אִיש אֶת אֵשֶת רֵעֵהוּ וְהָלְכוּ
לָהֶם, וְאֵין אוֹמֵר דָבָר. גַם כַּאֲשֶר בָּא אִיש נָכְרִי אֶל עָרֵיהֶם
וְהֵבִיא אִתּוֹ סְחוֹרָה לְמָכְרָהּ, אוֹ אֲשֶר קָנָה, וְהִתְקַבְּצוּ
אֵלָיו יוֹשְבֵי הָעִיר, אֲנָשִים נָשִים וָטַף, וְלָקְחוּ מִידֵי הָאִיש
אֶת סְחוֹרָתוֹ בְּחָזְקָה; וְכַאֲשֶר יָרִיב אִתָּם בַּעַל הַסְחוֹרָה עַל
הַגְזֵלָה אֲשֶר גָזְלוּ מִמֶנוּ וְגֵרְשוּהוּ מִן הָעִיר בְּחֶרְפָּה.
וַיְהִי אִיש אֶחָד מֵאֶרֶץ עֵילָם, עוֹבֵר־אֹרַח, וְעִמוֹ חֲמוֹר נוֹשֵׂא
שְׂמִיכָה אַחַת טוֹבָה וּצְבוּעָה בְּכָל מִינֵי צְבָעִים. וַיַעֲבֹר
הָאִיש לְדַרְכּוֹ וַתָּבוֹא לוֹ הַשֶמֶש בִּסְדוֹם, וַיֹאמֶר לָלוּן בָּהּ
הַלַיְלָה, וְאֵין אִיש מְאַסֵף אוֹתוֹ הַבָּיְתָה. וְשָם אִיש רָע
וּבְלִיַעַל וּשְמוֹ הֵידוֹד, וַיְהִי כִּרְאוֹתוֹ אֶת הָאוֹרֵחַ וַיֹאמֶר
לוֹ: מֵאַיִן תָּבוֹא וְאָנָה תֵלֵךְ? – וַיַעַן הַהֵלֶךְ: עוֹבֵר אָנֹכִי
מֵחֶבְרוֹן עַד עֵילָם, כִּי שָם בֵּיתִי. וַיֹאמֶר לוֹ הֵידוֹד: כָּל
מַחְסוֹרְךָ עָלַי, רַק בָּרְחוֹב אַל תָּלִין. וַיְבִיאוֹ הֵידוֹד
לְבֵיתוֹ וַיָלֶן שָם הָאִיש בַּלַיְלָה הַהוּא. וַיְהִי בַּבֹּקֶר
וַיַשְכֵּם הָאִיש לָלֶכֶת לְדַרְכּוֹ, וַיאֹמֶר לוֹ הֵידוֹד: סְעַד
לִבְּךָ וְאַחַר תֵּלֵךְ. וַיַעַשׂ הַהֵלֶךְ כֵּן וַיֹאכְלוּ וַיִשְתּוּ
שְנֵיהֶם יַחְדָו. וַיָקָם הָאִיש לָלֶכֶת, וַיאֹמֶר אֵלָיו הֵידוֹד: הִנֵה
רָפָה הַיוֹם לַעֲרֹב, הוֹאֶל נָא וָלִין אֶצְלִי גַם הַלַיְלָה
וְהֵיטִיבָה לִבֶּךָ. וַיִפְצַר בּוֹ וַיָלֶן הַהֵלֶךְ בְּבֵיתוֹ גַם
בַּלַיְלָה הַהוּא. וַיַשְכֵּם בַּיוֹם הַשֵנִי לָלֶכֶת, וַיַחֲזֵק בּוֹ
הֵידוֹד וַיאֹמֶר לוֹ: סְעַד נָא לִבְּךָ וְאַחַר תֵּלֵךְ. וַיֵשֶב הָאִיש
בְּבֵית הֵידוֹד גַם בַּיוֹם הַשֵנִי. וַיָקָם לִפְנוֹת עֶרֶב לָלֶכֶת
לְדַרְכּוֹ וַיאֹמֶר לוֹ הֵידוֹד: הִנֵה פָנָה הַיוֹם, שֵב אִתִּי
הַלַיְלָה – וּמָחָר תֵּלֵךְ לְדַרְכֶּךָ. וְלֹא אָבָה הָאִיש שְמֹעַ לוֹ
וַיָקָם וַיַחֲבֹש אֶת חֲמוֹרוֹ לָלֶכֶת, וַיַרְא וְהִנֵה אֵין הַשְׂמִיכָה
הַצִבְעוֹנִית וְהַחֶבֶל אֲשֶר הָיְתָה קְשוּרָה בוֹ אֶל הַחֲמוֹר.
וַיאֹמֶר הָאִיש אֶל הֵידוֹד: תְּנָה לִי אֶת שְׂמִיכָתִי וְאֶת הַחֶבֶל
אֲשֶר עָלֶיהָ וְהָלַכְתִּי לְדַרְכִּי. וַיַעַן הֵידוֹד: זֶה פִּתְרוֹן
חֲלוֹמְךָ אֲשֶר חָלָמְתָּ: הַחֶבֶל – יֶאֶרְכוּ חַיֶיךָ בָּאָרֶץ
כַּחֶבֶל, וְהַשְׂמִיכָה – כֶּרֶם יְהִי לְךָ אֲשֶר יְכַסֶה אֶת פְּנֵי
אַדְמָתְךָ בְּאֶשְכְּלוֹת עֲנָבִים כְּבַשְׂמִיכָה. וַיאֹמֶר הַהֵלֶך: לֹא
אֲדוֹנִי, כִּי בְהָקִיץ לָקַחְתָּ מִמֶנִי חֶבֶל וּשְׂמִיכָה וּבְהָקִיץ
הָשֵב תְּשִיבֵם לִי. וַיַעַן הֵידוֹד: הֲלֹא פָתַרְתִּי לְךָ חֲלוֹמְךָ
לְטוֹבָה, וְאַתָּה שַלֶם לִי בִּמְחִירוֹ שְלֹשָה שְקָלִים. וַיִצְעַק
הַהֵלֶךְ צְעָקָה גְדוֹלָה וַיָבֵא אֶת הֵידוֹד אֶל שוֹפֵט הָעִיר.
וַיִטְעַן הֵידוֹד לִפְנֵי הַשוֹפֵט לֵאמֹר: הִנֵה פָתַרְתִּי אֶת חֲלוֹם
הָאִיש הַנָכְרִי הַזֶה, וְהוּא מְמָאֵן לְשַלֵם לִי מְחִירוֹ. וַיאמֶר
הַשוֹפֵט: אֱמֶת וָצֶדֶק דוֹבֵר הֵידוֹד, כִּי שְמוֹ יָצָא לִתְהִלָה
לְפוֹתֵר־חֲלוֹמוֹת מֵאֵין כָּמוֹהוּ; וְאַתָּה, אוֹרֵחַ, שַלֵם תְּשַלֵם
לוֹ אֶת מְחִירוֹ. וַיִצְעַק הַהֵלֶךְ: לֹא, אֲדוֹנִי הַשוֹפֵט, כִּי לָקַח
מִמֶנִי הָאִיש הָרָע הַזֶה שְׂמִיכָה צִבְעוֹנִית וָחֶבֶל וְהוּא מְמָאֵן
לַהֲשִיבָם לִי. וַיאֹמֶר הֵידוֹד: גַם שַלֵם תְּשַלֵם לִי אֶת מְחִיר
הָאֹכֶל אֲשֶר אָכַלְתָּ בְּבֵיתִי יוֹמָיִם. וַיַעַן הַהֵלֶךְ: אֲנִי
אֲשַלֵם לְךָ אֶת מְחִיר הָאֹכֶל, אַךְ הָשֵב לִי אֶת שְׂמִיכָתִי וְאֶת
חַבְלִי. וַיאֹמֶר הֵידוֹד: זֶה פִּתְרוֹן חֲלוֹמֶךָ: חֶבֶל – אֲשֶר
יֶאֶרְכוּ יָמֶיךָ כְּחֶבֶל, וּשְׂמִיכָה – כֶּרֶם תִּטַע וַעֲנָבָיו
יְכַסוּ אֶת פְּנֵי אַדְמָתְךָ כְּבַשְׂמִיכָה. וַיָרִיבוּ שְנֵיהֶם
לִפְנֵי הַשוֹפֵט: זֶה טוֹעֵן בְּכֹה וְזֶה – בְּכֹה. וַיְהִי כַּאֲשֶר
נִשְמְעוּ קוֹלוֹתֵיהֶם גַם בַּחוּץ וַיִתְקַבְּצוּ מִסָבִיב לַהֵלֶךְ
אַנְשֵי סְדוֹם וַיְגָרְשוּהוּ מֵהָעִיר.

וּלְאַנְשֵי סְדוֹם וַעֲמוֹרָה הָיוּ אַרְבָּעָה שוֹפְטִים
לְאַרְבַּע עָרֵי הַכִּכָּר וְאֵלֶה שְמוֹתָם: שֶרֶק לְעִיר סְדוֹם,
שַרְקַר – לַעֲמוֹרָה, זַבְנָךְ לְאַדְמָה, מְנוֹן לִצְבוֹיִם. וַיַעֲשׂוּ
לָהֶם אַנְשֵי סְדוֹם אַרְבַּע מִטוֹת עַל שֵם אַרְבַּעַת שוֹפְטֵיהֶם,
וַיַעֲמִידוּן בִּרְחוֹב הָעִיר. וְהָיָה כָּל אִיש אֲשֶר יָבוֹא אֶל אַחַת
מֵהֶעָרִים הָאֵלֶה וֶהֱבִיאוּהוּ אֶל אַחַת הַמִטוֹת וְהִשְכִּיבוּהוּ
עָלֶיהָ בְּחָזְקָה. וְעָמְדוּ עָלָיו שְלֹשָה אֲנָשִים לִמְרַאֲשוֹתָיו
וּשְלֹשָה – לְמַרְגְלוֹתָיו וְהֶחֱזִיקוּ בוֹ וּמָדְדוּ אוֹתוֹ עַל
הַמִטָה. וְהָיָה אִם מִדַת הָאִיש הַהוּא קָצְרָה מִמִדַת הַמִטָה
וּמָתְחוּ אוֹתוֹ שֵשֶת הָאֲנָשִים, שְלֹשָה מִזֶה וּשְלשָה מִזֶה, עַד
אֲשֶר נָפְחָה נִשְמָתוֹ; וְאִם אָרְכָה מִדָתוֹ מִמִדַת הַמִטָה וְקָצְצוּ
אֶת כַּפּוֹת רַגְלָיו לְמַעַן תִּשְוֶינָה מִדוֹתֵיהֶם. וְכִי בָא אִיש
עָנִי אֶל עָרֵי סְדוֹם וְנָתְנוּ לוֹ יוֹשְבֵיהֶן כֶּסֶף וְזָהָב מְלֹא
חָפְנַיִם, אֲבָל אָסְרוּ עַל הָאוֹפִים וְעַל כָּל יוֹשְבֵי הָעִיר
לִמְכֹּר לוֹ אוֹ לָתֵת לוֹ פַּת לָחֶם. וְהָיָה כִּי יָשַב הָאֶבְיוֹן
בְּעָרֵי סְדוֹם יָמִים אֲחָדִים וּמֵת בָּרָעָב, וּבְמוֹתוֹ הוֹצִיאוּ
לָהֶם אַנְשֵי הָעִיר מִכִּיסוֹ אֶת כַּסְפָּם וּזִהָבָם אֲשֶר נָתְנוּ לוֹ
וְגַם אֶת כְּסוּתוֹ לָקָחוּ.

וַיְהִי מִיָמִים וַתִּשְלַח שָׂרָה אֵשֶת אַבְרָם אֶת
אֱלִיעֶזֶר עַבְדָם סְדוֹמָה אֶל לוֹט לְפָקְדוֹ לְשָלוֹם. וַיָבֹא
אֱלִיעֶזֶר סְדוֹמָה וַיַּרְא וְהִנֵה אֶחָד מֵאַנְשֵי הָעִיר פּוֹשֵט
מֵעַל אִיש עָנִי אֶת בִּגְדוֹ וְהוֹלֵךְ לוֹ. וַיִצְעַק הֶעָנִי אֶל
אֱלִיעֶזֶר כִּי יוֹשִיעוֹ בַצַר לוֹ. וַיאֹמֶר אֱלִיעֶזֶר אֶל הַסְדוֹמִי:
לָמָה תַּעֲשֶׂה כַּדָבָר הַזֶה לְאִיש עָנִי, אֲשֶר בָּא לִגְבוּל
אַרְצְכֶם? – וַיַעַן הַסְדוֹמִי: הֶאָחִיךָ הוּא אִם לְשוֹפֵט שָׂמוּךָ
פֹה אַנְשֵי הָעִיר כִּי תָרִיב אֶת רִיבוֹ? – וַיִגַש אֱלִיעֶזֶר אֶל
הַסְדוֹמִי לָקַחַת מִיָדוֹ אֶת בֶּגֶד הָאֶבְיוֹן, וַיַכֵּהוּ הַסְדוֹמִי
בְּאֶבֶן אֶל מִצְחוֹ, וַיָזָב מִמֶנוּ דָם. וַיַרְא הַסְדוֹמִי אֶת הַדָם
וַיַחֲזֵק בֶּאֱלִיעֶזֶר וַיאֹמֶר לוֹ: תְּנָה אֶת שְׂכָרִי כִּי הוֹצֵאתִי
דָם רַע מִמִצְחֶךָ. וְלֹא אָבָה אֱלִיעֶזֶר שְמוֹע לוֹ. וַיַחֲזֵק
הַסְדוֹמִי בֶּאֱלִיעֶזֶר וַיְבִיאֵהוּ אֶל שוֹפֵט הָעִיר. וַיאֹמֶר
הַשוֹפֵט אֶל אֱלִיעֶזֶר: תֶּן לוֹ אֶת שְׂכָרוֹ, כִּי חֹק הוּא
בְאַרְצֵנוּ: לְשַלֵם בְּעַד הַקָזַת דָם. וַיִשְמַע אֱלִיעֶזֶר אֶת
מִשְפַּט הַשֶקֶר וַיִקַח אֶבֶן וַיַךְ בָּהּ אֶת הַשוֹפֵט עַד שְפָךְ־דָם,
וַיאֹמֶר אֵלָיו: שַלֵם לוֹ מִכַּסְפִּי הַמַגִיעַ לִי מִמֶךָ…

וּבָעֵת הַהִיא יָלְדָה אֵשֶת לוֹט בַּת וַיִקְרָא אֶת
שְמָהּ פַּלְטִית, כִּי אָמַר: פִּלְטָנִי אֱלֹהִים מִידֵי מַלְכֵי עֵילָם.
וַתִּגְדַל פַּלְטִית וַתְּהִי לְנַעֲרָה טוֹבָה וְחוֹמֶלֶת־דַלִים.
וַיְהִי הַיוֹם וַיָבֹא סְדוֹמָה אִיש עָנִי לְבַקֵש שָם אֶת מִחְיָתוֹ
וַיֵשֶב בָּעִיר יָמִים אֲחָדִים. וַיַעֲבִירוּ אַנְשֵי סְדוֹם קוֹל
בָּעִיר כְּמִשְפָּטָם תָּמִיד, לְבִלְתִּי תֵּת לֶעָנִי הַהוּא פַּת לֶחֶם
עַד אֲשֶר יִפַּח נַפְשוֹ. וַתֵּרֶא פַּלְטִית בַּת לוֹט אֶת הֶעָנִי
הַהוּא מָשְלָךְ בָּרְחוֹב הָעִיר וְגוֵעַ בָּרָעָב, וַתַּחְמוֹל עָלָיו
וַתְּכַלְכְּלֵהוּ חֶרֶש בְּלֶחֶם וּמָיִם יָמִּים רַבִּים: וְהָיָה מִדֵי
צֵאתָהּ לִשְאֹב מַיִם מִן הַבְּאֵר וְשָמָה פַּת לֶחֶם בְּכַד הַמַיִם
וְנִגְשָה אֶל הֶעָנִי וְהוֹצִיאָה אֶת הַלֶחֶם מִתּוֹךְ הַכַּד
וּנְתָנַתּוּ לוֹ. וַיִתְמְהוּ אַנְשֵי סְדוֹם עַל הֶעָנִי הַהוּא, כִּי
חַי הוּא בְּלִי מָזוֹן זֶה יָמִים רַבִּים. וַיאֹמְרוּ אִיש אֶל רֵעֵהוּ:
אֵין זאֹת, כִּי אוֹכֵל וְשוֹתֶה הוּא, בַּאֲשֶר לֹא יוּכַל הָאָדָם
לִחְיוֹת בְּלִי אֹכֶל יָמִים רַבִּים וּפָנָיו לֹא יִדַלוּ.
וַיִסְתַּתְּרוּ שְלֹשָה מֵאַנְשֵי סְדוֹם וַיֶאֶרְבוּ לֶעָנִי וְלָאִיש
אֲשֶר יָבִיא לוֹ אֹכֶל. וַיִרְאוּ וְהִנֵה פַּלְטִית בַּת לוֹט יָצְאָה
אֶל הַבְּאֵר לִשְאֹב מַיִם וַתִּגַש אֶל הֶעָנִי וַתּוֹצֵא לֶחֶם
מִכַּדָהּ וַתִּתְּנֵהוּ לוֹ. וַיִתְפְּשׂוּ הַסְדוֹמִים אֶת פַּלְטִית
וַיְבִיאוּהָ אֶל הַשוֹפֵט וַיאֹמְרוּ אֵלָיו: כָּזאֹת וְכָזאֹת עָשְׂתָה
בַּת־לוֹט. וַיִתְקַבְּצוּ אַנְשֵי הָעִיר וַיִשְׂרְפוּהָ בָאֵש.

וּבְאַדְמָה הָיְתָה נַעֲרָה אַחַת, בַּת אֶחָד מֵעֲשִירֵי
הָעִיר. וַיְהִי הַיוֹם וַתֵּרֶא הַנַעֲרָה וְהִנֵה הֵלֶךְ יוֹשֵב נֶגֶד
פֶּתַח בֵּיתָם וְשֹאֵל מִמֶנָה מְעַט מָיִם, כִּי צָמֵא וְעָיֵף הָיָה
מַעֲמַל הַדָרֶךְ. וַתַּחְמוֹל עָלָיו הַנַעֲרָה וַתּוֹצֵא לוֹ בַסֵתֶר
לֶחֶם וּמָיִם. וַיִוָדַע הַדָבָר לְאַנְשֵי הָעִיר וַיָבִיאוּ אֶת
הַנַעֲרָה לִפְנֵי הַשוֹפֵט. וֵיאֹמֶר הַשוֹפֵט: מִשְפַּט מָוֶת
לְבַת־מַרְדוּת זוֹ. וַיוֹצִיאוּהוּ אַנְשֵי הָעִיר הַחוּצָה וַיַפְשִיטוּ
מֵעָלֶיהָ אֶת שִׂמְלָתָהּ וַיִמְשְחוּ אֶת בְּשָׂרָהּ וַיַצִיגוּהָ
לִפְנֵי עֲדַת דְבוֹרִים. וַיְכַסוּ הַדְבוֹרִים אֶת גוּף הַנַעֲרָה מִכַּף
רַגְלָהּ וְעַד קָדְקָדָהּ וַיִשְכוּהוּ עַד אֲשֶר צָבָה בְּשָׂרָהּ.
וַתִּצְעַק הַנַעֲרָה בְּמַכְאוֹבֶיהָ הַנוֹרָאִים, וַתַּעַל שַוְעָתָהּ
הַשָמָיְמָה.

וַיַרְא ה' אֶת הַמַעֲשִׂים הָרָעִים הָאֵלֶה וַיאֹמֶר אֶל
מַלְאָכָיו עוֹשֵי רְצוֹנוֹ: זַעֲקַת סְדוֹם וַעֲמוֹרָה כִּי רַבָּה –
כָּלָה אֶעֶשֶׂה אִתָּם". וַיִשְלַח ה' שְנַיִם מֵהַמַלְאָכִים,
וַיַשְמִידוּ אֶת עָרֵי סְדוֹם בָּאֵש וְגָפְרִית וְרַק לוֹט וּבְנוֹתָיו
נִמְלְטוּ מִשָם עַל נַפְשָם וַיֵלְכוּ וַיִתְיַשְּבוּ בִּמְעָרַת
עֲדוּלָם.
