


	
	חלום נמרד, כדרלעמר, רקיון = פרעה, שרי
	




וְאַבְרָהָם יָצָא מֵאֵת נִמְרֹד וַיִדְבְּקוּ בוֹ כִּשְלֹש מֵאוֹת אִיש מֵעַבְדֵי הַמֶלֶךְ. וַיֵלְכוּ אַחֲרָיו גַם כָּל בִּית אָבִיו וַיְלַמֵד אַבְרָם אֶת
כֻּלָם לַעֲזֹב אֶת עֲבוֹדַת הָאֱלִילִים וּלְהַאֲמִין רַק
בְּאֵל אֶחָד. בָּעֵת הַהִיא לָקְחוּ אַבְרָם וְנָחוֹר לָהֶם נָשִים
מִבְּנוֹת הָרָן אֲחִיהֶם, שֵׁם אֵשֶת נָחוֹר מִלְכָּה וְשֵם אֵשֶת אַבְרָם
– שָׂרָי.

מִקֵץ שְנָתַיִם יָמִים חָלַם נִמְרֹד חֲלוֹם: וְהִנֵה
הוּא עוֹמֵד עִם חֵילוֹ נֶגֶד כִּבְשַן הָאֵש, וְאַבְרָם יָצָא מִתּוֹךְ
הָאֵש וְחַרְבּוֹ שְלוּפָה בְּיָדוֹ. וַיָנָס הַמֶלֶךְ, וַיַשְלֵךְ אַבְרָם
עַל רֹאשוֹ בֵּיצָה וַתְּהִי הַבֵּיצָה לְנָהָר גָדוֹל וַיִטְבַּע בּוֹ
כָּל הַחַיִל, וְהַמֶלֶךְ בְּעַצְמוֹ נִמְלַט עַל נַפְשוֹ הוּא וּשְלֹשָה
מֵאֲנָשָיו עִמוֹ. וַיְהִי בְּבָרְחָם וַיָשָב הַנָהָר לְבֵיצָה
כְּבַתְּחִלָה וַיֵצֵא מִמֶנָה אֶפְרֹחַ קָטָן וַיָעָף עַל רֹאש הַמֶלֶךְ
וַיְנַקֵר אֶת עֵינָיו. וַיִיקַץ נִמְרֹד מִשְנָתוֹ וַתִּפָּעֶם רוּחוֹ
מְאֹד, וַיִשְלַח וַיִקְרָא לְכָל חַרְטֻמָיו וַיְסַפֵּר לָהֶם אֶת
חֲלוֹמוֹ הַמוּזָר. וַיַעַן חַרְטֹם אֶחָד וּשְמוֹ אֱנוֹקִי וַיֹאמֶר
לַמֶלֶךְ: “אֵין זֶה כִּי רָעַת אַבְרָם וְזַרְעוֹ אֲשֶר תִּצְמַח
בְּאַחֲרִית הַיָמִים עַל רֹאש אֲדוֹנִי הַמֶלֶךְ, כַּאֲשֶר חָזוּ
חֲכָמֶיךָ לִפְנֵי חֲמִשִים וּשְׁתַּיִם שָנָה. הִנֵה יָמִים בָּאִים
וְיִלָחֵם אַבְרָם עִם אֲדוֹנִי הַמֶלֶךְ וְהִכָּה אֶת כָּל צִבְאוֹתֶיךָ. וַאֲשֶר רָאִיתָ כִּי בָרַחְתָּ עִם שְלֹשָה
מֵאֲנָשֶיךָ – כִּי תִנָצֵל בַּמִלְחָמָה הַזֹאת רַק אַתָּה וְעוֹד שְלֹשֶת
מְלָכִים אֲשֶר יָבוֹאוּ לְעֶזְרָתֶךָ; וַאֲשֶר רָאִיתָ אֶת הָאֶפְרֹחַ
הַמְנַקֵר אֶת עֵינֵיךָ – אֵין זֶה כִּי אֶחָד מִזֶרַע אַבְרָם יַהֲרג אֶת הַמֶלֶךְ בְּאַחֲרִית הַיָמִים. וֶאֱמֶת
הַחֲלוֹם וְנָכוֹן הַפִּתְרוֹן. וְעַתָּה אִיעָצְךָ
לִתְפֹּשׂ אֶת אַבְרָם וְלַהֲמִיתוֹ כִּי כָל הָיָמִים אֲשֶר אַבְרָם חַי –
לֹא תִכּוֹן אַתָּה וּמַלְכוּתֶךָ”.

וַיִשְמַע נִמְרֹד בְּקוֹל אֱנוֹקִי וַיִשְלַח מְרַגְלֵי
חֶרֶש לִתְפֹּשׂ אֶת אַבְרָם וְלַהֲמִיתוֹ. וֶאֶלִיעֶזֶר עֶבֶד אַבְרָם,
אֲשֶר נָתַן לוֹ נִמְרד לְמַתָּנָה, עָמַד בָּעֵת הַהִיא לִפְנֵי הַמֶלֶךְ. וַיְהִי כְּשָמְעוֹ אֶת דִבְרֵי
אֱנוֹקִי וְאֶת עֲצַתוֹ הָרָעָה וַיָרָץ וַיַגֵד
לְאַבְרָם וַיֹאמֶר לוֹ: מַהֵר וְהִמָלֵט עַל נַפְשְךָ, כִּי כָזֶה וְכָזֶה
חָלַם הַמֶלֶךְ, וְכָזֶה פָתַר לוֹ אֱנוֹקִי יוֹעֲצוֹ. וַיִשְמַע אַבְרָם
בְּקוֹל אֶלִיעֶזֶר עַבְדוֹ וַיִבְרַח בַּסֵתֶר אֶל בֵּית שֵם בֶּן נֹחַ, וַיָבֹא שָמָה גַם תֶּרַח לִרְאוֹת אֶת בְּנוֹ. וַיֹאמֶר אַבְרָם אֶל תֶּרַח
אָבִיו: הֲלֹא יָדַעְתָּ כִּי מִתְנַחֵם נִמֱרֹד לְהָרְגֵנִי, וְעַתָּה מַה
לְךָ וּמִי לְךָ בְּאַרְצוֹ? קוּמָה וְנֵלְכָה יַחַד אַרְצָה כְנַעַן, פֶּן
תִּסָפֶה גַם אַתָּה בִּידֵי הַמֶלֶךְ הָרָשָע הַזֶה. כִּי לֹא מֵאַהֲבָתוֹ
אוֹתְךָ נָתַן לְךָ אֶת כָּל הַכָּבוֹד הַזֶה, כִּי אִם לְטוֹבָתוֹ וְלתוֹעַלְתוֹ הוּא, וְלֹא יוֹעִיל לְךָ הוֹנְךָ בְּיוֹם עֶבְרָה וָזָעַם.–וַיִשְמַע תֶּרַח בְּקוֹל
אַבְרָם וַיִקַח אֶת בְּנֵי בֵיתוֹ וְגַם אֶת לוֹט בֶּן הָרָן בְּנוֹ הַמֵת וַיֵצְאוּ מֵאוּר כַּשְׂדִים לָלֶכֶת אַרְצָה כְנָעַן וַיָבוֹאוּ עַד חָרָן וַיֵשְבוּ שָם, כִּי מָצְאוּ אֶת הָאָרֶץ טוֹבָה וְרַחֲבַת יָדָיִם. וַיִרְאוּ עַבְדֵי תֶרַח וְיוֹשְבֵי חָרָן אֶת אַבְרָם כִּי טוֹב וְיָשָר
הוּא עִם אֱלֹהִים וְעִם אֲנָשִים, וַיָבוֹאוּ אֵלָיו וַיְלַמְדֵם אַבְרָם
אֶת מוּסַר ה' וַיַעֲזְבוּ אֶת עֲבוֹדַת הַגִלוּלִים וַיִדְבְּקוּ
בְּתוֹרָתוֹ. וַיְהִי מִקֵץ שָלֹש שָנִים וַיֵרָא ה' אֶל אַבְרָם וַיֹאמֶר
אֵלָיו: אֲנִי ה' אֲשֶר הוֹצֵאתִיךָ מֵאוּר כַּשְׂדִים וְהִצַלְתִּיךָ
מִכַּף אוֹיְבֶיךָ; וְעַתָּה אִם שָמֹעַ תִּשְמַע בְּקוֹלִי וְשָמַרְתָּ
אֶת חֻקוֹתָי – וְהִרְבֵּיתִי אֶת זַרְעֲךָ
כְּכוֹכְבֵי הַשָמַיִם, וְשָלַחְתִּי לְךָ אֶת בִּרְכָתִי בְּכָל מַעֲשֶׂה
יָדֶיךָ. קוּם קַח אֶת אִשְתְּךָ וְאֶת כָּל אֲשֶר לְךָ וָלֵךְ
אַרְצָה כְנַעַן וְיָשַבְתָּ שָם. וַיָקָם אַבְרָם וַיִקַח אֶת שָׂרַי
אִשְתּוֹ וְאֶת כָּל אֲשֶר לוֹ וַיֵלֶךְ אַרְצָה כְנָעַן. רַק תֶּרַח
וְנָחוֹר וּבְנֵי בֵיתוֹ נִשְאֲרוּ בְחָרָן. וְאַבְרָם בֶּן חֲמִשִים
וּשְתַּיִם שָנָה בְּצֵאתוֹ מֵחָרָן.

מִקֵץ ג' שָנִים לְשֶבֶת אַבְרָם
בְּאֶרֶץ כְּנַעַן מֵת נֹחַ בֶּן תְּשַע מֵאוֹת וַחֲמִשִים שָנָה.
וּבַשָנָה הָעֲשִׂירִית לְשֶבֶת אַבְרָם בְּאֶרֶץ כְּנַעַן הָיְתָה
מִלְחָמָה בֵּין נִמְרֹד־אַמְרָפֶל מֶלֶךְ שִנְעָר וּבֵין כְּדָרְלָעֹמֶר
מֶלֶךְ עֵילָם, שַׂר צְבָאוֹ לְפָנִים. וַיִקְבֹּץ נִמְרֹד אֶת
כָּל חֵילוֹ וַיַעֲרֹךְ אִתּוֹ מִלְחָמָה בְּבִקְעַת
בָּבֶל. וַיִנָגֶף נִמרֹד לִפְנֵי כְדָרְלָעֹמֶר וַיִפְּלוּ מֵאֲנָשָיו
כְּשֵש מֵאוֹת אֶלֶף אִיש, וּבֵינֵיהֶם גַם מִרְדוֹן בְּנוֹ. וַיִבְרַח
נִמְרֹד וַיָשָב בְּבֹשֶת פָּנִים לְאַרְצוֹ וַיִכָּנַע מֵאָז
לִכְדָרְלָעֹמֶר.

וַיְהִי בִּשְנַת הַחֲמֵש עֶשְׂרֵה לְשֶבֶת אַבְרָם בְּאֶרֶץ כְּנַעַן, הִיא שְנַת ע' לְחַיָיו, וַיֵלֶךְ אַבְרָם חָרָנָה לִרְאוֹת אֶת תֶּרַח אָבִיו וְאֶת אִמוֹ וַיֵשֶב שָם כְּחָמֵש שָנִים. בָּעֵת הַהִיא מֵת
תֶּרַח בֶּן חָמֵש וּמָאתַיִם שָנָה, וְאַבְרָם שָב אַרְצָה כְּנַעַן הוּא וַאֲנָשָיו וְרַבִּים
מִבְּנֵי הָאָרֶץ אֲשֶר דָבְקוּ בַּה', וַיֵלֶךְ אִתּוֹ גַם לוֹט בֶּן
הָרָן אָחִיו, וַיִטוּ אָהֳלֵיהֶם עַד אֵלוֹן מוֹרֶה אֲשֶר בִּסְבִיבוֹת שְכֶם וַיֵשְבוּ שָם.

בַּיָמִים הָהֵם הָיָה בְּאֶרֶץ שִנְעָר אִיש חָכָם וִיפֵה־תֹאַר, אַךְ עָנִי מְאֹד, וּשְמוֹ
רֵקְיוֹן. וַיֵצֶר לָאִיש עַל מִחְיָתוֹ, וַיִמָלֵךְ לָרֶדֶת מִצְרַיְמָה
אֶל הַמֶלֶךְ אַשְוִירַש בֶּן עֲנָם לְהַרְאוֹתוֹ אֶת חָכְמָתוֹ, אוּלַי
יִמְצָא חֵן בְּעֵינָיו וּסְפָחָהוּ אֶל אַחַת מִמִשְׂרוֹת הַמַמְלָכָה.
וַיְהִי כְּבוֹא רֵקְיוֹן מִצְרַיְמָה וַיַגִידוּ לוֹ כִּי מִשְפַּט
הַמֶלֶךְ הוּא לָצֵאת מֵאַרְמוֹנוֹ וּלְהֵרָאוֹת לְעֵינֵי בְּנֵי עַמוֹ רַק
יוֹם אֶחָד בַּשָנָה. כִּשְמֹעַ רֵקְיוֹן אֶת הַדָבָר הַזֶה וַיִתְעַצֵב אֶל לִבּוֹ. וַיְהִי בָעֶרֶב וַיָבֹא
אֶל בַּיִת שָמֵם וַיָלֶן שָם מַר־נֶפֶש וְרָעֵב מְאֹד. אַךְ שְנָתוֹ
נָדְדָה מֵעֵינָיו, כִּי חָשַב כָּל הַלַיְלָה מַחֲשָבוֹת בַּמֶה יִחַיֶה
אֶת נַפְשוֹ עַד הֵרָאוֹתוֹ אֶת פְּנֵי הַמֶלֶךְ. וַיָקָם רֵקְיוֹן
בַּבֹּקֶר וַיִתְהַלֵךְ בִּרְחוֹבוֹת הָעִיר וַיִפְגַע בְּמוֹכְרֵי
יְרָקוֹת וַיִשְאַל לָהֶם וּלְמִחְיָתָם. וַיַגִידוּ לוֹ כִּי הֵם
מְכַלְכְּלִים אֶת נַפְשָם בִּירָקוֹת וּבְזֵרְעוֹנִים, אֲשֶר הֵם קוֹנִים וּמוֹכְרִים לְתוֹשְבֵי
הָעִיר. וַיַחְפֹּץ גַם רֵקְיוֹן לַעֲשׂוֹת כְּמוֹהֶם, אַךְ לֹא יָדַע אֶת
מִשְפְּטֵי הָאָרֶץ. וַיִתְקַבְּצוּ עָלָיו בְּנֵי בְלִיַעַל וַיִתְקַלְסוּ
בוֹ, וַיַחְטְפוּ מִיָדוֹ אֶת הַיְרָקוֹת, אֲשֶר
נָתְנוּ לוֹ סוֹחֲרֵי הַיָרָק לִמְכִירָה. וַיֵלֶךְ מִשָם רֵקְיוֹן
בְּמַר־נַפְשוֹ וַיָבֹא אֶל בֵּית־אוֹפֶה וַיִשְכַּב שָם. וַיִתְיָעֵץ כָּל
הַלַיְלָה הַהוּא אֵיכָה יְמַלֵט אֶת נַפְשוֹ מֵחֶרְפַּת רָעָב. וַיְהִי
בַּבֹּקֶר וַיִשְׂכֹּר רֵקְיוֹן שְלֹשִים אִיש גֳבּוֹרֵי חַיִל מְזֻיָנִים
וַיוֹלִיכֵם אֶל חֲצַר־הַקְבָרוֹת וַיְצַוֵם לֵאמֹר: כֹּה אָמַר הַמֶלֶךְ:
אלַ תִּתְּנוּ לְכָל מֵת לְהִקָבֵר פֹּה בִּלְתִּי
אִם יְשַלְמוּ קְרוֹבָיו בַּעֲדוֹ מָאתַיִם כֶּסֶף מַס לַגֻלְגֹלֶת.
וַיַעֲשׂוּ הָאֲנָשִים כִּדְבַר רֵקְיוֹן. לֹא אָרְכוּ הַיָמִים
וַיֶאֶסְפוּ רֵקְיוֹן וַאֲנָשָיו כֶּסֶף וְזָהָב וְאַבְנֵי שׁהַם וּבְדֹלַח
לְאֵין מִסְפָּר. וַיְהִי לִתְקוּפַת הַשָנָה, עֵת צֵאת הַמֶלֶךְ אֶל בְּנֵי עַמוֹ, וַיִתְקַבְּצוּ יוֹשְבֵי מִצְרַיִם
וַיִצְעֲקוּ אֵלָיו לֵאמֹר: מָה הַמַעֲשֶׂה אֲשֶר אַתָּה עוֹשֶׂה
לַעֲבָדֶיךָ לָקַחַת מַס גַם מִן הַחַיִים גַם מִן הַמֵתִים?! – כִּשְמֹעַ
הַמֶלֶךְ אֶת הַדְבָרִים הָאֵלֶה וַיִתְמַה מְאֹד, כִּי לֹא יָדַע עַל
אוֹדוֹת הַדָבָר הַזֶה מִטוֹב וְעַד רָע, וַיֹאמֶר אֶל בְּנֵי עַמוֹ: מִי
הוּא זֶה וְאֵיזֶה הוּא אֲשֶר מְלָאוֹ לִבּוֹ לַעֲשׂוֹת כֵּן וְאָנֹכִי לֹא צִוִיתִיו?! – וְיַגִידוּ לוֹ אֶת
מַעֲשֶׂה רֵקְיוֹן וַאֲנָשָיו, וַיְצַו הַמֶלֶךְ
לְהֲבִיאֵהוּ לְפָנָיו. וְרֵקְיוֹן שָׂכַר אֶלֶף יְלָדִים וִילָדוֹת
וַיַלְבִּישֵם שֵש, מֶשִי וְרִקְמָה וַיַרְכִּיבֵם עַל אֶלֶף עֲיָרִים
וַיִשְלַח בִּידֵי כָל אֶחָד מִנְחָה לַמֶלֶךְ;
וַיָבֹא אַחֲרֵיהֶם גַם הוּא וַיִשְתַּחוּ לְפָנָיו אַפַּיִם אָרְצָה.
וַתִּיטַב בְּעֵינֵי הַמֶלֶךְ מִנְחַת רֵקְיוֹן וַיִשְאַל לוֹ וּלְמַעֲשֵׂהוּ. וַיַעַן רֵקְיוֹן אֶת הַמֶלֶךְ בְּחָכְמָה וּבְטוּב־טַעַם, וַיִמְצָא חֵן בְּעֵינָיו,
וַיֹאמֶר לוֹ הַמֶלֶךְ: לֹא רֵקְיוֹן יִקָרֵא עוֹד שְמֶךָ, כִּי אִם
פַּרְעֹה, כִּי פָרַעְתָּ מַס גַם מִן
הַמֵתִים. וַיִתְיָעֲצוּ הַמִצְרִים לְהַמְלִיךְ עֲלֵיהֶם אֶת פַּרְעֹה
בִּמְקוֹם אַשְוִירַש, וַיַעֲשׂוּ כֵן. וַיִמְלֹךְ פַּרְעֹה עַל
אֶרֶץ מִצְרַיִם כָּל הַשָנָה וַיִשְפֹּט אֶת
עַמוֹ בְּהַשְׂכֵּל וּבְצֶדֶק מִדֵי יוֹם בְיוֹמוֹ, וְאַשְוִירַש–רַק יוֹם
אֶחָד בַּשָנָה, כְּמִשְפָּטוֹ תָמִיד.

וַיֶאֶהְבוּ מְאֹד הַמִצְרִים אֶת רֵקְיוֹן־פַּרְעֹה וַיִקְרְאוּ מֵאָז לְמַלְכֵיהֶם בְּשֵם פַּרְעֹה עַד הַיוֹם הַזֶה.

וּבַשָנָה הַהִיא הָיָה רָעָב כָּבֵד בְּאֶרֶץ כְּנַעַן
וְלֹא יָכְלוּ יוֹשְבֵי הָאָרֶץ לָשֶבֶת שָם מִפְּנֵי הָרָעָב וַיָקָם גַם
אַבְרָם הוּא וְכָל אֲשֶר לוֹ וַיֵצְאוּ מֵאֶרֶץ כְּנַעַן לָרֶדֶת
מִצְרַיְמָה, וַיַחֲנוּ עַל שְׂפַת הַנָחַל. וַיאֹמֶר אַבְרָם אֶל שָׂרַי:
הִנֵה נָא יָדַעְתִּי כִּי אִשָה יְפַת־תֹאַר וִיְפַת־מַרְאֶה אַתְּ,
וְהָיָה כִּי יִרְאוּךְ הַמִצְרִים וַהֲרָגוּנִי וְאוֹתָךְ יִקְחוּ לָהֶם,
כִּי אֵין יִרְאַת אֱלֹהִים בַּמָקוֹם הַזֶה.
אִמְרִי נָא: אֲחוֹתִי אַתְּ לְכָל אֲשֶר יִשְאָלוּךְ עָלַי וְחָיְתָה
נַפְשִי בִּגְלָלֵךְ. וַיְצַו אַבְרָם גַם לְכָל אֲנָשָיו אֲשֶר הָלְכוּ
אִתּוֹ לֵאמֹר: כִּי יִשְאֲלוּכֶם הַמִצְרִים: מִי הִיא הָאִשָה הַזֹאת –
וַאֲמַרְתֶּם: אֲחוֹת אַבְרָם הִיא. וּבְכָל זֹאת לֹא בָטַח אַבְרָם
בַּדְבָרִים הָאֵלֶה וַיִקַח אֶת שָׂרַי אִשְתּוֹ
וַיַסְתִּירָהּ בַּתֵּיבָה לְמַעַן לֹא יִרְאוּהָ הַמִצְרִים בְּבוֹאָם
אֶל הָאָרֶץ. הֵמָה בְּשַעֲרֵי הָעִיר וַיֹאמְרוּ
הַשוֹמְרִים לְאַבְרָם: תְּנָה מַעֲשֵׂר לַמֶלֶךְ מִכָּל אֲשֶר בְּיֶדְכֶם
וְאַחַר תָּבוֹאוּ הָעִירָה. וַיַעַשׂ אַבְרָם כַּאֲשֶר צִווּ עָלָיו
שוֹמְרֵי־הָעִיר. וַיְהִי כִּרְאוֹתָם אֶת הַתֵּיבָה וַיֹאמְרוּ אֶל
אַבְרָם: פְּתַח נָא אֶת הַתֵּיבָה וְנָתַתָּ גַם מִכָּל אֲשֶר בְּתוֹכָהּ
מַעֲשֵׂר לַמֶלֶךְ. וַיאֹמֶר אַבְרָם: כָּל אֲשֶר תֹּאמְרוּ עָלַי
אֶתֵּן, רַק אֶת הַתֵּיבָה אַל תִּפְתָּחוּ. וַיִגְשוּ עַבְדֵי פַרְעֹה אֶל הַתֵּיבָה
וַיִפְתְּחוּהָ בְּחָזְקָה וַיִרְאוּ וְהִנֵה אִשָה יְפַת־מַרְאֶה יוֹשֶבֶת
בְּתוֹכָהּ, וַיִשְתּוֹמְמוּ לְיָפְיָהּ וְהוֹדָהּ וַיְהַלְלוּ אוֹתָהּ אֶל
הַמֶלֶךְ. כִּרְאוֹת פַּרְעֹה אֶת שָׂרַי וְאֶת יָפְיָהּ הַמַפְלִיא
וַתִּיטַב מְאֹד בְּעֵינָיו וַתֻּקַח אֶל בֵּיתוֹ. וַיָבֹא מַלְאַךְ ה'
וַיְנַגַע אֶת פַּרְעֹה וְאֶת כָּל בֵּיתוֹ נְגָעִים גְדוֹלִים עַל דְבַר
שָׂרַי אֵשֶת אַבְרָם. וַיִקְרָא פַרְעֹה לְאַבְרָם וַיָשֶב לוֹ אֶת
שָׂרַי אִשְתוֹ, וַיִתֶן לָהֶם כֶּסֵֶף וְזָהָב וּמִקְנֶה
רָב. וְגַם אֶת הָגָר בִּתּוֹ, אֲשֶר יָלְדָה לוֹ פִּילַגְשוֹ, נָתַן
פַּרְעֹה לְשָׂרַי לְמִנְחָה. וְכָכָה אָמַר פַּרְעֹה לְהָגָר: טוֹב לָךְ,
בִּתִּי, לִהְיוֹת אָמָה בְּבֵית הָאִשָה הַזֹאת מִהְיוֹתֵךְ גְבִירָה
בְּבֵיתִי."
