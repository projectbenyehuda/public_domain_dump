


	
	מלחמת בני שעיר עם בני עשו
	




וַיְהִי אַחֲרֵי הִקָבֵר יַעֲקֹב וַיוֹסִיפוּ בְּנֵי
עֵשָׂו לְהִלָחֵם עִם בְּנֵי יַעֲקֹב בְּחֶבְרוֹן. וַתְּהִי הַמִלְחָמָה
חֲזָקָה וַיִנָגְפוּ שֵנִית בְּנֵי עֵשָׂו בַּמִלְחָמָה וַיִפְּלוּ מֵהֶם
שְמוֹנִים אִיש. וַתִּגְבַּר יַד יוֹסֵף וְגִבּוֹרָיו וַיִשְבּוּ אֶת צְפוֹ
בֶּן אֱלִיפַז בֶּן עֵשָׂו וַחֲמִשִים אִיש מֵעֲבָדָיו וַיַאַסְרוּם
בְּכַבְלֵי בַרְזֶל, וְהַנִשְאָרִים לָקְחוּ אֶת גְוִיַת עֵשָׂו
וַיִבְרְחוּ לְנַפְשָם. וַיְהִי בַּיוֹם הַשְלִישִי לַמִלְחָמָה
וַיֶאֱסְפוּ בְנֵי עֵשָׂו אֶת כָּל אֲחֵיהֶם וְאֶת כָּל בְּנֵי שֵׂעִיר
וְאֶת בְּנֵי קֶדֶם, כַּחוֹל אֲשֶר עַל שְׂפַת הַיָם, וַיֵרְדוּ
מִצְרַיְמָה לְהִלָחֵם עִם יוֹסֵף וְאֶחָיו וּלְהַצִיל אֶת אֲחִיהֶם
מִשִבְיָם. וַיִשְמַע יוֹסֵף וְכָל בֵּית יַעֲקֹב אֶת הַדָבָר הַזֶה
וַיֵצְאוּ לִקְרַאת בְּנֵי עֵשָׂו וּבְנֵי שֵׂעִיר וַיִלָחֲמוּ עִמָהֶם
בִּמְדִינַת רַעַמְסֵס. וַיַךְ עֵשָׂו וְאֶחָיו אֶת בְּנֵי שֵׂעִיר ואת
בְּנֵי קֶדֶם מַכָּה רַבָּה, לֹא נוֹתַר מֵהֶם בִּלְתִּי אִם
מְתֵי-מִסְפָּר אֲשֶר נִמְלְטוּ עַל נַפְשָם. וַיְהִי כַּאֲשֶר שָבוּ
הַפְּלִיטִים אַרְצָה שֵׂעִיר וַיִרְאוּ בְּנֵי שֵׂעִיר כִּי נָפְלוּ
חֲלָלִים כָּל אֲחֵיהֶם בִּגְלַל בְּנֵי עֵשָׂו, וַתִּבְעַר בָּהֶם חֲמָתָם
עַל בְּנֵי עֵשָׂו וַיאֹמְרוּ לָהֶם: צְאוּ לָכֶם מֵאַרְצֵנוּ וּלְכוּ
אַרְצָה כְּנַעַן, אֶל אֶרֶץ מְגוּרֵי אֲבוֹתֵיכֶם, כִּי לָמָה יִירְשוּ
בְנֵיכֶם אֶת בָּנֵינוּ בְאַחֲרִית הַיָמִים? – וְלֹא אָבוּ בְּנֵי עֵשָׂו
לִשְמוֹעַ בְּקוֹל בְּנֵי שֵׂעִיר, וַיִשְלְחוּ חֶרֶש אֶל אוֹגִיאַש מֶלֶךְ
קַרְתַּגָה לֵאמֹר: שְלַח לָנוּ אֲנָשִים מֵאֲנָשֶיךָ וְנִלָחֲמָה יַחַד
עִם בְּנֵי שֵׂעִיר הַחוֹרִי, כִּי נוֹעֲצוּ לְהִלָחֵם אִתָּנוּ
וּלְגָרְשֵנוּ מִן הָאָרֶץ הַזֹאת.

וַיַעַשׂ אוֹגִיאַש אֶת בַּקָשָתוֹ,
כִּי אוֹהֵב הָיָה אוֹגִיאַש לְעֵשָׂו וּלְבָנָיו, וַיִשְלַח לִבְנֵי
עֵשָׂו חֲמֵש מֵאוֹת אִיש רַגְלִי גִבּוֹרֵי חַיִל וּשְמוֹנֶה מֵאוֹת
רוֹכְבֵי סוּסִים. וּבְנֵי שֵׂעִיר שָלְחוּ אֶל בְּנֵי קֶדֶם וְאֶל מִדְיָן
לֵאמֹר: אַתֶּם רְאִיתֶם אֶת אֲשֶר עָשׂוּ לָנוּ בְּנֵי עֵשָׂו, אֲשֶר
בִּגְלָלָם תַּמְנוּ כֻלָנוּ בַּמִלְחָמָה עִם בְּנֵי-יַעֲקֹב; וְעַתָּה
בֹּאוּ אֵלֵינוּ וְעִזְרוּנוּ וּנְגָרְשֵם מֵאַרְצֵנוּ וְנִנָקְמָה מֵהֶם
אֶת נִקְמַת אַחֵינוּ וַאֲחִיכֶם הַחֲלָלִים. וַיִשְמְעוּ כָּל בְּנֵי
קֶדֶם לְדִבְרֵי בְּנֵי שֵׂעִיר וַיִתְקַבְּצוּ כִשְמוֹנֶה מֵאוֹת אִיש
שוֹלֵף חֶרֶב וַיִלָחֲמוּ בִּבְנֵי עֵשָׂו בְּמִדְבַּר פָּאָרן. וּמָאתַיִם
אִיש מִבְּנֵי עֵשָׂו בָּגְדוּ בַאֲחֵיהֶם וַיִפְּלוּ אֶל מַחֲנֵה
אוֹיְבֵיהֶם בְּנֵי שֵׂעִיר לַעֲזֹר לָהֶם. וַיִגְבְּרוּ בְּנֵי שֵׂעִיר
עַל בְּנֵי עֵשָׂו וַיַכּוּ מֵאַנְשֵי אוֹגִיאַש מָאתַיִם אִיש. וַיוֹסֶף
אוֹגִיאַש מֶלֶךְ קַרְתַּגָה וַיִשְלַח לִבְנֵי עֵשָׂו עוֹד שֵש מֵאוֹת
אִיש מִמִבְחַר גִבּוֹרָיו וַיַעַרְכוּ מִלְחָמָה עִם בְּנֵי שֵׂעִיר
וַיַכּוּם לְפִי חָרֶב. וְכָל בְּנֵי קֶדֶם וּמִדְיָן רָאוּ כִּי נִגְפוּ
בְּנֵי שֵׂעִיר בַּמִלְחָמָה וְלֹא נִשְאַר מֵהֶם שָׂרִיד וּפָלִיט
בִּלְתִּי אִם יַלְדֵיהֶם הַקְטַנִים, וַיַעַזְבוּ אֶת מַעֲרֶכֶת
הַמִלְחָמָה וַיִמָלְטוּ אֶל אַרְצָם. וַיִירְשוּ בְּנֵי עֵשָׂו אֶת כָּל
אֶרֶץ שֵׂעִיר וַיֵאָחֲזוּ בָהּ, וַיִקְרְאוּ לָהּ אֶרֶץ אֱדוֹם עַד הַיוֹם
הַזָה. וַיַחְלְקוּ בְּנֵי עֵשָׂו (אֱדוֹם) אֶת אַרְצָם לְחָמֵש מַחְלְקוֹת
לְפִי מִשְפְּחוֹתֵיהֶם וּבָתֵּי אֲבוֹתָם. וַיְהִי בַּיָמִים הָהֵם
וַיִתְיָעֲצוּ בְּנֵי עֵשָׂו לְהַמְלִיךְ עֲלֵיהֶם מֶלֶךְ אֲשֶר יִלָחֵם
אֶת מִלְחֲמוֹתֵיהֶם בְּאוֹיבֵיהֶם אֲשֶר מִסָבִיב. וַיִשָבְעוּ לֵאמֹר:
אֲשֶר לֹא יִמְלֹךְ עֲלֵיהֶם אִיש מֵאֲחֵיהֶם עַד עוֹלָם, בִּלְתִּי אִם
אִיש נָכְרִי. כִּי מָרָה נֶפֶש בְּנֵי עֵשָׂו אִיש עַל אָחִיו וְאִיש עַל
בְּנוֹ מִפְּנֵי הָרָעָה אֲשֶר עָשׂוּ לָהֶם אֲחֵיהֶם בְּנֵי עַמָם
בְּנָפְלָם אֶל בְּנֵי שֵׂעִיר וְעָזְרוּ לָהֶם בַּמִלְחָמָה. וַיְהִי שָם
אִיש אֶחָד מֵאַנְשֵי אוֹגִיאַש מֶלֶךְ קַרְתַּגָה וּשְמוֹ בֶּלַע
בֶּן-בְּעוֹר, וְהָאִיש גִבּוֹר חַיִל וִיפֵה תֹאַר וְחָכָם מְחֻכָּם
וּבַעַל עֵצָה וְתַחְבֻּלָה. וַיִקְחוּהוּ בְּנֵי עֵשָׂו וַיִמְשְחוּהוּ
לְמֶלֶךְ עֲלֵיהֶם, וַיִמְלֹךְ בֶּלַע עַל בְּנֵי עֵשָׂו שְלֹשִים
שָנָה.

וַיְהִי בִּשְנַת שְתַּיִם וּשְלֹשִים שָנָה לְרֶדֶת
בְּנֵי יִשְׂרָאֵל (יַעֲקֹב) מִצְרַיִמָה וַיָמָת מֶלֶךְ מִצְרַיִם
וַיִמְלֹךְ מִגְרוֹן בְּנוֹ תַּחְתָּיו. וַיִקְרְאוּ הַמִצְרִים גַם אֶת
שְמוֹ פַּרְעֹה כְּמִשְפָּטָם לְכָל מַלְכֵיהֶם, וְיוֹסֵף הָיָה מִשְנֵהוּ
וְהַמוֹצִיא וְהַמֵבִיא בְּכָל אֶרֶץ מִצְרַיִם, וְכָל מַלְכֵי כְנַעַן
וְעֵבֶר הַיַרְדֵן הָיוּ לוֹ לְמַס עוֹבֵד. וּבְנֵי יִשְׂרָאֵל יָשְבוּ
לָבֶטַח בְּאֶרֶץ גֹשֶן וַיִפְרוּ וַיִרְבּוּ וַיַעַצְמוּ בִּמְאֹד מְאֹד,
וַיַעַבְדוּ אֶת ה' אֱלֹהֵי אֲבוֹתָם כַּאֲשֶר צִוָם יַעֲקֹב אֲבִיהֶם
לִפְנֵי מוֹתוֹ.

וַיְהִי מִקֵץ יָמִים רַבִּים, כְּשֶבֶת בֶּלַע בֶּן
בְּעוֹר עַל כִּסֵא מַלְכוּתוֹ בְּאֶרֶץ אֱדוֹם, וּבְנֵי עֵשָׂו פָּרוּ
וְרָבוּ וַיַעַצְמוּ אָז מְאֹד, וַיִתְיָעֲצוּ לָרֶדֶת מִצְרַימָה
לַמִלְחָמָה עִם בְּנֵי יַעֲקֹב וּלְהַצִיל אֶת צְפוֹ בֶּן עֵשָׂו אֲחִיהֶם
וְאֶת אֲנָשָיו מִשִבְיָם. וַיִשְלְחוּ בְּנֵי עֵשָׂו מַלְאָכִים אֶל
בְּנֵי קֶדֶם וּבְנֵי יִשְמָעאֵל וַיִכְרְתוּ אִתָּם בְּרִית. וַיִתְקַבֵּץ
עַם רַב מְאֹד אֱדוֹמָה לַעֲלוֹת לַמִלְחָמָה עַל בְּנֵי יַעֲקֹב
בְּמִצְרָיִם. וַיְהִי הַמַחֲנֶה גָדוֹל מְאֹד כִּשְמוֹנֶה מֵאוֹת אֶלֶף
אִיש רַגְלִים וּפָרָשִים. וַיֵרְדוּ כָל הַגְדוּדִים הָאֵלֶה מִצְרַיְמָה
לְהִלָחֵם עִם בְּנֵי יַעֲקֹב וַיַחֲנוּ עַל רַעַמְסֵס. וַיֵצֵא יוֹסֵף
וְגִבּוֹרָיו וְאֶחָיו וְכָל בְּנֵיהֶם וְעַבְדֵיהֶם וַיִלָחֲמוּ בִּבְנֵי
עֵשָׂו וּבַעֲלֵי בְרִיתָם, וַיִנָגְפוּ בְּנֵי עֵשָׂו לִפְנֵי בְּנֵי
יִשְׂרָאֵל וַיִפְּלוּ מֵהֶם חֲלָלִים רַבִּים כְּמָאתַיִם אֶלֶף אִיש
וְגַם בֶּלַע בֶּן בְּעוֹר נָפַל חָלָל בַּמִלְחָמָה. וַיִרְאוּ בְּנֵי
עֵשָׂו כִּי מֵת מַלְכָּם וַתִּרְפֶּינָה יְדֵיהֶם – וַיָנוּסוּ עַל
נַפְשָם. וַיַמְלִיכוּ בְּנֵי עֵשָׂו עֲלֵיהֶם אַחֲרֵי מוֹת בֶּלַע אֶת
יוֹבָב בֶּן זֶרַח מִבָּצְרָה, וַיִמְלֹךְ בְּאֶרֶץ אֱדוֹם עֶשֶׂר שָנִים.
וְלֹא יָסְפוּ בְנֵי עֵשָׂו לְהִלָחֵם עִם בְּנֵי יַעֲקֹב. וַיְהִי אַחֲרֵי
מוֹת יוֹבָב וַיִמְשְחוּ בְּנֵי אֱדוֹם לְמֶלֶךְ עֲלֵיהֶם אִיש אֶחָד
מֵאֶרֶץ תֵּימָן וּשְמוֹ חוּשָם, וַיִמְלֹךְ בְּאֶרֶץ אֱדוֹם עֶשְׂרִים
שָנָה.

וְיוֹסֵף מוֹשֵל בְּמִצְרַיִם וְכָל בְּנֵי יִשְרָאֵל
יָשְבוּ אָז לָבֶטַח עַל אַדְמָתָם, אֵין שָׂטָן וְאֵין פֶּגַע רָע –
וַיְהִי שְלוֹם אֱמֶת בֵּינֵיהֶם וּבֵין יֶתֶר הָעַמִים.
