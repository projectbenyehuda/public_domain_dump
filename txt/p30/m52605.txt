


	
	מגפת בני אפרים
	




וּבִשְנַת מֵאָה וּשְמוֹנִים לְרֶדֶת בְּנֵי יִשְׂרָאֵל
מִצְרַיְמָה יָצְאוּ מִגֹשֶן שְלֹשִים אֶלֶף אִיש מִגִבּוֹרֵי בְנֵי
אֶפְרַיִם בֶּן יוֹסֵף, כִּי אָמָרוּ: תַּם הַקֵץ אֲשֶר יָעַד ה' אֱלֹהִים
לִבְנֵי יִשְׂרָאֵל בְּשִעְבוּד מִצְרַיִם וּלְשוּבָם אַרְצָה כְּנָעַן.
וַיִתְאַזְרוּ הָאֲנָשִים הָאַמִיצִים הָאֵלֶה וַיָשִׂימוּ אִיש־אִיש
כְּלֵי מִלְחָמְתּוֹ עָלָיו וַיִבְטְחוּ בִגְבוּרָתָם וַיֵצְאוּ יַחַד
מִמִצְרַיִם בְּיָד חֲזָקָה. וְלֹא לָקְחוּ אִתָּם גַם צֵידָה לַדֶרֶךְ
זוּלָתִי כֶסֶף וְזָהָב, כִּי אָמְרוּ לִקְנוֹת לָהֶם לֶחֶם בִּמְחִיר
מִבְּנֵי הַפְּלִשְתִּים אֲשֶר בִּגְבוּל אֶרֶץ כְּנָעַן. וַיֵלְכוּ יַחַד
וַיָשִׂימוּ פְּנֵיהֶם אַרְצָה כְּנַעַן דֶרֶךְ גַת. וַיִמְצְאוּ שָם
רוֹעֵי מִקְנֶה וַיאֹמְרוּ אֲלֵיהֶם: מִכְרוּ לָנוּ מִצֹאנְכֶם, כִּי
רְעֵבִים אֲנָחְנוּ. וַיְמָאֲנוּ רוֹעֵי גַת לִמְכּוֹר לִבְנֵי אֶפְרַיִם
מִצֹאנָם וּמִבְּקָרָם, וַיִגְשוּ בְנֵי אֶפְרַיִם לָקַחַת מִמִקְנֵה
הַפְּלִשְתִּים בְּחָזְקָה. וַיִצְעֲקוּ רוֹעֵי גַת לֶעֶזְרָה וַיַחְגְרוּ
אֶת כְּלֵי מִלְחַמְתָּם וַיִלָחֲמוּ עִם בְּנֵי אֶפְרַיִם, וַיִפְּלוּ
חֲלָלִים רַבִּים מִשְנֵי הַמַחֲנוֹת. וַיְהִי בַּיוֹם הַשֵנִי וַיִשְלְחוּ
בְנֵי גַת מַלְאָכִים אֶל יֶתֶר עָרֵי פְלִישְתִים לָבוֹא אֲלֵיהֶם
לְעֶזְרָה בְּמִלְחַמְתָּם עִם בְּנֵי אֶפְרָיִם. וַיָבֹאוּ מִכָּל עָרֵי
פְלִשְתִּים בְּאַרְבָּעִים אֶלֶף אִיש חֲלוּצֵי צָבָא וַיַכּוּ מַכָּה
רַבָּה אֶת בְּנֵי אֶפְרַיִם, אֲשֶר עָיְפָה נַפְשָם בְּרָעָב וּבַצָמָא,
כִּי לֹא אָכְלוּ וְלֹא שָתוּ זֶה שְלֹשֶת יָמִים. וְלֹא נִשְאָר
מִשְלֹשִים אֶלֶף בְּנֵי אֶפְרַיִם שָׂרִיד וּפָלִיט, זוּלָתִי עֲשָׂרָה
אֲנָשִים, אֲשֶר בָּרְחוּ עַל נַפְשָם וְשָבוּ גֹשְנָה; כִּי מֵאֵת ה'
הָיְתָה זֹאת לִבְנֵי אֶפְרַיִם, אֲשֶר דָחֲקוּ אֶת הַקֵץ. וַיִשָאֲרוּ
חַלְלֵי אַפְרַיִם בְּבִקְעַת גַת יָמִים וְשָנִים הַרְבֵּה, כִּי לֹא
נִתְּנוּ לִקְבוּרָה, וַתְּהִי הַבִּקְעָה מְלֵאָה עַצְמוֹת־אָדָם.



וַעֲשֶׂרֶת הָאֲנָשִים אֲשֶר נִמְלְטוּ עַל נַפְשָם מִמַעֲרֶכֶת
הַמִלְחָמָה שָבוּ גֹשְנָה וַיַגִידוּ לִבְנֵי יִשְׂרָאֵל אֶת כָּל
הַקוֹרוֹת אוֹתָם, וַיִתְאַבֵּל עֲלֵיהֶם אֶפְרַיִם אֲבִיהֶם יָמִים
רַבִּים. וַתֵּלֶד לוֹ אָז אִשְתּוֹ בֵן לִזְקֻנָיו וַתִּקְרָא אֶת שְמוֹ
בְרִיעָה, כִּי בִּימֵי רָעָה נוֹלַד לָהֶם, אַחֲרֵי הַכּוֹת בְּנֵי גַת
אֶת כָּל בְּנֵי אֶפְרַיִם.

וּמֹשֶה בֶן עַמְרָם עוֹדֶנוּ מוֹלֵךְ בְּאֶרֶץ כּוּש.
וַיְהִי בִּשְנַת הָאַרְבָּעִים לְמָלְכוֹ, וַתָּסֶת אֲדוֹנִית הַמַלְכָּה
אֶת בְּנֵי עַמָהּ, כִּי יַמְלִיכוּ אֶת בְּנָהּ, אֲשֶר יָלְדָה
לְקִיקָנוֹס, בַּאֲשֶר עַצְמָם וּבְשָׂרָם הוּא. וַיִשְמְעוּ לָהּ בְּנֵי
כוּש וַיִמְשְחוּ עֲלֵיהֶם אֶת מַנֵכְרִית בֶּן קִיקָנוֹס לְמֶלֶךְ
עֲלֵיהֶם. וַיִפָּרֵד מֹשֶה מִבְּנֵי כּוּש בֶּן שֶבַע וְשִשִים שָנָה.



וּמֵאֵת ה' הָיְתָה זֹאת, כִּי חִשֵב אֶת הַקֵץ לָעֳנִי בְּנֵי
יִשְׂרָאֵל וְעַבְדוּתָם בְּאֶרֶץ מִצְרָיִם. וַיֵלֶךְ מֹשֶה מִדְיָנָה,
כִּי יָרֵא לָשוּב מִצְרַיְמָה מִפְּנֵי פַרְעֹה אֲשֶר בִּקֵש לַהֲמִיתוֹ.
וַיֵלֶךְ וַיֵשֶב עַל בְּאֵר מָיִם. וַתָּבֹאנָה אָז בְּנוֹת רְעוּאֵל
הַמִדְיָנִי, כֹּהֵן מִדְיָן, לְהַשְקוֹת אֶת צֹאן אֲבִיהֶן. וַיְגָרְשוּן
רוֹעֵי מִדְיָן מֵעַל הַבְּאֵר; וַיָקָם מֹשֶה וַיוֹשִיעָן וַיַשְקְ אֶת
צֹאנָן. וַתָשֹבְנָה בְּנוֹת רְעוּאֵל אֶל אֲבִיהֶן וַתַּגֵדְנָה לוֹ אֶת
הַקוֹרוֹת אוֹתָן. וַיִשְלַח רְעוּאֵל וַיִקְרָא לְמֹשֶה וַיְבִיאֵהוּ
לְבֵיתוֹ. וַיְסַפֵּר מֹשֶה לִרְעוּאֵל עַל בָּרְחוֹ מִמִצְרַיִם וּמָלְכוֹ
אַרְבָּעִים שָנָה בְּאֶרֶץ כּוּש. וַיְהִי כִּשְמוֹעַ רְעוּאֵל אֶת
דִבְרֵי מֹשֶה וַיאֹמֶר בְּלִבּוֹ: אֶתֵּן אֶת זֶה אֶל בֵּית הַכֶּלֶא
וַאֲרַצֶה בוֹ אֶת פְּנֵי בְנֵי־כוּש, כִּי אֶל־נָכוֹן נוֹס נָס מֵאַרְצָם.
וַיִקַח רְעוּאֵל אֶת מֹשֶה וַיִתְּנֵהוּ בְּבֵית הַסֹהַר, וַיְהִי שָם
מֹשֶה עֶשֶׂר שָנִים. אַךְ צִפּוֹרָה בַּת רְעוּאֵל חָמְלָה עָלָיו
וְכִלְכְּלָה אוֹתוֹ חֶרֶש בְּלֶחֶם וָמַיִם כָּל יְמֵי שִבְתוֹ
בַּבּוֹר.
