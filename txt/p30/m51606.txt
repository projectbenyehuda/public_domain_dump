


	
	אברהם ונשי ישמעאל
	




הָגָר שִפְחַת שָׂרַי יָלְדָה לְאַבְרָהָם אֶת יִשְמָעֵאל,
וְכַעֲבוֹר י"ד שָנָה יָלְדָה לוֹ גַם שָׂרָה אִשְתּוֹ בֵּן וַיִקְרָא אֶת
שְמוֹ יִצְחָק. וְאַבְרָהָם בֶּן מְאַת שָנָה וְשָׂרָה אִשְתּוֹ בַּת
תִּשְעִים שָנָה בְּהִוָלֵד לָהֶם בְּנָם. וַיִגְדַל הַיֶלֶד וַיִגָמַל,
וַיַעַש אַבְרָהָם מִשְתֶּה גָדוֹל בְּיוֹם הִגָמֵל אֵת יִצְחָק. וַיָבֹאוּ
שֵם וָעֵבֶר וְכָל גְדוֹלֵי הָאָרֶץ, וְגַם אֲבִימֶלֶךְ מֶלֶךְ
פְּלִשְתִּים וּפִיכוֹל שַׂר צְבָאוֹ, אֶל הַמִשְתֶּה אֲשֶר עָשָׂה
אַבְרָהָם, וְגַם בְּנֵי מִשְפַּחַת נָחוֹר בָּאוּ מֵחָרָן וַיִשְׂמְחוּ
מְאֹד בְּשִׂמְחַת אַבְרָהָם וְשָׂרַי בְּיוֹם הִגָמֵל בֶּן־זְקֻנֵיהֶם.
וְכַאֲשֶר גָדַל יִשְמָעֵאל וַיְלַמֵד יָדוֹ לִירוֹת בַּקֶשֶת וַיְהִי
לְרוֹבֶה־קַשָת. וַיְהִי הַיוֹם, בִּהְיוֹת יִצְחָק כְּבֶן חָמֵש שָנִים,
וַיֵשֶב עִם יִשְמָעֵאל אָחִיו בְּפֶתַח הָאֹהֶל. וַיִקַח יִשְמָעֵאל אֶת
קַשְתּוֹ וַיִתֵּן בָּהּ חֵץ וַיֹאמֶר לִירוֹת בְּיִצְחָק. וַתֵּרֶא שָׂרָה
אֶת הַמַעֲשֶׂה הַזֶה וַתִּקְרָא לְאַבְרָהָם וַתֹּאמֶר לוֹ: גָרֵש אֶת
הָאָמָה וְאֶת בְּנָהּ, כִּי כָּזֹאת וְכָזֹאת אָמַר לַעֲשׂוֹת לְיִצְחָק
בְּנִי. וַיִשְמַע אַבְרָהָם בְּקוֹל שָׂרַי אִשְתּוֹ וַיִשְלַח אֶת הָגָר
וְאֶת יִשְמָעֵאל מִבֵּיתוֹ, וַיֵּלְכוּ וַיִתְיַשְבוּ בְּמִדְבַּר
פָּארָן. כַּעֲבוֹר יָמִים לָקְחָה לוֹ הָגָר אִמוֹ אִשָה מֵאֶרֶץ
מִצְרַיִם וּשְמָה מְרִיבָה. וַתֵּלֶד לוֹ ד' בָּנִים וּבַת אֶחָת. וַיֵשֶב
יִשְמָעֵאל בַּמִדְבָּר הוּא וְאִמוֹ וְאִשְתּוֹ וּבְנֵי בֵיתוֹ. וַיִפְרֹץ
הָאִיש מְאֹד בְּמִקְנֵה־צֹאן וּבְמִקְנֵה־בָקָר, וַיְהִי נוֹסֵעַ
בַּעֲדָרָיו וְאֹהָלָיו מִמָקוֹם לְמָקוֹם.

וַיְהִי מִיָמִים וַיֹאמֶר אַבְרָהָם אֶל שָׂרַי אִשְתּוֹ:
אֵלְכָה נָא וְאֶרְאֶה אֶת שְלוֹם יִשְמָעֵאל בְּנִי, כִּי לֹא רְאִיתִיו
זֶה יָמִים רַבִּים. וַיִרְכַּב אַבְרָהָם עַל אֶחָד מִגְמַלָיו וַיָשֶׂם
פָּנָיו אֶל מִדְבַּר עֲרָב לְבַקֵש שָם אֶת יִשְמָעֵאל בְּנוֹ, וַיַגַע
עַד אַחַד הָאֹהָלִים בַּחֲצוֹת הַיוֹם, וַיִמְצָא אֶת אֵשֶת יִשְמָעֵאל
יוֹשֶבֶת בְּפֶתַח הָאֹהֶל הִיא וִילָדֶיהָ וְיִשְמָעֵאל אִישָהּ וְהָגָר
אִמוֹ אֵין אִתָּהּ. וַיִשְאַל אַבְרָהָם אֶת הָאִשָה: אָנָה פָנָה
יִשְמָעֵאל? – וַתֹּאמַר: הָלַךְ הַשָׂדֶה לָצוּד צָיִד. וְאַבְרָהָם
עוֹדֶנוּ יוֹשֵב עַל גְמַלוֹ וְלֹא יָרַד מֵעָלָיו, וַיאֹמֶר אֶל אֵשֶת
יִשְמָעֵאל: תְּנִי לִי, בִּתִּי, מְעַט מַיִם וְשָתִיתִי כִּי צָמֵאתִי
וְעָיַפְתִּי מְאֹד מֵעֲמַל הַדָרֶךְ. וַתַּעַן הָאִשָה וַתֹּאמַר: אֵין
לָנוּ לֹא מַיִם וְלֹא לָחֶם. וְלֹא הִבִּיטָה אֶל אַבְרָהָם וְגַם לֹא
שָאֲלָה אוֹתוֹ מִי הוּא. וַיֵרַע הַדָבָר בְּעֵינֵי אַבְרָהָם וַיֹאמֶר
אֶל אֵשֶת יִשְמָעֵאל: כַּאֲשֶר יָשוּב אִישֵךְ לְבֵיתוֹ וְאָמַרְתְּ לוֹ,
כִּי בָא הֵנָה אִיש זָקֵן מֵאֶרֶץ פְּלִשְתִּים, כָּזֶה וְכָזֶה
מַרְאֵהוּ. וְכֹה אָמַר הָאִיש הַזָקֵן הַהוּא: הָסֵר אֶת יְתַד הָאֹהֶל
אֲשֶר בְּפֶתַח בֵּיתְךָ וְנָתַתָּ יָתֵד אַחֶרֶת תַּחְתֶיהָ, וַיְכַל
אַבְרָהָם לְדַבֵּר אֶל הָאִשָה וַיֵלֶךְ וַיָשָב לְדַרְכּוֹ.

וַיְהִי אַחֲרֵי כֵן, כְּשוּב יִשְמָעֵאל, הוּא וְאִמוֹ,
מִצֵידוֹ אֶל אָהֳלוֹ וַתְּסַפֵּר לוֹ אִשְתּוֹ כַּדְבָרִים הָאֵלֶה
לֵאמֹר: בָּא הֵנָה אִיש זָקֵן מֵאֶרֶץ פְּלִשְתִּים, כָּזֶה וְכָזֶה
מַרְאֵהוּ, וְכֹה אָמָר: הָמֵר אֶת יְתַד הָאֹהֶל אֲשֶר בְּפֶתַח בֵּיתְךָ
וְנָתַתָּ יָתֵד אַחֶרֶת תַּחְתֶּיהָ… כִּשְמֹעַ יִשְמָעֵאל אֶת דִבְרֵי
אִשְתּוֹ וַיֵדַע כִּי אָבִיו הָיָה פֹּה וְכִי אִשְתּוֹ לֹא כִבְדַתּוּ
לַהֲבִיאֵהוּ בְּצֵל קוֹרָתוֹ. וַיְגָרֶש יִשְמָעֵאל אֶת אִשְתּוֹ זוּ
וַיֵלֶךְ אַרְצָה כְּנָעַן וַיִקַח לוֹ מִשָם אִשָה אַחֶרֶת. וַיְהִי מִקֵץ
ג' שָנִים וַיֹאמֶר אַבְרָהָם שֵנִית אֶל שָׂרָה אִשְׁתּוֹ: אֵלְכָה לִי
וְאֶרְאֶה אֶת שְלוֹם יִשְמָעֵאל בְּנִי. וַיִרְכַּב אַבְרָהָם עַל גְמַלוֹ
וַיֵלֶךְ מִדְבָּרָה עֲרָב לְבַקֵש אֶת יִשְמָעֵאל, וַיַגַע אֶל אָהֳלוֹ
בַּחֲצִי הַיוֹם. וַתֵּצֵא אֵשֶת יִשְמָעֵאל הַחֲדָשָה מִן הָאֹהֶל
וַתֹּאמֶר לְאַבְרָהָם: יִשְמָעֵאל אִישִי הָלַךְ הַשָׂדֶה לָצוּד צַיִד,
אַךְ הוֹאִילָה וְסוּרָה אֶל אָהֳלֵנוּ וְאָכַלְתְּ וְשָתִית אִתָּנוּ
וְנַחְתָּ מֵעֲמַל הַדָרֶךְ. וַתָּשֶׂם לְפָנָיו מַיִם וָלֶחֶם, וַיֵשְתְּ
אַבְרָהָם וַיאֹכַל וַיִסְעַד לִבּוֹ. וַיְכַל לֶאֱכֹל וַיְבָרֶךְ אֶת ה‘,
וַיאֹמֶר אֶל אֵשֶת יִשְמָעֵאל: בְּשוּב יִשְמָעֵאל אִישֵךְ תֹּאמְרִי לוֹ:
בָּא הֵנָה אִיש זָקֵן מֵאֶרֶץ פְּלִשְתִּים, כָּזֶה וְכָזֶה מַרְאֵהוּ.
וְכֹה אָמַר הָאִיש הַזָקֵן הַהוּא: הַיָתֵד הַחֲדָשָה אֲשֶר שַׂמְתָּ
בְּפֶתַח אָהֳלֵךְ טוֹבָה הִיא, וְאַל תְסִירֶנָה מִשָם. וַיְכַל אַבְרָהָם
לְדַבֵּר וַיִרְכַּב עַל גְמַלוֹ וַיָשָב לְאֶרֶץ פְּלִשְתִּים. וַיְהִי
כְּבֹא יִשְמָעֵאל מִן הַשָׂדֶה וַתְּסַפֵּר לוֹ אִשְתּוֹ כַדְבָרִים
הָאֵלֶה לֵאמֹר: בָּא הֵנָה אִיש זָקֵן מֵאֶרֶץ פְּלִשְתִּים, כָּזֶה
וְכָזֶה מַרְאֵהוּ, וְכֹה אָמַר הָאִיש הַזָקֵן הַהוּא: הַיָתֵד הַחֲדָשָה
אֲשֶר שַׂמְתָּ בְּפֶתַח אָהֳלֵךְ טוֹבָה הוּא, וְאַל תְּסִירֶנָה מִשָם.
וַיֵדַע יִשְמָעֵאל כִּי הָיָה אָבִיו בְּאָהֳלוֹ, וַיְבָרֶךְ אֶת ה’
וָיָקָם וַיִקַח אֶת אִשְתּוֹ וְאֶת בָּנָיו וַיִסַע מִשָם וַיֵלֶךְ אֶל
אֶרֶץ פְּלִשְתִּים וַיֵשֶב שָם עִם אַבְרָהָם יָמִים רַבִּים.
