


	
	הולדת מרים ואהרן; חלום פרעה
	




וַיֵלֶךְ אִיש מִבֵּית לֵוִי, וּשְמוֹ עַמְרָם בֶּן
קְהָת בֶּן לֵוִי בֶּן יִשְׂרָאֵל וַיִקַח אֶת יוֹכֶבֶד אֲחוֹת אָבִיו לוֹ
לְאִשָה. וַתֵּלֶד לוֹ בַּת וַתִּקְרָא אֶת שְמָהּ מִרְיָם, לֵאמֹר:
בְּיָמֶיהָ מֵרֲרוּ הַמִצְרִים אֶת חַיֵי בְנֵי יִשְׂרָאֵל. וַיְהִי
כִּי אָרְכוּ הַיָמִים וַתֵלֶד יוֹכֶבֶד עוֹד בֵּן וַתִּקְרָא אֶת בְּנוֹ
אַהֲרֹן, לֵאמֹר: יְהִי לְאוֹר עַמוֹ. בַּיָמִים הָהֵם מֵת צְפוֹ מֶלֶךְ
כִּתִּים וַיִמְלֹךְ תַּחְתָּיו יַאנוּשׂ, אֶחָד מִגִבּוֹרֵי בְנֵי
כִתִּים. וַיְהִי אַחֲרֵי מוֹת צְפוֹ וַיִבְרַח בִּלְעָם בֶּן בְּעוֹר
מֵאֶרֶץ כִּתִּים וַיֵלֶךְ וַיָבֹא מִצְרָיְמָה. וַיְקַבְּלֵהוּ פַּרְעֹה
בְּכָבוֹד גָדוֹל, כִּי שָמַע אֶת חָכְמָתוֹ בַקְסָמִים, וַיִתֵּן לוֹ
מַתָּנוֹת רַבּוֹת וַיַפְקִידֵהוּ לְאֶחָד מִיוֹעֲצָיו. וַיְהִי בִּשְנַת
מֵאָה וּשְלֹשִים לְרֶדֶת בְּנֵי יִשׂרָאֵל מִצְרַיְמָה וּפַרְעֹה חוֹלֵם:
וְהִנֵה הוּא יוֹשֵב עַל כִּסֵא מַלְכוּתוֹ, וַיִשָׂא עֵינָיו וַיַרְא
וְהִנֵה אִיש זָקֵן עוֹמֵד לְנֶגְדוֹ וּבְיָדוֹ מֹאזְנָיִם. וַיִקַח
הַזָקֵן אֶת כָּל זִקְנֵי מִצְרַיִם וְשָׂרֶיהָ וּגְדוֹלֶיהָ וַיְשִׂימֵם
עַל כַּף הַמאֹזְנַיִם הָאַחַת, וַיִקַח טְלֵה־חָלָב אֶחָד וַיִתְּנֵהוּ
עַל כַּף הַמֹאזְנַיִם הַשְנִיָה – וַיַכְרַע הַטָלֶה אֶת כֻּלָם. וַיִיקַץ
פַּרְעֹה וְהִנֵה חֲלוֹם. וַיְהִי בַּבּקֶר וַיִקְרָא לְכָל עֲבָדָיו
וְשָׂרָיו וַיְסַפֵּר לָהֶם אֶת חֲלוֹמוֹ, וַיִירְאוּ הָאֲנָשִים יִרְאָה
גְדוֹלָה. וַיאֹמֶר פַּרְעֹה אֶל חֲכָמָיו: פִּתְרוּ נָא לִי אֶת חֲלוֹמִי
אֲשֶר חָלַמְתִּי וְאֵדָעֶנוּ. וַיַעַן בִּלְעָם בֶּן בְּעוֹר: אֵין זֶה,
כִּי רָעָה גְדוֹלָה אֲשֶר תִּצְמַח עַל אֶרֶץ מִצְרַיִם בְּאַחֲרִית
הַיָמִים. כִּי הִנֵה בֵן יִוָלֵד לִבְנֵי יִשְׂרָאֵל אֲשֶר יַחֲרִיב אֶת
אַרְצְךָ וִינַגַע אֶת יוֹשְבֶיהָ בִּנְגָעִים רַבִּים וְיוֹצִיא אֶת עַמוֹ
בְּיָד חֲזָקָה מִמִצְרָים. וְעַתָּה, אֲדוֹנִי הַמֶלֶךְ, הָבָה לְךָ עֵצָה
וְתַחְבֻּלָה בְּטֶרֶם תִּצְמַח הָרָעָה הַגְדוֹלָה הַזֹאת. וַיאֹמֶר
פַּרְעֹה: וּמַה לָנוּ לַעֲשׂוֹת לִבְנֵי יִשְׂרָאֵל? הֲלֹא כָזֹאת
וְכָזֹאת יָעַצְנוּ עֲלֵיהֶם וְלֹא יָכֹלְנוּ לָהֶם. וַיַעַן בִּלְעָם:
שְלַח נָא וּקְרָא גַם לִשְנֵי יוֹעֲצֶיךָ: לִרְעוּאֵל הַמִדְיָנִי
וּלְאִיוֹב הָעוּצִי וְנִשְמְעָה מַה בְּפִיהֶם, וְאַחֲרֵיהֶם יְדַבֵּר
עַבְדְךָ אֶת דְבָרָיו. וַיְצַו הַמֶלֶךְ וַיִקְרְאוּ לִרְעוּאֵל
וּלְאִיוֹב, וַיָבוֹאוּ וַיַעַמְדוּ לְפָנָיו. וַיאֹמֶר אֲלֵיהֶם הַמֶלֶךְ:
הֲלֹא כֻלְכֶם שְמַעְתֶּם אֶת דְבַר חֲלוֹמִי אֲשֶר חָלַמְתִּי, וְעַתָּה
עוּצוּ נָא: בַּמֶה נוּכַל לִבְנֵי יִשְׂרָאֵל. וַיַעַן רְעוּאֵל
הַמִדְיָנִי וַיאֹמַר: אִם עַל הַמֶלֶךְ טוֹב – יֶחְדַל נָא מֵהָעִבְרִים
וְיָנַח לָהֶם, כִּי אֱלֹהִים אַדִירִים לָהֶם וְהוּא בָחַר בְּאַבְרָהָם
אֲבִיהֶם וּבְבָנָיו אַחֲרָיו לְעַם סְגֻלָתוֹ, וְכָל הַנֹגֵע בָּהֶם–לֹא
יִנָקֶה. וַיְהִי כִּשְמֹעַ פַּרְעֹה אֶת דִבְרֵי רְעוּאֵל וַיִחַר אַפּוֹ
בוֹ.



וַיָקָם בַּיוֹם הַהוּא רְעוּאֵל הַמִדְיָנִי וַיֵצֵא מֵאֶרֶץ
מִצְרַיִם וַיָשָב אֶל אַרְצוֹ מִדְיָנָה. – וַיאֹמֶר פַּרְעֹה לְאִיוֹב:
וּמָה הִיא עֲצָתְךָ אָתָּה? וַיַעַן אִיוֹב: הֲלֹא כָל יוֹשְבֵי אַרְצְךָ
בְּיָדְךָ הֵם; כַּטוֹב בְּעֵינֵי הַמֶלֶךְ יַעֲשֶׂה בִּבְנֵי יִשְׂרָאֵל.
וַיאֹמֶר בִּלְעָם לְפַרְעֹה: כָּל אֲשֶר יָעַץ הַמֶלֶךְ יַעֲשֶׂה בִבְנֵי
יִשְׂרָאֵל. וַיאֹמֶר בִּלְעָם לְפַרְעֹה: כָּל אֲשֶר יָעַץ הַמֶלֶךְ
לַעֲשׂוֹת לָעִבְרִים הַחַיִים אִתָּנוּ לֹא יִצְלָח: כִּי אִם בְּלַהֶבֶת אֵש תּאֹמַר לְכַלוֹתָם – הֲלֹא אֱלֹהֵיהֶם
הִצִיל אֶת אַבְרָהָם אֲבִיהֶם מִכִּבְשַן הָאֵש בְּאוּר כַּשְׂדִים, וְאִם
בַּחֶרֶב תּאֹמַר לְהַשְמִידָם וּלְהַכְחִידָם – הֲלֹא יַעֲקֹב אֲבִיהֶם
עָבַד עֲבוֹדָה קָשָה עֶשְׂרִים שָנָה בְּבֵית דוֹדוֹ וְהִצְלִיחַ.
וְעַתָּה, אֲדוֹנִי הַמֶלֶךְ, שְמַע נָא לַעֲצָתִי אֲשֶר אִיעָצְךָ, וְאִם
כֹּה תַעֲשֶׂה יָכוֹל תּוּכַל לָהֶם: אִם עַל הַמֶלֶךְ טוֹב יְצָוֶה נָא
לְהַשְלִיךְ אֶת כָּל הַבֵּן הַיִלוֹד לָעִבְרִים הַיְאוֹרָה, וּבָזֶה
תּוּכַל לִמְחוֹת אֶת שְמָם, כִּי בַמַיִם לֹא נֻסָה עוֹד אֶחָד מֵהֶם,
לֹא הֵם וְלֹא אֲבוֹתֵיהֶם. – וַתִּיטַב עֲצַת בִּלְעָם בְּעֵינֵי פַרְעֹה
וַיִתֵּן צַו לְכָל עַמוֹ לֵאמֹר: כָּל הַבֵּן הַיִלוֹד לָעִבְרִים –
הַיְאוֹרָה תַּשְלִיכוּהוּ וְכָל הַבַּת תְּחַיוּן.
