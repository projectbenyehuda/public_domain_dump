


	
	משל לא: כֶּלֶב נַעַר וְזָקֵן
	




הַמִּתְגָּאֶה בִּכְבוֹד הַלְשָׁנוּתוֹ / נֶעֱלַם מִמֶּנּוּ רָעָתוֹ



כֶּלֶב קָטָן לְאִישׁ הָיָה / הַנֹּשֵׁךְ פֶּתַע כָּל
בְּרִיָּה / וּבַעַל הַכֶּלֶב תָּלָה לוֹ פַּעֲמוֹן / בְּצַוָּארוֹ
לַעֲשׂוֹת קוֹל הָמוֹן / וְכַאֲשֶׁר רָץ יִשָּׁמַע קוֹלוֹ / וְלֹא יִשֹּׁךְ
עוֹד פִּתְאֹם בְּאָהֳלוֹ / וַיָּרָם לְבָבוֹ מִיּוֹם הַפַּעֲמוֹן נָשָׂא /
כְּלָבִיא יָקוּם וְכַאֲרִי יִתְנַשָּׂא / חָשַׁב כִּי עַל כְּלָבִים
מֹלֵךְ / וּבְגַאֲוָה נְטוּי גָרוֹן הֹלֵךְ / כֶּלֶב זָקֵן הִשְׁקִיף
אֵלָיו / וַיָרֶם אֶת קוֹלוֹ עָלָיו / לֵאמֹר: לָמָּה תִתְגָּאֶה / מִן
הַפַּעֲמוֹן אֲשֶׁר תַּרְאֶה / כִּי לֹא לְטוֹבָתְךָ וּלְשַׁעֲשׁוּעֶךָ /
רַק לְהַזְהִיר עַם מֵרִשְׁעֶךָ / אֲשֶׁר מִשִּׁנֶּיךָ נְשׁוּכִים
וְנִשְׂרָטִים / עַל כֵּן יִהְיוּ דְבָרֶיךָ מְעַטִּים.

וְהַמָּשָׁל בְּדֹרֵךְ קַשְׁתּוֹ עַל לְשׁוֹן שֶׁקֶר /
וְעַל מוּמֵי שְׁכֵנָיו יְבַקֵּר / וְאָרְבּוֹ בְּקִרְבּוֹ יְגַבֵּר /
יֵצֵא לַחוּץ יְדַבֵּר / וּמֹשֵׁל הָעִיר יַמְתִּיק סוֹד אִתּוֹ /
וְיַשְׁלִיטֶנּוּ בְמֶמְשַׁלְתּוֹ / בַּאֲשֶׁר יְדָעוֹ מַלְשִׁין
וּמְחַנֵּף / וְזֶה יִתְגָּאֶה וְיִתְאַנֵּף / עַל הַדַּלִּים
בִּשְׁכוּנָתוֹ / יַשְׂכִּיל כִּי זֶה לֹא לְטוֹבָתוֹ.
