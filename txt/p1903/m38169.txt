


	
	משל סו: פֶּרֶד וְשֻׁעָל
	






הַנִּשְׁאָל לִשְׁמוֹ וּמוֹלַדְתּוֹ / לְהַגִּיד אָבִיו וְאִמּוֹ לֹא תְהֵא תָמִיד תִּפְאַרְתּוֹ



פֶּרֶד עֹבֵר בַּדֶּרֶךְ וַיִּפְגָּעֵהוּ / שֻׁעָל אֲשֶׁר
מֵעוֹלָם לֹא רָאָהוּ / וַיַּבֵּט הַשֻּׁעָל אֶל הַדְרַת פָּנָיו / וְכִי
אוֹרוּ עֵינָיו וְאָרְכוּ אָזְנָיו / וַיֹּאמֶר בְּלִבּוֹ: מִי הוּא אֲשֶׁר
עַתָּה אֶחֱזֶה / מַה טִיבוֹ שֶׁל עֻבָּר זֶה / כִּי לֹא רָאִיתִי כָמֹהוּ
כָּל תְּמוּנָה / וְלֹא רְאִיתִיו עַד הֵנָּה / בְּאָזְנַיִם אֲרֻכּוֹת
נְטוּיוֹת רָמָה / מָלֵא חָכְמָה דַּעַת וּמְזִמָּה / וַיֵּלֶךְ הַשֻּׁעָל
לִשְׁמֹעַ אִמְרוֹתָיו / לְהוֹסִיף עָרְמוֹת עַל עָרְמוֹתָיו / אוּלַי
יַעֲמֹד בְּסוֹדוֹ / וַיִּשְׁאַל לוֹ מִי יְלָדוֹ / וַיַּעַן הַפֶּרֶד:
דּוֹדִי הָיָה בְגַאֲוָה מִתְהַלֵּךְ / סוּס אֲשֶׁר רָכַב עָלָיו הַמֶּלֶךְ
/ מְדַלֵּג וּמְקַפֵּץ בְּיוֹם קְרָב וָקֶרֶץ / בְּרַעַשׁ וְרֹגֶז יְגַמֵּא
אָרֶץ / וְצַוָּארוֹ לָבוּשׁ רַעְמָה / וְהוֹד נַחֲרוֹ אֵימָה /
וּפַרְסוֹתָיו כַּצַּר נֶחְשָׁבוּ / לְהָרֵב קְרָבוֹת צָמְאוּ לְהַוּוֹת
רָעָבוּ / וְאֹרְחוֹתָיו לֹא יְעַבְּטוּ / כִּידוֹדֵי אֵשׁ יִתְמַלְּטוּ /
וְכָל אֶחָד כְּפַטִּישׁ סֶלַע יְפֹצֵץ / וְעֵינָיו כְּלַפִּידִים
וּכְבָזָק יְנֹצֵץ / מִגְדַּל עֹז לְרֹכְבוֹ וְהֹלֵךְ נְטוּי גָּרוֹן /
מֵרָחוֹק יָרִיחַ מִלְחָמָה וְיָרוֹן לְשִׁבָּרוֹן / רֹדֵף וּמַשִּׂיג
אֲשֶׁר בְּרֹכְבוֹ מֹרֵד. / וְאֵלֶּה תּוֹלְדוֹת פֶּרֶד.



וְהַמָּשָׁל בְּאִישׁ הָדוּר בִּלְבוּשׁוֹ / וּמִכַּף
רַגְלוֹ וְעַד רֹאשׁוֹ / אֵין בּוֹ מוּם וַיִּיף בְּגָדְלוֹ / אַךְ
כַַּאֲשֶׁר יִשְׁאֲלוּ לוֹ / לִשְׁמוֹ וּלְמוֹלַדְתּוֹ / אֶפֶס כִּי לֹא
תִהְיֶה תִפְאַרְתּוֹ / לְהַגִּיד אָבִיו וְאִמּוֹ / אַךְ יַזְכִּיר
קְרוֹבוֹ הַמְרִימוֹ / וַאֲשֶׁר יְלָדוֹ לֹא יַזְכִּירֶנּוּ / אוֹ דוֹדוֹ
אוֹ בֶן־דּוֹדוֹ יִגְאָלֶנּוּ / וּכְבוֹדוֹ וְהוֹדוֹ יַעֲלֶה עַל שְׂפָתוֹ
/ מַעֲשֵׂה תָקְפּוֹ וּגְבוּרָתוֹ / וְאִישׁ בְּכָל אֵלֶּה לֹא יָגַעְתִּי
וּמְצָאתִיו / הָאֹמֵר לְאָבִיו לֹא רְאִיתִיו / וּמִתְיַחֵשׂ לַגָּדוֹל
מִמִּשְׁפַּחְתּוֹ / בַּאֲשֶׁר נָפַל הוֹד הַמִּשְׁפָּחָה מֵאָבִיו
וּמֵאִתּוֹ / וַאֲנִי בֶּרֶכְיָה אָמַרְתִּי בְחָפְזִי / אוֹי לַאֲשֶׁר
יִקָּרֵא אִישׁ מִמִּשְׁפַּחַת בּוּזִי / כָּל אֲשֶׁר מִמַּעֲלַת אֲבוֹתָיו
נֹפֵל / טוֹב מִמֶּנּוּ הַנָּפֶל.
