


	
	משל עא: זַרְזִיר וּבַת־מֶלֶךְ
	




אִישׁ מָלֵא חָכְמָה וָדָעַת / לוֹ עַין רֹאָה וְאֹזֶן שׁוֹמַעַת



זַרְזִיר בַּת־מֶלֶךְ גִּדְלַתּוּ / וּלְדַבֵּר צַחוֹת
לִמְדַתּוּ / וַתְּבִיאֵהוּ בַּחֲדָרֶיהָ / בְּבֵית אָבִיהָ בִּנְעוּרֶיהָ
/ וַיֶּחְכַּם מִכָּל הָעוֹפוֹת / בְּדִבְרֵי צַחוֹת וּבִמְלִיצוֹת יָפוֹת
/ וַיִּתְבּוֹנֵן בְּשָׁמְעוֹ חֲדָשׁוֹת / עֵת לְדַבֵּר וְעֵת לַחֲשׁוֹת /
אַךְ לֹא יָכֹל לָצֵאת וְלָבֹא / וַיִּתְעַצֵּב אֶל לְבָבוֹ / כִּי תֵבַת
קְנֵי בַרְזֶל עָשְׂתָה לוֹ / וּבֵין הַקָּנִים נִשְׁמַע קוֹלוֹ / יוֹם
אֶחָד פָּרָשׁ מִתְהַלֵּךְ / בַּחֲצַר הַמַּלְכָּה וְהַמֶּלֶךְ / וְלַעֲבֹר
הַיָּם יָשִׂים פָּנָיו / וַיְצַו עַל בֵּיתוֹ וְעַל בָּנָיו / וְעַל
אַרְצוֹ אֶת הַמַּלְכָּה / וְחָלַק מִשֶּׁמֶן חִכָּה / לֵאמֹר: שָׁמֹעַ
אֶשְׁמַע בְּקוֹלְךָ / עֵינַי אָשִׁית עַל אֲשֶׁר לְךָ / זַרְזִיר הֵבִין
בַּפָּרָשׁ / כִּי לַעֲבֹר אֶת הַיָּם דָּרָשׁ / וּבְלֶכְתּוֹ דֶרֶךְ
אֶצְלוֹ / בִּשְׁמוֹ קָרָא וַיֹּאמֶר לוֹ / הִנְּךָ אֶת הַיָּם עֹבֵר /
נִכְבָּדוֹת בְּךָ אֱהִי דֹבֵר / אִם תִּרְאֶה עוֹף כִּדְמוּתִי /
וּכְצַלְמִי, אָז תִּזְכֹּר אֹתִי / וּבִשְׁמִי תִפְקְדֵהוּ לִשְׁלוֹמִים /
וּשְׁאַל לוֹ בְחֶסֶד וּבְרַחֲמִים / לָצֵאת מִמַּסְגֵּר כִּלְאִי עֵצָה /
בַּאֲשֶׁר נַפְשִׁי בוֹ קָצָה / כִּי סְבִיבוֹתַי גָּדֵר וְחוֹמָה / גַּם
פֹּה לֹא עָשִׂיתִי מְאוּמָה / כִּי מִמִּשְׁפַּחְתִּי גֻנַּבְתִּי /
וּבֵין אֲנָשִׁים סֻגַּרְתִּי / עֲשֵׂה עִמִּי חֶסֶד וֶאֱמֶת / עָצוּר
בְּאֵסוּר חָשׁוּב כְּמֵת / הִנְּנִי עָצוּר לֹא אוּכַל / לָצֵאת מִזֶּה
וּמַאֲכָל / לֹא יֶעֱרַב לִי, עַד לַחָפְשִׁי / אֵצֵא, אָז אֶשָּׂא רֹאשִׁי
/ עַל כֵּן אֲנִי מַשְׁבִּיעֲךָ / לְזָכְרֵנִי בְּמַדָּעֲךָ / וּבְשׁוּבְךָ
הַגֵּד לִי אִמְרוֹתָיו / וּמַה יִּהְיוּ תְשׁוּבוֹתָיו / הַפָּרָשׁ
מִשָּׁם נִפְרָד / וַיֵּלֶךְ וְאֶל הַיָּם יָרָד / וְעוֹף יְעוֹפֵף
לְעֻמָּתוֹ / כְּצֶלֶם עוֹף הַמְדַבֵּר אִתּוֹ / וּבְטֶרֶם יִקְרַב אֵלָיו
/ וַיָּרֶם אֶת קוֹלוֹ עָלָיו / לֵאמֹר: לְשָׁלוֹם פֹּקֵד אֹתְךָ / עוֹף
כְּצַלְמְךָ כִּדְמוּתְךָ / אֲשֶׁר הוּא בֵּין אֲנָשִׁים גָּר /
וּבְמַסְגֵּר כִּלְאָם סֻגָּר / הִשְׁבִּיעַנִי מִמְּךָ לִשְׁאוֹל / עֵצָה
אֵיךְ נַפְשׁוֹ לִגְאוֹל / לְהוֹצִיא מִמַּסְגֵּר אַסִּיר / וְיֵשַׁע
לְנַפְשׁוֹ לֹא תַחְסִיר / וְאִם יוּכַל עוֹד בַּיַּעַר לְהֵרָאֹה /
וְשִׁנָּה אֶת בִּגְדֵי כִלְאוֹ / וַיְהִי כְּכַלּוֹתוֹ לְדַבֵּר / הָעוֹף
אֶת כְּנָפָיו חִבֵּר / וַיִּפֹּל עַל הַפָּרָשׁ בְּעֻלְפֶּה / כַּאֲשֶׁר
הָיָה מִמָּוֶת נִסְפֶּה / וְהַפָּרָשׁ אֹתוֹ לְקָחוֹ / וּלְהָשִׁיב אֵלָיו
רוּחוֹ / שָׂם אֶת עֵינָיו לְמוּל עֵינָיו / וַיֵּז מַיִם חַיִּים עַל
פָּנָיו / וְאַחֲרֵי כֵן שָׂם אֹתוֹ בְחֵיקוֹ / אוּלַי יֵחַם לוֹ,
וּזְרָקוֹ / בִּרְאוֹתוֹ כִּי אֵין תִּקְוָה / עַל זֶה הָיְתָה נַפְשׁוֹ
דָוָה / כִּי לֹא יָדַע מֶה הָיָה לוֹ / וַיִּפָּרֵד מַהֵר מֵאֶצְלוֹ /
לָלֶכֶת אֶל מְחוֹז חֶפְצוֹ וַיָּשָׁב / עַד מִשְׁפַּחְתּוֹ בְּעִיר
מוֹשָׁב / וּבְשׁוּבוֹ לַמֶּלֶךְ שִׂמְּחָהוּ / וַיְחַבְּקֵהוּ
וַיְנַשְּׁקֵהוּ / וַיְהִי הוֹלֵךְ עַד כֹּה וְעַד כֹּה / הַזַּרְזִיר
הֵבִין כִּי מִדַּרְכּוֹ / שָׁב הַפָּרָשׁ וַיִּשְׁאָלוֹ / אִם זָכַר אֶת
יְמֵי אֶבְלוֹ / וְאִם מָצָא עוֹף כְּצַלְמוֹ / מַה פָּעַל וּמַה דִּבֵּר
עִמּוֹ / וַיְסַפֵּר לוֹ אֶת כָּל הַקֹּרוֹת / וְכִי הָיָה יוֹם עֲבָרוֹת /
לָעוֹף כִּשְׁמֹעַ אִמְרוֹתָיו / וְאֵיךְ נִתְּקוּ מְזִמּוֹתָיו / כִּי
מִבְּלִי אֹרֵב וְצֹדֶה / נָפַל מֵת עַל פְּנֵי הַשָּׂדֶה / וְכִי
לְהָשִׁיב רוּחוֹ בַּמַּיִם הִטְבִּילוֹ / וַיְכַסֵּהוּ בַּבְּגָדִים וְלֹא
יֵחַם לוֹ וּבְטֶרֶם תַּעֲלֶה צַחֲנָתוֹ וּבָאְשׁוֹ / הִשְׁלִיכוֹ בְּגֹעַל
נַפְשׁוֹ / וַיָּשֶׂם הַזַּרְזִיר עַל לִבּוֹ הַדְּבָרִים / וּכְבֹא בַּת
הַמֶּלֶךְ בַּחֲדָרִים / וְלֹא שָׁמְעָה הָעוֹף מְצַפְצֵף וַחֲמָתָהּ
בָּעֲרָה בָהּ בְּקֶצֶף / וַתִּפְתַּח אֶת הַתֵּבָה וַתִּרְאֵהוּ /
וַתִּשְׁלַח אֶת אֲמָתָהּ וַתִּקָּחֵהוּ / וְהוּא כְמֵת מָשְׁכָּב
וַתִּתְאַנַּף / וַתֵּרֵא כִּי לֹא הָיָה נֹדֵד כָּנָף / כַּאֲשֶׁר
אֲחָזַתּוּ הַשָּׁבָץ / בַּמָּקוֹם אֲשֶׁר כָּרַע רָבַץ / וַתִּזְרֹק מַיִם
עַל רֹאשׁוֹ / אוּלַי תּוּכַל לְהָשִׁיב נַפְשׁוֹ / וַתִּקָּחֵהוּ
וַתָּשֶׁת אֹתוֹ בְּחָצְנָהּ / וְלֹא קָם וְלֹא זָע, וְהִיא בְדוּמָה
שָׁכְנָה / וַתַּשְׁלִיכֵהוּ מִיָּדֶיהָ אַרְצָה / וּמִדִּמְעַת עֵינֶיהָ
פָנֶיהָ רַחֲצָה / וַתֵּבְךְּ עָלָיו וַתְּקוֹנֵן לוֹ / וַתִּפָּרֵד
מְבַכָּה מֵאֶצְלוֹ / כִּשְׁמֹעַ הָעוֹף כִּי הָלְכָה / וְעוֹד לוֹ אַךְ
הַמְּלוּכָה / וַיִּפְרֹשׂ כְּנָפָיו וַיָּעָף / וְלָנוּס בְּאֶבְרוֹתָיו
לֹא מוּעָף.



וְהַמָּשָׁל בְּאִישׁ מָלֵא חָכְמָה וָדַעַת / לוֹ עַיִן
רֹאָה וְאֹזֶן שֹׁמַָעַת / לִשְׁמֹעַ מַה בַּחוּץ יְדַבֵּרוּ /
וְעַפְעַפָּיו נֶגְדּוֹ יַיְשִׁירוּ / וּבְטֶרֶם יָבִינוּ זְמָמוֹ בַחוּץ /
לְהָחֵל וְכַלֵּה יִהְיֶה נָחוּץ / בְּמַשָּׂא וּבְמַתָּן בְּמֶקַח
וּבְמִמְכָּר / וְכַזַּרְזִיר יִהְיֶה זָרִיז וְנִשְׂכָּר / אֲשֶׁר
בְּחָכְמָתוֹ נִמְלַךְ וַיֵּלֶךְ הָלֹךְ / מִבֵּית הָסוּרִים יָצָא
לִמְלֹךְ.



וָאֶשָּׂא מְשָׁלִי וָאֹמַר:



אֱנוֹשׁ שֵׂכֶל לְפִי שִׂכְלוֹ יְהֻלַּל / וְהַסָּכָל לְפִי סִכְלוֹ יְחֻלָּל

וְיֵשׁ חָכָם בְּמַרְאֶה לֹא בְחִידוֹת / וְיֵשׁ פֶּתִי בְּסוּת סִכְלוֹ מְגֹלָל

וְאִוַּלְתּוֹ לְעֻמַּת מַחְבַּרְתּוֹ / וְלֹא תָסוּר וְאִם כָּל יוֹם יְקֻלָּל

וְנָבוֹן בִּרְמִיזוֹת בָּן דְּרָכָיו / וְיַכִּירוּ תְבוּנָתוֹ בְּמַעְלָל

כְּמוֹ זַרְזִיר אֲשֶׁר נָפַל לְחָלָל / וְנַפְשׁוֹ הָיְתָה אֵלָיו לְשָׁלָל.
