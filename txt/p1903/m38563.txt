


	
	משל קד: קוֹף וּשְׁנֵי קוֹפָיו וְנָמֵר
	




הַמֹּנֶה סֵדֶר הַבְדָּלוֹת בַּבָּנִים / לְסוֹף יִתְעָרְבוּ עִמּוֹ שׁוֹנִים



קוֹף אֶחָד וּשְׁנֵי קוֹפָיו עַל הַסֶּלַע סְבִיב הַסְּנֶה
/ הַקָּטָן אֹהֵב וְהַגָּדוֹל שֹׂנֵא / וְהִנֵּה נָמֵר שֹׁקֵד מִמְּעוֹנוֹ
יֹרֵד / וְהַקּוֹף עָלָיו וְעַל אֲשֶׁר אָהַב חָרֵד / בִּרְאוֹתוֹ כִּי בָא
מֵהַרֲרֵי נְמֵרִים / וַיֵּצֵא בְחֵמָה גְדוֹלָה לְהַשְׁמִיד וּלְהַחֲרִים
/ וַיִּשְׁאַג עַל טַרְפּוֹ / יְפַלֵּס נָתִיב לְאַפּוֹ / וַיֹּאמֶר
הַקּוֹף: מִן אֲשֶׁר אֶשְׂנָא אֵדַע שְׁכֹל / וְהִסְתַּרְתִּי פָנַי
מִמֶּנּוּ וְהָיָה לֶאֱכֹל / וַיִּקְָחֵהוּ וַיַּשְׁלִיכֵהוּ עַל גַּבּוֹ /
לְתִתּוֹ רִאשׁוֹן אֶל הַנָּמֵר שָׁת לִבּוֹ / וְהַקָּטָן אֲשֶׁר חָמַל
עָלָיו / בְּנוּסוֹ נְשָׂאוֹ בֵּין רַגְלָיו / וּבִרְאוֹתוֹ כִּי הַנָּמֵר
הֹלֵךְ וְקָרֵב רָצָה / לְהַשְׁלִיךְ אֶת אֲשֶׁר עַל גַּבּוֹ אָרְצָה /
וְזֶה יָדַע זְדוֹנוֹ וַיַּחֲזֵק בְּצִצִּית רֹאשׁוֹ / בַּאֲשֶׁר רָצָה
לְמַלֵּט אֶת נַפְשׁוֹ / וּבִרְאוֹתוֹ כִּי לֹא יָכֹל לְהַפִּילוֹ /
וְהַנָּמֵר נִגַּשׁ וּבָא אֶצְלוֹ / עָזַב אֶת אֲשֶׁר אָהַב וְהוּא נִמְלָט
/ עִם הַקּוֹף אֲשֶׁר עַל גַּבּוֹ שָׁלָט / וַיַּרְא הַקּוֹף כִּי
זִמּוֹתָיו נִתָּקוּ / כִּלְיוֹתָיו וְלִבּוֹ עִם הַקּוֹף אֲשֶׁר שָׂנֵא
דָבָקוּ / וַתֵּהָפֵךְ שִׂנְאָתוֹ לְאַהֲבָתוֹ / וַיַּחְמֹל עָלָיו
כַּאֲשֶׁר יַחְמֹל אִישׁ עַל בְּנוֹ הָעֹבֵד אֹתוֹ.



וְהַמָּשָׁל לֶאֱהֹב אֶת בָּנָיו בְּמִדָּה שָׁוָה / כִּי
אֶת אֲשֶׁר יֶאֱהַב יוֹתֵר וְשָׂם בּוֹ תִּקְוָה / בְּפֶתַע פִּתְאֹם
הַגַּלְגַּל יִטְרְפֶנּוּ / הֲתָעִיף עֵינֶיךָ בּוֹ וְאֵינֶנּוּ / וַאֲשֶׁר
שָׂנֵא וְרִחֵק אֹתוֹ / וְהָיָה לוֹ לְמֵשִׁיב נֶפֶשׁ וּלְכַלְכֵּל
שֵׂיבָתוֹ.
