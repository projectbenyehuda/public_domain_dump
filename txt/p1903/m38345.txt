


	
	משל פז: גָּמָל וְשִׁפְעַת גְּמַלִּים
	




הַחֲזֵק יָדַים רָפוֹת / אִם יָדָיו עִמּוֹ עָפוֹת



גָּמָל עִם שִׁפְעַת גְּמַלִּים בְּבֹאוֹ / כָּרַע רָבַץ
תַּחַת מַשָּׂאוֹ / וּלְבַעֲבוּר אֲשֶׁר לֹא הִתְחַזֵּק לָקוּם, מֵעָלָיו
כֹּל יָרוּמוּ / וְיַסְתִּירוּ מִמֶּנּוּ פָנֵימוֹ וְעֵינֵיהֶם הֶעְלִימוּ
/ וַיִּקְרָא לָהֶם: שְׁמָעוּנִי עַמִּי / הֲלֹא עַצְְכֶם וּבְשַׂרְכֶם
אֲנִי וְשִׁמְכֶם כִּשְׁמִי / לָמָּה יֹאמְרוּ הָרֹכְבִים וְעֹבְרֵי
הַדֶּרֶךְ בְּרֵגְלֵיהֶם / הַמּוּרָם מֵהֶם כַּיּוֹצֵא בָהֶם / וּבְנָפְלוֹ
כָּל אֶחָד עָלָה / כַּאֲשֶׁר הֹנָה מִן הַמְסִלָּה / רְאוּ כִּי אֵין
חֵטְא בְּכָל יְגִיעִי / חָנֻּנִּי חָנֻּנִי אַתֶּם רֵעָי / עַד אֲשֶׁר
תְּקִימוּנִי אַל תִּסָּעוּ / אַל נָא אַחַי תָּרֵעוּ / הֵן לִבִּי
עִמָּכֶם שָׁלֵם / אֵיךְ תּוּכְלוּ לְהִתְעַלֵּם / וַיַּעֲנוּ לוֹ: אִם
יָדֶיךָ וְרַגְלֶיךָ תֶּחֱזַקְנָה לַהֲקִימְךָ / גַּם אֲנַחְנוּ עָזֹב
נַעֲזֹב עִמְּךָ / וְאִם בְּשִׁפְלוּת תְּנַהֵג בַּעֲצַלְתַּיִם /
וּבִכְבֵדֻת כְּיָגֵעַ וּרְפֵה יָדַיִם / אֶל עֶזְרָתֵנוּ הֶבֶל צִפִּיתָ /
בַּאֲשֶׁר לַעֲזֹר אֵלֶיךָ הִתְרַפִּיתָ.



וְהַמָּשָׁל לְהָבִין וּלְהוֹרוֹת / לְחַזֵּק יָדַיִם
רָפוֹת אֲשֶׁר מִיֶּשַׁע קְצָרוֹת / בִּרְאוֹתְךָ דָּל וְאֵין יָדוֹ
מַשֶּׂגֶת / וּלְשַׁחֵר פָּנֶיךָ נַפְשׁוֹ מָרָה וְדֹאֶגֶת / בַּאֲשֶׁר
תַּם לָרִיק כֹּחוֹ / וְאָבַד תּוֹחַלְתּוֹ וְנִצְחוֹ / וְאִם תִּרְאֶה
כִּי תִגְבַּר עָלָיו הַדַּלּוּת / בְּעַצְלוּת יָדַיִם וּבְשִׁפְלוּת /
וּבְנֵי בֵיתוֹ בַעֲצַלְתַּיִם רָעֲבוּ וְרָשׁוּ / הֵם כָּל חֲבֵרָיו
יֵבֹשׁוּ / הַשֹּׁמְעִים צַעֲקָתוֹ / וּבְקָהָל תִּגָּלֶה רָעָתוֹ / כִּי
לֹא אִמֵּץ יָדָיו לֶאֱסֹף כְּמוֹהֶם / וּמֵצַח אִשָּׁה זוֹנָה לוֹ
בְּשָׁאֳלוֹ מֵהֶם / וְזֹאת תּוֹרַת הָאָשֵׁם / בֶּן נָבָל גַּם בֶּן
בְּלִי שֵׁם / וְאָמַר הֶחָכָם אֲשֶׁר בְּמַעֲלוֹת הַשֵּׂכֶל עָלָה /
טָעַמְתִּי כָּל מַר וְאֵין מַר כְּטַעַם הַשְּׁאֵלָה.
