


	
	משל נט: שַׁחַל וָצָב
	




אֱמֹר וְאָמַרְתָּ לְבַעֲלֵי אוֹנִים / לְהַזְהִיר גְּדוֹלִים עַל הַקְּטַנִּים



שַׁחַל אֲשֶׁר עַל כֹּל יִתְגַּבַּר / בְּצֵל סֶלַע יָשֵׁן
בַּמִּדְבָּר / וּבִנְקִיק הַסֶּלַע נִצָּב / מֵאַחֲרֵי עָרְפּוֹ הַצָּב /
וַיֹּאמֶר הַצָּב: מִי הוּא זֶה אֲשֶׁר לִפְנֵי חוֹרִי / כָּרַע שָׁכַב
כָּאֲרִי / וַיִּתְמַרְמַר וַיָּרָץ עָדָיו בַּחֲמַת רוּחוֹ /
וַיִּשְׁכֶנּוּ כְפֶתַע בְּמִצְחוֹ / וַיִּיקַץ מִשְּׁנָתוֹ הַשַּׁחַל /
וּבְעֵינָיו אֲשֶׁר כַּנַּחַל / וְכַלַּפִּידִים יְשׁוֹטֵט אוּלַי יֶחֱזֶה
/ אֵיזֶה הוּא וּמִי הוּא זֶה / אֲשֶׁר לַעֲשׂוֹת כֵּן לִבּוֹ מְלָאוֹ /
וְהַצָּב כְּבָר שָׁב דֶּרֶךְ בֹּאוֹ / וַיַּבֵּט אֶל הַשַּׁחַל וְהִנֵַּה
סָר וְזָעֵף / כַּאֲשֶׁר בְּמַעֲרָצָה נֹשְׁכוֹ יְסָעֵף / וַיִּקְרָא לוֹ
הַצָּב: הַתְּבַקֵּשׁ אֶת אֲשֶׁר הֵרַע לְךָ / אֲנִי הוּא הַצָּב הָעוֹמֵד
לְמוּלְךָ / אֲשֶׁר בְּאַפִּי וְזַעְפִּי נְשַׁכְתִּיךָ / וּמְאֹד חָרָה
לִי אֲשֶׁר לֹא הֲמִתִּיךָ / אַלְלַי לְךָ כִּי רָשַׁעְתָּ הֲלֹא תִמְצָא
רִשְׁעֶךָ / פֶּתַע יָקוּמוּ נֹשְׁכֶיךָ וְיִקְצוּ מְזַעְזְעֶיךָ /
הֲבִגְבוּרָתְךָ וּבִגְדֻלָּתְךָ תִּהְיֶה לִי לְשָׂטָן / לַאֲשֶׁר
אֵחָשֵׁב בְּעֵינֶיךָ נִבְזֶה וְקָטָן / כִּי נִגַּשְׁתָּ אֶל הַסֶּלַע
לִסְתֹּם מְבוֹא בֵיתִי / וַאֲנִי בַחֲרוֹנִי אֵלֶיךָ לְשָׂטָן
יָצָאתִי.

וְהַמָּשָׁל לְהַזְהִיר הַגְּדוֹלִים עַל הַקְּטַנִּים /
פֶּן בְּרֹב כֹּחַ וְעָצְמָה וְאוֹנִים / וּבְגַאֲוָה נֶגֶד קְטַנִּים
מֵהֶם יְאְשְׁמוּ / כִּי הַנִּרְדָּפִים חַיִּים עָצְמוּ / וְאִם קְטַנִּים
הֵם, גָּדְלָה גְבוּרַת לִבָּם / וּבְקִרְבָּם יָשִׂימוּ אָרְבָּם /
לִנְקֹם בִּשְׁאַט נֶפֶשׁ, וְיִלָּבֵב הַנָּבוּב / כִּי יֵשׁ עֲנָק נֶחֱנָק
בִּזְבוּב / וְיֵשׁ עֵת יִכָּשְׁלוּ הַנְּפִילִים בַּנְּפָלִים / וּמִן
הָרְחֵלִים יָגוּרוּ הַשְּׁחָלִים.
