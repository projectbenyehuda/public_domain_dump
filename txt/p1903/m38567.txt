


	
	משל קח: חֻלְדָּה וְעוֹף וְשֻׁעָל וְאַרְנֶבֶת
	




אִישׁ מָלֵא רַע וְתַהְפּוּכוֹת / מְשַׁלֵּחַ מְדָנִים בֵּין אַחִים וּמַכּוֹת

חֻלְדָּה דָּרָה בְעִקְּרֵי בָתִּים / וּמִן הַחֲרַכִּים
הֵצִיצוּ חִטִּים / וְהָעוֹף אֹכֵל אוֹתָם קִמְעָא / וַתַּחֲשֹׁב אֶת
נַפְשׁוֹ לְקָבְעָה / וַתֹּאמַר: מַדּוּעַ נַפְשִׁי קָצְרָה / וְלָעוֹף
הַרְבֵּה חִטָּה וּשְׂעוֹרָה / וִימֵי מִשְׁתֶּה וְשִׂמְחָה / כָּל יָמָיו
בְּלִי פֶּרֶץ וּצְוָחָה / לִשְׂחֹק אָמַרְתִּי מְהוֹלָל / וְהָיְתָה לִי
נַפְשִׁי וּבְשָׂרִי לְשָׁלָל / בְּאַוַּת נַפְשִׁי אֹכַל בָּשָׂר /
וּבָרָעָב מָזוֹן לֹא אֶחְסָר / וַתֵּלֶךְ אֵלָיו בַּלָּט / וּמִפַּח
יָקוּשׁ הַצִּפּוֹר נִמְלָט / וַיָעָף מִמֶּנּוּ דֶּרֶךְ קְצָרָה / פֶּן
יִהְיֶה לוֹ לַחֵץ מַטָּרָה / רִחַק כִּמְטַחֲוֵי קֶשֶׁת / וְהַחֻלְדָּה
קָרְאָה לוֹ בַאֲרֶשֶׁת / וַתֹּאמַר: לָמָּה בָרַחְתָּ מִמֶּנִּי / וְאִם
אִשָּׁה תַחְפֹּץ כָּמוֹנִי / וּפָרַשְׂתָּ כְנָפֶךָ עַל אֲמָתֶךָ / כִּי
אָהַבְתִּי מְעוֹן בֵּיתֶךָ / וְעָלֶיךָ הִשְׁלַכְתִּי יַהַב / וְאֵלַי
קְרָאתִיךָ בְּאַהַב / וְתִהְיֶה בְּכָל עֵבֶר רֹדֶה / לְכָה דוֹדִי נֵצֵא
הַשָּׂדֶה / וַאֲנִי דֹרֵס וְאֹכֵל / וְאַתָּה טֹרֵף וּמְשַׁכֵּל /
וּנְחַלֵּק בְּגוֹרָל הַשָּׁלָל יַחַד / טוֹבִים הַשְּׁנַיִם מִן הָאֶחָד /
וְהָעוֹף הֵשִׁיב לָהּ דָּבָר / כִּי הִנֵּה הַסְּתָו עָבָר / וְהִגִּיעַ
עֵת הַזָּמִיר / לֹא אַחֲלִיף וְלֹא אָמִיר / אִשְׁתִּי הַטּוֹבָה בְרָעָה
/ לֹא אוֹסִיף עָלֶיהָ לֹא אֶגְרָעָה / עֵינִי וְזוּגָתִי בְחַיֶּיהָ /
אֲפִלּוּ בִצְבִי אוֹ בִִצְבִיָּה / וְלֹא תִמְצְאִי בִי חֹבֵר חָבֶר /
וְלֹא תוֹסִיפִי עוֹד אֵלַי לְדַבֵּר / וַתַּעַן הַחֻלְדָּה / אֲנִי
שְׁכוּלָה וְגַלְמוּדָה / וּמִמֶּנִּי לֹא תִפְחַד / כִּי בְּבֵיתִי
הֵבֵאתִי לְךָ שֹׁחַד / וְלֹא יָדַעְתָּ אֶת שִׁפְלוּתִי / כִּי יָדַעְתִּי
אָמַרְתִּי אַחֲרֵי מוֹתִי / אֲשֶׁר פִּי זָרוֹת עֲמֻקָּה שׁוּחָה /
הִטַּתּוּ בְרֹב לִקְחָהּ / לָבֹא אֵלֶיהָ עוֹד וְלָגֶשֶׁת / וַתְּהִי לָהּ
לִכְלִמָּה וּלְבֹשֶׁת / וַיְהִי כַאֲשֶׁר הֵם מְדַבְּרִים / וְהִנֵּה מִן
סִבְכֵי הַיְּעָרִים / בָּאוּ שֻׁעָל וְאַרְנֶבֶת / וַיַּכּוּ אֶת
הַחֻלְדָּה עַד מָוֶת / וְהָעַיִט נִמְלַט מִכָּל פֶּגַע רַע / כִּי לֹא
אֹיֵב לוֹ וְלֹא הֵרַע / וְהַחֻלְדָּה אֲשֶׁר חָשְׁבָה נַפְשׁוֹ
לְקַחְתָּהּ / הוּשַׁב גְּמוּלָהּ בְּרֹאשָׁהּ / וְנַעֲשָׂה לָהּ כַּאֲשֶׁר
זָמְמָה / עֲנִיָּה סֹעֲרָה לֹא נֻחָמָה.

וְהַמָּשָׁל בְּאִישׁ מָלֵא רַע וְתַהְפֻּכוֹת /
מְשַׁלֵּחַ מְדָנִים בֵּין אַחִים וּמַכּוֹת / לְהַפִּיל יְדִידָיו
וְרֵעָיו / בְּפַחִים וּמִכְמָרוֹת מִדַּהֲרַת אַבִּירָיו / וְנִמְצֵאת
לַעֲשׂוֹת לוֹ כַּאֲשֶׁר זָמַם / הוּא לָאֲנָשִׁים עַד תֻּמָּם.
