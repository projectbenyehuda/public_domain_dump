


	
	משל פט: פָּרָשׁ נוֹרָא וְאָיֹם
	




אַל תִּתְהַלֵּל בְּגַאֲוָתֶךָ / כִּי לֹא תֵדַע אַחֲרִיתֶךָ



פָּרָשׁ נוֹרָא וְאָיֹם הָיָה / וְנוֹדַע שְׁמוֹ בְּכָל
קִרְיָה / וַהֲמוֹן עָם יֵחָתוּ לְקוֹלוֹ / וְרֹזְנִים מִשְׂחָק לוֹ /
מֵרָחֹק יָרִיחַ מִלְחָמָה וְיִירְאוּ מִגֶּשֶׁת אֵלָיו / בַּחֲגוֹרָתוֹ
אֲשֶׁר בְּמָתְנָיו וּבְנַעֲלוֹ אֲשֶׁר בְּרַגְלָיו / וַתְּהִי לוֹ
לִישׁוּעָה / רַעַם שָׂרִים וּתְרוּעָה / כִּי בְשָׁמְעוֹ אֶת קוֹל צְעָדָה
/ בְּרָאשֵׁי הַבְּכָאִים יָדוֹ הָדָה / לִשְׁפֹּךְ דַּם רְהָבִים
בַּיַּבֶּשֶׁת / לֹא יַבְרִיחֶנּוּ בֶן קָשֶׁת / וּמִצְחַת נְחֹשֶׁת עַל
מִצְחוֹ / וּמָשַׁךְ אַבִּירִים בְּכֹחוֹ / וַיִּשְׂחַק לְשֹׁד וּלְכָפָן /
קוֹל שׁוֹט וְקוֹל רַעַשׁ אוֹפָן / לְנַתֵּק וּלְשַׁבֵּר שִׁרְיוֹנִים
יְמִינוֹ נֶאְדֶּרֶת / כַּאֲשֶׁר יִנָּתֵק פְּתִיל הַנְּעֹרֶת / וְיָדָיו
בַּעֲדָיו יָרִיבוּ / פַּרְסוֹת סוּסָיו כַּצַּר נֶחְשָׁבוּ / וַיֵּצֵא
שְׁמוֹ עַד לְמֵרָחֹק / רֹדֵף וּמָשִּׂיג הָיָה לוֹ לְחֹק / עַל כִּרְאוֹת
לֹחֲמָיו כִּי בְרוֹשָׁיו הָרְעָלוּ / בָּאוּ בֶּעָבִים וּבַכֵּפִים עָלוּ
/ כֻּלָּם כְּאֶחָד מִפָּנָיו נָסִים / אֵלֶּה בָרֶכֶב וְאֵלֶּה בַסּוּסִים
/ וַתְּהִי עַל שִׁכְמוֹ מִן הַפָּרָשִׁים הַמִּשְׂרָה / וַיִּתְנוּ עָלָיו
הוֹד הַגְּבוּרָה / וַיִּפְגַּע בְּסֹחֵר כָּבֵד בְּכֶסֶף וּבְזָהָב /
וַיַּעַשׂ לוֹ מַטְעַמִּים כַּאֲשֶׁר אָהָב / גַּם בְּלִסְטִים מְזֻיָּן
שָׁלַח יָדוֹ / וַיַּהֲפֹךְ לָאָרֶץ הוֹדוֹ / וּבְעוֹד כְּבוֹדוֹ חָדָשׁ
עִמּוֹ / הַזְּמָן שָׁלַח יָדוֹ בִשְׁלוֹמוֹ / וַיִּפְרֹשׂ פַּח לִפְעָמָיו
/ גֶּבֶר לֹא יִצְלַח בְּיָמָיו / וּבְכָל עֵת אֲשֶׁר בַּקְרָב בָּא / אוֹ
נִשְׁבָּר אוֹ נִשְׁבָּה / וְאִם יָרוּץ יִכָּשֵׁל בַּהֲלִיכָתוֹ / אֵין
כֵּהֶה לְשִׁבְרוֹ נַחְלָה מַכָּתוֹ / בַּאֲשֶׁר לוֹ חַלְחָלָה
בְּמָתְנַיִם / וְלֵב נָמֵס וּפִיק בִּרְכַּיִם / וְקֹרָא לוֹ שֵׁם חָדָשׁ
חַלָּשׁ וְלֹא גִּבּוֹר / קָנֶה רָצוּץ לֹא יִשְׁבּוֹר / וַיְגֹרֵהוּ
בְחֶרְמוֹ כָּל שָׁפָל / וּמִמַּעֲלַת גְּבוּרָתוֹ נָפָל / וַיְהִי
כְנָפְלוֹ וְאֵין מֵקִים / וְשׁוֹקָיו רֻתְּקוּ בַזִּקִּים / כִּי עָלָה
מֵפִיץ עַל פָּנָיו / וַיְעַצֵּם אֶת עֵינָיו / וְיָדָיו מִקֶּשֶׁת
אֻסָּרוּ / וְהִבִּיטוּ אֵלָיו אֶת אֲשֶׁר דָּקָרוּ / גַּם הַסֹּחֵר
וְהַגַּזְלָן נִצְמְדוּ יַחַד / וּמִמְּכוֹן שִׁבְתָּם חָזַר וְנֵעוֹר כָּל
אֶחָד / וַיְהִי תָמִיד הֹלֵךְ וּבֹרֵחַ / וַיִּזְעֲקוּ אַחֲרָיו: עֲלֵה
קֵרֵחַ / וְלוּ כִּי נֻתַּן כֹּפֶר נַפְשׁוֹ / יָסִירוּ מֵעָלָיו רֹאשׁוֹ /
וּמִן הַיּוֹם הַהוּא הָלַךְ אֲחֹרַנִּית / מִפְּנֵי לַהַב חֶרֶב וּבְרַק
חֲנִית / וְתַחַת אֲשֶׁר עַל אֶחָיו עַד כֹּה עָצַם וְגָבַר / לֹא יֵצֵא
בַּצָּבָא וְלֹא יַעֲבֹר עָלָיו לְכָל דָּבָר / וַעֲבָדָיו מִקֶּדֶם
יְכַבֵּד וְיַעֲבוֹד / וְיִשְׂבַּע קָלוֹן מִכָּבוֹד / וַיִּשָּׁבַע
לְעֵינֵי כָל מִשְׁפַּחְתּוֹ / לִשְׂרֹף אֶת כָּל כְּלֵי מִלְחַמְתּוֹ /
וְלֹא יֵצֵא עוֹד לַמִּלְחָמָה / וַיַּגְדֵּל הַמְּדוּרָה לְהַשְׁלִיךְ
כֵּלָיו שָׁמָּה / כִּי אָמַר: אֶשְׂרֹף וְאַשְׁמִידָה / סוּס דֹּהֵר
וּמֶרְכָּבָה מְרַקֵּדָה / שִׁרְיוֹן וְכוֹבַע עִמָּם יִמַּח / מָגֵן אִם
יֵרָאֶה וָרֹמַח / חַרְבּוֹ תָקַע בְּבֶטֶן סוּסוֹ אֲשֶׁר אָהַב / וַיָּבֹא
גַּם הַנִּצָּב אַחַר הַלַּהַב / וְאַחֲרֵי כֵן הַחֶרֶב שִׁבֵּר /
וַיַּעַשׂ לְכֵלָיו כַּאֲשֶׁר דִּבֵּר / אִם מֵהֶם כָּל הַמַּחֲנוֹת
הֶחֱרִיד / לֹא הִשְׁאִיר מֵהֶם פָּלִיט וְשָׂרִיד / וּלְמָחֳרָתוֹ זָכַר
אֶת שׁוֹפָרוֹ / אֲשֶׁר בּוֹ תָקַע תּוֹךְ כַּרְמִלּו וְיַעְרוֹ /
וְכַאֲשֶׁר נְשָׂאוֹ בָאֵשׁ לְהַשְׁלִיכוֹ / הָלֹךְ יֵלֵךְ וּבָכֹה /
וַתֹּאמֶר לוֹ אִשְׁתּוֹ / מַדּוּעַ זֶה לִשְׂרֹף זְכַרְתּוֹ / וּמִכְּלֵי
מִלְחָמָה אֵינֶנּוּ / וְלָמָּה זֶּה בָּאֵשׁ תִּשְׂרְפֶנּוּ / לִבְּךָ
מִלְּהַשְׁחִיתוֹ יִתְיָאֵשׁ / הֲלֹא זֶה אוּד מֻצָּל מֵאֵשׁ / הִנְנִי
בְמַעֲשֶׂיךָ נִכְלֶמֶת / אַל נָא תְהִי כַּמֵּת / וַיְהִי כְדַבֵּר אֵלָיו
צַלְעָתוֹ לֹא אַחַת וְלֹא שְׁתַּיִם / וַיַּךְ אֶת הַצֶּלַע בְּמַטֵּהוּ
פַּעֲמָיִם / וַיֹּאמֶר לָהּ בְּחֵמָה: הִנֵּךְ בַּת הַמֶּרִי / כְּדַבֵּר
אַחַת הַנְּבָלוֹת תְּדַבֵּרִי / הִנֵּה גַּם הוּא הָיָה בְעָכְרָי /
לְהָעִיר עָלַי שַׁעֲשׁוּעֵי נְעוּרָי / כִּי כַּאֲשֶׁר קוֹל שֹׁפָר
שָׁמַעְתִּי / כָּל אַנְחָתִי הִשְׁבַּתִּי / בְּדֵי שֹׁפָר אָמַרְתִּי:
הֶאָח / עַל כֵּן אַשְׁלִיכֶנּוּ תוֹךְ הָאָח / כִּי קוֹל תְּרוּעָה
בְּשָׁמְעִי / עַל פִּיו אָכִין מַסָּעִי / וְהִתְעֹרַרְתִּי עַל אֹיְבִי
לְהִלָּחֵם / וְתָבֹא חַרְבִּי וְתִקָּחֵם / וּמִן הַמִּלְחָמָה בְשׁוּבִי
/ לָצוּד צַיִד לְהָבִיא / תָּקַעְתִּי בוֹ וַיֶּחְרְצוּ כְלָבַי
לְשׁוֹנוֹת / כִּי הוּא מְאַסֵּף לְכָל הַמַּחֲנוֹת.



וְהַמָּשָׁל לְהַזְכִּיר כִּי זֶה מִשְׁפַּט חָרוּץ
כַּזֹּרֵק כַּמּוֹשִׁיט / וְאֹחֵז הָרֶגֶל כַּמַּפְשִׁיט / כִּי גַם
הַשֹּׁפָר אֲשֶׁר לֹא מִכְּלֵי מִלְחֶמֶת הַפָּרָשׁ / לְשָׂרְפוֹ עִם
כְּלֵי מִלְחַמְתּוֹ דָּרֹשׁ דָּרָשׁ / אִם הַעְלֵם יַעֶלִים הָרֹאֶה
בְאֶשְׁנָב / דִּין הַמַּעְלִים כְּדִין גַּנָּב.
