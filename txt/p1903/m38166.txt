


	
	משל סג: תַּרְנְגֹל וְתַרְנְגֹלֶת
	






יֵשׁ אֲשֶׁר יָרִיבוּ יַחַד / רֹדְפֵי שַׁלְמוֹנִים וְשֹׁחַד



תַּרְנְגוֹל לָוָה מִתַּרְנְגֹלֶת כֹּר חִטִּים /
וַיְבִיאֵהוּ מִבֵּיתָהּ בְּמִטְפָּחוֹת וּבַחֲרִיטִים / וְהַמּוֹעֵד
אֲשֶׁר הַתַּרְנְגֹל לְשַׁלֵּם אָמַר / עָבַר הַזְּמָן וְהַיּוֹם לֹא
שָׁמַר / וְלֹא בָא וְלֹא שָׁאַל לָהּ זְמָן / וַתֹּאמֶר לוֹ: הֲלֹא
תִהְיֶה נֶאֱמָן / לְשַׁלֵּם כְּיוֹם אֶתְמוֹל כִּי יַעֲבֹר הַכֹּר / אֱמֶת
קְנֵה וְאַל תִּמְכֹּר / וַיֹּאמֶר לָהּ: חָלִילָה לִי לְשַׁקֵּר
בֶּאֱמוּנָתִי / וְחִטִּים צְבוּרִין וּמֻנָּחִין בְּבֵיתִי / בֹּאִי
אִתִּי הַבַּיְתָה וַאֲשַׁלֵּם בְּמִדָּה / כִּי לֹא אֲשַׁקֵּר בְּכָל
כְּלִי חֶמְדָּה / וַתֹּאמֶר לוֹ: הִנֵּה תִצְמַח רְמִיָּה בְרוּחֶךָ /
שִׂים כֹּה נֶגֶד אַחַי וְאַחֶיךָ / וְהָשֵׁב תְּשִׁיבֵם בַּמָּקוֹם אֲשֶׁר
קִבַּלְתָּם / כִּי אֶפְרֹחַי כַּאֲשֶׁר תִּמְדֹד הַחִטָּה יִלְקְטוּ
מַפַּלְתָּם / בַּאֲשֶׁר כִּמְעַט הָרָעָב הֲרָגָן / לְאִמּוֹתָם יֹאמְרוּ:
אַיֵּה דָגָן / כִּי אָזְלַת יָדָם וְדָגָן חָסְרוּ
/ וְעַתָּה יִלְקְטוּ אֶת אֲשֶׁר יִתְפַּזְּרוּ / עַל כֵּן אָמַרְתִּי
לַהֲבִיאָם בִּגְבוּלִי / וּמַפַּל בַּר יִמְצְאוּ הַנִּשְׁעָנִים עַל
מַאֲכָלִי / וַיֹּאמֶר הַתַּרְנְגֹל: לֹא כִי / בִּמְפֻזָּר וּמְפֹרָד לֹא
תְחַיִּי נַפְשֵׁךְ וְנֶפֶשׁ בָּנַיְכִי / אַךְ בֹּאִי וְאֶמְדֹד הַחִטִּים
בְּאָהֳלִי / קֹמְנָה שְׁמַעְנָה קוֹלִי / וַהֲמוֹנַי שָׁם יֹאכְלוּ אֲשֶׁר
יִפֹּל לְעֵינַיִךְ / כַּאֲשֶׁר זָמַמְתְּ לַעֲשֹׂת לְבָנַיִךְ.





וְהַמָּשָׁל מִן הַמְּלִיצִים וְהַשֹּׁפְטִים / אֲשֶׁר
מִשְׁפָּט וָצֶדֶק בָּעִיר מַטִּים / כִּי יֵשׁ אֲשֶׁר יָרִיבוּ יַחַד /
רֹדְפֵי שַׁלְמוֹנִים לֹקְחֵי שֹׁחַד / בְּבֹא דְבַר מִשְׁפָּט אֲלֵיהֶם /
וְיֵרָגְנוּ בְאָהֳלֵיהֶם / לֵאמֹר: זֶה הַמִּשְׁפָּט יִהְיֶה בְּבֵיתִי
וּבַחֲצֵרִי / וְלֹא יִפֹּל אַרְצָה מִן כְּבוֹדִי וִיקָרִי / וְזֶה אֹמֵר:
אַךְ תּוֹךְ גְּבוּלִי וְסִפִּי / וְלֹא יִשָּׁמַע רַק עַל פִּי / וְכָל
זֶה לָקַחַת שֹׁחַד וּלְהַטּוֹת צֶדֶק בְּתַרְמִית וּבְתֹאֲנָה / אֵין
קֹרֵא בְצֶדֶק וְאֵין נִשְׁפָּט בֶּאֱמוּנָה / וְאֵין אֶחָד מִתְעוֹרֵר
לְהַחֲזִיק בְּצֶדֶק / אַךְ לִטְחֹן אֹתוֹ הָדֵק / הֶחָפֵץ יְמַלֵּא אֶת
יָדוֹ וִיסַלֵּף אִמְרָתוֹ / הוֹי בּוֹצֵעַ בֶּצַע רַע לְבֵיתוֹ.
