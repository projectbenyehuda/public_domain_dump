


	
	משל צח: עֹרֵב וּנְבֵלָה
	




הַשְּׁתִיקָה לְבַעַל דַּעַת / תּוֹסֶפֶת וְלֹא מִגְרַעַת

עֹרֵב עַל נָהָר לֹא יְדָעוֹ עַיִט / וּבַל תֵּלֶךְ בּוֹ
אֳנִי שַׁיִט / וַיָּעָף אָנֶה וָאָנָה וַיִּרְעַב הָעֹרֵב / וַיִּלְאֶה
לִמְצֹא צֵידוֹ עַד הָעָרֶב / וַיִּפְגַּע בַּמָּקוֹם וַיִּמְצָא שָׁם
נְבֵלָה / וַיִּשְׂמַח עָלֶיהָ שִׂמְחָה גְדוֹלָה / וּמֵרֹב שִׂמְחָתוֹ
אוֹתָהּ יִצֶף / נֹדֵד כָּנָף וּפֹצֶה פֶה וּמְצַפְצֵף / וַיִּפְעַר פִּיו
לִבְלִי חֹק / וְהַקּוֹל נִשְׁמַע עַד לְמֵרָחֹק / וְאָזְנֵי הַנֶּשֶׁר
תְּצִלֶּינָה וַיֹּאמַר: לִרְאוֹת אָחִישָׁה / אִם הַקּוֹל עֲנוֹת
מִלְחָמָה אוֹ קוֹל עֲנוֹת חֲלוּשָׁה / וַיָּבֹא עַד הָעֹרֵב
וַיִּמְצָאֵהוּ עַל הַנְּבֵלָה יְנַתֵּר / וּבָשָׂר מִן הָעֲצָמוֹת יְפָרֵק
וִיבַתֵּר / וַיִּתְמַרְמֵר אֵלָיו בַּחֲמַת רוּחוֹ וַיַּכֵּהוּ / שָׁם
שָׂם לוֹ חֹק וּמִשְׁפָּט וְשָׁם נִסָּהוּ / וּמִן הַיּוֹם הַהוּא
וָהָלְאָה לֹא גָבַהּ קוֹלוֹ / לְהַשְׁמִיעַ בַּחוּץ אֶל הָעוֹפוֹת
שְׁלָלוֹ.

וְהַמָּשָׁל כִּי הַרְבֵּה נִלְכְּדוּ וְנוֹקְשׁוּ / אִם
הִקְרָה לָהֶם הַזְּמָן קְצִירָם לֹא הֶחֱרִישׁוּ / וְכַאֲשֶׁר לֹא
שָׁמְרוּ פִּתְחֵי פִיהֶם / וַיִּשָּׁמַע אֶל הַמּוֹשְׁלִים עֲלֵיהֶם / אָז
כָּל אֶחָד כָּאֲרִי נָהַם / כְּקוֹץ מֻנָד כֻּלָּהַם / לֶאֱרֹב כַּאֲרִי
בְּמִסְתָּרִים / וַיָּשִׂימוּ עֲלֵיהֶם עֲלִילוֹת דְּבָרִים / עַד לֹא
יַשְׁאִירוּ עוֹלֵלוֹת / וְהֵמָּה בִקְשׁוּ מוֹקְשִׁים בְּמַהֲתַלּוֹת /
וְלִמְצֹא שֶׁבֶר וָשֹׁד / אָפֵס הַמֵּץ כָּלָה שֹׁד / בְּמַאֲמַר פֶּה
וּבְחֹסֶר שְׁתִיקָה / רוּחַ אֲנָשִׁים נָבְקָה / וָאֹמַר לַאֲשֶׁר לֹא
אִמְרַת שְׂפָתָיו שְׁמוּרָה וַעֲרוּכָה / הִנּוֹ כְעֻגָּה בְּלִי הֲפוּכָה
/ וְאָמַר קֹהֶלֶת אֲשֶׁר הַתְּבוּנוֹת זָרַע / בְּפֶשַׁע שְׂפָתַיִם
מוֹקֵשׁ רָע / וְהַיִּשְׁמְעֵאלִי אָמַר בְּמוּסָרוֹ לִבְנוֹ בְכוֹרוֹ /
סוֹדְךָ אֲסִירְךָ, וְאִם תְּגַלֵּהוּ תִּהְיֶה אֲסִירוֹ.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

מוֹקֵשׁ־רֶגֶל קָטָן מוֹקְשׁוֹ / אַךְ מוֹקֵשׁ־פֶּה יָסִיר רֹאשׁוֹ
דַּע כִּי שֹׁמֵר פִּיו וּלְשׁוֹנוֹ / שֹׁמֵר מִצָּרָה אֶת נַפְשׁוֹ.
