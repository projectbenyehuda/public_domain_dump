


	
	משל עח: קוֹף וָמֶלֶךְ
	




אָרַח עִם שְׂפַת שֶׁקֶר לַחֲבוּרָה / נַפְשׁוֹ בְעֵינָיו לֹא יָקְרָה



קוֹף מֶלֶךְ גָּדוֹל גִּדְּלוֹ / יָמִים רַבִּים עָמַד
אֶצְלוֹ / עַד אֲשֶׁר הִכִּיר כָּל מַעְבָּדָיו / מַאֲכַל שֻׁלְחָנוֹ
וּמוֹשַׁב עֲבָדָיו / וַאֲשֶׁר מָלַךְ עֲלֵיהֶם וְגָבַר / לֹא נֶעְלַם
מִמֶּנּוּ כָל דָּבָר / לָצֵאת מִשָּׁם הָיָה נָחוּץ / וְלִמְשֹׁל עַל
אֶחָיו בַּחוּץ / וַיְהִי הַיּוֹם וַיָּכֶן אֶת מִבְרָחָיו / וַיֵּצֵא
הַשָּדֶה אֶל אֶחָיו / בְּמִדְבָּרוֹת עַל הֶהָרִים / בִּסְעִיפֵי סֶלַע
בַּיְעָרִים / וַיֹּאמֶר לִפְנֵי קוֹפֵי יָעַר / אִי לָךְ אֶרֶץ
שֶׁמַּלְכֵךְ נָעַר / עַל כֵּן יָרְדוּ טוֹבִים מַטָּה / כָּל רֹאשׁ
מֻקְרָח וְכָל כָּתֵף מְרוּטָה / וְאֵין אֶחָד מִכֶּם גֹּדֵר פָּרֶץ / מִי
יִתְּנֵנִי שֹׁפֵט בָּאָרֶץ / לְהָשִׁיב לַצַּדִּיק כְּצִדְקָתוֹ / וְלֹא
תוֹנוּ אִישׁ אֶת עֲמִיתוֹ / וְאֶעֱשֶׂה לַכֹּל כַּאֲשֶׁר זָמַם / אֲנִי
לֹכֵד חֲכָמִים בְּעָרְמָם / יֵאוֹת לִי עַל רֵעַי לִהְיוֹת מֹשְׁלָם
וְרֹדֵם / בֶּן חֲכָמִים אֲנִי בֶּן מַלְכֵי קֶדֶם / הַקּוֹפִים דְּבָרָיו
הֶאֱזִינוּ / וַיֹּאמְרוּ: קָצִין תִּהְיֶה לָנוּ / וַיִּבְחַר מִכָּל
הַקּוֹפִים / הַגִּבּוֹרִים הַיָּפִים / לִהְיוֹת נִצָּבִים עָלָיו /
וְיֵשׁ לְהָכִין מַאֲכָלָיו / וְהַנִּבְזִים וּרְפֵי מָתְנַיִם / נְתָנָם
חֹטְבֵי עֵצִים וְשֹׁאֲבֵי מַיִם / וְהַזְּקֵנִים הֵקִים שֹׁפְטִים /
וַיְהִי בִימֵי קְצִיר חִטִּים / שְׁנֵי אֲנָשִׁים בַּמִּדְבָּר תָּעוּ /
וַיָּבֹאוּ עַד הַמָּקוֹם אֲשֶׁר תָּקְעוּ / הַקּוֹפִים שָׁם אֶת
אָהֳלֵיהֶם / וַיִּרְאוּ כִּי בְקֶרֶב מַחֲנֵיהֶם / מֶלֶךְ וְכָל אֶחָד
הֹלֵךְ / לֵאמֹר: יְחִי אֲדֹנֵינוּ הַמֶּלֶךְ / וְקוֹף קָטָן עֹמֵד
לְפָנָיו / וַיִּמְצָא חֵן וָחֶסֶד בְּעֵינָיו / עַל כֵּן חֲשָׁבוּהוּ
הָאֲנָשִׁים לִבְנוֹ / נִצְּבָה שֵׁגָל לִימִינוֹ / אֵצֶל הַחֲרֻלִּים עַל
שִׂיחִים / אֹכְלִים וְשֹׁתִים וּשְׂמֵחִים / זֶה שַׂר הָאֹפִים וְזֶה שַׂר
הַמַּשְׁקִים / וְזֶה שֹׁמֵר כֵּלִים רֵיקִים / זֶה נִקְלֶה וְזֶה נִכְבָּד
/ כְּחֹק וּמִשְׁפָּט בֵּין אֲנָשִׁים / אֲשֶׁר רָשִׁים עֹבְדִים רָאשִׁים
/ וְהַקּוֹף בַּאֲנָשִׁים הִתְבּוֹנָן / וּבְכִלְיוֹתָיו יִשְׁתּוֹנָן /
וַיְהִי כַאֲשֶׁר רָאָם / בְּשִׂפְתֵּי רָצוֹן לוֹ קְרָאָם / לַגָּדוֹל
אָמַר: אֲנִי מַשְׁבִּיעֲךָ / לְהַגִּיד לִי מַה בְּמַדָּעֲךָ / אִם הוֹד
הַמַּלְכוּת עָלַי / אִם גִּבּוֹרִים נֹשְׂאֵי כֵלָי / וְאִם בְּנִי
לִמְאֹד יָפֶה / הֶחָזָק הוּא הֲרָפֶה / וְאִם יְפַת מַרְאֶה אִשְּׁתִּי /
אֲשֶׁר נִצְמֶדֶת אִתִּי / וְאִם לִמְאֹד נָעֲמוּ מְשָׁרְתַי / וְשֹׁמְרֵי
בֵיתִי וְאוֹצְרוֹתָי / הָאִישׁ הַנֶּאֱמָן אֲשֶׁר הִשְׁבִּיעַ / לְפִי
רְאוּת עֵינָיו יַבִּיעַ / וַיֹּאמַר: מִכָּל אֲשֶׁר רָאִיתִי / לֹא
אֲשַׁקֵּר בֶּאֱמוּנָתִי / כָּאָב כַּבֵּן כְּאִמָּהּ כְּבִתָּהּ /
כָּעֶבֶד כַּאֲדֹנָיו כְּשִׁפְחָה כִגְבִרְתָּהּ / קוֹפִים אַתֶּם
וְקוֹפִים אֲבוֹתֵיכֶם / בָּרוּךְ אֲשֶׁר מָאַס בָּכֶם / קוֹפִית
אִשְׁתְּךָ וְקוֹף בִּנְךָ / נִצְּבָה שֵׁגָל לִימִינְךָ / תָּאֳרָהּ
מָשְׁחָת וְהוּעָם / נִבְזִים וּשְׁפָלִים לְכָל הָעָם / אַתֶּם פְּרִיצֵי
חַיּוֹת וְעִקְּשִׁים / וּמֵעֲדַת כָּל חַי נִגְרָשִׁים / תַּחַת
כְּבוֹדְכֶם חֶרְפָּה / וְתַחַת חֲגוֹרָה נִקְפָּה / קוֹץ מַכְאִיב
וְסִלּוֹן מַמְאִיר / אַחֲרֵיכֶם נָתִיב לֹא יָאִיר / אַךְ תַּבְאִישׁוּ
כָּל מָלוֹן / יְרֻשַּׁתְכֶם בּוּז וְקָלוֹן / קוֹף הַמֶּלֶךְ הַדָּבָר
שָׁמָר / לָאִישׁ הַשֵּׁנִי שָׁאַל וְאָמַר / מִי יִתֵּן וְאֶשְׁכְּנָה
בְמִגְרָשֶׁיךָ / אַשְׁרֵי עֲבָדֶיךָ וְאַשְׁרֵי אֲנָשֶׁיךָ / הַשֹּׁמְעִים
אֶת חָכְמָתֶךָ / וְרוֹאִים יְפִי חֲבוּרָתֶךָ / אֲשֶׁר לִמְאֹד גָּבְרָה
יָדָם / יָפְיָפִיתָ מִבְּנֵי אָדָם / וְאִם כָּל יָפִים אֶצְלְךָ
נִקְהָלוּ / אוֹר פָּנֶיךָ לֹא יַפִּילוּ / וְאִשְׁתְּךָ מְאֹד נָעֲמָה /
כְּלִילַת יֹפִי וְחָכְמָה / לֵב כָּל מֶלֶךְ בִּרְאוֹתָהּ יִגְאֶה /
יְפֵַת תֹּאַר וִיפַת מַרְאֶה / בִּנְךָ לֹא נֶחְקַר הוֹדוֹ / מִכַּף
רַגְלוֹ וְעַד קָדְקָדוֹ / אֵין בּוֹ מוּם וְכָל דֹּפִי / לֹא דֹמֶה לוֹ
כָּל חַי בְּיֹפִי / קוֹפִים קָרְאוּ אֶל הַמְשַׁקֵּר / וַיְכַבְּדוּהוּ
עַד אֵין חֵקֶר / וְלַנֶּאֱמָן בְּגָדָיו קָרָעוּ / וְאַחֲרֵי כֵן
הִכּוּהוּ וּפָצָעוּ / וְהַמֶּלֶךְ צִוָּה לַהֲשִׁיבוֹ אֶל מְסִלּוֹתָיו /
לְהוֹדִיעַ לִבְנֵי אָדָם גְּבוּרוֹתָיו.



וְהַמָּשָׁל בְּאִישׁ אֲשֶׁר אֵין רְמִיָּה
בְּעֶשְׁתֹּנוֹתָיו / וְעַוְלָה לֹא נִמְצָא בִשְׂפָתָיו / וְאָרַח עִם
שְׂפַת שֶׁקֶר לַחֲבוּרָה / נַפְשׁוֹ בְעֵינָיו לֹא יָקְרָה / כִּי זֶה
לְהַחֲנִיף הָאֱמֶת יְסַלֵּף / וְזֶה לֹא יָמִיר וְלֹא יְחַלֵּף / וְאִם
בַּעֲבֹתוֹת הָאֱמֶת יִהְיוּ נִצְמָדִים / אַחֲרִיתָם לִהְיוֹת מְכֻבָּדִים
/ וְהָהּ לַיּוֹם אֲשֶׁר אִישׁ בְּפִיו נַפְשׁוֹ יְטַנֵּף / לִקְבֹּץ אָוֶן
לַעֲשׂוֹת חֹנֶף / וּלְבַעֲבוּר כִּי חֲנֻפָּה רַבּוּ אֹהֲבֶיהָ / הָאָרֶץ
חָנְפָה תַּחַת יֹשְׁבֶיהָ.
