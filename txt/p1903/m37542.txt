


	
	נְאֻם בֶּרֶכְיָה בַּ״ר נַטְרוֹנַאי הַנַּקְדָּן / אֲשֶׁר מְשָׁלִים אֵלֶּה הִצְמִיחָן וְהוֹלִידָן:
	




אֵיכָכָה אוּכַל וְרָאִיתִי בְּאָבְדָן / אִם לֹא
אֶכְתְּבֵם בְּסֵפֶר לְזִכָּרוֹן / בַּעֲמָלִי מַה־יִּתְרוֹן / וְאִם
אֵחָשֵׁב כְּסִיל אוֹ יָהִיר / לְשׁוֹנִי עֵט סוֹפֵר מָהִיר / מִי יִתֵּן
וְיִכָּתְבוּן מִלַּי / וְסִבַּת עֵט מְכַתֵּב מְשָׁלַי / עַל־אוֹפַן
גַּלְגַּל הָעוֹלָם / אֲשֶׁר מֵעֵינֵי שִׂכְלִי נֶעֱלָם / הַמִּתְגַּלְגֵּל
בְּאִי הַיָּם / אֵלֶּה הֵמִית וְאֵלֶּה חִיָּם / וַיִּפְשֹׁט בְּמֶרְחֲבֵי
אָרֶץ / לִפְרֹץ פֶּרֶץ עַל־פְּנֵי־פָרֶץ / וּבְגַלְגַּל עֶגְלָתוֹ הָמַם /
הַיְשָׁרִים עַד תֻּמָּם / וְאוֹפַן מַרְכְּבוֹתָיו / מִתְהַפֵּךְ
בְּתַחְבּוּלוֹתָיו / עַל הַטּוֹבִים וְיַכְנִיעֵם / וְעַל הָרָעִים
וְיוֹשִׁיעֵם / וַיַּרְא כִּי הָאֱמֶת עָמְדָה בַמִּישׁוֹר / וְכִי יָדֶיהָ
שִׁלְּחָה בַכִּישׁוֹר / וַיַּךְ אוֹתָהּ אֶל הַחֹמֶשׁ / וַיָּנֶף עַל
רַגְלֶיהָ חֶרְמֵשׁ / וַיְקַצְּצֵם וַיִּרְעֲצֵם / הַבָּשָׂר עִם הָעֶצֶם /
וַיְסִירֶהָ מִגְּבִירָה / כַּאֲשֶׁר לִזְנוּנִים הָרָה / וַתְּהִי לְדֹפִי
/ וַתֵּלֶךְ שֶׁפִי / וְתַחְתֶּיהָ אֶת הַשֶּׁקֶר אָהַב / וַיּוֹשֶׁט לוֹ
אֶת שַׁרְבִיט הַזָּהָב / וַיִּקְרָאֵהוּ לְפָנָיו / וַיִּמְצָא חֵן
וָחֶסֶד בְּעֵינָיו / וַיַּעֲמִידֵהוּ עַל
רַגְלַיִם / וַיְיַפֵּהוּ בִיפִי נַעֲלַיִם / וַיְקָרֵב כָּל קְרוֹבָיו
וְאֹהֲבָיו / וַיְרַחֵק בּוֹזָיו וְאֹיְבָיו / עַד כִּי עָלוּ בְּמַעֲלוֹת
/ פּוֹעֲלֵי עַוְלוֹת / וַיָּשִׂימוּ עֲלִילוֹת / לְהַכְרִית עוֹלֵלוֹת /
וּנְטִישׁוֹת וְזַלְזַלִּים / מֵאֶבְיוֹנִים וְדַלִּים / וַאֲשֶׁר לֹא
יַעֲשׂוּ כְמוֹהֶם / כְּלֵי קְרָב בְּפִיהֶם / חֲרָבוֹת שִׂפְתּוֹתֵיהֶם /
חֲנִית וְחִצִּים שִׁנֵּיהֶם / לְשׂוּמָם מִרְמָס כְּטִיט חוּצוֹת / עַל
כֵּן מָעֲטוּ גוֹדְרֵי פְרָצוֹת / כִּי הָרְשָׁעִים נוֹשָׁעִים / וְלֹא חֵן
הַיּוֹדְעִים / וְהַצַּדִּיקִים נֶאֱנָקִים / וְהַמָּרִים נִמְתָּקִים /
וּבְנֵי בְלִיַּעַל עָלוּ / וְהַגְּבֹהִים יִשְׁפָּלוּ / וְהַתְּפִלָּה
תִפְלָה / וְהַתְּהִלָּה תָהֳלָה / וְהָעוֹלָה עַוְלָה / וְהַמְּזִמָּה
זִמָּה / וְשָׁרְשָׁם פָּתוּחַ אֱלֵי מִרְמָה / יוֹצְאִים בְּיָד רָמָה /
וְקָרְאוּ בִשְׁמוֹתָם עֲלֵי אֲדָמוֹת / אֲשֶׁר נִמְשְׁלוּ כַבְּהֵמוֹת /
וְאַנְשֵׁי צֶדֶק / כִּמְשֻׂכַת חֶדֶק / וְטָחֵי תָפֵל / יוֹשְׁבִים
בָּעֹפֶל / וְרוּם אֲנָשִׁים שָׁפֵל / בַּחֹשֶׁךְ וַעֲרָפֶל / כְּפוּפֵי
רֹאשׁ כְּאַגְמוֹן / וְיִירְאוּ מִקּוֹל הָמוֹן / אִישִׁים אֲשֶׁר מָאֲסוּ
אֲבוֹתָם / הַיֹּשְׁבִים בְּחַצְרֵיהֶם וּבְטִירוֹתָם / עֲנוּקֵי גַאֲוָה
וָרָהַב / כּוֹסְפֵי כֶסֶף, אֹהֲבֵי זָהָב / הַיְרֵאִים לִשְׁלֹחַ יָד
בְּהוֹנָם / אַךְ לְשַׁקֵּר בִּלְשׁוֹנָם / וְלֹא הֵבִיאוּ צַוָּארָם /
בֵּעֲבוֹדַת צוּרָם / מִגַּאֲוָה נֶעֶנְקוּ / וְהַעֲנֵק לֹא הֶעֱנִיקוּ /
אַךְ שָׁמְנוּ עָשְׁתוּ / כֹּל אֲשֶׁר יָזְמוּ עָשְׁתוּ / בְּבֶצַע
מַעֲשַׁקּוֹת / לְהַחֲנִיף בַּחֲלָקוֹת / וַיַּעֲשׂוּ וַיַּצְלִיחוּ /
כַּיָּמִים אֲשֶׁר נָחוּ / וְהֹלֵךְ צֶדֶק וְדֹבֵר מֵישָׁרִים / צַלְמָוֶת
וְלֹא סְדָרִים / וּמִקּוֹל מְחַצְּצִים / הַמִּתְלוֹצְצִים / סְעִפַּי
בִּי נִצִּים וְאָצִים / לִנְטוֹת עֵטִי לְמָשָׁל / עַל הַשֶּׁקֶר אֲשֶׁר
מָשַׁל / לָשֶׁבֶת עַל מְרוֹמֵי קָרֶת / וַתְּהִי הָאֱמֶת נֶעְדָּרֶת /
וְסָר מֵרָע מִשְׁתּוֹלֵל / וְרָשָׁע עַל תַּאֲוָתוֹ הִלֵּל / וּמָלְאוּ
פְנֵי תֵבֵל / רֵיקִים עוֹשֵׂי תֶבֶל / וּמְכַבְּדֵי אֵל יֵקָלוּ /
וְרֹדְפֵיהֶם יִקַּלוּ / הָהּ כִּי נִלְכְּדָה בִמְצוּדָה / הָאֱמֶת אֲשֶׁר
עָמְדָה / וְהַנְּבָלִים עָלוּ לְמַעְלָה / וְהַנְּדִיבִים יָרְדוּ
שְׁאוֹלָה / וְשָׁב כָּל רֹאשׁ לְזָנָב / וַיִּשְׂתָּרֵר כָּל בְּלִיַּעַל
וְגַנָּב / וְקָצְרוּ דַרְכֵי אֱמֶת וְנֶחְבָּאוּ / וַיַּאֲרִיכוּ
הַבַּדִּים וַיֵּרָאוּ / וְהָעֲנִיִּים בְּעֵינֵי הָעֲשִׁירִים כְּדֻבִּים
/ וְאֹהֲבֵי עָשִׁיר רַבִּים / וְהָאֱוִיל מַשְׁרִישׁ / וְהֶחָכָם
יִשְׂבַּע רִישׁ / וְיִרְבּוּ וְיַעַצְמוּ / עֵינַיִם קָרְצוּ וְעָצְמוּ /
וּנְשִׁיקוֹת אַחִים / אֲשֶׁר בִּלְחֵיהֶם חַחִים / וְקֹוֵי לְמֶתֶק סוֹד
רֵעִים / כִּי יִמְצָאוּם רָעִים / וּמְלָשְׁנִי בַסֵּתֶר / יֵאָמֵר לוֹ:
אָכֹל וְהוֹתֵר / בְּפִיו מְהַלֵּל / וּבְקִרְבּוֹ מְקַלֵּל /
מִשְׁתַּחֲוֶה וְכֹרֵע / מְחַשֵּׁב אַךְ לְהָרֵעַ / וְהָאוֹפָן אֲשֶׁר כָּל
זֶה יַצְמִיד / עֹבֵר עָלֵינוּ תָמִיד / וְיַךְ בְּזַעְפּוֹ וְיַגֶּה /
כָּל חֵךְ בֶּאֱמֶת יֶהְגֶּה / וּמִתְהַלֵּל בְּמַתַּת שֶׁקֶר / לִכְבוֹדוֹ
אֵין חֵקֶר / נְשִׂיאִים וְרוּחַ וְאֵין גֶּשֶׁם / בְּנֵי נָבָל גַּם
בְּנֵי בְלִי שֵׁם / וּשְׂפָתַיִם דֹּלְקִים וְלֶב־רָע / יַאֲרִיךְ יָמִים
יִרְאֶה זֶרַע / וּבִמְעִיל בּוּץ מְכֻרְבָּל / וְזֵר עַל צַוָּארוֹ
סַרְבָּל / בְּלִי כֹל שָׁלוֹם, בְּלִי בֶגֶד / וְשָׁלוּ כָּל בֹּגְדֵי
בָגֶד / בְּפֶה חָנֵף יַשְׁחִית רֵעֵהוּ / וִיחַבְּקֵהוּ וִינַשְּׁקֵהוּ /
חֹנֶף בְּפִיו וְאָוֶן בְּלִבּוֹ / וּבְקִרְבּוֹ יָשִׂים אָרְבּוֹ / דְּבַר
בְּלִיַּעַל יָצוּק בּוֹ / וְכָל אָח עָקוֹב יַעֲקוֹב / וּבִרְאוֹתִי זֶה
הַזְּמָן אֶקּוֹב.





וָאֶשָּׂא מְשָׁלִי וָאֹמַר:



נְהֵה עֵטִי עֲלֵי אוֹפַן עֲגָלָה / וְלַגַּלְגַּל אֲשֶׁר נִגְלַל וְנִגְלָה

וּמִתְהַפֵּךְ עֲדֵי כָשְׁלָה וְנָפְלָה / שְׂפַת הָאֱמֶת וְשֶׁקֶר קָם לְמָעְלָה

וְהָיְתָה כָּל תְּפִלַּת זַךְ לְתִפְלָה / וְחַף רֹאשׁ חָף וְהַתְּהִלָּה לְתָהְלָה

וְהֻקַּם עָל בְּלִיַּעַל וְעוֹלָה / יְמִין עַוְלָה וְהַתָּם תַּם וְנִקְלָה

בְּאִי הַיָּם עֲדַת שֵׂכֶל שְׁכוּלָה / וְעֶטֶר בּוּז וּמִשְׁפָּחָה שְׁפָלָה

וְאֹזֶן רֹב עֲשִׁירֶיהָ עֲרֵלָה / לְכָל שֹׁאֵל וְלַנֹּתֵן מְהוּלָה

וְחִפְּשׂוּ בוֹגְדֵי בֶגֶד עֲלִילָה / שְׁלֹל בֶּגֶד יְשַׁר־דֶּרֶךְ וְשִׂמְלָה

עֲדַת חָנֵף שְׂמֵחָה הִיא וְצָהְלָה / וְתוֹךְ מַחְנֵה יְשָׁרִים קוֹל יְלָלָה

וְאִשְּׁרוּ פֶה אֲשֶׁר דֹּבֵר נְבָלָה / וְהַדֹּבֵר נְכֹחוֹת לִקְלָלָה

וְרֵעַ כַּטּוֹב וּבָלָה כַבְּתוּלָה / כְּמָתוֹק מַר וְאוֹרָה כַאֲפֵלָה

וּבֶרֶכְיָה זְמָן קִלֵּל וְאָלָה / וְסִבּוֹתָיו קְטַנָּה וּגְדוֹלָה

וְטוֹב לִי פַת חֲרֵבָה וַעֲמֵלָה / וְזוּלָתָם וְלֹא עִמָּם בְּנַחְלָה.





כְּחָשְׁבִי עַל זֶה יְבַהֲלוּנִי רַעְיוֹנַי / וְתִדַּד
שְׁנָתִי מֵעֵינַי / לָשׂוּם בְּסֵתֶר דִּבְרָתִי / מִלִּים אֲשֶׁר
מָלֵאתִי / וָאֹמַר לַאֲשֶׁר יָבֹא מִכְתָּבִי לַחֲזוֹת / אֱכֹל אֶת
הַמְּגִלָּה הַזֹּאת / כִּי מֵהָאֹכֵל יֵצֵא מַאֲכָל / וְשֵׂכֶל יֵצֵא
מִשָּׂכָל / וְאַל יֹאמַר אֲשֶׁר מְשָׁלַי יִזְכּוֹר / חוֹחַ עָלָה בְיַד
שִׁכּוֹר / וּמָשָׁל בְּפִי כְסִילִים / כְּרֹב חֲלוֹמוֹת וַהֲבָלִים /
כִּי מֵרְשָׁעִים יֵצֵא רֶשַׁע / וּמִן הַמּוֹשִׁיעִים יֶשַׁע / אִם אִישׁ
עָלַי יַמְרֶה חִכּוֹ / חָכְמַת עָרוּם הָבִין דַּרְכּוֹ.





וְהִנְנִי מְמַשֵּׁל בְּחַיּוֹת וְעוֹפוֹת / לְחַזֵּק
יָדַיִם רָפוֹת / וּבָרֶמֶשׂ וּבַשֶּׁרֶץ / הִנָּם לְמָשָׁל לְיוֹשְׁבֵי
הָאָרֶץ / וְאָחֵל בָּאַרְיֵה הַמֹּלֵךְ עַל כֻּלָּם / מִקְּטַנָּם וְעַד
גְּדוֹלָם / כִּי כְּבוֹדוֹ בְּקָלוֹן הֵמִירוּ / כֵּן לֶעָשִׁיר
שֶׁנּוֹרַשׁ חֲבֵרָיו לוֹ יִתְנַכָּרוּ.
