


	
	משל עו: שֶׁמֶשׁ וְיָרֵחַ
	




הִכָּבֵד בְּכָבוֹד וּמִקְנֶה / לֹא יִתְרַצֶּה בְּכָהֵנָה וְכָהֵנָה



הַשֶּׁמֶשׁ אֲשֶׁר בַּיּוֹם לְמֶמְשֶׁלֶת / פְּנֵי כָּל
הַיְקוּם מְחַלָּה וְשֹׁאֶלֶת / לָתֵת לָהּ הַלְּבָנָה לַחֲבוּרָה /
וַיַּעֲנוּ לוֹ: יָדְךָ לִמְאֹד עָצְמָה וְגָבְרָה / לֹא נַעֲשֶׂה לְךָ
עֵזֶר כְּנֶגְדְּךָ / הִנֵּה בִּהְיוֹתְךָ לְבַדְּךָ / אֶל הַשֶּׁלֶג
עָשִׂיתָ חָמָס / וְחַם הַשֶּׁמֶשׁ וְנָמַס / וּבְצֵאתְךָ בִּגְבוּרָתְךָ
תַּבְהִיל הַמֶּלֶךְ וְחֵילוֹ / אִם זָרְחָה הַשֶּׁמֶשׁ עָלָיו דָּמִים לוֹ
/ וְתֵצֵא מֵחֻפָּתְךָ כְּחָתָן / הֵן הוֹבַשְׁתָּ נַהֲרוֹת אֵיתָן /
וְאֵיךְ נַצְמִיד אֵלֶיךָ הַלְּבָנָה / לֹא תֵלֵךְ אִתְּךָ אָנֶה
וְאָנָה.



וְהַמָּשָׁל בְּאִישׁ אֲשֶׁר לוֹ עֹשֶׁר וָהוֹן וְכָבוֹד
וּמִקְנֶה / וְחָפֵץ לְהוֹסִיף כָּהֵנָּה וְכָהֵנָּה / לְהַעְדִּיף עַל
תְּכוּנָתוֹ לְהַגְבִּיר / בַּאֲשֶׁר לֹא מָצְאָה יָדוֹ כַבִּיר / לֹא
תֹאבֶה וְלֹא תִשְׁמַע בְּקוֹלוֹ / לְהוֹסִיף חַיִל עַל חֵילוֹ /
וּשְׁלֹמֹה הֶחָכָם יִסְּרָנוּ יָסוֹר / נֹתֵן לְעָשִׁיר אַךְ
לְמַחְסוֹר.
