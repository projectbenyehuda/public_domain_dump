


	
	משל כב: נָחָשׁ וְעָשִׁיר
	






מְשַׁלֵּם רָעָה תַּחַת טוֹבָה / מֵבִיא עַל נַפְשׁוֹ חוֹבָה



נָחָשׁ הָיָה בָּא וָשָׁב לֶעָשִׁיר בִּגְבוּלוֹ /
וַאֲרוּחָתוֹ אֲרוּחַת תָּמִיד נִתְּנָה לוֹ / מֵאֵת הֶעָשִׁיר עֵת בּוֹאוֹ
/ וְחָלָב לִשְׁתּוֹת לִצְמָאוֹ / וְהֶעָשִׁיר יוֹם אֶחָד נִפְעָם /
וּשְׂפָתָיו מָלְאוּ זַעַם / וְהִתְקַצֵּף וְקִלֵּל עַמּוֹ / וּבֵיתוֹ
וְאַנְשֵׁי שְׁלוֹמוֹ / וְהַנָּחָשׁ בָּא כְּחֻקּוֹ / לִמְצֹא בַּבַּיִת
סִפְקוֹ / וְרַגְלוֹ בִּפְנֵי הֶעָשִׁיר תַּחַשׁ / וַיַךְ בְּזַעְפּוֹ אֶת
הַנָּחָשׁ / כִּי לֹא נִסְתָּר מֵחֲמָתוֹ / וַיָּשָׁב הַנָּחָשׁ אֶל
בֵּיתוֹ / נָגוּעַ וּמֻכֶּה וּמְעֻנֶּה / לְחֶלְקוֹ שִׁבְרוֹן מִשְׁנֶה /
וְלֹא יָסַף עוֹד לְהֵרָאֹה / בִּגְבוּל הֶעָשִׁיר וּמְבוֹאוֹ / וַיְהִי
מִקֵּץ יָמִים בָּאִים / וַיֵּרֶד הֶעָשִׁיר פְּלָאִים / נָסַע עָשְׁרוֹ
בָּא לוֹ רִישׁוֹ / וּמֵת מִקְנֵהוּ וּרְכוּשׁוֹ / וְדַאֲגָתוֹ הָלֹךְ
וְקָשָׁה / תַּחַת שְׂעֹרָה בָּאְשָׁה / וַיָּסֹלּוּ אָרְחוֹת אֵידוֹ /
וַתִּקְצַר מֵהוֹשִׁיעַ יָדוֹ / וַיֹּאמַר: בְּפִשְׁעִי זֶה כִּי הִכֵּיתִי
/ אֶתְמוֹל הַנָּחָשׁ בְּבֵיתִי / וּפְצַעְתִּיו בְּרֹב חֵמָה / עַל לֹא
חָמָס עָשָׂה וּמִרְמָה / אֵלֵךְ לְשַׁחֵר פָּנָיו בְּאָהֳלוֹ / אוּלַי
יְפֻתֶּה וְאוּכְלָה לוֹ / בִּדְבַר פִּתּוּי וְחֵלֶק לָשׁוֹן / לָשׁוּב
כַּמִּשְׁפָּט הָרִאשׁוֹן / וַיָּבֹא עָדָיו וַיְקוֹנֵן / וַיֵּבְךְּ
לְפָנָיו וַיִּתְחַנֵּן / לְהַעֲבִיר מֵעָלָיו שִׂנְאָתוֹ / וְלֹא יָעִיר
כָּל חֲמָתוֹ / וְכֹה אָמַר הָאִישׁ בְּדֶמַע / נָחָשׁ, לַעֲצָתִי תִשְׁמַע
/ יַעַן הִתְרַגֶּזְךָ אֵלָי / שׁוּב וְכָל מַחֲסוֹרְךָ עָלַי / יָדַעְתָּ
כִּי אֲהַבְתִּיךָ / וַאֲשֶׁר בְּבֵיתִי קֵרַבְתִּיךָ / וְחַסְדִּי עָלֶיךָ
גָּבָר / לֹא חָסַרְתָּ כָּל דָּבָר / וְאִם בְּקִצְפִּי הִכִּיתִיךָ /
הִנֵּה בִּרְצוֹנִי רִחַמְתִּיךָ / שָׂא נָא לְפֶשַׁע מִן הַפֶּצַע /
וְאֶהְיֶה לְךָ עֶבֶד נִרְצָע / וַיַּעַן הַנָּחָשׁ לְמוּלוֹ / לֹא אָשִׁיב
לָאִישׁ כְּפָעֳלוֹ / אִם חָזָק עָלַי נִגְעִי / כִּי אִישׁ הָרַגְתִּי
לְפִצְעִי / לֹא יִתְמְהוּ עָלַי הַשּׁוֹמְעִים / כִּי כָּל בְּנֵי בֵיתְךָ
יוֹדְעִים / אֲשֶׁר לֹא בִעֲתָתְךָ אֵימָתִי / וּמַה תּוֹסִיף לְפַתּוֹתִי
/ לֹא אָשׁוּב עוֹד לְבֵיתֶךָ / וְלֹא אֶתְאַו לְמַטְעַמּוֹתֶיךָ /
בַּאֲשֶׁר רֹאשִׁי מָחַצְתָּ / וְגֶדֶר שָׁלוֹם פָּרַצְתָּ / לְהָנִיף
עָלַי חַרְבֶּךָ / וְאִם תִּשְׂחַק הַיּוֹם מָחָר אֶעֶצְבֶךָּ / וְלֹא
תִשְׂמַח אִם מוֹתִי תֶחֱזֶה / כִּי הֵבֵאתָ עָלַי אֶת הַמָּוֶת הַזֶּה /
וְלֹא אֶת עֲווֹנְךָ אֶפְקֹד / וְלֹא עַל רָעָתְךָ אֶשְׁקֹד / אַךְ בְּכָל
עֵת שֶׁאֲנִי זוֹכֵר וְרֹאֶה / פִּצְעִי גָדוֹל וְלִבִּי נִכְאֶה / וְאֵשׁ
קָדְחָה בְאַפִּי / סוּר פֶּן אֲמוֹתֶתְךָ בְּזַעְפִּי / כִּי עָצְמָה לִי
הַמַּחֲלָה / וְתַרְבֶּה שִׁבְרִי עַל נְקַלָּה / וְאִם לֹא, אֶפְקֹד
עֲווֹנְךָ בְחֵמָה / לִבְּךָ יֶהְגֶּה אֵימָה / מִבְּנִי הַגָּדוֹל אֶפַע /
כִּי מִשֹּׁרֶשׁ נָחָשׁ יֵצֵא צֶפַע / וּפִרְיוֹ שָׂרָף מְעֹפֵף / אֲשֶׁר
לְנַקְּמֵנִי עָלֶיךָ חֹפֵף / וּמִגַּעֲרָתוֹ פֶתַע תִּתְמַהּ
וְתִתְרוֹפֵף.



וְהַמָּשָׁל בְּאִישׁ יָזִיד עַל עֲמִיתוֹ / לַעֲשׂוֹת לוֹ
רָעָה בְּשִׂנְאָתוֹ / וְזֶה לֹא אֹיֵב לוֹ וְלֹא מְבַקֵּשׁ רָעָתוֹ /
יֹשֵׁב לָבֶטַח אִתּוֹ / וְלֹא הֵרַע לוֹ וּלְמִשְׁפַּחְתּוֹ / וּמִפְּנֵי
הָרָעָה אֲשֶׁר לוֹ הֵרַע / זֶרַע שִׂנְאָה בְלִבּוֹ זָרַע / אֲשֶׁר
יִקְצֹר כִּרְאוֹת מְקוֹמוֹ / כִּי יִרְאֶה כִּי יָבֹא יוֹמוֹ / שְׁתַּיִם
כְּאַחַת זֶה גְמוּלוֹ / וְאִם זֶה לֹא יְשַׁלֵּם לוֹ / אֶחָד מִשְּׁאֵר
בְּשָׂרוֹ / הַיּוֹדֵעַ אֶת יִצְרוֹ / לִנְקֹם נָקָם לֹא יִתְיָאֵשׁ /
בְּזַעַף אַף וְלַהַב אֵשׁ / עַל כֵּן עֹשֶׂה רָע לֹא יִתְמַהּ / אִם
יִמָּלֵא עָלָיו חֵמָה / הַמְּקַבֵּל רָעוֹת מִיָּדוֹ / יָדָיו
תְּקָרֵבְנָה אֵידוֹ / וּבְכָל דְּרָכָיו יְהִי לוֹ לְצוֹרֵר / יִשְׂמַח
בְּפִידוֹ וְיִתְעוֹרֵר / וְאִישׁ מֵרְאוֹת עֵינָיו טָח / אִם בַּאֲשֶׁר
הֵרַע לוֹ יִבְטָח / וְאָמַר הֶחָכָם: מִי שֶׁאֵינוֹ מוֹשֵׁל בַּחֲמָתוֹ /
אֵיךְ יִמְשֹׁל בְּאִישׁ זוּלָתוֹ / וְלִבִּי אַחֲרֵי הָאוֹמֵר נָטָה /
זֹרֵעַ שִׂנְאָה יִקְצֹר חֲרָטָה.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר / לְשָׂמִים אוֹר לְחֹשֶׁךְ וּמָתוֹק לְמָר:



כָּתַב בֶּרֶכְיָה בַחֶרֶט / זֶרַע שִׂנְאָה יִקְצֹר חֶרֶט

וּבְנַפְשׁוֹ הוּא נֹתֵן שֶׂרֶט / וַיָּצַר אֹתוֹ בַּחֶרֶט

פֹּצֵעַ רֵעוֹ אוֹ מֹרֵט / חֶרֶב מִלְחָמָה לוֹ מֹרֵט

אִם דֶּרֶךְ הָאֹיֵב יָרֵט / יִלְקֹט מִן אֵיבָתוֹ פֶּרֶט.
