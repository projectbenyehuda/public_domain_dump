


	
	משל פב: רָחֵל וְעֵז וְרֹעֶה
	




תִּשְׁמַע וְתַקְשֵׁב / הַמְקָרֶבְךָ קְרוֹבְךָ יֵחָשֵׁב



רָחֵל הָיְתָה תוֹךְ הָעֵדֶר / וְעֵז רָחוֹק מִן הַגָּדֵר
/ רֹעֶה לֵדַת רָחֵל מְיַחֵל / וַיְהִי כַאֲשֶׁר יָלְדָה רָחֵל / לָקַח
הַשֶּׂה וַיְבִיאֵהוּ / אֶל הָעֵז וַתְּנִיקֵהוּ / עַד אֲשֶׁר גָּדַל
וַיִּגָּמֵל / וְטִפְּחַתּוּ בְרֹב עָמָל / אָז אָמְרָה: רְאֵה
אֹרְחוֹתֶיךָ / שׁוּב אֶל אֶרֶץ אֲבֹתֶיךָ / וּלְמוֹלַדְתְּךָ וַעֲשֵׂה
חָיִל / אִמְּךָ רָחֵל אָבִיךָ אָיִל / אִם אֲנִי עַד כֹּה גִדַּלְתִּיךָ /
רִבִּיתִיךָ טִפַּחְתִּיךָ / וּבְשׁוּבְךָ אֶל עַמֶּךָ / יִשְׂמַח אָבִיךָ
וְאִמֶּךָ / הַשֶּׂה עָנָה: אַךְ אַתְּ אִמִּי / בַּאֲשֶׁר הֵיטַבְתְּ
חַסְדֵּךְ עִמִּי / לֹא אֶקְרָא אִמִּי הָרְחֵלָה / כִּי רִחֲקָה אֹתִי
מֵעַל גְּבוּלָהּ / וְלֹא יָסְפָה עוֹד לִרְאוֹתִי / וְאֵלֶיהָ לֹא
תְשׁוּקָתִי.



וְהַמָּשָׁל שְׁמַע וְהַקְשֵׁב / הַמְקָרֶבְךָ קְרוֹבְךָ
יֵחָשֵׁב / וּמַרְחִיקְךָ אַף אִם הוּא קָרוֹב / חֲשֹׁב אֹתוֹ בָּטֵל
בְּרוֹב / מוּזָר לְאֶחָיו וְנָכְרִי / עֹכֵר שְׁאֵרוֹ אַכְזָרִי / אֹמֵר,
כִּי אֵין אֹהֲבִי / אִם לֹא מֵאִמִּי וְאָבִי / הַמְכַבְּדִי עִם יֹשְׁבֵי
חָלֶד / וְאָח לְצָרָה יִוָּלֵד.
