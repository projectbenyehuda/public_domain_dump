


	
	משל לט: זַרְזִיר, נֶשֶׁר וְעוֹפוֹת
	




יְקַוּוּ הַקְּטַנִּים לִשְׂרָרָה / כִּי טוֹבָה חָכְמָה מִגְּבוּרָה



זַרְזִיר מָלַךְ בְּחָכְמָתוֹ / וְלֹא מֵרֹב גְּבוּרָתוֹ /
וְאִם עֲלֵיהֶם הוּרַם וְהוּנָף / קָטָן הוּא מִכָּל בַּעֲלֵי כָנָף /
וְכַאֲשֶׁר בַּתְּחִלָּה נִקְבָּצוּ / וְהָעוֹפוֹת בֵּינֵיהֶם נוֹעָצוּ /
לָשׂוּם מֶלֶךְ עֲלֵיהֶם / אֲשֶׁר יָרוּם לָעוּף בְּמַחֲנֵיהֶם / וַיֹּאמֶר
הַנֶּשֶׁר: מִי מִן הַמְעוֹפְפִים כָּמוֹנִי / וּמִי יָחוּשׁ חוּץ
מִמֶּנִּי / וְכִשְׁמֹעַ זֶה הַזַּרְזִיר זָמַם בְּמַחֲשָׁבוֹ / כִּי
בִּכְנַף הַנֶּשֶׁר אֵיתָן מוֹשָׁבוֹ / בַּאֲשֶׁר יָעוּף שָׁם תִּהְיֶה
תַחֲנוֹתוֹ / יִשָּׂאֵהוּ עַל אֶבְרָתוֹ / וַיִּתְחַבֵּא בְּאֶבְרַת
הַנֶּשֶׁר כַּאֲשֶׁר זָמָם / וְכַאֲשֶׁר יָעוּפוּ בַעֲלֵי כָנָף הַנֶּשֶׁר
עֲלֵיהֶם יִתְרוֹמָם / עַל כָּל עוֹף אֲשֶׁר יָעוּף פַּעֲמָיִם / דֶּרֶךְ
הַנֶּשֶׁר בַּשָּׁמַיִם / וַיֹּאמַר: עַל כָּל הָעוֹפוֹת מָלַכְתִּי / כִּי
הִתְגַּדַּלְתִּי וְהִתְרוֹמַמְתִּי / עוֹד זֶה מְדַבֵּר וְזֶה בָּא /
הַזַּרְזִיר אֲשֶׁר תַּחַת כְּנָפָיו נֶחְבָּא / אֲשֶׁר עַד עַתָּה נָח
עֲלֵי גַבּוֹ / עַל כֵּן טַעֲמוֹ עָמַד בּוֹ / כִּי בִרְאוֹתוֹ הַנֶּשֶׁר
יָגֵעַ וּרְפֵה יָדַיִם / וְהַזַּרְזִיר הִתְחַזֵּק וַיֶּאֱזֹר מָתְנַיִם /
לָעוּף מֵעַל הַנֶּשֶׁר מַעְלָה מַעְלָה / וַיִּתְנוּ לוֹ הוֹד מַלְכוּת
וּמֶמְשָׁלָה.



וְהַמָּשָׁל בְּאִישׁ מָלֵא כֶסֶף וְזָהָב / תַּחְתָּיו
שָׁחֲחוּ עֹזְרֵי רָהַב / יִתְפָּאֵר בְּחַדְרֵי מַשְׂכִּיתוֹ / וְאַל
יִתְהַלֵּל הַגִּבּוֹר בִּגְבוּרָתוֹ / לְהַבְזוֹת בְּעֵינָיו דַּל וְקָטָן
/ כִּי עֵת יֵהָפֵךְ לִהְיוֹת לְשָׂטָן / מִמּוֹעֲצוֹתָיו יַפִּילֶנּוּ /
וְדַל מֵבִין יַחְקְרֶנּוּ / וְאִמְרַת הֶחָכָם לַחְכִּים יְשָׁרָה /
טוֹבָה חָכְמָה מִגְּבוּרָה.
