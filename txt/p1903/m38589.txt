


	
	משל קיט: הַקַּנַּאי וְהַחֶמְדָּן
	







הַחֶמְדָּן וְהַמְקַנֵּא יַחְדָּו יִלְבְּשׁוּ בֹשֶׁת / וְאַחֲרֵיהֶם נֶזֶק עַל כָּל אֶחָד יָשִׁית



שִׁמְעוּ נָא דִּבְרֵי קְרִשְְׂפִּיָא הַנַּקְדָּן /
אֲשֶׁר שָׁפַט וָדָן / לֵב קַנַּאי וְחֶמְדָּן / לְחֶרֶב הֶרֶג וְאַבְדָן /
וּלְמַשְׁחִית אֵיבַת עוֹלָם / וּלְהָשִׁיב דַּךְ וְנִכְלָם / אֲשֶׁר
אַפּוֹ חָרָה / עַל הַמִּשְׁפָּט וְעַל הַגְּבוּרָה / וְעַל דֶּרֶךְ
הַגְּזֵרָה / לְעַוֵּת מִשְׁקָל וּמְשׂוּרָה / כַּאֲשֶׁר לָאָדָם הִקְרָה /
וּלְאַנְשֵׁי זָדוֹן וְעֶבְרָה / אֶחָד חֶמְדָּן וְאֶחָד מְקַנֵּא / אִישׁ
אָחִיו יְשַׂנֵּא / וּמְדַבְּרִים עַל יוֹצְרָם / הַיֹּדֵעַ רֹעַ יִצְרָם /
הַרְבֵּה מִתְלֹנְנִים / וּבְרָעָה מִתְאֹנְנִים / חֶמְדָּן כֹּה יֹאמַר /
רְאֵה כִּי רַע וָמַר / כָּל מַעֲשֵׂה הָאֱלֹהִים / הִשְׁפִּיל אֶת
הַגְּבֹהִים / מַדּוּעַ אֲנִי רָשׁ וְעָנִי / וְיַעֲשִׁיר אִישׁ פְּלוֹנִי
/ אֹיְבִי שְׁכֵנִי / הַיֹּשֵׁב לִימִינִי / וְחֹנֶה עַל דִּגְלוֹ / לָמָּה
יָרִים אֶת רַגְלוֹ / עַל סוּס וּמֶרְכָּבָה / וְנַפְשִׁי תְאֵבָה /
וְחוֹמֶדֶת לוֹ / כִּי מַחֲמַדִּים כֻּלּוֹ / וְלִבִּי אֵלָיו יִפְנֶה /
לִהְיוֹת בִּמְקוֹמוֹ חֹנֶה / וַיַּעַן הַמְקַנֵּא / בְּשִׂנְאָה אֲשֶׁר
שְׂנֵא / לֹא יִפְנֶה הַשֵּׁם לְךָ / וְלֹא יִשְׁמַע לְקוֹלְךָ /
לִהְיוֹתְךָ קְצִין עָם / אָמוּתָה הַפָּעַם / אִם תֶּעְשַׁר / וְאִם
תִּכְשַׁר / לִהְיוֹת כָּמוֹנִי / לָבוּשׁ שָׁנִי / אֹמַר לֵאלֹהִים אַל
תַּרְשִׁיעֵנִי / כִּי מִי יָחוּשׁ חוּץ מִמֶּנִּי / וְאוֹי מִי יִחְיֶה /
לֹא הוֹאַלְתִּי וְלֹא יִהְיֶה / לְכָל בְּנֵי אָדָם / דַּיֶּךָ מִחְיַת
בָּשָׂר וָדָם / וְאֶתְהַלֵּךְ לְבַדִּי / כָּל יְמֵי חֶלְדִּי /
בְּעָשְׁרִי וּבִכְבוֹדִי / וְכִי כַבִּיר מָצְאָה יָדִי / וּבְעֵת אֲשֶׁר
אֲנִי רָשׁ וְעָנִי / יִדַּל שְׁכֵנִי יוֹתֵר מִמֶּנִּי / וְכָל עֹבְרֵי
עַל פָּנַי / יִהְיוּ כַחֲגָבִים בְּעֵינַי / וְכֻלָּם יֵלְכוּ חֲשֵׁכִים /
וַיְהִי הֵמָּה הֹלְכִים / הָלֹךְ וְדַבֵּר / וְחָרָה אַף הַמְדַבֵּר /
הוּא יַעֲשֶׂה הַשֵּׁם / הוּא יָדַע וְאָשֵׁם / וַיִּמְצָאֵם מַלְאַךְ
הַשֵּׁם / בְּמִדְבַּר לֶשֶׁם / וַיְהִי כַּאֲשֶׁר רָאָם / הַמַּלְאָךְ
קְרָאָם / וַיֹּאמַר: מָה רְאִיתֶם / וּמַה בִּקַשְׁתֶּם / וְעֹבְרֵי
דֶרֶךְ מַה שְּׁאַלְתֶּם / אֶת יוֹצֶרְכֶם הַיְדַעְתֶּם / וַיֹּאמְרוּ:
יָדַעְנוּ / וְלֹא נַגִּיד מִמֶּנּוּ / כִּי זָר מַעֲשֵׂהוּ / וְאֵין מִי
יְכִילֵהוּ / בַּאֲשֶׁר נָתַן לִשְׁכֵנִי / כָּל מַחְמַד עֵינִי /
וּמִמֶּנִּי כֹל הֶחְסִיר / וְחַסְדּוֹ מֵאִתִּי הֵסִיר / עַד אֲשֶׁר
נַפְשִׁי מְעֻנָּה / מֵרֹב קִנְאָה וְתֹאֲנָה / אֲשֶׁר אֵלַי לֹא תְאֻנֶּה
/ וְזֶה הָיָה מַעֲנֶה / הַחֶמְדָּן בְּאִוַּלְתּוֹ שֹׁנֶה / וְסֵמֶל
הַקִּנְאָה הַמַּקְנֶה / לֹא יִהְיֶה וְלֹא יָבֹא / כֹּל אֲשֶׁר חֶמְדָּה
בוֹ / כִּי אִם יָרִיחַ בְּצֵידִי / עָשִׁיר כְּנֶגְדִּי / טוֹב מֵחַיַּי
מוֹתִי / וּרְקַב עֲצָמוֹת בְּקִנְאָתִי / וְאִם תִּשְׁמַע לִתְפִלָּה /
וְאִם תַּעֲלֶה לְמַעְלָה / לֵאלֹהִים אָשִׂים תִּפְלָה / וְתַחַת
תְּהִלָּה תָהֳלָה / וַיָּרִיבוּ שְׁנֵיהֶם / וְהֵם מְרִיקִים שַׂקֵּיהֶם /
וַיֹּאמֶר הַמַּלְאָךְ: הַחֲרִישׁוּ / לְמִפְלָט לָכֶם הָחִישׁוּ / הֲלֹא
תֵבוֹשׁוּ מִדַּבֵּר אֶל יוֹצֶרְכֶם / לָמָּה תְרִיבוּן אֵלַי כֻּלְּכֶם /
הִנֵּה שֻׁלַּחְתִּי לִפְנֵיכֶם / לָתֵת הַיּוֹם לָכֶם / לְכָל אֶחָד
שְׁאֵלָתוֹ / לַעֲשׂוֹת בַּקָּשָׁתוֹ / נֶגֶד אִמְרֵיכֶם דַּי / יִתֵּן
לָכֶם גְּמוּלְכֶם שַׁדַּי / וְלֹא תִשְּׂאוּ עָלַי נֶהִי / כִּי הוּא
אָמַר וַיֶּהִי / יָבֹא לָכֶם פִּתְאוֹם / וְיֶאֱמַץ מִלְּאוֹם וּלְאוֹם /
וְנָתַתִּי זֹאת לָכֶם / יְהִי לְאֶחָד מִשְּׁנֵיכֶם / כָּל אֲשֶׁר יִפְצֶה
פִיהוּ / בְּרֶגַע אֶחָד יִקְרָאֵהוּ / וּפִי שְׁנַיִם בְּדַבְּרוֹ /
יִהְיֶה בְּשֶׁל חֲבֵרוֹ / מִפְּנֵי שֶׁהוּא לְאַחֲרִית / וְזֹאת לָכֶם
הַבְּרִית / אֲשֶׁר לֹא תָפִירוּ / עִבְרוּ עֲבֹרוּ / וּלְכוּ לְדַרְכְּכֶם
/ כִּי מֵהֶם יָבֹא אֲלֵיכֶם / כֹּל אֲשֶׁר תִּבְחָרוּ / וַיִּשְׁמְעוּ לוֹ
וַיֹּאמְרוּ / עֲבָדֶיךָ הִנֵּנוּ / וְאַתָּה תִהְיֶה אֲדֹנֵינוּ / יְהִי
חַסְדְּךָ עָלֵינוּ / כִּי נִחַמְתָּנוּ / וַיְהִי כְדַבְּרָם דִּבְרֵיהֶם
/ וְהַמַּלְאָךְ הָלַךְ מֵעִמָּהֶם / כִּי הָרוּחַ נְשַָׂאַתְהוּ / וְעַיִן
לֹא רָאָתְהוּ / וְעִקְּבוֹתָיו לֹא נוֹדְעוּ / אָז הֵבִינוּ וַיֵּדְעוּ /
כִּי מַלְאַךְ הַשֵּׁם הוּא / וְתוֹרַת אֱמֶת בְּפִיהוּ / וַיַּעַן
הַחֶמְדָּן אֲשֶׁר חָמַד כִּפְלַיִם / וְטֻמְאָתוֹ בְשׁוּלַיִם /
וַיֹּאמַר: שְׁאַל נָא / וַיַּעַן לוֹ הַמְקַנֵּא / אֵיךְ אֶשְׁאַל דָּבָר
/ וְאַתָּה עָלַי תִּגְבָּר / לְהַכְפִּיל הַמִּשְׂרָה / לָקַחַת חֵלֶק
בְּכוֹרָה / וְאַתָּה הָאַחֲרוֹן / וְיִהְיֶה לְךָ הַיִּתְרוֹן / וְעָלַי
הַחֶסְרוֹן / וַאֲנִי בָאתִי רִאשׁוֹן / וְאֵיךְ אֲחַלֶּה לְטוֹב לְהֵיטִיב
גּוֹרָלֶךָ / וּלְהַרְחִיב גְּבוּלֶךָ / הֲזֶה מִשְׁפָּט פְּלִילִי /
וַיֹּאמַר: הַמְקַנֵּא אַתָּה לִי / הַעְמֵק שְׁאָלָה / אוֹ הַגְּבֵּהַּ
לְמָעְלָה / עַד אֲשֶׁר לֹא בּוֹ תֶחְסָר / וְלֹא תִהְיֶה מֵעִיר מוּסָר /
וְלָמָּה תְקַנֵּא קִנְאָה / וְנַפְשְׁךָ מִכָּל טוּב מְלֵאָה /
תִּקְוָתְךָ בָּאָה וְשַׂמְתָּ בַטֶּנֶא / וַיַּעַן לוֹ הַמְקַנֵּא / אִם
דִּבַּרְתָּ נְכוֹנָה / שְׁאַל נָא רִאשׁוֹנָה / מִטַּעַם דְּבָרֶיךָ /
וַאֲנִי אָבֹא אַחֲרֶיךָ / וּבְכִפְלַיִם לֹא תַּחְמֹד / וְאָנֹכִי לֹא
אֶעֱמֹד / וְלֹא אוּכַל לָשֵׂאת רֹאשִׁי / וַתִּבְחַר מַחֲנַק נַפְשִׁי /
אִם תּוֹסִיף עָלַי מָנָה / וְאִם תִּהְיֶה בָאַחֲרוֹנָה / וַיִּקְצֹף
הַחֶמְדָּן בְּלִבּוֹ / וַחֲמָתוֹ בָעֲרָה בוֹ / וַיִּפֶן אֵלָיו בְּחֵמָה
/ וַיַּכֵּהוּ בְיָד רָמָה / בְּאֶבֶן וּבְאֶגְרֹף / וּבְמַקֵּל יָד
עָרְפּוֹ יַעֲרֹף / וַיֹּאמַר: עַל פָּנֶיךָ תִשְׁאַל / וַאֲנִי אַחֲרֶיךָ
אֶגְאַל / וְאִם לֹא, אַשְׁחִיתֶךָ / וּמַהֵר אֲמִיתֶךָ / וַיְהִי
כִרְאוֹתוֹ / כִּי אָבְדָה תִקְוָתוֹ / וְעוֹדֶנּוּ מַחֲזִיק בְּטֻמְאָתוֹ
/ וּלְהַקְדִּים תְּפִלָּתוֹ / עוֹד לֹא יְקַו / וַיַּעַשׂ לוֹ מִשְׂפָּח
לְקָו / וּצְעָקָה לְמִשְׁקֹלֶת / בֶּקַע לַגֻּלְגֹּלֶת / הֵחֵל רֹאשׁוֹ
לִבְצֹעַ / הַכֵּה וּפָצֹעַ / כַּאֲשֶׁר גָּבַר חִילוֹ / וְכַאֲשֶׁר צַר
לוֹ / חָל אֶת פָּנָיו / וַיִּתְחַנֵּן לְפָנָיו / וַיֹּאמַר: אֲדֹנִי /
הֶרֶף מִמֶּנִּי / וְאֶשְׁאַל רִאשׁוֹנָה / וְכָפַלְתָּ בָאַחֲרוֹנָה /
לָשִׂים שָׁלוֹם בֵּינֵינוּ / וַיֶּרֶף מִמֶּנּוּ / אַךְ לֹא
יְחַפְּשֶׂנּוּ / וּבִכְנָפוֹ יֹאחֲזֶנּוּ / וּפֶן יְבַקֵּשׁ תֹּאֲנָה
לִבְרֹחַ אָנֶה וְאָנָה / וַיַּעַן הַמְקַנֵּא וַיֹּאמַר: אָנָּא / הַשֵּׁם
תִּגְמֹל לְעַבְדֶּךָ / הִפּוּךְ מֵחֲסָדֶיךָ / לְשַׁלֵּם לוֹ / אֶת
גְּמוּלוֹ / עַוֵּר לִי עַיִן אַחַת מֵהַשְּׁנַיִם / וּלְאֹיְבִי פִּי
שְׁנַיִם / וְהַפֵּל אַחַת מֵהַיָּדַיִם / וּלְשֹׂנְאִי זֶה תֵּן
כִּפְלַיִם / וַיְהִי כְדַבְּרוֹ כָּכָה / וְהִנֵּה אֵימָה חֲשֵׁכָה /
נָפְלָה עֲלֵיהֶם / וַתַּךְ אֶת שְׁנֵיהֶם / בְּעִוָּרוֹן / וַיְהִי
לָאֲחֲרוֹן / מִכֹּבֶד רֵעֵהוּ הַיִּתְרוֹן / לִהְיוֹת לוֹ הַכִּפְלַיִם
מֵחֲבֵרוֹ / וְלֹא נָפַל מִכָּל דְּבָרוֹ / וַיִּפֶן לְרֵעֵהוּ פָנָיו /
וְהִנֵּה חָשְׁכוּ שְׁתֵּי עֵינָיו / וּבְכַנְפֵי בְגָדָיו / נִשְׁאֲרוּ
שְׁתֵּי יָדָיו / וַיָּסַר כֹּחוֹ מֵעָלָיו / וְהִנֵּה קִטֵּעַ מִשְּׁתֵי
רַגְלָיו / וַיִּשָּׁאֲרוּ שְׁנֵיהֶם שָׁמָּה / לְבֹשֶׁת וְלִכְלִמָּה /
וַיָּסַר מֵהֶם אֶת חֶמְדָּתָם / וְגַם אֶת שִׂנְאָתָם / כִּי הַחֹמֵד נָח
מֵחֶמְדָּתוֹ / בָּחַר מִחְיָתוֹ / וּבְהַוָּתוֹ לֹא יָעוֹד / וְלֹא
יַחְמֹד עוֹד / הוֹן בֵּית גָּבֶר / כִּי אִם בֵּית הַקָּבֶר / כִּי עוֹד
לֹא יָכֹל / לָשִׂים בְּפִיו לֶאֱכֹל / כִּי אִם לִלְחֹךְ בְּמָרָק /
כִּלְחֹךְ הַשּׁוֹר אֶת יָרָק / כִּי בְחֶסְרוֹן הָעֵינַיִם / יְהִי לוֹ
נִקְיוֹן שִׁנַּיִם / לֹא יִרְאֶה טוּב טַעְמוֹ / וְלֹא יָמִישׁ
מִמְּקוֹמוֹ / מַאֲכָלוֹ וּמוֹצָאוֹ יַחַד / יִהְיוּ לְמָקוֹם אֶחָד / כִּי
נֶחְשַׁב כַּבְּהֵמָה / טוֹב לוֹ מָוֶת כַּמְּהוּמָה / וְגַם הַמְקַנֵּא /
אֶת אָחִיו שֹׂנֵא / לֹא יְקַנֵּא עוֹד בַּאֲחֵרִים / כִּי יוֹצֵא
בְרָאשֵׁי אֵבָרִים / אֲשֶׁר אָבַד בְּעַזּוּתוֹ / וַאֲשֶׁר הִפְסִיד
בְּקִנְאָתוֹ / וְיֻכַּת עַד הַחָרְמָה / וְעַל זֶה מָשְׁלָה חָכְמָה /
וְדִבְּרָה עַל הַמְקַנֵּא / בִּשְׂפָתָיו יִנָּכֵר שׂוֹנֵא / וּבְקִרְבּוֹ
יָשִׁית מִרְמָה / וְיִשְׁפֹּךְ עָלָיו חֵמָה / אַךְ הַמִּשְׁפָּט
כַמִּדָּה / וְאָז הָיָה אִישׁ הַחֶמְדָּה / הַחֶמְדָּה סוֹף עֲשֶׂרֶת
הַדְּבָרִים / וְהַקִּנְאָה סוֹף שִׁיר הַשִּׁירִים / שֶׁהִיא קָשָׁה
כִשְׁאוֹל / וְאֵין לָהּ גּוֹאֵל לִגְאוֹל / וְאֵין בָּהּ טוֹב מְאוּמָה /
כִּי אִם בְּקִנְאַת הַחָכְמָה / לְקַנֵּא בְמַעֲשֵׂה הַיְרֵאִים / וְלֹא
בְמַעֲשֵׂה הַחַטָּאִים / וְזוּלַת אִוֶּלֶת טוֹבָה אֵינָה / הַחֶמְדָּה
יוֹצְאָה מִמֶּנָּה / כִּי הַקִּנְאָה רִאשׁוֹנָה / וְאַחֲרִיתָהּ מָרָה
כַלַּעֲנָה / וּמָרָה תִהְיֶה הַחֶמְדָּה שֶׁבָּאָה / כִּי אִם הַמַּעֲשֶׂה
מְבִיאָהּ / וְהַכֹּל דָוָה אַחַת / וְהַצָּרַעַת פֹּרַחַת / רַבָּה
וְהֹלֶכֶת / עַד אֲשֶׁר מוֹלֶכֶת / וְלֵב הָאָדָם מוֹשֶׁכֶת / אֶל מִבֵּית
לַפָּרֹכֶת / כִּי בַסּוֹף תַּשְׁלִיכֶנּוּ / עַל פְּנֵי הַשָּׂדֶה
תַגְעִילֶנּוּ / בְּקַנְּאוֹ מָמוֹן אֲחֵרִים / יִכְרוּ עָלָיו רֵעִים /
וְזֶה מִשְׁפַּט הַקַּנַּאי / הֲלֹא מְשַׂנְאַי / אֲשֶׁר הַשֵּׁם
יִשְׂנָאֵהוּ / הַנַּקְדָן קְרִשְׂפִּיָּהוּ / מְבַקֵּשׁ מֵאֵת יוֹצְרֵהוּ
/ אֲשֶׁר מִקִּנְאָה יִשְׁמְרֵהוּ / וּמֵחֶמְדָּה יַצִּילֵהוּ / וְזֶה
הַכֹּתֵב מַעֲשֵׂהוּ / שֶׁל קְרִשְׂפִּיָּהוּ / הַשֵּׁם יְסִירֵהוּ /
מֵחֶמְדָּה וּמִקִּנְאָה / מִשְּׁאוֹל וּמִשִּׂנְאָה / וְכָל דָּבָר רָע /
אֵלָיו לֹא יֶאֱרַע / וּמֵחֲמֹד הוֹן אֲחֵרִים / בִּבְנֵי הַנְּעוּרִים /
וְגַם בִּימֵי הַזְּקֵנִים / מִלְּקַנֵּא בַשְּׁכֵנִים / בְּמָמוֹנָם
וּבְקִנְיָנָם / רַק בְּחָכְמָתָם וּבְעִנְיָנָם / וּבְמַעֲשֵׂיהֶם
הַטּוֹבִים / אֲשֶׁר לָעַד נִצָּבִים / לִשְׁמֹר לַעֲשׂוֹת כָּהֶם /
כְּמוֹהֶם יִהְיוּ עֹשֵׂיהֶם / וְזֹאת הַקִּנְאָה / מִכָּל טוּב מְלֵאָה /
וְכֵן דִּבְּרוּ הַנְּבִיאִים / אַל יְקַנֵּא לִבְּךָ בַּחַטָּאִים / כִּי
אִם בְּיִרְאַת אֲדֹנָי / וּבְדִבְרֵי הַר סִינַי / וּבְקִנְאַת הַנְּקָמָה
/ נֹתֵן שָׁלוֹם בִּמְרוֹמָה / וּבְשֶׂה תְמִימָה / בִּבְרִית עוֹלָם
חֲתוּמָה / הַצּוּר אֲשֶׁר שְׁמוֹ לְעֵילוֹם / הוּא יְבָרֵךְ אֶת עַמּוֹ
יִשְׂרָאֵל בַּשָּׁלוֹם.



אָמֵן



וְשָׁלְמוּ מְשָׁלִים, תְּהִלָּה לְמֹשֵׁל, / לְבֶרֶכְיָהוּ מְשָׁלָיו מְמַשֵּׁל

תְּבוּנוֹת מְאַסֵּף כְּזָהָב וְכֶסֶף / פְּתַיּוּת יְשַׁסֵּף וְאֹהְבֵם יְנַשֵּׁל

מְשָׁלוֹ עֲנָוִים מְעוֹדֵד וְגֵאִים / מְנוֹדֵד וְרָאשֵׁי גְאוֹנִים יְחַשֵּׁל

אֲמָרוֹת טְהוֹרוֹת יְלַקֵּט כְּאוֹרוֹת / וְנוֹתְנָם בְּסִירוֹת נְכוֹנִים לְבַשֵּׁל

כְּרָאשֵׁי בְשָׂמִים חֲשָׁבוּם חֲכָמִים / וְאִם הֵם רְתָמִים לְכָל לֵץ וְכֹשֵׁל.





סֻיְּמוּ

הַמְּשָׁלִים כֻּלָּם

שֶׁבַח

לְמֶלֶךְ עַל כָּל הָעוֹלָם.
