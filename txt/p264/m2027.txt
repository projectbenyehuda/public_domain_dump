


	
	[איה חכמינו ומוכיחנו]
	



1


אַיֵּה חֲכָמֵינוּ וּמוֹכִיחֵינוּ? / פַּסּוּ אֱמוּנֵינוּ וְגַם מוֹרֵינוּ

עִוְרִים נְמַשֵּׁשׁ בָּאֲפֵלָה מִבְּלִי / מוֹרֶה, וְהָיוּ עִוְרִים צוֹפֵינוּ

צוּר, מַה נְּיַחֵל אַחֲרֵי זֹאת, אוֹ לְמִי? / נִשְׁאַל לְרַצּוֹתָךְ וְתוֹשִׁיעֵנוּ

נַעְתִּיר וְתִתְרַצֶּה, וְתֵחָשֵׁב לְפָ- / נֶיךָ שְׂפָתֵנוּ כְּמוֹ פָרֵינוּ

5 אַנְשֵׁי אֱמוּנָה אָבָדוּ, לֹא שָׂרְדוּ / בָּאוּ בְכֹחַ מַעֲשִׂים אֵלֵינוּ

כָּל גּוֹדְרֵי גָדֵר וּפֶרֶץ צֻמְּתוּ / מֵרֹב עֲווֹנֵינוּ וְגַם מִרְיֵנוּ

הָיוּ כְּמוֹ חוֹמָה בְזַעַם, זוֹעֲכִים / הָאַף, וְחֵמָה עָצְרוּ בַעְדֵּנוּ

רֻבֵּי עֲווֹנֵינוּ, אֱמֶת, הֵם אִבְּדוּם, / גַּם נֶאְסְפוּ מֶנּוּ בְחַטֹּאתֵינוּ

חַלְנוּ וְחָלִינוּ – וְאֵין מָזוֹר, וְאֵין / רוֹפֵא לְחָלְיֵנוּ וְלִמְזוֹרֵנוּ

10 מִכַּף וְעַד קָדְקֹד חֳלִי, אֵין בּוֹ מְתֹם, / אַל יִקְרְכֶם מַכְאוֹב כְּמַכְאוֹבֵנוּ!

פֶּצַע וְחַבּוּרָה וּמַכָּה נַחֲלָה, / נוּדוּ לְחָלְיֵנוּ וּמַכּוֹתֵינוּ!

מַכָּה אֲנוּשָׁה, נוֹאֲשׁוּ הָרוֹפְאִים / מִתֵּת צְרִי מַרְפֵּא כְּפִי חָלְיֵנוּ

אֵיךְ תַּעֲלֶה לָנוּ אֲרוּכָה – אֵין צְרִי / גִלְעַד, וְרוֹפְאֵי הָאֱלִיל רוֹפְאֵינוּ

‘שָׁלוֹם’ וְאִין שָׁלוֹם, לְעוֹלָם יֹאמְרוּ / כִּי עַל נְקַלָּה רִפְּאוּ שִׁבְרֵנוּ

15 לוּ עַל צְחִיחֵי הַסְּלָעִים נֶהֱמֶה / נִבְכֶּה – אֲזַי דָּשְׁאוּ לְרֹב בִּכְיֵנוּ

אַל תִּתְמְהוּ אִם הַדְּמָעוֹת יֶאְדְּמוּ - / כִּי יִזְּלוּ בִּבְכִי דְמֵי לִבֵּנוּ

לִמְקוֹנְנוֹת קִרְאוּ וְתָבֹאנָה, וְלַחְ - / כָמוֹת – וְתִשֶּׁאנָה נְהִי עָלֵינוּ

מִסְפַּד הֲדַדְרִמּוֹן בְּבֵית מַלְכוּת לְמוּל / מִסְפֵּד אֲזַי סִפְדוּ לְזֶה, אַחֵינוּ

נֶהְפַּךְ לְקוֹל בּוֹכִים וְאֵל אֵבֶל נְעִים / הֶמְיַת נְבָלֵינוּ וְכִנּוֹרֵינוּ

20 תַּנִּים וֹבַת יַעְנָה וְהַכּוֹס יָדְעוּ / כִּי הֵם חֲבֵרֵינוּ וְאַלּוּפֵינוּ

לֹא אַב הֲמוֹן גּוֹיִים יְדָעָנוּ, וְלֹא / יַעְקֹב בְּנוֹ רָצָה לְהַכִּירֵנוּ

גַּם הַשְּׁבָטִים נוֹעֲדוּ יַחַד עֲשׂוֹת / מִסְפֵּד כְּעַל יָחִיד, לְגָלוּתֵנוּ

בָּכָה רְאוּבֵן עֵת רְאוֹת בָּנָיו בְּלִי / מוּסָר, וְשִׁמְעוֹן – מִשְּׁמוּעָתֵנוּ

וּנְעִים זְמִיר לֵוִי וְשִׁירוֹ נֶחֱשַׁב / קִינִים בְּמוֹ פּיו, לַאֲבֵלוּתֵנוּ

25 צִירִים כְּיוֹלֵדָה יְהוּדָה יֹאחֲזוּ / עֵת יֶחֱזוּ עֵינָיו לְתוֹלְדוֹתֵינוּ

אֶתְמַהּ לְיִשָּׂשכָר וּבָנָיו יוֹדְעִים / עִתִּים – וְאֵיךְ לֹא יָדְעוּ קִצֵּנוּ

בָּכָה זְבוּלוּן עַל זְבוּל אֵל, גַּם עֲלֵי / בָתֵּי זְבוּלֵינוּ וְחָרְבוֹתֵינוּ

לִבִּי לְבִנְיָמִין, וְדָן יַצְדִּיק לְדִין / צוּרוֹ, וְנָם: אֵל נָא רְפָא צִירֵינוּ

מַר לִי לְנַפְתָּלִי וְגָד, עֵת יִצְעֲקוּ: / אַיֵּה חֲמוּדֵינוּ וְאֵי בָנֵינוּ?

30 וַיַּעֲנֵם אָשֵׁר: אֲשׁוּרַי מָעֲדוּ, / וַיֵּבְךְּ יוֹסֵף מִמְּרוֹרוֹתֵינוּ

נָשָׁה מְנַשֶּׁה כָּל כְּבוֹד אָבִיו וְתִפְ - / אֶרֶת גְּדֻלָּתוֹ, לְעַנְיוּתֵנוּ

תַּחַת פְּאֵר אֵפֶר לְאֶפְרַיִם, וְשַׂק / יֻצַּע לְטַפֵּנוּ וְלִזְקֵנֵינוּ

שֶׁמַש וְיָרֵח וְגַלְגַּלֵּי זְבוּל / אָסְפוּ לְנָגְהָם, מֵאֲפֵלָתֵנוּ

כּוֹכָבִים וּמַזָּלוֹת יְנִיעוּן רֹאשׁ וְגַם / יִבְכּוּ וְיָנוּדוּ, לְנִידָתֵנוּ

35 טָלֶה – טְלָלִים יִזְּלוּ עֵינָיו, וְקוֹל / יִשָּׂא, וְשׁוֹר גָּעָה לְצַעְקָתֵנוּ

גַּם הַתְּאוֹמִים עַל יְדֵיהֶם סָפְקוּ, / כִּמְעַט וְנִפְרָדוּ לְפֵרוּדֵנוּ

סַרְטָן – דְּמָעָיו עַל לְחָיָיו יָרְדוּ, / אַרְיֵה בְמַר יִשְׁאַג לְשַׁאְגָּתֵנוּ

חָגְרָה בְּתוּלָה שַׂק עֲלֵי מָתְנָהּ, כְּעַל / בַּעַל נְעוּרֶיהָ, לְדַלּוּתֵנוּ

נִשְׁמַע לְמֹאזְנַיִם וְעַקְרָב קוֹל, בְּעֵת / נָהוּ נְהִי נִהְיָה לְקוֹל בִּכְיֵנוּ

40 קֶשֶׁת, גְּדִי עִם הַדְּלִי הִתְנַצְּלוּ / עֶדְיָם, וְצָרִים לָבְשׁוּ עֶדְיֵנוּ

דָּגִים בְּמֵימֵי הַדְּמָעוֹת יוֹם וְלֵיל / שׂוֹחִים וְנֶאְנָחִים לְאַנְחָתֵנוּ

גַּם מָאֲסוּ שָׁנִים חֳדָשִׁים, אָמְרוּ: / לֹא הֵם חֳדָשֵׁינוּ וְלֹא מוֹעָדֵינוּ

נִיסָן, בְּךָ מֵתוּ בְנֵי אַהְרֹן וְגַם / מִרְיָם, וְנִסְתַּלֵּק בְּאֵר מֵימֵינוּ

בִּן נוּן יְהוֹשֻׁעַ בְּךָ נֶאְסַף וְסָפְ - / דוּ בָךְ קְהָלֵנוּ בְנֵי עַמֵּנוּ

45 אִיָּר, בְּךָ עֵלִי וּבָנָיו נֶאְסְפוּ, / נִלְקַח אֲרוֹנֵנוּ וְלוּחוֹתֵינוּ

גַּם מֵת בְּנוֹ חַנָּה שְׁמוּאֵל בָּךְ, וְגַם / סָפְדָה לְמִיתָתוֹ קְהִלָּתֵנוּ

סִיוָן, בְּךָ מָלַךְ יְרוֹבְעָם בֶּן נְבָט, / מָנַע עֲלוֹת עַם אֵל לְבֵית קָדְשֵׁנוּ

נִשְׂרַף חֲנָנִי בֶּן תְּרַדְיוֹן בָּךְ, וְגַם / סֵפֶר אֱלֹהֵינוּ וְתוֹרָתֵנוּ

גַּם בָּךְ חֲנָנְיָה, הוּא סְגַן הַכֹּהֲנִים, / גַּם בֶּן אֱלִישַׁע נֶהֱרַג נֶגְדֶּנוּ

50 עוֹד בֵּן לְגַמְלִיאֵל בְּךָ נִשְׂרָף, שְׁמוֹ / שִׁמְעוֹן נְשִׂיאֵנוּ, נְגִיד דּוֹרֵנוּ

תַּמּוּז, בְּךָ נִשְׁתַּבְּרוּ לוּחוֹת בְּרִית, / בָּטְלוּ תְמִידֵנוּ וְגַם זִבְחֵנוּ

נִשְׂרַף בְּךָ סֵפֶר אֱלֹהֵינוּ וְגַם / הָעְמַד בְּךָ צֶלֶם בְּהֵיכָלֵנוּ

בָּךְ הָבְקְעָה הָעִיר וְחָרְבָה, שָׁמְמָה, / גַּם הַכְּלִמָּה כִסְּתָה פָנֵינוּ

אָב, אַתְּ בְּמִסְפַּר הַיְרָחִים אַל תְּהִי, / רָדוּ בְךָ רֹאשׁ אָב, לְאַנְחֹותֵינוּ

55 נִגְזַר עֲלֵי אָבוֹת לְבַל יִכָּנְסוּ / אֶרֶץ צְבִי, לֹא בָךְ תְּהִי מוֹתֵנוּ

נִסְתַּלְּקוּ בָךְ עַנְנֵי כָּבוֹד לְמוֹת / אַהְרֹן קְדוֹשֵׁנוּ וְרֹאשׁ כֹּהְנֵינוּ

נֶחְרַב דְּבִיר רִאשׁוֹן וְשֵׁנִי, נִלְכְּדָה / בִתֵּר, וְגָם בָּךְ נֶחְרְשָׁה עִירֵנוּ

כָּבָה בְּךָ נֵר מָעֲרָבִי, רַב לְךָ, / כָּבוּ מְאוֹרֵינוּ וּבָא שִׁמְשֵׁנוּ

חֹדֶשׁ אֱלוּל, מֵתוּ בְךָ כָּל חַי / אֲשֶׁרהוֹצִיא אֲזַי דִּבָּה עֲלֵי אַרְצֵנוּ

60 תִּשְׁרִי, בְּךָ נֶהְרַג גְּדַלְיָהוּ, וְגַם / גָּלְתָה וְאָבְדָה בָךְ שְׁאֵרִיתֵנוּ

רַבִּי עֲקִיבָא בֶן יְהוֹסֵף עוֹד בְּךָ / נִלְקַח וְנִמְסַר, בְּעֲווֹנוֹתֵינוּ

חֶרֶב וְרָעָב עוֹד, וְדֶבֶר, נִגְזְרוּ / בְךָ עַל אֲבוֹתֵינוּ וְקַדְמוֹנֵינוּ

מַר חֶשְׁוָן, אַתָּה כְשִׁמְךָ, כִּי בְךָ / מַר חָשׁ, וְנֶהְפַּךְ לַעֲנָה מָתְקֵנוּ

נִכְתַּב מְרוֹרוֹת עַל מְשִיחֵנוּ בְמַר / חֶשְׁוָן, וְגַם נִלְכַּד בְּפַח צָרֵינוּ

65 בָּךְ נִשְׁחֲטוּ בָנָיו, וְעֵינָיו עִוְּרוּ, / וַיִּשְׁחֲטוּ בָךְ כָּל בְּנֵי חוֹרֵינוּ

כִּסְלֵו, כְּסָלֵינוּ בְךָ נִקְלוּ, בְּעֵת / שָׂרַף יְהוֹיָקִים מְגִלָּתֵנוּ

מֵת בָּךְ נְחֶמְיָה בֶּן חֲכַלְיָהוּ, וְגַם 1 עֶזְרָא נְגִידֵנוּ, מְהִיר סוֹפְרֵינוּ

טֵבֵת, בְּךָ תוֹרָה יְוָנִית נִכְתְּבָה, / חָשְׁכוּ לְכָתְבָה כּוֹכְבֵי נִשְׁפֵּנוּ

סָמְכוּ בְךָ כַשְׂדִּים יְדֵיהֶם לַחֲרִיב / עָרֵי אֱלֹהֵינוּ וּמִקְדָּשֵׁנוּ

70 עַל אִישׁ וּפִילַגְשׁוֹ שְׁבָטִים נוֹעֲדוּ / בִשְׁבָט, וְכִלּוּ אֶת שְׁאָר שִׁבְטֵנוּ

גַּם בַּעֲווֹן פֶּסֶל לְמִיכָה נִקְבְּצוּ / לִלְחֹם בְּבִנְיָמִין שְׁאֵרִיתֵנוּ

אָדָר, בְּךָ נֶעְדַּר גְּבִיר נֶהְדָּר, שְׁמוֹ / משֶׁה, נְבִיאֵנוּ וְגַם רַבֵּנוּ

בָּךְ שֻּדְּדוּ יוֹם בָגְדוּ בָנוּ בְנֵי / יָמִים וְלָקְחוּ מַחֲמַד עֵינֵינוּ

צִיר נֶאֱמָן, מוֹפֵת זְמַן, עֵץ רַעֲנָן, / מָתַק מְאֹד פִּרְיוֹ אֱלֵי חִכֵּנוּ

75 חֶמְדַּת בְּרוּאֵי אֵל, יְקוּתִיאֵל שְׁמוֹ, / משֶׁה נְבִיאֵנוּ וּמוֹשִׁיעֵנוּ

עָלָה אֱלֵי הַר הָעֲבָרִים, הַר נְבוֹ, / אֵל אוֹהֲבוֹ, לַעֲשׁוֹת רְצוֹן יוֹצְרֵנוּ

יִתְרוֹן לְךָ עַל הַרְרֵי אֵל, כִּי בְךָ / חֶלְקַת מְחוֹקֵק, אַתְּ בְּחִיר הָרֵינוּ

נִרְצֶה אֲבָנֶיךָ, נְחוֹנֵן אֶת עֲפָר / מִדְרַךְ פְּעָמָיו, הוּא לְבַד מוֹרֵנוּ

שֶׁמֶש, הֲתִתְגָּאֶה? בְּגַלְגַּלָּךְ עֲמֹד, / משֶׁה כְבָר עָלָה אֱלֵי מַלְכֵּנוּ

80 יוּכַל אֱנוֹשׁ לַבִּיט לְאוֹרֶךָ – וְלֹא / יוּכַל לְהַבִּיט אֶל פְּנֵי דוֹדֵנוּ

עוֹד עֹז וְיִתְרוֹן לוֹ עֲלֵי אוֹרָךְ, וְאַתְּ / תַּקְדִּיר – וְיָאִיר תָּאֳרוֹ נִכְחֵנוּ

יוֹנַת רְחוֹקִים! כַּבְּרָקִים מַהֲרִי / אֵל דּוֹד אֲשֶׁר מִיַּם עֳנִי יַמְשֵׁנוּ

וּפְנִי סְעִי, אַל תִּרְגְּעִי, עַד תִּפְגְעִי / בִּבְחִיר מְשִׁיחֵנוּ, אֲהוּב נַפְשֵׁנוּ

אָז תֹּאמְרִי לוֹ כִּי בְּכָל שָׁעָה אֲנַחְ- / נוּ עוֹד מְחַכִּים בּוֹא יְשׁוּעָתֵנוּ

85 חַלּוּ פְנֵי הָאֵל וְאוּלַי יִשְלְחָה / לָנוּ מְבַשֵּׂר טוֹב וְיַשְׁמִיעֵנוּ

שָׁלוֹם וְנֶחָמוֹת אֲשֶׁר הֵמָּה כְתוּ- / בוֹת עַל יְדֵי חוֹזִים יְקַיֶּם לָנוּ

יָבוֹא מְשִׁיחֵנוּ בְּקָרוֹב הַזְּמַן,       נַעְלֶה לְבֵית קֹדֶשׁ וְאֶל אַרְצֵנוּ








(ברצלונה, פ“ו [1326] – אלג'יר, קס”ח [1407])  ↩
