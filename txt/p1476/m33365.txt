


	
	אִישׁ הָיָה בִכְפַר חָנָן
	








מתוך השער השלישי של משל הקדמוני – הכילי שבא על ענשו





אִישׁ הָיָה בִכְפַר חָנָן / בְּצֵל הַמָּמוֹן יִתְלוֹנָן,



/ וַיְהִי לוֹ בֵית חֶמְדָּה / וּמִקְנֵה בָקָר וַעֲבֻדָּה. / וַיְהִי



כִילַי בִּמְמוֹנוֹ וּמְאֹדוֹ, / מַשְׁחִית תִּפְאַרְתּוֹ וְהוֹדוֹ, /



וְדַעְתּוֹ בֶּאֱמוּנָתוֹ נִבְדֶּלֶת / בְּהַרְבּוֹת דְּבָרִים לְאֵין



תּוֹעֶלֶת, / וּמְמוֹן עָנִי וְאֶבְיוֹן יְפַזֵּר / וְיָתוֹם וְאֵין עוֹזֵר



/ בְרֹב דְּבָרָיו הָיָה עָצֵל וְכָבֵד / כִּילַי אוֹבֵד.

וַיְהִי הַיּוֹם וַיָּבוֹא אֶל בֵּיתוֹ הֵלֶךְ / כְּתֹאַר



בְּנֵי הַמֶּלֶךְ / אִישׁ אֶחָד נִכְבָּד מִבְּנֵי עַמּוֹ, / וּשְׁנֵי



נְעָרָיו עִמּוֹ. / וַיֹּאמֶר לוֹ ״הַאֶמְצָא מָלוֹן הַלַּיְלָה הַזֶּה



בְקִרְבֶּךָ?״ וַיֹּאמַר ״הוֹאֶל נָא וְלִין וְיִיטַב לִבֶּךָ!״



וְהָאוֹרֵחַ לֹא הֵבִין סוֹד תַּחְבּוּלוֹתָיו / שֶׁיּוֹרֵד עַל פִּי



מִדּוֹתָיו. / וַיִּתֵּן לְתַחֲנוֹתָיו מִטָּה מֻצַּעַת וְיָפָה, /



וַיַּעַשׂ לוֹ מִשְׁתֶּה וּמַצּוֹת אָפָה. / וַיּוֹסֶף לְכַבְּדוֹ



וּלְהַדְּרוֹ / וַיִּתֵּן מִסְפּוֹא לַחֲמוֹרוֹ, / וַיִּתֵּן מַיִם לִרְחֹץ



רַגְלָיו / וְרַגְלֵי הָאֲנָשִׁים הַנִּלְוִים אֵלָיו, / וַיֹּאכְלוּ



כֻלָּם בְּבֵיתוֹ וְאוּלַמּוֹ / וַיִּשְׁתּוּ וַיִּשְׁכְּרוּ



עִמּוֹ.

וַיְהִי אַחֲרֵי הַסְּעוּדָה / וַיִּפְתַח הָאוֹרֵחַ



בְּמִשְׁלֵי תְעוּדָה / וַיֹּאמֶר אֶל הָאִישׁ ״שְׁמַע נָא דְבָרַי, /



יֹשֶׁר לִבִּי אֲמָרַי: / הִנֵּה אַתָּה בַהֲדָרְךָ כִּבָּדְתָּנוּ /



וּמִטּוּב נָעָמְךָ הֶאֱכַלְתָּנוּ / וּמִיֵּין רֶקַח נְדִיבוּתְךָ



הִשְׂבַּעְתָּנוּ / וּמִמֵּיטַב פְּעֻלָּתְךָ שִׂמַּחְתָּנוּ. / עַל כֵּן



רָאוּי לְהוֹדִיעֲךָ דְבָרִים עֲרֵבִים, / גּוֹמֵל חֲסָדִים טוֹבִים, /



לְמַעַן תָּבִין בְּמַהֲלַל לִבִּי וְדַרְכּוֹ / לְמִי עוֹלַלְתָּ כֹּה״. /



וַיֹּאמַר ״דַּבֵּר כְּפִי צִדְקְךָ וְחַסְדֶּךָ, / כִּי שׁוֹמֵעַ



עַבְדֶּךָ״.

וַיֹּאמֶר הָאוֹרֵחַ ״אָמְרוּ הַקַּדְמוֹנִים / אֲשֶׁר



בְּמַחֲנֵה הַחָכְמָה חוֹנִים / טוּב הַמָּלוֹן וּשְׁלֵמוּתוֹ / קוֹנֶה



בֶן־חֹרִין בִּנְדִיבוּתוֹ, / וּפְרִי הַנְּדִיבוּת וְהָרֶוַח / הוּא



הַמַּהֲלָל וְהַשֶּׁבַח. / וְאָמְרוּ: נְדִיבוּת הַנִּכְבָּדִים – /



גְּמִילוּת חֲסָדִים, / וּמִפְעָלוֹת הַנֶּאֱהָבוֹת – / גֶּשֶׁם נְדָבוֹת,



/ כִּי הַזּוֹרֵעַ נְדָבָה / יִקְצֹר הוֹד וְאַהֲבָה. / וְאָמְרוּ:



הַנָּדִיב יְבַקֵּשׁ הַחֵן וְהַיֹּשֶׁר / וְהַכִּילַי – הַמָּמוֹן



וְהָעֹשֶׁר. / וְהַמִּתְנַדֵּב בְּנַפְשׁוֹ יְהֻלַּל / וְקוֹבֵץ עַל יָד –



חֶלְקוֹ יְקֻלַּל. / וְאָמְרוּ: הַנְּדִיבוּת יְחַזֵּק הַבֶּדֶק /



וּמִשָּׁם נוֹלַד הַצֶּדֶק. / הַצֶּדֶק יַצְמִיחַ הַשֶּׁבַח וְהַכֹּשֶׁר /



וּמִמֶּנּוּ יִוָּלֵד הַיֹּשֶׁר. / הַיֹּשֶׁר יוֹלִיד הַנֵּצַח וְהָעֶזְרָה



/ וּמִמֶּנּוּ תֵצֵא הָחֲבוּרָה. / הַחֲבוּרָה – גֶּפֶן חֲזָקָה /



תַּצְמִיחַ צְדָקָה. / הַצְּדָקָה תָּבוֹא הֲלֹם / וְתוֹסִיף אַהֲבָה



וְשָׁלוֹם. / הַשָּׁלוֹם – סוֹד הַתְּבוּנָה / וְעִקַּר הָאֱמוּנָה. /



הָאֱמוּנָה מוּכֶנֶת לְהָבִין וּלְהוֹרוֹת / וְהִיא רֹאשׁ הַתּוֹרוֹת. /



הַתּוֹרוֹת – מִשְׁפָּטִים וְחֻקִּים / מִצּוּף וּמִנֹּפֶת מְתוּקִים, /



וּבָהֶם יִשּׁוּב הָאֲרָצוֹת / קִרְבַת אֱלֹהִים חֲפֵצוֹת. / רְאֵה כַּמָּה



הַנְּדִיבוּת נִכְבֶּדֶת / אֲשֶׁר בְּרֹאשׁ הַמַּעֲלוֹת עוֹמֶדֶת /



וְכַמָּה גְדוֹלָה מַעֲלַת הַשְּׂמֵחִים / בְּהַכְנָסַת אוֹרְחִים!״





וַיֹּאמֶר הָאִישׁ אֶל הָאוֹרֵחַ / אֲשֶׁר בִּנְגִינוֹתָיו



יְנַצֵּחַ / ״הֲיֵּשׁ לְךָ מָחָר לְהַשְׁכִּים, / לְהִתְיַצֵּב בַּבֹּקֶר



לִבְּךָ מַסְכִּים? / עֵת הוּא לְהָכִין מִטּוֹת וּלְהַצִּיעַ – / אַחַר



תְּחַוֶּה דַעְתְּךָ וְתַבִּיעַ״. / וַיֹּאמֶר לוֹ ״עוֹד יֶשׁ לִי לְתַקֵּן



בְּאַמְתַּחְתִּי מִגְרָע / אֲשֶׁר הַיּוֹם נִקְרַע״. / וַיִּקַּח



אַמְתַּחְתּוֹ לְתָפְרוֹ, / וְשָׁם נִקְרַב הָאִישׁ בַּעַל הַבַּיִת



לְעָזְרוֹ, / וַיַּרְא בְּתוֹכוֹ צִנְצֶנֶת / לְתַכְלִית הַיֹּפִי



מְתֻקֶּנֶת. / וַיֹּאמֶר אֶל הָאִישׁ ״מַה זֶּה בְּיָדֶךָ / לְמַעֲלַת



כְּבוֹדְךָ וְהוֹדֶךָ?״ / וַיֹּאמַר ״אֲנִי רוֹכֵל בְּאֶרֶץ הַחֲוִילָה, /



נוֹרָא עֲלִילָה. / בַּשָּׁנָה הַזֹּאת חָסֵר בְּעִירִי צְרִי גִלְעָד /



לְמַרְפֵּא וּלְמִסְעָד, / וַאֲנִי הָלַכְתִּי אֶל גִּלְעָד בַּעֲבוּרוֹ /



וָאַרְבֶּה זָהָב בִּמְחִירוֹ, / וְזֶהוּ הַצֳּרִי הַגָּדוֹל אֲשֶׁר אֵין



חֵקֶר לְתִפְאַרְתּוֹ / לְכָל אֲשֶׁר נִמְצָא אִתּוֹ״. / וַיֹּאמֶר לוֹ



״בְּחַסְדְּךָ הוֹדִיעֵנִי בְתוֹעֲלוֹתָיו דָּבָר נִרְשָׁם / זְעֵיר שָׁם



זְעֵיר שָׁם״. / וַיֹּאמֶר הָאוֹרֵחַ ״וּמִי יוּכַל לְמַלֵּל גְּבוּרוֹתָיו



/ וְשֶׁבַח סְגֻלַּת מַעֲלוֹתָיו? / אֲבָל הָרוֹפְאִים נִמְנוּ וְגָמְרוּ



בְמַהֲלָלָיו אֶחָד וְתִשְׁעִים / לְכָל מַשְׂכִּיל נוֹדָעִים / וְאִם



תִּרְצֶה לֵדַע סִימָן בְּמַהֲלָלוֹ / ״צֵא״ תֹּאמַר לוֹ / כִּי בְכָל



מֵעֲלָה וּמַעֲלָה / יוֹעִיל לְכָל כְּאֵב וּלְכָל מַחֲלָה / וְהַשּׁוֹתֶה



סַם הַמָּוֶת – בְּטוּבוֹ יַצִּילֵהוּ / וּבִשְׁתוֹתוֹ מִמֶּנּוּ מִיָּד



יוֹעִילֵהוּ״. / וַיֹּאמֶר לוֹ ״תֶּן לִי מְעַט מִזְעָר לְזֵכֶר



מַעֲלוֹתָיו / לְמַעַן אֲסַפְּרָה כָּל נִפְלְאוֹתָיו״. / וַיֹּאמַר



״שָׁמַעְתִּי לְקוֹלֶךָ / אֲנִי אֲמַלֵּא אֶת מִשְׁאָלֶךָ״. / וַיִּתֶּן



לוֹ מִמֶּנּוּ בִּכְלִי קָטֹן / צְלוֹחִית שֶׁל פָּלְיָטוֹן. / וַיֹּאמֶר



לוֹ ״הֲרֵי לְךָ מֹהַר וּמַתָּן / בִּצְלוֹחִית קְטַנָּה נִתָּן / וְעַתָּה



שִׂים עֵינֶיךָ לְשָׁמְרֵהוּ / כִּי לֹא דָבָר רֵיק הוּא!״

וַיְהִי כְּכַלּוֹת הָאוֹרֵחַ לְתַקֵּן אַמְתַּחְתּוֹ /



וּבָא וְלָן בְּמִשְׁכַּב תִּפְאַרְתּוֹ – / וַיַּעַל בְּלֵב בַּעַל



הַבַּיִת / שָׁמִיר וָשַׁיִת. / וַיֹּאמֶר לוֹ ״אֲדוֹנִי, הוֹאִיל וְאֵין



רְצוֹנְךָ לְהִתְאַחֵר / וּבַבֹּקֶר אַתָּה תְשַׁחֵר – / צַוֵּה לִפְרֹעַ



הַהוֹצָאָה / אֲשֶׁר לְרַגְלְךָ בָאָה״. / וַיֹּאמַר ״מַה הוּא הוֹצֵאתִי /



אֲשֶׁר בַּסֵּתֶר עֻשֵּׂיתִי?״ / וַיֹּאמַר ״לֹא חִסַּרְתִּי דָבָר



מִמִּשְׁאָלֶיךָ / בַּל קְרֹב אֵלֶיךָ, / וְכָל דָּבָר בִּמְחִיר גָּדוֹל



נִרְצָה / וּבְסֵפֶר זֶה נִמְצָא – / כָּל שֶׁכֵּן בְּחִפָּזוֹן / בַּר



לֶחֶם וּמָזוֹן! / וְעַתָּה יֶשׁ לְךָ לָתֵת כְּכֹל אֲשֶׁר הוֹצֵאתָ פֹה /



עִם הַתֶּבֶן וְעִם הַמִּסְפּוֹא – / כֶּסֶף עֲשָׂרָה שְׁקָלִים /



בְּשֶׁקֶל הָעִיר נִשְׁקָלִים״. / וַיִּשְׁמַע הָאוֹרֵחַ הַדְּבָרִים



וַיִּתְעַצֵּב / וַיָּקָם עַל פֶּתַח הַמִּשְׁכָּן וַיִּתְיַצֵּב /



וַיֹּאמֶר בְּלִבּוֹ: וּמַה יּוֹעִיל כַּעְסִי וַחֲמָתִי / וְלָמָה



אֲחַלֵּל עִמּוֹ צֶדֶק אִמְרָתִי? / וַיֹּאמֶר לוֹ ״קַח אֶת כָּל אֲשֶׁר



תֹּאמַר, / כִּ יֵין עֲסִיסְךָ לְשׁוֹתָיו יֵמַר״. / וַיִּשְׁקֹל לוֹ



הַכֶּסֶף כַּאֲשֶׁר הָיָה בוֹחֵר / עוֹבֵר לַסּוֹחֵר.

וַיְהִי בְאַשְׁמֹרֶת הַבֹּקֶר וַיַּשְׁכֵּם הָאוֹרֵחַ



לַעֲמֹס חֲמוֹרוֹ / וְעִמּוֹ נְעָרָיו עוֹשִׂים דְּבָרוֹ. / וַיִּקְרָא אֶל



הָאִישׁ פֶּתַח הַשַּׁעַר – / וַיָּנַע לְבָבוֹ כְּנוֹעַ עֲצֵי הַיַּעַר /



וַיָּקָם בְּכֹבֶד וּבַעֲצַלְתַּיִם / כְּשִׁכּוֹר אָסוּר בַּנְחֻשְׁתַּיִם



/ וַיֹּאמַר ״תֶּן לִי שְׂכַר הַמָּלוֹן וְהַחֲנָיָה / וּשְׂכַר מִטָּה



יְפֵה־פִיָּה!״ / וַיֹּאמֶר הָאוֹרֵחַ ״כַּמָּה הַמְּחִיר אֲשֶׁר תִּדְרֹשׁ



/ וּמִפְּנֵי אָדָם לֹא תֵבוֹשׁ?״ / וַיֹּאמַר ״בְּיַד הָאִישׁ הַיְכֹלֶת –



/ בֶּקַע לַגֻּלְגֹּלֶת!״ / וַיְמַהֵר וַיִּתֶּן לוֹ הַמְּחִיר אֲשֶׁר



שָׁאַל, / כִּי הוּא מְטֻמָּא וּמְגֹאָל. / וַיִּפְתַּח, וַיֵּלֶךְ



הָאוֹרֵחַ לִנְתִיבוֹ / וּשְׁנֵי נְעָרָיו סְבִיבוֹ. / וַיֹּאמֶר לָהֶם



״בָּרְכוּ, בָּנַי, בְּכֹחַ לָאֵל / יְיָ מִמְּקוֹר יִשְׂרָאֵל, / אֲשֶׁר



הִצִּילָנוּ מִיַד הָאִישׁ הַנָּבָל / אֲשֶׁר הִרְשִׁיעַ לָנוּ וְחָבַל. /



וְאִם הָיִיתִי יוֹדֵעַ מַעֲלוֹ וְרִשְׁעוֹ / וַהֲבִינוֹתִי נִכְלוֹ



וּבִצְעוֹ – / לֹא הָיִיתִי לָן בִּמְלוֹנוֹ / וְלֹא קָרַבְתִּי אֶל



שֻׁלְחָנוֹ. / וְהִנְנִי מִתְנַחֵם לְרֹב פִּשְׁעוֹ וּמַעֲלוֹ / עַל



הַצֳּרִי אֲשֶׁר נָתַתִּי לוֹ״. / וַיֹּאמְרוּ לוֹ הַנְּעָרִים ״אַל יִחַר



בְּעֵינֵי יְחִידָתֶךָ / כַּתַר לוֹ זְעֵיר וְתִרְאֶה נִקְמָתֶךָ. / אוּלַי



הַצֳּרִי יִהְיֶה סִבָּה לַחֲלוֹתוֹ / חַרְצֻבּוֹת לְמוֹתוֹ״. / וַיִּשָּׂא



הָאִישׁ אֶת עֵינָיו אֶל רוּם מְעוֹנַי / וַיֹּאמַר ״אָמֵן, כֵּן יַעֲשֶׂה



יְיָ!״ / וַיֵּלֶךְ לְדַרְכּוֹ נִפְלָא וְרַעֶיוֹנָיו תְּמֵהִים – /



וַיִּפְגְּעוּ בוֹ מַלְאֲכֵי אֱלֹהִים.

וְהַכִּילַי קָם מִמִּשְׁכָּבוֹ / וַיֹּאכַל וַיֵּשְּׁתְּ



וַיִּטַב לִבּוֹ / וַיִּתְהַלֵּל וַיִּשְׁתַּבֵּחַ / עַל כָּל אֲשֶׁר



עָשָׂה אֶת הָאוֹרֵחַ / וְעַל דְּבַר הַצֳּרִי אֲשֶׁר שָׁאַל לִשְׂכָרוֹ /



וְלֹא הִרְבָּה בִמְחִירוֹ, / וַיֹּאמֶר בְּלִבּוֹ: יִקְרְבוּ יְמֵי



הַקַּיִץ וְאֵלֶךְ לִי אֶל הַמְּדִינָה / מְקוֹם הַדַּעַת וְהַבִּינָה /



וְאֶמְכֹּר אֶת הַצֳּרִי לְאַחַד הַסּוֹחֲרִים / אֲשֶׁר בִּינָה שׁוֹחֲרִים



/ וְאֶקְנֶה בְדָמָיו מַלְבּוּשִׁים / וְיֵין עֲנוּשִׁים.

וַיְהִי מִיָּמִים וְהִנֵּה אִישׁ עוֹבֵר / וְהַצָּמָא



בְּקִרְבּוֹ גוֹבֵר / וַיָּסַר אֶל בֵּיתוֹ לִשְׁתּוֹת / וַיּוֹאֶל



הַכִּילַי אוֹתוֹ לְפַתּוֹת / וַיֹּאמֶר לוֹ ״מַה מְּלַאכְתְּךָ וּמֵאַיִן



תָּבוֹא?״ / לָדַעַת עִסְקוֹ וּמַחֲשָׁבוֹ. / וַיֹּאמֶר לוֹ ״אָנֹכִי אִישׁ



צַיָּד, / בְּחַרְבִּי וְקַשְׁתִּי צְבִי מִיָּד. / וְהָלַכְתִּי לִקְנוֹת



לִמְלַאכְתִּי סַם / מֵאֵת אִישׁ אֶחָד בַּשָּׂם, / לָתֵת מִמֶּנּוּ



בְּרֹאשׁ הַחִצִּים / לִהְיוֹת נוֹגְשִׂים וְאָצִים / לְהָמִית כָּל



בְּהֵמָה וְחַיָּה / בְּכָל כְּלִי צְדִיָּה. / וְהַסַּם הַזֶּה הוּא חָזָק



/ וְכָל הַנּוֹגֵעַ בּוֹ יֻזַּק / וְלֹא יוֹעִיל לוֹ מַרְפֵּא וּמִסְעָד, /



זוּלָתִי הַצֳּרִי הַגָּדוֹל הַבָּא מִגִּלְעָד״. / וַיֹּאמֶר הַכִּילַי



״תֶּן לִי מְעַט מִמֶּנּוּ וְאָשִׂים מַאֲרָב / אֶל נָחָשׁ שָׂרָף



וְעַקְרָב / אֲשֵׁר הֵם לוֹחֲצִים אוֹתִי / וְאֶת בְּנֵי תְנוּבָתִי״. /



וַיִּתֶּן לוֹ מְעַט מִזְעָר / וְחָצְנוֹ נָעַר. / וַיִּפֶן הַצַּיָּד



וַיָּשָׁב לְבֵיתוֹ / וְהַחַיּוֹת יָגוּרוּ מִשֵּׁתוֹ.

וְהַכִּילַי חָשַׁב מַחְשְׁבוֹת אָוֶן בְּתַאֲוָתוֹ, /



כִּי נִתְמַלְאָה סְאָתוֹ, / וַיֹּאמַר: עַתָּה אֶבְחַן הַצֳּרִי



וְתִפְאַרְתּוֹ / וְאֵדַע מְחִירוֹ וְתוֹעַלְתּוֹ. / אִם סְגֻלַּת הַסַּם



וְהַצֳּרִי כַּאֲשֶׁר לָמַדְתִּי יָפֶה – / אֲנִי אֶמָּחֵץ וַאֲנִי



אֵרָפֵא. / וַיִּקַּח מִן הַסַּם וַיִּתֵּן בִּלְשׁוֹנוֹ / וַיִּלָּכֵד



בַּעֲווֹנוֹ / וַיִּגְוַע לִבּוֹ כְרֶגַע – / הַמִּקְרֶה וְהַפֶּגַע! – /



וַיִּתְהַפֵּךְ בְּצִירוֹ וַיִּתְחַלְחַל / וּפָחַד וְרָהָה וְזָחַל. /



וַיָּרָץ אֶל גְנָזָיו לָקַחַת אֶת הַצֳּרִי / לְהָסִיר אַף וָחֳרִי – /



וַיִּתְבַּלְבְּלוּ יָדָיו בַּמַּפְתֵּחַ, / אֵין סוֹגֵר וְאֵין פּוֹתֵחַ.



/ וַיִּשְׁאַג וַיִּנְהֹם בְּמַכְאוֹבוֹ / לְרֹעַ עֲצָתוֹ וּמַחֲשָׁבוֹ. /



וַתֵּרֶא אִשְׁתּוֹ כִּי נִלְכַּד בְּמַעֲלָלָיו / וְצִירָיו נֶהֶפְכוּ



עָלָיו, / וַתִּפְתַּח וַתִּקַּח הַצֳּרִי הַשָּׁמוּר כְּבַת עַיִן /



וַתָּבֵא לוֹ יַיִן – / וְאֵין שׁוֹתֶה וְאֵין לוֹקֵחַ, / אֵין מַבְרִיחַ



וְאֵין בּוֹרֵחַ! / וַיִּגְוַע וַיָּמָת בַּאֲשָׁמָיו / וְנָפַל



בַּעֲצוּמָיו, / וְהַמַּרְפֵּא לֹא הוֹעִיל / צָרְיוֹ עִבַּר – וְלֹא



יַגְעִיל.
