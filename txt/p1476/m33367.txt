


	
	אָמְרוּ כִּי הָיָה בְאַשְׁדּוֹד
	








מתוך השער השני של משל הקדמוני – החסיד ואשתו הנואפת





אָמְרוּ כִּי הָיָה בְאַשְׁדּוֹד אִישׁ תָּמִים



בְּמִדּוֹתָיו / וְצַדִּיק הָיָה בְּדוֹרוֹתָיו, / אִישׁ יִשְׁמָעאֵל /



נֶאֱמָן עִם אֵל / וְעִם קְדוֹשִׁים / זָקֵן בָּא בַאֲנָשִׁים. / וַתְּהִי



לוֹ אִשָּׁה מְיֻחֶסֶת וְתַמָּה, / יָפָה כַלְּבָנָה בָּרָה כַּחַמָּה, /



וּמֵרֹב צִדְקָתוֹ הָיָה נֶחְפָּר וְנִכְלָם / וְלֹא נִסְתַּכֵּל בְּהוֹד



יָפְיָהּ לְעוֹלָם / וּבְחֶזְקַת אֵשֶׁת חַיִל מַחֲזִיק אוֹתָהּ, / וְהוּא



לֹא רָאָה אֶת עֶרְוָתָהּ / וְלֹא הִכִּיר בִּפְעָלֶיהָ, / כִּי לֹא



הִבִּיט אֵלֶיהָ. / וַתַּחְשֹׁק הָאִשָּׁה בָּחוּר אֶחָד מִבְּנֵי הָעִיר /



אַדְמוֹנִי יְפֵה תֹאַר רַךְ בַּשָּׁנִים וְצָעִיר, / וַתִּבְעַר



בְּלִבָּהּ אֵשׁ תְּשׁוּקָתוֹ – / נִפְלְאַתָה אַהֲבָתוֹ / מַיִם רַבִּים



לֹא יָכְלוּ לְכַבּוֹת אַהֲבָתוֹ.

וַיְהִי בִימֵי קְצִיר חִטִּים / וְרַעְיוֹנֵי הָאִשָּׁה



אֶל תַּאֲוָתָהּ מָטִים / וּמִדֶּרֶךְ הַצְּנִיעוּת נָדִים וּמִתְמוֹטֲטִים



/ כְּמַעֲשֵׂה שִׁטִּים. / וַיִּגַּע עֶרֶב חַגָּם בְּבִטְחָה – / הֶחָלִיל



וְהַשִּׂמְחָה – / וַיִּהְיוּ שְׂמֵחִים וְעַלִּיזִים, / וַיִּפְקֹד



הָאִישׁ אֶת אִשְׁתּוֹ בִּגְדִי הָעִזִּים. / וַיְהִי כַּעֲלוֹת הַמִּנְחָה



/ לְעֵת מְצֹא נֹפֶשׁ וַהֲנָחָה / וַיֵּלֶךְ הָאִישׁ בֵּית הַתְּפִלָּה /



בְּמַעֲטֵה עֹז וּתְהִלָּה / בְּגִילָה וְצָהֳלָה.

וַיֹּאמֶר לְאִשְׁתּוֹ ״הָכִינִי הַשֻּׁלְחָן



וְהַמְּנוֹרָה / וְהִנְנִי הוֹלֵךְ לְהִתְפַּלֵּל בַּכְּנֶסֶת



הַהֲדוּרָה״.

וַתַּעֲמֹד אִשְׁתּוֹ לְהָבִין בַּבָּנִים חֲסַר לֵב



וּבַעַר / נֹכַח פֶּתַח הַשַּׁעַר – / וְהִנֵּה הַבָּחוּר בְּרִנָּה /



עוֹבֵר בַּשּׁוּק אֵצֶל פִּנָּה. / וַיִּצְעַד וַיָּבוֹא לִקְרָאתָהּ /



וַיִּקְרַב אֶל פֶּתַח בֵּיתָהּ, / וַתִּסְגֹּר הָאִשָּׁה בַּעֲדוֹ /



וַתִּתְפְּשֵׂהוּ בְּבִגְדוֹ. / וַתֵּרֶא אֶת הוֹד יָפְיוֹ וְגָדְלוֹ, /



הֵעֵזָה פָנֶיהָ וַתֹּאמַר לוֹ / ״זִבְחֵי שְׁלָמִים עָלַי מִנְּעוּרַי /



הַיּוֹם שִׁלַּמְתִּי נְדָרַי. / וְהָיִיתִי עֲצוּבָה עַל נְדָרַי /



וְלִבִּי הַמָּוֶת בּוֹחֵר / לְמֵיקַם עֲלֵהּ מִשּׁוּם ׳בַּל תְּאַחֵר׳. /



וּכְבָר הִזְהִירוּנִי חַכְמֵי הַדּוֹרוֹת / לְהָבִין וּלְהוֹרוֹת /



וְאָמְרוּ: אֶחָד הַנּוֹדֵר וְאֶחָד הַמַּעֲרִיךְ / וּמִלְּקַיֵּם



מַאֲמָרוֹ הֶאֱרִיךְ / עוֹבֵר בְּ׳בַל תְּאַחֵר׳ אִם אֵינוֹ מַשְׁלִים /



כֵּיוָן שֶׁעָבְרוּ עָלָיו שָׁלֹשׁ רְגָלִים. / וְאָמְרוּ: חֹמֶר



בִּנְדָרִים מִשְּׁבוּעוֹת / וְכָל אֵלּוּ הֲלָכוֹת יְדוּעוֹת /



שֶׁהַנְּדָרִים עַל גַּב שְׁבוּעוֹת עוֹלִים / וּלְבַטֵּל הַמִּצְוָה



חָלִים. / וְאָמְרוּ: בַּעֲווֹן נְדָרִים אִשְׁתּוֹ שֶׁל אָדָם מֵתָה /



כִּי הִיא מִצְוָה צְרוּפָה וְעַל לוּחַ לֵב חֲרוּתָה. / וְעַתָּה אַחֲרֵי



אֲשֶׁר תּוֹחַלְתִּי מִמְּךָ לֹא אָבָדָה, / בּוֹא אִתִּי הַבַּיְתָה



וְנִסְעָדָה, / וְתִשְׂבַּע מֵהוֹדִי וּמִזִּיו פָּנַי / וַאֲשַׁלֵּם אֶת



נְדָרַי אֲשֶׁר נָדַרְתִּי לַייָ״. / וַיֹּאמֶר הַבָּחוּר ״הָיָה לָךְ



לָדוּן אוֹתִי מִקַּל וָחֹמֶר, כִּי אֲנִי תָאֵב מְאֹד לַחֲבוּרָתֵךְ /



וּפִי פָעַרְתִּי לְמַלְקוֹשׁ יְדִידוּתֵךְ – / אֲבָל לֹא רָאִיתִי שְׁעַת



הַכֹּשֶׁר / בֶּאֱמֶת וּבְיֹשֶׁר, / וַאֲנִי מְפַחֵד מִבַּעַל הַבַּיִת /



אֲשֶׁר קִנְאָתוֹ שָׁמִיר וָשַׁיִת״. / וַתֹּאמֶר הָאִשָּׁה ״הֲבִינוֹתִי



הוֹד מַחְשְׁבוֹתֶיךָ / וְנִלְאֵיתִי נְשֹׂא עֹל אַהֲבָתֶךָ / וְלִבִּי



בַּעֲנַק חִשְׁקְךָ נַפְשִׁי הֶעֱנִיק / יוֹתֵר מִמַּה שֶּׁהָעֵגֶל רוֹצֶה



לִינֹק – פָּרָה רוֹצָה לְהָנִיק״.





וַתּוֹסֶף עוֹד לְפַתּוֹתוֹ בִּדְבָרֶיהָ / וַתַּחֲלִיק



עָלָיו אֲמָרֶיהָ / וַתֹּאמַר ״דַּע כִּי בַעְלִי – הַשֵּׁם יִשְׁמְרֵהוּ /



וְיוֹצְרוֹ מִבֶּטֶן יַעְזְרֵהוּ! – / הוּא בַעַל תְּשׁוּבָה / וְנֶפֶשׁ



נִכְבֶּדֶת וַחֲשׁוּבָה / וְנָקִי בְכָל עִנְיָנָיו, / וְעָנְתָה בּוֹ



הַכָּרַת פָּנָיו / כִּי לֹא נִסְתַּכֵּל בְּהוֹד יָפְיִי / לְהַשְׁלִים



אֶת נִשְׁיִי. / וְהוּא הַגָּדוֹל מִבְּנֵי מְתִיבְתָּא / מַקְדִּימֵי



וּמַחֲשִׁיכֵי לְבֵי כְנִישְׁתָּא / וְשָׁעָה אַחֲרֵיהֶם מִתְאַחֵר /



וּבַבֹּקֶר לִפְנֵיהֶם שׁוֹחֵר / מִתְפַּלֵּל וּמִתְחַנֵּן יוֹתֵר



מֵחֲבֵרָיו / מְבָרֵךְ לְפָנָיו וּלְאַחֲרָיו / בַּקָּשׁוֹת רַבּוֹת



וְתַחֲנוּנִים / וּמַרְבֶּה רְנָנִים. / עַתָּה בּוֹא אִתִּי אֶל הַבִּירָה



/ סוּרָה, אֲדוֹנִי, אֵלַי אַל תִּירָא״. / וַיָּסַר אֵלֶיהָ אֶל הַחֶדֶר /



בַּעֲלִיַּת מִגְדָּל עֵדֶר / וַיֹּאכְלוּ וַיִּשְׁתּוּ וַיִּשְׂמָחוּ /



וַיַּעֲלוּ הָעוֹלָה וַיִּצְנָחוּ. / וַתֹּאמַר ״שְׂמַח בְּיַלְדוּתְךָ



וְאַל תִּתְעַצֵּב / הַשְׁכֵּם בַּבֹּקֶר וְהִתְיַצֵּב, / וְעַתָּה בּוֹא



אֵלַי, זוּלַת יָפְיִי אַל תַּחְמֹד / מְהֵרָה חוּשָׁה אַל תַּעֲמֹד, /



כִּי הָאִישׁ יַשְׁכִּים לִתְפִלָּתוֹ לְהַשְׁלִים תַּחֲנוּנָיו



כְּמִשְׁפָּטוֹ וְדָתוֹ / לַעֲשׂוֹת מַעֲשֵׂהוּ – זָר מַעֲשֵׂהוּ



נָכְרִיָּה עֲבוֹדָתוֹ! / וְעַתָּה, חֲשוּקִי, בְּמַאֲמָרִי בִסְתַּכֵּל



וַחֲשֹׁב / לֵךְ וָשׁוֹב!״

וַיָסָב הַבָּחוּר אֶל בֵּית אָבִיו כְּחֻקּוֹ, /



וַתִּשָּׁאֵר הָאִשָּׁה לְכוּדָה בְחִשְׁקוֹ, / וַתְּמַהֵר וַתַּעַשׂ



מַטְעַמִּים / קִנְּמָן־בֶּשֶׂם וְסַמִּים / וַתְּכַבֵּד בֵּיתָהּ



בְּמִינֵי מַרְבַדִּים מְלֵאֵי עֵינַיִם / חֲטוּבוֹת אֵטוּן מִצְרַיִם, /



וַתַּעֲרֹךְ נֵרוֹתֶיהָ כְּפִי חֶפְצָהּ וּרְצוֹנָהּ, / טָבְחָה טִבְחָהּ



מָסְכָה יֵינָהּ / אַף עָרְכָה שֻׁלְחָנָהּ.

וַיְהִי כְּכַלּוֹת הָאִישׁ לְהִתְפַּלֵּל כְּדָתוֹ /



וַיִּפֶן וַיָּבוֹא בֵית מְנוּחָתוֹ, / וַיִּמְצָא רְפִידָתוֹ עֲרוּכָה



וּשְׁמוּרָה, / מִטָּה שֻׁלְחָן כִּסֵּא מְנוֹרָה. / וַיֹּאמֶר לְאִשְׁתּוֹ



״בְּרוּכָה אַתְּ בְּפִי כָל עַם וְלָשׁוֹן! / הֵיטַבְתְּ חַסְדֵּךְ



הָאַחֲרוֹן מִן הָרִאשׁוֹן״. / וַיְמַהֵר וַיֵּשֶׁב בִּמְסִבּוֹ /



וַיֹּאכַל וַיֵּשְּׁתְּ וַיִּיטַב לִבּוֹ. / וַיְהִי כַּאֲשֶׁר כִּלָּה



לֶאֱכֹל פִּתּוֹ, / וַתֹּאמֶר לוֹ הָעֲנֻגָּה אִשְׁתּוֹ / ״הוֹאִיל



וְשָׂבַעְנוּ מִטּוּב הָאָרֶץ וְתִפְאַרְתָּהּ, / לֵימָא לָן מַר מִלֵּי



דַאֲגַדְתָּא!״ / וַיָּחֶל הָאִישׁ לְדַבֵּר / וּבְרִיחֵי אֱמוּנָתוֹ



לְשַׁבֵּר / וַיֹּאמֶר ״שִׁמְעִי, בִּתִּי, וּרְאִי וְהַטִּי אָזְנֵךְ /



לְמַאֲמַר יוֹצְרֵךְ וְקוֹנֵךְ, / וְאוֹדִיעֵךְ מוּסָרִים מְפֹרָשִׁים /



אֲשֶׁר בֵּאֲרוּ זְקֵנִים וִישִׁישִׁים / בְּעִנְיְנֵי הַנָּשִׁים. /



אָמְרוּ: צְרִיכָה הָאִשָּׁה לִהְיוֹת אֵשֶׁת חַיִל בְּמִדּוֹתֶיהָ, /



צְנוּעָה וּשְׁלֵמָה בְמַעֲלוֹתֶיהָ, / וּבַעֲבוֹדַת בַּעְלָהּ מְהִירָה



וּזְרִיזָה / וּבְאוֹר פָּנָיו מְאִירָה וּבְשִׂמְחָתוֹ עַלִּיזָה, /



וּלְהִתְבּוֹנֵן תָּמִיד בִּדְבָרוֹ וְאִמְרָתוֹ / כְּדֵי לְהָפִיק



רְצוֹנוֹ בַּעֲבוֹדָתוֹ / וְשֶׁלֹּא יִצְטָרֵךְ לִקְרֹאתָהּ פַּעֲמַיִם /



וְתִירָאֶנּוּ כְּמוֹרָא שָׁמַיִם / וְתִשְׁמֹר עֵת כַּעְסוֹ וְשִׂמְחָתוֹ



/ עֵת אַהֲבָתוֹ וְשִׂנְאָתוֹ / עֵת שִׁבְתּוֹ וְקוּמוֹ / עֵת קִיצָתוֹ



וְנוּמוֹ, / וְאַל תִּשְׂנָא אֲשֶׁר יֶאֱהַב לְמָנָה / וְאַל תֶּאֱהַב אֶת



אֲשֶׁר יִשְׂנָא, / וְאַל תֶּחֱשֹׂף סוֹדוֹתָיו / וְאַל תָּבוּז



מַעֲלוֹתָיו / תּוֹלְדוֹתָיו וּמִשְׁפְּחוֹתָיו. / וְאָמְרוּ: צְרִיכָה



לַעֲזֹב חֻקָּהּ וּמִנְהָגָהּ, / תַּכְשִׁיטֵי עֶדְיָהּ וְתַעֲנוּגָהּ, /



וְאַל תַּקְשֶׁה לִשְׁאֹל מִמֶּנּוּ / וּמַתָּנוֹ לְאַל אַל תְּשִׂימֶנּוּ,



/ וְכַאֲשֶׁר יִתֵּן לָהּ אַל תִּבְעַט / בֵּין רַב לִמְעַט, / וְתֶאֱהַב



מַאֲכָלוֹ וּמְזוֹנוֹ / כְּפִי חֶפְצוֹ וּרְצוֹנוֹ. / וּבְקִימָתָהּ



וְשִׁבְתָּהּ / תִּשְׁמֹר עִתּוֹ וְלֹא עִתָּהּ, / קַשֶּׁבֶת וְלֹא



דַבְּרָנִית / מְיֻשֶּׁבֶת וְלֹא יַצְאָנִית, / שֶׁעַל שֵׂכֶל וְדֵעָה /



חָצְבָה עַמּוּדֶיהָ שִׁבְעָה, / צוֹפִיָּה הֲלִיכוֹת בֵּיתָהּ וּמְעוֹנָהּ



/ וְתוֹרַת חֶסֶד עַל לְשׁוֹנָהּ. / וְאָמְרוּ: צְרִיכָה לְקַשֵּׁט



עַצְמָהּ דֶּרֶךְ צְנִיעוּת / כְדֵי שֶׁלֹּא יִמָּצֵא בָהּ דֹּפִי



וְטָעוּת, / וְאַל תִּהְיֶה עָלָיו הוֹמִיָּה וְסוֹרֶרֶת / וּבְיַחְסָהּ



מִשְׂתָּרֶרֶת, / וְאַל תַּרְאֶה לוֹ פָּנִים זוֹעֲפִים / וְרוּחַ קִנְאָה



שׁוֹאֲפִים, / וְיִגְדַּל בְּעֵינֶיהָ הוֹדוֹ וְגָדְלוֹ / בִּקְרָב אִישׁ



לְהִשְׁתַּחֲווֹת לוֹ. / וְכָל מוּם וְכָל אֵיבָה מִלִּבָּהּ תַּשְׁמִיד /



וְתִתְפַּלֵּל בַּעֲדוֹ תָמִיד, / וּכְבָבַת עֵינָהּ תִּנְצְרֵהוּ / כָּל



הַיּוֹם תְּבָרְכֵהוּ. / וְהָאִשָּׁה אֲשֶׁר אֵלֶּה אֳרָחֶיהָ – / עָצְמוּ



מִסַּפֵּר שְׁבָחֶיהָ, / וְעָלֶיהָ אָמְרוּ זִמְרָה, / רָחוֹק מִפְּנִינִים



מִכְרָהּ, / וְעַל שְׁלֵמוּת גָּדְלָהּ וּמַהֲלָלָהּ / בָּטַח בָּהּ לֵב



בַּעְלָהּ. / וְעַתָּה אֲכוֹנֵן לִבִּי וְרַעְיוֹנַי, / הַפַּעַם אוֹדֶה



אֶת יְיָ: / כִּי רֹב הַמִּדּוֹת אֲשֶׁר דִּבַּרְתִּי / וְהַמַּהֲלָל



אֲשְֶׁר זָכַרְתִּי – / לָךְ הֵם לְמִקְנָה / מִימֵי נְעוּרִים



וְעֶדְנָה״.

וְכִשְׁמֹעַ הָאִשָּׁה אֶת דְּבָרָיו / וְנֹעַם אֲמָרָיו /



וַתָּקָם וַתַּחֲזֵק בְּבִגְדוֹ / וַתִּשְׁתַּחֲוֶה לְנֶגְדּוֹ / וַתֹּאמַר



״זֶהוּ מַאֲמַר חֲכָמַי וּנְבוֹנַי / זֶה הַשֻּׁלְחָן אֲשֶׁר לִפְנֵי יְיָ,



/ וְהִנֵּה אִם צָדַקְתָּ כִּי בְּמַהֲלָלְךָ תַּזְכִּירֵנִי – / אֶמְצָא



חֵן בְּעֵינֶיךָ, אֲדוֹנִי, לְהַכִּירֵנִי!״ / וַיְכַל הָאִישׁ עִנְיָנוֹ /



לְדַבֵּר עַל שֻׁלְחָנוֹ, / וַיָּקוּמוּ שְׁנֵיהֶם בְּחִפָּזוֹן / אַחַר



הַמַּאֲכָל וְהַמָּזוֹן / וַיָּסַר אֶל הָעֲלִיָּה וַיִּשְׁכְּבוּ שָׁם, /



וַיִּתְעַנְּגוּ כְּכָל תַּאֲוַת נַפְשָׁם. / וַיְהִי בְאַשְׁמֹרֶת



הַבֹּקֶר וַיֵּעוֹר הָאִישׁ מִשְׁנָתוֹ – / וַיָּפֹזוּ זְרוֹעֵי



תְנוּמָתוֹ. / וַיֹּאמַר ״אִשְׁתִּי, קוּמִי, אֲחוֹתִי, / הָכִינִי



שִׂמְלָתִי וּכְסוּתִי, / כִּי כְּבָר נִתְכְּנוּ רַעְיוֹנַי: / עֵת



לַעֲשׂוֹת לַייָ! / וְעַתָּה לְכַבֵּד בֵּיתֵךְ וְלַעֲרֹךְ מַאֲכָלֵךְ



עוּרִי / בְּחִטֵּי מִנִּית וּפַנַּג וּדְבַשׁ וָשֶׁמֶן וָצֹרִי, / וְאַחַר



תָּשִׂימִי לִבֵּךְ לַמְסִלָּה / לַעֲלוֹת אֶל בֵּית הַתְּפִלָּה / וְלָתֵת



הוֹדָאָה עַל טוּב חֶלְקֵנוּ / וְאוֹר נָגְהֵנוּ, / כִּי קָדוֹשׁ הַיּוֹם



לַאֲדוֹנֵינוּ״.

וַתֹּאמֶר הָאִשָּׁה בְּנֶפֶשׁ נְכֵאָה / ״אַל תָּשֶׁת



בִּי עָווֹן וּשְׁגִיאָה / כִּי לֹא אֵצֵא מִבֵּיתִי, / פֶּן אַשְׁחִית



נַחֲלָתִי. / גְּאַל לְךָ אַתָּה אֶת גְּאֻלָּתִי / וְהִתְפַּלֵּל



בַּעֲדֵנוּ / לְהַגְדִּיל מַהֲלָלֵנוּ וְהוֹדֵנוּ, / כִּי בְנוֹת הָעִיר



הַזֹּאת / מְאֹד הֵן קָשׁוֹת וְעַזּוֹת: / כִּי כָל אַחַת מֵהֶן



לְתַאֲוָתָהּ שׁוֹחֶרֶת / וְעַל הוֹד בַּעְלָהּ אֲמָרֶיהָ מְמָרֶרֶת /



הוֹמִיָּה וְסוֹרֶרֶת, / מִתְעַסֶּקֶת בְּדִבְרֵי כְלִמָּה / פְּתַיּוּת



וּבַל יָדְעָה מָּה. / וְכָל אַחַת בְּדִבְרֵי חֲבֶרְתָּהּ הֶחֱזִיקָה /



נָכְרִיָּה – אֲמָרֶיהָ הֶחֱלִיקָה. / וְאֵין בֵּינֵיהֶן נְבוֹנָה / כִּי



אִם פּוֹתָה וְדִבְרֵי עֲגָבִים מוֹנָה, / שִׁית זוֹנָה. / וְלֹא



תִתְחַבַּרְנָה בְּבָתֵי הַתְּפִלּוֹת / לִשְׁמֹעַ תַּחֲנוּנִים



וּתְהִלּוֹת, / כִּי אִם לִפְנֵי הַבַּחוּרִים לְהִתְנָאוֹת / לַעֲלוֹת



וְלֵרָאוֹת / וְהֵם לִרְאוֹתָן יֶאֱרֹבוּ / וּלְמַלֹּאת פְּגִימָתָן



יַחֲשֹׁבוּ / וְיִרְמְזוּ בְעֵינַיִם וּבְגַבּוֹתָם / יִצְפְּנוּ



לְנַפְשׁוֹתָם. / וְכַאֲשֶׁר שָׁמַעְתִּי סוֹדֵיהֶם / וָאֶתְבּוֹנֵן



עָדֵיהֶם, / אָמַרְתִּי: חוֹבָה לְהִתְרַחֵק מִכָּל לֵב הֵזִיד וְשָׁגַג –



/ טוֹב לָשֶׁבֶת עַל פִּנַּת גָּג! / וְנִשְׁבַּעְתִּי, אֲדוֹנִי,



בְּחַיֵּי רֹאשְׁךָ / חַי יְיָ וְחֵי נַפְשֶׁךָ, / אִם אֵצֵא מִפֶּתַח



בֵּיתִי / וְעָנְתָה בִּי צִדְקָתִי! / כָּל שֶׁכֵּן שֶׁאֵין הָאִשָּׁה



נֶחְשֶׁבֶת / וּבְמִצְוַת הַיּוֹם אַינָהּ חַיֶּבֶת / וְכֵן פֵּרְשׁוּ



בַעֲלֵי הַמְּסוֹרוֹת: / כָּל מִצְוַת עֲשֵׂה שֶׁהַזְּמָן גְּרָמָהּ –



נָשִׁים פְּטוּרוֹת״. / וַיֹּאמֶר הָאִישׁ ״בָּרוּךְ יְיָ אֲשֶׁר נָתַן



כָּזֹאת בְּלִבֵּךְ / וְהֵבִיא רוּחַ חֵן וְתַחֲנוּנִים בְּקִרְבֵּךְ /



וְתָשִׂימִי לְאַל דִּבְרֵי כְסִילוֹת וְנִבְזוֹת: / לְזֹאת יִקָּרֵא



אִשָּׁה כִּי מֵאִישׁ לֻקֳחָה זֹּאת! / וְעַתָּה שְׁבִי בֵית



תִּפְאַרְתֵּךְ וְהוֹדֵךְ – / וַאֲנִי אֶתְפַּלֵּל בַּעֲדֵךְ״. / וַיֵּלֶךְ



הָאִישׁ לְדַרְכּוֹ / לְהִתְפַּלֵּל לִפְנֵי מַלְכּוֹ.

וַיְהִי כַּאֲשֶׁר יָצָא מִבֵּית מְנוּחוֹ – / וְהַבָּחוּר



הִשְׁכִּים אֶל פִּתְחוֹ / וַיִּרְמֹז אֶל הָאִשָּׁה בְּמַטֵּהוּ, /



וַתִּפְתַּח וַתִּרְאֵהוּ / וַתֹּאמֶר לוֹ בְּקוֹל נָחוּץ / ״בּוֹא,



בְּרוּךְ יְיָ, לָמָּה תַעֲמֹד בַּחוּץ? / כִּי אֵין הָאִישׁ בְּבֵיתוֹ



וּבִזְבוּלוֹ / חָלַף וְהָלַךְ לוֹ״. / וַיָבוֹא וַיִּתְהַלֵּל



בַּעֲלוּמָיו / וַיִּגְרְשׁוּ מֵימָיו. / וַיֹּאכְלוּ וַיִּשְׁתּוּ כְּחֹק



הַחוֹשְׁקִים הַנֵּחָמִים / וַיִּצְנְחוּ לָאָרֶץ שֶׁבַע פְּעָמִים /



וַיִּשְׂבְּעוּ מַעֲדַנִּים לְפִי תַאֲוָתָם / הֵרֵעוּ מֵאֲבוֹתָם. /



וַיְהִי כַּאֲשֶׁר כִּלּוּ לֶאֱכֹל אֶת הַשֶּׁבֶר – / אֵשֶׁת מִדְיָנִים



וּבֵית חֶבֶר! – / וַיִּפֶן הַבָּחוּר אֶל דֶּרֶךְ גְּבוּלוֹ / וַיֵּלֶךְ



לוֹ. / וַתְּמַהֵר הָאִשָּׁה לְהָכִין מַאֲכָלָהּ וּמְזוֹנָהּ /



וַתִּתְעַטֵּף וַתֵּשֶׁב עַל מְכוֹנָהּ.

וַיָּבוֹא אִישָׁהּ מִבֵּית מִדְרָשׁוֹ / הָדוּר



בִּלְבוּשׁוֹ, / וַתָּקָם לְפָנָיו בְּשִׁפְלוּת עֵינַיִם / וַתִּקֹּד בַּת



שֶׁבַע אַפַּיִם / וַתְּסִירֵהוּ אֶל מְקוֹמוֹ וַהֲדָרוֹ / וְאֶל מוֹשַׁב



יְקָרוֹ. / וַיָּבוֹאוּ אוֹהֲבָיו וְכָל בְּנֵי מִשְׁפַּחְתּוֹ /



וַיִּפְקְדוּ אוֹתוֹ וְאֶת אִשְׁתּוֹ / וַיֹּאמְרוּ ״הֲשָׁלוֹם לָכֶם מֵאֵל



נוֹרָא וְעֵילוֹם?״ / וַיֹּאמְרוּ ״שָׁלוֹם!״ / וַיְהִי אַחֲרֵי כֵן



וַיֵּשֶׁב הָאִישׁ אֶל מְזוֹנוֹ / וְהָאִשָּׁה כְּכֶתֶם אוֹפִיר לִימִינוֹ,



/ וַיִּשְׂמְחוּ בְחַגָּם וּבְמוֹעֲדָם / כִּי זֶה כָּל הָאָדָם – / חֹק



וּזְמָן נִתַּן לָהֶם שֶׁלֹּא יְשַׁנּוּ אֶת תַּפְקִידָם.





וַיִּרְבּוּ הַיָּמִים / בַּחֲטָאִים וַאֲשָׁמִים / וְלֹא



הֵבִין הָאִישׁ עִנְיְנֵי אִשְׁתּוֹ וּמַעְבָּדֶיהָ, / כִּי מְצוֹדִים



וַחֲרָמִים – לִבָּהּ, אֲסוּרִים – יָדֶיהָ. / וַיְהִי לִתְקוּפַת



הַשָּׁנָה / בְּהִתְעוֹרֵר לֵב כָּל זוֹנָה / וְכָל מַמְרִים / וּלְמַלֵּא



תַאֲוָתוֹ יַעְרִים – / וַיִּתְאַו הַבָּחוּר לְצַחֵק / וַיֵּלֶךְ מֵאָבִיו



וַיִּתְרַחֵק / וַיֵּצֵא בֵית הַשְּׂחוֹק / כְּמִשְׁפַּט כָּל זוֹנֶה



וּכְחֹק. / וַתִּפֹּל עָלָיו גָּדֵר הַדְּחוּיָה – / וַיְצַחֵק בְּקֻבְיָא



/ וַיְחַסֵּר הוֹנוֹ וּכְסוּתוֹ גָרַע / וְאָבַד הָעֹשֶׁר הַהוּא



בְּעִנְיַן רָע. / וַיִּתְחַבֵּא כָל הַיּוֹם הַהוּא לִבְרֹחַ /



וַיִּשְׁכַּח בֵּית אָבִיו שָׁכוֹחַ / וַיִּשְׂכֹּר לוֹ בְהֵמָה לִרְכַּב /



וְלָן וְשָׁכַב.

וַיְהִי בַלַּיְלָה בְּהַגִּיעַ עֵת תְּפִלָּתָם וְעוֹנָה



/ רֹאשׁ הָאַשְׁמֹרֶת הַתִּיכוֹנָה / וַיֵּעוֹר הָאִישׁ מִשְּׁנָתוֹ /



וַיֵּלֶךְ לְהִתְוַדּוֹת עַל עֲווֹן אַשְׁמָתוֹ. / וַיַּשְׁכֵּם הַבָּחוּר



לָלֶכֶת לְדַרְכּוֹ / וְלָנוּס מֵעִירוֹ וּמִקְּהִלַּת פִּלְכּוֹ, /



וַיִּקְרָא אֶל הָאִשָּׁה אֶל הַפֶּתַח / לִפְתֹּחַ פֶּה בְּרֶצַח /



וַיֹּאמֶר לָהּ ״הִנְנִי הוֹלֵךְ מִבֵּיתִי וְאוּלַמִּי / אֶל מְלוֹן



אוֹרְחִים וְאֶעֶזְבָה אֶת עַמִּי״. / וַתֹּאמֶר לוֹ הָאִשָּׁה ״אַל



תִּמְנָעֵנִי, חֲשׁוּקִי, מֵעַצְמְךָ – / הָלוֹךְ אֵלֵךְ עִמְּךָ!״ /



וַתִּקַּח מִבֵּית בַּעְלָהּ כָּל אֲשֶׁר תֶּאֱהַב: / כֶּסֶף וְזָהָב, /



בְּגָדִים וּרְדִידִים, / נְזָמִים וּצְמִידִים. / וַיָּבוֹא הַבָּחוּר אֶל



הֵיכָלָהּ / וַיִּקַּח אֶת רְכוּשׁ בַּעְלָהּ, / עֲדָיִים וּשְׁבִיסִים, /



תִּפְאֶרֶת הָעֲכָסִים / כְּלִילָם וְאֶצְעָדָתָם וַעֲגִילָם / עֲשָׂרָה



זָהָב מִשְׁקָלָם. / וַיִּשְׂכְּרוּ לָהֶם בְּהֵמָה אַחֶרֶת / חֲזָקָה



וְנִבְחֶרֶת, / וַיִּקַּח כָּל אֶחָד מַשָּׂאוֹ וּמִבְחָרוֹ / וַיַּעֲמֹס



אִישׁ עַל חֲמוֹרוֹ, / וְלֹא הִנִּיחַ לְבַד הַכָּבֵד / וְכָל דָּבָר



הָאָבֵד / אֲשֶׁר לֹא הָיָה עוֹמֵס / וְכָל הַמְּלָאכָה נְמִבְזָה וְנָמֵס.



/ וַיָּקוּמוּ וַיֵּלְכוּ בַנֶּשֶׁף / מְזֵי רָעָב וּלְחוּמֵי רֶשֶׁף, /



וַיָּשִׂימוּ מְגַמַּת פְּנֵיהֶם אֶל אַדְמַת מִצְרַיִם לְחָגָּא / דֶּרֶךְ



אַשְׁדּוֹת הַפִּסְגָּה, / וַיָּסֹבּוּ אֶת כָּל עָרֵי הָאֱמוֹרִי



וּמִגְרָשָׁם / וַיָּבוֹאוּ עַד חָרְמָה וַיֵּשְׁבוּ שָׁם.





וַיְהִי בַבֹּקֶר עֵת שִׂמְחַת הַלְּבָבוֹת / בֹּקֶר לֹא



עָבוֹת – / וַיְכַל הָאִישׁ לְהִתְוַדּוֹת עַל מַעֲלוֹ – / דֹּם לַייָ



וְהִתְחוֹלֵל לוֹ – / וַיָּרָץ אֶל בֵּיתוֹ בִּמְרוּצָה / וּפֶתַח פָּתוּחַ



מָצָא. / וַיַּרְא – וְהִנֵּה בֵיתוֹ שָׁדוּד, אֵין בּוֹ מִחְיָה /



וְעֵירֹם וְעֶרְיָה. / וַיִּכָּנֵס אֶל גְּנָזָיו – וְהִנָּם נְסוֹגִים /



כַּמְּצוּלָה שֶׁאֵין בָּהּ דָּגִים. / וַיֹּאמַר: אֵין זֶה כִּי אִם



אִשְׁתִּי הַתְּמִימָה / הִשְׁאִילָה מֵיטַב חֲפָצֵינוּ לְכַלָּה יְתוֹמָה



/ כְּדֵי לִגְמֹל חֲסָדִים / בְּעֵת דּוֹדִים / (וְלֹא יָדַע הֶחָסִיד



הֶעָנָו / כִּי קָרַן עוֹר פָּנָיו!) / וְעַתָּה אֲיַחֵל עַד עֵת בּוֹאָהּ



/ וְאֵדְעָה מוֹצָאָהּ. / וַיְיַחֵל הָאִישׁ לִמְצֹא פִדְיוֹם / עַד נְכוֹן



הַיּוֹם / וַיַּרְא וְהִנָּהּ אֵחֲרָה לָבוֹא / וַיּתְעַצֵּב אֶל לִבּוֹ. /



וַיֹּאמַר: אֵלֵךְ לְהָבִין מִי הֶלְאַנִי / וְאֵדְעָה מֶה חָדֵל אָנִי. /



וַיָּקָם וַיִּדְרֹשׁ בַּעֲדָהּ בְּכָל מוֹשָׁב / נִכְבָּד לַכֹּל



וְנֶחְשָׁב – / וְלֹא מָצָא לִשְׁאֵלָתוֹ מַעֲנֶה / וְאֵין קוֹל וְאֵין



עוֹנֶה. / וַיָּבוֹא עַד לִפְנֵי שַׁעַר הָעִיר / שׁוֹאֵל לְכָל יָשִׁישׁ



וְצָעִיר, / וְהִנֵּה אִישׁ מִדֶּרֶךְ הַקָּדִים / בָּא וְהִקְדִּים /



וַיַּרְא אֶת הָאִישׁ וַיַּכִּירֵהוּ – / וַיִּקֶר מִקְרֵהוּ.

וַיֹּאמַר ״הֲלֹא אַתָּה הָאִישׁ בַּעַל הַתְּשׁוּבָה /



הוֹלֵךְ בְּדֶרֶךְ חֲשׁוּבָה, / וְאִשְׁתְּךָ הָיְתָה בְּעֵינַי בְּחֶזְקַת



הַצְּנוּעוֹת – / וְהִנֵּה הִיא מִן הַזּוֹנוֹת הַגְּרוּעוֹת! / כִּי



הַיּוֹם פְּגַשְׁתִּיהָ עִם פְּלוֹנִי הַבָּחוּר / רָצִים בְּאֵין אִחוּר,



/ וַחֲמוֹרֵיהֶם נוֹשְׂאִים מַשָּׂא כִפְלַיִם / הוֹלְכִים לְהוֹרִיד



מִצְרַיִם. / וְאֵין אִישׁ יָכוֹל לְהַשִּׂיג אוֹתָם / לְתֹקֶף מַהֲלָכָם



וּמְרוּצָתָם״. / וַיְהִי כִשְׁמֹעַ הֶחָסִיד עִנְיָנָיו, / כִּסְּתָה



כְלִמָּה פָּנָיו / וַיִּתֵּן אֶת קוֹלוֹ בִּבְכִי וַיִּתְנַהֵג כְּשׁוֹלָל



/ וְנָאַק נַאֲקַת חָלָל. / וַיִּצְעַק וַיֹּאמַר ״הֵן בַּעֲווֹן צִדְקִי



נִטְמֵאתִי, / אוֹי לִי כִי נִדְמֵיתִי! / וְסִבַּת בָּשְׁתִּי



וּכְלִמָּתִי / וִיגוֹן לִבִּי וְחֶרְפָּתִי / הָיְתָה לִי דִמְעָתִי, /



וְעִיּוּן תְּפִלָּתִי וּתְשׁוּבָתִי – / הָיְתָה מַחֲלָתִי. / כִּי אִם



הִתְנַהַגְתִּי בְּגַאֲוָה וּבְזָדוֹן / וְהָיִיתִי בְּבֵיתִי אִישׁ מָדוֹן



– / הָיִיתִי מַכִּיר אֶת אִשְׁתִּי וְאוֹדוֹתֶיהָ / מַלְקָחֶיהָ



וּמַחְתּוֹתֶיהָ / וּזְדוֹנִי בְּכָל יוֹם יֵדָעֶנָּה / וְגַאֲוָתִי בְּכָל



עֵת תִּדְרְשֶׁנָּה / וַתַּשְׁחֵת עַגְבָתָהּ מִמֶּנָּה. / אֲבָל



פִּתַּתְנִי הַמְּשׁוּבָה וָאֶתְפַּתֶה / וָאֶמְצָא כוֹס חֲמַת אִשְׁתִּי



וָאֶשְׁתֶּה, / וַיִּכְלוּ בַהֶבֶל יָמַי / וַיָּבוֹא רָקָב בַּעֲצָמַי. /



וְעַתָּה, שָׁבִים, סוּרוּ מִדֶּרֶךְ תְּשׁוּבָה / וּבַחֲרוּ לָכֶם דֶּרֶךְ



עַזּוֹת וּמְשׁוּבָה, / פֶּן יִקְרֶה אֶתְכֶם כַּאֲשֶׁר קְרָאַנִי /



וְיִשְׁבֹּר הַדְרַתְכֶם הַיָּגוֹן אֲשֶׁר שְׁבָרַנִי / וְתִהְיוּ לִמְשֹׁל



עַמִּים וְקִינָה / לְחֶרְפָּה לְמָשָׁל וְלִשְׁנִינָה״.

וַיָשָׁב הָאִישׁ לְבֵיתוֹ / כַּאֲשֶׁר כִּלָּה לְסַפֵּר



אַנְחָתוֹ / וַיִּשָּׁאֵר לְחֶרְפָּה וּלְבִזָּיוֹן / לְדַאֲבוֹן נֶפֶשׁ



וְכִלָּיוֹן. / וַיִּתֵּן תַּחַת שִׂמְחָתוֹ – אֲנָחָה / וּמְנוֹד רֹאשׁ –



תַּחַת הֲנָחָה / וְתַחַת מַעֲשֵׂה מִקְשֶׁה – קָרְחָה / וְתַחַת כָּבוֹד –



מוּם וָדֹפִי, / כִּי – תַחַת יֹפִי.
