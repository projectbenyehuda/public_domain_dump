


	
	הָיָה אִישׁ אֶחָד בְּאֶרֶץ הַתֵּימָנִי
	








מתוך השער השני [השודד והנזיר]





הָיָה אִישׁ אֶחָד בְּאֶרֶץ הַתֵּימָנִי / מִמִּשְׁפַּחַת



הַדָּנִי / וּשְׁמוֹ חֶבֶר הָאֲחוֹחִי / בֶּן-בִּלְדַּד הַשּׁוּחִי, /



וְלֹא לָמַד תּוֹרָה בְּקַטְנוּתוֹ / וְאֶל חֹק וּמִשְׁפָּט לֹא הִטָּה



אֱמוּנָתוֹ, / שָׁלַח יָדָיו בִּשְׁלוֹמָיו חִלֵּל בְּרִיתוֹ / וְהָיָה



מְזַלְזֵל אָבִיו וְאֶת בְּנֵי עַמּוֹ, / וּכְסִיל אָדָם בּוֹזֶה אִמּוֹ, /



וַיִּתְחַבֵּר אֶל בְּנֵי בְלִיַּעַל וַּיִּמְעֹל מַעַל. / וְכַאֲשֶׁר



רָאָה אָבִיו אָרְחוֹתָיו / מַהֲלָכָיו וּנְתִיבוֹתָיו, / הִדִּיחוֹ



מִבֵּיתוֹ לְרֹעַ מַעֲשָׂיו / וְנִדָּהוּ וְהוֹרִידוֹ מִנְּכָסָיו. /



וַיֵּצֵא הָאִישׁ מִבֵּית הָאִישׁ הַנֶּאֱמָן / אֶל מִדְבַּר צִן נֶגְבָּה



מִקְצֵה תֵימָן, / וַיְהִי לְכָל עוֹבֵר וָשָׁב מַשָּׂא וָטֹרַח / נָחָשׁ



עֲלֵי דֶרֶךְ שְׁפִיפוֹן עֲלֵי אֹרַח.

וַיְהִי הַיּוֹם וְהִנֵּה נָזִיר אֶחָד עוֹבֵר / לְיִרְאֵי



אֱלֹהִים רֵעַ וְחָבֵר, / וַיִּתְפֹּשׂ בּוֹ הָאִישׁ לְבַהֲלוֹ / וְלִגְזֹל



הוֹנוֹ וְחֵילוֹ – / וְלֹא מָצָא בְיָדוֹ דָבָר, / כַּאֲשֶׁר חָשַׁב



וְסָבַר. / וַיְהִי כִרְאוֹתוֹ כִּי הָאִישׁ הַתָּמִים / אֵין לוֹ דָּמִים,



/ הֵחֵל לְהַלְאוֹתוֹ וּלְהַכְאִיבוֹ / וּלְעַנּוֹתוֹ וּלְהַרְעִיבוֹ. /



וְכִרְאוֹת הַנָּזִיר צָרָתוֹ וְחִילוֹ, / וַיֵּבְךְּ וַיִּתְחַנֶּן לוֹ /



וַיֹּאמַר ״לָמָּה תַעֲשֶׂה כֵן לְעַבְדֶּךָ / וְלָמָה אָמוּת נֶגְדֶּךָ? /



לֵךְ עִמִּי, אַל תֶּחְדַּל חָדוֹל – / וַאֲנִי אַעְשֶׁרְךָ עֹשֶׁר גָּדוֹל



/ וְאֶתֵּן לְךָ כֶּסֶף וְזָהָב / וְכֹל אֲשֶׁר תֶּאֱהַב״. / וַיֹּאמֶר לוֹ



״אִם כֵּן / שָׁאַתָּה עָשִׁיר וְלֹא מִסְכֵּן, / הוֹדִיעֵנִי נָא אֶת



דְּרָכֶיךָ / וּבַמֶּה נֶעְשַׁרְתָּ וּמַה מְּלַאכְתֶּךָ / וּמֵאַיִן



תָּבוֹא וּבְאֵיזֶה מָקוֹם יָשַׁבְתָּ / וּמָה אַרְצְךָ וְאֵי מִזֶּה עַם



אַתָּה״.

וַיֹּאמַר ״אֲנִי הָיִיתִי מִתְעַסֵּק בִּסְחוֹרָה /



בְּצוֹר הַמַּעֲטִירָה, / וָאֱהִי רוֹכֵל בְּכָל מִינֵי תְמוּרָה /



בְּרֹאשׁ כָּל בֹּשֶׂם וּבְאֶבֶן יְקָרָה. / וַתִּגְדַּל מְאֹד יִפְעָתִי



וּתְבוּנַת מַחְשְׁבוֹתַי / וַתַּעַשׂ כֶּסֶף וְזָהָב בְּאוֹצְרוֹתַי. /



וַיִּבְעַט לִבִּי בְּיוֹצֵר אוֹר וּבוֹרֵא שַׁחַר / מֵרֹב כָּל הוֹן



בְּיֵין חֶלְבּוֹן וְצֶמֶר צַחַר, / וַיִּתְגַּבֵּר יִצְרִי עַל נַפְשִׁי



הַהֲדוּרָה / וַיְסִירֶהָ מִגְּבִירָה, / וַיֵּלֶךְ לוֹ לְבַקֵּשׁ גַּאֲוָה



/ אֶל קִבְרוֹת הַתַּאֲוָה. / וַיְהִי יוֹם אֶחָד בְּפָקְדִי עֲווֹנַי /



הוּא יוֹם יִוָּדַע לַייָ – / וְהִנֵּה כְאֵב גָּדוֹל – עַד לָמוּת



שָׁאֲלָה נַפְשִׁי / וַיִּקָּחֵנִי בְּצִיצִית רֹאשִׁי. / וָאֶשְׁלַח אֶל



הָרוֹפְאִים לִדְרֹשׁ / לְרַפּאוֹת כְּאֵב הָרֹאשׁ, / וָאַרְבֶּה לָהֶם



מֹהַר וּמַתָּן / בִּגְדֵי מֶשִׁי וְרִקְמָה וּשְׂרִיקוֹת פִּשְׁתָּן. /



וַיִּשְׁתַּדְּלוּ לַעֲשׂוֹת לִי רְפוּאָתָם, / אִישׁ אִישׁ כְּפִי



עֲבוֹדָתָם, / וַיְצַוּוּ עָלַי ״מִמַּאֲמָרֵינוּ אַל תִּגְרַע /



וְנִשְׁמַרְתָּ מִכָּל דָּבָר רַע – / וְהוּא כָל מַאֲכָל קָשֶׁה וְכָבֵד /



אֲשֶׁר לֹא יֵעָבֵד – / בְּקָנֶה וְקִדָּה לְהַשְׁלִים מַעֲלָתוֹ /



וְקִנְמָן-בֶּשֶׂם מַחֲצִיתוֹ״ / לְהִמָּלֵט מִדֶּבֶר וֶאָסוֹן / לִמְשֹׁחַ



רֹאשִׁי בְּשֶׁמֶן שָׂשׂוֹן. / וָאֶעֱשֶׂה כְּפִי מִצְוָתָם וְלֹא



אֵחַרְתִּי, / וּבְכָל אֵלֶּה נִשְׁמַרְתִּי וְנִמְשַׁחְתִּי – / וְלֹא



שָׁלַוְתִּי וְלֹא שָׁקַטְתִּי וְלֹא נַחְתִּי. / וָאֶשְׁלַח בְּעַד רוֹפֵא



אֶחָד מִגְּדוֹלֵיהֶם / שַׂר צִלָּם מֵעֲלֵיהֶם, / וְהִנֵּה הוּא בָקִי



בְכָל עִנְיָן / גָּדוֹל בְּחָכְמָה וּבְמִנְיָן. / וָאֹמְרָה אֵלָיו ״עַל



כָּל אֲשֶׁר הֵבִינוּ וְהִשְכִּילוּ – / כִּי לֹא יוֹעִילוּ וְלֹא



יַצִּילוּ. / וּמִיּוֹם עָשִׂיתִי מַאֲמָרָם – כִּמְעַט לֹא אָבַדְתִּי



אָבוֹד / כִּי הַכְּאֵב גָּדוֹל מְאֹד״. / וַיִּחַר אַף הָרוֹפֵא



וַיְרִיבֵנִי, / אַפּוֹ טָרָף וַיִּשְׂטְמֵנִי, / וַיֹּאמַר ״כָּל



הַמִּשְׁתַּדֵּל בִּרְפוּאַת פֶּתִי וְסָכָל / הַשְׁקֵט לֹא יוּכַל, / כִּי



לֹא יָבִין תַּבְנִית הַמַּרְפֵּא בְּעִנְיָנוֹ – / לֹא יָדַע שָׁלֵו



בְּבִטְנוֹ. / וְהִנְנִי אוֹדִיעֲךָ שֹׁרֶשׁ הָרְפוּאָה וְעִקָּרֶיהָ / אַף



הִיא תָּשִׁיב אֲמָרֶיהָ…

וַיְהִי בְשָׁמְעִי דִבְרֵי הָרוֹפֵא וְחָכְמָתוֹ /



וְהִתְבּוֹנְנִי בְּהַנְהָגָתוֹ וּתְבוּנָתוֹ, / וָאֶדֹּר עַל פְּשָׁעַי



וַעֲווֹנַי / נֶדֶר נָזִיר לְהַזִּיר לַייָ, / וָאֲחַפֵּשׂ מַעֲשַׂי /



וְהִקְדַּשְׁתִּי כָל נְכָסַי. / וַיַּעַשׂ לִי הָרוֹפֵא רְפוּאוֹת /



לְמוֹפֵת וּלְאוֹת / וַיִּשְׁלַח אֱלֹהִים מַרְגִּיעַ / מַלְאָךְ מֵלִיץ



וּמוֹשִׁיעַ, / וַיִּתְחַזֵּק הַטֶּבַע בְּלִבִּי / וַיֵּלֶךְ הַמִּקְרֶה



מִקִּרְבִּי / וַיָּנָס הַכְּאֵב וְהַנֵּזֶק מֵעַל נַפְשִׁי, / וָאָסִיר



עֲטֶרֶת רֹאשִׁי. / וָאָקוּם וָאֶבְנֶה בַיִת מְקֻדָּשׁ / לֵאלֹהִים מְעַט



מִקְדָּשׁ, / וָאָכִין אוֹתוֹ מִקֶּדֶם לָעִיר / לִקְרֹא לְכָל עוֹבֵר



וָשָׁב וּלְהָעִיר. / וָאֲחַבֵּר אֵלַי חֲבֵרִים וְתַלְמִידִים / רַבִּים



וְנִכְבָּדִים, / וָאֶמְאַס סַלּוֹנִים וְסָרָבִים / וָאַרְבִּיץ תּוֹרָה



בָּרַבִּים. / וָאֶתֵּן לְכָל אִישׁ כִּפְרִי מַעֲלָלָיו / בַּבַּיִת



אֲשֶׁר נִקְרָא שְׁמִי עָלָיו, / וָאַרְבֶּה בְתוֹכוֹ עֹשֶׁר וְכָבוֹד /



וְכֶסֶף וְזָהָב הַרְבֵּה מְאֹד, / וְהַכֹּל מוּכָן לַעֲנִיִּים /



וּלְפִדְיוֹן שְׁבוּיִים.

וַיְהִי הַיּוֹם וַאֲנִי יוֹשֵׁב עַל מְכוֹנִי /



שָׁמַעְתִּי וַתִּרְגַּז בִּטְנִי – / וְהִנֵּה אִישׁ מִבְּנֵי עַמֵּנוּ



נֶחְבָּא / וּבֵין הַגּוֹיִים נִשְׁבָּה. / וָאֵלֵךְ הַיּוֹם לִפְדּוֹתוֹ /



וּלְהָשִׁיב אֶת שְׁבוּתוֹ / וְהִנֵּה שַׁבְתִּי לְדַרְכִּי / לָלֶכֶת



לְבֵיתִי וְאֶל קְהִלַּת פִּלְכִּי. / וְעַתָּה מַהֵר לֵךְ עִמִּי / אֶל



בֵּיתִי וְאוּלַמִּי / וְאַרְבֶּה לְךָ מֹהַר וּמַתָּן לְהַשְׁכִּיחַ



רִישֶׁךָ / כְּכֹל אֲשֶׁר תִּשְׁאַל נַפְשְׁךָ״. / וַיֹּאמֶר הָאִישׁ



״אֵיךְ אֵלֶךְ? אוּלַי יִתְחַבְּרוּ תַלְמִידֶיךָ / וְאַתָּה וַעֲבָדֶיךָ /



וְיוֹדִיעוּ לַמֶּלֶךְ דְּבַר אִישׁ כָּמוֹנִי / וְיֹאחֲזוּנִי יְמֵי



עֹנִי. / וְעַתָּה הִשָּׁבְעָה לִי בְּצוּר מָעֻזִּי וְקוֹנִי / אִם



תִּשְׁקֹר לִי וּלְנִינִי!״ / וַיֹּאמֶר הַנָּזִיר ״אָנֹכִי אִשָּׁבַע, /



שָׂפָה לְפִי סָבִיב לֹא יִקָּרֵעַ״.





וַיָּקוּמוּ מִשָּׁם הָאֲנָשִׁים / וַיֵּלְכוּ הָלוֹךְ



וְהֵמָּה מַחְשִׁים / וַיָּגִיעוּ אֶל בֵּית הַוַּעַד / בְּאֵימָה וָרַעַד



/ וְכָל הַתַּלְמִידִים יוֹשְׁבִים בִּמְסִבָּם / יָסֹבוּ עַל דַרְכָּם



וּנְתִיבָם. / וַיִּשְׁתַּחֲווּ לוֹ וַיִּשְׂמְחוּ לִקְרָאתוֹ /



וַיֵּשְׁבוּ לְפָנָיו הַבְּכוֹר כִּבְכוֹרָתוֹ. / וַיִּשְׁאַל הַנָּזִיר



בִּשְׁלוֹמָם וּבִשְׁלוֹם תּוֹרָתָם / וּשְׁלוֹם תַּלְמוּדָם וּמִשְׁנָתָם.



/ וַיֹּאמְרוּ ״שָׁלוֹם – תְּהִלָּה לְשׁוֹכֵן מְעוֹנַי! – / בָּרוּךְ



אַתָּה לַייָ. / אָמְנָם נִסְתַּפֵּק עָלֵינוּ הֲלָכָה וְנִשְׁאֲרָה



לְפָרֵשׁ / אַרְבָּעָה חֲלוּקֵי כַפָּרָה שֶׁהָיָה רַבִּי יִשְׁמָעֵאל



דּוֹרֵשׁ״. / אָמַר לָהֶם ״שְׁלֹשָׁה הֵם / וּתְשׁוּבָה עַל כָּל אֶחָד
מֵהֶם״…

וַיִּשְׂמְחוּ הַתַּלְמִידִים בְּמִדְרָשׁוֹ / וְשׂוֹם



שֶׂכֶל וְהָבֵן בְּפֵרוּשׁוֹ / וַיֹּאמְרוּ לוֹ ״שָׁלוֹם לְךָ וְשָׁלוֹם

וְהוֹלֲלִים חִזְּקוּ לִשְׁפֹּט שְׁפָטַיִךְ / אַלּוֹנִים מִבָּשָׁן עָשׂוּ



מִשּׁוֹטַיִךְ! / וְעַתָּה, נַפְשִׁי, לְכִי וְנֵלְכָה בְּאוֹר אֱלֹהֵינוּ

/ וּבַתּוֹרָה מַחֲזִיקִים, / וְנִתְחַבֵּר לָהֶם בְּכָל מִינֵי חִבּוּר /



לָדַעַת וְלָבוּר – / וְאִם אֵינוֹ יָכוֹל לָלֶכֶת / מִפְּנֵי מַחֲלָה



וְאָלָה נִתֶּכֶת / אוֹ מִפְּנֵי אֻמָּה אַדִּירָה אוֹ מַמְלֶכֶת / לַהַט



הַחֶרֶב הַמִּתְהַפֶּכֶת – / יִתְנַהֵג יְחִידִי וְיִתְבּוֹנֵן



בְּמִפְעָלָיו / יֵשֵׁב בָּדָד וְיִדֹּם כִּי נָטַל עָלָיו. / וְאִם



לְבַדּוֹ לֹא יַנִּיחוּהוּ – יֵצֵא לְמִדְבָּרוֹת וּלְצִיּוֹת / לְהָרִים



וְלִגְבָעוֹת / לַאֲפִיקִים וּלְגֵיאָיוֹת, / וְיִתְנַהֵג בַּעֲנָוָה / עַל



פִּי הַמִּצְוָה / וְיִסְתַּכֵּל בְּמַעֲשֵׂה הַנִּשָּׂא וְהַגָּבֹהַּ /



וּמִבְּשָׂרוֹ יֶחֱזֶה אֱלוֹהַּ. / אֵלֶּה הֵם רָאשֵׁי מִדּוֹת הַשָּׁב



וּכְלָלָם – / וְתַלְמוּד תּוֹרָה כְּנֶגֶד כֻּלָּם!״





וַיֹּאמֶר הָאִישׁ ״אַחֲרֵי אֲשֶׁר מִצּוּף דְּבָרֶיךָ



הִשְׂבַּעְתַּנִי / וּמַעֲלוֹת מִדּוֹת טוֹבוֹת הוֹדַעְתַנִי, / הִנְנִי



מְקַיֵּם אִמְרוֹתֶיךָ הַצְּרוּפוֹת / בְּסֵבֶר פָּנִים יָפוֹת / וְכָל



יְמֵי חַיַּי אֶעֱבֹר לְפָנֶיךָ / כֹּה יַעֲשֶׂה לִי אֱלֹהִים וְכֹה



יוֹסִיף, כִּי הַמָּוֶת יַפְרִיד בֵּינִי וּבֵינֶךָ!״

וַיִּקָּחֵהוּ וַיְבִיאֵהוּ אֶל בֵּית רְפִידָתוֹ



הַנֶּאֱהָב / אֲשֶׁר שָׁם הַזָּהָב / וַיִּתֶּן לוֹ כִּכַּר כֶסֶף



בְּמַתָּנָה / וַחֲמֵשׁ חֲלִיפוֹת שְׂמָלוֹת לוֹ לְמָנָּה / וּמִזְּהַב



שְׁבָא לִטְרָא אַחַת מְזֻקֶּקֶת / טוּר אֹדֶם פִּטְדָה וּבָרֶקֶת. /



וַיֹּאמֶר לוֹ ״לֵךְ וְהַעֲבֵר קוֹל בְּכָל מְדִינוֹת הַנּוֹשָׁבוֹת /



בַּשְּׁוָקִים וּבָרְחוֹבוֹת / וְשַׁלֵּם לְכָל עָשׁוּק מָמוֹן וְאִמְרֵי



שֶׁפֶר / אֶת מִי עַשַׁקְתָּ אֶת מִי רַצּוֹתָ וּמִיַּד מִי לָקַחְתָּ



כֹפֶר, / וְאַל יִרַע לְבָבְךָ בְּתִתְּךָ לוֹ מְמוֹנו וְכֵלָיו /



וַחֲמִישִׁיתוֹ הוֹסֵף עָלָיו״.

וַיַּעֲבֹר הָאִישׁ וַיַּעַשׂ כְּכָל אֲשֶׁר צֻוָּה, /



כִּי מִדְרָשׁוֹ לְמוֹשָׁב לוֹ אִוָּה, / וַיְדַבֵּר עַל כָּל לֵב עָשׁוּק



וְגָזוּל עַד אֲשֶׁר נִתְרַצָּה, לְהָשִּׁיב אֶת הַגְּזֵלָה אֲשֶׁר גָּזַל



אוֹ אֶת הָעֹשֶׁק אֲשֶׁר עָשַׁק אוֹ אֶת הַפִּקָּדוֹן אֲשֶׁר הָפְקַד



אִתּוֹ אוֹ אֶת הָאֲבֵדָה אֲשֶׁר מָצָא, / וַיִּתֵּן לַעֲנִיֵּי



הַמְּדִינוֹת צְדָקָה / וַיֶּרֶב עַל חֶטְאוֹ תְּפִלָּה וּצְעָקָה /



וַיְשַׁנֶּה מַעֲשָׂיו לְטוֹבָה / כְּמִנְהַג כָּל בַּעַל תְּשׁוּבָה /



וַיָּעַד עַל פְּעָלָיו הַטּוֹבִים שְׁמוֹ / וַיְשַׁנֶּה אֶת טַעְמוֹ /



וַיִּלְבַּשׁ שְׁחוֹרִים וַיִּתְכַּס / וְגֹבַהּ לֵב שָׂם לְמִרְמָס, /



וַיֵּשֶׁב בְּבֵית הַנָּזִיר בְּאַהֲבָה / גָּרְסָה נַפְשׁוֹ לְתַאֲבָה. /



וְכִרְאוֹת הַנָּזִיר שִׁנּוּי פְּעָלָיו וּמַעֲשֵׂהוּ / וַיָּרַח אֶת



רֵיחַ בְּגָדָיו וַיְבָרְכֵהוּ / וַיֹּאמַר ״תִּחְיֶה וְתִזְכֶּה בַתּוֹרָה



כְּפִי חֶפְצֶךָ, / יְיָ אֱלֹהֶיךָ יִרְצֶךָ!״ / וַיְהִי מִמָּחֳרָת



וַיֵּשֶׁב הַנָּזִיר לְלַמְּדוֹ תוֹרָה / וַיִּלְמַד כָּל אֲשֶׁר הוֹרָה: /



תּוֹרָה נְבִיאִים כְּתוּבִים בְּיַחַד, / שְׁלֹשֶׁת הָרוֹעִים בְּיֶרַח



אֶחָד. / וַיִּלְמַד פֵּרוּשִׁים וְדִקְדּוּקִים, / חֻקִּים וּמִשְׁפָּטִים



צַדִּיקִים, / מִקְרָא מִשְׁנָה תַּלְמוּד וַאֲגַדְתָּא, / סִפְרֵי סִפְרָא



וְתוֹסֶפְתָּא. / וַיִּגְדַּל הָאִישׁ בְּמַעֲלַת הָעֹשֶׁר וְהַכָּבוֹד /



וַיֵּלֶךְ הָלוֹךְ וְגָדֵל עַד כִּי גָדַל מְאֹד. / וַיְשִׂימֵהוּ



הַנָּזִיר עַל תַּלְמִידָיו נָגִיד וּמְצַוֶּה / מַחְמִיר וּמֵקֵל



וּמַשְׁוֶה, / כִּי כָל דְּבָרָיו עַל פִּי הַתּוֹרָה שְׁקָלָן וַחֲלָקָן /



חֲזָקָה עַל חָבֵר שֶׁאֵינוֹ מוֹצִיא מִתַּחַת יָדוֹ דָּבָר שֶׁאֵנוֹ



מְתֻקָּן.

וַיִּשְׁמַע בִּלְדַּד אָבִיו עִנְיָנוֹ / וַיִּיטַב



הַדָּבָר בְּעֵינָיו עַל אוֹדוֹת בְּנוֹ / וַיִּשְׁלַח אֵלָיו סְפָרִים



וַאֲרֻחָה / מִנְחָה הִיא אֶל הַנָּזִיר שְׁלוּחָה / כֶסֶף וְזָהָב



וּבְגָדִים כַּאֲשֶׁר יֵאוֹת וְקִדָּה חֲמֵשׁ מֵאוֹת. / וַיִּתֶּן לוֹ



הַנָּזִיר בִּתּוֹ לְאִשָּׁה – / תְּהִלָּה תִפְאֶרֶת קְדֻשָּׁה – / כִּי



אָמַר: לְעוֹלָם יִמְכֹּר אָדָם כָּל אֲשֶׁר לוֹ / לְהַרְבּוֹת מַהֲלָלוֹ,



/ וְיַשִּׂיא בִתּוֹ לְתַלְמִיד חָכָם / וְיָבִין וְיֶחְכַּם, כִּי בַת



הֶחָכָם הַנְּשׂוּאָה לְתַלְמִיד / עוֹלָה תָמִיד: / כִּי הוּא דָבָר



מִתְקַבֵּל וְנָאֶה / יַעֲלֶה וְיָבוֹא יַגִּיעַ יֵרָאֶה, / מָשָׁל



לְעִנְבֵי הַגֶּפֶן בַּגֶּפֶן, / וְאֶל אָוֶן נַפְשׁוֹ אַל תֵּפֶן. / כִּי



הַתַּלְמִיד אִישׁ נֶחְמָד / וְדֶרֶךְ תְּשׁוּבָה לָמַד / וַיְשַׁמֵּשׁ אֶת



גְּדוֹל הַחָכְמָה / וַיְהִי לָהֶם לְחוֹמָה. / וְהַמַּשִּׂיא בִתּוֹ לְעַם



הָאָרֶץ כְּאִלּוּ כוֹפְתָהּ בִּפְנֵי הָאֲרִי / כָּעֵת בַּמָּרוֹם



תַּמְרִיא. / וּכְבָר אָמְרוּ, / נִמְנוּ וְגָמְרוּ: / אֵין בּוּר יְרֵא



חֵטְא וְלֹא עַם הָאָרֶץ חָסִיד. / וְאָמְרוּ: אִם תַּלְמִיד חָכָם נָחָשׁ



הוּא – חָגְרֵהוּ עַל מָתְנֶיךָ, / וְאִם עַם הָאָרֶץ חָסִיד – אַל תָּדוּר



בִּשְׁכוּנָתוֹ וְהַרְחֵק מִמֶּנּוּ מְעוֹנֶךָ. / וְעַל זֶה צָרִיךְ



לְהַשִּׂיא בַּת הַחֲמוּדָה / לְמַשְׂכִּיל עַל דָּבָר נוֹצֵר תְּעוּדָה /



״אֲשֶׁר יַכֶּה קִרְיַת סֵפֶר וּלְכָדָהּ״. / וַיַּעַשׂ הֶחָבֵר חֻפָּה



רַעֲנָנָה וּשְׁלֵמָה / עַד מְלֹאות לוֹ שָׁנָה תְּמִימָה, / וְכָל



הַתַּלְמִידִים גָמְלוּ אוֹתוֹ חֶסֶד לְרֹב אֵהֲבָתוֹ / וַיֹּאכְלוּ עִמּוֹ



לֶחֶם בְּבֵיתוֹ.





וַיְהִי אַחֲרֵי הַדְּבָרִים הָאֵלֶּה וַיִּשְׁמְעוּ בְנֵי



הַמְּדִינָה / מַהֲלַל הַכְּנֶסֶת וְהָעֲדִינָה / וְהַמִּדְרָשׁ הַקָּבוּעַ



וְהַמְּסִבָּה / אֲשֶׁר הַנָּבִיא הַזָּקֵן יוֹשֵׁב בְּקִרְבָּהּ – /



וַיִּתְקַבְּצוּ כֻלָּם בְנֶפֶשׁ נְכוֹנָה וְרוּחַ נְדִיבָה / אֶל הַנָּהָר



הַבָּא אֶל הָאַהֲבָה. / וַיִּשְׁלְחוּ מַלְאָכִים אֶל הַנָּזִיר



וְהֶחָבֵר, / מוֹר עוֹבֵר, /וַיֹּאמְרוּ ״רָאֹה רָאִינוּ כִּי הָיָה יְיָ



עִמָּכֶם, / וַנֹּאמַר: תְּהִי נָא אָלָה בֵּינוֹתֵינוּ וּבֵינֵיכֶם. /



בּוֹאוּ וּרְשׁוּ הָאָרֶץ הָעֲרוּכָה / וּשְׁבוּ בְתוֹכָהּ, / וְכָל אִישׁ



אֲשֶׁר פִּיכֶם יַמְרֶה – / יָרֹה יִיָּרֶה, / וְתִהְיוּ בֵינֵינוּ



לְמוֹפֵת וּלְאוֹת / שָׂרֵי אֲלָפִים וְשָׂרֵי מֵאוֹת״.

וַיֹּאמֶר הַנָּזִיר ״הִנֵּה נָא זָקַנְתִּי, / יוֹם



מוֹתִי לֹא יָדַעְתִּי, / וְאֵינִי יָכוֹל לְהַלְאוֹת עַצְמִי / וְלָנוּד



מִמְּקוֹמִי: / אָמְנָם חֲתָנִי הֶחָבֵר הַנֶּאֱמָן / הִנֵּה מוּכָן



וּמְזֻמָּן, / יָכוֹל לָלֶכֶת בַּהֲנָחָה / לָתוּר לָכֶם מְנוּחָה״. /



וַיֹּאמֶר הֶחָבֵר ״אֵיתָן מוֹשָׁבֶךָ, / חַי יְיָ וְחֵי נַפְשְׁךָ אִם



אֶעֶזְבֶךָּ!״ / וַיֹּאמְרוּ לוֹ הָאֲנָשִׁים ״אַל נָא תִּמָּנַע מֵהֲלֹךְ



אִתָּנוּ, / כִּי אֱלֹהִים הֲדִמָּנוּ, / וְלֹא נִשְׁאַר לָנוּ חָכָם



מוֹרֶה / וְלֹא תַלְמִיד קוֹרֵא, / וְאַתָּה תִהְיֶה לָּנוּ לְמֵלִיץ



וּלְסַבָּר – / עַל כֵּן יָדַעְתָּ חֲנוֹתֵנוּ בַּמִּדְבָּר״. / וַיֹּאמֶר



הַנָּזִיר ״הִנֵּה נָא הָעִיר הַזֹּאת קְרוֹבָה, / וְעָלֶיךָ הַמִּצְוָה



וְהַחוֹבָה / לְהַדְרִיכָם בְּדֶרֶךְ טוֹב וְנַעֲלָה / וְאֹרַח יְשָׁרִים



סְלוּלָה, / וְכָל זְמָן תּוּכַל לַעֲלוֹת אֶל בֵּית מְלוֹנַי, / כִּי



תָבוֹא לִרְאוֹת פָּנַי״.

וַיָקָם הֶחָבֵר וַיֵּלֶךְ בַּחֲבוּרָתָם / בִּמְחוֹקֵק



בְּמִשְׁעֲנוֹתָם. / וַיָּכִינוּ לוֹ בַּיִת וַעֲבָדִים מְשֻׁבָּחִים /



וַעֲלִיּוֹת מְרֻוָּחִים. / וַיְהִי לְכָל בְּנֵי הָעִיר מָגֵן וְצִנָּה, /



נָחָה שָׁקְטָה הָאָרֶץ פָּצְחוּ רִנָּה, / וַיִּנְהָגֵם עַל פִּי



הַתּוֹרָה וְהַמּוּסָר / וַיְהִי לָהֶם לְשַׂר, / וַיּוֹרֵם אֶת מִשְׁפַּט



הַתְּשׁוּבָה / כַּאֲשֶׁר הִיא עַל סֵפֶר הַיָּשָׁר כְּתוּבָה. /



וְלִפְעָמִים הָיָה מַקְבִּיל פְנֵי חֲבֵרוֹ בְּנַחַת וְזִמְרָה / חֹדֶשׁ



וְשַׁבָּת קְרֹא מִקְרָא. / וַיִּהְיוּ לוֹ בָּנִים וְתַלְמִידִים /



עוֹסְקִים בַּתּוֹרָה וּבִגְמִילוּת חֲסָדִים. / וַיְהִי רוֹכֵב פַּעַם



בַּשָּׁבוּעַ, / בִּזְמָן קָבוּעַ, / לְיַשֵּׁר אֶת בְּנֵי הַסְּחוֹרָה /



בַּמִּדָּה בַּמִּשְׁקָל וּבַמְּשׂוּרָה / וְשָׁב אֶל מִדְרָשׁוֹ שׁוֹקֵט



וְשָׂמֵחַ / וְרֹב שָׁלוֹם עַד בְּלִי יָרֵחַ.




מקור: שירמן – השירה העברית בספרד ובפרובאנס
