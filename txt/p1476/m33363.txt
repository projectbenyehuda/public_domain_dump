


	
	אָמְרוּ הַקַּדְמוֹנִים
	








מתוך השער הראשון של הספר משל הקדמוני.





אָמְרוּ הַקַּדְמוֹנִים / וּפוֹתְרֵי הַדִּמְיוֹנִים /



כִּי הָיָה בְאֶרֶץ צֹעַן אִישׁ מַשְׂכִּיל בְּעִנְיָנָיו, / אִישׁ חָסִיד



אִישׁ עָנָו. / וַהֲוָת שַׁעְתָּא דְחִיקָא לֵהּ / וּבָנָיו בְּנֵי דְּלָא



מַעֲלֵי, / לֹא הָיָה מַשִּׂיג דֵּי מַחְסוֹרוֹ / בְּרֹב חָכְמָתוֹ



וְיָשְׁרוֹ, / וְהָיָה חוֹקֵר כָּל תַּעֲלוּמָה / בְּרָעָב וּבְצָמָא, /



וְכָל הַיּוֹם צוֹעֵק וּמֵהִים / נָגוּעַ וּמֻכֵּה אֱלֹהִים, / וּבָנָיו



חוֹזְרִים עַל הַפְּתָחִים, / נֶאֱנָקִים וְנֶאֱנָחִים. / וְהָיוּ



שְׁכֵנָיו שְמֵחִים לְאֵידוֹ / וּמִתְעַנְּגִים בְּפִידוֹ, / כִּי אָמְרוּ:



מַה לּוֹ וְלַחָכְמָה / וְלַחְקֹר תַּעֲלוּמָה? / וּכְבָר קָדַם הַנָּבִיא



בִּנְבוּאָתוֹ / ״אַל יִתְהַלֵּל חָכָם בְּחָכְמָתוֹ!״



וַיְהִי כַאֲשֶׁר גָּדַל כְּאֵבוֹ וְצַעֲרוֹ / וְהַקַּיִץ



חִלֵּל נִזְרוֹ / וַיִּתְחַזֵּק בַּשָּׁנָה הַהִיא חֹם גָּדוֹל וְהַיֹּבֶשׁ



רַד, / כִּי חָדַל הַמָּטָר וְהַבָּרָד – / וַיֶּאֱרַע לוֹ חֳלִי הַיֹּבֶשׁ



בְּמֹחוֹ, / סָר כֹּחוֹ: / וְהוּא חֳלִי הַמַּחֲשָׁב וְהַזִּכָּרוֹן /



וְאֵין לוֹ עַל הַבְּהֵמָה מַעֲלָה וְיִתְרוֹן. / וַיָּחֶל הַיָּגֵעַ /



הוֹלֵךְ וּמִשְׁתַּגֵּעַ, / וַתִסָּתֵר מִלִּבּוֹ הַחָכְמָה וְהַתְּשׁוּקָה



/ וְנַפְשׁוֹ שׁוֹקֵקָה, / וַיַּחֲזֵק בִּבְגָדָיו הַצְּנוּעִים /



וַיִּקְרָעֵם לִשְׁנֵים עָשָׂר קְרָעִים. / וַיִּתְלַקְּטוּ אֵלָיו



אֲנָשִׁים רֵיקִים וּפוֹחֲזִים, / שׁוֹדְדֵי עֶרֶב וּבוֹזֲזִים, /



וַיסוֹבֲבוּהוּ הַנְּעָרִים בְּרִיבוֹ / בְּשָׁמְעָם צַעֲקוֹ וְנִיבוֹ, /



וְתַעֲלוּלִים מָשְׁלוּ בוֹ. / וַיָּבוֹא עַד לִפְנֵי שַׁעַר הַמֶּלֶךְ /



מְצַחֵק וְתוֹמֵךְ בַּפֶּלֶךְ – / וְהַמֶּלֶךְ יוֹשֵׁב בַּעֲלִיַּת



הַמְּקֵרָה אֲשֶׁר לוֹ / וְהַשֵּׁגַל יוֹשֶׁבֶת אֶצְלוֹ. / וַיִּתְאַו



לִשְׁמֹעַ רִבּוּי אֲמָרָיו אֲשֶׁר לָהַג, / כִּי בְשִׁגָּעוֹן יִנְהַג, /



וַיְצַו לַהֲבִיאוֹ בִּרְגִילוּת / וְלֶאֱחֹז בְּסִכְלוּת, /



וַיִּשְׁתַעֲשַׁע הַמֶּלֶךְ בְּמַהֲלָלָיו / וְהַשֵּׁגַל בְּמַעֲלָלָיו. /



וַיִּתֶּן לוֹ הַמֶּלֶךְ כֻּתָּנְתּוֹ / וְלֶחֶם אָמַר לוֹ לְהַבְרוֹתוֹ, /



וַיַּפְקִידֵהוּ עִם שָׂרָיו הַנִּכְבָּדִים / וַיַּלְבִּישֵׁהוּ בְגָדִים,



/ וַיְהִי לִפְנֵיהֶם כָּל הַזְּמַנִּים / שׁוֹתֶה מַמְתַּקִּים וְאוֹכֵל



מַעֲדָנִים. / וּכְמוֹ כֵן צִוָּה לָתֵת / לְאִשְׁתּוֹ וּלְבָנָיו



מַשְׂאֵת. / וַיְהִי יוֹשֵׁב בְּמוֹשַׁב הַשָּׂרִים / עִם בְּנֵי הַמֶּלֶךְ



וְהַנְּעָרִים, / כִּי הָיָה מְשַׁעַשְׁעָם בְּדִבְרֵי הִתּוּל וּשְׂחוֹק /



כְּמִשְׁפַּט כָּל כְּסִיל וּכְחֹק. / וְכָל יוֹשְׁבֵי הָעִיר, / יַחַד



יָשִׁישׁ וְצָעִיר, / יְשִׂימוּן עַיִן עָלָיו לְשָׁמְרֵהוּ, / כִּי



מִצְוַת הַמֶּלֶךְ הוּא. / וַיֹּאמְרוּ כָּל יוֹדְעָיו כַּאֲשֶׁר רָאוּ



מַעֲלָתוֹ / בַּחֲצַר הַמֶּלֶךְ עִם כָּל בֵּיתוֹ / ״לְטוֹבָתוֹ /



נִשְׁבְּרָה רֶגֶל פָּרָתוֹ!״





וַיְהִי מִקֵּץ שְׁנָתַיִם יָמִים – וַיֵּעָתֵר אֱלֹהִים



לָאֲדָמָה / תַּחַת אֲשֶׁר הָיְתָה שְׁמָמָה / וְהִשְׁגִּיחַ עַל כָּל



יוֹשְׁבֶיהָ / מִמְּכוֹן עֲרָבוֹת, / וַיָּנֶף גִּשְׁמֵי נְדָבוֹת /



לְטוֹבָה וּלְשֹׂבַע וְלִבְרָכָה / שְׁמוּרָה בַכֹּל וַעֲרוּכָה. /



וַיָחֵלוּ הַנִּצָּנִים לְהִתְנָאוֹת / לַעֲלוֹת וְלֵרָאוֹת / וְלִהְיוֹת



כָּל עֵץ פּוֹרֵחַ, / הַגְּפָנִים וְהַסְּמָדַר נָתְנוּ רֵיחַ. / וַיָּקָם



הַיֹּבֶשׁ וַיִּבְרַח / וַיָּשִׂישׂ כְּגִבּוֹר לָרוּץ אֹרַח, / וַיָּבוֹא



חֹדֶשׁ הָאָבִיב בְּשִׂמְחָה / וְנָסוּ יָגוֹן וַאֲנָחָה. / וַיִּכָּנֵס



הַשֶּׁמֶשׁ בְּמַזַּל טָלֶה / וְקָרְאוּ אַחֲרָיו ״מָלֵא!״ / בִּהְיוֹתוֹ



מַעֲלֶה דְשָׁאִים / בִּזְרוֹעוֹ יְקַבֵּץ טְלָאִים. / וְהַיּוֹם



וְהַלַּיְלָה בְּאַהֲבָה נֶחְבָּקוּ, / אִישׁ בְּאָחִיהוּ יְדֻבָּקוּ /



וְנִשְׁתַּוּוּ בְּמַעֲלָתָם וְגָבְהָם / בְּמַרְאֵיהֶם וְגֹבַהּ לָהֶם. /



וַיִּרְבֶּה הַזַּךְ וְהַנָּקִי מִן הַדָּם / וַיָּשָׁב עֵין כָּל בָּחוּר



מְאָדָּם, / וְנִרְאָה כָּל לָבָן אֲדַמְדַּם, / וְהַמָּרָה הַשְּׁחֹרָה



לֹא תֵחָשֵׁב / וְנִקָּתָה לָאָרֶץ תֵּשֵׁב. / וְהַטֶּבַע הֶחָכָם מֵסִיר



הַדְּאָגוֹת / וּמְשַׁוֶּה הַזּוּגוֹת / וְהָעֲמֵלִים מִתְחַזְּקִים



לִשְׂמֹחַ וַעֲמָלָם לִשְׁכֹּחַ. / וַיֶּחְסַר בָּעֵת הַהִיא הַכֶּסֶל



וַיִּגָּרַע / וְאָסַף הַמְּצֹרָע – / וַיָּשָׁב הַכְּסִיל לְדַעְתּוֹ



וּלְעִנְיָנוֹ / בִּפְנוֹת בֹּקֶר לְאֵיתָנוֹ.

וַיְהִי לְעֵת עֶרֶב / בְּהִתְעַצֵּב כָּל לֵב וְקֶרֶב. /



וַיִּדְרֹשׁ הַמֶּלֶךְ בְּעַד סְכָלוֹ / וַיַּרְא כִּי לֹא יָכוֹל לוֹ. /



וַיִּשְׁאָלוּהוּ הַשָּׂרִים וְהַשָּׂרוֹת / שִׂיחָה עֲמֻקָּה פִּי זָרוֹת,



/ וַיִּתְהוֹלְלוּ עִמּוֹ – וְלֹא עָנָה / וְאֶל שָׂטֵי כָזָב לֹא פָנָה./



וַיְהִי כַאֲשֶׁר הֶאֱלִיצוּהוּ / וּבְדִבְרֵי פַחֲזוּתָם הֱאִיצוּהוּ, /



וַיָּרֶם קוֹלוֹ וַיִּקְרָא בְשַׁוְעוֹ / ״אַךְ אֱוִילִים שָׂרֵי צֹעַן,



חַכְמֵי יוֹעֲצֵי פַרְעֹה!״ / וַיִּתְנַכֵּר אֲלֵיהֶם בְּמִלּוּלוֹ /



וּבֵאֵר שִׂיחוֹ הֵרִים קוֹלוֹ / וַיִּפְתַּח וַיִּשָּׂא מְשָׁלוֹ:





          תְרִיבוּנִי עֲלֵי חֶמְדַּת אֲמָרַי – / תְּבַקְשׁוּ חֵטְא וְסִכְלוּת מִדְּבָרַי:

          הֲלֹא סָכָל וְאִישׁ הוֹלֵל – בְּצָרַי, / בְּנֵי הֶבֶל, אֲשֶׁר שׂוֹנְאַי וְעָרַי!

          לְמָדוּנִי מְלִיצָה מִנְּעוּרַי, / וְהַשֵּׂכֶל וְהַבִּין – מִבְּחוּרַי.





וַיִּתְעַצְּבוּ הָאֲנָשִׁים עַל מַאֲמָרוֹ / כִּי חִלֵּל



דְּבָרוֹ / וַיֹּאמְרוּ ״לֹא בָא זֶה אֶלָּא לְרַמּוֹתֵנוּ / לִרְאוֹת



עֶרְוָתֵנוּ, / וְכָל סִכְלוּתוֹ הָיְתָה בְעָרְמָה / עַל פִּיו הָיְתָה



שׂוּמָה, / כַּאֲשֶׁר הֶעֱרִים הַמְשׁוֹרֵר נְאוּמוֹ / וַיְשַׁנֶּה אֶת



טַעְמוֹ / וַיּוֹרֶד רִירוֹ עַל זְקָנוֹ / וַיִּתְהוֹלֵל בִּרְצוֹנוֹ. /



וְהִנֵּה הוּא חַיָּב מַלְקוֹת אַרְבָּעִים / כְּמִשְׁפַּט הַמַּטְעִים״. /



וַיֹּאמְרוּ אֶל הַשּׁוֹטֵר לְהַלְקוֹתוֹ / וְכַמִּסְפָּר הַזֶּה



לְהַכּוֹתוֹ.





וַיְצַו הַמֶּלֶךְ לְהָסִיר מֵעָלָיו מַחֲלָצוֹת



הַטּוֹבוֹת / וַיַּלְבִּישֵׁהוּ בְלוֹיֵי הַסְּחָבוֹת / וּכְנֵצֶר נִתְעָב



מִבֵּיתוֹ הֻשְׁלַךְ / וַיְגָרְשֵׁהוּ וַיֵּלַךְ. / וַיֵּלֶךְ הָאִישׁ אֶל



בֵּיתוֹ סַר וְזָעֵף / יָגֵעַ וְעָיֵף, / וַיַּגֵּד לְאִשְׁתּוֹ וּלְבָנָיו



מַחֲלָתוֹ וְשִׁבְיוֹ / בַּחֲלוֹתוֹ וַיְחִי מֵחָלְיוֹ / וְאֶת כָּל אֲשֶׁר



אָמְרוּ הַשָּׂרִים לְהַכּוֹתוֹ / לְמַעַן עַנּוֹתוֹ. / וַיֹּאמַר ״עַתָּה



יָדַעְתִּי / וּסְבָרָתִי סָבַרְתִּי / כִּי הֶעְמִיק הֶחָכָם



בְּמִשְׁפָּטוֹ אֲשֶׁר שָׁפַט: / עָקָר מֵחָכְמָה מִכָּבוֹד סִכְלוּת



מְעַט!״
