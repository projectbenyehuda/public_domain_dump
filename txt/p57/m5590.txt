


	
	מן האדה האיסלנדית
	



מן האדה האיסלנדית / משוררים אלמונים /תרגום שאול טשרניחובסקי



הקדמה

האֶדה (אמא דסבתא) אינה אלא אוסף שירים של משוררים אלמונים מן המאה התשיעית ועד המאה השתים-עשרה למניין הנוצרים והיא נכללת בספרות האיסלנדית הואיל והשירים נקבצו באיסלנדיה. שפתם של כתבי-היד – איסלנדית – אמנם דומה עד היום ברובה לשפה הנורבגית הקדומה, אולם רוב השירים חוברו בנורבגיה.

איש איסלנדי אחד, עלום-שם, כינס את השירים האלה בתחילת המאה ה-13 לאוסף אחד; סבורים כי הוא-הוא שצירף גם את המבואות שבפרוזה, את דברי הנעילה ואת המאמרים המוסגרים, שאנו מוצאים ברוב השירים. כתב-היד העתיק ביותר הוא ה- Codex regius מסוף המאה ה-13 והוא שמור בספרייה אשר בקופנהגן.

מתוך שירי האֶדה אנו לומדים לדעת את המיתולוגיה הגרמנית העתיקה ואת אגדות הגיבורים שלה; אין צורך להרבות מלים עד כמה גדול ערכה של זו.

השירים כתובים במשקל קבוע שאת חוקיו גילה בחקירותיו החכם המפורסם בתחום זה – אדוארד סיבֶרס, והרי סיכום דבריו בקווים כלליים:

בדרך כלל יש להבחין בין שלושה משקלים. בשיר הסיפורי נהגו להשתמש בשני משקלים, ה- Fornydislag וה- Malahattr. השיר בנוי בתים-בתים, בני ארבעה חרוזים (שורות); מספר השורות הוא לפעמים יותר גדול; החרוז השירי משתמש באַליטרציה (דמיון ראשי התיבות), זאת אומרת הישנותה של הברה אחת בעלת נגינה בתחומו של חרוז אחד או זוג חרוזים.

ארבע השורות – ארוכות, והחתך ( Cäsura ) מפריד אותן לשתי שורות קצרות בעלות שתי הברות בנות-נגינה – “הגבהות”.

מספר ההברות קבוע: במשקל הראשון – ארבע, בשני – חמש. מעין דוגמה ישמש לנו החרוז (בגרמנית): Met trinkt Mimir / am Morgen taeglich שבו ההגבהה ניכרת יפה באות M הגדולה.

בעברית: מי -דבש יגמא / מי מיר לבקרים.



נְבוּאַת הָרוֹאָה


הַאֲזֵן הַאֲזִינוּ / דּוֹרוֹת הַקֹּדֶשׁ,



בָּנָיו שֶׁל הֵימְדַל / גְּדוֹלִים וּקְטַנִּים;



וַלְפָטֶר כִּי רָצָה, / אֲנִי אֲסַפֵּרָה



תּוֹלְדוֹת יְמֵי-קֶדֶם, / זִכְרוֹנוֹת רִאשׁוֹנִים.


  


שְׁמוּרִים בְּזִכְרוֹנִי – / אֲבוֹת-אֲבוֹת עֲנָקִים,



שֶׁהֵם יְלָדוּנִי / לִפְנֵי עִדָּנִים;



תִּשְׁעָה עוֹלָמוֹת יָדַעְתִּי / תִּשְׁעָה דּוֹרֵי עֵץ-הָעוֹלָם



הַמַּעֲמִיק שָׁרָשָׁיו / בְּבֶטֶן הָאֲדָמָה.


  


אָז בִּימוֹת-עוֹלָם / וְאִימִיר עוֹדוֹ חַי,



וְאֵין חָצָץ וְאֵין יָם / גַּם לֹא מִשְׁבָּר קַר;



לֹא הָיְתָה אֶרֶץ / וְלֹא רָקִיעַ,



אַךְ תְּהוֹם פּוֹעֶרֶת פֶּה, / וְאֵין זֵכֶר לַדֶּשֶׁא.


  


וְהִנֵּה בִּצְבְּצוּ וְעָלוּ / בְּנֵי בּוּר מִן הַבַּר



וְהֵם יָצְרוּ / אֶת מִידְגַרְד הַיָּפֶה,



מִדָּרוֹם הֵאִיר / הַשֶּׁמֶשׁ אֶת הָאָרֶץ,



וַיַּעַל בַּקַּרְקַע / עֵשֶׂב מוֹרִיק.


  


וְשֶׁמֶשׁ דָּרוֹם / נִלְוֶה אֶל יָרֵחַ,



נָגַע בִּימִינֵהוּ / בִּקְצֵה הַשָּׁמַיִם;



לֹא יָדַע הַשֶּׁמֶשׁ / אֵיפֹה זְבוּל לוֹ,



לֹא יָדַע יָרֵחַ / מַה כֹּחוֹ וְאוֹנוֹ.



לֹא יָדְעוּ הַכּוֹכָבִים / מְקוֹמָם אַיּוֹ.


  


אָז הָלְכוּ לַעֲצָרָה / הָאֵלִים כֻּלָּמוֹ,



מוֹשְׁלי הַקֹּדֶשׁ, / וְשָׁם נוֹעָצוּ:



קָרְאוּ שֵׁם לַלַּיְלָה, / לַיָּרֵחַ וְלַכֶּסֶא,



בֹּקֶר וָעֶרֶב, / צָהֳרַיִם וְעַרְבִית,



כָּל הַמּוֹעֲדִים / לִסְפִירַת הַשָּׁנִים.


  


לְאִידָפֶלְד בָּאוּ / הָאָזִים נִקְבָּצוּ



לַעֲשׂוֹת מִזְבְּחוֹת / וְלִבְנוֹת הֵיכָלוֹת;



יָסְדוּ כּוּרִים / לְחַשֵּׁל הַזָּהָב,



נִגְּדוּ צְבָתוֹת / וְכָל-כְּלֵי עֲבוֹדָה.


  


בֶּחָצֵר הִשְׁתַּעֲשָׁעוּ, / עַלִּיזִים בְּמִשְׂחַק-הַקֶּרֶשׁ,



כִּי לֹא חָסְרוּ / זְהַב פַּרְוָיִם –



עַד שֶׁבָּאוּ שְׁלָשׁ / הַנְּעָרוֹת הָאַדִּירוֹת,



בְּנוֹת הָעֲנָקִים / מִטּוּרְסֶנְהֵים.


 


[אָז הָלְכוּ לַעֲצָרָה / הָאֵלִים כֻּלָּמוֹ,



מוֹשְׁלי הַקֹּדֶשׁ, / וְשָׁם נוֹעָצוּ:



מִי וָמִי יִצּוֹר / הֲמוֹן הַנַּנָּסִים



מִדָּמָיו שֶׁל בְּרִימִיר / מֵאֵיבָרָיו שֶׁל בְּלָאִין.


 


וַיְהִי מוֹטְסוֹגְנִיר / הָאַדִּיר בֵּינֵיהֶם,



בְּכָל הַנַּנָּסִים, / וְדוּרִין הַשֵּׁנִי.



וַיַּעֲשׂוּ הַנַּנָּסִים / כַּמָּה צוּרוֹת בְּנֵי-אָדָם



בָּאֲדָמָה / כְּמִצְוַת דּוּרִין.


 


נִיאִי וְנִידִי, / נוֹרְדְרִי וְסוּדְרִי,



אוֹסְטְרִי וּוֶסְטְרִי / אַלְטְיוֹף, דְּוָלִין,



נַר וְנָאִין, / נִיפִּינְג, דָּאִין,



בִּיפוּר, בַּפוּר, / בּוֹמְבּוֹר, נוֹרִי,



אַן וְאוֹנַר, / אַיִי, מְיוֹדְבִיטְנִיר.





וִיג וְגַנְדַלְף / וִינְדוֹלְף, טוֹרִין,



טְרוֹר וּטְרָאִין / טֶק, לִיט וּוִיט,



נִיר וְרֶגִין / נִירַד וְרַדְסְבִיד –



נְכוֹנָה מָנִיתִי / אֶת הַנַּנָּסִים.


  


פִילִי וְקִילִי, / פוּנְדִין, נָלִי,



הֶפְּטִיפִילִי, / הָנַר, סְוִיאוּר,



פְרַר, הוֹרְנְבּוֹרִי, / פְרֶג וְלוֹנִי,



אָאוּרְוַנְג, יַארִי, / אֵיקִינְשַׁלְדִּי.


  


עֵת לְכַנּוֹת בְּשִׁיר / אֶת-הָאָדָם בִּשְׁמוֹ,



בְּנֵי עַם הַנַּנָּסִים / אֲבִי אֲבוֹת לוֹפַר;



גִּזְעָם עָלָה / מֵאֶרֶץ הַטְּרָשִׁים



דֶּרֶךְ בִּקְעוֹת-טִיט / אֶל שַׁדְמַת-הַחוֹל.


  


וְאֵלֶּה שְׁמוֹתָם: דְּרָאוּפְּנִיר / וְדוֹלְגְטְרָסִיר



הָר, הָאוּגְסְפּוֹרִי, / הֶלֶוַנְג, גְּלוֹאִין,



דּוֹרִי, אוֹרִי / דּוּף, אַנְדְּוַרִי,



שִׁירְפִיר, וִירְפִיר, / שָׁפִיד, אַיִי,


  


אַלְף וְאִינְגְוִי, / אֵיקִינְשַׁלְדִּי,



פְיַאלַר וּפְרוֹסְטִי / גִּנַר וּפִיד:



עַד בִּלְתִּי יָרֵחַ / יֵדְעוּ יָדֹעַ



שׁוֹשֶׁלֶת אֲרֻכָּה זֹאת / שֶׁל אֲבוֹת-לוֹפַר.]


  


וּשְׁלֹשָׁה אָזִים / נֶאְדָּרִים וְיָפִים



בְּנֵי הַזֶּרַע הַהוּא / עַד שְׂפַת-הַיָּם בָּאוּ;



בַּשָּׂדֶה הַנִּרְחָב / מָצְאוּ אֶת אַסְק וְאֶת אֶמְבְּלָה



תְּשׁוּשֵׁי-כֹח / וְגוֹרָלָם רוֹפֵף.


  


לֹא הָיָה בָּהֶם / לֹא רוּחַ לֹא נָפֶשׁ,



לֹא חֹם לֹא תְּנוּעָה, / לֹא צְבָעִים פּוֹרְחִים;



רוּחַ נָתַן אוֹדִין, / הוֹנִיר – הַנָּפֶשׁ,



לוֹדוּר – הַחֹם / וּצְבָעִים מְאִירִים.


 


אֲנִי יוֹדַעַת מֵילָה / אִיגְדְרָזִיל יְכַנּוּהָ,



מֹהַל לָבָן / מַרְטִיב אֶת עֵץ-הָאַדִּירִים;



מִשָּׁם בָּא הַטַּל / הַמַּרְטִיב עֲמָקִים;



מוֹרִיק הוּא תָּמִיד / עַל מַבּוּעַ הָאוּרְד


 


עַל יַד גֶּזַע הָעֵץ / עוֹמֵד אוּלָם,



שָׁלֹש בְּנוֹת-חָכְמָה / גָּרוֹת בְּתוֹכוֹ:



[הָאַחַת שְׁמָהּ אוּרְד, / הַשְּׁנִיָּה וֶרְדַּנְדִּי –



הֵן חָרְטוּ בְּרַעֲפֵי-עֵץ – / וְהַשְּׁלִישִׁית הִיא שׁוּלְד;]



הֵן פֹּה חָרְצוּ / גּוֹרַל הַחַיִּים



לִבְנֵי-אָדָם, / גּוֹרַל כָּל-גָּבֶר.


  


רִאשׁוֹנָה לְמִלְחֲמוֹת-עוֹלָמוֹת / אֲנִי יָדַעְתִּי



עֵת דָּקְרוּ / אֶת גּוּלְוֵיג בִּרְמָחִים



וַיִּשְׂרְפוּ בָּאֵשׁ / בְּאוּלַמּוֹ שֶׁל הָוִי,



שָׁלֹש פְּעָמִים שָׂרְפוּ / אֶת-הַיְּלוּדָה שָׁלֹש



[כַּמָּה וְכַמָּה פְּעָמִים / וְהִיא חַיָּה עוֹדָהּ].


  


הֵיד קָרְאוּ לָהּ / כְּשֶׁנִּכְנְסָה בָּיְתָה



אוֹתָהּ קוֹסֶמֶת / לֶב-חָכְמָה לָהּ וְחָזוֹן;



קְסָמִים בְּיָדֶיהָ / מַטְרִיפִים הַדַּעַת,



נָשִׁים קַלּוֹת-דַּעַת / חָשְׁקוּ בָּהּ תָּמִיד.





הָלְכוּ לַעֲצָרָה / הָאֵלִים כֻּלָּהַם,



מוֹשְׁלי הַקֹּדֶשׁ, / נוֹסְדוּ נוֹעָצוּ:



הֲיִשְׂאוּ מַס / הָאָזִים גַּם הֵם



אוֹ כָּל הָאֵלִים / יֵהָנוּ מִן הַקָּרְבָּנוֹת.


  


הֵטִיל אודִין / הָעֵץ בַּחַיִל –



גַּם זֶה אֵרַע / בַּקְּרָב הָרִאשׁוֹן –



אָז נִשְׁבְּרָה הַסּוֹלְלָה / שֶׁל בִּירָנִית הָאָזִים



הַוָּנִים אַנְשֵׁי-הַמִּלְחָמָה / בּוֹסְסוּ הַשָּׂדֶה.


  


הָלְכוּ לַעֲצָרָה / הָאֵלִים כֻּלָּהַם,



מוֹשְׁלי הַקֹּדֶשׁ / נוֹסְדוּ נוֹעָצוּ:



מִי הוּא אֲשֶׁר מִלֵּא / כָּל הָאֲוִיר רָעַל



וַיִּתֵּן לְזֶרַע הָעֲנָקִים / כַּלָּתוֹ שֶׁל אוֹד?


  


אַךְ טוֹר הוּא הִסְכִּים / בְּאֹמֶץ לִבּוֹ –



לְעִתִּים רְחוֹקוֹת / יֵשֵׁב כִּי יִשְׁמַע זֹאת –



אָז נִתְרוֹפְפוּ הָאָלוֹת / הַדְּבָרִים וְהַשְּׁבוּעוֹת,



חוֹזֵה-הַמִּבְטָחִים / אֲשֶׁר כָּרְתוּ קֹדֶם.


 


אֲנִי יוֹדַעַת מַחֲבוֹאָהּ / שֶׁל קֶרֶן הֵימְדַל



תַּחַת אִילַן-הַקֹּדֶש / הַשּׁוֹתֶה אֲוִיר-שָׁמַיִם;



רוֹאָה אֲנִי הַזְּרָמִים / הָרְטֻבִּים נוֹפְלִים עָלָיו



מֵעֵרְבוֹנוֹ שֶׁל אוֹדִין, / – הֲתָבִינוּ שְׁאָר-דְּבָרַי?


 


בָּדָד יָשַׁבְתִּי / כְּשֶׁבָּא אַלּוּף הָאָזִים



הַבָּא-בַיָּמִים / וְהִבִּיט בְּעֵינָי:



"מַה תְּאַוֶּה לָדַעַת? / מַה תִּבְחָנֵנִי?



אוֹדִין, אֲנִי יוֹדַעַת, / אֵיפֹה הִצְפַּנְתָּ עֵינְךָ!"


  


יָדַעְתִּי כִּי צְפוּנָה / עֵינוֹ שׁל אוֹדִין



בְּמַבּוּעוֹ שֶׁל מִימִיר / הַמְּהֻלָּל בַּכֹּל;



מֵי-דְבַש יְגַמֵּא / מִימִיר לַבְּקָרִים



מֵעֵרְבוֹנֹו שֶׁל אוֹדִין, / – הֲתָבִינוּ שְׁאָר דְּבָרַי?


  


אֲבִי-צְבָאוֹת חָנַנִּי / עֲנָק וְטַבָּעוֹת,



כִּי לִבִּי לֶב-שֵׂכֶל / וּבִי רוּחַ-חוֹזֶה.



לִי הַרְבֵּה חָכְמָה / אוּכַל לְהַרְחִיק שׁוּר,



עֵינִי חָלְפָה-עָבְרָה / כָּל הָעוֹלָמוֹת.


  


סָבִיב סָבִיב רָאִיתִי / הַוַּלְקִירוֹת בְּבוֹאָן,



מוּכָנוֹת לִרְכּוֹב / בְּמַעַרְכוֹת הַגִּבּוֹרִים;



שׁוּלְד נָשְׁקָה מָגֵן, / שׁוֹגוּל אַחֲרֶיהָ,



גּוּד, הִילְד, גּוֹנְדוּל / וְגֵירְשׁוֹגוּל.



הִנֵּה גִּלִּיתִי לָכֶם / אֶת נַעֲרוֹת הֶרְיַן,



נְכוֹנוֹת וְעוֹמְדוֹת / לָרֶדֶת אָרְצָה.


  


רָאֹה רָאִיתִי / נִגְזְרָה גְּזֵרָה



עַל בְּנוֹ-אוֹדִין / עַל בַּלְדְּר הָאָצִיל;



רָם עַל קַרְקָעוֹ / עָלָה מִתַּמֵּר



הַדִּבְקוֹן הַזָּקוּף / שֶׁצָּמַח יָפֶה.


  


אוֹתוֹ הֶעָנָף / הַדַּק לְמַרְאֶה



הָיָה לְחֵץ-צַעַר: / יָרֹה יָרָה הוֹד;



אַךְ תֵּכֶף הוּלַד / אָחִיו שֶׁל בַּלְדְּר,



בִּן-לַיְלָה נִלְחַם / בְּנוֹ שֶׁל אוֹדִין.


  


לֹא סָרַק רֹאשׁוֹ, / לֹא נָטַל יָדָיו עַד



שָׁכַב עַל גַּל-יְקוֹד / שׂוֹנֵא-בַּלְדְּר;



אַךְ פְרִיג בָּכְתָה / בְּפֶנְסָלִיר



אֲסוֹן וַלְהַלָה. / – הֲתָבִינוּ שְׁאָר דְּבָרַי?


  


אֶת לוֹקִי הָרָע, / אֶת דְּמוּת הַמְּאֵרָה,



רָאִיתִי עָקוּד / בְּחֹרֶשׁ שָׁבִיר,



שָׁם יוֹשֶׁבֶת סִיגִין / שְׁקוּעָה בְּצָעַר



עַל מַכְאוֹבֵי אִישָׁהּ / – הֲתָבִינוּ שְׁאָר דְּבָרַי?


  


מִמִּזְרָח שׁוֹפֵךְ מֵימָיו / הַסְּלִידְר הַמַּקְצִיף



עִם חֲרָבוֹת וּפִגְיוֹנוֹת / דֶּרֶך עִמְקֵי-תַרְעֵלָה.

בְּשַׁדְמוֹת נִידָה / הִתְנוֹסֵס בַּצָּפוֹן



אוּלָם בָּנוּי פָּז / לְצֶאֱצָאֵי סִינְדְרִי;



עַל הַר אוֹקוֹלְנִיר / עָמַד עוֹד אֶחָד –



בְּרִימִיר, בֵּית-הַשֵּׁכָר / לָעֲנָקִים.


  


רָאִיתִי אוּלָם / רָחוֹק מִשֶּׁמֶשׁ



בְּנַסְטְרַנְד, וּשְׁעָרָיו / קְרוּעִים צָפוֹנָה;



בְּעַד חוֹר-עֲשָׁנוֹ פָּרַץ / גֶּשֶׁם שֶׁל רָעַל,



כִּי כָּתְלֵי הָאוּלָם / חֲתוּלִים בִּנְחָשִׁים.


  


שָׁם רָאִיתִי גְּבָרִים / נִשְׁבָּעִים לַשֶּׁקֶר וּמַכֵּי-נָפֵשׁ



בַּחֲצוֹתָם / זִרְמֵי-פֶרֶא



[וְגַם אֵלֶּה / שֶׁפִּתּוּ אֵשֶׁת-אִישׁ];



שָׁם יָנַק נִידְהוֹג / מִגְּוִיּוֹת מֵתִים,



הַזְּאֵב טָרַף בְּנֵי-אָדָם, / – הֲתָבִינוּ שְׁאָר דְּבָרַי?


  


מִקֶּדֶם יָשְׁבָה הַזְּקֵנָה / בְּיַעַר-הַבַּרְזֶל,



צֶאֱצָאֵי פֶנְרִיר / פֹּה יָלָדָה;



עוֹד יָבוֹא יוֹם / וְאֶחָד מֵאֵלֶּה



יַחֲנֹק הַשֶּׁמֶשׁ / בִּדְמוּת הַזְּאֵב.


  


מַאֲכָלוֹ בְּשַׂר / גְּבָרִים נָפָלוּ,



בְּדָמִים יְגֹאַל / מוֹשַׁב הָאֵלִים;



יִקְדַּר אוֹר-שֶׁמֶשׁ, / הַקֵּיצִים יָבֹאוּ,



יְמֵיהֶם עֲכוּרִים, / – הֲתָבִינוּ שְׁאָר דְּבָרַי?


  


שָׁם עַל הַגִּבְעָה / יָשַׁב שׁוֹמֵר-הָעֲנָקִית,



אֶגְטֶר הֶעָלֵז, / פּוֹרֵט עַל נָבֶל;



לִמְרַאֲשׁוֹתָיו קָרָא / בְּיַעַר-הַצִּפֳּרִים



שֶׂכְוִי אָדֹם יָפֶה, / הַמְכֻנֶּה פְיַאלַר.


  


וְלָאֵלִים קָרָא / גּוּלִינְקַמְבִּי,



מְעוֹרֵר-הַגִּבּוֹרִים / בְּהֵיכַל אֵל-הָאֵלִים;



וְעוֹד אַחֵר קוֹרֵא / בְּמַעֲמַקֵּי הָאָרֶץ,



בַּחֲלַל-הַהֶל / וְנוֹצוֹתָיו אֲמֻצּוֹת.


 


גַּרְם צוֹעֵק בְּקוֹל / לִפְנֵי גְּנִיפַּהֶלִיר:



קוֹרֵעַ כְּבָלָיו, / בָּא-רָץ הַזְּאֵב!



לִי הַרְבֵּה חָכְמָה, / אוּכַל לְהַרְחִיק שׁוּר



גּוֹרַל הַזַּעַם / הַקָּרֵב לָאֵלִים.


  


מְדָנִים בֵּין אַחִים / וּמַפִּילִים זֶה אֶת זֶה,



קִשְׁרֵי דָמִים מְנַתְּקִים / בְּנֵי-הָאֲחָיוֹת;



רַע וּמַר בָּעוֹלָם, / רַבָּה הַתּוֹעֵבָה –



יְמוֹת-גַּרְזֶן, יְמוֹת חֶרֶב, / מָגִנִּים מִתְבַּקְּעִים,



יְמוֹת-סַעַר, יְמוֹת-זְאֵב, / עַד תֵּבֵל שׁוֹקַעַת,



לֹא יַחְמוֹל, לֹא יָחוּס / אָדָם עַל מִשְׁנֵהוּ.





בְּנֵי-מִים נֶחְפָּזִים, / קוֹלָהּּ הַצָּלוּל



שֶׁל קֶרֶן-גְּיַאלַר / מַכְרִיז עַל הַסּוֹף;



הֵימְדַל תּוֹקֵעַ בְּקוֹל, / הַשּׁוֹפָר בָּאֲוִיר,



אוֹדִין מְדַבֵּר / עִם רֹאשׁוֹ שֶׁל מִים.


  


אִיגְדְרָזִיל מַרְעִיד / הָרָמָה בַּמֵּילוֹת,



רוֹעֵש הָעֵץ הָעַתִּיק, / לָעֲנָק קֹרָא דְרוֹר;



אָחַז פַּחַד / כָּל-יוֹשְׁבֵי שְׁאוֹל-תָּחַת,



עַד שֶׁעָלָה בִּשְׁבִילוֹ / רֵעַ-דָּמָיו שֶׁל סוּרְט.


  


מַה נִּשְׁמַע בֵּין הָאָזִים? / מַה לָּהֶם לָאֶלְבִּים?



רִיזֶנְהֵים כֻּלָּהּ הוֹמָה, / הָאָזִים נוֹעָצִים;



הַנַּנָּסִים נֶאֱנָחִים / עַל שַׁעֲרֵי הָאֲבָנִים



שֶׁלְּמוֹשְׁלֵי הַר-הַיָּעַר, / – הֲתָבִינוּ שְׁאָר דְּבָרַי?


  


גַּרְם צוֹעֵק בְּקוֹל / לִפְנֵי גְּנִיפַּהֶלִיר:



קוֹרֵעַ אֶת-הַכְּבָלִים / בָּא-רָץ הַזְּאֵב!



הַרְבֵּה חָכְמָה לִי, / אוּכַל לְהַרְחִיק שׁוּר



גּוֹרַל הַזַּעַם / הַקָּרֵב לָאֵלִים.


  


מִקֶּדֶם בָּא הֲרִים / מָגִנּוֹ בִּזְרוֹעוֹ,



בַּגַּלִּים מִתְגּוֹלֵל / נְחַשׁ-הָעוֹלָם



בַּחֲמַת עֲנָקִים; / צְרוּדוֹת נוֹאֵק הָעַיִט הַשָּׂב,



בְּרַטְּשׁוֹ פְּגָרִים; / יִנָּתֵק נַגְלְפַר.


  


מִצָּפוֹן שָׁטָה / אֳנִיָּה בַּיָּם,



בָּהּ אַנְשֵׁי-הֶל / וְלוֹקִי בָּהּ נוֹהֵג;



אַחֲרֵי הַזְּאֵב בָּאִים / קְהַל פִּרְאֵי-אָדָם,



אִתָּם בִּקְהָלָם / אָחִיו שֶׁל בִּילֵיפְּט.


  


מִדָּרוֹם בָּא סוּרְט / עִם מַכְחִיד-הַזְּמוֹרוֹת,



שֶׁמֶשׁ אֱלֵי-קְרָב / מַזְהִירָה מִן הַחֶרֶב,



קוֹרְסִים הָרֵי-הָאָבֶן, / כּוֹשְׁלוֹת הָעֲנָקִיּוֹת,



הֶל בּוֹלֵעַ בְּנֵי-אָדָם, / נִבְקָע הָרָקִיעַ.


  


צַעַר חָדָשׁ / בָּא עַל הֲלִין,



בְּצֵאת וַלְפָטֶר / לַעֲמוֹד בִּפְנֵי הַזְּאֵב,



וּבִפְנֵי סוּרְט / חוֹנֵק-בֶּלִי הַלָּבָן;



נָפוֹל תִּפּוֹל אָז / שִׂמְחָתָהּ שֶׁל פְרִיג.


  


אָז יָבוֹא וִידַר / בְּנוֹ שֶׁל וַלְפָטֶר,



הַגִּבּוֹר הֶעָצוּם / לְהִלָּחֵם בַּזְּאֵב:



יִתְקַע הַלַּהַב / בְּלֵב בֶּן-הָעֲנָק,



בְּעַד גְּרוֹנוֹ, / וְיִנְקוֹם אֶת אָבִיו.


  


גַּם בְּנָהּ שֶׁל הֲלוֹדִין / הַנֶּהְדָּר יָבוֹא אָז;



חוֹבֵק-הָאָרֶץ / יִפְעַר מְפַהֵק



אֶת לֹעוֹ הָרָחָב / עַד כִּפַּת הַשָּׁמָיִם –



אַךְ בְּנוֹ שֶׁל אוֹדִין / יוֹצֵא לִקְרַאת פְּרִיץ-הַחַיּוֹת.


 


יִכָּנַע לַחֲמָתוֹ / גּוֹנֵן-הָעוֹלָם;



הַכֹּל אֲנוּסִים / לַעֲזוֹב מוֹלַדְתָּם;



בְּנָהּ שֶׁל פְיוֹרְגִין / נִרְתַּע תֵּשַׁע פְּסִיעוֹת



לִפְנֵי הַנָּחָשׁ, / לֹא יֵחַת מִפְּנֵי רֶשַׁע.


  


הַחַמָּה מַשְׁחִירָה, / הָאָרֶץ שׁוֹקַעַת בַּיָּם,



נוֹפְלִים מִשָּׁמַיִם / הַכּוֹכָבִים הַצַּחִים;



נִתָּז הַקִּיטוֹר / וְחוֹנֵן-הַחַיִּים,



הַלַּהַב הֶחָם / לוֹקֵק הָרָקִיעַ.


  


גַּרְם נוֹבֵחַ בְּקוֹל / לִפְנֵי גְּנִיפַּהֶלִיר,



קוֹרֵעַ הַכְּבָלִים / בָּא-רָץ הַזְּאֵב!



הַרְבֵּה חָכְמָה עִמָּדִי, / אוּכַל לְהַרְחִיק שׁוּר



הַגּוֹרָל הַמָּר / הַקָּרֵב לָאֵלִים.


  


רוֹאָה אֲנִי הָאָרֶץ / בַּעֲלוֹתָהּ שׁוּב



מִתּוֹך הַמַּיִם, / עוֹטָה יֶרֶק רַעֲנָן;



עַל פְּנֵי אַשְׁדּוֹת-קֶצֶף / מְרַחֵף הַנֶּשֶר,



שׁוֹלֶה דָּגִים / עַל יַד נֵד הַסָּלַע.


 


עַל שְׂדֵה-אִידָפֶלְד / נִקְבָּצִים הָאָזִים,



מְדַבְּרִים בָּעֲנָק / חוֹבֵק-הָאָרֶץ,



נִזְכָּרִים בַּעֲלִילוֹת / הַמְּאֹרָעוֹת הַגְּדוֹלִים



וּבִכְתַב-הָרוּנוֹת / הָעַתִּיק לְאֵל עֶלְיוֹן.


  


שָׁם שׁוּב יִגָּלוּ / בְּתוֹךְ הָעֵשֶׂב



לוּחוֹת-הַזָּהָב / הַנִּפְלָאִים



אֲשֶׁר הָיוּ / לָאָזִים לְפָנִים.


––––––– / –––––––




עַל שָׂדֶה לֹא זָרוּעַ / יַעֲלוּ שִׁבֳּלִים,



וְכָלָה כָּל-הָרַע, / כִּי בַּלְדְּר יוֹפִיעַ.



בְּבֵית-הַמְנַצְּחִים שֶׁל הֲרוֹפְּט, / מְעוֹן אֵלֵי-הַקְּרָב,



יָבוֹאוּ הוֹד וּבַּלְדְּר, / – הֲתָבִינוּ שְׁאָר דְּבָרַי?


 


אָז יָבוֹר לוֹ הוֹנִיר / זְמוֹרַת-הַנְּבוּאָה


––––––– / –––––––


וּבְנֵי אֵַחֵי-טְוֶגִי / יֵשְׁבוּ עַל כִּסְאָם



בְּבֵית-הָרוּחוֹת הָרָחָב, / – הֲתָבִינוּ שְׁאָר דְּבָרַי?


  


אֲנִי רוֹאָה אוּלָם / עַל רָמוֹת הַגִּמְלֶה



מְצֻפֶּה זָהָב, / יָפֶה מֵחַמָּה:



שָׁם יָגוּרוּ / גְּדוּדֵי גִּבּוֹרִים



יֵהָנוּ מֵאֹשֶׁר / אֲשֶׁר לֹא יָסוּף.


  


מִמָּרוֹם יָבוֹא / הַשַּׁלִּיט הַנֶּהְדָּר



הַכֹּל-יָכוֹל / לְיוֹם-הַדִּין הַגָּדוֹל


––––––– / –––––––

––––––– / –––––––




וּלְמַטָּה יָעוּף / הַדְּרָקוֹן הָרָע,



הֶעָכָן הַנּוֹצֵץ / מִסֶּלַע-הַלֵּיל:



בְּאֶבְרוֹתָיו יִשָּׂא / נִידְהוֹג פְּגָרִים –



טָס עַל פְּנֵי שָׂדוֹת / אַךְ שָׁקֹעַ יִשְׁקָע.


  

חֲלוֹמוֹתָיו שֶׁל בַּלְדְּר


חָשׁוּ הָאָזִים / כֻּלָּם אֶל הַטִּינְג,



גַּם הָאָזוֹת / כֻּלָּן בָּאוּ לָעֲצָרָה:



הָאֵלִים הָאַדִּירִים / רָצוּ לָדַעַת



לָמָּה חֲלוֹמוֹת רָעִים / מְעַנִּים אֶת בַּלְדְּר.


  


וַיָּקָם אוֹדִין / הַיּוֹצֵר הָעַתִּיק,



שָׂם אֶת הָאֻכָּף / עַל גַּבּוֹ שֶׁל סְלֵיפְּנִיר;



רָכַב יָרַד / לְמַעֲמַקֵּי הַנִּיפְלְהֶל,



וְיָצָא לִקְרָאתוֹ / מִן הַמְּחִלָּה כֶּלֶב.


  


הָיָה חָזֵהוּ / מְגֹאָל בְּדָמִים,



הָלַךְ וְצָעַק / עַל אֲבִי-הַקֶּסֶם;



הִמְשִׁיךְ דַּרְכּוֹ אוֹדִין, / וַתִּרְעַד הָאָרֶץ –



עַד שֶׁהִגִּיעַ / לְבֵית-הֶל הָרָם.


  


נָטָה אוֹדִין / עַד שַׁעֲרֵי מִזְרָח,



יָדַע כִּי יִמְצָא שָׁם / גִּבְעַת הַקּוֹסֶמֶת;



שָׁר לָעַרְמוּמִית / אֶת קֶסֶם הַפֶּגֶר,



עַד שֶׁקָּמָה / בְּעַל כָּרְחָהּ הָאֵשֶׁת;



מִשִּׂפְתֵי הַמֵּתָה / מִלִּים נָבָעוּ:


  

הָרוֹאָה


מִי הוּא הַגֶּבֶר, / אֲנִי לֹא יָדַעְתִּי,



הַמַּכְרִיחֵנִי / לָצֵאת בִּדְרָכִים קָשׁוֹת?



כִּסַּנִי שֶׁלֶג, / הִכַּנִי גֶּשֶׁם,



טִפְטֵף עָלַי הַטַּל, / זֶה כְּבָר אֲנִי מַתִּי.


  

אוֹדִין


וֶגְטַם זֶה שְׁמִי / אֲנִי בְּנוֹ-וֶלְטַם;



סַפְּרִי דִּבְרֵי שְׁאוֹל-תָּחַת / (וַאֲנִי – מַה לְּמָעַל):



בִּשְׁבִיל מִי הַסַּפְסָלִים / עִם טַבָּעוֹת מַבְרִיקוֹת,



הָרְצָפוֹת הַנּוֹצְצוֹת / הַמְצֻפּוֹת בַּפָּז?


  

הָרוֹאָה


מוּעָדוֹת לְבַּלְדְּר / פֹּה מֵי-דְבַשׁ הוּזָדוּ,



מָגֵן מְכַסֶּה / הַמַּשְׁקֶה הַמְזַהֲרֵר;



אָבְדָה תִּקְוָתָם / שֶׁל בְּנֵי-הָאָזִים.



עַל אַפִּי דִּבַּרְתִּי, / רְצוֹנִי וְאֶשְׁתּוֹק.


  

אוֹדִין


אַל דֳּמִי לָךְ, הָרוֹאָה, / הוֹסִיפִי וְאִמְרִי לִי,



לְשׁוֹאֲלֵךְ עֲנִי / עַד יֵדַע הַכֹּל:



מִי וָמִי יִשְׁפּוֹךְ / דָּמָיו שֶׁל בַּלְדְּר,



וּמִי יְקָרֵב קֵץ / בְּנוֹ שֶׁל אוֹדִין?


 

הָרוֹאָה


הוֹד יָדוֹ תָּבִיא / הַחֹטֶר הַנֶּהְדָּר,



הוּא-הוּא שֶׁיִּשְׁפֹּךְ / דָּמָיו שֶׁל בַּלְדְּר



וִיקָרֵב קִצּוֹ / שֶׁל בֶּן-אוֹדִין;



עַל אַפִּי דִּבַּרְתִּי, / רְצוֹנִי וְאֶשְׁתּוֹק.


  

אוֹדִין


אַל דֳּמִי לָךְ, הָרוֹאָה, / הוֹסִיפִי וְאִמְרִי לִי,



לְשׁוֹאֲלֵךְ עֲנִי / עַד יֵדַע הַכֹּל:



מִי וָמִי יִנְקוֹם / דְּבַר הַתּוֹעֵבָה,



יָבִיא עַל גַּל-הַיְקוֹד / אֶת הוֹרְגוֹ שֶׁל בַּלְדְּר?


  

הָרוֹאָה


רִינְד תֵּלֵד אֶת וָלִי / בָּאוּלָם הַמַּעֲרָבִי,



בִּן-לַיְלָה יִלְחוֹם / בְּנוֹ שֶׁל אוֹדִין;



לֹא יִסְרוֹק רֹאשׁוֹ, / יָדָיו לֹא יִטּוֹל



עַד יִשְׁכַּב עַל גַּל הַיְקוֹד / שׂוֹנֵא בַּלְדְּר.



עַל אַפִּי דִּבַּרְתִּי / רְצוֹנִי וְאֶשְׁתּוֹק.


  

אוֹדִין


אַל דֳּמִי לָךְ, הָרוֹאָה, / הוֹסִיפִי וְאִמְרִי לִי,



לְשׁוֹאֲלֵךְ עֲנִי / עַד יֵדַע הַכֹּל:



בְּנוֹתָיו שֶׁל מִי הֵן / הַנְּמַסּוֹת בִּדְמָעוֹת,



זוֹרְקוֹת הַשָּׁמַיְמָה / צִיצִיּוֹת הַצַּוָּארִים?


  

הָרוֹאָה


אֲבָל אֵינְךָ וֶגְטַם / כַּאֲשֶׁר דִּמִּיתִי קֹדֶם,



כִּי אַתָּה אוֹדִין / הַיּוֹצֵר הָעַתִּיק.


  

אוֹדִין


אֵינֵךְ חַכְמַת-לֵב / גַּם לֹא נְבִיאָה אַתְּ,



בְּחֵיקֵךְ כִּי נָשָׂאת / אֶת-שְׁלשֶׁת הַטּוּרְסִים.


 

הָרוֹאָה


שְׂמַח בִּתְהִלָּתְךָ / וְשׁוּבָה הַבָּיְתָה!



לֹא יְבַקְּרֵנִי / כָּל-גֶּבֶר אַחֵר,



עַד יִפְרֹק לוֹקִי / הַכְּבָלִים מֵעָלָיו,



וּבָאוּ הַמְהָרְסִים / לְמַגֵּר אֶת הָאֵלִים!


  

שִׁירַת טְרִים


בָּעַר אַף וִינְגְטוֹר, / כְּשֶׁנִתְעוֹרֵר



וְלֹא מָצָא אֶת מְיוֹלְנִיר, / הַפַּטִּישׁ הָאַדִּיר;



נִעְנַע אֶת זְקָנוֹ, / נִפְנֵף שְׂעָרוֹ,



מָשַׁשׁ בְּיָדָיו / בֶּן -הָאֲדָמָה.





הָיְתָה מִלָּתוֹ / הָרִאשׁוֹנָה אֲשֶׁר אָמָר:



"שְׁמַע נָא, לוֹקִי, / אֲשֶׁר אוֹדִיעַ –



אִישׁ לֹא יְשַׁעֵר זֹאת / לֹא בִּמְלֹא -אֶרֶץ,



לֹא בַּשָׁמַיִם מִמַּעַל – / נִגְזַל מִן הָאָז הַפַּטִּישׁ!"


  


הָלְכוּ אֶל חֲצֵרָהּ / הַנֶּהְדָּר שֶׁל פְרֵיָה.



הָיְתָה מִלָּתוֹ / הָרִאשׁוֹנָה אֲשֶׁר אָמָר:



"הֲתַלְוִי לִי, פְרֵיָה / שִׂמְלַת -נוֹצוֹתַיִךְ



לְמַעַן אָשִׁיבָה / פַּטִּישִׁי שֶׁגָּזָלוּ?"


  


"בְּחֵפֶץ -לֵב / לְךָ אַשְׁאִילֶנָּה,



לוּ גַּם הָיְתָה מִכֶּתֶם / אוֹ מִכֶּסֶף מֵאִיר!"



אָז טָס לוֹקִי, / וְהַשִּׂמְלָה רוֹעֶשֶׁת,



זָנַח אֲחוֹרָיו / אֶת חַצְרוֹת הָאָזִים,



תֵּכֶף הִגִּיעַ / לִמְכוֹרַת הָעֲנָקִים.


 


יָשַׁב עַל הַגִּבְעָה טְרִים / אַלּוּף הַטּוּרְסִים,



קָלַע מוֹסְרוֹת -כְּלָבִים / מִכֶּתֶם אָדֹם,



הֵישִׁיר הֶחֱלִיק / רַעֲמוֹת -הַסּוּסִים.


טְרִים


מַה שְּׁלוֹם הָאָזִים? / מַה שְּׁלוֹם הָאֶלְבִּים?



מַה תָּבוֹא לְבַדְּךָ / אֶל רִיזֶנְהֵים?


 

לוֹקִי


רַע לָאָזִים / רַע לָאֶלְבִּים;



הַאַתָּה טָמַנְתָּ / פַּטִּישׁ הֲלוֹרִידִי?


 

טְרִים


אֲנִי טָמַנְתִּי / פַּטִּישׁ הֲלוֹרִידִי



שְׁמוֹנָה מִילִין עָמֹק / בְּחֵיק הָאֲדָמָה.



אִישׁ לֹא יְקַבְּלֶנּוּ / עַד אִם הֵבִיאוּ



הֵנָּה אֶת פְרֵיָה / לְאִשָּׁה לִי.


 


אָז טָס לוֹקִי, / הַשִּׂמְלָה רָעָשָׁה,



מְכוֹרַת הָעֲנָקִים / זָנַח אַחֲרָיו,



וְנֶחְפַּז עָף / אֶל חַצְרוֹת הָאָזִים.



וְעֹוד בֶּחָצֵר / פְּגָשׁוֹ טוֹר,



הָיְתָה מִלָּתוֹ / הָרִאשׁוֹנָה אֲשֶׁר סָח:


 


"הַהִצְלַחְתָּ בַּדֶרֶך / וּכְדָאִית טִרְחָתְךָ?



הַגֵּד בַּחוּץ פֹּה / סִפּוּרְךָ הָאָרֹךְ!



הַיּוֹשֵׁב מַרְבֶּה לִשְׁכֹּחַ / אֶת אֲשֶׁר לוֹ לְהַגִּיד,



וְהַשׁוֹכֵב / מְשַׁקֵּר בְּרָצוֹן!"


  

לוֹקִי


"הִצְלַחְתִּי בַּדֶּרֶךְ, / כְּדָאִית הַטִּרְחָה:



טְרִים אַלּוּף הַטּוּרְסִים / בְּיָדָיו הַמַּקֶּבֶת;



אִישׁ לֹא יְקַבְּלֶנָּה / עַד אִם הֵבִיאוּ



שָׁמָּה אֶת-פְרֵיָה / לִהְיוֹת אִשָּׁה לוֹ."


  


הָלְכוּ לַחֲצֵרָהּ / הַנֶּהְדָּר שֶל פְרֵיָה.



הָיְתָה מִלָּתוֹ / הָרִאשׁוֹנָה אֲשֶׁר סָח:



"קַשְּׁטִי עַצְמֵךְ פְרֵיָה, / בִּצְעִיף כְּלוּלוֹתַיִךְ,



עָלֵינוּ הַשְּׁנַיִם / לָלֶכֶת לְאֶרֶץ עֲנָקִים!"





בַּחֲמַת קֶצֶף / נָשְׁמָה-נָשְׁפָה פְרֵיָה.



בִּירַת הָאָזִים / רָעֹד רָעָדָה,



נִשְׁבַּר-נִשְׁמַט / עֲנַק-הַבְּרִיזִינְגִים:



"הַמְשֻגַּעַת לִגְבָרִים – / כָּך הָיוּ לִי קוֹרְאִים,



לוּ נָסַעְתִּי / אִתְּךָ לְאֶרֶץ עֲנָקִים."


  


הָאָזִים כֻּלָּם / נֶחְפְּזוּ אֶל הַטִּינְג,



גַּם הָאָזוֹת בָּאוּ לַמּוֹעָצָה כֻּלָּן;



שָׁם נוֹעָצוּ / הָאֵלִים הַמְהֻלָּלִים,



אֵיךְ יַחֲזִירוּ / פַּטִּישׁ-הֲלוֹרִידִי.


  


עָנָה הֵימְדַל / הַלָּבָן בָּאָזִים,



יוֹדֵעַ הַבָּאוֹת / כְּאַחַד הַוָּנִים:



"נְקַשֵּׁט אֶת-טוֹר / בְּצָעִיף שֶׁל כַּלָּה,



יִשָּׂא אֶת עֲנַק / הַבְּרִיזִינְגִים הָרָחָב.


 


תְּנוּ לוֹ הַטַּבַּעַת / עִם הַמַּפְתְּחוֹת הַמְקַשְׁקְשִׁים,



תְּרַחֵשְנָה שִׂמְלוֹת-נָשִים / עַל בִּרְכּוֹ סָבִיב,



קַשְּׁטוּ אֶת-חָזוֹ / בַּאֲבָנִים רְחָבוֹת,



עַטְּרוּ לְרֹאשׁוֹ / עֲדִי-אָמָּנִים."


  


עָנָה-אָמַר טוֹר / הָאַדִּיר בָּאָזִים:



"אִשָּׁה יְכַנוּנִי / הָאֵלִים הַמּוֹשְׁלִים,



אִם אֶתְקַשֵּׁט / בְּצָעִיף שֶׁל כַּלָּה."


 


עָנָה לוֹקִי / בְּנָהּ שֶׁל לוֹפֵי:



"הַחֲרִישָׁה, טוֹר, / חֲדַל מִלַּהַג!



בִּמְהֵרָה יִמְלְכוּ / הַטּוּרְסִים בְּאַזְגַּרְד,



אִם אֵינְךָ מַחֲזִיר / פַּטִּישְׁךָ שֶׁלְּךָ."


  


קִשְּׁטוּ אֶת טוֹר / בְּצָעִיף שֶׁל כַּלָּה,



בָּעֲנָק הָרָחָב / שֶל הַבְּרִיזִינְגִים.


  


נָתְנוּ לוֹ הַטַּבַּעַת / עִם הַמַּפְתְּחוֹת הַמְקַשְׁקְשִׁים,



רַחֲשוּ שִׂמְלוֹת-נָשִים / עַל בִּרְכּוֹ סָבִיב,



קִשְּׁטוּ חָזֵהוּ / בַּאֲבָנִים רְחָבוֹת



וְעִטְּרוּ לְרֹאשׁוֹ / עֲדִי-אָמָּנִים.


 


אָז אָמַר לוֹקִי, / בְּנָהּ שֶׁל לוֹפֵי:



"אֶתְחַפְּשָׂה כְּאָמָה / וְאֵלֵךְ אִתְּךָ,



עָלֵינוּ הַשְּׁנַיִם / לָלֶכֶת לְאֶרֶץ-עֲנָקִים."


  


דָּפְקוּ הַבַּיְתָה / הַתְּיָשִׁים הַמַּקְרִינִים,



אָסְרוּ לַיָּצוּל / הָרָצִים הַקַּלִּים.



הֶהָרִים נִבְקְעוּ, / בָּעֲרָה הָאָרֶץ,



טוֹר בֶּן-אוֹדִין / הָלַךְ לְאֶרֶץ-טוּרְסִים.


  


אָז אָמַר טְרִים, / אַלּוּף הַטּוּרְסִים:



"קוּמוּ הָעֲנָקִים, / עִרְכוּ הַסַּפְסָלִים;



הִנֵּה מְבִיאִים לִי / אֶת פְרֵיָה לְאִשָּׁה,



אֶת-בִּתּוֹ שֶׁל נְיוֹרְד / מִנּוּאָטוּן.


  


פָּרוֹת עִם קַרְנֵי פָּז / הוֹלְכוֹת הַחֲצֵרָה,



פָּרִים שְׁחוֹרִים מִשְּׁחוֹר / לְשִׂמְחַת לֵב טוּרְס;



לִי הַרְבֵּה סְגֻלּוֹת / וְקִשּׁוּטִי-חֶמֶד,



אַךְ פְרֵיָה הָאַחַת / חָסְרָה לִי.


  


יָרַד הָעֶרֶב / בָּא בְּעִתּוֹ,



הֵבִיאוּ לַטּוּרְסִים / מַשְׁקֵה הַשֵׁכָר,



אָכַל טוֹר שׁוֹר-פָּר / וּשְׁמוֹנֶה אִלְתִּיּוֹת



וְכָל מִינֵי-תַּבְלִין / שֶׁנּוֹעֲדוּ לַנָּשִׁים,



שָׁתָה בַּעְלָהּ שֶׁל סִיף / עוֹד שָׁלֹש חָבִיּוֹת מֵי-דְבָשׁ.


  


אָז אָמַר טְרִים / אַלּוּף הַטּוּרְסִים:



"אֵיפֹה מָצָאתָ מֵעוֹלָם / כַּלּוֹת זוֹלְלוֹת כָּזֹאת?



מִיָּמַי לֹא רָאִיתִי כַּלּוֹת / בּוֹלְעוֹת בֶּלַע שֶׁכָּזֶה,



אוֹ נַעֲרָה מַרְבָּה / לִשְׁתּוֹת מֵי-דְבָשׁ."


  


יָשְׁבָה לְיַד פְרֵיָה / הָאָמָה הַשְּׁנוּנָה,



יָדְעָה מַה לְּהָשִׁיב / לְדִבְרֵי הָעֲנָק:



"מְאוּם לֹא אָכְלָה פְרֵיָה / זֶה שְׁמוֹנָה לֵילוֹת,



כִּי כָּמְהָה מְאֹד / לְאֶרֶץ-עֲנָקִים."


  


הֵרִים צְעִיפָהּ / כִּי רָצָה לְנַשְׁקָהּ,



נִבְעַת וְנִרְתַּע / לְאֲחוֹרֵי הָאוּלָם:



"מַדּוּעַ מַבְרִיקוֹת כֹּה / עֵינֶיהָ שֶׁל פְרֵיָה?



אָמֹר אָמַרְתִּי: אֵשׁ / מִתְלַקַּחַת בָּהֶן."


 


יָשְׁבָה לְיַד פְרֵיָה / הָאָמָה הַשְּׁנוּנָה,



יָדְעָה מַה שֶּׁתָּשִׁיב / לְדִבְרֵי הָעֲנָק:



זֶה שְּׁמוֹנָה לֵילוֹת / לֹא סָגְרָה עֵינֶיהָ,



כִּי כָּמְהָה מְאֹד / לְאֶרֶץ-עֲנָקִים."


  


נִכְנְסָה אֲחוֹת טוּרְס / הַבָּאָה-בַיָּמִים,



הֵעֵזָּה לְבַקֵּשׁ / מַתְּנַת הַכַּלּוֹת:



"חִלְצִי מִזְּרוֹעֵךְ / צְמִידַיִךְ הָאֲדֻמִּים,



אִם יֵש נַפְשֵׁךְ / לָקַחַת לֵב-הַיְשִׁישָׁה,



לֵב-הַיְשִׁישָׁה / וְכָל-חַסְדָּהּ."


  


אָז אָמַר טְרִים / אַלּוּף הַטּוּרְסִים:



"הָבִיאוּ הַמַּקֶּבֶת / לְקַדֵּשׁ הַכַּלָּה,



שִׂימוּ אֶת מְיוֹלְנִיר / בְּחֵיק הַנַּעֲרָה



וְקַדְּשׁוּ בְּרִיתֵנוּ / בִּשְׁמָה שֶׁל וָר!"


  


צָחַק בְּחֻבּוֹ / לֵב הֲלוֹרִידִי,



הָאִישׁ הַקָּשֶׁה / לְמַרְאֵה הַמַּקָּבֶת.



הָרַג רִאשׁוֹן אֶת-טְרִים, / אֶת-אַלּוּף הַטּוּרְסִים,



וַיְפוֹצֵץ אֶת-זֶרַע / הָעֲנָקִים כֻּלָּם.


  


הִכָּה גַּם אֶת אֲחוֹת / הַטּוּרְס הַיְשִׁישָׁה



אֲשֶׁר בִּקְשָׁה / מַתְּנַת הַכַּלוֹת;



סָפְגָה מַכּוֹת / תַּחַת שִׁילִינְגִים,



וּמַהֲלֻמּוֹת-פַּטִּישׁ / תַּחַת צְמִידִים.



כָּכָה הֶחֱזִיר לוֹ / בֶּן-אוֹדִין הַמַּקָּבֶת.


  

שִׁירַת הִימִיר


פַּעַם יָשְׁבוּ יַחְדָּיו / הָאֵלִים הַמְנַצְּחִים



עַל מֵי-דְבַשׁ וְצַיִד, / אַךְ הַסְּעוּדָה דַּלָּה;



הֵם נִעְנְעוּ זְמוֹרוֹת, / הִתְבּוֹנְנוּ בַּדָּם:



מָצְאוּ אֶת-אֵגִיר / וְהִנֵּה עִמּוֹ שֶׁפַע.


 


יָשַׁב שׁוֹכְנִי-הַסֶּלַע / עָלֵז כְּנָעַר



אֲשֶׁר לוֹ עֵינַיִם / סְמוּיוֹת עֲדַיִן;



הִבִּיט בְּעֵינָיו / בֶּן-אִיג בְּעֹז-לֵב:



“עֲרֹך-נָא לָאָזִים / כֵּרָה בְּשֶׁפַע!”





הִטְרִיחַ הַגִּבּוֹר / אֶת חוֹבֵב-הַמְּדָנִים,



שֶׁחָשַׁב רָעָה / עַל הָאָזִים;



בִּקֵּשׁ מִבַּעְלָהּ שֶׁל סִיף / לְהָבִיא הַקַּלַּחַת:



“אָז אֲבַשֵּׁל בְּרָצוֹן / לְכֻּלְכֶם הַשֵּׁכָר.”


  


לֹא עָלָה בִּידֵי / הָאֵלִים הַמְהֻלָּלִים,



חַכְמֵי-הַעֵצוֹת, / לְהַמְצִיא הַקַּלַּחַת,



עַד שֶׁיָּעַץ טוֹבוֹת / טִיר תַּם-הַלֵּבָב



לַהֲלוֹרִידִי / וַיֹּאמַר בְּסוֹד:


 


"בְּאֶלִיוָגַר / מִזְרָחָה שֶׁמֶשׁ



חַי הִימִיר הֶחָכָם / עַל קְצֵה הַשָּׁמַיִם,



יֵשׁ לוֹ לְאָבִי / הָאַבִּיר קַלַּחַת,



כְּלִי רַב וְעָצוּם, / מִיל אֶחָד עֻמְקוֹ."


 


“הֲנוּכַל לָקַחַת / אֶת מַרְתִּיחַ-הַמָּיִם?”



“נוּכַל יְדִידִי, / אִם נָבוֹא בְּעָרְמָה.”


  


נֶחְפְּזוּ נָסְעוּ / אוֹתוֹ יוֹם כֻּלּוֹ



מִן הָאַזְגַּרְד / עַד מְעוֹן אֶגִיל;



הוּא נָטַר הַתְּיָשִׁים / עֲטוּרֵי הַקַּרְנָיִם.



נִכְנְסוּ שְׁנֵיהֶם / בְּאוּלַם הִימִיר.


  


מָצָא טִיר אֶת אֵם-אָבִיו / אֲשֶׁר שְׂנֵאָהּ מְאֹד



הָיוּ לָהּ רָאשֶׁיהָ / מֵאָה פַּעַם תֵּשַׁע;



אַךְ הָאַחֶרֶת נוֹצְצָה / בְּזָהָב, וְהֵבִיאָה,



לִבְנַת-הַגַּבּוֹת / לִבְנָהּ אֶת הַשֵּׁכָר.


  


"קְרוֹבָם שֶׁל הָעֲנָקִים, / עַזֵּי-הַלֵּבָב,



אֶת שְׁנֵיכֶם אוֹשִׁיבָה/ תַּחַת הַדְּוָדִים,



כִּי, לְצַעֲרִי, צָרָה / עֵין דּוֹדִי בְּאוֹרְחִים



לְעִתִּים קְרוֹבוֹת, / וְסָר וְזָעֵף הוּא מְאֹד."


 


בְּשָׁעָה מְאֻחֶרֶת / שָׁב הָעֲנָק מִן הַצַּיִד,



הִימִיר הַמְכֹעָר / שָׁב נִכְנַס הַבָּיְתָה.



נִכְנַס בָּאוּלָם, / קִשְׁקְשׁוּ לְשׁוֹנוֹת-קֶרַח



עַל פְּנֵי הַבָּא, / כִּי קָפָא הַיָּעַר.


 


אַשְׁרֶיךָ, הִימִיר! / יִשְׂמַח מְאֹד לִבֶּךָ!



הַיּוֹם בָּאוּלָם פֹּה / נִכְנַס בְּנֵנוּ,



אֲשֶׁר לוֹ חִכִּינוּ, / מִדְּרָכִים רְחוֹקוֹת.



נוֹסַף-נִלְוָה אֵלָיו / אוֹיְבוֹ שֶׁל הֲרוֹדְר,



גוֹמֵל הַחֲסָדִים לְאָדָם, / וֵיאוֹר זֶה שְׁמוֹ.





"רְאֵה אֵיךְ הֵם יוֹשְׁבִים / עַל גַּמְלוֹן הָאוּלָם,



מֵאֲחוֹרֵי הַקּוֹרָה / שָׁם יִסְתַּתָּרוּ."



פָּקְעָה הַקּוֹרָה, / כִּי הִבִּיט הָעֲנָק בָּהּ.



נִשְׁבַּר לִשְׁבָבִים / הָעַמּוּד הֶחָזָק.


  


שְׁמוֹנָה דְּוָדִים / נִדַּרְדְּרוּ מִן הַקֶּרֶשׁ,



אֶחָד נָגוּד קָשֶׁה / נִשְׁאַר שָׁלֵם.



קָרְבוּ הָאוֹרְחִים. / הָעֲנָק הַשָּׂב



הִבִּיט בְּעֹז / בְּעֵינֵי שׂוֹנְאוֹ.


 


טוֹבוֹת לֹא נִבָּא לוֹ / לִבּוֹ כִּי זֶה בָּא,



הַמַאֲבִיל / נְשׁוֹת הַטּוּרְסִים.



לְהוֹצִיא מִן הָרֶפֶת / שְׁלֹשָׁה פָּרִים,



לִשְׁלוֹק בְּשָׂרָם / צִוָּה הָעֲנָק.


  


כֻּלָם הִתְקַצְּרוּ / כִּמְלֹא רֹאשׁ אֶחָד,



הֶעֱבִירוּ בְּשָׂרָם / בַּלֶּהָבָה.



בַּעְלָה שֶל סִיף / אָכַל הוּא לְבַדּוֹ



שְׁנַיִם פָּרִים / עַד שֶהִרְגִּיעַ.


  


לֹא קָטְנָה בְּעֵינֵי / אִישׁ-סוֹדוֹ שֶׁל הֲרוּנְגְנִיר



לְבֶן-הַשֵּׂעָר / סְעוּדָתוֹ שֶׁל טוֹר:



"אָנוּ הַשְּׁלֹשָה / אֲנוּסִים לֶאֱכֹל



בָּעֶרֶב הַבָּא / מִשְּׁלַל הַצַּיִד."


 


הָיָה וֵיאוֹר נָכוֹן / לַחְתּוֹר הַיָּמָּה



אִלּוּ נָתַן לוֹ הָעֲנָק / עַז-הַלֵּב הַשִּׁרְגוּג:



"לֵךְ אֶל הָעֵדֶר, / מַכְנִיעַ הַטּוּרְסִים,



אִם לִבְּךָ מְלָאֲךָ / לָקַחַת שִׁרְגוּג."





"סְבוּרַנִי, בְּנָקֵל / תִּקַּח יָדְךָ



פִּתְיוֹן-חֶמֶד / מִבְּשַׂר הַשּׁוֹר."



מִהֵר וּבָא / טוֹר אֶל הַיַּעַר,



שָׁם פָּגַע בּוֹ / שׁוֹר-פָּר שָׁחוֹר.


  


מַכְנִיעַ הָעֲנָקִים / קָטַם מִן הַפָּר



אֶת מִשְׂגַּבּוֹ / עֲטוּר הַקַּרְנָיִם.


——-  ——-




"רַב-הָאֳנִיָּה / רוֹאֶה אֲנִי שְׁקִידָתְךָ



נוֹרָאָה אַף / מִמְּנוּחָתְךָ בַּסְּעוּדָה."


——-  ——-




צִוָּה בַּעַל-הַתְּיָשִׁים / לְבֶן-הַקּוֹפִים



לְהוֹסִיף וּלְהַסִּיעַ / אֶת סוּס-הַגְּלִילִים;



אַךְ הַטּוּרְס הֶעָצֵל / עָנָה לוֹ וְאָמָר:



אֵין לִי כָּל חֵשֶׁק / לְהוֹסִיף וְלָשׁוּט.


  


שָׁלָה הִימִיר הָעַז / שְׁנַיִם תַּנִּינִים



בְּחַכָּה מִן הַיָּם / תּוֹךְ זִיזָה אַחַת;



אֲבָל בְּנוֹ-אוֹדִין, / וֵיאוֹר הֶעָרוּם,



כּוֹנֵן תֹּרֶן / מָתַח הֶחָבֶל.


  


גּוֹנֵן-הָאָדָם, / מַכֵּה-הַנְּחָשִׁים,



קָבַע בַּחַכָּה / רֹאשׁוֹ שֶׁל הַשּׁוֹר.



בְּתֵאָבוֹן חָטַף / שׂוֹנֵא-הָאֵלִים,



חוֹבֵק-הָאֲרָצוֹת / אֶת נֵתַח-הֶחָמֶד.


  


בְּיָד אַמִּיצָה / שָׁלָה טוֹר עַז-הַלֵּב



הַתּוֹלַעַת הָאַרְסִית / אֶל שְׂפַת הַסִּירָה;



מִמַּעַל הָלַם, / הִכָּה בַּמַּקָּבֶת



אֶת הַר-הַשֵּׂעָר / שֶׁל אֲחִי הַזְּאֵב.


 


הָמוּ הַמִּפְלָצוֹת, / נָתְנוּ קוֹל הֶהָרִים;



רָעֹד רָעֲדָה / אֶרֶץ-קְדוּמִים.


——-  ——-


וְשָׁמַט הַיָּמָּה / הַדָּג הַמּוּזָר


  

——-  ——-


הוֹסִיפוּ וְחָתְרוּ, / וַיָּלֶן הָעֲנָק,



יָשַב עַל הָרְצוּעָה, / שָׁתַק-הֶחֱרִישׁ.



וַיֵּט הַסִּירָה. / אֶל הַיַּבָּשָׁה.


  


"עָלֶיךָ לַחְלֹק / אִתִּי שְׁאָר הָעֲבוֹדָה,



הַדֵּק בַּחוֹף / אֶת תֵּישׁ-הַמָּיִם,



אוֹ שָׂא הַתַּנִּינִים / אֶל מְעוֹנֵנוּ



דֶּרֶך הַנָּקִיק / בְּצֵל הַיָּעַר."


 


אָז תָּפַס הֲלוֹרִידִי / אֶת קְנֵה הַחַרְטוֹם,



הֵרִים הַסִּירָה / עִם מֵי-הַקַּרְקַע,



נָשָׂא לְבַדּוֹ / עַד חֲצַר הַטּוּרְסִים



מְשׁוֹטִים, גִּיגִית-מַשְׁאֵב / וַחֲזִירֵי-הַדֳּכִי.


  


אוּלָם הַטּוּרְס / הָעִקֵּשׁ רָצָה



לְהוֹסִיף הִתְחָרוֹת / עִם טוֹר בְּכֹחַ,



אִם-גַּם הֵיטִיב / חֲתוֹר בִּמְשׁוֹטִים;



אַדִּיר יְכֻנֶּה / אַךְ מְפוֹצֵץ-הַכּוֹס.





תָּפַס הֲלוֹרִידִי / בְּכַפָּיו הַגָּבִיעַ,



נִפֵּץ בַּזְּכוּכִית / אֶת עַמּוּד הָאֲבָנִים,



בְּשִׁבְתּוֹ בָּקַע / הָעַמּוּד לִשְׁנָיִם;



אֶפֶס שְׁלֵמָה שָׁבָה / הַכּוֹס לִידֵי הִימִיר.


  


בְחָכְמָה יָעֲצָה / רַעְיָתוֹ הַנֶּחְמָדָה



שֶׁל הָעֲנָק / כִּי יָדָעָה:



"זְרֹק בְּגֻלְגֹּלֶת / הָעֲנָק הֶעָצֵל,



מִצְחוֹ כִּי קָשֶׁה / מִכָּל זְכוּכִית."


 


הִתְרוֹמֵם לְמֶחֱצָה / אֲדוֹן-הַתְּיָשִׁים,



קָם הִתְאַזֵּר / בְּכָל כֹּחַ הָאָזִים,



שָׁלֵם נִשְׁאַר / גַּרְדֹּם-הַקּוֹבַע,



אַךְ כּוֹס-הַיַּיִן / הַיָּפָה נִשְׁבָּרָה.


  


"הִנֵּה אָבְדָה לִי / סְגֻלַּת-חֶמֶד,



כִּי הַסֵּפֶל נֻפַּץ / שָׁמַט מֵחֵיקִי;



לֹא אוֹסִיף כְּקֶדֶם / לֵאמֹר בְּשִׂמְחָה:



הִנֵּה תָּסַסְתָּ / מַשְׁקֵה שֵׁכָרִי!"


  


" הִנֵּה הַתְּנַאי / הָאַחֲרוֹן בַּתְּנָאִים:



הוֹצֵא אֶת דּוּד / הַשֵּׁכָר הַחוּצָה!"



פַּעֲמַיִם תָּפַס / טִיר הֶחָבִית,



אַךְ זוֹ לֹא זָזָה, / נִשְׁאֲרָה עַל עָמְדָהּ.


 


תָּפַס אֲבִי-מוֹדִי, / אָחַז בַּשָּׂפָה



וְיָרַד לָאוּלָם / בְּמַעֲלוֹתָיו,



הֵרִים הַקַּלַּחַת / בַּעְלָהּ שֶׁל סִיף עַל רֹאשׁוֹ,



צִלְצְלוּ אָזְנֶיהָ / עַל עֲקֵבָיו.


  


לֹא הִרְבּוּ לִנְסוֹעַ / וּבְנוֹ שֶׁל אוֹדִין



צָפָה הִתְבּוֹנֵן / עוֹד פַּעַם לְאָחוֹר.



רָאָה הֲמוֹן עָם / מְרֻבֵּי-רָאשִׁים



בָּא עִם הִימִיר / מִמְּעָרוֹת הֶהָרִים.


  


אָז הֵרִים מֵעָרְפּוֹ / אֶת-הַדּוּד הַגָּדוֹל,



אֶת מְיוֹלְנִיר, / צְמֵא הַדָּמִים,



וַיַּךְ אֶת תַּנִּינֵי / הַשְּׁמָמָה, אֶת-כֻּלָּם.


 


לֹא הִרְבּוּ לִנְסוֹעַ, / וַיִּצְנַח אֶחָד



מִתְּיָשָׁיו שֶׁל הֲלוֹרִידִי / אַרְצָה, מֵת לְמֶחֱצָה:



הִתְעַקְּמָה רַגְלוֹ הַדַּקָּה / שֶׁל סוּס-הַיָּצוּל



בַּעֲווֹנוֹ שֶׁל לוֹקִי / רַע הַלֵּבָב.


  


הֲלֹא שְׁמַעְתֶּם: / מִי וָמִי הֵיטִיב



לְסַפֵּר דְּבָרִים / עַל סוֹד-הָאֵלִים?



מַה קִּבֵּל טוֹר / כִּתְמוּרָה



מִידֵי הַטּוּרְס – / שְׁנֵי בָּנָיו קְנָס נָתָן…


  


אֶל טִינְג הָאֵלִים / בָּא טוֹר הָאַדִּיר,



בְּיָדָיו הַקַּלַּחַת / שֶׁהָיְתָה לְהִימִיר;



עַתָּה יְכוֹלִים הָאֵלִים / לִסְבֹּא כַּהֲלָכָה



בְּאוּלַם אֵגִיר / עַד יְמוֹת הַסְּתָו.


  

מִלְחֶמֶת הַדְּבָרִים שֶׁל לוֹקִי

  

אֵגיר (הוא גם גִימִיר) עשה משתה בירה לאלים אחרי שקיבל לידיו את הדוּד הגדול (כמסופר ב“שירת הִימיר” – שיר IV ) אל המשתה באו אוֹדין ופְרִיג אשתוֹ. טוֹר לא בא כי נסע אז מזרחה שמש, אבל אשתו סִיף באה אל המשתה. היו שם בְּרַאגִי ואשתו אִידוּן; היה גַם טִיר גִידם ידו האחת, שכירסם הזאב פֶנְדִּיר, בהיותו כפות. בא נְיוֹרְד עם אשתו סְקַאדִי, פְרֵיר וּפְרֵיָה, בא גם וִידַר, בנו של אוֹדִין, כן בא לוֹקִי והמשרתים של פרֵיר – בִּיְגוִיר וּבֵּילָא, ועוד כמה מן הָאָזִים והָאֶלְבִּים. אֵגִיר היו לו שני משרתים, פִימַאפֶנג ואֶלְדִּיר. במקום האש האיר לאלים הזהב; השֵׁכר עצמו הגיש את עצמו. האורחים שיבחו מאוד את המשרתים הזריזים של אֵגִיר. לוֹקי לא יכול לשאת זאת והרג את פִימַאפֶנג. אז הרימו האלים את מגיניהם ופרצו בקריאות זעם כנגד לוֹקִי וגירשוהו מהבית היערה. אחר-כך ישבו שוב למשתה. לוקי חזר ומצא בחוץ את אֶלְדִּיר, פתח לוֹקִי ואמר לו:

לוֹקִי


אֱמֹר זֹאת אֶלְדִּיר, / עַד שֶתִּפְסַע



פְּסִיעָה אַחַת קָדִימָה:



מָה אוֹמֵר קְהַל אֵלֵי-הַקְּרָב / עַל כַּד הַשֵּׁכָר



בְּהֵיכַל הַטִּירָה פְּנִימָה?


  

אֶלְדִּיר


קְהַל אֵלֵי-הַקְּרָב / מְהַלֵּל כְּלֵי-מִלְחַמְתָּם



וְרוּחַ גְּבוּרָתָם;



אֵין בְּפִי הָאָזִים וְהָאֶלְבִּים / הַנִּמְצָאִים פֹּה



מִלָּה בְּשִׁבְחֶךָ.


  

לוֹקִי


רְצוֹנִי לְהִכָּנֵס / בְּהֵיכַל אֵגִיר



לִרְאוֹת קְהַל הַשּׁוֹתִים;



אָבִיא כַּעַס / וּסְלִידָה לָאָזִים



וְאַטִּיל מָרָה בִּתְמַד-זְהָבָם.


 

אֶלְדִּיר


בְּהִכָּנֶסְךָ / בְּהֵיכַל אֵגִיר



לִרְאוֹת קְהַל הַשּׁוֹתִים,



לְנַבֵּל בְּבוּז / אֶת הָאֵלִים הַנֶּהְדָּרִים,



נַגֵּב יְנַגְּבוּהוּ בָּךְ.


  

לוֹקִי


דַּע לְךָ אֶלְדִּיר / כִּי אִם נִלָּחֵם



בְּדִבְרֵי שִׂנְאָה שְׁנֵינוּ;



יָדִי בִּתְשׁוּבָתִי / תְּהִי עַל הָעֶלְיוֹנָה,



אִם תָּקִים שָׁאוֹן רַב מִדַּי.




בתוך כך נכנס לוקי בהיכל, וכשראו המסובים מי הנכנס, נשתתקו כולם.

לוֹקִי


צָמֵא נִכְנַס לוֹפְּט / בַּהֵיכָל הַזֶּה,



מִדֶּרֶךְ רְחוֹקָה בָּא



לְבַקֵּשׁ מִן הָאָזִים, / אוּלַי יִתֵּן לוֹ אֶחָד



מִמַּשְׁקֵה הַתְּמָד הָרַךְ.


  


אֵיךְ נֶאֱלַמְתֶּם, / הָאֵלִים הַגֵאִים?



אֵינְכֶם פּוֹצִים פֶּה?



תְּנוּ מָקוֹם / וּמוֹשָׁב לִי בַּמִּשְׁתֶּה,



אוֹ תְּצַוּוּ וְאֵלֵךְ מִפֹּה.




בְּרַאגִי


מָקוֹם וּמוֹשָׁב / לְךָ בַּמִּשְׁתֶּה



לְעוֹלָם לֹא יִתְּנוּ הָאָזִים;



כִּי יוֹדְעִים הָאָזִים / שֶׁלֹא כָּל יְצוּר



יָאֶה לוֹ לָבוֹא בֶּחָג.


  

לוֹקִי


הֲלֹא תִּזְכּוֹר אוֹדִין, / כִּי בִּימֵי קֶדֶם



עֵרַבְנוּ שְׁנֵינוּ דָּמֵינוּ?



נִשְׁבַּעְתָּ לֹא לִטְעוֹם / לְעוֹלָם מִן הַשֵּׁכָר,



אִם לֹא יַצִּיעוּ לִשְׁנֵינוּ גַּם יָחַד.


 

אוֹדִין


קוּמָה וִידַר, / יֵהָנֶה אֲבִי-הַזְּאֵב



מִמּוֹשָׁב שֶׁל אוֹרֵחַ;



וְלֹא יְנָאֲצֶנּוּ / לוֹקִי בַּחֲרָפוֹת



בְּהֵיכַל-אֵגִיר פְּנִימָה.


  

וִידַר קם ומזג ללוֹקִי, אבל קודם שזה שתה פנה אל האָזים ואמר:

  


שָׁלוֹם לָאָזִים! / שָׁלוֹם לָאָזוֹת!



וּלְכָל הָאֵלִים הַקְּדוֹשִׁים מְאֹד!



מִלְּבַד לָאֶחָד / בֵּין הָאָזִים אַךְ הוּא:



בְּרַאגִי, הַיּוֹשֵׁב שָׁם עַל הַסַּפָּה.




בְּרַאגִי


סוּס וְטַבַּעַת / מֵאוֹצְרוֹתַי אֲנִי נוֹתֵן



וּקְנָס בְּרַאגִי מַצִּיעַ – חֶרֶב,



לְבַל תַּרְעִים / מִתּוֹךְ צָרוּת-עַיִן;



סוּר מֵחֲרוֹן אַף הָאֵלִים!




לוֹקִי


לְעוֹלָם לֹא תֵּעָשֵׁר / בְּסוּסִים וְטַבָּעוֹת



מִזֶּה וּמִזֶּה תּוֹחַלְתְּךָ נִכְזָבָה;



אֵין בָּאָזִים וּבָאֶלְבִּים, / בְּכֻלָּם פֹּה יַחַד,



נִמְנָע מִקְּרָב כָּמוֹךָ,



מְפַחֵד מִפְּנֵי יְרִיָּה!


  

בְּרַאגִי


אִלּוּ הָיִיתִי / בַּחוּץ וְלֹא בִּפְנִים



הֵיכָלוֹ שֶׁל אֵגִיר,



מְהֵרָה נָשָׂאתִי / רֹאשְׁךָ בְּכַפִּי,



שִׁלּוּמִים בְּעַד הַכָּזָב.


  

לוֹקִי


עֹז רוּחֲךָ בַּכִּסֵּא / וּבָעֲלִילָה מְפַגֵּר



אַתָּה בְּרַאגִי, הֲדַר הַסַּפְסָל!



צֵא לִקְרַב-בֵּינַיִם / אִם חָרָה אַפֶּךָ:



אַמִּיץ-הַלֵּב אֵינוֹ מַרְבֶּה מַחֲשָׁבוֹת.


  

אִידוּן


הִתְעַשֵּׁת, בְּרַאגִי, / רְאֵה בִּקַּשְׁתִּיךָ



כִּי בוֹ בָּחַר אוֹדִין לְבֵן-כְּנַפְשׁוֹ,



עַל כֵּן אַל תִּשְׁעֶה / אֶל לוֹקִי בַּחֲרָפוֹת



בְּהֵיכָלוֹ שֶׁל אֵגִיר.


 

לוֹקִי


שִׁתְקִי, אִידוּן, / שֶׁאֵין בַּנָּשִׁים כְּמוֹתֵךְ



לִרְדּוֹף אַחֲרֵי גְּבָרִים!



בִּזְרוֹעוֹתַיִךְ הַצַּחוֹת / הִתְרַפַּקְתְּ עָלָיו,



אֲשֶׁר דָּקַר חֲזֵה אָחִיךְ.




אִידוּן


אֵינֶנִּי דּוֹבֶרֶת לְלוֹקִי / בְּדִבְרֵי חֲרָפוֹת



בְּהֵיכַל אֵגִיר פְּנִימָה;



אֲנִי פִּיַּסְתִּי אֶת בְּרַאגִי, / הֲלוּם הַשֵּׁכָר,



וְלֹא יָעִיר כַּעַס מִלְחָמָה.


  

גֶּפְיוֹן


אֵיכָה תַּרְהִיבוּ פֹּה פְּנִימָה / הָאָזִים הַשְּׁנַיִם,



לְהִלָּחֵם בְּדִבְרֵי-אֵיבָה?



יָצָא שֵׁם לְלוֹקִי – / פֶּה דּוֹבֵר נְבָלָה,



שׂוֹנֵא הוּא יְצוּרֵי הָעוֹלָם כֻּלָּם.


  

לוֹקִי


שִׁתְקִי אַתְּ גֶּפְיוֹן, / לְעוֹלָם לֹא אֶשְׁכַּח



מִי פִּתָּךְ לְתַאֲוָה;



הָעֶלֶם הַלָּבָן / הוּא נָתַן לָךְ עֲדִי,



וַתְּשַׁלְּבִי סְבִיבוֹ יְרֵכַיִךְ.


  

אוֹדִין


מְטֹרָף אַתָּה, לוֹקִי / וַחֲסַר-לֵב



אִם תָּעִיר קִצְפָּהּ שֶׁל גֶּפְיוֹן,



כִּי הִיא יוֹדַעַת / יָפֶה כְּמוֹתִי



אֶת גוֹרַל הָעוֹלָם.


  

לוֹקִי


שְׁתֹק, אַתָּה אוֹדִין, / הַמְחַלֵּק בְּלֹא צֶדֶק



אֹשֶׁר קְרָבוֹת בֵּין לוֹחֲמִים!



לְעִתִּים קְרוֹבוֹת / נָתַתָּ נִצָּחוֹן לָרַע בַּקְּרָב,



לַאֲשֶׁר לֹא הָיִיתָ צָרִיךְ לָתֵת.


  

אוֹדִין


אִם נָתַתִּי מִיָּמַי / לָרַע נִצָּחוֹן בַּקְּרָב,



לַאֲשֶׁר לֹא צָרִיךְ הָיִיתִי לָתֵת:



בִּפְנִים הָאֲדָמָה / עָשִׂיתָ שְׁמוֹנָה סְתָוִים,



חָלַבְתָּ פָּרוֹת בִּדְמוּת נַעֲרָה;



גַּם בָּנִים יוּלְדוּ לְךָ שָׁם,



הָיָה טִבְעֲךָ טֶבַע נָשִׁים.


  

לוֹקִי


אוֹמְרִים עָלֶיךָ: / בְּסַמְסֵי כָּשַׁפְתָּ,



מַעֲשֵי קוֹסְמִים בְּיָדֶיךָ,



מִבַּיִת אֶל בַּיִת / הָלַכְתָּ כִּמְכַשֵּׁפָה,



הָיָה טִבְעֲךָ טֶבַע נָשִׁים.


  

פְרִיג


מוּטָב שֶׁלֹּא תְּסַפְּרוּ / בְּאָזְנֵי הַגְּבָרִים



אֶת אֲשֶׁר עוֹלַלְתֶּם,



שְׁנֵי הָאָזִים, / בַּיָּמִים מִקֶּדֶם,



אֲשֶׁר הָיָה – לֹא יִזָּכֵר!


  

לוֹקִי


שִׁתְקִי אַתְּ, פְרִיג, / אִשְׁתוֹ שֶׁל פְיוֹרְגִין



אַתְּ, זוֹנָה, הָיִית תָּמִיד מְשֻׁגַּעַת לִגְבָרִים:



אִשְׁתוֹ שֶׁל וִידְרִיר / נָטְתָה אֶת חַסְדָּהּ



לְוִילִי וּלְוֶה – לַשְּׁנָיִם.


  

פְרִיג


אִלוּ הָיָה לִי / בְּהֵיכַל אֵגִיר פְּנִימָה



בֵּן שֶׁרוּחוֹ כְּרוּחוֹ שֶׁל בַּלְדְּר,



לֹא מָצָאתָ מוֹצָא / מִפְּנֵי בְּנֵי הָאָזִים



עַד שֶׁטָּעַמְתָּ טַעַם דּוּ-קְרָב.


 

לוֹקִי


רְצוֹנֵךְ, פְרִיג, / שֶׁאוֹסִיף וַאֲסַפֵּר



עַל הָאָסוֹן אֲנִי הֵמַטְתִּי?



הֲלֹא זֹאת אַשְׁמָתִי, / שֶׁאֵינֵךְ רוֹאָה לְעוֹלָם



אֶת-בַּלְדְּר עוֹשֶׂה דַּרְכּוֹ הֵיכָלָה.


  

פְרֵיָה


מְשֻׁגע אַתָּה, לוֹקִי, / שֶׁאַתָּה מְסַפֵּר עֲלִילוֹת,



שֶׁעֲשׂוּכֶם לְשֵׁם וּלִשְׁנִינָה,



יוֹדַעַת פְרִיג / אֶת-כָּל הַבָּאוֹת



אִם כִּי אֵינָה מַגֶּדֶת בְּעַצְמָהּ.


  

לוֹקִי


שִׁתְקִי אַתְּ, פְרֵיָה, / אֲנִי יְדַעְתִּיךְ הֵיטֵב,



אֵינֵךְ נְקִיָּה מֵעָוֹן:



כָּל הָאָזִים וְהָאֶלְבִּים / אֲשֶׁר פֹּה בַּבָּיִת,



אֶת כֻּלָּם בְּחֵפֶץ-לֵב הֶאֱשַׁרְתְּ.


 

פְרֵיָה


לְשׁוֹנְךָ כּוֹזֶבֶת, / אֲבָל יָמִים יָבוֹאוּ, אֲשׁוּרֵם,



וְהִיא תְּזַמֵּר לְךָ אָסוֹן.



יִזְעָמוּךָ הָאָזִים, / יִזְעָמוּךָ הָאָזוֹת,



בְּיָגוֹן עוֹד תָּשׁוּב בָּיְתָה.


  

לוֹקִי


שִׁתְקִי אַתְּ, פְרֵיָה, מְכַשֵּׁפָה רָעָה



מֻכְתֶּמֶת בְּזָדוֹן מְאֹד:



אוֹתָךְ מְצָאוּךְ הָאֵלִים / בְּמִטָּתוֹ שֶׁל אָחִיךְ



וְשָׁם נָפַחַתְּ נְפִיחָה.


  

נְיוֹרְד


קָטָן הַנֶּזֶק / אִם אִשָׁה יְפַת-תֹּאַר



מוֹצֵאת לָהּ יְדִיד אוֹ עוֹגֵב;



אַךְ תָּמוּהַּ הַדָּבָר / שֶׁאָז-נְקֵבָה,



שֶׁיָלַד יְלָדִים, אֵלֵינוּ בָּא.


  

לוֹקִי


שְׁתוֹק אַתָּה, נְיוֹרְד, / כַּמָּה פְּעָמִים שְׁלָחוּךָ



בֶּן-תַּעֲרוּבוֹת אֶל הָאֵלִים,



סִיר-שֶׁתֶן הָיִיתָ, וּבְנוֹתָיו שֶׁל הִימִיר



הֵטִילוּ מַיִם בְּפִיךָ.


  

נְיוֹרְד


תְּמוּרָה קִבַּלְתִּי / וְאִם הִרְחִיקוּ לְשַׁלְּחֵנִי



אֶל הָאֵלִים בֶּן-תַּעֲרוּבוֹת;



נַעַר קָנִיתִי, / שֶׁאֵין לוֹ שׂוֹנְאִים,



רִאשׁוֹן לָאָזִים יְכַנּוּהוּ.




לוֹקִי


אַל תַּגְבִּיהַּ גָּבֹהַּ / עַד מְאֹד, נְיוֹרְד!



אֶת אֲשֶׁר שָׁמַעְתִּי לֹא אַעְלִים:



מֵחֵיק אֲחוֹתְךָ / נוֹלַד לְךָ בֵּן



אֲשֶׁר יִצְלַח כַּמְקֻוֶּה.


  

טִיר


פְרֵי הוּא הָרִאשוֹן / בְּכָל הַגִּבּוֹרִים,



אֲשֶׁר בְּהֵיכַל הָאָזִים;



אֵינוֹ מַקְנִיט לֹא / אֵשֶׁת-אִישׁ אַף לֹא יַלְדָּה,



נוֹתֵן לַכְּפוּתִים דְּרוֹר.


  

לוֹקִי


שְׁתֹק, אַתָּה טִיר, / מֵעוֹלָם לֹא הִצְלַחְתָּ



לִהְיוֹת מְתַוֵּךְ בֵּין גֶּבֶר לְגָבֶר:



אֶת זְרוֹעַ יְמִינְךָ / קָרַע לְךָ פֶנְרִיר,



בְּרָצוֹן אֲנִי זוֹכֵר אֶת זֹאת.


  

טִיר


לִי אָבְדָה הַיָּד / לְךָ – הַזְּאֵב הַנֶּהְדָּר,



צַעֲרִי וְצַעַרְךָ – מַחְסוֹר.



גַּם לֹא טוֹב לְפֶנְרִיר, / הַמְחַכֶּה בְּשַׁלְשְׁלָאוֹת,



עַד שֶׁיַּגִּיעַ לֵיל-הָאֵלִים.


  

לוֹקִי


שְׁתֹק, אַתָּה טִיר, / הֲלֹא קָרָה לְאִשְׁתְּךָ



שֶׁיָּלְדָה יֶלֶד לִי;



לֹא שִׁלְּמוּ לְךָ / בְּעַד הַחֶרְפָּה גַּם פְּרוּטָה,



אוֹ שָׁוֶה אַמָּה, עָלוּב שֶׁכְּמוֹתְךָ!


  

פְרֵיר


פֶנְרִיר שׁוֹכֵב / מוּל שֶׁפֶך הַנָּהָר



עַד שֶׁיְּכֻלּוּ הָאֵלִים;



גַּם אַתָּה תָּבוֹא בִּמְהֵרָה, / אִם לֹא תִּשְׁתּוֹק,



בִּכְבָלִים, הוֹי, נוֹגֵד-הָרָעוֹת.


 

לוֹקִי


בְּזָהָב קָנִיתָ / אֶת בִּתּוֹ שֶׁל גִּימִיר,



נָתַתָּ מַתָּנָה חַרְבֶּךָ,



אַךְ בִּרְכֹב בְּנֵי-מוּסְפֶּל / דֶּרֶךְ מִירְקְוִיד,



אַתָּה, הַמִּסְכֵּן, חָסַרְתָּ כָּל כְּלִי-זַיִן.


  

בִּיגְוִיר


אִלּוּ אָצִיל אֲנִי / כְּמוֹ אִינְגוּנַאר-פְרֵיר,



וּמוֹשָׁב נֶהְדָּר לִי,



הָיִיתִי מוֹעֵךְ עַד-דַּק / אֶת עוֹרֵב-הָרָעָה,



וְרוֹעֵץ אֶת כָּל גְּרָמָיו.


  

לוֹקִי


מִי הוּא הַקְּטַנְטַן / הַמְכַשְׁכֵּשׁ בִּזְנָבוֹ,



רַב לַחְטֹף וְלִבְלֹעַ?



הַמַּתְמִיד וְשׁוֹהֶה / בְּאָזְנָיו שֶׁל פְרֵיר,



כְּשֶׁאֵינוֹ רוֹטֵן עַל רֵחָיִם.


  

בִּיגְוִיר


בִּיגְוִיר זֶה שְׁמִי / בְּפִי הָאָזִים וְהָאָדָם,



זֶה שִׁבְחִי שֶׁאֲנִי זָרִיז,



עַל זֹאת גַּאֲוָתִי, / כִּי בָּנָיו שֶׁל הֲרוֹפְּט



מְסֻבִּים פֹּה כֻּלָּם לַמִּשְׁתֶּה.


  

לוֹקִי


שְׁתֹק, אַתָּה בִּיגְוִיר, / מֵעוֹלָם לֹא הִמְצֵאתָ



לַגְּבָרִים טַרְפָּם בְּמִדָּה!



נֶחְבֵּאתָ-שָׁכַבְתָּ / בַּקַּשׁ עַל הָרִצְפָּה,



כְּשֶׁיָּרְדוּ לוֹחֲמִים לַקְּרָב.


  

הֵימְדַל


לוֹקִי, הַמַּשְׁקֶה / לְטִפֵּשׁ עָשָׂךָ,



מַדּוּעַ אֵינְךָ שׁוֹמֵעַ לָעֵצָה הַטּוֹבָה?



מִדַּת הַפְרָזָה / זֶה דַּרְכָּה בַּגֶּבֶר,



מַתִּירָה לַלָּשׁוֹן הָרְצוּעָה.


 

לוֹקִי


שְׁתֹק, הֵימְדַל, / מֵאָז מֵעוֹלָם נִגְזְרָה



עָלֶיךָ הַגְּזֵרָה הָרָעָה;



כָּל הַיָּמִים עָלֶיךָ / לַעֲמֹד בִּקְשִׁי עֹרֶף



עַל מִשְׁמַר הַשָּׁמָיִם.


  

סְקַאדִי


שָׂמֵחַ אַתָּה, לוֹקִי, / אֶפֶס לֹא תּוֹסִיף וְתַרְבֶּה



לְכַשְׁכֵּשׁ בִּזְנָבְךָ,



כִּי הָאָזִים / יְרַתְּקוּךָ אֶל סֶלַע



בְּמֵעָיו שֶׁל בֵּן קַר כַּכְּפוֹר.


  

לוֹקִי


דְּעִי לָךְ, אִם אֶל סֶלַע / יְרַתְּקוּנִי הָאָזִים



בְּמֵעָיו שֶׁל בֵּן קַר כַּכְּפוֹר,



אֲנִי רִאשׁוֹן וַאֲנִי אַחֲרוֹן / הָיִיתִי בַּקְּרָב,



בּוֹ נָפַל מֵת טִיאַצִי.


 

סְקַאדִי


דַּע לְךָ, אִם הָיִיתָ / אַתָּה רִאשׁוֹן וְאַתָּה אַחֲרוֹן



בַּקְרָב בּוֹ מֵת טִיאַצִי,



מִבָּתַּי וּמֵחֻרְשׁוֹתַי / תְּבוֹאֲךָ כָּל-הַיָּמִים



הָעֵצָה הַקָּרָה.


  

לוֹקִי


נָעֲמוּ דְּבָרַיִךְ יוֹתֵר / בְּדַבְּרֵךְ עִם בֶּן-לוֹפֵי



כְּשֶׁהִזְמַנְתִּנִי אֶל יְצוּעֵךְ;



אִם כָּל חֲטָאֵינוּ / יֵצְאוּ בְּמִסְפָּר,



אֶזְכְּרֶנּוּ גַּם אֶת זֶה.


  

לקחה סיף מן התמָד בגביע של גביש ותגֵש לו ותאמר:


שָׁלוֹם לְךָ לוֹקִי / קַח הַגָּבִיעַ



מָלֵא הַתְּמָד הַמְשֻׁמָּר,



וְאֶשָּׁאֵר אֲנִי לְבַדִּי / מִכָּל בְּנֵי הָאָזִים



חַפָּה מִכָּל פֶּשַׁע.


 

קיבל התמָד ושתה ממנו.


אַךְ עָלַיִךְ חַסְתִּי / כִּי בַּיְשָׁנִית וְקַפְּדָנִית הָיִית תָּמִיד



מִתְרַחֶקֶת מִגְּבָרִים;



רַק אֶחָד יָדַעְתִּי, / שֶׁהֵפִיר בְּרִית נִשּׂוּאָיו



גַּם עִם אִשְׁתּוֹ שֶׁל טוֹר:



מַעֲשֵׂה לוֹקִי הָרַע הוּא.


  

בֵּילָא


רָעֹשׁ יִרְעֲשׁוּ הָרִים, / סְבוּרַתְנִי, הַמַּרְעִים הוּא



הַהוֹלֵך מִבֵּיתוֹ הֵנָּה,



הוּא יַשְׁתִּיק מְהֵרָה / אֶת הַנָּבָל הַמְתֹעָב,



הַבָּז לָאָזִים וְלִבְנֵי-הָאָרֶץ.




לוֹקִי


שִׁתְקִי, אַתְּ בֵּילָא, / אַתְּ אִשְׁתּוֹ שֶׁל בִּגְוִיר,



וּמְכֻסָּה כֻּלָּךְ עַוְלָה;



מֵעוֹלָם לֹא בָּאָה בֵּין הָאָזִים / מִפְלֶצֶת רָעָה מִמֵּךְ,



אָמָה מְטֻנֶּפֶת בַּזֶּבֶל!


  

נכנס טוֹר ואמר:


שְׁתֹק, הָאֻמְלָל, / פֶּן יִסְגֹּר אֶת פִּיךָ



מְיוֹלְנִיר, הַפַּטִּישׁ הֶחָזָק,



אַתִּיז מִצַּוָּארְךָ / אֶת סֶלַע-הַכְּתֵפַיִם



וְכִבִּיתִי אוֹר חַיֶּיךָ.


  

לוֹקִי


הִנֵּה נִכְנַס / בֶּן הָאֲדָמָה;



בְּעַזּוּת פָּנִים תִּנְשׁוֹם וְתִנְשׁוֹף, טוֹר.



יָבוֹא יוֹם וְלֹא תָּעֵז / לְהִתְגּוֹשֵׁשׁ עִם הַזְּאֵב,



שֶׁיִּבְלַע אֶת אֲבִי-הַקְּרָבוֹת.


  

טוֹר


שְׁתֹק, הָאֻמְלָל, / פֶּן יִסְגֹּר אֶת פִּיךָ



מְיוֹלְנִיר, הַפַּטִּישׁ הֶחָזָק,



אֶזְרָקְךָ מַעְלָה / מִזְרָחָה שָׁמֶשׁ,



מָקוֹם לֹא תִרְאֲךָ נָפֶשׁ.


  

לוֹקִי


אַל תְּסַפֵּר לְעוֹלָם / לַגְּבָרִים, כִּי הָיִיתָ



מִיָּמֶיךָ בְּמִזְרַח הַשֶׁמֶשׁ;



בְּאֶצְבַּע קְטַנָּה שֶׁל כְּסָיָה / שָׁם יָשַׁב הַגִּבּוֹר,



לֹא הֶאֱמִין בְּעַצְמוֹ, כִּי הוּא טוֹר.


  

טוֹר


שְׁתֹק, הָאֻמְלָל, / פֶּן יִסְגֹּר אֶת פִּיךָ



מְיוֹלְנִיר, הַפַּטִּישׁ הֶחָזָק,



יָדִי תְּמִיתֶךָ / בַּכְּלִי בּוֹ הוּמַת הֲרוּנְגְנִיר



וְתִרְעַץ אֶת עַצְמוֹתֶיךָ כֻּלָּן.




לוֹקִי


אֲנִי זֹאת תִּקְוָתִי / כִּי עוֹד אַרְבֶּה שָׁנִים,



וְאַף אִם תָּרִים הַפַּטִּישׁ בִּגְעָרָה;



אָמַרְתָּ: קָשׁוֹת רְצוּעוֹת / יַלְקוּטוֹ שֶׁל סְקְרִימִיר



וְלֹא מָצָאתָ שֶׁבֶר,



אָבַדְתָּ בְּזַלְעֲפוֹת רָעָב.


  

טוֹר


שְׁתֹק, הָאֻמְלָל, / פֶּן יִסְגֹּר אֶת פִּיךָ



מְיוֹלְנִיר, הַפַּטִּישׁ הֶחָזָק,



שְׁאוֹל תַּחְתִּית יְשַׁלְּחֶךָּ / בַּכְּלִי בּוֹ הוּמַת הֲרוּנְגְנִיר



עָמֹק לְעוֹלַם-הַמֵּתִים.


  

לוֹקִי


כְּאַוַּת נַפְשִׁי שַׁרְתִּי / לִפְנֵי הָאָזִים



וְלִפְנֵי בְּנֵי-הָאָזִים;



רַק מִפָּנֶיךָ / בִּלְבַד אֶרְתַּע אָחוֹר,



אֲנִי יוֹדֵעַ שֶׁאַתָּה מֵיטִיב לִלְחֹם.


  


הֵזַדְתָּ, אֵגִיר / שֵׁכָר הַשָּׁנָה,



אוּלָם שׁוּב לֹא תִכְרֶה כֵּרָה;



כָּל טוּבְךָ, / אֲשֶׁר פֹּה אִתְּךָ



יֹאכַל יְקוֹד הָאֵשׁ,



אֵשׁ תִּצְלֶה אֶת גַּבֶּךָ!


  

אחרי הדברים האלה נחבא לוֹקי ולבש צורת אלתית. באשדות פְרַאנַאנְגְר צדוהו הָאָזִים ואסרו אותו במֵעיו של בנו וָלי, אבל את נַרְפִי, בנו השני, הפכו לזאב. סְקַאדִי נטלה נחש ארסי וקבעה אותו על פניו של לוֹקי, והיה ארסו של זה מטפטף עליו; אבל סִיגִין אשתו של לוֹקי ישבה על ידו וספל בידה לקבל את הארס, וכשנמלא הספל היתה מוציאה אותו החוצה. בינתיים היה הארס פוגע בלוֹקי וזה היה מושך בכבליו וכל הארץ היתה מזדעדעת - מכאן רעש האדמה הפוקד אותנו.



שִׁיר על אַלְוִיס

אַלְוִיס


עַתָּה אֶצְלִי תְּכַס / הַכַּלָּה סַפְסָלַי,



תֵּכֶף תֵּלֵךְ אִתִּי בָּיְתָה;



טִלְטוּלֵי הַשָּׁוְא / אָרְכוּ דַּי וְיוֹתֵר,



אִישׁ לֹא יַפְרִיעַ מְנוּחָתִי בַּבָּיִת.


  

טוֹר


מִי אַתָּה, הַבָּחוּר? / מַה חִוְרִים כֹּה פָּנֶיךָ?



הֲשָׁכַבְתָּ הַלַּיְלָה עִם מֵתִים?



אִישׁ מִן הָעֲנָקִים / לֹא יַגִּיעַ עָדֶיךָ,



לֹא לְךָ יָאָה הַכַּלָּה.




אַלְוִיס


אַלְוִיס - זֶה שְׁמִי, / מְעוֹנִי עֻמְקֵי אֲדָמָה,



שָׁם תַּחַת אֲבָנִים בֵּיתִי;



אַדִּיר חֶפְצִי / לִמְצֹא אֶת נַהַג-הַמֶּרְכָּבָה,



אַל יָפֵר אִישׁ מוֹצָא פִּיו.


  

טוֹר


אֲנִי אֲפִירֶנּוּ, / כִּי לִי הָאָב



יָאֶה לְמָהֳרֶנָּה;



לֹא הָיִיתִי בַּבַּיִת / עֵת יָעֲדוּ אוֹתָה לָךְ,



אֵין בִּלְתִּי אֵל לְתִתָּהּ.


 

אַלְוִיס


מִי הוּא הָעֲנָק / הַתּוֹבֵעַ לוֹ רְשׁוּת



לָתֵת הָעַלְמָה הַמַּבְרִיקָה?



נַוָּד שֶׁכְּמוֹתְךָ – / נָכְרִי לָאָדָם;



מִי נְשָׂאֲךָ בְּחֵיקוֹ, הוֹי גֹּלֶם?


  

טוֹר


וִינְגְטוֹר זֶה שְׁמִי / מִמֶּרְחַקִּים אֲנִי בָּא,



אֲנִי - בְּנוֹ שֶׁל סִידְגְרָנִי,



בִּרְצוֹתִי, לֹא תְּקַבֵּל / לְעוֹלָם הַנַּעֲרָה,



לְעוֹלָם לֹא אַסְכִּים לַבְּרִית.


  

אַלְוִיס


תֵּכֶף אֲקַבֵּל / הַסְכָּמָתְךָ שֶׁלָּךְ,



נָחֹג בִּשְׂמָחוֹת הַחֲתֻנָּה;



אֻמְלָל הָיִיתִי / בְּלִי הָעַלְמָה,



אֲשֶׁר לָבְנָהּ מַחְפִּיר הַשָּׁלֶג.


  

טוֹר


לֹא אֶמְנַע מִמְּךָ, / עֶלֶם חָכָם,



יָדָהּ שֶׁל הַנָּאָה,



אִם תּוֹדִיעֵנִי / אֶת אֲשֶׁר אֶרְצֶה לָדַעַת



מִכָּל הָעוֹלָמוֹת.


  

אַלְוִיס


[יְנַסֵּנִי וִינְגְטוֹר / אִם בִּרְצוֹנוֹ לְנַסּוֹתֵנִי,



אִם יֵשׁ חָכְמָה בַּנַּנָּס;



אֲנִי שׁוֹטַטְתִּי-עָבַרְתִּי / בְּתִשְׁעַת הָעוֹלָמוֹת כֻּלָּם,



וְעֵינַי פְּקוּחוֹת לְכָל דָּבָר]




טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ:



מַה שֵּׁם הָאֲדָמָה / רַחֲבַת-הַיָּדַיִם,



לָהּ יְכֻנֶּה בְּכָל הָעוֹלָמוֹת?


  

אַלְוִיס


“אֲדָמָה” בְּפִי אָדָם, / “הָאָרֶץ” בְּפִי הָאָזִים,



הַוָּנִים קוֹרְאִים לָהּ “הַדֶּרֶךְ”;



“הַיְרֻקָּה-לְעוֹלָם” – עֲנָקִים, / הָאֶלְבִּים – “הַכִּכָּר הַפּוֹרַחַת”



הָאֵלִים הָעֶלְיוֹנִים – “הַשָׂדֶה הַלַּח”.


  

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ:



מַה שֵּׁם הַשָּׁמַיִם, / (שֶׁרוֹאִים בָּם הַגִּבּוֹרִים)



לָהֶם יְקֹרָא בְּכָל הָעוֹלָמוֹת?


  

אַלְוִיס


“שָׁמַיִם” בְּפִי אָדָם, / “כִּפַּת-מָרוֹם” בְּפִי הָאֵלִים,



בְּפִי הַוָּנִים – “אוֹרְגֵי הָרוּחַ”,



“עוֹלָם-עֶלְיוֹן” בְּפִי הָעֲנָקִים, / בְּפִי הָאֶלְבִּים – “הַגַּג הַיָּפֶה”,



בְּפִי הַנַּנָּסִים – “אוּלָם חוֹנֵן-צוּף”.


  

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ:



מַה שֵּׁם הַיָּרֵחַ / שֶׁרוֹאִים בְּנֵי-הָאָדָם,



מַה שְּׁמוֹ בְּכָל הָעוֹלָמוֹת?


  

אַלְוִיס


“לְבָנָה” בְּפִי אָדָם, / “הָאֵשׁ הָרַכָּה” בְּפִי הָאֵלִים,



בְּפִי הֶל “הָאוֹפַן הַמִּתְגַּלְגֵּל”.



בְּפִי הָעֲנָקִים – “הָרָץ”, / “אוֹר” בְּפִי הַנַּנָּסִים,



בְּפִי הָאֶלְבִּים – “מוֹנֵה-הַזְּמָן”.


  

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר-דָּבָר לְכָל הַיֵּשׁ:



מַה שְּׁמוֹת הַשֶּׁמֶשׁ / שֶׁרוֹאִים בְּנֵי-אָדָם,



מַה שְּׁמוֹ בְּכָל הָעוֹלָמוֹת?


  

אַלְוִיס


בְּפִי הָאָדָם – “שֶׁמֶשׁ”, / בְּפִי אֵלִים – “חַמָּה”,



בְּפִי אֶצְבְּעוֹנֵי דְּוָלִין – כַּעַס,



“מְאוֹר-נֶצַח” בְּפִי הָעֲנָקִים, / בְּפִי הָאֶלְבִּים “הַגַּלְגַּל הַיָּפֶה”



“אוֹר צַח” בְּפִי עַם הָאָזִים.


  

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ:



מַה שְּׁמוֹת הָעֲנָנָה / הֲרַת הַחֲזִיזִים,



מַה שְּׁמָהּ בְּכָל הָעוֹלָמוֹת?




אַלְוִיס


“עֲנָנָה” בְּפִי אָדָם, / בְּפִי הָאֵלִים “תִּקְוַת-סַגְרִיר”



בְּפִי הַוָּנִים “שַׁעֲשׁוּע-הָרוּחַ”,



“מֵבִיא-מַיִם” בְּפִי הָעֲנָקִים, / “אוֹן חֲזִיזִים” בְּפִי הָאֶלְבִּים,



בְּפִי הֶל “קוֹבַע הַלּוֹט”.


  

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ:



מַה שְּׁמוֹת הָרוּחַ / הַמַּרְחִיק לַחְדֹּר,



מַה שְּׁמוֹ בְּכָל הָעוֹלָמוֹת?


 

אַלְוִיס


בְּפִי אָדָם – “רוּחַ”, / “הַמְשֻׁלְהָב” בְּפִי הָאֵלִים,



“הַצּוֹהֵל” בְּפִי הַמּוֹשְׁלִים-בַּכֹּל,



“הַנּוֹהֵם” בְּפִי הָעֲנָקִים, / “הָרוֹעֵש” בְּפִי הָאֶלְבִּים,



“סַעַר” – בְּמִשְׁכְּנוֹתֶיהָ שֶׁל הֶל.


  

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ: –



מַה שְּׁמוֹת הָאֲוִיר / הַלֹּא-נָע וְלֹא-זָע,



מַה שְּׁמוֹ בְּכָל הָעוֹלָמוֹת?


  

אַלְוִיס


בְּפִי אָדָם: “שֶׁקֶט-הָאֲוִיר”, / בְּפִי הָאֵלִים – “גַלִּינָה”,



“מָנֹחַ-הָרוּחַ” בְּפִי הַוָּנִים,



בְּפִי הַטּוּרְסִים – “מוּעָקָה”, / “נֶפֶשׁ הַיּוֹם” – בְּפִי הָאֶלְבִּים,



“מִפְלַט-הָרוֹם” בְּפִי הַנַּנָּסִים.


 

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ: –



מַה שְּׁמוֹתָיו שֶׁל יָם / בּוֹ עוֹבְרִים הָאָדָם,



מַה שְּׁמוֹ בְּכָל הָעוֹלָמוֹת?


  

אַלְוִיס


בְּפִי בְּנֵי-אָדָם – “יָם”, / בְּפִי אֵלִים – “אֲגָם”,



בְּפִי שֶׁבֶט הַוָּנִים – “מִשְׁבָּר”,



“בֵּית-הַצְּלוֹפוֹת” – בְּפִי הָעֲנָקִים, / בְּפִי הָאֶלְבִּים – “חֹמֶר-שְׁתִיָּה”,



נַנָּסִים יְכַנּוּהוּ “עֹמֶק”.


  

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ: –



מַה שְּׁמוֹת הָאֵשׁ / הַלּוֹהֶטֶת לָאָדָם,



מַה שְּׁמָהּ בְּכָל הָעוֹלָמוֹת?


  

אַלְוִיס


בְּפִי אָדָם “אֵשׁ”, / “לֶהָבָה” בְּפִי בְּנֵי-הָאָזִים,



בְּפִי הַוָּנִים “יְקוֹד רוֹתֵחַ”,



בְּפִי הָעֲנָקִים – “הָאוֹכֶלֶת”, / בְּפִי הַנַּנָּסִים – “הַשּׂוֹרֶפֶת”



בְּבֵיתָהּ שֶׁל הֶל – “הַזְּרִיזָה”.


  

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ: –



מַה שְּׁמוֹת הַיַּעַר / הַצּוֹמֵחַ לִבְנֵי-אָדָם,



מַה שְּׁמוֹ בְּכָל הָעוֹלָמוֹת?


 

אַלְוִיס


“יַּעַר” בְּפִי אָדָם, / “רַעְמַת-הַשָּׂדֶה” בְּפִי הָאֵלִים,



“אַצַּת-הַלְדֶּה” בְּעוֹלַם הַמָּוֶת,



“דֶּלֶק” בְּפִי הָעֲנָקִים, / בְּפִי הָאֶלְבִּים – “רַב-הָעֲנָפִים”,



“סְבַךְ” – בְּעוֹלַם הַוָּנִים.




טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ: -



מַה שְּׁמוֹת הַלַּיְלָה / בִּתּוֹ שֶׁל נוֹר,



מַה שְּׁמוֹ בְּכָל-הָעוֹלָמוֹת?


 

אַלְוִיס


“לַיְלָה” בְּפִי אָדָם, / “עֲרָפֶל” בְּפִי הָאֵלִים



“הַמַּעֲטֶה” בְּפִי הַשּׁוֹלְטִים בָּעוֹלָם,



“חֹסֶר-אוֹר” בְּפִי הָעֲנָקִים, / “עֹנֶג-שֵׁנָה” בְּפִי הָאֶלְבִּים,



“יוֹצֵר-הַחֲלוֹמוֹת” בְּפִי עַם-הַנַּנָּסִים.


 

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ: -



מַה שְּׁמוֹת הַמִּזְרָע / שֶׁזּוֹרְעִים בְּנֵי-אָדָם,



מַה שְּׁמוֹ בְּכָל הָעוֹלָמוֹת?


  

אַלְוִיס


“דָּגָן” – בְּפִי הָאָדָם, / “נֶבֶג” בְּפִי הָאֵלִים,



“לִבְלוּב-צְמִיחָה” בְּפִי שֵׁבֶט-הַוָּנִים,



“טֶרֶף” בְּפִי הָעֲנָקִים, “מַשְׁקֶה” בְּפִי הָאֶלְבִּים,



בְּבֵיתָהּ שֶׁל הֶל – “הַסּוּף הָרוֹפֵף”.


 

טוֹר


עֲנֵנִי, אַלְוִיס, / כִּי דּוֹמַנִי, לֹא נֶעְלַם



מִמְּךָ פֵּשֶׁר דָּבָר לְכָל הַיֵּשׁ: –



מַה שְּׁמוֹת הַמַּשְׁקֶה / שֶׁשּׁוֹתִים בְּנֵי-אָדָם,



מַה שְּׁמוֹ בְּכָל הָעוֹלָמוֹת?


 

אַלְוִיס


“בִּירָה” בְּפִי אָדָם, / “סוּדָן” בְּפִי הָאָזִים,



“לַחוּת-רֶקַח” בְּפִי הַוָּנִים,



“הַמַּשְׁקֶה הַזַּךְ” בְּפִי הַטּוּרְסִים, / בְּעוֹלַם-הַמֵּתִים “תְּמָד”



וּבְפִי בְּנֵי-סוּטְוִינְג – “עָסִיס”.


 

טוֹר


מֵעוֹלָם לֹא מָצָאתִי כָּל כָּךְ / חָכְמַת-זְקֵנִים



בְּלֵב הָאֶחָד;



אֲבָל עָלָה בְּיָדִי / לְכַלּוֹתְךָ בְּעָרְמָה.



אוֹר-יוֹם מֵמִית אֶת הַנַּנָּס,



הִנֵה הַחַמָּה מְאִירָה בַּהֵיכָל!


  

מִשְׁלֵי הָר


הִגִּיעָה הַשָּׁעָה לְדַבֵּר / מִכִּסֵּא הַנּוֹאֲמִים!



עַל הַמַּעְיָן הַקַּר שֶׁל אוּרְדְ



מַחֲרִישׁ הִשְׁקַפְתִּי, / מִשְׁתָּאֶה הִתְבּוֹנַנְתִּי



וְהֶאֱזַנְתִּי לְאִמְרֵי הָר;



הוּא דִּבֵּר עַל הָרוּנוֹת, / בְּלִי שֶׁתֹּאבַד לוֹ עֵצָה;



זֹאת שָׁמַעְתִּי בְּבֵיתוֹ שֶׁל הָר:


 


"אִיעָצְךָ, לוֹדְפַפְנִיר, / לַעֲצָתִי תִּשְׁמַע!



תִּצְמַח לְךָ טוֹבָה, כִּי תִּשְׁמַע לָהּ,



תִּהְיֶה לְךָ לְתוֹעֶלֶת, כִּי תְּקַבְּלֶנָּה:



אַךְ בַּלַּיְלָה תָּקוּם, / כִּי תִּתְחַקֶּה עַל אוֹיֵב



אוֹ תָּתוּר לְךָ מָקוֹם בַּחוּץ.


  


"אִיעָצְךָ, לוֹדְפַפְנִיר, / לַעֲצָתִי תִּשְׁמַע!



תִּצְמַח לְךָ טוֹבָה, כִּי תִּשְׁמַע לָהּ,



תִּהְיֶה לְךָ תּוֹעֶלֶת כִּי תְּקַבְּלֶנָּה:



לְעוֹלָם אַל תִּשְׁכַּב / בְּחֵיק מְכַשֵּׁפָה,



לֹא תִּלְפָּתְךָ שׁוֹקָהּ.


 


"כִּי תִּקַּח אֶת לִבְּךָ, / וְתִשְׁכַּח יוֹם מוֹעֵד,



תְּאַחֵר מוֹעֶצֶת-עָם;



תִּרְחַק מֵאָדָם, / לֹא יִמְתַּק לְךָ מַאֲכָל,



בִּדְאָגָה תְּבַקֵּשׁ שֵׁנָה.


 


"אִיעָצְךָ, לוֹדְפַפְנִיר, / לַעֲצָתִי תִּשְׁמַע!



תִּצְמַח לְךָ טוֹבָה, כִּי תִּשְׁמַע לָהּ,

שָׁלוֹם עַל כָּל הַשּׁוֹמְעִים!


  

נבואתו שֶׁל גְּרִיפִּיר

היה אדם אחד ושמו גְּרִיפִּיר, והיה בנו שֶׁל גִּילִימִי ואחיה שֶׁל הֲיוֹרְדִיס. הוא היה מלך במדינה אחת ועלה על כל האדם בּחוכמתו וגם ידע את הבאות. פּעם אחת רכב סִיגוּרד, הוא לבדו, ובא עד היכלו שֶׁל גְּרִיפִּיר. היה קל מאד להכיר את סִיגוּרד. בפתח ההיכל פגש אדם אחד והחליט לדבר אתו. שמו של אדם זה היה גֵּיטִיר.

  

שאל אותו סיגוּרד:


מִי הַיּוֹשֵׁב פֹּה / בְּאוּלַמֵּי הַבִּירָה?



מַה שֵּׁם קוֹרְאִים הַגִּבּוֹרִים / לְמוֹשֵׁל הָעָם?


  

גֵּיטִיר


גְּרִיפִּיר זֶה שְׁמוֹ / שֶׁל הָאַלּוּף הַטּוֹב,



הַמּוֹשֵׁל בַּיַּבָּשָׁה / וּבְעַם-הָאָרֶץ.


  

סִיגוּרְד


הַבְבֵיתוֹ הוּא / הַמֶּלֶךְ הֶחָכָם?



הֲנָכוֹן הַגִּבּוֹר / לְדַבֵּר אִתִּי?



נָכְרִי זָקוּק / לַעֲצָתוֹ בִּידִידוּת,



בְּחֵפֶץ-לֵב הָיִיתִי / נִפְגָּשׁ עִם גְּרִיפִּיר.


  

גֵּיטִיר


הַמֶּלֶךְ הָעַלִּיז / יִשְׁאַל אֶת גֵּיטִיר



מַה שֵּׁם הַגִּבּוֹר / הָרוֹצֶה שֶׁיִּשְׁמְעוּהוּ.


 

סִיגוּרְד


שְׁמִי סִיגוּרְד / יוֹרְשׁוֹ שֶׁל סִיגְמוּנְד,



וַהֲיוֹרְדִיס הִיא / אֵם נוֹטֵר-הַצָּבָא.


  


וַיֵּלֶךְ גֵּיטִיר / לְהַגִּיד לִגְרִיפִּיר:



"פֹּה בַּחוּץ / עוֹמֵד אִישׁ נָכְרִי;



נָאֶה וְיָפֶה / מְאֹד הַגִּבּוֹר,



הַדּוֹרֵשׁ לְדַבֵּר / אַךְ עִם הַנָּסִיךְ."





יָצָא מִן הַהֵיכָל / שׂוֹרֵר-הָעָם,



הִסְבִּיר פָּנָיו / לָאוֹרֵחַ הַזָּר:



"שָׁלוֹם בּוֹאֲךָ, סִיגוּרְד, / מַדּוּעַ לֹא בָּאתָ קֹדֶם?



אַתָּה, גֵּיטִיר, קַח / מִמֶּנּוּ אֶת גְּרָנִי."


  


הִתְחִילוּ לְדַבֵּר / עַל כַּמָּה וְכַמָּה דְּבָרִים,



בְּהִפָּגֵשׁ הַגִּבּוֹרִים / חַכְמֵי-הָעֵצָה.


  

סִיגוּרד


הַגֵּד לִי, אֲחִי-אִמִּי, / אִם אַךְ יָדַעְתָּ,



אֵיךְ יַעַבְרוּ / יְמֵי חַיֵּי סִיגוּרְד?


  

גְּרִיפִּיר


תִּהְיֶה הַנֶּהְדָּר / וּנְשׂוּא-פָנִים



מִכָּל הַנְּסִיכִים / תַּחַת הַשָּׁמַיִם,



נָדִיב בַּזָּהָב, / אַךְ עָצֵל לָנוּס,



תַּעֲנוּג לָעֵינָיִם, / חָכָם בַּדִּבּוּרִים.


 

סִיגוּרְד


מֶלֶךְ נַעֲלֶה, / הוֹסֵף וְגַלֵּה



לָאִישׁ סִיגוּרְד / אֶת אֲשֶׁר אַתָּה צוֹפֶה:



אֵיזֶה אֹשֶׁר / רִאשׁוֹן יִפְגְּשֵׁנִי



כְּשֶׁאֶטּוֹשׁ הַיּוֹם / חֲצֵרְךָ, שַׁלִּיט?


  

גְּרִיפִּיר


קֹדֶם כֹּל תִּקּוֹם / אָבִיךָ וְסָבְךָ,



תִּקּוֹם בְּדָמִים / בַּעֲקֵב הַתּוּגָה.



בַּקְּרָב תְּמַגֵּר / אֶת בְּנֵי-הוּנְדִינְג



עַזֵּי-הַלֵּב / וּתְנַצֵּחַ.


 

סִיגוּרְד


מֶלֶךְ חָכָם, / הַגֵּד לִקְרוֹבְךָ



אֱמֶת לַאֲמִתָּהּ / בְּשִׂיחָה גְּלוּיַת-לֵב:



הֲיֵשׁ עֲלִילוֹת זֹהַר / בְּגוֹרָלוֹ שֶׁל סִיגוּרְד



הַמִּתְרוֹמְמוֹת / עַד הָרָקִיעַ?


  

גְּרִיפִּיר


אַתָּה לְבַדְּךָ תַּכֶּה / הַתּוֹלַעַת הַנּוֹצְצָה



הַשּׁוֹכֶבֶת תְּאֵבָה / עַל שְׂדוֹת גְּנִיטָה;



תַּכֶּה נֶפֶשׁ / שְׁנֵי הָאַחִים



רֶגִין וּפַפְנִיר. / אֱמֶת דִּבַּרְתִּי.


  

סִיגוּרְד


יֵשׁ דֵּי זָהָב / אִם כָּךְ נִלְחַמְתִּי,



וְאֵצֵא בְּשָׁלוֹם / כִּדְבָרֶיךָ;



מָה אַתָּה רוֹאֶה עוֹד / בְּחַיֵּי סִיגוּרְד?


  

גְּרִיפִּיר


אַתָּה תִּמְצָא / מְעָרַת פַפְנִיר



וְתִשָּׂא הָאוֹצָר / הַמַּבְרִיק שָׁם;



תַּעֲמִיס הַזָּהָב / עַל גַּבּוֹ שֶׁל גְּרָנִי



וְתִסַּע אֶל גְּיוּקִי / גִּבּוֹר אֵיתָן.


  

סִיגוּרְד


הוֹסֵף וְהַגֵּד / בְּדִבְרֵי חָכְמָה



לַמֶּלֶךְ, / אַלּוּף עַז-לֵב:



הָיִיתִי אוֹרְחוֹ / שֶׁל גְּיוּקִי בְּצֵאתִי –



וְאֵיךְ עֲתִידִים / לַעֲבוֹר חַיָּי?




גְּרִיפִּיר


עַל סֶלַע יְשֵׁנָה / בִּתּוֹ שֶׁל הַנָּסִיךְ,



מְאִירָה בְּשִׁרְיוֹן, / כִּי מֵת הֶלְגִּי,



בְּחֶרֶב חַדָּה, / בְּמַכֵּה-פַפְנִיר,



תָּקוּם וְתַחְתּוֹךְ / הַשִּׁרְיוֹן הַנּוֹצֵץ.


 

סִיגוּרְד


הַשִּׁרְיוֹן נִשְׁבַּר, / הַכַּלָּה תְּדוֹבֵב,



הָאִשָּׁה הַנְּעִימָה, / שֶׁתָּקִיץ מִשְּׁנָתָהּ;



מַה תַּגִּיד לְסִיגוּרְד / הַיָּפָה-בַּטַּבָּעוֹת



וְיִהְיֶה לַמּוֹשֵׁל / לְאָשְׁרוֹ בֶּעָתִיד?


  

גְּרִיפִּיר


אוֹתְךָ הַגִּבּוֹר / תְּלַמֵּד הָרוּנוֹת,



לָמוֹ נִכְסָפִים / כָּל הָאָדָם;



וְעוֹד שְׂפוֹת / עַמִּים זָרִים,



וְדַעַת רִפְאוּת – / אַשְׁרֶיךָ, הַמּוֹשֵׁל!


  

סִיגוּרְד


זֶה עָלָה בְּיָדִי, / קִבַּלְתִּי הַתּוֹרָה,



אֲנִי נֶחְלָץ / הָלְאָה לָלֶכֶת;



הַעֲמֵק לִצְפּוֹת / הוֹסֵף וְהַגֵּד;



מָה אַתָּה רוֹאֶה עוֹד / בְּחַיֵּי-סִיגוּרְד?


 

גְּרִיפִּיר


תָּבֹא, הַגִּבּוֹר, / לִמְעוֹנוֹ שֶׁל הֵימִיר,



אוֹרֵחַ עַלִּיז / תִּהְיֶה לָאַלּוּף;



יוֹתֵר לֹא אֵדַע / לְסַפֵּר, סִיגוּרְד.



וְאַל תּוֹסִיף לִשְׁאֹל / אֶת אֲחִי-אִמֶּךָּ.




סִיגוּרְד


תּוּגָה תּוֹקְפַתְנִי / בַּעֲקֵב דְּבָרֶיךָ,



כִּי אַתָּה, הָאַלּוּף, / מַעֲמִיק רְאוֹת:



הַרְבֵּה עָמָל תִּרְאֶה / בְּגוֹרָלוֹ שֶׁל סִיגוּרְד



וְאֵינְךָ מֵעֵז / לֵאמֹר לִי, הַמֶּלֶךְ.


  

גְּרִיפִּיר


הָיוּ נְעוּרֶיךָ / מְאִירִים בְּזֹהַר,



גְּלוּיִים לָעֵינַיִם / וְעֵינִי בְּהִירָה;



אֲבָל לֹא בְּצֶדֶק / קְרָאוּנִי רַב-עֵצָה,



גָּדוֹל כִּנְבוּאָה: / הַקֵּץ לְחָכְמָתִי.


  

סִיגוּרְד


אִישׁ לֹא יָדַעְתִּי / בִּמְלֹא כָּל-הָאָרֶץ,



הַמֵּיטִיב לִרְאוֹת / מִגְּרִיפִּיר עֲתִידוֹת;



אַל תַּעְלִים דָּבָר / מִמֶּנִּי, וְאִם רָעוֹת;



וְאִם גַּם אֶאְשַׁם / בַּעֲלִילוֹת רָעוֹת.


 

גְּרִיפִּיר


לֹא נִכְתְּמוּ / חַיֶּיךָ בַּעֲווֹנוֹת,



הַאֲמֵן תַּאֲמִינָה / בָּזֹאת, סִיגוּרְד;



תְּהִלָּתְךָ עוֹמֶדֶת / עַד בְּלִי-יָרֵחַ,



וּלְעוֹלָם לֹא יָסוּף / שְׁמֶךָ, הַגִּבּוֹר.


 

סִיגוּרְד


לֹא יָשְׁרָה זֹאת בְּעֵינָי; / אֲבָל עַל סִיגוּרְד



לִפָּרֵד מִן הַמֶּלֶךְ, / כִּי כֵן נִגְזָרָה.



הַנַּח לָעֲתִידוֹת, / יָבוֹא מַה שֶּׁמֻּכְרָח,



אַךְ תַּרְאֶה לִי דַּרְכִּי, / דּוֹדִי הֶחָכָם!


 

גְּרִיפִּיר


עַתָּה עָלַי לְהַגִּיד / הַכֹּל לְסִיגוּרְד,



הָאַלּוּף הַמְהֻלָּל / מְאַלְּצֵנִי לֵאמֹר,



הֲלֹא יָדַעְתָּ, / כִּי אֱמֶת דִּבַּרְתִּי:



אֲנִי יוֹדֵעַ / הַיּוֹם בּוֹ תָּמוּת.




סִיגוּרְד


אֵין בִּרְצוֹנִי לְהַכְעִיס / אֶת-הַמֶּלֶךְ הָאַדִּיר;



אֶלָּא לִשְׁמֹעַ / עֵצָה טוֹבָה מִפִּי גְּרִיפִּיר.



רָצִיתִי לָדַעַת – / וְאִם גַּם אֵין זֶה שָׂמֵחַ –



מַה נָּכוֹן וְקַיָּם / בְּגוֹרַל סִיגוּרְד.




גְּרִיפִּיר


עַלְמָה נָאוָה / חַיָּה בְּבֵית הֵימִיר,



בְּסוֹד גִּבּוֹרִים / בְּרִינְהִילְד יְכַנּוּהָ.



בִּתּוֹ שֶׁל בּוּדְלִי; / הַנָּסִיךְ הַנַּעֲלֶה



הֵימִיר אוֹמֵן / אֶת הָעַקְשָׁנִית.


  

סִיגוּרְד


מַה לִּי וְלָהּ / לְבַת הַמֶּלֶךְ



יְפַת-הַתֹּאַר / הַגְּדֵלָה עִם הֵימִיר?



דָּבָר אֶל נָכוֹן / הַגֵּד לִי, הַמֶּלֶךְ,



כִּי אַתָּה צוֹפֶה / כָּל-הַגּוֹרָלוֹת.


  

גְּרִיפִּיר


תֹּאבַד מִמְּךָ / שִׂמְחַת הַחַיִּים.



בְּשֶׁל הַבַּת הַיָּפָה / שֶׁהֵימִיר אוֹמֵן;



תִּדַּד שְׁנָתְךָ, / קְרָב לֹא תַּעֲרֹךְ,



תִּרְחַק מֵאָדָם, / תִּרְאֶה אַךְ הַנַּעֲרָה.


  

סִיגוּרְד


הַאֵין מָזוֹר / לְתוּגַת אֲהָבִים?



הַגֵּד לִי, גְּרִיפִּיר, / מָה אַתָּה צוֹפֶה!



הַאֶקְנֶה הַנַּעֲרָה / בְּאוֹצְרוֹת-כֵּרָה,



אֶת בִּתּוֹ הַיָּפָה / שֶׁל הַשַּׂר הַנַּעֲלֶה?


 

גְּרִיפִּיר


כֻּלְּכֶם תִּשָּׁבְעוּ / שְׁבוּעוֹת זֶה לָזֶה,



אַךְ לֹא תֵּיטִיבוּ / לַעֲמֹד בְּנִדְרֵיכֶם.



לַיְלָה אֶחָד תִּשְׁהֶה / בְּנַחֲלַת גְּיוּקִי



וְתִשְׁכַּח הַיָּפָה / שֶׁגִּדֵּל הֵימִיר.


  

סִיגוּרְד


אֵיךְ קָרְתָה זֹאת? / אוֹתָהּ לִי תַּגִּיד.



כְּלוּם הֲפַכְפַּךְ / לִבּוֹ שֶׁל הַוִּיקִינְג?



הַאָפֵר דְּבָרַי / אָמַרְתִּי לְאֵשֶׁת



אֲשֶׁר אָהַבְתִּי / בְּכָל לְבָבִי?


  

גְּרִיפִּיר


תִּהְיֶה קָרְבָּן / תַּרְמִית אֲחֵרִים.



מְזִמּוֹת גְּרִימְהִילְד / אַתָּה תְּשַׁלֵּם;



אֶת-בִּתָּהּ שֶׁלָּהּ / צְהֻבַּת-הַשֵּׂעָר



תִּתֵּן לְךָ / בְּעָרְמָה רָעָה.




סִיגוּרְד


אֲנִי אֶתְחַתֵּן / בְּשֵׁבֶט גּוּנַר



וְגוּדְרוּן הִיא לִי / תִּהְיֶה לְאִשָּׁה;



הֲלֹא אֻשַּׁרְתִּי / בְּאִשָּׁה טוֹבָה,



לוּלֵא זִכְרוֹן / עָוֶל קְדוּמִים.


  

גְּרִיפִּיר


גְּרִימְהִילְד תִּקַּח / אֶת לִבְּךָ כָּלִיל,



הִיא תְּשַׁכֶּלְךָ / לְבַקֵּש יַד בְּרִינְהִילְד



בִּשְׁבִיל גּוּנַר, / מֶלֶךְ הַגּוֹתִים;



לְאֵם הַמֶּלֶךְ / תִּדּוֹר לִנְסֹעַ.


  

סִיגוּרְד


רָעוֹת תָּבֹאנָה; / אֲנִי צוֹפֶה יָפֶה,



בִּינַת סִיגוּרְד / תֹּאבַד מֶנּוּ כָּלִיל,



בִּרְצוֹתוֹ לְהַשִּׂיא / לְגֶבֶר זָר



הַנַּעֲרָה הַיָּפָה, / שֶׁכָּכָה אָהָב.


 

גְּרִיפִּיר


שְׁבוּעוֹת תִּשָּׁבְעוּ / כֻּלְכֶם זֶה לָזֶה



הוֹגְנִי וְגוּנַר, / וְאַתָּה, הַגִּבּוֹר, שְׁלִישִׁי;



עֲלֵיכֶם בַּדֶּרֶךְ / לְהַחֲלִיף תָּאָרְכֶם



אַתָּה וְגוּנַר. / גְּרִיפִּיר לֹא יְשַׁקֵּר.


  

סִיגוּרְד


אֵיך זֶה בָּזֶה תָּלוּי? / מַדּוּעַ שְׁנֵינוּ



מַחֲלִיפִים בַּדֶּרֶךְ / מַרְאֶה וּתְנוּעוֹת?



וַדַּאי רְמִיָּה / גּוֹרֶרֶת רְמִיָּה,



מְחַבֶּלֶת מְאֹד – / הַגֵּד לִי, גְּרִיפִּיר!


  

גְּרִיפִּיר


מַרְאֵה גּוּנַר / וּתְנוּעוֹתָיו תִּקַּח,



אֲבָל בִּינָתְךָ / וְקוֹלְךָ יִשָּׁאֲרוּ לָךְ;



אָז תְּאָרֵשׂ / הַנַּעֲרָה הַנָּאוָה



שֶׁגִּדֵּל הֵימִיר – / אִישׁ לֹא יוּכַל לִמְנוֹעַ זֹאת!




סִיגוּרְד


רַע הוּא בְּעֵינַי, / כִּי אָדָם רַע



הֲלֹא יְכַנּוּנִי / בְּסוֹד הַגִּבּוֹרִים;



עַל אַפִּי אֲרַמֶּה / בִּתּוֹ שֶׁל הַשַּׁלִּיט,



שֶׁהִיא בְּעֵינַי / הָעֲדִינָה בַּנָּשִׁים.


  

גְּרִיפִּיר


אַלּוּף-צְבָאוֹת / תַּעֲלֶה יְצוּעַ



הַנַּעֲרָה, כְּאִלּוּ / הִיא אִמְּךָ-שֶׁלְּךָ;



תְּהִלָּתְךָ עוֹמֶדֶת / עַד בְּלִי יָרֵחַ,



לְעוֹלָם לֹא יָסוּף / שְׁמֶךָ, הַגִּבּוֹר!


  

סִיגוּרְד


הֲתִהְיֶה לְגוּנַר / אִשָּׁה טוֹבָה,



לוֹ לַמְכֻבָּד מִכֹּל – / עֲנֵנִי, גְּרִיפִּיר! –



אַף כִּי תִּישַׁן / שְׁלשָׁה לֵילוֹת



עִמִּי כַּלַּת-הַחֶרֶב? / מֵעוֹלָם לֹא הָיְתָה כָּזֹאת!


  

גְּרִיפִּיר


שְׁנֵיכֶם בְּבַת-אַחַת / תָּחֹגּוּ חֲתֻנַּתְכֶם,



גּוּנַר וְסִיגוּרְד. / בְּאוּלַמֵּי גְּיוּקִי;



שׁוּב תַּחֲלִיפוּ / בַּבַּיִת אֶת בַּרְכֶם;



אוּלָם תּוֹכְכֶם / נִשְׁאַר כְּשֶׁהָיָה.


 

סִיגוּרְד


הַלְטוֹב לָנוּ, / לִשְׁנֵי הַגִּבּוֹרִים,



יִהְיֶה אֲשֶׁר הִרְהַבְנוּ? / הַגֵּד לִי, גְּרִיפִּיר!



הַאִם לְהַבָּא / יִצְמַח מִזֶּה



טוֹב לְגוּנַר / וּלְסִיגוּרְד גַּם הוּא?


 

גְּרִיפִּיר


תִּזְכּוֹר הַשְּׁבוּעָה, / הֲלֹא תַּחֲרִישָׁה,



וְתִרְאֶה חַיִּים טוֹבִים / עִם גּוּדְרוּן;



אוּלָם בְּרִינְהִילְד תֹּאמַר / כִּי לֹא טוֹב הִתְחַתְּנָה,



תִּהְיֶה שׁוֹאֶפֶת / לִנְקוֹם אֶת הַכַּחַשׁ.


  

סִיגוּרְד


הַתְפַיֵּס תְּשׁוּבָה / אֶת לֵב הָאִשָּׁה



אֲשֶׁר רִמִּינוּ / בְּעָרְמַת-רְמִיָּה?



לָהּ נִשְׁבַּעְתִּי / וְלֹא קִיַּמְתִּי,



וְאֹשֶׁר מְעַט / יָדְעָה הָאִשָּׁה.


  

גְּרִיפִּיר


אָמֹר תֹּאמַר / לְגוּנַר אִשְׁתּוֹ,



כִּי לֹא שָׁמַרְתָּ / לוֹ הַבְטָחוֹתֶיךָ,



כִּי הֶאֱמִין הַשַּׂר / הַנַּעֲלֶה בֶּאֱמוּנָה שְׁלֵמָה,



בְּנוֹ שֶׁל גְּיוּקִי, / לָאִישׁ לְסִיגוּרְד.


  

סִיגוּרְד


אֵיךְ יִפֹּל הַדָּבָר? / אַל תַּעְלֵם מֶנִּי דָּבָר!



הַאֻמְנָם עָשִׂיתִי / אֶת-הַתּוֹעֵבָה?



אוֹ לַעַז תּוֹצִיא / עָלַי הַמַּלְכָּה



וְגַם עַל עַצְמָהּ? / הַגֵּד לִי, גְּרִיפִּיר!


  

גְּרִיפִּיר


מֵעֶלְבּוֹן גַּאֲוָתָהּ / מִתּוֹךְ כְּאֵב וָכַעַס



תָּרַע לְךָ / הָעֲדִינָה;



לֹא הִכְלַמְתָּ / אֶת הַיָּפָה,



וְאִם גַּם בְּיוֹדְעִים / רִמִּיתֶם אֶת אֵשֶׁת הַמֶּלֶךְ.


  

סִיגוּרְד


הֲיִשְׁמַע גּוּנַר הֶחָכָם / גַּם גּוּטְהוֹרְם וְהוֹגְנִי



לְקוֹל דְּבָרֶיהָ / הַתּוֹבְעִים נָקָם?



הֲיַאְדִּימוּ הַחֲנִיתוֹת / בָּנָיו שֶׁל גְּיוּקִי



בְּלֵב גִּיסָם? / אַל תַּעְלִים, גְּרִיפִּיר!


 

גְּרִיפִּיר


אֵבֶל-זַעַם / מָלֵא לֶב-גּוּדְרוּן



כְּשֶׁיְמִיתוּךָ / הָאַחִים הַבּוֹגְדִים;



לֹא יִצְמַח שׁוּב / מָשׂוֹשׂ בְּחַיֵּי



חַכְמַת-הַנָּשִׁים – / גְּרִימְהִילְד זֹאת עוֹלְלָה.


  

סִיגוּרְד


אַשְׁרֵי הַנִּפְרָדִים! / אִישׁ לֹא יַעֲמוֹד בִּפְנֵי הַגְּזֵרָה.



יָפֶה מִלֵּא גְּרִיפִּיר / אֶת מִשְׁאֲלוֹתָי;



בְּרָצוֹן הָיִיתָ / מַגִּיד אַךְ שְׂמָחוֹת



מִיָּמִים יָבֹאוּ / אִלּוּ יָכוֹל פִּיךָ.




גְּרִיפִּיר


נְסִיךְ-צְבָאוֹת גִּבּוֹר, / זְכוֹר וְהִנָּחֵם,



כִּי אַךְ מַזָּל מַרְהִיב / נָפַל בְּגוֹרָלֶךָ:



לֹא יִתְהַלֵּךְ לְעוֹלָם / עַל עָפָר בְּאוֹר שֶׁמֶשׁ



גִּבּוֹר עָדִין / מִמֶּךָּ, סִיגוּרְד.
