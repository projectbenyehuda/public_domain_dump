


	
	וַיְהִי הַיּוֹם יוֹם
	



מתוך מלחמות העט והמספרים [המשורר מגלה שהעט עלול להפיג ממנו את השעמום.]









א.


וַיְהִי הַיּוֹם יוֹם צֵאת שֶׁלֶג וְקָרָה, / יוֹם
הַקֶּרַח הַנּוֹרָא / עַל פְּנֵי הַשָּׂדֶה נָטָה שַׁפְרִירוֹ / וַיּוֹרֶד
רִירוֹ, / נָס מִפָּנָיו כָּל חֹם וְכָל שָׁרָב / וְנָפַל מִמֶּנּוּ רַב –
/ כְּאַמָּתַיִם עַל פְּנֵי כָל הָאָרֶץ, / וְאֵין פּוֹרֵץ בְּעֻזּוֹ
פָּרֶץ. / הָגָה בְּרוּחוֹ הַקָּשָׁה / וְהָפַךְ יָם לְיַבָּשָׁה, /
וַיַּחֲבֹל רֵחַיִם, / וְלֹא יָכְלוּ הָעָם לִשְׁתּוֹת מַיִם, / כִּי כָל
מַעְיָן וְנַחַל נִקְפָּא, / הָגְלְדָה טִפָּה. / וְהַשֶּׁמֶשׁ לֹא עָצַר
עִמּוֹ כֹּחַ / וַיִּתְגַנֵּב בְּהֵחָבֵא לִבְרֹחַ, / יָשֶׁת חֹשֶׁךְ
סִתְרוֹ / וּמִפְרְשֵׂי עָב יִלְבַּשׁ עַל בְּשָׂרוֹ, / יְסֻכּוּהוּ
צֶאֱלִים צִלְלוֹ / וַיכַסּוּהוּ בַּבְּגָדִים וְלֹא יֵחַם לוֹ – / כִּי
הַקֶּרַח הוֹלֵךְ וְחָזֵק מְאֹד, / לִפְנֵי קָרָתוֹ מִי יַעֲמֹד?



וְכָל הָעָם אֶל הַכֵּלִים נֶחְבָּא – / אֵין יוֹצֵא
וְאֵין בָּא, / וּמֵעַמִּים אֵין אִישׁ אִתִּי / וְגַם מֵהַסָּרִים
לְמִשְׁמַעְתִּי, / אֵין רֵעַ וְאֵין חָבֵר / לְסַפֵּר עִמִּי וּלְדַבֵּר,
/ כִּי הַקֹּר סָגַר עֲלֵיהֶם / וְכֻלָּם קוֹפְאִים עַל שִׁמְרֵיהֶם. /
וְאַבִּיט – וְאֵין אִישׁ, וְאֶשְׁתּוֹמֵם – וְאֵין מַפְגִּיעַ, /
אָמַרְתִּי: תִּסְעָדֵנִי יְמִינִי וּזְרוֹעִי לִי תוֹשִׁיעַ, / לְבִלְתִּי
זְבוּב פְּרִידַת דּוֹדִים וְרֵעַ / שֶׁמֶן שְׂשׂוֹנִי יַבְאִישׁ, /
בְּמָקוֹם שֶׁאֵין אֲנָשִׁים תִּשְׁתַּדֵּל לִהְיוֹת אִישׁ, / וּמִי
שֶׁיֶּשׁ לוֹ עֵט וּדְיוֹ – / מִכָּל הַחֲבֵרִים דַּיּוֹ! / עַל בֹּהֶן
יָדוֹ הַיְמָנִית – / לַהַב חֲנִית, / לָשׁוֹן שֵׁנִית, / אֶצְבַּע
יְתֵרָה, / לוֹ מִשְׁפַּט הַבְּכוֹרָה, / לוֹ עֵזֶר כְּנֶגְדּוֹ / וּמַטֵּה
הָאֱלֹהִים בְּיָדוֹ. / הוּא אַחַד הַשִּׂיחִים / שְׂפָתַיִם יִשַּׁק
מֵשִׁיב דְּבָרִים נְכוֹחִים, / וּשְׁמוֹ לָמָּה יִגָּרַע / מֵעֵץ הַדַּעַת
טוֹב וָרַע? / כֹּל אֲשֶׁר בּוֹ נוֹגֵעַ / טוֹב וְרַע כֵּאלֹקִים יוֹדֵעַ.
/ וּבְמַהֲלָל הֲרִימוֹתִי קוֹלִי / וָאֶשָּׂא מְשָׁלִי:





סוֹפֵר, בִּימִינָךְ לַהַב אֵשׁ / אוֹ לַהַב חֶרֶב וַחְנִיתוֹת?

עֵץ הַדַּעַת טוֹב וָרַע אוֹ / מַטֶּה תַּעְשֶׂה בּוֹ אֶת הָאוֹתוֹת?




ב.

נסיון הכתיבה מסתיים בכשלון






אָז בִּמְסִבִּי יָשַׁבְתִּי / וְקַסְתִּי בְּמָתְנַי
שַׂמְתִּי / לִכְתֹּב כְּחֻקִּי וּכְמִשְׁפָּטִי, / יָדִי שָׁלַחְתִּי
לְעֵטִי / לִטְבֹּל אוֹתוֹ בִדְיוֹ – / וַיֹּאמֶר אֵלַי ״וְאַיּוֹ?״ / כִּי
בְרוּחַ הַקֶּרַח נֶעֶרְמוּ מֵימָיו, / קָפְאוּ בְלֶב קֶסֶת תְּהוֹמָיו. /
וְדָפַקְתִּי שֵׁנִית לְטָבְלוֹ בְחָזְקָה – / וְשִׁנָּיו נִתְעַוּוּ
וַיִּסְעוּ מִדָּפְקָה / וַיִּכְרַע עַל בִּרְכּוֹ – / וְהוּא צוֹלֵעַ עַל
יְרֵכוֹ! / וַאֲנִי קָצַפְתִּי עַל קַסְתִּי, / נָטַשְׁתִּי אֶת מְגִלָּתִי
/ וְכִמְעַט בַּחֲרוֹת אַפִּי קְרַעְתִּיהָ / וְאֶת הָאָח לְפָנַי
מְבֹעֶרֶת הִשְׁלַכְתִּיהָ. / וְהָעֵט יָרַד אֶל יַרְכְּתֵי קַסְתּוֹ /
וַיִּשְׁכַב עַל מִטָּתוֹ.

וּבַחֲמָתִי וְאַפִּי / לִבְלִי חֹק פִּי / עָלָיו
פָּעַרְתִּי, / וּמֵרֹב שִׂיחִי וְכַעְסִי דִּבַּרְתִּי / ״מְשַׁלֵּם רָעָה
תַּחַת טוֹבָה, צֵא מִן הַתֵּבָה! / לָמָּה נֶחְבֵּאתָ לִבְרֹחַ, / מֶה
עָזַרְתָּ לְלֹא כֹחַ? / הֲזֶה שְׂכָרִי עַל כִּי כִבַּדְתִּיךָ / וּמַטֵּה
הָאֱלֹהִים קְרָאתִיךָ / וְשֶׁבַע בַּיּוֹם הִלַּלְתִּיךָ / וּנְתַתִּיךָ
עֶלְיוֹן וְרִאשׁוֹן / עַל כָּל פֶּה וְלָשׁוֹן? / וְאַיֵּה הַגֶּרֶם
שֶׁשִׁבַּרְתָּ / וְאֵיפֹה הַצּוּר שֶׁבָּקַעְתָּ? – / וְלָשׁוֹן רַכָּה
תִּשְׁבָּר גֶּרֶם / וַאֲשֶׁר נָטָה יָדוֹ וַיָּרֶם / לְהַכּוֹת
בְּמַטֵּהוּ פַּעֲמַיִם – / בָּקַע צוּר וַיָּזוּבוּ מַיִם! / וְאַתָּה
שַׁבְתָּ מִפְּנֵי נְעֹרֶת שְׁרוּיָה, / וּמַה תַּעֲשֶׂה אֶל זְרוֹעַ
נְטוּיָה / וּבְצֶמֶר גֶּפֶן נָשְׁכָה שִׁנֶּךָ / וְנָס לֵחֶךָ, /
הִתְרַפִּיתָ בְּיוֹם צָרָה סָר כֹּחֶכָה! / וּתְלֻנּוֹתַי עֲלֵי נַפְשִׁי
/ וַחֲמָסִי עֲלֵי רֹאשִׁי, / כִּי בְחַרְתִּיךָ מִכָּל אֲהוּבִים וְרֵעִים
/ לִהְיוֹת לִי יוֹם יוֹם שַׁעֲשׁוּעִים, / וּמִשִּׂבְרִי הֱבִישׁוֹתַנִי /
עַתָּה רֵיקָם שִׁלַּחְתַּנִי. / וְכָכָה יֵעָשֶׂה לְמַנִּיחַ בְּרוֹשׁ
וְחוֹסֶה תַּחַת הַנַּעֲצוּץ, / וָאֶשָּׁעֵן עַל מִשְׁעֶנֶת קָנֶה
הָרָצוּץ!





בָּחַרְתִּי עֵט הִלַּלְתִּיהוּ / וּזְבַדְתִּיו זֶבֶד טוֹב זָבוֹד,

חָשַׁבְתִּי כִּי עַל כֵּן בִּשְׂכַר / זֶה מֵאַהְבָה אוֹתִי יַעְבֹד –

אַךְ לַשֶּׁקֶר שָׁמַרְתִּי, כִּי / טָרְחִי עִמּוֹ אָבַד אָבוֹד.

כִּצְרוֹר אֶבֶן בַּמַּרְגֵּמָה / כֵּן הַנּוֹתֵן לִכְסִיל כָּבוֹד!״





וְהָעֵט קָם בַּחֲמָתוֹ / מִיַּרְכְּתֵי קַסְתּוֹ /
וַיֹּאמַר ״שְׁמָעֵנִי, הוֹדִיעֵנִי, / עַל מַה תְּרִיבֵנִי / עַד
אָשִׂיחָה בְּמַר רוּחִי: / אִם כֹּחַ אֲבָנִים כֹּחִי? / אִם בְּשָׂרִי
נָחוּשׁ? / וּמִי יֹאכַל וּמִי יָחוּשׁ? / קָנֶה רָצוּץ רְפֵה יָדַיִם – /
הֲמִן הַסֶּלַע הַזֶּה נוֹצִיא לְךָ מַיִם? / אִם אֶפְצַר עוֹד לִהְיוֹת
לִי לְמוֹקֵשׁ – / וּמִקְרָא מָלֵא הוּא ״צִנִּים פַּחִים בְּדֶרֶךְ
עִקֵּשׁ״. / הַאִם כַּף רֶגֶל עֵגֶל חֲשַׁבְתַּנִי / בְּיַעַן לִשְׁתֵּי
פְרָסוֹת שִׁסַּעְתַּנִי / לְשַׂדֵּד פְּנֵי עֲמָקִים / כִּרְאִי מוּצָק
חֲזָקִים? / אוֹ קְרָאתַנִי מַטֵּה הָאֱלֹהִים / וְכִי דְרָכַי מִדַּרְכֵי
כָל לָשׁוֹן גְּבֹהִים – / לְהַכְבִּיד עָלַי מַשָּׂא וְהַוָּה עַל הַוָּה,
/ מִקַּל וָחֹמֶר וּמִגְּזֵרָה שָׁוָה? / מִקַּל וָחֹמֶר – לָשׁוֹן גֶּרֶם
לְרַצֵּץ, / וּמִגְּזֵרָה שָׁוָה – שֶׁל מַטֶּה סֶלַע לְפוֹצֵץ. / וְזֶה
לְאוֹת כִּי אַתָּה מְבַקֵּשׁ עָלַי תּוֹאֲנָה / אוֹ בְּפִי הַפָּסוּק אֵין
לְךָ הֲבָנָה, / וְכָל שַׁעַר עַמִּי יוֹדֵעַ, / כִּי הַנָּבִיא שֶׁהַצּוּר
בּוֹקֵעַ – / לֹא בְחֵיל הַמַּטֶּה וְלֹא בְכֹחוֹ, / כִּי אִם בְּרוּחוֹ. /
וְכֵן הַלָּשׁוֹן לֹא תִשְׁבֹּר הַגֶּרֶם בְּעָצְמָהּ, / כִּי אִם
בְּחֶלְקַת נְאוּמָהּ, / שֶׁנֶּאֱמַר ״מַעֲנֶה רַךְ מֵשִׁיב חֵמָה״. / אַף
כִּי אוֹי לִי / אֲשֶׁר לֹא נִסְּתָה כַף רַגְלִי / הַצֵּג עַל הָאָרֶץ
וּדְרֹךְ / מֵהִתְעַנֵּג וּמֵרֹךְ – / וְתוֹחִיל: כַּמָּטָר לִקְחִי
יַעֲרֹף / בְּהַכּוֹת בְּאֶבֶן אוֹ בְּאֶגְרוֹף! / וְעוֹד כִּי חֻלְשָׁתִי
אַתָּה סַבּוֹתָ – / וּמַדּוּעַ אֵלַי רִיבוֹתָ? / כִּי עַד דַּק
פְּסַלְתַּנִי / וְכַעֲצֵי עוֹלָה בִּקַּעְתַּנִי / וּכְשַׁסַּע הַגְּדִי
שִׁסַּעְתַנִי. / זְכָר נָא כִּי כַחֹמֶר עֲשִׂיתַנִי / לְצֹרֶךְ
הַכְּתִיבָה / עַד כִּי תָשַׁשׁ כֹּחִי כִּנְקֵבָה. / וְאַף גַּם זֹאת:
בְּכַחֲשִׁי וְרִפְיוֹן מֹחִי, / אַתָּה יָדַעְתָּ כִּי בְכָל כֹּחִי /
עֲבַדְתִּיךָ בְּתֹם לְבָבִי וְנִקְיוֹן כַּפִּי, / בְּזֵעַת אַפִּי, /
כְּעֶבֶד נִרְצָע וְחָבוּשׁ / עָרֹם הִלַּכְתִּי מִבְּלִי לְבוּשׁ. /
רַגְלִי מִיָּחֵף לֹא מָנַעְתִּי / וּבִמְצוּלַת הַדְּיוֹ טָבַעְתִּי, /
כִּי שִׁלַּחְתַּנִי כָּל הַיּוֹם לִבְדֹּק הַחֹר / לִשְׁתּוֹת מֵי
שִׁיחוֹר, / וָאָבוֹאָה בַמַּיִם / מֵי בִּרְכַּיִם. / וּבְכָל שׁוּטִי
וְלֶכְתִּי / הַגִּידָה לִי מַה מַּשְׂכֻּרְתִּי? / וּמֵאֲשֶׁר חַקּוֹתִי
בַסֵּפֶר / מִיַּד מִי לָקַחְתִּי כֹפֶר? / וְכָתַבְתִּי לְךָ שָׁלִישִׁים
/ לַבְּקָרִים חֲדָשִׁים, / מִכֹּל אֲשֶׁר לִבְּךָ עָבַר / לֹא נָפַל
דָּבָר. / אִם יִמָּלְאוּ עֲטִינַי גֶּשֶׁם דְּיוֹ – עַל הַמְּגִלָּה
אֲרִיקֶנּוּ, / חַיִל בָּלַעְתִּי וָאֲקִיאֶנּוּ, / וְלֹא דָבַק בְּיָדִי
מְאוּמָה – / עָרֹם יָצָאתִי מִבֶּטֶן קַסְתִּי וְעָרֹם אָשׁוּב שָׁמָּה /
וְאִם עַתָּה עֲצָרַנִי הַקֶּרַח מִלֶּדֶת, / הַקֻּשְׁיָא בִּמְקוֹמָהּ
עוֹמֶדֶת. / וְעַתָּה אִם לִכְתֹּב נִכְסַפְתָּ, / לָמָּה חֵרַפְתָּ /
וְאוֹתִי גִּדַּפְתָּ? / הֲלֹא טוֹב לְחַלּוֹת פְּנֵי אֵשׁ וּשְׁבִיבוֹ /
לְמוֹגֵג הַדְּיוֹ וּלְהַדְאִיבוֹ / וּלְהָסִיר לֵב הָאֶבֶן מִקִּרְבּוֹ. /
אוֹ כַתַּר נָא זְעֵיר וְחַם הַשֶּׁמֶשׁ וְנָמַס / וַאֲשִׁיתֵהוּ לְרַגְלִי
מִרְמָס / וְהַקֶּרַח מְפֻזָּר וּמְפֹרָד / וְיָעָה בָרָד, / וּמִן
הַמַּיִם הַנִּגָּרִים בְּמוֹרָד / עֲיָנוֹת וּתְהוֹמוֹת יוֹצְאוֹת
בַּבִּקְעָה, / וּשְׁאַבְתֶּם מַיִם בְּשָׂשׂוֹן מִמַּעְיְנֵי הַיְשׁוּעָה
/ לְהַשְׁקוֹת הַמְּגִלָּה וּלְכָתְבָהּ / לְאָרְכָהּ וּלְרָחְבָּהּ, /
וְלַעֲשׂוֹת לְךָ אוֹת בְּאוֹת תֵּבָה בְתֵבָה.





הֲיֵשׁ קָשׁוּר בְּלִי חֶבֶל / וְאִם אַסִּיר בְּלִי כֶבֶל?

וְכָל רֶכֶב בְּלִי מִלְוָה / וְרֵחַיִם הֲיֵשׁ חוֹבֵל?

וְרַק הָאֵשׁ וְחֹם שֶׁמֶשׁ / יְקַו לִקְרֹא שְׁנַת יוֹבֵל

וְאָז לַעְבֹר נְהָרוֹת – שׁוֹק / יְגַלֶּה, יֶחֱשֹׂף שֹׁבֶל.





זֹאת לֹא זֹאת: / אִם לֹא תַאֲמִין לְקוֹל הָאוֹת, /
הִנֵּה קָטָנְךָ עָבָה מִמָּתְנַי / וּמֵעֶיךָ חֲזָקִים מִשִּׁנַּי /
וְצֶמֶר הַדְּיוֹ רַךְ לְפִי דְבָרֶיךָ – / לְמַעַן תִּצְדַּק בְּדָבְרֶךָ
/ טְבֹל אַתָּה בְּאֶצְבְּעוֹתֶיךָ / וְתִרְאֶה מַה יִּהְיוּ
חֲלוֹמוֹתֶיךָ!״





הַדָּבָר מִפִּיו יָצָא / וּמִהַרְתִּי בִמְרוּצָה /
לִטְבֹּל אֶצְבַּע בַּדְּיוֹ / וְנִקְרַע עוֹר וּבָשָׂר בְּקָשְׁיוֹ /
וְצִפֹּרֶן נֶעְתַּק מִמְּקוֹמוֹ / וְנִמְצָה דָמוֹ. / וּבְבָשְׁתִּי
רָצִיתִי לְכַסּוֹת חָלְיִי / וְהָעֵט לוֹעֵג לִי / וְנִמְלָא שְׂחוֹק
פִּיהוּ / וַיֹּאמַר ״זֶה הַיּוֹם שֶׁקִּוִּיתִיהוּ!״ / וַיִּקְרָא אֵלַי
בְּלַעַג נְאוּמוֹ / ״אַל תָּדִין אֶת חֲבֵרְךָ עַד שֶׁתַּגִּיעַ
לִמְקוֹמוֹ!״ / וַאֲנִי בֹשְׁתִּי וְנִכְלַמְתִּי / וּמִשְׁתּוֹמֵם
עָמַדְתִּי / מֵעַזּוּת הַקֶּרַח וּמַעֲשֵׂה תָקְפּוֹ, / כִּי גִיד
בַּרְזֶל עָרְפּוֹ / וּמִצְחוֹ נְחוּשָׁה / וּמֵחֲמַס הָעֵט תְּכַסֵּנִי
בּוּשָׁה. / וָאֹמַר: חָטָאתִי הַפַּעַם / וְדִבְרֵי הָעֵט דְּבָרִים שֶׁל
טַעַם / וְתוֹדָתִי כְּמִנְחָה יָרַח / וַאֲהָהּ, הַקֶּרַח! / מִי יַאֲמִין
לִשְׁמוּעָתֵנוּ / שֶׁתְּמַשֵּׁשׁ כָּל כְּלֵי בֵיתֵנוּ / וְעַל קַן
הַדְּיוֹ תִשְׁטַח חֲרָמֶיךָ? / עַד כָּאן, כְּפוֹר, / יַגִּיעוּ
רְשָׁמֶיךָ / לָשׂוּם הַנְּעֹרֶת מִצּוּר חֲזָקָה / שְׁלֵמָה
מִמַּשְׂרֵקָה.





אוֹמֵר אֶמֶשׁ ״מָחָר אֶכְתֹּב״ / כֵּן כִּי אֵין לִדְבָרָיו פִּדְיוֹם –

וּכְפוֹר שָׁמַיִם יִשְׂחַק לוֹ, / יִלְעַג קֶרַח נוֹרָא אָיֹם:

אַל תִּתְהַלֵּל בְּיוֹם מָחָר / כִּי לֹא תֵדַע מַה יֵּלֶד יוֹם!




ג

המספריים מציעים את עזרתם






וְהֵבֵאתִי יָדִי לְחֵיקִי וְנִתְעַלַּפְתִּי /
וּבְחַלּוֹן בֵּיתִי בְּעַד אֶשְׁנַבִּי נִשְׁקַפְתִּי / לִרְאוֹת דְּבַר
מִי יָקוּם – מִן הַקֶּרַח אוֹ מִן הַשֶּׁמֶשׁ / וְאִם יַכֵּהוּ חֹם
הַיּוֹם אֶל הַחֹמֶשׁ. / וְעָבַר חֲצִי הַיּוֹם וּמַרְבִּיתוֹ / וְהַקֶּרַח
עוֹמֵד בְּחָזְקָתוֹ / עוֹדֶנּוּ מַחֲזִיק בְּתֻמָּתוֹ, / וְאוֹר
הַשֶּׁמֶשׁ – יְקָרוֹת וְקִפָּאוֹן, / מוֹצָאֵי מַיִם לְצִמָּאוֹן, /
וְעָצַר כָּל לֵחַ וְנִקְשַׁר כָּל מִיץ / וַיְהִי הַקֶּשֶׁר אַמִּיץ. /
וַאֲנִי יוֹשֵׁב נִכְלָם / עָצוּב וְנֶאֱלָם / עַל הֱיוֹתִי לְבַדִּי /
בְּלִי עֵזֶר חָבֵר לְנֶגְדִּי / וְעַל עֹצֶר יָדַי וְכֵלַי, / וָאֶשְׁמַע
אֵת מִדַּבֵּר אֵלַי / כְּתָמֵהַּ עַל מַצָּבִי / ״הַבֲרָכָה אַחַת הִיא
לְךָ, אָבִי? / לָמָּה יָדֶיךָ אַתָּה יוֹעֵט? / הֲמִבְּלִי הָעֵט /
רִקְמַת הַכְּתָב אֵין לִרְקֹם? / וְהַרְבֵּה פְתָחִים לַמָּקוֹם! / וְאִם
יְאוֹרֵי הַדְּיוֹ חָרְבוּ וְדָלָלוּ / וְקָנֶה וָסוּף קָמֵלוּ – / הֲלָזֶה
חָדְלוּ פְרָזוֹת בְּיִשְׂרָאֵל חָדֵלוּ? / וְאִם נִצְּבוּ כְּמוֹ נֵד
נוֹזְלִים וְנִקְרָשׁוּ / קֹר יַעְצֹר בַּמַּיִם וְיִיבָשׁוּ – / הֲלֹא
תִכְרֹת עַד תַּמּוּז עֵצִים יְבֵשִׁים / וְעָשִׂיתָ אֶת הַקְּרָשִׁים. /
וְאִם דָּבַק לְשׁוֹן קָנֶה לְחִכּוֹ וּדְיוֹ אַיִן – הֲלָזֶה תִשָּׁכַח
תּוֹרָה / וְאֵין נְבוּאָה שׁוֹרָה? / אַיֵּה אֵפוֹא חֲרָמֶיךָ, / אֵיפֹה
טוּב טַעְמֶךָ? / אִם נִשְׁבַּר הָעֵץ וְרֻצַּץ הַקָּנֶה – / לְבֵנִים
נָפְלוּ וְגָזִית נִבְנֶה, / וְאִם צִנָּה גַּם קֹר מֵימֵי הַדְּיוֹ גוֹזֵל
/ וְאֵין הָעֵט נוֹזֵל – / קַח לְךָ עַט בַּרְזֶל / יָעֹז לְךָ מֵעֲשָׂרָה
עֵטִים / וַעֲשֵׂה שִׁטִּים / וְתֵבוֹת כְּפִתּוּחֵי חוֹתָם מְפֻתָּחוֹת, /
מִכְתַּב אֱלֹהִים חָרוּת עַל הַלּוּחוֹת / עוֹמֵד לְנֵס עַמִּים, /
וְהָיוּ תוֹאֲמִים / בְּיָדְךָ רָצוֹא וָשׁוֹב כְּמַרְאֵה הַבָּזָק, /
אִישׁ אֶת רֵעֵהוּ יַעֲזֹרוּ וּלְאָחִיו יֹאמַר ״חֲזַק״. / רוֹכְבִים
צְמוּדִים / וְהָיוּ לַאֲחָדִים / בְּיָדְךָ שְׁנַיִם – / הֵם הֵם
הַמִּסְפָּרַיִם. / בַּרְזֶל בְּבַרְזֶל יַחַד, / טוֹבִים הַשְּׁנַיִם מִן
הָאֶחָד / תַּחַת הָעֵצִים בַּרְזֶל יְמַלֵּא מִשְׁאָלָךְ / וְתִגְזַר
אֹמֶר וְיָקָם לָךְ!״ / וַיְהִי כַּאֲשֶׁר כִּלָּה לְדַבֵּר אֵלַי /
וַיַּעַל מֵעָלַי / אֲנִי בְרֹב שִׂמְחָתִי / דָּלַגְתִּי בְיַרְכְּתֵי
בֵיתִי / וְאָמַרְתִּי: זֶה הַיּוֹם יוֹם בְּשׂוֹרָה / כִּי יֵשׁ לָעֵט
תְּמוּרָה. / וְהַמִּסְפָּרַיִם הֱשִׁיבוּנִי / הֵם הִרְהִיבוּנִי / ״יֵשׁ
וָיֵשׁ, תְּנָה יָדֶךָ, / נַס נָא אֶת עֲבָדֶיךָ / וּכְתֹב וְגַם תּוּכַל
כֹּל / וְלֹא נָשׁוּב מִפְּנֵי כֹל, / וְלֹא יָבוֹא לְעֶזְרָתֵנוּ /
בִּלְתִּי אִם גְּוִיָּתֵנוּ״.




ד

סיום המשפט וההכרעה






[ויאמר העט]: ״וְאִם לֹא תַאֲמִינוּ כֵּן כִּי לֹא
תֵאָמֵנוּ, / נְחַלֶּה פְנֵי אֲדוֹנֵינוּ / לְבַל יָשִׁית יָדוֹ עַל
שְׁנֵינוּ / וּבִמְקוֹמֵנוּ יַנִּיחֵנוּ, / וּנְבַקֵּשׁ בַּשּׁוּק / אָדָם
עָשׁוּק. / אָז בִּרְצוֹנוֹ אֶל הַבַּיִת נְבִיאֵהוּ / וְדָבָר מִכָּל זֶה
לֹא נוֹדִיעֵהוּ, / וְגַם כָּל אֶחָד מִמֶּנּוּ יִשָּׁבַע / אֲשֶׁר לֹא
יוֹרֶה בְאֶצְבַּע / וְלֹא יִקְרֹץ בְּעֵינַיִם / וְלֹא יִמְלֹל
בְּרַגְלַיִם, / וּנְבַלְבֵּל כָּל כְּלֵי הָאֹהֱלָה, / עֶלְיוֹנִים
לְמַטָּה וְתַחְתּוֹנִים לְמַעְלָה, / וַאֲנַחְנוּ בִכְלָלָם, / וְנֹאמַר
לָאִישׁ הַנֶּעְלָם: / מַשֵּׁשׁ כְּלֵי הַבַּיִת כֻּלָּם, / תֶּן לָהֶם
כְּפָעֳלָם! / וַאֲשֶׁר אֶת פִּיהוּ יַמְרֶה – / אוֹ סָקוֹל יִסָּקֵל אוֹ
יָרֹה יִיָּרֶה, / וְנִרְאֶה אִם יִכְתֹּב עִמִּי אוֹ עִמָּכֶם /
וְיִבָּחֲנוּ דִבְרֵיכֶם״. / וַיֹּאמְרוּ הַמִּסְפָּרַיִם ״כֵּן
דִּבַּרְתָּ / וֶאֱמֶת אָמַרְתָּ, / בּוֹא וַעֲשֵׂה כֵן!״ / וַיֵּצֵא הָעֵט
אֶל רְחוֹב הָעִיר וַיִּמְצָא בוֹ אִישׁ מִסְכֵּן, / חָכָם בְּכָל חָכְמָה
/ וְיוֹדֵעַ דַּעַת וּמְזִמָּה. / וַיִּגַּשׁ אֵלָיו וַיֹּאמַר ״בִּי
אֲדוֹנִי, / לְכָה אִתִּי אֶל מָקוֹם פְּלוֹנִי, / וּמִי יְמַלֵּל
גְּבוּרוֹת חַסְדֶּךָ / אֲשֶׁר תַּעֲשֶׂה אֶת עַבְדֶּךָ / כִּי תַשְׁבִּית
הַיּוֹם מְדָנִים עֲצוּמִים / וְתַחְשֹׂךְ רְצִיחָה וּשְׁפִיכוּת דָּמִים״.
/ וַיַעַן הָאִישׁ וַיֹּאמַר ״צִנִּים פַּחִים אֶדְרֹךְ / בְּלִי נַעַל
וּבְלִי שְׂרוֹךְ, / אַחֲלִיק פַּטִּישׁ וּפַעַם אֶהֱלֹם / מִשּׁוּם
דַּרְכֵי שָׁלוֹם״.

וַיְבִיאֵהוּ הָעֵט אֶל אָהֳלוֹ / וַיֹּאמֶר לוֹ /
״אֲדוֹנִי, כַּטּוֹב בְּעֵינֶיךָ / הִנֵּה הַבַּיִת לְפָנֶיךָ, /
לְאָרְכּוֹ וּלְרָחְבּוֹ, / שִׂים עֵינֶיךָ בּוֹ / וּבְכָל הַכֵּלִים /
הַקְּטַנִּים עִם הַגְּדוֹלִים, / וְשִׂים כָּל כְּלִי עַל כַּנּוֹ / אֵת
כָּל עוֹרֵב לְמִינוֹ״. / וַיָּבוֹא הָאִישׁ מַחֲזִיק בְּפִלְכּוֹ /
וַיִּפֶן כֹּה וָכֹה / וַיַּרְא צַפַּחַת בְּשֶׁמֶן זַיִת / וּמְנוֹרָה
בְיַרְכְּתֵי הַבַּיִת / וַיַּעַשׂ פְּתִילוֹתֶיהָ / וְהֶעֱלָה אֶת
נֵרוֹתֶיהָ. / וַיִּמְצָא מַחַט וּבְגָדָיו הָיוּ קְרוּעִים / וַיִּתְפֹּר
בּוֹ אֶת כָּל הַקְּרָעִים. / וְעוֹד נָתַן אֶת עֵינָיו כִּמְרַגְּלִים /
וַיַּרְא בְּתוֹךְ הַכֵּלִים עֵט יְרַקְרַק כֵּבָּרָק מַזְהִיר, / וְהִנֵּה
הוּא הָיָה סוֹפֵר מָהִיר, / וַיְמַהֵר לְהוֹצִיאוֹ מִבֵּינֵיהֶם /
וַיִּכְתֹּב בַּסֵּפֶר לְעֵינֵיהֶם. / וַיְהִי מְמַשֵּׁשׁ עַד צָהֳרַיִם /
וַיִּמְצָא הַמִּסְפָּרַיִם, / וַיַּעַשׂ בָּהֶם צִפָּרְנֵי הַיָּדַיִם /
וַיְגַלֵּחַ שְׂעָרוֹ וְעָשָׂה שְׂפָמוֹ. / וְהָעֵט קָם מִמְּקוֹמוֹ /
וְלֹא יָכֹל לְהִתְאַפֵּק, / כִּי לִבּוֹ דוֹפֵק, / וַיִּקְרָא בְּקוֹל רָם
וַיֹּאמֶר ״דַּי!״ / וְאֶל הַזְּקֵנִים אָמַר ״אַתֶּם עֵדַי!״ / וַיַּחֲזֵק
בָּאִישׁ וַיִּשָּׁקֵהוּ וַיֹּאמַר ״טָרַחְתָּ בַּעֲדֵנוּ כַּמָּה וְכַמָּה
/ תְּהִי מַשְׂכֻּרְתְךָ שְׁלֵמָה!״ / וְהִתְחִיל בְּשִׂמְחָה מְפַזֵּז
וּמְכַרְכֵּר / מְסַפֵּחַ חֲמָתוֹ וְאַף שַׁכֵּר, / יַזִּיל כַּטַּל
אֲמָרִים / וַיִּכְתֹּב בַּשָּׁעָה הַהִיא מֵאָה סְפָרִים, / כְּלַפֵּי
הַמִּסְפָּרַיִם דְּבָרִים יַטִּיחַ / יָרִיעַ אַף יַצְרִיחַ / וַעֲלֵיהֶם
יִתְגַּבָּר. / וְלֹא יָכְלוּ לַעֲנוֹת אוֹתוֹ דָבָר, / כִּי נִבְהֲלוּ
מִפָּנָיו וְנֶאֱלָמוּ / בּוֹשׁוּ וְנִכְלָמוּ, / כִּי הָעֵט שָׁת לָהֶם
מוֹרָא, / אֵין קוֹל עֲנוֹת גְּבוּרָה / וְאֵין קוֹל עֲנוֹת חֲלוּשָׁה /
וְהָיְתָה הַפְּצִירָה פִים לְמַחֲרִישָׁה.



וַיַּעַשׂ הָעֵט מִשְׁתֶּה גָּדוֹל לְכָל אוֹהֲבָיו /
בְּיוֹם הִצִּיל יְיָ אוֹתוֹ מִכַּף כָּל אוֹיְבָיו. / וַיָּקוּמוּ כָל
עֲדַת הַקָּנֶה / וַיַּעֲבִירוּ קוֹל בַּמַּחֲנֶה / ״מִטַּעַם הַמֶּלֶךְ
וּגְדוֹלָיו: / כָּל אָדָם יִפְקֹד כֵּלָיו, / כָּל כְּלִי כַּאֲשֶׁר
פָּעַל יְיָ לַמַּעֲנֵהוּ / וְלֹא יִשְׁלַח יָדוֹ בִּמְלֶאכֶת רֵעֵהוּ, /
פֹּעַל פָּעַל מִתְּחִלַּת בְּרִיָּתוֹ / לֹא יַחֲלִיפֶנּוּ וְלֹא יָמִיר
אוֹתוֹ, / לֹא יִהְיֶה כְלִי גֶבֶר עַל אִשָּׁה / וְלֹא יִלְבַּשׁ גֶּבֶר
שִמְלַת אִשָּׁה. / הָעֵט לִכְתֹּב וְהַמִּסְפָּרַיִם לְגַלֵּחַ, / וְאֵשׁ
עַל הַמִּזְבֵּחַ, / חֶרְמֵשׁ לִקְצֹר וּמֶתֶג לִבְלֹם – / וְכָל הָעָם עַל
מְקוֹמוֹ יָבוֹא בְשָׁלוֹם״. / וְהַקּוֹל נִשְׁמַע בְּכָל פִנָּה /
וַתַּעֲבֹר הָרִנָּה / וְכָל כְּלִי לְעֶבְרוֹ פָנָה / וְסֵדֶר חֻקּוֹ
וּמִנְהָגוֹ לֹא שִׁנָּה / וַתִּשְׁקֹט הָאָרֶץ אַרְבָּעִים שָׁנָה.
