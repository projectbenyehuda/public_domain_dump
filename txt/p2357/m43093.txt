


	
	אֲנִי אֶצְעַק
	



ואם אלה השירים לחן אבו עתאבא נאמרים. יכול לשנותם בשאר ניגונים. לפי שהם מרובעים. והבוחר יבחר איזו ניגון מכשירים הנבחרים.

סימן: א״ב – כמה אותיות כפולות או משולשות.






אֲנִי אֶצְעַק לְךָ בְלֵב חֳמַרמַר

וְלִנְדוּדְךָ הִנֵּה שָׁלוֹם לִי מַר מַר

וְהַחְזֵר לִי לְמָר דְּכֵי וּמֹר מַר

וְקִנְּמָן בְּשָׂמִים הַחֲשׁוּבָה.





אָשִׁירָה לָךְ עֲלֵי עָשׂוֹר וְנֵבֶל

וְהַשְׁמִידָה בְּבַקָּשָׁה וְנָא בֵל

תְּשִׁיתֵמוֹ כְמוֹ חָצִיר וְנָבֵל

בִרְצוֹנְךָ לְעִיר צִיּוֹן הֵיטִיבָה.





בְּלֵב יַמִּים אֲנִי נִטְרָף כָּאֳנִי

וְנָד כֹּחִי וְגַם חֵילִי וְאוֹנִי

שַׂמְּחֵנִי מֵעָצְבִּי גַם מֵאוֹנִי

וּתְשׁוּעָתְךָ מְהֵרָה הִיא קְרוֹבָה.





בִּכְבָלוֹת שָׂמוּ רַגְלַי לְנֶגְדִּי

וְגַם הָפְכוּ לְדִין תּוֹרָה לְנֶגְדִּי

וְשִׁוִּיתִי אֶת ה׳ לְנֶגְדִּי

יְדִינֵם יַעֲשֶׂה לִי אוֹת לְטוֹבָה.





בְּלֵב שָׁלֵם הָלוֹךְ אֵלֵךְ וַאֲזַמֵּר

וְאוֹיְבֵינוּ מַהֵר כַּלֵּה וְזַמֵּר

וְזִמְרַת אַרְצֵנוּ בָּרֵךְ וְזַמֵּר

וְתַשְׂבִּיעַ לְנֶפֶשׁ הָרְעֵבָה.





גְּדוֹל עֵצָה וְהוּא רַב עֲלִילִיָּה

מִבּוֹר שָׁאוֹן וְטִיט יָוֵן עֲלֵה לִי יָהּ

וּלְמִתְעוֹלֵל בְּרֶשַׁע הָעֲלִילִיָּה

עֲלֵי רֹאשׁוֹ הַכֵּה אוֹתוֹ מַכָּה רַבָּה.





דָּגוּל הַפֵּל אוֹיְבִי לַבּוֹר אֲשֶׁר כָּרָה

אֲשֶׁר אָכַל לְמָמוֹנִי עָלָיו כָּרָה

וְכָל מָמוֹן אֲשֶׁר קָנָה וְגַם כָּרָה

הַחְזֵר לִי לִנְחֹשֶׁת זְהַב שְׁבָא.





הַצִּילֵנִי מִיַּד תֵּימָן וְאוֹמָר

כִּי לֹא אֵדַע מַה אֶעֱנֶה וְאֹמַר

אֲרוֹמִמְךָ צוּר גּוֹאֲלִי וְאֹמַר

אַחְרִיתֵנוּ מֵרֵאשִׁיתֵנוּ הֵטִיבָה.





הַבֵּט וּרְאֵה כְּלִמָּתִי וְכָסְפִּי

אֲשֶׁר כָּלְתָה פְרוּטָתִי וְכַסְפִּי

בְּךָ חִשְׁקִי וְתַאְוָתִי וְכַסְפִּי

בְּחַצְרוֹת בֵּיתְךָ אֵלִי אֵשֵׁבָה.





הַצֵּל כָּחוּשׁ אֲשֶׁר נָשׁוּךְ וְנָחוּשׁ

מִיַּד חָזָק כְּמוֹ בַרְזֶל וְנָחוּשׁ

וּבִמְהֵרָה פְדֵה אוֹתִי וְנָא חוּשׁ

לְעֶזְרָתִי נֶחָמָה הַכְּתוּבָה.





הֲיִגְאֶה שּׁוֹר עֲלֵי מַאְכַל בְּלִילוֹ

וַחֲסַר לֵב כִּי יִבְטַח עַל כְּלִי לוֹ

וְלֹא יִזְכֹּר זְמַן יַעְכֹּר כְּלֵילוֹ

וְגַם יוֹמוֹ אֲשֶׁר הוֹלֵךְ וְלֹא בָא.





וְיִשְׁתּוֹמֵם לְבָבִי כִּי עֲלֵי אָח

אִשֶּׁה הִבְעִיר לְתוֹךְ קִרְבִּי עֲלֵי אָח

וּבְהֵאָנְחִי יֹאמַר הֶאָח עֲלֵי אָח

וְנַפְשִׁי אוֹמְרָה גַּם זוֹ לְטוֹבָה.





וְעוֹל מִצְווֹת שְׂאִי נָא גַּם שְׁבִי לִי

וּבָהֶם אֵלְכָה דֶרֶךְ שְׁבִילִי

וְאַתְּ רַעְיָה בְּתוֹךְ לִבִּי שְׁבִי לִי

וּמַשְׂכֻּרְתֵךְ בְּסִפְרָתוֹ כְתוּבָה.





וְקוֹרוֹת הַזְּמַן לִי בָא וּבָגַד

וְלֹא הִשְׁאִיר לָנוּ מִחְיָה וּבֶגֶד

וּמִצָּרָה קָרָאתִי לוֹ וּבָא גָד

וְאָז רִוָָּּה לְנַפְשִׁי הָעֲזוּבָה.





וְאִם יִהְיוּ חֲטָאֵינוּ כְשָׁנִים

מְחֹל וּסְלַח חֲדָשִׁים גַם יְשָׁנִים

וְכָל זְמָן וְכָל יָמִים וְשָׁנִים

אֲנִי לְךָ בְּכָל לִבִּי אָשׁוּבָה.





וְאוֹיְבִי זֶה כְמוֹ נָחָשׁ עֲלֵי צוּר

וּמִתְפָּאֵר וּמִתְרוֹנֵן עֲלֵי צוּר

זְמָמוֹ אַל תָּפֵר, אֵלִי, עָלַי צֹר

לִתְחִנָּתִי צוּרִי מַהֵר הַקְשִׁיבָה.





וְהֵן הִנֵּה הָיוּ עוֹכְרַי מֹעֲלַי

וְהֵרֵעוּ לְגַם טַפַּי וְעָלַי

רְדִידִי נָשְׂאוּ הֵמָּה מֵעָלַי

לְרַעְיָתִי מַהֵר צוּרִי הָשִׁיבָה.





וְתַאְוַת הָעוֹלָם תַּרְתִּי בְּרַגְלִי

וְעָרְבָה לִי תוֹרָתְךָ בְרַגְלִי

וְכָל חֶסְרוֹן אֲשֶׁר נִמְצָא בְרַגְלִי

הַסְתִּירֵנִי בְצֵל יָדְךָ הָרְחָבָה.





וְהִזָּהֵר מֵאֵשׁ יִצְרָךְ כִּי בָעַר

וְלֹא תִהְיֶה לְאִישׁ כְּסִיל וּבַעַר

וְרַע מִקִּרְבְּךָ תָסִיר וּבַעֵר

וָלֹא תָמִיר לְנַפְשָׁךְ הַחֲשׁוּבָה.





זַךְ וְיָשָׁר אֶת עַמְּךָ מַהֵר תִפְקֹד

וְעֹל אוֹיְבַי מֵעָלֵינוּ אוֹתוֹ תִפְקֹד

וְאֶת שׁוֹפְטַי כְּמִקֶּדֶם תָשׁוּב תִּפְקֹד

וְנָגִילָה נִתְעַלְּסָה בָאַהֲבָה.





זְכֹר צוֹרֵר יְהוּדִים וּתְפוֹרֵר פּוּר

אֲשֶׁר חָשַׁב לְאַבְּדָם בְּהַפִּיל פּוּר

וּפָנָיו קִבְּצוּ הֵמָּה כְפָרוּר פּוּר

מְחֵה שְׁמָם וְתִשְׂרְפֵם אֵשׁ לֶהָבָה.





חֹן עַם יָצָא מִן הַבֶּטֶן זָהָב סָגוּר

וְהַךְ אֻמִּים בְּשַׂפַּחַת וּשְׂאֵת סָגוּר

וְשָׁמְרֵם כַּחוֹתָם תּוֹךְ חוֹתָם סָגוּר

וְתוֹלִיכֵם לִמְחוֹז חֶפְצָם בְּחִבָּה.





טוֹב וְיָשָׁר דּוֹדִי יָקָר וְנָעִים

שִׂימָה אֶת אוֹיְבֶיךָ נָדִים וְנָעִים

וּבַשֵּׂר עַבְדְּךָ וּשְׁלַח לִי נָא עִם

אֵלִיָּה הַנָּבִיא בְשׂוֹרָה טוֹבָה.





יְחִידָתִי אֶזְכְּרֵכִי זֹאת הַפַּעַם

וְאֶת רוּחַ מְשַׂנְאַיִךְ אֶפְעֹם פַּעַם

וּמַה יָּפוּ פְעָמַיִךְ פַּעַם פַּעַם

בַּת נָדִיב אֲשִׁיבֵךְ לִי בְשׁוּבָה.





כְּגַל נָעוּל תִּסְגֹּר בַּעֲדִי וּנְעַל

וְשִׂים רוֹדְפַי תַּחַת רַגְלַי כְּנַעַל

וְרַחֵם נָא בְבַקָּשָׁה וְנָא עַל

שְׁאֵרִית הַפְּלֵיטָה הָאֲהוּבָה.





לְהַר צִיּוֹן מַהֵר תִּבְנֶה וְעָרַי

אֲשֶׁר הָרְסוּ יְסוֹדוֹתַי וְעָרַי

וְהַכְרִיתָה לְכָל שׂוֹנְאַי וְעָרַי

אֲשֶׁר אָמְרוּ עָרוּ עָרוּ עַד הַיְּסוֹד בָּהּ.





לְבֵית קָדְשָׁךְ מַהֵר אֵלֵךְ בְּרַגְלִי

וְאָרוּצָה כְאַיָּל לֹא כְרַגְלִי

וְאָשִׁיר לָךְ כְּעֵת עֲלוֹת לְרַגְלִי

שְׁבוּתֵנוּ הָשֵׁב וּמַהֲרָה שׁוּבָה.





לְבַת אֱדוֹם הָשֵׁב לָהּ אֶת גְּמוּלָהּ

וְחֹן נַפְשִׁי אֲשֶׁר מִטֹּב גְּמוּלָה

וְאַתְּ אֵלִי בְּרֹב טוּבְךָ גְמֹל לָהּ

וּכְמוֹ קֶדֶם שְׁכֹן עִמָּהּ בְּקִרְבָּהּ.





מְחוֹלֵל כֹּל שְׁמַע קוֹלִי וְשִׂיחִי

וּבָרֵךְ כָּל עֲצֵי פֶּרְיִי וְשִׂיחִי

וּמִטָּתִי עַרְשִׂי אַמְסֶה בְשִׂיחִי

עֲדֵי תִהְיֶה לְחוֹמַת אֵשׁ סְבִיבָהּ.





נָא חִישׁ חַזֵּק עַמְּךָ וְתִגְדֹּר פֶּרֶץ

חֶפְצָם וְקִנְיָנָם יִפְרְצוּ פֶרֶץ

וּשְׁלַח לִי צִיר נֶאְמָן עִמּוֹ בֶן פֶּרֶץ

וְאָז תִּקְרָא לְרַעְיָתִי חֶפְצִי בָהּ.





סְבָבוּנִי אֵלִי עֲדַת מְרֵעִים

וְהֵם נִרְאִים כְּאוֹהֲבִים וְרֵעִים

וּבְהָצֵר לִי בִתְרוּעָה הֵם מְרִיעִים

וְשָׁמְרֵנִי מִידֵי עוֹזֵר עִמּוֹ אֵיבָה.





עֲנֵנִי יָהּ עֲנֵנִי דָר מְעוֹנָה

וּפְדֵה גוֹלָה וְסוּרָה גַּם מְעֻנָּה

וְחִישׁ כּוֹנֵן לְבֵית מִקְדָּשׁ מְעוֹנָה

וְהָאַרְמוֹן כְּבָרִאשׁוֹן הָשִׁיבָה.





פְּדֵה נַפְשִׁי מִיַּד אֱדוֹם וַעֲרָב

וְשִׁירִי זֶה יָהְנֶה לְךָ וְעָרַב

וְהַבֵּט כִּי פָנָה הַיּוֹם וְעָרַב

הֲשִׁיבֵנוּ לְךָ אֵלִי נָשׁוּבָה.





צָעַקְתִּי לְעוֹשֵׂה כְסִיל וְאֶת עָשׁ

שִׂימָה אוֹיְבִי כְמוֹ בֶגֶד אֲכָלוֹ עָשׁ

וּמִכַּעַס עֲצָמָיו עָשְׁשׁוּ עָשׁ

בִּרְאוֹתוֹ לְיִנּוֹן וְאֵלִיָּה בָא.





קָרָאתִי יָהּ הַמַּעֲלֶה נְשִׂיאִים

וְאוֹהֵב הָעֲנָוִים לֹא נְשִׂיאִים

אֲבָל חָפֵץ לְהַנְשִׂיא נָשִׂיא אִם

שְׁפַל בֶּרֶךְ וְלֹא יִכְפֹּר בְּטוֹבָה.





רְאוּ מֶה עָשׂוּ לִי רֵעַי וּבָנַי

וְהִפְרִידוּ לְבֵין בָנַי וּבֵינַי

וְהַחְזֵר לִי צוּרִי שִׂכְלִי וּבִינַי

וְרַחֵם נָא לְנַפְשִׁי הָעֲצוּבָה.





רְאֵה עָנְיִי וְגַם לַחְצִי וְרוֹשִׁי

וּמִתְּהוֹם צָפוּ מַיִם לְרֹאשִׁי

וְהִשְׁקוּנִי לַעֲנָה גַם מֵי רֹאשִׁי

גְּמוּלֵיהֶם לְרָאשֵׁיהֶם הָשִׁיבָה.





רְצֵה שִׁירִי כְמוֹ אִשֶּׁה וְעוֹלָה

וְרַחֵם נָא עֲלֵי טַפָּהּ וְעוּלָהּ

וּבִמְהֵרָה שַׁבֵּר כַּבְלָה וְעֻלָּהּ

וֶאֱמֹר לָהּ כִּי הִנְּנִי בָא.





שׁוּבָה בְנִי מֵהַוָּתָךְ וְסִכְלָךְ

אֶבֶן שָׁלֵם צֶדֶק יִהְיֶה בְכִיס לָךְ

וְדַע כִּי ה׳ יִהְיֶה בְכִסְלָךְ

וְהוּא יִשְׁמֹר לְהַיּוֹצֵא וְלַבָּא.





שְׁעֵה נִיבִי כְקָרְבָּנִי וְלַחְמִי

וּבָרֵךְ כָּל מַעֲבָדִי וְלַחְמִי

וְעָזְרֵנִי עֲלֵי יִצְרִי בְלָחְמִי

עִמּוֹ תָמִיד אֲנִי כָל יוֹם אָרִיבָה.





שִׂמְחִי נַפְשִׁי וּבִטְחִי בוֹ עֲדֵי עַד

וְהוּא יַעְדֵּךְ עֲדָיִים וַעֲדֵי עַד

וּבֹקֶר תֹּאכְלִי עַד וַעֲדֵי עַד

וְגַם עֶרֶב נְהַלֵּךְ בּוֹ לְטוֹבָה.





תְּהִי מָהִיר כְּאִישׁ גִּבּוֹר וְלֹא נַעַר

וְתִמְשֹׁל כַּאֲשֶׁר יִצְחַק וְנָעַר

וְאֶת רוֹדְפָךְ יְשַׁקְּעֵם וְנִעַר

בְּתוֹךְ הַיָּם מֵי הַתְּהוֹם חֳרָבָה.
