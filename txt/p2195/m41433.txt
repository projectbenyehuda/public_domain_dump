


	
	נְזַמֵּר שִׁיר וְשֶׁבַח
	



לכבוד שיך ששון בן צאלח דוד

סימן: נשׂים.






נְזַמֵּר שִׁיר וְשֶׁבַח גַּם תְּהִלָּה / לְשַׂר גָּדוֹל וְהוּא כֶתֶר לְגוֹלָה

שְׂפָתָיו יִנְטְפוּ מֹר וַאֲהָלוֹת / בְּעֵת כִּי יִפְתְּחָה אֶת פִּיו בְּמִלָּה

יְהִי בָרוּךְ בְּצֵאתוֹ גַם בְּבוֹאוֹ / בְּרֹב עֹשֶׁר וּמֶמְשָׁלָה גְדוֹלָה

מְרוֹמָם יִהְיֶה עַל כָּל בְרָכוֹת / וְהַצְלָחָה מְרֻבָּה גַם כְּפוּלָה

אֲדוֹנִי שָׁלְחו לְמָר וְיִחְיֶה / כְּתָב עִבּוּר לְהוֹרוֹת אֶת שְׁבִילָהּ

וְנִכְתַּב עֵת מְאֹד צַר הוּא לְלָקוּת / בְּמַאְמַר צוּר וְהוּא נוֹרָא עֲלִילָה

וְיָדַעְתִּי כִי יֵשׁ הֶפְרֵשׁ וְהֶבְדֵּל / בְּחֶשְׁבּוֹנִי וְחֶשְׁבּוֹן רֹאשׁ קְהִלָּה

וְאָמַרְתִי אֲגַל אֹזֶן יְדִידִי / לְבַל יַחְשֹׁב עֲלֵי עַבְדּוֹ מְעִילָה

וְעוֹד שֶׁבַח וְלֹא עָמַד בְּחִלּוּף / הַהַבָּטָה לְעִיר וּכְרָךְ גְּדוֹלָה

וְיֵשׁ הֶפְרֵשׁ לְבֵין בָּבֶל וּמַארְדִין / כְּשֵׁשׁ מַעְלוֹת בְּאֹרֶךְ הַמְּסִלָּה

וְנִכְתֹּב לַאֲדוֹן כְּפִי יְרוּשְׁלֵם / לְלָקוּת צַר וְהוּא לָךְ הוּא תְחִלָּה



תְּשַׁע עֶשֶׂר דְּקֵי שָׁעָה וְאַחַר / לְיוֹ״ד שָׁעוֹת בְּיוֹם דָּלֶת בְּלַיְלָה



וְסוֹף לָקוּת שְׁלֹשׁ עֶשְׂרֶה בְשָׁעוֹת / וְכַ״ג דַקִּים וְתַעְלֶה לוֹ תְעָלָה



כְּפִי בָבֶל מְהֵרָה יִהְיֶה לִקּוּי / בְּחֵ״ת שָׁעוֹת וְנוּ״ן דַקִּים תְּחִלָּה



וְסוֹף לָקוּת אַחַת עֶשְׂרֵה בְשָׁעוֹת / וְרַ״ן דַקִּים וְנִגְלָה בַהֲמֻלָּה



וְיֵשׁ הֶפְרֵשׁ כְּמַ״ה דַקִּים בְּקֵרוּב / לְבֵין בָּבֶל וְצִיּוֹן עִיר תְּהִלָּה



לְמַארְדִּין יֵשׁ כְּמַ״ה דַקִּים בְּקֵרוּב / וְאִם תִּהְיֶה שְׁגִיאוֹת תֵּן מְחִילָה



לְאֵל עֶלְיוֹן אֲנִי פּוֹרֵשׂ לְכַפַּי / וְשׁוֹאֵל בַּתְּחִנָּה גַם תְּפִלָּה

שְׂשׂוֹן לִבִּי יְהֵא מֻצְלָח וְעוֹמֵד / וְדַרְכּוֹ יִהְיֶה דֶרֶךְ סְלוּלָה

וְיִמְצָא חֵן וְשֵׂכֶל טוֹב בְּעֵינֵי / מְלָכִים גַּם סְגָנִים חֵן וְחֶמְלָה

וְיַחְזֹר חִישׁ בְּרֹב מָמוֹן וְעֹשֶׁר / בְּלִי מִדָּה וְרֹב כֶּסֶף וְשִׂמְלָה

יְהִי מֻשְׂבָּע בְּשֹׂבַע הַבְּרָכוֹת / וְשָׂמֵחַ בְּרֹב שִׂמְחָה וְגִילָה

אֲדוֹנִי יַאֲרִיךְ יָמִים בְּמָלְכוֹ / וְכָל יָמָיו יְהִי נָשִׂיא לְגוֹלָה

יְהִי זַרְעוֹ בְדוֹמֶה לוֹ וְיוֹתֵר / וְכִנְטִיעָה עֲלֵי מַיִם שְׁתוּלָה

וְיַגְבִּיהַּ לְכוֹכַב מַעֲרַכְתּוֹ / וְגַל יָרוּם מְאֹד יַגְדִּיל בְּמַעְלָה

תְּפִלָּתִי לְבוֹרֵא רוּם וְתַחַת / יְקַבֵּל לִי וְיִשְׁמַע זֹאת שְׁאֵלָה.
