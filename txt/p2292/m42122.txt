


	
	עַל מוֹת
	



קינה. על מות משה מונטיפיורי






עַל מוֹת יְדִיד צְבִי נֵזֶר תִּפְאָרָה / אֶבְכֶּה בְמַר וְאֵבֶל זֶה לֹא אֶנְשֶׁה

כִּי שַׂר וְגָדוֹל נָפַל הוֹד וַעֲטָרָה / וַיַּעַל הַשָּׁמַיְמָה הָאִישׁ מֹשֶׁה.





אֲנִי לְקוֹל יַם הֲמוֹן צִירִי / מִיּוֹם נְפֹל יְדִידִי צְבִי נִזְרִי

נָשָׂאתִי קִינָה לְצִירִי וְשָׂרִי / הוּא הָאָדוֹן צִיר מֹשֶׁה מוֹנְטִיפְיוֹרִי.





הוֹי! הוֹי! עֲדַת לֹנְדֹן תִּזְעַק בְּחֶבֶל / וְקוֹל רַמַסְגֵיט תִּצְרַח בְקוֹל נֶהִי

מַקְהֲלוֹת יִשְׂרָאֵל כֻּלָּם אֶל אֵבֶל / נִקְבְּצוּ לִשָּׂא מִסְפֵּד הֶגֶה וָהִי.





אֵיךְ שַׁחַר שְׁמֵי גִיל הוֹדֵינוּ קָדָר / יַד אֱלֹהִים נָגְעָה בִנְשִׂיא הָאָרֶץ

אוֹי רוֹעֵה צֹאן קָדָשִׁים פִּתְאֹם נֶעְדָּר / אַיֵּה בֵית נָדִיב הָעוֹמֵד בַּפֶּרֶץ.





צִיּוֹן הַיְקָרָה חִישׁ קַדְרוּת לָבְשָׁה / כִּי רְדִיד תִּפְאַרְתָהּ נִשָּׂא מִמֶּנָּה

לִמְצֹא לָהּ יָדִיד כָּמוֹהוּ נוֹאֲשָׁה / צִיר אֲשֶׁר מֵימֵי הַיְשׁוּעָה יַשְׁקֶנָּה.





דַּמֶּשֶׂק עַל הוֹד מוֹשִׁיעָהּ מַר תִּצְרַח / וּמַארוֹקָה עַל מָעֻזָּהּ תֶּהֱמֶה

גַּם הַר מִשְׁכְּנוֹת שַׁאֲנַנִּים יִצְרַח / כִּי הַר פְּרָחִים מְכוֹנְנוֹ נִדְמֶה.





מַלְאֲכֵי הַשָּׁלוֹם בְּמַר יִבְכָּיוּן / אֱלֹהִים נַחֲלֵי דִמְעָה יַזִּילוּ

עַל מוֹת מֹשֶׁה שַׂר יִשְׂרָאֵל יֶהֱמָיוּן / וּלְשֶׁבֶר חוֹבְבֵי בַת צִיּוֹן יָחִילוּ.





שְׁחָקִים עָטְפוּ חֶשְׁכַת עָנָן אֹפֶל / סַהַר עֲלָטַת בַּלָּהוֹת הֶאֱפִילוּ

וְכוֹכְבֵי נֶשֶׁף יְפִי נָגְהָם אָפֵל / הֵן יִפְעַת שִׁמְשָׁם כָּבְתָה אֵיךְ יַצְהִילוּ.





עִיר חֶבְרוֹן גִּנְחִי וְנוּחִי בִדְאָבָה / לְמַכָּתֵךְ שָׂבְעָה נַפְשִׁי יְגוֹנִים

הֵן שִׂמְחָתֵךְ וְאוֹר שִׁמְשֵׁךְ עָרְבָה / כִּי נֵזֶר עֲדָיַיִךְ עָלָה לִמְעוֹנִים.





מִי יִתֵּן וְאַזִּיל דִּמְעוֹתַי כִּרְבִיב / וּמִבָּבַת עֵינַי יִזַּל כְּמִיכַל

אוּלַי אֲכַבֶּה יְקוֹד אֵשׁ בְּקִרְבִּי / כִּי רִשְׁפֵּי נְדוּדִי כַּלְכֵּל לֹא אוּכַל.





לָכֵן זֶרֶם דֶּמַע יַחְטֹף מַרְגּוֹעִי / וְעֵת אֶדְלֵם דָּלְלוּ מֵאֵשׁ כִּידוֹ

גַּם זֵכֶר פָּעֳלוֹ יַעְשֹׁק מַדָּעִי / וַיִּגְדַּל חֶבֶל צִירִי עַל נְדוּדוֹ.





הוֹי זְמַן הִכָּה כַחֵץ שְׂשׂוֹן לִבֵּנוּ / וּבְקִרְבֵּנוּ כִלְיוֹתֵינוּ יְפַלַּח

וְדִכָּנוּ וְשָׂם נֶצַח כְּאֵבֵינוּ / כִּי מַחְמַד עֵינֵינוּ מִמֶּנּוּ שֻׁלַּח.





נָא הַנְחֵה נַפְשׁוֹ תַחַת כְּנָפֶיךָ / גַּם לִימִינוֹ יְהוּדִית הַמַּעֲטִירָה

יַחְדָּו יְנוּבוּן בְּטַל גַּן עֲדָנֶיךָ / וַאֲלֵיהֶם אוֹר פָּנֶיךָ הָאִירָה.
