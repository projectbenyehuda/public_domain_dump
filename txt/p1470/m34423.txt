


	
	אוֹת לְטוֹבָה
	







א






אוֹת לְטוֹבָה חָשָׁבְתָּה / בְּאַחַר לַמַּבּוּל יָשָׁבְתָּה

גַּם בַּיְתָה יְחִידִים הוֹשָׁבְתָּה / דְּגָלִים בְּתֵיבָה יִשָּׁבְתָּה.





הִתַּרְתָּה אֲסוּרִים / וְהוֹצֵאתָה אֲ[סִירִי]ם



זָכַרְתָּה סְגוּרִים / חָשָׁבְתָּה סגו..ם.





טַפָּם בַּתֵּיבָה עֲבוּר הִצְנִיעוּ / יְשִׁיבָה עֲלֵי גֶיא נִצִּיעוּ

כִּי מְעַט הָיוּ וְהַרְבֵּה הָצְבָּאוּ / לְעֵת לְמִשְׁפְּחוֹתָם יָצָאוּ.





יַשְׁמִיד צָרָיו / יִשְׁמֹר צוּרָיו / יְמַגֵּ[ר] צוֹרְרָיו / יְמַגֵּן נְצוּרָיו.




ב






מֵחַלְצֵיהֶם אֲדָמָה הִשְׁרִיצָה / נְטִיעָתָם בָּאָרֶץ הִפְרִיצָה

שָׂרִיגִים גַּפְנָם הֵנִיצָה / עוֹלָתָה כְּעָלְתָה נִצָּהּ.





[פֻּלְּטוּ בְּ]טוֹב מְעַט לַצַּדִּיק / צִמְחֵי אִישׁ צַדִּיק



[ק]… יְסוֹד עוֹלָם הוּא הַצַּדִּיק / רַכָּיו עִמּוֹ הִצְדִּיק.





[ש]… וּכְחַלָּה הִפְרַשְׁתָּם / שֶׁפַע טוּב עוֹלָם הוֹרַשְׁתָּם.





פְּלִיטִים מִסַּגְרִירֵי שֶׁטֶף / בְּהִסָּכֵר מַעְיָנוֹת וּבְהִסָּ[תֵם]…



צְפוּנִים עַד יַעֲבֹר זַעַם קֶצֶף / וְיָצְאוּ נִצּוֹלִים מִשֶּׁצֶף.





[ק]….. שמותם כְּלוּלָה לָהֶם / וּמוֹרָא מוֹצִיאָם עֲלֵיהֶם



רֻחַק או…… בחם / וְנֶאֱרַר וְנִשְׁתַּנָּה מֵהֶם.





שְׁנַיִם מַאֶחָד הוּטְבוּ / כִּי מַעֲשֵׂ[יהֶם] הֵיטִיבוּ



תֵּבֵל וְחָרְבוֹתֶיהָ הוֹשִֵׁיבוּ … וּבִבְרָכָה זֶה לָזֶה הָשְׁווּ.




ז 1






נה….ו / מֵאֵלֶּה אַפְסֵי אֶרֶץ נֶאֱחָזוּ / מֵאֵלֶּה בָּתֵי בְרִיּוֹת נִבְנוּ



מֵאֵלֶּה גְּבוּלוֹת גֵּיא [נג…..] / מֵאֵלֶּה [ד….] דּוֹרוֹת נִדְרָכוּ



מֵאֵלֶּה הֲמוֹנֵי הֲמוֹנִים נה….ו / נה….ו / וִתּוּרֵי וְעָדִים [נו…]



מֵאֵלֶּה ז…… זֹאת נִזְרָעוּ / מֵאֵלֶּה חִדּוּשֵׁי חֶלֶד נֶחֱלָפוּ



מֵאֵלֶּה טֶכֶס טִירוֹת נִיט….. / מֵאֵלֶּה יקרות יל….ם נִרְאוּ



מֵאֵלֶּה כַּנַּת כֹּל נִכְלְלוּ / מֵאֵלֶּה לַהֲקוֹת למו…. [נִ]לְקָחוּ



מֵאֵלֶּה מַחֲנוֹת מוֹשָׁבוֹת נִמְלְאוּ / מֵאֵלֶּה נְווֹת נְחָלוֹת נִנְחָלוּ



מֵאֵלֶּה סְרִידֵי שׁ… [נס…] / מֵאֵלֶּה עוֹבְדֵי עֵדוּת [נַעֲ]שׂוּ


מֵאֵלֶּה פִּנוֹת פֹּעַל נִפְרָדוּ / מֵאֵלֶּה צִדְדֵי צִיָּה נצ….


מֵאֵלֶּה קִינֵי קְהָלִים נִקְרָאוּ / מֵאֵלֶּה רִבְבוֹת רַבִּים נִרְבָּבוּ



מֵאֵלֶּה שָׁרְשֵׁי שׁ…. [נשׁ…] / מֵאֵלֶּה תּוֹדוֹת תֵּבֵל נת….ו.




ז 2


בְּתִקְוָה / [אֲשֶׁר לָמוֹ ת….] / [בּשׁ….… / אֲשֶׁר לָמוֹ שׁ…]



[בְּרָצוֹן / אֲשֶׁר] לָמוֹ רָצִיתָה / בְּקוֹשֶׁט / אֲשֶׁר לָמוֹ [ק….]



[בצ…. / אֲשֶׁר לָמוֹ צ…../ [בּפ….] / אֲשֶׁר לָמוֹ פָּעַלְתָּה



בְּעֵידוּת / אֲשֶׁר לָמוֹ עֲרָכְתָּה / בְּסִימָן / אֲשֶׁר לָמוֹ סָדָרְתָּה



בְּנֹעַם / אֲשֶׁר לָמוֹ נָצָרְתָּה / בְּמִחְיָה / אֲשֶׁר לָמוֹ מִנִּיתָה



בְּלִוּוּי / אֲשֶׁר לָמוֹ לִבָּבְתָּה / בְּכַעַס / אֲשֶׁר לָמוֹ כָּבָשְׁתָּה



בְּיֵשַׁע / אֲשֶׁר לָמוֹ יִמָּנְתָּה / בְּטוֹב / אֲשֶׁר לָמוֹ טִיעָמְתָּה



בְּחַיִל / אֲשֶׁר לָמוֹ חִזָּקְתָּה / בִּזְכוּת / אֲשֶׁר לָמוֹ זָכָרְתָּה



בְּוָתִיקוּת / אֲשֶׁר לָמוֹ וִיעָדְתָּה / בְּהַשְׂכֵּל / אֲשֶׁר לָמוֹ הוֹרֵיתָה



בְּדָבָר / אֲשֶׁר לָמוֹ דָּנְתָּה / בִּגְזֵרָה / אֲשֶׁר לָמוֹ גָּזָרְתָּה



בִּבְרִית / אֲשֶׁר לָמוֹ בֵּירָרְתָּה / בַּאֲמִירָה / אֲשֶׁר לָמוֹ אִמָּנְתָּה.




ח


כִּי תַרְבֶּה לְהָשִׁיב אַפָּךְ / וְתַשְׂגֶּה לְהַחֲזִיר



זַעְפָּךְ / וְכָלָה לֹא תַעַשׂ / כִּי לְמַעֲנָךְ תַּעַשׂ / וְרַחֲמֶיךָ



עַל כָּל מַעֲשֶׂיךָ / וְחֶמְלָתְךָ עַל כָּל בִּירְיוֹתֶיךָ / וְדוֹר



לְדוֹר יְשַׁבַּח / וְדוֹר לְדוֹר יְתַנֶּה / וְזֶה מִזֶּה יִשְׁמָעוּ /



וְזֶה לָזֶה יְלַמֵּדוּ / כִּי לָרִיק לֹא נִבְרָאוּ / וּבַעֲבוּר זֹאת



נִבְרָאוּ / לְבָרֵךְ שְׁמָךְ / לְהַלֵּל זִכְרָךְ /וּלְהוֹדוֹת לָךְ /



וּלְהַאֲמִין בָּךְ / וּלְהַקְדִּישָׁךְ וּלְהַאְדִירָךְ / כְּאַדִּירֵי



מַעְלָה בְּאַדֶּרֶת קְדֻשָּׁה נֶאְדָּר בַקֹּדֶשׁ אַדִּיר



וְקָדוֹשׁ.




ט


יוֹצְאֵי מִמַּסְגֵּר אֲשֶׁר הוֹצֵאתָה לִרְוָחָה יְצוּאֵיהֶם יֹאמְרוּ:



קדוש קדוש קדוש / יי צבאות מלא כל הארץ כבודו



קָדוֹשׁ מִבַּעֲלֵי שֵׁם וּמַעֲשֶׂה / קָדוֹשׁ מִבַּעֲלֵי לָשׁוֹן מֻבְחָר / קָדוֹשׁ מִבַּעֲלֵי שָׂכָר טוֹב



קדוש…..



קָדוֹשׁ מִשֵּׁם הֶעֱמִיד עוֹטֶה צִיצִית בְּשֵׁם וּמַעֲשֶׂה.

קָדוֹשׁ מִיֶּפֶת הֶעֱמִיד מַמְלִיצֵי דָת בְּלָשׁוֹן מֻבְחָר.

קָדוֹשׁ מֵהֶם הֶעֱמִיד מַשְׂכִּילִים דְּבֵקִים בִּבְרוּכִים.



קדוש……





מִמְּקוֹמוֹ בָּחַר בְּשֵׁם כִּי דָרַשׁ אֵשׁ דָּת.

מִמְּקוֹמוֹ בָּחַר בְּיֶפֶת הוֹגֵי נֹפֶת צוּפִים.

מִמְּקוֹמוֹ קֵרַב מֵהֶם גֵּרִים וְחוֹשְׁבֵי שְׁמוֹ.





מִמְּקוֹמוֹ הוֹצִיא מִתֵּיבָה יַלְדֵי…

[מִמְּקוֹמוֹ] בָּנָה עוֹלָמוֹ מִגֶּזַע תָּמִים

מִמְּקוֹמוֹ הוֹרִישׁ ל….. עוֹלָם לַחֲלָקִים.
