


	
	אָב לִבּוֹ כְּלִבָּךְ
	



בראשית טו א. אחר הדברים באלה: במחזה









א


אָב לִבּוֹ כְּלִבָּךְ מָצָאתָה / בְּטֹהַר לִבּוֹ כְּלִבּוֹ לוֹ הִמְצֵאתָה

גַּאֲוַת חַרְבּוֹ בְּכָל צַר הִמְצֵאתָה / דִּבְרֵי חֲסִידוּתוֹ מִצִּיתָה.





הַמִּמִּלְחָמָה אַחַר הוּבָא / וַדַּיי הִרְהוּרִים בְּלִבּוֹ חֻבָּא

זֵדִים אֲשֶׁר קִטֵּל בְּדָן וּבְחוֹבָה / חָרַד פֶּן יֵחָשֵׁב לוֹ חוֹבָה.





טַעַם הֲשֵׁיבוֹתוֹ בַּעֲתִירָה / יוֹפִי רוֹב שָׂכָר [לָ]ךְ אוֹתִירָה



כִּי דָמָם תַּרְתִּי לָךְ לְהַתִּירָה / לָךְ אָנֹכִי מָגֵן אַל תִּירָא.





אָדָם בִּדְמוּת צַרְתָּה / וּבְצֶלֶם צִײַַּנְתָּה / מַגֵּן עַם זוּ יָצָרְתָּה.




ב


מַה יִּתְרוֹן לִי אָב נִתְחַנָּן / נַפְשִׁי עֲגוּמָה וְלִבִּי מִתְאוֹנָן

שִׂמְחָה בְלִבִּי מַה יְּרֻנָּן / עֵץ סְרַק אָנֹכִי וְלֹא רַעֲנָן.





פָּעֳלִי מַה בְּכָל יְקָרִי / צִיץ עִקָּרִי עֲקָרִי בִמְקוֹרִי

קִנְיַן בִּרְכָתִי יִירַשׁ אֲרוּרִי / רֵיק אִם אֶפְנֶה וַעֲרִירִי.





שֵׁם צֶדֶק יֵשׁ לִי / שֶׁתֶל צֶדֶק אֵין לִי

תַּתִּי לָךְ וְנָתַתָּה לִי / מָה אֶתֶּן לָךְ וּמַה תִּתֶּן לִי.





אָזְנֶיךָ קַשּׁוּבוֹת / לְכָל מַחֲשָׁבוֹת / מַטְלִיל תֶּחִי [בְּר]וּחוֹת נִשָּׁבוֹת.




ג


יָהּ מַאֲמָרוֹת עֲשָׂרָה חָ[תָכְ]תָּה / וּבִלְשׁוֹנוֹת עֲשָׂרָה נְבוּאָה קָרָאתָה



[נְבוּ]אָה חָזוֹן צִוּוּי דִּבֵּר אֲמִירָה / הַטָּפָה [מָשָׁ]ל מְלִיצָה מַשָׂא חִדָה



יְקָר לְאֵיתָן אָ[ז יִי]דָּעְתָּה / כִּי בִשְׁלֹשֶׁת הַיְקָרִים שֶׁבָּהֶם לוֹ נוֹדָעְתָּה



יִדּוּעַ דִּבֵּר וְאֹמֶר וְחָזוֹן / [לְדַבֵּר] וְלֹאמַר לִשְׁמֹעַ וְלֶאֱזוֹן.




ד


לְאָב עַד לֹא חָז מַחֲזֶה / וְהֵבִין בְּכָל מַעֲשֶׂה

נִרְאֵיתָה בַּמַּחֲזֶה / וְנִהְיֵתָה לוֹ מַחֲסֶה

וַאֲשֶׁר הִרְהֵר בַּלֵּב / וְחִשֵּׁב בַּכְּלָיוֹת

יִדַּעְתּוֹ חוֹקֵר לֵב / וּבוֹחֵן כְּלָיוֹת

מֵאֲשֶׁר נִתְיָרָא / נַמְתָּה לוֹ אַל תִּירָא

וּבְדָבָר אֲשֶׁר נֻהַם / בּוֹ בְדָבָר נֻחָם

וּבְאֹמֶר כִּי נִתְחַשֵּׁב / בּוֹ בְּאֹמֶר נִתְיַשָּׁב



אָנֹכִי מָגִינָךְ / מִי יַמְגִּינָךְ / [אָנֹ]כִי מַחֲזִיקָךְ / מִי יַזִּיקָךְ



אָנֹכִי מַשְׂכִּירָךְ / מִי יַחֲסִירָךְ / הִבְטַחְתּוֹ קָדוֹשׁ.




ה


יְיָ אֱלֹהִים הִנְנָךְ ……נך / אֲשֶׁר בְּרַחֲמִים וּבְדִין חֲנַנְתּוֹ בְּכָל מַתָּנָךְ



בְּכֵן פָּץ אִם אֵין לִי / בְּרַחֲמִים תֵּן לִי / וְנָם אִם יֵשׁ לִי / בְּדִין יִנָּתֵן לִי



גָּעָה מַה תִּתֵּן לִי / וְיֵשׁ לִי וְאֵין לִי / מוֹלִיד יֵשׁ לִי / וְיֶלֶד אֵין לִי



דֵּי שָׂכָר טוֹב בַּכֹּל נָתַתָּה לִּי / וְשָׂכָר פְּרִי הַבֶּטֶן לֹא נָתַתָּה לִּי



הֲלֹא מַה יִּתְרוֹן לִי / בְּכָל עֲמָלִי / [אִם] בֶּן אָחִי אוֹ בֶן בֵּיתִי / [יוֹרֵשׁ] אוֹתִי



וְנַמְתָּה לּוֹ אַל תִּירָא לֹא מִזֶּה וְלֹא מִזֶּה / לֹא יִירָשָׁךְ לֹא זֶה וְלֹא זֶה



זָחַל וְעָרַג / עַל אֲשֶׁר הָרַג / ….ה…. לִדְרוֹשׁ / פֶּן דָּמָם תִּדְרוֹשׁ



חָרַד פֶּן יִהְיֶה צַדִּיק בֵּינֵיהֶם / אוֹ נֵר עָתִיד לַעֲמֹד מִבֵּינֵיהֶם

טַעַם הֲשֵׁיבוֹתוֹ אַל תְּהִי צַדִּיק הַרְבֵּה / רְשָׁעִים הִכְחַדְתָּה שְׂכָרָךְ הַרְבֵּה

יְרֵאָךְ כְּחָכָם יָרֵא וְסָר מֵרָע / וּבְכָל רָע אֲשֶׁר יָרָא טוֹב מִפִּיךָ בֻשָּׂר.




ו


אֶלֶף הַמָּגֵן לְאָב תָּלִיתָה / וְעַל מָגִנֵּי אֶרֶץ עִלִּיתָה

בְּמָגֵן אֲשֶׁר אוֹתוֹ בֵירָכְתָּה / בְּמָגֵן אַבְרָהָם נִתְבָּרָכְתָּה.





ג……. שֶׁמֶשׁ וּמָגֵן לוֹ הָיִיתָה / וּמָגֵן [יֶשְׁ]עָךְ לוֹ נָתָתָּה



דָּגוּל בְּעֵת אֵלָיו נִרְאֵיתָה / מָגִנֵּינוּ א…….אִיתָה.





הֶרְאֵיתָה לּוֹ שְׂכָרוֹ……. [חו] / כִּי [לֹא הֵ]לִין שְׂכַר שָׂכִיר אִתּוֹ



וּבִשַּׂרְתּוֹ ל……… מסגורתו / וּלְשַׁלֶּם לוֹ מַשְׂכֻּרְתּוֹ.





[ז]……ך לְפָנָיו תֵּלֵךְ / בְּצֶדֶק אֲשֶׁר לְפָנָיו יְהַלֵּךְ



חָ[ז צִדְ]קוֹ לְפָנָיו מְהַלֵּךְ / בְּצֶדֶק אֲשֶׁר לְפָנָיו יְהַלֵּךְ.





טָרוּחַ עֲלֵי יְרֻשָּׁתוֹ / פֶּן לְזָר [תִּ]הְיֶה מ[וֹרַ]שְׁתּוֹ



יִדַּעְתּוֹ מֵרֵאשִׁיתוֹ / כִּי בְאַחֲרִיתוֹ / בְּנוֹ יִירַשׁ אוֹתוֹ.





כְּבוֹדָךְ עָלָיו גִּלִּיתָה / וְכוֹ[בֶד אָ]זְנוֹ גָּלִיתָה



לַעֲנוֹת אֲשֶׁר חָג פִּיו הִמְגָּדְתָּה / [בְּטוֹ]בוֹת אֲשֶׁר בְּפִיךָ לוֹ הִגָּדְתָּה.





מֵרֹאשׁ צוּרִים צוּר חֲצַבְתּוֹ / וּמֵרֹאשׁ אֲמָנָה מַאֲמִין הִצַּבְתּוֹ



נְוֵה מְרוֹמִים חוּצָה הוֹצֵאתוֹ / וּ[צְבָ]א כּוֹכָבִים הִצְפֵּיתוֹ.





סַחְתָּה [לּוֹ כֹּ]ה יִהְיֶה זַרְעָךְ / שְׁלוֹשִׁים כְּעֶרְכָּךְ מִגִּזְעָךְ



עֲדֵי [עַד] נֵרָךְ [לֹ]א יִדְעָךְ / אֲשֶׁר הֵיאַרְ[תָּה בְּמַ]דָּעָךְ.





פְּלָאוֹת הִפְלֵאתָה לְמַיְמִינָךְ / וְחָ[שַׁבְתָּה צְדָ]קָה לְמַאֲמִינָךְ



צְבִי לַצַּדִּיק…… בִּנְיָנָךְ / אֲשֶׁר הִקְנֵיתָ קִנְיָנָךְ.





קָרָאתָה לְמַ[קְ]רִיא ת……. לָךְ / וְקֵרַבְתָּה ………ם לָךְ



רַחֲמִים מִשֶּׁלָךְ אֶתֵּן לָךְ / וְצֶדֶק בִּשְׁמָךְ אֶקְרָא לָךְ.





שְׁמוּעָה טוֹבָה לְאָב [שִׁמַּעְתָּה] / וּמִשְּׁמוּעָה רָעָה אוֹתוֹ שִׁמַּרְתָּה



תְּשׁוּרַת שִׁלֵּישִׁים מֶנּוּ שַׁעְתָּה / וּמִשָּׁ[לִישִׁ]ים אוֹתוֹ [הוֹשָׁ]עְתָּה.




ז 1


אַל תִּירָא / אָנֹכִי תּוֹחַלְתָּךְ / אַל תִּשְׁתָּע / אָנֹכִי שׁוֹמְרָךְ.



אַל תִּרְעַד / אָנֹכִי ר……ך / אַל תְּקַנֵּא / אָנֹכִי קוֹנָךְ.



אַל תֵּצַר / אָנֹכִי צוּרָךְ / אַל תִּפְחַד / אָנֹכִי פּוֹדָךְ.



אַל תַּעֲרֹץ / אָנֹכִי עִמָּךְ / אַל תִּסְמַר / אָנֹכִי [סוֹמְכָ]ךְ.



אַל תָּנ[וּד] / אָנֹכִי נוּחָךְ / אַל תִּמּוֹג / אָנֹכִי מָגִינָךְ.



אַל תִּלְאֶה / אָנֹכִי לִוּוּיָךְ / אַל תִּכְעַס / אָנֹכִי כּוֹסָךְ.



אַל תֵּחַת / אָנֹכִי יִשְׁעָךְ / אַל תְּטֹרָף / אָנֹכִי טוּרָךְ.



אַל תֶּחֱלֶה / אָנֹכִי חֵילָךְ / [אַל תִּ]זְחַל / אָנֹכִי זְרוֹעָךְ.



אַל תְּהוֹלָל / אָנֹכִי הִלּוּלָךְ / אַל תִּדְאַג / אָנֹכִי דִּגְלָךְ.



אַל תָּגָר / אָנֹכִי גּוֹאֲלָךְ / [אַל תְּבוֹהָל / אָנֹכִי בּוֹרְאָךְ.



אַל ת[אפ]ל / אָנֹכִי אוֹרָךְ.




ז 2


[שְׂכָרָ]ךְ בָּאֵשׁ אֲשֶׁר בָּאתָה / הַרְבֵּה מְאֹד בְּנֵי[רִים] אֲשֶׁר דִּבָּקְתָּה



שְׂכָרָךְ בהלי… אֲשֶׁר וִעַדְתָּה / [הַרְבֵּה מְאֹד] בִּזְכוּת אֲשֶׁר חִשָּׁבְתָּה



שְׂכָרָךְ ב[ט….] אֲשֶׁר יִדָּעְתָּה / הַרְבֵּה מְאֹד בִּזְכוּת אֲשֶׁר לִבָּבְתָּה



[שְׂכָרָךְ] בִּמְלָכִים אֲשֶׁר נָצָחְתָּה / הַרְבֵּה מְאֹד בְּשִׂיחַ אֲשֶׁר עָנִיתָה



שְׂכָרָךְ בְּפוֹשְׁעִים אֲשֶׁר צִדָּקְתָּה / [הַרְבֵּה מְאֹד] בְּקוֹשֶׁט אֲשֶׁר רִבִּיתָה

שְׂכָרָךְ בִּשְׁבוּעָה אֲשֶׁר שָׁמָעְתָּה / הַרְבֵּה מְאֹד בִּתְּנָיִים אֲשֶׁר תָּבָעְתָּה




ח


[שְׂכָרָךְ] בְּדִין / הַרְבֵּה מְאֹד בְּרַחֲמִים /
כִּי אֱלֹהִים דִּין / וַיְיָ רַחֲמִים / בּוֹ קִדַּשְׁתָּה שְׁמָךְ
[בְּדִין] / וּבוֹ נִתְקַדַּשְׁתָּה בְּרַחֲמִים / כִּקְדוּשַׁת
עֶלְיוֹנִים / ככ וקרא




ֿ
