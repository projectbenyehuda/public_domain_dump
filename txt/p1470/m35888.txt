


	
	אוֹתוֹתַי בַּמָּה אֵדַע
	



לג / שמות ג א – ומשה היה





א


אוֹתוֹתַי בַּמָּה אֵדַע / בָּחוּר בְּעֵת אָץ לֵדַע

גַּלְתָּה אֲדוֹן כָּל מַדָּע / דְּרָכֶיךָ אוֹתוֹ לְיַדָּע.





הַבִּמְלַאכְתּוֹ הָיָה מָהִיר / וּבְמַעֲלָלָיו זַךְ וְזָהִיר

זִמַּנְתּוֹ צֹאן לְהַנְהִיג / חֵילֵי צֹאן אָדָם הֱיוֹת מַנְהִיג.





טֶרֶם נֵס יִרְאֶה / יִדַּעְתּוֹ כִּי נִסֶּיךָ יִרְאֶה

כְּלוּלַי נַמְתָּה זֶה יִרְעֶה / לְמַעַן כִּי בֶאֱמוּנָה הָיָה רוֹעֶה.





יָקוּם דְּבָרָךְ / יַעֲמוֹד דִּבּוּרָךְ / לְמַגְּנֵינוּ כדברך.




ב


מִדַּדֶּה הָיָה אִישׁ הָאֱלֹהִים / נוֹהֵג וּבָא בְּנִסֵּי אֱלֹהִים

שְׂאֵתוֹ עֵינָיו בְּהַר אֱלֹהִים / עָלָיו נָחָה רוּחַ אֱלֹהִים.





פְּלִידַת אֵשׁ אוֹכְלָה / צָפָה וּסְנֶה לֹא אָכְלָה

קִימַת לַפִּידִים שָׁר וְנִמְלָא / רַךְ יוֹנְקֹתָיו לֹא נֶאֶמְלָה.





שְׁבָח פָּעֳלָךְ בְּקָרוֹב תִּרְאֶה / שַׁדַּי מֵרָחוֹק תִּרְאֶה

תָּקְפָּךְ עַד לֹא יֵרָאֶה / תְּחִלָּה מַלְאַךְ פָּנָיו נִרְאָה.





חֶסֶד וְרַחֲמִים / תֵּט מִמְּרוֹמִים / לְהַחֲיוֹת בְּטַל רְדוּמִים.




ג


יָהּ חֲסוֹף יָדָךְ / וֶאֱסוֹף צֹאן יָדָךְ / בְּמַטֵּה עֻזָּךְ / וּבְשֵׁבֶט מָעֻזָּךְ



נִדְמִינוּ לְצֹאן תּוֹעֶה מְפוֹרָד וּפוֹעֶה / בְּאֵין לָהּ מִרְעֶה וְגַם לֹא רוֹעֶה.

יִרְעוּ כְדָבְרָם……..




ד


בְּכָל צָרוֹתֵינוּ לָךְ צָרָה / וְהִבְטַחְתָּנוּ עִמּוֹ אָנֹכִי בְצָרָה

כִּי אַתָּה אֱלֹהֵינוּ וַאֲנַחְנוּ עִמָּךְ / כִּי אַתָּה רוֹעֵנוּ וַאֲנַחְנוּ צֹאנְךָ

בְּמִצְרַיִם בְּעֵת הֵצַר לָנוּ שׂוֹנֵא / נִגְלֵיתָ בְּמָקוֹם צַר מִתּוֹךְ הַסְּנֶה

פְּרוּכִים בִּלְבֵנִים לָנוּ עֵת רָאִיתָ / תַּחַת רַגְלֶיךָ לְבֵנָה הֶרְאֵיתָ



רָאוֹ רָאִיתָה / וְיָדוֹעַ יָדַעְתָּה / פָּקוֹד פָּקַדְתָּה / וְקָרוֹא קָרָאתָה



לְמֹשֶׁה מִמִּרְעֶה / מִתּוֹךְ מַרְאֶה / וְעֵת אֲשֶׁר רָאָה / אֲשֶׁר לֹא הֶרְאָה



כְּבוֹדָךְ הַנִּרְאָה / בְּאֵימָה וְיִרְאָה / הֶאֱמִירְךָ קָדוֹשׁ.




ה


אַחַר הַמִּדְבָּר צִיר נִהֵג צֹאן / אַחַר הַמִּדְבָּר הִנְהִיג עַמּוֹ כַּצֹּאן

בְּלֹא רַגְלַיִם רָץ וְהֵרִיץ מַרְעִיתוֹ / לָלֶכֶת לִמְקוֹם חֶזְיוֹן מַרְאִיתוֹ

גִּדּוּלֵי דֶשֶׁא הָיוּ נִבְרָאִים לְפָנָיו / וְאַחַר כָּךְ הָיוּ נִבְלָעִים מִלְּפָנָיו

דֶּרֶךְ גְּדוֹלָה לְיוֹם אֶחָד הִלֵּךְ / כִּי אוֹהֵב מֵישָׁרִים יִשַּׁר לוֹ הֵלֶךְ

הַר הָאֱלֹהִים עֵת כִּי הֻגָּע / מִטָּרְחוֹ הוּנַח וּמִיגִיעוֹ הֻרְגָּע



וּתְחִלָּה נִרְאָה לּוֹ כִּדְמוּת מַלְאָךְ / ל…….. דְּמוּתוֹ כִּדְמוּת מַלְאָךְ



זֶה לְפִי דַרְכּוֹ חִנְּכוֹ לִרְאוֹת / לִהְיוֹת בָּקִי בְּכָל מרא מַרְאוֹת

חֻזַּק לִבּוֹ כְּשָׁר לַבַּת אֵשׁ / בַּעֲבוּר לְלַבְּבוֹ בְּכָל מִינֵי אֵשׁ

טָהוֹר בְּתוֹךְ טֻמְאָה יְקָרוֹ הוֹפִיעַ / גָּבוֹהַּ עַל סְנֶה כְּבוֹדוֹ הוֹדִיעַ

יַעַן כִּי צָרַת עַמּוֹ הִיא צָרָתוֹ / וִישׁוּעָתָם הִיא יְשׁוּעָתוֹ.




ו


אֲשֶׁר נִתְכַּנָּה יְקוּתִיאֵל / כִּי בָנִים הִקְוָה לָאֵל

בְּסִבְלוֹת אֶחָיו לִרְאוֹת כְּהוֹאֵל / נוֹדַע כִּי הוּא הַגּוֹאֵל.





גִּדּוּל מַרְבִּית בַּת פַּרְעֹה / וּמֶנּוּ וּבוֹ קָם וּפְרָעֹה

דָּגוּל אֲשֶׁר………… / דִּגְלוֹ אַהֲבָה…….





ה……………….. / ……..צֹאן אֵל אֱמוּנָה……



ו……………… / ……………..





ז………שר בְּמִפְעָלָיו / וְטוֹב מעשרו כּוֹנֵן עָלָיו



חִשּׁוּב הִרְהוּר גִּלָּה לוֹ וְנִגְלָה אֵלָיו.





טָעוּן אֶת משּׂאוי אַחִים / אֲשֶׁר הָיוּ כְּשׁוֹשָׁן בֵּין הַחוֹחִים

יָגַע לֵידַע עַד כַּמָּה הֵם נֶאֱנָחִים / וְיִדַּעְתּוֹ כִּי הֵם נִנּוֹחִים.





כּוֹלֵל הַקֵּץ וְהוּא קָרוֹב / וּפָרַט חֶשְׁבּוֹן וְהוּא רָחוֹק

לִבּוֹ הִשְׂכִּיל כִּי הוּא קָרוֹב / כְּקָרַב וַיַּאַת מֵרָחוֹק.





מִנֶּפֶשׁ בְּהֶמְתּוֹ הוֹדִיעַ / וּמָרַת נַפְשׁוֹ לִבּוֹ יוֹדֵעַ

נוֹדַע לוֹ מְתֵי שָׁוְא יוֹדֵעַ / קֵץ יָדוּעַ תֵּדַע לְהוֹדִיעַ.





סָמוּךְ וּבָטוּחַ בִּשְׁבוּעָה / אֲשֶׁר הֻבְטְחָה לָאָב בִּשְׁמוּעָה

עוֹדֶנּוּ מִשְׁתָּאֶה ברן יְשׁוּעָה / צָלְחָה וְצָמְחָה בָּהּ בְּשָׁעָה.





פָּקוּד מִיַּלְדוּתוֹ אִישׁ גָּדוֹל / לְהַפְקִיד עַל גּוֹי גָּדוֹל

צָעַד לִרְאוֹת הַמַּרְאֶה הַגָּדוֹל / מִיכָאֵל הַשַּׂר הַגָּדוֹל.





קָראוּי וּמְכֻנֶּה שֵׁמוֹת עֲשָׂרָה / בְּחָכְמָה אֲשֶׁר תָּעוֹז מֵעֲשָׂרָה

רִיצוּי זֶה שֵׁם אֲשֶׁר הֻקְרָא / חֻבַּב וּבוֹ פַעֲמַיִם נִקְרָא.





שָׁפָל וְעָנָיו וְנִבְחַר בּוֹ / וּבְהַשְׁפָּלָה וּבַעֲנָוָה תְּחִלָּה דֻבַּר בּוֹ

תַּקִּיף אֲשֶׁר מְלֹא כָל הָאָרֶץ כְּבוֹדוֹ / וְעַל הַסְּנֶה לָחַץ כְּבוֹדוֹ.





ובכן וירא מלאך יי אליו (בלבת אש)





אֵשׁ אֲשֶׁר הִיא אוֹכְלָה אֵשׁ אֵשׁ בּוֹעֶרֶת בִּיבֵשִׁים וּבְלַחִים

אֵשׁ גּוֹחֶלֶת בְּשֶׁלֶג וְקִיטוֹר אֵשׁ דּוֹמָה לַאֲרִי רוֹבֵץ

אֵשׁ הַמַּרְאָה כְמוֹ מַרְאוֹת אֵשׁ וַדַּי כִּי לֹא כָבָה

אֵשׁ זוֹרַחַת וּמְשׁוֹטֶטֶת אֵשׁ חוֹשֶׁשֶׁת וּמִתְלַקַּחַת

אֵשׁ טָסָה בְּרוּחַ סְעָרָה אֵשׁ יוֹקֶדֶת בְּאֶפֶס עֵצִים

אֵשׁ כִּי כָל יוֹם מִתְחַדֶּשֶׁת אֵשׁ לֹא נֻפְּחָה מֵאֵשׁ

אֵשׁ מַעֲלָה לוּלָבִין אֵשׁ נִצּוֹצֶיהָ הֵם הַבְּרָקִים

ח


[…….?] מִתְחַלֶּפֶת בְּרֹב דִּמְיוֹנוֹת / עַל הַסְּנֶה



בִּלְהָבִים / וְעַל סִינַי בְּלַפִּידִים לְמַעְלָה שְׁבִיבִים תְּלַהֵט /



וּלְמַטָּה רְשָׁפִים תְּלַחֵךְ / וּמֶנָּה גְּבִילַת הַשְּׂרָפִים / לָכֵן



בָּהּ לֹא נִשְׂרָפִים / וּמִזֵּעָה אֲשֶׁר הֵם מַזִּיעִים / נְהַר



דִּינוּר מוֹשֵׁךְ אוֹר / וּגְדוֹתָיו גּוּשֵׁי שֶׁלֶג / הָאֵשׁ לֹא



תַרְתִּיחַ שֶׁלֶג / וּמִשֶּׁלֶג לֹא יצין אֵשׁ / כִּי יוֹצֵר הָאֵשׁ



וּבוֹרֵא הַשֶּׁלֶג / מַתְוִיךְ שָׁלוֹם בֵּין אֵשׁ וָשֶׁלֶג / שׁוֹפֵט



בָּאֵשׁ וְדָן בַּשֶּׁלֶג.

וּמִי יִסָּתֵר מֵחַמָּתוֹ / וּמִי יַעֲמֹד בְּקָרָתוֹ /



כִּי אִם יִשְׂרָאֵל עַמּוֹ / אֲשֶׁר בָּאוּ בַמַּיִם וּבָאֵשׁ / וְהֵם



מְשׁוּלִים בַּמַּיִם וּבָאֵשׁ / לָכֵן לֹא יִשָּׁטְפוּ בַּמַּיִם וְלֹא



יִכָּווּ בָאֵשׁ / כִּי כִּבְלוּלֵי מַיִם וָאֵשׁ / יִפְתְּחוּ פִּיהֶם



בִּקְדֻשָּׁה / לְהַעֲרִיץ בַּקֹּדֶשׁ.




ט


בְּתוֹם וְצֶדֶק וְאוֹמֶן יַקְדִּישׁוּךָ צִבְאוֹת קֹדֶשׁ תָּמִיד יַעֲנוּ וְיֹאמְרוּ:



קדוש קדוש קדוש / יי צבאות מלא כל הארץ כבודו.



קָדוֹשׁ מֵהוֹלְכֵי תָּמִים / קָדוֹשׁ מִפּוֹעֲלֵי צֶדֶק / קָדוֹשׁ מִדּוֹבְרֵי אֱמֶת

קָדוֹשׁ לְמֹשֶׁה הוֹדִיעַ יֹשֶׁר וְהִלּוּךְ תָּמִים

קָדוֹשׁ חִכְּמוֹ עַנְוַת הָעֵקֶב וּפְעֻלַּת צֶדֶק

קָדוֹשׁ לִמְּדוֹ שְׁלִיחוּת הַקּוֹשֶׁט וְדִבְרַת אֱמֶת.





מִמְּקוֹמוֹ בְּתוֹם מֻפְרָכִים צָפָה וְדִלֵּג עַל קֵץ והוא במקומו

מִמְּקוֹמוֹ רָאָה צִדְקַת עַמְרָם וְקִפֵּץ חֶשְׁבּוֹן והוא במקומו

מִמְּקוֹמוֹ יָדַע אֱמוּנַת מֹשֶׁה וְנִגְלָה וְצָמַח והוא במקומו.
