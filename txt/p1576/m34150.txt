


	
	אָמַר אוֹיֵב לְקַדָּחוֹת
	



מאמרי הרופאים. שער שלישי: שער תפלות






אָמַר אוֹיֵב: לְקַדָּחוֹת דֶּבְרִיּוֹת / הָרוֹפֵא
יְדַבֵּר נִפְלָאוֹת לְרַמּוֹת בָּהֶם אֶת הַבְּרִיּוֹת, / יַעֲבִיר קוֹל
בְּכָל הָאָרֶץ גַּם בָּאִיִּים / ״מִי הָאִישׁ הֶחָפֵץ חַיִּים? / גְּשׁוּ
נָא אֵלַי – וְאֶתְּנָה לָכֶם הַנְהָגָה יְשָׁרָה / וְהִיא קֳדָמַי
מֵאַשְׁמְדַאי מַלְכָּא דְשֵׁדֵי עֲרוּכָה וּשְׁמוּרָה. / שִׁמְעוּ אֵלַי,
עַמִּי, / לְכוּ שִׁבְרוּ וְאִכְלוּ בַבֹּקֶר הַשְׁכֵּם וְלַחֲמוּ
בְלַחֲמִי, / עִרְכוּ מַאֲכָל מְלָכִים גְּדוֹלִים וְרָמִים – / אֶת
הַבְּצָלִים וְאֶת הַשּׁוּמִים / מְעֹרָבִים בְּקַשׁ וּגְבָבָא / לֶאֱכֹל
לְשָׂבְעָה בְּפַת חֲרֵבָה. / גַּם בַּצָּהֳרַיִם וּבָעֶרֶב תֹּאכְלוּ
בְשַׂר שְׁקָצִים / וּרְמָסִים, אֶפְרֹחִים אוֹ בֵיצִים, / וְכֵן
תִּשְׁכְּבוּ הַנָּשִׁים בְּכָרֵס מְלֵאָה – / זוֹ הִיא סְגֻלָּה נֶעֱלֶמֶת
וּרְפוּאָה! / זֹאת עֲשׂוּ וִחְיוּ מִדֵּי יוֹם בְּיוֹם, וַייָ עִמָּכֶם /
וְלֹא יִתֵּן הַמַּשְׁחִית לָבוֹא אֶל בָּתֵיכֶם. / וּמִי שֶׁלֹּא יַאֲמִין
וְיַעֲשֶׂה הַפְכּוֹ – / אִוֶּלֶת אָדָם תְּסַלֵּף דַּרְכּוֹ! / וְאִם
אוּלַי יָצָא הַדָּבָר כְּרֶגַע / כִּי פָשָׁה הַנֶּגַע – / נִשְׁלְחָה
אֲנָשִׁים אֶל הַגִּלְעָד וּנְבַקֵּשׁ שָׁם מְעַט צֳרִי וָלוֹט / אוֹ יַיִן
וְשֵׁכָר בִּמְעָרַת שְׁתֵּי בְנוֹת לוֹט: / כִּי מִגֶּפֶן סְדוֹם גַּפְנָם
וּמִשַּׁדְמוֹת עֲמֹרָה, / וּרְפוּאָה לָבוֹא קְרוֹבָה – לֹא יָדַע אֱנוֹשׁ
עֶרְכָּהּ וְלֹא יָדַע מְחִירָהּ. / וְאִם בְּאוּלַי גָּרַם מַזָּלוֹ – /



אַל תִּבְכּוּ לְמֵת וְאַל תָּנוּדוּ לוֹ!״.
אָמַר אוֹיֵב: לְמַחֲלַת הַכָּבֵד אוֹ לַלֵּב אוֹ לָרֵאָה
/ יֹאמַר הָרוֹפֵא ״אֲנִי בָקִי בִרְפוּאָתָם וְלִבִּי רָאָה הַרְבֵּה
חָכְמוֹת וָדַעַת – / בְּכָל הָאָרֶץ זֹאת מוּדַעַת״. / וּבֶאֱמֶת הוּא
כַחֲמוֹר נוֹשֵׂא מַרְדַּעַת!

כִּי יִפְתַּח פִּיו בְּדִבְרֵי תִפְלוּת, / בִּמְקוֹם
חָכְמָה – הוֹלֵלוּת וְשִׂכְלוּת, / וְכֹה יְדַבֵּר ״לְרָעַת הַכָּבֵד /
בְּחָכְמָתִי אֶכָּבֵד, / וְלִכְאֵב הַלֵּב / לֹא כָמוֹנִי בִלְתִּי אִם
כֶּלֶב, / וּלְכָל תַּחֲלוּאֵי הָרֵאָה / מִי יָחוּשׁ חוּץ מִמֶּנִּי
לָדַעַת תְּרוּפָה נִפְלָאָה? / וּבְחָנוּנִי נָא בָּזֹאת מַה תֶּחֱזוּ בִי
/ כִּי שֶׁבַע חָכְמוֹת בְּלִבִּי״ – / וּבֶאֱמֶת כִּי שֶׁבַע תּוֹעֵבוֹת
בְּלִבּוֹ / וְכָל רוּחַ אֵין בְּקִרְבּוֹ! / הַאַף אֻמְנָם כָּל עוֹד
אֲשֶׁר יַגְדִּיל לְדַבֵּר – / וְהַחוֹלִים בָּאוּ עַד מַשְׁבֵּר, / כִּי
מְצָאוּהוּ נָעוּר וָרֵיק מִכָּל חָכְמָה, / וְלֹא יַעֲזֹב מַעֲשָׂיו, אִם
יְבִיאוּהוּ לִכְלִמָּה.



אָמַר אוֹיֵב: לְכוּ, חֲזוּ מִפְעָלוֹת הָרוֹפֵא
וְתַחְבּוּלוֹתָיו / נוֹשֵׂא אֲלֻמּוֹתָיו, / פּוֹעֵר פִּיו לִבְלִי חֹק
גּוֹזֵר בְּאִמְרוֹתָיו, / כִּי הוּא יֵדַע חֳלִי וּמִקְרֶה וְסִבּוֹתָיו.
/ לְהָכִין עֵזֶר מָזוֹר וּמַרְפֵּא / יַגְדִּיל שָׂפָה וְיַרְחִיב פֶּה, /
יֵלֵךְ אֶל הַבַּשָּׂם בְּרוֹמָה בְּבוֹא עִתּוֹ, / יָרִים יָדוֹ לִכְתֹּב
לְהַרְאוֹת טַבַּעְתּוֹ, / יְסַדֵּר בְּלָשׁוֹן זָר כִּי נָכְרִיָּה
חָכְמָתוֹ / לְהָסִיר אֶת מֻרְאָתוֹ בְּנוֹצָתוֹ, / וּבֵין מִלָּה וּמִלָּה
זוֹרֵק מִפִּיו אֶת רֻקּוֹ – / כָּכָה מִשְׁפָּטוֹ וְחֻקּוֹ, / וּבְדָבָר
שֶׁהוּא קַל / יַחְמִיר בַּמִּדָּה וּבַמִּשְׁקָל. / אָז יֹאמַר אֶל
הַבַּשָּׂם ״קַח אֶת הַמַּשְׁקֶה וְאֶל הַחוֹלֶה תְּבִיאֶנּוּ, /
בְּעֶרְכְּךָ נְפָשׁוֹת תַּעֲרִיכֶנּוּ, / וּשְׁאַל אֶת מְחִירוֹ /
בְּטֶרֶם יִדְעַךְ נֵרוֹ. / וְאִם צָרָה וְיָגוֹן יִמְצָא, אֲנַחְנוּ בְנֵי
חֹרִין – / כָּל הַמְקַלְקְלִין פְּטוּרִין!״ / וּשְׁנֵיהֶם שְׂמֵחִים
בַּחֲלֻקָּה / כַּאֲשֶׁר אֲנָשִׁים אֲשֶׁר עָשׂוּ צְדָקָה. / וּמַה לָּהֶם
אִם בְּמוֹתוֹ יִתְקְעוּ בַשּׁוֹפָר / מֵאַחַר אֲשֶׁר יְכַסּוּהוּ בֶעָפָר?
/ ״וּמַה לָּנוּ – / נְקִיִּים אֲנַחְנוּ!״.



אָמַר אוֹיֵב: שִׁמְעוּ נָא כָּל הָעַמִּים / דִּבְרֵי
הָרוֹפֵא הַסָּכָל וְהוּא יִשְׁכֹּן מְרוֹמִים / אֲשֶׁר יֹאמַר ״בֶּן
מַלְכֵי קֶדֶם אֲנִי, בֶּן חֲכָמִים״. / מִיּוֹם בְּרֹא אֱלֹהִים אָדָם /
מִי הִגִּיעַ בַּחֲצִי יְדִיעָתוֹ בְּטִבְעֵי הָעֲשָׂבִים לָבוֹא בְסוֹדָם?
/ גַּם בַּשָּׁמַיִם הַנִּפְרָדִים / לְכַמָּה מַעֲלוֹת עוֹלִים
וְיוֹרְדִים – / כָּל אֶחָד בִּשְׁמוֹ / הוּא יָדַע מְקוֹמוֹ /
וּבְמַעֲשֵׂה הַהַקָּזוֹת / הֲלֹא יֹאמַר ״שִׂכְלִי פָעַל כָל זֹאת! / כִּי
הֲבִינוֹתִי לְהַבְחִין בֵּין שָׁעוֹת וּרְגָעִים / הַטּוֹבִים
וְהָרָעִים״. / וְהוּא לֹא יָדַע מוֹלַד הַיָּרֵחַ וּמְלוֹאוֹ, / כִּי לֹא
יָדַע מְבוֹאוֹ. / וּבְיָמִים שֶׁל סִימָן / יִתֵּן קוֹלוֹ ״גַּם אֲנִי
יָדַעְתִּי, וְיוֹם שֶׁנִּתְלָה בוֹ הָמָן / שֶׁיֶּשׁ בּוֹ סַכָּנָה דָּם
לְהוֹצִיא / אִם לֹא בְּבִרְכַּת יַיִן וְהַמּוֹצִיא״.

אָמַר אוֹיֵב: כַּמָּה הִרְשִׁיעַ הָרוֹפֵא לְדַבֵּר
פְּתַיּוּת וּבַל יָדַע מָה / לְהַעֲלוֹת עַצְמוֹ בְּרֹב חָכְמָה, / אוֹי
לְאוֹתָהּ בּוּשָׁה אוֹי לְאוֹתָהּ כְּלִמָּה! / גַּם כִּי יִשְׁאָלֵהוּ
אִישׁ מַהוּ גֶדֶר הָרְפוּאָה, / אָז יַעֲנֶה עַזּוֹת ״הֲמִמֶּנִּי דַעַת
פְּלִיאָה? / וּמַה זּוֹ שְׁאֵלָה? שְׁנַיִם הֵם וְאַתָּה תֶחֱזֶה: /
גָּדֵר מִזֶּה וְגָדֵר מִזֶּה״. / וְאִם יְבַקְשׁוּ מִמֶּנּוּ עַל
הַקַּדַּחַת / יָשִׁיב כִּי הִיא בְאֶפֶס עֵצִים מִתְלַקַּחַת. /
וּבְעוֹנוֹת חֲלוּקוֹת / שׁוֹקְדוֹת וּדְבֵקוֹת / יִגְזֹר ״הַמִּשְׁפָּט
לֵאלֹהִים, הוּא נֶעְלָם – / זֶה לֹא יָדַע אֱנוֹשׁ מֵעוֹלָם!״ /
וּבְהַתְחָלַת הֶחֳלִי וּבַעֲלִיָּתוֹ / וְהָעֶמְדָּה וּבִירִידָתוֹ – /
מִי זֶה בָא עַד תְּכוּנָתוֹ?״ / וְעַל יְמֵי הַגְּבוּל וְהֶקֵּפוֹ / וּמַה
יִּהְיֶה בְסוֹפוֹ – / כָּל דְּבָרָיו דִּבְרֵי תֹהוּ / וְהֶבֶל הֵבִיא גַם
הוּא!

אָמַר אוֹיֵב: שִׁגְגַת הָרוֹפֵא עוֹלָה זָדוֹן / וְרִיב
וּמָדוֹן, / מִן הַמֵּת אוֹמֵר שֶׁהוּא חַי וּמִן הַחַי מֵת, / אֵין
בְּפִיהוּ נְכוֹנָה וְלֹא דְבַר אֱמֶת, / אֲשֶׁר מִלֵּא כַפּוֹ קוֹצֵר /
כְּעַל סַלְסִלּוֹת בּוֹצֵר / הוּא מְשָׁרֵת אֶל מַלְאַךְ הַמָּוֶת
וּשְׁלוּחוֹ, / זֶה כֹחוֹ. / יְקַבֵּץ סִפְרֵי רְפוּאוֹת חֲדָשִׁים גַּם
יְשָׁנִים / מִכַּמָּה מִינִים, / מִשּׁוּם מַרְאִית עַיִן יַחֲזִיקֵם
בְּבֵיתוֹ / וְלֹא עַל קְרִיאָתוֹ. / וְאִם יֶשׁ בָּם שְׁגִיאוֹת – לֹא
יְבִינֵם, / יַעַמְדוּ כַּאֲשֶׁר הֵמָּה וְאֵין מִי שֶׁיְּתַקְּנֵם. /
וְאִם יִשְׁאָלוּהוּ פֶּרֶק אוֹ מַאֲמָר נִבְחָר – / יִדְחֶה אוֹתָם עַד
לְמָחָר / וְהָיָה לְעֵת מְצֹא יֹאמַר אֲלֵיהֶם / שֶׁיָּשׁוּבוּ
לְאָהֳלֵיהֶם / ״וְכִי לְנַסּוֹתִי אַתֶּם בָּאִים? / הַאִישׁ כָּמוֹנִי
יַעֲנֶה לִפְתָאִים? / וַאֲנִי הָרוֹפֵא לְכָל תַּחֲלוּאִים!״
אָמַר אוֹיֵב: זֶה הַכְּלָל בִּשְׁחִיטַת רוֹפְאֵי



הַזְּמָן – / שָׂכָר הַרְבֵּה לִטֹּל, / וְיִקָּרֵא רוֹפֵא-אֻמָּן. /
וְלַמְלַמְּדִים – לָהֶם אַזְהָרָתָם תְּחִלָּה: / כָּל הַמִּתְעַסֵּק
בִּרְפוּאָה וְאֵינוֹ נוֹטֵל שָׂכָר – חַיָּב סְקִילָה! / לִמְדוּ קִיחָה
קִיחָה מִשְּׂדֵה עֶפְרוֹן / וּבָזֶה יִהְיֶה לָכֶם תּוֹעֶלֶת וְיִתְרוֹן,
/ מִשְׁכוּ וּקְחוּ לָכֶם גַּם אַתֶּם / וּשְׁחַטְתֶּם בָּזֶה
וַאֲכַלְתֶּם. / הֱיוּ נְכוֹנִים לְבַקֵּר הַחוֹלִים / וְלֹא בִדְבָרִים
בְּטֵלִים / לִהְיוֹת עֶזְרָם וּמָגִנָּם – / בִּמְחִיר רַב וְלֹא
בְחִנָּם! / וְזֶה יִהְיֶה לָכֶם הָאוֹת: / חִזְקִיָּה גָּנַז סֵפֶר
רְפוּאוֹת / שֶׁלֹּא הָיוּ נוֹתְנִים לָרוֹפֵא שְׂכַר עֲמָלוֹ / וְדוֹחִים
אוֹתוֹ לֵאמֹר ״עַד כִּי יָבוֹא שִׁילֹה / שׁוּב לְבֵיתְךָ לְשָׁלוֹם /
וְלִשְׁאֹל שְׂכָרְךָ אַל תִּקְרַב הֲלֹם!״ / לָכֵן הִזָּהֲרוּ
בִפְעֻלַּתְכֶם / וּבֵית הַחוֹלֶה אַל תָּבוֹאוּ עַד אֲשֶׁר תָּרִימוּ
תְרוּמַתְכֶם, / וּבְיָדַיִם רֵיקָנִיּוֹת אַל תֵּצְאוּ מִן הַבַּיִת /
בִּמְעַט כַּבֵּיצָה אוֹ כַזַּיִת!
