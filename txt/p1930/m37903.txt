


	
	שמע נא מלך
	







מְשַׁל הָעוֹלָם הַזֶּה






שְׁמַע נָא! מֶלֶךְ אֶחָד הָיָה לוֹ פַרְדֵּס רִמּוֹנִים
וְעַרְמוֹנִים / וְהוֹשִׁיב שָׁם פּוֹעֲלִים שׁוֹנִים / לְהָרִים מִכְשׁוֹל וּלְהָסִיר קִמְשׁוֹנִים / וְהִבְטִיחַ מַשְׂכֻּרְתָּם מוֹנִים / הַיְשָׁרִים עַל דִּגְלֵיהֶם חוֹנִים / וּגְדָרִים בּוֹנִים / הַגֶּפֶן סְמָדָר / וְלָהֶם עֹז וְהָדָר / עַל מִשְׁמַרְתָּם עוֹמְדִים / וּבַעֲבוֹדָתָם חוֹמְדִים / בַּאדֹנֵיהֶם לֹא מָרוּ / וּמַיִם קָרוּ וּמִשָּׁם לֹא סָרוּ / עַד כָּל עֵץ נָשָׂא עָנָף / וְתַחְתֵּיהֶם שָׁכְנוּ כָּל כָּנָף / אֲלֵיהֶם נָסִים / וְתַחְתֵּיהֶם חֹסִים / שְׂמֵחִים וְשָׂשִׂים / הַדּוּדָאִים נָתְנוּ רֵיחַ / וְכָל עֵץ הִפְרִיחַ /
סְעַפּוֹתָם אָרְכוּ / וּלְפִרְיָם בֵּרְכוּ / וְהֵידָד כְּדוֹרְכִים צָרְחוּ / פּוֹעֲלִים כָּזֹאת עַל אֲנָשִׁים יָשׂרוּ / וְכָל בַּעֲלֵי הַשִּׁיר עֲלֵיהֶם יְשׁוֹרֵרוּ / וַאֲשֶׁר שָׁכְנוּ בְצֵל דָּלִיּוֹת / יִשְׁכְּנוּ בִּמְרוֹמֵי עֲלִיּוֹת.



אֲבָל פּוֹעֲלִים אֲשֶׁר בִּמְלַאכְתָּם עֲצֵלִים / בִּמְלֶאכֶת חֻקָּם מוֹעֲלִים / וּבַשִּׁלּוּם שׁוֹאֲלִים / כָּל אֶחָד בַּכּוּרָה בַּתְּאֵנָה חָטָף / רֹאשׁ יֹנְקוֹתָיו קָטָף / הֵן פּוֹעֵל כָּזֹאת כְּלִמָּה עָטָף / אוֹ כִי הַכֶּרֶם לֹא עִזֵּק / וַיִּפְקְדָהּ בְבֶזֶק / וְלֹא שְׁמָרָה מִנֵּזֶק / תּוֹכוֹ נָחָר / בַּטּוֹב לֹא בָחָר /
הַגֶּפֶן מִתְרוֹשֵׁשׁ / וְהוּא עֵצִים קוֹשֵׁשׁ / וְעַל כִּי בְּהִשְׁתַּדְּלוּת הַמְּלָאכָה בוֹשֵׁשׁ / עֲרוּגֹת צֶמַח עָבָשׁוּ / וְשָׁרָשֵׁיהֶם יָבָשׁוּ / וְשָׁם שָׁכַן כֹּל הוֹלֵךְ עַל גָּחוֹן / בֵּין הַפּוֹעֲלִים קוֹל עֲנוֹת מְהוּמָה / וּבְיָדָם אֵין כָּל מְאוּמָה / וְקַח עַל מַיִם רַבִּים צַפְצָפָה שָׂמוֹ / וּפוֹעֵל כָּזֹאת בְּרֹאשׁוֹ דָּמוֹ.
