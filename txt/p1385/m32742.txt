


	
	סיפור אליגורי על הנפש והגוף
	








נְאֻם לְמוּאֵל בֶּן-אִיתִיאֵל / רַב פְּעָלִים
מִקִּבְצְאֵל / אֲשֶׁר סִפֵּר מְשָׁלִים וְחִבֵּר / לָדַעַת לָעוּת אֶת
יָעֵף דָּבָר. וַיְהִי בְּעֵת צֵאת מַלְכֵי הַיְדִידוּת / וְהִתְעוֹרֵר
שָׂרֵי הַשַּׁחֲרוּת וְהַיַּלְדוּת / וְיָצָאתִי אֶל רְחוֹב מַרְכְּבוֹתָם
/ וּלְחוּצוֹת מְרוּצָתָם / וְהִנֵּה מַרְכְּבוֹת הַחֵשֶׁק בָּאִים /
יְרוּצוּן כִּצְבָאִים / חֲיָלֵי חֲיָלִים / דִּגְלֵי דְגָלִים /
יִתְהוֹלְלוּן וְיִִשְׁתַּקְשְׁקוּן / יִתְדַּפְּקוּן וְלֹא יִתְאַפְּקוּן
/ וְהִנֵּה אַחֲרֵיהֶם רִכְבֵי הָאַהֲבָה / מֶרְכָּבָה אַחֲרֵי מֶרְכָּבָה
/ וַעֲלֵיהֶם יְדִיד נַפְשִׁי שַׂר הַצָּבָא / דָּגוּל מֵרְבָבָה. /
בִּרְאוֹתָם אוֹתִי / וַיְרוּצוּן לִקְרָאתִי / וַיְחַבְּקוּנִי
וַיְנַשְּׁקוּנִי / וּבְכָל זֹאת נַפְשִׁי יְשֵׁנָה. וַתַּחֲלֹם וְהִנֵּה
הַשֶֶּׁמֶשׁ מְחַבֵּק חֵיק הַלְּבָנָה / וַתֹּאמֶר: אֵין זֹאת כִּי אִם
יְדִידִי יְבוֹאֵנִי / וְלֹא יִשְׁקֹט עַד אֲשְֶׁר יִרְאֵנִי.





וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

וְיָשַׁנְתִּי וְלִבִּי עֵר / וְהִנֵּה קוֹל צְבִי דוֹפֵק

צְבִיָּה, דַּי לְהֵרָדֵם, / לְכִי עַל דּוֹד לְהִתְרַפֵּק

וְכִשְׁמוֹעַ לְבָבִי זָע / וְלֹא יָכוֹל לְהִתְאַפֵּק.





וַיִּתְגָּעַשׁ וַיִּרְעַשׁ לִבִּי בְּחֶדְרוֹ /
בְּשָׁמְעוֹ מִסְפַּר הַחֲלוֹם וְאֶת שִׁבְרוֹ / וַיְמַהֵר
וַיַּחְבִּיאֵנִי בְּחַדְרֵי מוֹרָשָׁיו / לְבַל יִרְאוּנִי מַרְכְּבוֹת
הַצְּבִי וּפָרָשָׁיו. / וַיָּקָם דּוֹדִי וְכָל רִכְבֵי אוֹהֲבָיו / וְכָל
גְּדוּד הַחֵשֶׁק אֲשֶׁר סְבִיבוֹתָיו / וַיִּתְּנוּ כִסְאוֹתָם סְבִיבוֹת
לִבִּי / וַתִּשְׁמַע נַפְשִׁי וַתֹּאמַר: אַשְׁקֵף בְּעַד אֶשְׁנַבִּי /
וְאוּלַי אֶרְאֶה הַצְּבִי מְאוֹר שִׁמְשִׁ[י] / וּתְחִי נַפְשִׁי,
וַתִּדּוֹם וַתִּלְבַּשׁ בִּגְדֵי חָפְשִׁי / וַתִּפְתַּח גִּנְזִי
וַתִּיטַב נַפְשִׁי בִּבְשָׂמֶיהָ / וּמֹר וַאֲהָלוֹת וְקִנְמוֹנוֹת
וּקְצִיעוֹת כָּל בִּגְדוֹתֶיהָ / וַתֵּלֶךְ הִיא וְנַעֲרוֹתֶיהָ /
וְשׁוּלֵי עֲדָנֶיהָ / וּגְדִילֵי תַעֲנוּגֶיהָ אֲרוּכִים. וַתָּרָץ
יַעֲלַת חֵן / וַתַּעַל אֶל הָעֲלִיָּה / וַתַּשְׁקֵף אָנֶה וָאָנָה /
סְבִיבוֹתֵינָה / הַצְּבִי אֲהוּבָהּ, וַיָּבֹא לְעֵינֶיהָ / מֵאוֹר
לְחָיָיו פִּתְאוֹם / בָּרָק וַיֵּהוֹם.





וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

הֲבָרָק זֶה יְרוּצוּן אוֹ / פְּנֵי דוֹד זָהֳרָם זָרַח

הֲנָטַע בִּי יְדִיד חִשְׁקוֹ / בְּיוֹם נָסַע בְּיוֹם פָּרַח

יְרוּצוּן בִּי יְדִידוֹתָיו / כְּרוּץ גִּבּוֹר עֲלֵי אֹרַח

וְכָל דּוֹד יִשְׁכְּנָה לִבִּי / כְּגֵר תּוֹשָׁב – וְזֶה אֶזְרָח

וְזוּלָתוֹ כְּאוֹרֵחַ / כְּהֵלֶךְ יַהֲלֹךְ אוֹרֵחַ

וְזֶה יֵלֵךְ וְזֶה יָשׁוּב / וְזֶה יָבֹא וְזֶה יִבְרַח

וְזֶה יִכְבַּד וְזֶה יֵקַל / וְזֶה מַשָּׂא וְזֶה טֹרַח

וְרִפְאוּת פִּי צְבִי אֶל צִיר / וּמִלָּיו עַל שְׁחִין יִמְרָח.





וַיְהִי אַחֲרֵי כֵן וַתִּשְׁלַח יָדָהּ / וַתֶּחֱזַק&gt;
בִּידִידָהּ / וַיְהִי הוּא בָא כְּגוֹעֵר / וְחָצְנוֹ יְנָעֵר / וַיֹּאמֶר&gt;
אֵלֶיהָ: סוּרִי מִמֶּנִּי וְגוּרִי / בַּאֲשֶׁר תָּגוּרִי / הֲלֹא אַתְּ&gt;
יְדִידָתִי / לָמָּה לֹא יָצָאת לִקְרָאתִי / הֲלֹא יָדַעַתְּ / אִם לֹא&gt;
שָׁמַעְתְּ / כִּי בָאתִי לַחֲזוֹתֵךְ / לְהִשְׁתַּעֲשֵׁעַ בְּהַדְרָתֵךְ /
וּבְתִפְאֶרֶת יְפִי יִפְעָתֵךְ. / וְרָכַבְתִּי עַל עָבוֹת קַלּוֹת /
וְהָלַכְתִּי אֳרָחוֹת עֲקַלְקַלּוֹת / וַתֵּחָבְאִי וְלֹא רָצִית
לְהִתְגַלּוֹת / וַתֵּצֶר לִי לְהִתְחַלּוֹת / וּבְבוֹאִי לִרְאוֹת
פָּנַיִךְ / וַתָּסֵבִּי מִנֶּגְדִּי עֵינַיִךְ / וְעַתָּה אָשׁוּבָה לִי
אֲנִי וּמַרְכְּבוֹת חִשְׁקִי / כִּי לֹא הִשְׁלַמְתִּי חֻקִּי /
וַתְּחָרְפִי מַרְכְּבוֹת יְדִידוֹתַי / וּגְדוּדֵי הַחֵשֶׁק אֲשֶׁר
סְבִיבוֹתַי / אַךְ תְּנִי לִי נְשִׁיקָ[ה] אַחַת לְצֵידָה / בַּעֲבוּר
תִּהְיֶה לִי לְעֵדָה. / וַיָּשֹׁב אֶל עֵינוֹ וַיֹּאמַר: עֵינִי, עֵינִי,
קְחִי גַם [אַתְּ] מִן הַיְדִידָה צֵידָתֵךְ / לְהַחֲיוֹת אֶת נַפְשֵׁךְ
וּלְכַלְכֵּל אֶת שֵׂיבָתֵךְ / וְהַרְבִּי פְּנוֹת אֶל פָּנֶיהָ / כִּי
אֵין לָךְ מִן הָעָפְרָה כִּי אִם רְאוּת עֵינֶיהָ.&gt;





וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

קְחִי עֵינִי רְאוּת עָפְרָה לְצֵידָה / וְזֶה חֶלְקֵךְ לְבַד מִן הַיְדִידָה

וְגַם אַתְּ, פִּי, יְעַד לִבֵּךְ לְנָשְׁקָהּ / תְּהַלֶּלְךָ בְּכָל יוֹם הַיְעוּדָה

וְזֹאת תּוֹרַת יְדִידֵי הַתְּבוּנָה / וּבֵינֵיהֶם הֲלֹא זֹאת הַתְּעוּדָה

וּמַה מָּרָה פְּרִידַת יַעֲלַת חֵן / וּמַה קָּשֶׁה סְפוֹד עַל הַפְּרִידָה.





וַתָּקָם נַפְשִׁי וַתִּשְׁתַּחוּ לְמוּלוֹ / וַתֵּבְךְ
וַתִּתְחַנֵּן לוֹ / וַתֹּאמֶר: הֲלֹא יָדַעְתָּ כִּי לִבִּי לִי
וּבְדַבְּרִי עִם אָדָם יִתְאַנֶּה / וְעֹז פָּנָיו יְשֻׁנֶּה /
וּבִרְאוֹתוֹ אוֹתִי לְקָחַנִי / וְלַחֲדָרָיו הֵבִיאַנִי / וּבְמוֹרָשָׁיו
הֶחְבִּיאָנִי / וְאָמְנָם אֵין עַתָּה רוֹאָנִי. / וַיָּשָׁב הַצְּבִי
לְדַרְכּוֹ סַר וְזָעֵף / וּרְפֵה יָדַיִם וְיָעֵף. / וַיְהִי
בְּהַפְנוֹתוֹ שִׁכְמוֹ לָלֶכֶת / וְדִמְעַת עָפְרָה עַל לְחָיָיו נִתֶּכֶת
/ וַתָּשֵׁב אֶל לִבָּהּ וַתֹּאמַר: מַה לִּי וְלִבְכּוֹת / וַיִּרְבּוּ
בֵינֵיהֶם דִּבְרֵי רִיבֹת.





וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

הָאַהֲבִים סִפְּרוּ לִי / אֵיךְ דּוֹד יְחִי אֵין יְדִידוֹ

לִבִּי בְּיַד הַצְּבִי שָׂם / גַּם הַצְּבִי הוּא פְקִידוֹ

אָדָם בְּלִי לֵב הֲיִחְיֶה / וּצְבִי נְשָׂאוֹ בְיָדוֹ

נוֹעַ יְנוּעוֹ מְשַׂחֵק / יִלְעַג וְיִשְׂחַק לְאֵידוֹ

אִם עֵין צְבִי תֶּחֱזֵהוּ / חִצָּהּ יְפַלַּח כְּבֵדוֹ

יָלוּן בְּבוֹאוֹ לְחִשְׁקוֹ / אִם נָד יְשַׂחֵק לְנֹדוֹ

נִבְהַל לְבוֹאוֹ וְיִמַּס / כָּעָשׁ לְנוֹדוֹ חֲמוּדוֹ

עַל כֵּן יְמוֹתָיו דְּאָגוֹת / אִם בָּא וְאִם שָׁב בְּעוֹדוֹ.





וַיְהִי אַחַר הַדְּבָרִים הָאֵלֶּה וַתִּתְחַלְחַל
נַפְשִׁי מֵחִשְׁקָהּ / וַתָּרֶב אֶל לִבָּהּ בְּחָזְקָה / וְכַאֲשֶׁר
הֻגַּד לִי, כִּי גָדַל חִילָהּ / נִבְהַלְתִּי וְנֶחְפַּזְתִּי לִרְאוֹת
מַה-לָּהּ. / בָּאתִי עָדֶיהָ וְהִנָּהּ סָרָה וְזוֹעֶפֶת / וְאֶל דּוֹדָהּ
נִכְסֶפֶת / וּבִמְרִירוּת פֵּרוּד מִתְעַטֶּפֶת / וְרוּחַ מִמּוּל אַרְצוֹ
שׁוֹאֶפֶת. / שָׁאַלְתִּי לִשְׁלוֹמָהּ וְלֹא הֵשִׁיבָה / וְלֹא
הִקְשִׁיבָה / וָאֶקֹד וָאֶשְׁתַּחֲוֶה / וְכָל זֶה אֵינֶנּוּ שֹׁוֶה. /
וַתֹּאמַרְנָה לִי נַעֲרוֹתֶיהָ / דַּע לְךָ, כִּי הִיא חוֹלַת אֲהָבִים /
וַתְּדַבֵּר בְּךָ דְּבָרִים לֹא טוֹבִים. / וּכְשָׁמְעִי, כִּי דִבְּרָה
בִּי / וַיָּמָת לִבִּי בְּקִרְבִּי / וַיִּחַר אַפִּי וָאֹמַר: הֲלֹא זֶה
קְלוֹנִי / לִטְמוֹן בְּחֻבִּי עֲווֹנִי / וַיִּמָּלֵךְ לִבִּי עָלַי /&gt;
וַיֶחֱרַד חֲרָדָה בְּשָׁמְעוֹ מִלַּי / וַיֹּאמֶר: מְהֵרָה הִכָּנַע /&gt;
מֵהֲלֹךְ אֵלֶי[הָ] אַל נָא תִּמָּנַע / עַל כֵּן לֹא תַּבִּיט אֵלֶיךָ /
וְלֹא תִּרְאֶךָ. / עַל כִּי מָנַעְתָּ מִדּוֹדָהּ / שׁוּב אֶל
גְּבִרְתְּךָ וְהִתְעַנִּי תַּחַת יָדָהּ. / וַיְהִי טֶרֶם כִּלְּתָה
מֵעֲצָתוֹ / וַיִּיקַץ יַעֲקֹב מִשְׁנָתוֹ / וְאָמַר: אַט אֲנִי נֹגֵעַ
בְּבַת עֵינִי / הֲלֹא שֶׁקֶר בִּימִינִי.





וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

דּוֹד הֲיִקְצֹף עַל רַעֲיָתוֹ וְנִשְׁבַּע / חֵי יְדִידוֹת, לוּלֵי לְבָבִי הֲרַגְתִּיךְ

כָּזְבָה לָהּ נַפְשִׁי וְאֵינִי כְּחוֹשֵׁק / הֵן תְּמוֹל בַּעְלַת אֱמוּנִים יְדַעְתִּיךְ



זֹאת תְּעוּדַת דּוֹדִים: צְבִיָּה תְּחָרֵף – / יַעֲנֶה דּוֹדָהּ: הֵן בְּנַפְשִׁי פְּדִיתִיךְ.





וָאָקוּם בְּחָפְזָה, וּבִדְאָגָה וּבִרְעָדָה
הָכְנַעְתִּי / וְלֹא גָּבַהְתִּי / וְחַשְׁתִּי וְלֹא הִתְמַהְמַהְתִּי /
וָאָבֹא אֵלֶיהָ בַּלָּט וָאֶקְרָא: נַפְשִׁי, נַפְשִׁי, יוֹנָתִי תַמָּתִי
/ וָאֹמַר לָהּ וּלְנַעֲרוֹתֶיהָ: מַדּוּעַ קְצַפְתֶּן עַל עַבְדְּכֶן /
וַתְּדַבֵּרְנָה דְּבָרִים [אֲשֶׁר] לֹא כֵן / הֲכָזֶה יִתָּכֵן /
אָמְנָם כָּל הַיּוֹם יֶחֱטָא אִישׁ לְאִישׁ וּבֶן מַחְפִּיר וּמֵבִישׁ /
וִיהִי כְּמַחֲרִישׁ. / וַתַּעַן נַפְשִׁי וַתְּדַבֵּר אִתִּי קָשׁוֹת /
וַתֹּאמֶר: הֲלַאֲנָשִׁים תְּדַמֶּה הַנְּפָשׁוֹת? / וְאִם יֶחֱטָא אָדָם
לְאָדָם / הֲלֹא בֶּעָפָר יְסוֹדָם / וְאֵיךְ יֵעָרֵךְ בֶּן מֶלֶךְ / לְבֶן
מֶלֶך? / הֲלֹא גִּזְרָתִי גִּזְרַת סַפִּיר / וּמַחְצְבִי יַקִּיר /
וְחֶזְוֵיהּ דְּחִיל נְרֵוֵיהּ שַׁפִּיר.





וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמֶר:

גִּזְעִי בְּרוּם נִתַּן בְּתוֹךְ אוֹר הַיְּקָר / שָׁם הוּא לְמִקֶּדֶם לְיוֹם הַמַּעֲשֶׂה

אֵיךְ יוֹצְרִי שָׁלַח קְצָת גִּזְעִי לְגוּף / אָדָם לְהָסִיר מֵאֱנוֹשׁ רֹעַ מַעֲשֶׂה

כִּי אֵיךְ יְדַמֶּה גוּף לְגוּשׁ עָפָר? לְמִי / גִזְעוֹ כְּזֶה וּבְצֵל אֲדוֹנָיו יֶחֱסֶה.





וַיְהִי כִּשְׁמֹעַ לְבָבִי וְחָרָה אַפִּי / וְעָלָה
קִצְפִּי / וְחָזַק זַעְפִּי / וַיִּסְפֹּק אֶל כַּפִּי. / וַיַּעַן
בַּסְּעָרָה וַיֹּאמַר: אֵיךְ יִגְבְּהוּ שֹׁכְנֵי בָתֵי-חֹמֶר? וַיֹּאמֶר
לָהּ: גַּם אֲנִי אוֹדֵךְ / כִּי רַב הוֹדֵךְ / וְגָדוֹל כְּבוֹדֵךְ /
וִיסוֹד טָהוֹר יְסוֹדֵךְ / וְהֵן אִם יְסוֹדֵךְ נַעֲלָה / וּמַחְצְבֵךְ
מִמֶּנִּי נִפְלָא / וְגִזְרָתֵךְ בָּרָה / וְאַתְּ צְפִירַת תִּפְאָרָה /
הֲלֹא יָדַעַתְּ / כִּי כָל הוֹדֵךְ וְתִפְאַרְתֵּךְ / גָּנוּב הוּא
אִתֵּךְ. / הֲלֹא אִם גִּזְרָתֵךְ מִגִּזְרַת טוֹב נִגְזֶרֶת / וְהוּא
עָטַר לָךְ עֲטֶרֶת תִּפְאֶרֶת / עָלַיִךְ יֵאוֹר אוֹרוֹ / וְיִגַּהּ
נָגְהוֹ / כִּי גָבַֹהּ מֵעַל גָּבֹהַּ שֹׁמֵר / וְאִם יְסוֹדֵךְ וִיסוֹד
יְסוֹדֵךְ גָּבְהוּ שְׁנֵיהֶם / וּגְבֹהִים עֲלֵיהֶם / אַף כִּי כָל יְמֵי
מְגוּרֵךְ אֲנִי הֲדָרֵךְ / וּבְמַצָּבִי אַתְּ נִצֶּבֶת / וּבְקִרְבִּי
אַתְּ יוֹשֶׁבֶת / אַתְּ נֹשֵׂאת אוֹתִי – וַאֲנִי מַשָּׂאֵךְ / וְאַתְּ
לִי לֵאלֹהִים – וַאֲנִי נְבִיאֵךְ / אַגִּיד זִיוֵךְ וְיָשְׁרֵךְ /
וְנָגְהֵךְ וְאוֹרֵךְ / וְהוֹדֵךְ וַהֲדָרֵךְ / כְּבוֹד עָשְׁרֵךְ. /
וְאוּלָם לֹא חֲנוֹתֵךְ בְּגוּף אֲדָמָה / וְאָז תְּהִי נַעֲלָמָה /
וְעַתָּה הִנְנִי בֵּית מְנוּחֵךְ / וְאַתְּ לִי לֵאלֹהִים וַאֲנִי
מְשִׁיחֵךְ. / וְלָמָּה בְּזִיתַנִי / וּמַדּוּעַ הֱקִלַּתַנִי? / אָז
גָּעַרְתִּי בְלִבִּי וַאֲחַרְפֶנּוּ / בְּשָׁמְעִי אוֹתוֹ מְדַבֵּר
גְדוֹלוֹת וְנִפְלָאוֹת מִמֶּנּוּ / וָאֹמַר: מַה לְּךָ לְדַבֵּר
בְּמַעֲלוֹת נַעֲלוֹת / אַל תְּבַקֵּשׁ לְךָ גְּדוֹלוֹת / כֵּן
הַנְּפָשׁוֹת דּוֹבְרוֹת / הֲלֹא הֵנָּה בְּגִזְרַת בּוֹרְאָן חֹבְרוֹת /
אַךְ הַרְחֵק הִרְחִיקוּ מִבּוֹרְאָן נְוֵיהֶן / וְהַגַּיְא בֵּינֵיהֶן /
וְאַתָּה מַה לְּךָ לְדַבֵּר / וּמִלִּים לְחַבֵּר / הֲלֹא אַתָּה גּוּף
נָגוּף וּפֶגֶר מוּבָס / לְמִשְׁלָח וּלְמִרְמָס / וְאֵיךְ יִתְחָרֶה
בַנֶּפֶשׁ הַטִּיט וְהָרֶפֶשׁ.





וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

יְסוֹד אָדָם כְּמוֹ אֶפֶס / וּמַחְצָבוֹ הֲלֹא רַק רַע

וּמֵאַרְבַּע יְסוֹדוֹת הוּא / וְכָל מוֹסִיף הֲלֹא יִגְרַע.





וַיְהִי כִּשְׁמֹעַ נַפְשִׁי / בְּאָמְרִי לְלִבִּי
נִקְלֶה וְנִבְזֶה / אָז רָפְתָה רוּחִי מֵעָלַי בְּדַבְּרִי אֶת כָּל זֶה
/ וַתַּעֲנֵנִי נַפְשִׁי: אַל יֵרַע בְּעֵינֶיךָ זֹאת / כִּי עָנִיתִי לְךָ
עַזּוֹת / עַל כֵּן דִבַּרְתִּי דִּבְרֵי תְּבוּנוֹת / וְדִבְרֵי חֲכָמִים
כַּדָּרְבֹנוֹת / אֲבָל דִּבַּרְתִּי מִקֹּצֶר רוּחִי / וּמֵרֹב כַּעְסִי
וְשִׂיחִי / אַף אָמְנָם חֲמָסִי עָלֶיךָ / עַל כֵּן אָרִיב אֵלֶיךָ / עַל
כִּי כְלִיתַנִי בְּבָתֵי כְלָאִים / וְלֹא הִרְאִיתַנִי דּוֹדִי בֵּין
הַצְּבָאִים. / וַאֲנִי שָׁמַעְתִּי עָלָיו, כִּי בְיַעֲלַת חֵן זוּלָתִי
דּוֹבֵק / וְחֵיק אַיֶּלֶת אֲהָבִים אַחֶרֶת חוֹבֵק / וַאֲנִי שָׁמַעְתִּי
עָלֶיהָ לְהַשְׁמָעוּת אָזְנַיִם / כִּי תַאֲוָה הִיא לָעֵינַיִם /
מִי-יִתֵּן וְיַגִּיד לִי שְׁמָהּ וּדְמוּתָהּ / וּמִשְׁפָּטָהּ
וְצוּרָתָהּ. / וַיַעֲנֵנִי לִי לֵאמֹר: הֲלֹא הַחָכְמָה רַעְיָתוֹ /
וְעָלֶיהָ תְּשׁוּקָתוֹ / וְעַתָּה מַפִּיל אֲנִי תְּחִנָּתִי עַל
פָּנַיִךְ / וְתִיקַר-נָא נַפְשִׁי בְּעֵינַיִךְ / וְנֵלְכָה נָא יַחַד
וְנֵצֵא / וְנִשְׁאַל לִנְתִיבָהּ, אָנָה תִּמָּצֵא / וְשָׁם נִרְוֶה
דוֹדֵי הַחָכְמָה / וְנִתְעַלְּסָה בְּאַהֲבֵי עָרְמָה / וְדַדֵּי
שִׂכְלָהּ נְעַשֶּׂה / וּשְׂעַר דּוֹד נִתְכַּסֶּה / וְנִצְלַח וְנַעֲשֶׂה
/ וְנִרְעֶה בַגַּנִּים / וְנִלְקֹט שׁוֹשַׁנִּים / וּמֶשֶׁךְ חָכְמָה
מִפְּנִינִים / וְנֵרֵד לְגַן שַׁעֲשׁוּעִים / אוּלַי יִקָּרֶה
לִקְרָאתֵנוּ הַדּוֹד הַנָּעִים. / וָאֶצְעַק צְעָקָה גְּדֹלָה וּמָרָה /
וָאֹמַר: מֵאַיִן לִי גְבוּרָה / כִּי אֵיךְ דְּעִי וּרְאִי כִּי אַתְּ
מִתְאַנָּה / וְאֵיךְ תֹּאמְרִי נֵרֵד לַגַּן אוֹ נַעֲלֶה / וְאָנֹכִי
אִישׁ-רָשׁ וְנִקְלֶה.





וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

הֲמִסָּכָל יְבֻקָּשׁ בִּין / וְעָרְמָה לַפְּתָאִים מָה

הֲמֵרָעֵב שְׁאוֹל לֶחֶם / וּמַיִם מֵאֱנוֹשׁ יִצְמָא

מְבַקֵּשׁ בִּין בְּלֵב סָכָל / הֲלֹא יִשָּׂא עֲווֹן אַשְׁמָה

וּמַה בִּינָה לְפֶתִי כִּי / עֲלֵי כֶסֶל עֲשׂוֹת פִּימָה

הֲבַשַּׁחַק יְבֻקַּשׁ גּוּשׁ / וְאִם לִשְׁאֹל שְׁאוֹל רֵיקָה

וּמַה לַּגּוּף וְלַשֵּׂכֶל / לְנֶפֶשׁ יָאֲתָה חָכְמָה

וְלוֹ גַן יַעֲשֶׂה עַצְלוּת / וּפִרְיוֹ נוּם וְתַרְדֵּמָה.





וַתַּעַן נַפְשִׁי וַתֹּאמַר: גַּם אוֹדְךָ הוֹדֶךָ /
דַּבֵּר, כִּי חָפַצְתִּי צִדְקֶךָ / וַתָּקָם בִּצְוָחָה / גַּם הִיא
נֶאֱנָחָה / וַתֹּאמַר: וָלָמָּה אֲנִי מְצַעֶקֶת וְהֹמִיָּה / וְעַל מָה
אֲנִי בוֹכִיָּה? / הֲלֹא עַל הֱיוֹתְךָ עָצֵל בִּימֵי הַמַּעֲדָנוֹת /
וְלֹא עָשִׂיתָ לְךָ פַּרְדֵּסִים וְגַנּוֹת / הֲלֹא יָדַעְתָּ, כִּי
קְרָעִים תַּלְבִּישׁ נוּמָה / וְעַצְלָה תַּפִּיל תַּרְדֵּמָה / כִּי
זְכֹר בְּאָמְרִי לְךָ קְנֵה חָכְמָה / וּשְׁכֹן עָרְמָה / וַתַּעֲנֵנִי
דִבְרֵי תְנוּאוֹת / מְלֵאוֹת תְּשׁוּאוֹת. / וַתֹּאמַר: הִנֵּה הַחָכְמָה,
מִכָּל יְצִיר נִפְלָאָה / וְאֵיךְ אֲבַקְשֶׁנָּה וְהִיא לֹא נִמְצָאָה
/בְּאָזְנֵינוּ שָׁמַעְנוּ שִׁמְעָהּ / וְאִישׁ לֹא יְדָעָהּ / וְהַשֵּׂכֶל
הֲלֹא רָם יִשְׁכֹּן וְהוּא מְרוֹמִים יִשְׁכֹּן / עַד כִּי נִשְׁאַרְתָּ
מִן הַחָכְמָה עָרוֹם מִבְּלִי בֶגֶד / וַתֵּלֵךְ וַתֵּשֵׁב לָהּ מִנֶּגֶד
/ וְאָמְנָם כִּי שֶׁקֶר דִּבַּרְתָּ / וּמִלֵּי כְזָבִים חִבַּרְתָּ /
וּבְהַבְלֵי סִכְלוּת הִתְאֹנַנְתָּ / וּבְצֵל פְּתַיּוּת הִתְלוֹנַנְתָּ /
וַתִּבְחַר לָךְ דֶּרֶךְ מְשׁוּבוֹת / אָמַר עָצֵל: שַׁחַל בַּדֶּרֶךְ
אֲרִי בֵּין הָרְחֹבוֹת / כִּי רוֹדֵף שֵׂכֶל יַשִּׂיגֵהוּ וְלֹא יִרְאֶה /
אֶפֶס קָצֵהוּ תִרְאֶה / וְהֵן אִם נִתַּן הַשֵּׂכֶל בַּמְּרוֹמִים /
בְּתַחְבּוּלוֹת יוֹרִידוּהוּ מִשָּׁם חֲכָמִים / וְאַף כִּי הַנֶּפֶשׁ
אִתּוֹ תֵּשֵׁב וְלֹא תִהְיֶה / וּבְצִלּוֹ תִחְיֶה / וְתַחַת כְּנָפָיו
תֶּחְסֶה / לְהוֹדִיעַ לְאָדָם מַה יַּעֲשֶׂה / וְאִם תָּשִׁיב לָהּ
רוּחֶךָ / הִיא תְּשִׂיחֶךָ.

וַיְהִי כְּשָׁמְעִי דְבָרֶיהָ / וַתִּקְצַר רוּחִי, כִּי
נֶהֶפְכוּ עָלֶיהָ צִירֶיהָ / וְנֶחֶרַפְתִּי וְנֶאֶלַמְתִּי / וּבֹשְׁתִּי
וְגַם נִכְלַמְתִּי / וְיָדִי לְמוֹ פִי שַׂמְתִּי / וַיִּפְּלוּ פָנַי
וְגַם חֲמַרְמָרוּ / עַד כִּי עָלַי רַחֲמֶיהָ נִכְמָרוּ.

וַתֹּאמַר: לֹא עַתָּה יֵבוֹשׁ יַעֲקֹב וְלֹא עַתָּה
פָנָיו יֶחְוָרוּ. אִם אַתָּה הָיִיתָ עָצֵל וַתֵּרָדֵם / אַל יִפֹּל
עָלָיו לֵב אָדָם / הִנֵּה רָאִיתִי גַן חָכְמָה נָטוּעַ / וּבְזֵרְעוֹנֵי
בִינָה זָרוּעַ / וּמַעְיָן וּמַרְוֵהוּ יַשְׁקֵהוּ וּמַבּוּעַ / קוּם
וְנָשִׁיתָ לִבֵּנוּ לִמְסִלוֹת הֲלִיכוֹתָיו / וְיֹרֵנוּ מִדְּרָכָיו /
וְנִדְרֹשׁ וְנַחְקֹר לִנְתִיבוֹתָיו / וְנֵלְכָה בְאֹרְחוֹתָיו.
בְּשָׁמְעִי כָּל זֹאת אָמַרְתִּי: וּמִי הוּא זֶה אֲשֶׁר
נָגַע אֱלֹהִים בְּלִבּוֹ / וַתַּעֲנֵנִי: אִישׁ אֲשֶׁר רוּחַ בּוֹ; הֲלֹא
רָאִיתָ מֶרְכְּבוֹת הָאַהֲבָה / אֲשֶׁר עָלֶיהָ שַׂר הַצָּבָא / דָּגוּל
מֵרְבָבָה / הוּא הַנּוֹטֵעַ וְזוֹרֵעַ / הַחָכְמָה קְרָאתְהוּ רֵעַ /
וַאֲנִי כָל הַיּוֹם מִשְׁתּוֹמֵם / עַל כִּי יִשְׂטְמֵנִי / וְעַל
כְּבוֹדוֹ כִּי גָלָה מִמֶּנִּי / וְעַתָּה לָמָּה נִתְרָאֶה / לְכוּ
וְנֵלְכָה עַד הָרֹאֶה / הֲלֹא [בַ]אֲמָרָיו כְּאֵבִים יֶחְבָּשׁ /
וּמְתוּקִים מִדְּבַשׁ / וּשְׁמוֹ לֹא אֲכַנֶּה מִפְּנֵי אוֹיֵב מְקַנֵּא /
וְאָמְנָם אָלוּן עַל נוּדוֹ אֲשֶׁר הִכְאִיבַנִי / אַךְ עַתָּה
הֶלְאָנִי.

וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

נָע יְדִיד מֶנִּי וְעֵינִי / יֹרְדָה מַּיִם לְנוּעוֹ



בָּם יְשֻׁקֶּה גַן אֲהָבִים / גַּן [כְּ]נַפְשִׁי דוֹד נְטָעוֹ



בּוֹ הֲלֹא לִבּוֹ יְהַלֵּךְ, / בּוֹ הֲלֹא רִפֵּד יְצוּעוֹ

יִזְכְּרָה יָמִים קְדוּמִים / יִזְכְּרָה דוֹד שַׁעֲשׁוּעוֹ

אֶפְרְעָה לִבִּי וְאֶקְרַע / הֵן נְדוֹד דּוֹדִי קְרָעוֹ.





וָאֹמַר: אֲהָהּ נַפְשִׁי, מַה יּוֹעִילָךְ / בְּבֹא
חֲבָלִים לָךְ / וּמָה נֵחַנְתְּ / עַד מָתַי לְהִנָּחֵם מֵאַנְתְּ /
וַתַּעַן וַתֹּאמַר: הִנֵּה לְלַעַג תְּשִׂימֵנִי / הֲלֹא יָדַעְתִּי כִּי
הֶבֶל תְּנַחֲמֵנִי. / וָאֶתְחַנֵּן אֵלֶיהָ וָאֹמַר: הִנֵּה נָא
נוֹאַלְתִּי / וּלְדַבֵּר לְפָנַיִךְ הוֹאַלְתִּי / הָסִירִי מִמֵּךְ
כַּעַס וְזַעַם / וַאֲדַבְּרָה אַךְ הַפַּעַם / הַגִּידָה לִי יְסוֹדֵךְ /
וּמֵאַיִן חָכְמָתֵךְ וְהוֹדֵךְ. וַתֹּאמַר: אֲנִי לְךָ יְסוֹד
הַנְּפָשׁוֹת וּמַחְצַבְתָּן / וּנְצוּרוֹת לֹא יְדַעְתָּן.





וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

אֱלֹהַי נַעֲלָה עַל כֹּל / וְהוּא נִפְלָא לְכָל נִפְלָא

וְהוּא רֵאשִׁית לְכָל רֵאשִׁית / בְּאֵין רֵאשִׁית וְלֹא תִכְלָה

וּבָרָא כֹל בְּחָכְמָתוֹ / וְהוּא נוֹשְׂאָם בְּרֹב חֶמְלָה

וְשֵׂכֶל רֹאשׁ פְּעֻלּוֹתָיו / וְנֶפֶשׁ אַחֲרָיו עוֹלָה

שְׁלֹשֶׁת הַנְּפָשׁוֹת אַךְ / [וְ]חָכְמָה נִתְּנָה מַעְלָה

וְתַחְתֶּיהָ מְקוֹר חַיִּים / וְנֶפֶשׁ צָמְחָה אֶצְלָהּ

וּמִגִּזְרַת שְׁלָשְׁתָּם שֶׁ/שְׁלֹשׁ אֶרֶץ לְהַגְדִּילָהּ

לְהַחְכִּימָהּ וּלְהַחְיוֹתָהּ / לְהַצְמִיחָהּ….. לָהּ



[וְ]שָׁמַיִם וְכוֹכָבִים / בְּרָאָם סוֹבְבִים סֶלָה



בְּהָסֵב הַשְּׁחָקִים נוֹ/לְדוּ אַרְבַּע אָבוֹת תְּחִלָּה



[וְ]אַרְבַּעְתָּם יְסוֹד כָּל חַי / וְכָל אֶבֶן וְכָל אֵלָה



וְלֹא כָל חַי אֲבָל כָּל מֵת / הֲלֹא זֶה אֲשֶׁר הוּא יִכְלָה

אֲדָמָה שָׁת בְּתוֹךְ שַׁחַק / וְאֶרֶץ עַל בְּלִי תָלָה

וְכָל אֵלֶּה בְּרָאָם אֵל / בְּמַאְמָר עַד אֲשֶׁר כִּלָּה

וְדַע כִּי נֹאחֲזוּ כֻלָּם / וְזֶה מִזֶּה הֲכִי נִתְלָה

וּמַאְמָר יִהְיֶה צָעִיר / וְאוֹר מֵאוֹר עֲדֵי מַעְלָה

וְיִגְבַּהּ אוֹר עֲלֵי מָאוֹר / וּמִי יִגַּע לְאוֹר מַעְלָה

וְלֹא יִכְלוּ וְלֹא יִבְלוּ / וְלֹא יִרְאוּ לְיוֹם נִקְלָה

אֲבָל יִבְלוּ בְנֵי אַרְבַּע / יְסוֹדוֹת, הֵם בְּנֵי מַחְלָה



וּמַחְצָב זֶה בְּהֶפֶךְ זֶה / [וּ]מַה זָּעַם [וּ]מַה כָּלָה



וּמִפְעָלָם לְזֹאת יִבְלֶה / וּמַעֲשֵׂיהֶם כְּלֵי גוֹלָה

וְהַמָּשָׁל בְּסִיר אַרְבַּע / לְבַל תִּהְיֶה מְבֻשָּׁלָה?





נוֹצְרוּ אֵלֶּה הָעוֹלָמִים כֻּלָּם / וְהוּכְנוּ עַל
מְכוֹנוֹ כָּל עוֹלָם וְעוֹלָם / וּמַעֲרָכָה לִקְרַאת מַעֲרָכָה /
וַתִּשְׁלַם כָּל הַמְּלָאכָה.





מְשַׁל שֵׂכֶל כְּבָר נִשְׁלַם / בְּעִנְיַן אַהֲבָה נַעְשָׂה



חֲשֹׁב טוֹבָה, [וְ]לִדְבַר מוּם / יְדִיד אֶל נַפְשְׁךָ תִּשָּׂא.
