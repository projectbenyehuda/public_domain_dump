


	
	[אוֹיָה לִי כִּי־גַרְתִּי מֶשֶׁךְ אָהֳלֵי קֵדָר וְגַעְתָּם (סליחה)]
	




עַל־מְחַטְּטֵי שִׁכְבַי (סְלִיחָה)



אוֹיָה לִי כִּי־גַרְתִּי מֶשֶׁךְ אָהֳלֵי קֵדָר וְגַעְתָּם



בֵּין אֲרָיוֹת וּפָרִיצֵי חַיּוֹת הַחוֹרְקִים מְתַלְּעוֹתָם



גֵּאִים הַקֹּפְאִים עַל-שִׁמְרֵיהֶם בְּחַיֵּיהֶם וּבְמוֹתָם



הַשֹּׁכְבִים עַל-מִטּוֹת שֵׁן וּסְרֻחִים עַל- עַרְשׂתָם.



דָּחֹה הֻדַּחְתִּי וְאֵין דּוֹרֵשׁ בְּיַד עַמֵּי הָאֲרָצוֹת



הָיִיתִי חֶרְפַּת מָשָׁל בַּגּוֹיִם שִׁמֲמוֹת נֶחֱרָצוֹת



וַתִּשְׁתַּפֵּךְ לָאָרֶץ מְרֵרָתִי בְּהִשְׁתַּפֵּךְ לְקָלוֹן וּשְׁמָצוֹת



אַבְנֵי-קֹדֶשׁ בְּרֹאשׁ כָּל-חוּצוֹת.



זְאֵב עֲרָבוֹת שְׁדָדַנִי וְעוֹרֵר עָלַי חֲנִיתוֹ



חִצָּיו שְׁנוּנִים וְקַשְּׁתֹתָיו דְּרֻכוֹת בְּאַפּוֹ וַחֲמָתוֹ



טִנֵּף נְבֵלָתִי בְּסָחֹב וְהַשְׁלֵךְ אַךְ פִּגְרֵי נְבֵלָתוֹ



כֻּלֻּם שָׁכְבוּ בְכָבוֹד אִישׁ בְּבֵיתוֹ.



יָדַעְנוּ יְיָ כִּי-צֶדֶק מִשְׁפָּטֶיךָ עַל-אֲמִתָּם



כַּלֵּה בְחֵמָה צוֹרֲרֵי עַמְּךָ וְיִתַּמּוּ בְחַטּאֹתָם



לְלַעַג וָקֶלֶס שָׁלְחוּ יַד לִסְחֹב בְּחוּצוֹתָם



הַנֶּאֱהָבִים וְהַנְּעִימִם בְּחַיֵּיהֶם וּבְמוֹתָם.



מָתַי תֶּחֱשׂף זְרוֹעֲךָ לְבַקֵּשׁ שְׁאֵרִית הַנִּדָּחָה



נִדְחֵי עַמְּךָ אֲשֶׁר הִדַּחְתָּם בְּחַטֹּאתָם וְאֵין מְנוּחָה



סָפוּ תַּמּוּ בְּיַד אוֹיְבֵיהֶם נֶחְשְׁבוּ כְּצֹאן טִבְחָה



וַתְּהִי נִבְלָתָם כָּסּוּחָה.



עַצְמוֹת מֶלֶךְ-אֱדוֹם שׂרְפוּ הֲלֹא פָקַדְתָּ לְנָקְמוֹ



פְּקֹד עַצְמוֹת עֲבָדֶיךָ הַמּוּצָאִים מִקִּבְרוֹתָם יוֹם בְּיוֹמוֹ



צוֹרֲרִים הַצָּרִים הָשֵׁב פָּעֳלָם אֶל-חֵיקָם לְשַׁלְּמוֹ



לָכֵן פָּקַדְתָּ וַתַּשְׁמִידֵם וַתְּאַבֵּד כָּל-זֵכֶר לָמוֹ.



קוּמָה יְיָ בְּאַפֶּךָ הִנָּשֵׂא בְּעַבְרוֹת צוֹרֲרַי וְגַאֲוָתָם



רֶגֶשׁ צְבָאָם תַּכְרִית וְתִמְחֶה מִדּוֹר אַחֵר שְׁמוֹתָם



שֵׁם וּשְׁאֵרִית תִּתֵּן לְעַמְּךָ וְעוֹד בַּחַיִּים חַיָּתָם



יָבוֹא שָׁלוֹם יָנוּחוּ עַל-מִשְׁכְּבוֹתָם.



תֵּן יְיָ מַה-תִּתֵּן אַחֲרִית וְקָלוֹן וְשִׁמָּמוֹן



בְּשֶׁטֶף מַכָּה לִשְׁטֹף צָרִים מֵתֵי עַמְּךָ יַעֲרִימוּן



נַפְשׁוֹת חֲסִידֶיךָ עַל-רֹב שָׁלוֹם יִנְעָמוּן



יִחְיוּ מֵתֶיךָ נְבֵלָתִי יְקוּמוּן.



יִסַּח יְיָ בֵּית גֵּאִים מִתְקוֹמֲמִים הַמַּשְׁחִיתִים



מִקָּטֹן וְעַד-גָּדוֹל טַף וְנָשִׁים כָּל-עִיר מְתִים



נְבַרֵךְ שְׁמוֹ הַגּוֹמֵל טוֹבוֹת וְשׁוֹפֵט אֲמִתִּים



אֲשֶׁר לֹא-עָזַב חַסְדּוֹ אֶת-הַחַיִּים וְאֶת-הַמֵּתִים.



חֲסָדֶיךָ הָרַבִּים זְכֹר וְלֹא כְמַעֲשֵׂה הַשּׁוֹאֵל



זוּלָתְךָ יַעֲשֶׂה לִמְחַכֵּה-לוֹ כִּי אֵין גּוֹאֵל



קַנּוֹא וְנוֹקֵם לְשִׁמְךָ תֵּן כָּבוֹד וְהַכְרֵת צַר הוֹאֵל



וְיִוָּדַע כִּי-אַתָּה אֱלֹהִים בְּיִשְׂרָאֵל.
