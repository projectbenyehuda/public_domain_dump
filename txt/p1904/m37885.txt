


	
	ובעלי הכספין
	



מַסֶּכֶת עֲשִׁירִים


––– וּבַעֲלֵי הַכְּסָפִין / בַּכֹּל מְעָרְבִין וּמִשְׁתַּתְּפִין / וְהַחֲנֻיּוֹת פְּתוּחִים / לְעַרְבֵי פְּסָחִים / וּמְנָתָם נָכוֹנָה / מֵרֹאשׁ הַשָּׁנָה / כֻּלָּם אֲחוּזֵי חָרֶב / יֵשְׁבוּ בְסֻכָּה לְמוֹ אָרֶב / עוֹרְכֵי מַעֲרָכוֹת / רַב בְּרָכוֹת / וְהִנֵּה יַעֲלוּ בַשְּׂרָרָה / מְשָׁרְתֵי עֲבוֹדָה זָרָה / וּבַקֹּדֶשׁ
מַעֲלִין / הַקָּרוֹב לְחֻלִּין / וְכֶסֶף נָקִי יַחֲלֹף / כָּל הַיָּד הַמַּרְבָּה לִבְדֹק / וְהַהוֹגֶה בָהֶם אוֹמְרִים כִּי לֹא יֶחְסָר / לֶחֶם וּבָשָׂר / אֲשֶׁר בְּנֵיהֶם כִּנְטִיעִים / כַּעֲרוּגוֹת זְרָעִים /
וְקוֹרְאִים אֶת עַצְמָם צַדִּיקִים / הָעוֹסְקִים בִּנְזִיקִים / וְאִם אֵין בְּיָדָם כְּסָפִים / מַה יַּחַלְקוּ הַשֻּׁתָּפִים / הֲלֹא בְכֶסֶף וּבִרְכוּשִׁים / וּבַעֲדִי עֲדָיִים וּמַלְבּוּשִׁים / תִּתְפָּאֵרְנָה הַנָּשִׁים / וּבִפְרוּטָה הַנִּתֶּנֶת / הָאִשָּׁה נִקְנֵית / וְכָל
מַשְׂכִּיל יֶאֱהַב / פֶּרֶק הַזָּהָב / וְרֹאשׁ כָּל הַמִּשְׁנָאוֹת / אֵלּוּ מְצִיאוֹת / וּמָצָא שַׂק אוֹ מָעוֹת / עִקַּר הַדֵּעוֹת / וְאֵין עֵת עַתָּה לְהוֹרוֹת / בִּקְדָשִׁים וְטָהֳרוֹת / וְאִם יָעִיזוּ מִבְטָא עַל הַשְּׁחִיטָה / הֲלֹא הוֹרוּ הַגְּמָרַָא / הַכֹּל שׁוֹחֲטִין וּשְׁחִיטָתָן כְּשֵׁרָה / וְשָׁמַעְנוּ מִפִּי הַמּוֹרִים / שֶׁהַשְּׁחִיטָה כְּשֵׁרָה בְזָרִים / כַּנָּשִׁים וְכַעֲבָדִים / אַף כִּי אֲנַחְנוּ שָׂרִים וְנִכְבָּדִים / וְאִם יָדֵינוּ אֶת הַדָּם לֹא שָׁפְכוּ / פֶּן יִתְלַכְלְכוּ / בְּגָדֵינוּ הַחֲמוּדִים / הַצְּנִיפוֹת
וְהָרְדִידִים / וְאִם לְכָל דָּבָר נוֹסִיף בֵּאוּר / אֵלּוּ דְּבָרִים שֶׁאֵין לָהֶם שִׁעוּר / וְהַכֶּסֶף אֶת הַכֹּל יַעֲנֶה / הֵן יַהֲרֹס וְלֹא יִבְנֶה / וּבִנְיָנוֹ אֵין לַהֲרֹס וּלְמַגֵּר / וְסֻגַּר וְאֵין פּוֹתֵח וּפָתַח וְאֵין סֹגֵר / וְעִקַּר כָּל הַיְחָסִים / עֹשֶׁר וּנְכָסִים / וּמַה בֶּצַע בְּשֹׁרֶשׁ לֹא יַעֲשֶׂה פֶּרִי / וְהוּא בֵין
אַחִים יַפְרִיא / וְאִם הַחֲכָמִים גָּדְלוּ וְעַל עֲשָׂרָה יוֹחֲסִין עָלוּ / הֲלֹא הֲלָכָה כְּדִבְרֵי הָאֹמְרִים / הַכֶּסֶף מְטַהֵר אֶת הַמַּמְזְרִים / וּמַעֲלִין לְיוֹחֲסִין עַל פִּי עֵד אֶחָד / כָּל שֶׁכֵּן בְּהִתְחַבֵּר יַחַד / שְׁנֵיהֶם בְּהַצַּד הַשֹּׁוֶה שֶׁבָּהֶם.





וְהַמַּטִּים עֲקַלְקַלּוֹתָם / וּנְלוֹזִים בְּמַעְגְּלוֹתָם / זֶה דַרְכָּם בְּאָרְחוֹתָם / עַל־עָשְׁרָם אֲשֶׁר
גָּדָל / וּמַהֲלָכָם עָקֵב בְּצַד גּוּדָל / זוֹקְפִים קוֹמָתָם / וּמִתְהַפְּכִים בִּמְסִלּוֹתָם / כַּאֲשֶׁר רוּחַ רָעָה מְבַעִתָּם / פַּעַם עַל יְמִינָם / פַּעַם עַל שְׂמֹאלָם / מִתְנַעְנְעִים בְּמַעְגָּלָם / רַגְלֵיהֶם רַגְלַיִם יְשָׁרוֹת / וּשְׁתֵּי כְתֵפֹת
חֹבְרֹת / לָהֶם / בְּכָל דַּרְכֵיהֶם / הָרֶגֶל מְהַלֶּכֶת רִאשׁוֹנִית / וְהַכָּתֵף הַשֵּׁנִית / וְעֵינֵיהֶם רָמוֹת / עַל רְקָמוֹת / כָּל דְּמוּת
מַלְבּוּשֵׁיהֶם / אֲשֶׁר עֲלֵיהֶם / מִנְעָל דָּחוּק וּכְתֹנֶת דַּקָּה / גֹּמֶד אָרְכָּהּ / וְאֵפוֹד וּמְעִיל וּבְגָדִים אֲחֵרִים / מִפָּז יְקָרִים / רְחָבִים מִלְּמַטָּה עַל שׁוּלֵיהֶם / וּקְצָרִים מִלְּמַעְלָה עַל חֲזֵיהֶם / וְרִמּוֹנֵי תְּכֵלֶת וְאַרְגָּמָן סְבִיב רָאשֵׁיהֶם / וּפְתוּחִים מִצִּדֵּיהֶם / לְאַרְבַּעַת רִבְעֵיהֶם / וַאֲזוֹרֵיהֶם מְשֻׁבָּחִים / עַל רִקֻּעֵי פַחִים / וְהֵם מְחֻשָּׁקִים כֶּסֶף / וַחֲגוּרִים עַל הָאַצִּילִים / כִּי מָתְנֵיהֶם אָחֲזוּ חִלְחוּלִים / וְנִשְׁתַּבְּרוּ רֹב צַלְעוֹתֵיהֶם / וְנִלְאוּ מֵהָכִיל אֲזוֹרֵיהֶם / אָכֵן אֶל אֲחוֹרֵיהֶם / תְּלוּיִם עַד שְׁתוֹתֵיהֶם / וְכִיס תָּלוּ לְצִדּוֹ / אִישׁ אִישׁ כַּאֲשֶׁר תַּשִּׂיג יָדוֹ / וְכֶסֶף וְזָהָב יְמַלְּאוֹ / וְנִשְׁמַע קוֹלוֹ בְּבֹאוֹ / גַּם אֶצְבְּעוֹתָיו /
נוֹשְׂאוֹת טַבְּעוֹתָיו / מוּסַבֹּת מִשְׁבְּצֹת זָהָב / בְּמִלֻּאֹתָם / פִּתוּחֵי חוֹתָם.





וּבְרֹאשָׁם כֹּבַע שָׁשְׁרָה / דַּקָּה וּקְצָרָה / וְעָלֶיהָ מַאֲחִיזִים / מַטְוֵה הָעִזִּים / עַל־חֲצִי הָרֹאשׁ שׂוּמָה / אֲשֶׁר שָׁם מְלוֹן הַכְּנִימָה / וְאֵין חִשְׁקָם / רַק לְכַסּוֹת נִתְקָם / אַךְ שַׂעֲרוֹתָם גְּדוֹלִים / קְוֻצּוֹתָיו תַּלְתַּלִּים / וְסֶרַח
הָעֹדֵף / בְּצִיצַת רֹאשָׁם נוֹטֵף / עַל הַכָּתֵף / וְאִם בֹּהַק פָּרַח בְּצַוְּרוֹנוֹ / שְׂעַר רֹאשׁוֹ יְכַסֶּנּוּ / קְצוּצֵי פֵאָה וּבְלוֹרִית / וְהֵם בַּעֲלֵי בְרִית / לַשְּׂעִירִים אֲשֶׁר הֵם / זֹנִים אַחֲרֵיהֶם / וְאוּלַי יָעוּפוּ / בְּשַׂעֲרוֹתָם אֲשֶׁר עָדָפוּ / יֵלְכוּ וְלֹא
יִיעָפוּ / וְכָל אֶחָד מֵהֶם מַחֲזִיק בְּאִוַּלְתּוֹ / אֵין אִישׁ נִחַם עַל רָעָתוֹ / וְאִם אֶשְׁאָלֶנּוּ עַל דֶּרֶךְ הַמּוּסָר: / ׳דָּבָר לִי אֵלֶיךָ הַשָּׂר׳ / אֶקְרָא אֵלָיו וְלֹא יַעֲנֵנִי / אֲשֶׁר בִּשְׂעָרָה יְשׁוּפֵנִי / וּשְׁבוּעַת אָלָה אִם יַבְטִיא / אֶל הַשַּׂעֲרָה וְלֹא
יַחְטִא / כִּי פְרָעוֹ לְשִׁמְצָה / לֵאמֹר אִם אֶבְחַר לְהִתְלוֹצֲצָה / אוֹ אִם עַוְלָה בִשְׂפָתַי תִּמְצָא / יִפֹּל מִשַּׂעֲרַת רֹאשִׁי אַרְצָה / וְלוּלֵי רְאִיתִים לָעָיִן / הַשֹּׁתִים בְּמִזְרְקֵי יָיִן / וּמְטַמְּאִים רֹאשׁ נִזְרָם / בְּמַעֲשֵׂיהֶם וּבְמוּסָרָם / אָמַרְתִּי
עֲלֵיהֶם / כִּי נְזִירִים הֵם / וְאַשְׁרֵי אֶרֶץ צָדְקוּ שָׂרֶיהָ / זַכּוּ נְזִירֶיהָ / וְאוּלַי נִשְׁבְּרוּ הַמִּסְפָּרַיִם וְהַתַּעַר / אוֹ הָאוּמָנִים אָחֲזוּ שָׂעַר / עַל־כֵּן יַעַר שַׂעֲרָם גָּדֵל וְסוֹעֵר / וְחַיָּתוֹ אֵין דַּי בָּעֵר / וְכִי יָרִים כִּנָּיו / לֶאֱחֹז בִּסְבַךְ קַרְנָיו / וְהִנֵּה הַשְּׁוָרִים מַלִּינִים עֲלֵיהֶם / כִּי
לָקְחוּ קַרְנוֹתֵיהֶם / לַעֲשׂוֹת לָהֶם מַסְרֵקִים / קָשִׁים וַחֲזָקִים / כִּי בִסְבַךְ שַׂעֲרוֹתָם תָּמִיד מִשְׁתַּבְּרִים / כְּשֶׁבֶר נֵבֶל יוֹצְרִים / וּמַדּוּעַ נֶחְתְּכָה הָאֵלָה / אֲשֶׁר בָּהּ אַבְשָׁלוֹם נִתְלָה / אוּלַי תוֹרִיד לַמִּצְעָר / בַּעֲלֵי הַשֵּׂעָר. – – –
