


	
	ליקוי שני המאורות
	




קָמֵיעַ וְלַחַשׁ לְשִׁירִי מֵעַיִן-הָרַע ר"ל



מַשְׁבִּיעַ אֲנִי עֲלֵיכוֹן עֵינֵי דְנֻקְבָא וְדֻכְרָא,



עֵינָא צְהוּבָא וִירוּקָא תְּכֶלְתָּא וְשֻׁחְרָא,



עֵינָא דְגַדְיָא עֵינָא דְשֻׁנְרָא וְכַלְבַּיָּא,



עֵינָא “דְּמָרֵי קוֹלָא” וְעֵינָא דַחֲסִידַיָּא;



עֵינָא דְשֵׁדִין וְעֵינָא דְצַדִּיקִין,



עֵינָא דְבַעֲלֵי-תְרֵסִין וְעֵינָא דְמַזִּיקִין,



עֵינָא דְמָרָנָן וְרַבָּנָן וְרוּחִין תַּקִּיפִין,



עֵינָא דִרְדַקטוֹרִין הַנֵּי כַּלְבִּין דַּחֲצִיפֵין…



עֵינָא דְלִילִית וְעֵינָא דְסַמָּאֵל הַשָּׂטָן



דִּי לָא תָרֵעוּן וְלָא תַשְׁחִיתוּן לְשִׁירִי הַקָּטָן



הַנִּקְרָא בִשְׁמוֹ: לִקּוּי שְׁנֵי הַמְּאֹרוֹת!



כִּי מְחַבְּרוֹ, חַס וְשָׁלוֹם, אֵינֶנּוּ אֶפִּיקוֹרוֹס…



וּמַשְׁבִּיעַ אֲנִי עֲלֵיכוֹן מִתְיַמְּרִין לְיָדְעֵי בִינָה,



מְבַקְּרִין וּמְנַקְּרִין מִסִּטְרָא חוֹרִינָא,



דִּי לָא תִקְפְּצוּן עַל שִׁירִי כְּעַכְבְּרִין רַשִּׁיעִין,



– כְּמָא דְמֵילַף אֲלֵפְתּוּן מִזִּמְנִין סַגִּיאִין –



לְבַקְּרוֹ וּלְנַקְּרוֹ בְּלִי רַחֲמִים וַחֲנִינָה,



דַּאֲנָא “בַּר יֶנְטָא” מִזַּרְעָא דְיוֹסֵף קָאֳתִינָא,



דִּי לָא שָׁלְטָא בְהוֹן עֵינָא דִי בִישָׁא,



עֵינָא דִמְסָאֲבָא וְעֵינָא דִי קַדִּישָׁא,



עֵינָא דְשׁוֹקַעַת וּבוֹלֶטֶת וְעֵינָא דִכְווּצָה,



לִילִית וְכָל-כַּתְּ דִּי-לַהּ חוּצָה!…



סיני וסנסיני וסמנגלוף, מֶלַח! בָּצֵל! וּפִלְפְּלָא חֲרִפְתָּא!


א


גִּבְעַת-הָעֲרָלוֹת הַקְּרוֹבָה לְעִיר קְבוּרָה



הַמְּלֵאָה כָרִמּוֹן חֲסִידִים וְאַנְשֵׁי צוּרָה,



נוֹדַעַת הִיא לִתְהִלָּה בְּקֶרֶב הָאָרֶץ



לְעִיר מְלֵאָה מַאֲמִינִים כְּרֶמֶשׂ וָשָׁרֶץ.



בָּהּ נִמְצְאוּ מִתְקַדְּשִׁים רַבֵּי נְקִיּוֹת,



קֻנְדֵּסִים פַּרְנַסִים גּוֹזְלֵי הַבְּרִיּוֹת,



עֲשִׁירִים וּשְׂעִירִים חֲמֹר חֲמֹרָתָיִם,



מְיֻחָשִׂים וּתְיָשִׁים בַּעֲלֵי קַרְנָיִם,



קַנָּאִים גַּבָּאִים עַתּוּדִים וּכְבָשִׂים



חַזָּנִים שַׁדְכָנִים וְצֹאן קֳדָשִׁים,



חֲסִידִים מַגִּידִים צְבוּעִים וּבַעֲלֵי תְקִיעוֹת,



עֲטַלֵּפִים חֲנֵפִים וְגַבָּאֵי עֲלִיּוֹת,



פְּרוּשִׁים וּגְרוּשִׁים טַפְלָא וּנְשַׁיָּא,



גַּרְגְּרָנִים רַעַבְתָנִים וְחֶבְרָה לְוַיָּה



אֶבְיוֹנִים גְּאֵי-יוֹנִים וּמֹשְׁכֵי רַפְסוּדוֹת,



שִׁכּוֹרִים מַמְזֵרִים וּמַרְבִּים בִּסְעוּדוֹת,



וְכֻלָּמוֹ מַאֲמִינִים הַמְּפַזְּרִים כָּל-הוֹנָם



לְר' לֶמֶךְ הַצַּדִּיק רַבָּם וַאֲדוֹנָם!



כִּי תִהְיֶה בְּאִישׁ צָרַעַת אוֹ סַפַּחַת,



אָדָם כִּי יִלָּקֶה בְּשַׁחֶפֶת וְקַדַּחַת,



עַל מַשְׁבֵּר כִּי תָבֹא אִשָּׁה מְצֵרָה,



כִּי תֵצֵא בָעִיר מַגֵּפָה אוֹ תַבְעֵרָה,



שְׂעִירִים כִּי יְרַקְּדוּ בְּרֵחַיִם שֶׁל רוּחַ,



כִּי תִדְבַּק בְּאִישׁ קְלִפָּה אוֹ רוּחַ,



בָּנִים וּבָנוֹת עֲקָרָה לְמַעַן תַּזְרִיעַ,



לְרַגְלָיו יִתְנַפְּלוּ וְהוּא הוּא הַמּוֹשִׁיעַ!



וּבְכֵן שֵׁם גִּבְעַת-הָעֲרָלוֹת כֹּל יֵדָעוּ,



וְגַם מֵאֲרָצוֹת רְחֹקוֹת אֵלֶיהָ יִסָּעוּ,



חֲדָשִׁים לַבְּקָרִים חֲסִידִים נִלְבָּבִים,



וְכָל-הַיּוֹם הִיא הֹמִיָּה מֵעֹבְרִים וְשָׁבִים,



הַבָּאִים לָשֵׂאת בְּרָכָה וְלִמְצֹא חֲנִינָה,



מֵאֵת הַצַּדִּיק ר' לֶמֶךְ רֹאֶה פְּנֵי הַשְּׁכִינָה.



וְהוּא מֵאַשְׁמֹרֶת הַבֹּקֶר עַד בֵּין הָעַרְבָּיִם



יֹשֵׁב עֲטוּף-לְבָנִים עַד כַּפּוֹת הָרַגְלָיִם,



וּבְצִדְקָתוֹ הַגְּדוֹלָה הוּא פוֹתֵחַ לָרְוָחָה



שַׁעֲרֵי הָרַחֲמִים לְכָל-נֶפֶשׁ נֶאֱנָחָה.



לֹא יַדַּח נִדָּח מִכָּל-פֹּרֵשׂ כַּפָּיִם,



לְמַעֲנָם יַעֲבִיר נַפְשׁוֹ בָּאֵשׁ וּבַמָּיִם



לְמַעֲנָם יִלָּחֵם בְּמַזִּיקִים מִבֹּקֶר עַד נֶשֶׁף,



לְמַעֲנָם יַכֶּה חֵרֶם קְלִפּוֹת וּבְנֵי רֶשֶׁף,



לְמַעֲנָם יֵאָבֵק בְּשֵׁדִים וְרוּחוֹת לַאֲלָפִים,



לְמַעֲנָם יוֹרִיד מִפְּקֻדָּתָם אוֹפַנִּים וּשְׂרָפִים,



לְמַעֲנָם יוֹנֶה וִירַמֶּה כָּל-מַלְאֲכֵי חַבָּלָה,



לְמַעֲנָם יִרְתֹּם הַסַ"ם כְּסוּס בַּעֲגָלָה,



לְמַעֲנָם יַעֲבִיד אֶת מַלְאַךְ הַמָּוֶת כַּחֲמוֹר בְּמַרְדַּעַת



וּלְכָל1 טָמִיר וְנֶעְלָם יַקְשֶׁה עַצְמוֹ לָדַעַת.



וְכֵן עֶשְׂרִים שָׁנָה כֹּה הוּא יֹשֵׁב בְּטָהֳרָה,



בְּגִבְעַת-הָעֲרָלוֹת הָעֲדִינָה וְהַמְפֹאָרָה,



וְיַעֲשֶׂה מְלַאכְתּוֹ וְיַרְבֶּה נִפְלְאוֹתָיו,



אֵין עֲרֹךְ לִרְכוּשׁוֹ וְקֵצֶה לְאֹצְרוֹתָיו.



לוֹ גַּנּוֹת וּפַרְדֵּסִים כֶּרֶם וָזָיִת!



סוּסִים בָּאֻרָוֹת חֲסִידִים בַּבָּיִת,



גַּבָּאִים בִּפְרֹזְדוֹר עֲקָרוֹת בֶּחָדֶר,



כֶּסֶף בְּאַמְתַּחַת וּבַקְּעָרָה רֹאשׁ וָפָדֶר.



לֹא יֶחְסַר לוֹ מְאוּמָה מִתַּעֲנוּגוֹת הַחַיִּים,



גַּם בָּרִיא הוּא וְשָׁמֵן לוֹ זְרֹעַ וּלְחָיַיִם



יָדָיו מְטִילֵי בַרְזֶל כְּתֵפָיו עֲצֵי שִׁטִּים,



צַוָּארוֹ מִגְדַּל שֵׁן בִּטְנוֹ עֲרֵמַת חִטִּים,



דַּלְתֵי פָנָיו חֲזָקוֹת עֵינָיו אַבְנֵי קָלַע,



גִּידֵי פַחֲדָיו שֹרָגוּ וּגְרָמָיו כַּסָּלַע.



כֻּלּוֹ אַמִּיץ וְחָזָק נֶאְזָר בִּגְבוּרָה!



מִכַּף רֶגֶל עַד רֹאשׁ אֵין בּוֹ חַבּוּרָה,



לֹא יֵדַע כָּל-מַדְוֶה מַחֲלָה וָנֶגַע,



כָּל-יָמָיו לֹא קָרָהוּ אָסוֹן וָפֶגַע,



מֵעוֹדוֹ יִתְעַנֵּג עַל שְׂמָחוֹת וְגִיל,



וַהֲמוֹן כְּסִילִים לוֹ יִשְׁתַּחֲווּ בְּרֶתֶת וָחִיל.


ב


אַךְ פִּתְאֹם הָהּ! לָאָרֶץ הוּרַד גְּאוֹנוֹ!



שָׁבְתָה גַּם עָרְבָה הַשִּׂמְחָה בִּמְעוֹנוֹ,



גָּלָה כָל-כְּבֹדוֹ בְּיוֹם תּוֹכֵחָה וּנְאָצָה,



וְהוֹד פָּנָיו הָהּ כִּסְּתָה פַּלָּצָה.



כִּי בְעֶצֶם הַיּוֹם – הָהּ! תַּאֲנִיָּה וַאֲנִיָּה! –



עָבַר דֶּרֶךְ הָעִיר הַצַּדִּיק רַב-הָעֲלִילִיָּה



ר' דָץ מֵחֲצַר-מָוֶת, קָדֹשׁ מְאֹד נוֹרָא!



עִם גַּבָּאָיו הַשְּׁנַיִם אַדִּירֵי הַתּוֹרָה,



וּבַהֲבִינוֹ בַּצַּדִּיק כִּי מְאוּם אֵינֶנּוּ חָסֵר



בָּעִיר גִּבְעַת-הָעֲרָלוֹת הֶחְלִיט לְהִוָּתֵר!



הֶחְלִיט גַּם הֵקִים מְזִמַּת לְבָבוֹ,



כִּי שָׂכַר לוֹ מָעוֹן וַיְכוֹנֶן בּוֹ מוֹשָׁבוֹ,



וַיַּצֵּב גַּבָּאָיו הָכֵן עַל הַמִּשְׁמֶרֶת,



בִּפְרֹזְדוֹר הַבַּיִת לְכָבוֹד וּלְתִפְאֶרֶת



לִקְצֹב פִּדְיֹנוֹת וְלִכְתֹּב פִּתְּקָאוֹת,



לַבָּאִים הִוָּשַׁע מִבַּעַל הַנִּפְלָאוֹת.



וְהָעָם מִקָּצֶה הָהּ! קְטַנֵּי אֲמָנָה!



כְּבָר שָׁכְחוּ בְרִיתָם וֶאֱמוּנָתָם הַיְשָׁנָה,



בְּצַדִיקָם ר' לֶמֶךְ מוֹשִׁיעָם בְּכָל-צָרָה,



וְעַל רֹאשׁ ר' דָץ שָׂמוּ הָעֲטָרָה.



וּכְבָר מִכָּל-פִּנָּה נָהֲרוּ הֲמוֹנִים,



גְּבִירִים וּגְבִירוֹת אֶבְיוֹנוֹת וְאֶבְיוֹנִים,



הָרוֹת וַעֲקָרוֹת לַעֲשָׂרוֹת וּמֵאוֹת,



בֵּיתָה ר' דָץ בְּיָדַיִם מְלֵאוֹת.



וּכְבָר יָשַׁב בָּרְחָבָה כְּצַדִּיק יָשָׁן נוֹשָׁן,



בֵּין מַאֲמִינָיו וַחֲסִידָיו וַיִּפְרַח כַּשּׁוֹשָׁן;



וַיַּרְבֶּה נִפְלָאוֹת בְּגִבְעַת-הָעֲרָלוֹת,



וְעֵינֵי ר' לֶמֶךְ הָהּ! רֹאוֹת וְכָלוֹת!



רֹאוֹת וְכָלוֹת כִּי מִי יֵדַע כָּמֹהוּ,



הוֹנוֹ וַעֲמָלוֹ, כִּי יַעֲלוּ בַתֹּהוּ;



וְעַל דּוֹרְשָׁיו כִּי יִכְלוּ נַפְשׁוֹ נֶאֱנָחָה!



פֶּן יִהְיֶה בְּאַחֲרִיתוֹ כְּזוֹנָה נִשְׁכָּחָה.



גַּם הֶעֱמִיק לִרְאוֹת בַּשָּׁמַיִם מִמַּעַל,



כִּי מַלְאָכָיו וּשְׂרָפָיו גַּם הֵמָּה בַּמַּעַל,



וּמֵעַתָּה הָהּ! בּוֹ בָגֹד יִבְגֹּדוּ,



וּבְצָרָתוֹ זֹאת הוֹי, נֶגֶד נִגְעוֹ יַעֲמֹדוּ.


ג


אָז הִתְנָעֵר ר' לֶמֶךְ בְּאַפּוֹ וּבַחֲרוֹנוֹ,



וַיָּרֶם אֶת-קוֹלוֹ וַיִּשְׁרֹק לְצֹאנוֹ,



וַיִּקָּבְצוּ הַחֲסִידִים מֵרֹאשׁ וְעַד זָנָב,



לְעֶזְרַת צַדִּיקָם הֶחָסִיד הֶעָנָו,



וַיַּאַסְפוּ אֲסֵפָה וַיְקַדְּשׁוּ עֲצָרָה,



לְהִוָּעֵץ עִם רַבָּם בְּרוּחַ נִשְׁבָּרָה,



כְּדָת מַה-לַּעֲשׂוֹת בְּר' דָץ מֵחֲצַר-מָוֶת?



עַל מְקוֹם מוֹשִׁיעָם כִּי הִרְהִיב לָשֶׁבֶת!



אָז נִמְנוּ וְגָמְרוּ כְּאִישׁ אֶחָד חֲבֵרִים,



לִשְׁלֹחַ אֶל ר' דָץ שְׁנֵי אֲנָשִׁים כְּשֵׁרִים,



מִשִּׁבְעָה טוֹבֵי הָעִיר – וּלְהָעִיד בְּפָנָיו,



מִבְּלִי חַכּוֹת אַף רֶגַע לַדֶּרֶךְ יָשִׂים פָּנָיו,



כִּי בְעִיר מוֹשַׁב הַצַּדִיק לֹא יַכִּירֵהוּ מְעוֹנוֹ,



וְכָל-צְדָקָה לוֹ אַיִן לְהִתְחַמֵּם מִגֵּז צֹאנוֹ,



וְלַמְרוֹת אִם יָזִיד זֹאת הָאַזְהָרָה!



נְכוֹנִים בְּיָדוֹ יְמֵי חֹשֶׁךְ וְצָרָה.


ד


אַךְ עֲצָתָם הָהּ! לֹא עָשְׂתָה כָּל-פֶּרִי



הַצַּדִּיק ר' דָץ הָלַךְ עִמָּם קֶרִי,



וַיְחָרֶף וַיְגַדֵּף ר' לֶמֶךְ וְהַנִּבְחָרִים,



וּבְמִכְתָּב מָלֵא עַזּוֹת עָנָה אֵלֶּה הַדְּבָרִים:



"שִׁמְעָה-נָּא לֶמֶךְ סִטְרָא דְסַמָּאֵל



לוּקוּס הַמְּכַלֶּה מָמוֹנָם שֶׁל יִשְׂרָאֵל!



אֲנִי דָץ בַּר שִׁלְשֹׁם בֶּן-קְדֹשִׁים הִנֵּנִי!



הַזָּקֵן בְּעַצְמוֹ הָיָה אָבִי-זְקֵנִי;



הַקָּזַק2 הַנּוֹדַע הִנֵּהוּ אֲחִי-אָבִי,



וְאָחִי הַקָּטָן – הַיְהוּדִי הַפְּשִׁיסְחָבִי.



וְכֵן קְרוֹבַי כֻּלָּם אַךְ קְדֹשִׁים הִנָּמוֹ!



עַד דֹּרוֹת אַלְפַּיִם שַׁכֻּלָּה אֵין בָּמוֹ,



כֻּלָּם צַדִּיקִים, כֻּלָּם אַנְשֵׁי-פְלָאוֹת,



וּלְעֻמָּתָם הִנְּךָ אַך זָנָב לָאֲרָיוֹת3



וְאַחֲרֵי כִי תֵדַע מֶה וּמִי אָנֹכִי,



שְׁמַע וַאֲחַוְּךָ בַּמֶּה גָּדוֹל כֹּחִי!



דַּע לְךָ לֶמֶךְ כִּי כֹחִי מְאֹד נַעֲלָה,



לְהַטּוֹת לְחֶפְצִי כָּל-שָׂרֵי מַעֲלָה;



מִיכָאֵל וְגַבְרִיאֵל וּמְטַטְרוֹן שַׂר-הַפָּנִים,



יְשָׁרְתוּנִי בְּיִרְאָה כַּעֲבָדִים נֶאֱמָנִים,



סַמָּאֵל וְקוּשִׁיעֵאל וּמַלְאַךְ הַדּוֹמָה,



פַּצְפַּצְיָה וּפַצְפִּיאֵל וַהֲדַרְנִיאֵל רָם הַקּוֹמָה,



וְעוֹד מַלְאָכִים אֲחֵרִים טְמֵאִים גַּם קְדֹשִׁים



בֵּין כַּפְתֹּרַי יָלִינוּ כְּאֹרְחַת פַּרְעֹשִׁים.



וְכֻלָּם עֹשִׂים בְּאֵימָה עַל פִּי חֶפְצִי וּמִצְוָתִי,



טוֹב לְמֵטִיבַי וְרָעָה לִמְבַקְשֵׁי רָעָתִי,



לֹא יַבְדִּילוּ מְאוּמָה בְּחַרְבָּם הַקָּשָׁה,



בֵּין קָדֹשׁ לְנָבָל בֵּין צַדִּיק לְרָשָׁע.



כֹּפֶר וְשַׁלְמֹנִים כְּמוֹ דָוָה יְזָרוּ,



וּכְשׁוֹפְטִים נֶאֱמָנִים זְקֵנִים לֹא יֶהְדָּרוּ.



וְעַתָּה ר' לֶמֶךְ אֲהוִּבי וַחֲבִיבִי!



אֲמָרַי הַאֲזִינָה וּשְׁעֵה נָא אֶל נִיבִי,



בָּעִיר גִּבְעַת-הָעֲרָלוֹת דַּע לְךָ דַע,



אֲנִי וְגַבָּאַי עָשִׂינוּ דִירַת קְבַע!



בָּהּ מוֹפְתַי אֶתֵּנָה בְּכָל-אוֹנִי וְכֹחִי,



וְזֹאת תִּפְאַרְתְּךָ כִּי שֻׁתָּפְךָ אָנֹכִי!



אֵין הֶבְדֵּל בֵּינֵינוּ אַף כִּמְלֹא נִימָא,



וְכִי תִצְעַק גַּם מָרָה מִיָּמִים יָמִימָה,



בְּקוֹלְךָ לֹא אֶשְׁמַע לֹא אֶעֱזֹב מְעוֹנִי!



כִּי בְּקָרְךָ – בְּקָרִי וְצֹאנְךָ – צֹאנִי.



וְאַתָּה אֲהוּבִי! הֶרֶף נָא מֵאַף



שָׂהֲדִי בַּמְּרוֹמִים מִפֶּשַׁע אֲנִי חַף;



לֶחֶם בִּקַּשְׁתִּי! וְלֶחֶם מָצָאתִי!



וְלִלְחַם לֶחֶם לְעִירְךָ לֹא בָאתִי.



וְעָלֶיךָ עוֹד לְהוֹדוֹת לָאֵל גֹּמֵר עָלֶיךָ



בְּשָׁלְחוֹ לְךָ מוֹשִׁיעַ לְעֵת זִקְנָתֶךָ,



לְהָקֵל אֶת-עֻלְּךָ מִמַּעַל וּמִתַּחַת.



בָּעֵת אֲשֶׁר תַּעֲמֹד עַל יַרְכְּתֵי שַׁחַת.



רְאֵה נָא חֲבִיבִי, לְשָׁלוֹם הִנֵּנִי!



וְלַעֲרֹךְ מִלְחָמָה אַל נָא תְאַלְּצֵנִי.



כִּי רַבִּים אֲשֶׁר אִתִּי מֵאֲשֶׁר אִתֶּךָ,



וְאִם צַר כֹּחֶכָה עֲצָר-נָא יָדֶיךָ,



מִנָּאִי חֲבִיבְךָ מִתְּמֹל וּמִשִּׁלְשֹׁם,



אַסְקֻפָּה הַנִּדְרֶסֶת דָּץ בַּר שִׁלְשֹׁם.


ה


וְר' לֶמֶךְ בְּקָרְאוֹ אֶת-אֵלֶּה הַטּוּרִים,



הִתְנַפֵּל אַרְצָה וַיֵבְךְּ תַּמְרוּרִים,



וַיָּגָז שְׂעַר רֹאשׁוֹ וַיִּמְרֹט הַלֶּחִי,



וְעִמּוֹ קְהַל חֲסִידָיו יָרְדוּ בַבֶּכִי,



וַתְּהִי מְהוּמָה וּמְבוּכָה תַּאֲנִיָּה וַאֲנִיָּה!



כָּל-לֵבָב דַּוָי וְכָל-עַיִן בּוֹכִיָה.



בֹּשׁוּ מַלְאָכִים! אֶרְאֶלִּים עֲטוּ כְלִמָּה!



חִפְרוּ מַזָּרוֹת עָשׁ כְּסִיל וְכִימָה!



לִבְשׁוּ בִגְדֵי אֵבֶל אַתֶּם שַׂרְפֵי מַעְלָה,



וְהֵחָבְשׁוּ בַטָּמוּן טִפְטַפְיָה וְלַיְלָה![^4]



אַף מָרְרוּ בִּבְכִי עֶגְלוֹנֵי הַמֶּרְכָּבָה



כִּי נָפַל אֲדוֹנֵיכֶם הַדָּגוּל מֵרְבָבָה.



גַּם אַתְּ עֲלוּבָה תּוֹרָה יְקָרָה!



הִתְפַּלְשִׁי בָּאֵפֶר וְקוֹנְנִי מָרָה,



עַל שֶׁבֶר בֹּעֲלֵךְ ר' לֶמֶךְ הַנַּעֲלָה,



שֶׁעוֹנָתֵךְ לֹא גָרַע יוֹמָם וָלַיְלָה;



כִּי נִלְכַּד הָהּ, וַיְהִי לְדֵרָאוֹן!



וּמְשַׂנְּאוֹ עָלָיו יָרִים רֹאשׁוֹ בְּגָאוֹן.



אַךְ ר' לֶמֶךְ לֹא שָׁקַט וַיֶאֱסֹף בַּשְּׁנִיָּה,



אֲסֵפָה גְדוֹלָה בַּחֲדַר הָעֲלִיָּה,



וַיִּקָּבְצוּ לְבֵיתוֹ בְּמִדָּה מְאֹד גְּדוּשָׁה,



הֲמוֹן כִּתּוֹת וַחֲבוּרוֹת וְתַשְׁמִישֵׁי הַקְּדֻשָּׁה,



לְהַמְתִּיק סוֹד יַחַד וּלְהַמְצִיא יֶשַׁע,



לְרַבָּם הֶעָלוּב הַמְחוֹלָל בְּלִי פֶשַׁע.



וְהַצַּדִּיק בְּעַצְמוֹ בִּקְדֻשָּׁה וּבְטָהֳרָה,



קָם מֵעַל הַכִּסֵּא וַיִּקְרָא מִן-הַסְּעָרָה:



"שִׁמְעוּ נָא רַבּוֹתַי צַנְתְּרֵי דִּי-דַהֲבָא



בּוֹצִינִין קַדִּישִׁין בְּנֵי אָדָם וְחַוָּה!



עֶשְׂרִים שְׁנוֹת עָמָל בְּעִירְכֶם לִי מָנִיתִי,



אֵין פֶּגַע וְצָרָה שֶׁבָּהּ לֹא הָיִיתִי!



לְמַעַנְכֶם עָבַדְתִּי בְּזֵעַת אַפָּיִם,



לָחַמְתִּי בְּמַזִּיקִים וּבְמַלְאֲכֵי שָׁמָיִם,



נֶאֱבַקְתִּי בְּרוּחוֹת וְשֵׁדִים עַזֵּי-פָנִים,



וּלְמִרְמָס נָתַתִּי קְלִפּוֹת וְכַתְּ לֵצָנִים.



וּלְמַעַנְכֶם הָהּ מִבֹּקֶר עַד עָרֶב,



עַל מְקוֹמִי יָשַׁבְתִּי כְּאַסִּיר וּשְׁבוּי חֶרֶב,



לֹא יָדַעְתִּי שָׁלֵו וְרִגְעֵי מַרְגֵּעָה,



לֹא הִתְעַנַּגְתִּי מֵעוֹדִי עַל צוּרַת מַטְבֵּעָה



לֹא נֶהֱנֵיתֵי מֵעוֹלָמִי אַף כְּאֶצְבַּע קְטַנָּה,



וְאַחֲרִיתִי אֲהָהּ! מָרָה כַּלַּעֲנָה.



עוּרוּ רַבּוֹתַי! עוּרוּ צֹאן מַרְעִיתִי!



תִּבְעוּ עֶלְבּוֹנִי מִיַּד אוֹיְבִי וּמַצְמִיתִי,



הֲלֹא כֻּלְּכֶם צַדִּיקִים כֻּלְּכֶם בְּנֵי יִשְׂרָאֵל,



וְאֵיךְ תּוּכְלוּ לִרְאוֹת שְׂנוּאָה כִּי תִבָּעֵל?!



וְהַדְּבָרִים בְּצֵאתָם מִפִּי רַב הָעֲלִילִיָּה,



מִיַּד כָּל-הָעָם גָּעוּ בַּבְּכִיָּה,



וַיְמָרְרוּ כֻלָּמוֹ – מִשּׁוֹר עַד חֲמוֹר,



כְּאֵם רַחֲמָנִיָּה תָּמֵר עַל הַבְּכוֹר.


ו


אָז קָם מֵעַל הַשֻּׁלְחָן בְּקֶצֶף וְאֵימָה,



חָסִיד בַּעַל בֶּטֶן וַיִּקְרָא בְּחֵמָה:



הוֹי פְּתָאִים בּוֹעֲרִים תּוֹעִים וּנְבוֹכִים!



הַעֵת הִיא עַתָּה לְהָרִים קוֹל בּוֹכִים?



הֲכִי לֹא דִמְעָה וְקִינָה לְשָׁמִיר וָשָׁיִת



עֵת תֹּאחַז הָאֵשׁ בְּפִנּוֹת הַבָּיִת?!



הָבוּ גַבָּאִים, הָבוּ מְשָׁרְתֵי הַקְּהִלָּה!



אֶת-צוֹרֵר רַבֵּנוּ לָאָרֶץ נַשְׁפִּילָה,



נֵצֵא נָא לִקְרָאתוֹ בְּחֶרֶב וָשֶׁלַח,



וְנִזְרֶה עַל נָוֵהוּ גָפְרִית וָמֶלַח,



נִשָּׁפֵט נָא אִתּוֹ בָּאֵשׁ וּבַמַּיִם,



וְנַכְרִיתָה אֶת-שְׁמוֹ מֵאֶרֶץ הַחַיִּים.



אַךְ דְּבָרָיו אֵלֶּה הָאִישׁ עוֹד לֹא גָמַר,



וְהִנֵּה קָם עַל רַגְלוֹהִי חָסִיד אַחַר וְאָמַר:



אַל נָא רַבּוֹתַי! אַל זֶרַע יִשְׂרָאֵל!



לַעֲרֹךְ מִלְחָמָה עִם הָרָשָׁע סַמָּאֵל,



הֲשֵׁם דָּץ הָעַוָּל הַיּוֹם תִּשְׁמְעוּ רִאשׁוֹנָה



כִּי תֹאמְרוּ לִלְחֹם עִם מֵצַח אִשָּׁה זוֹנָה?!



אַחַי חִרְדוּ נָא, אַל תְּגָרוּ מְדָנִים!



פֶּן יַכְּכֶם חֵרֶם אָבוֹת עַל בָּנִים,



כִּי מֵעוֹדִי יְדַעְתִּיו לְחַיָּה טוֹרֶפֶת,



וּדְעוּ כִּי לְדִמְכֶם נַפְשׁוֹ שֹׁאֶפֶת,



וְאִם תֵּצְאוּ עוֹד נֶגְדּוֹ בְּחֵמָה וְעֹז מֶצַח,



אַחֲרִיתְכֶם נִכְרָתָה וְתֹאבְדוּ לָנֶצַח!



וּבְכֵן מוֹרַי וְרַבּוֹתַי עֲצָתִי לָכֶם אֱמוּנָה,



אִם אַךְ יֵשׁ בָּכֶם שְׁבִיב חָכְמָה וּתְבוּנָה,



שְׁקָלִים אַלְפַּיִם שִׁקְלוּ עַל יַד רַבֵּנוּ!



וּבָם יְשַׁחֵד אֶת-דָּץ כִּי יַעֲזֹב עִירֵנוּ.



אָז תֶּחְדַּל מַחְלֹקֶת מַשְׂטֵמָה וְאֵשׁ זָרָה,



וְעַל צַדִּיקֵנוּ לְיָשְׁנָהּ תָּשׁוּב הָעֲטָרָה.


ז


הֶחָסִיד טֶרֶם כִּלָּה עֲצָתוֹ וְקוֹל דְּבָרוֹ,



וְר' לֶמֶךְ הִתְנַפֵּל בְּאַהֲבָה עַל צַוָּארוֹ



וַיִּשָּׁקֵהוּ עַל לֶחֱיוֹ נְשִׁיקָה מְמֻשָּׁכָה,



וַיִּקְרָא בְּקוֹל: תָּבֹא עָלֶיךָ בְּרָכָה!



וְאַחַר רָחַץ יָדָיו וַיְבָרֶךְ בְּקוֹל רָם:



“בָּרוּךְ שֶׁנָּתַן מֵחָכְמָתוֹ לְבָשָׂר וָדָם!”



וְהַחֲסִידִים מַרְעִידִים כַּשָּׂטָן מִקּוֹל הַתְּרוּעָה,



קָמוּ מִמְּקוֹמָם בְּלִי כָל-רֶגֶשׁ וּתְנוּעָה,



וְאָמֵן בְּכַוָּנָה גְדוֹלָה עָנוּ אַחֲרָיו בֶּהָמוֹן!



וַאֲגַב אָרְחָא עַל הַשֻּׁלְחָן הִנִּיחוּ גַּם הַמָּמוֹן;



לֹא גָרְעוּ אַף גֵּרָה מִשְּׁקָלִים אַלְפָּיִם,



כִּי כֵן נִגְזַר הֶאָח! לִפְנֵי אָבִינוּ בַשָּׁמַיִם.



וְעוֹד בַּיּוֹם הַהוּא ר' לֶמֶךְ מְאֹד אָץ,



לִשְׁלֹחַ הַמִּנְחָה לַצַּדִּיק ר' דָּץ,



וְעוֹד בַּיּוֹם הַהוּא גַּם מֶרְכָּבָה מְרַקֵּדָה,



בְּלִוְיַת הַצַּדִּיק וְכָל-נְשִׂיאֵי הָעֵדָה,



מֵעִיר גִּבְעַת-הָעֲרָלוֹת בְּחִפָּזוֹן יָצָאָה;



וּמִי הֵמָּה הַנּוֹסְעִים? – הֶאָח, חִידָה נִפְלָאָה!



הַצַּדִּיק ר' דָץ עִם גַּבָּאָיו הַשְּׁנָיִם!



וּבִצְרֹרוֹ גַּם צְרוּרִים שְׁקָלָיו הָאַלְפָּיִם…



כֶּסֶף! כֶּסֶף! הֶאָח סְגֻלָּה נֶאֱהָבָה!



מַה-יָּקָר עֶרְכֶּךָ בַּתֵּבֵל הָרְחָבָה,



מַה-יָּקָר עֶרְכֶּךָ, הֶאָח! אוֹצַר כָּל-נָחַת!



אֵלֶיךָ אֵין עֲרֹךְ עַל הָאָרֶץ מִתָּחַת,



כָּל-פֹּעַל וּמַעַשׂ, בְּלִי מְנוּחָה וּמַרְגֵּעָה,



לְמַעַנְךָ אַךְ יַעַשׂ, בֶּן-אָדָם תּוֹלֵעָה;



לֹא יַבִּיט מְאוּמָה מַחֲשַׁבְתּוֹ אִם רְצוּיָה



וְרַק הַשִּׂיגְךָ בְּכִיסוֹ יָדוֹ נְטוּיָה.



לְמַעַנְךָ הָהּ! עָיֵף וּרְפֵה כֹחַ,



רֹאשׁוֹ עַל כְּרָעָיו יֵלֶךְ אִישׁ שְׁחֹחַ,



יַעֲבֹד וְיַעֲמֹל מִבֹּקֶר עַד עָרֶב,



וּבְמִסְתָּרִים יָבֹא כַּחַיָּה בְּמוֹ אָרֶב,



לִגְזֹר וְלַחְתֹּף עַל יָמִין וּשְׂמֹאל!



וְעֵינָיו לֹא תִשְׂבַּעְנָה כַּאֲבַדּוֹן וּשְׁאוֹל.



וּלְמַעַנְךָ הָהּ! נוֹכֵל וְעַיִּט צָבוּעַ,



יִלְבַּשׁ עַל בְּשָׂרוֹ בֶּגֶד קָרוּעַ,



יִתְחַפֵּשׂ לִמְעֻנֶּה וִידוּעַ מַכְאוֹבִים,



וּבְיָדוֹ יִחוּסוֹ כִּי הוּא עָנִי בֶּן-טוֹבִים,



וְכֹה יָסֹב בֶּעָרִים לֶאֱסֹף נְדָבוֹת,



וְיוֹנֶה וִירַמֶּה אֲנָשִׁים לִרְבָבוֹת. –



וּלְמַעַנְךָ הָהּ! זָקֵן אִישׁ תַּהְפֻּכוֹת,



יְגַדֵּל בְּרֹאשׁוֹ שְׁתֵּי פֵאוֹת אֲרוּכוֹת,



יַרְחִיב אֶת-זְקָנוֹ לְכָל-עֵבֶר וָרוּחַ,



וּמִבְּשָׂרוֹ לֹא יָמִישׁ בֶּגֶד סָרוּחַ,



יְקַשֵּׁט אֶת-רַגְלָיו בִּנְעָלִים וּפֻזְּמְקָאוֹת…



וְיוֹרֶה עַל אֶחָיו אַבְנֵי בְלִסְטְרָאוֹת!



כָּל קֹדֶשׁ יְחַלֵּל וּבְכָל-תּוֹעֵבָה יִתְגָּאֵל,



וִיהִי לְרֹאשׁ וְקָצִין עַל עֵדָה בְּיִשְׂרָאֵל…



וּלְמַעַנְךָ הָהּ! שְׁטְרַיְמֶל רַב-הַזְּנָבוֹת,



יִלְבַּשׁ אִישׁ מִרְמָה וְגוֹנֵב לְבָבוֹת,



וִיהִי לְרַב וְגָאוֹן בְּפֶתַח עֵינָיִם,



וְיִשְׂתָּרֵר עַל אֶחָיו כְּפַרְעֹה בְּמִצְרָיִם,



לֹא יָשִׁית כָּל-לֵב לַעֲנִיִּים מְרוּדִים,



וּבִגְזֵרוֹת חֲדָשׁוֹת יַשְׂבִּיעֵם נְדוּדִים…



כָּל-דִּבָּה וְרָכִיל כְּמִנְחָה יָרִיחַ!



וּמֵאֶנְקַת אֶבְיוֹנִים לִבּוֹ יַקְשִׁיחַ. –



וּלְמַעַנְךָ הָהּ! עָשִׁיר גְּדָל-בָּשָׂר,



יְקוֹמֵם בְּעִירוֹ מֶכֶס לַבָּשָׂר,



וְלֶאֱכֹל בְּשַׂר אֶחָיו יִקְנֶה מַחְשָׁבוֹת



וְיֶאֱרֹב לְדָמָם כִּזְאֵב עֲרָבוֹת,



לֹא יָחֹס עַל דָּל, חֲנִינָה לוֹ זָרָה!



וְעֵינָיו לֹא יָשִׂים לְאֹבֵד בַּצָּרָה,



כָּל-עִסְקוֹ וְעִנְיָנוֹ אַךְ הַרְבּוֹת הַזֶּבַח,



וַאֲנָשִׁים לְמֵאוֹת יַעֲלֶה לַטֶּבַח. –



וּלְמַעַנְךָ הָהּ! אִישׁ מִרְמָה וּנְבָלָה!



יְכַנֶּה לְעַצְמוֹ צַדִּיק בַּעַל קַבָּלָה,



וּבְקֶרֶב עֲדָתוֹ פְלָאוֹת הוּא מַשִׁית,



שֶׁלֹּא הָיוּ וְנִבְרָאוּ מִשֵּׁשֶׁת יְמֵי בְרֵאשִׁית;



יַעֲשֹׁק וְיִגְנֹב וְיַעֲרִים כַּנָּחָשׁ,



וּפְתָאִים לִרְבָבוֹת יְסַבֵּב בְּכָחַשׁ,



אַף עֵינוֹ לֹא תִשְׂבַּע מִמְּשָׁל מִמְשָׁל רַב,



וכִבְנוֹת עֲלוּקָה יִקְרָא: הַב! הַב!



וּלְמַעַנְךָ הָהּ! גַּם ר' דָץ מֵחֲצַר-מָוֶת,



בָּעִיר גִּבְעַת-הָעֲרָלוֹת אִוָּה לָשָׁבֶת,



לֹא חָס עַל צַדִּיקָהּ בֹּעֲלָהּ מִלְּפָנִים,



וַיָּבֹא בִּגְבוּלוֹ לְגָרוֹת מְדָנִים,



אַף הִכְבִּיד אֶת-אָזְנוֹ וַיְמָאֵן הִכָּנֵעַ!



וּבְעִירוֹ עַד עוֹלָם גָּזַר לְהִשְׁתַּקֵּעַ. –



וּלְמַעַנְךָ הָהּ! אוֹ הֶאָח! פִּלְאֵי פְלָאִים!



הַצַּדִּיק ר' דָץ עִם גַּבָּאָיו הַשְּׁנָיִם,



מִהֵר זֶה עַתָּה מִקִּנּוֹ לִבְרֹחַ,



וְלַעֲזֹב הָעֲבֻדָּה עִם כָּל-הַמַּלְקֹּחַ,



עַל יַד אוֹיְבוֹ וּמִתְנַגְּדוֹ מְבַקֵּשׁ רָעָתוֹ,



וְלֹא שָׁמַר לְמַלְּאֹת דִּבְרֵי תְשׁוּבָתוֹ. –



וּלְמַעַנְךָ הָהּ! גַּם פֹּשְׁעִים מִנֹּעַר,



לַאֲלָפִים וּרְבָבוֹת יֵשְׁבוּ בַסֹּהַר,



אֲסוּרִים וּקְשׁוּרִים בִכְבָלִים וָאֳזִקִּים,



וְעִמָּהֶם הָהּ! יִלָּכְדוּ גַם צַדִּיקִים.



גַּם צַדִּיקִים הָהּ! יִלָּכְדוּן בַּצָּרָה.



וְרַבּוֹתַי נָא שִׁמְעוּ מַעֲשֶׂה שֶׁקָּרָה:


ח


ר' דָץ מֵחֲצַר-מָוֶת הַמְקֻבָּל וְהַמְקַבֵּל,



צַדִּיק לַחֲסִידִים וּלְצַדִּיקִים מְחַבֵּל,



מֵעִיר מוֹשַׁב מְנַגְּדוֹ בְּנֶפֶשׁ שׂמַחַת,



טֶרֶם הִרְחִיק לִנְסֹעַ כִּבְרַת אֶרֶץ אַחַת,



עוֹדֶנּוּ בַדֶּרֶךְ עִם שְׁקָלָיו הַמְזֻמָּנִים,



וּבְגִבְעַת-הָעֲרָלוֹת עוֹד צָהֳלָה עַל כָּל-פָּנִים.



עוֹד הַחֲסִידִים מְרַקְּדִים כִּשְׂעִירִים וְאַיָּלִים,



שְׂמֵחִים בְּצַדִּיקָם וּבִתְשׁוּעַת הַשְּׁקָלִים,



עוֹד יָגִיל כָּל-אִישׁ, יִשְׂמַח כָּל-גָּבֶר,



וְהִנֵּה פִּתְאֹם לְפֶתֵע, הוֹי שֵׁאת וָשָׁבֶר!



מִקְרֶה לֹא-טָהוֹר לַצַדִּיק הָהּ קָרָה!



וּבְגִבְעַת-הָעֲרָלוֹת נִתְחַדְּשָׁה הַצָּרָה.



כִּי שְׁנֵי סוּסִים קַדִּישִׁין יַלְדֵי שַׁעֲשֻׁעִים,



סוּסִים שֶׁחַיֵּי הַצַדִּיק בָּמוֹ אַךְ תְּלֻאִים



בָּאֻרָוָה עַל אֲבוּסָם כַּאֲשֶׁר נִצָּבוּ,



כְּהֶרֶף-עַיִן הָהּ! מֵהָאֻרָוָה גֻנָּבוּ.



גֻנָּבוּ מֵהָאֻרָוָה – וּבְלִוְיַת הָאֲבֵדָה,



נֶאֶבְדָה הָהּ! גַּם מֶרְכָּבָה מְרַקֵּדָה!



לַשָּׁוְא עָמְלוּ סָרִיסִים וְשׁוֹטְרֵי הַמֶּמְשָׁלָה,



לִדְרֹשׁ וְלַחֲקֹר אַחַר עוֹשֵׂה הַנְּבָלָה,



לַשָּׁוְא גַּם שְׁתַּדְּלָנִים לְחַפֵּשׂ יָגָעוּ



הַגַּנָּב אֵינֶנּוּ וְעִקְבוֹתָיו לֹא נוֹדָעוּ!



וַתֵּהֹם כָּל-הָעִיר בְּרַעַשׁ וְקוֹל הֲמֻלָּה,



וַיִּתְאַבְּלוּ נַעַר וְזָקֵן בָּחוּר וּבְתוּלָה.



וַיָּקָם ר' לֶמֶךְ כְּגִבּוֹר מִלְחָמָה,



וַיֶּאֱסֹף אֶת-פְּתָאָיו וַיַּעַל עַל הַבָּמָה,



וַיָּרֶם אֶת-קֹלוֹ קוֹל אַדִּיר וָרָם:



"שִׁמְעוּ נָא רַבּוֹתַי עַם אֱלֹהֵי אַבְרָם!



רַבּוֹת נָשָׂאתִי, הֲלֹא גָלוּי וְיָדוּעַ,



צָרָה עַל צָרָה מִדָּץ הַפָּרוּעַ.



רַבּוֹת שָׂבַעְתִּי דֵּי מַצָּה וּמְדָנִים,



מִכָּל-מְעַוֵּל וְחוֹמֵץ וְכֶלֶב עַז-פָּנִים;



אַךְ עַתָּה נִלְאֵתִי, וּכְבָר כָּשַׁל בִּי, אֲהָהּ!



כֹּחַ הַסַּבָּל לִסְבֹּל אֶת-הָרָעָה.



דְּעוּ נָא רַבּוֹתַי כִּי סוּסַי הַגְּנוּבִים,



גִּלְגּוּלִים הִנָּמוֹ מִשְּׁנֵי צַדִּיקִים חֲשׁוּבִים,



הַנּוֹדָעִים בַּתֵּבֵל בְּשָׁנִים קַדְמֹנִיּוֹת,



לִקְדֹשִׁים וּטְהוֹרִים וּנְשָׁמוֹת נְקִיּוֹת,



אַךְ בַּעֲוֹן הַדּוֹר וּבְחַטַּאת אַנְשֵׁי-מֶרִי,



שֶׁהָלְכוּ בִּימֵיהֶם אֶת-אֱלֹהִים בַּחֲמַת קֶרִי,



נִשְׁמוֹתֵיהֶם הַקְּדֹשׁוֹת בַּסּוּסִים נִשְׁקָעוּ,



לִרְצוֹת עֲוֹנָם שֶׁהֶעֱווּ וּפָשָׁעוּ.



וְהֵן נִתְגַּלְגְּלוּ בְסוּסַי בִּשְׁבוּעוֹת וַחֲרָמוֹת,



כִּי אֲשִׁיבֵן לַמּוֹעֵד לְעוֹלָם הַנְּשָׁמוֹת,



וַאֲנִי עָרַבְתִּי הָהּ! לְשָׁמְרָן כְּאִישׁוֹן,



וְלַהֲשִׁיבָן בְּלִי נֶזֶק לִמְקוֹמָן הָרִאשׁוֹן,



וּלְיֶתֶר בִּטָּחוֹן, הוֹי צָרוֹת נִכְפָּלוֹת!



נָתַתִּי בָּעֵרָבוֹן עִיר גִּבְעַת-הָעֲרָלוֹת..



לָכֵן דְּעוּ רַבּוֹתַי לוֹמְדִים בַּעֲלֵי-תְרַסִים,



אִם הַגְּנֵבָה לֹא תִמָּצֵא כֻּלָּנוּ הַיּוֹם מֵתִים,



כֻּלָּנוּ הַיּוֹם מֵתִים – הוֹי פַּחַד וּמוֹרָא!



וְעִירֵנוּ תִשָּׁאֶה כִּסְדֹם וַעֲמֹרָה.



לָכֵן מוֹרַי וְרַבּוֹתַי, הֱיוּ לִבְנֵי חָיִל!



פִּשְׁטוּ עַל הַדְּרָכִים עוֹד בָּזֶה הַלָּיִל,



חַפְּשׂוּ וַחֲקָרוּ בְּכָל-עֵבֶר וָיֶרֶךְ,



וַאֲנִי בְרֹאשְׁכֶם לַנְחוֹתְכֶם הַדֶּרֶךְ…!"



וּכְרֶגַע הֶאָח! מִבְּלִי שְׁאֹל וַחֲשֹׁב,



נִרְתְּמוּ עֲגָלוֹת כְּדֵי אַרְבֶּה לָרֹב,



וּבְקִרְבָּן הֶאָח! שׁוֹטְרִים וְאַנְשֵׁי צָבָא,



עִם הַצַּדִּיק וַחֲסִידָיו הִתְיַשְּׁבוּ בָרְחָבָה,



מְסֻכְסָכִים וּמְעֹרָבִים כְּשַׁעַטְנֵז וְכִלְאָיִם,



וְעִמָּהֶם גַּם נִלְווּ מְשָׁרְתִים וְגַבָּאִים.



וּכְרֶגַע הֶאָח! נִמְהֲרוּ הָעֲגָלוֹת,



לִנְסֹעַ בְחִפָּזוֹן מִגִּבְעַת-הָעֲרָלוֹת,



קַלּוֹת כָּרוּחַ עַל הַדֶּרֶךְ הַפֹּנָה,



יְשָׁרָה כַּקָּו עִירָה קְבוּרֹנָה



שֶׁאֵלֶיהָ נָסַע ר' דָץ בְּשִׂמְחַת לְבָבוֹ,



בְּעָזְבוֹ לִפְנֵי שָׁעוֹת אֶת-אֵיתָן מוֹשָׁבוֹ,



לִדְרֹשׁ וְלַחֲקֹר אַחֲרֵי הַגְּנֵבָה,



וְלִתְפֹשׂ בַּכָּף אֶת-עוֹשֵׂה הַתּוֹעֵבָה.



וּבַלַּיְלָה הַהוּא, הוֹי שֹׁד וִילָלָה!



לִוְיָתָן מְאֹד גָּדוֹל בְּחַכָּה הָעֳלָה,



לִוְיָתָן מְאֹד גָּדוֹל אַדִּיר וְנָאוֹר,



ר' דָץ מֵחֲצַר-מָוֶת הַקָּדֹשׁ וְהַטָּהוֹר;



כִּי נִתְפַּשׂ הָהּ! מֵהַצַּדִּיק וְאַנְשֵׁי הַצָּבָא



שֶׁהִכִּירוּ בְּיָדוֹ הַסּוּסִים וְהַמֶּרְכָּבָה…



אָז קָמוּ הַחֲסִידִים כְּחֶבֶר אֲרָיוֹת,



וַיָּשִׂימוּ מִשְׁמָר עַל בַּעַל הַנִּפְלָאוֹת,



וַיִּפְשְׁטוּ עַל מֶרְכַּבְתּוֹ בִּתְלוּנָה וּמְרִיבָה,



וַיַּחַלְקוּ לוֹ מַהֲלֻמּוֹת בְּרוּחַ נְדִיבָה,



לֹא טָמְנוּ יְדֵיהֶם מֵהַאֲצִיל לוֹ בְּרָכָה,



וּמַכּוֹת-לֶחִי לוֹ הִמְטִירוּ כְּדִין וְכַהֲלָכָה.



וְאַחֲרֵימוֹ גַּם הַשּׁוֹטְרִים נִסּוּ דָבָר אֵלָיו,



וַיִּתְנַפְּלוּ עַל ר' דָץ וַיְמַשְׁשׁוּ כָּל-כֵּלָיו,



וַיִּבְדְּקוּ בְמִכְנָסָיו מַטָּה גַם מַעְלָה,



כְּדִין גּוֹנֵב סוּסִים וְשׁוֹדֵד מְאֹד נַעֲלָה;



וְהִנֵּה מָצְאוּ אֲהָהּ! בְּחַפְּשָׂם בֵּין הַכְּסָפִים:



שְׁקָלִים אַלְפַּיִם שְׁטָרוֹת מְזֻיָּפִים…



“הוֹי נָבָל בְּלִיַּעַל נוֹכֵל וְצָבוּעַ!”



– הִרְעִים רֹאשׁ הַשּׁוֹטְרִים עַל דָּץ הַפָּצוּעַ –



"הַמְעַט מִמְּךָ גָּנֹב וַעֲשׂוֹת כָּל-נְבָלָה?



כִּי תִשְׁלַח גַּם יָד בְּאוֹצַר הַמֶּמְשָׁלָה?!



אוֹי לְךָ גַנָּב עָרוּם בַּעֲרוּמִים!



הַיּוֹם נִמְלְאָה סְאָתְךָ, וְהַיּוֹם יוֹם שִׁלּוּמִים!



הַיּוֹם יוֹם שִׁלּוּמִים לְךָ אִישׁ עָוֶל!



שׁוֹטְרַי! קִרְבוּ נָא לְאָסְרֵהוּ בַכָּבֶל.



וּכְרֶגַע הָהּ! שׁוֹטְרִים מְזָרֵי אֵימָה,



עַל ר' דָץ בַּר שִׁלְשֹׁם קָפְצוּ בְחֵמָה,



וַיַּכְתִּירֻהוּ מִסָּבִיב כַּחֲרֻלִּים וָחָדֶק,



וַיָּסִירוּ מֵעַל רֹאשׁוֹ הַיַּרְמֻלְקָה וְהַסְפָּדֶק;



גַּם פָּשְׁטוּ מֵעָלָיו בְּקֶצֶף מְאֹד נוֹרָא,



הַזֻּפִּיצָה, הַדֶּלְיָה וְהָאָפַּנִּיצֶה הַטְּהוֹרָה;



וְתַחְתָּם אֲהָהּ! בְּאֹמֶץ יָדָיִם,



עַל בְּשָׂרוֹ הֵבִיאוּ זִקִּים וּנְחֻשְׁתָּיִם,



וַיַּאַסְרֻהוּ לָעֲגָלָה כְּפַר בֶּן-בָּקָר,



לְעֵינֵי הַחֲסִידִים וְר' לֶמֶךְ הַיָּקָר.



"אוֹי לָכֶם טְמֵאִים וּרְשָׁעִים קְבוּרִים!



אוֹי לָכֶם עֲרֵלִים וְזֵדִים אֲרוּרִים!



אוֹי לָכֶם מְחַבְּלִים וּפֹרְצֵי פָרֶץ!"



– נִשְׁמַע קוֹל ר' דָץ כְּאוֹב מֵאָרֶץ –



"עַד אָן מִדְּמֵי נָקִי שָׂבְעָה לֹא תֵדָעוּ?



וְעַל צַדִּיק עוֹבֵר בֶּטַח חִנָּם תִּתְמַלָּאוּ!



דְּעוּ נָא רַבּוֹתַי שׁוֹטְרִים וּמְמֻנִּים!



כִּי הַשְּׁטָרוֹת וְהַסּוּסִים מִלֶּמֶךְ לִי נְתוּנִים;



גַּם הַמֶּרְכָּבָה הָהּ! הַשָּׁמַיִם וְהָאֲדָמָה,



יְעִידוּן וְיַגִּידוּן כִּי מִמֶּנוּ לִי נִתָּנָה.



דְּעוּ נָא וּרְאוּ כִּי חַף אֲנִי מִפֶּשַׁע,



וְרַק אוֹיְבִי עָלַי יְעוֹלֵל עֲלִילוֹת בְּרֶשַׁע".



וְאַחַר אֶת-פָּנָיו גַּם אֶל ר' לֶמֶךְ שָׂם,



וַיִּתֵּן בְּקֹלוֹ קוֹל אַדִּיר וָרָם:



"וַי לָךְ לֶמֶךְ חֻטְרָא דִּי גַאֲוֻתָא,



"דִּי יְהַבְתְּ חֵילָא וְתָקְפָא לְרוּחַ מְסָאֲבוּתָא;



"אַף וַי לְמַשְׁרִיתָךְ חִוְיַן דִּי בִישִׁין,



"דִּי רָדְפִין עַל מַגָּנָא עִירִין קַדִּישִׁין,



"וַי לָךְ מִבֵּיתָא, וַי לְהוֹן מִבָּרָא!



"וַי וַי לְכוֹן כַּלְבַיָּא וְחַיוַת בָּרָא!



"עוֹד תִּטְרְדוּן מִן אֲנָשָׁא וּמִתְּרֵין עָלְמַיָּא



“עַד דִּי תִגְדְּעוּן רַשִּׁיעִין דִּי אִיתַי אֱלָהּ בִּשְׁמַיָּא!”



אַךְ לִדְבָרָיו הָהּ! לֹא הָיָה שׁוֹמֵעַ,



ר' לֶמֶךְ עָשָׂה שְׂפָמוֹ וַיְהִי כְּלֹא-יוֹדֵעַ,



וְהַשּׁוֹטְרִים הָהּ! עוֹד מִלְּאוּ פִיהֶם צְחֹק,



וַיְחַזְּקוּ עוֹד מוֹסָרָיו כַּמִּשְׁפָּט וְכַחֹק.



וּבַלַּיְלָה הַהוּא בְּלִוְיַת כָּל-הָעֲגָלוֹת,



הוּבָא ר' דָץ עִירָה גִּבְעַת-הָעֲרָלוֹת,



נָבוּךְ וְנִבְהָל וְאָסוּר בִּנְחֻשְׁתָּיִם;



גַּם גַּלְמוּד וְנֶעֱזָב מִגַּבָּאָיו הַשְּׁנָיִם,



– כִּי נִמְלְטוּ הָהּ מִידֵי אַנְשֵׁי הֶחָיִל



בְּתוֹךְ הַמְּהוּמָה וְהַמְּבוּכָה בְּחֶשְׁכַת הַלָּיִל –



וַיּוּבָא בְּבוֹר עָמֹק וַיִּסָּגֵר בַּחֲשֵׁכָה;



וְהָעִיר גִּבְעַת-הָעֲרָלוֹת צָהֲלָה וְשָׂמֵחָה. –


ט


אוֹי לָכֶם צַדִּיקִים נְחָשִׁים צִפְעוֹנִים,



אוֹי לָכֶם פְּתָיִם מִתְיַמְּרִים לִנְבוֹנִים,



אוֹי לָכֶם נוֹכְלִים מְעַקְּשֵׁי הַיְּשָׁרָה!



מַה-טָּפַשׁ לִבְּכֶם וְדַעְתְּכֶם מַה-קְּצָרָה?!



הֵן כֻּלְּכֶם אַנְשֵׁי מוֹפֵת, כֻּלְּכֶם מוֹשִׁיעִים,



כֻּלְּכֶם מַלְאֲכֵי עֶלְיוֹן, כֻּלְּכֶם נְבִיאִים,



כֻּלְּכֶם צַדִּיקִים, כֻּלְּכֶם אַנְשֵׁי פְלָאוֹת,



וְעֵינֵיכֶם הָהּ! לֹא תִרְאֶינָה אֶת-הַבָּאוֹת.



נָקֵל לָכֶם לִגְנֹב לְזַיֵּף וּלְשַׁקֵּר,



לַהֲרֹג וּלְאַבֵּד, לִנְתֹשׁ וּלְעַקֵּר,



לְהוֹנוֹת וּלְרַמּוֹת וּלְזַנֵּב נֶחֱשָׁלִים,



וְלַחֲשֹׁב הֲמוֹן בְּרִיּוֹת כִּבְהֵמוֹת וַעֲגָלִים;



וְאֵידְכֶם מִמִּסְתָּרִים כַּסּוּפָה כִּי יִתְנָעֵר,



טֶרֶם בֹּאוֹ הָהּ! לֹא תֵדְעוּ לְהִזָּהֵר.



אוֹי לְךָ דָץ בֶּן-קְדֹשִׁים וּטְהוֹרִים,



אוֹי לְךָ אַבִּיר לֵב אַמִּיץ בַּגִּבּוֹרִים



אוֹי לְךָ גְּלוּי עֵינַיִם וְאוֹי לְנַפְשֶׁךָ,



אַיֵּה חָכְמָתְךָ וְאֵיפֹה כֹחַ יָדֶיךָ?



אֵיפֹה כֹחַ יָדֶיךָ? הוֹי גִּבּוֹר מִלְחָמָה!



וּמִי שָׁדַד בִּינָתְךָ בְּיוֹם שִׁלֵּם וּנְקָמָה?



הֵן חֵילְךָ רַב וְכֹחֲךָ מְאֹד נַעֲלָה,



לְהַטּוֹת לְחֶפְצְךָ כָּל-שָׂרֵי מַעְלָה,



לִפְקוֹד וּלְצַוֹּת עַל מַלְאֲכֵי צְבָאוֹת,



וּלְהַרְגִּיז בְּקוֹלְךָ הָרִים וּגְבָעוֹת;



לַעֲבֹד בִּשְׂרָפִים וּבְאֶרְאֶלֵּי רָקִיעַ,



וְהָאֹתִיּוֹת לְאָחוֹר מֵרֹאשׁ לְהוֹדִיעַ,



וְאַיֵּה חָכְמָתְךָ? – הוֹי בַּעַל נִפְלָאוֹת!



כִּי כְּאַחַד הָאָדָם לֹא רָאִיתָ אֶת-הַבָּאוֹת.



גַּם אוֹי לְךָ לֶמֶךְ צַדִּיק פָּרוּעַ,



נוֹכֵל וּבְלִיַּעַל וְעַיִט צָבוּעַ!



הֵן עָרוּם גַּם אַתָּה כַּשּׁוּעָל בַּיָּעַר,



לְהוֹנוֹת וּלְהַשְׁלוֹת זָקֵן וָנָעַר,



לִמְשֹׁךְ בְּחַכָּה בָּחוּר וּבְתוּלָה,



וּלְהַתְעוֹת בַּתֹּהוּ יִשְׂרָאֵל עַם סְגֻלָּה;



וְאֵיךְ כֹּה עָשִׂיתָ תּוֹעֵבוֹת נוֹרָאוֹת,



וְעֵינֶיךָ גַּם אַתָּה לֹא רָאוּ אֶת-הַבָּאוֹת?



אֵיךְ לֹא הִתְבּוֹנַנְתָּ הוֹי כְּבַד פֶּשַׁע!



כִּי נִתְכְּנוּ גַּם לְאוֹיִבְךָ עֲלִילוֹת בְּרֶשַׁע?



אָכֵן פְּתָיִם אַתֶּם צַדִּיקִים וּקְדֹשִׁים!



כָּל-חָכְמַתְכֶם וְדַעְתְּכֶם אַךְ לִפְרוֹת כְּפַרְעֹשִׁים,



לֶאֱכֹל וְלִשְׁתּוֹת וְלִצְבֹּר הֲמוֹן שְׁקָלִים,



וּלְכַסּוֹת עֵין הָאָרֶץ בַּהֲבֵל הֲבָלִים;



וּבְהַרְבּוֹתְכֶם בַּמֶּחְשָׁךְ תֹּעֵבוֹת וְשַׁעֲרוּרִיּוֹת,



לֹא תִרְאוּ וְלֹא תָבִינוּ, גַּם תִּפְּלוּ בַּצָּרָה!



וְשִׁמְעוּ שֵׁנִית רַבּוֹתַי מַעֲשֶׂה שֶׁקָּרָה:


י


כְּמִשְׁלשׁ יְרָחִים אַחֲרֵי בֹא אֶל הַבּוֹר,



ר' דָץ בַּר שִׁלְשֹׁם צַדִּיק הַדּוֹר;



וּכְבָר עַל כִּסֵא קָדְשׁוֹ כְּמֶלֶךְ בִּמְסִבּוֹ,



הִתְיַשֵּׁב הַצַדִּיק וַהֲמוֹן פְּתָיִם מִסְּבִיבוֹ,



מַגִּישִׁים לוֹ מִנְחָה כֹּפֶר וְשַׁלְמֹנִים,



כְּחֻקָּם וּמִשְׁפָּטָם בַּיָּמִים הָרִאשׁוֹנִים;



כְּבָר נִשְׁכַּח ר' דָץ וּכְבָר זֵכֶר לוֹ אָיִן;



וְהַחֲסִידִים הַפְּרָאִים הֲלוּמֵי הַיָּיִן,



כְּבָר מִתְהַלְלִים בָּאֱלִילָם וּפְנֵיהֶם יִנְהָרוּ,



וּכְבָר כְּמִקֶּדֶם הֲמוֹנִים יִנְהָרוּ,



בְּרֶכֶב וּפָרָשִׁים בְּכִרְכָּרוֹת וַעֲגָלוֹת,



אֶל כִּסֵּא הַמּוֹשִׁיעַ בְּגִבְעַת-הָעֲרָלוֹת,



כְּבָר שָׁבַת נוֹגֵשׂ אָפַס הַמֵּץ!



וּכְבָר נִסִּים וְנִפְלָאוֹת וּמוֹפְתִים לְאֵין קֵץ,



יִתֵּן ר' לֶמֶךְ בָּאָרֶץ וּבַשָּׁמָיִם,



יַהֲפֹךְ בֹּקֶר לְעֶרֶב וְעֶרֶב לְצָהֳרָיִם;



כְּבָר יָרִיעַ וְיַצְרִיחַ עַל שֵׁדִים וּמַזִּיקִים,



וּכְבָר יֶאֱסֹר מְחַבְּלִים בִּכְבָלִים וָאזִקִּים,



כְּבָר יְרַצֵּץ וִיקַצֵּץ פֵּאוֹת וּכְנָפָיִם,



מִכְּרוּבִים וְאוֹפַנִּים וּמַלְאֲכֵי שָׁמָיִם;



כְּבָר יִכְתֹּב קָמֵעוֹת וְיִפְרַע פְּרָעוֹת,



בִּקְלִפּוֹת אֻמְלָלוֹת וּנְשָׁמוֹת עַרְטִלָּאוֹת.



וְהִנֵּה פִּתְאֹם הָהּ! בְּעֶצֶם יוֹם גְּאוֹנוֹ,



נוֹקַשׁ ר' לֶמֶךְ וַיִּלָּכֶד בַּעֲוֹנוֹ.


* *

*


בְּעֶצֶם יוֹם בָּהִיר עֵת עַל כִּסֵּא כְבֹדוֹ



יָשַׁב הַצַדִּיק וּסְבִיבָיו מְתֵי סוֹדוֹ,



עוֹשִׂים מְלַאכְתָּם עַל שֻׁלְחַן הַפִּדְיֹנוֹת,



שׁוֹקְלִים וְסוֹפְרִים וּמְחַשְּׁבִים חֶשְׁבֹּנוֹת,



וְהִנֵּה פִתְאֹם אִישׁ זָקֵן, הוֹדוֹ כַּזָּיִת.



בְּרַעַשׁ גָּדוֹל וְנוֹרָא פָּרַץ אֶל הַבָּיִת,



וַיִּתֵּן קוֹל גָּדוֹל: "רַבִּי, קְדֹשִׁי חֲמֹל!



חוּשָׁה, הַצִילָה, פְּדֵה מֵרֶדֶת שְׁאוֹל!



הָהּ… בִּתִּי… בִּתִּי… הַתַּמָּה כַּיּוֹנָה,



גֹּוַעַת… לְעֵינַי… וְהִיא בַּמִּסְדְּרֹנָה!



רַבִּי, קְדֹשִׁי, צַדִּיקִי, שָׂא תְחִנָּה וּתְפִלָּה!



כָּל-הוֹן… בֵּיתִי… אֶתֵּן… אַךְ אָנָּא הַצִּילָה!"



וּכְרֶגַע הָהּ! לְשׁוֹנוֹ בוֹ נֶאֱלָמָה



וַיִּפֹּל אַרְצָה כְּגוּף בְּלִי נְשָׁמָה.



“אַל לְאִישׁ חָסִיד הִתְרַפּוֹת בְּיוֹם צָרָה!”



– קָרָא הַצַדִּיק בִּקְדֻשָּׁה וּבְטָהֳרָה –



"אִם בָּאתָ בְּצֵל בֵּיתִי דַע לְךָ, דַּע,



כִּי בִתְּךָ תֵרָפֵא וְלֹא יִקְרְךָ רַע;



אַךְ קוּם וְהִתְעוֹרֵר – לָמָּה נָפְלוּ פָנֶיךָ?



הַאֲמֵן בְּרַבְּךָ וְתִוָּשַׁע אַתָּה וּבִתֶּךָ!…"



"בִּתִּי תִּוָּשַׁע? – הֶאָח רַבִּי וּקְדֹשִׁי!



כּל-הוֹנִי וַעֲמָלִי, חֵי נַפְשִׁי וְרֹאשִׁי!



לְךָ אַךְ לְךָ… מוֹשִׁיעִי אֶתֵּנָה"



– קָרָא הַזָּקֵן בְּמַהֲרוֹ הַמִּסְדְּרֹנָה.



וּכְרֶגַע אֲהָהּ! עַלְמָה יְפַת תֹּאַר,



זַכָּה וּבָרָה כַּלְּבָנָה לָטֹהַר,



קָפְצָה הַבַּיְתָה דְּחוּפָה וּמְבֹהָלָה,



וַתִּקְרַב אֶל הַשֻּׁלְחָן וַתַּהַפְכֵהוּ לְמָעְלָה;



וַתִּפֹּל אֲחֹרַנִּית כִּגְוִיָה בְּלִי רוּחַ



וַתְּפַרְפֵּר יָד וָרֶגֶל לְכָל עֵבֶר וָרוּחַ.



וְאַחַר, בְּקוֹל נוּגִים כְּקוֹל בַּת-הַיַּעֲנָה,



זִמְּרָה פִּיּוּטִים סְלִיחוֹת וְהוֹשַׁעְנָא,



וַתַּשְׁמַע מִגְּרוֹנָה בִּילֵל וְתַמְרוּרִים,



עַל-חֵטְא, אָשַׁמְנוּ וּקְרוֹבֵץ לְפוּרִים,



וַתְּסַכְסֵךְ פִּיּוּטִים בְּשִׂיחוֹת גַּנָּבִים,



וּסְלִיחוֹת וְקִינוֹת עִם שִׁירֵי עֲגָבִים.



"הוֹי – קָרָא הַזָּקֵן – רַבִּי נָא חוּסָה!



הַצִּילָה! הַצִּילָה! כִּי פֹה אָמוּתָה!



הַצִּילָה! הַצִּילָה! הוֹי בִּתִּי הַיְשָׁרָה!



הֵא כֶסֶף הֵא זָהָב וְכָל-סְגֻלָּה יְקָרָה".



"הַס! – קָרָא הַצַדִּיק – הַס קְטַן אֲמָנָה!



אִם כֹּה תֵתַע בְּשֶׂפֶק לֹא אֶקְחָה כָּל-מָנָה,



הַיּוֹם תִּוָּשַׁע בִּתְּךָ, הֱיֵה סָמוּךְ בָּטוּחַ!



וְעוֹד תִּרְאֶה בְּחוּשׁ בְּגָרְשִׁי אֶת-הָרוּחַ…"



"הוֹי צַדִּיק צָבוּעַ חוֹטֵא וּמִשְׁתַּגֵּעַ!



הוֹי שׁוֹגֵג וּמַשְׁגֶּה, אֱוִיל וַחֲסַר-דֵּעַ!



הוֹי מוֹרֶה מַתְעֶה עוֹבֵד עַכּוּם בְּטַהֲרָה!"



– נִשְׁמַע פִּתְאֹם קוֹל אַמִּיץ מִגְּרוֹן הַנַּעֲרָה –



"הַאוֹתִי תְגָרֵשׁ? – הוֹי שָׁגִיתָ בָרֹאֶה!



בּוּז לְךָ רֵיקָה לַעַג לְךָ תֹעֶה!



בּוּז לְךָ, לַעַג לְךָ, כֶּלֶב נוֹבֵחַ!



חֳלִי-מֵעַיִם בְּבִטְנְךָ בּוֹגֵד וּמְרַצֵּחַ;



הַשְׁמַעְתֶּם אַחַי? הַשְׁמַעְתֶּם שָׁמֹעַ?



צַדִּיק מִתְקַדֵּשׁ, עָרוּם לָרֹעַ,



צַדִּיק מְנֻוָּל… צַדִּיק מֻכֵּה-שְׁחִין,



יֹאמַר לְגָרְשֵׁנִי בְּלִי כָל-מִשְׁפָּט וָדִין.



לֹא, לֹא, כֶּלֶב, אַךְ שָׁוְא תִּתְהַלֵּלָה!



גַּם עַל גָּחוֹן כִּי תֵלֵךְ וְעָפָר תֹּאכֵלָה,



אַף עַל אַרְבַּע כִּי תִזְחַל עָיֵף וּשְׁחֹחַ,



מְקוֹמִי מֵהַנַּעֲרָה לֹא אֶזְנַח זָנֹחַ!"



“רָשָׁע מְרֻשָּׁע, מַמְזֵר וְזֵד אָרוּר!”



– קָרָא הַצַדִּיק בְּקֶצֶף וּפָנָיו קִבְּצוּ פָארוּר –



“רָשָׁע מְרֻשָּׁע, חֲ.. חֲ.. חֲצוּפָה וּמִרְשַׁעַת!”



– הוֹסִיף קְרֹא בְגָרוֹן כִּמְטֹרָף מִדַּעַת –



"הַיּוֹם תִּרְאֶה גָּדְלִי, הַיּוֹם הַכֵּר תַּכִּירֵנִי!



הַיּוֹם אוֹרְךָ נָבָל כִּי אַמִּיץ הִנֵּנִי,



הַיּוֹם הַיּוֹם כֶּלֶב תְּגֹרַשׁ מִזֶּה כָּלָה,



וְתִרְאֶה אֶפִּיקוֹרוֹס כִּי יָדִי זֹאת פָּעָלָה!



הָבוּ גַּבָּאַי, מַהֲרוּ סֵפֶר תְּהִלִּים!



הַגִּישׁוּ יַיִן-שָׂרָף, טַלִּית וּתְפִלִּים,



הַגִּישׁוּ רְצוּעָה, הָבִיאוּ חֲגֹרָה!



הָבוּ קִטֶּל וְלוּלָב הָבוּ סֵפֶר תּוֹרָה,



הָבוּ גָפְרִית וָמֶלַח, הָבוּ לְבֹנָה זַכָּה,



הָבוּ עֲרָבָה חֲבוּטָה הָבוּ תְּפִלָּה זַכָּה;



הָבוּ חֶלְבְּנָה, הָבוּ שׁוֹפָר, הָבוּ… גִּלָּיוֹן!…



וּפֹה, פֹּה עִם הַנָּבָל אֶעֱשֶׂה כִּלָּיוֹן".



"הוֹשַׁעְנָא! הוֹשַׁעְנָא! רַחְמָנָא מִן שְׁמַיָּא!



אֲנִי-וָהוֹ הוֹשִׁיעָה לְעָלַם עָלְמַיָּא4



אֲנִי-וָהוֹ הוֹשִׁיעָה רַחְמָנָא שְׁזִיב וּפְרֹק!"



– הִתְפָּרֵץ מֵהַנַּעֲרָה קוֹל פָּרוּעַ בִּצְחֹק –



"מָה אֶעֱשֶׂה אַלְלַי, מָה אֶעֱשֶׂה אֲנִי גֹוֵעַ?



מָה אֶעֱשֶׂה הַיּוֹם אֻמְלָל, לְמִי אֵבְךְּ וַאֲשַׁוֵּעַ?



אָנָה אֲנִי בָא? הָהּ רוּחִי חֻבָּלָה!



סַמְּכוּנִי! סַמְּכוּנִי! כִּי נַפְשִׁי נִבְהָלָה,



הָהּ יַיִן-שָׂרָף, הָהּ תְּקִיעָה וּתְרוּעָה!



הָהּ קִטֶּל וַעֲרָבָה, לוּלָב וּרְצוּעָה!



הָהּ גָּפְרִית וָמֶלַח, הָהּ תְּפִלָה זַכָּה!



יַרְעִישׁוּ כָּל-לִבִּי וִישִׁיבוּנִי עַד דַּכָּא.



חוּסָה נָּא לֶמֶךְ, חוּסָה צַדִּיק תָּמִים!



בַּחֲמָתְךָ אַל תְּיַסְּרֵנִי פֶּן אֶגְוַע לְעוֹלָמִים,



חוּסָה נָּא רַחֶם-נָּא אַל נָא בְּאַפֶּךָ!



פֹּה פֹה אָמוּתָה, פֹּה אֶגְוַע לְעֵינֶיךָ".



"דֶּרֶךְ-אֶרֶץ אֲרוּרָה, דֹם טִפָּה סְרוּחָה!



– הִרְעִים קוֹלוֹ הַצַּדִּיק בְּרַעַשׁ וְחֵמָה שְׁפוּכָה –



“דֶּרֶךְ-אֶרֶץ מְשֻׁמָּד, דֹּם זֹנָה וַחֲלָלָה!”



– הוֹסִיף תֵּת קֹלוֹ, וְכָל הָעֵדָה נִבְהָלָה –



"לְמִי תִּשְׂחַק תִּלְעַג? הוֹי בֶּן הַנִּדָּה!



עַל מִי תַּרְחִיב פֶּה? נָבָל נָא הַגִּידָה!



עַל מִי וָמִי מַמְזֵר תִּשְׁפֹּךְ לַעַג כַּמַּיִם?



וְתַהֲפֹךְ בְּזָדוֹן דִּבְרֵי אֱלֹהִים חַיִּים.



אוֹי לְךָ רָשָׁע, אוֹי לְךָ פּוֹשֵׁעַ!



אַף עַל פִּתְחוֹ שֶׁל גֵיהִנֹּם תְּמָאֵן הִכָּנֵעַ;



תְּמָאֵן הִכָּנֵעַ – וְעוֹד תַּרְבֶּה הַשַּׁעֲרוּרִיָּה,



לְחַלֵּל כָּל-קֹדֶשׁ בְּעָרְמָה וּבִצְדִיָּה.



דַּע לְךָ רָשָׁע, דְּעִי לָךְ אֲרוּרָה!



כִּי לְהָרַע וּלְהַשְׁחִית לִי עֹז וּגְבוּרָה,



גַּם לְהֵטִיב וּלְחוֹנֵן כֹּחִי עִמָּדִי



וּבְחַר לְךָ אַחַת טֶרֶם אָחֵל מַעְבָּדִי.



בְּחַר לְךָ אַחַת, אַל תַּעֲמֵק שְׁאָלָה!



הַשְׁמִיעֵנִי חֶפְצְךָ כִּי גַּם הוֹעֵל אוּכָלָה".



"הוֹי צַדִּיק מְשֻׁגָּע מַשְׁגֶּה וּמְתַעְתֵּעַ!



הוֹי קָדֹשׁ מִתְקַדֵּשׁ, חוֹטֵא וּפוֹשֵׁעַ



הוֹי טָמֵא מִתַּמָּם עָרוּם וְצָבוּעַ!"



– הִתְפָּרֵץ מֵהַנַּעֲרָה קוֹל אָיֹם פָּרוּעַ –



"עַד אָן בְּמִשְׁפָּטִים לִרְגָעִים תְּאַיְּמֵנִי?



עַד אָן בְּלַהַג הַרְבֵּה לָרִיק תְּיַגְּעֵנִי?



עַד אָן בַּהֲבָלִים תִּפְרַע בִּי פְּרָעוֹת?



וּבְקוֹל נָגִיד וּמְצַוֶּה עָלַי תַּפִּיל בַּלָּהוֹת.



דַּע לְךָ צַדִּיק, קָדֹשׁ עַז-פָּנִים,



דַּע לְךָ נוֹכֵל וּמְגָרֵה מְדָנִים,



דַּע לְךָ שַׁקְרָן מַרְבֵּה תַעְתּוּעִים,



כְּבָר נַסֹּה נִסּוּנִי צַדִּיקִים וּצְבוּעִים,



כְּבָר רֹגֶז שָׂבַעְתִּי מִנְּחָשִׁים צִפְעֹנִים,



כְּבָר הִשְׁקוּנִי מֵי-רוֹשׁ רַבָּנִים גְּאֵי-יוֹנִים,



כְּבָר נִדּוּנִי הֶחֱרִימוּנִי וַיְשִׂימוּנִי לְקֶלֶס,



צַדִּיקִים בַּעֲלֵי-שֵׁמוֹת וְרַבֵּי אִוֶּלֶת,



אַךְ דַּע כִּי כֻלָּמוֹ כָּשְׁלוּ וַיִּלְאוּ,



וּלְשָׂעֲרֵנִי מִמְּקוֹמִי מְאוּם לֹא הִשְׂכִּילוּ,



מְאוּם לֹא הִשְׂכִּילוּ – וְלַשָּׁוְא אַךְ יָגָעוּ!



וְכָמֹךָ גַּם הֵמָּה אַךָ קָלוֹן מָצָאוּ.



כִּי אָמְנָה כֶּבֶד-עָוֹן אֻמְלָל הִנֵּנִי!



וּפוֹשֵׁעַ מִבֶּטֶן כָּמֹנִי אֵין שֵׁנִי,



רַבּוֹת חָטָאתִי, רַבּוֹת עָוִיתִי!



בְּגוּף בֵּן אֲדָמָה עֵת חָיֹה חָיִיתִי,



וּבְכֵן נִלְאוּ צַדִּיקִים וְדַעְתָּם הִתְבַּלָּעָה



וּלְהוֹעִילֵנִי מֵהַוָּתִי יַד אִישׁ לֹא מָצָאָה".



"בִּתִּי – קָרָא הַצַדִּיק – בִּתִּי יַקִּירִי![^6]



חֶפְצֵךְ הַגִּידִי וְאַל מִלִים תַּכְבִּירִי,



הֵן אֵדַע בַּעֲלִיל כִּי הוֹשִׁיעֵךְ אוּכָלָה,



וְלָמָּה זֶה בְּנִי תְּדַבֵּר נְבָלָה?



קוּמִי הִתְעוֹרְרִי נְשָׁמָה עֲלוּבָה!



הִתְנַעֲרִי מֵעָפָר וְחִזְרִי בִּתְשׁוּבָה,



הֵן לְהַצִּילֵךְ וּלְתַקְּנֵךְ יָדִי הַיּוֹם פְּתוּחָה;



וְאַתְּ גַּלִּי עַל עֲוֹנֵךְ וְתִמְצְאִי מְנוּחָה.



"אוֹיָה! – קָרָא הָרוּחַ – בִּלְבַּלְתָּ אֶת-מֹחִי!



פּוֹרַרְתָּ כָּל-לִבִּי, הִכְשַׁלְתָּ בִּי כֹחִי,



מָחַצְתָּ רַקָּתִי וְעַד דַּכָּא הֲשִׁיבֹתָנִי.



וּשְׁמַע נָא בִקְצָרָה מֶה חָדֵל אָנִי!

רַב מוֹרֵה צֶדֶק נוֹדָע לְתִפְאֶרֶת,



אֲנִי אֻמְלָל הָיִיתִי בְּעִיר קְטַנָּה “עֲצֶרֶת”…



עֵת בְּבֵית חֹמֶר בַּתֵּבֵל חָיִיתִי



וְשֵׁם “מְאוֹר-הַגּוֹלָה” בְּיִשְׂרָאֵל לִי קָנִיתִי:



וְאָמְנָה שֵׁם, זֶה נָאֶה לִי הָיָה,



כִּי חָרִיף הָיִיתִי וּבָקִי בְּהַטָּיָה:



מִצַּד אֶל צַד אַחֵר בְּמִשְׁנָה וּבִגְמָרָה.



וְהָרִים עָקַרְתִּי בְּפִלְפּוּל וּסְבָרָא,



הִצְלַחְתִּי בְלִמּוּדִי וָאַעַשׂ פְּרִי הִלּוּלִים



בְּחִדּוּשִׁים וְהַלָכוֹת בְּשִׁטַּת דַּם בְּתוּלִים



וּבְכָל טָמִיר וְנֶעְלָם הִרְבֵּיתִי וִכּוּחַ



וְדִינִים חִדַּשְׁתִּי בְּפֶתַח פָּתוּחַ,



גַּם בְּכַוְרָא וּמַרְגְּנִיתָא יָדִי גְּדֹלוֹת עָשָׂתָה!



כִּי הָרִים תָּלִיתִי בְּשַׂעֲרוֹת נַעֲרָה הַמְאֹרָשָׂה,



וּבְכֵן “מְאוֹר-הַגּוֹלָה” יִשְׂרָאֵל קְרָאָנִי



כִּי שְׁמִי נָאֶה לִי וְלִשְׁמִי נָאֶה אֲנִי



גַּם עָשִׁיר הָיִיתִי מְטֻפָּל בִּכְסָפִים



וְכַבִּיר מָצְאָה יָדִי מֵאוֹת וַאֲלָפִים,



כִּי מֶכֶס הַבָּשָׂר וּבֵית מֶרְחַץ עֲדָתִי!



בִּרְשׁוּת רָאשֵׁי הַקָּהָל לָקַחְתִּי לְמִחְיָתִי,



וּמֵהֶם לַחְמִי הֵבֵאתִי גַּם עֹשֶׁר עָשִׂיתִי,



מְאוּם לֹא חָסַרְתִּי וּבַטּוֹבָה רָאִיתִי,



בַּטּוֹבָה רָאִיתִי – עָבִיתִי שָׁמַנְתִּי!



וּבְעֵינֵי בְּנִי עִירִי מְאֹד נֵחַנְתִּי,



לֹא יָדַעְתִּי כָּל-חֹסֶר שָׁלַוְתִּי בִמְכוֹנִי,



אַךְ פִּתְאֹם הָהּ! נִלְכַּדְתִּי בַּעֲוֹנִי:



רֵעַ דָּבֵק מֵאָח – רַב גָּדוֹל בְּיִשְׂרָאֵל –



הָיָה לִי בִקְבוּרָה וּשְׁמוֹ ר' סַמָּאֵל!



וְהָיָה הָאִישׁ הַהוּא תָּם יָשָׁר וְנָקִי,



סִינַי וְעוֹקֵר הָרִים חָרִיף וּבָקִי



כָּל-יָמָיו בַּתּוֹרָה עָשָׂה רַק חָיִל,



וּמִגִּרְסָא לֹא פָסַק יוֹמָם וָלָיִל,



קָרָא, לָמַד וְשָׁנָה בְּשַׁ"ס וּמִדְרָשִׁים



וּבְתוֹרָתוֹ הָיָה רָצוּי לֵאלֹהִים וַאֲנָשִׁים.



וַיְהִי הַיּוֹם, הָהּ יוֹם יְלָלוֹת!



עָבַר ר' סַמָּאֵל דֶּרֶךְ גִּבְעַת-הָעֲרָלוֹת,



– בְּלֶכְתּוֹ לְמַסָּעָיו מִמְּדִינָה לִמְדִינָה



לְהִשְׁתַּתֵּף בַּגּוֹלָה בְּצַּעְרָהּ שֶׁל שְׁכִינָה –



וּבְבֹאוֹ הָעִירָה הָהּ! תְּקָפַתּוּ קַדַּחַת,



וַיַחַל אֶת-חָלְיוֹ וַיָּמָת לַשַּׁחַת.



וּכְשָׁמְעִי הָאָסוֹן נְפוּגֹתִי, נִדְכֵּיתִי



וּמִקֶּרֶב לֵב עָמֹק בָּכִיתִי לְמֵתִי,



קָרַעְתִּי אֶת-בְּגָדַי וּבֶעָפָר הִתְפַּלָּשְׁתִּי,



וְאֶת-פָּנַי הָהּ! בְּקַרְקַע כִּבָּשְׁתִּי,



רָקַדְתִּי כַּמְּשֻׁגָּע דָּלַגְתִּי כָּאָיִל,



וּבְמַר נַפְשִׁי צָעַקְתִּי יוֹמָם וָלָיִל,



שִׁוַּעְתִּי, נֶאֱנַקְתִּי בְּכָל-עֹז וָכֹחַ,



קִפַּצְתִּי, כִּרְכַּרְתִּי וְלֹא מָצָאתִי מָנֹחַ.



וּלְמַעַן יִכַּבֵד הַמֵּת הָהּ, צוּקָה וְצָרָה!



בְּבֵית הַכְּנֶסֶת הַגָּדוֹל הִקְדַּשְׁתִּי עֲצָרָה,



וּבְנֶפֶשׁ נְכֵאָה הַבָּמָתָה עָלִיתִי,



הִסְפַּדְתִּיו כַּהֲלָכָה וּמְאֹד בָּכִיתִי



קָרָאתִי בִּשְׁמוֹתָיו: חָרִיף, שָׁנוּן וּבָקִי



תָּם, אָהוּב לַבְּרִיוֹת קָדֹשׁ וְנָקִי



עַמּוּד הַיְמָנִי, פְּרִי קֹדֶשׁ הִלּוּלִים,



וְעַל כָּל-שֵׁם מֵהַשֵּׁמוֹת הִרְבֵּיתִי פִלְפּוּלִים;



וּבַאֲרוֹן הַקֹּדֶשׁ הָהּ בְּרוּחַ נִשְׁבָּרָה,



הִתְנַפַּלְתִּי וָאֶקְרָא: הֲזוֹ תוֹרָה וְזֶה שְׂכָרָהּ?



וַיָּנֻעוּ אַמּוֹת הַסִּפִּים לְקוֹל צַעֲקָתִי,



וְקוֹל בֹּכִים הֵרִימוּ כָּל-אַנְשֵׁי עֲדָתִי,



וְאַחַר, שִׁבְחֵי הַמֵּת לְאַחַד אַחַד,



סָפַרְתִּי לִשְׁמוֹתָיו וְכִלַּלְתִּים יַחַד



וְהַשֵׁם הַגָּדוֹל “חָשׁוּב” בְּנוֹטָרֵיקוֹן הוֹצֵאתִי,



מִשְּׁלשָׁה הָרִאשׁוֹנִים בְּסָפְדִי לְמֵתִי,



וּמִמֶּנּוּ מִסְפֵּד חָדָשׁ לַעֲדָתִי עָרַכְתִּי,



וְגַם הַשֵּׁם ר' סַמָּאֵל מִשְׁמוֹתָיו הוֹכַחְתִּי



הַכֹּל עָשִׂיתִי יָפֶה-אַךְ הָהּ דֵּרָאוֹן!!



כִּי שָׁכַחְתִּי לְהַסְפִּידוֹ בְּשֵׁם רַב וְגָאוֹן…



וְהִנֵּה בָא אֵלַי רֵעִי בַּחֲלוֹם הַלַּיְלָה,



וַיּוֹעִידֵנִי לַמִּשְׁפָּט לִפְנֵי בּ"ד שֶׁל מַעְלָה,



וּבְצַוָּארִי אֲחָזַנִי כְּמֶלֶךְ בַּלָּהוֹת,



וַיּוֹלִיכֵנִי בַּחֲמָתוֹ הָרִים וּבְקָעוֹת,



וַיִּסְחָבֵנִי כְּפֶגֶר שְׁפָיִים וְיַמִּים,



גְּבָעוֹת וּנְהָרוֹת בִּצּוֹת וַאֲגַמִּים,



עֲדֵי רָאִיתִי הָהּ כִּי שָׁמַיְמָה הֶעֱלָנִי,



וּבְחֶדֶר רְחַב יָדַיִם עָרֹם הִצִּיגָנִי,



וְשָׁם לַמִּשְׁפָּט הָהּ! יֵשְׁבוּ כִּסָּאוֹת,



זְקֵנִים אֲיֻמִּים, פְּנֵיהֶם פְּנֵי אֲרָיוֹת,



כֻּלָם נוֹרָאִים, כֻּלָּם מְזָרֵי אֵימָה,



וְעָלַי אֻמְלָל הִבִּיטוּ בְּאַף וּבְחֵמָה.



וּפִתְאֹם קוֹל רֵעִי כְּבַגַּלְגַּל הָרָעַם,



עָלָה בְאָזְנַי בַּחֲמַת עֶבְרָה וָזָעַם:



"נָבָל בְּלִיַּעַל לָמָה עֲכַרְתָּנִי?



יַעְכָּרְךָ אֱלֹהִים כִּי רָעָה גְמַלְתָּנִי!"



וַאֲנִי נִבְהַלְתִּי נִבְעַתִּי וְרוּחִי חֻבָּלָה



נָפַלְתִּי לָאָרֶץ בְּרֶתֶת וְחַלְחָלָה,



חָפַצְתִּי הָרֵם קוֹלִי וְלַעֲרֹךְ תְּפִלָּה,



אַךְ כֹּחִי עֲזָבַנִי וּבִלְשׁוֹנִי אֵין מִלָּה.



וְקוֹל רֵעִי הָהּ! אֵלַי קוֹרֵא בְּמוֹרָא:



"קוּם נָא וַעֲמֹד, מַלְעִיב בַּתּוֹרָה!



לָמָּה תִפֹּל אַרְצָה תִּכְרַע בִּרְכָּיִם?!



מִשְׁפָּטֵנוּ נַגִּישׁ נָא! לִפְנֵי שׁוֹפְטֵי שָׁמָיִם."



וּפִתְאֹם פָּעַר פִּיו הָהּ! צוּקָה וְצָרָה!



וַיְדַבֵּר אֲרָמִית בְּשָׂפָה מוּזָרָה,



וַיְגַמְגֵּם בִּלְשֹׁנוֹ מִלּוֹת מְשֻׁנּוֹת,



וּבִשְׂפַת לֹא יָדַעְתִּי הִשְׂמִעַ קוֹל תְּלוּנוֹת.



וּכְרֶגַע נִשְׁמַע הָהּ! קוֹל עֲנוֹת גְּבוּרָה:



"דּוּשׁוּ אֶת-בְּשָׂרוֹ בִתְרֵין פֻּלְסִין דִּי-נוּרָא!



רַב וְגָאוֹן בְּעַד הַמִּלִּים הַשְּׁנָיִם



בִּתְרֵין פֻּלְסִין יִוָּסֵר עַל בְּשַׂר הָאֲחֹרָיִם!…



אוֹיָה! – קָרָאתִי – מַדּוּעַ וְלָמָּה?



וָאֶפֹּל אַרְצָה כְּגוּף בְּלִי נְשָׁמָה.



"עוֹד תִּשְׁאַל מַדּוּעַ? – הוֹי פֶּשַׁע וְחַטָּאָה!



– קָרָא זְקַן הַחֲבוּרָה בְּחֵמָה נוֹרָאָה –



עוֹד תִּשְׁאַל מַדּוּעַ? – הוֹי זִמָּה וּנְבָלָה!



הוֹסִיפוּ לוֹ אֶחָד! וְאַל יַעֲמֵק עוֹד שְׁאָלָה.."



וּכְרֶגַע הָהּ! שְׁנֵי סָרִיסִים אֲיֻמִּים,



גְּבֹהִים כַּעֲנָקִים גְּדוֹלִים וַעֲצוּמִים,



נִגְּשׁוּ אֵלַי אֻמְלָל בְּקֶצֶף אַכְזָרִי,



וּשְׁנֵי שְׁבָטִין רְתִיחִין הֶעֱבִירוּ עַל בְּשָׂרִי,



וְטֶרֶם יָד הֵרִימוּ לְהַכּוֹתֵנִי שְׁלִישִׁיָּה!



קִדְּמַנִי הַמָּוֶת וָאֵרֶד נְשִׁיָּה…



אָז קָמָה בַּשְּׁנִיָּה מְבוּכָה וּמְהוּמָה,



מַה-לַּעֲשׂוֹת בַּשְּׁלִישִׁי אִם יָרַדְתִּי דוּמָה,



אִם לְהַעֲבִיר עֲוֹנִי אַחֲרֵי כִּי עוֹד אֵינֶנִּי,



אוֹ גַּם בַּשְּׁלִישִׁי לְהִשָּׁפֵט מִמֶּנִּי,



אָז יָצָא הָרוּחַ, הָהּ! רוּחַ ר' סַמָּאֵל!



לַעֲמֹד לְשִׂטְנִי כַּשָּׂטָן עַל יִשְׂרָאֵל,



וַיָּרֶם קוֹל עֹז בְּכַעַס וּמִדְיָנִים.



"שִׁמְעוּ נָא רַבּוֹתַי מֻמְחִים דַּיָּנִים!



אֲנִי קָדשׁ וְצַדִּיק בְּיִשְׂרָאֵל הָיִיתִי,



לָמַדְתִּי תּוֹרָה וְדֵי חָכְמָה קָנִיתִי,



דָּרַשְׁתִּי הָגִיתִי בִּינֹתִי בַּסְּפָרִים,



וְתוּשִׁיָּה הוֹדַעְתִּי לִזְקֵנִים וּנְעָרִים



נָשָׂאתִי חֶסֶר וְכֶפֶן, רָעָב וְצִמָּאוֹן,



טֶרֶם הִקָּרֵא זָכִיתִי בְּשֵׁם רַב וְגָאוֹן



וְהִנֵּה בָּא זֶה הָעַוָּל, זֶה אִישׁ הָרְמִיָּה!



וַיַּפְשֵׁט אֶת-כְּבוֹדִי בְּעָרְמָה וּבִצְדִיָּה,



וַיְשִׂימֵנִי לְמָשָׁל בֵּין אַנְשֵׁי עִירֵהוּ,



וּבַמֶּה אֵפוֹא רַבּוֹתַי תּוּכְלוּ הַצְדִּיקֵהוּ?!



לֹא, לֹא, רַבּוֹתַי! הַצְדִּיקֵהוּ אַל תַּעֲמֹלוּ!



עַל רָשָׁע כָּמֹהוּ אַל נָא תַּחְמֹלוּ,



יַסְּרֻהוּ כְּרִשְׁעָתוֹ וּכְמַעֲשָׂיו לוֹ הַמְצִיאוּ,



פָּנָיו אַל תִּשָּׂאוּן וּמִשְׁפַּט אֱמֶת הוֹצִיאוּ,



הַכֻּהוּ! פְּצָעֻהוּ! פִּרְצוּ בוֹ פָּרֶץ!



וְיַדַע הָעַוָּל כִּי יֶשׁ שׁוֹפְטִים בָּאָרֶץ.



וּכְרֶגַע זְקַן הַחֲבוּרָה קָרָא בְּרָעַם:



צָדַקְתָּ ר' סַמָּאֵל, דְּבָרֶיךָ בְּטוּב טָעַם!



מָאתַיִם וְשִׁבְעִים שָׁנָה כְּמִסְפַּר רַב וְגָאוֹן!



יֶחְרַם זֶה הָרָשָׁע וְיִתְגַּלְגַּל בְּמַשָּׁאוֹן!..



גְּזַר דִּינִי נֶחְתַּךְ וַאֲנִי הָהּ! הוּטָלְתִּי,



וּבְרֵחַיִם שֶׁל רוּחַ תַּחַת שֹׁאָה הִתְגַּלְגָּלְתִּי,



וּמֵהַיּוֹם הַהוּא הוֹי זֶה שָׁנִים מֵאָה!



מִדֶּחִי אֶל דֶּחִי אֵלֵךְ וְלֹא אֶמְצָא מַרְגֵּעָה,



פַּעַם בְחַיָּה וּבְהֵמָה, פַּעַם בִּבְאֵר מָיִם



פַּעַם בְּיַנְשׁוּף וַעֲטַלֵּף, פַּעַם בְּרֵחָיִם



פַּעַם בְּאִישׁ וְאִשָּׁה, פַּעַם בְּשׁוֹר וְגָמָל



וּבְכֻלָּמוֹ אַלְלַי אֶשְׂבַּע רַק עָמָל.



צַדִּיקִים בַּעֲלֵי שֵׁמוֹת עֲרוּמִים כַּנְּחָשִׁים,



מְעוֹנְנִים וּמְנַחֲשִׁים וְלֹחֲשֵׁי לְחָשִׁים,



קוֹסְמִים וְקוֹסְמוֹת מְכַשְּׁפִים וְרַבָּנִים,



יְסֻבּוּנִי כִּדְבוֹרִים מֵאָחוֹר וּמִפָּנִים,



וְעָלַי אֻמְלָל הָהּ כֻּלָּמוֹ יִתְמַלָּאוּ,



יַשְׁבִּיעוּנִי יַחֲרִימוּנִי וּמִבְּשָׂרִי לֹא יִשְׂבָּעוּ,



זֶה יִכְתֹּב קָמַעוֹת וְזֶה מְרוֹרִים יַשְׁקֵנִי,



וְאֵין אִישׁ מִתְבּוֹנֵן כִּי אֻמְלָל הִנֵּנִי!..

"הוֹי! – קָרָא הַצַּדִּיק – הוֹי פְּצַמְתָּנִי!



קְשִׁי-יוֹמְךָ הָהּ! עַד אֶרֶץ דִּכְּאָנִי,



אֶרְחָמְךָ בֶּן יַקִּיר, רָאִיתִי עָנְיֶךָ!



וּמַהֵר הַגִּידָה מָה אֶעֱשֶׂה לְמַעֲנֶךָ.



"רַבִּי! – קָרָא הָרוּחַ – רַבִּי וַאֲדוֹנִי!



הֵן לִבִּי הִגַּדְתִּי, גִּלִּיתִי עַל עֲוֹנִי,



חַטָּאתִי הוֹדַעְתִּיךָ מֵהָחֵל וְעַד כָּלָה,



וּבְכֵן הַקְשֵׁב וּשְׁמַע בַּמֶּה הוֹשִׁיעֵנִי תּוּכָלָה!



פֹּה בַּחֲצַר-הַמָּוֶת, הֲלֹא תֵדַע צַדִּיקִי,



יִשְׁכַּב ר' סַמָּאֵל אִישׁ רִיבִי וּמְצִיקִי,



פֹּה יִשְׁכֹּן יָנוּחַ, מוּדַעַת זֹאת לַכֹּל,



זֶה שָׁנִים מֵאָה טָמוּן בַּחוֹל…



לָכֵן מַהֶר רַבִּי, קַח אֲנָשִׁים עֲשָׂרָה!



וְלֵךְ הִטָּבֵל בַּמַּיִם בִּקְדֻשָּׁה וּבְטָהֳרָה,



אֱמֹר פַּרְשַׁת קְטֹרֶת אַתָּה וַאֲנָשֶׁיךָ!



וְאֶל עֵמֶק הָרְפָאִים חִישׁ הָכֵן צְעָדֶיךָ



וְשָׁם בַּחֲצַר-הַמָּוֶת בְּקוֹל מַר צוֹרֵחַ,



עַל קֶבֶר ר' סַמָּאֵל נְפֹל וְהִשְׁתַּטֵּחַ,



קְרָאֵהוּ רַב וְגָאוֹן תָּם נוֹרָא עֲלִילָה,



וּבִשְׁמִי ר' זַנְבִיאֵל בַּקְּשֵׁהוּ מְחִילָה!



סְלַח, מְחַל, כַּפֵּר – תִּקְרָא בַּכֹּחַ –



לְפֶשַׁע רֵעֲךָ הַנָּד בְּלִי מָנֹחַ,



סְלַח לוֹ מְחַל לוֹ כִּי רוּחוֹ נְכֵאָה,



זְכָר-עָנְיוֹ וּמְרוּדוֹ זֶה שָׁנִים מֵאָה,



רַחֲמֵהוּ, הַצִּילֵהוּ – תִּשְׁנֶה אֶת-דְּבָרֶיךָ –



כִּי רַבּוֹת כְּבָר נָשָׂא עַל חִלּוּל שְׁמֶךָ,



רַבּוֹת כְּבָר שָׂבַע שֹׁד נֶגַע וְאָסוֹן,



וְחִישׁ תִּקַּח בְּיָדְךָ סֵפֶר “מַעֲנֵה-הַלָּשׁוֹן”



וְקַדִּישׁ דְּרַבָּנָן בְּקוֹל מַר תְּשַׁוֵּעַ,



וְאָז רַבִּי תְּתַקְּנֵנִי וּמִצָּרוֹתַי אִוָּשֵׁעַ!



מִצָּרוֹתַי אִוָּשֵׁעַ – לֹא אֵדַע עוֹד אֲנָקָה!



אֶת-הַנַּעֲרָה אֶעֱזֹבָה וּלְךָ תִּהְיֶה צְדָקָה".



"גַּבָּאִים! – קָרָא הַצַּדִּיק עָלֵץ בְּקוֹל תְּרוּעָה –



גַּבָּאִים הִכּוֹנוּ, כִּי הַיּוֹם יוֹם יְשׁוּעָה!



הַיּוֹם יִוָּדַע שְׁמִי לְכָבוֹד וּלְתִפְאָרֶת,



אַךְ נַעֲשֶׂה דַּרְכֵּנוּ כִּי הַשָּׁעָה עוֹבָרֶת,



מַהֲרוּ הִתְקַדָּשׁוּ שַׁמָּשִׁים וְגַבָּאִים!



הָבָה נֵצֵא הַשָּׂדֶה אֶל עֵמֶק הָרְפָאִים".



וּכְרֶגַע הֲמוֹן חֲסִידִים בְּחִפָּזוֹן וּבֶהָלָה,



בֵּיתָה רַבָּם הִתְלַקָּטוּ כְּמַלְאֲכֵי חַבָּלָה



וְיַחַד שְׁלוּבֵי זְרֹעַ דְּחוּפִים וּמְבֹהָלִים,



קַל חִישׁ אֶל הַמִּקְוָה רָקְדוּ כָּאַיָּלִים,



אֵין עָיֵף וְאֵין כּוֹשֵׁל בָּעֵדָה הַמְּפֹאַרַה,



וְשַׁ"י טְבִילוֹת נִטְבְּלוּ בִּקְדֻשָּׁה וּבְטָהֳרָה.



וּכְרֶגַע מִן הָרַחְצָה יְגֵעִים וַעֲמֵלִים



עָלוּ לְאַחַד אֶחָד כְּעֵדֶר הָרְחֵלִים,



וּבְלִוְיַת צַדִּיקָם בְּשִׁיר וּמְחֹא כַפָּיִם,



אָרְחוֹת דַּרְכָּם לָפָתוּ אֶל עֵמֶק הָרְפָאִים,



כֻּלָּם שָׁרִים וּמְרִיעִים בְּקוֹל עֲנוֹת גְּבוּרָה,



וְגַם הַזָּקֵן וּבִתּוֹ נִלְווּ אֶל הַחֲבוּרָה…



וּכְרֶגַע הֶאָח בְּעֹנֶג בְּלִי עַצָּבֶת,



נִפְתַּח לַצַּדִּיק שַׁעַר חֲצַר-הַמָּוֶת,



ור' לֶמֶךְ הַקָּדֹשׁ עַל אַחַד הַגַּלִים,



מְלֹא קוֹמָתוֹ הִתְנַפֵּל בְּפָנִים צֹהֲלִים,



וּפִתְאֹם קוֹל בָּלוּל מָהוּל בִּבְכִי וּרְנָנָה,



הֵרִים בִּקְדֻשָּׁתוֹ כְּעֵגֶל בֶּן-שָׁנָה,



ר' סַמָּאֵל! ר' סַמָּאֵל! רַחֶם-נָא, חֲמֹל!



לְפֶשַׁע שִׂפְתֵי רֵעֲךָ סְלַח, כַּפֵּר, מְחֹל!



אָנָּא רַב וְגָאוֹן תָּם נוֹרָא עֲלִילָה,



רְאָה עֳנִי ר' זַנְבִיאֵל הַמְבַקֶּשְׁךָ מְחִילָה,



עַד אָן תִּטוֹר אֵיבָה תִּמְשֹׁךְ אַפֶּךָ?



אָנָּא קָדשׁ עֲנֵנִי: סָלַחְתִּי כִּדְבָרֶיךָ!



חוּסָה נָּא רַחֶם, אֲהוּבִי ר' סַמָּאֵל!



הֲיָאֶה לְרַב וְגָאוֹן בַּקִּנְאָה לְהִתְגָּאֵל?



הֲנָאֶה לְמוֹרֵה צֶדֶק לִנְקֹם בְּאַף וְחֵמָה?



הֲכַךְ דַּרְכָּם שֶׁל צַדִּיקִים לִטּוֹר מַשְׂטֵמָה?



לֹא לֹא ר' סַמָּאֵל, מְחֹל עַל כְּבוֹדֶךָ!



וְכָלָה אַל תַּעַשׂ עִם ר' זַנְבִיאֵל רֵעֶךָ,



סְלַח לוֹ, מְחַל לוֹ, כִּי רוּחוֹ נְכֵאָה!



זְכָר-עָנְיוֹ וּמְרוּדוֹ זֶה שָׁנִים מֵאָה,



רַחֲמֵהוּ, הַצִּילֵהוּ, אַל נָא בְּאַפֶּךָ!



וְאִם לֹא לְמַעֲנֵהוּ עֲשֵׂה לְמַעַן שְׁמֶךָ".



וְהַחֲסִידִים הֶאָח! עוֹמְדִים מַרְעִידִים,



וְקוֹל רַבָּם יִשְׁמָעוּ בְּקֹלוֹת וְלַפִּידִים,



אִיש אֶת-רֵעֵהוּ יִתְלַחֲשׁוּ בְּאֵימָה וְיִרְאָה,



וְאֶל פָּנָיו יַבִּיטוּ כְּאֶל אַסְפַּקְלַרְיָה הַמְּאִירָה.



וְהַצַדִּיק עַל הַקֶּבֶר יִשְׁכַּב בְּלִי נֹעַ,



יַטֶּה אָזְנוֹ לַגַּל כְּמַקְשִׁיב לִשְׁמֹעַ,



שֹׁאֵל וְעוֹנֶה וּמְדַבֵּר בְּלִי סְדָרִים,



וּכְמֻכֵּה בְשִׁגָּעוֹן יֶרֶב אֲמָרִים,



יִתְאַנַּח וְיִתְאוֹנֵן וְאֶת-גֵּווֹ מוֹתֵחַ,



וּלְאָרְכּוֹ וּלְרָחְבּוֹ עַל הַקֶּבֶר יִשְׁתַּטֵּחַ.



וּפִתְאֹם אֲהָהּ! קוֹל אַדִּיר וְחָזָק



הִתְפָּרֵץ מֵהַקֶּבֶר כְּרַעַם אַחַר בָּזָּק



הוֹי נָבָל בְּלִיַּעַל הִנְּךָ בְיָדִי!



פֹּה חָבוּשׁ בַּטָּמוּן תִּשְׁכַּב עִמָּדִי,



פֹּה נָבָל תִּקָּבֵר! פֹּה תִּמְצָא מָנֹחַ!



מִיָּדִי, אִישׁ זִמָּה! לֹא תִבְרַח בָּרֹחַ.



וְיָד גְּדוֹלָה וַחֲזָקָה אֲרֻכָּה וּרְחָבָה,



אֲיֻמָּה וְנוֹרָאָה שְׁמֵנָה וְעָבָה,



מֵעֲרֵמַת הַקֶּבֶר הָהּ פִּתְאֹם הִבְרִיקָה,



וּשְׁתֵּי פְּאוֹת הַצַּדִּיק בְּכֹחַ הֶחֱזִיקָה,



וַתִּמְשְׁכֵן הָהּ עָמֹק עָמֹק בַּקָּבֶר,



וְאַחֲרֵיהֶן רֹאשׁ הַצַּדִּיק, הוֹי שֵׁאת וָשָׁבֶר!



נִדְחַק נִלְחַץ לַגַּל כַּכֶּלֶב בֵּין הַדֶּלֶת



וְלַהֲנִיעוֹ כֹּה וָכֹה הָהּ אֵין תּוֹחֶלֶת…



"אוֹיָה! – קָרָא הַצַּדִּיק – הַצִּילוּנִי רַבּוֹתַי!



רְאוּ רְאוּ… אֵיךְ בַּקֶּבֶר… נִסְחֲבוּ פְּאוֹתַי…



הָהּ… רֹאשִׁי רֹאשִׁי… אָנָא סַמְּכוּנִי



גַּבָּאַי… גַּבָּאַי… קוּמוּ הַצִּילוּנִי…



מוֹרְ.. רַ.. רַ וְרַ.. רַ.. בּוֹתַי זֶרַע יִשְׂרָאֵל!



מַלְּטוּנִי הַצִּילוּנִי… מִיַּד ר' סַמָּאֵל.



רַחְ.. חֲמוּ הַצִּילוּ… דְּגוּלִים מֵרְבָבָה,



הָהּ… רֹאשִׁי, צַוָּארִי… מַהֲרוּ אַנְשֵׁי צָבָא!



אוֹי לִי… אַלְלַי לִי… כֹּחִי… עֲזָבָנִי



הָהּ… רוּחִי… חֻבָּלָה… וְאֵל הֲדִמָּנִי



ר' סַמָּאֵל… ר' סַמָּאֵל… רַחֶם נָא… חוּסָה!



הָהּ רֹאשִׁי… יִתְפּוֹצֵץ… וּפֹה – – אָמוּתָה".



"פֹּה פֹּה תָמוּתָה – הוֹי חוֹטֵא וּפוֹשֵׁעַ!



– נִשְׁמַע מֵהַקֶּבֶר קוֹל לֵב קוֹרֵעַ –



פֹּה עָשׁ יֹאכַלְךָ בְּיַרְכְּתֵי בוֹר!



פֹּה נָבָל תִּקָּבֵר קְבוּרַת חֲמוֹר,



פֹּה רֹאשְׁךָ כֶּלֶב בְּעֶבְרֵי פִּי פַחַת,



תִּמְשֹׁךְ תִּסְחַב אַחֲרֶיהָ יָדִי הָאַחַת,



וְאַחַר גַּם הַשְּׁנִיָּה עָלֶיךָ אָרִימָה!



וּבָהּ טָמֵא לְצַוָּארְךָ מַחֲנַק אָשִׂימָה,



עֲדֵי תֵצֵא מִפִּגְרְךָ קְלִפָּתְךָ הָאֲרוּרָה,



וְתֵדַע בְּלִיַּעַל כִּי לִי עֹז וּגְבוּרָה".



"חָנּוּנִי! חָנּוּנִי! יִשְׂרָאֵל עַם סְגֻלָּה!



– חִנֵּן הַצַּדִּיק קֹלוֹ בְּרַעַשׁ וְשָׂפָה בְּלוּלָה –



חָנֻּנִי! חָנֻּנִי! עַם אֱלֹהֵי אַבְרָהָם!



– הוֹסִיף תֵּת קֹלוֹ קוֹל אַדִּיר וָרָם –



אַךְ מַמְזֵרִים רְשָׁעִים לָמָּה תַעַמְדוּ מִנֶּגֶד?!



בְּרַבְּכֶם קְדשְׁכֶם לָמָּה תִבְגְּדוּ בֶגֶד?



שְׁקָצִים! קֻנְדֵּסִים! נֹאֲפִים שִׁכּוֹרִים!



לָמָּה תִבְרְחוּ בְּהֵחָבֵא וְתִסָּתְרוּ בַּחוֹרִים?



כְּלָבִים מַהֵרוּ, אַנְשֵׁי חַיִל הָבִיאוּ!



לְרַבְּכֶם הָאֻמְלָל אוּלַי עֵזֶר תַּמְצִיאוּ.



וְהַחֲסִידִים אֲהָהּ! נִבְהֲלוּ נֶחְפָּזוּ,



וְהָעִירָה חִישׁ קַל דַּרְכָּם אָחָזוּ,



כָּל-לֵבָב דַּוָּי עַל כָּל-פָּנִים צַלְמָוֶת!



וְחִישׁ שׁוֹטְרִים הֵבִיאוּ מְלֹא חֲצַר-הַמָּוֶת.



בָּם פְּקִידִים, סָרִיסִים וְרַבֵּי הַמֶּלֶךְ,



נְצִיבִים שָׁלִישִׁים וְשָׂרֵי עִיר וָפֶלֶךְ,



וּכְרֶגַע לַצַּדִּיק כֻּלָּמוֹ הָהּ פָּנוּ!



וּבְלַעַג וּשְׁאָט נֶפֶשׁ אֵלָיו הִתְבּוֹנָנוּ,



וַיָּרִימוּ קוֹל גָּדוֹל בִּצְחֹק פָּרוּעַ;



וַיִּסֹּגוּ חִישׁ אָחֹר מֵהַצַּדִּיק הַנָּגוּעַ.



"אוֹיָה! – קָרָא הַצַּדִּיק – אֲדוֹנַי הוֹשִׁיעוּ!



אֶת רֹאשִׁי מִמַּסְגֵּר אָנָּא… הוֹצִיאוּ;



שׁוֹטְרִים אַנְשֵׁי חַיִל… קִרְבוּ נָא הֲלֹם!



הָהּ רֹאשִׁי יִשָּׁבֵר בִּבְשָׂרִי אֵין מְתֹם;



חוּסוּ הַצִּילוּ… פֶּן אֵלֵךְ… וְאֵינֶנִּי



הוֹי הוֹי ר' סַמָּאֵל… הֶרֶף מִמֶּנִּי!"



"נָבָל! בְּלִיַּעַל! פֹה תֵּרֵד לַשַּׁחַת!



– הִתְפָּרֵץ קוֹל אָיֹם מֵהַקֶּבֶר מִתַּחַת –



פֹּה שְׁאוֹל תֵּחַתָּה – אִישׁ זִמָּה וּמְזִמָּה!



פֹּה תֹּאכַלְנָה נִבְלָתְךָ תּוֹלֵעָה וְרִמָּה,



פֹּה עָלֶיךָ בְּלִיַּעַל תֶּאְטַר בְּאֵר פִּיהָ,



אַךְ מַהֵר הִתְוַדֵּה טֶרֶם תֵּרֵד נְשִׁיָּה!



"הוֹי! – קָרָא הַצַּדִּיק – ר' סַמָּאֵל חָנֵּנִי!



הָשֶׁב נָא אֶל לִבְּךָ כִּי קָדֹשׁ הִנֵּנִי,



חוּסָה נָּא, רַחֵם, עַל מַלְאַךְ צְבָאוֹת!



מַה-מֶּנִּי תַּחְפֹּצָה כִּי תִפְרַע בִּי פְּרָעוֹת?



מֶה עָוֶל פָּעַלְתִּי כִּי בְאַף תִּשְׂטְמֵנִי,



הוֹדִיעֵנִי עַוָּתָתִי בִּגְּלָלָהּ תִּרְדְּפֵנִי,



ר' סַמָּאֵל! ר' סַמָּאֵל! בְּצִדְקָתְךָ הַגְּדוֹלָה,



חוּסָה נָּא רַחֵם וְאַל אֵרֵד שְׁאוֹלָה".



"מַהֵר תֶּן-תּוֹדָה בֹּגֵד וּמְרַצֵּחַ!



– הִרְעִים מֵהַקֶּבֶר קוֹל מַר צֹרֵחַ –



מַהֵר תֶּן-תּוֹדָה – נוֹכֵל בְּלִיַּעַל!



פֶּן כֶּעָפָר אֶשְׁחָקְךָ פֹּה עַל הַגַּל.



הוֹי נָבָל מְנֻוָּל, הוֹי חַיָּה טֹרֶפֶת!



הוֹי לְטָאָה נוֹרָאָה! דָּם נָקִי שֹׁאֶפֶת



הוֹי חוֹמֵס וְרוֹמֵס מְקוֹר עָוֶל וָרֶשַׁע!



מִמִּי תַּסְתִּיר דָּבָר וּתְכַחֵשׁ עַל פֶּשַׁע?!



הַאוֹתִי תְרַמֶּה? הוֹי יְלִיד שַׁחַת!



בַּשָּׁמַיִם מִמַּעַל הֵן עֵינִי מַשְׁגַּחַת,



וְשָׁם עַל יְרִיעָה בְּעֵט בַּרְזֶל וְעֹפֶרֶת,



חַטָּאתְךָ לְר' דָץ חֲרוּתָהּ לְמַזְכֶּרֶת!…



לָכֵן מַהֵר הִתְוַדֵּה כָּל-עוֹד בְּךָ נְשָׁמָה,



כִּי דָמוֹ אֲהָהּ! אֵלַי צֹעֲקִים מִן הָאֲדָמָה…"



"אוֹיָה! – קָרָא הַצַּדִּיק – עָוִיתִי חָטָאתִי!



וְר' דָץ בְּמַהְפֶּכֶת יֵשֵׁב בְּאַשְׁמָתִי,



אוֹי לִי אֲבוֹי, הִשִּׁיגוּנִי עֲוֹנוֹתַי!!



וַחֲטָאַי בַּסֵּתֶר שִׁמְעוּ נָא רַבּוֹתַי:



אֲנִי לֶמֶךְ מֵעוֹדִי עָשִׂיתִי כָּל-תֹּעֵבָה!



כָּל-הוֹנִי וּרְכוּשִׁי אַךְ גְּזֵלָה וּגְנֵבָה,



בָּתַּי כְּרָמַי וְסוּסַי הָעוֹלִים לַאֲלָפִים,



כָּל-אֵלֶּה רָכַשְׁתִּי מִשְּׁטָרוֹת מְזֻיָּפִים.



גַּם אֲחַזֵּק בַּסֵּתֶר יְדֵי פוֹשְׁעִים וּמְרֵעִים,



וְהֵם כֻּלָּם יַחְשְׁבוּנִי לְצַדִּיק תְּמִים דֵּעִים,



הֵם יְעַקְּשׁוּ וְיַעַשְׁקוּ יְשָׁרְשׁוּ כָּל-יֹשֶׁר,



וּמִתְּבוּאַת עֲמָלָם אֶעֱשֶׂה אֲנִי עשֶׁר.



וְכֵן עֶשְׂרִים שָׁנָה בְּלִי מַעֲצָר זֹאת אַעַשׂ,



וְלַבְּקָרִים לִי רַבּוּ מַאֲמִינִים חַסְרֵי דַעַת.



פִּתְאֹם בָּא לְעִירִי ר' דָּץ מֵחֲצַר-מָוֶת,



וַיְקַנֵּא בְּעָשְׁרִי וּבָהּ אִוָּה לָשָׁבֶת!



וְאַךְ צָעַד בִּגְבוּלִי נָפַל מֶנִּי רָב,



כִּי כְבָר הָמָה בֵיתוֹ מֵעֹבֵר וָשָׁב,



כְּבָר קִבֵּל פִּדְיֹנוֹת וַיֵּשֶׁב לְמוּלִי,



וְאֶל לִבּוֹ לֹא שָׂם כִּי הִסִּיג אֶת-גְּבוּלִי,



אָז קַמְתִּי אֲהָהּ! בְּאַפִּי וַחֲרוֹנִי!



וְנָקָם לָבַשְׁתִּי כְּנָחָשׁ צִפְעֹנִי,



קָרָאתִי לַחֲסִידַי עֲלִיזֵי גַאֲוָתִי,



וּבַעֲצַת חָסִיד אֶחָד שׁוֹמֵר פְּקֻדָּתִי…



שִׁקְלֵי כֶסֶף לִי שָׁקְלוּ חֲסִידַי הַנִּבְעָרִים,



וַאֲנִי בִּמְזֻיָּפִים הֶחֱלַפְתִּים בַּמִּסְתָּרִים,



וְאוֹתָם לְר' דָץ נָתַתִּי לִבְרָכָה,



וְלֹא יָדַע, הָהּ, כִּי יִתְהַלֵּךְ עַל שְׂבָכָה!



עַל שְׂבָכָה הִתְהַלֵּךְ וּבָהּ נוֹקַשׁ אֲהָהּ!



וְגַם עֲלִילַת הַגְּנֵבָה מִיָּדִי לוֹ בָאָה



אֲנִי לוֹ נָתַתִּי הַסּוּסִים וְהַמֶּרְכָּבָה



וַאֲנִי אַחֲרָיו דָּלַקְתִּי בְּשׁוֹטְרִים אַנְשֵׁי צָבָא,



וּבְאַשְׁמָתִי וְחַטָּאתִי הוֹי שֹׁאָה וָאֶמֶשׁ!



הוּבָא בַסֹּהַר לְעֵינֵי הַשֶּׁמֶשׁ.



נָקִי! נָקִי! ר' דָץ, זַךְ הוּא מִפֶּשַׁע!



גִּזְרוּנִי בַּתְּרוּנִי כִּי בִי בִי הָרֶשַׁע!"



“בּוֹשׁ חֶרְפַּת-אָדָם!” – הִרְעִים קוֹל מֵהַקָּבֶר,



וּכְרֶגַע הוֹי נִבְקַע וּמִמֶּנּוּ קָפַץ גָּבֶר,



גֶּבֶר נוֹרָא וְאָיֹם, הוֹי חֹשֶׁךְ וְצַלְמָוֶת!



גֶּבֶר חָסוֹן כָּאַלּוֹן ר' דָץ מֵחֲצַר-מָוֶת!…



וְאֶל הַצַּדִּיק הַנִּבְהָל קָרָא בְּזָעַם:



"הֲתַכִּירֵנִי נָבָל? הֲתַכִּירֵנִי הַפָּעַם?!



אֲנִי אֲנִי הוּא דָץ הַמְחֹלָל מִפְּשָׁעֶיךָ!



גַּם גַּבָּאַי הַשְּׁנַיִם פֹּה הִנָּם לְפָנֶיךָ



הֲלֹא תִרְאֶה בָּזֶה הַנַּעֲרָה וְאָבִיהָ,



וְהֵם שׁוֹלָל הוֹלִיכוּךָ בְּעָרְמָה וּצְדִיָּה,



הֵם הֵמָּה גַּבָּאַי פּוּרִיאֵל וְיִשְׁמָעֵאל,



וְלֹא הָיָה וְלֹא נִבְרָא זַנְבִיאֵל וְסַמָּאֵל!



גַּם הַשּׁוֹפְטִים וְהַשּׁוֹטְרִים הָעוֹמְדִים נֶגְדֶּךָ,



עַל פִּי דַעְתָּם וּרְצוֹנָם קַמְתִּי לְרָדְפֶךָ,



הֵם יָדְעוּ מִכְּבָר אֶת-זִמַּת תַּבְלִיתֶךָ,



לָכֵן הִנִּיחוּנִי לְהַכְרִית אַחֲרִיתֶךָ.



רְאֵה נָבָל, בְּלִיַּעַל, רְאֵה וְהִוָּכֵחַ!



כִּי אַחֲרִית אֵינֶנָּה לְבֹגֵד וּמְרַצֵּחַ,



שֶׁבַע יֶחֱטָא וְיִפְשַׁע בְּמֶמֶר וְזָדוֹן,



יַעֲשֹׁק יִגְנֹב וִיגָרֶה כָּל-מָדוֹן,



יְשַׁדֵּד וְיִרְצַח בְּאַפּוֹ וַחֲרוֹנוֹ,



וּבְאַחֲרִיתוֹ יִתָּפֵשׂ וְיִלָּכֵד בַּעֲוֹנוֹ!



וְצַדִּיק הֶאָח! – שְׁמַע מוּסָר וַחֲכָם –



שֶׁבַע אִם יִפֹּל יִפֹּל-וָקָם,



כָּל-פֶּגַע וָנֶגַע יִסְעֲרוּ לַהֲפִיצֵהוּ,



הָפֵחַ וְאֵינָם – וּרְשָׁעִים כָּפְרֵהוּ;



כִּי אָמְנָה בַּמָּרוֹם דָּבָר לֹא יִפָּלֵא,



וְאִישׁ דָּמִים וּמִרְמָה בְּלֹא יוֹמוֹ יִמָּלֵא.



יֵשׁ אֹזֶן קַשֶּׁבֶת וְעַיִן מַשְׁגַּחַת,



וּבְנֵי אָדָם שׁוֹפֶטֶת בָּאָרֶץ מִתַּחַת,



הִיא תַבִּיט לַלֵּב וּבִכְלָיוֹת פְּנִימָה,



הִיא תָשׁוּר בָּאָרֶץ אֶת-נִרְדָּף עַל בְּלִימָה,



הִיא תִרְאֶה נִפְלָאוֹת לֹא שְׁזָפַתָּן עֵין כֹּל,



וְלַחוֹטֵא אֵין מָנוֹס עַד בְּרִיחֵי שְׁאוֹל!



פִּיךָ הִכְשִׁילְךָ! בּוֹשׁ נָא מֵחַטֹּאתֶיךָ!



בּוֹשׁ כִּי כָשַׁלְתָּ וְהֻדַּחְתָּ בְּרָעָתֶךָ.



הִכּוֹן נָא לֶמֶךְ, הִכּוֹן וְהָכֵן!!



לִפְנֵי מַלְכוּתָא דְאַרְעָא הַיּוֹם הֵן תִּבָּחֵן,



שָׁם תַּעֲמֹד בַּמִּשְׁפָּט עַל כָּל-גְּנֵבָה וּגְזֵלָה,



שָׁם יְחִתְּךָ חַטָּאתְךָ חָטָאתָ בָּאֲפֵלָה,



שָׁם לְגֵוְךָ נָכוֹנוּ שְׁבָטִים וּמַהֲלֻמּוֹת,



שָׁם בְּשָׂרְךָ יָדוּשׁוּ בְּמַכּוֹת אֲיֻמּוֹת,



שָׁם תִּגְוַע בַּעֲוֹנְךָ לֹא תִרְאֶה עוֹד אוֹר,



שׁוֹטְרִים אִסְרֻהוּ וּתְנֻהוּ בַּבּוֹר!"…



וּכְרֶגַע הָהּ! שׁוֹטְרִים אֲיֻמִּים



חֲזָקִים כַּעֲנָקִים וְכַתּוֹלָע אֲדֻמִּים,



עַל הַצַּדִּיק הִתְנַפְּלוּ כָּעַיִט עַל הַפְּגָּרִים,



וְעַל בְּשָׂרוֹ חִישׁ שָׂמוּ כְבָלִים וִיתָרִים,



וְעַל כַּפַּיִם אֲהָהּ! כְּעֵגֶל בֶּן-שָׁנָה,



מֵחֲצַר-הַמָּוֶת נְשָׂאֻהוּ בְּזֶמֶר וּרְנָנָה.



וְהַחֲסִידִים אֲהָהּ! כְּמֵתִים מֵאַשְׁמַנִּים



מְלַוִּים אֶת-רַבָּם בְּאֵבֶל כַּתַּנִּים,



הַלְוָיַת הַחַי – בִּילֵל וְתַמְרוּרִים



לַהֲשִׁיבוֹ אֶל עַפְרוֹ – בֵּית הָאֲסוּרִים,



שֶׁמִּמֶּנּוּ לְפָנִים בִּשְׂמָחוֹת וְצָהֳלוֹת,



לֻקַּח לְקָדשׁ בְּגִבְעַת-הָעֲרָלוֹת!



כֻּלָּם נוּגִים אֲבֵלִים עֲיֵפִים וְקֹדְרִים,



שְׁחֹחִים וּשְׁחֹרִים כְּשׁוֹכְנֵי חֲרֵרִים,



כָּל-רוּחַ נִפְעָמָה, כָּל-נֶפֶשׁ עַד דַּכָּא,



כָּל-פָּנִים נִזְעָמִים וְכָל-עַיִן מְפַכָּה,



כָּל-חֲסִידָה אֲבֵלָה, כָּל-חָסִיד לַמּוֹט



כִּי יִרְאוּ גְמוּל רַבָּם בְּעָלְמָא דִּקְשׁוֹט…



וְכֹה נוֹתְרָה לָהֶם הוֹי, בְּכִיָּה לְדֹרוֹת!



עַל יְדֵי לִקּוּי אֵלֶּה שְׁנֵי הַמְּאוֹרוֹת…



וּבְכֵן יִקָּרֵא בְיִשְׂרָאֵל שֵׁם שִׁירִי זֶה:



שְׁנֵי צַדִּיקִים שֶׁחָבְלוּ זֶה בָזֶה!


ת“ו, ש”ל“ב”ע.
