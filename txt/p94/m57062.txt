


	
	הַחֲלוֹם וְשִׁבְרוֹ - עוד דמעה אחת אל נאד דמעותינו
	



 מֻגָּשׁ

לידידי החביב הר"ר יצחק גאלדבערג

מאתי

המחבר.

א


עַל אַדְמַת רוּסִיָּה הָעֲקֻבִָה מִדָּמִים,

בְּשַׁוְלַן הָעִיר, בְּפִנָּה גַלְמוּדָה,

נִשְׁקָף שָׁם בַּיִת, מָט מֵרֹב יָמִים,

עוֹמֵד כְּזֵכֶר לְחֻרְבַּן בַּת יְהוּדָה.

שָׁתוֹתָיו נֶהֱרָסוֹת, כְּתָלָיו נְשַׁמִּים,

לֻחוֹתָיו נִשְׁבָּרִים, חַלוֹנָיו בָּאֲדָמָה,

גַּגּוֹ­ הֶהָרוּס עֹטֶה מְעִיל שְׁמָמָה,

לֹו מַרְאֶה בֵּין אֶחָיו – יִשְׂרָאֵל בָּעַמִּים…

מֵחָרְבָּה הַזֹאת קַו אוֹר קִפָּאוֹן

עַתָּה בַּלַּיִל כְּמוֹ לִפְרוֹץ יִבָּהֵל

דֶּרֶךְ הַחַלּוֹן אֶל תּוֹך הַמַּשָּׁאוֹן –

וְהָיְתָה לּוֹ דְּמוּת כְּתִקְוַת יִשְׂרָאֵל…

שָׁמָּה בַּחָרְבָּה בֵּין כָּתְלֵי חֻרְבָּנָהּ,

מִתְהַלֵּךְ אִישׁ שֵׂיבָה אָנֶה וָאָנָה,

כַּהֲלוֹךְ יִרְמְיָהוּ בֵּין קִבְרֵי צַלְמָוֶת;

עֲנָנִים עַל מִצְחוֹ יַעֲטוּ מַחֲשַׁכִּים,

וּכְמוֹ מִטְרוֹת עֻזָּם מֵעֵינָיו מְפַכִּים,

אֶגְלֵי מֵי רֹאשׁ, דִּמְעוֹת עַצָבֶת…

שָׁמָּה בַּפִּנָּה עַל מִטָּה מִתְנֹדֶדֶת

וְסַפְסָל צוֹלֵעַ – קְרָשִׁים נִשְׁעָנִים;

עֲלֵיהֶם רַעְיָתוֹ יְצוּעָהּ מְרַפֶּדֶת

וַחֲמִשָּׁה יְלָדִים עַל מַחְצֶלֶת שֶׁל קָנִים.

הַנְּפָשׁוֹת הָאֵלֶּה בִּזְרֹעוֹת הַשֵּׁנָה

עַתָּה אֲחוּזוֹת וּשְׁכֵחוֹת עָנְיָמוֹ,

וְהוּא רַק הָאֶחָד הָעֵר בֵּין כֻּלָּמוֹ,

כִּי הוּא נוֹתֵן לָחְמָן הַדּוֹאֵג לָהֵנָּה.

עֵר הוּא, כִּי אָב הוּא, וְאִשְׁתּוֹ וּבָנָיו

רְעֵבִים הָהּ שָּׁכְבוּ וּרְעֵבִים יָקוּמוּ,

וּבְלֵיל מָחָר יֵדַע, כִּי רְעֵבִים מוּל פָּנָיו,

כְּמִשְׁפַּט לֵיל זֶה, שֵׁנִית יָנוּמוּ. –

אָפְסָה כָל תִּקְוָה גַּם כֹּחו עֲזָבוֹ,

לֹא יוּכַל הִשָּׂגֵב וּלְהֵיטִיב מַצָּבוֹ

וּכְבָר גַּם הַיֵּאוּשׁ לִבּוֹ הִכְנִיעַ…

אִם עַל צַוַּאר הַיְּהוּדִי חֶרֶב מוּנַחַת,

גַּם אָז עוֹד יְקַוֶּה וְנַפְשׁוֹ בֹטַחַת –

עֳנִי עִבְרִי זֶה מִי יוּכַל יַבִּיעַ!..

יִשְׂרוֹק הָרוּחַ, יֵהוֹם יִתְחוֹלֵל

בִּבְקִיעֵי הַחָרְבָּה כְּכִנּוֹר בּוֹ שָׁבֶר,

עִם נַחֲרַת יְלָדָיו הֶמְיָתוֹ תִתְבּוֹלֵל –

וְהָיָה הַקּוֹל כְּקוֹל בּוֹכִים עַל קָבֶר.

אוֹר כֵּהֶה שָׁם נֵר יָפִיץ מִמְּנוֹרָה,

מֵרוּחַ בּוֹ יִגַּע שַׁלְהַבְתּוֹ תָנוּעַ,

תִּתְנַשֵּׂא, תִּתְאַבֵּךְ וְאוֹרָהּ זָרוּעַ

בִּנְטוֹתָה הַצִדָּה – אַךְ תֵּכַהּ מְהֵרָה…

תִּתְנַשֵּׂא, תִּתְאַבֵּךְ בַּנֵּר הַשַּׁלְהֶבֶת

בִּנְטוֹתָה הַצִדָּה לְעֵינֵי הֶעָנִי,

וּמַחֲשָׁבָה אֶל לִבּוֹ אַט מִתְגַּנֶבֶת:

אוּלַי הַצִדָּה אֶטָּה גַּם אָנִי?!..

אֶטָּה הַצִּדָּה – וְאוֹרִי יִבָּקֵעַ

וְהוֹדְפַי יְקָרְבוּנִי כְּאָח לָהֶם וְרֵעַ,

אָז נָכוֹן לֹא אֶהְיֶה לְמוֹעֲדֵי רָגֶל…

אַךְ מַה־זֹאת הַנֵּר יֵכַהּ בַּמְּנוֹרָה?

אִשָּׁהּ תִּתְנַשֵּׂא – אַחַר יִדְעַךְ אוֹרָהּ,

בּוּז לִי עַל אִוַלְתִּי, לֹא אֶעֶזוֹב הַדָּגֶל!! –

וּכְרֶגַע נִטּיוּ עֵינָיו אֶל בָּנָיו

וּשְׁתֵּי טִפּוֹת רֹתְחוֹת מֵהֶן הִתְגַּלְגָּלוּ,

לֹא נֶעֶצְרוּ בַתְּלָמִים הַחֲרוּשִׁים עַל פָּנָיו,

וּבִזְקָנוֹ הַלָבָן כַּעֹפֶרֶת צָלָלוּ.

וּכְעָנָן כִּי כִלָה הַשֶּׁמֶשׁ זֹרֵחַ,

כֵּן פָּנָיו הַקֹּדְרִים מִמְּצוּקוֹת וּתְלָאוֹת,

הֵאִירוּ כִּי שֻׁפְּכוּ מֵעֵינָיו הַדְּמָעוֹת,

וְגַאֲוַת עֹז לָבְשׁוּ כּלֹחֵם מְנַצֵחַ.

"לֹא! לֹא! יִקְרָא פִּתְאֹם, בָּנִים נֶאֶמָנִים

תִּהְיוּ יְלָדַי לְעַמְּכֶם וְדַתְכֶם!

שָׁוְא לָכֶם זֵדִים, הַבָּנִים קְשֵׁי פָנִים,

עָרְפָּם הַקָּשֶׁה לֹא תַכְנַע רִשְׁעַתְכֶם!

לְחָיֵינוּ לְמוֹרְטִים, גֵוֵנוּ לְמַכִּים,

דָּמֵנוּ בַחֻצוֹת כַּמַּיִם מְפַכִּים,

הוֹנֵנוּ לִמְשִׁסָה לְכָל חוֹמֵס יֶהִי!

אַך לָנוּ הַנְשָׁמוֹת הַיֹּשְבוֹת בַּקֶּרֶב,

אוֹתָן לֹא יַכְנַע גַּם אֶגְרוֹף גַּם חֶרֶב –

עִבְרִים הָיִינוּ וְעִבְרִים עוֹד נֶהִי!

יוֹסִיף אֹמֶץ הָרוּח, יֵהֹם יִתְחוֹלֵל,

יַחְדֹּר לַבְּקִיעִים כְּמַיִם זֵידוֹנִים,

עִם מַשַּׁק נְשִׁיקוֹת הֶמְיָתוֹ תִתְבּוֹלֵל –

הָאָב יִשַׁק עַתָּה בָּנָיו הַיְשֵׁנִים!ּ

"אַלְלַי לִי! אַנְחוֹתָיו תְּהִימֶנָּה,

מַה־שֻׁפּוּ עַצְמוֹתֵיכֶם, אִשְׁתִּי וּבָנָי!

רַק גַּל שֶׁל עֲצָמוֹת מוּטָל לְפָנָי,

אֵין בָּשָׂר אֵין עוֹר, וְהָעֲצָמוֹת תִּחְיֶינָה!

עֲצָמוֹת יְבֵשׁוֹת, מֵרְאוֹת הָהּ אֶבָּהֵל!

אֶרְאֶה וְאֶרְאֶה בְּצַלְמֵי בַלָּהוֹת:

פֹּה אַנְשֵׁי בֵיתִי, שָׁם כָּל בֵּית יִשְׂרָאֵל,

תִּהְיֶינָה אִם גַּם אֵינָן לְחַיִּים תּוֹצָאוֹת!..

הָהּ, בָּנִים עֲזוּבִים, הָהּ אֻמָּה עֲשׁוּקָה!

מִדַּת קְשִׁי יוֹמְכֶם מֵאֶרֶץ אֲרֻכָּה;

אֵין צֳרִי לָחֳלִי, תְּרוּפָה לַמַּכָּה!

רְאֵה אֲדֹנָי, צֹאן יַעֲקֹב הַנּוֹתָרוֹת

נָתַתָּ בְּאַפְּךָ לַכְּפִירִים לְבָרוֹת,

הֲכָלָה אַתָּה עֹשֶׂה בְּעַמְּךָ הַדַּכָּא?!..




ב


תִּכְלֶינָה לְסַפֵּר עֵינַי בִּדְמָעוֹת

תּוֹלְדוֹת אִישׁ זֶה, חַוָּתוֹ וַעֲמָלוֹ;

הוּא נוֹלַד לְהוֹרָיו בִּתְקוּפַת־בַּלָהוֹת

עֵת שָׁלַט הָאָדָם בָּאָדָם לְרָָעָה לוֹ.

אִם אָמְנָם שְׁנוֹת גֶּבֶר, הַלֵּילוֹת וְהַיָּמִים,

כָּל־יְמֵי צְבָאוֹ מֵאָז תֵּבֵל נִבְרָאָה,

הָיוּ מֵעוֹלָם תְּקוּפַת בַּלָּהָה,

תְּקוּפַת חֲמַס אַחִים עֲשׁוּקָה בְדָמִים…

פֹּה אַכְזָר יְדַכֵּא אֶבְיוֹנִים בַּשָּׁעַר,

יַעַן הֵם אֶבְיוֹנִים, חַלָּשִׁים מִמֶּנּוּ;

שָׁם עַם כֻּלוֹ חֶסֶר, עַם שָׁפֵל וָבָעַר

יַעֲשֹׁק עַם־רָם יַעַן שָׁפֵל אֵינֶנּוּ…

שָׁם מִזְרֹעוֹת עָרִיץ יְשַׁוְּעוּ עֲשׁוּקִים

מִתַּחַת שׁוֹט נוֹגֵשׁ – מְעֻנִּים וּמֻכִּים,

יְשַׁוְּעוּ – לֹא יֵעָנוּ, חֲנִינָה לָהֶם אָיִן;

וּבְנֵי אָדָם בִּמְנוּחָה הַקּוֹלוֹת יִשְׁמָעוּ,

“הוֹסִיפוּ לְהַכּוֹת!” עוֹד רַבִּים יִקְרָאוּ,

“הַכּוּ אַל תַּחְמֹלוּ, אַל תָּחֹס כָּל־עַיִן!”

כָּל־זֹאת יַעֲשֶׂה אָדָם לְאָח לוֹ וָרֵעַ

אַךְ אִישׁ לְפִי טַעְמוֹ יַצְדִּיק מַעֲשֵׂהוּ,

“לָזֶה הֵרַע בְּמִשְׁפָּט יַעַן הִרְבָּה הָרֵעַ”

“אֶת־זֶה הִכָּה בְישֶׁר כִּי חוֹטֵא הִנֵּהוּ”…

גַּם אַכְזָר, גַּם עָרִיץ, גַּם עַם כֻּלּוֹ חֶסֶר,

כִּי יַרְבּוּ הַוּת יִצְטַדְּקוּ עַל־פִּי טַעֲמָם…

אוּלָם מַה־יַּעֲנוּ אֹכְלֵי שְׁאֵר עַמָּם

שֹׁחֲטֵי יְלָדִים בְּנֵי שְׁמֹנֶה וָעֶשֶׂר?!..

וּבְכֵן תְּקוּפָה זֹאת נוֹרָאָה וַאֲיֻמָּה,

כָּמוֹהָ לֹא הָיְתָה עַל־פְּנֵי הָאֲדָמָה,

בְּדַם־יַלְדֵי עַמִּי הֲלֹא הִיא רְשוּמָה

בְּסֵפֶר הַתּוֹלָדָה – זְעֵיר פֹּה, זְעֵיר שָׁמָּה.

בִּהְיוֹת הָאִישׁ יֶלֶד בֶּן עשר שנים

מִּזְרֹעוֹת הוֹרָתוֹ גְּזָלֻהוּ “חַטְפָּנִים”

וַּיַעֲבִירֻהוּ לַמֹּלֶךְ בַּעֲבוּר נַעֲלָיִם…

מֵעָקַת־לֵב מֵתָה אִמּוֹ הָאַלְמָנָה

וְהוּא עָבַד בַצָבָא כִּשְׁלִישׁ מֵאָה שָׁנָה.

עָבַד – וַיָּשָׁב מִשָּׁמָּה בַּחַיִּים!…

אֵלֶה תּוֹלְדוֹת אִישׁ זֶה בִּקְצָרָה,

אַךְ מִדַּת קְשִׁי יוֹמוֹ מֵאֶרֶץ אֲרֻכָּה.

כִּי זֵדִים עָרִיצִים שָׂמוּהוּ מַּטָּרָה

לְחִצֵי אֱמוּנָה בְּדָמִים עֲשׁוּקָה…

בְּאֶרֶץ סִבִּרְיָה שָׁם קָרָה מִמְּזָרִים

שׂרֶרֶת בְּתָקְפָּהּ עִם סֹעָה וָסָעַר,

מֵעֶרֶשׂ יַלְדוּתוֹ שֻׁלַּח הַנָּעַר,

שָׁם נִתַּן “לְדֹד” פָּקִיד עַל־הַנְּעָרִים,

וְרַחֲמֵי דֹדֵהוּ מִי זֶה יְתַנֶּה!

הוּא הוֹרָה בְאַהֲבָה חֲנִיכוֹ הַיָּלֶד,

כִּי יַחֲלִיף הָאֵל וְדָתוֹ יְשַׁנֶּה

לְבִלְתִּי עוֹד יִהְיֶה שֶׂה אֹבֵד בֶּחָלֶד…

אַךְ לִהְיוֹת שֶׂה אֹבֵד, כִּבְשָׂה נִדָּחָה,

בָּחַר הַיֶּלֶד מִצֵּאתוֹ לָרְוָחָה

אֶל עוֹלָם שֶׁחָפְשִׁי מֵעָקַת יַד רָשָׁע…

כִּי זָכַר הַיֶּלֶד פֵּרוּשׁ כָּל־מִלָּה

לוֹ שִׁנְּנָה אִמּוֹ מִסֵּדֶר הַתְּפִלָּה:

תּוֹרָה צִוָּה לָנוּ משֶׁה, מוֹרָשָה…





רָאָה הַ“דֹּד” חֲנִיכוֹ מַקְשֶׁה עֹרֶף

וַיֶּהֱמוּ רַחֲמָיו לְנַפְשֹׁו הַנִּדָּחַת,

וְעָרֹם מִבֶּגֶד הִשְכִּיבֹו בַחֹרֶף

בְּרֶפֶת בְּנֵי בָקָר לַשֹּׁקֶת מִתָּחַת.

אוּלַי אִוְַּלּתוֹ יָסִיר מִמֶּנּוּ

הַקֶּרַח הַצְּפוֹנִי הַחוֹדֵר לְעַצְמוֹתָיו,

וְעָזַב עַד מְהֵרָה אֱמוּנַת אֲבוֹתָיו,

וְאָז גַּם קְשִׁי יוֹמוֹ לָנֶצַח יַעַזְבֶנוּ…

אַךְ דֹּדוֹ הָרַחוּם שָָׁגָה וַיִּוָּאֵל

כִּי טָפַשׁ מֵהָבִין לֵב זֶה הַבָּעַר,

כִּי כֹחַ הַמַּכְבִּים, רוּחַ יִשְׂרָאֵל

וְדַם גּוֹלֵי סְפָרַד צְפוּנִים בַּנָּעַר!…

כֹּחַ הַמַּכְבִּים לְנַצַּח עשְׁקֵהוּ,

רוּחַ יִשְׂרָאֵל לְכַלְכֵּל מַחֲלֵהוּ

וְדַם גּוֹלֵי סְפָרַד לָמוּת עַל דָּתוֹ:

הַתְּכוּנוֹת הָאֵלֶה יִינַק כָּל־יֶלֶד

מִזֶּרַע הַיְּהוּדִים לִיהֻֻדִּיָה יִוָּלֶד,

בְּרֶגַע הִוַּלִדוֹ – מִשְּׁדֵי אֵם הוֹרָתוֹ




ג


בִּרְאוֹת “דֹּר” הַיֶּלֶד כִּי אָבְדָה תִקְוָתוֹ,

יָמִים נָגֹזוּ וַעֲצָתוֹ לֹא קָמָה,

"קְשִׁי עָרְפְּךָ אַכְנִיעַ! – קָרָא בַחֲמָתוֹ

אֲעַנְּךָ בְּפַחַד בְּרָעָב וּבְצָמָא!" –

וּבְחָרְבָּה בַּיַּעַר, שָׁם זְאֵבִים וּכְפִירִים

יִתְעוּ לִבְלִי אֹכֶל, יְשַׁחֲרוּ לַטֶּרֶף,

מִבְּלִי הַאֲכִיל וְהַשְׁקֵה מִבֹּקֶר עַד עֶרֶב,

סָגַר אֶת־יַלְדוֹ יָמִים כַּבִּירִים…

שָׁם שָׁכַב בַּפִּנָּה הַיְּהוּדִי הַקָּטָן

וְרַעְיוֹנֵי בַלָּהוֹת לִבּוֹ לָחְצוּ,

זָכַר כִּי לִילִית וְסַמָּאֵל הַשָּׂטָן

בֵּין שֵׁדִים וְלֵצִים שָׁמָּה יִרְבָּצוּ…

וּכְרֶגַע גַּם מַרְאֵיהֶם רָאָה הַנָּעַר.

אַף שָׁמַע קוֹל נַחֲרָם בִּשְׁרִיקוֹת הַסָּעַר,

וּבְשָׂרוֹ הַקָּפוּא סָמַר מִפָּחַד…

הָהּ אָבִי!… הָהּ אִמִּי!… בְּחֶרְדַת־לֵב קָרָא,

וַיַּחֲבשׁ פָּנָיו בָּרִצְפָּה הַקָּרָה,

וַיִּישָׁן וַיַרְא שְׁנֵיהֶם גַּם יָחַד…

גּוּף עַצְמוֹת אָדָם־עֲצָמוֹת נִחָרוֹת,

עֲטוּף בְּטַלִּית נִצָּב לְפָנָיו,

מֵחוֹרֵי עֵינָיו דְּמָעוֹת נִגָּרוֹת,

וּכְרֶגַע הוּא נֹפֵל אַרְצָה עַל־פָּנָיו;

יִשַּׁק הַיֶּלֶד וִידַבֵּר בְּנַחַת:

אַל נָא, בֶּן־אוֹנִי, אַל תִּירָא, תִּבָּהֵל,

הִנֵּה לֹא יָנוּם שֹׁמֵר יִשְׂרָאֵל,

הוּא יַצִּיל מַאֲמִינָיו מֵרֶדֶת שָׁחַת!

“שֹׁמֵר יִשְׂרָאֵל” דַּע כִּי הִנֵּהוּ

אֵל יָחִיד וּמְיֻחָד, בְּלִי בֵן וּבְלִי אִשָּׁה…

רַק זֵדִים בְּנֵי נָבָל, לא יָדְעוּ דַרְכֵּהוּ,

יָצְרוּ לוֹ בֵן מֵעַלְמָה מְבִישָׁה…

אַל נָא, בֵּן יַקִּיר, יְפֻתֶּה לִבֶּך,

לַעֲזוֹב אֵל יָחִיד, אֱלֹהֵי אֲבוֹתֶיךָ,

וְלַאֲמִין בַּבֵּן לוֹ יָצְרוּ בְנֵי אָדָם;

אֱלֹהֵינוּ הַיָּחִיד יִשְׁכֹּן מְרוֹמִים

הוּא דַיַן אַלְמָנוֹת וַאֲבִי יְתוֹמִים

וְדַם עֲבָדָיו הַשָּׁפוּךְ עוֹד יִנְקוֹם מִיָדָם!…

הִנֵּה קַר לְךָ בְּנִי, גְוִיָּתְךָ רֹעֶדֶת…

בְּטַלִּיתִי אֲחַמֶּמְךָ עַד הָאִיר הֶחָרֶס…

וּרְאֵה נָא עַל־יָדְךָ אִמְךָ עֹמָדָת,

הִיא תַמְתִּיק לְךָ שְׁנָתְךָ בְּשִׁירֵי עָרֶשׂ…

דּוּמִיָּה שַׁלֶּטֶת מִסְפַּר רְגָעִים –

וּבְטַלִּיתוֹ בֶּן־אוֹנוֹ יָדָיו תַּעֲטֵפְנָה,

וְטִפּוֹת זֵעָה כַטַּל מַהֵר תִּטֹּפְנָה

מִמֵּצַּח הַיָּשֵׁן מֵחֹם מָתוֹק וְנָעִים…

והִנֵּה גַּם אִמּוֹ וְעַל־רֹאשָׁהּ יָדֶיהָ,

יֹשֶבֶת עַל־יָדוֹ אֲבֵלָה, נֶאֱנָחַת.

תַּכְרִיכֶיהָ רְטֻבִּים מִדִּמְעוֹת עֵינֶיהָ

וּבְרַחֲמֵי אֵם שְׁכוּלָה אֵלָיו מַשְׁגָּחַת…

"אֱלהִים מִקֹּדֶשׁ לְךָ יִשְׁלַח עֶזְרָתוֹ;

תִּקְרָא בִמְחוֹתָה מֵעַל מִצְחוֹ זֵעָתוֹ,

וּמֵעֻזּוֹ יְאַזֶּרְךָ עֹז וְתַעֲצֻמוֹת!…

וּתְנַשֵּׁק הַיֶּלֶד בְּאַהֲבָה נִצַּחַת,

וּשְׂחוֹק צַדִּיק וְטוֹב לוֹ, שְׂחוֹק עֵדֶן וָנַחַת,

יִרַחֵף לִרְגָעִים עַל־שְׂפָתָיו הַצִּנוּמוֹת…




ד


שְׁכַב יְהוּדִי עָשׁוּק, שְׁכַב בֶּן־יִשְׂרָאֵל,

וִיהִי נֹעַם אֵל רַחוּם עָלֶיךָ!

מִשְּׂמֹאלְךָ גַּבְרִיאֵל, מִיְמִינְךָ מִיכָאֵל,

שְׁכִינַת הָאֵל עַל רֹאשֶׁךָ!…

שְׁכַב בַּר נְדָרַי, תְּהִי שְׁנָתְךָ מְתוּקָה

כִּלְפָנִים בֵּין כְּסָתוֹת וְכָרִים…

הִנֵּה גַּם בְּרֹך עֶרֶשׂ שְׁנַת זֵדִים עֲשׁוּקָה

וַעֲרֵבָה עֲלֵי אֶבֶן שֵׁנָה לַיְשָרִים!

הִתְחַזֵּק, בֵּן יַקִּיר, שִׂים בַּרְזֶל לִבֶךָ

לָשֵׂאת בִּמְנוּחָה עָקַת רְשָׁעִים.

הֱיֵּה אֵת לְפָנִים הָיוּ אֲבוֹתֶיךָ,

עַזִּים מִנְּמֵרִים, גִּבּוֹרִים מִלְּבָאִים!

כְּמוֹהֶם כָּמוֹךָ צָוָּאְרְךָ הוֹשִׁיטָה

אֶל חֶרֶב הֹרֶגְךָ עַל־דָּתֶךָ;

וּקְרָא “שְׁמַע יִשְׂרָאֵל” קֹדֶם הַשְּׁחִיטָה,

וְהִתְפַּלֵּל כִּי יֵרָצֶה דָּמֶךָ…

שְׁכַב ­­­­­­­­­יֶלֶד נֵעֲרָץ, שְׁכַב בֶּן גִּבּוֹרִים,

בֶּן אֻמַּת עֲזַרְיָה, חֲנַנְיָה וּמִישָׁאֵל!

אַל תִּירָא מֵרְשָׁעִים סַבּוּךָ כִדְבֹרִים.

אַמֵּץ אֶת־לִבְּךָ בְּשֵׁם קְדֹש יִשְׂרָאֵל!…

הוא אֱלֹהֵינוּ זֶה בָּחַר בְּעַמּוֹ

לְשׂוּמוֹ בָאָרֶץ גּוֹי אֶחָד, גּוֹי תְּהִלָּה,

וְאִם רֶגַע יוֹכִיחַ אֹתוֹ בְּזַעְמוֹ,

הִנֵּה לָאֵל מֵעָוֶל חָלִילָה!

הוּא דַיַּן אֱמֶת, הוּא שֹׁפֵט מֵישָׁרִים,

אֵין נֶפֶשׁ בְּלֹא מִשְׁפָּט נִדַּחַת מִפָּנָי

וְדִמְעוֹת עֲשׁוּקָיו מֵעָקַת אַכְזָרִים

הִנָּן בְּנֹאדוֹ תָּמִיד לְפָנָיו.

הוּא רֹאָן, הוּא יוֹדְעָן וְסוֹפְרָן לִרְגָעִים,

וּבוֹ גַּם דִּמְעָתוֹ תִּזַּל כַּמָּיִם.

וְאִם תִּרְאֶה הוּא מַאֲרִיך אַפּוֹ לָרְשָׁעִים

הֲלֹא הוּא אֱלֹהֵינוּ, אֵל אֶרֶךְ אַפָּיִם!…

הוּא מַחֲשֶׂה מֵעוֹלָם, יִדֹּם כְּאִלֵּם,

לֹא יְיַסַּר, עֵת נַחְפּץ, זֵדִים בְּזַעְמוֹ;

אַך הוּא גַּם אֵל קַנָּא, לוֹ נָקָם וְשִׁלֵּם,

וּמְשַׁלֵּם עַל־יֶתֶר לְצֹרְרֵי עַמּוֹ…

מִבְּלִי נִקְמַת אֵל עַל־כָּל טִפַּת דָּם,

לֹא תְכַסֶּה אַדְמָתוֹ הֲרוּגֵינוּ.

יָקִיא כָּל־טִפָּה גַּם קַרְקַע הַיָּם

וּתְהִי לְרַעַל מֵמִית בְּכוֹס מְרַצְחֵינוּ…

אַל תַּאֲמִין, בֵּן יַקִּיר, אַל תַּאֲמִין בִּקְּדשָׁם,

לו מִזְבְּחוֹת אָדָם מְלֹא תֵבֵל עָרְכוּ;

הִנֵּה הוּא וְכֹל קְדשָׁיו זֵר קוֹצִים עַל רֹאשָׁם,

לְאוֹת כְּלִמּוּת עוֹלָם עַל דָּמִים שָׁפָכוּ…

שְׁכַב יְהוּדָה עָשוּק, שְׁכַב בֶּן־יִשְׂרָאֵל,

וִיהִי נֹעַם אֵל רַחוּם עָלֶיךָ!

מִשְּׂמֹאלְךָ דָּנִיּאֵל, מִימִינְךָ מִישָׁאֵל,

וַחֲנַנְיָה וַעֲזַרְיָה עַל־רֹאשֶׁךָ!..

לִקְדֹשֵי אֵל אֵלֶּה הָשְׁלְמוּ אֲרָיוֹת

וְאֵשׁ מִתְלַקַּחַת בָּם לֹא שָׁלָטָה;

בְּזכוּתָם גַּם לְךָ אֵל יַעֲשֶׂה נִפְלָאוֹת.

וּמִשֹּׁד רְשָׁעִים תִּנָּצֵל גַּם אָתָּה!

בְּיָדָם, בֵּן יַקִּיר, רוּחֲךָ אַפְקִידָה,

לְבִלְתִּי יַד זֵדִים אֹתוֹ תַדִּיחַ…

גַּם אַתָּה שָׁמְרֵהוּ מִנִּטְפֵי מֵי־נִדָּה…

הֵהָרֵג וְאַל תַּאֲמֵן בְּיַשּׁוּ הַמָּשִׁיחַ!!!.

שְׁכַב יְהוּדִי עָשוּק שְׁכַב בֶּן־ישְׂרָאֵל,

וִיהִי נֹעַם אֵל רַחוּם עָלֶיךָ!

מִשְׂמֹאלְךָ גַּבְרִיאֵל, מִימִינְךָ מִיכָאֵל,

שְׁכִינַת הָאֵל עַל רֹאשְךָ!..




ה


וַיִּישַׁן הַיֶּלֶד שְׁנַת מְנוּחָה שַׁאֲנָנָה

וַיַּרְא בַּחֲלֹמוֹ בִקְעָה מְרֻוָּחַת

ועָלֶיהָ מִמָּרוֹם תַּמְטִיר עֲנָנָה

מַבּוּל דָּם גָּדוֹל וְאֵשׁ מִתְלַקָּחַת.

טִפּוֹת דָּם, טִפּוֹת אֵשׁ שֹׁטְפוֹת בְּשָׁאוֹן

וְאַדְמַת הַבִּקְעָה בְּרַעַשׁ תְּרַוֶּינָה,

וְרָאשֵׁי אֲנָשִׁים עֹלִים מִמֶּנָּהּ;

יָצִיצוּ כְעֵשֶׂב מִגַּלֵּי מַשָּׂאוֹן…

וּכְרֶגַע וְהַבִּקְעָה מְלֵאָה אֲנָשִׁים,

מַה שׁוֹנִים וַאֲיֻמִּים דְּמוּתָם וְתָאֳרָם!

יֵש אֲשֶׁר לָמוֹ רָאשֵׁי נְחָשִׁים

וּכְעוֹר הַנָּמֵר חֲבַרְבֻּרוֹת בִּבְשָׂרָם…

וְיֵשׁ וּפָנֵימוֹ פְנֵי גְדָיִים וּכְשָׂבִים

וּפִימוֹ כְפִיּוֹת זְאֵבִים וּכְלָבִים.

וּכְקַרְנֵי הַשּׁוֹר קַרְנַיִם בְּמִצְחָם…

שִׂמְלוֹתָם כְשִׂמְלוֹת כְּמָרִים, כֹּהֲנִים,

וּצְלָבִים עַל לִבָּם גְדוֹלִים וּקְטַנִּים,

עִם דְּמוּת פֶּגֶר תָּלוּי – עֻזָּם וְנִצְחָם…

צִלְצַּל פַּעֲמוֹנִים פִּתְאֹם יִשָׁמֵעַ,

הֶמְיָתוֹ אֲיֻמָּה, קֹרַעַת לֵב וָקֶרֶב,

הֶהָמוֹן יִתְחַלְחַל וְעַל־בִּרְכָּיו כֹּרֵעַ

וְיִצְלֹב בְיָדָיו דְּמוּת שְׁתִי וָעֵרֶב…

וְהִנֵּה הָאָרֶץ תַּחְתָּיו תֵּאָנֵחַ –

וּכְקִיטֹר הָכָּלוּא בְּקֶרֶב קַלַּחַת

תִּרְגַז הַשְּׁאֹל, תִּרְעַשׁ הַשַּׁחַת,

וּבֶטֶן הָאֲדָמָה בְּרַעַשׁ תִּפָּתֵחַ;

וּכְשֵׁד עֵיפָתָה מֵקִּרְבָּהּ יִתְפָּרֶץ

אִישׁ נוֹרָא וְאָיֹם מַרְגִּיז לְבָבוֹת;

שַׁעֲרוֹתָיו עֲלוּקוֹת וְתַבְנִית כָּל־שָׁרֶץ,

מִצְחוֹ נְחוּשָׁה וְעֵינָיו לֶהָבוֹת,

אַפּוֹ אַף לָבִיא, וּשְׁתֵּי דַלְתֵי פִיהוּ

כִּדְמוּת שְׁנֵי נְחָשִׁים אִישׁ עַל־אָחִיהוּ.

וּקְוֻצוֹת זְקָנוֹ כִּקְבֻצַּת תּוֹלָעִים…

שִׁנַּיִם כַּחֲרָבוֹת מִפִּיו נִשְׁקָּפוֹת.

וּנְקֻדּוֹת קֶצֶף שָׁם רֹתְחוֹת וְצָפוֹת,

וּפָנָיו, פְּנֵי נָמֵר, כַּדָּם מְתֻלָּעִים…

הִנֵּה הוּא צֹעֵד!.. תִּרְעַשְׁנָה הַגְּבָעוֹת,

כָּהוּ כּוֹכָבִים נֶחְבָּא סַהַר בַּחֹדֶש,

וּבְקוֹל כַּשַּׁחַל יַרְעֵם נוֹרָאוֹת:

“בְּשֵׁם הָאָב וְהַבֵּן וְרוּח הַקֹּדֶשׁ!..”


קָדוֹש! קָדוֹשׁ! אַלְפֵי קוֹלוֹת יִרְעָמוּ,


יְחִי טוֹרְקוֹמַדוּ, יְהִי שְׁמוֹ לְתִפְאֶרֶת!

תִּנָּשֵׂא תְּרוֹמֵם דַּת הַנִּצָּרֶת

וְצֹרְרֵי יֵשׁוּע יִסָּפוּ, יִתָּמּוּ!!.. –

…"בְּשֵׁם הָאָב וְהַבֵּן אַרְצֵנוּ תִּגָּאֵל

מִכַּף כָּל־הַמִּינִים וְרוּחוֹת הַטֻּמְאָה!

עָרוּ עָרוּ עַד הַיְּסוֹד יְהוּדָה וְיִשְׂרָאֵל

וּפֹה וּבַמָּרוֹם אַל תְּהִי לָהֵם תְּקוּמָה!!…

מַהֲרוּ מְשָׁרְתַי, הִתְאַזְּרוּ בִגְּבוּרָה,

עִרְכוּ הָעֵצִּים הַצִּיתוּ הַמְּדוּרָה!

יַרְעֵם טוֹרְקוֹמַדּוּ בְּקוֹל מְשַׁבֵּר סְלָעִים

קִרְבוּ כֹהֲנֵי אֵל, הֵתָיוּ טַבָּחִים,

הַגִּישׁוּ מַאֲכלוֹת, חֲנִיתֹות וּרְמָחִים;

הִנֵּה יוֹם הַדִּין לִשְׁפּוֹט חַטָּאִים!!.."

וּפִתְאֹם הַבִּקְעָה שִׁנְּתָה פָנֶיהָ

לְעִינֵי הַנִּפְעָם מֵחֶזְיוֹן הַלָּיְלָה,

חוֹמַת אֲבָנִים סָביב עָלֶיהָ

וְסִפּוּן נְחוּשָׁה יְכַסָּהּ מִלְמָעְלָה.

שָׁם מְעָרוֹת וּמְחִלוֹת חֲשֵׁכוֹת וְצָרוֹת

וְעַל יְתֵדוֹת בּכְּתָלִים תְּלֻאִים בְּנֵי אָדָם,

וְרָאשֵׁי חֲלָלִים עֲלֵיהֶן כּוֹתָרוֹת…

פֹּה תְלוּיָה אִשָּׁה בְּמִקְלְעוֹת שְׂעָרָהּ

שָׁדֶיהָ כְרוּתוֹת עַד רֵיאָתָהּ פְּנִימָה,

וּלְחוּצָה אֶל לִבָּהּ יַלְדָּה מְדֻקָּרָה

וְעֵינֶיהָ הַקָּמוֹת נְשׂוּאוֹת אֶל אִמָּהּ…

וְשָׁם מִסְּגוֹר בַּרְזֶל נִשְׁקָף אִישׁ מֵת

וּתְלוּיָה בְּעַפְעַפָּיו דִּמְעָה קֹפֵאת

וְעֵינָיו כִּמְפַלֵּל שָׁמַיְמָה נִטָּיוּ…

כְּוִיָּה בְּמִצְחוֹ בִּדְמוּת שְׁתִי וָעֵרֶב

וּכְתֹבֶת צָרֶבֶת שָׁם עָמֹק בַּקֶּרֶב:

"בֵּן כָּל הַכֹּפרִים יֹאבְדוּ, יכִלָיוּ!!!…

הִנֵּה נַעַר רָתוּק אֶל עַמּוּד רָבוּעַ,

מִכִּתְמֵי דַם אָדָם לוֹ מַרְאֵה אַרְגָמָן,

וּלְפָנָיו עַל־שֻׁלְחָן כְּמַסְמֵר נָטוּעַ

צֶלֶם זָהָב גָּדוֹל, מַעֲשֵׂה יְדֵי אָמָן,

וְעַל קָרְבַּן זֶה נִצָּב אַחַד הַכְּמָרִים;

"שְׁמָעֵנִי נָא, יקְרָא, שְׁמָעֵנִי הַעֵלֶם,

שִׂים כָּבוֹד לָאֵל הִשְׁתַּחֲוֵה לַצֶּלֶם,

לָמָּה אֶרְאֶה דָמֶיךָ כְּמַיִם נִגָּרִים?!.."


“צַלְמְךָ” נִבְזֶה! יַעֲנֵהוּ בְגָאוֹן,


רַק אֵל אֶחָד אֵדַע, וְאֵין אַחֵר בִּלְעָדָיו…

בַּלֵּה בְשָׂרִי וְעוֹרִי – אַךְ יַד הַכִּלָּיוֹן

לֹא תִשְׁלוֹט עַד עוֹלָם בְּנַפְשׁוֹת עֲבָדָיו!"

"אִי לְךּ בְּנִי, יַעֲנֶה הַבְּלִיָעַל,

יָקוֹם יַצִּילְךָ אֱלֹהֶיךָ מִמָּעַל,"

וּבְבֶטֶן הָעַמּוּד יַבְעִיר גֶחָלִים.

יִתְאַדָּם הָעַמּוּד… יְפַרְפֵּר הָעֶלֶם…

הַכָּמָר אֶל פִּיהוּ יַגִּישׁ הַצֶּלֶם…


“בּוּז לְאֱלִילְכֶם זֵדִים וּנְבָלִים!!”.


שָׁם תִּבְעַר מְדוּרָה, יַעֲלֶה עֲשָׁנָהּ

וּשְׁנַיִם נְזִירִים עוֹמְדִים עָלֶיהָ,

וְאִשָּׁה עֲנִֻגָּה, יָפָה כַלְבָנָה,

אֲסוּרָה לַקִּיר עִם שְׁנֵי יְלָדֶיהָ.


" נִחוּמֵינוּ, עִבְרִיָּה, נִכְמְרוּ עָליִךְ,


יְדַבְּרוּ הַקְּדֹשִים אֵלֶיהָ בְרָגֶשׁ;

אִם תִּבְזִי דָתֵנוּ, הֲיִי לָנוּ פִּילָגֶשׁ,

וְתִנָצְלִי מִמָּוֶת אַתְּ וִילָדָיִךְ…

"הָלְאָה אַנְשֵׁי זִמָּה, הָלְאָה זֵדוֹנִים,

לָמִיר כְּבוֹדִי בְקָלוֹן, הֶבֶל תִּיגָעוּ!!…

וּשְׁנֵי קְדשֵׁי־אֵל שְׁנַיִם כִּידוֹנִים

בְּלֵב הַיְלָדִים לְעֵין אִמָּם יתְקָעוּ…

ישָּׂאוּם וִירִימוּם עַד לֵב הַלַּהֶבֶת,

יִתְאַבֵּךְ הֶעָשָׁן, תִּתְנַשֵּׂא הַשַּׁלְהֶבֶת,

וְרֵיחַ צָלִי עוֹלֶה בְּאַף הַסּוֹרָרֶת…

כְּשִׁכּוֹרָה הִיא חָגָה, כֹּרְעָה גֹּוַעַת,ֹ

וְרִנְּנַת הַקְּדֹשִים בְּרֶגֶשׁ נִשְׁמַעַת:

“בְּשֵׁם הָאָב וְהַבֵּן וְדַת הַנַּצָּרֶת!..”

הִנֵּה אִישׁ זָקֵן, שַׂעֲרוֹתָיו כַּשֶּׁלֶג,

אֶל עַמּוּד מְסֻמָּר אָסוּר בִּכְבָלִים

וְדָמוֹ לְרַגְלָיו יִשְׁתַּפֵּךְ כְּפֶלֶג

עַל־עֶלֶם וְעַלְמָה שָׁם נֹפְלִים חֲלָלִים.

"הָהּ בֶּן זִקּוּנַי, יְחִידִי עֲלֵי אֲדָמוֹת!

הֲזֶה אַפִּרְיוֹנְךָ בְּיוֹם חֲתֻנָּתְךָ?..

הֲזֶה גוֹרַל פְּנִינָה בַּת גִּילְךָ תַּמִּתְךָ?

“בָּרוּך דַּיַּן אֱמֶת!” יִקְרָא וְיָמוֹת.

לֹא הַרְחֵק מֵהֵמָּה כִּידוֹן תָּקוּעַ

תּוֹךְ רקּוֹת עוּל־יָמִים, בֵּין שְׁנַיִם עַמּוּדִים.

וְעל־קְרָסִים הַתְּקוּעִים בִּגֵווֹ הַפָּצוּעַ

יְלָדִים נִשְׁחָטִים תְּלֻאִים עֲקֻדִים…

קוֹל נַחֲרָם אֵימָה יַרְגִיז שָׁמָיִם

וְאֵד יַעֲלֶה מִדָּמָם יְעַוֵּר עֵינָיִם

וְאִמָּם עַל־יָדָם בְּרַתּוֹק מְפַרְפָּרֶת.

נְזִירִים וּכְמָרִים וּ“צְלָמִים” בְּיָדָם

רָצִים הֵנָּה וְהֵנָּה לַעֲבֹד מַעְבָּדָם

“בְּשֵׁם הָאֵם וְהַבֵּן וְדַת הַנַַָּּצָרֶת!…”

הִנֵּה טוֹרְקוֹמַדּוּ!.. מַה־נּוֹרָא מַרְאֵהוּ!

עַל־נָמֵר הוּא רֹכֵב בְּגאֲוָה וְגָאוֹן,

לְפָנָיו אֵשׁ תֵּלֵךְ, לֶהָבָה אַחֲרֵיהוּ,

קֶטֶב מִימִינוֹ, מִשְּׂמֹאלוֹ כִּלָּיוֹן.

צַלְצַּל נְחֻשְׁתַּיִם יִשָּׁמַע מֵעֲבָרִים,

אַסִּירֵי עֳנִי וּבַרְזֶל הוֹלְכִים כִּצְלָלִים.

רַגְלֵימוֹ בַּסַּד, יָדֵימוֹ בִּכְבָלִים, –

בָּם נָשִׁים וְטַף, זְקֵנִים וּנְעָרִים.


יִרְעַד הַנַּעַר, תִּרְחַפְנָה עַצְמוֹתָיו,


יִתְמַהּ עַל־הִשָּׁנוֹת לְעֵינָיו הָרָעָה,

לוֹ סִפְּרוּ מֵעוֹדוֹ לְפָנִים אֲבוֹתָיו,

אֵת עָשֹוּ הַכְּמָרִים, בִּסְפָרַד הָרְשָׁעָה –

יְזַמְּרוּ הַכְּמָרִים, יִשְׁפְּכוּ שִׂיחָם

בִּתְפִלּוֹת בֶּן יִשַּׁי לְיַשׁוּ מְשִׁיחָם –

וְהַנְחוּשָׁה תְצַלְצֵל לְרַגְלֵי נֹשְׂאֶיהָ…

הִנֵּה גֵיא הַתֹּפֶת, עֵמֶק הָרְפָאִים!

הִנֵּה גַם הַמְּדוּרָה לִשְׂרֹף חַטָּאִים:

בְּשֵׁם הָאָב וְהַבֵּן וְהָאֵם בִּבְתוּלֶיהָ!!..

הִנֵּה טוֹרְקוֹמַדּוּ עוֹלֶה הַבָּמָתָה,

שָׁם מִרְיָם הָאֵם נִצֶּבֶת עַל־כַּנָּהּ;

יִשַּׁק יָדֶיהָ וִיחוֹנֵן תְּמוּנָתָהּ

וּתְפִִּלָּה לִבְנָהּ שְׂפָתָיו תַּבַּעְנָה.

גַּם הָאֲסוּרִים יֶהְגּוּ בִגְרוֹנָם

תְּפִלָּה לָאָב עַל־יַד הַמְּרַצֵחַ;

סְבִיבָם הַכְּמָרִים, מְשָׁרְתֵי מִזְבֵּחַ,

עוֹמְדִים וּמְחַכִּים לִפְקֻדַּת אֲדוֹנָם…

עֲצֵי הַמַּעֲרָכָה יָשֹׁקּוּ, יִנְאָקוּ,

הָלַּבָּה הוֹמִיָּה כִּמְבַכָּה הַשָּׁבֶר…

הַשְׁבוּיִם יִתְנַגְּשׁוּ, אִישׁ אָחִיו יִשָּׁקוּ

נְשִׁיקוֹת הַפְּרִידָה עַל־עֶבְרֵי פִּי קָבֶר.

"מָוֶת לַיְּהוּדִים יִשָּׂרְפוּ כֻלָּהַם!

יִרְעַם הַנָּבָל פִּתְאֹם כָּרָעַם,

הֶרֶג וְאַבְּדָן לַפְּלֵיטָה הַנִּשְׁאָרֶת!"

יָחוּשׁוּ הַנְּזִירִים, יֵחָפְזוּ הַכְּמָרִים,

יִסְחֲבוּ אֶל־הַמְּדוּרָה נָשִׁים וּגְבָרִים:

בְּשֵׁם הָאֵם וְהַבֵּן וְדַת הַנַּצָרֶת!!

וּבֵינוֹת לַפְּגָרִים וַעֲרֵמוֹת הַחֲלָלִים

יִרְאֶה הַיֶלֶד שְׁתֵּי תְמוּנוֹת נוֹרָאוֹת:

אִשָּׁה וְאִישׁ אֲסוּרִים בִּכְבָלִים

אֶל שִׁקּוּץ מְשוֹמֵם, צֶּלֶם בַּלָהוֹת,

הָאִשָּׁה לְבוּשָׁה תַכְרִיכִים לְבָנִים

וְעָטוּף בְּטַלִּית הַגֶבֶר עַל־יָדָהּ,

יִרְאֵם הַיֶּלֶד וְיֶחֱרַד חֲרָדָה –

הִנֵּה הוֹרָיו לְפָנָיו פָּנִים אֶל פָּנִים!

"הָהּ אָבִי, הָהּ אִמּי, גַּם אַתֶּם אֵיכָכָה

הֵנָּה נֶהְדַּפְתֶּם, אֶל מְאוּרַת עַקְרַבִּים?!"

-"אַשְׁרֵי הָאִישׁ כָּמוֹנוּ לוֹ כָכָה

וְשֵׁם קְדֹש יִשְׂרָאֵל יְקַדֵּשׁ בָּרַבִּים!

"כְּבָר עָרוּךְ הַתָּפְתֶּה, תִּבְעַר הַמְּדוּרָה!

ִּפְתאֹם יִשָּׁמַע קוֹל עַנּוֹת גְּבוּרָה,

מָוֶת לַיְּהוּדִים, כִּלְיוֹן עוֹלָמִים!!"

אַחֲרָיו יִרְעֲמוּ אַלְפֵי פַעֲמוֹנִים

וּכְמָרִים יִנְהֲרוּ הֲמוֹנִים הֲמוֹנִים,

וּקְדוֹשָׁם בְּרֹאשָׁם, חֲמוּץ בְּגָדִים מִדָּמִים!

“אֶחָד אֱלֹהֵינוּ וְאֵין אַחֵר זוּלָתוֹ!”

אֶל אָזְנֵי הַיֶּלֶד קוֹלוֹת יַגִּיעוּ,

וּמַה־יָפוּג לִבּוֹ, תִּגְדַל חֶרְדָּתוֹ,

בְּשָׁמְעוֹ בְּתוֹכָם קוֹל אִמּוֹ וְאָבִיהוּ!

רוּחוֹ תִפְּעֵם מִבַּלְהוֹת צַלְמָוֶת –

וְהִנֵּה “דֹרֵהוּ” נִצָּב לְפָנָיו

וּצְלָב בִּדְמוּת רֹמַח יְעוֹפֵף עַל־פָּנָיו:

“הִתְנַצֵּר אוֹ רֵד אֶל תּוֹךְ הַלַּהָבֶת!”

“אֶחָד אֱלֹהֵינוּ, זוּלָתוֹ אֵל אָיִן!”

שֵׁנִית קוֹל הוֹרָיו אָזְנוֹ קַשָּׁבֶת,

וּבְרֶגַע זֶה יִרְאֵם עַיִן בֶּעָיִן

נִשָּׂאִים עַל־מוֹט אֶל הַשַּׁלְהָבֶת…


"אֶחָד אֱלֹהֵינוּ בַּשָּׁמַיִם וָאָרֶץ!


קוֹל אָיֹם וְנוֹרָא מִלִּבּוֹ יִתְפָּרֶץ,

אַשְׁרֵיכֶם כִּי בָאתֶם, הוֹרַי, עַד הֲלוֹם,

עִמְּכֶם אָמוּתָה מוֹת יְהוּדִים יְשָׁרִים!"

וּפְעָמָיו לַמְּדוּרָה בַּחֲמַת כֹּחוֹ יָרִים,

יִקְפֹּץ וְיִיקַץ וְהִנֵּה – חֲלוֹם!…






ו


דּוּמָם כַּמָּוֶת וּבְְנֶפֶשׁ נֶעְכָּרֶת

אֶל קִירוֹת הַחָרְבָּה יַבִּיט הַנָּעַר,

שָׁם רוּחַ המִיָּה שִׁיר אֵבֶל שֹׁרָרֶת,

וּכְיַנְשוּף בַּבְּקִיעִים יְקוֹנֵן הַסָּעַר.

אַך גַּם קַוֵּי פָז אֶל עֵבֶר פָּנֵיהוּ

יִשְׁלַח בְּטוּבוֹ אֲבִי־כָל־הָמְּאֹרוֹת,

הָרֹאֶה נִפְלֵנוּ זֶה אַלְפֵי דֹרוֹת,

וְלֹא יִמְנַע בְּחַסְדּוֹ מֵעֲשוּקִים אוֹרֵהוּ,

נִפְעָם הַיֶלֶד, חֲלוֹמוֹ יְחִיתּוֹ,

אַךְ פִּתְאֹם יִתְעוֹרֵר, יַעֲזֹב מִשְׁכָּבוֹ,

וְאֶל חֹר שָׁם הַשֶּמֶשׁ יִגַּשׁ לְאִטּוֹ

וּבִתְפִלַּת הַשַּׁחַר יִשְפֹּךְ לְבָבוֹ…

קַיּוֹס וְנִירוֹן – נִיקוֹלַי וְנֶכְדֵּהוּ!

הַבִּיטוּ לַיֶּלֶד חִקְרוּ רוּחֵהוּ

וּרְאוּ כִּי קַשׁ יָבֵשׁ אֶגְרוֹף־עַזּוּתְכֶם!

הַיֶלֶד הַזֶּה – כָּל־בֵּית יִשְׁרָאֵל,

מֵחֶרֶב רִשְעֲתְכֶם לֹא יֵחַת, יִבָּהֵל,

יוֹם יָבֹא וְיִצְעַד־עַל חָרְבוֹת מַלְכוּתְכֶם!..

בֵּית טַחֲנָה גָדוֹל הַתֵּבֵל וְהַחַיִּים

לָכֶם הֶעָרִיצִים, אֲדוֹנִים עֲצֵלִים!

תְּנִיעֶינָה בְּלִי הֶרֶף גַּלְגַּלֵּי הָרֵחַיִם

דִמְעוֹת עֲשׁוּקִים וְזֵעַת עֲמֵלִים;

תִּטְחַנָּה קִמְחֲכֶם מִמִּשְּׁמַנֵּי הָאָרֶץ;

עַל־כֵּן עַם כֻּלּוֹ בְּעֵינֵיכֶם כַּתֹּהוּ…

אַךְ חַכּוּ, עָרִיצִים, יָמִים יָבֹאוּ

וְרוּחַ הָאָדָם מִפְּנֵיכֶם יִתְפָּרֶץ,

אָז אוֹי לָכֶם אוֹי, עֲשׁוּקִים בְּדָמִים!

מִכֶּם וּמֶמְשַׁלְתְּכֶם יִשְׁאַר גַּל־אֵפֶר,

וְעַם זוּ דִּכִּאתֶם – בֵּכְתֹב דִּבְרֵי הַיָּמִים,

צִיּוּנֵי קָלוֹן לָכֶם יַשְׁאִיר בַּסֵּפֶר…

הַיְּהוּדִי הַקָּטָן לָאֵל יַעֲרֹךְ תְּפִלָּה

וּכְקוֹל כִּנּוֹר נָעִים תְּצַלְצֵל כָּל מִלָּה

בִּידֵי נַגֵּן יֵדַע שִׁירֵי עַצָּבֶת.

אֶל יוֹצֵר הָאָדָם יִשָּׂא עַפְעַפָּיו,

לַחֲסָדָיו יְיַחֵל וּבוֹ כָל־שַׂרְעַפָּיו,

לֹא יִשְׁמַע כִּי הַדֶּלֶת עַל־צִירָהּ סוֹבָבֶת…

וּכְבָר רֶגֶל “דֹּדוֹ” הַמִּפְתָּן עֹבֶרֶת

וּרְצוּעָה בְיָדוֹ עָבָה, מָשְׁזָרָה,


הָרְצוּעָה הַזֹאת הִיא רִצְפָּה בֹעֶרֶת


מִמּוקְדֵי סְפָרַד לְרוּסְיָה נִשְאָרָה..

וְהִיא הִיא הַשַּׁרְבִיט בִּידֵי מְלָכֵיהָ

מִימֵי יָאהָאן הַ“נּוֹרָא” עַד משְׁלָהּ עָתָּה,

הַ“צַּר” הַמִּתְחַבֵּא כַעֲטַלֵּף בַּעֲלָטָה

וּמוֹצִיא אֱלֵי הֹרֵג מִבְחַר בָנֶיהָ –

…בָּרוּךְ חַי לָעַד וְקַיָּם לָנֶצַח"…

וְדדוֹ יִתְקַצֵּף, אֵשׁ תּוּקַד בְּאַפּוֹ,

“קֶטֶב יְשָׁדךָ, כֶּלֶב עַז מֵצַּח!”

יִקְרָא וִיסלְסֵל הָרְצוּעָה בְכפוֹ.

“בָּרוּךְ פּוֹדֶה וּמַצִּיל, בָּרוּךְ שְׁמוֹ”…

יוֹסִיף הַיֶּלֶד תְּפִלָּה לֵאלֹהוֹ,

וּפִתְאֹם הָרְצוּעָה בִּבְשָׂרוֹ נֹגָעַת…

– "הִתְנַצֵּר בֵּן מֵבִיש! יִשְׁמַע קוֹל זַעָם,

וְאִם אַיִן הָרְצוּעָה תֶּהֶלְמְךָ כְּרַעַם

וּמִרְמָס לְרַגְלַי תְּהִי כַתּוֹלָעַת"



…“הָאָב הָרַחְמָן הַמְּהֻלָל בְּפִי עַמּוֹ”…

יִקְרָא הַיֶּלֶד וְיִשָּׂא מַכָּתוֹ,

וְדֹדוֹ הָרַחוּם לוֹ יוֹסִיף בְּזַעְמוֹ

עוֹד מִַכָּה אַחַת בִּקְצֵה־רְצוּעָתוֹ,

"הִנֵה שְֹכַר עַזּוּתְךָ רְצוּעָה בֹעֶרֶת!

וְיַלְדּוֹ לָאָרֶץ תַּחְתָּיו כֹּרֵעַ,

פָּנָיו יִתְעַוְּתוּ כִּפְנֵי אִישׁ גֹּוֵעַ,

וּמְרוּצַת דָּמוֹ מִלֶכֶת נֶעֱצָרֶת…

"גַּם עָנוֹג הַכֶּלֶב, רַךְ הוּא כְּרפֶשׁ,

חַלָּשׁ מִפַּרְעֹש עָשָׂהוּ אָבִיהוּ!

יִקְרָא דֹדֵהוּ בִּצְחוֹק וּשְׁאָט נֶפֶשׁ

וְנִטְפֵי יֵין־שְׂרֵפָה יַטִּיף אֶל־פִּיהוּ,

דִּמִּיטרִי בֶן־פֶּטֶר, אִישׁ רוּסִי מִלֵּידָה!

הַאֻמְנָם לֹא תָבִין, הַאֻמְנָם לֹא תֵדָע

כִּי רְצוּעָה בַת־אַרְצְךָ אֵינֶנָּה כְנַף־דְּרוֹר?

עָרִיץ אַחַר עָרִיץ חוּטִים בָּהּ טָוָה,

וַתִּכְבַּד הָרְצוּעָה – גַּם יָדְךָ מֶה־עָבָה!

וְלָמָּה תִּתְרַגֵּז בֶּן־פֶּטֶר חֲמוֹר?"



יִיקַץ הַיֶּלֶד וּבְחָזְקָה יְזוֹרֵר

וְקוֹל שִׁעוּל אַדִּיר יִפְרֹץ מִלִּבּוֹ

מַשְׁקֵה הָרוּסִים מִבַּקְבֻּק הַצּוֹרֵר

לֹא יִתֵּן לָאֻמְלָל לָמוּת בְּאִבּוֹ

מַשְׁקֵה רוּסִיָּה, פְּלָאֶיךּ מֶה רָבוּ!

הֲלֹא אַתָּה מֵעוֹלָם שִׁקּוּי־רְפוּאָה

גַּם בִּידֵי מְלָכֶיהָ תּוֹפְשֵׂי הָרְצוּעָה –

יַשְׁקוּ עֲשׁוּקֵיהֶם וְלֹא יִשְׁתּוֹבָבוּ!…

יִיַקץ הַיֶּלֶד, לְאַט יִפְקַח עָיִן

וּשְׂפָתָיו תַּבַּעְנָה דְבָרִים מוּזָרִים;

"הִנֵּה הַמְּדוּרָה… זוּלָתוֹ אֵל אָיִן…

עִמְּכֶם אָמוּתָה מוֹת יְהוּדִים יְשָׁרִים"…


כֶּלֶב מְשֻׁגָּע! דִמִּטְרִי יַבִּיעַ


וְרֹאשׁוֹ רֹאשׁ קִפּוֹד, בְּרֹגֶז יַנִיעַ,

הָבָה אֶשָּׂאֲךָ אֶל בֵּית־חָפְשִׁית הַצָּבָא!

שָׁם, יְהוּדִי אָרוּר, בְּמֶרְיְךָ תָמוֹת,

וְעַל כְּתֵפָיו יַנִּיחַ גַּל הָעֲצָמוֹת,

וְיִשָּׂא אֶת טַרְפּוֹ כִּזְאֵב מֵעֲרָבָה.




ז


חֲמִשִּׁים שָׁנָה מֵאָז נְשִׁיָּה צָלָלוּ

וּמְצָרֵי הַתֹּפֶת לֹא מָשְׁלוּ בַיָּלֶד,

בִּלְחָיֵי מְעַנָּיו עֲשָׂבִים כְּבָר עָלוּ

וְהוּא בָּרִיא וָחַי כְּאִלוּ עַתָּה יִוָּלֶד.

יִשְׂרָאֵל כַּסְנֶה בּוֹעֵר מֵעוֹלָם,

אַךְ אֵיי הָאוֹר שׁוֹלֵט בְּרוּחוֹ וּבְשָׂרוֹ;

הֶבֶל הַמְּדוּרוֹת, אֵל חָפֵץ בִּיקָרוֹ,

וַיְשִּׂימוֹ סַלְמַנְדְּרָא לְכָל בָּאֵי עוֹלָם:

וַיִּגְדַּל הַיֶּלֶד וַיְהִי הוֹד בַּחוּרִים

וַיָּשָׁב בְּשָׁלוֹם מֵעֲבוֹדַת מַלְכֵּהוּ

אל תְּחוּם מוֹשַׁב עַמּוֹ, כְּלוּב הָאֲסוּרִים,

הָאוֹכְלִים בַּמָּצוֹק אִישׁ בְּשַׂר רֵעֵהוּ…

וִיְהִי לְבַעַל אִשָּׁה וּבָנִים חֲמִשָּׁה,

וַתִּתֶּן לוֹ מִשְׂרָה מַמְלֶכֶת הָרִשְׁעָה

בְּלִשְׁכַּת הַפְּקִידִים עַל יַד שַׂר הַפֶּלֶךְ;



וַיְהִי הָאִישׁ בָּרוּךְ1 חַיֵי רֵישׁ וָלַחַץ



בֵּין רוּסִים רְשָׁעִים, חֲבֵרִים בְּנֵי שַׁחַץ,

על לֶחֶם נַקּוּדִים מִפַּתְבַּג הַמֶלֶךְ.

וּבָרוּךְ לֹא הִתְאוֹנֵן עַל רֹעַ מַצָבוֹ,

וַיִּשְׁמַח כִּי נִשְׁאַר בֵּן נֶאֱמָן לְדָתוֹ,

וְעַתָּה כִּי זָקֵן וְכֹחוֹ עֲזָבוֹ,

הֵן מִשְׂרַת הַמֶּלֶךְ תְּכַלְכֵּל שֵׂיבָתוֹ,

אַךְ מַה תִּקְוַת עִבְרִי בְּרוּסְיָה הַשְּׁפָלָה?

בְּעֶצֶם יוֹם בָּהִיר בָּאַתְהוּ אִגֶּרֶת:

"אִם תְּמָאֵן לְקַבֵּל דַּת הַנַּצֶרֶת

תּוּסַר מִכַּנְּךָ מִיּוֹם זֶה וָהָלְאָה"


* * *


עָל אַדְמַת רוּסִיָּה הָעֲקֻבָּה מִדָּמִים

בְּשַׁוְלַן הָעִיר, בְּפִנָּה גַלְמוּדָה,

נִשְׁקָף שָׁם בַּיִת מָט מֵרֹב יָמִים

עוֹמֵד כְּזֵכֶר לְחֻרְבַּן בַּת יְהוּדָה…

שָׁמָּה בַּחָרְבָּה, בֵּין כָּתְלֵי חֻרְבָּנָהּ,

מִּתְהַלֵּךְ אִישׁ שֵׂבָה אָנֶה וָאָנֶה,

כַּהֲלוֹךְ יִרְמְיָהוּ בֵּין קִבְרֵי צַלְמָוֶת

עֲנָנִים עַל מִצְחוֹ יְזָרוּ מַחֲשַׁכִּים,

וּכְמוֹ מִטְרוֹת עֻזָּם מֵעֵינָיו מְפַכִּים

אֶגְלֵי מֵי־רֹאשׁ, דִמְעוֹת עַצָּבֶת…

שָׁמָּה בַּפִּנָּה עַל מִטָּה מִתְנֹדֶדֶת

וְסַפְסָל צוֹלֵעַ קְרָשִׁים נִשְׁעָנִים;

עֲלֵיהֶם רַעְיָתוֹ יְצוּעָהּ מְרַפֶּדֶת

וַחֲמִשָּׁה יְלָדִים עַל מַחְצֶלֶת שֶׁל קָנִים.

אוֹר כֵּהֶה שָׁם נֵר יָפִיץ מִמְּנוֹרָה

מֵרוּחַ בּוֹ יִגַּע שַׁלְהַבְתוֹ תָנוּעַ,

תִּתְנַשֵׂא, תִּתְאַבֵּךְ וְאוֹרָהּ זָרוּעַ

בִּנְטוֹתָהּ הַצִדָּה – אַךְ תֵּכַהּ מְהֵרָה…

תְּנְשֵּׂא תִּתְאַבֵּך בַּנֵּר הַשַּׁלְהֶבֶת

בִּנְטוֹתָהּ הַצִּדָּה לְעֵינֵי הֶעָני,

וּמַחֲשָׁבָה אֶל לִבּוֹ אַט מִתְגַּנֶּבֶת

אוּלַי הַצִדָּה אֶטֶה גַּם אָנִי?..

אֶטֶה הַצִּדָּה וְאוֹרִי יִבָּקֵעַ

וְהֹדְפַי יְקָרְבוּנִי כְּאָח לָהֶם וָרֵעַ,

אָז אֶשְׁכַּח עֲמָלִי, רָגְזָה וָנֶהִי…

לֹא! לֹא! יִקְרָא פִתְאֹם וּפָנָיו מִשְׁתַּנִּים,

דָּתִי לֹא אָמִיר מֵעָקַת מְעַנִּים!

עִבְרִי הָיִיתִי וְעִבְרִי עוֹד אֶהִי!..





לֹא אַחֲלִיף הָאֵל בְּעַד לֶחֶם וּבִרְיָה!

וְאִם פֹּה עָלַי שׁוֹאָה זֵדִים הֵמִיטוּ –

אֶל עֶרֶשׂ יַלְדוּתִי אָשׁוּב סִבִּרְיָה,

שָׁם רֵעִים יְשָׁנִים לִי עֶזְרָה יוֹשִיטוּ…

שָׁם אֶמְצָא לִי מַעֲמָד, אֶמְצָא מַרְגוֹעַ…

אוֹי! אוֹי! הִרְעִים פִּתְאֹם קוֹל אִשְׁתּוֹ הַיְשֵׁנָה

אֲרָיוֹת וּכְפִירִים, אֵלִי רְאֵה־נָא,

סוֹחֲבִים יְלָדַי!.. פֹּה קָדְקֹד!.. שָׁם זְרֹעַ!..

הָהּ אִישִׁי יִטָּרֵף!.. הָהּ, כָּלָה!.. הָהּ כָּלָה!..

וּכְרֶגַע וְתִיקַץ, תִּפְקַח עֵינֶיהָ,

תִּקְפּוֹץ לָאָרֶץ וּבְנֶפֶש נִבְהָלָה

תְּעוֹרֵר מִשְׁנָתָם חֲמֵשֶׁת בָּנֶיהָ.

"נוּסוּ! הִמָּלְטְוּ! תִּקְרָא כְּמִשְׁתּוֹלֶלֶת,

מִשִּׁנֵּי לְבָאִים מֵחֶרֶב אֹכֶלֶת,

וּבְטֶרֶם דָּמֵנוּ לָאָרֶץ יִשָּׁפֵךְ!…

אִישָׁהּ יִבָּהַל, רוּחוֹ נֶעְכֶּרֶת,

הֵרָגְעִי נָא, יִקְרָא, בְרָגְזָה נִסְתֶּרֶת,

חֲלוֹם הֶבֶל חָלַמְתְּ וּלְטוֹבָה יֵהָפֵךְ!…

הַיְלָדִים יֵעוֹרוּ וְאֶל אִמָּם יִשּׁמּוּ,

בֵּין כֹּה תִּזַךְ דַּעְתָּהּ, תַּכִּיר מַעֲמָדָה;

–אוֹדְךָ אֲדֹנָי כִּי זֶה רַק חֲלוֹם הוּא!

תִּקְרָא וּתְחַבְּקֵם בְגִילָה וּרְעָדָה.

"אַל תִּשְׁעִי רַעְיָתִי אֶל מַחֲזוֹת הֲבָלִים,

יִשְׁעֵנוּ בַצָרָה אֱלֹהִים וְלֹא נִירָא!..

הֵן עָלָה עַל־לִבִּי לָשׁוּב סִבִּירָה:

שָׁם יִתַּם עָנְיֵנוּ, יָנוּסוּ הַצְּלָלִים"…


הָהּ, בָּרוּךְ מָחְמַדִּי, הֶרֶף נָא הֶרֶף!


בַּחֲלוֹמִי בָּדֶּרֶך נֵלֵךְ כֻּלָּנוּ,

וּפִתְאֹם מֵאַיִן?.. עֲדַת חַיְתוֹ טֶרֶף…

וּכְלָבִים וּזְאֵבִים טוֹרְפִים אוֹתָנוּ…

"הַרְפִּי מֵהֶבֶל! – אִישָּׁה יַעֲנֶנָּה,

עוֹד רוּסְיָה אַרְצֵנוּ הַפְקֵר אֵינֶנָה,

וּמַלְכָּהּ, שָׂרֶיהָ – חֲתַת בְּנֵי עַוְלָתָה…

אֶל עֶרֶשׂ יַלְדוּתִי נֵלְכָה נָשוּבָה,

שָׁם אֶרֶץ רְחָבָה, נֹאכַל מִטּוּבָהּ"

וַתִּדֹּם הָאִשָּׁה וַתִּבְלַע דִּמְעָתָהּ…




ח


פְּנֵי קָדִים כַּתּוֹלָע, גַלֵּי־פָז יֵרֹמוּ;

הַרְחֵק שָׁם הַרְחֵק יָצִיץ אוֹר בָּהִיר,

וְכַדּוּר אֵשׁ מִתְנַשֵּׂא, מֶלֶךְ הַיּוֹם הוּא

זֶה בָּרָא אֱלֹהִים לְחַמֵּם וּלְהָאִיר.

וְלֹא בֹשָה הַחַמָּה, אֵשׁ יָהּ בָּהּ בֹּעָרֶת,

לִשְלֹחַ אֶת אוֹרָהּ גַּם לְרוּסְיָה הַחוֹטֵאת…

וּמֵעַל לְצַלְמֵי נִיזְנִי עַתָּה נִשֵּׂאת –

עִיר מְלֵאֲתִי רוּסִים וֶאֱלִילֵי הַנַּצָּרֶת…

עַל־רָזֵי לֵיל אֶתְמוֹל, תַּעֲלֻמוֹת הָעֶרֶב

קַוֶּיהָ הַמְּאִירִים יַזְהִירוּ, יִצְהָלוּ,

וְאוֹתוֹת בַּנֶּתֶר, כִּדְמוּת שְׁתִי וָעֵרֶב,

וְעַל רֹב בָּתֵּי נִיזְנִי לְאוֹרָהּ יִגָּלוּ…

וּבְקֶרֶב הָעִיר מִתְהַלְּכִים חֲשֻׁכִּים,

וּבְחֻבָּם יַסְתִּירוּ סוֹדוֹת עֲמֻקִּים,

עֵין מֵבִין בְּרוּחָם סֶלֶף מַכָּרֶת…

עֵינֵיהֶם מִסָּבְאָם בַּדָּם מְמֻלָּאוֹת,

כִּמְשַׁחֲרוֹת לַטֶּרֶף הֵן נָדוֹת וְנָעוֹת,

"כאֵשׁ טֶרֶם תּפרוֹץ בְּקִרְבָּם נֶעֶצָרֶת…

אֶל בֵּית מְלוֹן אֹרְחִים בְּטַבּוּר הַקִּרְיָה

סָר בָּרוּךְ וּבֵיתוֹ זֶה שָׁעוֹת אֲחָדוֹת:

דֶּרֶךְ עִיר נִיזְנִי הוּא הוֹלֵךְ סִבִּרְיָה

וּלְפֶרֶם2 הָעִיר פָּנָיו מוּעָדוֹת.

נַפְשׁוֹ תְפַרְפֵּר בֵּין יֵאוּשׁ וְתוֹחֶלֶת,

אִשְׁתּוֹ נֶעֱצֶבֶת חוֹלָה וַעֲיֵפָה;

שַׁלְוָה עַל פָּנֶיהָ וְלִבָּהּ כְּהַר־שְׂרֵפָה,

שָׁם רָגְזָה גֹעֶשֶׁת וְהִיא כְצֹהָלֶת…

הָהּ, מִי יְתַנֶּה עֲמַל הַנִּדְכָּאִים!

מִי יִסְפּוֹר נוֹדָם עַד בֹּאָם הֲלוֹם?!

הוּא נִלְחַם בְּרִישׁוֹ וּבְחֹסֶר אֶמְצָעִים,

וְהִיא מַה־נִּלְחֲמָה בְּמַרְאוֹת הַחֲלוֹם!…

וְעַתָּה כַּף רַגְלָם עַד כֹּה כִי הִגִּיעָה,

אָזְלָה מִכִּיסָם הוֹצָאַת הַנְּסִיעָה

וְאֵין מַכִּיר וּמוֹדַע מִימִינָם, מִשְֹּמֹאלָם…

“הָהּ, אִישִׁי, מַה־נַּעֲשֶׂה בֵּין נָכְרִים וְזָרִים?”

"אֵצֵא, יַעֲנֶנָּה, לְבַקֵּשׁ מַכָּרִים,

עִבְרִי בֵּין עִבְרִים לֹא אָבַד מֵעוֹלָם!".


* * *


חַיָּה וּמַגֵּפָה, פִּרְצַת הַר גָּעַשׁ,

יָם יוֹצֵא מִדְּלָתָיו, קִיטֹר מִמִּסְגָּרֶת,

כַּדוּר מִתּוֹתָח עַל כַּנְפֵי הָרָעַשׁ

וּמֵאַף אֵל זֹעֵם סוּפָה סֹעָרֶת,

מוּל רָעַת הָאָדָם מַה־נֶּחְשְׁבוּ הֵמָּה

כִּי יְשַׁלַּח מִפָּנָיו רֶסֶן אִישׁוּתוֹ

וְעַל יִצְרוֹ הַטּוֹב תָּעֹז נַבְלוּתוֹ, –

תַּחְתָּיו אָז תִּרְגַּז הָאָרֶץ מִמְּקוֹמָהּ…

וּבְנִיזְנִי הִיא רֹגֶזֶת כְּנֵפֶץ הר פְּרָצִים

נֹצוֹת מְרַחֲפוֹת, תִּשְׁלַגְנָה בַּשְּׁוָקִים,

קוֹלוֹת מִתְנַשְׂאִים, שְׁאוֹן פְּרָאִים וּמַפָּצִים

וְשַׁוְעַת עֲשׁוּקִים עוֹלָה עַד שְׁחָקִים.

נָשִׁים תְּשֻׁגַלְנָה, יְרֻטְּשׁוּ עוֹלָלִים

וַהֲרוּגִים וּפְצוּעִים בַּדָּם מְגֹלָלִים

וּכְלָבִים יִסְחֲבוּ אֲבָרִים אֲבָרִים.

לְשׁוֹן בְּנֵי אָדָם נִכְרֶתֶת מִפִּיהֶם

וּשְׁחוּטִים יְלָדִים לְעֵין אִמָּם וַאֲבִיהֶם.

שֶּהֵמָּה בַחוּץ יָזוּבוּ מְדֻקָּרִים…

עַל יַד הַחַלּוֹן אֵשֶׁת בָּרוּךְ נִצֶבֶת;

וַתֵּרֶא וַיִקְפָּא דָמָהּ בַּקֶּרֶב:

עֲדַת זֹבְחֵי אָדָם אִישָׁהּ סֹבֶבֶת

וּלְפִיהוּ מַגִּישָׁה צֶלֶם שְׁתִי וָעֵרֶב3

רֶגַע הוּא עוֹמֵד נִפְעָם עַל יָדָם,

פּתְאֹם יִתְנָעֵר וּבְגאֲוָה יִתְרוֹמֵם

יִירַק בְּבוּז בִּפְנֵי שִׁקּוּץ מְשׁוֹמֵם,

אֱלֹהֵי הַנַּצֶרֶת – אֱלִיל אֹכְלֵי אָדָם,

אוֹיָה! מִלִבָּהּ הִתְמַלֵּט קוֹל נוֹרָא,

וּפִתְאֹם הַדֶּלֶת בְּרַעַשׁ הִתְפּוֹצָצָה

וְאחַת יְחֵפִים, מִדָּמִים שִׁכּוֹרָה,

כְּקֶטֶב מְרִירִי הַחֶדְרָה הִתְפָּרָצָה.

“מָוֶת לַיְהוּדִים, אַבְּדָן וְכָלָה!”

רָעם קוֹל אָיֹם, וּבְלֵב הָאֻמְלָלָה

תָּקַע שַׂכִּינוֹ רַב הַטַּבָּחִים.

חֲמֵשֶׁת בָּנֶיהָ עִמָּה נִרְצָחוּ

וְדֶרֶך הַחַלּוֹן עִם אִמָּם הָשְׁלָכוּ

אֶל עֲרֵמַת הַחֲלָלִים, גַּל הַנִּרְצָחִים…


* * *


וְזָרַח הַשֶׁמֶשׁ, אוֹרוֹ מַה־יָּהֵל!

חֻמּוֹ מַה־נָּעִים, פָּנָיו מַה־שָּׁלוּ!

וּשְנֵים עָשָׂר חֲלָלִים, הֲרוּגֵי יִשְׂרָאֵל,

מוּל נֹגַהּ זַרְחוֹ בָּרְחוֹב יִתְגַּלְגָּלוּ!..

וְזָרַח הַשֶּׁמֶשׁ עַל־בָּרוּךְ הַקָּדוֹש,

הַנוֹפֵל שָׁם חָלָל וְעֵינָיו הַקְּפוּאוֹת

לְיוֹצְרוֹ בּוֹ דָבַק, שָׁמַיְמָה נְשׂוּאוֹת –

וְזָרַח הַשֶּׁמֶש וְתַחְתָּיו לֹא נָדוֹשׁ!!..

וּבָא הַשֶּׁמֶשׁ וְשָׁקַע הַיָּמָה,

וְעָלָה וְזָרַח בְּאַרְצוֹת מֶרְחַקִּים –

וַהֲרוּגֵי יִשְׂרָאֵל יִרְאֶה גַּם שָׁמָּה

וּדְמֵיהֶם מוּל פָּנָיו כַּמַּיִם מְפַכִּים…

הָהּ אַבִּיר יַעֲקֹב, נֵצַח יִשְׁרָאֵל!

עַד אָן בְּדַם עַמְּךָ תֵּבֵל תִּתְגָּאֵל

וְלֹא יִכְבַּד עָלֶיהָ פִּשְעָהּ עַד נִפְלָהּ!?..

עִלְזִי בַּת רוּסְיָה, לַחֲטֹא הוֹסִיפִי,

כָּל רֶגֶב מֵעֲפָרֵךְ בְּדָמִים הַחֲנִיפִי –

עֲשׁוּקִים יְשַׁוְּעוּ וְאֵל לֹא יָשִׂים תִּפְלָה!..
