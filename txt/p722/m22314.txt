


	
	שירו לַיוי שיר חָדָש
	






אֱמוּנָה יוֹצְרָה אֶצְלוֹ אֲמָנָהּ / וְעַד לֹא נוֹסְדָה אֶרֶץ הֱכִינָהּ:



וְעַד לֹא טִפְּחָה יָדוֹ שְׁחָקִים / בְּחֶבְיוֹנֵי מְאוֹרוֹתָיו
צְפָנָהּ:



מְשַׂחֶקֶת לְמוּל יוֹצְרָה בְּכָל עֵת / מְשַׁעְשַׁעַת בְּטוּבוֹ
שֶׁחֲנָנָהּ:



לְאוֹרָהּ חָפְרוּ כָל כּוֹכְבֵי אוֹר / וְהֵשַׁח כָּל בְּנֵי אֵלִים
גְּאוֹנָהּ:



וְעָבִים נָטְפוּ מַיִם בְּרִדְתָּהּ / וְאֶרֶץ רַעֲשָׁה יוֹם צוּר
נְתָנָהּ:



וְשַׁחַק זָע וְרָהוּ כָל שְׁכֵנָיו / וְצוּרִים נִתְּכוּ מֵאֵשׁ
חֲרוֹנָהּ:



וְהַשֶּׁמֶשׁ בְּצֵאתוֹ שָׁב לְאָחוֹר / וְעַיִשׁ שָׁכְחָה רִבְצָהּ
וְקִנָּהּ:



תְּדַמּוּהָ בְּטוּב אֶרֶץ וְאֶרֶץ / נְקַלָּה מֵעֲרוֹךְ נֶגְדָּהּ
פְּנִינָהּ:



הֲתַעְרוֹךְ לָהּ זְהַב אוֹפִיר וְסַפִּיר / וְצוּפָהּ אוֹ חֲבַצֶּלֶת
שְׁרוֹנָהּ:



וְאֵלֶּה טִיט וְעָפָר מִשְׁכְּנוֹתָם / וְהִיא עַל כּוֹכְבֵי שַׁחַק
מְכוֹנָהּ:



פְּרִי עֵצָהּ מְבַכֵּר לֶחֳדָשָׁיו / וְלֹא יֵבוֹשׁ לְעוֹלָם מַעֲיָנָהּ:



אֲשֶׁר נַפְשׁוֹ לְפַת חָכְמָה רְעֵבָה / וְלִרְווֹת מִמְּקוֹר בִּינָה
רְצוֹנָהּ:



רְדָה אֶל גַּן סְתָרֶיהָ וְתִמְצָא / פְּרִי חָכְמָה יְחַיֶּה לֵב
מְבִינָהּ:



וּמֵעֵינָהּ שְׁאַב לְךָ מֵי תְבוּנָה / תְּרַוֶּה נַפְשְׁךָ בוֹ
צִמְּאוֹנָהּ:



וְיִזְרַח בַּחֲדָרֶיהָ לְךָ אוֹר / אֲשֶׁר יִפְקַח לְעֵינָךְ
עִוְּרוֹנָהּ:



וְתִמְצָא דַעְתְּךָ דַעַת אֱלֹהִים / וְתָנוּחַ וְתִשְׁקוֹט
מִשְּׁאוֹנָהּ:



לְבוּשָׁהּ שָׂם אֱלֹהִים עֹז וְהָדָר / וְתוֹרַת חֵן וְחֶסֶד עַל
לְשׁוֹנָהּ:



עֲטֶרֶת פָּז וְלִוְיַת חֵן לְרֹאשָׁהּ / וְרָבִיד וַעֲנַק הוֹד עַל
גְּרוֹנָהּ:



וּבִשְׂמֹאלָהּ מְצוֹא עֹשֶׁר וְכָבוֹד / וְחֵי עוֹלָם בְּמַתַּת יַד
יְמִינָהּ:



וְעֵץ חַיִּים לְמַחְזִיק בָּהּ, וּמַרְפֵּא / לְנֶפֶשׁ חָלְתָה מֵרֹב
יְגוֹנָהּ:



וְלִמְעַקֵּשׁ דְּרָכָיו אֵשׁ וְגָפְרִית / תְּצַו עָלָיו לְהַמְטִיר אֶת
עֲנָנָהּ:



כְּאֵשׁ תָּאִיר וְכֵן תִּשְׂרוֹף, וְכֵן הִיא / עֲנָוִים יִמְצְאוּ
אוֹרָהּ וְחִנָּהּ:



וְלֵצִים יֵרְדוּ לִשְׁאוֹל בְּעֹצֶב / וְיִהְיוּ מַאֲכֹלֶת אֵשׁ
בְּבִטְנָהּ:



לְזֹאת אַשְׁרֵי אֱנוֹשׁ מַחְזִיק בְּחֻקָּהּ / וְעוֹשֶׂה מַאֲמַר דָּתָהּ
וְדִינָהּ:



וְאַשְׁרֵי אִישׁ לְמִשְׁמַעְתָּהּ יְהִי סָר / וְלֹא יַמְרֶה דְבַר
צוּרָהּ וְקוֹנָהּ:



תְּהִי עָלָיו בְּיוֹם צָרָה לְמֵלִיץ / וְיִתְכַפֵּר לְנַפְשׁוֹ בָהּ
עֲוֹנָהּ:



וְיִשְׁגֶּה בַאֲהָבֶיהָ וְיִשְׁקוֹד / בְּכָל לִבּוֹ עֲלֵי דַלְתֵי
מְעוֹנָהּ:



הֲכִי בָהּ תִחֲיֶה נַפְשׁוֹ, וְתַשִּׂיג / רְצוֹן צוּרָהּ וְלֹא תֶחְסַר
מְזוֹנָהּ:



וְאַשְׁרֵי אִישׁ כְּתָבָהּ עַל לְבָבוֹ / וְעַל סֵפֶר חֲקָקָהּ
וֶהֱבִינָהּ:



כְּתַבְתִּיהָ אֲנִי [אַבְרָם] צְעִירָה / יְלוּד [חַלְפוֹן] קְטַן
חַבְרֵי הֲמוֹנָהּ:



חֲתַמְתִּיהָ שְׁנַת … לְמִנְיָן / לְאוּם זָרָה תְהֶא סוֹף חִשְּׁבוֹנָהּ:



וְתִהְיֶה רֹאשׁ לְפִדְיוֹן אוּם עֲנִיָּה / עֲדֶן גּוֹלָה וְסוּרָה
מִמְּכוֹנָהּ:



עֲדֵי יַשְׁקִיף בְּרַחְמָיו אֵל, וְיִקְבּוֹץ / פְּזוּרֶיהָ, וְעַד
יִרְאֶה בְעֵינָהּ:



וְיָחִישׁ לָהּ מְבוֹא יֶשַׁע, וְיָשִׁיב / עֲלֵי קָדְקֹד אֲמָתָה אֶת
זְדוֹנָהּ:



וְיִשְׁמַע צַעֲקָהּ חֵלֶף הֲגוֹתָהּ / בְּדָת צִדְקוֹ, וְיוֹשִׁיעָה
לְמַעְנָהּ:



וְיַעְלֶה לְעִיר קָדְשׁוֹ, וְכִימֵי / נְעוּרֶיהָ יְחַדֵּשׁ אֶת
שְׂשׂוֹנָהּ:



נִשלָם וְהַלֵּל לַיוי
