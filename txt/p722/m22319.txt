


	
	בֶּן אָדָם שְׁמַע אֶת אֲשֶׁר אֲנִי מְדַבֵּר אֵלֶיךָ
	



תוכחה






בֶּן אָדָם שְׁמַע אֶת אֲשֶׁר אֲנִי מְדַבֵּר אֵלֶיךָ:







בֶּן אָדָם אַל תְּהִי מֶרִי כְּמוֹ בָנִים סוֹרְרִים



הָבִיאָה לַמּוּסָר לִבָּךְ וְסוּרָה מִנְּתִיב זָרִים



שְׁמַע עֵצָה וַחֲכָם וְלֹא תָסוּר מִכָּל הַדְּבָרִים



אֲשֶׁר אֲנִי דוֹבֵר אֵלֶיךָ:







בֶּן אָדָם בִּינָה בְעֵין לִבָּךְ הֲלִיכוֹת יוֹצְרֶךָ



וְדַע כִּי לְאָהֳבוֹ וְלִירוֹא מִפָּנָיו יְצָרֶךָ



וְעַל זֹאת בְּרָאָךְ בְּצַלְמוֹ, וּבְכָל אֶרֶץ מְגוּרֶיךָ



אֵין עֲרוֹךְ אֵלֶיךָ:







בֶּן אָדָם גּוּר מֵחֲמַת חֵטְא סֵתֶר וּזְדוֹנָיו



וִירָא פֶּן תִּלָּכֵד בְּיַם תְּלָאוֹתָיו וּשְׁאוֹנָיו



וְדַע כִּי בוֹחֵן לִבּוֹת וּכְלָיוֹת תָּמִיד עֵינָיו



יַשְׁגִּיחוּ אֵלֶיךָ:







בֶּן אָדָם דְּרוֹךְ בִּנְתִיב צֶדֶק וּפַלֵּס מַעְגָּלָיו



וְאַל תֵט לִבָּךְ לְדָבָר רָע לְהִתְנַקֵּשׁ בְּמַעֲלָלָיו



וְאַל תִּתְרַע אֶת בַּעַל אַף וֶאֱמוֹר בְּכָל לֵב אֵלָיו



לֹא אֶעֱבוֹר אֵלֶיךָ:







בֶּן אָדָם הָסֵר מִלִּבָּךְ כַּעַס וְדִבְרֵי תוֹעָה



וְדַע כִּי הַיַּלְדוּת נוֹסַעַת וְהַשַּׁחֲרוּת נָעָה



וְגוּר מִזְּמָן אוֹמֶר כִּי אַחֲרֵיהֶם יְמֵי הָרָעָה



מָחָר אֶשְׁלַח אֵלֶיךָ:







בֶּן אָדָם וִיצִיר חֹמֶר עַד אָנָה תְּהִי נִרְדָּם



הָקֵץ רַעְיוֹנֶיךָ בְּטֶרֶם בּוֹא יוֹם אֵידָם



וּבְשׂוֹא גַלֵּי תַאֲוָתָם אִם תִּלָּחֵם נֶגְדָּם



יָשׁוּבוּ הֵמָּה אֵלֶיךָ:





בֶּן אָדָם זְכוֹר עַל מִשְׁכָּבָךְ כּוֹס הַמָּוֶת וְתַמְרוּרָיו



אֲשֶׁר אֵין שׁוֹעַ בְּפִידוֹ וּבִמְגוּרָיו



יוֹם יְהִי נָבוֹךְ לְבָבָךְ לֵאמֹר מַה יִּהְיוּ דְבָרָיו



לָאֵל הַנִּרְאֶה אֵלֶיךָ:







בֶּן אָדָם חוּל מִיּוֹם דִּין וַחֲרַד מִיּוֹם חֲרָדָה



יוֹם תְּהִי יְחִידָתָךְ בְּבִצַּת חֶטְאָךְ לְכוּדָה



יוֹם יִסַּבּוּ עָלֶיךָ מַלְאֲכֵי פַחַד וּרְעָדָה



וְנִלְחֲמוּ אֵלֶיךָ:







בֶּן אָדָם טַהֵר עַצְמָךְ מֵחֵשֶׁק הַגּוּף וְתַאֲוָתוֹ



וְאַל תֹּאמַר כִּי הוּא הִדִּיחַנִי בְחֵלֶק שְׂפָתוֹ



כִּי אַתָּה תִמְשָׁל-בּוֹ וְהָאֵל שָׂם תְּשׁוּקָתוֹ



לִשְׁמוֹעַ אֵלֶיךָ:





בֶּן אָדָם יְמוֹתֶיךָ מְעַטִּים וּכְרֶגַע נֶחְשָׁבוּ



וְאִם יַרְפּוּ נוֹגְשֶׂיךָ בַבֹּקֶר לָעֶרֶב יָשׁוּבוּ



סְבִיבָךְ חוֹנִים וְלֹא תֵדַע אִם יִרְחֲקוּ אוֹ יִקְרָבוּ



וּמֵאַיִן יָבוֹאוּ אֵלֶיךָ:





בֶּן אָדָם כַּמָּה תִישַׁן וְלֹא תֵעוֹר לְאַחֲרִיתָךְ



וְלֹא תִזְכֹּר יוֹם שׁוּב גּוּפָךְ לַעֲפַר אַדְמָתָךְ



וְיוֹם תִּתְיַצֵּב מִנֶּגֶד וְתִלְאֶה נִשְׁמָתָךְ



בַּל קְרוֹב אֵלֶיךָ:





בֶּן אָדָם לָמָּה תִהְיֶה בְאֵין לֵב כְּיוֹנָה פוֹתָה



הֲלֹא לְבַקֵּר בְּהֵיכַל אֲבוֹתֶיךָ הֵנָּה הוּבָאתָ



עַצְמֹתֶיךָ יַגִּידוּ זֹאת וְלִבָּךְ אִם לַתּוֹם תָּשִׁיתָה



כֻּלָּם יַעֲנוּ וְיֹאמְרוּ אֵלֶיךָ:





בֶּן אָדָם מַלֵּט נַפְשָׁךְ מִפַּח יָקוּשׁ וּפוּקָה



וְדַע כִּי הַמְּלָאכָה מְרֻבָּה וְהַדֶּרֶךְ רְחוֹקָה



וְלָכֵן קַח לָךְ מִכָּל מַאֲכָל מִפְעָל טוֹב וּמִדָּה חֲשׁוּקָה



וְאָסַפְתָּ אֵלֶיךָ:





בֶּן אָדָם נְטוֹשׁ נֵבֶל וּמִרְמָה וּשְׁמוֹר אֱמוּנוֹת



וְהַכְנַע גֹּבַהּ לְבָבָךְ וְהוֹאֵל מִפְּנֵי צוּר לֵעָנוֹת



וְאָז תֹּאמַר: שְׁמַע קוֹל תַּחֲנוּנַי מִשְּׁמֵי מְעוֹנוֹת



וְיָבֹאוּ אֵלֶיךָ:





בֶּן אָדָם סְעַד לִבָּךְ פַּת צֶדֶק וְדִבְרֵי תוֹרָה



כִּי בְיוֹם עֲמָדְךָ בְצָרָה תִּהְיֶה לְנַפְשָׁךְ לְעֶזְרָה



וְאִם תִּתְמַהְמַהּ מִמָּךְ רֶגַע, בְּרֹב יֶשַׁע מְהֵרָה



עַתָּה תָבוֹא אֵלֶיךָ:





בֶּן אָדָם עֲזוֹב הַבְלֵי הַזְּמָן וּזְנַח תַּאֲוָתוֹ



וְאַל יַדִּיחוּךָ בָתֵי חֶמְדָּתוֹ וּשְׂכִיָּתוֹ



וְהַאְטֵם אָזְנָךְ מִשְּׁמוֹעַ קוֹל תַּחֲנוּנָיו



אִם יִקְרָא אֵלֶיךָ:



בֶּן אָדָם פְּדֵה נַפְשָׁךְ הַיְקָרָה מִבּוֹא בְשׁוּחָה



הַמְצֵא לָהּ מַרְגֵּעָה וְתוּר לְשִׁכְנָהּ מְקוֹם מְנוּחָה



וְהַרְחֵק לְזוּת שָׂפָה וְעִקְשׁוּת פֶּה, וְיֹשֶׁר וּנְכוֹחָה



הַקְרֵב אֵלֶיךָ:



בֶּן אָדָם צִדְקָךְ לְבַדָּךְ הוּא יוֹעִילֶךָ



וְלָכֵן חַפֵּשׂ תָּמִיד בִּדְרָכֶיךָ וּמַעֲלָלֶיךָ



וֶאֱמוֹר לִפְנֵי אֵל: אִם לִבִּי תוֹעֶה מֵעָלֶיךָ



הֲשִׁיבֵנִי יוי אֵלֶיךָ:





בֶּן אָדָם קַח עִמָּךְ דְּבָרִים וְשׁוּבָה עַד יוי



וֶאֱמוֹר: לְמַעַן שִׁמְךָ יוי וְסָלַחְתָּ לַעֲוֹנַָי



וְאִם נְפוּגוֹתִי וּנְסוּגוֹתִי אָחוֹר אֲנִי וְרַעְיוֹנָי



עַתָּה שַׁבְנוּ אֵלֶיךָ:





בֶּן אָדָם רְאֵה הֵפֶךְ הַזְּמָן אֲשֶׁר בּוֹ נוּלַּדְתָּ



וְלָכֵן עֲשֵׂה תֹם וְהִנָּצֵל מֵאֲשֶׁר נִלְכַּדְתָּ



וּבְיוֹם צַר וּמָצוֹק קְרָא: הוֹשַׁע עַבְדְּךָ, אַתָּה



אֱלֹהַי, הַבּוֹטֵחַ אֵלֶיךָ:





בֶּן אָדָם שְׁפוֹךְ שִׂיחַ לִפְנֵי אֵל נוֹרָא וְאָיוֹם



לְהַצִּילָךְ כִּצְבִי מִפַּח יָקוּשׁ וְאֵין בּוֹ פִדְיוֹם



אוּלַי תְּהִי נַפְשָׁךְ לָךְ לְשָׁלָל וּתְחִי, יוֹם



יָבֹא הַפָּלִיט אֵלֶיךָ:





בֶּן אָדָם תַּבַּעְנָה שְׂפָתֶיךָ לְשׁוֹכֵן מַעַל:



תְּהִי יָדְךָ לְעָזְרֵנִי וְנַקֵּנִי מֵחֵטְא וּמִמַּעַל



וְשֵׁטֶף מַיִם רַבִּים אֵלַי לֹא יַגִּיעוּ, כִּי עַל



זֹאת יִתְפַּלֵּל כָּל חָסִיד אֵלֶיךָ:





אֵל בּוֹרֵא רוּחַ הַצִּילֵנִי מִשּׁוֹסִי



בְּגֹדֶל נָעָמְךָ חַדֵּשׁ לִי פְאֵרִי וְהָרֵם נִסִּי



אַל תְבִישֵׁנִי מִשִּׂבְרִי, כִּי בְךָ שַׂמְתִּי מַחְסִי, וּמְנוּסִי



בְּיוֹם צָרָה אֵלֶיךָ:
