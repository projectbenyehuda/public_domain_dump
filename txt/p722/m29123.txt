


	
	יוֹנוֹת אֲהָבִים מִתְעוֹפְפִים כֶּעָבִים
	



שבח לנדיב






יוֹנוֹת אֲהָבִים / מִתְעוֹפְפִים כֶּעָבִים:



רוֹעוֹת בְּגַנִּים / בֵּין זָהֳרֵי שׁוֹשַׁנִּים

הוֹגוֹת רְנָנִים / יָקְרוּ מְאֹד כִּפְנִינִים



דּוֹבֵ[ב יְשֵׁ]נִים / שִׁירָם, בְּעֵת כִּי עוֹנִים



וּבְעֵת עֲרָבִים / אוֹרָן כְּמוֹ כּוֹכָבִים:



לִבָּם יְסוֹבֵב / עַל מִשְׁכְּנוֹת הָרוֹעִים

פִּיהֶם יְנוֹבֵב / לִשְׁאוֹל לְכָל נוֹדָעִים

אַיֵּה מְשׁוֹבֵב / רוּחוֹת בְּיָגוֹן תּוֹעִים

כָּל הַנְּדִיבִים / לִדְבַר עֲצָתוֹ שָׁבִים:



עַל מִי נְתִיבוֹת / תַּרְתֶּם יְפוֹת עֵינַיִם

נָתְנוּ תְשׁוּבוֹת / לִרְמוֹז בְּעַפְעַפַּיִם

עַל עֹז לְבָבוֹת / יוֹסֵף נְקִי כַפַּיִם

לוֹ הַקְּרָבִים / הָמוּ וְלֹא נוֹשָׁבִים:



גַּן שַׁעֲשׁוּעִים / נִמְצָא לְלֵב כָּל דּוֹרֵשׁ

יָפֶה וְנָעִים / מִשְׂרָה וְכָבוֹד יוֹרֵשׁ

תָּמִים בְּדֵעִים / וּבְעַד תְבוּנוֹת חוֹרֵשׁ

מַרְפֵּא כְאֵבִים / שִׂמְחָה לְלֵב נֶעְצָבִים:



אֹהֶל יְקָרוֹ / עַל קָו אֱמוּנָה הִעְמִיד

טַל מַאֲמָרוֹ / כָּל נִפְרְדֵי לֵב הִצְמִיד

וּבְחֵץ דְּבָרוֹ / נִרְגָּן בְּזַעַם הִשְׁמִיד

כִּי עֵת שְׁבִיבִים / מִלָּיו, וְעֵת כִּרְבִיבִים:



לִבּוֹת נְבוֹכִים / רוֹעֶה בְתֹם כִּלְבָבוֹ

קַלּוּ נְסִיכִים / לִדְרוֹךְ עֲפַר מֶרְכָּבוֹ

הִבְהִיל מְלָכִים / חֹזֶק דְּבַר מַחְשָׁבוֹ

כִּי הַכְּרוּבִים / לוֹ סוֹד אֱנוֹשׁ מַקְרִיבִים:



הָאֵל יְצַוֶּה / שָׁלוֹם בְּתוֹךְ אֹהָלָיו

חַסְדּוֹ יְשַׁוֶּה / תָּמִיד מְחוֹפֵף עָלָיו

עַד כִּי יְחַוֶּה / דּוֹר דּוֹר נְעִים מַהְלָלָיו

וַעְלֵי כְּתָבִים / לָעַד יְהוּ נִכְתָּבִים:
