


	
	אֲלֵיכֶם אִישִׁים קְרוּאִים
	



מילה




אֲלֵיכֶם אִישִׁים קְרוּאִים



אֲשֶׁר מֵרֶחֶם נְשׂוּאִים



נִין אַב הֲמוֹן הַבָּאִים



בְּמָסוֹרֶת הַבְּרִית:





הֵמָּה עֲדַת מְבוֹרָכָיו



הַהוֹלְכִים בִּדְרָכָיו



וְהַשּׁוֹמְרִים מַהֲלָכָיו



וְהֵם בַּעֲלֵי בְרִית:





הַשּׁוֹפְכִים דַּם בְּנֵיהֶם



בַּשְּׁמִינִי לְמִנְיְנֵיהֶם



לַעֲשׂוֹת רְצוֹן קוֹנֵיהֶם



וְהָיְתָה לְאוֹת בְּרִית:





דְּעוּ כִי בָחַר בָּכֶם



הָאֵל, בַּעֲבוּר אֲבִיכֶם



אֲשֶׁר אֲהֵבוֹ אֱלֹהֵיכֶם



וַיִכְרָת-לוֹ בְרִית:





מִקְצֵה אֶרֶץ הֱבִיאוֹ



וּמֵאֲצִילֶיהָ קְרָאוֹ



וְאָמַר לוֹ עֵת מְצָאוֹ



זֹאת אוֹת הַבְּרִית:





שִׂימָהּ בִּבְשָׂרֶךָ



כְּחוֹתָם עַל שְׁאֵרֶךָ



תְּהִי לְזַרְעָךְ אַחֲרֶיךָ



לְדֹרֹתָם לִבְרִית:





בָּהּ תִּקָּרֵא תָמִים



אַתְּ וְזַרְעָךְ מִיְּקוּמִים



בִּהְיוֹתְכֶם נִרְשָׁמִים



בִּבְשַׂרְכֶם לִבְרִית:





עַתָּה קְהִלַּת יְשׁוּרוּן



אַהֲבַת אֲבִיכֶם תְּעִירוּן



וַעֲשׂוּ נָא וְאַל תְּפִירוּן



דִּבְרֵי הַבְּרִית:





בְּזֹאת צוּר יְגַדֵּל שִׁמְכֶם



וְיַפְרְכֶם מְאֹד וְיַעֲצִימְכֶם



וְיַעֲנֶה בְכָל עֵת שְׁלוֹמְכֶם



שׁוֹמֵר הַבְּרִית:





וְיִפְסַח עַל בָּתֵיכֶם



כְּפָסְחוֹ עַל אֲבוֹתֵיכֶם



וְיֹאמַר לְנַפְשׁוֹתֵיכֶם



הִנֵּה דַם הַבְּרִית:





יְהִי לָכֶם כַּפָּרַת



עָוֹן, וְצֵל מִתִּגְרַת



צָר, כַּכָּתוּב בְּתוֹרַת



סֵפֶר הַבְּרִית:





בְּיוֹם עָבְרִי בְּפִי בֶסֶת



וְאִמְכֶם טִיט רוֹמֶסֶת



וּבְדָם מִתְבּוֹסֶסֶת



וָאָבוֹא בִּבְרִית:





אוֹתָהּ, וְאָמַרְתִּי לָהּ



בְּדָמַיִךְ חֲיִי, כַּלָּה,



וּמֵהַיּוֹם וָמַעְלָה



לְכָה נִכְרְתָה בְרִית:





חִזְקוּ וְאַל יִרְפּוּ יְדֵי-



כֶם, כִּי אֶתְכֶם אֶפְדֶּה



וְעִם אַבְנֵי הַשָּׂדֶה



אֶכְרְתָ[ה] לָכֶם בְּרִית:





וְעַל יֶלֶד הַמּוּבָא



הַיּוֹם לְמִילָה אֲהוּבָה



אָשִׂים עֵינִי לְטוֹבָה



וְכָרוֹת אִתּוֹ בְרִית:





וְאוֹתוֹ מֵרָע אֶפְצֶה



וְיִצְלַח בְּכָל אֲשֶׁר יֵצֵא



וּבְמִילָתוֹ אֶרְצֶה



וּרְאִיתִיהָ לְזֵכֶר בְּרִית:





וְאַתְּ אָבִיו כִּי רַצְתָּ



לִרְצוֹנִי, וְחָפַצְתָּ



בּוֹ, וּבְלֵב טוֹב אַצְתָּ



לְעָבְרְךָ בִבְרִית:





בְּעֵין רָצוֹן לְךָ אֶפְנֶה



וּבֵיתְךָ לָעַד אֶבְנֶה



וּמוֹצָא פִי לֹא אֶשְׁנֶה



לְחַלֵּל בְּרִית:





וְלָךְ תִּקְוָה וְאַחֲרִית



אֶתֵּן, וְגַם לִשְׁאֵרִית



תִּהְיֶה, וְאֶכְרֹת לְךָ בְרִית



שָׁלוֹם בְּרִית:





אֲזַכֶּךָ לְתוֹרָתוֹ



וְחֻפָּתוֹ וּמִשְׂרָתוֹ



בְּכָרְתוֹ הוּא וְרַעְיָתוֹ



שְׁנֵיהֶם בְּרִית:





וְאַתֶּם יַלְדֵי אֱמוּנַי



נִכְבַּדַּי וְנֶאֱמָנַי



אֲשֶׁר לְמַלֹּאת אֶת רְצוֹנַי



בָּאוּ בִּבְרִית:





לָכֶם אָבִיא גוֹאֵל



וְתִנְהֲרוּ עַל טוּב הָאֵל



בְּכָרְתִי אֶת יִשְׂרָאֵל



וְאֶת בֵּית יְהוּדָה בְּרִית:





וְקִצֵּי יִשְׁעִי אֲקָרֵב



וְאָבִיא עַל צַר אוֹרֵב



וְעַל מַלְכוּת אֱדוֹם פִּי חֶרֶב



נוֹקֶמֶת נְקַם בְּרִית:





כִּי כָל גּוֹיִם עֲרֵלִים



מְאוּסִים וּמְגוֹאָלִים



אָהֳלֵי אֱדוֹם וְיִשְׁמְעֵאלִים



וּבְנֵי אֶרֶץ הַבְּרִית:





בָּהֶם אֶעֱשֶׂה כָלָה



וְאַכֵּם מַכַּת נַחְלָה



יַעַן בָּזוּ אָלָה



לְהָפֵר בְּרִית:





וְאַתֶּם שְׁאֵרִית עֲדָרַי



תִּירְשׁוּ קָדְשִׁי וְהָרַי



וְלָכֶם אֶשְׁלַח צִירַי



וּמַלְאָךְ הַבְּרִית:
