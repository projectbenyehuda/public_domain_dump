


	
	[יוֹדוּ לְשִׁמְךָ עֶלְיוֹן מוֹרֶה] (רְשׁוּת לְנִשְׁמַת)
	




Agnelo d’Ascoli


[חי בשנת 1436 בעיר קָמַרִינוֹ]





רְשׁוּת לְנִשְׁמַת שֶׁל יוֹם הַכִּפּוּרִים




יוֹדוּ לְשִׁמְךָ עֶלְיוֹן מוֹרֶה / דֶּרֶךְ לָשׁוּב לְךָ
כָּל־חָי



עוֹשֶׂה כֹל יוֹצֵר וּבוֹרֵא / נֶפֶשׁ רוּחַ נִשְׁמַת כָּל־חָי.





עַם זוּ בָחַר אֲדוֹן הַכֹּל / לִסְגֻלָּה
מִכָּל־עַמִּים



וְעַד־לֹא יָרָה פִּנַּת כֹּל / הִגְדִּיל יוֹם זֶה מִכָּל־יָמִים
וּבְכֵן מֵאָז תִּכֵּן הַכֹּל / אֲרִיכוּת אַפּוֹ בְרַחֲמִים





כִּי יֵצֶר לֶב־אִישׁ וְנִסְכּוֹ / יָדַע מֵאָז יוֹצֵר
כָּל־חָי



לָכֵן יֵיטִיב כָּל־אִישׁ דַּרְכּוֹ / וּבְכָל־לֵב יָשׁוּב
לְאֵל־חָי.





קְהִלּוֹת עַמְּךָ בִּתְפִלָּתָם / לְפָנֶיךָ הֵם
נֶעֱצָרִים



בַּחֲמֵשׁ עִנּוּיֵי תַאֲוָתָם / וִדּוּי פֶּשַׁע מַסְדִּירִים



אָנָּא סְלַח עַל־חַטֹּאתָם / בְּזָכְרָם מִדּוֹת הַיְקָרִים





כְּהִבְטַחְתָּם עַל־יַד חוֹזֶה / מֹשֶׁה עָנָיו מִבְחַר
כָּל־חָי



לְכַפֵּר עֲלֵיהֶם בַּיּוֹם הַזֶּה / וְלִקְרֹא לָהֶם בְּנֵי אֵל־חָי.

בְּרִיחַ מִסְגַּר גָּלוּת שִׁבְיָם / הַתֶּר־נָא לְמַעַן
שְׁמֶךָ



וְתַשְׁקִיעַ בִמְצוּלוֹת יָם / פֶּשַׁע וַעֲוֹן חַטַּאת עַמֶּךָ



הָפֵק לָהֶם כָּל־מַאֲוַיָּם / וְהַצְמֵת עָם זוֹנֶה מִמֶּךָּ





וּשְׁבוּת עַמְּךָ הָשֵׁב מַהֵר / וְאָז יוֹדֶה לְשִׁמְךָ
כָּל־חָי



וּבְאוֹר פָּנֶיךָ תַּנְהֵר / לְעָם חוֹסֶה בְשִׁמְךָ אֵל־חָי.







וּשְׁכִינָתְךָ בְּתוֹךְ צִיּוֹן / מְכוֹן שִׁבְתְּךָ
תְּשִׁיבֶנָּה



וּלְהָאִירָהּ מֵרוּם חֶבְיוֹן / עַל־נַחֲלָתְךָ תְּכוֹנֲנָהּ



וּלְתִתָּהּ עַל־כָּל־גּוֹי עֶלְיוֹן / כְּהִבְטַחְתָּ אֵל אֲמוּנָה







וְכֵן תְּקַיֵּם כָּל־נְאֻמֶּיךָ / מְהֵרָה לְעֵינֵי
כָל־חָי



וְכָל־לָשׁוֹן יִקְרָא בִשְׁמֶךָ / וְיַעֲנוּ יַחַד אֶל אֵל־חָי.





חַסְדְּךָ אֲשֶׁר הוּא מֵעוֹלָם / הַגְדֵּל עַל־עָם בְּךָ
חֲתוּמִים



מַהֵר לְגַלּוֹת קֵץ הַנֶּעְלָם / כִּי לֹא תִּזְנַח לְעוֹלָמִים



לְמִקְּטַנָּם וְעַד־גְּדוֹלָם / יֵדְעוּךָ רָם עַל־כָּל־רָמִים







וְכָל־עָם יוֹדֶה כִּי אֵין גּוֹאֵל / זוּלָתְךָ אֲדוֹן
כָּל־חָי



פּוֹדֶה וּמוֹשִׁיעַ יִשְׂרָאֵל / אֲשֶׁר חֶבֶל חֶלְקוֹ אֵל־חָי.





יִהְיוּ לְרָצוֹן תְּהִלּוֹת עַמְּךָ / וְלֹא תִבְזֶה
אֶת־עֱנוּתָם



וּשְׁלוּחֵיהֶם מְבַקְּשִׁים מִמְּךָ / רַחֲמִים עֲלֵיהֶם בִּתְחִנָּתָם



רְצוֹנָם יַשִּׂיגוּ מֵעִמְּךָ / וּפְנֵה־נָא אֶל־קוֹל שַׁוְעָתָם





בִטְחוֹנָם בְּךָ בִּזְכוּת הוֹרִים / פּוֹתֵחַ יָד
וּמַשְׂבִּיעַ לְכָל־חָי



רְצֵה בָם כְּעוֹלוֹת פָּרִים / רוֹמֲמִים לְךָ נִשְׁמַת כָּל־חָי.
