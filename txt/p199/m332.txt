


	
	מחברת "נאום אשר בן יהודה"
	



מחברת “נאום אשר בן יהודה” / שלמה אבן צקבל




הִכּוֹן, אֱנוֹשׁ, לִקְרַאת אֱלֹהֶיךָ, וְאַשְׁ־ / רֵי אִישׁ מְצָאוֹ בַר נְקִי כַפַּיִם!



הַיּוֹם לְיוֹם מָחָר חֲסָדִים קַדְּמָה – / וּלְמָחֳרָת תִּקַּח שְׂכַר יוֹמַיִם.



תִּנְעַם בְּעוֹלָם זֶה וְעוֹלָם בָּא, וּמִי / אוֹ מִי אֲשֶׁר יִתְחַתְּנָה בִשְׁתַּיִם!


  


נְאֻם אָשֵׁר בֶּן יְהוּדָה: וַיְהִי הַיּוֹם וָאֵשֵׁב בַּשַּׁעַר בְּקֶרֶב אַחַי וַחֲבֵרַי, וְכָל אִישׁ וְאִישׁ מֵשִׁיב בִּדְבָרָיו / וּמַגִּיד מִשְׁפְּטֵי אֲמָרָיו. / וָאֹמַר: אֶת חַטָּאתִי אֲנִי מַזְכִּיר הַיּוֹם! / הָיִיתִי בִימֵי נְעוּרִים / כְּאַחַד הָעֳפָרִים / דּוֹלֵג עַל הֶהָרִים / וְשׁוֹכֵן בַּיְעָרִים, / שָׁמָּה חָדְשִׁי וְשַׁבַּתִּי / וְאֵין אִישׁ אִתִּי, / כְּתֹנֶת פַּסִּים כֻּתָּנְתִּי / וְיַעֲלַת-חֵן חַבְלִי וּמַלְכֻּדְתִּי. / וְחָלְפוּ עָלַי זְמַנִּים / וְעָבְרוּ יָמִים וְשָׁנִים / עַד אֲשֶׁר בָּלְתָה מֵעָלַי כְּסוּתִי / וְנִכְסַפְתִּי לְבֵיתִי. / וָאָבוֹא הָעִיר לַילָה, וְכָל אַנְשֵׁי גְאֻלָּתִי / וְכֹל אֲשֶׁר בְּבֵיתִי / טֶרֶם יִשְׁכָּבוּ, בָּאתִי אֲלֵיהֶם / וְדָפַקְתִּי הַדֶּלֶת עֲלֵיהֶם. / הֵמָּה מִתְיָאֲשִׁים מִמֶּנִּי / וַאֲנִי אֶקְרָא אֲלֵיהֶם “הִנֵּנִי!” / וַיָּפָג לְבָבָם בְּקִרְבָּם / וְלֹא הֶאֱמִינוּ, / עַד אֲשֶׁר הֶאֱזִינוּ / וְהִכִּירוּ קוֹלִי. / וְאָז פָּתְחוּ לִי / וַיְנַשְּׁקוּ עַל פָּנַי / וְעַל עֵינַי / וַיֹּאמְרוּ לִי “נִחְיֶה אַחֲרֵי רְאוֹתֵנוּ פָנֶיךָ / וְעָמַדְנוּ לְפָנֶיךָ!” / וְרָחֲצוּ יָדַי וְרַגְלַי / וְהֵסִירוּ הַבְּגָדִים הַצּוֹאִים מֵעָלַי / וְהִלְבִּישׁוּ אוֹתִי חֲלִיפוֹת יָפוֹת / בְּאַהֲבָה רְצוּפוֹת / וְיָשַׁבְתִּי עִמָּהֶם שְׁלֹשָׁה יָמִים וּשְׁלֹשָׁה לֵילוֹת / לֶאֱכֹל וְלִשְׁתּוֹת.



וּבַיּוֹם הָרְבִיעִי יָצָאנוּ הֲלוּמֵי יַיִן / וְהַכֹּל לְנֶגְדֵּנוּ כְאַיִן, / מִתְנַהֲלִים עַל מֵי מְנוּחוֹת / וְשׁוֹאֲפִים רוּחוֹת צַחְצָחוֹת. / וְעַל הַתְּעָלוֹת – / יַעֲלוֹת / יְפֵפִיּוֹת / וְיוֹנִים הוֹמִיּוֹת, / אֲנִי מִתְבּוֹנֵן אֲלֵיהֶן / וּמַאֲזִין לְקוֹלֵיהֶן, / וְהִנֵּה כוֹכָב מַשְׁגִּיחַ מִן הַחַלּוֹנוֹת / וּמֵצִיץ מִן הַפִּנּוֹת, / קוֹרֵץ אֵלַי בְּעֵינוֹ / וּמוֹלֵל בִּימִינוֹ. / הִתְאַפַּקְתִּי לְעֵינֵיהֶם / וְהִתְהַלַּכְתִּי בֵינֵיהֶם, / עַד אֲשֶׁר עָבְרוּ אֶת הַמַּעֲבָר / וְיָצְאוּ אֶל יַד הַמִּדְבָּר. / נִמְלַטְתִּי מֵהֶם כִּצְבִי מִיָּד / וְשַׁבְתִּי אֶל הַמָּקוֹם מִיָּד. / אֲנִי חוֹלֵף וְעוֹבֵר / וְהִנֵּה קוֹל מְּדַבֵּר /“קְרַב עַד הֵנָּה וַאֲדַבְּרָה אֵלֶיךָ / וְאֶשְׁפֹּךְ אֶת רוּחִי עָלֶיךָ”. / קָרַבְתִּי / וּלְעֻמַּת הַקּוֹל נִצַּבְתּי – / וַיִּפֹּל עָלַי תַּפּוּחַ / בְּשֶׁמֶן מֹר מָשׁוּחַ, / עַל צִדּוֹ הָאֶחָד כָּתוּב לֵאמֹר:


  


אֲשֶׁר יָרוּץ לְצוֹדֵד הָעֳפָרִים / בְּמִדְבַּר יָם וְהָרֵי הַבְּתָרִים –



חֲדַל, כִּי הֵם כְּלוּאִים בַּחֲצֵרִים / וְנִמְצָאִים בְּחַדְרֵי הַחֲדָרִים!


  


וּבַצַּד הַשֵׁנִי כָּתוּב לֵאמֹר:


  


אֲשֶׁר שָׁכַר בְּיֵין חָמַר וְתִירוֹשׁ / וְיָנוּעַ כְּעֵץ תָּמָר וְכִבְרוֹשׁ –



לְךָ יִשְׁכַּר וְיָנוּעַ לְבָבִי / וְהָאַהְבָה תְּשַׂדֶּד־בִּי וְתַחְרֹשׁ!


  


וְכַאֲשֶׁר רְאִיתִים / וַהֲבִינוֹתִים, / גָּבְהָה נַפְשִׁי / וְנָגְהָה בְעֵינַי שִׁמְשִׁי, / וְאָמַרְתִּי: "עַל כֵּן נִקְרָא שְׁמִי אָשֵׁר, כִּי אִשְּׁרוּנִי בָנוֹת / וַיֶאֱהָבוּנִי עֲדִינוֹת. / מִי זֹאת אֲשֶׁר רָאֲתָה צִבְיִי / וְחָשְׁקָה בְיָפְיִי? / חֵי נַפְשִׁי, בְּאֹרֶךְ נְדוּדִים אַכְעִיסֶנָּה / וּבְאַהֲבָתִי אַנִּיחֶנָּה / וְאֶעֶזְבֶנָּה בְחִילָהּ / וָאֵדְעָה מַה יִּהְיֶה לָהּ!” / וְנָטִיתִי מֵעָלֶיהָ / וְלֹא פָנִיתִי אֵלֶיהָ. / וְכַאֲשֶׁר הִתְהַלַּכְתִּי / וּמִן הַמְסִלָּה הַהִיא נִמְשַׁכְתִּי, / הוֹצֵאתִי אֶת הַתַּפּוּחַ / אֲשֶׁר הָיָה אֵלַי שָׁלוּחַ, /שַׁבְתִּי לִקְרוֹתוֹ/וּבְיָפְיוֹ לַחֲזוֹתוֹ,/ דָּבְקוּ בְלִבִּי הַבָּתִּים / וּפָתוֹת אוֹתוֹ פִתִּים. / נִחַמְתִּי / עַל אֲשֶׁר לָלֶכֶת קִדַּמְתִּי / וְאָמַרְתִּי: מַה זֹּאת עָשִׂיתִי / אֲשֶׁר לֹא עָנִיתִי? / הֲרֵעוֹתִי / אֲשֶׁר לֹא הֲשִׁיבוֹתִי! / וָאֵשֵׁב הַיּוֹם וְהַלַּיְלָה, מַיִם לֹא שָׁתִיתִי / וְלֶחֶם לֹא אָכַלְתִּי / וּתְנוּמָה לֹא יָדַעְתִּי. / וְלַעֲלוֹת הַשַּׁחַר שִׁחַרְתִּי / וְלֹא הִתְאַחַרְתִּי / אֶל מְקוֹם הַקּוֹל / אֲשֶׁר מֵאֶתְמוֹל, / וְקוֹלִי נָשָׂאתִי / וְעַד הָעֲלִיָּה קָרָאתִי:


  


שָׁלוֹם לְבַת רַבּוּ מְצוּקֶיהָ, / שָׁלוֹם לְעַלְמָה מֵחֲשׁוּקֶיהָ! שָׁלוֹם לְיַעְלָה מִבְּנוֹת אַרְמוֹן – / הָאַהֲבָה דָתָהּ וְחֻקֶּיהָ!



אִם אָהֲבָה לַעְדּוֹת יְפִי שִׁירִי – / חוּטֵי חֲרוּזִים אַעֲנִיקֶהָ,



אוֹ צָֽמְאָה אֶל מֵי יְדִידוֹתַי – / הִנֵּה חֲלֵב אַהְבָה אֲנִיקֶהָ,



אוֹ נִכְסְפָה לִרְאוֹת הֲדַר פָּנַי – / אָבוֹא וְאֶשְׁתַּעְשַׁע בְּחֵיקֶהָ.


  


כִּלִּיתִי שִׁירוֹתַי / וְתַמּוּ דִבְרוֹתַי, / וְעָמַדְתִּי מִשְׁתָּאֶה / עַד אֲשֶׁר אֵדְעָה וְאֶרְאֶה – / וְהִנֵּה אֵין קוֹל וְאֵין עוֹנֶה / וְאֵין קֶשֶׁב וְאֵין פּוֹנֶה! / וְכַאֲשֶׁר גָּדַל כְּאֵבִי / וְרַב מַעֲצֵבִי, / שַׁבְתִּי בְמַחְשָׁבִי / וְגָעַרְתִּי בִלְבָבִי: / לָמָּה תִתְרָאֶה / לְכָל אֲשֶׁר אַתָּה רוֹאֶה, / וְלָמָּה תִתְגָּאֶה / וְתִגַּע עָדֶיךָ וְתִלְאֶה? / לָמָּה לֹא עָמַדְתָּ / כַּאֲשֶׁר חָמַדְתָּ? / וְאֵיךְ רַדְתָּ / וְעַתָּה נֶעְבַּדְתָּ? / הִתְקוֹשֵׁשׁ / וְהִתְאוֹשֵׁשׁ / אוֹ הִתְבָּאֵשׁ / וְהִתְיָאֵשׁ! / וְלִבִּי קָט לְגַעֲרָתִי / וְהֵשִׁיב אֶת אִמְרָתִי:





נַפְשִׁי – אֲמָתְךָ – נִבְהֲלָה / וּבְפַח מְאָרְבִים נָֽפְלָה.



מַה תִּגְעֲרָה בִי – וַאֲנִי / נִלְכָּד בְּחֵשֶׁק יַעֲלָה?



מַה יֵּשׁ בְּיָדִי לַעֲשׂוֹת? / עֵינִי לְנַפְשִׁי עוֹלֲלָה!


  


עַד אֲשֶׁר רִחַמְתִּיהוּ / וְדִבַּרְתִּי אֵלָיו וְנִחַמְתִּיהוּ / וְהוּא הוֹלֵךְ וְשׁוֹמֵם / וַיְמָאֵן לְהִתְנַחֵם. וְהִתְהַלַּכְתִּי כָּל הַיּוֹם הַהוּא עַל הַיְאוֹרִים וְעַל הָאֲגַמִּים / וְעֵינַי נוֹזְלוֹת זְרָמִים, / וְהָגִיתִי כָּל הַלַּיְלָה הַהוּא קִינִים / וָהֶגֶה וָהִי, אָח הָיִיתִי לְתַנִּים / וְרֵע לִיעֵנִים. / וְכַאֲשֶׁר זָרְחָה הַחַמָּה / וְהֵאִירוּ פְּנֵי הָאֲדָמָה, / שַׁבְתִּי אֶל הַמְסִלָּה / אֲשֶׁר עָמַדְתִּי שָׁם בַּתְּחִלָּה, / שַׂחְתִּי / וְנֶאֱנַחְתִּי / וְעָנִיתִי בְרֹן / וְקָרָאתִי בְגָרוֹן:


 


שְׁלוֹמוֹתַי שְׂאוּ נָא אֶל עֲדִינָה,/לְעַלְמָה נִשְׁקְפָה כִּדְמוּת לְבָנָה, וְאִמְרוּ נָא בְאָזְנָהּ, כִּי קְרוּאָהּ –/אֲשֶׁר קָרְאָה תְמוֹל – שָׁמַע וְעָנָה, וְהָאַהְבָה אֲשֶׁר חָרְשָׁה בְלִבָּה – / בְּלִבּוֹ נִמְצְאָה הַיּוֹם צְפוּנָה,



אֲבָל הוֹד תָּאֳרָהּ הֵנִיא לְבָבוֹ / בְּיוֹם נִגְלוֹת, וְלֹא מָצָא תְבוּנָה!


  


וְעַל מִשְׁמַרְתִּי עָמַדְתִּי / וְכָל הַיּוֹם שָׁקַדְתִּי, / צִפִּיתִי בָרָק יַעֲבָר־בִּי / לִרְאוֹת מַה יְּדַבֶּר בִּי. / וְכַאֲשֶׁר רָאִיתִי הַיּוֹם פּוֹנֶה / וְלֹא מָצָאתִי מַעֲנֶה, / צָעַקְתִּי / וְלֹא הִתְאַפַּקְתִּי / “מַה לִּי אֲיַחֵל, וְתוֹחַלְתִּי / הֲלֹא נִכְזָבָה! כִּי עֵת אֲקַוֶּה לְאוֹר – תִּוָּסֵף אֲפֵלָתִי. / נַפְשִׁי בְּפַח צְבִיָּה כַּצְּבִי אֲשֶׁר נִשְׁבָּה, / דָּרְכָה קַשְׁתָּהּ בְּאֵיתָן וְלֹא שָׁבָה / לְבַד כִּי לְקָרֵב עֵת אֲסִיפָתִי / חִצֵּי נְדוֹדִי / תִּשְׁלַח תּוֹךְ כְּבֵדִי, / וְאוּר תַּבְעִיר בְּקִרְבִּי וְלַבַּת אֵשׁ בְּלִבָּתִי, / גָּזְלָה שְׁנָתִי / וַתִּשָּׁבַע לְבַל תִּתֵּן פּוּגַת לְעֵינַי וְלֹא מַרְפֵּא לְמַחֲלָתִי”. / וְהִתְלוֹנַנְתִּי בְּאַחַד הָרְחוֹבוֹת / אֲשֶׁר הָיוּ אֵלַי קְרוֹבוֹת, / סַר וְזָעֵף / יָגֵעַ וְיָעֵף. / וְכַאֲשֶׁר כָּשַׁל כֹּחִי / וְקָצְרָה רוּחִי, / הוֹצֵאתִי אֶת הַתַּפּוּחַ בְּכַפִּי / וְהִגַּשְׁתִּיהוּ לְאַפִּי, / הֲרִיחוֹתִי בְשָׂמָיו / וְחָיִיתִי בְמַנְעַמָּיו. / וְכַאֲשֶׁר נָגֹלּוּ מִפְרְשֵׂי לַיִל וְנִטָּיוּ נַחֲלֵי יוֹמָם, נִקְרֵיתִי / אֶל הַמָּקוֹם אֲשֶׁר נִקְרֵאתִי / וְעָמַדְתִּי / בַמָּקוֹם אֲשֶׁר נִלְכַּדְתִּי / וְנָשָׂאתִי קוֹלִי בַבֶּכִי / בִּדְמָעוֹת עֲלֵי לֶחִי:


  


בְּחֵי הָאַהֲבָה, יַעְלַת אֲהָבִים, / בְּנִדְבַת יֽוֹלְדֵךְ, בַּת הַנְּדִיבִים,



בְּשׁוֹשַׁנֵּי לְחָיַיִךְ, בְּמִלֵּי / שְׂפָתַיִךְ אֲשֶׁר הֵמָּה עֲרֵבִים,



בְּתַפּוּחַ בְּרָאוֹ אֵל לְהָרְגִי, / בְּטוּרִים הֵם בְּאֶצְבַּע אֵל כְּתוּבִים – הֲלֹא תַטִּי לְדוֹדֵךְ רַחֲמִים, פֶּן / יְמִיתוּהוּ בְּפֵרוּדֵךְ אֲהָבִים,



וְיַחַרְקוּ לְחָלְיוֹ שֵׁן מְשַׂנְאִים / וְיַלְעִיגוּ עֲלֵי מוֹתוֹ מְרִיבִים?


  


וּבִקְצֵה מַעֲנַי / נִבְהֲלוּ רַעְיוֹנַי / וְנָפַלְתִּי עַל פָּנַי / וְלֹא מָצָאתִי חָזוֹן מֵאֲדֹנָי. / וְנֶאֱלַמְתִּי / וְנִרְדַּמְתִּי, / וְלֹא קַמְתִּי, / עַד אֲשֶׁר הֱקִימוּנִי / וֶהֱרִימוּנִי / יְדֵי נָשִׁים רַחֲמָנִיּוֹת / מִנְּשֵׁי הָעֲלִיּוֹת / וּפָקְחוּ עֵינַי / וְאָמְרוּ בְאָזְנַי / “בֶּן אָדָם, עֲמֹד עַל רַגְלֶיךָ / וּנְדַבְּרָה אֵלֶיךָ!” / וַתָּבוֹא בִי רוּחַ וַתַּעֲמִידֵנִי עַל רַגְלַי וְאָמַרְתִּי / “תְּדַבֵּרְנָה גְבִירוֹתַי, כִּי הִתְאַפַּקְתִּי / וּבְדִבְרֵיכֶן הִתְחַזַּקְתִּי”. / וַתֹּאמַרְנָה “שִׁפְחוֹתֶיךָ אֲנַחְנוּ / וְעָדֶיךָ שֻׁלַּחְנוּ: / שָׁלוֹם לְךָ שָׁלוֹם מֵאֵת הַצְּבִיָּה, / הַיּוֹשֶׁבֶת בָּעֲלִיָּה! / שָׁמְעָה בְכִיתֶךָ / וְרָאֲתָה דִמְעָתֶךָ, / וְהִנֵּה יָצְאָה אֵלֶיךָ בְסוֹדָהּ / וְשָׁלְחָה כְתָב יָדָהּ”. / וָאֶרְאֶה וְהִנֵּה בִידֵיהֶן אִגֶּרֶת / מְהֻדֶּרֶת / קְטַנָּה כְּכַף אִישׁ / יְקָרָה מִשֹּׁהַם וְתַרְשִׁישׁ / וְעָלֶיהָ שַׁמְנֵי מֹר, / כָּתוּב לֵאמֹר:


  


דּוֹפֵק עֲלֵי דֶלֶת וְלֹא / מֵאָז בְּפָנָיו נִנְעֲלָה –



רַק הוּא אֲשֶׁר תָּעָה וְחָ־ / לַף לוֹ בְּאֶפֶס תּוֹחֲלָה.



אִם אֶפְתְּחָה – אֵיךְ אֶשְׁכְּחָה? / אוֹ אֶנְעֲלָה – אֵיךְ אֽוּכְלָה?


  


וְכַאֲשֶׁר קְרָאתִים / וַהֲבִינוֹתִים, / נִכַּר אֵלַי מֵהֶם תּוֹכַחַת / וְאֵשׁ בְּקִרְבִּי מִתְלַקַּחַת / וְנֶהְפַּךְ עָלַי לִבִּי וְרַחֲמַי / יַחַד נִכְמְרוּ נִחוּמַי. / שָׁלַחְתִּי יָדִי / וְהֶחֱזַקְתִּי בְּבִגְדִי / וְאָמַרְתִּי לִקְרֹעַ אוֹתוֹ / וְנִבְצְרָה מִמֶּנִּי שְׂפָתוֹ, / כִּי לֹא הָיָה בִי כֹחַ. וַתֹּאמַרְנָה אֵלַי “מַה זֹּאת וְעַל מַה זֹּאת? / לָמָּה חָרַדְתָּ אֵלֵינוּ אֶת כָּל הַחֲרָדָה הַזֹּאת? / שׁוּב אֶל נַפְשֶׁךָ / וַעֲזֹב אֶת לְבוּשֶׁךָ! / הִתְעוֹרֵר / וְהִתְגַּבֵּר / וְהִתְאַפֵּק / וְהִתְחַזֵּק, / וּשְׁמֹר אֲשׁוּרֵינוּ / וְלֵךְ אַחֲרֵינוּ, / כִּי לְמַעַן הַנְחוֹתְךָ בָאנוּ”. / הֶאֱמַנְתִּי אֲלֵיהֶן / וְהָלַכְתִּי אַחֲרֵיהֶן, / וְלֹא זָכַרְתִּי דִבְרֵי קֹהֶלֶת (הוֹלֵךְ אַחֲרֶיהָ פִּתְאֹם כְּשׁוֹר אֶל טֶבַח יָבֹא), / עַד אֲשֶׁר עָבַרְתִּי הַדֶּלֶת / וּבָאתִי בָאֵשׁ וּבַמַּיִם. אָז רָפוּ יָדַי / וְצָדוּ צְעָדַי / וְנִבְהַלְתִּי / וְחָתַרְתִּי / לְהָשִׁיב אֶל הַיַּבָּשָׁה – וְלֹא יָכֹלְתִּי. / וַתֹּאחֲזֵנִי זִלְעָפוֹת / וְהִנֵּה קוֹל קוֹרֵא "חַזְּקוּ יָדַיִם רָפוֹת. / וְכַאֲשֶׁר שָׁמַעְתִּי, חָיְתָה נַפְשִׁי / וְכָבְתָה אִשִּׁי / וְעָנִיתִי “זְכָר־דָּבָר לְעַבְדֶּךָ עַל אֲשֶׁר יִחַלְתָּנִי, / זֹאת נֶחָמָתִי בְעָנְיִי כִּי אִמְרָתְךָ חִיָּתְנִי”. / וְהָאֲמָהוֹת הָהֵן בְּהַבְלֵי פְתַיּוּת מְשָׁכוּנִי / וְעַל בָּמוֹתַי יַעֲמִידוּנִי / וְהוֹלִיכוּנִי בְחַצְרֵיהֶן / וֶהֱבִיאוּנִי בְחַדְרֵיהֶן / וְאֶל הָעֲלִיָּה הֶעֱלוּנִי / וְאֶל פַּח יוֹקְשִׁים הִפִּלוּנִי. / אֲנִי הוֹלֵךְ בְּצַד הָעֲלִיָּה, / טֶרֶם אָבוֹא אֵלֶיהָ / וְאֶחְסֶה בְצִלֶּיהָ, / שָׁמַעְתִּי אַחַת מְשַׁנֶּנֶת / וּבְכִנּוֹרָהּ מְנַגֶּנֶת / “יָבֹא דוֹדִי לְגַנּוֹ / וְיֹאכַל פְּרִי מִגְדָּנוֹ” / וְעָנְתָה אַחֶרֶת / וְעַל נִבְלָהּ מְשׁוֹרֶרֶת / "מַרְבַדִּים רָבַדְתִּי עַרְשִׂי חֲטֻבוֹת אֵטוּן מִצְרַיִם. וְכַאֲשֶׁר בָּאתִי אֲלֵיהֶן / וְנִגְלֵיתִי עֲלֵיהֶן, / וַתֶּחֱרַדְנָה אֵלַי / וַתִּשְׁתַּחֲוֶינָה לְפָנַי / וַתֹּאמַרְנָה “בּוֹא, בְּרוּךְ יְיָ!” / וְיָשַׁבְתִּי עַל מִפְרְשֵׂי חוּר תְּכֵלֶת וְאַרְגָּמָן, / מַעֲשֵׂה יְדֵי אָמָּן. / הֵנָּה מְשַׂחֲקוֹת / וְהָאָרֶץ לְפָנַי מְנַשְּׁקוֹת / וְכָל אַחַת וְאַחַת מְבַשֶּׂרֶת / “עַתָּה תָבוֹא הַגְּבֶרֶת”, / עַד אֲשֶׁר רָאִיתִי פְנֵי כֻלָּן זוֹעֲפוֹת / כִּפְנֵי כִבְשָׁן שְׂרוּפוֹת. / שַׂחְתִּי אֲלֵיהֶן “מַה לָּכֶן / וְלָמָּה נִשְׁתַּנּוּ פְנֵיכֶן? / הֲיֵשׁ בִּלְבַבְכֶן מַטְמוֹן? / וַתֹּאמַרְנָה “הִנֵּה אֲדוֹן הָאַרְמוֹן!” / וָאֶשָּׂא אֶת עֵינַי / וָאֵרֶא בָּא לְפָנַי / אִיש כֻּלּוֹ מְלֵא זַיִן, / מִשְׁתַּכֵּר וְלֹא מִיַּיִן. / מִצְנֶפֶת בַּד עַל רֹאשׁוֹ / וּבְאֵשׁ קִנְאָה בָּעֲרָה נַפְשׁוֹ, / פָּנָיו לוֹהֲטִים, / כְּאֶחָד מִבְּנֵי פְלִשְׁתִּים / מִתְעַטֵּף בִּרְדִידוֹ, / וְחַרְבּוֹ שְׁלוּפָה בְיָדוֹ. / קַמְתִּי וְהִשְׁתַּחֲוֵיתִי / וְקָרָאתִי חָטָאתִי וְיָשָׁר הֶעֱוֵיתִי” / וְאַחַר כֵּן קִדַּמְתִּיהוּ “שָׁלוֹם, שָׁלוֹם!” / וַיֹּאמַר “חֲדַל לְךָ וּלְשָׁלוֹם / וֶאֱמֹר: מִי הֱבִיאֲךָ הֲלֹם? / כָּכָה יִגָּלוּ סְתָרִים / יִבָּעוּ צְפוּנֵי חֲדָרִים? / אֱמֹר, מִי זֶה הֶעֱנִיק אוֹתְךָ גַאֲוָה / וְכִסָּה אֶת עֵינֶיךָ בְּתַאֲוָה?” / נִפְעַמְתִּי / וְדוֹמַמְתִּי / וְלֹא מָצָאתִי כֹחַ לְדַבֵּר / וּמִלִּים לְחַבֵּר. / וַיּוֹסֶף וַיֹּאמַר “הֲבָגַדְתָּ – / וְעַתָּה חָרַדְתָּ? / וְכִחַשְׁתָּ – / וְעַתָּה הִתְבּוֹשַׁשְׁתָּ? / וְהִתְקוֹמַמְתָּ – / וְעַתָּה נֶאֱלַמְתָּ? / אֱמֹר, עַל מִי בָטַחְתָּ / בְּעֵת הִתְלַקַּחְתָּ? / וּבְמִי נֶעֱזַרְתָּ / וְלֹא יָגֹרְתָּ? / וּמִי הֲסִיתְךָ בַהֲבָלִים / וְהִפִּלְךָ בַחֲבָלִים? / בְּרִית נְדִיבִים הֵפַרְתָּ / וְסֵתֶר מְלָכִים קָרַעְתָּ – / חֵי נַפְשִׁי, כִּי בֶן־מָוֶת אַתָּה, / וְהַיּוֹם הַזֶּה בָּא עִתֶּךָ / וְאֶרְאֶה בְמַפַּלְתֶּךָ”. / וְהֶחֱזִיק בִּי וַיְעִירֵנִי / וַיִּתְפֹּשׂ חָצְנִי / וַיֹּאמֶר “פְּשֹׁט בְּגָדֶיךָ / וְתֹאכַל מִפְּרִי יָדֶיךָ”. / וַיְהִי לִבִּי כַדּוֹנַג / וְנַפְשִׁי עָלַי תִתְמוֹגָג, / חֶבְלֵי מָוֶת אֲפָפוּנִי / וּמְצָרֵי שְׁאוֹל מְצָאוּנִי. / וְכַאֲשֶׁר רָאָה כִּי יָרֵאתִי וְזָחַלְתִּי, / פֶּן תִּדְבָּקַנִי הָרָעָה וָמַתִּי, / הִתְגּוֹלֲלוּ בְדִמְעָה עֵינָיו / וַיָּסֶר אֶת הַמַּסְוֶה מֵעַל פָּנָיו / וַיִקְרָא בְאָזְנַי “בֶּן־אָדָם, פְּקַח עֵינֶיךָ / וּרְאֵה אֶת הַמַּרְאֶה אֲשֶׁר לְפָנֶיךָ!” / הִתְבּוֹנַנְתִּי – וְהִנֵּה אוֹר בָּהִיר / וְלֶחִי כְּזֹהַר הָרָקִיעַ יַזְהִיר. / הִתְעוֹרֲרָה לְשׁוֹנִי / וְאָמַרְתִּי “מַה זֶּה אֲדוֹנִי?” / וַיֹּאמַר "בִּי אֲדוֹנִי, אֲנִי מִיַּעֲלוֹת הַחֲדָרִים / וְאַיְלוֹת הַחֲצֵרִים, / לֹא שָׁכַנְתִּי בִיעָרִים / וְלֹא יָדַעְתִּי הָרֵי בְתָרִים. / אֲנִי מִבְּנוֹת אַרְמוֹן, / בְּחֵיק הַיְקָר אָמוֹן.



אֲנִי עֹפֶר בְּעַפְעַפָּיו יְצוֹדֵד / אֱנוֹשׁ חַיִל וְיִפֹּל בַּחֲבָלָיו – וְחַנֹּתִי אֲשֶׁר אָחֹן וְאֶשְׁמֹר, / לְבִלְתִּי אֶמְנְעָה טוֹב מִבְּעָלָיו".



וְכַאֲשֶׁר הִבְהִילַנִי מְאוֹרָהּ / כִּסַּתְהוּ בִשְׂעָרָהּ / וְאָמַרְתִּי “בְּרוּכָה אַתְּ לַיָי, בִּתִּי, הֵיטַבְתְּ חַסְדֵּךְ הָאַחֲרוֹן / מִן הָרִאשוֹן! / הַאַתְּ אֲשֶׁר מִן הָעֲלִיָּה עֲנִיתִנִי / וְאַחֲרֵי כֵן עִנִּיתִנִי / וְתַחַת הַתַּפּוּחַ עוֹרַרְתִּנִי / וְאַחֲרֵי כֵן הֲרַגְתִּנִי?” / וַתַּעֲנֵנִי “לֹא, אֲדוֹנִי, הִיא גְבֶרֶת הַגְּבִירוֹת / וּמַחְפֶּרֶת הַמְאוֹרוֹת, / הִיא נֹעַם הַשְּׂפָתַיִם / וְתַאֲוַת הָעֵינַיִם. / אַמְשִׁיל – וְחָלִילָה לְהַמְשִׁילָהּ! - לְאוֹר חַמָּה בַשָּׁמַיִם, / עֲדֵי תִהְיֶה לְשִׁבְעָתַיִם! / אֲנִי שִׁפְחָה מִשִּׁפְחוֹתֶיהָ / וְאַחַת מֵאַמְהוֹתֶיהָ, / שָׁלְחָה אוֹתִי לְשַׁעַשְׁעֶךָ / וּלְדַבֵּר עַל לִבְּךָ וְנַפְשֶׁךָ׃ / כִּי אָמְרָה אֲקַדְּמָה פָנָיו בַּמִּנְחָה הַהוֹלֶכֶת לְפָנַי / וְאַחֲרֵי כֵן אֶרְאֶה פָנָיו – אוּלַי יִשָּׂא פָנַי. / לִרְקָמוֹת תּוּבַל לַמֶּלֶךְ – / בְּתוּלוֹת אַחֲרֶיהָ רֵעוֹתֶיהָ מוּבָאוֹת לָךְ”. / וְטֶרֶם כִּלְּתָה דְבָרֶיהָ, שָׁמַעְתִּי קוֹל הָמוֹן / וְצִלְצַל פַּעֲמוֹן / וַהֲרִיחֹתִי קָנֶה וְקִנָּמוֹן / וָאֵרֶא וְהִנֵּה שִׁפְעַת עֲדִינוֹת / מִבְּחוּרֵי בָנוֹת / וּבְתוֹכָן



עַלְמָה מִשִּׁכְמָהּ וָמַעְלָה גְּבוֹהָה עַל כֻּלָּן / וְנָס מִפָּנֶיהָ צִלָּן. / כְּלִילֵי מַלְכוּת כְּלִילֶיהָ / וּפַעֲמוֹנִים בְּשׁוּלֶיהָ, / בִּצְעִיף גַּאֲוָה מִתְעַטֶּפֶת / וְכִסְעִיף הֲדַס מִתְנוֹפֶפֶת, / בַּלָּאט מִתְנַהֶלֶת / וְעַל רֵעוֹתֶיהָ מִתְגַּדֶּלֶת, / וְהֵנָּה יַעֲנוּ אוֹתָהּ בְּקוֹלוֹת, / בְּתֻפִּים וּבִמְחוֹלוֹת / “אִמְרִי נָא הֲמֵרֹךְ חֻמְשֵׁךְ / יִקְרֵךְ / אוֹ שָׁכַרְתְּ מִתִּירוֹשֵׁךְ”. / וְלֹא רָאוּ עַפְעַפַּי עֵינֶיהָ, / כִּי כִסְּתָה פָנֶיהָ. וְכַאֲשֶׁר הוֹשִׁיבוּהָ עַל מְכוֹנֶיהָ / יָצְאוּ מִלְּפָנֶיהָ, / וְנִשְׁאַרְתִּי אֲנִי לְבַדִּי / וְהָעַלְמָה יוֹשֶבֶת לְצִדִּי. / וְאָמַרְתִּי בְלִבִּי: הַחְלֵק לְשׁוֹנְךָ מַחֲמָאוֹתֶיךָ / תִּקְרַב וְעָשִׂיתָ כְּחָכְמָתֶךָ! / וְסַבֹּתִי פָנַי אֵלֶיהָ / וְקָרָאתִי עָלֶיהָ / “לִבַּבְתִּנִי, אֲחוֹתִי כַלָּה, לִבַּבְתִּנִי בְּאַחַת מֵעֵינַיִךְ, / בְּאַחַד עֲנָק מִצַּוְּרוֹנַיִךְ. / גַּלִּי, אֲחוֹתִי, מְאוֹרֵךְ / וּפִרְשִׂי עָלַי שַׂעְרֵךְ, / הָסִירִי הַמַּסְוֶה מִפָּנַיִךְ / וְהַרְאִינִי אֶת מַרְאַיִךְ!” / וְכַאֲשֶׁר הֲסִירֹתִי הַצָּנִיף / וְגִלִּיתִי הַצָּעִיף, / רָאִיתִי זָקָן אָרוּךְ / וְהַמָּוֶת עָלָיו עָרוּךְ, / וּפִיו פָּתוּחַ / כְּסִיר נָפוּחַ / וְהוּא יֹאמַר "אֵיךְ רָאִיתָ הַלְּחָיַיִם / וְאֹדֶם הַשְּׂפָתַיִם! / אָז נִבְהַלְתִּי / וּמְלֹא קוֹמָתִי נָפַלְתִּי / וְחָפוּ פָנַי / וּבָרְחוּ שְׂשׂוֹנַי. / וַיִּשָּׂא קוֹלוֹ בִּצְחוֹק, / עַד אֲשֶׁר נִשְׁמַע לְמֵרָחוֹק, / וְהֶחְזִיק בִּי / וַיְדַבֵּר עַל לִבִּי / וַיֹּאמַר “הִתְיַצֵּב / וְאַל תִּתְעַצֵּב! / אֲנִי רֵעֲךָ הָעֲדֻלָּמִי – / קוּם שְׁתֵה עִמִּי. / הֲלֹא יָדַעְתָּ כִּי נַחֵשׁ יְנַחֵשׁ אִישׁ אֲשֶׁר כָּמוֹנִי?” / וָאֶפְתַּח אֶת עֵינִי – / וְהִנֵּה הוּא חֲבֵרִי / וּבָשָׂר מִבְּשָׂרִי. / וָאֹמַר לוֹ “הֲלֹא בְרָחֵל עָבַדְתִּי עִמָּךְ וְלָמָּה רִמִּיתַנִי / וְעָכֹר עֲכַרְתַּנִי?” / וַיִפְצַר בִּי לְנַחֲמֵנִי / וּבְפַת לֶחֶם סְעָדַנִי, / וַיָּבֵא אֵלַי מַטְעַמִּים / וְשָׁהִיתִי עִמּוֹ יָמִים, / וְאַחֲרֵי כֵן הִשִּׂיאַנִי לְבִתּוֹ / וְהִשְׁלִיטַנִי בְּכָל אֲשֶׁר אִתּוֹ.





שְׁעוּ, דּוֹדַי, יְפִי שִׂיחִי וְטוּבוֹ / וְשִׁמְרוּ־נָא, לְבַל תִּנָּֽקְשׁוּ בוֹ:



הֲלֹא הֵם לַעֲגֵי עוֹגְבִים וּמִלִּים / אֲשֶׁר בָּדָא יְדִידְכֶם מִלְּבָבוֹ!
