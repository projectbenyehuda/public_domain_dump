


	
	עִם שְׁקִיעָה בִּרוּשָׁלַיִם
	




זֹאת לֹא הָיְתָה

שֶׁמֶשׁ שֶׁמֵּרָמַת

יְרוּשָׁלַיִם שָׁלְחָה שַׂרְפֵי פְּרִידָה בָּיְתָה

בְּטֶרֶם עֶמְמָה וַתָּמָת.



אַף לֹא הִשְׁתַּפֵּךְ עַל הֶהָרִים יְגוֹן בַּת צִיּוֹן שֶׁנִּתְאַרְמְלָה

מִמְּשִׁיחָהּ־דּוֹדָהּ־מַלְכָּהּ,

וְהִיא בְּדֶרֶךְ שְׁבִילִים בּוֹדְדִים אֶל כַּרְמָהּ לָהּ

נִסְתַּלְּקָה



לָצֶקֶת שִׂיחָהּ

אֶל כּוֹס שֶׁל כָּל עֲנָבָה.

זֹאת לֹא הָיְתָה שְׁקִיעָה. זֹאת הָיְתָה זְרִיחָה, פְּרִיחָה

עִם כָּל הַמַּרְדּוּת וְהַמְּתִיקוּת וְהָעֲנָוָה



וְהַהִשְׁתּוֹמְמוּת הַמְּשַׁכֶּרֶת, אֲשֶׁר יִשְׁאַב חֶרֶט

צַיָּר אָמָן

מִמָּזֶג תְּכֵלֶת, וֶרֶד,

חַכְלִיל וְאַרְגָּמָן



שֶׁל הַנּוֹף וְשֶׁל מִסְגַּרְתּוֹ.

זֶה הָיָה רֶנֶן גְּוָנִים, בֵּרוּץ יֵינוֹת, זִלּוּף בְּשָׂמִים

מֵאֲגַרְטֵל הַמִּתְלַפֵּד וּמִשְׁתַּלְהֵב – וּמְשַׁבַּרְתּוֹ

פִּתְאֹם יַד שֶׁל קְסָמִים



וְשׁוֹפַכְתּוֹ. וַתְּהִי שְׂרֵפָה

בְּחֵיקָהּ

שֶׁל יְרוּשָׁלַיִם, הַמְאֻבָּקָה וְהָעֲיֵפָה

וְהָרֵיקָה.



זֹאת הָיְתָה שְׂרֵפַת כָּל הַזַּעַם וְכָל הַצַּעַר

וְהַנְּקָמָה,

נִקְמַת אָב וָאֵם וָנַעַר,

אֲשֶׁר מִדָּמָם רָוְתָה הָאֲדָמָה.



זֹאת הָיְתָה שְׂרֵפַת הָעִנּוּיִים

שֶׁל בֶּן תְּרַדְיוֹן, אֲשֶׁר לִמֵּד קְהִלָּה רַבָּה

וּשְׂרָפוּהוּ בְּסֵפֶר תּוֹרָה וּבִסְפוֹגֵי צֶמֶר שְׁרוּיִים

לְעֵינֵי בִּתּוֹ הַצֹּוַחַת: בְּכָךְ אֶרְאֶךָ, אַבָּא.



זֹאת הָיְתָה שְׂרֵפַת כָּל הַשָּׁתוֹת

וְכָל הַשְּׁעָרִים,

אֲשֶׁר טָעָה בָּהּ הַמְקוֹנֵן בַּעֲלוֹתוֹ מֵעֲנָתוֹת

וְאָמַר: וַדַּאי שָׁם יִשְׂרָאֵל מַקְרִיב פָּרִים.



זֶה הָיָה אַחֲרוֹן יוֹמוֹ, פְּרִידַת־לֶכֶת

מֵעִם הַבַּיִת וּמֵעִם הַחוֹמָה,

אַחֲרֵי אֲשֶׁר שָׁלְמָה מְלֶאכֶת

חַיָּיו – זֹאת הָיְתָה תְּבוּסַת שִׁיר הַשִּׁירִים שֶׁלִּשְׁלֹמֹה.



זֹאת בָּעֲרוּ הָרִנּוֹת וְהַתְּפִלּוֹת וְהַשִּׁירוֹת

עַל כָּל הַתְּרוּמוֹת וְהַתְּנוּפוֹת,

אֲשֶׁר הֵנִיף עַם הַבְּחִירוֹת

עִם כָּל הַתְּקוּפוֹת.



זֹאת לֹא הָיְתָה שְׁקִיעָה. זֹאת יָקְדוּ כָּל הַגַּנִּים וְהַכְּרָמִים

מֵהִשְׁתַּפְּכוּת לֵב הָאֻמָּה וְהַדְלָקַת כָּל מַאֲוַיִּים.

זֹאת הָיְתָה אָז בִּגְדָל הַיָּמִים,

בְּעָמְדִי עִם הַשְּׁקִיעָה בִּירוּשָׁלַיִם.




(נוסח ראשון של השיר)
