


	
	יחידה עקבך כרוך
	



תוכחה. חתימת השם: יצחק הלוי בר זרחיה ירונדי זכרו לברכה אמן.

כל מחרוזות התוכחה מסתיימות בפסוק מקראי.






יְחִידָה עֲקֵבֵךְ כָּרוּךְ / בְּקֵץ וּזְמָן לֹא אָרוּךְ,

וּפָעֳלֵךְ לְמִשְׁפָּט עָרוּךְ – / יְהִי מַכִּירֵךְ בָּרוּךְ!





יְצוּרֵי אִישׁ נִרְדָּמִים, / סְכָלִים וְלֹא מְחֻכָּמִים,

לְבָשְׁתָּם וּכְלִמָּתָם / יֵעוֹרוּ מִשְּׁנָתָם.





צְמֵאִים לְהַשִּׂיג תַּאֲוָה / וְלַעֲלוֹת בָּמֳתֵי גַאֲוָה,

כְּאִלּוּ הוֹן יַצִּילֵם / וּמִמָּוֶת יִגְאָלֵם!





חֲתַת מָוֶת לֹא יִרְאֶה / אֱנוֹשׁ, בְּתַאֲוָה מִתְגָּרֶה,

רוֹדֵף אַחֲרֵי שְׁלוֹמִים / נֶאֱהָבִים וּנְעִימִים.





קָט בִּהְיוֹתוֹ רַעֲנָן, / בְּבֵיתוֹ שָׁלֵו וְשַׁאֲנָן,

יְצַו אֵל פֶּגַע זַעְמוֹ: / שָׂאֵהוּ אֶל אִמּוֹ!





הוֹלֵךְ עַל מִשְׁעַנְתּוֹ / בֶטַח לַעֲשׂוֹת מְלַאכְתּוֹ,

וּמָוֶת שׁוֹטֲטָה בַעֲדוֹ / וַתִּתְפְּשֵׂהוּ בְּבִגְדוֹ:





״לְכָה נָא אֶקָּחֶךָ, / הַרְפֵּה מֵרוּחֶךָ,

דְּבַר הַמֶּלֶךְ נָחוּץ – / לָמָּה תַעֲמֹד בַּחוּץ?





וּמַה זֶּה עוֹד תִּתְאַחֵר? / הֲיֶשׁ בֶּצַע כִּי תְשַׁחֵר?״

הֲיִשָּׂא כֹפֶר מְאוּמָה? / וְשֹׁחַד בַּחֵיק חֵמָה?





יִקְרָא לְאַנְשֵׁי אֱמוּנָיו, / יְצַוֶּה עַל קִנְיָנָיו,

יְמַהֲרוּ לַחְצֹב קֶבֶר / טֶרֶם כִּלָּה לְדַבֵּר.





בְּפֶתַע נַפְשׁוֹ תִפְרַח, / בְּדֶרֶךְ עוֹלָם תֶּאֱרַח,

וַתַּעֲבֹר מֵעָלָיו / וְלֹא יָסְפָה שׁוּב אֵלָיו!





רֵעָיו נִקְבְּצוּ לְסָפְדוֹ: / הוֹי אָדוֹן וְהוֹי הֹדֹה!

רַק הַלַּיְלָה בְּבֵיתוֹ / לֹא תָלִין נִבְלָתוֹ.





זוֹעֲפִים פְּנֵיהֶם וְלִבָּם, / יַעֲטֹף חָמָס בְּקִרְבָּם,

לְיוֹם מָחָר יַעֲלֹזוּ – / בֹּזוּ כֶסֶף בֹּזוּ!





רָצִים אִתּוֹ לְמִדְבָּר, / בְּאֶרֶץ אִישׁ לֹא עָבַר,

וְנוֹשְׂאוֹ עַל כְּתֵפַיִם / יַעֲשֶׂה לוֹ כְנָפַיִם.





חֶבֶל צוֹפְנוֹ יָמֹד, / אָץ לְהַעֲמִיק, לֹא יַעֲמֹד,

וְהִשְׁלִיכוֹ בְּבוֹר חָפַר, / וְכִסָּהוּ בֶּעָפָר.





יֵרֵד הַגּוּף שְׁאוֹלָה, / וְרוּחוֹ תָשׁוּב לְמַעְלָהּ,

יַחַָד שָׁבִים לִיסוֹדָם – / וְזֹאת תּוֹרַת הָאָדָם!





הִנֵּה אַחֲרֵי כָל זֹאת / בָּאוֹת חֲדָשׁוֹת רְגָזוֹת,

גְּדוֹלִים חִקְרֵי פֶלֶא / וְנִכְבָּדִים מֵאֵלֶּה –





יוֹם הָאֱלֹהִים יִדְרְשֵׁם / וְלַמִּשְׁפָּט יַגִּישֵׁם,

וְיַשְׁמִיעַ קוֹל חֲרָדָה / ״בָּאוּ יְמֵי הַפְּקֻדָּה!





רַב לָכֶם וְלִתְלָאוֹת! / הַמְעַט מִכֶּם הַלְאוֹת,



לָחַטָּאִים הָאֵלֶּה?״ / וַיֹּאמְרוּ ״לֹא נַעֲלֶה!״





וְרוּחַ מֵאֵל יַעֲנֶה: / עוֹד יֶשׁ שֶׁבֶר מִשְׁנֶה,

חָזְקוּ מְאֹד פִּשְׁעֵיכֶם / וַאֲשִׂימֵם בְּרָאשֵׁיכֶם!





נִצָּב הַגּוּף בְּאֵימָה, / דִּבְרֵי פִיהוּ מִרְמָה



״מִי זֶה יוֹסִיף סָרָה? / וּמִי כָמוֹנִי יִקְרָא:





דַּיִּי לָשֵׂאת חֲמָסִים / וְלִשְׁמֹעַ קוֹל נוֹגְשִׂים –

בְּחֹרֵי עָפָר אָהֳלִי / וּשְׁנָתִי עָרְבָה לִּי!





יֵשׁ מַעֲשֶׂה וָדַעַת / בְּרוּחִי הַמִּרְשַׁעַת,

פִּקְדוּ וַעֲנוּ עַזּוֹת / אֶת הָאֲרוּרָה הַזֹּאת!״





זִמּוֹת רוּחַ נִתְּקוּ / בְּשָׁמְעָהּ אֲמָרִים עָתְקוּ,



״מַה תְּרִיבֵנִי, חוֹרְפִי – / גַּם אֲנִי לֹא אֶחְשָׂךְ־פִּי:





כְּרִשְׁעוֹ, אֵל, שָׁפְטֵהוּ, / בְּרִיבוֹ אַל תְּנַקֵּהוּ,

בְּדַם פִּשְׁעוֹ מִתְגּוֹלֵל – / הוֹצֵא אֶת הַמְקַלֵּל!





רָעוֹת בְּכָל עֵת חָרַשׁ, / וְתַאֲוַת נְעוּרִים יָרַשׁ –

וְאֵיךְ הִנָּצֵל יֹאבֶה? / כָּזָב חָשֹׁד יַרְבֶּה!״





וְהַגּוּף יָשִׁיב גְּדוֹלוֹת, / עֶרְוַת רוּחַ לְגַלּוֹת

״אַל תַּחְמְלוּ עָלֶיהָ, / טֻמְאָתָהּ בְּשׁוּלֶיהָ,





לוּחַ כְּתָב נִפְרֶטֶת / ׳הַנֶּפֶשׁ הַחוֹטֵאת׳ –

וְאֵיכָה אֶל אֲדוֹנֶיהָ / הֵעֵזָה פָנֶיהָ?





בְּפִשְׁעָהּ פֹּה נִקְרֵאתִי, / אֲשֶׁר הֵסַתָּה אוֹתִי,

לֶכֶת בְּחֵלֶק דְּבָרִים / אַחֲרֵי הַבַּחוּרִים.





רְאוּ: מִיּוֹם נָסָעָה / הֲתִמָּצֵא בִי רָעָה?

בַּאֲשֶׁר אֶשְׁכַּב שָׁמָּה / לֹא עָשִׂיתִי מְאוּמָה!״





כְּשֶׁמֶן רַכּוּ דְבָרָיו, / אָכֵן סְבִיבָיו צָרָיו:

״אֲנִי אָשִׁיב כַּהֲתִמְךָ, / הַנִּצֶּבֶת עִמְּכָה!





הֵן בְּשׁוּבְךָ לָעֲפָרִים / וְלִמְחִלּוֹת הַקְּבָרִים

בָּאתִי אֲנִי אַחֲרֶיךָ – / לוּ יְהִי כִדְבָרֶךָ״.





״אַךְ אֶל שְׁאוֹל יָרַדְתָּ / וּמֵחַיִּים נִכְחַדְתָּ –

וְאָשׁוּבָה אֶל שָׁרְשִׁי / וְתִנָּצֵל נַפְשִׁי!״





מֵבִין כָּל נִסְתָּרוֹת / וְיוֹדֵעַ כָּל סְפוֹרוֹת,

בְּגָדְלְךָ, שׁוֹכֵן שַׁחַק, / מִדְּבַר שֶׁקֶר תִּרְחַק!





נוֹרָא בְּצֶדֶק יַעֲנֵם, / בְּמִשְׁפַּט הָאֱמֶת יְדִינֵם,

יְחַבְּרֵם כְּפַעַם בְּפַעַם, / וַיְשַׁלְּחֵם בְּיַד פִּשְׁעָם.





וְכַאֲשֶׁר חָטְאוּ יַחַד / בְּעוֹדָם לְבָשָׂר אֶחָד,

כֵּן יְצַו עֲשׂוֹתָם כָּלָה / אֶת אִשּׁוֹ הַגְּדוֹלָה!





מִי זֶה יָרֵא וְחָרֵד, / מֵחֲטָאָיו יִפָּרֵד,

לְבִלְתִּי יְנַהֲגוּ הֲתוּלִים / אִישׁ עַל הַגֶּחָלִים!





אִם עֵינִי לֹא שַׂמְתִּי / עָלַי וְעַל נִשְׁמָתִי

אוֹ יוֹם אוֹ יוֹמַיִם – / מִי לִי בַשָּׁמַיִם?
