


	
	טֶרְצִינוֹת (א – ט)
	







א


הוּרַם הַמָּסָךְ, הַכְּחֹל מָט לִנְפֹּל.

עֻקְּמוּ הַבָּתִּים בְּצַהֶבֶת.

מִבֵּין תִּפְאֹרוֹת מְצַיֶּצֶת בַּת־קוֹל.





אוֹר בְּאוֹר הִתְחַלֵּף, קָרְצוּ זְרַקּוֹרִים;

בְּחֶבְיוֹן מִשְׁכָּנָהּ שֶׁמֶשׁ נִכְלְמָה

מִתֹּהוּ בִמְאוֹר אֲנָשִׁים.





וַיָּחֶל הַקְּרָב, וְהַקֶּרֶב הוּרַק;

נִתְּזָה הַמָּרָה עַל מָתוֹק וְעַל עַז,

וּפֶרֶק אֱלֹהַּ בְּרֶטֶט פֹּרַק.





חוֹרְקִים כִּסְאוֹת, הַבָּמָה גוֹחֶכֶת,

בְּחֹם לֵבָב פְּנֵי שַׁעֲוָה נְמַסִּים,

לִגְבִיעַ שַׁמְפַּנְיָה הַשְּׁכִינָה מִתְמוֹגֶגֶת.





עַל צִירָם מִתְהַפְּכִים שְׁחָקִים.




ב


אֶחָד אֶחָד פָּרַק אֶת כְּלֵי־הַזַּיִן,

חִוְרוֹן סְפֵקוֹת זֶה כְבָר הֵחֵל מַצְהִיב,

בַּעֲלָטָה זוֹמֵם לוֹ שְׁחוֹר הַלַּיִל.





זוֹמֶמֶת אֵשׁ זָרָה הָרֶטֶט לְהַלְהִיב,

וּזְמַן נִשָּׂא עַל חִתְכֵי הַנֶּפֶשׁ;

מְעוּף רָחָב מַשִּׁיק אֶת הָאֲוִיר.





וְהַמֶּרְחָק גּוֹוֵעַ, שָׂדוֹת קוֹפְאִים.

לַמַּאֲדִים זִנּוּק חַד־דַּק בְּרֶקַע תְּכֵלֶת,

צוֹרְדִים קוֹלוֹת, הַמִּדְבָּרִים רוֹגְשִׁים.





וְכֹה נִפְסַע. הֲבַאי תֵּבֵל תּוֹסֵס כְּצוּף

בְּעֹמֶק לֵב: הַחוֹל כַּפּוֹת רַגְלֵינוּ מְלַחֵךְ,

וְכֹה נִפְשַׁע, לֹא נִתְחַסֵּד. נִיחוֹחַ אֵקָלִפְּטוּס





גַּם לִנְחִירֶיךָ יִתָּחֵב, שָׂטָן מְגַחֵךְ!




ג


תִּמְהוֹנוֹ שֶׁל מַבָּט עֵת דִּמְעָה הֶחְלִיקַתְהוּ,

תְּפִלָּה חוּמָה צוֹפָה לִנְבָטִים,

עִגּוּלוֹ שֶׁל הַצְּחוֹק – סְחַרְחֹרֶת פִּלְּחַתְהוּ.





וְנִבְּטוּ הַשָּׁמַיִם מְאֹד עֲנָוִים,

זַלְזַלֵּי עֶלְבּוֹנוֹת טָפְחוּ כֹּה קְלִילִים

וַיְטֻפְּחוּ מַרְדָּנִים – כְּאֵבִים;





הִצְטַלְּבוּ רָאשֵׁיהֶם אֶל הַכְּתִיב בִּגְוִילִים

לְקֶשֶׁב אֲנָקוֹת מֵאִחוּד חַשְׁמַלִּים

וְרַחַשׁ יָמִים בֶּעָלִים.





אֱלֹהֵי־רְגָשׁוֹת, הַכְנִיעֵם, אֱלֹהִים,

עוֹלָמְךָ בְּרוּךְ עֵצִים וְשֵׂיבָה, –

הַכְנִיעֵם, הֵם נוֹהִים





וּלְךָ נְשָׁמָה כֹּה שְׁלֵוָה!








ד


צֹמֶת־דְּרָכִים שֶׁנִּפְרֶדֶת,

חִצֵּי כִוּוּנִים מַחְטִיאִים,

בָּעַיִן רָעָב וְצָרָבֶת.





זָעִים הַגְּוָנִים שֶׁבַּמַּיִם,

נִצְחִי רַק הָאֹפֶק חֲסַר־הַגַּוָּן,

נִגְמַע הַתָּכֹל בְּקִצְפּוֹ, – כִּגְבִיעַ הַיָּיִן.





שִׁירַת הַזַּמֶּרֶת בּוֹקַעַת,

סוֹבֶבֶת קַוֵּי הַתַּקְלִיט;

מִסְּחַרְחֹרֶת מַשְׁקֶה נְשָׁמָה רוֹגַעַת.





וְיָם מַזְרִיק לְקִרְבּוֹ הַצְּלִילִים,

בְּנִימָיו יַהַפְכֵם – כִּתְמוּנַת סִינְיָאק –

לִצְבָעִים בְּנִקּוּד־נִקּוּדִים;





וְלֵב גוֹעֶה בְּקִיא יְרַקְרָק.




ה


שַׁלְוַת הַדִּמְדּוּמִים עַל אֲדָמָה,

וְעַל עֵצִים הָרוּחַ מְרַקֶּדֶת,

חָגָה בַמְּרוֹמִים צִפּוֹר נַעֲלָמָה.





נִשְׁעָן חֲמוֹר מָשְׁפַּל־אָזְנַיִם

עַל מְשׂוּכַת־גָּדֵר, בּוֹסֵס בְּאֵבוּסוֹ;

רוֹקֵעַ פֶּרֶד קְצַר־רֹאִי מוּל מֶרְחֲבֵי שָׁמָיִם.





מוּרָם עַל־נֵס עָמָל־בְּתַהֲלוּכוֹת

הֲמוֹן הַנְּמָלִים, צוֹבְאֵי פִתְחוֹ,

עָמוּס מַשָּׂא חַיִּים שֶׁל מְזוֹנוֹת.





וְעֵז בּוֹחֶנֶת עֹז קַרְנֶיהָ

בְּהִתְנַגְּשׁוּת עִם גֶּזַע־עֵץ,

וְזוֹ שַׁלְוָה צוֹפִיָּה לְפָנֶיהָ.





לְסַעֲרַת־נַפְשִׁי אֵי דִמְדּוּמֵי־הַקֵּץ?




ו


הַיּוֹם, שֶׁבָּחַל בַּמָּאוֹר, כִּי סֻחֲרַר

מֵאֶלֶף גַּלִּים וְרוּחוֹת,

הַיּוֹם, – בִּפְאוֹתָיו נִגְרָר;





רָכוּן עַל תְּחִנּוֹת שֶׁהָיוּ כֹה בְדוּחוֹת –

לֹא שָׁעָה לְיַשְׁרוּת קוֹמָתָם שֶׁל תְּרָנִים, –

הֵם נִצְּבוּ לְאֵידוֹ בִּטְפֹחַ בַּמִּפְרָשׂ אַכְזָבוֹת.





וַחֲלוֹם, שֶׁנִּצַּת עַל שׁוֹשָׁן וְעָלִים, –

צֻוָּה הִכָּחֵד בַּיַּמִּים, הִתְנַפֵּץ אֶל סַלְעֵי חֲלָקוֹת,

הַסִּיבִים, בְּרִשּׁוּם שֶׁנָּמֹג, מִתְפַּתְּלִים וְכָלִים.





עָלֶה שֶׁנִּמְעַךְ – עוֹד נָשַׁם יַרְקוּת,

לַחוּת שֶׁבּוֹ – רְגָבִים מְשַׁוַּעַת,

הָמוּ דְמָעוֹת בְּעֵינַיִם צְחִיחוֹת,





וְאֶצְבַּע מוֹחָה זָהֳרָן מְגַמַּעַת.




ז


חִיּוּךְ אָטוּם וּלְחִיצָה עַל הֶדֶק,

הַחוֹל נִסְעַר מֵאַדְמָתוֹ,

מַבָּט אָפֵל הִשְׁקִיעַ עַיִן מְדוֹבֶבֶת.





הֹוֶה מִשְׁקָע בַּהַכָּרָה, וּלְאִטּוֹ

נִדְחַס הַקּוֹל. מִלָּה כְּמוֹ נִסְחֶבֶת,

וְרוּחַ מְאַבֶּקֶת מֵיטִיבָה טַאַטְאוֹ.





הַחוֹל נִסְעַר מֵאַדְמָתוֹ. גּוּפָה נִרְבֶּצֶת.

חַיִּים בִּנְשִׁימָה נִתְעוּ לְהַאֲמִין.

חָם. וְאֵין תְּזוּזָה, עֲוִית נֶחְמֶצֶת.





שִׁבְעָה כוֹכְבֵי־הַשֶּׁבֶת. נִצְנוּץ לְהַאֲמִין.

תַּאֲוָה חֲרוּקָה הִתְנַפְּצָה לְהַרְקִיעַ,

גָּלְשׁוּ עָבִים וּמֵי־מָטָר לְהַאֲזִין.





חִיּוּךְ בְּאָזְלָתוֹ יָבֵשׁ מִלִּרְקוֹעַ.




ח


רוֹבֶצֶת עֵז בַּאֲפֵלַת הַלַּיִל,

וּגְדִי צָנוּם שׁוֹחֵר סְבִיבָה,

לוֹקֵק אֶת דְּלִי הַמַּיִם.





מִתְאָרְכִים צִלְלֵי עֵז וְגִדְיָהּ,

לְקֶצֶב הָאוֹר יוֹצְאִים בִּמְחוֹלוֹת,

כִּי פוֹזְלִים פַּנָּסִים לְמֶרְחֲבֵי־יָהּ.





בִּצְלִיל מַרְגִּיעַ נִשְׁמָעִים קוֹלוֹת,

עַל מַעֲקֶה מִרְפֶּסֶת נִצָּבִים שְׁנַיִם,

עֵץ עָנָף אוֹתָם סוֹכֵךְ בְּפֹארוֹת.





מִתְבַּלְּטִים רַק זוּגוֹת הָרַגְלַיִם,

וּמִתַּמֵּר אֶל־עָל עֲשַׁן־סִיגָרָה מִתְחַצֵּף,

מַמָּשׁ לְמוּל עֵינֵי כְבוֹדוֹ שֶׁבַּשָּׁמַיִם.





וְהוּא, כִּבְיָכוֹל, עַל עָשָׁן זֶה מִתְקַצֵּף.




ט


גָּלַשׁ אֶל הַמִּישׁוֹר נִצְנוּץ קַרְנֵי חַמָּה,

חָרִיץ פֹּה הִתְפַּתֵּל, וְאֵלֶם הַבְּרִיאָה,

וּבִטְרָשִׁים פְּרוּעָה־מֵאוֹר שְׁקִיעָה עָגְמָה.





סְלָעִים אֶבֶן גִשְׁשׁוּ, הִתְדַּפְּקוּ בִכְרִיעָה

עַל עַקְרוּתָהּ שֶׁל אֲדָמָה,

וּשְׁחוֹר גּוֹלֵל בַּלָּאט אֶת הַיְרִיעָה.





וְשָׁוְא. לַבֹּקֶר קָרַחְתָּם נוֹצֶצֶת מֵחֵימָה

וְאֹדֶם הַבּוּשָׁה אֶל הַמִּזְרָח יִגְלשׁ.

לַשָּׁוְא עַל כָּל צְבָעֶיהָ שְׁקִיעָה נִחֵמָה.





לַשָּׁוְא הַלֵּיל בִּשְׁלוּחוֹתֶיהָ יִתְלשׁ,

הֵן עֵץ בְּיֹרֶק רַעֲמָה, בְּחֶשֶׁב

לֹא הֶאֱמִין כִּי גַם עַל שַׁחֲרִית יַחֲלשׁ.





הוּא לְאוֹרוֹ שֶׁל יוֹם נִצָּב בְּקֶשֶׁב.
