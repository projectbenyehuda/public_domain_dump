י






         וּבְכֵן לַמְּדֵנִי צוּר דָּתֶךָ / בְּזָכְרִי סֵדֶר עֲבוֹדָתֶךָ

         אַחֲלַי בְּזָכְרִי צְפוּנֵי תַעֲלוּמוֹת

         אִמְרֵי עֲבוֹדַת הַיּוֹם צְפוּנוֹת וּסְתוּמוֹת

         בְּפָרָשַׁת אַחֲרֵי מוֹת.





         בֵּאֵר מֵאָז לְאַב הֳמוֹנָי:

5        בְּעֵת הָחֳדַל לְהַקְרִיב קָרְבָּנַי

         וּזְכַרְתֶּם אֶת כָּל מִצְוֹת ה'.





         גְּמוּל זֵכֶר כְּמַעֲשֶׂה אֲקִימֶנּוּ

         גַּם מִדֵּי דַבְּרִי בּוֹ לְנֶגְדִּי אֲשִׂימֶנּוּ

         הָמוּ מֵעַי לוֹ רַחֵם אֲרַחֲמֶנּוּ.





10       יִצְדָּק חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש





         דִּבֵּר הַפְרָשָׁה לְפַלְהֶדְרִין בְּהִתְחַדֵּשׁ:

         דְּרֹש שִׁבְעָה סֵדֶר הַיּוֹם וְתִתְקַדֵּשׁ

15       בְּזֹאת יָבוֹא אַהֲרֹן אֶל הַקֹּדֶשׁ.





         הַזּוֹת עָלָיו כָּל שִׁבְעָה וּלְכָל הַזָּאָה טְבִילָה

         הָכִין וְקַחַת מִשֶּׁלּוֹ לִהְיוֹת לִמְחִילָה

         פַּר בֶּן בָּקָר לְחַטָּאת וְאַיִל לְעוֹלָה





         וַעֲבוֹדַת הַיּוֹם בְּאַרְבָּעָה בִגְדֵי לָבָן יִלְבְּשָׁם

20      וְטוֹבֵל חָמֵשׁ טְבִילוֹת וּבַעֲשָׂרָה קִדּוּשִׁין מְקַדְּשָׁם

         וְרָחַץ בַּמַּיִם אֶת בְּשָׂרוֹ וּלְבֵשָׁם.





         יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





25       זִמֵּן לְהַעֲבִיר טֶנֶף מְגוֹאָל וּמְגָאֵל

         זוּג שְׂעִירִים וְאַיִל אֶ[חָד] יִקַּח כְּחֻקֵּי אֵל

         מֵאֵת עֲדַת בְּנֵי יִשְׂרָאֵל.





         חַטַּאת פָּרוֹ יַקְרִיב לְחַטָּאתוֹ

         חֻקֵּי וִדּוּי יַתְחִיל עָלָיו כְּחֻקָּתוֹ

30       בַּעֲדוֹ וּבְעַד בֵּיתוֹ.





         טָרַח וּבָא לוֹ לִצְפוֹן מִזְבֵּחַ עִם שְׁנֵי נֶאֱמָנָי

         טַעַם הַגּוֹרָל עַל שְׁנֵי שְׂעִירִים בְּפִי חֶזְיוֹנָי

         וְהֶעֱמִיד אֹתָם לִפְנֵי ה'.





35      יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





         יַכְנִיס יָדָיו לְקַלְפֵּי וּבָהּ שְׁנֵי גוֹרָלוֹת יְקָרִים

         יָד מַעֲלָה שֶׁל שֵׁם בַּתְּחִלָּה יָרִים

         וְנָתַן עַל שְׁנֵי הַשְּׂעִירִים.





40       כְּכַלּוֹתוֹ יַעֲמִיד נִשְׁחָט כְּנֶגֶד בֵּית שְׁחִיטָתוֹ

         כֵּן מִשְׁתַּלֵּחַ יַעֲמִיד כְּנֶגֶד בֵּית שְׁלִיחוּתוֹ

         לִפְנֵי ה' לְכַפֵּר עָלָיו לְשַׁלַּח אוֹתוֹ.





         לְפָנָיו בָּא שְׁנִיָּה לְהִתְוַדּוֹת מַעֲלוֹ

         לְכַפֵּר בַּעֲדוֹ וּבְעַד שִׁבְטוֹ מִלּוּלוֹ

45       וְשָׁחַט אֶת פַּר הַחַטָּאת אֲשֶׁר לוֹ





         יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





         מְקַבֵּל דָּמוֹ וְנוֹתְנוֹ לַמְּמָרֵס בְּלִי לְשַׁבֵּחַ

50       מַחְתָּה נוֹטֵל וְחָשׁ לְמַלְּאֹתָהּ בְּבֵית מַטְבֵּחַ

         גַּחֲלֵי אֵשׁ מֵעַל הַמִּזְבֵּחַ.





         נוֹטֵל כַּף שֶׁל זָהָב רֵיקָה

         נוֹתֵן לְתוֹכָהּ מְלֹא חָפְנָיו מִן הַדַּקָּה

         קְטֹרֶת סַמִּים דַקָּה.





55       סוֹבֵב וּבָא בֵין פָּרֹכוֹת הֵיכַל מְעוֹנָי

         סִדֵּר מַחְתָּה בֵין הַבַּדִּים לִפְנִים וְלִפְנָי

         וְנָתַן הַקְּטֹרֶת עַל הָאֵשׁ לִפְנֵי ה'.





         יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּו

60       אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





         עָשָׁן מִתְמַלֵּא הַבַּיִת שֶׁלֹּא לְהִסְתַּכֵּל בִּדְמוּת

         עֵין זֹהַר מַרְאֶה הַחוֹנֶה בִּתְמִימוּת

         עַל הָעֵדוּת וְלֹא יָמוּת





         פִּלֵּל בְּצֵאתוֹ קְצָרָה שֶׁלֹּא לְהַבְעִית קְדוֹשֵׁי אָדָם

65          פָּנָה וְלָקַח דַּם הַפָּר וְעַל הַכַּפֹּרֶת וְלִפְנָי בִּמְקוֹם הַדָּם

         יַזֶּה עָלָיו שֶׁבַע פְּעָמִים מִן הַדָּם.





         צֵאתוֹ לִשְׁחֹט שְׂעִיר הַחַטָּאת עַם בּוֹ יְכֻופָּר

         צָעַד לִפְנִים וְהִזָּה מִמֶּנּוּ בַמִּסְפָּר

         כַּאֲשֶׁר עָשָׂה לְדַם הַפָּר.

70       יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





         קָם וְעָמַד בְּהֵיכָל וּבִדְמֵי שְׁנֵיהֶם

         קֹדֶשׁ יְכַפֵּר בְּמִנְיָנֵי הֲזָיוֹתֵיהֶם

75       מִטֻּמְאוֹת בְּנֵי יִשְׂרָאֵל וּמִפִּשְׁעֵיהֶם.





         רָץ וְעָרוֹת דַּם הַפָּר לְדַם שָׂעִיר פְּעָלָיו

         רָץ לַמִּזְבֵּחַ הַזָּהָב מַעֲגָלָיו

         אֲשֶׁר לִפְנֵי ה' וְכִפֶּר עָלָיו





         שְׁיָרֵי הַדָּם כְּכַלּוֹתוֹ כָל אֵל

80       שׁוֹפֵךְ עַל יְסוֹד מִזְבַּח הַהַרְאֵל

         וְטִהֲרוֹ וְקִדְּשׁוֹ מִטֻּמְאוֹת בְּנֵי יִשְׂרָאֵל.





         יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





85        תַּמּוּ מֵחַטָּאת שְׁלָשְׁתָּם

         תּוֹמֵךְ מִשְׁתַּלֵּחַ וּמִתְוַדֶּה עָלָיו בְּעַד קְהַל עֲדָתָם

         וְנָשָׂא הַשָּׂעִיר עָלָיו אֶת כָּל עֲוֹנוֹתָם.





         יְמַהֵר תִּתּוֹ לְאִישׁ עִתִּי יָעִיר

         יְקָרִים מְלַוִּין אוֹתוֹ בְהוֹלִיכוֹ מִחוּץ לָעִיר

90       אֶל אֶרֶץ גְּזֵרָה וְשִׁלַּח אֶת הַשָּׂעִיר.





         צֵרוּף אֵמוּרֵי פָר וְשָׂעִיר לְהַקְטִירָם יִפְנֶה

         צִיר לִשְׂרֹף שְׁנֵיהֶם יְמַנֶּה

         מִחוּץ לַמַּחֲנֶה.





         יִצְדָּק [חִין עֶרֶךְ קְהָלִי

95       כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





         חָשׁ וְחָלַץ בִּגְדֵי בָד וּגְרָעָם

         חֲלִיצוֹת זָהָב לִלְבּשׁ תַחְתָּם יִנְעָם

         וְיָצָא וְעָשָׂה אֶת עוֹלָתוֹ וְאֶת עוֹלַת הָעָם.





100      קִדֵּם וּבָא לִלְבּשׁ בִּגְדֵי לָבָן וְצוֹעֵד

         קֹדֶשׁ קָדָשִׁים לְהוֹצִיא כַף וּמַחְתָּה יִוָּעֵד

         וּבָא אַהֲרֹן אֶל אֹהֶל מוֹעֵד.





         גּוֹמֵר וְחוֹלֵץ מַדֵּי בָד לְבֵשָׁם

         גְּנָזָם בַּקֹּדֶשׁ שֶׁלֹּא לְאַחַר כֵּן יִלְבְּשָׁם

105       בְּבוֹאוֹ אֶל הַקֹּדֶשׁ וְהִנִּיחָם שָׁם.





         יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





         יִלְבַּשׁ אַרְבַּע מַחְלְצוֹת זָהָב טְהוֹרוֹת

110      יַקְטִיר קְטֹרֶת בֵּין הָעַרְבָּיִם כְּחֻוקּוֹת סְדוּרוֹת

         בְּהֵיטִיבוֹ אֶת הַנֵּרוֹת





         אֶת בִּגְדֵי עַצְמוֹ לוֹבֵשׁ בְּהַפְשִׁיטוֹ מַדֵּי פָז וְיַהֲלוֹם

         אוֹתוֹ מְלַוִּים בְּהָלְכוֹ וַהֲלוֹם

         וְיוֹם טוֹב עָשָׂה לְאוֹהֲבָיו כְּשֶׁיָּצָא בְשָׁלוֹם.





115      תַמּוּ עֲבוֹדוֹת הַיּוֹם לַעֲרֹךְ כְּפִי דְבַר צוֹפִי

         תִּרְצֶה בְטוּבְךָ בְהַקְרִיבִי בִימֵי חָרְפִי

         אֲשֶׁר פָּצוּ שְׂפָתָי וְדִבֶּר פִּי.





         יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

120      אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]





         חַנּוּן רָצִיתִי כְבָר מַעֲשֵׂיכֶם

         חֻקֵּי עֲבוֹדָתִי אֶחֱשֹׁב בְּהַקְרִיבְכֶם

         יַעַן הַזְכַּרְכֶם.





         זִבְחֵי צֶדֶק זִבְחוּ וְהַעֲלוּ אֲמָרֶיךָ

125      זָכֹר אֶזְכֹּר מַהֵר וּבְאֶרֶץ מְגוּרֶיךָ

         יָקֵם ה' אֶת דְּבָרֶיךָ.





         קָרְבוּ לָבֹא יְמֵי יִשְׁעֲכֶם תָּשִׂישׂוּ

         קֶרֶב הָרֵי יִשְׂרָאֵל לְחַי תִּקָּבְצוּ וְתִכָּנְסוּ

         וְהַיּוֹצֵא מִפִּיכֶם תַּעֲשׂוּ.





130      יִצְדָּק [חִין עֶרֶךְ קְהָלִי

         כְּמַעֲשֶׂה סֵדֶר הִלּוּלִי

         אֱלֹהִים יַחֲשׁוֹב לִי קָדֹש]

         יִרְבּוּן צְפִירַת חֻקֵּי מוֹעֲדֵיכֶם

         גְּאוּלִים יוּבָלוּ אֵלָיו לִרְצוֹנְכֶם

135      תַּקְרִיבוּ אֶת קָרְבַּנְכֶם

         בְּעֹז קָדוֹשׁ.
