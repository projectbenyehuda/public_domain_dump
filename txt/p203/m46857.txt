


	
	מעמד למוסף לרבי יצחק בן גיאת זצ"ל
	



מגן. סימן: יצחק בן יהודה גיאת

מ


   יוֹנָה שׁוֹכֶבֶת בֵּין שְׁפַתָּיִם / יָסְפָה שֵׁנִית לָשׁוּב לְמִשְׂגַּבָּהּ

   צִלְצְלֵי תְהִלּוֹת לָקְחָה מְצִלְתַּיִם / צוּרָהּ מְגַמַּת פָּנֶיהָ וְנִיבָהּ

   חָשָׁה לְהַקְרִיב פָּרֵי שְׂפָתַיִם / חַטָּאת כִּפּוּרִים וּפִדְרָהּ וְקִרְבָּהּ

   קָרְבַּן מוּסַף מֵאֵין עֲצַלְתַּיִם / קָרְבָּהּ וּמִנְחָתוֹ מַשְׂאַת לִבָּהּ

5    בְּמִזְבַּח אִשֶּׁה לֵבָב וּמֵעַיִם / בָּם הִקְטִירָה אֵמוּרֵי חֶלְבָּהּ

   נִסְכֵי מַעֲרֶכֶת מֵי עַפְעַפַּיִם / נְתָחֶיהָ לְרָצוֹן הֶעֱלְתָה עַל גַּבָּהּ

   יָזְמָה לְחַפֵּשׂ בְּנֵר שַׂרְעַפִּים / יוֹצְרָהּ וְהִנֵּה אֱלֹהִים בְּקִרְבָּהּ

   הוֹד תַּבְנִיתוֹ בְּלִי רְאוֹת עֵינַיִם / הִכִּירָה וּמָצְאָה טָמוּן בְּחֻובָּהּ

   וּבִפְנוֹת הַיּוֹם בֵּין הָעַרְבָּיִם / וְשֶׁמֶשׁ לִיקוֹד מוּל כִּסְאֶךָ סֻבָּהּ

10    דָּבְקָה לִסְגֹּד אֶרֶץ אַפַּיִם / דֶּרֶךְ הֵיכָל שָׁמָּה מְסִבָּה

   הֵן לְפִתְחֵי אֹכֶל סוֹרְגוּ דְלָתַיִם / הָסֵר מַאֲכַל תַּאֲוָה מֵעֶרְבָהּ

   גִּשְׁתָּהּ בְּעִתּוֹת מִשְׁכַּב צָהֳרַיִם / גּוּפָהּ לְנַקּוֹת מִסֶּלֶף אָרְבָּהּ

   יֹקֶשׁ רָכַב וְנָשְׂאוּ כְתֵפַיִם / יוֹרֵד וְרִכְבֵּי סְלִיחָה שָׂם רִכְבָּהּ

   אֵל הַמַּאֲזִין מֵאֵין אָזְנַיִם / אוֹתָהּ תַּאֲזִין וְתִדְלֶה מֵעָצְבָּהּ

15    תַּעֲלֶה לְשִׁבְתְּךָ אֶל הַשָּׁמַיִם / תְּחִנָּתָהּ וְיֵחָשֵׁב מִעוּטָהּ כְּרֻובָּהּ

   צִדְקַת הוּבָא בָּאֵשׁ וּבַמַּיִם / וְאֵשׁ לֶהָבָה מֵי תֻּומּוֹ כִבָּה

   תִּזְכֹּר לְקוֹרְאִים כְּמַעֲשֵׂה אֶרֶץ מִצְרַיִם / אֲשֶׁר יְשַׁבְתֶּם בָּהּ.
