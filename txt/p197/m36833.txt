


	
	מקרובה ליום הכיפורים
	





1





הַיַּמִּים


הַיַּמִּים בְּחֶצְיָם קֵרָה עֲלִיּוֹת בְּעֶרֶץ

מִבְטַח כָּל קַצְוֵי אֶרֶץ2



הַיַּמִּים רָגְזוּ כְּהוֹבַשְׁתָּם כָּל הָאִיִּים

מֵרוֹמְמוּתֶךָ נָפְצוּ גּוֹיִם3



הַיַּמִּים טֶרֶם עָבְרְךָ לֹא זֹרְמוּ זָרוֹם

מַטֵּה עֻזְּךָ יִשְׁלַח מִמָּרוֹם4



הַיַּמִּים חִלְּקָם לָ[נוּ] וֶהֱשִׁיבָם עַל מַפְרִיכִים

מָחַץ בְּיוֹם אַפּוֹ מְלָכִים5 6



5 הַיַּמִּים זָכַרְתָּ בִשְׁאוֹלָם הַנָּס מִמִּשְׁלַחְתֶּךָ7

מִזְּבוּל קָדְשְׁךָ וְתִפְאַרְתֶּךָ8



הַיַּמִּים קְדוֹשִׁי תַּשְׁלִיךְ בָּם חֲטָאַי וַעֲוֹנַי

מִקְוֵה יִשְׂרָאֵל יְהֹוָה9








וּבְכֵן וְכָל


כָּל אֲהוּבֶיךָ יֹאמְרוּ בְקוֹל רִנּוּן

אַתָּה אֲדֹנָי אֵל רַחוּם וְחַנּוּן10





כָּל בּוֹחֲרֶיךָ יַבִּיטוּ כִּי דְבָרְךָ יֵאָמֵן

בָּרוּךְ יְהֹוָה לְעוֹלָם אָמֵן וְאָמֵן11



כָּל גְּאוּלֶיךָ יַגִּידוּ לְכָל הַבְּרִיאוֹת

גּוֹאֲלֵנוּ יְהֹוָה צְבָאוֹת12



כָּל דּוֹרְשֶׁיךָ יְדַבְּרוּ לִפְנֵי יוֹם זֶה בּוֹאוֹ

דִּרְשׁוּ יְהֹוָה בְּהִמָּצְאוֹ13



5 כָּל הֲדוּרֶיךָ יֶהְגּוּ לְאַמֵּץ קְהָלִי

הֵן אֲדֹנָי יֱהֹוָה יַעֲזָר לִי14



כָּל זוֹכְרֶיךָ יְזַמְּרוּ מִשְּׁמֵי שָׁמֶיךָ15

זָכְרֵנִי יְהֹוָה בִּרְצוֹן עַמֶּךָ16



כָּל חֲרֵדֶיךָ יְחַוּוּ נֶגְדְּךָ בְּמֶרֶץ

חַסְדְּךָ יְהֹוָה מָלְאָה הָאָרֶץ17



כָּל טְהוֹרֶיךָ יַטִּיפוּ בְּכָל עֲצָרָה

טוֹב יְהֹוָה לְמָעוֹז בְּיוֹם צָרָה18



כָּל יוֹדְעֶיךָ יְיַחֲדוּךָ בֶּאֱמֹר בְּחִירֶיךָ

יְבָרֶכְךָ יְהֹוָה וְיִשְׁמְרֶךָ19



10 כָּל כְּשֵׁרֶיךָ יַכִּירוּ כִּי נָעִים חֶלְקֵנוּ

כִּי יְהֹוָה שׁוֹפְטֵנוּ יְהֹוָה מְחוֹקְקֵנוּ20



כָּל לִמּוּדֶיךָ יְלַהֲגוּ21 בְּשָׂפָה בְרוּרָה

לְךָ יְהֹוָה הַגְּדֻלָּה וְהַגְּבוּרָה22



כָּל מְכַבְּדֶיךָ יְמַלְּלוּ עַל יָדֶיךָ הָעַזּוֹת

מֵאֵת יְהֹוָה הָיְתָה זֹּאת23



כָּל שְׂרִידֶיךָ יִשְׂאוּ קוֹלָם מִתְפַּלְּלִים

סוֹמֵךְ יְהֹוָה לְכָל הַנּוֹפְלִים24



כָּל עֲנָוֶיךָ יַעֲנוּ נֶאֱנָחִים נֶאֱנָקִים

עֵינֵי יְהֹוָה אֶל צַדִּיקִים25



15 כָּל פּוֹצֶיךָ יַפְצִיחוּ הַשּׁוֹמֵר חֲסִידָיו26

פּוֹדֶה יְהֹוָה נֶפֶשׁ עֲבָדָיו27



כָּל צְנוּעֶיךָ יִצְרְחוּ הָאוֹהֵב חוֹכָיו

צַדִּיק יְהֹוָה בְּכָל דְּרָכָיו28



כָּל קְהָלֶיךָ יִקְרְאוּ קָדוֹשׁ קָדוֹשׁ מַפְלִיא פְלָאוֹת

קָדוֹשׁ יְהֹוָה צְבָאוֹת29



כָּל רֵעֶיךָ יְרַנְּנוּ לְךָ נִשָּׂא וְגֵאֶה

רָם יְהֹוָה וְשָׁפָל יִרְאֶה30



כָּל שׁוֹחֲרֶיךָ יִשְׁאֲלוּ מִמְּךָ יוֹצֵר רוּחִי

שְׁמַע אֱלֹהִים קוֹלִי בְשִׂיחִי31



20 כָּל תְּמִימֶיךָ יְתַנּוּ אֲנָשַׁי וְנָשַׁי וְטַפִּי

תְּהִלַּת יְהֹוָה יְדַבֶּר פִּי32








וּבְכֵן אֲשֶׁר


אֲשֶׁר מַלְכוּתְךָ / בְּאַלְפֵי שִׁנְאַנִּים

וְחַיּוֹת וְאוֹפַנִּים

וּמַלְאָכִים מְרַנְּנִים / וּפַחְדְּךָ עֲלֵיהֶם



וְחָפַצְתָּ קְדֻשָּׁה / מִנְּפָשׁוֹת חַטָּאוֹת

בְּגוּפוֹת נִבְרָאוֹת

לְמַכְאוֹב וּתְלָאוֹת / וְהִיא קְדֻשָּׁתֶךָ



אֲשֶׁר מַלְכוּתְךָ / בִּיצוּרִים לְמַלְאֲכוֹתֶיךָ

לַעֲשׂוֹת מְזִמָּתֶךָ

וּלְבַצַּע אִמְרָתֶךָ / וּפַחְדְּךָ עֲלֵיהֶם



וְחָפַצְתָּ קְדֻשָּׁה / מִדּוֹאֲגִים לְמוֹתָם

וְרוֹאִים בְּרוֹאוֹתָם

מְקוֹם קִבְרוֹתָם / וְהִיא קְדֻשָּׁתֶךָ



5 אֲשֶׁר מַלְכוּתְךָ / בְּנִבְרָאִים מֵאִשִּׁים

טְהוֹרִים וּקְדוֹשִׁים

בִּדְבָרְךָ חָשִׁים / וּפַחְדְּךָ עֲלֵיהֶם



וְחָפַצְתָּ קְדֻשָּׁה / מִשְּׁקוּלֵי הֲבָלִים33

מֵאֲחוּזֵי חֲבָלִים

מְעֻנֵּי סְבָלִים / וְהִיא קְדֻשָּׁתֶךָ



אֲשֶׁר מַלְכוּתְךָ / בִּבְרָקִים רוֹצְצִים

וּבְזָקִים נוֹצְצִים

וּמַלְאָכִים מְלִיצִים / וּפַחְדְּךָ עֲלֵיהֶם



וְחָפַצְתָּ קְדֻשָּׁה / מִנְּבוּכִים בְּדֵעוֹתָם

מֵאֲמוּלִים לִבּוֹתָם

מִקְּצָרִים רוּחוֹתָם/ וְהִיא קְדֻשָּׁתֶךָ



אֲשֶׁר מַלְכוּתְךָ / בְּלוֹהֲגִים בִּכְבוֹדָךְ

וַעֲלֵיהֶם הוֹדָךְ

וְחֶזְקַת יָדָךְ / וּפַחְדְּךָ עֲלֵיהֶם



10 וְחָפַצְתָּ קְדֻשָּׁה / מִבְּרוּאֵי אֲבָקִים

בַּעֲוֹנוֹת דְּבֵקִים

וּמִתְּבוּנוֹת רֵקִים / וְהִיא קְדֻשָּׁתֶךָ



אֲשֶׁר מַלְכוּתְךָ / בְּרוֹעֲשִׁים מִפָּנֶיךָ

הָעוֹמְדִים בִּמְעוֹנֶיךָ

הָעוֹשִׂים רְצוֹנֶךָ / וּפַחְדְּךָ עֲלֵיהֶם



וְחָפַצְתָּ קְדֻשָּׁה / מִטְּבוּעֵי מְצוּלוֹת

מִמְּעוּטֵי תַּחְבּוּלוֹת

מִיּוֹשְׁבֵי אֲפֵלוֹת / וְהִיא קְדֻשָּׁתֶךָ



אֲשֶׁר מַלְכוּתְךָ / בִּגְדוּדֵי שָׁמֶיךָ

הַמַּקְדִּישִׁים לִשְׁמֶךָ

וּמְבָרֲכֶיךָ מִמְּקוֹמֶךָ / וּפַחְדְּךָ עֲלֵיהֶם





וְחָפַצְתָּ קְדֻשָּׁה / מִדְּבֵקִים בְּאַהֲבָתֶךָ

מִנּוֹדְדִים מִבֵּיתֶךָ

מִשּׁוֹאֲלִים סְלִיחָתֶךָ / וְהִיא קְדֻשָּׁתֶךָ








וּבְכֵן בָּהֶם


בַּיְיָ יִצְדְּקוּ / הָאוֹמְרִים יִרְאָתוֹ טְהוֹרָה

וּבָהֶם יִתְבָּרַךְ / הָאֵל הַגָּדוֹל וְהַגִּבּוֹר וְהַנּוֹרָא34





בַּיְיָ יִצְדְּקוּ / הַנּוֹפְלִים לוֹ עַל בֶּרֶךְ

וּבָהֶם יִתְבָּרַךְ / הַנּוֹתֵן בַּיָּם דֶּרֶךְ35





בַּיְיָ יִצְדְּקוּ / הַיּוֹשְׁבִים בְּבוּז וּמְבוּסָה

וּבָהֶם יִתְבָּרַךְ / הַיּוֹשֵׁב עַל כִּסֵּא רָם וְנִשָּׂא36





בַּיְיָ יִצְדְּקוּ / הַדּוֹחִים קָשְׁיָם וּמִרְיָם

וּבָהֶם יִתְבָּרַךְ / הַדּוֹרֵךְ עַל בָּמֳתֵי יָם37





5 בַּיְיָ יִצְדְּקוּ / הַנּוֹשְׂאִים אֵלָיו עַיִן

וּבָהֶם יִתְבָּרַךְ / הַנּוֹתֵן רוֹזְנִים לְאָיִן38





בַּיְיָ יִצְדְּקוּ / הַשּׁוֹאֲגִים הַיּוֹם בְּקוֹלָם

וּבָהֶם יִתְבָּרַךְ / הַשּׁוֹמֵר אֱמֶת לְעוֹלָם39





בַּיְיָ יִצְדְּקוּ / הַהוֹגִים לוֹ תוֹצָאוֹת לַמָּוֶת

וּבָהֶם יִתְבָּרַךְ / הַהוֹפֵךְ לַבֹּקֶר צַלְמָוֶת40





בַּיְיָ יִצְדְּקוּ /הַלּוֹהֲגִים יַם סוּף גָּזַר

וּבָהֶם יִתְבָּרַךְ / הַלּוֹבֵשׁ גֵּאוּת וְעֹז הִתְאַזָּר41





בַּיְיָ יִצְדְּקוּ / וְיִתְהַלֲלוּ כָּל זֶרַע אֲהוּבוֹ

וּבָהֶם יִתְבָּרַךְ / וָתִיק הַשָּׂם עָבִים רְכוּבוֹ42





10 בַּיְיָ יִצְדְּקוּ / הַיְרֵאִים מִשּׁוֹכֵן שְׁמֵי עֶרֶץ

וּבָהֶם יִתְבָּרַךְ / הַיּוֹשֵׁב עַל חוּג הָאָרֶץ43





בַּיְיָ יִצְדְּקוּ / הַבָּרִים הַמּוֹסִיפִים עַל תְּהִלּוֹתָיו

וּבָהֶם יִתְבָּרַךְ / הַבּוֹנֶה בַשָּׁמַיִם מַעֲלוֹתָיו44





בַּיְיָ יִצְדְּקוּ / הַנּוּגִים אֲשֶׁר לוֹ חוֹכִים

וּבָהֶם יִתְבָּרַךְ / הַנּוֹתֵן תְּשׁוּעָה לַמְּלָכִים45





בַּיְיָ יִצְדְּקוּ / הַלְּחוּצִים הַהוֹלְכִים בְּתֻמָּם

וּבָהֶם יִתְבָּרַךְ / הַלּוֹכֵד חֲכָמִים בְּעָרְמָם





בַּיְיָ יִצְדְּקוּ / הַבְּרוּכִים נְעָרַי וּזְקֵנַי

וּבָהֶם יִתְבָּרַךְ / הַבֹּונֵה יְרוּשָׁלַיִם יְהֹוָה46





15 בַּיְיָ יִצְדְּקוּ / הָרָצִים לְהַלֲּלוֹ עִמָּדִי47

וּבָהֶם יִתְבָּרַךְ / הָרוֹעֶה אוֹתִי מֵעוֹדִי





בַּיְיָ יִצְדְּקוּ / הַטְּהוֹרִים הָרְחוֹקִים מִשֶּׁקֶר

וּבָהֶם יִתְבָּרַךְ / הַטּוֹב שֶׁאֵין לְנִפְלְאוֹתָיו חֵקֶר48








וּבְכֵן וְאַתָּה


אַתָּה אֵל רוֹם וְתַחַת בְּדִבְּרָה יָצַרְתָּ



וּבִמְאוֹרוֹת אוֹתָם זִהַרְתָּ49



אַתָּה הָאֵל עוֹשֵׂה פֶלֶא 50





אַתָּה נָאוֹר בָּרָאתָ…













קרובה ליום הכיפורים: בכתב יד הגניזה מאוסף א“נ
אדלר (מס' 2678), הנמצא כעת בספריית בית המדרש לרבנים בניו־יורק. לפי
תיאור מרקוס (מזרח ומערב, ד, עמ' 5–6) כתב היד נכתב על קלף ישן נושן,
שני דפים (12.5 X 14.8 ). בכל עמוד שבע עשרה או  שמונה־עשרה שורה. כתב רש”י יפה נוטה למרובע.

על הנדפס עיין י‘ מרקוס, מזרח ומערב, ד, עמ’ 6– 9; ד',  מלואים, הימים.

הקרובה מיוסדת על הפסוק בנחמיה ט, ו: “אתה הוא ה' לבדך
אתה עשית את השמים שמי השמים וכל צבאם הארץ וכל אשר עליה הימים וכל אשר
בהם ואתה מחיה את כלם וצבא השמים לך משתחוים”. ממנה נשאר רק חלק קטן על
המלים: “הימים וכל אשר בהם ואתה” ואף באלה לקויים הפיוט הראשון והפיוט
האחרון בחסר. בראשון חסר: “אני דונש בן ל” ומן האחרון נשאר בידינו רק:
“אנ”.

הקרובה מיוחסת לדונש רק על־פי האקרוסטיכון.
האקרוסטיכון נמצא בחלק שנשאר בידינו שש פעמים ומהם שלוש פעמים בשלמות.
בחלק הימים נמצא פעמים “ברט חזק”, באשר מלכותך: אני דנש בן לברט
בגד(ו)די דנש; בה‘ יצדקו: אני דנש הלוי בן לברט (פעמים); אתה אל: “אנ”
בלבד. אין להתפלא שחתם כאן דונש חסר, כי לא השגיחו באותיות אהו“י חסרות
בימי־הבינים (עיין מתורת הלשון והשירה בימי הבינים, ירושלים תש"ד, עמ'
עב). גם באלף־בית באותו הפיוט החסיר את אות וי”ו, כי הכלל המקובל
בידיהם היה “רק הוא”ו לא ימצא בראש המלה…“ (ראב“ע, צחות, ויניציאה
ש”ו, קמט, ב; ועיין שם, קמט, ב – קנ, א). גם סימנו לאותיות השימוש הוא
בלא וי”ו בשמו “דנש הלוי אמת כטב”. עיין תת"מ, עמ’ 40–41, וזה חוזר
בתת“ד, עמ‘ 29, ובהקדמתו לתשובותיו על מנחם, עמ’ 5. בעניין זה עיין גם
ווילנסקי, מזרח ומערב, ד, עמ' 132. בסימן האחרון אנו מוצאים, כי חתם גם
הלוי, כמו שחתם ”בה' יצדקו“. חתימותיו בשירים אחרים הן בצורת דונש בלבד
בשירים הוודאיים: דבוק חתן; דוי הסר; דעה לבי; לדורש החכמות; דלה שובב;
דרור יקרא. בשיר ”לדורש החכמות“ יש גם חתימה: ”דנשש בן לברט“ גם בשירים
המסופקים אין חתימה ארוכה יותר מדונש (עוז תעצומך; ברכי יחידה) או ”דנש
חזק“ בשיר ”דחופים רדופים: או “אני דונאש בן יהודה חזק” בשיר “אלוה רצה
נא עדתך”, שאינו כנראה דונש שלנו. ד“ר שפיגל (מזרח מערב, ד, עמ' 6 הערה
16) הכיר כי בחלק ”אשר מלכותך“ רמוז במלה ”בגדודי“ גם התואר ”בגדדי“ זה
מתאים לדברי רמב”ע ב“שירת ישראל”, שמכנהו: “דונש בן לבראט הלוי
מבגדאד שרשו ובפאס גידולו” (ש"י, עמ' סד; ובמקור הערבי, לא, א).
וסעדיה אבן דנאן מכנהו: “דונש בן לברט הבבלי” (עי‘ מבוא, עמ’ ה). שימוש
במלה זו אנו מוצאים גם בשיר דעה לבי, בית 20, אולם שם בודאי שאינו משמש
רמז לשמו.

הקרובה היא ליום הכיפורים, כפי שמוכח מתוכה וכבר הוכיח  זאת מרקוס (שם, מזרח ומערב, ד, עמ' 6).

השיר ממשיך את דרכי הפייטנים בימי הבינים ואינו  שקול.

החרוז אף הוא לפי דרכי הפייטנים ולרוב חורזים שני  עיצורים ולפעמים רק עיצור אחד (כל, 4–5 ועוד).  ↩
