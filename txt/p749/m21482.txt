


	
	המשורר
	




וְיֵשׁ שֶׁיִּתְפָּאֵר בְּשִׁירִים וּמְלִיצוֹת / בָּהֶם יָרוּץ גְּדוּד,
כָּל מַעְיָנָיו בָּם בְּדַעַת וּבְמוֹעֵצוֹת, / יִתְגָּאֶה וְיִתְאַמָּר /
בְּמִשְׁלֵי חֵן בַּחֲמוּדוֹת הַמַּאֲמָר, / אִם חִכּוֹ מַמְתַּקִּים
ומִדְבָּרוֹ נָאוֶה כַּחֲלִי כֶתֶם כְּמִין חוֹמָר, / לְעֵינֵי בְנֵי
עַמּוֹ יָרִים רֹאשׁ יִתְיַצֵּב כְּחוֹמָה, / כִּי יְדַבֵּר בְּחִידוֹת
וּמִשְׁתָּעֵי בִּלְשׁוֹן חָכְמָה. לֹא יַעַרְכוּהוּ שְׂרִידִים וְרָאשֵׁי
מִשְׁפָּחוֹת, / כִּי יְמַהֵר לְדַבֵּר צַחוֹת / בִּבְנוֹתוֹ בָתִים
מְחֻטָּבִים תַּבְנִית הֵיכָל / כִּמְדֻמֶּה לוֹ יָשַׂר אֶל הַמַּלְאָךְ
וַיּוּכַל. / כִּי יִשְׁקֹל בְּמֹאזְנַיִם אִמְרֵי נֹעַם וּדְבָרִים
עֲרֵבִים – / יַחְשֹׁב שֶׁיִּפֹּל אַרְצָה מִן הַצָּבָא וּמִן
הַכּוֹכָבִים, / וְכִי יוֹצִיא מִלִּבּוֹ מִלִּים / כְּיַד הַשִּׁיר
הַטּוֹבָה עָלָיו תִּלֵּי תִלִּים – / בָּהֶם לְדַעְתּוֹ יִבְנֶה עוֹלָם
מָלֵא, / שָׁלֵם חָדָשׁ בְּרוּחַ פִּיו נַעֲשֶׂה, לִבְנוֹת וְלִנְטֹעַ
אָשְׁיוֹתָיו מִמֶּנּוּ לֹא יִפָּלֵא. / בִּלְשׁוֹנוֹ יְיַסְּדֶנּוּ,
יָצִיב דְּלָתָיו, / בְּרוּחַ חֵן יָסֹךְ לוֹ אֶת גַּגּוֹ וְאֶת
קִירוֹתָיו, / מִמַּעַל מְסֻלָּא בַפָּז מְצֻפֶּה זָהָב וּמַרְגְּנִיתָא
תוֹתֵה, / בָּזֶה יַחְשֹׁב יִמָּשֵׁךְ שְׁמוֹ הַגָּדוֹל, אֶפְשָׁר
דִּנְפִישׁ זְכוּתֵה, / יְהֵא שְׁמֵהּ רַבָּא בְּעָלְמָא דִבְרָא
כִרְעוּתֵהּ. / עַל כֵּן יִזְבַּח לְעֵטוֹ / זִבְחֵי הַבְהָבַי וּקְטֹרֶת
זָרָה יְקַטֵּר לִמְלֶאכֶת קַסְתּוֹ.



זֶה הֶחֳלִי הַמְגֻנֶּה הֵמִית לְפָנֵינוּ רַבִּים / הַנְּגוּעִים בּוֹ לֹא
יוּכְלוּ נִקָּיוֹן כִּנְשׁוּכֵי נְחָשִׁים וְעַקְרַבִּים. / גַּם הַיּוֹם
בִּזְמַנֵּנוּ זֶה הַנֶּהְפָּךְ כְּמַהְפֵּכַת זָרִים / מֻכִּים בְּזֶה
הַשִּׁגָּעוֹן וְתִמְהוֹן לֵבָב אֶלָּא שֶׁנִּתְמַעֲטוּ מִצָּרוֹת
וְנוֹתְרוּ מְפֻזָּרִים. / הַשִּׁבּוּשׁ הַזֶּה טְרָדָם מִן הָעוֹלָם, /
בְּזֶה הַקִּנְיָן הַפָּחוּת – נֶגֶד פְּנֵיהֶם נְבוֹנִים, אֵין עַל עָפָר
מָשְׁלָם. / יוֹדֵע אֲנִי מֵאֵלּוּ שֶׁכָּל יָמָיו זָרִיז וְחָרוּץ /
לְהָרִיק חֲנִיכָיו יְלִידֵי עֵטוֹ הֲמוֹנִים הֲמוֹנִים בְּעֵמֶק הֶחָרוּז.
/ מִדַּעְתּוֹ עָלָיו הוֹפִיעַ רוּחַ הַקֹּדֶשׁ וְחָזוֹן נִפְרַץ – / אִם
יָפֶה יִתְכַּוֵּן לַעֲשׂוֹת חָרָז. / בָּאֵלֶּה יְקַוֶּה לְהִתְנַחֵל
כִּסֵּא כָבוֹד וְיִשְׁתַּמֵּשׁ בְּתָגָא, / זֹאת הַחָכְמָה תָּאִיר
פָּנָיו יִנְהַר נְהוֹרֵה כִּשְׁרָגָא. / הָלוֹךְ וְטָפוֹף יֵלֵךְ,
יְדַקְדֵּק בִפְסִיעוֹתָיו בְּחוּצוֹת עִיר בִרְחוֹבוֹתֶיהָ, / עָקֵב
בְּצַד גּוּדָל, כְּיוֹנָה נֶחְפָּה בַכֶּסֶף וְאֶבְרוֹתֶיהָ, / כִּי
הִפְלִיא הַשֵּׁם חִדּוּשׁ לוֹ מְלִיצָה מְשֻׁנָּה מֵחַבְרוֹתֶיהָ. / וְאִם
קַן חֲמוּדוֹת בְּדַרְכּוֹ יִקָּרֵא – / דְּבָרִים לֹא שְׁמָעָתַן אֹזֶן
כֻּלָּם בְרוּרִים זֹרוּ בָרַחַת וּבַמִּזְרֶה, / לֹא כְאוֹתָן
שֶׁהָרְגְּלוּ בַהֲמוֹן מְלִיצִים וּנְהִיגֵי בִּבְנֵי יִשְׂרָאֵל וּדְלָא
כְטָפְסֵי דִשְׁטָרֵי, / גָּאֹה גָּאָה וְאָמַר ״מִי כָמוֹנִי מוֹרֶה?״ /
חַכְמֵי הָעִיּוּן לֹא זָכוּ בְעֵינָיו, / וְהַמְּלָכִים – רַבָּנָן
דְּאִקְּרוּ מַלְכֵי – לֹא עָמְדוּ לְפָנָיו. / וְכִי יְקַבֵּץ מֵאֵלּוּ
מְלִיצוֹת נִמְרָצוֹת, מְגִלַּת סֵפֶר בִּכְתָב רָשׁוּם – / חַכְמֵי
יִשְׂרָאֵל בְּעֵינָיו כִּקְלִפַּת הַשּׁוּם. / תָּמִיד יָדָיו מְלֵאוֹת /
מִמְּלִיצוֹת נָאוֹת / אֲגֻדּוֹת פִּתְקִין, / שֶׁבְּכָל הַלֵּילוֹת
כַּמָּה הוֹצִיא בְשֶׁמֶן עַד אֲשֶׁר אוֹתָם כִּרְצוֹנוֹ הִתְקִין! בַּחוּץ
יָרֹן, בִּרְחוֹבוֹת קִרְיָה / וְאֶל עַל יִקְרָאֵם לְפַרְסֵם מְלִיצָתוֹ
לְהַרְאוֹת הָעַמִּים וְהַשָּׂרִים אֶת יָפְיָהּ, / יַשְׁמִיעַ לְאָזְנָיו
מַה שֶּׁיּוֹצִיא בְפִיו, לֹא יִקְרָא בְיֵאוּשׁ וְלָאו אַדַּעְתָּא, /
וּבְעֵת קָרְאוֹ בְאָזְנֵי אֲנָשִׁים יְהוּדִים יֹאמַר בְּכָל שַׁעְתַָּא
וְשַׁעְתָּא / אֶל הָעוֹמְדִים לְפָנָיו ״פְּנוּ אֵלַי וְהָשַׁמּוּ, הֲלֹא
תָבִינוּ כַּמָּה מַעַלְיָא הַאי שְׁמַעְתָּא!״ / וְאִם יְשַׁבְּחוּהוּ
בְעֵינָיו, / תָּגֵל נַפְשׁוֹ וְיִצְהֲלוּ פָנָיו, / וְאָמַר ״בְּרוּכִים
אַתֶּם לַייָ, / כִּי הֲבִינוֹתֶם עֹמֶק עִנְיָנַי״. / וְאִם הֲחַרֵשׁ
יַחֲרִישׁוּ מִפֹּה וּמִפֹּה, / מִרְתַּח רְתַח יְדַבֵּר אֵלֵימוֹ
בְּאַפּוֹ / ״רְאוּ, קָרָאתִי עֲלֵיכֶם לְבַקֵּר, / בֵּין טוֹב לָרַע
תַּבְדִּילוּ וְתַפְרִישׁוּן – / וְאַתֶּם תַּחֲרִישׁוּן! / טָעִיתִי
בְמַחֲשַׁבְֶתִּי לִסְמֹךְ לְהִשָּׁעֵן עַל שִׂכְלְכֶם / מְלֶאכֶת תֻּפַּי
וּנְקָבַי לֹא תָבִינוּ – וְחָכְמָה מַה לָּכֶם?״



כֵּן מִנְהַג הָאִישׁ הַזֶּה וְסִיעָתוֹ: אֵין לָהֶם אֻמָּנוּת אֶלָּא
הוּא, / וְעִם כָּל זֶה גַּאֲוָתָם וּגְאוֹנָם רָחֲבָה וְנָסְבָה,
אַבִּירֵי לֵב נִנְהוֹ. / עֲלֵיהֶם אָמַרְתִּי: הוֹי הַמַּעֲמִיקִים וְלֹא
בְּעָמְקָהּ שֶׁל הֲלָכָה! / כָּלוּ יְמֵיהֶם בְּמַה שֶּׁאֵינוֹ לֹא
חָכְמָה וְלֹא מְלָאכָה / וּמוֹתָרוֹ לְאֹרֶךְ הַיָּמִים הֶבֶל הַכָּלָה
וְהַחֶרֶץ – / שֹׁד לָהֶם כִּי צָלֲלוּ בְמַיִם אַָדִּירִים וְהֶעֱלוּ
בִידֵיהֶם חֶרֶס!
