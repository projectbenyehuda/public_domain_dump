


	
	נורא עלילה
	




נוֹרָא עֲלִילָה / הַמְצֵא לָנוּ מְחִילָה / בִּשְׁעַת נְעִילָה

אֵלֶיךָ צוֹעֲקִים / יְשִׁישִׁים וְיוֹנְקִים / מִבֹּקֶר וְעַד לַיְלָה / בִּשְׁעַת נְעִילָה

נֶגְדְּךָ נוֹרָא וְאָיוֹם / עָמְדוּ כָּל הַיּוֹם / מִתְחַנְּנִים בִּתְפִלָּה / בִּשְׁעַת נְעִילָה

יִשְׁעֲךָ סוֹבְרִים / עִם חֲתִימַת סְפָרִים / שׁוֹאֲגִים בְּמִלָּה / בִּשְׁעַת נְעִילָה

יֵחָתְמוּ לְחַיִּים / וְהוֹצִיאֵם נְקִיִּים מִכָּל חֵטְא וְעַוְלָה / בִּשְׁעַת נְעִילָה

וּרְצֵה עִנּוּיָם / וְהָפֵק מַאֲוַיָּם / לַנִּצָּבִים בְּחִילָה / בִּשְׁעַת נְעִילָה

שַׂגְּבֵם מִפֶּשַׁע / וְנַקֵּם מֵרֶשַׁע / כִּי הֵם נַחֲלָה / בִּשְׁעַת נְעִילָה

פְּנֵה אֲלֵיהֶם בְּרַחֲמִים / צוּר שׁוֹכֵן מְרוֹמִים / לַעֲדֵי עַד סֶלָה / בִּשְׁעַת נְעִילָה

בָּרֵךְ שְׁאֵרִיתָם / וְקַבֵּץ גָּלוּתָם / בְּרִנָּה וְצָהֳלָה / בִּשְׁעַת נְעִילָה

רוֹמֵם אֶת רֹאשָׁם / וְשׁוֹבְבֵם לְבֵית מִקְדָּשָׁם / בְּשִׂמְחָה וְגִילָה / בִּשְׁעַת נְעִילָה

יָחִיד שׁוֹכֵן שְׁמֵי עֶרֶץ / בּוֹרֵא קְצוֹת הָאָרֶץ / אֲשֶׁר עַל כֹּל נָעֲלָה / בִּשְׁעַת נְעִילָה

צְפֵה עָנְיָם וְדִלְדּוּלָם / וּשְׁבֹר מוֹטוֹת עֻלָּם / וְשִׂימֵם לִתְהִלָּה / בִּשְׁעַת נְעִילָה

חֹן בְּחַסְדְּךָ עַם זֶה / וּתְפִלָּתָם אַל תִּבְזֶה / אֵל נֶאְדָּר בִּגְדוּלָה / בִּשְׁעַת נְעִילָה

קַבֵּל שַׁוְעָתָם / עֵת חֲתִימָתָם / צוּר שׁוֹכֵן מַעְלָה / בִּשְׁעַת נְעִילָה

חַזֵּק יָדָם וְאַמְּצֵם / וּמִכָּל צָרָה חַלְּצֵם / כִּי הֵם לְךָ סְגֻלָּה / בִּשְׁעַת נְעִילָה

זְמַן קָרֵב נָא לְפִדְיוֹנָם / וְשַׂמְּחֵם מִיְּגוֹנָם / וְחִישׁ יוֹם הַגְּאֻלָּה בִּשְׁעַת נְעִילָה.


מקור: הארץ עמ׳ 8 ערב יום כיפור 15.9.2021.
