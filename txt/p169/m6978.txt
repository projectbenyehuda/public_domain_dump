


	
	צִלְצְלֵי קָיִץ
	




לֹא שֶׁפִי כְגוֹנֵב הַגְּבוּלִים –



הוּא נִכְנָס בְּפֻמְבֵּי וּתְרוּעָה,



וַיִּשְׁלַח מַלְאָכִים מַרְנִינִים



לְהָבִיא אֵלֵינוּ הַשְּׁמוּעָה:


  


כִּי פָרְחָה בַשְּׁדֵמוֹת הַקָּמָה,



כִּי יָרֹן כָּל פֶּרַח לְמִינוֹ,



כִּי חָנְטוּ הַיְעָרִים מִסָּבִיב,



כִּי יֶהְגֶּה כְּבָר גּוֹזָל מִקִּנּוֹ;


  


וּבְגַלֵּי הָאָחוּ יִתְהוֹלְלוּ:



פֹּה עֵגֶל בֶּן־בָּקָר, פֹּה סַיָּח;



וּמְשֻׁבָּץ בְּחוֹפָיו כִּבְעֶרֶשׂ,



מִתְפַּנֵּק הַפֶּלֶג וּמְחַיֵּךְ.


  


וְקוֹרֵא קוֹל־קַיִץ אֵלֵינוּ:



צְאוּ־נָא מִכִּבְשַׁן הַכְּרַכִּים!



הַזְקִיפוּ שֶׁדְרַתְכֶם הַשְּׁחוֹחָה



וְחַזְּקוּ שְׁרִירֵיכֶם הַנְּמַקִּים.


  


הַדְלִיקוּ בְעֵינְכֶם נֵר־חַיִּים



וְרַפְּאוּ נִשְׁמַתְכֶם הָרְעוּעָה;



טִבְלוּהָ בְאוֹרוֹת אַרְגָּמָן



וּבְטַל הַשָּׁמַיִם רַחֲצוּהָ.


  


הִמָּלְטוּ מִתִּימְרוֹת הָאָבָק



וְלִינוּ בְנָוִים שַׁאֲנַנִּים,



הַטְּמִירִים בְּחֵיק הָעֲמָקִים



וְאוֹבְדִים בִּירָקוֹת וְגַנִּים.


  


תַּשְׁכִּימוּ עִם עַמּוּד־הַשַּׁחַר



וּתְטַיְּלוּ בִשְׁבִילֵי הַכְּרָמִים:



שָׁם תִּרְווּ מִדֶּשֶׁן הַתְּנוּבוֹת



וּשְׁכַרְתֶּם מֵחֲרִיפוּת־הַבְּשָׂמִים.


  


וּפְטוּרִים מֵחוֹבוֹת וְעֻלִּים



תִּתְפַּשְּׁטוּ הַגַּשְּׁמוּת הַשְּׁחוֹרָה –



וְקָרַם עֲלֵיכֶם עוֹר חָדָשׁ



וּטְהַרְתֶּם כִּכְרוּבֵי־הָאוֹרָה;


  


וְצָצוּ בִן־רֶגַע עַל גַּבְּכֶם



גַּפַּיִם קַלִּילִים אַוְרִירִים –



וּדְאִיתֶם לְרוּמוֹ שֶׁל עוֹלָם



וּצְלַלְתֶּם בְּזִיוִים אַדִּירִים.


  


וּרְאִיתֶם כָּל גִּנְזֵי הַבְּרִיאָה,



כָּל חָסְנָהּ וֶעֱזוּזֵי תִפְאַרְתָּהּ –



וּנְשַׁמְתֶּם בְּחָזוֹת אַבִּירִים,



וְלִבְּכֶם הַמְחֻדָּשׁ יְקַרְטָע.


  

תרס"ט
