


	
	סיפור חנוכה
	



סִפּוּר חֲנֻכּה: חִזּיוֹן לִקְטַנִּים

הַמּשְׁתַּתְּפִים:


סַבְתָּא,

נֶכֶד א',

נֶכֶד ב',

נֶכֶד ג',

נֶכֶד ד',

נֶכֶד ה'.

הַנְּכָדִים א‘, ב’, ג', מְדַּבְּרִים לְאַט לְאַט,

וְהַנְּכָדִים ד‘, ה’ – מַהֵר מַהֵר.

בְּהִתְרוֹמֵם הַמָּסָךְ נִרְאִים סַבְתָּא וּנְכָדֶיהָ מִסָּבִיב לָהּ.

נֶכֶד ג' מַרְבֶּה לֶאֱכוֹל.

סַבְתָּא: יְלָדַי, נְכָדַי. נֵרוֹת דוֹלְקִים עַל סַף הַחַלוֹן, אוֹר
בְּכָל הַבַּיִת, אֲנִי טוֹבָה וְאַתֶּם טוֹבִים, אַתֶּם נֶחְמָדִים
וַאֲנִי נֶחְמָדָה…

נֶכֶד א': כֵּן, כֵּן. סַבְתָּא נֶחְמָדָה.

כָּל הַנְּכָדִים (מְלַּטְפִים וּמְגַפְּפִים אֶת הַזְּקֵנָה): סַבְתָּא
נֶחְמָדָה, נֶחְמֶדֶת!

סַבְתָּא: אַתֶּם אוֹהֲבִים אוֹתִי?

הַנְּכָדִים: כֵּן, כֵּן.

סבתא: מִשּׁוּם מָה?

נֶכֶד א': מִשּׁוּם שֶאַתְּ סַבְתָּא שֶׁלָּנוּ.

נֶכֶד ב': מִשּׁוּם –

נֶכֶד ג': שֶׁאַתְּ –

נֶכֶד ד': סַבְתָּא –

נֶכֶד ה': שֶׁלָּנוּ! (צְחוֹק גָּדוֹל)

נֶכֶד א': גַּם –

נֶכֶד ב': מִשּׁוּם –

נֶכֶד ג': שֶׁאַתּ –

נֶכֶד ד': מְסַפֶּרֶת –

נֶכֶד ה': סִפּוּרִים יָפִים (מוֹחֵא כָּף)!

סַבְתָּא: כֵּן, כֵּן. כָּל הַיְּלָדִים וְכָל הַנְּכָדִים אוֹהֲבִים
סִפּוּרִים נֶחְמָדִים.

נֶכֶד א': אִם כֵּן –

נֶכֶד ב': סַבְתָּא –

נֶכֶד ג': סַפְּרִי לָנוּ –

נְכָדִים ד‘, ה’: גַּם הַיּוֹם סִפּוּר יָפֶה! (מוֹחֲאִים כָּף)

סַבְתָּא: אֲסַפֵּר, אֲסַפֵּר, אִם…

נֶכֶד א': נִהְיֶה יְלָדִים טוֹבִים –

נֶכֶד ב': וַחֲבִיבִים –

נֶכֶד ג': וְאוֹכְלִים הַרְבֵּה (מְמַלֵּא פִּיו אֹכֶל) –

נֶכֶד ד': וּמְשַׂחֲקִים הַרְבֵּה –

נֶכֶד ה': וּמְזַמְּרִים יָפֶה.

סַבְתָּא: כֵּן, כֵּן. אִם אַתֶּם תָשִׁירוּ שִׁיר יָפֶה, אֲנִי
אֲסַפֵּר לָכֶם סִפּוּר יָפֶה.

נֶכֶד ג': וְאִם אֲנַחְנוּ נֹאכָל הַרְבֵּה (מְמַלֵּא פִּיו אֹכֶל)
גַּם אַתְּ תּאֹכְלִי הַרְבֵּה.

סַבְתָּא: כֵּן, כֵּן. מַעֲשֵׂה זְקֵנִים סִימָּן לְבָנִים, וּמַעֲשֵׂה
בָנִים סִימָן לִזְקֵנִים.

נֶכֶד ה': גַּם אֲנִי יוֹדֵעַ לְהַגִּיד זֹאת: מַעֲשֵׂה זְקֵנִים
סִימָּן לִ…זְקֵנִים, וּמַעֲשֵׂה בָנִים סִימָּן לְ…בָנִים.

נֶכֶד א': כֵּן, כֵּן. זְקֵנִים וּבָנִים, מַעֲשִׂים וְסִימָּנִים.

נֶכֶד ב': הֲלֹא זֶה שִׁיר! הָבָה נָשִׁירָה יַחַד (שָׁרִים):

זְקֵנִים וּבָנִים,

מַעֲשִׂים, סִימָּנִים…

נוֹטְעִים בַּגַּנִּים -

יֵשׁ פְּרִי עֲדָנִים.

בָּנִים וּזְקֵנִים,

כָּל בְּנֵי יִשְׂרָאֵל,

עוֹבְדִים אֲדָמָה

בְּעֵמֶק יִזְרְעֶאל.

כָּל בְּנֵי יִשְׂרָאֵל,

וּבָנוֹת חֲבִיבוֹת,

עוֹבְדִים וּשְׂמֵחִים

וְ…

נֶכֶד ג': (מִתְרוֹנֵן לְבָדוֹ) וְ… אוֹכְלִים לְבִיבוֹת! (נוֹפֵחַ
בַּלְּחָיַיִם)

(כֻּלָם חוֹזְרִים עַל הַבַּיִת הָאַחֲרוֹן)

נֶכֶד א': עַתָּה, סַבְתָּא, סִפּוּר, סִפּוּר.

כֻּלָּם: עַתָּה, סִפּוּר, סִפּוּר…

סַבְתָּא: טוֹב. אֲבָל אֵיזֶה סִפּוּר רוֹצִים אַתֶּם לִשְׁמוֹעַ?
סִפּוּר “עָצוּב” או סִפּוּר “שָׂמֵח”?

נֶכֶד א': סִפּוּר עָצוּב, עָצוּב.

נֶכֶד ב': שֶׁיִּהְיוּ דְמָעוֹת בָּעֵינַיִם.

נֶכֶד ג': כָּכָה, שֶׁיִּהְיֶה כָּל כָּךְ עָצוּב, שֶׁלֹא נִרְצֶה
לֶאֱכֹל…

נֶכֶד ד': לא, סַבְתָּא. לא עָצוּב. סִפּוּר שָׂמֵח.

נֶכֶד ה': כָּכָה, שֶׁיִּהְיֶה כָּל כָּךְ שָׂמֵח, שֶׁיִּהְיֶה
צְחוֹק בָּעֵינַיִם. שֶׁנִרְצֶה לִרְקֹד, לִרְקֹד! אֲנִי נֶכֶד
שֶׁאוֹהֵב לִרְקֹד.

סַבְתָּא: אִם כֵּן, אֵיזֶה סִפּוּר אֲסַפֵּר לָכֶם?

נֶכֶד א': עָצוּב –

נֶכֶד ב': עָצוּב –

נֶכֶד ג': עָצוּב – – –

נכדים ד' ה': עַלִּיז, עַלִּיז, עַלִּיז, עַלִּיז – – –

סַבְתָּא: וַאֲנִי מָה אֶעֱשֶׂה? אֵלֶּה מְבַקְשִׁים עָצוּב, וְאֵלֶּה
מְבַקְשִׁים שָׂמֵח וְעַלִּיז…

נֶכֶד א': וְסַבְתָּא מָה תַעֲשֶׂה?

נֶכֶד ב': וְסַבְתָּא מָה תַעֲשֶׂה?

נֶכֶד ג': וְסַבְתָּא מָה תַעֲשֶׂה?

נֶכֶד ד': תְּסַפְּרִי לָנוּ סִפּוּר גַּם עָצוּב וְגַם שָׂמֵח…

נֶכֶד ה': שֶׁנִבְכֶּה וְנִצְחַק, שֶׁנִבְכֶּה וְנִרְקֹד!…

נֶכֶד א': כֵּן, הַרְבֵּה פְּעָמִים כָּכָה: אֲנוּ בּוֹכִים
וְצוֹחֲקִים, צוֹחֲקִים וּבוֹכִים…

נֶכֶד ב': רֶגַע בּוֹכִים וְרֶגַע רוֹקְדִים…

נֶכֶד ג': רֶגַע אוֹכְלִים וְרֶגַע לֹא אוֹכְלִים – דְּבָרִים
טוֹבִים…

נֶכֶד ד': בַּחֲצִי פָּנִים נִבְכֶּה וּבַחֲצִי פָּנִים נִצְחַק…

נֶכֶד ה': עַיִן אַחַת תִּבְכֶּה וְעַיִן אַחַת תִּצְחַק…

סַבְתָּא: אֶעֱשֶׂה יוֹתֵר טוֹב. אֲסַפֵּר לָכֶם סִפּוּר, שֶׁהוּא
בַּתְּחִילָה עָצוּב וְאַחַר כָּךְ – שָׂמֵחַ…

נֶכֶד ה': זֶה נֶחְמָד! הֶאָח! כֵּן!

כֻּלָּם: זֶה טוֹב, טוֹב מְאֹד.

נֶכֶד א': כָּל הַסִּפּוּרִים כָּךְ, תְּחִילָּה עֲצוּבִים וְאַחַר
כָּךְ שְׂמֵחִים…

סַבְתָּא: כֵּן, כֵּן. עַתָּה אֲסַפֵּר:

"הָיֹה הָיָה מֶלֶךְ יְוָנִי, מֶלֶךְ רָשָׁע, וּשְׁמוֹ – – – אַנְ – טִי
– יּוֹ – כוּס – אֶפִּי – פַ – נּוּס…

אָכֵן, זֶה עָצוּב, אִמְרוּ בְּעֶצֶב, אַנְ – טִי – יּוֹ – כוּס – אֶפִּי –
פַ – נּוּס…

נֶכֶד א' (בְּעֶצֶב. מַנְמִיךְ אֶת קוֹלוֹ): אַנְ – טִי – יּוֹ – כוּס
– אֶפִּי – פַ – נּוּס…

נֶכֶד ב' (בְּעֶצֶב. מַנְמִיךְ קוֹלוֹ יוֹתֵר): אַנְ – טִי – יּוֹ –
כוּס – אֶפִּי – פַ – נּוּס…

נֶכֶד ג' (מַנְמִיךְ קוֹלוֹ עוֹד יוֹתֵר): אַנְ – טִי – יּוֹ – כוּס –
אֶפִּי – פַ – נּוּס…

נֶכֶד ד' וה' (בְּעֶצֶב, אֲבָל בִּמְהִירוּת): אַנְ – טִי – יּוֹ –
כוּס – אֶפִּי – פַ – נּוּס…

סַבְתָּא: “צִוָּה אַנְטִיוֹכוּס עַל הַיְּהוּדִים לְהִשְׁתַּחֲוֹות
לַפֶּסֶל”.

גַּם זֶה עָצוּב, עָצוּב מְאֹד: לְהִשְׁתַּחֲוֹות לַפֶּסֶל… כָּאן
צָרִיךְ לִבְכּוֹת מְעָט…

נֶכֶד א' (בוכה): לְהִשְׁ – תַּ – חֲוֹות – – לַ – פֶּ – סֶל.

(שְׁאָר הַנְּכָדִים עוֹשִׂים כָּךְ זֶה אַחַר זֶה)

נֶכֶד א': סַבְתָּא, וְאֵיפֹה הַ“שָּׂמֵחַ”?

סַבְתָּא: “הָיֹה הָיָה אָז צַדִּיק גָּדוֹל, מַתְֵּתְיָהוּ
הַחַשְׁמוֹנָאִי”.

נְכָדַי, זֶה שָׂמֵחַ: מַתַּתְיָהוּ הַחַשְׁמוֹנָאִי!

נֶכֶד א' (בְּשִׂמְחָה, מַגְבִּיר קוֹלוֹ): מַתַּתְיָהוּ
הַחַשְׁמוֹנָאִי!

נֶכֶד ב' (בְּשִׂמְחָה, מַגְבִּיר קוֹלוֹ יוֹתֵר): מַתַּתְיָהוּ
הַחַשְׁמוֹנָאִי!

נֶכֶד ג' (בְּשִׂמְחָה, מַגְבִּיר קוֹלוֹ יוֹתֵר): מַתַּתְיָהוּ
הַחַשְׁמוֹנָאִי!

נֶכֶד ד' וה' (בְּשִׂמְחָה, מַגְבִּירִים קוֹלָם): מַתַּתְיָהוּ
הַחַשְׁמוֹנָאִי!

סַבְתָּא (בְּשִׂמְחָה): הָיוּ לְמַתַּתְיָהוּ חֲמִישָׁה בָּנִים
גִּבּוֹרִים (סוֹפֶרֶת עַל הָאֶצְבָּעוֹת): יוֹחָנָן, שִׁמְעוֹן,
אֶלְעָזָר, יוֹנָתָן וִיְהוּדָה הַמַּכַּבִּי…

זֶה שָׂמֵחַ, כָּאן צָרִיךְ לִשְׂמוֹחַ: וִיְהוּדָה הַמַּכַּבִּי!

כֻּלָם (מוֹנִים עַל הָאֶצְבָּעוֹת): יוֹחָנָן, שִׁמְעוֹן, אֶלְעָזָר,
יוֹנָתָן וִיְהוּדָה הַמַּכַּבִּי.

סַבְתָּא: אָז – אָז – אָז – מַתַּתְיָהוּ הַחַשְׁמוֹנָאִי וַחֲמֵשֶׁת
בָּנָיו הַגִּבּוֹרִים גֵּרְשׁוּ אֶת כָּל הַיְּוָנִים מֵאֶרֶץ
יִשְׂרָאֵל…

כֻּלָם: בְּאֱמֶת?

סַבְתָּא: בְּאֱמֶת! לֹא נִשְׁאַר אַף יְוָנִי אֶחָד!

נֶכֶד א': אַף לֹא רֹאשׁ אֶחָד?

נֶכֶד ב': אַף לֹא רֶגֶל אֶחָת?

נֶכֶד ג': אַף לֹא פֶּה אֶחָד?

נֶכֶד ד': אַף לֹא אֹזֶן אַחַת?

נֶכֶד ה': אַף לֹא אַף אֶחָד? (מַרְאֶה עַל הַחֹטֶם)

סַבְתָּא: “אֶת כָּל הַיְּוָנִים וְאֶת כָּל הַפְּסִילִים וְאֶת כָּל
הָאֱלִילִים, אֶת הַכָּל, אֶת הַכָּל שָׁלְחוּ וְהִשְׁלִיכוּ”…

נֶכֶד א': וּמֶה הָיָה אָז אַחַר כָּךְ?

סַבְתָּא: אָז – אָז – אָז – בָּאוּ הַיְהוּדִים אֶל בֵּית
הַמִּקְדָּשׁ וְהִדְלִיקוּ שֶׁמֶן טוֹב, שֶׁמֶן זַיִת זָךְ, וְעָשׂוּ אוֹר
גָּדוֹל, וְחַג גָּדוֹל: חֲנֻכָּה!

עַתָּה צָרִיךְ לִשְׂמֹחַ הַרְבֵּה, הַרְבֵּה!

אִמְרוּ כֻּלְכֶם:

"וְעָשׂוּ אוֹר גָּדוֹל וְחַג גָּדוֹל: חֲנֻכָּה!

כֻּלָם (יוֹצְאִים בְּמָחוֹל עִם סַבְתָּא): וְעָשׂוּ אוֹר גָּדוֹל וְחַג
גָּדוֹל: חֲנֻכָּה! (מִסְתּוֹבְבִים בַּמַעְגָּל):

אוֹר גָּדוֹל – אֲבוּקָה!

וְשִׂמְחָה – מְתֻקָה!

וְהַחַג – חֲנֻכָּה!

הַמָּסָךְ
