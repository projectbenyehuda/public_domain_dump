


	
	מגילת שמיטה
	



תמונה 1

נמצאה בגנזי החורבה.

נדפסה פעם ראשונה בשושן ועתה
יצאה לאור במהדורא תנינא ומתוקנת לקראה כדין וכמצוה בכל שנה של שמטה

ועם שני פרושים בעזרת החונן לאדם דעת ממני הק' קַל מִסֻּבִּין.





א


וַיְהִי בִּימֵי1 ר' דָוְצִי
הוּא ר' דָוְצִי הַמּוֹלֵךְ עַל חוּרְבָה וּמְזוּזָה שִׁבְעָה וְעֶשְׂרִים
וּמֵאָה וְיוֹתֵר חֲסִידִים: בַּיָּמִים הָהֵם כְּשֶׁבֶת הָרַב ר' דָוְצִי
עַל כִּסֵּא רַבָּנוּתוֹ אֲשֶׁר בִּירוּשָׁלַיִם הַבִּירָה: בִּשְׁנַת
עֶשֶׂר לְמָלְכוֹ עָשָׂה מִדְחֶה לְכָל הַיִּשׁוּב הֶחָדָשׁ חֵיל
הַפְּרוּשִׁים וְהַחֲסִידִים הַפַּרְתְּמִים וְשָׂרֵי הַמְּדָנִים
לְפָנָיו: בְּהַרְאוֹתוֹ אֶת עשֶׁר כְּבוֹד תּוֹרָתוֹ וְאֶת יְקָר
תִּפְאֶרֶת חוּמְרָתוֹ יָמִים רַבִּים אִסּוּרִים בְּכָל יוֹם: וּבִמְלֹאת
הָאִסּוּרִים הָאֵלֶה עָשָׂה ר' דָוְצִי לְכָל הָעָם הַנִּמְצְאִים
בִּירוּשָׁלַיִם הַבִּירָה אִסּוּר שְׁמִיטָה לְשֶׁבַע שָׁנִים2
בְּחוּר הַחוּרְבָה שֶׁל יְהוּדָה הֶחָסִיד: חוּר
יַרְמוּלְקָה וּתְכֵלֶת אָחוּז בְּחַבְלֵי־בוֹץ וְקַיְמָא־לָן עַל גְלִילֵי
כֶסֶף וְעַמּוּדֵי שַׁס שִׁטּוֹת הָרִיא וְהָרִיד3 עַל רִצְפַּת בַּטְלָנוּת וְדוּק וְסֹחָרֶת: וְהַקְשׁוֹת
מִכְּלִי יָקָר4 וְכֵלִים מִכֵּלִים שׁוֹנִים
וּפִלְפּוּל יֵשׁ לָרָב כְּיַד הַמֶּלֶךְ: וַיֵּצֵא הָאִסּוּר הַשְּׁתִיָה
לֹא כַדָּת אֲנִי אוֹנֵס כֵּן יִסַּד ר' דָוְצִי עִם כָּל רַבָּנֵי בֵּיתוֹ
שֶׁלֹא לַעֲשׂוֹת כִּרְצוֹן אִישׁ וָאִישׁ: ס וַיְמָאֵן הָרַב
קוּק5 לְמַלֵּא תּוֹרַת ר' דָוְצִי אֲשֶׁר בְּיַד הַסָּרִיסִים6 וַיְחַבֵּר וַיְפָרֵשׁ
הֵיתָר לַשְׁמִיטָה וְתוֹרָתוֹ בָּעֲרָה בוֹ: וַיֹּאמֶר ר' דָוְצִי
לַבַּטְלָנִים לֹא־יֹדְעֵי הָעִתִּים כִּי כֵן דְבַר ר' דָוְצִי לִפְנֵי
כָּל יוֹדְעֵי דָת וָדִין: וְהַקָּרֹב אֵלָיו כַּרְסָנִי7 שׁוֹטֵר אַדְמוֹנִי בַקְשִׁישׁ מֶרֶץ חַרְסִינָה מְהוּמָן
שִׁבְעַת שָׂרֵי פְרוּשִׁים וַחֲסִידִים רוֹאֵי פְּנֵי ר' דָוְצִי
הַיּוֹשְׁבִים רִאשׁוֹנָה בַּמַּלְכוּת8: כְּדָת
מַה לַעֲשׂוֹת בָּרַב קוּק עַל אֲשֶׁר לֹא עָשָׂה כְּמַאֲמַר הָרַב ר'
דָוְצִי בְּיַד הַסָּרִיסִים: ס וַיֹאמֶר מְהוּמָן לִפְנֵי ר' דָוְצִי
וְהַדַּיָנִים לֹא עַל ר' דָוְצִי לְבַדּוֹ עָוָה הָרַב קוּק כִּי
עַל־כָּל־הָאֲסֵפוֹת וְעַל־כָּל הָעִתּוֹנִים אֶת עַצְמוֹ יַחַז לְרֹאשׁ:
וְהַיּוֹם הַזֶּה תֹּאמַרְנָה קְהִלוֹת הַפְּרוּשִׁים וְהַחֲסִידִים
לְהַבְזוֹת בַּעֲלֵי הַתּוֹרָה בְּאָמְרָן הָרַב קוּק נָאוֹר9
וּמְדַבֵּר דְבָרִים שֶׁל טַעַם וּכְדַי בִּזָּיוֹן
וָקָצֶף: אִם־עַל־רַ' דָוְצִי טוֹב יֵצֵא דְבַר־הַגְּזֵרָה מִלְפָנָיו
בְּכָל תֹּקֶף וְיִכָּתֵב בְּחוֹתַם הַפְּרוּשִׁים וְהַחֲסִידִים בִּטּוּל
הַהֵיתָר שֶׁל קוּק וְכָל הָאֲנָשִׁים יִתְּנוּ יְקָר לְבַעֲלֵי הַתּוֹרָה
לְמִגָּדוֹל וְעַד קָטָן10: וַיִּיטַב הַדָּבָר
בְּעֵינֵי רַ' דָוְצִי וְהַדַּיָּנִים וַיַּעַשׂ רַ' דָוְצִי כִּדְבַר
מְהוּמָן: וַיִּשָּׁלְחוּ סְפָרִים אֶל־כָּל־חֲנוּיוֹת יְרוּשָׁלַיִם אֶל
חֲנוּת וַחֲנוּת כִּכְתָבָהּ וְאֶל מוֹזֵג וּמוֹזֵג
כִּלְשׁוֹנוֹ11 לִהְיוֹת כָּל אִישׁ סוֹרֵר בְּבֵיתוֹ וּמִתְנַגֵּד לְעִנְיָנֵי עַמּוֹ: ס






ב


בְּעֵת הַדְּבָרִים הָאֵלֶה גִדַּל הָרַב רַ' דָוְצִי אֶת־יַדְלֶר בֶּן־הַמְּדָנִים הַדֵּימַגּוֹגִי12 וַיְנַשְּׂאֵהוּ וַיָּשֶׂם אֶת כִּסְאוֹ אֵצֶל הַבַּדַּצְ אֲשֶׁר
אִתּוֹ: וַיִּבֶז בְּעֵינָיו לִשְׁלֹחַ יָד בַּשְׁתִיָּה לְבַדָּהּ
וַיְבַקֵּשׁ יַדְלֶר לְהַשְׁמִיד אֶת כָּל יִשּׁוּב הַיְּהוּדִים
וַעֲתִידָם13 אֶת הַסַּבּוֹן וְהַבֻּלְבּוּס14 וְאֶת הָעוֹלְשִׁין
וְהַמַּמְתַּקִּים: וַיֹאמֶר יַדְלֶר לָרַב רַ' דָוְצִי יֶשְׁנוֹ עַם־אֶחָד
מְפֻזָּר וּמְפֹרָד15 בֵּין הַמּוֹשָׁבוֹת
וְדָתֵיהֶם לְקוּחוֹת אֵצֶל קוּק וְאֶת דָּתֵי רַ' דָוְצִי אֵינָם עוֹשִׂים
וּלְרַ' דָוְצִי אֵין שׁוֶֹה לְהַנִּיחָם: אִם עַל רַ' דָוְצִי טוֹב
יִכָּתֵב לְאַבְּדָם וַעֲשֶׂרֶת אֲלָפִים פְרַנְקִים־כֶּסֶף נִשְׁקֹל
מִידֵי הָרִידְבָז16 לְהָבִיא אֶל גִנְזֵי הַחֲלוּקָה17: וַיֹּאמֶר רַ' דָוְצִי לְיַדְלֶר
הַכֶּסֶף נָתוּן לָנוּ וְהַיִּשּׁוּב לַעֲשׂוֹת בּוֹ כַּטּוֹב בְּעֵינֶיךָ:
ס וּבַלַּיְלָה הַהוּא נָדְדָה שְׁנַת רַ' חַיִּים־בֶּרְלִין וַיֹּאמֶר
לְהָבִיא אֶת הַחֻמָשׁ שֶׁל אַבָּא וַיְהִי נִקְרָא לִפְנֵי בֶרְלִין:
וַיִּמָצֵא כָתוּב כִּי אֵין הֵיתָר לָעֲנָבִים שֶׁל הַיְהוּדִים אֲבָל
יֵשׁ הֵיתָר לִבְנֵי בִיסְמַרְק וּשְׁטֶקֶר18 הַמְבַקְּשִׁים לִשְׁלֹחַ יַד בְּקִיּוּם הַיְהוּדִים: וַיֹּאמֶר
בֶּרְלִין מִי בַחוּרְבָה וְיַדְלֶר בָּא לְבֶרְלִין לְבַקֵּשׁ גַם
מִמֶּנּוּ לִתְלוֹת אֶת הַיִּשּׁוּב עַל הַקּוֹץ אַשֶׁר הֵכִינוּ
בְּפִלְפּוּל: וַיֹּאמֶר בֶּרְלִין לְיַדְלֶר מַהֵר קַח אֶת הַלְבוּשׁ
וְאֶת הַסְּמַג19 כַּאֲשֶׁר דִבַּרְתָּ וַעֲשֵׂה
כֵן לַיִּשּוּב הַיְהוּדִי אַל תַּפֵּל דָּבָר מִכֹּל אֲשֶׁר דִבַּרְתָּ:
וַיִּקָּרְאוּ סוֹפְרֵי רַ' דָוְצִי בְחֹדֶשׁ תַּמוּז בְּשִׁבְעָה יָמִים
בּוֹ וַיִּכָּתֵב כְּכָל־אֲשֶׁר־צִוָּה יַדְלֶר אֶל בֶּן שָׁמֶן וְאֶל
רִאשׁוֹן וְאֶל גְדֵרָה וּרְחוֹבוֹת מוֹשָׁבָה וּמוֹשָׁבָה כִּכְתָבָהּ
אִישׁ וְאִישׁ כִּלְשׁוֹנוֹ לְהַזִּיק לְכָל הַיְהוּדִים שֶׁל הַיִּשּׁוּב
הֶחָדָשׁ וּשְׁלָלָם לָבוֹז: פַּתְשֶׁגֶן הַכְּתָב לְהִנָּתֵן דָּת בְּכָל
מוֹשָׁבָה וְמוֹשָׁבָה לִהְיוֹת עַתּוּדִים הַיּוֹם
הַזֶּה20: ס וּמֹרִיָּה21 יָצְאָה מִלִּפְנֵי הַבַּדַּצְ בִּלְבוּשׁ אוֹרְגַן פְּסֹלֶת
וָבוּר וְתַכְרִיךְ בּוֹץ וְזֻהֲמָן22 וּמַאֲמָרֶיהָ מְלֵאִים אֶרֶס לְיִּשּׁוּב הַיְּהוּדִים:
וְהָרָצִים עִם הַפֵּאוֹת יָצְאוּ דְחוּפִים בִּדְבַר רַ' דָוְצִי וְהַדָּת
נִתְּנָה בִּירוּשָׁלַיִם הַבִּירָה וְרַ' דָוְצִי וְיַדְלֶר אָסְרוּ
לֶאֱכֹל וְלִשְׁתּוֹת וְהַמּוֹזְגִים הִתְקַשְּׁרוּ עִם הַגֶּרְמַנִּים
וְהָעִיר יְרוּשָׁלַיִם נָבוֹכָה: ס






ג


וְרִאשׁוֹן יָדְעָה אֶת־כָּל־אֲשֶׁר נַעֲשָׂה וַתִּקְרַע רִאשׁוֹן אֶת בִּגְדֵי הַמֶּשִׁי23 וַתַּשְׁחֵר
פָּנֶיהָ כָּאֵפֶר וַתִּשְׁלַח מַלְאָכִים לִירוּשָׁלַיִם וַיִּזְעֲקוּ
זְעָקָה גְדוֹלָה וּמָרָה: וַיָבֹאוּ עַד לִפְנֵי שַׁעַר הַחוּרְבָה לָבוֹא
אֶל הַבַּדַּצְ לְהִתְחַנֵּן24 לוֹ וּלְבַקֵּשׁ מִלְפָנָיו עַל עַמָּם25: וַיֹאמְרוּ לָהֶם כָּל
שַׁמָּשֵׁי הַבַּדַּצְ וְעַם יְרוּשָׁלַיִם יוֹדְעִים אֲשֶׁר כָּל־אִישׁ
וְאִשָּׁה אֲשֶׁר יָבוֹא אֶל הַבַּדַּצְ אֶל הֶחָצֵר הַפְּנִימִית אַחַת
דָּתוֹ לְהָמִית זְמַנּוֹ לְבַד מֵאֲשֶׁר יוֹשִׁיט שַׁרְבִיט הַזָּהָב
וְחָיָה: וַיֹאמְרוּ הַמַּלְאָכִים אַל תְּדַמּוּ בְנַפְשְׁכֶם לִמְלֹךְ
בְּאִסּוּרֵיכֶם עַל־כָּל עִסְקֵי הַיְּהוּדִים לִפְנֵי שִׁשָּׁה חֳדָשִׁים
עַל שֶׁמֶן שֻׁמְשֻׁם וְאַחֲרֵי שִׁשָּׁה חֳדָשִׁים עַל סַבּוֹן
קוּדְרַנְסְקִי26 וְתַמְרוּקֵי הַנָּשִׁים:
וּבְהַגִּיעַ תּוֹר עֲנָבִים וַעֲנָבִים לָבוֹא לִירוּשָׁלַיִם אַל־נָא
יַפְקִיד הַבַּדַּצְ פְּקִידִים בְּכָל רְחוֹבוֹת רַבָּנוּתוֹ לִבְלִי
קַחַת עִנְבֵי הַיְּהוּדִים וְסַבּוֹנָם וְתַמְרוּקֵיהֶם: כִּי אִם־הַחֲרֵם
תַּחֲרִימוּ בָּעֵת הַזֹּאת רֶוַח וְהַצָּלָה יַעֲמֹד לַיְּהוּדִים
מִמָּקוֹם אַחֵר27 וְעֶרְכֵי תוֹרַתְכֶם
יֹאבֵדוּ וּמִי יוֹדֵעַ אִם לְעֵת כָּזֹאת לֹא יַגִּיעַ הַדָּבָר
לַמַּלְכוּת28: כִּי אֵיכָכָה נוּכַל וְרָאִינוּ
בָּרָעָה אֲשֶׁר־יִמְצָא אֶת עַמֵּנּוּ וְאֵיכָכָה נוּכַל וְרָאִינוּ
בְּאָבְדַּן מוֹלַדְתֵּנוּ: וַיִּשְׁמְעוּ כָּתְלֵי הֶחָצֵר וְלֹא עָנוּ
דָבָר וְיַדְלֶר נִדְחַף אֶל הַבַּדַּצְ בְּיַרְמֻלְקָתוֹ חֲפוּי רֹאשׁ:
ס



אִישׁ יְהוּדִי הָיָה בְיָפוֹ שֶׁל קוּק וּשְׁמוֹ מָרְדְּכַי בֶּן הִלֵּל בֶּן לֵוִי29 בֶּן
יַעֲקֹב אִישׁ בּוּרְגָנִי: אֲשֶׁר הָגְלָה מֵהוֹמֶל עִם הַגּוֹלָה אֲשֶׁר
הָגְלְתָה בִּימֵי נִיקוֹלַי הַצָּר: וַיְהִי אֹמֵן אֶת רִאשׁוֹן־לְצִיוֹן
כִּי כֵן מְיוּחָס גַם הוּא וְגַם יֶשׁ לוֹ שָׁם פַּרְדֵּס: וַיֹאמֶר
מָרְדְּכַי לְוַעַד רִאשׁוֹן: לֵךְ כְּנוֹס אֶת כָּל הַיְּהוּדִים
הַנִּמְצְאִים בַּמּוֹשָׁבוֹת וְאֶת סוֹפְרֵי עִתּוֹנֵי
יְרוּשָׁלַיִם30 וְאֶת פְּרוֹפֵיסוֹר שַׁץ וְאֶת
רְפָאֵלִי וְגַם אֶת בַּרְזִלַּי יְדִידֵנוּ: וְאַל תֹּאכְלוּ וְאַל
תִּשְׁתוּ שָׁלשׁ שָׁעוֹת לַיְלָה וָיוֹם וְגַם אֲנִי וּבְנֵי מֵעַי אָצוּם
כֵּן: לְהִקָּהֵל31 וְלַעֲמֹד עַל נַפְשֵׁנוּ
וְעַל הָעֲנָבִים וְעַל הַשְׁקֵדִים וּלְהַשְׁמִיד וּלְאַבֵּד אֶת כָּל
חֵיל הַחֲלוּקָה הַצָרִים אוֹתָנוּ וְנָשִׂים שְׁלָלָם לָבוֹז: וְנָתן
הַלְבוּשׁ32 וְהַזְכוּת לָרַב קוּק הַמֵּקֵל
וְהִרְכִּיבוּהוּ עַל גַבֵּנוּ וְקָרְאוּ לְפָנָיו כָּכָה יֵעָשֶׂה לָאִישׁ
אֲשֶׁר אֲנַחְנוּ חֲפֵצִים בִּיקָרוֹ: הַדָּבָר יָצָא מִפִּי מָרְדְכָי
וַיִּיטַב הַדָּבָר לִפְנֵי הַוַּעַד וַיַּעַשׂ הָעֵצָה:






ד


וַיִּקָּהֲלוּ הַיְּהוּדִים בְּרִאשׁוֹן־לְצִיּוֹן בְּבֵית
הָעָם בְּחֹדֶשׁ תַּמוּז בְּשִׁבְעָה עָשָׂר יוֹם בּוֹ: לְהַשְׁמִיד
וּלְאַבֵּד33 אֶת הַחֲלוּקָה עַל יְדֵי
וִכּוּחִים וְהַחֲלָטוֹת וּפִטְפּוּטֵי דְבָרִים עַל עִבְרִית וְזַרְגּוֹן:
וַיֹאמְרוּ הַמַּלְאָכִים כִּי יַדְלֶר בֶּן הַמְּדָנִים הַדֵּימַגּוֹגִי
צוֹרֵר יִשּׁוּב הַיְהוּדִים וְהַבַּדַּצְ וְהָרִידְבָז חוֹשְׁבִים
לְאַבְּדָם בַּפִּלְפּוּל הוּא הַגּוֹרָל לְלוֹמְדֵי תּוֹרָה
לְהֻמָּם34 וּלְאַבְּדָם: וּבַרְזִלַּי
וּגְלוּסְקִין וְשַׁץ וְשֵׁנְקִין וְאַהֲרֹנוֹבִיץ דִּבְּרוּ דִּבְרֵי אֵשׁ
וֶאֱמֶת מָה רָאוּ עַל כָּכָה וּמָה הִגִּיעַ אֲלֵיהֶם: וַיָּרֶם
זַרְגוֹנִי אֶת קוֹלוֹ וַתִּתְחַלְחַל הָאִכָּרָה35 מְאֹד וַתִּשְׁלַח לְהָבִיא בְּגָדִים וּמַחֲלָצוֹת לְהַלְבִּישׁ
לִפְנֵי הָאוֹר36 אֶת רַעֲיוֹנוֹתֶיהָ: וַיֹּאמֶר אַחַד הַסָּרִיסִים37 בַּכַּרְמֶל
גַּם הִנֵּה עֵץ הַגֶּפֶן אֲשֶׁר אָמַר יַדְלֶר לִתְלוֹת אוֹתָנוּ עָלָיו
קָבוּר מִבְּלִי זֶה חֲמִשִּׁים אַמָּה: וַיֹאמֶר אַבְשָׁלוֹם
פַיְנְבֶּרְגְ38 הִנֵּה עֵץ הַשְׁקֵדִים שֶׁל
יַדְלֶר עוֹמֵד בְּפֶתַח תִּקְוָה תְּלוּהוּ עָלָיו: וְלֹא נָתַן הַוַּעַד
לַעֲשׂוֹת כֵּן: וְקִבְּלוּ הַיְּהוּדִים דִבְרֵי הַוַּעֲדָה אֲשֶׁר
בָּחֲרוּ לִכְתֹּב וְאֶת אֲשֶׁר כָּתַב מָרְדְכַי לִפְנֵיהֶם
וּבַבִּזָה39־הַחֲלוּקָה לֹא שָׁלְחוּ אֶת יָדָם40: קִיְּמוּ וְקִבְּלוּ הַיְּהוּדִים
עֲלֵיהֶם וְעַל זַרְעָם וְעַל כָּל הַנִּלְוִים אֲלֵיהֶם וְלֹא יַעֲבֹר
לִהְיוֹת עוֹשִׂים אֶת הַבַּדַּצְ בִּירוּשָׁלַיִם מְבֻטָּלִים וְדִבְרֵי
הַחֲרָמוֹת וְזַעֲקָתָם וּבַבִּזָּה הַחֲלוּקָה לֹא שָׁלְחוּ אֶת יָדָם:
וַיַּכּוּ יְהוּדֵי הָאֲסֵפָה וְגִבּוֹרֵי הַפּוֹעֵל הַצָּעִיר אֶת
הָרַבָּנִים אוֹיְבֵיהֶם מַכַּת מַחָאוֹת וַיַעֲשׂוּ בְשׂוֹנְאֵיהֶם
כִּרְצוֹנָם: וּבִירוּשָׁלַיִם הַבִּירָה הִדְבִּיק וַעַד שׁוֹמְרֵי

בַּת־בְּנֵי־הַחַיִּל וּמָרְדְּכַי הַיְהוּדִי אֶת כָּל תֹּקֶף לְקַיֵּם
אֶת אִגֶּרֶת הַמַּחָאָה הַזֹּאת הָרִאשׁוֹנִית: לְקַיֵּם אֶת דִינֵי
הַשְׁמִטָּה בִּזְמַנֵּיהֶם כַּאֲשֶׁר קִיַּם עֲלֵיהֶם הָרַב קוּק
וְכַאֲשֶׁר קִיְּמוּ עַל נַפְשָׁם וְעַל זַרְעָם: עַל־כֵּן הַיְהוּדִים
הַפְרַזֵיאוֹלוֹגִים הַיוֹשְׁבִים בְּעָרֵי הַפְרַזּוֹת45 עוֹשִׂים אֶת יוֹם שִׁבְעָה עָשָׂר לְחֹדֶשׁ תַּמוּז וְיוֹם
שְׁמוֹנָה עָשָׂר בּוֹ לְיָמִים גְדוֹלִים וּמִשְׁלֹחַ כִּבּוּדִים אִישׁ
לְרֵעֵהוּ: וְכָל מַעֲשֵׂי תָקְפָּהּ וּגְבוּרָתָהּ וּפָרָשַׁת גְדוּלַת
הַמַּחָאָה הֲלֹא הֵם כְּתוּבִים עַל־עַמּוּדֵי הָאוֹר וְהַחֵרוּת וְעַל
קִירוֹת יָפוֹ וִירוּשָׁלַיִם: וְהָעִיר יְרוּשָׁלַיִם לֹא
צָהֲלָה46 וְלֹא שָׂמֵחָה: וְהָרַב קוּק אֲשֶׁר
בְּיָפוֹ כּוֹתֵב סְפָרִים וּמִכְתָּבִים וְהוּא גָדוֹל לַיְהוּדִים
וְרָצוּי לְרֹב אֶחָיו דוֹרֵשׁ טוֹב לְעַמּוֹ וְדֹבֵר שָׁלוֹם לְכָל
זַרְעוֹ47:


ברכות שנוהגין בצבור לפני קריאת המגילה

ברוך אתה המקום ירושלים עיר העולם אשר קדשנו במצותיו וצונו על מקרֵה מגילה.

ברוך אתה המקום ירושלים עיר העולם שעשה אפוטרופסים לדתנו בימים ההם בזמן הזה.

ברוך אתה המקום ירושלים עיר העולם שהחינו וקימנו והגיענו למצב הזה.
