


	
	אַחְלַי וְיָמִים נוֹכְלִים
	



אל יהודה – הלוי? –






אַחְלַי וְיָמִים נוֹכְלִים יִהְיוּ וְאַל / יִשְׁלוּ, וְיָשֹׁלּוּ אֲשֶׁר זָבָדוּ,

עַל מִי אֲדַבֵּר וְאֶל מִי אֶזְעַק וְכָל / אַנְשֵׁי שְׁלוֹמִי אֶל שְׁאוֹל צָעָדוּ,

אוֹ אֵיךְ בְּעֹז אָגִיל וְלֹא אָגִיל, וְהֵם / גָּלוּ לְיַד גַּלִּים וְהִתְעַתָּדוּ.

אַךְ מַר וְרָע מִזֹּאת פְּלִיטֵיהֶם אֲשֶׁר / שָׂחְקוּ לְמִשְׁבַּתַּי וְלֹא חָרָדוּ.

בָּגְדוּ כְמוֹ נַחַל וְלֹא נֶחְלוּ עֲלֵי / שִׁבְרִי, וְתֵַם שִׂבְרִי וְלֹא סָלָדוּ.

בִּי נֶהְפְּכוּ יַחַד וְרָעוֹת חָשְׁבוּ / לֹא סֻפְּרוּ מֵאָז וְלֹא הֻגָּדוּ,

עַד רָחֲקוּ לִבּוֹת אֲחָדִים נוֹצְרוּ / כִּרְחֹק פְּגָרֵימוֹ וְהִתְאֶחָדוּ.

אָנוּד בְּאֶרֶץ נוֹד אֲשֶׁר תּוֹכָהּ יְדֵי / יָמִים בְּעָקְבָה אֶת־עֲקֵבַי נָּדוּ,

אֶרֶץ אֲשֶׁר לֹא אֶחְסַר בָּהּ כִּי אֲשֶׁר / יִתַּמְּמוּ עִמִּי וְיִתְחַסָּדוּ

חִלּוּ בְּלֵב יָחוּל בְּמַחְלֻיָּיו עֲדֵי / גַּבֵּי כְבֵדִי מִנְּשֹׂא כָבָדוּ.

חָכְמוּ בְּעֵינֵיהֶם וְעִם הַבֹּעֲרִים/ רֹאשׁ נָשְאוּ לָהֶם אֲשֶׁר פָּקָדוּ.

אִיגַע וְיַרְגִּיעוּ וְדַרְכִּי חָמְדוּ, / כִּי אֶשְׂנְאָה דֶרֶךְ אֲשֶׁר חָמָדוּ,

יֵעָצְבוּ לִי מֵאֲשֶׁר אֶשְׂמַח, הֲכִי / אֶדְאַג אֲנִי מִכֹּל אֲשֶׁר בּוֹ חָדוּ.

אֹהֶל מְשׂוֹשִׂי הַזְּמָן הֶחְרִיב בְּיוֹם / הָאָהֳלִי צָעַן וְגָרָיו נָדוּ,

עִמָּם שְׁנוֹתַי קָצְרוּ כֵּן אָרְכוּ / לֵילוֹת נְדוּדֵיהֶם וְהִתְמוֹדָדוּ.

לֵילוֹת נְדוּדִים נִמְשְׁכוּ לָהֶם וְלֹא / שָׁבוּ בְאוֹר שַׁחַר וְלֹא פָקָדוּ.

אִמְרוּ: הֲנֶאֶסְפוּ כְסִילֵיהֶם עֲלֵי / שַׁחַק וּבַעְבוֹת הַנְּדוֹד נֶעְקָדוּ,

אוֹ הֵם נְבֻכִים מִמְּבוּכָתִי וְאִם / נִנְהוּ נְהוֹת עָלַי וְהִתְגּוֹדָדוּ.

אֶחְשֶׁה וְאֶתְאַפֵּק, הֲכִי מִתְאַפְּקִים / עַל שַׁעֲרֵי מוּסָר וְתֹם שָׁקָדוּ.

אֵיכָה יְבַהְלוּנִי יְלִידֵי יוֹם, וְהֵן / יַחְדָּו יְסוּפוּן כַּאֲשֶׁר נוֹלָדוּ.

דֵּעָה וְהַשְׂכֵּל הָפְקְדוּ עָלַי לְבַל / אֵחַת לְמִפְעָלָם וּבִי הוּעָדוּ.

שִׂיאִי עֲלֵי אַף שׂוֹטְנַי הֶעְלוּ עֲדֵי / עֶרֶשׂ לְרַגְלִי עַל כְּסִיל רָבָדוּ,

אֶשְׂנָא אֲשִׁישׁוֹתַי אֲשֶׁר יִתְאוֹשְׁשׁוּ, / אֶבְחַל בְּמִזְרָקַי אֲשֶׁר יָקָדוּ,

עַד אֶשְׁאֲלָה מַשְׁקִי: הֲיֵין גֶּפֶן סְדוֹם / אוֹ רֹאשׁ פְּתָנִים בַּכְּלִי הָפְקָדוּ?

מַה לִּי וְלֵב יִכְלֶה בְּקוֹל שִׁירִים וְאַךְ / יִכְלֶה וְיָגִיל אֶל אֲשֶׁר סָפָדוּ.

כָּל־תַּעֲנוּגֵי הַזְּמָן אֶמְאַס, לְבַד / חִידוֹת חֲכָמִים שֶׁכְּבָר נִפְקָדוּ,

אוֹ מִכְתְּבֵי אָחִי יְהוּדָה, כִּי לְפִי / מָתְקוּ וּמִכֶּתֶם וּפָז נֶחְמָדוּ.

הֵמָּה אֲשֶׁר רִפְיוֹן יְמִינִי חִזְּקוּ, / הֵמָּה אֲשֶׁר לִבִּי לְאַט סָעָדוּ.

נֹפֶךְ עֲלֵי שֹׁהַם בְּיַד קָלַע, וְאִם / טוּרִים בְּמֹר עַל הַכְּפוֹר, הוּרָדוּ.

מִדְבָּר לְפַתֵּחַ סְתוּמָיו נֶאְסְפוּ / שָׁרִים וְאֶל סוֹדוֹת בְּפִיו נוֹסָדוּ,

דֵי מַחֲסוֹר שֵׂכֶל לְשַׁאְנַנִּים וּבוֹ / אָרְחוּ בְנֵי אֹרַח וְהִצְטַיָּדוּ.



שִׁירִים לְנֶגְדָּם יִמְחֲאוּ כָף כָּל־עֲצֵי / בֹשֶׂם וְהָרֵי מָר־דְּרוֹר רָקָדוּ.



הֵם נִלְחֲמוּ עִם עָשׁ וּבָנֶיהָ, אֲבָל / רֹאשׁ הָפְכוּ נֶגְדָּם וְהִתְלַכָּדוּ.

שֹׁמְעֵי חֲרוּזֵיהֶם יְשִׂיחוּן: אֵיךְ בְּנֵי / אִישׁ כּוֹכְבֵי מָרוֹם בְּלִי יָד צָדוּ?

אֵיתָן וְהֵימָן לוּ זְמָמָיו שָׁמְעוּ / הִתְפָּאֲרוּ בָהֶם וְהִתְעוֹדָדוּ.

בּוֹ מִבְּנֵי לֵוִי יְדֵיהֶם גָּבְהוּ, / כִּי לוֹ יְצוּעֵי מַאֲמָר רֻפָּדוּ,

אוֹ שָׂכְלוּ אוֹתָם לְשׁוֹנוֹת נִפְרְדוּ / עָזְבוּ נְאֻם דָּתָם וְהִתְיֶהָדוּ.

יִחְיוּ בְעוֹד שֶׁמֶשׁ – וְכַמָּה מִדְבְּרֵי / חָזוֹן בְּאֹרֶךְ הַזְּמָן נִכְחָדוּ.

לוּ כָל חֲסָדִים נִבְרְאוּ יִהְיוּ לְמוּל / חַסְדּוֹ – בְּפִי הַשּׁוֹמְעִים חֻסָּדוּ.

אַל תִּתְמְהוּ לִבְנֵי שְׂעִפָּיו, כִּי אֲבוֹת / מַרְאוֹת לְהָעִיד מַעֲנָיו נוֹעָדוּ,

כַּאְשֶׁר בְּנֵי אַשְׁפָּה לְעֵטָיו יַעֲטוּ / בֹּשֶׁת בְּעֵת לִפְנֵי כְתָב סָגָדוּ,

עֵטִים כְּצִפְעוֹנִים, וְאֵיךְ הִשְׁתַּעְשְׁעוּ / בָּם אֶצְבְּעוֹתָיו וַעֲלֵיהֶם הָדוּ,

צָמוּ וּמֵרֹק הַדְּיוֹ יִמְצוּ, וְרֹק / לָרִיק בְּשִׁנַּיִם כְּצוּר חֻדָּדוּ.

נֹשְׁקִים וְהֵם נֹשְׁכִים, וּמִפִּיהֶם מְתֵי / שֵׂכֶל וְכֶסֶל רֹאשׁ וְנֹפֶת רָדוּ.

יוֹם אֶחֱזֶה פִּלְאֵי פְלָאִים אֹמְרָה: / מִמַּעֲשָׂיו לִבּוֹת אֱנוֹשׁ עֻוָּדוּ,



אוֹ אֶשְׁמְעָה מוֹפְתֵי חֲרָזִים אֶעֱנֶה: / מִפִּיו לְרִקְמַת [הַ]שְּׂרָד שָׂרָדוּ.



אֵלֶּה קְצוֹת דַּרְכֵי תְהִלּוֹתָיו אֲשֶׁר / אוֹתָם בְּצִבְאוֹתָם שְׁחָקִים פָּדוּ,

תִּסְפֹּר עֲפַר אֶרֶץ וְרוּם תָּמֹד – וְלֹא / יִסָּפְרוּ מֵרֹב וְלֹא יִמַדּוּ.
