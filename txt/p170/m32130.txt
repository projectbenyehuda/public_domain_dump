


	
	תְּנוּ צִיץ לְשִׁירוֹת
	



שיר הקדשה לאברהם אבן מֻהאג׳ר בראש ספר הענק.


תְּנוּ צִיץ לְשִׁירוֹת / וּמִלּוֹת נְכוֹחוֹת / לְאַחֵז בְּתֵבֵל / קְצָווֹת וְרוּחוֹת,



לְלַמֵּד לְכָל נִמְ/הֲרֵי לֵב תְּבוּנוֹת, / וְעִמְקֵי שְׂפָתוֹת / לְדַבֵּר בְּצַחוֹת,



וְלָשׁוּר נְגִידֵי / אֲמָרִים, בְּמַעְבֵה / לְבָבִי יְצַקְתִּים, / צְרַפְתִּים בְטוּחוֹת,



הֱיוֹתָם כְּאֹזֶן / לְשֵׂכֶל, וְעַיִן / לְדֵעָה, וְאֶל גּוּף / תְּעוּדוֹת כְּרוּחוֹת.



יְפִיחוּן כְּקָנֶה / לְאַפֵּי מְלָכִים / וְיִהְיוּ מְלָכוֹת / בְּרָקְחָם רְקוּחוֹת,



וְהֵם שַׁעֲשׁוּעִים / לְכָל שַׁאֲנַנִּים / וְצֵדָה לְדֶרֶךְ / לְעוֹבְרֵי אֳרָחוֹת,



וְאֹזֶן לְחֵרְשִׁים / גְּלוּיָה לְשָׁמְעָם / וְרוֹאוֹת לְעִוְרִים / לְשׁוּרָם פְּקוּחוֹת



וְנֹפֶשׁ לְנֶפֶשׁ / מְבִינִים, וְקֶרֶב / כְּסִילִים – כְּמוֹ אֵשׁ / קְרָב וַאֲנָחוֹת,



וְחָלְקוּ מְאֹד מַ/חֲמָאוֹת וְרַכּוּ / כְשֶׁמֶן – וְהֵמָּה / לְחוֹמְדִים פְּתִיחוֹת.



נְתַתִּים כְּמִכְתַּב / אֱלֹהִים לְאַב כָּל / הֲמוֹנִים וְגוֹיִים / לְנָשְׂאָם כְּלוּחוֹת,



וְשַׂמְתִּים כְּכֶתֶר / עֲלֵי רֹאשׁ חֲסָדָיו, / יְרִימוֹ עֲלֵי כָל / סְגָנִים וּפַחוֹת,



וְעַל גַּרְגְּרוֹת מַ/עֲלָתוֹ – עֲנָק חֵן, / וְטוּרָיו מְלֵאִים / בְּאַבְנֵי שְׁבָחוֹת.



גְּבִיר יַהֲלֹךְ עַל / אֲדָמָה וְעַל רֹאשׁ / כְּסִיל נִתְכְּנוּ לוֹ כְּחֶפְצוֹ מְנוּחוֹת,



יְרוֹצֵץ כְּבָרָק / עֲשֹׂה תֹם וְצֶדֶק – / בְּעֵת מַהְלְכֵי כָל / גְּבִירִים טְפָחוֹת,



וּמִשְׂרָה בְּנוּיָה / בְּאַבְנֵי תְהִלּוֹת – / וּבִלְתָּהּ בְּתָפֵל / בְּנוּיוֹת וְטוּחוֹת.



וְטוֹבוֹת בְּיָדָיו, / וְדַלְתוֹת חֲסָדָיו / לְכָל גֵּר וְאֶזְרָח / יְמוֹתָיו פְּתוּחוֹת,



וּמַצִּיל לְמָטִים / לְהֶרֶג בְּהוֹנוֹ / וּמַשְׂגִּיב נְפָשׁוֹת / לְמָוֶת לְקוּחוֹת,



נְעִימוֹת לְשׁוֹאֵל / בְּמַתָּן יְמִינוֹ / וְלוֹ גִיל בְּפָנָיו / וְשֹׂבַע שְׂמָחוֹת.



קְחָה לָךְ תְּשׁוּרָה / בְּדֹלַח אֲמָרִים – / וּמַה טּוֹב מְנָחוֹת / כְּמוֹ זֹאת שְׁלוּחוֹת,



וְכַנּוֹת אֲהָבִים, / יְסוּפוּן אֲרָזִים / בְּגַן אֵל – וְהֵן רַ/עֲנָנוֹת וְלַחוֹת.



וְהֵא לָךְ גְּבֶרֶת / לְשִׁירוֹת יְלָדוּן / בְּנֵי שִׁיר – וְכֻלָּן / לְנֶגְדָּהּ שְׁפָחוֹת.



אֱמֶת, אַתְּ וְשֶׁמֶשׁ / תְּאוֹמִים בְּעוֹלָם, / וְלָכֶם תְּהִי זֹאת / לְבַדָּהּ לְאָחוֹת.
