


	
	הקיצתי תנומת רעיוני
	



הקיצתי תנומת רעיוני / ר' משה אבן-עזרא




הֱקִיצֹתִי תְּנוּמַת רַעֲיוֹנַי/ לְיַשֵּׁן תַּאֲוַת נַפְשִׁי וְעֵינָי



וְחֹלְפוֹת הַזְּמָן תַּרְִתִּי בְלִבִּי/ לְהַשְׁמִיעַ עֲתִדֹתָיו לְאָזְנָי



וּפִי דֵעִי יְחַוֵּנִי גְדֹלוֹת/ וְיַצִּיב מִפְלְאוֹת צוּרִי לְפָנָי



וְיַגֶּד־לִי צְפוּנֵי תַעֲלֻמוֹת/ עֲדֵי אֶחְשֹׁב בְּנֵי עֶלְיוֹן שְׁכֵנָי



וְאַחַז מַחֲזֵה שַׁדַּי בְּשִׂכְלִי/ וְאָבִין כִּי בְקִרְבִּי יֵשׁ אֲדֹנָי



אֲשֶׁר נִסְתָּר יְקָר עַצְמוֹ וְנִגְלָה/ בְמִפְעָלָיו לְעֵינֵי רַעֲיוֹנָי



בְּפִגְרִי הֶעֱלָה נֵר מִכְּבוֹדוֹ/ יְשִׂיחֵנִי שְׁבִילֵי תַחְכְּמֹנָי



וְהוּא הָאוֹר אֲשֶׁר יֵלֵךְ וְיֵאוֹר/ יְמֵי־נֹעַר וְיוֹסִיף עֵת־זְקֻנָי



וְאִלּוּ לֹא־יְהִי מִסּוֹד מְאוֹרוֹ/ אֲזַי יָסוּף סְפוֹת כֹּחִי וְשָׁנָי



אֲחַפֶּשׂ־בּוֹ חֲדַר חָכְמָה וְאַעַל/ בְּלִי סֻלָּם אֱלֵי גִנַּת עֲדָנָי



הֲכִי חֶלְדִּי כְּאַיִן שָׂם וְאֵלֵךְ/ בְּדֶרֶךְ כֹּל אְשֶׁר הָלְכוּ לְפָנָי



וְאֶסַּע בַּאֲשֶׁר נָסְעוּ אֲבוֹתַי/ וְאַחַן בַּאֲשֶׁר חָנוּ זְקֵנָי



וְעַל־כֹּל אֵל יְבִיאֵנִי בְמִשְׁפָּט/ וּמַעְשַׂי יִֽהְיוּ עֵדָי וְדָנָי



לְזֹאת אֶמְאַס שְׁכֹן תֵּבֵל לְבִלְתִּי/ תְסִיתֵנִי וְתַשִּׁיא בִי זְדוֹנָי



נְטַשְׁתִּיהָ בְּטֶרֶם תִּטְּשֵׁנִי/ וְתַחְלֹץ נַעֲלִי לָרֹק בְּפָנָי



וְלוּ שֶׁמֶשׁ תְּשַׁו נֵזֶר לְרֹאשִׁי/ וְסַהַר יִֽהְיֶה מִשַּׂהֲרֹנָי



וְאֶצְעָדָה תְהִי עָשׁ עַל־זְרֹעִי/ וּבָנֶיהָ רְבִידֵי צַוְּרֹנָי



אֲזַי לֹא־אֶחֱמֹד עֻזָּהּ וְאִלּוּ/ תְשַׁוֶּה מַעֲלוֹת גַּלְגַּל מְעוֹנָי



אֲבָל חֶפְצִי נְתֹן סִפִּי לְסִפֵּי/ מְתֵי מוּסָר וּבֵינֵימוֹ מְלוֹנָי



וְכַמָּה אֶשְׁאֲפָה לָסוּר אֲלֵיהֶם/ וְרַגְלַי יִמְשְׁכוּ חַבְלֵי עֲוֹנָי



בְּתוֹךְ עַם לֹא־יְדָעוּנִי וְכֵן לֹא/ אֲנִי מֵהֶם וְהֵם לֹא מֵהֲמוֹנָי



בְּעֵת אֹמַר נְשַׁקְתִּימוֹ לְשָׁלוֹם/ יְשִׂיחוּן כִּי־נְשַׁכְתִּימוֹ בְשִׁנָּי



תְּעוּדַת חֹֽלְפִים שַׂמְתִּי מְנָתִי/ וְסִפְרֵיהֶם צְרִי צִירֵי יְגוֹנָי



וְעִמָּהֶם אֲנִי מַמְתִּיק לְסוֹדִי/ בְּיַעַן הֵם סְגֻלַּת נֶאֱמָנָי



וְאִם אֶשְׂחֶה בְּיָם שִׂכְלָם אֲלַקֵּט/ לְהַעְדוֹת גַּרְגְּרֹת יָמִים פְּנִינָי



וּבָם יִשְׁתַּעְשְׁעוּ עֵינַי וְלִבִּי/ וְלָהֶם יַעֲנוּ שִׂפְתֵי רְנָנָי



מְאוֹר עֵינַי וְשִׁיר אָזְנַי וְצוּף פִּי/ וְאֶל־אַפִּי כְּרֵיחַ קְנְּמוֹנָי



וּבָם אֵתְעַל וְאֶתְהַלַּל וְאֶהְגֶּה/ יְמֵי חַיַּי וּבָם כֹּל מַעֲיָנָי.
