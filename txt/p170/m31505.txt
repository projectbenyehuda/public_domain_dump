


	
	לִקְחֵי שְׂפַת מָשׂוֹשׂ
	



אל אבו־אסחק [אברהם] בן מהאנר






לִקְחֵי שְׂפַת מָשׂוֹשׂ כְּמָטָר עָרְפוּ / הַיּוֹם וְכַטַּל מִדְבְּרֵי שִׁיר נָטְפוּ,

וַיֵּלְכוּ מֵימֵי שְׂמָחוֹת עַל פְּנֵי / אֶרֶץ וְרוּחוֹת גִּיל עֲלֵיהֶם רָחֲפוּ.

הֵן הַזְּמָן גוֹנֵן בְּאֵין לָשׁוֹן וְהַ/שָּׁעוֹת בְּלִי יָד כַּעֲלָמוֹת תָּפְפוּ.

אֹרוּ שְׁתֵּי עֵינָיו – וּמֵרֹאשׁ חָשְׁכוּ, / וַיִּצְהֲלוּ פָנָיו – וְקֶדֶם זָעֲפוּ.

אַרְצָה שְׁחָקִים יִזְּלוּ בֹּשֶׂם וּמוֹר / עֹכֵר עֲנָנִים עַל אֲדָמָה רָעֲפוּ,

אוֹ אַב הֲמוֹן גּוֹיִם דְּגָלָיו קָרְבוּ, / כִּי מֶרְקְחֵי גַן עַל אֲדָמָה נוֹפְפוּ.

שֶׁמֶשׁ אֲשֶׁר זָרַח וְצַחוּ כוֹכְבֵי / עֶלֶץ וּמַחְשַׁכֵּי יְגוֹנִים נֶחְשְׂפוּ,

יָרַד כְּאֶגְלֵי טַל בְּעֵת קָצִיר וְקֹר / מַיִם לְיוֹם חֹם עַל נְפָשִׁים עָיְפוּ.

יָצָא לְיֶשַׁע עַם אֱלֹהָיו בַּחֲזוֹת / יָמִים עֲלֵיהֶם פֶּה לְהָרַע אָכְפוּ,



הִבִּיט בְּעֵין חַנּוֹת עֲלֵיהֶם, כִּי עֲלֵי / בּוֹר הַשְּׁבִי [כִּמְ]עַט אֲסֵפוֹת אֻסְּפוּ.



עוֹרֵר יְקַר רוּחַ נְדִיבָה לַעֲזֹר / אוֹתָם בְּמוֹעֲצוֹת שְׁחָקִים רוֹפְפוּ.

עֵצוֹת יְלָדוּם רַעֲיוֹנִים מִכְּלִי / הָרֹה עֲלֵי כַנְפֵי כְרוּבִים עוֹפְפוּ,

בָּם נַחֲלֵי צוּפִים לְחַפִּים, אַךְ מְתֵי / עָוֶל הֲמוֹן מֵימָם בְּלִי פֶה גָּרְפוּ,

צֶדֶק אֲשֶׁר נִשְׁקָף וְהֵרִיחַ דְּמֵי / אָוֶן נְאוֹת מִשְׁפָּט אֲשֶׁר טִנְּפוּ,

חֶלְאַת בְּנֵי חֹנֶף אֲמִתּוֹ טִהֲרָה, / כִּי עָבְרוּ תוֹרוֹת וְחֻקִּים חָלְפוּ,

וַיְחַלְלוּ קֹדֶשׁ וְתוֹרָה חָמְסוּ / מֵאָז וְתַחְתֵּיהֶם אֲרָצוֹת חָנְפוּ,

סָפוּ בְשֵׁבֶט פִּיו וְכָלוּ כַּעֲשַׁן / כִּבְשָׁן וּמֵרוּחַ שְׂפָתָיו נִשְׂרְפוּ.

מֵעֹז גְּאוֹנוֹ עָרְצוּ עַזִּים וְכָל / גֵּאִים חֲפוּיֵי רֹאשׁ לְפָנָיו נִדְחֲפוּ.

הֵרִים יְדֵי חֶסֶד אֲשֶׁר בָּם מִשְׁבְּרֵי / מָוֶת וְחַיִּים עַל בְּנֵי אִישׁ אָפְפוּ,

עַיִן תְּשַׂבֵּר אֶל עֲנַן אוֹרָם וּמֵי / זָהָב בְּלִי מִשְׁאָל כְּמָטָר סָחֲפוּ.

גַּן אֵל בְּצֵל שִׂיחָיו אֲשֶׁר לֹא עָמְמוּ / מִקֹּר וְחֹם כָּל הַיְקוּם הִסְתּוֹפְפוּ,

שָׁמָּה יְגִיעֵי אוֹן וְכֹחַ נָפְשׁוּ, / כִּי בָם נְעִים רֵיחֵי בְשָׂמָיו נָשְׁפוּ.

שָׁמַט לְאוֹרִים מִגְדְּנֵי שֵׂכֶל וְתֹם / וּבְלִי מְחִיר פֵּרֵי מְזִמָּה קָטְפוּ.

מַעְיַן חֲסָדִים לֹא יְכַזֵּב, הוֹי, לְכוּ / אֵלָיו וְצִחֵה בִין בְּצָמָא עֻלְּפוּ,



וּלְשֻׁלְחֲנוֹת מ[וּסָר] עֲרוּכִים מַהֲרוּ / אָכֹל וְהוֹתֵר מִי בְרָעָב נֶעְטְפוּ.



וּכְסִיל לְתוֹעֶה בֶאֱשׁוּן כֶּסֶל, וְאוֹר / מוֹרֶה אֲשֶׁר לַאְפֵל פְּתַיּוּת נֶהְדְּפוּ,

רֶקַח מְלָכִים מֵאֲבַק רַגְלָיו וְכֵן / בִּשְׂרוֹךְ נְעָלָיו הַנְּשִׂיאִים צָנְפוּ.

שָׂרִים בְּמִלִּים עָצְרוּ נֶגְדּוֹ וכָל / רָאשִׁים לְפָנָיו רֹאשׁ כְּאַגְמוֹן כָּפְפוּ.

דּוֹבֵר נְגִידִים יוֹם לְשׁוֹנוֹת בַּעֲלֵי / דָבָר בְּאַדִּיר הַדֳּמִי יִנָּקְפוּ.

תָּמִיד יְרוֹצְצוּ כְבָרָק עֵת לְאַט / יֵלֵךְ וּמֵהַשִּׂיג אֲבָקוֹ יִיעֲפוּ.

מִפִּיו חֲכָמִים בִּקְּשׁוּ חָכְמָה וְכָל / עָרְמָה פְּתָאִים מֵאֳרָחָיו אָלְפוּ.

יַסְתִּיר בְּעַנְוָתוֹ חֲסָדָיו, אַךְ כְּמוֹ / שַׁחַר בְּעֵת יִפְרֹשׂ כְּנָפָיו נִשְׁקְפוּ.

נָשָׂא בְחֵיקוֹ מַקְהֲלוֹת צוּרוֹ כְּמוֹ / אוֹמֵן וְחַנּוֹתָיו עֲלֵיהֶם עָדְפוּ.

חוֹמָה עֲלֵיהֶם וְרוֹעֶה נֶאֱמָן / יָדָיו עֲלֵי חֶבֶל עֲדָרָיו חוֹפְפוּ,

קָצוּ בְמַרְבֵּץ טוֹב וְנָוֶה שַׁאֲנָן / שָׁלֵו וְעַל מֵימֵי מְנוּחוֹת חָרְפוּ.

יִשְׁכַּב וְאֵין לוֹ מַחֲרִיד, אָכֵן בְּעֹז / שִׂיאוֹ טְלָאָיו הַכְּפִירִים טָרְפוּ.

יַלְדֵי זְמָן, אַל תֵּצְאוּ לָרִיב וְאַל / תִּלָּחֲמוּ עִמּוֹ וְלֹא תִנָּגְפוּ,

כִּי כֵן זְכוּת הוֹרָיו לְמָגֵן לוֹ, וְהוֹן / שֶׁיַּעֲנִיק טוֹב מֵחֲרָבוֹת נִשְׁלְפוּ.

אִמְרוּ לְשָׂרִים לֶאֱנוֹשׁ בִּלְתּוֹ, הֲלֹא / תָעוּ נְתִיב יֹשֶׁר וְצֶדֶק סִלְּפוּ.

רוֹפְאֵי אֱלִיל הֵמָּה וְהוֹלְכִים אַחֲרֵי / הֶבֶל, וְלֹא יוֹעִיל יְמוֹתָם רָדְפוּ.

קַח נָא, מְשִׁיחַ אֵל, עֲנַק אֹמֶר אֲשֶׁר / עַל הוֹד פְּנִינִים יַהֲלֹמָיו יָסְפוּ,

כִּי פֵאֲרוּ שִׁמְךָ עֲלֵי כָל־שִׁיר וְכָל / טוּרָיו בְּתַרְשִׁישֵׁי כְבוֹדָךְ עֻלְּפוּ,

וּשְׁמַע נְאֻם נֶטַע אֲהָבֶיךָ אֲשֶׁר / עָלָיו בְּמֵי חַסְדְּךָ וְצִצָּיו חָלְפוּ.

פֶּלֶג חֲסָדֶיךָ מְאֹד גָּבַר עֲלֵי / נַפְשׁוֹ וּמֵימָיו אַט מְכוֹנָיו שָׁטְפוּ.



[יִרְ]אֶה בְטַלֶּיךָ כְּעֵין שֶׁמֶשׁ הֲכִי / שֶׁבַע בְּמַצְרֵף הַיְדִידוּת נִצְרְפוּ,



מֵאָז פְּנֵי הָאֵל יְחַלֶּה לַחֲזוֹת / הוֹדָךְ וְלוֹ נֶפֶשׁ וְעַיִן דָּלְפוּ,

עַד כִּי עֲלֵי אַף הַזְּמָן הֵפִיק זְמַם / לִבּוֹ וְעֵינָיו מַאֲוַיָּיו שָׁזְפוּ.

בֹּשׁוּ בְנֵי יָמִים וְרַכּוֹת דִּבְּרוּ / עִמּוֹ, וְקֶדֶם בּוֹ בְּלִי חֵטְא דָּחֲפוּ.

תָּרוּם, נְשִׂיא הָאֵל, בְּעוֹד אִישִׁים עֲלֵי / רוּחוֹת נְדִיבוּתָךְ כְּתַנִּים יִשְׁאֲפוּ,

נָכוֹן יְהִי כִסְאָךְ בְּעוֹדָם כְּמוֹ / עָגוּר חֲסָדֶיךָ וְטוּבָךְ צִפְצְפוּ.
