


	
	הֲיַעְמֹד לְבָבִי וְנַפְשִׁי
	



על מות בת הנגיד אברהם






הֲיַעְמֹד לְבָבִי / וְנַפְשִׁי מְחוּצָה / וְתֶחְזַק יְמִינִי / וְהִנֵַּה רְצוּצָה,



וְאֶרְגַּע בְּנוּדִי / וְיָגוֹן מְנִידִי / וְתֶעְרַב שְׁנָתִי / וְתוּגָה מְקִיצָה,



וְאַסְתִּיר יְגוֹנַי / וּמוֹפְתָם בְּעֵינַי / וְאַשְׁקִיט שְׁאוֹנַי / וְדִמְעָה מְאִיצָה,



לְכַבּוֹת שְׁבִיבִי / וְלַהַב בְּקִרְבִּי / וּמִבֵּית לְלִבִּי / מְלַהֵט וְחוּצָה,



וְיִשְׂגּוּ מְצָרַי / כְּאָחוּ בְמַיִם / וְיִגְאוּ מְצוּקַי / כְּגֹמֶא בְּבִצָּה,



אֲקַוֶּה הֲנָחָה / וְתָחוּשׁ אֲנָחָה / וְלִהְיוֹת רְוָחָה / וְהִנֵּה נְאָצָה.



וְאֶשְׂנָא חֲיוֹתִי / וְאֶשְׁאַף לְמוֹתִי / וּמַה־לִּי בְחַיִּים / אֲשֶׁר הֵם לְשִׁמְצָה.



יְמוֹתַי בְּתֵבֵל / מְרוֹרִים וְאֵבֶל / וְחֵפֶץ לְהֶבֶל / וְאַוָּה חֲרוּצָה.



הֲתִשְׂנָא אֲדָמָה / מְתֵי בִין וְעָרְמָה / וְתָקוּץ בְּחָכְמָה / כְּשֶׁמֶשׁ מְצִיצָה.



בְּשִׂימָהּ שְׁחוֹתִי / כְּפִי גַאֲוָתִי / וְהוֹד מַעֲלָתִי / בְּלִי חֵטְא נְתוּצָה.



וְיוֹם יוֹם תְּגַבֵּר / הֲמוֹן שֹׁד וָשֶׁבֶר / וְצוּקִים תְּחַבֵּר / וְגִילִים מְפִיצָה.



וְיָמִים כְּקָמִים / מְצִיקִים וְקָמִים / לְהַגִּישׁ עֲצֻמִּים / ומלה חמוצה(?)



וְעִם בַּת מְלָכִים / וְזֶרַע נְסִיכִים / יְגָרוּ מְדָנִים / וְרִיבוֹת וּמַצָּה.



לְהַשְׁמִיד שְׁאֵרִית / בְּנֵיהֶם וְנַצֵּל / עֲדֵיהֶם וְשׂוּמָם / כְּחוֹמָה פְרוּצָה.



וְשָׂרַת גְּבִירִים / וְצָרַת מְאוֹרִים / לְאֹפֶל קְבָרִים / שְׁלוּחָה וְאָצָה.



וְאִשָּׁה חֲכָמָה / וְתַמָּה אֲיֻמָּה / וּבָרָה כְּחַמָּה / וְיָפָה כְּתִרְצָה.



וְקוֹמָה כְּתָמָר / וּפָנִים כְּסַהַר / וְלֶחִי בְּמֵי חֵן / וּבֹשֶׁת רְחוּצָה.



וּבֵינוֹת עֲלָמוֹת / כְּשֶׁמֶשׁ מְרוֹמוֹת / בְּיָדָהּ מְקוֹרוֹת / לְנֹעַם וּמוֹצָא,



וְגֶפֶן עֲנֵפָה / וּפִרְיָהּ תְּרוּפָה / וּפֹארָה עֲלֵי גַב / בְּנוֹת עָשׁ וְנִצָּה.



פְּנִימָה כְּבוּדָה / בְּנֶפֶשׁ שְׁפָלָה / וּמִמּוֹר וְקִדָּה / לְבַד הִיא קְרוּצָה.



מְבִישָׁה לְבָנוֹת / וְשָׂרָה לְבָנוֹת / לְשַׁמֵּר אֱמוּנוֹת / וְהָסֵר מְרוּצָה.



לְשֹׁד הַכְּבוּדוֹת / חֲלִיצוֹת וּמִדּוֹת / מְסִירוֹת וְקוֹרְעוֹת / כְּתֹנֶת חֲמוּצָה.



וּמַפְרִיד אֱמוּנִים / עֲלֵי מֵי אֱמוּנִים / וְחֶבְרַת עֳפָרִים / כְּרֶגַע נְפוֹצָה.



וְחָל בֶּן נְדִיבִים / לְעָפְרַת אֳהָבִים / וְנַפְשׁוֹ בְחַיִּים / מְאוּסָה וְקָצָה,



מְצַעֵק בְּתוּגָה / לְצִיצַת עֲרוּגָה / אֲשֶׁר פָּרְחָה בֵּין / צְלָעָיו וְצָצָה.



כְּעֹפֶר בְּצִיָּה / עֲלֵי נֹד צְבִיָּה / אֲשֶׁר בֵּין זְרָעָיו / בְּפֶתַע חֲלוּצָה.



מְרִי פִי וְשִׂיחִי / אֶשְׁלַח לְאָחִי / וּמֶרֶד בְּרוּחִי / בְּשִׁירָה נְחוּצָה



לְאָבִי וְאַב כָּל / בְּנֵי בִין וְהַשְׂכֵּל, / וְאִם לֹא יְכִילוּן / לְפָנָיו מְרוּצָה.



וְאֵיכָה אֲנַחֵם / אֲרִי עָז וְלֹחֵם / בְּתוּגוֹת וְיַשְׁחֵם / בְּנֶפֶשׁ חֲפֵצָה.



הֲלָרִיק מְאֹד בָּן / תְּעוּדַת חֲכָמִים / וְיָדַע מְשָׁלִים / וְהֵבִין מְלִיצָה,



וּמִבּוֹר תְּבוּנָיו / וְהוֹד רַעֲיוֹנָיו / בְּנֵי הַזְּמָן שָׁ/אֲבוּ בִין וְעֵצָה.



וְנִמְהָר יְבוֹנֵן / וְיֹאמַר לְאוֹנֵן / בְּעַד יָד נְטוּיָה / וְעֵצָה יְעוּצָה,



הֲלֹא חַי לְחֶרֶץ / וְנוֹצָר לְקֶרֶץ / וְתֵבֵל לְכָלָה / וְגַם נֶחֱרָצָה.



שְׁקֹט, כִּי בְכִי לֹא / יְשׁוֹבֵב אֲבֵדָה / וְלֹא יַחֲזִיק לֵב / וְנֶפֶשׁ רְעוּצָה.



וְאֵיךְ נָד לְמִי כוֹס / פְּרִידָה בְיָדוֹ / זְמָן שָׂם וְשָׁתָה / שְׁמָרָיו וּמָצָה,



וְיֵאוֹת הֱיוֹת לוֹ / כְּנַף נֵץ וְיָעוּף / רְאוֹתְךָ כְּבַעַל / אֲבָרִים וְנוֹצָה.



אֲקַוֶּה לְאֵלִי / לְקַבֵּץ נְדוּדִים / וְאֶבְרַת פְּרִידָה / לְשׂוּמָהּ קְצוּצָה,



וְאוּלַי תְּהִי לַ/נְּפִילָה תְּקוּמָה / וְיָשִׂים לְנוּמַת / שְׂשׂוֹנִים יְקִיצָה.
