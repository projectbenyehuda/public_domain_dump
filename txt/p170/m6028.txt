


	
	עַל אַף מְשַׂנְאֵינוּ וּמַקְנִיאֵינוּ
	




עַל אַף מְשַׂנְאֵינוּ וּמַקְנִיאֵינוּ / אֶל עָל בְּכָל עֵת יַעֲלֶה שִׂיאֵנוּ



הוֹרוּ בְחֵץ לָשׁוֹן סְגוֹר לִבּוֹת אֲבָל / הוֹרוּם יְדֵי הָאֵל לְהַחְטִיאֵנוּ



חַתּוּ וּבשׁוּ מֵחֲתֻלִים דִבְּרוּ / לַכְאוֹת נְפָשֵׁינו וְלַהְנִיאֵנוּ



אָרְבָּם בְּקִרְבָּם נָתְנוּ וַיֵּלְכוּ / רָכִיל בְּלֵב וָלֵב לְהַשִּׁיאֵנוּ



5    קָשׁוֹת בְּמַצְפֻּנָם וּפִימוֹ יַעֲנֶה / רַכּוֹת וְהֵם אִנּוּ לְהַפְלִיאֵנוּ



כָּאֵשׁ אֲשֶׁר תִּשָּׂא קְנֵה בֹשֶׂם / תִּשָּׂא עֲתַד רֵיחֹו וְתַמְצִיאֵנוּ



וַתֶּחֱזַק רוּח וְאֶל אַף תַּעֲבִיר / רֵיח נְרָדֵינו וְדוּדָאֵינוּ



מַה לַּחֲצִיר גַּגּוֹת וְאַרְזֵי אֵל וְאֵיךְ / אֶל עֵץ הֲדַסִּים עָרְכוּ דִשְׁאֵנוּ



אוֹ מַה לְּשֻׁעָלִים קְטָנִים אַחֲרֵי / גוֹרֵי אֲרָיֵינוּ וְלִלְבָאֵינוּ



10   אֵיךְ תִּתְמְהוּ מֵהֶם וְקֶדֶם דִּבְּרוּ / עָתָק בְּעֹשֵׂינוּ וּבִנְבִיאֵנוּ



נוֹדֶה לְצוּר הֵפֵר עֲצַת בַּדִּים וְאִם / קָצְרוּ לְשׁוֹנֵנוּ וּמִבְטָאֵינוּ



הִטָּה לְפָנֵינוּ נְהַר שָׁלוֹם בְּעֵת / בֹּצֶר וְאֵין מַיִם לְהַגְמִיאֵנוּ



וַתַּעֲלֶה שֶׁמֶשׁ עֲלֵי חוּג הַיְקָר / לִקְפֹּץ יְדֵי אֹפֶל אֲשֶׁר יַפְאֵנוּ



הָאִישׁ אֲשֶׁר יַשֶּׁה עֲמַל יָמִים וְרָב- / רַעַם גְּדוּדֵיהֶם לְהַצְבִּיאֵנוּ



15   תָּמִיד יְרַחֵף עַל מְתֵי הַשְׂכֵּל וְגַם / יִפְרשׁ כְּנַף חַסְדּוֹ וְיִשָּׂאֵנו



יִשָּׂק לְמִלְחֶמֶת זְמָן יִשָּׂק עֲלֵי / פִיהוּ לְיָרְאֵנו וְלֹא יִרְאֵנוּ



כִּי יוֹם אֲשֶׁר יִרְצֶה לְהַדְפֵנוּ אֲזַי / יִרְצֶה בְּצֵל יָדוֹ לְהַחְבִּיאֵנוּ



שָׁווּ בְעֵינֵינוּ תְּנוּאוֹת הַזְּמָן / יוֹם יֶאֱהָבֵנו וְיִשְׂנָאֵנוּ



מִכְלַל יְמוֹתָיו לַהֲנִיעֵנו וְהוּא / חָשׁ עַל שְׁמָרֵינוּ לְהַקְפִּיאֵנוּ



20   זֶה יוֹם אֲשֶׁר עָשָׂה אֱלֹקִים לִהְיוֹת / זֵכֶר לְבָנֵינוּ וְצֶאְצָאֵינוּ



קִרְאוּ מְתֵי אַהַב וְאַל יִתְמַהְמְהוּ / לָבוֹא וְיַחַד נִסְבְּאָה סָבְאֵנוּ



סֹבֶא כְּמוֹ לַהַב בְּכוֹס מַיִם וְלוֹ / נֹגַהּ בְּמַרְאָיו נֶחֱזֶה מַרְאֵנוּ



לַגַּן אֲשֶׁר נָשָׂא פְּרִי חֶסֶד אֱמֹר / מֵרֹאשׁ תְּבוּאָתוֹ לְהַבְרִיאֵנוּ



אוֹצָר וּמַעְיָן הֵם שְׁתֵי יָדָיו לְעֵת / צוֹק רַעֲבוֹנֵנוּ וְלִצְמָאֵנוּ



25   אָרְכוּ יְמֵי פְרוּד שְׁלֹמֹה עַד אֲשֶׁר / קָצְרוּ כְאֵבֵינוּ וּמַחְלֻאֵינוּ



הַגּוּף יְכַלְכֵּל מַחֲלֵהוּ אַךְ חֳלִי / נֶפֶשׁ כְּנֵטֶל חוֹל נְשׂא יַלְאֵנוּ



אִם פַּעֲמֵי רִכְבּוֹ כְּהַיּוֹם אֶחֱרוּ / יִשְׁלַח דְּבַר פִּיהוּ וְיִרְפָּאֵנוּ



בּוֹ חֻדְּשׁוּ לָנוּ יְמֵי עֶדְנָה וְכֵן / כִּימֵי בְחֻרֵינוּ יְמֵי דָבְאֵנוּ



בִּשְׁמוֹ נְשַׁבֵּר מַלְתְּעוֹת יָמִים וּבוֹ / נָעֹז בְּצֵאתֵנוּ וְיוֹם בֹּאֵנוּ



30   נִקְרָא פְּנֵי הָאֵל בְּעַד חַיָּיו וְהוּא / יִשְׁמַע וְיַעְנֵנוּ בְּיוֹם קָרְאֵנוּ.
