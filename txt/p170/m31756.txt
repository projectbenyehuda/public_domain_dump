


	
	אַט עַל אֱנוֹשׁ
	



על מות רפיא בן בזאז


כְּתָב נֹחַם שְׁלָחוֹ אֶל אֲהוּבָיו / אֲשֶׁר יִרְאֵם בְּעֵין לִבּוֹ סְבִיבָיו,

בְּנוֹ הָרַב וְיַלְדֵי אָח וְאָחוֹת / וְתַלְמִידָיו וְאֶל יֶתֶר קְרוֹבָיו.



אַט עַל אֱנוֹשׁ חָל מְגוֹרוֹ / פֶּתַע וְצַר בּוֹ מְגוּרוֹ,

אֵין לוֹ רְפוּאַת תְּעָלָה, / אֵין דָּן לְמָזוֹר מְזוֹרוֹ,

חָלְיוֹ מְאֹד רַע וְנֶאְמָן, / אָכֵן צֳרִי אֵין לְצִירוֹ,

בֵּינוֹת יְעֵנִים יְאַוֶּה / לָשׂוּם יְמוֹתָיו מְדוֹרוֹ.

אָסַר זְמָן רָע לְהַכְאִיב / לִבּוֹ וְקָם עַל אֱסָרוֹ,

אוֹתוֹ בְּהַנְחִיל יְגוֹנִים / הִכִּיר וְשָׂמוֹ בְּכוֹרוֹ.

אָמְרוּ ״אֱלֹהִים עֲזָבוּ״ / רֵעָיו ״וְצוּרוֹ מְכָרוֹ״,

וַיִּבְגְּדוּ בוֹ, וְשֶׁפַע / יַמִּים בְּרֹאשׁוֹ עֲזָרוֹ,

קִרְבּוֹ כְתֹפֶת, וְנַחַל / מֵימֵי דְמָעָיו גְּבָרוֹ,

מוֹצָא אֲשֶׁר לֹא יְכַזֵּב, / כִּי מִדְּמֵי לֵב מְקוֹרוֹ,

[א]וֹ מֵאֲגָמִים בְּנוֹף יוֹם / נֶהְפַּךְ לְדָם כָּל יְאֹרוֹ,

דֶּמַע יְפוֹצֵץ סְלָעִים, / יִשְׁחַק אֲבָנִים בְּאוֹרוֹ.

עָמְקוּ מְאֹד רַעֲיוֹנַי / בִזְמָן וְשִׂכְלִי חֲקָרוֹ,

מִיּוֹם מְעַט מֵחֲזוֹת אֶת / הָפְכּוֹ כְּרֶגַע יְעִירוֹ,

בּוֹגֵד בְּבַעַל בְּרִיתוֹ / שׁוֹדֵד לְהוֹן אִם צְבָרוֹ,

אִם יַעֲמֹד אִישׁ עֲלֵי קַו / יֹשֶׁר לְהַמְעִיד אֲשֻׁרוֹ,

יַחְבִּיר לְהַפְרִיד וְעֹשֶׁר / יִתֵּן בְּרֵישׁ לַהֲמִירוֹ,

יַרְנִין בְּנֶשֶׁף וְעֶרֶב / יַחְלִיף בְּמִסְפֵּד זְמִירוֹ,

כַּסְפּוֹ כְרֶשֶׁת לְמַעְגָּל / יִפְרֹשׂ וּמִכְמָר סְגוֹרוֹ.

יִכְסַל אֱנוֹשׁ שָׂם בְּהוֹנוֹ / כִּסְלוֹ וּבָטַח בְּעָשְׁרוֹ,

לָרִיק יְגִיעוֹ וְכֹחוֹ / יָלוּד לְעָמָל וְהֹרוֹ.

מִי יַאֲמִין בַּזְּמָן אוֹ / לִבּוֹ יְפַתֶּה בְשִׁקְרוֹ

יִזְכֹּר תְּמוּתַת רְפָיָא / הָרַב וְיוֹדֶה לְצוּרוֹ,

יוֹדֶה וְיֹאמַר: הֲכִי כָל / נוֹצָר לְכָלָה יְצָרוֹ,

אֵיךְ הַזְּמָן קָם וְנִצֵּל / עֶדְיוֹ וְחִלֵּל [לְ]נִזְרוֹ!

אֵיכָה לְהַר הָאֱלֹהִים / יָכֹל וְאֵיכָה עֲקָרוֹ!

קָם לִי עֲלֵי אִישׁ בְּרוּחַ / פִיו מִהְיוֹתוֹ גְעָרוֹ.

עוֹלָם לְמוֹתוֹ כְגֶבֶר / נִדְהָם וְיַיִן עֲבָרוֹ,

וַיַּעֲמֹד חוּג שְׁחָקִים / מֵרוּץ וּפַחַד עֲצָרוֹ,

אָסְפוּ כְסִילִים לְמוֹתוֹ / נָגְהָם וְשֶׁמֶשׁ מְאוֹרוֹ,

מַסְוֵה אֲפֵלָה עֲלֵי אוֹר / סַהַר וְקַדְרוּת אֲפֵרוֹ,

וַיִּלְבְּשׁוּ שַׂק בְּנוֹת עָשׁ / וּכְסִיל חֲבָלִים חֲגוֹרוֹ,

וּלְיָם תְּהִלּוֹת מְאֹד זָע / הַיָּם וְשָׁב לַאֲחוֹרוֹ.

נָעוּ גְבָעוֹת וְסִינַי / חָרַד עֲלֵי מוֹת חֲבֵרוֹ,

כָּל הַחֲסָדִים יְתַנּוּ / כִּי נִבְרְאוּ בַעֲבוּרוֹ,

הֵיכָל פְּנִימִי יְיֵלִיל / כִּי סַף כְּבוֹדוֹ וְנֵרוֹ,

נֵר מַעֲרָבִי, וְיֵלְכוּ / שׁוֹכְנִים כְּקֵדָר לְאוֹרוֹ,

שׁוֹכֵן פְּאַת יָם וְצָפוןֹ / בִּשְׁמוֹ יְרַנֵּן וְזִכְרוֹ.

הֶאְמִיר אֱלֹהִים וְכֵן עַל / כָּל הַיְּקוּם הֶאֱמִירוֹ,

בָּחַר בְּדָתוֹ וּבָזֹאת / מִכָּל בְּנֵי תֹם בְּחָרוֹ.

חָכָם, וְלֹא קָם לְפָנָיו / חָכָם בְּתֻמּוֹ וְיָשְׁרוֹ,

כָּל חַף בְּתֵבֵל בְּצִדְקוֹ / יִצְנֹף וְיַכְתִּיר בְּכִתְרוֹ.

מָשְׁלָה עֲנָוָה בְּלִבּוֹ / כִּמְשֹׁל פְּעָלָיו בְּיִצְרוֹ.

עָצְרוּ חֲכָמִים בְּמִלִּין / נֶגְדּוֹ וְהוֹדוּ לְאָמְרוֹ,

שָׂמוּ יְדֵיהֶם לְפִיהֶם / יַחַד עֲדֵי בֹא דְבָרוֹ.

עוּשׁוּ בְכוּ נָא עֲלֵי אוֹת / עוֹלָם וּמוֹפֵת בְּדוֹרוֹ,

וּנְהוּ לְשׁוֹפֵט בְּצֶדֶק / בֵּין אִישׁ לְאָחִיוּ וְגֵרוֹ,

עוֹטֵה אֱמוּנָה וְחֶסֶד / כִּמְעִיל וְצֶדֶק אֲזוֹרוֹ,

אוֹצַר תְּעוּדָה, וְאָכְלוֹ / חִנָּם לְרָעֵב וְשִׁבְרוֹ.

מֵשִׁיב לְשׁוֹאֵל כְּחֵפֶץ / לִבּוֹ וְדֵי מַחֲסוֹרוֹ,

מַשְׁקֶה צְמֵאִים לְמֵי בִין / מֵאֵין דְּלִי מֵי בְאֵרוֹ,

גַּן הַתְּבוּנוֹת וּפִרְיוֹ / שִׂכְלוֹ וּמַדָּע עֲבוּרוֹ,

שָׁמַט לְכָל בָּא רְדוֹת אֶת / דִּבְשׁוֹ וְאָכוֹל לְיַעְרוֹ,

וּבְלִי מְחִיר יֶאֱרֶה אֶת / בֹּשֶׂם תְּבוּנָיו וּמֹרוֹ.

רוֹעֶה מְנַהֵל לְצֹאנוֹ / עַל מֵי מְנוּחוֹת לְדָבְרוֹ

לֹא הֶחֱרִידוֹ אֲרִי כָל / יָמָיו וְאִישׁ לֹא שְׂעָרוֹ.

נָטַר לְכֶרֶם אֱלֹהָיו / וּכְעֵין יְמִינוֹ נְצָרוֹ,

סִקֵּל נְתִיבָיו וְכִלָּה / קוֹצָיו וְאֶת־כָּל־שְׁמִירוֹ,

הֵן אַחֲרָיו הָיָה לְבָעֵר / חָרֹץ וּפָרֹץ גְּדֵרוֹ.

נוֹשְאֵי אֲרוֹנוֹ בְּכָתֵף / אַט בּוֹ, וְסוֹבְלִים לְפִגְרוֹ,

כִּי אֶת־אֲרוֹן הָאֱלֹהִים / תִּשְׂאוּ וְלוּחוֹת דְּבִירוֹ,

לֹא שָׁעֲרוּ כִּי שְׂרָפִים / צָבְאוּ לְלַוּוֹת הֲדָרוֹ,

נַפְשׁוֹ לְהֵיכַל קְדוֹשׁוֹ / נָשְׂאוּ וְאֶל כֵּס יְקָרוֹ.

יָדוֹ תְּחוֹפֵף עֲלֵי גוּף / טָהוֹר וְיִרְגַּע בְּשָׂרוֹ,

כִּי אֵין יְכֹלֶת בְּעָפָר / בַּלּוֹת יְצוּרָיו וְצוּרוֹ.

בִּלְשׁוֹן כְתָבִי יְלָדָיו / אֶקְרָא וְאֶל כָּל שְׁאֵרוֹ:

הִנָּחֲמוּ, כִּי פְאֵר כָּל / מַשְׂכִּיל וְדֵעוֹ בְשָׂבְרוֹ

חַיֵּי בְנֵי אִישׁ בְּעוֹלָם / כַּחְלוֹם וּמָוֶת כְּשְִׁרוֹ,

יִנְהַג יְקוּמִים עֲלֵי פִי / שַׁחַת כְּרוֹעֶה לְעֶדְרוֹ,

מָוֶת כְּמַגָּל, וְנַחְנוּ / שָׂדֶה וּבָשַׁל קְצִירוֹ.

אִלּוּ זְמָן רָע יְבַקֵּשׁ / כֹּפֶר וּפִדְיוֹן גְּבִירוֹ

שַׁתְנוּ לְבָבוֹת וּבָבוֹת / תַּחְתָּיו שְׂמֵחִים בְּכָפְרוֹ.

אַחְלַי יְדִידַי, וְיִתֶּן / לִי עוֹף כְּנָפוֹ וְאֶבְרוֹ

לָעוּף לְאֶרֶץ אֲשֶׁר בָּהּ / נֶחְצַב מְכוֹנוֹ וְקִבְרוֹ,

קֶבֶר יְנוֹצֵץ לְתוֹעֶה / דֶרֶךְ בְּאֹפֶל לְאוֹרוֹ,

לִקְרַאת מְשַׁחֵר תְּשַׁלַּח / רֵיחַ בְּשָׂמוֹ וְזִכְרוֹ,

רֵיחַ יְחַיֶּה נְשָׁמוֹת, / כִּי מָר־דְּרוֹר הוּא עֲפָרוֹ.

עִמְדוּ עֲלֵי עָמְדְּכֶם כָּל / רֵעָיו לְהָשִׂיר צְעִירוֹ,

יָקוּם לְמַלֹּאת מְקוֹמוֹ / מַהֵר וְלַחְבֹּשׁ פְּאֵרוֹ,

יִפְרֶה, לְמַעַן כְּשֹׁרֶשׁ / עֵץ גַּן אֱלֹהִים אֲמִירוֹ,

יִשְׁאַג בְּדָת אֵל, וְכַאְרִי / חָכְמָה יְהִי קוֹל כְּפִירוֹ.

יִגְאֶה וְעַד יָם תְּבוּנָה / לָעַד יְשַׁלַּח קְצִירוֹ.

עַד יֹאמְרוּ מוֹשְׁלִים, כִּי / בֶן־הָאֱמוּנָה כְּהוֹרוֹ,

הוּא מֵאֲשֶׁר רָם וְנִכְבָּד / תָּאַב וְנִכְסַף לְצוּרוֹ,

יִשְׁכֹּן לְפָנָיו בְּחֵלֶק / אַהְרֹן וּמֹשֶׁה בְחִירוֹ,

קִבְרוֹ יְרַוֶּה בְּכָל עֵת / מַלְקוֹשׁ מְרוֹמוֹ וְיוֹרוֹ,

יֵלְכוּ לְפָנָיו פְּעָלָיו / לָאוֹר וְאִתּוֹ שְׂכָרוֹ.
