


	
	תֵּבֵל תְּנִי לִבֵּךְ
	




תֵּבֵל, תְּנִי לִבֵּךְ וְאֶת עַינַיִךְ / פִּקְחִי וְהַטִּי לַאֲזִין אָזְנָיִךְ,

שִׁמְעִי אֱמֶת מִפִּי אֲשֶׁר רָאָה בְעֵין / לִבּוֹ סְתָרַיִךְ וּמַצְפּוּנַיִךְ,

אִזֵּן וְחִקֵּר רֹב כְּשָׁפַיִךְ – הֲכִי / שָׁוְא דַּבְּרוֹתַיִךְ וְחֶזְיוֹנָיִךְ!

כַּמָּה תְפַתִּי הַפְּתָאִים, תַּחֲלִי/קִי מַאֲמָרַיִךְ וְעִנְיָנָיִךְ?

רַכּוּ דְבָרַיִךְ וּמִצּוּף מָתְקוּ – אָכֵן פְּתִיחוֹת הֵם לְעֵת פָּנָיִךְ,

אוֹתָם תְּסִיתִין בַּעֲדִי זָהָב אֲשֶׁר / תַּעְדִּי, וּבַפּוּךְ תִּקְרְעִי עֵינָיִךְ.

חִנָּם רְקָמוֹת תַּעֲטִי לָהֶם וְתִתְ/יַפִּי, וְלַשָּׁוְא תִּלְבְּשִׁי שָׁנָיִךְ,

הֵם הַנְּחָשִׁים הַשְּׂרָפִים תִּשְׁלְחִי / בָהֶם – וְאֵין לַחַשׁ לְצִפְעוֹנָיִךְ.

אִשָֻּׁה מְנָאֶפֶת קְרָאתִיךְ בַּחֲזוֹת / רֹב נַאֲפוּפַיִךְ וּבִזְנוּנָיִךְ,

תִּשְׂטִי לְבַד תַּחַת בְּעָלַיִךְ – וְאַךְ תִּגְאִי בְטֻמְאָתֵךְ וּבִזְדוֹנָיִךְ.

כָּל הַיְקוּם תַּשְׁוִי בְּבִגְדֵךְ בָּם, אֲבָל / תֵּרְעִי לְדוֹדַיִךְ כְּמוֹ מוֹנָיִךְ,

וּבְכוֹס חֲמַת מָוֶת בְּיָדֵךְ יִרְוְיוּן / יַחַד עֲשִׁירַיִךְ וְאֶבְיוֹנָיִךְ.

תַּקְשִׁי מְאֹד עָרְפֵּךְ, עֲדֵי לֹא תֶהְדְּרִי / פָנִים לְשָׂבַיִךְ וְלִמְבִינָיִךְ.

תִּתְחַתְּנִי בָהֶם – וְצוּקוֹת תִּשְׁלְחִי / שַׁי אֶל בְּנוֹתַיִךְ וְלַחְתָנָיִךְ,

וַתִּקְרְאִי שָׁלוֹם בְּעַגְבָתֵךְ לְחוֹשְׁ/קַיִךְ – וְאוֹתָם יִשְּׁכוּ שִׁנָּיִךְ,

וּתְכַחֲדִי כִּי אַתְּ הֲרַגְתִּים – וַהֲלֹא / דָמָם בְּבָתַּיִךְ וּבִמְעוֹנָיִךְ.

שַׂמְתְּ אֶת יְמוֹתֵיהֶם חֲלוֹם סָתוּם – וְאַךְ / מָוֶת שְׁבָרַיִךְ וּפִתְרוֹנָיִךְ.

אִם תַּעְרְכִי שֻׁלְחָן לְנֶגְדָּם – תִּפְרְשִׂי / לֶחֶם כְּזָבַיִךְ בְּשֻׁלְחָנָיִךְ,

אוֹ תִמְסְכִי מִיֵּין גְּפָנִים מִזְרְקֵי / כֶסֶף – חֲמַת תַּנִּין בְּתוֹךְ יֵינָיִךְ,

אוֹ תִפְרְשִׂי צִיצִים – נְחָשִׁים יֵשְׁבוּ / בֵּינוֹת הֲדַסַּיִךְ וְשׁוֹשַׁנָּיִךְ,

אוֹ תִרְבְּדִי לָהֶם חֲטוּבוֹת – תִּמְלְאִי / אוֹתָם בְּשִׂכַּיִךְ וּבִצְנִינָיִךְ,

אוֹ אִם תְּשׁוֹרֵרִי – כְּפֶשַׂע יִהְיֶה / בֵּין שִׁיר זְמִירַיִךְ וְקוֹל קִינָיִךְ,

טֶרֶם תְּכִינִין אַרְמְנוֹתָם – תַּחְצְבִי / תוֹכָם קְבָרַיִךְ וְצִיּוּנָיִךְ,

הַתֹּף עֲלֵי כַפֵּךְ והַכִּנּוֹר עֲלֵי / יָדֵךְ – וְהַחֶבֶל עֲלֵי מָתְנָיִךְ.

חֻקָּה כְּבוֹד כָּל אֵם, לְבַד מִמֵּךְ! הֲכִי / נַבֵּל כְּבוֹדֵךְ חֹק עֲלֵי בָנָיִךְ!

חֻקָּה לְהַרְאוֹת אֶת קְלוֹנֵךְ, וַחֲשֹׂף / שׁוּלֵי מְעִילַיִךְ עֲלֵי פָנָיִךְ!

הִתְעוֹרֲרִי, נַפְשִׁי, וְאַל תִּתְגּוֹרֲרִי / תוֹכָהּ וְשׁוּבִי נָא לְמִשְׁכָּנָיִךְ,

הִנָּזְרִי לָשִׂישׂ בְּשׁוּבֵךְ אֶל נְוֵה / צִדְקֵךְ – וְאַל תִּבְכִּי לְעִזְבוֹנָיִךְ,

שָׁם תִּחְיִי אַחַר תְּמוּתָתֵךְ, וְשָׁם / יָפִיץ מְאוֹר יוֹצְרֵךְ לְאִישׁוֹנָיִךְ,

מִמַּעְיְנֵי יֵשַׁע, יְחִידָה, תִּשְׁאֲבִי / אֶת מֵי שְׂשׂוֹנַיִךְ וּמַעְדַנָּיִךְ.

הִתְקַדְּשִׁי, בָּרָה, וְחִישׁ הִתְקַדְּשִׁי / אֶל כֵס אֱלֹהַיִךְ וְלָאְדֹנָיִךְ.
