


	
	ירעני זמן לא בן לרעי
	



ירעני זמן לא בן לרעי / ר' משה אבן-עזרא




יְרֹעֵנִי זְמָן לֹא־בָּן לְרֵעַי / וְיָקָם־בִּי בְלִי־פֶשַׁע כְּקָמִי



עֲדֵי בִידֵי מְגִנָּתִי יְמַגֵּן / יְחִידָתִי וְיִטְפֹּל עַל־אֲשָׁמִי



מְנָתִי מִשְּׂכִיּוֹת מַתְּנוֹתָיו / מְנָתִי מִנְּדִיב נוֹכֵל כְּנוּמִי



וְחַטָּאתִי בְּעֵינֵי כָל־אֲנָשָׁיו / עֲדוֹת עֶדְיֵי תְעוּדָתִי וְתֻמִּי



וְהֵם פִּקְחִים וְאַךְ לֹא־יֶחֱזוּ מִ־ /תְּהִלּוֹתַי לְבַד גָּלְמִי וְצַלְמִי



לְזֹאת אֶשְׁתֶּה דְמֵי לִבִּי וְאֹכַל / שְׁאֵר גּוּפִי מְקוֹם יֵינִי וְלַחְמִי



וְקֹדֵר אֶעֱבֹר בָּם כַּאֲבֶל־אֵם / וְכִמְעַט קָט וְאֶעְטֶה עַל־שְׂפָמִי



תְּנוּבָתִי תְבוּנָתִי וְדָתִי / וְעָרְמָתִי תְּכַסֶּה מַעֲרֻמִי



בְּאָזְנַי עֵט כְּפִי־עַיִט יְצַפְצֵף / וְדַיִּי בַדְּיוֹ מוֹרִי וּבָשְׂמִי



וְלוּ אֶסְכַּל אֲזַי אֶשְׂכַּל וְיִקְרָא / זְמַנִּי זֶה לְבַד יַעְצֹר בְּעַמִּי



אְנִי גֹלֶה בְּלֵב חֹלֶה וְאֹלֶה / לְמִי אֶצְרַח וְאָן אֶבְרַח וְאֶל־מִי



תְּבוֹאֵנִי אֲחוֹת חֹלִי בְּלַיְלָה / וְתֵבוֹשׁ מֵֽרְאוֹת פָּנַי בְּיוֹמִי



אֲיַחֵל לָהּ וְלֹא מֵאַהֲבָה כִּי / תְחַל פָּנַי וְלֹא לִדְרֹשׁ שְׁלוֹמִי



וְתָשׁוּר כָּל־בְּנוֹת יָמִים בְּחֵיקִי / וְהִיא תָתוּר לְהִתְגּוֹרֵר בְּעַצְמִי



וְתִמְצֶה מִבְּלִי־פֶה דַּם לְבָבִי / וְתֹאכַל מִבְּלִי־שֵׁן אֶת־לְחוּמִי



וְיוֹם הַחֹם לְקָרָתִי כְּשֶׁלֶג / וְלֵיל הַקֹּר כְּגַחֶלֶת לְחֻמִּי



וְאָזוּעַ לְמֵי זֵעָה וְאֵיכָה / אֱחִי בֵּינוֹת שְׁבִיב אִשִּׁי וְגִשְׁמִי



נְדוּד נוּמָה אֲחִי־רַעַד וְאֵימָה / שְׁתֻם עַיִן רְעוֹת כּוֹכְבֵי מְרוֹמִי



וְדַלֹּתִי וְאֵין מַכִּיר דְּמוּתִי / עֲלֵי עַרְשִׂי עֲדֵי יִשׁמַע נְאֻמִי



וְהַלַּיִל כְּמוֹ־רֶשֶׁת לְשַׁחַר / וּמִי־יִתֵּן וְיִלָּכֵד בְּחֶרְמִי



יְחִתּוּנִי בְּעֵת שִׁכְבִי חֲלֹמוֹת / וּפִתְרֹנָם יְבַעְתֵּנִי בְקוּמִי



וְכָל־רֹפֵא אֱלִל נַפְשִׁי יְנַחֵם / לְחַיּוֹתִי וְלֹא יֵדַע עֲלֻמִי



לְמַעַן רָחֲקוּ מֵהֶם חֲפָצַי / כְּרֹחַק צַד צְפוֹנִי מִדְּרוֹמִי



וְלֹא תָכִיל אֲדָמָה רַעֲיוֹנַי / וְאֵיךְ חֶדֶר לְבָבָם הַפְּנִימִי



וְנֶחְלֵיתִי וְלֹא־חָלוּ שְׂעִפַּי / וְנִהְיֵיתִי וְלֹא נִהְיָה זְמָמִי



וְלוּ אֶרְאֶה אֲחִי מוּסָר דְּבָרוֹ / יְהִי מִמַּחֲלָתִי מַחֲלִימִי



הֲשָׁכְחוּ אַחֲוָתִי כָל־מְתֵי־תֹם / וְאִם־תָּעוּ וְלֹא־יָדְעוּ מְקוֹמִי



וְאִם חַבְלֵי תְנוּמוּת לַתְּמוּנוֹת / אֱהִי פֹרֵשׂ רְאוֹת צַלְמָם בְּצַלְמִי



יְצַוּוּ עַל־יְצֻרֵיהֶם לְבִלְתִּי / יְבֹאוּנִי בְּלֵילִי בַחֲלוֹמִי.
