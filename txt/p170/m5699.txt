


	
	יַלְדֵי יָמִים חָשׁוּ לִצְבֹּא
	




יַלְדֵי יָמִים / חָשׁוּ לִצְבֹּא / עַל לֵב הִכְחִיד / יָגוֹן רֻבּוֹ



כִּמְעַט מִזְעָר / מֶנְהוּ נִשְׁאַר / שָׁלַף עָלָיו / פְּרוּד חַרְבּוֹ



אַךְ בּוֹ יָשֻׁב / יַהְפֹּךְ יָדוֹ / יָשׁוּט אָכֵן / אֵלָיו סֻבּוֹ



קָשַׁר עָלָיו / קֶשֶר אַמִּיץ / וַיָשֶׂם תּוֹךְ / קִרְבּוֹ אָרְבּוֹ



כָּל מִבְטֶחָיו / בִּדְמֵי שִׂיחָיו / וּדְמֵי עֵינָיו / רַק מִשְׂגַּבּוֹ



וּכְתָב דּוֹדוֹ / תָּמַךְ יָדוֹ / הָעֵת פִּידוֹ / הֵעִיר עָצְבּוֹ



מִכְתָּב דָּמָה / לִפְנֵי שַׁחַר / פָּרַשׁ עָלָיו / טוּרֵי עַרְבּוֹ



טוּרֵי שִׁירִים / יָקְרוּ מִפָּז / אֶחָד מֵהֶם / יִרְדֹּף רִבּוֹ



מֶשִׁי חִבֵּר / מֵהַדִּבֵּר / הַבִּין שִׁתְיוֹ / הַתֹּם עֶרְבּוֹ



הִנֵּה כָּל שִׁיר / יָקָר כְּגוּף / נֶגְדּוֹ אַךְ הוּא / רוּח קִרְבּוֹ



לוּלֵי נָתַן / הוֹדוֹ עָלָיו / הָיָה כִכְלִי / אֵין חֵפֶץ בּוֹ



אֵיךְ בֵּן נָעִים / וּצְעִיר שָׁנִים / יַעְמֹס הָרֵי / בִּין עַל גַּבּוֹ



אוֹ עוֹל יָמִים / יֶהְדֹּף רָמִים / וּבְעוֹד נֹעַר / הוּא וּבְאִבּוֹ



הֵן מִשֵּׂעִיר / זָרַח לָעִיר / אֹרֶךְ עוֹלָם / וּמְכוֹן רָחְבּוֹ



וּבְיָד רָמָה / מֵרֹאשׁ כִּימָה / נִשְׁקַף כִּי לֹא / מֵאֶשְׁנַבּוֹ



אִישׁ כָּמוֹנִי/ בֶּאְשׁוּן שִׂכְלוֹ / רָאָה אֹתוֹ / וּבְעֵין לִבּוֹ



עַל יָד שָׁמְעוֹ / הֵבִין רֵעוֹ / וַיִּרְאֶה אֵת / מַטְמוֹן חֻבּוֹ



עֹבֵר אֹרַח / בַּעְדוֹ יִשְׁאַל / אָז מַדּוּעַ / בְּשֵׁשׁ רִכְבּוֹ



נָא אִמְרו לוֹ / לָרוּץ אֶל גַּן / אַהַב וּרְעוּת / בָּשְׂמֵי עֶשְׂבּוֹ



אֶל גַּן נָתַן / נִרְדּוֹ רֵיחוֹ / רִבְעוֹ יָשִׁים / בּוֹ וּמְסִבּוֹ



וֶאְרוֹת מִגְדֵי / דֹדִים בָּשְׁלוּ / וֶאְכֹל חִנָּם / אֶת טוּב חֶלְבּוֹ



וּבְבֵית מִדּוֹת / לָגוּר תּוֹכוֹ / יוֹם כָּל בַּיִת / סֻגַּר מִבּוֹא.
