


	
	אֲמָרֵינוּ כַּיּוֹם לָעוּ
	




אֲמָרֵינוּ כַּיּוֹם לָעוּ /  חַוֹּת צֶדֶק לֹא יְסֻפָּר



כִּי אָפְפוּ עָלֵינוּ   / רָעוֹת עַד אֵין מִסְפָּר:


תה' מ, יג


נוֹעֲדוּ צָרֵינוּ יַחַד כָּאֲרִי יָדָיו וְרַגְלָיו



כִּי נוֹעֲצוּ   /   לֵב יַחְדָּו:


שם פג, ו


יֶהֱמוּ קְרָבַי   /  וְעַל חֲטָאַי אֶתְאוֹנָן



כִּי יִתְחַמֵּץ לְבָבִי /  וְכִלְיוֹתַי אֶשְׁתּוֹנָן:


שם עג, כא


מִפְּנֵי חַסְדּוֹ יָעִיר וְיִתְּנֵם לְמִרְמָס



כִּי מָלְאוּ מַחֲשַׁכֵּי אֶרֶץ נְאוֹת חָמָס:


שם עד, כ


שׁוּר עָנְיֵנוּ כִּי יָגַעְנוּ / נְשׂוֹא עֹל מוֹקְשֵׁנוּ



כִּי שָׁחָה    /  לֶעָפָר נַפְשֵׁנוּ:


שם מד, כו


הוֹפִיעָה כִּי פָרְשׂוּ / לְרַגְלֵי מַעְגָּלִי רֶשֶׁת



כִּי הִנֵּה הָֹרְשָׁעִים  / יִדְרְכוּן קֶשֶׁת


שם יא, ב


קָמִים הֱצִיקוּנִי /  בְּלִי תִכְלָה וְחֵקֶר



כִּי קָמוּ בִּי  /   עֵדֵי שֶׁקֶר:


שם כ


טָהוֹר, הַכְנִיעֵם  /  אֲשֶׁר בְּךָ חָסוּ



כִּי טָבְעוּ גוֹיִם  /  בְּשַׁחַת עָשׂוּ:


תה' ט, טז


נְחֵה עַם זוּ בְישֶׁר  / לְהַשִּׂיג מַאֲוַיָּו



כִּי נִחַם אֲדֹנָי  /  עַמּוֹ וַעֲנִיָּו:


ישע' מט, יג


בְּפִשְׁעִי אֲנִי נֶחְלָשׁ /  וְצָרִי מִתְגַּבֵּר



כִּי בָאוּ בָנִים  /   עַד מַשְׁבֵּר:


שם לז, ג


רַחֵם הָעִיר הַהֻלָּלָה / וְקַבֵּץ בָּנֶיהָ



כִּי רָצוּ עֲבָדֶיךָ  /  אֶת אֲבָנֶיהָ:


תה' קב, טו


יִוָּדְעוּ חֲסָדֶיךָ  /  לְעַם זוּ מִשְּחָקִים



כִּי יוֹדֵעַ אֲדֹנָי  /  דֶּרֶך צַדִּיקִים:


שם א, ו


עֻזְּךָ אֲנִי אֲשַנֵּן  /  בְּזֶמֶר לְשׁוֹנִי



כִּי עָשִׂיתָ   /   מִשְׁפָּטִי וְדִינִי:


שם ט, ה


קוֹל שְׂפַת חוֹרְפֵיךָ /  וְחֶרְפָּתָם שְׂרָפַתְנִי



כִּי קִנְאַת בֵּיתְךָ /  אֲכָלָתְנִי:


שם סט, י


בְּחַסְדְּךָ פְשָׁעַי /   הַפְאֵה לְכָל רוּחַ



כִּי בְךָ בָּטּוּחַ:


ישע' כו, ג
