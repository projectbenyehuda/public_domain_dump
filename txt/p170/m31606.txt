


	
	עֵינוֹת תְּהוֹם הֵמָּה
	



על מות ר׳ ברוך בן יצחק אבן אלבאליה


עֵינוֹת תְּהוֹם הֵמָּה וְלֹא עֵינַיִם / וּמְקוֹם יְאוֹרִים רַחֲבֵי יָדַיִם,

אוֹ נִבְקְעוּ נִבְלֵי דְמָעוֹת לַעֲרֹף, / אוּלַי יְכַבּוּ נַחֲלֵי מֵעָיִם.

מֵאֵין הֲפוּגוֹת יִגְּרוּ, כִּי לֹא לְיוֹם / אֶחָד יְהִי הַשֹּׁד וְלֹא לִשְׁנָיִם.

אֵבֶל וְלֹא יִכְלוּ יְמֵי בִכְיוֹ עֲדֵי / יִכְלוּ כְעָשָׁן כּוֹכְבֵי שָׁמָיִם.

רָעָה אֲשֶׁר חָלָה וְלֵיל יָגוֹן, וְלֹא / שַׁחַר יְקַו לִרְאוֹת וְעַפְעַפָּיִם.

הִתְרַגְּזוּ לָהּ מוֹסְדוֹת תֵּבֵל, וְחוּג / אֶרֶץ מְרוּצָתוֹ לְיַרְכָתָיִם.

כָּסְלוֹ כְסִילִים קַו הֲלִיכוֹתָם, כְּמוֹ / תָעוּ שְׁנֵים עָשָׂר וְשִׁבְעָתָיִם,

וַתַּעֲטֶה כִימָה מְעִיל זַעַף, וְעַל / אַפֵּי בְנוֹת עַיִשׁ צְעִיף עַרְבָּיִם.

וּפְנֵי שְׁחָקִים לָבְשׁוּ קַדְרוּת, וְשַׂק / שָׂמוּ כְסוּתָם, אוֹ לְבוּשׁ כִּירָיִם;

יֵאָנְקוּ עַל הַשְּׁמוּעָה כֹאֲבִים / דּוּמָם וְרָפוּ מִצְּעֹק יָדָיִם.

יִפְעוּ כְיוֹלֵדָה וְיִשְּׂמוּ, וְאַט / יֵלְכוּ בְּלֵב נָמֵס וּפִיק בִּרְכָּיִם.

צֶפֶת יְהוּדָה סָפְתָה, גַּם נָפְלָה / אֶרֶץ עֲטֶרֶת גַּאֲוַת אֶפְרָיִם.

אֶל כָּל־מְדִינָה כִּתְבוּ לִמְתֹפְפוֹת / עַל לֵב, וְלַסֹּפְדִים עֲלֵי שָׁדָיִם.

יִוָּעֲדוּ לִצְבֹּא בְּאֶרֶץ הַצְּבִי / קִינִים, וְעִם הַבַּת יְרוּשָׁלָיִם

תַּזִּיל עֲדַת עֵינָם דְּמֵי עֵינָם וְלֹא / תִתֵּן דֳּמִי עַל שַׁעֲרֵי שַׁעְרָיִם.

עִמָּם מְתֵי רָמָה יְרִימוּן קוֹל בְּכִי / וִינַצְּלוּ עֶדְיָם וּבַעְדִיתָיִם.

אָרַח מְשׂוֹשׂ מִזְרָח, וְגִיל הַמַּעֲרָב / עָרַב, וְתַחְתָּם יִלְלַת אֶגְלָיִם.

יִתְפָּרְקוּ נִזְמָם בְּאַפְרִיקֵי, וְעַל / כָּל רֹאשׁ יְהִי חֶבֶל וּבַחְלָצָיִם.

מִסְפֵּד בְּבַת בָּבֶל לְבַל יִבֶל, וְאַל / אַדְמַת אֱדוֹם תִּדֹּם וְלֹא מִצְרָיִם.

עַל אֵם סְפָרַד יִקְרְחוּ קָרְחָה, הֲכִי / מִיַּד יְיָ לָקְחָה כִפְלָיִם.

אָנוּ פְתָחֶיהָ לְמִגְדַּל עֹז אֲשֶׁר / נָפַל, וְלִבְרִיחַ וְלִדְלָתָיִם.

וּכְעִיר פְּרוּצָה נִשְׁאֲרָה לָהּ מִבְּלִי / חוֹמָה, וְהוּא הָיָה לְחוֹמוֹתָיִם.

אָרוּר יְהִי יוֹם הֶאֱבִיד בָּרוּךְ, אֲבָל / שֶׁמֶשׁ אֲשֶׁר בָּאָה בְּתוֹךְ צָהְרָיִם.

חֹשֶׁךְ וְצַלְמָוֶת יְבַעְתוּהוּ, וְאַל / מָחָר יְהִי לוֹ לִהְיוֹת יוֹמָיִם.

שְׂרַר לְהִתְנוֹסֵס בְּרֹאשׁ הַדּוֹר וְכָל / מִשְׂתָּרְרִים בִּלְתּוֹ כְּמוֹ נַעְלָיִם.

הוּא מֵאֱנוֹשׁ נִקְרָא בְּשֵׁם גִּבּוֹר בְּלִי / שֶׁלַח, וְיוֹם לָחֶם – כְּאִישׁ בֵּינָיִם.

מֵכִין כְּלֵי אֹמֶר לְכָל מֶמֶר, וְיוֹם / רִיבוֹת יְהִי מֵנִיס לְרִבּוֹתָיִם.

אֵימָה לְפָנָיו יַחְגְּרוּ עַד יַחְרְגוּ, / כִּי אֵיךְ יְרוּצוּן מִבְּלִי שׁוֹקָיִם?

אוֹ אֵיךְ בְּלִי שֵׁן יִהְיֶה טוֹרֵף אֲרִי, / וּרְאֵם הֲיִגַּח מִבְּלִי קַרְנָיִם?

הִמְשִׁיל גְּאוֹן דֵּעוֹ עֲלֵי חֶפְצוֹ, וְכֹּל / מֹשֵׁל בְּרוּחוֹ מַאֲרִיךְ אַפָּיִם.

שַׁח עִם מְתֵי אֶרֶץ, וְאִם עַל בָּמֳתֵי / שַׁחַק זְמָמָיו סָחֲבוּ שׁוּלָיִם.

נָשָׂא אֲרוֹן לִבּוֹ אֲרוֹן עֵדוּת, וְכָל / דָּתוֹ חֲרוּתָה לוֹ בְלוּחוֹתָיִם.

גָּאוּ בְעֵינֵיהֶם רְחוֹקִים, אַךְ בְּעֵת / עָמְדוּ לְפָנָיו כָּרְעוּ אַפָּיִם.

רָאוּ אֲשֶׁר לֹא שָׁמְעוּ, עַד חָמְדוּ / רָאשִׁים לְרַגְלָיו לִהְיוֹת רַגְלָיִם.

כָּבֵד כְּבוֹדוֹ בוֹ וְהֵם קַלּוּ כְקַשׁ / נִדָּף בְּעֵת עָלוּ בְכַף מֹאזְנָיִם.

שָׂחָה בְיָם הַשְׂכֵּל וְלָכַד מַעְבְּרוֹת / חָכְמָה, וְהֵם טָבְעוּ בְמֵי אַפְסָיִם.

כָּל מַחְשְׁבוֹת יָמִים בְּיַד יִצְרוֹ כְּמוֹ / חֹמֶר בְּיַד יוֹצֵר עֲלֵי אָבְנָיִם.

הֶאְרִיךְ לְשׁוֹן צֶדֶק, וּמֶתֶג הַדֳּמִי / שָׂם עַל שְׂפַת שֶׁקֶר וּבִלְחָיָיִם.

עֵטָיו בְּפִיהֶם רוֹשׁ לְפִי שֵׂטִים, וְעַל / חֵךְ הַיְּשָׁרִים צוּף וְלִשְׂפָתָיִם.

סוּת אַלְמְנוּתָה לָבְשָׁה תוֹרָה, וְשַׂק / לֹא פִתְּחָה לָהּ מֵעֲלֵי מָתְנָיִם,

הָלוֹךְ וְצֹעֶקֶת וְחֶבֶל תַּחֲגֹר / תַּחַת שְׁלַל צֶבַע וְרִקְמָתָיִם,

כִּי נֶאֱסַף אִישׁ הַמְּאַסֵּף מַחֲנוֹת / עֻזָּהּ בְּתַחְבֻּלּוֹת וּבַעְצָתָיִם.

מִמִּי לְהָשִׁיב הַנְּפָשִׁים תִּשְׁאֲלוּ / לֶחֶם, וּמַיִם חָדְלוּ רֵחָיִם,

וּבְמִי תְגַבֵּרוּ הֲמוֹן צָרִים, וְאֵין / אֶחָד יְרַדֵּף בַּקְרָב אַלְפָּיִם?

מַמְשִׁיל הֲדָרוֹ אֶל מְאוֹר שֶׁמֶשׁ – כְּאִישׁ / דִּמָּה בְּלִי אֶחָד סְפוֹר מָאתָיִם.

אוֹ מַעֲרִיךְ דֵּעוֹ לְדַעַת הָאֱנוֹשׁ – / אֵיפָה יְהִי חֹשֵׁב כְּמוֹ סָאתָיִם.

יָדוֹ אֲשֶׁר יָרוּץ כְּבָרָק בַּכְּתָב – / אֵיכָה לְשִׁפְלוּת סָר וְלַעְצַלְתָּיִם?

גַּם צוּר לְשׁוֹנוֹ הַמְּרַדֵּף לוֹחֲמִים – / אֵיךְ הַזְּמָן הִדְבִּיק לְמַלְקוֹחָיִם?

אֵיכָה אֲנָשִׁים סָבְלוּ סִינַי עֲלֵי / שֶׁכֶם בְּלִי כֹבֶד, וּבִכְתֵפָיִם?

צָרָה אֲדָמָה מִנְּשׂוֹא שִׂיאוֹ, וּמֵ/אַמּוֹת שְׁאוֹל נִקְרָא לְאַמָּתָיִם,

עַד פָּגְעוּ בוֹ מַלְאֲכֵי שָׁלוֹם וְאָז / נִקְרֵאת מְנוּחָתוֹ בְשֵׁם מַחְנָיִם.

הַקּוֹבְרִים אִם בַּעֲפָרוֹ הָרְקְחוּ / אוֹ מִבְּשָׂמִים מִלְאוּ חָפְנָיִם.

אִם לַאֲפֵלָה יָעֲטוּ פָנָיו, הֲלֹא / מֵאוֹר פְּעָלָיו יַעֲטוּ קַרְנָיִם.

תּוּאַר אֲדָמָה כִּי יְלָדֶיהָ תְּפַת / חִנָּם וְתִקָּחֵם בְּעַפְעַפָּיִם.

תַּשְׂגִּיא כְּחוֹל סִיגִים עֲלֵי גַבָּהּ, אֲבָל / תַּחְבִּיא בְּבִטְנָהּ טוּב זְהַב פָּרְוָיִם.

יֵלְכוּ אֲנָשֶׁיהָ בְּמוֹ אִשָּׁהּ וְלֹא / יִקְחוּ רְצָפֶיהָ בְּמֶלְקָחָיִם,

יִרְבּוּ עֲשֹׂה חַיִל וְלֹא לָהֶם, עֲדֵי / יִמָּלְטוּ מֶנְהוּ בְעוֹר שִׁנָּיִם.

אֵיךְ חַף יְהִי מֵהֶם, וְחוּט אַהַב בְּפִיו / יִטְוֶה, וּבַלֵּב יֶאֱרֹג כִּלְאָיִם.

מִיַּד שְׁאוֹל דִּמּוּ לְהִנָּצֵל, וְיַד / מָוֶת עֲלֵי צַוָּאר וּבַעְקֵבָיִם,

יַשְׁקֶה בְּכוֹס אֶחָד אֲשֶׁר יִישַׁן בְּבֵית / הַשֵּׁן וּבֵין תַּנּוּר וְלִשְׂפַתָּיִם.

אֵיךְ תֶּחֱזֶה תֵבֵל וְאֵיךְ תִּשְׁמַע – / וְהוּא הָיָה לְרֹאוֹת לָהּ וְלָאָזְנָיִם.

גָּדוֹל עֲווֹן הַדּוֹר, וְאִישָׁיו חִבְּלוּ / חֶבֶל בְּמַעְשֵׂיהֶם וּבִדְרָכָיִם.

צַר לִי עֲלֵי אָחִי וְנַפְשִׁי, וַעֲלֵי / דוֹדַי אֲשֶׁר לִי נֶחְשְׁבוּ עֵינָיִם.

סָפוּ יוֹם אֶחָד – לְזֹאת לֹא אֶחְרַד / כִּי יַעַקְבֵנִי הַזְּמָן פַּעְמָיִם.

הִתְפָּרְצוּ אַחֲרָיו עֲבָדִים סֻגְּרוּ / עַל בּוֹר חֲרוֹן אַפּוֹ וּבִנְחֻשְׁתָּיִם.

הֵמָּה אֲשֶׁר בִּרְאוֹת יְצוּרָיו בַּחֲלוֹם / רָהוּ, וּמִגּוֹר סָפְקוּ כַפָּיִם,

הַיּוֹם לְפָנָיו יֵלְכוּ רוֹמָה, אֲשֶׁר / אֶתְמוֹל שְׁחוֹחַ הֵם וְלַאְחוֹרָיִם.

אִם נִסְגְּרוּ דַלְתֵי אֲדָמָה בַּעֲדוֹ, / הֵן נִפְתְּחוּ לוֹ שַׁעֲרֵי שָׁמָיִם,

וַתַּעֲלֶה רוּחוֹ מְרוֹמִים, וַעֲנַן / כָּבוֹד יְרַו קִבְרוֹ וְחַשְׁרַת מָיִם.

יִרְגַּע, וְעָלָיו הַכְּרוּבִים סוֹכְכִים / יִהְיוּ בְאֶבְרַת חֵן וּבִכְנָפָיִם.
