


	
	[אֵשׁ קָדְחוּ אוּרָיו וְלֹא נֻפָּחוּ]
	




אֵשׁ קָדְחוּ אוּרָיו וְלֹא נֻפָּחוּ / תוֹךְ מִזְרְקֵי בָרָד וְהִתְלַקָּחוּ –

אֵיךְ עָבְרוּ דָתָם עֲדֵי הִתְגּוֹדֲדוּ, / אֵיךְ חָלְפוּ חֻקָּם עֲדֵי נִסְפָּחוּ?

חֶמֶר אֲשֶׁר חָמַר וְלֹא נָמַר – כְּעֵין / אֹדֶם, וּמֵחָלָב גְּבִיעָיו צַחוּ.

עַל גַּרְגְּרוֹת הַכּוֹס עֲנָק מִגַּרְגְּרֵי / שֹׁהַם, וְכִמְעַט תֶּחֱזֵם – יִמָּחוּ.

תִּבְעַר חֲמָתוֹ בוֹ וְאַפּוֹ יֶחֱרֶה – / וּבְרֹךְ לְשׁוֹן מַיִם לְאַט יִסָּחוּ,

יִשַּׁק – וְיִשַּׁךְ, יַחֲלִיא – גַּם יַחֲלִים, / וּבְתַהֲפוּכוֹתָיו אֱנוֹשׁ יִשְׂמָחוּ,

נָפְלוּ נְפִילֵיהֶם הֲרוּגִים מִיְּדֵי/ יֶלֶד אֲחִי עָנֵר – וְלֹא רֻצָּחוּ!

יַכֵּם – וְלֹא יֶחְלוּ, וְיֶהֶלְמֵם – וְלֹא / יֵדְעוּ, וְעֵת קוּם – בַּעֲדוֹ יִצְוָחוּ:

כִּי נֶאֱמָנִים הֵם פְּצָעִים מִיְּדֵי / אוֹהֵב, וּפִשְׁעֵי שׁוֹגֲגִים – יִסְלָחוּ.

יֵמַר – וְכָל מָרֵי נְפָשִׁים יֶעֲרַב / לָהֶם, וְרִישָׁם אוֹבְדִים יִשְׁכָּחוּ,

יַעְשִׁיר מְתֵי חֶסֶר בְּלִי מַתָּן, וּמֵ- / אֵין הוֹן דְּמֵי אַנְשֵׁי עֳנִי יוּדָחוּ,

יַרְהִיב בְּמַרְאֵהוּ אֲשֶׁר יִנָּזְרוּ, / עַד יִמְאֲסוּ נִזְרָם וְיִתְגַּלָּחוּ.

נִשְׁמַת סְבוּאָיו – קִנְּמָן-בֶּשֶׂם, וְכַ-/ פֵּיהֶם – בְּאַבְקַת רוֹכְלִים רֻקָּחוּ,

שָתוּ, וְהַיּוֹתֵר כְּשֶׁמֶן טוֹב עֲלֵי / רֹאשָׁם לְהִתְעַדֵּן בְּיָד מָשָׁחוּ.

יַשְׁקֶה אֱלֹהִים הַזְּמוֹרָה מִמְּטַר / עֻזּוֹ וּבַדֶּיהָ – עֲדֵי יִצְלָחוּ

כִּי הֶאֱרִיכָה גִּיל אֲנָשִׁים קָצְרוּ / מַדֵּי שְׂשׂוֹנֵיהֶם וְלֹא סָרָחוּ,

יֵאָלְמוּ שִׂפְתֵי יְגוֹנוֹת מֵעֲנוֹת – / כַּאְשֶׁר גְּבוֹהָה הָעֲלִיצוּת שָׂחוּ.

מַשְׁקֶה, לְטֹש עֵינַי בְנָגְהוֹ, וַעֲלֵי / אָזְנַי קְרָא בִשְׁמוֹ – וְיִתְפַּתָּחוּ,

כַּאְשֶׁר בְּתָמְכוֹ נָעֲמוּ יָדַי, וּפִי / מָתַק, וְאַפַּי מָר דְּרוֹר יָרָחוּ.

מֵסַב כְּרָקִיעַ, וּמִזְרָקָיו כְּכוֹ-/כָבִים בְּכַפֵּי סוֹבֲבִים זָרָחוּ:

אָכֵן לְפִיהֶם עֵת פְּנוֹתָם לַעֲרֹב – / יָדָם וְלֶחְיֵיהֶם בְּזָהָב טָחוּ.

כָּבְדוּ – וְהֵם רֵקִים, וְהָעֵת מָלְאוּ – / קַלּוּ וְאֶל פִּיהֶם כְּעוֹף פָּרָחוּ!

כֵּן הַפְּגָרִים בַּנְּפָשִׁים נִמְצְאוּ – / קַלִּים, וּבִלְתָּם – אַרְצָה צָנָחוּ.

יָרוּם וְיִתְנַשֵּׂא לְבָבִי, בַּחֲזוֹת/ כּוֹסִי – לְרַגְלָיו הַכְּלִי יִשְׁתָּחוּ.

אֶקְרָא לְמֵרֵעַי לְהָרִיעַ עֲלֵי / נָהָר, וְאֶל אַחַי שְׁתוֹת עַל אָחוּ,

כִּי הַסְּתָיו עָבַר וְהֵנֵצוּ עֲצֵי / בֹשֶׂם וְצִצֵּי הַהֲדַס פָּרָחוּ.

חוּשׁוּ לְהִסְתּוֹפֵף בְּצֵל אָמִיר, וְעַל / שׁוֹשָׁן יְדֵי רוּחַ בְּגַן שָׁטָחוּ,

וַיִּתְּנוּ רֵיחַ עֲרוּגוֹת נָשְׁפוּ / רוּחוֹת נְשָׁפִים בָּם עֲדֵי נָפָחוּ.

כָּל עִלְגֵּי הָעוֹף לְמִינוֹ הֶחֱלִיף, / זָמִיר לְעֻמַּת נוֹגְנִים פָּצָחוּ.

בִּי הַמְּנַגֵּן הַמְּמַגֵּן מַחְשְׁבוֹת / עָצְבִּי וְתוּגָתִי – כְּצֵל יִבְרָחוּ!

כִּנּוֹר – דְּמוּת פַּעַם לְיַד פַּחַד בְּלִי / שׁוֹק, דָּבְקוּ יַחַד וְלֹא נֻתָּחוּ–

יִתַּר לְבָבִי אֶל יְתָרָיו, עֵת אֲשֶׁר / נָעוּ וְלִקְרָאתָם אֲחֵרִים נָחוּ.

אֶתְמַהּ לְהוֹד יָדָיו אֲשֶׁר הָדוּ וְעַל / קַו קָפְצוּ בִסְפָר וְגַם שֻׁלָּחוּ–

גִּיל הַנְּשָׁמוֹת הַנְּשַׁמּוֹת וַעֲלֵי / אִישִׁים אֲנוּשִׁים מִנְּגֹף פָּסָחוּ:

הַקּוֹל וְהַפֹּעַל בְּמוֹ מִשְׁקָל וְעַל / מִסְפָּר בְּמוֹפֵת הָאֱמֶת הֻנָּחוּ.

יִסָּגְרוּ דַלְתֵי אֲפֵל, אַךְ מִשְׁכְּנֵי / עֶלְיוֹן לְעֵינֵי יְדוּעִים נִפְתָּחוּ,

יַעְלוּ לְעוֹלָם הַנְּפָשִׁים מִבְּלִי / מַעְלוֹת וְנַחְלֵי הַיְקָר יִצְלָחוּ,

זַכּוּ שְׂעִפֵּיהֶם וְכִמְעַט יֹאמְרוּ: / רוּחוֹת מְשָׁרְתֵי אֵל עֲלֵיהֶם נָחוּ!

עִם תּוֹפְשֵׂי כִנּוֹר וְעוּגָב יִצְהֲלוּ / אוֹנִים, וְלָהֶם מִבְּכִי יִרְוָחוּ –

בִּלְתִּי כְאֵבִי עַל בְּנֵי אָבִי אֲשֶׁר / סָפוּ וּמוֹדָעַי אֲשֶׁר הֻדָּחוּ.

דִּמְעִי לְבַד חֻקִּי – וְהוּא נִשְׁקִי, וְאִם / מֵעַי בְּרֻבֵּי נוֹזְלָיו יִרְתָּחוּ.

עֵינַי בְדִמְעִי נִגְּרוּ מֵאֵין דֳּמִי / לִדְמִי יְמוֹתַי עִם תְּמוֹל אָרָחוּ

יִכְלוּ וְלִמְאֹד יֶחְמְרוּ מֵימָם, הֲכִי / אוֹתָם דְּמֵי עַיִן וְלֵב דָּלָחוּ

כַּמָּה שְׁאִלְתִּימוֹ לְהֵעָצֵר – וְלֹא / אָבוּ, וְיִסַּרְתִּים – וְלֹא נוֹכָחוּ,

וּבְשׁוּב שְׂעִפַּי מִבְּלִי תִקְוָה וְלֹא / אוֹן, עַל סְעִפֵּי הַדֳּמִי בָטָחוּ.

אָנוּד עֲלֵי שֶׁמֶשׁ נְעוּרִים עָרְבָה / וּשְׁמֵי בְחוּרוֹתַי אֲשֶׁר נִמְלָחוּ,

בָּגְדוּ כְנַחַל בִּי וְהֵפֵרוּ בְרִית / אַהַב, וְאוֹתִי מֵאֱנוֹשׁ זָנָחוּ,

עֵץ הַנְּדוֹד – לֹא נָבְלוּ עָלָיו, הֲכִי / עַל מֵי דְמָעַי כַעֲרָבִים קָחוּ!

רָחְקוּ מְתֵי סוֹדִי, וְאַךְ יִשְׁתַּעְשְׁעוּ / נֶגְדִּי יְצוּרֵיהֶם וְלֹא יִזָּחוּ,

בָּשְׂמֵי שְׁמַנֵּיהֶם, אֲשֶׁר בָּם נִרְקְחוּ, / בַּחְלוֹם עֲלֵי פִצְעִי נְדוֹד מָרָחוּ.

אִם צָעֲנוּ עִם אָהֳלֵיהֶם – אֶשְׁכְּנָה / תוֹךְ אָהֳלֵי אַהַב אֲשֶׁר מָתָחוּ.

עַל הַזְּמָן אֶלּוֹן? וְאֵין לוֹ חֵטְא! אֲבָל / עַל מוֹעֲצוֹת אִישָׁיו אֲשֶׁר נִסְרָחוּ –

נִטְעֵי זְמוֹרֵי זָר, אֲשֶׁר שָׂגוּ בְתוֹךְ / מָקוֹם אֲשֶׁר לֹא נִטְּעוּ צָמָחוּ,

קָפְצוּ עֲלֵי הוֹנָם – וְאַךְ לֹא קָפְצוּ / בִזְכֹר פְּעָלֵימוֹ אֲשֶׁר נֶאְלָחוּ!

אַפִּי עֲלֵי חַיַּי מְאֹד יִחַר, הֲכִי / הָאֵשׁ בְּאַפִּי בּוֹעֲרִים קָדָחוּ.

יֵצַר לְבָבִי בָם – וְאִם הוּא יָם אֲשֶׁר / לֹא יֶחֱזוּ חוֹפוֹ אֲשֶׁר בּוֹ שָׂחוּ.

אִם נִדְבְּרוּ בִי – הַכְּלָבִים חָמְדוּ / מֵאָז מְאוֹר סַהַר וְלוֹ נָבָחוּ.

אֵיכָה יְבִינוּן מַחְשְׁבוֹת דֵּעִי – וְהֵם / יֵאָבְקוּ עִם עָשׁ וְיִתְנַגָּחוּ!

יִבְכּוּ צְבָא מָרוֹם עֲלֵי מִקְרִי, וְלִי / יִתְנוֹדֲדוֹ אוֹרָיו וּמַר יִצְרָחוּ.

לָקְטוּ חֲצִירֵי מַאֲמָר לָהֶם – בְּיוֹם / אֶקְטֹף אֲנִי צִצָּיו בְּעֵת פִּתָּחוּ,

עַד עָלְצוּ בָהֶם בְּנֵי הַשִּׁיר – וְאִם / יֵאָנְקוּ דוּמָם וְיֵאָנָחוּ –

יִתְאַמְּרוּ בָהֶם אֲדוֹנֵי מַאֲמָר, / יִתְהַלֲלוּ בִשְׁמָם וְיִשְׁתַּבָּחוּ!
