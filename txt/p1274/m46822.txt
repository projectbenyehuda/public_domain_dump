


	
	בְּאֵר מַיִם חַיִים
	




רַבִּי יִשְׂרָאֵל בַּעַל שֵׁם טוֹב מְיַסֵּד הַחֲסִידוּת
הָיְתָה לוֹ חֲבוּרָה שֶׁל תַּלְמִידִים, שֶׁהָיָה מְטַיֵּל עִמָּהֶם
בַּשָּׂדֶה וּבַיַּעַר, כְּדֵי לְהִסְתַּכֵּל בְּמַעֲשֵׂי הַבּוֹרֵא
וְלָבוֹא מִתּוֹךְ כָּךְ לִידֵי הַשָּגָה גְבוֹהָה בְּעִנְיְנֵי שָׁמַיִם
וְתַכְלִית חַיֵּי הָאָדָם.

יוֹם אֶחָד הָיוּ מְהַלְּכִים בַּעֲרָבָה גְדוֹלָה וְהָיוּ
שְׁקוּעִים בְּשִׂיחָה עֲמֻקָּה, וְלֹא הִשְׁגִיחוּ שֶׁהַיּוֹםֹ נִָטָה
לַעֲרֹב וְהִגִּיעָה הַשָׁעָה לְהִתְפַּלֵּל תְּפִלַּת מִנְחָה,
שֶׁזְּמַנָּהּ קָבוּעַ לִפְנֵי שְּׁקִיעַת הַחַמָּה. בִּקְשׁוּ מְעַט מַיִם
לִטְֹּל אֶת יְדֵיהֶם וְלֹא מָצְאוּ. נִתְפַּזְּרוּ הַתַּלְמִידִים
וְחִפְּשׂוּ כֹּה וָכֹה וְשָבוּ בְּיָדַיִם רֵיקוֹת – אֵין מַעְיַן מַיִם
בְּכָל הָעֲרָבָה מִסָּבִיב.

וְהַמּוֹרֶה הַנַּעֲרָץ עוֹמֵד וְרוֹאֶה בָּאֲוִיר קְהַל
נְשָׁמוֹת עֵירֻמּוֹת מְרַחֲפוֹת, מִתְקָרְבוֹת וּמִתְרַחֲקוֹת, כְּיוֹנִים
מִתְדַפְּקוֹת עַל שׁוֹבָךְ סָגוּר, מְחַכּוֹת לִתְפִלַּת הָאִישׁ
הַקָּדוֹשׁ, שֶׁתָּבִיא לָהֶן תִּקּוּן וְתַחֲזִירֵן לִמְקוֹמָן תַּחַת
כִּסֵּא הַכָּבוֹד.

נָתַן אוֹת לַתַּלְמִידִים וְעָמְדוּ מִסָבִיב לוֹ
בְּעִגּוּל. הֵרִים הַבַּעַל שֵׁם טוֹב אֶת הַמַּטֶּה שֶׁבְּיָדוֹ וְהִכָּה
בְּאֶבֶן שְׁהָיְתָה מֻטֶּלֶת עַל הַקַּרְקַע. מִיָּד נִתְבַּקְעָה הָאֶבֶן
וְזֶרֶם מַיִם חַיִּים עָלָה מִתּוֹכָהּ בְּסִילוֹן מֵאִיר.

רַחֲצוּ כֻּלָּם אֶת יְדֵיהֶם וְאֶת פְּנֵיהֶם בְּמַיִם
וְנִגְּבוּם בַּמִּטְפָּחוֹת שֶׁבְּכִיסֵיהֶם וְעָמְדוּ לְהִתְפַּלֵּל,
וּפְנֵיהֶם מִזְרָחָה לְעֵבֶר הַר הַבַּיִת, כַּמִּנְהָג.

וּכְשֶׁכִּלּוּ אֶת תְּפִלָּתָם רָאוּ שֶׁהַמַּיִם
הַנּוֹבְעִים מִן הָאֶבֶן חַפְרוּ לָהֶם חֲפִירָה עֲגֻלָּה וּמִלְאוּהָ,
וְהַמַּיִם נוֹבְעִים וְרוֹעֲדִים בְּתוֹךְ הַחֲפִירָה כִּבְאֵר מַיִם
חַיִּים.

לְָמָחֳרַת הַיּוֹם עָשְׂתָה לָהּ הַשְׁמוּעָה כְּנָפַיִם
בָּעִיר וּבַסְּבִיבָה: בָּעֲרָבָה הַחֲרֵבָה, שֶׁלֹא רָאֲתָה מַעְיַן
מַיִם מֵעוֹלָם, פָּתַח הַצַּדִיק מַעְיַן מַיִם חַיִּים. מִהֲרוּ רַבִּים
אֶל הַמָּקוֹם לִרְאוֹת בְּמַעֲשֵׂה הַנֵּס, וְלֹא יָכְלוּ עֵינֵיהֶם
לִשְׂבֹּעַ מִמַּרְאֵה הַמַּיִם הַזַּכִּים המְַפַכְפְַכִים
בְּשָׂשׂוֹן.

הָיָה שָׁם מְצֹרָע אֶחָד, שָׁאַב מִן הַמַּיִם
בְּחָפְנָיו וְרָחַץ אֶת פָּנָיו וְאֶת צַוָּארוֹ וְאֶת יָדָיו,
נִתְעַלְּמָה הַצָּרַעַת וּבְשָׂרוֹ שָׁב כִּבְשַׂר נַעַר קָטֹן.

מֵאָז הָיְתָה הַבְּאֵר בָּעֲרָבָה מַעְיַן יְשׁוּעָה
וּמַרְפֵּא לְכָל אִישׁ יְדוּעַ חֹלִי וּלְכָל אִישׁ נְכֵה רוּחַ. הַחוֹלֶה
שֶׁשָּׁתָה מִן הַבְּאֵר שְָב וְרָפָא לוֹ, וְהָאִישׁ מַר הַנֶּפֶש בָּאוּ
הַמַּיִם אֶל קִרְבּוֹ לְהַחֲיוֹת אֶת רוּחוֹ וּלְמַלְּאוֹתוֹ מִבְטַח עֹז.
הַפִּסֵּח טָבַל אֶת רַגְלוֹ בְּמֵי הַבְּאֵר – וְלֹא הָיָה לוֹ עוֹד
צֹרֶךְ לָצֵאת בְּקַב אוֹ בְּמִשְׁעֶנֶת; הָעִוֵּר טָבַל אֶת
אֶצְבְּעוֹתָיו וּנְתָנָן עַל עֵינָיו – מִיָּד נִפְקְחוּ עֵינָיו וְרָאָה
עוֹלָם וּמְלוֹאוֹ; הַחֵרֵשׁ טָבַל אֶת רֹאשׁוֹ בַּמַּיִם – מִיָד
נִפְתְּחוּ אָזְנָיו וְקוֹלוֹת הָעוֹלָם בָּאוּ אֶל קִרְבּוֹ לְהַרְנִין
אֶת נַפְשׁוֹ.

הָרוֹעִים הָיוּ מְבִיאִים אֶת עֶדְרֵיהֶם לְהַשְׁקוֹתָם
מִמֵּי הַבְּאֵר, וְכָל בְּהֵמָה אֲשֶׁר שָׁתְתָה מִן הַבְּאֵר שָׁמְנָה
וְכָשְׁתָה לְהַרְבּוֹת חֲלָבָהּ הַמָּתוֹק וּבְשָׂרָהּ הֶעָרֵב. אַנְשֵׁי
גַנִּים בָּאוּ לִשְׁאֹב מִן הַבְּאֵר לְהַשְׁקוֹת אֶת עֲצֵיהֶם, כְּדֵי
שֶׁיַּעֲשׂוּ פֵּרוֹת גְדוֹלִים, נֶחְמָדִים לְמַרְאֶה וְטוֹבִים
לְמַאֲכָל, רְאוּיִים לַעֲלוֹת עַל שֻׁלְחַן מְלָכִים. גַם הַפְּרָחִים
מִמֵּי הַבְּאֵר שִׂגְשְׂגוּ וְעָשׂוּ עֲתֶרֶת צְבָעִים מַרְהִיבֵי עַיִן.
מֵרֵיחַ הַמַּים שִׁנְּתָה כָּל הָעֲרָבָה אֶת פָּנֶיהָ וְנִתְכַּסְּתָה
דֶשֶׁא וּפְרָחִים וְכָל צִמְחֵי הוֹד. בָּאוּ הָאִכָּרִים וְחָרְשׁוּ אֶת
אַדְמָתָהּ וְהָיוּ לָהֶם שְׂדוֹת־חֶמֶד, עַיִן לֹא רָאֲתָה
כְּמוֹהֶם.

וְעוֹד סְגֻלָּה אַחַת הָיְתָה לַמַּיִם בַּבְּאֵר.
כְּשֶׁרָבוּ שְׁנֵי בְּעָלִים בְּנֵי מֶצֶר עַל גְבוּל שְׂדוֹתֵיהֶם
וְהַשּׁוֹפֵטִ לֹא יָדַע מִי הִסִּיג גְבוּל רֵעֵהוּ, כִּי שְׁנֵיהֶם
הֵבִיאוּ עֵדִים לְצִדְקַת טַעֲנָתָם – צִוָּה הַשׁוֹפֵט לְמַלְּאוֹת מִן
הַבְּאֵר שְׁנֵי כַּדִּים עַד הַחֵצִי, וְאֶת הַכַּדִּים שָׂמוּ – הָאֶחָד
בִּפְאַת הַשָּׂדֶה הָאֶחָד וְהַשֵּׁנִי בִּפְאַת הַשָּׂדֶה הָאַחֵר
וְשׁוֹמְרִים הֶעֱמִידוּ עַל שְׁנֵי הַכַּדִּים לִשְׁמֹר כָּל הַלַּיְלָה
שֶׁלֹא יִגַּע בָּהֶם אִישׁ. וּכְשֶׁעָלָה הַשַּׁחַר הִבִּיטוּ לְתוֹךְ
הַכַּדִּים וּמָצְאוּ כִּי נֶעְתְּקוּ הַמַּיִם מִן הַכַּד הָאֶחָד אֶל
הַכַּד הָאַחֵר, זֶה מָלֵא וְזֶה רֵיק – וְיָדַע הַשּׁוֹפֵט וְיָדְעָה כָּל
הָעֵדָה, כִּי בַּעַל הַכַּד הָרֵיק שָׁלַח אֶת יָדוֹ בִּשְׁדֵה רֵעֵהוּ
לִגְזֹל מֵאַדְמָתוֹ.

וּבְכֵן הָיְתָה הַבְּאֵר לְעֵד וּלְשׁוֹפֵט בְּכָל דְבַר
פֶּשַׁע, וְהָיְתָה אֵימָתָהּ עַל כָּל מִי שֶׁהִשִּׁיאוֹ יִצְרוֹ לַעֲשֹׁק
וְלִגְזֹל, שֶׁמָּא יָבוֹאוּ הַמַּיִם וְיוֹכִיחוּ אֶת אַשְׁמָתוֹ.
שָׁבְתָה הָרָעָה וְהַשָּׁלוֹם נָכוֹן בָּאָרֶץ. עָבְרוּ שָׁנִים רַבּוֹת,
הַבַּעַל שֵׁם טוֹב נִסְתַּלֵּק מִן הָעוֹלָם, וְיוֹשְׁבֵי הָאָרֶץ נֶהֱנוּ
מִבִּרְכַּת הַבְּאֵר וְהָעֹשֶׁר בָּא לִשְׁכֹּן בְּבָתֵּיהֶם, וְלֹא הָיָה
אִישׁ עָנִי בְּכָל הָאָרֶץ הַהִיא. אֲבָל עִם הָעֹשֶׁר שָׁמַן לִבָּם, לֹא
עָשׂוּ רָעָה מֵאֵימַת הַמַּיִם, אַךְ גַם אֶת הַטּוֹב לֹא עָשׂוּ. כָּל
אֶחָד אָמַר: שֶׁלִּי – שֶׁלִּי, שֶׁלְּךָ – שֶׁלְּךָ, וְלֹא הֵרַע וְלֹא
הֵיטִיב. לֹא טוֹב הַדָּבָר בְּעֵינֵי מַנְהִיג הָעוֹלָם וְאָמַר לְזַעֲזַע
אֶת בְּנֵי הָאָדָם מִשַּׁלְוָתָם. מֶה עָשָׂה? שָׁלַח בְּאַחַד הַיָּמִים
סוּפָה גְדוֹלָה. הֵגִיחָה הַסּוּפָה פִּתְאֹם וְהֵבִיאָה עִמָּהּ
תִּימְרוֹת אָבָק כַּעֲנָנִים, נִשָׂא הָאָבָק בֵּין הַשָּׁמַיִם וּבֵין
הָאָרֶץ וְהִתְגַלְגֵל כָּל הַיּוֹם וְכָל הַלָּיְלָה עַל פְּנֵי
הַשָּׂדוֹת וְהָעֲרָבָה הַגְּדוֹלָה.

וּכְשֶׁנִּשְׁתַּתְּקָה הַסּוּפָה יָצְאוּ הָאֲנָשִׁים
בַּבֹּקֶר לִשְׁאֹב מַיִם מִן הַבְּאֵר – וְלֹא מְצָאוּהָ. הַדֶּשֶׁא
בָּעֲרָבָה נֶחֱרַךְ בְּלַהַט הָרוּחַ, הַפְּרָחִים נָבְלוּ וְאַדְמַת
הָעֲרָבָה חָרְבָה כְּמוֹ שְׁהָיְתָה לִפְנֵי הַבְּאֵר. הָאָבָק סָתַם אֶת
הַבְּאֵר וְלֹא נוֹדַע מְקוֹמָהּ.

בָּאוּ הַחֲסִידִים מִן הָעִיר הַסְּמוּכָה, אָמְרוּ
הַלֵּל הַגָּדוֹל וְחָפְרוּ בָּאֲדָמָה, חָפְרוּ פֹּה וְחָפְרוּ שָׁם וְלֹא
מָצְאוּ אֶת הַבְּאֵר. חוֹלִים ודַכְּאֵי רוּחַ בָּאוּ מִמֶּרְחָק וְלֹא
מָצְאוּ אֶלָּא עֲרָבָה גְדוֹלָה, חֲרֵבָה וְשׁוֹמֵמָה, וְאֵין בָּה סִמָּן
שֶׁל בְּאֵר. נָפְלוּ עַל פְּנֵיהֶם וְצָעֲקוּ אֶל הַשָּׁמַיִם וְלֹא
נַעֲנוּ.

וְרַבִּים מִיּוֹשְׁבֵי הָאָרֶץ יָרְדוּ מִנִּכְסֵיהֶם
וְנַעֲשׂוּ עֲנִיִּים. הִתְחִילוּ הָאֲנָשִׁים רָבִים עַל דָבָר גָדוֹל
וְעַל דָבָר קָטֹן, וְהָיוּ רָעִים וְאַכְזָרִים וְהָיוּ גַם טוֹבִים
וִישָׁרִים, עוֹשֵׂי עַוְלָה וְעוֹשֵׂי צְדָקָה וָחֶסֶד, וְהַטּוֹבִים
מִתְאַמְּצִים לְהַצִיל עָנִי מִגּוֹזְלוֹ.

וְהָיָה שָׁם אִישׁ אֶחָד, תָּם וְיָשָׁר, וְלוֹ בָּנִים
רַבִּים וְאֶת כֻּלָּם הִדְרִיךְ בְּאֹרַח מִישׁוֹר, אַךְ נַחֲלָתוֹ לֹא
מָצְאָה לוֹ לְפַרְנֵס אֶת כֻּלָּם, וּכְשֶׁגָּדְלוּ לֹא הָיָה בְּיָדוֹ
מַה לְהוֹרִישׁ לָהֶם. הִָיָה הָאִישׁ מִצְטָעֵר, זוֹכֵר אֶת יְמֵי
הַבְּאֵר הַמַּפְלִיאָה לַעֲשׂוֹת וּבוֹכֶה עַל הַטּוֹבָה שֶׁבָּרְחָה מִן
הָעוֹלָם.

וְלַיְלָה אֶחָד רָאָה בַּחֲלוֹמוֹ זָקֵן לָבוּשׁ לְבָנִים
עוֹמֵד עָלָיו, מַחֲזִיק בִּכְתֵפוֹ וּמְנַעֲרוֹ בְּחָזְקָה: מַה לְּךָ,
נִרְדָם, קוּם קַח אֶת הַמַּעְדֵר וְצֵא אֶל שָׂדְךָ וַחֲפֹר בְּטַבּוּר
הַשָּׂדֶה וּמָצָאתָ אֶת הַבְּאֵר שֶׁנִּתְעַלְּמָה. קוּם וְאַל
תְּאַחַר!

קָם הַאִישׁ בְּבֶהָלָה. רָץ אֶל הֶחָצֵר וְלָקַח אֶת הַמַּעְדֵר וּבָא אֶל הַשָּׂדֶה עִם הָנֵץ הַחַמָּה.

מִשֶּׁנָּעַץ אֶת הַמַּעְדֵר בְָאֲדָמָה קָפְצוּ הַמַּיִם
לִקְרָאתוֹ, מַיִם זַכִּים, מַיִם חַיִּים, נוֹבְעִים וּמְפַכְפְּכִים
בְּשָׂשׂוֹן.

נָפַל הָאִישׁ עַל פָּנָיו וְהוֹדָה לְהַשֵּׁם חַסְּדוֹ.

יָצְאָה הַשְּׁמוּעָה בְּכָל הָאָרֶץ, וּמִכָּל
הַמְּקוֹמוֹת נָהֲרוּ לִרְאוֹת אֶת הַבְּאֵר שְׁחָזְרָה וְנִתְגַלְּתָה
בִּשְׂדֵה הָאִישׁ הַיָשָׁר.

וְהַבְּאֵר הָיְתָה לִבְרָכָה לָאָרֶץ וְלִמְקוֹר מִחְיָה לְבַעַל הַשָּׂדֶה וּלְזַרְעוֹ אַחֲרָיו עַד הַיּוֹם הַזֶּה.
