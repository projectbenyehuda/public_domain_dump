


	
	קַבָּלַת שַׁבָּת בִּירוּשָׁלַיִם
	








בָּאתִי יְרוּשָׁלַיְמָה בְּעֶרֶב שַׁבָּת, וְהַשֶּׁמֶשׁ לַעֲרֹב פּוֹנָה.

פּוֹעֲלִים מִהֲרוּ הַבַּיְתָה, זְקֵנִים לִתְפִלָּה, בַּשָּׁמַיִם רִחֲפָה יוֹנָה.





טְהוֹרֵי יְהוּדִים לְבוּשֵׁי שַׁבָּת הִזְכִּירוּ עַל הַמְּנוּחָה מֵחֲנוּת לַחֲנוּת.

קִשְׁקְשׁוּ תְּרִיסֵים, רְחוֹב יָפוֹ נָדַף בְּשָׂמִים וְצִפִּיָּה לִפְדוּת.





עָלִיתִי בְּמַעֲלֵה הַכִּכָּר. הִבְהִיק לִי בַּשְּׁקִיעָה חֲצִי הַיָּרֵחַ.

עֵינַי לֹא נִכְווּ בִּרְאוֹתִי גַם עָלָיו חֶסֶד־אֱלֹהִים זוֹרֵחַ.





פָּנִיתִי עוֹד וְרָאִיתִי אֶת שְׁמֵי־הַכְּחוֹל יִדְקֹר הַצְּלָב,

וְלִבִּי הָגָה הָגוּת־חֲשָׁאִים וְלֹא הָיָה נֶעֱלָב.





לָחַשְׁתִּי בְּרֶטֶט: הַגִּידִי, הַמִּתְגַלָּה וְהַמִּתְכַּסָּה,

הֲזֶה קֵץ הַיָּמִין וְזֹאת אַחֲרִית הַמַּסָּה?





לֹא אֶקְבֹּל, אַךְ הַגִּידִי, שֶׁאֵינֵךְ מְצִיאוּת וְאֵינֵךְ חֲלוֹם:

לְמִי תִּהְיִי בַּדּוֹרוֹת הָעוֹלִים כִּסֵּא וּלְמִי הֲדוֹם?





כְּמוֹ שֶׁאַתְּ: מַחֲצִיתֵךְ חֲרֵבָה וּמַחֲצִיתֵךְ בְּנוּיָה,

טֻמְאָתֵךְ בְּשׁוּלַיִךְ וְיַד־אֵל נְדִיבָה עָלַיִךְ נְטוּיָה —





הֲצִירֵי כָּל הַגּוֹיִים יִשְׁלוּ בָּךְ עוֹלָמִית בְּלֹא אַהֲבָה?

וְלֵב בָּנִים מֵאַפְסֵי אֶרֶץ יֶחֱרַד לָךְ, הַנָּאוָה?





לֹא אֶקְבֹּל, אַךְ הַגִּידִי, מָה אָנוּ לָךְ וּמַה לָנוּ אָתְּ,

צְלָב בִּימִינֵךְ, בִּשְׂמֹאלֵךְ חֲצִי־יָרֵחַ — וְאָנוּ פֹּה בְּאֶפֶס יָד?





וְעוֹד אֲנִי הוֹגֶה וְעִם לְבָבִי רַכּוֹת נִדְבָּר,

יָרַד הַלַּיְלָה וּכְנַף־רוּחַ הִשִּׁיבָה צִנָּה מֵעֵבֶר הַמִּדְבָּר.





וְאוֹרִים רַבִּים עָלוּ וְהִבְלִיחוּ עַלִּיזִים בָּעֵמֶק וּבָהָר,

וְלֹא יָדַעְתִּי אֵיזֶה מֵהֶם יָאִיר לֶעָתִיד וְאֵיזֶה לֶעָבָר.





וְזָכַרְתִּי שֶׁיֵּשׁ לִי פֹּה אַחִים בְּנֵי אֶרֶץ וּמוֹלֶדֶת,

אַךְ אֵין טַבַּעַת־הַזָּהָב שֶׁתְּהֵא לִבּוֹתֵינוּ מְאַחֶדֶת.





וְזָכַרְתִּי שֶׁצֶּמֶד יְלָדִים לִי בְּתֵל־אָבִיב, בֵּן עִם בַּת,

וְהֵם אֵינָם יוֹדְעִים רֵיחֵךְ, יְרוּשָׁלַיִם, בְּעֶרֶב שַׁבָּת.





וְיָדַעְתִּי כִּי לֹא פַּיְטָנֵךְ אֱהִי, תּוֹשָׁב בְּתוֹכֵכִי עִיר־הַקֹּדֶשׁ,

רַק עוֹלֶה לִתְהוֹת עַל חִידָתֵךְ פַּעֲמַיִם בַּשָּׁנָה אוֹ פַּעַם בַּחֹדֶשׁ.





וְרִחַמְתִּי אֶת כֻּלָּם וְאוֹתִי וָאֶצְנַח עַל אֶבֶן חֲצוּבָה,

וָאָלִיט פָּנַי כִּמְבָרֵךְ עַל הַנֵּרוֹת וְרוּחִי בִּי עֲצוּבָה.




תרפ"ד
