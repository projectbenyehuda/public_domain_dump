


	
	אֱנוֹשׁ רִמָּה
	



שיר על דרך המוסר






אֱנוֹשׁ רִמָּה עָפָר מִן הָאֲדָמָה / עַד אָן תִּישַׁן בְּחַבְלֵי הַתַּרְדֵּמָה

כִּי אֶשְׁאֲלָה מִמְּךָ, אֱנוֹשׁ, שְׁאֵלָה / אֲדַבְּרָה דִבְרֵי שֵׂכֶל תְּחִלָּה

אִם יֵיטִיב לָךְ, יְלוּד אִשָּׁה, בֶּן גִּילָךְ / כַּמָּה תַּעַשׂ שֵׁרוּתוֹ בְּכָל חֵילָךְ

וְאִם תִּלְוֶה כֶּסֶף שָׁוֶה בְשָׁוֶה / הֲלֹא תִהְיֶה כְּעֶבֶד לְאִישׁ מַלְוֶה

הֲלֹא צוּרָךְ מִן הָרֶחֶם שְׁמָרָךְ / וְעַד הֵנָּה תָמִיד הָיָה בְּעוֹזְרָךְ

הוּא הֱכִינָךְ וּמִטּוּבוֹ חֲנָנָךְ / וּמֵאַיִן הִרְבָּה לָךְ כָּל קִנְיָנָךְ

אֵין כָּל רֶגַע אֲשֶׁר אֵין בּוֹ כָּל פֶּגַע / וְהוּא מֵסִיר מִמְּךָ חֳלִי וָנֶגַע

גָּדוֹל חַסְדּוֹ וּבָרוּךְ שֵׁם כְּבוֹדוֹ / וּבְחֶמְלָתוֹ עוֹשֶׂה, נִפְלָא לְבַדּוֹ

מִי הִקְדִּימוֹ כִּי יָחוּשׁ לְשַׁלְּמוֹ / יָצַר הַכֹּל, בָּרוּךְ וּמְבֹרָךְ שְׁמוֹ

כָּל צְבָאָיו וּמַלְאָכָיו וּנְבִיאָיו / אֵין דֵּי סַפֵּר עֱזוּזוֹ וְנִפְלָאָיו

אֵיךְ בְּכָל זֶה לֹא תַּבִּיט וְתֶחֱזֶה / בְּעֵין שִׂכְלָךְ, אֱנוֹשׁ פָּחוּת וְנִבְזֶה

חוּשָׁה עוּרָה לְנַפְשָׁךְ הַשִּׁכּוֹרָה / לִזְכֹּר תָּמִיד אֶת מַחְצָבָה וּמְקוֹרָהּ

פַּנֵּה דַּעְתָּךְ לְהַבִּיט בְּשִׁפְלוּתָךְ / וּמָה אַתָּה וְעַצְמְךָ וּמַהוּתָךְ

שִׂים לְנֶגְדָּךְ מֵאַיִן הוּא יְסוֹדָךְ / וְלַהֶבֶל נֶחְשַׁבְתָּ גַּם בְּעוֹדָךְ

גַּם כְּבוֹדָךְ וְרֹב עָשְׁרָךְ וּמְאֹדָךְ / אֲשֶׁר תָּכִין לָךְ בְּכָל יְמֵי חֶלְדָּךְ

הַבֵּט הַסּוֹף לְמִי כָּל זֶה תֶּאֱסֹף / גַּם לְבַסּוֹף מַה יִּהְיֶה בּוֹ סוֹף כָּל סוֹף?

אִם תַּחְקֹרָה בְּכָל זֹאת וְתַסְבִּירָה / הֲלֹא תִלְהַב כְּלַהַב הַמְּדוּרָה?

גַּם בְּנָקָל בְּלִי חֶשְׁבּוֹן וּמִשְׁקָל / הֲלֹא תִמְצָא עַצְמָךְ תּוֹעֶה וּמְקֻלְקָל

מַהֵר סוּרָה מִדַּרְכֵי הָעֲבֵרָה / וְחִישׁ שַׁלֵּם לְנַפְשָׁךְ דֵּי מַחְסוֹרָהּ

קוּם לְרָחְצָהּ מִכָּל לִכְלוּךְ שִׁקּוּצָהּ / טַהֵר אוֹתָהּ מַהֵר נָא וּגְדֹר פִּרְצָהּ

רְפָא שִׁבְרָהּ וְהַתֵּר אֶת מַאְסָרָהּ / וְאָז תִּהְיֶה בִצְרוֹר חַיִּים צְרוּרָה

שְׁאַל עֶזְרָה מֵאֵל גָּדוֹל וְנוֹרָא / מַהֵרה יָחִישׁ לָנוּ בִנְיַן הַבִּירָה

וְנִשְׂמְחָה בְּקוֹל שָׂשׂוֹן וְשִׂמְחָה / וְנַקְרִיבָה שַׁלְמֵי זֶבַח וּמִנְחָה

קוֹל לְוִיִּים וַעֲבוֹדַת כֹּהֲנִים / וְיִשְׂרָאֵל שָׁמָּה כְאֶחָד חוֹנִים.
