


	
	[מֶלֶךְ עוֹלָם / אֲשֶׁר נֶעְלָם] / רְשׁוּת לְקַדִּישׁ
	



[מאִי מָיוֹרְקָה, חי בפָּלֶרמוֹ ונהרג שם
בן כ"ד שנה סמוך לשנת 1430]





רְשׁוּת לְקַדִּישׁ


מֶלֶךְ עוֹלָם / אֲשֶׁר נֶעְלָם / שִׁיר וּתְהִלָּה / לְךָ
יָאָתָה



בְּאֵימָה וּבְמוֹרָא / אֲסַפְּרָה / מַעֲשֶׂיךָ / אֲשֶׁר פָּעָלְתָּ



בָּרָאתָ / וְיָצַרְתָּ / בְּשֵׁשֶׁת יָמִים / כֹּל כִּלִּיתָ



הַשְּׁפָלִים / וְגַלְגַּלִים / וּשְׂכָלִים / בָּם כּוֹנָנְתָּ



טוּבְךָ יַתְמִידֵם / וְיַעֲמִידֵם / וְאֶת־כָּל־אֵלֶּה / יָדְךְ
עָשָׂתָה



בְּךָ קַיָּמִים / צוּר עוֹלָמִים / אַתָּה הוּא / בָּרוּךְ אָתָּה



יִתְהַלֵּל שְׁמַהּ / דְּקוּדְשָׁא בְּרִיךְ הוּא / לְעֵילָא וּלְעֵילָא /
מִן־כָּל־בִּרְכָתָא.





שֶׁפַע טוּבוֹ / יַעֲבָר־בּוֹ / עַל־שְׂכָלִים /
רֵאשִׁיתוֹ



וּמִשְּׂכָלִים / עַל־גַּלְגַּלִים / תַּעֲבֹר אֵת / הַשְׁפָּעָתוֹ



וְהַגַּלְגַּלִים / בְּעוֹלָם שְׁפָלִים / כִּי בְיָדָם אֵל / נָתַן
מֶמְשַׁלְתּוֹ



וְהֵם עֲגֻלִּים / וּכְגִלְדֵּי בְצָלִים / זֶה בְּתוֹךְ זֶה / וְרוּחַ
אֵין אִתּוֹ



וְכֻלָּם סוֹבֲבִים / וּבָהֶם כּוֹכָבִים / וּמְהַלֲלִים / כָּל־יוֹם
אוֹתוֹ



וְזֶה מִסְפָּרָם / לְפִי סִדּוּרָם / תִּשְׁעָה הֵמָּה / מֵאוֹר
עָשִׂיתָ.







הַגַּלְגַּל הָעֶלְיוֹן / בְּרוּם חֶבְיוֹן / הוּא
עֲרָבוֹת / בָּחַר אוֹתוֹ



שְׁנֵים עָשָׂר / נְשִׂיאִים קָשַׁר / יַעֲקֹב שָׁם / חֲקוּקָה צוּרָתוֹ



שַׁבְּתַי וְצֶדֶק / יְחַזֵּק כָּל־בֶּדֶק / מַאֲדִים מוּל /
מִלְחַמְתּוֹ



לַשֶּׁמֶשׁ שָׁם / אֹהֶל שָׂם / וְאֵין נִסְתָּר / מֵחַמָּתוֹ



וְגַם נֹגַהּ / אֲשֶׁר יִגַּהּ / וְכוֹכַב חַמָּה / בִּכְתִיבָתוֹ



וְהַלְּבָנָה / הִיא תַחְתּוֹנָה / תּוֹסִיף תִּגְרַע /
בְּטוֹבָתָהּ.





רֹאשׁ וְשִׁלְטוֹן / הֵם בַּתַּחְתּוֹן / כִּי־כֵן רָצָה /
בָּרוּךְ הוּא



חֹמֶר הָרִאשׁוֹן / הַתְחָלָה וְרִאשׁוֹן / אֶל־גֹּלֶם כֹּל / מֻרְכָּב
הוּא



וּמִתְפָּעֵל / אֵינוֹ פוֹעֵל / כְּמַעֲשֵׂה לִבְנַת / סַפִּיר
מַעֲשֵׂהוּ



כָּל־צוּרָה / מֶנּוּ נֶעְדָּרָה / וְהָאָרֶץ / הָיְתָה תֹהוּ



פּוֹשֵׁט צוּרָה / וְלוֹבֵשׁ צוּרָה / כִּי הַהֶעְדֵּר / דָּבֵק לוֹ הוּא



קִבֵּל צוּרוֹת / סְפוּרוֹת / הֵם הַיְסוֹדוֹת / אַתָּה יָצָרְתָּ.





מֵהַיְסוֹדוֹת / קַלּוֹת וּכְבֵדוֹת / נִפְעַל בְּחָמְרָם / פּוֹעֲלִים בְּצוּרָתָם



זַךְ עָב חָבֵר / אֵשׁ וַאֲוִיר / מַיִם וְעָפָר / הֵם אַרְבַּעְתָּם



טֶבַע הָאֵשׁ / חַם וְיָבֵשׁ / עֶלְיוֹן לְכֻלָּם / וּמַקִּיף אוֹתָם



הָאֲוִיר הַלַּח / הוּא חַם וְלַח / שְׁנֵי אֵלֶּה הֵם /
עוֹלִים לְקַלּוּתָם



וּמַיִם נִבְעָם / קַר לַח טִבְעָם / וְעַל־אֶרֶץ / מוֹשְׁבוֹתָם



הַיַּבָּשָׁה / קָרָה וִיבֵשָׁה / וְעַל־בְּלִימָה / אוֹתָהּ
תָּלִיתָ.





וְהַמֻּרְכָּבוֹת / הַמִּתְהַוּוֹת / מֵעֵרוּבָם /
אַרְבָּעָה הֵם



הַדּוֹמָם / עַד תֻּמָם / וְהַצְּמָחִים / לְמִינֵהֶם



וּמַרְגִּישִׁים / בַּעֲלֵי חוּשִׁים / אֲשֶׁר בִּלְתִּי / מְדַבְּרִים
הֵם



מַרְגִּישׁ מְדַבֵּר / אֲשֶׁר סוֹבֵר / בְּצֶלֶם אֱלֹהִים / עָשָׂה אֶתְהֶם



הוּא הָאָדָם / אֲשֶׁר בְּיָדָם / נָתַן הַכֹּל / וַיִּרְדוּ בָהֶם



וּבָחַרְתָּ אֶל / בְּיִשְׂרָאֵל / וְאֶת־תּוֹרָתְךָ / לָהֶם
נָתַתָּ.





סֵתֶר וּמָגֵן / בַּעֲדָם הָגֵן / וּפְדֵה אוֹתָם /
מִכָּל־צָרָה



לְמַעַן שִׁמְךָ / תְּכוֹנֵן עִירְךָ / וּתְקַבֵּץ שֶׂה / הַפְּזוּרָה



וְגַם תָּבִיא / הַנָּבִיא / אֱלִיָּהוּ / בִּבְשׂרָה



וְיָשִׁיב לֵבָב / הוֹלֵךְ שׁוֹבָב / לִפְנֵי בּוֹא יוֹם / אֵל הַנּוֹרָא



וְהַגּוֹאֵל / שַׁלַּח הוֹאֵל / בְּיָמֵינוּ / בִּמְּהֵרָה



תִּפְדֵּנוּ / וּתְקַבְּצֵנוּ / וּתְבִיאֵנוּ / אֶל־עִיר
בָּחָרְתָּ.
