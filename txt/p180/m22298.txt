


	
	הֲשָׁבַר הַזְּמַן
	






הֲשָׁבַר הַזְּמַן קַרְנֵי תְיָשָׁיו / וְאִם הִתְעַשְּׁתוּ צִירֵי
חֲרָשָׁיו



אֲשֶׁר הִתְעָרְבוּ בוֹ הַדְּבָרִים / וְאֵין פֵּרוּד לְנָשָׁיו
מֵאֲנָשָׁיו



אֲשׁוֹטֵט בָּם וְאֶתְבּוֹנֵן, יְדִידִי, / וְלֹא אַכִּיר נְעָרָיו
מִיְּשִׁישָׁיו



וְלוּלֵי כִי [זְקָ]נֵיהֶם אֲרֻכִּים / אֲזַי לֹא נֶחְשְׁבוּ רַַגְלָיו
כְּרֹאשָׁיו



וְאב…ם לְךָ מִלּוֹת קְנָאוֹת / וְעֵת כִּי יִצְדְּקוּ אָז יַעֲנוּ שָׁוְא



….תי כהשל עם שארי / אבד לירש חרש חֲרָשָׁיו



וּמַה זֶּה נֶעְדְּרָה חָכְמָה בְּתֵבֵל / הֲמֵת גַּלְגַּל וּמֵתוּ כָל
עֲיָשָׁיו



זְרָמִים נָשְׂאוּ אַנְשֵׁי תְבוּנוֹת / וְסָרוּ יוֹדְעֵי הַשִּׁיר
וְדוֹרְשָׁיו



אֲבָל בִּפְנוֹתְךָ פָּנוּ חֲכָמִים / כְּחוּט נִנְעָר - וְהִתְפָּרְדוּ
גְבִישָׁיו



בְּךָ אָרִיד וְאוֹרִיד רֹאשׁ זְמַנִּי / וְאֶתְחַפֵּשׂ בְּרָב כֹּחַ
לְבוּשָׁיו



וְעָלַי מַעֲמָד בֵּינִי וּבֵינוֹ / כְּמוֹ לַיִשׁ טְרָפוּהוּ לְיָשָׁיו



וְנָפַלְתִּי בְרִשְׁתּוֹ הַפְּרוּשָׂה / וְיַפִּיל הַקְּ[רָב] עִתִּים
חֲמוּשָׁיו



וּפָתַחְתָּ לְנַפְשִׁי שַׁעֲרֵי גִיל / וְנִצַּלְתִּי בְךָ מִפַּח
יְקוּשָׁיו



בְּךָ אֶבְטַח וְלֹא אֶבְטַח בְּבֶן אִישׁ / וְדַי לֹא יַאֲמִין הָאֵל
קְדוֹשָׁיו



וְאַתְּ שַׁחְקִי, וְאַתְּ חֵשֶׁק לְבָבִי / וְתַאְוָתוֹ, וְאַתְּ קֹדֶשׁ
קְדָשָׁיו.
