


	
	שְׁזוּפֵי פְלֵטַי
	








שְׁזוּפֵי פְלֵטַי / אֲפוּפֵי עֵיטַי / שׁוֹקְדַי וְעוֹטַי / עַל סַף רָהִיטָי



בְּחוּרֵי חֲנִיטַי / יְלָדַי מְעוּטַי / בַּעֲדַת קְשׁוּטַי / כְּלָלַי וּפְרָטָי



תַּמּוּ מְקַשְּׁטַי / וְחוֹגְרֵי אַבְנֵטַי / תּוֹפְשֵׂי מוֹטַי / וּמְנַצְּחַי וּפוֹרְטָי



שָׁלְמוּ שַׁלִּיטַי / וְנוֹשְׂאֵי שְׁלָטַי / שֻׁפְּכוּ בְהָטַי / וְשֻׁדְּדוּ חֲרִיטָי



בַּלַּע מַלְעִיטַי / רוֹשׁ וּמַעֲלִיטַי / בַּזֵּר שׁוֹמְטַי / וְחוֹבְטַי וְלוֹבְטָי



תְּטָאֵט מְטָאטַאי / תְּלַהֵט מְלַהֲטַי / תְּפַשֵּׁט מְפַשְּׁטַי / תְּקַמֵּט מְקַמְּטַי



וְשׁוֹעֲטַי וּבוֹעֲטַי / וְשׂוֹחֲטַי וְשׁוֹחֲטַי / וּמוֹרְטַי וְיוֹרְטַי / וְחוֹלְטַי וְשׁוֹלְטָי



נַהֵר מְקַלְּטַי / וּמַהֵר מְפַלְּטַי / נַחֵת לְקוּטַי / מִבֵּין שׂוֹרְטָי



הָעֵז מְשׁוֹטְטַי / מִפּוּלַי וּפוּטַי / הָשֵׁב שׁוֹפְטַי / וְתוֹמְכֵי שַׁרְבִיטָי



וּמִיַּד מָטַי / הָקֵל עַבְטִיטַי / וּמִיוֵן טִיטַי / מְשֹׁךְ מִשּׁוֹטָי



אַשֵּׁר מֶבָּטַי / וְיַשֵּׁר מַעְבָּטַי / אַמֵּץ קוֹלְטַי / מִזְּדוֹן שֵׂטָי



שַׁבֵּר שָׁאטַי / שַׁפֵּר לָאטַי / לְהַלְבִּין חוּטַי / לְסוּרַי וְחוֹטָאי



מְפַצֵּל בְּעֵטַי / מַקְלוֹת רְהָטַי / הָלֵץ שְׁבָטַי / הַָקּוֹרְאִים אֶת מִשְׁפָּטָי.
