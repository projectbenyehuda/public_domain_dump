


	
	וְאוֹמֵר אַל תֶּחְשֶׁה
	




וְאוֹמֵר: אַל תֶּחְשֶׁה / לְבָבְךָ פֶּן יִקְשֶׁה /
עֲנִיתִיו: אֵיךְ אֶחְשֶׁה / וְכִלְיוֹתַי חָרוֹת



וְעִמִּי תוֹחֶלֶת / לְבָבִי אוֹכֶלֶת / כְּלַהַב גַּחֶלֶת
/ תְּקַרְקֵר הַקִּירוֹת



קְרָבַי תֶּהְמֶינָה / וְעֵינַי תִּבְכֶּינָה / וְאוּלַי
תִּכְבֶּינָה / כְּלָיוֹת בּוֹעֵרוֹת



אֲשֶׁר אֵשׁ בָּם נֶחְבָּא / בְּיוֹם הַלֵּב נִשְׁבָּה /
וְלוּ תֹפֶת נִכְבָּה ־ / שְׁטָפוּהוּ נְהָרוֹת



מְאֹד נִכְחַד עָצְבִּי / וְלֹא יָדַע לִבִּי / בְּעָרוֹת
מִקִּרְבִּי / דְּמָעוֹת נִגָּרוֹת



וְזֶה מִכְתַּב יָדִי / יְהִי לָכֶם עֵדִי / וּמוֹפֵת עַל
סוֹדִי / וְאוֹתוֹת נִכָּרוֹת



בְּסוֹדִי כִּיד נִכְתַּם / בְּמוֹ סֵפֶר נֶחְתַּם / תַּם
/ לְהָבִין וּלְהוֹרוֹת



וְאוּלַי אִישׁ נוֹסָר / אֲשֶׁר בַּתֹּם נֶאְסָר /
וּמִדֶּרֶךְ לֹא סָר / לְנַתֵּק מוֹסֵרוֹת



וְאָרוּם עַל אוֹיְבִי / בְּנַפְשִׁי וּלְבָבִי / וְלוּ
הָיָה אָבִי / גְּעַרְתִּיו גְּעָרוֹת



וְשָׁחַח עַל קִצְפּוֹ / וְכָפַף עַל כַּפּוֹ / וְלוּ
שָׁחָה עָרְפּוֹ / רוֹת



עֲדֵי מָתַי אוֹחִיל / וְהַתּוֹחֶלֶת ־ חִיל / וְנַפְשִׁי
לָהּ תָּחִיל / כְּחִילֵי מַבְכִּירוֹת



הֲכֹל הַיּוֹם אֶפְנֶה / כְּמוֹ נִגַּשׂ נַעֲנֶה / וְלֹא
אֶמְצָא מַעֲנֶה / כְּפִיּוֹת נִסְכָּרוֹת



וְהָשִׁיב עַל יִשְׁאַל / זְמַנִּי הַנּוֹאָל / וְאֵיכָכָה
אֶשְׁאַל / גְּדוֹלוֹת וּבְצוּרוֹת



זְמַנִּי הַלּוֹחֵץ / אֲשֶׁר רֹאשִׁי מוֹחֵץ / יְרֵה
אוֹתִי כַחֵץ ־ / אֲנַחְנוּ מַטָּרוֹת!



אֲשֶׁר קַרְנִי גָדַע / וְאֵלַי יִתְוַדַּע / כְּאִלּוּ
לֹא יָדַע / לְבָבִי נִסְתָּרוֹת



זְמַן שֶׁקֶר עָמַס / וּבִי קָם הֶחָמָס / וְהִשְׁפִּיל
אַף רָמַס / נְפָשׁוֹת הַיְקָרוֹת



וְהֶחְזִיק בַּתַּרְמִית / וּבֵין רֵעִים הִצְמִית /
וּפֵרוּדוֹ הֶעְמִית / לְעֻמַּת הַצָּרוֹת



וְאֵינֶנִּי גֶבֶר / יְקַוֶּה הַשֵּׂבֶר / אֲבָל אֶחְצֹב
קֶבֶר / לְנַפְשִׁי בַצּוּרוֹת



וְאִלּוּ חִכִּיתִי / וְסֵבֶר אִוִּיתִי / מְעַט
שֶׁקִּוִּיתִי / לְהַשִּׂיג מַזָּרוֹת



וְאֵיךְ לֹא אֶוָּאֵשׁ – / וְהַנֶּפֶשׁ כָּאֵשׁ / וְשִׂים
עַל עֵצִים אֵשׁ / יְקוּמוּן תַּבְעֵרוֹת



אֲקַוֶּה לִצְדָקָה –/ וְאֶמְצָא מוּעָקָה / וְתֵבֵל
כִּמְתוּקָה – / וּמָתְקָהּ כִּמְרוֹרוֹת

וְהִיא הַנּוֹסֶדֶת / לְעוֹלָם עוֹמֶדֶת / וְהִיא
כְּיוֹלֶדֶת / וְהַיָּמִים ־ הָרוֹת



וְהִיא גַם רַע מִמְשַׁק / וְהַכֹּל בָּהּ חָשַׁק /
לְמַעַן כִּי תִשַּׁק / נְשִׁיקוֹת נַעְתָּרוֹת



וְהָעוֹלָם רוֹעֵץ / וְהָאָדָם כָּעֵץ / וְהַמָּוֶת יוֹעֵץ
/ יְקַצֵּץ הַפֹּארוֹת



וְעַיִן לֹא יַעְלִים / אֲנָשִׁים מֵהַשְׁלִים / אֲבָל הֵם
זַלְזַלִּים / וְהוּא כַמַּזְמֵרוֹת



כְּצוֹדֵד מֵגִיחַ / אֲלֵיהֶם יַשְׁגִּיחַ / וְכֻלָּם
יָגִיחַ / אֱלֵי פִיו כִּנְהָרוֹת



יְגַלֶּה אָזְנֵיהֶם / בְּעֵת יִקְרָא לָהֶם / וְיִפְקַח
עֵינֵיהֶם – / וְהָיוּ נִסְגָּרוֹת



וְתֵבֵל תַּנְחֵנוּ / עֲדֵי תַּנִּיחֵנוּ / בְּיוֹם
תִּשְׁלָחֵנוּ / לְבִתָּהּ בִתְשׁוּרוֹת



אֲנַחְנוּ נִפְזָרִים / בְּפִיהָ נִקְבָּרִים / כְּעֵצִים
נִגְזָרִים / מְכֻסִּים בִּגְדֵרוֹת



וְאֵל שׁוֹכֵן עֶרֶץ / לְךָ יִגְדֹּר פֶּרֶץ / וְתִהְיֶה
בָאָרֶץ / בְּרָכָה לַדּוֹרוֹת



שְׁלוֹמָךְ כַּנָּהָר / וְכַשַּׁחַק תִּזְהַר / כְּמוֹ
אַתָּה נִזְהָר / בְּקַלּוֹת וַחְמוּרוֹת



גְּעַר לָךְ בִּדְאָגָה / וְתֵן לַלֵּב פּוּגָה / וּמַה
תּוֹעִיל תּוּגָה / וְאַף עַל הָעֲבָרוֹת?



הֱיֵה דוֹמֵם זוֹמֵם – / לְעוֹלָם תֵּרוֹמֵם / וְלָמָּה
תִשְׁתּוֹמֵם? / תְּחֻלַּץ מִצָּרוֹת



הֲלֹא מֻדַּד חֶבֶל / עֲלֵי אַנְשֵׁי תֵבֵל / לְהַשְׁחִית
וּלְחַבֵּל / וְלִזְרוֹת וּלְזָרוֹת



אֱנוֹשׁ כַּצִּיץ נָמֵל / וְכַקָּנֶה קָמֵל / וּמַה
יִּתְרוֹן עָמֵל – / וְיוּבַל לִקְבָרוֹת!



וּמָוֶת יִרְעֶנּוּ / הֲלֹא יַשִּׂיגֶנוּ / וְלוּ
יִמְצָאֶנּוּ / בְּבֵין הַמַּזָּרוֹת!



וְרוּחוֹ לֹא יַצֵּל / אֲשֶׁר יָמָיו כַּצֵּל / וְאֵיכָה
יִנָּצֵל – / וְעָלָיו מִשְׁמָרוֹת!



וְהַמָּוֶת מָסָךְ / עֲלֵי כָל חַי נָסַךְ / וְהָיָה לָהֶם
סָךְ / וְעָבְרוּ הָעְבָרוֹת



כְּאִלּוּ הוּא מִסְעָד / לְכָל אָדָם נוֹעָד / וּמִי זֶה
חַי לָעַד / וְעָמַד לַדּוֹרוֹת?



וּמִי עֹז הִתְאַזָּר / וּמִמָּוֶת נִגְזַר / וּבַחַיִּים
נִגְזַר / וְנִמְלַט מִבּוֹרוֹת?



וְשֹׁרֶשׁ כֹּל יָקוֹט / וְלֹא יוּכַל לִשְׁקֹט /
וְהָאָדָם יִלְקֹט / כְּאוֹרֶה הָאוֹרוֹת



וְכֹל יִשְׁאַל שָׁאוֹל / לְבַל יֵרֵד לִשְׁאוֹל – / וּמִי
יוּכַל לִגְאֹל / נְפָשׁוֹת מִנְּקָרוֹת



גְּבִיר אַחִים תִִּהְיֶה – / הֲלָעַד אִישׁ יִחְיֶה /
וְהִתְבּוֹנֵן אַיֵּה / מְלָכִים וּגְבִירוֹת?



וְאַיֵּה אַבִּירִים / וְאַיֵּה כַבִּירִים / וְאַיֵּה
הַשָּׂרִים / וְאַיֵּה הַשָּׂרוֹת



וְאַיֵּה כָל הָמוֹן / וְאוֹצַר כָּל מַטְמוֹן / אֲשֶׁר
בָּנוּ אַרְמוֹן / וְיִשְׁבוּ הַטִּירוֹת?



הֲלֹא מוֹתָם אָרַב / לְעִתָּם כִּי קָרַב – / וְעָשׂוּ
חַיִל רַב / וּמִלְאוּ אוֹצָרוֹת!



וְיֵקַר פִּדְיוֹנָם / וְנָקֵל כָּל הוֹנָם / וְהַמָּוֶת
חִנָּם / לְקָחָם מִמְכָּרוֹת



וְהַחַי כֶּעָגוּר / בְּתוֹךְ מַסְגֵּר סָגוּר / כְּאֶחָד
בָּא לָגוּר / וְיָצָא מִמְגוּרוֹת.



אֲשֶׁר אֵין לוֹ מַעְצוֹר / לְךָ חֶסֶד יִצֹּר / אֲשֶׁר
רוּחוֹת יִבְצֹר / וְנוֹדַע בִּגְבוּרוֹת



וְיִבְנֶה הַהֶרֶס / וְיָקִים הַקֶּרֶס / וְתִהְיֶה
כַּחֶרֶס / עֲלֵי כָל הַמְאוֹרוֹת



יְמַלֵּא יָמֶיךָ / וְשָׂם נֵר לִשְׁמֶךָ / וְיַעְבִיר
קָמֶיךָ / בְּשׁוֹאוֹת וַעְבָרוֹת!
