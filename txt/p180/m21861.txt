


	
	תְּרִיבוּן אִישׁ
	




תְּרִיבוּן אִישׁ אֲשֶׁר לוּלֵי דְמָעָיו - / קְרָבָיו נִשְׂרְפוּ מֵאֵשׁ
צְלָעָיו



אֲמַרְתֶּם: דֹּם וְהִתְאַפֵּק - עֲדֵי כִי / בְּאֹרֶך הַנְּדוֹד כָּלוּ
רְגָעָיו



וְדַל מֵרֹב חֳלִי, עַד כִּי נְשָׂאוֹ / זְבוּב קָטָן בְּאַחַת
מִזְּרוֹעָיו



וְהוֹמֶה בַבְּכִי תָמִיד, כְּאִלּוּ / פְרָת עֵינָיו וְכִנֶּרֶת בְּמֵעָיו



וְצִירָיו הָפְכוּ עָלָיו לְבָבוֹ / וְנֶהֶפְכוּ עֲלֵי אַרְצוֹ רְקִיעָיו



וְנָדַר, אַחֲרֵי בֶגֶד יְדִידוֹ, / וְלֹא עָלָה עֲלֵי עֶרֶשׂ יְצוּעָיו



הֲבָגַדְתָּ, יְדִידִי, בָּאֳהָבִים - / וְאֵיכָה אִישׁ אֱמֶת יִבְגֹּד
בְּרֵעָיו?



וּמִיּוֹם סוּרְךָ - לֹא סָר יְגוֹנִי, / וְהִצִּיב הַכְּאֵב עָלַי
כְּרָעָיו



וְשַׁאְלוּ עוֹד לְעַפְעַפִּי וְיַגִּיד / אֲשֶׁר הֻמְסוּ בְּמֵימֵיהֶם
סְלָעָיו



אֲהַבְתַּנִי, וְהָיָה לִבְּךָ לִי / כְּלֵב זָקֵן לְיֶלֶד שַׁעֲשׁוּעָיו



נְטַעְתַּנִי וְאַחְרֵי כֵן תְּמָאֵן – / וְאֵיךְ יַעְקֹר אֱנוֹשׁ חָכָם
נְטָעָיו?



וְאֵיךְ תִּגְרַע וְתִמְנַע הַיְּדִידוּת - / וְאַתָּה מִנְּדִיבֵי עַם
וְשׁוֹעָיו?



וְאֵיכָה מִמְּךָ פִשְׁעָךְ יְכַסֶּה / לְבָבִי - וַאֲנִי אֵדַע
פְּשָׁעָיו?



וְלוּ יִכְתֹּב לְךָ סֵפֶר יְדִידוּת - / קְרַעְתָּהוּ וְשָׂרַפְתָּ
קְרָעָיו.



בְּאַנְשֵׁי רִיב יְהִי חֶרֶץ וְכָלָה / וְיָשׁוּף הַזְּמַן רָאשֵׁי
צְפָעָיו



אֲשֶׁר פָּתְחוּ כְּמוֹ חֶרֶב לְשׁוֹנָם / וְכַקַּשׁ נֶחְשְׁבוּ אַבְנֵי
קְלָעָיו



וְאַל יַשִּׂיאֲךָ מֵהֶם שְׁלוֹמָם - / הֲיַשִּׂיא מֵחֲמַת תַּנִּין
צְבָעָיו?
