[אפלס מעגלי] / שלמה אבן גבירול

ומן כלמאתה אלוסאס וקצאידה אלג͘האר קצידה יד͘ם ّ פיהא קומא ויפתכ͘ר בנפסה [ומדבריו הנאים ומשירותיו המוכרות שירה שבה הוא מגנה אחרים ומשבח את עצמו]

אֲפַלֵּס מַעְגָּלַי / וְאֵיטִיב מַעְלָלָי / וְאֵלֶךְ-תֹּם אוּלַי / לְאֵל יֶעְרַב שִׂיחִי



וְאֶתֵּן תּוֹדָתוֹ / וְאֶשְׁמֹר תּוֹרָתוֹ / וְאֶעְמֹד בִּבְרִיתוֹ / וְאֶשְׁמֹר מַסָּחִי



וּמֵעַי אַרְתִּיחָה / וְלֵב לֹא-אַקְשִׁיחָה / בְּזִכְרוֹ אָשִׂיחָה / וְתִתְעַטֵּף רוּחִי



זְמָמִי לֹא-נִבְצַר / וְיָדִי לֹא-תִקְצַר / בְּקָרְאִי מִמֵּצַר / אֲדֹנִי מִבְטַחִי



מְגַלֶּה תוּשִׁיּוֹת / וְחֹפֵשׂ מַשְׂכִּיּוֹת / לְבָבוֹת וּכְלָיוֹת / וּמַגִּיד מַה-שֵּׂחִי



בְּאַף יִזְכֹּר רַחֵם / לְאָדָם יִנָּחֵם / אֱלֹהַי מֵרֶחֶם / וּמִבֶּטֶן גֹּחִי



וְהִצִּיל שֶׂה תֹעָה / בְּחַסְדּוֹ מֵרָעָה / וְעֵינִי מִדִּמְעָה / וְרַגְלִי מִדֶּחִי



וְהוּא יִתְמֹךְ יָדִי / וְיַכִּיר מַעְבָּדִי / וְיָכִין מִצְעָדִי / כְּמוֹ זֵרָה אָרְחִי



בְּלִבִּי אֵשׁ תִּיקַד / לְעֹל פֶּשַׁע נִשְׂקַד / וּבַצַּוָּאר נֶעְקַד / כְּהֵעָקֵד מִזְחִי



וְלִי רַעְיוֹן נִדְבָּר / יְעִירֵנִי דָבָר / לְךָ לֵבָב נִשְׁבָּר / וְנַפְשִׁי הִוָּכְחִי



וְשׁוּבִי מִגְּאוֹנֵךְ / וּמַחְשֶׁבֶת אוֹנֵךְ / וְהַטִּי אֶת-אָזְנֵךְ / וְעֵינַיִךְ פִּקְחִי



שְׁמֹר מֵחֵטְא מַשְׂכִּיל / אֲשֶׁר עָשֹׂה יַסְכִּיל / אֱלִי אִם-לֹא אַשְׂכִּיל / וְאִם-אַשְׂכִּיל פִּצְחִי



הֲלֹא נֵפֶל אֵשֶׁת / מְזֹרָה הָרֶשֶׁת / וְלָמָה עוֹד בּשֶׁת / עֲלוּמִים תִּשְׁכָּחִי



וְאַתָּה מִשְׁתּוֹלֵל / וּבָרָע מִסְתּוֹלֵל / אֱמֹר אֵיךְ תִּתְהַלֵּל / וְתֹאמַר זַךְ לִקְחִי



וְהִסְכַּנְתָּ הַסְכֵּן / לְחַשֵּׁב רַע מִכֵּן / וְהִנֵּה אַךְ עַל-כֵּן / סְבִיבֹתַי פַּחִי



לְךָ יָשֵׁן עוּרָה / וּמִשַּׁדַּי גּוּרָה / וּמִתֵּבֵל סוּרָה / וְאִם-תָּמוּת תֶּחִי



תְּפָרֶשׂ-לָךְ כַּפָּהּ / וְיוֹם תַּקְשֶׁה עָרְפָּהּ / וְתִלְטשׁ עַפְעַפָּהּ / תְּהִי מָאוֹס וּסְחִי



לְזוּת שָׂפָה הָסֵר / וְאוּלַי תִּוָּסֵר / וּמָה-אַתָּה חָסֵר / וְדֶרֶךְ אֵל נִכְחִי



וְדַרְכֵי אֵל שָׁמְרֵם / וּמִצְוֹתָיו זָכְרֵם / וְעַל-אֶצְבַּע קָשְׁרֵם / וְכָתְבֵם עַל-לוּחִי



וְשִׁפְלֵי כָל-הַיְקוּם / בְּסוֹדָם אַל-תָּקוּם / וְהַשְׁקֵט אִם-יָקוּם / בְּךָ כָּל-פִּרְחָחִי



הֱשִׁיבַנִי עַל-זֹאת / וְעָנָנִי עַזּוֹת / אֲחָזוּנִי רְגָזוֹת / וְנָחַר מַפֻּחִי



בְּקִרְבִּי חִיל כָּבֵד / יְפַלַּח הַכָּבֵד / וְתִקְוָתִי תֹאבֵד / וְלֹא-אָבַד נִצְחִי



אֲנַחְנוּ עִם-אַנְשֵׁי / זְמַנִּי הַקָּשֶׁה / כְּמִצְרִים עִם-משֶׁה / וְכַשְׂדִּים עִם-תַּרְחִי



תְּמוּנוֹת נִמְשָׁלוֹת / וְלִבּוֹת נִבְדָּלוֹת / וְאֵינָם נִקְהָלוֹת / כְּשֶׁלֶג עִם-אָחִי



הֲתִהְיֶה הַבְּרָכָה / לְעוֹלָם כִּבְרָכָה / וְאִם-צֶלֶם מִיכָה / כְּצַלְמוֹן הָאחֹחִי



וְאִם-יִהְיֶה עַמְרָם / כְּמוֹ מִשְׁפַּחַת רָם / וְאִם-יִהְיֶה אַבְרָם / אֲבִירָם הָקָּרְחִי



וְאִם-יִהְיֶה הֵידָד / אֲשֶׁר בּוֹ יוֹד כַּהְדַד / בְּלִי יוֹד אוֹ אֶלְדָּד / כְּבִלְדַּד הַשּׁוּחִי



וְחָלִילָה לִהְיוֹת / שְׁוָרִים כָּארָיוֹת / וְאֵיזֶה קַל חַיּוֹת / אֲשֶׁר יִצְעַד פִּתְחִי



מְיֻדָּעֵי זִמְרִי / וְרֵעֵי בֶן-עָמְרִי / וְשׂנְאֵי-בֶן-אוּרִי / וְאֵיתָן אֶזְרָחִי



וְלוּ יֵשׁ מִשְׁאֶלֶת / וּמָצְאוּ קֹהֶלֶת / לְחָצוּהוּ דֶלֶת / וְהִכּוּהוּ לֶחִי



בְּרִיתָם לֹא יִמּוֹט / לְשׂוֹם שֶׂכֶל שָׁמוֹט / וְעַל-כֵּן בָּם אָקוֹט / וְלֹא-אֶשְׁכַּח שַׁחִי



וּבִרְאוֹתָם אָחוּשׁ / לְקַלֵּל הַיּוֹם חוּשׁ / וְלוּ בְשָׂרִי נָחוּשׁ / וְכָאבָנִים כֹּחִי



נְפָשׁוֹת נִדְאָבוֹת / וְגַבּוֹתָם עָבוֹת / כְּגַלֵּי מַצֵּבוֹת / וְשִׂפְתֵי נַפְתֻּחִי



וְגָבְהָם כַּתִּדְהָר / וְלִבָּם אָץ נִמְהָר / כְּעַתּוּדֵי הָהָר / וְקֹטְפֵי מַלֻּחִי



כְּחֹסֶן אַלּוֹנִים / כְּקוֹמַת עַרְמוֹנִים / כְּסִירֵי סַלּוֹנִים / כְּקִמֹּשֵׂי חוֹחִי



יְשִׁיבוּן אֶת-מִלַּי / וְיָסֹבּוּ עָלַי / עֲדֵי אַשְׁלִיך כֵּלַי / וְאַשְׁלִיךְ תּוֹתָחִי



פְּנֵיהֶם נִפְעַמְתִּי / וְדוּמָם נֶאְלַמְתִּי / וְלוּ אֶרְצֶה שַׂמְתִּי / בְּאַפֵּיהֶם חַחִי



וְרִפִּיתִי קָשְׁיָם / וְרִפִּאתִי חָלְיָם / אֲבָל שַׂמְתִּים כַּיָּם / וְסִפְרִי מִזְרָחִי



הֲכִי אֵין דּוּמִיָּה / לְנֶפֶשׁ הֹמִיָּה / וְקַצְוֵי תוּשִׁיָּה / יִגֹרֵמוֹ חֹחִי



וְאָנִיף יָד רָמָה / בְּדַעַת וּמְזִמָּה / וְצִיץ נֵזֶר חָכְמָה / אֲשַׁוֶּה עַל-מִצְחִי



וְאֵלֶיהָ אֶתְאָב / וּמֵימֶיהָ אֶשְׁאָב / וְלַבִּין אֶקְרָא אָב / וְלַמּוּסָר אָחִי



וְדַרְכֵי בִין אֶצֹּר / וּבַשֵּׂכֶל אֶעְצֹר / וּמִמֶּנוּ אֶקְצֹר / שְׁחִיסִי וּסְפִיחִי



וְהוּא גַּן פַּרְדֵּסִי / וְיֵינִי וַעְסִיסִי / וְכָפְרִי וַהְדַסִּי / וְרֵיחַ נִיחֹחִי.
