


	
	אֱלֹהִים אִלַּפְתָּ
	








אֱלֹהִים אִלַּפְתָּ סְגֻלַּת שְׁבָטֶיךָ / אֵל צַדִּיק וְחֶסֶד מַעֲטֶךָ

צַדִּיק אַתָּה יְיָ וְיָשָׁר מִשְׁפָּטֶיךָ





אֱלֹהִים נִצַּבְתִּי נֶגְדְּךָ עוֹדְדֵנִי / נָאוֹר קוּמָה וְהָגֵן בַּעֲדֵנִי

נִדְבוֹת פִּי רְצֵה נָא יְיָ וּמִשְׁפָּטֶיךָ לַמְּדֵנִי





אֱלֹהִים יִקַּרְתָּ עַם זוּ לְהַפְאִיר עֶרְכּוֹ / יִסַּדְתָּ יוֹם זֶה לְהָאִיר חָשְׁכּוֹ

יַדְרֵךְ עֲנָוִים בַּמִּשְׁפָּט וִילַמֵּד עֲנָוִים דַּרְכּוֹ





אֱלֹהִים שָׁבֵי פְשָׁעִים בְּצִלְּךָ יָלִינוּ / לִשְׁמֹעַ סְלִיחָה אָזְנָם יָכִינוּ

מָרוֹם הַשְׁמִיעֵם הַיּוֹם אִם יִהְיוּ חֲטָאֵיכֶם כַּשָּׁנִים כַּשֶּׁלֶג יַלְבִּינוּ





אֱלֹהִים שׁוֹכֵן בְּאַפַּדְנֵי גְבוֹהַי / שְׁעֵנִי בְּקָרְאִי בְּעַד כְּמֵהַי

שָׁפְטֵנִי כְצִדְקְךָ יְיָ אֱלֹהַי





אֱלֹהִים לְשִׁמְךָ נוֹרָא וְאָיֹם / לְהוּקֶיךָ נִצְּבוּ לְבַקֵּשׁ פִּדְיוֹם

לְמִשְׁפָּטֶיךָ עָמְדוּ הַיּוֹם





אֱלֹהִים מַעַן לְשׁוֹנִי תִרְצֶה / מַגֵּן כַּחְשִׁי וְנַפְשִׁי תִפְצֶה

מִלְּפָנֶיךָ מִשְׁפָּטִי יֵצֵא





אֱלֹהִים הֲכִינוֹתָ לְמִקְלָטִי וּמִפְלָטִי / הַעֲבֵר מַעְבָּטִי וְהָבֵר מַבָּטִי

הָעִירָה וְהָקִיצָה לְמִשְׁפָּטִי





אֱלֹהִים בְּשִׁבְתְּךָ בְּכִסֵּא מְרוֹם עֶרֶץ / בְּהַעֲלוֹתְךָ יְצוּרִים בְּמֹאזְנֵי תֶרֶץ

כִּי בָא לִשְׁפֹּט הָאָרֶץ





אֱלֹהִים נְצֹר חֶסֶד לְאֹם עֲמוּסָה / נָא אַל תַּחְקֹר חַטָּאת כְּמוּסָה

נוֹדַע יְיָ מִשְׁפָּט עָשָׂה





אֱלֹהִים יָעַדְתָּ כֹּפֶר לְחַזֵּק כָּל בֶּדֶק / יִשַּׁרְתָּ מַסְלוּל לְיַשֵּׁר כָּל חֵדֶק

יָשַׁבְתָּ לְכִסֵּא שׁוֹפֵט צֶדֶק





אֱלֹהִים הַיּוֹם יְעוֹדֵד מְעוּטָיו / הַצֵּל יַצִּיל שְׁבָטָיו מִשְּׁפָטָיו

הוּא יְיָ אֱלֹהֵינוּ בְּכָל הָאָרֶץ מִשְׁפָּטָיו





אֱלֹהִים וַעֵד נֶגְדִּי חַסְדֶּךָ / וְגוֹנְנִי הַיּוֹם בְּעָמְדִי נֶגְדֶּךָ

וְאַל תָּבוֹא בְמִשְׁפָּט אֶת עַבְדֶּךָ





אֱלֹהִים דּוֹרֵשׁ לְבָבוֹת וְרַעְיוֹנִים / דְּלֵה הַיּוֹם מֵעָווֹן אֱמוּנִים

דִּין עָנִי מִשְׁפַּט אֶבְיוֹנִים





אֱלֹהִים הֲכִינוֹתָ יוֹם זֶה לְיַשֵּׁר כָּל עָקֹב / הֵן רְשַׁמְתּוֹ לְחֹק נָקוּב

הוּא נִשְׁפָּט לֵאלֹהֵי יַעֲקֹב





אֱלֹהִים גּוֹנְנִי מִתִּגְרַת שְׁפוּטֶיךָ / כַּל עֵינַי וְאַבִּיטָה נִפְלָאוֹת מִקָּשְׁטֶךָ

גָרְסָה נַפְשִׁי לְתַאֲוָה אֶל מִשְׁפָּטֶיךָ





אֱלֹהִים בָּאתִי לְסַלְסֵל חֻקֶּיךָ / בְּקָרְאִי יִפְגַּע שַׁוְעִי חִשְׁקֶךָ

בְּלָמְדִי מִשְׁפְּטֵי צִדְקֶךָ





אֱלֹהִים יָעַד יוֹם זֶה לִשְׁפֹּט מִבְרָאוֹ / יוֹם לְהַרְאוֹת לַכֹּל מִפְלָאוֹ

יֵשֵׁב כּוֹנֵן לַמִּשְׁפָּט כִּסְאוֹ





אֱלֹהִים רַחְשִׁי עֲרֹב וַעֲנֵנִי / רוֹחֵף מִפַּחְדְּךָ הִנֵּנִי

רַחֲמֶיךָ רַבִּים יְיָ כְּמִשְׁפָּטֶיךָ חַיֵּנִי





אֱלֹהִים וַיְיָ יְאַמֵּץ כָּל בֶּדֶק / וְיוֹפִיעַ לִלְקֹט שׁוֹשָׁן מֵחֵדֶק

וְהוּא יִשְׁפֹּט תֵּבֵל בְּצֶדֶק





אֱלֹהִים לִפְצוֹת שְׁאֵרִית פְּלִיטֶיךָ / לְחַלְּצֵם מֵרִשְׁפֵי לְהָטֶיךָ

לְמַעַן תִּצְדַּק בְּדָבְרֶךָ תִּזְכֶּה בְשָׁפְטֶךָ





אֱלֹהִים חֶבְיוֹן לְבָבִי אָעִירָה / חֹסֶן גְּדֻלָּתְךָ אַאֲמִירָה

חֶסֶד וּמִשְׁפָּט אָשִׁירָה





אֱלֹהִים זָכְרֵנִי מִשְׂגַּב דַּךְ נִכְלָם / זַכֵּה לְבָבִי מֵעָווֹן הַנֶּעְלָם

זָכַרְתִּי מִשְׁפָּטֶיךָ מֵעוֹלָם





אֱלֹהִים קְלֹט שִׁבְיְךָ מִקֶּרֶץ / קָמִים לְהוֹדוֹת לְךָ בְּמֶרֶץ

קוּמָה אֱלֹהִים שָׁפְטָה הָאָרֶץ





אֱלֹהִים יְנַתֵּק חַבְלֵי זָדוֹן מְקֻשָּׁרִים / קְהַל מֵישָׁרִים לָצֵאת מְאֻשָּׁרִים

וְהוּא יִשְׁפֹּט תֵּבֵל בְּצֶדֶק יָדִין לְאֻמִּים בְּמֵישָׁרִים.
