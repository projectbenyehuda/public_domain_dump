


	
	שׁוֹכֵן עַד מֵאָז נִשְׂגָּב
	








מַלְכוּתְךָ מַלְכוּת כָּל עוֹלָמִים

שׁוֹכֵן עַד מֵאָז נִשְׂגָּב לְבַדּוֹ / שֵׁם מַלְכוּתוֹ מְיֻחָד וְאֵין שֵׁנִי לְצִדּוֹ

שִׁכְלֵל עוֹלָמוֹ מֵאוֹר לְבוּשׁ מַדּוֹ / בִּשְׁלֹשָׁה סְפָרִים נֶחְתָּמִים

לַעֲצַת אָמוֹן נִכְסַף נִכְסוֹף / לַהֲקַת עֶשֶׂר סְפִירוֹת יָזַם לַחְשֹׂף

לִבְלֵר כְּנֶגְדָּם עֲשָׂרָה בְאֵין סוֹף / וְחָמֵשׁ בְּחָמֵשׁ מְהֻסְכָּמִים

מֵבִין סוֹדָם יִבָּהֵל וְיִפְחָד / מֵהֶם יַשְׂכִּיל כִּי הַיּוֹצֵר מְיֻחָד

מָה אַתָּה סוֹפֵר לִפְנֵי אֶחָד / וְהוּא רֹאשׁ לְכָל רֹאשׁ וְרָם עַל כָּל רָמִים

הֲכִי עֶשֶׂר הֵם אֲחוּזוֹת בְּמָצוֹר / הַמַּשְׂכִּיל בָּהֶם יֵדַע וְיָצוֹר

הוּא הַיּוֹצֵר אֲשְֶׁר בָּם יַעֲצֹר / וְעֵדָיו גְּלוּיִים וּמְפֻרְסָמִים

הֵן בְּאוֹתִיּוֹת עֶשְׂרִים וּשְׁתַּיִם / הִגְבִּיל אֵשׁ לְמַעְלָה וּלְמַטָּה מַיִם

הִבְּדִילָם בְּרוּחַ חֹק מַכְרִיעַ בֵּינְתַיִם / וּשְׁנֵים עָשָׂר מַזָּלוֹת בַּמְּרוֹמִים

קִיַּם מֵאַיִן יֵשׁ וְיָצַר מֶמֶשׁ מִתֹּהוּ / קִיֵּם עַמּוּדִים גְּדוֹלִים מֵאֵין כָּמוֹהוּ

קָבַע בְּתוֹךְ קַו יָרֹק אַבְנֵי בֹהוּ / וּמֵהֶם יוֹצְאִים מֵימֵי תְהוֹמִים

טָבַע שֵׁשׁ קְצָווֹת חֲתוּמִים בִּשְׁמוֹ / טִלְטֵל אֵשׁ מִמַּיִם בְּתָקְפּוֹ וְעַצְמוֹ

טִכֵּס בָּם כִּסְאוֹ וּצְבָא מְרוֹמוֹ / לְאוֹתוֹת וּלְמוֹעֲדִים וּלְיָמִים

נִשָּׂא עַל כֹּל שְׁמוֹ אֲשֶׁר שָׁת בַּכֹּל / נִתְלָה עוֹלָם בְּיָדוֹ כְאֶשְׁכֹּל

נִשְׂגָּב מִמָּקוֹם וְהוּא מָקוֹם לַכֹּל כִּי בְיָהּ יְיָ צוּר עוֹלָמִים

בְּרוּחַ מָרוֹם אֲשֶׁר שָׁת כִּסְאוֹ נָכוֹן / בּוֹ כְּבוֹד מַלְכוּתוֹ מֵאָז יִשְׁכֹּן

בְּכֵן מוֹשֵׁל בַּכֹּל וְתִיכוֹן / הוּא רוּחַ אֱלֹהִים חַי הָעוֹלָמִים

רָם עַל כֹּל וּמֵהַכֹּל תַּקִּיף / רוֹאֶה הַכֹּל וְעַל הַכֹּל מַשְׁקִיף

רוֹדֶה בַכֹּל וְעַל הַכֹּל מַקִּיף / וְכָל הַיְּצוּרִים בִּשְׁמוֹ קַיָּמִים

יוֹצֵר הַכֹּל בְּמַאֲמָר בְּלִי פֶגַע / יָחִיד מַנְהִיג הַכֹּל כְּרֶגַע

יָהּ סוֹבֵל הַכֹּל בְּלִי יֶגַע / לוֹכֵד בְּעָרְמָם חֲכָמִים

הוּא הַמְּגַלְגֵּל אֲפֻדַּת שְׁחָקִים / הוּא הַתּוֹלֶה אֲגֻדַּת אֲרָקִים

הוּא הָאוֹמֵר יְהִי כֵן וְיָקִים / וְיוֹצִיא לָאוֹר כָּל תַּעֲלוּמִים

וְהוּא אֲשֶׁר יַקְהִיל וְיַפְרִישׁ / וְהוּא אֲשֶׁר יַעֲשִׁיר וְיוֹרִישׁ

וְהוּא אֲשֶׁר יְפַצְפֵּץ וְיַקְרִישׁ / וְיַחְבִּיר תְּמוּנוֹת לַגְּלָמִים

דְּעוּ כִּי הוּא אֲשֶׁר יָאִיר וְיַאֲפִיל / וְהוּא אֲשֶׁר יָרִים וְיַשְׁפִּיל

וְהוּא אֲשֶׁר יִסְמֹךְ וְיַפִּיל / כָּל הֶהָרִים הָרָמִים

הוּא אֲשֶׁר יְמַלֵּא לַיְּצוּרִים סִפְקָם / הוּא אֲשֶׁר יַטְרִיפֵם לֶחֶם חֻקָּם

הוּא אֲשֶׁר יִתֵּן מַאֲכָלָם וּמַשְׁקָם / הוּא אֲשֶׁר יוֹרִיד הַגְּשָׁמִים

הוּא אֲשֶׁר יחַיֶּה מְתֵי חֶלֶד / הוּא אֲשֶׁר יְכוֹנֵן גּוּף הַיֶּלֶד

הוּא אֲשֶׁר יִקְרֹם הַגֶּלֶד / הוּא אֲשֶׁר יְסוֹכֵךְ הָעֲצָמִים

הוּא אֲשֶׁר יִפַּח בַּגּוּף נְשָׁמָה / הוּא אֲשֶׁר יְכוֹנְנֵהוּ בְעָצְמָה

הוּא אֲשֶׁר יְשִׁיבֵהוּ לָאֲדָמָה / הוּא אֲשֶׁר יְעוֹרֵר נִרְדָּמִים.
