


	
	שַׁחַק כַּדֹּק
	








יְיָ אֱלֹהַי גָּדַלְתָּ מְאֹד / הוֹד וְהָדָר לָבַשְׁתָּ





– – – – – – – –



שַׁחַק כַּדֹּק נָטִיתָ / אוֹר כַּשַּׂלְמָה עָטִיתָ

בִּשְׂפָתֶיךָ בִּטִּיתָ / מֵחֶפְצְךָ לֹא שָׂטִיתָ / כִּי הוֹד כִּפְאֵר חָבַשְׁתָּ





לְךָ יְרַנְנוּ תָבוֹר וְחֶרְמוֹן / שַׁדַּי הַמַּשְׁלִיג בְּצַלְמוֹן

יוֹם צָעַדְתָּ בִישִׁימוֹן / כִּי אַתְּ קַדְמוֹן לְכָל קַדְמוֹן / וְהֵמָּה חֲדָשִׁים חִדַּשְׁתָּ



מוֹסְדוֹת תֵּבֵל עַל בְּלִימָה / תָּלִיתָ בְּעֹז וּבְעָצְמָה

וְיָשְׁבָה תַחְתֶּיהָ וְרָאֲמָה / אַךְ בַּעֲלוֹת בְּאַפְּךָ חֵמָה / עַמּוּדֶיהָ הִרְעַשְׁתָּ



הִרְבֵּיתָ חֵיל כּוֹכָבִים / הָאִיר לְתֵבֵל וְלַיּוֹשְׁבִים

בָּהּ לֹא מְרֵעִים וְלֹא מְטִיבִים / וְעֵת תְּכַס שָׁמַיִם בְּעָבִים / פְּנֵי הוֹבְרֵיהֶם הוֹבַשְׁתָּ



הֵן עַם בָּנֶיךָ קְרָאתָם / יֵין תַּרְעֵלָה הִשְׁקִיתָם

וְצֹאן מַרְעִיתְךָ כִּנִּיתָם / אֵיךְ לְכָל רוּחַ זֵרִיתָם / עַל מִי צֹאנְךָ נָטַשְׁתָּ



חֶזְקַת יָדְךָ תּוֹדִיעַ / וּשְׁאוֹן זָרִים תַּכְנִיעַ

וְאֶת עַם עָנִי תוֹשִׁיעַ / יוֹם מִצִּיּוֹן תּוֹפִיעַ / וְהִתְגַּדִּלְתָּ וְהִתְקַדִּשְׁתָּ.
