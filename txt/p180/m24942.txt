


	
	שִׁנְאַנֵּי אֵלִים
	








שִׁנְאַנֵּי אֵלִים / חֲצוּבֵי לֶהָבָה

לָז לַלָּז שׁוֹאֲלִים / אַיֵּה מְקוֹם הַמֶּרְכָּבָה

מַעֲרִיצִים וּמְהַלְלִים / דָּגוּל מֵרְבָבָה





שְׂרָפִים / אוֹזְרֵי עֹז / עֲטוּיֵי הוֹד כְּלִילוֹ

עוֹמְדִים / בְּכֵם מְפֻרְשָׁז / בַּחֲתֻלַּת זְבוּלוֹ

מִמַּעַל / גַּלְגַּל / וּמִתַּחַת גְּלִילוֹ

לוֹ / דֶגֶל יִשְׂאוּ / וְיָסֹבּוּ לְהַלְלוֹ

שְׂרָפִים עוֹמְדִים מִמַּעַל לוֹ





שֵׁשׁ כְּנָפַיִם / הֵם / מַרְעִידִים בְּפַחַד

שֵׁשׁ / וְאַרְבַּע פְּרוּדוֹת / לְאַרְבַּע חַיּוֹת יַחַד

כְּנָפַיִם / זְקוּפוֹת / וּפְנֵי חַד אֶל חַד

לְאֶחָד / חוֹקְרִים שִׁיר / וְזֶמֶר בְּלִי כַחַד

שֵׁשׁ כְּנָפַיִם שֵׁשׁ כְּנָפַיִם לְאֶחָד





בִּשְׁתַּיִם / טָסִים / סְבִיבוֹת כֵּס מְכוֹנָיו

יְכַסֶּה / יְקַר עֻזּוֹ / מִצִּבְאוֹת הֲמוֹנָיו

פָּנָיו / כְּמוֹ יְבַקְּשׁוּ / נִסְתָּר בְּחֶבְיוֹנָיו

לַהַט יְקוֹד פַּרְגּוֹד / בֵּינֵימוֹ וּבֵינָיו

בִּשְׁתַּיִם יְכַסֶּה פָנָיו





וּבִשְׁתַּיִם / מַחֲנוֹת / מִיָּמִין מְהַלְלָיו

יְכַסֶּה / נֹגַהּ שְׂמֹאל / בִּשְׁתֵּי מַחֲנוֹת זְבוּלָיו

רַגְלָיו / סְבִיבֵי זִיו / נִיצוֹצֵי חֲיָלָיו

עֶרְכָּם כִּצְפוֹת בָּזָק / רָצִים וְשָׁבִים אֵלָיו

וּבִשְׁתַּיִם יְכַסֶּה רַגְלָיו





וּבִשְׁתַּיִם / פֹּארוֹת / זִיו הוֹלֵךְ וְטוֹפֵף

יְעוֹפֵף / צִיץ כִּסֵּא / וְעָלֵימוֹ יְחוֹפֵף

קֶרֶן עֱזוּז חֶבְיוֹן / סְבִיבוֹתָם יְנוֹפֵף

רִפְיוֹנָם צָפוּי / נְסוֹגִים לְהִסְתּוֹפֵף

וּבִשְׁתַּיִם יְעוֹפֵף





וְקָרָא זֶה / שִׁנְאָן / לְמוּל שִׁנְאָן בְּמִשְׁמָר

אֶל / תַּקִּיף יְרוּצוּן / בְּכָל מַהֲלָל וּזְמָר

זֶה / שָׁלֹשׁ וּשְׁתַּיִם / וְזֶה אֶחָד בְּמִגְמָר

וְאָמַר / לָז שֵׁנִי / לְלָז חֹק לֹא יוּמָר

וְקָרָא זֶה אֶל זֶה וְאָמָר





קָדוֹשׁ / מֵאַרְבַּע / חַיּוֹת נִשָּׂאוֹת

קָדוֹשׁ / הוּא בְּאַרְבַּע / מַחֲנוֹת נוֹרָאוֹת

קָדוֹשׁ / בְּאוֹפַנִּים / עֵינַיִם מְלֵאוֹת

יְיָ צְבָאוֹת / רָם / בְּסוֹד שַׂרְפֵי פְלָאוֹת

קָדוֹשׁ קָדוֹשׁ קָדוֹשׁ יְיָ צְבָאוֹת





מְלֹא / יְצוּקֵי חֲצָצִים / וְחַשְׁמַלֵּי גְדוּדוֹ

כָּל / הַיּוֹם יְרוּצוּן / זֶמֶר לְהַעֲנִידוֹ

הָאָרֶץ / וְחֵילָהּ / הַתְּלוּיִים בְּיָדוֹ

כְּבוֹדוֹ / דוֹבְבִים כֵּן / הֲגוֹת הַכֹּל בְּיָדוֹ

מְלֹא כָּל הָאָרֶץ כְּבוֹדוֹ.
