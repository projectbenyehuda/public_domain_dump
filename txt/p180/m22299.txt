


	
	לִבִּי בְּאֵימַת אַהֲבָה
	




לִבִּי בְּאֵימַת אַהֲבָה גוֹוֵעַ / כִּי הַזְּמַן אַבְנֵי נְדֹד קוֹלֵעַ



חֵילֵי שְׂעִפַּי אַחֲרֶיהָ נֶחְשְׁלוּ / וּזְמַן יְזַנֵּב וַאֲנִי יָגֵעַ



יִכְלוּ יְמֵי נֹעַר וְעַד הֵן אֶחֱשֹׁב / כִּי לֹא זְמַן יֵיטִיב וְלֹא
יָרֵעַ



אָקוּ בְּרַעְיוֹנַי וְאֶלְחַם בַּהֲמוֹן / יָגוֹן בְּלִבִּי וַאֲנִי
כוֹרֵעַ



וָאֹמְרָה: דַּי לִי הֲלוֹךְ עַל מַהְלְכֵי / יָמִים, וְכַמָּה אֶהְיֶה
פוֹרֵעַ



נִבְהָל בְּחֶטְאוֹ מֵעֲמֹד לִפְנֵי גְבִיר / וּבְיַם כְּסִילוּת מִתְּמוֹל
טוֹבֵעַ -



מִי זֶה מְהַלֵּךְ עַל כְּנַף לִבִּי, וְעֵץ / אַהַב בְּגַן הָאַהֲבָה
נוֹטֵעַ



מֵכִין בְּצַלְעוֹתַי מְכוֹן שִׁבְתּוֹ, וְעַל / לוּחוֹת לְבָבִי אָהֳלוֹ
תוֹקֵעַ



הוּא הַגְּבִיר יַעֲקֹב חֲסָדָיו גָּבְרוּ / מֵרֹאשׁ וְעַד סוֹף עַל יְדִיד
וָרֵעַ



אוֹר נַעֲרָץ יָאִיר פְּנֵי כָל אוֹהֲבָיו / לַיִל וְיוֹמָם בַּל יְהִי
גּוֹרֵעַ



וּנְעִים זְמִירוֹת כָּל מְתֵי שִׁיר וַאֲדוֹן / הַשִּׁיר וּמִבְצַר
אוֹיְבִים קוֹרֵעַ



בָּאוּ לְךָ דּוֹדִי חֲרוּזִים נָעֲמוּ / יֵרַךְ לְבָבִי בָּם וְיִקָּרֵעַ



נַפְשִׁי פְּדוּת שִׁיר מַהֲלָל, מֵעֵין יְקָר / יוֹצֵא וּמִנַּחַל פְּאֵר
נוֹבֵעַ



יֵרֵד כְּטַל עַל גֵּז וְנַפְשׁוֹת אוֹהֲבָיו / מֵשִׁיב, וּמֵרֶדֶת שְׁאוֹל
פּוֹדֵעַ



קַלּוּ לְפָנָיו עִנְיְנֵי שִׁירַי, הֲכִי / רוּם מַעֲלוֹתָיו עַד זְבוּל
נוֹגֵעַ



וּבְמַחֲנַיִם פָּגְעוּ בוֹ מַלְאֲכֵי / הָאֵל, בְּטֶרֶם בִּי יְהִי
פוֹגֵעַ



פָּנִים בְּפָנִים עֵת חֲזִיתִיו נִצְּלָה / נַפְשִׁי, וְהָיִיתִי תְמוֹל
גוֹוֵעַ



אַךְ אַחֲרֵי עָבְרִי פְנוּאֵל זָרְחָה / שִׁמְשִׁי בְּפָנַי - וַאֲנִי
צוֹלֵעַ.
