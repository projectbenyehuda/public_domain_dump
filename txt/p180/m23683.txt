


	
	נִשְׁמַת שְׁאָר יַעֲקֹב
	








נִשְׁמַת שְׁאָר יַעֲקֹב אֲשֶׁר בְעֵדוּת נֶאֱמָנָה הַעַדְתּוֹ תְּשַׁבְּחָךְ

שׁוֹמֵר הַבְּרִית וְהַחֶסֶד, יוֹם לַמִּשְׁפָּט יְעַדְתּוֹ

יְיָ לְמִשְׁפָּט שַׂמְתּוֹ וְצוּר לְהוֹכִיחַ יְסַדְתּוֹ.





נִשְׁמַת לַהֲקַת יְשׁוּרוּן בְּכָל יוֹם מְיַחֲדֶיךָ תְּלַבְּבָךְ

לְהַצְדִּיקָם וּלְכָתְבָם לְחַיִּים בִּכְתָב יָדֶךָ

לְמִשְׁפָּטֶיךָ עָמְדוּ הַיּוֹם כִּי הַכֹּל עֲבָדֶיךָ.





נִשְׁמַת מֻפְרֶשֶׁת מִשִּׁבְעִים, נְכוֹחָה בְמֹאזְנֵי צֶדֶק תַּמְלִיכָךְ

מֶלֶךְ הָעוֹשֶׂה מִשְׁפָּט וָצֶדֶק

יָשַׁבְתָּ לְכִסֵּא שׁוֹפֵט צֶדֶק.





נִשְׁמַת הֲמוֹנֵי סְגֻלָּה דִּגְלֵי שְׁבָטֶיךָ תְהוֹדָךְ

הַפּוֹשֵׁט יָד לְקַבֵּל עוֹבְרֵי מִשְׁפָּטֶיךָ

לְמַעַן תִּצְדַּק בְּדָבְרֶךָ תִּזְכֶּה בְשָׁפְטֶךָ.





נִשְׁמַת חֲנִיטֵי יְשׁוּרוּן עֲבָדֶיךָ יְרֵאֶיךָ תְחַסְּנָךְ

זַךְ קָרוֹב לְכָל קוֹרְאֶיךָ

צֶדֶק וּמִשְׁפָּט מְכוֹן כִּסְאֶךָ.





נִשְׁמַת קְדוֹשִׁים קְדוּשָׁתְךָ עוֹנִים בְּכָל מַאֲוַיָּם תַּקְדִּישָׁךְ

קָדוֹשׁ אֵל מֶלֶךְ חַי וְקַיָּם

אוֹמְרִים וְאִילוּ פִינוּ מָלֵא שִׁירָה כַּיָּם.
