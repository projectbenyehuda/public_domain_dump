


	
	נַפְשִׁי דְּעִי
	








נַפְשִׁי דְּעִי מַה תִּפְעֲלִי / טֶרֶם לְצוּרֵךְ תַּעֲלִי

שׁוּבִי לְכִי מִמַּעֲלִי / חֶטְאֵךְ וְחוֹבַיִךְ גְּלִי

בּוֹאִי בְחַדְרֵךְ וּסְגוֹר / דַּלְתֵי פְשָׁעִים נַעֲלִי

תַּחַת כְּנַף צוּרֵךְ חֲבִי / עַד יַעֲבֹר עֵת זַחֲלִי.





הוֹדִי עֲלֵי פִשְׁעֵךְ לְצוּר / עֻזֵּךְ וְאָז תִּנָּצְלִי

וּרְאִי פְקֻדַּת רֹב עֲווֹן / וְעוֹד לֹא תִמְעֲלִי

קוּמִי בְעוֹד לַיִל וְאַל / תִּתְיַשְּׁנִי תִּתְעַצְּלִי

רֹנִּי וְשִׁירָה דַבְּרִי / הַלֵּל וְזִמְרָה מַלְּלִי.





לִפְנֵי אֱלֹהַיִךְ שְׁחִי / הִשְׁתַּטְּחִי הִתְנַפְּלִי

הִתְבּוֹנְנִי הִכּוֹנְנִי / הִתְרוֹנְנִי הִתְפַּלְּלִי

הִתְנַעֲרִי הִתְעוֹרְרִי / הִשְׂתָּעֲרִי הִתְחוֹלְלִי

צֶדֶק וּמִשְׁפָּט בַּקְּשִׁי / גַּם הָעֲנָוָה שַׁאֲלִי.





אוּלַי יְחָנֵּךְ צוּר וְלֹא / תֵרְדִי אֱלֵי שַׁחַת בְּלִי

לָמָּה יְחִידָתִי וְעַל / מַה זֶּה וְהֶבֶל תֶּהְבְּלִי

הִתְאַזְּרִי הִתְעַזְּרִי / הִנָּזְרִי הֵאָצְלִי

עִזְבִי הֲבֵל עוֹלָם וְלֹא / עַל רַהֲבוֹ תִּתְאַבְּלִי.





אִם תִּשְׁמְעִי מוּסָר וְגַם / בִּנְתִיב יְשָׁרִים תִּנְהֲלִי

טוּב הָאֲדָמָה תֹּאכְלִי / חַיִּים וְכָבוֹד תִּנְחֲלִי

מֵאֵת אֲדוֹנַיִךְ וְגַם / בּוֹ תִּצְדְּקִי תִּתְהַלְּלִי

אִם תִּמְאֲנִי תִּמָּנְעִי / מִטּוֹב וְרַע תִּתְעוֹלְלִי

נֶגַע וְקָלוֹן תִּמְצְאִי / חֶרְפָּה וּבוּז לֹא תֶחְדְּלִי.





זִכְרִי בְיוֹם הַדִּין וְאַל / תִּסְתַּכְּלִי וּתְהוֹלְלִי

עַל פָּעֳלֵךְ הַטּוֹב וְגַם / הָרַע פְּנֵי צוּר תִּשְׁאֲלִי

תִּתְפַּלְלִי תִּתְנַפְּלִי / וּבְתֹף לְפָנָיו חַלְּלִי

אִם תִּפְעֲלִי טוֹב לָךְ וְלִי / אַל תֵּלְכִי בֵית הָאֱלִי



[עַד] בֵּית אֱלֹהַי אֶהֱלִי / תַּעְדִּי חֳלִי תַּעֲדִּי חֲלִי.
