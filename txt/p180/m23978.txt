


	
	מָעוֹן אַדִּיר
	








מָעוֹן אַדִּיר מֵאָז רִאשׁוֹן / נִסְתָּר מֵרְאוּת אִישׁוֹן

יוֹדוּךָ כָּל פֶּה וְלָשׁוֹן / יְיָ אֱלֹהֵי הַשָּׁמַיִם





מָעוֹן נִמְצָא בַכֹּל דַּרְכֶּךָ / וְנִפְלָא לִמְצֹא עֶרְכֶּךָ

מְאַחֵז שַׁפְרִיר סֻכֶּךָ / רוּמָה עַל הַשָּׁמַיִם





מָעוֹן יוֹשֵׁב קֶדֶם דֹּק הֶעֱמַדְתָּ / וְגָבְהֵי שָׁמַיִם רִפְסַדְתָּ

לְפָנִים הָאָרֶץ יָסַדְתָּ / וּמַעֲשֵׂה יָדֶיךָ שָׁמַיִם





מָעוֹן שִׁפְרֵר זְרוּתֵי מַעַל / וּמָדַד מַיִם בַּשֹּׁעַל

בְּמַאֲמָר וְלֹא בְּפֹעַל / עוֹשֶׂה כֹּל נוֹטֶה שָׁמַיִם





מָעוֹן לְשִׁמְךָ הַכֹּל יְהוֹדוּן / וְגֹדֶל זְרוֹעֲךָ יַגִּידוּן

שְׁחָקִים וַאֲרָקִים יְעִידוּן / וּצְבָא הַשָּׁמַיִם





מָעוֹן מֵאֵין לְבָבוֹת יְכִילוּךָ / לְמִי יְדַמּוּךָ וְיַמְשִׁילוּךָ

וְהִנֵּה לֹא יְכַלְכְּלוּךָ / הַשָּׁמַיִם וּשְׁמֵי הַשָּׁמַיִם





מָעוֹן הִנֵּה נִקְרָֻא שְׁמֶךָ / וְהַכֹּל נִהְיָה מִמֶּךָ

וְאֵי זֶה יִהְיֶה מְקוֹמֶךָ / הֲלֹא אֱלוֹהַּ גֹּבַהּ שָׁמַיִם





מָעוֹן הֲלֹא מִנִּי עַד תָּקַפְתָּ / וְהַכֹּל מֵאָז הִקַּפְתָּ

וּבְעֵת אֲשֶׁר הִשְׁקַפְתָּ / אֶרֶץ רָעֲשָׁה גַּם שָׁמַיִם





מָעוֹן קוֹרֵא הַכֹּל מִמְּךָ / וְהַכֹּל שָׁר תַּעֲצוּמְךָ

יְיָ אֲדוֹנֵינוּ מָה אַדִּיר שִׁמְךָ / אֲשֶׁר תְּנָה הוֹדְךָ עַל הַשָּׁמַיִם





מָעוֹן טְפוּחֵי חַיּוֹת טִכֵּס / וְאַחַת לְאַחַת רִכֵּס

וְנִרְאִים כְּנוֹשְׂאֵי הַכֵּס / מֵעַל הַשָּׁמַיִם





מָעוֹן נִפְלֵאתָ מְאֹד וְנִשְׂגַּבְתָּ / מִשְׁכַּן שְׁחָקִים רָכַבְתָּ

וְעֻזְּךָ עֲלֵיהֶם הִרְכַּבְתָּ / אֵין כָּאֵל יְשׁוּרוּן רוֹכֵב שָׁמַיִם





מָעוֹן בְּגַאֲוָה וָעֹז נֶאְדַּרְתָּ / גַּם בַּעֲנָוָה נֶהְדַּרְתָּ

כִּי עַל כְּנַף רוּחַ גַּרְתָּ / וֵאלֹהֵינוּ בַּשָּׁמַיִם





מָעוֹן רָם עַל כֹּל וּמִכֹּל נִפְרַדְתָּ / וְעַל הַר סִינַי יָרַדְתָּ

יוֹם אֲשֶׁר הֶחֱרַדְתָּ / מוֹסְדוֹת הַשָּׁמַיִם





מָעוֹן יָחִיד לוֹ הָעֹז נִתְכָּן / וּבְרֹב עֲנָוָה נִסְכָּן

לִשְׁכֹּן בְּאֹהֶל וּמִשְׁכָּן / הוֹדוֹ עַל אֶרֶץ וְשָׁמַיִם





מָעוֹן הָדוּר אֲשֶׁר לוֹ נִתְוֵַיתָ / וְצִיּוֹן לְמוֹשָׁב אִוִּיתָ

וְכִסֵּא לְכִסֵּא הִשְׁוֵיתָ / הַנּוֹטֶה כַדֹּק שָׁמַיִם





מָעוֹן וְעַל כָּל אֱלֹהִים נַעֲלֵיתָ / וְנוֹרָאוֹת מְאֹד נִפְלֵיתָ

וְהַכֹּל עֻזְךָ מִלֵּאתָ / מִקְצֵה הַשָּׁמַיִם וְעַד קְצֵה הַשָּׁמַיִם





מָעוֹן דּוֹרְשֶׁיךָ עֵת קְרָאוּךָ / הִנֵּה לְנֶגְדָּם מְצָאוּךָ

עַל כֵּן עַמִּים יְרֵאוּךָ / תַּחַת כָּל הַשָּׁמַיִם





מָעוֹן הִנֵּה בְּמָעוֹן יָשַׁבְתָּ / וּמָעוֹן לַכֹּל נֶחְשַׁבְתָּ

מִמְּעוֹן קָדְשְׁךָ הִקְשַׁבְתָּ / וְאַתָּה תִּשְׁמַע הַשָּׁמַיִם.
