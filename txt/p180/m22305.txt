


	
	לִבִּי שְׂמַח
	




לִבִּי שְׂמַח / וּבְשִׁיר פְּצַח / דָּבְרֶךָ



בֵּית גַּן פְּתַח / סֹב נָא וְקַח / מֹרֶךָ



וֶרֶד קְטֹף / שׁוֹשָׁן חֲטֹף / וַהְדַס טְרֹף



זַמֵּר בְּתֹף / עֵת יַעֲרֹף / אוֹרֶךָ



וּבְיַד תְּפֹשׂ / מִזְרָק וְכוֹס / וּמְנָת וְכוֹס



קַח דּוֹד, וְשׂוֹשׂ / תָּשִׂישׂ בְּכֹס / שָׂרֶיךָ



שָׂשׂוֹן דְּרֹשׁ / יָגוֹן גְּרֹשׁ / גִּיל תַּחֲרֹשׁ



וּכְמוֹ בְרוֹשׁ / תִּהְיֶה לְרֹאשׁ / מַמְרֶיךָ



רֹן תַּעֲנֶה / וּבְמַחֲנֵה / בִין תַּחֲנֶה



חָכְמָה קְנֵה / תָּאִיר פְּנֵי / חוֹרֶךָ



תָּשִׁיר בְּשִׁיר / אֶל הַגְּבִיר / אִישׁ תָּם בְּחִיר



וָשֵׁם וְנִיר / לוֹ מֵאֲבִיר / צוּרֶךָ



כִּי עַל שְׁמוֹ / הוֹד טַעֲמוֹ / זִיו נָעֳמוֹ



צוּף טַעֲמוֹ / עָרֵב כְּמוֹ / יַעְרֶךָ



מַה נֶהֱדָר, / נָקִי וּבַר / לוֹ מַאֲמָר



רֹאשׁ לַיְּקָר / קָצִין וְשַׂר / דּוֹרֶךָ



שָׂם גִּיל לְגִיל / מַחְכִּים כְּסִיל / גַּם כִּכְסִיל



עַל הַגְּלִיל / יָאִיר שְׁבִיל / שִׁירֶךָ



עֵץ רַעֲנָן / צִיץ נַעֲמָן / צִיר נֶאֱמָן



מָתוֹק כַּמָּן, / הוּא בַזְּמַן / שִׁמְרֶךָ



דּוֹבֵב בְּנִיב / יַצִּית סְבִיב / צָרָיו שְׁבִיב



אַךְ לוֹ רְבִיב / מֵאִיר נְתִיב / אִמְרֶךָ



הוּא הַיְּסוֹד / אֶל רָז וְסוֹד / עָנָו מְאוֹד



לוֹ לַעֲבֹד / הֵן יַחֲמֹד / סַהְרֶךָ



לוֹ חֵן צְבִי / אֵימַת אֲרִי / וּבְפִיו צְרִי



בּוֹ בַחֲרִי / יַךְ בָּחֳרִי / עָרֶיךָ



יַעְלֶה בְלִי / סֻלָּם תְּלִי / מִלָּיו חֲלִי



הֵם לָךְ וְלִי / חוֹבֵשׁ חֳלִי / שִׁבְרֶךָ



וּבְמַעֲמַד / הַמַּחֲמָד / עָמַד לְבָד



לוֹ שֵׁם וְיָד / עוֹטֶה כְּמַד / אוֹרֶךָ



אוֹר שַׁחֲרוֹ / מִתָּאֳרוֹ / הוֹד סַהֲרוֹ



מִצָּהֳרוֹ / מִזָּהֳרוֹ / זָהְרֶךָ



טוֹב מַעֲלָל / עָנָו, שְׁפַל / רוּחַ, אֲבָל



עָלָה לְגַל/גַּל רוּם וְעַל / שִׁירֶיךָ



דֵּעָה יְחַו / שׁוֹקֵל בְּקַו / גַּן בִּין יְרַו



פִּרְיוֹ תְּאַו, / אִם בּוֹ תְקַו / אַשְׁרֶיךָ



עֵינָיו סְגוֹר / נָאֶה כְתוֹר / כָּל עֵת וְתוֹר



עוֹפֵף כְּתוֹר / מוֹצִיא סְגוֹר / הָרֶיךָ



לַדָּת כְּקֵן / לִבּוֹ, וְאֵין / בּוֹ תֹךְ, וְחֵן



מָצָא כְבֶן / לֶמֶךְ בְּעֵין / צוּרֶךָ



אָדֹם וְצַח / הוֹד דּוֹד וְאָח / גִּיל נֶאֱנָח



לִבִּי שְׂמַח / וּבְשִׁיר פְּצַח / דָּבְרֶךָ.
