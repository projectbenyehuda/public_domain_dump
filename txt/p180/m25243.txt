


	
	שַׁחַר קַמְתִּי גַּם
	








שַׁחַר קַמְתִּי / גַּם נִכְלַמְתִּי / לָשֵׂאת לְךָ פָנָי

בְּךָ חָסִיתִי / וְלֹא כִסִּיתִי / עֲווֹנַי וּזְדוֹנָי

כִּי יָדַעְתִּי / גַּם שָׁמַעְתִּי / אֵל עוֹלָם אֲדֹנָי

בַּנְתָּ לְרֵעֵנוּ / יְיָ מַלְכֵּנוּ / הוּא יוֹשִׁיעֵנוּ





לְךָ צוֹפִיָּה / בַּת הוֹמִיָּה / מִבֵּית מְגוּרֶיהָ



תַּשְׁחֶה עָרְפָּהּ / תִּפְרֹשׂ כַּפָּהּ / תִּקַּח אֲמָרֶיהָ



עָנְיָהּ תִּפְנֶה / בֵּיתָהּ תִּבְנֶה / נִפְרְצוּ גְדֵרֶיהָ

עַל רֹב פְּשָׁעֵינוּ / יְיָ מַלְכֵּנוּ / הוּא יוֹשִׁיעֵנוּ





מֶלֶךְ מַקְשִׁיב / מָתַי תָּשִׁיב / צִפּוֹר גֵּרַשְׁתָּ

נָא אַל תִּבְזֶה / עֱנוּת עָנִי זֶה / כַּמֶּה נָטַשְׁתָּ

בַּעֲבוּר יְדִידְךָ / אַתָּה יָדְךָ / גּוֹיִים הוֹרַשְׁתָּ

וַתִּטָּעֵנוּ / יְיָ מַלְכֵּנוּ / הוּא יוֹשִׁיעֵנוּ





הַיּוֹם תִּגְאַל / אֶבְיוֹן יִשְׁאַל / גּוֹאֵל חֲסָדֶיךָ

אִם עָנָה בוֹ / רֹעַ לְבָבוֹ / הִנּוֹ בְיָדֶךָ

קַח שַׁוְעָתוֹ / וְגַם דִּמְעָתוֹ / שִׂימָה בְנֹאדֶךָ

אֱלֹהֵי יִשְׁעֵנוּ / יְיָ מַלְכֵּנוּ / הוּא יוֹשִׁיעֵנוּ





חַזֵּק רִפְיוֹן / מֶלֶךְ עֶלְיוֹן / וְרַחֵם עַמֶּךָ

זַכֵּה אוֹתָם / בִּזְכוּת אֲבוֹתָם / שְׁלֹשֶׁת תְּמִימֶיךָ

קָרֵב קִצָּם / וּגְדֹר פִּרְצָם / לְמַעַן רַחֲמֶיךָ

שְׁעֵה שַׁוְעָתֵנוּ / יְיָ מַלְכֵּנוּ / הוּא יוֹשִׁיעֵנוּ.
