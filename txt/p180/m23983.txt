


	
	אָמְרָה גוֹלָה
	








וּבְכֵן יְיָ מָעוֹן אַתָּה הָיִיתָ לָנוּ

פִּזְמוֹן יְיָ תִּשְׁפֹּת שָׁלוֹם לָנוּ





אָמְרָה גוֹלָה וְסוּרָה / אֲשֶׁר בְּיַד צָרִים עֲצוּרָה

בְּעוֹדִי בְּבוֹר גָּלוּת אֲסוּרָה / שָׁם אַדִּיר יְיָ לָנוּ





בְּחוּנָה בְגָלוּת יוֹשֶׁבֶת / כָּאֳנִיָּה בְלֶב יָם שׁוֹכֶבֶת

בְּכָל עֵת תְּעוֹרֵר מַחֲשֶׁבֶת / לֵאמֹר מִי יַעֲלֶה לָּנוּ





גְּרוּשָׁה מֵחֵל וָעֹפֶל / סְגוּרָה בְּמַחְשָׁךְ וָאֹפֶל

כִּמְעַט תֹּאבַד מִשֶּׁפֶל / לוּלֵי יְיָ צְבָאוֹת הוֹתִיר לָנוּ





דְּחוּיָה עִם כָּל מַמְלֶכֶת / כְּיוֹנָה קְצוּצָה מָשְׁלֶכֶת

צָדוּ צְעָדֶיהָ מִלֶּכֶת / בַּמִּדְבָּר אָרְבוּ לָנוּ





הֲפוּכָה כְּצִפּוֹר בְּתוֹךְ רֶשֶׁת / בְּכָל עֵת לִשָּׁחֵט נִדְרֶשֶׁת

וְהִיא בְּכָל זֹאת אוֹרֶשֶׁת / אוּלַי יִתְעַשֵּׁת הָאֱלֹהִים לָנוּ





וְכַמָּה זְמַנִּים נֶחְשָׁבוּ / וְעַמְּךָ בְּגָלוּת יָשָׁבוּ

וְאוֹיְבִים מֵאַחֲרֵיהֶם לֹא שָׁבוּ / כִּי עַתָּה יֹאמְרוּ אֵין מֶלֶךְ לָנוּ





זָמְמוּ שֶׁקֶר לְהַחְבִּיר / וְכִחֲשׁוּ בְדַת אֵל כַּבִּיר

אֲשֶׁר אָמְרוּ לִלְשׁוֹנֵנוּ נַגְבִּיר / שְפָתֵנוּ אִתָּנוּ מִי אָדוֹן לָנוּ





חֲסִין הוֹפַע בְּהוֹדֶךָ / וּשְׁלַח כְּמֵאָז יָדֶךָ

וְקַנֵּא לְשֵׁם כְּבוֹדֶךָ / לֹא לָנוּ יְיָ לֹא לָנוּ





טָאֵט מַלְכוּת נְבָיוֹת / וּטְרֹף פְּרִיצֵי חַיּוֹת

טוֹרְפֵי פְזוּרִים כְּשֵׂיוֹת / אֲשֶׁר אָמְרוּ נִירְשָׁה לָּנוּ





יָרַשׁ שֵׂעִיר הִגְלָנוּ / וְהַיּוֹם כַּמָּה אֲכָלָנוּ

הָאִישׁ אֲשֶׁר כִּלָּנוּ / וַאֲשֶׁר דִּמָּה לָנוּ





כְּאַיִן תַּמְנוּ וְאָבַדְנוּ / מֵרֹב אֲדוֹנִים עָבַדְנוּ

כִּמְעַט מֵהֶם נִכְחַדְנוּ / לוּלֵי יְיָ שֶׁהָיָה לָנוּ





לְרַגְלֵי עֲבָדִים כָּפַפְנוּ / וְעַל בּוֹר וּמַסְגֵּר אֻסַּפְנוּ

עַל צַוָּארֵנוּ נִרְדָּפְנוּ / יָגַעְנוּ לֹא הוּנַח לָנוּ





מָרוֹם אַיֵּה חֲסָדֶיךָ / גָּמַלְתָּ מֵאָז יְדִידֶיךָ

אֲשֶׁר שָׁמְעוּ עֲבָדֶיךָ / וַאֲבוֹתֵינוּ סִפְּרוּ לָנוּ





נָאֵר גֵּאִים וְשַׁקֵּץ / וְגַלֵּה פְלִיאַת הַקֵּץ

וְצָעִיר רוֹדֵם הָקֵץ / קוּמָה עֶזְרָתָה לָנוּ





סַגֵּב שְׁפָלִים מִשְּׁחוֹחַ / וְשִׂיחָם יֶעֱרַב כְּנִיחוֹחַ

וּלְקֹט שׁוֹשָׁן מִבֵּין חוֹחַ / וּלְכָה לִישׁוּעָתָה לָּנוּ





עֲצוּבָה עֲזוּבָה תְּשַׂמַּח / וְדִמְעַת לֶחְיָהּ תֶּמַח

וְצֶמַח צַדִּיק יִצְמַח / כִּי יֶלֶד יֻלַּד לָנוּ





פְּרֹק עֻלִּי וְהָסֵר / וְזָרִים תּוֹסִיף לְיַסֵּר

וּשְׁלַח מְהֵרָה מְבַשֵּׂר / וְיָבוֹא כַגֶּשֶׁם לָנוּ





צַהֵל יַקִּיר נִבְזֶה / וּמֶלֶךְ בְּיָפְיוֹ נֶחֱזֶה

וְיוּשַׁר הַשִּׁיר הַזֶּה / בְּאֶרֶץ יְהוּדָה עִיר עָז לָנוּ





קִרְיַת עֻזִּי תִצְלַח / וְאַתָּה תִּשְׁמַע וְתִסְלַח

וְתִשָּׂא עָווֹן נֶאֱלָח / אֵל יְיָ וַיָּאֶר לָנוּ





רִנַּת עוֹשֵׂי תְשׁוּבָה / תִּהְיֶה כְעוֹלָה חֲשׁוּבָה

וְיוֹדוּ נְשׁוּיֵי מְשׁוּבָה / לֹא כַחֲטָאֵינוּ עָשָׂה לָנוּ





שָׁלוֹם וְחֶדְוָה עָשִׂיתָ / וַעֲווֹן עַמְּךָ נָשִׁיתָ

פֶּשַׁע וַחֲטָאָה כִּסִּיתָ / כִּי גַּם כָּל מַעֲשֵׂינוּ פָּעַלְתָּ לָנוּ





תָּמִיד תִּהְיֶה נֶגְדֶּךָ / שַׁוְעַת עַמְּךָ וַעֲבָדֶיךָ

הַרְאֵנוּ יְיָ חַסְדֶּךָ / וְיֶשְׁעֲךָ תִּתֵּן לָנוּ.
