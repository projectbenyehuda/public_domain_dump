


	
	נְשִׂיא עַמּוֹ
	




נְשִׂיא עַמּוֹ, נְגִיד אֻמּוֹ, נְסִיכוֹ / וְהַגּוֹאֵל, אֲשֶׁר בּוֹ אֵל
סְמָכוֹ



וְהַנֵּזֶר וְהָעֵזֶר וְהָעֹז / וְהַמָּעֹז לְמִגְדַּל כֹּל מְשָׁכוֹ



אֲרִי וּצְרִי, חֳרִי וּפְרִי תְעָלָה, / כְּזָר דַּרְכּוֹ - וְכֹה וָכֹה
עֲרָכוֹ



קָרָא מִכְתָּב לְמִתְאַוֶּה וְנֶאְנָח, / בְּעֵת כָּתְבוֹ - בְּדִמְעָתוֹ
מְסָכוֹ



בְּדִמְעַת לֶחְיוֹ וּדְיוֹ מְסָכוֹ / וְלוּ יָכֹל עֲבֹר - הָיָה בְתוֹכוֹ



וְשָׁלוּחַ בְּיַד שָׁפֵל וְאָפֵל / אֲשֶׁר הַבַּז וְהַבּוּז הֶחֱשִׁיכוֹ



וּבַשַּׁעַר כְּבַיַּעַר מְדֻכֶּה, / וּמֵחָמָס, וְאִם הָלוֹךְ וּבָכֹה



וְאוּלַי נָס וּבִי נֶאְסַף לְעוֹלָה, / וְאֵל עָלַי בְּמָעְלָלַי הֲפָכוֹ



כְּטוֹבֵעַ אֲשֶׁר נֶאְחָז בְּנִטְבָּע / וְכָעִוֵּר אֲשֶׁר עִוֵּר
מְשָׁכוֹ.



אֲדוֹנִי, הַט יְמִינָךְ לוֹ, וְלִבָּךְ / אֱזֹר קִנְאָה וְהִתְעוֹרֵר
לְתָמְכוֹ.



תְּהִי שָׁלוֹם, אֲדוֹן נַפְשִׁי, לְעֵילוֹם / נְשִׂיא עַמּוֹ, נְגִיד
אֻמּוֹ, נְסִיכוֹ!
