


	
	אָז בְּאֵין זְרוּתֵי מַעַל
	








אָז בְּאֵין זְרוּתֵי מַעַל / אַף בְּאֶפֶס כָּל פֹּעַל

אַצְתָּ וַתַּעַל / אֲרוּכָה גַם תַּעַל

אַף קָרָאתָ מִמַּעַל / אֹמֶר לְשָׁבֵי מַעַל / אָנֹכִי אָנֹכִי הוּא





נָזוֹרוּ מִנְּתִיבֶיךָ / נָסוֹגוּ שִׁבְעַת סָרָבֶיךָ

נָאוֹר כְּשׁוּב שׁוֹבָבֶיךָ / נִצְלוּ מֵעֹז לְהָבֶיךָ

נַקֵּה כְּמוֹ כֵן שָׁבֶיךָ / נַהֵר מִשְׁפַּט אוֹהֲבֶיךָ / כִּי הַמִּשְׁפָּט לֵאלֹהִים הוּא





יְצִיר כַּף עֵת שָׁגָה / יִסַּרְתּוֹ בְּקוֹל שְׁאָגָה

יָצָא מִסּוּגָה / יַעַן הֶעֱמִיק מְשׁוּגָה

יוֹם שָׁב מִפּוּגָה / יָהּ חִלַּצְתּוֹ מִתּוּגָה / כִּי בֶן מָוֶת הוּא





שַׁי נִמְאָס כְּהוּשַׁב / שְׁאוֹן זַעְפּוֹ נָשַׁב

שְׁרִירוּת לֵב חָשַׁב / שִׁכֵּל אָח כְּנִקְשָׁב

שָׁפַךְ שִׂיחַ וְשָׁב / שִׁוַּע לְךָ בְּמוֹשָׁב / כִּי אָמַר פֶּן יָמוּת גַּם הוּא





לִי בְּכוֹר הַנִּכְסָף / לִשְׁנֵי כְתָרִים אָסַף



לְעֵת שָׁקַד עַל סַף / [ל סַף]



לַהַג תְּשׁוּבָה יָסַף / לַאֲשֶׁר יָצוּעַ חָשַׂף / וְאִם עַל הַמִּשְׁכָּב הוּא





מָעַד אֲשׁוּר חַכְלִיל / מוּל הוֹמִיָּה כְּחָלִיל



מִ[שֶּׁבְּחַ]רְתּוֹּ נָגִיד [וּפָ]לִיל / מוּכָן לְנֵזֶר וְכָלִיל



מוּמוֹ מְזֻקָּק בַּעֲלִיל / מֵאָז יְהוּדָה יְהוֹדֶה לִיל / כִּי פָרוּעַ הוּא





הֵן סָח לְפֶסֶל וּמַזָּל / העליו רָבִיב לְנֶאֱזָל

הָלַם אִישׁ וְגָזַל / הַדְרַת פְּרִי וְזַלְזַל

הָעֵת אַט אָזַל / הוֹד כִּסְאוֹ לֹא הוּזַל / מֶלֶךְ יִשְׂרָאֵל הוּא





בּוֹגֵד פָּרַח וְגַם צָץ / בִּמְאֹד זָר וְהִתְלוֹצָץ

בְּכֵן נִזְרוֹ קֻצַּץ / בְּיַד רוֹעֲדִים רֻצַּץ

בְּעֵת נִכְנַע לְחָצָץ / בְּצִיץ מַמְלַכְתּוֹ נוֹצַץ / וַיֹּאמֶר דְּבַר יְיָ הוּא





נִכְרַת מֵאָז בְּמַזְמֶרֶת / נוֹף כִּנִּיתוֹ צַמֶּרֶת



נִתַּן בְּשׁוּחָה נִכְמֶרֶת / נאם רח משה בְּמִשְׁמֶרֶת



נִפְקַד לִצְפִירָה נִגְמֶרֶת / נוֹעַד לְמִשְׂרָה נֶאֱמֶרֶת / אֶת הַנָּשִׂיא נָשִׂיא הוּא





גְּעוּלִים מֵאָז כְּשָׁבוּ / גְּנֹן כְּמוֹ כֵן עַם הוּשָׁבוּ

גְּאַל אֲסִירִים בְּבוֹר נוֹשָׁבוּ / גּוֹעִים לְךָ כַּצֹּאן נֶחְשָׁבוּ

גַּם בַּצּוֹם לְפָנֶיךָ יָשָׁבוּ / [גּ שָׁבוּ] / כִּי חֹק לְיִשְׂרָאֵל הוּא





בְּטִיט יָוֵן צָנַחְנוּ / בְּכֵן נֶאֱלַחְנוּ וְנֶאֱנַחְנוּ

בַּשָּׁוְא עָמַלְנוּ וְלֹא נַחְנוּ / בַּאֲשֶׁר לֹא הוּנַחְנוּ

בְּרִיתְךָ אִם זָנַחְנוּ / בִּין כִּי עָפָר אֲנָחְנוּ / כִּי בַמֶּה נְֶחְשָׁב הוּא





יְדִידִים צִדְקָם יִזָּרַע / יָשְׁרְךָ עֲלֵיהֶם יִשְׂתָּרַע

יוֹם לְבָבָם יִקָּרַע / יֹקֶשׁ עֲווֹנָם יִגָּרַע

יָהּ יְגַל וְיִפְרַע / יֵצֶר לֵב הָאָדָם רַע / כִּי מִתְאַנֶּה הוּא





רְאֵה וְהַבֵּט לַחַץ / רְפָא שֶׁבֶר וּמַחַץ

רֹאשׁ מְסִבַּי תַּחַץ / רָם עָלַי כְּשַׁחַץ

רְעַץ כְּאֵבִי בְנַחַץ / רְחַץ לְבָבִי רַחַץ / וְסָלַחְתָּ לַעֲווֹנִי כִּי רַב הוּא





וּכְמוֹ הִצְלִיל תַּנִּין / וְנִעֵר סִלּוֹן וּצְנִין

וְכֵן יַצִּיל אָדֹם מִפְּנִין / וְיִמְחֶה כָעָב וְיַעֲנִין

וְלִבּוֹת קוֹוָיו יַרְנִין / וְרִחַם עַל אָב וְנִין / כִּי חַנּוּן וְרַחוּם הוּא





לְעַם לְךָ יִכְרַע וְיִקְרֹס / לְחָצָץ שִׁנָּיו יִגְרֹס

לְמַיִם טְהוֹרִים תִּפְרֹשׂ / לִזְרֹק עֲלֵיהֶם וְלָרֹס

לְהוֹדוֹת בְּחָלִיל וְקַתְרוֹס / לְךָ כְּמֵאָז בְּפַתְרוֹס / עֶזְרַת אֲבוֹתֵינוּ אַתָּה הוּא.
