


	
	יוֹסְפִים שֵׁנִית
	








יוֹסְפִים / שֵׁנִית לַעֲמֹד בְּמִשְׁמֶרֶת תְּהִלָּתָם

וְלִפְנֵי יְיָ לִבָּם לִשְׁפֹּךְ וּתְחִנָּתָם

וְלֶחָי לְצִיּוֹן יַעֲלוּ / לְהַקְרִיב אֶת קָרְבְּנוֹתָם / תְּמִידִים כְּסִדְרָם וּמוּסָפִים כְּהִלְכָתָם





יוֹסְפִים / לְבֵיתְךָ יַעֲלוּ וְלָקְחוּ עִמָּם אֲמָרִים

הֱיוֹתָם מֵלִיץ בֵּינְךָ וּבֵין לֹא אַנְשֵׁי דְבָרִים

יִהְיוּ כֹפֶר בַּעֲדָם וּסְלִיחָה וְכִפּוּרִים

וְלֶחָי לְצִיּוֹן יַעֲלוּ / עַל מִזְבַּחֲךָ פָרִים / וּשְׂעִירִים כְּחֻקּוֹתָם וְנִסְכֵּיהֶם וּמִנְחָתָם





יוֹסְפִים / מִמָּגוֹר חָלוּ צְפוּן לִבָּם יִבָּדֵק

וְאֵיךְ חֶסְרוֹן יִמָּנֶה וְיַעֲלֶה בְחֶשְׁבּוֹן לְדַקְדֵּק

בִּדְקָם תַּחֲזִיק כִּי אֵין בִּלְתְּךָ מַחֲזִיק בֶּדֶק

וְלֶחָי לְצִיּוֹן יַעֲלוּ / וְשָׁם יִזְבְּחוּ זִבְחֵי צֶדֶק / וְהַקְרָבַת נִשְׁמָתָם לְךָ תְהִי כַפָּרָתָם





יוֹסְפִים / הַיּוֹם לִבָּם הִקְרִיבוּ חַטָּאתָם

וְיָד פָּרְשׂוּ וְהֵרִימוּ מְקוֹם תְּרוּמַת מִנְחָתָם

וְחֶלְבָּם חֵלֶב חַטָּאת וְנִסְכֵּיהֶם דִּמְעוֹתָם

וְלֶחָי לְצִיּוֹן יַעֲלוּ / בְּרִנָּתָם וְשִׁירָתָם / וְיִהְיוּ לְרָצוֹן כְּזִבְחֵי כִפּוּר עַל מִזְבְּחוֹתָם





יוֹסְפִים / כַּיּוֹם אִוּוּ צֵאת מִדַּרְכֵי אָדָם

וְדַרְכֵי מַלְאֲכֵי עֶלְיוֹן אָחֲזוּ וּבָאוּ בְסוֹדָם

וְשִׁירוֹת עוֹרְכִים כְּמוֹהֶם וְהֵם עוֹמְדִים עַל עָמְדָם

וְלֶחָי לְצִיּוֹן יַעֲלוּ / יִשְׂרָאֵל בְּמַעֲמָדָם / וּלְוִיִּים בְּדוּכָנָם וְכֹהֲנִים בַּעֲבוֹדָתָם.





שִׁמְךָ לְהַלֵּל מִשְׂגַּב הוֹדִי

בְּקָהָל רוֹמַמְתִּי בְּמֹעַל יָדִי

יֶעֱרַב הַשִּׂיחַ וְיִנְעַם דְּבַר הֻיְּדִי

וְאַתָּה יְיָ מָגֵן בַּעֲדִי.
