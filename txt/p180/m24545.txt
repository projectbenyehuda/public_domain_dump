


	
	שְׁלִישִׁית שׁוֹקֶדֶת
	








שְׁלִישִׁית שׁוֹקֶדֶת מְשַׁלֶּשֶׁת צְוָחָה / שׁוֹקְדָה עַל דֶּלֶת וְעוֹמְדָה עַל מִשְׁמָר

לְפָנֶיךָ מֵאֶמֶשׁ שׁוֹפֶכֶת שִׂיחָהּ / לִבָּהּ וּבְשָׂרָהּ מִפַּחְדְּךָ סָמָר

מְחוּצָה לְחוּצָה בְּשַׁחַת וְשׁוּחָה / מִבֹּץ מְצַעֶקֶת וּמִבּוֹר הַחֵמָר

הָלְכוּ שְׁנוֹתֶיהָ בְּיָגוֹן וַאֲנָחָה / הוֹגָה בְמַר נֶפֶשׁ הִנֵּה מַר לִי מָר

הֲלָעַד אֶהְיֶה בְגָלוּת מְשֻׁלָּחָה / הֲשָׁכַח חַנּוֹת אֵל וְאִם אֹמֶר גָּמָר

קַבֵּץ נִדְּחֵי הַשֶּׂה הַנִּדָּחָה / קֶרֶן בֶּן שֶׁמֶן תִּרְעֶה וְתִתְיַמָּר

טָהוֹר הַבֵּט כִּי כִפְלַיִם לָקָחָה / טָעֲמָה קֻבַּעַת וְתַרְעֵלָה חָמָר

נְטַע כִּימֵי קֶדֶם גֶּפֶן כְּסוּחָה / נִצָּנֶיהָ יַעֲלוּ וְתִפְרַח כַּתָּמָר

חֻקַּת אֶזְרָח תְּהִי לָהּ מְפֻעְנָחָה / חֲקַרְתּוֹ בְעֶשֶׂר וְרֵיחוֹ לֹא נָמָר

זַךְ תִּכּוֹן כִּקְטֹרֶת רְקוּחָה / זַעֲקַת מְצַפְצֶפֶת כְּצִפּוֹר מִמִּכְמָר

קוֹל קְרִיאַת קוֹדֶרֶת וְשַׁחָה / קַבֵּל כְּתִשְׁבִּי חֹק בַּל יוּמָר

הָעֵת וַיְהִי בַּעֲלוֹת הַמִּנְחָה / וַיִּגַּשׁ אֵלִיָּהוּ הַנָּבִיא וַיֹּאמַר:





שַׁחוֹתִי לְפָנֶיךָ מוּל הֵיכָלָךְ

בְּתוֹךְ רַבִּים יִסַּדְתִּי הִלּוּלָךְ

וַאֲקַוֶּה דְבַר הַבְטָחַת אֲצִילָךְ

אַל תִּירָא אַבְרָם אָנֹכִי מָגֵן לָךְ.
