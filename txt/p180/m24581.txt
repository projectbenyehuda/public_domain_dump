


	
	שְׁחָקִים רָקַע
	








שְׁחָקִים רָקַע / אֲרָקִים תָּקַע / נִיצוֹץ בָּקַע / אֵלִי מַלְכִּי בַקֹּדֶשׁ



לָפַת רָקִיעַ / וְעֵץ פְּרִי הִתְקִיעַ / מְאוֹרוֹת הִבְקִיעַ / הִשְׁתַּחֲווּ לַיְיָ בְּהַדְרַת קֹדֶשׁ



מִקְוֶה וְיַבֶּשֶׁת / גַּם אִישׁ וְאֵשֶׁת / כִּלָּה בְּשֵׁשֶׁת / וּבַיּוֹם הַשְׁבִיעִי מִקְרָא קֹדֶשׁ



הוּסַת נִפְרָד / וּמִגַּן נִטְרַד / יַעַן אֲשֶׁר מָרַד / וְאֵת אֲשֶׁר חָטָא מִן הַקֹּדֶשׁ



בּוֹגֵד נִזְעַק / לָשׂוּם מוּעָק / דַּם אָח צָעַק / וְנִשְׁמַע קוֹלוֹ בְּבוֹאוֹ אֶל הַקֹּדֶשׁ



רָעִים פִּשַּׁח / רָמִים הֵשַׁח / אֵיתָן נִמְשַׁח / שֶׁמֶן מִשְׁחַת קֹדֶשׁ



יְמִינוֹ יָדָה / בְּקָנֶה וְקִדָּה / יְחִידוֹ נִפְדָּה / בְּצֵאתוֹ בַיּוֹבֵל קֹדֶשׁ



הַבְּאֵר נִסְתַּם / עַד צֵאת אִישׁ תָּם / בְּכֵס יָהּ נֶחְתַּם / פִּתּוּחֵי חוֹתָם קֹדֶשׁ



וּמֵעֵת הוּסַר / לָגוּר בֻּשַּׂר / בִּשְׁנֵים עָשָׂר / שׁוֹמְרֵי מִשְׁמֶרֶת הַקֹּדֶשׁ



דְּחוּיֵי חָנֵס / בְּרַחֲמָיו כִּנֵּס / הֵרִימוּ נֵס / וְקָרְאוּ לָהֶם עַם הַקֹּדֶשׁ



הָגוּ הֲמוֹנַי / בְּהַצְלִיל מוֹנַי / מִי כָמוֹכָה בָּאֵלִים יְיָ / מִי כָּמוֹכָה נֶאְדָּר בַּקֹּדֶשׁ.
