


	
	מִי כָמוֹכָה
	








מִי כָמוֹכָה וְאֵין כָּמוֹךָ

מִי דוֹמֶה לְךָ וְאֵין דּוֹמֶה לָּךְ





מִי כָמוֹכָה שׁוֹכֵן עַד לֹא רֵאשִׁית / וּמִי דוֹמֶה לָּךְ

שְׁבִילוֹ וְדַרְכּוֹ בַּיָּם הֵשִׁית / יְיָ אֱלֹהֵינוּ יְיָ אֶחָד





מִי כָמוֹכָה לְפָנֶיךָ יַעֲרֹךְ וְיוֹעִיד / וּמִי דוֹמֶה לָּךְ

לְשִׁמְךָ הַכֹּל יוֹדֶה וְיָעִיד / יְיָ אֱלֹהֵינוּ יְיָ אֶחָד





מִי כָמוֹכָה מָרוֹם לֹא רָאֲתָה עַיִן / וּמִי דוֹמֶה לָךְ

מוֹצִיא כָל יֵשׁ מֵאַיִן / יְיָ אֱלֹהֵינוּ יְיָ אֶחָד





מִי כָמוֹכָה הַנּוֹטֶה כַדֹּק שָׁמַיִם / וּמִי דוֹמֶה לָּךְ

הָרוֹקֵעַ הָאָרֶץ עַל הַמַּיִם / וּמִי כְּעַמְּךָ יִשְׂרָאֵל גּוֹי אֶחָד





אָז בְּרֵאשִׁית קִנְיָן / אֵל הִפְלִיא עִנְיָן

יָסַד רֹאשׁ וּמִנְיָן / וְעָשָׂה אֶת הָאֶחָד





בִּהְיוֹת הַכֹּל בָּנוּי / נוֹדַע כִּי הוּא קָנוּי

גַּם לְכָל דָּבָר מָנוּי / הֲלֹא אָב אֶחָד





גָּלַם שַׁפְרִיר נִבְצָר / וְאַרְבַּע חַיּוֹת עָצַר

וּבְהוֹד עֻזּוֹ יָצַר / אַרְבָּעָה פָנִים לְאֶחָד





דִּפֵּן אֶבֶן אֲרָקָיו / וְנָטָה עָלֶיהָ קָו

וַיֵּט גַּלְגַּלֵּי שְׁחָקָיו / וַיְהִי הַמִּשְׁכָּן אֶחָד





הֲלִיכוֹת אֲפֵלָה פוֹצֵץ / וַיִּהְיֶה אוֹר נוֹצֵץ

סִבֵּב עֲלֵי חוּג מְרוֹצֵץ / וַיְהִי עֶרֶב וַיְהִי בֹקֶר יוֹם אֶחָד





וַיְצַו נֵבֶךְ לְהַשְׁקִיעַ / וַיְתַו תּוֹכוֹ רָקִיעַ

וְלִשְׁנֵי גַלִּים הִבְקִיעַ / אֶת הַיָּם הָאֶחָד





זָמַם בִּמְלֹאת יוֹמַיִם / וְנָאַם יִקָּווּ הַמָּיִם

מִתַּחַת הַשָּׁמַיִם / אֶל מָקוֹם אֶחָד





חֲנִיטֵי דְשָׁאִים הִפְרִיץ / וּמְאוֹרֵי אוֹר הֵרִיץ

וּמִינֵי חַיּוֹת הִשְׁרִיץ / זְאֵב וְטָלֶה יִרְעוּ כְאֶחָד





טָהוֹר יְצוּרָיו קָדַם / וְאָמַר נַעֲשֶׂה אָדָם

לִהְיוֹת מוֹשֵׁל בְּסוֹדָם / וַהֲקִימוֹתִי עֲלֵיהֶם רוֹעֶה אֶחָד





יָצַק עָפָר וְסָפַח / וְעֶשֶׁת מְרֻקָּם טָפַח

וְנֶפֶשׁ בְּאַפָּיו נָפַח / וַיְהִי אִישׁ אֶחָד





כְּבוּדָה הֵשִׁיב לְצִדּוֹ / לִהְיוֹת עֵזֶר כְּנֶגְדּוֹ

לֹא טוֹב הֱיוֹתוֹ לְבַדּוֹ / טוֹבִים הַשְּׁנַיִם מִן הָאֶחָד





לְטִירָה יְקָרָה שִׁבֵּץ / וְתַרְשִׁישׁ וְשֹׁהַם רִבֵּץ

וְאוֹתָם לְתוֹכָהּ קִבֵּץ / וְהָיוּ לְבָשָׂר אֶחָד





מָרוֹם הִזְהִיר מִשְׁמַעַת / יָדְךָ בְכָל עֵץ נוֹגַעַת

אַךְ מֵעֵץ הַדַּעַת / תִּהְיוּ לְעֵבֶר אָחָד





נִפְתָּה הֲפַכְפַּךְ וָזָר / וַיֵּצֵא לָמוּת וְנִפְזַר

עָלָיו הָאֵל גָּזַר / שְׁתֵּי אֵלֶּה רֶגַע בְּיוֹם אֶחָד





סִנְסֵן שְׁנַיִם סְפִיחִים / בּוֹקֵר וְעוֹבֵד שִׂיחִים

קִדְּמוּ תְשׁוּרַת מֵחִים / וְדִבֶּר חַד אֶת אַחַד





עָרְכוּ שָׁמֵן וְרָזֶה / זֶה נִכְבָּד וְזֶה נִבְזֶה

וַיָּקָם זֶה עַל זֶה / וַיַּכּוֹ הָאֶחָד אֶת הָאֶחָד





פִּלֵּל לְצוּרוֹ נִכְנָע / אַל נָא חַסְדְּךָ תִמְנַע

וְהָיִיתִי נָד וָנָע / עַתָּה אֶסָּפֶה יוֹם אֶחָד





צוּרִי עֲווֹנִי עוֹדֵף / וְהִנֵּה אוֹתִי הוֹדֵף

אַחֲרֵי מִי אַתָּה רוֹדֵף / אַחֲרֵי פַרְעוֹשׁ אֶחָד





קֻבַּל בְּנַחַת וְשׁוּבָה / בְּשׁוּבוֹ לְהוֹדוֹת מְשׁוּבָה

עַל כֵּן יְצַוּוּ תְשׁוּבָה / הַנְּבִיאִים פֶּה אֶחָד





רַבָּה רָעַת מְתֵי גַעַל / וְהוּחַל לִקְרֹא בְּשֵׁם בַּעַל

כִּחֲשׁוּ לָאֵל מִמָּעַל / וַיֹּאמְרוּ גֶּשׁ הָלְאָה וַיֹּאמְרוּ הָאֶחָד





שָׁקַף אֵל טוֹב וְסַלָּח / עַל דּוֹר אֲשֶׁר לֹא יִצְלַח

כֻּלּוֹ סָג יַחְדָּו נֶאֱלָח / אֵין עוֹשֵׂה טוֹב אֵין גַּם אֶחָד





תַּקִּיף הִכְחִיד סוֹדָם / וְנָהָר יוּצַק יְסוֹדָם

כִּי מִקְרֶה בְנֵי הָאָדָם / וּמִקְרֶה הַבְּהֵמָה וּמִקְרֶה אֶחָד





אָזְנוֹ פָקַח וַיְגַל / טֶרֶם יְסוֹבֵב גַּלְגַּל

וַיְצַו מְפַלֵּס מַעְגָּל / קַח לְךָ עֵץ אֶחָד





אִנָּה לְיָדוֹ עֲצֵי גֹפֶר / וְהָיָה הַכֹּפֶר כֹּפֶר

סְגוּרִים נִצְּלוּ כְעֹפֶר / אִישׁ אֶחָד אִישׁ אֶחָד





נִקְבְּצוּ אַנְשֵׁי בִקְעָה / וְנוֹעֲדוּ אַנְשֵׁי רִשְׁעָה

יָזְמוּ לְפַסֵּג גִּבְעָה / וְשָׂמוּ לָהֶם רֹאשׁ אֶחָד





יָזְמוּ מְזִמָּה לְהָעִיר / הָבָה נִבְנֶה לָּנוּ עִיר

וְנִבְצַר גָּדוֹל וְצָעִיר / וַיֹּאמֶר יְיָ הֵן עַם אֶחָד





שׁוֹאַת מִגְדָּל חָפַץ / וּשְׂפַת עַלִּיזִים קָפַץ

וְאוֹתָם מִשָּׁם נָפַץ / וְלֹא נוֹתַר מֵהֶם אֶחָד





לְעֶשְׂרִים דּוֹר פָּרַח / צִיץ הַנִּקְרָא אֶזְרָח

יָצָא כַּשֶּׁמֶשׁ וַיִּזְרָח / וְהוּא אִישׁ אֶחָד





מַסֵּכַת אֱלִילִים פָּרַץ / וּמֵאֵשׁ זָרִים לֹא עָרַץ

נָטָה אַחֲרֶיךָ רָץ / לֵב אֶחָד וְדֶרֶךְ אֶחָד





הִצַּלְתּוֹ בְשִׁבְעָה דְרָכִים / וְהוֹכַחְתָּ עָלָיו מְלָכִים

חֲשַׂכְתּוֹ מֵהֲלֹךְ חֲשֵׁכִים / אֶת פְּנֵי פַחַת אֶחָד





הָעֵת אֵיתָן סָקַר / כִּי הוּא עֲרִירִי עָקָר

דָּרַשׁ הָאֵל וְחָקַר / חֲלוֹם בְּלַיְלָה אֶחָד





קָרָא שׁוֹשָׁן בְּתוֹךְ שַׁיִת / הִנְנִי נִשָּׁל כַּזַּּיִת

וְהָיָה מָחָר בֶּן בַּיִת / כֹּהֵן לְבֵית אִישׁ אֶחָד





טָעַם שָׂא זְרוֹעֶךָ / וּסְפֹר כּוֹכְבֵי יִשְׁעֶךָ

כֹּה יִהְיֶה זַרְעֶךָ / וְעָשִׂיתִי אוֹתָם לְגוֹי אֶחָד





נָדִיב מוֹפֵת דָּרַשׁ / בַּמָּה אֵדַע כִּי אִירַשׁ

כַּאֲשֶׁר צוּר אָרַשׁ / נָתַתִּי לְךָ שְׁכֶם אַחַד





בֹּאַר בֶּן סוֹד לוֹעֵז / בְּעֶגְלָה וְאַיִל וָעֵז

וְתוֹר וְגוֹזָל הָעֵז / וְקָרֵב אוֹתָם אֶחָד אֶל אֶחָד





רָזַם לוֹ בְּמַבָּע / סוֹד מַלְכֻיּוֹת אַרְבַּע

וַיָּקָם אֲשֶׁר נִשְׁבַּע / לֹא נָפַל דָּבָר אֶחָד





בְּשֵׂיבוֹ צָהַר צָהוֹר / וַיּוֹלֶד זַךְ וְטָהוֹר

שָׁמְרוּ כַּבֵּן כַּהוֹר / תּוֹרָה אַחַת וּמִשְׁפָּט אֶחָד





יָזַם חוֹקֵר רָזֶה / לְהַרְאוֹת לְכָל צַר בּוֹזֶה

אֵי זֶה יִכְשַׁר הֲזֶה אוֹ זֶה / וְאִם שְׁנֵיהֶם כְּאֶחָד





יָהּ בְּחָנוֹ עֲשִׂירִיָּה / כְּגֶפֶן עֲנֵפָה פּוֹרִיָּה

לֶךְ לְךָ אֶל אֶרֶץ הַמּוֹרִיָּה / וֶאֱחֹז לְךָ אֶחָד





הִקְדִּים מִבְחַר בָּדָיו / וְעָמַס עֵצִים בְּבַדָּיו

וַיֵּלְכוּ שְׁנֵיהֶם יַחְדָּו / לְפָנֶיךָ בְּדֶרֶךְ אֶחָד





וַיַּעַן בֵּן מִשְׁתָּאֶה / אָבִי הַזֶּבַח אַיֵּה

הֱשִׁיבוֹ אֱלֹהִים יִרְאֶה / אֶת הַכֶּבֶשׂ הָאֶחָד





דִּמְעַת אָב תְּפַכֶּה / וְהַבֵּן מָוֶת יְחַכֶּה

זֶה יֵילִיל וְזֶה יִבְכֶּה / לִפְאַת הָאֶחָד





הָהּ לְשֹׁד אָבִי וּפַחְדּוֹ / כִּי אֶפֹּל חָלָל לְצִדּוֹ

וְנוֹתַר הוּא לְבַדּוֹ / וְאִילוֹ הָאֶחָד





אָבִי אַל נָא תִבְכֶּה / וְקַוֵּה לַיְיָ וְחַכֵּה

כִּי לְשָׂכָר טוֹב תִּזְכֶּה / לַכֹּל מִקְרֶה אֶחָד





בְּמוֹתִי נַחֵם אִמִּי / כִּי אֵאָסֵף אֶל עַמִּי

וּמָחָר אַתֶּם עִמִּי / הַכֹּל הוֹלֵךְ אֶל מָקוֹם אֶחָד





נָעוּ וְזָעוּ צְבָא עֶרֶץ / בְּשׁוּרָם כִּי בָא הַקֶּרֶץ

וְכִמְעַט תִּפֹּל עַל הָאָרֶץ / קוֹמַת הַכְּרוּב הָאֶחָד





גָּעֲשׁוּ לִפְנֵי אָיֹם / חַנּוּן הָפֵק פִּדְיוֹם

וְלֹא יִשָּׁחֵט כַּיּוֹם / טְלֵה חָלָב אֶחָד





בָּכוּ בִרְאוֹת הַתַּעַר / נִלְקַח לִשְׁחֹט הַנַּעַר

עַד קָרְאוּ פִּתְחוּ הַשַּׁעַר / שִׁלְחוּ מִכֶּם אֶחָד





יָדִיד בְּקַחְתּוֹ הַזַּיִן / נָם עֶזְרִי מֵאַיִן

וַיַּרְא בְּשֵׂאת לְךָ עַיִן / וְהִנֵּה אוֹפַן אֶחָד





רָחַשׁ אֱסֹף יָדֶיךָ / מֵאֵת בִּנְךָ יְחִידֶךָ

כִּי כְבָר נוֹדַע סוֹדֶךָ / וַיִּמָּלֵט בֵּן אֶחָד





וּבְעֵת רִדְתּוֹ מֵעֵרֶךְ / נָשָׂא עֵינָיו שְׁפַל בֶּרֶךְ

וַיַּרְא בִּסְבַךְ הַדֶּרֶךְ / וְהִנֵּה אַיִל אֶחָד





לָהַג בְּתִתּוֹ פִדְיוֹם / וַיִּצְעַק לְנוֹרָא וְאָיֹם

לְזַרְעִי זְכֹר זֶה הַיּוֹם / וְהָיָה יוֹם אֶחָד





מַעְיַן חָכְמָה נִסְתַּם / עַד כִּי נוֹלַד אִישׁ תָּם

בְּכִסֵּא כָבוֹד נֶחְתַּם / וַיְכוּנֶנּוּ בָּרֶחֶם אֶחָד





אָזַן הוֹלֵךְ וָשָׁב / וּלְחוּט מְשֻׁלָּשׁ נֶחְשַׁב

בְּאֹהֶל תְּבוּנָה יָשַׁב / לְחַבֵּר אֶת הָאֹהֶל לִהְיוֹת אֶחָד





לְבַדּוֹ לָגוּר נִמְסַר / וּבְעֹז וְשָׁלוֹם בֻּשַּׂר

נִכְתַּר בִּשְׁנֵים עָשָׂר / נָשִׂיא אֶחָד נָשִׂיא אֶחָד





קָרְבוּ יָמִים וְלֵילוֹת / גַּם נִתְכְּנוּ הָעֲלִילוֹת

הִפְלִיא שַׁדַּי פְּלִילוֹת / וַיֵּצֵא הָאֶחָד





יוֹסֵף לְפַתְרוֹס הוּרַד / וּמֵחֵיק אָבִיו נִפְרַד

זָעַק עָלָיו וְחָרַד / נִגְדַּע הַיּוֹם שֵׁבֶט אֶחָד





חָבוּשׁ בְּמַסְגֵּר אֻסַּר / עַד עֵת לְמֶלֶךְ הוּשַׂר

שָׂמוֹ לְבֵית טִפְסָר / מֶלֶךְ עֲלֵיהֶם וְרוֹעֶה אֶחָד





זֻלְזַל כְּצִפּוֹר מִקֵּן / זֶרַע מְאַשֵּׁר מְתַקֵּן

יַחְדָּו נַעַר וְזָקֵן / בְּנֵי אִישׁ אֶחָד





קִנְּנָה בְחָנֵס צִפּוֹר / וְרָבְתָה עַד אֵין לִסְפֹּר

יָשְׁבָה שְׁלֵוָה מֵחֲפֹר / כִּימֵי מֶלֶךְ אֶחָד





וַיְהִי כְּנֶאֱסַף יוֹסֵף / חָשַׂף זְרוֹעוֹ חוֹשֵׂף

וַיֹּאמֶר רַב שׁוֹסֵף / יֶשְׁנוֹ עַם אֶחָד





וְהִנּוֹ עָצוּם וָרָב / וְהָיָה בִּהְיוֹת הַקְּרָב

וְנוֹסַף עַל צַר וְאָרַב / אֶל הָעִיר כְּאִישׁ אֶחָד





יְחִידִים עִנּוּ בְמַס / וּבְטִיט וּבְחֹמֶר לְמִרְמָס

וַיִּזְעֲקוּ שֹׁד וְחָמָס / טַף וְנָשִׁים בְּיוֹם אֶחָד





אָתָה צְפִירַת שִׁיחוֹר / וְהַקּוֹץ יִסּוֹב לְאָחוֹר

שָׁלַח יָדוֹ מִן הַחוֹר / וַיְמַלֵּא אֶת יַד אֶחָד





מֵעֵץ שְׁלִישִׁי הַנָּעִים / צִמַּח שְׁלֹשָׁה נְטִיעִים

כְּמַרְאֵה שְׁלֹשָׁה גְבִיעִים / מְשֻׁקָּדִים בְּקָנֶה אֶחָד





צוֹפֶה בְּלֶכְתּוֹ בַדֶּרֶךְ / פָּגַע מֵצִיץ מֵחֶרֶךְ

שְׁלָחוֹ לְהוֹצִיא מִפֶּרֶךְ / בֵּית אָב אֶחָד





לֶךְ נָא אֶל אִישׁ חֶרְמִי / וְדַבֵּר שְׁלַח עַמִּי

וַעֲשֵׂה אוֹתוֹת בִּשְׁמִי / וּלְקַח עֵץ אֶחָד





בָּא וַיִּקְרָא בְּמֶרֶץ / שְׁלַח עַמִּי מֵאֶרֶץ

בִּדְבַר אֶחָד בְּרוּם עֶרֶץ / וַיֹּאמֶר מִי אֶחָד





בְּדָרְשׁוֹ מוֹפֵת לְהַזְמִין / הִשְׁלִיךְ מַטֶּה בַיָּמִין

וַיֹּאמֶר לֹא אַאֲמִין / עַל פִּי עֵד אֶחָד





כַּשָּׁפִים עָשׂוּ בַלָּט / וּמַטֵּה אַהֲרֹן שָׁלַט

מַטּוֹת בַּדִּים חָלַט / וְהָיוּ לְעֵץ אֶחָד





מַרְגִּישׁ עָצְמוֹ יִשְׂטֹם / אָזְנוֹ אָטוֹם יֶאֱטֹם

דִּבֵּר גַּם זֶה חַרְטֹם / וְאִם יִתְקְפוֹ הָאֶחָד





כָּפַשׁ הַיְּאוֹר וְהִכּוֹ / וְלֹא שָׁת לִבּוֹ עַד כֹּה

הוֹסִיף אֹמֶץ בְּדַרְכּוֹ / הָלַךְ בְּדֶרֶךְ אֶחָד





לֹהֲקוּ צְפַרְדְּעִים בִּפְנִים / בְּבָתֵי פַחִים וְצִנִּים

וַיַּךְ בָּהֶם כִּנִּים / שְׁנֵי הָעֹמֶר לְאֶחָד





הֵבִיא עֲלֵיהֶם עָרוֹב / אָרַב לְנַפְשָׁם אָרוֹב

וְדֶבֶר מֵמִית לָרֹב / וּמִמִּקְנֵה בְנֵי יִשְׂרָאֵל לֹא מֵת אֶחָד





מֵרְחָם בִּשְׁחִין וְגָרָד / וְהִמְטִיר עֲלֵיהֶם בָּרָד

וְאַרְבֶּה בְּאַרְצָם יָרַד / לֹא נִשְׁאַר אַרְבֶּה אֶחָד





יָעַץ חוֹמֵד לָצוֹן / בְּהַחֲשִׁיךְ נְוֵה קִיצוֹן

יֻצַּג בָּקָר וְצֹאן / וְאַתֶּם תְּלֻקְּטוּ לְאַחַד אֶחָד





חוֹקֵק מַעֲנוֹ הִפְאִיר / בְּשָׁמְעוֹ סִלּוֹן מַמְאִיר

בְּמִצְרַיִם לֹא אַשְׁאִיר / פַּר בֶּן בָּקָר אֶחָד





לָהַג תַּנִּין וְנִסְכַּל / דַּע כִּי חֶרֶב תְּאֻכַּל

לֵךְ מֵעָלַי לָמָה אֶשְׁכַּל / גַּם שְׁנֵיכֶם יוֹם אֶחָד





יָדִיד עָנָה עַכְשׁוּב / כֵּן לֹא אוֹסִיף לָשׁוּב

אַךְ יֵשׁ לְמוֹעֵד חָשׁוּב / עוֹד נֶגַע אֶחָד





מָחָר אֶגֹּף יְלָדֶיךָ / וְהִשְׁתַּחֲווּ לִי עֲבָדֶיךָ

צֵא אַתָּה וְכָל גְּדוּדֶיךָ / לְעָבְדוֹ שְׁכֶם אֶחָד





לְכוּדִים נִזְכְּרוּ זָכוֹר / וְלוּדִים נֶעְכְּרוּ עָכוֹר

וַיַּהֲרֹג בָּהֶם כָּל בְּכוֹר / לֹא נִשְׁאַר אֶחָד





יוֹם גָּדְלָה הַצְּוָחָה / אָז הָיְתָה הָרְוָחָה

וַיֵּצְאוּ מֵאֲנָחָה לְשִׂמְחָה / כָּל הַַקָּהָל כְּאֶחָד





הֲמוֹנִים כְּחָנוּ בְחִירוֹת / חֻדְּשׁוּ צָרוֹת אֲחֵרוֹת

חָפְרוּ חֲנֵסִים בְּאֵרוֹת / וַתִּקָּהֵל הָעֵדָה כְּאִישׁ אֶחָד





וַיַּעֲנוּ חֲפֵצֵי אֵידֵנוּ / מַה זֹּאת עָשְׂתָה יָדֵנוּ

כִּי שָׁלַחְנוּ מֵעָבְדֵנוּ / אֶת הָעָם הַזֶּה כְּאִישׁ אֶחָד





הַוּוֹת יָעַץ וְחָרַשׁ / וְלָקַח גַּלְגַּל וּפָרַשׁ

לִרְדֹּף עָנִי וָרָשׁ / וַיֵּצְאוּ כְּאִישׁ אֶחָד





יְדִידִים סָחֲרוּ סָחוֹר / וַיִּפְנוּ הָעָם אָחוֹר

וְהִנֵּה מַרְכְּבוֹת שִׁיחוֹר / עוֹלוֹת בְּקָנֶה אֶחָד





וַיְהִי כִּרְאוֹת רִכְבָּם / חָרְדוּ וַיֵּצֵא לִבָּם

הַיָּם בְּקִרְבָּם וְהַצַּר לְגַבָּם / מִזֶּה אֶחָד וּמִזֶּה אֶחָד





מְחוֹקֵק צָעַק וְצָרַח / כִּי הַצַּר הֵצִיץ וְזָרַח

הֱשִׁיבוֹ הַךְ יָם וּרְקַע / וּנְטֵה מַטְּךָ וּתְקַע





כָּפַף יְהוּדָה וְנִכְנַס / וְאֵזוֹר חֵילוֹ שִׁנַּס

הַיָּם רָאָה גַּם נָס / מִפְּנֵי גַעֲרַת אֶחָד





פָּשׁוּ אוֹפַן וְגַלְּגַּל / לִרְדֹּף סְלוּלֵי מַעְגָּל

וַיְכַס עֲלֵיהֶם הַגַּל / לֹא נִשְׁאַר מֵהֶם עַד אֶחָד





רוֹאֶה חֲנֵסִים וְגִוְיָם / מֵתִים עַל שְׂפַת הַיָּם

יֹאמַר אֵיפֹה אַיָּם / כִּפָּה וְאַגְמוֹן בְּיוֹם אֶחָד





יָעֲצוּ לְהַשְׁלִיךְ בְּמַיִם / נִשְׁלְחוּ בְּמֵימֵי תְהוֹמַיִם

שְׁפָטָם שׁוֹכֵן שָׁמַיִם / מִדָּה אַחַת וְקֶצֶב אֶחָד





מְצוּלוֹת יְדִידִים הִדְרִיכוּ / וְנַעֲלָם עַל צַר הִשְׁלִיכוּ

רוֹמְמוּ לָאֵל וְהִמְלִיכוּ / יְיָ אֶחָד וּשְׁמוֹ אֶחָד.
