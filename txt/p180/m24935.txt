


	
	יִגָּל חָזוֹן
	








יִגָּל חָזוֹן וּבְשׂוֹרָה / אֶשְׁמַע וּשְׁנוֹת מִסְפָּר

יֶאֱתָיוּ גִּיל אֶחְגֹּרָה / מִשִּׁבְרִי לֹא אֶחְפָּר





שׁוּעִי יַעֲרֹךְ צוּר יִשְׁעִי / נָא וּלְהַרְגִּיעִי / צִיר תִּשְׁלָח

דַּי יָשֹׁר בַּעֲווֹן בִּצְעִי / צַר וּבְיַד פִּשְׁעִי / חֶפְצוֹ יִצְלָח





וַאֲנִי לַבְּרִית אֶנְצֹרָה / פָנַי וּבְתוֹךְ מִשְׁמָר

קוּמְךָ לָעַד אֶשְׁמֹרָה / עֹז וּצְדָקוֹת אֹמַר





לִדְבָרִים עַל לֵב אַעֲלֶה / אֶצְהַל לֹא אֶחֱלֶה / לִתְלָאוֹת

אַאֲמִין כִּי אַזְכִּיר פֶּלֶא / חָנֵס מִכֶּלֶא / חֹפֶשׁ לִרְאוֹת





שָׁם יָדְךָ עֶלְיוֹן נוֹרָא / מַטֵּה שִׁכְמִי שָׁבָר

יָמִין רָמָה וּבְמוֹרָא / גָדוֹל חֶסֶד גָּבָר





מִשֵּׂעִיר בָּבֶל בָּם אוֹת / תַּעֲשֶׂה נוֹרָאוֹת / לִי נָאוֹר

תַּצְמִית קָמַי וּלְהַרְאוֹת / גָּדְלְךָ נִפְלָאוֹת / תּוֹצִיא לָאוֹר





נוֹדֵד כֹּחַ אֶעֱצֹרָה / בּוֹטֵחַ בָּךְ נֶעֱזָר

יוֹם יוֹם וּלְהַגְדִּיל תּוֹרָה / בִּגְבוּרָתְךָ נֶאְזָר





הוֹד קוֹל בִּמְגִלָּה אָרִים / וַאֲקַדֵּם שִׁירִים / וּבְתוֹדָה

אָבוֹא וִימֵי הַפּוּרִים / יִהְיוּ נִזְכָּרִים / לִי לִתְעוּדָה





וּרְכוּשׁ עַל נֵס אֶבְזֹרָה / אֶתֵּן אֹמֶר יִשְׁפָּר

וּלְאֵין נָכוֹן אֶעְזֹרָה / בַּאֲשִׁישָׁה וּבְאֶשְׁפָּר





וַעֲלֵי עָשׂוֹר אֶזְכֹּרָה / חֶסֶד אֵל מַה יָּקָר

לַיְּהוּדִים הָיְתָה אוֹרָה / שִׂמְחָה שָׂשׂוֹן וִיקָר.
