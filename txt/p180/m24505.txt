


	
	אַאֲמִיר אַאְדִּיר אֲפֻדַּת
	








אַאֲמִיר / אַאְדִּיר / אֲפֻדַּת / אֲגֻדַּת / אוֹרָתֶךָ



אֲבָאֵר / אֲבָרֵר / בְּרִיקַת / בְּהִיקַת / בְּהִירָתֶךָ



אַגְדִּיל / אַגְבִּיר / גֻּלַּת / גְדֻלַּת / גְּבוּרָתֶךָ



אֶדְגֹּל / אֶדְרֹשׁ / דַּהֲרַת / דִּבְרַת / דִּירָתֶךָ



אֶהְגֶּה / אֶהְדַּר / הִלַּת / הֲמֻלַּת / הַדְרָתֶךָ



אוֹדֶה / אוֹדִיעַ / וַעַד / וָתִיקוּתְךָ / וְיִשְׁרָתֶךָ



אֲזַמֵּר / אֶזְכֹּר / זָהֳרַת / זִמְרַת / זְכִירָתֶךָ



אָחִישׁ / אֲחַסֵּן / חֲתֻלַּת / חֲלִילַת / חֲבוּרָתֶךָ



אַטִּיף / אֵטִיב / טִכְסַת / טִיסַת / טָהֳרָתֶךָ



אֲיַשֵּׁר / אֲיַקֵּר / יִתְרַת / יִקְרַת / יִשְׁרָתֶךָ



אַכְבִּיר / אַכְתִּיר / כִּשְׁרַת / כִּתְרַת / כִּתְרָתֶךָ



אֲלַהֵג / אֲלַמֵּד / לַהֲקַת / לְצִדְקַת / לְאִמְרָתֶךָ



אַמְלִיץ / אַמְרִיץ / מְלוּכַת / מְסוּכַת / מִשְׂרָתֶךָ



אֲנוֹבֵב / אֲנוֹפֵף / נְתִינַת / נְגִינַת / נַהֲרָתֶךָ



אֲסַפֵּר / אֲסַבֵּר / שִׂיחַת / שִׂמְחַת / סִתְרָתֶךָ



אַעֲמִיד / אַעֲנִיד / עֲטֶרֶת / עֲתֶרֶת / עֲתַרְתֶּךָ



אֶפְתַּח / אֶפְצַח / פְּקֻדַּת / פַּחְדַּת / פְּלִיאָתֶךָ



אֶצְוַח / אֶצְרַח / צְרִיפַת / צְנִיפַת / צְפִירָתֶךָ



אֲקַשֵּׁט / אֲקַדֵּשׁ / קִצְבַת / קִרְבַת / קִשְׁרָתֶךָ



אֶרְעַשׁ / אֶרְחַשׁ / רְפִידַת / רְבִידַת / רוֹמְמוּתֶךָ



אֲשַׁחֵר / אֲשׁוֹרֵר / שְׁמִירַת / שִׁירַת / שִׁפְרָתֶךָ



אֲתַכֵּן / אֲתַקֵּן / תְּכוּנַת / תְּמוּנַת / תִּקְרָתֶךָ





תִּקְרָתְךָ אַדִּיר בַּמָּרוֹם נִצֶּבֶת

נִצֶּבֶת מֶמְשַַׁלְתְּךָ וְלָנֶצַח יוֹשֶׁבֶת

יוֹשֶׁבֶת סִתְרָתְךָ בַּחֲתֻלַּת שַׁלְהֶבֶת

שַׁלְהֶבֶת מִפֹּה וּמִפֹּה לַהֶבֶת

לַהֶבֶת תְּלַהֵט לְעֻמַּת מִרְכֶּבֶת

מִרְכֶּבֶת עֻזְּךָ הִיא הַנִּצֶּבֶת

הַנִּצֶּבֶת מוּל טִפְסָר יוֹשֵׁב בַּשֶּׁבֶת

בַּשֶּׁבֶת מִיָּמִין שְׁחָקִים רוֹכֶבֶת

רוֹכֶבֶת אֲפֻדַּת שְׁבִיעִי בְּרֹאשׁוֹ בְּמַצֶּבֶת

בְּמַצֶּבֶת שִׁבְעָה בְּפַרְגּוֹד יוֹשֶׁבֶת

יוֹשֶׁבֶת מִלְּפָנִים נִשְׁקְפָה יוֹבֶבֶת

יוֹבֶבֶת מוּל מִרְכַּבְתְּךָ הַנִּשְׁלֶבֶת

הַנִּשְׁלֶבֶת בְּרוּחַ וְעָנָן וְצָרֶבֶת

וְצָרֶבֶת תּוֹךְ קֶרַח מֵאֵין דּוֹאֶבֶת

דּוֹאֶבֶת וְרוֹפֶפֶת עֲלִיָּה הַנֶּחְצֶבֶת

הַנֶּחְצֶבֶת לְקוֹל הֲמֻלָּה אוֹתְךָ אוֹהֶבֶת

אוֹהֶבֶת עֻזְּךָ גַּם בְּקוֹל גַּם בְּמַחְשֶׁבֶת

בְּמַחְשֶׁבֶת וּבְקוֹל רָנֵּי זְמִירוֹת נוֹבֶבֶת

נוֹבֶבֶת רִגְשַׁת הָאַרְבַּע סָבִיב גּוֹבֶבֶת

גּוֹבֶבֶת שְׁלֹשִׁים וָשֵׁשֵׁ נֶגְדְּךָ בְּמִקְצֶבֶת

בְּמִקְצֶבֶת יָמִין וּשְׂמֹאל לְךָ עֹז יוֹהֶבֶת

יוֹהֶבֶת קֶדֶם וְאָחוֹר בְּאַלְפֵי רְבָבוֹת רוֹכֶבֶת

רוֹכֶבֶת עַל מְקוֹמְךָ הוֹלֶכֶת וְסוֹבֶבֶת

וְסוֹבֶבֶת לִרְאוֹת בְּזִיו הַמִּרְכְֶּבֶת

מִי כַּיְיָ אֱלֹהֵינוּ הַמַּגְבִּיהִי לָשֶׁבֶת





לָשֶׁבֶת נַאֲמִיר אֱיָלָךְ / נַבִּיעַ בְּגָדְלָךְ

כְּמוֹ גְּדוּדֵי גַלְגַּלִּים דְּגוּלִים דִּגְלָךְ

נְהַדֵּר הִלּוּלָךְ / נוֹדֶה וְנַגְדִּילָךְ

כְּמוֹ זְקוּפֵי זְרוּתִים חוֹקְרִים חֵילָךְ

מַטִּיף טִלּוּלָךְ / נְיַקֵּר יִחוּלָךְ

כְּמוֹ כִתֵּי כְרוּבִים לוֹהֲגִים לְמוּלָךְ

נַמְרִיץ מַהֲלָלָךְ / נְנוֹבֵב נִהוּלָךְ

כְּמוֹ סִדְרֵי שְׂרָפִים עוֹמְסִים עֻלָּךְ

נִפְצֶה פָעֳלָךְ / נְצַפֶּה צִלָּךְ

כְּמוֹ קְרוּנֵי קְרוּנִים רָצִים רַגְלָךְ

נָשׁוּר שְׁבִילָךְ / נְתַכֵּן תִּלּוּלָךְ

כְּמוֹ שְׁבִיבֵי שְׁחָקִים לוֹמְדִים לְהַלְלָךְ

נְמַלֵּל מַכְלוּלָךְ / נְהַלֵּל הִלָּךְ

כְּמוֹ חֵילֵי זְקוּפִים כֶּתֶר יִתְּנוּ לָךְ.
