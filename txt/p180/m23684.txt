


	
	תָּעָה לְבָבִי
	








תָּעָה לְבָבִי אַחֲרֵי הַתֹּהוּ / וְתוּשִׁיָּה נִדְּחָה מִמֶּנִּי

עָזַבְתִּי אָרְחוֹת יֹשֶׁר / וּבְהַבְלֵי עוֹלָם נִסְמַכְתִּי

כְּסָלַי מָלְאוּ נִקְלֶה / וּמֹחַ עַצְמוֹתַי מִרְמָה

עֲטִינַי מָלְאוּ זִמָּה / טָבְעוּ רַעְיוֹנַי בִּמְצוּלוֹת חֵטְא

כִּלְיוֹתַי כָּלוּ מֵאָוֶן וּמִרְמָה / וּמִתֹּךְ וּמֵחָמָס יַחְדָּו אֻמְלָלוּ

שֶׁקֶר גַּם שָׁוְא וּדְבַר כָּזָב / יַחְדָּו יִכּוֹנוּ עַל שְׂפָתַי

מֵאֲנָה לִנְגֹּעַ נַפְשִׁי / דִּבְרֵי חֲכָמִים וְחִידוֹתָם

לֹא יִישְׁנוּ יְצוּרַי אִם לֹא יָרֵעוּ / וְנִגְזְלָה שְׁנָתָם אִם לֹא יַכְשִׁילוּ

עָרֹם הִצִּיגוּנִי מִבְּלִי צְדָקָה / כְּמַדִּי הִלְבִּישׁוּנִי פֶּשַׁע וּפוּקָה

בִּשְׁאוֹן גַּלֵּיהֶם עֲלֵיהֶם הִגְדַּלְתִּי / שָׂחוּ זִמּוֹתַי וְשׁוּב לֹא יָכֹלְתִּי

עַל כֵּן בָּאתִי לְהוֹדוֹת לְפָנֶיךָ / אוּלַי אוּכַל לַעֲלוֹת מִבּוֹר שְׁאוֹנָם

וְהִנְנִי שׂוֹחֶה נֶגֶד הֲדוֹם רַגְלֶךָ / וּמִתְוַדֶּה נֹכַח אֲרוֹן עֻזֶּךָ

וּמַבְהִיל עֲצָמַי עַל עֹצֶם אֲשָׁמַי / וּמַזִּיל דְּמָעַי עַל רֹב פְּשָׁעַי

וּמִתְחָרֵט בְּלִבִּי עַל עֲווֹן חֻבִּי / וּבוֹכֶה בְמִסְתָּרִים עַל חֵטְא נְעוּרִים

כִּי לְגַלּוֹתָם בְּשָׂפָה לֹא אוּכַל שְׂאֵת / וּלְזָכְרָם בְּלָשׁוֹן לֹא אֶעְצָר כֹּחַ

כִּי אִם אָבוֹא לְפָרֵט מִקְצָתָם / הַזְּמַן יִכְלֶה וְהֵם לֹא יִכְלוּ

עַל אֵי זֶה מֵהֶם אֶדְרֹשׁ אוֹ אֲבַקֵּשׁ / עַל אֵי זֶה מֵהֶם אֶתְבַּע אוֹ אֶתְוַדֶּה

עַל הַזְּדוֹנוֹת אוֹ עַל הַשְּׁגָגוֹת / עַל הַחֲבוּיוֹת אוֹ עַל הָרְאוּיוֹת

יוֹדֵעַ אֲנִי בְנַפְשִׁי יְיָ אֱלֹהַי / כִּי רֵק לִקְחִי מִנִּי צֶדֶק

אֲשֶׁר שָׂנֵאתָ אָהַבְתִּי / וַאֲשֶׁר אָהַבְתָּ שָׂנֵאתִי

אֲשֶׁר רִחַקְתָּ קֵרַבְתִּי / וַאֲשֶׁר קֵרַבְתָּ רִחַקְתִּי

אֲשֶׁר אָסַרְתָּ הִתַּרְתִּי / וַאֲשֶׁר הִתַּרְתָּ אָסַרְתִּי

אָהַבְתִּי רַע מִטּוֹב / שֶׁקֶר מִדַּבֵּר צֶדֶק סֶלָה

פָּנִיתִי אֶל רְהָבִים וְשָׂטֵי כָזָב / וְאַחֲרֵי בִצְעִי לִבִּי הוֹלֵךְ

אָשַׁמְתִּי [בְתוֹרָתֶךָ] נִאַצְתִּי מֵעֲשׂוֹת



חֻקֶּיךָ / גָּעַלְתִּי [דְבָרֶךָ] מָאַסְתִּי מִשְׁפְּטֵי צִדְקֶךָ

הֵפַרְתִּי בְרִית וּבִטַּלְתִּי מִצְוָה / זָרַעְתִּי אָוֶן חָרַשְׁתִּי מַעֲוָה

טָפַלְתִּי שֶׁקֶר יָעַצְתִּי דִבָּה / כִּזַּבְתִּי לַצְתִּי מָלֵאתִי חוֹבָה

נָעַלְתִּי סָגַרְתִּי עָלַי פִּתְחֵי הַתְּשׁוּבָה / צָרַרְתִּי חַטָּאתִי קָשַׁרְתִּי מִרְמָה

רָדַפְתִּי שִׁחַתִּי תָּאַבְתִּי זִמָּה / שָׁגִיתִי לוֹצַצְתִּי מָעַלְתִּי הִגְדַּלְתִּי עָווֹן פְּלִילִי

חָטָאתִי זַדְתִּי וְיָשָׁר הֶעֱוֵיתִי / קַצְתִּי בְכָל מִצְווֹתֶיךָ וְלֹא שָׁוָה לִי

בְּעַזּוּת מֵצַח בָּאתִי לְבַקֵּשׁ סְלִיחָה / וְלִתְבֹּעַ מְחִילָה עַל כֹּבֶד עֲווֹנִי

אֲבָל עַל רֹב רַחֲמֶיךָ נִשְׁעַנְתִּי / וָאֵדַע כִּי לֹא אֵבוֹשׁ

וָאֶשָּׂא בְיָדִי מִנְחָה מִשִּׁיר עֶרֶב / וָאָבוֹא לִדְפֹּק דַּלְתֵּי רַחֲמֶיךָ

וָאֶקַּח עִמִּי זְכוּת אֲבוֹתַי / וְחַסְדֵּי נְבִיאַי הַקַּדְמוֹנִים

וָאֶחְקֹר וָאֶחְפֹּשׂ סִתְרִי / וָאֶתפֹּר קֶרַע חִסְרִי

וָאֹמַר אוֹדֶה עֲלֵי פְשָׁעַי נֶגְדְּךָ / אוּלַי תִּשָּׂא עֲווֹן חַטָּאתִי סֶלָה

וּבְעָמְדִי לְפָנֶיךָ רָחֲפוּ כָּל עַצְמוֹתַי / וְנִבְהֲלוּ קְרָבַי וְנִתְּקוּ זִמּוֹתַי

וַיֶּחֱרַד לִבִּי וַיִּתַּר כְבֵדִי / וַיִּמְעֲדוּ אֲשׁוּרַי וַתִּרְפֶּינָה יָדָי

יָגֹרְתִּי כָל עַצְּבוֹתָי / יָדַעְתִּי כִּי לֹא תְנַקֵּנִי

אֲדֹנָי שְׂפָתַי תִּפְתָּח / וּפִי יַגִּיד תְּהִלָּתֶךָ.
