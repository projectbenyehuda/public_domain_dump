


	
	אֲשֶׁר תֵּשֵׁב
	



ולו בנעוריו יקונן עח אדם הרוג, זכרו לברכה.






אֲשֶׁר תֵּשֵׁב שְׁכוּלָה, כַּנְּהָרוֹת / דְּמָעוֹת עַל
לְחָיֶיהָ פְזוּרוֹת



וְתֵאָנַח וְיָד תָּשִׂים לְלֶחְיָהּ / וְהִיא מָרָה
כְּהָמֵר עַל בְּכוֹרוֹת



וּפָנֶיהָ יְדַעְתִּים כַּמְּאוֹרוֹת – / וְנֶהְפְכוּ
בְּבִכְיָתָהּ שְׁחֹרוֹת –



לְמָה תִבְכִּי, כְּלִילַת הַנְּעָרוֹת, / וְלֹא מִטַּל
עֲנָנַיִךְ עֲצוּרוֹת



עֲלֵי פֵרוּד יְדִידַיִךְ, כְּאִלּוּ / גְוִיּוֹתָם
לְפָנַיִךְ קְבוּרוֹת?



וּמִדֵּי חַי אֱנוֹשׁ רֵעָיו יְקַוֶּּה – / וְאֵין מִקְוֶה
לְשׁוֹכֵב בַּקְּבָרוֹת



בְּכִי בָכֹה עֲלֵי חַיִּים וּמוֹת אִישׁ / אֲשֶׁר נָפַל
בְּאַחַת הַיְּעָרוֹת!



וְאִלּוּ מֵת כְּמוֹת רַבִּים בְּבֵיתוֹ / וְיִקָּבֵר
בְּכָבוֹד וַהֲדָרוֹת –



אֲזַי הָיָה לְלִבִּי קָט הֲנָחָה / וְנִחוּמִים וְחֹבֶשׁ
לַשְּׁבָרוֹת



וְלֹא יָמוּת מְטֹעָן בַּחֲרָבוֹת / וּבַחְרוּצוֹת אֲשֶׁר
בָּאֵשׁ בְּעוּרוֹת



וְלֹא תֵעָשׂ גְּוִיָּתוֹ כְבָרָה / וְיִדָּקֵר בְּכָל
גּוּפוֹ דְקָרוֹת



אֲשֶׁר נִשְׁלַח הֱיוֹת עֶזְרָה בְצָרוֹת / וְהָלַךְ
לַהֲדֹף חֵילֵי מְצוּרוֹת ־



טְרָפוּהוּ הֲדָפוּהוּ לְבָאִים / וְהָאְרָיוֹת וְגוֹרֵי
הַכְּפִירוֹת



אֲשֶׁר עָמְדוּ וְרָמְחֵיהֶם מְרוּקִים / וְחַרְבוֹתָם
שְׁלוּפוֹת מִתְּעָרוֹת



חֲרָבוֹת נוֹשְׁכוֹת כָּאֵשׁ תְּלַהֵט / תְּגוֹרֵר
הָאֲבָנִים כַּמְּגֵרוֹת



אֲדוֹמִים שָׁפְכוּ דַם אִישׁ נְקִי כַף / וְנֶפֶשׁ
הָרְגוּ מֵהַבְּחוּרוֹת



וְ״הֶאָח!״ אָמְרוּ עָלָיו בְּהָרְגוֹ / וְלִבּוֹ סָר
וְעוֹפֵף בָּאֲוֵירוֹת



שְׁחָטוּהוּ כְּאַחַת הָעֲגָלוֹת / וְגַם הוּבַל אֱלֵי
טֶבַח כְּפָרוֹת



וְהִתְגּוֹלֵל בְּדָמוֹ בַמְּסִלָּה / וְהִרְתִּיחַ
כְּבָשָׂר בַּקְּדֵרוֹת



לְדָמוֹ רַעֲשָׁה אֶרֶץ בְּשָׁפְכוֹ / וְהָרִים רָגְזוּ
עָלָיו, וְהָרוֹת.



וְלוּ רָאָה וְיָדַע מַחְשְׁבוֹתָם – / אֲזַי בָא
מִפְּנֵיהֶם בַּמְּעָרוֹת



אֲבָל עֶת בּוֹא פְקֻדַּת אִישׁ וְיוֹמוֹ / הֲיִמָּלֵט,
וְאִם יִמְצָא עֲזָרוֹת?



וְאֵין שִׁלְטוֹן בְּיוֹם מָוֶת לְאָדָם / וְלֹא יַעְבֹר
אֱנוֹשׁ מִיּוֹם עֲבָרוֹת



אֲהָהּ! אֵיךְ הֻתְּשָׁה חוֹמָה בְחֵמָה / אֲשֶׁר הָיְתָה
בְּצוּרַת הַבְּצוּרוֹת



וְחוֹמַת אֵשׁ סְבִיב כָּל הַמְּדִינוֹת / וּמִגְדַּל עֹז
עֲלֵי כָל הָעֲיָרוֹת



וְהַר תָּבוֹר מְחַזֵּק לֵב חֲשוּלִים / וְלִצְמֵאִים
בְּצַחְצָחוֹת מְקוֹרוֹת



וְצוּף אוֹהֵב וְרוֹשׁ אוֹיֵב, אֲבָל כִּי / בְּכָל
כֹּחוֹ, אֱמֶת, הִמְתִּיק מְרוֹרוֹת!



תְּמִים לֵבָב וּבַר מִכָּל נְבַר לֵב / אֲשֶׁר אָחַז
אֲשׁוּרוֹת בַּיְּשָׁרוֹת ־



לְכֵן כָּל יוֹדְעָיו חָלוּ לְשִׁמְעוֹ / וְאָחְזוּ צִיר
כְּצִירֵי הַמְּצֵרוֹת



וְקִינָתָם בְּדִמְעָתָם עֲשׂוֹתָם / בְּחַיּוּתָם ־
תְּמוּר יַיִן וְשִׁירוֹת



וְתַחַת מַעֲטֵה כָבוֹד וְהָדָר – / חֲגוּרִים מִלְּבוּשׁ
שַׂקִּים חֲגוֹרוֹת



חֲפוּיֵי רֹאשׁ ־ מְקוֹם הוֹד הָעֲטָרוֹת / וְשִׁית אֵפֶר
עֲלֵיהֶם וַעֲפָרוֹת



וְשֶׁבֶת דֹּם בְּלֵב יֵחַם כְּתַנּוּר / מְקוֹמָם ־
מֵעֲלִיּוֹת הַמְּקֵרוֹת



וְכָל בָּנוֹת עֲדִינוֹת שַׁאֲנַנּוֹת / אֲשֶׁר בָּרֹךְ
וּבַתַּפְנוּק זְכוּרוֹת



מְקוֹנְנוֹת הֵן וְאַפֵּיהֶן גְּדוּדוֹת / וְכָל עֶרֶב
וְכָל שַׁחַר שְׁחֹרוֹת



וּמַר צָרְחוּ וְרוֹשׁ לַעֲנָה בְּמוֹת רֹאשׁ / לְכָל
רָאשִׁים וְשָׂבְעוּ מַמְּרוֹרוֹת



וְכוֹס רַעַל ־ תְּמוּרַת הָרְעָלוֹת / חֲבֹשׁ חֶבֶל ־
מְקוֹם נֵבֶל וְשִׁירוֹת



וְהֵן הוֹלְכוֹת וְזוֹ אֶל זוֹ סְמוּכוֹת / כְּיוֹלֶדֶת
בְּצִירִים, לֹא עֲצוּרוֹת



וּבוֹכוֹת עַל אֲדוֹנֵיהֶן אֲבִיהֶן / וְהוֹמִיּוֹת
זְקֵנוֹת עִם נְעָרוֹת



וְכֻלָּן נוֹשְׂאוֹת קוֹלָן בְּחִילָן / כְּאִלּוּ יָלְדוּ
אוֹתוֹ עֲקָרוֹת



וְאֵיכָה לֹא יְכֻבּוּ הַמְּאוֹרוֹת / לְאוֹר תֵּבֵל
אֲשֶׁר כִּבָּה נְהָרוֹת



וְכִמְעַט קָט ־ וְתִבְכֶּינָה אֲבָנִים / וְכִמְעַט קָט ־
תְּפוּצֶינָה נְהָרוֹת!



בְּכוּ עָלָיו, קְהִלּוֹת הַיְּשָׁרִים, / וְיִהְיוּ
מַעְיְנֵיכֶם כַּיְּאוֹרוֹת



וְשִׂימוּ יוֹם כְּיוֹם גָּלוּת בְּכִיתוֹ / וְיוֹם אֵבֶל
יְהִי לָכֶם לְדוֹרוֹת



וּמַהְרוּ נָא וְקִרְאוּ לַחֲכָמוֹת / וְשִׁלְחוּ עוֹד
וְהוֹסִיפוּ אֲחֵרוֹת



וְסִפְדוּ לוֹ כְּמִסְפַּד הַמְּדֻקָּר / מְחַדֵּשׁ בֵּית
אֱלֹהִים בַּטְּהָרוֹת



וְכִתְבוּ אֶל פְּלִילֵי דָת, גְּאוֹנֵי / עֲגֻלּוֹת
כַּחֲצִי גֹרֶן וְשׁוּרוֹת



וְשִׁלְחוּ שִׁיר אֲלֵיהֶם, חִבְּרוֹ שָׁר / דְּבָרִים
מִבְּדֹלָחִים יְקָרוֹת



אֲשֶׁר בָּחַן אֲמָרוֹת הַצְּרוּפוֹת / וְגַם לִבֵּן
וּבֵרַר הַבְּרוּרוֹת



וְנִשְׁקָלוֹת בְּפֶלֶס מַחֲשָׁבוֹת / וּמֵאָסְפֵי
תְעוּדוֹת הֵן אֲגוּרוֹת



תְּמִימוֹת הֵן בְּמִשְׁפָּטָן וְחֻקָּן /
בְּמַתְכֻּנְתָּן וְאֵינֵיהֶן שְׁבוּרוֹת



יְשִׂיחוּהֶן קְצִינֵי הַחֲבוּרוֹת / כְּשִׂיחָתָם בְּחֹק
דָּתוֹת וְתוֹרוֹת



אֲשֶׁר יַטּוּ בְחִנָּם הַלְּבָבוֹת / וְכָל נֶפֶשׁ
יְשֵׁנָה הֵן מְעִירוֹת



בְּכוֹת עַל אִישׁ אֲשֶׁר הֵסִיר מְשֻׂכָּה / בְּיוֹם
מוֹתוֹ וּפָרַץ הַגְּדֵרוֹת



וְעַל אֶרֶז אֲשֶׁר הָיָה מְגוּרוֹת / אֲשֶׁר הֶאְרִיךְ
סְעַפּוֹת לוֹ וּפֹארוֹת



וְכָל עוֹף שָׁכְנוּ תַחְתָּיו וְקִנְנוּ / וְשָׂחוּ בֵּין
סְעַפּוֹתָיו מְדוֹרוֹת



כְּזַיִת רַעֲנָן הָיָה ־ וְנִכְרַת / בְּיַד אוֹיֵב
וְשָׁרָשָׁיו עֲקוּרוֹת



יְעַַקֵּר עוֹקְרָיו הָאֵל מְעַקֵּר / חֲמֵשֶׁת
הַמְּלָכִים בַּמְּעָרוֹת!



נְתָשׁוּהוּ, נְתָצוּהוּ סְעָרוֹת –/ אֲחָזוּנוּ לְשֹׁד
שִׁבְרוֹ שְׂעָרוֹת



לְשִׁמְעוֹ צָלְלוּ אָזְנוֹת סְגוּרוֹת / וְעָלָיו
רָעֲשׁוּ בָתִּים וְקוֹרוֹת



אֲסָפוֹ אֵל וְרוּחוֹ הַטְּהוֹרָה / וְשָׂמוֹ בַנְּפָשׁוֹת
הַטְּהוֹרוֹת



וְנֵרוֹ יַעֲלֶה תָמִיד לְנֵרוֹת / עֲלוֹת שֶׁמֶשׁ עֲלֵי
כָל הַמְּאוֹרוֹת.
