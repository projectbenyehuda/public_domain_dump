


	
	שַׁלִּיט יוֹצֵר
	








מִי כָמוֹךָ וְאֵין כָּמוֹךָ מִי דוֹמֶה לָךְ וְאֵין דּוֹמֶה לָךְ





שַׁלִּיט יוֹצֵר כָּל פֹּעַל / בְּתַחַת וּבְמַעַל / מוֹדֵד מַיִם בַּשֹּׁעַל / הַנּוֹטֶה כַדֹּק שָׁמַיִם



לוֹבֵשׁ צְדָקָה כַּשִּׁרְיוֹן / אֵל מִסְתַּתֵּר בְּחֶבְיוֹן / נוֹרָא וְעֶלְיוֹן / בַּהֲכִינוֹ שָׁמַיִם



מוֹשֵׁל בִּגְבוּרָתוֹ עוֹלָם / מְחַיֶּה יְצוּרִים כֻּלָּם / בָּרָא שְׁחָקִים וּכְלָלָם / וּצְבָא הַשָּׁמַיִם



הַשָּׂם עָבִים רְכוּבוֹ / וּבְמַיִם עַזִּים נְתִיבוֹ / עָנָן וַעֲרָפֶל סְבִיבוֹ / הַיּוֹשְׁבִי בַּשָּׁמַיִם



– – – – – – / – – – – – / – – – – – / – – – – – 



אַחֲרִישׁ הֶגְיוֹן / וְאֶשְׁאַל רִשְׁיוֹן / אֱלֹהִים עֶלְיוֹן / נוֹטֶה שָׁמַיִם



בָּרָא בְּמַבָּע / יְסוֹדוֹת אַרְבַּע / תָּקַע וְקָבַע / גָּבְהֵי שָׁמַיִם



גָּזַר נָאוֹר / וַיְהִי אוֹר / הוֹלֵךְ וָאוֹר / תַּחַת הַשָּׁמַיִם



דִּלָּה זְרוּמִים / לְגָבְהֵי מְרוֹמִים / וְהִקְוָה תְהוֹמִים / מִתַּחַת הַשָּׁמַיִם



הֵצִיץ דְּשָׁאִים / וְכָל צֶאֱצָאִים / וְזִיז וּפְרָאִים / וְעוֹף הַשָּׁמַיִם



וְהֵרִיץ מְאוֹרוֹת / וְהִשְׁרִיץ יְאוֹרוֹת / וְרִקֵּם צוּרוֹת / וַיְכֻלּוּ הַשָּׁמַיִם



זְהוּרִים עָדוּ / וְצָדוּ וְזָדוּ / נָעוּ וְנָדוּ / מִתַּחַת הַשָּׁמַיִם



חִבְּקוּ אֶשְׁכָּר / רוֹעֶה וְאִכָּר / וְצָעִיר נִזְכַּר / לְאֵל הַשָּׁמַיִם



טָחֹה טָחָה / וְאֶחָיו דָּחָה / זִכְרוֹ מָחָה / מִתַּחַת הַשָּׁמַיִם



יְהִירִים עָמְסוּ / וְנָתִיב חָמְסוּ / נִחֲשׁוּ וְכִחֲשׁוּ / בֵּאלֹהֵי שָׁמַיִם



כֶּמֶס רִבָּה / וְהִפְלִיא סִבָּה / בִּתְהוֹם רַבָּה / וַאֲרֻבּוֹת הַשָּׁמַיִם



לְאֻמִּים הִתִּיר / וְעַבְדּוֹ הִסְתִּיר / קִטֵּר וְהֶעְתִּיר / לֵאלֹהֵי הַשָּׁמַיִם



מְרֵעִים שָׁלוּ / וְגָאוּ וְעָלוּ / וּמִגְדָּל תָּלוּ / וְרֹאשׁוֹ בַשָּׁמַיִם



מְצָדָם נִשְׁחַק / נָדוּ לְמֶרְחָק / לָהֶם שָׂחַק / יוֹשֵׁב בַּשָּׁמַיִם



נָדִיב נוֹדָע / מֵאוּר נִפְדַּע / אֶזְרָח [יָ]דַע / חֻקּוֹת שָׁמַיִם



סָחַף שָׂרִים / וְסֵעַף זָרִים / יָדוֹ הֵרִים / אֶל אֵל בַּשָּׁמַיִם



עָמַד וְשָׁקַד / וּבְנוֹ עָקַד / זִכְרוֹ פָקַד / כִּימֵי הַשָּׁמַיִם



פָּצָה עֲתֶרֶת / לְנֹכַח גְּבֶרֶת / זָכַר עֲקֶרֶת / בּוֹרֵא הַשָּׁמַיִם



צֶאֱצָאֵי תְאוֹמִים / שְׁעָרִים וְיָמִים / זְכָרוֹ לְרַחֲמִים / נִצָּב בַּשָּׁמַיִם



קָפַץ תָּמִים / מֵאִישׁ דָּמִים / הוֹלִיד רְחוּמִים / עַמּוּדֵי שָׁמַיִם



רְגִישַׁת שְׁבָטִים / יָרְדָה לְפוּטִים / וְרַבּוּ מְעוּטִים / כְּכוֹכְבֵי הַשָּׁמַיִם



שָׁאוֹן תַּקִּיף / מְצוּדוֹ הִקִּיף / רָאָה וְהִשְׁקִיף / יְיָ מִשָּׁמַיִם



שִׁבַּר גְּאוֹנָם / וְשִׁקַּע שְׁאוֹנָם / כָּל[וּ] וְאֵינָם / עַד בִּלְתִּי שָׁמַיִם



תַּמָּה חֲפוּשָׁה / בַּיָּם חִדְּשָׁה / שִׁירָה חֲדָשָׁה / לְעוֹשֵׂה הַשָּׁמַיִם.
