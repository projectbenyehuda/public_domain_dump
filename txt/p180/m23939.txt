


	
	שַׁפְרִיר וָאָרֶץ
	








שַׁפְרִיר וָאָרֶץ בְּטֶרֶם יָצַרְתָּ אֵל

וְעַד לֹא נוֹדַעְתָּ לְעַמְּךָ יִשְׂרָאֵל

מָרוֹם הָיִיתָ אֵל פּוֹדֶה וְגוֹאֵל

אֵל קָדוֹשׁ יוֹשֵׁב קְהִלּוֹת יִשְׂרָאֵל





יְיָ אֵל רִאשׁוֹן הֵאִיר בְּמֶרֶץ

יְיָ בְּאֵין רֵאשִׁית וְאֵין תַּכְלִית וְחֶרֶץ

יְיָ גָּדַר חוּג שַׁפְרִירֵי עֶרֶץ

יְיָ דִּבֶּר וַיִּקְרָא אָרֶץ





יְיָ הוּא הַשֵּׁם אֲשֶׁר שַׁדַּי פְּסָקוֹ

יְיָ וֵאלֹהִים לְשֵׁשׁ פְּשׁוּטוֹת חֲלָקוֹ

יְיָ זִהֵר בָּם קַצְוֵי חוּג בְּחֻקּוֹ

יְיָ חָפֵץ לְמַעַן צִדְקוֹ





יְיָ טִכֵּס רֵאשִׁית מוּל מַעְלָה לְמָדוֹר

יְיָ יְסָדָהוּ בְּיוֹד הֵי וָו לִגְדֹּר

יְיָ כֹּחֲךָ זֶה מִי יוּכַל לִסְדֹּר

יְיָ לְעוֹלָם תֵּשֵׁב כִסְאֲךָ לְדוֹר וָדוֹר





יְיָ מָדַד שֵׁנִית מוּל מַטָּה בְּיָדוֹ

יְיָ נָטָה בְּיוֹד וָו הֵי יְסוֹדוֹ

יְיָ סָתַם סוֹד וּמִי יֵדַע בְּסוֹדוֹ

יְיָ עַל הַשָּׁמַיִם כְּבוֹדוֹ





יְיָ סָתַם שְׁלִישִׁית מוּל מִזְרָח בְּמֶרֶץ

יְיָ עִטְּרָהוּ בְּהֵי וָו יוֹד בְּתֶרֶץ

יְיָ פָּעַל כֹּל וּמִי יִפְרֹץ פֶּרֶץ

יְיָ צְבָאוֹת מְלֹא כָל הָאָרֶץ





יְיָ קָבַע רְבִיעִית מוּל מַעֲרָב וּפְרָזוֹ

יְיָ רָם וְנִשָּׂא בְּהֵי יוֹד וָו אֲחָזוֹ

יְיָ שׁוֹכֵן עַד הַפּוֹעֵל כָּל זוֹ

יְיָ בִּימִינוֹ וּבִזְרוֹעַ עֻזּוֹ





יְיָ תִּכֵּן חֲמִישִׁית מוּל דָּרוֹם צוֹעֵד

יְיָ שָׁתוֹתָיו בְּוָו יוֹד הֵי סוֹעֵד

יְיָ לוֹ הָעֹז אֲשֶׁר הַכֹּל וִעֵד

יְיָ מֶלֶךְ עוֹלָם וָעֶד





יְיָ הִתְאִיר שִׁשִּׁית מוּל צָפוֹן וְדָכְיָהּ

יְיָ חָתַם בְּוָו הֵי יוֹד שִׁפְיָהּ

יְיָ זֶה קָרָא לְכָל פְּעֻלָּה דַיָּהּ

יְיָ אֱלֹהֵי צְבָאוֹת מִי כָמוֹךָ חֲסִין יָהּ.
