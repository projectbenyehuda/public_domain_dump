


	
	מָעוֹן לְךָ
	








וּבְכֵן מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יַאֲמִירוּן יֹאמֵרוּן / אֱמוּנִים אֵיתָנִים / אֲצִילֵי תֵבֵל



וּלְךָ לְמַעְלָה / יְאַדְּרוּן יְאַשְּׁרוּן / אֵלִים אֶרְאֶלִּים / אוֹפַנֵּי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְנוֹבְבוּן יְנוֹפְפוּן / נְדִיבִים נְצִיבִים / נְדוּדֵי תֵּבֵל



וּלְךָ לְמַעְלָה / יְנַצְּחוּן יְנַשְּׂאוּן / נִשְׂגָּבִים נִצָּבִים / נוֹצְצֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְיַקְּרוּן יְיַשְּׁרוּן / יְדִידִים יְחִידִים / יְדוּעֵי תֵבֵל



וּלְךָ לְמַעְלָה / יְיַסְּדוּן יְיַחֲדוּן / יְקָרִים יְשָׁרִים / יְקוּדֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְשׁוֹרְרוּן יְשַׁחֲרוּן / שְׁחוּיִים שְׁאוּיִים / שְׁכוּנֵי תֵבֵל



וּלְךָ לְמַעְלָה / יַשְׁפִּיעוּן יַשְׁמִיעוּן / שִׁנְאַנִּים שַׁאֲנַנִּים / שְׁקוּפֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְלַהֲגוּן יְלַמְּדוּן / לְהוּקִים לְקוּקִים / לְכוּדֵי תֵבֵל



וּלְךָ לְמַעְלָה / יְלַבְּדוּן יְלַבְּבוּן / לוֹהֲטִים לוֹאֲטִים / לְאַפְסֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְמַלְּלוּן יַמְשִׁילוּן / מְרוּדִים מְעוּדִים / מְעוּטֵי תֵבֵל



וּלְךָ לְמַעְלָה / יַמְלִיכוּן יִמָּכְכוּן / מְנוֹצְצִים מְרוֹצְצִים / מַלְאֲכֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְהוֹדוּן יְהַדְּרוּן / הֲדוּכִים הֲמוּכִים / הֲלוּמֵי תֵבֵל



וּלְךָ לְמַעְלָה / יֶהְגּוּן יֶהֱמוּן / הִלּוֹת הֲמֻלּוֹת / הֲמוֹנֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְקוּמוּן יְקַדְּמוּן / קְשׁוּטִים קְמוּטִים / קְרוּאֵי תֵבֵל



וּלְךָ לְמַעְלָה / יִקָּפְצוּן יִקָּבְצוּן / קְצִינִים קְרוּנִים / קְדוֹשֵׁי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְטַיְלוּן יְטִיבוּן / טְפוּחִים טְרוּחִים / טְחוּיֵי תֵבֵל



וּלְךָ לְמַעְלָה / יְטוֹפְפוּן יַטִּיפוּן / טָסִים טְכוּסִים / טַפְסְרֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יִנְאֲמוּן יַנְעִימוּן / נְכוֹנִים נְבוֹנִים / נְפוּצֵי תֵבֵל



וּלְךָ לְמַעְלָה / יְנוֹפְפוּן יְנִיפוּן / נִשְׁקָפִים נִזְקָפִים / נְצִיבֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְגִילוּן יַגְדִּילוּן / גְּאוּלִים גְּמוּלִים / גְּזוּרֵי תֵבֵל



וּלְךָ לְמַעְלָה / יִגְמְרוּן יַגְבִּירוּן / גָּשִׁים גּוֹעֲשִׁים / גְּדוּדֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְבָאֲרוּן יְבָרְרוּן / בְּחִירִים בְּהִירִים / בְּרוּרֵי תֵבֵל



וּלְךָ לְמַעְלָה / יְבַצְּרוּן יְבַשְּׂרוּן / בְּרָקִים בּוֹהֲקִים / בִּזְקֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יַיְשִׁירוּן יַיְקִירוּן / יְשָׁרִים יְקָרִים / יְסוֹדֵי תֵבֵל



וּלְךָ לְמַעְלָה / יֵיטִיבוּן יְיַבְּבוּן / יְקוּדִים יְעוּדִים יְצוּקֵי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יִרְחֲשׁוּן יִרְעֲשׁוּן / רְעוּצִים רְצוּצִים / רְמוּסֵי תֵַבֵל



וּלְךָ לְמַעְלָה / יְרוֹמְמוּן יְרַנְּנוּן / רְחוּפִים רְצוּפִים / רִשְׁפֵּי מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יוֹדוּן יִוָּּעֲדוּן / וְלָדִים וְלִמּוּדִים / וְתִיקֵי תֵבֵל



וּלְךָ לְמַעְלָה / יו יו / ו ו / ו מָעוֹן





מָעוֹן לְךָ לְמַטָּה / יְלַהֲגוּן יְלַחֲשׁוּן / לְעוּטִים לְקוּטִים / לְתַנִּינֵי תֵבֵל



וּלְךָ לְמַעְלָה / יְלַבְּבוּן יְלַבְּדוּן / לַפִּידִים לְמוּדִים / לְהוּקֵי מָעוֹן.
