


	
	וְאָלִין וַאֲנִי נִבְהָל
	






וְאָלִין וַאֲנִי נִבְהָל כְּאִלּוּ / שְׁנַת עֵינַי עֲלֵי עֵינַי אֲסוּרָה



וְהַשַּׁחַק כְּאֹהֶל בַּחֲבָלָיו / אֲשֶׁר הַלַּיְלָה בָהֶם קְשׁוּרָה



וּמַרְאֵה הַלְּבָנָה בַעֲיָשִׁים / כְּכַלָּה נַעֲרוֹתֶיהָ מְאִירָה



כְּאַיֶּלֶת סְבָבוּהָ צְבָאִים / אֲשֶׁר הַלַּיְלָה לָהֶם כְּפִירָה



וְהַכִּימָה בְּכוֹכָבִים אֲגוּדָה / כְּנֵרוֹת הֹעֲלוּ עַל הַמְּנוֹרָה



וְרֹאשׁ הַלַּיְלָה כָּפַף בְּשֵׂיבוֹ / כְּבֶן שִׁשִּׁים וְעֶשְׂרִים
וַעֲשָׂרָה



וְהִנֵּה כוֹכְבֵי שַׁחַק לְנֶגְדִּי / כְּרָצִים שׁוֹאֲלִים קַחַת
בְּשׂוֹרָה



וְעָלוּ כַעֲלוֹת שָׂרֵי תְעוּדָה / עֲלוֹת שֶׁמֶשׁ בְּאֶרֶץ לֹא סְדוּרָה



מְלִיצֵי הַנְּבוֹנִים הַגְּאוֹנִים / וְחַכְמֵי דָת יְשׁוּרוּן
הַטְּהוֹרָה



שְׁנֵיהֶם הַמְּאוֹרִים הַהֲדוּרִים / שָׁלֹשׁ קָנִים שְׁנֵי צִדֵּי
מְנוֹרָה



וְזֶה סַפִּיר, וְזֶה נֹפֶךְ וְיַהְלוֹם, / וְזֶה תַרְשִׁישׁ, וְזֶה אֶבֶן
יְקָרָה



בְּנֵי הַחֵן וּבַעְלֵי הָאֱמוּנָה, / אֲחֵי כָבוֹד, בְּנֵי מַשְׂאֵת
וְיִתְרָה



בְּדָת אֵל סָחֲרוּ תָמִיד, אֲשֶׁר לָהּ / מְאֹד יִתְרוֹן, וְעָזְבוּ כָל
סְחוֹרָה



וְהִנֵּנוּ מְחַכִּים בּוֹא אֲלֵיהֶם / שְׁאֵלוֹת מִנְּהַרְדֵּעָא וְסוּרָא



יְשִׁיבוּהֶם [בְּחָכְמָתָם] תְּשׁוּבוֹת / בְּמִלִּים יִפְתְּחוּ דֶלֶת
סְגוּרָה



וְהֵם שָׂמוּ בְיָדֵינוּ חֲרָבוֹת / שְׁלוּפוֹת לַעֲמֹד לָנוּ בְצָרָה.
