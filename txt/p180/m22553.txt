


	
	כשרש עץ
	








כְּשֹׁרֶשׁ עֵץ יְהִי אֹרֶךְ אֲמִירָיו / וְכִתְבוּנַת אֱנוֹשׁ יֹשֶׁר
אֲמָרָיו



וְכִמְזִמַּת לְבַב אִישׁ כֵּן יְגוֹנוֹ / בְּשׁוּר, טֶרֶם יְבוֹאוּהוּ,
מְגוּרָיו.



יְדִידַי! אָהֳלוֹ תָקַע בְּלִבִּי / וּמוֹרָשֵׁי לְבָבִי הֵם יְתָרָיו



וְנִשְׁבַּע, כָּל אֲשֶׁר יִסַּע וְיֹאבֶה / נְסֹעַ הַכְּאֵב, לִתְמֹךְ
אֲשׁוּרָיו



וְשָׂם תֵּבֵל בְּיָדִי כּוֹס יְגוֹנִים / וְעָלַי הָפְקְדָה לִמְצוֹת
שְׁמָרָיו



וְהִפְקִיד הַזְּמַן כָּל עַצְּבוֹתָיו / בְּנַפְשִׁי, כִּי אֲנִי נֶפֶשׁ
פְּגָרָיו



וּפָגַשׁ קָדְקָדִי בַעְנַן תְּלָאוֹת / וְעָלַי סָחֲפוּ גִשְׁמֵי מְטָרָיו



וּפֶתֶן הַזְּמַן נָשַׁךְ עֲקֵבַי / וְהָיִיתִי מְשַׁעְשֵׁעַ לְחוּרָיו



יְמֵי רִיבוֹת יְעָדַנִי ־ וְהִנֵּה / כְּבָר שָׁלֵם, וְלֹא שִׁלֵּם
נְדָרָיו



וְכֹל אָעוּף עֲלֵי גָבְהֵי מְזִמּוֹת ־ / יְקַצֵּץ בַּעֲדִי כַּנְפֵי
נְשָׁרָיו



וְהָיִיתִי כְּמוֹ נֶשֶׁר אֲשֶׁר כָּל / בְּרוּם יִגְבַּהּ ־ וְיִמָּרְטוּ
אֲבָרָיו



וְרַק אֶשְׂחֶה בְּיָם הַבִּין לְבַדִּי / לְלַקֵּט כָּל בְּדֹלָחָיו
וְדָרָיו



וְדַי לִתְפֹּשׂ בְּיָדִי מֶחְקְרֵי בִין / וְדַי לָלִין לְבַדִּי בֵּין
מְצָרָיו



כְּמוֹ חוֹשֵׁב אֲשֶׁר יַעֲשֶׂה מְלַאכְתּוֹ / וְכַסּוֹפֵר יְחַפֵּשׂ אֶת
סְפָרָיו



וְאֵיךְ יוּכַח אֱנוֹשׁ יַמְרֶה לְבָבוֹ / וְלָמָּה יֹאמְרוּ: ״הַתֵּר
חֲגוֹרָיו,



מְגוּרָיו קָרְבוּ בִרְחֹק זְמָמוֹ / וְרֵעָיו רָחֲקוּ מִבֵּית מְגוּרָיו״.



וְלָמָּה יֹאמְרוּ: ״יִיגַע לְנַפְשׁוֹ / וְלִלְבָבוֹ כְּהַמָּס
לַאֲסוּרָיו



וְיִרְאֶה הָאֱנוֹשׁ חוֹבוֹת אֲחֵרִים / וְיַעְלִים עֵין לְבָבוֹ
מִמְּעוֹרָיו״.



וְאֵיךְ יַתִּיר חֲגוֹר שֵׂכֶל וּבִינָה / אֱנוֹשׁ מֵהַמְּרִי פִּתַּח
אֲזוֹרָיו



אֲשֶׁר סִגַּר בְּיַד נַפְשׁוֹ בְשָׂרוֹ / וְהִפְקִיד אֶת לְבָבוֹ עַל
נְעוּרָיו



וְנִפְשַׁט מִמְּתֵי בוּז כַּשְּׁפִיפוֹן / מְצָאוּהוּ בְיוֹם קַיִץ
מְרִירָיו



וְנַפְשׁוֹ גָדְרָה פִרְצֵי שְׁחָקִים / וְתִפְרֹץ מִמְּקוֹם עַיִשׁ
גְּדֵרָיו



דְּבָרָיו מִפְלְאוֹת דֵּעִים וְשָׁת עִם / דְּבִירֵי מַעֲלוֹת שֶׁמֶשׁ
דְּבִירָיו



וְקָט לָבוּס כְּסִיל תַּחַת פְּעָמָיו / וְתַחַת פַּעֲמֵי עֶשְׂרִים
סְפָרַָיו



בְּלֵב יִסֹּב עֲלֵי גַלְגַּל אֲדָמָה / וְיֵחָלֵק עֲלֵי שֵׁשֶׁת עֲבָרָיו



וְיִכּוֹן כָּל אֲשֶׁר יִרְחַק זְמָמוֹ ־ / לְהַשִּׂיג, מִבְּלִי הָפֵר,
אֱסָרָיו



אֲשֶׁר לוּ אִם יְבַקֵּשׁ קָט הֲנָחָה / גְּזַרְתִּיהוּ וְשָׂרַפְתִּי
גְזָרָיו



אֱמֹר אִם לֹא בְעֵת אוֹרֶה לְבָבִי / בְּאַף ־ יִבְעַר עֲשַׁן אֵשׁ
מִנְּחִירָיו



וְאֹמַר: מִדְּרֹשׁ שֵׂכֶל הֲתָסוּר / בְּאֵין כֹּחַ בְּךָ לִמְצֹא
חֲקָרָיו



דְּרֹךְ עַל בָּמָתֵי יַמֵּי תְבוּנָה / וּבַקַּע מִשְּׂדֵה שֵׂכֶל
הֲרָרָיו



וְדַע כִּי בַעֲלוֹתְךָ הַר תְבוּנָה / אֲבַקֵּשׁ מִמְּךָ לִמְנוֹת
עֲפָרָיו



וְעוֹד לֹא אֶשְׁקְטָה מֵהַר מְזִמּוֹת / עֲדֵי כִי אֶעֱלֶה לִטְרֹף
כְּפִירָיו



וְלִי עַיִן מְשׁוֹטֶטֶת בְּתֵבֵל / אֲשֶׁר מִלֵּב אֱנוֹשׁ תִּרְאֶה
סְתָרָיו



וְלִבִּי פִתְּחוּ תָּמִיד שְׁעָרָיו / וְשִׁירִי לֻטְּשׁוּ שֶׁבַע
תְּעָרָיו



וְיָדִי בַזְּמַן תָּמִיד קְשׁוּרָה / וְאֶקְשֹׁר יַד לְבָנָיו עַל
שְׁחֹרָיו



וְשַׂמְתִּי מִלְּבוּשׁ לַיִל לְבוּשִׁי / וְתָפַרְתִּי כְסוּת כִּכְסוּת
שְׁחָרָיו



וְאָסִיר מֵעֲלֵי עֵינֵי לְבָבִי, / בְּהִתָּלוֹת אֲפֵר לַיִל, אֲפֵרָיו



בְּמוֹת שֶׁמֶשׁ וְקוּם לַיִל כְּתַנִּין / לְהָשִׂים עַל פְּנֵי אֶרֶץ
חֳמָרָיו



וְהַסַּהַר לְבוּשׁ חֹשֶׁן יְרַקְרַק / שְׁאָלוֹ מֵהֲדַר שֶׁמֶשׁ וְאוֹרָיו



וְשָׁת עָנָן עֲלֵי פָנָיו לְמַסְוֶה / כְּאִלּוּ יִירְאוּ אוֹרִי
מְאוֹרָיו



וּפֶתַח אָהֳלוֹ נִצָּב כְּנָגִיד / לְמִלְחָמָה יְאַסֵּף אֶת גְּבִירָיו



וְעָמַד מַעֲמַד מֶלֶךְ, סְבִיבוֹ / כְּחֹם אֵשׁ מַעֲרֶכֶת מִנְּזָרָיו



וְשָׁלַח כּוֹכְבָיו סָבִיב, כְּרוֹעֶה / יְשַׁלַּח עַל פְּנֵי שָׂדֶה
עֲדָרָיו



וְכִרְאוֹת מַרְכְּבוֹת יוֹם הֵן יְנוּסוּן / וְאִישׁ יִפָּרְדוּ מֵעַל
חֲבֵרָיו



וְלַיִל קָדְקָדוֹ יִמְשַׁח בְּמוֹ טַל / לְפָנָיו, כִּי יְגַלֵּחַ
שְׂעָרָיו



וּבֶן שַׁחַר יְגַלְגֵּל אֶת כְּסוּתוֹ / כְּאוֹרֵג הַיְּרִיעָה עַל
מְנוֹרָיו



וְלוּלֵי כִּי אֲנִי שֵׂכֶל מְשַׁחֵר / וְדַעַת ־ לֹא יְדָעוּנִי שְׁחָרָיו



וְדִמִּיתִי קְנוֹת וּסְפוֹג חֲבֵרִים / הֲכִי לָאִישׁ תְּעָלָה בַחֲבֵרָיו



וְשַׁבְתִּי מִזְּרֹעַ הַיְדִידוּת / כְּקוֹצֵר שָׁדְפָה רוּחַ קְצִירָיו



וּמֵרֵעַ קְנוֹת מֵעַם זְמַנִּי ־ / בְּקֶרֶן הַצְּבִי יִתֵּן מְחִירָיו



״בְּחַר מֵהַקְּרָב שָׁלוֹם וְהֶרֶף / וְאִם עַזּוּ וְגַם רַבּוּ חֲגוּרָיו



וְדַע כִּי לֹא אֱנוֹשׁ נִצָּל כְּמוֹ אִישׁ / צְרָפוֹ הַזְּמַן שֶׁבַע
בְּכוּרָיו



יְדוּעַ רַע וְאָסוּר כֻּתְּנוֹת דַּל / עֲדֵי כִי נִמְתְּקוּ
עִמּוֹ מְרוֹרָיו״



וְאֵיךְ יַמְרֶה אֱנוֹשׁ דַּעַת לְבָבוֹ / וְיִשַּׁח לַחֲצִיר יַעַר
וְסִירָיו



וְאֵיךְ אָבוֹא בְלַהֲקַת הַפְּתָאִים / וְהַשֵּׂכֶל הֱבִיאַנִי חֲדָרָיו



וְאֵיךְ אֶלְאֶה מְצֹא פֵתַח דְּבָרִים / וְעִמִּי נִפְתְּחוּ דַלְתֵי
שְׁעָרָיו



וּבָרָא אֵל בְּלִי לֵב עַם זְמַנִּי / וְיָצַָר מִבְּלִי קֶרֶן שְׁוָרָיו



וְיָרוּם כָּל אֱוִיל בּוֹ וַחֲסַר לֵב / וְכָל יוֹנֵק חֲלֵב אֵם
מִבְּחִירָיו



וְכָל נָטָה לְבָבוֹ אַחֲרֵי הוֹן / וְעָבַד בִּלְעֲדֵי שַׁדַּי סְגוֹרָיו



וְלָאֵל יוֹם יְהִי כָל סָר וְחָנֵף / כְּמוֹ קַשׁ מִפְּנֵי רוּחַ סְעָרָיו



וְיָעֹזּוּ וְיַעְלֹזוּ יְשָׁרָיו / וְיִבֹּלוּ וְיִמָּלוּ יְהִירָיו.
