


	
	אֲזַי בְּהַר הַמֹּר
	








אֲזַי בְּהַר הַמֹּר דָּץ יוֹנַת אֵלֶם

בְּקַחְתּוֹ אֵיתָן לְעוֹלָה וָשֶׁלֶם

גֵּאֶה בְחָנוֹ קַח זֶה הָעֶלֶם / תְּרוּמָה לַיְיָ קֹדֶשׁ





דָּת הַטֶּנֶא לְפָנַי לְהַנִּיחֵהוּ

הִנֵּה בִנְךָ לְאֶשְׁכָּר סְפָחֵהוּ

וַאֲנִי בְחַרְתִּיו קוּם מְשָׁחֵהוּ / שֶׁמֶן מִשְׁחַת קֹדֶשׁ





זַךְ בְּשָׁמְעוֹ לְבָבוֹ לֹא דָאָב

חֵפֶץ צוּרוֹ חָשַׁק וְתָאָב

טָפְלוּ שְׁנֵיהֶם כַּבֵּן כָּאָב / הִשְׁתַּחֲווּ לַיְיָ בְּהַדְרַת קֹדֶשׁ





יָעַד עֵצִים וְלִשְׁפּוֹת דָּרַךְ

כָּמַהּ וְתָמַהּ יָחִיד וָרָךְ

לָהַג אַיֵּה הַשֶּׂה הַנֶּעֱרָךְ / לְהַקְדִּישׁוֹ לַיְיָ קֹדֶשׁ





מָרוֹם יִרְאֶה לּוֹ הַשֶּׂה וְיָשִׁיב

נֶפֶשׁ עֲקֶרֶת שְׂמֵחָה לְהוֹשִׁיב

סוֹקֵר צְפוּנוֹת שִיחֲךָ יַקְשִׁיב / יִשְׁלַח עֶזְרְךָ מִקֹּדֶשׁ





עֲמֹד נָא אָבִי וּקְשֹׁר הַטֶּבַח

פֶּן אֶנָּעֵר מִפְּנֵי הָאֶבַח

צָמְדֵנִי פֶּן אֲחַלֵּל הַזֶּבַח / לְהַבְדִּיל בֵּין הַקֹּדֶשׁ





קָפַץ לִבְכּוֹת כֹּפֶר אֶשְׁכֹּל

רְאוֹתוֹ אָבִיו נִשְׁכַּל נִשְׁכּוֹל

שֶׁלַח תָּפַשׂ לַעֲשׂוֹת אֶת כָּל / מְלֶאכֶת עֲבוֹדַת הַקֹּדֶשׁ





תַּרְשִׁישִׁים צָעֲקוּ בִשְׁמֵי מְרוֹמִי

שְׁחֹט יוֹנַת אֵלֶם מִי יֶחֱזֶה מִי

לָז לְלָז הָגוּ אַל תִּתְּנוּ דֳמִי / שְׂאוּ יְדֵיכֶם קֹדֶשׁ





מָרוֹם הִבִּיט עוֹקֵד וְנֶעֱקָד

הַשֶּׂה עֲרוּכָה וְהָאֵשׁ תּוּקָד

בֹּחַן טְהָר לֵב זָכַר וּפָקַד / וְנִשְׁמַע קוֹלוֹ בְּבוֹאוֹ אֶל הַקֹּדֶשׁ





רָחַשׁ מֵלִיץ בְּקַחְתּוֹ מִשְׁלַחַת

יְחִידְךָ פְּדָעִֵהוּ מֵרֶדֶת שַׁחַת

הִנֵּה כְתוּבָה לְפָנָי מְפֻתַּחַת / פִּתּוּחֵי חוֹתָם קֹדֶשׁ





וַאֲנִי יָדַעְתִּי תֹּם דַּרְכְּךָ בְּשֶׁפֶר

דּוּץ וְרוּץ וְקַח אַיִל תַּחַת עֹפֶר

הֲכִינוֹתִי כֹפֶר לְאֶשְׁכֹּל הַכֹּפֶר / וּתְמוּרָתוֹ יִהְיֶה קֹדֶשׁ





גָּמוּל נִמְלַט מֵאֶבַח נִשְׁחָז

בְּפָשְׁטוֹ צַוָּארוֹ דָּמוֹ לֹא נִפְחָז

יְמִינוֹ אָחַז בְּיוֹבֵל נֶאֱחָז / בְּצֵאתוֹ בַיּוֹבֵל קֹדֶשׁ





רַחוּם תַּתָּה שָׁלוֹם אַחֲרִיתוֹ

וּזְכֹר הַיּוֹם לְגִזְעוֹ בְרִיתוֹ

לְיֶתֶר הַפְּלֵטָה וּשְׁאָר שְׁאֵרִיתוֹ / תִּהְיֶה פְלֵטָה וְהָיָה קֹדֶשׁ





שָׁלוֹם לָרֹב מִזְּבוּל הַנֶּעֱרָב

בַּשֵּׂר רְחוּמִים בְּיֶשַׁע יִקְרָב

יְקָרְךָ הַרְאֵם וְקוֹמֵם דְּבִיר הַנֶּחֱרָב

לְמַעַן עֲשֹׂה כַּיּוֹם הַזֶּה לְהַחֲיוֹת עַם רָב.
