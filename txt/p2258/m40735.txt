


	
	[דֶּרֶךְ־אֳנִיָּה בְלֶב־יָם שִׂים פְּעָמֶיךָ]
	



[נולד בספרד בסוף המאה הי"ד, חי בסָרָגוֹסָה,
ואחרי כן בפִּיזָה, פֵרָרָה, מוֹדֶנָה, פָּדוֹבָה, צֶ֗נְטוֹ אַנְקוֹנָה, רֵקָנָטִי, רֶג֗וֹ ומַנְטוֹבָה. מת אחרי שנת 1457]

דֶּרֶךְ־אֳנִיָּה בְלֶב־יָם שִׂים פְּעָמֶיךָ / לִבִּי
רְאֵה אֶת־שְׁלוֹם נָשִׂיא בְעַמֶּךָ



לֹא מִפְּנֵי מִשְׁבְּרֵי נָהָר וְגַלֵּי תְהוֹם / תָּהֹם בְּיוֹם תִּתְּךָ
דֶּרֶךְ בְּיַמֶּךָ



כִּי מוּל שְׁאוֹנִי לְיוֹם נָדַד אֲדֹנִי בְּלִי / נִשְׁמַע דְבָרוֹ
כְּאוֹב קוֹל מֵי תְהוֹמֶךָ



הִכּוֹן לְפָנָיו וְהִתְהַדֵּר עֲלֵי רִצֲפַת / דַּר רוֹמֲמוּתוֹ וְעָשִׂית
שְׂפָמֶךָ



יַעְרֹף מְטַר לִקְחֲךָ לִבִּי וְנִיב שִׂיחֲךָ / אֵלָיו תְּנָה הוֹֽדְךָ
רֵיחַ בְּשָׂמֶיךָ



נַחַל זְמִיר נַחֲלָה נִתַּן לְךָ עַל־פְּנֵי / חוּצוֹת שְׁלַח־נָא כְּאָז
נַחַל קְדוּמֶיךָ



עַל־כֵּס כְּבוֹדוֹ וְנֵס הוֹדוֹ לְאֵל תַּעֲתִיר / יָרוּם וְנִשָּׂא
וְגָבַהּ מִמְּרוֹמֶיךָ



וַחְסֵה בְצִלּוֹ וְסֵתֶר אָהֳלוֹ יִֽהְיֶה / מַחְסֶה לְךָ מִמְּשַׂנְאֶיךָ
וְקָמֶיךָ



מִמַּעֲמַקִּים לְהָקִים חֵיל עֲשׁוּקִים שְׁמוֹ / נוֹדָע כְּהַיּוֹם
יְקִימְךָ לוֹ וְרַחְמֶךָ



כָּעֵת בְּמָשְׁכוֹ בְּמוֹ־רֶשֶׁת אֲרֶשֶׁת שְׂפָ־ / תָיו אוֹֽתְךָ כִּי
בְךָ בָּחַר וְגָלְמֶךָ



זָכַר בְּשִׂכְלוֹ בְּרִית אָבוֹת וְאַחַר בְּלוֹת / חַדֵּשׁ יְמוֹתַי
יְצַו כִּימֵי עֲלוּמֶיךָ



וּזְבוּל לְךָ כּוֹנֲנוּ יָדָיו וְיָדָיו תְּבַ־ / צַּעֲנָה וְצִוָּה
בְרָכָה בַאֲסָמֶיךָ



אָבִי רְאֵה גַּם רְאֵה עֵינִי וְעֵינִי אֲשֶׁר / תִּדְמַע וּבִכְיִי
שְׁמַע יַעַל לְשָׁמֶיךָ



אֶנְקַת אֲסִיר חִשְׁקְךָ תִּשַׁע אֲשֶׁר מֵאֲזַי / נִלְכַּד כְּצִפּוֹר
בְּמוֹ מִשְׁטַח חֲרָמֶיךָ



תָּכִין לְבָבְךָ לְקוֹל עַנּוֹת עֱנוּתִי וּפִי / יִשַּׁק עֲפָרְךָ
וְהָאָרֶץ הֲדֹמֶךָ



אַל־נָא תְכַזֵּב בְּרַב וּמְעַט וְקָט עַל־דְּבַר / מֹהַר וּמַתָּן
וְיֵאָמֵן נְאֻמֶּךָ



אָגוּר לְעוֹלָם בְּאָהָלְךָ וְעַל־לִבְּךָ / אֶהְיֶה בְּחוֹתָם
יְדִידוּתְךָ חֲתוּמֶךָ



אֵבוּס בְּעָלִים וּמַן זוּלָֽתְךָ הֵן גְּבִיר / אָבוּס וְאֶתְאַו
בְּצָלֶיךָ וְשׁוּמֶיךָ



אַשְׁרֵי מְחַכֶּה וְיַגִּיעַ לְיוֹם יֶחֱזוּ / עֵינָיו נְוֵה קָדְשְׁךָ
וּקְצֵה תְחוּמֶךָ

הָגוּת לְבָבִי וְנִיבִי קַח וְלֹא אֶחֱפַר / יִשְׁפַּר
כְּאֶשְׁפָּר וּפַר זִבְחֵי שְׁלָמֶיךָ



יֻלַּד זְמִיר זֶה עֲלֵי בִרְכֵּי יְהוֹסֵף וְנִיב / צַח שִׂיחֲךָ לִי
מְיַלֶּדֶת וְטַעֲמֶךָ



נִכְתַּב וְנֶחְתַּם הֲדַר שִׁירִי בְּחֹדֶשׁ אֲדָר / יִקֹּד לְנֹכַח חֲדַר
גִּנְזֵי בְרֹמֶיךָ



שֻׁלַּח לְפָנֵי לְהוֹרוֹת בֵּית מְעוֹנִי וּפָ־ / נֶיךָ לְקַדֵּם
וְלִדְרֹשׁ אֶת־שְׁלוֹמֶךָ



כִּי אַעֲתִיר לוֹ בְהָקִיץ יוֹם וְלֵיל בַּחֲלוֹם / לֶךְ־נָא רְאֵה
אֶת־שְׁלוֹם נָשִׂיא בְעַמֶּךָ.
