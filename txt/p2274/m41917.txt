


	
	עֲלֵי עָשׂוֹר
	



לחתן. סימן: עבדאללה ח׳זק






עֲלֵי עָשׂוֹר / שִׁיר תְּהִלָּה / בְּשָׂרִי כָּמַהּ



אֲהַלְּלָה / דָּר עֲלִילָה / תּוֹךְ סְגֻלָּה / עֵדָה תְמִימָה



שָׂם יָהּ חָתָן / וְכַלָּה / יָסַד בְּחָכְמָה

אֵל רַב עָצְמָה / אֲרוֹמְמָה

בֵּין יְשָׁרִים / שְׁמוֹ בְּעֶזְרִי.





בָּחוּר מֵעָם / נַפְשׁוֹ תִנְעַם / טוֹב מָצָא אִשָּׁה



בַּת נְעִימָה / וַאֲיֻמָּה / וְאוֹר חַמָּה / שָׂמָה לְבוּשָׁהּ



וּמַחְלָצוֹת / מִמִּשְׁבְּצוֹת / זָהָב לְבוּשָׁה

וְחֵן אִישָׁהּ / כֶּתֶר רֹאשָׁהּ

רַב מְאוֹרִים / גִּבּוֹר כַּאֲרִי.





דּוֹדִי צְלַח / בְּכָל מִשְׁלַח / יָדְךָ מִטּוּב יָהּ



בְּרֹב שִׂמְחָה / וְהַצְלָחָה / לָךְ שְׁלֻחָה / בִּרְכַּת אֵלִיָּה



בְּהוֹנֶיךָ / וְאִשְׁתְּךָ / גֶּפֶן פּוֹרִיָּה

כָּעֵת חַיָּה / תִּתֵּן פִּרְיָהּ

צִיץ כִּתְמָרִים / עֲשׂוֹת כָּל פֶּרִי.





אֵל נֶאֱמָן / וְאָב רַחְמָן / לְפָנָיו אֲבָרֵךְ



חָתָן תָּמִים / אֹרֶךְ יָמִים / לְעוֹלָמִים / בָּרֵךְ יְבָרֵךְ



וּבְתוֹרָתוֹ / וּבְצִדְקָתוֹ / עֲנָוִים יַדְרֵךְ

אוֹיְבִים דּוֹרֵךְ / יָשִׂים מֹרֶךְ

בְּלֵב צָרִים / אֵל עֹז נֶאְדָּרִי.





לַבְנוּת עֶלְיוֹן / הָאַפִּרְיוֹן / מִקְדַּשׁ הַהַרְאֵל



בְּמוֹצָאָיו / וּבְמוֹבָאָיו / וְזִיו מַרְאָיו / מַרְאוֹת יְחֶזְקֵאל



מִמְּרוֹמוֹ / בְּשׁוּב עַמּוֹ / יִשְׂמַח יִשְׂרָאֵל

שָׁאוֹל שֹׁאֵל / יִשְׁלַח גּוֹאֵל

בָּנִים יָרִים / יִשְׁעִי וְאוֹרִי.





חָזָק חֲמֹל / חֶסֶד גְּמֹל / אַתָּה תְשַׁלֵּם



לְאִישׁ נֶעְצָר / עֶבֶד נִמְסָר / בְּיַד אִישׁ צַר / כְּרָחֵל אִלֵּם



שׁוּר לוֹחֲמִי / וֶאְמֹר בֶּן מִי / זֶה הוּא הָעֶלֶם

מֶלֶךְ שָׁלֵם / אַל תִּתְעַלֵּם

תֵּיטִיב שִׁירִים / מִשּׁוֹר פָּר וּמְרִיא.
