


	
	שירת דודים
	



לכבוד איש אמונים הנגיד וקצין, עשה צדקה בכל עת, ירא אלהים מרבים,

מוהר"ר

יוסף אליעזר לאַצאַרוס ני"ו ביום מלאת לו שבעים שנה,

מאת אנשי דבה"כ ליסר פה ברעסלויא יום ה' ח' אייר עוד ימים רבים יחי' בנעם לפ"ק.






חזן. יָמִים כַּבִּירִים דַּרְכֵּי אִישׁ תָּמִים רָאִינוּ,



הִטָה שִׁכְמוֹ לִסְבֹּל עָמַל בְּנֵי עַמֵּינוּ,

תֵּת לוֹ תּוֹדָה בְּכָל נַפְשֵׁנוּ צִפִּינוּ,

מִשֶּׁמֶשׁ צִדְקָתוֹ הוֹפִיעַ עָלֵינוּ,

חִכִּינוּ עֵת נבֹאָה לְפָנָיו בִּרְנָנָה,

עַד מוֹעֵד יִמְלְאוּ לוֹ שִׁבְעִים שָׁנָה.





קהל. עוֹד יָמִים רַבִּים יְחִי אִישׁ הַבֵּינָיִם,



כְּמֵאָז וּכְמִקֶּדֶם יְהִי לָנוּ לָעֵינָיִם.





חזן. בִּמְשֹׁל אִישׁ זֵד בְּיָדוֹ שֵׁבֶט הָרֶשַׁע,



יֵאָנַח הָעָם הַיּוֹשֵׁב תַּחַת מֶמְשַׁלְתּוֹ,

בִּזְדֹנוֹ בַּל יַבְחִין בֵין צֶדֶק וָפֶשַׁע,

בַּל יָסִיר מֵעָלָיו נִכְלֵי אִוַלְתּוֹ,

אִם יַלְבִּין כְשֶׁלֶג שְׂעָרוֹת רֹאשֵׁהוּ,

אֱנוֹשׁ בַל יִשְׂמַח בוֹ גֶבֶר בַל יְכַבְּדֵהוּ.





קהל. מֶמְשֶׁלֶת זָדוֹן מְחִתָּה חֲמַת תַּנִּינִים,



אַנְשֵׁי אֱמֶת שֹׁפְטֵי אֶרֶץ בְּעֵינָיו לִצְנִינִים.





חזן. לֹא כֵן אַתָּה יְדִיד לִבֵּנוּ אִישׁ תָּמִים,



לַעֲשׂוֹת צְדָקָה וּמִשְׁפָט אָהַבְתָּ,

יָצָאתָ לְיֵשָׁע עַמְךָ כָּל הַיָּמִים,

בְּכָבוֹד הָדָר וָעֹז זָקַנְתָּ וָשַׂבְתָּ,

הַשְׂעָרוֹת הַלְבָנוֹת לְךָ לָעֲטֶרֶת

מִפְעָלֶיךָ הַטוֹבִים לְךָ לְתִפְאֶרֶת.





קהל. מַה טוֹב וּמַה נָעִים בְּתוֹכֵנוּ אֹהֲבֵי צֶדֶק,



רוּחַ אֱלֹהִים בְּקִרְבָּם לְהַחֲזִיק כָּל בֶּדֶק.





חזן. הַבַּיִת הַנֶחְמָד הַזֶה יָדָיו יָסָדוּ



הָלַךְ אַחֲרֵי צָו יִרְאַת ה' עַל פָּנָיו,

אַנְשֵׁי אֱמֶת מִנִּי אָז פֹּה נוֹעָדוּ

צִדְקוֹת ה' הֵן מִשְׁעַנְתּוֹ הֵן בָּנָיו. –

בְּכָל לְבָבוֹ פִּזֵר נָתַן לָאֶבְיּוֹנִים,

בְּכָל נָפְשׁוֹ עָצְמָה הִרְבָּה לְאֵין אוֹנִים.





קהל. בָּרוּךְ הָאִישׁ פִּזֵר נָתַן לָאֶבְיוֹנִים,



לֹא יָשַׁב עִם מְתֵי שָׁוְא וְעִם שׁוֹנִים.





חזן. חָבֵר נֶאֱמָן! שָׂא נָא סָבִיב עֵינֶיךָ,



רְאֵה מָה נֶחְמָד פְּרִי צְדָקָה וָישֶׁר,

רְאֵה אִשְׁתְּךָ הַיְּקָרָה וְאֵלֶּה מִפְעָלֶיךָ,

הַטוֹבִים מִפָּז מִכָּל הוֹן וָעשֶׁר,

וְאֵלֶּה הָאֲנָשִׁים הַבָּאִים לְכַבְּדֶךָ

שְׂמֵחִים בִּישׁוּעָתֶךָ שְׂמֵחִים בְּהוֹדֶךָ.





קהל. בָּרוּךְ הָאִישׁ פִּזֵר נָתַן לָאֶבְיוֹנִים



צִדְקוֹת ה' עָשָׂה לֹא הִתְעָרֶב עִם שׁוֹנִים.





חזן. חֲזַק וֶאֱמָץ אָחִינוּ עֲמוֹד לִימִינֵנוּ



כַּאֲשֶׁר הִסְכַּנְתָּ וְכַחוֹל תַּרְבֶּה יָמִים

יַחַד נַעַזְרָה נָא אֶת בְּנֵי עַמֵינוּ

וְנִהְיֶה קֶשֶׁר אַמִיץ לְעוֹלָמִים.

עוֹד תְהִי מִשְׁעַן לָעֲנִיִּים וְאֶבְיּוֹנִים

וְנִשְׂמְחָה יַחַד בְּיּוֹם מְלֹאת לְךָ שְׁמוֹנִים.





קהל. יִחְיוּ לָעַד נְדִיבֵי לֵב יִחְיוּ אֲנָשִׁים נִכְבָּדִים,



הַנּוֹתְנִים מִלַּחֲמָם לַדָּל לָאֶבְיּוֹנִים בְּגָדִים.




חֻבר מאת דוד זאַמושטש.





נדפס אצל הירש זולצבאַך.
