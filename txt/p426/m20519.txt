וּבְנֵי יִשְֹרָאֵל בָּאוּ מִכַּרְמֶל וּמֵהַשָּרוֹן. מִשְֹנִיר
וּמֵחֶרְמוֹן, מִדָן וּמִבְּאֵר־שָבַע לַחְַנֵךְ אֶת הַבַּיִת אְַשֶר
בָּנָה שְׁלֹמֹה מֶלֶךְ יִשְֹרָאֵל לְשֵם ה'.

וַיָבִיאוּ אִתָּם זָהָב וָכֶסֶף וּבִכּוּרֵי אַדְמָתָם וּמֵיטַב כַּרְמָם
וְגַפְנָם וּמִבְחַר עֶדְרָם, הֵבִיא אִיש אִיש מֵהֶם יְרוּשָלַיְמָה, כִּי
אָמְרוּ: אַל נָא נִרְאֶה רֵיקָם אֶת פְּנֵי ה'.

וַיַרְא ה' כִּ נֶאְֶסַף יִשְֹרָאֵל עַמוֹ מֵאַרְבַּע כַּנְפוֹת הָאָרֶץ
לְשֵם קָדְשׁוֹ, וַיִיטַב הַדָבָר לְפָנָיו, וַיִקְרָא לְכָל מַלְאָכָיו
וּכְרוּבָיו וּשְֹרָפָיו וַיֹאמֶר: נֵרְדָה נָא יְרוּשָׁלַיְמָה
וְשָׁכַנְתִּי בְבֵית מִקְדָשִׁי, אְַשֶׁר עָשֹוּ לִי בָּנַי, וַיַעַשֹ ה'
כֵן כַּאְַשֶׁר אָמַר וַיָשֶֹם שְׁכִינָתוֹ עַל כַּפֹּרֶת הַזָהָב אְַשֶׁר
עַל אְַרוֹן הַקֹדֶש בֵּין הַכְּרוּבִים.

אָז יִפָּתְחוּ שַעְַרֵי הַהֵיכָל וַיֵרָאוּ רִבְבוֹת אַלְפֵי יִשְֹרָאֵל
כַּחוֹל אְַשֶר עַל שְֹפַת הַיָם וּכְכוֹכְבֵי הַשָמַיִם לָרוֹב,
וַיִשְֹמַח ה' וַתִּמָלֵא כָל הָאָרֶץ בִּרְכַּת ה'.

וְעֵינֵי כָל הָעָם לְמִקָטוֹן וְעַד גָדוֹל נְשֹוּאוֹת הָיוּ אֶל
הַהֵיכָל, אְַשֶר יְרִיעָה נֶחְמָדָה מְאֹד כִּסַּתּוֹ מֵהָעָם, כִּי
אָמְרוּ: אַךְ תּוּסַר הַיְרִיעָה וְרָאִינוּ אֶת אְֶלֹהֵי יִשְֹרָאֵל,
אְַשֶר לִשְׁמוֹ בָאנוּ מֵאַרְבַּע כַּנְפוֹת הָאָרֶץ.

וַיְהִי עוֹדָם מְצַפִּים לִרְאוֹת אֶת יְיָ וַתּוּרַם הַיְרִיעָה,
וַיִתְנוֹדֵד הָעָם וַיַבִּיטוּ וַיִשְׁתוֹמָמוּ.
וַתְּשׁוֹטַטְנָה1עֵינֵיהֶם אָנֶה וָאָנֶה וַתְּבַקֵשְׁנָה אֶת ה'
בְּכָל הַהֵיכָל וְלֹא מְצָאוּהוּ.


הַלִרְאוֹת אֶת גְדוּדֵי הַלְוִיִים, הַלְבוּשִים כָּתְנוֹת בַּד
עָלִינוּ ירוּשָלַימָה וַנַעְַבוֹר הָרִים וּגְבָעוֹת
עְַמָקִים וְגֵאָיוֹת? קָרְאוּ יְהוּדֵי חֶרְמוֹן.
הְַלִשְׁמוֹעַ בְּקּוֹל נֵבֶל וְכִנוֹר וּמְצִלְתַּיִם, בְּקוֹל
תְִּפִים וַחְַצוֹצְרוֹת עְזַבְנוּ אֶת אָהְָלֵינוּ וְאֶת בָּתֵּינוּ
וַנִפְצַע רַגְלֵינוּ בְקוֹצִים בְקוֹצִים וּבְאַבָנִים עַד בּוֹאֵנוּ
הְַלוֹם? נִרְגְנוּ2יְהוּדֵי הַלְבָנוֹן וְהַגָּלִיל.
הְַלְהִתְעַנֵּג עַל בִּגְדֵי הַשְֹרָד אְַשֶר לָבַש הַכֹּהֵן
הַגָּדוֹל, עַל מַּחְתוֹת הַזָּהָב, עַל שְׁלֹמֹה הַמֶלֶךְ


וְרַבֵּי הַמְלוּכָה הָעוֹטְרִים 3אוֹתוֹ, נֶאֶסַפְנוּ בָּזֶה? הֵלִינוּ
יְהוּדֵי הַכַּרְמֶל וְעֵבֶר הַיַּרְדֵן, וַיוֹסֶף הָעָם לְהַלִין4
וּלְבַקֵש אֶת ה', וַיַבִּיטוּ אֶל כָּל זָוִית וְאֶל כָּל פִּנָה וְלֹא
רָאוּ בִּלְתִּי אִם אֶת מִזְבַּח־הַנְחֹשֶת. וְאֶת אֶבְיָתָר הַכֹּהֵן
הַגָּדוֹל הַעוֹמֵד אֶצְלוֹ, אֶת שֻׁלְחַן הַזָּהָב וְאֶת לֶחֶם הַפָּנִים,
אֶת מְנוֹרַת הַזָּהָב וְאֶת נֵרוֹתֶיהָ, אֶת הָאָרוֹן וּכְרוּבָיו,
הַסוֹכְכִים5 אוֹתוֹ בְּכַנְפֵיהֶם מִן הַתִּקְרָה וְעַד הָרִצְפָּה.


רַמֵה רִמָנוּ הַמֶלֶךְ! נִשְׁמַע פִּתְאֹם קוֹל אַחַד הָעָם.
מִרְמָה! תַּרְמִית! נִשְׁמַע קוֹל הָעָם בְּהֵרָגְנוֹ.
יִסָקֵל הַמֶלֶךְ בָּאְַבָנִים! הִרְעִימוּ אַלְפֵי קוֹלוֹת.
וַיֶחֱרַד שְׁלֹמֹה מֶלֶךְ יִשְֹרָאֵל חֲרָדָה גְדוֹלָה, וַיִבָּהְַלוּ
שָֹרָיו וַיֶחְוְרוּ6 פְּנֵי הַכֹּהְַנִים וְהַלְוִים מְשָׁרְתֵי ה'.


וַיִרְמֹז הַמֶלֶךְ לִבְנָיָהוּ בֶן יְהוֹיָדָע וַיָבֹא וַיִשְׁתַּחוּ
לַמֶלֶךְ: וַיֹּאמֶר הַמֶלֶךְ: בַּמָה אַשְׁקִיט לָעָם? וַיִשְׁתַּחוּ
בְנָיָהוּ שֵׁנִית וַיֹאמֶר: לֹא יָדַעְתִּי, אְַדוֹנִי הַמֶלֶךְ, וַיאֹמֶר
הַמֶלֶךְ לְאֶבְיָתָר: עֻצָה נָא עֵצָה לְהַרְגִיעַ אֶת עְַבָדַי
הַסוֹאְַנִים כְּגַלֵי יָם! וַיַעַן אֶבְיָתָר וַיֹאמֶר בִּרְעָדָה: חֵי
נַפְשִׁי אִם יָדָעְתִּי!

וַיִקְצֹף הַמֶלֶךְ קֶצֶף גָּדוֹל וַיִקְרָא לְשָֹרָיו הָעֹטְרִים אוֹתוֹ
לֵאמֹר: מִי הָאִישׁ הַיוֹדֵעַ לְהַשְׁקִיט לָעָם וְלֹא יַהַס 7אוֹתָם
וּשְׁפַטְתִּיהוּ מִשְׁפַּט מָוֶת! וַיִשְׁתַּחוּ בְנָיָהוּ בֶן יְהוֹיָדָע
שֵׁנִית לַמֶלֶךְ וַיֹּאמֶר: לוּ לַעְַצָתִי יִשְׁמַע הַמֶלֶךְ, כִּי
עַתָּה צִוָה אֶת אְַחִיָה הַנָּבִיא לֵאמֹר: לֵךְ נַבֵּא אֶת הָעָם.
וַיִשְׁתַּחוּ אְַחִיָה לַמֶלֶךְ וַיּאמֶר; אֵיכָה אְַנַבֵּא לָעָם
וְלִבּוֹ זוֹעֵף8וְנַפְשוֹ מָרָה? הְַיִשְׁמַע לִי? הְַיִקַח מוּסָר.


רִגְמוּהוּ בָאְַבָנִים! סִקְלוּהוּ! נִשְמַע קוֹל הָעָם כְּקוֹל
אַלְפֵי רְעָמִים.
הַקְשַׁרְתֶּם כֻּלְכֶם עָלַי? קָרָא שְׁלֹמֹה בְּכַעַס אֶל שָֹרָיו
הָעֹמְדִים לְפָנָיו, הַחְַרִישׁוּ אֶת הַעַם!


וַיֹאמֶר בְּנָיָהוּ בֶן יְהוֹיָדָע: כִּי יָעַצְתִּי – יִגְשׁוּ הַכֹּהְַנִים לַעְַבוֹדָתָם, וְשָֹם הָעָם אֶת עֵינוֹ בַּכֹּהְַנִים וְחָדַל מֵאְַדוֹנִי הַמֶלֶךְ! וַיַרְא אֶבְיָתָר כִּי טוֹבָה הָעֵצָה הַזֹאת לִפְנֵי הַמֶלֶךְ וַיְקַח מַחְתַּת זָהָב וַיְשֶֹם בָּהּ קְטֹרֶת וַיִגַּשׁ בִּרְעָדָה אֶל הַמִזְבֵּחַ, וַיִרְאוּ הַכֹּהְַנִים כִּי כֵן, וַיִקְחוּ אִישׁ מַחְתָּתוֹ, וַיִגַּשׁ אָסָף, הֵימָן וִידוּתוּן אֶל כְּלֵי שִׁירָם וַיִתְעוֹרְרוּ מַחְַנוֹת הַלְוִיִם וַיְמַהְַרוּ לָקַחַת אֶת נִבְלֵיהֶם וְכִנוֹרוֹתֵיהֶם וּמִנֵיהֶם וְעוּגְבֵיֵהֶם…


מוֹת יָמוּת! רָגֹם יִרְגְמוּ 9אוֹתוֹ כָּל יִשְֹרָאֵל: נִשְמְעוּ
אַלְפֵי רִבְבוֹת קוֹלוֹת, אַךְ פַּעַם רִמָנוּ וְלֹא יוֹסִיף.


וְקוֹל רַעַשׁ שֶׁל אַלְפֵי רַגְלַיִם הָרוֹקְעוֹת 10וְרִבְבוֹת
הָאֶגְרוֹפִים הַמְתוֹפְפִים מִלֵא אֶת כָּל הַהֵיכָל, וַתִּפֹּל אֵימַת
מָוֶת עַל הַכֹּהְַנִים וְלֹא יָכְלוּ יְדֵיהֶם לְהַחְַזִיק
בְּמַחְתּוֹתֵיהֶם, וַיִבָּהְַלוּ הַלְוִיֹם וַיָמָת לִבָּם וַתִּהְיֶינָה
אֵצְבְּעוֹתֵיהֶם לָאָבֶן.


וַיְהִי כָל הַמָקוֹם לְחֶרְדַת אְֶלֹהִים.


וּלְהֵימָן הַלֵוִי רֹאשׁ הַמְשׁוֹרְרִים בְּמִקְדַש ה' בֵּן, וַחְַנַנְיָה
שְמוֹ; וְהוּא נַעַר כְּבֶן אַרְבַּע עֶשְֹרָה שָׁנָה. וְיָפֶה הָיָה
חְַנַנְיָה וְטוֹב תֹּאַר וְתָם דֶרֶךְ וּמוֹצֵא חֵן בְּעֵינֵי כָל רוֹאָיו
וְאָהֹב אָהַב חְַנַנְיָה אֶת הַיַעַר וְלַחַשׁ11 עֵצָיו, אֶת הַנָּהָר
וְשִֹיחַת גַלָיו, אֶת זִמְרַת הַצִּפֳּרִים, אֶת הַכָּרִים12הַמְכֻסִים
דֶשֶׁא עֵשֶֹב וְאֶת הַפְּרָחִים הַנֶחְמָדִים הַמְשֻבָּצִים בְּתוֹכָם.


מַה לְךָ, בְּנִי, כִּי תִתְהַלֵךְ יוֹם, יוֹם בַּגַּנִים וּבַשָֹדוֹת?
הָיְתָה אִמוֹ שׁוֹאֶלֶת אֶת חְַנַנְיָה.
לִרְאוֹת אֶת ה' אְַנִי הוֹלֵךְ אִמִי הַטוֹבָה, הָיָה חְַנַנְיָה
עוֹנֶה, וְקוֹלוֹ צָלַל13כְּפַעְַמוֹנֵי כֶּסֶף.
הְַאֵין ה' בִּירוּשָׁלַיִם עִיר קָדְשֵנוּ; כִּי הָלֹךְ תֵּלֵךְ
לְבַקְשֵׁהוּ בַּכַּרְמֶל וְבַשָׁרוֹן?
מְלֹא כָל הָאָרֶץ כְּבוֹדוֹ! הָיָה חְַנַנְיָה עוֹנֶה אֶת אִמוֹ;
אוּלָם מִחוּץ לָעִיר, בַּמָקוֹם אְַשֶר יָשֹךְ14


הְַמוֹן הָעִיר וְשְׁאוֹנָהּ, רָאֹה אֶרְאֶה אֶת ה', גַם שָׁמֹעַ אֶשְׁמַע
קוֹלוֹ מְדַבֵּר אֵלָי.


חִידוֹת לִי דְבָרֶיךָ בֵן יַקִיר! הָיְתָה אִמוֹ אוֹמֶרֶת
לַחְַנַנְיָה וַחְַדֵלָה לְדַבֵּר אֵלָיו בַּדָּבָר הַזֶּה.
שִׁמְעִי נָא, אִמִי הַטוֹבָה, הָיָה חְַנַנְיָה אוֹמֵר לְאִמוֹ
הַמַבִּיטָה אֵלָיו בְּשִֹמְחָה וּבִדְאָגָה. שִׁמְעִי
נָא אֶת אְַשֶׁר אְַסַפֵּר לָךְ: אֶתְמוֹל יָרַדְתִּי מֵהַכַּרְמֶל לְשׁוּב
הַבַּיְתָה, הַשֶׁמֶשׁ יָרְדָה הַיָמָה וַתִּשְׁלַח קַרְנֵי 15זְהָבָה
עַל הָעֵצִים וְהַדְשָאִים, עַל הַצֹאן וְהַבָּקָר אְַשֶר רָעוּ בָעֵמֶק,
נֶהְדָר הָיָה הַמַרְאֶה מְאֹד וָאֵלֵךְ וָאְַשׁוֹרֵר:



שַׁבְּחִי, יְרוּשָׁלַיִם, אֶת ה'!



הַלְלִי אְֶלֹהַיִךְ, צִיוֹן!



כִּי חִזַק בְּרִיחֵי שָׁעָרַיִךְ



בֵּרַךְ בָּנַיִךְ בְּקִרְבֵּךְ



הַשָֹם גְבוּלֵךְ שָׁלוֹם…


וְהִנֵה קוֹל אְַנָשִים מְרִיבִים אִישׁ אֶת אָחִיו בְּחָזְקָה, עוֹלֶה
בְאָזְנָי, וָאֶחְדַל לְשׁוֹרֵר וָאְַמַהֵר אֶל מְקוֹם הַמְרִיבָה, וָאֵרֶא
אִישׁ וְאִשָׁה מִתְקוֹטְטִים וּפְנֵיהֶם אְַדוּמִים כַּדָם, וְעֵינֵיהֶם
לַפִּידֵי אֵש וְאֶגְרוֹפֵיהֶם מוּנָכִים לְמַעְלָה, וָאֶקְרָא בְּחֵן
וּבִנְעִימוּת:



שָׁלוֹם לְךָ אָחִי! וְשָׁלוֹם לָךְ אְַחוֹתִי!



וַיְהִי פֶלֶא, כִּי שָׁבְתוּ הָאְַנָשִים כְּרֶגַע מֵרִיב וּפְנֵיהֶם
נָהַרוּ,16 וַיַבִּיטוּ אַחְַרֵי עַד הַרְחִיקִי מֵהֶם.

וָאוֹסִיף לָלֶכֶת עַד בּוֹאִי אֶל שַׁעְַרֵי הָעִיר וְהִנֵה יְלָדִים
נֶאְֶבָקִים אִישׁ אֶת רֵעֵהוּ וָאֶתֵּן קוֹלִי בַּשִׁיר, – וַיֶחְדְלוּ
לְהֵאָבֵק, וָאוֹסִיף לָשִׁיר – וַיֵצְאוּ בַמְחוֹלוֹת.


טוֹב אַתָּה בְּנִי, וְטוֹבִים מַעְַשֶֹיךָ! קָרְאָה אֵם חְַנַנְיָה
וְבְעֵינֶיהָ הִזְהִירוּ דִמְעוֹת שִֹמְחָה. וַיְהִי
בְרֶגַע הַזֶה וְקוֹל רַעַשׁ וְשָׁאוֹן נִשְׁמַע בַּחוּץ, וַיֹּאמֶר
חְַנַנְיָה, אָרוּצָה הַחוּצָה וְהָיָה כִּי אֶרְאֶה חְַדָשָׁה וְשַׁבְתִּי
וְהִגַדְתִּי לָךְ. וַיָרָץ חְַנַנְיָה לְקוֹל הָרַעַשׁ וַיָבֹא לְבֵית ה'
וַיַרְא אֶת הָעָם הַמִתְגָעֵשׁ, וַיֵדַע אֶת אְַשֶר נִהְיָה בְבֵית
הַמִקְדָשׁ, וַיִקַח כִּנוֹר וַיַעְַמוֹד בֵּין בֵּין הָעָם וּבֵין
הַכֹּהְַנִים וַיַשְׁמִיעַ קוֹל שִׁיר, וַיִשְׁקוֹט הַשָׁאוֹן וַיִגַּע
חְַנַנְיָה בְמֵיתְרֵי הַכִּנוֹר וַיִפְתַּח פִיו בְּזִמְרָה: – וַיֵאוֹרוּ
פְנֵי הָעָם וַיְזַמֵר חְַנַנְיָה:



הַלְלוּ אֵל בְּקָדְשׁוֹ



הַלְלוּ אֵל בִּרְקִיעַ עֻזוֹ



וַיִשְתַּחוּ הָעָם מוּל אְַרוֹן הַקֹדֶשׁ


וַיוֹסֶף חְַנַנְיָה לְזַמֵר וּלְנַגֵן:


הַלְלוּהוּ בְּתוף וְמָחוֹל.


וַיֵצֵא כָל הָעָם בִּמְחוֹלוֹת.
וַיִרְאוּ הַכֹּהְַנִים וְהַלְוִיִם כִּי כֵן, וַיִגְשׁוּ לַעְַבוֹדָתָם
וּלְשִׁירָם וַיִמָלֵא בֵית ה' אוֹרָה וְשִֹמְחָה וְזִמְרָה, וַיֵרָא
כְבוֹד ה' בֶּעָנָן, וַיַרְא כָּל הָעָם וַיִפְּלוּ עַל פְנֵיהֶם אָרְצָה
וַיִקְרְאוּ:

ה' הוּא הָאְֶלֹהִים!
