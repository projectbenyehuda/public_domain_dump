


	
	אֲזַמֶּרְךָ אֱלֹהֵי כָל יְצוּרִים
	




סימן: אני ישראל בר משה נגארה חזק






אֲזַמֶּרְךָ אֱלֹהֵי כָל יְצוּרִים / בְּזִמְרַת חֵן וְשִׁירִים נֶהֱדָרִים

נְגִינָתִי כְּקוֹל נֵבֶל אֲעוֹרֵר / וְאָשִׁיר אַחֲרֵי נוֹגְנִים וְשָׁרִים

יְקַר הוֹדָךְ לְכָל יָבֹא אֲסַפֵּר / וְאַגִּידָה בְּסוֹד שָׁרִים יְשָׁרִים

יְצוּרֶיךָ כְּבוֹד שִׁמְךָ יְעִידוּן / וְיוֹדוּךָ זְקֵנִים עִם נְעָרִים

שְׁחָקִים בֶּאֱמוּנָתְךָ יְסַדְתָּם / וְכוֹכְבֵי אוֹר עֲלֵי אֶרֶץ מְאִירִים

רְקִיעֵי רוֹם וְכָל-מַלְכֵי צְבָאוֹת / בְּנֵי אֵלִים וְאֶרְאֶלִּים בְּהִירִים

אֲשֶׁר חָשִׁים וְרָצִים הֵם בְּמֶרֶץ / לְמִצְוָתְךָ, וְלֹא אָחוֹר נְזוֹרִים

לְךָ יוֹדוּ בְּנִיב שָׂפָה בְרוּרָה / בְּרָן יַחַד כְּאִישׁ אֶחָד חֲבֵרִים

בְּכָל זֹאת עֹז הֲדָרְךָ לֹא יְכִילוּן / וְאַף גֶּבֶר יְמוֹתָיו כַּשְּׂכִירִים

רְבָבוֹת מַאֲלִיפוֹת כָּל לְשׁוֹנוֹת / וְאִלּוּ יִפְצְחוּ כַּיָּם זְמִירִים

מְרוֹם עָזָּךְ בְשִׁירִים לֹא יְסֻלֶּה / וְלֹא הוּא תּוֹךְ גְּבוּלִים אוֹ גְדֵרִים

שְׁבָחֶיךָ בְּלִי קֵץ סוֹף וְתִכְלָה / פְּלָאֶיךָ חֲדָשִׁים לַבְּקָרִים

הֲכִי תַפְלִיא חֲסָדֶיךָ לְעֶבֶד / אֲשֶׁר יִקְרָא לְךָ מִן הַמְּצָרִים

נְהֹם יִנְהֹם כְּנַהְמַת יָם לְצִירָיו / וְיִשְׁאַג כַּכְּפִירִים בַּיְעָרִים

אֲשֶׁר יִרְאֶה לְעַבְדּוֹ בוֹ יְשׁוֹרֵר / וְהוּא שַׁאְנָן וְשׁוֹקֵט עַל שְׁמָרִים

גְּדָל-חֵמָה וּמִלְחָמָה יְגָרֶה / וְלוֹ מִמְשָׁל בְּעַם מָשַׁל בְּשָׂרִים

אֲבִיר רוֹעִים רְעֵה עַמָּךְ וְקַבֵּץ / נְפוּצוֹתָיו בְּקַצְוֵי יָם פְּזוּרִים

רְצֵה צוּרִי בְּקוֹל שִׁיר וְשַׁלַּח / אֲסִיר תִּקְוָה יְקַוֶּה עֵת דְּרוֹרִים

הֲלָנֶצַח יְהִי עֶבֶד לְעוֹבְדָיו / הֲלָעַד זֶה יְהִי נִכְנָע לְזָרִים

חֲבֹשׁ שִׁבְרוֹ הֲלֹא שִׂבְרוֹ לְבַד בָּךְ / הֲכִי אֵין בִּלְתְּךָ מַרְפֵּא לְצִירִים

זְכָר-לוֹ אַהֲבַת וּזְכוּת אֲבוֹתָיו / וְאַל תָּפֵר בְּרִית הוֹרִים טְהוֹרִים

קְחָה מִפִּיו מְקוֹמָן שֶׁל זְבָחִים / תְּשׁוּרַת שִׁיר וּמִנְחַת מַאֲמָרִים

וְשׁוּב אַרְמוֹן לְמִשְׁפָּטוֹ כְּקֶדֶם / אֲשֶׁר יָשִׁיב נְשָׁמוֹת לַפְּגָרִים

וַאֲזַי שָׁם בְּקוֹל רִנָּה וְתוֹדָה / אֲזַמֶּרְךָ אֱלֹהֵי כָל יְצוּרִים.
