


	
	[אָדוֹן מַגִּיד מֵרֵאשִׁית אַחֲרִית (יוצר)]
	




יוֹצֵר לַחֲתֻנַּת כַּסְיָאה אֲחוֹת הַמְשׁוֹרֵר





אָדוֹן מַגִּיד מֵרֵאשִׁית אַחֲרִית שָׁנִים



מִקֶּדֶם אֲשֵׁר לֹא נַעֲשׂוּ מְזֻמָּנִים



אוֹמֵר עֲצָתִי תָקוּם לְדוֹרוֹת אַחֲרוֹנִים.

בְּמַחְשַׁבְתּוֹ אָז כְּעָלָה לִבְראוֹת חֲתָנִים



בְּטֶרֶם יְצָרָם הִקְדִּים כָּל תִּקּוּנִים



לְלַמֵּד כָּל טַכְסִיסֵי חֻפָּה לַבָּנִים.

גָּדַר וְקֵרָה אֵת גַּג תִּקְרָתָם



תָּכַף וְיָסַד אֵת בֵּית חֲתֻנָּתָם



וְאוֹר לְהָאִיר עֲלֵי הַסֵּבַת סְעוּדָתָם.

אֱמֶת מַה נּוֹרָאִים מַעֲשֶׂיךָ נוֹרָא



מִדַּת בָּשָׂר וָדָם צָר צוּרָה



תְּפִיסַת רֵיק בְּלֹא רְאוּת וַחֲקִירָה



יוֹצֵר צַר צוּרָה בְּתוֹךְ צוּרָה.

דִּבֵּק בַּשֵּׁנִי עֲלִיָּה נָאָה וּמְיֻפָּה



וְהֵכִין שׁוֹשְׁבִינִים לְצַלְצֵל וּלְתוֹפֲפָה



וְהִסִּיק אַתּוּנָא לִמְעַרְבְּבֵי סְעוּדָתָם וּלְשָׂרְפָה.

הִקְוָה בַּשְּׁלִישִׁי וּפִנָּה תֵּיאַטְרוֹן לִמְשַׂחֲקִים



וְהֵצִיץ אִילָנוֹת תְּמוּר פַּרְפְּרָיוֹת וּפִנּוּקִים



וְגִּנַּת בִּיתָן גִּנּוּנָם בּוֹ לְהָקִים.

וּבָרְבִיעִי הִדְלִיק שְׁנֵי נֵרוֹת כְּפַנָּסִיּוֹת



מִלְּהִכָּבוֹת מֵהַפָּחַת רוּחַ וּמְטַר מֵיוֹת



וְהַשְּׁאָר לְנוֹי כְּעִטּוּרֵי סֻכָּה לִהְיוֹת.

זִמֵּן בַּחֲמִישִׁי מִינֵי עוֹפוֹת וְדָגִים



וְהִתְקִין מֵהֶם לְמַאֲכַל יְשִׁישִים וּמְעֻנָּגִים



וְזִיז לִוְיָתָן לְהָבִיא לַחֲבוּרַת זוּגִים.

חִיֵּל בַּשִּׁשִּׁי מִינֵי חַיָּה וּבְהֵמָה



תַּבְשִׁילֵי שָׂבוֹעַ וְהוֹתֵר לַעֲשׂוֹת מֵהֵמָּה



וּבְהֵמוֹת גָּנַז לַצַּדִּיקִים לְיוֹם נֶחָמָה.

טוֹב שָׁר וְהִנֵּה הַכֹּל נִתְיַפָּה



וְכָל מִינֵי סְעוּדָה צְרוּרִין בְּמַפָּה



וַעֲדָיִן אֵין חָתָן וְכַלָּה בַּחֻפָּה.

יִפָּה זִיו חָתָן מִגַּלְגַּל חַמָּה



נִשָּׂא גָלְמוֹ מֵהֲדוֹם וְעַד רוֹמָה



וְאַחֲרֵי כֵן הֶעֱמִידוֹ לְמֵאָה אַמָּה.

כִּוֵּן שְׁתֵּים עֶשְׂרֵה חֻפּוֹת מָלוֹן



שֶׁל-אֹדֶם פִּטְדָה זָהָב תַּרְשִׁישׁ וְיַהֲלוֹם



וְהִכְרִיזוּ לוֹ מַלְאָכִים בֹּאֲךָ לְשָׁלוֹם.

לְעֵת הִכְנִיסוֹ לְכִלַּת חֲתָנִים הַמְיֻקָּרָה



הִרְדִּימוֹ וְלָקַח צַלְעוֹ וְקִשְׁטָהּ כְּנַעֲרָה



הִרְחִיצָהּ וְסָכָהּ וּפִרְכְּסָהּ וְקָלַע שַׂעֲרָהּ.

מְסָרָהּ לַאֲלָפִים וְלִרְבָבוֹת לַהֲבִיאָהּ בִּזְמִירוֹת



וְכָל שַׁמָּשִׁים עוֹמְדִים שׁוּרוֹת שׁוּרוֹת



חַמָּה וּלְבָנָה לְעֻמָּתָם מְרַקְּדוֹת כְּנַעֲרוֹת.

נָשְׂאָה חֵן וָחֶסֶד לִפְנֵי בַעְלָהּ



וְהוּא בִּכְבוֹדוֹ בֵּרְכָם בְּרָכָה מְכֻלָּלָה



וְהַכֹּל יַעֲנוּ יִשְׂמַח חָתָן וְכַלָּה.

סָח לְשׁוֹשְׁבִינִים לְסַדֵּר שֻׁלְחָנוֹת שֶׁל-מַרְגְּלִיּוֹת



וְסָבִיב סָבִיב הוֹשִׁיב סַפְסָלִים וְקַתֶּדְרָיוֹת



וְלִשְׂמֹחַ בַּבָּשָׂר וּלִמְזֹג יֵין עֲסִיסִיּוֹת.

עוֹדָם בְּחֻפָּתָם הִתְרָם הַתְרָיָה אֶחָת



וְלֹא קִיְמוּהָ וְכִמְעַט יָרְדוּ לַשָּׁחַת



לוּלֵי מִזְמוֹר שִׁיר לְיוֹם נָחַת.

פֻּקַּד זְמַנָּם וְגַע קֵץ עִתָּם



וּמֵאָז וְעַד עַתָּה נִמְסַר לְתוֹלְדוֹתָם



לִפְרוֹת וְלִרְבּוֹת וְסוֹף לַמָּוֶת אַחֲרִיתָם.

צְנוּעִים נָהֲגוּ אֵלֶּה לְדוֹרֵי דוֹרוֹתָם



לְשַׁדֵּךְ בְּעֵת אֵרוּסִין וּלְקַדֵּשׁ בְּטַבְּעוֹתָם



וּבְעֵת חֻפָּתָם לִשְׂמֹחַ בְּשִׂמְחַת חֲתֻנָּתָם.

קְרֵבִים עָדֶיךָ בְּשִׂמְחָתָם עֲמוּסֵי מֵעָיִם



לָגִיל בְּמוֹרָא וְלֹא בְּעַזּוּת אַפָּיִם



מְסַיְּמִים בִּרְכָתָם כְּהַיּוֹם הַזֶּה בִּירוּשָׁלָיִם.

רַחֲמִים מְבַקְשִׁים מִמְּךָ נוֹרָא וְנִקְדָּשׁ



לֶאֱסֹף נִדָּחִים וְלָשִׁיר שִׁיר חָדָשׁ



שָׂשִׂים וּשְׂמֵחִים בְּבִנְין בֵּית הַמִּקְדָּשׁ.

שֶׁבַח וּגְדֻלָּה לְשִׁמְךָ אֵלִי זֶה



אֲשֶׁר רִחַמְתָּ דּוֹר עָנִי וְנִבְזֶה



וְנאֹמַר שֶׁהֶחֱיָנוּ וְהִגִּיעָנוּ לַזְּמָן הַזֶּה.

תָּמִיד הֵיטִיבָה אַחֲרִית עֲלֵי יְדִידִים



לְדוֹרֵי דוֹרִים הֱיוֹת שִׁמְךָ מְיַחֲדִים



לְהַקְדִּישְׁךָ בְּרַעַד וּבְרֶתֶת כֻּלָּם עוֹמְדִים.
