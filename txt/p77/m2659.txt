


	
	מושך בחוט אהבה
	



מושך בחוט אהבה / וידאל בנבנשת

כתובת: עוד שלחתי לידידי דון וידאל בן לביא




מוֹשֵׁךְ בְּחוּט אַהֲבָה נַפְשִׁי יְחִידָתִי / הוֹאֵל מְשׁוֹךְ עִם יְחִידָתִי גְּוִיָּתִי



הַאִם לְדַת אַהֲבָה תַחְשֹׁב יְדִיד לַעֲשׁוֹק / רוּחִי לְהַשְׁאִיר כְּמֵת אוֹתִי בְאַדְמָתִי



קָרַךְ כְּחָתוּל אֲשֶׁר אוֹכֵל בְּאַהְבָה בְשַׂר / בָּנָיו וְתַשְׁחִית תְּמוּנָתִי לְאַהְבָתִי



תִּקַּח סְגוֹרִי וְנִשְׁאַרְתִּי כְדוֹמֵם בְּלִי / חוּשִׁים וְהֶרְגֵּשׁ וְיֵז נִצְחִי וְחַיָּתִי



יוֹדְעֵי יְדִידוֹת יְדוּעֵי מַחֲלָתִי הֲיֵשׁ / בָּכֶם דְּמוּתִי יְהִי רֵעַ לְצָרָתִי                   5



אוֹ אִם יְדַעְתֶּם הֲיִמָּצֵא בְגִלְעָד צֳרִי / יָשֹׂר לְהָסִיר חֳלִי חִילִי וּמַכָּתִי



וַיּאמְרוּ לִי אֲשֶׁר לֹא שָׁמְעוּ עַד הֲלֹם / הֲלוּם יְדִידוֹת אֲשֶׁר יִשָּׂא תְלוּנָתִי



גַּם שׁוֹטְטוּ בַעֲדִי יוֹמָם וְלַיִל קְצוֹת / תֵּבֵל וְלֹא מָצְאוּ רִפְאוּת לְמַחְלָתִי



לוּלֵי לְחַלּוֹת פְּנֵי דוֹדִי לְהָשִׁיב יְחִי- / דָתִי וְחָשְבוּ לְהַעְלוֹת אֶת אֲרוּכָתִי



הַאָמְנָם חָיְתָה נַפְשִׁי בְצִלּוֹ וְאִם / אֵלַי יְשִׁיבָהּ יְקָרֵב יוֹם אֲסִיפָתִי               10



הַאִם בְּרֶגַע תְּחִי רוּחִי בְהִפָּרְדָהּ / מֶנְהוּ וְחַיֵּי בְשָׂרִי הוּא וְנִשְׁמָתִי



כִּמְעַט בְּשׁוּבָהּ אֱלֵי גֵוָהּ כְּאֵבָהּ יְצִי -/ קֶהָ וְיִצָּמְתוּ נַפְשִי וְגוּפָתִי



טוֹב לִי אֲשֶׁר תִּחְיֶה נַפְשִׁי בְצֵל אָהֳלֵי / דוֹדִי וְיִסְבֹּל לְבַד גּוּפִי מְצוּקָתִי



וּתְחִי בְצִלּוֹ עֲדֵי יַשְׁקִיף וְיַרְא מִמְּעוֹן / חַסְדּוֹ וְיִשְׁמַע שְׁאוֹן קוֹלִי וְצַעְקָתִי



יַשְׁקִיף וְיֵרֵא בְעֵין חֶסֶד וְאַהְבָה וְיֵט / אֹזֶן חֲסָדָיו לְרַוְחָתִי לְשַׁוְעָתִי              15



הֵא לָךְ יְדִידִי זְמִיר נִבְרָא בְמִצְוַת שְׂפַת / חֵשֶׁק לְהַגִּיד לְךָ שִׂיחִי וְאַנְחָתִי



שֻׁלַּח לְךָ עַל שְׁמִי אַךְ לא יְסַדְתִּיו אֲנִי / כִי הַנְּדוֹד הֶאֱבִיד שִׁירִי נְגִינָתִי



וּלְךָ לְבָבִי וְהֵיךְ בִּלְתּוֹ אֲנִי אוּכְלָה / לַעְרוֹךְ רְנָנוֹת וְאִתּוֹ כָל מְזִמָּתִי



חֵשֶׁק בְּרָאוֹ כְּחָס עַל הַיְּדִידוּת אֲשֶׁר / בֵּינִי וּבֵינָךְ לְבַל יִמַּס בְּדִמְעָתִי



גַּם חָס עֲלֵי גֵו אֲשֶׁר סוֹבֵל כְּאֵב חִשְׁקְךָ / תָּמִיד וְהִפִּיל לְפָנֶיךָ תְּחִנָּתִי              20



הִגִּיד לְךָ רב תְּלָאוֹתַי כְדוֹבֵר בְּפִי / הֵא תַחֲנוּנַי וְהֵא עִנְיַן תְּפִלָּתִי



אַחְלַי שְׁמַע נָא בְקוֹל גֵּו דַּל וְנִכְאֶה וְחוֹן / עָלָיו וְהַסֵּר כְּלֵי שִׁבְיִי וְגוֹלָתִי



הַלֵּב וְהַגּוּף מְשׁוֹךְ יַחַד וְיִהְיוּ לְךָ / וּרְאֵה בְהִפָּרְדָם נָשְׁתָה גְבוּרָתִי



שִׁיתָם כְּאֶחָד סְבִיב מִשְׁכַּן יְקָרָךְ וְאַל / תַּגְדֵּל בְּלַבַּת נְדוֹד רִשְׁפֵּי מְדוּרָתִי



וּבְחוּט יְדִידוּתְךָ גַּם בַּעֲבוֹת חִשְׁקְךָ / הוֹאֵל מְשׁוֹךְ עִם יְחִידָתִי גְוִיָּתִי              25
