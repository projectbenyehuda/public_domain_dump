


	
	הכיתי בית מאהבי
	



הכיתי בית מאהבי – לחיך צבית חן / וידאל בנבנשת


הֻכֵּיתִי בֵית מְאַהֲבַי / בַּהֲתוּלֵי הַחֵשֶׁק וּבְדִבְרֵי הֲבַאי / בַּהֲפוֹךְ ה' אֶת עֵשֶׂב הַשָּׂדֶה לְגָאוֹן וּלְתִפְאֶרֶת / בְהִקָּבֵץ נְעָרוֹת לְהוֹצִיא מִמַּסְגֵּר אַהֲבָה מְסֻתֶּרֶת / וּבְבָתֵי הַנֶּפֶשׁ וְהַלְחָשִׁים / מְצוֹדְדוֹת נְפָשִׁים / וּבְחֵלֶק שְׂפָתַיִם / מושְׁכוֹת כְּסִיל וְנֶעֱקָשׁ דְּרָכַיִם / וַיַּאַסְרוּהוּ בִנְחֻשְׁתַּיִם / וּבְזֹהַר הַלְּחָיַיִם / וְלֹבֶן הַשִּׁנַּיִם / וְשַׁחֲרוּת הָעַפְעַפַּיִם / הָאָדָם יִרְאֶה לַעֵינַיִם / עֶרֶב וָבֹקֶר וְצָהֳרַיִם / וַאֲרֻבּוֹת הַשָּׁמַיִם / וּבְמַרְאוֹת הַנָּשִׁים / הַצּוֹבְאוֹת לִרְעוֹת בַּגַּנִים / רָאוּ עֵינַי נַעֲרָה עֵינֶיהָ יוֹנִים / נוֹשֵׂאת חֵן בְּעֵינֵי כָּל רוֹאֶיהָ / בְּאַהֲבָתָהּ הֶחֱזַקְתִּי וְלֹא אַרְפֶּהָּ / הִיא יָשְׁרָה בְעֵינַי מִכָּל בְּנוֹת עִירִי / לֶאֱחוֹז בַּסִּכְלוּת וְלַחְטִיא אֶת בְּשָׂרִי / וַתֵּשֵׁב בְּאֵיתָן / גִּנַּת הַבִּיתָן / בִּלְבוּשׁ שָׁנִי וְצֶבַע רִקְמָתַיִם / כְּסוּת עֵינַיִם / וּבֶגֶד כִּלְאַיִם / וַתַּעַד נִזְמָהּ וְחֶלְיָתָהּ / לַבָּאִים לַחֲקוֹר מִמְסַךְ יְדִידוּתָהּ / וְהָיָה בְמַחְשָׁךְ מַעֲשֵׂיהֶם / אוֹר נָגָהּ עֲלֵיהֶם / מַה גָּדְלוּ מַעֲשֶׂיהָ / מֶה עָצְמוּ רָאשֵׁי תְהִלּוֹתֶיהָ / יָפָה כַלְּבָנָה בָרָה כְזֹהַר הָרָקִיע / מִבְטַא שְׂפָתֶיהָ לְבָבוֹת יַבְקִיעַ / וּבְיַם אַהֲבָתָהּ לֵב שׁוֹמְעֶיהָ יַשְׁקִיעַ / וּבַחוּרֵי יִשְׂרָאֵל תַּכְרִיעַ / גְּוִיָּתָהּ כְתַרְשִׁישׁ סַפִּיר גִּזְרָתוֹ / מִבֵּינוֹת לְגַלְגַּל צְבִי תִּפְאַרְתּוֹ / יוֹצֵא מֵחֻפָּתוֹ / כְצֵאת הַשֶּׁמֶשׁ בִּגְבוּרָתוֹ / וּמִי הָעוֹמֵד בְּהֵרָאוֹתוֹ / הַתְשַׁלַּח בְּרָקִים מִבֵּין עֵינֶיהָ / וְהָיוּ לִמְאוֹרוֹת לְעֵינֵי מְאַהֲבֶיהָ / הֲתוֹצִיא מַזָּרוֹת מִבֵּין שָׁדֶיהָ וּמִחֵיקָה / וּמִבַּעַד לְצַמָּתָהּ שֶׁמֶשׁ צְדָקָה / הֲתַלְבִּישׁ צַוָּארָהּ / צְפִירַת תִּפְאָרָה / וְתוֹפַע עָלָיו נְהָרָה / הֲתְקַשֵּׁר מַעֲדָנוֹת כִּימָה בְּקִשּׁוּרֵי עֶדְיָהּ / וְדִמְדּוּמֵי חַמָּה עַל לֶחְיָהּ / מִי יַעֲצֹר כֹחַ אַהֲבַת כְּלוּלוֹתֶיהָ / וּמִי יִתְיַצֵּב לְפָנֶיהָ / וְלַיְלָה כַיוֹם תָּאִיר אֶל עֵבֶר פָּנֶיהָ / אֵין קֵצֶה לִתְכוּנוֹת נֹעַם תְּכוּנוֹתֶיהָ / אֵין חֵקֶר אֶל תַּכְלִית חֶמְדַּת מִדּוֹתֶיהָ / עַל כֵּן כַּאֲשֶׁר רְאִיתִיהָ / אַהֲבַת עוֹלָם אֲהַבְתִּיהָ / וּבְזִמְרַת זְמִירוֹתַי לַיְלָה וָיוֹם הִלַּלְתִּיהָ וָאַעַן וָאֹמַר:

לֶחְיֵךְ צְבִיַּת חֵן כְּיוֹם תָּאִיר וּבַת / עֵינֵךְ וְאִישׁוֹנֵךְ כְּאִישׁוֹן לַיְלָה



לַיְלָה וְיוֹם אֶרְאֶה בְפָנַיִךְ אֲבָל / גָּבַר מְאוֹר הַיּוֹם לְהָאִיר לַיְלָה
