


	
	ה' שבור זרוע רשע
	



ה' שבור זרוע רשע / רבי שמעון בן צמח דוראן (הרשב"ץ )


ה', שְׁבוֹר זְרוֹעַ רָשָׁע / מִבֶּטֶן פָּשַׁע / יִרְחֲקוּ בָּנָיו מִיֵּשַׁע



חֲיָלִים מְגַבֵּר / וְשָׁלוֹם כִּי אֲדַבֵּר / עוֹבֵר וּמִתְעַבֵּר



‘מַה לְּךָ וּלְשָׁלוֹם?’




מַכִּים עֲבָדִים / וְנוֹגְשִׁים וְרוֹדִים / בּוֹזְזִים וְשׁוֹדְדִים



וּפַסּוּ אֱמוּנֵנוּ / וּמֵרֹב עֲווֹנֵנוּ / כִּלִּינוּ שָׁנֵינוּ



קַוֵּה לְשָׁלוֹם.




עָתָק מִפִּיו מוֹצֵא / הֶבֶל פִּיהוּ יִפְצֶה / עוֹשֶׂה אֲשֶׁר יִרְצֶה



אוֹיֵב כְּמוֹץ יִדְּפֵם / יֶהְדְּפֵם / יִרְדְּפֵם



יַעֲבֹר שָׁלוֹם.




וְהוּא מִתְעַקֵּשׁ / לִהְיוֹת לִי לְמוֹקֵשׁ / לְהַדִּיחֵנִי מְבַקֵּשׁ



אוֹמֵר לִי: לֵךְ עֲבֹוד אֵל זָר / וְיָקֶם לָךְ אוֹמֶר תִּגְזָר / וּבוֹ תִּהְיֶה נֶעְזָר



וְיָשֵׂם לְךָ שָׁלוֹם.



נָחָשׁ הִשִּׁיאַנִי / אַךְ עַתָּה הֶלְאַנִי / וּבְכִלְאוֹ הֱבִיאַנִי



לִהְיוֹת לִמְחִיתָּה / אוֹמֵר לִי אִם עַתָּה / תָּמִיר כְּבוֹדְךָ אַתָּה



תָּבוֹא אֶל אֲבוֹתֶיךָ בְּשָׁלוֹם.





בּוֹנֶה בָּמָה / טְלוּאָה וְרָמָה / וּבְקוֹל דְּמָמָה



מַשְׁכִּים וּמַעֲרִיב /מִבַּיִת מָלֵא זִבְחֵי רִיב / לֶחֶם כְּזָבִים מַקְרִיב



וַיִּקְרָא לוֹ ‘ה’ שָׁלוֹם'.




רָבוּ כְּמוֹ רָבוּ / בְּמֵי מְרִיבָה רָבוּ / בְּרִיבָם עִמָּדִי נִצְּבוּ



וַאֲנִי אָמַרְתִּי בִּדְמִי יָמַי / אֵלֵךְ שְׁאוֹל בַּאֲשָׁמַי / כִּי מִפְּנֵי חַטָּאתִי בַּעֲצָמַי



אֵין שָׁלוֹם.




צוּר לְבָבִי וְחֶלְקִי, / אֶרְחָמְךָ, ה' חִזְקִי! / שָׂם לִי לַחֲרָדָה נֶשֶׁף חִשְׁקִי



אוֹיֵב שִׁנָּיו חוֹרֵק / וְאֵין מַצִּיל פּוֹרֵק / וְזֶרַע אֱמֶת נְטַעְתִּיו שׂוֹרֵק



זֶרַע הַשָּׁלוֹם.




מְבַשֵּׂר גְּאֻלָּה / לְשֵׁם וְלִתְהִלָּה / לְעֵדָה סְגֻלָּה



אֶשְׁלַח בְּבוֹא עִתּוֹ / בְּאֵיתָן קַשְׁתּוֹ / בְּרִיתִי הָיְתָה אִתּוֹ



הַחַיִּים וְהַשָּׁלוֹם.




חַיִּים וְאֹרֶךְ יָמִים / נָתַתִּי לוֹ לְעוֹלָמִים / וְכִי בֶּאֱמֶת וּבְתָמִים



קִנֵּא לֵאְלֹהָיו בְּחֵילוֹ / יְכַפֵּר עַל עַמּוֹ וּמִמָּוֶת גְּאָלוֹ / הִנְנִי נוֹתֵן לוֹ



אֶת בְּרִיתִי שָׁלוֹם.
