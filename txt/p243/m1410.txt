


	
	ה' שוועת יום
	



ה' שוועת יום / רבי שמעון בן צמח דוראן (הרשב"ץ)


ה', שַׁוְעַת יוֹם וְיוֹמַיִם / בְּלֵב נָמֵס כַּמַּיִם / כִּי כָּלְאוּ מִטַּל שָׁמַיִם



וַיִכָּלֵא הַגֶּשֶׁם.

מִקְוֵה יִשְׂרָאֵל / עַמְּךָ לְבָרֵךְ הוֹאֵל / רַחֲמֶיךָ הַיּוֹם שׁוֹאֵל



לְהַחְיוֹת לֵב נִדְכָּאִים / מַעֲשֶׂיךָ מְאֹד נִפְלָאִים / מַעֲלֶה נְשִׂיאִים



5    וְרוּחַ וְגֶשֶׁם.



עֶזְרָה נִמְצָא בְּצָרוֹת / הָשֵׁב אַפְּךָ מֵחֲרוֹת / וְעַל דִּבְרֵי הַבַּצָּרוֹת



עֲווֹנוֹת עָם לְנֶגְדְּךָ שַׁתָּה / לְשׁוֹנָם בַּצָּמָא נָשָׁתָּה / בַּאֲבוֹד הָאֲדָמָה חַתָּה



כִּי לֹא הָיָה גֶּשֶׁם.

וּלְמִי אֶזְעַק לְרַחֲמִים / הֲיֵשׁ מַגְשִׁימִים / בְּהַבְלֵי לְאֻמִּים



10   הֲלֹא לְךָ נְקַוֶּה / צְמֵאִים מַרְוֶה / וְעַל הָעָב תְּצַוֶּה



וְלֹא יַעְצְרֶנָּה הַגֶּשֶׁם.

נִפְלָאוֹת עוֹשֶׂה / שָׁמַיִם בְּעָבִים תְּכַסֶּה / וּבָם יָדִין עַמִּים וִינַסֶּה



וְאִם בְּרִיתוֹ שָׁמַר / וּמִפַּחֲדוֹ בְּשָׂרוֹ סָמַר / לַשֶּׁלֶג יֹאמַר



הֱוֶה אֶרֶץ וְגֶשֶׁם.

15   בְּלֵב נִשְׁבָּר וְנִדְכֶּה / בָּאתִי לְךָ מְבַכֶּה / כְּנַחַל דִּמְעָה מְפַכֶּה



עָבֵי שְׁחָקִים חַשֵּׁר / וְהָאָרֶץ רַבַּת תְּעַשֵּׁר / נַחֵית גְּדוּדֶיהָ



כַּאֲשֶׁר יָרַד הַגֶּשֶׁם

רַוֵּה תַּלְמֵי שָׂדַי / וּלְחַזֵּק רִפְיוֹן יָדַי / הָרִיק בְּרָכָה עַד בְּלִי דַי



שַׁוְעָתִי עָדֶיךָ תֶּאֱתֶה / וּבַשֵׂר תּוֹלַעַת יַעֲקֹב מְתֵי / יִשְׂרָאֵל עֲלֵה אֱכֹל וּשְׁתֵה



20   כִּי קוֹל הֲמוֹן הַגֶּשֶׁם.

צֶמַח הָאֲדָמָה / בָּרֵךְ עַד גָּדִישׁ וְקָמָה / וּבַשֵּׂר עֵדָה עֲגוּמָה



בְּבוֹא בְּאָזְנֶיךָ שַׁוְעָתִי / צִדְקָתִי רָאוּ וִישׁוּעָתִי / וְנָתַתִּי אוֹתָם וּסְבִיבוֹת גִּבְעָתִי



בְּרָכָה וְהוֹרַדְתִּי הַגֶּשֶׁם.



(מאיורקה, 1361 – אלג’יר, 1444)
