מָאִם בְּפִלְכַּיָּא

כתאב עזא לרבנו חננאל ז“ל אנפדה לה מר שמואל הנגיד הלוי ז”ל אלענואן

מכתב תנחומים לרבנו חננאל ז“ל שהעבר אליו מר' שמואל הנגיד הלוי ז”ל הכתבת


לִדְזִיו לֵהּ כְּבַר בִּתְיָה, דְּנָטְלִין אַפִּין בְּדִילֵה ּ לְאֻמְּתֵהּ, רַבַּנָא חֲנַנְאֵל רָחֲמִין , מִן שְׁמוּאֵל זוּטָא דְכֻלֵּי חַכִּימִין, שְׁלָם טָב וְנִחוּמִין.

מוֹדַעְנָא לֵהּ לְמַר דְּאַף עַל גַּב דְּאַנְפַּע עַדְבֵּה ּ, וּשְׁדָא בֵהּ זִמְנָא נִשְׁבֵּהּ, וְדָמְיָן לְיַמָּא כֵיבֵיה ּ, כַּד אִתְבְּעִי לִמְתִיבְתָא דִרְקִיעָא סָבֵהּ, לָאו מִנֵּהּ בִּלְחֹד אִעֲקַרָא שְׁרָגָא, וְאִטְּמַרָא בוֹצִינָא, וְאִפְרַדָא פָתוֹרָא, וְאִבְּטַשָׁא [מַדְבְּחָא ], אֶלָּהֵן מִכָּל דָּיְרֵי אַרְעָא דְיָתְבִין תְּחוֹת כָּל שְׁמַיָּא, מִשּׁוּם דִּבְרִיר לְהוֹן דְּ’רָעָה בָּאָה לָעוֹלָם' וּמִשּׁוּם דְּ’חָכָם שֶׁמֵּת הַכֹּל קְרוֹבָיו' , כָּל שֶׁכֵּן אֲנַחְנָא, בְּנֵי אוֹרָיְתָא, דְּאִצְּנַעָא מִנַּנָא אִצְטְלִיתָא, וְאִכְּסַיָא מַרְגָּלִיתָא, וּנְפַל מֵרֵישַׁנָא כִתְרָא, וְאִסְתַּלֵּק מִבֵּינַנֶא שֻׁפְרָא, וְאִחֲסַרְנָא גַבְרָא, וְאִתְיְהֵב בְּחוֹבִין לְעַפְרָא, וְלוּא יֶהְוֵה חַי בְּפֻרְקָן מַרְגָּלְיָן סְמִיקָן יָרְקָן הֲוָה מַפִּישְׁנָא טִימֵי, וּמַצְדִּיקְנָא אכלתי, וּמַעְתַּרְנָא אָסֵי, וְיָהִיְבָנא נִכְסֵי, וּמָכַרְ[נָא] תוֹתֵהּ אֻכְלוֹסֵי, כְּדִכְתִיב: מֵאֲשֶׁר יָקַרְתָּ בְעֵינַי נִכְבַּדְת ָּ, אֶלָּא מַאי כּו ֹ נַעֲבֵד? חֲלוֹטְתָא מֵאֱלָהּ שְׁמַיָּא, וּגְזֵרְתָא דְמָרָא עִלָּאָה, כְּדִכְתִיב: אָח לֹא פָדֹה יִפְדֶה אִישׁ, לֹא יִתֵּן [לֵאלֹהִים ] כָּפְרוֹ. וְרַבָּא וְגִבָּרָא יָדַע דְּכַד שְׁמַעִית שִׁמְעֵהּ וְאִתְחַוִּי לִי אִסְתַּלָּקוּתֵה ּ אִתַּמֵּס לִבַּאי, וַחֲלַשׁ רוּחַאי, וְאִתְחַלְחֲלו ּ חַרְצַאי, וְאַבַּע רֵישִׁי מֵימֵיהּ, וְאַחֵית עֵינִי דִּמ ְ[עָה ], וּבְזַעְתִּנּוּן לְמָאנֵא, וְאַכְנֵישִׁית חַבְרוּרָאִי וְאִשְׁתָּעִינָא בְמִלֵּיה ּ, [וְא ]וֹדֵעִית [מוֹקְרֵה ּ] לְמַאן דְּלָא יְדַע, וְאַשְׁמַעִית שְׁבָחֵהּ לְמַאן דְּלָא שְׁמַע, וְאַכְלֵית בְּגַנֵהּ: ‘[רָאוּי הָיָה] רַבֵּנוּ שֶׁתִּשְׁרֶה עָלָיו שְׁכִינָה, אֶלָּא שֶׁבָּבֶל גָּרְמָה לוֹ’ וַאֲרַמְנָא קָלָא וַאֲמַרְנָא: אַנְת ְּ, רַבַּנֶא חוּשִׁיאֵל, ‘עַל אֲדָמָה טְמֵאָה תָמוּת’, וֱכֻלַּנָא [יָדְעִינַן ] וּמִשְׁתְּמוֹדְעִין דְּ’נִתְפָּרְדָה הַחֲבִילָה' וַחֲסַר עַמָּא יִשְׂרָאֵל…[וְכַנַּפִית] כְּנוּפְיָא, וּגְזַרִית תַּעֲנִיתָא, וְאוֹקִימֵית אַשְׁכָּ[בְתָּא]…, וְאַתְקֵנָּא הֶסְפֵּדָא, וְאִתְעַתַּדְנָא ‘חֲלוּצֵי כָתֵף’… בִּדְבָרִים תְּחִלָּה כִּי הֵיכִי דַעֲבַדִית… וְכִדְ’אָמַר ר' יְהוּדָה אָמַר רַב: אֵין הַמְּנַחֲמִין רַשָּׁאִין לְדַבֵּר עַד שֶׁיִּפְתַּח אָבֵל תְּחִלָּה, שֶׁנֶּאֱמַר וְאֵין דּוֹבֵר אֵלָיו דָּבָר כִּי רָאוּ כִּי גָדַל הַכְּאֵב מְאֹד אַחֲרֵי כֵן פָּתַח אִיוֹב אֶת פִּיהו ּ' וְלִשָּׁן חַד נוּמֵינָא: ‘אִי עָנָו, אִי חָסִיד’, אִי גָדוֹל בְּאַפְרִיקָתָא ‘וְשֵׁם לוֹ בְרַקַּת’. וְכִי הֵיכִי דְאִעַסַּקִית בְּהֶסְפֵּדֵהּ בִּגְרָנָטָא הָכִי אִעַסַּקִית נַמִּי בְאַלְיֻסֶנָה, וְאַכְשַׁרִית פְּתִיחֵי, וְשַׁדַּרִית שְׁלוּחֵי ברתקרי לְקָרְטָבָה וּלְרֻבָּא דְהַלֵּי כַרְכֵי וּבְכֻלְּהֵי אִעֲבַד כִּי הֵיכִי דַעֲבַדִית. וּבְוַדַּאי נְהִירְנָא דִרְגִיז רָעְיָא עַל עָאנֵה ּ, וּתְקֵף מַלְכָּא עַל אֻמְּתֵהּ, וּכְדִכְתִיב ‘הַצַּדִּיק אָבָד וְאֵין אִישׁ שָׂם עַל לֵב’. וְאָמְרִינַן בְּהַנְהו ֹ הֶסְפֵּדֵי: ‘קָצַף עַל עוֹלָמוֹ וְחָמַס נְפָשׁוֹת וְשָׂמַח בָּם כְּכַלָּה חֲדָשָׁה, רוֹכֵב עֲרָבוֹת שָׂשׂ וְשָׂמֵחַ בְּבוֹא אֵלָיו נֶפֶשׁ נָקִי וְצַדִּיק’. וּמֵרֹב צַעֲרֵין חַבַּרִית הַנֵּי מִלּוּלֵי קַלִּילֵי וְשַׁדַּרְתִּנּוּן לְגַבֵּי מָר:


מָאִם בְּפִלְכַּיָּא וּפַצְחַיָּא / יֻצְפָּא, דְּכֹלָּא פוֹן צְוַח אִלְיָא?

לָא קָל דְּמִתְגַּבְּרִין בְּכִילוֹזָא / אוּף לָא דְּמִתַּבְּרִין בְּגַפַּיָא.

מָאִם שׁנוֹ סַגְיָא בְאַפְרִיקֵי, / לֵהּ אֳתְרְגִשׁ מָפִיס וְאִסְפַּמְיָא?

זִמְנָא, בְּלָא חֻלְפֵי וְלָא טֻפְרֵי, / הַדֵּם לְהַאי לֵיתָא כְּהַאי גַּדְיָא.

אֵכְדֵין מְרַע אָתֵא לְמַאן נוּמֵהּ / מֵיסֵי לְכָל מַרְעִין בְּלָא אָסְיָא?

אֵכְדֵין בְּלָא אַרְזוֹב כְּתֵפָא מִן / פַּרְוָא קְטַף מִנַּן כְּהַאי קַנְיָא?

שָׁתַן בְּפִילָוֵי חֲמַר, אָתַת / בִּישְׁתָּא, בְּלָא חַמְרָא לְכוֹן רָוְיָא

שִׁמְשָׁא בְמַדִנְחָא בְלָא זִיוֵהּ / מִשּוּםּ דְּסָט חַבְרֵה בְּגוֹ אוּרְיָא

מַקְשֵׁין וְכָל קֻשְׁיָא דְמַקְשׁו ֹ לֵהּ / פָּרֵק, וּמַלִּי יָת א2תַר קֻשְׁיָא.

הַאי מַן דְּנָקֵט לַהּ לְכַוְרָתָא / אִיהו ּ לְחֹד נָסַח לְכָל חוּלְיָא.


וְסוֹף כָּל מִלֵּי נֵימָא אַדִּילֵה ּ כִּי הֵיכִי דַאֲמַרו ּ עַלֵהּ צַדִּיקַיָּא וּמַלְאֲכֵי הַשָּׁרֵת: 'יָבוֹא שָׁלוֹם יָנוּחוּ עַל מִשְׁכְּבוֹתָם ' וְעַל רַבַּנָא חֲנַנְאֵל לְאַעֲדוֹיֵי אִבּוֹלָא וּלְקַבּוֹלֵי תַנְחוּמָא ‘סוֹף כָּל אָדָם לָמוּת וְהַכֹּל לְמִיתָה עוֹמְדִין, אַשְׁרֵי מִי שֶעֲמָלוֹ בַּתּוֹרָה וְעָשָׂה נַחַת לְיוֹצְרוֹ וְגָדַל בְּשֵׁם טוֹב וְנִפְטַר בְּשֵׁם טוֹב מִן הָעוֹלָם’ וְטוּבָךְ, [רַבַּנָא ] חוּשִׁיאֵל, דְּאִתְנַגַּדְת ְּ וּשְבַקְתְּ דֻּכְרָאנָא בִּבְרָא מַלְּפָנָא, שִׁכְלוּלָא דְאוֹרָיְתָא, שִׁיחוֹרָא דְתַלְמוּדָא, אַרְיֹוך ְ וְשָׁבוּר מַלְכָּא, סִינַי וְעוֹקֵר הָרִים דְּהַאי דָּרָא, דָּמֵי לְסוֹכָאָה אַוְלָא דְמִתְּקֵיל כְּלַפֵּי שִׁתִּין רִבּוֹא. דְּיָלֵד כְּוָתֵה ּ – [יוֹלִיד], וְאִי לָא – לָא יוֹלִיד, וּמַאן דְּאָזֵל וּמַנַּח דִּכְוָתֵה ּ – לָא אֲזַל, כָּל שֶׁכֵּן בְּצ ֵ[דִּיקַיָּא ] כְוָתֵה ּ טַלִּיל עֲלֵיהוֹן גַּנּוּאָן דִּי רַמְשָׂא וְאִתְרַבִּי לְהוֹן בְּרָא [עַד דִּי] לָא אִפְּרֵד אַבָּא, וְכִדְאָמְרִינָן הָתָם: 'עַד שֶלֹּא כָבְתָה שִׁמ ְ[שׁו ֹ שֶׁל עֵלִי] זָרְחָה שִׁמְשׁוֹ שֶׁל שְׁמוּאֵל הָרָמָתִי '. וּמִבָּתַר הָכִין אָמֵנָא [לֵהּ לְמָר דִּגְלֵי] קַמֵּי שְׁמַיָּא דמדאיתני אנא מרנא לְאִשְׁתָּעוֹיֵי… וּמִשּׁוּם גָּלְוָתָא דִּטְרַדו ּ לָנוּ וְעָק ָ[תָא] דֵּשְׁ[מַיָּא ]… בְּתִקְרוֹבֵי דַּח ֲ[קִירָא וְ]לָא אִחֲבַבִית ב… חַ[כְמֵי בָבֶל וְכָל אֶרֶץ יִ]שְׁרָאֵל… אַהֲדָדֵי וְכָל שֶׁכֵּן מוֹר דְקִים לִי בְגַוֵּה ּ דְּאָדָם גָּדוֹל הוּא וִיחִיד הַמַּעֲרָב הוּא הַשְׁתָּא וַדַּאי מִשּׁוּם תַּרְתֵּי אִחֲשִׁיבְנָא כְּאַחֵי וְאִדְּמִינָא לִבְרֵי אַבָּא קָרִיב: חֳדָא, קֻרְבָא דְאוֹרָיְתָא דְהִיא סְחוֹרָתַנָא וַאֲרֵי לֵהּ לַאוּתַנָא, וְכֵיוָן דְּאִיהִי הַוְיָא אֳחָתַן אֲנַן וַדַּאי אַחֵי אֲנַן, וּכְדִכְתִיב : ‘אֱמֹר לַחָכְמָה אֲחוֹתִי אַתְּ וּמוֹדָע לַבִּינָה תִקְרָא’. וְעוֹד, דְּהָא דְלֳיתִי לַאֲבוּי בְּאַשְׁכָּבְתָא וַאַחֵמִית לֵהּ הֶסְפֵּדָא וְקִים לִי דְכָל מָן דַּהֲוָה בְאַשְׁכָּבְתֵהּ מְזֻמָּן לְחַיֵּי הָעוֹלָם הַבָּא, וּמִשּׁוּם הַנֵּי תַרְתֵּי בָעֵינָא מִנֵּהּ דְּמָר דְּלָא לְפַגֵּר אִגַּרְתֵּהּ, וְלָא לְעַכֵּב תְּיוּבְתֵה ּ, בְּאוֹדָעוּת שְׁלֵמָה וּשְׁלָם אוֹרָיְתֵה ּ, וְחַבְרוֹהִי חַכִּימַיָּא , וְתַלְמִידוֹהִי סַבְרָנַיָּא. וּמִשּׁוּם דִּכְתַבִית כְּתָבָא דְנָן מִגּוֹ שִׁגּוֹשָׁא, וְנַפְשָׁא שַׁבִּישָׁא, וְדַעְתָּא חַלִּישָׁא, וְלִבָּא בְאִישָׁא, אַהַנֵּי עָקֵי דְאַקִּיפו ּ עֲלֵנָא עַד דְּ 'אֵין הַשּׁוֹטֶה נִפְגָּע וְאֵין בְּשַׂר הַמֵּת מַרְגִּישׁ בְּאִזְמֵל ' הם הָכִי אַקְבֵּלִית אַפֵּי מָר בְּמִלָּא קַלִּילָא וּשְׁבָחָא פוּרְתָא. וְכֵיוָן דְּיָדַע מָר דִּצְלוֹתָא דַּחֲסִידֵי טְפֵי מְהַנְּיָא קַמֵּי שְׁמַיָּא מִחַיָּב לְמִבְעֵי עֲלַן רַחֲמֵי, כִּדְאָמְרִינָן בְּמִי שֶׁאֵרְעו ֹ דָבָר, וּכְדִבְתִיב : וַיִּתְפַּלֵּל אַבְרָהָם אֶל הָאֱלֹהִים וַיִּרְפָּא אֱלֹהִים וג '. וּשְׁלָם לָעַד עַל שָׁלֵם. שְׁמוּאֵל הַלֵּוִי ב“ר יוֹסֵף נ”ע.
