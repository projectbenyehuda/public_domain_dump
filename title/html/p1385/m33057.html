המחברת התשיעית – וכימה סהר אהבת

נְאֻם לְמוּאֵל בֶּן־אִיתִיאֵל: וַיְהִי כַאֲשֶׁר הַסְּתָיו עָבַר וְתַמּוּ יְמֵי חֶלְדּוֹ / הַגֶּשֶׁם חָלַף וְהָלַךְ לוֹ / וְכִי הָפַךְ עֹרֶף / הַקַּר וְהַחֹרֶף / כִּי הַקַּיִץ שִׂפַּח קָדְקָדָם וְהִפִּיל אָשְׁיוֹתָם / וְנֶהְפַּךְ יְסוֹדָם / וַיִּסְבֹּר עַל הַזָּקֵן בֶּגֶד / הַכַּר עַל הַגֶּלֶד / וַיָּחָם בְּשַׂר הַיֶּלֶד / וְהָעֵצִים נָתְנוּ פִרְיָם / וְהָאֳנִיּוֹת דּובְרוֹת בַּיָּם. / וַיְהִי אִישׁ מִבְּנֵי הַנְּדִיבִים וְאִישׁ מִמִּשְׁפַּחַת מְלָכִים וּשְׁמוֹ צַלְמוֹן / מִגַּת רִמּוֹן / וְלוֹ בֵן אֶחָד וּשְׁמוֹ סַהַר / וְלֶחְיוֹ מְלֻבָּשׁ אוֹר וְזֹהַר. / וַיִּבְרַח מֵאִתּוֹ / כִי עֲצָבוֹ וְגִדְּפוֹ וַיֵּלֵךְ אֶל יָם יָפוֹ / וַיִּמְצָא עַל מְבוֹאוֹת הַיָּם אֲנָשִׁים / בּוֹרְחִים מִפְּנֵי רֹב נָשִׁים / וַיִשְׂכֹּר הָאֳנִיָּה עַל כָּל סוֹחֵר וְיָם עוֹבֵר / וְהָאֳנִיָּה חִשְּׁבָה לְהִשָּׁבֵר / כִּי הִתְגֹעַשׁ וְרָעַשׁ הַיָּם / וְנָשְׂאוּ נְהָרוֹת דָּכְיָם / וַיִּצְעֲקוּ לְמַשְׁבִּיחַ שְׁאוֹן יַמִּים לְהַשְׁבִּיחַ שְׁאוֹן גַּלֵּיהֶם. / וַיָּטִילוּ אֶת הַכֵּלִים אֲשֶׁר בָּאֳנִיָּה אֶל הַיָּם לְהָקֵל מֵעֲלֵיהֶם / וְהָעָם צוֹעֲקִים וּמַר צוֹרְחִים / וְנֶאֱנָקִים וְנֶאֱנָחִים. / וַיִּירְאוּ הַמַּלָּחִים וַיַּשְׁפִּילוּ הַקְּרִיּוֹת / אֲשֶׁר בָּאֳנִיּוֹת / וְרַב הַחֹבֵל בַּיְרִיעוֹת / מִפְּנֵי רוּחַ הַנֵּס / בַּל יְחַזְּקוּ תָּרְנָם בַּל פָּרְשׂוּ נֵס / וְכָל אִישׁ מֵהֶם יִדֹּר לָתֵת לַעֲנִיִּים כַּסְפּוֹ / וְאוּלַי יַעֲמֹד הַיָּם מִזַּעְפּוֹ / וְגַם הַנַּעַר בֹּהַר וּמַר צוֹרֵחַ / כִּי מִפְּנֵי אָבִיו הוּא בוֹרֵחַ / וְהָאֳנִיָּה רָפְתָה מֵרוּחַ גְּדוֹלָה וְחָזָק מְפָרֵק הָרִים / וּשְׁבָרָהּ כְּשֵׁבֶר נֵבֶל יוֹצְרִים / וַיַּחֲזֵק סַהַר בְּלוּחַ / וַיִּשָּׂאֵהוּ רוּחַ / וַתְּבִיאֵהוּ עַד יַם הָעֲרָבָה / וְהִנֵּה עַל שְׂפַת הַיָּם עִיר וּשְׁמָהּ צוֹבָה. / וַיְהִי בְּצֵאתוֹ אֶל הַיַּבָּשָׁה, / וַיִּשָּׂא מְשָׁלוֹ בְשִׁירָה חֲדָשָׁה:

וּמִי רָאָה גְבוּרוֹת אֵל / וְנוֹרְאוֹתָיו כְּיוֹרְדֵי־יָם

יְחִילוּן מִשְּׁאוֹן גָּלָיו / וְיוֹדוּ לַאֲשֶׁר לוֹ יָם

וְיֶחְזוּן נִפְלְאוֹת שַׁדַּי / וְיֶהְמוּ כַּהֲמוֹת הַיָּם

וּמִי יַעְמֹד בְּעֵת יִרְעַשׁ / וְיָכִיל מִשְׁבְּרֵי הַיָּם

וּמִי נִסְתַּר בְּיוֹם אַף אֵל / וְלוּ נֶחְבָּא בְקַרְקַע יָם

וְלוּ יִסַּק לְרוּם־שַׁחַק / וְיַצִּיעַ תְּהוֹמוֹת יָם

וְלוֹ אֶרֶץ וְשָׁמַיִם / וְיַבָּשָׁה וְיָם עַד־יָם.

וְהוּא גוֹעֵר בְּזַעַף יָם / וּמַשְׁבִּיחַ שְׁאוֹן הַיָּם

וְנָהָר יַחֲרִיב יַבֵּשׁ / וְאָזְלוּ מֵי תְהוֹם מִיָּם

וְהוּא בוֹנֶה עֲלִיּוֹת רוֹם / וְהוּא קוֹרֵא לְמֵי הַיָּם

וְהוּא צִוָּה לְהָרְעִישׁ יָם / נְהָרוֹת נָשְׂאוּ דָכְיָם

וְשָׁבַר הָאֳנִיָּה, כִּי / כְקַשׁ נִדָּף יְשַׁו עָבְיָם

וְאַנְשֵׁי הָאֳנִי בוֹכִים / וְדִמְעָתָם עֲלֵי לֶחְיָם

כְּמִסְפֵּד עַל בְּכוֹר סָפְדוּ / וְנָהוּ אָז נְהִי כַּיָּם

וּמֵהֶם יִדְּרוּ נֵדֶר / וְיוֹדוּ עַל עֲווֹן מִרְיָם

לְבַל־יַקְשׁוּ לְבָבָם עוֹד / וְיַעַזְבוּ הֲלֹךְ קִרְיָם

וּבַל־יֵלְכוּ בִקְרִי עוֹד / וּבַל יִשְׁגוּ בְחֵטְא קָשְׁיָם

וּמֵהֶם יִדְּרוּ הַעְנֵק / עֲנִיֵּיהֶם לְפִי עָנְיָם

עֲדֵי רָגַע בְּכֹחוֹ יָם / אֲזַי הִתְנַצְּלוּ עֶדְיָם

וְהִטִּילוּ אֱלֵי יָם כָּל / כְּלֵי כַסְפָּם כְּלֵי צִבְיָם

וּבִגְדֵי מַחֲמַדֵּיהֶם / לְבוּשֵׁי שֵׁשׁ לְבוּשׁ מִשְׁיָם


וְהִפִּילוּ אֲזַי חוֹבְלִים / קְרִיּוֹתָם, כָל־כִּלְיָם

וְהִשְׁחוּ אֶת יְרִיעוֹתָם / וְכָל עַרְבָּם וְכָל שִׁתְיָם

וְלֹא הוֹעִיל כְּלִי נִשְׁקָם / וְחֵץ קַשְׁתָּם וְכָל־כִּלְיָם

גְּבוּרָה נָשְׁתָה מֵהֶם / וְרוֹפְאִים שָׁכְחוּ צָרְיָם

עֲדֵי כִי מָעֲדָה רַגְלָם / וְלֹא עָמְדוּ עֲדֵי דָחְיָם

כְּאֶבֶן יָרְדוּ טָבְעוּ / וְלוּ אַדִּיר יְהִי עָיָם.

כְּאַיִן כָּל־הֲדָרָם שָׁב / וְשָׁוְא הוֹדָם וְכָל־יָפְיָם

וְחַיּוֹת יָם קְטַנּוֹת עִם / גְּדוֹלוֹת אָכְלוּ דַּיָּם

וְקִבְרֵיהֶם מְעֵי דָגִים / לְשָׁמָּה נָהֲגוּ שִׁבְיָם

בְּלֶב־יָם יָצְאָה רוּחָם / וְשָׁמָּה שִׁלְּמוּ נִשְׁיָם

וְכֻלָּם אָבְדוּ רֶגַע / וְרוֹאִים יֹאמְרוּ אַיָּם

אֲרוֹמֵם אֵל אֲשֶׁר קֶדֶם / בְּבֶטֶן כּוֹנֲנָם חִיָּם

אֲשֶׁר הַיּוֹם גְּאָלַנִי / בְּחַסְדּוֹ מִשְּׁכֹן עִיָּם

מְבֹרָךְ הוּא לְעוֹלְמֵי־עַד / וְעַד־נֵצַח שְׁמוֹ קַיָּם.


וַיְהִי הוּא מְדַבֵּר אֶת הַשִּׁירָה / וַתַּשְׁקֵף אֵלָיו נַעֲרָה מֵהַבִּירָה / וַתֵּרֶא יָפְיוֹ וַתִּתְמַהּ / וַתָּרָץ וַתַּגֵּד לְאִמָּהּ / וַתָּבֹא הָאֵם וַתִּרְאֵהוּ וַתַּגֵּד לַחֲבֶרְתָּהּ / וַתָּרָץ גַּם הִיא וַתַּגֵּד לִגְבִירְתָהּ / וַתָּבֹאנָה כָל נַעֲרוֹת הַבִּירָה לִרְאוֹתוֹ נוֹעָדוּ / אִשָּׁה רְעוּתָהּ לֹא פָקָדוּ / וַתִּתְיַצֵּבְנָה לַחֲזוֹת הֲדַר יָפְיוֹ וְלָשׁוּר / בָּנוֹת צָעֲדָה עֲלֵי שׁוּר / וַיִּרְאוּהוּ וְהִנֵּה פָּנָיו פְּנֵי אַרְיֵה / לְפָנָיו לֹא קָם כָּמֹהוּ וְאַחֲרָיו לֹא יִהְיֶה / וְשִׁנֵּי אֲרָיוֹת שִׁנָּיו / וְכִי קָרַן אוֹר פָּנָיו / וּשְׂפָתָיו בִּהְיוֹת מְדַבֵּר / נוֹטְפוֹת מוֹר עוֹבֵר. / וַיְהִי כְּכַלּוֹת הַנַּעַר שִׁירָתוֹ / לְהוֹדוֹת עַל־תְּשׁוּעָתוֹ / וַיִּשָּׁא עֵינָיו לִרְאוֹת הַבִּירָה / וַיִשְׁמַע בְּקִרְבָּהּ קוֹל הָמוֹן שָׁלֵיו בָּא / וַיֹּאמֶר: ״מֶה קוֹל הֶהָמוֹן בְּתוֹךְ הָאַרְמוֹן״ / וַיַרְא וְהִנֵּה דַלְתוֹת הַבִּירָה צָלָלוּ / וְכִרְאוֹתָם יְפִי סַהַר נִבְהֲלוּ / וּלְבוֹרֵא הַדְרָתוֹ הִלֵּלוּ / וּכְשָׁמְעוֹ קוֹל הֲמֻלָּה / וַיֹּאמַר: ״אֵין זֶה כִּי אִם בֵּית הַתְּפִלָּה״. / וַיַּרְא וְהִנֵּה פָנִים נוֹצֲצִים / וְעֵינַיִם מִתְלוֹצֲצִים / וְשָׁם שְׁנֵי בְנֵי כוּשִׁים חֲלוּצִים / וַיְמָרֵרֻהוּ וָרֹבּוּ וַיִּשְׂטְמֻהוּ בַּעֲלֵי חִצִּים / וַיִּטְמְנוּ לוֹ רָשֶׁת / וִיִּמְצָאוּהוּ הַמּוֹרִים בַּקֶּשֶׁת / וַיֹּאמַר לָהֶם בִּרְאוֹתוֹ כִּי בְשַׁחְתָּם נִתְפַּשׂ: ״תַּמְנוּ חֵפֶשׂ מְחֻפָּשׂ״ / כִּי אָמַר: ״אֵיךְ מִיָּם נֶחְלַצְתִּי / וְעַתָּה בַבַּיִת נִמְחַצְתִּי / וְאֵיךְ לִבִּי וְנַפְשִׁי יַבְרִיחֵם בֶּן־קֶשֶׁת / וְהָיוּ לְדָם בַּיַּבֶּשֶׁת / וְאֵיךְ יְרִיבוּן עֵינַיִם / בְּאִישׁ יָגֵעַ וּרְפֵה יָדַיִם.״


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

הָהּ כַעֲמָלֵק נֶחְשְׁבוּ עֵינַיִם / לִלְחֹם בְּיָגֵעַ יָדַיִם

הִנֵּה עֲמָלֵק מִלְּפָנִים נִקְבְּצוּ / וַיִּלְחֲמוּ אֶת יוֹצְאֵי מִצְרַיִם

כֵּן בַּעֲלֵי חִצִּים לְחָצוּנִי בְּיוֹם / הִנָּצְלִי מֵעִמְקֵי הַמָּיִם

אָנָּא עֲלֵי יָמִין נְטוּ אוֹ עַל שְׂמֹאל / אַל תִּשְׁפְּכוּ דַם בָּר נְקִי כַפָּיִם

חֵץ קַשְׁתּוֹתֵיכֶם גְּדוּד יָמִית וְכָל / גִּבּוֹר וְלוּ יִהְיֶה כְּאִישׁ בֵּינָיִם

לֵב רוֹאֲכֶם אָמְנָם בְּחַכָּה תַעֲלוּ / רֶגַע וְהַמַּזְלֵג שְׁלֹשׁ שִׁנָּיִם

אִלּוּ יְהִי הַלֵּב כְּלֵב אַרְיֵה יְמַס / צִירִים אֲחָזוּהוּ וּפִיק בִּרְכָּיִם.




שָׁמְעוּ שִׁירוֹ וַתִּבָּהֵלְנָה / וְלֹא יָדְעוּ עַל מָה תַבֵּטְנָה / אִם לְשִׁירוֹ / אִם לַהֲדָרוֹ / אִם לְטוּב מַעֲנָיו / אִם לְזֹהַר פָּנָיו / וְהִנֵּה תַפּוּחַ עָבַר וְחָלַף / בְּמוֹר וּבִנְרָדִים מְעֻלָּף / וַיֻּשְׁלַךְ אֵלָיו בְעַד הַחוֹמָה / מֵאֵת צְבִיָּה וּשְׁמָהּ כִּימָה / בָּרָה כַּחַמָּה / וּדְבַר אַהֲבָה בּוֹ מְחֻקָּק / שִׁבְעָתַיִם מְזֻקָּק / כִּי שָׁמְעָה שִׁירָתוֹ / וְנֹעַם זְמִירָתוֹ.


וַתִּשָּׂא מְשָׁלָהּ / אַף־הִיא אֲמָרֶיהָ תָשִׁיב לָהּ: שָׁלוֹם לְךָ שָׁלוֹם הֲרוּג עֵינַיִם / מִכּוֹאֲבָה מִשְׁתַּחֲוָה אַפָּיִם

חוֹלַת אַהֲבֶיךָ הֲרוּגַת חִשְׁקְךָ / לָקְחָה כְפִי כָל־חִשְׁקְךָ כִּפְלָיִם

מִי יִתְּנֵנִי יוֹם וְלַיִל עִם צְבִי / בִּשְׂחוֹק חֲבֹק אַעְנִיקְךָ שָׁדָיִם.


וַיִּשָּׂא עֵינָיו וַיִּרְאֶהָ / וַיִּבָּהֵל מַהֲדַר מַרְאֶהָ / וְאֶל יְפִי פָנֶיהָ / וַתִּשְׁתַּחוּ לוֹ וַתִּשַּׁק לוֹ עַל־יָדָהּ / כְּמִשְׁפַּט כָּל עָפְרָה אֶל־יְדִידָהּ / וַתִּשָּׁקֵהוּ וְהוּא נְשָׁקָהּ מֵרָחוֹק / כְּמִשְׁפַּט יְדִידִים הַנִּפְרָדִים כְּחֹק / וְהִנֵּה סְרִיס הַמֶּלֶךְ גָּעַר בָּהּ וּבִשְׁאָר הַנְּעָרוֹת / וַתִּשָּׁאַרְנָה דַלְתוֹת הַבִּירָה לְפָנָיו סְגוּרוֹת / וְנִשְׁאֲרוּ בְצַלְעוֹתָיו לֶהָבוֹת וְכִידוֹדִים / בּוֹעֲרִים כְּמַרְאֵה הַלַּפִּידִים / וַיְהִי כִפְנוֹתָהּ לְדַרְכָּהּ / וַתִּתְחַלְחַל הַמַּלְכָּה / וְלִבּוֹ בְּקִיר לִבָּהּ נֶחְבָּא / וַיָּמָת לִבָּהּ בְּקִרְבָּהּ. / וַיֵּלֵךְ אֶל דֶּרֶךְ הַמְּדִינָה / בְּלֹא דַעַת וּתְבוּנָה / וַיֹּאמֶר לְנַעַר אֶחָד צָעִיר: / ״הַרְאֵנִי נָא אֶת מְבוֹא הָעִיר״! / וַיָבֹא וַיֵּשֵׁב לוֹ בִרְחוֹב הָעִיר בּוֹכֶה / וְדָבָר מֵהַנַּעֲרָה חוֹכֶה. / וַיְהִי לְעֵת עֶרֶב / כִּקְדֹחַ לֵב הַחוֹשֵׁק וְהַקֶּרֶב / וַיִּגְדַּל כְּאֵבוֹ / וַיֵּאָנַח לְבָבוֹ / וַיִּשָּׂא אֶת קוֹלוֹ בִּבְכִי וַתִּגְדַּל צַעֲקָתוֹ / וְהִנֵּה אִשָּׁה לִקְרָאתוֹ / וַתִּגְעַר בּוֹ וַתֹּאמַר ״הֲצִירִים אֲחָזוּךָ / אִם בְּכִימָה חָשַׁקְתָּ / מַה לְּךָ כִּי נִזְעַקְתָּ״? / וַיְהִי כְּשָׁמְעוֹ אֶת שֵׁם כִּימָה חֲשׁוּקָתוֹ / וַתֵּצֵא נִשְׁמָתוֹ / וַיַּךְ לֵב הַנַּעַר אוֹתוֹ / וַיֹּאמַר: ״מִי אַתְּ גְּבִרְתִּי וּמִי כִימָה?״ / וַתֵַּעֲנֵהוּ לֵאמֹר: ״הֲלֹא כִימָה / שֵׁם אֲהוּבָתְךָ הַנְּעִימָה / בַּעֲלַת הַתַּפּוּחַ / אֲשֶׁר בְּשֶׁמֶן הַמּוֹר מָשׁוּחַ״ / וְכַאֲשֶׁר הִגִּידָה לוֹ אֶת דְּבַר הַנַּעֲרָה אֲשֶׁר חָשַׁק וַחֲשָׁקַתְהוּ / לְבָבוֹ פַּלָּצוּת בִּעֲתָתְהוּ / וַתֹּאמֶר לוֹ: ״כִּי חָשַׁקְתָּ כִּימָה / מִי יַעֲלֶה לְךָ הַשָּׁמַימָה / הֲלֹא יָבֹא בְלִבְּךָ מֹרֶךְ / כִּי יִרְחַק מִמְּךָ הַדֶּרֶךְ / הֲלֹא הִיא כִימָה אֲשֶׁר רָאִיתָ בַבִּירָה / הִיא בַּת מֶלֶךְ וְהִיא הַגְּבִירָה / הֲלֹא כָל אוֹהֵב כִימָה / בְּחִצִּים וּבְקֶשֶׁת יָבֹא שָׁמָּה״. / וַיְהִי כְּשָׁמְעוֹ מִלֶּיהָ וַיִּפֹּל כְּמֵת אַרְצָה / כִּי יָצְאָה רוּחוֹ וְנַפְשׁוֹ קָצָה. / וַתְּדַבֵּר הָאִשָּׁה עַל לִבּוֹ / אוּלַי תְּנִיחֵהוּ מֵרָגְזוֹ וּמֵעָצְבּוֹ / וַיֵּאָנֵק וַיֵּאָנַח / וַיֹּאמַר: ״נַפְשִׁי מִשָּׁלוֹם תִּזְנַח״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

רְאִיתִיהָ וְרָאַתְנִי / חֲשַׁקְתִּיהָ חֲשָׁקַתְנִי

חֲקַקְתִּיהָ עֲלֵי לִבִּי / וְעַל לִבָּה חֲקַקַתְנִי

וְעַל־יַד מַלְאֲכֵי דֹדִים / נְשַׁקְתִּיהָ נְשָׁקַתְנִי

צְרָפַתְנִי בְּכוּר־אַהַב / וְעַל־לִבָּהּ יְצָקַתְנִי

וּבִגְדוּדָהּ הֱבִיאַתְנִי / וְשָׁמָּה הֶחֱזִיקַתְנִי

וְאֵיךְ עַתָּה גְזָלַתְנִי / וְקֹדֶם הֶעֱנִיקַתְנִי

תְּמוֹל תִּגְעַר בְּעוֹשֵׁק דַּל / וְשָׁם הַיּוֹם עֲשָׁקַתְנִי

הֱנִיאַתְנִי וְהֶלְאַתְנִי / מְחָצַתְנִי מְחָקַתְנִי

וְיוֹם מַתִּי. דְּעוּ־נָא כִּי / פְּרִידָתָהּ הֱמִיתַתְנִי.


וְהָאִשָּׁה עָשְׂתָה בְעָקְבָה לִרְאוֹת אַהֲבָתוֹ / וַתָּבֹא לְנַסּוֹתוֹ / כִּי אָמְרָה כִּימָה אֶשְׁלְחָה וַאֲנַסֵּהוּ / מַה־יִּהְיֶה מִשְׁפַּט הַנַּעַר וּמַעֲשֵׂהוּ / כַּאֲשֶׁר רָאֲתָה הָאִשָּׁה חִשְׁקוֹ בָהּ / וַתִּתֵּן לוֹ כְתָבָהּ / וַיִּפְרְשֵׂהוּ וְהִנֵּה רֵיחֵי בְשָׂמֶיהָ הֵפִיחוּ / וּמִלֵּי אַהֲבָה יָרִיעוּ אַף יַצְרִיחוּ.


וַתִּשָּׂא מְשָׁלָהּ בִּכְתָבָהּ / וַתָּלֶן עַל סַעַר אֲהוּבָה:

דַּי צְבִי דַּי – הֲלַעֲשֹׁק / יַעֲלַת־חֵן? – לְאַט לְאַט

אֵיךְ תְּעַנֶּה לְבַב צְבִ/יָּה וְנָמֵס מְעַט מְעַט

הָהּ הֲמוֹעִיל בְּדַכְּאוֹ? / רַפְּאוֹ אוֹ שְׁחַט שְׁחַט

לֵב יְשַׁוַּע בְּמַר נְדוֹד / אָזְנְךָ הַט וְהַט וְהַט

וְיִהְיֶה רָץ וּרְפָא / כִּי כְבָר מֵת וְקָט וְקָט

אוֹ הֲתִשְׂמַח בְּשָׁאֲטְךָ / אֶל אֲסִיר דּוֹד שְׁאָט שְׁאָט

בַּעֲבֹר דּוֹד חֲשָׁקוֹ עֲלֵי / לִבְּךָ שָׁט וּמָט וְנָט.


וַיְהִי כִּקְרֹא הָעֹפֶר / אֶת דִּבְרֵי הַסֵּפֶר / וְכִרְאוֹתוֹ כִּי מָצָא חֵן בְּעֵינֶיהָ / וְכִי הֵפִיק רְצוֹנֶיהָ / וַיַּעֲמֹד לִבּוֹ מִזַּעְפּוֹ / וַתֹּאמֶר לוֹ הָאִשָּׁה: ״הַעֲבֵר דְּאָגָה מִלִּבֶּךָ / וְהָסֵר כַּעַס מִקִּרְבֶּךָ״ / וַיֹּאמֶר לָה: ״הֶחֱיִתִינִי כִּי חֶסֶד גְּמוּלֵךְ / וְחָיְתָה נַפְשִׁי בִּגְלָלֵךְ / וְהִנֵּה שָׂנֵאתִי אֶת הַחַיִּים וּבְחַיַּי אָקוּצָה / מָשְׁכֵנִי אַחֲרֶיךָ נָרוּצָה״. / וַתּוֹלִיכֵהוּ אֶל הַבִּירָה / אֲשֶׁר שָׁם כִּימָה הַגְּבִירָה / וַתּוֹשִׁיבֵהוּ תּוֹךְ הֶחָצֵר / וְנִשְׁאַר כְּאִישׁ אֵין כֹּחַ עוֹצֵר / וּכְעָמִיר אַחֲרֵי הַקּוֹצֵר / וְלִבּוֹ בּוֹעֵר בְּחֹם אַהֲבָתוֹ / בְּתוֹךְ הָאַרְמוֹן / כְּדוּד נָפוּחַ וְאַגְמוֹן.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

לֵב כָּל אֱנוֹשׁ דָּם וּבָשָׂר / לֵב חוֹשְׁקִים מֵרְשָׁפִים

לִבּוֹת יְעוּפוּן כְּנֶשֶׁר / כִּי כֵן רְשָׁפִים כְּעָפִים

כָּל יָד תְּמַשֵּׁשׁ לְבָבָם / תֵּצֵא כְעוּגַת רְצָפִים.


וַיְהִי כִמְיַחֵל וְיִצְרוֹ יִכְבֹּשׁ / וְיָחִיל עַד בּוֹשׁ / וְהִנֵּה בָאוּ נְעָרוֹת שְׁתַּיִם / כְּאוֹר הַשֶּׁמֶשׁ בַּחֲצִי הַשָּׁמַיִם / וַיֵּלֵךְ עִם הַנְּעָרוֹת / אֶל מִבֵּית לַשְּׂדֵרוֹת / וַיַשְׁכִּיבוּהוּ בְמִטָּתוֹ / כָּל הַלַּיְלָה עַד לְמָחֳרָתוֹ / וַיְהִי כִּזְרֹחַ הַשֶּׁמֶשׁ / וְהִנֵּה נְעָרוֹת שְׁתַּיִם יָפוֹת מִנַּעֲרוֹת אֶמֶשׁ / וְהֵרִיחוּ בִגְדֵיהֶן כְּבוֹאָן אֶל הָאַרְמוֹן / מֹר אֲהָלִים וְקִנָּמוֹן / וַתִּלְעַגְנָה לוֹ לְהוֹסִיף עַל חִילוֹ / וְתֹּאמַרְנָה לוֹ: / ״אֵיךְ הֵנָּה הוּבֵאתָ / הַבְּכִסְלָתְךָ בָאתָ / וְאִם בְּאַהֲבָה בִקַּשְּׁתָּ הֲלֹא יָדַעְתָּ / כִּי הַבָּא אֶל חֲדַר הַנְּעָרוֹת נִכְרָת וְנִצְמָת / וְהַבָּא אֶל הַשְּׂדֵרוֹת יוּמָת״. / וַיַּעַן סַהַר לֵאמֹר: ״הִנֵּה הִסְכַּלְתִּי אִם בְּחִשְׁקִי אִם בְּשִׂכְלִי / וְהִנֵּה אִם רַע בְּעֵינֵיכֶן אָשׁוּבָה לִּי״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

מְנָעוּנִי מְתוּקֵי דוֹד / הֱנִיאוּנִי מְרוֹרֵי נוֹד

וְיֹאמְרוּ כִּי חֲשֹׁק דּוֹד חֵטְא / וְהַחוֹטֵא אֲשֶׁר יִבְגֹּד

וְדוֹד בּוֹגֵד בְּדוֹד כִבְגֹד / בְּצוּר מִי שֶׁלְּכָל יִסְגֹּד.


וַתִּשְׁתַּחֲוֶינָה לוֹ וַתִּפֹּלְנָה לִפְנֵי רַגְלָיו / כְּשָׁמְעָן יְפִי מִלָּיו / וַתֹּאמֶר הָאַחַת: ״נִרְאֲתָה / כִּי כִמְעַט רוּחוֹ נִכְרְתָה / וְכִמְעַט נַפְשׁוֹ נִשְׁמְדָה / בֹּאָה אִתִּי הַבַּיְתָה וּסְעָדָה״ / וַתֹּאמֶר הָאַחַת: ״בֹּא תָבֹא עִמִּי / אֶל
הֶחָצֵר הַפְּנִימִי / כִּי הַנַּעֲרָה הַנִּצֶּבֶת עִמְּךָ / בָּזָה לְךָ לַעֲגָה לְךָ״. / וְהִנֵּה אַרְבַּע נְעָרוֹת כִּצְבָאוֹת / מֵהַחֶדֶר יוֹצְאוֹת / וּבְצֵאתָן נִמְלָא הַבַּיִת מִזָּהֳרָן / וַיֹּאמַר בִּרְאוֹתוֹ רַב מְאוֹרָן / וְכִי זָרַח עַל־כָּל־פִּנָּה וְעַל כָּל פָּרֶץ: / ״הַשֶּׁמֶשׁ יָצָא עַל הָאָרֶץ״. / וַיָּקָם וַיִּשְׁתַּחוּ אֲלֵיהֶן / וַיִּשְׁתַּחוּ עַל מִדְרַךְ רַגְלֵיהֶן / וַיֹּאמַר: ״מִי מִכֶּם גְּבִרְתִּי כִימָה / בַּעֲלַת הַשִּׁירָה הַנְּעִימָה / הַחֲקוּקָה בַתַּפּוּחַ הַמֻּשְׁלָךְ בְּעַד הַחוֹמָה״? / וַתֹּאמַרְנָה לוֹ:


״שְׁמַע צְבִי הַנָּאֶה / שָׂא נָא עֵינֶיךָ וּרְאֵה״: / וַיִּשָּׂא עֵינָיו אֶל הַחֶדֶר / וְהִנּוֹ נִסְתַּר בְּפָרֹכֶת יְקָר וָאֶדֶר / וְכָתוּב בָּהּ ״קְרָא אֶת הַשִּׁירָה / בְּטֶרֶם תָּבֹא הֶחָדְרָה״ / וְכָתְבָה בָהּ שִׁיר אַהֲבָה / וַתַּגֶּד לוֹ אֶת כָּל לִבָּהּ:


יְדִידִי אַחֲרַי רַעְיָה / אֲנִי עַל לֶחֱיָהּ מַסְוֶה

וְאוֹר תַּלְוֶה לְכָל מָאוֹר / וְשֶׁמֶשׁ זָהֳרָהּ לַוֶּה

לְכָה אִיעָצְךָ דּוֹדִי / בְּשֵׂכֶל בֹּא וְהִשְׁתַּחְוֵה

וְהִכָּנַע אֱלֵי יָפְיָהּ / וּמִבְחַר שִׁיר תְּהַחֲוֵה

רְצוֹנָהּ אָז יְדִיד תָּפִיק / אֲזַי תִּכּוֹן וְתִלָּוֶה

תְּשִׂיחֶךָ תְּנִיחֶךָ / וְדֹדֵי רַעֲיָה תִרְוֶה

וְאִם תַּעְנֶה בְמִלֶּיךָ / ״אֲבָל אֵין לַיְדִיד שֹׁוֶה״

תְּגֹרַשׁ אָז וְתִמָּאֵס / וְנִשְׁאַר לִבְּךָ דָוֶה

וְיוּקַד בָּךְ יְקוֹד־אַהַב / עֲדֵי תֵחַם וְתִדָּוֶה

וְאִם תֵּיטִב דְּבָרֶיךָ / בְּרַעְיָה יֵשׁ לְךָ מִקְוֶה.


וַיָּבֹא וְהִנֵּה נְעָרוֹת מוֹשְׁכוֹת הַדְרָתָן / וְלֹא נוֹדַע הֲתַרְשִׁישׁ גְּוִיָּתָן / וְאִם סַפִּיר גִּזְרָתָן / וְהִנֵּה נַעֲרָה אַחֶרֶת בָּאָה מִגִּנַּת הַבִּיתָן / וַיֹּאמֶר לָהֶן: ״אֵי הַנַּעֲרָה אֲשֶׁר תּוֹרֶה בְּחִצִּים / אֲשֶׁר בִּי חִצֶּיהָ יֵלְכוּ שָׁבִים וְרָצִים״. / וַתַּעַן הַנַעֲרָה אֲשֶׁר מִגִּנַּת הַבִּיתָן בָּאָה: / ״הִנֵּה הַחִצִּים מִמְּךָ וְהָלְאָה.״ / וַיַּעֲנוּ לוֹ לִבּוֹ וְנַפְשׁוֹ מֵחִצִּים מְחוּצִים: / ״רָץ מְצָא נָא אֶת הַחִצִּים״. / וַיִּגְעַר בָּם לֵאמֹר: ״אֵיךְ תִּשְׁלְחוּנִי אֶל הוֹרְגַי / וּתְשִׂימוּנִי בְּיַד מוֹגָי!״ / וַיֹּאמֶר כַּאֲשֶׁר רָאָה כִּי יְבַקְשׁוּ הֲמִיתוֹ: / ״אוֹיְבֵי אִישׁ אַנְשֵׁי בֵיתוֹ״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

נַפְשִׁי וְלִבִּי יְקוּמוּן / כַּיּוֹם מְמִיתָם לְאוֹרֵב

מוֹתֵת בְּחֵץ עֵין צְבִיָּה / מַה־טּוֹב וּמָתוֹק וְעָרֵב

תּוֹרֶה בְחִצִּים שְׁנוּנִים / בָּאִים עֲמֹק לֵב וְקֶרֶב

הֵמָּה לְשָׁלוֹם וְאַךְ הִיא / תִּלָּחֵם בְּכָל יוֹם וְתָרֵב

כִּי אַהֲבָתָהּ בְּכָל־עֵת / תִּגְדַּל וְתִפְרֶה וְתֶרֶב

תִּכּוֹן בְּעֵצָה לְהָרְגָם / קֶשֶׁת וְכִידוֹן וְחֶרֶב

תַּאְוַת לְבָבוֹת סְבִיבוֹת / בָּבוֹת שְׁחוֹרוֹת כְּעוֹרֵב

לַרְחִיק וְלָנוּד רְצוֹנָהּ / וּרְצוֹן שְׁנֵיהֶם לְקָרֵב

תִּשְׂחַק וְהֵמָּה יְחִילוּן / בֹּקֶר וְצֹהַר וְעֶרֶב

עַל־לֵב יְצוּרִים יְהִי טַל / עַל־חוֹשְׁקִים חֹם וְחֹרֶב.




וַיֹּאמֶר אֱלֵיהֶן: ״אָן כִּימָה אֲחוֹתִי / שְׁאֵלָתִי וּבַקָּשָׁתִי״ / וַתֹּאמַרְנָה לוֹ: ״עַד מָתַי תַּהֲלֹךְ נַפְשְׁךָ לְמוֹקֵשׁ / גְּדֹלוֹת אַל תְּבַקֵּשׁ / וְאֵיךְ אַתָּה בוֹטֵחַ עַל־תֹּהוּ וְעַל בְּלִימָה / הַתְּקַשֵּׁר מַעֲדַנּוֹת כִּימָה״? / וַיַּעַן וַיֹּאמַר: ״עַד מָתַי תְּהַתְּלוּ בִי / לְהַכְאִיב אֶת לִבִּי / עַד אֲשֶׁר יִשָּׁבֵר וְיֵהָרֵס הֶרֶס / וְלֹא יִמָּצֵא בְּמִכְתָּתוֹ חֶרֶשׂ״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

לְבַב הַחוֹשְׁקִים נִשְׁבָּר / מְכֻתָּת כִּכְתוֹת חֶרֶס

וְיִשָּׁבֵר כְּמוֹ נֵבֶל / בְּנוּד דּוֹד נֶהֱרַס הֶרֶס

וְיִזָּרֶה עֲדֵי כִּי כָל / יְצוּרָיו צֵל בְּתוֹךְ עֶרֶס

וְלִמְצֹא חֵן בְּעֵין עָפְרָה / יְפַלֵּל צוּר בְּכַף פּוֹרֵס

וְלֹא יִמְצָא לְבַד מֵרִיב / לְחִי מַכָּה וְשֵׁן יַגְרֵס

יְרִיבוּהוּ עֲלֵי חִשְׁקוֹ / בְּרַעְיָה תַּעֲשֹׁק חֶרֶס

וְרָב חוֹשֵׁק בְּרֹב גֶּרֶב / יְהִי מֻכֶּה וּבַחֶרֶס

וְיִהְיוּ שׁוֹאֲלִים לֶחֶם / עֲלֵי דֶלֶת וְאֵין פּוֹרֵס

וְיִכָּנְעוּ מְרִיבֵי דוֹד / וּמַקְלֶה אוֹהֲבִים קוֹרֵס.


וַתֹּאמַרְנָה לוֹ: אַל יִחַר בְּעֵינֶיךָ / וְאַל־תַּעַל חֲמָתֶךָ וַחֲרוֹנֶךָ / הֲלֹא אֲנַחְנוּ בָאוֹת לְנַסּוֹתֶךָ / הֲיֶשׁ [אֶת] לְבָבְךָ יָשָׁר כַּאֲשֶׁר לְבָבָהּ עִם לְבָבֶךָ״? / וַיּוֹדִיעוּהוּ כִּי בָתֵי הָאַרְמוֹן וְהַבִּירוֹת / חֲצֵרוֹת קְטֻרוֹת / וְהֶרְאוּהוּ מְבֹא הַבִּירָה / וּלְפָנָיו פָּרֹכֶת כְּתוּבָה בָהּ זֹאת הַשִּׁירָה:


טֶרֶם יְדִידִי בוֹאֲךָ לַחֶדֶר / שָׂא עֵינְךָ וּקְרָא כְתָב פָּרֹכֶת

בֹּא בַעֲנָוָה אֶל דְּבִירֶיהָ אֲזַי / יֻתַּן בְּבֵית עָפְרָה לְךָ מַהְלֶכֶת

וּבְחַר קְצַר אֹמֶר עֲרֵב עִנְיָן וְאָז / נַפְשְׁךָ לְבַב רַעְיָה תְהִי מוֹשֶׁכֶת

תִּשְׂמַח בְּעָפְרָה אָז וְתִנְעַם נַפְשְׁךָ / וּבְכָל־אֲשֶׁר תִּתְאַב תְּהִי מוֹלֶכֶת

תֵּשֵׁב בְּמַעְדָנִים וּמֵרִיחַ בְּמֹר / זִכְרָהּ בְּפֶה נוֹשֵׁק וְיָד מוֹעֶכֶת.


וַיָּבֹא וְהִנֵּה שְׁמֹנֶה נְעָרוֹת / בְּאוֹרָן הַשֶּׁמֶשׁ מַחְפִּירוֹת / וּכְזֹהַר הָרָקִיעַ מַזְהִירוֹת. / וַיֹּאמַר: ״הִשְׁבַּעְתִּי אֶתְכֶם הַמְּאוֹרוֹת אֲשֶׁר לִקְרָאתִי יְצָאתֶן / אִם אֶת שֶׁאָהֲבָה נַפְשִׁי רְאִיתֶן״. / וַתַּעֲנֶינָה לֵאמֹר: / ״הֵן מִשְׁאָלְךָ נִפְלָא מִמֶּנּוּ / עָמֹק עָמֹק מִי יִמְצָאֶנּוּ / אֵיךְ כִּימָה תְבַקֵּשׁ בָּאָרֶץ הֲלֹא תִּרְגַּז וְלֹא תֶחֱרַד / מִי עָלָה שָׁמַיִם וְיֵרַד״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

כִּימַת אֲדָמָה תּוֹר בְּעֵינָהּ חֵץ / כִּי לֹא לְכִימַת רוֹם יְרוֹת חִצִּים

כִּימַת אֲדָמָה הִיא תְעַנֶּה דוֹד / תִּגַּשׁ בְּעֵינַיִם מְאֹד אָצִים

כִּימַת אֲדָמָה תַּרְאֶה לֶחְיָהּ / כִּשְׁנֵי אֲנָשִׁים לוֹחֲצִים נִצִּים

כִּימַת אֲדָמָה תַּךְ בְּעֵינַיִם / לִשְׁפֹּךְ דְּמֵי נָקִי וְתָם רוֹצִים

כִּימַת אֲדָמָה אֵשׁ נְדוֹד תַּדְלִיק / בַּלֵּב וְתוּקַד בּוֹ בְּלִי עֵצִים

אֵין קוֹדְחֵי אֵשׁ הַנְּדוֹד זוּלַת / כִּימָה הֲלֹא תִדְעַךְ כְּאֵשׁ קוֹצִים

כִּימַת אֲדָמָה חוֹמְסָה חוֹשְׁקִים / כִּי לֹא כְסִיל כִּימָה וְעָשׁ חוֹמְצִים.


וַיִּצְעַק וַיֹּאמַר: ״רַחֲמוּנִי כָּל מְרִיבָי / וְנַחֲמוּנִי כָּל אוֹהֲבָי / וְחִמְלוּ עַל לֵב נִגְזָל אֲשֶׁר עָשׁוּק / וְהַצִּילוּ גָזוּל מִיַּד עָשׁוּק״. / וַתֹּאמַרְנָה לוֹ: ״הִתְבַּשֵּׂר וְגִיל וּשְׂמַח / כִּי יְשׁוּעָתְךָ עַתָּה תִצְמַח / הֲלֹא כִימָה הַגְּבִירָה הוֹלֶכֶת / אֶל־מִבֵּית לַפָּרֹכֶת״. / וַיְמַהֵר לָבֹא וַתֹּאמַרְנָה לוֹ: שָׂא נָא עֵינֶיךָ וּקְרָא זֹאת הַשִּׁירָה / וְאַחַר תָּבוֹא אֶל הַגְּבִירָה״.


יְדִיד בֹּא תֶּחֱזֶה עָפְרָה / וְאַךְ אֶל סוֹד הֱיֵה נֶאְמָן

שְׂפָתֶיהָ כְּחוּט שָׁנִי / רֹק פִּיהָ כְּצוּף וּמָן

בְּשָׂרָהּ עָב וְשׁוֹקֶיהָ / חֲלָאִים מַעֲשֵׂה אָמָּן

וְלוּ שָׁת אִישׁ יְמִינוֹ בֵּין / עֲבוֹת בִּטְנָהּ יְהִי נִטְמָן

זְרֹעֶךָ לְחַנְּקָהּ צַו / שְׂפָתֶיךָ לְנָשְׁקָהּ מַן

וְכָל אִישׁ יַחֲשֹׁק לוֹ בָּהּ / כְּמוֹ אִישׁ יַעֲבֹד חַמָּן

וּמִי יוּכַל סְפֹר הוֹדָהּ / הֲיִסָּפֵר וְאִם יִמָּן

אֲשֶׁר תַּצִּיל מְאוֹר שֶׁמֶשׁ / וּמְאֹר זָהֳרוֹ אַלְמָן

וְלָהּ עַיִן לְבַב חוֹשְׁקִים / כְּשׁוֹדֵד לָהּ כְּשֹׁד שַׁלְמָן

הֲרוּג חִצָּהּ בְּלִי יֶחִי / וְלוּ רוֹפְאוֹ יְהִי הֵימַָן

וְלָהּ לֵב עַז כְּצוּר חָזָק / וּמִי יִתֵּן יְהִי רַחְמָן

וְכֵן כָּל־דּוֹד וְכָל־חוֹשֵׁק / כְּבַד־אֹזֶן וְלֵב הָשְׁמָן

וְאִם תּוֹעִידְךָ מוֹעֵד / יְמוֹתֶיךָ תְכַל בַּזְּמָן

וְדַע כִּי זֹאת גְּבוּלַת־חֵן / וְשׂוֹרַת־הוֹד בְּשִׁיר נִסְמָן.


וַיָּבֹא וְהִנֵּה עֲדַת נְעָרוֹת כִּצְבָאוֹת / מֵאֵת הַגְּבִירָה יוֹצְאוֹת / וְלִקְרַאת סַהַר בָּאוֹת / מַחְפִּירוֹת הַשֶּׁמֶשׁ בְּזֹהַר פְּנֵיהֶם וּמְבִישׁוֹת הַחַמָּה וְנוֹצְצִים עַל הַלְּחָיַיִם / פֶּרַח וְנֵרוֹת וּמֶלְקָחַיִם / וְכָךְ הָיוּ קוֹרְאוֹת וּמְדַבְּרוֹת / וּבְתַחֲנוּנֵי אֲמָרִים אֹמְרוֹת: / ״לֵךְ אָהוּב כִּימָה / וְעִמְּךָ חֲשׁוּק חַמָּה.״ / וַיַּעַן וַיֹּאמַר: ״אִם יֶשְׁכֶן אַיֵּה כִּימָה גְבִרְתִּי, הֲלֹא רְאִיתֶן כִּי רוּחִי בְּחִשְׁקָהּ חֻבָּל / וְכָשַׁל כֹּחַ הַסַּבָּל״. / כִּרְאוֹתָן כִּי רוּחוֹ קָצְרָה קְצֹר / וַתָּבֹא נַפְשׁוֹ בַמָּצוֹר / וַתֹּאמַרְנָה לוֹ ״קְרַב אֶל הַפָּרֹכֶת / אֲשֶׁר אַחֲרֶיהָ כִּימָה הוֹלֶכֶת / וּקְרָא הַמִּכְתָּב הַזֶּה״. / וַיִּקְרָא וְהִנֵּה כָתוּב בִּמְגִלַּת סֵפֶר: / ״קְרָא נָא אֶשְׁכֹּל הַכֹּפֶר״. / וַיַּרְא בִּכְתָבָהּ / כִּי הִגִּידָה לוֹ אֶת כָּל לִבָּהּ:


אֲנִי סֵתֶר עֲלֵי רַעְיָה / כְּהַסְתִּיר עָב מְאוֹר חַמָּה

בְּבוֹאָךְ הַעֲלֵנִי דוֹד / וְתִמְצָא אַחֲרַי כִּימָה

וְלֹא כִימָה אֲחוֹת עַיִשׁ / אֲשֶׁר עִם כּוֹכְבֵי רוֹמָה

אֲבָל כִּימָה יְפַת־עַיִן / יְפַת מַרְאֶה יְפַת קוֹמָה

בְּכִימַת הָאֲדָמָה לֹא / יְדִידַי תֵלְכוּ רוֹמָה.


וַיָּבֹא אֶל הָאַרְמוֹן אֲשֶׁר שָׁם כִּימָה הַגְּבִירָה / וְאֶל סַהַר הָיְתָה אוֹרָה / וְהִנֵּה לִקְרָאתוֹ נְעָרוֹת קְהִלּוֹת / לָחוּל בַּמְּחוֹלוֹת / וַתַּעֲרֹךְ כָּל הַקְּהִלָּה / וַתְּהַלֵּלְנָה אוֹתוֹ בְּשִׁירֵי תְהִלָּה / וּבִידֵיהֶן כְּלֵי שִׁיר מִכָּל־מִינִים / וְכֵלִים מִכֵּלִים שׁוֹנִים / וּבְהַגִּיעַ כָּל הַקְּהִלָּה אֵלָיו / וַתִּשְׁתַּחוּ לְרַגְלָיו / וּלְיָפְיוֹ לִבָּם יִסְגֹּד / וַיֵּלֵךְ כְּמֶלֶךְ בַּגְּדוּד / וְכֹה אָמַר בְּלֶכְתּוֹ / בְזָכְרוֹ כִימָה גְבִרְתּוֹ: / ״מִי יִתֵּן וְאֶרְאֶה רַעְיָתִי כִימָה / וְיַעֲבֹר עָלַי מָה / הֲלֹא בַעֲדָהּ נִתְּכָה כַמַּיִם שַׁאֲגָתִי. / מִי יִתֶּן תָבוֹא שֶׁאֱלָתִי / כִּי נִשְׁאַרְתִּי בְּאֵין לֵב וְאֵין דַּעַת״. / וְכִימָה שׁוֹמַעַת / וַתַּעֲנֵהוּ בְמַעֲנֶה כָּל־כְּאֵב יֶחְבָּשׁ / מַה־מָּתוֹק מִדְּבַשׁ / וּדְבָרִים עֲרוּכִים בְּטוּב טַעַם / צוּף דְּבַשׁ אִמְרֵי נֹעַם: / ״הֲלֹא לִבִּי נָד כְּפִי נוּדְךָ כִּפְלָיִם / כַּאֲשֶׁר יָנוּד הַקָּנֶה בַּמָּיִם / וְיָצָא לִבִּי מֵאִתִּי / כִּי בְּאֵשׁ נוּדְךָ שֹׂרָף / וָאֹמַר:״אַךְ טָרֹף טֹרַף / וַיֵּלֵךְ אֶל הַמִּדְבָּר כְּעָרוֹד / וַיֵּשֵׁב בְּאֶרֶץ נוֹד / וַאֲנִי חָשַׁבְתִּי כִּי לִבִּי אַחֲרַי יַעֲמֹד / וְהִנֵּה הִסְכַּלְתִּי וָאֶשְׁגֶּה הַרְבֵּה מְאֹד״.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

וְחָשַׁבְתִּי לְהִתְנוֹדֵד בְּנוּד דּוֹד / וְהוֹחַלְתִּי וְהוּפַר מַחֲשָׁבִי

וּבָטַחְתִּי בְעֹז לִבִּי וְתָקְפִּי / וְהוּא נִסְחַף וְלֹא עָמַד לְבָבִי

וְאָמַרְתִּי כְּבָר צַדְתִּיו בְּרִשְׁתִּי / וְהוּא עוֹדֶה וְהוּא טוֹרֵף כְּלָבִיא

וְגֵרַשְׁתִּיו וְנִגְרַשׁ גַּם לְבָבִי / וְהוֹלִיכוֹ כְּאִישׁ הוֹלֵךְ בְּשֶׁבִי

בְּהִתְנוֹדֵד צְבִי חָשַׁךְ מְאוֹרִי / וּמִזְרָחִי אֲזַי שָׁב מַעֲרָבִי

וְנִחַמְתִּי וְנַחוּמַי כְּאַיִן / לְאַחַר נוֹד אֲבָל נִכְפַּל כְּאֵבִי

וְהִדְלִיק אֵשׁ נְדוֹדוֹ תוֹךְ כְבֵדִי / וְהִגְדִיל הַמְּדוּרָה עַל שְׁבִיבִי

אֲחַזְתִּיךְ יְדִיד לֹא אַרְפְּךָ עוֹד / וְעוֹד לֹא אֶשְׁמְעָה בָּךְ קוֹל מְרִיבִי

קְרַב דּוֹדִי עֲדֵי מָתַי קְרָבוֹת / וְכַבֵּה אֵשׁ נְדוֹדִים בָּעֲרָה בִי

נְדוֹדִים תַּחֲלִיף הַיּוֹם בְּדֹדִים / וּמֵי קִרְבָתְךָ תֵּן עַל שְׁרָבִי

אֱרֵה מוֹרִי יְדִידִי עִם בְּשָׂמִי / אֱכֹל דִּבְשִׁי וְיַעְרִי עִם חֲלָבִי.


וַתִּקְרַָב אֵלָיו / וַתִּשְׁתַּחוּ עַל שׁוּלָיו / וַתְּנַשְׁקֵהוּ עַל־יָדָהּ כְּמוֹ מִשְׁפַּט יְדִידִים נֶאֱמָנִים / וְלֹא נְשָׁקַתְהוּ בְּפִיו מִפְּנֵי הַנִּרְגָּנִים / רַק עֵינֶיהָ הֵם מְרַקְּדִים וּמְשַׂחֲקִים / כְּאִילוּ הֵם מְחַבְּקִים / וַיַּרְא סַהַר כִּי לֹא נְשָׁקַתְהוּ וַיִּבֶז בְּעֵינָיו / וַיִּחַר לוֹ מְאֹד וַיִּפְלוּ פָּנָיו.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

נְשַׁקְתִּיו עַל יְדֵי מַלְאָךְ יְדִידִים / וְעָנָנִי: ״הֲלֹא פִי צִיר אֱמוּנִים

הֲכָמוֹנִי בְיַד מַלְאָךְ לְנַשֵּׁק / הֲזֹאת תּוֹרַת יְדִידִים הָעֲדִינִים״.


וַתֹּאמֶר לוֹ: ״אַל יֵרַע בְּעֵינֶיךָ / לָמָּה חָרָה לָךְ וְלָמָּה נָפְלוּ פָנֶיךָ / הֲלִנְשֹׁק וּלְחַבֵּק הוּא חִשְׁקֵנוּ / לֹא יֵעָשֶׂה כֵן בִּמְקוֹמֵנוּ / אֲבָל חִשְׁקֵנוּ וְחֵשֶׁק הַנְּדִיבוֹת / לְצָרֵף וּלְלַבֵּן הַלְּבָבוֹת / לֹא כְמִשְׁפַּט בְּנֵי הַתַּעֲרֻבוֹת / כִּי יֵשְׁבוּ יַחַד לֹא לְנַשֵּׁק לֹא לְחַבֵּק / אֲבָל לֵב [זְה] בְּלֵב זֶה דוֹבֵק / וְכֵן סִבַּת אַהֲבַת בְּנֵי הַיְקָרִים / לָקַחַת מוּסַר הַשְׂכֵּל, צֶדֶק וּמִשְׁפָּט וּמֵשָׁרִים / וַיְהִי יָדָם אֱמוּנָה / וְאִם מְסַפְּרִים דִּבְרֵי חִשְׁקָם וְאַהֲבָתָם / תִּגְבַּר צִדְקָתָם / וְאִם הַגְּבוּרוֹת יְסַפְּרוּ אוֹדוֹת חִשְׁקָם וְהַתְּלָאוֹת וְהַקּוֹרוֹת. / אֵלֶּה הַחֻקִּים וְהַתּוֹרוֹת / וְזֶה כָל מְתוּקֵי חִשְׁקָם / צוּפָם וְדִבְשָׁם / וְהָיוּ לְבָרוֹת לָמוֹ וְיָשִׁיבוּ אֶת נַפְשָׁם.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

אֲהַבְתִּיךְ אֲבָל לֹא עַל דְּבַר מוּם / וְכָמוֹנִי יְדִידוּתָם אֲהוּבָה

וְחִשְׁקֵנוּ נְקִי־כַף בַּר פְּעָלִים / וְרוּחַ אַהֲבָתֵנוּ נְדִיבָה

וְסִפְרֵיהֶם מְלֵאִים תֹּם וְיֹשֶׁר / וְיִרְאָה וַעֲנָוָה בָּם כְּתוּבָה

וְכָל־חֵשֶׁק בְּנֵי נָבָל נְבָלָה / וְסוֹף כָּל אַהֲבָתָם מַעֲצֵבָה

וְתִתְמֹגֵג כְּרֶגַע אַהֲבָתָם / וְתִתְהַפֵּךְ יְדִידוּתָם לְאֵיבָה

וְאִם תִּשְׁמַע דְּבַר מוּם עַל נְדִיבִים / חֲשֹׁב טוֹב וֶאֱמֹר גַּם זוּ לְטוֹבָה.


וַיֹּאמֶר סַהַר בְּשָׁמְעוֹ דְבַר לַחְשָׁהּ / צָדְקָה נַפְשָׁהּ / וַיֹּאמַר: ״טוֹב הַדָּבָר אֲשֶׁר דִּבַּרְתְּ / בָּרוּךְ טַעְמֵךְ וּבְרוּכָה אַתְּ״ / וַיָּבוֹאוּ אֶל הָאַרְמוֹן וְדַלְתּוֹתָיו בְּרוֹשִׁים / מְמֻלָּאִים בַּתַּרְשִׁישִׁים / מַעֲשֵׂה חָרָשִׁים / וְקַרְקָעוֹ וְקִירוֹתָיו וְגַגּוֹ זְכוּכִית כֻּלָּם / וְהַמַּיִם לָהֶם חֹמָה מִימִינָם וּמִשְּׂמֹאלָם / וַתָּבֹא אֵלָיו בַּתְּחִלָּה / [וַיִּפָּלֵא] מִמֶּנּוּ הַמַּעֲשֶׂה הַנִּפְלָא / וְלִשְׂחוֹת פָּשַׁט הַבְּגָדִים וְהַנַּעֲלַיִם / כִּי יַחְשְׁבֵהוּ פְלָגִים יִבְלֵי מַיִם / וְכִי יָרֵא הַמַּיִם הַנִּגָּרִים תַּחַת הַקַּרְקַע / חָשַׁב כִּי יִטְבַּע בַּמַּיִם וְיִשְׁקַע / וַיִּדְלַח וַיִּרְקַע בְּרַגְלָיו / עַד אֲשֶׁר מָעֲדוּ קַרְסֻלָּיו. / וַיְהִי כִּרְאוֹת סַהַר הַבַּיִת וְצוּרָתוֹ / וְהַמַּיִם אֲשֶׁר מִתַּחַת הַזְּכוּכִית וּתְכוּנָתוֹ / וַיָּבֹאוּ בְלִבּוֹ מוֹרָא וְחִתִּית, / בִּרְאוֹתוֹ גֻלּוֹת עִלִּית גֻּלּוֹת תַּחְתִּית / וַיֹּאמַר: ״הֲלֹא הָרוּחַ אֶל זֶה הַיָּם שֵׁנִית תִּדְּפֶךָ / אִם לֹא יָם שָׂנֵאתָ וְיָם יִרְדֳפֶךָ.״ / וַתּוֹדִיעֵהוּ כִימָה דְבַר הַבַּיִת וְהַזְּכוּכִית וְצוּרוֹתָיו / וְגֻלּוֹת הַמַּיִם אֲשֶׁר תַּחְתָּיו / וַיֵשְׁבוּ בוֹ בְּשִׂמְחָה וַהֲנָחָה / וַיָכִינוּ אֶת הַמִּנְחָה / וַיֹּאכְלוּ וַיִּשְׁתּוּ / וְלִבּוֹתָם מִתַּאֲוָתָם חַתּוּ. / וַיָּלִינוּ שְׁנֵיהֶם מְדַבְּרִים מִכָּל־טוּב מִלָּה / וְלֹא קָרַב זֶה אֶל זֶה כָּל הַלַּיְלָה / וְנֶפֶשׁ אִישׁ לְאָחִיו כָּלְתָה וְנִכְסְפָה / נְשֻׂאוֹתֵיהֶם עֲמוּסוֹת מַשָּׂא לַעֲיֵפָה / וְאִישׁ מֵרֵעֵהוּ אֵין לוֹ מִכָּל חֶפְצוֹ וְרַעְיוֹנָיו / כִּי רְאוּת עֵינָיו. / וַיִּהְיוּ כֵן בְּרוּרִים מִכָּל־מוּם וְתָהֳלָה / וַיִּתְחָרְשׁוּ כָּל־הַלַּיְלָה / בִּנְגִינוֹת שִׁיר אַהַב וּתְהִלָּה / עַד כִּי בַעֲרָה בָּם אֵשׁ תְּשׁוּקוֹתֵיהֶם נָחָר / עַד עֲלוֹת הַשַּׁחַר. / וַתִּגְדַּל אַהֲבָתָם עַד אֵין חֵקֶר / וַתִּתְבּוֹנֵן אֵלָיו בַּבֹּקֶר,


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

הֲסַהַר יַחֲנֶה מָרוֹם / הֲשֶׁמֶשׁ תִּהֲלַךְ אָרְצָה

אֲשֶׁר זֶה… עָמַד / וְזֹאת לָבֹא בְּלִי אָצָה

מְבַקֵּשׁ יוֹם וְלֵיל יַחַד / יְגַל תֹּאַר צְבִי יִמְצָא

בְּעֵינָיו יַהֲרֹג אִישׁ תָּם / וְהֵא דָמִי כְּבָר נִמְצָה.


וַתּוֹסֶף פְּנוֹת אֶל הַדְרָתוֹ / וְאֶל יָפְיוֹ וְאֶל יִפְעָתוֹ / וְזֹהַר לֹא יְתֹאַר כְּבָרָק לְחָיָיו וְאֶל עֵינֶיהָ תִתְמַרְמַר.


וַתּוֹסֶף מְשָׁלָהּ וַתֹּאמַר:

מְאוֹר שֶׁמֶשׁ וְאוֹר סַהַר / יְקַנְּאוּ בַצְּבִי סַהַר

שְׁנֵיהֶם יַעְשְׁקוּ אוֹרוֹ / וְיוֹסִיף לֶחְיוֹ זֹהַר

יְרִיבוּהוּ וְאַךְ כָּל־יוֹם / לְחָיָיו יָסְפוּ סֹהַר

וְכָרְעוּ לוֹ אֲזַי אוֹרִים / לְזִיווֹ תָּאֳרָם צָהַר

וְגַם הוֹדוֹ עֲדֵי נָגְהָם / עֲלֵי אוֹר לֶחְיוֹ יִגְהָר

אֲשֶׁר סַהַר דְּמוּת עֹפֶר / וְאֵי אֹדֶם וְאֵי טֹהַר

יְדַמֶּה דוֹד אֱלֵי שֶׁמֶשׁ / כְּסִיל בַּעַר בְּלֵב נִמְהָר

וְזִכְרוֹ מָר־דְרוֹר וּשְׁמוֹ / כְּשֶׁמֶן טוֹב וְטוּב יִצְהָר

וְנִדְבוֹת לֵב אֲמָהוֹת לוֹ / וְיִמְהָרָם הַצְּבִי מֹהַר

וְלוּ שׁוֹאֵל יְאַו הוֹנוֹ / לְתִתּוֹ לוֹ אֲזַי מִהַר

וְלוּ שָׁאַל אֱנוֹשׁ נֶפֶשׁ / נְתָנָהּ אַךְ הֱיֵה נִזְהָר

לְזֹאת אִם נוֹד יְנִידֶנּוּ / וְאֶתְשׁוֹטֵט בְּרֹאשׁ כָּל־הַר

אֲצַפְצֵף לִי עֲלֵי נוּדוֹ / כְּסוּס עָגוּר, כְּסוּס אֶדְהַר

וְאֶהְגֶּה אֶהְמֶה כָל־יוֹם / כְּיוֹנִים עַל־בְּרוֹשׁ תִּדְהָר

אֲצַו עָבִים נְשׂוֹא שָׁלוֹם / לְדוֹד חֵן רַב כְּמֵי נָהָר.


וַיִּשְׁמַע סַהַר וַיֵּאָנַח וַיֹּאמַר: ״אֲהָהּ כִּי לִבִּי יֵקַד יְקֹד / וְיִצְמָא עַד מְאֹד / רְאִי רַעְיָתִי כִּי אֵשׁ נְדוֹדֵךְ בְּחֵיקוֹ יַחְתֶּה / הַטִּי־נָא כַדֵּךְ וְאֶשְׁתֶּה / טֶרֶם מֵחַיִּים נִגְזַרְתִּי / פֶּן־תִּדְבָּקַנִי הָרָעָה וָמַתִּי״. / וַתַּעַן כִּימָה וַתֹּאמַר: ״אַל־נָא זֶה הַדֶּרֶךְ תֵּט / וְיִהְיֶה בְּךָ חַטְא״.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

עֲווֹן דּוֹד רַב מְאֹד מִנְּשׂוֹא / וְאֵין כֹּפֶר אֱלֵי חֶטְאוֹ

כְּבֵדִי סָהֳרוֹ אֶתֵּן / וְלִבִּי יִהְיֶה כִלְאוֹ

וְלִי לֵבָב כְּבַד־אֹזֶן / בְּלִי יִשְׁמַע לְקוֹל קוֹרְאוֹ.


וַיְהִי סַהַר עִם־לִבּוֹ בְמִלְחָמָה / בְּאַהֲבָתוֹ אֶת־כִּימָה.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

חֵשֶׁק וּמִלְחָמָה שְׁנֵיהֶם הוֹרְגִים / הוֹי עַל בְּנֵי חַיִל וְהוֹי עַל חוֹשְׁקִים

אֵל בָּרְמָחִים וַחֲרָבוֹת נִלְחֲמוּ / גַּם אֵל בְּחִצִּים לוֹחֲצֵי לֵב דּוֹחֲקִים

רֹמַח הֲכִי נִשְׁבָּר וְחֶרֶב תַּחֲטִיא / אַךְ בַּעֲלֵי חִצִּים לְעוֹלָם דּוֹלְקִים

וּבְאַף וְחֵמָה נִלְחֲמוּ [הֵמָּה] אֲבָל / עֵינֵי צְבִי יוֹרוּ וְהֵם כִּמְשַׂחֲקִים.


וַתִּשְׁמַע כִּימָה טוּב שִׁירָתוֹ / וַתּוֹסֶף מְאֹד בְּלִבָּהּ אַהֲבָתוֹ / עַל כִּי זֶה לְבִזָּה נָתַן נַפְשָׁהּ הַצְּבִי / לְבָבָהּ נָהַג בַּשְּׁבִי.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

לְמִי אֶלּוֹן עֲלֵי חִשְׁקִי / וְעֵינִי הָיְתָה סִבָּה

לְבָבִי בַשְּׁבִי הָלַךְ / בְּיַד שַׂר אַהֲבָה נִשְׁבָּה

וְהוּא לֹא שָׁב עֲדֶן אֵלַי / וְהוּא שַׂר הַיְדִידֹת בָּא

וְאִם אֹמַר לְהִפָּדוֹת / הֲלֹא דוֹד בַּמְּחִיר הִרְבָּה

וּבִקַּשְׁתִּי וְעָנָנִי / בְּתוֹךְ לֵב הַיְדִיד נֶחְבָּא

הֲבַמַּרְאוֹת נְעִים שִׁיר דּוֹד / הֲיֵשׁ נָבִיא בְּשִׁיר נִבָּא.


וּבְכָל־זֹאת כִּימָה לֹא תִּרְאֶנּוּ / וּכְמַסְתֵּר פָּנִים מִמֶּנּוּ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

רַעֲיָה אֵיךְ תְּסַתְּרִי / לֶחֱיֵךְ נָא גְלִי גְלִי

אוֹ חֲבִי נָא לְבַעֲבוּר / זָהֳרֵי רוֹם תְּנַצְּלִי

לֵב יְדִידֵךְ תְּנַתְּחִי / תִּזְבְּחִי גַּם־תְּבַשְּׁלִי

בִּי רְצוֹנֵךְ עֲשִׂי עֲשִׂי / בַּשְּׁלִי אוֹ צְלִי צְלִי

אַהֲבַת אוֹהֲבִים הֲלֹא / מוּל אֲהָבֵךְ כְּמַר דְּלִי.


וַיְהִי כֵן יָמִים רַבִּים בֵּין מְרִיבָה וּתְשׁוּבָה / וְאֵיבָה וְאַהֲבָה / וּבֵין דִּבְרֵי אַהֲבוֹת / וְדִבְרֵי מְרִיבוֹת.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

רַעֲיָה יוֹם לְיוֹם תְּלֻנּוֹת וְרִיבוֹת / יַעֲצֹר הַזְּמָן וְנַחְנוּ מְרִיבִים

הַאֲנַחְנוּ לְבַד מְעֻנִּים כְּאֵלֶּה / אוֹהֲבִים אוֹ הֲזֹאת תְּעוּדַת אֲהוּבִים

יַעֲנֵנִי יְדִיד בְּלִי רִיב כְּשׁוֹכֵן / עִם־שְׁכֵנוֹ וְאֵי מְתוּקֵי אֲהָבִים

הָהּ וְאִם אֵין בְּאַהֲבָה רִיב וּמָדוֹן / אֵין נְעִים שִׁיר אוֹ נְעִימַת אַהֲבִים.


וַיְדַבֵּר כָּזֹאת וְכָזֹאת אוּלַי לִרְצוֹנוֹ יְשִׁיבָהּ / וְלֹא עָנְתָה וְלֹא שָׁתָה עַל־לִבָּהּ / וַיְהִי כִרְאוֹתוֹ מְשׁוּבָתָהּ / וְכִי הִיא מַחֲזֶקֶת בְּתֻמָּתָהּ / וַיָבֹא אֶל הָאַרְמוֹן וַיִּקְרַב אֶל הַדֶּלֶת / וַתֵּצֵא מִכְּבֵדוֹ גֶּחָלֶת / וַתֵּצֵא כִימָה נִבְהֶלֶת וִיחֵפָה / לִרְאוֹת אֶת הַשְּׂרֵפָה. / וַתֹּאמֶר לְנַעֲרוֹתֶיהָ: / ״מָה הָאֵשׁ אֲשֶׁר בַּדֶּלֶת נִשְּׂקָה?״ / וַיְרִיבוּן אִתּוֹ בְּחָזְקָה / וַיַעַן לֵאמֹר: ״הֲלֹא שְׂרֵפַת גַּחֶלֶת כְּבֵדִי / מַה־תְּרִיבוּן עִמָּדִי״! / וַיִּתְחַנֵּן אֵלֶיהָ / וַיֹּאמַר: ״אַחֲלַי לִפְנֵי הַנְּעָרוֹת הָעוֹמְדוֹת אִתִּי / שֶׁתֹּאמַרְנָה בְאָזְנֵי כִימָה גְבִרְתִּי / וְאוּלַי תִתְרַצֶּה נַפְשִׁי אֲשֶׁר בְּנַפְשָׁהּ נִקְשְׁרָה / וְכַבּוּ גַחֶלֶת אֲשֶׁר נִשְׁאָרָה״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וַיְהִי בְּנוֹעַ מִנִּי צְבִיָּה / יוֹם יוֹם וְתִיקַד אֵשׁ בְּקִרְבִּי

בָּתִי לְדַלְתֵּי בֵיתָהּ כְּשׁוֹאֵל / כָּעוֹף יְעוּפוּן רִשְׁפֵּי שְׁבִיבִי

וַיִּשְׂרְפוּ אֶת דַּלְתָּהּ וְלֹא עַל / חֶפְצִי וְזֹאת לֹא עָלְתָה בְלִבִּי

חָרָה לְרַעְיָה עַד־מְאֹד וְתִשְׁכַּח / כִּי אֵשׁ נְדוֹדָהּ אָז בָּעֲרָה בִי / שַׁלֵּם אֲשַׁלֵּם דַּלְתָּהּ בְּכֶפֶל / גַּם הִיא תְכַבֶּה אִשִּׁי וְלַהְבִי.


וַתֵּרֵא כִימָה כִּי בָהּ הִתְאַנֵּף יְדִידָהּ / וַיְפַלַּח חֵץ חֲמָתוֹ כְּבֵדָהּ.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

לְבָבִי זָע וְצִיר אָחַז / וְאִם בּוֹ הַיְדִיד אָנַף

אֲשֶׁר יֶחְרַד וְהִתְעוֹפֵף / הֲיֵשׁ לֵב עָף כְּעוֹף כָּנָף

נְטַע חִשְׁקוֹ בְּיוֹם פָּרַח / בְּנִטְעוֹ הֶעֱלָה עָנָף

עֲמֹד לִבִּי הֲלֹא לֵב דּוֹד / בְּךָ הוּרַם בְּךָ הוּנַף

וְגַם לִבּוֹ יְצוּעֶךָ / אֲשֶׁר [בּוֹ] מֹר אֲהָלִים נָף

וְלֵב רַעְיָה אֲשֶׁר יִדְבַּק / בְּזוּלַת אַהֲבָה טֻנָּף

וְאִם יִדְבַּק בְּזוּלָתוֹ / וְשָׁב אֵלָיו הֲלֹא יֶחְנָף

מְעִיל חִשְׁקוֹ יְעָטַנִי / צְנִיף דֹּדָיו הֲלֹא צָנַף.


וַיְּסַפֵּר אֵלֶיהָ סַהַר תּוֹרַת הַיְדִידוֹת / וְכָל תְּלָאוֹת הַפְּרִידוֹת / וַיֹּאמֶר לָהּ: ״לְכִי שִׁמְעִי הָאוֹהֲבִים הָעֲמֵלִים / וְתַחֲלוּאֵי הַחוֹשְׁקִים הָאֲמֵלָלִים / וְצָרוֹת לִבּוֹתָם הַנֶּהֱלָמִים / אֲשֶׁר יִקְרָא אוֹתָם בְּאַחֲרִית הַיָּמִים״.



וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

חוֹשֵׁק עֲשׁוּק נֶפֶשׁ רְצוּץ / לֵבָב יְהַלֵּךְ צַו לְצָו

יִבְטַח בְּקִרְבַת רַעֲיָה / אַךְ תִּמְשְׁכֶנּוּ קַו לְקָו

כִּי אֵין בְּפִי רַעֲיָה אֱמֶת / לֹא תַעֲמִיד רַגְלָהּ בְּשָׁוְא

אֵיךְ בַּאֲהָבִים יַאֲמִין / חוֹשֵׁק דְּבַר כָּזָב וְשָׁוְא

כִּי אִם יְעָדַתְהוּ, עֲדֵי / שֵׂיבַת כְּנַף עוֹרֵב יְקַו

לֵילָיו אֱלֵי עוֹרְבִים וְכָל / יָמָיו בְּשִׁיבָתָם יְאַו

יֹאמַר יְשׁוּבוּן עוֹרְבִים / אוּלַי צְמָאוֹ אָז יְרַו

אֵין בִּין וְאֵין דֵּעָה עֲדֵי / בֵּין טוֹב וּבֵין הָרָע יְשַׁו

יִקְרַב מְדוּרוֹת אֵשׁ יְמֵי / קַיִץ וְחֹרֶף גַּם סְתָו

נִבְהָל בְּלִי לֵבָב וְאִם / יַחְשֹׁב לְתַקֵּן אָז יְעַו

יִבְעַר וְיִכְסַל עַד בְּרִיר / פִּיו מִדְבְּרֵי אַהַב יְתָו

אִם יַאֲרִיךְ [יָמִים] כְּרוֹ / קִבְרוֹ וְאֶל בֵּיתוֹ יְצַו

תַּכִּיר בְּנִי חוֹשֵׁק כְּאִישׁ / הִתְוָה עֲלֵי מִצְחוֹ כְתָו

אֵלֶּה קְצוֹת דַּרְכֵי חֲשׁוֹק / קוֹרוֹת אַהֲבִים מִי יְחַו.


עוֹד הֵם מְדַבְּרִים דִּבְרֵי חִשְׁקָם / וּמִתְחַלְחֲלִים בְּעֵת צָרָתָם וּמְצוּקָם / וְהִנֵּה נַעֲרָה בָאָה / נִבְהֲלָה וְנִלְאָה / וַתֹּאמֶר לְכִימָה: ״הָכִינִי בֵיתֵךְ וּנְוָתֵךְ / כִּי הִנֵּה אָבִיךְ בָּא לִרְאוֹתֵךְ / וְהַחְבִּיאִי סַהַר וּדְבַר חִשְׁקֵךְ מֵאָבִיךְ סַתְּרִי וְכַסִּי / וְעַתָּה דְעִי מַה תַּעֲשִׂי״. / וַתִּשְׁמַע כִּימָה וַתָּקָם [וָתֹּאמַר]: ״אַסְתִּיר אֶת סַהַר יְדִידִי וְאַחְבִּיא / כִי כָלָה הִיא מֵעִם אָבִי״.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

יְדִידַי אָן אֲשַׁו דּוֹד אָן / אֲבִיאוֹ מִפְּנֵי רוֹדְפָי

אֲבִיאוֹ תּוֹךְ סְגוֹר לִבְּהֶן / רְשָׁפִים שָׁם וְהֵם שׂוֹרְפָי

וְאָן אַחְבִּיא צְבִי לִפְנֵי / מְקַנְאַי אוֹיְבַי צוֹפָי

וְכָבֵד רָתְחָה כַסִּיד / בְּגֶחָלַי וּבִרְצָפָי

וְאֵין כִּי אִם לְהַסְתִּירוֹ / לְהַחְבִּיאוֹ בְּסַרְעַפָּי

וְלוּ [בָא] בֶאֱשׁוּן עֵינִי / אֲכַסֶּנּוּ בְעַפְעַפָּי

מְקוֹם בֶּכִי בְּגִיל אֶשְׂחַק / בְּשִׂמְחָה אֶסְפְּקָה כַפָּי.


וַיְהִי הַמֶּלֶךְ לָבוֹא וַיִּירְאוּ וְהִנֵּה תַּחְבִּיאֶנּוּ בְמַחֲבוֹא / וַיַךְ הַמֶּלֶךְ סַהַר מַכָּה / וַיֹּאמַר: ״הֲגַם לִכְבֹּשׁ אֶת הַמַּלְכָּה״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וּמַה־יּוֹעִיל בְּעֹז אַרְיֵה / וְהֵן עֹפֶר יְצוּדֶנּוּ

וּמַה־יּוֹעִיל בְּעֹז מֶלֶךְ / וְהֵן עַבְדּוֹ יְגוּדֶנּוּ

וְאִם הֵנִיא אֲנִיאֶנּוּ / וּמִתְנַכֵּל אֲנִידֶנּוּ

וְאִם צֶדֶק אֲקָרְבֶנּוּ / וּמִשִּׁירִי אֲהוֹדֶנּוּ.



וַיָּשָׁב הַמֶּלֶךְ אֶל כִּימָה וַיִּגְעַר בָּהּ / וַיְחָרְפָהּ וַיַעֲצִיבָהּ / וַיֹּאמֶר הַמֶּלֶךְ: ״קְחוּ לִי אֶת סַהַר / וְהוֹלִיכוּהוּ אֶל בֵּית הַסֹּהַר״! / וַיָּשָׁב פָּנָיו אֶל כִּימָה / וְהִנֵּה כִסְּתָה פָּנֶיהָ כְּלִמָּה.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וְהִנֵּנוּ שְׁנֵי יוֹנִים בְּבַיִת / בְּמַעֲדַנִּים וְעֹנֶג הַיְדִידוֹת

מְצָאוּנוּ תְלָאוֹת הַזְּמַנִּים / וְנִפְרַדְנוּ וְהָהּ עַל הַפְּרִידוֹת.


וַתֹּאמֶר כִּימָה: ״יָדַעְתִּי כִּי אַהֲבָתְךָ לִי כַיּוֹם תִּשְׁבֶּךָ / הֵן קַלֹּתִי וּמָה אֲשִׁיבֶךָ״.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

וְהָיִינוּ שְׁנֵי עַנְפֵי הֲדַסִּים / בְּגַן רָוֶה דְּשֵׁנִים רַעֲנָנִים

מְחֻבָּקִים מְחֻשָּׁקִים דְּבֵקִים / וְנִפְרַדְנוּ בְקוֹרוֹת הַזְּמַנִּים.


וַיְהִי כִּשְׁמוֹעַ הַמֶּלֶךְ וַתָּחָס עֵינוֹ עֲלֵיהֶם / בְּשָׁמְעוֹ טוּב מִלֵּיהֶם / וַיֹּאמֶר: ״הֲשִׁיבוּהוּ וְנִדְרְשָׁה בְעַד חִשְׁקָם / וְנַחְקֹר בְּעַד רִשְׁעָם וְצִדְקָם״. / וַתַּעַן כִּימָה וַתֹּאמַר: ״אֲדֹנִי הַמֶּלֶךְ כִּי זֶה הַנַּעַר לֹא אֲמוּשֵׁהוּ וְלֹא הֱמִישַׁנִי / כִּי צֶדֶק לָבַשְׁתִּי וַיִּלְבָּשֵׁנִי / וּמֵאֵין חִשְׁקִי מִבִּגְדֵי יִרְאָתְךָ פְּשָׁטַנִי / סַהַר מְעִיל צְדָקָה יְעָטַנִי / וִיצִיעֵנוּ מִצְוָה / וּמַלְבּוּשֵׁנוּ עֲנָוָה / לְבוּשׁ יִרְאָה לְבוּשֵׁנוּ / וּרְדִידֵנוּ דַּעַת קְדוֹשֵׁינוּ״.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

וְיָעַטְנוּ מְעִיל־חֵן גַּם עֲנָוָה / וְלָבַשְׁנוּ לְבוּשׁ־אַהַב וְיִרְאָה

וְכָרַתְנוּ בְרִית־חֵשֶׁק בְּלִי מוּם / אֲשֶׁר לֹא יַעֲלֶנָּה רִיב וְקִנְאָה

אֲנַחְנוּ נוֹשְׂאֵי עֹל תֹּם וְחֵשֶׁק / וְלֵב נוֹשֵׂא שְׁתֵּי אֵלֶּה הֲנִרְאָה

וְרוּחַ אַהֲבָה תָּעִיר לְהָרַע / אֲבָל רוּחַ חֲכָמָה הִיא מְנִיאָה

וְרוּחַ אַהֲבָתִי נִכְבְּשָׁה בִי / וְחָכְמָה לָבְשָׁה גָאוֹן וְגֵאָה

רְוֵי דֹדִים חֲשָׁבוּנוּ מְקַנְאִים / וְהֵא יָבֵשׁ גְּרוֹנֵנוּ בְצִמְאָה.


וַיֹּאמֶר לָהּ הַמֶּלֶךְ: ״הַגִּידִי נָא לִי מָה הַחֵשֶׁק הַזֶּה וּמַה־סִבָּתוֹ / וּמִי הַחוֹשֵׁק וּמָה אַהֲבָתוֹ״?


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

בְּנֵי אַהַב / יְקוֹד לַהַב / וְאֵם הַב הַב / בְּקִרְבָּם בְּלִי יִכְבֶּה / אֲבָל יִרְבֶּה / כְּדֵי־אַרְבֶּה / שְׁבִיבָם וְיַשְׁפִּילוּ / וְיָחִילוּ / יְחוֹלָלוּ / לְחַרְבָּם דְּוֵי נֶפֶשׁ / בְּלִי נֹפֶשׁ / וְאֵין חֹפֶשׁ / לְעָצְבָּם הֲכִי דוֹדִים / כְּכִידוֹנִים / וְלַפִּידִים / בְּלִבָּם וְלֵב יִקְלוּ / וְלֹא יִצְלוּ / וְאָכְלוּ אֵין / מְכַבָּם הֱדִיחוּהוּ / שְׁחָטוּהוּ / זְבָחוּהוּ / בְחַרְבָּם יְכַתּוּהוּ / יְחַתּוּהוּ / יְפַתּוּהוּ / בְּעָקְבָּם וְעֵינֵיהֶם / מְעַנֵּיהֶם / וּמוֹנֵיהֶם / בְּעָגְבָם


וְדַע וַחְרֹשׁ / חֲקֹר וּדְרֹשׁ / הֲלֹא הֵם רֹאשׁ / מְסִבָּם וְשָׁם חָשִׁים / בְּנֵי כוּשִׁים / יְדוֹת אִשִּׁים / בְּגַבָּם וְדוֹד יַלְאוּ / וְלֵב מָצְאוּ / וְלֹא חָטְאוּ / בְרִיבָם וְלֹא דָמוּ / וְלֹא שָׁלְמוּ / אֲבָל הָלְמוּ / עֲקֵבָם בְּנֵי־חֵשֶׁק / בְּנֵי־עֹשֶׁק / כְּלֵי־נֶשֶׁק / סְבִיבָם לְבַב דּוֹד דַּל / מְאֹד יִדַּל / עֲדֵי יִגְדַּל / כְּאֵבָם וְעֵינַיִם / כְּעֵין מַיִם / כְּנַחְלַיִם / שְׁאָבָם וְצָרוֹתָם / וְקוֹרוֹתָם / קְצוֹתָם, מִי / כְתָבָם וְדֹדֵיהֶם / מְנַדֵּיהֶם / יְקוֹדֵיהֶם / וְלַהְבָם וְיָד יַהְפֹךְ / וְדוֹד יִמְשֹׁךְ / וּמִי יִדְרֹךְ / נְתִיבָם וְיַגִּידוּ / וְיָעִידוּ / וְיַפְחִידוּ / בְנִיבָם דְּבַר־שֶׁקֶר / בְּפֶה יִקֶּר / לְאֵין חֵקֶר / כְּזָבָם וְזֶה יִמְחַץ / וְזֶה יִלְחַץ / עֲדֵי יַחַץ / לְבָבָם וְלוּלֵי דוֹד / אֲשְֶׁר יִדּוֹד / גְּבִיר כָּל־הוֹד / נְצִיבָם אֲשֶׁר פָּרַח / כְּאוֹר זָרַח / בְּהוֹד אֶזְרָח / נְדִיבָם אֲזַי מַתִּי / וְגָוַעְתִּי / וְנִצְמַתִּי / לְכִזְבָם כְּלִיל יֹפִי / בְּלִי דֹפִי / דְּבַשׁ צוּפִי / עֲרֵבָם וְיָד שִׁוָּה / לְדַל תִּקְוָה / עֲדֵי רָוָה / שְׁרָבָם לְמוֹשִׁיעַ / לְיָגֵעַ / לְמַשְׂבִּיעַ / רְעָבָם נְדִיבֵי עָם / וְכָל־שׁוֹעָם / בְּלִי טַעַם / עֲזָבָם.


וַיֹּאמֶר הַמֶּלֶךְ לְסַהַר: ״וְאִם בְּכִימָה חָשַׁקְתָּ / רְאֵה חַיִּים עִם הָאִשָּׁה אֲשֶׁר אָהַבְתָּ״. / וַיֹּאמֶר הַמֶּלֶךְ לְכָל שָׂרָיו וַעֲבָדָיו / וּלְכָל רָאשֵׁי חַיָּלָיו וּגְדוּדָיו / וּלְכָל הָעָם הַנִּמְצָאִים סְבִיבוֹ: / ״הֲנִמְצָא כָזֶה אִישׁ אֲשֶׁר רוּחַ אֱלֹהִים בּוֹ״? / וַיַעַשׂ לָהֶם מִשְׁתֶּה גָדוֹל, וַיֵּשֶׁב סַהַר עִם כִּימָה / בְּמִשְׁתֶּה וּבְשִׂמְחָה שָׁנָה תְמִימָה. / וַיִּנָּגֵף הַמֶּלֶךְ וַיָּקָם סַהַר מִיֵּין מִשְׁתָּיו / הוּא וְְהַשָּׂרִים וְכָל מְשָׁרְתָיו / וַיִּקְבְּרוּהוּ וַיַּמְלִיכוּהוּ תַּחְתָּיו / וַיֵּשֶׁב סַהַר הַמֶּלֶךְ עִם כִּימָה הַגְּבִירָה / מוֹשֵׁל בְּצוֹבָה הַבִּירָה.

וַיְהִי אַחֲרֵי כֵן וַיָּקָם סַהַר בְּעָרְמָה / וַיָּרֶב בְּכִימָה / וַיְהִי רִיב מְקַדֵּשׁ / וְדִבְרֵי חִשְׁקָם מְחַדֵּשׁ / כִּי רִיב אֲהָבִים הַיְדִידֹת לוֹטֵשׁ / וְחוֹשֵׁק בְּלִי רִיב חִשְׁקוֹ נוֹטֵשׁ. / וַיִּפְקֹד עָלֶיהָ עֲווֹן מִרְיָהּ / וּנְדוֹדָהּ מִמֶּנּוּ וְקָשְׁיָהּ / בְּאַהֲבָתָם אֲשֶׁר נֶאֱהָבוּ / טֶרֶם יִשְׁכָּבוּ / וַיֹּאמֶר לָהּ: ״הָיִיתִי יוֹשֵׁב בְּאַרְמוֹנֵךְ / וַתָּפִיחִי לִי רֵיחַ קִנְמוֹנֵךְ / וְתִכָּבְשִׁי כְּאַוָּתִי וּרְצוֹנִי / וְלֹא עָשִׂית כַּטּוֹב בְּעֵינִי״. / וַתַּעַן כִּימָה וַתֹּאמַר / ״הֲכָזֶה יַעֲשׂוּ בַאֲהָבִים / דּוֹדִים הַנְּדִיבִים / הֲלֹא זֹאת חֶרְפָּה וּבוּשָׁה / כִּי תִפְקֹד עָלַי עֲווֹן הָאִשָּׁה. / אַל תִּזְכָּר לִי עֲווֹנוֹת רִאשׁוֹנִים / הֲלֹא דִבַּרְתָּ מִיָּמִים קַדְמוֹנִים / מִי יִתֵּן וְאֶרְאֶה כִימָה / וְיַעֲבֹר עָלַי מָה / וְאִם לְבָבְךָ מֵעָלַי נָטָה / אוֹ לִבִּי מֵעָלֶיךָ שָׂטָה / וְאִם בְּאַדְמַת חִשְׁקֵנוּ הָיְתָה שְׁמִטָּה / יֵצֵא חוֹחַ תַּחַת חִטָּה / וְאִם תִּרְצֶה אַהֲבָתֵנוּ בְרִיב לִלְטֹשׁ / לִפְנֵי הַתְּלָאָה הִתְגַּלַּע הָרִיב נְטֹשׁ / הֲלֹא הָאַרְמוֹן אַרְמוֹנֶךָ / וְהַקִּנָּמוֹן קִנְמוֹנֶךָ / וּמַדּוּעַ תְּנִידֵם מִמֶּנִּי / וּבִנְדוֹד תַּחְלִימֵנִי / שׁוּב וַהֲשִׁיבֵם עַד אֲרִיחֵם בְּאַפִּי / כַּאֲשֶׁר הָיִיתִי בִּימֵי חָרְפִּי״.



וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

רֵיחַ בְּשָׂמִים נָדֲדָה מֶנִּי / גַּם נִחֲרוּ בִי אֶשְׁכְּלוֹת כֹּפֶר

מִטְרוֹת חֲרוֹן דּוֹד גָּבְרוּ אֵין לִי / כִּי לַעֲשׂוֹת תֵּיבַת עֲצֵי גֹפֶר

אֶשְׁתּוֹלֲלָה אֶשּׁוֹם וְאֶשְׁתָּאֶה / אַבִּיט מְקוֹם דוד… שֶׁפֶר

אֶתְחַבְּאָה עַד יַעֲבֹר זַעַם / עַד מֵחֲרוֹנוֹ יִהְיֶה צוֹפֵר

אִם לֵב יְדִיד תַּתָּה עֲלֵי דוֹד זָר / אוֹ הֶחֱלִיף תַּחַת פְּאֵר אֵפֶר

חָטָא וְגָנַב לֵב יְדִיד עַל־כֵּן / לֹא יִמְצְאָה עַל מַעֲלוֹ כֹפֶר

לֵבָב קְרַעְתּוֹ בַנְּדוֹד קוּם־נָא / אוּלַי קְרָעָיו תִּהְיֶה תוֹפֵר

אֵיךְ חוֹשְׁקִים יֵהָפְכוּ עוֹשְׁקִים / דּוֹד שַׁעֲשׁוּעִים אֵיךְ בְּרִית יָפֵר

וָאֶשְׁאֲלָה עַל־זֹאת וְעָנוּנִי / כֵּן מִשְׁפַּט חָשׁוּק וְחֹק עֹפֶר

אִם אֵין בְּאַהֲבָה רִיב וְאֵי מִתְקִי / תּוֹכְחוֹת יְדִיד אוֹ אֵי נְעִים סֵפֶר.


וַיֹּאמֶר סַהַר: ״אַל תָּשִׁיתִי לִבֵּךְ וּדְבַר הַקִּנְאָה הֲלֹא מְשַחֵק אָנִי״. / וַיִּנָּחֵם סַהַר עַל אֲשֶׁר דִּבֵּר קָשׁוֹת וְעַזּוֹת / וַיֹּאמֶר: ״מֶה חָרַדְתָּ אֶת כָּל הַחֲרָדָה הַזֹּאת״. / וַיִּהְיוּ כֵן מְרִיבִים סַהַר וְכִימָה / מִיָּמִים יָמִימָה / וְכָזֹאת הָיוּ מוֹלִיכִים אַהֲבָתָם וּמְחַדְּשִׁים / וְחַרְבוֹת חִשְׁקָם מְלַטְּשִׁים / וְחֵמָה בֵינֵיהֶם לֹא עָלְתָה / וְנֶפֶשׁ אִישׁ לְאָחִיו נִכְסְפָה וְגַם כָּלְתָה.
