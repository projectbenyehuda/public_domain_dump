אָנָּא חִישׁ

זה השיר הוא על סדר א״ב. ופסוק אחד אומרת כנסת ישראל להקב״ה, והקב״ה משיב לה בפסוק שני.

סימן: א״ב. משה ישועה יצ״ו. לחן: יאמר נא ישראל.


אָנָּא חִישׁ יִשְׁעֵנוּ / אָבִינוּ מַלְכֵּנוּ / וּשְׁעֵה הֶגְיוֹנֵנוּ / דּוֹד אֶשְׁכֹּל הַכֹּפֶר

בָּחַרְתִּי בָךְ כַּלָּה / יְסוֹד עִיר תְּהִלָּה / אֲיַסֵּד בְּחֶמְלָה / מִגְדָּלָהּ אֲסַפֵּר

גָּדוֹל, לִישׁוּעָתָךְ / הוֹדָךְ וְזִמְרָתָךְ / תְּיַחֵל עֲדָתָךְ / אֶת פִּשְׁעָם תְּכַפֵּר

דְּעִי כִי אֵל תָּמִים / לָנֶצַח לָךְ יָשִׂים / גְּבֶרֶת עוֹלָמִים / וְדַרְכֵּךְ יְשַׁפֵּר

הָשֵׁב שְׁבִיתֵנוּ / תִּפְאֶרֶת רָאשֵׁינוּ / כִּי אַתָּה אָבִינוּ / אֵל, כּוֹכָבִים סוֹפֵר

וְאֶפְדֵּךְ בִּמְהֵרָה / בְּעֹז עִם גְּבוּרָה / וּלְזֶרַע קְטוּרָה / אָשִׂים עָפָר אֵפֶר

זְכֹר לִי אַדִּירִי / אֶת חֶסֶד נְעוּרִי / וְתִשְׁלַח אֶת צִירִי / קַל כְּמוֹ הָעֹפֶר

חֲמַדְתִּיךְ, רַעְיָתִי / אַנְחֵךְ בְּבִינָתִי / וְאֶשְׁפֹּךְ חֲמָתִי / עַל נִין עֵיפָה עֵפֶר

טוֹב וְיָשָׁר שִׂימָה / עָלַי רוּחַ חָכְמָה / וְרֹאשִׁי הָרִימָה / עַל צַר גֻּמָּץ חוֹפֵר

יְחִידָה וְתַמָּה / בְּכֶתֶר אַחְלָמָה / אֲשִׂימֵךְ רְשׁוּמָה / גִּלּוּף בְּעֵט סוֹפֵר

כָּל בֹּשֶׂם טוֹב וָרֹאשׁ / וְכֶתֶר שִׂים בָּרֹאשׁ / בֶּן יִשַׁי הוּא בָּרֹאשׁ / בְּשִׁירָיו נְסַפֵּר

לְעֵדָה שְׁפָלָה / חָכְמָה עִם נַחֲלָה / אֶתֵּן וְאֶשְׁלַח לָהּ / פִּדְיוֹן וְגַם כֹּפֶר

מָרוֹם, חוּס עָלֵינוּ / בְּבִינָה תַנְחֵנוּ / וּשְׁעֵה הֶגְיוֹנֵנוּ / שִׁיר בְּאִמְרֵי שֶׁפֶר

נָאווּ לְחָיַיִךְ / בְּחֶסֶד הוֹרַיִךְ / אֶקְבֹּץ קְהָלַיִךְ / מֵעִיר וּמִכֹּפֶר

סַלְעִי וּמְצוּדָתִי / אֵל גִּבּוֹר, תִּקְוָתִי / בְּהֶגְיוֹן תּוֹרָתִי / כְּדַם זִבְחִי כַפֵּר

עוּרִי מִתְּנוּמוֹת / תִּפְאֶרֶת עֲלָמוֹת / אֶקְבְּצֵךְ מֵאֻמּוֹת / אֲבִיאֵךְ הַר שֶׁפֶר

פְּאֵר בֵּית הַמִּקְדָּשׁ / לָנֶצַח יִתְחַדָּשׁ / אָשִׁירָה שִׁיר חָדָשׁ / וְאֶכְתֹּב בְּסֵפֶר

צְבִיָּה אֲהוּבָה / אֶת הוֹדֵךְ אָשִׁיבָה / לְצִיּוֹן בְּחִבָּה / עִם כֹּהֵן הַמְכַפֵּר

קוּמָה, דּוֹדִי, וְשׁוּב / יַסֵּד יְסוֹד חָשׁוּב / לְעִיר קָדְשָׁךְ נָשׁוּב / בְּשִׁבְחָךְ נְסַפֵּר

רַנְּנִי בְשִִׁירָה / מַלְכוּת גַּם עֲטָרָה / אָשִׁיב לָךְ גְּבִירָה / עִם נֵרְדְּ וּמֹר כֹּפֶר

שְׁלַח לִי הַגּוֹאֵל / צוּר עוֹנֶה לְשׁוֹאֵל / עִם זֶרַע יִשְׂרָאֵל / בְּרִיתָךְ אַל תָּפֵר

תַּמָּתִי יְדִידָתִי / כַּלָּה הַיְּחִידָה / אֶת אוֹיְבֵךְ אַשְׁמִידָה / וְלָךְ יִהְיוּ כֹפֶר

מַלְכִּי שׁוּר הֶגְיוֹנִי / יָהּ שְׁעֵה וּפְדֵנִי / עֶלְיוֹן הַקַּדְמוֹנִי / יִדְמֶה צְבִי וָעֹפֶר.
