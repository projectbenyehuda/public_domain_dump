ד אַלְפַּיִם קְדוּמָה

א

אַלְפַּיִם קְדוּמָה בְּחָלְשִׁי

בִּמְשִׁיבַת הֵשִׁיב נַפְשִׁי

גָּעֲשׁוּ רָעֲשׁוּ דּוֹק וְגַם נֶשִׁי

דָּת הֶחָצוּב מִלַּהַב הוֹרִישִׁי


הוֹרִישִׁי / מְתוּקִים מִדְּבַשׁ לְהַנְפִּישִׁי

וְכַכַּלָּה לֶחָתָן בְּיוֹם חֲתֻנָּה קִדְּשִׁי

זֵר תִּפְאֶרֶת וַעֲטָרָה הִלְבִּישִׁי

חִנְּכִי חִבְּרִי לְהַעְלִים טִפְשִׁי


טִפְשִׁי / הֶעֱבִיר מִנָּשִׁים כְּהִפְרִישִׁי

יֵרָאֶה לְלַמְּדִי מַלְכִּי וּקְדוֹשִׁי

כְּבוֹדִי וּמֵרִים רֹאשִׁי

לְמָאתַיִם וָעֶשֶׂר הֶעֱלִי מֵרִפְשִׁי


מֵרִפְשִׁי / רְחָצִי וְאִזֵּן צָקוּן לַחְשִׁי

נְשָׂאַנִי עַל אֵבֶר לְטַבּוּר מִקְדָּשִׁי

סִינַי בְּיוֹם זֶה הִגִּישִׁי

עֵדוּת נֶאֱמָנָה בְּכָל לֵב פִּלְּשִׁי


פִּלְּשִׁי / פִקּוּדִי כְּהוֹצִיאִי חָפְשִׁי

צוֹעֲנִים מָחַץ וּפָדַנִי בַּחֲמִישִׁי

קֵץ חָתַךְ וּמִלּוּד תְּלָשִׁי

רַעֲנָן עַל מַיִם הִפְרִיחַ שָׁרְשִׁי


שָׁרְשִׁי / הֶעֱנִקַנִי וְכַנֶּשֶׁר חִדְּשִׁי

שֵׁשׁ מֵאוֹת וּשְׁלֹשׁ עֶשְׂרֵה לְקַבֵּל בְּפֵרוּשִׁי

תּוֹלְדוֹת שְׁלִישִׁיָּה שָׁלִישִׁים שִׁלְּשִׁי

תִּרְגְּלִי הִלְפִּידִי בַּשִּׁשִּׁי בַּחֹדֶשׁ הַשְּׁלִישִׁי


א1

לָמוֹ / נָחֲלוּ מְבוֹנַי

עַל הַר סִינַי

זָרַח וְהוֹפִיעַ לְהָאִיר עֵינַי

רֵעִים לְשַׂגֵּב בְּמָגִנַּת אֵיתָנַי.


ב

תּוֹרָה תְמִימָה יָרְשׁוּ יְדִידִים

שְׁבוּיַת מְרוֹמִים בְּקוֹל פְּחָדִים

רוֹעֲדִים וּמִתְיָרְאִים וּלְאָחוֹר צוֹעֲדִים

קְצִינִים נְגִידִים כְּאֶחָד צְמוּדִים


צְמוּדִים / כִּשְׁנֵי עֵדִים שְׁנַיִם עַל נַפְשָׁם מְעִידִים

פִּקּוּדִים יְשָׁרִים לוֹמְדִים וּמְלַמְּדִים

עִזּוּז וְנָאוֹר עַל דְּבָרָיו חֲרֵַדִים

סְבִיבוֹת חוֹרֵב לְהַקִּיף נוֹעָדִים


נוֹעָדִים / לֶאֱחֹז דְּבָרִים עֲתִידִים

מִזָּהָב וּמִפַּז נֶחֱמָדִים

לְשֶׁמַע אָנֹכִי וְלֹא יִהְיֶה לְךָ נוֹדְדִים

כְּהִרְעִים בְּקוֹל מוֹשִׁיב יְחִידִים


יְחִידִים / הוֹשִׁיב כְּפָדָם מִבֵּית עֲבָדִים

טַבּוּר גֵּיא קִרְיַת מוֹעֲדִים

חִכּוֹ מַמְתַּקִּים וְכֻלּוֹ מַחֲמַדִּים

זָבַד זֶרַע חָז עוֹלִים וְיוֹרְדִים


וְיוֹרְדִים / וּמִתְוַעֲדִים וּבְחִילָה מְסַלְּדִים

הַשְּׁכִינָה כְחָזוּ וַהֲמוֹן גְּדוּדִים

רגליו הַיּוֹם בְּחוֹרֵב נִצָּבִים וְעוֹמְדִים

גָּשִׁים בְּקִדּוּשׁ וּבְכִבּוּס בְּגָדִים


בְּגָדִים / רְחוּצִים כְּמַלְאָכִים אֲפוּדִים

בָּאֲשִׁישׁוֹת סְמוּכִים וּבַתַּפּוּחִים רְפוּדִים

אֲמוּנָה לִטֹּל בְּקוֹלוֹת וְלַפִּידִים

אָתוּ לְסִינַי מֵרְפִידִים


ב1

יִשְׂרָאֵל / אִזֵּן דִּבְּרוֹת עֲשָׂרָה

לְיִרְאָה אֶת הַשֵּׁם הַנִּכְבָּד וְהַנּוֹרָא

עַל רֹאשָׁם הִצְנִיף זֵר וַעֲטָרָה

לְהַחֲיוֹתָם בְּטַלְלֵי אוֹרָה

ברוך… מחיה


ג

אָמוֹן שַׁעֲשׁוּעִים הוּרָשׁוּ נַחֲלָה

תַּעֲלוּמוֹת חָכְמָה לְתוּשִׁיָּה כְפוּלָה

בְּמוֹעֵצוֹת וָדַעַת מִיָּד לְיָד קִבְּלָהּ

שָׁלִישִׁים כָּתַב וַחֲתוּמִים גִּלָּה


גִּלָּה / סְתוּמִים וְחָשַׂף בְּגִילָה

רֶכֶב רִבּוֹתַיִם כְּרָכַב בִּגְדֻלָּה

דּוֹק הִרְעִישׁ וְהִרְעִים […]

קוֹלוֹת שֶׁבַע כְּקוֹל הֲמֻלָּה


הֲמֻלָּה / גְדוֹלָה צוֹפָה בְחִילָה

צִיר נֶאֱמָן מַחֲנַיִם עִלָּה

וַיְהִי שָׁם עִם יְיָ אַרְבָּעִים יוֹם וְאַרְבָּעִים לַיְלָה

פֶּה אֶל פֶּה כְּקָשְׁבָה מִמַּצָּבָהּ זָחֲלָה


זָחֲלָה / וְנִרְתָּעָה וְאָחֲזָה חַלְחָלָה

עַל הַר חֶמֶד כְּנִגְלָה

חֻקִּים וּמִשְׁפָּטִים עָרַךְ נוֹרָא עֲלִילָה

סִינַי גָּשׁ וּבְטָהֳרָה טְבִילָה


טְבִילָה / כִלְבוֹנָה זַכָּה אוֹם נִדְגָּלָה

נֶאֱמָנָה לְהַנְחִיל מֵעַמִּים סְגֻלָּה

יְקָרָה מִפְּנִינִים לְהַעֲצִים תְּחִלָּה

מֵחוֹרֵב קִבְּלָה כְתֻבָּה כְּכַלָּה


כְּכַלָּה / אֲיֻמָּה נִמְשְׁלָה

כְּתוּרָה בְעֶשְׂרִים וּשְׁנַיִם מִי זֹאת עוֹלָה

לִוְיַת חֵן לְרֹאשָׁהּ כַּעֲגֻלָּה

לְהוֹרִיד לֶקַח טוֹב וּמֹשֶׁה עָלָה


ג1

בַּחֹדֶשׁ הַשְּׁלִישִׁי בֵּאֵר מְשׁוּלַת מַיִם

יְקָרָה הָיְתָה אֶצְלוֹ מִיּוֹמַיִם

בְּשִׁשָּׁה לַחֹדֶשׁ הִרְעִים קוֹל עַל הַמַּיִם

יָרַד לְדַבֵּר עִמָּהֶם מִשָּׁמַיִם


ג2

קְדוּמָה לָמְדוּ שִׁשִּׁים רִבֹּא

לְלַוּוֹתָהּ יָרְדוּ שִׁשִּׁים רִבּוֹא
