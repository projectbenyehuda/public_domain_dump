בַּחֲלוֹם חֶזְיוֹן לָיְלָה

עַל מִשְׁכָּבִי בַלֵילוֹת נָדְדָה שְׁנָתִי,

בִּקַשְׁתִּי תְנוּמָה וְלֹא מְצָאתִיָה;

חֶזְיוֹנֵי לַיְלָה עָלוּ מִטָתִי

וַיַרְאוּנִי תְמוּנוֹת מֵאֶרֶץ נְשִׁיָה!


תְּמוּנוֹת זָרוֹת, צוּרוֹת אֲיוּמוֹת

כִּתְרוּנִי מִסָבִיב וַיַרְגִיזוּ מְנוּחָתִי;

הֲמוֹן שֵׁדִים וּשְׂעִירִים וּשְׁאוֹן מְהוּמוֹת

יַחְדָו הִתְלַכְּדוּ לְהַגְדִיל פַּחֲדָתִי!


אַחַת לְאַחַת הַתְּמוּנוֹת נָשְׂאוּ כַנְפֵיהֶן

וַיָעוּפוּ לָמוֹ, לֹא יָדַעְתִּי אָנָה;

שְׁמוּרוֹת עֵינַי סָגְרוּ עַפְעֲפֵיהֶן,

וְחוּשַׁי בִי עֲזָבוּנִי, וָאִישָׁנָה!


הֶקִיצוֹתִי, וְרוּחִי בְקִרְבִּי חֻבָּלָה,

וּלְבָבִי גַם הוּא כְּהוֹלֵם פָּעַם;

אֵימָתָה גְדוֹלָה עָלַי נָפָלָה

וַחֲרָדָה כִבְיוֹם זַעַם וָרָעַם!


וְלָמָה זֶה אִירָא, מַדוּעַ נִדְהָמְתִּי?

מַה זֶה הַחֲרָדָה, וּמֶה הַבְּהָלָה?

חָזוּת קָשֶׁה הָרְאֵיתִי, חֲלוֹם חָלַמְתִּי,

וּלְזִכְרוֹ תֹאחֲזֵנִי פַּלָצוּת וְחַלְחָלָה!


בַּחֲלוֹמִי, וַאֲנִי בְתוֹךְ עַמִי יָשַׁבְתִּי,

וְאֵת חֲבֵרַי, בְּנֵי גִילִי הִתְהַלַכְתִּי;

לְכָל הֶגֶה מִפִּיהֶם קֶשֶׁב הִקְשַׁבְתִּי,

וּבְדַבְּרָם, נַחֲלֵי דְמָעוֹת שָׁפַכְתִּי!


שׁוֹד וָרֶצַח, הָהּ! נַעֲוֵתִי מִשְׁמוֹעַ,

הִרְבּוּ בְּעַמִי יְדֵי זֵדִים אֲרוּרִים;

שֵׂה פְּזוּרָה יְהוּדָה חָשׁוּ לִבְלוֹעַ,

לְהַעֲלוֹתָהּ כָּלִיל, חַטַאת כִּפּוּרִים!


וּמַה פִּשְׁעָם, שָׁאַלְתִּי, מֶה חֲטָאתָם,

לָמָה זֶה רֻדְפוּ בַגוֹיִם חִנָם?

אֵין רַע בְּעַמֵנוּ, יָשְׁרָה פְּעוּלָתָם,

אֲבָל, יַעַן כִּי יְהוּדִים הֵמָה!


עוֹדָם מְדַבְּרִים, וְהִנֵה הִקִיפָה זְעָקָה

אֶת כָּל רְחוֹבוֹת הָעִיר וּמִגְרָשֶׁיָה;

נַאֲקַת שֶׁבֶר וְהֹוָה, יְלָלָה וּצְעָקָה

עָלְתָה בְּאָזְנַי מִכָּל חוּצוֹתֶיָה!


הַכּוּ בַיְהוּדִים, הַכּוּ אַל תַּחֲמוֹלוּ,

הֵרִיעוּ שׂוֹנְאֵי יִשְׂרָאֵל לִקְרָאתָם!

זָקֵן גַם נַעַר בַּחֶרֶב יִפּוֹלוּ,

אִישׁ לֹא יִמָלֵט אַךְ בְּהָמֵר דָתוֹ!…


וּלְעֵינַי דַם עַמִי כְּמַיִם נִגָרִים,

וְאֶנְקַת חַלָלִים מִכָּל עֵבֶר נִשְׁמָעָה;

כָּשְׁלוּ וָנָפְלוּ בְנֵי צִיוֹן הַיְקָרִים,

וְכִמְעַט עַד נַפְשִׁי הַחֶרֶב נָגָעָה


הַרְפּוּ מִמֶנִי! אַל הָמֵת תְּמִיתוּנִי!

מֵאֵימַת הַמָוֶת לְרוֹצְחַי קָרָאתִי!

חוּסִי עָלַי, רַחֲמוּנִי, נָא רַחֲמוּנִי!

וּתְנִי לִי אֵת נַפְשִׁי כֹּפֶר דָתִי!


מֵרֹאשׁ הַמִגְדָל יַכּוּ הַפַּעֲמוֹנִים

וַיְמַלְּאוּ עִיר וּמְלוֹאָהּ בִּשְׁאוֹנָם;

לְבֵית הַתִּפְלָה יְרוּצוּן הֲמוֹנִים,

חֲרָבוֹת בְּיָדָם, וְרוֹמְמוּת אֵל בִּגְרוֹנָם!


וְהַבַּיִת מָלֵא אֲנָשִׁים מִבְּנֵי הָעֲרֵלִים,

וְאָנֹכִי בְתוֹכָם לְכַחֵשׁ בֶּאֱמוּנָתִי;

וּכְהֲרִימִי קוֹלִי לְהֹדוֹת בְּמַקְהֵלִים,

וְהִנֵה אִמִי הַמֵתָה לִקְרָאתִי!


דוֹם נָבָל, קָרְאָה, יְצוּר תּוֹלֵעָה וְרִמָה,

וְהִשָׁמֵר מֵחֲטוֹא לֵאֱלֹהֵי אֲבוֹתֶיךָ!

אֲמִיתְךָ לְעֵינַי טֶרֶם עֲשׂוֹתְךָ זִמָה

לְבָשְׁתְּךָ וּלְבוֹשֶׁת בְּנֵי אִמֶךָ!


וּכְדַבְּרָהּ חִבְּקוּנִי זְרוֹעוֹת שְׁתַּיִם

וְאֶצְבְּעוֹתֵיהֶן מַחֲנֵק לְצַוָארִי שָׂמוּ;

עוֹדֶנִי מְפַרְפֵּר בֵּין מָוֶת וְחַיִים,

הֱקִיצוֹתִי, וְחֶזְיוֹנַי עָפוּ, נֶעֶלָמוּ!…
