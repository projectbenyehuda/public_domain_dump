ספר תחכמוני - שַׁעַר שִׁבְעָה עָשָׂר

בְּוִכּוּחַ הַמִּין וְהַמַּאֲמִין. זֶה עַל שְׂמֹאל וְזֶה עַל יָמִין.


נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי בִּימֵי נְעוּרַי מִתְדַפֵּק עַל דַּלְתֵי חֲכָמִים וּמַשְׂכִּילִים. וְשׁוֹקֵד עַל שַׁעֲרֵי הַמְּבִינִים וְהַפְּלִילִים. לִלְקֹט בְּדָלְחֵי אִמְרֵיהֶם. וְלִשְׁאֹב מִמַּעְיְנֵי מִשְׁלֵיהֶם. וְלִרְווֹת צוּף דְּבַשׁ מִיַּעֲרוֹת מִלֵּיהֶם. וַיְהִי הַיּוֹם וַאֲנִי בְּמַעֲמַד חֲכָמִים יוֹשֵׁב עִמָּהֶם. וְשׁוֹמֵעַ אִמְרֵיהֶם. וְהָיָה חֶבֶר הַקֵּינִי בְּרֹאשׁ קְרוּאֵיהֶם. וַיִּשְׁאָלוּהוּ הָעָם הָעוֹמְדִים לְפָנָיו. וְהַשּׁוֹאֲבִים מִמַּעְיָנָיו. עַל דְּבַר הַכּוּתִים. הַכְּרוּתִים. וֶאֱמוּנָתָם הָרָעָה. הַגְּרוּעָה.

אָמַר לָהֶם: דְּעוּ כִּי הַכּוּתִים בֵּין הַבְּרִיוֹת. כְּשֶׁרֶץ בֵּין הַחַיּוֹת. עִם הַיְּהוּדִים לֹא יִמָּנוּ. וְעִם הָעֲרֵלִים לֹא נִתְכְּנוּ. וְעִם הַיִּשְׁמְעֵאלִים לֹא יְכֻנּוּ. אֲבָל הֵם בְּרִיָה בִּפְנֵי עַצְמָהּ. נוֹדַע לְכָל הָאֻמּוֹת חֶסְרוֹנָהּ וּמוּמָהּ. וּמַה תִּשְׁאַל עַל עֵדָה גַלְמוּדָה וְאֻמָּה זְעוּמָה. כְּעָנָף בְּלִי חֹרֶשׁ. וְאִילָן בְּלִי שֹׁרֶשׁ. נְבִיאָם סַנְבַלַּט הַחֹרֹנִי. וְשָׂרָם טוֹבִיָּה הָעֶבֶד הָעַמּוֹנִי. וּמִלְּבַד זֶה יַחְזִיקוּ בְתוֹרָתֵנוּ. וְיִתְגָּאוּ בָהּ עָלֵינוּ. כְּעֶבֶד מִתְגָּאֶה עַל אֲדוֹנָיו בְּעַזּוּת פָּנִים. [וּמְגַלִּים בָּהּ פָּנִים.] וְהֵם חֲשׁוּדִים בִּגְנֵבַת תּוֹרָתֵנוּ וּמַרְאִים בְּגָלוּי צִדְקָם. וְהַגְּנֵבָה בְּתוֹךְ חֵיקָם. יַחֲזִיקוּ בְתוֹרָתֵנוּ. וְיַעֲשׂוּ הֵפֶךְ פְּעֻלָּתֵנוּ. וְכֻלָּם כַּשִּׁכּוֹרִים נְבוּכִים. וְהוֹלְכֵי חֲשֵׁכִים. תּוֹרָתָם גְּרוּעָה. וֶאֱמוּנָתָם רָעָה. וּקְהִלָּתָם מְעוּטָה וְאֵינָם לְאֶלֶף אִישׁ מַגִּיעִים. כִּי הֵם חֲסֵרִים וּגְרוּעִים. קְלוּטִים וְלֹא שְׂרוּעִים. מְעַט וְרָעִים.

אָמְרוּ לוֹ: מַה תֹּאמַר עַל הַקָּרָאִים הַמִּינִים. אֲשֶׁר הֵם בְּדִבְרֵי רַבּוֹתֵינוּ אֵינָם מַאֲמִינִים. אָמַר לָהֶם: הֵם מִגּוֹאֲלֵינוּ. וְיוֹתֵר קְרוֹבִים אֵלֵינוּ. רַק לֹא יַאֲמִינוּ בְּדִבְרֵי פְּלִילֵנוּ. וְאַף עַל פִּי שֶׁיֵּשׁ בְּיָדָם תּוֹרָתֵנוּ. לֹא יִמָּנוּ עִם עֲדָתֵנוּ. כִּי נֶעֶבְדוּ לְתַאֲוָתָם בְּפֶרֶךְ. וְסָרוּ מַהֵר מִן הַדֶּרֶךְ.

אָמְרוּ לוֹ: וְאֵיךְ יִתְעַלֵּם מֵהֶם הָאֱמֶת. וְכָל מִי שֶּיֵּשׁ לוֹ שֵׂכֶל יוּכַל לְהַבְדִּיל בֵּין הַחַי וּבֵין הַמֵּת. הֲלֹא יְדְעוּ כִּי הַתּוֹרָה. כְּמוֹ טִירָה סְגוּרָה. וּבַחֲדָרֶיהָ כָּל אֶבֶן יְקָרָה. וּלְשׁוֹן הַחֲכָמִים כְּמַפְתֵּחַ לָבֹא חֲדָרֶיהָ. וְלִלְקֹט בְּדֹלָחֶיהָ וְדָרֶיהָ. וְאִם יֹאבַד הַמַּפְתֵּחַ. יִהְיֶה הַפֶּתַח סָגוּר וְלֹא יִפָּתֵחַ.

אָמַר לָהֶם: יַכִּירוּ כָּל זֶה וְיָבִינוּ. אַךְ לֹא יַאֲזִינוּ וְלֹא יַאֲמִינוּ. כִּי אֲבוֹתֵיהֶם רָצוּ לַעֲשׂוֹת לָהֶם תּוֹרָה חֲדָשָׁה. וְחִלְּלוּ דָת הַקְּדוֹשָׁה. וַתְּהִי בְתוֹכָם הָאֱמֶת נֶעְדֶּרֶת. וְאֱמוּנָתָם חוֹלָה וְנִשְׁבֶּרֶת. אָמְרוּ לוֹ: מַה הַיא טַעֲנָתָם. עַל אֱמוּנָתָם. אָמַר לָהֶם: יֵשׁ לָהֶם טְעָנוֹת בְּזוּיוֹת. וְעַל מוֹסְדֵי שָׁוְא בְּנוּיוֹת. וְכֻלָּם דְּחוּיוֹת. אָמְרוּ לוֹ: הֲלֹא תַגִּיד לָנוּ מִקְצָת עִנְיְנֵיהֶם. וְתַשְׁמִיעֵנוּ טַעֲנוֹתֵיהֶם. אָמַר לָהֶם: דְּעוּ כּי מִימֵי קֶדֶם נִקְבְּצוּ מִתּוֹפְשֵׂי הַתּוֹרָה שְׁנֵי מִינִים. קְצָתָם כּוֹפְרִים בְּחַכְמֵי יִשְׂרָאֵל וּקְצָתָם מַאֲמִינִים. אֵלּוּ יַחֲזִיקוּ בְדִבְרֵי רַבּוֹתֵינוּ כְּמוֹ רָבִינָא וְרַב אָשֵׁי. וְאֵלּוּ לֹא יַאֲמִינוּ בָם וְיַאֲמִינוּ בַּה' וּבְמשֶׁה. כִּי יִקְחוּ הַפָּסוּק עָרֹם מִבִּגְדֵי עֶדְיוֹ. וְיַפְשִׁיטוּ מֵעָלָיו כְּתֹנֶת יָפְיוֹ. כְּמִי שֶׁיִּשְׁמֹר הָאִילָן וְיַשְׁלִיךְ פִּרְיוֹ. וַיַּעֲנוּ הַמַּאֲמִינִים. וַיֹּאמְרוּ אֶל הַמִּינִים: יֵשׁ לִתְמֹהַּ עַל מִי שֶׁיֶּשׁ לוֹ דַעַת. וְנַפְשׁוֹ הָרַע וְהַטּוֹב יוֹדַעַת. וְאֵינָּה אֶל הָאֱמֶת נִכְנַעַת. כִּי אֵיךְ יַאֲמִין אִישׁ בְּתוֹרַת משֶׁה וּמִצְווֹתֶיהָ. אִם לֹא יַאֲמִין בְּפֵרוּשֵׁי הִלְכוֹתֶיהָ. וְלוּלֵי הַתַּלְמוּד. הָיְתָה הַתּוֹרָה כְּאֹהֶל בְּלִי עַמּוּד. וְלוּלֵי הַמִּשְׁנָה. נָמַר רֵיחַ הַתְּעוּדָה וְשָׁנָה. וְלוּלֵי הַגְּמָרָא. נָסְעוּ לוֹמְדֶיהָ מִמָּתְקָה וַיַּחֲנוּ בְּמָרָה. וְלוּלֵי הַפְּרוּשִׁים. רַבּוּ בַתּוֹרָה הַסְּפֵקוֹת וְהַשִּׁבּוּשִׁים. וְלוּלֵי הַחֲכָמַים אֵיךְ יֵדַע אִישׁ בְּדִבְרֵי תוֹרָה. לְהִשָּׁמֵר מֵהָעֲבֵרָה. אוֹ לְהִזָּהֵר בְּמִצְוָה קַלָּה אוֹ חֲמוּרָה. אָמְרוּ הַמִּינִים: הַמִבְּלִי אֵין אֱלֹהִים בְּיִשְׂרָאֵל. כִּי נְבַקֵּשׁ מֵחַכְמֵיכֶם עוֹזֵר וְגוֹאֵל. חָלִילָה לָנוּ לְחַלֵּל בַּעֲבוּר חַכְמֵיכֶם תּוֹרַת צוּר בְּרָאָנוּ. וְדָתוֹ אֲשֶׁר הִנְחִילָנוּ. אֲבָל נִשְׁמֹר דְּבָרוֹ וּמִבְטָאוֹ. כַּאֲשֶׁר קִבַּלְנוּ מִמּשֶׁה נְבִיאוֹ. וְלֹא נַחֲלִיפֶנּוּ וְלֹא נְמִירֶנּוּ. וְלֹא נוֹסִיף עָלָיו וְלֹא נִגְרַע מִמֶּנּוּ. אֲבָל חַכְמֵיכֶם. שִׁנּוּ הַתּוֹרָה בְּפֵרוּשָׁהּ אֲשֶׁר פֵּרְשׁוּ לָכֶם. כִּי יָצְאוּ מִן הַפְּשָׁט אֶל הַדְּרָשׁ. וּמִן הַדְּרָשׁ טָבְעוּ בְּיָם נִגְרָשׁ. וּפֵרְשׁוּ הַפָּסוּק עַל דֶּרֶךְ אֲשֶׁר לֹא פֹרַשׁ. אָמְרוּ לָהֶם הַמַּאֲמִינִים: נִרְפִּים אַתֶּם נִרְפִּים. וְעַד מָתַי אַתֶּם פּוֹסְחִים עַל שְׁתֵּי הַסְּעִיפִּים. אַחֲרֵי אֲשֶׁר הֻכָּה הַכַּפְתּוֹר וַיִּרְעֲשׁוּ הַסִּפִּים. וְעַד אָנָה יִפְתֶּה לְבַבְכֵם בִּשְׁקָרָיו.

אִם ה' הָאֱלֹהִים לְכוּ אַחֲרָיו. וְאִם הַבַּעַל לְכוּ אַחֲרָיו. וְעַתָּה יֵצְאוּ נָא שְׁנֵי אֲנָשִׁים אֶחָד מִכֶּם וְאֶחָד מִמֶּנּוּ. וְיוֹכִיחוּ בֵּין שְׁנֵינוּ. וְאָז יִוָּדַע הָאֱמֶת בֵּינֵינוּ. וַיֹּאמְרוּ כָל הָעָם: טוֹב הַדָּבָר אֲשֶׁר דִּבַּרְתָּ. וְיָפֶה גָזַרְתָּ. וַיֵּצְאוּ שְׁנֵי אֲנָשִׁים אֶחָד מִין. וְאֶחָד מַאֲמִין.

וַיִּנָּצוּ שְׁנֵיהֶם. וּשְׁתֵּי הַשּׁוּרוֹת שׁוֹמְעִים דִּבְרֵיהֶם.

וַיַּעַן הַמַּאֲמִין וַיֹּאמַר:

כָּל אִישׁ אֲשֶׁר יַאֲמִין בְּדָת משֶׁה לְבַד

וּדְבַר חֲכָמִים בָּחֳרִי יִבְזֵהוּ.

יִדְמֶה לְאִישׁ יֹאמַר אֲנִי אֶעֱשֶׂה דְבַר

מַלְכִּי וְאֶעְבֹר עַל דְּבַר מִשְׁנֵהוּ.

אָמַר הַמִּין:

אֲנִי מַאְמִין בְּדָת משֶׁה לְבַדָּהּ

וְלֹא אַכְזִיב דְּבַר חוֹזִים וְצוֹפִים.

וְלֹא אֶשְׁמַע לְחַכְמֵיכֶם לְמַעַן

אֲשֶׁר הֵמָּה דְבַר אֵל מַחֲלִיפִים.

וְאֶשָּׁעֵן בְּדָת אֵל הַמְּרִימָה

לְמַחְזִיק בָּהּ וְעוֹזְבֶיהָ כְּפוּפִים.

כְּאִישׁ נִסְמָךְ בְּעֵץ אַדִּיר וְחָזָק

וְלֹא יָשִׁית לְבָבוֹ לָעֲנָפִים.

אָמַר הַמַּאֲמִין:

שְׁאַל אַנְשֵׁי זְמַנִּים הַקְּדוּמִים

הֲיֵשׁ חָכְמָה אֲשֶׁר אֵין לָהּ מְפָרְשִׁים.

הֲלֹא סִפְרֵי רְפוּאוֹת נִסְתְּמוּ עַד

בְּשִׂכְלָם פֵּרְשׁוּם חַכְמֵי חֲרָשִׁים.

וְאִלּוּ עָזְבוּ אֹתָם סְתוּמִים

יְמוּתוּן בָּרְפוּאוֹת הָאֲנָשִׁים

וְכֵן הַדָּת תְּמוֹל הָיְתָה סְתוּמָה

וְאוֹתָהּ פֵּרְשׁוּ אָבוֹת קְדוֹשִׁים.

לְמַעַן כִּי בְתוֹכָהּ יֵשׁ כְּלָלִים

וְתַחְתָּם עִנְיְנֵי מִצְוֹת חֲמֻשִׁים.


וְסוֹדֵיהֶם בְּפֵרוּשׁ הַחֲכָמִים

מפֹרָשִׁים לְחֶפְצֵיהֶם דְּרוּשִׁים.

וְאִלּוּ נַעֲשׂוּ עַל פִּי פְשָׁטָם

נְהִי חוֹטְאִים וּמִן הַדָּת פְּרוּשִׁים.

ולוּלֵי הַחֲכָמִים נִהְיֶה בָם

כְּמוֹ עִוְרִים בְּלִי מוֹרֶה נָטוּשִׁים.


אָמַר הַמִּין:

דְּבַר משֶׁה מְבֹאָר הוּא וְגָלוּי

וּמִצְוֹתָיו לְכָל מַשְׂכִּיל יְדוּעוֹת.

וְהָאוֹמֶר סְתוּמִים הֵם יְגַלֶה

כְּסִילוּתוֹ וְהָיָה רַק זְוָעוֹת.

וְאֵין צֹרֶךְ לְפֵרוּשׁ הַחֲכָמִים

וְהֵם כַּמַסְמְרוֹת בַּלֵּב נְטוּעוֹת.

וְעַל כֵּן הַחֲכָמִים פֵּרְשׁוּ רִיק

וְנַפְשׁוֹתָם בְּדִבְרֵי שָׁוְא יְגֵעוֹת.


אָמַר הַמַאֲמִין:

בְּנִי אִם נַפְשְׁךָ בַּדָּת סְמוּכָה

וְתוֹךְ לִבְּךָ שְׁמוּרָה הִיא עֲרוּכָה.

קְחָה פָּסוּק וְצֵא אֵלַי כְּגִבּוֹר

עֲדֵי נַעֲרֹךְ קְרָב בַּמַּעֲרָכָה.

וְאֶשְׁאָלָךְ בְּעַד דִּינוֹ וְחֻקָּיו

וּמִשְׁפָּטָיו וְחַרְבּוֹ הַדְּרוּכָה.

וְאִם תּוּכַל לְגַלּוֹת אֶת סְתוּמָיו

בְּלֹא מִשְׁנָה וְתַלְמוּד וַהֲלָכָה

אֲנִי אָשׁוּב לְטַעֲנָתָךְ וְאוֹדֶה

וְאֹמַר כִּי לְךָ תֵאוֹת מְלוּכָה.


אָמַר הַמִּין:

בְּעוֹדִי חַי אֲנִי מַחְזִיק בְּדָת אֵל

וְאַנְחִילָה בְּעֵת מוֹתִי לְבָנִים.

וְלֹא אֶפְנָה לְדִבְרֵי הַחֲכָמִים -

וְלֹא אַבִּיט לְפֵרוּשׁ הַמְּבִינִים

אֲבָל אֶעְשֶׂה אֲשֶׁר עָשׂוּ אֲבוֹתַי

אֲשֶׁר שָׁכְנוּ בְצִיוֹן מִלְפָנִים.

אֲשֶׁר חָיוּ בְטֶרֶם הַחֲכָמִים

וְלֹא הָיוּ בְדוֹרוֹתָם גְאוֹנִים.

וְלֹא חָטְאוּ וְלֹא הִמְרוּ דְבַר אֵל

וּמַעֲשֵׂיהֶם לְאֵל הָיוּ נְכוֹנָים.


אָמַר הַמַּאֲמִין:

אֲבוֹתֶיךָ אֲשֶׁר הָיוּ לְפָנִים

דְּבַר אֵל שָׁמְרוּ אָבוֹת וּבָנִים.

וְאִם לֹא נִמְצְאוּ עִמָּם גְאוֹנִים

הֲלֹא הָיוּ לְוִיִּם שָׁם וְכֹהְנִים

וְהָיוּ הַנְּבִיאִים אָז בְּעַם אֵל

להוֹרוֹת מִשְׁפְּטֵי אֵל הַנְּכוֹנִים.

וְעַתָּה אָבְדוּ אֵלֶּה וְאֵלֶּה

וְהֵקִים צוּר מְקוֹמָם הַגְאוֹנִים.


אָמַר הַמִּין:

לוּ הַחֲכָמִים לֹא יְשַׁנּוּ דָת

הָאֵל. אֱהִי עוֹשֶׂה כְמִצְוָתָם.

רַק הֶחֱלִיפוּ עִנְיְנֵי תוֹרָה

עַל כֵּן אֲנִי אַחְלִיף נְתִיבוֹתָם.


דָּת אָמְרָה: אֵשׁ לֹא תְבַעֲרוּ יוֹם

שַׁבָּת וְהֵם רָדְפוּ סְבָרוֹתָם

וַיְבַעֲרוּ הָאֵשׁ לְהַדְלִיק נֵר

וַיֵּלְכוּ חשֶׁך לְנֵרוֹתָם.

עַל כֵּן אֲנִי לֹא אֶחֱטָא לְאֵל

כָּל עֵת אֲשֶׁר אַחְליף אֱמוּנָתָם.


אָמַר הַמַּאֲמִין:

אֲשֶׁר לוֹ רַעְיוֹן בָּהִיר וּמַזְהִיר

דְּבַר צֶדֶק בְּעֵין לֵב יֶחְזֶהוּ.

וּבִמְצוֹא הַחֲכָמִים נִיב מְבַטֵּל

לְנִיב אַחֵר יְבִינוּן מַעֲנֵהוּ.

וְלוּ עָמְדוּ בְלֵיל שַׁבָּת בְּאֹפֶל

יְהִי עָוֹן וְהָאֵל יִדְרשֵׁהוּ.

וְהַנֵּר לא דְלַקְנוּהוּ לְבַד כִּי

בְּיוֹם שִׁשִּׁי וְנִגְמַר פָּעֳלֵהוּ.

כְּאִישׁ עָשָׂה בְּחֹל מַעֲשֶׂה וּבָטֵל

בְּבוֹא שַׁבָּת וְנִשְׁאָר מַעֲשֶׂהוּ.

וְאֵשׁ קָרְבָּן הֲלֹא דָלְקָה בְשַׁבָּת

וְשׁוֹאֶלְךָ בְזֶה מַה תַּעֲנֵהוּ?

וְלָכֵן שׁוּב פְּקַח עֵינִי לְבָבָךְ

וְאָז אוֹר הַנְכוֹחָה תֶּחֱזֶהוּ.


אָמַר הַמִּין:

אֶמֶת כָּל הַדְּבָרִים הֵם יְגֵעִים

וְחָק־הָאֵל מְלֵא שֵׂכֶל וְדֵעִים.

וְדִבְרֵי הָאֱלֹהִים הֵם שְׁלֵמִים

לְכָל מַשְׂכִּיל וְלַסָּכָל גְּרוּעים.

וְאִמְרֵי הַגְּאוֹנִים הֵם פְּרִי בִּין

וְשָׁרָשָׁיו בְּלֵב מַשְׂכִּיל נְטוּעים.

וְהַתּוֹרָה מְאִירָה כַמְּנוֹרָה

וְאוּלָם הַחֲכָמִים לָהּ גְבִיעִים.

וְאֵדַע כִּי אֲשֶׁר יָמִיר דְּבָרָם

זָמָמָיו הֹלְכִים אָחוֹר וְתוֹעִים.

אֲבָל אֵיךְ אַחֲלִיף מִנְהָג אֲבוֹתַי

וְהֵם הָיוּ מְבִינִי דָת וְיוֹדְעִים.

וְאִם אֶעֱזֹב אֱמוּנָתָם וְאַקְשִׁיב

לְנִיבָךְ. אֶעֶשֶׂה אוֹתָם רְשָׁעִים.


אָמַר הַמַּאֲמִין:

אֲבוֹתֵיךְ אֲשֶׁר תָּעוּ תְחִלָּה

הַתִתְעֶה אַף כָּמוֹתָם בַּמְּסִלָּה.

רְחַק מֵהֶם וְאַל תִּקְרַב אֲלֵיהֶם

וְסָקֵל דַּרְכְּךָ וַחְזֹר חֲלִילָה.

וְשִׂים לִבָּךְ לְחֻקֵּי הָאֱלֹהִים

הֲכִי לֹא נִכְתְּבוּ לָךְ עַל נְקַלָּה.

וְהִתְבּוֹנָן בְּמִצְוַת אֵל כְּסֻכָּה

אֲשֶׁר הִנְחִיל לְעַם שָׂמָם סְגֻלָּה.

ולוּלֵי הַחֲכָמִים אֵיךְ תְּבִינָהּ

וְתֵדַע אִם כְּשֵׁרָה אוֹ פְסוּלָה.

וְכַמָּה יִהְיֶה אָרְכָּהּ וְרָחְבָּהּ

וְקוֹמָתָהּ וְחַמָּתָהּ וְצִלָּה.

וְאִם תִּהְיֶה עֲמֻקָּה אוֹ סְמוּכָה

וְאִם תִּהְיֶה רְבוּעָה אוֹ עֲגֻלָּה.

וְאֵיךְ תֵּדַע בְּשַׁבָּת הַיְצִיאוֹת

וְעֵרוּבוֹ וְשִׁעוּר הַמְּסִלָּה.

וְאוּלָם בֵָּאֲרוּ כָל זֶה חֲכָמִים

וְסַנְהֶדְרֵי קְטַנָּה עִם גְּדוֹלָה.

וְלָכֵן שׁוּב לְדֶרֶךְ אוֹר וְתִצְלַח

וְתַצִיל נַפְשְׁךָ מִבּוֹר אֲפֵלָה.


אָמַר הַמַגִיד: וְכַאֲשֶׁר הִרְבּוּ הַדְּבָרִים. נוֹדַע הָאֱמֶת מִן הַשְׁקָרִים. וְנִבְדַּל הַחשֶׁך מִן הַמְּאוֹרים. וְיָדְעוּ הַקָרָאִים כִּי דִבְרֵי גְאוֹנֵינוּ הֵם אוֹר עוֹלָם. וְכִי הֵם כְּעֵץ חַיִּים וְהַטּוֹעֵם פִּרְיָם וְאָכַל וָחַי לְעוֹלָם. וַיְמַהֲרוּ לצֵאת מִמּוֹקְשָׁם. לְהַצִּיל מִמָּוֶת נַפְשָׁם. וְעָזַב כָּל אִישׁ מִנְהַג אָבִיו. הָאוֹמֵר לְאָבִיו וּלְאִמּו לֹא רְאִיתִיו. וְאָכְלוּ מִפְּרִי עֵץ הַדַּעַת לְהַחְכִּים רַעְיוֹנֵיהֶם. וַתִּפָּקַחְנָה עֵינֵיהֶם, וְהֵבִיאוּ בְעֹל הָאֱמֶת צַוְּארֵיהֶם. וְעָשׂוּ זִקְנֵיהֶם. שָׁלוֹם בֵּינֵיהֶם.
