ספר תחכמוני - שַׁעַר שׁלֹשִׁים וְאֶחָד

כַּפָּרָשׁ אֲשֶׁר הֵתֵל בְּגִבּוֹרִים. וַיַּשְׁקֵם כּוֹס לַעֲנָה וְתַמְרוּרִים


נְאֻם הֵימָן הָאֶזְרָחִי: שַׂמְתִּי מְגַמָּתִי. בְּעֵת צֵאתִי מֵאַדְמָתִי. לִרְאוֹת פְּלִיאוֹת הָעוֹלָם. וְכָל סוֹד מִמֶּנִּי בַּל יִתְעַלָּם. וְסָבַבְתִּי אַפְסֵי תֵבֵל. וְלָקַחְתִּי הַנְּדוֹדִים לְנַחֲלָה וָחֶבֶל. וַיְהִי בִקְצֵה הַיָּמִים וַאֲנִי נוֹסֵעַ מִסּוּרְיָא לְאַנְטוֹכִיָא. וְעִמָּנוּ פְּאֵר הַגְּבִירִים. וְנֵזֶר הַחֲבֵרִים. חֶבֶר הַקֵּינִי. וְהָיָה מְסַפֵּר לָנוּ מְשָׁלִים וְחִידוֹת. וּמִלִּים חֲמוּדוֹת. וַיֹּאמֶר לָנוּ: אַגִּידָה נָא לָכֶם דָּבָר נִפְלָא. כָּמוֹהוּ עַל לֵב אִישׁ לֹא עָלָה. אָמַרְנוּ: הַגֵּד פְּלִיאָתְךָ וְנֵדָעֶנָּה. וְחוּדָה חִידָתְךָ וְנִשְׁמָעֶנָּה. אָמַר: הָיִיתִי נוֹסֵעַ מִדִּיבֹן. עַד אֶרֶץ חֶשְׁבּוֹן. עִם חֲבֵרִים מְאִירִים כְּכוֹכְבֵי לַיִל. גִּבּוֹרֵי חַיִל. בַּעֲלֵי זְרוֹעַ רָמָה. אֲחוּזֵי חֶרֶב מְלֻמְּדֵי מִלְחָמָה. וּבְעוֹד אֲנַחְנוּ נוֹהֲגִים מַרְכְּבוֹתֵינוּ. בִּנְתִיבוֹתֵינוּ. רָאִינוּ אִישׁ רוֹכֵב עַל סוּסוֹ וְהוּא נָחוּץ. וְכַבָּרָק יָרוּץ. מַשְׁקִיף מֵעַל רֹאשׁ הַגִּבְעַה. וְכַאֲשֶׁר רָאָה אוֹתָנוּ מִהַר לָרֶדֶת אֶל הַבִּקְעָה. וּמַסְוֶה עַל פָּנָיו. וְלֹא יְגַלֶּה כִּי אִם עֵינָיו. וְכַאֲשֶׁר רְאִינוּהוּ רַצְנוּ לִקְרָאתוֹ בִזְרוֹעוֹת חֲשׂוּפוֹת. וַחֲרָבוֹת שְׁלוּפוֹת. וּמִלְחֲמוֹת תְּנוּפוֹת. וְכַאֲשֶׁר הִדְבִּיקוּהוּ רָאשֵׁי חֲנִיתוֹתֵינוּ. הִתְנּפֵּל אַרְצָה לְעֻמָּתֵנוּ. וְנָשַׁק פַּעֲמֵי מַרְכְּבוֹתֵינוּ. וַיֹּאמֶר לָנוּ: שִׁמְעוּ אֵלַי. גִּבּוֹרַי וַאֲצִילַי. וְאָז תִּגְזְרוּ אֲשֶׁר תִּרְצוּ עָלַי. אָמַרְנוּ לוֹ: דַּבֵּר דְּבָרֶיךָ. וְגַלֵּה סְתָרֶיךָ. וַיְגַל סֵתֶר פָּנָיו. וְהֵסִיר הַמַּסְוֶה מִבֵּין עֵינָיו. וְהִנֵּה הִיא עַלְמָה. אוֹר הַשֶּׁמֶשׁ לְיָפְיָהּ לֹא דָמָה. וּמִפָּנֶיהָ לָבַשׁ הַשַּׁחַר אֵימָה. וְעָטָה כְלִימָּה. שְׂפָתֶיהָ כְּצִנְצֶנֶת הַמָּן. וְשִׁנֶּיהָ נֶטַע נַעֲמָן. וּשְׂעָרָהּ תְּכֵלֶת וְאַרְגָּמָן. מַעֲשֵׂה יְדֵי אָמָּן. וְגַבֵּי עֵינֶיהָ כִּשְׁנֵי טוּרִים בְּאֶצְבַּע אֵל כְּתוּבִים. מַעֲשֵׂה חשֵׁב כְּרוּבִים. וַנֹּאמֶר לָהּ: אִמְרִי לָנוּ מַה לָּךִ. וְהַגִּידִי אֲשֶׁר קָרָה לָךְ. אָמְרָה: לֹא אֲלֵיכֶם כָּל עוֹבְרֵי דֶרֶךְ הַבִּיטוּ וּרְאוּ אִם יֵשׁ מַכְאוֹב כְּמַכְאוֹבִי. וְחָנּוּנִי כִּי יַד אֱלֹוהַּ נָגְעָה בִי. אֲנִי הָיִיתִי עַלְמָה נֶעְלֶמֶת בְּחַדְרֵי מְלָכִים. וְהֵיכְלֵי נְסִיכִים. מִן הַשֶּׁמֶשׁ וְהַיָּרֵחַ. וּמִן הָאוֹר הַזּוֹרֵחַ. וְלֹא יָכְלוּ הָרוּחוֹת לִגְנֹב מִבְּשָׂמַי רֵיחַ. וְשׁוֹאַת רְשָׁעִים לֹא בָאַתְנִי. וְיָדָם לֹא הֱנִידַתְנִי. עַד אֲשֶׁר שָׁמַע שִׁמְעִי אֶחָד מִן הַפָּרִיצִים. וְאִישׁ מִן הֶעָרִיצִים. וּבָא בַחֲצוֹת הַלַּיְלָה וְנָקַב קִירוֹת בֵּיתִי. וְהוֹצִיאַנִי מִבֵּית אִמִּי וּמֵחֶדֶר הוֹרָתִי. וְאֶל הַמִּדְבָּר הַזֶּה הֳבִיאַנִי. וּבְתוֹךְ הַמְּעָרָה הִשְׁכִּיבַנִי. וּבְמַחֲשַׁכִּים הוֹשִׁיבַנִי. וְהִתְעוֹלֵל בִּי וְהֶעֱצִיבַנִי. וְכַמַּטָּרָה לַחֵץ הִצִּיבַנִי. וְזֶה לִי שָׁנָה תְמִימָה אֲשֶׁר אֲנִי יוֹשֶׁבֶתֶ בַּמִּדְבָּר הַזֶּה לְבַדִּי. וְאֵין נֶפֶשׁ מִבְּרוּאֵי עוֹלָם לְצִדִּי. עַד קָצְרָה נַפְשִׁי וְקַצְתִּי בְחַיַּי. וְהִכֵּיתִי לְחָיַי. וְאָמַרְתִּי טוֹב מוֹתִי מֵחַיַּי. וְטוֹב לִי לְסַכֵּן בְּעַצְמִי. וְלִמְסֹר לַמָּוֶת נַפְשִׁי וְדָמִי. מִהְיוֹתִי יוֹשֶׁבֶת בַּמִּדְבָּר הַזֶּה עַד יוֹם מוֹתִי. וָאָקוּם וְשַׂמְתִּי נַפְשִׁי בְכַפִּי וְאָמַרְתִּי: אֶבְטַּח עַל רַחֲמֵי ה' כִּי לֹא תָמְנוּ. אִם יְחַיּוּנוּ נִחְיֶה וְאִם יְמִיתוּנוּ וָמָתְנוּ. וַיְהִי בַחֲצִי הַלַּיְלָה וְהוּא שׁוֹתֶה שִׁכּוֹר עַל מִטָּתוֹ. כְּפִי מִנְהָגוֹ וְדָתוֹ. וּבָאתִי אֵלָיו בַּלָּאט. וְקָרַבְתִּי אֵלָיו מְעַט. וְשָׁלַפְתִּי חַרְבּוֹ. וּתְקַעְתִּיהָ בְלִבּוֹ. וְיָצְאָה מִגַּבּוֹ. וַיָּקָם עָלַי וְהֶחֱזַקְתִּי בִזְקָנוֹ. וְהִפַּלְתִּיו אַרְצָה עַל גְּחוֹנוֹ. וְהוֹצֵאתִי הַחֶרֶב מִבִּטְנוֹ. וְחָתַכְתִּי צַוְרוֹנוֹ. וְחַבְלֵי גְרוֹנוֹ. אַחֲרֵי כֵן מִהַרְתִּי. וְעַל סוּסוֹ רָכַבְתִּי. וּבְמִדְבָּרוֹת אָרַחְתִּי. וּמֵהַר אֶל גִּבְעָה בָּרַחְתִּי. וְלֹא שָׁקַטְתִּי וְלֹא נַחְתִּי. עַד אֲשֶׁר בָּכֶם פָּגַעְתִּי. וְעַתָּה הִנְנִי בְיֶדְכֶם. עֲשׂוּ לִי כַּטּוֹב בְּעֵינֵיכֶם. וּכְשָׁמְעֵנוּ דְבָרֶיהָ. חָמַלְנוּ לְצִירֶיהָ. וַיֶּהֱמוּ רַחֲמֵינוּ לַאֲמָרֶיהָ. וְאָמַרְנוּ: תֵּרֵד אֵשׁ ה' עַל הָרָשָׁע הַהוּא וְעַל פִּגְרוֹ. וּבָרוּךְ שֶׁהִפִּילוֹ וְשֶׁעֲקָרוֹ. וְעַתָּה אַל תִּירְאִי כִּי רַע לֹא יְאֻנֶּה לָךְ. לְכִי אִתָּנוּ וְהֵטַבְנוּ לָךְ. וַתֹּאמַר: הִנְנִי אֲמַתְכֶם. לְשָׁרֵת אֶתְכֶם. בְּכָל תַּאֲוַתְכֶם. וְכַאֲשֶׁר הָלַכְנוּ יַחַד. וְלִבָּהּ שָׁלוֹם מִפַּחַד. אָמְרָה לָנוּ: דְּעוּ כִּי יֵשׁ אַחֲרֵי הָהָר הַזֶּה עֵינוֹת מַיִם. מְתוּקִים כְּנֹפֶת לַשְּׂפָתָיִם. וְאִם תָּשִׂימוּ מִשָּׁם דַּרְכְּכֶם. וְתִקְחוּ מַיִם לְצָרְכְּכֶם. תַּשְׂכִּילוּ בַּאֲשֶׁר תַּעֲשׂוּ כִּי הַדֶּרֶךְ רְחוֹקָה. וְהַשֶּׁמֶשׁ חֲזָקָה. וְהַנֶּפֶשׁ שׁוֹקֵקָה. וַיִּיטַב הַדָּבָר בְּעֵינֵינוּ וּבָאנוּ אֶל הַמַּעְיָן כְּחֹם הַצָּהֳרַיִם. וְהִנֵּה כָּל הָאָרֶץ פְּלָגִים יִבְלֵי מַיִם. וְשָׂמַח מְאֹד לִבֵּנוּ. וְשָׁקַט עָצְבֵּנוּ. וַתֹּאמֶר לָנוּ: אַתֶּם עֲיֵפִים מִן הַהֲלִיכָה וְהַיְגִיעָה. וְתֹקֶף הַחֹם וְהַנְּסִיעָה. וְאִם תִּשְׁכְּבוּ מִשְׁכַּב הַצָּהֳרַיִם. וְתִתְּנוּ מִן הַמְּנוּחָה חֵלֶק לָעֵינַיִם. עַד שֶׁהַיּוֹם יָפוּחַ. וְנָשְׁבָה הָרוּחַ. אָז תֵּלְכוּ בְשִׂמְחָה. בִּרְאוֹתְכֶם כִּי הָיְתָה הָרְוָחָה.

וַיִּיטַב הַדָּבָר בְּעֵינֵי חֲבוּרָתֵנוּ. וְיָרַדְנוּ מִמַּרְכְּבוֹתֵינוּ. וְשַׂמְנוּ סְבִיב הַמַּעְיָן מוֹשְׁבוֹתֵינוּ. וְגַם הִיא יָרְדָה מֵעַל סוּסָהּ וְהֵסִירָה מְעִילֶיהָ. וּשְׁבִיסֶיהָ וַעִגילֶיהָ. וְקָשְׁרָה סוּסָהּ וְסוּסֵינוּ. וְהֵכִינָה לָנוּ מְקוֹמוֹת כְּחֶפְצֵנוּ. וּבִנְאוֹת דֶּשֶׁא תַּרְבִּיצֵנוּ. וַנֹּאמֶר לָהּ: אַשְׁרֵיךְ יַעֲלַת הַחֵן כִּי אָסַפְתְּ עִם הַצּוּרָה הַהֲדוּרָה. נֶפֶשׁ יְקָרָה. וְהֶחְבַּרְתְּ עִם הַתְּמִימוּת. זְרִיזוּת וּנְעִימוּת. אָמְרָה: תְּמַהְתֶּם עַל נְעִימוּתִי. וְעוֹד לֹא רְאִיתֶם זְרִיזוּתִי. כִּי יֶשׁ-לִי כַמָּה מְלָאכוֹת עֲצוּמוֹת וַחֲמוּדוֹת. וְכַמָּה מִדּוֹת נִכְבָּדוֹת. וְעַתָּה אֲגַלֶּה לָכֶם קְצָתָם. וַתָּקָם לְהַרְאוֹת לָנוּ גְבוּרָתָהּ וִיכָלְתָּהּ. וְדָרְכָה קַשְׁתָּהּ. וְזָרְקָה חֵץ עַד לֵב הַשָּׁמַיִם. וּבְטֶרֶם נָפַל זָרְקָה אַחֲרָיו חֵץ שֵׁנִי וַיִּבְקָעֵהוּ לִשְׁנַיִם. וַנִּתְמַהּ עַל גְּבוּרָתָהּ. וְעַל זְרִיזוּתָהּ. אָמְרָה לָנוּ: וְעוֹד אַרְאֶה לָכֶם מְלָאכָה אַחֶרֶת. תִּהְיֶה לְדוֹר אַחֳרוֹן מַזְכֶּרֶת. וּבְיֶדְכֶם לְמִשְׁמֶרֶת. וַתִּקַּח אַשְׁפָּתָהּ בִּמְרוּצָה. וְרָכְבָה עַל סוּסָהּ. וְדָרְכָה קַשְׁתָּהּ. וְזָרְקָה חֵץ בְּלֵב אֶחָד מֵחֲבֵרֵינוּ. וְנָפַל אַרְצָה מֵת וָאֶצְעַק אֵלֶיהָ וְאָמַרְתִּי: אוֹי לָךְ מַה זֹּאת עָשִׂית. אָמְרָה לִי: הַחֲרֵשׁ בֶּן הַזּוֹנָה. נִפְתָּה לְבָבְךָ בַתְּמוּנָה. הַלְּבָנָהּ. וּבַשָּׂפָה. הַיָּפָה. וּבַמַּרְאֶה. הַנָּאֶה. וְהַרַקָּה. הַחֲשׁוּקָה. וְלִבְּבוּךָ הָעֵינַיִם. וְתָאֳמֵי הַשָּדַיִם. וְחֹלֶשׁ הַמָּתְנַיִם. וַחֲמוּקֵי הַיְרֵכַיִם. וְלִבְנַת הַיָּדַיִם וְהַשּׁוֹקַיִם. וְחָשַׁבְתָּ לַחֲבֹק וְלִנְשֹׁק. וְלָשׂוּם חָזֶה עַל חָזֶה וְשׁוֹק עַל שׁוֹק. חַי ה' תָּעִיתָ בְּחֶשְבּוֹנְךָ. וְנִפְתָּה לִבְּךָ וְרַעְיוֹנְךָ. וְאִם לֹא יִקְשֹׁר כָּל אִישׁ מִכֶּם אֶת חֲבֵרוֹ. אֶקְרַע אֶת סְגוֹרוֹ. וְאֶתֵּן לְעוֹף הַשָּׁמַיִם אֶת בְּשָׂרוֹ.

אָמַר הַמַּגִּיד: וּבִשְׁמֹעַ הָעָם אֶת הַדָּבָר הָרַע הַזֶּה וַיִּתְאַּבָּלוּ. וְנָפְלוּ עַל פְּנֵיהֶם אַרְצָה וְנִבְהָלוּ. וְנֶהֶפְכָה הַשִּׂמְחָה. אֲנָחָה. וְהָרִנָה. קִינָה. וְשָׁבָה הָאַהֲבָה. אֵיבָה. וְנֶהֱפַךְ הַחֵשֶׁק. לְמִלְחָמָה וּכְלֵי נֶשֶׁק. וְנוֹדַע לָנוּ מִכָּל אֲשֶׁר זָכַר. כִּי הוּא זָכָר. וְלֹא יָדַעְנוּ בְּצֵל מִי נֶחֱסֶה. וְלֹא עֵצָה מַה נַּעֲשֶׂה. כִּי לֹא הָיָה בָנוּ כֹחַ. לֹא לְהִלָּחֵם וְלֹא לִבְרֹחַ. כִּי הָיוּ סוּסֵינוּ עֲצוּרִים. וּכְלֵי מִלְחֲמוֹתֵינוּ קְשׁוּרִים. וְכָל אִישׁ מִמֶּנּוּ שׁוֹכֵב. וְהוּא עַל סוּסֹו רוֹכֵב. וְקַשְׁתּוֹ דְרוּכָה עַל יָד. וְאֵין לְאִישׁ לְפָנָיו מַעֲמָד. וְלֹא מָצָאנוּ עֵצָה כִּי אִם לַעֲשׂוֹת דְּבָרוֹ. קָשַׁר כָּל אֶחָד מִמֶּנּוּ חֲבֵרוֹ. עַד אֲשֶׁר נִשְׁאַרְתִּי לְבַדִּי. וְאֵין שֵׁנִי לַצִדִּי. וַיֵּרֶד מֵעַל סוּסוֹ וְהֶחֱזִיק בְּכַפִּי. וְהִכָּה עַל עָרְפִּי. וַיֹּאמֶר לִי: זֶרַע מְנָאֵף חָשַׁבְתָּ לִרְווֹת בְּצוּף הַשְּׂפָתַיִם. וְלַחֲבֹק הַזְּרוֹעַ וְהַלְּחָיַיִם. שַׁל נְעָלֶיךָ מֵעַל רַגְלֶיךָ. אָמַרְתִּי לוֹ: זֶה הַמִּנְעָל בְּעֵת לְבַשְׁתִּיהוּ הָיָה חָדָשׁ. וְעַתָּה נָס לֵחֹה וַיִּיבַשׁ. וְלֹא אוּכַל לִשְלֹף אוֹתוֹ. אָמַר לִי: אֲנִי אֶחֱלֹץ הַיּוֹם מִנְעָלְךָ. כִּי זֶה הַכָּבוֹד עָלָה בְגוֹרָלְךָ. וְכַאֲשֶׁר עָמַד אֶצְלִי לִשְׁלֹף נַעֲלִי הָיָה לִי בְתוֹךְ הֶמִּנְעָל מַאֲכֶלֶת. לְעֵין בָּרָק נִמְשֶׁלֶת. וּמְשַׁכְתִּיהָ בְּטֶרֶם יִרְאֶנָּה בְעֵינוֹ. וּתְקַעְתִּיהָ בְּבִטְנוֹ. וְנָפַל אַרְצָה עַל גְּחוֹנוֹ. וָאֶעֲמֹד עָלָיו וַאֲמוֹתְתֵהוּ. וְהֶחֱזַקְתִּי בְרֹאשׁוֹ וָאֶכְרְתֵהוּ. וּמִהַרְתִּי לְהַתִּיר חֲבֵרַי מִמַּאַסְרֵיהֶם. וְנִתַּקְתִּי הַקֶּשֶׁר אֲשֶׁר עַל יְדֵיהֶם. וְקָבַרְנוּ בֶעָפָר חֲבֵרֵנוּ. וְהָלַכְנוּ נֶעֱצָבִים בְּאֵידֵנוּ וְצִירֵינוּ. וְעָרַכְנוּ שֶׁבַח לְצוּרֵנוּ. אֲשֶׁר הִתִּיר מַאֲסָרֵנוּ.

וָאַעַן וָאֹמַר:

לוּלֵי אֲדֹנָי שֶׁעֲזָרָנוּ

כִּמְעַט שְׁאוֹן מָוֶת עֲבָרָנוּ.

בָּאנוּ בְיָם צָרוֹת וְלוּלֵי אֵל

פֶּתַע שְׁאוֹן מֵימָיו בְּלָעָנוּ.

אַךְ אַחֲרֵי בָאנוּ בְּפִי מָוֶת

צִוָּה אֱלֹהִים וֶהֱקִיאָנוּ.

נוֹדֶה לְשֵׁם הָאֵל אֲדוֹן עוֹלָם

עַל רֹב חֲסָדִים שֶׁגְּמָלָנוּ.
