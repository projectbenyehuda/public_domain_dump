פורים

וּבְאַרְבָּעָה עָשָׂר לְחֹדֶשׁ אֲדָר / בַּחוּרֵי יִשְׂרָאֵל לְכָבוֹד וּלְהָדָר / יִתְפָּאֲרוּ וְיִתְהַלָּלוּ / כִּי יִשְׁתַּגְּעוּ וְכִי יִתְהוֹלָלוּ. / זֶה זֵכֶר עָשָׂה לַנִּפְלָאוֹת וּפִרְסוּם לַנִּסִּים / כִּי יַעְטֶה כַמְעִיל תִּפְאֶרֶת הָעֲכָסִים / וְזֶה לִכְבוֹד מָרְדְּכַי וְאֶסְתֵּר / סְחִיף דִּקּוּלָא אַרֵישֵׁהּ וְכוֹפֶה עָלָיו פְּסַכְתֵּר. / זֶה יִלְבַּשׁ שִׂמְלַת אִשָּׁה וּלְגַרְגְּרוֹתָיו עֲנָקִים, / וְזֶה יִתְחַקֶּה כְּאַחַד הָרֵיקִים, / תֹּף וּמָחוֹל שִׂמְחָה וְשָׁלִישִׁים, / אֵלּוּ עִם אֵלּוּ אֲנָשִׁים עִם נָשִים. / וּלְעֵת עֶרֶב סְעוּדַת פּוּרִים / אַיָּל וּצְבִי וְיַחְמוּר וּבַרְבּוּרִים, / מִשְׁלוֹחַ מָנוֹת אִישׁ לְרֵעֵהוּ מִשִּׁבְעַת הַמִּינִים – / אֲבָל מַתָּנוֹת לָאֶבְיוֹנִים מִתְמַעֲטוֹת בְּכָל הַשָּׁנִים!

בַּלַּיְלָה הַהוּא יִתְּנוּ רוֹזְנִים לְאַיִן / הַבָּאִים לַחְקור מִמְסָךְ הַשּׁוֹתִים בְּמִזְרְקֵי יַיִן. / אָז יְרַנֲנוּ אוֹהֲבֵי אֲשִׁישֵׁי עֲנָבִים / יַזְהִירוּ כְּזֹהַר הָרָקִיעַ וְכַכּוֹכָבִים, / וְנִכְרַת זֵכֶר בֵּית הָרֵכָבִים. / בֵּהּ בְּלֵילְיָא יְחֻפַּשׂ בֵּית הַיַּיִן בַּנֵּרוֹת / וְנִשְּׁתוּ מַיִם מִן הַבָּתִים וּמִן הַחֲצֵרוֹת, / גְּזֵרָה הִיא מִלִּפְנֵי חֲמַת אֱלֹהֵי סְפַרְוַיִם: / בַּלַּיְלָה הַהוּא אַל תֹֹּאמְרוּ: “מַיִם,מַיִם!” / הַשּׁוֹתֶה הַיַּיִן חַי – אוֹר כַּשַּׂלְמָה עוֹטֶה, / וַאֲשֶׁר אַחַר הַמַּיִם נוֹטֶה – / כְּמַדְקְרוֹת חֶרֶב בּוֹטֶה. / רָאוּי לִבְרֹחַ מֵהֶם כְּמִי שֶׁנְּשָׁכוֹ כֶלֶב שׁוֹטֶה, / הַקָּרֵב הַקָּרֵב אֲלֵיהֶם דָּמוֹ בְרֹאשׁוֹ, / הַנּוֹגֵעַ בָּם הֲרֵי זֶה מִתְחַיֵּב בְּנַפְשׁוֹ. / לָכֵן הִתְקִינוּ הַקַּדְמוֹנִים / וְיָעֲצוּ חֲסִידִים הָרִאשׁוֹנִים / שֶׁמִּי שֶׁהִנִּיחַ לוֹ אָבִיו מָעוֹת / וְרוֹצֶה לְהַרְוִיחַ בָּם יוֹתֵר מִכָּל עֵסֶק וְקַרְקָעוֹת – / וְקָנָה מֵהֶם יַיִן חָשׁוּב וּמֻפְלֶא / בַּאֲדָר יַעֲשֶׂה וְהִצְלִיחַ דִּבְרִיא מַזְּלֵהּ. / כָּל אוֹתוֹ הַלַּיְלָה מְסַפְּרִים כְּבוֹד יַיִן וְשֵׁכָר / וְדוֹרְשִׁים מֵהֶם דָּרוֹשׁ וְקַבֵּל שָׂכָר.

בְּעִדָּנָא הַהִיא קָרְבוּ פְּקֻדּוֹת כָּל חָבִית וְקַנְקַן / מָלֵא וְרֵיקָן – / וְהָרֵיק לַחֲרָפוֹת מִנָּוֶה יְגֹרַש / וְהַמָּלֵא יַסִּיעוּ וְגַם דָּמוֹ הִנֵּה נִדְרָשׁ. / יִשְׁתּוּ וְיִמְצוּ / אֵין אוֹנֵס וּמְעַכֵּב אֲבָל לִכְשֶׁיִּרְצוּ, / אֲדֻמִּים שְׂרֻקִּים וּלְבָנִים, / חֲדָשִׁים גַּם יְשָׁנִים./ זֶה יִשְׁתֶּה עַד דִּי אַפּוֹהִי יִשְׁתַּנֵּא / וְזֶה עַד דִּנְפִיק קִיּוּהָא מֵאוּנֵהּ, / זֶה יִשְׁתֶּה בְגָבִיעַ גָּדוֹל יַיִן חַי וּמָלֵא / עַד שֶׁתֶּחְסַר הַצַּפַּחַת וְהַכַּד יִכְלֶה, / וְזֶה לְמִדַּת בִּטְנוֹ עַד דִּקְפֵי טַחֲלֵהּ. / כָּל פֶּתִי וְכָל חֲסַר מֹחַ / שׁוֹרֶה בְנֵי מֵעָיו בְּיַיִן, עַד אִם יִהְיֶה בַרְזֶל – יֵרַךְ, וְאִם אֶבֶן – הוּא נִמּוֹחַ. / וַאֲלֵיהֶם יִלָּווּ בְשָׂמִים וּזְנִים מְשֻׁבָּחִים / וּמִגְדְּלוֹת מֶרְקָחִים.

כְּלָלוֹ שֶׁל דָּבָר: / הַגִּבּוֹר – גִּבּוֹר לִשְׁתּוֹת יַיִן וּדְאַלִּים גְּבַר, / הַמִּשְׁתַּכֵּר – מִשְׁתַּכֵּר עַד שֶׁלֹּא יַכִּיר בֵּין נָבָל לְנָבָר / וּבֵין עָתִיד לְעָבָר, / עַד שֶׁלֹּא יֵדַע בֵּין הָמָן לִמְהוּמָן / וּבֵין “בָּרוּךְ מָרְדְּכַי” לְ“אָרוּר הָמָן”, / וַאֲשֶׁר יִשְׁכַּר בְּתוֹךְ הָאָהֳלוֹ / עַד שֶׁלֹּא יֵדַע בֵּין יְמִינוֹ לִשְׂמֹאלוֹ – / מַה טּוֹב חֶלְקוֹ וּמַה נָּעִים גּוֹרָלוֹ! / חֹק לְיִשְׂרָאֵל הוּא לְהִשְׁתַּכֵּר מֵאָדֹם כַּשָּׁנִים וּמִלָּבָן כְּצֶמֶר צַחַר / לֵיל אַרְבָּעָה עָשָׂר בַּאֲדָר עַד שֶׁיַּעֲלֶה עַמּוּד הַשַּׁחַר, / וּלְהָפִיק רְצוֹן נוֹרָא וְאָיֹם, / יִנָּתֵן גַּם מָחָר לַיְּהוּדִים לַעֲשׂוֹת כְּדָת הַיּוֹם! / כִּי כְדַאי הוּא הַחֹדֶשׁ / לְהוֹסִיף מֵחֹל עַל הַקֹּדֶשׁ /

וְאַף כִּי טוֹב בַּיּוֹם הַהוּא לְזַנֵּק כְּאַרְיֵה מִן הַבָּשָׁן / לִכְבוֹד הַיְּהוּדִים אֲשֶׁר בְּשׁוּשָׁן. /

אַשְׁרֶיךָ יִשְׂרָאֵל, מִי כָמוֹךָ וּמִי לְךָ דוֹמֶה / לְקַיֵּם מִצְוַת “מְחַיֵּב אֱנָשׁ לִבְּסוּמֵי”! /
