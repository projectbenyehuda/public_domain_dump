אני בעודי במאסרי (תוכחה)

אֲנִי בְּעוֹדִי בְמַאֲסָרִי / תִּבְכֶּה רוּחִי לְמוֹסֵרִי, / אֵין מְתֹם בִּבְשָׂרִי.

אָחֲזָה רְעָדָה מָתְנָי / בְּקָרְאִי בְּסוֹד חֶזְיוֹנָי, / כִּי אֱלֹהֵי מִשְׁפָּט אֲדֹנָי.

בְּעֵת קָדוֹשׁ וְנוֹרָא שְׁמוֹ / קָרָא מִשְּׁמֵי מְרוֹמוֹ / אֶל הָאָרֶץ לָדִין עִמּוֹ –

גָּדַל עֹל הַמּוּסָר, / וְעֹל כָּל רוּחַ יֶחְסָר, / אֵין שָׁלוֹם לְכָל בָּשָׂר.

דִּין נֶעֱרַךְ לְפָנָי / וְקוֹל קוֹרֵא בְאָזְנָי: / קָרְבוּ רִיבְכֶם, יֹאמַר אֲדֹנָי.

הֲגִי, רוּחִי, בְּחַיָּי: / הֲזֹאת אַחֲרִית מַאֲוַיָּי, / זְכֹר כִּי רוּחַ חַיָּי.

וְיָדַעְתָּ עֵת תְּשִׁיבֵנִי / עֵת בְּמַסְגֵּר תַּכְבִּישֵׁנִי, / עוֹר וּבָשָׂר תַּלְבִּישֵׁנִי.

זְדוֹן הַגֵּו נָשָׂאתִי, / כָּלוּא הֲלֹא יָצָאתִי / וּמְנוּחָה לֹא מָצָאתִי.

חַשְׁתָּ לְהַשְׁלִיכֵנִי, / בְּדָם וּבָשָׂר לְמָסְכֵנִי / וּבַעֲצָמוֹת וְגִידִים תְּשֹכְכֵנִי.

טֹרַפְתִּי בִּנְהִי וָבֶכִי; / אַל תְּדִינֵנִי כְּתוֹכִי, / כִּי עֵירֹם אָנֹכִי.

כֻּסִּיתִי בְּבוּז וָחֵפֶר / וְתִכְתֹּב עָלַי סֵפֶר – / וְאָנֹכִי עָפָר וָאֵפֶר.

לֹא דַיִּי אַנְחוֹתָי, / כִּי תִסְפֹּר שִׂיחוֹתָי / וְתִשְׁמוֹר כָּל אָרְחוֹתָי?

מְשָׁכַנִי הֶבְלִי, / רָזֶה חֶלְקִי בְּכָל עֲמָלִי / וְלֵילוֹת עָמָל מִנּוּ לִי.

נוֹרָא הוֹד בְּאוֹר חַמָּה / וּבַלְּבָנָה מָאוּס רוּמָהּ, / אַף כִּי אֱנוֹשׁ רִמָּה.

סֵתֶר אֱנוֹשׁ אַל תַּחֲשֹׂף, / כִּי אַחֲרִית סוֹף / לְמַעֲשֵׂה יָדֶיךָ תִכְסֹף.

עֶלְיוֹן בִּגְאוֹן עָצְמוֹ / יָשִׁיב רוּחַ לְגָלְמוֹ / וְעֶצֶם אֶל עַצְמוֹ.

פְּעֻלָּתִי לְעֻמָּתוֹ / יַעֲרֹךְ אָז בְּעָצְמָתוֹ – / וְלֹא יָעִיר כָּל חֲמָתוֹ.

צֶוַח פְּנֵי אָיוֹם / אֵין כֹּפֶר וְאֵין פִּדְיוֹם – / הֵילִילוּ הָהּ לַיּוֹם!

קָרַב יוֹם אָנוּשׁ וְעָצְבּוֹ – / וּמִמּוֹצָא סֵתֶר וּמַחֲבוֹא / סֻגַּר כָּל בַּיִת מִבּוֹא.

רָחוּם, רַחֲמִים שְׁלָחָה / וְדַלְתֵי רַחֲמִים פְּתָחָה, / יְיָ שְׁמָעָה יְיָ סְלָחָה!

שְׁלַח אוֹרְךָ, אֲדוֹנֵנוּ, / תַּחְשֹׂךְ לְמַטָּה מֵעֲוֹנֵינוּ, / כִּי עָלֶיךָ עֵינֵינוּ.
