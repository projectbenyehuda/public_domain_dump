משל קי: נְמָלָה וְעַכְבָּר

כְּמוֹ אֵשׁ עַל עֵץ רַעֲנָן / לַפִּיד בּוּז לְעַשְׁתּוֹת שַׁאֲנָן

הַנְּמָלָה עַם לֹא עָז / הוֹצִיאָה עַל הָעַכְבָּר לַעַז / כִּי אָגְרָה בַקַּיִץ לַחְמָהּ / וַתָּבֹא בִקְצֵה הָעֲרֵמָה / בִּתְחִלַּת קְצִיר שְׂעֹרִים / לְלַקֵּט בֵּין הָעֳמָרִים / וַתַּחְמֹל עַל נַפְשָׁהּ בַּעֲמָלָהּ / לְהָכִין בַּקָּצִיר מַאֲכָלָהּ / וַתֵּרֶא שָׁמָּה עַכְבָּרִים רַבִּים / חִטִּים וּשְׂעֹרִים מַרְבִּים / וַיָּמֹדוּ שֵׁשׁ שְׂעֹרִים לָהֶם / לְהָבִיא אֶל חוֹרֵיהֶם / עַד אֲשֶׁר הוֹתִירוּ מִשָּׂבְעָם / וְהַנְּמָלָה מְלֵאָה זָעַם / וַתֹּאמֶר לָהֶם דִּבְרֵי רִיבוֹת / וַתִּקְרָא לְמוֹתָם חַרְצֻבּוֹת / לַעֲשׂוֹת לִי כֹה לָמָּה הִסְכַּנְתֶּם / מִפְּנֵי אַפִּי וַחֲמָתִי לֹא יְגָרְתֶּם / כִּי אֲנִי הַנְּמָלָה / אֲנִי קָצִין לִי שִׂמְלָה / אֲנִי מֹשֵׁל אֲנִי שֹׁטֵר / אֲנִי נֹקֵם אֲנִי נֹטֵר / וְאֵין בִּי וְעָלַי מֶלֶךְ / מִכָּל שֶׁרֶץ עַל גָּחוֹן הֹלֵךְ / לָמָּה הִרְגַּזְתֶּם אֹתִי / לָקַחַת מִגָּרְנִי וּמְדֻשָּׁתִי / וַיַּעַן הָעַכְבָּר / לִי הַגֹּרֶן לִי הַבָּר / לָמָּה הוֹצֵאת עָלַי דִּבָּה / וַתְּהִי לִי לְחָרְבָּה / הֲלֹא בְזוּיָה אַתְּ לְנֶגְדִּי / וְאֵיךְ לְהִלָּחֵם בִּי לָמֹד תִּלְמְדִי / שְׁבִי וְקוּמִי מֵעָפָר הִתְנַעֲרִי / וְגוּרִי בַאֲשֶׁר תָּגוּרִי / וְלֹא תִדְרְכִי עֹז נַפְשִׁי / אָכַלְתִּי יַעְרִי עִם דִּבְשִׁי / וְהַמְּלָאכָה הָיְתָה דַיָּם / מִי יַעֲבֹר לָנוּ אֶת הַיָּם / וְהַנְּמָלָה תָשִׁיב אֲמָרֶיהָ / מִיָּם וְעַד יָם קְצִירֶיהָ / לְשַׁלַּח צְעִירֶיהָ בְאַדִּירֵיהֶם / בְּעַכְבָּרִים לְמִשְׁפְּחוֹתֵיהֶם / וּבַעֲמָלָם אֲשֶׁר עָמְלוּ / בִּימֵי שֶׁלֶג מַיִם יִגְזְלוּ / וַתָּשָׁב אֶל חַבְרוֹתֶיהָ / לְסַפֵּר לָהֶן כָּל קוֹרוֹתֶיהָ / וַתֹּאמַר: יָגַעְתִּי וּמָצָאתִי / וְאָכַלְתִּי וְטָעַמְתִּי וְשָׂבַעְתִּי / מַאֲכָל וְלֶחֶם מַטְעַמִּים / עַל הָרֵי בְשָׂמִים / וְהָעַכְבָּרִים הָרָעִים / וְהַשְּׁרָצִים הָרְשָׁעִים / בָּאוּ עֲדֵי נַפְשִׁי / וּמֵהֶם לֹא אֶשָּׂא רֹאשִׁי / וַיִּלְקְטוּ בַשִּׁבֳּלִים וּבַמַּאֲכָלִים / וַיַּחְבְּלוּ כְרָמֵינוּ כִּקְטַנִּים שֻׁעָלִים / צְאֶינָה אִתִּי לִקְרָאתָם / בִּקְרָב יָדַעְתִּי תַחֲנוֹתָם / וַיִּשְׁמְעוּ אֵלֶיהָ הַנְּמָלִים /וַיָּבֹאוּ אִתָּן הַחֲתוּלִים / לִהְיוֹת עִמָּן בְּעֶזְרָתָן / לֶאֱרֹב דָּמָם בְּמַחֲבוֹאָתָן / וַיָּבֹאוּ עַד מְקוֹמָן / וַיַּחֲנוּ שָׁם עַד תֻּמָּן / וַיִּכָּנְסוּ בְחוֹרֵיהֶן הַנְּמָלִים / וַיַּכּוּ פַעֲמֵיהֶם בַּנְּעָלִים / וַיַּעֲלוּ עַל גַּב הָעַכְבָּרִים / כַּאֲשֶׁר תַּעֲשֶׂינָה הַדְּבוֹרִים / עָקְצוּ בִּבְשָׂרָם וּבְנַפְשָׁם / עַד אֲשֶׁר יָצְאוּ מִמִּגְרָשָׁם / וַיֹּצִיאוּ הַחִטִּים וְהַשְּׂעֹרִים / וַיִּצְבְּרוּ אוֹתָם חֳמָרִים חֳמָרִים / וּמִן הָעַכְבָּרִים מִלְּאוּ הַחֲתוּלִים בִּטְנָם / וַיַּעֲשׂוּ בְשֹׂנְאֵיהֶם כִרְצוֹנָם / וְהַנְּמָלִים עַם לֹא עָז וְנִבְזֶה / הִשְׁמִידוּם וַיֵּשְׁבוּ תַחְתָּם עַד הַיּוֹם הַזֶּה.


וְהַמָּשָׁל בְּאִישׁ עָנִי וְנִכְאֶה / אֵצֶל אִישׁ עָשִׁיר גֵּאֶה / וְתַחֲנוּנִים יְדַבֵּר לוֹ רָשׁ / כִּי בְאַהֲבָתוֹ דָרֹשׁ דָּרַשׁ / וְהֶעָשִׁיר יַעֲנֶה עַזּוֹת / מִי שָׁמַע כָּזֹאת / לְבַקֵּשׁ עָלָיו עוֹלֵלוֹת / לְהַמְצִיא לוֹ עֲלִילוֹת / לְגָרְשֵׁהוּ מִמִּחְיָתוֹ / וְתַעַל עָלָיו צַחֲנָתוֹ / וּכְמוֹ אֵשׁ עַל עֵץ רַעֲנָן / לַפִּיד בּוּז לְעַשְׁתּוֹת שַׁאֲנָן / נָכוֹן לְמוֹעֵד רַגְלָיו / כֵּן הֶעָשִׁיר בְּמִכְשֹׁלָיו / וְרָדַף אֹתוֹ כָּל עָלֶה נִדָּף / וְהָאֱלֹהִים יְבַקֵּשׁ אֶת נִרְדָּף / וָאֶשָּׂא מְשָׁלִי / וָאֹמַר בְּמַעֲנִי / אֱלֹהִים שֹׁפֵט צֶדֶק מוֹרִישׁ אֶת הֶעָשִׁיר / וְהֶעָנִי מַעֲשִׁיר.
