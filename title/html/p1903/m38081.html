משל כ: נֶשֶׁר, חֹמֶט וְעֹרֵב

הַחֲכָמִים בְּעָרְמָתָם וְסוֹדָם / יִתְחַכְּמוּ עַל בְּנֵי אָדָם


חֹמֶט הָלַךְ לְפִי תֻמּוֹ / וּבְכָל אֲשֶׁר יֵלֵךְ בֵּיתוֹ עִמּוֹ / כִּי הוּא נֹשְׂאוֹ עַל גַבּוֹ / וְהִנֵּה יָצָא רֹאשׁוֹ וְרֻבּוֹ / קַרְנָיו זְקוּפוֹת בְּתוֹצְאוֹתָיו / מִמֶּנּוּ הָיוּ קַרְנוֹתָיו / וְהִנֵּה הַנֶּשֶׁר מְרַחֵף לְפָנָיו / וַיִּתְבּוֹנֵן לוֹ וַיַּשְׁפִּיל קַרְנָיו / וְלִבּוֹ מֵחֶרְדָּתוֹ יִתָּר / עָרוֹם רָאָה רָעָה וְנִסְתָּר / וַיִּתְחַבֵּא מַהֵר בְּאָהֳלוֹ / בַּעֲלִיַּת הַמְּקֵרָה אֲשֶׁר לוֹ / וַיַּרְא הַנֶּשֶׁר כִּי נִסְתַּר וְנֶחְבָּא / וְלֹא נִרְאָה דֶרֶךְ הַקֻּבָּה / וַיָּבֹא עַד הָאֹהֶל וַיַּהַפְכֵהוּ לְמַעְלָה / וַיָּבֶן כִּי שָׁם נֶחְבָּא אֲשֶׁר רָאָה בַתְּחִלָּה / וְכִי שָׁמָּה תַחֲנָתוֹ / יִשָּׂאֵהוּ עַל אֶבְרָתוֹ / וַיֹּאמַר כִּי לֹא יִמָּלֵט מִקִּרְבּוֹ / עַד עֲשׂוֹתוֹ וְעַד הֲקִימוֹ מְזִמּוֹת לִבּוֹ / וְכַאֲשֶׁר נְשָׂאוֹ בֵּין הַשָּׁמַיִם וּבֵין הָאָרֶץ / וַיְנַסֶּה לִפְרֹץ בּוֹ פָרֶץ / וּלְשַׁבְּרוֹ בְּפִיו וּבְרַגְלָיו / וַיַּרְא כִּי לֹא יָכֹל אֵלָיו / וְהִנֵּה נִצָּב לִקְרָאתוֹ הָעֹרֵב / וַיֵּלֶך הָלֹךְ וְקָרֵב / וַיִּקְרָא לַנֶּשֶׁר וַיֹּאמַר: בָּגַד בְּךָ הַשֵּׂכֶל / הִנְּךָ רֹאֶה בְעַינֶיךָ וּמִשָּׁם לֹא תֹאכַל / וּמַה תָּעִיר עַל זֶה חֵמָה / לֹא לַגִּבּוֹרִים הַמִּלְחָמָה / כִּי מִי פִתֵּחַ דַּלְתֵי פָנָיו / גַּאֲוָה אֲפִיקֵי מָגִנָּיו / יָצוּק כְּמוֹ אֶבֶן גַּבּוֹ / הָעֹשׂוֹ יַגֵּשׁ חַרְבּוֹ / וְלֹא לְךָ חָכְמַת הַשְּׁבִירָה / מֵעוֹף הַשָּׁמַיִם נִסְתָּרָה / רַק מִמֶּנִּי עַתָּה בָאתִי / בְּחָכְמָתִי כִּי נְבוּנוֹתִי / וְהַשָּׁלָל מְאֹד נֶחְמָד / אֲשֶׁר אֵלֶיךָ נִצְמָד / וְאִם תִּתֵּן בּוֹ לִי חֵלֶק וָחֶבֶל / תִּשְׁבְּרֶנּוּ כְּשֵׁבֶר נֵבֶל / וְתִרְאֶה הַנִּסְתָּר בְּעָרְמָתוֹ / אָז תָּבִין נְתִיבוֹת בֵּיתוֹ / וַיַּעַן הַנֶּשֶׁר: אַחֲרֵי אֲשֶׁר רֹאשׁוֹ מָחַצְתָּ / כֵּן מִשְׁפָּטְךָ אַתָּה חָרַצְתָּ / וּלְהָסִיר מִכְסֵהוּ הוֹרֵנִי / חֵלֶק כְּחֵלֶק לְךָ כָמוֹנִי / וַיֹּאמֶר הָעֹרֵב: לֵךְ עַל שֶׁן סֶלַע וּמְצוּדָה / וּתְנַפְּצֵהוּ מִשָּׁם בִּירִידָה / וּתְקַלְּעֶנּוּ אַרְצָה כִּבְקֶלַע / וּתְנַפְּצֵהוּ אֶל הַסֶּלַע / וְהָיָה כְאָבָק דַּק מִכְסֵהוּ / וְהַבָּשָׂר שָׁם תֹּאכְלֵהוּ / וְהַנֶּשֶׁר שָׁמַע לָעֵצָה / וַיַּשְׁלִיכֵהוּ מִן הַסֶּלַע אַרְצָה / וְטֶרֶם מִיָּדוֹ הָשְׁלַךְ / עֹרֵב בְּתַחְתִּית הַסֶּלַע הָלַךְ / וַיִּמְצָא הַחֹמֶט מְפֻזָּר וּמְפֹרָד / וַיֹּאכְלוֹ וְלֹא פָלַט וְלֹא שָׂרַד / וְלַנֶּשֶׁר צַר מְאֹד אֲשֶׁר שָׁמַע בְּסוֹדוֹ / כִּי יָכִין לָעֹרֵב צֵידוֹ / וּשְׁלָלוֹ בְּעָרְמָתוֹ יֶרֶף / וַיַּשְׁלֵךְ מִשִּׁנָּיו טֶרֶף.

וְהַמָּשָׁל מְאֹד יֶעֱרָב / טוֹבָה חָכְמָה מִכְּלֵי קְרָב / וְהָעֲרוּמִים בְּחָכְמָתָם וּבְסוֹדָם / יַשְׁחִיתוּ אֲשֶׁר בָּנוּ בְּנֵי אָדָם / אַף כִּי יִבְטַח אָדָם בְּמִבְצָרָיו אֲשֶׁר עָשָׂה וּבְמִגְדָּלָיו / אִם שׂוֹנְאָיו יַעֲרִימוּ סוֹד עָלָיו / וִיְהִי בְלִבּוֹ מַסְלוּל חָכְמָה מְיַשֵּׁר / פֶּן יִתְחַכְּמוּ לוֹ כְּחָכְמַת הָעֹרֵב לַנֶּשֶׁר.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


בְּהוֹן יָרִיב אֱנוֹשׁ רֵעוֹ וְהוֹדְפוֹ / וְעָבַר זָר וְאֹסְפוֹ בִכְנָפוֹ

כְּמוֹ עֶצֶם יְרִיבוּן בּוֹ שְׁנַיִם / כְּלָבִים, בָּא שְׁלִישִׁי וַחֲטָפוֹ.
