משל קיח: בָּחוּר וְרֵיקִים וְאִשָּׁה וְשֹׁפֵט

אֲשֶׁר שֵׁבֶט יְבַקֵּשׁ לְגֵו אָחִיו בְּשִׂטְמָה / הוּא נִקְלֶה בְשִׁבְטוֹ וְיַעֲטֶה כְלִמָּה

וַיָּבֹא הֵלֶךְ לְאִישׁ עָשִׁיר / יֹדֵעַ לְנַגֵּן בְּכָל כְּלֵי שִׁיר / הֶעָשִׁיר הָיָה יְפֵה תֹאַר / כַאֲשֶׁר בַּמְּחוּגָה מְתֹאָר / וַיִּיף בְּגָדְלוֹ בְּאֹרֶךְ דָּלִיּוֹתָיו / וּבָחוּר מֵעָם הֱרִימוּהוּ אֲבוֹתָיו / וַיִּשְׁאַל לְאִישׁ הֵלֶךְ / מֵאַיִן בָּאתָ וּלְאָן אַתָּה הֹלֵךְ / וַיֹּאמַר: מֵאֶרֶץ רְחוֹקָה בָאתִי / וְשָׂרִים רֹכְבִים עַל סוּסִים רָאִיתִי / וַיִּיטַב בְּעֵינַי אֶרֶץ וְטוֹב סַחֲרָהּ מִסְּחַר כָּסֶף / לֹא יֶחְסַר בָּהּ כָּסֶף / וַיֹּאמֶר לוֹ: מָה הָעִיר וּמַה שְּׁמָהּ / מִיּוֹם הֱיוֹתְךָ שָׁמָּה / וַיֹּאמֶר לוֹ: מֵעֵבֶר לַאֲרַם נַהֲרַיִם / אֲלֶכְּסַנְדְרִיָּה שֶׁל מִצְרַיִם / וַיֹּאמַר: כָּךְ שָׁמַעְתִּי אֹמְרִים / אֶסְפָה לִּי חֲבֵרִים / וְאֵלְכָה לִי שָׁם בִּסְחוֹרָה / בְּבֶגֶד עִזִּים וּבְאֶבֶן יְקָרָה / וּבְבִגְדֵּי צֶבַע תּוֹלַעַת שָׁנִי / אֵלְכָה מִצְרַיִם בַּיָּם וּבָאֳנִי / וּתְכֵלֶת וְאַרְגָּמָן וּבְגָדִים וּכְלֵי יְקָר / עַד מִשְׁקָל לֹא נֶחְקָר / וְהַכֹּל הֵכִין לֹא חָסֵר דָּבָר / וְהִנֵּה הַסְּתָו עָבַר / וַיְמַלֵּא אֶת הָאֳנִיָּה / עַד אֲשֶׁר הוֹתֵר וְדַיָּהּ / וַיָּשֶׂם בָּהּ צִנָּה וּמָגֵן וּרְמָחִים / וַיִּסַּע עִם הַמַּלָּחִים / וַתִּשָּׂאֵם רוּחַ קָדִים / וּבָאוּ לָאָרֶץ שָׂשִׂים וְחָדִים / וַיָבֹאוּ עַד לִפְנֵי אֲלֶכְּסַנְדְרִיָּה / וַתֵּהֹם הַקִּרְיָה / לִרְאוֹת צִי אַדִּיר / כִּי לֹא נִמְצָא לְהַאְדִּיר / לְחוֹף יַמִּים וּלְחוֹף אֲנִיּוֹת / כָּל כֹּחַ נֶגְדָּהּ דַּיּוֹת / וְהַבָּחוּר מִן הַסְּפִינָה יָצָא / וַיִּקַּח בְּתוֹכָהּ פֶּרֶד אֲשֶׁר מָצָא / וַיִּרְכַּב עָלָיו וַיֵּלֶךְ לְהִתְאַכְסֵן / בַּעֲדִי בְּמֶתֶג וּבְרֶסֶן / וַיֵּצֵא הַקּוֹל בָּרְחוֹבוֹת וּבַשְּׁוָקִים / וַיִּתְלַקְּטוּ אֲנָשִׁים רֵיקִים / וַיִּקְשְׁרוּ קֶשֶׁר / לִרְאוֹת הַסְּפִינָה וְהָעֹשֶׁר / יַעַן הִתְחַבְּרוּת אֵלָיו / לְהִתְגּוֹלֵל וּלְהִתְנַפֵּל עָלָיו / וּלְבַקֵּשׁ תֹּאֲנָה וּמְזִמָּה / כִּי הֵם אַנְשֵׁי מִרְמָה / וְהַבָּחוּר הָלַךְ אָנֶה וְאָנָה / לִרְאוֹת אֶת הָעִיר וְאֶת הַמְּדִינָה / וַיִּקֶר אִישׁ בָּא מִבּוֹא / וַיֹּאמֶר לוֹ: אֵלַי תָבוֹא / הַצּוּר לְפָנַי אוֹתְךָ הִקְרָה / כִּי שְׁנֵינוּ בַעֲלֵי סְחוֹרָה / וְאֶקְנֶה כָּל סְחוֹרָתְךָ אִם תֹּאבֶה / כִּי יֵשׁ אֳנִי תַרְשִׁישׁ עִם אֳנִיּוֹת אֵבֶה / וַיַּעַן הַבָּחוּר: אִם יֵשׁ לְךָ רְוָחָה / אֶמְכֹּר לְךָ הַכֹּל בְּשִׂמְחָה / וְאִם רְצוֹנְךָ בְּמִקְנָה / מַה תִּתֵּן מִכֹּל אֲשֶׁר בַּסְּפִינָה / וַיֹּאמַר: כַּאֲשֶׁר תֹּאמַר וְכַאֲשֶׁר תִּבְחָר / אֲמַלֵּא אֹתָהּ מִכָּל מִסְחָר / וַיֹּאמַר: כִּדְבָרְךָ לְפָנֶיךָ הִנֵּה / בֹּא אֶל הַסְּפִינָה מְשֹׁךְ וּקְנֵה / וַיִּקֶּן הַסְּחוֹרָה מֵאִתּוֹ / וַיִּפָּרְדוּ אִישׁ לְבֵיתוֹ / וַיְהִי בַבֹּקֶר לְמָחֳרָתוֹ / וַיֵּלֶךְ לִרְאוֹת אֶת סְפִינָתוֹ / וַיִּרְאֵהוּ אִישׁ לְפָנָיו / וַחֲגֹרָתוֹ אֲשֶׁר בְּמָתְנָיו / וְהַסַּכִּין אֲשֶׁר בְּצִדּוֹ / חָגוּר עֲלֵי מַדּוֹ / וַיֹּאמֶר לוֹ: סוּרָה אֵלַי סוּרָה / כִּי הַסַּכִּין אֲשֶׁר בַּחֲגֹרָה / גָּנַבְתָּ מִמֶּנִּי / וְאֶת כָּל מְמוֹנִי / וַיּוּשַׂם בְּעֶרְכּוֹת לִוָּעֵד / לִפְנֵי הַשֹּׁפֵט לְיוֹם מוֹעֵד / וְהַבָּחוּר הָיָה עָנָיו / וַיַּרְא וַיִּפְּלוּ פָנָיו / וַיְהִי בַיּוֹם הַשְּׁלִישִׁי בִּהְיוֹת הַבֹּקֶר / קָם הַבָּחוּר לִהְיוֹת חֹקֵר / עֵזֶר כִּי גָזַלְתָּ עֵינִי / וְאָשִׂים נַפְשִׁי נֶגֶד נַפְשְׁךָ / כִּי זֶה עֵינִי אֲשֶׁר בְּרֹאשֶׁךָ / וְלַשֹּׁפֵט מָצָא עָרָב / לִהְיוֹת נָכוֹן לְיוֹם קְרָב / וְלֹא יֶהְדַּר פְּנֵי דַלִּים / וְנָתֹן וַתִּמְצָאֵהוּ זְקֵנָה / וַתֹּאמַר: מַדּוּעַ פָּנֶיךָ זֹעֲפִים / אֲשֶׁר הָיוּ יָפִים / וַיְסַפֵּר לָהּ כָּל הַקֹּרוֹת / וַיִּתְחַנֵּן לָהּ לְהוֹרוֹת / מַה יַּעֲשֶׂה / וַתֹּאמַר: לֹא אֶשְׁאַל וְלֹא נֶאֱמָנִים / אֲשֶׁר הַסַּכִּין הִכִּירוּ / פִּסְקוּ דִינָיו וּגְמֹרוּ / וִישַׁלֵּם לִי מִבֵּיתוֹ / וְאִם אֵין לוֹ וְנִמְכַּר בִּגְנֵבָתוֹ / וַיָּקָם לִפְנֵי הַשֹּׁפֵט וַיַּעַן / אֲדֹנִי, בָּאתִי מֵאֶרֶץ כְּנַעַן / וּמִצָּרְפַת וּמִסְּפָרָד / לְזֹאת לִבִּי לֹא חָרַד / וְהַדֶּרֶךְ לְנֶגְדִּי לֹא יָרַט / עַד בֹּאִי אֶל אֶרֶץ אֲרָרָט / וּמִשָּׁם הָלַכְתִּי לְאַסְפַּמִיָּה / וְעַתָּה בָּאתִי לַאֲלֶכְּסַנְדְרִיָּה / לְגַלּוֹת הַסַּכִּין וּלְהַרְאוֹת / הֲנִמְצָא אִישׁ אֲשֶׁר יֵאוֹת / לְהוֹרוֹת בּוֹ וּלְהַכִּירוֹ / יִתְבָּרַךְ הַצּוּר וְזִכְרוֹ / אֲשֶׁר הִצְלִיחַ נְתִיבָתִי / יָגַעְתִּי וּמָצָאתִי / הֵן בְּסַכִּין זֶה נֶהֱרַג אָבִי / וַיִּגְדַּל כְּאֵבִי / וְגַם נִמְצָא בְּבִטְנוֹ / אֲדֹנִי הַשֹּׁפֵט הִנּוֹ / בְּיָדְךָ רַק אֶת נַפְשׁוֹ שְׁמוֹר / עַד אֶרְכַּב עַל הַחֲמוֹר / וְאָבִיא עֵדִים בַּדָּבָר / כִּי לֹא יִהְיֶה נָבָר / וְהַשְּׁלִישִׁי קָם בַּחֲרוֹן אַף / לַהֲרֹג הַבָּחוּר שָׁאַף / וַיְדַבֵּר עָלָיו גְּבוֹהוֹת / בְּעֵינֵי כֹל הָיוּ תְמוּהוֹת / אֲדֹנִי זֶה הָאִישׁ / בָּא לִטְרֹף אֹתִי כְלַיִשׁ / וְהִגְדִּיל עָלַי בִּגְבוּרָה / וְעֵין יְמִינִי נִקְּרָהּ / וְשָׂם עֵינִי בְרֹאשׁוֹ / בִּכְשָׁפָיו וְלַחֲשׁוֹ / וְעַתָּה אִם לֹא יוֹדֶה / בְּפֶרֶךְ בּוֹ אֶרְדֶּה / וְאֶלְחַם בּוֹ וַאֲנַצְּחוֹ / וְאֶקְלַע אֶבֶן בְּמִצְחוֹ / אַךְ גְּמֹר דִּינִי / וִישַׁלֵּם לִי עֵינִי / וְצַעַר וְרִפּוּי וְשֶׁבֶת וּבֹשֶׁת וְנֶזֶק / כִּי בָא אֵלַי בְּחֹזֶק / וַיֹּאמֶר אֶל הַשֹּׁפֵט: בִּי אֲדֹנִי / יָדַעְתָּ מֵאֶרֶץ רְחוֹקָה אָנִי / וְרָחֹק מֵאָב וְאָח וּבָנִים / שְׁמַע נָא אִם דְּבָרַי כֵּנִים / וְאִם מִשְׁפַּט אַרְצִי יָשָׁר / וּבְעֵינֶיךָ יִכְשָׁר / נַקֵּר נָא אֶת עֵין שְׂמֹאלוֹ / וְגַם יְמִינִי לְפִי דְבָרָיו שֶׁהָיָה שֶׁלּוֹ / תָּשִׂים בְּמֹאזְנַיִם / אֶת שְׁתֵּי הָעֵינַיִם / אִם שׁוֹקְלוֹת בְּשָׁוֶה / מִדְבָּרוֹ נָאוֶה / וְאִם תִּכְבַּד זוּ עַל זוּ / כְּדִין רָאוּי לֹא יֶחֱזוּ / לַעֲשׂוֹת בּוֹ מִשְׁפָּט / לְעֵין כֹּל בְּעֵמֶק יְהוֹשָׁפָט / וְעִמִּי לֹא יִלְחַם / אַךְ מוֹדֶה וְעֹזֵב יְרֻחָם / וַיְנֻצְּחוּ הָרַמָּאִים / וַיִּשָׁאֲרוּ חֵלְכָאִים / וַיִּתְנוּ לוֹ כָל הוֹנָם / וַיִּתַּשׁ כֹּחָם וְאוֹנָם.

מְשָׁלִי זֶה הֲלֹא נִמְשַׁל וְנִדְמָה / לְבֶן אָדָם מְלֵא אָוֶן וּמִרְמָה

אֲשֶׁר בִּקֵּשׁ עֲשׂוֹת אָוֶן וְעָוֶל / לְאִישׁ תָּמִים, פְּעֹל תּוֹךְ וּמְזִמָּה

וּבָא עִתּוֹ לְשַׁלֶּם לוֹ גְמוּלוֹ / וְנִתְמַלְּאָה סְאָתוֹ גַּם שְׁלֵמָה

בְּתוֹךְ גוּמָץ אֲשֶׁר חָפַר וְכָרָה / הֲלֹא יִפֹּל בְּרֹגֶז וּמְהוּמָה

וְיֵשׁ גֶּבֶר אֲשֶׁר שֵׁבֶט יְבַקֵּשׁ / לְגֵו אָחִיו בְּזָדוֹן וּשְׂטִימָה

וּבָא הָעֵת וְהוּא נִקְלֶה בְּשִׁבְטוֹ / אֲשֶׁר בִּקֵּשׁ וְיַעְטֶה בוֹ כְלִמָּה

וְיֵשׁ פֹּרֵץ גְּדֵרוֹת לִשְׁכֵנָיו / עֲשׂוֹת בָּהֶם חֲרָפוֹת גַּם נְשָׁמָה

וּבָא יוֹמוֹ וְהַנָּחָשׁ נְשָׁכוֹ / בְּתוֹךְ פִּרְצוֹ וְעָשָׂה בוֹ נְקָמָה

תְּנוּ לֵב לִמְשָׁלִי זֶה בְּנֵי אִישׁ / וְאַל תַּרְבּוּ עֲלֵיכֶם רִיק וְאַשְׁמָה

כְּמוֹ עָשׂוּ שְׁלֹשָׁה בַּכְּתוּבִים / וְנֶהְפַּךְ הוּא וְנִשְׁאָרוּ בְלִי מָה.


לְכוּ מְשָׁלַי לְאִישׁ תְּבוּנוֹת / חֲזוֹת אֲמָרַי הֱיוֹת נְכוֹנוֹת

וְגַם יְשָׁרוֹת כְּדֵי לְאַסֵּף / בְּתוֹךְ צְרוּפוֹת וְאַף בְּחוּנוֹת

וְאִם אֱמֶת הוּא אֲשֶׁר בְּעֵינָיו / מְצָאתֶם חֵן בְּלִי תְּלֻנּוֹת

אֲזַי אֲחַלֶּה פְּנֵי יְבֹרָךְ / לְהַרְבִּיצְכֶם בְּתוֹךְ מְעוֹנוֹת

וְגַם כְּמוֹ כֵן יְחַו וְיַגִּיד / הֲיֵשׁ בְּתוֹכְכֶם מְאוּם זְדוֹנוֹת

וְיַצְפִּין אֶת כְּתָב יְדֵיכֶם / לְבַל רְאוֹת בָּם זְכֹר עֲווֹנוֹת.
