משל קטו: זָקֵן וּבָנָיו וְהֶגְמוֹן וַאֲנָשָׁיו

הַמְכַבֵּד אָבִיו וַאֲפִלּוּ בִשְׁתִיָּה / לֹא יִהְיֶה שְׂכָרוֹ רְמִיָּה

אִישׁ אֶחָד בָּאָרֶץ הָיָה / זָקֵן וְעָשִׁיר מִכָּל בִּרְיָה / אֲשֶׁר בַּכֹּל תְּכוּנָתוֹ / וּבָנָיו הָיוּ בְּבֵיתוֹ / וַיִּקְרָא לָהֶם לְצַוּוֹת / שִׁמְעוּ בָנִים מִצְווֹת / אֲבִיכֶם כִּי זָקַנְתִּי / וְלֹא יָדַעְתִּי יוֹם מוֹתִי / הַאֲזִינוּ נָא אֵלַי רוֹזְנִים / וְשִׁמְעוּ אִם דְּבָרַי כֵּנִים / הַקְשִׁיבוּ לִי כָּל בָּנַי / מָמוֹן יֵשׁ לִי בְמַטְמֹנַי / וּקְחוּ נָא הַכֹּל לָכֶם / וְחַלְּקוּ אֹתוֹ בֵינֵיכֶם / וְלֹא תַעַזְבוּ אִתִּי מְאוּמָה / רַק הַיַּיִן תְּנוּ לִי מִיָּמִים יָמִימָה / וְהַזָּקֵן צִוֻּיוֹ גָמַר / וַיַּעֲשׂוּ לוֹ בָנָיו כַּאֲשֶׁר אָמַר / וְיַיִין בְּכָל יוֹם לוֹ מָצְאוּ / הַרְבֵּה שָׁנִים עַד כִּי נִלְאוּ / וַיִּקְחוּ עֵצָה בֵינֵיהֶם / וַיְשַׁכְּרוּ אֶת אֲבִיהֶם / מִיַּיִן לָבָן וַאֲדַמְדָּם / וְהוּא לִפְנֵיהֶם נִרְדָּם / וַיִּקְרְאוּ אֶל הַשְּׁכֵנִים / וַיֹּאמְרוּ: בֹּאוּ הֵנָּה רֹדְפֵי שַׁלְמוֹנִים / וּרְאוּ מָה הִקְרָה לָנוּ / כִּי בַּעֲווֹנוֹתֵינוּ מֵת אָבִינוּ / וַיֵּלְכוּ וַיִּבְכּוּ אֹתוֹ / וַיַּעֲשׂוּ לוֹ כְּהִלְכָתוֹ / וַיַּלְבִּישׁוּהוּ תַכְרִיכִים בָּנָיו / וַיִּשְׂאוּ אֹתוֹ בַמְּעָרָה אֲשֶׁר לְפָנָיו / וַיְהִי בַיּוֹם הַהוּא וַיָּבֹא שָׁם הָהֶגְמוֹן / וְעִמּוֹ הָיָה רָב וְקוֹל הָמוֹן / וַיֵּשְׁבוּ בַמְּעָרָה כִתּוֹת כִּתּוֹת / וַיַּתְחִילוּ לֶאֱכֹל וְלִשְׁתּוֹת / וַיָּקָץ הָאִישׁ מִיֵּינוֹ / וַיָּקָם לֵילֵךְ לִמְעוֹנוֹ / וַיִּרְאוּ אֹתוֹ מֵרָחֹק כֻּלָּם / וַיַּנִּיחוּ אֶת מַאֲכָלָם / וַיֹּאמֶר לָהֶם הַהֶגְמוֹן: בֶּאֱמֶת / הִנֵּה הָאִישׁ הַלָּזֶה מֵת / וַיָּקוּמוּ הָאֲנָשִׁים וַיִּירְאוּ מְאֹד / וַיֵּלְכוּ וַיַּעַזְבוּ אֶת כָּל הַמְּאֹד / וּבָנָיו בָּאוּ בַמְּעָרָה לִרְאוֹתוֹ / וַיִּמְצְאוּ עֹשֶׁר גָּדוֹל אִתּוֹ / וַיִּשְׂמְחוּ שָׁם שִׂמְחָה גְדוֹלָה / וַיֵּלְכוּ לְבֵיתָם עַם סְגֻלָּה.

וְהַמָּשָׁל לְאִישׁ הַמְּמַלֵּא מִמַּעֲדַנִּים בִּטְנוֹ וְהַמֵּעַיִם / וְלַשּׁתֶה נֹתֵן יָיִן.
