בִּבְנֵי אָדָם שֶׁאֵינָם מְהֻגָּנִים / וְשָׂרֵי הַצָּבָא / בִּטְנָם צָבָה / וְשָׂרֵי אֲלָפִים וּמֵאוֹת וַחֲמִשִּׁים וַעֲשָׂרוֹת / מַנִּיחִים אֶת־חֲיָלוֹתֵיהֶם וְאֶת־הַשְּׁמִירוֹת / מִשְׁמַרְתָּן וּמִשְׁמֶרֶת הָעֲיָרוֹת / וְהוֹלְכִין אַחַר הַבְּתוּלוֹת הַיָּפוֹת וְהַנָּשִׁים הַיְקָרוֹת / הָאַפִּיפְיוֹר / שׁוֹתֶה יַיִן בְּמִזְרְקֵי־כֶסֶף וּבְכִיּוֹר / וְהוּא עוֹמֵד וּמְטַיֵּל עַל־שְׂפַת הַיְאוֹר / הַחַשְׁמַנִּים / אוֹתָן דִּרְבוּ יַתִּירָא הָיָה נוֹתֵן עֲלֵיהֶם סִימָנִים / בָּאַשְׁמַנִּים / שְׁמֵנִים וּדְשֵׁנִים וְרַעֲנַנִּים / הַהֶגְמוֹן / אֲשֶׁר כְּאַגְמוֹן / רֹאשׁוֹ יַכְנִיעַ וְיָכֹף / גַּם הוּא בְמַעֲלָלָיו דּוֹמֶה לְקוֹף.

וְהַכֹּהֲנִים וְהָעָם הָעוֹמְדִים בַּעֲבוֹדָה זָרָה / נֶהֱנִין וְנִזּוֹנִין מִזִּיו הַשְּׁכוּנָה וְהַחֲבוּרָה / וְחָטְאוּ וְעָווּ וּפָשְׁעוּ וְעָבְרוּ הָעֲבֵרָה / אֵין יוֹם שֶׁאֵין סְעוּדָתוֹ מְרֻבָּה מֵחֲבֶרְתָּהּ / וְלָא אִיבְּרִי סִיהֲרָא לְשֵׁינְתָא / אֶלָּא לִזְנוּתָא וְהֶפְקְרוּתָא.

שֶׁבִּן־לַיְלָה הָיָה וּבִן־לַיְלָה אָבְדָה שִׂמְחָתָם / וְסָרָה מֵעֲלֵיהֶם אִוַּלְתָּם / וְיָקוּמוּ אִישׁ אִישׁ מִמְּסִבָּתָם / וְרִקּוּדָם וּצְחִיקוּתָם / בְּשָׁמְעָם אֶת־רִנָּתָם / וְאֶת־קוֹל תְּרוּעָתָם / בְּבֵית אוֹן אֱלֹהֵיהֶם / וְיִפָּרְדוּ אִישׁ מֵעַל אֲחֵיהֶם / רֵעֵיהֶם וְחַבְרֵיהֶם / וְיֵרָגְנוּ בְאָהֳלֵיהֶם.

וַיְהִי מִמָּחֳרָת זֶה נָגוּעַ וְזֶה מֻכֶּה / זֶה נִשְׁבָּר וְזֶה נִדְכֶּה / זֶה כּוֹאֵב וְזֶה דּוֹאֵג / זֶה מִתְאַנֵּחַ וְזֶה שׁוֹאֵג / זֶה חוֹשֵׁשׁ בִּגְרוֹנוֹ / וְזֶה בִּשְׁרִירֵי בִטְנוֹ / זֶה מִמִּלּוּי אִצְטָמְכָתוֹ צוֹעֵק כְּאִלּוּ נִשְׁתַּטָּה / וְזֶה מִזְּרֹעוֹתָיו וּמַרְגְלוֹתָיו דּוֹאֵב כְּאִלּוּ נָפְלָה עֲלֵיהֶם מוֹטָה / וְחוֹלֶה וְנוֹפֵל עַל־הַמִּטָּה / זֶה חָשׁ בְּרֹאשׁוֹ / וְזֶה אָבַד תַּאֲוָתוֹ וְאֶפְשׁוֹ / וְלֹא יָדַע כִּי הוּא בְנַפְשׁוֹ / כֻּלָּם אֲחֻזֵי חֹרֶב וְקֶרַח בָּאוּ בֵית שֶׁמֶץ אֱלֹהֵיהֶם / וְהֵמָּה כָּרְעוּ עַל־בִּרְכֵיהֶם / וַיִּזְרְקוּ עָפָר עַל־רָאשֵׁיהֶם.
