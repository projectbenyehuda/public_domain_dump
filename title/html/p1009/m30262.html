שָׁלוֹם לְבוֹאֶךָ קֵץ הַפְּלָאוֹת

שָׁלוֹם לְבוֹאֶךָ / קֵץ הַפְּלָאוֹת

כִּי לִי בְיָמֶיךָ / חֵן חֵן תְּשׁוּאוֹת


מַהֵר לְבַשֵּׂר טוֹב לִקְהַל עֲנָוִים / בִּגְלָלְךָ אָבוֹת בָּנִים מְחַוִּים

וּלְיִשְׁעֲךָ מִיּוֹם אֶל יוֹם מְקַוִּים / כִּי בָךְ יְסַפֵּרוּ דִּבְרֵי נְבוּאוֹת


לָךְ נִכְסְפָה נֶפֶשׁ צִיר לֶאֱמוּנִים / הִפְצִיר וְעָנָהוּ פָּנִים בְּפָנִים

הָאֵל וְלֹא גִלָּה הַקֵּץ לְשָׁנִים / כַּמָּה וְעַד מָתַי אֶמְצָא תְלָאוֹת


לָכֵן הֲמוֹן מוֹנַי יִתְקַלְּסוּ לִי / כִּי תַם לְרִיק כֹּחִי אִיגַע בְּהֶבְלִי

לָאוֹר אֲקַו עַד בּוֹשׁ אוֹסִיף בְּאָפְלִי / לִבִּי וְכִלְיוֹתַי מָלְאו תְנוּאוֹת


קִצִּי הֲלֹא אָרַךְ עַל רֹב עֲווֹנִי / אֶשָּׂא לְאֵל נַפְשִׁי לִבִּי וְעֵינִי

אוּלַי וְהוּא יִרְאֶה יַבִּיט בְּעֵינִי / כַּפַּי פְּרוּשׂוֹת עַד לִזְבוּל נְשׂוּאוֹת


נַפְשִׁי וְגַם עֵינַי כָּלוּ לְיַחֵל / כִּי גֵרְשׁוּנִי מֵחוֹמָה וְהַחֵיל

אֶשְׁתַּחֲוֶה לָאֵל נוֹשֵׂא וּמוֹחֵל / שָׁם בֶּאֱמֶת נִמְצָא הַמּוֹפֵת וְהָאוֹר


הֵן מַחֲלָה לַלֵּב תִּקְוָה מְשׁוּכָה / עַד כִּי בְדִמְעָתִי עַרְסִי מְסוּכָה

מָתַי לְמַכָּתִי תַּעְלֶה אֲרוּכָה / אֶלֶף בְּכוֹשָׁרוֹת גָּלִיּוֹת כְּלוּאוֹת


נִשְׁמַע בְּסוֹד הַקֵּץ מִלָּה נְכוֹחָה / יוֹם בּוֹא שְׁנַת הַשִּׁי״ן דֶּלֶת פְּתוּחָה

הָעֵר וְהַיָּשֵׁן יִשְׁנֶה בְּשִׂמְחָה / שׁוּבוּ לְבִצָּרוֹן בֵּאלֹהִים צְבָאוֹת.
