[אֲשׂוֹחֵחַ נִפְלְאוֹתֶיךָ צוּר עוֹלָמִים (סדר עבודה)]

סֵדֶר הָעֲבוֹדָה לְיוֹם הַכִּפּוּרִים (קֶטַע)


אֲשׂוֹחֵחַ נִפְלְאוֹתֶיךָ צוּר עוֹלָמִים

אַהַג מִפְעֲלוֹתֶיךָ אַדִּיר בִּמְרוֹמִים

אֲדַבְּרָה גְּבוּרֹתֶיךָ בְּאֶרֶשׁ נְאֻמִּים

אֲזַמְּרָה תְהִלֹּתֶיךָ קְצָת מֵעֲצוּמִים.


בְּטֶרֶם חָצַבְתָּ גְּדוּדֵי מְעוֹנָה

בְּעֻזְּךָ נֶאֱזַרְתָּ אֵל אֱמוּנָה

בְּרוּחֲךָ שֻׁפְּרוּ חֲזָקִים בִּתְבוּנָה

בֻּסְּסָה תֵבֵל עַל מְכוֹנָהּ.


גִּיהַ קָרָאתָ לְהָאִיר עֵיפָה

גֵּאֶה דַּרְכְּךָ בִּסְעָרָה וְסוּפָה

גְּלַלְתּוֹ וְהֻצְפַּן לְנוֹחֲלֵי צְרוּפָה

גָּנוּז לָמוֹ לְתַעַל וְלִתְרוּפָה.


דּוֹד מְקָרֶה בַמַּיִם עֲלִיּוֹת

דִּלִּיתָ בוֹדֵל לְתִוּוּךְ מֵיוֹת

דֶּמַע בְּכֵן הִזִּילוּ תַחְתִּיּוֹת

דְּבָרְךָ קָשְׁבוּ וְהֻקְווּ בְדָכְיוֹת.


הֵנַצְתָּ צִיָּה צִצֵּי פְרָחִים

הֵנִיבָה מֶגֶד גֶּרֶשׁ יְרָחִים

הָחְנְטוּ לְמִינֵימוֹ דְשָׁאִים וְשִׂיחִים

הֲכִינֹתָ גַּן לִגְדֻלֹּתֶיךָ שָׂחִים.


וְהֵאַרְתָּ נָאוֹר אוֹרִים גְּדוֹלִים

וְהִשְׁוִיתָם יַחַד לְעִתּוֹתָם חָלִים

וְהֻקְטַן סַהַר כְּרִגֵּל מִלִּים

וְרִבִּיתָ צְבָאוֹ עָשׁ וּכְסִילִים.


זֶרֶם הִשְׁרַצְתָּ שָׂחִים וְעָפִים

זִיז וּבָרִיחַ לְכֵרַת יָפִים

זָבַת הִשְׁרִיצָה צֹנֶה וַאֲלָפִים

זִמַּנְתָּ אָבוּס לְמֵסַב אַלּוּפִים.


חַי בְּשׁוּרְךָ מְלַאכְתְּךָ תַּמָּה

חֹצֶב עַמּוּדִים לְרִקּוּעַ בְּלִימָה

חֵשֶׁק עִדּוּנִים מֶסֶךְ וּטְעִימָה

חַשְׁתָּ לִבְראוֹת יְצִיר מֵאֲדָמָה.


טֶכֶס תַּבְנִיתוֹ הֵכַנְתָּ בְחוֹתָמֶךָ

טֶפַח הֵכַנְתָּ גִּבַּלְתּוֹ מֵהֲדוֹמֶךָ

טֹהָר נִשְׁמָתוֹ עָרַכְתָּ מִמְּרוֹמֶךָ

טוֹב חִכַּמְתּוֹ כְּמַלְאֲכֵי מְרוֹמֶיךָ.


יָהּ רְאִיתוֹ גַּף בְּעוֹלָמוֹ

יִעַדְתָּ לוֹ עֵזֶר מִגָּלְמוֹ

יָשֵׁן וְנִנְעַר וְהִנָּהּ לְעֻמּוֹ

יָדַע וְהֵבִין שֶׁהִיא מֵעַצְמוֹ.


כּוֹנַנְתָּ לָמוֹ בָעֵדֶן חֻפּוֹת

כָּל יְקָרָה סוֹכַכְתָּם לְחַפּוֹת

כִּוַּנְתָּ כְּרֵעַ כְּבוּדָּה לְיַפּוֹת

כִּרְכְּרוּ כְשָׁרִים אֵילֵי תוֹעָפוֹת.


לִנְצֹר צִוּוּיְךָ לוּ חָכְמוּ

לְחַיֵּי נְצָחִים חַיָּתָם קִיְּמוּ

לְעִיטַת הַמָּוֶת אִוּוּ וַיִּטְעֲמוּ

לָכֵן נִמְשַׁל כַּבְּהֵמוֹת נִדְמוּ.


מִגַּן אֱלֹהִים גֹּרְשׁוּ פוֹתִים

מַעֲקַשֵּׁי גַיא לַחֲרֹשׁ בְּאִתִּים

מַעֲמַקֶּיהָ לְשַׂדֵּד בִּבְנֵי רְפָתִים

מַאֲכָלָם בְּזֵעַ וְלֶעָפָר שְׁפוּתִים.


נֶאֶרְרָה יוֹלֶדֶת כְּסִיל לְתוּגָה

נַעֲצוּץ וְדַרְדַּר לְהַצְמִיחַ בַּעֲרוּגָה

נִפְתֶּה וּפוֹתָה בְּעֶצֶב לְמוֹגֲגָה

נָחָש בְּגָחוֹן לֶכֶת בְּלִי-פוּגָה.


שְׂרִיגִים שֻׁלְּחוּ נוֹקֵד וְאִכָּר

שָׂשׂוּ וַיַּקְרִיבוּ לַמֶּלֶךְ אֶשְׁכָּר

שַׂגִּיא שַׁעְתָּ לְהֵבִיא מִמְבֻכָּר

סִיג נוֹכֵל מִלִּשְׁעוֹת נִתְנַכָּר.


עַז בְּגָבְהוֹ וְשִׁחֵת רַחֳמָיו

עָרַף נָקִי וְשָׁפַךְ דָּמָיו

עֲנַשְׁתּוֹ לָנוּעַ וְלָנוּד בַּאֲשָׁמָיו

עֵת הִתְוַדָּה הִתְוִיתוֹ בְרִשּׁוּמָיו.


פָּרִיצֵי שְׁלִישִׁים הֵחֵלּוּ לִמְעֹל

פְּסִילִים הִמְלִיכוּ וַיִּפְרְקוּ עֹל

פָּרַצְתָּ בַּעֲדָם גָּדֵר מִלִּנְעֹל

פָּשַׁט רַהַב לְבָלְעָם כִּשְׁאֹל.

צָצוּ פּוֹעֲלֵי אָוֶן וּמִרְמָה

צָרְחוּ סוּר יָדֵנוּ רָמָה

צִירֵי אֲרֻבּוֹת פָּתַחְתָּ לְגֻשְׁמָה

צוּל הִבְקַעְתָּ מֵהַשְׁאִיר נְשָׁמָה.


קָפְאוּ גְוִיָּתָם בְּשֶׁטֶף הוֹמִים

קָלַטְתָּ מֵהֶם צַדִּיק תָּמִים

קִנִּים עָשָׂה לְהֵחָבֵא מִזְּעָמִים

קֻלְּעָה אֳנִיָּתוֹ מִיָּמִים יָמִים.


רִצּוּי תְּשׁוּרָה לְפָנֶיךָ הֶעֱלָה

רָצִיתָ נִיחוֹחוֹ וְשַׁוְעוֹ עָלָה

רָם נִשְׁבַּעְתָּ מֵעֲשׂוֹת כָּלָה

רַעֲנַנְתּוֹ לְמַלְּאוֹת תֵּבֵל כֻּלָּהּ.


שָׁלוּ בוֹגְדִים וַיִּוָּעֲדוּ לְמָרְדָה

שִׁכְנָם שִׁית בַּשַּׁחַק לְהֵאָגְדָה

שָׂחַקְתָּ לָמוֹ כְּפַצְתָּ נֵרְדָה

שֵׂעַרְתָּם בַּנֶּפֶץ וַחֲבִילָתָם נִפְרְדָה.


תֻּעְתְּעוּ יְצוּרִים אַחֲרֵי הֲבָלִים

תָּעוּ מִמַּסְלוּל וְשָׁגוּ כֶאֱוִילִים

תָּמָר הִפְרַחְתָּ בְּתוֹךְ חֲרֻלִּים

תְּרוּפָה לְנִמְחָצִים צֳרִי לְנֶחֱלִים.


תִּרְגַּלְתָּ יְדִיד לְהַעֲמִיד נְשִׁיָּה

תִּכַּנְתָּ עַמּוּדֶיהָ מֵהִתְמוֹגֵג תּוּשִׁיָּה

תֻּמְּכוּ אֲשׁוּרָיו בְּמַעְגְּלוֹתֶיךָ יָהּ

תּוֹעִים הִסְלִיל בְּדֶרֶךְ תְּחִיָּה.
