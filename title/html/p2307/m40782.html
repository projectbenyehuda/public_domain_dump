[הָהּ כִּי פְנֵי חַמָּה לְקַדְרוּת נֶהְפְּכוּ]

Sabbato Vita Marini

[נולד בפָּדוֹבָה לערך 1690 ומת שם בשנת 1748]


– – ͝– – – ͝– – – ͝– / – – ͝– – – ͝– – – –

הָהּ כִּי פְנֵי חַמָּה לְקַדְרוּת נֶהְפְּכוּ / הָיָה לְאִישׁ אַחֵר מְקוֹר הַדָּעַת

הַדְרַת צְבִי הַחֵן בְּאֵבֶל נָסְעָה / בֵּין אוֹֽצְרוֹת עֶבְרָה וְשָׁם נִכְנָעַת

חַוָּה אֲשֶׁר כָּל־חַי בְּבִטְנָהּ נִֽהְיָה / בֵּין מִשְׁבְּרֵי הַמָּֽוְתָה כּוֹרָעַת

עֵצָה וְתוּשִׁיָּה אֲשֶׁר בָּהּ נַאֲוָה / בֵּין מַעְיְנוֹת הַדּוֹמְעִים טוֹבָעַת

מַה־תַּעֲשׂוּ אִישִׁים גִּבִירֵי גַאֲוָה / עֵת אֵם לְכָל־בָּשָׂר בַּבּוֹר שׁוֹקָעַת

אֵיךְ נָֽפְלוּ חִצִּים וְחָרְבוֹת נוֹֽצְחִים / קֶשֶׁת נְחוּשָׁה הָיְתָה מִרְשָׁעַת

לִבּוֹת מְתֵי צָבָא כְּדוֹנַג נוֹֽזְלִים / הֶמְיַת שְׁאוֹנָם לֹא תְהִי נִשְׁמָעַת

כָּל־מַרְכְּבוֹתֵיהֶם כְּחָצִיר אֻמְלְלוּ / יִהְיוּ כְמֹץ כַּקַּשׁ לְאֵשׁ מוּצָעַת

חֵץ וַחֲנִית יַחְדָּו כְּעָשָׁן יֹאֽבְדוּ / כִּי לַהֲבַת אָסוֹן בְּעֹז נוֹבָעַת

לֹא נָֽצְחָה דִמְעָה וּפוּגַת־לֵב אֲשֶׁר / מוּל מַחֲנֵה הַמָּֽוְתָה נוֹסָעַת

כָּל־נַחֲלֵי תֵבֵל לְאֶפֶס נֶחְשְׁבוּ / כִּי גַחֲלָתוֹ לַזְּבוּל נוֹגָעַת

כָּל־אוֹֽצְרוֹת זָהָב כְּנִבְלֵי יוֹֽצְרִים / כִּי לַעֲנָה וָרוֹשׁ בְּכַף מַכְרָעַת

כָּל־נַהֲרוֹת חֵמָה סְבִיבָיו עוֹֽמְדִים / עָנָן וְחֹשֶׁךְ רַב וְכָל־מִגְרָעַת

לֹא זָֽרְחוּ נֶגְדּוֹ שְׁבִיבֵי יַהֲלֹם / כֶּסֶף אַרְגָּמָן שֳׁנִי תוֹלָעַת

לוּ כוֹֽכְבֵי מָרוֹם לְמוּלוֹ נִמְשְׁלוּ / רַגְלָם יְשָׁרָה תִהֲלַךְ צוֹלָעַת

לוּ מַרְכְּבוֹת חַמָּה לְפָנָיו יִשְׁכְּנוּ / יִפְעַת גְּדֻלָּתָם תְּהִי שׁוֹקָעַת

וַיִּקְדְּרוּ נֶגְדּוֹ פְנֵי מֹשֶׁה, בְּעֵת / אוֹרָה בְפָנָיו הָיְתָה שׁוֹפָעַת

לֹא הֶאֱמַנְתִּי עַד־אֲשֶׁר עוֹד אֶחֱזֶה / כִּי אֵשׁ חֲמָסִים עַל־אֱנוֹשׁ נִזְרָעַת

חִצִּים שְׁנוּנִים לוֹ וְיָדוֹ תִזְרְקֵם / לֹא תַּחֲטִיא הַשַּׂעֲרָה קוֹלָעַת

בָּם נִכְרְתוּ אִישִׁים וְנָשִׁים נִגְדְּעוּ / שִׂמְחַת קְצִירֵיהֶם אֲזַי נִמְנָעַת

לֹא הֶאֱמַנְתִּי כִּי לְחַוָּה אָפֲפוּ / חֶבְלֵי מְרוֹרָתוֹ וְכוֹס קֻבָּעַת

עוֹד מַלְתְּעוֹת נָחָשׁ שְׁלוֹמָהּ עָֽשְׁקוּ / עַד־כִּי מְאֵרַת מָֽוְתָה קוֹבָעַת

מָה אֹֽמְרָה אִם זֹאת לְנַפְשָׁהּ סִבְּבָה / כִּי עֵץ בְּחִירוֹ הָֽיְתָה נוֹגָעַת

מָה אֶעֱשֶׂה אִם עֵץ תְּבוּנָה לָֽקְחָה / אַךְ עַל־דְּבַר צוּר נַעֲלֶה פּוֹשָׁעַת

אֵם פָּֽשְׁטָה יָדָהּ בְּאֶדֶר הַֽיְקָר / וּבְעֵץ חֲמַת עַכְשׁוּב תְּהִי מוּקָעַת

אֵם חָֽשְׁבָה לִקְנוֹת פְּרִי הַשְׂכֵּל, אֲבָל / עַל־עֹז סְעִפִּים בּוֹֽגְדִים פּוֹסָעַת

לוּלֵי שְׁדֵי חַסְדָהּ מְלָכִים הֶחֱיוּ / לֹא תַחֲרִישׁ נַפְשִׁי לְאֵם גֹּוָעַת

לוּלֵי בְנֵי־בִטְנָה נְשִׂיאֵי הַזְּמָן / אֶלְבַּשׁ מְעִיל קַדְרוּת וְלֹא מִגְבָּעַת

אַל־תִּשְׁקְטִי רוּחִי וְאַל־נָא תֶחֱשִׁי / לִבֵּךְ וְעֵינֵךְ אַל־תְּהִי רוֹגָעַת

אַךְ כָּל־יְמוֹת עוֹלָם בְּבִכְיִי אַחֲזִיק / נֶפֶשׁ בְּמַר בּוֹכָה וְעֵין דּוֹמָעַת

כִּי מִפְּנִינֵי פָז פְּתָנִים יָֽצְאוּ / כָּל־מַעֲלָה טוֹבָה בְּאַף נִגְדָּעַת

הָהּ כִּי הֲלֹם דֵּעָה לְרָעָה נֶחְלְפָה / הָיָה לְאִישׁ אַחֵר מְקוֹר הַדָּעַת.
