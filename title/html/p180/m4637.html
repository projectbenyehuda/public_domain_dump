[כנשר את מרחף על-ילדיו] / שלמה אבן גבירול

ולה תצאנע אלי בעץ ͘ תעל ّ ם עלם אלאדב [ולו דברי מחמאות לאיש שהיה בקי בחכמת המוסר]

כְּנֶשֶׁר אתְּ מְרַחֵף עַל-יְלָדָיו / וְכָאָדוֹן אֲשֶׁר משֵׁל עֲבָדָיו

וְאַתְּ בַּלָּט עֲלוֹת הַשִּׁיר מְנַהֵג / כְּמוֹ-מֶלֶךְ יְנַהֵג אֶת-גְּדוּדָיו

וְהַמּוּסָר בְּךָ נֶאְסָר וְלֹא-סָר / בְּךָ יֵלֵךְ וְלֹא-יֵצַר צְעָדָיו

וְאִם-הִתְפָּֽרְדוּ קֶדֶם פְּזוּרָיו / הֲלֹא אַתָּה תְּקַבֵּץ אֶת-נְדוּדָיו

וְעַל-אֹזֶן תְּבוּנָה אַתְּ כְּעָגִיל / וְגַם עַל-צַוְּרֹנֶיהָ רְדִידָיו

וְאַתְּ נִמְשָׁל כְּעֵץ חַיִּים וְקִצּוֹ / פְּני-שַׁחַק וּבָאָרֶץ יְסוֹדָיו

וּמִי יִתֵּן יְשִׂימֵנִי יְדִידִי / כְּמוֹ זֵרוֹ וְכַחוּט עַל-בְּגָדָיו

וְיָבִין מִתְּבוּנָתוֹ לְבָבִי / וְיִרְאַנִי כְּאַחַד מִיְּלָדָיו

אֲשׁוֹרֵר שִׁיר תְּהִלָּתוֹ לְעוֹלָם / וְיִתֵּן עֵץ תְּהִלָּתוֹ מְגָדָיו.
