בלק, ובלעם, המלחמה במדין, מות משה

וַיִסְעוּ בְנֵי יִשְׂרָאֵל וַיָבֹאוּ עַד עַרְבוֹת מוֹאָב אֲשֶר מֵעֵבֶר לְיַרְדֵן יְרֵחוֹ. וַיִשְמְעוּ בְנֵי מוֹאָב אֶת כָּל אֲשֶר עָשׂוּ בְנֵי יִשְׂרָאֵל לִשְנֵי מַלְכֵי הָאֱמוֹרִי, לְסִיחוֹן וּלְעוֹג, וַיִירְאוּ מִפְּנֵיהֶם מְאֹד. וַיֹאמְרוּ זִקְנֵי מוֹאָב: הִנֵה סִיחוֹן וָעֹוג הַחֲזָקִים מִכָּל מַלְכֵי הָאָרֶץ לֹא עָמְדוּ בִּפְנֵי בְנֵי יִשְׂרָאֵל וְאֵיךְ נַעֲמוֹד בִּפְנֵיהֶם אֲנָחְנוּ? ­– וַיָקוּמוּ בְנֵי מוֹאָב וַיִבְחֲרוּ בָעֵת הַהִיא בְּאֶחָד מֵאַנְשֵיהֶם וּשְמוֹ בָּלָק בֶּן צִפּוֹר לְמֶלֶךְ עֲלֵיהֶם. וַיִשְלְחוּ מַלְאָכִים אֶל בְּנֵי מִדְיָן וַיַעֲשׂוּ אִתָּם שָלוֹם, כִּי מִלְחָמוֹת גְדוֹלוֹת וְשִׂנְאָה עַזָה הָיְתָה תָמִיד בֵּין מוֹאָב וּמִדְיָן מִימֵי הֲדַד בֶּן בְּדַד מֶלֶךְ אֱדוֹם הַמַכֶּה אֶת מִדְיָן בִּשְׂדֵה מוֹאָב. וַיָבֹאוּ זִקְנֵי מִדְיָן אֶל אֶרֶץ מוֹאָב, וַיִתְיָעֲצוּ אִתָּם זִקְנֵי מוֹאָב מַה לַעֲשׂוֹת לִבְנֵי יִשְׂרָאֵל וַיֹאמְרוּ אֲלֵיהֶם זִקְנֵי מִדְיָן: שָמַעְנוּ כִּי בַעֵת אֲשֶר נִלְחַם אִתְכֶם סִיחוֹן מֶלֶךְ הָאֱמוֹרִי וַתִּגְבּוֹר יָדוֹ עֲלֵיכֶם, כִּי שָלַח אֶל בְּעוֹר בֶּן יַנוּס וְאֶל בִּלְעָם בְּנוֹ מֵאֲרַם־נַהֲרַיִם, וַיָבֹאוּ וַיְקַלְלוּ אֶתְכֶם, עַל כֵּן גָבְרָה יַד סִיחוֹן עֲלֵיכֶם וַיִלְכֹּד אֶת עִירְכֶם הַגְדולָה חֶשְבּוֹן. וְעַתָּה שִלְחוּ נָא גַם אַתֶּם אֶל בִּלְעָם בְּנוֹ, וּנְתַתֶּם לוֹ שְׂכָרוֹ לְמַעַן אֲשֶר יְקַלֵל גַם אֶת עַם יִשְׂרָאֵל וְהִכִּיתֶם אוֹתוֹ לְפִי חָרֶב. וַתִּיטַב עֲצַת זִקְנֵי מִדְיָן בְּעֵינֵי זִקְנֵי מוֹאָב, וַיִשְלַח בָּלָק בֶּן צִפּוֹר מַלְאָכִים אֶל בִּלְעָם בֶּן בְּעוֹר בֶּן יַנוּס לֵאמֹר: הִנֵה עַם יִשְׂרָאֵל הַיוֹצֵא מִמִצְרַיִם ­– כִּסָה אֶת עֵין הָאָרֶץ, וְהוּא יוֹשֵב מִמֻלִי. וְעַתָּה לְכָה אָרָה לִי אֶת הָעָם הַזֶה לְמַעַן אֲשֶר אַכֶּה אוֹתוֹ וַאֲגָרְשֶנוּ מֵאַרְצִי. וְלֹא יָדְעוּ זִקְנֵי מוֹאָב, כִּי יְיָ צִוָה עַל בְּנֵי יִשְׂרָאֵל לְבַל יִתְגָרוּ מִלְחָמָה בְּמוֹאָב וּבְיֶתֶר גוֹיֵי הָאֲרָצוֹת, זוּלָתִי בְּעַמֵי כְנַעַן לְבַדָם. וַיֵלְכוּ זִקְנֵי מוֹאָב וְזִקְנֵי מִדְיָן וַיָבֹאוּ פְתוֹרָה אֶל בִּלְעָם וַיַגִידוּ לוֹ אֶת דִבְרֵי בָלָק. וַיֹאמֶר אֲלֵיהֶם בִּלְעָם: לִינוּ פֹה הַלַיְלָה וַהֲשִיבוֹתִי אֶתְכֶם דָבָר, כַּאֲשֶר יְדַבֵּר אֵלַי יְיָ. וַיָבֹא אֱלֹהִים אֶל בִּלְעָם בַּחֲלוֹם הַלַיְלָה וַיֹאמֶר לוֹ: לֹא תֵלֵךְ עִמָהֶם וְלֹא תָאוֹר אֶת הָעָם, כִּי בָרוּךְ הוּא. וַיְהִי בַּבֹּקֶר וַיֹאמֶר בִּלְעָם לְזִקְנֵי מוֹאָב: לְכוּ וְשוּבוּ אֶל אַרְצְכֶם, כִּי מֵאֵן יְיָ לְתִתִּי לַהֲלֹךְ עִמָכֶם. וַיָשוּבוּ מַלְאֲכֵי מוֹאָב וַיָבוֹאוּ אֶל בָּלָק וַיֹאמְרוּ לוֹ: מֵאֵן בִּלְעָם הֲלֹךְ עִמָנוּ. וַיוֹסֶף בָּלָק לְחַלוֹת אֶת פְּנֵי זִקְנֵי מִדְיָן, כִּי יֵלְכוּ וְיָאִיצוּ בְּבִלְעָם לָבוֹא אַרְצָה מוֹאָב וּלְקַלֵל אֶת בְּנֵי יִשְׂרָאֵל. וַיָבאֹ אֱלֹהִים גַם הַפַּעַם בְּמַחֲזֵה הַלַיְלָה אֶל בִּלְעָם וַיֹאמֶר לוֹ: אִם לִקְרֹא לְךָ בָּאוּ הָאֲנָשִים ­– קוּם לֵךְ אִתָּם, אַךְ אֶת אֲשֶר אֲדַבֵּר אֵלֶיךָ אוֹתוֹ תַעֲשֶׂה.

וַיְהִי בְּלֶכֶת בִּלְעָם עִם זִקְנֵי מוֹאָב וַיִתיַצֵב מַלְאַךְ ה' לְשָׂטָן לוֹ בַדֶרֶךְ. וַתֵּרֶא הָאָתוֹן אֶת מַלְאַךְ ה' וַתִּבָּהֵל מִפָּנָיו וַתֵּט מִן הַדֶרֶךְ וַתֵּלֶךְ בַּשָׂדֶה, וַיַךְ בִּלְעָם אֶת אֲתוֹנוֹ לְהַטוֹתָהּ אֶל הַדָרֶךְ. וַיִתְיַצֵב מַלְאַךְ ה' בְּמִשְעֹל הַכְּרָמִים גָדֵר מִזֶה וְגָדֵר מִזֶה. וַתֵּרֶא הָאָתוֹן אֶת מַלְאַךְ ה' וַתִּלָחֵץ אֶל הַגָדֵר וַתִּלְחַץ אֶת רֶגֶל בִּלְעָם, וַיוֹסֶף לְהַכּוֹתָהּ. וַתִּרְבַּץ הָאָתוֹן תַּחַת בִּלְעָם בְּמָקוֹם צַר, וַיַךְ אוֹתָהּ בִּלְעָם בַּמַקֵל. וַתִּפְתַּח הָאָתוֹן אֶת פִּיהָ וַתְּדַבֵּר בִּלְשוֹן בְּנֵי אָדָם אֶל בִּלְעָם וַתֹּאמַר: מֶה עָשִׂיתִי לְךָ כִּי הִכִּיתַנִי זֶה שָלֹש פְּעָמִים? ­– וּמַדוּעַ לֹא שַׂמְתָּ לִבְּךָ, כִּי ה' עוֹמֵד לְשָׂטָן עַל דַרְכֶּךָ"?!… וַיְגַל ה' אֶת עֵינֵי בִלְעָם וַיַרְא אֶת הַמַלְאָךְ הַנִצָב לְנֶגְדוֹ וְחַרְבּוֹ שְלוּפָה בְיָדוֹ­ – וַיִקֹד וַיִשְתַּחוּ לְפָנָיו אָרְצָה. וַיֹאמֶר לוֹ בִלְעָם: חָטָאתִי, וְאִם רַע בְּעֵינֶיךָ ­– וְאָשוּבָה. וַיֹאמֶר הַמַלְאָךְ. לֵךְ עִם הָאֲנָשִים, אֶפֶס אֶת הַדָבָר אֲשֶר אֲדַבֵּר אֵלֶיךָ אוֹתוֹ תְדַבֵּר.

וַיֵלֶךְ בִּלְעָם עִם שָׂרֵי מוֹאָב וַיָבֹא אֶל בָּלָק. וַיַעֲלֵהוּ בָּלָק עַל אַחַד הֶהָרִים אֲשֶר מִשָם יוּכַל לִרְאוֹת אֶת מַחֲנֶה יִשְׂרָאֵל, אוּלַי יִמְצָא בוֹ שֶמֶץ דֹפִי. וַיְהִי כִּרְאוֹת בִּלְעָם אֶת מַעַרְכוֹת יִשְׂרָאֵל וַיִשָׂא מְשָלוֹ וַיאמַר:

מִן אֲרָם יַנְחֵנִי בָלָק מֶלֶךְ מוֹאָב, מֵהַרְרֵי קֶדֶם

לְכָה אָרָה לִי יַעֲקֹב וּלְכָה זֹעֲמָה יִשְׂרָאֵל ­–

מָה אֶקוֹב לֹא קַבּוֹ אֵל, וּמָה אֶזְעֹם לֹא זָעַם יְהוָֹה.

כִּי מֵרֹאש צוּרִים אֶרְאֶנוּ וּמִגְבָעוֹת אֲשוּרֶנוּ:

הֵן עַם לְבָדָד יִשְכֹּן וּבַגוֹיִם לֹא יִתְחַשָב.

מִי מָנָה עֲפַר יַעֲקֹב וּמִסְפָּר אֶת רֹבַע יִשְׂרָאֵל

תָּמוּת נַפְשִי מוֹת יְשָרִים וּתְהִי אַחֲרִיתִי כָמוֹהוּ.

וַינַס בָּלָק לְהוֹבִיל אֶת בִּלְעָם אֶל מְקוֹמוֹת אֲחֵרִים אוּלַי יִרְאֶה מִשָם אָוֶן בְּיִשְׂרָאֵל וְיִמְצָא תוֹאֲנָה לְקַלֵל אוֹתָם. אַךְ בִּלְעָם לֹא הִבִּיט אָוֶן בְּיַעֲקֹב וְלֹא רָאָה עָמָל בְּיִשְׂרָאֵל וּבִרְאוֹתוֹ אֶת הַסֵדֶר וְהַמִשְטָר הַשוֹרֵר בְּמַחֲנֵה הָעִבְרִים נוֹכַח כִּי בָלָק לֹא יוּכַל לָעָם הַזֶה, וַיוֹסֶף לְבָרֵךְ אֶת יִשְׂרָאֵל וּלְנַבֵּא לוֹ עֲתִידוֹת תּוּשִיָה וְרָצוֹן. וַיִחַר אַף בָּלָק בְּבִלְעָם וַיְשַלְחֵהוּ מֵעַל פָּנָיו בְּחֶרְפָּה.

וַיָקָם בִּלְעָם וַיָשָב אֶל אַרְצוֹ. אוּלָם בְּטֶרֶם לֶכְתּוֹ יָעַץ אֶת זִקְנֵי מוֹאָב וּמִדְיָן, כִּי יְצַווּ עַל בְּנוֹתֵיהֶם אֲשֶר תְּפַתֶּינָה אֶת נַעֲרֵי בְנֵי יִשְׂרָאֵל לַעֲבוֹד אֶת הָאֱלִיל פְּעוֹר, וּבָזֶה יַעֲלוּ עֲלֵיהֶם אֶת חֲמַת ה‘. וַיַעֲשׂוּ כֵן בְּנֵי הַמוֹאָבִים. וַיְהִי בְּשֶבֶת יִשְׂרָאֵל בְּנַחַל הַשִטִים, וַתֵּצֶאנָה בְּנוֹת מוֹאָב וַתִּקְרֶאנָה לִבְנֵי יִשְׂרָאֵל לִזְבֹּחַ לֶאֱלֹהֵיהֶן. וַיִצָמֵד יִשְׂרָאֵל לְבַעַל פְּעוֹר, וַיֹאכְלוּ וַיִשְתּוּ עִם בְּנוֹת מוֹאָב. וַיִחַר אַף ה’ בְּיִשְׂרָאֵל עַל הַדָבָר הַזֶה וַיִשְלַח בָּהֶם מַגֵפָה וַיָמוּתוּ בַּיוֹם הַהוּא אַרְבָּעָה וְעֶשְׂרִים אֶלֶף אִיש. וַיֹאמֶר ה' אֶל מֹשֶה: נְקוֹם אֶת נִקְמַת בְּנֵי יִשְׂרָאֵל מֵאֶת הַמִדְיָנִים, כִּי צוֹרְרִים הֵם לָכֶם בְּנִכְלֵיהֶם. וַיִצְבְּאוּ בְנֵי יִשְׂרָאֵל עַל בְּנֵי מִדְיָן וַיִלָחֲמוּ בָהֶם וַיַכּוּם לְפִי חָרֶב, וְגַם אֶת חֲמֵשֶת נְסִיכֵי מִדְיָן וְאֶת בִּלְעָם בֶּן בְּעוֹר הָרְגוּ בַמִלְחָמָה הַהִיא, וַיָשוּבוּ אֶל מַחֲנֵיהֶם בְּשָלָל רָב.

בָּעֵת הַהִיא אָמַר ה' אֶל מֹשֶה: עֲלֵה עַל הַר הָעֲבָרִים וּרְאֵה אֶת הָאָרֶץ אֲשֶר נָתַתִּי לִבְנֵי יִשְׂרָאֵל, וְרָאִיתָ אוֹתָהּ וְנֶאֱסַפְתָּ אֶל עַמֶךָ. וַיֹאמֶר מֹשֶה אֶל ה‘: יִפְקֹד יְהֹוָה אֱלֹהֵי הָרוּחוֹת אִיש עַל הָעֵדָה, אֲשֶר יֵצֵא לִפְנֵיהֶם, וַאֲשֶר יָבֹא לִפְנֵיהֶם, וְלֹא תְהִי עֲדַת ה’ כְּצֹאן אֲשֶר אֵין לָהֶם רוֹעֶה. וַיֹאמֶר ה' אֶל מֹשֶה: קַח לְךָ אֶת יְהוֹשֻעַ בִּן נוּן מְשָרְתְךָ וְסָמַכְתָּ אֶת יָדְךָ עָלָיו וְנָתַתָּ מֵהוֹדְךָ עָלָיו, לְמַעַן אֲשֶר יִשְמְעוּ לוֹ כָל עֲדַת בְּנֵי יִשְׂרָאֵל. וַיַעַשׂ מֹשֶה כַאֲשֶר צִוָהוּ ה' וַיִקַח אֶת יְהוֹשֻעַ וַיַעֲמִידֵהוּ לִפְנֵי אֶלְעָזָר בֶּן אַהֲרֹן הַכֹּהֵן וְלִפְנֵי כָל הָעֵדָה, וַיִסְמֹךְ אֶת יָדוֹ עָלָיו. ­– וּשְלֹשִים יוֹם לִפְנֵי מְלֹאת אַרְבָּעִים שָנָה לְצֵאת בְּנֵי יִשְׂרָאֵל מֵאֶרֶץ מִצְרַיִם אָסַף מֹשֶה אֶת הָעָם וְאֶת זְקֵנָיו וְשוֹפְטָיו וְשוֹטְרָיו וַיַשְמִיעֵם אֶת מִשְנֵה הַתּוֹרָה וְחֻקֶיהָ וּמִשְפָּטֶיהָ הַצַדִיקִים אֲשֶר יִחְיוּ בָהּ בְּהֵאָחֲזָם בָּאֲדָמָה, אֲשֶר נִשְבַּע ה' לַאֲבוֹתָם לָתֵת לָהֶם. וַיְבָרֶךְ מֹשֶה אֶת בְּנֵי יִשְׂרָאֵל וַיָעַר לְמוּסָר אָזְנָם לָלֶכֶת בְּדֶרֶךְ הַטוֹב וְהַיָשָר כְּכָל הַכָּתוּב בְּסֵפֶר תּוֹרַת יְיָ. וַיַעַל מֹשֶה עַל הַר נְבוֹ רֹאש הַפִּסְגָה לְהַר הָעֲבָרִים אֲשֶר בְּאֶרֶץ מוֹאָב. וַיַשְקֵף מִשָם עַל אֶרֶץ כְּנַעַן וַיַרְא אֶת הֲדָרָהּ, וַיִתְחַנֵן אֶל ה' כִּי יִתֵּן לוֹ לַעֲבוֹר אֶת הַיַרְדֵן. וַיַעֲנֵהוּ ה': מִנֶגֶד תִּרְאֶה אֶת הָאָרֶץ וְשָמָה לֹא תָבוֹא, וּמוּת בָּהָר הַזֶה.

וַיְהִי בְּבוֹא הַיוֹם הָאַחֲרוֹן לְחַיֵי מֹשֶה, וַיֹאמֶר ה' לְגַבְרִיאֵל הַמַלְאָךְ: לֵךְ וְהָבֵא לִי אֶת נִשְמַת מֹשֶה. וַיַעַן גַבְרִיאֵל וַיֹאמַר: אֲדוֹן הָאֲדוֹנִים וּמֶלֶךְ הַמְלָכִים, אֵינֶנִי יָכוֹל לִרְאוֹת בְּמוֹת מֹשֶה אִיש־הָאֱלֹהִים. וַיֹאמֶר ה' לַמַלְאָךְ מִיכָאֵל: לֵךְ אַתָּה וְהָביאָה לִי אֶת נִשְמַת מֹשֶה. וַיַעַן מִיכָאֵל וַיֹאמַר: אֵינֶנִי יָכוֹל לִרְאוֹת בְּמוֹת מֹשֶה אִיש־הָאֱלֹהִים. וַיֹאמֶר ה' לְסַמָאֵל הַשָׂטָן: לֵךְ אַתָּה וְהָבִיאָה לִי אֶת נִשְמַת מֹשֶה. וַיֵלֶךְ סַמָאֵל וַיָבוא אֶל הַר נְבוֹ וַיַרְא וְהִנֵה מֹשֶה יוֹשֵב וְעוֹסֵק בַּתּוֹרָה, וַיִירָא הַשָׂטָן לָגֶשֶת אֵלָיו. וַיְהִי כַּאֲשֶר פָּסַק מֹשֶה מִמִשְנָתוֹ, וַיִגַש אֵלָיו הַשָׂטָן וַיֹאמֶר לוֹ: מֹשֶה! משֶׁה! לָקַחַת אֶת נַפְשְךָ בָאתִי, כַּאֲשֶר צִוַנִי הָאֱלֹהִים. וַיַעֲנֵהוּ מֹשֶה: יִגְעַר יְיָ בְּךָ, הַשָׂטָן! לֶךְ לְךָ מִזֶה, כִּי בְּיָדְךָ לֹא אֶתֶּן אֶת נִשְמָתִי! וַיִירָא הַשָׂטָן מִפְּנֵי מֹשֶה וַיִבְרַח וַיָשָב הַשָמַיְמָה וַיֹאמֶר אֶל ה‘: מְמָאֵן בֶּן־עַמְרָם לִמְסוֹר בְּיָדִי אֶת נִשְמָתוֹ. וַיֵרֶד ה’ מִשְמֵי הַשָמַיִם בַּעֲנַן כְּבוֹדוֹ אֶל הַר נְבוֹ, וַיֹאמֶר אֶל מֹשֶה: מֹשֶה בְנִי, הִגִיעַ הָרֶגַע הָאַחֲרוֹן לִימֵי חַיֶיךָ! הֱיֵה נָכוֹן לְהָשִיב לִי אֶת נִשְמָתְךָ, אֲשֶר נָפַחְתִּי בִגְוִיָתְךָ לִשְכּן בָּהּ רַק מֵאָה וְעֶשְׂרִים שָנָה. וַיְהִי בְּבוֹא הָרֶגַע הָאַחֲרוֹן לְחַיֵי מֹשֶה וַיֹאמֶר ה' אֶל נִשְמָתוֹ: בִּתִּי, צְאִי מִגְוִיַת מֹשֶה בְנִי וְשוּבִי אֵלָי. וַתַּעַן הַנְשָמָה: אֲדוֹנִי אֱלֹהִים, הֲיֵש לְךָ גוּף יוֹתֵר טָהוֹר בַּעֲדִי מִגְוִיַת מֹשֶה אִיש־הָאֱלֹהִים? וַיֹאמֶר לָהּ ה': אֲנִי אֲשִׂימְךָ בִּגְנָזַי וַאֲפָאֵר בָּךְ אֶת כִּסֵא כְבוֹדִי.

וּבָרֶגַע הַזֶה נָשַק יְהֹוָה אֶת מֹשֶה­ – וַתֵּצֵא נִשְמָתוֹ בִּנְשִיקָה.
