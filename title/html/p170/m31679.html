אֲלֵיכֶם אֱמוּנִים

אל יוסף


אֲלֵיכֶם אֱמוּנִים / בְּחֵיק הָאֱמוּנִים / מְתֵי רַעֲיוֹנִים / כְּבָרָק לְטוּשִׁים,

אֲדוֹנֵי קְהִלּוֹת / וְאַדְנֵי תְהִלּוֹת / מְפִיצִים תְּהִלּוֹת / לְכָל אוֹר מְבִישִׁים,

וְנֵרוֹת בְּאֹפֶל / זְמַנִּים, וְעֹפֶל / וּמִסְעָד לְנוֹפֵל / וּמִשְׁעַן יְקוּשִׁים,

וְצַהְלַת עֲצֵבִים / וְרִפְאוּת כְּאֵבִים / וְאוֹצַר רְעֵבִים / וְעֹשֶׁר לְרָשִׁים,

יְדֵי תֹם לְחַזֵּק / וְעָוֶל לְהָדֵק / וְחַפִּים לְצַדֵּק / חֲנֵפִים לְהַאֲשִׁים,

פְּזוּרִים לְהַקְהִיל, / עֲגוּמִים לְהַצְהִיל, / חֲרֵדִים לְהַאְהִיל, / חֲרֵרִים לְהַנְשִׁים,

עֲתִידִים נְכוֹחוֹת / לְהוֹרוֹת וְצַחוֹת, / וְיַשֵּׁר אֳרָחוֹת / לְשֵׂטִים וְעִקְשִׁים,

צְנוּעֵי לְבָבוֹת / כְּפִירֵי קְרָבוֹת / מְתוּקֵי אֲמָרוֹת / לְמָרֵי נְפָשִׁים,

וְשָׁרִים, אֲבָל הֵם / כְּשָׂרֵי אֱלֹהִים / קְדוֹשִׁים, וְאָכֵן / כְּקֹדֶשׁ קֳדָשִׁים,

בְּקִרְבָּם נְשָׁמוֹת / וְהֵם עַל מְרוֹמוֹת / בְּמִשְׁכַּן נְעִימוֹת / וְחַיּוֹת קְדוֹשִׁים,

צְמֵאִים כְּמֵהִים / לְטוּב הָאֱלֹהִים / וּמַעְלוֹת גְּבוֹהִים / לְאַחֵז מְחִישִׁים,

וְגוּפוֹת בְּתֵבֵל / רְאוֹת שֹׁד וְתֶבֶל / יְמֵיהֶם בְּהֶבֶל / וּמִגֵּו גְּרוּשִׁים,

תְּפַתֶּה בְגִילָהּ / וְתַשִּׁיא בְחֵילָהּ, / אֲסוּרִים בְּכַבְלָהּ / וְאִם הֵם נְטוּשִׁים,

וְתוֹלִיד פְּתָאִים / בְּנֵי אִישׁ קְרוּאִים, / בְּעֵין כֹּל פְּרָאִים, / בְּעֵינָם אֲנָשִׁים,

וְהֵם הַמְּרֵעִים / וְנִרְאִים כְּרֵעִים, / וְהֵם הַצְּבוּעִים / בְּצֶבַע נְחָשִׁים.

לְהַרְחִיב גְּרוֹנָם / וְהַאְרִיךְ לְשׁוֹנָם / וְלִנְשֹׁךְ בְּשִׁנָּם / וְאֵין לָם לְחָשִׁים,

אֲנוּשִׁים לְשַׁבֵּר / וְזֵדִים לְגַבֵּר / וְאַיֵּה מְדַבֵּר / עֲלֵי לֵב אֲנוּשִׁים?

וְאָזְנָם פְּתוּחָה / וְעֵינָם פְּקוּחָה / וְלִפְנֵי נְכוֹחָה / כְּעִוְרִים וְחֵרְשִׁים,

לְהָרַע יְהִירִים / שְׂבֵעִים שְׁכוּרִים / בְּלֶחֶם שְׁקָרִים / וּמִיֵּין עֲנוּשִׁים,

חֲנֵפִים וְרֵקִים / בְּפָנִים חֲזָקִים / כְּנָחוּשׁ יְצוּקִים / וְנַפְשׁוֹת קְדֵשִׁים,

עֲלִילוֹת לְבַקֵּשׁ / סְלוּלוֹת לְעַקֵּשׁ / וְחִנָּם לְנַקֵּשׁ / לְחַיִל כְּנוֹשִׁים,

וְאֹרְגִים בְּגָדִים / בְּחָמָס אֲפוּדִים / וְעָגִים מְרוּדִים / לְהַאְכִיל וְלָשִׁים,

זְמוֹרֵי פְתָנִים / וְסוּרֵי גְפָנִים / יְקַו אִישׁ עֲנָבִים / וְיַעְשׂוּ בְאֻשִּׁים.

צְאוּ מִנְּוֵיהֶם / וְזַעַף פְּנֵיהֶם / וְטוֹב מִשְּׁנֵיהֶם / מְעוֹנוֹת לְיָשִׁים.

וְסוּרוּ, תְּמִימִים, / לְחֶזְיוֹן חֲכָמִים / וְלֹא אֶל קְסָמִים / וְלִקְרַאת נְחָשִׁים,

לְגִבְעַת לְבוֹנָה / וּמֹר הָאֱמוּנָה / וְגִלְיוֹן תְּבוּנָה / לְזוֹכְרִים וְנוֹשִׁים,

כְּשֶׁמֶן חֲלָקִים / וְכִרְאִי חֲזָקִים / לְגָרוֹן עֲנָקִים / וְצֶפֶת לְרָאשִׁים,

כְּנִטְעֵי עֲרָבִים / לְהוֹרוֹת עֲרֵבִים / בְּצַחוֹת עֲרָבִים / נְבוֹנֵי לְחָשִׁים,

וְחָכְמַת יְוָנִים / כְּנִטְפֵי עֲנָנִים / וְנַחְלֵי עֲדָנִים / כְּיָם לֹא יְבֵשִׁים,

עֲגִילֵי מְלִיצוֹת / נְזָמִים לְעֵצוֹת / צְבִי מַחֲלָצוֹת / וְעֶדְיֵי לְחָשִׁים,

יְסוֹדוֹת וְיָדוֹת / לְכִסְאוֹת תְּעוּדוֹת / מְשָׁלִים וְחִידוֹת / עֲלֵיהֶם קְרָשִׁים,

חֲלִי צַוְּארֵיהֶם / וְתוֹרֵי לְחֵיהֶם / חֲכָמִים עֲלֵיהֶם / מְיַדִּים חֲלָשִׁים.

נְגִידֵי אֲמָרִים / לְהַחְכִּים וְלַעְרִים / מְתוּקִים וּמָרִים / וְרַכִּים וְקָשִׁים,

לְגַלּוֹת כְּסוּיִים / וְכַסּוֹת גְּלוּיִים / לְחַדֵּשׁ בְּלוּיִים / וּבַלּוֹת חֲדָשִׁים,

לְיַשֵּׁב מְעוֹנִים / לְאִישִׁים מְבִינִים / וְהַחְרִיב לְשׁוֹנִים / נְוֵיהֶם וְלַשִּׁים.

וְאֵלֶּה יְלָדִים / לְשִׂכְלִי מְעִידִים / וְאִם הֵם עֲבָדִים / בְּצוּר פִּי כְבוּשִׁים,

בְּנֵי רַעֲיוֹנַי / וְנוֹזְלֵי עֲנָנַי / לְפִי צוּר לְשׁוֹנִי / וְעֵטַי חֲלוּשִׁים,

לְדֵעָה וְחָכְמָה / כְּדֶלֶת וְחוֹמָה / וְסִתְרֵי מְזִמָּה / לְחֶפְצָם דְּרוּשִׁים,

בְּפִיהֶם פְּלָאִים / וְדִבְרֵי נְבִיאִים, / תְּרוּפַת חֳלָאִים / וְעֹז נֶאֱנָשִׁים.

לְאִישׁ מֵעֲלוּמִים / בְּבָאֵר עֲלֻמִּים / רְגָעָיו כְּיָמִים / וְיָמָיו חֳדָשִׁים,

מְפָאֵר בְּמַעְנָיו / אֲבוֹת שִׁיר וּבָנָיו / וְנָעִים לְפָנָיו / הֲרָרִים וּמָשִׁים,

וְנוֹסְעִים וְחוֹנִים / בְּאַרְצוֹת מְבִינִים / וְהֵם שַׁאֲנַנִּים / בְּבֵיתָם כְּנָשִׁים,

וְלִמֵּד לְרֵעִים / תְּבוּנוֹת וְדֵעִים / וְאֵיךְ הֵם רְצוּעִים / וְיָצְאוּ כְּחָפְשִׁים.

כְּשֻׁלְחָן עֲרַכְתִּיו / כְּלֶחֶם פְּרַשְׂתִּיו / וּמוֹפֵת נְתַתִּיו / לְחַכְמֵי חֲרָשִׁים,

שְׁלַחְתִּיו כְּמַשְׂאוֹת / לְשָׂרֵי בְרִיאוֹת / כְּשֵׂיוֹת בְּרִיאוֹת / וְיֵין הָאֲשִׁישִׁים,

וְלִפְנֵי שְׁוָרִים / – אֲבָל הֵם עֲיָרִים – / כְּלֶחֶם שְׂעוֹרִים / וְכִנְזִיד עֲדָשִׁים,

כְּקֹר מֵי גְבָאִים / לְחֹם הַצְּמֵאִים / וּבִלְבַב מְקַנְאִים / כְּחֻמָּם כְּאִשִּׁים,

מְדַמִּים פְּתָאִים / לְנַכֵּר לְבָאִים / וְלַעְרֹךְ דְּשָׁאִים / לְגֹבַהּ בְּרוֹשִׁים.

הֲיִהְיוּ חֲנֵפִים / בְּתֵבֵל כְּחָפִים / וְשָׂרֵי אֲלָפִים / כְּשָׂרֵי חֲמִשִּׁים?

וְנַחַל עֲדָנִים / כְּמוֹ רוֹשׁ פְּתָנִים / וְעוֹרוֹת גְּדָיִים / כְּעוֹרוֹת תְּחָשִׁים?

וּמִשְׁלֵי חֲכָמִים / כְּהַבְלֵי חֲלוֹמִים / וְאִם אֶלְגְּבִישִׁים / כְּדַר אוֹ גְּבִישִׁים?

וְרֵעֵי מְזִמָּה / כְּרוֹעֵי בְהֵמָה / וְאִם עָשׁ וְכִימָה / כְּעָפָר וְגוּשִׁים?

וְיִפְרוּ צְמָחִים / בְּגַבֵּי צְחִיחִים / וּמַיִם שְׁלוּחִים / הֲיִהְיוּ נְתוּשִׁים?

וְאִם אִישׁ יְאַסֵּף / חֲסָדִים כְּיוֹסֵף / אֲשֶׁר פָּז וְכֶסֶף / לְרַגְלָיו נְדוֹשִׁים,

לְגֶזַע חֲכָמִים / וְנֵצֶר עֲצוּמִים / וְצָעִיר לְיָמִים / וּמַחְכִּים יְשִׁישִׁים,

אֲשֶׁר מִפְעֲלוֹתָיו / כְּפִי עֹז אֲבוֹתָיו / וְאֹרֶךְ עֲבֹתָיו / כְּחֹזֶק שֳׁרָשִׁים,

וְשֹׂרֵף כְּגַחַל / וְשֹׁטֵף כְּנַחַל / וְשֹׁאֵג כְּשַׁחַל / בְּעֶדְרֵי תְיָשִׁים,

קְחָה סוֹד וְחֶבֶר / לְעֶרֶב וְעֵבֶר / וּמָזוֹן וְשֶׁבֶר / לְפִי כָל מְבַקְשִׁים.

עֲלֵה בָם וְהַצְלַח, / הֲכִי הֵם כְּמֶלַח / לְאֹמֶר, וְשֶׁלַח / לְרַדֵּף חֲמוּשִׁים,

וְרוּם עַל גְּבוֹהִים / וְכוֹכְבֵי נְגוֹהִים, / שְׂרֵה עִם אֱלֹהִים / שְׂרֵה עִם אֲנָשִׁים.
