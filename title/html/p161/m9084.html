[אליכם קורא]

אֲלֵיכֶם קֹרֵא

וְאֶתְכֶם יִזְרֶה

קוֹל מִבּוֹרֵא

קְצוֹת הָאָרֶץ

מִפִּיו שֹׁפְטִים

בְּצֶדֶק שֹׁלְטִים

הֵמָּה מְשׁוֹטְטִים

בְּכֹל הָאָרֶץ

פְּנֵה לְךָ קֵדְמָה

וְהַבֵּט יָמָּה

וְשׁוּר כִּי בְּחָכְמָה

יָסַד אָרֶץ

יְעִידוּן נְגֹהִים

עֲלֵיהֶם גְּבֹהִים

כִּי יֵשׁ אֱלֹהִים

שֹׁפְטִים בָּאָרֶץ

וִיהִי לְאַכְזָב

דְּבַר מְתֵי כָזָב

כִּי אֵל עָזַב

אֶת הָאָרֶץ

מַעְלָה מְעוּנָיו

וּמַטָּה עֵינָיו

הַס מִפָּנָיו

כָּל הָאָרֶץ

וְרָאוּ עֵינַי

בְּעַד רַעְיוֹנַי

כִּי לַאדֹנָי

מְצֻקֵי אָרֶץ

טְפוּחִים בְּכַפּוֹ

וְיֹאבְדוּן בְּאַפּוֹ

כִּי מִקִּצְפּוֹ

תִּרְעַשׁ הָאָרֶץ

גָּזַר וּפָעַל

תַּחַת וּמַעַל

וְהוּא עֶלְיוֹן עַל

כָּל הָאָרֶץ

עָפָר צִבֵּר

וְרוּחַ חִבֵּר

אֲדֹנָי דִּבֵּר

וַיִקְרָא אָרֶץ

הֲתַעֲמֹד בְּסוֹדוֹ

וְהַכֹּל בְּיָדוֹ

וּמָלֵא כְבוֹדוֹ

אֶת כָּל הָאָרֶץ

תְּמוֹל אָתִיתָ

וּמָה רָאִיתָ

אֵי הָיִיתָ

בְּיָסְדִי אָרֶץ

הַיוֹם הִנְּךָ

וּמָחָר אֵינְךָ

הַלְמַעַנְךָ

תֵּעָזַב אָרֶץ

שְׁחֵה וְהִכָּנַע

וְלֶכֶת הַצְנַע

בְּתֹם וְהִמָּנַע

מִשּׁוּט בָּאָרֶץ

רֵאשִׁית תֵּבֵל

וְאַחֲרִית הֶבֶל

לְשֹׁכֵן אֵבֶל

כְּגֵר בָּאָרֶץ

יוֹם הִוָּלְדוֹ

וְיוֹם הִפָּרְדוֹ

בְּיַד אֲשֶׁר בְּיָדוֹ

מֶחְקְרֵי אָרֶץ

קָצַב חֶדְלוֹ

וְגָזַר עֲמָלוֹ

מֵאָז עַד לֹא

עָשָׂה אָרֶץ

יָמָיו יִסְפֹּר

וְאֹכֶל יַחְפֹּר

וְיִפֹּל כְּצִפּוֹר

עַל פַּח הָאָרֶץ

הוּא מֵרֶחֶם

בְּעָמָל לֹחֵם

לְהוֹצִיא לֶחֶם

מִן הָאָרֶץ

וְאִם יִמָּנַע

פַּת יִכָּנַע

אָרְצָה וְהוּא נָע

וָנָד בָּאָרֶץ

וְנַפְשׁוֹ נָקְטָה

עֲדֵי פַת לָקְטָה

נָחָה שָקְטָה

כָּל הָאָרֶץ

וּמַעֲבִיד וְנֶעֱבָד

וְנִבְזֶה וְנִכְבָּד

זִכְרָם אָבַד

מִנִּי אָרֶץ

וְאָנָה הֹלְכִים

וְסֻגְּרוּ דְרָכִים

וְהִנֵּה נְבֻכִים

הֵם בָּאָרֶץ

אָוֶן הוּדָם

וְעָמָל כְּבוֹדָם

מִדְּמֵי אָדָם

וַחֲמַס אָרֶץ

שָׁנְנוּ לְשׁוֹנָם

וּפָתְחוּ גְרֹנָם

וְשָׁתוּ עֵינָם

לִנְטוֹת בָּאָרֶץ

לֶאֱסֹף תְּכוּנָה

לָמְדוּ תְבוּנָה

וְלֹא לֶאֱמוּנָה

גָּבְרוּ בָאָרֶץ

וְאֵין כֹּחַ בָּם

עֲשׂוֹת מַה בְּלִבָּם

כִּי לֹא בְּחַרְבָּם

יָרְשׁוּ אָרֶץ

מִדָּה קְצוּצָה

לְכָל אִישׁ חֲרוּצָה

עֵצָה יְעוּצָה

עַל כָּל הָאָרֶץ

רָשׁ וְאִישׁ תְּכָכִים

יַחַד הֹלְכִים

שְׁאוֹל עִם מְלָכִים

וְיוֹעֲצֵי אָרֶץ

שְאוֹל יָנוּמוּ

וְלֹא יָרוּמוּ

וּבַל יָקֻמוּ

וְיָרְשוּ אָרֶץ

תֵּצֵא רוּחָם

וְיִבְאַשׁ רֵיחָם

וּמִהֲרוּ לְשַׁלְּחָם

מִן הָאָרֶץ

לְזֹאת הִזָּכְרוּ

וְאַל תִּתְגַּבְּרוּ

וְגַם הִוָּסְרוּ

שֹׁפְטֵי אָרֶץ

יְראוּ יוֹם יָדוֹן

שְׁגָגָה וְזָדוֹן

מִלִּפְנֵי אָדוֹן

חוּלִי אָרֶץ

אֹמֵר וְעֹשֶׂה

עַל חֵטְא מְכַסֶּה

בְּיוֹם תִּנָּשֵׂא

לִשְׁפֹּט הָאָרֶץ

אוֹת חֹתָמְךָ

בְּכָל עוֹלָמְּךָ

אַדִּיר שִׁמְךָ

בְּכָל הָאָרֶץ.
