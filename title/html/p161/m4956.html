אל בחלום ליל תשמחי דוה

ליוסף (אבו אלחגאג) בן לאיר

אַל בַּחֲלוֹם לֵיל תִּשְׂמְחִי דָוָה

בֹּקֶר וְנִפְרַע מֵאֲשֶׁר הִלְוָה

נַפְשִׁי פְדוּת עֹפֶר אֲשֶׁר טֶרֶם

בֹּקֶר מְבַקֵּר מִחְיַת מִכְוָה

לִפְנֵי צְבָא שַׁחַק חֲצוֹת לַיִל

עָבַר וְאֶל בֵּית אוֹהֲבוֹ נִלְוָה


כִּמְעַט גְּזָלוּהוּ בְנֵי עַיִשׁ

לוּלֵי הֱיוֹתוֹ מִבְּנֵי חַוָּה

דִּמָּה לְדַלּוּתִי חֲלָצָיו וּלְ-

קַדְרוּת לְבָבִי שַׂעֲרוֹ הִשְׁוָה

וּבְאֵשׁ צְלָעַי עִם בְּכִי עֵינַי

מֵאֵשׁ וְשֶׁלֶג לֶחֱיוֹ נִקְוָה

הִתְגַּנְּבוּ עֵינַי רְאוֹת עֵינָיו

בַּלָּאט עֲדֵי יֵדְעוּ הֲיֵשׁ תִּקְוָה

לִרְדּוֹת דְּבַשׁ מִבֵּין בְּדֹלָחִים

תַּחַת שְׂפַת שָׁנִי כְמוּ תִקְוָה

הָהּ עִם שְׁנָתִי עָף כְּרֶגַע וַ-

יֵבשׁ לְבָבִי מֵאֲשֶׁר קִוָּה

מִי לִי בְּפֹתֵר הַחֲלוֹם דּוֹדַי

מִי יִמְשְׁכֵהוּ לִי בְחוּט אַחֲוָה

מִי לִי בְּבֶן רָחֵל וְאֵינֶנּוּ

מִמִּי לְפִתְרוֹן אֶשְׁמְעָה אַחְוָה

אֶפְדֶּה בְכָל חֶמְדָּה מְבַשֵּׂר טוֹב

בָּא בִּבְשׂרָה אַחֲרֵי חִוָּה

לֵאמֹר הֲלֹא יוֹסֵף מְקוֹם יוֹסֵף

לִהְיוֹת צְבִי הַמַּעֲרָב צֻוָּה

הוּא הַחֲלוֹם הַבָּא וּפִתְרוֹנוֹ

הוּא מַחֲמַד עַיִן וְהוּא תַאֲוָה

בֹּשֶׂם תְּבוּנָה וַהֲדַס חָכְמָה

לַחֵן וְלַתּוֹרָה וְלַמִּצְוָה

בָּא הַכְּתָב מִמָּר דְּרוֹר טוּרָיו

אוֹ מֵעֲרוּגוֹת בִּשְׂמְךָ הָלְוָה

מֶתֶק אֲמָרָיו בֶּאֱמֶת הֵעִיד

כִּי מִדְּבַשׁ אַהֲבָתְךְ רָוָה

הַאִם בְּלָאטֵי יָדְךָ רֻקַּם

אוֹ אִם בְּקֶסֶם עֵינְךָ הָתְוָה

סֵפֶר בְּסַפִּיר לֶחְיְךָ סֻפָּר

וּדְיוֹ בְנֹפֶךְ שַׂעֲרָךְ חֻוָּה

אֵלֶּה דְבָרַי מַעְיְנֵיהֶם בָּךְ

הִנֵּה תְהִלָּתִי לְךָ נָאוָה

הִנֵּה נְאֻם נִמְכָּר בְּאַהֲבָתְךָ

מִיּוֹם עֲלוֹתוֹ בֵּין בְּנֵי עַלְוָה

בָּא מֵאֱדוֹם אֶל מַעֲרַב קֵדָר

וּפְנֵי מְגַמָּתוּ לְךָ שִׁוָּה

לֹא מִמְּךָ הִתְאַחֲרוּ רַגְלָיו

אַךְ אֶת נְתִיבוֹתָיו זְמָן עִוָּה

לָכֵן קְרָא עַל נַפְשְׁךָ שָׁלוֹם

מֵאֵת יְדִיד יַרְבֶּה בְךָ חֶדְוָה

וּלְרַב וְתַלְמִידָיו קְרָא כָזֹאת

מֵאִישׁ הֱיוֹת בִּמְשָׁרֲתָיו אִוָּה

וּלְכֻלְּכֶם יִקְרָא אֱלֹהֵי עַד

רֹב עֹז וְרֹב שִׂמְחָה וְרֹב חֶדְוָה.
