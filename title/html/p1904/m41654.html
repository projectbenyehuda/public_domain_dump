[בַּת צִיּוֹן עַד אָנָה נוֹדֶדֶת כַּמְּלוּנָה (תחנה)]

תְּחִנָּה


בַּת צִיּוֹן עַד אָנָה / נוֹדֶדֶת כַּמְּלוּנָה

יוֹצֵאת אָנֶה וְאָנָה / מְשׁוֹטֶטֶת בְּכָל-פִּנָּה

יָדֶיהָ פְרוּשׂוֹת לִמְעוֹנָה / נוֹשְׂאָה עֵינֶיהָ לְקוֹנָהּ

אַתָּה תָקוּם תְּרַחֵם / צִיּוֹן כִּי עֵת לְחֶנְנָהּ.


אֲהוּבָה מִכָּל-עֲלָמוֹת / אָחֲזָה רְעָדָה וְאֵימוֹת

בְּזָכְרָהּ עִתּוֹת קְדוּמוֹת / דָּרְכָה עַל-כָּל-בָּמוֹת

גָּבְרָה בְעֹז וְתַעֲצוּמוֹת / לַעֲלוֹת מַעֲלוֹת רָמוֹת

וְעַתָּה רַבַּת שָׁכְנָה / הַיָּצִיעַ הַתַּחְתֹּנָה.


דּוֹדָהּ אַלּוּף נְעוּרֶיהָ / אָז בִּימֵי נְעוּרֶיהָ

הֵעִיר וְדָפַק שְׁעָרֶיהָ / לְיַעֲדָהּ לוֹ לְשַׁחְרְרֶהָ

וּמַלְכוּת עֶגְלָה וְשָׂרֶיהָ / הִכְנִיעַ בַּעֲבוּרָה

וְעַתָּה אוֹיְבָהּ עִנָּהּ / וַיִּקְרָא שְׁמָהּ שִׂטְנָה.


זֶה דַרְכָּה בְּמֵישָׁרִים / בְּיָד רָמָה לְעֵין זָרִים

חָלְפָה וְעָבְרָה גְזָרִים / בְּשִׂמְחַת לֵבָב וּבְשִׁירִים

טָעֲמָה לֶחֶם אַבִּירִים / וְלָמְדָה חֻקִּים יְשָׁרִים

וְעַתָּה שָׂבְעָה בָּשְׁנָה / שֹׁרֶשׁ פֹּרֶה רֹאשׁ וְלַעֲנָה.


יָרְשָׁה אֶרֶץ חֶמְדָּה / נַחֲלַת צְבִי מְיֻחָדָה

כָּל-הָאָרֶץ חָרְדָה / וְהִתְגָּעֲשָׁה מִפַּחְדָהּ

לְכָל-בָּנוֹת עָלְתָה בְהוֹדָהּ / כְּמוֹ בַּת-מֶלֶךְ כְּבוּדָּה

וְעַתָּה שָׁכְנוּ בְמַחֲנָהּ / תַּנִים וּבְנוֹת יַעֲנָה.


מֵרֹב שֶׁקֶט שַׁלְוָתָהּ / וּמִמִּשְׁמַן מַרְעִיתָהּ

נִאֲצָה לְהָפֵר בְּאִוַּלְתָּהּ / בְּרִית אֱלֹהֵי יְשׁוּעָתָהּ

סָרֲרָה דָּבְקָה בְחַטָּאתָהּ / לֹא זָכְרָה אַחֲרִיתָהּ

וְעַתָּה כָשְׁלָה בַעֲוֹנָהּ / וְיָרַד הֲדָרָהּ וַהֲמוֹנָהּ.


עֲשָׁקוּהָ בְנֵי עַוְלָה / מִיּוֹם צֵאתָהּ בַּגּוֹלָה

פּוֹעֲרִים פִּיהֶם לְמוּלָהּ / שׁוּבִי שׁוּבִי סְגֻלָּה

צְבִי תִפְאַרְתֵּךְ בָּלָה / וּמַה-לָּךְ עוֹד לְהוֹחִילָה

וְעַתָּה בַת-עוּץ עֲדִינָה / הָאֱלֹהִים אֲשֶׁר נְתָנָהּ.


קוֹלָהּ נָשְׂאָה בִנְדָבָה / מַה-תְּעוֹרֲרוּ אֶת-הָאַהֲבָה

רְאוּ זֹאת הָיְתָה סִבָּה / מִפְּנֵי מוֹצִיאֵי דִבָּה

שְׁטָרִי קוֹדֵם וּכְתֻבָּה / בְּחוֹתַם כָּל-חֹזֶה נִבָּא

וְעַתָּה יִזְכֹּר בְּרִית יְשָׁנָה / אֱמֶת לֹא-יָשׁוּב מִמֶּנָּה.


תַּמָּה תַּמּוּתָה בְּהִתְחַזְּקָהּ / וְעַל-דּוֹדָהּ מִתְרַפְּקָה

בְּאַלּוּף נְעוּרֶיהָ לְדָבְקָה / וְדַלְתֵי דֹדָיו לְדָפְקָה

נִדְבַת רְצוֹנוֹ הֵפִיקָה / לְבִלְתִּי הֱיוֹת עוֹד לְפוּקָה

וְעַתָּה תָשׁוּב אֶל-מְלוֹנָהּ / וְלֹא יִכָּרֵת מְעוֹנָה.


יֵאָמֵן דְּבַר כָּל-נָבִיא / וְצוֹפֶה אֲשֶׁר חָזוּ-בִי

מַלְכֵּךְ יָבוֹא לְהָבִיא / נִדַּחֲךְ מִבֵּית הַשְּׁבִי

נְוֵה-צֶדֶק אֶרֶץ צְבִי / לְבָדָד לָבֶטַח תֵּשְׁבִי

וְעַתָּה שׁוּבִי וּבֹאִי נָא / וְאֵרַשְׂתִּיךְ לִי בֶּאֱמוּנָה.


חוֹרְפַי אָשִׁיב בְּמִלַּי / הַמַּגְדִּילִים עָלָי

זֹאת הַפַּעַם בְּמַכְלֻלַי / יִלָּוֶה אִישִׁי אֵלָי

קָרְבָה בָּאָה שְׁנַת גְּאוּלַי / וְיוֹם נִקְמַת הֵיכָלָי

וְעַתָּה אַחֲרֵי זִקְנָה / הָיְתָה-לִּי עֶדְנָה.
