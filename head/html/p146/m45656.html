מִנְחַת כִּפּוּר

חתימה: משתא אלשבזי


מִנְחַת כִּפּוּר תֵּן מְנוּחָה / צוּרִי, לְאַיָּלָה שְׁלוּחָה

מֵחֵטְא וּמֵחֶרֶב פְּתוּחָה / רוּחִי וְנִשְׁמָתִי בְּטוּחָה

בְּחֶסֶד אֵל שַׁדַּי מְשׁוּחָה / כִּי מִכִּסְאוֹ הִיא לְקוּחָה

לְהָאִיר גּוּף בְּאַפּוֹ נְפוּחָה / הֻגְלָד מִלֵּחָה סְרוּחָה

חָטָא וַיִּפּוֹל בְשׁוּחָה / וְשָׁבָה נֶפֶשׁ נֶאֱנָחָה

יְוֹיָ שְׁמָעָה יְוֹיָ סְלָחָה:


שׁוּבִי יְחִידָתִי שׁוּבִי / מֵהִרְהוּר לְבָבִי

אַחֲרֵי עֵינִי מָרַד בִּי / גָּרַם חַטָּאתִי וְחוֹבִי

כִּי לוּלֵי אֱלֹהֵי אָבִי / אֲזַי אָבַדְתִּי בְּמַכְאוֹבִי

וּבְעוֹד נִשְׁמָתִי בִּי / אֲיַחֵד אֱלֹהֵי אָבִי

יְמִין אֵל חֶסֶד אֵל הַרְבִּי / וְלִי בְּסֵפֶר חַיִּים כִּתְבִי

עוֹרְרִי זְכוּת מֹשֶׁה רַבִּי / יָם הֵשִׂים כַּמֶּרְקָחָה:


תָּגִיל נַפְשִׁי אִם בָּאתִי / בְּכִפּוּר וְשַׁבָּת מְנוּחָתִי

עוֹלַת תָּמִיד תְּפִלָּתִי / וְשִׂיחָתִי הִיא מִנְחָתִי

יְוֹיָ נֶגְדְּךָ כָּל תַּאֲוָתִי / שִׂים רוֹב שָׁלוֹם לַעֲדָתִי

וּבְצֵל כַּנְפֵי רַעְיָתִי / תָּלִין נַפְשִׁי בְּהַרְוָחָה:


אַהֲבַת נַפְשִׁי וְרַעְיוֹנִי / לִלְמוֹד תּוֹרַת הֶגְיוֹנַי

מִיּוֹם נִתְּנָה מִסִּינַי / הוֹסִיף אוֹר שִׂכְלִי וְעֵינִי

עָלָה מֹשֶׁה רֹאשׁ גְּאוֹנִי / וְהִגְבִּיל עַמִּי וּזְקֵנִי

[….נִי] / וְנַעֲרוֹ צִוָּה בְּהַצְלָחָה:


אֵל חַי בְּחָרוֹ וְזִכָּהוּ / וְאָמַר לְמֹשֶׁה: צַוֵּהוּ

קוּם חַזְּקֵהוּ וְאַמְּצֵהוּ / וּמֵהוֹדְךָ הַלְבִּישֵׁהוּ

וְהִתְבּוֹנֵן כִּי אֲנִי אֲנִי הוּא / בְּרֹאשׁ קְהַל עַמִּי אַעֲלֵהוּ

אֲחַלְּצֵהוּ וַאֲכַבְּדֵהוּ / בְּצָפוֹן וְדָרוֹם יָם וּמִזְרָחָה:


לָקַח יְהוֹשֻׁעַ מְשָׁחוֹ / הֶאֱצִיל עָלָיו מֵרוּחוֹ

וְכָל עַם קְדוֹשָׁיו הוֹכִיחוֹ / כִּי הוּא זַךְ וְיָפֶה כּוֹחוֹ

לֹא כָהֲתָה עֵינוֹ בְּנוּחוֹ / וּבְמוֹתוֹ לֹא נָס לֵחוֹ

הוּא נֶאֱמָן אֵל וּשְׁלוּחוֹ / בְּמַרְאֵה הַסְּנֶה הִבְטִיחוֹ

[…חוֹ] / לְדַבֵּר בִּנְבוּאָה נְכוֹחָה:


שְׁמַע יָהּ עָם בָּא לְהִתְוַדּוֹת / אֲסִירִים תָּחִישָׁה לִפְדּוֹת

בַּעֲבוּר שְׁלֹשׁ עֶשְׂרֵה מִדּוֹת / וּמַנְעִימִים שִׁיר לְךָ לְהוֹדוֹת

וּמִצְווֹת סִגַּלְתָּנוּ וְעֵדוֹת / לֵידַע בְּכָל מָשָׁל וְסוֹדוֹת

נָשִׁירָה שִׁיר עַל יְדִידוֹת / בְּשִׂמְחָה וְצַהֲלָה וְהַצְלָחָה:


בְּמִנְחַת כִּפּוּר הָבִינוּ / זֶרַע יְשׁוּרוּן וְהַאֲזִינוּ

שִׁיתוּ לִבְּכֶם וְהַאֲמִינוּ / וְלֵאלֹהִים חַי הַרְנִינוּ

זֶה הַיּוֹם אֲשֶׁר קִוִּינוּ / וְלִסְלִיחָה הִתְאַוִּינוּ

בְּשֶׁקֶט וּבְשַׁאֲנָן לִינוּ / וְחַטָּאת עַמִּי נִשְׁכָּחָה:


זִכְרוֹן טוֹב לִבְנֵי נְדִיבִים / הַיּוֹם אַתֶּם נִצָּבִים

בְּשַׁבָּתוֹת וְיָמִים טוֹבִים / מִקְרָא קֹדֶשׁ נֶאֱהָבִים

שׁוּבוּ בָּנִים שׁוֹבָבִים / בְּסֵפֶר חַיִּים נִכְתָּבִים

בְּקֶרֶב עַמִּים רַבִּים / אֶחָד מֵהֶם לְמִשְׁפָּחָה:


יָקוּמָה בְּעֶזְרָתֵנוּ / אֱלֹהֵי קֶדֶם וְיִרְצֵנוּ

מְהֵרָה נַעֲלֶה לְאַרְצֵנוּ / בְּחֶסֶד וְרַחֲמִים יְנַהֲלֵנוּ

יָאֵר פָּנָיו אֵלֵינוּ / בְּמַעְגְּלֵי צֶדֶק יַנְחֵנוּ

מַה נָּעִים גּוֹרָלֵנוּ / שֶׁמֶשׁ צְדָקָה זָרָחָה:
