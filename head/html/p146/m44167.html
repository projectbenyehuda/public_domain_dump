שחר אהלל צורי ואעיר כל נרדם

חתימה: שלם ברב יוסף אלשבזי


שַׁחַר אֲהַלֵּל צוּרִי, וְאָעִיר כָּל נִרְדָּם

לְיַחֵד אֶת שְׁמוֹ לֹא יְשַׁנּוּ אֶת תַּפְקִידָם

וְהִיא סִבַּת כָּל סִבָּה וְעִלַּת יְסוֹדָם

בִּתְפִלַּת הַשַּׁחַר יִזְכֹּר לָנוּ זְכוּת אָדָם:


לְחַסְדּוֹ מְיַחֲלִים הַגּוּפוֹת וְהַנְּפָשׁוֹת

וּבָחַר בְּיִשְׂרָאֵל אֲשֶׁר נַפְשׁוֹתָם מְקֻדָּשׁוֹת

וּמֵאוֹר שְׁכִינָתוֹ הֵמָּה מְלֻבָּשׁוֹת

בִּתְפִלַּת הַשַּׁחַר, יִזְכֹּר לָנוּ זְכוּת שֵׁת:


מְרוֹמָם עַל כָּל בְּרָכָה וּתְהִלָּה נֶאְדָּרִי בַּכּוֹחַ

בָּרָא הַכֹּל בְּשִׁשָּׁה יָמִים וְיוֹם הַשְּׁבִיעִי יוֹם מָנוֹחַ

בּוֹ תַּכְלִית הַכֹּל וְתָצִיץ הַנְּפָשׁוֹת בְּסוֹד שַׁדַּי לִזְרֹחַ

בִּתְפִלַּת הַשַּׁחַר, יִזְכֹּר לָנוּ זְכוּת נֹחַ:


בְּרֵאשִׁית בָּרָא מַלְאָכִים וְגַלְגַּלִּים וְנָגְהָם

וְכִוַּן אֶת סֵדֶר הַכּוֹכָבִים וְגָבְהָם

וְגָלַל אוֹר מִפְּנֵי חֹשֶׁךְ לְהַגִּיהָם

בִּתְפִלַּת הַשַּׁחַר, יִזְכֹּר לָנוּ זְכוּת אַבְרָהָם:


יוֹדוּהוּ גְּדוּדָיו וְאִיֵּי הַיָּם מִמֶּרְחָק

וְעַמּוֹ אֲשֶׁר בָּחַר לוֹמְדֵי תּוֹרָתוֹ אֲשֶׁר חָק

וְיָחוֹן בְּרַחֲמָיו, לְעַם עָנִי הַנִּדְחָק

בִּתְפִלַּת הַשַּׁחַר יִזְכֹּר לָנוּ זְכוּת יִצְחָק:


רוֹצֶה עֲבוֹדָה יִשְׁמֹר עַל נַחֲלָתוֹ בְּלֶכְתָּם

וְיִשְׁמַע וְיַקְשִׁיב בְּעֵת רָצוֹן תְּפִלָּתָם

וּבְחַסְדּוֹ וְטוּבוֹ יְעוֹרֵר אֶת אַהֲבָתָם

בִּתְפִלַּת הַשַּׁחַר יִזְכֹּר לָנוּ זְכוּת יַעֲקֹב אִישׁ תָּם:


בְּשֶׁקֶט וְשַׁאֲנַן תִּשְׂמַח יְחִידָתִי וְתָרוֹן

וְתַחְמֹד לֵישֶׁב בְּצִלּוֹ, חֲבַצֶּלֶת הַשָּׁרוֹן

אָז יִכְתְבֵנוּ בְּסֵפֶר חַיִּים וְזִכָּרוֹן

בִּתְפִלַּת הַשַּׁחַר, יִזְכֹּר לָנוּ זְכוּת אַהֲרֹן:


יְקַבֵּל תְּפִלָּתֵנוּ כְּקָרְבַּן אִשֶּׁה

יִזְכּוֹר זְכוּת אֲבוֹתֵינוּ וּבְרִיתוֹ יִנְשֶׁה

וְיַצִּיל אוֹתָנוּ מִיַּד אֲדוֹנִים קָשֶׁה

בִּתְפִלַּת הַשַּׁחַר יִזְכֹּר לָנוּ זְכוּת מֹשֶׁה:


וִדּוּיֵנוּ יַאֲזִין וְיַצִּילֵנו מִגּוֹי אַכְזָר

וְנִשְׂמַח, וּבְשִׂמְחָתֵנוּ אַל יִתְעָרֵב זָר

לָבֵשׁ יְוֹיָ עוֹז הִתְאַזָּר

בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת אֶלְעָזָר:


סוֹמֵךְ נוֹפְלִים אֲשֶׁר עַל עַמּוֹ חָס וְ

וְהִצִּילוֹ עַל יַד מֹשֶׁה רֹאשׁ הַנְּבִיאִים הַמְיוּחָס

וְעֵשָׂו וְחֵילוֹ יְהֶא בְהַר שֵׂעִיר נִפְחָס

בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת פִּנְחָס:


פְּדוּת יִָחִישׁ צוּרֵנוּ לְעָם בְּדַרְכּוֹ צָנוּעַ

וְיַשְׁפִּיעַ בְּרָכָה עַל כָּל זֶרַע זָרוּעַ

וְעָנִי וְאֶבְיוֹן יְהֶא שְׁטַר חוֹבוֹ קָרוּעַ

בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת יְהוֹשֻׁעַ:


אָז יָגִיל יַעֲקֹב יִשְׂמַח יִשְׂרָאֵל

וְנִשְׁמַע בְּשׂוֹרָה כִּי בָא לְצִיּוֹן גּוֹאֵל

תָּגִיל יְחִידָתִי יוֹם בִּנְיַן אֲרִיאֵל

בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת עָתְנִיאֵל:


לִתְפִלָּתֵנוּ יַאֲזִין וֶעֱנוּת עָנִי אַל יִבְזֶה

וְעָלֵינוּ מֵי טוֹהַר מֵחֲסָדָיו יַזֶּה

אָז נָגִיל בִּמְנוּחָתֵנוּ כְּהַיּוֹם הַזֶּה

בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת שְׁמוּאֵל הַחוֹזֶה:


שַׁדַּי אֲשֶׁר שׁוּלָיו אֶת הַהֵיכַל מְלֵאִים

וְכָל זֶרַע קְדוֹשָׁיו בְּצִלּוֹ הֵם נֶהְנָאִים

נוֹרָא וְאָיוֹם וְרָם עַל כָּל גֵּאִים

בִּתְפִלַּת הַשַּׁחַר, יִזְכֹּר לָנוּ זְכוּת נְבִיאִים:


בְּטוּבוֹ יְשַׂגְּבֵנוּ אֱלֹהֵי עוֹלָמִים

וִינַקֵּנוּ מִכָּל עָווֹן וְחֵטְא וַאֲשָׁמִים

וְיַצִּילֵנוּ מֵאִישׁ מִרְמָה וְדָמִים

בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת חֲכָמִים:


זְמָן יְשׁוּעָתֵנוּ יְקָרֵב, וִיבָרְכֵנוּ בַּבְּרָכוֹת

אֲשֶׁר אֶל תּוֹךְ יָם הָאַחֲרוֹן מְשׁוּכוֹת

וִיכוֹנֵן בֵּית קָדְשֵׁנוּ וְהַלְּשָׁכוֹת

בִּתְפִלַּת הַשַּׁחַר יִזְכֹּר לָנוּ זְכוּת שׁוֹנֵי הֲלָכוֹת:


יְזַמֵּן פַּרְנָסָתֵנוּ וְיָסִיר כָּל מְגִנָּה

וְיִשְׁפּוֹת שָׁלוֹם לָנוּ מִכָּל צַד וּפִנָּה

וְיִשְׁלַח לָנוּ מַלְאַךְ חֶסֶד הַמְּמֻנָּה

בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת אַנְשֵׁי מִשְׁנָה:


יִשְׁמַע תְּפִלָּתֵנוּ, בְּיָחִיד וּבַצִּבּוּר

יִתְבָּרַךְ שְׁמוֹ בּוֹרֵא נִיב שְׂפָתַיִם לְדִבּוּר

אֲשֶׁר גִּלָּה לָנוּ אֶת סוֹד הַעִבּוּר

בִּתְפִלַּת הַשַּׁחַר יִזְכֹּר לָנוּ זְכוּת סִפְרֵי הַחִבּוּר:


סֻכַּת שְׁלוֹמוֹ יִפְרֹס עַל גּוֹלִים הַנִּלְבָּטִים

מְקַיְמֵי מִצְווֹתָיו בִּכְלָלִים וּפְרָטִים

אֲשֶׁר נָתַן לָהֶם תּוֹרָה חֻקִּים וּמִשְׁפָּטִים

בִּתְפִלַּת הַשַּׁחַר יִזְכּוֹר לָנוּ זְכוּת שְׁנֵים עָשָׂר שְׁבָטִים:
