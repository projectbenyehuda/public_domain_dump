עוֹרְרִי בַת

חתימה: שלם בן שבזי משתא


עוֹרְרִי בַת נָדִיב / עָם פְּזוּרֵי שִׁבְיוֹן

שִׁיר וְהַלֵּל תֵּיטִיב / מְשַׁבֵּחַ לְאֵל עֶלְיוֹן

כִּי לְבָבִי מַקְשִׁיב / הוּא וְשֵׂכֶל רַעְיוֹן

אֶחְפְּצָה סוֹד הֶגְיוֹן / בַּחֲצַר הַר צִיּוֹן:


שׁוֹבְבִי עָם גּוֹלִים / רַעְיָה כִּי שָׁאוּ

לָךְ בְּאַהֲבָה חוֹלִים / אָז כְּצִיץ יִזְרָחוּ

חַזְּקִי נִכְשָׁלִים / זֶה בְזֶה יִתְאָחוּ

וְתִמְכִי כָּל רִפְיוֹן / בַּחֲצַר הַר צִיּוֹן:


לִי תְהֶא תַקָּנָה / תּוֹךְ סְגֻלַּת עַמִּי

רַחֲמָן עַד אָנָה / צַר יְהֶא מַשְׂטִינִי

מִמְּךָ תַּכְּנָה / קוּם דְּרוֹשׁ נָא דָמִי

אָז יְהֶא לִי רִתְיוֹן / בַּחֲצַר הַר צִיּוֹן:


מִמְּקוֹר הַצֶּדֶק / עַל פְּזוּרִים שׁוֹרָה

חַזְּקָה כָּל בֶּדֶק / דּוֹד וְנִשְׂמַח פּוּרָה

אֵיךְ אֲהוּבִי חָמַק / וַאֲנִי צַמְרוּרָה

רַחֲמָה דָּל אֶבְיוֹן / בַּחֲצַר הַר צִיּוֹן:


כָּךְ בְּטוּחִים יַחַד / עָם אֲשֶׁר חָקַקְתָּ

כִּי אֲחָזָם פַּחַד / יוֹם אֲשֶׁר הִרְחַקְתָּ

בֶּן אֲמָתִי הַכְחֶד / כִּי מְרוֹמִים אַתָּה

כּוֹנַנְתָּ אַפִּרְיוֹן / בַּחֲצַר הַר צִיּוֹן:


נַחֲמָה דָּל צוֹפֶה / חַסְדְּךָ צוּר הַכֹּל

כִּי שְׁאוֹן צָר חוֹפֶה / חוּץ לְאוֹהֶל בַּת כֹּל

קוּם צְלַח דּוֹד יָפֶה / וַעֲצָתוֹ סַכֵּל

יֹאחֲזֶנּוּ שִׁגָּיוֹן / בַּחֲצַר הַר צִיּוֹן:


שֵׁם אֱלֹהַי עֶזְרִי / מִכְּלַל הַקָּמִים

זִיו סְגֻלַּת יָשְׁרִי / חַי עֲדֵי עוֹלָמִים

חִיל מְשַׂנְאַי יַפְרִיא / אָז יְהוּ נִדְהָמִים

יִשְׁכְּרוּ בַּחֲמַת יַיִן / בַּחֲצַר הַר צִיּוֹן:


בֵּאלֹהַי עֶזְרִי / מִכְּלַל זֵדוֹנִים

בַּת נְדִיבִים אִמְרִי / לוֹ בְּשִׁיר וּרְנָנִים

יַעֲלֶה טוֹב זִכְרִי / בֵּין שְׂרִידִים חוֹנִים

סוֹדְרֵי דָּת הֶגְיוֹן / בַּחֲצַר הַר צִיּוֹן:


זָרְזוּ שִׂכְלֵיכֶם / כָּל בְּנֵי אִישׁ תָּמִים

אֶל מְחוֹז חֶפְצֵיכֶם / רָם עֲלֵי נֶעְצָמִים

הוּא כָּעֵת יַנְחֵיכֶם / עִם גְּבוּלוֹת עַמִּים

נֶחְלְקוּ בְּחִזָּיוֹן / בַּחֲצַר הַר צִיּוֹן:


יוֹם לְיוֹם יַבִּיעַ / מֵאֱלֹהַי אוֹמֶר

אָז לְבֵן יוֹשִׁיעַ / בַּעֲבוּר דָּת שָׁמַר

נִפְזָרִים יַשְׁמִיעַ / שִׁיר וְהַלֵּל זֶמֶר

נִמְצְאָה סוֹד חֶבְיוֹן / בַּחֲצַר הַר צִיּוֹן:


מְהֵרָה בִגְאֻלָּה / מִמְּעוֹן קָדְשֶׁךָ

רַחֲמָה לִסְגֻלָּה / דּוֹרְשָׁה חַסְדֶּךָ

תֵּעָנֶה בִתְהִלָּה / נִכְסְפָה אֵלֶיךָ

תַּחְפְּשָׂה […] / בַּחֲצַר הַר צִיּוֹן:


שַׁחֲרִי נִשְׁמָתִי / בַּתְּפִלָּה עִמְדִי

עוֹרְרִי אַהְבָתִי / סוֹד בְּתוֹרָה לִמְדִי

כִּי בְזֶה חָמַדְתִּי / מִתְּחִלַּת עָמְדִי

אֶמְצְאָה גִּלָּיוֹן / בַּחֲצַר הַר צִיּוֹן:


תּוֹךְ אֲרוֹן הַקֹּדֶשׁ / אוֹר מְהֻדָּר נִגְלָל

תִּשְׁמְעָה קוֹל רַעַשׁ / מִמְּפִיצֵי מַהְלָל

אַךְ אֲהוּבִי הָחֶשׁ / דַּע פְּרָטִים וּכְלָל

מִלּוֹמְדֵי הִגָּיוֹן / בַּחֲצַר הַר צִיּוֹן:


אֶחְתְּמָה שִׁיר נוֹעַם / מִשְּׂפָתַי נִפְצָה

בּוֹ תְנוּעַת טַעַם / סוֹד לְצוּרָה נִרְצָה

יָהּ פְּדֵה לֵב הוּעָם / עַמְּךָ חֵן יִמְצָא

תֶּן לְעָנְיִי נִשְׁיוֹן / בַּחֲצַר הַר צִיּוֹן:
