אֲיַחֵל חֶסֶד

חתימה: אלשבזי


אֲיַחֵל חֶסֶד אֵל קוֹנִי / אָשִׁירָה שִׁיר לַאדוֹנִי

וַאֲנִי גּוֹלֶה בַּעֲווֹנִי / כִּי הִטְעַנִי הַצְּפוֹנִי

לְשִׂכְלִי טִבַּע

בְּקִרְיַת אַרְבַּע

וְאָהֳלוֹ קָבַע

וְקָלוֹן שָבַע

אֲלוּלֵי בַת שֶׁבַע

בָּאַהֲבָה תַנְחֵנִי / דְּרָכֶיךָ יְוֹיָ הוֹדִיעֵנִי

וְאוֹרְחוֹתֶיךָ לַמְּדֵנִי:


לְבָבִי הִכּוֹן לְעֻמָּתִי / וְאַשְׂכִּילְךָ דָּת תּוֹרָתִי

אוּלַי יְקַבֵּל תְּשׁוּבָתִי / אֱלֹהֵי עָזִּי וְזִמְרָתִי

הַמְרוֹמָם עַל תְּהִלָּתִי / מוֹחֶה פְשָׁעַי וְחַטָּאתִי

יְהֶא עִיר וּמִגְדָּל

וּמִשְׂגָּב לַדָּל

בְּעֵת כִּי יֶחְדָּל

בְּשֵׂכֶל נִבְדָּל

אָז יִשְׁתַּדָּל

יָאִיר נְתִיבָתִי / יָקוּם בְּעֶזְרָתִי

[…] וְיוֹשִׁיעֵנִי:


שָׁמֵם לְבָבִי בְּתוֹךְ גּוֹלָה / בְּהִרְהוּר חֵטְא נַפְשִׁי חוֹלָה

יִרְצֶה שִׂיחִי יָהּ כְּעוֹלָה / וְיִדְלֶה נֶפֶשׁ צוֹלֵלָה

חַסְדּוֹ וְטוּבוֹ אֲיַחֵלָה / עַל בֶּן אֲמָתוֹ יַחְמוֹלָה

יְקַנֵּא לִשְׁמוֹ

וְיַצִּיל עַמּוֹ

וְיָסִיר זַעְמוֹ

מִקְּהַל עַמּוֹ

וְיַךְ צָרֵימוֹ

וְיָסִיר טַעְמוֹ

וְלָנוּ חֶסֶד יִגְמוֹלָה / יָאִיר חוֹשֶׁךְ וַאֲפֵלָה

וּבִישׁוּעָתוֹ יְבַשְּׂרֵנִי:


בֵּין הַהֲדַסִּים שָׁמַעְתִּי / דּוֹדִי דּוֹפֵק וְשָׂמַחְתִּי

וּמֵרוֹב פְּשָׁעַי נִכְלַמְתִּי / מִלְּהַקְבִּילוֹ רָחַקְתִּי

אֲבָל מֵרָחוֹק צָעַקְתִּי / וּבֵין הַשָּׁבִים עָמַדְתִּי

וְקוֹל אוֹמֵר לִי

בְּעֵינִי שִׂכְלִי

הֻרְצָה חֶדְלִי

וְהוּרָם דִּגְלִי

הוּא נֵר לְרַגְלִי

אֲשֶׁר בְּאוֹתוֹת חָקַקְתִּי / כִּי בֵאלֹהַי דָּבַקְתִּי

קָרָאתִי וַיַּעֲנֵנִי:


זִרְמֵי יַַרְדֵּן נִלְכָּדוּ / יוֹם אֲבוֹתַי בּוֹ עָמָדוּ

בְּיֵחוּד יְוֹיָ נִצְמָדוּ / וְלַנִסִּים נִתְעַתָּדוּ

אֶרֶץ וְהָרִים רָעָדוּ / בְּמַתָּן תּוֹרָה הֻפְקָדוּ

וּמַלְאֲכֵי מַעְלָה יָרָדוּ / בְּהַר סִינַי הִתְפָּרָדוּ

וּמֹשֶׁה עָמַד

בְּלוּחוֹת יָרַד

וְתוֹרָה לִמַּד

אֲשֶׁר בָּהּ חָמַד

וּבָהּ שְׁבָטַי לָמְדוּ / בְּסוֹד תַּרְיַ״ג נִתְיַחֲדוּ

וּבִזְכוּתָהּ תִּסְעֲדֵנִי:


יוֹם עָשׂוֹר לְחֹדֶשׁ תִּשְׁרֵי / אָשׁוּב לֵאלֹהֵי עֶזְרִי

בּוֹ בַיּוֹם בָּאוּ לְחֶדְרִי / לוּחוֹת שְׁנִיּוֹת סוֹד יָשְׁרִי

בְּעֵת רָצוֹן יַעֲלֶה זִכְרִי / לְטוֹבָה וְלֹא יֵבוֹשׁ שִׂבְרִי

בְּסוֹד הָאַהֲבָה

קוֹל דּוֹדִי בָּא

בְּחֵן וּבְחִבָּה

בְּשִׂמְחָה רַבָּה

וְשַׁלְוָה וְטוֹבָה

וּבְצֵל סֻכָּתוֹ סִתְרִי / עֲבוּר אַבְרָהָם הָעִבְרִי

וְיִצְחָק וֶיַעֲקֹב רֹאשׁ הֲמוֹנִי:
