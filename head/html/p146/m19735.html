אַהְבַת גְּבֶרֶת נִכְבְּדָה תָּאִיר לְעֵין שִׂכְלִי

חתימה: אלשבזי


אַהְבַת גְּבֶרֶת נִכְבְּדָה / תָּאִיר לְעֵין שִׂכְלִי וְרַעֲיוֹנִי

וַאְנִי לְיָפְיָהּ אֶחְמְדָה / כִּי הִיא בְּגָלוּתִי תְנַחֲמֵנִי

נַפְשִׁי כְצִפּוֹר בּוֹדְדָה / תַּקְבִּיל בְּכָל לַיְלָה פְנֵי אֲדוֹנִי

כִּי לָהּ מְלַוֶּה שַׂר צְבָא

לַעְלוֹת בְּחֵי הָאַהֲבָה

בֵּין הַכְּרוּבִים נִצְּבָה

וּתְחַדְּשָׁה גִילִי וְרֹב שְׂשׂוֹנִי

מִגּוּף טְבָעִי נִפְרְדָה / וּבְעוֹד עֲלוֹת שַׁחַר תְּקוֹמְמֵנִי:


לִאלנַּפס גַאיַה מֻדרִכַּה / מַעשׁוּקְהַא אַלעַקל פִי הְוַאהַא

בַּין אַלדְּרַארִי סַאלִכַּה / תַעלוּ אִלַי אַלכֻּרסִי תְכֻׄץ רְצַׄאהַא

בִּאלדִּין וַאלאִימַאן מֻמסִכַּה / וַאלגִׄסם פִי כַׄטִּ אלשְּׁמַאל בְּלַאהַא

דַאיֵם חְיַאתֻה מֻפסְדַּא / מֻזגִׄי אַלנֻּפוּס בִּאלגַׄהל וַאלהֲוַאנִי

וַאלנַּפס תַקצֻד עִלמְהַא

תַזכּוּ וְתַעבֻּד רַבְּהַא

תַשׁתַאק אִלַי בֻּסתַאנְהַא

תַנגׄוּ מִן אַלאַתַׄאם וַאלאִמתִחַאנִי

תַרגַׄע לִדַאר אַלאִבּתִדַא / תַלבַּס לִנוּר אַלעַקל בִּאלגִׄנַאנִי:


שַׁאְלִי אֲיֻמָּתִי לְדוֹד / יִרְצֶה עֲלֵי גוֹלִים שְׁהֵן סְגֻלָּה

בֵּין הַהֲדַסִּים יַעֲמוֹד / יָשִׁיב שְׁבוּתֵנוּ כְּבַתְּחִלָּה

שׁוּבִי יְחִידָה מִנְּדוֹד / עָלַי בְשַׁחַר סַדְּרִי תְפִלָּה

עִזְבִי לְשִׁפְחָה מוֹרְדָה / בִּדְבַר תְּשׁוּבָה עוֹרְרִי הֲמוֹנִי

אֶל טוּב אֱלֹהַי תִּנְהֲרִי

שִׁירָה חֲדָשָׁה דַבְּרִי

יֶשַׁע אֱלֹהִים סַבְּרִי

אוּלַי בְּעֵת רָצוֹן יְשַׂגְּבֵנִי

אַזְכִּיר שְׁמוֹ וַאְיַחֲדָה / עָזִּי וְזִמְרָתִי יְנַהֲלֵנִי:


בַּיֵּן רְצַׄאךּ יַא כַׄאלִקִי / וִאנעִם סְלִיכַּא נַסתְנִיר בִּפַצׄלַךּ

וַאנגִׄי אַלאֲסִיר אַלגַארִקִי / וִאעִדְנַא פִי אַלכַׄיר קַבּל נַהלַךּ

תַעלוּ אַלאַנפֻס תַרתְקִי / וְתַלְחַץׄ אַלאִימַאן תַחת צִׄלַּךּ

תַשׁתַאק אִלַי דַּאר אַלהֻדַא / דִׄי קַד כֻׄלִק פִי אַוַּל אַלזְּמַאנִי

וִאדׄכֻּר לִעַהד אַבַּאיְנַא

וַאתִם לִמַא אַועַדתְנַא

נַרגַׄע אִלַי בַּית קֻדסְנַא

נַסמַע לִנַגם אַלשִּׁיר וַאלמְעַאנִי

וַאכּפִי אַלשְּׁמַאל אַלחַאסִדַּא / וַאעִזְנַא גַׄמעַה מִן אַלהֲוַאנִי:


זָכְרָה לְדָל הַנֶּעֱלָב / מִשֶּׁלְּחִנָּם זַמְּנָה מְזוֹנוֹ

אַתָּה אֲדוֹן הַכֹּל וְאָב / עֵינַי כְמוֹ עֶבֶד לְיַד אֲדוֹנוֹ

רַפֵּא לְלֵב הַנֶּעֱצָב / מֵעוֹל יְלוּד אָמָה וּמֵחֲרוֹנוֹ

הַצֵּל לְנֶפֶשׁ חָרְדָה / מִבַּעֲלֵי עַוְלָה תְמַלְּטֵנִי

חַזֵּק עֲנִיִּים נִדְכְּאִים

מִזִּיז כְּבוֹדָךְ נִהנְאִים

תָּמִיד בְּפִתְחָךְ קוֹרְאִים

קוּמָה בְעֶזְרָתִי וְאַמְּצֵנִי

עַמָּךְ בְּאַהְבָה תִפְקְדָה / וּבְצֵל כְּנָפֶיךָ יְהֶא מְלוֹנִי:


יַא צַאח בַּלִּג נִיְּתִי / שַׁא נַעקִד אַלרַאחַה תְטִיבּ אַלאַרוַאח

מַא בַּין אַחבַּאר סַאדְתִי / נַשׁרַבּ וְנַתהַיַּא בִּשֻׁרבּ אַלאַקדַּאח

סֻבּחַאן חַאפֵץׄ חַיוְתִי / גַאפִר כְׄטַא מִן תַאבּ אִלַיה וּסַמַּאח

אַלמֻלךּ לֻה מֻתסַרמִדַּא / רַבֻּן מֻהַימִל לַא סִוַאה תַׄאנִי

פַצׄלֻה עֲלַינַא דַּאִימַא

לִאלכֻּל בִּמֻלכֻּה חַאכִּמַא

בַּאסִט לִאַרצֻׄה וַאלסְּמַא

לַא תַנצֻׄרֻהּ עַינִי וְהוּ יְרַאנִי

וֻאוַחִד אִסמֻהּ סַאגִּדַּא / וַאשׁרַבּ לִכַּאסִי אַלצַּאפִי אַלזֻמַּאנִי:
