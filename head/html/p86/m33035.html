ספר תחכמוני - שַׁעַר אַרְבָּעִים

הָעֵט וְהַחֶרֶב וּמַחְלְקוֹתָם. מִי יוֹתֵר נִצְרָךְ לְאִישִׁים וּלְמִלְחַמְתָּם


נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי בַלַּיְלָה שׁוֹכֵב עַל מִטָּתִי. וְנָדְדָה מֵעֵינַי שְׁנָתִי. וּבְעוֹדִי מִתְחוֹלֵל בְּצִירִי וְנִגְעִי. וּמִתְגּוֹלֵל עַל יְצוּעִי. שָׁמַעְתִּי קוֹל דּוֹפֶק. וְעַל הַדֶּלֶת מִתְדַּפֵּק. וְאֵינוֹ מִתְאַפֵּק. אָמַרְתִּי: מִי הַקּוֹרֵא בְּאִישׁוֹן לַיְלָה וַאֲפֵלָה. אָמַר לִי: אִישׁ תּוֹעֶה מִן הַמְּסִלָּה וּמִקֹּצֶר יָדוֹ יְסַלֵּד בְּחִילָה. וּכְשָׁמְעִי לְשׁוֹנוֹ מְלֻטָּשׁ כַּתַּעַר. קָרָאתִי אֶל הַנַּעַר. לִפְתֹּחַ הַשַּׁעַר. וְכַאֲשֶׁר בָּא הָאִישׁ בְּמִשְׁעַנְתּוֹ וְכֵלָיו. וּבְגָדִים בְּלוּיִם עָלָיו. הִתְבּוֹנַנְתִּי אֵלָיו. וְהִנֵּה הוּא חֲבֵרֵנוּ הָרַב. וְשָׂמַחְתִּי בוֹ כְּמוֹצֵא שָׁלָל רָב. וְסָר יְגוֹנִי וְעָרַב. וְגָדַל שְשׂוֹנִי וְקָרַב. וְשַׂמְתִּי לְפָנָיו הַנִּמְצָא בְּבֵיתִי לֶאֹכל. וְהֵבֵאתִי לוֹ וַיֹּאכַל מִכָּל. וְכַאֲשֶׁר כִּלָּה אֲכִילָתוֹ. וְעָרַךְ לָאֵל תְּהִלָּתוֹ. הֵחֵל לְסַפֵּר מַחְמַדֵּי מְלִיצָתוֹ. וּפָתַחִּ גִּנְזֵי חָכְמָתוֹ. וָאֶקַּח הַדְּיוֹ וְהַסֵּפֶר. לִכְתֹּב מִפִּיו אִמְרֵי שֶׁפֶר. וְכַאֲשֶׁר הַחִלּוֹתִי לִכְתֹּב נִשְׁבַּר הָעֵט בְּיָדִי וְהֶחְלַפְתִּיהוּ. וְנִשְׁבַּר פַּעַם שֵׁנִית עַד הִשְלַכְתִּיהוּ. אָמַר לִי חֶבֶר הַקֵּינִי: לָמָּה תַשְלִיכֵהוּ וְהָאֵל בָּחַר בּוֹ. אַל תַּשְׁחִיתֵהוּ כְּי בְרָכָה בּוֹ. אִלּוּ יָדַעְתָּ מַעֲלָתוֹ. לֹא הִשְׁלַכְתָּ אוֹתוֹ. הֲלֹא יָדַעְתָּ נֹעַם חִידָתוֹ. וְאִמְרֵי תְעוּדָתוֹ. וְאִם תִּרְצֶה לְךָ אֲגִידֶנָּה. וּמִמְּךָ לֹא אַכְחִידֶנָּה. אָמַרְתִּי: הִנֵּה אָזְנִי לְדֶלֶת נִיבְךָ רְצוּעָה. וְעֵינִי בְּפוּךְ מֹרְךָ קְרוּעָה. אָמַר: הִתְוַכְּחוּ בִימֵי קֶדֶם סופֵרי הַמֶּלֶךְ הַגְּדוֹלִים. עִם שָׂרֵי הַחֲיָלִים. וַּיֲענּו סוֹפְרֵי הַמֶּלֶךְ וַיֹּאמְרוּ: אֲנַחְנוּ גִבּוֹרֵי הַמְּלִיצוֹת. וְלָנוּ דֵעוֹת וְעֵצוֹת. יָפִיצוּ מִפִּינוּ אִמְרֵי תְעוּדוֹת. שַׂמְנוּם לַמְּלוּכָה יְסוֹדוֹת. וּלְקַרְשֵׁי הַגְּדֻלָּה יָדוֹת. וּבְיָדֵינוּ הָעֵט הַנֶּחְמָד. אֲשֵׁר אֵין לְפָנָיו מַעֲמָד. יַפִּיל רְפָאִים. וְיַחְכִּים פְּתָאיִם. וְהוּא קְצֶר קוֹמָה וְנִבְזֶה. וְחַלָּשׁ וְרָזֶה. אֲבָל הוּא רָם וְרֹאשׁוֹ בַשָּמַיִם. וּבוֹ מַחְמַד עֵינַיִם. יָשִׁיב לְאָחוֹר כָּל שׁוֹלְפֵי זַיִן. וְהוּא הַנּוֹתֵן רוֹזְנִים לָאָיִן.

וַיִּשְׂאוּ מְשָׁלָם וַיֹּאמְרוּ:

אֲנַחְנוּ לַיְּקָר עַמּוּד. וְתָמִיד

בְּעֵטֵנוּ נְפָאֵר הַמְּלוּכָה.

וְלָנוּ הַגְּדֻלָּה הַגְּדוֹלָה

אֲשֶׁר עַל כּוֹכְבֵי שַׁחַק נְסוּכָה.

וּבַעְלֵי הַקְּרָב לָנוּ עֲבָדִים

וְתוֹךְ לִבָּם חֲנִיתֵנוּ מְעוּכָה.

וַיַּעֲנוּ חֲלוּצֵי הַצָּבָא וַיֹּאמְרוּ: אֲנַחְנוּ כְפִירֵי קְרָבוֹת. וְאַמִּיצֵי לְבָבוֹת. נַחְצֹב מִצּוּר הַחֲרָבוֹת לֶהָבוֹת. וְנַעֲזֹב הָאֲרָצוֹת מִפַּחְדֵּנוּ חֲרֵבוֹת. עַד יִבְרְחוּ שׁוֹכְנֵיהֶם בִּנְפָשׁוֹת עֲצוּבוֹת. וְלֹא הִפְנוּ בָנִים אֶל אָבוֹת. וְלָנוּ הַחֶרֶב אֲשֶׁר תַּבִּיעַ בְּלִי לָשׁוֹן. וְתַשְׁגִּיחַ בְּלִי אִישׁוֹן. וְתִגְרֹף כְּנַחַל קִישׁוֹן וּפִישׁוֹן. וּבְהִקָּרֵא אֲצִילֵי הַמְּלוּכָה אֶל עָל הִיא תַעֲבֹר רִאשׁוֹן. כִּי הִיא כֶתֶר לַמְּלָכִים. וַעֶטֶרת לַנְּסִיכִים. וְהִיא שׁוֹמֶרֶת מִכַּל נֵזֶק בְּעָלֶיהָ. וּכְחוֹל הַיָּם חֲלָלֶיהָ.

וַיִּשְׂאוּ מְשָׁלָם וַיֹּאמְרוּ:

אֲנַחְנוּ יַד יְקָר נָנִיף תְּנוּפָה

וְחַרְבֵּנוּ לְאוֹיְבֵנוּ שְׁלוּפָה.

וְעֵת יִתְחַבְּאוּ בֶעְלֵי גְבוּרָה

זְרוֹעֵנוּ לְהִלָּחֵם חֲשׂוּפָה.

וְחַרְבֵּנוּ כְּמוֹ גֶפֶן וְאוּלָם

בְּמֵימֵי הַחֲלָלִים הִיא עֲנָפָהּ.

תְּרוֹצֵץ עַל פְּנֵי אֶרֶץ כְּבָרָק

וְאֶל רֹאשׁ אוֹיִבְים חָשָׁה וְעָפָה.

וַיִּוָּעֲדוּ הַחֶרֶב וְהָעֵט שְׁנֵיהֶם. לַעֲרֹךְ דִּין בֵּינֵיהֶם. וַתַּעַן הַחֶרֶב וַתֹּאמַר: אֲנִי נוֹתֶנֶת כֹּחַ לַגִּבּוֹרִים. וּבְיָדִי מִחְיַת הַנְּשָׁרִים וְהַכְּפִירִים. וְכָל הַיָּמִים אֲשֶׁר הַנְּשָׁרִים אוֹתִי יִמְצָאוּ. לֹא יִרְעֲבוּ וְלֹא יִצְמָאוּ. כִּי אֲנִי אַטְרִיפֵם בְּשַׂר גִּבּוֹרִים. וְאַשְׁכִּירֵם בִּדְמֵי אַבִּירִים. וְאֵיךְ יַעֲרֹךְ הָעֵט אֵלַי. וְהוּא שְׂרֵפָה לְגֶחָלַי. וּמִדְרָךְ לְרַגְלַי. וְאֵיךְ יִתְגָּאֶה קָנֶה רָצוּץ. כְּסִרְפָּד וְנַעֲצוּץ. אִם תִּגַּע בּוֹ יָד תִּשְׁבְּרֶנּוּ. וְרוּחַ עָבְרָה בוֹ וְאֵינֶנּוּ.

וַיַּעַן הָעֵט וַיֹּאמַר: אֲנִי נוֹתַרְתִּי נָבִיא. בְּאֹהֶל הַתְּעוּדָה מוֹשָׁבִי. כִּי אֲנִי אִישׁ תָּם וְיָשָׁר. וְכָל תּוֹמְכַי מְאֻשָּׁר. אֲבָאֵר רָזֵי הַחָכְמוֹת. וַאֲגַלֶּה סוֹדוֹת הַמְּזִימּוֹת. וְאַבִּיעָה חִידוֹת. וְאַגִּיד נִכְבָּדוֹת. וּבִי מְלָכִים יִמְלֹכוּ. וְסוֹפְרִים בְּשֵׁבֶט סוֹפֵר יִמְשֹׁכוּ. וּבִי רוֹזְנִים יְחוֹקְקוּ צֶדֶק. וִיחַזְּקוּ כָל בֶּדֶק. וְלוּלֵי עֵט הַסּוֹפֵר אָבְדוּ כָל הַחָכְמוֹת. וְגַם נֶעְדְּרוּ הַחִידוֹת וְכָל הַמְּזִמּוֹת.

וַתַּעַן הַחֶרֶב וַתֹּאמַר: אֲנִי מַחֲצֶבֶת רַהַב. חוֹצֶבֶת לַהַב. וּמִי מִכָּל אֲצִילֵי עוֹלָם. וְרוֹזְנֵי תֵבֵל כֻּלָּם. אֳשֶׁר יִרְאֶה צִלִּי. וְלֹא יִפְחַד מִקְּרָב-לִי. וַהֳלֹא בְצֵאתִי מִתַּעֲרִי לֵב הָאַרִי יִמַּס. וְיָשׁוּבוּ הַגִּבּוֹרִים לְמִרְמָס. וְהַנְּפִילִים. יִפְּלוּ לְפָנַי חֲלָלִים. וְגִבּוֹרִים. יִתְחַבְּאוּ בַחוֹרִים. וּמִי הוּא הָעֵט כִּי יַעֲמֹד לְפָנַי. אוֹ יִתְיַצֵּב לְנֶגֶד עֵינַי. וְיַעֲלֶה שְׁאוֹנוֹ בְאָזְנַי. וְהוּא קָנֶה נִשְׁבָּר. וְתֶבֶן בְּלִי בָר. לֹא לִזְרוֹת וְלֹא לְהָבַר. אִם אִישׁ יִמְצָאֵהוּ. בְּכַף רַגְלוֹ יִדְרְכֵהוּ. וְעִם הַדֹּמֶן יַשְׁלִיכֵהוּ. וּמְקוֹמוֹ נוֹדַע בְּתוֹךְ הָאַשְׁפָּה. כִּי כֻלּוֹ בוּז וְחֶרְפָּה. וּמְקוֹמִי בְחֵיק מְלָכִים. וְעַל צַוְּארֵי נְסִיכִים. וּמֵאֵימָתִי הַגִּבּוֹרִים יִירָאּו. וְכָל רוֹאַי מִפָּנַי יִתְחַבָּאוּ.

וַיַּעַן הָעֵט וַיֹּאמַר: כֵּן דִּבַּרְתָּ. וְצֶדֶק אָמַרְתָּ. כִּי בְצֵאתְךָ מִתַּעַרְךָ כָּל רוֹאֶיךָ יִפְחֲדוּ מִמְּךָ. וְאְישׁ לֹא יַעֳמֹד עִמְּךָ. כִּי שׁוֹפֵךְ דָּם אַתָּה. וְאַכְזָרִי וְעָרִיץ נוֹדַעְתָּ. כַּמָּה דָמִים שָפַכְתָּ. וְכַמָּה נְקִיִּים הָרַגְתָּ. וּמִיּוֹם הֱיוֹתְךָ לֹא תֶחְדַּל לְהַחֲרִיב הָאַרָצוֹת. וּלְמַלֵּא בַחֲלָלִים חוּצוֹת. וּלְהַפְרִיד בָּנִים מֵאֲבוֹתָם. וְהַיְלָדִים מֵחֵיק אִמּוֹתָם. וְאִם תִּתְפָּאֵר עָלַי בְּכֹחֲךָ אֵין גְּבוּרָתִי בְכֹחִי. כִּי אִם בְּרוּחִי. וְאֵיךְ תַּעֲרֹךְ אֵלַי וַאֲנִי אִישׁ תָּם יוֹשֵׁב אֹהָלִים. וְאַתָּה אִישׁ שָׂדֶה וְרַע מַעֲלָלִים. וּמַרְבֶּה חֲלָלִים. וּמְקוֹמְךָ בֶהָרִים הַתְּלוּלִים. וְעַל פְּנֵי צוּרֵי הַיְּעֵלִים. וּבְאֶשֶד הַנְּחָלִים. וְהַיְּעָרִים הַגְּדוֹלִים. כָּל רוֹאֶיךָ מִמְּךָ יִבְרָחוּ. וְכָל רוֹאַי בִּי יִשְׂמָחוּ. וּבְחֶבְרָתִי יִבְטָחוּ. וְאַתָּה חָבֵר לְכָל חָנֵף וּמֵרַע. וְחָסֵר וְנִגְרָע. וְלֹא יִדְבַּק בְּךָ כִּי אִם גַּנָּב וְרָשָׁע. וְכָל אֲשֶׁר מִבֶּטֶן פָּשָׁע. וַאֲנִי כָל רָע לֹא יְגוּרֵנִי. וְרָשָׁע לֹא יִתְחַבֵּר אֵלַי וְלֹא יְשׁוּרֵנִי. הוֹלֵךְ בְּדֶרֶךְ תָּמִים הוּא יְשָׁרְתֵנִי. וַחֲסִידִים הֵם אֲשֶׁר בְּיָדָם יַעֲלוּנִי. וְנִכְבָּדִים יְכַבְּדוּנִי. וְלִי יְגַלּוּ הַמְּלָכִים סוֹדוֹתָם. וּבִי תִשְׁלַם עֲצָתָם. וּבְעֵת אֶתְיַחֵד עִם הַמֶּלֶךְ בְּהֵיכָלוֹ. אַתָּה לֹא תִקְרַב לוֹ.

וַתַּעַן הַחֶרֶב וַתֹּאמַר: אֵין לְהָשִׁיב מַעֲנֶה עַל הֶגְיוֹנֶיךָ. וְשִׁקְרֵי לְשׁוֹנֶךָ. כִּי שְׁאַל נָא לְיָמִים רִאשׁוֹנִים אֲשֶׁר הָיוּ לְפָנֶיךָ. הֲלֹא הֵם יוֹרוּךָ. יֹאמְרוּ לְךָ. כִּי בִי יַכְנִיעַ הַמֶּלֶךְ קָמָיו וּמוֹרְדָיו. וּבִי יִכְבּשׁ אוֹיְבָיו וּבוֹגְדָיו. וּבִי יִלָּכְדוּ הֶעָרִים הַבְּצוּרוֹת. וְיִפָּתְחוּ הֶמִּגְדָּלוֹת הַסְּגוּרוֹת. וּבִי יִשְׁמֹר הַמֶּלֶךְ מַלְכוּתוֹ. וְלוּלֵי פַחְדִּי לֹא עָמְדָה רֶגַע גְּדֻלָּתוֹ. אֲבָל אֲנִי אֶשְמְרֵהוּ מִמּוֹנָיו. וְאֶת אֵימָתִי אֶשְׁלַח לְפָנָיו. וְהַמּוֹתִי צָרָיו וְכָל גְּדוּדֵיהֶם. וְאֶת כָּל הָעָם אֲשֶׁר יָבוֹא בָהֶם. וּבִרְאוֹתָם צוּר חַרְבְּי בְיָדִי נִצָּב. מִי זֶה לְפָנַי יִתְיַצָּב. וְכִשְמֹעַ הָעֵט דִּבְרֵי הַחֶרֶב וּמִלֶּיהָ. הִתְגָּאָה עָלֶיהָ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמֶר אֵלֶיהָ:

אֲנִי מַחְרִישׁ אֲבָל אַחְרִיד בְּמִלַי

מְתֵי גַאְוָה בְּעֵת אַצְבִּיא חֲיָלַי.

וּמַאְמָרֵי פְּאֵר עַל רֹאשׁ מְלָכִים

וְשִׂמְחַת הַלְּבָבוֹת טוּב מְשָׁלַי.

הֲכִי בִי תַעֲמֹד אֶרֶץ בְּמִשְׁפָּט

וְאֵין עָוֶל בְּמַעְשַׂי אוֹ פְעָלַי.

וּבִי צוּר חַק עֲשֶׂרֶת הַדְּבָרִים

וְהִנְחִילָם בְּחוֹרֵב אֶל קְהָלִי.

וְהַחֶרֶב בְּעֵת תָּרוּם וְתִגְאֶה

עֲלֵי רֹאשָׁהּ אֲנִי אָרִים דְּגָלַי.

וְיוֹם בָּאָה לְמִלְחַמְתִּי בְּגַאְוָה

אֲנִי קַמְתִּי וְהִיא כָרְעָה לְרַגְלָי.

אָמַר הַמַּגִּיד: וּכְשָׁמְעִי חִידָתוֹ. וּמִשְׁלֵי מְלִיצָתוֹ. כָּתַבְתִּי עַל לִבִּי דְבָרָיו. וְחָקַקְתִּי בְּעֵט בַּרְזֶל אֲמָרָיו. וְיָשַׁבְתִּי עִמּוֹ יָמִים. לְבַלּוֹת יָמַי בַּטּוֹב וּשְׁנוֹתַי בַּנְּעִימִים. עַד הוֹרַנִי הַזְּמַן בְּחִצֵּי פְרִידָתוֹ. וּגְמָלַנִי מִשְּׂדֵי חֶבְרָתוֹ.
