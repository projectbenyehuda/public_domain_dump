ספר תחכמוני -שַׁעַר שְׁנֵים עָשָׂר

מֵהַכִּילוּת וְהַנְּדִיבוּת וּמְרִיבוֹתֵיהֶם. מַחְלְקוֹתָם וּמִלְחֲמוֹתֵיהֶם.


נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי בְּמִבְחַר יְדִידִים. רַבִּים וְנִכְבָּדִים. בַּעֲלֵי מַעֲשִׂים נֶחְמָדִים. עַל טוּב שָׁרְשָׁם מְעִידִים. וְהָיוּ מְדַבְּרִים עַל מִדַּת הַכִּילוּת וְאוֹמְרִים כִּי אֵין כָּמוֹהָ מִדָּה רָעָה. וּמְאוּסָה וּגְרוּעָה. כְּאֶרֶץ לֹא זְרוּעָה. וְהָיָה בְּתוֹכָם רֹאשׁ הַמְדַבְּרִים חֶבֶר הַקֵּינִי. הַגּוֹזֵר בְּנִיבוֹ וְאֵין מֵבִיא. וַיִּשְׁאָלוּהוּ הָעָם. לְהוֹדִיעָם. סוֹד הַכִּילוּת בְּאֵיזֶה מָקוֹם נִבְרָא. וּמֵאַיִן נִקְרָא. וַיֹּאמֶר לָהֶם: דְּעוּ כִּי הַכִּילוּת בִּימֵי קֶדֶם. הָיָה שׁוֹכֵן בְּאֶרֶץ בְּנֵי קֶדֶם. וַתְּהִי רֵאשִׁית מַמְלַכְתּוֹ בָּבֶל. וּמִשָּׁם פָּשְׁטָה עַל יוֹשְׁבֵי תֵבֵל. וּתְחִלָּתוֹ הָיְתָה כִכְלִי אֵין חֵפֶץ בּוֹ. וּכְאִישׁ צָרוּעַ אֲשֶׁר לֹא יִגַּע בּוֹ. וִּמחוּץ לַמַּחֲנֶה מוֹשָׁבוֹ. עֵת בֹּא דְבָרוֹ. וְרָצָה הַבּוֹרֵא לְהַגְדִּיל כְּבוֹדוֹ וַהֲדָרוֹ.

וַיֹּאמֶר הָאֵל לַנְּדִיבוּת: מַהֵר קַח אֶת לְבוּשׁ הוֹדִי. וּמִרְכֶּבֶת כְּבוֹדִי. וְהִלְבַּשְׁתָּ אֶת הַכִּילוּת. וְתַמְלִיכֵהוּ עַל בְּנֵי הַגָּלוּת. וַיִּקַח הַנְּדִיבוּת אֶת הַלְּבוּשׁ וְאֶת הַסּוּס כְּדִבְרֵי יוֹצְרוֹ. וַיַּלְבֵּשׁ אֶת הַכִּילוּת וַיָּשֶׂם רְבִיד הַזָּהָב עַל צַוָּארוֹ. וַיַּרְכִּיבֵהוּ בְּמִרְכֶּבֶת הֲדָרוֹ. וַיִּקְרָא לְפָנָיו: כָּכָה יֵעָשֶׂה לָאִישׁ אֲשֶׁר הַמֶּלֶךְ חָפֵץ בִּיקָרוֹ. וַיָּשָׁב הַכִּילוּת אֶל שַׁעַר הַמֶּלֶךְ נְשׂוּא רֹאשׁ. וְהַנְּדִיבוּת נִדְחַף אֶלֵּ בֵּיתוֹ אָבֵל וַחֲפוּי רֹאשׁ. וַיִּשְׁאֲלוּ כָל הָעָם. לְחֶבֶר הַקֵּינִי לְהוֹדִיעָם. אֵיךְ הָיְתָה מַלְכוּת הַכִּילוּת מִתְּחִלָּתוֹ. וְאֵיךְ עָלְתָה מַעֲלָתוֹ. וּמָתַי מָשְׁחוּ הַיְּהוִּדים לְמֶלֶךְ אוֹתוֹ.

אָמַר לָהֶם: דְּעוּ כִּי בִימֵי קֶדֶם. הָיָה הַכִּילוּת וְהַנַּבְלוּת שְׁנֵיהֶם שׁוֹכְנִים בְּאֶרֶץ קֶדֶם. בְּבַיִת אֶחָד מוֹשָׁבָם. וְעַל עֶרֶשׂ אֶחָד מִשְׁכָּבָם. וּבִתְחִלַת צֵאתָם מִמִּגְרָשָׁם. וַיִּמְצְאוּ בִקְעָה בְּאֶרֶץ שִׁנְעָר וַיֵּשְׁבוּ שָׁם. כִּי שָׁם יְצָרָם שׁוֹכֵן עָרֶץ. וּמִשָּׁם הֱפִיצָם ה' עַל פֵני כָל הָאָרֶץ. וַיְהִי הַיּוֹם וַיֹּאמֶר הַכִּילוּת לְרֵעָיו. וּלְחֶבְרַת מְיֻדָּעָיו: רָאִיתִי הַלַּיְלָה בַּחֲלוֹמִי וְהִנֵּה אֲנִי עוֹמֵד עַל רֹאשׁ הָהָר הַמָּעוֹז. וְעָלָיו מִגַּדל עֹז. וְהַמִּגְדָּל הָיָה עוֹמֵד עַל עַמּוּדִים. כַאֲרָזִים עוֹמְדִים. וָאֶשְׁמַע אַחֲרַי קוֹל. רַעַשׁ גָּדוֹל. כְּמוֹ עֵגֶל הִרְקִידָם וְהִרְגִּיעָם. וְלָאָרֶץ הִפִּילָם לְפָנַי וְהִגִּיעָם. אָמְרוּ לוֹ חֲבֵרָיו: יָדוֹעַ תֵּדַע כִּי הַמִגְדָּל הַמָּעוֹז. הִיא תּוֹרַת ה' אֲשֶׁר הִיא מִגְדַּל עֹז. וְהָעַמּוּדִים הָעוֹמְדִים. הֵם גְּדוֹלֵי הַיְּהוּדִים. אֲשֶׁר מִמִּצְרַיִם וְעַד אֶרֶץ כַּשְׂדִּים. וַאֲשֶׁר רָאִיתָ כִּי נָפְלוּ לְעֻמָּתְךָ. כְּאִלּוּ הֵם כּוֹרְעִים וּפְנֵיהֶם מְגַמָּתְךָ. יָדוֹעַ תֵּדַע כִּי מָלוֹךְ תִּמְלֹךְ עַל עַם הָאֵל. וְקָמָה בְּיָדְךָ מַמְלֶכֶת יִשְׂרָאֵל. וְכָל גְּדוֹלֵיהֶם יָבִיאוּ צַוָּארָם בְּעֻלָּךְ. אַפַּיִם אֶרֶץ יִשְׁתַּחֲווּ לָךְ. וְכָל צִבְאוֹתָם תְּהִי כּוֹבֵשׁ וְנוֹצֵחַ.

תִּפְתַּח וְאֵין סוֹגֵר תִּסְגֹר וְאֵין פּוֹתֵחַ. וּבִקְצָת הַיָּמִים. עָבַר בַּשּׁוּק עַל אִישׁ מֵהַחֲכָמִים. וּבְיָדוֹ סֵפֶר הָעֲתִידוֹת וְכָתוּב בּוֹ: וְהָיָה בְּאַחֲרִית הַיָּמִים. חוֹמַת בָּבֶל הָרְחָבָה עַרְעַר תִתְעַרְעָר. וְיִצְמַח בְּאֶרֶץ שִׁנְעָר. בְּרִיָּה חֲדָשָׁה מְשֻׁנָּה. וְצוּרָה רָעָה וּתְמוּנָה. מֵאֶרֶץ עֳדִינָה יְצִיאָתוֹ. וְעֶל כָּל קְהִלּוֹת מִזְרָח תִּהְיֶה מַלְכוּתֹו. וּשְׁמוֹ הַכִּילוּת. וְשֵׁם יוֹלְדָיו הַנַּבְלוּת וְהַזֻּלּוּת. וַיִּתְמַהּ הַכִּילוּת עַל הַדָּבָר הַזֶּה וְהַזְּמַן עָבַר. וְהוּא שָׁמַר אֶת הַדָּבָר.

וַיְהִי מִקֵּץ יָמִים רַבִּים נוֹדַע הַחֲלוֹם אֲשֶׁר חָלַם. לִקְהִלּוֹת הַקֹּדֶשׁ כֻּלָּם. אֲשֶׁר מִנֹּא אָמוֹן וְעַד עֵילָם. וַיִּשְׂמְחוּ שִׂמְחָה גְדוֹלָה כִּי אָמְרוּ עַתָּה יִהְיֶה לָנוּ מֶלֶךְ אֲשֶׁר יָאִיר חָשְׁכֵּנוּ. וִייַשֵּׁר דַּרְכֵּנוּ. וְיָקִים סֻכָּתֵנוּ. וּשְׁפָטָנוּ מַלְכֵּנוּ.

וַיְהִי אַחֲרֵי יָמִים רַבִּים יָצָא הַכִּילוּת וַאֲחֻזַּת מֵרֵעָיו אֶל אַחַת הֶעָרִים. לְהִתְהַלֵּךְ בַּכְּפָרִים. הֵם יָצְאוּ אֶת הָעִיר לֹא הִרְחִיקוּ. וְאַרְבָּעָה פָּרָשִׁים מִן הַנְדִיבוּת אוֹתָם הִדְבִּיקוּ. וְכִרְאוֹת הַנְּדִיבִים אוֹתוֹ בְעֵינֵיהֶם. וּבְטֶרֶם יִקְרַב אֲלֵיהֶם. אָמְרוּ אִישׁ אֶל אָחִיו: הִנֵּה בַּעַל הַחֲלוֹמוֹת הַלָּזֶה בָא. אֲשֶׁר עָלֵינוּ בְּמַלְכוּתוֹ הִתְנַבָּא. וְעַתָּה לְכוּ וְנַהַרְגֵהוּ. וּבְאַחַד הַבּוֹרוֹת נַשְׁלִיכֵהוּ. וְנָסִיר מֵעַל הָאֲדָמָה חֶרְפַּת כְּלִמּוֹתָיו. וְנִרְאֶה מַה יִהְיוּ חֲלוֹמוֹתָיו. אָמַר לָהֶם גְּבִירָם וּבַעַל סוֹדָם. אַל תִּשְׁפְּכוּ דָם. רַק תִּקְשְׁרוּהוּ בַעֲבוֹתוֹת חֲזָקִים. יִהְיוּ לִזְרוֹעוֹתָיו צְמִידִים וּלְצַוָּארוֹ עֲנָקִים. וְשִׂימוּהוּ בְּאַחַת הַמְּצָדוֹת בְּלֶחֶם צַר וּמַיִם לָחַץ. עֲדֵי יָמוּת בְּלִי פֶּגַע וּבְלִי מָחַץ. וְכָל יָמָיו יִהְיֶה בְּמַאֲסָר בְּלִי גוֹאֵל. לְמַעַן לֹא יִשָּׁמַע עוֹד קוֹלוֹ בְּהָרֵי יִשְׂרָאֵל. וַיְמַהֲרוּ כֻלָּם וַיִּקָּחוּהוּ. וּמֵעַל מֶרְכַּבְתּוֹ הוֹרִידוּהוּ. וְהִכּוּהוּ פְּצָעוּהוּ. וּבִנְחֻשְׁתַּיִם אֲסָרוּהוּ. וּבְמִגְדָּל וְרֹאשׁוֹ בַשָּׁמַיִם עֲצָרוּהוּ. וְהוֹשִׁיבוּהוּ בְחֶדֶר קָטָן בְּמַחֲשַׁכִּים נֶחְבָּא. אֵין יוֹצֵא וְאֵין בָּא. רַק הָיָה נִכְנָס עָלָיו מְעַט אוֹר מִן הָאֲרֻבָּה. וְהִקִּיפוּ עָלָיו לְשָׁמְרוֹ מִן הַנְּדִיבִים הַנִּבְחָרִים. חֲמֵשׁ מֵאוֹת גִּבּוֹרִים. בַּעֲלֵי זְרוֹעַ רָמָה.

כֻּלָּם אֲחוּזֵי חֶרֶב מְלֻמְּדֵי מִלְחָמָה. וְהַקּוֹל נִשְׁמַע לַיְּהוּדִים אֲשֶׁר בַּעֲדִינָה וּבִסְפַרְוָיִם. וּבְאַשּׁוּר וּבַאֲרַם נַהֲרָיִם. וּבְאֶרֶץ הַצְּבִי וּמִצְרָיִם. כִּי הַכִּילוּת אֲדוֹנֵיהֶם הָיָה לָמַס. וּבִידֵי אוֹיְבָיו נִתְפַּס וְנִרְמַס. וַיִּתְעַצְּבוּ הָאֲנָשִׁים וַיִּחַר לָהֶם מְאֹד. כי הָלַךְ מַלְכָּם בַּגּוֹלָה. וַיֹּאמְרוּ: זוֹ רָעָה חוֹלָה. הִנֵּה חָשַׁבְנוּ כִּי יִהְיֶה לָנוּ לְמֶלֶךְ לִמְשֹׁל עַל קְהִלָּתֵנוּ. וְלִלְחֹם אֶת כָּל מִלְחֲמוֹתֵינוּ. וְלֹא יִהְיֶה עַם ה' תּוֹעֶה. כַּצֹּאן אֲשֶׁר אֵין לָהֶם רוֹעֶה. וְעַתָּה פָרִיצֵי בְנֵי עַמֵּנוּ כְּבָשׁוּהוּ בְּאַכְזְרִיּוּתָם. רוּחַ אַפֵּינוּ מְשִׁיחַ ה' נִלְכַּד בִּשְׁחִיתוֹתָם. אֲשֶׁר חֲשַׁבְנוּהוּ כְּעֵץ חַיִּים. אֲשֶׂר אָמַרְנוּ בְּצִלּוֹ נִחְיֶה בַגּוֹיִם. אֲבָל אֵין לְאֵל יָדֵנוּ רַק נִשָּׂא לְבָבֵנוּ אֶל כַּפָּיִם. עַד יַשְׁקִיף וְיֵרֶא ה' מִשָּׁמָיִם. וַיְהִי כִּי אָרְכוּ לוֹ שָׁם הַיָּמִים. וְהוּא בְּמַאֲסָרוֹ יוֹשֵׁב מַשְׁמִים. וְעָלָיו יַהֲלֹךְ אֵימִים. עָלָה עַל לִבּוֹ כִּי הָיְתָה לוֹ בְּבֵיתוֹ יוֹנָה חֲשׁוּבָה. יְפַת קוֹל וַעֲרֵבָה. מְצַפְצֶפֶת בְּכָל עֵת לְפָנָיו. וְהָיְתָה חַיֵּי נַפְשׁוֹ וְאוֹר עֵינָיו. וְכַאֲשֶׁר זָכַר אוֹתָהּ בָּכָה בְכִי גָדוֹל עַל פְּרִידָתָהּ. וְנִרְדַּם בִּמְרִירוּת נַפְשׁוֹ. וַיַּחֲלֹם וְהִנֵּה אִישׁ עוֹמֵד לְרֹאשׁוֹ. וְהוּא אוֹמֵר: אִם תִּשְׁכְּבוּן בֵּין שְׁפַתָּיִם. כַּנְפֵי יוֹנָה נֶחְפָּה בַכֶּסֶף וְאֶבְרוֹתֶיהָ בִירַקְרַק חָרוּץ. בְּפָרַשׂ שַׁדַּי מְלָכִים בָּהּ תַּשְׁלֵג בְּצַלְמוֹן. וְהֵקִיץ מֵחֲלוֹמוֹ בְּנֶפֶשׁ כּוֹאֲבָה. וְעַיִן דּוֹאֲבָה. וַיַּרְא וְהִנֵּה הַיּוֹנָה עַל הָאֲרֻבָּה. הוֹגָה וּמְצַפְצֶפֶת. וּבְרַגְלֶיהָ מְטוֹפֶפֶת. כְּאִלּוּ הִיא מְסַפֶּרֶת יְגוֹנֶיהָ. לַאדוֹנֶיהָ.

וְזוֹכֶרֶת עִצְּבוֹנֶיהָ. וְאֵיךְ יְצָאוּהָ בָנֶיהָ. וְכִרְאוֹתוֹ אוֹתָהּ שָׁלַח יָדוֹ וַיִּקָּחֶהָ. וּבָכָה בְכִי גָדוֹל עָלֶיהָ. וַיְנַשְׁקֶנָּה וַיְחַבְּקֶהָ.

וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר:

יוֹנַת רְחוֹקִים אֵיךְ זְכַרְתִּנִי

וּבְאַהֲבָתֵךְ אֵיך פְּקַדְתִּנִי.

אַשְׁקֵךְ בְּדִמְעִי כֹּל אֲשֶׁר צָמִית

אוֹ שַׁאֲבִי דִמְעֵךְ וְהַשְׁקִינִי.

צוּר יִגְמְלֵךְ טוֹב עַל אֲשֶׁר עָשִׂית

כִּי עִם יְדִידַי לֹא שְכַחְתִּנִי.

כָּל־אֹהֲבַי שָׁכְחוּ יְדִידוֹתַי

בִּלְתֵּךְ וְאֵין מוֹצִיא לָאוֹר דִּינִי.

וַיֹּאמֶר לָהּ: בִּתִּי. מִי הִגִּיד לָךְ מְקוֹם שִׁבְתִּי. כִּי מִהַרְתְּ וַתָּבֹאִי לִקְרָאתִי. לְדַבֵּר עַל לִבִּי וּלְנַחֵם אוֹתִי. יְשַׁלֵּם ה' פָּעֳלֵךְ עַל אַהֲבָתֵךְ הַתְּמִימָה. וּתְהִי מַשְׂכֻּרְתֵּךְ שְׁלֵמָה. כִּי לֹא בָגַדְתְּ בִּי עִם הַיְדִידִים הַבּוֹגְדִים. וְלֹא מָרַדְתְּ עִם הַמּוֹרְדִים. וְעַתָּה הַטִּי נָא חַסְדֵּךְ אֵלַי. וּפִרְשִׂי כַנְפֵי אַהֲבָתֵךְ עָלַי. וּשְׂאִי שְׁלוֹמִי. לְכָל אַנְשֵׁי מְקוֹמִי. וְתַגִּידִי לָהֶם בְּטוּב עִנְיָנֵךְ. וּבְמֶתֶק לְשׁוֹנֵךְ. צָרָתִי אֲשֶׁר רָאַתָה עֵינֵךְ. וַיִּקַּח מְעַט פֶּחָם וַיְעָרְבֵהוּ בְּבִכְיוֹ. וַיַּעַשׂ מִמֶּנּוּ דְּיוֹ. וּמָצָא חֲתִיכַת אִגֶּרֶת. כְּפִי רְבִיעַ זֶרֶת. וְכָתַב בּוֹ אַרְבָּעָה טוּרִים. וְשָׂמָם תַּחַת אֶבְרוֹתֶיהָ קְשׁוּרִים. הַשֶּׁמֶשׁ לַעֲרֹב פָּנָה. וַיְשַׁלַּח אֶת הַיּוֹנָה. וַתָּבוֹא הַיּוֹנָה לְעֵת עֶרֶב אֶל קִנַּהּ. וְלִנְוֵה מְעוֹנָהּ. אֲשֶׁר בְּאֶרֶץ עֲדִינָה. וַתַּעֲמֹד נֹכַח פֶּתַח הַבַּיִת. לְעֵינֵי אֶנְשֵׁי הַבַּיִת. וְהִיא מְצַפְצֶפֶת. וּסְבִיבָם מְעוֹפֶפֶת. וְהוֹלֶכֶת וְטוֹפֶפֶת. וַיִּתְבּוֹנְנוּ אֵלֶיהָ וְהִנֵּה סֵפֶר קָשׁוּר בְּאֶבְרָתָהּ. וְתָלוּי בְּנוֹצָתָהּ. וַיְמַהֲרוּ וַיִּקָּחוּהוּ. וְכַאֲשֶׁר פְּתָחוּהוּ. כִּמְעַט בְּדִמְעָתָם מָחוּהוּ. וּמֵעִיר לְעִיר שְׁלָחוּהוּ. וְלָקְחוּ הַיּוֹנָה וַיְחַבְּקוּהָ וַיְנַשְׁקוּהָ. וּבְשַׁרְשֶׁרֶת זָהָב כִּסּוּהָ וְהֶעֱנִיקוּהָ. וְנִתְקַיֵּם הַחֲלוֹם כַּנְפֵי יוֹנָה נֶחְפָּה בַּכֶּסֶף.

וְזֶה פַּתְשֶׁגֶן הַכְּתָב אֲשֶׁר שָׁלַח הַכִּילוּת מִמַּאֲסַר שְׁבוּתוֹ. לְאַנְשֵׁי בְּרִיתוֹ: שָׁלוֹם מִשְּׁמֵי מְעוֹנָה. עַל אַנְשֵׁי הָאֱמוּנָה. וְאִם הֵם מְעַט מִזְעָר יָדוֹעַ תֵּדְעוּ אַחַי וְרֵעַי. וְחֶבְרַת מְיֻדָּעַי. כִּי נִמְכַּרְתִּי לָמוּת בְּלִי חֹפֶשׁ. וְנָגַע חֶרֶב עַד הַנֶּפֶשׁ. וְכִסָּה עַל עֵינַי חשֶׁךְ וְצַלְמָוֶת. כִּי כְפֶשַׂע בֵּינִי וּבֵין הַמָּוֶת. וּפַסּוּ אֱמוּנִים מִבְּנֵי אָדָם. וְשָׁכְחוּ הַיְדִידִים יְדִידָם. וְאָפְסוּ הַחוֹצְבִים לִקְרָבוֹת לֶהָבוֹת. וְשָׁבוּ הַזְּכָרִים נְקֵבוֹת. הַשְׁכַחְתֶּם יְדִידוּתִי. וְאִם בְּגַדְתֶּם בְּאַהֲבָתִי. הֲלֹא מִיּוֹם נוֹלַדְתֶּם הָיִיתִי מְזֻמָּן לַעֳבוֹדַתְכֶם. וּכְבָנִים טִפַּחְתִּי וְרִבִּיתִי אֶתְכֶם. אוֹי עַל יְדִידִים לֹא זָכְרוּ טוּבִי. וְלֹא רִחֲמוּ עַל כְּאֵבִי. בָּנִים גִּדַּלְתִּי וְרוֹמַמְתִּי וְהֵם פָּשְׁעוּ בִי. וְעַתָּה בָנַי אַל תִּשְכְּחוּ אַהֲבָתִי. וְאַל תָּפֵרוּ בְּרִיתִי. זִכְרוּ הָאֱמוּנָה. וְהָאַהֲבָה הַיְשָׁנָה. וַעֲלוּ אֵלַי וְעִזְרוּנִי. כִּי נָס לֵחִי וְכָהֲתָה עֵינִי. וּמַהֲרוּ לְהוֹצִיאֵנִי מִן הַפַּחַת. בֶּטֶרם אֵרֵד אֶל הַשַּׁחַת. וּדְעוּ כִּי אִם תִלָּחֲמוּ עַל אֲדוֹנֵיכֶם. ה' יִלָּחֵם לָכֶם.

וְכַאֲשֶׁר הִגִּיעַ הַכְּתָב הַזֶּה לְכָל הַקְּהִלּוֹת בָּכוּ בְּכִי גָדוֹל. וְשָׁבְרוּ כָּל כִּנּוֹר וּמָחוֹל. וּפָשְׁטוּ בִּגְדֵי רִקְמָתָם הַשָּׂרִים הַנִּכְבָּדִים. וּבְכָל מְדִינָה וּמְדִינָה אֲשֶׁר דְּבָרוֹ מַגִּיעַ אֵבֶל גָּדוֹל לַיְּהוּדִים. וְקָרְאוּ עֲצָרָה הַגְּדוֹלִים וְהַחֲשׁוּבִים. וְשַׂק וָאֵפֶר יֻצַּע לָרַבִּים. וַתִּצְלַח רוּחַ אֱלֹהִים עֲלֵיהֶם. וְנוֹעֲצוּ בְּאֶרֶץ הַצְּבִי כָּל גְּדוֹלֵיהֶם. וְנִקְבְּצוּ בְּעֵמֶק הַבָּכָא. בְּעַיִן בּוֹכָה. וְנֶפֶשׁ נְבוֹכָה. וְכָתְבוּ סְפָרִים לְכָל קְהִילּוֹת יִשְׂרָאֵל. לְהוֹדִיעַ כִּי פָּרַץ בָּהֶם פֶּרֶץ הָאֵל.

וְזֶה פַּתְשֶׁגֶן הַכְּתָב: שָׁלוֹם לִקְהִלּוֹת הַקֹּדֶשׁ אַחֵינוּ. הַעֵת לָכֶם אַתֶּם לָשֶׁבֶת בְּבָתֵּיכֶם סְפוּנִים. וְלִלְבּשׁ בִּגְדֵי עֲדָנִים. הָאוֹכְלִים לְמַעֲדַנִּים. וַאדוֹנֵיכֶם תַּחַת בְּרִיחֵי בַרְזֶל מֻכֶּה בַשּׁוֹטִים. מֶלֶךְ אָסוּר בָּרְהָטִים. לָכֵן דְּעוּ כִּי הֵעִיר אֱלֹהִים רַעְיוֹנֵינוּ. לְהִלָּחֵם עַל אֲדוֹנֵנוּ. וְעַתָּה מִי בָכֶם מִכָּל עַמּוֹ. ה' אֱלֹהָיו יְהִי עִמּוֹ. וְיֵצֵא אִתָּנוּ לַמִּלְחָמָה. בְּיָד רָמָה. לַעֲשׂוֹת בְּאוֹיְבֵנוּ נְקָמָה. לָכֵן חַזְּקוּ יָדַיִם וִהְיוּ לִבְנֵי חַיִל. וּבַחֲרוּ לָכֶם מִכָּל יִשְׂרָאֵל אַנְשֵׁי חַיִל. וְהֵחָלְצוּ מֵאִתְּכֶם אֲנָשִׁים. נוֹשְׁכִים כָּנְּחָשִׁים. טוֹרְפִים כַּכְּפִירִים. מְעוֹפְפִים כַּנְּשָׁרִים. שׁוֹטְפִים כַּנְּהָרִים. בַּעֲלֵי זְרוֹעַ אֲרֻכָּה. וְקֶשֶׁת דְּרוּכָה. וּמִלְחָמָה עֲרוּכָה. וְעוֹרְכֵי מַעֲרָכָה. וְכֻלָּם יָבוֹאוּ אֵלֵינוּ אֶל עֵמֶק הַבָּכָא. וּנְחַדֵּשׁ שָׁם הַמְּלוּכָה. וְכַאֲשֶׁר הִגִּיעַ הַכְּתָב הַזֶּה לְאַנְשֵׁי כָל עִיר וָעִיר. הָאֵל לִבּוֹתָם הֵעִיר. הָרַב מֵהֶם וְהַצָּעִיר. וַיִּמָּסְרוּ מֵאַנְשֵׁי יִשְׂרָאֶל בְּאַהֲבָה. כַּחֲמִשִׁים אֶלֶף חֲלוּצֵי צָבָא. גְּדוּדֵיהֶם מִמִּצְרַיִם וְעַד יָם הָעֲרָבָה. וּמֵאֶרֶץ צְבִי וּמִדַּמֶּשֶׂק עַד חֲמַת רַבָּה. וּמֵחָרָן עַד אֲרַם נַהֲרַיִם עַד אַשּׁוּר הַחֲשׁוּבָה. וּמֵאַרְבֵּל עַד עֲדִינָה וְעַד חוֹמַת בָּבֶל הָרְחָבָה. כָּל אֵלֶּה חָבְרוּ לַמִּלְחָמָה. בְּאַף וּבְחֵמָה. וּבְיָד רָמָה. וַיַּחֲנוּ הָמוֹן אֵצֶל הָמוֹן. וְתָקְעוּ אָהֳלֵיהֶם כְּשֶׁלֶג לְבָנוֹן עַל הַר צַלְמוֹן. וּמִתַּחַת הַר חֶרְמוֹן. וְנִתְקַיֵּם הַחֲלוֹם הָאַחֲרוֹן וְהַקָּדְמוֹן. בְּפָרֵשׂ שַׁדַּי מְלָכִים בָּהּ תַּשְׁלֵג בְּצַלְמוֹן. וַיְהִי כַּאֲשֶׁר בָּאָה הַשְּׁמוּעָה לְשׁוֹמְרֵי הַמְּצוּדָה. נָפְלָה עֲלֵיהֶם חֲרָדָה. וְשָׁלְחוּ שְׁלוּחֵיהֶם. לְכָל הַנְּדִיבִים אֲחֵיהֶם. לֵאמֹר: דְּעוּ כִּי הֻכָּה הַכַּפְתּוֹר וְרָעֲשׁוּ הַסִּפִּים. וְעַד אָנָה אַתֶּם מִתְרַפִּים. וּפוֹסְחִים עַל שְׁתֵּי הַסְּעִיפִּים. לָכֵן מַהֲרוּ אַל תֵּעָצְלוּ. וְאוּלַי תִּנָּצְלוּ. וְאִם לֹא תְמַהֲרוּ לָבוֹא נָקוּם לַיְלָה וְנִבְרָחָה. וְנַעֲזֹב הָעִיר פְתוּחָה. וַיָּשִׁיבוּ לָהֶם דָּבָר: אִישׁ מִמְּקוֹמוֹ אַל יָמֵשׁ. וְעַד שְׁלשָׁה יָמִים תִּהְיֶה לָכֶם תְּשׁוּעָה כְּחֹם הַשֶּׁמֶשׁ. וַיְהִי בַּיּוֹם הַשְׁלִישִׁי בִּהְיוֹת הַבֹּקֶר וַיְהִי קוֹלוֹת וּבְרָקִים. מִזִּיו הַחֲרָבוֹת וְהָרְמָחִים הַמְּרוּקִים. וּכְלֵי הְקָרב בַּחשֶׁךְ יָפִיצוּ זֹהַר. וְעָנָן כָּבֵד עַל הָהָר. לְפָנָיו אִישׁ לֹא יַעֲמֹד. וְקוֹל הַשּוֹפָר חָזָק מְאֹד. וְהִנֵּה מַחֲנֵה הַנְּדִיבִים בָּאוּ הֵם וְכָל חֵילֵיהֶם. וְהִצִּיבוּ סְבִיב הַמְּצוּדָה אָהֳלֵיהֶם. וְדִגְלֵיהֶם וְאוֹיְבֵיהֶם.

מִבְּנֵי יְהוּדָה וְאֶפְרַיִם חָנוּ לְנֶגֶד פְּנֵיהֶם. וַיִהְיוּ מַחֲנֵה הַנְּגִידִים וְהַנְּדִיבִים לְנֶגְדָּם כִּשְׁנֵי חֲשִׂיפֵי עִזִּים. וְאֵלֶה מָלְאוּ אֶת הָאָרֶץ כִּשְׁאוֹן מַיִם עַזִּים. וְכַאֲשֶׂר הִתְקַבְּצוּ צִבְאוֹת יְהודָה וְאֶפְרַיִם. כְּחוֹל הַיָּם וְכוֹכְבֵי הַשָּׁמָיִם. קָמוּ שָׂרֵיהֶם וְרוֹזְנֵיהֶם. וַיֹּאמְרּו לְכָל אַנְשֵׁי מַחֲנֵיהֶם. דְּעוּ אַחֵינוּ כִּי כָתוּב בַּתּוֹרָה: כִּי תִקְרַב אֶל עִיר לְהִלָּחֵם עָלֶיהָ. וְקָרָאתָ לְשָׁלוֹם אֵלֶיהָ. וְאַנְשֵׁי הַמְּדִינָה הֵם אַחֵינוּ. וְנַפְשָׁם כְּנַפְשֵׁנוּ. וְרוּחָם כְּרוּחֵנוּ. וְלֹא טוֹב לַעֲלוֹת לְמִלְחָמָה עֲלֵיהֶם. עַד נִקְרָא לְשָׁלוֹם אֲלֵיהֶם. אוּלַי יָשׁוּבוּ מִדַּרְכֵיהֶם. וַיֹּאמְרוּ כֻלָּם: טוֹב הַדָּבָר. וַיִּשְׁלְחוּ מַלְאָכִים בְּנֵי אֶפְרַיִם וִיהוּדָה.

לְאַנְשֵׁי הַמְּצוּדָה. לֵאמֹר: אַתֶּם אַחֵינוּ וְאַנְשֵׁי גְּאֻלָּתֵנוּ. וּמִבְחַר סְגֻלָּתֵנוּ. וְעַתָּה תְּנוּ לָנוּ מַלְכֵּנוּ. וְנֵלֵךְ לְשָׁלוֹם לְדַּרְכֵּנוּ. וְאִם כֹּה תַעֲשׂוּ תַּשְׂכִּילוּ. וְלֹא תַסְכִּילוּ. וּקְחוּ מִמֶּנּוּ כָל אֲשֶׁר תִּשְׁאֲלוּ. וְאִם תְּמָאַנוּ וּמְרִיתֶם חֶרֶב תְּאֻכְּלוּ. וְעַתָּה מַדּוּעַ תִּתְמַכְּרוּ לַמָּוֶת בְּלִי גוֹאֵל. שׁוּבוּ שׁוּבוּ מִדַּרְכֵיכֶם הָרָעִים וְלָמָּה תָמוּתוּ בֵּית יִשְׂרָאֵל. וְכַאֲשֶׁר הִגִּיעוּ שְׁלוּחֵיהֶם לַאֲחֵיהֶם. הָיוּ צוֹחֲקִים עֲלֵיהֶם. וְלוֹעֲגִים עַל דִּבְרֵיהֶם. עַד נִמְלְאוּ עֲלֵיהֶם אֲחֵיהֶם. כַּעַס וְחֵמָה. וְיָצְאוּ לְמִלְחָמָה. בְּיָד רָמָה. וְלָבְשׁוּ שִׁרְיוֹנוֹתֵיהֶם. וּמֵרְקוּ חֲנִיתוֹתֵיהֶם. וַיִּגְּשׁוּ בָרִאשׁוֹנָה. שָׂרֵי עֲדִינָה.

וַיַּעֲנוּ בְקוֹל נְגִינָה וַיֹּאמְרוּ:

מִי זֶה יְהִי עוֹמֵד לְפָנֵינוּ

אוֹ יֶחֱרַץ נֶגֶד הֲמוֹנֵנוּ.

נַחְנוּ בְנֵי הֶבֶל פְּאֵר תֵּבֵל

יָמִים תְּהִלּוֹתָם יְרַנֵּנוּ.

לָנוּ חֲרָבוֹת. תַּעֲלֶה שִׁמְשָׁן

אִם תֶּחֱשַׁךְ שֶׁמֶשׁ לְעֵינֵינוּ.

לֹא נִשְׁקְטָה לוּ בַקְּרָב נָמוּת

עַד נַעֲזֹב שֵׁם טוֹב לְבָנֵינוּ.

עַד כִּי נְמַלֵּט אֶת־אֲדוֹנֵנוּ

לֹא תֶעֱרַב שֵׁנָה לְעֵינֵינוּ.

וַיִּגְשׁוּ גְדוֹלֵי אַשּׁוּר וַיֹּאְמרוּ:

נַחְנוּ נְעוֹרֵר הַקְּרָב עַד כִּי

מוּל הַחֲרָבוֹת תֶּחֱשַׁךְ שִׁמְשׁוֹ.

אִם לַקְּרָב צוּרָה אֲנַחְנוּ רֹאשׁ

אוֹ הוא כְגוּף נַחְנוּ לְבַד נַפְשׁוֹ.

וּבְרַק חֲנִיתֵנוּ – בְּעֵת יִבְעַר –

הַיּוֹם יְהִי נִשְׂרָף בְּאוּר אִשּׁוֹ.

נִקְמַת אֲדוֹנֵנוּ נְבַקֵּשׁ. עַד

רַגְלוֹ נְשַׁלַּח מִכְּלוּא מוֹקְשׁוֹ.

לֹא תֶעֱרַב לָנוּ מְנוּחָה עַד

נָשִׂים לְנַפְשֵׁנוּ פְּדוּת ַנְפשׁוֹ.

וַיַּעֲנוּ אֲצִילֵי אַרְבֵּל וַיֹּאמְרוּ:

אִם יֶחֱשַׁךְ הַזְּמַן חֶרֶב יְמִינֵנוּ

תָּפִיץ מְאוֹרָהּ חֲצוֹת לַיִל לְעֵינֵינוּ.

תַּמְטִיר דְּמֵי הַחֲלָלִים מֵעֲנָנִים עֲדֵי

כִי לָמְדוּ הָעֲנָנִים מֵעֲנָנֵינוּ.

לֹא נִשְׁקְטָה עַד אֲשֶׁר נָמוּת בְּתוֹךְ הַקְּרָב

אוֹ מִיְּדֵי צַר נְחַלֵּץ אֲדוֹנֵנוּ.

בּוֹשֵׁש בְּחִיר אֵל וְלִזְמַן נִקְרְאָה קוּם עֲשֵׂה

לָנוּ אֱלֹהִים אֲשֶׁר יֵלְכוּ לְפָנֵינוּ.

וַיַּעֲנוּ רוֹכְלֵי חָרָן וַיֹּאמְרוּ:

נַחְנוּ קְהַל חָרָן כְּלֵי חָרוֹן

נַעְרֹךְ וְנַעְזֹב שֵׁם לְדוֹר אַחְרוֹן.

לָנוּ חֲרָבוֹת אָדְמוּ מִדָּם

יִדְמוּ חֲבַצֶלֶת בְּתוֹךְ שָׁרוֹן.

נִקְמַת אֲדוֹנֵנוּ נְבַקֵּשׁ עַד

עַל הַזְּמַן לָנוּ יְהִי יִתְרוֹן.

אִם בַּעֲדוֹ נָמוּת אֲזַי נַעְזֹב

שֵׁם טובֹ לְבָנֵינוּ לְזִכָּרוֹן.

וַיַּעֲנוּ בְנֵי אֲרַם נַהֲרַיִם וַיֹּאמְרוּ:

נַחְנוּ יְדוּעִים מֵאֲרַם נַהְרַיִם

לַעְרֹךְ קְרָבוֹת עִם צְבָא שָׁמָיִם.

לֹא נִשְׁקְטָה עֶד כִּי נְחַדֵּשׁ לַאדוֹ־

נֵינוּ מְלוּכָה עַל בְּנֵי אֶפְרַיִם.

נִשְׁפֹּךְ דְּמֵי כָל־אוֹיְבָיו עַד נַעְזֹב

אֶרֶץ בְּדָמִים כַּאֲפִיקֵי מָיִם.

וַיַּעֲנוּ גִבּוֹרֵי כַּלְנֵה וַיֹּאמְרוּ:

נַחְנוּ פְּאֵר הַדּוֹר בְּנֵי כַלְנֵה

עִם כּוֹכְבֵי רוּם נַעֲרֹךְ מַחְנֶה.

נִטְרֹף מְשַׂנְאֵינוּ וְאוֹיְבֵנוּ

כִּכְפִיר בְּעֵת עָבְרוֹ בְּתוֹךְ מִקְנֶה.

נָרוּץ בְּרֹב אַהֲבָה לְמוּל מָוֶת

טֶרֶם אֲשֶׁר הוּא יִקְרְאָה נַעְנֶה.

לֹא נֶחֱשֶׁה עַד כִּי אֲדוֹנֵנוּ

נִגְאַל בְּכֹחַ מִיְּדֵי מוֹנֶה.

עוֹד אַחֲרֵי נָפַל מְכוֹן הוֹדוֹ

יָרוּם עֲדֵי שַׁחַק וְיִבָּנֶה.

אִם נִפְזְרוּ צֹאנוֹ לְכָל־פֵּאָה

עוֹד תַּעֲבֹרְנָה עַל יְדֵי מוֹנֶה.

וַיַּעֲנוּ גִבּוֹרֵי הַפִּסְגָּה וַיֹּאמְרוּ:

נַחְנוּ בְנֵי פִסְגָה אֲצִילֵי עָם

עוֹרְכֵי קְרָבוֹת יֹצְאֵי צָבָא.

אִם בַּקְּרָב נָבוֹא תְּדַמֶּה כִּי

נַחְנוּ חֲצוּבֵי אֵשׁ וְלֶהָבָה.

לוּ נַעֲבֹר בַּיָּם שְׁתִינוּהוּ

וַנַּהֲפֹךְ אוֹתוֹ לְחָרָבָה.

לֹא נֶחֱשֶׁה עַד כִּי אֲדוֹנֵנוּ

נַצִּיל בְּעֹז מִיַּד מְתֵי אֵיבָה.

נַחְלֹץ גְּדוּד צָרִים וְאִישׁ אֶחָד

בָּנוּ לְאֶלֶף קָם וְלִרְבָבָה.

וַיַּעֲנוּ אַנְשֵׁי חֲמָת וַיֹּאמְרו:

חֲמַת אַנְשֵׁי חֲמָת תַּחְצֹב לְהָבִים

וְתַצִּית אֵשׁ קְרָבוֹת בַּקְּרָבִים.

אֲנַחְנוּ כַכְּפִיר נִטְרֹף לְאוֹיְבִים

וְנִשְׁלַח בַּמְשַׂנְאִים שֵׁן זְאֵבִים.

עֲדֵי נַצִּיל אֲדוֹנֵנוּ וְנַעְזֹר

חֲיָלָיו מִידֵי צָרִים וְאוֹיְבִים.

בְּשֶׁטֶף הַקְּרָב לָנוּ לְבָבוֹת

אֲשֶׁר מִלַּהֲבוֹת אֵשׁ הֵם חֲצוּבִים.

וַיַּעֲנוּ גְדוּדֵי דַמֶּשֶׂק וַיֹּאמְרוּ:

הָאֵל קְרָאָנוּ בְּעֹז אֶל־עָל

עֵת כִּי בְשִׁרְיוֹן הַקְּרַב נִתְעַל.

וּבְרַק חֲנִיתֵנוּ בְּעֵת יָפִיץ

לַהַב יְלַהֵט כּוֹכְבֵי מָעַל.

מִיּוֹם אֲשֶׁר נִתְפַשׂ אֲדוֹנֵנוּ

נָקוּץ בְּחַיֵּינוּ וְגַם נִגְעַל.

לֹא נִשְׁקְטָה עַד נַחֲלשׁ קָמָיו

יַחַד וְנַשְׁקֵמוֹ בְּסַף רַעַל.

עַד שׁוֹמְעֵי שִׁמְעוֹ בְּסוֹף אֶרֶץ

יֹאמְרוּ רְאוּ גוֹי זֶה וּמַה פָּעַל.

וַיַּעֲנוּ צִבְאוֹת אֶרֶץ הַצְּבִי וַיֹּאמְרוּ:

נַחְנוּ צְבָא הָאֵל בְּאֶרֶץ הַצְּבִי

עַזִּים כְּמוֹ אַרְיֵה וְקַלִּים כַּצְּבִי.

לוּ נִלְחֲמוּ בָנוּ צְבָא כוֹכְבֵי זְבוּל

הָלְכוּ גְדוּדֵיהֶם לְפָנֵינוּ שְׁבִי.

מִיּוֹם אֲדוֹנֵנוּ מְשַׂנְאִים תָּפְשׂוּ

נָפַל גְּאוֹנֵנוּ וְהוּסַר כָּל צְבִי.

נֹאמַר לְנַפְשֵׁנוּ חֲכִי עַד יַחֲמֹל

הָאֵל וְעַד כִי יַעֲבֹר זַעַם חֲבִי.

וַיַּעֲנוּ חֲלוּצֵי מִצְרַיִם וַיֹּאמְרוּ:

הִנֵּה אֲדֹנָי שׁוֹכְנִי שָׁמַיִם

רוֹכֵב עֲלֵי עָב קַל וּבָא מִצְרַיִם

לַעְזֹר כְּמִקֶּדֶם בְּמִלְחֶמֶת קְרָב

זֶרַע יְהוּדָה עִם בְּנֵי אֶפְרַיִם.

נַחְנוּ כְפִירֵי הַקְּרָב עֵת נִשְׁאֲגָה

יַלְדֵי עֲנָק לֹא יִמְצְאוּ יָדַיִם.

אֵיךְ יַעֲלֹז הַלֵּב וּמַלְכֵּנוּ בְיַד

צָרִים בְּתוֹךְ כֶּבֶל וּבִנְחֻשְׁתַּיִם.

לֹא נִשְׁקְטָה עַד כִּי בְשֶׁטֶף הַקְּרָב

הָאוֹיְבִים יִמָּאֲסוּ כַמַּיִם.

וַיַּעֲנוּ אַדִּירֵי נֹא אָמוֹן וַיֹּאמְרוּ:

מִקּוֹל שְׁאוֹן רַעַם גְּדוּדֵנוּ

הָמוּ וְרָגְזוּ נַחֲלֵי שִׁיחוֹר.

כִּי צוּר חֲנִיתֵנוּ כְּצֹר בָּרָק

יָרוּץ לְלֵב אוֹיְבִים וְיִקֹּב חֹר.

קַצְנוּ בְחַיִּים כִּי אֲדוֹנֵנוּ

נִתְפַּשׂ וּמָוֶת נֶאֱהַב לִבְחֹר.

לֹא נִשְׁקְטָה עַד כִּי בְחַרְבֵּנוּ

הַצָּר לְאָחוֹר יִסְחֲרָה סָחוֹר.

אִם אוֹר פְּנֵי הַיּוֹם יְהִי לָבָן

יָשׁוּב בְּאַבְקַת הַקְּרָב שָׁחוֹר.

בָּעֵת כְּיָם יֵצְאוּ גְדוּדֵינוּ

יַם־סוּף הֲלֹא יֶחְרַב וְשָׁב אָחוֹר.

אָמַר הַמַּגִּיד: וַיְהִי אַחֲרֵי כֵן הִתְקַבְּצוּ הַחֲבוּרוֹת. בִּגְבוּרוֹת. וְנוֹעֲדוּ בִקְשָׂתוֹת דְּרוּכוֹת. וּמַעֲרָכוֹת מוּל מַעֲרָכוֹת. וַיַּעַל הָאָבָק עַד לֵב הַשָּׁמָיִם. וּבָאָה הַשֶּׁמֶשׁ בַּצָּהֳרָיִם. וְצָעֲקוּ הַגִּבּוֹרִים כַּיּוֹלֵדָה עַל אָבְנָיִם. וְנִגְרְשׁוּ הַדָּמִים כִּיאוֹר מִצְרָיִם. וְלַהַב הַחֲרָבוֹת בּוֹקֵעַ אוֹר מֵאֹפֶל. וְאוֹר הַיּוֹם זוֹעֵף וְאָפֵל. וְהָהָר בּוֹעֵר בָּאֵשׁ עַד לֵב הַשָּׁמַיִם חשֶׁךְ עָנָן וַעֲרָפֶל. וַיְהִי בַּעֲלוֹת הַצָּהֳרָיִם. הֵרִיעוּ תְרוּעָה גדולה כָל בְּנֵי הַקְּהִלּוֹת הֵם בְּנֵי יְהוּדָה וְאֶפְרַיִם. וַתַּעַל שַׁוְעָתָם אֶל הָאֱלֹהִים לַשָּׁמַיִם. וְנִשְׁבְּרוּ לִפְנֵיהֶם כָּל הַנְּדִיבִים אוֹיְבֵיהֶם וּשְׁטָפוּם כְּמַיִם אַדִּירִים. וַיַּכּוּם בַּמּוֹרָד עַד הַשְּׁבָרִים. וַיְעוֹלְלוּ אוֹתָם בַּגֵּאָיוֹת וּבֶהָרִים. וְלָכְדוּ לָהֶם הַמַּעֲבָרִים. וְהַשְׂרִידִים בָּרְחוּ וְנֶחְבְּאוּ בִּמְעָרוֹת צוּרִים. וְאוֹיְבֵיהֶם גָּבְרוּ עֲלֵיהֶם וּבָאוּ כָל קְהִלּוֹת הַקֹּדֶשׁ יָחַד. וַיְמַהֲרוּ כְאִישׁ אֶחָד. וְשָׁבְרוּ שַׁעֲרֵי הַמְּצוּדָה. בְּקוֹל שָׁאוֹן וַחֲרָדָה. וְכִמְעַט לְקוֹל שְׁאוֹנֵיהֶם. תִּבָּקַע הָאֲדָמָה תַּחְתֵּיהֶם. וְהוֹצִיאוּ הַכִּילוּת מִבֵּית כִּלְאוֹ. וַיְרִימוּהוּ עַל כִּסְאוֹ. וּפָנָיו מְשֻׁנִּים כְּשׁוֹכְנֵי קֶבֶר. מֵרֹב הַיָּגוֹן וְהַשֶּׁבֶר. וְהִלְבִּישׁוּהוּ בִגְדֵי מַלְכוּת. וַיִּתְּנוּ עָלָיו אֶת הַנֵּזֶר וְאֶת הָעֵדוּת. וְהִקִּיפוּ סְבִיבוֹ כָל רָאשֵׁי הַקְּהִלּוֹת וּגְדוֹלֵיהֶם. וַיַּעַמְדוּ מִימִינוֹ שָׂרֵי עֲדִינָה וְרוֹזְנֵיהֶם. וּמִשְּׂמֹאלוֹ הֲמוֹנֵי אַשּׁוּר וּקְצִינֵיהֶם. וּלְפָנָיו נֶאֶמְנֵי כַלְנֵה וַחֲסִידֵיהֶם. וּמֵאַחֲרָיו גִבּוֹרֵי הַפִּסְגָּה וַאֲצִילֵיהֶם. וְנֶגְדּוֹ רוֹפְאֵי דַמֶּשֶׂק וַעֲשִׁירֵיהֶם. וַיַּעֲבֹר מַלְכָּם לִפְנֵיהֶם. וַיְהַלֲלוּ אֵת ה' אֱלֹהֵיהֶם. עַל כָּל הַטּוֹבָה אֲשֶׁר עָשָׂה לָהֶם. וְהִשְׁתַּחֲווּ לוֹ גְדוֹלֵיהֶם וּקְטַנֵּיהֶם. אָז נָשָׂא עֵינָיו לָהֶם. וַיִּזְכֹּר חֲלוֹמוֹ אֲשֶׁר חָלַם לָהֶם. וַיִּשְׂמְחוּ כֻלָּם בַּמֶּלֶךְ. הָאֶזְרָח וְהַהֵלֶךְ. וְהֶעָשִׁיר וְהַחֵלֶךְ.

וַיֹּאמְרוּ לוֹ: יִחְי הַמֶּלֶךְ. וַיַּעֲשׂוּ יוֹם גְּאֻלָּתוֹ יוֹם מִשְׁתֶּה וְשִׂמְחָה. וּמִדֵּי שָׁנָה בְשָׁנָה יַעֲשׂוּ בוֹ יוֹם הַשְׁקֵט וּמְנוּחָה. וּמִן הַיּוֹם הַהוּא גָבְרָה יַד הַכִּילוּת. עַל כָּל בְּנֵי הַגָּלוּת. וּמֵאָז וְעַד עַתָּה פָּשְׁטָה מַלְכוּתוֹ עַל כָּל הָעוֹלָם וְלֹא נִשְאַר לִנְדָבָה בְּכָל אַרְצוֹת מִזְרָח זִכָּרוֹן. וְלֹא שֵׁם וְלֹא יִתְרוֹן. מֵעֳדִינָה וְעַד אֶרֶץ צְבִי וְעַד חֶרְמוֹן. וּמֵהַמִּדְבָּר הַגָּדוֹל וְהַלְּבָנוֹן עַד הַיָּם הָאַחֲרוֹן. כִּי אִם כְּלֶקֶט מִן הֶעָמִיר. שְׁנַיִם שְׁלשָׁה גַרְגְּרִים בּרֹאשׁ אָמִיר.

אָמַר הַמַּגִּיד: עַל זֶה הָיָה דָוֶה לִבִּנוּ. וְגָדַל עָצְבֵּנוּ. וְנָמַס רוּחֵנוּ בְּקִרְבֵּנוּ. עַל הַר הַנְּדָבָה שֶׁשָּׁמֵם נְתִיבוֹ. שׁוּעָלִים הִלְּכוּ בוֹ. הֲשִׁיבֵנוּ ה' אֵלֶיךָ וְנָשׁוּבָה וּשְׁבֹר מַלְכוּת הַכִּילוּת הַשְּׁדוּדָה. וְהַצְמַח קֶרֶן לַנְּדָבָה הַחֲמוּדָה. בְּיָמָיו תִּוָּשַׁע יְהוּדָה.
