ספר תחכמוני - שַׁעַר שְׁמֹנָה עָשָׂר

בִּידִיעַת הַשִּׁיר מֵאַיִן עָלָה וְיָרָד. אִם מִבְּנֵי עֲרָב אוֹ מִבְּנֵי סְפָרָד


נְאֻם הֵימָן הָאֶזְרָחִי: נָסַעְתִּי בְעוֹדִי נַעַר רַךְ. מֵאֶרֶץ חַדְרָךְ. לְעִיר הַקֹּדֶשׁ אֲשֶׁר שָׁם נֵר הַמְּלוּכָה נֶעֱרָךְ. וְאוֹר הַשְּׁכִינָה דָרָךְ. וְנִתְחַבַּרְתִּי בָהּ עִם בְּחִירֵי אֵל. וְנוֹעַדְתִּי שָׁמָּה אֶל בְּנֵי יִשְׂרָאֵל. וַיְהִי בִקְצָת הַיָּמִים. וַאֲנִי יוֹשֵׁב בִּקְהַל חֲכָמִים. נִכְבָּדִים וְרָמִים. מְחַבְּרִים מִן הַשִּׁיר לְשָׁמִים. וּמְפִיצִים מִנּוֹזְלֵי עִנְיָנָיו גְּשָׁמִים. וְהָיוּ מְדַבְּרִים בְּעִנְיַן הַשִּׁיר וְסוֹדוֹתָיו. לָדַעַת אֵיךְ נִבְנוּ מִתְּחִלָּה יְסוֹדוֹתָיו. וּמִי הֵקִיר לַיְּהוּדִים מְקוֹרוֹתָיו. וְהִזְרִיחַ לָהֶם מְאוֹרוֹתָיו. וַיַּעֲנוּ אֲנָשִׁים מֵהֶם וַיֹּאמְרוּ: הִנֵּה יָדַעְנוּ כֻלָּנוּ. וַאֲבוֹתֵינוּ סִפְּרוּ לָנוּ. כִּי הַשִּׁיר בִּבְנֵי עֲרָב הוּא נַחֲלָתָם. מִיּוֹם הֱיוֹתָם עַל אַדְמָתָם. אַךְ לֹא יָדְעוּ מָתַי צָלְחָה רוּחַ הַשִּׁירִים. עַל הָעִבְרִים. וּמִי פָתַח לָהֶם בַּתְּחִלָּה שְׁעָרָיו. וּבָא בַחֲדָרָיו. לִלְקֹט בְּדֹלָחָיו וְדָרָיו. וּמִי בָקַע פְּלָגָיו וְאַגְמֵיהֶם. וּמִי מָדַד בְּשָׁעֳלוֹ מֵימֵיהֶם. וּבַזֶּרֶת תִּכֵּן שְׁמֵיהֶם. וְהִרְבּוּ לְדַבֵּר כָּל אִישׁ לְפִי דַרְכּוֹ. זֶה אוֹמֵר בְּכֹה וְזֶה אוֹמֵר בְכֹה. וְשָׁם נִקְרָא אִישׁ זָקֵן וְחָכָם. וְהוּא יוֹשֵׁב בְּתוֹכָם. וְשׁוֹמֵעַ מִלֵּיהֶם. וְנוֹדַע מִמַּרְאֵה עֵינָיו כִּי הוּא שׂוֹחֵק וְלוֹעֵג עֲלֵיהֶם. וְכַאֲשֶׁר נָתְנוּ דִבְרֵיהֶם וְנָשָׂאוּ. וְשׁוֹטְטוּ לְבַקֵּשׁ וְלֹא מָצָאוּ. וַיִּיגְעוּ לָצֵאת מִיָּם הַסִּכְלוּת וַיָּצֹלּוּ. וַיַּחְתְּרוּ הָאֲנָשִׁים לְהָשִׁיב אֶל הַיַּבָּשָׁה וְלֹא יָכֹלוּ. אָמַר לָהֶם: הוֹי הַבָּאִים בְּנַחַל עָלָה וְגָבֲר. וְשָׁטַף וְעָבַר. נְבוּכִים אַתֶּם בָּאָרֶץ בְּלִי מַעֲבָר. וְסָגַר עֲלֵיכֶם הַמִּדְבָּר. וְהַשְּׁאֵלָה אֲשֶׁר שְׁאֶלְתֶּם הִיא מִשַּׁחַק רְחוֹקָה. וּמִנִּי יָם עֲמֻקָּה.

וְאֵין אִישׁ זוּלָתִי יוֹדֵעַ סוֹדָהּ. וְנוֹגֵעַ עַד יְסוֹדָה.

אָמַר הַמַּגִּיד: וְכִשְׁמֹעַ הָעָם נִיבוֹ. הִתְקַבְּצוּ כֻלָּם סְבִיבוֹ. לִקְדֹּחַ לָהֶם שְׁבִיבוֹ. וּלְהַרְעִיף עֲלֵיהֶם רְבִיבוֹ. וַיֹּאמֶר לָהֶם: דְּעוּ כִּי הַשִּׁיר הַנִּפְלָא. אֲשֶׁר בִּפְנִינִים מְמֻלָּא. הָיָה בַתְּחִלָּה לִבְנֵי עֲרָב לְנַחֲלָה. וְהֵם חִזְּקוּ בוֹ כָל בֶּדֶק. וּשְׁקָלוּהוּ בְּמֹאזְנֵי צֶדֶק. וְהִצִּיבוּהוּ עַל מְכוֹנוֹ. וְהֵקִימוּ מִשְכָנוֹ. וְעָרְכוּ שֻׁלְחָנוֹ. וְהֶאֱכִילוּ מָנוֹ. וּבָנוּ עֲלִיּוֹתָיו. וּפִסְּגוּ אַרְמְנוֹתָיו. הֵם קֵרוּהוּ וַיַּעֲמִידוּ דַלְתוֹתָיו. וְלָהֶם עַל מְשׁוֹרְרֵי תֵבֵל הַמַּעֲלָה הָעֶלְיוֹנָה. וְכִי יִקָּרְאוּ אַנְשֵׁי הַשִּׁיר אֶל עָל הֵם יַעֲנוּ וְיַעֲלוּ רִאשׁוֹנָה. וְאַף עַל פִּי שֶׁיֵּשׁ גַּם בְּכָל אֻמָּה מְשׁוֹרְרִים. וּמִתְעַסְּקִים בִּמְלֶאכֶת הַשִּׁירִים. כָּל שִׁירֵיהֶם לְנֶגֶד שִׁירֵי יִשְׁמְעֵאלִים לֹא לְעֵזֶר וְלֹא לְהוֹעִיל. וְכָל מִלֵּיהֶם הֶבֶל הֵם וְאֵין בָּם מוֹעִיל. כִּי אֵין הַשִּׁיר הַנָּעִים בְּמִבְטָאוֹ. הֶעָרֵב בְּמִקְרָאוֹ. כַּשֶּׁמֶשׁ בְּשִׂיאוֹ. וְכַשַׂחַר נָכוֹן מוֹצָאוֹ. כִּי אִם לִבְנֵי עֲרָב לְבַדָּם. וְכָל הַגּוֹיִם כְּאַיִן נֶגְדָּם. רַק בְּנֵי עַמֵּנוּ אַחַר גְּלוֹתָם מֵאַדְמָתָם. שָׁכְנוּ רַבִּים מֵהֶם עִם בְּנֵי עֲרָב בְּאַרְצוֹתָם. וְנָהֲגוּ לְדַבֵּר בִּלְשׁוֹנָם. וְלַהֲגוֹת בְּהֶגְיוֹנָם. וּבְהִתְעָרְבָם עִמָּהֶם. לָמְדוּ מְלֶאכֶת הַשִּׁיר מֵהֶם. כְּמוֹ שֶׁנֶּאֱמַר: וַיִּתְעָרְבוּ בַגּוֹיִם וַיִּלְמְדוּ מַעֲשֵׂיהֶם. כִּי בְעוֹד הָיוּ אֲבוֹתֵינוּ שׁוֹכְנִים בְּעִיר הַקֹּדֶשׁ. לֹא נוֹדַע לָהֶם שִׁיר שָׁקוּל בִּלְשׁוֹן-הַקֹּדֶשׁ. רַק סֵפֶר אִיּוֹב וּמִשְׁלֵי וּתְהִלִּים. הֵם פְּסוּקִים קְצָרִים וְקַלִּים. וְיִדְמוּ לַחֲרוּזֵי שִׁיר אֲבָל אֵין לָהֶם חֲרוּז וְאֵינָם שְׁקוּלִים. וּבַעֲבוּר כִּי קְצָת שֵׁבֶט יְהוּדָה לִסְפָרַד שָׂרָד. כְּמוֹ שֶׁנֶּאֱמַר: וְגָלוּת יְרוּשָׁלִִַם אֲשֶׁר בִּסְפָרָד. וּגְבוּל סְפָרַד הוּא כְּנֶגֶד אֶמְצַע הַשָּׁמַיִם מְכֻוָּן מְיֻשָּׁר. מִתַּחַת הַקַּו הַשָּׁוֶה וְהַיָּשָׁר. כִּי בְנֵי בָבֶל בֶּקַּו הַשָּׁוֶה בַּמִּזְרָח. וּבְנֵי סְפָרַד בְּסוֹף קַו הַשָּׁוֶה בַּמַּעֲרָב. וְעַל כֵּן פָּשְׁטוּ הַחָכְמוֹת בִּשְׁתֵּי הַקְּצָווֹת הָאֵלֶּה לְמִינֵיהֶם. כִּי הֵם מִשְּׁנֵי צדּי הַגַּלְגַּל וְעַמּוּדֵיהֶם. אֲשֶׁר הַבַּיִת נָכוֹן עֲלֵיהֶם. וּבִשְׁנֵי הַמְּקוֹמוֹת הָאֵלֶּה הִתְפָּאֵר הָעוֹלָם וְהִתְגַּבָּר. שְׁתֵּי כְּתֵפוֹת יִהְיֶה לוֹ עַל שְׁנֵי קְצוֹתָיו וְחֻבַּר. כִי הָיוּ מִקֶּדֶם בְּבָבֶל. חַכְמֵי תֵבֵל. וְהָיוּ כְּמוֹ כֵן בִּסְפָרָד. בַּעֲלֵי חָכְמוֹת בְּכָל חָכְמָה וּבְסוֹד הַשִּׁיר מְחֻבָּר וּמְפֹרָד. וְעַל כֵּן לְשׁוֹנָם בְּצַחוּתוֹ נִשְׁמָר. וְעָמַד טַעְמוֹ בוֹ וְרֵיחוֹ לֹא נָמָר. וַיְהִי לַשִּׁיר בְּתוֹךְ אַנְשֵׁי סְפָרַד מַחֲנֶה. וּבְטֶרֶם יִקְרָאוּהוּ יַעֲנֶה. מִלִּבָּם יִסַּע וְעַל פִּיהֶם יַחֲנֶה.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

עֵת הַחֲכָמִים מֵעֲנוֹת שִּׁיר נֶחְבְּאוּ

חַכְמֵי סְפָרַד לַעֲנוֹתוֹ נִצְבְּאוּ.

וּבְעֵת בְּנֵי מִזְרָח בְּשִׁיר לֹא מָצְאוּ

חָזוֹן בְּנֵי הַמַּעֲרָב הִתְנַבְּאוּ.

וַיְהִי בִשְׁנַת אַרְבַּעַת אֲלָפִים וּשְׁבַע מֵאוֹת לַיְצִירָה. צָלְחָה עַל בְּנֵי סְפָרַד בַּמֵּאָה הַשְּׁבִיעִית רוּחַ עֵצָה וּגְבוּרָה. וְשָׂפָה בְּרוּרָה. וְאָז הִתְחִילוּ הָעִבְרִים. לְהַרְגִּיל לְשׁוֹנָם בְּדַרְכֵי הַשִּׁירִים. וְהִתְחִילוּ יְקוֹדֵי הַמְּלִיצוֹת אֶת לִבָּם לְהַבְעִיר. וְהַמַּשְׂאֵת הֵחֵלָה לַעֲלוֹת מִן הָעִיר. אַךְ מְלִיצָתָם בָּעֵת הַהִיא הָיְתָה גְרוּעָה. וּלְשׁוֹנָם בִּנְתִיב הַשִּׁיר צְלוּעָה. עַד בּוֹא הַמֵּאָה הַשְּׁמִינִית. וְאָז הִתְעוֹרְרוּ לְנַגֵּן עַל הַשְּׁמִינִית. וְהֵכִינוּ בֵית הַשִּׁיר כַּמִּשְׁפָּט וְכַתַּבְנִית. וּמָדְדוּ אֶת תָּכְנִית. בַּיָּמִים הָהֵם זָרַח בִּסְפָרַד שֶׁמֶשׁ הַתְּהִלָּה. בִּרְקִיעַ הַגְּדֻלָּה. הוּא הַנָּשִׂיא הַגָּדוֹל. רַבִּי יִצְחָק הַסְּפָרַדִי בֶּן חִסְדָּאי.

יָנוּחַ בְּצֵל שַׁדָּי. כִּי הֵרִיק לְכָל שׁוֹאֲלָיו בְּרָכָה עַד בְּלִי דָי. וּבְיָמָיו הָמוּ גַּלֵּי הַחָכְמוֹת. וַיִּגְרְשׁוּ מֵימֵיהֶם גָּבִישׁ וְרָאמוֹת. וְנִצְּבוּ נוֹזְלֵיהֶם כְּחוֹמוֹת. וְכָל נֶפֶשׁ אֲשֶׁר צָמְאָה לְמֵי חָכְמָה רִוְּתָה בוֹ צִמְאוֹנֶיהָ. וַיִּפְקַח אֱלֹהִים אֶת עֵינֶיהָ. וַתֵּרֶא בְּאֵר מַיִם מְתוּקִים מִצּוּף לַשְּׂפָתַיִם. וַתֵּלֶךְ וַתְּמַלֵּא אֶת הַחֵמֶת מַיִם. כִּי הַנָּשִׂיא הַהוּא הֶחְיָה בְטַלְלֵי חֲסָדָיו מֵתֵי הַסִּכְלוּת. וּמָשַׁךְ בַּעֲבוֹת נִדְבוֹתָיו לִבּוֹת בְּנֵי הַגָּלוּת. וְהֶעֱבִיר קוֹל: מִי לַה' אֵלַי. וְכָל מַחְסוֹרוֹ עָלַי. וַיֵּאָסְפוּ אֵלָיו כָּל גָּאוֹן וְכָל רָב.

מֵאַרְצוֹת אֱדוֹם וַעֲרָב. וּמִמִּזְרָח וּמִמַּעֲרָב. וְעָרַךְ שֻׁלְחַן חֲסָדָיו לִפְנֵיהֵם. וַעֲנַן הוֹדוֹ עוֹמֵד עֲלֵיהֶם. וְעַמּוּד אֵשׁ כְּבוֹדוֹ יָאִיר לְעֵינֵיהֶם. וַיִּתְקַבְּצוּ אֵלָיו חַכְמֵי הַדּוֹר מְאִירִים בְּאוֹרֵי נְגוֹהִים. לַעֲרֹךְ מַעֲרֶכֶת חָכְמָה לִפְנֵי דוֹרְשֵׁי אֱלֹהִים. וַיְמַלֵּא אוֹתָם רוּחַ אֱלֹהִים. בְּחָכְמָה בִּתְבוּנָה וּבְדַעַת נְסוּכַה. וּבְכָל מְלָאכָה. לַחְשֹׁב מַחֲשָׁבוֹת. וְלַחְצֹב מִן הַלְּבָבוֹת. מוֹקְדֵי לֶהָבוֹת. וּמִן הָעֵת הַהִיא פָּרְצוּ הַחָכְמוֹת בִּסְפָרַד פָּרֶץ. וּמֵאֵלֶּה נָפְצָה כָּל הָאָרֶץ.

וּמֵאָז לָמְדוּ לֶקַח רוֹגְנִים. וְקִדְּמוּ שָׁרִים אַחַר נוֹגְנִים. כִּי חֲסָדָיו הִרְנִינוּ לְשׁוֹן אִלְמִּים. וּפָתְחוּ הַלְּבָבוֹת הַסְּתוּמִים. עַד חִבְּרוּ עָלָיו שִׁירִים נְעִימִים. יָאִירוּ כְּכוֹכְבֵי מְרוֹמִים. וְאָז נִפְקְחוּ בִּמְלֶאכֶת הַשִּׁיר הָעֵינַיִם. וְנִפְתְּחוּ הַשָּׁמַיִם. וְנִרְאוּ מַרְאוֹת אֱלֹהִים. וּבְיָמָיו פָּשְׁטָה הַחָכְמָה בְּיִשְׂרָאֵל. כִּי הוּא הָיָה לַמַּדָּע מוֹדַע וְגוֹאֵל. וְאַחֲרֵי הַזְּמַן הַהוּא. כָּהוּ הַמְּאוֹרוֹת אֲשֶׁר נָגְהוּ.

וְנָפְלוּ הָעַמּוּדִים אֲשֶׁר גָּבְהוּ. וַיָּמָת הַנָּשִׂיא וְכָל אֶחָיו וְכָל הַדּוֹר הַהוּא. וּבָאָה הַמֵּאָה הַתְּשִׁיעִית וּבָהּ נוֹלַד רַבִּיִ יְצְחָק בֶּן כַּלְפוֹן. אֲשֶׁר מִלֵּא בְשִׁירָיו פְאַת יָם וְצָפוֹן. וְהוּא הָעוֹבֵר בְּמַעֲרֶכֶת הַשִּׁיר רִאשׁוֹן. וְנָתַן לוֹ הָאֵל בִּקְצָת שִׁירָיו מַעֲנֵה לָשׁוֹן. וְהוּא הֵחֵל לִהְיוֹת גִּבּוֹר בַּשִּׁירִים. וְהוּא פָתַח דְּלָתָיו הַסְּגוּרִים. לִבְנֵי הָעִבְרִים. וְאָז יָרְדוּ לַשְּׁעָרִים. אַף עַל פִּי אֲשֶׁר מְעַט מִשִּׁירָיו כִּפְרִי עֵץ הָדָר נֶהְדָּרִים. וְרֻבָּם קוֹצִים וְדַרְדָּרִים. וְהָיוּ בְדוֹרוֹ מְשׁוֹרְרִים רַבִּים. יֶהֱמוּ כְדֻבִּים. אַךְ שִׁירוֹ עָלָה עַל כֻּלָּם. וְנִמְשַׁח לְמֶלֶךְ עַל קְהָלָם. וּבְיָמָיו הָיָה הַנָּגִיד הַגָּדוֹל רַבִּי שְׁמוּאֵל הַלֵּוִי זַ"ל הַחוֹשֵׂף בִּמְלֶאכֶת הַשִּׁיר זְרוֹעַ הַבָּאִים אַחֲרָיו. רָצוּ לִלְמֹד מִשִּׁירָיו. וְלֹא הִשִּׂיגוּ אֲבַק רַגְלָיו. לְתֹקֶף מְשָׁלָיו. וְחֹזֶק מִלָּיו. וְאִלּוּ הֶאֱרִיךְ יָמִים. חִבֵּר בְּסוֹדוֹת הַשִּׁיר פְּלָאִים עֲצוּמִים. אַךְ נִקְטַף זֵרוֹ בְּאִבּוֹ. וְנֶחְטַף וְלַחוּת הַנַּעֲרוּת בּוֹ. וּבֶן תֵּשַׁע וְעֶשְׂרִים נֵרוֹ כָבָה. וְעַד הַשְׁלשִׁים לֹא בָא. וּבְתוֹךְ הַמֵּאָה הַתְּשִׁיעִית אָבְדוּ הַמְשׁוֹרְרִים הָהֵם רַק נִשְׁאֲרוּ חֲמֻדוֹתָם. וְנֶאֶסְפוּ הֵם לְעַמָּם וְחָיוּ תְהִלּוֹתָם. וְגַם כָּל הַדּוֹר הַהוּא נֶאֶסְפוּ אֶל אֲבוֹתָם. וַיָּקָם אַחֲרֵיהֶם דּוֹר הַמְשׁוֹרְרִים הַנַּעֲמָנִים. וְנִקְרָא זְמַנָּם זְמַן הַנִּצָּנִים. כִּי הָיָה הַדּוֹר כְּמוֹ קוֹצִים וְהֵם כְּשׁוֹשַנִּים. וְהוּא דוֹר הַמְשׁוֹרְרִים הַנִּפְלָאִים. הַקְּרוּאִים חֶבֶל נְבִיאִים. אֲשֶׁר הָיָה מֵהֶם הַלֵּוִי רַבִּי יְהוּדָה. תּוֹר הַתּוֹרָה וַעֲדִי הַתְּעוּדָה. וּשְׁאָר מְשׁוֹרְרֵי דוֹרוֹ יִלָּוֶה עֲלֵיהֶם הַשָּׁלוֹם וְאַף עַל פִּי שֶׁלֹּא הִגִּיעַ אֶחָד מֵהֶם בְּשִׁירָיו לְמַעֲלוֹת רַבִּי שְׁלֹמֹה. וְלֹא הִשִּׂיג לְתֹקֶף נָאֳמוֹ. וּמֶתֶק טַעְמוֹ. אַךְ הָיוּ לָהֶם מִלִּים נִפְלָאִים. וַחֲרוּזִים נוֹרָאִים. וְעַל פִּיהֶם נִרְאוּ נוֹרְאוֹת הַשִּׁירִים הָעֲרֵבִים. כְּאִלּוּ הֵם לְקוּחִים מִפִּי כְרוּבִים. אוֹ מִלְּשׁוֹן חוֹזִים גְּנוּבִים. וַתָּנַח עֲלֵיהֶם הָרוּחַ וְהֵמָּה בַכְּתוּבִים. וַיִּתְנַבְּאוּ וְלֹא יָסָפוּ. וְלִשְׁמֹעַ נוֹרְאוֹת שִׁירָם נְדִיבֵי עַמִּים נֶאֱסָפוּ.

בְּחֶזְיוֹן שִׁיר בְּמַרְאֶה וְלֹא בְחִידוֹת.


וְאַחֲרֵי הַדּוֹר הַהוּא נִסְתְּמוּ מְקוֹרֵי הַשִּׁירִים. בִּלְשׁוֹן הָעִבְרִים. וּמְלִיצַת הַשִּׁיר שֻׁלַּח בְּמִשְׁמַנֶּיהָ רָזוֹן. גַּם נְבִיאֶיהָ לֹא מָצְאוּ חָזוֹן. וְכָל הַבָּאִים אַחֲרֵי הַדּוֹר הַהוּא יָגְעוּ לַחֲקֹר מְקוֹר הַשִּׁירִים. וְחָצְבוּ בוֹרוֹת נִשְׁבָּרִים. וּמֵימֵיהֶם מָרִים. וּמֵאַחַר אֲשֶׁר נֶאֱסַף שְׁלֹמֹה הַיּוֹשֵׁב עַל כִּסְאוֹ. וַיִּגְוַע וַיָּמָת אַבְרָהָם נְשִׂיאוֹ. וִיהוּדָה שַׂר צְבָאוֹ. וּמשֶׁה נְבִיאוֹ. נִסְתָּם חֲזוֹן הַשִּׁיר וּמוֹצָאוֹ. וְנִסְתַּלֵּק הַכָּבוֹד הַהוּא וּמוֹרָאוֹ. וְלֹא יָסַף עוֹד מַלְאַךְ ה' לְהֵרָאֹה. כִּי לֹא קָם אַחֲרֵיהֶם. מִי יְחַבֵּר כְּשִׁירֵיהֶם. וְהִנֵּה אֲנַחְנוּ אֵלֶּה פֹּה הַיּוֹם מְלַקְּטִים שִׁכְחַת שִׁבֳּלֵיהֶם. וּמְזַנְּבִים נֶחֶשְׁלֵיהֶם. וְיוֹם וָלַיְלָה נָרוּץ בְּמַעְגְּלֵיהֶם. וְנִיגַע וְלֹא נִגַּע אֲלֵיהֶם. כִּי כָל אֶחָד מִמֶּנּוּ יֶהְגֶּה רִיק. וְחֶרֶב סִכְלוּת יָרִיק. וְהָרִאשׁוֹנִים אָכְלוּ הַסֹּלֶת. וְעָזְבוּ לָנוּ הַפְּסֹלֶת. וְגַם אָמְנָה רָאִיתִי קְהִלּוֹת מִזְרָח. אֲשֶׁר שָׁם כְּבוֹד ה' זָרָח. וּקְהִלּוֹת צָרְפַת.

וְאַשְכְּנַז וְרִיפַת. וּקְהִלּוֹת פָּרַס. וּמֶשֶׁךְ וְתִירָס. וּקְהִלּוֹת יָוָן. הַשּׁוֹכְנִים בֵּין דִּישׁוֹן וְעַלְוָן. כִּי אֵין חָכְמָה אֲשֶׁר לֹא יְרֵשׁוּהָ. וְאֵין מִדָּה טוֹבָה אֲשֶׁר לֹא כְבָשׁוּהָ. רַק אַדְמַת הַשִּׁיר לְבַדָּהּ רָאוּהָ וְלֹא בָאוּהָ. וּמְלֶאכֶת חֲרוּזִים חָשְׁבוּ כִי לְמָדוּהָ. וְסָבְרוּ כִּי שְׂכָלוּהָ. וְהַשִּׁיר רָחוֹק מֵהֶם וְלָהֶם לֹא יִקְרָב. כִּרְחוֹק מִזְרָח מִמַּעֲרָב. וְנַפְשָׁם מְסוֹבֶבֶת סְבִיב אַרְצוֹ וּמוֹשָׁבוֹ. אַךְ לֹא תַעֲבֹר בּו. כִּי מִנֶּגֶד תִּרְאֶה אֶת הָאָרֶץ וְשָׁמָּה לֹא תָבוֹא.

וְאִם תִּשְׁאָלֵם יֹאמְרוּ כִּי אֵין מְשׁוֹרְרִים כָּהֵם. וְאֵין שִׁיר כְּשִׁירֵיהֶם. אֲבָל אֵין אִישֹ רוֹאֶה מוּמוֹ. וְאֵין אָדָם רוֹאֶה חוֹבָה לְעַצְמוֹ. כְּי יֵשׁ לַשִּׁיר שִׁבְעָה תְנָאִים וּבָהֶם יֶעֱרַב וְיִנְעַם. וְיִהְיוּ דְרָכָיו דַּרְכֵי נֹעַם. וְכָל מִי שֶׁלֹּא יִשְׁמֹר הַתְּנָאִים הָאֵלוּ יְהִי שִׁירוֹ לְמִרְמָס. וְיִדְמֶה לְשִׁכְבַת הַטַּל וְחַם הַשֶּׁמֶשׁ וְנָמָס.

הַתְּנַאי הָרִאשׁוֹן: צָרִיךְ לַמְשׁוֹרֵר לְבָרֵר מִכָּל סִיג דִּבְרֵי שִׁירוֹתָיו. וּלְהָסִיר כָּל מִלָּה זָרָה מִמִּלוֹתָיו. פֶּן יִדְמֶה לִמְשׁוֹרְרֵי יָוָן אֲשֶׁר יְמַלְּאוּ בְשִׁירָם מִלָּה זָרָה עִם מִלָּה יְקָרָה. וַיְחַבְּרוּ צִצִים. עִם קוֹצִים. וּפְנִינִים עִם אֲבָנִים. וְדָרִים. עִם דַּרְדַּרִים. וְעַל כֵּן כָּל שִׁירוֹתָם מְשֻׁנּוֹת. וְהַשָּׂפָה רָפָה וְהַלְּשׁוֹנוֹת שׁוֹנוֹת.

הַתְּנַאי הַשֵּׁנִי: צָרִיךְ לַמְשׁוֹרֵר לִשְׁמֹר מִשְׁקַל הַשִּׁיר וּתְנוּעוֹתָיו. וְאַל יוֹסִיף וְעַל1 יִגְרַע עַל מִשְׁקַל אִמְרוֹתָיו. פֶּן יִהְיֶה כִּמְשׁוֹרְרֵי דוֹרֵנוּ אֲשֶׁר מִשְׁקְלֵי שִׁירָם שְׁבוּרִים. פַּעַם חֲסֵרִים. וּפַעַם יְתֵרִים. וְלֹא יָשִׂימוּ לֵב לַמִּשְׁקָל. וְלֹא יֵדְעוּ אִם יִכְבַּד אוֹ יֵקַל.

הַתְּנַאי הַשְּׁלִישִׁי: רָאוּי לַמְשׁוֹרֵר לִהְיוֹת עִנְיְנֵי שִׁירוֹ חֲזָקִים. בְּמַעֲבֵה הַמְּלִיצוֹת יְצוּקִים. פֶּן יִדְמֶה לִמְשׁוֹרְרֵי אֶרֶץ שִׁנְעָר אֲשֶׁר עִנְיְנֵיהֶם צְנוּמִים דַּקִּים. וּמִמְּלִיצָה טוֹבָה רֵיקִים. וְאֵין בָּם אִישׁ יִדְלֶה מִשִּׂכְלוֹ מַיִם עֲמֻקִּים. כִּי הַחֶבֶל נִכְרַת וְהַדְּלִי נִקְרַע וְהַמַּיִם נִשְפְּכוּ וְהַכֵּלִים רֵיקִים.

הַתְּנַאי הָרְבִיעִי: צָרִיךְ לַמְשׁוֹרֵר לִהְיוֹת שִׁירָיו קַלִּים לְהָבִין. וְכֻלָּם נְכוֹחִים לַמֵּבִין. פֶּן יִדְמֶה לִמְשׁוֹרְרֵי צָרְפַת אֲשֶׁר כָּל שִׁירֵיהֶם מְלֵאִים שִׁבּוּשׁ. וּצְרִיכִים לְפֵרוּשׁ.

הַתְּנַאי הַחֲמִישִׁי: צָרִיךְ לַמְשׁוֹרֵר לִשְׁמֹר שִׁירוֹ מִטָּעוּת הַדִּקְדּוּק. פֶּן יִדְמֶה לִמְשׁוֹרְרֵי דַמֶּשֶׂק. וְכָל שֶׁכֵּן יִצְחָק בֶּן בָּרוּךְ הָרוֹפֵא אֲשֶׁר כָּל שִׁירָיו כְּחֶרֶשׂ נִשְׁבָּר. וּכְתֶבֶן בְּלִי בָּר. וְכָל מִלָּיו זָרִים. וּמֵימָיו מָרִים. וְעִנְיָנָיו מִשֶּׁלֶג קָרִים. וְגִבּוֹרָיו חֲלוּשִׁים. וַאֲנָשָׁיו נָשִׁים. וְקָדָשָׁיו קְדֵשִׁים. וּדְבָרָיו מְלֵאִים טָעֻיּוֹת. בְּמִלּוֹת וּבְאוֹתִיּוֹת. וְיִבְטַח בַּסְּכָלִים הַמְשַׁבְּחִים שִׁירָיו. וְהַמְרַפְּאִים עַל נְקַלָּה צִירָיו. וְהֵם לֹא יָדְעוּ אִם יְרַנֵּן. אוֹ יְקוֹנֵן. וְאִם הוּא מְשׁוֹרֵר. אוֹ חֲמוֹר נוֹעֵר.

הַתְּנַאי הַשִּׁשִּׁי: צָרִיךְ לַמְשׁוֹרֵר בְּהַשְׁלִימוֹ שִׁירָתוֹ לְבַל יְמַהֵר לְהַרְאוֹתָהּ. עַד יְבָרֵר אוֹתָהּ. וְאַל יַשְׁאִיר חֶלְאָתָהּ עָלֶיהָ. וְטֻמְאָתָהּ בְּשׁוּלֶיהָ. רַק יַעֲבִיר מִמֶּנָּה כָּל הֲתֻלֶיהָ. וְיָסִיר כָּל בְּדִילֶיהָ. פֶּן יִדְמֶה לַמְשׁוֹרְרִים הַסְּכָלִים אֲשֶׁר בְּטֶרֶם יַשְׁלִימוּ הַשִּׁירָה יְגַלּוּהָ לִפְנֵי קָהָל וְעֵדָה. וּמְזִמָּתָם בְּטֶרֶם תָּחִיל יָלְדָה.

הַתְּנַאי הַשְּׁבִיעִי: צָרִיךְ לַמְשׁוֹרֵר לִבְחֹר מִשִּׁירָתוֹ קְצָתָהּ. וְיַשְׁלִיךְ קְצָתָהּ. וְיִבְחַר לוֹ מִן הַחֲמִשִּׁים. שְׁלשִׁים. וּמִן הַשְּׁבְעִים. אַרְבָּעִים. וְאָז תִּהְיֶה שִירָתוֹ נְקִיָּה. כְּכַלָּה יְפֵה-פִיָּה. וּבַתְּנָאִים הָאֵלֶּה הַתִּפְאָרָה הַחָכְמָה וְהַדֵּעָה. חָצְבָה עַמוּדֶיהָ שִׁבְעָה. וּבְאֵלֶּה הַמִּדּוֹת הָיְתָה מְלֶאכֶת הַשִּׁיר לִבְנֵי סְפָרַד כְּבוּשָׁה. וַתְּהִי לָהֶם לְאִשָּׁה. וּמִבַּלְעֲדֵיהֶם גְּרוּשָׁה. וְהָיְתָה עִמָּם מְקֻדֶּשֶׁת וּקְדוֹשָׁה. וְעִם זוּלָתָם חֲלָלָה וּקְדֵשָׁה. וְדַע כִּי בְנֵי אִישׁ נֶחֱלָקִים. לִשְׁלשָׁה חֲלָקִים. כַּת כְּסִילִים. וְכַת מַשְׂכִּילִים. וְכַת מְשׁוֹרְרִים פְּלִילִים. וְעַל כֵּן צָרִיךְ לְכָל מְשׁוֹרֵר לַעֲשׂוֹת בְּשִׁירוֹ עִנְיָנִים פְּשׁוּטִים וְקַלִּים. כְּדֵי שֶׁיִּיטְבוּ בְּעֵינֵי הַכְּסִילִים. וְעִנְיָנִים חֲזָקִים וְתֹקֶף מִלִּים. כְּדֵי שֶׁיִּיטְבוּ בְּעֵינֵי הַמַשְׂכִּילִים.

וְהַמְּלָאכָה נְקִיָּה. וּמְלִיצָה יְפֵה-פִיָּה. כְּדֵי שֶׁתִּיטַב בְּעֵינֵי הַמְשׁוֹרְרִים. הַמַּכִּירִים מִבְחַר הַשִּׁירִים. וְעַל כֵּן תִּמְצָא שִׁירֵי הַמְשׁוֹרְרִים נֶחֱלָקִים. לְאֵלֶּה הַחֲלָקִים. כִּי יֶשׁ שִׁיר אֲשֶׁר בּוֹ הָעִנְיָן הָרִאשׁוֹן וְיִּיטַב בְּעֵינֵי הַכְּסִילִים. וְלֹא בְּעֵינֵי הַמַּשְׂכִּילִים. וְשִׁיר אֲשֶׁר בּוֹ הָעִנְיָן הַשֵּׁנִי יִיטַב בְּעֵינֵי הַמַּשְׂכִּילִים. וְלֹא בְּעֵינֵי הַכְּסִילִים. וְשִׁיר אֲשֶׁר בּוֹ הָעִנְיָן הַשְּׁלִישִׁי וְיִיטַב בְּעֵינֵי הַמְשׁוֹרְרִים. וְלֹא בְּעֵינֵי הָאֲחֵרִים. וְהַשִּׁיר הָאוֹסֵף שְׁלשֶׁת הָעִנְיָנִים הָאֵלֶּה אֵין לְמַעְלָה מִמֶּנּוּ. כְּשִׁירֵי רַבִּי יְהוּדָה הַלֵּוִי זַ"ל אֲשֶׁר בּוֹ שְׁלשֶׁת הָעִנְיָנִים כִּי תִּמְצָא שִׁירָיו פְּשׁוּטִים וַחֲלָקִים. וְרַכִּים וַחֲזָקִים. וַעֲרֵבִים וּמְתוּקִים. וְשִׁיר רַבִּי שְׁלֹמֹה הַקָּטָן יִיטַב בְּעֵינֵי הַמַּשְׂכִּילִים. וְלֹא בְּעֵינֵי הַכְּסִילִים. וְשִׁיר רַבִּי משֶׁה בֶּן עְזָרא יִיטַב בְּעֵינֵי הַמְשׁוֹרְרִים. יוֹתֵר מִן הָאֲחֵרִים. בַּעֲבוּר מְלִיצָתוֹ. וְנֹעַם טַעַם מְלַאכְתּוֹ. וְרָאִיתִי בִּגְלִילוֹת הַמַּעֲרָב אֲרָצוֹת סְמוּכוֹת מִשְנֵי צְדָדִים לִסְפָרַד.

הַצַּד הָאֶחָד הֵם גְּלִילוֹת פְּרוֹבִינְצָא וְלָהֶם שִׁירִים עֲמֻקִּים. וְעִנְיְנֵיהֶם טוֹבִים וַחֲזָקִים. אֲבָל אֵינָם כְּשִׁירֵי סְפָרַד עֲרֵבִים וּמְתוּקִים. וּמִן הַצַּד הַשֵּׁנִי אַרְצוֹת יִשְׁמְעֵאלִים. אֲשֶׁר בְּמַלְכוּת הַמֶּלֶךְ הַמּוּרַד יִמָּח. וּמַפַּלְתּוֹ מְהֵרָה תִצְמַח. וְשָׁם הָיוּ בְּטֶרֶם זְמַן הַשְּׁמַד חַכְמֵי עוֹלָם. בְּמִדּוֹת הַחָכְמָה כֻּלָּם. רַק מְלֶאכֶת הַשִּׁיר חֲלוּשָׁה וְרַקָּה. לֹא חֲזָקָה וְלֹא מְתוּקָה. וְלֹא רָאִיתִי בָהֶם שִׁירָה חֲזָקָה.

בְּמַעֲבֵה הַשֵּׂכֶל יְצוּקָה. כִּי אִם שִׁירֵי הֶחָכָם הָעֶלְיוֹן. רַבִּי יוֹסֵף בַּר יְהוּדָה בַּר שִׁמְעוֹן. כִּי מְלֶאכֶת שִׁירוֹ נְקִיָּה וַחֲזָקָה. וְעַל לֵב חֲקוּקָה. וְשִירוֹ לְבַדּוֹ קַב וְנָקִי. וְשִׁירֵי זוּלָתוֹ מָלְאוּ קִיא. וְלוֹ הַמַחְבֶּרֶת הַיְפֵה-פִיָּה. הַבְּנוּיָה לְתַלְפִיָּה. אֲשֶׁר תְּחִילָּתָהּ נְאֻם טוֹבִיָּה בֶּן צִדְקִיָּה. הֶרְאָה בָהּ לָשׁוֹן נִמְרָץ. וְעַל פִּיהָ חֲזוֹן הַשִּׁיר נִפְרָץ. וּלְפָנֶיהָ אִישׁ לְשׁוֹנוֹ לֹא יֶחְרָץ. לְמַעַן תֵּדְעוּ אֲשֶׁר יַפְלֶה ה' בֵּינוֹ. וּבֵין מְשׁוֹרְרֵי זְמַנּוֹ. וְחוּץ מִשִּׁירָיו לֹא רָאִיתִי בִּגְלִילוֹת מַעֲרָב. שִׁיר יִנְעַם וְיֶעֱרָב. אֲבָל שִׁיר זוּלָתוֹ בְּחַיֵּי בְעָלָיו יִצָּמֵת. כִּי אֵין בִּית אֲשֶׁר אֵין שָׁם מֵת. וְרָאִיתִי מִבְּנֵי צָרְפַת חֲכָמִים. מְאִירִים כְּכוֹכְבֵי מְרוֹמִים. וְלִבָּם בְּכָל חָכְמָה כַּיָּם הַגָּדוֹל. וְלֹא הִנִּיחוּ לִלְמֹד כָּל דָּבָר קָטָן אוֹ גָדוֹל. אֲבָל הַשִּׁיר אֲשֶׁר יַחְבִּירוּ. הִקְשַׁבְתִּי וָאֶשְׁמַע לֹא כֵן יְדַבֵּרוּ. כִּי עִנְיְנֵיהֶם בְּזוּיִם. וְאֵינָם לִשְׁמֹעַ רְאוּיִם. וְשִׁירֵיהֶם קָשִׁים כַּבַּרְזֶל. עֲמוּסִים מֵעֲוֹנוֹת כִּשְׂעִיר עֲזָאזֵל. וַחֲרוּזֵיהֶם מְלֵאִים שִׁבּוּשׁ. וְהֵם מַעֲמִיקִים בְּשִׁירָם לַעֲשׂוֹת חִדּוּשׁ. וְלֹא יְבִינֵם אָדָם וְדַעַת וּמְזִימָה. וְישֶׁר וְתֻמָּה. וְכָל מִדָּה נְעִימָה. אֲבָל שִׁירוֹתֵיהֶם מְאוּסוֹת. כְּבֵדוֹת כְּאֶבֶן מַעֲמָסוֹת. כְּעֵץ הָאֶשְׁכּוֹל אֲשֶׁר לֹא יִצְלַח לַכֹּל. אוֹ כְּאִילָן סוֹרֵק בְּלִי פּוֹרֵחַ. וְלֹא טַעַם וְלֹא רֵיחַ. וְלִפְעָמִים יִמָּלֵט אֶחָד מֵהֶם אֲשֶׁר הָאֵל שִׂכְלוֹ יָעִיר. כְּגוֹן שִׁירֵי מִיכָאֵל בֶּן כָּלֵב מֵעִיר תֵּבֵץ. כִּי שִׁירָיו יֵשׁ קְצָת נֹעַם עֲלֵיהֶם. בַּעֲבוּר כִּי הָלַךְ מִנְּעוּרָיו לִסְפָרַד וְלָמַד מֵהֶם. וְכֵן רַבִּי משֶׁה בֶּן אַבְטַלְיוֹן. שִׁירוֹ בֵּין אַנְשֵׁי דוֹרוֹ עֶלְיוֹן. וּבְעָבְרִי לְאַרְצוֹת מִזְרָח רָאִיתִי שָׁם קְהִלוֹת. עוֹלִים צֹאן אוֹבְדוֹת. רֵיקִים מִתְּבוּנוֹת וּתְעוּדוֹת. וְיַזְרִיעוּ בְשִׁירֵיהֶם יְלָדִים. אֲשֶׁר הֵם מְאוּסִים כְּבֶגֶד עִדִּים. וַתֵּלַדְןָ הַצֹּאן עֲקֻדִּים נְקֻדִּים וּבְרֻדִּים. וּבְעִיר נֹא אָמוֹן כִּי קְצָת שִׁירָיו בְּצֵל הַשֵׂכֶל חָסוּ. וְכַנֵּס עַל הָהָר הִתְנוֹסְסוּ. וְהַנִּשְׁאָרִים הֶרָה נָסוּ. גַּם בְּדַמֶּשֶׂק רָאִיתִי אֲנָשִים נִכְבָּדִים. וּבַעֲלֵי מִדּוֹת טוֹבוֹת וַחֲסָדִים. וְלָהֶם פְּעָלִים נֶחְמָדִים. רַק הַשִּׁירִים אֲשֶׁר יְחַבְּרוּ מְלֵאִים חֶלְאָה. וְרוּחַ טְמֵאָה וְנִגְאָלָה וּמוֹרָאָה.

וְאִלּוּ יַשְׁלִיכוּ אוֹתָם לֶאֱכֹל לַחֲזִירִים. הָיוּ בְעֵינֵיהֶם אֲסוּרִים. וּמִבְחַר מְשׁוֹרְרֵיהֶם יִצְחָק בֶּן בָּרוּךְ הָרוֹפֵא וְכָל שִׁירָיו פּוֹחֲזִים וְרֵיקִים. כִּי בְקָקוּם בּוֹקְקִים. אֲבָל הֵם עֲרֵבִים לִסְכָלִים כִּצְרוֹר הַמּוֹר. וְיִיטְבוּ דִבְרֵיהֶם בְּעֵינֵי חֲמוֹר. וְיֵשׁ בָּהֶם מְשׁוֹרְרִים יְגַנְּבוּ שִׁיר רַבִּי יְהוּדָה וְרַבִּי שְׁלֹמֹה. וִישַׁנּוּ אֶת טַעְמוֹ. וְיַשְׁחִיתוּ מִשְׁקְלֵיהֶם. וְיוֹסִיפוּ מִלּוֹת זָרוֹת בָּהֶם. וְיַפְשִׁיטוּ בִגְדֵי רִקְמָתָם מֵהֶם. וְיָשִׂימוּ הַבְּגָדִים הַצּוֹאִים עֲלֵיהֶם. וְיַחְבִּירוּ עִם בְּשָׂמִים רֹאשׁ. לַעֲנָה וָרוֹשׁ. וְעִם בְּדֹלַח וָדָר. קוֹץ וְדַרְדָּר. וִינַגְּנוּ אוֹתָם בְּבֵית מִשְׁתֵּיהֶם. וְיֹאמְרוּ לַסְכָלִים כִּי הֵם שִׁירֵיהֶם. וּבִמְשׁוֹרְרֵי דַמֶּשֶׂק אִישֹ מִצְרִי יְחַבֵּר שִׁירִים אֲסוּרִים לִשְׁמֹעַ. וְהַשּׁוֹמֵעַ אוֹתָם חַיָּב לִקְרֹעַ. כַּתְּאֵנִים הַשּׁוֹעָרִים אֲשֶׁר לֹא תֵאָכַלְנָה מֵרֹעַ. לֹא רָאִיתִי כָהֵנָּה בְּכָל אֶרֶץ מִצְרַיִם לָרֹעַ. וְהוּא חוֹשֵׁב בְּגַאֲוָתוֹ וּגְאוֹנוֹ. כִּי שְׁלֹמֹה הַקָּטָן קָטןֹ מֵהָכִיל הֶגְיוֹנוֹ. וְאַבְרָם כָּבֵד מְאֹד בְּשִׁירָיו לְנֶגֶד צַחוּת לְשׁוֹנוֹ.

וּמשֶׁה הָיָה רוֹעֶה צֹאנוֹ. וִיהוּדָה יְהוֹדֶה לְשֵׂכֶל רַעְיוֹנָיו. וְכָל הַמְשׁוֹרְרִים הָעֲנָקִים הֵם כַּחֲגָבִים בְּעֵינָיו. וְהוּא הָאוֹמֵר בְּשִׁירָתוֹ. לְגַלּוֹת בָּשְׁתּוֹ. “וְשִׁירָתִי כְמוֹ שִׁירַת סְפָרַדִּי וְכָל דָּבָר בְּעִתּוֹ”ֹ. וּמִי יִתֵּן וְתָשׁוּב לִסְפָרַדִּי נִשְׁמָתוֹ. וּשְלֹמֹה יֵשֵׁב עַל כִּסֵּא מַלְכוּתוֹ. עַד יִרְאֶה טֶנֶף שִׁירָתוֹ. וְרוּחַ טֻמְאָתוֹ. כִּי הָיָה מֹחוֹ מְרַצֵּץ. וְרֹאשׁוֹ מְפוֹצֵץ. וּזְנָבוֹ מְקַצֵּץ. וּבְרַגְלוֹ מְשָׁכָהוּ. וְעַל פָנָיו הֲפָכָהוּ. וּבְבֵית הַטִּנֹּפֶת הִשְׁלִיכָהוּ. גַּם רָאִיתִי בִמְשׁוֹרְרֵי דַמֶּשֶׂק אִישׁ עִבְרִי. מַכֶּה אִישׁ מִצְרִי. וְהוּא בֶן בָּרוּךְ הָרוֹפֵא יִצְחָק. אֲשֶׁר יָרַד הַשִּׁיר עִמּוֹ לְאֶרֶץ מֶרְחָק. וַיִּפֹּל עַל פָנָיו וַיִּשְׂחָק. וַיֹּאמֶר שְׂחוֹק עָשָׂה לִי אֱלֹהִים וְכָל הַשּׁוֹמֵעַ לִי יִצְחָק. כִּי נְעָרִים נִתְעָבִים יִצְחֲקוּ בי. וְתַעֲלוּלִים יִתְעַלְּלוּ בִי. וְזֶה יִצְחָק הַנִּזְכָּר הוּא הַמְחָרֵף שִׁירֵי הַמִּצְרִי.

וַיַּעַן וַיֹּאמַר:

פְּנֵה אֵלַי פְּאֵר רוֹזְנִים וְשָׂרִים

וְאַל תֵּפֶן אֱלֵי שִׁירוֹת אֲחֵרִים.

חֲצֹב קֶבֶר לְשִׁירוֹתָם וְטָמְנֵם

וְאִם רַבּוּ חֲצֹב לָהֶם קְבָרִים.


וַאֲנַי תָמֵהַּ אֵיךְ יַחְצֹב הַנִּקְבָּר קֶבֶר לַאֲחֵרִים. וְאֵין חָבוּשׁ מַתִּיר עַצְמוֹ מִבֵּית הָאֲסוּרִים. וְצָרִיךְ לוֹמַר לוֹ: הֶעְתַּרְתָּ עַל אָחִיךָ דְבָרִים. וְרָאִיתָ מוּמוֹ וּמוּמֶיךָ מִמְּךָ נִסְתָּרִים. וְעָלֶיךָ אָמְרוּ חַכְמֵי יִשְׂרָאֵל הַיְשָׁרִים. שֶׁלְךָ אִי אַתָּה רוֹאֶה אֲבָל אַתָּה רוֹאֶה שֶׁל אֲחֵרִים. וּבְעוֹד תַּחְצֹב אַתָּה לְשִׁיר הַמִּצְּרִי קְבָרִים. הֲלֹא שִׁירְךָ מִצְרַיִם מְקַבְּרִים. כִּי כֻלּוֹ כְדֹמֶן בְּאַשְׁפָּה. לֹא לְעֵזֶר וְלֹא לְהוֹעִיל כִּי לְבשֶׁת וְגַם לְחֶרְפָּה. וְרָאִיתִי בְנֵי צוֹבָה. יִזְכְּרֵם הָאֵל לְטוֹבָה. כִּי אָסְפוּ כָּל מִדָּה טוֹבָה. אֲבָל הַשִּׁיר אֲשֶׁר יְחַבְּרוּ נָפַל בְּיָדָם כְּצִפּוֹר בְּמִכְמָר. וְטָבַע בְּתוֹךְ בֶּאֱרוֹת חֵמָר. וּבְעֵת יְנַגְּנוּהוּ. הוּא יְקוֹנֵן: הִנֵּה לְשָׁלוֹם מַר לִי מָר. וְשָׁם הָיָה בַתְּחִלָּה מְשׁוֹרֵר וּשְׁמוֹ רַבִּי יוֹסֵף בַּר צֶמַח. וְהָיוּ בוֹ מִדּוֹת טוֹבוֹת אֲבָל שִׁירוֹ צֶמַח. בְּלִי יַעֲשֶׂה קֶמַח. וּמִמְּשׁוֹרְרֵיהֶם הָיָה רַבִּי שֵׁלָה. לָקַח מְלֶאכֶת הַשִּׁיר לְכַלָּה. וְכַל יָמָיו הָיְתָה עִמּוֹ בְּתוּלָה וְלֹא בְּעוּלָה. וְהִיא צוֹעֶקֶת בְּנֶפֶשׁ אֲנוּשָׁה. כִּי רָאֲתָה כִּי גָּדַל שֵׁלָה וְהִיא לֹא נִתְנָה לוֹ לְאִשָּׁה. וְאַחֲרָיו הֶחִזיק יַחְיָא בֶּן אֶלְקָצִיר הוּא חַזַּן עֲרָב. וְאֶל מִשְׁכַּן ה' יִקְּרַב. וְהוּא בַעַל מִדּוֹת חֲשׁוּקוֹת. וּמִפְעָלוֹת מְתֻקָּנוֹת. אֲבָל שִׁירוֹ כִּשְׁאָר מְשׁוֹרְרֵי דוֹרוֹ לֹא סָר מִדַּרְכָּם. וְלֹא נָטָה מִמַּהְלָכָם. וְיֵשׁ מְשׁוֹרְרִים אֲחֵרִים כָּמוֹהוּ. וְכָל שִׁירֵיהֶם רוּחַ וָתֹהוּ. מֵהֶם חֲדָשִׁים וּמֵהֶם יְשָׁנִים. מֵהֶם חַזָּנִים וּמֵהֶם דַיָּנִים. וְאֶפֶס וָתֹהוּ שִׁירֵיהֶם. וְיֵשׁ מֵהֶם יִתְפָּאֲרוּ בְשִׁירֵי בֶּן עַבָּאס וְיֹאמְרוּ כִּי לֹא קָם אַחֲרָיו מִי יְחַבֵּר שִׁיר. וְאִלּוּ יָסִיר הָאֱלֹהִים אֶת לֵב הָאֶבֶן מֵהֶם. וְנִפְקְחוּ עֵינֵי רַעְיוֹנֵיהֶם. לְהַכִּיר הַשִּיר הָעֶלְיוֹן. הַמַּבְהִיל כָּל רַעְיוֹן. הִכִּירוּ כִּי לֹא הִכִּירוּ הָאֱמֶת. וְיָדְעוּ כִּי לֹא יָדְעוּ לְהַבְדִּיל בֵּין הַחַי וּבֵין הַמֵּת. וְרָאִיתִי בְעִיר כַּלְנֹה הִיא אַלָרְקָה אִישׁ עוֹלֶה בְּמַעֲלֵה הַשִּׁיר הָעֶלְיוֹנָה. וְהוּא מִבַּעֲלֵי הַמִּשְׁמָרֶת הָרִאשׁוֹנָה. וּפִיהוּ בִמְלֶאכֶת הַשִּׁיר פִוֹעֵר. וְהוּא חֲמוֹר נוֹעֵר. וְלֹא יַכִּיר מִן הַחָכְמָה וְלֹא יָבֵן. כִּי אִם הַמִּסְפּוֹא וְהַתֶּבֶן. וְלֹא לָמַד מִן הַתְּבוּנָה וְהַדַּעַת. רַק לָשֵׂאת הָאֻכָּף וְהַמִּרְדַּעַת. וְשִׁירוֹ טוֹב בְּעֵינֵי עֲדָתוֹ. וְנֶחְמָד לְפִי שֵׂכֶל קְהִלָּתוֹ. כִּי הַבְּהֵמָה וְהַכֵּלִים. כְּרַגְלֵי הַבְּעָלִים.

וּשְׁמוֹ בְּרָכוֹת בֶּן יְשׁוּעָה. וְהָיָה רַק זְוָעָה הָבִין שְׁמוּעָה.

וְעָלָיו אָמַרְתִּי:

קְלָלוֹת הוּא מְחַבֵּר שִׁיר כְּצוֹאָה

וְהַצּוֹאָה בְּפִיו יַחְשֹׁב כְּנֹפֶת.

וְהֶגְיוֹנוֹ יְגַלֶּה אֶת קְלוֹנוֹ

וְעֶרְוָתוֹ בְשִׁירוֹ נֶחֱשֶׂפֶת.

וְאָבִיו הוּא מְשֻׁעְבָּד לְנַבְלוּת

וְאִמּוֹ לַחֲרָפוֹת נֶחֱרֶפֶת.

וְנֶאֶסְפוּ לְעוֹלָמָם שְׁנֵיהֶם

וְחֶרְפָּתָם לְאוֹת לֹא נֶאֱסֶפֶת.

וְהַבֵּן עֵד עֲלֵי טֻמְאַת אֲבוֹתָיו

כְּרֵיחַ רַע עֲלֵי בֵית הַטְּנֹפֶת.


וּבִנְבִיאֵי אַרְבֵּל רָאִיתִי שַׁעֲרוּרָה. כִּי אִישׁ מִמְּשׁוֹרְרֵיהֶם לָקַח יְצִיאוֹת הָעֲנָק אֲשֶׁר לְרַבִּי משֶׁה בֶּן עֶזְרָא. וַיְחַבֵּר עַל כָּל יְצִיאָה מִיצִיאוֹת חֲרוּזָיו שִׁירָה עֲרוּכָה. כְּדֹמֶן בְּאַשְׁפָּה דְרוּכָה. וַיָּצַר אותֹו בַּחֶרֶט וַיַּעֲשֵׂהוּ עֵגֶל מַסֵּכָה. וְנִקְרָא שְׁמוֹ בֶּן הַשֶּכְוִי. וּמֵאַיִן תָּבוֹא לְעוֹפוֹת הַשָּׂדֶה תְבוּנָה. אוֹ מִי יִתֵּן לַשֶׂכְוִי בִינָה. וּמְשׁוֹרְרֵי מִזְרָח הַקַּדְמוֹנִים הָרַב שְׁמַרְיָה. וְרֹאשׁ הַיְשִׁיבָה זְכַרְיָה. הָיוּ שְׁנֵיהֶם בַּתַּלְמוּד גִּבּוֹרִים וְנִכְבָּדִים. וְיִרְאֵי שָׁמַיִם וַחֲסִידִים. וְאִלּוּ שִׁיר לֹא הִזְכִּירוּ. הִשְׂכִּילוּ וְדִבָּתָם הִסְתִּירוּ. אֲבָל לָקְחוּ שִׂמְלַת הַשִּׁיר בֵּינֵיהֶם. וַיָּשִׂימוּ עַל שֶׁכֶם שְׁנֵיהֶם.

וַיֵּלְכוּ אֲחֹרַנִּית וְעֶרְוַת שִׁירֵיהֶם לֹא רָאוּ וּפָשְטוּ סוּת הַמּוּסַר אֲשֶׁר לָבְשׁוּ. וַיִּהְיוּ שְׁנֵיהֶם עֲרֻמִּים וְלֹא הִתְבּוֹשְׁשׁוּ. וְרָאִיתִי מְשׁוֹרְרֵי בָבֶל. אֲשֶׁר שָׁם הָיוּ גְאוֹנֵי תֵבֵל. כִּי כְפִי עֲלוֹתָם בַּתְּעוּדוֹת. כֵּן שִׁפְלוּתָם בַּשִּירוֹת הַכְּבֵדוֹת. וְשִׁירֵיהֶם יְקוֹנְנוּ: אוֹי לָנוּ כִּי נִדְמִינוּ. וְכֻלָּנוּ כַצֹאן תָּעִינוּ. וְעֶל נַהֲרוֹת בָּבֶל שָׁם יָשַׁבְנוּ גַּם בָּכִינוּ. וְעַל עֳרָבִים בְּתוֹכָהּ כִּנּוֹרוֹתֵינוּ תָּלִינוּ. וּמִבְחַר מְשׁוֹרְרֵיהֶם רַבִּי יִצְחָק בַּר יִשְׂרָאֵל רֹאשׁ הַיְשִיבָה קְצָת שִׁירָיו חֲזָקִים וְטוֹבִים. וְרֻבָּם כְּחֶרֶשׂ נִשְׁבָּר חֲשׁוּבִים. לֹא מְתוּקִים וְלֹא עֲרֵבִים. גַּם הוּא חִבֵּר סֵפֶר מַחְבָּרוֹת. וּבוֹ שִׁירוֹת וְאִגְּרוֹת. חֲלוּשׁוֹת וְקָרוֹת. כְּאִלּוּ מִשֶּׁלֶג שְׂנִיר גְּזוּרוֹת. וְכָל עִנְיָנָיו גְּרוּעִים. וַיָּרֻם תּוֹלָעִים. וְאִלּוּ הַחֲרֵשׁ יַחֲרִישׁ. רִמָּה לֹא הָיְתָה בוֹ וְלֹא הִבְאִישׁ. וְאַחֳרָיו הֶחֱזִיק רַבִּי יִצְחָק בֶּן אֶלְאוֹנִי. הוּא מְאֹד עָשִׁיר אֲבָל שִׁירוֹ דַּל וְעָנִי. וְהוּא קָנָה בְאֶלֶף זְהוּבִים הַיְשִׁיבָה וְזוּלָתוֹ יִרְכְּכֶנָּה. אִשָּׁה יְאָרֵשׂ וְאִישׁ אַחֵר יִשְׁכָּבֶנָּה. וְהוּא מְחַבֵּר שִׁירִים מְנֻפָּצִים בְּשַׁחַת בְּלִי. וְנִשְׁחַת הַכְּלִי. וְאִם יִשְׁאַל שׁוֹאֵל אֵי זֶה הַשִּׁיר הַנִּמְאָס לַמַּשְׁכִּילִים. וַיֹּאמֶר ה' הִנֵּה הוּא נֶחְבָּא אֶל הַכֵּלִים. וְכָל מְשׁוֹרְרֵי עֲדִינָה הוֹזִים נִדְהָמִים. בְּחֵיק הַסִּכְלוּת נִרְדָּמִים. וּמֵהֶם נְעָרִים וַעֲלָמִים. פָרְצוּ בְשִׁירָם דִּבְרֵי חַכָמִים. שׁוּעָלִים קְטָנִּים מְחַבְּלִים כְּרָמִים. וְשִׂמְלַת שִׁירָם מְגוֹלְלָה בְדָמִים. וְרַבִּי משֶׁה בַר שֵׁשֶׁת הַהוֹלֵךְ שָׁמָּה מֵאַרְצֵנוּ הִדְרִיךְ בְּנֵי עֲדִינָה בְדֶרֶךְ הַשִּׁיר. אוּלַי יְרַפֵּא מֵהֶם מַכְאוֹב וְכָל צִיר. וְלָהֶם אֲרוּכָה לֹא עָלְתָה. וְרִפֵּאנוּ אֶת בָּבֶל וְלֹא נִרְפְּתָה.

וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֱמֶת כִּי כָל בְּנֵי מִזְרָח גְּבוֹהִים

וְחַכְמֵי הַזְּמַן נֶגְדָּם בְּמוֹרָד.

וְכָל-חַכְמֵי בְנֵי צָרְפַת וְיָוָן

לְבָבָם לַתְּעוּדוֹת רָץ וְחָרָד.

וְכָל חָכְמָה וּבִין לָקְחוּ לְנַפְשָׁם

וְהַשִּׁיר עָזְבוּ לִבְנֵי סְפָרָד.

אָמַר הַמַּגִּיד: וּבְשָׁמְעִי פְלִיאוֹת אִמְרוֹתָיו. וְנֹעַם מִלּוֹתָיו. שְׁאִלְתִּיו עַל שְׁמוֹ וְאֶרֶץ מוֹלַדְתּוֹ.

וַיַּעַן וַיֹּאמַר:

אֲנִי חָבֵר לְכָל מִי יִדְרְשֵׁנִי

וְאֵלוֹן צַעֲנַנִּים עִיר מְעוֹנִי.

וּבִקְרָב בִּין שְׁבָחַי הֵם שְׁלָחַי

וְשַׂמְתִּי אֶת לְשׁוֹנִי קִלְּשׁוֹנִי.


וְכַאֲשֶׁר שָׁמַעְתִּי נְאֻמוֹ. הִכַּרְתִּי שְׁמוֹ. וּבָאתִי לִשְאֹל בִּשְלוֹמוֹ. אֲבָל הָיָה נֶחְפָּז לָלֶכֶת. וְנִפְרַד מֵאִתִּי וְנַפְשִׁי בַּעֲבוֹת אַהֲבָתוֹ נִמְשֶׁכֶת.
