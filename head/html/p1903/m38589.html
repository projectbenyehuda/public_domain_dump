משל קיט: הַקַּנַּאי וְהַחֶמְדָּן

הַחֶמְדָּן וְהַמְקַנֵּא יַחְדָּו יִלְבְּשׁוּ בֹשֶׁת / וְאַחֲרֵיהֶם נֶזֶק עַל כָּל אֶחָד יָשִׁית

שִׁמְעוּ נָא דִּבְרֵי קְרִשְְׂפִּיָא הַנַּקְדָּן / אֲשֶׁר שָׁפַט וָדָן / לֵב קַנַּאי וְחֶמְדָּן / לְחֶרֶב הֶרֶג וְאַבְדָן / וּלְמַשְׁחִית אֵיבַת עוֹלָם / וּלְהָשִׁיב דַּךְ וְנִכְלָם / אֲשֶׁר אַפּוֹ חָרָה / עַל הַמִּשְׁפָּט וְעַל הַגְּבוּרָה / וְעַל דֶּרֶךְ הַגְּזֵרָה / לְעַוֵּת מִשְׁקָל וּמְשׂוּרָה / כַּאֲשֶׁר לָאָדָם הִקְרָה / וּלְאַנְשֵׁי זָדוֹן וְעֶבְרָה / אֶחָד חֶמְדָּן וְאֶחָד מְקַנֵּא / אִישׁ אָחִיו יְשַׂנֵּא / וּמְדַבְּרִים עַל יוֹצְרָם / הַיֹּדֵעַ רֹעַ יִצְרָם / הַרְבֵּה מִתְלֹנְנִים / וּבְרָעָה מִתְאֹנְנִים / חֶמְדָּן כֹּה יֹאמַר / רְאֵה כִּי רַע וָמַר / כָּל מַעֲשֵׂה הָאֱלֹהִים / הִשְׁפִּיל אֶת הַגְּבֹהִים / מַדּוּעַ אֲנִי רָשׁ וְעָנִי / וְיַעֲשִׁיר אִישׁ פְּלוֹנִי / אֹיְבִי שְׁכֵנִי / הַיֹּשֵׁב לִימִינִי / וְחֹנֶה עַל דִּגְלוֹ / לָמָּה יָרִים אֶת רַגְלוֹ / עַל סוּס מֵהַשְּׁנַיִם / וּלְאֹיְבִי פִּי שְׁנַיִם / וְהַפֵּל אַחַת מֵהַיָּדַיִם / וּלְשֹׂנְאִי זֶה תֵּן כִּפְלַיִם / וַיְהִי כְדַבְּרוֹ כָּכָה / וְהִנֵּה אֵימָה חֲשֵׁכָה / נָפְלָה עֲלֵיהֶם / וַתַּךְ אֶת שְׁנֵיהֶם / בְּעִוָּרוֹן / וַיְהִי לָאֲחֲרוֹן / מִכֹּבֶד רֵעֵהוּ הַיִּתְרוֹן / לִהְיוֹת לוֹ הַכִּפְלַיִם מֵחֲבֵרוֹ / וְלֹא נָפַל מִכָּל דְּבָרוֹ / וַיִּפֶן לְרֵעֵהוּ פָנָיו / וְהִנֵּה חָשְׁכוּ שְׁתֵּי עֵינָיו / וּבְכַנְפֵי בְגָדָיו / נִשְׁאֲרוּ שְׁתֵּי יָדָיו / וַיָּסַר כֹּחוֹ מֵעָלָיו / וְהִנֵּה קִטֵּעַ מִשְּׁתֵי רַגְלָיו / וַיִּשָּׁאֲרוּ שְׁנֵיהֶם שָׁמָּה / לְבֹשֶׁת וְלִכְלִמָּה / וַיָּסַר מֵהֶם אֶת חֶמְדָּתָם / וְגַם אֶת שִׂנְאָתָם / כִּי הַחֹמֵד נָח מֵחֶמְדָּתוֹ / בָּחַר מִחְיָתוֹ / וּבְהַוָּתוֹ לֹא יָעוֹד / וְלֹא יַחְמֹד עוֹד / הוֹן בֵּית גָּבֶר / כִּי אִם בֵּית הַקָּבֶר / כִּי עוֹד לֹא יָכֹל / לָשִׂים בְּפִיו לֶאֱכֹל / כִּי אִם לִלְחֹךְ בְּמָרָק / כִּלְחֹךְ הַשּׁוֹר אֶת יָרָק / כִּי בְחֶסְרוֹן הָעֵינַיִם / יְהִי לוֹ נִקְיוֹן שִׁנַּיִם / לֹא יִרְאֶה טוּב טַעְמוֹ / וְלֹא יָמִישׁ מִמְּקוֹמוֹ / מַאֲכָלוֹ וּמוֹצָאוֹ יַחַד / יִהְיוּ לְמָקוֹם אֶחָד / כִּי נֶחְשַׁב כַּבְּהֵמָה / טוֹב לוֹ מָוֶת כַּמְּהוּמָה / וְגַם הַמְקַנֵּא / אֶת אָחִיו שֹׂנֵא / לֹא יְקַנֵּא עוֹד בַּאֲחֵרִים / כִּי יוֹצֵא בְרָאשֵׁי אֵבָרִים / אֲשֶׁר אָבַד בְּעַזּוּתוֹ / וַאֲשֶׁר הִפְסִיד בְּקִנְאָתוֹ / וְיֻכַּת עַד הַחָרְמָה / וְעַל זֶה מָשְׁלָה חָכְמָה / וְדִבְּרָה עַל הַמְקַנֵּא / בִּשְׂפָתָיו יִנָּכֵר שׂוֹנֵא / וּבְקִרְבּוֹ יָשִׁית מִרְמָה / וְיִשְׁפֹּךְ עָלָיו חֵמָה / אַךְ הַמִּשְׁפָּט כַמִּדָּה / וְאָז הָיָה אִישׁ הַחֶמְדָּה / הַחֶמְדָּה סוֹף עֲשֶׂרֶת הַדְּבָרִים / וְהַקִּנְאָה סוֹף שִׁיר הַשִּׁירִים / שֶׁהִיא קָשָׁה כִשְׁאוֹל / וְאֵין לָהּ גּוֹאֵל לִגְאוֹל / וְאֵין בָּהּ טוֹב מְאוּמָה / כִּי אִם בְּקִנְאַת הַחָכְמָה / לְקַנֵּא בְמַעֲשֵׂה הַיְרֵאִים / וְלֹא בְמַעֲשֵׂה הַחַטָּאִים / וְזוּלַת אִוֶּלֶת טוֹבָה אֵינָה / הַחֶמְדָּה יוֹצְאָה מִמֶּנָּה / כִּי הַקִּנְאָה רִאשׁוֹנָה / וְאַחֲרִיתָהּ מָרָה כַלַּעֲנָה / וּמָרָה תִהְיֶה הַחֶמְדָּה שֶׁבָּאָה / כִּי אִם הַמַּעֲשֶׂה מְבִיאָהּ / וְהַכֹּל דָוָה אַחַת / וְהַצָּרַעַת פֹּרַחַת / רַבָּה וְהֹלֶכֶת / עַד אֲשֶׁר מוֹלֶכֶת / וְלֵב הָאָדָם מוֹשֶׁכֶת / אֶל מִבֵּית לַפָּרֹכֶת / כִּי בַסּוֹף תַּשְׁלִיכֶנּוּ / עַל פְּנֵי הַשָּׂדֶה תַגְעִילֶנּוּ / בְּקַנְּאוֹ מָמוֹן אֲחֵרִים / יִכְרוּ עָלָיו רֵעִים / וְזֶה מִשְׁפַּט הַקַּנַּאי / הֲלֹא מְשַׂנְאַי / אֲשֶׁר הַשֵּׁם יִשְׂנָאֵהוּ / הַנַּקְדָן קְרִשְׂפִּיָּהוּ / מְבַקֵּשׁ מֵאֵת יוֹצְרֵהוּ / אֲשֶׁר מִקִּנְאָה יִשְׁמְרֵהוּ / וּמֵחֶמְדָּה יַצִּילֵהוּ / וְזֶה הַכֹּתֵב מַעֲשֵׂהוּ / שֶׁל קְרִשְׂפִּיָּהוּ / הַשֵּׁם יְסִירֵהוּ / מֵחֶמְדָּה וּמִקִּנְאָה / מִשְּׁאוֹל וּמִשִּׂנְאָה / וְכָל דָּבָר רָע / אֵלָיו לֹא יֶאֱרַע / וּמֵחֲמֹד הוֹן אֲחֵרִים / בִּבְנֵי הַנְּעוּרִים / וְגַם בִּימֵי הַזְּקֵנִים / מִלְּקַנֵּא בַשְּׁכֵנִים / בְּמָמוֹנָם וּבְקִנְיָנָם / רַק בְּחָכְמָתָם וּבְעִנְיָנָם / וּבְמַעֲשֵׂיהֶם הַטּוֹבִים / אֲשֶׁר לָעַד נִצָּבִים / לִשְׁמֹר לַעֲשׂוֹת כָּהֶם / כְּמוֹהֶם יִהְיוּ עֹשֵׂיהֶם / וְזֹאת הַקִּנְאָה / מִכָּל טוּב מְלֵאָה / וְכֵן דִּבְּרוּ הַנְּבִיאִים / אַל יְקַנֵּא לִבְּךָ בַּחַטָּאִים / כִּי אִם בְּיִרְאַת אֲדֹנָי / וּבְדִבְרֵי הַר סִינַי / וּבְקִנְאַת הַנְּקָמָה / נֹתֵן שָׁלוֹם בִּמְרוֹמָה / וּבְשֶׂה תְמִימָה / בִּבְרִית עוֹלָם חֲתוּמָה / הַצּוּר אֲשֶׁר שְׁמוֹ לְעֵילוֹם / הוּא יְבָרֵךְ אֶת עַמּוֹ יִשְׂרָאֵל בַּשָּׁלוֹם.

אָמֵן

וְשָׁלְמוּ מְשָׁלִים, תְּהִלָּה לְמֹשֵׁל, / לְבֶרֶכְיָהוּ מְשָׁלָיו מְמַשֵּׁל

תְּבוּנוֹת מְאַסֵּף כְּזָהָב וְכֶסֶף / פְּתַיּוּת יְשַׁסֵּף וְאֹהְבֵם יְנַשֵּׁל

מְשָׁלוֹ עֲנָוִים מְעוֹדֵד וְגֵאִים / מְנוֹדֵד וְרָאשֵׁי גְאוֹנִים יְחַשֵּׁל

אֲמָרוֹת טְהוֹרוֹת יְלַקֵּט כְּאוֹרוֹת / וְנוֹתְנָם בְּסִירוֹת נְכוֹנִים לְבַשֵּׁל

כְּרָאשֵׁי בְשָׂמִים חֲשָׁבוּם חֲכָמִים / וְאִם הֵם רְתָמִים לְכָל לֵץ וְכֹשֵׁל.


סֻיְּמוּ

הַמְּשָׁלִים כֻּלָּם

שֶׁבַח

לְמֶלֶךְ עַל כָּל הָעוֹלָם.
