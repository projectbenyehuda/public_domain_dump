משל סט: זְאֵב וְשֻׁעָל וְיוֹנָה

חֲנֵפִים וְאַנְשֵׁי תֹעָה / חֶבְרָתָם תָּמִיד לְרָעָה


זְאֵב וְשֻׁעָל וְיוֹנָה / בָּאוּ בִּבְִרִית וּבֶאֱמוּנָה / לִטְרֹף טֶרֶף לְמַאֲכָלָם / וּלְחַלֵּק בְּשָׁוֶה אֶת שְׁלָלָם / וַיָּבֹאוּ בְמוֹ אֹרֵב / כִּי יִנָּטוּ צִלֲלֵי עָרֶב / וַיֵּלְכוּ יַחַד וְלֹא מָצְאוּ / רַק עֲטַלֵּף אֲשֶׁר הוֹצִיאוּ / מֵחוֹרֵי עָפָר וְכֵפִים / וַיָּשׁוּבוּ רְעֵבִים וִיעֵפִים / וַיֹּאמֶר הַשֻּׁעָל: אֵיךְ יְחֻלַּק שְׁלָלֵנוּ / הֲיִשָּׁחֵט וּמָצָא לָנוּ / הָלַכְנוּ בָעֲרָבָה בָּהָר וּבַשְּׁפֵלָה / וְלֹא מָצְאָה הַיּוֹנָה מָנוֹחַ לְכַף רַגְלָהּ / עַל כֵּן יֵאוֹת לָתֵת לָהּ חֶלְקָהּ / וְלֹא תִהְיֶה בֵינֵינוּ מְעֻשָּׁקָה / וְלֹא נָשִׂים עָלֶיהָ עֲלִילוֹת / לְכוּ וְנַפִּילָה גוֹרָלוֹת / מִי יַגִּיד רִאשׁוֹן זִקְנָתוֹ / וַאֲשֶׁר יִהְיֶה זָקֵן מֵעֲמִיתוֹ / שָׁמֹעַ נִשְׁמַע בְּקוֹלוֹ / וְהָעֲטַלֵּף יִנָּתֵן לוֹ / וְהַשְּׁנַיִם אֲשֶׁר לֹא כָמוֹהוּ בַיָּמִים בָּאִים / וְהָיוּ מִן הַשָּׁלָל חַטָּאִים / וַיַּעֲנוּ הַשְּׁנַיִם: עֲצָתְךָ טוֹבָה וּנְכוֹנָה / וַיַּפִּילוּ גּוֹרָלוֹת וַיִּפֹּל הַגּוֹרָל עַל יוֹנָה / וַתֹּאמֶר הַיּוֹנָה: אִם עִמָּכֶם יָגַעְתִּי וְלֹא מָצָאתִי מָנוֹחַ / אֲנִי הַיּוֹנָה אֲשֶׁר יָצָאתִי מִתֵּבַת נֹחַ / וּבֵין עֲצוּמִים אֲחַלֵּק שָׁלָל וַאֲשַׁסֵּף / כַּנְפֵי יוֹנָה נֶחְפָּה בַכֶּסֶף / וַיֹּאמֶר הַשֻּׁעָל: זָקַנְתִּי וְאֵין מַגִּיד מִסְפַּר שְׁנוֹתַי / אַךְ כָּךְ מְקֻבְּלַנִי מֵאֲבוֹתַי / אֲשֶׁר הָיוּ מִיָּמִים קַדְמֹנִים / סִמָּן לָדַעַת מִסְפַּר הַשָּׁנִים / אֲשֶׁר עָבְרוּ עָלַי מִיּוֹם הִוָּלְדִי / כָּל שֵׂעָר לָבָן אֲשֶׁר בִּי עֵדִי / לְכָל שֵׂעָר שָׁנָה, וַאֲשֶׁר הָאֱמֶת יְבַקֵּר / לְמִסְפַּר שָׁנַי אֵין חֵקֶר / וַיַּעַן הַזְּאֵב וַיֹּאמֶר בָּחֳרִי / שֻׁעָל וְיוֹנָה, שִׁמְעוּ נָא דְבָרִי / זָקֵן מִמֶּנִּי אֵין אִתִּי / וַאֲנִי זָקַנְתִּי וְשַׂבְתִּי / אַךְ סִמַַּן שְׁנוֹתַי לֹא בְאֹדֶם וְלֹא בְלִבּוּן / אֶסְפְּרֵם מֵחוֹל יִרְבּוּן / וַאֲנִי יָגַעְתִּי לָדַעַת וּמָצָאתִי / כַּאֲשֶׁר מִשְּׁתֵּי שָׁנִים יָצָאתִי / בָּאתִי אֶל הַשְּׁלִישִׁית / אַךְ לֹא אֵדַע אַחֲרִית וְרֵאשִׁית / וּמִי יוֹרֶה אֹתִי וִיאַלֵּף / לַעֲזֹב לַקָּטָן מִמֶּנִּי הָעֲטַלֵּף / וַיֹּאכַל אוֹתוֹ לְעֵינֵיהֶם / וּמְאוּמָה לֹא הִשְׁאִיר לָהֶם.


וְהַמָּשָׁל בַּחֲנֵפִים וְאַנְשֵׁי תֹעָה / זֵדִים וְעוֹשֵׂי רִשְׁעָה / הַמְקָרְבִים עַמִּים בְּפִיהֶם / וּמְרַחֲקִים מִכִּלְיוֹתֵיהֶם / כִּי יִהְיֶה לְאִישׁ דָּבָר אֶצְלָם / יָשִׁיבוּ בְמַעֲנֶה רַךְ כְּרָךְ וְנִכְלָם / וְאֹרְחוֹת יֹשֶׁר מֵעֵינֵיהֶם יַלִּיזוּ / אַלְמָנוֹת שְׁלָלָם וִיתוֹמִים יָבֹזוּ / וּבַפֶּה יֹאמַר: חֶפְצִי לְחֶפְצוֹ נִקְשָׁר / לַעֲשׂוֹת הַטּוֹב וְהַיָּשָׁר / וּבְהַגִּיעַ דְּבַר מִשְׁפָּט לְאֶחָד מֵהֶם וּשְׁפָטוּם / וַיֹּאמַר הַשֹּׁפֵט: שַׁלֵּם, אָז יְהִי אָטוּם / מִלִּשְׁמֹעַ דָּבָר אֲשֶׁר לֹא יַחְפֹּץ / יַחֲרֹק שֵׁן וּפֶה יִקְפֹּץ / וְאָז יָעִיר כָּל חֲמָתוֹ / כִּי יִבְטַח כְּגִבּוֹר בִּגְבוּרָתוֹ / לְהָעִיר גְּזֵלוֹת וַחֲבֹלוֹת / לָלֶכֶת אֳרָחוֹת עֲקַלְקַלּוֹת / כַּאֲשֶׁר עָשָׂה הַזְּאֵב לַיּוֹנָה וְשֻׁעָל / אֲשֶׁר הִמְתִּיק עִמָּהֶם סוֹד וּבְאַחֲרִית מָעַל / לָקַחַת שֶׁלֹּא כִרְצוֹנָם חֶלְקָם / תַּחַת רְשָׁעִים סְפָקָם / אַךְ שִׁבְעָתַיִם אֶל חֵיקָם זֹרְעֵי הָאֱמֶת יִקְצֹרוּ / אֶל מוּל פְּנֵי הַמְּנוֹרָה יָאִירוּ / כִּי חוֹבָה לַחַיָּב וּזְכוּת לַתָּם / בְּפָנַי נִכְתַּב וּבְפָנַי נֶחְתָּם.
