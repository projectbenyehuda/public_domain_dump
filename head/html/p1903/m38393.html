משל צז: שַׁחַל וְעִזִּים וְשֻׁעָל

תִּמְחֹל לְמִי שֶׁהֵרַע לְךָ / וְתֵן לְמִי שֶׁמָּנַע מִמְּךָ בִּשְׁאָלַךְ

שַׁחַל שָׁלַח בָּעִזִּים חֲרוֹן אַפּוֹ / וַיְהִי כַאֲשֶׁר טָרַף הֶחְבִּיא טַרְפּוֹ / וְהַשֻּׁעָל הֵבִין נְתִיבוֹת בֵּיתוֹ / וַיִּתְגַּנֵּב וַיָּבֹא בַלָּט בְּעָרְמָתוֹ / וּמִבְּשָׂרָיו אָכַל ומִגְּרָמָיו גָּרַם / וַיִּשְׂבַּע מֵהֶם וְלֹא נָתַן מִכְרָם / כַּאֲשֶׁר בַּשָּׂדֶה מָצָא מְצִיאָה / לִנְתָחֶיהָ לִנְתָחֶיהָ הוֹצִיאָהּ / וְהַשַּׁחַל רָאָה כִּי הַשֻּׁעָל / לֹא שָׁמַר אֱמוּנִים וַיִּמְעַל מַעַל / וַיִּקְרָאֵהוּ בִדְבָרִים רַכִּים וְלֹא בָחֳרִי / זֹאת הַפַּעַם עֶצֶם מֵעֲצָמַי וּבָשָׂר מִבְּשָׂרִי / אֲשֶׁר טָרַפְתִּי בַעֲמָלִי וּבִיגִיעַ כַּפַּי / אַךְ כִּלְיוֹתַי יְעָצוּנִי וֶהֱשִׁיבוּנִי שְׂעִפַּי / לֵאמֹר: לֵךְ עֲזֹב צֵידִי וֶאֱכֹל מִצֵּידְךָ / וְאִם כֹּה תוֹסִיף שֵׁנִית יָדְךָ / אֶשְׁנֶה לְךָ ׳לֹא תִגְנֹב׳ וְ׳לֹא תַחְמֹד׳ / וְשִׁלַּשְׁתָּ תֵּרֵד מְאֹד / כַּאֲשֶׁר תִּהְיֶה בְּמֶרְיָךְ עֹמֵד / וְטֵאטֵאתִיךָ בְּמַטְאֲטֵא הַשְׁמֵד / וְלִשְׁבֹּר עֲצָמֶיךָ וּלְנַתֵּק גִּדֶּיךָ לֹא אֶתְיָאֵשׁ / כַּאֲשֶׁר יִנָּתֵק פְּתִיל הַנְּעֹרֶת בַּהֲרִיחוֹ אֵשׁ / הֲיֹאכַל שֻׁעָל וְשַׁחַל יִמְחַץ / לֹא הִדְרִיכוּהוּ בְנֵי שַׁחַץ / וְאִם יִפֹּל בְּיָדְךָ אֲשֶׁר הִקִּיף מְצוּדוֹ / לֹא יַחֲרֹךְ רְמִיָּה צֵידוֹ / וְאֵיךְ עָלָה עַל רוּחֲךָ לְחַפֵּשׂ מַצְפּוּנִי / וְלָמָּה כָּכָה רִמִּיתַנִי / אֶת מוּסָרִי הַשְׂכִּילָה / וְלֹא תָשׁוּב עוֹד לְכִסְלָה / וַיַּעַן הַשֻּׁעָל בְּמֶתֶק פִּתּוּי חִכּוֹ / הַהַסְכֵּן הִסְכַּנְתִּי לַעֲשׂוֹת לְךָ כֹּה / שָׁמַעְתִּי מוּסָרְךָ וּמֶתֶק נִיבֶךָ / הֵן קַלּוֹתִי מָה אֲשִׁיבֶךָ / אַךְ בְּמַעֲשַׂי בּוֹשְׁתִּי וְנִכְלַמְתִּי / לֹא אוֹסִיף עוֹד אָוֶן כַּאֲשֶׁר פָּעַלְתִּי.

וְהַמָּשָׁל לְהַצִּילְךָ מִפַּח יוֹקְשִׁים / הֲלֹא כָתַבְתִּי לְךָ שָׁלִישִׁים / חִידוֹת מְשָׁלִים וּמְלִיצוֹת / לִהְיוֹת קָשֶׁה לִכְעֹס וְנֹחַ לִרְצוֹת / אַךְ תָּשִׂים לְךָ בְּלֵב הַתְּבוּנָה שְׁבִילִים / כִּי מַרְפֵּא יַנִּיחַ חֲטָאִים גְּדוֹלִים / וְאִם תִּרְאֶה רֵעֲךָ הֹלֵךְ דֶּרֶךְ פָּרִיצִים / מָשַׁךְ יָדוֹ אֶת לוֹצְצִים / וְלֹא יִשְׁבֹּת לְהוֹסִיף עַל חַטָּאתוֹ / מִלּוּשׁ בָּצֵק עַד חֻמְצָתוֹ / וּדְבַר בְּלִיַּעַל יָצוּק בּוֹ / בְּאִמְרֵי נֹעַם קְרַע סְגֹר לִבּוֹ / וְשַׁנֵּה וְשַׁלֵּשׁ בְּמוּסָרְךָ לַהֲשִׁיבוֹ / כִּי יָדַעְתִּי אֶת מַכְאוֹבוֹ / וֶאֱמֹר לוֹ: שֹׁנֶה בְאִוַּלְתּוֹ וְעֹשֵׂה שַׁעֲרוּרִיָּה / אַחַת דָּתוֹ תַּאֲנִיָּה וַאֲנִיָּה / לְמִי אוֹי לְמִי אֲבוֹי לְמִי מִדְיָנִים / לַאֲשֶׁר בְּאִוַּלְתָּם כָּמוֹךָ שֹׁנִים / וְאִם עַד הַשְּׁנַיִם בָּא וְעֹבֵר / פַּעֲמַיִם שָׁלוֹשׁ עִם גָּבֶר / אָז אֵין תִּקְוָה לְאַחֲרִיתוֹ / כִּי גַם הַשּׁוֹר בִּנְגִיחָתוֹ / מוּעָד בְּשָׁלֹשׁ נְגִיחוֹת / חָרוּת עַל הַלֻּחוֹת / וְאִם לֹא יָשִׁיב אֶל לִבּוֹ לֹא דַעַת וְלֹא תְבוּנָה / יָדְךָ תִּהְיֶה בּוֹ בָרִאשׁוֹנָה / וְאָמַר הֶחָכָם: מְחֹל לְמִי שֶׁהֵרַע לְךָ / וְתֵן לְמִי שֶׁמָּנַע מִמְּךָ בִּשְׁאָלְךָ.
