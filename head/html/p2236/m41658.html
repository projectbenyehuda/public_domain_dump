[אֲנִי גֶבֶר בְּסִבּוֹת הַזְּמַנִּים] (שִׁיר חֲתִימָה לְסֵפֶר מַעֲלוֹת הַמִּדּוֹת)

שִׁיר חֲתִימָה לְסֵפֶר מַעֲלוֹת הַמִּדּוֹת (קֶטַע)

̮̮ _ _ _ ̮ _ _ _ ̮̮ _ _

אֲנִי גֶבֶר בְּסִבּוֹת הַזְּמַנִּים / שְׂבַע רֹגֶז שְׂבַע בוּז עִם-קְלוֹנִים

וְדָלְקוּ אַחֲרַי צָרוֹת מְצֵרוֹת / וְשֹׁד עַל-שֹׁד וְעַל-שֶׁבֶר חֲרוֹנִים

וְצָלַלְתִּי בְּעִמְקֵי הַדְּאָגוֹת / וְעָלַי עָֽבְרוּ גַּלֵּי יְגוֹנִים

וְטֶרֶם אֶמְצְאָה מַרְפֵּא לְצִירַי / בְּצִדָּם צָֽמְחוּ הָאַחֲרוֹנִים

לְהַוָּתִי אֱמֶת שָׂחְקוּ צְעִירִים / לְמוֹ יָמִים וְשׁוּעָלִים קְטַנִּים

מְאַסְתִּימוֹ וְתִעַבְתִּי אֲבוֹתָם / לְשִׁיתָם שׁוֹֽמְרֵי עִזִּים וְצֹאנִים

וְתִבְחַר מַחֲנַק נַפְשִׁי וּמָוֶת / רְאוֹתִי בוּז וְלַעַג שַׁאֲנַנִּים

בְּשִׁמְרֵיהֶם מְאֹד שׁוֹקְטִים וְיוֹשְׁבִים / בְּנָוֵיהֶם דְּשֵׁנִים רַעֲנַנִּים

אֲשֶׁר אָמְרוּ לְנַפְשִׁי אֵין תְּשׁוּעָה / וְלֹא חֵלֶק בְּיָהּ רוֹכֵב עֲנָנִים

נְגִינָתָם אֲנִי שִׁבְתָּם וְקוּמָם / וְכִנּוֹרוֹת וְעוּגָבִים וּמִנִּים

וְהָיִיתִי לְמָשָׁל וּֽשְׁנִינָה / לְשָׁתוּקִים אֲסוּפִים וּֽנְתִינִים

וְכָל-רוֹאַי הֲכִי שָׁמְמוּ וְתָמְהוּ / לְקוֹרוֹתַי וְיוֹדְעַי מִלְּפָנִים

וְיֹאמְרוּ לִי חֲלָצֶיךָ תְּשַׁנֵּס / כְּאִישׁ גִּבּוֹר בְּרָב-כֹּחַ וְאוֹנִים

וְחֵץ בִּינָֽתְךָ תּוֹרֶה בְּלֵב הַ-/זְּמָן בָּגַד בְּךָ בֶּגֶד כְּמוֹנִים

חֲנִית חָכְמָֽתְךָ הַכֵּה בְקִרְבּוֹ / וְחַרְבוֹת שִׂכְלְךָ הָרֵק וְצִנִּים

וְזַמֵּן שִׂכְלְךָ תָּמִיד לְנֶגְדּוֹ / וְתִתְלוֹנֵן בְּסֵתֶר דַּר מְעוֹנִים

וְתָנוּחַ וְתִשְׁקֹט מִזְּמַנְךָ / וְתִנָּפֵשׁ לְךָ מִדַּאֲבוֹנִים

וְשָׁמַעְתִּי לְמִצְוָתָם לְמַעַן / חֲשַׁבְתִּימוֹ לְרֵעִים נֶאֱמָנִים

וְלָקַחְתִּי לְנַפְשִׁי תוֹֽכְחוֹתָם / וְהָיוּ לִי יְקָרִים מִפְּנִינִים

וְאָמַרְתִּי בְלִבִּי אֵֽלְכָה נָּא / בְּעִקְבֵי בַעֲלֵי שֵׂכֶל נְבוֹנִים

וְאוֹכִיחַ לְנַפְשִׁי בַתְּחִלָּה / וּבִי יִתְיַסְּרוּ שָׂרִים וְרוֹזְנִים

וְאַחְרֵי-כֵן אֲיַסֵּר לַהֲמוֹן עָם / וְאוֹכִיחַ פְּתָאִים עִם-מְבִינִים

וּמוּסָרַי לְנַפְשָׁם יֶעֱרָבוּן / כְּטַל יוֹם חֹם וְעֵת שֶׁלֶג וְצִנִּים

וְאַף אָמְנָה חֲמָסִי עַל-זְמָן רַע / וּבוֹ אָרִיב וְאֶשָּׂא רֹב תְּלֻנִּים

דְּלָקַנִי בְיַלְדוּתִי וְרָדַף / עֲדֵי בַחְרוּת וְנָגַע עַד-זְקֻנִים

וְשָׁלַח-אֵשׁ בְּעַצְמֹתַי וְתִיקַד / שְׁאוֹל תַּחְתִּית וְהִרְגִּיזָה אֲדָנִים

וְהַנִּיר לִחֲכָה גַּם דִּלְדְּלָה הָר / וְסִכְסָכָה לְעֵצִים וַאֲבָנִים

הֲכִי בָנַי יְצָאֻנִי וְאֵינָם / לְנָכְרִים נֶחְשְׁבוּ-לִי גַּם לְשׂוֹטְנִים

עֲזָבוּנִי וְנוֹתַרְתִּי לְבַדִּי / וְכִכְלִי רִיק חֲשָׁבוּנִי כְעוֹיְנִים

וְלֹא-אָבוּ שְׁמוֹעַ לִי וְהָלְכוּ / בְּמוֹעְצוֹתָם לְאָחוֹר לֹא לְפָנִים

לְזֹאת אֶסְפֹּד אֲקוֹנֵן וַאֲיֵלִיל / וְאֵבֶל אֶעֲשֶׂה כִּבְנוֹת יְעֵנִים

לְקוֹל בִּכְיִי אֲעוֹרֵר כָּל-מְקוֹנְנוֹת / וְכָל-יוֹדְעֵי נְהִי עִם-כָּל-מְקוֹנְנִים

וְאֶתְבַּיֵּשׁ לְהָרִים רֹאשׁ לְמַעַן / כְּלִמָּה כִסְּתָה פָנַי כְּשָׁנִים

וְעִם-כָּל-זֶה הֱעִירַנִי לְבָבִי / יְשִׁיבוּנִי סְעִפַּי הַנְּכוֹנִים

וְיֹאמְרוּ לִי חֲמֹל עַל-הַפְּתָאיִם / לְהָסִיר מִפְּנֵיהֶם כִּֽשְׁלוֹנִים

וְחִבַּרְתִּי יְסוֹד סֵפֶר יְסַדְתִּיו / בְּאַגָּדוֹת אֲמִתִּיִּים וְכֵנִים

בְּמִקְרָאוֹת וּמִשְׁלֵי הַחֲכָמִים / וְחִידוֹתָם גְּלוּיִם עִם-צְפוּנִים

חֲצוּבִים מִמְּקוֹר בִּינָה וְהַשְׂכֵּל / בְּמוֹ-דֵעַ מְיֻסָּדִים נְכוֹנִים

כְּלַלְתִּיהוּ בְּכָל-חָכְמָה וּמוּסָר / לְהוֹכִיחַ לְבָנִים הַהֲגוּנִים

לְהָגֵן בַּעֲדָם מִן-הַזְּמָן מַר / לְמָתוֹק שָׂם וּמָתוֹק רֹאשׁ פְּתָנִים

לְבִלְתִּי יִמְצְאוּם נֶגַע וּמַחְלָה / בְּחַיֵּיהֶם וְאַף-לֹא קוֹל שְׁאוֹנִים

וְאַחְרִיתָם בְּעֵדֶן גַּן אֶלֹהִים / בְּצַד אָבוֹת יְהוּ שׁוֹכְנִים וְחוֹנִים.
