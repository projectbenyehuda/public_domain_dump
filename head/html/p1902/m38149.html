ויאמר השכמתי בבוקר: מקאמה

מִשְׁפַּט הַיַּיִן


– – – וַיֹּאמַר: הִשְׁכַּמְתִּי בַבֹּקֶר וָאֵצֵא לְדַרְכִּי / לַעֲשׂוֹת חֶפְצִי וְצָרְכִּי / וּבְעוֹדִי מִתְהַלֵּךְ בִּרְחוֹב הַמְּדִינָה / אָנֶה וָאָנָה / מָצָאתִי צִבְאוֹת בְּנֵי אָדָם כִּצְבָאִים רָצִים / וְעַם רַב מִכָּל פִּנָה נִקְבָּצִים / וּשְׁנֵי אֲנָשִׁים עִבְרִים נִצִּים / הָאֶחָד זָקֵן יַאֲמִיר בְּאִמְרָתוֹ הַמְּזִמּוֹת / וְהַשֵּׁנִי עֶלֶם יְגַלֶּה בִּמְלִיצָתוֹ הָעֲלוּמוֹת / זֶה אוֹמֵר: ׳תֶּן לִי פִקְדוֹנִי / פֶּן אֶקְרַע לִבְּךָ בְּקִלְשׁוֹנִי׳ / וְזֶה אוֹמֵר ׳תָּשֵׁב גְּזֵלָתִי / פֶּן אֲכַלֶּה בְּךָ בְּנֵי אַשְׁפָּתִי׳ / וְכַאֲשֶׁר גִּלּוּ הָאֵיבָה / וְגָדְלָה הַמְּרִיבָה / אָרוּ לָהֶם הָעָם: ׳עַד מָתַי אַתֶּם נִרְפִּים / וּפוֹסְחִים עַל שְׁתֵּי הַסְּעִפִּים? / עֲשׂוּ בֵינֵיכֶם פְּשָׁרָה / אוֹ לְכוּ עַד הַשּׁוֹפֵט אֲשֶׁר בַּשַּׁעְרָה׳ / וַיָּהִינוּ לָלֶכֶת אֶל הַשּׁוֹפֵט / וְלֵב כָּל אֶחָד מֵהֶם בְּחַיָּיו נוֹקֵט / וָאוֹמַר בְּלִבִּי: ׳אֵלְכָה וְאֶחֱזֶה / הַמַּרְאֶה הַגָּדוֹל הַזֶּה׳ / וְכַאֲשֶׁר עַד הַָשּׁוֹפֵט הִגַּעְנוּ / וּבִמְכוֹן כִּסְאוֹ פָּגַעְנוּ / נִגַּשׁ הַזָּקֵן בְּעָרְמָה וַיִּתְמַרְמֵר / וַיַּעַן וַיֹּאמַר: ׳בִּי הָאָדוֹן / אֲשֶׁר בֵּין עֲצוּמִים יַפְרִיד וְיָדוֹן / יֶשׁ לִי עִם זֶה הַבָּחוּר פִּקָּדוֹן / בָּרוּךְ הוּא וּבָרוּךְ טַעְמוֹ / וְהַשִּׂמְחָה עַל שִׁכְמוֹ / וּכְבָר נִשְׁבַּעְתִּי בְּזֹהַר צַלְמוֹ / וְזֵכֶר בָּשְׂמוֹ / לֹא אֶזְכְּרֶנּוּ וְלֹא אֲדַבֵּר עוֹד בִּשְׁמוֹ / אֲבָל אֲסַפֵּר לַאדֹנִי קְצָת מַהֲלָלָיו / אוּלַי תַּכִּירֶנּוּ בְּיֹפִי פְעָלָיו׳.


דַּע כִּי הָיְתָה לִי עַלְמָה תַמָּה / אֲדֻמָּה אֲיֻמָּה / בָּרָה כַּחַמָּה / רֻקָּהּ – צוּף לְמַלְקֹחַיִּם וְעֵינָהּ – רִצְפָּה בְמֶלְקָחַיִם / בַּת יוֹשֶׁבֶת בְּפֶתַח עֵינַיִם / לֹא תִתְעַלֵּם מֵרוֹאָהּ / וְאִם הִיא צְפוּנָה בְּכַרְמָהּ בְּכִלְאָהּ / וְלֹא תַסְתִּיר מַצְפּוּנֶיהָ / וְלֹא תַעְלִים רֵיחַ שְׁמָנֶיהָ וְהַכְּלִמָּה לֹא כִסְּתָה פָנֶיהָ / דַּלָּה וְדַקָּה / לִיגוֹן תַּחֲלִיף רַקָּה / לֶחְיָהּ כְּאֹדֶם הַשּׁוֹשָׁן וְעֵינָהּ כַּדָּם אָדַם / וּבָהּ שִׂמְחַת אֱלֹהִים וְאָדָם / תַּפִּיל רְפָאִים / וְתָשִׂים הַפְּלֵטִים / בְּחַרְבָּהּ פְּלֵטִים / וּתְלַבֵּב לְבָאִים / וְשֻׁלְחָנָהּ עָרוּךְ לַיּוֹצְאִים וְלַבָּאִים / דַּלָּה יְרַקְרֶקֶת / כְּאִלּוּ הִיא חוֹשֶׁקֶת / וּמְעִילֶיהָ רְצוּפִים אֶקְדָּח וּבָרֶקֶת / תַּרְשִׁיעַ צַדִּיקִים וְהִיא מַצְדֶּקֶת / מָרָה וְנִמְתֶּקֶת / כִּי רוֹדֶיהָ תָּמִיד נוֹשֶׁקֶת / וַעֲלֵיהֶם מֵאוֹרָהּ חוֹלֶקֶת / הֵם מַגִּישִׁים אֵלֶיהָ וְהִיא מוֹצֶקֶת / בָּלְתָה וְלֹא נָמַר רֵיחָהּ / וְרַבּוּ יָמֶיהָ וְלֹא כָהֲתָה עֵינָהּ / וְלֹא נָס לֵחָהּ / בְּקִרְבָּהּ שִׂמְחוֹת תֵּבֵל צְפוּנוֹת / רַבָּתִי בַגּוֹיִם שָׂרָתִי בַמְּדִינוֹת.


עַלְמָה בְּזִיוָהּ פַּאֲתֵי הָאָרֶץ

תָּאִיר, עֲדֵי תַחְפִּיר מְאוֹרֵי עֶרֶץ

אַחַר צְבָא יָגוֹן חֲנִית תִּשְׁלֹף, וְיָד

תַּחְשֹׂף, עֲדֵי תַחְרֹץ חֲיָלָיו חֶרֶץ.

תַּזְקִין – וְעֶדְנָה יוֹם לְיוֹם תּוֹסִיף, וְאִם

תִּבְלֶה – תְּחַדֵּשׁ הוֹד וְתִפְרֹץ פֶּרֶץ

רֻקָּהּ צְרִי גּוּפוֹת נְגוּפוֹת הוּא וְצִיר

נֶאְמָן לְרַפֵּא צִיר וְלַחְלִים קֶרֶץ

אִם תַּעֲלִימֶנָּה – בְּרִקְחָהּ תַּהֲלֹךְ

רָכִיל וְאַף תִּקְרֹץ בְּעֵינָהּ קֶרֶץ.


וָאַפְקִידָהּ אֵצֶל זֶה הַנַּעַר / הַסָּכָל וְהַבַּעַר / וַיָּבוֹא אֵלֶיהָ בַּלָּאט / וַיִּקְרַב לָהּ מְעַט / וַיָּרַח רֵיחַ בָּשְׂמָה / וּתְפָשָׂהּ וְשָׁכַב עִמָּהּ / וְצַעֲקָה הַנַּעֲרָה וְאֵין מוֹשִׁיעַ לָהּ / וַיָּבֹא בִזְדוֹנוֹ עָלֶיהָ / וַיּוֹצִיא דַּם בְּתוּלֶיהָ / וַתַּהַר וַתֵּלֶד בֵּן נֶחְמָד וְנָאֶה / יְפֵה תֹאַר וִיפֵה מַרְאֶה / אַדְמוֹנִי עִם יְפֵה עֵינַיִם / יָשִׁיב הַחֲשֵׁכָה צָהֳרַיִם / וּסְבִיבוֹ יָפִיץ זֹהַר / כְּעֶצֶם הַשָּׁמַיִם לָטֹהַר / בּוֹ גָדַל מְשׂוֹשׂוֹ וְחָדַל כַּעְסוֹ / וַתִּהְיֶה הַשִּׂמְחָה מְנָתוֹ וְכוֹסוֹ / וַיָּשֶׂם אוֹתוֹ עַל זְרוֹעוֹ / וַיִּתֶּן בּוֹ שַׁעֲשׁוּעוֹ / וַיְכַבְּדֵהוּ כְאָבִיו / וַיָּשֶׂם פִּיו אֶל פִּיו / וַיִּקְרַב עוֹד אֵלֶיהָ וַתּוֹלֶד בֵּן / מְעֻטָּף בִּרְדִידֵי חֵן / זָקֵן בָּא בַּיָּמִים / לא ישנה… / וַיִּשְׂמַח בּוֹ כִּפְלַיִם / וַיָּשָׁב אוֹר גִּילוֹ לְשִׁבְעָתַיִם / וַיּוֹסֶף עוֹד לָדַעַת אֶת הַבְּתוּלָה / כִּי לֹא יִשְׁקֹט הָאִישׁ כִּי אִם כִּלָּה / וַתִּדְבַּק נַפְשׁוֹ בְּאַהֲבָתָהּ… קְדֹשָׁה / וַתֵּלֶד לוֹ בָּנִים שִׁשָּׁה / וַתִּשָּׁאֵר נִבְזֵית אַחַר הֱיוֹתָהּ נִכְבֶּדֶת / וְנִמְאֶסֶת עֵקֶב הֱיוֹתָהּ נֶחְמֶדֶת / וַתַּעֲמֹד מִלֶּדֶת / וְכִרְאוֹתוֹ כִּי נִשְׁאֲרָה כְּנִבְלֵי חֶרֶשׂ / אַחַר הֱיוֹת אוֹרָהּ כְּאוֹר חֶרֶס / מִהַר וַיִּקָּחֶנָּה / וַיְדַבֵּר עַל לִבָּהּ וַיּוֹכִיחֶנָּה / לֵאמֹר: ׳לֹא טוֹב עָזְבֵךְ אָבִיךְ וְצֵאתֵךְ מִגְּבוּלוֹ / כִּי הוּא אֲדוֹנַיִךְ וְהִשְׁתַּחֲוִי לוֹ / כִּי אֲנִי לִנְסֹעַ מִזֶּה הַמָּקוֹם צָרִיךְ / לְכִי שְׁבִי, אַלְמָנָה, בֵּית אָבִיךְ׳ / וַתָּבוֹא אֵלַי לְעֵת עֶרֶב / מִתְעַלֶּפֶת כְּחַלְלֵי חֶרֶב / וָאֹמַר לָהּ: ׳הֲמֵחֳלִי תַּם כֹּחֵךְ / אוֹ מִן הַחֹם נָס לֵחֵךְ / אוֹ נָפְלָה שְׁבָא וַתִּקָּחֵךְ / כִּי מַדּוּעַ שֻׁנּוּ פָנַיִךְ לְחוֹשְׁקָם / וְיֵרָאוּ פָנַי רֵיקָם?׳ / וַתֹּאמַר: ׳הָאִישׁ אֲשֶׁר אֶצְלוֹ הִפְקַדְתַּנִי / הוּא אֲכָלַנִי וְכִלַּנִי / וְלֹא עָזַב בִּי אוֹנִים / כִּי יָלַדְתִּי לוֹ שִׁשָּׁה בָּנִים׳ / וּכְשָׁמְעִי מִקְרֵה הָעַלְמָה הַמְפֹאֶרֶת / יָרְדָה דִמְעָתִי כְּיָם כִּנֶּרֶת / וַיְהִי בְלִבִּי כְּאֵשׁ בּוֹעֶרֶת / וְעַתָּה הוֹשִׁיעָה לְבֶן אֲמָתֶךָ וְעָשִׂיתָ כְּחָכְמָתֶךָ.


וַיַחֲרֵשׁ הַשּׁוֹפֵט שָׁעָה תַמָּה עַל חִידָתוֹ / כִּי לֹא הֵבִין הַסּוֹד הַצָּפוּן בִּמְלִיצָתוֹ / אֲבָל לִפְנֵי הָעָם הִתְחַזֵּק / וַיִּרְחַץ פָּנָיו וַיִּתְאַפֵּק / יֹּאמַר: ׳אִם דְּבָרֶיךָ נְכוֹנִים / וְטַעֲנוֹתֶיךָ נֶאֱמָנִים / יִתֵּן כְּמֹהַר הַבְּתוּלוֹת הַשּׁוֹכֵב אוֹתָהּ וַיְעַנֶּהָ / וְהָאִשָּׁה וִילָדֶיהָ / תִּהְיֶה לַאדֹנֶיהָ׳.


וַיִּגַּשׁ הַנַּעַר / וַיֹּאמַר: / ׳בִּי אֲדֹנִי / שְׁמָעֵנִי / וְאַחַר תְּצַדְּקֵנִי / אוֹ תַרְשִׁיעֵנִי / אֲדֹנִי יוֹדֵעַ, כִּי מֶלֶךְ הַתַּאֲוָה יְפַתֶּה בְּנֵי אִישׁ בִּשְׁקָרָיו / וְיֵצֶר לֵב הָאָדָם רַע מִנְּעוּרָיו / וְכַאֲשֶׁר הִפְקִיד הָעַלְמָה אֶצְלִי מָצְאָה חֵן בְּעֵינַי / וְשָׁגוּ בָהּ רַעְיוֹנַי / וָאֶשְׁאַל אוֹתָהּ עַל מִסְפַּר יָמֶיהָ / לָדַעַת הֲיֵשׁ לְאָבִיהָ מִמְשָׁל עָלֶיהָ / וַתֹּאמַר: ׳מִימֵי הַמַּבּוּל נוֹלַדְתִּי / וּבְחֵיק נֹחַ נֶאֱמַנְתִּי / וְלִבְנוֹת לוֹט בְּהִתְיָעֲצָם הִתְחַבַּרְתִּי / וַאֲנִי זָקַנְתִּי וָשַׂבְתִּי׳ / וּכְשָׁמְעָהּ יָדַעְתִּי כִּי הִיא מַשְׂכֶּלֶת / וְעַל נַפְשָׁהּ מוֹשֶׁלֶת / וָאֹמַר לָהּ: ׳אִם טוֹב בְּעֵינַיִךְ, הַבַּת הַקְּדוֹשָׁה / אֲשֶׁר לְרוֹאָהּ מְבִישָׁה / שֶׁתִּהְיִי אֶצְלִי אֱמוּנָה / אֵרַשְׂתִּיךְ לִי בֶאֱמוּנָה׳ / וַתֹּאמַר ׳בְּצִלְּךָ אֶחֱסֶה / וְכָל אֲשֶׁר בִּלְבָבְךָ עֲשֵׂה׳ / וְלֹא לְקַחְתִּיהָ אַךְ כְּמִשְׁפַּט הַנָּשִׁים / עַל יַד חֻפָּה וְקִדּוּשִׁים / וַתֵּלֵד שִׁשָּׁה בָנִים / הֶחֱלִיפוּ לִי כֹּחַ וְאוֹנִים / וַיֵּצֵא הָרִאשׁוֹן אַדְמוֹנִי / וְאַחֲרָיו יָצָא אָחִיו אֲשֶׁר עַל יָדוֹ הַשָּׁנִי / הָרִאשׁוֹן / הֵאִיר לְעֵינַי כָּל אִישׁוֹן / וְצִוָּה לִפְקֹחַ לוֹ הָאִישׁוֹן / וְלַעֲרוֹךְ שִׁיר עוֹרֵר הַלָּשׁוֹן / וְהַשֵּׁנִי / דָּלַק אַחַר צְבָא יְגוֹנִי / וְהַשְּׁלִישִׁי / שִׁלַּח סוֹד לְבָבִי חָפְשִׁי / וְהָרְבִיעִי הִשְׁכִּירַנִי בְרֻקּוֹ עַד חֲשַׁבְתִּיו שְׁבִיעִי / וַהַחֲמִישִׁי / הֵאִיר נֹגַהּ שִׁמְשִׁי / וְהַשִּׁשִּׁי / הִפִּיל לָאָרֶץ רֹאשִׁי / וְכִרְאוֹתָהּ כִּי פִתַּתְנִי וְהִשִּׂיאַתְנִי / עֲזָבַתְנִי בְּאַהֲבָתָהּ חוֹלֶה / וַתָּרָץ הַנַּעֲרָה וַתַּגֵּד לְבֵית אָבִיהָ כַּדְּבָרִים הָאֵלֶּה / וְעַתָּה, אֲדֹנִי, אֵיךְ אֶעֱזֹב מְאוֹר עֵינַי / וְאֶתֶּן לוֹ מְשׂוֹשׂ רַעְיוֹנַי / אֶת אִשְׁתִּי וְאֶת בָּנַי? / וְהִנֵּה עֹשֶׁק עָשַׁק (?) זֶה (?) וּמָעַל מַעַל / עַל הָאִשָּׁה אֲשֶׁר לָקַח וְהִיא בְּעוּלַת בַּעַל.


אָמַר הַמַּגִּיד: וְכִרְאוֹת הַשּׁוֹפֵט כִּי טַעֲנוֹתָם חֲזָקוֹת / וְחִידוֹתָם עֲמֻקּוֹת / וְכִרְחוֹק מִזְרָח מִמַּעֲרָב מִמֶּנּוּ רְחוֹקוֹת / נִבְהַל רַעְיוֹנוֹ / וַיִּדְבַּק לְחִכּוֹ לְשׁוֹנוֹ / וַתַּעַל חֲמַת קִצְפּוֹ בְּאַפּוֹ / וַיִּטְרֹף נַפְשׁוֹ בְּאַפּוֹ / וַיּוֹשֶׁט כַּפּוֹ אֶל צְרוֹר כַּסְפּוֹ / וַיּוֹצִיא שִׁבְעִים כֶּסֶף / וַיֹּאמַר ׳כִּרְתּוּ בְאֵלֶּה הַמְּרִיבָה וְהַקֶּצֶף!׳ / וַיַחְטְפֵם הַזָּקֵן בְּמִרְמָה / וְלַנַּעַר לֹא נָתַן מְאוּמָה / וַיֹּאמַר: ׳חֶצְיָם חֶלְקִי מִן הַמַּתָּנָה / וְחֶלְקְךָ – מֹהַר הָעַלְמָה הַנֶּאֱמָנָה / מַהֵר לָקַחַת אִשְׁתְּךָ אֲשֶׁר לָקַחְתָּ / וּרְאֵה חַיִּים עִם אִשָּׁה אֲשֶׁר אָהַבְתָּ׳.


וְכִשְׁמֹעַ הַנַּעַר / גָּדַל עָלָיו הַצַּעַר / וְרַעְיוֹנָיו נִשְׂעֲרוּ שַׂעַר / וַיֶּהֱמוּ מֵעָיו / וַיֵּרְדוּ פַּלְגֵי דְמָעָיו / וַיִּגַּשׁ אֶל הַשּׁוֹפֵט בְּעַיִן נוֹטֶפֶת / וְדִמְעָה שׁוֹטֶפֶת / וַאֲנָחָה שׂוֹרֶפֶת / וְעָרְמָה עוֹדֶפֶת / אֵשׁ בְּקִרְבּוֹ – וּמַרְאֶה כִּי לִבּוֹ חֳמַרְמַר / וַיַּעַן וַיֹּאמַר:


בִּי אֲדוֹנִי / הִנְנִי וַאֲגַלֶּה לְךָ סוֹדִי / כִּי נֶהְפַּךְ עָלַי לְמַשְׁחִית הוֹדִי / דַּע כִּי הָעַלְמָה אֲשֶׁר לָקַחְתִּי לִי לְאִשָּׁה / אֲשִׁישָׁה הַשְּׂמָחוֹת לַלֵּב מְחִישָׁה / וְהַיְלָדִים הַנֶּחְמָדִים / גְּבִיעִים מְשֻׁקָּדִים / וְעַתָּה מְחִירָהּ יִקַּח לוֹ לְמָנָה / וְאֶחֱלֹק עִמּוֹ בַּמַּתָּנָה / וְיַעֲמֹד שָׁמֵם מְחַכֶּה / וְדִמְעוֹ מֵעֵינָיו מְפַכֶּה / עַד הָמוּ רַחֲמֵי הַשּׁוֹפֵט עָלָיו / וַיִּקְרָא אֵלָיו / וַיּוֹצִיא מִתַּחַת בִּגְדוֹ שְׁלֹשִׁים כֶּסֶף וַיִּתְנֵם אֵלָיו בַּלָּאט / וַיֹּאמֶר: ׳קַח אֵלֶּה, וְאַל תָּשׁוּבוּ לַעֲמֹד לְפָנַי בַּמִּשְׁפָּט!׳


וְכַאֲשֶׁר פָּרְשׂוּ לוֹ מְצוּדוֹתָם / וְנִלְכַּד בִּשְׁחִיתוֹתָם / וְהֶאֱכִילוּהוּ לֶחֶם קֶצֶף / וְעָנְשׁוּ אוֹתוֹ מֵאָה כֶּסֶף / יָצְאוּ מֵאִתּוֹ בְּשָׂשׂוֹן שָׁלֵם / וָאֵצֵא אַחֲרֵיהֶם מִתְעַלֵּם / וְכַאֲשֶׁר הִרְחִיקוּ, הִגְדִּילוּ שִׂמְחָה וּשְׂחוֹק / עַד נִשְׁמַע קוֹלָם מֵרָחוֹק / וָאֶקְרַב אֶל הַזָּקֵן וָאֹמַר: ׳הִשְׁבַּעְתִּיךָ, בַּאֲשֶׁר שָׂם מֶתֶק שְׂפָתֶיךָ כְּיַעַר / וּלְשׁוֹנְךָ לָטוּשׁ כְּתַעַר / וְחֵמֶת צוּף וְרוֹשׁ פֶּתֶן תְּעַר / וּמוּם אוֹהֲבֶיךָ תַּסְתֵּר וּמוּם אוֹיְבֶיךָ תְּעַר / וְלֵב מְקַנְאֶיךָ לְחַרְבּוֹ תַעַר / הֲבִנְךָ זֶה הַנַּעַר / אֲשֶׁר הוּא בְּגִילוֹ כְּתַפּוּחַ בַּעֲצֵי הַיַּעַר?׳ / וַיֹּאמַר: ׳בְּנִי וְתַאֲוַת עֵינִי / אֲנִי יָחִיד בַּמְּלִיצוֹת וְאֵין שֵׁנִי / וְהוּא יִגְזֹר עֲלֵיהֶם וְאֵין מֵנִיא / טֶרֶם נִקְרָא לָשִׁיר יַעֲנֶה / וְעַל פִּינוּ יִסַּע צְבָאוֹ וְיַחֲנֶה / וּבְמִלְחֲמוֹתֵינוּ נִיבֵנוּ – חַרְבֵּנוּ / וְקִלְּשׁוֹנֵנוּ / לְשׁוֹנֵנוּ / וְקַשְׁתוֹתֵינוּ – שִׂפְתוֹתֵינוּ / וַחֲנִיתוֹתֵינוּ – אִמְרוֹתֵינוּ / וְלָנוּ הַגְּבוּרָה בְּמַעֲרַךְ הַמִּבְטָא / וְשָׁם נִקְלַע אֶל הַשַּׂעֲרָה וְלֹא נֶחֱטָא / וְלֹא יַשִּׂיגוּ הַבְּרָקִים אֲבַק מַרְכְּבוֹתֵינוּ / וְלֹא יֵדְעוּ מְקוֹם תַּחֲנוֹתֵינוּ / כְּאִלּוּ סוּסֵינוּ חֲצוּבִים מִלֶּהָבָה / אֶחָד יִרְדֹּף אֶלֶף וּשְׁנַיִם יָנִיסוּ רְבָבָה׳ / וַיִּשָּׂא מְשָׁלוֹ

וַיֹּאמַר:


נָרִיק דְּבַשׁ הַשִּׁיר בְּפִי אוֹהֵב, וְצוּר

חָזָק נְשַׁבֶּר בּוֹ וְכָל הַר נַחֲרִיב

נָרִיק עֲלֵי צָרִים חֲנִיתוֹת, וּבְצוּר

חֶרֶב לְשׁוֹנֵנוּ לְשׁוֹנָם נַחֲרִיב.


וַיִּקְרְאוּ עָלַי שְׁלוֹמָם / וַיָּסֹבּוּ לָלֶכֶת לִמְקוֹמָם / וָאֶעֶזְבֵם וְזִכְרָם בַּעֲדִי עָרֵב / וַיֵּלְכוּ לְאָהֳלֵיהֶם שְׂמֵחִים וְטוֹבֵי לֵב. – – –
