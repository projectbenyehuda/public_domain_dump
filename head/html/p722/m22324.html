דּוֹדִים שׁוֹרְרוּ אַהְבָה עוֹרְרוּ

לשר שלמה אבי משה לרגל חתונת בנו אברהם


דּוֹדִים שׁוֹרְרוּ / אַהְבָה עוֹרְרוּ

וּבְשִׁירֵי נְגִידִים / לָאֵל זַמְּרוּ:


שִׁירוּ שִׁיר יְדִידוֹת / לִצְבִי נֶחֱמָד

כִּי הוּא יוֹם חֲרָדוֹת / לַלֵּב מָעֳמָד

וּבְכָל הַחֲמוּדוֹת / לוֹ שֵׁם טוֹב וְיָד

מַעֲשָׂיו שָׁפְרוּ / טוּבָיו יָקְרוּ

וַעְנִיִּים מְרוּדִים / בִּשְׁמוֹ נֶעְזְרוּ:


שָׁמְעוֹ בַמְּדִינוֹת / הוֹלֵךְ הוּא וְרָב

וּלְכָל הַלְּשׁוֹנוֹת / זִכְרוֹ יָעֳרָב

כִּי מַעְיָן אֱמוּנוֹת / לוֹ לֹא יֶחֱרָב

מֵימָיו גָּבְרוּ / עַד לֹא חָסְרוּ

וּשְׂדֵי הַחֲסָדִים / לַשְׁקוֹת חֻבְּרוּ:


הוּא הַשָּׂר שְׁלֹמֹה / נוֹדָע בַּזְּמָן

כִּי הָאֵל הֱקִימוֹ / נֵטַע נַעֲמָן

אֵיתָן, שָׁם מְקוֹמוֹ / צִלּוֹ נֶאֱמָן

שׂוֹנְאָיו נֶעְדְּרוּ / בּוֹשׁוּ חָפְרוּ

מִירוֹא קוֹל פְּחָדִים / פִּתְאוֹם נִשְׁבְּרוּ:


יַגְדִּיל אֵל כְּבוֹדוֹ / וּבְטוֹב יֶחֱנַן

אֶת מֹשֶׁה חֲמוּדוֹ / וִיהִי רַעֲנָן

גַּם זַכַּאי בְּהוֹדָם / יֵשֵׁב שַׁאֲנָן

מַעְשָׂיו יִכְשְׁרוּ / יָדָיו יִצְּרוּ

צֶדֶק עִם יְדִידִים / בּוֹ יִתְפָּאֲרוּ:


גִּיל הַיּוֹם בְּשִׂמְחַת / בִּנְךָ אַב הֲמוֹן

וּבְרַנֵּן וְגִילַת / צַלְצֵל פַּעֲמוֹן

וַאְנָחָה וּמַחְלַת / לֵב וְגוּף טְמוֹן

כִּי טוֹב סִפְּרוּ / בָּךְ, עָם נִדְבְּרוּ

וּשְׁלֹשָׁה שְׂרִידִים / בָּךְ הִתְחַבְּרוּ:


הֵמָּה הַשְּׁלֹשָׁה / כִּשְׁלֹשֶׁת קְנֵי

גֻּלָּה הַקְּדוֹשָׁה / מוּל אוֹרָם חֲנֵה

בָּם שִׁירָה חֲדָשָׁה / כָּל פֶּה יַעֲנֶה

הָדָר יַחְגְּרוּ / עֹז יִתְאַזְּרוּ

וּכְלַל הַיְּהוּדִים / בָּם יִתְאַשְּׁרוּ:


אֵל עוֹשֶׂה פְלָאוֹת / הָסֵר זַעְמְךָ

וּפְדֵה מִתְּלָאוֹת / נִטְעֵי כַרְמְךָ

וּבְחֵן חֵן תְּשׁוּעוֹת / תִּקְרָא עַמְּךָ

לַעְלוֹת מַהֲרוּ / אַל תִּתְאַחֲרוּ

וּלְאִסּוּף נְדוּדִים / דֶּרֶךְ יַשְּׁרוּ:
