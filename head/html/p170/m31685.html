עַל הַדֳּמִי תִבְטַח

עוד על מות אחיו יוסף [אבו־אלחנאנ]


עַל הַדֳּמִי תִבְטַח בְּחוּל חֵילֶךָ / אוֹ בַּבְּכִי, כֹּאֵב, תְּשַׁו כִּסְלֶךָ?

עַד מֶה לְבַכּוֹת תִּהְיֶה שֹׁאֵל דְּמֵי / כָבֵד, וְהוּא נָדִיב בְּהַשְׁאִילֶךָ!

מַה־בִּצְעֲךָ תָמִיד בְּהִתְנוֹדֵד, וּמַה / יִסְכֹּן לְךָ דֶמַע וְיוֹעִילֶךָ?

יוֹרִישֶׁךָ הַלֵּב יְקֹד הָאֵשׁ, וְכֵן / עַיִן הֲמוֹן מַיִם תְּהַנְחִילֶךָ,

הֵם מִבְצְרֵי תִקְוָתְךָ יוֹם הַקְּרָב, / הֵם תֶּלְיְךָ לִלְחֹם וְהֵם כֵּלֶיךָ!

נָפַל בְּהִתְאוֹנֵן וְקוֹנֵן חֶבְלְךָ / וּסְפוֹד וְהַקְרִיחַ בְּגוֹרָלֶךָ,

וּזְמָן לְךָ הֵצִיק, וְחוֹתָם צַר בְּעֵין / מַדָּעֲךָ הֵם מֶרְחֲבֵי חֶדְלֶךָ.

נוֹעַ וְנוֹחַ הֵם חֲלִיפוֹתָיו, וְכֹל / הַיּוֹם יְשַׁחֶרְךָ לְהַבְהִילֶךָ,

וַיַּשְׁקְךָ מֵי רֹאשׁ בְּעֵת צִמְאָה, כְּמוֹ / מִנָּה פְּרִי לַעְנָה לְהַאְכִילֶךָ.

הִכְחִיד יְדִידֶיךָ וְאַחֶיךָ שְׁכֶם / אֶחָד וְאֵין שָׂרִיד בְּאֹהָלֶךָ.

עָפוּ מְשׂוֹשֶׂיךָ כְּמוֹ קַשׁ מִפְּנֵי / רוּחַ וְנָגֹלוּ שְׁמֵי גִילֶיךָ,

עַד שָׂמְךָ מָשָׁל בְּפִי תֵבֵל, וְשָׁת / חִידָה לְכָל־יָבוֹא מְרִי אֶבְלֶךָ.

מִי רַגְלְךָ מִמּוֹט יְהִי תוֹמִיךְ וּמִי / בַּצַּר יְהִי עִמְּךָ לְהַצִּילֶךָ,

אוֹ מִי יְהִי עֶזְרָה בְּיוֹם עֶבְרָה וְעַל / מִי יוֹם מְהוּמָה תַּעֲזֹב חֵילֶךָ?

הַיּוֹם אֲשֶׁר בּוֹ צָלֲלָה אָזְנִי בְּמוֹת / יוֹסֵף, מְנֻדָּח יִהְיֶה אָפְלֶךָ,

תֹּאבַד וְלֹא תִחַדְּ בְּסוֹד יָמִים, אֲבָל / יִקַּחֲךָ אֹפֶל וְיִגְאָלֶךָ.

עֵץ רַעֲנָן, הָשְׁקָה בְמֵי שֵׂכֶל, אֱמֹר: / אֵיכָה בְּפֶתַע נָבְלוּ עָלֶיךָ,

סוּפָה גְנָבַתְךָ וְשָׁרְשָׁךְ שֵׁרְשָׁה / מִתּוֹךְ עֲרוּגָתְךָ וְסָר צִלֶּךָ!

הֵילֵל בְּנוֹ שַׁחַר, הֲנָפַלְתָּ וְאֵיךְ / לֹא זָע צְבָא מָרוֹם לְקוֹל נִפְלֶךָ?

אַבִּיט פְּנֵי שֶׁמֶשׁ וְאַךְ לֹא אֶחֱזֶה / הַהוֹד אֲשֶׁר אָז נָתְנוּ עָלֶיךָ.

אֶשְׁאַף לְרֵיחַ מֹר, וְאִם לֹא יַעֲרֹךְ / בָּשְׂמֵי זְמָמֶיךָ וּמַעְלָלֶיךָ,

וָאֶטְעֲמָה נֹפֶת וְלֹא יִמְתַּק לְפִי / כִּדְבַשׁ אֲמָרֶיךָ וְצוּף מִלֶּיךָ.

לֹא נָעֲמָה אָזְנִי בְּקוֹל כִּנּוֹר בְּעֵת / טוּב לִקְחֲךָ אֶזְכֹּר וְהוֹד קוֹלֶךָ.

אֶלְעַג לְנוֹזְלֵי עָב בְּזָכְרִי מִשְׁבְּרֵי / יָם אֶצְבְּעוֹתֶיךָ וּבִשְׁעָלֶיךָ.

יִתְיַעֲצוּ שָׂרִים וְאַךְ לֹא אֶשְׁמְעָה / בֵּין מוֹעֲצוֹתָם תֹּלְדוֹת שִׂכְלֶךָ.

עָצְמָה תְלָאָתְךָ כְּפִי עָצְמָךְ וְכֵן / מַכְאוֹבְךָ גָּדַל כְּפִי גָּדְלֶךָ.

תִּכְלֶה – וְתִתְחַדֵּשׁ תְּהִלָּתְךָ, אֲבָל / תִּרְקַב – וְהוֹדְךָ חַי וּמַהְלָלֶיךָ.

מִי יִתְנֵנִי כָפְרְךָ, אָחִי, וּבִי / תִפְגַּע תְּמוּתָה בַּל קְרֹב אֵלֶיךָ,

נַדְתִּי יְמֵי חַיַּי עֲלֵי נוֹדְךָ וְהֵן / אֶתְאַו לְקֶבֶר לִהְיוֹת חַבְלֶךָ.

אֵלָיו אֲנִי קוֹרֵא לְרָחוֹק: אַחֲלַי, / קֶבֶר, וְקִבְרִי אֶחֱצֹב אֶצְלֶךָ!

צִדְקָךְ לְפָנֶיךָ יְהִי הֹלֵךְ וְשָׁם / יָכִין צְעָדֶיךָ וּמַעְגָּלֶיךָ,

תִּשְׁכַּב וְלֹא תִפְחַד וְרַחְמִי יֹצְרָךְ / הֵם מַאֲמִינֶיךָ וּמַשְׂמִילֶיךָ,

דַּי קִבְרְךָ כָּל־עֵת עֲנַן כָּבוֹד יְרַו / אֹתוֹ וְטַל אֹרוֹת יְהִי טַלֶּךָ!
