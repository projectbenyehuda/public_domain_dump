אֲהַלְלָה אֱלֹהַ

אֲהַלְלָה אֱלֹהַי / אָשִׁירָה עֻזּוֹ / אֲסַפְּרָה כְבוֹדוֹ / אֲאַפְּדֶנּוּ מְלוּכָה

אֲשַׂגֵּב לְפוֹעֵל / אֲשֶׁר שָׂח וּפָעָל / אֲאַנְוֵיהוּ כִּי לוֹ / יָאֲתָה מְלוּכָה.


בְּעֻזּוֹ נֶצַח אֲשַׁנֵּן / כִּי צְבָאוֹ אָנִי / וְלוֹ נָאֶה שִׂיחַ / גֹּדֶל הַמְּלוּכָה

בְּקָהָל אֲבַשֵּׂר / בְּרָב עָם אֲהַלֵּל / לְמִי שְׂאֵת וְיֶתֶר עָז / וּלְמִי הַמְּלוּכָה.


גֹּשׁוּ גּוֹיִם / וּבוֹאוּ מַמְלָכוֹת / רְאוּ מַה נֶּהְדָּר / בְּמֵזַח הַמְּלוּכָה

גַּדְּלוּהוּ אִתִּי / וּנְרוֹמְמֶנְהוּ יָחַד / וְאַל תִּתְגָּאוּ / בְּנֵזֶר הַמְּלוּכָה.


דְּרָכִים בְּעֵת / נֶעֶשְׂתָּה מְצוּלָה / הִתְבּוֹנְנוּ יַחַד / לְמִי נִזְרְקָה מְלוּכָה

דָּרַךְ סוּס בַּיָּם / כְּנֶגֶד שֵׁשׁ מֵאוֹת רָכֶב / וּמַה-יּוֹעִיל גֶּבֶר / עוֹז בַּמְּלוּכָה.


הֶאֱזִינוּ רוֹזְנִים / אָז וַיִּרְגָּזוּ / הִבִּיטוּ חֲתַת / וּמָאֲסוּ מְלוּכָה

הִגִּידוּ כוֹחוֹ / לְאֻמִּים וְדִבֵּרוּ / לָזֶה יִכָּתֵב / שֵׁם הַמְּלוּכָה.


וְנִלְחַם רֵאשִׁית / גּוֹיִם וְאָבָד / כִּי נִשְׁבַּע חַי / בְּכִסֵּא מְלוּכָה

וְיֻלְעַג בְּכָל-דּוֹר / כִּי לֹא לָמַד / מִי נִלְחַם בַּיָּם / וְעָטָה מְלוּכָה.


זָד עַל אֲדוֹנָיו / עֶבֶד יֹשֵׁב נֶגֶב / בְּזֹאת תִּרְגַּז אֶרֶץ / בִּשְׂאֵת עֶבֶד מְלוּכָה

זֶרַע בְּרוּכִים / הֶחֱרִימוּ אֲרוּרִים / כִּי נָתְנוּ קוֹל / לְאַדִּיר בַּמְּלוּכָה.


חֶשְׁבּוֹן וּבָשָׁן / עוֹרְרוּ מִלְחָמֶת / בְּלִי לָתֵת דֶּרֶךְ / לְצִבְאוֹת מְלוּכָה

חֵילָם נִשְׁמַד / וְאַרְצָם חֻלָּקָה / וּמֵעַל זְרוֹעָם / נָפְלָה מְלוּכָה.


טָפְשׁוּ בְּנֵי כְנַעַן / כִּי נָכְרִים הֵם / בְּאַדְמַת בְּנֵי שֵׁם / זֶרַע הַמְּלוּכָה

טְבָחָם בִּן-נוּן / עַד יְפַנֶּה אֶרֶץ / לִפְנֵי אֲרוֹן הַבְּרִית / אֲדוֹן הַמְּלוּכָה.


יוֹשְׁבֵי חֲרוֹשֶׁת / אָז הִקְשָׁה לָחַץ / עֲזָרוּהוּ בְּלִי בֶצַע / אַפְסֵי מְלוּכָה

יָהּ הִלְּחִים בָּם / צָבָא בְּלִי בָצַע / כֵּן יֹאבְדוּ שְׁאָר / וְלָאֵל הַמְּלוּכָה.


כְּאֶרֶז בַּלְּבָנוֹן / אַשּׁוּר גָּדַל / וְחֵירֵף: אוֹרִיד / כַּבִּיר מְלוּכָה

כְּלִיל אֵשׁ הֲמָמָם / בְּלֵיל שִׁמּוּרִים / וְאָז יָדְעוּ הַכֹּל / כִּי לָאֵל הַמְּלוּכָה.


לַשַּׁחַת כָּרַע בֵּל / כְּחָשְׁבוֹ עֲלוֹת לַשָּׁחַק / וְסָר מֶנּוּ לְבַב אֱנוֹשׁ / וְרָד מִמְּלוּכָה

לְכַנּוֹ הוּשַׁב / וְכֹחַ אֵל הִכִּיר / לְמֵרִים וּמַשְׁפִּיל / הִשְׁלִים מְלוּכָה.


מֻגָּרָה צֹאן לַטֶּבַח / וְנִתְכְּנוּ עֲלִילוֹת / בִּלְבוּשׁ צָעִיר / רוֹדֵם הַמְּלוּכָה

מְכוּרֵי בְלֹא הוֹן / פְּדוּיֵי בְלֹא כֶסֶף / סוֹלוּ לְמַטֶּה כַמַּיִם / לֵב הַמְּלוּכָה.


נִמְכְּרוּ יוֹנִים / לִבְנֵי יְוָנִים / וְרִיחֲקוּם מֵעַל / גְבוּל מְלוּכָה

נִיאֲרוּ בְּרִית וָדָת / וְהִמְרוּ עָם בְּאֵל / וּמִגְּרוּם בְּלֹא כֹחַ / מְכַהֲנֵי מְלוּכָה.


שָׂעִיר הֶחֱנִיף / לְהוֹרֵיהוּ בְּצֵידוֹ / וַיִּירַשׁ בְּקוֹל בֶּכִי / חֶרֶב וּמְלוּכָה

סֻגַּב חָלָק / הֱיוֹת גְּבִיר לְאַחִים / וְעוֹד תִּיסוֹב / לִישׁוּרוּן מְלוּכָה.


עֲשֵׂה לְךָ בְּצִיּוֹן / שֵׁם נוֹרָאוֹת / כְּאָז תַּצְלִיחֶנָּה / בְּכִסֵּא מְלוּכָה

עוֹרֵר וְהָקֵץ / מְשׂוֹשׂ כָּל-הָאָרֶץ / וְכוֹנֵן כִּסְאֲךָ / בְּקִרְיַת מְלוּכָה.


פְּנֵי מְאוֹר לְבָנָה / וְחַמָּה תַּחְפִּיר / וְיֵבוֹשׁוּ עוֹבְדֵימוֹ / בִּשְׂאֵיתָךְ מְלוּכָה

פָּאֵר עִיר יוֹפִי / לְבָרָה כַּחַמָּה / וְגַלֵּה לְנֶגְדֵּינוּ / כְּבוֹד הַמְּלוּכָה.


צִבְאוֹת גְּאוּלֵי צֹעַן / שׁוֹרְרוּ בְּלֵיל חָג / וְהוּא לַיְלָה נִשְׁמָר / לְסַחֵף מְלוּכָה

צָעֲדוּ בְּמֵי שַׁעַל / צָפוּ בְּרוּחַ שֵׂכֶל / אָנָה יֻנְטָעוּ / וִיקַבְּלוּ מְלוּכָה.


קֻמְּטוּ שַׁעֲרֵי זְבוּל / בֵּית עוֹלָמִים / כִּי מִבֵּינֵימוֹ / שָׁבְתָה מְלוּכָה

קָדוֹשׁ יָבוֹא / בָם לְעוֹלָמִים / וְאָז יִשְּׂאוּ רֹאשׁ / בְּחַדְּשָׁךְ מְלוּכָה.


רָבְצָה עֲדִינָה / שָׁקְטָה מֵאַלְמוֹן / כִּי אָרַךְ לָהּ / קֵץ וּמְלוּכָה

רִיבוּ מוֹשִׁיעִים / שְׂאוּ אֶדֶר מֶאֱדוֹם / וְשִׁיתוּ עַל אָדוֹן / הוֹד הַמְּלוּכָה.


שָׁוְא שָׂנֵא אֵל / וְהוּא עַל לְשׁוֹנֵינוּ / בִּקֵּשׁ אֱמֶת וָאַיִן / וְרִיחֲקָה מְלוּכָה

שַׁדַּי הָסֵר / אָוֶן מִצְּבָאֶיךָ / וְיָרִיעוּ לְךָ / תְּרוּעַת מְלוּכָה.


תַּחְגּוֹר גֵּאוּת / תִּתְאַזָּר עוֹז / לְבַל יִשְׂתָּרֵר / זָר בִּמְלוּכָה

תִכּוֹן תֵּבֵל / כִּי יֻנְעַר רָשָׁע / וְשָׂם צֶדֶק לְרַגְלָיו / וְיֻצְנַף מְלוּכָה.


תִּקּוֹם גּוֹיִם / תּוֹכִיחַ לְאֻמִּים / תִּשְׁבֹּר מַטֵּה רֶשַׁע / מוֹשֵׁל בִּמְלוּכָה

תַּחֲלִיף אֱלִילִים / תֻּשְׂגַּב לְבַדֶּךָ / תִּקָּרֵא נֶצַח / יָחִיד בִּמְלוּכָה.
