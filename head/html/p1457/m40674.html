אַפְסֵי חוּג

קדושה לשבועות


אַפְסֵי חוּג פִּלְּצָה אֵימָה

בְּפַעְנְחָךְ קִנְיְנֵי אַלְפַּיִם קְדוּמָה

גְּמוּלִים עַל קֶרֶת לְרוֹמְמָה

דָּתוֹתֵי פְלִיאָה בְּמוֹעֵצוֹת לְהַחְכִּימָה


הִטִּיתָה לְסִין מֶתַח זְרוּתֶיךָ

וְהִשְׁגַּחְתָּה בְּרִבְבוֹת שִׁנְאַנֵּי מַרְכְּבוֹתֶיךָ

זַרְזִיפִים הִטִּיפוּ לֶקֶשׁ רְבִיבוֹתֶיךָ

חַשְׁמַנִּים בֶּאֱתוֹת לְשַׁעֲשׁוּעַ אִמְרוֹתֶיךָ


טֶפַח יְסוֹדוֹת שָׁקַט בְּשִׁכְלוּלוֹת

יְמִינְךָ כְּחָרְטָה גְלִיפַת כִּלּוּלוֹת

כְּלָל וּפְרָט מְפֻתָּחוֹת בְּלִי נִגְלָלוֹת

לְהִנָּתֵן מְזִמָּה בַּחֲמוּרוֹת וְקַלּוֹת


מֵחֶמֶד לְאָלוּשׁ קִדַּמְתָּה עֲזוּרִים

נְחוֹתָם בְּעִדּוּן לֶחֶם אַבִּירִים

סְכוּכִים בַּסָּךְ בְּדִדּוּי עִטּוּרִים

עֲנָקִים לְקַבֵּל אִמְרֵי שְׁפָרִים


פְּלוּטֵי כוּר בְּעֹמֶס אֶבְרָה

צִיעָתָם הֻצָּגָה בְּיִרְאָה טְהוֹרָה

קֶצֶב מוֹר בְּחֻפָּתָם קֻטְּרָה

רְצוּיָה מֵחֲפָצִים וּמִפְּנִינִים יְקָרָה


שֶׁבֶת גַּבְנוּן אִוִּיתָה לְשַׁלְּשִׁי

שֶׁנִּגְעֲלוּ רוֹצְדִים וְחֶמֶד רָבַץ שָׁרְשִׁי

תֹּקֶף כְּהֻנָּה וּמַמְלֶכֶת לְהוֹרִישִׁי

תַּלְאוּבֹת בִּקְדֻשָּׁה שִׁכַּנְתַנִי בַּשְּׁלִישִׁי


בְּשָׁלוֹם / יִפְרַח כֵּס נְגִידִי

בְּהוֹד תִּפְאֶרֶת לְאָפְדִי רְדִידִי

בִּבְרִית חֲדָשָׁה נֵצַח לְעֹדְדִי

וְאַתָּה יְיָ מָגֵן בַּעֲדִי


תַּמָּה מִפִּיתוֹם הָאָרְשָׂה בֶּאֱמוּנָה

שִׁבּוּץ מְשַׂחֶקֶת קְנוֹת מַתָּנָה

רוֹגְלָהּ שָׁרָה מֵרֹאשׁ אֲמָנָה

קוֹבֵץ בַּאֲבוּקוֹת לְשָׁתְלָהּ כַּנָּה


צְבָאֶיהָ רִבֵּב כַּצֶּמַח בִּתְקוּמוֹת

פְּאוּרִים אֲזוּרִים חֵשֶׁק עֲלָמוֹת

עִלְּסָם בְּדִקְדּוּקֵי עוֹז וְתַעֲצוּמוֹת

סֵפֶר כְּתֻבַּת חַיֵּי עוֹלָמוֹת


נוֹבְבָה בְּיֵין מְסִיכָה לְהִתְיַפּוֹת

מִתְרַפֶּקֶת עַל מוֹצִיאָהּ כְּתוֹעֲפוֹת

לִמְּדָהּ מְאַמֶּצֶת יָדַיִם רָפוֹת

כַּלָּה לְקַשֵּׁט בְּזִיּוּן מְעוֹפְפוֹת


יָאֲתָה בְּמַצַּב קֶשֶׁב דִּבְּרוֹת

טַעַם עֲשִׂיָּה וּשְׁמִיעָה עֲגִילֶיהָ מַזְהִירוֹת

חֲמוּקֶיהָ חֻסְּנוּ מִשְּׁתֵּי שְׁתֵּי עֲטָרוֹת

זִקִּים וּבְרָקִים פְּרָחֶיהָ מַכְתִּירוֹת


וְרִקְמָה וְתַחַשׁ בְּתִלְבֹּשֶׁת נִתְכַּלְּלָה

הַשְּׁכִינָה בַּחֲנוֹת בָהּ בִּתְהִלָּה

דַּהֲרַת רָבִיד פֵּיאֲרוּהָ מַחֲנֶה הֲמֻלָּה

גַּבְנוּן בַּחֲנוֹתָהּ בְּרִנָּה וְצָהֲלָה


בְּקוֹלוֹת שִׁבְעָה מָטוּ יְסוֹדִים

בָּהִיר כְּהִשְׁמִיעַ לִמְכּוֹר אֲגוּדִים

אוֹר הִבְהִיק מַרְאֵה לַפִּידִים

אֲיֻמָּה בְּנָסְעָהּ מִסִּינַי לִרְפִידִים


עֵת / אֲרוּכָה הִרְגִּיל עֲמוּסָיו מִמַּעַל

תְּעָלָה לִתְלָאָה בִּרְטִיָּה הַעַל

בְּטַל מִבָּשָׁן יָשִׁיב וּמִמְּצוּלוֹת יַעַל

יְיָ מֵמִית וּמְחַיֶּה מוֹרִיד שְׁאוֹל וַיָּעַל


אֲבִי גְדוֹר כְּפָגַשׁ עַרְפִלֵּי מְרוֹמִים

תּוּשִׁיָּה שִׁילוֹל שִׁקּוּי לְחוּג הֲדוֹמִים

בֵּינְתַיִם צָגוּ אֵלִים נוֹהֲמִים

שַׁעֲשׁוּעַ מִשְּׁבוֹת קְצַר יָמִים


גַּחַלְתָּם הִבְעִירוּ וְחִצֵּי זִקִּים

רָשׁוּף רֻפַּק בַּעֲדָם עַד שְׁחָקִים

דָּאַג בְּשׁוּרוֹ אֵימוֹת בְּרָקִים

קוֹשִׁי לֶהָבָה לְמוּלוֹ מַבְרִיקִים


הִמָּלֵט בִּזְרוֹעַ קוֹל לוֹ צָרַח

צוֹעֶה בְכֹחַ כְּעֵין הַקֶּרַח

וּבְפִרְשׁוּז עָנָן בְּמַחְסֶה נִתְאָרַח

פָּתַח מַעֲרָכָיו וּמַעֲנוֹ זָרַח


זְמַן שִׁלּוּחַ אֶזְכּוֹר לְבִלְתִּי תינשה

עֲנִיַּת הננו מֵחֶלֶד רוֹגְשִׁי

חֶמְאָה הוֹבִיל בְּמַצַּג תּוֹם יְשִׁישִׁי

סְעוּדָה עֲרֹךְ בְּשַׁעַן רַעֲנָן דִּשְׁאִי


טוֹבַת בְּעָלָיו מִדָּה בְּמִדָּה לְשַׁלְּמָה

נְתִיבוֹ שֶׁל עוֹלָם סַעַד לְקַדְּמָה

יְקוֹד פַּעֲמֵיהֶם גְּבוֹהוֹת נְאוֹמָה

מִלּוּל שֶׁלֹּא כַדָּת עֲרוֹךְ מִלְחָמָה


כֶּתֶר תּוֹרָה לָאָדָם מֵאָז הִכְלְלָה

כְּמָסָהּ אֶצְלוֹ נוֹרָא עֲלִילָה

לִוְיַת חֵן לְסַגֵּל מְאֹד נַעֲלָה

לַמַּלְאָכִים הִפְצִיחַ צִיר וְלִמְחִיצָה עָלָה


מִבְטֶיחָהּ / מֵאַבִּירִים שָׁבָה נֶאֱמָן

חוֹבֶשׁ לְדִגְלֵי עַם לֹא אַלְמָן

כְּמֵרֵאשִׁית יֵאתֵא בַחֲבוּרָה לְהִתְיַמַּן

עֵת יָבוֹא אֱלוֹהַּ מִתֵּימָן


מִדָּבָר אֲשֶׁר כְּפַטִּישׁ נְשִׁיָּה נִתְפּוֹצְצָה

אֲגֻדָּה תַּחְתִּית צִין עֵת נִקְמְצָה

הָמוּ פְלִיאֵי שֵׁמוֹת בְּכָל מְחִצָּה

כְּהֵן אֶרְאֶלָּם צָעֲקוּ חוּצָה

עַלְמָה בַת גַּלִּים נָחֲלָה מוֹעֵצָה

הַשְּׂבֵיעֵי רֹגֶז כָּלָה וְנֶחְרָצָה

וֶהֱשִׁיבָם נְכוֹחוֹת גְּדוֹל הָעֵצָה

אָמוֹן מִקֶּדֶם חֲרוּטָה בְּיָפָה כְתִרְצָה

גִּזְעֵי נִצְרַף וּלְנִיחוֹחַ נִרְצָה

וְנֶכְדֵי גָלוּם בְּכֵס מִרְכֶּבֶת אֲמֻצָּה

הִנֵּה הָעֵת בְּלֶקַח כְּלוּלוֹת לְהִתְאַמְּצָה

אֲשֶׁר בַּמַּחֲנֶה שכך בְּלִי תַחֲרוּת וּמַצָּה

בַּעֲשֶׂרֶת הַדִּבְּרוֹת לְהִפָּרַע מִשִּׁמְצָה

בְּדִקְדּוּקֵי אִסּוּר וְהֶתֵּר לְהַשִּׂיג דִּיצָה

בְּחִידוֹתֵיהֶם לְהָבִין מָשָׁל וּמְלִיצָה

וּמֵחִל קוֹלוֹ נָס לְאָחוֹר בִּצָּה

וְעֵת קָשְׁבוּ גְדוּדִים הֶגֶה מִפִּיו יָצָא

הֶאְדִּירוּ קְדֻשָּׁה וַעֲרִיצָה

קָדוֹשׁ לְפָאֵר וְחַי לְהַעֲרִיצָה

וּמְשַׂחֶקֶת פַּלְגֵי מַעְיְנוֹתֶיהָ הֵפִיצָה

בְּשַׁעֲשׁוּעֵי נוֹתְנָהּ לָעָם אֲשֶׁר נִתְרַצָּה

קָדוֹשׁ
