אַאֲמִיר אַאְדִּיר עֲשָׂרָה

וּבְכֵן פִּזְמוֹן הַלְלוּ אֶת שֵׁם יְיָ


אַאֲמִיר אַאְדִּיר עֲשָׂרָה שֵׁמוֹת בְּמֶרֶץ

אֲבָאֵר פְּלִיאַת כָּל אֶחָד וְאֶחָד בְּתֶרֶץ

אֶשְׁאַל רִשְׁיוֹן מָרוֹם שׁוֹכְנֵי עֶרֶץ

אֲיַסֵּד עֹז קוֹנֵה שָׁמַיִם וָאָרֶץ אוֹדֶה שִׁמְךָ יְיָ


אֲבָאֵר אֲבָרֵר הַשֵּׁם הָרִאשׁוֹן לְהָאִיר עֵינָיִם

בָּאֵר אֲבָרֵר יָהּ לְהַשְׁמָעוּת אָזְנָיִם

בְּאֶחָד נִקְרָא וְהוּא מִתְחַלֵּק לִשְׁנָיִם

בּוֹ זִמֵּר מְשׁוֹרֵר מְחוֹלַת מַחֲנָיִם


אֲנַצַּח אֲנַשֵּׂא הַשֵּׁם הַשֵּׁנִי בְּמוֹרָאוֹת

נַזְכִּיר יְיָ צְבָאוֹת שֶׁהוּא בְצִבְאוֹתָיו אוֹת

נֶגְדּוֹ הִטִּיף חוֹזֶה בְּמַרְאוֹת נְבוּאוֹת

נְאוּם הָאָדוֹן יְיָ צְבָאוֹת מְהֻלָּל שֵׁם יְיָ


אַגְדִּיל אַגְבִּיר הַשֵּׁם הַשְּׁלִישִׁי בְּעִתּוֹת כָּאֵל

גְּדֻלַּת אֱלֹהֵי יִשְׂרָאֵל לְהַזְכִּיר אוֹאֵל

גַּם שָׂר בִּפְנִיאֵל כְּבָנָה אֲרִיאֵל

גָּשׁ וַיִּקְרָא לוֹ אֵל אֱלֹהֵי יִשְׂרָאֵל לְהַגִּיד כִּי יָשָׁר יְיָ


אֲבָאֵר אָבִין הַשֵּׁם הָרְבִיעִי הַנֶּעְלָם

בְּסוֹד אֱלֹהִים חַיִּים יָרֹן כָּל נֶאֱלָם

בְּכֵן נָם חוֹזֶה הוּא אֱלֹהִים חַיִּים וּמֶלֶךְ עוֹלָם

בּוֹ נִשְׁבַּע אֵל וַיִּגְזֹר מַאֲמָר וַיִּשְׁלָם חַי אָנִי נְאוּם יְיָ


אֲיַקֵּר אֲיַסֵּד הַשֵּׁם הַחֲמִישִׁי פַּעֲמָיִם

יִפְעַת אֵל הַכָּבוֹד הִרְעִים עַל מָיִם

יְעָדוֹ מְחוֹקֵק דָּת אֲצוּלַת יוֹמָיִם

יִסַּד אֵין כָּאֵל יְשׁוּרוּן רוֹכֵב שָׁמַיִם כִּי אֵל גָּדוֹל יְיָ


אֲרַחֵשׁ אֲרוֹמֵם הַשֵּׁם הַשִּׁשִּׁי בְּפֶלֶץ

רוֹמְמוּת שַׁדַּי אֶפְרֹשׁ וְאֶחְגֹּר חֶלֶץ

רָאוּי דֵּי עֶרְכּוֹ לְמוֹצְאוֹ בְעֶלֶץ

רָז תַּעֲלוּמוֹ נִגְלָה לְאֵיתָן בְּמֶלֶץ בְּאֵל שַׁדַּי וּשְׁמִי יְיָ


אוֹדֶה אוֹקִיר הַשֵּׁם הַשְּׁבִיעִי הַנִּדְגָּל

וְשֵׁם רָם אֲבָאֵר בְּסוֹד עִם הַנִּסְגָּל

וְסוֹדוֹ שֶׁהוּא יוֹשֵׁב בְּרוּם הַגַּלְגַּל

וְעַל כֵּן זִמֵּר מְשׁוֹרֵר מְפַלֵּס מַעְגָּל רָם עַל כָּל גּוֹיִים יְיָ


אֲלַהֵג אֲלַמֵּד הַשֵּׁם הַשְּׁמִינִי לְאֹם עֲמוּסָה

לַחַשׁ נִשָּׂא אֲפַעְנֵחַ בְּמִלָּה פְרוּשָׂה

לְהוֹדִיעַ כִּי הוּא נוֹשֵׂא זְרוּתָה וְקַצְוֵי פְלוּסָה

לָכֵן נֶאֱמַר כִּי כֹה אָמַר רָם וְנִשָּׂא אַדִּיר בַּמָּרוֹם יְיָ


אֲחַסֵּן אֲחַקֵּר הַשֵּׁם הַתְּשִׁיעִי לַהֲרִימוֹ

חֲסִין שׁוֹכֵן עַד נִקְרָא בְתַעֲצוּמוֹ

חֵלֶף כִּי לָנֶצַח יִשְׁכֹּן מְרוֹמוֹ

חוֹזֶה עַל כֵּן הִטִּיף שׁוֹכֵן עַד וְקָדוֹשׁ שְׁמוֹ וְאַתָּה מָרוֹם לְעוֹלָם יְיָ


אֲזַמֵּר אֶזְכֹּר הַשֵּׁם הָעֲשִׂירִי וְאֶתְאוֹשֵׁשׁ

זִמְרַת קָדוֹשׁ אֲצַפְצֵף וְלֹא אֶתְבּוֹשֵׁשׁ

זֶה כִּי הוּא קָדוֹשׁ וּמְשָׁרְתָּיו קְדוֹשִׁים לְהִתְקוֹשֵׁשׁ

קְדֻשָּׁתוֹ בְכָל יוֹם יְשַׁלְּשׁוּ שַׂרְפֵי שֵׁשׁ קָדושׁ קָדוֹשׁ קָדוֹשׁ יְיָ.
