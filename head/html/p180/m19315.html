עִנְיַן חֲמִישִׁית רַק לְשָׂפָה נֽוֹדְעָה / וּבְאַחֲרִית כֻּלָּֽהְנָה נִגְמֶרֶת

פִּתְרוֹן פְּקוּדֵיהֶם יְסוֹד אַהְחַ“ע לְנִיב / גָּרוֹן יְסוֹד גִּיכָ”ק לְחֵק צֹרֶרֶת

צוּרָה שְׁלִישִׁית דַּטְלְנַ"ת הִיא נִקְרְאָה / לַחֵךְ וְלַלָּשׁוֹן אֱמֶת נִסְפֶּרֶת

קֶצֶב יְצִיאֵי שֵׁן חֲקוּקִים זַ“ץ שְׁשַׂרְ”ס / וּמְקוֹם יְסוֹד בּוּמָ"ף שְׂפַת נִסְגֶּרֶת

רַק נִשְׁלְמוּ חָמֵשׁ וְעוֹדָם נֶחְלְקוּ / לִשְׁנֵי חֲלָקִים עַל פְּאַת אַחֶרֶת

שֶׁבָּם נְטוּעִים כַּמְּלָכִים נִצְּבוּ / וּמְשָֽׁרְתִים עֹבְרִים בְּלִי מַעְבֶּרֶת

תַּתִּים אֲגֻדָּה וָאֲשִׂימֵם עַל שְׁמִי / לִהְיוֹת לְכָל מַשְׂכִּיל לְאֹות מִשְׁמֶרֶת

שֵׁבֶט מְשָׁרְתִים הוּא אֲנִי שְׁלֹמֹה וְגַם / כּוֹתֵב מְקֻשָּׁרִים בְּתוֹךְ מָסֹרֶת

לַהְקַת מְלָכִים רַק תְּהִי קָט צַח וְגַם / גֶּזַע סְפָרַד בַּעֲבוֹת נִקְשֶׁרֶת

מָקוֹם מְשָׁרְתִים יַחֲלִיפוּן אִישׁ וְאִישׁ / עַל פִּי עֲבֹדָתָם אֲשֶׁר נִסְדֶּרֶת

הִנְנִי אֲבָאֵר כָּל עֲבֹדָתָם הֲלֹם / עַל מִשְׁמְרוֹת כָּל אוֹת וְאוֹת נֶעְדֶּרֶת

אֵיךְ מִשְׁמְרוֹת אָלֶף אֲשֶׁר בָּם תַּעֲבֹד / תָּמִיד בְּמִשְׁמַרְתָּה תְּהִי נֶאְסֶרֶת

בְּאֶחָד לְאַרְבָּעָה עֲתִידִים תַּעֲבֹד / לַשֵּׁם וְלַפֹּעַל תְּהִי נִמְסֶרֶת

גַּם תַּעֲבֹד פֹּעַל אֲשֶׁר הָיָה כְבָר / אָלֶף בְּאֶתְחַבַּר לְהֵ"א נִפְתֶּרֶת

דָּתָהּ לְהוֹסִיפָהּ בְּשֵׁם פֹּעַל כְּמוֹ / אָדוֹשׁ יְדוּשֶׁנּוּ וּבוֹ נֶעְצֶרֶת

הֵן נִסְפְּחָה גַּם עַל יְסוֹד הַשֵּׁם כְּמוֹ / אֶתְמוֹל וְאֶזְרוֹעַ הֱיוֹת נִבְרֶרֶת

וּבְכֵן אֲבָאֵר מַחְלְקוֹת הַנּוּן לְכָל / פָּנִים אֲשֶׁר בָּם נִמְצְאָה נִפְזֶרֶת

זֹאת גַּם לְאֶחָד מֵעֲתִידִים תַּעֲבֹד / תָּמִיד עֲבוּר לִהְיוֹת לְאוֹת לַעְצֶרֶת

חָקוּק לְהֶעָשׂוּי כְּבָר גַּם לַאֲשֶׁר / עַתָּה כְּמוֹ נֶעְכַּר וְגַם נֶעְכֶּרֶת

טָפוֹל לְוָו רַבִּים בְּפֹעַל שֶׁכְּבָר / גַּם בַּאֲשֶׁר לִהְיוֹת תְּהִי נִכְתֶּרֶת

יְדוּעָה בְנוּן צָקוּן וְנוּן וַיַּחֲנוּן / עֵת תַּעֲמֹד בָּהֶן וְעֵת נוֹסֶרֶת

כֵּן נֽוֹסְפָה בִּלְשׁוֹן נְקֵבָה כִּי תְהִי / אַחַת וְהִיא כָל עֵת וְעֵת מוֹסֶרֶת

לִקְהַל נְקֵבוֹת נָתְנָה חֹק וַתְּהִי / בַשֵּׁם וּבַפֹּעַל לְוָו נִכֶּרֶת

מֵאַחֲרֵי הַיּוֹד וְהַתָּיו תַּעֲבֹד / וּבְמַאֲמָר גַּם הִיא לְחֹק נִגְזֶרֶת

נוּן נִתְּנָה לַעְבֹד וְלִבְנוֹת בָּהּ בְּשֵׁם / רַבִּים וְשֵׁם אֶחָד בְּלִי אַחֶרֶת

שׂוּמָה תְּמוּרַת מֵם וְנוֹסֶפֶת בְּתוֹךְ / מִלִּים כְּנוּן מִנְלָם כְּנוּן נִפְאֶרֶת…
