שִׁנְאֲנֵי מְרוֹמִים

שִׁנְאֲנֵי מְרוֹמִים / אַדְרְךָ יִרְאוּ

וְלַהֲקַת הֲדוֹמִים / בְּשִׁמְךָ יִקְרְאוּ

מֵאָז הַבֹּקֶר / גָּשִׁים יִתְרָאוּ

יְהַלְלוּ אֶת שֵׁם יְיָ / כִּי הוּא צִוָּה וְנִבְרָאוּ


שִׁנְאַנֵּי מְרוֹמִים / דָּרִים בְּרוּם תֶּקֶר

וְלַהֲקַת הֲדוֹמִים / הוֹגֵי שִׁיר הַבֹּקֶר

מֵאָז הַבֹּקֶר / וְעֶרֶב וָבֹקֶר

גָּדוֹל יְיָ וּמְהֻלָּל מְאֹד / וְלִגְדֻלָּתוֹ אֵין חֵקֶר


שִׁנְאַנֵּי מְרוֹמִים / זִקֵּי הִלָּתֶךָ

וְלַהֲקַת הֲדוֹמִים / חֲשׁוּבִים סְגֻלָּתֶךָ

מֵאָז הַבֹּקֶר / טוֹעֲמִים גְּדֻלָּתֶךָ

וַאֲנִי תָּמִיד אֲיַחֵל / וְהוֹסַפְתִּי עַל כָּל תְּהִלָּתֶךָ


שִׁנְאַנֵּי מְרוֹמִים / יָד לְיָד תְּסַלְסְלֶךָ

וְלַהֲקַת הֲדוֹמִים / כְּמוֹ כֵן תְּצַלְצְלֶךָ

מֵאָז הַבֹּקֶר / לָעַד אַגְדִּילֶךָ

אֲסַפְּרָה שִׁמְךָ לְאֶחָי / בְּתוֹךְ קָהָל אֲהַלְלֶךָ


שִׁנְאַנֵּי מְרוֹמִים / מַנְעִימִים שִׁיר בְּקוֹל צְעָדָה

וְלַהֲקַת הֲדוֹמִים / נִצָּבִים בִּרְעָדָה

מֵאָז הַבֹּקֶר / שׂוֹחֲחִים בְּקוֹל חֲרָדָה

אֲהַלְלָה שֵׁם אֱלֹהִים בְּשִׁיר / וַאֲגַדְּלֶנּוּ בְתוֹדָה


שִׁנְאַנֵּי מְרוֹמִים / עוֹנִים אַיֵּה כְבוֹדוֹ

וְלַהֲקַת הֲדוֹמִים / פּוֹצְחלִים יִחוּדוֹ

מֵאָז הַבֹּקֶר / צוֹעֲקִים לְנֶגְדּוֹ

יְהַלְלוּ אֶת שֵׁם יְיָ / כִּי נִשְׂגָּב שְׁמוֹ לְבַדּוֹ


שִׁנְאַנֵּי מְרוֹמִים / קַרְנֵי זִיו תְּהִלָּה

וְלַהֲקַת הֲדוֹמִים / רְצוּפִים בְּרֹב קְהִלָּה

מֵאָז הַבֹּקֶר / שְׁמוֹ יְהַלְלוּ בְּמִלָּה

מְרוֹמֵם / עַל כָּל בְרָכָה וּתְהִלָּה


שִׁנְאַנֵּי מְרוֹמִים / יָד לְיָד תְּסַלְסְלֶךָ

וְלַהֲקַת הֲדוֹמִים / כְּמוֹ כֵן תְּצַלְצְלֶךָ

מֵאָז הַבֹּקֶר / לָעַד אַגְדִּילֶךָ

אֲסַפְּרָה שִׁמְךָ לְאֶחָי / בְּתוֹךְ קָהָל אֲהַלְלֶךָ


שִׁנְאַנֵּי מְרוֹמִים / תִּפְאַרְתּוֹ מְעִידִים

וְלַהֲקַת הֲדוֹמִים / שְׁבָחָיו מַגִּידִים

מֵאָז הַבֹּקֶר / סִלְסוּלוֹ מְסַלְּדִים

שִׁירוּ לַיְיָ שִׁיר חָדָשׁ / תְּהִלּ7תוֹ בִּקְהַל חֲסִידִים.
