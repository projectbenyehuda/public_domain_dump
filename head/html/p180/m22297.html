יְגוֹן חֵשֶׁק

ולו בתהלה


יְגוֹן חֵשֶׁק וְאַהֲבַת הַנְּעוּרִים / עֲזָבוּנִי בְלִי לֵב בַּיְּצוּרִים

וְעֵינַי נִפְתְּחוּ מֵאֵין שְׁנָתָם / לְדַלְתֵי הָאֲהָבִים הַסְּגוּרִים

וְשָׂם אֵל כּוֹכְבָיו עִמִּי, כְּאִלּוּ / אֲנִי רוֹעֶה וְהֵמָּה הָעֲדָרִים

וְנִסַּנִי כְּאַבְרָם הַמְּנֻסֶה / בְּאָמְרוֹ: קוּם סְפֹר שֶׁאֵין סְפוּרִים

רְאוֹת אֵלֶּה וְכָאֵלֶּה, יְדִידִי, / יְבִיאוּנִי בְרָעָה לַקְּבָרִים

צְרָרוּנִי, הֱצִיקוּנִי יְגוֹנִים / וְאַךְ שֶׁבַע צְרָפוּנִי בְכוּרִים

וְהַיּוֹם, שָׁת שְׂפָתָיו צוּף לְחִכִּי, / הֲלֹא נֶהְפַּךְ וְהָיָה מַמְּרוֹרִים

וְחִצֵּי הַזְּמַן - שִׁירַי, וְאִלּוּ / אֲהֵבַנִי - עֲשִׂיתִים לוֹ כְתָרִים

וְהֵם הָגוּת בְּפִי כָל הַחֲכָמִים / וּבִלְשׁוֹן כָּל מְתֵי מוּסָר שְׁמוּרִים

וְרֹאשָׁם יִהְיֶה שָׁמוּר לְעוֹלָם / וְיִפָּדֶה בְכָל שׂוֹטְנִים וְעָרִים

אֲשֶׁר בָּחַר בְּשֵׁם טוֹב מִשְּׁמָנִים / וְטוֹב מִכָּל בְּדֹלָחִים וְדָרִים

אֲשֶׁר נִתַּן לְכֶתֶר הַמְּלָכִים / וְלַעְטֶרֶת, וְנֵזֶר מִנְּזָרִים

מְבַלֶּה שׂוֹטְנֵי שֵׁרִית יְשׁוּרוּן / וְהַמִּתְיָעֲצִים אֵינָם אֲזוּרִים

וְלוּ מָרוּ וְלֹא אָבוּ שְׁמֹעַ / לְמוֹעְצוֹתָיו - אֲכָלָם כַּכְּפִירִים

הֲלֹא יֵחָפְזוּ מִגַּעֲרָתוֹ / וְגַם יִבָּהֲלוּ זֵדִים אֲרוּרִים

אֱדוֹם וּבְנֵי צְפוֹ צוֹפִים לְדַעְתּוֹ / וְקֵדָר שׁוֹחֲחִים אֵלָיו וְקוֹדְרִים

יְלִיד מִשְׂרָה וּבֶן יִתְרָה, וְאָמוּן / בְּחֵיקִי הַסְּגָנִים הַגְּבִירִים

נְדִיב רוּחַ, אֲשֶׁר צִוָּה עֲנָנָיו / לְבַל יִהְיוּ לְעוֹלָמִים עֲצוּרִים

רְצוּץ מַתָּן וּמַזְהִיר כֹּל לְתִתֵּן / לְכָל שׁוֹאֵל, וְאִם אֵינָם זְהִירִים

וְדִינֵי אֵל וְחֻקּוֹתָיו וְדָתָיו / עֲלֵי פִיהוּ עֲרוּכִים גַּם שְׁמוּרִים

אֲשֶׁר הֵמָּה כְתוּבִים עַל לְבָבוֹ / וּבֵין עֵינָיו לְטוֹטָפוֹת קְשׁוּרִים

אֲשֶׁר נִיבָיו יְקוֹדִים עֵת יְנִיבֵם / וְאֵינָם אֵשׁ אֲבָל כָּאֵשׁ בְּעוּרִים

אֲשֶׁר לָבַשׁ כְּמוֹ שִׁרְיָן צְדָקָה / וְשָׂם כָּל מִשְׁפְּטֵי עַמּוֹ יְשָׁרִים

אֲשֶׁר אִם נָע שְׂפָתָיו קָט לְדַבֵּר - / לְאַט, מִתּוֹךְ שְׂפָתָיו קַח דְּבָרִים

כְּפִיר יַצִּיל טְרֵפַת עַם יְשׁוּרוּן / וְיָדֵק בַּחֲמָתוֹ הַכְּפִירִים

וְכָל אוֹבֵד וְכָל נִדָּח מְבַקֵּשׁ / וְחוֹבֵשׁ בַּחֲסָדָיו הַשְּׁבוּרִים

וּמֵסִיר מֵעֲלֵיהֶם שֶׁן בְּהֵמוֹת / וְחַיּוֹת עִם חֲמַת זוֹחְלֵי עֲפָרִים

וּמָה אַגִּיד בְּתוֹרוֹ הַמְּעֻלֶּה / וּמָה אֹמַר - וְרַעְיוֹנַי קְצֵרִים

בְּשָׁכְנוֹ מַעֲלוֹת שֶׁמֶשׁ, וְהָאֵל / הֱבִיאוֹ מִלְּפָנִים בַּחֲדָרִים

וְאוּלָם מִפְלְאֵי הַשִּׁיר יְעִירוּן / מְזִמּוֹתַי וְנוֹקָצִים וְעֵרִים

לְקַבֵּץ מִגְּדוּדָיו הַפְּזוּרִים / וּמֵהַשִּׁיר לְגַלּוֹת הַסְּתָרִים

וְהַשִּׁיר גִּדְּלַנִי מִנְּעוּרַי / כְּמוֹ אָב וַאֲנִי לוֹ בֵּן בְּכוֹרִים

וְקָרַן עוֹר פְּנֵי שִׁירִי, כְּאִלּוּ / עֲלֵיהֶם מִתְּהִלָּתָךְ אֲפֵרִים

וְאֵלָיו יִחֲלוּ כָל הַמְּשׁוֹרְרִים / וּפִיהֶם פָּעֲרוּ לַטַּל פְּעוּרִים

אֲשֶׁר לָנוּ בְּנַפְשׁוֹתָם אֲרָיוֹת / וְהִשְׁכִּימוּ לְפָנָיו כַּשְּׁוָרִים

וּמִמִּשְׂרָה וּמִכָּבוֹד קְנוּיִים / וְלַלַּעַג וְלַחֶרְפָּה מְכוּרִים

אֲחֵי מוּסָר, מְעַט קָט כַּתְּרוּ לִי / עֲדֵי יִנָּעֲרוּ מִנִּי נְעוּרִים

וְתִרְאוּ שִׁיר יְבַהֵל רַעֲיוֹנִים / אֲשֶׁר בָּתָּיו מְמֻלָּאִים בְּדָרִים

בְּזָהַָב וַחֲרוּזֵי הַבְּדֹלַח / בְּמִדְבָּרִים וְעִנְיָנִים יְקָרִים

יְשִׂימוּנִי בְעֵין אַנְשֵׁי זְמַנִּי / כְּמוֹ אַרְיֵה בְּעֵינֵי הַשְּׁוָרִים.
