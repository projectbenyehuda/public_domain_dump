סִפּוּר עָזִיז וַעֲזִיזָה

א. עָזִיז בְּבֵית הוֹרָיו.

וַיֹּאמֶר הַצָּעִיר: דַּע לְךָ, אֲדוֹנִי, כִּי אָבִי הָיָה אַחַד הַסּוֹחֲרִים הַגְּדוֹלִים, וּבֵן לֹא הָיָה לוֹ מִבַּלְעָדָי, וּבְבֵיתֵנוּ דוֹדַנְתִּי וַיְגַדְּלֶנָה אָבִי עִמִּי יַחַד בַּבַּיִת, כִּי אָבִיהָ מֵת וְלִפְנֵי מוֹתוֹ כָרַת בְּרִית עִם אָבִי כִּי יִקָּחֶנָּה לִי לְאִשָּׁה בְהַגִּיעַ לָהּ עֵת דּוֹדִים. וַנִּגְדַּל שְׁנֵינוּ יַחַד, אֶת-פָּנֶיהָ לֹא הִסְתִּירוּ מִמֶּנִי וּפָנַי לֹא הִסְתִּירוּ מִמֶּנָּה. וַיְהִי כִי גָדַלְנוּ וַיֹּאמֶר אָבִי אֶל אִמִּי: הַשָּנָה נִכְתֹּב אֶת-כְּתֻבַּת עָזִיז לַעֲזִיזָה. וְאַחַר הֵחֵל אָבִי לְהָכִין כָּל-צָרְכֵי הַחֲתֻנָּה. וְאָנֹכִֹי וּבַת דּוֹדִי יְשֵׁנִים יַחַד בְּמִטָּה אַחַת וְלֹא יָדַעְנוּ דָבָר. וַיְהִי כִּי הֵכִין אָבִי כָל-צָרְכֵי כְלוּלוֹתֵינוּ וְלֹא נִשְׁאַר דָּבָר בִּלְתִּי כְתִיבַת הַכְּתֻבָּה וּבוֹאִי אֶל בַּת דּוֹדִי, וַיַּחְפֹּץ אָבִי כִי תִכָּתֵב הַכְּתֻבָּה אַחֲרֵי תְפִלַּת יוֹם הַשִּׁשִּׁי. וַיֶּלֶךְ אֶל מַכָּרָיו הַסּוֹחֲרִים וְיֶתֶר רֵעָיו וַיּוֹדִיֵעם אֶת-הַדָּבָר, וַיַּזְמִינֵם לָבוֹא בַיּוֹם הַשִּׁשִּׁי אַחֲרֵי תְפִלַּת הַצָּהֳרַיִם, וְאִמִּי הָלְכָה אֶל יְדִידוֹתֶיהָ וּקְרוֹבוֹתֶיהָ וַתִּקְרָאֵן לָבוֹא אֶל בֵּיתֵנוּ. וַיְהִי בְּהַגִיעַ יוֹם הַשִּׁשִּׁי וַיִּרְחֲצוּ אֶת-הָאוּלָם אֲשֶׁר הֵכִינוּ לְמוֹשַׁב הַקְּרוּאִים, וַיִּשְׁטְפוּ אֶת-אַבְנֵי הַשַּׁיִשׁ, וַיִּפְרְשׂוּ אֶת-הַשְּׁטִיחִים, וְעַל הַקִּירֹוֹת פָּרְשׂוּ יְרִיעוֹת רְקוּמוֹת חוּטֵי זָהָב. וְאָבִי הָלַךְ וַיְצַו לַעֲשׂוֹת הַמַּמְתַּקִּים וּסְרִידוֹת הַמִּגְדָּנוֹת, לֹא נִשְׁאַר בִּלְתִּי אִם כְּתִיבַת הַכְּתֻבָּה. וַתִּשְׁלָחֵנִי אִמִּי אֶל הַמֶּרְחָץ, וְאָחֲרַי שָׁלְחָה חֲלִיפָה חֲדָשָׁה מִמֵּיטַב הַמַּלְבּושִׁים. וַיְהִי בְּצֵאתִי מֵהַמֶּרְחָץ וָאֶלְבַּשׁ אֶת-הַחֲלִיפָה הַמְּפֹאָרָה, וְהִיא מְבֻשָּׂמָה מְקֻטָּרָה. וַיְהִי בְּלָבְשִׁי אוֹתָה וַיָפוּץ רֵיחָהּ הַטּוֹב עַד לְמֵרָחוֹק. וָאֹמַר לָלֶכֶת אֶל הַמִּסְגֵּד לְהִתְפַּלֵּל. וְהִנֵּה נִזְכַּרְתִּי בְאַחַד יְדִידַי אֲשֶׁר שָׁכַחְתִּי לְהַזְמִינוֹ, וָאֹמַר בְּלִבִּי: הָבָה אֲחַפֵּשׂ אַחֲרָיו עַד בּוֹא עֵת הַתְּפִלָּה. וָאָבֹא אֶל סִמְטָה אֲשֶׁר לֹא עָבַרְתִּי בָהּ מֵעוֹלָם, וְאָנֹכִי מֵזִיעַ אַחֲרֵי רְחִיצָתִי בַּמֶּרְחָץ וְאַחֲרֵי לָבְשִׁי הַחֲלִיפָה הַחֲדָשָׁה, וַתְּהִי זֵעָתִי נוֹזֶלֶת וְרֵיחַ בְּשָׂמַי נוֹדֵף. וָאֵשֵׁב בְּרֹאש הַסִּמְטָה לָנוּחַ עַל אִצטְבָה, וָאֶפְרֹשׂ תַּחְתַּי מִטְפַּחַת רְקוּמָה אֲשֶׁר הָיְתָה אִתִּי. וַיֵחֱזַק עָלַי הַחֹם וַיָּזַע מִצְחִי, וַתְּהִי הַזֵּעָה נִגֶּרֶת עַל פָּנַי וְלְֹא יָכֹלְתִּי לְנַגְּבָהּ מֵעַל פָּנַי בּמִּטְפַּחַת כִּי הָיְתָה מִטְפַּחְתִּי פְרוּשָׂה תַחְתָּי. וָאֹמַר לָקַחַת כְּנֵף בִּגְדִי וּלְנַגֵּב פָּנַי, וְהִנֵּה מִטְפַּחַת לְבָנָה נוֹפֶלֶת עָלַי מִמָּעַל, וְהַמִּטְפַּחַת דַּקָּה מֵרוּחַ צַח שְׁפָיִים וּמַרְאֶהָ נֶחְמָד מִצֳּרִי הַחוֹלֶה. וָאֶקָּחֶנָּה בְיָדִי וָאָרִים רֹאשִׁי מַעְלָה לִרְאוֹת מֵאַיִן נָפְלָה הַמִּטְפַּחַת, וְהִנֵּה נִפְגַּשׁ מַבָּטִי בְמַבַּט בַּעֲלָת הַצְּבִי הַזֶּה הַנִּשְׁקֶפֶת מִבַּעַד לַחַלוֹן, לֹא רָאִיתִי יָפָה מִמֶּנָּה מִיָּמָי. וַיְהִי בִּרְאוֹתִי אוֹתָהּ וַתָּשֶׂם אֶצְבָּעָהּ בְּפִיהָ, וְאַחַר חִבְּרָה שְׁתַּיִם מֵאֶצְבְּעוֹתֶיהָ וַתְּשִׂימֵן עַל חָזֶהָ בֵין שָׁדֶיהָ, וְאַחַר הִכְנִיסָה רֹאשָׁהּ בְּחַלוֹנָהּ וַתִּסְגְּרֵהוּ וַתִּבְעַר בְּקִרְבִּי אֵשׁ הָאַהֲבָה וַתִּגְדַּל הַבְּעֵרָה, וָאֱאָנַח וָאֱהִי נָבוֹךְ וְלֹא שָׁמַעְתִּי אֵת אֲשֶׁר אָמְרָה וְלֹא הֲבִינוֹתִי אֶת אֲשֶׁר רָמְזָה. וָאַבִּיט אֶל הַחַלוֹן שֵׁנִית וְהִנֵּה הוֹא סָגוּר. וָאֲחַכֶּה עַד בּוֹא הַשֶּׁמֶשׁ, וְלֹא שָׁמַעְתִּי כָּל-קוֹל וְלֹא רָאִיתִי פְנֵי כָל-אִישׁ.

וַיְהִי כִּי נוֹאַשְׁתִּי מֵרְאוֹתָה וָאָקוּם מִמְקוֹמִי, וָאֶקַּח אֶת-הַמִּטְפַּחַת אִתִּי וָאֶפְתָּחֶנָּה וְהִנֵּה נָדַף מִמֶּנָּה רֵיחַ מֹר, וַיְעוֹרֵר בְּקִרְבִּי רוּחַ עֹנֵג וָנַחַת, וָאֱהִי בְעֵינַי כְּיוֹשֵׁב בְּגַן –הָעֵדֶן. וְאַחַר פֵּרַשְׁתִּי אֶת-הַמִּטְפַּחַת וְהִנֵּה נָפְלָה מִמֶּנָּהּ פִּסַּת נְיָר, וְהִיא מְקֻטֶּרֶת מֹר וּלְבוֹנָה, וּבָהּ כָּתוּב לֵאמֹר:

יוֹנִים תְּעוּפֶינָה כְעָב אֶל מַעֲרָב / אֶתְכֶן בְּחֵי הָאַהֲבָה הִשְׁבַּעְתִּי

לָשֵׂאת שְׁלוֹמִי אֶל יְדִיד הִרְחִיק נְדוֹד, / אַךְ בֵּין צְלָעַי אָהֳלוֹ תָקַעְתִּי,

וּתְסַפְּרוּ נָא לוֹ הֲכִי לִבִּי עֲלֵי / נוּדוֹ כְשַׁסַּע הַגְּדִי שִׁסַּעְתִּי,

יָחֹן וְרוּחַ חֵן יְהִי מֵעִיר עֲלֵי / נַפְשִׁי, וְאֶחְיֶה אַחֲרֵי גָוַעְתִּי.

וַיְהִי אַחֲרֵי קָרְאִי הַבָּתִים הָאֵלֶה וָאָעִיף עֵינַי שֵׁנִית בַּמִּטְפַּחַת וְהִנֵּה כְתוּבִים בְּאַחַת שְׂפָתֶיהָ הַטּוּרִים הָאֵלֶּה:

לָמָּה צְבִי תָצוּד בְּעֵינֶיךָ, / תַּהֲרגֹ בְּעַפְעָפֵּי אֱשׁוּנֶיךָ?

אֹמֶר הֲשִֹיבֵנִי, הֲבִינֵנִי, / מַה-זֶּה אֲשֶׁר הֵצִיץ בְּפָנֶיךָ?

כַּרְכֹּם וְקִנָּמוֹן וּמֵי וֶרֶד / נַעֲשׂוּ נְקֻדִּים עַל גְּרוֹנֶךָ.

אָנָּא תְּנָה לִי כוֹס אֲהָבֶיךָ / אַהֲבָה תְהִי בֵינִי וּבֵינֶיךָ.

וּבִשְׂפָתָהּ הַשְּׁנִיָּה כְתוּבִים הַבָּתִּים הָאֵלֶּה:

יְמוֹתַי בִּלְעֲדֵי עָפְרִי כְשַׂעֲרוֹ, / וְלֵילוֹת חָלְפוּ עִמּוֹ כְפָנָיו,

וְכָל-צוּף אַחֲרָיו בִּלְתּוֹ כְקִצְפוֹ, / וְרוֹש עִמּוֹ כְיֵין חִכּוֹ וְשִׁנָּיו.

וְנִהְיֵיתִי וְנֶחְלֵיתִי לְנוּדוֹ / וְדַלּוֹתִי עֲדֵי שַׁבְתִּי כְמָתְנָיו.

וַיְהִי בְּקָרְאִי כָל-בָּתֵּי הַשִּׁיר אֲשֶׁר בַּמִּטְפַּחַת וַתִּתְלַקַּח בְּלִבִּי לַהֶבֶת שַׁלְהֶבֶת, וְהַתְּשׁוּקָה לִהֲטָה בְקִרְבִּי כְאֵשׁ צָרֶבֶת. וָאֶקַּח אֶת הַמִּטְפַּחַת וְאֶת-פִּסַּת הַנְּיָר וָאֲבִיאָן אִתִּי הַבָּיְתָה, וְלֹא יָדַעְתִּי אֵיךְ אַשִּׂיג מַאֲוַיָּי.

וָאַגִּיעַ הַבַּיְתָּה אַחֲרֵי עֲבֹר הָאַשְׁמֹרֶת הָרִאשׁוֹנָה, וָאֵרֶא וְהִנֵּה דוֹדַנְתִּי יוֹשֶׁבֶת וֹבוֹכָה. וַיְהִי כִרְאוֹתָהּ אוֹתִי וַתִּמַח דִּמְעוֹתֶיהָ וַתִּגַּשׁ אֵלַי וַתַּפְשִׁיטֵנִי בְגָדַי, וַתִּשְׁאָלֵנִי עַל דְּבַר הֵעָדְרִי. וַתֹּאמֶר: הִנֵּה נֶאֶסְפוּ בְּבֵיתֵנוּ כָל-הָאֲנַשִׁים, הַשָּׂרִים וְהַגְּדוֹלִים, וְהַסּוֹחֲרִים וְהַחֹרִים, וּבָא הַקָֹדִי וְהָעֵדִים, וַיֹּאכְלוּ וַיִּשְׁתּוּ, וַיֵשְׁבוּ זְמָן רָב מְחַכִּים לְבוֹאֲךָ, לְמַעַן יִכְתְּבוּ אֶת-הַכְּתֻבָּה, וְכַאֲשֶׁר נוֹאֲשׁוּ מִבּוֹאֲֹךָ קָמוּ וַיֵּלְכוּ לְדַרְכָּם. וַתּוֹסֶף וַתֹּאמֶר: וְאָבִיךָ קָצַף קֶצֶף רַב וַיִּשָּׁבַע כִּי לֹא יִכְתֹּב כְּתֻבָּתֵנוּ עַד הַשָׁנָה הַבָּאָה, כִּי הוֹצִיא עַל הַתְּכוּנָה הַזֹּאת הוֹן רָב. וְאַחַר אָמְרָה אֵלַי: הֲלֹא תַגִּיד לִי, חֲבִיבִי, מַה-קָּרְךָ הַיּוֹם כִּי אֵחַרְתָּ לָבוֹא כָּכָה. וָאֹמַר אֵלֶיהָ: כָּזֶה וְכָזֶה קָרָנִי. וָאֲסַפֵּר לָהּ דְּבַר הַמִּטְפַּחַת וְכָל-אֲשֶׁר קְָרַאַנִי מֵרֹאש וְעַד סוֹף. וַתִּקַּח אֶת-פִּסַּת הַנְּיָר וְאֶת-הַמִּטְפַּחַת וַתִּקְרָא כָל-הַכָּתוּב בָּהֵן וַתֵּרַדְנָהּ עֵינֶיה דִמְעָה, וְאַחַר אָמְרָה אֵלַי: וּמַה אָמְרָה לְךָ? וּמָה רָמָזָה? וָאֹמַר: לֹא הִגִּידָה דָבָר רַק שָׂמָה אֶצְבָּעָהּ בְּפִיהָ, וְאַחַר חִבְּרָה שְׁתֵּי אֶצְבְּעוֹתֶיהָ וַתְּשִׂימֵן עַל חָזֵהָ וַתּוֹרֶה לָאָרֶץ, וְאַחַר סָגְרָה הַחַלוֹן וְלֹא יָסַפְתִּי לִרְאוֹתָהּ, רַק לִבִּי הָלַךְ אַחֲרֶיהָ שֶׁבִי. וָאֵשֵׁב עַד בּוֹא הַשֶּׁמֶּשׁ, יוֹשֵׁב וּמְצַפֶּה אוּלַי תַשְׁקִיף מֵהַחֲלּוֹן שֵׁנִית. אַךְ רָאִיתִי כִי לַשָּׁוְא צִפִּיתִי וָאֳקוּם וָאָבוֹא. זֶה סִפּוּרִי, וְעַתָּה הִנְנִי מְחַכֶּה לְעֶזְרָתֵךְ.

וַתָּרֶם רֹאשָׁהּ אֵלַי וַתֹּאמֶר: בֶּן-דּוֹדִי, לוּ בִקַּשְׁתָּ מִמֶּנִּי אֶת עֵינַי הוֹצֵאתִין לְךָ מִנִּקְרָתָן, וְאָנֹכִי בוֹא אָבוֹא לְעֶזְרַת שְׁנֵיכֶם, כִּי אָכֵן נַפְשָׁהּ חָשְׁקָה בְךָ כַּאֲשֶׁר חָשַׁקְתָּ אַתָּה בָהּ.

וָאֹמַר: הֲתֵדְעִי מַה-כִּוְּנָה בִרְמִיזוֹתֶיהָ? וַתֹּאמֶר: יָדָעְתִּי. אֶת-אֶצְבָּעָהּ שָׂמָה בְּפִיהָ לֵאמֹר: הִנְּךָ לִי כְרוּחִי בִגְוִיָּתִי, וּשְׁתֵּי הָאֶצְבָּעוֹת אֲשֶׁר שָׂמָה עַל חָזֶהָ בֵין שָׁדֶיהָ שְׁנֵי יָמִים הֵמָּה, לֵאמֹר: אַחֲרֵי עֲבֹר יוֹמַיִם שׁוֹב תָּשׁוּב אֵלֵינוּ. וְעַתָֹּה דַע לְךָ כִי אָכֵן אוֹהֶבֶת הִיא אוֹתְךָ, וּמִי יִתֵּן וְיָכֹלְתִּי לָצֵאת וְלָבוֹא כִּי עַתָּה אִחַדְתִּיכֶם וְהִסְתַּרְתִּיכֶם בְּצִילִּי.

וַיְהִי כְּשָׁמְעִי דְבָרֶיהָ וָאוֹדֶה לָהּ, וַאֹמַר לָהּ: אֲחַכֶּה יוֹמָיִם.

וָאֵשֵׁב יוֹמַיִם בְּבֵיתִי, לֹא יָצָאתִי וְלֹא בָאתִי, וְלֹא אָכַלְתִּי וְלֹא שָׁתִיתִי. וְאָשִׂים רֹאשִׁי בְחֵיק בַּת-דּוֹדִי, וְהִיא מְעוֹדֶדֶת רוּחִי וְאוֹמֶרֶת אֵלַי: אַמֵּץ רוּחֲךָ וְהֵיטֵב לִבְּךָ וּלְבַש בְּגָדֶיךָ וְהָלַכְתָּ אֵלֶיהָ בְּמוֹעֲדָהּ הַמְּיֻעָד. וְאַחַר קָמָה וַתַּחֲלֵף בְּגָדַי וַתְּקַטְּרֵנִי וַתְּבַשְּׂמֵנִי וָאֲחַזֵק רוּחיִ וָאֵצֵא.

וָאֵּלֵךְ עַד אֲשֶׁר הִגַּעְתִּי אֶל הַסִּמְטָה, וָאֵשֵׁב עַל הָאִצְטְבָה מֶשֶׁךְ שָׁעָה, וְהִנֵּה נִפְתַּח הַחַלוֹן, וָאַבִּיט בְּעֵינַי אֵלֶיהָ. וַיְהִי כַּאֲשֶׁר רְאִיתִיהָ וְהִנֵּה אִתָּה מַרְאָה וּמִטְפַּחַת אֲדֻמָּה, וַתַּפְשֵׁל שַׁרְוֻלֵּי זְרוֹעֶיהָ וַתְּפַשֵּׂק חֲמֵשׁ אֶצְבְּעוֹתֶיהָ, וַתַּךְ בָּהֶן וּבְכַפָּהּ עַל חָזֶהָ, וְאַחַר הֵרִימָה יָדָהּ וַתּוֹצֵא אֶת-הַמַּרְאָה דֶרֶךְ הַחַלּוֹן, וְאֶת-הַמִּטְפַּחַת הָאַדֻמָּה שִׁלְשְׁלָה מִן הַחַלּוֹן שָׁלֹש פְּעָמִים שַׁלְשֵׁל וְהַעֲלֵה, וְאַחַר כְּרָכַתָּהּ עַל יָדָהּ, וַתְּנַעֲנֵעַ בְּרֹאשָׁה, וְאַחַר סָגְרָה אֶת-הַחַלּוֹן וַתֵּעָלֵם וְלֹא דִבְּרָה אִתִּי אַף מִלָּה, וְאָנֹכִי נִדְהַמְתִּי וְלֹא יָדַעְתִּי מַה-כָּל הַרְמָזִים הָאֵלֶּה, וָאֶשֵׁב מַשְׁמִים עַד הָעֶרֶב, וְאַחַר שַׁבְתִּי הַבָּיְתָה, וָאֶמְצָא אֶת-בַּת דּוֹדִי מִתְיַפַּחַת וְעֵינֶיהָ יוֹרְדוֹת בַּבֶּכִי, וַתּוֹסֶף יָגוֹן עַל יְגוֹנִי, וָאֶפֹּל בִּקְצֵה הַחֶדֶר. וַתָּקָם דּוֹדַנְתִּי אֵלַי וַתַּפְשִׁיטֵנִי בְגָדַי, וְאֶת-פָּנַי נִגְּבָה, וְאַחַר שְׁאֵלַתְנִי לְקוֹרוֹתַי וַאֲסַפְּרֵן לָהּ, וַתֹּאמֶר: דּוֹדָנִי, חָמֵשׁ הָאֶצְבָּעוֹת אֲשֶׁר טָפְחָה בָהֶן עַל חָזֶהָ – חֲמִשָׁה יָמִים הֵם, לֵאמֹר: אַחֲרֵי עֲבֹר חֲמֵשֶׁת יָמִים שׁוּבָה אֵלַי, וְכָל-רְמִיזוֹתֶיהָ הָאֲחֵרוֹת זֶה פִתְרוֹנָן: שְׁבָה עַל דּוּכַן חֲנוּת הַצּוֹבֵעַ עַד אֲשֶׁר יְבוֹאֲךָ שְׁלִיחִי.

וַיְהִי כְּשָׁמְעִי דְבָרֶיהָ וַתִּבְעַר בְּקִרְבִּי כְאֵשׁ יוֹקֶדֶת, וָאֹמַר: חֵי הָאֱלֹהִים, בַּת דּוֹדִי, כִּי צָדַקְתְּ בְּפִתְרוֹנֵךְ, כִּי אָכֵן רָאִיתִי בַסִּמְטָה חֲנוּת צוֹבֵעַ יְהוּדִי. וְאַחַר שַׁבְתִּי וָאֶבְכֶּה. וַתֹּאמֶר בַּת דּוֹדִי: הִתְאוֹשֵׁשׁ וְאַל יִפֹּל לִבְּךָ, כִּי אָכֵן יֶשְׁנָם חוֹשְׁקִים הַשּוֹגִים בְּאַהֲבָתָם יָמִים וְשָׁנִים, וְאַתָּה רַק שָׁבוּעַ נֻסֵּיתָ בְּמַסָּה וַתֵּלֶא. וַתּוֹסֶף לְדַבֵּר עַל לֳבִּי, וַתָּבֵא לִי אֹכֵל וָאֶקַּח לְגִימָה וָאֹמַר לְאָכְלָהּ וְלֹא יָכֹלְתִּי, גַּם שָׁתֹה לֹא שָׁתִיתִי, וְשֵׁנָה לֹא רָאִיתִי בְעֵינַי. וַיֶּחְוְרוּ פָנַי וַיִּקָּדָּרוּ, כִּי לֹא חָשַׁקְתִּי מֵעוֹדִי וְעַד אָז, וָאֶחֱלַשׁ, וַתֶּחֱלַשׁ גַּם בַּת דּוֹדִי בִּגְלָלִי. וַתָּחֶל לְסַפֵּר לִי סִפּוּרֵי הַחוֹשְׁקִים לְשַׁעְשְׁעֵנִי בַלַּיְלָה עַד יָשְׁנִי. וּבְכָל-עֵת הֲקִיצִי רְאִיתִיהָ עֵרָה וְדִמְעוֹתֶיהָ עַל לְחָיֶיהָ. וָאֱהִי כָּכָה עַד עֲבֹר חֲמֵשֶׁת הַיָּמִים.

אָז קָמָה בַת דּוֹדִי וַתְּחַמֵּם לִי מַיִם, וַתַּלְבִּישֵׁנִי מַחֲלָצוֹת וַתֹּאמֶר אֵלַי: לְכָה אֵלֶיהָ וְהָאֱלֹהִים יַעַזְרֶךָ. וָאֵלֵךְ עַד אֲשֶׁר בָּאתִי אֶל רֹאש הַסִּמְטָה. וְהַיּוֹם יוֹם הַשַּׁבָּת, וַחֲנות הַצּוֹבֵעַ הַיְּהוּדִי נְעוּלָה. וָאֵשֵׁב שָׁם יוֹשֵׁב וּמְחַכֶּה עַד אֲשֶׁר קָרָא הַכָּרוֹז לִתְפִלַּת הַמִּנְחָה. וְאַחַר נָטְתָה הַשֶּׁמֶשׁ לַעֲרֹב וַיַּכְרִיזוּ לִתְפִלַּת הָעֶרֶב. וַיְהִי לָיְלָה, וְאָנֹכִי טֶרֶם רָאִיתִי דָבָר וְכָל-קוֹל לֹא שָׁמָעְתִּי, וָאִירָא לְנַפְשִׁי בְּשִׁבְתִּי שָׁם לְבַדִּי, וָאָקוּם וָאֶתְהַלֵּךְ וְלִבִּי כַשִׁכּוֹר עַד שׁוּבִי הַבָּיְתָה. וַיְהִי כְבוֹאִי וָאֵרֶא אֶת-דּוֹדַנְתִּי עֲזִיזָה וְיָדָה הָאַחַת אוֹחֶזֶת בְּיָתֵד תְּקוּעָה בַקִּיר וְיָדָהּ הַשְׁנִיָּה עַל חָזֶהָ, וְהִיא מִתְיַפַּחַת וּבוֹכָה, וַתֵּפֶן אֵלַי וַתִּמַּח דִּמְעוֹתֶיהָ וְדְמְעֹותַי, וַתְּחַיֵּךְ בְּפָנַי וַתֹּאמֶר אֵלַי: בָּרוּךְ אַתָּה לֵאלֹהִים, בֶּן-דּוֹדִי, אֲשֶׁר חֲנָנְךָ וַיִּתֶּן לְךָ מִשְׁאֶלֶת לִבֶּךָ, וְלָמָּה זֶה לֹא לַנְתָּ בְּבֵית אֲהוּבָתֶךָ? וַיְהִי כְּשָׁמְעִי דְבָרֶיהָ וָאֶבְעָטֶנָּה בְרַגְלִי וַתִּפֹּל אָרְצָה וּבָרִצְפָה יָתֵד וַתִּפְגַּע הַיָּתֵד בְּמִצְחָה וַיִזַּל הַדָּם. אַךְ הִיא הֶחֱרִישָׁה וְלֹא הוֹצִיאָה כָל הֶגֶה מִפִּיהָ. וְאַחַר קָמָה וַתִּשְׂרֹף חֲתִיכַת אֲרִיג וַתְּמַלֵא בוֹ אֶת-חֹר הַפֶּצַע וַתַּחֲבֹש תַּחְבֹּשֶׁת, וְאֶת-הַדָּם אֲשֶׁר נָזַל עַל הַשְּׁטִיחַ נִגְּבָה, וּבַת-צְחוֹק עַל שְׂפָתֶיהָ, וַתֹּאמֶר אֵלַי: חֵי הָאֵל, בֶּן-דּוֹדִי, כִּי לֹא חָשַׁבְתִּי לְהַרְעִימְךָ אוֹ לְהִתְעַלֵּל בָּךְ. וְעַתָּה הוֹדִיעֵנִי נָא מֶה הָיָה לְךָ הַיּוֹם. וָאֲסַפֵּר לָהּ כָּל-אֲשֶׁר קָרַנִי בַיּוֹם הַהוּא, וְאַחַר בָּכִיתִי. וַתֹּאמֶר אֵלַי: בֶּן-דּוֹדִי, הִתְבַּשֵׂר כִּי הַצְלֵחַ תַּצְלִיחַ וְתִקְוָתְךָ בוֹא תָבֹא, אַךְ הַיּוֹם נֶעֶלְמָה מִמְּךָ, כִּי אָמְרָה לְנַסּוֹתְךָ לָדַעַת אִם יָקוּם בְּךָ לִבְּךָ אוֹ לֹא, וְאִם הִנְךָ נֶאֱמָן בְּאַהֲבָתְךָ לָהּ אִם לֹא. וְעַתָּה חַכֵּה עַד יוֹם מָחָר, וְהָלַכְתָּ אֶל הַמָּקוֹם הַהוּא וְרָאִיתָ אֶת אֲשֶׁר תִּרְמֹז לְךָ הַפַּעַם, כִּי אָכֵן קָרְבָה עֵת שִׂמְחָתְךָ וְיִסּוּרֶיךָ יָסוּרוּ.

וַתְּהִי הִיא מַרְגִּיעָה רוּחִי וְלִבִּי נֶעְכָּר, וַתַּגֵּשׁ אֵלַי הָאֹכֶל וָאֶבְעַט בּוֹ בְרַגְלִי, וַתִּשָּׁפֵךְ כָּל-כּוֹס בְּפִנָּה אַחֶרֶת, וָאֶקְרָא: אָכֵן כָּל-נִלְכָּד בְּרֶשֶׁת הָאַהֲבָה הוּא כִמְשֻׁגָּע, וְלִבּוֹ לֹא יֵלֵךְ לֹא לַאֲכִילָה וְלֹא לְשֵׁנָה. וַתֹּאמֶר אֵלַי בַּת-דּוֹדִי: אָכֵן אֵלֶּה הֵם אוֹתוֹת הָאַהֲבָה. וַתִּזַּלְנָה דִמְעוֹתֶיהָ וּבְיָדָהּ אָסְפָה אֶת-שִׁבְרֵי הַזְּכוּכִית, וַתָּשָׁב לְשַׁעְשְׁעֵנִי וְאָנֹכִי מִתְפַּלֵּל לָאֵל כִּי יַעֲבֹר הַלָּיְלָה. וַיְהִי בְאוֹר הַבֹּקֶר וָאָרוּץ וָאָבוֹא אֶל הַסִּמְטָה מַהֵר וָאֵשֵׁב עַל הָאִצְטְבָה. וְהִנֵּה נִפְתַּח הַחַלּוֹן וְהַצְּעִירָה הוֹצִיאָה רֹאשָׁה וַתִּצְחַק, וְאַחַר נֶעֶלְמָה, וַתָּשָׁב וְאִתָּהּ מַרְאָה וְכִיס וְקַעֲרִית מְלֵאָה זֵרְעוֹנִים יְרֻקִּים, וּבְיָדָה הַשֵּׁנִית מְנוֹרָה. וַתִּקַּח אֶת-הַמַּרְאָה וַתְּשִׂימֶנָּה בְתוֹךְ הַכִּיס וַתִּקְשְׁרֵהוּ וַתַּשְׁלִיכֵהוּ הַחַדְרָה. וְאַחַר סָתְרָה מִקְלַעַת שְׂעָרָה עַל פָּנֶיהָ, וְאֶת-הַמְּנוֹרָה שָׂמָה עַל הַזֵּרְעוֹנִים, וְלָאַחֲרוֹנָה לָקְחָה אֶת-כָּל-הַדְּבָרִים וַתֵּלֶךְ וַתִּסְגֹּר אֶת-הַחַלּוֹן. וַתֶּחֱזַק אַהֲבָתִי וּתְשׁוּקָתִי, וָאָשׁוּב עַל עֲקֵבַי וְעֵינַי בּוֹכִיּוֹת וְלִבִּי כוֹאֵב. וָאָשׁוּב הַבַּיְתָה וְהִנֵּה בַת דּוֹדִי יוֹשֶׁבֶת וּפָנֶיהָ אֶל הַקִּיר וּבְלִבָּהּ כְּאֵשׁ צוֹרֶבֶת מִדַּאֲגָתָהּ וְקִנְאָתָהּ, אַךְ אַהֲבָתָהּ אֵלַי שָׂמָה מַחְסוֹם לְפִיהָ בִרְאוֹתָהּ אֶת-סִבְלִי. וָאַבִּיט אֵלֶיהָ וְהִנֵּה עַל רֹאשָׁהּ וְעַל פָּנֶיהָ שְׁתֵּי תַחְבְּשׁוֹת, הָאַחַת מֵעֵת נָפְלָהּ עַל מִצְחָהּ וְהַשֵּׁנִית עַל עֵינָהּ מִגֹּדֶל כְּאֵבָהּ וּמִבְּכִיָּתָהּ.

וַיְהִי כִּרְאוֹתָהּ אוֹתִי וַתִּמַּח דִּמְעוֹתֶיהָ וַתָּקָם אֵלַי, וַתֹּאמֶר לְדַבֵּר וְלֹא יָכְלָה כִּי נִכְמְרוּ נִחוּמֶיהָ. וְאַחַר הִתְאוֹשְׁשָׁה וַתֹּאמֶר: בֶּן דּוֹדִי, הוֹדִיעֵנִי אֵת אֲשֶׁר קָרְךָ בַּפַּעַם הַזֹּאת. וָאֲסַפֵּר לָהּ כָּל-אֲשֶׁר קָרָנִי. וַתֹּאמֶר: הִתְבַּשֵּׂר כִּי אָכֵן הִגִּיעָה כְבָר עֵת הַפְּגִישָׁה אִתָּהּ. הַמַּרְאָה אֲשֶׁר שָׂמָה בְתוֹךְ הַכִּיס אוֹמֶרֶת: חַכֵּה עַד בּוֹא הַשֶּׁמֶשׁ. וּבְהוֹרִידָהּ שְׂעַר רֹאשָׁהּ עַל פָּנֶיהָ אָמְרָה אֵלֶיךָ: בְּבֹוא הַלַּיְלָה וּשְׁחֹורוֹ פָרוּשׂ בּוֹא הֵנָּה, וּבְקַעֲרִית הַזֵּרְעוֹנִים אֲשֶׁר בְּיָדָהּ רָמְזָה לְךָ כִּי בְּבוֹאֲךָ תָבוֹא דֶרֶך הַגָּן אֲשֶׁר מֵאַחֲרֵי הַסִּמְטָה. וְהַמְּנוֹרָה אֲשֶׁר בְּיָדָהּ רָמְזָה לְךָ כִּי בְּבוֹאֲךָ אֶל הַגָּן תֵּלֵךְ עַד הַמָּקוֹם אֲשֶׁר בּוֹ תִמְצָא מְנוֹרָה דוֹלֶקֶת, וְיָשַׁבְתָּ שָׁם וְחִכִּיתָ לָהּ. וַיְהִי בְשָׁמְעִי אֶת-דִּבְרֵי דוֹדַנְתִּי וָאֶצְעַק וָאֶקְרָא: כַּמָּה פְעָמִים הִבְטַחְתִּינִי וְאָנֹכִי הָלַכְתִּי וְלֹא הִשַּׂגְתִּי חֶפְצִי, וּפִתְרוֹנֵךְ לֹא הָיָה פִתְרוֹן אֶמֶת. וַתִּצְחַק בַּת דּוֹדִי וַתֹּאמֶר: עָלֶיךָ לְהִתְאַזֵּר בְּסַבְלָנוּת רַק הַיּוֹם הַזֶּה עַד בּוֹא הַלַּיְלָה. וַתָּשָׁב לְנַחֲמֵנִי בִדְבָרֶיהָ הַטּוֹבִים, וְלֹא הֵעֵזָה לְהָבִיא לִי אֹכֶל פֶּן אֶקְצֹף עָלֶיהָ, וַתֹּאמֶר: דֹוֹדָנִי, שְׁבָה נָא אִתִּי וְשִׁעֲשַׁעְתִּיךָ עַד פְּנוֹת הַיּוֹם בְּשִׂיחָתִי וְאַחַר תֵּלֵך אֶל בֵּית אֲהוּבָתֶךָ אַךְ אָנֹכִי לֹא שָׁעִיתִי לִדְבָרֶיהָ וָאֵשֶׁב מַשְׁמִים מְחַכֶּה עַד בּוֹא הַלַּיְלָה. וָאֹמַר: אָנָּא הָאֵל, הָחִישָׁה אֶת-בּוֹא הָעֶרֶב.

וַיְהִי בְּבוֹא הַלַּיְלָה וַתִּתֶּן לִי גַרְגִּיר מֹשֶׁק טָהוֹר וַתֹּאמֶר אֵלַי: בֶּן דּוֹדִי, אֶת-גַּרְגִּיר הַמֹּשֶׁק הַזֶּה שִׂים בְּפִיךָ, וְהָיָה בְּמַלֵּא אֲהוּבָתְךָ תַּאֲוַת לֳבְּךָ וְשַׁרְתָּ הַבַּיִת הַזֶּה:
הַחוֹשְִׁקִים, נָּא לִי הַגִּידוּ

כִּגְבֹר חֵשֶׁק אִישׁ מַהּ-יָעַשׂ?

וְאַחַר נְשָׁקַתְנִי וַתַּשְׁבִּיעֵנִי, כִּי לֹא אָשִׁיר אֶת-בֵּית הַשִּׁיר בִּלְתִּי אִם בְּצֵאתִי מִבֵּית אֲהוֹבָתִי, וָאֹמַר: כִּדְבָרַיִךְ כֵּן אֶעֱשֶׂה.

וָאֵצֵא בַנֶשֶׁף וָאֵלֵךְ עַד הַגִּיעִי אֶל הַגַּן וָאֶמְצָא פִתְחוֹ פָתוּחַ, וָאִכָּנֵס וָאֵרֶא אוֹר מֵרָחוֹק וָאֵלֵךְ עַד הַגִּיעִי שָׁמָּה, וָאֶמְצָּא לִשְּׁכָּה מְקֹרָה כִּפַּת עֲצֵי הָבְנִים, וּמְנוֹרָה תְלוּיָה בְאֶמְצַע הַתִּקְרָה. וְהַלִּשְׁכָּה מֻצָּעָה שְׁטִיחֵי מֶשִׁי רְקוּמִים בְּזָהָב וּבְכֶסֶף, וְנֵר דּוֹנַג דּוֹלֵק בְּתוֹךְ פָּמוֹט זָהָב מִתַּחַת לַמְּנוֹרָה, וּבְאֶמְצַע הַלִּשְׁכָּה מְזַנֶּקֶת מְצֻיָּרָה כָּל-מִינֵי צִיּוּרִים, וְעַל יַד הַבְּרֵכָה שֻׁלְחָן מְכֻסֶּה מַפַּת מֶשִׁי, וְעַל יָדָהּ כַּד חַרְסִינָה גָדוֹל מָלֵא יַיִן, וְכוֹס זְכוּכִית דַּקָּה וְעָלֶיהָ צִיּוּרֵי זָהָב, וְשָׁם טַס כֶּסֶף גָּדוֹל מְכֻסֶּה, וָאָרִים מִכְסֵהוּ וְהִנֵּה בוֹ כָל-פְּרִי לְמִינֵהוּ: תְּאֵנִים וְרִמּוֹנִים וְצִמּוּקִים וְאֲפַרְסְקִים, וּבֵינֵיהֶם פְּרָחִים שׁוֹנִים: וְרָדִים וּגְרָדִים, וַהֲדַסִּים וְנַרְקָסִים, וָאֶתְפַּעֵל לְמַרְאֶה עֵינַי וָאֶשְׂמַח תַּכְלִית שִׂמְחָה וִיגוֹנִי חָלַף עָבָר. אַךְ בְּכָל-הַמָּקוֹם הָזֶּה לֹא מָצָאתִי אַף יְצִיר מִיצִירֵי הָאֵל, וְלֹא עֶבֶד וְלֹא שִׁפְחָה. וָאֵשֶׁב יָשוֹב וְחַכּוֹת לְבוֹא אֲהוּבָתִי עַד עֲבֹר אַשְׁמֹרֶת הַלַּיְלָה הָרִאשׁוֹנָה. וַתַּעֲבֹר גַּם הַשָׁעָה הַשְּׁנִיָה וְהַשְּׁלִישִׁית וְאֵין בָּא. וַיֶחֱזַק עָלַי רַעֲבוֹנִי, כִּי זֶה זְמַן רַב לֹא אָכַלְתִּי; וְרֵיחַ הַמַּטְעַמִּים אֲשֶׁר עַל הַשֻּׁלְחָן עוֹרֵר בִּי יוֹתֵר אֶת-תְּשוּקַת הָאֲכִילָה. וָאֶגַּש אֶל הַשֻּׁלְחָן וָאָרִים אֶת-הַמִּכְסֶה וְהִנֵה בְאֶמְצָעוֹ טַס חַרְסִינָה וְעָלָיו אַרְבַּע תַרְנְגֹלוֹת מְטֻגָּנוֹת וּמְתֻבָּלוֹת בִּתְבָלִים שׁוֹנִים וּמִסָּבִיב לוֹ אַרְבָּעָה בְזִיכִים, בָּאֶחָד מֵהֶם חַלְוָה1, וּבַשֵּׁנִי גַרְגִירֵי רֳמּוֹנִים, וּבַשְׁלִישִׁי כְּרִיכוֹת, וּבָרְבִיעִי תוּפִינֵי דְבָשׁ. וָאֹכַל מִן הַתּוּפִינִים וָאֶטְעַם מִן הַבָּשָׂר, וְאַחַר פָּנִיתִי אֶל הַכְּרִיכוֹת וָאֶטְעַם מֵהֶן; וְגַם מִן הַחַלְוָה אָכַלְתִּי לְגִימָה וּשְׁתַּיִם וְשָׁלֹשׁ וְאַרְבַּע, וַתִּמָּלֵא כְּרֵשִׂי, וְרֹאשִׁי אָחוּז בְּחַבְלֵי שֵׁנָה; וָאֶרְחַץ כַּפַּי, וָאָשִׂים רֹאשִׁי עַל כָּר, וַתִּגְבַּר עָלַי הַשֵׁנָה וָאֵרָדַם. וְלֹא יַדַעְתִּי נַפְשִׁי עַד אֲשֶׁר הֱקִיצוֹתִי בִשְׂרֹף אוֹתִי הֶשֶּׁמֶשׁ בְּחֻמָּהּ, כִּי זֶה לִי לֵילוֹת אֲחָדִים לְלֹא שֵׁנָה. וַיְהִי בַהֲקִצִי וָאֶמְצָא עַל בִּטְנִי מֶלַח וּפֶחָם, וָאָרוּם וְאֲנָעֵר בְּגָדַי, וָאֵפֵן יָמִינָה וּשְׂמֹאלָה וְלֹא מָצָאתִי אִישׁ. וַאֲנִי יָשֵׁן עַל רִצְפַּת שֵׁשׁ מִבְּלִי כָל-יְצוּעַ. וָאֶהִי נָבוֹך וְכוֹאֵב וְדִמְעוֹתַי נוֹזְלוֹת עַל לְחָיַי, וָאָקוּם וָאֶפְנֶה אֶל בֵּיתִי.

וַיְהִי בַּהֲגִיעִי הַבַּיְתָה וְהִנֵּה בַת דּוֹדִי מְתוֹפֶפֶת בְּיָדָהּ עַל חָזֵהָ וּבוֹכָה וְשָׁרָה: לְכָהּ נַחַל אֲשֶׁר יֵלֵךְ לָאֶרֶץ

מְגוֹרֵי דוֹד, קְרָא עָלָיו שְׁלוֹמִי,

וּמֵימֶיךָ – וְאִם רֻבָּם אֲדֻמִּים

אֱמָר-לוֹ כִּי דְמָעַי הֵם וְדָמִי,

מְסַכְתִּימוֹ עֲלֵי רַחְקוֹ וְחִשְׁקוֹ

שְׁנֵיהֶם שָׁלְחוּ רָקָב בְּעַצְמִי,

וְנָא רַכּוֹת תְּדַבֵּר לוֹ לְבַל יִכְ–

בְּשׁוּ זָרִים בְּתוֹך לִבּוֹ מְקוֹמִי.

וַיְהִי בִרְאוֹתָהּ אוֹתִי וַתְּמַהֵר וַתָּקָם, וַתִּמַּח דִּמְעֹתֶיהָ, וַתִּגַּשׁ אֵלַי בְּרֹך אֲמָרֶיהָ וַתֹּאמֶר: בֶּן דּוֹדִי, אָכֵן חֲנָנְךָ הָאֵל בְּתִתּוֹ אַהֲבָתְךָ בְּלֵב אֲהוּבָתֶךָ, וְאָנֹכִי יוֹשֶׁבֶת עֲצוּבָה וְנוּגָה לִפְרִידָתֶךָ. וְאַחַר הִפְשִׁיטַתְנִי בְגָדַי וַתִּפְרְשֵׂם וַתֹּאמֶר: חֵי הָאֵל, אֵין זֶה רֵיחַ הַמְּאֻשָּׁר בְּהַשָּׂגַת חֶפְצוֹ, וְעַתָּה הוֹדִיעֵנִי נָא מַה-קוֹרוֹתֶיךָ, בֶּן דּוֹדִי. וָאַגִּיד לָהּ כָּל-אֲשֶׁר קָרַנִי, וַתְּחַיֵךְ חִיוּךְ קֶצֶף וַתֹּאמֶר: חֵי הָאֵל, בֶּן-דּוֹדִי, כִּי יְרֵאָה הִנְנִי מִפְּנֵי הָאִשָּׁה הַזֹּאת אֲשֶׁר מִלְּאָה חַדְרֵי לִבֶּךָ. וְהַמֶּלַח וְהַפֶּחָם אֲשֶׁר שָׂמָה עַל בִּטְנְךָ – זֶה פִּתְרוֹנָם: חֲסַר-טַעַם הִנְּךָ, כִּי בְעֶצֶם אַהֲבָתְךָ גָבְרָה הַשֵּׁנָה עָלֶיךָ, וְעַתָּה הִתְאַמֵּץ וְהֱיֵה לְאִישׁ מְמֻלָּח פֶּן תִּקַּעְנָה הַנְּפָשׁוֹת מִמֶּךָ, וְאִם כָּל-תַּאֲוָתְךָ הִיא רַק לֶאֱכֹל וְלִשְׁתּוֹת וְלִישׁוֹן יַשְׁחִיר הָאֵל פָּנֶיךָ כִּשְׁחוֹר הַפֶּחָם. זֶה פִתְרוֹן רְמִיזוֹתֶיהָ, יַצִּילְךָ הָאֵל מִמְּזִמּוֹתֶיהָ.
וַיְהִי כְּשָׁמְעי דְּבָרֶיהָ וְאַךְ בְּיָדַי עַל חָזִי וָאֹמַר: חֵי הָאֵל, כִּי כֵנִים דְּבָרַיִך, וְאָנֹכִי אָשֵׁם בְּנַפְשִֹי כִּי יָשַׁנְתִּי וְהַחוֹשְׁקִים לֹא יִישְׁנוּ. וְעַתָּה מָה אֶעֱשֶׂה כַּיוֹם? רַחֲמִינִי נָא, בַּת-דּוֹדִי, יְרַחֲמֵךְ הָאֵל. וַתֹּאמֶר אֵלָי: בֶּן-דּוֹדִי, הֵן אָמַרְתִּי לְךָ, כִּי לוּ יָכֹלְתִּי לָצֵאת וְלָבוֹא הָיִיתִי מְאַחֶדֶת אֶתְכֶם וְסוֹכֶכֶת עֲלֵיכֶם. וְעַתָּה זֹאת עֲשֵׂה: אֱכֹל עַתָּה עַד אֲשֶׁר תִּשְׂבַּע וְלֹא תָשׁוּב לְהִשְׁתּוֹקֵק עוֹד לְמַטְעַמִּים אֲשֶׁר תִּמְצָא שָׁם. וָאֹכַל וָאֶשְׂבָּע.

וְכַאֲשֶׁר הִגִּיעַ הַלַּיְלָה הֵבִיאָה אֵלַי חֲלִיפָה נֶהְדָּרָה וַתַּלְבִּישֵׁנִי, וַתַּשְׁבִּיעֵנִי עוֹד הַפַּעַם, לְהַזְכִּיר אֶת-בֵּית הַשִּׁיר אֲשֶׁר לִמְּדַתְנִי, וְאַחַר שָׁבָה וַתַּזְהִירֵנִי מִפְּנֵי הַשֵּׁנָה. וָאֵלֵך אֶל הַגַּן וָאַגִּיעַ עַד הַלִּשְׁכָּה. וָאֵשֵׁב כְּשָׁעָה יוֹשֵׁב וְּמְחַכֶּה. וְהִנֵּה נָשְׁבָה הָרוּחַ וַתָּבֵא לְאַפִּי אֶת-רֵיחַ הַמַּטְעַמִּים. וַאֶגַּש אֶל הַשֻּׁלְחָן וָאֲגָלֶּה אֶת-מִכְסֵה הַקַּעֲרִיּוֹת וְהִנֵּה שָׁם תַּרְנְגֹלוֹת וְכָל-מֵיטַב הָמַּאֲכָלִים וְהַמַּשְׁקִים וְהַקְּלָיוֹת, וַתְּכַל נַפְשִׁי אֲלֵיהֶם, אַךְ הִתְאַפַּקְתִּי פַעַם וּפַעֲמַיִם, אַךְ לָאַחֲרוֹנָה כָּשַׁל כֹּחַ סַבְלְנוּתִי וָאֶגַּשׁ אֶל הַשֻּׁלְחָן וָאֹכַל מִכָּל-מַאֲכָל לְגִימָה. וְאַחַר נִגַּשְׁתִּי אֶל כַּד הַיַּיִן, וָאֹמַר בְּלִבִּי: אֶשְׁתֶּה רַק כּוֹס אַחַת, וָאְשְׁתְּ אַחַת וּשְׁתַּיִם וְשָׁלֹשׁ, וְהָרוּחַ נָשְׁבָה עַל פָּנַי וָאֶפֹּל אַרְצָה וָאֵרָדֵם תַּרְדֵּמָה חֲזָקָה, וָאִישָׁן כָּל-הַלַּיְלָה עַד עֲלוֹת הַשֶּׁמֶשׁ. וַיְהִי בַהֲקִיצִי וָאֵרֶא וְהִנֵּה אָנכֹיִ מִחוּץ לַגָּן, וְעַל בִּטְנִי לַהַב שַׂכִּין וְדִרְהַם בַּרְזֶל. וָאֶקָּחֵם וַאֲבִיאֵם אִתִּי הַבָּיְתָה. וּבְבוֹאִי נָפַלְתִּי מְלוֹא קוֹמָתִי וְאֶת-הַשַּׂכִּין וְהַדִּרְהַם הִשְׁלַכְתִּי מִיָּדִי וָאֶתְעַלָּף. וַיְהִי בַהֲקִיצִי וָאַגִּיד לְבַת דּוֹדִי כָל-אֲשֶׁר קָרַנִי, וַיִּכָּמְרוּ נִחוּמֶיהָ עָלַי וַתֹּאמֶר: הֵן הַזְהֵר הִזְהַרְתִּיךָ מִפְּנֵי הַשֵּׁנָה וְאַתָּה לֹא שָׁמַעְתָּ בְקוֹלִי, וְכָל-דְּבָרַי הָיוּ לְלֹא הוֹעִיל. וָאֶשְׁאָלֶנָּה לְפֵשֶׁר הַדִּרְהָם וְלַהַב הַשַּׂכִּין, וַתֹּאמֶר: דִּרְהַם הַבַּרְזֶל מוֹרֶה עַל עֵינָהּ הַיְמָנִית, וְהִיא נִשְׁבַּעַתּ לֵאמֹר: חֵי הָאֱלֹהִים וְחֵי עֵינִי הַיְמָנִית, כִּי אִם שׁוּב תָּשׁוּב לִישַׁן אֶשְׁחָטְךָ בַּמַּאֲכֶלֶת הַזֹּאת, וְעַתָּה, בֶּן-דּוֹדִי, אִם הִנְּךָ בָטוּחַ בְּנַפְשְׁךָ כִּי בְּשוּבְךָ אֵלֶיהָ לֹא תִגְבַּר שְׁנָתְךָ עָלֶיךָ – שׁוֹבָה אֵלֶיהָ, וְאִם אַיִן, דַּע לְךָ כִּי שָׁחוֹט תִּשְׁחָטְךָ וְלֹא תָחוּס וְלֹא תְרַחֵם, וָאֹמַר: וּמֶה עָלַי אֵפוֹא לַעֲשׂוֹת? וַתִּקָּחֵנִי וַתַּשְׁכִּיבֵנִי עַל הַמִּטָה וַתְּיַשְּׁנֵנִי וְאַחַר לָקְחָה מְנִיפָה וַתֵּשֶׁב לְיַד רֹאשִׁי וַתְּנַפְנֵף בָּהּ עַל פָּנַי כָּל-הַיּוֹם עַד לִפְנוֹת עֶרֶב, וּכְבוֹא הָעֶרֶב הֱקִיצַתְנִי וַתָּבֵא לִי לֶאֱכֹל וַאֲמָאֵן וַתֹּאמֶר אֵלַי: שְׁמַע בְּקוֹלִי וֶאֱכֹל. וָאֹכַל כַּאֲשֶׁר צִַּוַּתְנִי, וַתְּהִי מְשִׂימָה הָאֹכֶל בְּפִי וְאָנֹכִי לוֹעֵס עַד אֲשֶׁר שָׂבָעְתִּי. וַתַּשְׁקֵנִי עֲסִיס רִמּוֹנִים מְתוּקִים, וְאַחַר רָחֲצָה יָדַי וַתְּנַגְּבֵן וַתְּזַלֵּף עָלַי מֵי וְרָדִים. וּבַחֲשֹׁךְ הַלַּיְלָה הִלְבִּישַׁתְנִי בְגָדֵי הַחֲמוּדוֹת וַתְּקַטְרֵנִי בְכָל-רֹאשֵׁי בְשָׂמִים וַתֹּאמֶר אֵלַי: בֶּן-דּוֹדִי, הַלַּיְלָה הַזֶּה יְהִי לְךָ לֵיל שִׁמּוּרִים, אַל תָּנוּם וְאַל תִּישָׁן, וְאֶת-בֵּית הַשִּׁיר אֲשֶׁר צִוִּיתִיךָ לְהַגִיד בְּלֶכְתְּךָ מִמֶּנָּה אַל נָא תִשְׁכָּח. וּבְדַבְּרָה בָּכְתָה וְלִבִּי נִשְׁבַּר עָלֶיהָ.

ב. עָזִיז בְּבֵית חֲשׁוּקָתוֹ.

וָאֵצֵא מִבֵּיתִי וָאָבוֹא אֶל-הַגָּן, וָאַעַל עַל מוֹשָׁבִי וְאָנֹכִי שָׂבֵעַ. וָאֵשֵׁב עַד אֲשֶׁר עָבַר רֶבַע הַלַּיְלָה וְלֹא בָאָה, וֵָאֵוֹסִיף לָשֶׁבֶת וּלְחַכּוֹת, וְהַשָּׁעוֹת אֲרֻכּוֹת כְּשָׁבוּעוֹת, וָאֱהִי עֵר עַד עֲבֹר שְׁלֹשֶׁת רִבְעֵי הַלַּיְלָה, וְהַתַּרְנְגֹלִים הֵחֵלוּ כְבָר לִקְרֹא קְרִיאָתָם. וַיֶחֱזַק הָרָעָב עָלַי מְאֹד, וָאֶגַּשׁ אֶל הַשֻּׁלְחָן וָאֹכַל עַד אֲשֶׁר שָׂבָעְתִּי. וַיְּכְבַּד רֹאשׁיִ וָאֹמַר לִישׁוֹן, וְהִנֵּה קוֹל רַעַשׁ נִשְׁמַע מֵרָחוֹק, וָאָקוּם וָאֶרְחַץ יָדַי וּפִי מַהֵר וָאֲעוֹרֵר נַפְשִׁי.

וַיְהִי אַחֲרֵי עֲבֹר רְגָעִים אֲחָדִים וְהִנֵּה הִיא בָאָה וְאִתָּהּ עֶשֶׂר עֲלָמוֹת וְהִיא בֵינֵיהֶן כַּיָּרֵחַ בֵּין הַכּוֹכָבִים. וְעָלֶיהָ שִׂמְלַת קְטִיפָה יְרֻקָּה רְקוּמָה בְּזָהָב אָדֹם, וְהִיא כְדִבְרֵי הַמְשׁוֹרֵר:

הָלוֹךְ וְטָפוֹף צָעֲדָה וּבְגַאֲוָה,

הֻתְּרוּ קְרָסֶיהָ וּפֶרַע שַׂעֲרָהּ,

וָאֹמְרָה: הַיַּעֲלָה! רַחְמִי לְבָב

כָּלָה בְּאַהְבָתֵךְ וְנִשְׂרָף. אָמְרָה:

כָּזֹאת וְכָזֹאת אֶשְׂרְפָה לֵב חוֹשְׁקִים

וּבְלִי חֲמֹל אוֹתוֹ קְרָעִים אֶגְזְרָה.

וָאֹמְרָה: עַד אָן לְבָבֵךְ תַּקְשִׁי?

עָנְתָה: הֲלַסֶּלַע דְּבָרָךְ תֹּאמְרָה?

וָאֶעֱנֶה: לִבֶּךְ לְסֶלַע אִם תְּדַ–

מִּי, בִּשְׁמִיר חִשְׁקִי הֲלֹא יִשָּׁבְרָה.

וַיְהִי כִרְאוֹתָהּ אוֹתִי וַתֹּאמֶר: אֵיכָכָה זֶה לֹא נִרְדַּמְתָּ גַּם הַפָּעַם? וְעַתָּה כִי עַרְתָּ רֹב הַלַּיְלָה אוֹת הוּא כִּי אָכֵן חוֹשֵׁק הִנֶּךָ וְאַחַר פָּנְתָה אֶל נַעֲרוֹתֶֹיהָ וַתִּרְמֹז לָהֶן וַתֵּלַכְנָה. אָז קָרְבָה אֵלַי וַתִּלְחָצֵנִי אֶל חָזֶהָ וַתִּשָׁקֵנִי וָאֶשָּׁקֶהָ, וַנִּתְעַלֵּס בַּאֲהָבִים כָּל-שְׁאֵרִית הַלַּיְלַה, וַיְהִי בַבֹּקֶר וָאֹמַר לָלֶכֶת, וַתַֹּחַזֵק בִּי וַתֹּאמַר: עֲמֹד עַד אֲשֶׁר אוֹדִיעֲךָ דָבָר, וָאֶעֱמֹד. וַתַּתֵּר מִטְפַּחַת וַתּוֹצִיא אֶת-הַפִּתְקָה הַזֹּאת מִמֶּנָּה, וַתִּפְרְשֶׂנָּה לְנֶגֶד עֵינַי וָאֵרֶא בָהּ צוּרַת צְבִי, וְאֶתְפַּלֵּא מְאֹד לִיפִי הַצּוּרָה וָאֶקָּחֶנָּה, וַיְהִי הַמּוֹעֵד בֵּינֵינוּ כִּי אָבוֹא לַיְלָה-לַיְלָה אֶל הַגָּן.

וְאַחַר הָלַכְתִּי מִבֵּיתָהּ וְלִבִּי עָלֵז וְשָׂמֵחַ, וּלְרֹב שִׂמְחָתִי שָׁכַחְתִּי אֶת-דְּבַר בֵּית הַשִׁיר אֲשֶׁר צִוַּתְנִי בַּת דּוֹדִי לִקְרֹא בְאָזְנֶיהָ לִפְנֵי לֶכְתִּי. וְכַאֲשֶׁר נָתְנָה לִי אֶת-הַמִּטְפַּחַת אֲשֶׁר בָּהּ צוּרָת הַצְּבִי אָמְרָה אֵלַי: זֶה מַעֲשֶׁה יְדֵי אֲחוֹתִי, וָאֹמַר: וּמַה-שֵׁם אֲחוֹתֵךְ? וַתֹּאמַר: שְׁמָהּ נוּר אַל הוּדֶה, וְעַתָּה שְׁמָר-נָא מִטְפַּחַת זֹאת. וָאֲבָרְכֶנָּה לְשָׁלוֹם וָאֵלֵךְ, וָאָשׁוּב אֶל בֵּיתִי וָאֵרֶא וְהִנֵּה בַּת דּוֹדִי שׁוֹכֶבֶת, וְכַאֲשֶׁר רָאַתְנִי קָמָה וְדִמְעוֹתֶיהָ יוֹרְדֹות עַל לְחָיֶיהָ. וַתִּגַּשׁ אֵלַי וַתִּשַּׁק אֶת-חָזִי וַתֹּאמַר: הֶעָשִׂיתָ אֵת אֲשֶׁר צִוִּיתִיךָ בִדְבַר בֵּית הַשִּׁיר? וָאֹמַר: אֲבָל שְׁכַחְתִּיו, וּתְמוּנַת הַצְּבִי הַזֹּאת הִיא אֲשֶׁר הִשְׁכִּיחַתְנִי אוֹתוֹ. וָאַשְׁלִיך הַמִּטְפַּחַת לְפָנֶיהָ, וָתָּקָם וַתֵּשֶׁב וַתֹּאמֶר אֵלַי: הָבָה הַמִּטְפַּחַת הַזֹּאת, וָאֶתְּנֶנָּה לָהּ, וַתִּקָּחֶנָּה וַתִּפְרְשֶׂנָּה וַתֵּרֵא אֵת אֲשֶׁר בָּהּ.

וַיְהִי בָעֶרֶב וַתֹּאמֶר אֵלַי: לֶךְ-לְךָ בְשָׁלוֹם; אַךְ לִפְנֵי לֶכְתְּךָ מֵעִמָּהּ קְרָא נַא בְאָזְנָהּ אֶת-בֵּית שִׁירִי וְאַל תִּשְׁכָּח. וָאֵלֵךְ אֶל הַגָּן, וָאָבֹא אֶל הַלִּשְׁכָּה, וְהִנֵּה הַצְּעִירָה יוֹשֶׁבֶת וּמְחַכָּה לִי. וַיְהִי כַאֲשֶׁר רָאַתְנִי וַתָּקָם אֵלַי וַתִּשָׁקֵנִי, וְאַחַר הוֹשִׁיבַתְנִי בְּחֵיקָהּ, וַנֹּאכַל וַנֵּשְׁתְּ וַנֵּיטִיב לִבֵּנוּ כָל-הַלַּיְלָה. וְכַאֲשֶׁר הֵאִיר הַבֹּקֶר קָרָאתִי בְאָזְנָהּ אֶת-הַבַּיִת אֲשֶׁר לִמְּדַתְנִי בַת דּוֹדִי לֵאמֹר:

הַחֹוֹשְׁקִים, נָא לִי הַגִּידוּ כִּגְּבֹר חֵשֶׁק אִישׁ מַה-יָעַשׂ? וַיְהִי כְּשָמְעָהּ אֶת-הַבַּיִת וַתִּזַּלְנָה עֵינֶיהָ דִמְעָה וַתָּשָׁר: יֵלְכָה אַחְרָיו, יַסְתִיר סוֹדוֹ, יִשָּׂא עָמָל, יִסְבֹּל כָּעַשׂ.

וָאֶלְמַד אֶת-הַבַּיִת עַל-פֶּה, וְלִבִּי שָׂמֵחַ כִּי מִלֵּאתִי הַפַּעַם אֶת-בַּקָּשַׁת דּוֹדַנְתִּי. וְאַחַר יָצָאתִי וָאָשׁוּב אֶל בַּת דּוֹדִי, וָאֶמְצָאֶנָּהּ שׁוֹכֶבֶת בַּמִּטָּה וְאִמִּי עַל יַד רֹאשָׁהּ יוֹשֶׁבֶת וּבוֹכָה. וַיְהִי בְבוֹאִי הַחַדְרָה וַתֹּאמֶר אֵלַי אִמִּי: אֹוי לְךָ, בֶּן-דּוֹד, אֵיכָכָה תַּעֲזֹב אֶת-דּוֹדַנְתְּךָ חוֹלָה וְלֹא תִשְׁאַל לְמַחֲלָתָהּ? וַיְהִי כַאֲשֶׁר רָאַתְנִי דוֹדַנְתִּי וַתָּרֵם רֹאשָׁהּ וַתֵּשֶׁב וַתֹּאמֶר אֵלַי: עָזִיז, הֲקָרָאתָ אֶת-בֵּית הַשִּׁיר? וָאֹמַר: כֵּן. וָאֶקְרָא לְפָנֶיהָ אֶת-הַבַּיִת אֲשֶׁר הִשְׁמִיעַתְנִי חֲשׁוּקַתִי. וַתִּקְרָא בַת דּוֹדִי בְאָזְנַי בֵּית שִׁיר שֵׁנִי לֵאמֹר:

גָּזַר לִסְבֹּל, אַךְ לֹא מָצָא

רַק לֵב נוֹאָשׁ מִתּוֹחֶלֶת.

וַתֹּאמֶר: בְּלֶכְתְּךָ אֵלֵיהָ קְרָא בְאָזְנֶיהָ אֶת-הַבַּיִת הַזֶּה אֲשֶׁר שָׁמָעְתָּ. וָאֹמַר: כֵּן אֶעֱשֱׂה.
וַיְהִי בָעֶרֶב וָאֵלֵךְ אֶל הַגָּן כְּמִשְׁפָּטִי, וָאֵשֶׁב שָׁם כָּל-הַלַּיְלָה וּבַבֹּקֶר עָשִׂיתִי כַאֲשֶׁר צִוַּתְנִי בַת דּוֹדִי וָאֶקְרַא בְאָזְנֶיהָ אֶת-הַבַּיִת הַשֵׁנִי. וַיְהִי כְשָׁמְעָהּ אוֹתוֹ וַתִּזַּלְנָה עֵינֶיהָ דִמְעָה, וַתָּשַׁר לֵאמֹר:
אִם לֹא יוּכַל אַמֵּץ לִבּוֹ

רַק הַמִּיתָה הַגּוֹאֶלֶת.

וָאֶלְמַד אֶת הַבַּיִת וָאָשׁוּב אֶל בֵּיתִי. וַיְהִי בְּבוֹאִי וָאֶמְצָא אֶת בַּת-דּוֹדִי שׁוֹכֶבֶת מִתְעַלֶּפֶת, וְאִמִּי יוֹשֶׁבֶת לְיַד רֹאשָׁהּ. וַיְהִי בְּשָׁמְעָהּ קוֹלִי וַתִּפְקַח עֵינֶיהָ וַתֹּאמֶר: עָזִיז, הֲקָרָאתָ לְפָנֶיהָ אֶת-בֵּית שִׁירִי? וָאֹמַר: כֵּן, בַּת דּוֹדִי, וְכַאֲשֶׁר שְׁמָעַתּוּ בָּכְתָה בְּאָזְנַי וַתָּשַׁר אֶת-הַבַּיִת הַזֶּה לֵאמֹר:
יֵלְכָה אַחֲרָיו, יַסְתִּיר סוֹדוֹ,

יִשָּׂא עָמָל, יִסְבֹּל כָּעַשׂ.

וַיְהִי כִשְׁמֹעַ בַּת דּוֹדִי אֶת-הַבַּיִת הַזֶּה וַתֵּבְךְּ וַתָּשָׁב וַתִּתְעַלֵּף וְאַחַר שָׁרָה בָּיִת אַחֵר לֵאמֹר:
אָכֵן הִקְשַׁבְנוּ וַמָתְנוּ / שָׁלוֹם לַאְשֶׁר הִפְרִידָתְנוּ.
וַיְהִי בָעֶרֶב וָאֵלֵךְ אֶל הַגָּן כְּמִשְׁפָּטִי וָאֶמְצָע אֶת-הַצְּעִירָה יוֹשֶׁבֶת וּמְחַכָּה לִי. וַנֶּשֶׁב וַנֹּאכַל וַנֵּשְׁתְּ וַנִּתְעַנֵּג וַנִּישַׁן עַד הַבֹּקֶר. וַיְהִי בְאָמְרִי לָלֶכֶת וָאֶקְרָא בְאָזְנֶיהָ אֶת-הַבַּיִת שֶׁל בַּת דּוֹדִי. וַיְהִי אַךְ שָׁמְעָה אוֹתוֹ וַתִּצְעַק צְעָקָה גְדוֹלָה וּמָרָה, וַתֹּאמֶר: חֵי הָאֵל כֵּי מְחַבֶּרֶת הַבַּיִת הַזֶּה מֵתָה. וָאֹמַר לָהּ: הֵן כַּלָּתִי הִיא. וַתֹּאמֶר: אָכֵן מְשַׁקֵּר הִנְךָּ, לוּ הָיְתָה לְךָ כַלָּה הָיָה לִבְּךָ מָלֵא אַהֲבָתָהּ כַּאֲשֶׁר אֲהֵבָתְךָ הִיא. וְעַתָּה אַתָּה הוּא אֲשֶׁר הֲסִבּוֹתָ בְּמוֹתָהּ, יַהֲרֶגְךָ אֱלֹהִים כַּאֲשֶׁר הֲרַגְתָּהּ, וְלוּ הִגַּדְתָּ לִי כִּי כַלָּה לְךָ לֹא נְתַתִּיךָ לָגֶשֶׁת אֵלַי. וָאֹמַר אֵלֶיהָ: אָכֵן כַּלָּתִי הִיא, וְהִיא אֲשֶׁר בֵּאֲרָה לִי תָמִיד אֶת-רְמִיזוֹתַיִךְ, וְהִיא אֲשֶׁר לִמְּדַתְנִי בְכָל פַּעַם אֶת אֲשֶׁר עָלַי לַעֲשׂוֹת לְמַעַן אַגִּיעַ אֵלָיִךְ. וַתֹּאמֶר: הֲיָדְעָה דָבָר עַל אוֹדוֹתַי? וָאֹמַר: כֵּן, וַתֹּאֶמר: יִגְמָלְךָ הָאֵל כִּגְמוּלְךָ עָלֶיהָ. וְאַחַר אָמְרָה: לֵךְ וְרָאִיתָ אֶת אֲשֶׁר אִתָּה. וָאֵלֵךְ וְלִבִּי נִפְעָם עַד אֲשֶׁר הִגַּעְתִּי אֶל סִמְטַת בֵּיתֵנוּ, וְהִנֵּה קוֹלוֹת יְלָלָה בְאָזְנָי. וָאֶשְׁאַל לַדָּבָר וַיֻגַּד לִי לֵאמֹר: מָצָאנוּ אֶת-עֲזִיזָה מֵאַחֲרֵי הַדֶּלַת מֵתָה. וָאָבֹא הַבָּיְתָה. וַיְהִי כַאֲשֶׁר רָאַתְנִי אִמִּי וַתִּקְרַא: דָּמָהּ בְּרֹאשֶׁךָ! אַל יִסְלַח לְךָ הָאֵל מוֹתָהּ, אוֹי לְךָ, חֲתַן דָּמִים! וְאָבִי בָא וַיַעֲרֹך כָּל-דִּבְרֵי טָהֳרָתָהּ וְהַלְבָּשָׁתָהּ תַּכְרִיכֶיהָ, וַנְּלַוֶהָ וַנִּקְבְּרֶנָּה וַנִּקְרָא עַל קִבְרָהּ אֶת-הַתְּפִלּוֹת לַנִּפְטָרִים, וְאַחַר יָשַׁבְנוּ לְיַד קִבְרָה שְׁלֹשָׁה יָמִים. וְכַעֲבֹר שְׁלֹשֶׁת הַיָּמִים שַׁבְנוּ הַבַּיְתָה וְלִבִּי נִשְׁבָּר עָלֶיהָ. וַתִּגַּשׁ אִמִּי אֵלַי וַתֹּאמֶר: מִי יִתֶּן וְיָדַעְתִּי מֶה עָשִׂיתָ לָהּ עַד כְּדֵי לְפַקֵּעַ מְרֵרָתָהּ. וְאָנֹכִי שָׁאוֹל שְׁאַלְתִּיהָ בְכָל-עֵת לְסִבַּת חָלְיָהּ, אַךְ הִיא שָׂמָה מַחֲסוֹם לְפִיהָ וְלֹא גִלְּתָה לְפָנַי דָבָר. וְעַתָּה אַשְׁבִּיעֲךָ בֵאלֹהִים כִּי תוֹדִיעֵנִי אֶת אֲשֶׁר עָשִׂיתָ לָהּ עַד אֲשֶׁר מֵתָה. וָאֹמַר: לֹא עָשִׂיִתי דָבָר. וַתֹּאמֶר: יִקֹּם הָאֵל נִקְמָתָהּ מִמְךָּ. הִיא לֹא הִגִּידָה דָבָר וְלֹא הִתְאוֹנְנָה עָלֶיךָ, עַד אֲשֶׁר מֵתָה וְלִבָּהּ טוֹב עָלֶיךָ, וּבְעֵת מוֹתָהּ יָשַׁבְתִּי עַל יָדָהּ וַתִּפְתַּח עֵינֶיהָ וַתֹּאמֶר: חֲמוֹתִי, אַל יָשִׂים הָאֵל דָּמִי עַל בְּנֵךְ וְאַל יִגְמְלֵהוּ כַּגְּמוּל אֲשֶׁר גְּמָלָנִי, אַשְׁרַי כִּי הִנְנִי עוֹבֶרֶת הַיּוֹם מֵעוֹלַם הַהֶפְסֶד וְהַכִּלָּיוֹן הַזֶּה אֶל הָעוֹלָם הַנִּצְחִי. וָאֹמַר לָהּ: בִּתִּי, שָׁלוֹם לָךְ וְשָׁלוֹם לִנְעוּרַיִךְ. וָאָחֵל לְשָׁאֲלָה לְסִבַּת מַחֲלָתהּ וְלֹא עָנָתְנִי. וְאַחַר חִיְכָה וָתֹּאמֶר: חֲמוֹתִי, אִם יֹאמַר בְּנֵךְ לָשׁוּב וְלָלֶכֶת אֶל הַמָּקוֹם אֲשֶׁר אֵלָיו הוּא הוֹלֵךְ יוֹם יוֹם, הַגִּידִי לוֹ כִּי בְלֶכְתוֹ מִשָׁם יָגִּיד אֶת-הַמִּלִּים הָאֵלֶּה:
“הָאֱמוּנָה מְפֹאָרָה / וְהַבְּגִידָה מְכֹעָרָה”, כִּי נִחוּמַי נִכְמְרוּ עָלָיו בְּחַיַּי וּבְמוֹתִי. וְאַחַר נַתְנַה לִי דָבָר לְמַעֲנֶךָ, וַתַּשְׁבִּיעֵנִי כִּי לֹא אֶתְּנֶנוּ לְךָ בִּלְתִּי אִם בִּרְאוֹתִי אוֹתְךָ בּוֹכֶה עָלֶיהָ וּמִתְאַבֵּל בְּלֵב תָּמִים. הִנֵּה הוּא שָׁמוּר אִתִּי, וְכַאֲשֶׁר אֶרְאֲךָ כַּאֲשֶׁר אָמְרָה לִי אֶתְּננּוּ לָךְ. וָאֹמַר אֵלֶיהָ: הַרְאִינִי אֶת-הַדָּבָר וַתְּמָאֵן.
וְאָנֹכִי שַׁבְתִּי וָאֶשְׁתַּעֲשֵׁעַ בְּתַעֲנוּגָי, וְאֶת-מוֹת כַּלָּתִי לֹא זָכַרְתִּי, כִּי הָלַכְתִּי שׁוֹבָב בְּדַּרְכֵי לִבִּי, וָאֶשְׁתּוֹקֵק לִהְיוֹת כָּל-הַיּוֹם וְכָל-הַלַּיְלָה עַל יַד חֲשּוּקָּתִי.

וַיְהִי אַךְ הִגִּיעַ הַלַּיְלָה וָאֵלֵךְ אֶל הַגָּן וָאֶמְצָא אֶת-הַצְּעִירָה יוֹשֶׁבֶת כְּעַל גֶּחָלִים בִּצְפִיָּתָהּ, וְהִיא אַךְ רָאַתְנִי וַתְּמַהֵר וַתִּפֹּל עַל צַוָּארִי וַתִּשְׁאָלֵנִי לִשְׁלוֹם כַּלָּתִי. וָאֹמַר אֵלֶיהָ: כַּלָּתִי מֵתָה וַאֲנַחְנוּ עָרַכְנוּ לְעִלּוּי נִשְׁמָתָהּ כָּל-הַתְּפִלוֹת וְהָאַזְכָּרוֹת, וּכְבָר עָבְרוּ אַרְבָּעָה יָמִים מִיוֹם מוֹתָהּ. וַיְהִי כְשָׁמְעָהּ דְּבָרַי וַתִּצְעַק וַתִּקְרָא: הַאִם לֹא אָמַרְתִּי לְךָ כִּי אַתָּה הֲרַגְתָּהּ? וְלוּ הוֹדַעְתַּנִי דָבָר עַל אוֹדוֹתֶיהָ לִפְנֵי מוֹתָהּ כִּי אָז גְּמַלְתִּיהָ עַל גְּמוּלָה הַטּוֹב אִתִּי, כִּי לוּלֵא הִיא לֹא נִפְגַּשְׁנוּ עַד הַיּוֹם. וּמִי יוֹדֵעַ אִם לֹא תִפֹּל עוֹד בְּרָעָה עַל חַטָּאתְךָ הַגְּדוֹלָה אֲשֶׁר חָטָאתָ לָהּ. וָאֹמַר אֵלֶיהָ: אָכֵן סָלְחָה בַת דּוֹדִי לַעֲוֹנִי לִפְנֵי מוֹתָהּ. וְאַחַר הִגַּדְתִּי לָהּ אֵת אֲשֶׁר אָמְרָה אֵלַי אִמִּי. וַתֹּאמֶר: חֵי הָאֱלֹהִים, בְּלֶכְתְּךָ אֶל אִמְּךָ הִוָּדַע מִפִּיהָ מָה הוּא הַדָּבָר אֲשֶׁר נָתְנָה בְיָדָהּ לְמַעֲנֶךָ. וָאֹמַר לָהּ: אַף זֹאת הִגִּידָה לְאִמִּי לֵאמֹר: אִם יֹאמַר בְּנֵךְ לְהוֹסִיף לָלֶכֶת אֶל הַמָּקוֹם אֲשֶׁר הוּא הוֹלֵךְ אֵלָיו יוֹם יוֹם, הַגִידִי לוֹ כִּי לִפְנֵי לֶכְתּוֹ מִשָּׁם יַגִּיד אֶת-הַמִּלִּים הָאֵלֶּה:

“הָאֱמוּנָה מְפֹאָרָה / וְהַבְּגִידָה מְכֹעָרָה”.

וַיְהִי כִשְׁמֹעַ הַצְּעִירָה אֶת-הַמִּלִים הָאֵלֶה וַתִּקְרָא: יְרַחֵם הָאֵל אֶת-הַנֶּפֶשׁ הַטְּהוֹרָה הַזֹּאת! אָכֵן הִצִּילַתְךָ מִידֵי זַעְמִי, וְאָנֹכִי אָמַרְתִּי לְהָרַע לְךָ כְרָעָתֶךָ, וָאֶתְפַּלֵא לִדְבָרֶיהָ וָאֹמַר: וּמָה אָמַרְתְּ לַעֲשׂוֹת לִי וּבְרִית הָאַהֲבָה כְרוּתָה בֵינִי וּבֵינָיִךְ? וַתֹּאמֶר: אָכֵן נַפְשְׁךָ נְתוּנָה לְאַהֲבָתִי, אַךְ עוֹד צָעִיר הִנֶּךָ, וְלִבְּךָ אֵינֶנוּ יוֹדֵעַ עוֹד מִרְמָה וְעָרְמָה, וְתַרְמִיתֵנוּ וְתַחְבּוּלוֹתֵינוּ עוֹד נִסְתָּרוֹת מִלְפָנֶיךָ. וְעַתָּה זֹאת מִצְוָתִי אֲשֶׁר אֲצַוֶּךָּ: אַל תְּדַבֵּר עִם אִשָּׁה וְאַל תְּשׂוֹחֵחַ אִתָּהּ, אִם גְּדוֹלָה אִם צְעִירָה, כִּי טֶרֶם יָדַעְתָּ מִרְמַת הַנָּשִׁים וְנִכְלֵיהֶן,וְזֹאת אֲשֶׁר הָיְתָה פּוֹתֶרֶת לְךָ רְמִיזוֹתַי הֵן מֵתָה וְאַחֲרֶיהָ לֹא תִמְצָא עוֹד מִי שֶֹיְחַלֶצְךָ בַצָּר. וְעַתָּה עוֹד בַּקָּשָה אַחַת לִי אֵלֶיךָ, כִּי תֹבִילֵנִי אֶל קֶבֶר כַּלָּתְךָ, כִּי חָפַצְתִּי לְבַקְּרֵהוּ וְלִכְתֹּב עָלָיו בָּתִּים אֲחָדִים. וָאֹמַר: מָחָר אֶעֱשֶׂה כִדְבָרָיִך. וָאָלִין בְּבֵיתָהּ כָּל-הַלַּיְלָה, וּבְכָל-שָׁעָה הֵקִיצָה וַתֹּאמַר: מִי יִתֵּן וְהִגַּדְתָּ לִי עַל אוֹדוֹתֶיהָ לִפְנֵי מוֹתָהּ.

וָאֶשְׁאָלֵנָּה: מַה-פְּעֻלַּת הַמִּלִּים: הָאֱמוּנָה מְפֹאָרָה, וְהַבְּגִידָה מְכֹעָרָה? וְלֹא עָנַתְנִי.

וַיְהִי בַּבֹּקֶר וַתִּקָּח כִּיס דִּינְרֵי זָהָב וַתֹּאמֶר אֵלַי: קוּם וְהַרְאֵנִי אֶת-קִבְרָהּ. וָאֵלֵךְ לְפָנֶיהָ וְהִיא אַחֲרָי. וּבְלֶכְתָּהּ חִלְּקָה כֶסֶף לַעֲנִיִּים, וְכָל-עֵת תִּתָּהּ לְעָנִי אָמְרָה: זֹאת לְנִשְׁמַת עֲזִיזָה הַמַּסְתִּירָה סוֹדָהּ עַד מוֹתָהּ. וַיְהִי כַאֲשֶׁר הִגַּעְנוּ אֶל הַקֶּבֶר וַתִּפֹּל מְלוֹא קוֹמָתָהּ עָלָיו וַתֵּבְךְּ בְּכִי גַדוֹל וַתִּתְיַפֵּחַ וַתְּקוֹנֵן. וְאַחַר קָמָה וַתּוֹצִיא מִכִּיסָהּ אִזְמַל פְּלָדָה וּפַטִישׁ קָטָן וַתָּחָק בָּאֶבֶן הָרֹאשָׁה בְּאוֹתִיּוֹת דַּקּוֹת וְיָפוֹת אֶת-הַבָּתִּים הָאֵלֶּה:

אֵיךְ עֻפְּרָה עָפְרָה בְגוּשׁ עָפָר, וְעִם

בַּדֵּי שְׁאוֹל אֵיךְ הָיְתָה נֶחְבֶּקֶת,

אֵיךְ נָגְעָה בָהּ יָד אֲשֶׁר כֶּתֶם וְסִיג

אִישִׁים בְּעָפָר חָמְדָה לָצֶקֶת.

עָנָן עֲלֵי קִבְרָהּ יְסוֹכֵךְ כָּל-יְמֵי

אֶרֶץ, וְהָעָב לוֹ תְהִי מוּצֶקֶת.

וְאַחַר שָׁבָה וַתִּתְיַפֵּחַ, וּבְהַרְגִּיעַ רוּחָהּ קָמָה לָלֶכֶת וָאָקוּם גַּם אֲנִי, וַנָּשָׁב יַחְדָּו אֶל הַגָּן. וַתֹּאמֶר אֵלַי: רְאֵה בִקַּשְׁתִּיךָ בְשֵׁם הָאֱלֹהִים כִּי לֹא תַעַזְבֵנִי לְעוֹלָם. וָאַעַן: כֵּן, גְּבִרְתִּי.

וָאֱהִי בָא אֵלֶיהָ מֵעֵת לְעֵת, וּבְכָל-פַּעַם הֵיטִיבָה פָנֶיהָ אֵלַי. וַתִּשְׁאָלֵנִי לְמַּאֲמָר אֲשֶׁר אָמְרָה בַת דּוֹדִי לְאִמִּי לִפְנֵי מוֹתָהּ, וְאָנֹכִי שַׁבְתִּי וָאַגִּידֶנּוּ לָּהּ.

וָאֱהִי אוֹכֵל וְשׁוֹתֶה כָּל-הַיָּמִים וְלוֹבֵשׁ מַחֲלָצוֹת וּמִתְעַנֵּג עַל רֹב טוּב עַד כִּי שָׁמַנְּתִּי וְעָבִיתִי, וּבְלִבִּי אֵין דְּאָגָה וְאֵין דְּאָבָה. וְכַלָּתִי הַמֵּתָה נִשְׁכְּחָה מִלֵב בְּשָׁקְעִי בְיָם הַתַּעֲנוּגוֹת.

כָּכָה עָבְרָה עָלַי שָׁנָה תְמִימָה.

* * *

עוֹד שָׁהֵר-זָד מְסַפֶּרֶת וְהִנֵּה הֵאִיר בֹּקֶר יוֹם הַמֵּאָה וְעֶשְׁרִים וּשְׁנַיִם, וַתֶּחְדַּל מִסַּפֵּר סִפּוּרָהּ. וַתֹּאמֶר אֵלֶיהָ דֻנְיָה-זָד אֲחוֹתָהּ: מַה-טוֹב וּמַה נָּעִים סִפּוּרֵךְ זֶה! וַתֹּאמֶר שָהֵר-זָד: אַךְ אַחֲרִית סִפּוּרִי יָפֶה מִכָּל אֲשֶׁר סִפַּרְתִּי עַד כֹּה. וּמָחָר אֲסַפְּרֶנָּה לַאֲדוֹנִי הַמֶּלֶךְ.
