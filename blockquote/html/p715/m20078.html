כֹּה אָמַר אֲדֹנָי אֱלֹהִים

כֹּה אָמַר אֲדֹנָי אֱלֹהִים לַמִּתְבּוֹלְלִים הַלְּאֻמִּיִּים,

אֲשֶׁר לְאֻמִּיּוּתָם בְּפִיהֶם וּבְמַעֲשֵׂיהֶם הִתְבּוֹלְלוּת,

אֲשֶׁר יַהֲדוּתָם בִּלְשׁוֹנָם וּמִן הַשָּׂפָה וּלְחוּץ

וּבְקֶרֶב לִבָּם וּבְחַיֵּיהֶם דַּרְכֵי זָרִים וְהֹלֵלוּת:


יַעַן אֲשֶׁר מְאַסְתֶּם בְּרוּחַ יִשְׂרָאֵל וּבְקָדָשָׁיו

וְאֶת שֹׁרֶשׁ נִשְׁמָתוֹ עֲקַרְתֶּם וַתִּזְרוּ הָלְאָה,

וְאֶת נֵצַח יִשְׂרָאֵל הִשְׁלַכְתֶּם אַחֲרֵי גֵוְכֶם

כְנֵצֶר נִתְעָב, כִּכְלִי אֵין חֵפֶץ וּכְשִׂמְלָה בָּלָה,


וַתִּשְׁתַּחֲווּ לֶאֱלִילִים חֲדָשִׁים מִן הַגּוֹיִם

וּמִן הָאֲרָצוֹת אֲשֶׁר הָגְלִיתֶם שָׁמָּה,

וַתִּלְמְדוּ אֶל מִנְהֲגֵיהֶם וְאֶל מַחְשְׁבוֹתֵיהֶם,

וַתָּזִידוּ לַעֲשׂוֹת כְּמַעֲשֵׂיהֶם בְּיָד רָמָה;


יַעַן אֲשֶׁר עֲזַבְתֶּם מְקוֹר מַיִם חַיִּים טְהוֹרִים

וְנַפְשְׁכֶם נָקְעָה מִבּוֹר נֻקַּרְתֶּם הַיִּשְׂרְאֵלִי,

וַתֵּלְכוּ לַחְצֹב לָכֶם בֹּארוֹת חֲדָשִׁים לֹא־טְהוֹרִים

וַתֹּאמְרוּ לְכָל הַר וְעֵץ וְאָדָם יָפֶה בְעֵינֵיכֶם – אֵלִי,


וַתִּתְּשׁוּ מִקִּרְבְּכֶם אֶת רוּחַ אֲבוֹתֵינוּ הַקְּדוֹשִׁים

וַתִּשְׁפְּכוּ לָאָרֶץ אֶת נַפְשְׁכֶם הַיְּהוּדִית הַטְּהוֹרָה,

וַיִּתְרוֹקֵן לְבַבְכֶם מֵהַתֹּכֶן הַנַּעֲלֶה הַנִּבְחָר

וַתִּתְּנוּהוּ בִמְחִיר וַתַּעֲשׂוּ בוֹ סְחוֹרָה,


וַתְקַבְּלוּ תְמוּרָתוֹ נְשָׁמָה זָרָה, וְרוּחַ אַחֵר:

תְּשׁוּקוֹת וְתַאֲוֹת חֲדָשׁוֹת לֹא שְׁעָרוּן רִאשׁוֹנִים,

וַתִּרְדְּפוּ כָל הַיָּמִים אַחֲרֵי חֶמְדַּת בְּשָׂרִים

וּמַאֲוַיִּים שְׁפָלִים, בְּנֵי רֶגַע, הָיוּ לָכֶם לַאֲדוֹנִים,


וַתּוֹתִירוּ לָכֶם מִכָּל יַהֲדוּתְכֶם רַק מְעַט עִבְרִית

וּשְׁאִיפָה לָאָרֶץ, שֶׁאַתֶּם מְטַמְּאִים בְּתַעֲלוּלֵיכֶם,

כְּאוּדִים מֻצָּלִים מֵאֵשׁ תַּאֲוַתְכֶם הַשְּׁפָלָה,

הַבּוֹעֶרֶת וְאֵין מְכַבֶּה בִלְבַבְכֶם וּבְעַצְמוֹתֵיכֶם;


יַעַן שִׁאוֹל חֲטָאתֶם בְּכַבּוֹתְכֶם אֶת הַנִּיצוֹץ הָאַחֲרוֹן

בְּלֵב יְלָדֵינוּ הָרַכִּים וְאֶת נִשְׁמָתָם אַתֶּם מְמוֹתְתִים,

בְּמָסְרְכֶם אֹתָם לְמֹּלֶךְ הַכְּפִירָה, לְתֹהוּ וָבֹהוּ,

לֶאֱלִיל הַבָּשָׂר וְהָרֶגַע וְאֵין אַתֶּם חוֹשְׁשִׁים

בִּכְבֹת חָלִילָה הַגַּחֶלֶת הָאַחֲרוֹנָה

לַיַּהֲדוּת הָאֲמִתִּית, הָרוֹמֵמָה וְהַקְּדוֹשָׁה,

שֶׁעָלֶיהָ מָסְרוּ אֲבוֹתֵינוּ אֶת נַפְשָׁם בְּשִׂמְחָה וַעֲלִיצוּת,

וְעוֹד אַתֶּם לוֹעֲגִים לָהּ, לְגוֹרָלָהּ וְלִמְנַת כּוֹסָהּ;


יַעַן שֶׁהִתְמַכַּרְתֶּם לַעֲבָדִים, הַלְּאֻמִּיִּים מִּתְבּוֹלְלִים,

לֶאֱלִילֵי הַבָּשָׂר וְהָרֶגַע, לְאוֹיְבֵי רוּחַ יִשְׂרָאֵל,

וְאֶת נַפְשְׁכֶם הַיְּהוּדִית וְאֶת נֶפֶשׁ הָאֻמָּה אַתֶּם זוֹלְלִים

וּמְכַלִּים וּמַשְׁמִידִים, וְכֹה אַתֶּם חוֹשְׁבִים לְהִגָּאֵל;


יַעַן וּבְיַעַן שֶׁחֵרַפְתֶּם מַעַרְכוֹת אֱלֹהִים חַיִּים

וַתֹּאמְרוּ: רַק בְּכֹחַ עַצְמֵנוּ נִתֵּן לָנוּ פְדוּת – נִגָּאֵל,

וַתַּחֲלִיפוּ הַרוּחַ בַּחֹמֶר וּבָאָרֶץ הַשָּׁמַיִם,

וַתַּעֲמִיקוּ כִשְׁאוֹל חַטָּאָה לְרוּחַ, לְנֵצַח יִשְׂרָאֵל;


לָכֵן, כֹּה אָמַר אֲדֹנָי, הִנְנִי עֲלֵיכֶם בָּנִים סוֹרְרִים,

תִּקְצְרוּ לַבֶּהָלָה בִידֵיכֶם אֵת אֲשֶׁר זְרַעְתֶּם,

תִּמַּקּוּ בַעֲוֹנֵיכֶם וְתָסוּפוּ בְתַאֲוֹתֵיכֶם

וְתַעֲלוּ עַל הַמּוֹקֵד בְּאוּר אֶשְׁכֶם אֲשֶׁר הִצַּתֶּם.


וְרוּחִי אֶשְׁפֹּךְ עַל נֶאֶמְנֵי וְגִבּוֹרֵי הָרוּחַ הָרָמִים,

שְׂרִידֵי עַמִּי הַנִּבְחָר, וְלִרְגָעִים אֹתָם אֲבַקֵּר,

וְשַׂמְתִּי לָהֶם פְּדוּת בְּהַר קָדְשִׁי וְהָיוּ לְאוֹר עוֹלָמִים,

כִּי פִי אֲדֹנָי דִּבֵּר, וְנֵצַח יִשְׂרָאֵל לֹא יְשַׁקֵּר!
