[איכה כלילת יופי מתנכרה לעין כל בשר]

מגן

אֵיכָה כְּלִילַת יֹפִי מִתְנַכְּרָה לְעֵין כָּל בָּשָׂר

בָּא שִׁמְשָהּ בְּעוֹד יוֹמָם וּמַחְמַד עֵינֶיהָ הוּסָר

גָּעַר בָּה בֹּעֲלָה וַיִּתְּנָהּ בְּמַאֲסָר

אֵין מֶלֶך וְאֵין שָׂר

דְּלָקוּהָ אוֹיְבִים וַיְעוֹרֲרוּ עָלֶיהָ אֵיבָה

הָטְבְּעוּ בַבֹּץ רַגְלֶיהָ וַתִּשְׁכַּב לְמַעֲצֵבָה

וְהָעִיר הַהֻלָּלָה נוֹתְרָה כַּמִּדְבָּר חֲרֵבָה

אֵין זֶבַח וְאֵין מַצֵּבָה

זְמִירֵי עָרִיצִים לִסְפּוֹתָהּ נֶאֱסָפִים

חָלְפוּ זְמַנֶּיהָ וִיגוֹנֶיהָ לֹא נֶחֱלָפִים

טְרָפוּהָ לְבָאִים וּמַכְאֹבֶיהָ נֶחְשָׂפִים

אֵין אֵפוֹד וּתְרָפִים.

כרוג

יָחִיד הֲנִחַם לְהוֹתִיר לָהּ פָּלִיט

כִּי בָאוּ בָנִים עַד מַשְׁבֵּר וְכֹחַ אֵין לְהַפְלִיט

לְגוֹנֵן עֲלֵיהֶם בְּרַחֲמֶיךָ שַׁדַּי הַשָּלִּיט

גָּנוֹן וְהִצִּיל פָּסוֹחַ וְהִמְלִיט.

מחיה

מִי נָתַן לִמְשִׁסָּה יַעֲקֹב בְּקַצְוֵי תֵבֵל

נֶעֱזָב כָּאֳנִי בְלֶב יָם מִבְּלִי חֹבֵל

סָפַדְנוּ בְגָלוּתֵנוּ חֲפוּי רֹאשׁ וְאָבֵל

עַל נַהֲרוֹת בָּבֶל

עַל נַהֲרוֹת בָּבֶל עֵינֵינוּ בַּדְּמָעוֹת כִּלִּינוּ

פְּקַדְנוּךָ אָהֳלִיבָה וְאָהֳלָה עַל לֵב הֶעֱלִינוּ

צִלְצְלֵי שֶׁמַע הֶחֱרַשְׁנוּ וְכִנוֹרוֹתֵינוּ תָלִינוּ

שָׁם יָשַׁבְנוּ גַּם בָּכִינוּ

קָרָאנוּ לִנְהִי עַד אֵין לָנוּ בִּבְנוֹת יַעֲנָה דִּמְיוֹן

רַבּוּ אַנְחֹתֵינוּ עַל שְׂרֵפַת מִקְדַּשׁ עֶלְיוֹן

שִׁבְרֵנוּ גָדַל וְכִמְעַט הָיִינוּ לְכִלָּיוֹן

בְּזָכְרֵנוּ אֶת צִיּוֹן.

כרוג

תִּשְׁכַּח יְמִינֵנוּ אִם נִשְׁכָּחֵךְ יְרוּשָׁלַיִם

יְמֵי הוֹדֵךְ וְקֵץ דְּרוֹרֵךְ הוֹחַלְנוּ מִשּׁמָיִם

הִתְנַפַּלְנוּ לִפְנֵי אֵל וְעַפְעַפֵּינוּ יִזְלוּ מָיִם

יְחַיֵּנוּ מִיֹמָיִם.

סלוק

וּבְכֵן לְךָ תַעֲלֶה קְדֻשָּה כִּי אַתָּה אֵל מֶלֶך מֹחֵץ וְרֹפֵא

אֲהָה אֲדֹנָי אֱלֹהִים הֲלָעַד זְנַחְתָּנוּ פְּרַצְתָּנוּ

וְרַחֲמֶיךָ לְעַמֶּךָ מֵרָחֹק עָמָדוּ

נִאַצְתָּ בֵן לְפָנִים אִמַּצְתָּ אֵין קֶרֶן וְאֵין סֶרֶן אֵין שָׂר

וְאֵין טִפְסָר כִּי מִפְּנֵי אֵשׁ קֶצְפְּךָ וְרוּחַ אַפְּךָ יִתְפָּרָדוּ

יַחַד מְבַלְּעַי וּמְגַדּעַי רַבּוּ וְנֵרוֹתַי כִּבּוּ וְכִדְבוֹרִים סַבּוּ

וְעַל שְׁאֵרִית בְּנֵי הַבְּרִית לְהַכְרִית וּמִגּוֹי לְהַכְחִיד שָׁקָדוּ

יִסְּרוּנִי בָעַקְרַבִּים חֲבוּרֵי עֲצַבִּים וְנַפְשִׁי בְּמוֹקְשִׁי רְצוּצָה

לְחוּצָה קְצוּצָה רְעוּצָה וּפַח וָפַחַד חַד בְּחַד יִתְלַכָּדוּ

הוֹרוּנִי בְּחִצֵּי מָוֶת וַיְנַהֲגוּנִי לְצַלְמָוֶת רֻטַּשְׁתִּי וְנֻטַּשְׁתִּי

מֵרְוָיָה אֶל אֶרֶץ צִיָּה וְרַגְלַי וְקַרְסֻלַּי מָעָדוּ

וַאֲצַפֶּה וְאוֹחִיל לְמַרְפֵּא וְהִנֵּה חִיל וְלִישׁוּעָתָה וְהִנֵּה בְעָתָה

וָאֱהִי כְעוֹף נוֹדֵד עַל גַּג בּוֹדֵד וְקוֹל נְהִי אָרִים לְאַדִּירִים אֲשֶׁר שֻׁדָּדוּ

דִּגְלִי נָפַל וְאוֹרִי אָפַל וְגָבְהִי שָׁפַל וְלִבִּי בְקִרְבִּי

נֶחֱשַׁל וּבִרְכִּי נִכְשַׁל וְנֶחֱשַׂף סֻכִּי וְכִבָּה חָשְכִּי נֵרוֹת עֶרְכִּי

אַף רִשְׁפֵּי יְגוֹנַי נוֹקָדוּ

הֲדִמּוּנִי וּפִרְפְּרוּנִי וּפִצְפְּצוּנִי מְרַדְּפַי וּמְגַדְּפַי

וּמְנַגְּפַי בֹּזְזֵי יְקָרִי וְגֹזְזֵי נִזְרִי וְהֹזִים

עַלִּיזִים לְשַבֵּר הַבַּת-עֵבֶר וּבָנַי נִפְקָדוּ

בָּרְחוּ וְאָרְחוּ וְשָׁכְחוּ רִבְצָם וְכֻלָּם כִּמְעַט קָט

אָבָדוּ כִּי מִפְּנֵי חֲרָבוֹת נָדָדוּ

נָדְדוּ וְאֵינָם וְהֻמְּכוּ בַּעֲוֹנָם וְרַגְלָם בְּפֵח זְדוֹנָם יְקוּשָׁה

שְׁלוֹמִים רִחַק וְהַצָּר דָּחַק וְעַצְמָם שִחֵק וְשָׂמָהּ כְּתוּשָׁה

מִשְׂרָה לְקֻחָה וְעָרְבָה כָל שִׂמְחָה וַתְּהִי כְּשִׁפְחָה כְּבוּשָׁה

וְנִזְרִי חֻלַּל וְעַמִּי שֻׁלַּל וְחֶלְקִי קֻלַּל וַיְהִי לְזָר יְרֻשּׁה

אֶשְׁכְּבָה לֹהֲטִים דְּרָכַי מַטִּים מְיַסְּרִים בְּשׁוֹטִים וּמַכָּתִי אֲנוּשָׁה

לִי קִוּוּ רְשָׁעִים לְאַבְּדֵנִי וְכָל יוֹם יָדָם תְּנִדֵנִי וַאֲנִי

בְּכָל יוֹם חֲלוּשָׁה מִפְּנֵי חֶרֶב נְטוּשָׁה

הַנְּטוּשָׁה וְהַצֹּלֵעָה מֵאֵין מַרְגֵּעָה וְהַנֵּהֲלָאָה מִכֹּבֶד

מַשָּׂאָה בְּחוּגַת עֲנִיוּת וְאַלְמְנוּת חַיּוּת כְּאִשָּׁה מֵאִישָׁהּ גְּרוּשָׁה

לְכוּדָה בְּמוֹקֵש פְּתַלְתֹּל וְעִקֵּש נְשֻׂאַת סֵבֶל מְעֻנַּת כֶּבֶל

אֲחֻזַת חֶבֶל סְפָרַד וּבָבֶל וּבְגְדֵי אֵבֶל לְבֻשָׁה

וּבְכָל זֹאת אֱלֹהֶיהָ לֹא שָׁכְחָה רַק בִּשְׁמוֹ בָטְחָה וְטוּבוֹ

לֹא זָנְחָה וְתוֹחִיל פִּדְיֹום בְּכָל יוֹם וָיוֹם אוּלַי יְמַהֵר יָחִישָׁה

יְדוּעַת חֳלָאִים מְשׁוּלַת פְּרָאִים חֲלוּצַת עֲדָיִים לְבֻשַׁת בְּלוֹיִים

אֳזוֹרָהּ נִתַּק וּמַחֲגֹרֶת שַׂק חֲבוּשָׁה

הֹגֶה פִּלְאֵי אֵל קְדוֹשׁ יִשְׂרָאֵל וְעַל יִחוּד שְׁמוֹ תִּשּׂא

אֶבֶן מַעֲמָסָה וּבַאֲרָיוֹת וְדֻבִּים וְצִפְעֹנִים וְעַקְרַבִּים פְּגוּשָׁה

קְדֻשָּׁה מְשַׁלֶּשֶת וְאִם בָּאֵפֶר וּבְשַׂק מִתְפַּלֶּשֶׁת חֲרֵדָה לִדְבַר צוּר

וְיִצְרָה מִתַּרְמִית נָצוּר וּבְכָל יוֹם לְמִקְדָּשׁ מְעַט רְגוּשָׁה

טוּב טַעַם תְּנוֹבֵב וְלִקְרַאת אֱלֹהֶיהָ תְּיַבֵּב וּתְמוּרַת

קָרְבְּנוֹתֶיהָ פְּרִי שְׂפָתֶיהָ הִגִּישָׁה

נֹאֶמֶת בְּאֵימָה בְּפַחַד בְּזַחַל בְּרַעַד מְשַׁלֶשֶׁת קְדֻשָּה לְאֵל

נַעֲרָץ בְּסוֹד קְדֹשִׁים רַבָּה וְנוֹרָא עַל כָּל סְבִיבָיו.
