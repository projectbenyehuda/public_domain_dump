וִינָה

עִם הַתְּרוּמָה לְפוֹעֲלֵי וִינָה

הַתְּרוּמָה, הַמְפֹרָטָה בְּסֵפֶר זֶה, הִיא רֻבָּהּ כְּכֻלָּהּ פְּרִי הִתְנַדְּבוּת שֶׁל יוֹם אֶחָד, יוֹם שִׁשִּׁי, א' אֲדָר תרצ"ד. זוֹ לֹא הָיְתָה מַגְבִּית בְּנֻסָּח מְקֻבָּל, כִּי מַשְׂאַת אַחֲוָה שֶׁל פּוֹעֲלֵי אֶרֶץ־יִשְׂרָאֵל, לְהַבָּעַת אֹמֶן־רוּחָם לְחַבְרֵיהֶם בְּאוֹסְטְרִיָּה, אוֹסְרֵי הַקְּרָב עַל הֲגָנַת קִנְיְנֵי הָאָדָם־הַפּוֹעֵל וְעַל שְׁמִירַת עֲרֻבּוֹת חֵרוּתוֹ וְקוֹמְמִיּוּתוֹ, לְבַל תּוּבַסְנָה בְּרַגְלֵי עָרִיץ.

אוֹסְרֵי הַקְּרָב הַנּוֹאָשׁ וּמְקַדְשָׁיו בְּדָמָם, בְּנַפְשָׁם וּבְכָל מְאֹדָם.

רַק צִבּוּר פּוֹעֲלִים, אֲשֶׁר גּוֹרָלוֹ מִנָּה לוֹ לָשֹר אֶל תְּנָאִים הִסְטוֹרִיִּים חֲמוּרִים בְּיוֹתֵר וּלְהַתְמִיד בְהַשְׁקָעַת כֹּחוֹת יְצִירָה וְחָזוֹן בַּמַּהְפֵּכָה שֶׁלּוֹ – וְהִיא מַהְפֵּכַת־בִּנְיָן – עָלוּל לַחֲרֹד כָּכָה לְפִרְפּוּרֵי צִבּוּר־עוֹבְדִים־אָח, הַנָּתוּן בְּאֵשׁ גּוֹרָלוֹ.

בְּלִי אֹמֶר וּבְלִי דְבָרִים נֶהְפַּךְ מַחֲנֵנוּ כֻּלּוֹ, כִּמְעַט לְאֵין יוֹצֵא מִן הַכְּלָל, מִצְּעִיר שְׂכִירֵי־הַיּוֹם בָּעִיר וּבַמּוֹשָׁבָה וְעַד זְקַן וְתִיקֵי הַמֶּשֶק הָחַקְלָאִי הָעַצְמִי, לַאֲבוּקָה אַחַת, דּוֹלֶקֶת וּמְאִירָה מִבִּפְנִים.

נֶחְשְׂפָה וְנִתְגַלְּתָה, מִבַּעַד לְכָל רָבְדֵי־חֻלִּין אֲפוֹרִים, הַמַּהוּת הַיְּהוּדִית שֶׁל תְּנוּעַת הַפּוֹעֵל הָעִבְרִי בָּאָרֶץ:

אֱמוּנָה לְלֹא־חָת בִּגְאֻלַּת עוֹלָם, אֲשֶׁר גְּאֻלַּת יִשְׂרָאֵל מְשֻלֶּבֶת וּמְעֹרָה בְּתוֹךְ תּוֹכָהּ, לִבְלִי הַפְרִיד בֵּינֵיהֶן, וְנֶאֱמָנוּת לְאַחְדוּת בְּרִיתָם הָעוֹלָמִית שֶׁל הַפּוֹעֲלִים, נוֹשְׂאֵי דִגְלָהּ שֶׁל גְּאֻלָּה זוֹ.

כִּי עָלֶיהָ, עַל גְּאֻלַּת עוֹלָם זוֹ, וְעַל דֶּגֶל חֲזוֹן אַחֲרִית הַיָּמִים מָסְרוּ אֶת נַפְשָׁם וְהוּצְאוּ לַהוֹרֵג קְדוֹשֵׁי־לוֹחֲמֵי־פּוֹעֲלֵי־אוֹסְטְרִיָּה.

אדר תרצ"ד.
