אֶתֵּן לְאֵלִי

סימן: אלף בית; אני שלמה הקטן (ב)ן יהודה בן גבירול חזק חרזתי הענק; אלף בית; שלמה; אלף בית;…


אֶתֵּן לְאֵלִי עֹז וְגַם תִּפְאֶרֶת / הָאֵל אֲשֶׁר תִּכֵּן זְבוּל בַּזֶּרֶת

בָּרָא שְׂפָתַיִם וְשָׂם פֶּה לֶאֱנוֹשׁ / וַיְּעַטְּרֵם הָדָר וְהוֹד לַעֲטֶרֶת

גַּם לִמְּדָם דַּעַת לְהָבִין נִפְלְאוֹת / אֵל יוֹצְרָם בָּזֹאת וּבָאַחֶרֶת

דִּבֵּר שְׁלֹמֹה הַסְּפָרַדִּי אֲשֶׁר / אָסַף שְׂפַת קֹדֶשׁ לְאֹם נִפְזֶרֶת

הִשְׁקִיף לְבָבִי עַל עֲדַת צוּר, וָאֱהִי / סוֹבֵר פְלֵטָתָם אֲשֶׁר נִשְׁאֶרֶת

וָאֵדְעָה כִּי נִשְׁמְרָה מֵהֶם שְׂפַת / קֹדֶשׁ וְכִמְעַט הָיְתָה נֶעְדֶּרֶת

זָרָה לְשׁוֹנָם מִלְּשׁוֹן עִבְרִית וְלֹא / לִשְׂפַת יְהוּדִית הָיְתָה מַכֶּרֶת

חֶצְיָם מְדַבֵּר בָּאֲדוֹמִית וַחֲצִי / בִלְשׁוֹן בְּנֵי קֵדָר אֲשֶׁר קוֹדֶרֶת

טָבַע סְגוֹר לִבָּם בְּמַעְמַקֵּי תְהוֹם / גַּם צָלְלוּ בָהֶם כְּמוֹ עוֹפֶרֶת

יָסַף כְּאֵבָם עַל כְּאֵבִי וַיְהִי / עָצוּר בְּכִלְיוֹתַי כְּאֵשׁ בּוֹעֶרֶת

כִּרְאוֹת פְּתָאִים יֶהֱמָיוּן יֶהֱמֶה / לִבִּי כְּכִנּוֹר אוֹ כְיָם כִּנֶּרֶת

לֹא יָדְעוּ חָזוֹן וְגַם לֹא יָדְעוּ / סֵפֶר, וְאֵיכָה יִקְרְאוּן אִגֶּרֶת

מִי יַעֲלֶה עִוְּרִים בְּלֵב יָם טָבְעוּ / יַד מִי לְהוֹצִיא הָאֳנִי חוֹתֶרֶת

נוֹעַץ לְבָבִי: עֵינְךָ כִּי נִפְקְחָה / לֹא טוֹב הֱיוֹת עֵין עַמְּךָ עִוֶּרֶת

שִׂים פֶּה לְפִיּוֹת נִסְכְּרוּ כָאִלְּמִים / כִּי יֵשׁ בְּזֹאת מֵאֵל לְךָ מַשְׂכֹּרֶת

שַׂחְתִּי, בְּשׁוּרִי כִּי אֲנִי צָעִיר לְיוֹם, / וָאֶגְעֲרָה בוֹ בַחֲמַת מִגְעֶרֶת

עַל כִּי לְבַב נַעַר כְּבַעַר נֶחְשָׁב / יַד בֶּן תְּשַׁע עֶשְׂרֵה מְאֹד קָצֶרֶת

פָּנָה לְבָבִי עַד קְרָאַנִי חֲלוֹם / וָאֶשְׁמְעָה קוֹל יַעֲבֹר עַל קֶרֶת

צוֹעֵק בְּאָזְנַי בֶּאֱשׁוּן לֵיל: קוּם עֲשֵׂה / כִּי יַד אֱלוֹהַּ אוֹתְךָ עוֹזֶרֶת

קוּמָה וְאַל תֹּאמַר הֲלֹא נַעַר אֲנִי / כִּי לֹא זְקֵנִים יִתְמְכוּ כוֹתֶרֶת

רָמָה לְזֹאת קַרְנִי מְאֹד וָאֵדְעָה / כִּי מֵאֲדֹנָי הָיְתָה נִגְזֶרֶת

שַׁתִּי לְפִי שִׂכְלִי, וְאָמְנָם יַד אֱנוֹשׁ / עַל פִּי אֲשֶׁר תַּשִּׂיג תְּהִי נוֹדֶרֶת

תַּרְתִּי בְלִבִּי לַעֲשׂוֹת סֵפֶר בְּסוֹד / דִּקְדּוּק שְׂפַת קֹדֶשׁ הֱיוֹת נִבְצֶרֶת

אוּלָם בְּדַעְתִּי כִּי לְשׁוֹן עִבְרִית לְבַד / מִכָּל לְשׁוֹנוֹת הִיא לְטוֹב נִכֶּרֶת

נִמְלַךְ לְבָבִי וָאֲכִינֵהוּ בְמוֹ / עִבְרִית, בְּיַעַן הִיא לְבַד נִבְחֶרֶת

יַעַן רְאוֹתִי הַחֲרוּז יֶתֶר לְכָל / דָּבָר, וְאִם דַּרְכּוֹ מְאֹד נִפְצֶרֶת

שָׁקוֹל שְׁקַלְתִּיו וָאֲשִׂימֵהוּ לְשִׁיר / חָרוּז, לְמַעַן לִהְיוֹת מַזְכֶּרֶת

לָכֵן חֲלַקְתִּיהוּ לְעֶשֶׂר מַחְלְקוֹת / וָאֶעֱשֶׂה-לוֹ זֵר כְּמוֹ מִסְגֶּרֶת

מִכְתָּם-עֲלֵי מִכְתָּם וְחָרוּז עַל-פְּנֵי / חָרוּז כְּמוֹ דַר עַל פְּנֵי סֹחֶרֶת.

הִיא כַעֲרוּגַת גַּן אֲשֶׁר תֵּרָא בְכָל / עַיִן, וְכָל עַיִן תְּהִי שׂוֹבֶרֶת

הַהוֹלְכִים בָּהּ יִמְצְאוּ מִינֵי הֲדַס / כֹּפֶר וְשׁוֹשָׁן וַעֲצֵי צַמֶּרֶת

קְרָאתִיו ״עֲנָק״ שֶׁהֶעֱנַקְתִּיהוּ אֱנוֹשׁ / גַּם כַּעֲנָק שַׂמְתִּיו עֲלֵי גַרְגֶּרֶת

טַל יִזְּלוּ מִלָּיו בְּיוֹם שָׁרָב וְחֹם / גַּם חַשְׁרַת מַיִם בְּעֵת בַּצֹּרֶת

נִצָּב לְפִי קֶרֶת כְּמוֹ חָכְמָה אֲשֶׁר / עַל בֵּית נְתִיבוֹת נִצְּבָה אוֹמֶרֶת

נִבְחָר מְאֹד מִכָּל כְּלִי חֵפֶץ וְלֹא / יֻתַּן סְגוֹר תַּחְתָּיו מְחִיר מִמְכֶּרֶת

יִשְׂבַּע קְלוֹן נָבוּב, וְאִם יֶעְשַׁר מְאֹד, / וִימִין יְקַר לֵבָב יְקָר עוֹזֶרֶת

הֵחָלְצוּ, בָנִים, וְאַל תֵּעָצְלוּ / כִּי נַפְשְׁכֶם אָמְנָם לְזֹאת נוֹצֶרֶת

וּבְכֵן דְּעוּ יִתְרוֹן לְשׁוֹן עִבְרִית אֲשֶׁר / הִיא מִלְּשׁוֹן כָּל עַם וְעָם יוֹתֶרֶת

דָּרֵי שְׁחָקִים יִתְּנוּ בָהּ יוֹם בְּיוֹם / אֶדֶר לְעוֹטֶה אוֹר כְּמוֹ אַדֶּרֶת

הִיא הָיְתָה מֵאָז שְׂפַת כָּל הַיְּקוּם / עַד נִפְלְגוּ אַנְשֵׁי עֲצַת נִבְעֶרֶת

בָּלַל שְׂפָתָם אֵל, לְבַד בִּלְשׁוֹן בְּנֵי / עֵבֶר לְבַדָּהּ הָיְתָה נוֹתֶרֶת

נָחְלָהּ וְהִנְחִילָהּ לְבָנָיו אַב הֲמוֹן / גַּם הָיְתָה מִדּוֹר לְדוֹר נִשְׁמֶרֶת

גַּם לֹא עֲזָבָה גָּר בְּאֶרֶץ לוֹעֲזִים / יוֹם נִלְכְּדָה רַגְלוֹ בְתוֹךְ כּוֹשֶׁרֶת

בָּהּ נִתְּנָה אֵשׁ דָּת וְגַם בָּהּ נִשְׁלְחוּ / כָל הַנְּבִיאִים לַחֲבֹשׁ נִשְׁבֶּרֶת

יַד תּוֹפְשֵׂי כִנּוֹר הֲלֹא בָהּ נִגְּנוּ / יוֹם הָיְתָה אֶרֶץ צְבִי לִגְבֶרֶת.

רִיב יֵשׁ לָאֵל בָּכֶם, שְׁאֵרִית יַעֲקֹב, / אִם תִּשְׁכְּחוּ שָׂפָה מְאֹד נִבְחֶרֶת

וּרְאוּ אֲבוֹתֵיכֶם בְּעֵת כִּי מָרְדוּ / וַיִּתְּנוּ כָתֵף אֲשֶׁר סוֹרֶרֶת

לָכֵן הֱבִישָׁם בֶּאֱמֹר כִּי לַעֲגֵי / שָׂפָה בְּסוֹד חָזוּת אֲזַי נִדְבֶּרֶת.

חִזְקוּ וְאַל תַּרְפּוּ יְדֵיכֶם לַעֲזֹב / כִּי לֹא נְקַלָּה הִיא אֲבָל נוֹקֶרֶת.

רַב לַעֲזֹב מָקוֹר וְלַחְצֹב כָּל בְּאֵר / עַל חוֹצְבָהּ פִּיהָ תְהִי אוֹטֶרֶת.

זִכְרוּ לְקִנְאַת בֶּן חֲכַלְיָה יוֹם רְאוֹת / בָּנִים לְשׁוֹנָם נֶהְפְּכָה אַחֶרֶת

תָּפַשׂ אֲנָשִׁים בָּהּ וְהִכָּם בָּחֳרִי / קֶצֶף וּבִמְרִיבָה מְאֹד נֶעְתֶּרֶת.

יִיטַב הֱיוֹת אֻמָּה גְבִירָה שׁוֹכְחָה / דָבָר וּפִלַּגְשָׁהּ תְּהִי שׁוֹמֶרֶת.

הָהּ לָהּ אֲשֶׁר לֹא נָטְרָה כַרְמָהּ אֲבָל / כַּרְמֵי אֲחֵרִים הָיְתָה נוֹטֶרֶת

עַד יַעֲשֶׂה שַׁדַּי וְיַשְׁלִים מוֹעֲדוֹ / צַחוֹת לְשׁוֹן עִלְגִּים תְּהִי דוֹבֶרֶת

נֶגְדּוֹ אֲנִי מַפִּיל תְּחִנָּתִי וּבוֹ / אֶבְטַח וְנַפְשִׁי בּוֹ מְאֹד נֶעְזֶרֶת

קוֹל יַאֲזִין לָשׂוּם בְּפִי עַבְדּוֹ אֱמֶת / כִּי אֶפְתְּחָה פִי לַעֲשׂוֹת מַחְבֶּרֶת.


אוּלָם אֲבָאֵר סוֹד, בְּטֶרֶם אֶחְקְרָה / מִלִּין, בְּעֹז יָמִין מְאֹד נֶאְדֶּרֶת

בִּהְיוֹת יְסוֹד כָּל יֵשׁ וְשֹׁרֶשׁ כָּל יְקוּם / עֶצֶם וְקָרְמָה בַּעֲדוֹ נֶאֱסֶרֶת

גָּרַם הֱיוֹת הַשֵּׁם וְהַפֹּעַל וְגַם / מִלָּה שְׁלִישִׁית זֶה בְּזֶה קוֹשֶׁרֶת

הָיוּ לְאַרְבָּעָה חֲלָקִים, וַאֲנִי / גַּם כֵּן חֲלַקְתִּים לָךְ בְּזֹאת אִגֶּרֶת

וָאֶחֱלֹק אוֹתָם לְאַרְבַּע מַחְלְקוֹת / אֶת זוֹ לְעֻמַּת זוֹ תְהִי חוֹבֶרֶת

זָכוֹר בְּרִאשׁוֹנָה לְאוֹתִיּוֹת וְגַם / כָּל תּוֹלְדוֹתֵיהֶם בְּדָת נִכְשֶׁרֶת

חֹק הַשְּׁנִיָּה הַחֲקוּקָה אַחֲרָיו / בַּשֵּׁם וְתוֹצְאוֹתָיו תְּהִי עוֹצֶרֶת

טַעַם שְׁלִישִׁיָּה בְּפֹעַל יֻסְּדָה / וּבְכָל נְסִבָּה בַעֲדוֹ נֶאְמֶרֶת

יִקְרַת רְבִיעִית דָּת חֲקַקְתִּיהָ / מִלָּה אֲשֶׁר פֹּעַל לְשֵׁם אוֹסֶרֶת

כְּפִי תוֹצְאוֹת אַחַת לְמוּל אַחַת יְהִי / מִסְפַּר שְׁעָרֶיהָ אֲשֶׁר אוֹגֶרֶת

לָכֵן הֲכִינוֹתִי בְרֹאשׁ הַמַּחְלְקוֹת / שִׁשָּׁה שְׁעָרִים בַּעֲדָהּ גּוֹדֶרֶת

מִסְפַּר שְׁתֵים עֶשְׂרִים וּמוֹצָאָם בְּפֶה / חָקוּק בְּרֹאשָׁם אוֹת לְאוֹת נִזְכֶּרֶת

נְצֹר כִּי לְחָמֵשׁ מַחְלְקוֹת יֵחָלְקוּ / בַפֶּה כְּעֵדוּת בַּעֲלֵי מַסֹּרֶת

סְפוּחוֹת עֲלֵי גָרוֹן וְעַל חֵךְ אַחֲרָיו / גַּם עַל קְצָת לָשׁוֹן וְשֵׁן זוֹרֶרֶת

עִנְיַן חֲמִישִׁית רַק לְשָׂפָה נוֹדְעָה / וּבְאַחֲרִית כֻּלָּהְנָה נִגְמֶרֶת

פִּתְרוֹן פְּקוּדֵיהֶם יְסוֹד אַהְחַ״ע לְנִיב / גָּרוֹן, וְסוֹד גִּיכָ״ק לְחֵךְ צוֹרֶרֶת

צוּרָה שְׁלִישִׁית דַּטְלְנַ״ת הִיא נִקְרְאָה / לַחֵךְ וְלַלָּשׁוֹן אֱמֶת נִסְפֶּרֶת

קֶצֶב יְצִיאֵי שֵׁן חֲקוּקִים זַסְצְרַ״שׁ / וּמְקוֹם יְסוֹד בּוּמָ״ף שְׂפַת נִסְגֶּרֶת

רַק נִשְׁלְמוּ חָמֵשׁ וְעוֹדָם נֶחְלְקוּ / לִשְׁנֵח חֲלָקִים עַל פְּאַת אַחֶרֶת

שֶׁבָּם נְטוּעִים כִּמְלָכִים נִצְּבוּ / וּמְשָׁרְתִים עוֹבְרִים בְּלִי מַעְבֶּרֶת

תַּתִּים אֲגֻדָּה וָאֲשִׂימֵם עַל שְׁמִי / לִהְיוֹת לְכָל מַשְׂכִּיל לְאוֹת מִשְׁמֶרֶת

שֵׁבֶט מְשָׁרְתִים הוּא אֲנִי שְׁלֹמֹה וְגַם / כּוֹתֵב, מְקֻשָּׁרִים בְּתוֹךְ מָסֹרֶת

לַהֲקַת מְלָכִים רַק תְּהִי קָט צַח וְגַם / גֶּזַע סְפָרַד בַּעֲבוֹת נִקְשֶׁרֶת

מָקוֹם מְשָׁרְתִים מַחֲלִיפִים זֶה לְזֶה / עַל פִּי עֲבוֹדָתָם אֲשֶׁר נִסְדֶּרֶת

הִנֵּה אֲבָאֵר כָּל עֲבוֹדָתָם הֲלוֹם / עַל מִשְׁמְרוֹתָם אוֹת לְאוֹת נֶעְדֶּרֶת

אֵיךְ מִשְׁמְרוֹת אָלֶף אֲשֶׁר בָּם תַּעֲבֹד / תָּמִיד בְּמִשְׁמַרְתָּהּ תְּהִי נֶאְסֶרֶת

בְּאֶחָד לְאַרְבָּעָה עֲתִידִים תַּעֲבֹד / לַשֵּׁם וְלַפֹּעַל תְּהִי נִמְסֶרֶת

גַּם תַּעֲבֹד פֹּעַל אֲשֶׁר הָיָה כְבָר / אָלֶף בְּ״אֶתְחַבֵּר״ לְהֵי נִפְתֶּרֶת

דָּתָהּ לְהוֹסִיפָהּ בְּשֵׁם פֹּעַל כְּמוֹ / אָדוֹשׁ יְדוּשֶׁנּוּ וּבוֹ נֶעְצֶרֶת

הֵן נִסְפְּחָה גַם עַל יְסוֹד הַשֵּׁם כְּמוֹ / אֶתְמוֹל וְאֶזְרוֹעַ הֱיוֹת נִבְרֶרֶת

וּבְכֵן אֲבָאֵר מַחְלְקוֹת הַנּוּן לְכָל / פָּנִים אֲשֶׁר בָּם נִמְצְאָה נִפְזֶרֶת

זֹאת גַּם לְאֶחָד מֵעֲתִידִים תַּעֲבֹד / תָּמִיד, עֲבוּר לִהְיוֹת לְאוֹת לַעְצֶרֶת

חָקוּק לְהֶעָשׂוּי כְּבָר, גַּם לַאֲשֶׁר / עַתָּה, כְּמוֹ נֶעְכָּר וְגַם נֶעְכָּרֶת

טְפוּלָה לְוָו רַבִּים בְּפֹעַל שֶׁכְּבָר, / כֵּן בַּאֲשֶׁר לִהְיוֹת תְּהִי נִכְתֶּרֶת

יְדוּעָה בְנוּן צָקוּן וְנוּן וַיַּחֲנוּן, / עֵת תַּעֲמֹד בָּהֶן וְעֵת נֶחְסֶרֶת

כֵּן נוֹסְפָה בִלְשׁוֹן נְקֵבָה כִּי תְהִי / אַחַת וְהִיא כָל עֵת וְעֵת מוֹסֶרֶת

לִקְהַל נְקֵבוֹת נִתְּנָה חֹק, וַתְּהִי / בַשֵּׁם וּבַפֹּעַל לְוָו נִכֶּרֶת

מֵאַחֲרֵי הַיּוֹד וְהַתָּו תַּעֲבֹד / וּבְמַאֲמָר גַּם הִיא לְחֹק נִגְזֶרֶת

נוּן נִתְּנָה לַעֲבֹד וְלִבְנוֹת בָּהּ בְּשֵׁם / רַבִּים וְשֵׁם אֶחָד בְּלִי אַחֶרֶת

שׂוּמָה תְּמוּרַת מֵם, וְנוֹסֶפֶת בְּתוֹךְ / מִלִּים כְּנוּן מִנְלָם בְּנוּן נִפְאֶרֶת.
