[בִּימֵי יְקוּתִיאֵל אֲשֶׁר נִגְמָרוּ]

ולה מרת͘יה מטאולה בדיעה רת͘י עלי חסאן רי"ת [ולו קינה ארוכה ונמלצת שקונן בה על אבן חסאן ירחמהו אלוהים]


בִּימֵי יְקוּתִיאֵל אֲשֶׁר נִגְמָרוּ / אוֹת כִּי שְׁחָקִים לַחֲלֹף יֻצָּרוּ

שִׂימוּ לְבַבְכֶם תֵּֽדְעוּ כִּי הַזְּמַן / יָכִין לְעַם שַׁחַת וְלֹא נוֹצָרוּ

יִקְחוּ יְמוֹתָם עַם וְגַם עַם יִתְּנוּ / עַד אֶחֱשֹׁב כִּי עִם אֱנוֹשׁ יִסְחָרוּ

לֹא אֵֽדְעָה־זֹאת כַּאֲשֶׁר לֹא יָֽדְעוּ / אַנְשֵׁי לְבָבוֹת כָּל אֲשֶׁר נִזְכָּרוּ

אִם הַגְּוִיּוֹת הַנְּשָׁמוֹת יִמְאֲסוּ / אוֹ הַנְּשָׁמוֹת לַעֲלוֹת יִפְצָרוּ

מַה נּוֹאֲלוּ הָאֹֽמְרִים תֵּבֵל חֲלוֹם / אֵיךְ שָֽׁכְחוּ דָבָר וְלֹא זָכָרוּ

כֶּן־הוּא כְּדִבְרֵיהֶם אֲבָל הוּא מֵאֲשֶׁר / שָׁוְא דִּבְּרוּ לָאִישׁ וְלֹא נִפְתָּרוּ

הוּכַן בְּקִרְבָּהּ לַתְּלָאוֹת מַאֲרָב / כִּי מִפְּנֵי הַבֹּעֲרִים נִסְתָּרוּ

כֻּפַּר בְּרִית מַלְכֵי אֲדָמָה עִם שְׁאוֹל / גַּם מַחְשְׁבוֹת הָרֹֽזְנִים הוּפָרוּ

מַה קָּֽרְאוּ עַמִּים וְלֹא עָנוּ עֲדֵי / קָרָא שְׁאוֹל לָהֶם וּפִיו לֹא מָרוּ

שָׁבַת גְּאוֹן עַזִּים וְשָׂרִים שָׁחֲחוּ / כִּי כַעֲבָדִים אֶל שְׁאוֹל נִמְכָּרוּ

פִּתְאוֹם וּפֶתַע יָֽרְדוּ יַחַד לְבוֹר / לֹא מָֽצְאוּ כֹחַ וְלֹא עָצָרוּ

לֹא מָֽצְאוּ מֵלִיץ יְכַפֵּר בַּעֲדָם / מִכָּל סְגֻלָּתָם אֲשֶׁר אָצָרוּ

אֶשְׁתּוֹלֲלוּ יַחַד וְרֵיקָם הָֽלְכוּ / מִתֹּעֲפוֹת כַּסְפָּם אֲשֶׁר צָבָרוּ

לֹא נָֽשְׂאוּ מִכָּל כְּבוֹדָם בִּלְעֲדֵי / מִדֵּי בְלוֹאִים בַּאֲשֶׁר נִקְבָּרוּ

בָּאוּ בְמַחְשַׁכִּים וְאָזְנָם פִּתְחָה / תַּחַת בְּרִיחִים בַּעֲדָם סָגָרוּ

כַּתַּר מְעַט מָוֶת וְיָדֶיךָ אֱסֹף / מַה תַּעֲרֹץ דַּלִּים בְּיָד נִסְכָּרוּ

דִּמּוּ יְצֻרִים כִּי בְצֶדֶק תִּשְׁפְּטֵם / וּדְבָֽרְךָ לֹא כֵן כְּמוֹ שָׁעָרוּ

כִּי בָךְ אֲנָשִׁים נֶעֱזָרִים כָּֽשְׁלוּ / וּבָךְ אֲנָשִׁים כֹּֽשְׁלִים נֶעְזָרוּ

אִם לָעֲגוּ עַזִּים וְעַלִּיזִים לְךָ / הִנֵּה מְאֹד מִפַּחְדְּךָ סָמָרוּ

אֵשׁ בַּאֲרָזִים נָֽפְלָה מַה יַּעֲשׂוּ / אֵפוֹא אֱזוֹבֵי קִיר בְּיָד נִנְעָרוּ

אֶרְאֶה פְּנֵי תֵבֵל יְדִידַי זֹעֲפִים / יִתְנַכְּרוּ עָלַי וְלֹא נִכָּרוּ

עַל כֵּן שְׁעוּ מִנִּי אֲמָרֵר בַּבְּכִי / מַה תִּקְצְפוּ עָלַי וּבִי תִגְעָרוּ

אַל תִּשְׁאֲלוּ עַל מַה שְּׂפָתַי צָלֲלוּ / כִּי מֵעֲלֵי רֹאשִׁי צְלָלַי סָרוּ

אָבַד יְקוּתִיאֵל אֲשֶׁר הָאֹֽבְדִים / בּוֹ עָֽתְקוּ חַיִל וּבוֹ גָבָרוּ

כַּפְתּוֹר אֲשֶׁר הֻכָּה וְסִפִּים רָעֲשׁוּ / יָכִין אֲשֶׁר נָפַל וְגֻלּוֹת סָרוּ

חשֶׁךְ וְצַלְמָוֶת מְאוֹרוֹת נִגְאֲלוּ / וִימֵי אֲדָמָה אַחֲרָיו יוּאָרוּ

לֹא יָֽדְעָה נַפְשִׁי אֲבָל הַיּוֹם לְבָד / אִם יִמְאֲסוּ יָמִים וְאִם יִבְחָרוּ

עַד לָֽקְחוּ מִבְחַר אֱנוֹשׁ וָאֵֽדְעָה / כִּי יָֽלְדוּ אָוֶן וְעָמָל הָרוּ

אָכֵן לְעֻמָּתִי פְּנֵיהֶם חִזְּקוּ / כִּי אֵֽדְעָה פִשְׁעָם וְלֹא יֶחְוָרוּ

אַל תַּאֲמֵן בָּהֶם וְאַל תִּכְרֹת בְּרִית / כִּי עַל דְּמוּת מַלְאָךְ בְּרִית קָשָׁרוּ

אִם נָֽדְרוּ נֵדֶר לְהַשְׁחִית כָּל יְקַר / בּוֹ שִׁלְּמוּ נִדְרָם אֲשֶׁר נָדָרוּ

הִוָּעֲדוּ הִוָּעֲצוּ הַמֹּֽשְׁלִים / הִתְנַגְּשׁוּ הִתְקַדְּשׁוּ הִבָּרוּ

חוּשׁוּ לְעֶזְרָתִי וְָדַי חַזְּקוּ / כִּי מַחְשְׁבֹתַי אַחֲרָיו נִמְהָרוּ

גַּם הַדְּמָעוֹת גָּֽרְעוּ חֹק נַחֲלָה / כִּי כַאֲפִיקֵי נַחֲלָה נִגָּרוּ

אַךְ הַלְּבָבוֹת כַּשְּׁמִירִים חָֽזְקוּ / כִּי נָֽשְׂאוּ כָזֹאת וְלֹא נִגְזָרוּ

הֹוָה אֲשֶׁר לֹא תֽוּכְלוּן עוֹד כַּפְּרָהּ / הֹוָה אֲשֶׁר לָהּ כָּל לְבָבוֹת חָרוּ

יִקָּפְצוּ יִנָּפְצוּ יִנָּֽתְצוּ / יִנָּֽתְסוּ יִמָּאֲסוּ יֻתָּרוּ

יִתְמֹלֲלוּ יִתְבּוֹלֲלוּ יִתְעוֹלֲלוּ / יִתְהֹלֲלוּ יִתְחוֹלֲלוּ יֵמָרוּ

תֵּקַל בְּעֵינֵינוּ עֲדֵי יִתְפַּלְּצוּ / בָהּ מֽוֹסְדֵי תֵבֵל וְיִתְפּוֹרָרוּ

אַל תַּעֲבִירוּ קוֹל בְּאֶרֶץ הַצְּבִי / פֶּן יִרְגְּזוּ הָרִים וְיִתְעַרְעָרוּ

עַמִּי סְפֹד כִּי הַכְּרוּב סָר מִדְּבִיר / וּכְלֵי קְרָב וּכְלֵי יְקָר נֶעְדָּרוּ

וּזְעַק לְפִי קֶרֶת וְהֵילֵל כִּי אֲרוֹן / נִלְקַח וְלֻחוֹת הַבְּרִית נִשְׁבָּרוּ

נֶהְפַּךְ לְךָ נִיסָן אֲשֶׁר הָיָה לְנֵס / נָסוּ צְלָלֶיךָ וְגַם עָבָרוּ

הַיּוֹם אֲשֶׁר עָלָה אֲרִי מִסֻּבְּכוֹ / הַיּוֹם אֲרָיוֹת עַל אֲרִיאֵל צָרוּ

הַיּוֹם עֲטֶרֶת עַם צְבִי בּוֹ נָֽפְלָה / הַיּוֹם תְּמִידֵי מִזְבְּחוֹ הוּסָרוּ

הַיּוֹם יְסוֹד הֵיכָל פְּנִימִי הָעֳרָה / הַיּוֹם בְּתוֹכוֹ מֽוֹקְדִים בָּעָרוּ

הַיּוֹם בְּתוּלַת בַּת יְהוּדָה הָגְלְתָה / הַיּוֹם שְׁלָלָה בּוֹזֲזִים פִּזָּרוּ

הַיּוֹם בְּקֶרֶב מוֹעֲדֵי אֵל שָׁאֲגוּ / הָאֹֽמְרִים לָהֶם עֲרוּ גַּם עָרוּ

נִשְׁכַּח כְּאֵב שָׁאוּל וְיוֹנָתָן בְּנוֹ / וּכְאֵב בְּנוֹ אָמוֹן אֲשֶׁר דָּקָרוּ

וּכְאֵב גְּדַלְיָה בֶן אֲחִיקָם אַחֲרָיו / וּכְאֵב עֲשֶׂרֶת שֹֽׁפְטִים אֻסָּרוּ

אַל תִּתְמְהוּ אַנְשֵׁי אֲדָמָה עַל דְּבַר / שֶׁמֶשׁ וְיָרֵחַ אֲשֶׁר קָדָרוּ

לוּ חָֽרְדוּ לִקְצָת כְּאֵבִי יָֽרְדוּ / לִקְצֵה תְהוֹם וּלְפִי שְׁאוֹל נִפְזָרוּ

שָׁבַת מְשׂוֹשׂ תֻּפִּים וְהַגִּיל נֶאֱסַף / יוֹם רֻתְּקוּ רַגְלָיו וְיוֹם נֶאְסָרוּ

נָפַל בְּרוֹשׁ עַל בֵּין עֲבֹתִים גָּֽבְהָה / קוֹמַת סְעַפּוֹתָיו אֲשֶׁר נִצְמָרוּ

נָפַל שְׁנַת תִּשְׁעִים וְתֵשַׁע שַׂר גְּבִיר / לִפְנֵי מְרֵעִים אַחֲרָיו נִמְסָרוּ

יִגְדַּל לְעוֹלָם שֵׁם אֱלֹהֵי יַעֲקֹב / כִּי מִשְׁפְּטֵי צִדְקוֹ מְאֹד יָשָׁרוּ

רָאָה דְמֵי עַבְדּוֹ לְפָנָיו צֹעֲקִים / וַיִּדְרְשֵׁם מִיַּד רְשָׁעִים זָרוּ

יוֹם תֵּת פְּקֻדָּתָם שְׁנַת תֵּ"ת הָֽיְתָה / הוּא יוֹם אֲשֶׁר שׁוּחָה לְפָנָיו כָּרוּ

הָיָה יְקוּתִיאֵל כְּסֶלַע יָקְתְאֵל / בּוֹ נִשְׁמְטוּ זָרִים וְגַם זֹרָרוּ

הָיָה יְקוּתִיאֵל כְּתֹרֶן הֹעֲלָה / עַל הַר אֲשֶׁר בּוֹ עֹֽבְרִים אֻשָּׁרוּ

הָיָה יְקוּתִיאֵל כְּזַיִת רַעֲנָן / אוֹ כַאֲרָזִים בַּלְּבָנוֹן פָּרוּ

הָיָה יְקוּתִיאֵל כְּמֶלֶךְ בַּגְּדוּד / אֶל אָהֳלוֹ הַמַּחֲנוֹת צָפָרוּ

שָׁם נִקְבְּצוּ דַיּוֹת וְשָׁמָּה קִנֲנוּ / כָּל צִפֳּרֵי כָנָף וְשָׁם דָּגָרוּ

מִשְׂרַת מְלָכִים עַל כְּתֵפוֹ הָֽיְתָה / וּלְפִי דְבָרוֹ הַגְּבִירִים שָׂרוּ

וּלְמֹעֲצוֹתָיו יִדְּמוּ הַיֹּעֲצִים / וּלְמַחְשְׁבֹתָיו נוֹאֲלוּ נִבְעָרוּ

שָׂמוּ מְלָכִים כַּף לְפִיהֶם וַהֲמוֹן / שָׂרִים בְּמִלֵּיהֶם לְךָ עָצָרוּ

יָתֵד אֲשֶׁר מֵאָהֳלֵנוּ נִגְדְּעָה / הָהּ לַחֲבָלַיִךְ אֲשֶׁר הֻתָּרוּ

גֶּפֶן אֲשֶׁר שֹרֵק בְּכַרְמִי נִטְּעָה / הָהּ לַעֲנָפַיִךְ אֲשֶׁר נִזְמָרוּ

לִבִּי לְצִיּוֹן כַּחֲלִילִים יֶהֱמֶה / וּלְקוֹל בְּכִיתָהּ רַחֲמַי נִכְמָרוּ

תִּזְעַק אֲהָהּ כִּי נָֽפְלָה חוֹמָה אֲשֶׁר / בָּהּ נִשְׁגְּבוּ עַמִּי וּבָהּ נִבְצָרוּ

סָרוּ נְטִישֶׁיהָ וְנִגְלָה הַֽיְסוֹד / בָּהּ וַאֲבָנֶיהָ לְגַיְא הֻגָּרוּ

מִי יִתְּנֵנִי עֹז וְיִפְקֹד יוֹם לְיוֹם / אִם יִכְבְּדוּ בָנָיו וְאִם יִצְעָרוּ

מִי יִמְשְׁכֵנִי בַּעֲבֹתוֹת אַהֲבָה / מִי יִתְמְכֵנִי עֵת צְעָדַי צָרוּ

מִי יַעֲמֹד לִפְנֵי עֲדַת מִתְקוֹמֲמִים / יִתְיָעֲצוּ עָלַי וְיִתְיַמָּרוּ

מִי יַעֲרֹךְ בַּצַּר לְהָשִׁיב צֹרֲרִים / כִּי יַעֲרִימוּ סוֹד וְיִתְעֹרָרוּ

מִי יַהֲרֹס שִׁנֵּי כְפִירִים שֹׁאֲגִים / לִטְרֹף שְׁאָר צֹאן אֹֽבְדוֹת נָעָרוּ

יִבְכּוּ בְּנֵי דָוִד עֲלֵי אֶדֶר יְקָר / בּוֹ לָֽבְשׁוּ יֶשַׁע וְעֹז אָזָרוּ

יִבְכּוּ בְּנֵי אַהְרֹן עֲלֵי אֵפֹד וּבַד / אוּרִים וְגַם תֻּמִים אֲשֶׁר יָקָרוּ

יִבְכּוּ בְּנֵי לֵוִי עֲלֵי כִנּוֹר אֲשֶׁר / בּוֹ נִצְּחוּ שִׁירִים לְאֵל יָשָׁרוּ

יִבְכּוּ מְתֵי אֶרֶץ עֲלֵי כֶתֶם חֲלִי / יִתְהַלֲלוּ בִשְׁמוֹ וְיִתְפָּאָרוּ

יִבְכּוּ פְלִילֵינוּ בְּשִׁנְעָר אַחֲרָיו / כִּי אָֽזְלַת יָדַם וְכֹל חָסָרוּ

יִבְכּוּ מְלָכֶיהָ עֲלֵי צִיר נֶאֱמָן / יִוָּעֲצוּ אִתּוֹ וְיִצְטַיָּרוּ

יִבְכּוּ גְבִירֶיהָ עֲלֵי מַקֵּל פְּאֵר / תָּעֹז יְמִינָם בּוֹ וְיִתְגַּבָּרוּ

יִבְכּוּ זְקֵנֶיהָ עֲלֵי שֵׂיבַת יְקָר / בָּהּ נִשְּׂאוּ פָנִים וּבָהּ נֶהְדָּרוּ

יִבְכּוּ מְרוּדֶיהָ עֲלֵי עֵץ רַעֲנָן / יֵאָסְפוּ תַחְתָּיו וְיִתְגּוֹרָרוּ

יִבְכּוּ רְעֵבֶיהָ עֲלֵי שֻׁלְחָן אֲשֶׁר / יִתְעַדְּנוּ נֶגְדּוֹ וְיִתְאַמָּרוּ

יִבְכּוּ צְמֵאֶיהָ עֲלֵי עָנָן אֲשֶׁר / בּוֹ נֹפֲפוּ יָדָם וּבוֹ נִזְכָּרוּ

קִרְבָּם כְּכִנּוֹר יֶהֱמֶה מֵאֵין דֳּמִי / עַד כִּי כְבֵדָם שָֽׁפְכוּ וַיְעָרוּ

הִתְנוֹדֲדוּ הִתְגֹּדֲדוּ גַּם הֶעֱלוּ / אֵפֶר בְּרָאשֵׁיהֶם וְשַׂק חָגָרוּ

גַּם יָֽשְׁבוּ דוּמָם וְשַׂק לֹא פִתְּחוּ / עַל שַׁעֲרֵי צֶדֶק אֲשֶׁר נִסְגָּרוּ

יִרְהוּ וְיֶחרְדוּ לְשִׁמְעוֹ מִקְּצוֹת / אֶרֶץ מְגוּרֵיהֶם אֲשֶׁר בָּהּ גָּרוּ

יִשְׂאוּ נְהִי נִהְיָה וּמָרָה יִצְעֲקוּ / כִּי כַֽיְתוֹמִים אַחֲרָיו נִשְׁאָרוּ

לֹא יָֽדְעוּ עַד כִּי שְׁנוֹתָיו חָֽלְפוּ / כִּי חָֽלְפוּ תוֹרוֹת וְחֹק עָבָרוּ

תִּשְׁכַּח יְמִינָם אִם חֲסָדָיו יִשְׁכְּחוּ / עַד יִסְּחוּ הָרִים וְעַד יוּסָרוּ

שֵׁבֶט בְּנִי אוֹ כוֹס עֲלֵיכֶם עָֽבְרָה / אַל תִּשְׁכְּרוּ עוֹד בָּהּ וְאַל תִּתְעָרוּ

אַל תִּפְחֲדוּ כִּי הַנְּבוּאָה דִבְּרָה / בִּכָּֽשְׁלָם עֵזֶר מְעַט נֶעְזָרוּ

הִנֵּה יְקוּתִיאֵל יְכַפֵּר פִּשְׁעֲכֶם / לִפְנֵי יְיָ מַלְכְּכֶם תִּטְהָרוּ

יִכּוֹן וְיֵחָשֵׁב לְאֵל דָּמוֹ כְּמוֹ / עֹלוֹת עֲלֵי יַד אַהֲרֹן הָקְטָרוּ

יָקִיץ וְגַם יַעְמֹד לְגוֹרָלוֹ בְּיוֹם / יִצָּֽרְפוּ רַבִּים וְיִתְבָּרָרוּ

פַּסּוּ אֱמוּנִים וַחֲסִידִים גָּֽמְרוּ / בִּימֵי יְקוּתִיאֵל אֲשֶׁר נִגְמָרוּ.
