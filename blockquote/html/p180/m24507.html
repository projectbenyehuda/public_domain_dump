אֱלֹהִים אֵל רִאשׁוֹן

אֱלֹהִים אֵל רִאשׁוֹן וְאַחֲרוֹן / נַעֲלֶה עַל כָּל גְּבוֹהִים

אֲיַחֲדֶנּוּ וְאֶקְרָא בְגָרוֹן / בְּרֵאשִׁית בָּרָא אֱלֹהִים.


אֱלֹהִים אֲדֹנָי אֵל עוֹלָם / יְעִידוּהוּ לִבּוֹת אֱמוּנָי

אַף מַעֲשָׂיו יוֹדוּ כֻלָּם / אָכֵן יֵשׁ אֲדֹנָי.


אֲדֹנָי אֶחָד מֵאָז נִקְדָּשׁ / בַּלִֵּב נִמְצָא וְלֹא נִכְחָד

אֱמֶת יוֹרֶה כִּי כָל יְצִיר חֻדָּשׁ / נִתּנוּ מֵרוֹעֶה אֶחָד.


אֶחָד אֲדֹנָי רָם וְנִשָּׂא / עַד לֹא שַׁפְרִירֵי מְעוֹנָי

אָכֵן הוּא פָעַל וְעָשָׂה / כֹּל אֲשֶׁר חָפֵץ אֲדֹנָי.


אֲדֹנָי אֵל אֶפֶס בִּלְתּוֹ / יִוָּדַע לְכָל שׁוֹאֵל

אַף הוּא רִאשׁוֹן וְהַכֹּל בְּעִתּוֹ / וְאֶל מִי תְּדַמְּיוּן אֵל.


אֵל אוֹר אָז קָרָא בְּמַאֲמָר / וַיְגָרֶשׁ מַאֲפֵל וָאוֹר

אֲפֻדַּת עֹז לָבַשׁ וְאָמַר / יְהִי אוֹר וַיְהִי אוֹר.


אוֹר אֶרֶץ וְשַׁחַק לַיְלָה וְיוֹמָם / יַבִּיעוּ אֹמֶר בְּמֶרֶץ

אֵין קָדוֹשׁ כָּאֵל מְרוֹמָם / אֲשֶׁר בְּיָדוֹ מֶחְקְרֵי אָרֶץ.


אֶרֶץ בְּמִדָּה יָסַד וְקִלְקַל / חוּג מִתְהַלֵּךְ עַל אֲגֻדָּה

בַּעֲשׂוֹתוֹ לָרוּחַ מִשְׁקָל / וּמַיִם תִּכֵּן בְּמִדָּה.


בְּמִדָּה בְרָאתָם תּוֹכֵן שְׁחָקָיו / וְעָמְדוּ יַחְדָּו עֵת קְרָאתָם

בְּהַטְבִּיע אֶבֶן בְּתוֹךְ קָו / צָפוֹן וְיָמִין אַתָּה בְרָאתָם.


בְּרָאתָם בְּלִימָה לִהְיוֹת בֹּהוּ / לְעֵת תַּחֲלִיף כַּלְּבוּשׁ אֲדָמָה

בִּנְטוֹת צָפוֹן עַל תֹּהוּ / תּוֹלֶה אֶרֶץ עַל בְּלִימָה.


בְּלִימָה בִּגְבוּרָה שָׁת מְמַדִּים / בְּקֶרֶב סַגְרִירֵי סְעָרָה

בְּהַקְצוֹתוֹ פִנּוֹת וְצִדִּים / נֶאְזָר בִּגְבוּרָה.


בִּגְבוּרָה בְּמֹאזְנַיִם הֵרִים / זְרוּמֵי תַחְתִּיּוֹת שְׁנָיִם

בַּפֶּלֶס שָׁקַל הָרִים / וּגְבָעוֹת בְּמֹאזְנָיִם.


בְּמֹאזְנָיִם בְּכֹחוֹ לַעֲשׂוֹת הִגְדִּיל / בְּצִנּוֹר הַנִּגְדָשׁ בְּגִיחוֹ

בָּרָקִיעַ בֵּינָיו הִבְדִּיל / מֵכִין הָרִים בְּכֹחוֹ.


בְּכֹחוֹ בְּעִתּוֹ קֵרָה עֲלִיּוֹת / מְפַלֵּג שֶׁטֶף לְקוֹל תִּתּוֹ

בְּרוּחוֹ יַעַרְפוּ בַצִּיּוֹת / יוֹרֶה וּמַלְקוֹשׁ בְּעִתּוֹ.


בְּעִתּוֹ גְדוֹלוֹת תִּכֵּן מְזָרָיו / כָּמוֹס בָּמוֹ שֶׁלֶג מְצוּלוֹת

גַּפֵּי רֶשֶׁף הֶעְמִיק לְצָרָיו / עוֹשֶׂה גְדוֹלוֹת.


גְּדוֹלוֹת גְּבוּרוֹת שָׁת בְּאוֹצָר / אַבְנֵי אֶלְגָּבִישׁ יְקָרוֹת

גָּנַז קִפָּאוֹן לְעֵת צָר / וּדְבַר גְּבוּרוֹת.


גְּבוּרוֹת גָּדוֹל מִסַּפֵּר עֲצוּמוֹת / בְּהַבְדִּיל תַּחְתִּיִּים בָּדוֹל

גַּם בְּקָרְאוֹ לְמִקְוֵה תְהוֹמוֹת / זֶה הַיָּם גָּדוֹל.


גָּדוֹל גַּלֵּיהֶם יַחְדָּו הִגִּיר / וַיִּשְׁבֹּר חֻקּוֹ עֲלֵיהֶם

גְּבוּל גָּדַר בַּעֲדָם לְהַסְגִּיר / שְׁאוֹן יַמִּים שְׁאוֹן גַּלֵּיהֶם.


גַּלֵיהֶם גְּאוּלִים שָׁת וְזִקֵּק / מְחוֹז יַבֶּשֶׁת מִמְּצוּלִים

גְּרָפָם עַד יְפַלֵּס מְחוֹקֵק / דֶּרֶךְ לַעֲבֹר גְּאוּלִים.


גְּאוּלִים גַּלָּיו אָז לַעֲבֹר / הוֹלִיךְ בִּנְתִיבֵי שְׁבִילָיו

גְּזָרָם בְּשֵׁבֶט אֵל גִּבּוֹר / רוֹגַע הַיָּם וַיֶּהֱמוּ גַּלָּיו.


גַּלָּיו גְּבוּרוֹתָיו בָּם יְקַוּוּ / לְהַרְאוֹת שֵׁנִית נוֹרְאוֹתָיו

גְּדֻלָּתוֹ אָז אִיִּים יְחַוּוּ / וְרַעַם גְּבוּרוֹתָיו.


גְּבוּרוֹתָיו דֶּשֶׁא דִּמּוּ לְהַפְרוֹת / וְעֵשֶׂב לַעֲבוֹדַת נִקְשֶׁה

דִּשְׁנֵי אֶרֶץ צָצוּ לְהַבְרוֹת / וַתּוֹצֵא הָאָרֶץ דֶּשֶׁא.


דֶּשֶׁא דֶשֶׁן לְאֶלֶף מְנוּיִים / וּמָאתַיִם וְתִשְׁעִים לְדַשֵּׁן

דָּעַץ גַּם שְׁלֹשִׁים שְׁנוּיִים / וּמַעְגָּלֶיךָ יִרְעֲפוּן דָּשֶׁן.


דֶּשֶׁן דְּרָכָיו יָזַם לְהַפְאִיר / צוּר מָעֹז וּמַחֲסֶה לְחוֹכָיו

דְּבִיר עֵדֶן לָהֶם הִתְאִיר / הֵן אֵלֶּה קְצוֹת דְּרָכָיו.


דְּרָכָיו דְּבָרוֹ הִשְׁלִים וְזָרַח / שֶׁמֶשׁ וְסַהַר לֵאוֹר בְּאוֹרוֹ

דָּצִים כְּגִבּוֹר לָרוּץ אֹרַח / לִשְׁמֹעַ בְּקוֹל דְּבָרוֹ.


דְּבָרוֹ דִּבֵּר לָרוּץ תָּמִיד / עַד בִּן נוּן חֲיָלִים יְגַבֵּר

דַּהַר מְרוּצָתָם הֶעֱמִיד / כִּי פִּי יְיָ דִבֵַּר.


דִּבֵּר דִּגְלוֹ שִׁבְעָה לְהַצְמִיד / בְּשִׁבְעָה גַלְגַּלֵּי זְבוּלוֹ

דְּגָלִים בְּגַלְגַּל רוֹם הֶעֱמִיד / אִישׁ עַל דִּגְלוֹ.


דִּגְלוֹ דָּגִים תְּהוֹם שָׁרַץ / וְעוֹפוֹת מֵרֶקֶק פְּלָגִים

דֶּגֶר תַּנִּינִים פָּרַץ / וּבְצִלְצַל דָּגִים.


דָּגִים הָאֹכֶל הִתִּיר לְחוֹסָיו / וְהֵכִין לִוְיָתָן לְתֶכֶל

הֵן עָלָיו יִכְרוּ עֲמוּסָיו / לְעֵת הָאֹכֶל.


הָאֹכֶל הַבָּא מִטּוּב יְמִינוֹ / עָרַךְ עַל שֻׁלְחָן וְהִרְבָּה

הֵן כָּל זֶה שַׁדַּי הֱכִינוֹ / לַעֲשׂוֹת לָאוֹרֵחַ הַבָּא.


הַבָּא הַבַּיְתָּה עֵת קְרָאוֹ / שָׂם נֶפֶשׁ בְּגֵו עֲמוּתָה

הִזְמִין הֵלֶךְ לַהֲבִיאוֹ / וַיָּבוֹא הַבָּיְתָה.


הַבַּיְתָּה הוֹדוֹ עָלָיו יֵאוֹר / בְּהַכְתִּירוֹ יִפְעַת כְּבוֹדוֹ

הֱפִיקוֹ נְוַת בַּיִת נָאוֹר / כִּסָּה שָׁמַיִם הוֹדוֹ.


הוֹדוֹ הַזֶּה נָתַן עָלָיו / לַעֲשׂוֹת לוֹ עִם שְׁאוֹל חוֹזֶה

הֵשִׁית מִתַּחַת לְרַגְלָיו / אֶת כָּל הַכָּבוֹד הַזֶּה.


הַזֶּה הַגַּן הִפְקִיד בְּיָדוֹ / וְנַחַת שֻׁלְחָנוֹ מְמֻגָּן

הִשְׁלִיטוֹ לֶאֱכֹל מְעוֹדְדוֹ / מִכָּל עֵץ הַגָּן.


הַגַּן הִיא תִתֵּן מֶגֶד / וְזֶה יֻתַּן לָכֶם וִיהִי

הַרְחֵק מִן הָעֵץ נֶגֶד / כִּי מִצְוַת הַמֶּלֶךְ הִיא.


הִיא וָהוּא אִוּוּ אֹכֶל / וּשְׁנֵיהֶם שִׁנֵּיהֶם קָהוּ

וְהִשִּׁיאָם עָרוּם נוֹכֵל / וַתֹּאכַל הִיא וָהוּא.


וָהוּא וְנִסְתָּר חָשַׁב דַּרְכּוֹ / מִמְּגַלֶּה מַצְפּוּן וּמִסְתָּר

וּבְהִתְהַלֵּךְ קוֹל בְּסֻכּוֹ / רָאָה רָעָה וְנִסְתָּר.


וְנִסְתָּר וְנָלוֹז גָּעַר צוּרִי / לֹא תוֹסִיפוּ עוֹד לַעֲלֹז

זִמֵּר לְשֵׁם יוֹשֵׁב תְּהִלּוֹת / עַד יַעֲבָר זָעַם.


זַעַם זֶה מָלְאוּ נְסוֹגִים / עַד שׁוּב מֵישָׁרִים חוֹזֶה

זְבָדָם שְׁנַיִם שָׂרִיגִים / אֶת זֶה לְעֻמַּת זֶה.


זֶה זִכְרָם שָׁלְחוּ לְמִקְנֶה / לְהִתְעַדֵּן בִּפְרִי סַחֲרָם

זְרֹעַ יְבוּל וּרְעוֹת צוֹנֶה / מֵאֶרֶץ זִכְרָם.


זִכְרָם זִמְרָה לָקְחוּ לְקָרֵב / פְּרִי וּבִכְרַת צֹאן תְּשׁוּרָה

זֻמַּן לְמוֹבִיל שַׁי עָרֵב / תּוֹדָה וְקוֹל זִמְרָה.


זִמְרָה זָמַם עֵת רָצוֹן / וְרַב נִיבוֹ נִבְזֶה נִשְׁמַם

זָר לַעֲשׂוֹת חוֹמֵד לָצוֹן / לוֹ כַּאֲשֶׁר זָמַם.


זָמַם זְוָעָה יָד לַהֲפֹךְ / חוֹרֵשׁ עַל אָחִיו רָעָה

זָעַף דָּם נָקִי לִשְׁפֹּךְ / וְהָיָה רַק זְוָעָה.


זְוָעָה זָר מֵרֹאשׁ רוֹחֵשׁ / פֶּשַׁע כִּסָּה וְנָזָר

זַךְ עָנָהוּ מַה תְּכַחֵשׁ / וְעֵינַי רָאוּ וְלֹא זָר.


זָר חֶרֶב שָׁלַף נִכְנַע / וְאוֹת שָׂם חוֹפֵשׂ קֶרֶב

חֻלַּץ בְּלֶכְתּוֹ נָד וָנָע / וּבְמִלְחָמָה מִידֵי חֶרֶב.


חֶרֶב חַיִל לֹא קָרַב / וַיַּחְלְפוּ יוֹם וָלָיִל

חֲבוּר עֲצַבִּים מְאֹד רַב / עָתְקוּ גַּם גָּבְרוּ חָיִל.


חַיִל חָנֵף רַב תָּקְפּוֹ / וּמַטֵּה רֶשַׁע צָץ וְעָנֵף

חֲבָלִים חִלֵּק אֵל בְּאַפּוֹ / כִּי כֻלּוֹ חָנֵף.


חָנֵף חַי נִאֵץ וְאָמַר / לָאֵל סוּר מִנִּי אֳרָחַי

חוֹקֵר סֵתֶר אָז גָּמַר / לְהַכּוֹת אֶת כָּל חָי.


חַי חָשַׂךְ יַד מְתֵי אָוֶן / וּמַסֵּכָה עֲלֵיהֶם נָסָךְ

חִבְּלָם יַחַד בְּטִיט יָוֵן / וְאֶת עַבְדּוֹ חָשָׂךְ.


חָשַׂךְ חֹק זֶרֶם כַּבִּיר / וְקֻמְּטוּ וְלֹא עֵת מֵרָחוֹק

חַשְׁרַת מַיִם עֲלֵיהֶם הֶעֱבִיר / כִּי עָבְרוּ תוֹרוֹת חָלְפוּ חֹק.


חֹק חָסִיד לֹא הִכְרִית / בִּמְחוֹתוֹ יְקוּם מַפְסִיד

חִיָּה לָאָרֶץ שְׁאֵרִית / כִּי הִפְלָה יְיָ חָסִיד.


חָסִיד טָרַף מִכֹּל שְׁנַיִם / וּבְעֵץ גֹּפֶר קָץ וְחָרָף

טִלְטֵל יוֹנָה לָעֵינַיִם / וְהִנֵּה עֲלֵה זַיִת טָרָף.


טָרַף טוֹב כָּל בְרוּאָיו / וְהִצִּילוֹ מִשְּׁ[אוֹל קָטֹב]

טֶרֶף נָתַן לִירֵאָיו / וְדוֹרְשֵׁי יְיָ לֹא יַחְסְרוּ כָל טוֹב.


טוֹב טוּבָם לֹא נָעַר / וְשָׂם אֵיתָן מוֹשָׁבָם

טָחוּ תָפֵל יוֹשְׁבֵי שִׁנְעָר / הֵן לֹא בְיָדָם טוּבָם.


טוּבָם טִיט הֻשְׁלַךְ בִּשְׁאָט / בְּאָמְרָם לְהַכְבִּיד עַבְטִיט

טִאֵט מָעֻזָּם וְלֹא אָט / וּכְמוֹ יוֹצֵר יִרְמָס טִיט.


טִיט טַבּוּר זֵדִים רָמַס / לְבַל יָקוּם דָּבָר דָּבוּר

טִלְטֵל אַנְשֵׁי תוֹךְ וְחָמָס / יוֹשְׁבֵי עַל טַבּוּר.


טַבּוּר טָהוֹר לְאוֹר יָצָא / וּמֵאֵלָיו נִזְהַר נִזְהוֹר

טוֹעִים נִמְצְאוּ סִיג וְנִמְצָא / זָהָב טָהוֹר.


טָהוֹר טוֹב בָּחַר בָּחוֹר / בֶּן שָׁלֹשׁ כְּגַן רָטֹב

טִעֵם לִנְסוֹגִים אָחוֹר / טַעֲמוּ וּרְאוּ כִּי טוֹב.


טוֹב יָעַץ לְהָעִיר אֵיתָן / וּמֵאֵשׁ גִּבּוֹר לֹא נִרְעַץ

יָזַם כְּרֹת גִּלּוּל מִפְתָּן / וְנָדִיב נְדִיבוֹת יָעַץ.


יָעַץ יָשְׁרוֹ מִנְּעוּרָיו / לֶאֱחֹז רְֶגֶל בַּאֲשׁוּרוֹ

יָהּ צְרָפוֹ מֵבִין סְתָרָיו / לְהַגִּיד לְאָדָם יָשְׁרוֹ.


יָשְׁרוֹ יִבְחַן לִשְׁחֹט יָחִיד / עַד יֵקַר דָּמוֹ וְיוּחַן

[יִרְאַת שַׁ]דַּי לֹא הִכְחִיד / יְיָ צַדִּיק יִבְחָן.


יִבְחַן יִרְאֶה מִמְּכוֹן כִּסְאוֹ / תְּבוּסַת עַם עָנִי וְנִכְאֶה

יִזְכֹּר בְּרִית אֵיתָן בְּקָרְאוֹ / יְיָ יִרְאֶה.


יִרְאֶה יִשְׁעוֹ גֶזַע אֱמוּנָי / וְיִמְחֶה בִשְׂכַר זֹאת פִּשְׁעוֹ

יִשָּׂא בְרָכָה מֵאֵת יְיָ / וּצְדָקָה מֵאֱלֹהֵי יִשְׁעוֹ.


יִשְׁעוֹ יֶהְגֶּה בֵּן נִבְחָר / כְּאֶרֶז בַּלְּבָנוֹן יִשְׂגֶּה

יוֹשֵׁב אֹהֶל דָּת שָׁחַר / וּבְתוֹרָתוֹ יֶהְגֶּה.


יֶהְגֶּה יִפְרַח בֵּית מְגוּרָיו / תָּם נָעוּר וָרֵיק בָּרָח

יַעֲטוּהוּ בַדֵּי אֲמִירָיו / צַדִּיק כַּתָּמָר יִפְרָח.


יִפְרַח כַּדָּת גֶּזַע יְדִידוּת / עַד בְּנֵי חָם הִקְשׁוּ עֲבוֹדָת

כָּבְשׁוּ אֲדוֹנֵיהֶם לְעַבְדוּת / אֲשֶׁר לֹא כַדָּת.


כַּדָּת כֻּלּוֹ גַן נִנְעָל / הֵצִיץ וַיֶּעְשַׁר חֵילוֹ

כֶּרֶם לְחַבֵּל יָזַם שׁוּעָל / וְהִנֵּה עָלָה כֻלּוֹ.


כֻּלּוֹ כַלְּבָנוֹן זָרַח / מִתַּחַת סִִבְלוֹת מָנוֹן

כְּשׁוֹשָׁן בֵּין חוֹחִים פָּרַח / וַיַּךְ שָׁרָשָׁיו כַּלְּבָנוֹן.


כַּלְּבָנוֹן כָּבוֹד רַב הֶעְדָּם / וַיֹּאבְדוּ זָרִים אָבוֹד

כָּרְעוּ שָׂרֵי נֹף נֶגְדָּם / וְנֶגֶד זְקֵנָיו כָּבוֹד.


כָּבוֹד כָּעֵדֶר שִׁלְּחָם / בַּיְשִׁימוֹן חֶדֶר בְּחָדֶר

כַּסּוּס בַּמִּדְבָּר הִנְחָם / וַיְנַהֲגֵם כָּעֵדֶר.


כָּעֵדֶר כָּאֵלֶּה בְּנָסְעָם / עֵת מָסָךְ פָּרוּשׂ יַעֲלֶה

כָּזֹאת מִי שָׁמַע עַל עָם / מִי רָאָה כָּאֵלֶּה.


כָּאֵלֶּה כֹּהֲנִים הֵם לְאֵלִי / קְרוּאֵי רֵאשִׁית אוֹנִים

כָּאָמוּר וְאַתֶּם תִּהְיוּ לִי / מַמְלֶכֶת כֹּהֲנִים.


כֹּהֲנִים לִסְבֹּל קֹדֶשׁ אִנָּה / וְהַזָּר הַקָּרֵב יִבֹּל

לֵוִי לַעֲמֹד לְפָנָיו מִנָּה / וַיֵּט שִׁכְמוֹ לִסְבֹּל.


לִסְבֹּל לְעוֹלָם פִּנַּת יִקְרַת / אֲבִיר יַעֲקֹב הֶאֱצִילָם

לְאַהֲרֹן קְדוֹשׁוֹ בְּרִית כָּרַת / אַתָּה כֹהֵן לְעוֹלָם.


לְעוֹלָם לַיְיָ נִבְדַּל / לְכַהֵן בְּהֵיכָל לִפְנָי

לוֹ וּלְזַרְעוֹ כִּי יִגְדַּל / יִכְתֹּב יָדוֹ לַיְיָ.


לַיְיָ לַעֲשׂוֹת חֻקִּים נְעִימִים / תִּהְיֶינָה כַּפָּיו פְּרוּשׂוֹת

לְמַלֵּא יָדוֹ שִׁבְעַת יָמִים / כַּאֲשֶׁר צִוָּה יְיָ לַעֲשׂוֹת.


לַעֲשׂוֹת לְשׁוֹלְחָיו דָּת יְפַנֶּה / שְׁאֵר גָּדוֹל מֵאֶחָיו

לְלִשְׁכַּת פַּלְהֶדְרִין יִפְנֶה / צִיר נֶאֱמָן לְשׁוֹלְחָיו.


לְשׁוֹלְחָיו לֶקַח טוֹב יְזֻמַּן / בְּהַקְהִיל תַּחְכְּמוֹנֵי פֶקַח

לְהוֹרוֹת חֹק עָשׂוֹר נֶאֱמָן / יִשְׁמַע חָכָם וְיוֹסֶף לֶקַח.


לֶקַח לְשׁוֹלְחֶיךָ תַקְשִׁיב / וְאָז יִישַׁר כּוֹחֶךָ

לָכֵן הִתְאַמֵּץ לְהָשִׁיב / אֲמָרִים אֱמֶת לְשׁוֹלְחֶיךָ.


לְשׁוֹלְחֶיךָ מָתְנֶיךָ תְשַׁנֵּס / וְיִשְׁמַע אֵל וְיַעֲנֶךָ

מִמָּרוֹם יִתֵּן לְךָ נֵס / וְאַתָּה תֶּאְזֹר מָתְנֶיךָ.


מָתְנֶיךָ מִפַּח מוּעָק יְמַלֵּט / בְּאֶפֶס תִּקְוַת מַפָּח

מַעְגְּלוֹתֶיךָ יִתְמֹךְ פַּלֵּט / כִּי הוּא יַצִּילְךָ מִפָּח.


מִפַּח מֵלִיץ טוֹב נֶאֱצָל / עֵת יְכוֹנֵן נַפְשׁוֹ לְהַחֲלִיץ

מִמָּוֶת פִּתְאֹם נִצָּל / מַלְאָךְ מֵלִיץ.


מֵלִיץ מַרְפֵּא [וַאֲרוּכָה] / וַעֲבוֹדוֹת אַרְבַּע יְיַפֶּה

מִשְׁמָ[ר הַלַּילָה וְהַיּוֹם] מְלָא[כָ]ה / וְצִיר אֱמוּנִים מַרְפֵּא.


מַרְפֵּא מְשִׁיחוֹ יִשְׁלַח תָּמִיד / בְּהַרְאוֹתוֹ זִבְחֵי עַם [נְכוֹחוֹ]

[מַעֲבִירִים] פַּר וָאַיִל לְהַעֲמִיד / נֶגֶד יְיָ וְנֶגֶד מְשִׁיחוֹ.


מְשִׁיחוֹ מִזֶּה סָר לְבֵית רֶקַח / עִם זִקְנֵי כְהֻנָּה בַמַּחֲזֶה

מַשְׁבִּיעִים עָלָיו שֵׁם בְּלֶקַח / אַל נָא תָמוּשׁ מִזֶּה.


מִזֶּה מְכֻבָּד בְּקוֹל תְּשׁוּאָה / וְנֹעַם צְרָדָה בַּפֶּה בִלְבַד

מְמַעֵט אֹכֶל פֶּן תִּקֶּר תְּנוּאָה / לִקְדוֹשׁ יְיָ מְכֻבָּד.


מְכֻבָּד נְכוֹחִים יֶהְגּוּ לְפָנָיו / בְּסִפְרֵי קֹדֶשׁ אוֹ בְשִׂיחִים

נְדוּדִים יִשְׂבַּע בֵּית מְלוֹנָיו / מֵשִׁיב דְּבָרִים נְכוֹחִים.


נְכוֹחִים נַפְשָׁם חֲצוֹת לַיִל / יְעִירוּן לַעֲבוֹדַת קְדוֹשָׁם

נִצְבְּעוּ בְּפַיִס לֹא בְחַיִל / לְהַצִּיל מִמָּוֶת נַפְשָׁם.


נַפְשָׁם נְכוֹחָה רִאשׁוֹנִית / תְּרֹם דֶּשֶׁן עוֹלָה קְדוּחָה

נִתְכַּן פְּנִימִי וְנֵרוֹת שֵׁנִית / יָדְעוּ עֲשׂוֹת נְכוֹחָה.


נְכוֹחָה נֶאֱמָנָה שְׁלִישִׁית / [קְטֹרֶת סַמִּים בְּאָמְנָה]

נְתָחִים בָּרְבִיעִית לָשִׁית / עֵדוּת יְיָ נֶאֱמָנָה.


נֶאֱמָנָה נֶחְפְּזוּ צְנוּפִים / לַעֲשׂוֹת [ זוּ]

[נְגִיד מְלֶאכֶ]ת יִשְׁלַח צוֹפִים / נִבְהֲלוּ נֶחְפָּזוּ.


נֶחְפְּזוּ נֵס לַעֲשׂוֹת וְלַחֲזוֹת / אִם שַׁחַר קָדִים [עָלָה כְּנֵס]

נֹגַהּ כְּצֵאתוֹ עַל פְּרָזוֹת / נָתַתָּה לִּירֵאֶיךָ נֵס.


נֵס נֶאֱמָנִים צוּר יְצַוֶּה / בְּהִוָּסְדָם עַל צִיר [אֱמוּנִים]

נָכוֹן לָצֵאת חוּץ לְמִקְוֶה / מֵימָיו נֶאֱמָנִים.


נֶאֱמָנִים סְבָבוּהוּ בְנָסְעָם / וּלְשַׁעַר הַמַּיִם קֵרְבוּהוּ

סוּת פָּרְשׂוּ בֵּינוֹ לְבֵין עַם / אֲשֶׁר סְבָבוּהוּ.


סְבָבוּהוּ שְׂמָלוֹת וְנֶחְפַּז / לִטְבֹּל וּלְהִסְתַּפֵּג וְלַעֲלוֹת

סְגָן יַעֲטֵהוּ כְלֵי פָז / חֲלִיפוֹת שְׂמָלוֹת.


שְׂמָלוֹת סִבָּה הֵם לִדְחוֹת / בְּמִכְנְסֵי בַד שֶׁמֶץ וְדִבָּה

סְאוֹן חֶרְפַּת שִׁטִּים לִמְחוֹת / כִּי הָיְתָה סִבָּה.


סִבָּה סְלוּלָה בִכְתֹנֶת בַּד / לִפְנֵי נוֹרָא עֲלִילָה

סוֹד פַּסִּים תָּסִיר וּמַעְבַּד / דֶּרֶךְ לֹא סְלוּלָה.


סְלוּלָה סְבִיבָיו אִמְרַת צוּרוֹ / בְּאַבְנֵט יַחְגֹּר שַׂר קְרֵבָיו

סְחוֹת צַלְמֵי כַשְׂדִּים בְּחָגְרוֹ / עַל כָּל סְבִיבָיו.


סְבִיבָיו סָבִיב מִצְנֶפֶת תַּעֲבִיר / פֶּרַע וְתַצְדִּיק אוֹת חָבִיב

סִכְלוּת לָשׁוֹן בְּקוֹל יַגְבִּיר / עַל שׁוּלֵי הַמְּעִיל סָבִיב.


סָבִיב סֻבֳּלוֹ אֵפוֹד עָרֵב / וְחֹשֶׁן שְׂאֵת לִפְנֵי מְחוֹלְלוֹ.

סֵמֶל הֶעָשׂוּי בְּחוֹרֵב / בַּיּוֹם הַהוּא יָסוּר סֻבֳּלוֹ.


סֻבֳּלוֹ עוֹלָם בְּצִיץ יַצְמִיד / רַצּוֹת דָּם וָנֶסֶךְ וּמַעֲלָם

עַל מִצְחוֹ יִהְיֶה תָמִיד / כְּיָרֵחַ יִכּוֹן עוֹלָם.


עוֹלָם עֲדֵי עַד צִיר נִקְדָּשׁ / עֵת לַעֲשׂוֹת תָּמִיד יִצְעַד

עֲשָׂרָה בֶעָשׂוֹר יְקֻדַּשׁ / בַּיְיָ עֲדֵי עַד.


עַד עִמּוֹ עֹז כְּקָרַץ / הַתָּמִיד וְקִבֵּל דָּמוֹ

עָמַד לְהַקְטִיר וּלְהֵטִיב רָץ / וַיְיָ עִמּוֹ.


עִמּוֹ עֲטֶרֶת רָצוֹן בְּעָרְכוֹ / רֹאשׁ וּנְתָחִים לַעֲתֶרֶת

עֵת יַקְרִיב סֹלֶת וְנִסְכּוֹ / תָּשִׁית לְרֹאשׁוֹ עֲטֶרֶת.


עֲטֶרֶת עַמּוֹ אָץ כְּכַלּוֹת / תְּמִיד הַשַּׁחַר מֵהֲקִימוֹ

עַד פַּרְוָה יְמַהֵר לַעֲלוֹת / עִם נְדִבֵי עַמּוֹ.


עַמּוֹ עָלָיו סַךְ וְקִדֵּשׁ / וְיָרַד וְטָבַל מוּל אֲצִילָיו

עָטָה בַד וְעֵת קִדּוּשׁ חִדֵּשׁ / הוֹד וְהָדָר תְּשַׁוֶּה עָלָיו.


עָלָיו עוֹמֵד אֵל עוֹלָם / צֶדֶק עֵדְווֹתָיו לְלַמֵּד

עֵת בֵּין מִזְבֵּחַ לְאוּלָם / הַכֹּהֵן הַגָּדוֹל עוֹמֵד.


עוֹמֵד פָּנָיו בְּעֵת יִקְרַב / לְחַלּוֹת וּלְגַלּוֹת טְמוּנָיו

פַּר הִקְרִיב וּפְאַת מַעֲרָב / אֶל עֵבֶר פָּנָיו.


פָּנָיו פֶּשַׁע בַּל יְכַסֶּה / וּבְסָמְכוֹ יִסְמֹךְ יֶשַׁע

פֹּה יַעֲנֵהוּ צוּר מַחְסֶה / אַשְׁרֵי נְשׂוּי פֶּשַׁע.


פֶּשַׁע פֹּה כָעֵת יְכֻפַּר / סְגָן וְרֹאשׁ בֵּית אָב אֲפָפוֹ

פָּנָה וַיַּצֵּג הַפָּר / עַד בּוֹאוֹ פֹה.


פֹּה פְנֵיהֶם שָׂמוּ כְּהוֹאֵל / לָלֶכֶת עִמּוֹ שְׁנֵיהֶם

פֶּתַח יְסוֹד לִצְפוֹן אֲרִיאֵל / מְגַמַּת פְּנֵיהֶם.


פְּנֵיהֶם פְּאֵר עָטָה בְּמָצְאוֹ / שְׂעִירִים וְקַלְפֵי פוּר מְבָאֵר

פּוּרֵי פַז טָרַף בְּנָשְׂאוֹ / כְּחָתָן יְכַהֵן פְּאֵר.


פְּאֵר פָּעַלְתָּ עֵת רְצוֹנְךָ / וְשֵׁם עַל יְמִינוֹ הֵטַלְתָּ

פָּץ סְגָן הַעַל יְמִינְךָ / כָּל מַעֲשֵׂינוּ פָּעַלְתָּ.


פָּעַלְתָּ פַּחַד כְּקָמַץ / בִּשְׂמֹאל וַיִּרְהוּ יַחַד

פִּי רֹאשׁ בֵּית אָב חֲזַק וֶאֱמָץ / לֹא הָיָה פַחַד.


פַּחַד צִדְקוֹ צִיר יַעְטֶה / תָמִיד כִּי יִשְׁעֶה בְחֻקּוֹ

צוּר שָׁלוֹם אֵלָיו יַטֶּה / יְיָ חָפֵץ לְמַעַן צִדְקוֹ.


צִדְקוֹ צְבִי וּפְאֵר נִכְתָּם / בְּתָפְשׂוֹ שְׁנֵי שְׂעִירִים כְּלָבִיא

צִמֵּד גּוֹרָלוֹת בְּתִתָּם / לַעֲטֶרֶת צְבִי.


צְבִי צְבָאוֹת תַּעֲלֹז בְּנָשְׂאוֹ / לַשֵּׁם חַטָּאת בְּקוֹל תְּשׁוּאוֹת

צִלְצְלוּ צַדִּיקִים בְּקָרְאוֹ / בְּשֵׁם יְיָ צְבָאוֹת.


צִבְאוֹת צִיּוֹן רָאוּ אוֹרָה / בְּשָׁמְעָם שֵׁם אֱלֹהִים עֶלְיוֹן

צִפְצְפוּ יַחַד בְּקוֹל זִמְרָה / וְרִנְּנוּ בִמְרוֹם צִיּוֹן.


צִיּוֹן צִדְקָהּ לֹא לְהַכְרִית / כְּדַת שְׂעִיר הָעֲזָאזֵל חֻקָּהּ

צִמֵּד בּוֹ תִקְוַת זְהוֹרִית / עַד יֵצֵא כַנֹּגַהּ צִדְקָהּ.


צִדְקָהּ צָפַנְתָּ לְהַרְאוֹת / מֵלִיץ לְשַׁוְעוֹ הֶאֱזַנְתָּ

צָעַד בֵּין אוּלָם לִרְאוֹת / טוּבְךָ אֲשֶׁר צָפַנְתָּ.


צָפַנְתָּ צִדְקִי בְּדָרְשׁוֹ / לָצֵאת עִם שִׁבְטוֹ נָקִי

צָעַק עַל הַפָּר בְּרַחֲשׁוֹ / עֲנֵנִי אֱלֹהֵי צִדְקִי.


צִדְקִי קְדוֹשׁוֹ אָז יַעֲנֶה / כְּכַלּוֹת לָצֶקֶת לַחְשׁוֹ

קָם לִשְׁחֹט הַפָּר לִפְנֵי / גּוֹאֵל יִשְׂרָאֵל קְדוֹשׁוֹ.


קְדוֹשׁוֹ קוֹלוֹ יַאֲזִין רָב / כְּקִבֵּל דָּם לִרְצוֹת פָּעֳלוֹ

קָרָא לְמָרְסוֹ וְאָז יֶעֱרַב / אֶת הוֹד קוֹלוֹ.


קוֹלוֹ קָדְשׁוֹ אָז יַעֲלֶה / בְּמַחְתַּת פָּז לַחְתּוֹת אִשּׁוֹ

קָרַב לְמַלֵּא אֵשׁ לְמַלֵּא / אֶת דְּבַר קָדְשׁוֹ.


קָדְשׁוֹ קְטֹרֶת בִּשְׂמֹאל נְתוּנָה / מְלֹא חָפְנָיו לִמְצֹא עֲתֶרֶת

קָח מַחְתָּה בְיָמִין וּפָנָה / לְהַקְטִיר קְטֹרֶת.


קְטֹרֶת קֶדֶם מִזְבֵּחַ פְּנִימָה / לִפְנֵי הַפָּרֹכֶת יְקַדֵּם

קְצֵה צָפוֹן יָתוּר בְּעָצְמָה / לָרוֹכֵב בִּשְׁמֵי שְׁמֵי קֶדֶם.


קֶדֶם קֶרֶן יִשְׁעוֹ יָרִים / לָשֵׂאת נֵס וּלְחַזֵּק תֹּרֶן

קִדֵּם מַחְתָּה בֵּין שְׁנֵי עֳפָרִים / הָנּוֹשְׂאִים קֶרֶן.


קֶרֶן קָדְשׁוֹ כִסְּתָה עֲנָנָה / בְּתִתּוֹ קְטֹרֶת עַל אִשּׁוֹ

קִצַּר עֵת צֵאתוֹ רְנָנָה / בְּהֵיכַל קָדְשׁוֹ.


קָדְשׁוֹ רַבּוֹת רָץ נָחוּץ / קַחַת דַּם פַּר מִמְּסִבּוֹת

רָץ לְהַזּוֹתוֹ פְּנִים וָחוּץ / פְּעָמִים רַבּוֹת.


רַבּוֹת רָצָה צוּר עֶרְכּוֹ / עֵת לִשְׁחֹט שָׂעִיר יָצָא

רָחַשׁ לְהַזּוֹת וְשָׁב לְדַרְכּוֹ / כִּי כְבָר רָצָה.


רָצָה רָצוֹן לְכַפֵּר מְשׁוּחִי / וְהִצְלִיף עַל קִיצוֹן וְחִיצוֹן

רִצֵּף הַשָּׂעִיר הֶחָי / שׁוֹחֵר טוֹב יְבַקֵּשׁ רָצוֹן.


רָצוֹן רֹאשׁ פָּעֳלוֹ יְחַבֵּר / שְׁתֵּי יָדָיו עָלָיו וְיִדְרֹשׁ

רָחַשׁ פִּשְׁעֵי עָם וְדִבֵּר / יָחוּלוּ עַל רֹאשׁ.


רֹאשׁ רֹאשׁ מִדְבַּר צוֹק יְשֻׁלַּח / שָׂעִיר בְּיַד עִתִּי נִפְרַשׁ נִפְרוֹשׁ

רֻצַּץ בְּהַשְׁלִיכוֹ וּפֻלַּח / מִכַּף רֶגֶל וְעַד רֹאשׁ.


רֹאשׁ רַבּוֹת יַעַשׂ בְּהַקְטִיר / חַטָּאת לְצוּר תּוֹכֵן לִבּוֹת

רָחַץ וְקִדַּשׁ וְעוֹלָה הֶעֱטִיר / וְכָהֵנָה רַבּוֹת.


רַבּוֹת רַע לֹא יִמְצָא / וְחֹק טְבִילָה לֹא יִגְרָע

רִגֵּל כַּף וּמֶחְתָּה וְיָצָא / וְאֵין פֶּגַע רָע.


רַע שָׁמוּר בְּפָשְׁטוֹ כְלֵי בַד / וְטָבַל וְקִדַּשׁ כְּחֹק אָמוּר

שָׁקַד לִלְבֹּשׁ פַּז מְכֻבָּד / כִּי לַמּוֹעֵד שָׁמוּר.


שָׁמוּר תִּרְצֶה תָּמִיד כְּעוֹרֵךְ / וְעֵת יַקְטִיר וְיֵיטִיב תִּפְצֶה

תָּר לָשֵׂאת יָדוֹ לְבָרֵךְ / וּפֹעַל יָדָיו תִּרְצֶה.


תִּרְצֶה שָׁלוֹם לְעַם מְאֻשָּׁר / בְּצֵאתוֹ בְאֵין פֶּגַע לִבְלֹם

שְׁמָר תָּם וּרְאֵה יָשָׁר / כִּי אַחֲרִית לְאִישׁ שָׁלוֹם.


שָׁלוֹם לְעַמּוֹ צִיר יִזְבֹּד / וְהֵם יְסוֹבְבוּהוּ לְרוֹמְמוֹ

לְבֵיתוֹ יְלַוּוּ בְכָבוֹד / דּוֹרֵשׁ טוֹב לְעַמּוֹ.


לְעַמּוֹ מוֹפֵת טוֹב יְבֻשַּׂר / בְּאָמְרָם לְצִיר שָׂעִיר שׁוֹפֵת

מִי יוֹדֵעַ כִּי עָווֹן סָר / תְּנוּ לָכֶם מוֹפֵת.


מוֹפֵת הָאוֹת עֵת יְעִירוּן / אָז יְבַשֵּׂר צִיר פְּלָאוֹת

הוּסַר עֲווֹנְךָ עַם יְשׁוּרוּן / וְזֶה לְךָ הָאוֹת.


הָאוֹת לָבָן מְאֹד כְּבֹרִית / וּטְהוֹרִים נִצְּבוּ כְדָרְבָן

לִרְאוֹת פֶּלֶא חוּט זְהוֹרִית / כֻּלּוֹ הָפַךְ לָבָן.


לָבָן עָווֹן וְסָר פֶּשַׁע / דַּבְּרָה בַת קוֹל בַּהֲדוֹמוֹ

חֲסִידִים זַמְּרוּ קוֹל יֶשַׁע / כִּי רוֹצֶה יְיָ בְּעַמּוֹ.
