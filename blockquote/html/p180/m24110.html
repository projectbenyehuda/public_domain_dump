הָרִים אִדַּנְתָּ

  וּבְכֵן הָרִים


הָרִים אִדַּנְתָּ מָעוֹן בְּשַׁפְרִיר זְבוּלוֹ

  אֲשֶׁר בְּיָדוֹ מֶחְקְרֵי אָרֶץ וְתוֹעֲפוֹת הָרִים לוֹ


הָרִים בְּטֶרֶם יָצַקְתָּ עֲפָרָם בְּכוּר צוֹרֵף

  נָאוֹר אַתָּה אַדִּיר מֵהַרְרֵי טָרֶף


הָרִים גָּלַמְתָּ עַד רֹאשׁ מוּצַק שָׁמַיִם

  תְּהוֹם כַּלְּבוּשׁ כִּסִּיתוֹ עַל הָרִים יַעַמְדוּ מָיִם


הָרִים דִּשַּׁנְתָּ לְהוֹצִיא חָצִיר וּבָלִיל וָגֶרֶץ

  מַשְׁקֶה הָרִים מֵעֲלִיּוֹתָיו מִפְּרִי מַעֲשֶׂיךָ תִּשְׂבַּע הָאָרֶץ


הָרִים הִצְמַחְתָּ עֵץ פְּרִי וְזֶרַע קָצִיר

הַמֵּכִין לָאָרֶץ מָטָר הַמַּצְמִיחַ הָרִים חָצִיר


הָרִים וְעַל בְּלִימָה תָלִיתָה צוּר הַמָּעֹז

  יְיָ בִּרְצוֹנְךָ הֶעֱמַדְתָּה לְהַרְרֵי עֹז


הָרִים זָעוּ מֵאֵימַת גַּעֲרָתְךָ וְנִגְרָעוּ

  הַמַּעְתִּיק הָרִים וְלֹא יָדָעוּ


הָרִים חוּל תָּחוּל חֶזְקָתָם בְּיַד הַנִּפְלָא

  יִרְעֲשׁוּ הָרִים בְּגַאֲוָתוֹ סֶלָה


הָרִים טֻלְטְלוּ מִמַּבָּטְךָ וְשָׁנוּ

  הַמַּבִּיט לָאָרֶץ וַתִּרְעָד יִגַּע בֶּהָרִים וְיֶעֱשָׁנוּ


הָרִים יוֹם צֵאת לְיֵשַׁע עַמֶּךָ נֶחְפָּזוּ

  וַתִּגְעַשׁ וַתִּרְעַשׁ הָאָרֶץ וּמוֹסְדֵי הָרִים יִרְגָּזוּ


הָרִים כְּקָרַעְתָּ יַם סוּף עַד נֵבֶךְ קִיצוֹן

  הֶהָרִים רָקְדוּ כְאֵילִים גְּבָעוֹת כִּבְנֵי צֹאן


הָרִים לָמָּה תְּרַצְּדוּן עַל הַר תִּפְאַרְתּוֹ

  הָהָר חָמַד אֱלֹהִים לְשִׁבְתּוֹ


הָרִים מָעֲדוּ יוֹם הֵנַפְתָּ גִשְׁמֵי עֶלְיוֹנַי

  הָרִים נָזְלוּ מִפְּנֵי יְיָ


הָרִים נָמוֹגוּ בְּתִתְּךָ רֵאשִׁית קִנְיָנַי

  הָרִים כַּדּוֹנַג נָמָסוּ מִלִּפְנֵי יְיָ


הָרִים שׂוֹשׂ יָשִׂישׂוּ בְּבוֹא מְדַלֵּג עַל הֶהָרִים

  מִמְּעוֹנוֹת אֲרָיוֹת מֵהַרְרֵי נְמֵרִים


הָרִים עָלוֹץ יַעַלְצוּ בְּבוֹא לְעַמְּךָ יִשְׁעָם

  יִשְׂאוּ הָרִים שָׁלוֹם לָעָם


הָרִים פְּאֵרְךָ יְתַנּוּ וְעֹז מַלְכוּתְךָ יְשַׁנֵַּנוּ

  נְהָרוֹת יִמְחָאוּ כָף יַחַד הָרִים יְרַנֵּנוּ


הָרִים צָהוֹל יִצְהֲלוּ יוֹם תִּלְקֹט שׁוֹשָׁן מֵחֵדֶק

  עַמִּים הַר יִקְרָאוּ שָׁם יִזְבְּחוּ זִבְחֵי צֶדֶק


הָרִים קוֹל צוֹפַי יִשְׂאוּ לְהִתְבַּשֵּׂר

  מַה נָּאווּ עַל הֶהָרִים רַגְלֵי מְבַשֵּׂר


הָרִים רִנָּה יִפְצְחוּ כְּנַמְתָּ לַשּׁוֹשַׁנָּה

  הֶהָרִים וְהַגְּבָעוֹת יִפְצְחוּ לִפְנֵיכֶם רִנָּה


הָרִים שְׁבָטִים יַעֲלוּ חַג וְשַׁבָּת וְחֹדֶשׁ

  וְהִשְׁתַּחֲווּ לַיְיָ בְּהַר הַקֹּדֶשׁ


הָרִים תִּכֵּן גֵּיא חִזָּיוֹן עֲלֵיהֶם תָּרִים

  וְהָיָה בְּאַחֲרִית הַיָּמִים נָכוֹן יִהְיֶה הַר בֵּית יְיָ בְּרֹאשׁ הֶהָרִים.
