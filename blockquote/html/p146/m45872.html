אֶקְרָא לְאֵל

חתימה: אלשבזי


אֶקְרָא לְאֵל מֶלֶךְ נֶאֱמָן / הַנּוֹרָא אָב הָרַחֲמָן

חַסְדּוֹ וְטוּבוֹ לִי יְזֻמָּן / וְיִתֵּן לְקֵץ יִשְׁעִי סִימָן

אֶרְאֶה שְׁנֵי לוּחוֹת וְרִשְׁמָן / וּמַטֶּה וְצִנְצֶנֶת הַמָּן

וּמִשְׁפְּטֵי מִקְדָּשׁ

וְקוֹל שִׁיר חָדָשׁ

וְכֹהֵן מְקֻדָּשׁ

וְסֵדֶר יוֹם הַכִּפּוּרִים וּזְמָן

וְקָרְבַּן הַתּוֹדָה

וְסֵדֶר הָעֲבוֹדָה

וְלֵוִי יַגִּידָה

וְאֶתְבּוֹנֵן עַל שִׁיר הֵימָן / עוּרִי צָפוֹן וּבוֹאִי תֵימָן:


לַוִּי אַיֶּלֶת אֲהָבִים / חֶבְרַת עַם קֹדֶשׁ אֲהוּבִים

וְזִכְרִי זְכוּת אָבוֹת נְדִיבִים / דֵַּבְּרִי עַל לֵב נֶעְלָבִים

קַבְּצִי רְחוֹקִים וּקְרוֹבִים / יִהְיוּ לְחַיִּים נִכְתָּבִים

נַפְשִׁי מַהֲרִי

גּוּפִי טַהֲרִי

לְשִׁבְטַי תֹּאמְרִי

שׁוּבוּ בָּנִים שׁוֹבָבִים

וְהִכּוֹנִי דֶרֶךְ

לַעֲלוֹת אֶל חַדְרַךְ

מֵעֲבוֹדַת פֶּרֶךְ

בְּנֵי זֵדִים וּמְרִיבִים

כְּשׁוּלֵי בִּרְכַּיִם / תְּחַזְּקִי לַהֲקִימָן:


שׁוּב דָּל מִתְנַחְשֵׁל / וְהִתְאַמֵּץ כּוֹשֵׁל

וְלָמָּה תִתְרַשֵּׁל / כְּלֵי חֶרֶב תִּפְשׁוֹל

וְתָסִיר הַמִּכְשׁוֹל / וּלְבוּשׁ מַלְכוּת שַׁלְשֵׁל

כְּבִנְיָמִין צָעִיר

לְשַׂר הַיָּם הֵעִיר

וּנְשִׂיא יְהוּדָה הִנְעִיר

קָפַץ לְתוֹךְ הַנַּחְשׁוֹל

וְנִקְרָא בִשְׁמוֹ

וַיוֹיָ עִמּוֹ

וּבָרוּךְ טַעְמוֹ

אִישׁ כְּכֶרֶם טוֹב בָּשֵׁל / וְזָכָה לִמְלוּכָה

רֹאשׁ לַנְּשִׂיאִים קְדָמָן:


שׁוֹשַׁנַּת עֲמָקִים הַשְׁקִי / נְטִיעוֹתַיִךְ וּבָם דִּבְקִי

וּבְגַן הַבִּיתָן דִבְקִי / כִּי בוֹ תְשׁוּקָתִי וְחָשְׁקִי

וְאִמְרִי לִמְשִׁיחַ צִדְקִי / הַטְרִיפֵנִי לֶחֶם חֻקִּי

וְחֶמְדַּת עֵינִי

בְּפִי וּלְשׁוֹנִי

רְצֵה נָא אֲדוֹנִי

וְהַצְלִיחָה לְבַחוּרֵי פִרְקִי

עֶזְרִי בְצָרָה

בְּאֶרֶץ זָרָה

בַּצַּר לִי אֶקְרָא

וְלֹא יַנִּיחַ אָדָם לְעָשְׁקִי

כִּי הוּא עֻזִּי וּמָעֻזִּי / וְאוֹיְבֵי נַפְשִׁי יַכְלִימָן:


בִּימֵי נְעוּרַי חָשַׁקְתִּי / בְּצוּרִי וּבִשְׁמוֹ בָטַחְתִּי

יְקִימֵנִי וְאִם נָפַלְתִּי / כִּי בוֹ לְעוֹלָם דָּבַקְתִּי

וְרֵעַי וְאוֹהֲבַי עָזַבְתִּי / וּבְאַהֲבָתוֹ שָׂמַחְתִּי

וְחָשׁוּ רַעְיוֹנַי

וּלְבָבִי וְעֵינַי

נַפְשִׁי לַיוֹיָ

יְחַזְּקֵנִי אִם כָּשַׁלְתִּי

מְאַהֲבַי עָבְרוּ

וְנַפְשׁוֹתָן יָקְרוּ

לְעוֹלָמָן נִפְטְרוּ

וַאֲנִי זָקַנְתִּי וְשַׂבְתִּי

חֶלְקָם בַּחַיִּים / הַבָּאִים לְבִית עוֹלָמָן:


זִכְרוֹן טוֹב יְהֶא לִידִידִי / הַאֲזִינוּ שְׂרִידִים עָדַי

אֶשְׂמַח בְּשִׁבְתִּי וּמַעֲמָדִי / וְצוּרִי מַחֲזִיק אֶת יָדַי

אַחֲלַי בִּסְגֻלַּת שְׂרִידַי / רֹאשֵׁי קְהָלִי וּגְדוּדַי

לַיְלָה וְיוֹם

הַחָכְמָה תְּחַיֵּם

בִּרְצוֹן אֵל אָיוֹם

גּוֹלֶה עֲמוּקוֹת סוֹדִי

וְיֵין טוֹב יֵינָם

לְהַרְגִיל לְשׁוֹנָם

בְיֵחוּד קוֹנָם

וּבְעֶזְרַת אֵל שַׁדַּי

אָקוּם עֲלוֹת שַׁחַר / וְאֵלְכָה לִמְקוֹמָן:


יְקָרֵךְ יְלַוֵּנִי / בַּת אַבְרָהָם אֲדוֹנִי

בַּכֹּל אֵם הֲמוֹנִי / חֲמַד חַשֶׁק רְצוֹנִי

וּבְצִלֵּךְ מְלוֹנִי / כִּי מִמֵּךְ מְזוֹנִי

הַפְצִיחִי לְשׁוֹנִי / וְטַהֲרִי רַעְיוֹנִי

וְהַגִּינִי עַל רָשׁ

מִמְּקוֹמוֹ נִגְרָשׁ

וּלְפַרְנָסָה יִדְרָשׁ

וְטֶרֶף תִּתְּנִי

וְהַשְׂכִּילִי עַל דָּל

בִּתְשׁוּבָה חָדָל

לַיוֹיָ נִבְדָּל

אֶחָד וְאֵין בִּלְתּוֹ שֵׁנִי

הַבּוֹחֵר בְּעַמּוֹ / וְשׁוֹאֵל בִּשְׁלוֹמָן:
