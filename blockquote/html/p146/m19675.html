יוֹם אֶזְכְּרָה מַלְכִּי כִּסְאוֹ בְּהַר קָדְשִׁי

החתימה: אלשבזי (בראשי החרוזות); שלם סאלם שלם (בראשי טורי האזור הראשונים בחרוזות הראשונה, השנייה והשישית)


יוֹם אֶזְכְּרָה מַלְכִּי / כִּסְאוֹ בְּהַר קָדְשִׁי

יֻמְתַּק בְּשִׁיר חִכִּי / מִצּוּף וְטוּב דִּבְשִׁי

רַעְיָה לְדוֹד חַכִּי / וּבְאוֹיְבַי בִּטְשִׁי

רַחְצִי וְהִזַּכִּי / חֵן אַהֲבָה דִּרְשִׁי

שׁוּרִי פְנֵי

שִׁבְטֵי בְּנֵי

אִישׁ תָּם, בְּנוֹ

עִם שׁוֹכְנֵי / תֵּימָן וְגַם אוּזָל

לָהֶם רְצִי

כֹּפֶר מְצִי

דָּל אַמְּצִי

גַּם קַבְּצִי / הָמוֹן אֲשֶׁר נִגְזָל

מִשְׁחִי גְּבִיר

כֹּהֵן אֲבִיר

יִהְיֶה סְבִיר

יָבֹא דְּבִיר / אָז אוֹיְבִי יוּזָל

פֶּרֶא אֲשֶׁר אָרַב / יִפּוֹל לְפִי חָרֶב:


אַהְבַת יְפַת קוֹמָה / חָשְׁקָה בְּאוֹת נִקְמָר

יָפְיָהּ אֲשֶׁר נִדְמָה / הָדָר כְּמוֹ תָּמָר

מֵרֹב יְפִי עַלְמָה / שֶׁמֶשׁ בְּזִיו נִגְמָר

מִלֶּחֱיָה חָכְמָה / לְנוֹגְנִים נִשְׁמָר

סוֹד תָּאֳרָהּ

מִיּוֹצְרָהּ

וּבְיוֹשְׁרָהּ

שָׁם זָהֳרָהּ / בַּיּוֹם וּבַלַּיְלָה

אֵשׁ דָּת רְפָא

דַּל יִכְסְפָה

מוֹר מִשְּׂפָ־

תָיו יִטְּפָה / מֵאַהֲבַת כַּלָּה

לוֹבֵשׁ כְּלִיל

מֵאוֹר עֲלִיל

וּבְזוֹג גְּלִיל

יִשְׁמַע חֲלִיל / נוֹגֵן בְּקוֹל צַהְלָה

מִנָּעֳמוֹ

טוּב טַעֲמוֹ

קוֹל פַּעֲמוֹ

שִׁיר נָאֳמוֹ / יִרְפָּא לְלֵב נֶחְלָה

בֹּקֶר וְגַם עֶרֶב / יֶאְתֶה וְיִתְקָרַב:


לָבִיא וְהַלִּבְיָא / מִתְחַבְּרִים הַיּוֹם

וַאְנִי בְמוֹשָׁבִי / צוֹפֶה פְדוּת, אָיוֹם

צוּרִי שְׁלַח תִּשְׁבִּי / וּבְרִיתְךָ קַיֵּם

רַפֵּא לְנֶעְצָבִי / מַהֵר שְׁלַח פִּדְיוֹם

אֵלִי תְּנָה

אֶתְכּוֹנְנָה

אֵיךְ אֶשְׁכְּנָה

עִם בֶּן עֲנָה / תָּחִישׁ רְצוֹנֶךָ

לָנוּ קְבוֹץ

בָּנוּ חֲפוֹץ

אֹיְבִים נְפוֹץ

הָעִיר לְקֵץ / וּשְׁלַח מְשִׁיחֶךָ

מַיִם דְּלֵה

רָשׁ הַעֲלֵה

אֶל הֵיכְלוֹ

יָהּ קַבְּלוֹ / בִּזְכוּת שְׁבָטֶיךָ

פֶּרֶא אֲשֶׁר זוֹרַב / יִצְמַת וְיִשְׁתָּרַב:


שׁוּבִי יְחִידָתִי / הִתְאַזְּרִי חַיִל

זַכִּי לְעַם דָּתִי / לַעְמוֹד בְּעוֹד לַיִל

אָעִיר בְּאַהְבָתִי / חָכְמַת אֲבִיגַיִל

אָתָא בְּעוֹנָתִי / תּוֹר בַּת אֲבִיחַיִל

צוּרִי צְלַח

חוּן נֶאֱלַח

יִנּוֹן שְׁלַח

אֹיְבִי דְּלַח / רַפֵּא לְנִכְאָבִים

לִידִידְךָ

צַו חַסְדְּךָ

בִּרְצוֹנְךָ

וּבְיִשְׁעֲךָ / תָּשִׁיב לְשׁוֹבָבִים

יִתְרַפְּאוּ

עַם נִדְכְּאוּ

חֵן יִמְצְאוּ

כִּי צָמְאוּ / יַחְדָּו לְמֵימֶיךָ

סַפְסוֹף וְעֵרֶב רַב / זִיו תָּאֳרָם יִצְרַב:


בִּחְרִי לְדָל רַעְיָה /חַלִּי לְדוֹד יָפֶה

כַּמָּה בְּתוֹךְ שִׁבְיָה / חַסְדֵּךְ אֲנִי צוֹפֶה

אֵיכָה צְבִי צִבְיָה / פֶּרֶא בְּעֹל כּוֹפֶה

עֵירֹם וְגַם עֶרְיָה / קָלוֹן עֲלֵי כָל פֶּה

יוֹם גָּלְתָה

אָז נִשְׁחֲתָה

כִּי טָעֲתָה

לֹא עָלְתָה / לַחְזוֹת מְקוֹם שַׁלְוָה

שַׁדַּי קְרָא

בַּת נִבְחֲרָה

חֵטְא כַּפְּרָה

חוּשׁ יַשְּׁרָה / דֶּרֶךְ וְשִׂים תִּקְוָה

רַחֵם עֲנִי

עָם נִצְפְּנִי

מִשּׁוֹכְנִי

מִתְאוֹנְנִי / שָׁכַח לְכָל חֶדְוָה

שָׁר שִׁיר וְלֹא עָרַב / כִּי אָהֳלוֹ חָרַב:


זֶה עֵת מְבוֹא דוֹדִים / יָבֹא אֲהוּב לִבִּי

וַאְנִי לְשִׁיר אַקְדִּים / יוֹם אֶשְׁמְעָה תִשְׁבִּי

שִׁבְטַי בְּרוֹן מוֹדִים /רוּחַ אֱלֹהַי בִּי

חוֹזִים וּמַגִּידִים / בַּת רַעֲיָה הַרְבִּי

שִׁמְרִי בְּרִית

עַד אַחֲרִית

בָּנִים זְרִית

אֵיכָן שְׂרִית / לָהֶם תְּבַקֵּשִׁי

לִבְשִׁי לְבוּשׁ

זָהָב חֲבוּשׁ

קִבְצִי נָתוּשׁ

מִפּוּט וְכוּשׁ / לַצּוֹלְלִים הַמְשִׁי

מַלְכִּי גְּשָׁה

דָּת חַדְּשָׁה

בַּת בּוֹשְׁשָׁה

קוּם קַדְּשָׁה / נַעְלֶה לְהַר קָדְשִׁי

וּמְאַהֲבִי קָרֵב / כִּשְׁתִי וְכָעֵרֶב:


יַא טַאלִבּ אַלחִכּמַה / מִן גַירִ אסתַאדִי

אַלצַבּר לַךּ חִשׁמַה / לַא תַנעְכִּס עַאדִי

קֵף קַיִּד אַלנִּעמַה / חִתַּא יְגִ̇יךּ בַּאדִי

יֻפתִיךּ עַן כִּלמַה / מַא שַׁית תַתקַאדִי

בַּאבּ אַלעלוּם

תַקוַא אַלגֻ̇סוּם

תַובַּה תְרוּם

קַבּל אַלעֻזוּם / וִאחדַ̇ר מִן אַלבַּטרַה

תֻבּ תַסתְרִיח

אַנַּךּ גְׄרִיח

גִ̇סמַךּ טְרִיח

וַאלנַּפס רִיח / הַאוִי מְעַא אַלסַכּרַה

רֻדּ אַלכְּלַאם

כַ̇וף אַלמְלַאם

קַבּל אַלצְ̇לַאם

כֻ̇ץ·בִּאלסַּלַאם / מִן כַּאן פִי אַלחַצׄרַה

כֻ̇ץ שַׁרק וַאלמַגרֵבּ / וַאלקֻדס וּחַד יַתרֵבּ:
