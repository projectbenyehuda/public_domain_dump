פליאה דעת העט

תְּהִלַּת הָעֵט


פְּלִיאָה דַעַת הָעֵט מִמֶּנִּי, זָר מַעֲשֵׂהוּ / אֶת מִי נוֹעַץ וַיְבִינֵהוּ / מַחְשְׁבוֹת לִבִּי בְּלִי הַגִּידִי אוֹתָם / יִרְאֶה יַעֲשֶׂה כְּתַבְנִיתָם / בְּלִי אָזְנַיִם יַקְשִׁיב בְּכָל רָזֶה / וּבִלְתִּי עֵינַיִם כָּל צָפוּן חוֹזֶה / מַזֶּה בֶן מַזֶּה יַזֶּה / נַחַל נוֹבֵעַ מְקוֹר חָכְמָה / לָתֵת לִפְתָאִים עָרְמָה / יַעֲרֹף כַּמָּטָר לִקְחוֹ / וְלִפְנֵי אֲדֹנָיו יִשְׁפֹּךְ שִׂיחוֹ / לָרוּץ לְפָנָיו צַו לָצָו / יֶאֱזֹר כְּגֶבֶר חֲלָצָיו / אֶשְׁאָלֶנּוּ וְיוֹדִיעֵנִי / בְּיוֹם קָרָאתִי וַיַּעֲנֵנִי.


עֵט בְּלִי לֵב יָבִין לַחֲשִׁי / וּבְלִי אֹזֶן יִשְׁמַע רַחֲשִׁי

יֶאֱזֹר מָתְנָיוּ לָרוּץ אֹרַח / בִּשְׁלִיחוּתִי כִּרְצוֹן נַפְשִׁי

כָּל אוֹת יִפְגֹּשׁ יַשְׁלִיךְ אַחֲרֵי / גַּבּוֹ יְשַׁלַּח אוֹתוֹ חָפְשִׁי

דֶּרֶךְ כִּכָּר הַגִּלָּיוֹן / רָץ וַיַּעֲבֹר אֶת הַכּוּשִׁי.


לָכֵן הָעֵט בְּחִיר דּוֹדִים וְרֵעִים / בֵּן יַקִּיר יֶלֶד שַׁעֲשׁוּעִים / מִדֵּי דַּבְּרִי בּוֹ זָכֹר יַזְכִּירֵנִי הַנִּשְׁכָּחוֹת / וִיקַבֵּץ הַנִּדָּחוֹת / מֵאַרְבַּע רוּחוֹת / וְיִתְּנֵם בְּמִשְׁמָר לְבִלְתִּי יִשְׁתַּכָּחוּ / בְּדוֹר אַחֵר פֶּן יִמָּחוּ / כִּי לְמִן הָעֵת יִכָּתְבוּן מִלִּין בַּסֵּפֶר וְיֻחָקוּ / כִּרְגָבִים יְדֻבָּקוּ / לֹא נָדִים מִשָּׁם וְלֹא נָעִים / כְּמַסְמֵרוֹת נְטוּעִים / שָׁם יִשְׁמֹר עֵט מוֹצָא שְׂפָתוֹ / לֹא יַחֲלִיפֶנּוּ וְלֹא יָמִיר אוֹתוֹ / לְעוֹלָם דְּבָרוֹ נִצָּב בְּמִשְׁמֶרֶת / לְאוֹת וּלְמַזְכֶּרֶת / יָכִין צַעֲדוֹ בְּהַצְנֵעַ לֶכֶת / יֵשׁ רֶגֶל מְבֹרֶכֶת / כָּל הַמָּקוֹם אֲשֶׁר תִּדְרֹךְ כַּף רַגְלוֹ / בַּנְּעִימִים נָפַל חֶבְלוֹ / גִּלָּיוֹן מְצֹרָע כַּשֶּׁלֶג בְּעָבְרוֹ בוֹ מִצָּרַעְתּוֹ יֶאֶסְפֵהוּ / מִנַּחַל עֲדָנָיו יַשְׁקֵהוּ / יִשְׁלַח מְרַגְּלִים אֶל הַשִּׁטִּים לֶחָרָשׁ לֵאמֹר וְקַו אֶל קָו אֶל הַמְּגִלָּה רוֹקֵעַ / וְהוּא מְדַבֵּר עַל לִבּוֹ, רַק שְפָתָיו נָעוֹת וְקוֹלוֹ לֹא יִשָּׁמֵעַ / וְיַחְשִׁיבוּהוּ לְשִׁכֹּרָה / בְּשׁוּבוֹ לִשְׁאֹב פַּעַם וַעֲשָׂרָה / וְהוּא בְאַהֲבָה מְכַלְכֵּל אֶת הַשּׁוּרָה / אוֹכֵל וְאֵין לְשָׂבְעָה שׁוֹתֶה וְאֵין לְשָׁכְרָה / רַק לְהַשְׁקוֹת הַסֵּפֶר וּלְהַלְעִיטוֹ / כְּבוֹלְעוֹ כָּךְ פּוֹלְטוֹ / וּמְאוּמָה לֹא יִשָּׂא בַעֲמָלוֹ / יְשַׁלֵּם ה׳ פָּעֳלוֹ / וְעוֹד מָה אָרְחִיב פֶּה בִתְהִלּוֹתָיו / וְאַאֲרִיךְ לָשׁוֹן בַּחֲמוּדוֹתָיו / וְדָוִד מַלְכֵּנוּ בְּרוֹב רְצוֹנוֹ / לְהַפְלִיג בְּמַהֲלַל לְשׁוֹנוֹ / עֵט סוֹפֵר מָהִיר יִקְרָאֶנּוּ / לַעֲשׂוֹת לוֹ יְקָר יוֹתֵר מִמֶּנּוּ / כִּי כֵן דַּרְכָּם שֶׁל בְּנֵי אָדָם / וּמִנְהַג בָּשָׂר וָדָם / כָּל רוֹצֶה לְהַגְדִּיל כָּל בִּרְיָה / וְהֶחָפֵץ לְהַלֵּל תַּבְנִית כָּל חֶמְדַּת שְׂכִיָּה / לֹא יַעַרְכֶנָּה / כִּי אִם לִרְעוּתָהּ הַטּוֹבָה מִמֶּנָּה.
