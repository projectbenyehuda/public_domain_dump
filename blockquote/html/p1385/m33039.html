מעשה בעכבור הזקן


נְאֻם לְמוּאֵל בֶּן־אִיתִיאֵל: הָיִיתִי מִנְּעוּרַי מִתְהַלֵּךְ בַּחוּצוֹת וְסוֹחֵר / מִגּוֹי אֶל גּוֹי וּמִמַּמְלָכָה אֶל עַם אַחֵר / פַּעַם אָלִין בַּמִּדְבָּר וּבַיְּעָרִים / פַּעַם בְּבָתֵי הַחֲצֵרִים / כְּמִשְׁפַּט הוֹלְכֵי עַל דֶּרֶךְ וְהוֹלְכֵי נְתִיבוֹת / פַּעַם בַּחוּץ פַּעַם בָּרְחוֹבוֹת / וְלַיְלָה אֱהִי יָשֵׁן / עַל מִטּוֹת שֵׁן / וְשׁוֹכְבִים עַל מִטּוֹתָם / וּסְרוּחִים עַל עַרְשׂוֹתָם / וְלַיְלָה סוּף לְרֹאשָׁם חָבוּשׁ / וְעָרוֹם יָלִינוּ מִבְּלִי לְבוּשׁ, / וְיוֹם יְהַלֵּךְ יָעֵף וּנְכֵה רַגְלַיִם / אֵין לֶחֶם וְאֵין מַיִם, / כָּזֹאת וְכָזֹאת יִקְרֶה לְעוֹבְרֵי דְּרָכִים / פַּעַם עֲדִינִים, פַּעַם נְבוּכִים / פַּעַם כִּנְבָלִים וּפַעַם כִּמְלָכִים.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

דְּבַר אָדָם בְּלִי יַעְמֹד / בְּעֵינוֹ, יָד לְיָד יִשְׂנֶא

וְעֵת יָקוּם וְעֵת יֵשֵׁב / וְעֵת יִסַּע וְעֵת יַחְנֶה

וְעֵת יִגְדַּל וְעֵת יִדַּל / וְעֵת יַעְשִׁיר וְעֵת יַעְנֶה

וְעֵת יִשְׂחַק וְעֵת יִזְעַק / וְעֵת יַהְרֹס וְעֵת יִבְנֶה

וְעֵת יָגִיל וְעֵת יָחִיל / וְעֵת יִצְדַּק וְעֵת יוֹנֶה

וְשַׁחַר יַעֲבֹד צוּרוֹ – / וְעֶרֶב לֹא לָאֵל יִפְנֶה,

וְיֵשׁ נִשְׁעָן עֲלֵי הוֹנוֹ – / וְהוּא עַל בַּעֲלוֹ יִזְנֶה,

וְיֵשׁ נִשְׁעָן בְּהוֹד תֵּבֵל – / וּמִשְׁעַנְתּוֹ רְצוּץ קָנֶה,

וְיֵשׁ יִיגַע לְהַעְשִׁיר, אַךְ / לְלִבּוֹ מַחֲלָה יִקְנֶה.

דְּבַר אִישׁ כָּל יְמֵי חַיָּיו / הֲיִסָּפֵר הֲיִמָּנֶה?

וּמַשְׂכֻּרְתּוֹ בְּעֵת וָעֵת / עֲשָׂרָה יַחֲלִיף מֹנֶה,

תְּבַקְּשׁוֹ אַחֲרֵי כָל זֹאת / וְהֵן אֵין קוֹל וְאֵין עוֹנֶה!


וְרָאִיתִי נִפְלָאוֹת / גְּדֻלּוֹת וְנוֹרָאוֹת / וּפָגַעְתִּי אֲנָשִׁים וְרָשִׁים / נְבָלִים וּנְדִיבִים. / [וּבָאתִי לָעִיר] יוֹשְׁבֶיהָ עֲשִׁירִים / וְסוֹחֲרֶיהָ שָׂרִים / וְעַל זָהָב לֹא יַחֲמֹלוּ / וְכֶסֶף בַּקָּנֶה יִשְׁקֹלוּ / וַתִּישַׁר הַמְּדִינָה בְעֵינַי / וַיִּבְרְחוּ וַיֵּלְכוּ כָל יְגוֹנַי / וָאֹמַר כַּאֲשֶׁר רְאִיתִיהָ: / פֹּה אֵשֵׁב כִּי אִוִּיתִיהָ.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

בְּחַר־נָא לָךְ שְׁכוֹן אָרֶץ / שְׁכוּנֶיהָ נְדִיבֵי עָם

עֲשִׁירֵי הוֹן שְׂמֵחִים בּוֹ, / מְתֵי שֵׂכֶל וְטוּב טַעַם

בְּטוּב לֵב תִּפְגְּשֵׁם תָּמִיד, / בְּאוֹר פָּנִים בְּלִי זַעַם

וְאִם בָּא אֵיד לְאִישׁ מֵהֶם – / יְבוֹאוּהוּ בְּלֵב נִפְעָם

בְּעַם זֶה תִּשְׁכְּנָה בֶטַח / וּבֵינָם נַפְשְׁךָ תִּנְעַם.


וָאַשְׁכִּים בַּבֹּקֶר לִרְאוֹת אֲנָשֶׁיהָ / וְיוֹשְׁבֶיהָ וּרְחוֹבוֹתֶיהָ וּמִגְרָשֶׁיהָ / וְהִנֵּה כֻלָּם אַנְשֵׁי מִרְמָה / וִילִידֵי אַשְׁמָה. / וָאֹמַר בִּרְאוֹתִי רִשְׁעָם: / אֵלְכָה לִי אֶל הַגְּדוֹלִים וַאֲדַבְּרָה עִמָּם / כִּי הֵמָּה יוֹרוּ חוּקוֹת וְתוֹרוֹת / אַךְ הֵמָּה שָׁבְרוּ עֹל נִתְּקוּ מוֹסֵרוֹת. / וָאֶרְאֶה וְהִנֵּה שׁוֹפְטֵיהֶם שׁוֹפְטֵי שֶׁקֶר / זְאֵבֵי עֶרֶב לֹא גָרְמוּ לַבֹּקֶר / וַיִּתְלַקְּטוּ [אֲלֵיהֶם] כָּל טָחֵי תָפֵל / וְכָל אֲשֶׁר שֶׁקֶר טוֹפֵל / וְאִישׁ בְּאָחִיהוּ בָגַד וּפָשַׁע / וְצַדִּיק מָט לִפְנֵי רָשָׁע / וּכְעֶגְלָה דָּשָׁה יְפוּשׁוּן / וְלַמּוֹכִיחַ בַּשַּׁעַר יְקוֹשׁוּן / וְלֹא נִמְצָא כִּי אִם רָשָׁע וְחוֹטֵא – / וְהָעִיר מָלְאָה מֻטֶּה.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

כַּמָּה אֲהַלֵּךְ נְתִיבוֹת, / אָסֹב וְאֶדְרֹךְ דְּרָכִים

קַצְוֵי אֲדָמָה אֲמוֹדֵד, / אָבוֹא קְצוֹת מַהֲלָכִים

אַרְחִיק נְדוֹד בַּיְּעָרִים, / אַקִּיף שְׁבִילִים אֲרֻכִּים!

פַּעַם בְּנֶפֶשׁ חֲרֵדָה / וּבְרַעֲיוֹנִים נְבוּכִים

פַּעַם חֲבֵרַי נְבָלִים, / פַּעַם חֲבֵרַי נְסִיכִים

פַּעַם מְלוֹנִי בְאַשְׁפָּה, / פַּעַם בְּבָתִים עֲרוּכִים

פַּעַם אֲמַשֵּׁשׁ בְּלִי נֵר, / אֶתְעֶה וְאֵלֵךְ חֲשֵׁכִים

לִדְרֹשׁ וְלָתוּר מְקוֹמוֹת / אִישִׁים בְּשֶׂכֶל סְמוּכִים.

אַנְשֵׁי אֱמֶת אֹהֲבֵי טוֹב / מִשָּׁוְא וּמֵחֵטְא חֲשׂוּכִים

הֵן אַחֲרֵי מִשְׁאֲלוֹתַי / הֵם מַאֲוַיַּי מְשׁוּכִים

וָאֶפְגְּשָׁה עִם חֲשַׁבְתִּים / זַכִּים וּבַתּוֹם תְּמוּכִים –

הֵן תַּאֲוָתְךָ לְבָבִי – / הִיא עִיר סְגֻלַּת מְלָכִים

הֵן אִם זְמַן יִנְשְׁכֵנִי, / פֹּה יִחְיוּ הַנְּשׁוּכִים.

וָאֶחְקְרָה – וְהִנֵּה / פֹּה מַחֲשָׁבוֹת הֲפוּכִים

כִּי הִיא מְלֵאָה מְתֵי שְָׁוְא / מִזֵּד וּמֵאִישׁ תְּכָכִים

וַתַּעֲלֶה קִמְּשׁוֹנִים / כֻּלָּהּ כְּסִירִים סְבוּכִים

עוֹבְדֵי מְלֶכֶת מְרוֹמִים / וַיִּסְּכוּ לָהּ נְסָכִים

כָּל מַאֲכָלָם פְּתָנִים, / יֵינָם חֲמָסִים מְסוּכִים

עֹשֶׁק וְגֶזֶל בְּיָדָם / נִמְצָא וְדָמִים שְׁפוּכִים

חֶלְקָם בְּאֶרֶץ יְקֻלַּל, / יִהְיוּ כְרֶגַע הֲדוּכִים

יֹאבְדוּ מְהֵרָה וְיִהְיוּ / נֵרוֹת מְאוֹרָם דְּעוּכִים

רָמְחֵי זְרוֹעוֹת נְטוּיוֹת / יִהְיוּ בְלִבָּם מְעוּכִים!

וּמְתֵי אֱמוּנָה וְצֶדֶק / יִהְיוּ לְצוּרָם בְּרוּכִים

וִירַנֲנוּ אָז בְּטוּב לֵב / ה׳ צּוֹעֲקִים מַר וּבוֹכִים.

אֵלֵךְ אֲבַקֵּשׁ שְׁכֵן טוֹב / תּוֹךְ עַם וּמְנֹעָרִים חֲנוּכִים

עַם יִשְׁפְּטוּ דַּל וְיָתוֹם, / כְּרָשׁ כְּעָשִׁיר כְּמָכִים

לֹא יֶהְדְּרוּ דַל בְּרִיב, לֹא / יִשְׂאוּ פְּנֵי דַּל וְדַכִּים

צוֹפִים רְצוֹן בּוֹרְאָם וּלְ/חַסְדּוֹ מְקַוִּים וְחוֹכִים

יוֹשְׁבִים בְּסֵתֶר שְׁלוֹמִים / בִּכְנַף חֲנִינָה סְכוּכִים

לֹא יִירְאוּ יוֹם חֲרוֹן אַף, / כִּי אַף וְחָרוֹן שְׂכוּכִים.


וָאֹמַר: כִּי רָאִיתִי מַטֵּי מִשְׁפָּט לוֹקְחֵי שׁוֹחַד / עַתָּה אֶסְָּפֶה יוֹם אֶחָד / כִּי דַרְכָּם מְשׂוּכַת חֵדֶק וְחַתְחַתִּים. / אֵין לִי טוֹב כִּי אִם הִמָּלֵט אֶל אֶרֶץ פְּלִשְׁתִּים.

אֲנִי טֶרֶם אֲכַלֶּה לְדַבֵּר אֶל לִבִּי / וָאֶשְׁמַע קוֹל שָׁאוֹן וְקוֹל מִלְחָמָה / וָאֹמַר: מַדּוּעַ קוֹל הַקִּרְיָה הוֹמָה? / וְהִנֵּה אִישׁ זָקֵן [יוֹצֵא] אֶל רְחוֹב הָעִיר אֶל הַמַּעְגָּלָה / וַיִּצְעֲקוּ מִכֹּל חוֹגֵר חֲגוֹרָה וָמַעְלָה. / הוּא מִתְנַהֵל לְאַט לְרַגְלוֹ / וְנִשְׁעָן עַל מַקְלוֹ / וְרוּחַ לֹא יָבֹא בֵּין פֶּשַׂע לְפֶשַׂע / כְּאִישׁ חַף תָּם בְּלִי פֶּשַׁע. / וָאֹמַר: עַתָּה תִצְמַח צְדָקָה וְיִכְלֶה הָרֶשַׁע / תִּפְתַּח הָאָרֶץ וְיִפְרֶה יֶשַׁע.

וּשְׁמוֹ עַכְבּוֹר / מֵהַר תָּבוֹר, / וְלוֹ זָקָן וַעֲנָפִים יוֹצְאִים מִמֶּנָּה / וַיֵּחָצוּ הֵנָּה וָהֵנָּה / כִפְרֹשׂ מִבְרָחֶיהָ וַאֲגַפֶּיהָ / וְאֶבְרוֹתֶיהָ וּכְנָפֶיהָ / שָׁם יִרְעֶה עֵגֶל וְשָׁם יִרְבַּץ וְכִלָּה סְעִיפֶיהָ / וַיִּתְמְהוּ בְמַרְאֶיהָ / וַיִּבָּהֲלוּן כָּל רוֹאֶיהָ / וַתָּבֹא כָל הָעֵדָה לַחֲזוֹתָהּ וַתִּקָּהֵל / וַתָּבֹא אֵלֶיהָ וַתֵּלֶא וַתִּגַּע עָדֶיהָ וַתִּבָּהֵל. / וּבִכְבֵדוּתָהּ יֵלֵךְ שְׁחוֹחַ / וְלֹא עָצַר כֹּחַ / וַיֵּלֵךְ הָלֹךְ בֵּין צְבָאוֹ / רוֹבֵץ תַּחַת מַשָּׂאוֹ / כִּי הָעָם אַחֲרָיו כַּחוֹל אֲשֶׁר לֹא יִמַּד / עַד בּוֹאוֹ אֶל הָרְחוֹב אֲשֶׁר הָעָם שָׁם נִצְמָד / וְכָל הַבָּא אֵלָיו וְעָמַד.> וָאַבִּיט וְהִנֵּה מַצִּיב לוֹ יָד, / כִּי הֵכִינוּ בָרְחוֹב כִּסֵּא / אֲשֶׁר יַעֲלֶה שָׁם בְּיוֹם הַמַּעֲשֶׂה. / וַיֵּלֵךְ וַיַעֲמֹד כְּמַלְאָךְ / וְעַל מְשָׁרְתוֹ הָיָה נִסְמָךְ./ וָאֹמַר: עַתָּה יֵיטִיב [עִמָּכֶם] / אֲשֶׁר מַלְאָךְ שָׁלַח לָכֶם / וְהָיָה מֵכִין עַל כְּתֵפוֹ לְבוּשׁוֹ / וַיִּשְׁרֹק וַיָּנִיד בְּרֹאשׁוֹ / וָאֶבָּהֵל וָאֶשְׁתָּאֶה / וָאֶשְׁתּוֹמֵם עַל הַמַּרְאֶה.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

אִם זְקָנוֹ זְקַן אֱמֶת / מִשְּׁנַת חֵטְא יְעוֹרֲרֵם

אִם פְּשָׁעִים כְּחוֹל, הֲלֹא / זֶה בְצִדְקוֹ יְכַפְּרֵם

סָרֲרוּ אוֹ יְעַוְּתוּ / זֶה כְבָנִים יְיַסְּרֵם

נִפְרְדוּ אוֹ יְפֻזְּרוּ / זֶה כְרוֹעֶה יְפַקְּדֵם

הֶעֱווּ אוֹ יְעַוְְּּתוּ / בַּסְּלִיחָה יְבַשְּׂרֵם,

בַּחֲטָאִים נְגֹאֲלוּ / אֵל בְּלֵב־יָם יְנַעֲרֵם

כִּי בְסֻכּוֹ יְחַבְּאֵם / גַּם בְּסִתְרוֹ יְסַתְּרֵם.


וְכָל הָעָם סְבִיבָיו מֵהֶם מְדַבְּרִים וּמֵהֶם מַחֲרִישִׁים / מֵהֶם מְצַפְצְפִים וּמֵהֶם מִתְלַחֲשִׁים. / וַיִּקְרָא לַנַּעַר מְשָׁרְתוֹ וּפְקִידוֹ / אֲשֶׁר עַכְבּוֹר נִשְׁעָן עַל יָדוֹ. / וַיֹּאמֶר בִּגְעָרָה: ״אִם לְצַפְצֵף אוֹ לְהִתְלַחֵשׁ הָיְתָה קְבֻצַּתְכֶם / פְּנוּ וּסְעוּ לָכֶם!״ / וַיְהִי כִרְאוֹתָם כִּי הִתְקַצֵּף מְאֹד / עָמְדוּ וְלֹא עָנוּ עוֹד. / וַיִּפְתַּח שְׂפָתָיו עַכְבּוֹר בְּמִלּוֹת נְכוֹחוֹת / וְדִבְרֵי תוֹכֵחוֹת / וַיִּשָּׂא קוֹלוֹ / וַיָּרֶם יְמִינוֹ וּשְׂמֹאלוֹ. / וַיֹּאמֶר [בְּלָשׁוֹן] רַכָּה / וּבְעַיִן מְבַכָּה: / ״אַשְׁרֵי הָאִישׁ אֲשֶׁר לִבּוֹ זִכָּה / וְדָאַג מֵחַטָּאתוֹ וּמֵעֲווֹנוֹתָיו מְדֻכָּא / וְיָשַׁב בְּסֵתֶר עֶלְיוֹן וּלְרַחֲמָיו חִכָּה / וַיָּשֶׂם לוֹ שָׁם סֻכָּה!


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

לִבִּי פְּנֵי צוּר יְחַלֶּה, / תָּמִיד יְשַׁוַּע לְמַלְכּוֹ

יֶעְרַב שְׁמוֹ עַל לְשׁוֹן אִישׁ / זִכְרוֹ דְּבַשׁ צוּף לְחִכּוֹ.

אֶשְׁרֹק לְחֹרֶב וְזֶרֶם, / כִּי יִצְפְּנֵנִי בְסֻכּוֹ,

קַוּוּ לְיִשְׁעוֹ וְחַכּוּ – / אַשְׁרֵי אֱנוֹשׁ הַמְחַכּוֹ!


אֲבָאֵר אֲמָרַי בַּתְּחִלָּה / בְּהַלֵּל לוֹ נָאוָה תְּהִלָּה / וְשָׁמַיִם וְכוֹכָבִים בָּרָא בְמִלָּה / וְהָאָרֶץ [עַל] בְּלִימָה תָלָה / וְהָאָדָם עָלֶיהָ בָרָא לִהְיוֹת לוֹ סְגֻלָּה. / עַל כֵּן שִׁמְעוּ נָא תְהִלָּה! / הַטִּי נָא אָזְנֵךְ, [עֲדָתִי], וְשִׁמְעִי וְהִוָּסְרִי וְהִכַּנְעִי / וְדִרְכִי בֵית תְּפִלָּה / וְשִׁיתִי לִבֵּךְ אֶל הַמְסִלָּה. / וְהִטַּהֲרִי וְהִתְבָּרֲרִי / וְהִשָּׁמְרִי וְהִזַּהֲרִי, / פֶּן יַשִּׁיאוּךְ בַּעֲלֵי זְקַן חֲרָפוֹת וּמְרָדִים / שְׁדוּפוֹת קָדִים / וְאַבְלִיגָה. וְאַל יְבַהֲלוּךָ אַנְשֵׁי כַחַשׁ וְכַחַד / עוֹלוֹת בְּקָנֶה אֶחָד / הַמְפַתִּים בְּדִבְרֵי סֵפֶר וּמִלָּה / וַיְרַפּוּ אֶת שֶׁבֶר בַּת עַמִּי עַל נְקַלָּה / אֲשֶׁר הֵמָּה נֶאֱנָקִים וְנֶאֱנָחִים / וַיֶּחֱזוּ לוֹ מַשְׂאוֹת שָׁוְא וּמַדּוּחִים. / הֲלֹא יְדַעְתֶּם אֲשֶׁר נֶאֱמַר עַל זְקַן חֲרָפוֹת / אֲשֶׁר מִקָּדִים שְׁדוּפוֹת: / וּשְׁפַטְתִּיךְ מִשְׁפְּטֵי נוֹאֲפוֹת!


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

זָקָן וְשֵׂיבָה אַל יְבַהְלוּךָ / אַל תַּאֲמֵן בָּם עַד אֲשֶׁר תִּבְחַן,

כִּי רֹב פְּעָמִים תַּחֲשֹׁב לִמְצוֹא / חִטָּה – וְתִמְצָא חוֹחַ וְלֹא יוּחַן.

כֵּן תַּחֲשֹׁב אִישׁ מִבְּנֵי חֶסֶד – / הִנּוֹ פְּנֵי עָנִי וְדַל יִטְחַן,

יִתְאַב אֱכֹל פַּת דַּל, כְּכֶלֶב שָׁב / אֶל קִיא וּמִקִּיא יַעֲרֹךְ שֻׁלְחָן.

אַל תֹּאמְרוּ יִצְלַח וְיִמָּלֵט, / כִּי לֹא יְכֻפַּר לוֹ וְלֹא יוּחַן!


עַל כֵּן תַּשְׁפִּילִי גְּאוֹנֵךְ / וְהַרְחִיקִי זְדוֹנֵךְ / וְאַל תִּתְּנִי עַל בִּצְעֵךְ עֲווֹנֵךְ / וְאַל תִּבְטְחִי בְהוֹנֵךְ / כִּי הוֹנֵךְ מוֹנֵךְ / וּבִצְעֵךְ בּוֹצְעֵךְ / הֲלֹא בְיוֹם פְּקֻדַּת עֲווֹנֵךְ / הַרְבָּה יַרְבֶּה עִצְבוֹנֵךְ. / קְחִי מִצְוָה / וְהַרְחִיקִי גַאֲוָה / וְאַל תֵּלְכִי אַחֲרֵי הַתַּאֲוָה / פֶּן תּוֹרִישְׁכֶם צָרֶבֶת הַמִּכְוָה / כַּבְּסוּ מֵרָעָה לִבְּכֶם / שִׁמְעוּ וּתְחִי נַפְשְׁכֶם / עַד מָתַי יִצְעַק עֲשׁוּקְכֶם / וְיִלּוֹן גְזוּלְכֶם / אִם עַל יָתוֹם תַּפִּילוּ וְתִכְרוּ עַל רֵיעֲכֶם? / שִׁמְעִי נָא הָעֵדָה הַקְּבוּצָה, / לְכִי אִיעָצֵךְ נָא עֵצָה / וְכַפְּרִי עֲווֹנֵךְ בַּהֲמוֹנֵךְ / וּפְרִי קְלוֹנֵךְ בְּהוֹנֵךְ, / כִּי אִם הַעֲנִיק תַּעֲנִיקִי לֶעָנִי / בִּצְדָקָה תִכּוֹנָנִי / וְאִם פָּרֹשׂ תִּפְרֹשׂ לָרָעֵב לַחְמְךָ / יְבָרֵךְ [יי] אֶת לַחְמְךָ וְאֶת מֵימֶיךָ. / וּבַדַּל וּבָאֶבְיוֹן אַל תִּגְעַר / וְאַל תְּדַכֵּא עָנִי בַשַּׁעַר / וְאַל יָשֹׁב דַּךְ נִכְלָם / מִבְּשָׂרְךָ אַל תִּתְעַלָּם, / פָּתֹחַ תִּפְתַּח לַעֲנִיֶּךָ אֲשֶׁר בְּקִרְבֶּךָ / וְהַעֲנִיק תַּעֲנִיק לוֹ מִגָּרְנְךָ וּמִיִּקָּבֶךָ, / וְאַל נָא בְרֹאשׁ דַּלִּים תִּשְׁאַף / מַתָּן בַּסֵּתֶר יִכְפֶּה אַף.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וְאִם תִּרְצֶה מְצֹא מֵאָה / לְכָל בֶּצַע – שְׁחַד עַם אֵל

לְעָנִי הַעְנִיק תַּעְנִיק / פְּתַח יָדָךְ לְדַל שׁוֹאֵל

וְיֹאמַר אֵל: בְּרָכָה עַד / בְּלִי דָי קַח לְךָ הוֹאֵל

וּמֵחֵטְא יִפְדְּךָ, יִשָּׂא / פְּשָׁעֶיךָ וְתִגָּאֵל.


וַאֲנִי מַשְׁבִּיעֲכֶם בְּסֵתֶר עֶלְיוֹן / שֶׁתְּרַחֲמוּ עָנִי וְאֶבְיוֹן / וְאַל תִּבְזוּ סֵתֶר אֵל / עֵת יַשְׁבִּיעֲכֶם בּוֹ שׁוֹאֵל – / מְשַׁל הֶעָנִי שֶׁשָּׁאַל עַל דֶּלֶת / לְפִתֵּי מַכֹּלֶת. / וַיִּתֵּן לוֹ בַּעַל הַבַּיִת פְּרוּטָה / וַיַּשְׁבִּיעֵהוּ בְּסֵתֶר עֶלְיוֹן לֵאמֹר ״הוֹסֵף לִי מְעַט חִטָּה / כִּי יֵשׁ לִי נְעָרִים קְטַנִּים / בָּרָעָב וּבַצָּמָא מְעֻנִּים.״ / אֲזַי בַּעַל הַבַּיִת בַּחֲצֵרוֹ פָּשַׁע שִׁקַּר / וְכָפַר בָּעִקָּר / וַיֹּאמַר: ״וְאֵיךְ תַּשְׁבִּיעֵנִי בְסֵתֶר עֶלְיוֹן? / הֲלֹא תִרְאֶה כִּי הִנְנִי עָנִי וְאֶבְיוֹן?״ / וַיְהִי טֶרֶם כִּלָּה לְדַבֵּר / וַיֹּאחֲזוּהוּ צִירִים וָשֶׁבֶר, / וַיַּכֵּהוּ בוֹרְאוֹ בְּעִוָּרוֹן וּבְשִׁגָּעוֹן וּבכִלָּיוֹן / וַיַּעֲבֹר קוֹל: כֹּה יֵעָשֶׂה לְבוֹזֵי סֵתֶר עֶלְיוֹן! / וְאַתֶּם אַחַי הִוָּסְרוּ / אוּלַי בְּיוֹם אַף וְחָרוֹן בְּסֵתֶר עֶלְיוֹן תִּסָּתְרוּ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אַשְׁרֵי אֱנוֹשׁ יֶעְתַּר וְיֵעָתֵר! / צֵל בּוֹרְאוֹ עָלָיו כְּמוֹ כֶתֶר

יִבְטַח בְּיוֹם חָרוֹן וְלֹא יִירָא, / כִּי צֵל ה׳ לוֹ יְהִי סֵתֶר.

מִדּוֹרְכֵי קֶשֶׁת יְהִי נִצָּל / יוֹם כּוֹנֲנוּ חִצָּם עֲלֵי יֶתֶר

נִצָּל וְיִמָּלֵט בְּלֵב יַמִּים / אַךְ מִבְּלִי חוֹבֵל בְּלִי חוֹתֵר.

אִישׁ בֶּן לְסֵתֶר אֵל וּבוֹזֵהוּ – / יִמַּס בְּחַטָּאתוֹ מְסוֹס נֵתֶר

הִנֵּה כְיִתְרוֹן אוֹר עֲלֵי חֹשֶׁךְ / נִסְתָּר עֲלֵי נָחָשׁ יְהִי יֵתֶר.

אֶחְלֹם עֲלֵי שִׁבְתִּי בְסֵתֶר אֵל / אַךְ צוּר יְשׁוּעָתִי יְהִי פוֹתֵר

עִם רַחֲמָיו אֶכְרֹת בְּרִית עֵגֶל / אֶכְרֹת בְּתָוֶךְ אֶעֱבֹר בֶּתֶר

וַאְהַלֲּלָה וַאְזַמְּרָה, אֹמַר: / אַשְׁרֵי אֱנוֹשׁ יֶעְתַּר וְיֵעָתֵר!


וַיְדַבֵּר כָּל דָּבָר תָּמִים וְנָבָר / וְלֹא עָנוּ אוֹתוֹ דָּבָר. / וַיֹּאמַר לָהֶם: ״אַחַי, לָמָּה תְבַקְּשׁוּ בֶצַע וְלָמָּה תַכְבְּדוּ לִבְּכֶם? / הֲלֹא שְׁמַעְתֶּם מְשַׁל שְׁנֵי אֲנָשִׁים שֶׁהָיוּ בְאֶחָד מִגְרָשׁ / אֶחָד עָשִׁיר וְאֶחָד רָשׁ. / וַיְהִי הֶעָשִׁיר סוֹחֵר / בְּכָל יוֹם הָיָה מְשַׁחֵר, / וַיְעַנֶּה הָרָשׁ נַפְשׁוֹ בְּצוֹם / שְׁלֹשָׁה יָמִים לַיְלָה וָיוֹם / וְכִמְעַט לֹא נוֹתְרָה בּוֹ נְשָׁמָה / בְּרָעָב וּבְצָמָא. וַיְהִי בָּרְבִיעִי כִּי צַר לוֹ / וַיִּקַּח בְּיָדוֹ מַקְּלוֹ / וַיֵּצֵא בְּרֹב דְּאָבוֹת / נִשְׁעָן עַל חַסְדֵי רוֹכֵב עֲרָבוֹת / עַד אֲשֶׁר [בָּא] לָאִישׁ הַסּוֹחֵר / וַיֹּאמַר לוֹ: ״סְעַד נָא לְבַב עַבְדְּךָ וְאַל תְּאַחֵר!״ / וַיִּגְעַר בּוֹ וַיַּחְשְׁבֵהוּ לְאַיִן / כִּי הָיָה מִתְרוֹנֵן מִיַּיִן. / וַיֹּאמַר: ״וְאֵיךְ תָּבוֹא אֵלַי בַּשַּׁחַר?״ / וַיַּעֲנֵהוּ: ״כִּי הָיָה לְבָבִי סְחַרְחַר״. / וַיְהִי לְרַגְלָיו סוֹגֵד / וַיֵּלֵךְ וַיֵּשֵׁב לוֹ מִנֶּגֶד. / וַיִּגְעַר בּוֹ שֵׁנִית וַיֹּאמֶר לוֹ: ״הֲלֹא אַתָּה מְבַקֵּשׁ קְלוֹנֶךָ, / יָדַעְתִּי אֶת זְדוֹנֶךָ! / מַה תַּעֲמֹד וּמַה תְּחַשֵּׁב / וְעַל מָה אַתָּה חוֹשֵׁב?״ / וַיַּעֲנֵהוּ הָאוֹרֵחַ: ״הַעַל הֲמוֹנְךָ אֲנִי בוֹטֵחַ? / הֲלֹא אֲנִי עוֹמֵד עַל הוֹפֵךְ לְבָבוֹת / וּמֵפֵר מַחֲשָׁבוֹת / הוּא יַהֲפֹךְ לִבֶּךָ״. / כִּשְׁמוֹעַ הַסּוֹחֵר / וַיַּהֲפָךְ לוֹ אֱלֹהִים לֵב אַחֵר. / וְלוֹ הָיָה אוֹהֵב וְשָׂמוֹ פְקִידוֹ / וְכָל אֲשֶׁר יֵשׁ לוֹ נָתַן בְּיָדוֹ / וַיִּכְתָּב לוֹ לָתֵת אַרְבָּעָה שְׁקָלִים לָרָשׁ / וַתֵּתַע יָדוֹ וְלִבּוֹ נִגְרָשׁ / וַיַּחְשֹׁב לִכְתֹּב לוֹ: תֶּן־לוֹ אַרְבָּעָה שְׁקָלִים וְאַל תּוֹסֶף / וַיִּכְתֹּב: תֶּן־לוֹ אַרְבַּע מֵאוֹת שֶׁקֶל כֶּסֶף. / וַיִּקַּח הָרָשׁ הַסֵּפֶר מִיָּדוֹ / וַיָּבֹא אֶל פְּקִידוֹ / וַיִּשְׁקֹל לוֹ הַכֶּסֶף וַיָשֶׂם בִּסְדִינָיו / וַיָבֹא אֲדוֹנָיו. / וַיָּבֵא הַסּוֹחֵר הַכֶּסֶף וְהַכְּתָב דָּרַשׁ / וַיֹּאמֶר לֶעָנִי: ״הֲלֹא אֶת [אֲשֶׁר] יוֹרִישֶׁךָ אֱלֹהֶיךָ אוֹתוֹ תִירַשׁ / וְהוּא הִתְעָה אֶת יָדִי / לָתֵת לְךָ אֶת הוֹנִי וְהוֹדִי / וְהוּא הַמַּעֲשִׁיר וְהַמּוֹרִישׁ / וְהַנּוֹטֵעַ וְהַמַּשְׁרִישׁ / עוֹצֵר אֱמוּנִים / וּמֵסִיר שָׂפָה לְנֶאֱמָנִים. / וְלוֹ כָל נִשָּׂא וְשָׁפָל וְכָל גֹּאֲלוֹ שׁוֹגֵג וּמַשְׁגֶּה.״


וְאָדָם יַחֲשֹׁב נַפְשׁוֹ / כְּלֵב חָכָם – וְהוּא מִשְׁגֶה, וְיֵשׁ חָכָם אֲשֶׁר יִדַּל / וְיֵשׁ סָכָל אֲשֶׁר יִשְׂגֶה, וְיֵשׁ צַדִּיק אֲשֶׁר יִכְשַׁל / וְיֵשׁ מַעְשִׁיר וְהוּא מוּגֶה, וְיֵשׁ שׁוֹאֵל שְׁאֵר וּכְסוּת / עֲלֵי כָּל הַר וְעַל כָּל גֵּיא וְאַחֵר יִשְּׂבְּעָה טוּב אֵל / בְּרֹב שָׁלוֹם לְרֹב יִדְגֶּה וְיֵשׁ יָרוֹן בְּרֵאשִׁיתוֹ – / בְּסוֹפוֹ לַעֲנָה יֶהְגֶּה אֱלֹהִים הוּא אֲשֶׁר יִשְׁפֹּט / יְנַשֵּׂא שַׁח וּמַשְׁפִּיל גֵּא, אֲשֶׁר לוֹ כָל אֱנוֹשׁ חוֹכֶה / וְלוֹ יוֹחִיל וְיֵעָגֶה.


וַיְהִי כְכַלּוֹתוֹ לְחַוּוֹת הַמָּשָׁל / וַיָּקָם אִישׁ מִזִּקְנֵי הַמַּעְגָּל / וַיֹּאמַר: ״הֲלֹא יְדַעְתֶּם זֶה הָאִישׁ בֶּן שְׁמוֹנִים / זָקֵן וּנְשׂוּא פָּנִים / הִכָּלְמוּ מֵאֹרֶךְ זְקָנוֹ וּמֵרוֹב קְצִירֶיהָ / אֲשֶׁר בְּכָל פּוֹארוֹתֶיהָ רָבְצוּ גוּרֶיהָ / וְעַתָּה גִּמְלוּ חֲסָדִים אֶצְלוֹ / וּתְנוּ לוֹ אִישׁ נֶזֶם שְׁלָלוֹ.״ / וַיְהִי בְעָמְדוֹ בֵּין הַמִּשְׁפְּתַיִם / וַיְדַבֵּר כָּזֶה לֹא אַחַת וְלֹא שְׁתַּיִם / וַיִּמַּס לְבַב הָעָם וַיְהִי לְמָיִם. / וַיִּקְרָא הָאִישׁ שְׁנֵי נַעֲרֵי עַכְבּוֹר / וַיֵּלְכוּ אֶל הָעָם כֶּסֶף לִצְבֹּר / וַיֵהָפֵךְ לְבַב הָעָם וָרַךְ וָנָחַת / וַיִּתְּנוּ לוֹ אִישׁ קְשִׂיטָה אַחַת / וְאִישׁ בִּגְדּוֹ וּרְדִידוֹ / כַּאֲשֶׁר תַּשִּׂיג יָדוֹ. / וַיִּשְּׂאוּ שְׁנֵי נְעָרָיו כַּסְפָּם וּזְהָבָם / וַיִּשְׁתַּחוּ עַכְבּוֹר וַיְבָרְכֵם וַיֹּאמֶר אֲלֵיהֶם: ״אָמַרְתִּי לָכֶם כִּי לֹא שָׁאַלְתִּי מֵאִתְכֶם / כִּי אִם אֶת נִדְבַת לְבַבְכֶם.״


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

בּוֹרֵא לְבָבוֹת הוּא אֲשֶׁר / זַכִּים לְחֶפְצוֹ יַהְפְּכֵם

הִכְבִּיד וְהִקְשָׁה חִזְּקָם, / גַּם אֶל רְצוֹנוֹ יִמְשְׁכֵם,

יַקְשֵׁם יְכִינֵם, כַּאֲשֶׁר / יִקְרָא לְמֵי יָם יִשְׁפְּכֵם –

כֵּן נַפְשְׁכֶם עִמִּי וְכֵן / הָפְקַד בְּיָדִי רוּחֲכֶם.


כֵּן יְהִי רַעְוָה שֶׁיַּטֶּה לִרְצוֹנוֹ לְבַבְכֶם / וִיכַוֵּן לְחֶפְצוֹ תַאֲוַתְכֶם / וְלָכֵן יְחַכֶּה לַחֲנַנְכֶם / וְיָרוּם לְרַחֶמְכֶם / וְיִצְרוֹר בִּצְרוֹר הַחַיִּים נַפְשְׁכֶם.״ / וַיִּשְׁתַּחוּ לָהֶם עַכְבּוֹר / וַיֵּלֵךְ וַיַּעֲבֹר, / וְהָעָם הִתְפַּזְּרוּ מֵעָלָיו / וַיֵּלֵךְ אִישׁ לְאֹהָלָיו.

וָאֹמַר בְּלִבִּי: [אֵלֵךְ] אַחַר הַזָּקֵן הַזֶּה / וְאֵדְעָה אִם נָבִיא הוּא אוֹ חוֹזֶה / אִם יָקָר אוֹ נִבְזֶה. / וַיֵּלֵךְ עַכְבּוֹר עִם שְׁנֵי נְעָרָיו / הוּא הוֹלֵךְ וַאֲנִי אַחֲרָיו / וַיָבֹא אֶל בֵּיתוֹ וְנָשַׁל נַעֲלוֹ / וְהִשְׁלִיךְ מַקְלוֹ / וְכֵלָיו שָׂם בְּאָהֳלוֹ. / וּלְבֵיתוֹ עֲשָׂרָה מְבוֹאוֹת / כְּבָתֵי שָׂרֵי צְבָאוֹת. / הָיִיתִי מְסַפֵּר צוּרוֹתָיו / וְחַצְרוֹתָיו וְטִירוֹתָיו / לוּלֵא כִי קָצְרָה לְשׁוֹנִי / לְסַפֵּר רְאוּת עֵינִי. / וּמִי יוּכַל לְסַפֵּר הַמַּיִם וְהַבְּרֵכָה / וְהָעוֹף וְהַשְּׂבָכָה / וּמִינֵי הָעֵצִים / וְהַשּׁוֹשַׁנִּים וְהַצִּצִּים / וְהַכַּפּוֹת וְהַמַּחְתּוֹת / וּכְלֵי הַיַּיִן לִשְׁתּוֹת / וּמִינֵי בְגָדִים לְתִלְבּוֹשֶׁת / וּכְלֵי כֶסֶף וְזָהָב וּנְחֹשֶׁת? / וָאֹמַר בְּלִבִּי: הַמְשָׁאלְתְּךָ מִנְּבָלִים אוֹ מִנְּדִיבִים / עֹשֶׂה עֹשֶׁר וּנְכָסִים רַבִּים. / הֲלֹא יִכָּלֵם לִשְׁאֹל מִיָּקָר וְנִבְזֶה / וְעָשָׂה לוֹ כָל הַחַיִל הַזֶּה / וָאֵדַע כִּי הוּא מִבְּנֵי הַנְּבָלִים / וְהָאִישׁ קָשֶׁה וְרַע מַעֲלָלִים.

וְיֵשׁ עוֹשֶׂה נְכָסִים מֵחֲמָסִים / וְהִתְעַשֵּׁר בְּדִמְעַת הָעֲשׁוּקִים

וְיֵשׁ אִישִׁים בְּהוֹד שֵׂיבָה וְזָקָן – / וְטוֹב מֵהֶם בְּקִיר אַנְשֵׁי מְחֻקִּים

וְיֵשׁ זָקֵן הֲדוּרוֹת שַׂעֲרוֹתָיו – / וְטוֹב מֵהֶם בְּמוֹ זֶפֶת דְּבוּקִים

וְיֵשׁ זָקַָן אֲרוּכָה – וְאֵין אֲרוּכָה / בְּחַטָּאתָהּ, שְׁכֵנֶיהָ נְמַקִּים.

בְּמֹחֵיהֶם הֲלֹא יַשְׁקוּ זְקָנָם / וְרָאשֵׁיהֶם בְּמוֹ מוֹחָם בְּקוּקִים

וְיֵשׁ אִישִׁים בְּעֵינֵיהֶם חֲכָמִים / וּמֵחָכְמָה הֲלֹא הֵמָּה רְחוֹקִים

וְיֵשׁ אִישִׁים כְּחוֹל יַמִּים כְּבֵדִים / וְתַחְתֵּיהֶם מְתֵי שֶׂכֶל מְעִיקִים

וְיֵשׁ אִישִׁים בְּצַד חָכָם לְשִׂיכִּים / וְכִצְנִינִים בְּעֵינָיו הֵם מְצִיקִים

וְיֵשׁ זָקָן וּבַעְלָהּ אִישׁ תְּכָכִים / וְנִכְסֵי דַּל לְחִכּוֹ מַמְתַּקִּים

וּמִחְיָתוֹ גְּזֵלַת דַּל, וְיַעְשִׁיר / וְיִבְלַע כָּל שְׂאוֹר מַצּוֹת רְקִיקִים.

וְלֹא זָכְרוּ לְמַקְשִׁיב צַעֲקַת דַּל / וְשׁוֹמֵעַ לְשַׁוְעַת נֶאֱנָקִים

אֲשֶׁר הָיוּ מְמַלְּאִים שַׂק בְּגֶזֶל / לְמָחָר אַמְתְּחוֹתֵיהֶם מְרִיקִים.


וַיְהִי עַד כֹּה וְעַד כֹּה – וַתֵּצֵאנָה אַרְבַּע נְעָרוֹת / כְּיָרֵחַ הוֹלְכוֹת יְקָרוֹת / וַתִּשַּׁקְנָה בְיָדָיו / וַתַּפְשִׁיטְנָה בְגָדָיו. / וַיֵּשְׁבוּ עַל הַשֻּׁלְחָן וַיָּשִׂימוּ

לְפָנָיו כָּל מַאֲכָלָיו / וּפְרִי מַעֲדַנָּיו / וְכַפּוֹת וּגְבִיעִים / מִקְלָעוֹת פְּקָעִים / וְיַיִן יָשָׁן / וְצִיצִים וְשׁוֹשַׁן / וּמַיִם מְהַלְּכִים / וּסְגֻלַּת מְלָכִים / וַיֵּשֵׁב עַכְבּוֹר עַל מְסִבּוֹ

וַיְהִי כְטוֹב לִבּוֹ / וַיֹּאמַר לְנַעֲרוֹתָיו: ״הִתְעוֹרַרְנָה / וְשִׁיר דַּבֵּרְנָה!״ / וּבִידֵיהֶן כִּנּוֹרוֹת וּנְבָלִים / וּמְנַעְנְעִים וְצִלְצְלִים.


וַתַּעַן הָאַחַת וַתֹּאמַר:

יַיִן כְּעֵין בָּרָק כְּעֵין לַהַב / יַדְלִיק לְבַב שׁוֹתָיו בְּכִידוֹדוֹ, /

יֵמַר, וְאָכֵן מִדְּבַשׁ יִמְתַּק, / רֵיחוֹ יְחַיֶּה כִּי כְמֹר הוֹדוֹ.

אִם יִמְצְאָה יָגוֹן בְּלֵב כָּל אִישׁ, / יִשְׁלַח גְּדוּד מָשׂוֹשׂ לְהַשְׁמִידוֹ,

יָשִׁית לְבַב כִּילַי כְּלֵב נָדִיב / כִּי בוֹ קְפוֹץ יָד יִפְתְּחָה יָדוֹ.


וַתַּעַן הַשֵּׁנִית וַתֹּאמַר:

יַיִן אֲשֶׁר יָאִיר בְּעַד כּוֹסוֹת / דָּמוּ אֱלֵי מַיִם אֲשֶׁר קָפְאוּ

לֹא נִסְתְּרָה עֵינוֹ בְּעַד הַכּוֹס / וּשְׁבִיב בְּעַד סֵתֶר הֲלֹא תִרְאוּ.

וָאֹמְרָה עֵת נִפְלְאוּ מֵהֶם / סוֹד כּוֹס וְאוֹר [יֵינָהּ] וּבָם נִלְאוּ:

כַּאֲשֶׁר יְאִירוּן כּוֹכְבֵי חַמָּה / וּבְעַד שְׁחָקִים הֵם יְחֻבָּאוּ.


וַתַּעַן הַשְּׁלִישִׁית וַתֹּאמַר:

וְעָפְרָה כָּסְתָה לֶחְיָהּ בְּשָׁנִי / בְּיָדָהּ כּוֹס כְּאִילוּ דַם חֲלָלִים

וּמֵאֹדֶם שְׁתֵיהֶן אָדְמוּ כָל / פְּנֵי רוֹאִים וּמַלְבּוּשִׁים וְכֵלִים

עֲדֵי לֵיל מֵרֹב מָאוֹר יְכֻסֶּה / מְאוֹרֵי רוֹם פְּנֵי אוֹרָם מְחַלִּים

וְשִׁית עָפְרָה וְכוֹסָהּ בֶּאֱשׁוּן [לֵיל] / אֲזַי יֵאוֹר וְנָסוּ הַצְּלָלִים.


וַתַּעַן הָרְבִיעִית וַתֹּאמַר:

וּבַת כֶּרֶם בְּעֵת תּוּרַק בִּדְמָעוֹת / צְבִיַּת חֵן בְּהִתְנוֹדֵד יְדִידָהּ

דְּאָגוֹת תַּהֲרֹג רֶגַע, כְּאִלּוּ / לְהַשְׁמִיד כָּל [כְּאֵב] חֶרֶב בְּיָדָהּ.

וְלֵב נָמֵס כְּלֵב אַרְיֵה תְּשִׁיתוֹ, / וְתָעִיר אַהֲבַת דּוֹדִים בְּהוֹדָהּ.

וְאִם תֹּאמַר: וְאֵיךְ אוֹרָהּ יְנוֹצֵץ / בְּעַד הַכּוֹס וְנִגְלֶה כָל יְסוֹדָהּ?

הֲלֹא תִרְאֶה יְפַת גַּן יַעֲלָה תּוֹךְ / רְדִּידָהּ – כֵּן חֲשֹׁב הַכּוֹס רְדִידָהּ.

בְּכֵן קוּמָה, הֲלֹא זֶה עֵת שְׁתִיָּה / וְהִתְעַנֵּג, יְדִיד, מִזִּיו כְּבוֹדָה!


וַיֹּאמֶר עַכְבּוֹר: ״וְהִנֵּה בְפִיכֶן הֲטִיבוֹתֶן /

וּבִידֵיכֶן מִלֵּאתֶן / וּבְטוּב שִׁירֵיכֶן / הִרְבֵּיתֶן חֲלָלֵיכֶן. /

וְכַאֲשֶׁר הֲעִירוֹתֶן חִשְׁקִי / אַעֲנֶה אַף אֲנִי חֶלְקִי.״


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

כּוֹסוֹת בְּדֹלַח מִבְּלִי יַיִן / צַחוֹת לְבָנוֹת הֵן מְזֻקָּקוֹת

יִתְמַלְּאוּן דִּמְעַת כְּרָמִים, אָז / אֶרְאֵם אֲדַמְדַּמּוֹת יְרַקְרַקּוֹת,

וַיִּרְקְדוּ תוֹכָן בְּנֵי רֶשֶׁף, / יִתְמַלְּאוּ כִּמְאַזְרֵי זִיקוֹת

הִתְבּוֹשֲׁשׁוּ אוֹ נִכְלְמוּ אוֹ אֵיךְ / לֹא עַיִן מְלֵאוֹת הִיא כְעֵין רֵקוֹת?

כֵּן חֹק צְבָאוֹת עֵת פְּגֹשׁ דּוֹדִים / יִתְאַדְּמוּ פָּנִים וְגַם רַקּוֹת:

יִתְאַדְּמוּ מִדַּם לְבַב חֹשְׁקִים, / כִּי לֵב יְדִידִים גֹּזְלוֹת עֹשְׁקוֹת.

קוּמִי, צְבִיַּת חֵן, וְהַשְׁקִינִי / כֹּסוֹת גְּדוֹלוֹת אַחֲרֵי רַקּוֹת!

מִי־יִתְּנָה כוֹסִי שְׂפַת רַעְיָה / כִּי מָתְקָה מִכָּל כְּלֵי הַשְׁקוֹת!

כִּי־כֵן תְּעוּדַת חֹשְׁקִים לִשְׁתּוֹת / בִּשְׂפַת צְבִיַּת חֵן וְכֵן חֻקּוֹת.


וַתֹּאמַרְנָה לוֹ הַנְּעָרוֹת: ״הֲטִיבוֹתָ וְשִׁירֶיךָ מִשִּׁירִים נֶחְמָדִים / רַבִּים וְנִכְבָּדִים!״ / וַיָּקָם אֲלֵיהֶן / וַיִּשְׂחַק עִמָּהֶן / וְיְהִי כְמִתְנַפֵּל מֵחֵיק אֶל חֵיק / וַיָּקוּמוּ לְשַׂחֵק / וּבְשִׂמְחַת לִבּוֹ הָיָה מְרַקֵּד / כִּי הָיָה כְּאֵשׁ יוֹקֵד / עַד כִּי יָגַע וְשָׁכַב וְסָגַר עַפְעַפָּיו / וְכִסָּה אַפָּיו / וַיֵּרָדֵם בְּתַרְדֵּמוֹת. וַתָּבוֹאנָה הַנְּעָרוֹת בֶּחָצֵר הַפְּנִימִית – וּבְכָל זֹאת אֵינוֹ רוֹאֶה / מַחֲרִישׁ וַיְהִי בְלֶכְתָּן וַיִּקְרָא חֲרִישִׁית / וַתֵּצֵא מֵהַחֶדֶר אִשָּׁה כֻלָּה כּוּשִׁית, / וְלָהּ שָׂפָה / כְּאוּד מוּצָל מִשְּׂרֵיפָה / וְעֵינֶיהָ כִרְשָׁפִים / וְאַפֶּיהָ שְׁקוּפִים. / וָאֹמַר ״לַכֹּל נִיאוּפִים!״ / וָאֶרְגַּז לְרוֹעַ מַעַלְלֵיהֶם / וָאַכֵּם וָאַפְשִׁיט לְבוּשָׁם מֵעֲלֵיהֶם / וָאֶעֶזְבֵם עֲרֻמִּים וְשׁוֹלֲלִים / כִּי קִנֵּאתִי בַּהוֹלְלִים. / וַיֹּאמֶר עַכְבּוֹר: ״הֶרֶף יָדֶיךָ! / לָמָּה תַעֲשֶׂה כֹה לַעֲבָדֶיךָ? / כִּי אֵין לֵב לַבּוֹעֲרִים / בַּתְּאֵנִים הַשּׁוֹעֲרִים״.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

יוֹם יֶחֱלוּ מָתְנַי אֲזַי / סוֹד עַל תְּאֵנִים אַעֲרִים

אַךְ שֹׁעֲרֵיהֶן לִי צֳרִי – / לִמְדוּ רְפוּאוֹת, בֹּעֲרִים!


וָאֶגְעַר עַל סִכְלוֹ וְעַל מַעֲלוֹ / וַיּוֹסֶף שֵׂאת מְשָׁלוֹ:

מַה תִּמְאֲסוּ כוּשִׁית וְעֵינְכֶם לֹא / תִּכּוֹן בְלִי הַדְרַת שְׁחַרְחֹרֶת

אֶבְחַר שְׁחַרְחֹרֶת וְאַל בָּרָה / כִּי אֶחְשְׁבֶנָּה לִי לְבַהֶרֶת

כתמו בשוק צלמה וחתולה / אֶל יַד שְׁעָרִים אוֹ לְפִי קֶרֶת.


וָאֶעֱנֵהוּ עַל חֶסְרוֹן דַּעְתּוֹ: / ״עֲנֵה כְסִיל כְּאִוַּלְתּוֹ!״

מַה לַשְּׁחַרְחֹרֶת אֲשֶׁר תִּבְאַשׁ / וּלְנַעֲרָה בְּמוֹר מְקֻטֶּרֶת?

בָּרָה כְשׁוֹשַׁנָּה בְּיַד חוֹשֵׁק / מַחְמָד לְכָל עַיִן וְתִפְאֶרֶת

כִּי כָל שְׁחוֹרָה הִיא בְלִי לֵבָב / פּוֹתָה וְהוֹמִיָּה וְסוֹרֶרֶת

אֹמַר לְכָל כּוּשִׁית דְּבַר סֵתֶר: / שׁוּבִי, לְכִי בִמְקוֹם אֲשֶׁר נִבְרֵאת!


וָאֹמַר לוֹ: ״מַה לְּךָ, עַכְבּוֹר, / לִשְׁתּוֹת מֵי שִׁיחוֹר?״ / וַיַּעֲנֵנִי לֵאמֹר: / ״הֲלֹא כָל אוֹהֲבֵי נְעָרוֹת בָּרוֹת / רוֹכְבֵי אֲתוֹנוֹת צְחוֹרוֹת!״ / וָאֶצְעַק צְעָקָה מָרָה / וַתָּבֹאנָה הַנְּעָרוֹת כִּי הֲרִימוֹתִי קוֹלִי וָאֶקְרָא. / וָאַגִּיד לָהֶן דְּבַר עַכְבּוֹר וְגִדְּפוּהוּ / וּבְחֲרָפוֹת גְּדוֹלוֹת חֵרְפוּהוּ / וַתִּלְעַגְנָה עַל שַׂעֲרוֹת זְקָנוֹ אֲשֶׁר אָרְכוּ וְהִלְבִּינוּ / וְלֹא יָדְעוּ וְלֹא הֵבִינוּ, / וְהַבָּרָה כְנֶגְדּוֹ כַּשְּׁחוֹרָה / וְהַשְּׁחוֹרָה כָּאוֹרָה, / וְתַאֲוָתוֹ מֵהַשְּׁחוֹרוֹת מִי יְשִׁיבֶנָּה / כִּי אֲפֵלוֹת הֵנָּה.


וַתֹּאמֶר הָאֶחָת: ״מְרוֹחַ אֶשֶׁךְ / חָתַר בַּחֹשֶׁךְ! / וְהִנֵּה רָאִיתִי זְקָנוֹ כִּי נָתְנָה בְלֵב רוֹאֶיהָ מוֹרָא וְחִתִּית / וַתִּגַּע עַד שְׁאוֹל תַּחְתִּית / וַעֲנָפֶיהָ רַעֲנַנִּים / מַחֲסֶה לַשְּׁפָנִים, / בֵּין פֹּארָה וּפֹארָה מַחֲבוֹא / שׁוּעָלִים הִלְּכוּ בוֹ / וּפַרְעוֹשִׁים שָׁם נִרְדָּמִים, / כִּי אָרְכוּ לָהֶם שָׁם הַיָּמִים.״


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

זְקַן עַכְבּוֹר – זְקַן רוֹעֵי אֱוִילִים / סְעִיפֶיהָ לְכָל עֵבֶר וּפָנִים,

וְאָרְכָּהּ עַד אֲשֶׁר נָגְעָה שְׁאוֹלָה / בְּרֹב אוֹנִים עֲדֵי הַלְאוֹת תְּאוּנִים.

בְּתוֹכָהּ יֵלְכוּ חַיּוֹת קְטַנּוֹת / וְשָׁאַלְתִּי: הֲקוֹפִים הַסְּפוּנִים

וְרָאִיתִי כְּעוֹף תּוֹכָהּ יְעוּפוּן / וּפַרְעוֹשִׁים יְרָשׁוּהָ לְקִנִּים

הֱשִׁיבוּנִי: זְקַן עַכְבּוֹר כְּיַעַר / וְשָׁם צִיִּים וְשׁוּעָלִים קְטַנִּים

וְשָׁמָּה קִּנֲּנָה קִפּוֹז תְּמַלֵּט / וְשָׁמָּה מַחֲסֶה אַף לִשְׁפַנִּים,

וְשָׁמָּה יִשְׁכְּנוּ בֶּטַח וְהַשְׁקֵט / וְאִישׁ תַּחַת גְּפָנִים שַׁאֲנַנִּים,

וְאִישׁ תַּחַת תְּאֵנָתוֹ בְּצֵל שֶׁל / זְקַן עַכְבּוֹר דְּשֵׁנִים רַעֲנַנִּים.


וַתֹּאמַר הַשֵּׁנִית: ״הִנֵּה רָאִיתִי עַכְבּוֹר כִּי הִלְבִּין מִשֵּׂיבוֹ / וְשֵׂעָר שָׁחֹר אֵין בּוֹ / וְאֶחְשְׁבֶנּוּ חַף מִפְּשָׁעָיו / וּבַחֹשֶׁךְ רִפֵּד יְצוּעָיו / וְזָנָה אֶת בְּנוֹת חָם / וּפָעַל בַּפֶּחָם. / עַל כֵּן קָרָאתִי עַל זְקָנוֹ הַנְּבָלָה / הוֹי מוֹרְאָה וְנִגְאָלָה! / וְאָמְנָם קֹדֶם חֲשַׁבְתִּיהָ זְקַן דַּעַת וְחָכְמָה / וְהִנֵּה זְקַן בֹּשֶׁת וְחֵמָה.


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

זְקַן עַכְבּוֹר חֲשַׁבְתִּיהָ / זְקַן אִישׁ מַחֲלָה יִרְפָּא,

זְקַן צֶדֶק יֹשֶׁר / זְקַן כָּבוֹד זְקַן חֻפָּה,

וְהִנֵּה הִיא זְקַן בֹּשֶׁת / זְקַן קָלוֹן זְקַן חֶרְפָּה

וְנָאָה עַל שְׁחֹר פָּנָיו / כְּדֹמֶן עַל פְּנֵי אַשְׁפָּה.

תְּגֻלַּח אַךְ בְּלִי תַעַר / בְּמוֹ חֶרֶשׂ תְּהִי נִסְפָּה,

וְשָׂטָן יַעֲמֹד נֶגְדּוֹ / בְּעֵינָיו הַשְּׁחוֹר יִפָּה,

עֲדֵי חֹשֶׁךְ לְאוֹרָה שָׂם / וְיַד יֹפִי וְחֵן רִפָּה

וּבָחַר מַק מְקוֹם בֹּשֶׂם / וְתַחַת הַחֲגוֹר נִקְפָּה.


וַתֹּאמֶר הַשְּׁלִישִׁית: ״הִנֵּה רָאִיתִי זְקַן עַכְבּוֹר הֲדוּרָה וְיָפָה / פּוֹרִיָּה וַעֲנֵפָה, / וְחָשַׁבְתִּי שַׂעֲרוֹתֶיהָ כְּשֵׁרוֹת וּנְבָרוֹת / וַתַּעַשׂ בַּדִּים וַתְּשַׁלַּח פּאוֹרוֹת / וַתֶּאֱרַכְנָה עַד קְצֵה שׁוּלָיו / וַתִּגַּע לְרַגְלָיו. / וְהִגִּיעַ שַׂעַר לְשַׂעַר עָנָף / נוֹגְעוֹת כָּנָף אֶל כָּנָף. / וָאֹמַר: אֶחְשֹׁב – וְאֵיךְ אֶחְשֹׁב? – שַׂעֲרוֹת זְקָנוֹ שַׂעֲרוֹת דַּעַת וְכִשְׁרוֹן – וְהֵן בְּמוֹרַד בֵּית חוֹרוֹן!״


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

וּמִי גֶבֶר יְפֵה אִישִׁים כְּעַכְבּוֹר / כְּאִילוּ עַל זְקָנוֹ הוֹד מְלוּכָה?

וּבוֹ נִפְתָּה לְבָבִי כִּי רְאִיתִיו / אֲרֹךְ זָקָן – וְהֵן אֵין אֲרוּכָה.

חֲקַרְתִּיהָ וְהִנֵּה הִיא שְׁתוּלָה / בְּאַשְׁפַּתּוֹת עֲלֵי דֹמֶן מְעוּכָה,

וְשָׁרְשָׁהּ עַל לְחִי, אוּלָם קְצוֹתָהּ / עֲדֵי מוֹרָד לְבֵית חוֹרוֹן מְשׁוּכָה.

עֲנָפֶיהָ לְכָל עֵבֶר פְּזוּרִים, / וְכָל פֹּארָה לְכָל פֵּאָה נְבוֹכָה.

וּבָהּ צֶמֶר לְמִרְדַּעַת פְּרָדִים / וּמַסֵּכָה, לְגִדְרֹת צֹאן שְׂבָכָה

שְׂמִיכָה תִהְיֶה לָךְ עַל אֲזֵנָךְ / וְצֵאתָךְ תְּכַסֶּה בַשְּׂמִיכָה.


וַתֹּאמַר הָרְבִיעִית: ״מִי יִתֵּן וְיִרחֹץ אֲשֶׁר בָּהּ חֶלְאָתָהּ / וְהֵסִיר אֶת מַרְאָתוֹ בְנוֹצָתָהּ! / הֲלֹא כִּי שַׂעֲרוֹתֶיהָ חֲטָאִים וְרָעִים – / הֶבֶל הֵמָּה מַעֲשֵׂה תַּעְתּוּעִים. / וְאִם הִזְקִינָה לְהֶבֶל וְהִלְבִּינָה כַבָּרָד, אַךְ אֶל שְׁאוֹל תּוּרַד.״


וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:

וְרָאִיתִי זְקַן עַכְבּוֹר / וְלִבִּי זַע וְגַם חָרַד

חֲשַׁבְתִּיהוּ לְאִישׁ תָּמִים / וְהִנֵּה הוּא בְצוּר מָרַד

וְאִם הִזְקִין זְקָנוֹ עָשׁ, / כְּשֶׁלֶג שָׁב וְכַבָּרָד

וְאִם הֻשְׁקָה בְּמֵי חַטָּאת, / הֲלֹא אַךְ אֶל שְׁאוֹל תּוּרַד.


וַתָּבוֹאנָה אַרְבָּעְתָּן וְהִכּוּהוּ וּמְרָטוּהוּ / וּמִשְׁפַּט נוֹאֲפוֹת שְׁפָטוּהוּ, / זֹאת תִּמְשֹׁךְ זְקָנוֹ וְזֹאת תִּמְרְטֶנּוּ / וַיִּלְחָצוּהוּ וְלֹא הִרְפּוּ מִמֶּנּוּ / וַיָּמָת מִן הַמַּכִּים אֲשֶׁר הִכּוּהוּ / כִּי הָמֵת הֱמִיתוּהוּ. / וַיַּאַסְרוּהוּ בַעֲבוֹתִים / וַיַשְׁלִיכוּהוּ בְאַחַד הַפְּחָתִים. / וַתֹּאמַרְנָה: ״אַל יֵשֵׁב אִישׁ נֶאֱלָח עִמָּנוּ / כִּי חֶרְפָּה הִיא לָנוּ.״

וַיְהִי מִיָּמִים / וַתֵּצֵאנָה אֶל הַכְּרָמִים / לִרְעוֹת בֵַּגַּנִּים / וְלִלְקֹט שׁוֹשַׁנִּים / מִתְהַלְּכוֹת בַּפַּרְדֵסִים / מִתְנַהֲלוֹת בֵּין הַהֲדַסִּים / עֲלֵיהֶן כָּתְנוֹת פַּסִּים / מְסַפְּרוֹת מְשָׁלִים וְחִידוֹת / וְשִׁירוֹת חֲמוּדוֹת / עַל הַסְּמָדַר וְהַגְּפָנִים / וְעַל כָּל מִינֵי הַנִּצָּנִים / וַתְּדַבֵּרְנָה עַל הָעֵצִים / וְעַל הַשּׁוֹשַׁנִּים וְהַצִּיצִים. / וַיְהִי לְעֵת הַצָּהֳרַיִם / וַתֵּשֵׁבְנָה עַל בְּרֵיכוֹת הַמַּיִם / יוֹשְׁבוֹת תַּחַת הַצְּלָלִים / מְנַגְּנוֹת בְּכִינוֹרוֹת וּבִנְבָלִים.

וְהִנֵּה אַרְבָּעָה נְעָרִים מַקְשִׁיבִים לְקוֹל רִנָּתָן / מַאֲזִינִים מְחוֹלָן וּנְגִינָתָן / מִשְׁתַּעְשְׁעִים לְקוֹל הַנְּגִינוֹת / מַשְׁגִּיחִים מִן הַחַלּוֹנוֹת. / וַתֵּדַעְנָה הַדָּבָר וַתֹּאמַרְנָה: / ״מִי הוּא אֲשֶׁר יַקְשִׁיב לְקוֹלֵינוּ / וְעוֹמֵד אַחַר כָּתְלֵינוּ? / הֲלָנוּ אַתָּה אִם לְצָרֵינוּ?״ / וַיְהִי לְעֵת עֶרֶב / וַיָקוּמוּ מִן הַמַּאֲרָב וַיֵּלְכוּ הָלֹךְ וְקָרֵב / וְכַאֲשֶׁר אֲלֵיהֶם נִלְווּ / וַיִּקְדוּ וַיִּשְׁתַּחֲווּ / וְהוֹשִׁיבוּם אִישׁ עַל כַּנּוֹ וְאִישׁ עַל מַחֲנֵהוּ / וַיִּתְמְהוּ אִישׁ אֶל רֵעֵהוּ / עַל יָפְיָן וּמַרְאֵיהֶן / וְעַל זֹהַר פְּנֵיהֶן / וַיֹּאמְרוּ כְמִתְלַחֲשִׁים: / ״מִי יִתֵּן וְתִהְיוּ לָנוּ לְנָשִׁים!״

וַתִּשְׁמַעְנָה הַדָּבָר וַתֹּאמַרְנָה: / ״הֲבִינוּנוּ אִמְרֵיכֶם / וְיִבָּחֲנוּ דִבְרֵיכֶם! / וּבַמֶּה יִוָּדַע כִּי אֱמֶת דִּבַּרְתֶּם וְנֵדְעָה כִּי לֹא מְרַגְלִים אַתֶּם? / וְאֵיךְ [בָּאתֶם] לְגַלּוֹת עֶרְוָתֵנוּ וְלֹא נִכְלַמְתֶּם? וְלָמָּה [לֹא] אֲמַרְתֶּם: אִם עַד כֹּה בְעֵינֵיהֶן נִטְמֵינוּ / עַתָּה נְדַבֵּר וְאַחַר [כַּךְ] אֲלֵיהֶם נִגְלֵינוּ / וְאִם כֹּה יֹאמְרוּ: עֲלוּ אֱלֵינוּ! – וְעָלִינוּ.״


וַתִּשָּׂא אַחַת מֵהֶן מְשָׁלָהּ וֵַתֹּאמַר:

הֶחָמְסוּ דוֹדִים אֲשֶׁר / דַּלְתּוֹת עֳפָרוֹת דָּפְקוּ?

הָהּ נִבְעֲרוּ אוֹ יִכְסְלוּ / אוֹ נוֹאֲלוּ – אוֹ יֶחְשְׁקוּ!

כִּי אֵין לְדוֹדִים בֹּא עֲלֵי / עָפְרָה לְבַד אִם עַשְּׁקוּ,

לֹא לַעֲשׁוּקִים לַעֲשֹׁק / כִּי אִם בְּפִיהֶם יֶחְלְקוּ.

אִם חוֹשְׁקִים הֵם, יִתְּנוּ / מוֹפֵת וְאוֹת – יוֹם יִצְדְּקוּ.

אִם הֵם חֲבִיבֵי אַהֲבָה / אֵיךְ אֵין יְדִיד יִתְאַפֵּקוּ

כִּי חוֹשְׁקִים תָּמִיד דְּוֵי / לֵב הֵם לְהָבִים נִשְּׂקוּ.

מִתְאוֹנֲנִים מִתְהוֹלֲלִים, / כִּי מִכְּאֵב לֵב יִצְעֲקוּ,

אִם חוֹשְׁקִים אַתֶּם – וְאֵי / דִמְעוֹת לְחֵיכֶם תִּצְּקוּ?

אֵי תַחֲנוּנֵי אוֹהֲבִים / אֵי מַאֲמָרִים מָתְקוּ?

אֵי חוֹשְׁקֵי עוֹלָם אֲשֶׁר / מֵעַל עֳפָרוֹת חָזְקוּ,

הַנּוֹבְבִים חִשְׁקָם עֲלֵי / לוּחַ לְבָבָם חָקְקוּ?

כַּלּוּ וְהַעְלוּתָם בְּכָל / אִם גָּבְרוּ גַּם עָתְקוּ.

דּוֹדִים, בְּזֹאת תִּבָּחֲנוּ / כִּי לֹא בְרִיב תִּתְעַסְּקוּ,

קוּמוּ וְחוּשׁוּ מַהֲרוּ / מוֹסְרוֹת נְדוּדִים נַתְּקוּ,

הִתְנַחֲלוּ אִישׁ רַעְיָה / אָכֵן בְּגוֹרָל תַּחְלְקוּ.

כִּי יַעֲזֹב אִישׁ אָב וְאֵם / וּבְיַעֲלַת חֵן יִדְּבְּקוּ,

כִּי אָז יְמַהְרוּ וִינְחֲשׁוּ, / כָּל אִישׁ בְּאִשְׁתּוֹ יֶחְזְקוּ,

וִיהִי נְעוּרִים חַדְּשׁוּ, / עַל אוֹהֲבִים הִתְרַפְּקוּ!
