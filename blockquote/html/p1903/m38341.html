משל פג: גַּנָּב וּמְכַשֵׁפָה

לֹא תִּשְׁמַע לָרְשָׁעִים / מְחַזְּקֵי יְדֵי מְרֵעִים


גַּנָּב יִנְהַק בֵּין הַשִּׂיחִים / וְיִקְטֹף מֵהֶם מַלּוּחִים / בָּעֶרֶב כְּבֹא הַשֶּׁמֶשׁ יַעֲמוֹד / בְּחֶסֶר וּבְכָפֶן גַּלְמוּד / מִקֶּטֶב יָשׁוּד צָהֳרָיִם / עָיֵף וְיָגֵעַ וּרְפֵה יָדַיִם / וְלִשְׁתּוֹת נַפְשׁוֹ נִכְסָפָה / דֶּרֶךְ שָׁם בָּאָה מְכַשֵּׁפָה / וַתִּשְׁמַע כִּי יִצְעַק לַמַּיִם / וַתְּמַלֵּא הַחֵמֶת מִמְּטַר הַשָּׁמַיִם / וְהוּא נִכְנַע וּמִתְוַדֶּה פְשָׁעָיו / מְקַלֵּל בְּזַעְמוֹ אֶת רְגָעָיו / כַּאֲשֶׁר כִּלָּה בְּבֹשֶׁת יָמָיו / וְחֶטְאוֹ פָרַשׂ רֶשֶׁת לִפְעָמָיו / אוֹי לִי אִמִּי כִּי יְלִדְתִּנִי / כִּי כָל הַיְקוּם מְקַלְלָוְנִי / בִּפְשָׁעַי וְחַטֹּאתִַי / הִשִּׂיגוּנִי עֲווֹנוֹתַי / וְהַמְּכַשֵּׁפָה מִן הַחֵמֶת תַּשְׁקֵהוּ / וְעַל אֲשֶׁר נִחַם עַל הָרָעָה תְרִיבֵהוּ / וַתְּמַהֵר לָגֶשֶׁת אֶצְלוֹ / הֵעֵזָּה פָנֶיהָ וַתֹּאמֶר לוֹ / הִנֵּה מְצָאתִיךָ מִתְבּוֹדֵד / וּשְׁמַעְתִּיךָ מִתְנוֹדֵד / עַל אֲשֶׁר רַבּוּ פְשָׁעֶיךָ / וַתְּקַלֵּל יָמֶיךָ וּרְגָעֶיךָ / עַל כֵּן נִגַּשְׁתִּי אֵלֶיךָ / פֶּן תִּמְנַע מֵרָע רַגְלֶיךָ / אַךְ סְעִפֶּיךָ אוֹתְךָ יָאִיצוּ / רַגְלֶיךָ לָרַע יָרוּצוּ / וְתִרְצַח וְתִגְנֹב וְתִנְאָף / וְאֶל כָּל מְלֶאכֶת רְמִיָּה תִשְׁאָף / וְאָז תַּצְלִיחַ דְּרָכֶיךָ / וַאֲבָרֲכָה מְבָרֲכֶיךָ / וְאַל תִּירָא מֵרֶדֶת שַׁחַת / חֲזַק וֶאֱמַץ אַל תַּעֲרֹץ וְאַל תֵּחָת / כִּי גַּם בָּעֵת אֲשֶׁר תִּמָּצֵא בְמַחְתֶּרֶת / יָדִי תְהִי אוֹתְךָ עֹזֶרֶת / וּזְרוֹעִי לְךָ מֹשְׁלָה / לְמַלֶּטְךָ מִמַּכְשֵׁלָה / בְּלַהַט וְקֶסֶם וְלָחַשׁ / עַל מִרְמָה רַגְלְךָ תָחַשׁ / וּמִמְּלַאכְתְּךָ רִאשׁוֹנָה אַל תִּרְחָק / לְשֹׁד וּלְכָפָן תִּשְׂחָק / וַאֲנִי אֵלֵךְ אֶל חֲבוּרָתְךָ / וְשָׁמַעְתָּ בְּקוֹל אֲמָתְךָ / לֹא תוֹסִיף וְלֹא תִגְרָע / נֶפֶשׁ רָשָׁע אִוְּתָה רָע / לַעֲשֹׂת מְלַאכְתּוֹ רִאשׁוֹנָה / חָזַר לְטֻמְאָתוֹ יְשָׁנָה / וְעֶבְרָתוֹ שְׁמָרָהּ נֶצַח לִפְתֹּחַ פֶּה בְּרֶצַח / וּלְהֵיטִיב נַפְשׁוֹ לֹא זָכַר / גָּנַב וְטָבַח וּמָכַר / יִבְחַר לוֹ לֹא יוֹשִׁיעַ / כִּי הִיא הָיְתָה יוֹעַצְתּוֹ לְהַרְשִׁיעַ / וַתְּלַמְּדֵהוּ כְשָׁפִים וּקְסָמִים / עַד אֲשֶׁר מֵאִתּוֹ לֹא נַעֲלָמִים / קִלְקַל בַּחִצִּים שָׁאַל בַּתְּרָפִים / וַיֹּאמֶר לַבָּלָה נִאֻפִים / וַיְהִי הַיּוֹם וַיֵּצֵא הַשָּׂדֶה / וְאֶת הָרֹכְלִים אֹרֵב וְצֹדֶה / וַיִּשְׁמְעוּ אֵלָיו גּוֹיִם בְּשַׁחְתָּם נִתְפַּשׂ / טָמְנוּ חֵפֶשׂ מְחֻפָּשׂ / זֶה מַכִּיר עָלָיו בִּגְדּוֹ / וְזֶה אֹמֵר כִּי הָרַג דּוֹדוֹ / וַיְעַצְּמוּ עֵינָיו וַיַּכּוּהוּ בְמַקְלוֹת / וַיּוֹלִיכוּהוּ אֶל הָעֵץ לִתְלוֹת / וַיִּקְרָא לַמְכַשֵּׁפָה: מַהֲרִי נָא לְמַלְּטֵנִי / פֶּן הָרָעָה תִּדְבָּקֵנִי / כִּי הִנֵּה אֹיְבִי בִי שֹׁלֵט / וְאָנֹכִי לֹא אוּכַל לְהִמָּלֵט / וּפָרַשְׂתְּ עָלַי כְּנָפַיִךְ / עִמְדִי נָא בַחֲבָרַיִךְ וּבְרֹב כְּשָׁפַיִךְ / וַתַּעַן לוֹ: עַד הֵנָּה עֲזַרְתִּיךָ / וּבִקְסָמַי תְּמַכְתִּיךָ / וַעֲנַקְתִּיךָ גַאֲוָה / לְמַעַן סְפוֹת הָרָוָה / כֹּה פָעַלְתִּי עִמְּךָ פְּעָמִים רַבּוֹת / וּבִרְאוֹתִי כִּי אֵין לְמוֹתְךָ חַרְצֻבּוֹת / לָמָּה זֶה הֶבֶל אִיגָע / כִּי תִהְיֶה לְשַׁמָּה כְרָגַע / וְלָזֹאת עֲצָתִי סָכְלָה וִימִינִי קָצְרָה / וּבָא עָלֶיךָ רָעָה אֲשֶׁר לֹא תֵדַע שַׁחְרָהּ.

וְהַמָּשָׁל לְבִלְתִּי שְׁמֹעַ אֶל הָרְשָׁעִים / מַחֲזִיקֵי יְדֵי מְרֵעִים / וְלַמְעֹנְנִים וְלַקֹּסְמִים / וְלַמְכַשְּׁפִים וְלַחֹלְמִים / לֹא תָבֹא בַעֲדָם הָרָעָה / וְגַם נִבְנוּ עֹשֵׂי רִשְׁעָה / חַח הַזְּמָן בְּאַפֵּיהֶם / וּמִתְגוֹ בְשִׂפְתּוֹתֵיהֶם / אַנְשֵׁי דָמִים וּמִרְמָה לֹא יֶחֱצוּ יְמֵיהֶם / כִּי בְאַחַת יִכָּשְׁלוּ / וְצַדִּיקִים יָקוּמוּ אִם שֶׁבַע יִפֹּלוּ.
