משל סח: אֲרִי וְאִישׁ וּבוֹר וְנָחָשׁ

הַזְּמָן קָצָר וְהַמְּלָאכָה מְרֻבָּה / וּבַעַל הַבַּיִת דֹּחֵק לְקֻבָּה


אַרְיֵה מִסֻּבְּכוֹ עָלָה / אֶל אֵם הַדֶּרֶךְ בַּמְסִלָּה / מְצַפֶּה אֶת טַרְפּוֹ וְאֹרֵב / מִן הַבֹּקֶר וְעַד הָעֶרֶב / אוּלַי יְגֹרֵהוּ בְחֶרְמוֹ / וְהִנֵּה אִישׁ לְפִי תֻמּוֹ / וַיַּרְא הָאֲרִי עָלָיו שׁוֹאֵג / וּבְלֵב רַגָּז מְפַחֵד וְדוֹאֵג / כִּי רָץ אַחֲרָיו בְּזַעַם אַפּוֹ / כַּאֲשֶׁר יֶהְגֶּה הָאֲרִי עַל טַרְפּוֹ / וּבְמָצְאוֹ מָוֶת מַר / אֶל בֶּאֱרֹת בֶּאֱרֹת חֵמָר / שָׁם יָנוּס אֶל בּוֹר אֶגְלַיִם / וְהַבּוֹר רֵיק אֵין בּוֹ מַיִם / וּבְקִיר הַבּוֹר בַּתָּוֶךְ מוֹצֵא / יָתֵד מִזֶּה וּמִזֶּה יוֹצֵא / וַיִּשְׂמַח מְאֹד עַל הַיְתֵדוֹת / אֲשֶׁר שָׁם רַגְלָיו הָיוּ עֹמְדוֹת / כִּי חָשַׁב לִמְצֹא מִפְלָט וּמִקְלָט / מִן הָאֲרִי אֲשֶׁר עָלָיו שָׁלָט / וְכִי נִמְלַט לָעַד מֵרֶצַח / וּבַל יִסַּע יְתֵדֹתָיו לָנֶצַח / אַךְ בְּיַרְכְּתֵי בוֹר רוֹאֶה בְמַאֲרָב / נָחָשׁ שָׂרָף וְעַקְרָב / מְצַפֶּה לְבָלְעוֹ כִּשְׁאוֹל חַיִּים / וַיְהִי לוֹ פְּחָדִים שְׁנַיִם / בְּתוֹךְ צָהֳרַיִם כַּלַּיִל צִלּוֹ / כִּי לֹא מוּעָף לַאֲשֶׁר מוּצָק לוֹ / נָס מִן הָאֲרִי וּפְגָעוֹ / נָחָשׁ אֲשֶׁר פָּעַר פִּיו לְבָלְעוֹ / אִם מִדֶּרֶךְ הָאֲרִי תוֹעֶה / תַּהַרְגֵהוּ לְשׁוֹן אֶפְעֶה / מָצָא נָחָשׁ אַחֲרֵי אַרְיֵה / וַיֹּאמַר: אוֹי מִי יִחְיֶה / אִם אֶעֱלֶה יַכֵּנִי הָאֲרִי / אִם אֵרֵד פֶּה נָחָשׁ קִבְרִי / וּמֵאַיִן יָבֹא עֶזְרִי / בֵּין אֲרִי וְנָחָשׁ לִי מַחֲנָיִם / גַּם כִּי לִי נִקְיוֹן שִׁנַּיִם / מֵחֹסֶר לֶחֶם וְכָל מַאֲכָל / וְלִמְצֹא מַיִם לֹא אוּכָל / אַךְ זֹאת הִיא נֶחָמָתִי / כִּי יָתֵד נֶאֱמָן מָצָאתִי / וּמַטּוֹת עֹז לִסְמֹךְ עַל רַגְלַי / עַד אֲשֶׁר יָסוּר הָאֲרִי מֵעָלַי / וְאֶתְיַצֵּב עֲלֵיהֶם כְּגִבּוֹר / וְלֹא אֵרֵד אֶל יַרְכְּתֵי בוֹר / וּבְדַבְּרוֹ כָזֶה עַל לִבּוֹ / עַכְבָּר יוֹצֵא מֵעַל גַּבּוֹ / וַיַּבֵּט אֵלָיו וַיָּבֶן / כִּי אֵין שֵׂעָר שָׁחוֹר בּוֹ רַק לֹבֶן / וּמִצַּד הַשֵּׁנִי כְּנֶגֶד הַחוֹר / יוֹצֵא עַכְבָּר כֻּלּוֹ שָׁחוֹר / וּבַיְתֵדוֹת שִׁלְּחוּ שִׁנֵּיהֶם / אֲשֶׁר עָמְדוּ רַגְלָיו עֲלֵיהֶם / וּמְעַט מְעַט בָּהֶם אֹכְלִים / מִדֵּי יוֹם בְּיוֹם מַצָּבוֹ מְכַלִּים / וְכִרְאוֹת הָאִישׁ כִּי הִתְרוֹעֲעוּ / יָדַע כִּי הַיְתֵדוֹת יָנוּעוּ / שְׁלֹשָׁה הֵמָּה קֹרוֹתָיו / וַיִּתְאוֹנֵן עַל סִבּוֹתָיו / אוֹי מֵאֲרִי אוֹי מִנָּחָשׁ / אֲשֶׁר בְּפָנָיו רַגְלִי תָחַשׁ / אוֹי לִי מִשְּׁנֵי הָעַכְבָּרִים / אֲשֶׁר מְקוֹם מַצָּבִי נֹקְרִים / אוֹי מִצָּמָא אוֹי מֵרָעָב / וְכִי אֶהְיֶה מַהֵר לְלַעַב / וְתִשַּׁח נַפְשִׁי בְּבוֹר וּבָעֳנִי / כְּפֶשַׂע בֵּין הַמָּוֶת וּבֵינִי / וְלֹא אָשׁוּב לִרְאוֹת רֵעִי / בַּחֹשֶׁךְ רִפַּדְתִּי יְצוּעִי / וְאַתְּ בּוֹר חֹשֶׁךְ אֲשֶׁר אֶעֱמֹד בִּגְבוּלָךְ / אִם אֶרְעַב לֹא אֹמַר לָךְ / הֲמִן הַסֶּלַע אוֹצִיא לִי / לֶחֶם וּמַן מַאֲכָלִי / וּכְדַבְּרוֹ פַעֲמַיִם כָּכָה / וּתְלֻנָּתוֹ לֹא שָׁכָכָה / כַּאֲשֶׁר הוּא כִּבְתוֹךְ כַּף קֶלַע / וַיְנִקֵּהוּ דְּבַשׁ מִסֶּלַע / הַצּוּר אֲשֶׁר הַכֹּל יָכֹל / וַיֹּאמַר: פְּתַח פִּיךָ וֶאֱכֹל / וְלֹא יֶחְסַר כָּל יָמֶיךָ / דְּבַשׁ מָצָאתָ אֱכֹל דַּיֶּךָ / וַיִּתְעַדֵּן בְּטוּב דִּשְׁנוֹ / דְּבַשׁ וְחָלָב תַּחַת לְשׁוֹנוֹ / וַיִּפְשֹׁט חֲרָדוֹת אֲשֶׁר לָבַשׁ / וַיֹּאמַר: מַה מָּתוֹק מִדְּבַָשׁ / שָׁכַח מַה לְּמַעְלָה מַה לְּמַטָּה / וְכִי הוּא יֹשֵׁב בַּעֲלָטָה / וּמַה לְּפָנִים וּמַה לְּאָחוֹר / עַכְבָּר לָבָן וְשָׁחוֹר / אֲשֶׁר מִמַּצָּבוֹ יֶהְדְּפֻהוּ / וּמִתֵּבֵל יְנִדֻּהוּ / לֹא זָכַר אֲרִי מִמַּעַל לוֹ / וְלֹא נָחָשׁ מִתַּחַת רַגְלוֹ / בִּדְבַשׁ אֲשֶׁר יֹאכַל יִתְעַנַּג / כְּמוֹ בְּחִטֵּי מִנִּית וּפַנַּג / מִצַּפִּיחִת בִּדְבַשׁ לָעַט / וַיִּשְׁמַן יְשֻׁרוּן וַיִּבְעָט / וּבִהְיוֹתוֹ שָׁמֵן וְדָשֵׁן הָעַכְבָּרִים אֲשֶׁר שִׁלְּחוּ שֵׁן / בְּכָל יוֹם בִּמְקוֹם מַצָּבוֹ / הֶעֱבִירוּ מִמֶּנּוּ טוּבוֹ / פֶּתַע פִּתְאֹם נִשְׁבָּרוּ / וּמִמַּקֶּבֶת בּוֹר נִגְזָרוּ / וְנִתְּקוּ מִקִּיר בּוֹר הַיְתֵדוֹת / בְּמַרְאֶה וְלֹא בְחִידוֹת / וְנִגְדְּעָה כָּל יָתֵד וְנָפְלָה / וְהָאִישׁ אֲשֶׁר עָלֶיהָ עָלָה / בָּא בְיַרְכְּתֵי בוֹר בְּפֶה נָחָשׁ / אֲשֶׁר לֹא יִשְׁמַע לְקוֹל לָחַשׁ / לֹא יִרְאֶה עוֹד בֵּיתוֹ וּבְנֵי עַמּוֹ / וְלֹא יַכִּירֶנּוּ עוֹד מְקוֹמוֹ.


וְהַמָּשָׁל מוּסַר טוֹב לַנֶּפֶשׁ / לִלְבֹּשׁ בִּגְדֵי חֹפֶשׁ / הָאֲרִי אֲשֶׁר הֹלֵךְ וְקָרֵב / הוּא הַמָּוֶת אֲשֶׁר לָאִישׁ כְּאֹרֵב / לֹא יַהֲפֹךְ אֵלָיו עֹרֶף / וְלֹא יִשְׁכַּב עַד יֹאכַל טָרֶף / וְהַבּוֹר הוּא עוֹלָם אֲשֶׁר נִצָּב בְּקִרְבּוֹ / עַד בּוֹר יָנוּס אַל יִתְמְכוּ בוֹ / כָּל גֵּר בּוֹ בְעֵינָיו תּוֹשָׁב / וְהוּא כְמוֹ יָתֵד נֶחְשָׁב / לְהַאֲרִיךְ יָמִים לִרְאוֹת זֶרַע / יֵצֶר לֵב הָאָדָם רָע / הַנָּחָשׁ אֲשֶׁר בְּתַחְתִּיתוֹ / פֶּה שְׁאוֹל הַפְּתוּחָה לְעֻמָּתוֹ / הָאוֹמֶרֶת תָּמִיד: הַב הַב / עַתָּה תִמְאַס אֲשֶׁר תֶּאֱהָב / הָעַכְבָּר אֲשֶׁר לָבָן כֻּלּוֹ / הוּא יוֹם הַמֵּאִיר לְמוּלוֹ / וְהַשָּׁחוֹר הוּא הַלַּיְלָה / שְׁנֵיהֶם חָבְרוּ לַעֲשׂוֹת כָּלָה / בָּאָדָם עַד תֻּמּוֹ / כִּי הֹלֵךְ הָאָדָם לְבֵית עוֹלָמוֹ / הֹלְכִים סְבִיבָיו חוֹזְרִים וְסוֹבֲבִים / כִּי מִמֶּנּוּ הֵם עֲרֵבִים / לְכַלּוֹת מְקוֹם מַצָּבוֹ לְהָדְפוֹ / וְהוּא לֹא יָבִין אֶל סוֹפוֹ / וְהַדְּבַשׁ אֲשֶׁר כָּל זֶה יַשְׁכִּיחוֹ / הֲנָאַת הָעוֹלָם הָעוֹלָה עַל רוּחוֹ / וּמְשַׁעְשְׁעֵי הֲבָלָיו / אֲשֶׁר פָּרְשׂוּ פַחִים לְרַגְלָיו / צָמַח גִּילוֹ אֵידוֹ עָבַשׁ / וְטַעְמוֹ כְצַפִּיחִת בִּדְבָשׁ / וַיִּתְאַו אֶל מַטְעַמּוֹתָיו / וּכְרֶגַע שָׁרְשֵׁי שִׂמְחוֹתָיו / יָבְשׁוּ וְצַמֶּרֶת קְצִירוֹ יִשָּׁבֵר / עִם אֲבוֹתָיו שֹׁכְנֵי קָבֶר / וְאָמַר הֶחָכָם אֲשֶׁר חִבֵּר תְּבוּנוֹת כְּלֻלָאוֹת בְּקֶרֶס / הֲנָאַת הָעוֹלָם כִּדְבַשׁ מָסוּךְ בְּאֶרֶס / וַאֲנִי אֶתֵּן לְצוּרִי שֶׁבַח וּתְהִלָּה / עַל הַמִּחְיָה וְעַל הַכַּלְכָּלָה / וַאֲשֶׁר לֹא יַעֲשֶׂה כָמוֹנִי, לוֹ יוֹם מְהוּמָה וּמְבוּכָה / וְכַלְכָּלָה וְהֶָאֶבֶן בְּתוֹכָהּ / בַּאֲשֶׁר מָלְאוּ חָלָב עֲטִינֵיהֶם / וַיֵּרָגְנוּ בְאָהֳלֵיהֶם / וְלֹא שָׂמוּ בְמַדָּעָם / לְהַלֵּל צוּרָם וְרֹעָם.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


אֲרִי מָוֶת שְׁאוֹל נָחָשׁ / אֲשֶׁר לֹא יִשְׁמַע לָחַשׁ

וְלֹבֶן עַכְבָּר הַיּוֹם / וְשָׁחוֹר לַיְלָה, רָחַשׁ

לְבָבִי כִּי עֲרֵבִים הֵם / לְכַלּוֹת חַי אֲשֶׁר נִחַשׁ

יְמֵי חַיָּיו וְעוֹלָם בּוֹר / וּפִרְיוֹ רַק פְּרִי כָחַשׁ

יְתֵד בּוֹר נִשְׁבַּר וָמֵת / וְלֹא יִכְסֶה בְעוֹר תַּחַשׁ.
