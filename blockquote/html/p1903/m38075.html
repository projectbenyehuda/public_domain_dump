משל יד: חֲמוֹר, כֶּלֶב וְאָדָם

הַקִּנְאָה תָבִיא לְהִתְאַוּוֹת / דָּבָר בִּלְתִּי יֵאוֹת


כֶּלֶב קָטָן הָיָה / בֵּית הַפָּקִיד מִן הַקִּרְיָה / פִּתּוֹ יֹאכַל חֵיקוֹ יִשְׁכַּב / עִמּוֹ יֵלֵךְ הָעֵת יִרְכָּב / וּבַבַּיִת חֲמוֹר אֹרֵב / כִּי הָאָדוֹן הַכֶּלֶב יְקָרֵב / וְכִי יְשַׂחֵק בּוֹ לִרְגָעִים / כַּאֲשֶׁר הָיוּ שְׁנֵי רֵעִים / וַיֹּאמֶר הַחֲמוֹר בַּלֵּב / לֹא נוֹפֵל אָנֹכִי מִכֶּלֶב / אֵלֵךְ גַּם אֲנִי אֶל אֲדֹנִי / אוּלַי כַּכֶּלֶב יְקָרְבֵנִי / וַאֲחַבְּקֵהוּ וַאֲנַשְׁקֵהוּ / וּבִלְשׁוֹנִי אֲלַחֲכֵהוּ / וְאָשִׂים עֵינַי אֵצֶל עֵינָיו / וְאֶת פָּנַי לְמוּל פָּנָיו / וְאֶתְחַבֵּר אִתּוֹ / לְעֻמַּת מַחְבַּרְתּוֹ / וְכָל מַלְבּוּשָׁיו אֶגְאַל וְאֶדְרֹךְ / מֵחוּט וְעַד הַשְּׂרוֹךְ / וְלִמְצֹא חֵן עָלָיו אֶזְעַק / כַּאֲשֶׁר הַכֶּלֶב חָרַץ וְצָעַק / בְּזֶה הַדָּבָר אֶבָּחֵן / אִם אֲנִי כַכֶּלֶב אֶמְצָא חֵן / וְכַאֲשֶׁר בָּא אֲדֹנָיו הַבַּיְתָה / הַחֲמוֹר הִשְׁלִים לִבּוֹ עָלְתָה / וַיָרָץ מַהֵר אֵצֶל אֲדֹנָיו / וַיְלַחֲכֵהוּ סְבִיב אָזְנָיו / וַיְחַבְּקֵהוּ בִּשְׁתֵּי רַגְלָיו / וַיָּרֶם קוֹלוֹ עָלָיו / וְרֹאשׁוֹ אֶל רֹאשׁוֹ נִצְמָד / כַּאֲשֶׁר מִן הַכֶּלֶב לָמָד / הָאָדוֹן יָרֵא וְחָרֵד / כִּי הַחֲמוֹר בּוֹ עוֹלֶה וְיוֹרֵד / וּמִפֶּה אֶל פֶּה הִקִּיפוֹ / עַד כִּי מִכִּסְאוֹ הֲדָפוֹ / וּבְצַוָּארוֹ חִבְּקוֹ כִּרְבִיד / וּמַשָּׂאוֹ עָלָיו הִכְבִּיד / וַיֹּאמֶר הָאָדוֹן: מִי אִתִּי אֵלַי / הוֹצִיאוּ אֶת הַחֲמוֹר מֵעָלָי / כִּי גָדְלָה לוֹ הַחֲרָדָה / וַתִּקָּהֵל כָּל הָעֵדָה / זֶה בְּרֹמַח וְזֶה בְמַקֵּל / וְזֶה בַאֲבָנִים יְסַקֵּל / זֶה בְּשֵׁבֶט מַכֶּה הַחֲמוֹר / זֶה אוֹמֵר הַךְ אַךְ אֶת נַפְשׁוֹ שְׁמֹר / וַיִּרְדְּפוּ הַחֲמוֹר עַד אֲבוּסוֹ / הָלֹם כָּל גֵּווֹ וּכְרֵסוֹ.

וְהַמָּשָׁל לְאִישׁ הוֹלֵךְ שׁוֹבָב / רְחַב עֵינַיִם וּגְבַהּ לֵבָב / וְיִשְׂתָּרֵר בְּגַאֲוָתוֹ / כִּי לֹא יִתָּכֵן לְאִישׁ כְּמוֹתוֹ / וּבִרְצוֹתוֹ לִהְיוֹת מוֹשֵׁל / פֶּתַע פִּתְאֹם יָבֹא כֹּשֶׁל / וְיֶהֶדֳפֶנּוּ מִמַּצָּבוֹ / וְאֵין אִישׁ מַאֲזִין אֶל נִיבוֹ / וּמְשַׁל שְׁלֹמֹה לֹא נִגְרָע / כִּי לֹא תִהְיֶה אַחֲרִית לָרָע.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


לִפְנֵי שֵׁבֶר אִישׁ דְּרָכָיו מְעַקֵּשׁ / לִפְנֵי כָבוֹד שִׁפְלוּת לוֹ לְמוֹקֵשׁ

נָא זְכֹר מִי מָצָא מְלוּכָה וְנִבָּא / כַּאֲשֶׁר שָׁב מִן הָאֲתוֹנוֹת לְבַקֵּשׁ.
