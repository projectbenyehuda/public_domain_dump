ספר תחכמוני - שער שלישי

הַמּוֹדִיעַ תֹּקֶף הַשִּׁיר וּמַהֲלָלוֹ, וּשְׁמוֹת הַמְשׁוֹרְרִים אִישׁ אִישׁ עַל דִּגְלוֹ


נְאֻם הֵימָן הָאֶזְרָחִי; נָטִיתִי לִבִּי בִּימֵי נַעֲרוּתִי. וּזְמַן בַּחֲרוּתִי. לִרְדוֹף הַתַּעֲנוּגִים עִם חֲבֵרִים. עַל הָרֵי בְתָרִים. כַּאֲשֶׁר יִרְדֹּף הַקּוֹרֵא בֶהָרִים. וְקָשַׁרְתִּי עַפְעַפַּי בְּעַפְעַפֵּי שְׁחָרִים. וּכְנָפַי בְּכַנְפֵי נְשָׁרִים. לְהִתְעַלֵּס בַּכְּפָרִים. בִּמְקוֹם נְרָדִים עִם כְּפָרִים. וּמִרְעֵה עֳפָרִים. בְּעוֹד הַזְּמַנִּים לִי לַעֲבָדִים. וּבָנָיו לְשָׁרְתֵנִי יָצְאוּ גְדוּדִים. בְּטֶרֶם אֶקְרָאֵם יַעֲנוּ. וְאַחַרַי דְבָרִי לֹא יִשְׁנוּ. וּבְנֵי יוֹם בְּרֶסֶן חֶפְצִי יִמְשְׁכוּ. אֶל אֲשֶׁר יִהְיֶה שָׁמָּה הָרוּחַ לָלֶכֶת יֵלְכוּ. וְכַאֲשֶׁר הֶעֱבִיר הַגְדוּד שׁוֹפַר תְּרוּעָה. לְהָמִישׁ הַיָתֵד הַתְּקוּעָה. שִׁנַּסְתִּי מָתְנַי בְּאֵזוֹר הַנְּסִיעָה. וּפָשַׁטְתִּי סוּת הַמַּרְגֵּעָה. וְלָבַשְׁתִּי כְּסוּת הַיְגִיעָה. וְגָעַרְתִּי בְדוֹדִים וּבָחַרְתִּי נְדוֹדִים. וְשַׂמְתִּי מְגַמָּתִי לְשׁוֹטֵט אַפְסֵי עוֹלָם. עַד בָּאתִי גְבוּל שִׁנְעָר וְעֵילָם. וַיְהִי הַיּוֹם וַאֲנִי בְעִיר מֵעָרֵי בָבֶל. מִתְהַלֵּךְ כָּאֳנִיָה בְּלִי חוֹבֵל. וַיִקְרָאֵנִי לִסְעֻדָּתוֹ אִישׁ מִבְּנֵי הַנְּגִידִים. וּמַאֲצִילֵי אֶרֶץ הַנִכְבָּדִים. וְנָתַן לִי מַהְלְכִים בֵּין הָעוֹמְדִים. וַיְבִיאֵנִי לְאַרְמוֹן. בְּחֵיק הַיֳפִי אָמוֹן. וּמִמַּחֲצַב הַנֹּעַם גָּזוּר וְחָצוּב. וּמִסְּלָעָיו דְּבַשׁ וְחָלָב יָזוּב. וּבְתוֹכוֹ בָתִים מְלֵאִים כָּל טוּב וְחַיִל יָנוּב. וּסְבִיב הֶחָצֵר בִּירוֹת. וְתוֹךְ הַבִּירוֹת טִירוֹת. בְּתוֹרֵי זָהָב מְפֻתָּחִים עַל הַקִּירוֹת. בָּהֶם הַנְפָשׁוֹת תִּצְהַלְנָה. וְהָאָזְנַיִם תִּצַּלְנָה. לְרִגְשַׁת קוֹל הַכִּנּוֹר. וְשִׁפְעַת מֵי הַצִּנּוֹר, וְהַבָּתִּים בְּמִשְׁבְּצוֹת זָהָב מְכֻסִּים. וְכָל בַּיִת מְעֻטָּר בְּמִינֵי שְׁבִיסִים. וּבְמִפְרְשֵי רִקְמָה עָלָיו פְּרוּשִׂים. וְעָשָׂה לוֹ כְתֹנֶת פַּסִּים. וּבְתוֹךְ הַבָּתִּים שֻלְחָנוֹת. בְּכָל טוּב מוּכָנוֹת. דְּשֵׁנוֹת וְרַעֲנַנּוֹת. וּסְבִיבָם מַרְבָדִים. הַלֵב מְחַדִּים. וְהַיָגוֹן מְנַדִּים. לְפָנֵינוּ אֲשִׁישׁוּת וְכַדִּים. בָּם לְכָל עַיִן מַחֲמַדִּים. וְהַיַּיִן כְּלַפִּידִים תּוֹךְ הַכַּדִּים. וְהַגְּבִיעִים. לְכוֹכְבֵי הַיַּיִן רְקִיעִים. וְהַמִּזְרָקִים. כְּאִלּוּ מִן הָאוֹר יְצוּקִים. וּבִזְהַב הַחֵשֶׁק מְחֻשָׁקִים.

וָאַעַן וָאֹמַר:


תְּמַהּ אֵיך נִצְמְדוּ גַחַל וּבָרָד

וּמוֹקְדֵי אֵשׁ יְפוּצוּן מֵאֲפִיקִים.

בִּחוּטֵי שַׁרְשְׁרוֹת זָהָב מְרֻקָּע

בְּכִכַּר מַעֲבֵה הַלֵב יְצוּקִים.

עֲסִיסִים הֵם בְּכוֹסוֹת דַּר מְכֻסִּים

וְיָפִיצוּ מְאוֹרִים עַד שְׁחָקִים.

יְרַוֻּךָ בְטַל אוֹרוֹת וְאוּלָם

בְּפִיךָ נוֹזְלֵי לַהַב מְרִיקִים.

כִּאִלוּ הַגְּבִיעִים הֵם רְקִיעִים

וְהַיַיִן בְּתוֹכָם כַּבְּרָקִים.

וַיוּשָׂם לְפָנֵינוּ מִינֵי מַעֲדַנִים. וּמִשׁתֵּה שְׁמָנִים.

וּמַטְעַמִּים דְּשֵׁנִים. וּסְבִיבֵנוּ מְשָׁרְתִים חֲמוּדִים. כִּמְנוֹרוֹת לְפָנֵינוּ עוֹמְדִים. לְהָבִיא כָל מַאֲכָל נָעִים וְנֶאֱהָב. וְכָל אֶחָד מֵהֶם מְמַהֵר וְאִתּוֹ כְלִי מֻצְהָב. הַמַּשְׁקֶה בְּיָדוֹ כּוֹס זָהָב. וְיָּבֹא גַם הַנִּצָּב אַחַר הַלַּהַב. וּבְיָדָיו גְּדָיִים צְלוּיִים. וּזְרוֹעוֹ יְקַבֵּץ טְלָאִים. רַכִּים וְנָאִים. אֲשֶׁר חֹם הָאֵשׁ הֶאֱדִיבָם. עָד הֱשִיבָם. צַחִים וְזַכִּים. לַחִים וְרַכִּים. גְּאוֹן הָרָעָב מְדַכִּים. גַּאֲוָתוֹ בְּשׁוֹט לָשׁוֹן מַכִּים. וְהָיָה בְתוֹך הַקְּרוּאִים אִישׁ זָקֵן. נוֹדֵד כְּצִיפּוֹר מִקֵּן. וְהוּא מֵסֵב אֶל הַשֻׁלְחָן הֶעָרוּך. כְּנָחָשׁ כָּרוּך. וּזְרוֹעוֹ אָרֹך. לְהַקְרִיב כָּל מַאֲכָל רָחוֹק. וְלִטְחוֹן אוֹתוֹ וְלִשׂחוֹק. וְלוֹ חֵך. אֶת כָּל סְבִיבוֹתֵינוּ לוֹחֵך. וְלָשׁוֹן. יִגְרֹף הַמַּאֲכָל כְּנַחַל קִישׁוֹן. וְלוֹ פֶה. לֹא יְמַלְּאֵהוּ הַמַשְׁקֶה וְהָאוֹפֶה. תָּמִיד יִתֵּן בַּכּוֹס עֵינוֹ. וּבַלֶּחֶם שִׁנוֹ. וַיִּמָּצֵא הַגָּבִיעַ סוֹבֵב עַל יְמִינוֹ. וְהַבָּשָׂר. אִלוּ בְּכַבְלֵי בַרְזֶל נֶאֱסָר. יִמְשְׁכֵהוּ אֵלָיו בִּלְשׁוֹנוֹ. וְלַחָפְשִׁי יִשַׁלְּחֵנּוּ תַּחַת שִׁנּוֹ. וְהוּא לוֹקֵחַ מִכָּל לֶחֶם חֶלְבּוֹ. וּמִכָּל מַאֲכָל טוּבוֹ. וְאֶל פִּיו דּוֹחֵף וְהוֹדֵף. כָּל נֵתַח טוֹב וְכָתֵף. וְהַבָּשָׂר הַשָׁמֵן יְמְשְׁכֵהוּ בִּשְתֵּי יָדַיִם. וְיִלְקְטוּהוּ הַשְּׂפָתַיִם. וְטָחֲנוּ בָרֵחָיִם וּזְרוֹעוֹתָיו הַקְּעָרָה יִתְמֹכוּ. וְכָל אֲשֶׁר בָּהּ אֵלָיו יִמְשֹׁכוּ. וּמִפִּיו לַפִּידִים יַהֲלוֹכוּ. וְהַנְּתָחִים. הַמְשֻׁבָּחִים. אִם יֹאחֲזֵם בְּכַפּוֹ. יְשַׁלֵחַ בָּם חֲרוֹן אַפּוֹ. וְכָל חֲתִיכָה שְׁמֵנָה. אֲשֶׁר מֵרָחוֹק יִרְאֶנָּה. בַּעֲבוֹתוֹת אַהֲבָה יִמְשְׁכֶנָּה. בְּעוֹדָהּ בְּכַפּוֹ יִבְלָעֶנָּה. וְהַטְּלָאִים. מֵחֶרֶב שִׁנָּיו יְרֵאִים. וְהַכְּבָשִׂים. מִפַּחְדּוֹ נָסִים. וְהַדָּגִים. מֵאֵימָתוֹ דוֹאֲגִים. וְהַבָּשָׂר עוֹדֶנוֹ בֵּין שִׁנָּיו וְהוּא צוֹעֵק לְנֶגֶד הַיְקוּם. נְתָנַנִי ה' בִּידֵי לֹא אוּכַל קוּם. כְּאִלוּ הוּא נוֹשֶׁה עַל הַצַּלַחַת. וּבַעַל חוֹב עַל הַקַּלַחַת. וַיְהִי בְּבוֹאָם לַשֻׁלְחָן וְהַנוֹשֶׁה בָא לָקַחַת. עַד מָחָה הַצַּלַחַת. וּמָצָה הַצַפַּחַת. מִמַּעַל וּמִתַּחַת. וְלֹא נִמְלְאָה הָאַמְתַּחַת. הוּא מְכַבֵּד הַיַּיִן כְּאָבִיו. וְנוֹשֵׁק הַכּוֹס כְּאָחִיו. וְיָּשֶׂם פִּיו עַל פִּיו. וְלֹא שָׁקַט. עַד כָּל הַנִּמְצָא לָקָט. וְהַמַּאֲכָל. אָכָל. וְהַמָּרָק. בְּפִיהוּ זָרַק. וְְלֹא הִשְׁאִיר לְפָנֵינוּ כָּל מְאוּמָה. כְּגַן עֶדֶן הָאָרֶץ לְפָנָיו וְאַחֲרָיו מִדְבַּר שְׁמָמָה. עַד כִּמְעַט לְרֹע פְּעָלָיו. עָלְתָה חֲמָתֵנוּ עָלָיו. וְלוּלֵי אֲשֶׁר מְנָעַתְנוּ הָעֲנָוָה מֵעַל הַשֻׁלְחָן הִשְׁלַכְנוּהוּ. וּמִן הַבַּיִת גֵּרַשְׁנוּהוּ. וְאָמַרְנוּ זֶה לָזֶה. הַרְאִיתֶם אִישׁ כָּזֶה. אֲשֶׁר שְׂפָתוֹ. חֲנִיתוֹ. וּלְשׁוֹנוֹ. קִלְשׁוֹנוֹ. וְשִׁנָּיו. אֲזֵנָיו. וְהוּא הִכִּיר מִמַּרְאֶה עֵנֵינוּ. צְפוּנֵי רַעְיוֹנֵינוּ. וְשָׁמַע קְצָת הֶגְיוֹנֵינוּ. אַךְ בְּמֶתֶק הָאֲכִילָה קִבֵּל הַנֵּזֶק וְהַקְּטוֹב. וְלֹא דִבֶּר לָנוּ מֵרַע וְעַד טוֹב. וְאֲנַחְנוּ בְּכֹח הָאֲכִילָה הַהִיא הָיִינוּ מְדַבְּרִים בְּעִנְיַן מְלֶאכֶת הֲשִׁירִים. וְהָיִינוּ חוֹקְרִים עַל הַמְשׁוֹרְרִים. אֲשֶׁר הָיוּ בִּלְשׁוֹן הָעִבְרִים. וְהָאִישׁ שׁוֹמֵעַ וּמַחֲרִישׁ. אֲבָל בַּקְעָרוֹת כַּצִּפְעוֹנִי יַפְרִישׁ. וְהָיָה מַּבִּיט אֵלֵינוּ בּעַיִן לַעַג וּבִזָּיוֹן. וּמִתְגָּאֶה עָלֵינוּ גַּאֲוַת הָעֶלְיוֹן עַל הָאֶבְיוֹן. אוֹ כְּנוֹשֶׁה. עַל אִישׁ רָשׁ וְנִקְשֶׁה. וְכַאֲשֶׁר הִכָּה הַקְּעָרוֹת בּאַבְנֵי נֶגֶף וְכִלָּיוֹן. וְהָיוּ כֻלָּם לְשִׁנָּיו נִקָּיוֹן. וְהִשְׁקִיט בַּאֲכִילָתוֹ כִלְיוֹן הָעֵינַיִם וּמַחֲלוֹן הָרַעְיוֹן. וְיָּמוּתוּ גַם שְׁנֵיהֶם מַחְלוֹן וְכִילְיוֹן. אָז נָשָׂא עֵינָיו אֵלֵינוּ וְהִשְׁקִיף מִשְׁמֵי הַגַאֲוָה עָלֵינוּ.

וַיַּעַן אֶל הַיוֹשֵׁב מִשְׂמֹאלוֹ וַיֹּאמֶר לוֹ אֵלֶּה הַצֹּאן הָאוֹבְדִים וְהַתּוֹעִים. הַגִידָה לִי אֵיפֹה הֵם רוֹעִים? וְעַל מָה הֵם מְדַבְּרִים. וְאֵי זֶה עִנְיָן הֵם מְסַפְּרִים?

אָמַרְתִּי לוֹ: הֵם מְדַבְּרִים עַל גִבּוֹרֵי הַשִׁיר אֲשֶׁר הַיוּ בִּסְפָרָד. וְכֹחַ מְלִיצָתָם בִּמְחֻבָּר וּמְפֹרָד. וְכֻּלָּם הִסְכִּימוּ וְגָזְרוּ. וּפֶה אֶחָד אָמְרוּ. כִּי אֵין כְּשִׁירֵי שְׁלֹמֹה חֲזָקִים. וְלֹא מִשִּירֵי הַנָּגִיד עֲמֻּקִים. וְלֹא כְּשִׁירֵי בֶן חַסְדַּאי עֲרֵבִים וְחֲלָקִים. וְלֹא כְּשִׁירֵי בֶן תַּקְנָה בְּכוּר שֶׂכֶל יְצוּקִים. וְלֹא כְשִׁירֵי בֶן חַלְפוֹן לְטוֹב וְרַע נֶחֱלָקִים. וְלֹא כְשִׁירֵי יִצְחָק בֶּן גִּיאַת מֵהָבִין רְחוֹקִים. וְלֹא כְשִׁירי ר' יוֹסֵף בֶּן שַׁטְנַאש בְּדוּקִים. וְלֹא כְשִׁירֵי ר' מֹשֶׁה הַכֹּהֵן עַתִּיקִים. וְלֹא כִשִׁירֵי בֶן בָּרוּךְ בְּפִי הַלְבָבוֹת נְשׁוּקִים. וְלֹא כְשִׁירֵי רַבִּי אֲדוֹנִים בְּיַד הַחֵן חֲבוּקִים.

וְלֹא כְּשִׁירֵי רַבִּי יְהוּדָה בֶּן גִיאַת בַּלֵּב חֲקוּקִים. וְלֹא כְּשִׁירֵי רַבִּי מֹשֶה בֶּן עֶזְרָא מְזֻקָּקִים. וְלֹא כְשִׁירֵי רַבִּי מֹשֶה בֶּן שַׁשָּׁת לְנֶפֶשׁ דְּבוּקִים. וְלֹא כְּשִׁירֵי בֶן תַּבַּאן חֲשׁוּקִים. וְלֹא כְשִׁירֵי רַבִּי דָוִד בֶּן בַּקּוּדָה דוֹלְקִים. וְלֹא כְשִׁירֵי בֶן רְאוּבֵן מִצְּבָא רוֹם עֲשׁוּקִים. וְלֹא כְשִׁירֵי רַבִּי אַבְרָהָם אַלְחֲרִיזִי מוּצָקִים. וְלֹא כְשִׁירֵי רַבִּי יוֹסֵף בֶּן סַהַל מִזַּן אֶל זַן מְפִיקִים. וְלֹא כְשִׁירֵי רַבִּי אַבְרָהָם בֶּן עֶזְרָא בְּהִירִים בַּשְּׁחָקִים. וְלֹא כְשִׁירֵי בֶּן אַלמֻעַלָּם בִּזְהַב הַחֵשֶׁק מְחֻשָּׁקִים. וְלֹא כְשִׁירֵי בֶּן בַּרְזֶל לְכֹל צַוָּאר עֲנָקִים. וְלֹא כְשִׁירֵי בֶן צַדִיק נֶאֱמָנִים צַדִיקִים. וְלֹא כְשִׁירֵי ר' יְהוּדָה הַלֵּוִי מִדְּבַשׁ מְתוּקִים.

וְבְשָׁמְעוֹ דְבָרַי הֶחֱרִישׁ כְּפִי רֶגַע לְמִילָּי. אַחֲרֵי כֵן נָשָׂא עֵינָיו אֵלָי. וַיֹאמֶר לִי: הֲלֹא אַתֶּם שְׂנֵאתֶם אוֹתִי וּבַעֲבוּר אֲכִילָתִי. בְּזִיתֶם גְדֻלָּתִי. וְכִמְעַט בְּרֹעַ מוּסַרְכֶם יָד בִּי שְׁלַחְתֶּם. וְחֶרֶב הָרֶשַׁע עָלַי פְּתַחְתֶּם. וְלוּלֵא הָעֲנָוָה אֲשֶׁר תִּכְבְּשֵׁנִי. וּבְרֶסֶן הַמוּסָר תִּקְשְׁרֵנִי. עָזַבְתִּי אֶתְכֶם בְּיָם הַסִּכְלוּת טוֹבְעִים. וּבְטִיט הַפְּתַיוּת שׁוֹקְעִים. לְמַעַן תִּקְצְרוּ פְּעֻלַּתְכֶם. וְתֹאכְלוּ פְּרִי חַטַּאתְכֶם. אַךְ לֹא אֶעֱשֶׂנָּה. וְחַטַּאתְכֶם מִלִבִּי אַעֲבִירֶנָּה. וְהָאֵל יָשִׁיב גְּמוּלָהּ לְעוֹשֶׂנָּה. וְהַמְחַטֵּא אוֹתָהּ יֹאכְלֶנָּה.

אָמַר הַמַּגִיד: וְכִשְׁמֹע הָעָם נֹעַם דְּבָרָיו. חָרְדוּ לְרַעַם אֲמָרָיו. וְהֵבִינוּ מִבְּרָקָיו מְטָרָיו. וְהִכִּירוּ מִשְׁחָרָיו מְאוֹרָיו. וְיָדְעוּ כִי אַחֲרֵי בְרָקָיו נְשִׂיאִים. בְּגִשְׁמֵי שֵׂכֶל מְלֵאִים. וְעַל אֲשֶׁר עָשׁוּ בוֹשׁוּ מִפָּנָיו וְנִכְלָמוּ. וּמֵעֲנוֹת נֶאֱלָמוּ. וְיָד לְפֶה שָׂמוּ.

וַיַּעַן וַיֹּאמֶר לָהֶם: אַנְשֵׁי לֵבָב שִׁמְעוּ לִי. וּפַעֲרוּ פִּיכֶם לְמַלְקוֹשׁ שִׂכְלִי. כִּי אֲנִי אַתִּיר לָכֶם כָּל חָתוּם. וְאֶפְתַּח לָכֶם כָּל סָתוּם. וְהַמְשׁוֹרְרִים אֲשֶׁר זְכַרְתֶּם וְעַל סוֹדָם חֲקַרְתֶּם. נִקְרֹא נִקְרֵיתִי בַמִּלְחָמָה הָעֲרוּכָה וְאָנֹכִי הַבָּא מִן הַמַּעֲרָכָה. וְלִבִּי מְגִילַּת עִנְיְנֵיהֶם. וַאֲנִי סֶפֶר הַזִכָּרוֹן לְשִׁירֵיהֶם. וְעַתָּה הַטּוּ אֵלַי אָזְנְכֶם. וְשִׁמְעוּ וּתְחִי נַפְשְׁכֶם. וַיַאֲזִינוּ כָל הָעָם אֵלָיו. וְהִקְשִׁיבוּ לְמִלָּיו.

וַיַּעַן וַיֹּאמֶר: דְּעוּ כִּי הַשִּׁיר הַמְעֻלֶּה. אֲשֶׁר הוּא בִּפְנִינִים מְמֻלֶּה. וּבְכֶתֶם אוֹפִיר לֹא יְסֻלֶּה. מִסְפָרַד שָׂרַד. וְעַל כָּל אַפְסֵי הָעוֹלָם נִפְרַד. כִּי שִׁירוֹת בְּנֵי סְפָרַד חֲזָקוֹת וְעֲרֵבוֹת. כְּאִלוּ מִלַּהַב אֵשׁ חֲצוּבוֹת. וּמְשׁוֹרְרֵיהֶם כִּזְכָרִים. וְכָל מְשׁוֹרְרֵי עוֹלָם כִּנְקֵבוֹת. אֲבָל בִּימֵי קֶדֶם הָיוּ בָהֶם מְשׁוֹרְרִים רַבִּים. צָמְחוּ כַעֲרָבִים. וְשׁוּרוֹתָם סְתוּמוֹת וְלֹא נִקְּבוּ בְשֵׁמוֹת. כְּגֹון מְנַחֵם בֶּן סָרוּק וְדוֹנַשׁ בֶּן לִבְרַט וּשְׁמוּאֵל וְאָבוּן וַאֲחֵרִים רַבִּים מִלְבַד אֵלֶּה אֲשֶׁר אֵין זִכְרוֹן לְשִׁירֵיהֶם. מִפְּנֵי חֹלֶשׁ עִנְיְנֵיהֶם. עַד בּוֹא הַנָּגִיד הַגָּדוֹל רַבִּי שְׁמוּאֵל הַלֵּוִי זַצַ"ל וְהָיוּ שִׁירָיו נְגִידִים וְרָאשִׁים. וְעִנְיָנָיו חֲזָקִים וַחֲדָשִׁים. אַךְ רֹב שִׁירָיו עֲמֻקִּים וְקָשִׁים. וּצְרִיכִים לְפֵרוּשִׁים. וּבְדוֹרוֹ הָיָה רַבִּי יִצְחָק בֶּן חַלְפוֹן וּקְצָת שִׁירָיו חֲזָקִים. וּקְצָתָם צְנוּמִים דַּקִים כִּי בְקָקוּם בּוֹקְקִים. וְרַבִּי יוֹסֵף בֶּן חַסְדַּאי הוּא אֲבִי הַשִׁירָה הַיְּתוֹמָה. אֲשֶׁר תְּחִילָּתָהּ „הֲלִצְבִי הֵן גְּבוּרַת אוֹן וְעָצְמָה ”. אֲשֶׁר אַחַת הִיא לְאִמָּה. וְאֵין כָּמוֹהָ בֵּין בְּנֵי עַמָּהּ. וְשִׁירֵי רַ' מֹשֶה בֶּן תַּקְנָה בִּידֵי הַשֵּׂכֶל מְתֻקָּנִים. נְטוּעִים נִטְעֵי נַעֲמָנִים. וְשִׁירֵי רַבִּי שְׁלֹמֹה הַקָּטָן . לְפָנָיו כָּל גָּדוֹל יִקְטָן. וְלֹא קָם כָּמוֹהוּ. מִימוֹת עֵבֶר אֲבִי יָקְטָן. וְכָל מְשׁוֹרְרֵי דוֹרוֹ הָיוּ לְפָנָיו רִיק וְסֶלֶף. וְהַקָּטֹן יִהְיֶה לְאֶלֶף. הוּא לְבַדּוֹ עָלָה בְּמַעֲלַת הַשִּׁיר הָעֶלְיוֹנָה. וְהַמְּלִיצָה יְלָדַתְהוּ עַל בִּרְכֵּי הַתּבוּנָה. וַתִּקְשׁוֹר עַל יָדוֹ שָׁנִי לֵאמֹר: זֶה יָצָא רִאשׁוֹנָה. כִּי כָל הַמְשׁוֹרְרִים אֲשֶׁר הָיוּ לְפָנָיו שִׁירָם לְנֶגְדוֹ רוּחַ וָתֹהוּ. וְאַחֲרָיו לֹא קָם כָּמוֹהוּ. וְכָל הַבָּאִים אַחֲרָיו לָמְדוּ מִטַּעְמוֹ. וְקִבְּלוּ רוּחַ הַשִׁיר מְעִימוֹ. וְהָאֵל מְשָׁחוֹ לְמֶלֶך עָל שִירֵי עַמּוֹ. שִׁיר הַשִׁירִים אֲשֶׁר לִשְׁלֹמֹה. וְדַי לַמְשׁוֹרְרִים הַחֲזָקִים. לְהָבִין שִׁירָיו הָעֲמֻקִּים. כִּי מְלִיצָתוֹ נִשְׂגְּבָה מִדַּעַת כֻּלָּנוּ. וּמִי יַעֲלֶה הַשָׁמַיְמָה וְיִקָחֶהָ לָנוּ. גַּם בְּמַעֲמַד הַצוֹמוֹת. הֶרְאָה נוֹרָאוֹת עֲצוּמוֹת. וְחִבֵּר סֵדֶר לְיוֹם הַכִּפּוּרִים. כָּל עִנְיָנָיו נְרָדִים עִם כְּפָרִים. מְעֻלָּפִים סַפִּירִים. וְשִירֵי אֲדוֹנִים. לְקַרְשֵי הַשֵּׂכֶל אֲדָנִים. וְשִירֵי הָרַב רַבִּי יִצְחָק. עַל מֵצַח הַיְקָר הוּחַק. וּבְסִפְרוֹ אֲשֶׁר חִבֵּר לְיוֹם הַכִּיפּוּרִים הֶרְאָה נוֹרָאוֹת. הָיוּ לוֹ לְאוֹת. כִּי שִׁירוֹ צָלְחָה עָלָיו רוּחַ נְבוּאוֹת. וְשִירֵי הַגָּאוֹן רָבִּי יוֹסֵף בֶּן שַׁטְנַאשׁ נְעִימִים וִיקָרִים. וְהוּא הָרִאשׁוֹן אֲשֶׁר חִבֵּר בִּסְפָרַד מַעֲמָד לְיוֹם הַכֵּפּוּרִים. וְשִׁירֵי הָרַב בָּרוּך. מִקוֹרָם בָּרוּךְ. וְשֻׁלְחָנָם לְכָל בָּא עָרוּךְ. וְשִׁירֵי רָבִּי יְהוּדָה בֶן גִיאַת יְהוּדוּהוּ אֶחָיו. וְיָעִידוּ שְׁבָחָיו עָל שְׁבָחָיו. וְרַ' מֹשֶה בֶּן עֶזְרָא מוֹשֶה הַפְּנִינִים. מִמְצוּלוֹת הָרַעְיוֹנִים. וְשִׁירוֹ אֲשֶׁר חִבֵּר לְלֵילוֹת הַתַּחֲנוּנִים. דּוֹבֵב שִׂפְתֵי יְשֵׁנִים. גַם הוּא עָשָׂה סֵדֶר לְיוֹם הַכִּפּוּרִים. מִלָּיו מִכְּלֵי פָּז יְקָרִים. וְרַבִּי יוֹסֵף בֶּן שַׁשָּׁת. הִצִיב שִׁירוֹ לְמַטָּרָה וַיְהִי רֹבֶה קַשָּׁת. וְרַבִּי לֵוִי וְרַבִּי יַעֲקֹב בֶּן תַּבַּאן. יָדוּשּׁוּ הַמְּלִיצוֹת כְּחִידוּש מַתְבֵּן. וְרַבִּי דָוִד בֶּן בַּקּוּדָה. לְרַגְלָיו הַמְּלִיצוֹת יִקֹּדוּ. וְעַל דַּלְתֵי שְׂפָתָיו יִשְׁקֹדוּ. וְשִׁירֵי רַבִּי יִצְחָק בֶּן רְאוּבֵן. סוֹדָם מִי חָכָם וְיָבֵן. כִּי הִפְלִיא לַעֲשׂוֹת מְלִיצוֹת נָאווֹת. וְהֵבִיא בִּמְלֹא חֲרוּזָיו זִכְרוֹן כָּל הַמִּצְוֹת. וַיְשִׂימֵם בַּפְּסוּקִים דְּבֵקִים. כְּאִלוּ הֵם בְּרוּחַ הַקֹדֶשׁ חֲקוּקִים. וְשִׁירֵי רַבִּי אַבְרָהָם בֶּן חֲרִיזִי בִּידֵי שֵׂכֶל אֲחוּזִים. וְנָאווּ לְחָיָיו בַּתּוֹרִים צַוָּארוֹ בַּחֲרוּזִים. וְרַבִּי יוֹסֵף בֶּן סַהַל לְמַחֲנֵה הַשִׁיר מְאַסֵּף. וְהַמְּלִיצוֹת יֻלְּדוּ עַל בִּרְכֵּי יוֹסֵף. וּקְרוֹבוֹ רַבִּי שְׁלֹמֹה בֶּן צַקְבַּל. גַּם הוּא לָמַד מְלֶאכֶת הַשִּׁיר וְקִבֵּל. וְהוּא עָשָׂה הַמַּחְבֶּרֶת הַחֲמוּדָה. אֲשֶׁר תְּחִילָּתָהּ „ נְאֻם אָשֶׁר בֶּן יְהוּדָה“. וְשִׁירֵי הֶחָכָם רַבִּי אַבְרָהָם בֶּן עֶזְרָא עֶזְרָה בַצָּרוֹת. וְגִשִׁמֵי נְדָבוֹת בִּזְמַן בַּצָּרוֹת. וְכָל פִּיוּטָיו נוֹרָאִים. וְעִנְיְנֵיהֶם נִפְלָאִים. בָּהֶם לֹא רָאוּ הָרוֹאִים. גַם הוּא חִבֵּר סֵדֶר לְיֹום הַכִּפּוּרִים עִנְיָנָיו בְּכוּר שֶׂכֶל יְצוּקִים. וְעַל הַלְּבָבוֹת חֲקוּקִים. וְיִצְחָק בְּנוֹ גָּם הוּא מִמְּקוֹר הַשִׁיר שָׁאָב. וְעַל שִירֵי הַבֵּן מִזִּיו הָאָב. אַךְ בְּבוֹאוֹ לְאַרְצוֹת מִזְרַח. כְּבוֹד ה' עָלָיו לֹא זָרָח. וְהֵסִיר מֵעָלָיו מְעִילֵי הַדָּת הַיְקָרִים. וּפָשַׁט אֶת בְּגָדָיו וְלָבַשׁ בְּגָדִים אֲחֵרִים. וְרַבִּי יְהוּדָה בֶּן עַבַּס גַּם הוּא בָא מְאַרְצוֹת מַעֲרָב לְמִזְרָח. וּבְדֶרֶךְ הַשִּׁיר אָרָח. וְלוֹ שִׁירִים חֲמוּדִים וְעֲרֵבִים. וּקְצָתָם כְּאֶפֶס נֶחְשָׁבִים. לֹא עֲרֵבִים וְלֹא טוֹבִים. גַּם הוֹלִיד בֶּן פָּרִיץ. הִפְלִיא נַבְלוּת וְהִמְרִיץ. וְשִׁירֵי רַבִּי שְׁלֹמֹה בֶּן אַלְמֻעַלָּם. נִיבוֹ יְרַנֵן לְשׁוֹן נֶאֱלָם. וְאוֹרוֹ מִן הַעִוְרִים לֹא נֶעְלָם. וְשִׁירֵי רַבִּי יוֹסֵף בֶּן בַּרְזֶל כַּבַּרְזֶל חֲזָקִים. וְכַנֹּפֶת חֲלָקִים. וּמֵאֲצִילֵי סְפָרַד הָיָה הַשַׂר הַגָּדוֹל רַבִּי יִצְחָק בֶּן קְרִשְׁפִּין בַּעַל סֶפֶר הַמּוּסָר וְהוּא סֵפֶר אֲשֶׁר כָּל שִׁירָיו חֲמוּדִים. בְּיַד הַשֵּׂכֶל צְמוּדִים. וְהָיוּ לַאֲחָדִים. וְרַבִּי יוֹסֵף בֶּן צַדִיק בְּרוּחַ שִׁירוֹ יִקְרַע יָם הַמַדָּע קְרָעִים. מְפָרֵק הָרִים וּמְשַׁבֵּר סְלָעִים. וּבְתֹקֶף מִלָּיו הַרֲרֵי עַד יַעְתִּיק. וּמֵי מָרָה יַמְתִּיק. וּבְעַת יִשַׁחוּ כָל בְּנוֹת הַשִׁיר תְּרוֹמַמְנָה קַרְנוֹת צַדִּיק. וְשִׁירֵי הַלֵּוִי רַבִּי יְהוּדָה. לִוְיַת חֵן לְרֹאש הַתְּעוּדָה. וְעַל צַוָּארָה טוּר אֹדֶם פִּטְדָה. הוּא לְבֵית הַשִׁיר הָעַמּוּד הַיְמָנִי. יוֹשֵׁב בְּשֶׁבֶת תַּחְכְּמוֹנִי. הוּא עֲדִינוֹ הָעֶצְנִי והוּא עוֹרֵר אֶת חֲנִיתוֹ עַל הַנְּפִילִים. וְעָזַב גִּבּוֹרֵי הַשִׁיר חֲלָלִים. וְכָל שִׁירָיו לֵב הַחֲכָמִים יְחִתּוּן. וְכִמְעַט לְפָנָיו סָף אָסָף וְאָזְלַת יַד יְדּותוּן. וַיְהִי שִׁיר בְּנֵי קֹרַח. לָטֹרַח. הוּא בָא בְאוֹצַר הַשִׁיר וְשָׁלַל בֵּית נְכֹתֹה. וְלָקַח כָּל כְּלֵי חֶמְדָּתוֹ. וְיָצָא וְסָגַר הַשַּׁעַר אַחֲרֵי צֵאתוֹ. וְכָֹל הַיוֹצְאֶים בְּעִקְבוֹתָיו. לִלְמוֹד מְלֶאכֶת שִׁירוֹתָיו. לֹא הִשִׁיגוֹ אֲבַק מַרְכְּבוֹתָיו. וְכָל הַמְשׁוֹרְרִים יִשְׂאוּ מִדַבְּרוֹתָיו. וְיִשְּׁקוּ מַרְגְּלוֹתָיו. כִּי בִמְלֶאכֶת הַפִּיוּט. לְשוֹנוֹ חֵץ בָּרוּר וְשָׁחוּט. וּבְשִׁירֵי תְהִילּוֹתָיו אֵין מִי יוֹעִידֶנּוּ. וּבְשִׁירֵי תְפִילוֹתָיו יִמְשֹׁךְ כָּל לֵב וְיִכְבְּשֶׁנוּ. וּבְשִׁירֵי חֵשֶׁק נִיבוֹ כְּשִׁכְבַת הַטַּל וְגֶחָלִים בָּעֲרוּ מִמֶּנּוּ. וּבְקִינוֹתָיו יַזִּיל עֲנַן הַבְּכִי וְיִבְקָעֶנּוּ. וְאִם יְחַבֵּר אִגֶּרֶת אוֹ מְגִילָּה. תִּמְצָא כָּל מְלִיָצה יָפָה בָּה כְּלוּלָה. כְּאִלוּ מִכּוֹכְבֵי רוֹם גְּלוּלָה. אוֹ מֵרוּם הַקֹּדֶשׁ אֲצוּלָה. וּבְאֹהֶל הַשִּׁיר נִפְתְּחוּ לוֹ שַׁעֲרֵי גְבוֹהִים. כִּי שָׁם נִגְלוּ אֵלָיו הָאֱלֹהִים. וּבְרוּח פִּיו גַּלֵּי חָכְמוֹת יַשְׁבִּיחַ. וְכָאֲרִי יָרִיעַ אַף יַצְרִיחַ. וְכָל אֲשֶׁר יַעֲשֶׂה יַצְלִיחַ.

וַיִּשָּׂא מְשָׁלוֹ וַיֹאמַר:

לְלֵוִי כָל יְקָר נִלְוָה וְנִקְוָה

וְהוּא לִוְיַת פְּאֵר עַל רֹאשׁ מְשִׁיחָיו.

וְיוֹם זָרַח בְּחוּג מַעְרָב מְאוֹרוֹ

בְּמִזְרָח מָלְאוּ כָל צַד רְקָחָיו.

וּבִקְרָב שִׁיר שְׁבָחָיו שָׁם שְׁלָחָיו

לְהִלָּחֵם וְשִׂיחָיו הֵם רְמָחָיו.

וְרַבִּים רוֹצְצוּ אַחֲרָיו וְנִלְאוּ

מְצֹא דַרְכּוֹ וְלֹא יָדְעוּ אֳרָחָיו.

וְהוּא בָא בַחֲדַר הַשִׁיר לְבַדּוֹ

עֲדֵי צֵאתוֹ וְסָגַר פְּתָחָיו.

וְהָאֵל עַל שְׁעָרָיו שָׂם בְּרִיחִים

וְהוּא שִׁבַּר בְּיַד שִׂכְלוֹ בְרִיחָיו.

וְהִצְמִיחַ בְּטַל מִלָּיו צְמָחָיו

וְהִפְרִיחַ בְּמֵי שִׂכְלוֹ פְּרָחָיו.

וְצָרוּ הַלְבָבוֹת מֵהֲדָרוֹ

כְּמוֹ קָצְרוּ לְשׁוֹנוֹת מִשְׁבָחָיו.

וְיוֹם עָרְכוּ חַכָמִים מִלְחֲמוֹת שִׁיר

יְהוּדָה הוּא אֲשֶׁר גָבַר בְּאֶחָיו.

אָמַר הַמַּגִּיד: וְכַאֲשֶׁר כִּלָה דְבָרָיו הֵשִׁיב כָּל לָשׁוֹן

נֶאֱלֶמֶת. וְכָל חָכְמָה נֶעְלֶמֶת. וְכָל נֶפֶשׁ נִכְלֶמֶת. וַיֵּדְעוּ

כִּי כָל מִשְׁפָּטָיו צֶדֶק וֶאֱמֶת. וַיִּשְׁאָלוּהוּ הָעָם עַל שְׁמוֹ.

וְאַיֵּה נְוֵה הֲדוֹמוֹ.

וַיַּעַן וַיֹּאמַר:

נָוְתִי הִיא בְּאַלוֹן צַעֲנַנִּים

וְלִי נֶפֶשׁ עֲלֵי רֹאֹש עָשׁ מִכוֹנָהּ.

וְתִשְתּוֹחַח בְּעַנְוָתָהּ לְעָפָר

וְיִשַׁח רֹאשׁ תְּלִי לִפְנֵי גָאוֹנָהּ.

וְאִם יֵשׁ לַמְּלִיצָה רֹאשׁ אֲנִי צִיץ

וְאִם לָשׁוֹן אֲנִי מַעְנֵה לְשׁוֹנָהּ.

אָמַר הַמָּגִּיד: וְכַאֲשֶׁר שָׁמַעְתִּי תֹקֶף שִׁירוֹ וְהֶגְיוֹנוֹ.

וְצַחוּת לְשׁוֹנוֹ. יָדַעְתִּי כִּי הוּא הַיָּחִיד בְּדוֹרוֹ מֵאֵין שֵׁנִי.

חֲבֵרֵנוּ חֶבֶר הַקֵּינִי. וְרָצִינוּ לְהִתְעַלֵּס בְּאַהֲבָתוֹ.

וּלְהִשְׁתַּעֲשֵׁעַ בְּחֶבְרָתוֹ. אֲבָל נָטַר בְּלִבּוֹ אֵיבָתֵנוּ. וּמָאַס

חֶבְרָתֵנוּ. וַיִסַּע בְּכַעַס מֵחֲבוּרָתֵנוּ. אַחֲרֵי אֲשֶׁר הִבְהִיל

מַחְשְׁבוֹתֵינוּ. וְהִדְבִּיק לְחִכֵּנוּ לְשׁוֹנֵנוּ. וְנִשְׁאַרְנוּ נִכְלָמִים

עַל הָרַע אֲשֶׁר עָשִׂינוּ. וְאָמַרְנוּ: אוֹי לָנוּ כִי נִדְמִינוּ.

אֲבָל אֲשֵׁמִים אֲנַחְנוּ עַל אָחִינוּ.
