בָּרְכִי נַפְשִׁי אֶת יְיָ

בָּרְכִי נַפְשִׁי אֶת יְיָ

בָּרְכִי אֲיֻמָּה מִכָּל נוּצָּר וְנִבְרָא / אֶת שֵׁם אָדוֹן אֲשֶׁר לוֹ הַגְּדֻלָּה וְהַגְּבוּרָה

אֶחָד וּמִסּוֹד קְדֻשָּׁתוֹ נִפְחַד וְנִירָא / אֵין קָדוֹשׁ כַּיְיָ


בָּרְכִי בְחוּנָה מִכָּל נֶפֶשׁ חַיָּה / אֶת שֵׁם בָּרוּךְ שֶׁאָמַר לָעוֹלָם וְהָיָה

בְּחָרֵךְ וְהִנְחִילֵךְ דָּת וְתוּשִׁיָּה / בְּרוּכָה אַתְּ לַיְיָ


בָּרְכִי גוֹמֶלֶת חֶסֶד לַגּוּף לְהַחֲיוֹתוֹ / אֶת שֵׁם גּוֹאֲלֵךְ לְמַעַן תּוֹרָתוֹ וּמִצְוָתוֹ

גָּבוֹהַּ וּמַשְׁגִּיחַ לַכֹּל בַּעֲבוּר בְּרִיתוֹ / גְּדוֹלִים מַעֲשֵׂי יְיָ


בָּרְכִי דְּבֵקָה בַּגּוּף בְּלֹא רָצוֹן / אֶת שֵׁם דּוֹבְרֵךְ וְהִזְהִירֵךְ מִלָּצוֹן

דְּגָלֵךְ בְּשֵׂכֶל פְּנִימִי וְחִיצוֹן / דַּעַת צִדְקוֹת יְיָ


בָּרְכִי הַשּׁוֹכֶנֶת בְּחַדְרֵי חֲדָרִים / אֶת שֵׁם הַמֵּכִין מְקוֹמֵךְ לְעִתִּים סְפוּרִים

הִטַּהֲרִי וְעִמְדִי בִמְקוֹם טְהוֹרִים / הַנִּגָּשִׁים אֶל יְיָ


בָּרְכִי וְעוּדָה לְהַשִּׂיג הַמַּעֲלוֹת הָרָמוֹת / אֶת שֵׁם וָעֵד הַמְחַכְּמֵךְ בְּחָכְמוֹת נַעֲלָצוֹת

וְהַחוּשִׁים קָרְאוּ בִשְׁמוֹתָם עֲלֵי אֲדָמוֹת / וְעָלַיִךְ יִזְרַח יְיָ


בָּרְכִי זַכָּה מִטֶּנֶף הַגּוּף וּמַעֲלָלוֹ / אֶת שֵׁם זוֹכֵר חֶסֶד נְעוּרַיִךְ בְּהֵיכָלוֹ

זַמְּרִי לִשְׁמוֹ בְּכָל רֶגַע בְּמַהֲלָלוֹ / זֶה הַשַּׁעַר לַייָ


בָּרְכִי חֲגוּרָה בְעֹז לַעֲבוֹדַת יוֹצְרֵךְ / אֶת שֵׁם חוֹנְנֵךְ דַּעַת מְלַמְּדֵךְ ומוֹרֵךְ

חָפֵץ לְזַכּוֹתֵךְ אִם תְּקַבְּלִי מוּסָרֵךְ / חֵן וְכָבוֹד יִתֵּן יְיָ


בָּרְכִי טְהוֹרָה כְּגַלְגַּלֵּי מַעְלָה / אֶת שֵׁם טָהוֹר בָּחַר בְּעַמּוֹ סְגֻלָּה

טוֹב וּמֵטִיב וְלוֹ דוּמִיָּה תְהִלָּה / טַעֲמוּ וּרְאוּ כִּי טוֹב יְיָ


בָּרְכִי יוֹשֶׁבֶת בַּגּוּף עַד יוֹם פָּקְדֵךְ / אֶת שֵׁם יוֹשֵׁב תְּהִלּוֹתַיִךְ וְיוֹדֵעַ סוֹדֵךְ

יִרְאוּ הַחַטָּאִים כְּבוֹדֵךְ וְהוֹדֵךְ / יִזְכְּרוּ וְיָשׁוּבוּ אֶל יְיָ


בָּרְכִי כְלוּלָה מִן הַמַּעֲלוֹת כֻּלָּם / אֶת שֵׁם כַּבִּיר כֹּחַ מְכַפֵּר לַחַטָּאִים מַעֲלָם

כּוֹנֵן כִּסְאוֹ בְּחֶסֶד וּמִשְׁפָּט וְצִוָּה לְעוֹלָם / כִּי בְאֵלֶּה חָפַצְתִּי נְאֻם יְיָ


בָּרְכִי לוֹבֶשֶׁת כָּבוֹד מֵהֲדַר אֵל וְהוֹדוֹ / אֶת שֵׁם לוֹבֵשׁ גַּאֲוָה וְלֹא נוֹדַע סוֹדוֹ

לְקָחֵך מִתַּחַת כִּסֵּא כְבוֹדוֹ / לַעֲמֹד לְשָׁרֵת בְּשֵׁם יְיָ


בָּרְכִי מְיַחֶלֶת לַעֲלוֹת לְמַחְצֵבָהּ הָרִאשׁוֹן / אֶת שֵׁם מוֹרֵךְ מְצַוֶּה בְלֹא פֶה וְלֹא לָשׁוֹן

מְחַכָּה לְמַלְכָּהּ הַשּׁוֹמְרָהּ כְּאִישׁוֹן / מִדֵּי עֲלוֹתָה בְּבֵית יְיָ


בָּרְכִי נִשְׁקֶפֶת בְּכָל עֵת עַל הַגּוּף לְעוֹרְרוֹ / אֶת שֵׁם נִצָּב בַּשָּׁמַיִם עַד עֵת בּוֹא דְבָרוֹ

נוֹעָדִים מַלְאָכָיו לְפָאֲרוֹ בְהִלּוֹ נֵרוֹ / נוֹשְׂאֵי כְלֵי יְיָ


בָּרְכִי סוֹפֶרֶת לְעַצְמָהּ יָמִים וְשָׁנִים / אֶת שֵׁם סוֹלֵחַ עֲווֹנוֹת אֲדֻמִּים כַּשָּׁנִים

שְׂמֵחָה לְהַצִּילָהּ מִמַּיִם הַזֵּידוֹנִים / סוֹמֵךְ צַדִּיקִים יְיָ


בָּרְכִי עֲלוּבָה בְאוֹדוֹת הַגּוּף וְקוֹרוֹתָיו / אֶת שֵׁם עֶלְיוֹן עַל כָּל גָּבוֹהַ וּגְבוֹהִים מְשָׁרְתָיו

עוֹנֶה שַׁוְעַת צוֹעֵק מִצָּרוֹתָיו / עֶזְרִי מֵעִם יְיָ


בָּרְכִי פּוֹתַחַת בְּכָל עֵדָה בְּמַקְהֵלוֹת / אֶת שֵׁם פּוֹדֶה לְעַמּוֹ מְמַלֵּא כָל שְׁאָלוֹת

פּוֹנֶה אֶל פִּלּוּלֵךְ בְּקוּמֵךְ לְחַלּוֹת / פְּנֵי הָאָדוֹן יְיָ


בָּרְכִי צוּרָה מְצֻיֶּרֶת בְּצֶלֶם אֱלֹהִים / אֶת שֵׁם צוּר עוֹלָם הַמּוֹפִיעַ מֵרוּם גְּבוֹהִים

צָדַק בְּמַאֲמָרוֹ וְכָל יוֹדְעָיו תְּמֵהִים / צַדִּיק הוּא יְיָ


בָּרְכִי קְרוֹבָה אֶל מַעֲלוֹת הַסֻּלָּם / אֶת שֵׁם קָרוֹב לְכָל קוֹרְאָיו נִגְלֶה וְנֶעֱלָם

קַנָּא וּמִי הִקְשָׁה אֵלָיו וַיִּשְׁלָם / קָדוֹשׁ וּמְקֻדָּשׁ יְיָ


בָּרְכִי רָמָה מִמַּחְצֵב הַיְּקָר חֲצוּבָה / אֶת שֵׁם רוֹכֵב שָׁמַיִם לֹא יִשְׁבְּתוּ מִמְּסִבָּה

רָחַשׁ וְהִקְדִּים לַחַטָּאִים תְּשׁוּבָה / רָם עַל כָּל גּוֹיִים יְיָ


בָּרְכִי שׁוֹקֶלֶת מַעֲשֶׂיהָ בְּכָל שָׁנָה / אֶת שֵׁם שׁוֹמֵעַ צַעֲקַת יָתוֹם וְאַלְמָנָה

שׁוֹקֵל בְּדַעְתּוֹ וְדַעְתּוֹ לֹא תִּשְׁתַּנֶּה / שָׁם יְתַנּוּ צִדְקוֹת יְיָ


בָּרְכִי תַמָּה וְהַשֵּׂכֶל לִכְבוֹדָהּ מִשְׁתָּאֶה / אֶת שֵׁם תּוֹלֶה עוֹלָם בְּעַמּוּד נֶעְלָם מִמַּרְאֶה

תִּקַן גּוּף וּנְשָׁמָה סוֹבֶלֶת לוֹ וְלֹא תִלְאֶה / תְּחִלַּת דִּבֶּר יְיָ


זוֹֹקֵף כְּפוּפֵי רֹאשׁ יָחִיד הֲקִימֵנוּ / יְהִי שִׁמְךָ לְעוֹלָם בְּכָל נְתִיבוֹתֵינוּ

סוֹד עֶשֶׂר דִּבְּרוֹת יָה נְסָה עָלֵינוּ / אוֹר פָּנֶיךָ יְיָ

בָּרְכִי נַפְשִׁי אֶת יְיָ
