לִפְנֵי אֱלֹהִים אֶעֱרֹךְ שִׂיחִי

לִפְנֵי אֱלֹהִים אֶעֱרֹךְ שִׂיחִי / אָשִׁיר בְּמַהְלָלוֹ כְּפִי כֹחִי

עֵת אֶחֱזֶה יֹשֶׁר פְּעֻלּוֹתָיו / אָשִׂישׂ בְּשַׁבַּתִּי וְגַם יַרְחִי

אַפְטִיר בְּשִׁיר שָׂפָה וְאָנִיעַ / לָשׁוֹן וְלֹא אַדְבִּיק לְמַלְקֹחִי

אָבוֹא חֲדַר הַשִּׁיר לְבַדִּי כִּי / מִי זֶה אֲשֶׁר יוּכַל סְגֹר פִּתְחִי

אֶתְבּוֹנְנָה רַעַם גְּבוּרוֹתָיו / בַּשִּׁיר וּמִי הוּא יַעֲמֹד נִכְחִי

יִלְחַם לְמִלְחַמְתִּי בְנִיב שָׂפָה / עַל מַהֲלַל דָּוִד בְּנוֹ קִמְחִי

נַפְשׁוֹ צְרוּרָה תּוֹךְ צְרוֹר חַיִּים / לָהּ אוֹמְרִים רָנִּי וְגַם שִׂמְחִי

סִקַּל מְסִלּוֹתַי בְּכָל מִקְרָא / לֹא כָהֲתָה עֵינִי וְלֹא נָס לֵחִי

אָבִין בְּדִקְדּוּקִים וְנִפְעָלִים / הֵאִיר אֲפֵלָתִי וְצָץ פִּרְחִי

לַמַּעֲנֵהוּ כָּל אֲשֶׁר פָּעַל / בִּנְיָן מְיֻסָּד לַעֲקֹר טָרְחִי

הוּא לַעֲנָוָה רֹאשׁ בְּכָל אָפְנָיו / שֹׁרֶשׁ פְּרִי שִׂכְלִי בְּטוּב לִקְחִי

תִּקַּן לְשָׁרָשִׁים מְפֹרָשִׁים / כִּדְבַשׁ וְצוּף מָתוֹק בְּכָל לֵחִי

עֵדָה פְּזוּרָה צַהֲלִי קוֹלֵךְ / פִּתְחִי בְּשִׁיר עָרֵב בְּרֹן פִּצְחִי

גִּילוּ בְּנֵי יַעֲקֹב בְּכָל מִקְרָא / אִכְלוּ לְמַשְׁמַנִּים בְּמוֹ טִבְחִי

סִימָן לְכָל פָּסוּק כְּפִי סִדְרוֹ / עָרוּךְ וְלֹא אִירָא קְצָר רוּחִי

מִי יִתְּנָה לִי לֵב בְּפֵרוּשָׁיו / אָבִין בְּמַרְגֹּעִי וּבִמְנוּחִי

אֶשְׁתַּעְשְׁעָה בָּהֶם בְּגָלוּתִי / עַד הָאֱלֹהִים יֶאֱסֹף נִדְחִי

כִּי הוּא לְעֶזְרָתִי וְהוּא כִסְלִי / הוּא בֶאֱמֶת גּוֹחִי וּמִבְטַחִי

הוּא יַאֲרִיךְ אַפּוֹ לְצֹאן יָדָיו / תָּמִיד יְכַפֵּר עַל קְשִׁי מִצְחִי

יִרְפָּא לְמַכְאוֹבִי וְהוּא יִמְרַח / גַּם עַל שְׁחִין לִבִּי עֲדֵי יֶחִי

יִשְׁלַח מְבַשֵּׂר טוֹב לְכָל עַמּוֹ / עֵינַי תְּלוּיוֹת לוֹ וְלִמְשִׁיחִי

יִשְׁמַע לְקוֹל שִׁירִי וְגַם יִרְצֶה / אוֹתוֹ כְּקָרְבָּנִי וְנִיחוֹחִי

נַפְשִׁי בְּהָקִיץ תַּעֲנֶה כַיּוֹם / לִפְנֵי אֱלֹהִים אֶעֱרֹךְ שִׂיחִי.
