אמרתי אספרה כמו

          למען ידעו דור אחרון צדיק הולך בתמו, ואת יקר תפארת

          גדולת גבר חכם בעוז דורם טוב לעמו, יחיד בדורו לכתוש את המקבלים

          ולשפוך עליהם זעמו, משורר אשר נודע ביהודה ובישראל גדול שמו,

          כמוהר"ר יהודה אריה

          ממודינא ז"ל בעדן גן אלהים עד יום קומו:


מִי זֶה אֲשֶׁר לִבּוֹ יֵרַךְ כַּמָּיִם,

יִתֵּן דֳּמִי לַגֶּבֶר

חָכָם וְאַמִּיץ כֹּחַ;

אֶל הַרְרֵי זִמְרָה בַל יִפְרשׁ אֵבֶר

לָשִׁיר בְּאֵין מָנוֹחַ

לִכְבוֹד נְקִי כַפָּיִם,

נָבוֹן טְהוֹר לֵבָב, אֶרֶךְ אַפָּיִם?

אוּלָם אֱנוֹשׁ כָזֶה סָכָל וָבַעַר

נֶחְשָׁב כְּחַיְתוֹ יָעַר;

אַךְ אִם בְּקֶרֶב אִישׁ יָשְׁרוֹ הִבִּיעַ,

גַּם מַעֲשֵׂה יָדָיו מַגִּיד רָקִיעַ.


כֵּן אֶאֱזֹר מָתְנַי, וּבְחֹזֶק מֵצַח

אָקוּם לְסַפֵּר הֵנָה

עַל תֹּף בְּאִמְרֵי ישֶׁר

חַיֵּי יְהוּדָה גוּר אַרְיֵה מוֹדֵינָא,

עַתָּה בְגַן הָאשֶׁר,

לִשְׁכֹּן שְׁלֵיו לָנֶצַח

מִמַּעֲשָׂיו לִקְצֹר רָב-בַּר וָקֶצַח.

שִׂישִׂי וְשִׂמְחִי בַזָר קִרְיַת פֵירָארָה,

הָעִיר אֲשֶׁר נִבְחָרָה,

לִגְלֹם בְּמִסְתָּרִים גֹלֶם הָעֶלֶם,

לֵאוֹר דְּמוּת אֵל בּוֹ, חַיִּים וָצֶלֶם.


גָּלָה אֲרִי טֶרֶם צֵאתוֹ מֵרֶחֶם;

אָבִיו וְאִמּוֹ יָחַד

נָסוּ לְחוֹף הַיָּמָה

כִּי נָפְלָה בָהֶם אֵימָה וָפָחַד;

עַל הָרְעָשִׁים שָׁמָה

הֵם נָדְדוּ לַלֶּחֶם;

אַךְ לָקְחוּ חֶלְקָם, לִשְׂכָרָם שֶׁכֶם,

וּבְסוֹף מְגוּרֵיהֶם מוֹשָׁב מָצָאוּ

יוֹם כִּי בְנֶצְיָה בָּאוּ:

יֻלַּד יְהוּדָה אָז [אַחַר כִּי שָׁטוּ]

בֵּית יַעֲקֹב זָקֵן מִבֵּית לוּצָאטוּ.


כִּי חָדְלוּ קֹלוֹת, רַעַשׁ וָסָעַר

שָׁבוּ לְעִיר פֵירָארָה

לָתֵת עֱזוּז לַיָּלֶד

אַךְ,הָהּ! בְּאָבִיו רוּחַ לֹא נִשְׁאָרָה!

עָזַב בְּאִבּוֹ חָלֶד!

אָז גִּיר בְּנוֹ עוֹד נָעַר

הָלַךְ לְוֵינֶצְיָה לִדְרשׁ בַּשָׁעַר;

שָׁם מִשְׂדֵה עַמּוֹ בִּעֵר כָּל-חֶדֶק

לִלְקֹט פְּרִי הַצֶּדֶק,

כִּי מִנְּעֻרָיו בֶּן עֶשְׂרִים וּשְׁתָּיִם

לֶחֶם בְּפִיו הִמְטִיר מִן הַשָּׁמָיִם.


אַשְׁרֵיךָ וּמַה טוֹב לָךְ, עִיר הַתִּפְאֶרֶת,

מָשׂוֹשׂ לְכָל-הָאָרֶץ,

הָעִיר כְּלִילַת-יֹפִי,

עֹז בָּךְ, וְאֵין בִּרְחֹבֹתַיִךְ פָּרֶץ

אֵין מוּם וְאֵין בָךְ דּוֹפִי,

בִּמְאֹד מְאֹד נֶהְדֶּרֶת,

אַתְּ אַתְּ יְחִידָה מֵאִיִּים נִבְחֶרֶת;

טוֹבָה מְנַת חֶלְקֵךְ, כִּי בָךְ שָׁכָנָה

נֶפֶשׁ אֲשֶׁר נֶאְמָנָה,

זַכָּה כְמֵי נָהָר, מַעְיָן וָנָחַל.

אֶל תּוֹךְ יְהוּדָה, גּוּר, אַרְיֵה וָשָׁחַל.


אַרְיֵה לְמַעְלָה רֹאשׁ, אֶרֶךְ הָאֵבֶר,

גָּבַר לְנֵס הָעִירָה;

[הָעִיר פְּלִיאַת חָלֶד

כִּי בָהּ אֱמוּנַת אֵל תָּמִיד הֵאִירָה,

לֵאוֹר בְּשָׂב וָיָלֶד

שֶׁמֶשׁ יְקָר וָשֶׁבֶר,

חָכְמָה וְתוּשִׁיָּה לִמְסִלַּת גֶּבֶר.]

וּבְסוֹד יְשָׁרִים, רָע הֵסִיר מִנָּעַר,

הִשְׂכִּיל כְּסִיל וָבָעַר;

חֹק לַיְּהוּדִים שָׂם, וּלְחַזֵק בֶּדֶק,

בִּלְשׁוֹן בְּנֵי עַמּוֹ הִצְמִיחַ צֶדֶק.


מִדְבָּר אֲשֶׁר הָיָה לִישִׁימוֹן דֶּרֶךְ,

מָעוֹן לְגוּר וָשָׁחַל,

נֶהְפַּךְ לְמוֹצָא מָיִם;

מַיִם מְפַכִּים מִמַּעְיָן לַנָּחַל

מִפִּי נְקִי כַפָיִם,

חָכָם אֲשֶׁר כָּל-בֶּרֶךְ

תִּכְרַע לְפָנָיו; הֵן מִכֹּבֶד פֶּרֶךְ

גֵּאוּת, לְדֹר וָדֹר הֵסִיר לֵב גָּבֶר;

כִּי לַאֲחֻזַּת קָבֶר

הֶרְאָה לְרָם לֵבָב ישֵׁב בָּאָרֶץ,

רַק צַחֲנָה, בָאְשָׁה, רִמָּה וָשָׁרֶץ.


דָּרַשׁ לְעַמּוֹ טוֹב, פָּתַח כָּל-צֹהַר

לֵאוֹר בְּחֶשְׁכַּת לָיִל

לַיּשְׁבִים בַּכֶּלֶא.

לִקְרַאת נְחָשִׁים, אִישׁ גִּבּוֹר הֶחָיִל

דִּבֶּר וְעָשָׂה פֶלֶא.

הֶחְבִּיא בְּבֵית הַסֹּהַר

סִפְרֵי מְקֻבָּלִים עִם סֵפֶר זֹהַר

בַּל יֶחֱזוּ עוֹד שֶׁמֶשׁ אוֹ יָרֵחַ;

שׁוֹאֵג וּמַר צֹרֵחַ

בָּם כַּאֲרִי נֹהֵם, קוֹלוֹ קוֹל שָׁחַל,

נֶגֶד אֲמִתַּי הוּא שׁוֹטֵף כַּנָּחַל.


חָכָם כְּבֶן-דָּוִד, חֹקֵר בַּסֵּתֶר

הִפְלִיא לְמַלֵּל אֹמֶר:

כִּי פַעֲמַיִם, נֶפֶשׁ

לֹא לָבְשָׁה גֶבֶר קֹרַץ מֵחֹמֶר;

כִּי לֹא יְכֻבַּס רֶפֶשׁ

אוֹ טִיט אֱנוֹשׁ בַּנֶּתֶר,

רַק אֵל גְּמֻלּוֹת וּמְשַׁלֵּם עַל יֶתֶר

נֹתֵן לְרָשָׁע רָע, וּלְצַדִּיק אשֶׁר:

כִּי כֵן נְתִיב הַיּשֶׁר,

כָּל מִפְעֲלוֹת אִישׁ אִישׁ לִשְׁקֹל בַּפֶּלֶס,

לָתֵת לְטוֹב הַטּוֹב, וּלְרָע רַק קֶלֶס.


לִבּוֹ כְלֵב אַרְיֵה, לֹא פָחַד אֶרֶב,

הִכָּה בְאֶגְרוֹף רֶשַׁע,

עֹשֵׂה פְעֻלַּת שֶׁקֶר.

פִּיהוּ כְפִי אַרְיֵה עַל חֵטְא וָפֶשַׁע,

דֹּבֵר עֲדֵי אֵין חֵקֶר

לָנוּס מְנֻסַּת חֶרֶב

יֵצֶר לְבַב אָדָם עַד בּוֹא הָעֶרֶב.

מִמִּיץ חֲלֵב תַּלְמוּד הוֹצִיא כָל-רֶקַח,

שָׁמַע וְהֹסִיף לֶקַח,

לָתֵת לְאוֹר עוֹלָם סִפְרוֹ בַּגֹּבַהּ,

חָכָם בְּחַשְׁמַנִּים מִנִי לוֹדוֹבָה.


מִי יַעֲלֶה אֶל הַר זִמְרָה כָמֹהוּ?

שִׁירָיו בְּמַעְגַּל ישֶׁר,

מִכְתַּב אֱלֹהִים הֵמָּה.

וַתַּךְ בְּשֵׁבֶט פִּיו גֵּאוּת וָעשֶׁר

בַּת שִׁיר אֲשֶׁר נָעֵמָה;

בִּי בֹטְחִים עַל בֹּהוּ,

מִתְהַלְלִים בָּהוֹן, נוֹטִים עַל תֹּהוּ,

יִתְנַגְּפוּ פִתְאֹם עַל הָרֵי נֶשֶׁף;

כֻּלָם לְחֻמֵּי רֶשֶׁף,

קֶטֶב מְרִירִי בָם, רֹאשׁ פֶּתֶן לָמוֹ,

רַק גַּאֲוָה, קָלוֹן וָבוּז פִּרְיָמוֹ.


הֵלִיץ בְּעַד פָּלְקָאר, וּבְדִבְרֵי זָעַם

מָגֵן וְאִבְחַת חָרֶב

נֶגֶד מְנַבֶּל-צֶדֶק

נָתַן בְּרָב-כֹּחוֹ בֹקֶר וָעֶרֶב;

הָיָה כְקוֹץ וָחֶדֶק,

פַּטִּישׁ בְּהוֹלֶם פָּעַם,

לָדִין בְּנֵי עַמּוֹ עַל אִמְרֵי טָעַם,

בַּל יִפְרְצוּ חוֹמַת הַדָּת וָגֶדֶר;

שׁוֹמֵר כְּרֹעֶה עֵדֶר,

מִפִּי זְאֵב מַצִּיל צֹאן, עֵז וָכֶשֶׂב,

כִּי שַׁלְאֲנָן אוֹכֵל כָּל-יֶרֶק עֵשֶׂב.


הוֹצִיא לְעֹז עַמּוֹ רֹב אִמְרֵי שֶׁפֶר

מֵאֹצְרוֹת שָׁמָיִם;

שֶׁמֶשׁ לְשׁוֹן בֶּן-עֵבֶר

הֵאִיר בְּמִקְרָא קֹדֶשׁ לָעֵינָיִם;

כֵּן מִצְעֲדֵי הַגֶּבֶר

הִדְרִיךְ לְבֵית הַסֵּפֶר,

נֶחְמָד לְמַרְאֶה, רַק מִמִּשְׁלֵי אֵפֶר,

עַד כִּי בְּרֹב גֻּדְלוֹ הֵשִׁיב מִפֶּשַׁע

אִישׁ רִיב וּמַטֵּה רֶשַׁע;

וַיִּצְעֲקוּ אֶל אֵל יוֹם בַּצַּר לָמוֹ,

מִתּוֹךְ וּמֵחָמָס יִגְאַל נַפְשָׁמוֹ.


שָׁנָה אֲשֶׁר בָּהּ, אוֹי! בָּא חִיל וָרָעַד

אֶל ישְׁבֵי פֵרָארָה,

הֵן הִיא שְׁנַת הֻלֶּדֶת

גֶּבֶר בְּלֵב תָּמִים, נֶפֶשׁ יָקְרָה;

נֶפֶשׁ אֲשֶׁר שֹׁקֶרֶת

יוֹם יוֹם בְּאַנְשֵׁי סָעַד,

הֵמָּה בְּדַלְתֵי דָת לִמְטִיבֵי צָעַר.

לָתוּר וְלָסוּר מִדּוֹרוֹ הַתֹּהוּ

לֹא קָם אֱנוֹשׁ כָּמֹהוּ;

יָחִיל בְּקֹלוֹ אִישׁ גַּם שָׂב גַּם יָלֶד,

שָׁאַג אֲרִי מִי לֹא יִירָא בֶחָלֶד?


אֵין עוֹד –אֲרִי אֵין עוֹד – אֵין עוֹד בָּאָרֶץ!

גֵּווֹ סְחִי וָשָׁחַק

וּבְטִיט לְמִרְמַס רֶגֶל.

[רוּחוֹ וְנִשְׁמָתוֹ אַךְ בִּשְׁמֵי שָׁחַק

חֹסוֹת בְּצֵל הַדֶּגֶל.]

הָהּ שִׁמְמָה וָקָרֶץ!

אֵין לוֹ תְמוּרָה, אֵין עֹמֵד בַּפָּרֶץ!

שָׁנָיו בְּמוֹתוֹ רַק שִׁבְעִים וָשֶׁבַע,

בָּעִיר לְזֹאת, בַּגֶּבַע

הִתְגוֹדְדוּ אִישִׁים, אַרְצָה נָפָלוּ,

בָּכוּ בְקוֹל גָּדוֹל וַיִתְאַבָּלוּ.


מִשְׁפָּט וְחֹק מִי כָמֹהוּ יֹדֵעַ? –

יָדִי לְמוֹ פִי שָׂמְתִּי,

כִּי רַק שְׁכֶם מֵאֶלֶף

מֵרֹב חֲסָדָיו אָנֹכִי רוֹמָמְתִּי.

בֹּזֶה לְזוּת וָסֶלֶף,

אֹהֵב יְדִיד וָרֵעַ,

מַשְׂכִּיל עֲנִיֵּי עָם, בָּם מִשְׁתַּעְשֵׁעַ.

אַשְׁרֵי אֱנוֹשׁ פָעַל כָּל-זֹאת בָּאָרֶץ

צַדִּיק וְגֹדֵר פָרֶץ!

בַּגָּן[ אֲשֶׁר לָשׁוֹן תִּקְצַר וַתֵּלַהּ]

נַפְשׁוֹ בְטוֹב תָּלִין עַד עוֹלָם סֶלָה.
