הָיָה בְצִיּוֹן בָּחוּר אֶחָד

מתוך השער הרביעי של משל הקדמוני – מעשה כשפים


הָיָה בְצִיּוֹן בָּחוּר אֶחָד נֶחְמָד / אֲשֶׁר דֶּרֶךְ

הַמְּלִיצָה לָמַד, / וַיְהִי לוֹ בְּכָל חָכְמָה מָבוֹא / כִּי לְכָל

תְּבוּנָה הֵכִין לְבָבוֹ. / וַיְבַקֵּשׁ לָדַעַת בְּעִנְיַן הַכְּשָׁפִים

/ וַיְחַפֵּשׂ וְלֹא מָצָא אֶת הַתְּרָפִים: / כִּי נֶאֱסַר הַכִּשּׁוּף

לְאֻמָּה חֲמוּדָה / לְיוֹשֵׁב יְרוּשָׁלַיִם וּלְאִישׁ יְהוּדָה. /

וְכִרְאוֹת הַבָּחוּר כִּי הַתְּשׁוּקָה לְלִבּוֹ נוֹגַעַת / וּבַקָּשָׁתוֹ

בְּאֶרֶץ הַצְּבִי נִמְנַעַת, לָקַח מֵהוֹנוֹ דָּבָר נִרְשָׁם / וַיֵּרֶד

מִצְרַיְמָה וַיָּגָר שָׁם.

וַיְהִי בִתְחִלַּת בּוֹאוֹ אֶל הָעִיר / וּמַחֲשַׁבְתּוֹ

הֵעִיר – / וְהִנֵּה זָקֵן אֶחָד יוֹשֵׁב / חָרָשׁ וְחוֹשֵׁב. / וַיִּשָּׂא

עֵינָיו וַיַּרְא אֶת הָאִישׁ הָאוֹרֵחַ / אוֹר זוֹרֵחַ, / כִּי כֻלּוֹ

מַחֲמַדִּים / לָבוּשׁ בַּדִּים. / וַיִּשְׁאָלֵהוּ הַזָּקֵן לָדַעַת

מַחֲשָׁבוֹ / ״אָנָה תֵלֵךְ וּמֵאַיִן תָּבוֹא?״ / וַיֹּאמֶר לוֹ ״עִבְרִי

אָנֹכִי נוֹצֵר תְּעוּדָה / מִבֵּית לֶחֶם יְהוּדָה״. / וַיֹּאמֶר הַזָּקֵן

״סוּרָה, אֲדוֹנִי, אֵלַי – / רַק כָּל מַחְסוֹרְךָ עָלַי״. / וַיָּסַר

אֵלָיו וַיְבִיאֵהוּ אֶל בֵּיתוֹ וַחֲצֵרוֹ / וַיִּתֵּן מַאֲכָל

לַחֲמוֹרוֹ, / וַיִּרְחֲצוּ רַגְלֵיהֶם וַיֹּאכְלוּ וַיִּשְׁתּוּ /

וּפִיהֶם בְּדִבּוּר שָׁתוּ. / וַיֹּאמֶר הַזָּקֵן ״בְּהִתְחַיֵּב כָּל

אָדָם בְּאַהֲבַת הַחֲכָמִים / וְלִבְחֹר מוֹשַׁב זְקֵנִים וְתַמִּים – /

כְּמוֹ כֵן נִתְחַיֵּב לְהִתְנַהֵג עִמּוֹ בְּשֵׂכֶל טוֹב וּמוּסָר / וְכָל

טוּב מִמֶּנּוּ לֹא יֶחְסַר, / וּלְקָרֵב לְבֵיתוֹ כָּל אוֹרֵחַ /

וְלִהְיוֹת בִּכְבוֹדוֹ שָׂמֵחַ, / וּלְדַבֵּר דְּבָרִים עַל לִבּוֹ /

לַהֲנִיחוֹ מֵרָגְזוֹ וּמֵעָצְבּוֹ, / וּלְהַדְרִיכוֹ בְּדֶרֶךְ יְשָׁרָה /

כְּפִי הַדַּעַת וְהַסְּבָרָה: / וְלֹא יִתָּכֵן זֶה זוּלָתִי בְדַעְתּוֹ

רַעְיוֹנָיו / וּלְאֵיזֶה עִנְיָן מְגַמַּת פָּנָיו – / וְעַל פִּי

דַרְכּוֹ יִיעָצֵהוּ וְיוֹעִילֵהוּ / וְעַל מֵי מְנוּחוֹת יְנַהֲלֵהוּ. /

וְעַתָּה הוֹאִיל וּבְבוֹאֲךָ כִּבַּדְתַּנִי / וּלְחָבֵר וּלְרֵעַ לְךָ

שַׂמְתַּנִי / וְאָרַחְתָּ הַלַּיְלָה בִּמְעוֹנִי / סָבִיב לְשֻׁלְחָנִי –

/ הַשְׁמִיעֵנִי מַצְפּוּן לִבְּךָ וּמַהֲלָכֶיךָ, / הוֹדִיעֵנִי נָא אֶת

דְרָכֶיךָ וְאֵדָעֶךָ, / לְמַעַן אֶמְצָא חֵן בְּעֵינֶיךָ. / אִיעָצְךָ

כְּפִי שִׂכְלִי אֶעֱנָדְךָ עֲטָרוֹת לִי״.


וַיֹּאמֶר הַבָּחוּר ״בֶּאֱמֶת צָדַק הֶחָכָם בְּמוּסָרוֹ

/ וְגָמַר אֹמֶר בְּמַאֲמָרוֹ, / כִּי כָל אָדָם צָרִיךְ לְהִוָּעֵץ /

מִפִּי זָקֵן וְיוֹעֵץ – / וְאַף אִם יִהְיֶה מַשְׂכִּיל וּבַעַל מְלִיצָה,

/ אֵין לוֹ יִתְרוֹן בְּלִי עֵצָה: / כָּל שֶׁכֵּן אִישׁ אוֹרֵחַ / נוֹדֵד

מִמְּקוֹמוֹ וּבוֹרֵחַ / וְדַעְתּוֹ בַּנְּדוֹד דְּחוּיָה / בְּאֶרֶץ

נָכְרִיָּה. / וְעַתָּה אוֹדִיעֲךָ מַחְשְׁבוֹת לִבִּי, / אִם דֶּרֶךְ

עֹצֶב בִּי: / אֲנִי הָיִיתִי שׁוֹכֵן בְּצִיּוֹן / כְּפַלְגֵי מַיִם

בְּצָיוֹן, / וְנַפְשִׁי הָיְתָה לְכָל מְלִיצָה מוּדַעַת / וְלִבִּי רָאָה

חָכְמָה וָדַעַת. / וָאֲבַקֵּשׁ לָדַעַת בְּעִנְיַן הַכְּשָׁפִים דָּבָר /

וּבַמֶּה כֹּחָם גָּבַר – / וְלֹא מָצָאתִי בְעִירִי מַעֲנֶה / וְאֵין קוֹל

וְאֵין עוֹנֶה. / וַאֲחַשְּׁבָה לָדַעַת זֹאת / בְּאַחַת מֵעָרֵי

הַפְּרָזוֹת, / וְעַל זֶה מֵאֶרֶץ הַצְּבִי יָצָאתִי / וְעַתָּה בָאתִי /

לְבַקֵּשׁ מַחְסוֹרִי דַּי / לְכָל חַרְטֹם וְאָשַׁף וְכַשְׂדָּי. /

וְעַתָּה אִם יֶשְׁךָ עוֹשֶׂה חֶסֶד וֶאֱמֶת עִמִּי / אַחֲרֵי אֲשֶׁר כָּל

זֹאת נָדַדְתִּי מֵעַמִּי – / הוֹדִיעֵנִי הֲיִמָּצֵא אִישׁ בָּקִי בְזֹאת

הַחָכְמָה, / וְיַעֲבֹר עָלַי מָה, / וּמֹהַר וּמַתָּן יַרְבֶּה עָלַי /

וְאֶתְּנָה כַּאֲשֶׁר תֹּאמְרוּ אֵלַי / וְיַאֲצֵל עָלַי מֵהוֹדוֹ

וִילַמְּדֵנִי / וְאוֹתִיּוֹת הַחְָמָה יַגִּידֵנִי!״

וַיֹּאמֶר הַזָּקֵן ״אַחֲרֵי אֲשֶׁר עַד רַחֲבֵי אֶרֶץ

הִתְבּוֹנַנְתָּ / וְהַלַּיְלָה בְּסֵתֶר אָהֳלִי הִתְלוֹנַנְתָּ /

בְּבוֹאֲךָ לְבַקֵּשׁ חָכְמָה / מֵעֵינֵי יוֹשְׁבֵי אַרְצְךָ נַעֲלָמָה – /

עַל כֵּן יְהָבְךָ עָלַי הַשְׁלֵךְ / וַאֲנִי אֲלַמֶּדְךָ לְהוֹעִיל,

אַשְׂכִּילְךָ בְּדֶרֶךְ זוּ תֵלֵךְ, / וְאוֹדִיעֲךָ עִקַּר הַכְּשָׁפִים /

וְאָוֶן וּתְרָפִים: / כִּי הַחָכְמָה הַזֹּאת לִי לִמְלָאכָה / וּמֵאָבוֹת

אֲבוֹתַי לִי נִמְשָׁכָה״. / וַיִּשְׁמַע הַבָּחוּר עִנְיָנָיו / וַיְהִי

כִמְצַחֵק בְּעֵינָיו, / וַיַּחְשֹׁב בְּלִבּוֹ מַחֲשָׁבוֹת / אֲשֶׁר

תְּבוּנַת הַזָּקֵן מְתַעֲבוֹת. / וַיֹּאמַר: אֵין זֶה כִּי אִם הִטְעָהוּ

יֵין מְסִבּוֹ / וּלְהוֹרוֹת נָתַן בְּלִבּוֹ, / וְיַחְמֹד הוֹנִי וְחֵילִי

/ וְיַחְפֹּץ לְרַמּוֹת דַּעְתִּי וְשִׂכְלִי. / וַיֹּאמֶר לוֹ

״מֵהָאֱלֹהִים תִּמְצָא מַשְׂכֹּרֶת / וְיֶעֱרַב שִׂיחֲךָ כְּשִׁיר

וּקְטֹרֶת, / כִּי דִבַּרְתָּ עַל נֶפֶשׁ אֲנוּשָׁה וּמָרָה, / רוּחַ

נִשְׁבָּרָה. / אֲבָל אֵינִי חָפֵץ לְהַכְבִּיד עֻלֶּךָ / וּלְהַטְרִיחַ

עָלֶיךָ, / כִּי הַחָכְמָה הַזֹּאת גְּדוֹלָה / כַּנָּחָשׁ תִּתֵּן

קוֹלָהּ, / וּרְצוֹנִי לְבַקֵּשׁ בָּחוּר בְּגִילִי / וְיַגִּידֶהָ

וְיַעַרְכֶהָ לִי: / וְהוּא יִשְׁאָלְךָ עַל הַסָּפֵק / לְהַגְדִּיל רָצוֹן

וּלְהָפֵק – / וְאַתָּה תַעְזְרֵהוּ וְתוֹעִילֵהוּ בְּחַסְדֶּךָ / וְלִי

אֲנִי עַבְדֶּךָ״. / וַיֵּדַע הַזָּקֵן כִּי לֹא הֶאֱמִין הַבָּחוּר

לְחָכְמָתוֹ / וְלֹא חָשַׁשׁ לִתְבוּנָתוֹ, / וְנָדַר בְּלִבּוֹ נֶדֶר: /

אִם אָבוֹא בְאֹהֶל אוֹ בְחֶדֶר / וְאִם אֶעֱלֶה עַל עֶרֶשׂ יְצוּעִי, /

עַד אַרְאֵהוּ חָכְמָתִי וְדֵעִי / וְאוֹדִיעוֹ אֶת גְּבוּרָתִי / שִׂכְלִי

וּתְבוּנָתִי. / וַיָּכֶן תַּחְבּוּלוֹת וַיְחַנֵּן קוֹלוֹ / וַיַּסֵּב

פָּנָיו וַיֹּאמֶר לוֹ / ״לִין הַלַּיְלָה וְהָיָה בַבֹּקֶר אַרְאֲךָ אִישׁ

אֲשֶׁר לִתְבוּנָתוֹ אֵין חֵקֶר: / הוּא יוֹרְךָ וִילַמְּדֶךָ – / וְגַם

אֲנִי אוֹדֶךָּ״.

וַיִּיטֶב הַבָּחוּר שַׂרְעַפָּיו / וַיִּשְׁתַּחוּ

לְאַפָּיו / וַיֹּאמַר ״בָּרוּךְ הַשֵּׁם אֲשֶׁר מַהֲלָלוֹ לֹא יִצָּמֵת /

אֲשֶׁר הִנְחַנִי בְּדֶרֶךְ אֱמֶת!״ / וַיֵּשְׁבוּ שָׁעָה אַחַת

מִתְחַבְּרִים – / עוֹדָם מְדַבְּרִים, / וַיִּשְׁאַל הַזָּקֵן לִשְׁתּוֹת

/ לְרַמּוֹת הַבָּחוּר וּלְפַתּוֹת. / וַיִּקַּח הַכּוֹס בִּימִינוֹ /

וַיֵּשְׁתְּ מִן הַיַּיִן לְרַוּוֹת צִמְאוֹנוֹ. / וַיֹּאמֶר אֶל הַבָּחוּר

״שְׁתֵה – וְיִשְׁתַּכְּחוּ תְלָאוֹתֶיךָ, / וְשָׁכַבְתָּ – וְעָרְבָה

שְׁנָתֶךָ״. / וַיִּשְׁתֶּה הַבָּחוּר, וַיִּשְׁתַּנֶה מַחֲשָׁבוֹ /

וַיָּקָם וַיֵּלֶךְ אֶל בֵּית מִשְׁכָּבוֹ, / וַיְהִי הוֹלֵךְ – /

וְהַזָּקֵן לְנֶגְדּוֹ פַּחִים יַשְׁלֵךְ. / וַיַּרְא וְהִנֵּה בְאֵר

בַּשָּׂדֶה / וְהִנֵּה שָׁם מָקוֹם אֲשֶׁר עֲמָלוֹ צוֹדֶה, / וַיַּחְפֹּץ

לְרַגְלוֹ וַיֵּחָשֵׁל / וַיִּפֹּל בְּבוֹר סִיד וַיִּכָּשֵׁל. / וְהַבּוֹר

רֵיק אֵין בּוֹ מַיִם / וַיִּשָּׂא לְבָבוֹ אֶל כַּפַּיִם / וַיִּצְעַק

וַיֶּהֱמֶה בְכַחֲשׁוֹ / כַּאֲשֶׁר רָאָה צָרַת נַפְשׁוֹ. / וַיֵּלֶךְ כָּל

הַלַּיְלָה בְּמַחֲשַׁכִּים / וְגֵווֹ נָתַן לְמַכִּים, / וַיְהִי בַבֹּקֶר

וַיַּרְא בְּיַרְכְּתֵי הַבּוֹר דֶּלֶת, / וְנַפְשׁוֹ עָלָיו מְיַלֶּלֶת. /

וְאָנָה תֵלֵךְ, / אֲשֶׁר חֵץ יְדִידוֹת בְּלֵב רוֹאֶיךָ תַּשְׁלֵךְ?״ /

אֱלֹהִים בַּשָּׁמַיִם, / וּבָאתִי לְמִצְרַיִם לְבַקֵּשׁ חָכְמַת

הַכִּשּׁוּף – / אֶת הַכּוֹס וְאֶת הַיַּנְשׁוּף״. / וַיִּצְחַק הַסּוֹפֵר

מִדְּבָרָיו / וַיִּקְרָא אֶל הַסּוֹפְרִים חֲבֵרָיו / וַיֹּאמַר

״הַיְדַעְתֶּם עַל עִיר – יְרוּשָׁלַיִם שְׁמָהּ, / וְאוֹרֵחַ יְבַקֵּשׁ

לוֹ בִּירוּשָׁלַיִם״. / וַיִּתְמְהוּ הַסּוֹפְרִים עַל דְּבָרוֹ /

וַיָּבִיאוּ לוֹ מִגְדָּנוֹת, / וַיֵּשְׁבוּ כֻלָּם סְבִיבוֹ /

בִּנְגִינוֹתָיו מְנַצֵּחַ / וַיִּכְתֹּב לָהֶם עוֹד שְׁתֵּי שׁוּרוֹת /

שִׁירוֹ וְהוֹדוֹ / וַיֹּאמְרוּ ״הֲכָזֶה רָאוּי לְבֵית הַמַּלְכוּת /

/ לֵאמֹר ״פְּעָמֶיךָ לְמַסָּעוֹת הָרִימָה, / וְתִרְאֶה סוֹפֵר מָהִיר /

כְּזֹהַר הָרָקִיעַ יַזְהִיר, / הָאִישׁ אֲשֶׁר בָּא הַיּוֹם אֵלֵינוּ – /

הִנֵּה זֶה עוֹמֵד אַחַר כָּתְלֵנוּ״. / וַיִּשְׁלַח הַמֶּלֶךְ

בַּעֲבוּרוֹ, / וַיָּכֶן לְנֶגְדּוֹ פַּעֲמֵי אֲשׁוּרוֹ. / וַיָּבוֹא

וַיִּשְׁתַּחוּ הָאִישׁ הַנֶּאֱהָב / וַיּוֹשֶׁט לוֹ הַמֶּלֶךְ אֶת

שַׁרְבִיט הַזָּהָב. / וַיֵּשְׁבוּ הַסּוֹפְרִים לְהוֹדוֹת מֵהֲלָלוֹ /

מִימִינוֹ וּמִשְׂמֹאלוֹ / וַיְדַבְּרוּ לַמֶּלֶךְ אֲרָמִית / בְּאָזְנֵי

כָל רֵעַ וְעָמִית / ״הִנֵּה זֶה הָאִישׁ אֲשֶׁר בּוֹ רוּחַ יַתִּירָה /

וּמִלְתָא דִי מַלְכָּה שָׁאַל יַקִּירָה / לְשַׁמָּשָׁא קֳדָמוֹהִי גְּבַר

קַדִּישִׁין בֵּהּ״. / וַיְהִי בְהִתְבּוֹנֵן הַמֶּלֶךְ מַעֲנֵיהֶם /

וְרוּחַ לֹא יָבוֹא בֵינֵיהֶם. / וְאָמְרוּ: אַל תִּשְׁכֹּן בְּעִיר אֲשֶׁר

וְרַעְיוֹנַי / כִּי מָצָאתָ חֵן בְּעֵינַי״. / וַיֶּאֱסֹף הַמֶּלֶךְ

פָּרָשָׁיו וּגְדוּדָיו / וַיַּעַשׂ מִשְׁתֶּה לְכָל שָׂרָיו וַעֲבָדָיו /

וַיְדַבֵּר בְּאָזְנֵיהֶם דְּבָרִים נִמְרָצִים, / אַהֲבַת הָאִישׁ

חֲפֵצִים, / מַקְדִּישִׁים וּמַעֲרִיצִים. / וַיִּיטְבוּ בְעֵינֵיהֶם

מְלִיצוֹתָיו / וְדַעַת שְׂפָתָיו / וַיְקַבְּלוּ אוֹתוֹ לְאָדוֹן

וּלְמֶלֶךְ / כָּל עָשִׁיר וָחֵלֶךְ / וַיִּתֶּן לוֹ הַמֶלֶךְ אֶת בִּתּוֹ

לְאִשָּׁה / וְהוֹצִיא אֶת אֶבֶן הָרֹאשָׁה. / וַיִּשְׂמַח כָּל עַם

הָאָרֶץ וְהָעִיר שָׁקְטָה, / רִפָּא שְׁבָרֶיהָ כִּי מָטָה, /

וַיִּקְרְאוּ לְפָנָיו ״אַבְרֵךְ״ / כַּאֲשֶׁר כִּלּוּ לְבָרֵךְ.


וַיַעַבְרוּ יְמֵי הַמִּשְׁתֶּה וְהַשִּׂמְחָה /

בְּהַשְׁקֵט וּבְבִטְחָה, / וַיִּבְנֶה לוֹ הָאִישׁ בֵּית מִדָּה / אֵת

כָּל שְׂכִיּוֹת הַחֶמְדָּה, / וַיִּכְרֶה בוֹר אֶחָד תּוֹךְ חֲצֵרוֹ /

לְהַשְׁלִים הוֹדוֹ וַהֲדָרוֹ, / וַיִּבֶן לְאִשְׁתּוֹ בֵּית מִדָּה, /

מָשׂוֹשׂ וּמַחְמַד עֵינַיִם, וּמִגְדָּל וְרֹאשׁוֹ בַּשָּׁמַיִם. /

וַיִּנְשֶׁה בְיָפְיָהּ אֶת בֵּית אָבִיו וַיִּשְׁכַּח – / וְשִׂמַּח אֶת

אִשְׁתּוֹ אֲשֶׁר לָקַח. / וַתַּהַר הָאִשָּׁה וַתֵּלֶד בֵּן וַתֹּאמַר

״יֻלַּד בֵּן לְנָעֳמִי /לִהְיוֹת נָגִיד עַל עַמִּי״. / וַיִּקְרָא שְׁמוֹ

עוֹבֵד לְהַרְאוֹת מַעֲלָתוֹ – / וְהָיָה בִכְזִיב בְּלִדְתָּהּ אוֹתוֹ! /

וַיִּגְדַּל הַיֶּלֶד וַיִּגָּמַל / וְאָבִיו חֶסֶד לְכָל בְּנֵי

הַמַּלְכוּת גָּמַל, / וַיַּעַשׂ מִשְׁתֶּה וַהֲנָחָה לַמְּדִינוֹת /

וּמִשְׁלוֹחַ מָנוֹת.

וַיְהִי הַיּוֹם וַיַּעֲמֹד הָאִישׁ בְּפֶתַח אָהֳלוֹ /

בַּחֲצַר הַכָּבוֹד שֶׁלּוֹ / וְהָיָה מִשְׁתַּעֲשֵׁעַ בְּסֵפֶר אֶחָד

נִמְרָץ, / וְהַנַּעַר רָץ / וַיִּשְׁחֶה אֶל הַבּוֹר וַיִּשְׁפַּל /

יִדְכֶּה יָשֹׂחַ וְנָפַל. / וַיִּרְגַּז הָאִישׁ לְקוֹל הַיֶּלֶד / כִּי

נָסַע מֵאִתּוֹ אֶל מְתֵי חֶלֶד / וַיֹּאמַר ״מִי יִתֵּן מוּתִי אֲנִי

תַחְתֶּיךָ / בְּבוֹא עִתֶּךָ!״ / וַיְמַהֵר וַיֵּרֶד אַחֲרָיו אֶל הַבּוֹר

וּזְרוֹעַ עֻזּוֹ נִשְׁבַּר, וַיְבַקֵּשׁ וְלֹא מָצָא הַנַּעַר, / וַיָּנַע

לְבָבוֹ כְּנוֹעַ עֲצֵי הַיַּעַר. / וַיִּפֶן כֹּה וָכֹה / הָלוֹךְ יֵלֵךְ

וּבָכֹה, / וַיַּרְא אֶת הַמְּכַשֵּׁף אֻשְׁפִּיזוֹ / גַּאֲוָתוֹ וְעֻזּוֹ

/ וַיֹּאמֶר לוֹ ״לָמָּה תִבְכֶּה / וְדֶמַע תְּפַכֶּה?״ / וַיֹּאמַר ״כֹּה

קְרָאַנִי הַיּוֹם רְעָדָה וָפַחַד / לְהַשְׁמִיד אוֹתִי וְאֶת בְּנִי

יַחַד״. / וַיֹּאמַר ״וּמֵאַיִן לְךָ בֵּן? / אַל תִּהְיֶה כְּסוּס

וּכְפֶרֶד אֵין הָבֵן, / כִּי כָל מַעֲשֶׂיךָ הֶבֶל / צִירִים וָחֵבֶל!״ /

וַיֹּאמֶר הָאִישׁ ״הֲלֹא אָנֹכִי גוֹדֵר פָּרֶץ / נוֹרָא לְמַלְכֵי אָרֶץ:

/ כִּי בַת מֶלֶךְ גָּדוֹל נָשָׂאתִי / לִי כֹפֶר מָצָאתִי, / וַיְהִי לִי

מִמֶּנָּה בֵּן גֵּאֶה / יְפֵה תֹאַר וִיפֵה מַרְאֶה / וַיִּגָּמֵל

הַיֶּלֶד וַיְהִי רוֹכֵב וְחוֹנֶה / עַל מִשְׁעֶנֶת הַקָּנֶה, / וַיִּפֹּל

אֶל הַטִּיט וְאֶל הַיָּוֵן נֶחְבַּר / אֶל הַבּוֹר הַזֶּה אֲשֶׁר

בַּמִּדְבָּר, / וְעַד לָא מְטָה לְאַרְעִית גֻּבָּא / מֵת אוֹ נִשְׁבַּר

אוֹ נִשְׁבָּה. / וְעַל זֶה בָכִיתִי לִקְשֵׁה יוֹמִי / אָרִיד בְּשִׂיחִי

וְנָאֳמִי: / הִנֵּה אָנֹכִי בְּשִׁבְרִי תוֹעֶה / כַּיּוֹלֵדָה אֶפְעֶה /

כְּעָלֶה נִדָּף אֶבֹּלָה, / כִּי אֵרֵד אֶל בְּנִי אָבֵל שְׁאוֹלָה״. /

וַיֹּאמֶר לוֹ הַמִּצְרִי ״אַל תֵּעָצֵב! / קוּם וְהִתְיַצֵּב / וְשׁוּב

אֶל לִבְּךָ וְתָנוּחַ – / שֶׁגַּם זֶה הֶבֶל וְרַעְיוֹן רוּחַ: / וְכָל

הָעִנְיָן אֲשֶׁר סִפַּרְתָּ / וְהַיָּמִים אֲשֶׁר עָבַרְתָּ / וְכָל

הַמִּקְרֶה הַגָּדוֹל וְהַפֶּגַע – / לֹא הָיָה אֶלָּא כְרֶגַע. / וְהִנֵּה

בְיָדִי הַכּוֹס אֲשֶׁר שָׁתִיתָ / וְהַתַּרְעֵלָה אֲשֶׁר מָצִיתָ, /

וּבַעֲבוּר תָּבִין חָכְמָתִי וְתַגִּידֶנָּה / לְמַעַן הַרְאוֹתְךָ

הוּבָאתָ הֵנָּה!״


וַיִרְגַּז הָאִישׁ מֵאִמְרָתוֹ וְנִיבוֹ / יָשֹׁב

יִתְפַּלָּא בוֹ / וַיֹּאמֶר לוֹ ״נַעֲנֵיתִי לְךָ, כִּי אֵין כָּמוֹךָ

אִישׁ מְכַשֵּׁף / אֲשֶׁר רוּחַ חוֹבֵר נוֹשֵׁף / וְאֵין דּוֹמֶה לְךָ

בְּכָל תַּעֲלוּמוֹת נִרְמָזוֹת / אַחֲרֵי הוֹדִיעַ יְיָ אוֹתְךָ כָּל

זֹאת. / וּלְךָ מִשְׁפַּט הַבְּכוֹרָה / וּבְיָדְךָ כֹּחַ וּגְבוּרָה /

לְהַרְאוֹת עִנְיָנִים / כְּנַף רְנָנִים – / בְּמוֹשַׁב חֶפְצְךָ

יִשְׁכְּנוּ וְיִחְיוּ, / וְשָׁתוּ וְלָעוּ – וְהָיוּ כְּלֹא הָיוּ. /

וְעַתָּה אֶעֱבָדְךָ כְּפִי כֹחִי / וּלְכָל בָּהֶם חַיֵּי רוּחִי /

וְתַחֲיֵנִי וְתַחֲלִימֵנִי, / וְעַתָּה אַל תַּכְלִימֵנִי / הַעֲמִידֵנִי

עַל אֱמוּנַת עִתֶּךָ / אֶתְהַלֵּךְ בַּאֲמִתֶּךָ״. / וַיֵּשֶׁב עִמּוֹ

יָמִים אֲחָדִים / לְלַמְּדוֹ סֵפֶר וּלְשׁוֹן כַּשְׂדִּים.
