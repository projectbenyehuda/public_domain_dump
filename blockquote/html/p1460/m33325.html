לְהָבִין מָשָׁל וּמְלִיצָה

לְהָבִין מָשָׁל וּמְלִיצָה דִּבְרֵי חֲכָמִים וְחִידוֹתָם / לָדַעַת מוּסָר נְבִיאֵי הַשִּׁיר וְתוֹכַחְתָּם / חִפֵּשׂ הַמְבַקֵּשׁ מְשׁוֹרֵר אֲשֶׁר לוֹ בַּשִּׁיר עֶשֶׂר יָדוֹת / מְנַצֵּחַ בִּנְגִינוֹת וּמַשְׂכִּיל שִׁיר יְדִידוֹת / וְלוֹמֵד מִנְּעוּרָיו לְדַבֵּר בְּנִכְבַּדֵּי אֶרֶץ נִכְבָּדוֹת. / וַיִּמְצָא מְשׁוֹרֵר אֲשֶׁר הַמְּלִיצָה הַיְקָרָה בְּפִיו וְנִכְבָּד נִיבוֹ / וַחֲזוֹן הַשִּׁיר נִפְרָץ בִּלְבָבוֹ, / אִישׁ דְּבָרִים וְרַב מַהֲלָלִים / אִזֵּן וְחִקֵּר תִּקֵּן מְשָׁלִים, / נַהֲרֵי נַחֲלֵי דְבַשׁ וְחֶמְאָה בְּזִמְרוֹתָיו, / נֹפֶת תִּטֹּפְנָה שִׂפְתּוֹתָיו / וְצוּף יָזוּב מִשִּׁירוֹתָיו, / יִמָּלֵא הַפֶּה שְחוֹק וְרִנָּה בְּאָמְרוֹ אֲמָרָיו / וְצוּר יָצוּק פַּלְגֵי שֶׁמֶן בְּדַבְּרוֹ דְבָרָיו. / בְּשׁוֹרֲרוֹ שִׁירָיו וְעַל פִּיו יַעֲלֵם / אָזיְדַלֵּג כָּאַיָּל פִּסֵּחַ וְתָרֹן לְשׁוֹן אִלֵּם, / יְכַפֵּר כָּל עָווֹן וְכָל חַטָּאת וְאִם מִנְּשֹׂא גְּדוֹלָה וּכְבֵדָה / בְּאָמְרוֹ מִזְמוֹר לְתוֹדָה. / בְּחֵן שְׂפָתָיו יָצוּד הַנְּשָׁמוֹת / וְיִמְשֹׁךְ לוֹ בְנֹעַם מִלָּיו שָׂרֵי אֲדָמוֹת, / יִתְּנוּ לוֹ מִפְּרִי אֲמָרָיו וּבַשְּׁעָרִים יְהַלֲלוּהוּ, / רֵיחַ שְׁמָנָיו טוֹבִים, שֶׁמֶן תּוּרַק שְׁמוֹ – עַל כֵּן עֲלָמוֹת אֲהֵבוּהוּ. / בְּשִׁירוֹתָיו יְנַשֵּׂא לְמַעְלָה נְדִיבֵי עַם וְשׁוֹעֵיהֶם / וְיַשְׁפִּיל לְמַטָּה בְנֵי בְלִיַּעַל וְהָעוֹשִׂים כְּמַעֲשֵׂיהֶם.


מְשׁוֹרֵר מַעֲשָׂיו תֵּמַהּ לְרוֹאִים, / לְמַעַן יַעֲשֶׂה בַשִּׁיר פְּלָאִים

בְּהַגְבִּיהוֹ עֲדֵי מָרוֹם שְׁפָלִים / וְהַשְׁפִּילוֹ עֲדֵי אֶרֶץ לְגֵאִים.


אָמַר לוֹ הַמְבַקֵּשׁ ״אָדָם הַמְשׁוֹרֵר, / אֲשֶׁר בְּחָכְמַת שִׁירָיו יְשֵׁנִים יְעוֹרֵר, / כִּתְמֵי הַתּוּגוֹת בִּמְלִיצוֹתָיו יַלְבֵּן וִיבָרֵר / וְכָל אִמְרוֹתָיו אֲמָרוֹת טְהוֹרוֹת / וְהוּא נְעִים זְמִירוֹת! / צוֹפֶה הָיִיתִי בְשִׁירוֹתֶיךָ אֲשֶׁר מִפְּנִינֵי הַצַּחוֹת חֲצוּבוֹת, / וְהִסְתַּכַּלְתִּי בִזְמִירוֹתֶיךָ – וְהֵם אוֹר זָרוּחַ וּבֹקֶר לֹא עָבוֹת, / וְדִבְרוֹתֶיךָ מִצּוּף דְּבַשׁ עֲרֵבוֹת / וְשִׁירוֹתֶיךָ רְאוּיוֹת לְהַלֵּל בָּהֶן לְרוֹכֵב עֲרָבוֹת / וְכָל דְּבָרֶיךָ בְּמִשְׁפָּט תְּכַלְכֵּל / לְהָבִין אִמְרֵי בִינָה וְלָדַעַת מוּסַר הַשְׂכֵּל. / וְאַחֲרֵי שֶׁחֲנָנְךָ יְיָ דַּעַת / חוֹבָה עָלֶיךָ לְהִתְעַסֵּק בַּחָכְמוֹת הָאֲמִתִּיּוֹת וְסִתְרֵיהֶן לָדַעַת, / כִּי הֵן צוּרַת הָאָדָם וְצַלְמוֹ / וּבַשֵּׂכֶל נִבְדָּל הָאָדָם עַל כָּל נִבְרְאֵי מַטָּה וַייָ יִהְיֶה עִמּוֹ. / וְאֵינוֹ מֵהַנָּכוֹן לְבַלּוֹת הַיָּמִים בְּהַבְלֵי הַמְשׁוֹרֲרִים / אֲשֶׁר פִּיהֶם דִּבֶּר שָׁוְא וּלְשׁוֹנָם תֶּהְגֶּה שְׁקָרִים / וְלַטּוֹב רַע וְלָרַע טוֹב אוֹמְרִים / הַבּוֹנִים עַל יְסוֹד הַשָּׁוְא שִׁירֵיהֶם / וַתְּהִי הָאֱמֶת נֶעְדֶּרֶת בְּדִבְרֵיהֶם, / הַנּוֹשְׁכִים בִּלְשׁוֹנָם וְחִקְקֵי אָוֶן חוֹקֲקִים / וּבְכִזְבֵי הַדִּמְיוֹן מַחֲזִיקִים / וַיִּתְלַקְּטוּ לָהֶם אֲנָשִׁים רֵיקִים. / וְצָדְקוּ בְאָמְרָם הַמּוֹשְׁלִים / כִּי הַחָכְמָה כִּפְרִי וְהַשִּׁיר כְּעָלִים. / וְאָמַר הֶחָכָם: הַמַּצְלִיחַ מְאֹד מִבְּנֵי אָדָם מִי שֶׁמִּתְעַסֵּק מִנְּעוּרָיו בְּחָכְמַת הַמּוֹפֵת לֹא בְהַבְלֵי הַמְשׁוֹרֲרִים. וְאָמַרְתִּי אֲנִי:


כַּלֵּה יְמוֹתֶיךָ בְּחַדֵּשׁ סוֹד / חָכְמָה וְדַעַת – לֹא בְדִבְרֵי שִׁיר,

וַעְמֹד לְהִתְחַכֵּם, וְלָדַעַת / דַּעַת קְדוֹשִׁים דַּרְכְּךָ תַיְשִׁיר.


רָאִיתִי וְהִנֵּה הֲפַכְפַּךְ דֶּרֶךְ הַמְשׁוֹרֵר וָזָר / נָכְרִי לְחַכְמֵי לֵב וּלְאַנְשֵׁי אֱמֶת מוּזָר, / כִּי לְכִילַי יֹאמַר שׁוֹעַ וְנָדִיב אֶל נְבָלִים / דְּבַר בְּלִיַּעַל יָצוּק בּוֹ וְעִמּוֹ הֲתוּלִים, / זֶה דַרְכּוֹ כֵּסֶל לוֹ וְחֵמָה / וְשָׁוְא וְדִבְרֵי כָזָב בְּשִׁירוֹתָיו וּמְזִמָּה, / וְעַל יְסוֹד הַשֶּׁקֶר בּוֹנֶה בַיִת אָפֵל / וְטָח קִירוֹתָיו תָּפֵל / וִיצַפֶּה אוֹתָם סִיגִים מִחוּץ וּמִבַּיִת / נוֹטֵעַ כֶּרֶם וְהִנֵּה עָלָה כֻלּוֹ קִמְּשׁוֹנִים כָּסוּ פָנָיו חֲרֻלִּים וְעָלָה שָׁמִיר וָשַׁיִת, / וְהַמְבֻקָּשׁ מִדְּבָרָיו וְהַנִּדְרָשׁ / לִהְיוֹת בָּהֶם הַכָּזָב מְפֹרָשׁ. / אוֹמְרִים אָמוֹר שִׁיר עֲגָבִים, / וּפוֹנִים אֶל רְהָבִים וּמִלְאוּ בָתֵיהֶם



שֶׁקֶר, וְשָׂטֵי כְזָבִים. / חֹמֶר הַבָּתִים גָּרוּעַ וְשָׁפֵל – / מָאֲסוּ הַבּוֹנִים בִּיסוֹד הָאֱמֶת וְהִנָּם טָחִים אוֹתוֹ תָּפֵל, / וְהוּא מֵחֹמֶר הַהֶקֵּשׁ הַחֲמִישִׁי הַגָּרוּעַ וְהָאָפֵל – / אֵין נֹגַהּ לוֹ כִּי אִם חֹשֶׁךְ עָנָן וַעֲרָפֶל. / וּבַעֲבוּר כִּי חָכְמַת הַשִּׁיר מֵהָאֱמֶת רְחוֹקָה / וְרוּחַ שֶׁקֶר בְּפִי נְבִיאֶיהָ וּמֵהָאֱמֶת נְעוּרָה וְרֵיקָה – / לֹא יְשַׁמְּשׁוּ בְעָלֶיהָ אֶלָּא בַשֵּׁמוֹת הַמֻּשְׁאָלִים וְהַמָּעְתָּקִים / אֲשֶׁר הֵם מֵהוֹרוֹת עַל הָאֱמֶת רְחוֹקִים, / וְלֹא יְשַׁמְּשׁוּ בַשֵּׁמוֹת הַמֻּסְכָּמִים / אֲשֶׁר יִשְׁמְרֵם פִּי חֲכָמִים.

וְכִשְׁמֹעַ הַמְשׁוֹרֵַר מִלָּיו גָּעַר בּוֹ גְּעָרָה / וְחַם לִבּוֹ בְּקִרְבּוֹ וּבַהֲגִיגוֹ אֵשׁ בָּעֲרָה / וַיֹּאמַר ״צָדַק שֶׁהִקְדִּימַנִי בִמְשָׁלוֹ / כִּי מִי שֶׁלֹּא יַשְׂכִּיל הַדָּבָר הוּא שׂוֹנֵא לוֹ, / וְלוּלֵי שֶׁאֲנִי נוֹשֵׂא פְנֵי הַמּוּסָר – בַּהֲבֵל פִּי שְׂרַפְתִּיךָ / וְלַחֲרָפוֹת לְדִרְאוֹן עוֹלָם שַׂמְתִּיךָ, / כִּי חֵרַפְתָּ מַעַרְכוֹת הַמְשׁוֹרֲרִים / אֲשֶׁר כָּל דִּבְרֵיהֶם עַל אָפְנֵיהֶם דְּבוּרִים, / וְזֶה דַרְכָּם לְהַזְהִיר הָאָדָם לִקְרוֹת מוּסָר וְחָכְמָה / וּלְבַקֵּשׁ סְגֻלַּת הַנֶּפֶשׁ הַחֲכָמָה וְכָל מַעֲשֵׂה רָמָה / וּלְשַׁבֵּחַ מַעֲשֵׂה הַטּוֹבִים / וּלְהַלֵּל וּלְאַשֵּׁר מַעַלְלֵי הַנְּדִיבִים / וּלְנַבֵּל וּלְגַנּוֹת בְּנֵי בְלִיַּעַל / הָאוֹמְרִים לָאֵל ״סוּר מִמֶּנּוּ!״ וְיִשְׁעוּ לְבָמוֹת בַּעַל, / וּלְפַרְסֵם הַחֲנֵפִים אֲשֶׁר בְּיוֹצְרָם מָעֲלוּ מַעַל / וּמַעֲשֵׂי הָרָעִים לְהוֹדִיעַ וּלְהַלְשִׁין / וּכְמוֹ שֶׁאָמְרוּ חֲכָמִים זִכְרוֹנָם לִבְרָכָה: הַאי מַאן דִּסְנֵי שֻׁמְעָנֵהּ שְׁרֵי לְבַזּוֹיֵהּ בְּגִימֶ״ל וְשִׁי״ן.


אֱמֹר אֵיךְ תַּעֲרֹךְ אִישׁ אֶל מְשׁוֹרֵר? / וְהוּא הַלֵּב יְשַׂמַּח עֵת יְשׁוֹרֵר

וְיַמְתִּיק עִם חֲבֵרָיו סוֹד וְרֵעָיו / וְחַיֵּי הַמְקַנְאִים בָּם יְמָרֵר

וְלִרְדֹּף הוֹד מְיֻדָּעָיו יְעוּרוּן / וְלִדְרֹשׁ כָּל יְקָר אוֹתָם יְעוֹרֵר.

וְיָשִׂים מַחֲשַׁכֵּי הַדְּבָרִים / בְּאוֹר, וּדְבַר שְׂפָתַיִם יְבָרֵר,

וְרֹאשׁ תַּנִּין יְלִידֵי יוֹם יְשַׁבֵּר / בְּעֹז שִׁירָיו, וְיָם תּוּגוֹת יְפוֹרֵר,

וְיָנִיחַ בְּמִלָּיו אוֹהֲבֵי תֹם / וְהַיֹּשֶׁר, וְשׂוֹנְאֵיהֶם יְצוֹרֵר,

יְבָרֵךְ מַעֲשֵׂה נָדִיב וּפָעְלוֹ / כְּמוֹ כָל מַעֲשֵׂה כִילַי יְאָרֵר,

יְחָרֵף הַזְּמָן, יַעַן יְעַקֵּשׁ / יְשַׁר לֵבָב וְאֶת דַּרְכּוֹ יְסוֹרֵר


קוּם, הַמְבַקֵּשׁ, וּשְׁמַע וְעָדַי הַאֲזִינָה, / וְלֵץ, שְׁמַע גְּעָרָה, וּפֶתִי, חָכְמָה הָבִינָה! / רְאֵה כִּי בַשִּׁיר אוֹמְרִים תְּהִלּוֹת לָאֵל / אֲשֶׁר שׁוֹרֲרוּ אֲנָשִׁים מִלְּפָנִים מֹשֶׁה וּבְנֵי יִשְׂרָאֵל, / וְשִׁירַת ״הַאֲזִינוּ״ שֶׁהִיא כוֹלֶלֶת כָּל עִקְּרֵי הַתּוֹרָה / כֻּלָּהּ עַל דֶּרֶךְ הַשִּׁיר אֲמוּרָה. / וּמֵהַמְשׁוֹרֲרִים הָיוּ אֲנָשִׁים צַדִּיקִים וִישָׁרִים / בְּרוּחַ הַקֹּדֶשׁ מְדַבְּרִים, / וְסֵפֶר תְּהִלִּים בְּרוּחַ הַקֹּדֶשׁ כָּתוּב וְכֻלּוֹ מִזְמוֹרִים / וַחֲכָמֵינוּ קָרְאוּ קֹדֶשׁ הַקֳּדָשִׁים לְ״שִׁיר הַשִּׁירִים״, / יַעַן בּוֹ נֶאֱמַר דְּבֵקוּת הַנֶּפֶשׁ בִּמְקוֹרָהּ / וְאַהֲבָתָהּ וְחִשְׁקָהּ אֶל יוֹצְרָהּ. / וְדֶרֶךְ הַשִּׁיר לְהָעִיר הָאַהֲבָה וּלְהַמְתִּיק סוֹד הָאוֹהֲבִים / וּלְפַרְסֵם הַשִּׂנְאָה וְאֵיבַת הָאוֹיְבִים / וְהַחֲכָמִים הַקַּדְמוֹנִים אָמְרוּ כִּי הֶפְסֵד הַיְצוּרִים וַהֲוָיָתָם – / הַשִּׂנְאָה וְהָאַהֲבָה סִבָּתָם: / כִּי בְשִׂנְאָה יִתְפָּרְדוּ וְזֶה מֵעַל זֶה יִרְחָקוּ / וּבְאַהֲבָה יִקְרְבוּ זֶה לָזֶה וְאִישׁ בְּרֵעֵהוּ יְדֻבָּקוּ, / וּמִטֶּבַע הַמְשׁוֹרֵר לִרְדֹּף אַחֲרֵי הַנָּאוֹת בְּדִבּוּרוֹ וּבְפָעֳלוֹ / וּשְׂפַת יֶתֶר נָאוָה לוֹ. / וְהַשִּׁיר הַטּוֹב – רְאָיָה כִּי טֶבַע זַךְ טֶבַע אוֹמְרוֹ, / וְשֶׁהוּא מִגְּדוֹלֵי הַנֶּפֶשׁ וַאֲשֶׁר הַמְּלָכִים חֲפֵצִים בִּיקָרוֹ. / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


בַּשִּׁיר בְּעֵת יִפֹּל לְנַפְשְׁךָ חֶבֶל – / רַבִּים יְהוֹדוּךָ וְשָרֵי תֵבֵל.

תִּתֵּן אֲלֵיהֶם אֶת רְעוּת רוּחַ וְהֵם / לָךְ יִתְּנוּ כַסְפָּם – וְגַם זֶה הֶבֶל!



וְזָכְרוּ כִּי בְאֻמָּה מִן הָאֻמּוֹת / מֶלֶךְ רַב וְאוֹזֵר חַיִל וְתַעֲצוּמוֹת, / הֶעֱמִיד בְּמִשְׁפָּט אָרֶץ / וַיִּפְרֹץ בְּאוֹיְבָיו פָּרֶץ, / וְהָרְשָׁעִים לֹא שָׁכְנוּ אַרְצוֹ וַיִּתְנֵם לְכָלָה וָחָרֶץ. / וַיְהִי כַשֶּׁמֶשׁ הוֹדוֹ, / יְרֵאוּהוּ אֲנָשִׁים וְלֹא מָחַל עַל כְּבוֹדוֹ, / וּבְתָקְפּוֹ וּבִגְבוּרָתוֹ / נִשֵּׂאת לְמַעְלָה מַלְכוּתוֹ, / בֶּן־חֹרִין וְחָכָם בְּכָל חָכְמָה / וּבְתַחְבּוּלוֹת עוֹשֶׂה לוֹ הַמִּלְחָמָה, / גִּבּוֹר חַיִל נוֹדְעוּ בָעַמִּים פְּעָלָיו – / וַאֲנִי הוּא הָאוֹמֵר עָלָיו:


לֹא עָצְרו כֹחַ לְנֶגְדּוֹ לוֹחֲמָיו, / אַךְ נֶחְבְּאוּ כַצֹּאן [לְמוּל] טוֹבֵחַ.

לֹא יֶחֱרַד בִּרְבוֹת חֲיָלֵיהֶם – וְאֵיךְ / בִּרְבוֹת תְּיָשִׁים יֶחֱרַד זוֹבֵחַ?


וְהוֹלִיד בֵּן וְאֵין בְּיָדוֹ מְאוּמָה מִמִּדּוֹת אָבִיו הַמְעֻלּוֹת / וְנוֹלַד רָשׁ מִקִּנְיְנֵי מַעֲלוֹתָיו הַגְּדוֹלוֹת. / וַיִּתְבּוֹנֵן אָבִיו בִּדְרָכָיו / וְהִבִּיט אָוֶן בּוֹ וְלֹא יָשְׁרוּ בְעֵינָיו מַהֲלָכָיו / כִּי אֵינָם כְּדָת הַמְּלָכִים וְכַהֲלָכָה / וְהוּא נַעַר וָרַךְ וְלֹא יִצְלַח לִמְלוּכָה. / וְהַמֶּלֶךְ נֶעְצַב אֶל עַמּוֹ, / בִּרְאוֹתוֹ כִּי אֵין מְמַלֵּא מְקוֹמוֹ: / כִּי בִהְיוֹת הַמֶּלֶךְ מְכַלְכֵּל בְּמִשְׁפָּט דְּבָרָיו / וְהוּא חָכָם וּמֵבִין לַאֲשׁוּרָיו – / יִשְׂמַח כָּל עַם הָאָרֶץ וְתִשְׁקֹט כָּל עִיר, / וְאִם הוּא סָכָל – יֵאָנַח עָם וְקוֹל מְהוּמוֹת בְּאַרְצוֹ וּבַלָּהוֹת בָּעִיר. / וּבַחֲלוֹם הַלֵּב יִדְווּ כָל הָאֲבָרִים וּתְהִי אֲנוּשָׁה מַכָּתָם – / וְכֵן הָעָם לְפִי שֵׂכֶל מַלְכָּם תִּהְיֶה הַצְלָחָתָם. / וְאָמַר שְׁלֹמֹה ״מֶלֶךְ בְּמִשְׁפָּט יַעֲמִיד אָרֶץ״, / וְאָמַר הֶחָכָם: כְּשֶׁיִרְצֶה הַשֵּׁם לְהֵיטִיב לְאֻמָּה יָשִׂים הַשֵּׂכֶל בִּמְלָכֶיהָ / וְהַמַּלְכוּת בַּחֲכָמֶיהָ. / וְאָמַרְתִּי אֲנִי:


בִּרְצוֹת הָאֵל דַּרְכֵי הָעָם / יָשִׂים הַשֵּׂכֶל בִּמְלָכָיו,

כִּי טוּב עַם וָעָם יִהְיֶה עַל / פִּי שֵׂכֶל שָׂרָיו וּנְסִיכָיו.


וַיִּשְׁלַח הַמֶּלֶךְ לַחֲכָמִים יוֹדְעֵי הָעִתִּים לְלַמְּדוֹ חֻקֵּי הַמַּלְכוּת וּלְהַדְרִיכוֹ בָהֶם / וּלְהוֹרוֹת אֵלָיו מוּסַר הַמְּלָכִים וְהַנְהֲגוֹתֵיהֶם, / וְיִסְּרוּ אוֹתוֹ – וְלֹא הָיָה שׁוֹמֵעַ אֲלֵיהֶם / וְהָיָה הוֹלֵךְ בִּשְׁרִירוּת לִבּוֹ וַחֲבוּר עֲצַבִּים. / וַיִּתְאַבֵּל הַמֶּלֶךְ עַל בְּנוֹ יָמִים רַבִּים / עַד שֶׁנּוֹאַשׁ מִמְּצֹא מַרְפֵּא לְמַחֲלַת נַפְשׁוֹ / וַיְשַׁלְּחֵהוּ מֵעָלָיו וְגֵרְשׁוֹ.

וּלְקֵץ יָמִים בָּא מְשׁוֹרֵר גָּדוֹל בְּמַלְכוּתוֹ / מִקַָּצְוֵי אֶרֶץ אֲשֶׁר אֵין כְּמוֹתוֹ, / וַיַּגִּידוּ לוֹ דַאֲגַת הַמֶּלֶךְ עַל סִכְלוּת בְּנוֹ וְרֹעַ הַנְהָגָתוֹ. / וַיָּבוֹא לִפְנֵי הַמֶּלֶךְ וַיִּקֹּד לְפָנָיו וַיִּשְׁתַּחֲוֶה / וַיֹּאמַר ״יְחִי אֲדוֹנִי הַמֶּלֶךְ, וּבוֹרְאוֹ יָחִישָׁה מִפְלָט לוֹ וְכֹל אֲשֶׁר יְאַוֶּה!״ / וַיֹּאמֶר הַמֶּלֶךְ לוֹ ״מַה מְּלַאכְתְּךָ וּמֵאַיִן תָּבוֹא וּמֵאֵיזֶה עַם אַתָּה?״ / וַיֹּאמַר ״עַרְבִי אָנֹכִי וּמְלֶאכֶת הַשִּׁיר מְלַאכְתִּי מֵאָז וְעַד עַתָּה, /

וַאֲנִי בָא מֵהָאִי הַנִּקְרָא אִי הַשִּׁיר, / יַעַן כָּל שׁוֹכֵן בּוֹ יִפְתַּח שְׂפָתָיו בִּזְמִירוֹת וְיָשִׁיר. / וַאֲנִי הֻגֵּד הֻגַּד לִי דַּאֲגָתְךָ וִיגוֹנְךָ עַל אוֹדוֹת בְּנֶךָ, / וְאִם כָּשֵׁר הַדָּבָר בְּעֵינֶיךָ – / יֵשֵׁב הַנַּעַר אִתִּי יָמִים אוֹ עָשׂוֹר / וַאֲנִי אַדְרִיכֵהוּ בְּמוּסַר הַמְּלָכִים וַאֲיַסְּרֵהוּ יַסֹּר. / רוּחַ הַשִּׁיר אֶתֵּן בְּקִרְבּוֹ – / וְאָז תָּבוֹא חָכְמָה בְּלִבּוֹ / וְנָחָה עָלָיו רוּחַ דַּעַת וְחָכְמָה / וְרוּחַ עֵצָה וּגְבוּרָה לַמִּלְחָמָה״. / וַיֹּאמֶר הַמֶּלֶךְ ״קָחֶנּוּ וְעֵינֶיךָ שִׁית עָלָיו / וְאוּלַי בְּלַמֶּדְךָ אוֹתוֹ יִתְנַכֵּר הַנַּעַר בְּמַעֲלָלָיו / וְיִהְיוּ יְשָׁרִים וְזַכִּים פְּעָלָיו, / וּתְהִי מַשְׂכֻּרְתְּךָ שְׁלֵמָה מֵעִמִּי / וְאַתָּה תִהְיֶה עַל בֵּיתִי וְעַל פִּיךָ יִשַּׁק כָּל עַמִּי״.

וַיָּחֶל הַמְשׁוֹרֵר לְפַעֲמוֹ בְּמַחֲנֵה הַמְשׁוֹרֲרִים / וּלְלַמְּדוֹ סֵפֶר הַצַּחוֹת וּלְשׁוֹן הַשִּׁירִים, / וַיְהִי בַדֶּרֶךְ וַיִּמְצָא לַהֲקַת נְבִיאֵי הַשִּׁיר / וְנֶהְפַּךְ לְאִישׁ אַחֵר וַיָּשִׁיר. / אָז אָמְרוּ הָרוֹאִים / ״מַה זֶּה הָיָה לְבֶן־הַמֶּלֶךְ? הֲגַם הוּא בַנְּבִיאִים?״ / וְיְהִי כְנוֹחַ הָרוּחַ עָלָיו / דִּבֶּר נִכְבָּדוֹת וַיִּשָּׂא מְשָׁלָיו:


לִי לֵב יְאַוֶּה לַעֲלוֹת אֶל רֹאשׁ / עָשׁ אוֹ הֱיוֹת נִקְבָּר בְּתַחְתִּית הַר,

נַפְשׁוֹ לְהַשִּׂיג מַעְלָה עָלָיו / תָּקֵל, וְלִרְדֹּף הַיְקָר מִהַר.

כִּי מִי בְעָפְרָה חָשְׁקָה נַפְשׁוֹ / יֵקָר בְּעֵינָיו לוּ יְהִי מֹהַר?


וְרָאָה אָדָם מְיַסֵּר בַּשּׁוֹטִים / לְאֶחָד מֵהַשּׁוֹטִים, / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


בְּזֵד יָהִיר בְּלִי תֵחַת גְּעָרָה – / וְאַךְ כִּי לַיְלָה וָיוֹם תְּשִׂיחוֹ,

וְאַל תַּכֶּה בְשֵׁבֶט גֵּו חֲסַר לֵב / לְיַשֵּׁר מַעֲוֵה לִבּוֹ וְרוּחוֹ,

אֲבָל הַכֵּה וּפָצוֹעַ לְרֹאשׁוֹ, / לְמַעַן כִּי מְקוֹם הַשֵּׁד בְּמֹחוֹ!


אַחֲרֵי כֵן גֵּרָה הַמְשׁוֹרֵר מָדוֹן בֵּינוֹ וּבֵין אֶחָד מֵרֵעָיו / וַיְהִי רִיב בֵּינוֹ וּבֵין אֶחָד מִמְּיֻדָּעָיו. / וַיֹּאמֶר בֶּן־הַמֶּלֶךְ לְהַכּוֹתוֹ בְּחֶרֶב שִׁירָיו / וְיָרָה אֵלָיו חִצֵּי אֲמָרָיו, / וַאֲנִי עִנְיָנָם הֶעְתַּקְתִּי / וּבְזֶה הַסֵּפֶר אוֹתָם חָקַקְתִּי:


אֶשְׁאַל לְאָדָם בַּחֲלוֹם אֶחְזֶה / אוֹתוֹ ״הֲזֶה הָרֵיק וְהַנִּבְזֶה

מִצֶּאֱצָאֶיךָ?״ וְעָנַנִי / ״חַוָּה אֲגָרֵשׁ, אִם אֱמֶת הוּא זֶה!״


וַיֵּלֶךְ בֶּן־הַמֶּלֶךְ הָלוֹךְ וְגָדוֹל וַייָ עִמּוֹ / וְנִשְׁמַע שְׁמוֹ בַגּוֹיִים וַיִּגְדַּל שְׁמוֹ, / וַיִּשְׁלַח הַמֶּלֶךְ בַּעֲדוֹ / וַיָּבוֹא אֵלָיו עִם הַמְשׁוֹרֵר אֲשֶׁר לִמְּדוֹ. / וְכִרְאוֹת הַמֶּלֶךְ פָּנָיו אָמַר ״הִתְנַכֵּר הַנַּעַר הַזֶּה בְּמַעֲלָלָיו / וְהַכָּרַת פָּנָיו עָנְתָה בּוֹ כִּי צָלְחָה רוּחַ אֱלֹהִים עָלָיו״. / וַיַּגֵּד לוֹ הַמְשׁוֹרֵר כֹּל אֲשֶׁר קָרָה לוֹ / וַיִּתֵּן לוֹ כְּכָל חֶפְצוֹ וּמִשְׁאָלוֹ.
