תִּתְבָּרֵךְ אֱלֹהֵי הָאֱלֹהִים

בקשה (?)


תִּתְבָּרֵךְ אֱלֹהֵי הָאֱלֹהִים וַאֲדֹנֵי הָאֲדֹנִים

וְתִשְׁתַּבַּח בְּפִי עֶלְיוֹנִים וְתַחְתּוֹנִים,

כִּי אֵין כָּמוֹךָ שׁוֹכֵן מְעוֹנִים –

בא״י שׁוֹמֵעַ קוֹל תַּחֲנוּנִים


תִּתְבָּרֵךְ בּוֹרֵא הָעוֹלָם בְּדָעַת

וְתִשְׁתַּבַּח מֵבִין לְאָדָם דָּעַת,

כִּי אֵין כָּמוֹךָ מֵבִין כָּל דָּעַת –

בא״י הַמְלַמֵּד לָאָדָם דָּעַת.


תִּתְבָּרֵךְ גְּדוֹל הָעֵצָה וְרַב הָעֲלִילִיּוֹת

וְתִשְׁתַּבַּח בְּפִי כָּל בְּרִיּוֹת,

כִּי אֵין כָּמוֹךָ בֹּחֵן לְבָבוֹת וּכְלָיוֹת –

בא״י הַמְקָרֶה בַמַּיִם עֲלִיּוֹת.


תִּתְבָּרֵךְ דָּר מְרוֹמִים

וְתִשְׁתַּבַּח בַּעַל הָרַחֲמִים,

כִּי אֵין כָּמוֹךָ אֲבִי יְתוֹמִים –

בא״י אֵל חַי הָעוֹלָמִים.


תִּתְבָּרֵךְ הָאֵל אֲדֹנָי

וְתִשְׁתַּבַּח בְּפִי הֲמוֹנָי,

כִּי אֵין כָּמוֹךָ שׁוֹכֵן מְעוֹנָי –

בא״י שׁוֹמֵעַ קוֹל תַּחֲנוּנָי.


תִּתְבָּרֵךְ וְאֵין דּוֹמֶה לְּךָ נוֹרָא עֲלִילוֹת

וְתִשְׁתַּבַּח יוֹשֵׁב תְּהִלּוֹת,

כִּי אֵין כָּמוֹךָ שׁעוֹשֶׂה גְדוֹלוֹת –

בא״י גִּבּוֹר בַּמַּעֲלוֹת.


תִּתְבָּרֵךְ זוֹכֵר הַבְּרִית, מַזְכִּיר הַנִּשְׁכָּחוֹת

וְתִשְׁתַּבַּח אֱלֹהֵי הָרוּחוֹת,

כִּי אֵין כָּמוֹךָ דּוֹלֶה מִמַּעֲמַקִּים וּמַעֲלֶה מִשִּׁיחוֹת –

בא״י הַמְּהֻלָּל בַּתִּשְׁבָּחוֹת.


תִּתְבָּרֵךְ חֲסִין יָהּ, אֱלֹהִים רוֹכֵב גְּבֹהִים

וְתִשְׁתַּבַּח צוּר יוֹצֵר נְגוֹהִים,

כִּי אֵין כָּמוֹךָ נוֹרָא עַל כָּל אֱלֹהִים –

בא״י אֲדֹנֵי הָאֲדֹנִים וֵאלֹהֵי הָאֱלֹהִים.


תִּתְבָּרֵךְ טָהוֹר וְקָדוֹשׁ, בָּהִיר בַּשְּׁחָקִים

וְתִשְׁתַּבַּח בְּפִי עוֹלְלִים וְיוֹנְקִים,

כִּי אֵין כָּמוֹךָ דּוֹבֵר וּמֵקִים –

בא״י עֹשֶׂה מִשְׁפָּט לַעֲשׁוּקִים.


תִּתְבָּרֵךְ יֹשֵׁב הַכְּרֻבִים

וְתִשְׁתַּבַּח מוֹנֶה מִסְפָּר לַכּוֹכָבִים,

כִּי אֵין כָּמוֹךָ מְכַסֶּה שָׁמַיִם בְּעָבִים –

בא״י נֹתֵן לֶחֶם לָרְעֵבִים.


תִּתְבָּרֵךְ כֹּנֵס כַּנֵּד מֵי הַיָּם

וְתִשְׁתַּבַּח מוֹשֵׁל בְּגֵאוּת הַיָּם,

כִּי אֵין כָּמוֹךָ דּוֹרֵךְ עַל בָּמֳתֵי יָם –

בא״י אֵל חַי וְקַיָּם.


תִּתְבָּרֵךְ לוֹבֵשׁ צְדָקוֹת,

וְתִשְׁתַּבַּח מְגַלֶּה עֲמֻקוֹת,

כִּי אֵין כָּמוֹךָ שׁוֹמֵעַ צְעָקוֹת –

בא״י מַאֲזִין נְאָקוֹת.


תִּתְבָּרֵךְ מֵמִית וּמְחַיֶּה

וְתִשְׁתַּבַּח אֶהְיֶה אֲשֶׁר אֶהְיֶה,

כִּי אֵין כָּמוֹךָ יוֹדֵעַ מַה שֶּׁיִּהְיֶה –

בא״י שֶׁבָּרָאתָ פַּת כָּל יְצוּר וְיִחְיֶה.


תִּתְבָּרֵךְ נֶאְזָר בִּגְבוּרָה

וְתִשְׁתַּבַּח לוֹבֵשׁ עֹז, וְתִפְאָרָה,

כִּי אֵין כָּמוֹךָ עֶלְיוֹן וְנוֹרָא –

בא״י עוֹטֶה כַּשַּׂלְמָה אוֹרָה.


תִּתְבָּרֵךְ שַׂגִּיא כֹחַ

וְתִשְׁתַּבַּח נוֹתֵן לִבְרוּאָיו כֹּחַ,

כִּי אֵין כָּמוֹךָ יוֹצֵר הָרִים וּבֹרֵא רוּחַ –

בא״י עוֹשֶׂה מִשְׁקַָל לָרוּחַ.


תִּתְבָּרֵךְ עוֹזֵר לָאִלְּמִים וּמַתִּיר כְּבָלִים

וְתִשְׁתַּבַּח מָעוֹז לַדַּלִּים,

כִּי אֵין כָּמוֹךָ סוֹמֵךְ נוֹפְלִים –

בא״י מַתִּיר אֲסוּרִים וְרֹפֵא חֹלִים.


תִּתְבָּרֵךְ פּוֹדֶה וּמַצִּיל, עֶזְרָה בְצָרוֹת

וְתִשְׁתַּבַּח מעוֹנֶה בְּכָל עֵת עֲתִירוֹת,

כִּי אֵין כָּמוֹךָ פּוֹעֵל גְּבוּרוֹת –

בא״י אֵל קֹרֵא הַדֹּרוֹת.


תִּתְבָּרֵךְ צַדִּיק וְחָסִיד בְּכָל פְּעֻלָּה

וְתִשְׁתַּבַּח מַרְבֶּה, סְלִיחָה וּמְחִילָה,

כִּי אֵין כָּמוֹךָ מְרוֹמָם עַל כָּל בְּרָכָה וּתְהִלָּה –

בא״י כִּי לְךָ נָאוָה הַגְּדֻלָּה.


תִּתְבָּרֵךְ קָדוֹשׁ, אֱלֹהֵי הַצְּבָאוֹת

וְתִשְׁתַּבַּח מַפְלִיא פְלָאוֹת,

כִּי אֵין כָּמוֹךָ אֱלוֹהַּ נוֹרָאוֹת –

בא״י עוֹשֶׂה גְדוֹלוֹת וְנִפְלָאוֹת.


תִּתְבָּרֵךְ רָם וְנִשָּׂא, שֹׁכֵן עַד וְקָדוֹשׁ

וְתִשְׁתַּבַּח הַמִּתְנַשֵּׂא לְרֹאשׁ,

כִּי אֵין כָּמוֹךָ שׁוֹכֵן עַד, מָרוֹם וְקָדוֹשׁ –

בא״י הַמֶּלֶךְ הַקָּדוֹשׁ.


תִּתְבָּרֵךְ שַׁדַּי, אֵל מַעַרְכֹת יִשְׂרָאֵל

וְתִשְׁתַּבַּח קְדוֹשׁ יִשְׂרָאֵל,

כִּי אֵין כָּמוֹךָ מוֹשִׁיעַ וְגוֹאֵל –

בא״י אֱלֹהִים אֱלֹהֵי יִשְׂרָאֵל.


תִּתְבָּרֵךְ תּוֹכֵן רוּחוֹת וּבוֹרֵא עוֹלָם

וְתִשְׁתַּבַּח הַמֵּאִיר בְּכֹחוֹ לְכָל בָּאֵי עוֹלָם,

כִּי אֵין כָּמוֹךָ אֱלֹהִים חַיִּים וּמֶלֶךְ עוֹלָם –

בא״י לְעוֹלָם.
