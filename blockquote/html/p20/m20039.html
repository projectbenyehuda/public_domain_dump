אָרַךְ זְמַנִּי כַּמֶּה וְכַמָּה

אהבה


אָרַךְ זְמַנִּי / כַּמֶּה וְכַמָּה, // סָר הוֹד וּפָנָה / שֶׁמֶשׁ וּבָא.

נִחַר גְּרוֹנִי, / יָבֵשׁ בְּצָמָא // לִדְבַר אֱמוּנָה / שֶׁנִּכְתְּבָה.


בִּשְׁבִי אֲיַחֵל / עַד בֹּא דְבָרִי, // יָמִים וְשָׁנִים / אֵין לִי דֳמִי,

יָרֵא וְזוֹחֵל / שַׁבְתִּי בְצִירִי, // לָקַח מְעַנִּי / כָּל נָעֳמִי.

אֵיךְ צַר, יְנַהֵל / עַמּוֹ דְבִירִי, // יֹאמַר לְבָנִים: /׳מִי צוּר וּמִי?

חוֹזְכֶם פְּלוֹנִי / פִּתָּה בְעָרְמָה // אֶתְכֶם – וְעָלָה; / שָׁוְא כִּי צְבָא

אֵל קָם לְהָנִיא / לֵב אָח וְאָמָה, // אַךְ זֹאת דְּעוּ נָא / כִּי נִכְזְבָה.׳


׳רֻכְסֵי אֲנָשִׁים / בָּאוּ לְרָגְלִי,׳ // אָמְרָה כְבוּדָּה / תּוֹךְ בֵּית שְׁבִי,

׳בָּנַי גְּרוּשִׁים / מִבֵּית זְבוּלִי, // וִיקָר וְחֶמְדָה, / אֶרֶץ צְבִי,

תּוֹפְשִׂים קְדֵשִׁים, / וַיֹּאמְרוּ לִי: // בָּא יוֹם פְּקֻדָּה, / שִׁפְלִי שְׁבִי!

אָנָּא, אֲדֹנִי, / רַחֵם אֲיֻמָּה // מִהְיוֹת שְׁנִינָה; / לֹא שֻׂגְּבָה,

בִּכְאֵב וָעֳנִי / נָשְׂאָה כְלִמָּה, // בָּדָד בְּחוּנָה / הִיא יוֹשְׁבָה.׳


׳הַרְבּוֹת דְּבָרִים / עִזְבִי, גְּבִירָה, // עָלָה בְאָזְנַי / קוֹל צַעֲקֵךְ,

שָׁם בַּסְּפָרִים / חָזוּת שְׁמוּרָה: // שׁוֹפְטֵךְ יְיָ / וּמְחוֹקְקֵךְ,

אֶשָּׂא וְאָרִים / יָדִי מְהֵרָה // לִשְׁפֹּךְ חֲרוֹנַי / עַל דּוֹחֲקֵךְ

וּבְרֹב גְּאוֹנִי / לַעְשׂוֹת נְקָמָה // בִּבְנֵי עֲדִינָה / אֶתְיַצְּבָה,

לִשְׁמִי תְּתַנִּי / שִׁירָה נְעִימָה // כִּי עֵת חֲתֻנָּה / הֵן קָרְבָה.׳


׳מֹר וַאֲהָלִים, / קָנֶה וְקִדָּה // עִם קִנְמָן בֹּ־/שֶׂם תִּרְקְחִי,

נָסוּ צְלָלִים / כַּיּוֹם וְהִפְדָּה // אוֹתָךְ גְּבִיר בּוֹ / עוֹד תִּשְׂמְחִי,

וַהְמוֹן נְבָלִים / שָׂמוּךְ לְנִדָּה // מַלְכִּי בְחַרְבּוֹ / יָשִׂים סְחִי.

יוֹם צָר חֲרוֹנַי / אַהְבָה קְדוּמָה // אֶזְכֹּר לְיוֹנָה, / וַאֲשׁוֹבְבָה

הוֹר בַּת הֲמוֹנַי, / גַּם אֶקְרְאָה: מַה // יָפִית וּמַה נָּ־/עַמְתְּ אַהֲבָה.׳
