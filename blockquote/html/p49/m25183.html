[באשמרת ערב לבבי]

ולה אלי ראש כלה ר שמואל אלבגדאדי גואב ען שער כאטבה בה מן דאניה

ולו (שיר) לראש הכלה רב שמואל הבגדדי, והוא תשובה על שיר שבו פנה אליו מן העיר דניה

בְּאַשְׁמֹרֶת עֶרֶב / לְבָבִי בִּי יָרֶב / וְתוֹכָחוֹת הֶרֶב / בְּמוּסָר הִכְתִּירָם

וְאָמַר לִי: קוּמָה, וְהַנַּח תַּרְדֵּמָה, / וְהַבֵּט שָׁמַיְמָה, / וּבָרֵךְ צוּר קֵרָם!

רְאֵה אֶת צִבְאוֹתָם,/ וְסַפֵּר פִּלְאוֹתָם / וְדַע יָצַר אוֹתָם / וְיָדַע מִסְפָּרָם;

וְשֶׁמֶשׁ מִתְהַלֵּךְ / בְּתוֹכָם כַּמֶּלֶךְ, / וְכֻלָּם עַל פֶּלֶךְ / אֱלֹהֶיךָ זֵרָם,

כְּשׁוֹשַׁנֵּי סוּגָה / נְפוֹצוֹת בַּעְרוּגָה / יְרוּצוּן בִּמְחוּגָה / עֲלֵי קַו כַּבִּירָם;

וְהַבֵּט רָמֵיהֶם / בְּהִלָּם בִּשְׁמֵיהֶם / זְכֹר חָבֵר לָהֶם / וּמֵאִיר כִּמְאוֹרָם

אֲשֶׁר יָשַׁב שַׁעַר / בְּתוֹךְ אֶרֶץ שִׁנְעָר וּבִגְדוֹלָם גָּעַר / וּמָאַס שִׁנְעָרָם

זְכֹר חָסִיד וָתָם / תְּבוּנוֹת לֹא חָתַם, / אֲבָל פִּזַּר אוֹתָם / וְרִפֵּא נִשְׁבָּרָם

שְׁמוּאֵל בֶּן יוֹסֵף / שְׁמַע דִּבְרֵי כוֹסֵף / שְׁמוּאֵל בֶּן יוֹסֵף / וְעָנְדָה סַפִּירָם:

חֲכָמִים נוֹעָדִים / בְּכָל יוֹם מַגִּידִים / כְּבוֹד אֶרֶץ כַּשְׂדִים / בְּשִׂיחַ סִפּוּרָם;

וְלֹא לָאַדִּירִים / מְקוֹמוֹת מַפְאִירִים / וְגַם לֹא נֶהְדָּרִים / בְּאַרְצָם וּמְגוּרָם,

אֲבָל אַרְצָם תָּרוֹם / בְּכָל נָבוֹן מָרוֹם / בְּצָפוֹן וּבְדָרוֹם / שְׁמָם נִשָּׂא וָרָם.

וְיָסוּרוּ מִשָּׁם – / וְאַדְמָתָם תֶּאְשָׁם, / וְתִרְאֶה מִדְרָשָׁם / כְּיַעַר מִדְבָּרָם;

וְעַל כֵּן מִי יִבְזֶה / לְאַלּוּף בֶּן מַזֶּה, / אֲשֶׁר הוּא תּוֹךְ עַם זֶה / קְצִינָם וּגְבִירָם

וּמַחְסֵיָה בָתָה / לְנוּדוֹ, וּבְדִיתָא, / וְאִסְפַּמְיָא שׁוֹתָה / וְרָוָה מִבּוֹרָם.

וְלוּ עַתָּה יִהְיֶה / בְּאֶרֶץ־אַבַּיֵּי / אֲזַי בִּמְקוֹם הַאיֵי / תְּעוּדַת אֵל הוֹרָם,

וּפָשְׁטָה תּוֹרָתוֹ / וְתַרְשִׁישֵׁי דָתוֹ / בְּפִיּוֹת אֻמָּתוֹ / וְנָהַג אֶת דּוֹרָם,

וְשָׂמוּהוּ עַל רֹאשׁ / בְּנֵי דּוֹרוֹ לִדְרשׁ / וְקָרְאוּ אוֹתוֹ רֹאשׁ / וְנִזְרָם וּנְזִירָם.

בְּשִׁירָךְ שָׂמַחְתִּי / וּבִדְבָרָיו שַׂשְׂתִּי / וְכַדָּת אִשַּׁרְתִּי / תְּבוּנַת חִבּוּרָם.

בְּטוּרָיו נִשֵּׂאתִי, / וְחָכְמוֹת מָלֵאתִי, / וְרֹאשִׁי נָשָׂאתִי / לְהַתִּיר נִקְשָׁרָם.

אֲמָרָיו סַפִּירִים, / וְאוֹתוֹתָיו שָׂרִים, / וּמִלוּאַת טוּרִים / בְּחשֶׁן – סִדּוּרָם,

וְעִנְיָן כִּפְרָחִים / וּמַעְשֵׂה פִּתּוּחִים / מְכֻסִּים כִּירֵחִים / בְּלֵילֵי עִבּוּרָם.

לְךָ חֵלֶק עֶלְיוֹן / בְּעֵט וּבְגִלָּיוֹן, / וְגוֹלַת עַם אֶבְיוֹן / בְּצִלָּךְ מִסְתּוֹרָם.

וְסוֹד אַמְתִּיק עִמָּךְ / לְלַמֵּד אֶת עַמָּךְ / כְּמִתְנַשֵּׂא כַּמָּךְ / כְּרַבָּם כִּצְעִירָם,

לְהַשְׂכִּיל וּלְהוֹרוֹת / עֲצֵלִים בַּתּוֹרוֹת / וְלִזְרֹעַ אוֹרוֹת / וְלָנִיר אֶת נִירָם.

בְּךָ וּבְמַכָּרַי / אֲנַגֵּחַ צָרֵי, / וְלָךְ יִתְרוֹן שִׁירַי / וְעָלַי מַחְסוֹרָם!
