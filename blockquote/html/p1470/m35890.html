אָז נוֹלָד

לז / שמות ז ח – כי ידבר אליכם


א

אָז נוֹלָד רָאִיתָ / בְּחִירָךְ אָז הֶרְאֵיתָ

גִּנְזֵי לֵב צַר יָדָעְתָּ / דָּבָר אֲשֶׁר יְדַבֵּר לָהֶם הוֹדָעְתָּה.


הַמַּטֶּה אֲשֶׁר נִבְרָא לְעַנּוֹתוֹ / וּבוֹ כְאִוַּלְתּוֹ לַעֲנוֹתוֹ

זְחִילַת תַּנִּין לְהַרְאוֹתוֹ / חֲמַת תַּנִּינִים לְהַרְווֹתוֹ.


טֶפֶשׁ לִבּוֹ יִתְגַּבֵּר / יִרְגַּז כְּזוֹחֵל וְיִתְעַבֵּר

כֹּחַ מַטֶּה אוֹתוֹ יַדְבֵּר / לְעֵת יֻעַשׂ מוֹפֵת אֲשֶׁר יְדַבֵּר.


מַעֲנֶה לָנוּ תִּפְתַּח / וּמוֹסְרוֹתֵינוּ תְפַתַּח / בְּצֶדֶק יוֹשֵׁב פֶּתַח.


ב

מַטֶּה הַנִּקְרָא לְשֵׁם אֱלֹהִים / נִקְרָא עוֹד לְשֵׁם אַחִים

שָׂמוּ בָם אוֹתוֹת וּמוֹפְתִים / עֲלֵי נוֹפִים וּמוֹפִים וְנַפְתּוּחִים.


פִּתְגָּם מַלְכָּם לֹא אִיחֵרוּ / צִירִים וּמִיָּד מִיהֵרוּ

קִירוֹת לִבּוֹתָם נִמְהָרוּ / רָגְזוּ וּמִדַּעַת נִבְעָרוּ.


שְׁלוּחִים עֵת אֲשֶׁר נִרְאוּ / שְׁלִיחוּתָם מִיָּד הֶרְאוּ

תַּעֲצוּמוֹת מִפִּי מֶלֶךְ נִבָּאוּ / תְּפוּסֵי עוֹז בְּעַזּוּת אֵלָיו בָּאוּ.


לִבּוֹ בְהִתְבּוֹנֵן / בְּבִינַת אֵל מְכוֹנֵן / טַל תֶּחִי יְחוֹנֵן.


ג

יַנְחֵינוּ שִׁבְטָךְ / וּמִשְׁעֶנֶת מַטָּךְ / עַל אֱדוֹם בְּהַטָּךְ / מוּסַר שׁוֹטָךְ

נְשָׂא נֵס גָּדוֹל / לְתַנִּין יָם הַגָּדוֹל / כְּתַתָּה אוֹת גָּדוֹל / לַתַּנִּין הַגָּדוֹל

יְשֻׁלַּח מַטֵּה עֻזָּךְ / וְשֵׁבֶט מָעֻזָּךְ / חֹטֶר גַּאֲוָתָךְ / מַקֵּל גְּבוּרָתָךְ

יָהּ מוֹפֵת אַחַת שִׁלַּחְתָּ עַל מִצְרַיִם / מוֹפְתִים הַרְבֵּה תְּשַׁלַּח עַל מְצֵירִים.


ד

אֱמֶת לֹא נָאֶה לִכְסִיל כָּבוֹד / וְלִכְסִיל נַמְתָּה לַחֲלֹק כָּבוֹד

בְּכָבוֹד לְהֵרָאוֹת לוֹ / וְכָבוֹד לְהַרְאוֹת לוֹ

אוֹת לְהַפְלִיא לוֹ / וּמוֹפֵת לִפְעֹל לוֹ

בַּעֲבוּר לְרַפּוֹת לוֹ מוֹסֵירוֹ / וּלְהַחֲזִיק עָלָיו מוּסָרוֹ

לִבּוֹ וְכִלְיוֹתָיו לְחַסְּרוֹ / בְּשֵׁבֶט מוּכָן לְנֵיו כְּסִילִים לְיַסְּרוֹ

לְסַלֵּף חֲכָמָיו / לְהַטְעוֹת חַרְטֻמָּיו / לְהוֹבִישׁ אֲשָׁפָיו / לְהַחְפִּיר כַּשָּׁפָיו

כִּי גַם כְּשָׁפִים אֵין בִּלְעָדֶיךָ בִלְתְּךָ נוֹרָא וְקָדוֹשׁ.


ה

אָז שֵׁבֶט מוּסָר יַרְחִיק אִוֶּלֶת / וּלְפַרְעֹה וּלְעַמּוֹ הוֹסִיף אִוֶּלֶת בַּעֲבוּר לָתֵת עֲווֹן עַל עֲווֹנָם / לְנַצֵּל מֵהֶם נַפְשָׁם וְהוֹנָם גֶּבֶר בָּעוֹז / אֲשֶׁר קָח מַטֵּה עוֹז / וְנַעֲשָׂה נָחָשׁ לְפָנָיו / וְנָס מִפָּנָיו דִּבַּרְתָּ לּוֹ לְדַבֵּר לְאָחִיו לְשֵׂאתוֹ / וְלִפְנֵי מוֹשֵׁל תַּנִּין לַעֲשׂוֹתוֹ הִשְׁלִיכוֹ וְנַעֲשָׂה לִשְׁלֹשָׁה מִינִים / לְעֶכֶס וּלְתַנִּין וּלְרֹאשׁ פְּתָנִים וַיִּבֶז צַר וְהַדָּבָר לֹא בָא לוֹ / כִּי בָּז לַדָּבָר יֵחָבֶל לוֹ זָד וְקָרָא לְכָל בַּעֲלֵי כְשָׁפִים / וְעָשׂוּ כֵן מַקְלוֹת נְחָשִׁים שׁוֹאֲפִים חָזַר עֲלֵיהֶם מַטֵּה אֵל וּבָלַע אוֹתָם / וְנוֹאֲלוּ שָׂרֵי צֹעַן כֵּן בִּרְאוֹתָם טָפַשׁ וְכָבַד לִבּוֹ מִדָּעַת / וְנַעֲשָׂה כְּכָבֵד פּוֹלֶטֶת וְלֹא בּוֹלַעַת יָהּ לְהַעֲבִיר עָלָיו עֶשֶׂר מַכּוֹת / אֲשֶׁר עַל מַטֶּה הָיוּ כְתוּבוֹת.


כי ידבר


ו

אֱוִיל בְּרוֹעַ מַעְבָּדָיו / דִבֵּר אִוֶּלֶת בְּמַעֲמַד עֲבָדָיו

בִּינוּ עֲנוֹת אִוַּלְתּוֹ לְעֵינָיו / פֶּן יִהְיֶה חָכָם בְּעֵינָיו.


גַּסּוּת לִבּוֹ אִם נִתְהַפַּךְ / וְיִשְׁאַל מִכֶּם אוֹת הֲפַכְפַּךְ

דְּבָרִי יַהֲפֹךְ דְּבָרוֹ הַמְּהֻפָּךְ / בַּמַּטֶּה אֲצֶר לְנָחָשׁ נֶהְפָּךְ.


הַאִם בִּקֵּשׁ מוֹפֵת תְּנוּ לוֹ / וְאִם שָׁאַל שְׁאֵלָה תַּנּוּ לוֹ

וְעַל אִוַּלְתּוֹ אִוֶּלֶת שְׁנוּ לוֹ / וְלֹא תִּרְאוּ כִּי לֹא תִשְׁווּ לוֹ.


זָר הַזָּהוּם בְּזִרְמַת סוּס / בְּמַטֶּה יִירְדוֹ כְּשׁוֹט לַסּוּס

חָשׁוּב הוּא בְשָׂרוֹ כִּבְשַׂר חֲמוֹר / לָכֵן מַתְּנוּהוּ כְּמֶתֶג לַחֲמוֹר.


טוֹעֶה אֲשֶׁר הִתְעַתּוּ רוּחוֹ / בְּנַחַת רוּחֲכֶם הִלְכוּ לְרוּחוֹ

יוֹצִיא כִּכְסִיל כָּל רוּחוֹ / וְתִשָּׁבֵר גַּסּוּת רוּחוֹ.


כּוֹחַ מַטֶּה בְּשׁוּרוֹ יִירָא / וְעֵת יֻעַשׂ מוֹפֵת לֹא יִירָא

לָכֵן חַכְמֵי עַמּוֹ יִקְרָא מְהֵרָה / וְיַעֲשׂוּ כֵן בְּלַהֲטֵי מְאֵרָה.


מַלְעִיג מְבַזֶּה יְהִי עַל כֵּן / בֵּעָשׂוֹת כֵּן וְיַעֲשׂוּ כֵן

נִבְלְעוּ מַטּוֹת בְּמַטֶּה וְתָמְהוּ עַל כֵּן / כִּי לֹא יוּכְלוּ כֵּן לְתַכֵּן.


נֵס הָאַחֲרוֹן יִהְיֶה נָוֶה / מִן הָרִאשׁוֹן יְהִי מְנֻוֶּה

עוֹז מַטֶּה בְּמַעֲמָדוֹ יְחַוֶּה / כִּי לֹא…… וְלֹא יֵעָבֶה.


פֶּלֶא זֶה בִּרְאוֹת אֱוִיל לֵב / כִּי חֲכָמָיו וְשָׂרָיו אֱוִילֵי לֵב

צוּר יְצוּר …….. שי לֵב / וְיִפּוֹל בְּרָע כָּל מַקְשֶׁה לֵב.


קוֹצֵף וְאוֹמֵר …… עָצְמִי / לִי יְאוֹרִי וַאֲנִי בָרָאתִי עַצְמִי.
