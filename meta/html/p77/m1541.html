אמת עולם

כתובת: עוד לו

אֱמֶת עוֹלָם סְדָרָיו הֶחֱלִיף אוֹ / בְקִרְבּוֹ חֻדְּשׁוּ נִסּוֹ וּפִלְאוֹ

הֲכִי בָא הָאֲרִי לִשְׁאוֹל לְשָׂדֶה / אֱלֵי שֵׂיוֹ וְלָבִיא אֶל טְלָאוֹ

וְשָׁאַל מִדְּוֵה לֵבָב וְנִכְאָב / אֲרוּכָה לוֹ חֲכָם לֵבָב וְרוֹפְאוֹ

וּמֵעֻלְפֶּה לְרֹב חִילוֹ יְצַפֶּה / תְּרוּפָה אֶל בְּרִיאוּתוֹ בְרִיאוֹ

וְיוֹשֵׁב בֵּין בְּרֵכוֹת בָּא לְשָׁרָב / וּמִדְבָּר נֶעֱזָב לִשְׁבֹּר צְמָאוֹ 5

וּמֵאֵלֶּה הֲלֹא חָדָשׁ וְנִפְלָא / בְּהִשָּׁמַע דְּבַר שִׁירָךְ בְּקָרְאוֹ

בְּקָרְאוֹ לִי אֲשֶׁר חֻלַּל חֲלִילִי / וְשַׁח קוֹלוֹ וְסָר כֹּחוֹ וְדָבְאוֹ

וְשָׁב נִבְלִי כְמוֹ נֵבֶל מְנֻפָּץ / עֲדֵי כִי דַק וְאֵיךְ יִהְיֶה נְכֵאוֹ

זְמָן חָנֵף נְשָׂאַנִי בְמִרְמָה / בְאֶבְרָתוֹ וְהָיִיתִי נְשׂוּאוֹ

וְעוֹד מִזְעָר וְהַמִּשְׁפָּט מְהֻפָּךְ / נְשָׂאתִיהוּ וְנֶהְפַּכְתִּי לְנוֹשְׂאוֹ 10

זְמָן בּוֹגֵד קְרָאָנִי וְהֵבִיא / מְגוֹרוֹ לִי בְיוֹם הִקְדִּישׁ קְרוּאוֹ

וְאָסַף מַחֲנוֹת יָגוֹן לְנֶגְדִּי / כְאִלּוּ אֵין בְּתֵבֵל עוֹד שְׂנוּאוֹ

תְּמוֹל הוּתַל לְבָבִי מֵהֲבָלָיו / וְהַיּוֹם הָאֱמֶת שַׁבְתִּי וְרָאֹה

וְאֶמְצָא רוֹשׁ בְּיַעְרָתוֹ וְלַעְנָה / בְּצוּף דִּבְשׁוֹ וְרֵיקוּת עִם מְלוֹאוֹ

וְאָכֵן אַחֲרִישׁ בָּדָיו וְלֹא לוֹ / אֲבָל בִּגְלָל כְּבוֹד יוֹצְרוֹ וּבוֹרְאוֹ 15

וְאַךְ כִּי אֶבְחֲנָה דַרְכּוֹ וְעֶרְכּוֹ / מְאַסְתִּיהוּ וְהוֹסַפְתִּי לְשָׂנְאוֹ

וְעָזַבְתִּי מְשׁוֹשׁוֹ עִם יְגוֹנוֹ / וְהִרְחַקְתִּי טְהוֹרוֹ עִם טְמֵאוֹ

וְתָלִיתִי חֲלִילִי עַל עֲרָבִים / וְכִנּוֹרִי עֲלֵי רָאשֵׁי בְכָאוֹ

עֲדֵי זָנַח לְבָבִי בֵית זְמִירִים / וְאַרְמוֹן שִׁיר וְאוּלַמּוֹ וְתָאוֹ

וְאִם לִשְׁאוֹל לְשָׁלוֹם לִי יְבַקֵּשׁ / אֲכַחֵשׁ בּוֹ וְלֹא אַכִּיר בְּבוֹאוֹ 20

וְעַתָּה מַה לְּךָ תִּזְעַק לְנֶגְדִּי / ומַה יִתֵּן חֲסַר לֵב אֶל בְּקִיאוֹ

וּמִי זֶה מֵאֱוִיל יִשְׁאַל תְּבוּנָה / וּמִי יִדְרֹשׁ דְּרוֹר מִיַּד כְּלוּאוֹ

וְאִישׁ חֶפְצוֹ עֲשֹׂה כָזוֹ יְעָצוֹ / הֲלֹא מַיִם יְאַו לִצְרוֹר בְּטַנְאוֹ

וּמַה לָּךְ הַיְּדִיד לָתוּר בְּקִרְבָּךְ / צְפוּן לִבָּךְ וְחַפֵּשׂ מַחֲבוֹאוֹ

וְתַעַשׂ מִשׁאֲלוֹת לִבָּךְ בְּיָדָךְ / וּמִי גֶבֶר יְאַו חַפֵּשׂ בְּמָצְאוֹ 25

לְךָ מִמְשָׁל עֲלֵי מוֹשׁלֵי מְשָׁלִים / וְעָלִיתָ מְרוֹם הַשִׁיר וְשִׂיאוֹ

וְעֵת דִּבֵּר זְמִיר בָּם בַּחֲלוֹמוֹת / לְךָ נִרְאָה וְהָיִיתָ נְבִיאוֹ

וְאִם יֵשׁ בַּזְּמָן דּוֹבֵר נְגִידִים / נְגִידוֹ לִבְּךָ וּנְשִׂיא נְשִׂיאוֹ

וְאִם יָשַׁב לְכִסֵּא שָׁר וְנוֹגֵן / הֲלא מֵאִתְּךָ רוּחַ נְשָׂאוֹ

וּמָלַכְתָּ בְּמַלְכֵיהֶם וְהָיוּ/ לְעֻמָּתָךְ כְּכֶבֶשׂ מוּל לְבִיאוֹ 30

הֲכִי לָךְ נִתְּנוּ מֵאָז לְמָנָה / מְלוּכַת שִׁיר וְהַמֶּלֶךְ וְכִסְאוֹ

סימן

גִּילָה וְרַע בִּיקָר בַּל / יָלִין וְסוּת הוֹדוֹ בָּל

נִשְׁאָר כְּיוֹנָה פוֹתָה / אֵין לֵב וְאִם יֵשׁ חֻבַּל

אַךְ בַּעֲדָךְ הוֹסִיף עֹז / עַד כִּי בְּלֹא לֵב שָׂם בל
