דּוֹרֵשׁ דְּבַר צַחוֹת

מתוך הספר עפר ודינה


א


דּוֹרֵשׁ דְּבַר צַחוֹת וְאִמְרֵי שֶׁפֶר,

שׁוּר מַעֲשֵׂה דִינָה וְתוֹלְדוֹת עֵפֶר!

דָּת שַׁעֲשׁוּעִים שָׁם לְיַלְדֵי הַזְּמָן

לִקְרֹא בְהַבְלֵי שָׁוְא וּמִשְׁלֵי אֵפֶר.

שָׁם תּוֹכְחוֹת מוּסָר, וְשָׁמָּה יִמְצְאוּ

חָזוֹן נְבוֹנֵי לֵב וְיוֹדְעֵי סֵפֶר.


אִישׁ הָיָה בְאֶרֶץ נוֹד / וְיְהִי לוֹ עֹשֶׁר וְכָבוֹד / וַעֲבֻדָּה רַבָּה מְאֹד / וְשָׂדוֹת וּכְרָמִים שְׁפָחוֹת וַעֲבָדִים, / צֹאן וּבָקָר וּגְמַלִּים וְסוּסִים וּפְרָדִים, / אֵין קֵצֶה לְקִנְיָנָיו וְחֵילוֹ / וְאֵין חֵקֶר לִכְבוֹד עָשְׁרוֹ וְרֹב גֻּדְלוֹ. / וְשֵׁם הָאִישׁ עֵפֶר / מִמִּשְׁפַּחַת בּוּז מֵאֶרֶץ חֵפֶר, / מִפְּנֵי שֵׂיבָה חָשַׁךְ מִשְּׁחוֹר תָּאֳרוֹ / וּמֵרֹב יָמִים בִּלָּה בְשָׂרוֹ וְעוֹרוֹ, / וְהָאִישׁ קָשֶׁה וְרַב מַעֲלָלִים / אַחֲרֵי שְׁרִירוּת לִבּוֹ הוֹלֵךְ וְאֵין מַכְלִים / וְכָל חֲפָצָיו יַשְׁלִים. / ״אֱכֹל וּשְׁתֵה!״ יֹאמַר לָךְ – / וְאַחֲרֵי לֹא יוֹעִיל הָלַךְ. / מִשְּׁמֹעַ מוּסָר אוֹטֵם אָזְנָיו / וְאֵין פַּחַד יְיָ לְנֶגֶד עֵינָיו, / בְּפֹעַל יָדָיו חוֹמֵס נַפְשׁוֹ / וּמוֹרָא לֹא יַעֲלֶה עַל רֹאשׁוֹ. / וְשֵׁם אִשְׁתּוֹ מְהֵיטַבאֵל, / בַּת אִישׁ חַיִל רַב פְּעָלִים מִקַּבְצְאֵל, / אִשָּׁה חַכְמַת לֵב וְנֶאֱמָנָה / מְכַבֶּדֶת אֶת יְיָ מֵהוֹנָהּ, / וְתוֹרַת חֶסֶד עַל לְשׁוֹנָהּ, / מְאֻשֶּׁרֶת בְּמִפְעָלֶיהָ / יָפָה בְּמַעֲלָלֶיהָ / וַתְּהִי נוֹשֵׂאת חֵן בְּעֵינֵי כָל רוֹאֶיהָ. / עַל כֵּן יֹאמְרוּ בְמַהֲלָלָהּ / ״אֵשֶׁת חַיִל עֲטֶרֶת בַּעְלָהּ!״ / בְּטוּב לִקְחָהּ מֵעֵת לְעֵת תּוֹכִיחֶנּוּ / וּבְחֶלְקַת אֲמָרֶיהָ לִרְגָעִים תִּבְחָנֶנּוּ, / אַף לֵילוֹת עַל עֶרֶשׂ יְצוּעֶיהָ תְיַסְּרֶנּוּ / וּמִנַּחַל עֲדָנֶיהָ תַשְׁקֶנּוּ – / וְהוּא כְמוֹ פֶתֶן חֵרֵשׁ בְּקוֹלָהּ אֵינֶנּוּ שׁוֹמֵעַ / שׂוֹנֵא תוֹכַחַת וּמוּסָר פּוֹרֵעַ / בְּכָל עֵת אוֹהֵב הָרֵעַ. / וְיְהִי כְדַבְּרָהּ אֵלָיו יוֹם יוֹם וְלֹא שָׁמַע אֵלֶיהָ / וְלֹא הִטָּה אָזְנָיו לְנֹעַם מִלֶּיהָ – / מֵרֹב כַּעְסָהּ וְשִׂיחָהּ / אֶתְכְּרִיַּת רוּחָהּ / וְתִגְרַע שִׂיחָה / וּלְשׁוֹנָהּ מֻדְבָּק מַלְקוֹחָהּ / וְלֹא תוֹסֵף תֵּת כֹּחָהּ, / וַתָּמָת הָאִשָּׁה מֵעִצְּבוֹן מַעֲשֵׂה אִישָׁהּ, / בָּאָה שִׁמְשָׁהּ גַּם פָּרְחָה נַפְשָׁהּ. / וַיִּשָּׁאֶר הָאִישׁ בְּיַרְכְּתֵי בֵיתוֹ / וַעֲרָפֶל חֲתֻלָּתוֹ.


ב

עפר מחזר אחרי דינה


וַיְהִי מִקֵּץ יָמִים וַיַּעַבְרוּ יְמֵי בְכִיתוֹ / וַיִּנָּחֶם הָאִישׁ אַחֲרֵי אִשְׁתּוֹ / וַיֵּצֵא אֶל רְחוֹב הָעִיר נִשְׁעָן עַל מִשְׁעַנְתּוֹ. / וַיַּרְא נַעֲרָה מִלִּבְנַת הַסַּפִּיר יוֹצְרָהּ הֱכִינָהּ / וּשְׁמָה דִּינָה, / יְפַת תֹּאַר וּבָרָה כַּחַמָּה, / בַּעֲלַת חֵן וּנְעִימָה, / בְּחַרְבוֹת לְשׁוֹנָהּ כָּל הַנִּמְצָא יִדָּקֵר / וּבְחִצֵּי אִישׁוֹנָהּ עֵינֵי הָאֲנָשִׁים תְּנַקֵּר. / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

עָפְרָה אֲשֶׁר חֵן כַּמְּעִיל לוֹבֶשֶׁת

אֶחְשֹׁק – וְנַפְשִׁי יָרְאָה מִגֶּשֶׁת.

הֵן אַהֲבָתָהּ עוֹרֲרָה וַיְהִי סְגוֹר

לִבִּי כְמַטָּרָה וְעֵינָהּ – קֶשֶׁת.

חִכָּהּ לְצוּדֵנִי כְּחַכָּה הֹעֲלָה

וַאֲנִי כְבֶן צִפּוֹר וְנִיבָהּ – רֶשֶׁת.


וַיּוֹסֶף עֵפֶר לְהַבִּיט נֹכַח פְּנֵי הַנַּעֲרָה / וְהִנֵּה כְּעֶצֶם הַשָּׁמַיִם תָּאֳרָהּ / וְכָאַסְפַּקְלַרְיָא הַמְּאִירָה / זַכָּה וּבָרָה. / וַתִּיקַר בְּעֵינָיו מִכֹּל כְּלִי חֶמְדָּה / וְכָל אוֹצְרוֹת זְהָבוֹ כְּאַיִן נֶגְדָּהּ / וְלֹא יַעַרְכֶנָּה אֹדֶם וּפִטְדָה. / וַתִּדְבַּק נַפְשׁוֹ בְדִינָה / וַיֹּאמַר: הָאֱלֹהִים לְיָדִי אִנָּה / לְעֵת זִקְנָתִי מָגֵן וְצִנָּה, / מִפְּנֵי הַצִּנָּה, / בִּימֵי הַסְּתָו תִּהְיֶה לְנַפְשִׁי מִשְׁעֶנֶת / וּתְהִי לִי סוֹכֶנֶת, / הִיא הָאִשָּׁה אֲשֶׁר הוֹכִיחַ יְיָ / לְהִתְעַלֵּס בָּהּ כְּחֶפְצִי וּכְמַאֲוַיַּי / וּלְמַלֵּא כְרֵסָהּ מַעֲדַנַּי.

וַיַּחְמֹד עֵפֶר אֶת יָפְיָהּ בִּלְבָבוֹ / וּבְקִרְבּוֹ יָשִׂים אָרְבּוֹ, / בְּאַהֲבָתוֹ אוֹתָהּ עַל לוּחַ לִבּוֹ יִכְתֳּבֶנָּה, / מַשְׁחִית נַפְשׁוֹ הוּא יַעֲשֶׂנָּה. / וְלָכֵן אָמַר הַמְּשׁוֹרֵר:


אַל נָא בְמֶתֶק מַעֲנֶה תִּבְטָחוּ,

גַּם לֹא לְיֹפִי מַחֲזֶה תִשְׂמָחוּ –

פֶּן יִקְרְכֶם כָּעוֹלֲלִים עֵת יֶחֱזוּ

אוּר גַּחֲלֵי אֵשׁ כִּי בְיָד יִקָּחוּ.


וַיֵּלֶךְ עֵפֶר וַיְדַבֵּר בְּאָזְנֵי אָבִיהָ / וַיֹּאמֶר לוֹ ״אֶת דִּינָה / רָאִיתִי וְאַהֲבַת עוֹלָם אֲהַבְתִּיהָ, / תְּנָה אוֹתָהּ לִי לְאִשָּׁה כִּי אִוִּיתִיהָ! / הִיא יָשְׁרָה בְעֵינַי מִכֹּל בְּנוֹת עִירִי / לַעֲשׂוֹת עָנָף וְלָשֵׂאת פֶּרִי, / בָּהּ שַׂמְתִּי מִבְטַחִי וְכִסְלִי – / אֵין כָּמוֹהָ, תְּנֶנָּהּ לִי. / עַל כַּפַּיִם אֶשָּׂאֶנָּהּ / וְעַל בֵּיתִי אֲצַוֶּנָּה / וְאִבָּנֶה גַּם אָנֹכִי מִמֶּנָּה. / כִּי מַה חֶפְצִי בְּבֵיתִי בְּרֹב עָשְׁרִי, / וְאָנֹכִי הוֹלֵךְ עֲרִירִי?״ / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


נֶפֶשׁ נְכֵאָה תַּעֲמֹד נִבְהֶלֶת

גַּם נוֹהֲמָה, כִּי אָבְדָה תוֹחֶלֶת.

אֵיךְ תַּעֲלֹז אִם יוֹם וְלַיִל תַּעֲמֹל

רוּחָהּ, וְלֹא תֵדַע לְמִי עוֹמֶלֶת?


וְעוֹד אָמַר עֵפֶר ״וְהָיָה אִם שָׁמוֹעַ תִּשְׁמַע בְּקוֹלִי – / בְּיָדְךָ אַפְקִיד אוֹצְרוֹתַי וְחֵילִי, / וְאִם תַּטֶּה אָזְנְךָ לְשַׁוְעָתִי / לָתֵת אֶת שְׁאֵלָתִי – / אַתָּה תִּהְיֶה עַל בֵּיתִי, / וְאִם יִכָּנְסוּ דְבָרַי בְּאָזְנֶךָ / וְלַעֲשׂוֹת אֶת בַּקָּשָׁתִי יִיטַב בְּעֵינֶיךָ – / הִנֵּה כָל אֲשֶׁר לִי בְּיָדֶךָ, / וְאָנֹכִי אֲכַלְכֵּל אֶת שֵׂיבָתֶךָ / וְנָתַתִּי לְךָ עֵרֶךְ בְּגָדִים וּמִחְיָתֶךָ, / וְחָיִיתָ אַתָּה וּבֵיתֶךָ״. וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


הַקְשֵׁב עֲצַת זָקֵן יוֹעֵץ בְּלַחְשׁוֹ

וּשְׁמַע לְקוֹל יָשִׁישׁ גַּם הַאֲזֵן רַחְשׁוֹ:

אֶת בִּתְךָ דִינָה יָפָה תְּנָה אֵלָיו,

כִּי כָל אֲשֶׁר לָאִישׁ יִתֵּן בְּכַד נַפְשׁוֹ.


וַאֲבִי הַנַּעֲרָה אִישׁ דַּל וְנַעֲנֶה / מֻכֵּה אֱלֹהִים וּמְעֻנֶּה. / וַיַּעַן ״כֹּל אֲשֶׁר דִּבֶּר אֲדוֹנִי אֶעֱשֶׂה / וּבְצֵל כְּנָפֶיךָ אֶחְסֶה / לֶאֱכֹל לְשָׂבְעָה וְלִמְכַסֶּה. / אוּלָם אֶת פִּי הַנַּעֲרָה אֶשְׁאַל, / וְיָדַעְתִּי: לֹא תָשִׂים מִלָּתִי לְאַל. / בְּחֵלֶק שְׂפָתַיִם אֲפַתֶּנָּה / וּלְכֹל אֲשֶׁר אֶחְפֹּץ אַטֶּנָּה. / וּבְעוֹד אֲנִי מְדַבֵּר עַל לִבָּהּ, / כְּיַד אֱלֹהַי הַטּוֹבָה – / קַשֵּׁט עַצְמְךָ וְהִתְיַפֵּה / וְגַם אֶת הַזָּקָן תִּסְפֶּה, / וְלֵךְ הָסֵר מֵעָלֶיךָ שִׂמְלַת אַלְמְנוּתֶךָ וּבִגְדֵי אֶבְלֶךָ / וּלְבַשׁ בְּגָדֶיךָ הַחֲמוּדוֹת הַחֲדָשׁוֹת / לְכַסּוֹת אֶת עֶרְוַת הָעֲצָמוֹת הַיְּבֵשׁוֹת״. / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


מַעְטֵה תְהִלָּה אוֹ צְנִיף תִּפְאֶרֶת

כִּסּוּ גְוִיָּה תִּהֲלַךְ קוֹדֶרֶת –

כֵּן כִּסְּתָה רַקָּה אֲדַמְדֶּמֶת עֲלֵי

טוּחוֹת בְּקִרְבָּם תַּעֲמֹד בַּהֶרֶת.


וַיּוֹסֶף עוֹד שְׂאֵת מְשָׁלוֹ וַיֹּאמַר:


מַכְאוֹב לְבָבְךָ בְּטוּב גֵּהָה תְּכַסֶּנּוּ

וּרְקַב עֲצָמוֹת בְּחֵן מַרְאֶה תְחַפֶּנּוּ.

תִּדְמֶה לְצוֹרֵף אֲשֶׁר נִשְׁחַת בְּיָדוֹ כְּלִי

כֶּסֶף – וְזָהָב לְכַסּוֹת מוּם יְצַפֶּנּוּ.


וַיֵּלֶךְ הָאִישׁ שָׂמֵחַ וְטוֹב לֵב לְבֵיתוֹ / וַיְסַפֵּר לְדִינָה בִתּוֹ אֶת כָּל הַמּוֹצְאוֹת אוֹתוֹ / וַיֹּאמֶר אֵלֶיהָ ״בִּתִּי, בִּתִּי, מָצָאתִי לָךְ מָנוֹחַ, / לֹא בְחַיִל וְלֹא בְכֹחַ: / כִּי הִקְרָה יְיָ לְפָנֵינוּ / אֶת עֵפֶר שְׁכֵנֵנוּ / לְהַחֲיוֹת אֶת נַפְשֵׁנוּ / וְלִמְאוֹר עֵינֵינוּ. / אֶמֶשׁ אָמַר אֵלַי בְּמַעֲנֶה רַךְ וּבִלְשׁוֹן בַּקָּשָׁה / לְתִתֵּךְ לוֹ לְאִשָּׁה – / וַאֲנַחְנוּ יָדַעְנוּ אֶת כְּבוֹד עָשְׁרוֹ וּמֶמְשַׁלְתּוֹ / בַּאֲשֶׁר יְיָ אִתּוֹ, / כִּי מָלְאָה כָל הָאָרֶץ תְּהִלּוֹתָיו, / וְאֵין קֵצֶה לְאוֹצְרוֹתָיו. / וְעַתָּה, בִּתִּי, מַלְאִי מִשְׁאֲלוֹתָיו / בְּנֶפֶשׁ חֲפֵצָה, / כִּי מֵיְיָ הַדָּבָר יָצָא. / וְכֵן אָמְרוּ קַדְמוֹנֵינוּ זַ״ל / ״הַכֹּל תָּלוּי בַּמַּזָּל״. / וְאִם יְיָ חָפֵץ לַעֲשׂוֹתוֹ – / מִי יָעַץ לְהָפֵר עֲצָתוֹ / אוֹ מִי יוּכַל לְתַקֵּן אֵת אֲשֶׁר עִוְּתוֹ?״ / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


יוֹצֵר אֱנוֹשׁ כָּל מַעֲשָׂיו שִׁוָּה / עַל חֹק וּמִי יָפֵר אֲשֶׁר צִוָּה?

מִי זֶה יְעַוֵּת אֶת אֲשֶׁר תִּקֵּן / אוֹ מִי יְתַקֵּן אֶת אֲשֶׁר עִוָּה?


וַתַּעַן דִּינָה וַתֹּאמַר / ״מַר לִי מַר! / הַעַל יְפִי הַזָּקֵן וּצְבִי תִפְאַרְתּוֹ / לִבְּךָ תָּשִׁית לְדַעְתּוֹ? / וְאִם עַל חֶמְדַּת פָּנָיו וְהַדְרָתוֹ / פָּקַחְתָּ עֵינֶיךָ לְעֻמַּת מַחְבַּרְתּוֹ? / וּבַמֶּה יִוָּדַע כִּי עֵת דֹּדִים אִתּוֹ – / בְּקָרַחְתּוֹ אוֹ בְגַבַּחְתּוֹ? / אֵין בּוֹ רַק שִׁבְתּוֹ / לֶאֱכֹל וְלִשְׁתּוֹת וְיִשְׁבֹּת מִכָּל מְלַאכְתּוֹ! / וְאֵיךְ תֹּאמַר שֶׁאֵל יַעֲקֹב בְּעֶזְרוֹ – / וַיֵּאָכֵל חֲצִי בְשָׂרוֹ! / יֵשֵׁב בָּדָד מִמְּקוֹמוֹ לֹא יָמִישׁ, / חַי נִזּוֹן בִּלְתִּי מַרְגִּישׁ, / וְלֹא נוֹדַע אִם בְּהֵמָה אִם אִישׁ. / וּמַה בֶּצַע בְּכַסְפּוֹ וּזְהָבוֹ / וְעַל חֶמְדַּת נָשִׁים לֹא יָבוֹא? / יָמָיו נִזְעֲכוּ וְרוּחוֹ חֻבָּלָה – / וּמַה יִּתְבּוֹנֵן עַל בְּתוּלָה? / וְאֵיךְ אֶשָּׂא בִימֵי עֲלוּמִים / כַּבִּיר מֵאָבִיךָ יָמִים! / הַאֶמְנַע מֵחֵפֶץ בַּחוּרֵי חֶמֶד לְמִשְׁכַּב דֹּדִים בִּימֵי בְחוּרוֹתַי, / וְשָׁכַבְתִּי עִם אֲבוֹתַי? / אָבִי, אָבִי! רְאֵה גַם רְאֵה לְחֻלְשַׁת הָאִישׁ וְתוֹלַדְתּוֹ, / מִתְהַלֵּךְ בַּחוּץ עַל מִשְׁעַנְתּוֹ, / כִּי תַם כֹּחוֹ וְחֵילוֹ / וּמַקְלוֹ יַגִּיד לוֹ!״ / וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:


נֹחַם לְנַפְשָׁהּ מֵאֲנָה לָקַחַת

דִּינָה בְקָרַחַת וְגַם גַּבַּחַת:

כִּי רָאֲתָה שֵׂיבָה בְּדֶגֶל אַהֲבָה

עַל קִיר לְבָבָהּ הוּא כְמוֹ מִסְפַּחַת

לֹא יַעֲלֹז רוּחָהּ בְּחֶמְדַּת אוֹצְרוֹת

כֶּסֶף וְלֹא מָצְאָה לְנַפְשָׁהּ נַחַת.

יֶעְרַב מְשׂוֹשׂ גִּילָהּ בְּבֶן גִּילָהּ וְתוֹךְ

חֶבְרַת צְבִי עֹפֶר זְמִיר פּוֹצַחַת.

מֵאֵין מְחִיר דּוֹד נֶאֱהָב נֶחְשָׁב זְהַב

אוֹפִיר, וּפָנָיו – מִגְדְּלוֹת מֶרְקַחַת.

אָכֵן מְלֵא יָמִים בְּפָנִים זוֹעֲפִים

עִם דֵּי זְהָבוֹ נֶחֱשָׁב אַמְתַּחַת!


וַיַּעַן אָבִיהָ וַיֹּאמֶר לָהּ ״בִּתִּי,בִּתִּי, / חֶלְקַת חֶמְדָּתִי, / יָדַעְתִּי אֶת הָאִישׁ וְאֶת שִׂיחוֹ, / עוֹד עָמַד בּוֹ טַעְמוֹ וְרֵיחוֹ, / לֹא כָהֲתָה עֵינוֹ וְלֹא נָס לֵחֹה. / וְאִם שֵׂיבָה זָרְקָה בּוֹ – / עוֹדֶנּוּ בְּאִבּוֹ, / אַחֲרֵי בְלוֹתוֹ הָיְתָה לּוֹ עֶדְנָה בְּצֶלֶם דְּמוּת תַּבְנִיתֵךְ, / וְלֹא יִהְיֶה זָקֵן בְּבֵיתֵךְ! / הֵן עוֹד הַיּוֹם בְּתוֹךְ עֲלָמוֹת יִתְיַצָּב, / וּבָא וְלָן וְשָׁכַב.

עוֹד יָדוֹ נְטוּיָה / לָקַחַת חֲנִית מַסָּע וְשִׁרְיָה, / בִּגְבוּרָה נֶאְזָר / וּבְכָבוֹד יִתְיַמָּר וּבְאִשָּׁה שָׁמַר, / אֵשֶׁת חֵן יִתְמֹךְ וּבְחֵיקוֹ יִשָּׂאֶנָּה, / גַּם כִּי יַזְקִין לֹא יָסוּר מִמֶּנָּה!״ / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


אַהְבַת בְּנֵי נֹעַר לְלֹא תוֹעֶלֶת,

כִּי עַל לְבָבָם נִקְשְׁרָה אִוֶּלֶת:

מֵעֵת לְעֵת נִתְהַפְּכָה רוּחָם וְלֹא

יִפְנוּ לְיַעְלַת חֵן וְאֶל אַיֶּלֶת.

אֶל הַזְּקֵנִים אָמְרָה נַפְשִׁי ״שְׁבוּ

בָזֶה!״ וְחֶבְרָתָם תְּהִי שׁוֹאֶלֶת,

כִּי אַהֲבָתָם תַּעֲמֹד לָעַד וּבָם

מִתְפָּאֲרָה נַפְשִׁי וּמִתְהַלֶּלֶת.


״בִּתִּי,בִּתִּי! אַתְּ אִשָּׁה חַכְמַת לֵב וּמַשְׂכֶּלֶת, / אַל תְּהִי דֶלֶת מֻשְׂכָּלוֹת לְפָנַיִךְ נִנְעֶלֶת, / וְלֹא תָשִׂימִי הַקְדָּמָה חֶלְקִית כְּכוֹלֶלֶת: / כִּי לֹא כָל הַבַּחוּרִים אַבִּירֵי הַטֶּבַע חִזְקֵי הַהַרְכָּבָה / וְלֹא כָל זִקְנֵי אֶרֶץ חֲלוּשֵׁי מֶזֶג מִפְּנֵי שֵׂיבָה: / יֵשׁ בָּחוּר מֵעַם וְטַעַם זְקֵנִים יִקַּח, / גַּם שָׂב, גַּם יָשִׁישׁ כְּגִבּוֹר לָרוּץ אֹרַח״. / וַיִּשָּׂא מְשָׁלוֹ:


לֹא מִפְּנֵי שֵׂיבָה בְּנֵי אִישׁ נִחֲתוּ,

גַּם לֹא לְיַלְדוּת עֹז וְחַיִל אָזְרוּ:

כִּי אֶחֱזֶה בָנִים – וְחָשַׁךְ מִשְּׁחוֹר

הוֹד תָּאֳרָם, וּפְנֵי זְקֵנִים נֶהְדְּרוּ!


״הֲלֹא תֵדְעִי לָךְ: אָבִינוּ הָרִאשׁוֹן אַחֲרֵי מוֹת שָׂרָה, / לָקַח אִשָּׁה וּשְׁמָהּ קְטוּרָה / וַיּוֹלֶד מִמֶּנָּה שִׁשָּׁה בָנִים, / וְהוּא בֶן יוֹתֵר מִמֵּאָה שָׁנִים. / שִׁמְעִי, בַּת, וּרְאִי כִּי רוּת הַמּוֹאֲבִיָּה / אֵשֶׁת חַיִל וְרַבַּת הָעֲלִילִיָּה / כָּמוֹנוּ הַיּוֹם יָרְדָה לַשְּׂעֹרִים / וְלֹא נָטְתָה לָלֶכֶת אַחֲרֵי הַבַּחוּרִים, / וּלְטוּב מִדּוֹתֶיהָ וְכִשְׁרוֹן פְּעָלֶיהָ / מְלָכִים יָצְאוּ מֵחֲלָצֶיהָ / וִיהַלֲלוּ בַשְּׁעָרִים מַעֲשֶׂיהָ״. / וַיִּשָּׂא מְשָׁלוֹ:


כָּל בַּת כְּבוּדָה לְפִי שִׂכְלָהּ מְיֻקֶּרֶת

וּכְפָעֳלָהּ כֵּן שְׁמָהּ יַעְמֹד לְמַזְכֶּרֶת,

כִּי אִם לְיָפְיָה וְנֹעַם תָּאֳרָהּ יָקְרָה

כִּיקָר תְּמוּנָה בְּקִיר הֵיכָל מְצֻיֶּרֶת.


״בִּתִּי, בִּתִּי! זֶה לִּי עֶשְׂרִים שָׁנָה בְּבֵיתִי גִדַּלְתִּיךְ, / עַל כַּפַּיִם חַקּוֹתִיךְ / וְעַל לוּחַ לִבִּי שַמְתִּיךְ / וְאַהֲבַת עוֹלָם אֲהַבְתִּיךְ. / יָגַעְתִּי בְּאַנְחָתִי לְמַלֹּאת מִשְׁאֲלוֹתַיִךְ – / הַנְּטִיפוֹת וְהַמַּעֲטָפוֹת וְרֵיחַ שַׂלְמוֹתַיִךְ / מֹר וַאֲהָלוֹת קְצִיעוֹת כָּל בִּגְדּוֹתַיִךְ – / וַאֲנִי בְעָנְיִי הֲכִינוֹתִי בָתֵּי הַנֶּפֶשׁ וְהַלְּחָשִׁים / וְתַמְרוּקֵי הַנָּשִׁים, / וַתִּרְבִּי וַתִּגְדְּלִי וַתָּבוֹאִי בַּעֲדִי עֲדָיִים / וָאַלְבִּישֵׁךְ מֶשִׁי וְצֶבַע רִקְמָתַיִם. / אַךְ עַתָּה הֶלְאוּנִי יְמֵי הַשֵּׂיבָה / וְתָשַׁשׁ כֹּחִי כִּנְקֵבָה, / מֵעֹנִי וּמֵרֹב עֲבוֹדָה, / אֲסַלְּדָה בְחִילָה, / וּבְבֵיתִי אֵין לֶחֶם וְאֵין שִׂמְלָה. / וְעַתָּה, בִּתִּי, שִׁמְעִי בְקוֹלִי, לְמַעַן יִיטַב לִי / וִיהִי שָׁלוֹם בְּחֵילֵךְ / וְחָיְתָה נַפְשִׁי בִּגְלָלֵךְ: / כִּי זֶה הָאִישׁ אֱלֹהִים שְׁלָחוֹ לְפָנֵינוּ לְמִחְיָה, / בָּרוּךְ שֶׁאָמַר וְהָיָה״. וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


הַצּוּר עֲלֵי כָל הַיְצוּר מַשְׁגִּיחַ

לֹא יַעְזֹב מַעְשָׂיו וְלֹא יַזְנִיחַ,

לִסְפֹּר תְּהִלּוֹתָיו וְכֹחַ מַעֲשָׂיו

כָּלוּ תְפִלּוֹתַי וְכָלָה שִׂיחַ.

הִפְלִיא חֲסָדָיו לִי וְעַל רֹב רַחֲמָיו

עָמָל וְרִישׁ מֵאָהֳלִי הִשְׁכִּיחַ,

תַּחַת הֱיוֹתִי אִישׁ שְׂבַע רֹגֶז – אֱהִי

הַיּוֹם שְׂבַע רָצוֹן וְאִישׁ מַצְלִיחַ!


וְדִינָה נִשְׁקְפָה בְּעַד הַחַלּוֹן / וַתֵּרֶא אֶת עֵפֶר יָכֹף רֹאשׁוֹ כְּאַגְמוֹן, / שְׁפַל רוּחַ וּמֵרִים קָלוֹן. / וַתֹּאמַר ״אָמוּתָה הַפַּעַם אַחֲרֵי רְאוֹתִי, / וְאַל אֶרְאֶה בְרָעָתִי! / הֲתִבְטַח כִּי רַב כֹּחוֹ / אִישׁ אֵין מַעְצָר לְרוּחוֹ / וַיּוֹרֶד עֹז מִבְטַחוֹ? / הֲתַאֲמִין כִּי יָשׁוּב לִימֵי עֲלוּמָיו – / וְאֵין שָׁלוֹם בַּעֲצָמָיו / וְנָפַל בַּעֲצוּמָיו? / כֻּלּוֹ כְּאַדֶּרֶת שֵׂעָר מְלֵאָה נוֹצָה, / עַל עַפְעַפָּיו צַלְמָוֶת וְאֹפֶל פָּנָיו כִּסָּה. / כֵּן יִהְיֶה תָמִיד – הֶעָנָן יְכַסֶּנּוּ, / חֲבוּר עֲצַבִּים וּמִי יִשָּׂאֶנּוּ? / חַכְלִילִי עֵינַיִם, / רְפֵה יָדַיִם, / נִשְׁבָּר וְנִדְכֶּה מִמָּתְנַיִם / וְעַד יְרֵכַיִם, / כְּחֹם הַיּוֹם שׁוֹכֵב בֵּין שְׁפַתַּיִם, / מִבֹּקֶר לָעֶרֶב לֹא יָקוּם פַּעֲמַיִם. / מְרֻבֶּה בְגָדִים וּמִקֹּר מִזְגּוֹ גְּוִיָּתוֹ תְהַלֵּךְ קוֹדֶרֶת – / וְאֶת הָאָח לְפָנָיו מְבֹעֶרֶת, / כָּל רוֹאֵהוּ יִדֹּד וְיַפְלִיא מְאֹד / אִם יוֹם אוֹ יוֹמַיִם יַעֲמֹד. / וְאֵיךְ הֶבֶל תְּנַחֲמֵנִי / וְלַעֲפַר מָוֶת תִּשְׁפְּתֵנִי! / וְאִם אָבִינוּ הָרִאשׁוֹן בִּגְדוֹלוֹת וּבְנִפְלָאוֹת נָשָׂא, / לָאו בְּכָל שַׁעֲתָא וְשַׁעֲתָא מִתְרַחֵשׁ נִסָּא. / וְגַם רוּת הַמּוֹאֲבִיָּה הָיְתָה אִשָּׁה אַלְמָנָה / וְכִפָּתָהּ לֹא רַעֲנָנָה: / וְהִיא גַם הִיא בִּימֵי בְחוּרוֹתֶיהָ שׂוֹשׂ תָּשִׂישׂ בְּאַלּוּף נְעוּרֶיהָ / וְכַכַּלָּה תַּעְדֶּה כֵלֶיהָ. / וַאֲנִי עֲנִיָּה אָנָה אוֹלִיךְ אֶת חֶרְפָּתִי / וְאָנָה מִפָּנֶיךָ אֶבְרַח לְהִנָּצֵל מֵרָעָתִי? / בָּרְכֵנִי גַם אָנִי, אָבִי, / בְּאִישׁ עוֹשֶׂה מְזִמּוֹת כַּאֲשֶׁר עִם לְבָבִי: / כִּי אִם עַנֵּה תְעַנֶּה אוֹתִי נֶגֶד אַחַי / לְחַלֵּל צְבִי עֶדְיִי, נְטַע שַׁעֲשׁוּעַי – / טוֹב מוֹתִי מֵחַיַּי!״ / וַתִּשָּׂא מְשָׁלָהּ וַתֹּאמַר:


דִּינָה לְנַפְשָׁהּ מַחֲנָק בָּחֲרָה

תִּצְעַק בְּמַר שִׂיחָהּ וְכֹה אָמְרָה:

אַיֵּה מְתֵי בִין חוֹשְׁבִים כִּי הָאֱמֶת

לֹא נֶעְלְמָה מֵהֶם וְלֹא נִסְתָּרָה?

אַיֵּה מְדַמִּים כִּי פְעֻלַּת תּוֹלְדוֹת

כָּל הַיְצוּרִים לֹא לְשָׁוְא חֻבָּרָה?

לוּ יֶחֱזוּ עָפְרָה לְעֵפֶר נִתְּנָה –

לֹא כִחֲשׁוּ בָהּ, כִּי לְרִיק נוֹצָרָה!

עפר נושא את דינה לאשה


וַיְהִי מִמָּחֳרָת וַיֶּאֱסֹף עֵפֶר אֶת כָּל זִקְנֵי עַמּוֹ / וְאֶת כָּל בְּנֵי שַׁעַר מְקוֹמוֹ, / לִשְׂמֹחַ בְּשִׂמְחַת חֻפָּתוֹ / וּבְיוֹם חֲתֻנָּתוֹ. / וַיֵּלְכוּ לְבֵית דִּינָה בְּתֻפִּים וּבִמְחוֹלוֹת / וְכָל הָעָם רוֹאִים אֶת הַקּוֹלוֹת. / וְכִשְׁמֹעַ דִּינָה אֶת קוֹל הָעָם בְּכִנּוֹרוֹת וּבִמְצִלְתַּיִם / וַיִּמַּס לְבָבָהּ וַיְהִי לְמַיִם, / וְהִיא יוֹשֶׁבֶת בֵּין הַהֲדַסִּים בְּקִשּׁוּרֵי עֶדְיָהּ / וְדִמְעָתָהּ עַל לֶחֱיָהּ. / וַיִּגַּשׁ עֵפֶר וַיָּשֶׂם אֶת הַטַּבַּעַת עַל יַד יְמִינָהּ / וַיֹּאמַר ״שִׁמְעִי נָא זֹאת, עֲדִינָה, / בְּטַבַּעַת זֹאת תְּהֵא לִי לְמִקְנָה״.

וַיָּקָם אֶחָד מֵהָעוֹמְדִים אֶצְלוֹ / וַיִּשָּׂא אֶת קוֹלוֹ / וַיֹּאמַר: בְּשִׁשִּׁי בְשַׁבָּת / קוֹל חֲתָנִים מִמִּשְׁתֶּה שָׁבַת, / בְּתִשְׁעָה יָמִים לְחֹדֶשׁ אָב / הַבְּתוּלוֹת תִּתְעַלַּפְנָה וּבְשָׂרָן עֲלֵיהֶן יִכְאַב. / אֵיךְ עֵפֶר דְּנָן הֶחָתָן / חִלֵּל יְקָרָן / וְחִבֵּל הֲדָרָן / וְשֹׁד פָּנָיו יְחִיתַן / אֲמַר לַהּ לְדִינָה אֲשֶׁר בְּתוֹךְ עֲלָמוֹת מְהֻלֶּלֶת: / הֲוִי לִי לְאִנְתּוּ וְהָיִית כְּאִשָּׁה זֶה יָמִים מִתְאַבֶּלֶת / וּכְאֵלָה עָלֶהָ נוֹבֶלֶת. / וַאֲנָא אוֹקִיר וַאֲזוּן וְאֶפְלַח – / נֵצֶר נִתְעָב וְנֶאֱלָח / וְרוּחוֹ כֶּעָשָׁן נִמְלַח, / וּמֵעַל לְוָתַיְכִי כְּאֹרַח כָּל אַרְעָא – / אַל תַּאֲמֵן בַּשָּׁוְא נִתְעָה. / וְדָא נְדוּנְיָא דְּהַנְעָלַת לֵהּ כַּלְּתָא: / קָל קַרְנָא מַשְׁרוֹקִיתָא, / וַתַּעֲבֹר הַמִּנְחָה עַל פָּנָיו – / קֶרֶן חָזוּת בֵּין עֵינָיו. / וְעוֹד הֵבִיאָה לוֹ בִּנְדוּנְיָא: שַׁמָּה וּשְׁאִיָּה / וְהַהֶעְדֵּר הַקּוֹדֵם לַהֲוָיָה, / וּלְהָסִיר הַשְּׁחִין מֵעַל גַּבּוֹ – / חֶרֶשׂ לְהִתְגָּרֵד בּוֹ. / וְתַחַת הַסְּדִינִים – / פַּחִים וְצִנִּים, / וְתַחַת הַגִּלְיוֹנִים – / רֹבַע הַקַּב חִרְיוֹנִים. / וּצְבִי הֶחָתָן וְהוֹסִיף לַהּ מִדִּילֵהּ / פֶּגֶר מוּבָס כְּרָקָב יִבְלֶה / וּכְסָלָיו מָלְאוּ נִקְלֶה, / וְלֵב נִשְׁבָּר וְנִדְכֶּה, / נָגוּעַ וּמֻכֶּה / בְּשִׁבְרוֹן מָתְנַיִם / וְכָל מַדְוֵי מִצְרַיִם / וּשְׁחִין פּוֹרֵחַ עַל הַבִּרְכַּיִם / וְעַל הַשּׁוֹקַיִם. / סַךְ הַכֹּל בֵּין כְּתֻבָּה נְדוּנְיָא וְתוֹסֶפֶת / אֶת הַקַּדַּחַת וְאֶת הַשַּׁחֶפֶת / אֶת הַסַּפַּחַת וְאֶת הַבַּהֶרֶת / אֶת הַמְּהוּמָה וְאֶת הַמִּגְעֶרֶת / וְצָרַעַת מַמְאֶרֶת. / וּקְנֵינָא מִנֵּהּ בְּמָנָא דִכְּשַׁר לְמִקְנְיָא / כִּי בִמְהֵרָה יָשׁוּב אֶל הָאָרֶץ כְּשֶׁהָיָה, / וְכֵן קִבֵּל עַל עַצְמוֹ בְּדֶרֶךְ פְּשָׁרָה / דְּלֶהֱוֵי בְטִיל וַחֲשִׁיב כְּעַפְרָא. / וְאִלּוּ פִינוּ מָלֵא שִׁירָה כַיָּם, / אֵין דַּי לְהַעֲלוֹת עַל סֵפֶר כֹּבֶד מַכְאוֹבָיו וְעֹז דָּכְיָם – / וְהַכֹּל שָׁרִיר וְקַיָּם. / וַיִּכְתְּבוּ בְסֵפֶר הֲדַדְרִמּוֹן / בֶּן טַבְרִמּוֹן / מֵרַבַּת בְּנֵי עַמּוֹן, / וַיָּעִידוּ עָלָיו עֵדִים: פַּגְעִיאֵל בֶּן עָכְרָן / וְעוּץ וַאֲרָן / וְאֶשְׁבָּן וְיִתְרָן וְחֶמְדָּן וּכְרָן. / וַיְשׁוֹרֵר:


רְאוּ חָתָן שְׂבַע רֹגֶז וְחֶרְפָּה / וְחֶרְפָּתוֹ בְחֻפָּתוֹ מְחֻפָּה,

וְיַעְלָתוֹ כְּלִילַת חֵן בְּצִדּוֹ – / כְּבָרֶקֶת אֲשֶׁר נִמְצָא בְאַשְׁפָּה!


וַיְשׁוֹרֵר עוֹד:


חָתָן נְכֵה רוּחַ, וּפָנָיו קִבְּצוּ / פָארוּר, וּבֵין עֵינָיו תְּמַשֵּׁשׁ חֹשֶׁךְ,

נֶגְדּוֹ יְפַת תֹּאַר כְּאוֹר שֶׁמֶשׁ, וּבָם / יַבְדִּיל אֱנוֹשׁ בֵּין אוֹר וּבֵין הַחֹשֶׁךְ.



וַיַּעַשׂ עֵפֶר מִשְׁתֶּה גָּדוֹל / לְאֶחָיו וּקְרוֹבָיו וּלְכָל יוֹדְעָיו מֵאֶתְמוֹל / וַיֹּאמֶר לַאֲשֶׁר עַל בֵּיתוֹ בְּשִׂמְחָה וּבְשָׂשׂוֹן / ״הָרוֹג בָּקָר וְשָׁחוֹט צֹאן!״ / וַיִּתֵּן לְכָל הַנִּמְצָאִים מַתָּנוֹת / וּמִשְׁלוֹחַ מָנוֹת, / לְכֻלָּם נָתַן אֲשִׁישָׁה וְאֶשְׁפָּר / וְחָדַל לִסְפֹּר כִּי אֵין מִסְפָּר. / וַיֵּלְכוּ לְבֵיתָם אַחֲרֵי אָכְלָם וְאַחֲרֵי שׁוֹתָם / וַיְבָרְכוּ אֶת עֵפֶר וְאֶת דִּינָה אֲחוֹתָם, / אִישׁ כְּבִרְכָּתוֹ בֵּרַךְ אוֹתָם.

וַיְהִי בַלַּיְלָה וַיִּקַּח עֵפֶר אֶת דִּינָה אִשְׁתּוֹ / וַיְבִיאֶהָ הַחַדְרָה אֶצְלוֹ לַעֲשׂוֹת מְלַאכְתּוֹ / וַיְחַבְּקֶהָ וַיְנַשְּׁקֶהָ וּלְפַיְסָהּ בִּדְבָרִים הִרְבָּה / וַיְדַבֵּר עַל לִבָּהּ. / וַיֹּאמֶר אֵלֶיהָ ״אֲחוֹתִי רַעְיָתִי, מְשׂוֹשִׂי וְגִילִי! / שִׁיתִי לִבֵּךְ לְחֵילִי, / יֶשׁ לִי רַב לְהָפִיק מִשְׁאָלֵךְ – / מֶשִׁי אַלְבִּישֵׁךְ וְתַחַשׁ אֶנְעֲלֵךְ, / וּבַאֲשֶׁר תֵּלְכִי אֵלֵךְ. / וְאָנֹכִי פִּנִּיתִי הַבַּיִת / מִחוּץ וּמִבַּיִת / וְלָשׂוּשׂ עָלַיִךְ כְּחֶפְצִי / מַרְבַדִּים רָבַדְתִּי עַרְשִׂי״. וַיִּשָּׂא מְשָׁלוֹ:


עֶרֶשׂ יְקָר תּוֹךְ אָהֳלִי רָבַדְתִּי,

עָלָיו רְפִידַת אַהֲבָה רָפַדְתִּי.

גַּם מָר דְּרוֹר נַפְתִּי בְמִשְׁכָּבִי, וּבוֹ

לִבּוֹת אֲסִירֵי תַאֲוָה עוֹרַרְתִּי,

שִׁכְבִי, צְבִיַּת חֵן, וְאֶתְעַלֵּס בְּהוֹד

מַרְאֵךְ וְחֵן יָפְיֵךְ אֲשֶׁר חִמַּדְתִּי!


וַיִּשְׁכַּב עֵפֶר וַיֶּאֱסֹף אֶל הַמִּטָּה רַגְלָיו / וְתַרְדָּמָה גְדוֹלָה נוֹפֶלֶת עָלָיו, / כָּל הַלַּיְלָה שָׁכַב עָיֵף וְיָגֵעַ / וְאֵין מֵקִיץ וְאֵין יוֹדֵעַ. / וְדִינָה בִּפְאַת הַמִּטָּה שָׁכְבָה לְמַעֲצֵבָה / וְלֹא יָדַע בְּשִׁכְבָהּ. / וַתֹּאמֶר דִּינָה בְּלִבָּהּ: / דָּנַנִי אֱלֹהִים מִשְׁפַּט מָוֶת בָּאִישׁ הַזֶּה, / כָּל הַלַּיְלָה אוֹהֵב לָנוּם שׁוֹכֵב הוֹזֶה, / נִמְאָס בְּעֵינַי וְנִבְזֶה. / וַתִּשָּׂא מְשָׁלָהּ:

דִּינָה בְחֵיק שֵׂיבָה שָׁכְבָה לְמַעְצֵבָה

וּבְמַעֲטֵה זֹקֶן עוֹטְיָה וְנִכְלֶמֶת,

תִּשְׁכַּב בְּחֵיק אִישׁ רַע זָקֵן וְגַם מָאוּס

רוּחוֹ בְיַד שֵׁנָה נְתוּנָה – וְנִפְעֶמֶת,

דָּמְתָה בְּבֵית עֵפֶר אֶל שׁוֹכְבֵי עָפָר –

אֵל נָא, רְפָא נָא לָהּ, אַל נָא תְהִי כַּמֵּת!


ד

קצו העגום של עפר


וַיִּשְׁמַע עֵפֶר וַיִּתְעַצֵּב אֶל לִבּוֹ / וְלֹא יָכֹל לַעֲנוֹת מֵרָגְזוֹ וּמֵעָצְבּוֹ. / וַיֹּאמַר: עַתָּה יְרַנֲנוּ בְקָהָל רַב בָּשְׁתִּי וּכְלִמָּתִי / וְעַל פְּנֵי כָל חוּצוֹת יַגִּידוּ חֶרְפָּתִי / וְיָגִילוּ מִפְּנֵי רָעָתִי. / וְאֶל מִי אֶשָּׂא תְלֻנָּתִי? / נוֹקַשְׁתִּי בְּפַחֵי מְזִמָּתִי, / נִלְכַּדְתִּי בְּרֶשֶׁת תַּאֲוָתִי, / בְּקִבְרוֹת הַתַּאֲוָה פָּרַשְׂתִּי פַח לְמַעְגָּלִי / וְרֶשֶׁת טָמַנְתִּי לְרַגְלִי / בְּקִבְרִי אֲשֶׁר כָּרִיתִי לִי. / הַצֳּרִי אֵין בְּגִלְעָד לְמַחַץ מַכָּתִי? / אִם רוֹפֵא אֵין שָׁם לְהַעֲלוֹת אֲרוּכָה לְמַחֲלָתִי? / וְאִם בְּרֹב חֵילִי וְעָשְׁרִי / לֹא יִמָּצֵא מָזוֹר לִמְזוֹרִי / וּצְרִי לְצִירִי? / אוֹי לִי עַל שִׁבְרִי! / זֹאת הָיְתָה לִּי, / כִּי שַמְתִּי זָהָב כִּסְלִי / בְּרֹב אִוַּלְתִּי וְסִכְלִי / וְהוּא עָווֹן פְּלִילִי. / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


שָׂמוּ יְלִידֵי הוֹלֲלוּת כִּסְלָם – זְהַב

חֵילָם, וּמִבְטֶחָם – בְּרַצֵּי כֶסֶף.

הֵם בָּטְחוּ עַל רִיק, וְלָהֶם נֶאֱמַר

״לֹא תַעֲשׂוּן אִתִּי אֱלֹהֵי כֶסֶף!״


וַיֹּאמֶר חַרְבוֹנָה אֶחָד מֵעֲבָדָיו / הַיּוֹשְׁבִים בְּצִדָּיו / ״יְצַוֶּה אֲדוֹנִי לְבַקֵּשׁ וְלַחְקֹר / מֵאַיִן יִמָּצֵא הַדָּג שֶׁנִּקְרָא סְקַנְקוֹר / כִּי יֶשׁ לוֹ כֹּחַ יָפֶה וּמְסֻגָּל / לְעוֹרֵר תַּאֲוַת הַמִּשְׁגָּל, / בְּחֻמּוֹ יָעִיר חֹם הַטֶּבַע הַנִּכְבֶּה / וּלְאֵין אוֹנִים עָצְמָה יַרְבֶּה. / הָאִישׁ הָאוֹכֵל מְעַט מִטַּבּוּרוֹ / יֵצֵא חָתָן מֵחֶדְרוֹ. / וְשִׁעוּר הַלְּקִיחָה לָרֹב / כִּשְׁנֵי גַרְעִינֵי חָרוֹב, / וּבְיַיִן חַי יִשְׁתֶּה / וְכָל הַיָּשָׁר בְּעֵינָיו יַעֲשֶׂה: כִּי בָזֶה דַי לְמַלֹּאת תַּאֲוַת הָאִשָּׁה עַד פְּעָמִים שִׁשָּׁה. / אוּלָם יִשָּׁמְרוּ מִטַּבּוּרוֹ וּלְמַעְלָה / כִּי הוּא חָזָק מְאֹד וְאוּלַי יָבִיא לִידֵי מַחֲלָה״.

וַיֹּאמֶר עֵפֶר לְאֶחָד מֵהַנְּעָרִים ״לְבַקֵּשׁ הַדָּג הַזֶּה קוּם צֵא״ / וַיְבֻקַּשׁ הַדָּבָר וַיִּמָּצֵא. / וַיִּשְׂמַח עֵפֶר שִׂמְחָה גְדוֹלָה / (וּבִמְקוֹם גִּילָה שָׁם תְּהֵא רְעָדָה וְחִילָה!) / וּלְהַרְאוֹת אֶת תָּקְפּוֹ, / לָקַח לוֹ כְזַיִת מִמּוּל עָרְפּוֹ, / לִהְיוֹתוֹ יוֹתֵר תַּקִּיף וְחָזָק, / כִּי אָמַר: הַזָּקֵן אֲשֶׁר מִזְגּוֹ קַר בָּזֶה לֹא יֻזַּק. / וַיֹּאמַר: בַּלַּיְלָה הַזֶּה תִּהְיֶה נַפְשִׁי עֲדִינָה / וְאוֹצִיא לָאוֹר מִשְׁפָּטָהּ וְדִינָהּ / בְּתִתִּי לְדִינָה / חֵלֶק לְשִׁבְעָה וְגַם לִשְׁמֹנָה. / וְהוּא לֹא יָדַע כִּי בְזֶה הַדָּג כֹּחַ אַרְסִיִּי / יוֹרִישׁ לְאוֹכְלָיו אֵשׁ פַּרְסִיִּי. / וַיֹּאכַל מִמֶּנּוּ וַיִּמָּלֵא כָל גּוּפוֹ נְגָעִים גְּדוֹלִים / וַיֹּאחֲזוּהוּ מַכְאוֹבִים וַחֲבָלִים, / כָּל הַלַּיְלָה הַהוּא צוֹעֵק וּמֵהִים / וַיְהִי לְחֶרְדַּת אֱלֹהִים, / וְתַחַת מַעֲשֵׂה מִקְשֶׁה / הָיָה כְאֵבוֹ נֶצַח וּמַכָּתוֹ אֲנוּשָׁה. / וַיָּמָת הָאִישׁ בְּצִירִים כְּצִירֵי יוֹלֵדָה / וַיֵּלֶךְ בְּלֹא חֶמְדָּה.

וַיִּשְׁמַע שַׂר הָעִיר אוֹדוֹת עֵפֶר וְעִנְיָנָיו / וַיִּתֵּן עֵינָיו לָקַחַת אֶת הוֹנוֹ וְקִנְיָנָיו / וַיֹּאמֶר לְאַנְשֵׁי בֵיתוֹ ״אֶת הָאִישׁ הַזֶּה הֱמִתֶּם / וְכוֹס רַעַל אוֹתוֹ הִשְׁקִיתֶם, / הֲרֵעוֹתֶם אֲשֶׁר עֲשִׂיתֶם! / וְעַתָּה, חַי יְיָ, כִּי בְנֵי מָוֶת אַתֶּם, / כִּי אֶת אֲדוֹנֵיכֶם לֹא שְׁמַרְתֶּם, / וְאַתֶּם הֵאָסְרוּ בְּבֵית מִשְׁמַרְכֶם / וְיִבָּחֲנוּ דִּבְרֵיכֶם. / וְעַל כָּל אֲשֶׁר לוֹ – יֶשׁ לִי זְכוּת, / כִּי כֵן חֹק הַמַּלְכוּת, / וּמִשּׁוֹר עַד חֲמוֹר וְעַד שֶׂה / לֹא אֶחְמֹל וְלֹא אֲכַסֶּה״. / וַיִּתֵּן אֶת כָּל אוֹהֲבָיו וּקְרוֹבָיו בְּבֵית הָאֲסוּרִים / וַיָּשֶׂם עֲלֵיהֶם עֲלִילוֹת דְּבָרִים / וַיְעַנֵּם בְּכַבְלֵי בַרְזֶל וּמִינֵי יִסּוּרִים. / עַל כֵּן אָמַר הֶחָכָם לִבְנוֹ: / אוֹי לָרָשָׁע וְאוֹי לִשְׁכֵנוֹ! / וְכָל חֵיל עֵפֶר הָיָה לְבַז וּלְשַׁמָּה / וְלֹא נוֹתַר לְנָבָל מִכָּל אֲשֶׁר לוֹ מְאוּמָה, / בְּרֹעַ מַעֲלָלָיו עָלָיו וְעַל יוֹרְשָׁיו אַחֲרָיו רָעָה הִדִּיחַ – / וְלָנוּ וּלְכָל יִשְׂרָאֵל שָׁלוֹם הִנִּיחַ. / עַל כֵּן אָמַר הַמְּשׁוֹרֵר:


לִבּוֹת מְתֵי כֹשֶׁר בְּמוֹתָם נָחוּ / וַהְמוֹן לְאֻמִּים בַּעֲדָם נֶאְנָחוּ,

אָכֵן בְּמוֹת עַוָּל לְבָבוֹ יֶאֱבַל / עָלָיו – וְהָעָם אַחֲרָיו יִשְׂמָחוּ!
