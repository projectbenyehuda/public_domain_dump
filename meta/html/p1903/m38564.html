משל קה: חֲזִיר וַאֲרִי וְשֻׁעָל

הַמַּכְבִּיד לִבּוֹ לְפִי שָׁעָה / יִהְיֶה לוֹ תָמִיד רָעָה

חֲזִיר יַעַר הֹלֵךְ וְשָׁב כְּדַרְכּוֹ / בַּחֲצַר אֲרִי הַמֶּלֶךְ וּבְתוֹךְ סֻכּוֹ / וְהָאֲרִי אָמַר לוֹ: מַה לְּךָ אֶצְלִי / אַל תּוֹסֵף לָבֹא עוֹד בִּגְבוּלִי / וְלֹא תְטַמֵּא מִשְׁכָּנִי אֲשֶׁר חָמַדְתִּי / וַיַּעַן לוֹ הַחֲזִיר: אֲדֹנִי שָׁמַעְתִּי / וַיְהִי בַבֹּקֶר וְהִנֵּה הַחֲזִיר מִתְהַלֵּךְ / סְבִיבוֹת סֻכַּת אֲרִי הַמֶּלֶךְ / וַיִּזְעַף אֵלָיו וַיִּכְרֹת אֶחָד מֵאָזְנָיו / וַיִּקְרָא אֶל הַשֻּׁעָל וַיֹּאמֶר לוֹ בְּפָנָיו / אִם הַחֲזִיר לֹא יוּסַר מֵעָלָיו / לֹא תַחְמֹל וְלֹא תְכַסֶּה עָלָיו / כִּי סֻכָּתִי אֲשֶׁר מַחְסֶה מִזֶּרֶם וּמִסָּעַר / יְכַרְסְמֶנָּה חֲזִיר מִיָּעַר / גַּם כִּי נָכְרִיָּה עֲבֹדָתוֹ / וְעַתָּה אַל תְּנַקֵּהוּ וַהֲמִיתוֹ / וְהַחֲזִיר הִכְבִּיד לִבּוֹ וַיֶּקֶשׁ עָרְפּוֹ / וְהָאֲרִי בָחֳרִי אַפּוֹ / כָּרַת מֵרֹאשׁוֹ אָזְנַיִם / וַיְנַקֵּר אֶת הָעֵינַיִם / וְעוֹדֶנּוּ מַכְעִיסוֹ בְּגֶלְלֵי צֵאָתוֹ / וַיְטַמֵּא סְבִיבוֹת סֻכָּתוֹ / וַיֹּאמֶר הָאֲרִי אֶל הַשֻּׁעָל: סֹב פְּגַע בּוֹ / וְאֶת רֹאשׁוֹ וְאֶת כְּרָעָיו וְקִרְבּוֹ / וְכָל נֵתַח טוֹב כָּתֵף וְיָרֵךְ / שִׂים לִקְרָאתִי עַל הַדֶּרֶךְ / וְהַשֻּׁעָל לִנְתָחָיו נִתְּחוֹ / וַיַּרְא כִּי לִבּוֹ שָׁמֵן וּלְקָחוֹ / לְדֵעָה מַה יֵּעָשֶׂה לוֹ / וְלֹא יָכֹל לְהִתְאַפֵּק וַאֲכָלוֹ / וַיַּעֲרֹךְ עֵרֶךְ אֵבָרָיו / וְהָאֲרִי בָא לִרְאוֹת בְּתָרָיו / וַיַּרְא אֲשֶׁר מֵהֵנָּה לֹא נֶעְדָּרָה / כָּל עַצְמוֹתָיו אַחַת מֵהֵנָּה לֹא נִשְׁבָּרָה / אַךְ מְצָאָם כְּסֵדֶר / אֶת הָרֹאשׁ וְאֶת הַפֶּדֶר / וּבְהַגִּיעוֹ לְקִרְבּוֹ / לֹא מָצָא אֶת לִבּוֹ / וַיֹּאמֶר בָּחֳרִי אֶל הַשֻּׁעָל / אֵיךְ בַּאֲדֹנֶיךָ תִּמְעֹל מָעַל / אַיֵּה לֵב הַחֲזִיר אֲשֶׁר מָחַצְתָּ / וְאֵיךְ לְהַסְתִּירוֹ מִמֶּנִּי חָפַצְתָּ / וְעַתָּה אָשִׁיב לְךָ כִּגְמוּלָךְ / כַּאֲשֶׁר עָשִׂיתָ כֵּן יֵעָשֶׂה לָךְ / וְהַשֻּׁעָל בְּחָכְמָתוֹ עָנָה / אֲדֹנִי לֹא עָלַי הַתְּלֻנָּה / אַךְ אֶל אֲמָרַי הַטֵּה אָזְנֶךָ / הֶן זֹאת לֹא צָדַקְתָּ אֶעֱנֶךָּ / תָּבִין בְּמִרְיוֹ וְתַשְׂכִּיל בְּסִכְלוֹ / כִּי לֵב בְּקִרְבּוֹ לֹא הָיָה לוֹ / וּבְעָרְכִי אֵבָרָיו לְעֻמָּתִי / יָגַעְתִּי וְלֹא מָצָאתִי / עַל כֵּן בִּדְבַר הַמֶּלֶךְ לֹא יִוָּסֵר / וְגַם בַּדֶּרֶךְ כְּשֶׁהַסָּכָל הֹלֵךְ לִבּוֹ חָסֵר / וּבְחֹסֶר לְבָבוֹ לָקַחְתָּ אָזְנָיו וְעֵינָיו / כַּאֲשֶׁר עַל מִצְוָתְךָ הֵעֵז פָּנָיו / וַאדֹנִי הַמֶּלֶךְ חָכָם יִתְהַלֵּךְ בְּמֵישָׁרִים / וְאַל יָשִׂים עַל עַבְדּוֹ עֲלִילוֹת דְּבָרִים / וַיַּעַן הָאֲרִי: שָׁמַעְתִּי בְּקוֹלָךְ / אִם חָכַמְתָּ חָכַמְתָּ לָּךְ / וְצַדִּיק אַתָּה מִמֶּנִּי / עַל כֵּן כָּל יָמַי תֵּשֵׁב לִימִינִי / אֵין חָכָם כָּמוֹךָ וְשִׁמְךָ אֲכַנֶּה נָעֳמִי / אַתָּה תִּהְיֶה עַל בֵּיתִי וְעַל פִּיךָ יִשַּׁק כָּל עַמִּי.


וְהַמָּשָׁל אִם יִצְרְךָ תּוֹקֶפְךָ לַעֲשׂוֹת שֶׁלֹּא כְהֹגֶן / חַפֵּשׂ בְּשָׁרְשֵׁי מְזִמּוֹתֶיךָ לְהִנָּצֵל כְּאִישׁ מָגֵן / וְאַל תְּקַנֵּא בַאֲנָשִׁים אֲשֶׁר לִבָּם וְאָזְנֵיהֶם הָעֳרָלוּ / בְּצֵאתָם מִמִּצְרַיִם לֹא מָלוּ / כִּי אִישׁ אֲשֶׁר לִבּוֹ עָרֵל וְאָזְנוֹ עֲרֵלָה / רַעְיוֹנָיו הָרָעִים צִיצִין הַמְּעַכְּבִין אֶת הַמִּילָה / אַךְ רְאֵה דֶרֶךְ אֲשֶׁר תֵּצֵא מֵחַטָּאתְךָ לָרְוָיָה / וְאָז נַפְשְׁךָ תִּכּוֹנֵן וָחָיָה / כִּי חָכְמָה תְּחַיֶּה הַבְּעָלִים / וּמַרְפֵּא יַנִּיחַ חַטָּאִים גְּדוֹלִים / וַאֲנִי מָצָאתִי בְּסֵפֶר הַנְּסִיכִים / הַמְּלָכִים שֹׁפְטֵי אָרֶץ וְהַחֲכָמִים שֹׁפְטֵי הַמְּלָכִים.
