יִשְׁמָעֵאל־בֵּי: שירה מזרחית

פְתִיחָה

שׁוּב הֵחֵל רוּחַ לְפַעֲמֵנִי,

שׁוּב נִנְעַר בִּי הַלֵּב הַקָּר,

וַיְהִי לְמִזְמוֹר דְּמִי-יְגוֹנִי,

מַאֲוַיַּי בִּי, שֶׁשָּׁמְמוּ כְּבָר.

כֵּן בּוֹדְדָה לָהּ בִּישִׁימוֹן זָר,

מֵאֵין כָּל נִשְׁמַת-חַי מַקְשֶׁבֶת,

עַל עַרְעָר צִחֵה-יוֹם יוֹשֶׁבֶת

זֹה צִפּוֹר-עֵדֶן, צִפּוֹר-דְּרוֹר,

וּתְנוֹסֵס כְּנָפָהּ – תְּכֵלֶת-אוֹר.

וְלוּ יִסָּעֵר מֵעֲבָרִים

בְּקוֹל יְלֵל וְסוּפַת-קֹר –

לָהּ אֶחָד שִׁיר עַל שֶׁמֶש-דָּרוֹם.

גַּם אַתְּ, כּוֹכֶבֶת אַהֲבַת-תֹּם,

תּוֹמֶכֶת חַיֵּי-זַעְפִּי, שִׁמְעִי

שִׁירָתִי זֹאת מִתְּמוֹל-שִׁלְשׁוֹם:

חֲדָשָׁה, הָהּ, זֶה-כְּבָר אֵין עִמִּי.

עוֹד יַד-הַזְּמַן לֹא חִבְּלָה בִי

לֹא קְדוֹשׁ הֶגְיוֹנִי, גַם עַצְּבוֹתָי:

אַךְ אַת צְלִיל-אֹשֶר בִּנְגִינוֹתָי,

וּבְאַחֵר צְלִיל יִוָּדַע שְׁמִי…


חלק ראשון

א

שָׂא שָׁלוֹם רָב, קְשִׁישׁ-עַד, קַוְקַז הֹר-הָר!

בֵּין בָּמוֹת-עֻזְּךָ תָּר אֲנִי לֹא זָר:

בְּשַׁחֲרִית-חַיַּי הֵן בְּחֵיק נִטְּלוּנִי

וּלְמַרְאֵה שְׁמֵי-יְשִימוֹת הִרְגִּילוּנִי.

מֵאָז בַּחֲלוֹם אֶחֱזָיָה יוֹם וָלֵיל

אַךְ שְמֵי-הַדָּרוֹם וְצוּקֵי הַרְרֵי-אֵל.

מַה-נֶהְדָּר אַתָּה, נוֹף בֶּן דְּרוֹר וָזָעַף,

וְאַתֶּם, כִּסְאוֹת-עוֹלָם, בָּמוֹת-רָהַב,

בִּדְאוֹת מִמֶּרְחָק עִם עֲרֹב הַיּוֹם

אֲלֵיכֶם עָבֵי עֶשֶׁן-תְּכֵלֶת, דֹּם

תִּרְעַדְנָה עַל רָאשֵׁיכֶם כְּשִׁיחַ-צְלָלִים

וּכְנוֹצוֹת-הוֹד עַל קוֹבְעֵי עִנְקֵי-סְמָלִים,

וְרָם עַל רָמִים שָלֵו צָף עַל-פְּנֵי

שַׁפְרִירִים כְּחֻלִּים הוֹד-יָרֵחַ גֵּא!


ב

וּמָה אָהַבְתִּי אָז, קַוְקַז נֶאְדָּרִי,

אֶת אֵשׁ-הַקְּרָב בְּלֵב הָעָם הַהֲרָרִי!

וּתְכֵלֶת עִמְקֵי רוּם וּמְגוֹר-הַיְלֵל

שֶׁל סוּפוֹת-רֶגַע-קָט, בְּתֵת הֵד-קוֹל

מְעָרַת-צוּר לִמְעָרָה, הַר לֶהָרִים,

כִּקְרֹא קוֹל שׁוֹמְרֵי-לַיְלָה מֵעֲבָרִים.

וּפִתְאֹם יָצִיץ שֶׁמֶשׁ, וְהַיְאוֹר

גִּיל-זָהָב יַעַט, רֹאשוֹ מֹר

יְנַשֵּׂא צִיץ-הַשָּׂדֶה מוּל קַרְנָיִם

וְעָלֵז כְּפֶרַח עֵדֶן וְשָׁמָיִם…

בִּשְׁעוֹת עֲרָבִים דֹּם אֶתְבּוֹנֵן לִי

בִּירִיעַת עָבִים פְּרוּמָה שָׁם, וְהִיא,

חַכְלִילַת-חוֹפִים וַחֲבַצֻּלַת-צְבָעִים,

תִּצָּבֵר בְּרֹגֶז עַל מָעֻזְנֵי סְלָעִים –

וְטִירַת-כְּשָׁפִים, פֶּלֶא-קְדוּמִים, רְאֵה,

לְפֶתַע צָצָה; אַךְ חִישׁ-קַל – מֵאֵי

זֶה רוּחַ שׁוֹבָב נָשַׁב בָּהּ – וַתְּסֹעַר…

כֹּה יַפְסִיק צְלִיל-הַקּוֹלָר בְּבֵית-הַבּוֹר

הֶגְיוֹנֵי אַסִּיר דָּךְ, הֲגִיג-הַזֹּהַר

עַל תַּלְמֵי שְׂדֵי-מוֹלַדְתּוֹ, שְׂדֵי-דְרוֹר…

בֵּין-כֹּה מַעֲרָבָה יִנְהֲרוּ עֲנָנִים

אֲחֵרִים, וְהֵם מֵהַרְרֵי-שֶׁלֶג לְבָנִים,

יִלַּוּוּ יוֹם – וְנִצְפָּף אָח אֶל אָח,

וְאָח מֵאַחֲרֵי אָח אָז יַבִּיט, כֻּלּוֹ צַח,

וְכֹה שְׁלֵוִים הֵם וּמְדֻשְּׁנֵי-אוֹר, כְּאִלּוּ

עַד-עוֹלָם חָיֹה יִחְיוּ כֹה, יָגִילוּ.


ג

וּסְפוּנִים בָּעֲרוּצִים שִׁבְטֵי עָם,

הַדְּרוֹר לוֹ אֵל, וְתוֹרָתוֹ – קְרַב-דָּם.

תּוֹךְ שֹׁד וְחָמָס יַפְרִיא הַהֲרָרִי,

תּוֹךְ עִנְיַן-חַיִּים מוּזָר וְאַכְזָרִי.

שָׁם עוֹלָל עוֹד בְּעַרְשׂוֹ תְּבַעֲתוֹ אֵם:

“יְשַן, הָרוּסִי בָּא!” לְחָגָא זֶה הַשֵּׁם.

שָׁם מַכֵּה אוֹיְבוֹ – עָוֶל אֵין בְּכַפּוֹ;

שָׁם רֵעַ – בְּרִיתוֹ עַד, אַךְ אָרוּר אַפּוֹ;

טוֹב תַּחַת טוֹב, דָּם תַּחַת דָּם,

וֶאֱהֹב גַּם שְׂנֹא – אֵין חֵקֶר לָמוֹ שָׁם.


ד

אֲפֵלוֹת שָׁם אַגָּדוֹת. צֶ’צֶ’ן עָנִי

מִילִידֵי דַבְּשוֹת-קַזְבֵּק עֵת לִוָּנִי

בַּדֶּרֶךְ בֵּין הֶהָרִים – הוֹאִיל שָׂח

לִי סִפּוּר מִנִּי-קֶדֶם. זֶה הַמָּךְ

תְּהִלַּת דוֹר-אֲבוֹתָיו הִשְׁמִיעַנִי,

אֶל תַּחַת צוּק רֻסְלַמְבֵּק הוֹלִיכַנִי:

מֵעַל לִנְתִיב עֲקַלָּתוֹן, כְּמוֹ

בְּעֶצֶם נָפְלָהּ דְּבַר אַלַּהּ עֲצָרָהּ,

כְּגִגִּית כְּפוּיָה עָמְדָה אֶבֶן זֹה.

אַדֶּרֶת-יֶרֶק עָטְפָה לְתִפְאָרָה;

וּבְצִלָּהּ, חָבוּי-לוֹ מִדֶּלֶף קָר,

מִבְּלִי יְגֹר, פֶּן יִמּוֹט צוּק מֵהַר

יָפְיָפֶה מִנִּי-תְכֵלֶת עֵין-הַבָּנוֹת

שֶׁבְּאַרְצוֹת-צָפוֹן הַמָּשְׁלָגוֹת-לְבָנוֹת,

וְנִשְׁעָן עַל גִּבְעוֹלוֹ כְּחֹם הַיּוֹם –

שָׁם צִיץ-מַזְכֶּרֶת יִפְרַח בְּעַנְוַת-דֹּם.

וְתַחַת צוּק-הַקְּדוּמִים, צוּק-יֵרָקוֹן,

לְפָנַי יָשֹׁב יָשַׁב זֶה הַזָּקֵן;

הוּא קָדַר, שָׂב כִּפְנֵי חַלָּמִיש קָר.

וּמוּרַד-רֹאש דֹּם הָגָה דָבָר מָר…

הַעַל מוֹלַדְתּוֹ שָׁפַךְ לֵב בְּצִנְעָה?

אָנֹכִי אָז הַפְרִיעוֹ לֹא אָהִינָה;

אַךְ דְּמִי הַשָּׂב מִשְּׁתִיקַת סַלְעֵי-עַד

לְהַבְחִין בַּל יָכֹלְתִּי אָז כִּמְעַט…


ה

זֶה סִפּוּר הָעֱזוּז וְהָעִצָּבוֹן

אָמַרְתִּי: אָבִיא-נָא עַד יַרְכְּתֵי-צָפוֹן,

וִיהִי גַם זָר וּמוּזָר לָנוּ פֹה –

כַאֲשֶׁר קִבַּלְתִּיו, אֶמְסְרֶנּוּ כֹה.

כִּי צַר לִי, אִם כְּרָז-מַשָּׁאוֹן

עִמָּדִי, גֵּר-הָעוֹלָם, דְּבָרַי דֹּם יִגְוָעוּן.

וְאַף אִם לֹא יַקְשִׁיבוּ לִי – אֲנִי

עַד תֹּם אֲסַפֵּר. מִי לוֹ נִשְׁמַת-גָּאוֹן,

לֹא יִרְדֹּף אַחֲרֵי זָר וְשֵׁם שַׂגִּיא,

רַק שִׁיר וְאַהֲבָה – רַק בָּם יְחִי

הַמְשׁוֹרֵר, בִּלְתָּם – חַיָּיו פֹּה עֲגוּמִים-שְׁחוֹרִים,

כְּעֶרְיַת שַׁחַק בְּלִי כָּל עָב וּמְאוֹרִים…


ו

מִקַּדְמֵי-דוֹר, עַל חוֹף, בִּמְקוֹם

בֵּין סַלְעֵי-צוּר תַּת-קוּם קוֹלֵחַ,

מִצַּד מַשּׁוּק יִנָּשֵׂא יוֹם

וּלְקִצְבֵי בֶּשְׁטוּ הוּא צוֹנֵחַ,

עַל גְּבוּלוֹת אַדְמַת שֵׁבֶט זָר

נְאוֹת שָׁלוֹם פָּרְחוּ, כְּפָר וּכְפָר.

תִּפְאַרְתָּם – אַחֲוָה, בְּרִית-נֶאֱמָנִים;

אָז יִמְצָא נוֹדֵד בְּזֶה הַנּוֹף

גַּם מָלוֹן חַם גַּם סֵבֶר-פָּנִים.

צֶ’רְקֶס אָז חַי בֶּן-דְּרוֹר כָּעוֹף:

בִּיפִי מַרְאֵיהֶן בְּנוֹת-חֶמְדָּתָם

לִתְהִלָּה נוֹדְעוּ עַד קְצֵה-יָם,

וִישִׁישִׁים, עִמָּם כְּבוֹד-שֵׂיבָתָם,

עוֹד שָׁפְטוּ בֵין בַּחוּרֵי-עָם;

לִצְלִיל-שִׁירֵיהֶם לִבּוֹת חָדוּ,

אֶת בְּרַק-הַזָּהָב טֶרֶם יָדְעוּ,

גַּם יַד-הָרוּסִים, בְּרַק-הַפְּלָדָה…


ז

אַךְ לֹא לָנֶצַח בִּרְכוֹת-טוֹב:

לַכֹּל יֶשׁ-יוֹם, לַכֹּל יֶשׁ-סוֹף.

וּפַעַם – שֶׁמֶש רַד הַיָּמָה,

סְבִיב רַגְלֵי-הָר עֲרָפֶל שָׁת –

בְּלַיְלָה זֶה הַכְּפָרִים שָׁמָּה

לֹא יָדְעוּ שֶׁקֶט-לֵיל וּשְׁנָת.

עֲדָרִים צְפוּפִים הָמוּ, שָׁרְקוּ,

וּמְגֻשָּׁמֵי אוֹפַן קְרוֹנוֹת חָרְקוּ;

בִּרְעָדָה בָּלְמָה אֵם אֶת פִּי

עוֹלָלָהּ, הִרְעִישׁ כְּפָר בִּבְכִי;

וְאָבוֹת, עוֹטֵי מִכְבְּרֵיהֶם,

דֹּם עָלוּ אִישׁ וָאִישׁ עַל סוּס

וַיְּמַלְּאוּ אַבְקַת-אֵשׁ רוֹבֵיהֶם,

וּלְתוֹךְ יְקוֹד חִישׁ זָרְקוּ אֵת

לֹא יָכְלוּ קַחַת עוֹד וָשֵׂאת.

וַיְהִי בְּהָקִיץ עֵין הַשַּׁחַר

עַל תֵּל-הָעֵד רְחוּץ הַטַּל,

וְרוּחַ נֵעוֹר צַח וְקַל

וַיָּפֶץ אֵד עֲנַן-הַצַּחַר –

וַיְגַל הַיּוֹם תַּחְתִּיּוֹת-הָר,

עֵין שָׂדֶה רֵיק וּנְאוֹת-שְׁאִיָּה,

עֲשַׁן יְקוֹד הַדּוֹעֵךְ כְּבָר

וְעִקְּבוֹת אוֹפַן – וְדוּמִיָּה…


ח

אַךְ מָה הִמְרִיצָם כֹּה עֲזֹב

עָפַר-אֲבוֹתָם לְבִלְתִּי שׁוֹב,

וּלְמִדְבָּר זָר מַה-פִּתְאֹם חָשׁוּ

נְדֹד וּגְלוֹת מֵרָצוֹן טוֹב?

הַחֲרוֹן מֻחֲמַד אִם חָזוּת קָשָׁה?

לֹא-זֹאת! כִּי שָׁמְעוּ שְׁמוּעַת שֹׁד,

כִּי גוֹי אֶל אַרְצָם שָׂם פְּעָמָיו,

גּוֹי אַבִּיר-לֵב, לֹא יָחֹן קָמָיו,

גּוֹי, הַכֹּל חָת מִפְּנֵי רְעָמָיו,

גּוֹי, חֵילוֹ רָב מְאֹד-מְאֹד…

צֶ’רְקֶס, מֵאָז אַבִּיר-מִלְחָמוֹת,

כְּבָר לָמַד מוֹת-גִּבּוֹרִים לָמוּת;

וְגַם לְאִשְׁתּוֹ, בַּת-קַוְקַז,

חֲנִית-פִּיפִיּוֹת וְרוּחַ עַז;

וַחֲתַת-הַשֹּׁד וַחֲמַת-מְצִיקִים

לֹא יָכְלוּ אוֹתָם נְתֹשׁ מִפֹּה.

וְאוּלָם אוֹי! כִּי זֶה הַגּוֹי

זֶה – הֵכִין לָמוֹ קְלוֹן-אֲזִקִּים.

חֲבוּר-מוֹלַדְתּוֹ בֶּן-קַוְקָז

וְחוֹמֵד שֶׁקֶט הוּא מֵאָז,

וְאוּלָם דְּרוֹר, הַדְּרוֹר לוֹ חָבִיב

מִכָּל מַנְעַמֵּי אַדְמַת אָבִיו.

"לְבֹשֶׁת צַר, עַל אַף הַצָּר

נַעַזְבָה-נָא עֲרוּצֵי הָר;

וְאוֹתְךָ, בֶּשְׁטוּ, נִמְרָץ, אָיֹם,

יְנַס-נָא לְהַגִּישׁ לִנְחֻשְׁתָּיִם!"

כֹּה גָמַר אֹמֶר אִישׁ וָאִישׁ

וּבֶשְׁטוּ בָּן לְרֵעָם עַתָּה,

רַעְמָתוֹ-הוֹד בְּעָנָן יַעֲטֶה

וְעַל הָרוּסִים שׁוֹכְנֵי-מַטָּה

יְכוֹנֵן מַבָּט קַר וָבִיש…


ט

בֵּין-כֹּה – וַיַּאַרְכוּ הַיָּמִים,

הַשְּׁבָטִים חָרְשׁוּ נִקְמַת-דָּמִים;

וְשָׁנִים חָמֵש נָטוּ כְּבָר –

אַךְ טֶרֶם שֻׁפַּךְ דַּם הַצָּר.

בְּמִדְבַּר-דְּמִי בֵּין סַלְעֵי-נֵכָר

צֶ’רְקֶס הַגּוֹלֶה הֶחֱלִיף אוֹן

וַיִּבֶן כְּפָר וּגִדְרוֹת-צֹאן

(כַּיּוֹם אֵין שְׁאָר לָהֶם וָזֵכֶר),

גַּם בַּחוּר-קְרָב גַּם שָׂב בָּעָם,

מַשְׂטֵמַת-מָוֶת יָקְדָה בָם.

רֻסְלַמְבֵּק כְּבָר יְחַכֶּה שָׁמָּה

עַל גְּדוֹת קֻבַּן לִנְסִיכֵי-בְרִית;

לֶזְגִּין הֵרִיחַ אֵשׁ-מִלְחָמָה

וַיְמַהֵר לִלְטֹשׁ חֵץ וַחֲנִית –

וַיִּגְמֹל חֶרֶשׁ זְמַם-אַכְזָרִים:

נְקַם-שְׁאוֹל מִשַּׁאֲנָן צָר.

כֵּן חַשְׁרַת-שֶׁלֶג בְּרֹאשׁ הֶהָרִים,

מִתְנוֹסְסָה בְּשֶׁפַע זְהָרִים,

עָתִידָה גְלֹש עַל רֹאשׁ עֲדָרִים,

בְּשַׁלְוָה רוֹעִים לְרַגְלֵי הָר.


י

בָּעֵת הַהִיא, בְּיוֹם-בָּצִיר קָר,

בֵּין הַר-בַּרְזִלִּי וְהַר שְׁפִיפוֹנִי,

בְּנָתִיב, עִקְּבוֹתָיו נִכְּרוּ מְעָט,

בְּתוֹךְ גֵּיא-תְנוּבָה – אִישׁ אַלְמוֹנִי

עַל סוּסוֹ עָבַר לוֹ בַלָּט.

מִסָּבִיב הָרִים כְּעִנְקֵי-שְׁאִיָּה

אֶל נִבְכֵי-רוּם יַעְפִּילוּ עֹז;

וְאִישׁ מֵעַל לְגַב אָחִיהוּ

יֶהֶרְסוּ זָקֹף רֹאשׁ וּרְאוֹת.

וְרָם עַל-כֹּל הַמְחֻמָּשׁ-שִׂיא

מִתְּכֵלֶת-אֵד כְּמוֹשֵׁל עָז,

עַל סְבִיבָיו יַפִּיל אֵימַת-רָז.


יא

עוֹד טֶרֶם מָחָה שַׂר-הַמְּאוֹרוֹת

מֵעֵין הַכַּר אֶת טַל-הָאוֹרוֹת;

אֶל תּוֹךְ הַגַּיְא מִצַּלְעוֹת הָר

נִטָּיוּ סִבְכֵי פִרְאֵי-עֵנָב

וַיַּזּוּ אֶגְלֵי כֶסֶף קָר

עַל גַּב הַסּוּס וְרֹאשׁ אֲדוֹנָיו.

וּפִתְאֹם עָמַד זֶה, וּכְמוֹ

בְּרַק-רַעְיוֹן פִּתְאֹם נִצְנֵץ בּוֹ,

וְנָבוֹךְ יִתְעֶה מַבַּט-עָיִן,

כִּמְבַקֵּשׁ פֹּה דְבַר-מָה – וָאָיִן.

יֵשׁ פִּתְאֹם יַרְתַּח סוּס וְטָס,

וְיֵשׁ אֶת שִׁטְפּוֹ יַעֲצֹר, אָז

יִזְדַּקֵּף חִישׁ, וְהוּא זָע, קוֹדֵחַ –

שְׁאִיָּה. הוּא מִסּוּס צוֹנֵחַ

וּלְאַדְמַת טַל יַט רֹאשׁ, יָשֹּחַ –

אַךְ רַחַשׁ-דֶּשֶׁא יַאֲזִין, הָס!

כְּמוֹ פַס כָּל חַי לָעַד בַּשְּמָמָה…

עִצָּבוֹן יָעִיק לֵב בַּלָּט.

אֲהָהּ! קוֹל דַּהֲרַת-סוּס לוּ שָׁמָע,

לוּ מָצָא שְׁאֵרִית נָוֶה קָט,

בְּשִׂמְחָה נָתַן כָּל עוֹלָמוֹ.


יב

אַךְ מִי הוּא זֶה הַמְשָׂרֵךְ שְׁבִיל?

בֶּשְׁמֶט עַל כְּתֵפָיו, פְּלַג-מְעִילוֹ,

עַל מִצְחוֹ יִצְנַח קוֹבַע-שֵׂעָר;

צוּר-חַרְבּוֹ מָךְ: לֹא-נוֹי, לֹא-צְלִיל

וּנְדַן פִּגְיוֹנוֹ לֹא מְפֹאָר;

וּמְשֻׁנָּס הוּא פַּס-אֵזוֹר דָּל

וְרוֹבוֹ זֶה עַל גַּב מֵעָל –

נַרְתִּיקוֹ יַלְבִּין מִמֶּרְחַקִּים,

מִטַּלְטֵל חִישׁ עַל רוֹכֵב-סוּס;

וְאֵיךְ לֹא תִבְחַן עֵינִי דְמוּת

הַיּוֹשֵׁב-הָר מִדְּמוּת קוֹזַקִּים?

וּבְכֵן – צֶ’רְקֶס הָאִישׁ הַלָּז.

אַךְ שְׁחַרְחֹר פָּנָיו כָּלִיל גָּז,

כִּי סַעֲרוֹת צְפוֹן-הַקֹּר וּשְׁלָגָיו

וְרֹגֶז שְׁמֵי הַכְּפוֹר הָעָז

כְּבָר מָחוּ מֶנּוּ שְׁחַרְחֹר נָגֶב,

אַךְ נִכָּר עוֹד: צֶ’רְקֶס – הַלָּז.

עֲבֻתִּים גְּבִינָיו, עֵינָיו – נְשָׁרִים,

אֲרֻכּוֹת-שְׁחוֹרוֹת שַׁעֲרוֹת-רִיס,

וּפָזִיז-קַל כָּל נִיד: כָּל זִיז.

וְלֹא גֻּלְחוּ עוֹד, כְּמִשְׁפַּט זָרִים,

שְׁחוֹר-זָקָן לוֹ וּשְׂפַם-הַשְּׁחוֹר;

מִפִּיו תַּבְרֵקְנָה שִׁנֵּי-צְחוֹר;

כְּקֶצֶף, יֵז עַל שְׂפַת הַיְאוֹר.

כֵּן רֻבֵּי תוֹרוֹת עַם הֶהָרִים

עוֹד טֶרֶם יִזְנַח זֶה בֵּין-זָרִים…

אַךְ אוֹי לוֹ, אוֹי, אִם לָמַד כְּבָר

אֶל דַּרְכֵי גוֹיֵי-הַהַשְׂכָּלָה;

אִם טָהֳרַת-לִבּוֹ כְּבָר חֻלָּלָה

וַיֵּשְׁתְּ מִכּוֹס הַסַּם הַמָּר!

אֲהָהּ! לַחֶמְדָּה זָר, הוּא בָּלָה

בְּטֶרֶם שְׂעָרוֹ הָפַךְ שָׂב –

עַל-מָה לְאֶרֶץ-הַשַּׁעֲשׁוּעִים,

אֶל נוֹף הַחֶמְדָּה וְהָאִוּוּיִים

כְּפוֹר לִבּוֹ הֵבִיא זֶה לַשָּׁוְא?


יג

עוֹד צָעִיר הוּא כִּבְנֵי-עֲלוּמִים,

אַךְ בְּרַק מַבָּטָיו מַשִּׁיב קֹר.

כֹּה עָטֹה יַעֲטֶה קְרוּם הַכְּפוֹר

אֶת אֹפֶל פְּנֵי הַיָּם הָרְדוּמִים

עַד פְּרָץ הַסְּעָרָה הָרִאשׁוֹנָה.

גַם אֵשׁ, שֶׁשָּׁקְעָה בְּעֵין אַבִּיר,

תִּסְתַּתֵּר לָהּ, כִּבְסֹבֶךְ-כְּפִיר,

בְּנִבְכֵי-לֵב; אַךְ עֹז-שִׁלְטוֹנָהּ

יַכְנִיעַ גַּם הַלֵּב-הַצּוּר:

זֹה אֵשׁ-הַתֹּפֶת אִם תִּתְנַעֵר,

בְּחֶשְׁכַת לֵב תְהִי לְבָעֵר

וּתְלַהֲטוֹ חִישׁ לְגַחֲלֵי-אוֹר.


יד

וְכָל שֶׁעָבַר צָף לְמוּלוֹ

כְּצֵל הַמֵּת מִשַּׁחַת בָּא;

וכְמוֹ מֵאָז נִשְׁתַּנָּה כֻלּוֹ

לֹא נוֹדַע, אֵיךְ זֶה וּבְשֶׁל-מָה.

אָז יֵצֵא-לוֹ לִשְׂדֵה-הַשְּׁאִיָּה

וּפִתְאֹם – מַה-זֶּה? – קוֹל-יְרִיָּה

וְאַחֲרָיו אַט מֵהַר אֶל הַר

כְּגִלְגוּל-רַעַם בְּלַעַג זָר

מִתְגַּלְגֵל הֵד מְשֻׁלָּשׁ, חוֹלֵף.

צֶ’רְקֶס חִישׁ-קַל פִּגְיוֹנוֹ שׁוֹלֵף;

וּפִתְאֹם יַרְא בַּסְּבָךְ קוֹזַק

וְחָמוּשׁ הוּא קְנֵה-רוֹבֶה רַק;

לְפָנָיו פַּסְיוֹן בְּדָם מִתְבּוֹסֵס,

זְנַב צִבְעֵי-קֶשֶׁת לוֹ מִתְנוֹסֵס,

וּמְפַרְפֵּר עוֹד בְּרֶטֶט קַל,

הֲלוּם-הַבְּדִיל וּרְחוּץ-הַטָּל…

וְרוֹכֵב-סוּס חִישׁ אֵלָיו קָרֵב.

"קוֹזַק, אֱמֹר – הוּא פוֹצֶה פֶה –

אֱמֹר: הַאִם מִכְּבָר פֹּה חָרֵב

וְשׁוֹאֶה דֹם הַנּוֹף הַזֶּה?"

– "מֵאָז הָרוּסִים הִלְּכוּ פָחַד

עַל עַמְּךָ, עָשׂוּי לִבְלִי חָת,

אֶל הַרְרֵי-נֵכָר חֻבְּאוּ יָחַד

וְעִדָּן זֶה חֲמִשִּׁי גָז".


טו

אֵךְ מַה-לְּךָ, צֶ’רְקֶס בֶּן-הָרִים?

מַדּוּעַ יָד הֵנַפְתָּ עָל

וְאֶת פִּגְיוֹנְךָ עַל-מָה תָרִים?

מְחַל לִלְצוֹן קוֹזַק, מְחָל!

אֲהָהּ! כִּי בָא, כִּי בָא הַשִּׁלֵּם:

כְּבָר מֻטָּל שָׁדוּד קַר וְאִלֵּם

וּמְגֹאַל-דָּם הַלֵּץ הַזֵּד.

צֶ’רְקֶס פְּנֵי-צַר הַקְּפוּאִים יֵרֶא

וּבְקִרְבּוֹ רֻתְּחוּ דְמֵי הַמֶּרִי.

הָהּ, לֹא! עֲצֹר לֹא יָכֹל עוֹד,

לֹא יָכֹל שְׂאֵת הַבּוּז הַנּוֹרָא!

וּכְהַאְדִּים שַׁחַק בְּלֵיל-בְּעֵרָה,

וּכְהָצִיף זִרְמַת לַבַּת-הָר

מְלֹא-רֹחַב גַּיְא – כֵּן אֹדֶם מַר

עַל פָּנָיו הוּצַק, וּבְמַבָּטָיו

חִישׁ הֻצַּת בְּרַק פִּי-חֶרֶב-קָטֶב

וּבְתוֹךְ הַבָּרָק חָזוּת כֹּל:

תְּהוֹם הַנֶּפֶש וְהַשְּאוֹל!

צֶ’רְקֶס לַצָּפוֹן שָׁלַח עַיִן

וְצֵל שְׂחוֹק בִּישׁ עַל פָּנָיו רַד;

וּמָה רוֹשׁ-פֶּתֶן וְאֶרֶס-עַד

לְעֻמַּת זֶה הַשְּׂחוֹק? – כְּאָיִן!

וּכְגַל יַךְ לִבּוֹ רוֹם וָרוֹם

וְרוּחוֹ צַר מִסְּעֹר וּפְעֹם

וְאֶגְלֵי יֶזַע קַר מַזִּילָה

פַּדַּחְתּוֹ, אַךְ בְּפִיו אֵין מִלָּה.


טז

וּפִתְאֹם נִנְעַר, רָעַד, שַׁח

וַיֵּט וַיַּרְעֵשׁ סוּסוֹ: צְלַח!

וּבְרֶגַע קָט כְּבָרָק אָפֵל

בִּמְרוֹם הַגַּבְנוּן צָף וָגָז –

וַיּאֹבַד חִישׁ בְּעַב-עֲרָפֶל.

עַל כֵּפֵי-מָגוֹר סוּסוֹ טָס

וּכְמוֹ עַל רוּחַ יִשָּׂאֶנּוּ:

כֹּה דָהֹר יִדְהַר אַךְ הַנָּס

וּמוּסַר-כְּלָיוֹת יִרְדְּפֶנּוּ…


יז

לְאָן צֶ’רְקֶס פְּעָמָיו שָׂם?

אֵי יִשְׁקֹט בּוֹ שְׁאוֹן-הַדָּם

וְתָקוּם סַעֲרַת-לֵב לִדְמָמָה?

הֲיִשְׁקַע צְמָא הַלֵּב הֶחָם,

אִם תּוּכַל שְקֹט זֹה אֵשׁ-הַנְּקָמָה?

הַכְּפָר, בּוֹ רָאָה יַלְדוּת, דְּרוֹר,

וּמִסְגַּד-אֵל וּנְאוֹת שְׁלוֹם-אוֹר –

כָּל-זֶה, אֲהָהּ, הַצַּר בִּלֵּעַ!

לֹא! הוּא לֹא יֵדַע עוֹד מַרְגּוֹעַ

עַד אִם מִגַּל עַצְמוֹתָם-צְחוֹר

הוּא אַרְמוֹן-קְבָרִים לָמוֹ יַצֵּב,

לִהְיוֹתוֹ אוֹת לְדוֹר וָדוֹר

וְנִקְמַת-עַד עַל בּוּז מְנַאֲצָיו,

וְיָגֹל חֶרְפַּת נוֹף-הַדְּרוֹר!

"כְּדֹמֶן-חוּץ לָעַד אֲשִׂימְכֶם

וְהָיָה שְׁמִי לָכֶם לְחֵת;

זֶה-כְּבָר, זֶה-כְּבָר אֲתָעֵב שִׁמְכֶם,

וְאַךְ לְנַפְשְׁכֶם וְדִמְכֶם

צָמֵאתִי פֹה בְּזֹה הָעֵת!"

וְשָׁט הַסּוּס הָאַבִּיר רְתֵת

וְגַעַשׁ נְחִירָיו כְּאֵד הַסִּירִים

וּמְזָרוֹת עֵינָיו בְּרַק דִּי-נוּר;

מֵרָחוֹק הַרְרֵי-עַד כְּבָר נִרְאִים,

מָעֻזְנֵי-שֵׁן, פְּתַלְתֻּלֵי-צוּר,

וּמְעֻלַּף-צְחוֹר וּזְקוּף-רֹאשַׁיִם

יִתְגָּאֶה שַׁט עַד רוֹּם שָׁמַיִם,

וְרוֹכֵב-סוּס בְּלִבּוֹ נָם:

“בְּעוֹד אַךְ שָׁעָה אָבֹא שָׁם”.


יח

לְפָנָיו עוֹד בִּקְרוּם-עֲרָפֶל,

כְּקִיר שֶׁל אַוִּיר כְּחַלְחַל-אָפֵל,

נִמְשָׁכִים צוּקֵי דַבְּשׁוֹת-רוֹם:

פַּלְמוֹנִים-זָרִים כְּמוֹ חֲלוֹם,

שׁוּב יִתְפָּרְדוּ, שׁוּב יִדְבָּקוּ…

וְשָׁעוֹת שְׁתַּיִם עָבְרוּ כְּבָר!

וּכְאִלּוּ עַל הָאִישׁ יִצְחָקוּ,

עֲדֶן מַרְאֵיהֶם פֶּלִאי-זָר.

פְּנֵי הָרוֹכֵב יִתְרָעָמוּ

וְכוֹשֵׁל כְּבָר הַסּוּס מִזְּרֹם,

וּכְבָר מַעֲרָבָה יֵפֶן יוֹם

וְצִנַּת לֵיל בָּאַוִּיר קָמָה, –

וְאֵלֶּה עִנְקֵי הָרֵי-שְׁמָמָה,

אַף-כִּי הִגְבִּירוּ שִׂיא וָצֵּל,

עֲדֶן מַרְאֵיהֶם – חִידַת-אֵל.


יט

אַךְ הִנֵּה כְּבָר כְּעָנָן כַּבִּיר

בֹּו יִפְגַּע פִּתְאֹם גּוּשׁ הַר-הֹר;

כְּתַשְבֵּץ-כְּבִיר יִתְגַּוֵּן סְחוֹר,

וְשִׂיא מִשִּׂיא עֲלוֹת יִתְגַּבֵּר.

מְיֻזָּע פִּגֵּר סוּס מִשְּׁטָף

וַיַעֲמֵק פֹּה נְשִׁימַת-אָף.

וְצֵל הַיַּלְדוּת, תֹּר-שַׁעֲשׁוּעִים,

לִפְנֵי צֶ’רְקֶס פֹּה צָף מִתְּהוֹם;

בַּלֵּב הֵקִיצוּ הַגַּעְגּוּעִים

וּבְעֵינָיו צָצוּ אֶגְלֵי-תֹם.

וַתִּשְׁקַע אֵיבַת לֵב-הַמֶּרִי,

וּמְצוּקוֹת מַחְשְׁבוֹת הַקֶּרִי

כְּמוֹ קַלּוּ פֹה. אֶת מִצְחוֹ-אוֹר

צֶ’רְקֶס אָז יָרִים, יַשְׁקִיף סְחוֹר –

וְרוּחַ-גֵּאוּת לְבָשַׁתְהוּ,

כִּי הוּא צֶ’רְקֶס, פֹּה אֵם חִבְּלַתְהוּ:

יְחִידִי בֵּין מְצוּקֵי-הָר,

כְּמוֹ עָמְדָה פִתְאֹם זִרְמַת-חַיָּיו,

בַּהֲגִיגֵי-עוֹלָם הוּא הַשָּׂר

וְּסְפֹג אַלְמוּתָם – כָּל מָאֲוַיָּיו.

וַיִשְׁכַּח פֹּה אֵת עָלָיו בָּא,

וְרוּחוֹ פִתְאֹם כְּמוֹ הִתְעַלָּה;

וּכְחַבֵּק אִישׁ אֲחוֹתוֹ-כַּלָּה

כֵּן חִבְּקָה נַפְשׁוֹ עוֹלָם-יָהּ!


כ

כְּבָר אָדַם קָדְקֹד הַעֲנָקִים

מֵחַכְלִיל שְׂחוֹק בִּפְאַת דִּי-נוּר,

וְחֹשֶׁךְ רַד בִּנְקִיקֵי-צוּר.

מִמְּצָרֵי-גַיְא, מִנְּתִיב-עֲמָקִים

עֲרָפֶל נָם בִּדְמָמָה צָף,

וְאַךְ קוֹל צַעֲדֵי-סוּס חֲזָקִים

בַּמֶּרְחָק יִמּוֹג אַט וָסָף.

וַיִּדְעַךְ אִלֵּם יוֹם-הַבָּצִיר;

וְנוֹבֵל חֶצְיוֹ קִפֵּל דֹּם

עָלֵהוּ-מֹר כָּל צִיץ וְחָצִיר

וַיִּישַׁן שְׁנָת בְּאֵין חֲלוֹם.

אָז חֶרֶשׁ אַט מִנְּקִיק-חַלָּמִישׁ

כַּחֲרוּז טַבָּעוֹת צֶפַע גָּח,

יִתְמֵשֵּׁךְ עָצֵל, קַר וְגָמִישׁ,

אַף יֵשׁ יִתְקַמֵּט חָח עַל חָח.

וּמַכְסִיף גַּבּוֹ – עוֹר הַפַּסִּים –

כְּהַכְסֵף דֹּם מִמַּעֲבֶה-סְבָךְ

הַחֲנִית אוֹ שִׁרְיוֹן הַקַּשְׂקַשִּׂים

(הַזְּנוּחִים מִנִּי גִבּוֹר-חַיִל

בִּשְׂדֵה-הַקְּרָב) אֶל מוּל יָרֵחַ,

שֶׁמְּצָאָם שָׁם בְּקַדְרוּת-לַיִל…


כא

כְּבָר אָתָא לַיְלָה, לֵיל אֲפֵלָה;

הַנּוֹדֵד עָטַף גְּלִימָה גְדוֹלָה.

מֵאַחֲרֵי אַלּוֹן עָבֹת, גּוּץ,

הִתְחַבֵּא שְׁבִיל; הַרוּחַ סוֹעָה;

הַסּוּס הִתְנַגֵּף תּוֹךְ-כְּדֵי-רוּץ

וַיִּנְחַר כְּחָשׁ לְפֶתַע שׁוֹאָה,

וַיַּעֲמֹד… חִישׁ הָרוֹכֵב רַד –

וּלְרַגְלָיו שְׁאוֹל פְּעוּרָה כֻלָּהּ

וְשָׁם בַּתְּהוֹם, בְּמַעֲמַק-עַד,

יְשׁוֹקֵק פֶּלֶג רַב-הֲמֻלָּה

(זֶה דְכִי הֲמֻלָּה לִי לֹא רָז:

בִּישִׁימוֹן-לֵיִל הִקְשַׁבְתִּי זַעְמָהּ

וְרֹב, הָהּ, רֹב מַחֲשָׁבוֹת אָז

הֵקִיצוּ בְנַפְשִׁי זוֹ הַנְּשַׁמָּה!).

הָרוֹכֵב תּוֹהֶה עַל הַכֵּף

בְּאֵין-אֱיָל וּבְאֵין מְגַמָּה;

אַךְ פִּתְאֹם – שָׁם בְּלֵיל-אֵין-סוֹף

מְרַטֵּט אוֹר. וְשׁוּב עַל סוּסוֹ

הַנֶּאֱמָן לוֹ, אַבִּירוֹ, מְנוּסוֹ,

הוּא עוֹלֶה חִישׁ, וְטוּס וּדְהֹר

לְשָׁם, לְשָׁם, אֶל שְבִיב-הָאוֹר.


כב

לֹא שֵׁד-מְזִמּוֹת בָּעֲלָטָה

בְּרִשְׁתּוֹ מָשַׁךְ אֶת הַזָּר;

לֹא עֵין בְּכוֹר-שָׂטָן שָׁם לָהָטָה

וַתְּאַזֵּר שְׁבִיב בַּנְּקִיק הַצָּר –

שְׁנֵי אָהֳלֵי-עָנְוָה צְחוֹרֵי-קִירוֹת

שַׁאֲנַנּוּ דֹם לְרַגְלֵי הָר;

עֲלֵיהֶם כִּפּוֹת-גַּג מַשְׁחִירוֹת,

וּכְטוּר צִיצִיּוֹת יִצְנַח רָב

מִשּׁוּלֵי-גַג סָפִיחַ רָב,

וְרוּחַ טַחַב, רוַּח-סְתָו

מְלַחֵשׁ לָמוֹ נוּגוֹת זְמִירוֹת.

וּמַקִּיף אֶת הַחֲצֵרִים סְיָג

חֲרוּלֵי-עֵץ, וְכֻלּוֹ מָק

וַיִּשַּׁח כְּבָר מֵעִדָּן כַּבִּיר.

וּתְנוּמַת-עַד וּדְמָמָה סְחוֹר,

אַךְ צֹהַר קָט יַגִּיהַּ אוֹר…

וַיִּצְהַל עֹז הַסּוּס הָאַבִּיר

וּבְרַגְלוֹ יַךְ עֲפַר-הַחוֹל

וּלְקוֹלוֹ עָנָה סוּס בְּקוֹל..

אָז יֵצֵא מִי מִפֶּתַח נָווֹ

וַיִּשְׁאַל אֶת הַבָּא: "אֶת-מִי

נְבִיא-אַלַהּ לְעֵת-לֵיִל יָבֵא?"

וַיַּעַן: “נוֹדֵד זָר אֲנִי!”

וְלֹא יָסַף עוֹד הַסָּב הֶעָנָו

חֲקֹר מֵאֵי וּמִי הָאִישׁ

וּלְנָווֹ סוּר הוּא קוֹרְאוֹ חִישׁ

(כִּי כֵן הִיא תוֹרַת אָב לְבָנָיו);

וּכְבָר הוּא עוֹשֶׂה כֹּה וָכֹה

לְהַתִּיר סוּס אַף תֵּת מִסְפּוֹא.


כג

לְנָוֶה סָפוּג עָשָׁן כֻּלּוֹ,

אָז יוֹאִיל הָאוֹרֵחַ סוּר;

הַשָּׂב יוֹשִׁיבוֹ מוּל הַנּוּר –

וְלִבּוֹ טוֹב וְשָׂשׂ לְמוּלוֹ.

עַל-קִיר יִתְנוֹסֵס כְּבוֹד עֱזוּז

בֶּן-הַר: קְנֵי-רוֹבִים, חֵץ וַחֲרָבוֹת

עִם כְּתֹבֶת-מוּסָר בַּת חֲרוּז;

וּבַרְדַּס שׁוֹדֵד-לֵיל בַּזָּוִית

וּגְלִימָה, שׁוֹט וְכַר לַסּוּס.

וְאִישׁ עִם רֵעוֹ אָז יִדָּבְרוּ

עַל דְּרוֹר, עַל קְרָב, עַל יָמִים עָבְרוּ;

וְחַי וְסוֹעֵר זֶה הַשִּׂיג,

וְטָסִים הֵם עַל-כְּנַף-הֲגִיג

אֶל קַדְמֵי-דוֹר וּלְיָמִים בָּאִים.

כֹּה טָסוֹת שָׁעוֹת וְלַיְלָה עָף, –

מוּל אוֹר הַקָּמִין יוֹשֵׁב שָׂב

וְכֻלּוֹ אוֹמֵר קֶשֶׁב רָב

וְדִבְרֵי זָר בְּעֵינָיו – פְּלָאִים.


כד

הוּא הָיָה אִישׁ לֶזְגִּין, זֶה-כְּבָר

(כֹּה גָמַר אֵל) בַּנּוֹף הַזָּר

יִתְעַגֵּם דֹּם. וְלֹו בַת נָאוָה

וּבָנִים שְׁלֹשָׁה, בְּנֵי-אֱיָל.

עִמָּהֶם – יֵשׁ וְשָׁכַח דְּאָבָה

וְנָעַם לוֹ זֶה אֹהֶל דָּל.

בְּהַבְרֵק יַחַד כּוֹכְבֵי-עֶרֶב

וְיָצְאוּ חִישׁ לְשַׁחֵר טֶרֶף

בְּנֵי-שַׁחַץ אֵל עִם נֶשֶׁק חַד –

וְהִרְבּוּ שֹׁד וְאָכְלוּ עַד.

לִפְנֵיהֶם יֵלֵךְ צֵל-מְחִתָּה,

וְאַחַת הִיא – הֲרֹג אוֹ גְנֹב;

בַּחֶרֶב יִקְנוּ חֵלֶב-חִטָּה

וּבְחֵץ יְשַלְּמוּ יַיִן טוֹב.

מֵעֵדֶר אוֹ מִכְּפָר קוֹזַקִּים

יְנַהֲגוּ סוּס בַּלָּט – וָגָז;

רַק אוֹר-הַיּוֹם הוּא אוֹיְבָם עָז;

כֹּה פְּרוּשָׂה רִשְׁתָּם עַד מֶרְחַקִּים…

הַלַּיְלָה שָׁאַר עִם הַשָּׂב

רַק בְּחִיר-בַּחוּרָיו, בְּנוֹ הָרָב…

כֹּה סִפֵּר… אַךְ הַלָּז לֹא שָׁמָע!

כִּמְעַט לֹא נוֹתְרָה בוֹ נְשָׁמָה;

אַף קָמָה פִתְאֹם עֵין הַזָּר,

כִּבְרֶגַע שְׁקֹעַ מֵטֵאוֹר:

לְפָנָיו נִצְּבָה דְמוּת בַּת-הָר

עֲדִינַת-חֵן וּצְעִירַת-שָׁנִים,

יְצִיר אֲדָמָה וַעֲדָנִים…


כה

וּבְעֵינֵי מִי כָּזֹאת לֹא תִשָּׂא חֵן?

מִי חֵן-שָׁמַיִם, וְלוּ רַק מֵעֵין

שָׁמַיִם עַלְזֵי-אוֹר, הוּא חֵן-הַטֹּהַר

הַגָּנוּז בְּסֵבֶר-שְׂחוֹק וּבְנִיד עֵין-אוֹר, –

כָּל מַה-שֶּׁחָבִיב כֹּה בַחֲלוֹם הַנֹּעַר

וְאַךְ פַּעַם נִזְכֶּה לוֹ בְחַיֵּי-שְׁחוֹר –

מִי הַבְחֵן זֶה לֹא יַבְחִין תּוֹךְ-כְּדֵי-רְאִיָּה

מִיפִי הֲבֵל-הֲבָלִים, חֵן-רְמִיָּה?

וּמִי לֵב יַקְשֶׁה, יָהִין כֹּה אֱמֹר:

“הֵן אָמְנָם קְלַסְתֵּר-חֵן, אַךְ נֶפֶשׁ-קֹר!”

בִּרְאוֹתוֹ פִתְאֹם סֵמֶל, שֶמִּתְּחִלָּה

חֲשָׁבוֹ לְנִשְׁמַת כְּרוּבֵי דְרוֹר וְגִילָה,

שֶׁיָּרְדוּ מִמְּרוֹמֵיהֶם עַד הֲלוֹם

לְנַחֵם לֵב וְלִקְסֹם לוֹ חֲלוֹם?

אֶל בַּת לֶזְגִּין לוּ יִגַּשׁ, יִתְוַדֵּעַ:

בְּלֶחְיָהּ דַּם-הַקֶּדֶם יִשְׁתַּעֲשֵׁעַ

וְחַיֵּי אֶרֶץ יְרַנְּנוּ מִגֵּו;

וְאַךְ תִּרְחַק מֶנּוּ זֹה הַדְּמוּת הַנָּאָה

וְתָמַהּ עַל שֶׁרָאוּ עֵינָיו וְתָהָה,

וְאָמַר: קֶסֶם הוּא! וְלֹא יַאֲמִין לֵב…


כו

עֲדִינַת-חֵן וּצְעִירַת-שָׁנִים,

יְצִיר אֲדָמָה וַעֲדָנִים;

מְאִירַת-עַיִן, קְרוֹבַת-לֵב,

כִּצְלִיל שְׂפַת-אֵם, שֶׁצָּף מִנְּשִׁיָּה

בְּאֶרֶץ-נוֹד, בְּעִיר נָכְרִיָּה;

עֲנֻגָּה, מְלֵאָה תַנְחוּם-אוֹר

כִּנַחֵם שִׁירַת צִפּוֹר-דְּרוֹר

מִבַּעַד לְשִׂבְכַת אֶשְׁנָב אָפֵל

אֶת לֵב אֲסִיר-הַבּוֹר בַּשָּׁפֶל –

כֹּה נִצְּבָה זַהְרָה מוּל הָאָח

בֵּין אוֹר וָצֵל, וּמִצְחָהּ שָׁח,

יְהִירָה וְקַלַּת-גֵּו. בִּלְבוּשָׁהּ

אֵין הוֹד וּפְאֵר, אַךְ תֹּם בִּלְבָד

וְטַעְמוֹ בוֹ… לָרֹאשׁ חֲבוּשָׁה

מִטְפַּחַת-בַּד בַּחֲרִיצוּת-יָד;

מִתַּחְתָּהּ עַד לִשְׁדֵי-חֶמְדָּתָהּ

שְׁתֵּי צַמּוֹת-אֹפֶל פִּזְּזוּ-נָסוּ…

זֶה שָׁעָה תְטַפֵּל יָד בָּהֵן:

קוֹלַעְתָּן, פּוֹרְעָה, וְשׁוּב קוֹלַעַת;

זֶה דֶרֶךְ אִשָּׁה – כָּכָה תַעַשׂ

בִּגְבֹר תַּאֲוָתָהּ לִמְצֹא חֵן…


כז

בְּיַד רוֹעֵדָה הִיא מַגֶּשֶת

בִּצְנִיעוּת-חֵן כְּמִתְבּוֹשֶׁשֶׁת

לִפְנֵי אָבִיהָ נְזִיד דַּל,

וַתָּפֶק סֵבֶר חִיּוּךְ קַל

וַתֹּאמֶר צֵאת, אַךְ טֶרֶם יָדְעָה:

הֲצֵאת אִם לֹא? זְעֵיר-זְעֵיר

עַל פְּנֵי הַלֵּב הַסּוּת רָעָדָה.

לִשְׁמֹעַ חָמֹד מֶה חָמָדָה

מַה-יֹאמַר זֶה הָאִישׁ הַגֵּר,

אַךְ מַחֲשֶׁה הוּא, מַבָּטָיו רָצִים,

נִמְשָׁכִים אַחֲרֵי בְרַק-אֲזֵן

שֶׁעַל הַקִּיר, חֲנִית וַחֲצָצִים;

אַךְ הִנֵּה פִתְאֹם נָפְלָה עֵין

הַזָּר אֶל עֵבֶר פְּנֵי הַנַּעֲרָה

וַתִּבּוֹךְ קְצָת, אַךְ הִיא נִשְׁאֲרָה

תַּחְתֶּיהָ, בְּלִי יְרֹא מִגְּעָר-

בָּהּ אָב, וַתּוֹסֶף קְשֹׁב לַזָּר.

וְנִרְאֶה, כְּמוֹ דְבַר-מָה יַדְרִיכוֹ

לִרְגָעִים מְנוּחָה עַד מְאֹד:

עַל מִפְתַּן פִּיהוּ נֶאֱלַם שִׂיחוֹ;

עוֹד יִשְׂחַק שָׂחֹק אַךְ לֹא עוֹד

תָּצֹדְנָה עֵינָיו אוֹתָהּ צוֹד.

עוֹד יִשְׂחַק, יָלִיץ אַךְ עִצָּבוֹן

בַּצְּחוֹק, וּכְמוֹ מֵאֹנֶס בָּא;

וּבְהַחֲרִישׁ זֶה – וּמָלְאָה דְאָבוֹן

וְיָרְעָה נַפְשָׁהּ לָהּ עַל-מָה…

צֶ’רְקֶס מֵרֹאש הֵן צָד בִּרְעָבוֹן

כָּל זִיז, כָּל בְּרַק עֵינֵיהָ-לַהַב;

וְעַתָּה קָמוּ עֵינֵי-מְשׁוּבָה,

כְּמוֹ מְיַחֲלוֹת הֵן לִתְשׁוּבָה,

כְּיַחֵל עֵינֵי בַת עֲלוּבָה –

וְהוּא לֹא יָשִׂים לִבּוֹ – רַב!

כִּי גְדוֹלָה כְלִמַּת-לֵב וּמָרָה,

לֹא תוּכַל נַעֲרָה אוֹתָהּ נְשׂא:

כִּמְּדֻמֶּה, לְמִפְגָּע לוֹ הִיא זַהְרָה –

הִיא תֵשְׂטְ מִפֹּה, הִיא תֵשְׁטְ מִפֹּה!


כח

מִי אִישׁ, אֲרָצוֹת הִרְבָּה לָתוּר,

מִי אִישׁ, שֶׁהִסְכִּין רַבּוֹת רְאוֹת;

מִי אִישׁ, לוֹ רָז-הַתַּאֲווֹת פָּתוּר

וַיִּלְמַד שְׂפַת כָּל לֵב, כָּל אוֹת;

מִי אִישׁ, שֶׁהֻטַּל עוֹד בְּאִבּוֹ

אֶל-בֵּין מַשְׂכִּילֵי עָם וָגוֹי

וּכְמוֹהֶם תֵּת לֹא נָתַן לִבּוֹ

בְּתִתּוֹ יָד בְּיַד בַּת-נוֹי –

הוּא שׂוֹשׂ לֹא יָשִׂישׂ עוֹד בַּחַיִּים

לִתְשׁוּקַת אֵשֶׁת-מַאֲוַיִּים,

וּבְיֹשֶׁר פֶּרֶא-לֵב לֹא עוֹד

הֲתוּלִים יַחְמֹד לוֹ חֲמֹד.

הוּא יַעֲנֶה בִצְחוֹק קַל עַל דִּמְעָה,

עַל חִיּוּךְ-חֵן – בְּמַבָּט קָר

וְחִבַּת-מָה בְּעֵינָיו – כְּלִמָּה…

יָגֵעַ מֵהִתְכַּחֵש לוֹ –

גַּם לִבּוֹ הוּא לֹא יַאֲמִין בּוֹ.

כִּי נֶפֶשׁ רָמָה אֵין לָהּ נַחַת

בִּשְׂרִידֵי-אֵשׁ שֶׁל נֹעַר-הוֹד,

וּמְצִיקָה לָהּ מַחֲשָׁבָה אַחַת,

כִּי אֵין בָּהֶם כְּדֵי-יְקוֹד…

אֲנָשִׁים אֵל בְּחֶבְרַת אָדָם –

מְקוֹר מַמְרוֹרֵי-עַד וּכְאֵב;

כִּילָדִים יַעֲשׂוּ כָל מַעְבָּדָם

בְּפִיק הַיָּד וּבְמֹרֶךְ-לֵב:

אַף גְּנֹב לֵב אִשָּׁה לֹא יָהִינוּ,

אַף שְׂטוֹת מִמֶּנָּה לֹא יַסְכִּינוּ;

וְיֵש – אֶת לִבָּם יַשְׁגֶה רָז:

כִּי נוֹף-מֶרְחָק – צֳרִי לַנְּשָׁמָה,

אוֹ מִדְבַּר חוֹל, אוֹ גִבְעָה רָמָה,

אוֹ נָוֶה מֻצְנָע בְּבִקְעַת-דְּמָמָה,

בִּמְקוֹם חֲלוֹם-הַנֹּעַר גָּז;

אַךְ אֵין בְּתִקְוָה זֹה נְכוֹנָה:

בִּמְקוֹם הַנֶּפֶש שָׁם שִׁלְטוֹנָהּ…


כט

כְּבָר יָצֹא יָצְאָה בַת-הַחֵן,

צֶ’רְקֶס אַחֲרֶיהָ מֵעִיף עֵין-

דְּאָבָה, הוֹגֶה: "בַּת-הַפְּלָאוֹת!

הֵן עוּלַת-יָמִים הִיא עֲדֶן

וּכְבָר מִשְׁאָלוֹת לָהּ וּדְמָעוֹת.

אַל, עַפְעַף-שַׁחַר, תִּזְרַע אוֹר

לַשָּׁוְא עַל אֹפֶל עַנְנַת-שְׁחוֹר,

פֶּן זָעְפָּהּ עָכֹר אַךְ יַעְכָּרְךָ,

הִיא תָּעוּף לָהּ, כִי אָץ לָהּ אָרְחָהּ".


ל

"עוֹד טֶרֶם תֵּדְעִי, מִי אֲנִי –

הִנָּחֲמִי! לֹא נְוֵה-מִבְטַחִים,

אַךְ קְרָב וּדְרוֹר וְאֶרֶץ-אַחִים

מִן הוּא גוֹרָלִי מִנָּה לִי.

הֵן אָמְנָם אַהֲבַת-עַד עִמָּדִי

לְאָהֳבֵךְ, וְאוּלָם אֵל

אֶל-נָכוֹן מָגֵן לָךְ וָצֵל.

כִּי הִנֵּה דָמִים שָׁפְכָה יָדִי –

הַהִיא תְחַבְּקֵךְ, בַּת-יְפִי?

הַבְּחֵיקִי תִרְוִי חֶמְדַּת-אַהֲבָה,

וּלְפִיךְ הַאֻמְנָם יִשַּׁק פִּי

הַמָלֵא תָמִיד קִלְלוֹת-זַעֲוָה?"

– – – – – – – – –


לא

הַבֹּקֶר אוֹר. כְּבָר הִשְׁכִּים עוּר

לַצֵאת לְדַרְכּוֹ הָאוֹרֵחַ;

עַל-גַּב הָאָח עוֹד כָּחַל נוּר,

הַשָּׂב הֶעָגוּם בּוֹ יָפֵחַ,

יְבַשֵּׁל דֹּחַן לוֹ, אָז יוֹר

לָאִיש, אֵי יָבֹר אֶת הַדֶּרֶךְ

אַף הוּא עֲדֵי הַסַּף לוֹ יֶאֱרַח,

יְלַוֶּה גֵר בְּפָנִים-אוֹר.

בְּלֵב צֶ’רְקֶס דְּבַר-מָה יִתְרַגֵּשׁ

וְיָעִיק דֹּם בִּנְהִי-הַכְּאֵב:

אֶל נַעֲרַת-חֵן יִנָּהֶה לֵב…

אַךְ מִי הֲלוֹם אֶת סוּסוֹ יַגֵּשׁ?


לב

עֲצוּבַת-רוּחַ בַּת-הַחֵן

אֶת סוּס-הַקְּרָב לָאִישׁ הִקְרִיבָה;

בִּתְחִנָּה אֵלָיו נָשְׂאָה עֵין

וּבְקוֹל כָּאוֹב הִשְׁמִיעָה נִיבָהּ:

"כְּבָר מוּכָן הַכֹּל – קוּם וּצְלַח!

בְּרֶסֶן קְרָב פֵּאַרְתִּי סוּסְךָ

וּכְמוֹ קַשְׂקֶשֶׂת כֶּסֶף צַח

יִתְנוֹסֵס בְּרַק חֲנִית-עֱזוּזְךָ;

וּלְאֻכָּף פֹּה בִרְצוּעַת-עוֹר

קָשַׁרְתִּי קָשֹׁר גְּלִימַת-שְׁחוֹר;

רְאֵה, כָּל-זֶה לֹא חָדָשׁ עִמִּי,

כְּבָר מוּכָן הַכֹּל, אוֹרְחִי-נְעִימִי!

וְאוֹן לַסּוּס וְחֵן וָנוֹי

וּפְתַלְתֹּל-הַר אַךְ מִשְׂחָק לוֹ.

וְאִם גַּם גָּדַל בַּמֶּרְחַקִּים,

אַךְ פֶּרֶא-גָאוֹן לוֹ וָעֹז,

וְנוֹצֵץ הוּא כְּאֶבֶן זֹאת,

שֶׁשָּׁחֲקוּ זִרְמֵי מַיִם מְפַכִּים,

וּבְרַק מַבָּטָיו – אֵשׁ יְקוֹד,

כִּמְעַט אַךְ שְׁחֵה – וְטָשׁ כַּשֵּׁד.

רְאֵה, אֶת סוּסְךָ רַק בִּשְּׁבִילְךָ

סִלְסַלְתִּי כֹה, אוֹרֵחַ זָר,

מֵחֶרֶב אוֹיֵב כִּי יַצִּילְךָ

בִּישִׁימוֹן-דְּמִי, בְּעִדָּן מָר".


לג

"וְאוּלָם כַּתֵּר-נָא רְגָעִים

מִשִּׂים כַּף-רַגְלְךָ בַּכָּר;

לֹא תִשְׁמַע לִי, אוֹרֵחַ זָר!

מִי יַגִּיד? אוּלַי יָמִים בָּאִים

וְסֵמֶל הַנִּצֶּבֶת פֹּה

יִזָּכֵר לְךָ בְּנוֹף-הַנּוֹי?

אָז אִם יְשֻׁפַּך יֵין-שְׂשׂוֹנְךָ

בִּפְנִינֵי קֶצֶף-זִיו עַל-פְּנֵי

הַכּוֹס – אַל בִּי יְהִי הֶגְיוֹנְךָ,

וְזֵכֶר שְׁמִי מִלִּבְּךָ זְרֵה!

אַךְ אִם זִכְרוֹנִי יְבוֹאֶךָ,

אִם צִלִּי יַחֲלֹף עַל פָּנֶיךָ

בְּלֵילֵי-דְוָי וְעֶצֶב-דְּמִי –

זֶה קוֹל תַּחֲנוּנַי, לוּ שְׁמָעֵנִי:

אַל-נָא בְלָצוֹן קַר תִּדְחֵנִי,

וְרֶגַע קַל הֱגֵה-נָא בִי!"


לד

"נָוֵנוּ דַל, אַךְ קִרְבּוֹ שַׁלְוָה,

לֹא יָנִיף עָלָיו יָד בֶּן עַלְוָה.

מַה-יִמְצָא פֹּה? אַךְ כַּר וָשׁוֹט

וְצֶמֶד-סוּס וּמִכְבַּר-גְּלִימָה.

תֵּן אֵמוּן בִּי וְגַל לִי סוֹד:

אָן פַּעֲמֵי-עֹז כָּעֵת תָּרִימָה

וְתִקְוַת-שָׁוְא תוֹלִיכְךָ אָן?

מַה-תְּבַקֵּש-לָךְ? – הִשָּׁאֵר כָּאן!

אִתָּנוּ שְׁבָה בִּנְוֵה-מִבְטַחִים!

הֵן כְּבָר נָשִׁיתָ אָבוֹת, אַחִים,

וּשְׂפָתָם הֵם לֹא תִשְׁמַע עוֹד

וַתְּהִי לָהֶם כַּזָּר כַּנּוֹד.

מַה-יִסְכֹּן לְךָ עֲפָר-מוֹלֵדָה

וּמַה-לְּךָ מִנּוּ שָׁם, מִי יֵדָע?

וִיהִי דְבַר-אָבִי נָא יוֹם-יוֹם,

כִּי חוֹבַת אִישׁ לְאֶרֶץ-אָבִיו

לְחָרֵף נַפְשׁוֹ לָהּ וּנְקֹם –

עַל-דַּעְתִּי: שָׁם הַנּוֹף הֶחָבִיב,

בִּמְקוֹם נֵאָהֵב אַהֲבַת-חֹם,

בִּמְקוֹם נֵאָמֵן אֹמֶן-תֹּם".


לה

"עוֹד צְחוֹר-עֲרָפֶל יַעֲטֹף שָׂדֶה

וְצִנָּה יִשְּׂאוּ הַרְרֵי-רוֹם.

הַקְשִׁיבָה לִי, לוּ אֶחָד יוֹם,

לוּ אַחַת שָׁעָה שְׁבָה עִמָּדִי!

עִמָּדִי שֵׁב, אַכְזָרִי גֵר!

אֲנִי אֶת סוּסְךָ אַלְעִיט זְעֵיר,

אֲפַתְּחוֹ דְרוֹר, גַּם אַשְׁקוֹ מָיִם,

יִנָּפֶשׁ-לוֹ בַמִשְׁפְּתָיִם –

וְאַתָּה פֹּה בְצֵל נְוֵה-

מְנוּחוֹת רֹאשׁ עַל זְרֹעִי שְׁחֵה.

רַק פַּעַם לוּ אֶת נִדְבוֹת פִּיךָ

יָכֹלְתִּי קְשֹׁב וָשׂוֹשׂ בִּדְמִי!

מִי רִגְעֵי-אֹשֶׁר יֵעָצֶר-לִי

וּשְׁעַת-פְּרִידָתְךָ, מִי יַבְרִיחָהּ?…"

אַךְ חִנָּם רָטְטָה בַת-הַנּוֹי

וַתּוֹחֶל לוֹ, כִּי יָשִׁיב דָּבָר

וְלֹא יָכְלָה הַבְלִיג עוֹד עַל דְּוָי:

אֶת מִפְתַּן-עֵינָהּ דֶּמַע עָבַר…

אֲהָהּ! בִּדְמִי-אֲבָנִים קָר

עַל סוּסוֹ עָלָה כְּבָר הַזָּר;

כְּבָר נָכוֹן הוּא לְהָרִים פָּעַם,

אַךְ הִפְנָה רֹאשׁוֹ – וַיִּדָּהַם,

וַיֵּצֶר לוֹ עַד רַחֲשֵׁי כְאֵב,

וַיִּפְתַּח פִּיו לְנַחֲמָהּ לֵב.


לו

"אַל-נָא תִּרְאִינִי כֹּה, בַּת-פְּלָאוֹת!

מַה-לִּבֵּךְ שׁוֹאֵל? דִּמְעוֹת-חֹם?

הֵן אָמְנָם הָיָה לִי נֹאד-דְּמָעוֹת –

מִכַּעַס דָּלַל כְּבָר עַד-תֹּם.

מַה-יְּחָבְרֵךְ, כְּלִילַת-אַהַב,

כָּמוֹנִי אִישׁ מְלֵא עִנְיַן-זַעַף?

עֲרִירִי אֹבַד, אַךְ בֶּן-דְּרוֹר,

אִם שַׂר בָּעָם, אִם עֶבֶד-שְׁחוֹר…

מִי נֶפֶשׁ עָרְגָה לִי בִּדְחִילוּ

וּסְחֹף לֹא נִסְחֲפָה בִמְרִי?

מִנִּשְׁמַת-אַפִּי תִיבַש גִּילָה,

וְחֶמְלַת-מָה – לֹא בִי, לֹא בִי…

וַאֲנִי, דְּעִי לָךְ, לֹא בֶן-עָם רֵיק

(וְאִם אֲנִי בִּלְבוּש בֶּן-עָם)

הִנָּחֲמִי, זַהְרָה, יַחֲשִׂי רָם:

הַדּוֹבֵר בָּךְ – אֲחִי רֻסְלַמְבֵּק.

זֶה נִדְרִי לְהַקְרִיב אֹשֶׁר-חַיַּי, דְּעִי!

אַל יֵרַע לִבֵּך שָׁוְא… הָהּ, סִלְחִי לִי!"…


לז

כֹּה סָח וַיְנוֹפֵף יָד – וְקוֹל שַׁעֲטַת-עֹז

כְּבָר גִּמֵּא שְׁבִיל וַיֵּלֶךְ הָלֹךְ וְנָמֹג;

וְהִיא – בְּלִי בְכִי, בְּלִי נִיב וּבְלִי הֲגוֹת

הָעָמְדָה בְּתִמְהוֹן-לֵב וְקֶשֶׁב עָמֹק,

כְּמוֹ לְאַפְסֵי-עַד זֹה שַׁעֲטַת-עֹז

חֲמוּדוֹת כָּל עֲתִידָהּ נָשְׂאָה יַחַד.

לַמִּצְעָר, זַהְרָה, לָךְ הֲגִיג טִפַּחַתְּ:

עוֹד טֶרֶם הֱקִיצוֹת – וְאֵין הֲגִיג!

וְעַתָּה עֵינֵךְ – דִּמְעָה, נַפְשֵׁךְ – רִיק,

וְרִגְעֵי-דְוָי קִדְּמוּךְ וּקְשִׁי-צִפִּיָּה

וּזְכֹר רִאשׁוֹנוֹת, עָבְרוּ וְלֹא-יִהְיוּ…

מִיּוֹם לְיוֹם הִיא צוֹפָה אַךְ לִמְקוֹם

זֶה כּוֹכַב-אַהֲבָתָהּ צָץ – וָדֹם.

בַּכֹּל, בַּכֹּל הִיא קְלַסְתֵּר-פָּנָיו רוֹאָה:

בְעָבֵי עֶרֶב-יוֹם שָׁם נַפְשׁוֹ תּוֹעָה;

אוֹ דַהֲרַת-סוּס כִּי-תִשְׁמַע בְּלֵיל נְדֹד –

מִיצוּעַ-שְׁנָתָהּ תֶּחֱרַד לָהּ חֲרֹד;

וּרְדוּפָה לִפְנֵי רוּחַ זֹה הַדְּהָרָה

הֵן קְרוֹבָה כְּבָר – וְאֵינָהּ עוֹד – עָבָרָה!

כֹּה יִמֹּג פִּתְאֹם מֵטֵאוֹר בִּנְפֹל:

הִנֵּהוּ פֹה! – וְאֵיפֹה הוּא? – אֵין-כֹּל!..
