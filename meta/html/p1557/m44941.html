מִקְרֵי הַזְּמָן רְדָפוּנִי


מתוך ספר עזר הדת. ראשית הוויכוח בין התורני והפילוסוף.



מִקְרֵי הַזְּמָן רְדָפוּנִי וַהֲדָפוּנִי / וְכַמַּטָּרָה

לַחִצִּים תּוֹלְדוֹת הַיָּמִים שָׂמוּנִי. / וְהוֹצִיאוּנִי מִמְּקוֹמִי

וְהִרְחִיקוּנִי / מֵאַנְשֵׁי עַמִּי / וּנְוֵה הֲדוֹמִי, / וַתִּשָּׂאֵנִי

רוּחַ עַל כְּנָפַיִם / וַתַּשְׁלִיכֵנִי עִיר יְרוּשָׁלַיִם. / וּבְעוֹדִי

מְהַלֵּךְ בִּדְרָכֶיהָ / וּמְסוֹבֵב אֶת מַעְגָּלֶיהָ / לְבַקֵּשׁ סִפְקִי

וַאֲרֻחָתִי / אוֹ נָדִיב מֵסִיר אַנְחָתִי, / וּבָבַת עֵינִי תִנְהַר /

כְּעוֹף הַמְשׁוֹטֵט סְבִיבוֹת הַנָּהָר – / נָשָׂאתִי אֶת עֵינַי וָאֵרֶא

וְהִנֵּה לַהֲקַת אֲנָשִׁים יוֹשְׁבִים / וּבִרְחוֹב הָעִיר נִצָּבִים, /

כָּעֲגֻלָּה הַמְסוֹבֶבֶת סְבִיב הַיָּרֵחַ / אוֹ כִקְלִפַּת הַפְּרִי

הַפּוֹרֵחַ, / וּבְתוֹכָם נִשְׁמַע קוֹל הֲבָרָה / קוֹל הֲמֻלָּה

נֶעְכָּרָה, / קוֹל עֲנוֹת גְּבוּרָה.

וָאֶקְרַב לְהִתְיַצֵּב עֲלֵיהֶם / לִרְאוֹת מַה

מַּעֲשֵׂיהֶם, / וְהִנֵּה בְתוֹכָם שְׁנֵי אֲנָשִׁים נִצִּים, /

מִתְקוֹטֲטִים וּמִתְלַחֲצִים, / שָׁבִים וְרָצִים, / כָּל אֶחָד לְנַצֵּחַ

לָאַחֵר אָצִים. / הָאֶחָד זָקֵן וּנְשׂוּא פָּנִים / בַּעַל שֵׂיבָה וּבָא

בַשָּׁנִים, / עַד אֲשֶׁר סָרוּ מִמֶּנּוּ שְׁנֵי הַמַּעֲדַנִּים. /

מַרְאֵהוּ נֶהְדָּר וּמְתֻקָּן, / בַּעַל כָּרֵס וְזָקָן, / גְּבֹהָה מְאֹד

קוֹמָתוֹ, / מְעֻטָּף בְּטַלִּיתוֹ, / בְּכֹבֶד רֹאשׁ כָּפוּף /

כְּאַגְמוֹן יָכֹף. / וְהַשֵּׁנִי נַעַר נָאֶה / יְפֵה תֹאַר וִיפֵה

מַרְאֶה, / כָּל רוֹאֶה עָלָיו יִשְׁתָּאֶה, / עֵינָיו כְּיוֹנִים /

לְחָיָיו שׁוֹשַׁנִּים / וְשִׂפְתּוֹתָיו שְׂשׂוֹנִים.

וַיִּשָּׂא הַזָּקֵן אֶת קוֹלוֹ וַיֹּאמַר ״שִׁמְעוּ

אֵלַי, עַמִּים, / וְהַטּוּ אָזְנְכֶם, לְאֻמִּים! / אַשְׁמִיעֲכֶם מַה

שֶּׁנִּסִּיתִי בִימֵי חָרְפִּי / הַאֲזִינוּ, הַשָּׁמַיִם, וַאֲדַבֵּרָה,

וְתִשְׁמַע הָאָרֶץ אִמְרֵי פִי. / שִׁמְעוּ אֵלַי, כָּל אַנְשֵׁי

הַקְּהִלָּה, / כִּי לֹא תְדַבֵּרְנָה שְׂפָתַי עַוְלָה / הֶן כָּל אֵלֶּה

שָׁמְעָה אָזְנִי וַתָּבֶן לָהּ. / הָאֵל אֲשֶׁר מִבֵּין הָאֻמּוֹת

בְּחָרָנוּ / וּבְרַחֲמָיו גְּאָלָנוּ / מֵחֹשֶׁךְ הוֹצִיאָנוּ / וּבְרֹב

חֲסָדָיו עַל צָרֵינוּ הִמְשִׁילָנוּ – / תּוֹרָה תְמִימָה נָתַן לָנוּ /

וְחֻקִּים יְשָׁרִים הֶעֱנִיקָנוּ / לְהַדְרִיכֵנוּ לְהַיְשִׁירֵנוּ /

בְּכָל מַעֲשֵׂה יָדֵינוּ, / וְגַם לִהְיוֹתָהּ סִבָּה / לִקְנוֹת בָּהּ

חַיֵּי עוֹלָם הַבָּא. / וּבִרְצוֹתוֹ לְזַכּוֹתֵנוּ / הִרְבָּה

בְחֻקּוֹתֵינוּ וּבְמִצְווֹתֵינוּ / לְהֵיטִיב אוֹתָנוּ בְּאַחֲרִיתֵנוּ. /

וְעַתָּה הִנֵּה עַם יָצָא מִמִּצְרַיִם / עַרְלֵי שְׂפָתַיִם, /

מְבַקְּשִׁים לִפְרֹשׂ רִשְׁתָּם עָלֵינוּ / לְבַטֵּל אֶת אֱמוּנָתֵנוּ /

וּלְהָפֵר עֲצָתֵנוּ / לְבָשְׁתֵּנוּ וּלְחֶרְפָּתֵנוּ, / לְהַבְאִישׁ אֶת

רֵיחֵנוּ / בֵּין מְבַקְשֵׁי רָעָתֵנוּ, / לְהַאֲרִיךְ אֶת גָּלוּתֵנוּ /

וּלְהַכְבִּיד אֶת עֻלֵּנוּ. / בְּחֻקּוֹתֵינוּ הַיְשָׁרִים שׁוֹגִים /

וּבַחֲכָמֵינוּ מַלְעִיגִים, / לְשׁוֹנָם נָכְרִית / מוֹאָבִית וְהַגְרִית,

/ דֵּעוֹתֵיהֶם מְשֻׁנִּים / וְסִפְרֵיהֶם חִיצוֹנִים, / מְחֻבָּרִים

וּלְקוּטִים מֵהַיְדִיעָה הַזָּרָה הַחִיצוֹנִית / הִיא הַחָכְמָה

הַיְּוָנִית / אֲשֶׁר הָאַפִּיקוֹרְסִין חִבְּרוּהָ / וְהַמִּינִין

יִלְמְדוּהָ / וְהַכּוֹפְרִים יֵדָעוּהָ, / וְהָיוּ לָנוּ לְשִׂטְנָה /

לַעֲקֹר וְלִסְתֹּר יְסוֹדוֹת הָאֱמוּנָה. / כָּל הַיּוֹם עָלֵינוּ צָרִים

וְצוֹרֲרִים / מְבִישִׁים וּמַחְפִּירִים, / גּוֹבְרִים עָלֵינוּ

וּמִשְׂתָּרֲרִים / דּוֹלְקִים אַחֲרֵינוּ וְרוֹדְפִים – / הֵם

הַנִּקְרָאִים פִּילוֹסוֹפִים. / וְכָל אִישׁ חָסִיד אֲלֵיהֶם לֹא

יִתְרוֹעָע, / חֲכָמִים הֵמָּה לְהָרַע. / וְהִנֵּה הַנַּעַר הַזֶּה עֲצַת

זְקֵנִים פָּרַע, / כִּי בָא אַחֲרֵיהֶם / וְהוֹלֵךְ בְּעִקְבוֹתֵיהֶם /

וּמִתְעַסֵּק לִקְרֹא בְסִפְרֵיהֶם / וּמְמַלֵּא אֶת דִּבְרֵיהֶם /

וְשׁוֹאֵב בִּכְלִי מֵימֵיהֶם / וְנוֹהֵג בְּמִנְהֲגֵי שִׁגְעוֹנֵיהֶם, /

יַסִּית וְיַדִּיחַ / יָרִיעַ אַף יַצְרִיחַ / יַבְאִישׁ יַסְרִיחַ, /

גַּלֵּי שְׁאוֹן דָּתֵנוּ יַשְׁבִּיחַ / וְכֹל אֲשֶׁר יַעֲשֶׂה לֹא

יַצְלִיחַ. / וְעַתָּה לְכוּ וְנַהַרְגֵהוּ / וּבָאֲבָנִים נִרְגְּמֵהוּ /

אוֹ בָאֵשׁ נִשְרְפֵהוּ. / וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:


רְאוּ נַעַר אֲשֶׁר דַּרְכּוֹ לְהַשְׁחִית / וְכָפַר חֹק אֲשֶׁר שָם אֵל וְהַבְּרִית

וְיַדִּיחַ מְתֵי מוּסָר וְיַסִּית / לְהַאְמִין שָׁוְא בְּבֶן מוֹאָב וְהַגְרִית.

לְזֹאת הִתְרַחֲקוּ מֶנּוּ, אֱמוּנִים, / וְהִזַּכּוּ וְהִתְרַחְצוּ בְּבֹרִית

וְרָגוֹם תִּרְגְּמוּהוּ בָאֲבָנִים / לְבַל יֵרָא וְיִשְׁאַר לוֹ שְׁאֵרִית

וְשָׂרוֹף תִּשְׂרְפוּ תּוֹךְ אֵשׁ וְגָפְרִית / וְיִשָּׁכַח שְׁמוֹ וִיהִי לְהַכְרִית.

מקור: שירמן – השירה העברית בספרד ובפרובאנס; הוצאת מוסד ביאליק ודביר תשי״ז תשכ״א.
