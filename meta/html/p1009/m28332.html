אֱלֹהַי וֵאלֹהֵי אֲבוֹתַי יָמַי חָלְפוּ

אֱלֹהַי וֵאלֹהֵי אֲבוֹתַי יָמַי חָלְפוּ עִם אֳנִיּוֹת אֵבֶה / אֵיכָכָה אוּכַל לַחְשֹׂף מַיִם מִגֶּבֶא

בָּרָאתָ אָדָם קְצַר יָמִים וּשְׂבַע רֹגֶז נֶאֱלָח וְנִתְעָב / אִם יַעֲלֶה לַשָּׁמַיִם שִׂיאוֹ וְרֹאשׁוֹ יַגִּיעַ לָעָב

גֶּבֶר יָהִיר כַּכְּפִיר שָׁאֹג יִשְׁאַג / וְאַחֲרִיתוֹ כַּמַּיִם יִשְׁתֶּה־לַּעַג

דְּבָרוֹ בְחַיָּיו אַחֲרָיו הֵידָד / וּבְמוֹתוֹ שֶׁמֶשׁ זָרְחָה וְנוֹדָד

הַגּוּף הַנָּגוּף יִשְׁכַּב לְמַעֲצֵבָה / הַנֶּפֶשׁ הַהִיא עֲוֹנָהּ בָּהּ

וְזֶה לָהּ תַּחַת אֲשֶׁר הֶאֱמִינָה לַשָּׁוְא / וְעָלֶיהָ יְחַזְּקוּ כְקַו לָקָו

זְכֹר כִּי רוּחַ חַיָּי וְהָיִיתִי לָבַז / אַחַר בִּסְבַךְ הַתְּלָאוֹת נֶאֱחַז

חָתוּם בִּצְרוֹר פִּשְׁעִי וְעָלָיו מַר אֶצְרָח / חַסְדְּךָ אֱלֹהִים בְּמֵי יֶשַׁע תִּמְרָח

טֶרֶם אֶסָּפֶה בְשָׁכְבִי בַלָּט / וּמִן הַמָּוֶת אֵין מָנוֹס וְאֵין מִפְלָט

יוֹצֵר כְּלִי דַעַת הֲלֹא מְעַט יָדָי / וּלְפָנֶיךָ תָּמִיד עוֹלֶה שְׁאוֹן רוֹדָי

כַּבִּיר כֹּחַ מִי לֹא יִירָאֶךָּ / אִם לֹא עַל פָּנִים יְבָרְכֶךָּ

לֹא אוּכַל גְּבוּרוֹתֶיךָ לְמַלֵּל / וְאִם אוֹסִיף לְהוֹדוֹת וּלְהַלֵּל

מִמְתִים יָדְךָ יי הַמְהֻלָּל בָּאִיִּים / מִמְתִים מֵחֶלֶד חֶלְקָם בַּחַיִּים

נַחְנוּ פָשַׁעְנוּ וּמָרִינוּ וְאַתָּה רַחוּם וְחַנּוּן / מִתְפַּיֵס בְּרַחֲמִים מִתְרַצֶּה בְּתַחֲנוּן

סָרַרְנוּ וְהָיִינוּ לָמַס / כִּמְעַט הָיִינוּ לְמִרְמָס

עַוְלָתָה קָצַרְנוּ וְאָוֶן נִזְרָע / לָכֵן מֵעַבְדוּתֵנוּ לֹא נִגְרָע

פָּחַדְתִּי פֶּן תְּדוּשֵׁנִי בְאַף / כִּי הִסְתַּרְתָּ פָּנֶיךָ בְּשֶׁצֶף קֶצֶף

צָמְתוּ בַבּוֹר חַיָּי וְטָבַעְתִּי בַבּוֹץ / וְאַתָּה תִּצְפֶּה מִבַּיִת וּמִחוּץ

קַצְתִּי בְּחַיַּי וְהָיִיתִי לִשְׂחוֹק / וּפָעַרְתִּי פִי לִבְלִי חוֹק

רוֹכֵב שָׁמַיִם אַתָּה אַדִּיר וְנָאוֹר / אָמַרְתָּ יְהִי אוֹר וַיְהִי אוֹר

שׁוֹמֵעַ תְּפִלָּה מַה בֶּצַע לְעַנּוֹת נָפֶשׁ / דְּלֵה בֶן אֲמָתְךָ מִטִּיט וָרֶפֶשׁ

תַּחֲנוּן הַקְשִׁיבָה וְאֵלַי הִתְעַשֵּׁת / פְּצֵנִי מִיַּד בְּנֵי רוֹבֶה קַשָּׁת

אֶעֱבָדְךָ טֶרֶם אֶהְיֶה לְזָרָא / אַרְיֵה שָׁאָג מִי לֹא יִירָא

בַּעֲלוֹתִי עַל לְבָבִי יְמֵי גִּדְּלַנִי כְּאָב / בֵּן חָכָם יְשַׂמַּח אָב

נוֹצֵר חֶסֶד מִשְׁפָּטֶיךָ בַּל יִהְיוּ כְאַיִן / נֶפֶשׁ בְּנֶפֶשׁ עַיִן בְּעַיִן

רַחוּם חַיָּתִי אַל תַּעְכּוֹר / בְּרוֹגֶז רַחֵם תִּזְכּוֹר

בָּכִיתִי בַּלַּיְלָה וְדִמְעִי כַמַּיִם זָב / אַל תִּהְיֶה לִי כְּמוֹ אַכְזָב

נַפְשִׁי עָלֶיהָ תִּלְעַג כָּל עָיִן / וּשְׁכוּרַת וְלֹא מִיָּיִן

חַטָּאתִי וַעֲוֹנִי תִסְלַח / חֶסֶד תַּחְפֹּץ וְלֹא זָבַח

לִישׁוּעָתְךָ קִוִּינוּ יָשִׁישׁ זָקֵן וְעוֹלֵל / יָגֵל יַעֲקֹב יִשְׂמַח יִשְׂרָאֵל

קַבֵּל רִנַּת עֶבֶד דַּל אֲשֶׁר בָּךְ חָשַׁק / וְאַל תַּעֲמֹד עַל הַפֶּרֶק

מוֹחֵל וְסוֹלֵחַ פֶּשַׁע וְחַטָּאת וַאֲשָׁמִים / אֵל מֶלֶךְ יוֹשֵׁב עַל כִּסֵּא רַחֲמִים.
