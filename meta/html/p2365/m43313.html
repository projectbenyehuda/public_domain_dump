אָשִׁירָה כִנּוֹר

שיר תפארת לכבוד השר אליה דוד יוסף עזרא הי״ו ונות ביתו מזל טוב הגברת.

סימן: אליה מזל ט׳וב.


אָשִׁירָה כִנּוֹר, הַנֵּבֶל אָעִירָה / שִׁיר חָדָשׁ אֲזַמְּרָה אָשִׁירָה

לִכְבוֹד רוֹדֵף חֶסֶד רַב פְּעָלִים / אֵלִיָּה, מַלְאָךְ מוֹשִׁיעַ רַב מַהְלָלִים

לִי מַה טּוֹבוּ דְרָכָיו, מַה מְּאֹד נָעֲמוּ / מַה גָּדְלוּ חֲסָדָיו, מַה מְּאֹד עָצְמוּ

נֶפֶשׁ רְעֵבָה מֵרֹב טוּבוֹ יַעֲנִיקָה / וּלְעָנִי וְאֶבְיוֹן שֹׁד צִדְקוֹתָיו יֵינִיקָה

יָדִיד, מָגֵן לְעַמּוֹ הִנֵּהוּ, עֵזֶר בְּצָרָתוֹ / כְּבוֹד אֶחָיו יָרִים וְזֹאת תְּהִלָּתוֹ

בֵּית תְּפִלָּה לִבְנֵי עַמּוֹ יָקִימָה / לְפָאֲרוֹ לְהַדְּרוֹ נִדְבַת לִבּוֹ יָרִימָה

הֵן רָמִים יִבְנֵהוּ בְעִיר כַּלְכַּתָּה הַבִּירָה / שֵׁם אָבִיו הַמְרוֹמָם בּוֹ נֵצַח לְהַזְכִּירָה

אָבִיו רַב הַחֶסֶד לְאֶחָיו עֶזְרָה / הַשַּׂר הַנַּעֲלֶה דָּוִד יוֹסֵף עֶזְרָא

מִבְּנוֹת מְלָכִים נְוַת בֵּיתוֹ מַזָּל טוֹב הַגְּבֶרֶת / כְּגֶפֶן פּוֹרִיָּה צְפִירַת תִּפְאֶרֶת

כָּמוֹהוּ תְחוֹנֵן דַּלִּים מַאֲלִיפוֹת צִדְקוֹתֶיהָ / וּבְמִפְעָלֶיהָ הָרַבִּים אֵין עָרוֹךְ אֵלֶיהָ

זֹאת מִי לֹא יִתְעוֹרֵר גְּדֻלָּתָם לְהַשְׁמִיעַ / מִי לֹא יִתְחוֹבֵב תִּפְאַרְתָּם לְהַבִּיעַ

מִי לֹא יַעְתִּירָה בַיּוֹם שִׁבְעָתַיִם / בְּעַד אֹרֶךְ יְמֵיהֶם לְשׁוֹכֵן שָׁמַיִם

לָכֵן כִּימֵי הַשָּׁמַיִם יַאֲרִיךְ יְמֵיהֶם / וְרֹב גְּדֻלָּתָם כְּרֹב חַסְדֵיהֶם

שְׁמָם הַנַּעֲלֶה כְאוֹר יוֹמָם זוֹרֵחַ / וּבֵיתָם לְעוֹלָם כְּשׁוֹשָׁן יַפְרִיחַ

טוֹב לְיִשְׂרָאֵל, הִנָּם כַּנֵּזֶר עַל מֵצַח / תִּשְׂגֶּה תִּפְאַרְתָּם, יִחְיוּ לָנֶצַח

יַעֲלוּ מַעְלָה, כַּתָּמָר יִפְרָחוּ / יָרוּם הוֹדָם וְכַשֶּׁמֶשׁ יִזְרָחוּ

שְׁמָם לְטוֹב וּלְתִפְאֶרֶת אַזְכִּירָה / אָנָּא ה׳ צְבָאוֹת אֵלֶיךָ אַעְתִּירָה

מִמְּרוֹם שָׁמֶיךָ עֲלֵיהֶם תַּשְׁגִּיחָה / בִּימֵיהֶם וּבְיָמֵינוּ אֵלִיָּה הַצְמִיחָה

מַזָּל צוֹמֵחַ לִיהוּדָה וִירוּשָׁלַיִם / כַּעֲתֶרֶת פַרַג׳ חַיִּים בֶּן הר״ש יְהוּדָה וִירוּשָׁלַיִם.
