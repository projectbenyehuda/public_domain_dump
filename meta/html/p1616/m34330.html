הוֹלֵךְ הָלוֹךְ וְנָסוֹעַ

מתוך קונטרס המליצות והשירים נגד נכבדי סאראגוסה.

עליו רשום בכתב יד: וזאת המליצה בלתי שקולה שלחתיה לשרי סאראגוסה הסוררים.


א

הוֹלֵךְ הָלוֹךְ וְנָסוֹעַ / בְּהָרֵי הַגִּלְבֹּעַ / פָּגַעְתִּי בְעִיר גְּדוֹלָה אֲשֶׁר אֲבָנֶיהָ יֻסְּדוּ עַל דָּמִים וּמִרְמָה / וְהָעָם אֲשֶׁר רָאִיתִי בְתוֹכָהּ אַנְשֵׁי מִדּוֹת מְגֻנּוֹת יָמוּתוּ וְלֹא בְחָכְמָה, / אַנְשֵׁי בוֹגְדוֹת וְאֶשְׁתַּדּוּר עָבְדִין בְּגַוַּהּ וְקֶשֶׁר שֶׁל קְיָמָא, / לֹא יָרֵעוּ לֹא יַשְׁחִיתוּ כִּי אִם גָּנוֹב וְרָצוֹחַ וְנָאוֹף וַעֲשׂוֹת הַמְזִמָּה. / לֹא חֲמוֹר אֶחָד מֵהֶם, כִּי כָל הָעֵדָה פְּתַיּוּת וּבַל יָדְעָה מָּה! / עֵדָה פְחוּתָה מֵעֲשָׂרָה בְאֵיכוּתָהּ – יֶשׁ בָּהּ לְאַלְפֵי רְבָבָה / אָדָם, אֲשֶׁר לֹא יָדַע בֵּין יְמִינוֹ לִשְׂמֹאלוֹ וּבְהֵמָה רַבָּה. / עַם כֶּבֶד עָווֹן הֵמָּה, שָׂרֵיהֶם וְשׁוֹפְטֵיהֶם אֶת יְיָ לֹא דָרָשׁוּ / וְנוֹדְעוּ כִּי עֲרוּמִים הֵם וְלֹא יִתְבּוֹשָׁשׁוּ, / כֵּלִים רֵיקִים חֲלוּלִים חֲלוּלִים / כְּעוֹלְלִים, / לֹא רָאוּ אוֹר רַק הַחוּצָה אֶל הָעַיִן, / בְּנֵי בְלִי שֵׁם נִכְּאוּ מֵאַיִן / וּגְוִיּוֹת מְחֻטָּבוֹת כְּתַבְנִית אִישׁ, וְאָדָם – אַיִן! / בּוֹעֲרִים בָּעָם וְהֶדְיוֹטוֹת – / רָאשֵׁי הַמַּטּוֹת, / הַחַיָּטִים – / שׁוֹפְטִים, / וְרַצְעָנִים – / דַּיָּנִים / וְרָאשֵׁי סַנְהֶדְרָאוֹת, / אֵין אִישׁ בְּהַשְׂכֵּל – דְּבָרָיו יְכַלְכֵּל – וְאֵין לַדַּיָּן אֶלָּא מַה שֶּׁעֵינָיו רוֹאוֹת. / כָּל בּוּר אֵינוֹ יְרֵא / חֵטְא – הוּא יוֹרֶה, / וְכָל תָּפוּשׂ זָהָב וָכֶסֶף וְכָל רוּחַ אֵין בְּקִרְבּוֹ / עוֹשֶׂה רוֹשׁ וּבְרֹאשׁ הַקָּהָל יָבוֹא. / סוּסִים מְיֻזָּנִים / מַשְׁכִּימֵי קוּם בִּתְפִלּוֹת וּבְתַחֲנוּנִים, / וְאַחַר אֲשֶׁר יִתְפַּלָּלוּ – / אִישׁ אֶל אֵשֶׁת רֵעֵהוּ יִצְהָלוּ. / קִדְּמוּ עֵינֵיהֶם אַשְׁמוּרוֹת – / וְטֻמְאוֹתָם בְּשִׂמְלוֹתָם צְרוּרוֹת: / בְּיַד שְׂמֹאלָם אֲגֻדּוֹת קֶשֶׁר שֶׁל תְּפִילִין קְשׁוּרוֹת / וּבְיַד יְמִינָם הַשּׁוֹפָרוֹת. / וְאִם בְּאַחַת יִתְקָעוּ, / יִתְהַלְּלוּ בְיוֹם מָחָר וְנִשְׁבָּעוּ / כִּי בְרַבּוֹת בָּנוֹת כָּשְׁלוּ וּפָשָׁעוּ. / וְכָל אִשָּׁה חַכְמַת לֵב יְפַת תֹּאַר וִיפַת מַרְאֶה תֵּשֵׁב בְּפֶתַח עֵינַיִם / וְעִם הָעוֹגְבִים – הִתְנוּ אֲהָבִים – וּמְשַׂקְּרוֹת עֵינַיִם. / תַּמְרוּק בְּרַע עַל עֵינֵיהֶם וְהַזּוֹנוֹת רָחָצוּ / וּבְיַלְדֵי נָכְרִים בַּעֲלֵי נְעוּרִים מָאָסוּ, / וְכִי תְחִלֶּינָה לִזְנוֹת / תֹּלוֹת עַצְמָן בְּגוֹי וְדָתֵיהֶן שׁוֹנוֹת. / וְרֹב הַצְּנוּעוֹת שֶׁבָּהֶן / יִתְּנוּ יְקָר לְבַעֲלֵיהֶן: / כִּי אִישׁ עִבְרִי לְבַד יְעָרֶה פָתְהֵן / וְכָל בֶּן נֵכָר לֹא יֹאכַל בּוֹ, / וְהַבַּעַל הֶעָלוּב יִשְׂמַח וְיֹאמַר בְּלִבּוֹ: / מִמִּשְׁפַּחְתֵּנוּ הוּא! וְכִפֵּר עַל קַרְנוֹתָיו וְחֶרְפָּה לֹא נָשָׂא עַל קְרוֹבוֹ. / וְהַבַּחוּרִים יֵצְאוּ מִן הַחֹרִים וְנַפְשׁוֹתָם לֹא תִשְׂבַּעְנָה, / יַחְתְּרוּ בַחֹשֶׁךְ בָּתִים וְהַנָּשִׁים תִּשָּׁגַלְנָה, / הַקְּטַנִּים עִם הַגְּדוֹלִים גָּדְלוּ מַעֲשֵׂיהֶם בְּדִבְרֵי רְכִילוּת, / עָבְרוּ מַשְׂכִּיּוֹת לֵבָב בְּגַאֲוָה וּבְגֹדֶל לֵבָב כְּרוּם זֻלּוּת. / וְזִקְנֵי הָעָם וְשׁוֹטְרָיו נִשְׁבְּעוּ לַשָּׁוְא עַל פַּת לֶחֶם וּלְכַחֵשׁ יַלְבִּישׁוּ אַדֶּרֶת, / נִכְרְתָה הָאֱמוּנָה וַתְּהִי הָאֱמֶת נֶעְדֶּרֶת. / וּבִרְאוֹתִי שַׁמָּה וְשַׁעֲרוּרָה / וְכִי נָגַע לַשָּׁמַיִם מִשְׁפָּטָם וְנִתְמַלְאָה סְאָתָם וְחַטָּאתָם גָּבְרָה, / יָגֹרְתִּי מִפְּנֵי הָאַף וְהָעֶבְרָה / וְקִנֵּאתִי לֵאלֹהַי, נָשָׂאתִי תְפִלָּה בְּעַד הַשְּׁאֵרִית הַנִּשְׁאָרָה / וְנָשָׂאתִי קוֹלִי כְּקוֹל אִשָּׁה מְצֵרָה / וּבַבַּיִת פֵּרַטְתִּי עֲווֹנוֹתָם פְּעָמִים עֲשָׂרָה – / וְאֵין מֵקִיץ כִּי כֻלָּם יְשֵׁנִים, / עֲרֵלָה אָזְנָם שׁוֹכְבִים בְּחֵיק הַזְּדוֹנִים, / וְתוֹכַחְתִּי בִּדְרָשׁוֹתַי לֹא הֵשִׁיבוּ פוֹשְׁעִים עַל לֵב / וַיֶּהֱמוּ כַכֶּלֶב / אֵין עוֹשֵׂה טוֹב אֵין נָבוֹן וְחָכָם…


ב

מעל למליצה זו רשום: וטרם אכלה לדבר אכתוב התקנות והחקים לא טובים, אשר עשו ביניהם שטי כזב פונים אל רהבים.


עוֹד רָאִיתִי תּוֹעֵבוֹת גְּדוֹלוֹת / בְּהִתְאַסֵּף רָאשֵׁי עָם רָשֵׁי תְבוּנוֹת לְתַקֵּן תַּקָּנוֹת אוֹ תַקָּלוֹת / לָשִׂים מִכְשׁוֹלוֹת בַּמַּקְהֵלוֹת / וְלִגְזֹל בְּחִירָתָם אֲשֶׁר לֹא צִוָּה אוֹתָם יוֹשֵׁב תְּהִלּוֹת. / וַעֲנִיֵַּי הַצֹּאן לֹא רָאוּ / כִּי בְנַפְשׁוֹתָם הוּא, / לְהִתְעוֹלֵל בָּהֶם עֲלִילוֹת / וּלְעוֹלֵל אוֹתָם בַּמְּסִלּוֹת, / וְלֹא נָתְנוּ לָהֶם כָּתֵף סוֹרֶרֶת, / נִגְזֶרֶת עֲצֶרֶת, / בּוֹגְדִים קִיְּמוּ וְקִבְּלוּ / וְכָל עֵינֵיהֶם עַל עֲשֶׂרֶת בְּנֵי הָמָן תָּלוּ, / וְהֵם חָשְׁבוּ לִכְתֹּב עַל קְצָת יְחִידֵיהֶם שִׂטְנָה, / וַיִּפֹּל הַגּוֹרָל עַל יוֹנָה: / כִּי אוֹתִי אֲשֶׁר לֹא עָשִׂיתִי עִמָּהֶם רַק טוֹבָה עַד כִּי חָדַל לִסְפֹּר צוֹד צָדוּנִי כַּצִּפּוֹר. / לָכֵן קֹדֶם אֲפָרֵט הַהַסְכָּמוֹת, שִׁבְעָה מֵהֶם בְּסִפְרִי אֶזְכְּרֵם / וּבִמְחוּגַת עֵטִי אֲתָאֲרֵם / וּבְצִיּוּר רִשְׁעָתָם אֲצַיְּרֵם, / לְמַעַן יָרוּץ הַקּוֹרֵא בִּכְתָבִי, וְאַנְשֵׁי אֱמֶת עֵת דִּבְרֵי הַסֵּפֶר הַזֶּה יִקְרְאוּ / דְּמוּת פְּנֵיהֶם עַיִן בְּעַיִן יִרְאוּ. / אֵל אֱלֹהִים יַצְמִיתֵם וִימִיתֵם כְּאֶחָד / וְיִפְּלוּ שְׁבַעְתָּם יַחַד!

הָרִאשׁוֹן מֹשֶׁה עָפְיָה הַמִּתְנַשֵּׂא לְכֹל לְרֹאשׁ רֹאשׁ וּתְלָאָה, / כָּמוֹהוּ לֹא נִרְאָה, / מִשְׂתָּרֵר גַּם הִשְׂתָּרֵר עַל הַנֵּי יְהוּדָאֵי טִפְּשָׁאֵי, / הוֹלֵךְ רָכִיל וְאוֹכֵל קַרְצֵיהוֹן דִּיהוּדָאֵי, / לוֹקֵחַ נְפָשׁוֹת בַּאֲרֶשֶׁת רֶשֶׁת שְׂפָתָיו וְנָפַל בַּעֲצוּמָיו חֵלְכָּאִים, / רַגְלָיו לָרַע יָרוּצוּ כְּאַחַד הַצְּבָאִים. / עוֹשֶׂה עֹשֶׁר / וְלֹא בְיֹשֶׁר, / פִּיהוּ מָלֵא חִצִּים וּמָוֶת כְּמִתְלַהְלֵהַּ הַיּוֹרֶה זִקִּים, / הוּא אָמוֹן הִרְבָּה אַשְׁמָה וּבְרִית לֹא יָקִים. / וְכִי יִכְרֹת בְּרִית עִם רֵעֵהוּ – / לָעֶרֶב יְמוֹלֵל וְיָבֵשׁ / וַיְהִי יָדָיו אֱמוּנָה עַד בּוֹא הַשֶּׁמֶשׁ! / לַהַט הַחֶרֶב הַמִּתְהַפֶּכֶת, / אֲשֶׁר בְשַׁלֶּכֶת / הָרַע מִתְנַשֵּׂא וּמִתְגָּאֶה – / אֶפֶס קָצֵהוּ תִרְאֶה וְכֻלּוֹ לֹא תִרְאֶה.

וְהַשֵּׁנִי אַבְרָהָם אַבִּיּוּד הָאָדֹם הָאָדֹם / הוּא עֵשָׂו אֲבִי אֱדוֹם, / הַבּוֹגֵד בּוֹגֵד לִבּוֹ יִקְבָּץ־אָוֶן לוֹ, וְכָל רַע לִבּוֹ רָחַשׁ, / שְׁפִיפוֹן עֲלֵי אֹרַח וְעָרוּם כַּנָּחָשׁ. / וּבְשָׂרוֹ נָחוּשׁ וְגִיד בַּרְזֶל עָרְפּוֹ, / יֶחְשְׁכוּ כּוֹכְבֵי נִשְׁפּוֹ, / סַעַר מִתְחוֹלֵל – / אַיְלוֹת הַקְּהִלּוֹת יְחוֹלֵל. / וְהֵן שָׁמַיִם יָעִיב בְּאַפּוֹ, / פּוֹרֵק עֹל וְנַפְשׁוֹ כַּחַשׁ מְלֵאָה וּזְדוֹנִים, / וְגַם הִיא הָרָה לִזְנוּנִים. / קְטַן קוֹמָה בְדֵעָה, וְעֵת יִמָּלֵא עֶבְרָה וָזַעַם – / מִשִּׁכְמוֹ וָמַעְלָה גָּבֹהַּ מִכָּל הָעָם!

וְהַשְּׁלִישִׁי / דּוֹמֶה לְעוֹרֵב וְכוּשִׁי / הוּא מֹשֶׁה בֶן־יָמִין / הֵן בִּקְדוֹשָׁיו לֹא יַאֲמִין, / זֶפֶת בּוֹעֵרָה, / גֹּלֶם בְּלִי צוּרָה. / הָרוֹכֵל הַנּוֹכֵל, הַשֶּׁרֶץ הַשּׁוֹרֵץ / כֻּלּוֹ אֶרֶס, / וּבְנֶשֶׁךְ וּבְמַרְבִּית כָּל עַמּוֹ אוֹכֵל. / עוֹשֶׂה סַמִּים / מִגַּחֲלֵי רְתָמִים / וּמִרְמוֹת וָתֹךְ יִתֵּן בְּרָאשֵׁי בְּשָׂמִים. / אִישׁ שׁוֹתֶה כַמַּיִם עַוְלָה / אֲשֶׁר חֲמָתוֹ חֲמַת תַּנִּינִים וְיֵין תַּרְעֵלָה, / הֲפַכְפַּךְ דַּרְכּוֹ וָזָר / וְרֹאשׁ פְּתָנִים אַכְזָר, / אִישׁ רִיב וּמָדוֹן / בְּיוֹם הִבָּרְאוֹ כּוֹנֲנוּ כַחַשׁ וְכָזָב וּפָרַח הַזָּדוֹן. / עֲרִירִי יָמוּת וַחֲמָסוֹ בְּרֹאשׁוֹ יִדָּרֵשׁ / וּבֵיתוֹ שָׂדֶה תֵחָרֵשׁ!

וְהָרְבִיעִי וִידַאל אַבּוּלְפֵידָה / עַל זִכְרוֹנוֹ פַּחַד קְרָאַנִי וּרְעָדָה. / אִישׁ צַר וְאוֹיֵב, הָמָן הָרַע, / הַסּוֹפֵר הַכּוֹפֵר כֻּלּוֹ חָנֵף וּמֵרַע. / עֵינִי יוֹרְדָה מַיִם וְתָנוּד כְּנוּד הַקָּנֶה כִּי אַעֲלֶה עִל סֵפֶר זִכְרוֹנוֹ, / כִּי בָרֶשַׁע וּבַפֶּשַׁע עָבַר עַל אַנְשֵׁי מִינוֹ. / אֲבִי אֲבוֹת הַטֻּמְאָה וְאוֹנוֹ בִּשְׁרִירֵי בִטְנוֹ, / עָפָר וָאֵפֶר / לֹא לָמַד סֵפֶר / וְלֹא נִשְׁמַר מֵחֲטֹא בִלְשׁוֹנוֹ, / קֻלְמוֹסוֹ כִּנְשֹׂא יַעֲרֹף כַּמָּטָר שָׁוְא / וּדְבַר כָּזָב, / וַתָּשָׁב קֶשֶׁת רְמִיָּה בְּרִשְׁתּוֹ וְקִנּוֹ / וַיְהִי כִמְצַחֵק כִּי יַקִּיף הַשְּׁטָר וְיַקְדִּים זְמַנּוֹ, / וּמְכַתְּבִים עָמָל יִלְמְדוּ לֶקַח מֵרֹעַ מַעֲלָלוֹ וּזְדוֹנָיו, / כִּי יְדַבֵּר הַוּוֹת מוֹלֵל בְּרַגְלָיו וְקוֹרֵץ בְּעֵינָיו, / אָוֶן יַחְשֹׁב עַל חֲבֵרוֹ / וּטְבָחוֹ אוֹ מְכָרוֹ / וַיִּשָּׂא בְשִׁנָּיו בְּשָׂרוֹ. / אֶבֶן מִקִּיר תִּזְעַק עַל זְדוֹנוֹ, / יְגַלּוּ שָׁמַיִם עֲווֹנוֹ!

וְהַחֲמִישִׁי אִישׁ שְׁלוֹמִי אֲשֶׁר בָּטַחְתִּי בוֹ הוּא בּוֹנְאַשְׁטְרוּק דֵּי שָׂלִינֵשׂ אֲשֶׁר בְּרִיתִי אִתּוֹ בְּרִית מֶלַח – / הוּא יָם הַמֶּלַח, / מֶלַח סְדוֹמִית, / מְסַמֵּא אֶת הָעֵינַיִם בְּתַרְמִית. / הוּא הִקְהִיל עָלַי אֶת כָּל הָעֵדָה, כִּי כֹה מִשְׁפָּטוֹ לְהָרֵעַ לְכָל רֵעַ וְעָמִית / הָעֲצָמוֹת הַיְּבֵשׁוֹת אִישׁ כָּזֶה לְכָל דָּבָר אֱמוּנָה אֹמֶן / וְהָיָה דָשֵׁן וְשָׁמֵן / לְחַרְחֵר רִיב וּלְכָל דִּבְרֵי בֶלַע / וְנֶפֶשׁ אוֹהֲבָיו יְקַלְּעֶנָּה בְּתוֹךְ כַּף הַקֶּלַע. / כְּלִי רֵיק מָלֵא מִרְמוֹת כְּרִמּוֹן, / תּוֹלַעַת וְלֹא אִישׁ וְקוֹל דְּבָרָיו כְּקוֹל הָמוֹן, / וְהוּא מִתְקַדֵּשׁ וּמִטַּהֵר וְשֶׁרֶץ בְּיָדוֹ, / וְיִתֵּן לְמוֹפֵת כִּי אָבִיו שֹׂרַף – וְהִנֵּה צֹרַף בְּיוֹם אֵידוֹ. / בְּמֹאזְנֵי מִרְמָה לְבָבוֹ, זְכוּת אֲבוֹתָיו יָשִׂים לְקָלְבּוֹן, / אוֹהֵב פֶּשַׁע וְאוֹהֵב מַצָּה לְתֵאָבוֹן, / וְכַפָּיו מְלֵאוֹת חָמָס וְגוֹזֵל קְהָלוֹ בְּעֵת יוֹשֵׁב בְּחֶשְׁבּוֹן.


אֵלֶּה הַחַטָּאִים בְּנַפְשׁוֹתָם / נִמְנוּ וְגָמְרוּ בִסְכָמוּתָם / לְצוֹדֵד נְפָשׁוֹת בְּתַרְמִיתָם, / וְזֹאת תּוֹרַת הָאָדָם לְקַיְּּמָהּ וּלְחַזְּקָהּ נְתָנוּהָ לִשְׁנֵי חַזָּנִים זְהִירִים / לְקָרְאָהּ בְּכָל חֹדֶשׁ בְּשִׂמְחָה וּבְשִׁירִים / לְפַרְסוּמֵי נִסָּא דְּלָא תִשְׁתַּכַּח בְּאִגְֶּרֶת פּוּרִים, / וְאַזְכִּיר תָּאֳרֵי שְׁנֵיהֶם עַל מַתְכֹּנֶת פִּשְׁעָם וְרִשְׁעָם / לְאַחַד קָרָאתִי חוֹבְלִים וְהָאַחֵר בִּלְעָם. הַשִּׁשִּׁי אַבְרָהָם שׁוּעַיְבּ נִקְשֶׁה וְרָעֵב, רַב הַחוֹבֵל, שַׂר הַטַּבָּחִים דְּאִתְיְלִיד בְּמַאְדִּים, / שׁוֹפֵךְ דָּמִים מִכַּמָּה צְדָדִים, / וְיָצָא מֵחֵלֶב עֵינָיו וּפָנָיו עַל זְדוֹנָיו עֵדִים. / מִצְוַת ״לֹא תִנְאַף״ / יִרְדֹּף בְּאַף, / אַף כִּי אָמַר אֱלֹהִים ״לֹא תִקַּח שֹׁחַד״ – / כֻּלּוֹ רוֹדֵף שַׁלְמוֹנִים וְאוֹהֵב שֹׁחַד. / חַזָּן הַזָּן כָּל הָעוֹלָם בִּזְנוּנָיו / וּשְׂעִירַת עִזִּים לְחַטָּאת מָצָא חֵן בְּעֵינָיו. וְהַשְּׁבִיעִי הַטָּמֵא וְהֶעָרֵל / אַבְרָהָם מוֹרִיאֵל, / דּוֹמִים פָּנָיו לִמְגַדֵּל חֲזִירִים וְנוֹתֵן לִפְגָם טַעַם, / שָׁמֵן כְּעֶגְלוֹן, מַמְאִיר כְּסִלּוֹן, וְסַר טַעַם / אוֹכֵל תַּרְטֵימַר בָּשָׂר וְלֹג יַיִן וְעֶבְרָה וָזַעַם, / אֲחִי אֶשְׁכּוֹל פִּיכֹל שַׂר צְבָאוֹ / וּכְהִתָּעוֹת שִׁכּוֹר בְּקִיאוֹ. / הוֹלֵם הַבְּרִיאוֹת כְּהֹלֶם פַּעַם – / קוֹל יְשׁוֹרֵר בַּחַלּוֹן כְּקוֹל רַעַם, / וְזֶה הַסָּכָל יַעֲלֶה לַתֵּבָה כְּפַעַם בְּפַעַם / וְיִפְתַּח פִּיו לִקְרֹא הַתַּקָּנוֹת וִיקַלֵּל אֶת יוֹמוֹ וּכְפִתְחוֹ עָמְדוּ כָל הָעָם / וְהֶהָמוֹן הֶעָלוּב בְּשָׁמְעָם אֶת דִּבְרֵי הָאָלָה וְהַקּוֹל הוֹלֵךְ וְחָזֵק יִלְבְּשׁוּ חֲרָדוֹת / וּפְנֵיהֶם אִישׁ אֶל אָחִיו יֹאמַר ״מָה הַחֻקִּים הָאֵלֶּה וְהָעֵדוֹת הַנִּכְבָּדוֹת?״ / וְלֹא יָדְעוּ כִּי כָל אֵלֶּה מִשְׁטַח חֲרָמִים וּמְצוּדוֹת!


ג

מעל למליצה זו רשום: טופס התקנות שהסכימו ואותם החכימו.

לפי סגנונה הרי זו פארודיה על נוסח התקנות המקובל.


זֶה טֹפֶס הַחֻקִּים / הַמְיֻסָּדִים בִּמְחוֹל מְשַׂחֲקִים, / וְהֵם רֶמֶז לְמוּמֵיהֶם דְּכָלְּהוֹ אִיתַנְהוֹ בְהוֹ אִישׁ לֹא נֶעְדַּר, / אָמְנָם לַשָּׁוְא צָרַף צָרוֹף, כִּי לֹא יָדְעוּ הִכָּלֵם וְאֵין תְּבוּנָה בָם וְלֹא הָדָר.

תְּחִלָּה הָעֲשָׂרָה נִבְרָרִים הִסְכִּימוּ / שֶׁכָּל מְלֶאכֶת יְיָ יַחֲרִימוּ / וּבִעֲרוּ הַקֹּדֶשׁ מִבֵּיתָם וְאֵשׁ לֹא יָשִׂימוּ.

עוֹד הִסְכִּימוּ תֵּת קֹדֶשׁ וְצָבָא מִרְמָס / וְלָבִיא מַמְזֵר בַּקָּהָל וּלְהַתִּיר אֶת הַשֶּׁרֶץ וְאֶת הָרָחָמָה וְאֶת הַתַּחְמָס, / וְכִי יָקוּם עֵד חָמָס / שֶׁיִּהְיֶה פָטוּר מִן הַמַּס, / וְכִי יִכְרֹת בְּרִית אִישׁ עִם רֵעֵהוּ – וְחַם הַשֶּׁמֶשׁ וְנָמַס! / וְכָל אִשָּׁה יְפַת תֹּאַר לֹא תַעְדֶּה עֲדִי כִּי אִם עָגִיל וְכוּמָז.

עוֹד הִסְכִּימוּ שֶׁכָּל הוֹלֵךְ רָכִיל, / אַלְפַּיִם בַּת יָכִיל, / וְכָל דַּיָּן שֶׁיַּטֶּה אֶת הַדִּין וְשֹׁחַד עַל נָקִי יִקַּח – / שֶׁיִּתֵּן דַּנְקָא לִצְדָקָה וְיַשְׁקוּהוּ מִיֵּין הָרֶקַח. / וְכָל הַגּוֹנֵב תַּפּוּחֵי סֵפֶר תּוֹרָה / אוֹ הֵסִיר הַמִּצְנֶפֶת וְהֵרִים הָעֲטָרָה / יְעַטְּרוּהוּ בְּקוֹצֵי הַמִּדְבָּר וּבְשׁוֹשַׁנִּים לִצְפִירַת תִּפְאָרָה.

עוֹד הִסְכִּימוּ שֶׁכָּל מַקְרִין מַפְרִיס יִשָּׂא קֶרֶן חָזוּת בֵין עֵינָיו, גַּנָּב / שֶׁיִּכָּנֵס בַּלַּיְלָה בְּעַד הָאֶשְׁנָב – / בְּיוֹמוֹ יִנָּתֵן שְׂכָרוֹ וְשָׂמוּ מַסְוֶה עַל פָּנָיו.

עוֹד הִסְכִּימוּ שֶׁקֹּדֶם שֶׁתָּבוֹא הַכַּלָּה לַבַּעַל / שֶׁכָּל הָעָם מִקָּצֶה אוֹתָהּ יִבְעַל, / וְאִם אֵין לוֹ דָּמִים – / נָא יִתֵּן לְעֵרָבוֹן הַטַּלִּית וְהַמַּאֲכֶלֶת וְלֹא יִשְׁפֹּךְ עוֹד דָּמִים.

עוֹד הִסְכִּימוּ שֶׁכָּל מִי שֶׁיִּמְצָא אִישׁ אֶחָד עִם אִשְׁתּוֹ בַמִּטָּה / תַּחַת הַשִּׂמְלָה לוּטָה – / שֶׁלֹּא יִהְיֶה רְשׁוּת בְּיַד הַבַּעַל לְהַתִּיר אֲגֻדּוֹת מוֹטָה, / וְיַעַן הַבָּחוּר לָלוּן בְּצֵל קוֹרָתוֹ נָטָה, / מִמְּקוֹמוֹ לֹא יָמִישׁ וְהוּא יַעֲלֶה מַעְלָה מַּעְלָה וְהַבַּעַל יֵרֵד מַטָּה מַטָּה.

עוֹד הִסְכִּימוּ שֶׁכָּל מִי שֶׁמְּמַעֵט אֶת הַדְּמוּת וְלֹא יַעֲסֹק בִּפְרִיָּה וּרְבִיָּה לְהַרְבּוֹת יַלְדֵי הָעִבְרִים / שֶׁיַּשְׁקוּהוּ מִן הַמַּיִם הָמְאָרְרִים. / בְּאִשְׁתּוֹ אָמְרוּ, קַל וָחֹמֶר – בְּאֵשֶׁת חֲבֵרִים! / וְכָל אִשָּׁה שֶׁתִּתֵּן עֵינֶיהָ בְאַחֵר וְתִשְׁאַל גִּטָּהּ / כּוֹפִין אֶת הַבַּעַל לְגָרְשָׁהּ וְיַאֲכִילוּהוּ שְׂעוֹרִים תַּחַת חִטָּה.



עוֹד הִסְכִּימוּ שֶׁלֹּא תִכָּנֵס אִשָּׁה לַמַּרְחֵץ / כִּי אִם עִם בָּחוּר שֶׁיּוֹרֶה כַחֵץ, / וְשֶׁלֹּא תִכָּנֵס כַּלָּה לְחֻפָּה / כִּי אִם עִם אַגְמוֹן וְכִפָּה / וְתַחַת חֲגוֹרָה – נִקְפָּה.

עוֹד הִסְכִּימוּ שֶׁלֹּא יִהְיוּ לַכַּלָּה שׁוֹשְׁבִינִים / כִּי אִם בְּנֵי זְנוּנִים, /וּלְצִדֵּי הַחֲתָנִים יָקוּמוּ אֲשֵׁרִים וְחַמָּנִים, / וְשֶׁלֹּא יֹאכְלוּ בִסְעוּדַת הֶחָתָן / כִּי אִם תֶּרֶשׁ וּבִנְתָן / וַאֲבִירָם וְדָתָן / בְּגִנַּת הַבִּיתָן.

עוֹד הִסְכִּימוּ שֶׁכָּל אִישׁ טְהָר־לֵב / עָלוּב וְאֵנוֹ עוֹלֵב / שֶׁיַּשְׁלִיכוּ אוֹתוֹ לַכֶּלֶב, / וְכָל מִי שֶׁלֹּא יִשָּׁבַע לַשָּׁוְא / וּבְמוֹשַׁב לֵצִים לֹא יָשַׁב / שֶׁיִּהְיֶה נֶחְשָׁב כְּגֵר תּוֹשָׁב / וּבַגּוֹיִים לֹא יִתְחַשָּׁב. עוֹד הִסְכִּימוּ שֶׁבִּמְקוֹם שֵׁשׁ וְרִקְמָה וְשָׁנִי עִם עֲדָנִים / וְהַמַּחֲלָצוֹת וְהַגִּלְיוֹנִים / שֶׁתֵּצֶאנָה הַנָּשִׁים בְּמַחֲצֶלֶת שֶׁל קָנִים, / וְתַחַת בֹּשֶׂם – מַק, / וְתַחַת פְּתִיגִיל – מַחֲגֹרֶת שַׂק, / וְהַטַּבָּעוֹת וְנִזְמֵי הָאַף / יֵלְכוּ לָחֳרִי אַף.

וְעוֹד הִסְכִּימוּ שֶׁכָּל מִי שֶׁלֹּא יֵלֵךְ בָּרְחוֹבוֹת בְּגִלּוּי הָרֹאשׁ / וְיִשָּׂא כּוֹבַע עַל רֹאשׁ וּבֵית רֹאשׁ / שֶׁיִּשְׁתֶּה קֻבַּעַת לַעֲנָה וָרֹאשׁ / וְיֵלֵךְ אָבֵל וְחָפוּי רֹאשׁ.

עוֹד הִסְכִּימוּ שֶׁכָּל נַעֲרָה יָפָה וַהֲגוּנָה הַלּוֹבֶשֶׁת תִּלְבֹּשֶׁת בֹּשֶׁת וַעֲנָוָה כְּנֶזֶם עַל אַפָּהּ / וּכְנָפֶיהָ בְּכֶסֶף הַצְּנִיעוּת נֶחְפָּה / וְלֹא שָׁמְעָה קוֹל נוֹגֵשׂ וְלֹא הֵחֵל לְפַעֲמָהּ רוּחַ הַחֵשֶׁק וְלֹא בָעֲרָה בָהּ אֵשׁ וְתַבְעֵרַת קִבְרוֹת הַתַּאֲוָה וְרִשְׁפָּהּ / וּתְצַעֵר כָּל נֶפֶשׁ מְאַהֲבֶיהָ וְהִיא אִשָּׁה בִּבְתוּלֶיהָ הִגִּיעָה לְחֻפָּה – / וְיַעַן צִעֲרָה אֶת הַנֶּפֶשׁ וְרַבִּים חֲלָלִים הִפִּילָה בְּיָפְיָהּ – תִּהְיֶה לְבֹשֶׁת וְגַם לְחֶרְפָּה / וְתָבֹאנָה תִמְנַע פִּילֶגֶשׁ וְרִצְפָּה / וּבְיָדָן רִצְפָּה / וְהָיָה לְמַאֲכֹלֶת אֵשׁ עַל כָּל כָּבוֹד חֻפָּה. / וְאִם עוֹד תַּקְשֶׁה עָרְפָּהּ, / תֵּצֵאןָ עָרְפָּה וְזִלְפָּה / וּמָלְקוּ אֶת רֹאשָׁהּ מִמּוּל עָרְפָּהּ, / וּתְקַצֵּצְןָ אֶת כַּפָּהּ / וַיִּגַּשׁ גֵּיחֲזִי לְהָדְפָהּ.
