יְיָ רִאשׁוֹן וְאַחֲרוֹן

וּבְכֵן יְיָ מָעוֹן אַתָּה


יְיָ רִאשׁוֹן וְאַחֲרוֹן בְּלִי כַחַד

יְיָ אֱלֹהֵינוּ יְיָ אֶחָד


יְיָ אֲזַי בִּשְׁתֵּי יְסוֹדוֹת הַכֹּל גַּלּוֹתָ

יְיָ אֱלֹהִים אַתָּה הַחִלּוֹתָ


[יְיָ בּ שׁלשׁ

יְיָ בּ ]


יְיָ גִּלְגַּלְתָּ אַרְבַּע בְּצוּרָה בְצוּרָה

יְיָ גָּדוֹל אַתָּה וְגָדוֹל שִׁמְךָ בִּגְבוּרָה


יְיָ דִּגֵּל חָמֵשׁ קְבוּעוֹת בְּמֶרֶץ

יְיָ דִּבֶּר וַיִּקְרָא אָרֶץ


יְיָ הִפְאִיר בְּעֻזּוֹ שֵׁשׁ פְּשׁוּטוֹת וַיְכַסֶּה

יְיָ הוּא הַטּוֹב בְּעֵינָיו יַעֲשֶׂה


יְיָ וִעֵד שִׁבְעָה נְגוֹהִים גְּבוֹהִים

יְיָ וּמֶלֶךְ גָּדוֹל עַל כָּל אֱלֹהִים


יְיָ זִקַּקְתָּ שְׁמוֹנָה זְקוּפֵי מָדוֹר

יְיָ זִכְרְךָ לְדוֹר וָדוֹר


יְיָ חָקַרְתָּ תִשְׁעָה עַד לֹא הַרְרֵי טֶרֶף

יְיָ חַסְדְּךָ לְעוֹלָם מַעֲשֵׂי יָדֶיךָ אַל תֶּרֶף


יְיָ טִכַּסְתָּ עֲשָׂרָה בְּאֵין סוֹף

יְיָ טוֹב וְסַלָּח וְרַב חֶסֶד לְכָל קוֹרְאֶיךָ


יְיָ יָחִיד בְּאֶחָד הַכֹּל תּוֹמֵךְ וְסוֹעֵד

יְיָ יִמְלֹךְ לְעוֹלָם וָעֶד


יְיָ כְּרוּבֵי כוֹכְבֵי זְרוּתִים יַמְלִיכוּךָ

יְיָ כָּל מַעֲשֶׂיךָ וַחֲסִידֶיךָ יְבָרְכוּךָ


יְיָ לָבַשְׁתָּ הוֹד וַתְּפַרְשֵׁז שַׁפְרִיר מָדוֹר

יְיָ לְעוֹלָם תֵּשֵׁב כִּסְאֲךָ לְדוֹר וָדוֹר


יְיָ מָרוֹם בְּאַרְבַּע צוּרוֹת כִּי גֵאוּת חוֹבֵשׁ

יְיָ מָלָךְ גֵּאוּת לֵָבֵשׁ


יְיָ נִשָּׂא קֶרֶן עֲמוּסָיו לְהָרִים נֵס

יְיָ נִדְחֵי יִשְׂרָאֵל יְכַנֵּס


יְיָ שִׂגֵּב מְצוּדוֹת סְלָעִים לְמִקְדַּשׁ עֲדָתִי

יְיָ סַלְעִי וּמְצוּדָתִי


יְיָ עִטֵּר צְפִירַת אוֹמֵן מְחוֹקֵק בְּתֶרֶץ

יְיָ עוֹשֵׂה שָׁמַיִם וָאָרֶץ


יְיָ פֵּאֵר אֵשְׁדָּת מְתוּקָה מִנֹּפֶת צוּפִים

יְיָ פּוֹקֵחַ עִוְרִים יְיָ זוֹקֵף כְּפוּפִים


יְיָ צְדָקָה הֵכִין לְחוֹסָיו לְהָאִיר עֵינֵימוֹ

יְיָ צְדָקוֹת אָהֵב יָשָׁר יֶחֱזוּ פָנֵימוֹ


יְיָ קִדַּשׁ לְקוֹוָיו מְנוּחָה הָנְחָלְתִּי

יְיָ קִוְתָה נַפְשִׁי וְלִדְבָרוֹ הוֹחַלְתִּי


יְיָ רוֹמֵם קַרְנִי לְעֵינֵי כָל בָּשָׂר

יְיָ רוֹעִי לֹא אֶחְסָר


יְיָ שְׁבָטָיו יִרְעֶה וְעַמִּים תַּחְתָּם יְרוֹדֵד

יְיָ שׁוֹמֵר אֶת גֵּרִים יָתוֹם וְאַלְמָנָה יְעוֹדֵד


יְיָ תְּחִנָּתִי יַקְשִׁיב וְעֵינַי יִפְקָח

יְיָ תְּפִלָּתִי יִקָּח


יְיָ שַׁדַּי לְחֶלְקוֹ מֵאָז הִבְדִּילָנוּ

יְיָ מָעוֹן אַתָּה הָיִיתָ לָנוּ.
