שׁוֹמַמְתִּי בְּרֹב יְגוֹנִי

שׁוֹמַמְתִּי בְּרֹב יְגוֹנִי / לְיוֹם יִפָּקֵד זְדוֹנִי / מָה אֹמַר לַאדוֹנִי

אֻמְלַלְתִּי וְנֶאֱלַמְתִּי / בְּזָכְרִי אֲשֶׁר אָשַׁמְתִּי / בֹּשְׁתִּי וְגַם נִכְלַמְתִּי

בְּהֶבֶל כָּלוּ יָמַי / מִפְּנֵי בֹשֶׁת עֲלוּמַי / אֵין שָׁלוֹם בַּעֲצָמַי

גַּחַלְתִּי קוֹדַחַת / כִּי מְגִלָּה נִפְתַּחַת / וְהַנּוֹשֶׁה בָּא לָקַחַת

דָּבַקְתִּי בְמַחְשַׁכִּי / וְנַפְשִׁי לֹא יָדְעָה כִּי / גֵר וְתוֹשָׁב אָנֹכִי

הוֹי כִּי יָבוֹא יוֹמִי / אֲזַי אִיקַץ מֵחֲלוֹמִי / וְאָשׁוּבָה אֶל מְקוֹמִי

וְעַל חֶטְאִי אֲשֶׁר עָבָר / וְעַל פִּשְׁעִי אֲשֶׁר גָּבָר / מָה אָשִׁיב שׁוֹלְחִי דָּבָר

זְדוֹן לִבִּי הִשִּׂיאַנִי / וְעָווֹן אֲשֶׁר הֶלְאָנִי / מִבֶּטֶן קְרָאָנִי

חוֹשֵׁב בְּנַפְשׁוֹ נָבָל / כְּעֵץ שָׁתוּל עַל יוּבַל / וְהוּא לִקְבָרוֹת יוּבָל

טָפַל שֶׁקֶר בְּתוֹכוֹ / וְנָטָה מִמַּהֲלָכוֹ / וַיִּפֶן כֹּה וָכֹה

יֻשְׁלַךְ כְּאֶבֶן דּוּמָה / וְלֹא יִשָּׂא לְבֵית מְהוּמָה / מִכָּל אֲשֶׁר לוֹ מְאוּמָה

כֹּחוֹ לֹא סְמָכַתְהוּ / עֵת נַפְשׁוֹ נְשָׂאַתְהוּ / אַף כִּי אֵשׁ אֲכָלָתְהוּ

לְכָדַתְנִי אַשְׁמָתִי / הֲלֹא זֶה דְבַר נִשְׁמָתִי / עַד הֱיוֹתִי עַל אַדְמָתִי

מְשׁוּגָתִי נָשָׂאתִי / וּבִלְבָבִי קָרָאתִי / אָנֹכִי חָטָאתִי

נְכוֹחָה לֹא חָשַׁקְתָּ / וּבִשְׁרִירוּת דָּבַקְתָּ / מַה לְּךָ כִּי נִזְעָקְתָּ

שְׂאִי נַפְשִׁי אַשְׁמָתֵךְ / רְאִי חֶטְאֵךְ לְעֻמָּתֵךְ / גַּם אַתְּ שְׂאִי כְלִמָּתֵךְ

עֵת אֶדְאַג מֵעֲווֹנַי / הֱשִׁיבוּנִי רַעְיוֹנַי / נִפְּלָה נָּא בְיַד אֲדֹנָי

פְּנֵה מִמְּכוֹן שִׁבְתֶּךָ / וּפְתַח לִי דְלָתֶיךָ / כִּי אֵין בִּלְתֶּךָ

צוּר הָגֵן בַּעֲדֵנִי / וּמֵעֲווֹנוֹתַי פְּדֵנִי / אֹרְחוֹתֶיךָ לַמְּדֵנִי

קוֹלִי שְׁמַע כְּחַסְדֶּךָ / בְּיוֹם אֶעֱמֹד נֶגְדֶּךָ / אַל תַּט בְּאַף עַבְדֶּךָ

רְאֵה עָנְיִי וַעֲנֵנִי / אֲנִי בְיָדְךָ הִנֵּנִי / וְאַתָּה יְיָ חָנֵּנִי

שְׁלַח אֲמִתְּךָ וְחַסְדֶּךָ / לְעַם צוֹעֲקִים נֶגְדֶּךָ / וְלִי אֲנִי עַבְדֶּךָ

תִּסְלַח לַאֲשָׁמֵינוּ / וְאַל תִּפְקֹד עֲלוּמֵינוּ / כִּי צֵל יָמֵינוּ.
