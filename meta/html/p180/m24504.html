שָׁאוֹל יִשָׁאֲלוּן

שָׁאוֹל יִשָׁאֲלוּן / לָז לְלָז יְמַלְּלוּן / מְהַלְּלִים אֶהְיֶה / וְאוֹמְרִים אַיֵּה


בְּכִסֵּא רָם וְנִשָּׂא / אֶדֶר אַרְבַּע חַיּוֹת

וּבָאָרֶץ כְּנֶגְדָּם / בְּחִירֵי דָר עֲלִיּוֹת

גְדֻלַּת אֵל מְעִידִים / וּלְפָנָיו עוֹמְדִים

וְכֵן רָאָה חוֹזֶה / דְּמוּת קָמִים לְהַלְלוֹ / שְׂרָפִים עוֹמְדִים מִמַּעַל לוֹ.


בְּכִסֵּא רָם וְנִשָּׂא / הוֹד כְּנָפַיִם אַרְבַּע

וּבָאָרֶץ כְּנֶגְדָּם / וְעוּדֵי חוּט מְרֻבָּע

זְרוּיִים בְּאַרְבַּע כְּנָפוֹת / כְּנֶגֶד אַרְבַּע זְקוּפוֹת

וְכֵן רָאָה חוֹזֶה / חֹסֶן אֵל מְיֻחָד / שֵׁשׁ כְּנָפַיִם שֵׁשׁ כְּנָפַיִם לְאֶחָד.


בְּכִסֵּא רָם וְנִשָּׂא / טֶכֶס אָדָם נִקְדָּם

וּבָאָרֶץ כְּנֶגְדָּם / יְעוּדִים צֹאן אָדָם

כְּזָכְרָם עֲווֹנֵיהֶם / יְכַסּוּ אֶת פְּנֵיהֶם

וְכֵן רָאָה חוֹזֶה / לִפְנֵי דָר מְעוֹנָיו / בִּשְׁתַּיִם יְכַסֶּה פָנָיו.


בְּכִסֵּא רָם וְנִשָּׂא / מְכוֹנַת אַרְיֵה יְעוּדָה

וּבָאָרֶץ כְּנֶגְדָּם / נִצָּנֵי יְהוּדָה

סוֹלְדִים בְּלֵב מְאֻשָּׁר / וּבְרֶגֶל מְיֻשָּׁר

וְכֵן רָאָה חוֹזֶה / עֹז דִּגְלֵי זְבוּלָיו / וּבִשְׁתַּיִם יְכַסֶּה רַגְלָיו.


בְּכִסֵּא רָם וְנִשָּׂא / פְּאֵר דְּמוּת שׁוֹר בְּמִישׁוֹר

וּבָאָרֶץ כְּנֶגְדָּם / צֶאֱצָאֵי בְכוֹר שׁוֹר

קְהָלָם יְאַסְּפוּן / וְכַיּוֹנִים יְעוּפוּן

וְכֵן רָאָה חוֹזֶה / רוּם כָּבוֹד מְחוֹפֵף / וּבִשְׁתַּיִם יְעוֹפֵף.


בְּכִסֵּא רָם וְנִשָּׂא / שְׁבִיב נֶשֶׁר נִכְתָּם

וּבָאָרֶץ כְּנֶגְדָּם / תְּמִימִים כַּנֶּשֶׁר בְּלֶכְתָּם

שִׁיר לְיוֹצֵר וּבוֹרֵא / זֶה אֶל זֶה קוֹרֵא

וְכֵן רָאָה חוֹזֶה / לָעוֹמְדִים עַל מִִשְׁמָר / וְקָרָא זֶה אֶל זֶה וְאָמַר.


בְּכִסֵּא רָם וְנִשָּׂא / מַרְאֵה גַחֲלֵי אֵשׁ

וּבָאָרֶץ כְּנֶגְדָּם / הַהוֹלְכִים בְּמוֹ אֵשׁ

בְּמוֹנִים שְׁלֹשָׁה / יְשַׁלְּשׁוּ לְךָ קְדֻשָּׁה

וְכֵן רָאָה חוֹזֶה / רַחַשׁ שַׂרְפֵי פְלָאוֹת / קָדוֹשׁ קָדוֹשׁ קָדוֹשׁ יְיָ צְבָאוֹת.


בְּכִסֵּא רָם וְנִשָּׂא / יָדוֹת רָצִים וְשָׁבִים

וּבָאָרֶץ כְּנֶגְדָּם / הָמוֹן רָצִים בְּנִיבִים

וְשָׁבִים לְצוּר עֶרֶץ / מְלֹא חוּג וָאָרֶץ

וְכֵן רָאָה חוֹזֶה / דִּבְרֵי הֲמוֹן גְּדוּדוֹ / מְלֹא כָל הָאָרֶץ כְּבוֹדוֹ.
