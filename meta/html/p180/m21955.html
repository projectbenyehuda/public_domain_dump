תְּהִלַּת אֵל

למנויו של אחד מתלמידי החכמים


תְּהִלַּת אֵל בְּרֹאשׁ כָּל הַתְּהִלּוֹת / וְצִדְקוֹתָיו לְתַנּוֹת, הַגְּדוֹלוֹת

אֲשֶׁר נַעֲלָה עֲלֵי עֶרֶץ וְנִשְׂגָּב / וְשָׁת בֵּינוֹ וּבֵינֵינוּ גְבוּלוֹת

יְמִינוֹ רוֹמְמָה בִמְאֹד פְּלָאָיו / אֲשֶׁר גָּדְלוּ עֲלֵי כָל הַגְּדֻלּוֹת

אֲשֶׁר תִּתְמֹךְ אֲדָמָה עַל בְּלִימָה / וְתַחְרִיב בֶּעֱזוּזָהּ הַמְּצוּלוֹת

אֲשֶׁר תָּעוּר וְתִפְדֶּה הַיְּשֵׁנִים / בְּגָלוּתָם, וְתָחִישׁ הַגְּאֻלּוֹת

לְשֶׂה גוֹלָה, וּמִבָּנִים שְׁכוּלָה / וְגַלְמוּדָה, שְׁכוּלָה מִשְּׁכוּלוֹת

נְדוּדָה מֵהֲנָחָה לַאֲנָחָה / אֲכוּלָה גַם בְּנוֹתֶיהָ אֲכוּלוֹת

וְהַיּוֹם לָהּ כְּמוֹ אֶלֶף בְּגָלוּת / וְכִמְעַט קָט וְאַלְפַּיִם כְּפוּלוֹת

בְּכָל פִּיהֶם אֲכָלוּהָ יְרִיבִים / וְעוֹד יִדְמוּ שְׁאֵרִיתָהּ נְטוּלוֹת

בְּעֹצֶם עִצְּמוּהָ הָעֲצוּמִים / עֲלֵי חֵילָהּ אֲשֶׁר הָיָה נְחָלוֹת

וְהִיא מִיָּד לְיָד נִדְחָה וְשָׁחָה / בְּשׁוּחוֹת וַחֲלַקְלַקּוֹת אֲפֵלוֹת

בְּחַבְלֵי הָעֳנִי תֵשֵׁב אֲחוּזָה / עֲנִיָּה פַּעֲמוֹתֶיהָ כְבוּלוֹת

מְפָרֵק עֹל! פְּרֹק עֹל צַוְּארוֹנָהּ / וְהַמּוֹטוֹת תְּמוֹטֵט בַּחֲיָלוֹת

וּמַשְׂכֻּרְתָּהּ לְגָלוּתָהּ תְּשַׁלֵּם / וְתָשֵׁב לָהּ גְּמוּל, מֵשִׁיב גְּמוּלוֹת!

וְתִבְנֶה מֵהֲרוּסֶיהָ בְּצוּרוֹת / וְתָקִים מִנְּתוּצֶיהָ תְּלוּלוֹת

וְתִבְעַל מִבְּנוֹתֶיהָ עֲזוּבוֹת / אֲשֶׁר הָיוּ לְצוֹדֵיהֶן בְּעוּלוֹת

וְגָלוּת חֵיל מְדִינָה הָעֲדִינָה / תְּחַדֶּהָ בְּמַכְלוּלוֹת כְּלוּלוֹת

אֲשֶׁר גָּלוּ אֱלֵי אֶרֶץ סְפָרַד / וְהִשִּׂיגוּ, וְהִשִּׂיגוּם קְלָלוֹת

וּמַשְׂאוֹת נָשְׂאוּ כִּי אַתְּ גְּזַרְתָּהּ / וְהֵכִילוּ עֲדֵי אֵין בָּם יְכוֹלוֹת

רְעֵבִים, מִבְּלִי מַיִם צְמֵאִים, / כְּאַיָּלוֹת חֲשׂוּכוֹת מֵאֱיָלוֹת

וְלוּלֵי נָחֲלוּ נַחַל אֱמוּנָה / שְׁאֵרִית יַעֲקֹב עָשׂוּ פְלִילוֹת

אֲזַי בָם נֶעֱקַר עִקַּר תְּבוּנָה / וְיָבְשׁוּ הָאֲשָׁלִים הַשְּׁתוּלוֹת

וְהָיוּ הָאֲסוּרוֹת כַּפְּטוּרוֹת / וְהָיוּ הַכְּשֵׁרוֹת כַּפְּסוּלוֹת

אֲשֶׁר הֶעְמִיד יְסוֹדוֹת מִשְׁפְּטֵי דָת / וְהֵשִׂים מַפְּלוֹתֶיהָ בְּטֵלוֹת

וְלָנוּ עַם יְשׁוּרוּן בַּחֲלָקוֹת / וְהִשְׁכִּימוּ בְאוֹרֵי הָאֲפֵלוֹת

כְּדַיַּן דִּין אֱמֶת, הוֹלֵךְ נְכוֹחוֹ / וְהַצֶּדֶק לְבַקֵּשׁ גַּם פְּעֻלּוֹת

אֲשֶׁר יַפְסִיק לְדִין רָאוּי לְהַפְסִיק / בְּצִדְקָתוֹ בְּלִי אָוֶן וְעוֹלוֹת

וּמַמְתִּין בּוֹ, וְאִם זָקוּק לְנַחֵץ, / כְּאַזְהָרַת יְשִׁישֵׁי הָעֲגֻלּוֹת

וְדִינוֹ מִבְּלִי טָעוּת מְתֻקָּן / כְּאֹרַח שֶׁאֲבָנֶיהָ סְקוּלוֹת

וְהַקַּלּוֹת הֱשִׁיתָן כַּחֲמוּרוֹת / וְרָאָה כַּכְּבֵדוֹת הַנְּקַלּוֹת

וְאִם יִשָּׁפְטוּ נִצִּים לְפָנָיו / וּמִתְעוֹלֵל אֲזָרָם בַּעֲלִילוֹת

יְזַכֶּה זֶה, וְאִם אֶת זֶה יְחַיֵּב / וְלֹא יִשָּׂא פְּנֵי פָנִים צְהוּלוֹת

וְכָל מִשְׁפָּט לְאוֹר יוֹצִיא בְצִדְקוֹ / וּמִשְׁפָּטָיו וְאֹרְחוֹתָיו סְלוּלוֹת

וּבִמְקוֹם הַשְּׁבוּעָה שָׁם פְּשָׁרָה / וּבָהּ גָּזַר מְקוֹם גָּזְרוּ בְאָלוֹת

וְיַמְצִיא תַעֲלוּמֵי הָאֲבֵדוֹת / וּמֵהָעוֹשְׁקִים יָשִׁיב גְּזֵלוֹת

וְשָׂם מַלְוֶה לְאִישׁ עָנִי לְהָשִׁיב, / בְּעוֹד שֶׁמֶשׁ, בְּגָדָיו הַחֲבוּלוֹת

וְהִלְבִּישׁ מַעֲרֻמֵּי הַתְּבוּנָה / וְהָיוּ הַיְּחֵפוֹת בּוֹ נְעוּלוֹת

וּפָתַח לַחֲכָמִים הָאֲטוּמוֹת / וּמִי יִפְתַּח דְּלָתוֹת שָׂם נְעוּלוֹת?

וְהוּא נִבְחָר, בְּעֵת אֵין שָׁם זְבָחִים / לָאֵל, מִשַּׁי וְקָרְבָּנוֹת וְעוֹלוֹת

וְהוּא הִזְהִיר עֲלֵי צֶדֶק וְצִוָּה / - הֲיִרְצֶה בַעֲגָלִים אוֹ עֲגָלוֹת -

אֱמֶת כִּי בוֹ עֲווֹנֵינוּ מְחוּיִים / וְחַטֹּאת דּוֹרְשֵׁי חֶפְצוֹ מְחוּלוֹת

וְכָל חָכְמוֹת לְחָכְמָתוֹ עֲזוּבוֹת / וְגַם מִבַּעֲלֵיהֶן הֵן שְׁלוּלוֹת

וְנִקְרָאוֹת תְּשׁוּבוֹתָיו בְּבָבֶל / וְנִפְרָשׁוֹת בְּרָאשֵׁי הַקְּהִלּוֹת

בְּסוֹד רָאשֵׁי נְהַרְדֵּעָא וְסוּרָא / עֲיָרוֹת הַפְּלִילוֹת הַגְּדוֹלוֹת

הֱרִימוּהֶן עֲלֵי כָל הַתְּשׁוּבוֹת / וּבַזָּהָב כְּתָבוּם בַּמְּגִלּוֹת

וְאוֹמְרִים: מָצְאוּ אַנְשֵׁי סְפָרַד / פְּלִיאֵי הַתְּבוּנוֹת הַמְּעֻלּוֹת

וְחָזוּ הָאֱמֶת גָּבַר בְּרַבָּם / וְחָזִינוּ אֲנַחְנוּ מַהֲתַלּוֹת

וְלָבַשְׁנוּ לְעֻמָּתָם קְרָעִים / וְהֵמָּה לָבְשׁוּ חֶמְדַּת שְׂמָלוֹת

וְהָיְתָה גַאֲוָתָם כַּאֲרָיוֹת / וְהָיִינוּ לְעֻמָּתָם רְחֵלוֹת

וְשָׂרִיגֵי תְבוּנָה הֵם קְטוּפִים / וְאֵין עוֹלֵל, וְקָטַפְנוּ מְלִילוֹת

וְעֵינֵי עִוְרֵיהֶם נִפְקְחוּ בָם / וְאָזְנֵיהֶם, וְאָזְנֵינוּ צְלוּלוֹת

בְּרַב הִרְבָּה שְׂשׂוֹנֵיהֶם וְחִדָּם / וְחֶדְוָתָם לְהִגָּלוֹת תְּהִלּוֹת

וְרַב הַאיֵי כְּלֹא הָיָה לְפָנָיו / וְקָטֹן מֵעֲנוֹת אוֹתוֹ בְּמִלּוֹת

מְזִמּוֹתָיו יְפִיצוּן כָּל מְזִמּוֹת, / וְתַחְבּוּלוֹת לְכָל אָדָם - חֲבוּלוֹת

וְחָכְמַת שׁוֹטְרֵי חָכְמוֹת גְּדוֹלוֹת / אֲשֶׁר הוֹרָה עֲלֵי דֶרֶךְ גְּמוּלוֹת

פְּנֵי רַב לִמְּדָנוּ הַכְּעוּרוֹת

וּמִלֵּא הַנְּפָשׁוֹת הַיְּבֵשׁוֹת / עֲדֵי הָיוּ בְּמֵי הַבִּין טְבוּלוֹת

בְּמַיִם לֹא אֱנוֹשׁ יִשְחֶה בְדָכְיָם / וְאִם תֵּרֵד לְסוֹף תִּמְצָא מְצוּלוֹת

וְחָכְמָתוֹ רְחָבָה מֵאֲדָמָה / וְאִם יֵשׁ לָהּ גְּבוּל - אֵין לָהּ גְּבוּלוֹת

יְפַצֵּל מִפְּלִיאֶיהָ פְּצָלוֹת / כְּאִישׁ פִּצֵּל לְיַחֵם בָּם בְּמַקְלוֹת

וְתֵלַדְנָה תְבוּנוֹתָיו בְּלִי צִיר / וְלֹא קָשׁוֹת וְלֹא רוֹגְזוֹת וְחָלוֹת

וְכָל עֵת הֵן מְחֻדָּשׁוֹת וְעָלוֹת / וְאִם הֵנָּה מְיֻשָּׁנוֹת וּבָלוֹת

אֲשֶׁר הֵנָּה לְכָל צָמֵא וְעָיֵף / מְזֵי רָעָב כְּצִמּוּקֵי דְבֵלוֹת

וְדֵעוֹת הַגְּלוּיוֹת - לַחֲכָמִים, / וְהוּא יָדַע לְנִסְתָּרוֹת וְנִגְלוֹת

וְהַכֹּל פּוֹצְחִים רִנּוֹת לְשׁוֹנוֹת / וּבִלְבָבָם לְרַבָּם רַב מְסִלּוֹת

אֲשֶׁר הִזְהִיר לְהַדְלִיק נֵר נְשֵׁיהֶם / וְהַפְרָשַׁת תְּרוּמִיּוֹת וְחַלּוֹת

וּמִצְווֹתָיו וְחֻקּוֹתָיו וְדָתָיו / מְיֻשָּׁרוֹת נְכוֹחוֹת כַּשְּׁפֵלוֹת

וְתוֹכְחוֹתָיו לְכָל רָאשִׁים עֲטָרוֹת / וְכִצְפִירוֹת עֲלֵי קָדְקֹד סְבוּלוֹת

אֲשֶׁר הֵנָּה בְּעֵינֵינוּ בְתוּלוֹת / וְכָמוֹנוּ וְעִמָּנוּ שְׁגוּלוֹת

וְהֵן רִפְאוּת שְׁרִירֵי נוֹסְסֵיהֶן / וְלִמְזוֹרוֹת מְזוֹרוֹת גַּם תְּעָלוֹת

וְהֵנָּה לַמְּגַשְּׁשִׁים כַּמְּאוֹרוֹת / וְכַנֵּרוֹת דְּלוּקוֹת בָּאֲפֵלוֹת

וּמִימֵי אַב אֲבִי מַלְאַךְ עֲתָרוֹת / וְתַלְמִיד רַב, אֲשֶׁר הִנְחִיל נְחָלוֹת -

אֱמֶת לֹא קָם בְּיִשְׂרָאֵל כְּמוֹתוֹ / אֲבָל כִּי אִישׁ בְּגִבְעַת הָעֲרָלוֹת

אֲשֶׁר מָל הַלְּבָבוֹת הָעֲרֵלוֹת / וְהִשְכִּיל הַנְּפָשׁוֹת הַסְּכָלוֹת

וְקָרַע כָּל סְגוֹרֵי לֵב אֱוִילִים / וְנַפְשׁוֹתָם מְפֻלָּשׁוֹת חֲלוּלוֹת

וְכָל סָתוּם וְכָל נֶעְלָם וְעָמוּם / מְגֻלּוֹת הֵן לְקֻשִּׁיּוֹת עֲפֵלוֹת

וְהוּא אֶחָד וְאֵין שֵׁנִי לְעָרְכוֹ / וְנַפְשׁוֹ בָם כְּמֶלֶךְ בַּחֲיָלוֹת

עֲלוּלָה הַיְּקָרָה מִתְּבוּנָה / אֲצוּלָה מֵאֲצִילֵי הָאֲצוּלוֹת

מְהִירָה לַעֲשׂוֹת תַּחְבּוּל וְלוֹ הַ-/חֲרוּצָה נֶחְשְׁבָה מֵהָעֲצֵלוֹת

לָכֵן יֵקַל וְיִקְטַן מִכְּבוֹדוֹ / אֱגֹר שֶׁבַח וְחַבֵּר מַהֲלָלוֹת

וְקָצִיתִי קְצָת הוֹדוֹ, וְחָסֵר / דְּלִי מִיאוֹר וְנֵטֶף מִתְּעָלוֹת

וּבִרְאוֹתִי יְקָר הַשִּׁיר וְיִתְרוֹ / הֱכִינוֹתִיו לְנָוֵהוּ מְסִלּוֹת

וְהַעְדִּיתִיו בְּעֶדִי הַבְּדֹלַח / וְהַעְנַדְתִּיו עֲטָרוֹת מִתְּהִלּוֹת -

וְאִמְרוֹתַי אֲמָרוֹת הַטְּהוֹרוֹת / בְּמֹאזְנֵי שִׁיר וּמִשְׁקָלוֹת שְׁקוּלוֹת

וּמָצָאתִי בְּצוּרוֹת כָּעֲזוּבוֹת / וְחָשַׁבְתִּי בְּמַחְשָׁבוֹת כְּלוּלוֹת

וְהָיו יוֹדְעֵי שָׂפָה בְרוּרָה / וְכָל אַמִּיץ לְדַבֵּר צַח בְּמִלּוֹת

סְחַרְחָרִים וְלֹא מָצְאוּ יְדֵיהֶם / וְהֵנָּה לֶעֱזוּז יָדַי אֲזוּלוֹת

וּמָשַׁלְתִּי לְשׁוֹן עִבְרִית, וְעַזֵּי / כְבֵדֶיהָ לְעֻמָּתִי נְקַלּוֹת

וְעָשִׂיתִי לְפוּצֶיהַָ מְסִלּוֹת, / כְּמוֹ כֵן לַעֲרוּצֶיהָ תְּעָלוֹת

וְלוּ עוֹכֵר עֲכָרַנִי עֲכַרְתִּיו / כְּחִדֶּקֶל וְכִפְרָת בַּהֲמֻלּוֹת

וְרַגְלַי בָּהּ כְּרַגְלֵי הָרְאֵמִים / וְרַגְלָיו בָּהּ כְּרַגְלֵי הַנְּמָלוֹת

וְעִנְנוּהוּ וְכִסּוּהוּ אֲבָקִים / וְלֹא יָכֹל לְצָעְדוֹ בַנְּעָלוֹת

וְהָיוּ לִי לְבִיאוֹתָיו כְּשֵׂיוֹת / וְגַם נָפְלוּ לְאוֹתוֹתַי קְטוּלוֹת.

וְאַל תִּתְמַהּ בְּשִׁירוֹת הַיְּקָרוֹת - / תְּמַהּ מֵהַשְּׁמוֹנֶה הַכְּפוּלוֹת.
