אָבֹא שַׁעְרֵי

אָבֹא שַׁעְרֵי יְרוּשְׁלֵם הַבְּצוּרָה

אֲשֶׁר חָרְבָה וּמִטּוֹבָה בְצוּרָה

וְאִם תָּשׁוּב וְהִיא תִבְטַח בְּצוּרָהּ

אֲזַי צוּרָהּ יְדַבֶּר נָא שָׁלוֹם בָּהּ:


אָבוֹא שַׁעְרֵי תְּשׁוּבָתָךְ, וְאָשִׁית

לְנֶגְדִּי יוֹצְרִי תָמִיד, וְאָשִׁית

בְּיִצְרִי, לֹא אֲנִי אֶחְשֶׁה וְאָשִׁית

בְּכָל יָמַי עֲדֵי זִקְנָה וְשֵׂיבָה:


אָבוֹא שַׁעְרֵי בְּצוֹק עִתִּים וְחָרוּץ

אֲשֶׁר הַיּוֹם בְּכִלָּיוֹן וְחָרוּץ

וְחִישׁ יִהְיוּ מְמֻלָּאִים בְּחָרוּץ

וּמַרְגָּלִית, זָהָב שָׁחוּט וּשְׁבָא.


אָבוֹא שַׁעְרֵי מְקוֹם צִיּוֹן, וְאָגוּר

מְקוֹם מַלְכֵי בְנֵי דָוִד וְאָגוּר

וְאַךְ אוֹיְבִי וְלֹא אֶפְחַד וְאָגוּר

בְּכָל מָקוֹם אֲשֶׁר אֵלְכָה בָהּ:


אָבוֹא שַׁעְרֵי יְרוּשְׁלֵם הָאֲרֻכָּה

בְּנֵה אוֹתָהּ וְהַעְלֵה לָהּ אֲרֻכָה

וְכָל בָּנַי בְּךָ יִתְפָּאֲרוּ כַ-

אֲשֶׁר יִרְאוּ שְׁכִינָתְךָ בְּקִרְבָּהּ:


אָבוֹא שַׁעְרֵי יְרוּשְׁלֵם הַבְּנוּיָה

אֲזַי אֶשְׂמַח וְאָגִילָה בְנוֹיָה

וְכָל כֶּתֶם אֲשֶׁר כָּמוּס בְּנוֹא יַ-

אֲצִיל לַבַּת אֲהוּבָה וַחֲבִיבָה:


אָבוֹא שַׁעְרֵי כְנֵסִיּוֹת בְּרִנָּה

וְחִישׁ כַּבְלֵי חֲטָאַי שַׁבְּרָה נָא

וְלִירוּשְׁלֵם מְהֵרָה דַבְּרָה נָּא

לְמַעַנְךָ וְחוּשָׁה, אֵל, שָׁלוֹם בָּהּ:


אָבוֹא שַׁעְרֵי צְפַת חֶבְרוֹן וְשָׁלֵם

בְּמָמוֹן רַב וְגוּף בָּרִיא וְשָׁלֵם

וְתִבְנֶה אֶת מְקוֹם קָדְשָׁךְ וְשַׁלֵּם

לְבַת אֱדֹם גְּמוּלָהּ לָהּ וְרִיב בָּהּ:


אָבוֹא שַׁעְרֵי תְפִלָּתָךְ מְכֻתָּר

וְלַיְשׁוּעָה אֱהִי תָמִיד מְכֻתָּר

רְאֵה אוֹיְבַי לְעַבְדָּךְ הִמְּכוּ, תַּרְ-

אֵינִי נִפְלָם וְשָׁכְבָם בַּעֲצִיבָה:


אָבוֹא שַׁעְרֵי יְרוּשְׁלֵם וּבָרְחוֹב

וְאֶעְמֹד עִם בְּנֵי צִיּוֹן וּבַר חוֹב

וְכָל יָשָׁר וְזַךְ לֵבָב וּבַר חוֹב

עָלָיו יַעְלֶה בְּחַגִּים הַכְּתוּבָה:


אָבוֹא שַׁעְרֵי דְלָתֶיךָ וּפִשְׁפֵּשׁ

מְקוֹם לֹא נִמְצְאוּ תוֹלָע וּפִשְׁפֵּשׁ

וְאַשְׁרֵי אִישׁ אֲשֶׁר יָשׁוּב וּפִשְׁפֵּשׁ

דְּרָכָיו אָז יְשׁוּעָתוֹ קְרוֹבָה:
