וִכּוּחַ הָאָדָם

הוכחת השלושה עשר


וִכּוּחַ הָאָדָם עִם אַדְמָתוֹ / וַיִּזְכֹּר לָהּ יְמֵי קַדְמָתוֹ

וַיִּצְעַק עָלֶיהָ בְּהֶגְיוֹן אִמְרָתוֹ / וּמֵרֹב אַנְחָתוֹ בִּצַּע אִמְרָתוֹ

מוּסָר לָאָדָם הַנִּרְדָּם אֲשֶׁר לַהֶבֶל דָּמָה / יַעֲרֹךְ אֶת תּוֹכְחוֹתָיו בְּנַחַת עִם הָאֲדָמָה

וְיוֹצִיא אֶת כָּל מַצְפּוּנָיו בְּקוֹל דְּמָמָה / כִּי מָחָר יִשְׁכַּב בַּעֲפָרָהּ כְּאֶבֶן דּוּמָה

לָכֵן קֹדֶם אִמְרֵי הַוִּכּוּחַ / בְּרֹב אוֹנִים וְאַמִּיץ כֹּחַ

מִנֶּגֶד גַּאֲוָתוֹ הַנֶּעְדֶרֶת יִבְרוֹחַ / וְלֹא יְדַבֵּר בְּגַסּוּת הָרוּחַ

יַכְנִיעַ גְּאוֹנוֹ וְיִדְכֶּה יָשֹׁחַ / וּשְׁאוֹן אִמְרוֹתָיו בְּעַנְוָה יָשׂוּחַ

יְבַקֵּשׁ לְנַפְשׁוֹ הַיְקָרָה מָנוֹחַ / וְיַכִּיר מְקוֹמוֹ אָנָה יָנוּחַ

אִם לְבָבוֹ כְסִיר נָפוּחַ / יְהַלֵּךְ לְאַט עַד שֶׁיָּפוּחַ.


בֶּן אָדָם אֵיךְ תְּהַלֵּךְ בְּיָד רָמָה / וְלֹא תִּכָּלֵם מִלִּפְנֵי שׁוֹכֵן רוּמָה

מָחָר אֶת אִישׁוֹן עֵינֶיךָ תֹאכַל הָרִמָּה / וְיִצְרָךְ הָרָע יִרְדֹּף אַחֲרֶיךָ עַד חָרְמָה

וְאַחֲרָיו תָּבוֹא אֵלֶיךָ הָאָרֶץ הַנְּשַׁמָּה / וּבָעֲפָרָהּ תַּדְאִיב מִגּוּפָךְ אֶת הַנְּשָׁמָה

תַּשְׁפִּיל מֵעָלֶיךָ אֶת גֹּבַהּ הַקּוֹמָה / וְלֹא תַשְׁאִיר לְךָ שְׁאֵרִית וּתְקוּמָה

הִיא וּשְׁתֵּי כַלּוֹתֶיהָ עִמָּהּ / וְתַשְׁחִיר פָּנֶיךָ לְנֶגֶד עַמָּהּ

אָכֵן הָסֵר מֵרֹאשָׁךְ אֶת מִצְנֶפֶת הָאֲדֻמָּה / וְאַל תִּתְוַכַּח בְּעַזּוּת פָּנֶיךָ עִם הָאֲדָמָה

אֲשֶׁר מִמֶּנָּה לֻקָּחְתָּ / וְלִבְגָדֶיךָ בִבְשָׂמֶיהָ הִרְקַחְתָּ

אֶת כַּפֶּיךָ עָלֶיהָ שִׁטָּחְתָּ / וְקוֹרוֹת טִירוֹתֶיךָ בַּעֲפָרָהּ הִטָּחְתָּ

בִּרְחוֹבוֹתֶיהָ יָמִין וּשְׂמֹאל פָּרָצְתָּ / וְאֶת בִּנְיַן גְּדֵרֶיהָ פָּרָצְתָּ

מֵחֹמֶר עֲפָרָהּ קֹרָצְתָּ / וְעֵינֶיךָ עָלֶיהָ קֹרָצְתָּ

תַּחַת צֵל אִילָנוֹתֶיהָ רָבָצְתָּ / וּלְחַצְרוֹתֶיךָ בְּמֵי נְהָרוֹתֶיהָ רִבָּצְתָּ

אֶת יְעָרוֹתֶיהָ בְּקַרְדֻּמֶּיךָ קִצָּצְתָּ / וְאֶת אֲבָנֶיהָ בְמַקֵּבוֹתֶיךָ רִצָּצְתָּ

עַל בָּמוֹתֶיהָ נַעֲלֶיךָ חָלָצְתָּ / וְאֶת אַבְנֵי יְסוֹדוֹתֶיהָ חִלָּצְתָּ

בְּפַעֲמֵי רַגְלֶיךָ עָלֶיהָ אַצְתָּ / וְאֶת כָּל פֵּירוֹתֶיהָ נִאָצְתָּ

פִּיךָ לְרָע לְנֶגְדָּה פָּצְתָּ / וְכָל צֶמֶר גַּפְנָהּ נִפָּצְתָּ

עַל גִּבְעוֹתֶיהָ כִצְבִי קִפָּצְתָּ / וְלִשְׁכֹּן בְּתוֹךְ אַרְמְנוֹתֶיהָ חָפָצְתָּ

וְכִרְאוֹתְךָ חֶסְרֹנְךָ כְּכֶלֶב חָרָצְתָּ / כִּי מִשְׁפָּטֶיךָ אַתָּה חָרָצְתָּ

בְּצֵל חוֹמוֹתֶיהָ הִסְתּוֹפָפְתָּ / וְלִתְבוּאָתָהּ נִכְסֹף נִכְסָפְתָּה

לֶאֱכֹל מִפִּרְיָהּ וְלִשְׂבֹּעַ מִטּוּבָהּ / וּלְהִתְעַלֵּס בְּתוֹךְ עֲלִיּוֹתֶיהָ הַמְּחֻטָּבָה

מִצֶּמֶר גַּפְנָהּ תַּעֲשֶׂה לְגוּפָךְ תִּלְבֹּשֶׁת / וּמֵעֲשָבֶיהָ תִקַּח לָחֳלִי מַכּוֹתָךְ תַּחְבֹּשֶׁת

כָּל יְמֵי חַיֶּיךָ עַל בָּמוֹתֶיהָ תִּכְבֹּשׁ / וְהִיא תוֹצִיא לְךָ לֶחֶם לֶאֱכֹל וּבֶגֶד לִלְבֹּשׁ

תִּתְעוֹרֵר מִשֵּׁנָתָךְ בִּתְחִלַּת הָאַשְׁמוּרָה / לַחֲרֹשׁ שְׂדוֹתֶיהָ וּלְעָבְדָהּ וּלְשָׁמְרָהּ

לֶאֱכֹל מִדְּגַן תִּירוֹשָׁהּ וְעִבּוּרָהּ / וּלְמַעַן יִיטַב לְךָ בַעֲבוּרָהּ

תָּרוּץ כִּצְבִי בְקֶרֶב חֲלָלָהּ / וְתִחְיֶה נַפְשָׁךְ הַיְקָרָה בִגְלָלָהּ

בְּפֵרוֹתֶיהָ הַנְּעִימִים תְּצַהֲלֶךָ / וּבְחוֹמוֹתֶיהָ מֵהָאוֹיְבִים תַּצִּילֶךָ

כָּל יָמֶיךָ תִדְרֹךְ עָלֶיהָ בְמַנְעָלֶיךָ / וְהִיא בְחַיֶּיךָ וּבְמוֹתָךְ תִּשְׁמֹר עָלֶיךָ

לַיְלָה וְיוֹמָם תַּרְשִׁיעַ בְּתוֹכָה בְּכָל מַפְנֶיךָ / וְאֵיךְ בָּאתָ לְהוֹכִיחַ אוֹתָהּ בְּעַזּוּת פָּנֶיךָ

מִגַּנּוֹתֶיהָ תֹאכַל שׁוּמֶיךָ וְלוּפְנֶיךָ / וְכַשִּׁפְחָה תָשִׂים לַחְמָהּ לְפָנֶיךָ.


וַיְהִי כִשְׁמֹעַ הָאָדָם אֶת דִּבְרֵי הַצִּירָה הַזֹּאת / עָמַד בְּעֵינָיו קֶרֶן חָזוּת / לְבָבוֹ נִמְלָא חֵמָה וְרַגְזוּת / וַיֵּשֶׁב לוֹ בְּעָרֵי הַפְּרָזוֹת / וַיָּקָם בַּחֲמָתוֹ וַיִּכְתֹּב אֶת הַמְּגִילָה / וְאֶת כָּל מַצְפּוּנָיו אֵלֶיהָ גִלָּה / וַיָּשֶׂם שַׂק בְּמָתְנָיו וַיֵּלֶךְ בַּגּוֹלָה / וַיִּצְעַק בַּחֲרוֹנוֹ וַיַּעַן וַיֹּאמֶר לָהּ:


עֲנִיָּה סוֹעֲרָה לָמָּה אַתְּ מִתְנַכְּרָה / וְקוֹלֵךְ מַר כְּקוֹל יוֹלֵדָה בְמַבְכִּירָה

לְבָבֵךְ צַר כְּלֵב אִשָּׁה מְצֵרָה / וְנַפְשִׁי הַיְקָרָה מֵחֲרוֹנֵךְ תָּחִיל בְּצִירָהּ

אֵיךְ נֶהְפַּכְתָּ עָלַי כְּאֶרֶץ גְּזֵרָה / וְאֶת אֵיבָרַי תָּשִׂימִי בְּחַרְבוֹתַיִךְ גְּזוּרָה

נִלְאֵיתִי מִקַּחַת לִיךְ תְּשׁוּרָה / וְאַתְּ תַּשְׁקִינִי מַיִם הַמָּרִים בִּמְשׂוּרָה

בִּסְעִיפֵךְ תִּתְכַּסִּי וְתִתְעַלְּפִי / וְכַיּוֹלֵדָה בַחֲבָלַיךְ תִּתְעַלְּפִי

טִירָתֵךְ נִשְׂגָּבָה עוֹמֶדֶת לְתַלְפִּיּוֹת / וַאֲבָנַיִךְ עַל רֹאשִׁי חוֹפִיּוֹת

לְשׁוֹנֵךְ חַד כְּחֶרֶב פִּיפִיּוֹת / וַעֲפָרֵךְ יְבַלֶּה כָל הַפֵּיוֹת

שְׁאוֹנֵךְ יַעֲלֶה כִשְׁאוֹן הַיַּמִּים / וְעֶבְרָתֵךְ צְפוּנָה לְאַחֲרִית הַיָּמִים

קוֹלֵךְ יֶהֱמֶה כְקוֹל הֶהָמוֹן / וְלָנֶצַח אֵין בָּךְ אֵמוּן

מֵעַפְרוֹת גִּבְעוֹתַיִךְ יֵצֵא פָז הֶחָרוּץ / וְרָאשֵׁי הַבְּרִיּוֹת תִּכְרְתִי בְמוֹרַג חָרוּץ

כָּל הַנִּבְרָאִים בְּתוֹכֵךְ קְבוּרִים / וּבַחֲרוֹנֵךְ תְּכַלִּי הַנְּבוֹנִים וְהַבּוּרִים

אָהֳלֵיהֶם עַל בָּמוֹתַיִךְ תְּקוּעִים / וְנִבְלָתָם בְּתוֹךְ עֲפָרֵךְ שְׁקוּעִים

בַּחֲמָתֵךְ תְכַלִּי כָל שַׂר וָמֶלֶךְ / וְלֹא תָחוּסִי עַל עָשִׁיר וָהֵלֶךְ

בַּעֲפָרֵךְ יִכְהוּ הָעֵינַיִם הַמְּאִירוֹת / וְלֹא יוּכְלוּ לִרְאוֹת הַמְּאוֹרוֹת

מֵעֲבוֹדָתֵךְ הַקָּשָׁה כָל נִבְרָא יִתְחוֹלָל / וּבְתוֹךְ חָרְבוֹתַיִךְ יִשָּׁמַע נַאֲקַת חָלָל

הַנְּפָשׁוֹת הַיְקָרוֹת בְּקִרְבֵּךְ לְחוּצוֹת / וְלֹא יוּכְלוּ לָצֵאת לַחוּצוֹת

חָרְבוֹת חוֹמוֹתַיִךְ דָּם יִרְעָפוּ / וְכָל הַיְצוּרִים מִתַּחֲלֻאַיִךְ יִזְעָפוּ

כָּל הָרוּחוֹת אֵלַיִךְ יוּבָלוּ / וְעַל נַפְשׁוֹתָם הָעֲדִינוֹת יִתְאַבָּלוּ

דַּלְתּוֹתַיִךְ עֲלֵיהֶם יִנְעָלוּ / וּבְחֶלְאַת עֲפָרֵךְ יִגְעָלוּ

פִּנּוֹתַיִךְ עַל פְּנֵיהֶם יִקְצֹפוּ / וּבְחַדְרֵי מִשְׁכְּנוֹתָיִךְ כַּנּוֹדוֹת יָצוּפוּ

מֵחֲמַס עֶבְרָתֵךְ הַיְקָרִים יָסוּפוּ / וְכָל הַמְּלָכִים בַּחֲדָרַיִךְ יֶאֱסֹפוּ

תָּסֹבִּי עֲלֵיהֶם בְּצִירָיִךְ / וְאֵין מַרְפֵּא לְצִירָיִךְ

תַּעֲמִיקִי לָהֶם אֶת הַתְּעָלָה / וּלְמַכּוֹתַיִךְ לֹא יִמָּצֵא תְעָלָה

תְּכַסִּי אֶת פְּנֵיהֶם בְּשׁוּלָיִךְ / וְעַצְמוֹתֵיהֶם תְּבַשְּׁלִי בִקְדֵרוֹת תַּבְשִׁילָיִךְ

בַּחֲמָתֵךְ תְּכַלִּי הָרַכָּה וְהָעֲנֻגָּה / וְכָל בְּתוּלָה מִקִּצְפֵּךְ נוּגָה

הַקְּצִינִים בִּיתֵדוֹתַיִךְ תְּלוּאִים / וְלָנֶצַח בְּקִרְבֵּךְ כְּלוּאִים

חַצְרוֹת מִשְׁכְּנוֹתַיִךְ חֹשֶׁךְ וַאֲפֵלָה / וְכָל קוֹמָה בְתוֹכֵךְ טְפוּלָה

גִּבְעוֹתַיִךְ קָשִׁים כְּעֶשֶׁת שֵׁן / וְכָל נִבְרָא בַעֲפָרֵךְ יָשֵׁן

כָּל הַפְּגָרִים אֵלַיִךְ נִקְרָאוֹת / וְלֹא יוּכְלוּ לַעֲלוֹת וְלֵרָאוֹת

מִפֵּרוֹתַיִךְ יִשְׂבְּעוּ עֲשִׁירִים וְרָזִים / וּמֵאִילָנוֹתַיִךְ יִצְמְחוּ כָל אֲרָזִים

מִתּוֹךְ מַעְיְנוֹתַיִךְ יַבִּיעוּ הַבְּאֵרוֹת וְהַבְּרֵכוֹת / וְעַל פֵּרוֹתַיִךְ יָחוּלוּ כָל הַבְּרָכוֹת

מִיּוֹם הֱיוֹתֵךְ אַתְּ לֹא מְטֹהָרָה / וּבִטְנֵךְ מְלֵאָה חָמָס כְּאִשָּׁה הָרָה

תְּלָאוֹתַיִךְ יָבֹאוּ עָלַי כְּנֶשֶׁר מְהֵרָה / וּמֵעֶלְבּוֹנִי לֹא תוֹפַע עָלַיִךְ נְהָרָה

לְמָתַי אְֶקְרָאֵךְ יַעְלָה יְפֵיפִיָּה / וְאַתְּ חֲמָתֵךְ עָלַי חוֹפִיָּה

קוֹלֵךְ יָבֹא בְאָזְנַי כְּקוֹל הַמְּרִיָּה / וּתְלָאוֹתַיִךְ כְּבֵדִים עָלַי כְּהַר הַמּוֹרִיָּה

אָנֹכִי נְטַעְתִּיךְ שׂוֹרֵק כְּגֶפֶן פּוֹרִיָּה / וְאֵיךְ נֶהְפַּכְתְּ כְּשִׁפְחָה נֶחֱרֶפֶת נָכְרִיָּה

כְּרֵסִי תְמַלְּאִי מִתּוֹרֵי יוֹנָיִךְ / וּתְשַׁכְּרִי אוֹתִי בְשִׁמְרֵי יֵינָיִךְ

עַצְמוֹתַי תִּשְׂרְפִי כְאֵזוֹב יְוֵנָיִךְ / וְתִמְכְּרִינִי לְעֶבֶד לִבְנֵי יְוָנָיִךְ

כְּגַנָּב אֶרְדֹּף אַחֲרֵי פְנִינָיִךְ / וּמִפַּחֲדִי אֶאֱרֹב בְּעֵבֶר פִּנָּיִךְ

לָרַע תָּשִׂימִי לִקְרָאתִי פוֹנָיִךְ / וְאֵיךְ אוּכַל לִבְרֹחַ מִפָּנָיִךְ

תִּרְדְּפִי אַחֲרַי בְּכָל עוֹנָיִךְ / וְתַחֲרִישִׁי לְגוּפִי עִמְקֵי מַעֲנָיִךְ

תַּרְהִיבִי נַפְשִׁי בְעַפְעַפֵּי עֵינָיִךְ / וְתַרְבִּי עָלַי אֶת מְעַנָּיִךְ

הִכָּלְמִי מִלִּפְנֵי הָאָדוֹן קוֹנָיִךְ / וְאַל תִּלְחָצִינִי בְתוֹךְ קִנָּיִךְ

הָסִירִי מֵעַל רֹאשִׁי קָנָיִךְ / וְאַל תִּמְסְרִינִי בְיַד קוֹנָיִךְ.


לָכֵן שִׁמְעִי הַתּוֹכָחוֹת אֲשֶׁר עָרַכְתִּי / וְאֶת נַפְשִׁי בְדָמִים יְקָרִים עָרָכְתִּי

בִּנְתִיבוֹתַיִךְ הַמְּקֻלְקָלִים בִּגְאוֹנִי הָלָכְתִּי / וְעַתָּה בְיִצְרִי הַטּוֹב נִמְלָכְתִּי

אֶת נַפְשִׁי מֵחֲטֹא חָשָׂכְתִּי / וְאֶת יֵינֵךְ לֹא מָסָכְתִּי

בְּחַצְרוֹת אֱלֹדַי יָשַׁבְתִּי / וְיִרְאָתוֹ בְּלִבִּי חָשָׁבְתִּי

אֶת כַּפַּי מֵהַגֶּזֶל הֵשָׁבְתִּי / וּבְאָזְנַי לְדִבְרֵי הַתּוֹרָה הִקְשָׁבְתִּי

עִם הָרְשָׁעִים לֹא הִתְעָרָבְתִּי / וּלְלֶחֶם חָמָס לֹא עָרָכְתִּי

לְבָתֵי הַחוֹטְאִים לֹא קָרָבְתִּי / וְאֶת חֶלְבִּי כְּקָרְבָּן הִקְרָבְתִּי

אֶת לְבָבִי לִתְשׁוּבָה הֵסָבְתִּי / כִּי מִתְּלָאוֹתַיִךָ זָקַנְתִּי וָשָׂבְתִּי

אָכֵן בְּלִבִּי תַרְתִּי / וְדֶרֶךְ אֱמוּנָה בָחָרְתִּי

כְּאַרְיֵה מִמְּקוֹמִי נִעָרְתִּי / וְחָצְנִי מֵהַגֶּזֶל נִעָרְתִּי

אֶת הַמִּצְווֹת עָלַי הֶחֱמָרְתִּי / וְאֶת עֵינַי בִּדְמָעוֹת הֶחֱמָרְתִּי

עַל יִצְרִי הָרַע קָבָלְתִּי / וְתוֹכְחוֹת הַמּוּסָר עָלָי קִבָּלְתִּי

לְכָל אֶבְיוֹן לֹא חִבָּלְתִּי / וּבִגְדֵי הָעֲרֻמִּים לֹא חָבָלְתִּי

בְּהֶגְיוֹן שְׂפָתַי לְיוֹצְרִי הִלָּלְתִּי / וְעַל אַנְשֵׁי עַוְלָה הִתְהוֹלָלְתִּי

רְאֹה פָנַיִךְ לֹא פִלָּלְתִּי / וְלִפְנֵי חֵי הָעוֹלָמִים הִתְפַּלָּלְתִּי

וְעַתָּה אַל תְּגַלִּי לִקְרָאתִי נַבְלוּתֵךְ / כִּי קָרוֹב הוּא יוֹם בְּלוֹתֵךְ

הִנֵּה לְמַעְלָה סִפַּרְתִּי אֶת גְּדֻלָּתֵךְ / הִנֵּה בְאֶפֶס תִּקְוָה יִפְּלוּ מִגְדְּלוֹתֵךְ

הִתְרַחֲקִי מִנֶּגֶד עֵינַי פֶּן אֶשְׁבֹּר אֶת קֻלָּתֵךְ / כִּי מִפְנֵי תַאֲוַת פֵּרוֹתַיִךְ בָּאָה עָלַי קִלְלָתֵךְ

שְׂאִי עֵינַיִךְ וּרְאִי דַלּוּתֵךְ / וְהָסִירִי מֵרֹאשִׁי עֲצֵי דָלִיּוֹתֵךְ

אַל תַּעֲלִי עָלַי בְּמַעֲלוֹתֵךְ / כִּי מָחָר תִּפְּלִי מִמַּעֲלָתֵךְ

אוֹיָה לִיךְ אֶרֶץ אוֹכֶלֶת יוֹשְׁבֶיהָ / וְכַלְבִיָּא טוֹרֶפֶת אֶת הַבְּרִיּוֹת בְּנִיבֶיהָ

שׁוֹד וָשֶׁבֶר יִתְהַלֵּךְ עַל סְבִיבֶיהָ / וּמֵחֲמָסָהּ בָּגְדוּ בָהּ כָּל אוֹהֲבֶיהָ

אָנָה מִפָּנַיִךְ אֶבְרַח / וּמֵעֲבוֹדָתֵךְ הַקָּשָׁה אֶטְרַח

בְּרֹב תְּלָאוֹתַיִךְ אֶת נַפְשִׁי תַזִּיקִי / וְיַד עָנִי וְאֶבְיוֹן לֹא תַחֲזִיקִי

קוֹל הֲרוּגַיִךְ צוֹעֲקִים בְּחֵמָה / וּפִגְרֵיהֶם מֻשְׁלָכִים חוּץ לַחוֹמָה

כָּל בָּאַיִךְ לֹא יְשׁוּבוּן / וְאֶת עֲפָרֵךְ בְּפִיהֶם יִשְׁאָבוּן

עַל סִפֵּי חָרְבוֹתַיִךְ יִשְׁכָּבוּן / וּמִפְּנֵי זַעַם עֶבְרָתֵךְ יִדְאָבוּן

חוֹרֵי חוֹמוֹתַיִךְ מְלֵאִים דִּבְיוֹנִים / וּבִכְנָפַיִךְ נִמְצְאוּ דְמֵי הָאֶבְיוֹנִים

בַּחֲרוֹנֵךְ תַּשְׁחִיתִי הַמְּסֻלָּאִים והַיְקָרִים / וּבְתוֹךְ חוּצוֹתַיִךְ כֻּלָּם מְדֻקָּרִים

הַתִּינוֹקוֹת יָבוֹאוּ אֵלַיִךְ בַּחֲצִינִים / וְלֹא תַשְׁאִירִי הַפַּחוֹת וְהַקְּצִינִים

כָּל חַכְמֵי לֵב וְהַיּוֹעֲצִים / תַּחַת בָּמוֹתֵךְ כַּיְתֵדוֹת נְעוּצִים

הַמְּלָכִים לַעֲבוֹדַת שְׂדוֹתַיִךְ נֶעֱבָדִים / וְאֶת עֲפָרֵךְ יְנַשְּׁקוּ כָעֲבָדִים

הַפְּגָרִים רוֹחֲשִׁים בְּקִרְבֵּךְ כַּנְּדָלִים / וְלֹא תַשְׁאִירִי הַקְּטַנִּים וְהַגְּדוֹלִים

מִתְּלָאוֹתַיִךְ יִצְעֲקוּ הָאֲנָשִׁים וְהַנָּשִׁים / וּמִקִּלְלָתֵךְ הַקְּדוּמָה הֵמָּה נֶעֱנָשִׁים

תֹּאַר חֶמְדַּת הַחֲתָנִים וְהַכַּלּוֹת / תַּחַת כַּפּוֹת מַנְעָלַיִךְ כָּלוֹת

בִּמְקוֹם מִבְחוֹר בְּרוֹשָׁם / עָפָר יָשִׂימוּ בְרֹאשָׁם

בִּמְקוֹם הַדְרַת יְקָרָם / יִבָשׁ כַּחֶרֶס מְקוֹרָם

בִּמְקוֹם תִּפְאֶרֶת כְּבוֹדָם / רִמָּתֵךְ תְּפַלַּח כְבֵדָם

בִּמְקוֹם מִרְקַחַת רֵיחָם / חֲמָתֵךְ שׁוֹתָה רוּחָם

בִּמְקוֹם מֵי הַוְּרָדִים / הַנְּחָשִׁים בְּגוּפָם רוֹדִים

בִּמְקוֹם נְעִילַת הָעֲכָסִים / יִשְׁתּוּ אֶרֶס הָעֲכָסִים

בִּמְקוֹם שֶׁמֶן מִשְׁחָתָם / יִפְּלוּ בְּבוֹר שַׁחֲתָם

בִּמְקוֹם הַפְּאֵרִים וְהַקִּשּׁוּרִים / בְּצִירֵי דַלְתוֹתַיִךְ קְשׁוּרִים

בִּמְקוֹם הַנְּטִיפוֹת וְהָרְעָלוֹת / יִשְׁתּוּ כוֹס הַתַּרְעֵלוֹת

חֶמְדַּת הַשְּׁבִיסִים וְהַשַּׂהֲרוֹנִים / כְּלוּאִים בְּתוֹךְ הַסַּהֲרוֹנִים

נִזְמֵי הָאַף וְהַטַּבָּעוֹת / בְּתוֹךְ עֲפָרֵךְ טְבוּעוֹת

בָּתֵי הַנֶּפֶשׁ וְהַלְּחָשִׁים / הַנְּחָשִׁים עֲלֵיהֶם מִתְלַחֲשִׁים

בִּגְדֵי הַמִּטְפָּחוֹת וְהַמַּחֲלָצוֹת / תַּחַת אֲבָנַיִךְ מְחֻלָּצוֹת

לוֹבְשֵׁי הַחֲרִיטִים וְהַמַּעֲטָפוֹת / בְּתַכְרִיךְ בּוּז מְעֻטָּפוֹת

אַנְשֵׁי הַתַּעֲנוּג וְהַמְּנוּחָה / תְּכַלִּים בְּיָגוֹן וַאֲנָחָה

רָעוֹתַיִךְ אַחֲרֵיהֶם תִּדְבָּק / וּלְשׁוֹנָם לְחִכָּם יִדְבָּק

כְּאֱנֹק חֲלָלִים יֶאֱנָקוּ / וְכוֹס שְׁמָרַיִךְ יִינָקוּ

עֲפָרֵךְ לִבְשָׂרָם יִטְחַן / וְתוֹלַעְתֵּךְ תֹּאכַל בַּטּוּחַן

אֶת קִשְׁרֵי הַבְּרִיּוֹת תְּשַׁבְּרִי בַּיְתֵדוֹת / וְתִקְחִי מַהֶם נְקָמָה עֶשֶׂר יָדוֹת

כָּל הַפְּגָרִים בְּתוֹכֵךְ נִקְבָּרוֹת / וְנִבְלָתָם תִּתְּנִי לָרִמָּה לְבָרוֹת

בַּעֲלֵי הַגְּבוּרָה בְקִרְבֵּךְ סְרוּחוֹת / וְאֵלַיִךְ יָבֹאוּ כָל הָרוּחוֹת

הַנְּפָשׁוֹת יִרְבְּצוּ עָלַיִךְ כַּעֲדָרִים / וּמֵחַדְרֵי טִירוֹתָם תְּשִׂימִים נֶעְדָּרִים

כָּל הַגַּמָּלִים וְכָל הַחַמָּרִים / תִּצְבְּרִי פִגְרֵיהֶם חֳמָרִים חֳמָרִים

כָּל הַנִּבְרָאִים אֲשֶׁר הֵמָּה בְּמִשְׁכְּנוֹתַיִךְ דָּרִים / תַּאַסְרִי אוֹתָם בְּקִרְבֵּךְ לָעַד וּלְדוֹרֵי דוֹרִים.


אִי לָךְ אֶרֶץ שֶׁמַּלְכֵּךְ נַעַר / וְעַל כָּל הַבְּרִיּוֹת חֲמָתֵךְ תַּעַר

קוֹלֵךְ יַחֲרִיד כְּקוֹל הָאַרְיֵה הַשּׁוֹאֵג בַּיַּעַר / וַעֲפָרֵךְ לֹא יַשְׁאִיר כָּל נָבוֹן וָבַעַר

מִיּוֹם הֱיוֹתֵךְ תֹּהוּ וָבֹהוּ / בְּכָל חוּצוֹת יֹאמְרוּ הוֹ הוֹ

יָדַיִךְ בְּרָאשֵׁי הַיְקָרִים הוֹיָה / וְלֹא תַשְׁאִירִי נֶפֶשׁ וּגְוִיָּה

הַמָּוֶת בְּרָאשֵׁךְ תֶּחְבָּשִׁי / וּבִגְדֵי נָקָם תִּלְבָּשִׁי

פְּרוּדוֹתַיִךְ בַּדָּם תֶּעְבָּשִׁי / וְהַבְּרִיּוֹת לַעֲבָדִים תִּכְבָּשִׁי

סַם הַמָּוֶת דְּבַשׁ נְחִילָתֵךְ / וְקוֹץ וְדַרְדַּר חֲמַס נַחֲלָתֵךְ

עַל הַמְּלָכִים תְּקוֹנְנִי בִמְחוֹלָתֵךְ / וְתִקְבְּרִי אוֹתָם בְּעַפְרוֹת מְחִלָּתֵךְ

רוֹשׁ פְּתָנִים רֵאשִׁית חַלָּתֵךְ / וְאֵין מַרְפֵּא לְצִירֵי מַחֲלָתֵךְ

בּוֹשִׁי וְהִכָּלְמִי הָאָרֶץ הַנְּשַׁמָּה / כִּי מַה בֶּצַע תִּרְדְּפִי אַחֲרֵי הַנְּשָׁמָה

הִנֵּה יִרְעוּךְ קָדְקֹד תְּיָשֵׁי אֲשִׁימָא / וּמֵעֶלְבּוֹנִי יִגְדַּל עָלַיִךְ פֶּשַׁע וְאַשְׁמָה

כַּלֵּךְ מִדְבָּרוֹתַיִךְ נְגָעִים וְחַלְחָלוֹת / וְלֹא תְּתוֹפְפִי לִקְרָאתִי בִמְחוֹלוֹת

הַבִּיטִי וּרְאִי חַסְדֵי נוֹרָא עֲלִילוֹת / הֲלֹא בְרַחֲמָיו יַשְׁאִיר לָנוּ עוֹלֵלוֹת

שִׁמְעִי קוֹלִי אֶרֶץ רַחֲבַת יָדַיִם / וְלֹא תָבוֹאִי עָלַי בַעֲדִי עֲדָיִים

כִּי מִתְּגְרָתֵךְ רָפוּ כָל יָדַיִם / אֲנָשִׁים וְנָשִׁים עוֹלְלִים וְיוֹנְקֵי שָׁדַיִם

תְּלָאוֹתַיִךְ יָבוֹאוּ עָלַי כַּגְּדוּדִים / וּמֵחֲרוֹן אַפֵּךְ בְּגָדַי גְּדוּדִים

אֲבָנַיִךְ יִפְּלוּ עַל רֹאשִׁי כַּגְּדוּדִים / וְרֶמֶשׂ אַדְמָתֵךְ יָשִׂים אֵיבָרַי גְּדוּדִים

בַּעֲבוּר קִצְפֵּךְ אַרְחִיק נְדוּדִים / וְעַפְעַפַּי לֹא יִשְׂבְּעוּ נְדוּדִים

מֵאֵימָתֵךְ כָּל עַצְמוֹתַי מִתְנוֹדְדִים / וְלַיְלָה וְיוֹמָם בְּבִכְיָתָם מִתְנוֹדְדִים

לְבִיּוֹתַיִךְ יִתְהַלְּכוּ לְפָנַי כְּעֵדֶר הַקְּצוּבָה / וּפְעָמָי בְּתוֹךְ עֲפָרֵךְ כְּעֵצִים קְצוּבָה

לְהָרַיִךְ מִי יוּכַל לֵירֵד לְקִצְבָה / וְקוֹרוֹתַיִךְ אֵין לָהֶם תַּכְלִית וְקִצְבָה

צִירֵי דַלְתוֹתַיִךְ סְגוּרִים עָלַי מִבֹּא / וּמִפַּחְדֵּךְ לֹא אוּכַל לָצֵאת וְלָבוֹא

מִי הוּא זֶה אֲשֶׁר מְלָאוֹ לִבּוֹ / שֶׁיִשְׂרֹף אֶת גָּרְנֵי תְבוּאָתֵךְ בְּגַחֶלֶת לִבּוֹ

אַתְּ יוֹשֶׁבֶת כְּמֶלֶךְ בִּמְסִבּוֹ / מְשָׁרְתָיו וַעֲבָדָיו הוֹלְכִים סְבִיבוֹ

וַאֲנִי כְעָנִי שׁוֹכֵב בְּקֶרֶב הַמַּחֲבוֹא / וּבֵיתִי סָגוּר עָלַי לְאָרְכּוֹ וּלְרָחְבּוֹ.


אֶרֶץ בֹּאִי בֶעָפָר וְהִטָּמְנִי בַצּוּר / עַד מָתַי עָרַיִךְ לִקְרָאתִי תָצוּר

חַרְבֵּךְ הַקָּשָׁה יֹאכַל הַנִּשְׁאָר וְהַנָּצוּר / וְכָל יְצִיר נִבְרָא בְמַגָּלֵךְ קָצוּר

נְתִיבוֹת דְּרָכַיִךְ הָעֲקַלְקַלּוֹת סְבוּכִים כִמְשׂוּכַת סִיר / וּבִטְנֵךְ מְלֵאָה אַף וְחֵמָה וּנְפוּחָה כַסִּיר

אֱלֹקַי בְּרֹב רַחֲמָיו אֶת גְּדֻלָּתֵךְ יַחֲסִיר / וִיבַלַּע הַמָּוֶת לָנֶצַח וְחֶרְפַּת עַמּוֹ יָסִיר

וַאֲנִי כָל יְמֵי צְבָאִי אֲסַלְּדָה בְצִיר / לֹא אָנוּחַ וְלֹא אֶשְׁקֹט בֶּחָרִישׁ וּבַקָּצִיר

אֶשָּׂא עֵינַי לִפְנֵי קוֹנִי וּבְתַחֲנוּנַי אַפְצִיר / יִשְׁלַח רְפוּאָה לְעַמּוֹ וְיַצִּילֵם מִנֵּזֶק וּמִצִּיר

וְצֹאן מַרְעִיתוֹ יְקַבֵּץ עַל יְדֵי צִיר / וִיקָרֵב הַגְּאֻלָּה וְאַרְיֵה כַבָּקָר יֹאכַל חָצִיר.


וְעַתָּה הַקְשִׁיבִי לְתוֹכַחְתִּי וְלֹא תָגוּרִי / וְהִתְהַלְּכִי לִמְנוּחָתֵךְ וְגוּרִי בַאֲשֶׁר תָּגוּרִי

מַלְּאִי גְרוֹנִי מִדְּגַן תִּירוֹשֵׁךְ הָאָגוּרִי / וְהָאִירִי לְפָנַי בְּשֶׁמֶן זֵיתֵךְ הָאָגוּרִי

הַשְׁקִינִי מַיִם מִתּוֹךְ קָלָלֵךְ / וְלֹא תְקַלְּלִי אוֹתִי בִקְלָלֵךְ

רוּצִי לְפָנַי כְּצִבְיָה בִּמְרוּצַת קוֹלֵךְ / וְהַשְׁמִיעִינִי נָא אֶת נְעִימוּת קוֹלֵךְ

אִם מָצָאתִי חֵן בְּעֵינָיִךְ / אַל תַּרְבִּי עָלַי מְעַנָּיִךְ

הַרְחִיבִי לִי מִשְׁכְּנוֹת מְעוֹנָיִךְ / וַהֲשִׁיבִינִי דָבָר בִּלְשׁוֹן מַעֲנָיִךְ.
