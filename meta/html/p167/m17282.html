סֵפֶר הַהִגָּיוֹן

סֵפֶר הַהִגָּיוֹן שחבר החכם הנעלה משה חיים לוצאטו נר"ו באמשטרדם

שנת התק“ב לפ”ק

יצא לאור בפעם הראשונה בדפוס על ידי בית מערכת הצפירה.

ווארשא בדפוס ה' מ. לעווינסקי. שנת תרנ“ז לפ”ק 1897.


הַקְדָּמָה

כְּשֶׁנַּשְׁקִיף עַל בְּרִיּוֹת הָאָרֶץ וּמַה שֶּׁמַּגִּיעַ אֲלֵיהֶם, נִרְאֶה, שֶׁנָּתַן אֱלֹהִים לִבְנֵי הָאָדָם חֵלֶק גָּדוֹל בְּיִשּׁוּבוֹ שֶׁל הָעוֹלָם הַזֶּה וּבְהַשְׁלָמַת כָּל מְלוֹאוֹ, עַד שֶׁנּוּכַל לוֹמַר שֶׁבְּרִיאַת בְּרֵאשִׁית בְּמַה שֶּׁנּוֹגֵעַ לַחֵלֶק הַזֶּה לֹא הָיְתָה אֶלָּא כְּמוֹ הַתְחָלָה לַנִּמְצָאוֹת הָאֵלֶּה וְכֹחָנִיּוֹת לְהַשְׁלָמָתָם; אָמְנָם הַשְׁלָמָתָם בְּפֹעַל נִמְסְרָה לָאָדָם לְגַמְרֵי. וּכְבָר הֶעִירוּנוּ חֲכָמֵינוּ זַ"ל עַל אֲמִתַּת הָעִנְיָן הַזֶּה שֶׁאָמְרוּ, שֶׁכָּל שֶׁנִּבְרָא בְמַעֲשֵׂה ברֵאשִׁית צְרִיכִין תִּקּוּן, חִטִּים צְרִיכִים טְחִינָה, תֻּרְמוֹסִין צְרִיכִים מִתּוּק, וְכֵן כֻּלָּם. וּמָצִינוּ עוֹד שֶׁשָּאַל טוֹרְנוֹס־רוּפוֹס אֶת ר' עֲקִיבָא: אֵיזֶה מַעֲשִׂים נָאִים, שֶׁל שָׁמַיִם אוֹ שֶׁל בָּשָׂר וָדָם? וֶהֱשִׁיבוֹ: שֶׁל בָּשָׂר וָדָם נָאִים, וּלְאַמֵּת דְּבָרָיו הֵבִיא לוֹ שִׁבֳּלִים וּגְלֻסְקָאוֹת וְאָמַר לוֹ: אֵיזֶה מֵהֶם נָאִים? וּבֶאֱמֶת עֵינֵינוּ הָרוֹאוֹת שֶׁכָּל הַדְּבָרִים הַטִּבְעִים כְּשֶׁיֻּנְּחוּ בְטִבְעָם בְּלִי עֲבוֹדַת הָאָדָם וּמְלַאכְתּוֹ הִנֵּה יַעֲשׂוּ מִמַּה שֶׁבְּחֻקָּם הַטִּבְעִי אַךְ לֹא בְהַשְׁלָמָה, וּכְשֶׁיַּעֲסֹק בָּהֶם הָאָדָם וְיַעֲזֹר אֶת טִבְעָם וִיסַדֵּר תּוֹצְאוֹתֵיהֶם עַל הַסֵּדֶר הֶהָגוּן, הִנֵּה יֵרָאוּ בְּבֵרוּר יוֹתֵר שְׁלֵמִים וְנָאִים, וְזֶה פָּשׁוּט. דֶּרֶךְ מָשָׁל, אִילָן מִן הָאִילָנוֹת אִם יֻנַּח לְטִבְעוֹ בְּלִי עֲבוֹדָה, אַף אִם יִתֵּן מִמִּין הַפְּרִי הַמְּיֻחָד לוֹ, לֹא יִשְׁוֶה לֹא בְטַעְמוֹ וְלֹא בְיָפְיוֹ אֶל אֲשֶׁר יִתֵּן אִם יְעֻבַּד, וְהָאֲדָמָה עַצְמָהּ אֲשֶׁר לֹא תֵעָבֵד, לֹא תִזָּבֵל וְלֹא תֵעָדֵר – אוֹ לֹא תִּתֵּן יְבוּלָהּ כָּל עִקָּר אוֹ תִתְּנֵהוּ רָזֶה וְחָסֵר, זוּלָתִי קְצָת הַמְּקוֹמוֹת אֲשֶׁר תִּרְבֶּה בָם הַהֲכָנָה הַטִּבְעִית רִבּוּי גָּדוֹל עַד שֶׁלֹּא יִצְטָרְכוּ לַעֲבוֹדָה וְלֹא יִהְיֶה זֶה עַל צַד הָרֹב כְּלָל. כְּלָלוֹ שֶׁל דָּבָר: מָקוֹם הִנִּיחַ הַבּוֹרֵא יִתְבָּרֵךְ לִבְנֵי הָאַָדָם בְּמַה שֶּׁיָּדָם מַגַּעַת, שֶׁיִּתְגַּדְּרוּ בוֹ בְּגַדֵּל הֵמָּה בַּעֲמָלָם אֵת אֲשֶׁר תֵּלֵד הַטֶּבַע. וְאוּלָם כַּמִּקְרֶה הַזֶּה יִקְרֶה גַּם לְשִׂכְלוֹ שֶׁל הָאָדָם עַצְמוֹ, שֶׁעִם הֱיוֹת הַהַשְׂכָּלָה מֻטְבַּעַת בּוֹ בְּטִבְעוֹ מִתְּחִלָּה, אִם לֹא יְעֻבַּד יִהְיֶה כְאִילָן אֲשֶׁר לֹא יֵעָבֵד וְכַאֲדָמָה אֲשֶׁר לֹא שָׂמוּ לֵב עָלֶיהָ; שֶׁאַף עַל פִּי שֶׁלֹּא יֶחְדַּל מֵהַשְׂכִּיל דְּבָרִים רַבִּים, כַּאֲשֶׁר לֹא יֶחְדַּל הָאִילָן מִתֵּת מִפִּרְיוֹ הַמְּיֻחָד בּוֹ, אָמְנָם לֹא תִשְׁוֶה הַשְׂכָּלָתוֹ אַחֲרֵי הֵעָבְדוֹ לְהַשְׂכָּלָתוֹ זוּלַת זֶה, כַּאֲשֶׁר לֹא יִשְׁוֶה פְּרִי הָאִילָן הַנֶּעֱבָד עִם פִּרְיוֹ זוּלַת זֶה. וּכְמוֹ שֶׁהוּא צָרִיךְ לִמּוּד בְּנוֹשְׂאֵי הַמְּצִיאוּת, דְּהַיְנוּ הַמֻּשְׂכָּלוֹת כֻּלָּם שֶׁהֵם הָעֵסֶק שֶׁלּוֹ וּבְלֹא זֶה לֹא תֵּצֵא הַשְׂכָּלָתוֹ מִכֹּחָהּ אֶל הַפֹּעַל, כֵּן צָרִיךְ לִמּוּד בַּהַשְׂכָּלָה עַצְמָהּ כְּדֵי שֶׁתִּהְיֶה הֵשְׂכָּלָתוֹ בְמֻשְׂכָּלָיו הַשְׂכָּלָה שְׁלֵמָה וּמְתֻקֶּנֶת וְלֹא חֲסֵרָה וּמְבֻלְבֶּלֶת. וְאָמְנָם, לְתַכְלִית זֶה טָרְחוּ הַחֲכָמִים שֶׁקָּדְמוּ לָנוּ לְסַדֵּר סִדְרֵי הַהַשְׂכָּלָה וּלְבָאֵר חֻקּוֹתֶיהָ עַל נָכוֹן, בְּהַשְׁקִיף עַל מַה שֶּׁיִּקְרֶה לַשֵּׂכֶל בְּהַשְׂכָּלוֹתָיו מִן הַהַשָּׂגָה וּמִן הַסִּכְלוּת, מִן הַצִּיּוּר הָאֲמִתִּי וּמִן הַכּוֹזֵב, לְמַעַן דַּעַת מִמַּה יִּשָּׁמֵר בְּהִתְבּוֹנְנוּתוֹ כְּדֵי שֶׁלֹּא יִטְעֶה, וּבַמֶּה יֵעָזֵר כְּדֵי שֶׁיַּשִּׂיג הָאֱמֶת. וּכְלָל הַלִּמּוּד הַזֶּה הוּא הַנִּקְרָא מְלֶאכֶת הַהִגָּיוֹן. וּבִרְאוֹתִי הַצֹּרֶךְ הַגָּדוֹל אֲשֶׁר לָנוּ בֶּאֱמֶת בַּדָּבָר הַזֶּה, שֶׁבִּלְעָדָיו אִי אֶפְשָׁר לִכָּנֵס אֶל חַדְרֵי הַחָכְמוֹת וּלְהִשְׁתַּעֲשֵׁעַ בְּתַעֲנוּגֵיהֶם כָּרָאוּי, בָּחַרְתִּי לְסַדֵּר זֶה הַלִּמּוּד בְּקִצּוּר כְּפִי מַה שֶּׁרְאִיתִיו מֻכְרָח לְתַשְׁלוּם הָעִנְיָן. רֻבּוֹ הֶעְתַּקְתִּיו מִסְּפָרִים שֶׁקְּדָמוּנִי בִּלְשׁוֹנוֹת אֲחֵרוֹת וַהֲבֵאתִיו אֶל לְשׁוֹנֵנוּ לְהוֹעִיל לִבְנֵי עַמֵּנוּ. וְעִנְיָנִים הוֹסַפְתִּי, גָּרַעְתִּי וְשִׁנִּיתִי מֵאֲשֶׁר מָצָאתִי לְפָנַי, כְּמוֹ שֶׁנִּרְאֶה לִי יוֹתֵר נָאוּת. אָנָּא אַתָּה, קוֹרֵא נָעִים, עֲמֹד וְהִתְבּוֹנֵן, שְׁמַע וְהוֹסֵף לֶקַח וָדַעַת לְנַפְשְׁךָ יִנְעָם.


פֶּרֶק רִאשׁוֹן

כְּלָל מְלֶאכֶת הַהִגָּיוֹן מַה הִיא וְהַצֹּרֶךְ בָּהּ. חִלּוּף הַמְּצִיאֻיּוֹת לְמֻרְגָּשׁ וּמֻשְׂכָּל, וְהַמֻּשְׂכָּל לְמֻפְרָד וּמְדֻמֶּה וְהַמְדֻמֶּה לְמֻשָּׂג וּבָדוּי.


א הִנֵּה כָּל עֲמַל הַשֵּׂכֶל וְהִשְׁתַּדְּלוּתוֹ הוּא לְהַשִּׂיג הַדְּבָרִים לַאֲמִתָּם. אָמְנָם, יֵשׁ שֶׁיִּקְרֶה לוֹ לִטְעוֹת בָּהֶן וְיִהְיֶה מַה שֶּׁיַּשִּׂיג מֵהֶן כּוֹזֵב, עַל כֵּן יִצְטָרֵךְ לוֹ לִמּוּד לָדַעַת לְהַכִּיר מְקוֹמוֹת הַמִּכְשׁוֹל שֶׁאֶפְשָׁר לוֹ לִכָּשֵׁל בָּם וְיֵדַע אֵיךְ יִשָּׁמֵר מֵהֶם עַד שֶׁתִּהְיֶה הַשָּׂגָתוֹ אֲמִתִּית. וּכְלָל הַלִּמּוּד הַזֶּה הוּא הַנִּקְרָא מְלֶאכֶת הַהִגָּיוֹן.

ב עִקַּר מְלֶאכֶת הַשֵּׂכֶל בְּבַקָּשָׁתוֹ יְדִיעַת הַדְּבָרִים הוּא הַהַבְחָנָה עַד שֶׁיַּכִּיר כָּל דָּבָר וְדָבָר בִּפְנֵי עַצְמוֹ נִבְחָן וְנִבְדָּל מִכָּל זוּלָתוֹ. אָמְנָם בִּהְיוֹת שֶׁכְּלָל הַמְּצִיאֻיּוֹת יֵשׁ בָּהֶם מַה שֶׁדּוֹמֶה הָאֶחָד לְזוּלָתוֹ וְיֵשׁ בָּהֶם מַה שֶׁנִּבְדָּל הָאֶחָד מִזּוּלָתוֹ, עַל כֵּן לֹא יַגִּיעַ הַשֵּׂכֶל לְסוֹף מְלַאכְתּוֹ עַד שֶׁיִּשְׁתַּלֵּם בִּשְׁתֵּי הַפְּעֻלּוֹת, שֶׁהֵם: הַדִּמְיוֹן וְהַהֶבְדֵּל.

ג בְּכָל אֶחָד מִשְּׁתֵּי הַפְּעֻלּוֹת יָכוֹל לִפֹּל הַטָּעוּת אִם יְדַמֶּה מַה שֶּׁאֵינוֹ דוֹמֶה, אוֹ יַבְדִּיל מַה שֶּׁלֹּא נִבְדָּל. עַל כֵּן לִשְׁתֵּיהֶן צָרִיךְ לִמּוּד וְאִמּוּן וְחֻקִּים מְסֻדָּרִים, שֶׁיַּעֲמִידוּהוּ בַּנָּתִיב הַיָּשָׁר וְלֹא יִשְׁגֶּה מִמֶּנּוּ, עַד שֶׁיַּשִּׂיג תַּכְלִיתוֹ עַל נָכוֹן.

ד כְּלָל הַמְּצִיאֻיּוֹת יִתְחַלֵּק לִשְׁנֵי חֲלָקִים. הָרִאשׁוֹן: מֻרְגָּשׁ אוֹ מוּחָשׁ; וְהַשֵּׁנִי: מֻשְׂכָּל. הַמֻּרְגָּשׁ הוּא שֶׁמֻּשָּׂג מִן אַחַד הַחוּשִׁים; הַמֻּשְׂכָּל הוּא אֲשֶׁר יְדַמֵּהוּ הַשֵּׂכֶל בְּדִמְיוֹנוֹ, אַךְ אֵין מְצִיאוּתוֹ כֵּן חוּץ מִמֶּנּוּ, וִיחֻלַּק גַּם הוּא לִשְׁנַיִם, הָאֶחָד מֻפְרָד וְהָאֶחָד מְדֻמֶּה. פֵּרוּשׁוֹ: הַמּוּחָשׁ הוּא, דֶּרֶךְ מָשָׁל, הָאֶבֶן וְהָעֵץ, שֶׁרוֹאִים אוֹתוֹ בְחוּשׁ; וְהַמֻּשְׂכָּל – כְּגוֹן הַחָכְמָה, הַזִּכָּרוֹן, הַגְּבוּרָה, שֶׁהֵם דְּבָרִים שֶׁמַּשְׂכִּילִים וּמְצַיְּרִים אוֹתָם בַּשֵּׂכֶל, אַךְ אֵין הַחוּשׁ רוֹאֶה אוֹתָם.

ה הַמֻּפְרָד הוּא – שֶׁמַּה שֶּׁאֵינוֹ מֻשָּׂג מִן הַחוּשִׁים אֶלָּא בְחִבּוּר עִם זוּלָתוֹ, יַפְרִידֵהוּ הַשֵּׂכֶל בְּדִמְיוֹנוֹ וְיִקָּחֵהוּ לְבַדּוֹ. הַמְדֻמֶּה הוּא – אֲשֶׁר אֵינוֹ מֻשָּׂג מִן הַחוּשִׁים כְּלָל לֹא לְבַדּוֹ וְלֹא בְחִבּוּר אֶלָּא הַשֵּׂכֶל הוּא אֲשֶׁר יְדַמֶּה אוֹתוֹ, וְיִתְחַלֵּק גַּם הוּא לִשְׁנַיִם, הָרִאשׁוֹן מֻשָּׂג וְהַשֵּׁנִי בָדוּי. פֵּרוּשׁוֹ: הַמַּרְאֶה, דְּהַיְנוּ הַלֹּבֶן וְהַשַׁחֲרוּת וּשְׁאָר הַגְּוָנִים אֵין הַחוּשׁ רוֹאֶה אוֹתָם לְבַדָּם אֶלָּא נְתוּנִים בְּאֶחָד מִן הַגּוּפִים, אַךְ הַשֵּׂכֶל מְצַיֵּר הַלֹּבֶן הֱיוֹתוֹ עִנְיָן אֶחָד נוֹסָף עַל הַגּוּף שֶׁהוּא בְתוֹכוֹ, וְנִקְרָא שֶׁמַּפְרִידוֹ בְדִמְיוֹנוֹ מִן הַגּוּף שֶׁהוּא נָתוּן בּוֹ וְזֶהוּ הַמֻּפְרָד, אַךְ הַמְדֻמֶּה אֵינוֹ כֵן, וְיִתְבָּאֵר בַּסִּמָּן שֶׁלְּפָנֵינוּ.

ו הַמֻּשָּׂג הוּא הַדָּבָר שֶׁאַף עַל פִּי שֶׁאֵין מְצִיאוּתוֹ הֹוֶה בְּמַה שֶׁמּוּחָש, רָאוּי לִמְצֹא לְפִי הַהַדְרָגָה, וְנִמְשָׁךְ הַמּוּחָשׁ אַחֲרָיו בְּסֵדֶר, וְהַבָּדוּי הוּא אֲשֶׁר אֵינֶנּוּ לֹא הֹוֶה וְלֹא רָאוּי וְלֹא הַמּוּחָשׁ נִמְשָׁךְ אַחֲרָיו בְּסֵדֶר. פֵּרוּשׁוֹ: הַמֻּשָּׂג הוּא – שֶׁמִּמַּה שֶּׁאָנוּ רוֹאִים בְּחוּשׁ אֲנַחְנוּ דָנִים עַל מְצִיאוּת שֶׁאֵין אָנוּ רוֹאִים. דֶּרֶךְ מָשָׁל: אָנוּ רוֹאִים שֶׁיֵּשׁ בְּנֵי אָדָם וְהֵם חַיִּים, בְּהֵמוֹת וְגַם הֵם חַיִּים, עוֹפוֹת וְגַם הֵם כָּךְ, נֹאמַר שֶׁיֵּשׁ מִין אֶחָד שֶׁל בְּרִיאָה בֵּין הַבְּרִיאוֹת שִׁנִּקְרָא מִין הַבַּעֲלֵי חַיִּים, וְאֵין הַמִּין הַזֶּה דָּבָר הֹוֶה מַמָּשׁ זוּלַת אֵלֶּה הַפְּרָטִים שֶׁלּוֹ שֶׁזָּכַרְנוּ, אֶלָּא הַשֵּׂכֶל מְדַמֶּה כְּאִלּוּ יֵשׁ דָּבָר אֶחָד כְּלָלִי שֶׁמִּתְחַלֵּק לַחֲלָקִים פְּרָטִים, לְפִי שֶׁמִּסֵּדֶר הַהַדְרָגָה הוּא זֶה שֶׁיִּהְיֶה הַכְּלָל וְאַחַר כָּךְ הַפְּרָט, כְּמוֹ הָאִילָן וַעֲנָפָיו, וְנִמְצָא שֶׁהַמּוּחָשׁ נִמְשָׁךְ בְּהַדְרָגָה אַחַר הַמְדֻמֶּה, כְּאִלּוּ הָיָה הַמְדֻּמֶּה הַהוּא עִקָּר וְהַמּוּחָשׁ עֲנָפִים מִמֶּנּוּ. מַה שֶּׁאֵין כֵּן הַבָּדוּי, שֶׁהוּא דִמְיוֹן לְבַד, וְאֵין דָּבָר בַּמּוּחָשׁ שֶׁנָּדוּן מִמֶּנּוּ הֱיוֹת הַבָּדוּי הַהוּא.


פֶּרֶק שֵׁנִי:

הָעֶצֶם וְהַמִּקְרֶה וּכְלָלָיו


א עַד הֵנָּה בֵּאַרְנוּ עִנְיְנֵי הַהַשָּׂגָה וּמַחְלְקוֹתֶיהָ, עַתָּה נְבָאֵר הַמֻּשָּׂגִים לְמַחְלְקוֹתֵיהֶם.

ב כָּל מֻשָּׂג הוּא אוֹ עֶצֶם אוֹ מִקְרֶה.

פֵּרוּשׁוֹ: בֵּין הַמֻּשָּׂגִים הַמּוּחָשִׁים וְהַמֻּשׂכָּלִים שֶׁנִּתְבָּאֲרוּ לְמָעְלָה.

ג עֶצֶם – הוּא הַדָּבָר הַמַּמָּשִׁיִּי שֶׁעֲמִידָתוֹ בְעַצְמוֹ וְלֹא בְאַחֵר, וּבוֹ עוֹמְדִים הַמִּקְרִים הַמִּתְחַבְּרִים לוֹ. פֵּרוּשׁוֹ: דֶּרֶךְ מָשָׁל, עֶצֶם הָאֶבֶן, הָעֵץ, הַבַּעַל־חַיִּים וְכָל הַנִּמְצָאִים רוּחָנִים וְגַשְׁמִים.

ד מִקְרֶה – הוּא עִנְיָן יִמָּצֵא בְעֶצֶם וְעוֹמֵד בּוֹ, וְלֹא יַעֲמֹד זוּלַת עֶצֶם.

דֶּרֶךְ מָשָׁל, הַמַּרְאֶה שֶׁזָּכַרְנוּ לְמַעְלָה, הַיֹּפִי, הַחָכְמָה שֶׁאֵינָם נִמְצָאִים בִּפְנֵי עַצְמָם, אֶלָּא נְתוּנִים בְּאֶחָד מִן הָעֲצָמִים.


כְּלָלֵי הַמִּקְרֶה

1 מְצִיאוּת הַמִּקְרֶה הוּא הִתְחַבֵּר לְעֶצֶם.

פֵּרוּשׁוֹ: כִּי הִנֵּה הַמִּקְרֶה אֵינוֹ אֶלָּא עִנְיַן הֱיוֹת בְּעֶצֶם, בְּחִינָה אַחַת אוֹ בְּחִינוֹת רַבּוֹת נוֹסָפוֹת עַל הֱיוֹתוֹ עֶצֶם. אֲשֶׁר הֱיוֹת הַבְּחִינָה הַהִיא נִמְצֵאת בְּעֶצֶם זֶהוּ הַמִּקְרֶה שֶׁקָּרָה לָעֶצֶם.

2 הַמִּקְרֶה תָּלוּי בָּעֶצֶם בֵּין בַּהֲוָיָה בֵּין בַּמְּצִיאוּת.

פֵּרוּשׁוֹ: אִם לֹא הָיָה עִנְיָן הָעֶצֶם לֹא הָיְתָה בְּחִינָה נוֹסֶפֶת לוֹ, וְנִמְצָא שֶׁלֹּא הָיָה עִנְיַן הַמִּקְרֶה כְּלָל, עַל כֵּן נֹאמַר שֶׁהַמִּקְרֶה תָּלוּי בָּעֶצֶם בַּהֲוָיָה. וְכֵן אִם הָיָה נֶעְדָּר, לֹא הָיָה הַמִּקְרֶה קוֹרֶה, עַל כֵּן נֹאמַר שֶׁהַמִּקְרֶה תָּלוּי בָּעֶצֶם בַּמְּצִיאוּת.

3 מִקְרֶה מִבִּלְתִּי עֶצֶם נִמְנָע.

4 הַמִּקְרֶה לֹא יִתְגַּלְגֵּל מִנּוֹשֵׂא לְנוֹשֵׂא אוֹ מֵעֶצֶם לְעֶצֶם.

פֵּרוּשׁוֹ: כִּי הָעֶצֶם הוּא סוֹבֵל מִקְרִים רַבִּים וְיִסְבֹּל מִקְרֶה אֶחָד בִּזְמַן אֶחָד וְאַחַר כָּךְ יִסְבֹּל מִקְרֶה אַחֵר, וְהָעֶצֶם שֶׁסָּבַל הָאֶחָד הוּא שֶׁסּוֹבֵל הַשֵּׁנִי. אַךְ הַמִּקְרֶה שֶׁקּוֹרֶה לָעֶצֶם לֹא יוּכַל לָצֵאת מִמֶּנּוּ וּלְהִתְחַבֵּר אֶל עֶצֶם אַחֵר, אֶלָּא מַה שֶׁקָּרָה לְאֶחָד כְּבָר כָּלָה בְּהֶעְדֵּרוֹ מִמֶּנּוּ, וּמַה שֶּׁיִּקְרֶה לְעֶצֶם אַחֵר – דָּבָר חָדָשׁ וְאֵינוֹ הָרִאשׁוֹן.


פֶּרֶק שְׁלִישִׁי

הַכַּמּוּת הַדָּבֵק וְהַנִּפְרָד. הָאֵיכוּת וְאַרְבַּעַת מִינָיו. הַקִּנְיָן וּשְׁלשֶׁת מִינָיו. הָעִיּוּנִי וּשְׁלֹשֶׁת מִינָיו. הַטֶּבַע וּמַה שֶׁאַחַר הַטֶּבַע. הַתְּכוּנוֹת. הַמּוּסָר, הַמְּדִינוּת וְהַפַּרְנָסָה. הַמְּלָאכוֹת הַחָפְשִיּוֹת וּבִלְתִּי חָפְשִׁיּוֹת. הַהֲכָנָה הַטִּבְעִית. הַתְּכוּנָה הָרִאשׁוֹנִית וְהַשֵּׁנִית. הַצּוּרָה הַטִּבְעִית וְהַמְּלַאכְתִּית. כְּלָלֵי הָאֵיכוּת. הַפְּעֻלָּה וּכְלָלֶיהָ. הַהֶפְעֵל וּכְלָלָיו. הַיַּחַס וּכְלָלָיו. הַזְּמַן, הָאָנָה, הַמַּצָּב וְהַקִּנְיָן. הַפְּרָטִים וְהַכּוֹלְלִים. הַסּוּג וְהַמִּין. סוּג עֶלְיוֹן וְאֶמְצָעִי, רָחוֹק וְקָרוֹב. כְּלָלֵי הַסּוּג וּכְלָלֵי הַמִּין. הַהֶבְדֵּל וּכְלָלָיו. הַסְּגֻלָּה וּכְלָלֶיהָ וְהַמִּקְרֶה.


א כְּבָר בֵּאַרְנוּ הַמִּקְרֶה בִכְלָלוֹ, עַתָּה נְבָאֵר חֲלָקָיו.

ב חֶלְקֵי הַמִּקְרֶה תִּשְׁעָה, וְהֵם: כַּמּוּת, אֵיכוּת, פְּעֻלָּה, הֶפְעֵל, יַחַס, זְמַן, מַצָּב, קִנְיָן, אָנָה.

ג כַּמּוּת הוּא מִקְרֶה יְשֹׁעַר בּוֹ הַדָּבָר וּבַעֲבוּרוֹ יִשְׁאַל כַּמָּה, וְיִתְחַלֵּק לִשְׁנֵי חֲלָקִים: הָאֶחָד – הַמִּדָּה שֶׁקְּרָאוּהָ כַּמּוּת דָּבֵק, וְהַשֵּׁנִי – הַמִּסְפָּר שֶׁקְּרָאוּהָ כַּמּוּת נִפְרָד.

פֵּרוּשׁוֹ: כִּי רִבּוּי הַחֲלָקִים וְהִדָּבְקָם זֶה לָזֶה עוֹשִׂים גּוּף בַּעַל שִעוּר שֶׁיּוּכַל לִמָּדֵד, וְזֶה מַה שֶׁכִּוְנוּ בְּאָמְרָם: כַּמּוּת דָּבֵק. וְרִבּוּיָם בְּפֵרוּד זֶה מִזֶּה מוֹלִיד הַמִּסְפָּר, וְזֶה מַה שֶׁכִּוְנוּ בְּקָרְאָם אוֹתוֹ: כַּמּוּת נִפְרָד.

ד אֵיכוּת – מִקְרֶה, בּוֹ יֵאָמֵר עַל הַדָּבָר מָה וְאֵיךְ, יִתְחַלֵּק לְאַרְבָּעָה מִינִים: הַקִּנְיָן וְהַהֲכָנָה הַטִּבְעִית, הַתְּכוּנָה וְהַצּוּרָה.

ה הַקִּנְיָן הוּא אֵיכוּת אֶחָד בְּאָדָם, שֶׁעַל יָדוֹ הוּא רָאוּי לְאֵיזֶה פְּעֻלּוֹת, וְיִתְחַלֵּק לִשְׁלֹשָׁה מִינִים: עִיּוּנִי, מַעֲשִׂיִּי וּמְלָאכוּתִיִּי.

ו הָעִיּוּנִי הוּא מַה שֶּׁעִנְיָנוֹ יְדִיעָה וְהַשָּׂגָה, וּמִינָיו שְׁלֹשָׁה: הַטֶּבַע, וּמַה שֶּׁאַחַר הַטֶּבַע, וְהַתְּכוּנוֹת.

ז הַטֶּבַע הוּא יְדִיעַת הַדְּבָרִים הַטִּבְעִיִּים בִּבְחִינַת טִבְעִיִּים וּמֻשָּׂג מֵהַחוּשִׁים; מַה שֶּׁאַחַר הַטֶּבַע – יְדִיעַת הַדְּבָרִים שֶׁלְּמַעְלָה מִכָּל הַחוּשִׁים וְלֹא מֻשָּׂגִים מֵהֶם. הַתְּכוּנוֹת – נִכְלָלִים בָּהֶם חָכְמַת הַגַּלְגַּלִּים, הַתִּשְׁבֹּרֶת, הַהַנְדָּסָה, הַמַּבָּט, הַנִּגּוּן, וְכַיּוֹצֵא.

ח הַמַּעֲשִׂיִּי הוּא יְדִיעַת תִּקּוּן מַעֲשֶׂה, וּמִתְחַלֵּק לִשְׁלֹשָׁה מִינִים, וְהֵם: הַמּוּסָר, הַמְּדִינוּת, וְהַפַּרְנָסָה.

ט הַמְּלַאכְתִּיִּי הוּא יְדִיעַת מְלָאכָה אַחַת, וּמִתְחַלֵּק לִשְׁנַיִם: מְלָאכוֹת חָפְשִׁיּוֹת, וְהַשֵּׁנִי – הַלֹּא חָפְשִׁיּוֹת.

י הַחָפְשִׁיּוֹת הֵם הָרְאוּיוֹת לְשָׂרִים וְנִכְבָּדִים, וְהֵם שְׁלֹשָׁה: הַדִּקְדּוּק, הֲלָצָה, הִגָּיוֹן. הַלֹּא הַחָפְשִׁיּוֹת הֵן הַבִּלְתִּי הָרְאוּיוֹת לְשָׂרִים וְנִכְבָּדִים, כְּגוֹן: הַתְּפִירָה, הַבִּנְיָן, הָאֲרִיגָה וְכַיּוֹצֵא.

יא הַהֲכָנָה הַטִּבְעִית הַנִּמְצָא בַנּוֹשֵׂא מִצַּד טִבְעוֹ, כְּגוֹן: חִדּוּד הַשֵּׂכֶל וְהַזִּכָּרוֹן, חֹזֶק הַגּוּף וּבְרִיאוּתוֹ, קַלּוּת הַתְּנוּעָה.

יב הַתְּכוּנָה הִיא מַה שֶׁבְּתֹכֶן הַדָּבָר וּמִזְגּוֹ, וְתִתְחַלֵּק לִשְׁנַיִם: הָאֶחָד – רִאשׁוֹנִית, וְהַשֵּׁנִי – שֵׁנִית.

יג הָרִאשׁוֹנִית מִתְחַלֵּק לְאַרְבָּעָה, וְהֵם: הַחֹם וְהַקֹּר, הַלַּחוּת וְהַיֹּבֶשׁ, וְהַשֵּׁנִית – כְּגוֹן: הַקֹּשִׁי וְהָרֹךְ, הַקַּלּוּת וְהַכֹּבֶד, וְכַיּוֹצֵא.

יד הַצּוּרָה הִיא צוּרַת הַדָּבָר שֶׁבָּהּ נִרְאֶה וּמִצְטַיֵּר בַּשֵּׂכֶל, וּמִתְחַלֶּקֶת לִשְתַּיִם: הָאַחַת – טִבְעִית, וְהַשְּׁנִיָּה – מְלַאכְתִּית.

טו הַטִּבְעִית – צוּרַת הָאָדָם, צוּרַת הַסּוּס, וְכֵן שְׁאָר הַמִּינִים; הַמְּלַאכְתִּית – צוּרַת הַנַּעֲשֶׂה בִּידֵי אָדָם.


כְּלָלֵי הָאֵיכוּת:

1 בְּאֵיכוּת שַׁיָּךְ פָּחוֹת וְיָתֵר.

2 בֵּין הָאֵיכֻיּוֹת יֵשׁ הַהַפְכִּיּוּת.

דֶּרֶךְ מָשָׁל: חֹם וָקֹר.

3 מִצַּד הָאֵיכוּת יֵאָמֵר בִּדְבָרִים דּוֹמִים וְלֹא דוֹמִים.

טז פְּעֻלָּה הוּא הִתְעוֹרְרוּת הָעֶצֶם אוֹ גִלּוּי וִיצִיאַת כֹּחַ הַפּוֹעֵל שֶׁבָּעֶצֶם. (אוֹ תֹאמַר: הֶמְשֵׁךְ הַמְּסֻבָּב מִן הַסִּבָּה הַפּוֹעֶלֶת, וּכְמוֹ שֶׁאָמַרְנוּ לְעֵיל. דֶּרֶך מָשָׁל: הַחִמּוּם הוּא הֶמְשֵׁךְ הַחֹם מֵהָאֵשׁ).


כְּלָלֵי הַפְּעֻלָּה:

1 כָּל פּוֹעֵל בְּרָצוֹן פּוֹעֵל לְתַכְלִית מָה.

2 בִּפְעֻלּוֹת יֵשׁ פָּחוֹת וְיָתֵר.

3 בִּפְעֻלָּה שַׁיָּךְ הַפְכִּיּוּת.

דֶּרֶךְ מָשָׁל: מְחַמֵּם – מְקָרֵר.

יז הַהֶפְעֵל הוּא רֹשֶׁם הַפְּעֻלָּה.


כְּלָלֵי הַהֶפְעֵל:

1 בְּהֶפְעֵל יֵשׁ פָּחוֹת וְיָתֵר.

2 בְּהֶפְעֵל שַׁיָּךְ הַפְכִּיּוּת.

יח יַחַס הוּא מִין קֶשֶׁר בֵּין שְׁנֵי דְבָרִים, בַּעֲבוּרוֹ יֵאָמֵר בְּאֶחָד מֵהֶם שֶׁהוּא שֶׁל חֲבֵרוֹ.

פֵּרוּשׁוֹ: כְּגוֹן אָב שֶׁהוּא עִנְיָן מִתְיַחֵס אֵצֶל בֵּן וּבֵן הוּא עִנְיָן מִתְיַחֵס אֵצֶל אָב. כִּי אֵין אָב אֶלָּא שֶׁיֵּשׁ לוֹ בֵּן וְאֵין בֵּן אֶלָּא שֶׁיֵּשׁ לוֹ אָב.


כְּלָלֵי הַיַּחַס:

1 מִקְרֵה הַיַּחַס גּוֹבֵר בְּרִבּוּיוֹ בְּכָל הַמִּקְרִים.

2 שְׁנֵי הַמִּתְיַחֲסִים שָׁוִים בְּהֶכְרַח בְּהֶעְדֵּר וּבִמְצִיאוּת.

פֵּרוּשׁוֹ: בְּהִמָּצֵא אֶחָד מְצִיאוּת, מֻכְרָח הַשֵּׁנִי. וְכֵן בְּהֵעָדֵר הָאֶחָד, מֻכְרָח בִּטּוּל הַשֵּׁנִי. כִּי אֵין אָב אֶלָּא שֶׁיֵּשׁ לוֹ בֵּן, וְאִם יֵשׁ אָב וַדַּאי יֵשׁ בֵּן, וְאִם אֵין אָב אֵין בֵּן.

יט זְמָן – מִקְרֶה זְמַנִּיִּי בָּעֶצֶם, דְּהַיְנוּ: שִעוּר מֶרְחָקוֹ מִן הַתְּחִלָּה אוֹ הַסּוֹף.

כ אָנָה – מִקְרֶה מְקוֹמִי בָּעֶצֶם, דְּהַיְנוּ: הַחֵלֶק מִן הָעוֹלָם שֶׁמְּמֻלָּא מִמֶּנּוּ.

כא מַצָּב – מִקְרֶה מַצָּבִי בָּעֶצֶם.

פֵּרוּשׁוֹ: אֹפֶן הִמָּצֵא חֶלְקֵי הָעֶצֶם בְּמָקוֹם – זֶהוּ הַנִּקְרָא מַצָּב, כְּגוֹן הָעֲמִידָה, הַיְשִׁיבָה, הַנְּטִיָּה, וְכַיּוֹצֵא.

כב קִנְיָן – מִקְרֶה מוֹרֶה מִין קְנִיָּה נִקְנֵית בָּעֶצֶם, אוֹ הַלְבָּשָׁה עָלָיו.

דֶּרֶךְ מָשָׁל: הַמַּלְבּוּשִׁים שֶׁעַל הָאָדָם, אוֹ הַמִּדּוֹת הַקְּנוּיוֹת בְּנַפְשׁוֹ – הָעֲנָוָה, הַנְּדִיבוּת, וְכַיּוֹצֵא.

כג אֵלֶּה הָעֲשָׂרָה כְּלָלֵי עִנְיָנִים שֶׁזָּכַרְנוּ – אֶחָד עֶצֶם וְתִשְׁעָה מִקְרִים – תַּחְתֵּיהֶם נִכְלָלִים כָּל עִנְיְנֵי הַמְּצִיאוּת לְמַדְרֵגוֹתֵיהֶם.

כד וְהַנִּדְרָשִׁים בַּעֲשָׂרָה הָעִנְיָנִים שֶׁזָּכַרְנוּ – שְׁנַיִם, וְהֵם: הַפְּרָטִים וְהַכּוֹלְלִים.

כה הַפְּרָטִים הֵם הַנִּמְצָאִים כָּל אֶחָד בִּפְנֵי עַצְמוֹ, כְּגוֹן: אִישׁ מִן הָאֲנָשִׁים, עוֹף מִן הָעוֹפוֹת, אִילָן מִן הָאִילָנוֹת, וְכַיּוֹצֵא בָזֶה, וְנִקְרָאִים אִישִׁים.

כו הַכּוֹלְלִים – שְׁנַיִם, רִאשׁוֹן: כּוֹלְלֵי הַמְּצִיאוּת, וְהַשֵּׁנִי: מַשִּׂיגֵיהֶם.

כז כּוֹלְלֵי הַמְּצִיאוּת גַּם הֵם שְׁנַיִם, הָרִאשׁוֹן – סוּג, וְהַשֵּׁנִי – מִין.

כח הַסּוּג וְהַמִּין עִנְיָנָם אֶחָד, וְהוּא כְּלָל כּוֹלֵל פְּרָטִים מִצַּד שִׁתּוּף שֶׁהֵם מְשֻתָּפִים בְּעִנְיָן אֶחָד. וְהַהֶפְרֵשׁ שֶׁבֵּין הַסּוּג וְהַמִּין, שֶׁמַּה שֶׁכּוֹלֵל יוֹתֵר – יִקָּרֵא סוּג, וּמַה שֶּׁאֵינוֹ כוֹלֵל אֶלָּא מִקְּצָת מִמָּה שֶׁנִּכְלָל בַּסּוּג – יִקָּרֵא מִין.

דֶרֶךְ מָשָׁל: בַּעֲלֵי חַיִּים הוּא סוּג, כִּי נִכְלָלִים בָּזֶה הַחַיּוֹת וְהָעוֹפוֹת, הַדָּגִים וּבְנֵי אָדָם מִצַּד הַשִּתּוּף שֶׁיַּשׁ בֵּינֵיהֶם בְּעִנְיָן הַחִיּוּת, שֶׁכֻּלָּם חַיִּים. אַךְ כְּלָל בְּנֵי הָאָדָם יִקָּרֵא מִין, כִּי גַם הוּא כוֹלֵל כָּל אִישֵׁי בְנֵי הָאָדָם, דְּהַיְנוּ: רְאוּבֵן, שִׁמְעוֹן וְכו', וְאֵינוֹ כוֹלֵל הַחַיּוֹת וְהָעוֹפוֹת וְהַדָּגִים, שֶׁנִּכְלְלוּ בְּשֵׁם בַּעֲלֵי חַיִּים.

כט הַסּוּג מִתְחַלֵּק לְעֶלְיוֹן אוֹ אֶמְצָעִי, לְרָחוֹק וְקָרוֹב.

ל סוּג הָעֶלְיוֹן הוּא מַה שֶּׁאֵין סוּג אַחֵר עָלָיו, פֵּרוּשׁוֹ: שֶׁכּוֹלֵל יוֹתֵר מִמֶּנּוּ; הָאֶמְצָעִי הוּא שֶׁיֵּשׁ סוּג כּוֹלֵל מִמֶּנּוּ שֶׁלְּגַבָּיו יִקָּרֵא סוּג הָאֶמְצָעִי – מִין, וּלְגַבֵּי הַמִּין שֶׁתַּחְתָּיו יִקָּרֵא סוּג.

לא סוּג רָחוֹק הוּא – שֶׁרָחוֹק מִן הַמִּין, וְקָרוֹב – שֶׁקָּרוֹב לוֹ.

פֵּרוּשׁוֹ: סוּג הַסּוּג הוּא רָחוֹק מִן הַמִּין, כִּי יֵשׁ מַדְרֵגָה אַחֶרֶת בֵּינוֹ לַמִּין.

דֶּרֶךְ מָשָׁל: הַתַּרְנְגֹל הוּא מִין מִמִּינֵי הָעוֹפוֹת, עוֹף הוּא סוּג לְמִינֵי הָעוֹפוֹת, בַּעֲלֵי חַיִּים הוּא סוּג הַסּוּג לַתַּרְנְגֹל וְנִקְרָא סוּג רָחוֹק, וְעוֹף נִקְרָא סוּג קָרוֹב.


כְּלָלֵי הַסּוּג:

1 הַסּוּג גָּדוֹל מִכָּל מִין וּמִין שֶׁתַּחְתָּיו.

פֵּרוּשׁוֹ: כִּי הַסּוּג הוּא מַה שֶׁכּוֹלֵל מִינִים הַרְבֵּה.

2 הַסּוּג אֵינוֹ מִתְפָּרֵד מִמִּינָיו.

פֵּרוּשׁוֹ: כִּי אֵין הַסּוּג אֶלָּא הַחֹק הַכְּלָלִי הַשָּׁוֶה לְמִינָיו, וְנִמְצָא שֶׁאִי אֶפְשָׁר לְשׁוּם מִין לִהְיוֹת בְּלֹא הַחֹק הַזֶּה.

3 מַה שֶּׁיִּקְרֶה לַסּוּג יִקְרֶה לְכָל מִין.

4 כְּשֶׁיֻּנַּח סוּג יֻנְּחוּ בְּהֶכְרַח מִינִים, וּבְהֵעָדֵר הַסּוּג נֶעְדָּרִים כֻּלָּם.


כְּלָלֵי הַמִּין:

1 הַמִּין נוֹלָד מֵהִגָּבֵל הַסּוּג עַל יְדֵי אֵיזֶה הֶבְדֵּלִים שֶׁבָּהֶם נִבְחָן מִין מִמִּין אַחֵר.

דֶּרֶךְ מָשָׁל: מִסּוּג הַבַּעֲלֵי חַיִּים יִוָּלֵד מִין הָאָדָם בְּחַבֵּר אֵלָיו הֶבְדֵּל הַדִּבּוּר, וְנֹאמַר: בַּעַל חַי מְדַבֵּר.

2 הַמִּין מְאֻחָר בַּטֶּבַע אֶל סוּגוֹ.

3 כְּשֶׁיֻּנַּח מִין בְּהֶכְרַח יֻנַּח הַסּוּג, וּבְהֵעָדֵר מִין אֶחָד אֵין הֶעְדֵּר הַסּוּג מֻכְרָח.

לב מַשִּׂיגֵי הַכּוֹלְלִים שְׁלשָׁה: הֶבְדֵּל, סְגֻלָּה וּמִקְרֶה.

לג הֶבְדֵּל הוּא עִנְיָן עַצְמִיִּי שֶׁבּוֹ יִבָּחֲנוּ זֶה מִזֶּה שְׁנַיִם שֶׁיֵּשׁ לָהֶם שִׁתּוּף בַּסּוּג אוֹ בַמִּין. נִמְצָא הַסּוּג וְהַמִּין מוֹדִיעִים מֵעֲצָמוֹת הַנִּמְצָא מַה שֶׁיְּשֻׁתַּף בּוֹ עִם זוּלָתוֹ, וְהַהֶבְדֵּל מוֹדִיעַ מִמֶּנּוּ מַה שֶׁיִּבָּדֵל מִזּוּלָתוֹ

דֶּרֶךְ מָשָׁל: הָאָדָם וְהַבְּהֵמָה מְשֻׁתָּפִים בַּחִיּוּת, וְהֶבְדֵּל הָאָדָם מִן הַבְּהֵמָה הוּא הַדִּבּוּר. וְנִקְרָא הָאָדָם בַּעַל חַי מְדַבֵּר, וְהַבְּהֵמָה – בַּעַל חַי בִּלְתִּי מְדַבֵּר.


כְּלָלֵי הַהֶבְדֵּל:

הַהֶבְדֵּל הוּא עִנְיָן עַצְמִי מְיֻחָד בַּסּוּג אוֹ בְמִינוֹ, וְנִמְצָא בוֹ בְּרֻבּוֹ וּבְכָל זְמָן.

דֶּרֶךְ מָשָׁל: הַדִּבּוּר הוּא עַצְמִי לְמִין הָאָדָם שֶׁזּוּלָתוֹ לֹא הָיָה אָדָם, וְנִמְצָא בְּרֹב הַמִּין וּבְכָל זְמָן.

לד הַסְּגֻלָּה הוּא הֶבְדֵּל בִּלְתִּי עַצְמִיִּי, דְּהַיְנוּ שֶׁמּוֹדִיעַ מֵהַנִּמְצָא מַה שֶׁנִּבְחָן בּוֹ מִזּוּלָתוֹ, אַךְ אֵינוֹ מֵעַצְמוּתוֹ.

דֶּרֶךְ מָשָׁל: הַשְּׂחוֹק לְמִין הָאָדָם, כִּי אֵינוֹ נִמְצָא אֶלָּא בוֹ, אַךְ אֵינוֹ מֵעַצְמוּתוֹ, כִּי אֵין הֱיוֹתוֹ אָדָם תָּלוּי בָּזֶה, שֶׁבְּלֹא זֶה הָיָה אָדָם.


כְּלָלֵי הַסְּגֻלָּה:

הַסְּגֻלָּה מְיֻחֶדֶת בַּמִּין אוֹ בַסּוּג, וְנִמְצָא בְרֻבּוֹ וּבְכָל זְמָן, וְאֵינָה עַצְמִית בּוֹ.

לה הַמִּקְרֶה הוּא עִנְיָן בְּנִמְצָא בִּלְתִּי מְיֻחָד בּוֹ וּבִלְתִּי עַצְמִי לוֹ.


כְּלָלֵי הַמִּקְרֶה:

הַמִּקְרֶה נִמְצָא בְּמִעוּט הַסּוּג אוֹ מִין, וְלֹא בְּכָל זְמָן, וְאֵינוֹ עַצְמִי בוֹ. שְׁאָר עִנְיְנֵי הַמִּקְרֶה נִתְבָּאֲרוּ לְמָעְלָה.


פֶּרֶק רְבִיעִי

שָׁלֹש פְּעֻלּוֹת הַשֵּׂכֶל. הַשֵּם הַמְּיֻחָד וְהַנִּרְדָּף. מְשֻׁתָּף מֻחְלָט. מְשֻׁתָּף מִתְיַחֵס: כּוֹלֵל, מֻשְׁאָל, מֻעְתָּק, מוּדָע. שֵׁמוֹת הַמְּשַׁמְּשִׁים לְהִגָּיוֹן. הַסִּבָּה וּכְלָלֶיהָ. הַפּוֹעֵל וְהַתַּכְלִית: הַפּוֹעֵל מִצַּד עַצְמוֹ וּפוֹעֵל בְּמִקְרֶה, הַפּוֹעֵל לְבַדּוֹ וּבְשֻׁתָּפוּת. הַפּוֹעֲלִים הַשְּׁקוּלִים וּבִלְתִּי שְׁקוּלִים, הַשֻּׁתָּפוּת הַמֻּכְרָח וּבִלְתִּי מֻכְרָח. הַסִּבָּה הַמְעוֹרֶרֶת וְהַכֵּלִיִּית, הַמּוֹלִידָה, הַמְקַיֶּמֶת, וְהַמַּפְסֶדֶת. הַפְּעֻלָּה הַטִּבְעִית וְהָרְצוֹנִית. הַסִּבָּה הָרְחוֹקָה וְהַסְּמוּכָה.


א עַד הֵנָּה בֵּאַרְנוּ עֵסֶק הַהִגָּיוֹן; עַתָּה נְבָאֵר חִלּוּקוֹ לְמַחְלְקוֹתָיו, וְהֵם שְׁלֹשָׁה: הָרִאשׁוֹן – בֵּאוּר הַשֵּׁמוֹת, הַשֵּׁנִי – בֵּאוּר הַמַּאֲמָרִים, הַשְּׁלִישִׁ – הַהֶקֵּשִׁים.

פֵּרוּשׁוֹ: כִּי פּעֻלּוֹת הַשֵּׂכֶל שְׁלֹשָׁה. הָרִאשׁוֹן – הַשָּׂגַת הַנּוֹשְׂאִים הַמִּזְדַּמְּנִים לְפָנָיו כְּמוֹ שֶׁמִּזְדַּמְּנִים.

דֶּרֶךְ מָשָׁל: הָאֶבֶן, הָעֵץ, הַבַּרְזֶל, וְכַיּוֹצֵא.

הַשֵּׁנִי – חַבֵּר אֶל נוֹשֵׂא מַה יְחֻיַּב בּוֹ אוֹ יְשׁוֹלַל מִמֶּנּוּ עַד שֶׁיִּבָּנֶה מִזֶּה הַמַּאֲמָר.

דֶּרֶךְ מָשָׁל: הָעֵץ הוּא גָדוֹל – חִיֵּב הַגֹּדֶל אֶל הָעֵץ, הַנְּיָר הַזֶּה אֵינוֹ לָבָן – שָׁלַל הַלֹּבֶן מִן הַנְּיָר.

הַשְּׁלִישִׁי הוּא – חַבֵּר מַאֲמָר אֶל מַאֲמָר וְהוֹלִיד מֵהֶם שְׁלֹשָׁה.

דֶּרֶךְ מָשָׁל: הָאָדָם הוּא בַּעַל חַי; רְאוּבֵן הוּא אָדָם, אִם כֵּן הוּא בַּעַל חַי. לַפְּעֻלָּה הָרִאשׁוֹנָה מִשְׁתַּמֵּשׁ מִן הַשֵּׁמוֹת, לַשֵּׁנִי – מִן הַמַּאֲמָרִים, לַשְּׁלִישִׁי – מִן הַהֶקֵּשִׁים.

פֵּרוּשׁוֹ: חִבּוּר הַמַּאֲמָרִים וְהוֹלָדַת תּוֹלַדְתָּם.

ב הַשֵּׁם הוּא מוֹרֶה עַל אֶחָד מֵהָעֲצָמִים אוֹ אֶחָד מִן הַמִּקְרִים וְיִתְחַלֵּק לְפִי הוֹרָאָתוֹ וּלְפִי עִנְיָנוֹ.

לְפִי הוֹרָאָתוֹ חֲלָקָיו שְׁלֹשָׁה: מְיֻחָד, נִרְדָּף, מְשֻׁתָּף.

פֵּרוּשׁוֹ: כִּי יֵשׁ מִן הַשֵּׁמוֹת שֶׁיּוֹרוּ עַל עִנְיָן בְּדֶרֶךְ פָּשׁוּט, וְיֵשׁ שֶׁיּוֹרוּ בְּדֶרֶךְ דִּמְיוֹן, וְיֵשׁ שֶׁיּוֹרוּ בְּדֶרֶךְ הַסְכָּמָה, וּכְמָה שֶׁאֶכְתֹּב לְפָנִים. וּמִצַּד זֶה יִבָּחֲנוּ הַשֵּׁמוֹת לְפִי הוֹרָאָתָן.

אָמְנָם עוֹד יִבָּחֲנוּ הַשֵּׁמוֹת מִצַּד הָעִנְיָנִים הַנִּרְמָזִים בָּם, וּכְמָה שֶׁנִּכְתֹּב לְפָנִים. וְהִנֵּה מִצַּד הַהוֹרָאָה יִתְחַלֵּק לְאֵלֶּה הַשְּׁלֹשָׁה: מְיֻחָד וכו'.

ג הַמְיֻחָד הוּא – שֶׁהוּא לְבַדּוֹ מוֹרֶה עַל עִנְיָן אֶחָד לְבָד; הַנִּרְדָּף הוּא – שֶׁיֵּשׁ שֵׁמוֹת רַבִּים מוֹרִים עַל עִנְיָן אֶחָד, כָּל אֶחָד מֵהֶם נִקְרָא נִרְדָּף.

מְשַׁל הַמְיֻחָד – שַׁם הָעֶצֶם, כְּגוֹן: אַבְרָהָם, מֹשֶׁה, יְרוּשָׁלַיִם.

שֵׁם הַיַּחַס, כְּגוֹן: עִבְרִי, מִצְרִי, וְכַיּוֹצֵא.

שֵׁם הַמִּסְפָּר: אֶחָד, שְׁנַיִם.

מְשַׁל הַנִּרְדָּף: לֶחֶם וּפַת.

ד הַמְּשֻתָּף הוּא שֵׁם אֶחָד שֶׁמּוֹרֶה עַל עִנְיָנִים רַבִּים וּמִתְחַלֵּק לִשְׁנַיִם: לִמְשֻׁתָּף מֻחְלָט וּמְשֻׁתָּף מִתְיַחֵס.

ה מְשֻׁתָּף מֻחְלָט הוּא – שֶׁמּוֹרֶה עַל שְׁנֵי עִנְיָנִים שֶׁאֵין בֵּינֵיהֶם שׁוּם יַחַס וְדִמְיוֹן.

דֶּרֶךְ מָשָׁל: סַף, שֶׁרוֹצֶה לוֹמַר – כְּלִי, וְרוֹצֶה לוֹמַר – מְזוּזָה.

מְשֻׁתָּף מִתְיַחֵס – שֶׁמּוֹרֶה עַל שְׁנֵי עִנְיָנִים שֶׁיֵּשׁ בֵּינֵיהֶם אֵיזֶה יַחַס וְדִמְיוֹן, וְיִתְחַלֵּק לְאַרְבָּעָה: כּוֹלֵל, מֻשְׁאָל, מֻעְתָּק, מוּדָע.

ו הַכּוֹלֵל הוּא שֵׁם שֶׁיּוֹרֶה עַל עִנְיָנִים רַבִּים, כָּל אֶחָד מֵהֶם רָאוּי לָהֶם זֶה הַשֵּׁם, כְּמוֹ שֶׁרָאוּי לַחֲבֵרוֹ בְּשָׁוֶה, מִצַּד הֱיוֹתָם כֻּלָּם עֲנָפִים מִשֹּׁרֶשׁ אֶחָד, וְהַיְנוּ הַמִּינִים שֶׁתַּחַת הַסּוּג, וְהָאִישִׁים שֶׁתַּחַת הַמִּין.

דֶּרֶךְ מָשָׁל: רְאוּבֵן, שִׁמְעוֹן, לֵוִי – כָּל אֶחָד מֵהֶם יִקָּרֵא אָדָם, מִפְּנֵי הֱיוֹתָם אִישִׁים מִמִּין הָאָדָם, וְנִקְרָא שֵׁם אָדָם – שֵׁם כּוֹלֵל.

ז הַמֻּשְׁאָל הוּא שֵׁם שֶׁיּוֹרֶה עַל עִנְיָן אֶחָד בְּהוֹרָאָה רִאשׁוֹנָה, וְעַל עִנְיָן אַחֵר בְּהוֹרָאָה שְׁנִיָּה, מִפְּנֵי שֶׁיֵּשׁ לוֹ דִּמְיוֹן מָה עִם הָעִנְיָן הָרִאשׁוֹן.

פֵּרוּשׁוֹ: הוֹרָאָה רִאשׁוֹנָה נִקְרֵאת מַה שֶּׁמִּשְׁתַּמְּשִׁים מֵהַמִּלָּה בַּלָּשׁוֹן עַל הָרֹב, וְהוֹרָאָה שֵׁנִית – מַה שֶׁמִּשְׁתַּמְּשִׁים מִמֶּנָּה לִפְעָמִים.

דֶּרֶךְ מָשָׁל: אֵשׁ, הוֹרָאָה רִאשׁוֹנָה שֶׁלּוֹ הִיא עַל הָעֶצֶם הַשּׂוֹרֵף הַיָּדוּעַ, וּבְהוֹרָאָה שֵׁנִית יוֹרֶה עַל מִי שֶׁהוּא מָהִיר הַרְבֵּה בִּתְנוּעָתוֹ וְאֵינוֹ שׁוֹקֵט – נִקְרָאֵהוּ אֵשׁ לַדִּמְיוֹן שֶׁיֵּשׁ לוֹ עִם הָאֵשׁ הָאֲמִתִּי, שֶׁמִּתְנוֹעֵעַ בִּמְהִירוּת וְאֵינוֹ שׁוֹקֵט.

ח הַמֻּעְתָּק הוּא שֵׁם שֶׁיּוֹרֶה עַל שְׁנֵי עִנְיָנִים בְּשָׁוֶה מִפְּנֵי דִמְיוֹן שֶׁבֵּינֵיהֶם, עַד שֶׁלֹּא נוּכַל לוֹמַר עַל שׁוּם אֶחָד מֵהֶם הֱיוֹת הוֹרָאָה שְׁנִיָּה בוֹ. פֵּרוּשׁוֹ: שֶׁכְּבָר נָהֲגוּ כְּמוֹ כֵן בַּלָּשׁוֹן שֶׁיּוֹרֶה עַל שְׁנֵי הָעִנְיָנִים בְּשָׁוֶה.

דֶּרֶךְ מָשָׁל: הַגָּדֵר, כָּךְ יוּבַן עַל הַגָּדֵר הַבִּנְיָנִיִּי הֶעָשׂוּי סְבִיב הַשָּׂדוֹת, כְּמוֹ שֶׁיּוּבַן עַל בֵּאוּר הַשָּׁלֵם וְהַצִּיּוּר הָאֲמִתִּי אֵצֶל הַהֶגְיוֹנִים. וְכֵן הֵם כָּל שְׁמוֹת הַדִּקְדּוּק וְהַהִגָּיוֹן וּשְׁאָר הַחָכְמוֹת, שֶׁלָּקְחוּ שֵׁמוֹת לְבָאֵר מִשְׁפְּטֵיהֶם מִצַּד הַדִּמְיוֹן אֲשֶׁר לָהֶם עִם מַה שֶׁתּוֹרֶינָה הַמִּלּוֹת בְּהוֹרָאָתָן הָרִאשׁוֹנָה.

ט הַמּוּדָע הוּא שֵׁם נֶאֱמָר עַל רַבִּים בְּשָׁוֶה וּבְיִחוּד עַל אֶחָד, לִהְיוֹת הָעִנְיָן הַהוּא חָזָק בּוֹ מֵהָאֲחֵרִים. וְהַדָּבָר מוּדָע לַכֹּל, שֶׁהַכַּוָּנָה בַּשֵּׁם הַהוּא עַל עִנְיָן הַפְּרָטִי הַהוּא.

דֶּרֶךְ מָשָׁל: הַגִּבּוֹר סְתָם יֵאָמֵר עַל שִׁמְשׁוֹן, אַף עַל פִּי שֶׁשֵּׁם זֶה שַׁיָּךְ לְכָל גִּבּוֹר, לְפִי שֶׁהִפְלִיג בַּגְּבוּרָה; מְשׁוֹרֵר נֶאֱמַר עַל דָּוִד, אַף עַל פִּי שֶׁיֵּשׁ מְשׁוֹרְרִים רַבִּים, לְפִי שֶׁהִפְלִיג בַּשִּׁיר.

י הַשֵּׁמוֹת לְפִי הָעִנְיָנִים הַנִּרְמָזִים בָּם יִתְחַלְּקוּ לִשְׁנַיִם, לְדִבּוּרִיִּים וְלִמּוּדִיִּים. הַדִּבּוֹרִיִּים הֵם הַמְּשַׁמְּשִׁים לַדִּבּוּר, לְהַזְכִּיר בָּם הַמְּדַבֵּר מַה שֶּׁבְּמַחֲשַׁבְתּוֹ מִן הָעִנְיָנִים הַנִּמְצָאִים; הַלִּמוּדִיִּים הֵם הַמְיֻחָדִים לַחָכְמוֹת אוֹ לַמְּלָאכוֹת שֶׁעַל יְדֵיהֶם מַגִּיעִים לְתַכְלִית הַמְכֻוָּן בַּחָכְמוֹת אוֹ מְלָאכוֹת הָהֵם, וְהִם כ"ב מִינִים, כְּמִינֵי הַחָכְמוֹת וְהַמְּלָאכוֹת.

יא הַשֵּׁמוֹת הַמְּשַׁמְּשִׁים לְהִגָּיוֹן הֵם אֶחָד וְעֶשְׂרִים, וְאֵלֶּה הֵם: סִבָּה וּמְסֻבָּב, נוֹשֵׂא וּמִתְחַבֵּר, כֹּל וָחֵלֶק, גְּזֵרָה וְנִגְזָר, גָּדֵר וְנִגְדָּר, בִּנְיָן וְנִבְנֶה, חִלּוּק וְנֶחֱלָק, נֶעֱרָכִים, בִּלְתִּי שָׁוִים, הַפְכִּיִּים, עֵדוּת וּמוּעָד, סוּג וָמִין.

פֵּרוּשׁוֹ: כִּי אֵלֶּה הֵם הַשֵּׁמוֹת שֶׁבָּהֶם נַגִּיעַ לַתַּכְלִית הַמְכֻוָּן בַּהִגָּיוֹן, שֶׁהוּא לָדַעַת הַדְּבָרִים לַאֲמִתָּם בְּכָל הַהַבְחָנוֹת שֶׁיֵּשׁ לְהַבְחִין בָּהֶם.

יב הַסִּבָּה הִיא הַדָּבָר שֶׁמִּכֹּחוֹ נִמְצָא הַמְסֻבָּב.


כְּלָלֵי הַסִּבָּה:

1 הַסִּבָּה קוֹדֶמֶת לִמְסֻבָּבָהּ תָּמִיד לְפָחוֹת קְדִימָה טִבְעִית.

פֵּרוּשׁוֹ, כִּי הַקְּדִימוֹת הֵן שְׁתַּיִם, קְדִימָה זְמַנִּית וּקְדִימָה טִבְעִית. הַזְּמַנִּית הִיא קְדִימָה בִּזְמַן, וְהַטִּבְעִית הִיא – שֶׁאַף שֶׁיִּהְיֶה הַקּוֹדֵם הַטִּבְעִי יַחַד בִּזְמַן עִם מְאֻחָרוֹ, הִנֵּה תֻשְׂכַּל לוֹ קְדִימָה בַשֵּׂכֶל, שֶׁאִם הוּא לֹא הָיָה נִמְצָא לֹא הָיָה מְצִיאוּת לִמְאֻחָרוֹ, וְזֹאת הִיא הַנִּמְצֵאת תָּמִיד לְסִבָּה לְגַבֵּי מְסֻבָּבָהּ, אַךְ בְּרֹב הַדְּבָרִים תִּקְדֹּם הַסִּבָּה לִמְסֻבָּבָהּ גַּם בִּזְמַן.

דֶּרֶךְ מָשָׁל: זְרִיחַת הַשֶּׁמֶשׁ וְאוֹר הַיּוֹם הֹוִים בִּזְמַן אֶחָד. אָמְנָם, בִּהְיוֹת שֶׁמְּצִיאוּת אוֹר הַיּוֹם תָּלוּי בִּמְצִיאוּת זְרִיחַת הַשֶּׁמֶשׁ, נִקֱרָא לָהּ קוֹדֶמֶת, וּלְאוֹר הַיּוֹם – מְאֻחָר בַּטֶּבַע.

2 אֵין דָּבָר נִמְצָא בְּלֹא סִבָּה שֶׁתַּמְצִיאֵהוּ.

3 אֵין בַּסִּבּוֹת הֶמְשֵׁךְ לְאֵין תַּכְלִית.

פֵּרוּשׁוֹ, כִּי הַסִּבָּה הָעֶלְיוֹנָה הוּא הַבּוֹרֵא, יִתְבָּרֵךְ שְׁמוֹ, אֲשֶׁר אֵין סִבָּה לְמַעְלָה מִמֶּנּוּ.

4 אִם תֻּנַּח סִבָּה בַּפּוֹעֵל יֻנַּח הַמְּסֻבָּב גַּם כֵּן, אַךְ בַּהֲסָרָתָהּ לֹא יוּסַר אֶלָּא הַמְּסֻבָּב, הֶעָתִיד, לֹא הֶעָבָר.

פֵּרוּשׁוֹ: דֶּרֶךְ מָשָׁל, אֵשׁ שֶׁחִמֵּם אֶת הַבַּרְזֶל, אַף אִם יִכְלֶה הָאֵשׁ לֹא יָסוּר הַחִמּוּם מֵהַבַּרְזֶל, אַךְ לֹא יִוָּלֵד מִמֶּנּוּ חִמּוּם אַחֵר.

וּלְפָנִים נְבָאֵר עוֹד שֶׁזֶּהוּ בַּסִּבָּה הַסְּמוּכָה לְבָד.

יג הַסִּבָּה נֶחֱלֶקֶת לְחִיצוֹנָה וּפְנִימִית. הַחִיצוֹנָה הִיא שֶׁחוּץ לָעֶצֶם הַמְּסֻבָּב. וְתֵחָלֵק גַּם כֵּן לִשְׁנַיִם, לְפוֹעֵל וְתַכְלִית.

יד הַפּוֹעֵל הוּא שֶׁעַל יָדוֹ הַנִּמְצָא הֹוֶה, וּמִינָיו שְׁנַיִם: פּוֹעֵל מִצַּד עַצְמוֹ וּפוֹעֵל בְּמִקְרֶה.

טו הַפּוֹעֵל מִצַּד עַצְמוֹ הוּא הַפּוֹעֵל בְּטִבְעוֹ אוֹ בִרְצוֹנוֹ, וְיִתְחַלֵּק לְעִקָּר וְטָפַל.

טז הָעִקָּרִי – שֶׁעִקַּר תְּלִיַּת הַמְּסֻבָּב בּוֹ, וְיִתְחַלֵּק לְפוֹעֵל לְבַדּוֹ וּפוֹעֵל בְּחֶבְרָה אוֹ בְּשֻׁתָּפוּת.

דֶּרֶךְ מָשָׁל: עִקַּר הַנְהָגַת הָעָם בִּמְלֹךְ תַּחְתָּיו הַשָּׂרִים וְהַשּׁוֹטְרִים.

יז הַפּוֹעֵל לְבַד שֶׁמַּמְצִיא הַמְסֻבָּב לְבַדּוֹ מִבְּלִי שֶׁיִּהְיֶה אַחֵר שָׁוֶה לוֹ בְּהַמְצָאָה זֹאת.

פֵּרוּשׁוֹ: כִּי לֹא יַרְחִיק הַפּוֹעֵל לְבַדּוֹ הַסִּבָּה הַכֵּלִיִּית, שֶׁנְּבָאֵר לְקַמָּן בְּסִיַּעְתָּא דִשְׁמַיָּא, שֶׁאֵינָהּ מִשְׁתַּוָּה לָהּ בְּהַמְצָאַת המְהִבָּב, אֶלָּא תַחְתֶּיהָ הִיא בְּמַדְרֵגָה, אַךְ יִשְׁלֹל שֶׁיִּהְיֶה אַחֵר שָׁוֶה לוֹ בְּהַמְצָאַת הַמְסֻבָּב בְּמַדְרֵגָה אַחַת.

יח הַפּוֹעֵל בְּחֶבְרָה הוּא אֲשֶׁר יְסַיֵּעַ אַחֵר בְּהַמְצָאַת הַמְסֻבָּב, בְּשִׁוּוּי אֵלָיו. וִיכוֹלִים הַמְשֻׁתָּפִים לִהְיוֹת שְׁקוּלִים וּבִלְתִּי שְׁקוּלִים.

יט הַשְּׁקוּלִים הֵם שֶׁחֵלֶק כָּל אֶחָד מֵהֶם שָׁקוּל בְּהַמְצָאַת הַמְסֻבָּב, וְהַבִּלְתִּי שְׁקוּלִים – שֶׁאֵין כָּל אֶחָד מֵהֶם שָׁקוּל. וְזֶה וָזֶה יָכוֹל שֻׁתָּפוּתָם שֶׁיִּהְיֶה מֻכְרָח אוֹ בִלְתִּי מֻכְרָח.

כ שֻׁתָּפוּת מֻכְרָח הוּא כְּשֶׁאֵין אֶחָד מֵהֶם לְבַדּוֹ יָכוֹל לְהַמְצִיא מְסֻבָּבוֹ, וּבִלְתִּי מֻכְרָח – שֶׁיֵּשׁ כֹּחַ בְּאֶחָד מֵהֶם לְהַמְצִיא מְסֻבָּבוֹ.

כא הַסִּבָּה הַטְּפֵלָה הִיא שֶׁכֹּחָהּ בְּהַמְצָאַת הַמְּסֻבָּב אֵינוֹ עִקָּרִי, וְתֵחָלֵק לִמְעוֹרֶרֶת וְכֵלִיִּית.

כב הַמְעוֹרֶרֶת הִיא שֶׁמְּעוֹרֶרֶת הָעִקָּרִי לְשֶׁיִּפְעַל.

דֶּרֶךְ מָשָׁל: תַּאֲוַת הָרוּחַ הִיא הַמְעוֹרֶרֶת אֶת הַסּוֹחֵר לִטְרֹחַ בִּסְחוֹרָתוֹ.

הַכֵּלִיִּי – שֶׁמְּשַׁמְּשָׁה לַפּוֹעֶלֶת לְהַמְצִיא מְסֻבָּבָהּ.

דֶּרֶךְ מָשָׁל: הַגַּרְזֶן לַחוֹצֵב.

כג עוֹד הַפּוֹעֶלֶת מִתְחַלֶּקֶת לִשְׁלֹשָׁה מִינִים: מוֹלִידָה, מְקַיֶּמֶת וּמַפְסֶדֶת.

כד הַמּוֹלִידָה הִיא שְׁמְּחַדֶּשֶׁת וּמַמְצִיאָה הַמְסֻבָּב, הַמְקַיֶּמֶת – שֶׁמַּתְמֶדֶת אוֹתוֹ, וְהַמַּפְסֶדֶת – שֶׁמְּבַטַּלְתּוֹ.

כה עוֹד מִתְחַלֶּקֶת הַפּוֹעֶלֶת לְטִבְעִית וּרְצוֹנִית אוֹ בְּחִירִיִּית. הַטִּבְעִית הִיא שֶׁפּוֹעֶלֶת מֵחֲמַת טִבְעָהּ בְּהֶכְרַח; הָרְצוֹנִית – שֶׁפּוֹעֶלֶת מֵחֲמַת עֵצָה וּבְחִירָה, מַה שֶּׁאֵינָהּ מֻכְרַחַת.

דֶּרֶךְ מָשָׁל: שְׂרֵפַת הָאֵשׁ פְּעֻלָּה טִבְעִית שֶׁאִי אֶפְשָׁר לוֹ זוּלָתָהּ; כְּתִיבַת הָאָדָם – רְצוֹנִית.

כו עוֹד מִתְחַלֶּקֶת בְּכָל שְׁאָר הַסִּבּוֹת, – שְׁנְּבָאֵר בְּסִיַּעְתָּא דִשְׁמַעְיָא – לִרְחוֹקָה, קְרוֹבָה וּסְמוּכָה. בֵּין הָרְחוֹקָה וְהַמְּסֻבָּב – אֶמְצָעִיּוֹת הַרְבֵּה, בֵּין הַקְּרוֹבָה לִמְסֻבָּב אֶחָד, בֵּין הַסְּמוּכָה וְהַמְסֻבָּב וְלֹא כְלוּם.

דֶּרֶךְ מָשָׁל: אַבְרָהָם הוּא סִבָּה רְחוֹקָה לִרְאוּבֵן, יִצְחָק – סִבָּה קְרוֹבָה, יַעֲקֹב – סִבָּה סְמוּכָה.


כְּלָלֵי הַסִּבָּה הַסְּמוּכָה:

כְּשֶׁתֻּנַּח הַסִּבָּה הַסְּמוּכָה בַּפּוֹעֵל, מֻכְרָח שֶׁיִּמָּצֵא הַמְסֻבָּב בַּפּוֹעֵל, מַה שֶּׁאֵין כֵּן בָּרְחוֹקָה וְלֹא בַקְּרוֹבָה.

כז הַסִּבָּה הַפּוֹעֶלֶת בְּמִקְרֶה הִיא, שֶׁאֵין הַמְּסֻבָּב נוֹלָד מִמֶּנָּה לֹא מִטֶּבַע וְלֹא מֵרָצוֹן, וְתִמָּצֵא בֵּין בַּסִּבּוֹת הַטִּבְעִיּוֹת בֵּין בַּבְּחִירִיּוֹת.

כח פּוֹעֶלֶת בְּמִקְרֶה בְּטִבְעִיּוּת הִיא שֶׁפּוֹעֶלֶת לֹא מִצַּד טִבְעָהּ אֶלָּא מִצַּד מִקְרֶה שֶׁקָּרָה לָהּ.

מָשָׁל: הַמִּקְרִית בְּטִבְעִיּוּת הַנְּיָר הַלָּבָן מְיַגֵּעַ הָרְאִיָּה מִצַּד מִקְרֵה הַלֹּבֶן שֶׁבּוֹ.

כט הַמִּקְרִית בַּבְּחִירִיּוֹת הִיא שֶׁפּוֹעֶלֶת שֶׁלֹּא בְכַוָּנָה אוֹ בִשְׁגָגָה.


פֶּרֶק חֲמִישִׁי

הַתַּכְלִית הָעִקָּרִי וְטָפֵל. אַחֲרוֹן וְאֶמְצָעִי. כְּלָלֵי הַתַּכְלִית.


א עַד הֵנָּה בֵּאַרְנוּ הַסִּבָּה הַפּוֹעֶלֶת, עַתָּה נְבָאֵר הַתַּכְלִיתִית, וְהִיא – שֶׁבַּעֲבוּרָהּ סוֹבֵב הַמְּסֻבָּב אוֹ נַעֲשֶׂה הַדָּבָר; וְיִתְחַלֵּק לְעִקָּרִי וְטָפֵל. הָעִקָּרִי הוּא הַכַּוָּנָה הָרִאשׁוֹנָה שֶׁל הַפּוֹעֵל, הַטָּפֵל הוּא הַכַּוָּנָה אֲשֶׁר לוֹ מִלְּבַד הָרִאשׁוֹנָה, בְּמַדְרֵגָה שְׁנִיָּה אֵלֶיהָ.

ב עוֹד יִתְחַלֵּק לְאַחֲרוֹן וְאֶמְצָעִי. הָאַחֲרוֹן כְּמַשְׁמָעוֹ – שֶׁדַּי לַפּוֹעֵל בּוֹ, הָאֶמְצָעִי – שֶׁהוּא תַכְלִית לְדָבָר אֶחָד, אַךְ יִהְיֶה אֶמְצָעִי לְהַגִּיעַ עוֹד אֶל תַּכְלִית אַחֵר שֶׁכִּוֵּן לוֹ הַפּוֹעֵל.

דֶּרֶךְ מָשָׁל: תַּכְלִית חָכְמַת הָרְפוּאָה הוּא לָדַעַת הֶחֳלָאִים לְמִינֵיהֶם בְּסִבּוֹתֵיהֶם, אָמְנָם הַתַּכְלִית הָאַחֲרוֹן הוּא לְרַפֵּא אֶת הַחוֹלִים.


כְּלָלֵי הַסִּבָּה הַתַּכְלִיתִית:

1 הַתַּכְלִית מֵנִיעַ אֶת הַפּוֹעֵל לִפְעֹל.

2 בְּדָבָר אֶחָד יְכוֹלִים לִמָּצֵא תַכְלִיּוֹת רַבִּים.

3 הַתַּכְלִיּוֹת מְשַׁנִּים הַפְּעֻלּוֹת.

פֵּרוּשׁוֹ, כִּי פְעֻלָּה אַחַת נַעֲשֵׂית מִפּוֹעֵל אֶחָד לְתַכְלִית אֶחָד וּמִפּוֹעֵל אַחֵר לְתַכְלִית אַחֵר יֵחָשְׁבוּ לִפְעֻלּוֹת שׁוֹנוֹת, וּכְבָר תּוּכַל לִהְיוֹת אֶחָד מֵהֶם טוֹבָה וְאֶחָד מֵהֶם רָעָה, אַף עַל פִּי שֶׁשְּׁתֵיהֶן אֶחָד, לְפִי שֶׁתַּכְלִיּוֹתֵיהֶן שְׁתַּיִם.

דֶּרֶךְ מָשָׁל: שְׁנַיִם מַאֲרִיכִין בִּתְפִלָּתָן, אֶחָד לְשֵׁם שָׁמַיִם וְאֶחָד לִגְנֹב דַּעַת הַבְּרִיּוֹת שֶׁיַּאֲמִינוּ לוֹ; זֶה עוֹשֶׂה מִצְוָה רַבָּה, וְזֶה – אָשֵׁם אַשְׁמָה רַבָּה.

4 הַתַּכְלִית קוֹדֵם לָאֶמְצָעִיִּים שֶׁלּוֹ בְּמַחֲשָׁבָה, מַה שֶּׁאֵין כֵּן בְּמַעֲשֶׂה.

פֵּרוּשׁוֹ, כִּי אֵין שׁוּם אָדָם חוֹשֵׁב בָּאֶמְצָעִי אֶלָּא אִם כֵּן חָשַׁב בַּתְּחִלָּה עַל הַתַּכְלִית, שֶׁבַּעֲבוּרוֹ יִצְטָרְכוּ הָאֶמְצָעִיּים הָהֵם; אַךְ בְּמַעֲשֶׂה אִי אֶפְשָׁר לְהַגִּיעַ אֶל הַתַּכְלִית אֶלָּא אַחַר כָּל הָאֶמְצָעִיִּים.

5 שֶׁהַתַּכְלִית הֹוָה, כָּלְתָה הַפְּעֻלָּה שֶׁנַּעֲשֵׂית לְהַגִּיעַ אֵלָיו. מִי שֶׁרוֹצֶה בֶאֱמֶת בַּתַּכְלִית, צָרִיךְ שֶׁיִּרְצֶה בָּאֶמְצָעִיִּים.

פֵּרוּשׁוֹ, כִּי הַשָּׂגַת הַתַּכְלִית בְּלִי אֶמְצָעִיִּים אִי אֶפְשָׁר.


פֶּרֶק שִׁשִּׁי

הַחֹמֶר הָרָחוֹק וְהַקָּרוֹב. כְּלָלֵי הַחֹמֶר. הַצּוּרָה הָעַצְמִית וְהַמִּקְרִית. כְּלָלֵי הַצּוּרָה.


א עַד הֵנָּה נִתְבָּאֲרָה הַסִּבָּה הַחִיצוֹנָה, עַתָּה נְבָאֵר הַפְּנִימִית, הַנִּכְנֶסֶת בִּמְצִיאוּת הַמְסוֹבָב עַצְמוֹ, וּמִתְחַלֵּק לִשְׁנַיִם: חֹמֶר וְצוּרָה.

ב הַחֹמֶר הוּא מַה שֶּׁמִּמֶּנּוּ נַעֲשָׂה הַמְסֻבָּב, וְיִתְחַלֵּק לְרָחוֹק וְקָרוֹב. הַרָחוֹק הוּא – מַה שֶׁמִמֶּנּוּ נַעֲשָׂה הַמְסֻבָּב בְּחֶבְרַת חֹמֶר אַחֵר קָרוֹב מִמֶּנּוּ; הַקָּרוֹב – אוֹתוֹ שֶׁמִּמֶּנּוּ נַעֲשָׂה הַמְסֻבָּב בְּלֹא חֹמֶר בָּאֶמְצַע.

מָשָׁל: הַחֹמֶר הָרָחוֹק בָּאָדָם – הֶעָפָר, כִּי לֹא מֵעָפָר מִיָּד נוֹצָר הַוָּלָד, אֶלָּא שֶׁרֵאשִׁית עִנְיָנוֹ הוּא עָפָר, אַךְ חֹמֶר הַקָּרוֹב הוּא בָּשָׂר וָדָם וּשְׁאָר הַרְכָּבוֹתָיו.


כְּלָלֵי הַחֹמֶר:

הַחֹמֶר קוֹדֵם לַצּוּרָה בַּטֶּבַע.

פֵּרוּשֹוֹ, כִּי אִי אֶפְשָׁר לְצוּרָה לִמָּצֵא אֶלָּא בַחֹמֶר.

ג הַצּוּרָה הִיא – מַה שֶּׁבּוֹ הַמְסֻבָּב הוּא מַה שֶׁהוּא, וְתִתְחַלֵּק לְעַצְמִית וּמְדֻקְדֶּקֶת, וּלְמִקְרִית אוֹ בִלְתִּי־מְדֻקְדֶּקֶת, הָעַצְמִית הִיא – מַה שֶׁעוֹשֶׂה הַמְסֻבָּב מַה שֶּׁהוּא בַאֲמִתַּת עִנְיָנוֹ וּמַבְדִּילוֹ מִשְּׁאָר כָּל הַנִּמְצָאִים.

דֶּרֶךְ מָשָׁל: צוּרַת הָאָדָם הוּא הַכֹּחַ הַמְדַבֵּר, שֶׁכּוֹלֵל הַשֵּכֶל וְהָרָצוֹן, שֶׁבָּזֶה הוּא מַה שֶׁנִּבְדָּל בָּעֶצֶם.

ד הַבִּלְתִּי מְדֻקְדֶּקֶת הִיא כְּלָל אֵיזֶה מִקְרִים מְיֻחָדִים לְאֵיזֶה נִמְצָא שֶׁנְּצַיְּרֵהוּ בָהֶם, כְּשֶׁאֵין צוּרָתוֹ הָעַצְמִית נוֹדַעַת.

דֶּרֶךְ מָשָׁל שֶׁנֹּאמַר: הַכֶּלֶב הוּא שֶׁיֵּשׁ לוֹ כֹּחַ הַנִּבּוּחַ, וְאֵין זֶה עִקַּר עִנְיָנוֹ שֶׁל הַכֶּלֶב.


כְּלָלֵי הַצּוּרָה:

1 הַצוּרָה נִכְבֶּדֶת תָּמִיד מֵהַחֹמֶר.

פֵּרוּשׁוֹ, כִּי בָהּ תָּלוּי כָּל אֲמִתַּת עִנְיַן הַנִּמְצָא, מַה שֶּׁאֵין כֵּן בַּחֹמֶר.

2 הַצּוּרָה הָעַצְמִית אֵינָהּ מֻשֶּׂגֶת מֵהַחוּשׁ.

3 פְּעֻלּוֹת הַנִּמְצָא תְּלוּיוֹת בַּצּורָה שֶׁלּוֹ.


פֶּרֶק שְׁבִיעִי

מְסֻבְּבֵי אַרְבַּע הַסִּבּוֹת. כְּלָלֵי הַמְסֻבָּב.


א עַד הֵנָּה בֵּאַרְנוּ הַסִּבָּה, עַתָּה נְבָאֵר הַמְסֻבָּב, וְהוּא מַה שֶׁנּוֹלַד מִכֹּחַ הַסִּבָּה.

ב לְכָל אַרְבַּע הַסִּבּוֹת יֵשׁ מְסֻבָּב. מְסֻבָּב הַפּוֹעֶלֶת הוּא הַפָּעוּל, מְסֻבָּב הַתַּכְלִיתִית הוּא הַנִּגְזָר, מְסֻבָּב הַצּוּרִיִּית הוּא הַמְצֻיָּר, מְסֻבָּב הַחָמְרִית הוּא הַחֹמֶר.


כְּלָלֵי הַמְסֻבָּב:

1 הַמְסֻבָּב לֹא יִהְיֶה לוֹ כֹּחַ יוֹתֵר מִסִּבָּתוֹ.

פֵּרוּשׁוֹ, כִּי אֵין הַמְסֻבָּב אֶלָּא מַה שֶּׁנּוֹלַד מִן הַסִּבָּה, וְהַסִּבָּה אֵינָה יְכוֹלָה לְהוֹלִיד מַה שֶּׁאֵין בָּהּ, אִם כֵּן אִי אֶפְשָׁר שֶׁיִּהְיֶה בַּמְסֻבָּב מַה שֶׁלֹּא יִמָּצֵא בְסִבָּתוֹ.

2 כְּמַהוּת הַסִּבָּה כָּךְ מַהוּת הַמְסֻבָּב.

פֵּרוּשׁוֹ: זֶה לֹא יוּבַן אֶלָּא בַסִּבּוֹת הַטִּבְעִיּוֹת, עִקָּרִיּוֹת וּמְיֻחָדוֹת, וּבִמְסֻבָּב שֶׁיּוּכַל לְקַבֵּל מַהוּת סִבָּתוֹ.

3 הַדָּבָר שֶׁבַּעֲבוּרוֹ – אַחֵר הוּא מַה שֶּׁהוּא, כָּל שֶׁכֵּן שֶׁהוּא עַצְמוֹ יִהְיֶה כֵן.

דֶּרֶךְ מָשָׁל: אִם הַסְּפָרִים הַנֶּחְמָדִים בַּעֲבוּר הַחָכְמָה הַנֶּחְקֶקֶת בָּהֶם, כָּל שֶׁכֵּן שֶׁתִּהְיֶה נֶחְמֶדֶת הַחָכְמָה עַצְמָהּ.


פֶּרֶק שְמִינִי

הַנּוֹשֵׂא וְהָעֵסֶק. הַנּוֹשֵׂא הַמֻּחְלָט וְהַמֻּגְבָּל. מְיֻחָד וּבִלְתִּי מְיֻחָד.


א הַנּוֹשֵׂא הוּא הַדָּבָר שֶׁיִּתְחַבֵּר אֵלָיו מָה, וְיִתְחַלֵּק לְנוֹשֵׂא וָעֵסֶק. הַנּוֹשֵׂא הוּא שֶיְּקַבֵּל בּוֹ, אוֹ עָלָיו, מָה.

דֶּרֶךְ מָשָׁל: הָאֵשׁ לַחֹם.

עֵסֶק – שֶעָלָיו יַעֲסֹק עוֹסֵק.

דֶּרֶךְ מָשָׁל: הַמַּרְאֶה – עֵסֶק לִרְאוֹת.

ב הַנּוֹשֵׂא יִתְחַלֵּק לְמֻחְלָט וּמֻגְבָּל. הַמֻּחְלָט הוּא שֶיִּתְחַבֵּר לוֹ מִתְחַבֵּר בְּלִי שׁוּם גְּבוּל וּתְנָאי.

דֶּרֶךְ מָשָׁל: הָאֵשׁ שֶׁמִּתְחַבֵּר לוֹ הַחֹם בְּהֶחְלֵט.

הַמֻּגְבָּל – שֶׁיִּתְחַבֵּר לוֹ הַמִּתְחַבֵּר לְפִי עִנְיָן אֶחָד אוֹ הַבְחָנָה אֶחָד.

דֶּרֶךְ מָשָׁל: הַכּוּשִׁי נוֹשֵׂא לַלֹּבֶן לֹא בִּבְחִינַת כֻּלּוֹ, אֶלָּא בִּבְחִינַת אֶחָד מֵחֲלָקָיו, דְּהַיְנוּ הַשִּׁנַּיִם.

ג עוֹד מִתְחַלֵּק הֵנּוֹשֵׂא לִמְיֻחָד אוֹ בִלְתִּי מְיֻחָד.

הַמְיֻחָד הוּא שֶמִּתְיַחֵד לְמִקְרֵהוּ, שֶלֹּא יִמָּצְאוּ זֶה בְלֹא זֶה.

דֶּרֶךְ מָשָׁל: הַחֹם וְהָאֵשׁ לֹא יִמָּצְאוּ זֶה בְלִי זֶה.

הַבִּלְתִּי מְיֻחָד הוּא שֶאֵינוֹ מְיֻחָד עִם מִקְרֵהוּ, אֶלָּא יִמָּצֵא הַמִּקְרֶה הַהוּא גַּם בְּזוּלָתוֹ.

דֶּרֶךְ מָשָל: הַבַּעַל חַי וֶהֱיוֹת נִרְאֶה, כִּי כָל בַּעַל חַי נִרְאֶה, אַךְ יֵשׁ נִרְאֶה שֶאֵינוֹ בַעַל חָי.


פֶּרֶק תְּשִׁיעִי

הַמִּתְחַבֵּר, תּוֹכִיִּי וְנִסְמָךְ. הַנִּצָּב וְהַמְהַלֵּךְ. הַקּוֹדֵם וְהַנִּמְשָׁךְ. הַנָּשׁוּא הָעַצְמִי וְהַמִּקְרִי בְּכֹחַ וּבְפֹעַל. כְּלָלֵי הַמִּתְחַבֵּר.


א הַמִּתְחַבֵּר הוּא מַה שֶּׁמִּתְחַבֵּר אֶל נוֹשֵׂא, וּמִתְחַלֵּק לְתוֹכִיִּי וְנִסְמָךְ.

ב הַתּוֹכִיִּי הוּא שֶׁמִּתְחַבֵּר בְּתוֹךְ נוֹשֵׂא וְלֹא יוּכַל לִהְיוֹת בִּלְתּוֹ, וּמִתְחַלֵּק לִמְיֻחָד וּבִלְתִּי מְיֻחָד.

הַמְיֻחָד הוּא שֶׁלֹּא יִמָּצֵא אֶלָּא בְנוֹשְׂאִים מִמִּין אֶחָד, וְיִתְחַלֵּק לְשָׁלֵם וּבִלְתִּי שָׁלֵם.

ג הַמְיֻחָד הַשָּׁלֵם הוּא הַנִּמְצָא בְכָל נוֹשְׂאוֹ וּבְכָל זְמָן, וְלֹא בְזוּלָתוֹ.

דֶּרֶךְ מָשָׁל: הַהֶרְגֵּשׁ לְבַעַל חָי.

ד הַבִּלְתִּי שָׁלֵם הוּא הַנִּמְצָא בְנוֹשֵׂא אֶחָד בְּיִחוּד.

פֵּרוּשׁוֹ: לֹא בְזוּלָתוֹ, אַךְ לֹא בְכֻלּוֹ, אוֹ לֹא בְכָל עֵת.

דֶּרֶךְ מָשָׁל: הַשְּׂחוֹק הוּא מְיֻחָד לָאָדָם, כִּי לֹא נִמְצָא לְזוּלָתוֹ, אַךְ אֵינוֹ בָאָדָם בְּכָל עֵת.

ה הַבִּלְתִּי מְיֻחָד הוּא שֶׁיִּמָּצֵא לְנוֹשְׂאִים מִמִּינִים שׁוֹנִים.

דֶּרֶךְ מָשָׁל: הַהֶעְתֵּק מִמָּקוֹם שֶׁיִּמָּצֵא לָאָדָם וְיִמָּצֵא גַם כֵּן לְבַעֲלֵי חַיִּים וְגַם לְדוֹמְמִים.

ו הַנִּסְמָךְ מִתְחַלֵּק לְנִצָּב וּמְהַלֵּךְ.

הַנִּצָּב – אוֹ חוֹנֶה עַל הַנּוֹשֵׂא עַצְמוֹ, אוֹ נִמְצָא יַחַד עִמּוֹ, אוֹ סוֹבֵב וּמַגְבִּיל אוֹתוֹ.

דֶּרֶךְ מָשָׁל: הַלְּבוּשׁ חוֹנֶה עַל הַגּוּף; עִם לֵדַת מֹשֶה נִמְצָא יַחַד מִלּוּי הַבַּיִת אוֹרָה; הַזְּמַן מַגְבִּיל הַפְּעֻלָּה שֶׁנַּעֲשֵׂית בְּיוֹם פְּלוֹנִי בְּשָׁעָה פְלוֹנִית.

הַמְהַלֵּךְ מַקְדִּים לַנּוֹשֵׂא, אוֹ בָא אַחֲרָיו, עַל כֵּן יִתְחַלֵּק לְקוֹדֵם וְנִמְשָׁךְ.

דֶּרֶךְ מָשָׁל: הָאָבִיב קוֹדֵם לַקַּיִץ, וְהָאָסִיף בָּא אַחֲרָיו.

ז עוֹד יִתְחַלֵּק הַמִּתְחַבֵּר אוֹ הַנָּשׂוּא לְעַצְמִי וּמִקְרִי.

ח הָעַצְמִי הוּא מְיֻחָד לַנּוֹשֵׂא לְפִי צוּרָתוֹ הָעַצְמִית וְנִמְצָא בוֹ עַל הָרֹב; הַמִּקְרִי הוּא – שֶׁאֵינוֹ מְיֻחָד לוֹ וְאֵינוֹ נִמְצָא בוֹ אֶלָּא בְמִעוּטוֹ.

מְשַׁל הָעַצְמִי: הַחֹם לָאֵשׁ.

מְשַׁל הַמִּקְרִי: הַלֹּבֶן לַבֶּגֶד.

ט עוֹד מְצִיאוּת הַנָּשׂוּא בַנּוֹשֵׂא יִתְחַלֵּק לִשְׁנַיִם, הָאֶחָד בְּכֹחַ וְהַשֵּׁנִי בְּפֹעַל.

י הַמְּצִיאוּת בְּכֹחַ הוּא כָּל זְמַן שֶׁהַנָּשׂוּא בַנּוֹשֵׂא בַּהֲכָנָה לְבַד, וְאֵינוֹ בְמַעֲשֶׂה; בְּפֹעַל – כְּשֶׁהִגִּיעַ לְמַעֲשֶׂה.

מְשַׁל הַכֹּחַ: הָאָדָם יֵשׁ בְּכֹחוֹ לִהְיוֹתוֹ כוֹתֵב אַף בְּשָׁעָה שֶׁאֵינוֹ כוֹתֵב; וּבְשָׁעָה שֶׁהוּא כוֹתֵב נִקְרָא: כּוֹתֵב בְּפֹעַל.

יא הַכֹּחַ מִתְחַלֵּק לְרָחוֹק וְקָרוֹב.

הָרָחוֹק הוּא שֶׁחֲסֵרוֹת לוֹ מַדְרֵגוֹת רַבּוֹת לִהְיוֹת נִמְצָא בְּפֹעַל; הַקָּרוֹב הוּא – שֶׁאֵין חָסֵר לוֹ כִּי אִם מַדְרֵגַת צַאתוֹ לְפֹעַל.


כְּלָלֵי הַמִּתְחַבֵּר:

1 הַמִּקְרֶה אֲשֶׁר הוּא מְיֻחָד לְמִין אֶחָד אִי אֶפְשָׁר שֶׁיִּתְיַחֵס לְמִין אַחֵר בִּלְתִּי דוֹמֶה לוֹ.

דֶּרֶךְ מָשָׁל: הַדֵּעָה מְיֻחֶדֶת לְמִין הָאָדָם, לֹא תִתְיַחֵס לָאֲבָנִים.

2 הַמִּקְרֶה הַמְיֻחָד הַשָּׁלֵם לֹא יִפָּרֵד מִנּוֹשְׂאוֹ אֲפִילוּ רֶגַע אֶחָד.

3 אִי אֶפְשָׁר לְצַיֵּר נוֹשֵׂא עִם שְׁלִילַת הַמִּקְרֶה הַמְיֻחָד הַשָּׁלֵם שֶׁלּוֹ.

4 הַמִּתְחַבֵּר יֵאָמֵר בַּנּוֹשֵׂא בְחִבּוּר, לֹא בְהֶחְלֵט.

פֵּרוּשׁוֹ, כִּי הַמִּתְחַבֵּר שֶׁיֵּאָמֵר הֱיוֹתוֹ נָשׂוּא בְנוֹשְׂאוֹ, יוּבַן בּוֹ הֱיוֹתוֹ אֶחָד מִן הַמִּקְרִים מִתְדַּבֵּק בַּנּוֹשֵׂא עַד שֶׁיְתֹאַר נוֹשְׂאוֹ בוֹ, וְהוּא הַנִּקְרָא לַמְדַקְדְּקִים: שֵׁם תֹּאַר, כְּגוֹן – חָכָם, גִּבּוֹר; אַךְ הַמִּקְרֶה בְהֶחְלֵט הוּא שֶׁקְּרָאוּהוּ הַמְדַקְדְּקִים: שֵׁם דָּבָר, שֶׁאֵין הַנּוֹשֵׂא מְתֹאָר בְּמִקְרֶה, כְּגוֹן – חָכְמָה, גְּבוּרָה.


פֶּרֶק עֲשִׂירִי

הַכֹּל עַצְמִי וְהַמִּקְרִיִּי מֻרְכָּב וּבָנוּי. הַמִּתְדַּמֶה וּבִלְתִּי מִתְדַּמֶה. כְּלָלֵי הַכֹּל. הַחֵלֶק וּכְלָלָיו.


א כֹּל הוּא מַה שֶּׁמְחֻבָּר מֵחֲלָקִים שֶׁמֵּהֶם נִבְנָה, וְיִתְחַלֵּק לְכֹל מִצַּד עַצְמוֹ וְכֹל מִקְרִיִּי.

ב כֹּל מִצַּד עַצְמוֹ הוּא מַה שֶׁיֵּשׁ בּוֹ חֲלָקִים מַמָּשׁ שֶׁבּוֹנִים אוֹתוֹ, וְיִתְחַלֵּק לְמֻרְכָּב, אוֹ עַצְמִי, וּבָנוּי.

ג הַמֻּרְכָּב אוֹ עַצְמִי הוּ שֶׁחֲלָקָיו עַצְמִיִּים בּוֹ, דְּהַיְנוּ חֹמֶר וְצוּרָה.

פֵּרוּשׁוֹ: כָּל נִמְצָא, שֶׁהוּא מֻרְכָּב מֵחֹמֶר וְצוּרָה, בִּבְחִינַת הֱיוֹתוֹ הַרְכָּבַת שְׁנֵי אֵלֶּה, נִקְרְאוּ כֹּל עַצְמִי.

ד הַבָּנוּי הוּא – שֶׁיֵּשׁ חֲלָקִים בִּנְיָנִיִּים שֶׁמֵּחִבּוּרָם הוּא נִבְנֶה, וְיִתְחַלֵּק לְמִתְדַּמֶּה וּבִלְתִּי מִתְדַּמֶּה.

ה הַמִּתְדַּמֶּה הוּא – שֶׁכָּל חֵלֶק מִמֶּנּוּ מִתְדַּמֶּה לַכֹּל בַּשֵּׁם וּבַטֶּבַע.

דֶּרֶךְ מָשָׁל: הָאֵשׁ.

הַבִּלְתִּי מִתְדַּמֶּה הוּא – שֶׁאֵין כָּל חֵלֶק מִמֶּנּוּ מִתְדַּמֶּה אֶל הַכֹּל בַּשֵּׁם וּבַטֶּבַע.

דֶּרֶךְ מָשָׁל: הָאָדָם אוֹ הַבַּיִת שֶׁאֵין כָּל חֵלֶק נִקְרָא אָדָם אוֹ בַיִת.

ו הַכֹּל הַמִּקְרִיִּי – שֶׁאֵינוֹ כֹּל עַצְמִי אֶלָּא בְדֶרֶךְ מִקְרֶה.

דֶּרֶךְ מָשָׁל: הַנּוֹשֵׂא הַנֶּחֱלָק בִּנְשׂוּאָיו.


כְּלָלֵי הַכֹּל:

1 יֵשׁ הֶפְרֵשׁ בֵּין הַכֹּל לַחֲלָקָיו, וַאֲפִילוּ לְכֻלָּם בְּיַחַד בִּהְיוֹתָם חֲלָקִים.

פֵּרוּשׁוֹ, כִּי בִהְיוֹת הַחֲלָקִים מִתְחַבְּרִים יַחַד וְעוֹשִׂים כֹּל מְחֻבָּר מִכֻּלָּם, נַעֲשֶׂה מִזֶּה מְצִיאוּת שֶׁאֵינוֹ נִמְצָא כְלָל, אֲפִילוּ כְּשֶׁהָיוּ כֻלָּם יַחַד אַךְ בִּלְתִּי מִתְחַבְּרִים.

דֶּרֶךְ מָשָׁל: אִם תְּסַדֵּר כָּל אֵיבְרֵי הַבְּהֵמָה זֶה אֵצֶל זֶה, לֹא יִקָּרְאוּ בְהֵמָה, אֶלָּא חֶלְקֵי בְהֵמָה.

2 הַכֹּל לְגַבֵּי חֲלָקָיו הוּא יוֹתֵר שָׁלֵם, וְהַחֲלָקִים לְגַבֵּי הַכֹּל יוֹתֵר חֲסֵרִים.

3 כְּשֶיֻנַּח הַכֹּל, יֻנְּחוּ הַחֲלָקִים בְּהֶכְרַח; אַךְ בִּשְׁלֹל הַכֹּל, לֹא תֻכְרַח שְׁלִילַת הַחֲלָקִים.

ז הַחֵלֶק הוּא מַה שֶּׁמִּמֶּנּוּ יִבָּנֶה הַכֹּל בְּחֶבְרַת שְׁנַיִם אוֹ יוֹתֵר, וְחִלּוּקָיו כְּחִלּוּקֵי הַכֹּל.


כְּלָלֵי הַחֵלֶק:

1 כָּל חֵלֶק לְגַבֵּי הַכֹּל הוּא חָסֵר.

2 לֹא כָל מַה שֶּׁיֵּשׁ לַכֹּל יֵשׁ לְכָל חֵלֶק.

3 מַה שֶּׁיֵּשׁ לַחֵלֶק מִצַּד עַצְמוֹ יֵשׁ לַכֹּל מִצַּד אוֹתוֹ הַחֵלֶק שֶׁבּוֹ.


פֶּרֶק אַחַד־עָשָׂר

הַגִּזְרָה וְהַנִּגְזָר וּכְלָלֵיהֶם. הַבִּנְיָן וּכְלָלָיו.


א הַגִּזְרָה הִיא פֵּרוּשׁ הַמִּלָּה וּמְקוֹרָהּ.

ב גִּזְרוֹת הַמִּלּוֹת נִקָּחוֹת מִן הַסִּבּוֹת, מִן הַמְסֻבָּבִים, מִן הַנּוֹשְׂאִים, וּמִן הַמִּתְחַבְּרִים, וּמִן הַהַפְכִּיִּים.

דֶּרֶךְ מָשָׁל: אָדָם מִן הָאֲדָמָה; נֹחַ מִן הַמְּנוּחָה.

ג נִגְזָר הוּא הַנִּבְנֶה לְפִי הַגִּזְרָה.


כְּלָלֵי הַגִּזְרָה:


לֹא כָל הַמִּלּוֹת יְבֻקָּשׁ לָהֶם גִּזְרָה.

פֵּרוּשׁוֹ, כִּי יֵשׁ מִלּוֹת רִאשׁוֹנוֹת וְהֵן שֶׁאֵינָן נִגְזָרוֹת מִמִּלָּה אַחֶרֶת.

דֶּרֶךְ מָשָׁל: אֵשׁ, מַיִם, לֶחֶם.

ד הַבִּנְיָן הוּא מַה שֶּׁנִּבְנֵית הַמִּלָּה עַל יְסוֹד עִנְיָנָהּ הָרִאשׁוֹן.

דֶּרֶךְ מָשָׁל: שֵׁם דָּבָר, שֵׁם הַתֹּאַר וּמָקוֹר – כֻּלָּם נִבְנִים עַל יְסוֹד הַפֹּעַל, דְּהַיְנוּ זִכָּרוֹן, זַכְרָן, זָכוֹר – כֻּלָּם מִינֵי בִנְיָן עַל הַיְסוֹד זָכֹר.

ה הַמִּלּוֹת הַנִּבְנוֹת מִתְחַלְּקוֹת לִדְבֵקוֹת וּמֻחְלָטוֹת.

ו הַדְּבֵקוֹת הֵם שֶׁמּוֹרוֹת עַל צוּרָה אַחַת בַּנּוֹשֵׂא, וְהַיְנוּ הַתֹּאַר.

דֶּרֶךְ מָשָׁל: חָכָם, נָבוֹן, וְכַיּוֹצֵא.

וְהַמֻּחְלָטוֹת – שֶׁמּוֹרוֹת עַל עִנְיָן אֶחָד בִּלְבָד, בְּלִי שֶׁיֻּזְכַּר עִמּוֹ נוֹשְׂאוֹ.

דֶּרֶךְ מָשָׁל: חָכְמָה, תְּבוּנָה.


כְּלָלֵי הַבִּנְיָן:

1 כְּשֶׁיֻּנְּחוּ אֶחָד מִדַּרְכֵי הַבִּנְיָן, יֻנְּחוּ הָאֲחֵרִים.

2 מַה שֶּׁשַּׁיָּךְ לְאֶחָד מִדַּרְכֵי הַבִּנְיָן, שַׁיָּךְ לְכֻלָּם.

דֶּרֶךְ מָשָׁל: אִם הַחֶסֶד נֶחְמָד, הֶחָסִיד יִהְיֶה נֶחְמָד.


פֶּרֶק שְׁנֵים־עָשָׂר

הַגָּדֵר, הַסּוּג וְהַהֶבְדֵּל. חִיּוּבֵי הַגָּדֵר. כְּלָלֵי הַגָּדֵר. הַחִלּוּק, הַנֶּחֱלָק וּכְלָלֵיהֶם.


א גָּדֵר הוּא בֵּאוּר שָׁלֵם שֶׁל הַדָּבָר הַנִּגְדָּר.

ב בַּגָּדֵר יֵשׁ לְהַבְחִין הַחֲלָקִים, הַחִיּוּב וְהַמִּין.

ג חֶלְקֵי הַגָּדֵר שְׁנַיִם: סוּג וְהֶבְדֵּל; בַּסּוּג מִשְׁתַּוֶּה הַנִּגְדָּר לְזוּלָתוֹ, וּבַהֶבְדֵּל – נִבְחָן מִכָּל זוּלָתוֹ.

דֶּרֶךְ מָשָׁל: גֶּדֶר הָאָדָם – בַּעַל חַי מְדַבֵּר; בַּעַל חַי – סוּג שֶׁבּוֹ מִשְׁתַּוֶּה עִם הַבִּלְתִּי מְדַבְּרִים, מְדַבֵּר – הוּא הַהֶבְדֵּל שֶׁנִּבְדָּל בּוֹ מֵהֶם.

ד חִיּוּבֵי הַגָּדֵר שְׁנַיִם. הָאֶחָד, שֶׁמִּלּוֹתָיו תִּהְיֶינָה רְאוּיוֹת וּמְבָאֲרוֹת; הַשֵּׁנִי, שֶׁהַגָּדֵר יִהְיֶה לְמִדַּת הַנִּגְדָּר, שֶׁלֹּא יֹאמַר פָּחוֹת מִמַּה שֶּׁהוּא וְלֹא יוֹתֵר מִמַּה שֶּׁהוּא.

פֵּרוּשׁוֹ, שֶׁלֹא יִכָּנֵס בּוֹ זוּלָתוֹ וְלֹא יֶחְסַר מִמֶּנּוּ.

דֶּרֶךְ מָשָׁל: אִם נִגְדֹּר הָאָדָם: בַּעַל חַי הַהוֹלֵךְ עַל שְׁתֵּי רַגְלָיו, הִנֵּה יִכָּנְסוּ בָּזֶה הָעוֹפוֹת, אוֹ אִם נִגְדֹּר הַנֶּפֶשׁ: הַדָּבָר הָרוֹצֶה וְהַמַּשְׂכִּיל, הִנֵּה נִשְׁאָר חֵלֶק אֶחָד מִן הַנֶּפֶשׁ שֶׁלֹּא נִזְכַּר בַּזֶּה שֶׁהוּא הַזִּכָּרוֹן.

ה מִינֵי הַגָּדֵר שְׁנַיִם, שָׁלֵם וּבִלְתִּי שָׁלֵם.

הַשָּׁלֵם הוּא הַנִּבְנֶה מִסּוּג. וְהֶבְדֵּל הַבִּלְתִּי שָׁלֵם הוּא, שֶׁמְּבָאֵר הַנִּגְדָּר מִצַּד מַה שֶּׁאֵינוֹ עִקָּר בּוֹ, כְּגוֹן: הַמְסֻבָּב, הַחֲלָקִים, הַחֶבְרָה, וְכַיּוֹצֵא.

דֶּרֶךְ מָשָׁל: אִם נִגְדֹּר הַיַּיִן: הַמַּשְׁקֶה הַמְשַׁכֵּר, אוֹ אִם נִגְדֹּר הַתֹּמֶר: הָאִילָן אֲשֶׁר בּוֹ לוּלָבִים וְחָרִיּוֹת וּתְמָרִים, אוֹ אִם נִגְדֹּר הַלֶּהָבָה: הַדָּבָר אֲשֶׁר הֶעָשָׁן קוֹדֵם לוֹ.


כְּלָלֵי הַגָּדֵר:

1 לְמַה שֶּׁיְּקֻיַּם אוֹ יְשׁוֹלַל הַגָּדֵר, יְקֻיַּם אוֹ יְשׁוֹלַל הַנִּגְדָּר; וְכֵן לְהֵפֶךְ.

דֶּרֶך מָשָׁל: בְּמִי שֶׁנֶּאֱמַר הֱיוֹתוֹ בַעַל חַי מְדַבֵּר, נֶאֱמַר הֱיוֹתוֹ אָדָם.

2 מַה שֶּׁשַּיָּךְ אוֹ לֹא שַׁיָּךְ בַּגָּדֵר, שַׁיָּךְ אוֹ לֹא שַׁיָּךְ בַּנִּגְדָּר.

דֶּרֶךְ מָשָׁל: מַה שֶּׁנֶּאֱמַר בְּבַעַל חַי מְדַבֵּר, נֶאֱמַר בָּאָדָם.

ו חִלּוּק הוּא בָּאֵר אֶת הַכֹּל וְהַבְחִינוֹ בַחֲלָקָיו, וְיִתְחַלֵּק לְעִקָּרִי וּבִלְתִּי עִקָּרִי.

ז הָעִקָּרִי הוּא הִתְחַלֵּק הַכֹּל בַּחֲלָקִים עִקָּרִים, וְהַבִּלְתִּי עִקָּרִי – לַחֲלָקִים בִּלְתִּי עִקָּרִים, וְהַיְנוּ הַכֹּל מִצַּד עַצְמוֹ וּבַמִּקְרֶה הַנַּ"ל.

ח חִיּוּבֵי הַחִלּוּק הָעִקָּרִי שְׁנַיִם: רִאשׁוֹן, שֶֹלֹּא יִפְחֲתוּ וְלֹֹא יוֹסִיפוּ הַחֲלָקִים עַל הַמִּתְחַלֵּק; שֵׁנִי, שֶׁיִּהְיוּ הַחֲלָקִים מִתְדַּמִּים עִם הַכֹּל וּבִלְתִּי מִתְדַּמִּים בֵּינֵיהֶם.

פֵּרוּשׁוֹ: אָדָם וּבְהֵמָה הֵם חֶלְקֵי הַמַּרְגִּישׁ בִּלְתִּי מִתְדַּמִּים, אַךְ אָדָם וּבַעַל חַי אֵינָם בִּלְתִּי מִתְדַּמִּים, כִּי הָאָדָם הוּא בַעַל חַי גַּם כֵּן.


כְּלָלֵי הַחִלּוּק:

1 הַחִלּוּק צָרִיךְ שֶׁיִּהְיֶה לַחֲלָקִים קְרוֹבִים.

פֵּרוּשׁוֹ, שֶׁלֹּא תְחַלֵּק הַסּוּג לְאִישִׁים שֶׁיֵּשׁ כַּמָּה מִינִים בֵּין הַסּוּג וּבֵינֵיהֶם, אֶלָּא לְמִינִים שֶׁבָּאִים מִיָּד אַחֲרָיו בְּהַדְרָגָה.

2 הַחִלּוּק צָרִיךְ שֶׁיִּהְיֶה לְמַעֵט הַחֲלָקִים כָּל מַה שֶׁאֶפְשָׁר.

ט הַנֶּחֱלָק – הַבְחָנוֹתָיו כְּהַבְחָנוֹת הַחִלּוּק.


כְּלָלֵי הַנֶּחְלָק:

1 מַה שֶׁנִּרְצֶה לְחַלֵּק צָרִיךְ שֶׁיּוּסַר מִמֶּנּוּ תְחִלָּה כָּל הַסְּפֵקוֹת וְשִׁתּוּפֵי הַלָּשׁוֹן.

2 מַה שֶּׁנֶּחֱלָק כְּהֹגֶן מֻכְרָח שֶׁיֵּשׁ לוֹ חֲלָקִים.


פֶּרֶק שְׁלשָׁה־עָשָׂר

הַנֶּעֱרָכִים. הַשָּׁוִים וּכְלָלֵיהֶם. הַדּוֹמִים וּכְלָלֵיהֶם.


א הַנֶּעֱרָכִים מִתְדַּמִּים אוֹ בְסוּג, אוֹ בְמִין, אוֹ בְמִקְרִים. בְּסוּג מִתְדַּמִּים הַמִּינִים, בְּמִין – הָאִישִׁים, בְּמִקְרִים – מִי שֶׁיֵּשׁ לָהֶם מַהוּת אַחַת אוֹ כַּמּוּת אַחַת, וּמִתְחַלְּקִים לְשָׁוִים וְדוֹמִים.

ב הַשָּׁוִים הֵם שֶׁמִּשְׁתַּוִּים בְּכַמּוּת אֵיכוּתָם.

דֶּרֶך מָשָׁל, אִם נֹאמַר: כָּל כָּךְ נֶאֱהָב הַחֶסֶד כְּמוֹ הַמִּשְׁפָּט, הַכַּוָּנָה בָזֶה שֶׁכָּל כָּךְ טַעַם יֵשׁ בַּחֶסֶד שֶׁיִּהְיֶה נֶאֱהָב כְּמוֹ שֶׁיֵּשׁ בַּמִּשְׁפָּט, וְנִמְצְאוּ הַחֶסֶד וְהַמִּשְׁפָּט שָׁוִים בְּכַמּוּת אֵיכוּת זֶה שֶׁבָּהֶם, דְּהַיְנוּ הֱיוֹת נֶאֱהָב.


כְּלָלֵי הַשָּׁוִים:

1 מַה שֶּׁנּוֹפֵל בְּאֶחָד מִן הַשָּׁוִים נוֹפֵל בַּחֲבֵרוֹ.

פֵּרוּשׁוֹ: זֶה צָרִיךְ שֶׁיּוּבַן בְּמַה שֶּׁנּוֹפֵל בְּאֶחָד מֵהֶם מִצַּד הַבְחָנָה שֶׁיֵּשׁ בּוֹ, שֶׁבָּה הוּא מִשְׁתַּוֶּה לַחֲבֵרוֹ.

דֶּרֶך מָשָׁל, הַמִּשְׁפָּט: כְּחֶסֶד לִהְיוֹתוֹ נֶאֱהָב יְצַוּוּ עָלָיו הַשְּׁלֵמִים, וְזֶה כִּי צִוּוּי הַשְּׁלֵמִים עָלָיו תָּלוּי בִּהְיוֹתוֹ נֶאֱהָב. אָמְנָם, אִם תֹּאמַר: לַחֶסֶד צָרִיךְ רֹךְ לֵב, – לֹא תִּגְזֹר מִפְּנֵי זֶה שֶׁלַּמִּשְׁפָּט צָרִיךְ, כִּי אֵין זֶה תָּלוּי בְּמַה שֶּׁהֵם שָׁוִים, שֶׁהוּא הֱיוֹת נֶאֱהָב, אֶלָּא בְּמַה שֶּׁיֵּשׁ לַחֶסֶד בִּפְרָט לְפִי עִנְיָנוֹ.

2 מַה שֶּׁשַּׁיָּךְ בּוֹ אֶחָד מִן הַשָּׁוִים, גַּם חֲבֵרוֹ יִהְיֶה שַׁיָּךְ בּוֹ.

דֶּרֶךְ מָשָׁל: בְּלֵב דּוֹרֵש הַטּוֹב יָנוּחַ הַחֶסֶד, אִם כֵּן גַּם הַמִּשְׁפָּט.

3 שְׁנַיִם שֶׁמִּשְׁתַּוִּים אֶל שְׁלִישִׁי, מִשְׁתַּוִּים בֵּינֵיהֶם.

ג הַדּוֹמִים הֵם אוֹ דוֹמִים בְּאֵיכוּת, אוֹ בִפְעֻלָּה וְהֶפְעֵל.

ד הַדִּמְיוֹן הוּא פָשׁוּט אוֹ מֻרְכָּב; הַפָּשׁוּט – בֵּין שְׁנַיִם, הַמֻּרְכָּב – בֵּין אַרְבָּעָה.

דֶּרֶך מָשָׁל: הַפָּשׁוּט – הָאָדָם וְהָעֵשֶׂב דּוֹמִים בְּחֻלְשָׁה, וְאֵין כָּאן אֶלָּא שְׁנֵי נוֹשְׂאִים; הַמֻּרְכָּב – הַשַּׂר בְּעַמּוֹ כְּחוֹבֵל בָּאֳנִיָּתוֹ, וְהַיְנוּ לְעִנְיַן הַהַשְׁקָפָה וְהַהַנְהָגָה, וְאֵין הַדִּמְיוֹן מִשְׁתַּלֵּם אֶלָּא בְאַרְבָּעָה נוֹשְׂאִים: שַׂר וְחוֹבֵל, עַם וָאֳנִיָּה.


כְּלָלֵי הַדּוֹמִים:

1 כָּל שְׁנַיִם דּוֹמִים צָרִיךְ שֶׁיִּהְיוּ גַם כֵּן בִּלְתִּי דוֹמִים.

פֵּרוּשׁוֹ, כִּי הַדִּמְיוֹן אֵינוֹ הֱיוֹתָם אֶחָד, אֶלָּא הֱיוֹתָם שְׁנַיִם, שֶׁיִהְיֶה דִמְיוֹן בֵּינֵיהֶם. אִם כֵּן מֻכְרָח הוּא שֶׁיִּבָּדְלוּ בְאֵיזֶה הֶבְדֵּל, וְיִתְדַּמּוּ אַחַר כָּךְ בְּאֵיזֶה דִמְיוֹן.

דֶּרֶךְ מָשָׁל, אִם תֹּאמַר: אָדָם דּוֹמֶה לְעֵשֶׂב הַשָּׂדֶה – לֹא מִפְּנֵי זֶה תִּגְזֹר אִם כֵּן צוֹמֵחַ מִן הֶעָפָר, כִּי אֵינוֹ אֶלָּא דוֹמֶה לְעֵשֶׂב בְּאֵיזֶה עִנְיָנִים, אַךְ לֹא בַכֹּל.

2 מַה שֶּׁנּוֹפֵל בְּאֶחָד מִן הַדּוֹמִים בִּבְחִינַת מַה שֶּׁמִּתְדַּמֶּה, נוֹפֵל גַּם בַּחֲבֵרוֹ.

דֶּרֶךְ מָשָׁל: הַצִּנּוֹרוֹת הַדַּקִּים וּמְבוֹאוֹת הַהַשְׂכָּלָה בִּילָדִים הֵם שָׁוִים; כְּמוֹ שֶׁאֵלֶּה לֹא יוּכְלוּ לִכָּנֵס בָּהֶם מַיִם רַבִּים בְּבַת אַחַת, כָּךְ אֵלֶּה לֹא יוּכְלוּ לִכָּנֵס בָּהֶם מֻשְׂכָּלוֹת רַבּוֹת בְּבַת אַחַת.


פֶּרֶק אַרְבָּעָה־עָשָׂר

הַבִּלְתִּי שָׁוִים, הַמְשֻׁנִּים, הַנֶּעֱרָכִים, הַיְתֵרִים, וְהַפְּחוּתִים. כְּלָלֵיהֶם. הַמֻּבְדָּלִים וּכְלָלֵיהֶם. הַמֻּחְלָטִים.


א הַבִּלְתִּי שָׁוִים הֵם אוֹתָם, שֶׁמַּה שֶּׁיֵּשׁ בְּאֶחָד אֵינוֹ בַחֲבֵרוֹ, וְנֶחֱלָקִים לִמְשֻׁנִּים וְהַפְכִּיִּים.

ב הַמְשֻׁנִּים הֵם הַמִּתְפָּרְשִׁים אֶחָד מֵחֲבֵרוֹ בְּלֹא הִתְנַגְּדוּת חָזָק, וְנֶחֱלָקִים לְנֶעֱרָכִים וּמֻחְלָטִים.

ג הַנֶּעֱרָכִים הֵם שֶׁיֵּשׁ הֶפְרֵשׁ בְּעֶרְכָּם זֶה לָזֶה, וְנֶחֱלָקִים לַחֲלוּקִים וּמֻבְדָּלִים.

ד הַחֲלוּקִים הֵם שֶׁחֲלוּקִים בְּכַמּוּת אֵיכוּתָם, וְהֵם יְתֵרִים אוֹ פְחוּתִים.


כְּלָלֵי הַיְתֵרִים:

1 מִי שֶׁיִּפֹּל בּוֹ הַיָּתֵר, יִפֹּל בּוֹ הַפָּחוֹת.

דֶּרֶך מָשָׁל: אִם הָאָדוֹן, בָּרוּךְ הוּא, בּוֹרֵא יֵשׁ מֵאַיִן, כָּל שֶׁכֵּן שֶׁיָּכוֹל לְקַיֵּם אֶת הַיֵּשׁ.

2 מַה שֶּׁלֹא יִפֹּל בַּיָּתֵר, לֹא יִפֹּל בַּפָּחוֹת בִּשְׁמִירַת הַיַּחַס.

פֵּרוּשׁוֹ, נֶאֱמַר דֶּרֶך מָשָׁל: בִּקְדוֹשָׁיו לֹא יַאֲמִין נִגְזַר מִזֶּה – כָּל שֶׁכֵּן בִּבְנֵי הָאָדָם. וּפֹה נִשְׁמַר הַיַּחַס שֶׁבֵּין לִבְנֵי אָדָם בֵּין לִקְדוֹשִׁים, מַה שֶּׁנְּקַיֵּם יִהְיֶה חֶסְרוֹן הַאֲמָנָתוֹ, יִתְבָּרַךְ בָּהֶם.


כְּלָלֵי הַפָּחוֹת:

1 מִי שֶׁלֹּא יִפֹּל בּוֹ פָּחוֹת, לֹא יִפֹּל בּוֹ יָתֵר.

2 מַה שֶּׁמְחֻיָּב בַּפָּחוֹת, כָּל שֶׁכֵּן בַּיָּתֵר.

ה עוֹד צָרִיךְ לְהַבְחִין בֵּין הֱיוֹת הַדָּבָר פָּחוֹת אוֹ יָתֵר וּבֵין הֱיוֹתוֹ יוֹתֵר אֶפְשָׁרִי, אוֹ לֹא כָּל כָּךְ אֶפְשָׁרִי.

פֵּרוּשׁוֹ, אִם נֹאמַר: יָתֵר הוּא הַמְצִיא דָבָר חָדָשׁ שֶׁלֹּא יְדָעוּהוּ הָרִאשׁוֹנִים – מְבָאֵר יָפֶה מַה שֶּׁיְדָעוּהוּ הָרִאשׁוֹנִים. נִמְצָא שֶׁהַחִדּוּשׁ יָתֵר וְהַבֵּאוּר פָּחוֹת. אַךְ לֹא תִגְזֹר מִזֶּה שֶׁהַחִדּוּשׁ יוֹתֵר אֶפְשָׁרִי, לְפִי שֶׁהוּא יָתֵר וְהַבֵּאוּר לֹא כָּל כָּךְ, לְפִי שֶׁכָּל פָּחוֹת בְּעַצְמוֹ הוּא יוֹתֵר אֶפְשָׁרִי, וְהַחִדּוּשׁ שֶׁהוּא יָתֵר בְּעַצְמוֹ אֵינוֹ כָּל כָּךְ אֶפְשָׁרִי.

ו הַמֻּבְדָּלִים הֵם הַנֶּעֱרָכִים הַמִּתְפָּרְשִׁים זֶה מִזֶּה בְּאֵיכוּתָם וּבִפְעֻלָה וְהֶפְעֵל.

דֶּרֶךְ מָשָׁל, אִם נֹאמַר: מִצְרַיִם אֵינָהּ כִּשְׁכֶם. הָאָדָם אֵינוֹ כַּחוֹמָה.


כְּלָלֵי הַמֻּבְדָּלִים:

מַה שֶּׁנּוֹפֵל בְּאֶחָד מֵהֶם, אֵינוֹ נוֹפֵל בַּחֲבֵרוֹ בִּבְחִינַת מַה שֶּׁהוּא מֻבְדָּל מִמֶּנּוּ.

ז הַמֻּחְלָטִים הֵם מַה שֶּׁנִּשְׁלַל מִן אֵיזֶה נוֹשֵׂא פְּרָטִי בְּלֹא עֵרֶךְ.

דֶּרֶךְ מָשָׁל: אֵיזֶה אָדָם אֵינוֹ לָבָן.


פֶּרֶק חֲמִשָּׁה־עָשָׂר

הַהַפְכִּיִּים. כְּלָלֵיהֶם. הַנֶּגְדִּיִּים וּכְלָלֵיהֶם. הַמַּעְדִּירִים וּכְלָלֵיהֶם. הַנִּבְחָנִים. הַמִּתְנַגְּדִים וּכְלָלֵיהֶם. הַמִּצְטָרְפִים וּכְלָלֵיהֶם.


א הַהַפְכִּיִּים הֵם שֶׁמִּתְפָּרְשִׁים זֶה מִזֶּה בְּהִתְנַגְּדוּת חָזָק שֶׁאִי אֶפְשָׁר לְיַחֲדָם.

דֶּרֶךְ מָשָׁל: שַׁחֲרוּת וָלֹבֶן.

ב מַה שֶּׁיֵּשׁ לְהַבְחִין בַּהַפְכִּיִּים הוּא הַסְּגֻלָּה וְהַמִּין.

ג סְגֻלַּת הַהַפְכִּיִּים שֶׁלֹּא יִמָּצְאוּ בְנוֹשֵׂא אֶחָד, בִּבְחִינָה אַחַת, בְּחֵלֶק אֶחָד, בִּזְמַן אֶחָד.

פֵּרוּשׁוֹ, כִּי בְחֶסְרוֹן אֶחָד מֵאֵלֶּה הַתְּנָאִים יוּכְלוּ לִמָּצֵא.

דֶּרֶךְ מָשָׁל: יוּכַל אָדָם לִהְיוֹת לָבָן לְגַבֵּי מִי שֶׁהוּא שָׁחוֹר מִמֶּנּוּ, וְלִהְיוֹת שָׁחוֹר לְגַבֵּי מִי שֶׁלָבָן מִמֶּנּוּ, וְזֶהוּ בִּבְחִינוֹת שׁוֹנוֹת; עוֹד יוּכַל אָדָם לִהְיוֹת שָׁחוֹר דֶּרֶךְ מָשָׁל בְּעוֹרוֹ וְלָבָן בְּשִׁנָּיו, וְזֶהוּ בַחֲלָקִים שׁוֹנִים; עוֹד יוּכַל לִהְיוֹת שָׁחוֹר בִּזְמַן אֶחָד וְלָבָן בִּזְמַן אַחֵר, וְזֶהוּ בִזְמַנִּים שׁוֹנִים.


כְּלָלֵי הַהַפְכִּיִּים:

1 בְּמַה שֶּׁיְּקֻיַּם אֶחָד מִן הַהַפְכִּיִּים, יְשׁוֹלַל חֲבֵרוֹ.

2 בְּמַה שֶּׁיְּשׁוֹלַל אֶחָד מִן הַהַפְכִּיִּים, אֵינוֹ מֻכְרָח שֶׁיְּקֻיַּם חֲבֵרוֹ אִם לֹא יִהְיוּ הַפְכִּיִּים בִּלְתִּי מִתְמַצְּעִים.

דֶּרֶךְ מָשָׁל, אִם נֹאמַר: זֶה הַדַּף אֵינוֹ לָבָן, – לֹא יְחֻיַּב הֱיוֹתוֹ שָׁחוֹר, כִּי יָכוֹל לִהְיוֹתוֹ יָרֹק. אַךְ אִם תֹּאמַר: עַכְשָׁו אֵינוֹ יוֹם – מֻכְרָח שֶׁיִּהְיֶה לַיְלָה, כִּי אֵין אֶמְצַע בֵּינֵיהֶם.

3 הַהַפְכִּיּוּת צָרִיךְ שֶׁיִּבָּחֵן לְפִי הַהֶבְדֵּל, לֹא לְפִי הַשִּׁתּוּף.

דֶּרֶך מָשָׁל, אִם נֹאמַר: הָאָדָם מַרְגִּישׁ, – לֹא נִגְזַר אִם כֵּן: הַבַּעַל חַי אֵינוֹ מַרְגִּישׁ, כִּי הַהֶרְגֵּשׁ מְשֻׁתָּף לָזֶה וְלָזֶה; אַךְ אִם נֹאמַר: הָאָדָם מַמְצִיא מְלָאכוֹת, – נוּכַל לִגְזֹר מִזֶּה אִם כֵּן: הַבַּעַל חַי אֵינוֹ מַמְצִיא, כִּי זֶה דָבָר תָּלוּי בַּשֵּׂכֶל, שֶׁהוּא הֶבְדֵּלוֹ שֶׁל הָאָדָם.

ד מִינֵי הַהַפְכִּיִּים חֲמִשָּׁה: נֶגְדִּיִּים, מַעְדִּירִים אוֹ שׁוֹלְלְים, נִבְחָנִים, מִתְנַגְּדִים, מִצְטָרְפִים.

ה הַנֶּגְדִּיִּים הֵם שֶׁאֶחָד מְקַיֵּם וְאֶחָד שׁוֹלֵל.

דֶּרֶך מָשָׁל: הוֹלֵךְ, לֹא הוֹלֵךְ.


כְּלָלֵי הַנֶּגְדִּיִּים:

1 הַנֶּגְדִּיִּים אֵין לָהֶם אֶמְצָעִי.

פֵּרוּשׁוֹ, כִּי אֵין בֵּין שְׁלִילָה לְקִיּוּם אֶמְצָעִי.

2 בְּהַנִּיחַ אֶחָד מִן הַנֶּגְדִּיִּים, בְּהֶכְרַח יְשׁוֹלַל הַשֵּׁנִי.

3 הַנָּשׂוּא שֶׁל הַמַּאֲמָר הַנֶּגְדִּיִּי לֹא יִתָּכֵן עִם נוֹשְׂאוֹ לְעוֹלָם.

דֶּרֶךְ מָשָׁל, יִהְיֶה הַמַּאֲמָר: אָדָם אֵינוֹ לֹא אָדָם אָדָם וְלֹא אָדָם, – אִי אֶפְשָׁר לָהֶם לְהִתְחַבֵּר בְּשׁוּם פָּנִים.

ו הַמַּעְדִּירִים הֵם הַמּוֹרִים עַל קִנְיָן וְהֶעְדֵּרוֹ, וְיִתְחַלְּקוּ לִגְמוּרִים וִּבְלְתִּי גְמוּרִים.

ז הַגְּמוּרִים – הַמַּעְדִּירִים אֲפִילוּ הַכֹּח.

דֶּרֶך מָשָׁל: הָעִוָּרוֹן בְּמִי שֶׁנִּקְּרוּ שְׁתֵּי עֵינָיו.

ח הַבִּלְתִּי גְמוּרִים הֵם שֶׁמַּעְדִּירִים פּוֹעֵל הַקִּנְיָן, אַךְ הַכֹּח נִשְׁאָר בַּנּוֹשֵׂא.

דֶּרֶךְ מָשָׁל: הַשֵּׁנָה מַעְדֶּרֶת הַיְקִיצָה בַּפּוֹעֵל, אַךְ הַכֹּחַ נִשְׁאָר בַּפּוֹעֵל שֶׁיֵּעוֹר.


כְּלָלֵי הַהֶעְדֵּר:

1 הַקִּנְיָן טוֹב מִן הַהֶעְדֵּר.

2 הַקִּנְיָן וְהַהֶעְדֵּר סוֹבְבִים עַל נוֹשֵׂא אֶחָד.

3 מִי שֶׁשַּׁיָּךְ בּוֹ הַקִּנְיָן, שַׁיָּךְ בּוֹ הַהֶעְדֵּר, וּמִי שֶׁאֵין הַקִּנְיָן שַׁיָּךְ בּוֹ – אֵין הַהֶעְדֵּר שַׁיָּךְ בּוֹ.

דֶּרֶךְ מָשָׁל: אָדָם שֶׁשַּׁיָּךְ בּוֹ הָרְאִיָּה, שַׁיָּךְ בּוֹ הָעִוָּרוֹן; הַכֹּתֶל שֶׁאֵין שַׁיָּךְ בּוֹ הָרְאִיָּה, אֵין שַׁיָּךְ בּוֹ הָעִוָּרוֹן.

4 כְּשֶׁיֻּנַּח אֶחָד מֵהֶם, יְשׁוֹלַל חֲבֵרוֹ בְּהֶכְרַח.

ט הַנִּבְחָנִים הֵם הַמִּינִים הַחֲלוּקִים שֶׁתַּחַת סוּג אֶחָד, אוֹ אִישִׁים תַּחַת מִין אֶחָד.

י הַמִּתְנַגְּדִים הֵם שֶׁבְּאֵיכוּתָם אוֹ בִפְעֻלָּתָם מִתְנַגְּדִים זֶה לָזֶה.

דֶּרֶךְ מָשָׁל: הַחֹם וְהַקֹּר, הַחַם וְהַקַּר.

יא הַמִּתְנַגְּדִים מִתְחַלְּקִים לְמִתְמַצְּעִים וְלֹא מִתְמַצְּעִים. הַמִּתְמַצְּעִים – שֶׁיֵּשׁ בֵּינֵיהֶם אֶמְצָעִי, הַלֹּא־מִתְמַצְּעִים – שֶׁאֵין בֵּינֵיהֶם אֶמְצָעִי.

דֶּרֶךְ מָשָׁל: הַחֹם וְהַקֹּר יֵשׁ בֵּינֵיהֶם הַפּוֹשֵׁר, אַךְ הַיּוֹם וְהַלַּיְלָה – אֵין בֵּינֵיהֶם אֶמְצָע.


כְּלָלֵי הַמִּתְנַגְּדִים:

1 אִם מִתְנַגֵּד אֶחָד יְנַצֵּח לְגַמְרֵי, יְבַטֵּל אֶת חֲבֵרוֹ לְגַמְרֵי.

דֶּרֶךְ מָשָׁל: אִם הַחֹם יְנַצֵּחַ לְגַמְרֵי, יְבַטֵּל הַקֹּר לְגַמְרֵי.

2 הַמִּתְנַגְּדִים – שְׁנֵיהֶם מְקַיְּמִים וְלֹא שׁוֹלְלִים.

פֵּרוּשׁוֹ, כִּי אִם תֹּאמַר: לֹא חַם אֵינוֹ מִתְנַגֵּד לְלֹא קַר, – שֶׁכְּבָר יָכוֹל לִהְיוֹת לֹא חַם וְלֹא קַר, אַךְ חַם וְקַר אִי אֶפְשָׁר לִהְיוֹת.

3 הַמִּתְנַגְּדִים – סִבּוֹתֵיהֶם וּמְסֻבְּבֵיהֶם, סְגֻלּוֹתֵיהֶם וְנוֹשְׂאֵיהֶם כֻּלָּם מִתְנַגְּדִים.

יב הַמִּצְטָרְפִים הֵם הַמִּתְיַחֲסִים שֶׁאִי אֶפְשָׁר לְאֶחָד לִהְיוֹת בְּלֹא חֲבֵרוֹ, וְהֵם הַפְכִּיִּים זֶה לָזֶה.

דֶּרֶךְ מָשָׁל: הָאָב הֵפֶךְ הַבֵּן, וְהַבֵּן הֵפֶךְ הָאָב.


כְּלָלֵי הַמִּצְטָרְפִים:

1 יְדִיעַת הַמִּצְטָרְפִים אֶחָד, כִּי אֵין אֶחָד בְּלֹא חֲבֵרוֹ.

2 הַמִּצְטָרְפִים בִּבְחִינַת הִצְטָרְפָם הֵם כְּמוֹ סִבָּה זֶה לָזֶה.

3 בְּהֵעָדֵר אֶחָד יֵעָדֵר חֲבֵרוֹ.

4 בְּמַה שֶּׁיְּקֻיַּם הָאֶחָד יְשׁוֹלַל חֲבֵרוֹ.


פֶּרֶק שִׁשָּׁה־עָשָׂר

הָעֵדוּת וְהַמּוּעָד. כְּלָלֵי הָעֵדוּת.


א עֵדוּת וּמוּעָד הוּא קִיּוּם דָּבָר מִחוּץ מִמֶּנּוּ, וְיִתְחַלֵּק לֶאֱלֹקִי וֶאֱנוֹשׁי.

ב הָאֱלֹקִי – שֶׁדִּבְרֵי הָאֵל, יִתְבָּרַךְ, מְקַיְּמִים אוֹתוֹ.

ג הָאֱנוֹשִׁי – שֶׁאָדָם מְקַיְּמוֹ, וְיִתְחַלֵּק לִמְרֻבֶּה וּמְיֻחָד. הַמְּרֻבֶּה – שֶׁהַרְבֵּה מְעִידִים עָלָיו, הַמְּיֻחָד – שֶׁאֶחָד מֵעִיד עָלָיו.


כְּלָלֵי הָעֵדוּת:

כָּעֵד כָּךְ כֹּח הָעֵדוּת.

פֵּרוּשׁוֹ: כְּעֵרֶךְ הָעֵד וַחֲשִׁיבוּתוֹ, כָּךְ כֹּחַ הָעֵדוּת שֶׁלּוֹ לְשֶׁנַּאֲמִין בָּהּ.


פֶּרֶק שִׁבְעָה־עָשָׂר

הַמַּאֲמָר הַפָּשׁוּט וְהַמֻּרְכָּב. הַנּוֹשֵׂא וְהַנָּשׂוּא. הַקֶּשֶׁר. הַמַּאֲמָר הַכְּלָלִי, הַחֶלְקִיִּי וְהָאִישִּיִּי. הַמְּחַיֵּב וְהַשּׁוֹלֵל, הַסְּתָמִיִּי וְהַתְּנָאִיִּי. הַמְּמַעֵט, הַמּוֹצִיא וְהַמַּגְבִּיל. הָאֲמִתִּי וְהַכּוֹזֵב. הַמֻּכְרָח, הָאֶפְשָׁרִי, הַמְּדֻקְדָּק וִּבִלְתִּי מְדֻקְדָּק.


א הַמַּאֲמָר הוּא חִבּוּר דְּבָרִים, יְקֻיַּם בּוֹ דָּבָר בְּדָבָר, אוֹ יְשׁוֹלַל דָּבָר מִדָּבָר.

דֶּרֶךְ מָשָׁל: הַלֶּחֶם הוּא לָבָן, – קַיָּם הַלֹּבֶן בַּלֶּחֶם; הַלֶּחֶם אֵינוֹ שָׁחוֹר, – שָׁלַל הַשַּׁחֲרוּת מִמֶּנוּ.

ב הַמַּאֲמָר יִתְחַלֵּק לְפָשׁוּט וּמֻרְכָּב.

ג הַפָּשׁוּט הוּא הַמַּאֲמָר שֶׁיְקַיֵּם אוֹ יִשְׁלֹל דְּבַר מָה. וְאַרְבַּע הַבְחָנוֹת יֵשׁ לְהַבְחִין בּוֹ, וְהֵם: הַחֲלָקִים, הַקֶּשֶׁר, הַמִּין וְהַמִּקְרִים.

ד חֶלְקֵי הַמַּאֲמָרִים – נוֹשֵׂא וְנָשׂוּא.

ה הַנּוֹשֵׂא הוּא שֶׁיְקֻיַּם בּוֹ הַדָּבָר הַמְקֻיָּם, אוֹ יְשׁוֹלַל הַמְשׁוֹלָל.

ו הַנָּשׂוּא הוּא אֲשֶׁר יְקֻיַּם אוֹ יְשׁוֹלַל בַּנּוֹשֵׂא.

דֶּרֶךְ מָשָׁל: הַלֶּחֶם הוּא לָבָן. לֶחֶם – נוֹשֵׂא, לָבָן – נָשׂוּא.

ז הַקֶּשֶׁר הוּא מַה שֶּׁיְקַשֵּׁר הַנָּשׂוּא בַנּוֹשֵׂא, וְיוּכַל לִהְיוֹת מְפֹרָשׁ, וְיוּכַל לִהְיוֹת נֶעְלָם.

הַמְפֹרָשׁ, שֶׁתִּהְיֶה מִלַּת הוּא אוֹ מַה שֶּׁיּוֹרֶה עָלֶיהָ בֵּין הַנּוֹשֵׂא וְהַנָּשׂוּא, וְהַנֶּעְלָם – שֶׁיִּהְיֶה הַנָּשׂוּא פּוֹעֵל.

מְשַׁל הַמְפֹרָשׁ: הַלֶּחֶם הוּא לָבָן; מְשַׁל הַנֶּעְלָם: רְאוּבֵן אָכַל, – יוּבַן בּוֹ, רְאוּבֵן פָּעַל פְּעֻלַּת הָאֲכִילָה.

ח הַמִּין, בּוֹ יִבָּחֵן הַמַּאֲמָר לְפִי הַכַּמּוּת וּלְפִי הָאֵיכוּת. לְפִי הַכַּמּוּת יִהְיוּ בּוֹ שְׁלֹשָׁה מִינִים: כְּלָלִי, חֶלְקִיִּי וְאִישִׁיִּי.

ט הַכְּלָלִי הוּא, שֶׁנּוֹשְׂאוֹ כּוֹלֵל כָּל שֶמֵּעִנְיָנוֹ, וְיֹאמַר הַנָּשׂוּא בְכֻלָּם.

דֶּרֶךְ מָשָׁל: כָּל בַּעַל חַי מַרְגִּישׁ, הַנּוֹשֵׂא כּוֹלֵל כָּל מִי שֶׁיֵּשׁ בּוֹ עִנְיַן הַחִיּוּת, וְיֹאמַר הַנָּשׂוּא שֶׁהוּא מַרְגִּישׁ בְּכֻלָּם.

י הַחֶלְקִיִּי הוּא, שֶׁנּוֹשְׂאוֹ כּוֹלֵל קְצָת מִמַּה שֶּׁמֵּעִנְיָנוֹ שֶׁבָּהֶם יֹאמַר הַנָּשׂוּא.

דֶּרֶךְ מָשָׁל: קְצָת בַּעֲלֵי חַיִּים מְדַבְּרִים.

יא הָאִישִׁיִּי, שֶׁנּוֹשְׂאוֹ אִישׁ.

דֶּרֶךְ מָשָׁל: יְרוּשָׁלַיִם הִיא עִיר יָפָה.

יב לְפִי הָאֵיכוּת יִהְיוּ מִינֵי הַמַּאֲמָר שְׁנַיִם: מְחַיֵּב אוֹ שׁוֹלֵל. הַמְּחַיֵּב הוּא שֶׁיְחַיֵּב הַנָּשׂוּא לַנּוֹשֵׂא, הַשּׁוֹלֵל יִשְׁלְלֵהוּ מִמֶּנּוּ.

יג כְּלָל הַמַּאֲמָרִים מִתְחַלֵּק לִשְׁנֵי מִינִים: סְתָמִיִּי וּתְנָאִיִּי.

יד סְתָמִיִּי הוּא, שֶׁיְחַיֵּב אוֹ יִשְׁלֹל בְּלִי שׁוּם תְּנַאי; תְּנָאִיִּי – שֶׁיִּהְיֶה תְנַאי עִם הַחִיּוּב אוֹ הַשְּׁלִילָה, וְיִתְחַלֵּק לְרִאשׁוֹן וְשֵׁנִי.

טו הַתְּנָאִיִּי הָרִאשׁוֹן הוּא, שֶׁיִּהְיֶה בוֹ אֶחָד מֵאַרְבָּעָה, וְהֵם: הֶכְרַח, מְנִיעָה, אֶפְשָׁרוּת, שִׁקּוּל.

דֶּרֶךְ מָשָׁל: הֶכְרַח – הָאֵשׁ מֻכְרָח שֶׁיִּשְׂרֹף, מְנִיעָה – הַחֵלֶק נִמְנָע שֶׁיִּהְיֶה כְּכֹל, אֶפְשָׁרוּת – הָאָדָם אֶפְשָׁר שֶׁיִּחְיֶה הַרְבֵּה, שִׁקּוּל – הָאָדָם שָׁקוּל לְשֶׁיֵּשֵׁב אוֹ לֹא יֵשֵׁב.

טז הַתְּנָאִיִּי הַשֵּׁנִי יִתְחַלֵּק לִשְׁלֹשָׁה מִינִים: תְּנָאִיִּי מְמַעֵט, תְּנָאִיִּי מוֹצִיא, תְּנָאִיִּי מַגְבִּיל.

מְמַעֵט הוּא, שֶׁמְּמַעֵט מֵהַנּוֹשֵׂא אוֹ הַנָּשׂוּא כָּל זוּלָתָם. וְנַעֲשֶׂה בְּמִלַּת בִּלְבַד, אוֹ לְבַדּוֹ, וְכַיּוֹצֵא בָהֶם. מוֹצִיא הוּא, שֶׁמּוֹצִיא מִן הַנּוֹשֵׂא אוֹ מִן הַנָּשׂוּא מָה, וְזֶה בְּמִלַּת חוּץ וְכַיּוֹצֵא בוֹ. מַגְבִּיל הוּא, שֶׁמַּגְבִּיל הַנָּשׂוּא בַנּוֹשֵׂא בְּאֵיזֶה בְחִינָה, וְנַעֲשֶׂה בְּמִלַּת לְפִי, אוֹ בְּעֵרֶךְ, אוֹ לְגַבֵּי, וְכַיּוֹצֵא.

מְשַׁל הַמְמַעֵט: הָאָדָם לְבַדּוֹ מְדַבֵּר. מוֹצִיא: כָּל שְׁתִיקָה טוֹבָה חוּץ מִדִּבְרֵי תוֹרָה.

מְשַׁל הַמַּגְבִּיל: הָאָדָם הוּא שָׁפָל בְּעֵרֶךְ הַמַּלְאָכִים.

יז הַמַּאֲמָר יוּכַל לִהְיוֹת אֲמִתִּי אוֹ כוֹזֵב.

הָאֲמִתִּי הוּא, שֶׁיַּגִּיד הַדָּבָר כְּמוֹת שֶׁהוּא, כּוֹזֵב – שֶׁיַּגִּיד כְּמוֹת שֶׁאֵינוֹ. וּבִשְׁנֵיהֶם יוּכַל לִהְיוֹת מֻכְרָח אוֹ אֶפְשָׁרִי, מְדֻקְדָּק אוֹ לֹא מְדֻקְדָּק.

יח הַמֻכְרָח הוּא שֶׁיִּהְיֶה אֱמֶת שֶׁאִי אֶפְשָׁר לִהְיוֹת כָּזָב, אוֹ כָּזָב שֶׁאִי אֶפְשָׁר לִהְיוֹת אֱמֶת.

דֶּרֶךְ מָשָׁל, אִם תֹּאמַר: הַחַי מֻכְרָח שֶׁיַּרְגִּישׁ.

יט הָאֶפְשָׁרִי שֶׁהוּא אֱמֶת וְיָכוֹל לִהְיוֹת שֶׁקֶר, אוֹ לְהֵפֶךְ.

דֶּרֶךְ מָשָׁל: הָאָדָם אֶפְשָׁר שֶׁיִּחְיֶה הַרְבֵּה.

כ מְדֻקְדָּק – מַאֲמָר שֶׁיּוֹרֶה בְּמִלּוֹתָיו כְּפִי הוֹרָאָתָם בַּדִּקְדּוּק, וְלֹא מְדֻקְדָּק הוּא – שֶׁיּוֹרֶה בְּמִלּוֹתָיו שֶׁלֹא כְפִי הוֹרָאָתָם בַּדִּקְדּוּק.

דֶּרֶךְ מָשָׁל, אִם תֹּאמַר עַל גִּבּוֹר אֶחָד: זֶה הָאִישׁ הוּא אַרְיֵה, – לֹא תָבִין אַרְיֵה מַמָּשׁ, אֶלָּא לִבּוֹ כְּלֵב אַרְיֵה.


פֶּרֶק שְׁמוֹנָה־עָשָׂר

הַחִלּוּף הַפָּשׁוּט, הַמִּקְרִי וְהַהַפְכִּי. כְּלָלֵי הַחִלּוּף. הַנִּגּוּד, תְּנָאֵי הַנִּגּוּד, מִינָיו וּכְלָלָיו.


א שִׁנּוּי הַמַּאֲמָר שְׁנַיִם: הָאֶחָד חִלּוּף וְהָאַחֵר נִגּוּד.

חִלּוּף הַמַּאֲמָר הוּא שׂוֹם הַנּוֹשֵׂא בִּמְקוֹם הַנָּשׂוּא וְהַנָּשׂוּא בִּמְקוֹם הַנּוֹשֵׂא, וְיִהְיֶה הַמַּאֲמָר אֲמִתִּי. וְיֵשׁ בּוֹ שְׁלֹשָׁה מִינִים: פְּשׁוּטִי, מִקְרִי וְהַפְכִּיִּי.

ב הַפָּשׁוּט הוּא חִלּוּף הַנּוֹשֵׂא בַנָּשׂוּא וְנִשְׁמָר הַכַּמּוּת וְהָאֵיכוּת.

פֵּרוּשׁוֹ, שֶׁנִּשְׁאָר מְחַיֵּב אִם הָיָה מְחַיֵּב, אוֹ שׁוֹלֵל אִם הָיָה שׁוֹלֵל. וְכֵן בַּכַּמּוּת.

דֶּרֶךְ מָשָׁל: כָּל אָדָם הוּא בַעַל חַי מְדַבֵּר; כָּל בַּעַל חַי מְדַבֵּר הוּא אָדָם.

ג מִקְרִי שֶׁיִּשְׁמֹר הָאֵיכוּת וְלֹא הַכַּמּוּת.

דֶּרֶךְ מָשָׁל: כָּל מְדַבֵּר הוּא חַי; אֵיזֶה חַי הוּא מְדַבֵּר. כִּי הָרִאשׁוֹן מַאֲמָר כְּלָלִי, וְהַשֵּׁנִי – חֶלְקִיִּי.

ד הַפְכִּיִּי – שֶׁיִּשְׁמֹר הַכַּמּוּת וְלֹא הָאֵיכוּת.

דֶּרֶךְ מָשָׁל: כָּל מְדַבֵּר הוּא בַעַל חַי; כָּל שֶׁאֵינוֹ בַעַל חַי אֵינוֹ מְדַבֵּר.


כְּלָלֵי הַחִלּוּף:

1 חִלּוּף הַכְּלָלִי הַמְחַיֵּב – חֶלְקִיִּי מְחַיֵּב וּכְלָלִי שׁוֹלֵל.

2 חִלּוּף הַחֶלְקִיִּי הַמְּחַיֵּב – חֶלְקִיִּי מְחַיֵּב.

3 חִלּוּף הַכְּלָלִי הַשּׁוֹלֵל – כְּלָלִי שׁוֹלֵל.

4 הַחֶלְקִי שׁוֹלֵל אֵין חִלּוּף מֻכְרָח לוֹ.

דֶּרֶךְ מָשָׁל: כָּל אָדָם חַי; אֵיזֶה חַי אָדָם; כָּל שֶׁאֵינוֹ חַי אֵינוֹ אָדָם.

אֵיזֶה אָדָם חַי; אֵיזֶה חַי אָדָם.

אֵין שׁוּם אָדָם אֶבֶן; אֵין שׁוּם אֶבֶן אָדָם.

אֵיזֶה אָדָם אֵינוֹ חָכָם, – אֵין חִלּוּפוֹ מֻכְרָח, כִּי אִם הַהַפְכִּיִּי.

ה הַנִּגּוּד הוּא מַאֲמָר נֶגְדִּיּי לַמַּאֲמָר.

פֵּרוּשׁוֹ, שֶׁהָאֶחָד יְקַיֵּם וְהָאֶחָד יִשְׁלֹל דָּבָר אֶחָד בְּנוֹשֵׂא אֶחָד.

ו הַבְחָנוֹת הַנִּגּוּד שְׁתַּיִם: הַתְּנָאִים וְהַמִּין.

ז תְּנָאֵי הַנִּגּוּד שְׁלשָׁה:

1 הַמַּאֲמָרִים הַנֶּגְדִּיִים צָרִיךְ שֶׁיִּהְיֶה נוֹשְׂאֵי שְׁנֵיהֶם וּנְשׂוּאֵיהֶם אֶחָד.

דֶּרֶךְ מָשָׁל: אָדָם הוּא בַעַל חַי; אָדָם אֵינוֹ בַעַל חַי.

2 צָרִיךְ שֶׁיּוּבְנוּ בְּנוֹשֵׂא אֶחָד, בִּבְחִינָה אַחַת, בְּמָקוֹם אֶחָד, בִּזְמַן אֶחָד.

דֶּרֶךְ מָשָׁל: יִצְחָק הָיוּ לוֹ בָנִים; יִצְחָק לֹא הָיוּ לוֹ בָנִים, – אֵינָם נֶגְדִּיִּים, כִּי יוּבְנוּ בִּשְׁנֵי זְמַנִּים, קֹדֶם תְּפִלָּתוֹ וְאַחַר תְּפִלָּתוֹ. וְכֵן: הַזְּרָעִים צוֹמְחִים עַל יְדֵי הַמָּטָר; הַצְּמָחִים אֵינָם צוֹמְחִים עַל יְדֵי הַמָּטָר, – יוּבְנוּ בְּכָל הָעוֹלָם וּבְמִצְרָיִם.

3 הַנֶּגְדִּיִּים צָרִיךְ שֶׁיּוּבְנוּ בַּהֲבָנָה פְּשׁוּטָה בְּלִי שׁוּם שִׁתּוּף.

דֶּרֶךְ מָשָׁל: סַרְטָן הוּא בַעַל חַי; סַרְטָן אֵינוֹ בַעַל חַי, – יוּבַן הַדָּג סַרְטָן בָּרִאשׁוֹן וְהַמַּזָּל בַּשֵּׁנִי.

ה מִינֵי הַנֶּגְדִּיִּים שְׁנַיִם, וְהֵם: נִגּוּד גָּמוּר וּבִלְתִּי גָמוּר.

הַגָּמוּר הוּא, שֶׁבְּהֶכְרַח אֶחָד מֵהֶם שֶׁקֶר וְאֶחָד מֵהֶם אֱמֶת. וּמִתְחַלֵּק לִשְׁנַיִם: רִאשׁוֹן וְשֵׁנִי.

ו הָרִאשׁוֹן הוּא נִגּוּד בֵּין כְּלָלִי וּפְרָטִי, אֶחָד מְחַיֵּב וְאֶחָד שׁוֹלֵל; הַשֵּׁנִי – בֵּין שְׁנֵי אִישִׁים.

מָשָׁל, הָרִאשׁוֹן – כָּל אָדָם לָבָן; אֵיזֶה אָדָם אֵינוֹ לָבָן. הַשֵּׁנִי – רְאוּבֵן לָבָן; רְאוּבֵן אֵינוֹ לָבָן.


כְּלָלֵי הַנִּגּוּד הַגָּמוּר:

שְׁנֵי נֶגְדִּיִּים גְּמוּרִים אִי־אֶפְשָׁר לִהְיוֹת לֹא שְׁנֵיהֶם שֶׁקֶר וְלֹא שְׁנֵיהֶם אֱמֶת.

הַבִּלְתִּי גָמוּר הוּא נִגּוּד בֵּין שְׁנֵי כְלָלִים.

דֶּרֶךְ מָשָׁל: כָּל אָדָם הוּא צַדִּיק; שׁוּם אָדָם אֵינוֹ צַדִּיק.


כְּלָלֵי הַבִּלְתִּי גָמוּר:

שְׁנֵי נֶגְדִּיִּים בִּלְתִּי גְמוּרִים אִי־אֶפְשָׁר לִהְיוֹת אֱמֶת וִיכוֹלִים לִהְיוֹת שֶׁקֶר.


פֶּרֶק תִּשְׁעָה־עָשָׂר

הַמַּאֲמָר הַתְּנָאִיִּי, הַמְחַבֵּר, הַמְחַלֵּק, הַמַּבְדִּיל וְהַמִּתְיַחֵס.


א הַמַּאֲמָר הַמֻּרְכָּב הוּא שֶׁעִנְיָנוֹ מֻרְכָּב, וְיִתְחַלֵּק לַחֲמִשָּׁה מִינִים: תְּנָאִיִּי, מְחַבֵּר, מְחַלֵּק, מַבְדִּיל, מִתְיַחֵס.

ב חֶלְקֵי הַתְּנָאִיִּי שְׁנַיִם: קוֹדֵם וְנִמְשָׁךְ; הַחֵלֶק שֶׁבּוֹ הַתְּנַאי – קוֹדֵם, הַשֵּׁנִי – נִמְשָׁךְ.

דֶּרֶךְ מָשָׁל, מַאֲמָר תְּנָאִיִּי הוּא: אִם תִּיגַע בַּלִּמּוּדִים תִּתְחַכֵּם וַדַּאי. אִם תִּיגַע בַּלִּמוּדִים – הוּא הַקּוֹדֵם, תִּתְחַכֵּם, וַדַּאי – הוּא הַנִּמְשָׁךְ.

ג חִיּוּבֵי הַתְּנַאי שְׁנַיִם:

1 שֶׁכָּל הַקִּיּוּם אוֹ הַשְּׁלִילָה יִהְיֶה תָלוּי בַּתְּנָאי.

דֶּרֶךְ מָשָׁל, הַמַּאֲמָר שֶׁזָּכַרְנוּ: אִם תִּיגַע וְכוּ' אֵינוֹ מְקַיֵּם שֶׁתִּיגַע בַּלִּמּוּדִים וְגַם לֹא שֶׁתִּתְחַכֵּם וַדַּאי, אֶלָּא מַנִּיחַ הַנִּמְשָׁךְ בְּהַנָּחַת הַקּוֹדֵם, דְּהַיְנוּ: שֶׁאִם תִּיגַע בַּלִּמּוּדִים יִמָּשֵׁךְ מִזֶּה שֶׁתִּתְחַכֵּם וַדַּאי.

2 שֶׁיִּהְיֶה אֱמֶת, וְלֹא מִצַּד הַחֲלָקִים בִּפְנֵי עַצְמָם, אֶלָּא מִצַּד קִשְׁרָם הַתְּנָאִיִּי.

דֶּרֶךְ מָשָׁל, אִם תֹּאמַר: אִם הָאָדָם יֵשׁ לוֹ שְׁתֵּי רַגְלַיִם, הוּא מְדַבֵּר, – הִנֵּה שְׁנֵי הַחֲלָקִים כָּל אֶחָד בִּפְנֵי עַצְמוֹ אֱמֶת, אַךְ קִשְׁרָם שֶׁקֶר, כִּי לֹא יִמָּשֵׁך מִהְיוֹת לוֹ שְׁתֵּי רַגְלַיִם שֶׁיְדַבֵּר. וְאִם תֹּאמַר: אִם הַכֹּתֶל יֵשׁ לוֹ עֵינַיִם, וַדַּאי יִרְאֶה, – הִנֵּה שְׁנֵי הַחֲלָקִים שֶׁקֶר וּמַאֲמָרְךָ אֱמֶת, כִּי בְהַנִּיחַ שֶׁיִּהְיֶה לוֹ עֵינַיִם יִמָּשֵׁךְ שֶׁיִּרְאֶה.

ד הַמְחֻבָּר הוּא שֶׁיִּתְחַבְּרוּ חֲלָקָיו בִּקְשָׁרִים אֵלֶּה בַּמִּלּוֹת: בְּעוֹד, אָז, לֹא בִלְבַד, אֶלָּא אַף.

דֶּרֶךְ מָשָׁל: לֹא עַל הַמִּדְרָשׁ בִּלְבַד נִצְטַוִּינוּ, אֶלָּא אַף עַל הַמַּעֲשֶׂה.

ה קִיּוּם וּשְׁלִילַת הַמְחֻבָּר תָּלוּי בַּקֶּשֶׁר, אַךְ הָאֱמֶת – בְּכָל חֵלֶק מִן הַחֲלָקִים.

פֵּרוּשׁוֹ: לִהְיוֹת הַמַּאֲמָר שֶׁזָּכַרְנוּ אֲמִתִּי צָרִיךְ שֶׁיִּהְיֶה אֱמֶת, שֶׁנִּצְטַוִּינוּ עַל הַמִּדְרָשׁ וְנִצְטַוִּינוּ גַם עַל הַמַּעֲשֶׂה גַּם כֵּן.

ו הַמְחַלֵּק, קִשְׁרוֹ: מִלַּת אוֹ, וַאֲמִתָּהּ בְּהֶכְרַח הַחִלּוּק.

דֶּרֶךְ מָשָׁל: עַתָּה הוּא יוֹם אוֹ לָיְלָה.

ז הַמַּבְדִּיל הוּא שֶׁמַּבְדִּיל עִנְיָנוֹ, וְקִשְׁרוֹ: לֹא, אֶלָּא, אַף עַל פִּי.

דֶּרֶךְ מָשָׁל: אַף עַל פִּי שֶׁאָנוּ בְגָלוּת אָרֹךְ כָּזֶה, מַחֲזִיקִים בְּתוֹרָתֵנוּ. לֹא עַבְדוּת הִיא הַתּוֹרָה, אֶלָּא חֵרוּת.

ח הַמְיַחֵס, שֶׁיִּתְיַחֲסוּ חֲלָקָיו זֶה לָזֶה, וּקְשָׁרָיו: כְּמו – כֵּן, וְכַיּוֹצֵא.

דֶּרֶךְ מָשָׁל: כְּמוֹ הַמֶּלֶךְ כֵּן הָעָם; מְקוֹם הַזָּהָב שָׁם הַכָּבוֹד.


פֶּרֶק עֶשְׂרִים

הַהֶקֵּשׁ וְכֹחַ הוֹלָדָתוֹ. הַהֶקֵּשׁ הַשָּׁלֵם. הַקְדָּמָה גְדוֹלָה, קְטַנָּה וְתוֹלָדָה. הַבִּלְתִּי שָׁלֵם, פָּשׁוּט וּמֻרְכָּב. הַהֶקֵּשׁ הַתְּנָאִיִּי. הַמְחַלֵּק.


א הַהֶקֵּשׁ הוּא הַרְכָּבַת שְׁתֵּי הַקְדָּמוֹת מְאֻמָּתוֹת וְהוֹלִיד מֵהֶם תּוֹלָדָה אֲמִתִּית.

פֵּרוּשׁוֹ, הַהֶקֵּשׁ הוּא הַפְּעֻלָּה הַשְּׁלִישִׁית שֶׁהַשֵּׂכֶל פּוֹעֵל וְהוּא הִתְבּוֹנֵן דָּבָר מִתּוֹךְ דָּבָר, וְהַיְנוּ שֶׁמִּתּוֹךְ מֻשְׂכָּל שֶׁכְּבָר נִתְאַמֵּת וְנוֹדָע יוֹלִיד וִיחַדֵּשׁ אַחֵר נִמְשָׁךְ בְּהֶכְרַח מִן הָרִאשׁוֹן. וְהִנֵּה שֹׁרֶשׁ הַהֶקֵּשׁ הִיא יְדִיעַת מַה שֶּׁתַּחַת הַנּוֹשֵׂא אֲשֶׁר יָדַעְתָּ נְשׂוּאוֹ, אוֹ שֶׁיִּהְיֶה אֵלָיו בְּעֵרֶךְ הַמִּין עִם סוּגוֹ, אוֹ מַה שֶּׁתַּחַת נָשׂוּא שֶׁל נוֹשַׂאֲךָ, כִּי מַה שֶּׁיֻּכְלַל תַּחַת נוֹשַׂאֲךָ הִנֵּה תָדִין בְּהֶכְרַח שֶׁיִּפֹּל בּוֹ הַנָּשׂוּא שֶׁנָּפַל בְּנוֹשַׂאֲךָ, וְאִם מִינוֹ הוּא – הִנֵּה מַה שֶּׁנוֹפֵל בַּסּוּג יִפֹּל בְּכָל מִין מִמֶּנּוּ. וְכֵן בַּנָּשׂוּא.

פֵּרוּשׁוֹ, כִּי מַה שֶּׁיֻּכְלַל בַּנָּשׂוּא, אֲשֶׁר יָדַעְתָּ הֱיוֹתוֹ בְנוֹשַׂאֲךָ, תִּגְזֹר הֱיוֹתוֹ בוֹ.

דֶּרֶךְ מָשָׁל: יָדַעְתָּ שֶׁכָּל אָדָם הוּא בַעַל חַי. כְּשֶׁתֵּדַע שֶׁרְאוּבֵן נִכְלָל בָּאָדָם כְּמִין תַּחַת סוּגוֹ, תֵּדַע שֶׁבְּהֶכְרַח גַּם רְאוּבֵן יִהְיֶה בַעַל חַי. וְזֶה הַיְסוֹד שֶׁעָלָיו נִבְנִים כָּל הַהֶקֵּשִׁים. אָמְנָם יֵשׁ לַהֶקֵּשִׁים דְּרָכִים רַבִּים וְכֻלָּם שָׁבִים אֶל הַדֶּרֶךְ הַזֶּה, כְּשֶׁתִּתְבּוֹנֵן עֲלֵיהֶם.

דֶּרֶך מָשָׁל, אִם תֹּאמַר: שׁוּם אָדָם אֵינוֹ שׁוֹר,

אֵיזֶה בַעַל חַי הוּא אָדָם,

אִם כֵּן, אֵיזֶה בַעַל חַי אֵינוֹ שׁוֹר.

אוֹ אִם תֹּאמַר:

כָּל מְדַבֵּר אָדָם; שׁוּם אֶבֶן אֵינוֹ אָדָם,

אִם כֵּן, שׁוּם מְדַבֵּר אֵינוֹ אֶבֶן.

אוֹ תֹאמַר:

כָּל מְדַבֵּר גּוּף, שׁוּם מְדַבֵּר אֵינוֹ אֶבֶן,

אִם כֵּן, אֵיזֶה גוּף אֵינוֹ אֶבֶן.

הִנְּךָ רוֹאֶה, שֶׁכֻּלָּם יָשׁוּבוּ לַדֶּרֶךְ שֶׁאָמַרְתִּי.

כִּי כְּשֶׁתֵּדַע שֶׁשּוּם אֶבֶן אֵינוֹ אָדָם, בְּהֶכְרַח שֶׁשּׁוּם אָדָם אֵינוֹ אֶבֶן, וְאָז תֹּאמַר:

כָּל מְדַבֵּר אָדָם, נִמְצָא שֶׁאֵינוֹ אֶבֶן.

וְכֵן כָּל מִינֵי הַהֶקֵּשִׁים יְבֹאֲרוּ בְּכָל חֶלְקֵיהֶם, אוֹ לֹא יְבֹאֲרוּ, אוֹ יְסֻדְּרוּ בְּאֵיזֶה סֵדֶר שֶׁיִּהְיֶה. הִנֵּה הֵם שָׁבִים לַדֶּרֶךְ הַזֶּה, וְזֶה כָּל כֹּחַ הוֹלָדָתָם.

ב מִינֵי הַהֶקֵּשִׁים אַרְבָּעָה: הַהֶקֵּשׁ הַשָּׁלֵם, הַבִּלְתִּי שָׁלֵם, הַתְּנָאִיִּי, וְהַמְחַלֵּק.

ג הַהֶקֵּשׁ הַשָּׁלֵם הוּא אֲשֶׁר יָבֹאוּ בוֹ שְׁתֵּי הַקְדָּמוֹת הַמּוֹלִידוֹת וְהַתּוֹלָדָה שֶׁלָּהֶם.

דֶּרֶךְ מָשָׁל: כָּל אָדָם חַי; רְאוּבֵן אָדָם. אִם כֵּן הוּא בַעַל חַי.

ד הַהַבְחָנוֹת שֶׁיֵּשׁ לְהַבְחִין בַּהֶקֵּשׁ הַשָּׁלֵם שָׁלֹשׁ: הַנּוֹדָע, הַמְבֻקָּשׁ וְהַמַּשְׁוֶה, וּקְרָאוּם: גְּדוֹלָה, קְטַנָּה וְתוֹלָדָה.

ה הַנּוֹדָע הִיא הַהַקְדָּמָה אֲשֶׁר יָדַעְתָּ נוֹשְׂאָהּ וּנְשׂוּאָהּ; הַמְבֻקָּשׁ הוּא הַנּוֹשֵׂא אֲשֶׁר יְדַעְתּוֹ וּתְבַקֵּשׁ לָדַעַת נָשׂוּא לוֹ; וְהַמַּשְׁוֶה הִיא הַהַקְדָּמָה שֶׁבָּה תַּשְׁוֶה הַנּוֹשֵׂא הַמְבֻקָּשׁ לַנּוֹשֵׂא הַנּוֹדָע, לָדוּן בּוֹ עַל יְדֵי זֶה הַנָּשׂוּא הַנּוֹדָע.

דֶּרֶךְ מָשָׁל: כָּל אָדָם בַּעַל חַי, – הוּא הַנּוֹדָע, שֶׁיָּדַעְתָּ נוֹשְׂאוֹ וְהוּא אָדָם, וּנְשׂוּאוֹ שֶׁהוּא בַעַל חַי. רְאוּבֵן הוּא בַעַל חַי, – זֶהוּ הַמְבֻקָּשׁ שֶׁיָּדַעְתָּ נוֹשְׂאוֹ, וְהוּא רְאוּבֵן, וּבִקַּשְׁתָּ לָדַעַת נְשׂוּאוֹ אִם הוּא בַעַל חַי. וּרְאוּבֵן הוּא אָדָם – הוּא הַמַּשְׁוֶה שֶׁמַּשְׁוֶה נוֹשַׂאֲךָ, וְהוּא רְאוּבֵן, אֶל הַנּוֹשֵׂא הַנּוֹדָע, וְהוּא אָדָם.

ו הַבִּלְתִּי שָׁלֵם הוּא שֶׁיִּתְעַלְּמוּ בוֹ אֶחָד מִן הַהַקְדָּמוֹת הַמּוֹלִידוֹת, וְיִבָּנֶה מִשְׁנַיִם לְבָד, וְיִתְחַלֵּק לְפָשׁוּט וּמֻרְכָּב.

ז הַפָּשׁוּט הוּא שֶׁיֵּשׁ לוֹ שְׁנֵי חֲלָקִים לְבָד. הָרִאשׁוֹן נִקְרָא קוֹדֵם, וְהַשֵּׁנִי – נִמְשָׁךְ.

דֶּרֶךְ מָשָׁל: הַחָכְמוֹת מְיַקְּרוֹת אֶת הָאָדָם, אִם כֵּן רָאוּי שֶׁתֵּחָמַדְנָה.

אוֹ: מַה שֶּׁמְּיַקֵּר אֶת הָאָדָם רָאוּי שֶׁיֵּחָמֵד, אִם כֵּן הַחָכְמוֹת רָאוּי שֶׁתֵּחָמַדְנָה. וְתִרְאֶה שֶׁהַמָּשָׁל הָרִאשׁוֹן נֶעֶלְמָה בוֹ הַהַקְדָּמָה הָרִאשׁוֹנָה, וּבַשֵּׁנִי – הַשְּׁנִיָּה.

ח הַמֻּרְכָּב יִתְחַלֵּק לִשְׁנַיִם. הָרִאשׁוֹן הוּא – שֶׁהַקּוֹדֵם שֶׁלּוֹ מֻרְכָּב מִנּוֹשְׂאִים רַבִּים וְנָשׂוּא אֶחָד, וְיֵשׁ מִי שֶׁקְּרָאוּהוּ חִפּוּשׂ.

דֶּרֶךְ מָשָׁל: יֵין אֶרֶץ צָרְפַת, אֶרֶץ אִיטַלִיאָה, אֶרֶץ אַשְׁכְּנַז, אֶרֶץ סְפָרַד, אֶרֶץ תּוֹגַרְמָה – מְחַמֵּם, אִם כֵּן כָּל יַיִן מְחַמֵּם.

ט הַמֻּרְכָּב הַשֵּׁנִי הוּא – שֶׁהַקּוֹדֵם מֻרְכָּב מִנּוֹשְׂאִים וּנְשׂוּאִים רַבִּים מִתְחַלְּפִים זֶה אַחַר זֶה בְּהַדְרָגָה, שֶׁנָּשׂוּא הָאֶחָד נַעֲשֶׂה נוֹשֵׂא בַשֵּׁנִי.

דֶּרֶך מָשָׁל: כָּל אָדָם בַּעַל חַי, כָּל בַּעַל חַי מַרְגִּישׁ, כָּל מַרְגִּישׁ גּוּף חַי, כָּל גּוּף חַי מֻרְכָּב, אִם כֵּן כָּל אָדָם מֻרְכָּב.

י הַהֶקֵּשׁ הַתְּנָאִיִּי הוּא שֶׁתּוֹלַדְתּוֹ מֻנַּחַת בַּתְּנַאי שֶׁתֻּנַּח בָּרִאשׁוֹנָה.

דֶּרֶךְ מָשָׁל: אִם הָאָדָם בַּעַל חַי – הוּא מַרְגִּישׁ; אַךְ הָאָדָם וַדַּאי שֶׁהוּא בַּעַל חַי, אִם כֵּן הוּא מַרְגִּישׁ.

יא הַהֶקֵּשׁ הַמְחַלֵּק – שֶׁהַקְדָּמָתוֹ הָרִאשוֹנָה מְחֻלֶּקֶת בְּמִלַּת “אוֹ”, וְיֻנַּח אֶחָד מִן הַחֲלָקִים, וְיֻכְחַשׁ הַשֵּׁנִי.

דֶּרֶךְ מָשָׁל: עַכְשָׁו הוּא יוֹם אוֹ לַיְלָה,

אָמְנָם יוֹם אֵינוֹ,

אִם כֵּן הוּא לַיְלָה.

אוֹ: עַכְשָׁו הוּא יוֹם אוֹ לַיְלָה,

אָמְנָם יוֹם הוּא,

אִם כֵּן אֵינוֹ לַיְלָה.


פֶּרֶק עֶשְׂרִים וְאֶחָד

שִׁמּוּשֵׁי הַהֶקֵּשׁ. הֶקֵּשׁ יָשָׁר וְהֶקֵּשׁ הַחִלּוּף.


א עַד הֵנָּה בֵּאַרְנוּ הַהֶקֵּשִׁים עַצְמָם, עַתָּה נְבָאֵר הַהִשְׁתַּמֵּשׁ מֵהֶם.

ב הַהֶקֵּשׁ יְשַׁמֵּשׁ לִשְׁנֵי דְבָרִים, הָאֶחָד – לְחַדֵשׁ מֻשְׂכָּל, הַשֵּׁנִית – לְהַעֲמִיד גְּזֵרָה וּלְאַמְּתָהּ.

פֵּרוּשׁוֹ, כִּי הִנֵּה כְּשֶׁיַּגִּיעוּ אֶל שִׂכְלְךָ שְׁתֵּי הַהַקְדָּמוֹת הָרִאשׁוֹנוֹת, מֵאֵלָיו יִתְבּוֹנֵן שִׂכְלְךָ וְיַקִּישׁ וְיוֹלִיד הַתּוֹלָדָה, וְאָז יֵצֵא לְךָ מֻשְׂכָּל חָדָשׁ.

דֶּרֶךְ מָשָׁל: כְּשֶׁרָאִיתָ שֶׁעָבַר לְפָנֶיךָ צֵל אֶחָד, הִנֵּה מִיָּד יִתְבָּרֵר לְךָ שֶׁעָבַר שָׁם גּוּף אֶחָד, כִּי תַקִּישׁ וְתֹאמַר: אֵין צֵל אֶלָּא בְּמָקוֹם שֶׁיֵּשׁ גּוּף, כָּאן עָבַר צֵל, אִם כֵּן כָּאן עָבַר גּוּף, וְכַיּוֹצֵא בָזֶה. זֶהוּ הַשִּׁמּוּשׁ הָרִאשׁוֹן שֶׁל הַהֶקֵּשׁ. אַךְ הַשִּׁמּוּשׁ הַשֵּׁנִי – כְּשֶׁתִּרְצֶה לְאַמֵּת מַאֲמָר שֶׁתֹּאמַר, אָז תִּשְׁתַּמֵּשׁ מִן הַהֶקֶּשׁ.

דֶּרֶךְ מָשָׁל: הָיָה הַמַּאֲמָר שֶׁרָאוּי שֶׁיִּהְיֶה עִקַּר הִשְׁתַּדְּלוּתוֹ שֶׁל הָאָדָם עַל הַתּוֹרָה, כְּשֶׁתִּרְצֶה לְאַמְּתוֹ תִּשְׁתַּמֵּשׁ מֵהֶקֵּשׁ כָּזֶה:

עִקַּר הִשְׁתַּדְלוּתוֹ שֶׁל הָאָדָם רָאוּי שֶׁיִּהְיֶה לְמַה שֶּׁהוּא הַיּוֹתֵר טוֹב לוֹ. הַתּוֹרָה הִיא מַה שֶׁהוּא הַיּוֹתֵר טוֹב לָאָדָם. אִם כֵּן עִקַּר הִשְׁתַּדְּלוּתוֹ שֶׁל הָאָדָם רָאוּי שֶׁיִּהְיֶה עַל הַתּוֹרָה.

ג הַאֲמָתַת הַגְּזֵרָה יִהְיֶה אוֹ עַל יְדֵי הֶקֵּשׁ יָשָׁר, אוֹ עַל יְדֵי הֶקֵּשׁ הַחִלּוּף. הֶקֵּשׁ הַיָּשָׁר הוּא הֶקֵּשׁ שֶׁמֵּהַקְדָּמוֹתָיו תִּוָּלֵד הַתּוֹלָדָה שֶׁנְּבַקֵּשׁ לְאַמְּתָהּ.

הֶקֵּשׁ הַחִלּוּף הוּא הֶקֵּשׁ עַל חִלּוּף הַגְּזֵרָה שֶׁנְּבַקֵּשׁ לְאַמֵּת, שֶׁמֵּהַקְדָּמוֹתָיו תִּוָּלֵד תּוֹלָדָה מְפֻרְסֶמֶת הַכָּזָב, וְאָז נִשְׁפֹּט הֱיוֹת גְּזֵרָתֵנוּ אֲמִתִּית.

דֶּרֶךְ מָשָׁל: הָיָה הַמַּאֲמָר שֶׁהָעוֹלָם מְחֻדָּשׁ.

וּלְהַרְאוֹת אֲמִתָּתוֹ נֹאמַר: אִלּוּ הָיָה קַדְמוֹן יָצָא הַהֶקֵּשׁ בְּדֶרֶךְ זֶה:

כָּל מַה שֶּׁהוּא קַדְמוֹן אִי אֶפְשָׁר לוֹ לְהִשְׁתַּנּוֹת;

הָעוֹלָם הוּא קַדְמוֹן,

אִם כֵּן הָעוֹלָם אֵינוֹ מִשְׁתַּנֶּה.

וְהֲרֵי הַתּוֹלָדָה הַזֹּאת מְפֻרְסֶמֶת הַכָּזָב, כִּי אֲנַחְנוּ רוֹאִים שֶׁהָעוֹלָם מִשְׁתַּנֶּה, אִם כֵּן מֻכְרָח שֶׁמַּאֲמָר שֶׁהָעוֹלָם קַדְמוֹן – יִהְיֶה שֶׁקֶר, וְשֶׁהוּא מְחֻדָּש – אֱמֶת.


פֶּרֶק עֶשְׂרִים וּשְׁנַיִם

סְתִירוֹת הַהֶקֵּשׁ. סְתִירַת הַתּוֹלָדָה מִצַּד סֵדֶר הַהֶקֵּש וּמִצַּד קְצָת הַהֶקֵּשׁ. סְתִירַת הַהַקְדָּמָה. סְתִירָה בִּבְחִינָה אַחַת. הַטְעָאוֹת הַשִּׁתּוּף וְהָעִרְבּוּב.


א עַד הֵנָּה בֵּאַרְנוּ הַרְכָּבַת הַהֶקֵּשׁ, עַתָּה נְבָאֵר סְתִירָתוֹ.

ב יִסָּתֵר הַהֶקֵּשׁ אוֹ לַחֲלוּטִין, אוֹ בִּבְחִינָה מָה.

ג לַחֲלוּטִין תִּהְיֶה הַסְּתִירָה אוֹ בַתּוֹלָדָה, אוֹ בְאֵיזֶה הַקְדָּמָה.

ד הַתּוֹלָדָה תִסָּתֵר מִצַּד כָּל הַהֶקֵּשׁ, אוֹ מִצַּד מִקְצָתוֹ.

ה תִּסָּתֵר הַתּוֹלָדָה מִצַּד כָּל הַהֶקֵּשׁ אִם סֵדֶר הַהֶקֵּשׁ> אֵינוֹ כְהֹגֶן.

פֵּרוּשׁוֹ, שֶׁלֹּא יִהְיֶה יַחַס לַהַקְדָּמוֹת עִם הַתּוֹלָדָה, אוֹ זוֹ עִם זוֹ.

דֶּרֶךְ מָשָׁל: שֶׁיֻּזְכַּר בַּתּוֹלָדָה מַה שֶּׁלֹא בָא בַּנּוֹדָע אוֹ בַמַּשְׁוֶה, שֶׁיִּתֵּן בַּנּוֹשֵׂא הַמְבֻקָּשׁ נָשׂוּא שֶׁאֵינוֹ מִנָּשׂוּא הַנּוֹדָע, אוֹ שֶׁיִּתֵּן נָשׂוּא הַנּוֹדָע לְנוֹשֵׂא שֶׁאֵינוֹ נִכְנָס תַּחַת נוֹשֵׂא הַנּוֹדָע.

דֶּרֶךְ מָשָׁל שֶׁתֹּאמַר: כָּל אָדָם בַּעַל חַי, רְאוּבֵן אָדָם,

אִם כֵּן אֵינוֹ בַעַל חַי, אוֹ: אִם כֵּן הוּא מוֹלִיד.

אוֹ: אִם כֵּן שִׁמְעוֹן הוּא בַעַל חַי.

אוֹ שֶׁתֹּאמַר: כָּל אָדָם מְדַבֵּר, זֶה הָעוֹף מְדַבֵּר, אִם כֵּן הוּא אָדָם. הִנֵּה תִסָּתֵר הַתּוֹלָדָה מִצַּד כָּל הַהֶקֵּשׁ, כִּי אֵין הַנָּשׂוּא שֶׁל הַהַקְדָּמָה אָדָם, אֶלָּא מְדַבֵּר.

ו תִּסָּתֵר הַתּוֹלָדָה מִצַּד מִקְצָת הַהֶקֵּשׁ, אִם נָשׂוּא אֶחָד מִן הַהַקְדָּמוֹת אֵינוֹ מֻכְרָח.

דֶּרֶךְ מָשָׁל שֶׁתֹּאמַר: מַתֶּכֶת הוּא אוֹ כֶסֶף אוֹ זָהָב, הַבַּרְזֶל הוּא מַתֶּכֶת, אִם כֵּן הַבַּרְזֶל הוּא אוֹ כֶסֶף אוֹ זָהָב, – הִנֵּה הַנָּשׂוּא שֶׁל הַהַקְדָּמָה רִאשׁוֹנָה אֵינוֹ מֻכְרָח, כִּי יָכוֹל לִהְיוֹת מַתֶּכֶת וְלֹא יִהְיֶה לֹא כֶסֶף וְלֹא זָהָב.

ז סְתִירַת הַהַקְדָּמָה תִּהְיֶה אִם תִּהְיֶה הַהַקְדָּמָה כּוֹזֶבֶת, דְּהַיְנוּ אִם יְחַיֵּב בְּנוֹשֵׂא אֶחָד נָשׂוּא שֶׁרָאוּי לִשָּׁלֵל מִמֶּנּוּ, אוֹ יְשֻׁלַּל מַה שֶּׁרָאוּי לְחַיֵּב, לְפִי כְלָלֵי הַתְּאָרִים הַהֶגְיוֹנִים שֶׁנִּתְבָּאֲרוּ לְמָעְלָה.

ח סְתִירָה בִּבְחִינָה אַחַת הוּא אִם יֵאָמֵר בַּהַקְדָּמָה דֶּרֶךְ כְּלָל מַה שֶּׁאֵינוֹ אֶלָּא בְּעֵרֶךְ וּבִבְחִינָה אַחַת, שֶׁאָז נֹודֶה אוֹתָה בִּבְחִינָה אַחַת וְנַכְחִישׁ אוֹתָהּ בִּבְחִינָה אַחֶרֶת, וְזֶה יִהְיֶה בַנּוֹשֵׂא אוֹ בַנָּשׂוּא.

דֶּרֶךְ מָשָׁל: כָּל אָדָם מוֹלִיד, זֶה הַיֶּלֶד אָדָם,

אִם כֵּן הוּא מוֹלִיד.

נִסְתֹּר הַהַקְדָּמָה הָאַחַת בְּמִקְצָת וְנֹאמַר: כָּל אָדָם גָּדוֹל וּבִלְתִּי עָקָר מוֹלִיד, אַךְ לֹא כָּל אָדָם בְּהֶחְלֵט, אִם כֵּן אֵינוֹ בְהֶכְרַח שֶׁזֶּה הַיֶּלֶד יִהְיֶה מוֹלִיד. וְזֶה הַחִלּוּק בַּנּוֹשֵׂא.

אוֹ שֶׁתֹּאמַר: הָאָדָם מְדַבֵּר,

הָאִלֵּם הוּא אָדָם,

אִם כֵּן הָאִלֵּם מְדַבֵּר.

תִּסָּתֵר הַהַקְדָּמָה הָאַחַת בְּמִקְצָת וְנֹאמַר: כָּל אָדָם שֶׁאֵין לוֹ מְעַכֵּב לְדִבּוּרוֹ מְדַבֵּר, אַךְ הָאָדָם שֶׁיֵּשׁ עִכּוּב לְדִבּוּרוֹ אֵינוֹ מְדַבֵּר.

ט כָּל הַהֶקֵּשִׁים לִהְיוֹתָם אֲמִתִּים צָרִיךְ שֶׁיִּהְיוּ מְנֻקִּים מִן הַהַטְעָאוֹת, וְהֵם שְׁנֵי מִינִים, הָרִאשׁוֹן – הַשִּׁתּוּף, הַשֵּׁנִי – הָעִרְבּוּב.

י הַשִּׁתּוּף הוּא שִׁתּוּף הַמִּלּוֹת בְּהוֹרָאָתָם, שֶׁלִּפְעָמִים לְפִי הוֹרָאָה אַחַת יָאוּת לָהֶם נָשׂוּא אֶחָד, וּלְפִי הוֹרָאָה אַחֶרֶת – לֹא. וְאִם בְּאַחַת מִן הַהוֹרָאוֹת תִּקְדַּם וְהַתּוֹלָדָה בַּהוֹרָאָה הָאַחֶרֶת, הֲרֵי זֶה הַטְעָאָה.

יא הָעִרְבּוּב הוּא שֶׁלֹּא יִבָּחֵן הַנָּשׂוּא בְּאֵיזֶה יַחַס הוּא עִם נוֹשְׂאוֹ, אִם עַצְמִי וְאִם מִקְרִי, אִם בְּכֹחַ וְאִם בְּפֹעַל, וְכַיּוֹצֵא.

כְּלָלו שֶׁל דָּבָר: הַתּוֹלָדָה צְרִיכָה שֶׁתִּדְמֶה לְהַקְדָּמוֹתֶיהָ דִּמְיוֹן עַצְמִי וְשָׁלֵם. וְאִם אֵינָהּ דּוֹמָה לָהֶם אֶלָּא בְמִקְצָת, וְנִבְדֶּלֶת מֵהֶם בְּעִקַּר אֲמִתָּה – הֲרֵי זֶה הַטְעָאָה.

דֶּרֶךְ מָשָׁל אִם תֹּאמַר: הָאָדָם מְדַבֵּר,

פְּלוֹנִי הַשּׁוֹתֵק הוּא אָדָם,

אִם כֵּן פְּלוֹנִי הַשּׁוֹתֵק הוּא מְדַבֵּר, –

הֲרֵי זֶה הַטְעָאָה, כִּי הָאָדָם מְדַבֵּר בְּכֹח אַךְ לֹא תָמִיד בְּפֹעַל.


פֶּרֶק עֶשְׂרִים וּשְׁלֹשָה

הַסֵּדֶר הַכְּלָלִי. הַשִּׂכְלִיִּי וְהַמַּעֲשִׂיִּי. הַפְּרָטִי. הַהַמְצָאָה. הַמְצָאַת הַגְּזֵרוֹת. הַמְצָאַת הַמּוֹפְתִים. הַבְחָנַת הַגְּזֵרָה. הַוִּכּוּחַ הַמְּדֻמֶּה. הַוִּכּוּחַ הַגּוּפִיִּי. נוֹשֵׂא הַוִּכּוּחַ וְאֹפֶן הַוִּכּוּחַ. הַמִּתְוַכְּחִים וּמַשָּׂאָם. הַדְּחִיָה וְהַסְּתִירָה. הַתַּכְלִית הָעִקָּרִי וְהַטָּפֵל. הַהַבְחָנָה.


א עַד עַתָּה בֵּאַרְנוּ הַהֶקֵּשׁ. עַתָּה נְבָאֵר הַסֵּדֶר, וְהוּא מַה שֶּׁבּוֹ נִסְדָּרִים חֶלְקֵי הַלִּמּוּד כְּהֹגֶן, לְפִי מַדְרֵגָתָם, וְיִתְחַלֵּק לִכְלָלִי וּפְרָטִי.

פֵּרוּשׁוֹ, הַסֵּדֶר הוּא דֶּרֶךְ יְשָׁרָה שֶׁהַהוֹלֵךְ בָּה יַגִּיעַ אֶל תַּכְלִיתוֹ עַל נָכוֹן, בְּלִי נְטוֹת מִמֶּנּוּ.

ב הַסֵּדֶר הַכְּלָלִי הוּא סֵדֶר לִמּוּד מִן הַלִּמּוּדִים, אֲשֶׁר בְּחֻקִּים מְיֻחָדִים יַדְרִיךְ אֶת הָאָדָם לִידִיעַת אַחַת מִן הַחָכְמוֹת, אוֹ מִן הַמְלָאכוֹת, בְּהִתְהַלְכוֹ מִן הָעִנְיָנִים הַקּוֹדְמִים וְהַיּוֹתֵר נוֹדָעִים אֶל הַמְאֻחָרִים וְיוֹתֵר נֶעֱלָמִים.

פֵּרוּשׁוֹ, הַקּוֹדְמִים וְיוֹתֵר נוֹדָעִים הֵם הַכְּלָלִים.

דֶּרֶךְ מָשָׁל: הַסּוּגִים יוֹתֵר נוֹדָעִים מֵהַמִּינִים וְקוֹדְמִים לָהֶם, הַפְּשׁוּטִים – לַמֻּרְכָּבִים. וּלְפִי זֶה בַּלִּמּוּד הַמְסֻדָּר צָרִיךְ שֶׁיִּקְדַּם לִמּוּד הַכְּלָלִים אֶל לִמּוּד הַפְּרָטִים, וְשֶׁל הַפְּשׁוּטִים לְשֶׁל הַמֻּרְכָּבִים.

דֶּרֶךְ מָשָׁל: בַּדִּקְדּוּק הִנֵּה תְחִלָּה צָרִיךְ לְדַבֵּר עַל הָאוֹתִיּוֹת, אַחַר כָּךְ עַל הַתֵּבוֹת וְהַנְקֻדּוֹת, אַחַר כָּךְ עַל חֶלְקֵי הַדִּבּוּר – הַשֵּׁמוֹת, הַפְּעָלִים וְהַמִּלּוֹת, אַחַר כָּךְ עַל חִבּוּרֵי הַמַּאֲמָרִים.

ג עֹוד מִתְחַלֵּק לְשִׂכְלִיִּי וּמַעֲשִׂיִּי. שִׂכְלִיִּי הוּא – כְּשֶׁיְּסֻדַר חֶלְקֵי אֶחָד מִן הַלִּמּוּדִים הַשִּׂכְלִיִּים בְּאֹפֶן שֶׁמִּידִיעַת נוֹשֵׂא הַלִּמּוּד הַהוּא נֵלֵךְ אֶל יְדִיעַת סִבּוֹתָיו, וּמֵהֶם אֶל הַהַבְחָנוֹת הַסּוּגִיּוֹת, וּמֵהֶם אֶל הַמִּינִיּוֹת.

דֶּרֶךְ מָשָׁל: בְּחָכְמַת הַטֶּבַע בַּתְּחִלָּה יִתְבָּאֵר גֶּדֶר הַגּוּף שֶׁהוּא עֵסֶק הַחָכְמָה הַהִיא, אַחַר כָּךְ יִתְבָּאֲרוּ הַסִּבּוֹת הַפְּנִימִיּוֹת וְהַחִצוֹנוֹת, אַחַר כָּךְ יִתְבָּאֲרוּ הַמִּינִים כֻּלָּם, דְּהַיְנוּ הַפְּשׁוּטִים – הַיְסוֹדוֹת – בַּתְּחִלָּה, אַחַר כָּךְ אֶל הַמֻּרְכָּבִים לְפִי מַדְרֵגוֹתֵיהֶם.

ד הַמַּעֲשִׂיִּי הוּא – כְּשֶׁיְּסֻדַּר חֶלְקֵי אֶחָד מִן הַלִּמּוּדִים הַמְלַאכְתִּיִּים בְּאֹפֶן זֶה, שֶׁמִּידִיעַת הַתַּכְלִית נָבוֹא אֶל יְדִיעַת הָאֶמְצָעִיִּים הַמִּצְטָרְכִים לְהַשִּׂיגוֹ.

דֶּרֶךְ מָשָׁל: בַּהִגָּיוֹן יִתְבָּאֵר תְּחִלָּה הַתַּכְלִית, שֶׁהוּא לְהַבְחִין הַדְּבָרִים לַאֲמִתָּם, אַחַר כָּךְ יִתְבָּאֲרוּ הָאֶמְצָעִיִּים הַמַּגִּיעִים לָזֶה, דְּהַיְנוּ הַתְּאָרִים, הַהֶגְיוֹנִים, הַמַּאֲמָרִים, וְהַהֶקֵּשִׁים וְהַסֵּדֶר.

ה הַפְּרָטִי הוּא שֶׁיַּדְרִיךְ אֶת הָאָדָם אֶל יְדִיעַת אֵיזֶה מֻשְׂכָּל מִן הַמֻּשְׂכָּלוֹת בִּפְרָט, וְיִתְחַלֵּק לְהַמְצָאָה וְהַבְחָנָה.

ו הַהַמְצָאָה הוּא – הַמְצָאַת הַגְּזֵרוֹת וְהַהֶקֵּשִׁים שֶׁנִּתְבָּאֲרוּ חֻקּוֹתֵיהֶם לְמָעְלָה.

ז הַמְצָאַת הַגְּזֵרוֹת הוּא – הַבְחֵן בְּנוֹשֵׂא מִן הַנּוֹשְׂאִים אַחַת מִן הַהַבְחָנוֹת הַכּ"א שֶׁנִּתְבָּאֲרוּ לְמָעְלָה.

פֵּרוּשׁוֹ: כְּבָר בֵּאַרְנוּ לְמַעְלָה, אֵיךְ הַכּ"א הַבְחָנוֹת אֵלֶּה הֵם כְּלָל כָּל מַה שֶּׁיֵּשׁ לְקַיֵּם אוֹ לִשְׁלֹל בְּאֵיזֶה נוֹשֵׂא שֶׁיִּהְיֶה, וּמִזֶּה נִבְנוֹת הַגְּזֵרוֹת שֶׁאֵינָם אֶלָּא קִיּוּם אוֹ שְׁלִילַת מַה בַּנּוֹשֵׂא. וְאָמְנָם הַסֵּדֶר שֶׁרְגִילִים לִשְׁמֹר בַּחֲקִירַת הַבְחָנוֹת אֵלֶּה בְּאֵיזֶה נוֹשֵׂא שֶׁיִּהְיֶה, כְּפִי מַה שֶּׁיִּפֹּל בּוֹ מֵהֶם, הוּא זֶה:

1 הַשֵּׁם אִם הוּא מְיֻחָד, אוֹ מְשֻׁתָּף, אוֹ מֻשְׁאָל, וְכַיּוֹצֵא מִשְּׁאָר מִינֵי הַשֵּׁם. וְאוּלָם זֶה רָאוּי שֶׁיוּבַן רִאשוֹנָה, כְּדֵי שֶׁלֹּא נִסְתַּפֵּק בַּנּוֹשֵׂא שֶׁבּוֹ נַעֲסֹק.

2 הַסּוּג. וְיִבָּחֵן בָּזֶה תַּחַת אֵיזֶה סוּג מֵהָעֶשֶׂר סוּגִים הָרִאשׁוֹנִים יִכָּנֵס, כִּי לְפִי הִשְׁתַּנּוּת הַסּוּג יִשְׁתַּנּוּ חֻקָּיו, כִּמְבֹאָר בִּכְלָלֵי הָעֶשֶׂר סוּגִים שֶׁזָּכָרְנוּ.

3 הַסִּבּוֹת כֻּלָּם.

4 הַמְסֻבָּבִים הַיּוֹתֵר רְשׁוּמִים.

5 הַמִּתְחַבְּרִים, הַמְּיֻחָדִים, הַמְּשֻׁתָּפִים, הַקּוֹדְמִים וְהַנִּמְשָׁכִים וְכָל הַשְּׁאָר.

6 הַגָּדֵר.

7 הַחִלּוּק בַּחֲלָקִים בִּנְיָנִיִּם, אִם הַנּוֹשֵׂא כָּל בִּנְיָנִיִּי.

8 הַחִלּוּק בַּמִּינִים אִם הַנּוֹשֵׂא סוּג.

9 הַנֶּעֱרָכִים בְּכָל חֶלְקֵיהֶם.

10 הַמִּשְׁתַּנִּים וְהַהַפְכִּיִּים בְּכָל חֶלְקֵיהֶם, שֶׁגַּם יְדִיעָתָם מוֹעֶלֶת הַרְבֵּה לְבֵרוּר יְדִיעַת הַנּוֹשֵׂא.

ח הַמְצָאַת הַהֶקֵּשִׁים הוּא הַמְצָאַת הַמּוֹפְתִים לְהַאֲמָתַת הַגְּזֵרוֹת, אוֹ לְהַכְחָשָׁתָן, וּמְקוֹרָה הַכּ"א הַבְחָנוֹת שֶׁבֵּאַרְנוּ.

פֵּרוּשׁוֹ, כִּי אֵין שׁוּם הֶקֵּשׁ בָּעוֹלָם שֶׁיַּרְאֶה אֲמִתַּת גְזֵרָה אַחַת, אוֹ שֶׁקָּרָה שֶׁלֹּא יִוָּלֵד מֵאַחַת מִן הַהַבְחָנוֹת הָאֵלֶּה.

דֶּרֶךְ מָשָׁל: אִם תִּהְיֶה הַגְּזֵרָה שֶׁרָאוּי לְכַבֵּד אֶת הַתּוֹרָה כָּבוֹד גָּדוֹל, הִנֵּה תוֹכִיחַ אֲמִתַּת הַגְּזֵרָה בְּהֶקֵּשׁ כָּזֶה: כָּל מַה שֶּׁיִתְיַחֵס אֶל הַבּוֹרֵא יִתְבָּרַךְ רָאוּי לְכַבְּדוֹ כָּבוֹד גָּדוֹל, הַתּוֹרָה מִתְיַחֶסֶת לַבּוֹרֵא יִתְבָּרַךְ, אִם כֵּן רָאוּי לְכַבְּדָהּ כָּבוֹד גָּדוֹל. הִנֵּה הֶקֵּשׁ זֶה נוֹלַד מֵהַמִּתְחַבְּרִים שֶׁפֵּרַשְׁתִּי בִמְקוֹמוֹ. כִּי אֶחָד מִמִּתְחַבְּרֵי הַתּוֹרָה הוּא הֱיוֹתָהּ אֱלֹהִית, פֵּרוּשׁוֹ, מִתְיַחֶסֶת לַבּוֹרֵא יִתְבָּרַךְ. וְכָזֶה תָּדִין בְּכָל הַהֶקֵּשִׁים.

ט הַבְחָנַת הַגְּזֵרָה, לְאַמְּתָה אוֹ לְהַכְחִישָׁהּ, תִּהְיֶה אוֹ בִּפְנֵי עַצְמָהּ, אוֹ בְּוִכּוּחַ. בִּפְנֵי עַצְמָהּ הוּא – שֶׁנִּבְחַן גְּזֵרָה אַחַת לְבַדֵּנוּ, בְּלִי מִתְנַגֵּד, עַל יְדֵי הַבְחָנוֹת הַנְּאוּתוֹת.

פֵּרוּשׁוֹ: עַל יְדֵי הַהַבְחָנוֹת הַכּ“א שֶׁזָּכַרְנוּ לְמַעְלָה, שֶׁמִּשָּׁם נוֹצִיא כָּל הַהֶקֵּשִׁים. וְהִנֵּה כְּשֶׁתִּרְצֶה לִבְחֹן גְּזֵרָה אַחַת, תִּצְטָרֵךְ לְהַבִּיט תְּחִלָּה לִשְׁנֵי חֲלָקִים, דְּהַיְנוּ הַנּוֹשֵׂא וְהַנָּשׂוּא, וּלְהָסִיר מֵהֶם כָּל סְפֵקוֹת הַשִּׁתּוּף, עַד שֶׁיִּוָּדְעוּ לְךָ לַאֲמִתָּם; אַחַר כָּךְ – כַּמּוּת הַגְּזֵרָה: אִם הִיא כְּלָלִית, חֶלְקִית אוֹ אִישׁיִּית; אַחַר כָּךְ תַּבִּיט אֶל מַהוּתָהּ שֶׁל הַגְּזֵרָה, דְּהַיְנוּ אִם הִיא מְחַיֶּבֶת אוֹ שׁוֹלֶלֶת, אִם הִיא מֻכְרַחַת אוֹ אֶפְשָׁרִית; אַחַר כָּךְ תִּתְבּוֹנֵן אִם הִיא אֲמִתִּית אוֹ כוֹזֶבֶת, וְהַיְנוּ אִם תֵּאוֹת עִם הַכְּלָלִים אֲשֶׁר הִקְדַּמְנוּ בְּכ”א הַהַבְחָנוֹת, אוֹ אִם תִּתְנַגֵּד לָהֶם, אוֹ אִם תִּוָּלֵד מֵהֶם בְּהֶכְרַח: אִם תִּוָּלֵד מֵהֶם בְּהֶכְרַח תִּהְיֶה אֲמִתִּית מֻכְרַחַת, אִם תִּתְנַגֵּד תִּהְיֶה כוֹזֶבֶת, אִם תֵּאוֹת וְלֹא תֻכְרַח תִּהְיֶה אֶפְשָׁרִית.

י בְּוִכּוּח הוּא – כְּשֶׁתֵּאָמַרְנָה שְׁתֵּי גְזֵרוֹת שׁוֹנוֹת אוֹ הַפְכִּיּוֹת וְנַעֲרֹךְ בֵּינֵיהֶם מִלְחָמָה עַל יְדֵי הֶקֵּשִׁים שֶׁיַּעַזְרוּ לְאַחַת מֵהֶם, אוֹ לַחֲבֶרְתָּהּ, וְיִהְיֶה הַהִשְׁתַּדְּלוּת לִסְתֹּר הַהֶקֵּשִׁים הַמֻּטְעִים וּלְהַעֲמִיד הָאֲמִתִּיִּים, עַד שֶׁתְּקֻיַּם הַגְּזֵרָה הָאֲמִתִּית, וְזֶה יֵעָשֶׂה אוֹ בְּוִכּוּחַ מְדֻמֶּה אוֹ בְּוִכּוּחַ גּוּפִיִּי.

יא הַוִּכּוּחַ הַמְדֻמֶּה הוּא – שֶׁאִישׁ אֶחָד יַעֲשֶׂה פְּעֻלַּת הַשֵּׁנִי, דְּהַיְנוּ שֶׁיִּגְזֹר גְּזֵרָתוֹ בֶאֱמֶת, וְאַחַר כָּךְ יַקְשֶׁה עָלֶיהָ בְּהֶקֵּשִׁים יַכְזִיבוּהָ, אַחַר כָּךְ יִסְתֹּר הַהֶקֵּשִׁים הָהֵם, וִיקַיֵּם גְּזֵרָתוֹ וְיַרְאֶה אֲמִתָּהּ.

יב הַוִּכּוּחַ הַגּוּפִיִּי הוּא – שֶׁיִּהְיֶה בֵּין שְׁנֵי גוּפִים מַמָּשׁ, אֶחָד מַנִּיחַ וְאֶחָד מַקְשֶׁה. וְיִבָּחֲנוּ בָזֶה אַרְבַּע הַבְחָנוֹת: רִאשׁוֹנָה – הַנּוֹשֵׂא, שְׁנִיָּה – הָאֹפֶן, שְׁלִישִׁית – הַמִּתְוַכְּחִים, רְבִיעִית – הַתַּכְלִית.

יג הַנּוֹשֵׂא הוּא הָעִנְיָן אוֹ גְזֵרָה, שֶׁעָלָיו יִשְׂאוּ וְיִתְּנוּ הַמַּקְשֶׁה וְהַמֵּשִׁיב. הַמַּשָּׂא וּמַּתָּן הַוִּכּוּחִי אֵין רָאוּי שֶׁיִּהְיֶה אֶלָּא בְעִנְיָן שֶׁאֲמִתּוֹ בִּלְתִּי מְפֻרְסֶמֶת, כִּי בַמְפֻרְסָמוֹת אֵינוֹ אֶלָּא מְיֻתָּר וּבְעִנְיָן שֶׁיְּדִיעָתוֹ אֶפְשָׁרִית, כִּי בְנִמְנְעֵי הַהַשָּׂגָה הַמַּשָּׂא וּמַתָּן בָּטֵל וּמְיֻתָּר וּבִלְתִּי רָאוּי; וְשֶׁיִּהְיֶה עִנְיָן שֶׁיְּדִיעָתוֹ מוֹעֶלֶת לְאֵיזֶה חָכְמָה אוֹ מְלָאכָה, שֶׁאִם לֹא כֵן אֵינוֹ אֶלָּא שִׂיחָה בְטֵלָה.

יד הָאֹפֶן הוּא – שֶׁיִּהְיֶה הַמַּשָּׂא וּמַתָּן בָּרוּר, בְּלִי הַטְעָאוֹת; שֶׁיִּהְיֶה בְיִשּׁוּב וּבְנַחַת, בְּלִי בֶהָלָה וָכַעַס; שֶׁיִּהְיֶה בְדֶרֶךְ קְצָרָה וְהֶגְיוֹנִית.

פֵּרוּשׁוֹ, כִּי אִם לֹא יְסֻדְּרוּ הַקֻּשְׁיוֹת וְהַתֵּרוּצִים בְּדֶרֶךְ הֶגְיוֹנִי, שֶׁיִּבָּחֲנוּ הַנּוֹשְׂאִים וְהַנְּשׂוּאִים, הַחִיּוּבִים וְהַשְּׁלִילוֹת וּשְׁאָר כָּל תְּנָאֵי הַמַּאֲמָרִים, הִנֵּה לֹא יִתְפַּרְסֵם הַכָּזָּב וְהַטָּעוּת, וְלֹא יִוָּדַע מְקוֹמוֹ, וְיִמָּלֵט מִן הַשֵּׂכֶל בְּלִי שֶׁיַּרְגִּישׁ בּוֹ; מַה שֶּׁאֵין כֵּן בִּהְיוֹת כָּל הַגְּדֵרִים נִבְחָנִים כָּרָאוּי, שֶׁאָז יִגָּלֶה מִיָּד הַבִּלְתִּי נָאוּת, אֲשֶׁר לֹא יִשְׁמֹר חֹק הָעֵרֶךְ הָרָאוּי בַּגְזֵרוֹת. וְאוּלָם גַּם רִבּוּי הַדְּבָרִים, אֲפִילוּ שֶׁיִּהְיוּ הֲגוּנִים, מְיַגֵּעַ הַשֵּׂכֶל וְיִלְאֶה בָּם וְלֹא יוּכַל לְהַבְחִין אוֹתָם בַּדִּקְדּוּק הַמִּצְטָרֵךְ, כִּי יִרְבֶּה מַשָּׂאָם עָלָיו וְלֹא יַעֲצֹר כֹּחַ לְהַבְדִּיל בֵּינֵיהֶם וְלַעֲמֹד עַל פְּרָטֵיהֶם הַדַּקִּים, וְנִמְצָא מְחַיֵּב מַה שֶּׁרָאוּי לִשְׁלֹל, אוֹ לְהֵפֶךְ, וְהוּא לֹא יֵדַע.

טו הַמִּתְוַכְּחִים הֵם – הָאֶחָד הַמַּנִּיחַ וְהוּא עַצְמוֹ הַמֵּשִׁיב, וְהַשֵּׁנִי – הַמַּקְשֶׁה לוֹ.

טז הַמַּנִּיחַ צָרִיךְ שֶׁיְּבָאֵר גְּזֵרָתוֹ הֵיטֵב בְּדֶרֶךְ קְצָרָה, וְלֹא יוֹתֵר מִדַּאי, וְלֹא יִגְזֹר לְחִיּוּב אֶלָּא מַה שֶּׁיֻּכְרַח לוֹ עַל פִּי הֶקֵּשׁ אֲמִתִּי.

יז מַשָּׂא הַמַּקְשֶׁה הוּא בְּמַה שֶּׁקֹּדֶם הַקְשׁוֹתוֹ, בְּעֵת הַקְשׁוֹתוֹ, וְאַחֲרֵי הַקְשׁוֹתוֹ.

יח קֹדֶם הַקְשׁוֹתוֹ הוּא – הַאֲזֵן הֵיטֵב לְהַנָּחַת הַמַּנִּיחַ בְּכָל חֲלָקֶיהָ, לְהָבִין אוֹתָה כְּהֹגֶן, אַחַר כָּךְ דַּקְדֵּק בָּהֶם אִם יֵשׁ לְהִתְנַגֵּד לְאֶחָד מֵהֶם, אִם בִּדְחִיָּה אִם בִּסְתִירָה.

יט בִּדְחִיָּה הוּא – כְּשֶׁיֵּעָדֵר הַהֶכְרַח בִּגְזֵרָתוֹ שֶׁל הַמַּנִּיחַ, אֲפִילוּ אִם לֹא יַכְחִישָׁהּ.

כ בִּסְתִירָה הוּא – בִּהְיוֹת גְּזֵרָתוֹ כוֹזֶבֶת, מִכֹּח הֶקֵּשׁ חָזָק שֶׁיִּסְתֹּר אוֹתָה.

כא בְּעֵת הַקְשׁוֹתוֹ – שֶׁיְּסַדֵּר קֻשְׁיָתוֹ בְּדֶרֶך קְצָרָה וְהֶגְיוֹנִית, לְפִי מַה שֶּׁהִיא, אִם דְּחִיָּה וְאִם סְתִירָה.

פֵּרוּשׁוֹ, כִּי לֹא כַדְּחִיָּה הַסְּתִירָה. כִּי הַסְּתִירָה צָרִיךְ שֶׁתִּהְיֶה בְּכֹחַ הֶקֵּשׁ חוֹתֵךְ, חָזָק, כְּמוֹ הַהַנָּחָה; אַךְ הַדְּחִיָּה – כָּל זְמַן שֶׁלֹּא יִהְיֶה מוֹפֵת מַכְרִיחַ אֲמִתַּת הַגְּזֵרָה, נוּכַל שֶׁלֹּא לְקַבְּלָהּ, בִּדְחוֹת אוֹתָה בִּדְחִיָּה לְבָד.

כב אַחֲרֵי הַקְשׁוֹתוֹ – הַאֲזֵן הֵיטֵב אֶל הַמַּנִיחַ שֶׁיָּשִׁיב אֶל קֻשְׁיָתוֹ, וְהוֹדוֹת עַל הָאֱמֶת.

כג מַשָּׂא הַמֵּשִׁיב הוּא קֹדֶם הֲשִׁיבוֹ וּבְעֵת הֲשִׁיבוֹ. קֹדֶם הֲשִׁיבוֹ – לְהַאֲזִין הֵיטֵב אֶל הַקֻּשְׁיָא וּלְהַבְחִין עִנְיָנָהּ: אִם בֶּאֱמֶת מִתְנַגֵּד הוּא לְהַנָּחָתוֹ אִם לֹא, אִם בִּנְיָנָהּ הֶגְיוֹנִי אִם לֹא, אִם הַקְדָּמוֹתֶיהָ וְתוֹלַדְתָּהּ מֻכְרָחוֹת אִם לֹא.

פֵּרוּשׁוֹ, כִּי אִם לֹא עִנְיָנָהּ מִתְנַגֵּד מַמָּשׁ לְהַנָּחָתוֹ, אֵין סְתִירָתָהּ נוֹגַעַת לוֹ, וְלֹא יִצְטָרֵךְ לִטְרֹחַ בָּה שֶׁלֹּא לָצֵאת מֵעִנְיָנוֹ; אִם בִנְיָנָהּ אֵינוֹ הֶגְיוֹנִי, יַרְאֶה לַמַּקְשֶׁה מְקוֹם הַטָּעוּת וְיֹאמַר לוֹ שֶׁיְּסַדֵּר קֻשְׁיָתוֹ; אִם הַקְדָּמוֹתֶיהָ אוֹ תוֹלַדְתָּהּ בִּלְתִּי מֻכְרָחוֹת – יִסְתְּרֵם, וְאִם הַקֻּשְׁיָא אֲמִתִּית – יוֹדֶה עַל הָאֱמֶת וְיָשׁוּב מֵהַנָּחָתוֹ.

כד הַתַּכְלִית יִתְחַלֵּק לְעִקָּרִי וְטָפֵל.

כה הָעִקָּרִי – הַשָּׂגַת הָאֱמֶת, הַטָּפֵל – חִדּוּד הַשֵּׂכֶל.

כו עַד עַתָּה בֵּאַרְנוּ הַהַמְצָאָה. עַתָּה נְבָאֵר הַהַבְחָנָה, וְהִיא – הַבְחֵן חֶלְקֵי הַמַּאֲמָרִים וְהַעֲמִידָם עַל מַדְרֵגוֹתֵיהֶם הַהֶגְיוֹנִיּוֹת.

פֵּרוּשׁוֹ, כִּי הִנֵּה אַחַר שֶׁיָּדַעְנוּ מַה נִּשְׁלֹל אוֹ נְחַיֵּב בְּאֶחָד מִן הַנּוֹשְׂאִים וְהַהֶקֵּשִׁים הַמּוֹכִיחִים אֲמִתַּת גְּזֵרָה מִן הַגְּזֵרוֹת, מִזֶּה נִבְנֶה הַדְּרוּשִׁים וְהַהַגָּדוֹת בַּמְּלִיצָה כָּרָאוּי, וְּנַסֵּדר הַדְּבָרִים בְּהַרְחָבַת הַלָּשׁוֹן וּשְׁאָר הַדְּבָרִים הַהֲלָצִיִּים הָעֲשׂוּיִים לְהַמְתִּיק. אַךְ לִבְחִירַת הַדְּרוּשִׁים – נִצְטָרֵךְ לְנַתְּחָם, וּלְּהַבְדִּיל עִנְיְנֵי הַמְלִיצָה וְדַרְכֵי הַסִּפּוּר לְבַד וְהַחֲלָקִים הַהֶגְיוֹנִים לְבָד, מֻפְשָׁטִים וַעֲרֻמִּים מִכָּל שְׁאָר עִנְיְנֵי הַסִּפּוּר וְהַהֶקֵּשִׁים עַל צוּרָתָם וְסִדְרֵיהֶם, וְאָז נוּכַל לִשְׁפֹּט אוֹתָם וְלִבְחֹן אוֹתָם. וְהִנֵּה עֲשִׂיַּת הַנִּתּוּחַ הַזֶּה הִיא הַהַבְחָנָה הַזֹּאת שֶׁזָּכַרְנוּ. אָמְנָם סֵדֶר מַעֲשֵׂה הַהַבְחָנָה הַזֹּאת הוּא זֶה: תְּחִלַּת הַכֹּל צָרִיךְ שֶׁתִּקְרָא וְתִשְׁנֶה וּתְשַׁלֵּשׁ הַדְּרוּשׁ אוֹ הַהַגָּדָה אֲשֶׁר תִּרְצֶה לְנַתֵּחַ, וְתִתְבּוֹנֵן בָּה הֵיטֵב עַד שֶׁתַּכִּיר עִנְיָנָהּ, אִם הוּא אֶחָד אוֹ יוֹתֵר, אִם הוּא פָשׁוּט אוֹ מְחֻבָּר. כִּי אִם לֹא תַכִּיר כָּל זֶה עַל בֻּרְיוֹ, הִנֵּה תִטֶּה מִדֶּרֶךְ הַהַגָּדָה הַהִיא וְהַדְּרוּשׁ הַהוּא נְטִיָּה רַבָּה. וְאַחַר שֶׁהִכַּרְתָּ הָעִנְיָן אוֹ הָעִנְיָנִים אֲשֶׁר בַּהַגָּדָה, אָז תַּבְדִּיל הַבְּחִינוֹת, אוֹ הַמּוֹפְתִים הַנִּכְלָלִים בָּהּ, וְתַבְדִּיל בֵּין הַיּוֹתֵר עִקָּרִיִּים לָעִנְיָן וֵּבין הַבִּלְתִּי עִקָּרִיִּים כָּל כָּךְ. וְאִם הָעִנְיָנִים רַבִּים, תַּפְרִיד כָּל עִנְיָן לְעַצְמוֹ עִם בְּחִינוֹתָיו וּמוֹפְתָיו, שֶׁלֹּא לְעָרֵב עִנְיָן בְּעִנְיָן וְלֹא תְנָאֵי עִנְיָן בִּתְנָאֵי זוּלָתוֹ. אַחַר כָּךְ תַּעֲמִיד הַמּוֹפְתִים לְמַחְלְקוֹתֵיהֶם עַל סִדְרָם וְצוּרָתָם, וְתַבְחִין הַמּוֹפְתִים הַמְאַמְּתִים הָעִנְיָנִים וְהַמּוֹסִיפִים בֵּאוּר וּמַרְחִיבִים הָעִנְיָן הָעִקָּרִי, וְנִמְצְאוּ כָּל הַחֲלָקִים מֻצָּגִים לְפָנֶיךָ, שֶׁתּוּכַל לְהִתְבּוֹנֵן בָּהֶם כָּרָאוּי וּלְהַכִּיר הָאֱמֶת לַאֲמִתּוֹ, בְּסִיַּעְתָּא דִשְׁמַיָּא.
