עִיר קֹדֶשׁ

שיר לאהבת עיה״ק חברון תו״ב עשיתיו ביום כ״ה סיון


עִיר קֹדֶשׁ חֶבְרוֹן קִרְיַת אַרְבַּע אֶרֶץ מוֹלַדְתֵּנוּ

יְפֵה נוֹף מְשׂוֹשׂ כָּל תֵּבֵל חֶבֶל נַחֲלָתֵנוּ.


רוּחֵךְ עֵדֶן וּבֹשֶׂם רֵיחֵךְ רֵיחַ נִיחוֹחִים

מַה יָּפִית וּמַה נָּעַמְתְּ שׁוֹשַׁנָּה בֵינוֹת שִׂיחִים

חוֹנָה בֵין הֲרָרַיִךְ כַּעֲרוּגַת מֶרְקָחִים

בָּךְ הַדְרַת קֹדֶשׁ זִיו טָהֳרָה נֹגַהּ חֶמְדָּתֵנוּ.


בָּךְ מִבְצַר עֹז מִגְדָּל הוֹד הֵיכַל דָּוִד מַלְכֵּנוּ

נִצָּב כְּחוֹמָה נִשְׂגָּבָה בוֹ גִיל חַיָּתֵנוּ

שָׁם שָׁכְבוּ קְדוֹשֵׁי עֶלְיוֹן הֵם אֲבוֹתֵינוּ

בּוֹ נִקְשְׁרָה שִׂמְחַת שַׁעֲשׁוּעֵי תַאֲוָתֵנוּ.


אֶזְכֹּר אָשְׁרֵךְ עֵת שֶׁמֶשׁ וְסַהַר בָּךְ נָהֲרוּ

עֵת אַלְפֵי רִבְבוֹת יִשְׂרָאֵל עָדַיִךְ נָהֲרוּ

לְהַמְשִׁיךְ קְדוֹשׁ מְשִׁיחוֹ בְּאֶחָד נִקְשָׁרוּ

לַהֲבוֹת אֵשׁ שַׁלְהֶבֶת יָהּ תּוּקַד בְּלִבּוֹתֵינוּ.


לָךְ מִמֶּרְחַקִּים כָּלְתָה נַפְשָׁם אֵלַיִךְ

רוּחִי וְרַעְיוֹנַי נִקְשְׁרוּ בְצוּף הוֹדַיִךְ

מִי יִתֵּן וּכְמוֹ רֶגַע אֲחוֹנֵן רְגָבַיִךְ

אֲחַבְּקֵם וַאֲרַוֵּם מִנַּחַל דִּמְעוֹתֵנוּ.


אֵיךְ אֶרְאֶה בְאוֹר חַיִּים וַאֲנִי נָד מֵחֵיקֵךְ

יֵקַל בְּעֵינַי הוֹן כָּל טוּב תֵּבֵל בִרְחוֹקֵךְ

לִקְרָאתֵךְ אָרוּץ קֵץ אִיִּים לָשֶׁבֶת בְּחֵיקֵךְ

עַבְדֵּךְ אָסוּר בְּכַבְלֵי חִנֵּךְ צִבְיַת תִּפְאַרְתֵּנוּ.


מְאֹד נִכְסְפָה נַפְשִׁי לַהֲלֹךְ בָּךְ אַף צַעַד

עֲלֵי אַדְמָתֵךְ וְלוּ חַתְחַתִּים רַגְלַי תִּמְעַד

וְעֵת אֶזְכְּרָה נְדוּדַיִךְ יֹאחֲזֵנִי רַעַד

כִּי בָךְ חַיַּי תְּלוּאִים וְאַתְּ הִיא עֵץ חַיָּתֵנוּ.


מַה נָּעֲמוּ לִי נַחֲלַת קוֹצֵךְ וְחֹחַיִךְ

מִכָּל גַּנִּי חֶמֶד אֲשֶׁר אֶמְצָא בִנְדוּדַיִךְ

בָּךְ אֶרְאֶה תְמוּנַת צִיץ שַׁדַּי כֹּכְבֵי שָׁמַיִךְ

אָבוֹת וּנְבִיאִים קְדוֹשִׁים זִקְנֵי עֲדָתֵנוּ.


יְדִיד עֶלְיוֹן כַּמָּה בָךְ אַהֲבָתוֹ בָעֲרָה

הוּא הִרְבָּה כֶסֶף לִבְלִי חֹק אַךְ בְּעַד מְעָרָה

גַּם יִצְחָק וְיַעֲקֹב בָּךְ בָּחֲרוּ לִקְבוּרָה

וְאֵיךְ לֹא אֶזְכְּרֵךְ תָּמִיד עַל רֹאשׁ שִׂמְחָתֵנוּ.


גַּם נְבִיאִים בָּךְ עָשׂוּ לְמֵתֵיהֶם צְרִיחִים

וַיִּטְּשׁוּ אֲרָצוֹת עֲדִינוֹת וְשֶׁבֶת אַחִים

יַעַן אַתְּ לְשׁוֹכְבֵי חֵיקֵךְ חַיֵּי נְצָחִים

בָּךְ צַעֲרֵי גַן עֵדֶן נִפְתְּחוּ לְנִשְׁמוֹתֵנוּ.


חִזְקִי לִבְשִׁי עֹז עִיר קֹדֶשׁ אֲהוּבַת עוֹלָמִים

סוֹדוֹ לִגְאֻלָּתֵךְ יַגִּיהַּ מַהֵר אֵל נִיחוּמִים

עוֹד תַּעְדִּי כְלֵי תִפְאָרָה כִימֵי עֲלוּמִים

וְאָז לְאֵל חַי גּוֹאֲלֵנוּ נַקְרִיב שַׁי שִׁירוֹתֵנוּ.
