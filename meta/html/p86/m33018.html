ספר תחכמוני - שַׁעַר עֶשְׂרִים וּשְׁלשָׁה

בְּצָרַת חֶבֶר הַקֵּינִי וְדַאֲגָתוֹ. וּמֶמְשַׁלְתּוֹ וּמַעֲלָתוֹ.


נְאֻם הֵימָן הָאֶזְרָחִי: לְקָחַנִי הַנְּדוֹד עַל אֶבְרָתוֹ. וּקְשֶׁרַנִי בַעֲבוֹת חֶבְרָתוֹ. וְעַל כַּנְפֵי פֵרוּד נְשָׂאַנִי. וְהִשִּׁיאַנִי. לְשׁוֹטֵט מְדִינוֹת. וּלְסוֹבֵב מְעוֹנוֹת. וְלְמַדֵּד אֲרָקִים. וּלְשַׂדֵּד עֲמָקִים. תֶּהְדְּפֵנִי אֶרֶץ אֶל אֶרֶץ. וַיִּפְרְצֵנִי הַנְּדוֹד פֶּרֶץ עַל פְּנֵי פֶּרֶץ. וַיְהִי בִקְצָת הַיָּמִים וַאֲנִי בָאתִי מִסּוּרְיָה. לִטְבֶרְיָה. פָּגַעְתִּי בְשַׁעַר הָעִיר חֶבֶר הַקֵּינִי וּמַדָּיו קְרוּעִים. וְגוּפוֹ מָלֶא פְּצָעִים וּנְגָעִים. וַיָּרֻם תּוֹלָעִים. וָאֲמַהֵר לְחָקְרוֹ וּלְשָׁאֳלוֹ. עַל הַמִּקְרֶה אֲשֶׁר קָרָה לוֹ. וָיֹּאמֶר לִי כִּי הָיָה נוֹסֵעַ מִן הַשִּׁטִּים. וְעַד תֵּל חִטִּים. וַיָּבוֹאוּ עָלָיו לִסְטִים. וְעָשׂוּ בוֹ שְׁפָטִים. וַחֲבָטוּהוּ בְשׁוֹטִים. כְּחוֹבֵט חִטִּים. וְהִכּוּהוּ פְצָעוּהוּ. וְלָקחוּ כֹל אֲשֶׁר לוֹ וְלַמָּוֶת הִגִּיעוּהוּ. וְשׁוֹלָל וְעָרֹם הִנִּיחוּהוּ. וְנִמְלַט מֵהֶם וְרוּחוֹ בֵין שִׁנָּיו. וְחֶרֶב הַמָּוֶת מִתְהַפֶּכֶת בֵּין עֵינָיו. וּכְשָׁמְעִי מִלָּיו. בָּכִיתִי לְצִירָיו וְחִילָיו. וַיֶּחֱרַד לִבִּי עָלָיו. וּבְיָדוֹ הֶחֱזַקְתִּיהוּ. וּלְבֵיתִי הֳבֵאתִיהוּ. וְדִבַּרְתִּי עַל לִבּוֹ וְנִחַמְתִּיהוּ. וְלָקַחְתִּי בֶגֶד מִבְּגָדַי וְהִלְבַּשְׁתִּיהוּ. וּבְלֶחֶם וּבָשָׂר קִדַּמְתִּיהוּ. וְיֵין הָרֶקַח הִשְׁקִיתִיהוּ. אוּלַי תִּשְׂמַח נַפְשׁוֹ. וְיִשְׁתֶּה וְיִשְׁכַּח רִישׁוֹ. וְכַאֲשֶׁר שָׁתִיתִי עִמּוֹ שָׁכַרְתִּי. וְגָבַר עָלַי הַיַּיִן וְנִרְדַּמְתִּי. וּבְעוֹדִי יָשֵׁן בָּרַח מִמֶּנִּי וַאֲנִי לֹא יָדַעְתִּי. וְכַאֲשֶׁר הֲקִיצוֹתִי בִקַּשְׁתִּיהוּ. וְלֹא מְצָאתִיהוּ. וְלֹא עָנַנִי בְּעֵת קְרָאתִיהוּ. וַיִּחַר עָלָיו אַפִּי. וְהִכֵּיתִי כַפִּי אֶל כַּפִּי. עַל אֲשֶׁר עִמּוֹ הֵיטַבְתִּי. וְיָדִי לְפִי שַׂמְתִּי. וְעַל חֶבְרָתִי עִמּוֹ נִחַמְתִּי. וְשַׂמְתִּי עַל לִבִּי. לִנְטֹר אֵיבָתוֹ בְקִרְבִּי. וְלִטְמֹן עֲוֹנוֹ בְּחֻבִּי. וְכַאֲשֶׁר הָיָה הַיּוֹם הַשֵּׁנִי. וְיָצָאתִי מִמְּעוֹנִי. וָאֶשָּׂא עֵינַי. וְהִנֵּה חֶבֶר הַקֵּינִי. עַל סוּס רוֹכֵב. וּבִגְדֵי מַלְכוּת סוֹחֵב. וּשְׁנֵי עֲבָדִים רָצִים לְפָנָיו. וַיֶּחֱרַד לִבִּי לְשִׁנּוּי עִנְיָנָיו. כִּי בַלַּיְלָה רְאִיתִיהוּ בְיָגוֹן וּמְגִנָּה. בָּעֶרֶב יָלִין בֶּכִי וְלַבֹּקֶר רִנָּה. וְכַאֲשֶׁר רְאִיתִיהוּ. נָטִיתִי מִן הַדֶּרֶך וַעֲזַבְתִּיהוּ. וַיְמַהֵר וַיֵּרֶד מֵעַל מֶרְכַּבְתּוֹ וַיְחַבְּקֵנִי. וַיִּפֹּל עַל צַוָּארִי וַיִּשָּׁקֵנִי. וַיֹּאמֶר לִי: הִנֵּה יָדַעְתִּי. כִּי מוּסָר פָּרַעְתִּי. וּבשְׁתִּי וְנִכְלַמְתִּי. עַל אֲשֶׁר אָשַמְתִּי. אָמַרְתִּי לוֹ: אֵיךְ הֵעַזְתָּ פָּנֶיךָ. לְחַלֵּל בְּרִית אֱמוּנֶיךָ. וְעַד מָתַי לֹא תֶחְדַּל מֵרֹעַ פְּעָלֶיךָ. וְשִׁנּוּי מַעֲלָלֶיךָ. אָמַר לִי: הָאֵל יוֹדֵעַ כִּי אַהֲבָתְךָ חֲקוּקָה עַל לוּחַ לִבִּי. כָּל עוֹד נִשְׁמָתִי בִי. וְרוּחַ אֱלֹוהַּ בְּקִרְבִּי. אֲבָל צֹרֶךְ גָּדוֹל הִצְרִיכַנִי לְהִפָּרֵד מִמְּךָ. וְלָצֵאת מֵעִמְּךָ. אָמַרְתִּי: הַגֵּד נָא לִי אֵיךְ עָלָה מַזָּלְךָ וְגָבָר. וּמֶה הָיָה הַדָּבָר. כִּי אֶמֶשׁ הָיִיתָ כְנֵצֶר נִתְעָב. מִתְעַטֵּף בָּרָעָב. וְהִנְּךָ הַיּוֹם עַל סוּס רוֹכֵב. וּבִגְדֵי מַלְכוּת סוֹחֵב. אָמַר לִי: כָּכָה מַעֲשֵׂה הָאֱלֹהִים. פַּעַם יַגְבִּיהַּ שְׁפָלִים וּפַעַם יַשְׁפִּיל גְּבוֹהִים. וּפַעַם יַעֲשִׁיר רָשִׁים. וּפַעַם יְרוֹשֵׁשׁ רָאשִׁים. אֲבָל אַגִּיד לְךָ סִבַּת עָשְׁרִי וּכְבוֹדִי. וְאֵיךְ מָצְאָה כַבִּיר יָדִי. כִּי כְשֶׁנִּפְרַדְתִּי מֵעִמְּךָ. וְיָצָאתִי בַלַּיְלָה מֵהֲדוֹמְךָ. סוֹבַבְתִּי בְּכָל הָעִיר. אוּלַי הָאֵל רַחֲמָיו לִי יָעִיר. וְשׁוֹטַטְתִּי עַל הַחֳצֵרִים.

וְעָמַדְתִּי עַל שַׁעַר אֶחָד מִן הַשָּׂרִים.

וָאֶשָּׂא מְשָׁלִי:

מִי זֶה יְרַחֵם דַּל יְלִיד שׁוֹעִים

צָנַח לְאֶרֶץ מִגְּדֻלָּתוֹ.

הָיָה תְמוֹל דּוֹרֵךְ עֲלֵי רֹאשׁ עָשׁ

וַיְסַקְּלוּ אוֹרִים מְסִלָּתוֹ.

אַךְ שַׁח וְשָׁפַל עַד שְׁאוֹל הַיּוֹם

וַיֶחְשְׁכוּ אוֹרֵי תְהִלָּתוֹ.

וַיִּפְגְּעוּ בוֹ שׁוֹלְלֵי דֶרֶךְ

וַיְשׁוֹדְדוּ מַחְמַד סְגֻלָּתוֹ.

יִתְאַו שְׁכוֹן עִם שׁוֹכְנֵי קֶבֶר

לַחְצֹב בְּנִקְרַת צוּר מְחִלָּתוֹ.

עָטָה לְרֹב עָנְיוֹ מְעִיל קַדְרוּת

אָכֵן עֲרָפֶל שָׂם חֲתֻלָּתוֹ.

יִלְאֶה כְּתֹב סוֹפֵר תְּלָאוֹתָיו

לוּ יִהְיֶה שַׁחַק מְגִלָּתוֹ.

עַתָּה הֲיֵשׁ נָדִיב יְרַחְמֵהוּ

יַעְמִיד לְדוֹר אַחְרוֹן תְּהִלָּתוֹ.


וַיְהִי כִּשְׁמֹעַ הַשַּׂר נֹעַם שִׁירָתִי. וְהֵבִין תֹּקֶף מְלִיצָתִי. צִוָּה לַהֲבִיאֵנִי אֵלָיו וַיֹּאמַר: אֶת קוֹלְךָ שָׁמַעְתִּי. וְכָל סוֹדְךָ יָדַעְתִּי. וְנוֹדַע לִי מִטּוּב שִׁירָתְךָ. כֹּחַ תְּבוּנָתְךָ. וְעַתָּה אִם תּוּכַל לָתֵת אֶת שְׁאֵלָתִי וְלַעֲשׂוֹת אֶת בַּקָּשָׁתִי. אֶשְׁבֹּר עֹל הַזְּמַן מֵעָלֶיךָ. וַאֲצַוֶּה הַיְגוֹנִים בַּל קְרֹב אֵלֶיךָ. אָמַרְתִּי לוֹ: כֹּל אֲשֶׁר תִּרְצֶה שְׁאַל וְנַסֵּה. וְכֹל אֲשֶׁר תֹּאמַר אֵלַי אֶעֱשֶׂה. אָמַר לִי: יָדוֹעַ תֵּדַע כִּי אִישׁ מִמִּבְחַר הַשָּׂרִים. וְהַגְּאוֹנִים הַטַּפְסָרִים. מֵתָה אֶתְמוֹל בִּתּוֹ. וְהַיּוֹם יָלְדָה בֵן אִשְׁתּוֹ. וְכָל הַיּוֹם הַזֶּה לְחַבֵּר שִׁירִים יָגַעְתִּי. וְלֹא רָגַעְתִּי. מִיָּם הַסִּכְלוּת חָתַרְתִּי. לְהָשִׁיב אֶל הַיַּבָּשָׁה וְלֹא יָכֹלְתִּי. כִי חָשַׁבְתִּי לַעֲשׂוֹת לוֹ אִגֶּרֶת. מִשְׁתֵי דְרָכִים מְדַבֶּרֶת. וּשְׁנֵי עִנְיָנִים מְחַבֶּרֶת. מְנַחֶמֶת וּמְבַשֶּׂרֶת. וּבוֹכָה וּמְזַמֶּרֶת. מְקוֹנֶנֶת וּמְשׁוֹרֶרֶת. וְלֹא מָצְאוּ חָזוֹן נְבִיאֵי שִׂכְלִי. וַה' הֶעְלִים מִמֶּנִּי וְלֹא הִגִּיד לִי. וְאִם יֵשׁ לְאֵל יְדֵי שִׂכְלְךָ. לְהַשְׁלִים שְׁאֵלָתִי לְךָ. תִּמְצָא כָּל מִשְׁאָלְךָ. אָמַרְתִּי לוֹ: מָצָאתָ הָעִנְיָן. וְשָׁאַבְתָּ מִן הַמַּעְיָן. וּפָגַשְׁתָּ מְחַזֵּק הַבֶּדֶק. וּפָגַעְתָּ אֶת שָׂשׂ וְעוֹשֶׂה צֶדֶק. וְעַתָּה הָבִיאָה לִי הַדְּיוֹ וְהַסֵּפֶר. וַיְצַו לַהֲבִיאָם. וּמָשַׁכְתִּי יָדִי בְשֵׁבֶט סוֹפֵר.

וָאֶשָּׂא מְשָׁלִי:

דִּין יוֹצְרָךְ קַבֵּל וְחֶמְלָתוֹ דְרשׁ

תִּזְכֶּה לְרֹב כָּבוֹד וְגַם תֵּשֵׁב בְּרֹאשׁ.

כּוֹס גִּיל וְתַנְחוּמִים אֱלֹהִים הִשְׁקְךָ

וַיַּשְׁקְ אוֹיְבֶיךּ חֲמַת תַּנִּין וְרוֹשׁ.

וַתַּחֲלֵף בָּנוֹת בְּבָנִים יִנְחֲלוּ

כָבוֹד וְתַחַת נַעֲצוּץ יַעֲלֶה בְרוֹשׁ.


וְכָתַבְתִּי תַּחַת הַשִּׁירִים. אֵלֶּה הַדְּבָרִים. אַחַר אַשֶׁר נִבְקְעוּ עַנְנֵי הַדִּמְעָה. וְעָלְתָה עַד לַשָּׁמַיִם הָאֲנָקָה וְהַשַּׁוְעָה. רָאָה יְיָ' וַיִּנָּחֶם עַל הָרָעָה. וְנֶעֱצַר עֲנַן הַבְּכִי אֲשָׁר פָּרַץ פָּרֶץ. וַיִּכָּלֵא הַגֶּשֶׁם מִן הָאָרֶץ. וְחָרְבוּ הַמַּיִם. מִן הַלְּחָיַיִם. וְנִבְנָה הַהֶרֶס. וְנִקְשַׁר הַקֶּרֶס. וְתַחַת אוֹר הַלְּבָנָה עָלָה אוֹר חֶרֶס. וְאַחַר אֲשָׁר סָר צִלֵּנוּ. וְנָפַל דִּגְלֵנוּ. וְנֶהְפַּךְ לְאֵבֶל מְחוֹלֵנוּ. הֵאִיר יְיָ' פָּנָיו אֵלֵינוּ. וְהֵסִיר אָפְלֵנוּ. וְהֵקִים נָפְלֵנוּ. וְהֶחֱלִיף בְּשָׂשׂוֹן אֶבְלֵנו. וְהָפַךְ לְמָחוֹל חִילֵנוּ. וּמֵאַחֲרֵי אֲשֶׁר רָפוּ הַיָּדַיִם. וְכָהוּ הָעֵינַיִם. שָׁבוּ הָעַרְבַּיִם צָהֳרַיִם. וַיַּעֲמדֹ הַשֶּׁמֶשׁ בַּחֲצִי הַשָּׁמַיִם. וּבְפֶתַע נֶחְבַּשׁ הַפֶּגַע. וּכְרֶגַע נִרְפָּא הַנֶּגַע. וְשָׁבָה הַקִּינָה. נְגִינָה. וְהָאֲנָחָה. שִׂמְחָה. וְהַתַּמְרוּרִים. לְקוֹל שִׁירִים. וְנֶהְפַּךְ הָאָסוֹן. לְשָׂשׂוֹן.
