הקדמה

נְאֻם גֶּבֶר אֲשֶׁר עָלָה וְיָרַד

בְּחוּג הַשִּׁיר, וְאִם שִׂיאוֹ בְּמוֹרָד.

וְהִתְלַקַּח יְקֹד מִלָּיו בְּסִפְרוֹ

כְּהִתְלַקַּח יְקֹד לַהַב בְּבָרָד.

וְצֵל הַמַּעֲלוֹת הֵשִׁיב לְאָחוֹר,

בְּאוֹרוֹ אַחֲרֵי היּוֹם מְאֹד רַד.

וְחִבֵּר מַחְבְּרוֹת מוּסָר חֲמִשִּׁים,

בְּכָל עִנְיָן מְחֻבָּר אוֹ מְפֹרָד.

וְאָסַף כָּל־פְּזוּרֵי הַמְּלִיצוֹת,

וְאִישׁ מֵהֶם בְּלִי נִמְלַט וְשָׂרָד.

וְהַשִּׁיר הוּא לְעֶבֶד נֶאֱמָן לוֹ

וְאִם קָשַׁר עֲלֵי בִלְתּוֹ וּמָרָד

וְשִׁבֵּץ מִשְׁבְּצוֹת מִלִּים אֲרוּגִים

בְּשֵׂכֶל לֹא בְאַרְגָּמָן וּבִשְׂרָד.

בְּאַרְצוֹ רָם לְרוּם עֻזּוֹ וְהוֹדוֹ,

וְאוּלָם בַּנְּדוֹד שָׁפַל וְהוּרָד,

וְשִׂיאוֹ עַד תְּהוֹם שָׁפַל, וְאֶתְמוֹל

לְרוּם גָּבַר וְעִם אֵל וֶאֱנוֹשׁ רָד

וְהָיָה כַעֲנָק לַשִּׁיר – וְאוּלָם

בְּקוֹרוֹת הַנְּדוֹד נִתַּק וְנִפְרָד.

וְאַרְצוֹ הָיְּתָה גַּן הָאֱלֹהִים

וְגָלָה מִנְּאוֹתֶיהָ וְנִטְרָד.

וְעָזַב מִשְׁכְּנוֹת אֶרֶץ מְגוּרָיו,

וְאֶל אֶרֶץ אֲדֹנָי רָץ וְחָרָד.

שְׁמוֹ נוֹדַע יְהוּדָה בֶן־שְׁלֹמֹה,

וְשֵׁם אַרְצוֹ וּמוֹלַדְתּוֹ סְפָרָד.

בְּשֵׁם הַמְלַמֵּד אָדָם דַּעַת

הַחָכְמָה תָּעֹז לְחָכָם בְּאֹרַח חַיִּים תַּנְחֵהוּ. וּמֵעֲמָלוֹ תְּנִיחֵהוּ. בְּשָכְבוֹ תִשְׁמֹר עָלָיו וְלֹא תַנִּיחֵהוּ. וּבַהֲקִיצוֹ הִיא תְשִׂיחֵהוּ. וְהִיא תַזְהִירֵנוּ. לָשׂוּם תְּחִלַּת אֲמָרֵינוּ. תְּהִלַּת יוֹצְרֵנוּ. וּפֶתַח דְּבָרֵינוּ. שֶׁבַח לְצוּרֵנוּ. עָשָׂה שָׁמַיִם. בְּלִי עֲמַל יָדַיִם. וּבְלֹא מִבְטָא שְׂפָתַיִם. אֲחוּזִים בְּלִי כַפַּיִם. תְּלוּיִם כְּמֹאזְנַיִם. מְסוֹבְבִים כְּרֵחַיִם. צִוָּה אוֹתָם וְנִבְרָאוּ. וּקְרָאָם וְנִמְצָאוּ. וְחַיּוֹת מֶרְכָּבָה לָשֵׂאת כִּסְאוֹ נִצְבָּאוּ. כִּי עֲבוֹדַת הַקֹּדֶשׁ עֲלֵיהֶם בְּכָתֵף יִשָּׂאוּ. הוּא הַנִּמְצָא וְעַיִן לֹא תִרְאֶנּוּ. וְהוּא הַנֶּעְלָם וְכָל דּוֹרְשׁוֹ יִמְצָאֶנּוּ. הוּא הַקָּרוֹב וְגַלְגַּלֵּי מַעְלָה לֹא יֵדְעוּ מְכוֹנוֹ. וְהוּא הָרָחוֹק וּבְתוֹךְ הַלֵּב מִשְׁכָּנוֹ. הַבּוֹרֵא גַלְגַּלֵּי שְׁחָקִים. כִּרְאִי מוּצָק חֲזָקִים. וַעֲלֵיהֶם כִּסֵּא הַכָּבוֹד. וְחֶבְיוֹן הַהוֹד. וְשָׁם נְוֵה הַשְּׂרָפִים וְהַתַּרְשִׁישִׁים. וְהַנְּשָׁמוֹת אֲשֶׁר לָאֵל מְשַׁבְּחִים וּמַקְדִּישִׁים. וּבָתֵּי הַנֶּפֶשׁ וְהַלְּחָשִׁים. וְשָׁם יְסוֹד כָּל סוֹד. וְעָדָיו יַגִּיעַ שֵׂכֶל כָּל מַשְׂכִּיל וְיַעֲמֹד. וְאִם יִרְצֶה עַל הַכָּבוֹד הָעֶלְיוֹן לְהַשְקִיף. דְּלָתָיו בְּפָנָיו יָגִיף. וְיֹאמַר לוֹ: עַד פֹּה תָבוֹא וְלֹא תוֹסִיף. וְאֵין לְךָ רְשׁוּת לָדַעַת אֲשֶׁר בְּקִרְבִּי. וְלֹא תַעֲבֹר בִּי. וְנִשְׁמוֹת הַצַּדִּיקִים בְּעוֹדָם בְּגוּפוֹתֵיהֶם. יִרְצוּ לָעוּף אֶל הַכָּבוֹד הַהוּא כְּיוֹנִים אֶל אֲרֻבּוֹתֵיהֶם. אֲבָל בְּכֹבֶד הַגּוּף תִּרְפֶּינָה כַנְפֵיהֶם. וּבְצֵאתָם מִן הַגּוּף אֶל הֶחָצֵר הַחִיצוֹנָה יִפְשְׁטוּ אֶת בִּגְדֵיהֶם. אֲשֶׁר הֵמָּה מְשָׁרְתִים בָּהֶם. וּמִבֵּית גִּנְזֵי הַמֶּלֶךְ. יָבִיאוּ לְבוּשׁ מַלְכוּת אֲשֶׁר לָבַשׁ בּוֹ הַמֶּלֶךְ. וּבְבוֹאָם אֶל בֵּית הַמֶּלֶךְ בִּגְדֵי קֹדֶשׁ יִלְבָּשׁוּ. וְהַנִּגָּשִׁים אֶל ה' יִתְקַדָּשׁוּ. לְפָאֵר לְיוֹצְרָם. וּלְרוֹמֵם לְבוֹרְאָם. הוּא הָאֵל הַבּוֹרֵא נְשָׁמָה טְהוֹרָה. מִבּוֹר הַבּוֹר נֻקָּרָה. וּמִצּוּר הַצּוּר נִגְזָרָה. חָצַב אוֹתָהּ מִן הַיְסוֹד הַקַּדְמוֹן. וַתִּהְיֶה אֶצְלֹו אָמוֹן. עַד עֵת בּוֹא דְבָרָהּ. אָז רָאָהּ וַיְסַפְּרָהּ. הֱכִינָהּ וְגַם חֲקָרָהּ. וַיִּמְצָאָהּ מַלְאַךְ ה' עַל עֵין הַחָכְמָה בְּדֶרֶךְ שׁוּר. אֲשֶׁר מֵרֹאשׁ אֲמָנָה תָשׁוּר. וַיֹאמֶר לָהּ מַלְאַךְ ה‘: עִזְבִי יְסוֹדֵךְ אֲשֶׁר מִשָּׁם חֲצַבְתִּיךְ. וְשִׁכְחִי עַמֵּךְ וּבֵית אָבִיךְ. וִהְיִי לַגּוּף שִׁפְחָה נֶחֱרֶפֶת לוֹ. כִּי הוּא אֲדוֹנַיִךְ וְהִשְׁתַּחֲוִי לוֹ. וַתֵּלֶךְ וַתֵּשֶׁב לָהּ מִנֶּגֶד. עַד נִבְרָא הַגּוּף וַתִּלְבַּשׁ אוֹתוֹ כַבֶּגֶד. וְהֵפִיצָה חֹם יְקוֹדָהּ בַּבָּשָׂר וּבַגֶּלֶד. וַיָּחָם בְּשַׂר הַיֶּלֶד. כִּי בְחֹם הַנְּשָׁמָה יָצָא הָאָדָם מֵאַיִן וַיְהִי יֵשׁ. מִפְּנֵי אֲשֶׁר יָרַד עָלָיו ה’ בָּאֵשׁ. וְהַנֶּפֶשׁ מִגִּזְרָה טְהוֹרָה נֶחְצֶבֶת. וּבְרֹאשׁ הַגּוּף כְּצוֹפָה נִצֶּבֶת. כִּי בֵיתָהּ בְּקִיר הַחוֹמָה וּבַחוֹמָה הִיא יוֹשֶׁבֶת. זָרְחָה בַגּוּף כְּכוֹכָב. וְלָקְחָה אוֹתוֹ לְמֶרְכָּב. מִפִּתּוֹ תֹאכַל וּמִכּוֹסוֹ תִשְׁתֶּה וּבְחֵיקוֹ תִשְׁכָּב. וְנִבְרְאוּ מִצֵּל מְאוֹרָהּ שְׁתֵּי נְפָשׁוֹת חַיָּה וְצוֹמַחַת. חוֹבְרוֹת אַחַת אֶל אַחַת. וּשְׁתֵּיהֶם לְשָׁרֵת הַנֶּפֶשׁ הָעֶלְיוֹנָה תִּלָּוֶינָה. וַתִּגַּשְׁןָ הַשְּׁפָחוֹת הֵנָּה וְיַלְדֵיהֶן וַתִּשְׁתַּחֲוֶינָה. וְהַשֵּׂכֶל נָתַן לְמַעְלָה מִן הַנֶּפֶשׁ מְעוֹנָתוֹ. וּתְשׁוּבָתוֹ הָרָמָתָה כִּי שָׁם בֵּיתוֹ. וְהוּא כְחָתָן יוֹצֵא מֵחֻפָּתוֹ. וְכֶל חֶפְצוֹ וּתְשׁוּקָתוֹ. לְהִצָּמֵד עִם הַנֶּפֶשׁ חֲשׁוּקָתוֹ. וְהוּא אוֹהֵב אוֹתָהּ כְּבִתּוֹ. כַּאֲשֶׁר הָיְתָה בְאַמְנָה אִתּוֹ. וִאם תִּהְיֶה טְהוֹרָה וּקְדוֹשָׁה. מָהֹר יִמְהָרֶנָּה לוֹ לְאִשָּׁה. וְעַל עֶרֶשׂ אֲהָבִים יִתְעַלֵּס בִּנְעִימָה. וּתְפָשָׂהּ וְשָׁכַב עִמָּהּ. וְאָז הַיֵּצֶר יְהִי נִרְדָּם וְלֹא יֵדַע בְּשִׁכְבָהּ וּבְקוּמָהּ. וְהַשֵּׂכֶל יוֹרֶנָּה דַרְכֵי אֱלֹהִים. וִירִימֶנָּה מִן הֶעָפָר לְמַעֲלַת גְּבוֹהִים. וּבוֹ תִנָּצֵל מֵרֶדֶת שִׁאוֹלָה. וְתַעֲלֶה אֶל הַמַּעֲלָה הַגְּדוֹלָה. וְזֹאת תּוֹרַת הָעוֹלָה. כִּי אִם תִּהְיֶה נְקִיָּה. בְּצֵאתָהּ מִן הַגְּוִיָּה. וְתָעוּף לְקִנָּהּ וְקוֹנָהּ בְּלִי כְנָפַיִם. גַּם חֲסִידָה בַשָּׁמָיִם. וְאִם תֶּחֱטָא תֵּשֵׁב בִּשְׁאוֹל גַּלְמוּדָה וְסוּרָה. בְּכָל קֹדֶשׁ לֹא תִגַּע וְאֶל הַמִּקְדָּשׁ לֹא תָבוֹא עַד מְלֹאת יְמֵי טָהֳרָהּ. וְשָׁם תְּקַבֵּל גְמוּלוֹת עָנְשָׁהּ הָרְאוּיוֹת לָהּ מִבֵּית הַמֶּלֶךְ. עַל אֲשֶׁר לֹא עָשְׂתָה אֶת מַאֲמַר הַמֶּלֶךְ. וְלֹא תָבוֹא עוֹד אֶל הַמֶּלֶךְ. כִּי אִם חָפֵץ בָּהּ הַמֶּלֶךְ. כְּשֹׁךְ חֲמַת הַמֶּלֶךְ. וּבָזֶה הַנַּעֲרָה בָאָה אֶל הַמֶּלֶךְ. וְעַל כֵּן שָׁלַח אֱלֹהִים הַשֵּׂכֶל אֵלֶיהָ. לִפְרֹשׂ כְּנָפָיו עָלֶיהָ. וּלְהָאִיר בְּאוֹרוֹ עִוְרוֹנֶיהָ. וּלְרַוּוֹת בְּמֵימָיו צִמְאוֹנֶיהָ. וַיִּפְקַח אֱלֹהִים אֶת עֵינֵיהָ. וַתֵּרֶא בְּאֵר מַיִם. מְשִׁיבֵי נֶפֶשׁ מְאִירֵי עֵינָיִם. וְתָמִיד תִּשְׁאַב מִמְּקוֹרוֹ סוֹד אֱלוֹהַּ מִמַּעַל.

וַתֵּרֵד הָעַיְנָה וַתְּמַלֵּא כַדָּהּ וַתָּעַל. כִּי הַשֵּׂכֶל יְרַוֵּנוּ מִמְּקוֹר הַחַיִּים. מַעְיַן גַּנִּים בְּאֵר מַיִם חַיִּים. וְהוּא יִפְקַח עֵינֵי לִבּוֹתֵינוּ. לִרְאוֹת הָאוֹר הָעֶלְיוֹן בְּמַחְשְׁבוֹתֵינוּ. וּלְהִתְעַדֵּן בִּפְרִי עֵץ הַדַּעַת נִשְׁמוֹתֵינוּ. לְטוֹב לָנוּ כָּל הַיָּמִים לְחַיּוֹתֵנוּ. וְעַל כָּל אֵלֶּה הַטּוֹבוֹת הַגְּדוֹלוֹת. חוֹבָה עָלֵינוּ לַעֲרֹךְ תְּהִלּוֹת. לְנוֹרָא עֲלִילוֹת. וְלֵאמֹר לְפָנָיו: ה' גָּדוֹל אַתָּה וְגָדוֹל שְׁמֶךָ. וּמֶרְחֲבֵי הָעוֹלָם לֹא יְכִילוּן תַּעֲצוּמֶךָ. כִּי אַתָּה מְקוֹם עוֹלָמֶךָ. וְאֵין עוֹלָמְךָ מְקוֹמֶךָ. בְּטֶרֶם הָרִים יֻלָּדוּ.

וְעַמּוּדֵי שָׁמַיִם עָמָדוּ. מוֹשַׁב אֱלֹהִים יָשַׁבְתָּ. וּבְאֵין מָקוֹם נִצַּבִתָּ. וְהַשֵּׂכֶל הָעֶלְיוֹן אֲשֶׁר מִצּוּר כְּבוֹדְךָ חָצַבְתָּ. כְּבֵן יַקִּיר אֵלֶיךָ הִקְרַבְתָּ. וּלְמֶלֶךְ עַל כָּל בְּרוּאֶיךָ הִצַּבְתָּ. אֶת בִּנְךָ יְחִידְךָ אֲשֶׁר אָהָבְתָּ. וּבוֹ יָצַרְתָּ אֶרֶץ וְעֶרֶץ. כְּמוֹ שֶׁנֶּאֱמַר: ה' בְּחָכְמָה יָסַד אָרֶץ. וּבוֹ בָרָאת עוֹלָמְךָ בְּשֵׁשֶׁת יְמֵי הַמַּעֲשֶׂה. וְנָתַתָּה אֶרֶץ לִבְנֵי אִישׁ לְמַחֲסֶה. וְכוֹנַנְתָּ בַּשַּׁחַק חֶבְיוֹן הַמִּכְסֶה וְהַכִּסֵּא. אֲשֶׁר נִסְתַּרְתָּ שָׁם בְּיוֹם הַמַּעֲשֶׂה. וְכָל נֶפֶשׁ מַשְׂכֶּלֶת תִכְסֹף לְהַשִּׂיג סוֹדְךָ אֲבָל תִּלְאֶה. אֶפֶס קָצֵהוּ תִרְאֶה. וְכֻלּוֹ לֹא תִרְאֶה. וְהַחוֹשְׁבִים לְהָבִין סוֹדוֹת שְׁמֵי עֶרֶץ. נְבוּכִים הֵם בָּאָרֶץ. וְשַׁתּוּ בַשָּׁמַיִם פִּיהֶם וּלְשׁוֹנָם תִּהֲלַךְ בָּאָרֶץ. וּבְסוֹדְךָ נִלְאוּ רַעְיוֹנֵינוּ לַעֲמֹד, כִּי עָצַמְתָּ מִמֶּנּוּ מְאֹד. וַאֲשֶׁר מֵהֶם נֶעְלַמְתָּ בְּעֵין לִבָּם רָאוּךָ, וּבְקִרְבָּם מְצָאוּךָ. כִּי אַתָּה קָרוֹב לִמְבַקְשֶׁיךָ.

וְנִמְצָא לְדוֹרְשֶׁיךָ. וְעַתָּה אֲלֹהֵינוּ, בּוֹרְאֵנוּ. הִמָּצֵא לָנוּ בְּכָל קָרְאֵנוּ. וְאַל נִכָּשֵׁל בְּכָל אֲשֶׁר נְחַבֵּר. וֶהְיֵה עִם פִּינוּ בְּכָל אֲשֶׁר נְדַבֵּר. וְאַל תְּבִישֵׁנוּ מֵאֲשֶׁר נְשַׂבֵּר. וּנְצֹר רַעְיוֹנֵינוּ. מִשְּׁגִיאוֹת הֶגְיוֹנֵינוּ. שְׁמֹר דְּרָכֵינוּ. מֵחֲטוֹא בִלְשׁוֹנֵנוּ. עַד נִהְיֶה מִן הַמּוֹצְאִים חֵן בְּעֵינֶיךָ. וְאַל תְּשִׂימֵנוּ מַטָּרָה לְחִצֵּי חֲרוֹנֶךָ. וְהַסְתִּירֵנוּ מֵרֻכְסֵי אִישׁ בְּסֵתֶר פָּנֶיךּ. וּבְמַחֲשַׁכִּים נְסָה עָלֵינוּ אוֹר פָּנֶיךָ. וְנִהְיֶה מֵרְצוּיֶיךָ וְיִרְאֵי שְׁמֶךָ.

נְאֻם יְהוּדָה בַּר שְׁלֹמֹה (בֶּן אַלְחֲרִיזִי) זַ"ל: ה' אֱלֹהִים נָתַן לִי לְשׁוֹן לִמּוּדִים. לְמַלֹּאות עִנְיָנִים חֲמוּדִים. בְּיַד הַשֵּׂכֶל צְמוּדִים. וְהָיוּ לַאֲחָדִים. לִקְשֹׁר מִפְּנִינֵי מִלֵּיהֶם לְצַוְּארֵי הַנְּגִידִים. רְבִידִים. וּבִזְרוֹעוֹת הַיְדִידִים צְמִידִים. בְּשַׁעֲשׁוּעֵיהֶם יִשְׂמְחוּ מְרוּדִים. וּבְעִנְיְנֵיהֶם יַעְלְזוּ חֲסִידִים. וּבְכָל מְדִינָה וּמְדִינָה מַגִּיעַ שִׂמְחָה וְשָׂשׂוֹן לַיְהוּדִים:

(וָאֶשָּׂא מְשָׁלִי):

אֲחַבֵּר חֲמוּדוֹת, מְשָׁלִים וְחִידוֹת,

וְעֶדְרֵי תְעוּדוֹת, אֲשׁוֹבֵב לְמַרְבֵּץ.

וְהַשִּׁיר אֲרִימוֹ, וּמֶלֶךְ אֲשִׂימוֹ,

אֲכַס מַעֲרֻמּוֹ, בְּרִקְמָה וְתַשְׁבֵּץ.

וְיִטְוֶה לְשׁוֹנִי, פְּתִיל רַעְיוֹנִי,

וּבִגְדֵי זְמַנִּי, בְּשִׁירַי אֲשַׁבֵּץ.

וְנִדְּחֵי מְלִיצוֹת, אֲשִׁיבֵם קְבוּצוֹת,

וּמֵהֶם נְפוּצוֹת יְהוּדָה יְקַבֵּץ.

אֲנִי יָשֵׁן וְלִבִּי עֵר. וּבְיָם הַתְּשׁוּקָה סוֹעֵר וּבוֹעֵר. וַיְעִירֵנִי מִשְּׁנַת סִבְלִי, שִׂכְלִי. וַיּוֹרֵנִי וַיֹּאמֶר לִי: אַתָּה בֶן אָדָם. מַה לְּךָ נִרְדָּם. פְּקַח עֵינֵי רַעְיוֹנֶיךָ. וְהַצְבֵּא גְּדוּדֵי הֶגְיוֹנֶיךָ. וַחֲלוּצֵי לְשׁוֹנֶךָ. וְאַתָּה תֶאְזֹר מָתְנֶיךָ. וּלְבַשׁ קְנָאוֹת. לֵאלֹהֵי הַצְּבָאוֹת. וְלִלְשׁוֹן הַקֹּדֶשׁ אֲשֶׁר הִיא לְשׁוֹן הַנְּבוּאוֹת. הַיּוֹרֶדֶת פְּלָאוֹת. וְחַדֵּשׁ כַּנֶּשֶׁר נְעוּרֶיהָ. וּרְפָא שְׁבָרֶיהָ. וּשְׁבֹר מַלְתְּעוֹת כְּפִירֶיהָ. הַנּוֹשְׁכִים אוֹתָהּ בְּשִׁנֵּיהֶם.

וְהַפּוֹעֲרִים עָלֶיהָ פִּיהֶם. וְהַצֵּל טַרְפָּהּ מִבֵּין שִׁנֵּי אֲרָיוֹת. וּפְרִיצֵי חַיּוֹת. וָאֹמַר: אֲהָהּ אֲדֹנִי. הַנְּדוֹד בָּלַל שְׂפָתַי וְהֶגְיוֹנִי. וְהִדְבִּיק לְחִכִּי לְשׁוֹנִי. וּמִי אָנֹכִי כִּי אַצִּיל שֶׂה פְּזוּרָה מִבֵּין שִׁנֵּי כְפִירִים. וְכִי אוֹצִיא לְשׁוֹן הַקֹּדֶשׁ מִבֵּין הַמְּצָרִים. וַיֹּאמֶר לִי: כִּי אֶהְיֶה עִמְּךָ. וּזְרוֹעִי תְּאַמֶּצְךָ בֵּין בְּנֵי עַמְּךָ. וְרָאָה כָל הָעָם אֲשֶׂר אַתָּה בְקִרְבּוֹ אֶת מַעֲשֵׂה ה' כִּי נוֹרָא הוּא אֲשֶׁר אֲנִי עוֹשֶׂה עִמָּךְ. וַיִּשְׁלַח הַשֵּׂכֶל אֶת יָדוֹ וַיַּגַּע עַל פִּי. וְקָדַח לַהֲבֵי רִשְׁפִּי. וַיֹּאמֶר לִי: הִנֵּה נָתַתִּי דְבָרִי בְּפִיךָ. וּבַחֲזוֹן הַשִּׁיר נָבִיא לַגּוֹיִם נְתַתִּיךָ. רְאֵה הִפְקַדְתִּיךָ עַל בָּתֵּי הַסִּכְלוּת לִנְתוֹץ וְלִפְצֹעַ. וְעַל בָּתֵּי הַמְּלִיצוֹת לִבְנוֹת וְלִנְטֹעַ. וַיָּעַר אֱלֹהִים אֶת רוּחִי לְהַעֲלוֹת מִזִּיו הַשְּׁכִינָה לִלְשׁוֹן הַקֹּדֶשׁ מְנוֹרוֹת. לִפְקֹחַ עֵינַיִם עִוְרוֹת. וְהָיוּ לִמְאוֹרוֹת.לְמַעַן דַּעַת כָּל עַמֵּי הָאָרֶץ כִּי לְשׁוֹן הַקֹּדֶשׁ אֵין עֲרוֹך אֵלֶיהָ. בְּצַחוֹת מִלֶּיהָ וְנֹעַם מְשָׁלֶיהָ. כִּי הִיא כַכַּלָּה תַּעְדֶּה כֵּלֶיהָ. הַמְּלִיצוֹת מְעִילֶיהָ. וְרֶקַח הַמּוֹר בְּשׁוּלֶיהָ. וּבְעֵת תָּסִיר צְעִיפָהּ מֵעָלֶיהָ. כָּל עַיִן תְּלוּיָה אֵלֶיהָ. וְכָל לֵב קָשׁוּר בַּחֲבָלֶיהָ.

וּבְנֵי עֲנָק אֲסִירֵי כְבָלֶיהָ. וְגִבּוֹרֵי קְרָב חֲלָלֶיהָ. וְהִיא כְעֵץ חַיִּים תְּרוּפָה בְּעָלֶיהָ. וְהַחָכְמָה תְּחַיֶּה לֵב בְּעָלֶיהָ. וּמִימֵי קֶדֶם הָיוּ חֲסִידֵי עוֹלָם בִּלְשׁוֹן הַקֹּדֶשׁ שְׂמֵחִים. וְהַיּוֹם פָּרִיצֵי עַמֵּנוּ הֶנֶּאֱלָחִים. דְּקָרוּהָ מִצּוּר לְשׁוֹנָם בַּחֲרָבוֹת וּבִרְמָחִים. וְצֶדֶק יָלִין בָּהּ וְעַתָּה מְרַצְּחִים. וְכָל בָּנֶיהָ עָזְבוּ לְשׁוֹנָם וְשָׁבוּ עִלְּגִים. וְכַסְפָּם הָיָה לְסִיגִים. וְהִיא בְּכָל יוֹם בִּבְכִי מְמָרֶרֶת. וְעַל אֹזֶן סֹגֶרֶת וּמְסֻגֶּרֶת. אֲמָרֶיהָ אוֹמֶרֶת. לֵאמֹר: מַה מְּצָאתֶם עָוֶל בְּמִפְעָלַי,כִּי רְחַקְתֶּם מֵעָלָי. הֲלֹא בְיוֹם דִּבֶּר ה' עִם אֲבוֹתֵיכֶם, אָנֹכִי עָמַדְתִּי בֵּין ה' וּבֵינֵיכֶם.

וְלוּחֹות הַבְּרִית הֲלֹא בְרוּחַ פִּי נִכְתָּבוּ, וּמִצּוּר לְשׁוֹנִי נֶחְצֶבוּ. וְכָל הַיָּמִים אֲשֶׁר שָׁכַן כְּבוֹד הָאֵל בְּהֵיכָלוֹ. אֲנִי הָיִיתִי כְשֵׁגָל יוֹשֶׁבֶת אֶצְלוֹ. לְהַגִּיד לָכֶם סוֹדוֹ. וְלָתֵת עֲלֵיכֶם הוֹדוֹ. וּבְנֵיכֶם מְאָסוּנִי וְחִבְּקוּ לְשׁוֹן זָרִים. אֲשֶׁר עֲזָבוּנִי וַיְקַטְּרוּ לֵאלֹהִים אֲחֵרִים. הֶעֱבִידוּ לִלְשׁוֹן קֵדָר לְשׁוֹן יִשְׂרְאֵלִים. וְאָמְרוּ: לְכוּ וְנִמְכְּרֶנּוּ לַיִּשְׁמְעֵאלִים. וְאָמְרוּ לוֹ: שְׁחֵה וְנַעֲבוֹרָה. וַיִּקָּחוּהוּ וַיַּשְׁלִיכוּ אוֹתוֹ הַבּוֹרָה. עַד אָבַד בְּתוֹכָם וּלְשׁוֹן קֵדָר הִקְדִּירַתְהוּ. וְכָאֲרִי טְרָפַתְהוּ. חַיָּה רָעָה אֲכָלַתְהוּ. וְכֻלָּם מָאֲסוּ לְשׁוֹן עִבְרִיָּה. וְחָשְׁקוּ לְשׁוֹן הַגְרִיָּה. וְחִבְּקוּ חֵיק נָכְרִיָּה. וּבְאֵשֶׁת זָרִים חָשָׁקוּ. וְחֵיקָהּ נָשָׁקוּ. כִּי מַיִם גְנוּבִים לָהֶם יִמְתָּקוּ. וְנִפְתָּה לְבָבָם בִּרְאוֹתָם כַּמָּה מְלִיצָה יְקָרָה. אֲשֶׁר יָלְדָה הָגָר הַמִּצְרִית שִׁפְחַת שָׂרָה. וַתְּהִי שָׂרַי עֳקָרָה.

בְּנַפְשִׁי עָבְרָה רוּחַ קְנָאוֹת

לְחָכְמָה נֶעְדְּרָה מֶנּוּ וְסָרָה;

בְּשׁוּרִי, יָלְדָה הָגָר יְלָדִים

חֲמוּדִים וַתְּהִי שָׂרַי עֲקָרָה;

וְעַל כֵּן הִתְעוֹרַרְתִּי. וְעֹז הִתְאַזַּרְתִּי. לְהָשִׁיב עֲקֶרֶת הַבֵּיִת אֵם הַבָּנִים שְׂמֵחָה. וְלָתוּר לָהּ מְנוּחָה. וְלִשְׁאֹב מִמְּקוֹרָהּ הַמְרַוֶּה צְמֵאִים. כִּי מֵימָיו מִן הַמִּקְדָּשׁ הֵמָּה יוֹצְאִים. וָאָבוֹא הַיּוֹם אֶל הָעַיִן. לִרְווֹת מִדּוֹדֵי הַחָכְמָה הַטּוֹבִים מִיָּיִן. וְאָמַרְתִּי: אָנָּא אֵל אָיֹם. הַקְרֵה נָא לְפָנַי הַיּוֹם. בְּנוֹת הַבִּינוֹת. וְעַלְמוֹת הַתַּעֲלוּמוֹת. הִנֵּה אָנֹכִי נִצָּב עַל עֵין הַמַּיִם וְהָיָה הַנַּעֲרָה. הַיּוֹצֵאת הַשַּׁעְרָה. וְאָמַרְתִּי אֵלֶיהָ הַשְׁקִינִי נָא מִנּוֹזְלֵי מְלִיצוֹתַיִךְ. הַנּוֹטְפִים מִבֵּין שִׂפְתוֹתַיִךְ. וְאָמְרָה: שְׁתֵה. וְאֵשׁ לְהָבִים מִבֵּין שְׂפָתַי חֲתֵה. אוֹתָהּ לְעַבְדְּךָ הוֹכַחְתָּ, וּבָהּ אֵדַע כִּי חֶסֶד עָלַי גָּמַלְתָּ. אֲנִי טֶרֶם אֲכַלֶּה לְדַבֵּר אֶל לִבִּי. וָאֲצַפֶּה לִרְאוֹת מַה־יְדַבֶּר־בִּי. וְהִנֵּה עַלְמָה. בָּרָה כַּחַמָּה. מִבְּנוֹת הַחָכְמָה. יוֹצֵאת לִשְׁאוֹב מַיִם וְכַדָּהּ עַל שִׁכְמָהּ. טֶרֶם קְרָאתִיהָ עָנַתְנִי. וּבְצוּף שְׂפָתֶיהָ הִרְוַתְנִי. וַתֹּאמֶר לִי: שְׁתֵה אֲדוֹנִי. מִנּוֹזְלֵי הֶגְיוֹנִי. כִּי דְבַשׁ וְחָלָב תַּחַת לְשׁוֹנִי. וָאֶשְׁאַל אוֹתָהּ וָאֹמַר: בַּת מִי אָתְּ. וּמֵאֵיזֶה מַחְצָב חֻצָּבְתְּ? אָמְרָה: אֲנִי יְתוֹמָה וְאָבִי חָי. וְנִחֲרוּ בִי בְּנֵי אִמִּי וְאֶחָי. אֲנִי הָיִיתִי כֶּתֶר מְלוּכָה. וְהִנְנִי הַיּוֹם לְכָל רֶגֶל דְּרוּכָה. אֲנִי לְשׁוֹן הַקֹּדֶשׁ גְּבִרְתֶּךָ. וְאִם אִיטַב בְּעֵינֶיךָ אֶהְיֶה חֲבֶרְתֶּךָ. אַךְ לְמַעַן תְּקַנֵּא לְשֵׁם הָאֵל. וְקִדַּשְׁתָּ אֶת שְׁמִי הַגָּדוֹל אֲשֶׁר חִלְּלוּהוּ בֵּית יִשְׂרָאֵל. וְהָיִיתָ לִי לְגוֹאֵל. מִכָּל מְגָאֵל וּמִתְגָּאֵל. וָאֶקֹּד וָאֶשְׁתַּחֲוֶה לָאֵל. כִּי מָצָאתִי חֵן בְּעֵינֶיהָ. וְשַׂמְתִּי נִזְמֵי תְהִלּוֹתַי בְּאָזְנֶיהָ. וַעֲנַק מְלִיצוֹתַי בְּצַוְּרוֹנֶיהָ. וְאֵרַשְׂתִּיהָ לִי בְּצֶדֶק וּבְיִרְאָה. מִבְּלִי שְׁטָר וּבִיאָה. וָאֶקְרַב אֶל הַנְּבִיאָה. וַתַּהַר וַתֵּלֶד בֵּן בִּדְמוּת אֱלֹהִים וְצַלְמוֹ. וַתְּהִי הַמִּשְׂרָה עַל שִׁכְמוֹ. וַתָּחֶל רוּחַ ה' לְפַעֲמוֹ. וַיֶּאֱסֹף אֵלָיו כָּל עַמּוֹ. וַיַּעַל לָצוּר עַל מִבְצְרֵי הַמְּזִמּוֹת הָרָמוֹת. וְלִלְכֹּד מְצָדוֹת הַתְּעוּדוֹת. בְּמַחֲשָׁבוֹת. חֲצוּבֵי לֶהָבוֹת. לְטוּשֵׁי חֲרָבוֹת. וְשִׂיחִים. מְרוּקֵי שְׁלָחִים. אֲחוּזֵי רְמָחִים. וְלָשׁוֹן רַכָּה. לֵב הָאֲרִי מְדַכָּה. וְתִמְשֹׁךְ לִוְיָתָן בְּחַכָּה. וּבְשׁוֹט חֲמָתָהּ מַכָּה. תָּשֵׁב אֱנוֹש עַד דַּכָּא. תְּלַבֵּב לְבָאִים. וְתִכְבּשׁ פְּרָאִים. וְתַפִּיל רְפָאִים. וְתָקִים רְפָאִים. בְּנִיבִים. מִיקֹד לַהַב חֲצוּבִים. וּמִמְּקוֹר עֵדֶן שְׁאוּבִים. פַּעַם יָפִיצוּ שְׁבִיבִים. וּפַעַם יַטִּיפוּ רְבִיבִים. וְעִנְיָנִים כֻּלָּם גִּבּוֹרִים חֲלוּצֵי צָבָא. אֶחָד יִרְדֹּף אֶלֶף וּשְׁנַיִם יָנִיסוּ רְבָבָה.

וְהָעִנְיָן אֲשֶׁר הֵעִיר אֶת רוּחִי לְחַבֵּר הַסֵּפֶר הַזֶּה, כִּי אִישׁ חָכָם מֵחַכְמֵי יִשְׁמְעֵאלִים. וּמִמִּבְחַר הַמַּשְׂכִּילִים. לְשׁוֹנוֹ בִמְלִיצוֹת עֲרָב נִמְרָץ. וְעַל פִּיו חֲזוֹן הַשִּׁיר נִפְרָץ. וְנוֹדַע שְׁמוֹ אַלְחֳרִירִי. וְכָל אֲבִי מְלִיצוֹת זוּלָתוֹ הוֹלֵךְ עֲרִירִי. חִבֵּר בִּלְשׁוֹן עֲרָב סֵפֶר. נוֹתֵן אִמְרֵי שֶׁפֶר. אַף עַל פִּי שֶׁעִנְיָנָיו מִלְשׁוֹן הַקֹּדֶשׁ גְּזוּרִים. וְכָל מְשָׁלָיו הַיְקָרִים. מִסְּפָרֵינוּ לְקוּחִים וּמְסוּרִים. וְאִם תִּשְׁאַל לְכָל מְלִיצָה מִמְּלִיצוֹתָם מִי הֱבִיאֵךְ לִלְשׁוֹן הַגְרִים. תֹּאמַר כִּי גֻנֹּב גֹּנַּבְתִּי מֵאֶרֶץ הָעִבְרִים. וּבִרְאוֹתִי הַסֵּפֶר הַהוּא נָגֹלּוּ כַסֵּפֶר שְׁמֵי גִילִי. וְנָזֹלוּ יִבְלֵי אֶבְלִי. יַעַן כָּל עַם וָעָם נִזְהָרִים בְּהֶגְיוֹנָם. נִשְמָרִים מֵחֲטוֹא בִלְשׁוֹנָם. וּלְשׁוֹנֵנוּ אֲשֶׁר הָיָה מַחְמָד לְכָל עַיִן. נֶחְשָׁב כַּאֲחִי קַיִן. וְשָׁב הַיֹפִי דֹּפִי. וְהַנֹּעַם זַעַם. וּכְבוֹדוֹ נֶעְדָּר וְאוֹרוֹ קָדָר. וְלֹא תֹאַר לוֹ וְלֹא הָדָר. כִּי בְנֵי עַמֵּנוּ דִבְּרוּ עָתָק עָלָיו. וְהוֹרוּ חֵץ הַלַּעַג אֵלָיו וּפִלְּחוּ כְּסָלָיו. וְאָמְרוּ כִּי לְשׁוֹנֵנוּ קְצֵרָה. וּמְלִיצָתָהּ חֲסֵרָה. וְהֵם לֹא יָדְעוּ כִּי הַחֶסְרוֹן מֵהֶם אֲשֶׁר לֹא יָבִינוּ אִמְרוֹתֶיהָ. וְלֹא יַכִּירוּ חֲמוּדוֹתֶיהָ. כְּמוֹ מִי שֶׁיֵּשׁ לוֹ חֳלִי בְּעֵינָיו. וְתֶחֱשַׁךְ לוֹ הַשֶּׁמֶשׁ הַמְּאִירָה לְפָנָיו. וְהוּא חוֹשֵׁב כִּי בַשֶּׁמֶשׁ מִקְרֶה אוֹ פֶגַע. וְלֹא יָדַע כִּי בְּעֵינָיו עָמַד הַנֶּגַע. וְכֵן רֹב בְּנֵי עַמֵּנוּ יִבְזוּ לְשׁוֹן הַקֹּדֶשׁ כִּי עִנְיָנָיו מֵהֶם נִפְלָאוּ. וְעֵינַיִם לָהֶם וְאוֹרוֹ לֹא רָאוּ. וְהַמָּן לִפְנֵיהֶם וַיֵּצְאוּ לִלְקֹט וְלֹא מָצָאוּ. וְנַהֲרֵי עֵדֶן לִפְנֵיהֶם וְהֵם יִצְמָאוּ.

אוֹי עַל פְּתָאִים כַּפְּרָאִים יֶהֱמוּ,

אֶצְלָם מְקורֹ עֵדֶן וְהֵם יִצְמָאוּ.

הַמָּן לְעֵינֵיהֶם וְעֵינָם נִסְגְּרָה

וַיֵּצְאוּ לִלְקֹט וְלֹא מָצָאוּ.

עַל כֵּן חִבַּרְתִּי הַסֵּפֶר הַזֶּה לְהַרְאוֹת כֹּחַ לְשׁוֹן הַקֹּדֶשׁ. לְעַם הַקֹּדֶשׁ. אֲשֶׁר טַח מֵרְאוֹת עֵינֵיהֶם. מֵהַשְׂכִּיל רַעְיוֹנֵיהֶם. וְכָל מְבַקֵּשׁ לִרְאוֹת נֹעַם עִנְיָנָיו. וּלְהִשְׁתַּעֲשֵׁעַ בְּגַנֵּי הֶגְיוֹנָיו. וְלִשְׁאֹב מִנּוֹזְלֵי מַעְיָנָיו. יִמְצָא בְתוֹכוֹ גַנּוֹת וּפַרְדֵּסִים. וַעֲרוּגוֹת הֲדַסִּים. וְנִטְפֵי רְסִיסִים. וְתוֹרִים וְסִיסִים. וְכוֹסוֹת עֲסִיסִים. כַּמַּיִם לַיָּם מְכַסִּים. וּבָתֵּי מִדּוֹת. וְשִׁדָּה וְשִׁדּוֹת. זֹאת מְנַגֶּנֶת. וְזֹאת מְרַנֶּנֶת. וְזֹאת תַּעִיר אֲהָבִים. וְזאֹת תַּבְעִיר בַּקְּרָבִים לְהָבִים. בְּשִׁירֵי עֲגָבִים. וְנִיבִים עֲרֵבִים. מִסַּלְעֵי צְלָעַי חֲצוּבִים. וּמֵעֲצֵי עֲצָתִי קְצוּבִים. וּבִמְלִיצוֹת בְּתוּלוֹת. לֹא הָיוּ לְאִישׁ בְּעוּלוֹת. וְזָר לֹא עָלָה עַל עֶרֶשׂ יְצוּעָם. וְאִישׁ לֹא יְדָעָם. יָשִׁיבוּ הַנְּפָשׁוֹת בְּרוּחָם וְרֵיחָם. וִיחַיּוּ הַלְּבָבוֹת בְּלִקְחָם וְרִקְחָם. וְיָעִירוּ לֵב נִרְדָּם בְּנִרְדָּם. וְכָלַלְתִּי בַּסֵּפֶר הַזֶּה מְשָׁלִים רַבִּים. וְעִנְיָנִים עֲרֵבִים. בָּם מִינֵי מְלִיצוֹת. וְחִידוֹת נִמְרָצוֹת. וְדִבְרֵי תְעוּדוֹת. וְשִׁירֵי יְדִידוֹת. וּמִשְׁלֵי נְכוֹחוֹת. וּמִלֵּי תוֹכָחוֹת. וְקוֹרוֹת זְמַנִּים. וִחִדּוּשֵׁי הַשָּׁנִים. וְזִכְרוֹן הַמָּוֶת. וּמְקוֹם הַצַּלְמָוֶת. וְדִבְרֵי תְשׁוּבוֹת. וּסְלִיחַת הַחוֹבוֹת. וְתַעֲנוּגֵי אֲהָבִים. וְשִׁירֵי עֲגָבִים. וְנִשּׂוּאֵי נָשִׁים וְחֻפָּה וְקִדּוּשִׁים. (וְעִנְיְנֵי גֵרוּשִׁים). וְשִׁכְרוּת שִׁכּוֹרִים.

וּנְזִירוּת נְזִירִים. וּמִלְחֲמוֹת הַגִּבּוֹרִים. וְקוֹרוֹת מְלָכִים. וַהֲלִיכוֹת דְּרָכִים. וְשִׁירֵי תְהִלּוֹת. וְתַחֲנוּנֵי תְפִלוֹת. וּמוּסַר חֲכָמִים. וְחֶבְרַת תְּמִימִים. וְחֵשֶׁק עֳפָרִים. וְגַנִּים וּכְפָרִים. וְדִבְרֵי נְגִידִים. וְשִׂיחַת יְלָדִים. וְצֵיד הַצַּיָּדִים. וְעָרְמַת הָרַמָּאִים. וְסִכְלוּת הַפְּתָאִים. וְחֵרוּף הַמְחָרְפִים. וְגִדּוּף הַמְגַדְּפִים. וְשִׁירִים נִפְלָאִים. וְאִגְּרוֹת מִלִּים נוֹרָאִים. לְמַעַן יִהְיֶה הַסֵּפֶר הַזֶּה כְּגַנָּה אֲשֶׁר בָּהּ כָּל מִינֵי מַעֲדַנִּים. וְנִטְעֵי נַעֲמָנִים. וּבוֹ יִמְצָא כָל שׁוֹאֵל מִשְׁאָלוֹ. וְיַשִּׂיג מִתַּאֲוָתוֹ דֵּי מַחֲסוֹרוֹ אֲשֶׁר יֶחְסַר לוֹ. וַהֲכִינוֹתִי בוֹ לְכָל אִישׁ מְכוֹנוֹ וּמִקְרָאוֹ. אִישׁ אִישׁ עַל עֳבוֹדָתוֹ וְעַל מַשָּׂאוֹ. הַיָּרֵא אֶת דְּבַר ה‘. יִמְצָא בוֹ תוֹכָחוֹת וּתְפִלּוֹת וְיִרְאַת ה’. אֲשֶׁר לֹא שָׂם לִבּוֹ אֶל דְּבַר ה' יִמְצָא בוֹ תַעֲנוּגֵי עוֹלָם וְטוֹבוֹתָיו וְנָהֲרוּ אֶל טוּב ה'. וְכָל כְּסִיל וּבַעַר. יָשִׁישׁ אוֹ נַעַר. יִלְמַד מִמֶּנּוּ חָכְמָה. לָתֵת לִפְתָאִים עָרְמָה. לְנַעַר דַּעַת וּמְזִמָּה. וְכַל חֲכַם לֵב אִם עֵינָיו עָלָיו יִפְקַח. יִשְׁמַע חָכָם וְיוֹסִיף לֶקַח.

וְכָל מְבַקֵּשׁ לִלְמֹד דֶּרֶך אֶרֶץ בּוֹ יַשְׂכִּיל בְּכָל אֲשֶׁר יִפְנֶה. וְנָבוֹן תַּחְבּוּלוֹת יִקְנֶה. וְכָל הַמְחַבֵּר שִׁירִים וְיַחְצֹב בּוֹרוֹת נִשְׁבָּרִים. וּמֵימֵיהֶם מָרִים. בַּסֵּפֶר הַזֶּה יִפְקַח עֵינַיִם. וְיֹרֵהוּ ה' עֵץ וְיַשְׁלֵךְ אֶל הַמַּיִם.

וְיִמְתְּקוּ הַמָּיִם. וְכָל אִישׁ אֲשֶׁר תִּרְעַב נַפְשׁוֹ לְלֶחֶם חֲמוּדוֹת. וּמַטְעַמֵּי שִׁיר יְדִידוֹת. וּמַמְתַּקֵּי מְשָׁלִים וְחִידוֹת. יִמְצָא בַסֵּפֶר הַזֶּה עֶשֶׂר יָדוֹת. כַּד וְצַפַּחַת. וּבָהֶם בִּרְכַּת ה' פּוֹרַחַת. וְאִם יִקַּח מֵהֶם שֵׂכֶל וּמוּסָר. כַּד הַקֶּמַח לֹא תִּכְלֶה וְצַפַּחַת הַשֶּׁמֶן לֹא תֶחְסָר. כִּי שֻׁלְחָנוֹ לְכָל בָּא עָרוּךְ. וּבֵיתוֹ לְעוֹלָם בָּרוּךְ. וְלַחְמוֹ בְּכָל פֶּה אָכוּל. וְכָל מַאן דִּכְפִין יֵיתֵי וְיֵיכוּל. וְכַמָּה יוֹעִיל הַסֵּפֶר הַזֶּה לְאַנְשֵׁי כָל מְדִינָה. אֲשֶׁר מִמִּצְרַיִם וְעַד עֲדִינָה. אֲשֶׁר רַבִּים מֵהֶם לְשׁוֹנָם לְדַבֵּר בִּלְשׁוֹן הַקֹּדֶשׁ צוֹלֵעָה. וּקְרִיאָתָם גְרוּעָה וְרָעָה. כִּי בַּסֵּפֶר הַזֶּה דַּרְכֵיהֶם יַיְשִׁירוּ. וּנְכוֹחוֹת יְדַבֵּרוּ. וְהֵיטִיבוּ אֲשֶׁר דִּבֵּרוּ. וּבוֹ יָקִיצוּ מִשְּׁנַת הַסִּכְלוּת רַעְיוֹנֵיהֶם. וְתִפָּקַחְנָה עֵינֵיהֶם. וְיֵדְעוּ כִּי עֲרֻמִּים הֵם. וְיִתְפְּרוּ עֲלֵה מְלִיצוֹת וְשִׁירוֹת. וְיַעֲשׂוּ לָהֶם חֲגוֹרוֹת. וּבְכָל־הַדְּבָרִים אֲשֶׁר בַּסֵּפֶר הַזֶּה זָכַרְתִּי, דָּבָר מִסֵּפֶר הַיִּשְׁמְעֵלִי לֹא לָקַחְתִּי. לְבַד אִם שָׁכַחְתִּי. אוֹ אִם בָּאוּ לְיָדַי דְּבָרִים בְּמִקְרֶה וְאָנֹכִי לֹא יָדַעְתִּי. אַךְ כָּל עִנְיְנֵי הַסֵּפֶר הַזֶּה מִלְבָבִי נִבְרָאוּ. חֲדָשִׁים מִקָרוֹב בָּאוּ. וּמִמֵּי יְהוּדָה יָצָאוּ. וְרַבִּים מִיְּשֵׁנֵי אַדְמַת הַסִּכְלוּת הֵקִיצוּ. וְרִכְבֵי לְשׁוֹנָם בִּרְחוֹב הַשִּׁיר הֵרִיצוּ. וְחָשְׁבוּ לַהֲפֹךְ סֵפֶר הַיִּשְׁמְעֵאלִי הַזֶּה מִלְּשׁוֹן עֲרָב לִלְשׁוֹן הַקֹּדֶשׁ. וּבָאוּ בְּבִגְדֵי חֹל לְשָׁרֵת בַּקֹּדֶשׁ. וּבְצֵאתָם לְמִלְחֶמֶת הַמְּלִיצוֹת חֲמוּשִׁים. לֹא שָׁלְלוּ כִּי אִם אֶחָד אָחוּז מִן הַחֳמִשִּׁים. כְּי לְחֹזֶק מְלִיצוֹת הַסֵּפֶר נִבְהֲלוּ וַיִּבָּעֵתוּ. וּלְקוֹל רְעָמָיו וְאַבְנֵי אֶלְגְּבִישָׁיו גָּוְעוּ וַיִּצָּמֵתוּ. וַיֵּרֶד עֲלֵיהֶם הַבָּרָד וָמֵתוּ. וּבְעֵת לִרְדֹּף אַחֲרָיו נִצְבָּאוּ. דָּפְקוּ שְׁעָרָיו וּבָהֶם לֹא בָאוּ. וַיְהִי הַשַּׁעַר לִסְגֹּר בַּחשֶׁךְ וְהָאֲנָשִׁים יָצָאוּ. וַיְבַקְשׁוּ הָרוֹדְפִים בְּכָל הַדֶּרֶך וְלֹא מָצָאוּ. עַד שַׁקַּמְתִּי. וּמַחְלְצוֹתָיו רָקַמְתִּי. הֶעְתַּקְתִּי כָּל הַסֵּפֶר בְּמִלִּים נְכֹחִים. וְשִׁירִים כִּבְדֹלָחִים. טְהוֹרִים מְמֻלָּחִים. יַשְׁכִּיבוּ כָל הַמְשׁוֹרְרִיִם הַצַּחִים. תַּחַת אַחַד הַשִּׂיחִים. וְקָרָאתִילִמְרוֹם הַשֵּׂכֶל וְנִפְתְּחוּ שָׁמָיו. וַיֵּרְדוּ גְשָׁמָיו. וְרָווּ תְלָמָיו. וְהֵפִיחוּ בְשָׂמָיו. וְדִבַּרְתִּי אֶל סֶלַע הַשִּׁיר וְנָתַן מֵימָיו. כִּי נְדִיבֵי סְפָרַד בְּשָׁמְעָם דִּבְרֵי סֵפֶר הַיִּשְׁמְעֵאלִי תָּמְהוּ עֲלֵיהֶם. וּבִקְשׁוּ מִמֶּנִּי בִּהְיוֹתִי בֵינֵיהֶם. לְהַעְתִּיק הַסֵּפֶר הַזֶּה לָהֶם. וְלֹא יָכֹלְתִּי לְהָשִׁיב אֶת פְּנֵיהֶם. וְכַאֲשֶׁר חֶפְצָם הִשְׁלַמְתִּי. וְהַסֵּפֶר הֶעְתַּקְתִּי. בֵּיתִי נָטַשְׁתִּי. וּדְרָכִים אָרַחְתִּי. וָאֳנִיּוֹת רָכַבְתִּי. וְיַמִּים עָבַרְתִּי. וּמִמַּעֲרָב בָּרַחְתִּי. וּבְמִזְרָח זָרַחְתִּי. וָאֵרָא כִּי הִסְכַּלְתִּי עֲשׂוֹ. וְגָדַל עֲוֹנִי מִנְּשׂוֹא. בְּעָזְבִי לְחַבֵּר סֵפֶר מְלִיצוֹתֵינוּ. וְהָלַכְתִּי לְהַעְתִּיק סֵפֶר זוּלַתֵנוּ. כְּאִלּוּ אֵין דִּבְרֵי אֱלֹהִים חַיִּים בֵּינוֹתֵינוּ. וְכֶרֶם זָרִים לִנְטֹר מִהַרְתִּי. וְכַרְמִי שֶׁלִּי לֹא נָטָרְתִּי. וְעַל כֵּן חִבַּרְתִּי הַסֵּפֶר הַזֶּה בִּמְלִיצוֹת חֲדָשׁוֹת. מִלְּשׁוֹן הַקֹּדֶשׁ קְדוֹשׁוֹת. וְעִנְיָנִים חֲדָשִׁים יְחַוּוּ הַנְּפָשׁוֹת. וְיַחֲלִיצוּ הָעֲצָמוֹת הַיְבֵשׁוֹת.

וְחִלַּקְתִּים לַחֲמִשִּׁים מַחְבֶּרֶת. כָּל אַחַת מֵהֶם כְּאַבְקַת רוֹכֵל מְקֻטֶּרֶת. וּבִלְשׁוֹן נְבוּאוֹת מְדַבֶּרֶת. וְכָל דִּבְרֵי הַסֵּפֶר הַזֶּה בִּלְשׁוֹן הֵימָן הָאֶזְרָחִי עֲשִׂיתִים. וְעַל שֵׁם חֶבֶר הַקֵּינִי יְסַדְתִּים וּבְנִיתִים. וְאַף עַל פִּי שֶׁאֶחָד מֵהֶם בְּדוֹרֵנוּ לֹא הָיָה. וְכָל אֲשֶׁר דִּבַּרְתִי בִּשְׁמָם לֹא הָיָה וְלֹא נִבְרָא אֶלָּא מָשָׁל הָיָה. וַאֲנִי מְחַלֶּה פְּנֵי כָל קוֹרֵא בוֹ בְּמָצְאוֹ דְבַר חֵטְא וְחוֹבָה. יָדִין אוֹתִי לְטוֹבָה. וְעַל כָּל פְּשָׁעִים תְּכַסֶּה אַהֲבָה. וְאַל תְּהִי בְלִבּוֹ חַטַּאת יְהוּדָה כְתוּבָה. כִּי אֵין אָדָם אֲשֶׁר לֹא יֶחֱטָא. בְּמִפְעָל אוֹ בְּמִבְטָא. כִּי הַחֵטְא מָצוּי בְּסִפְרֵי הַגְּאוֹנִים בְּעֵת יִבָּחֲנוּ. וְהִנֵּה הַמְּלָכִים לֹא עָמְדוּ לְפָנָיו וְאֵיְך נַעֲמֹד אֲנַחְנוּ. וְאִם הַגִּבּוֹרִים יִכָּשְׁלוּ בְדִבְרֵיהֶם. וְלֹא יִפָּלְטוּ מִשְּׁגִיאוֹת סִפְרֵיהֶם. מַה יַּעֲשֶׂה נָעוּר וְרֵק מִכָּל חָכְמָה כָּמוֹנִי. וְאִם אֵשׁ הַשְּׁגִיאוֹת נָפְלָה בְּכָל אֶרֶז אַדִּיר. מַה יַעֲשׂוּ אֲזוֹבֵי קִיר. וְהָאֵל הַמְלַמֵּד אָדָם דַּעַת לְדִבְרֵי ישֶׁר יְזַכֵּנוּ. וּמִשְּגִיאוֹת יְנַקֵּנוּ. וְיִהְיֶה עִם פִּינוּ בְּכָל דְבָרֵינוּ. וְיִשְׁלַח מִקֹּדֶשׁ עֶזְרֵנוּ. וִימַלֵּא מִשְׁאֲלוֹתֵינוּ.

אָמַר הַמְחַבֵּר בִּרְאוֹתִי הַסֵּפֶר הַזֶּה נִפְלָא בְעֶרְכּוֹ. וְהַטּוֹעֵם אוֹתוֹ יִמְצָא נֹפֶת מָתוֹק עַל חִכּוֹ. בִּקַּשְׁתִּי נְדִיבֵי תֵבֵל. אֲשֶׁר מִמִּצְרַיִם וְעַד בָּבֶל. אִישׁ אֲשֶׁר אֲפָאֵר הַסֵּפֶר בִּשְׁמוֹ וְיִהְיֶה חָתוּם בְּחוֹתָמוֹ. וּבְעֵת בּוֹאִי בְּאַלְפֵי הַזְּמַן דְּרַשְׁתִּיהוּ. בִּקַּשְׁתִּיהוּ וְלֹא מְצָאתִיהוּ. וְלֹא עָנָנִי אִישׁ בְּעֵת קְרָאתִיהוּ. כִּמְעַט שֶׁעָבַרְתִּי מֵהֶם עַד שֶׁמָּצָאתִי אֵת שֶׁאָהֲבָה נַפְשִׁי. אִישׁ אֲשֶׁר בָּחַר בּוֹ ה' הַשַּׂר הַיָּחִיד בְּדוֹרוֹ. הַחֶסֶד אֲזוֹרוֹ. וְהַיֹּשֶׁר חֲגוֹרוֹ. וְהַחָכְמָה פְאֵרוֹ. כַּשֶּׁמֶשׁ בְּצֵאתוֹ. וְכַיָּם נִדְבָתוֹ. וְכָעֹפֶר בִּנְעִימָתוֹ. וְכָאֲרִי בְּאֵימָתוֹ. וּכְיוֹסֵף בְּיִרְאָתוֹ. וְדָוִד בְּצִדְקָתוֹ. וְקֹהֶלֶת בְּחָכְמָתוֹ. אִם יִפְתַּח פִּיו יְסַכֵּל חֲכָמִים. וְיָשִׁיב בַּעֲלֵי לָשׁוֹן נֶאֱלָמִים. וְאִם יִתְיַחֵס יַעֲזֹב כָּל בַּעֲלֵי הַיַּחַס שְׁפָלִים. יִתְנַדֵּב יָשִׂים הַנְּדִיבִים נְבָלִים. אֹסֵף הַמִּדּוֹת הַחֲמוּדוֹת הוּא הַגְּבִיר הַנֶּאְדָּר. הַכְּרוּב הַנֶּהְדָּר. פְּאֵר דּוֹרֵנוּ. וְהַדְרַת גְּבִירֵנוּ. רוּחֶ אַפֵּינוּ. וְאוֹר עַפְעַפֵּינוּ. וְכוֹכַב נִשְׁפֵּנוּ. יָדוֹ וּבֵיתוֹ פָתוּחַ לִרְוָחָה. לְקַבֵּל כָּל אוֹרֵחַ בְּשִׂמְחָה. הוּא הַשַּר הַגָּדוֹל. אֲשֶׁר הוּקַם בְּיִשְׂרָאֵל חוֹמָה וּמִגְדּוֹל. הַגְּבִיר הַמְעֻלֶּה רַבִּי שְׁמוּאֵל בֶּן אַלְבַּרְקוֹלִי זצ"ל. מִצְרַיִם תִּתְאַוֶּה לִהְיוֹת מִדְרַךְ אַדְמָתוֹ. וְדַמֶּשֶׂק מְנוּחָתוֹ. וְצוֹבָא תְקַנֵּא בִּשְׁמוּעָתוֹ. וְאַשּׁוּר תָּעִיד גְּדֻלָּתוֹ. וַעֲדִינָה תִתְפָּאֵר כִּי הִיא אֲמָתוֹ. וּבִרְאוֹתִי כִּי הוּא יָחִיד בְּעַמּוֹ. וַתְּהִי הַמִּשְׂרָה עַל שִׁכְמוֹ. עָשִׂיתִי הַסֵּפֶר הַזֶּה עַל שְׁמוֹ. וַחֲתַמְתִּיו בְּחוֹתָמוֹ. וְשַׂמְתִּיו כְּמוֹ רְבִיד הַזָּהָב עַל צַוָּארוֹ. וְהִלְבַּשְׁתִּיו מְעִיל פְּאֵרוֹ. וְהִרְכַּבְתִּיו עַל סוּס הֲדָרוֹ. וְקָרָאתִי לְפָנָיו: כָּכָה יֵעָשֶׂה לָאִישׁ אֲשֶׁר הַמֶּלֶךְ חָפֵץ בִּיקָרוֹ. כִּי יְדַעְתִּיו כִּי בִרְאוֹתוֹ יְלָדַי. מַעֲשֵׂי יָדָי. בִּידֵי אַהֲבָה יִסְמְכֵנִי. וְעַל בָּמֳתֵי שַׁחַק יַדְרִיכֵנִי. וּלְזוּלָתוֹ לֹא יַצְרִיכֵנִי. הָאֵל יַעֲמִיד הוֹדוֹ. וּבְדוֹר דּוֹרִים כְּבוֹדוֹ. וְהוּא מְרוֹמִים יִשְׁכֹּן. כְּיָרֵחַ יִכּוֹן. לְפָנָיו יַעֲמֹד זִכְרוֹ הַטּוֹב לְדוֹר אַחֲרוֹן. וְכָל הַשּׁוֹמֵעַ אוֹתוֹ יָרֹן. וְיִהְיֶה בּוֹ לִבְעָלָיו תַּגְמוּל וְיִתְרוֹן. כִּי זֶה הַסֵּפֶר הוּא כַעֲנָק לֹא יִתְיַפֶּה אֶלָּא עַל גְּרוֹנוֹ. וְלֹא יִתְכַּבֵּד רַק עַל צַוְּרוֹנוֹ. כִּי הוּא בְשִׂכְלוֹ יֵדַע מַעֲלַת הַחָכְמָה וִיכַבְּדֶנָּה. וּמִנְּעוּרָיו הִיא לוֹ גְבֶרֶת וְתָמִיד יַעַבְדֶנָּה. כִּי בָהּ תִּגְדַּל מַעֲלָתוֹ. וְתִנַּשֵּׂא מַלְכוּתוֹ. וּבַסֵּפֶר הַזֶּה יַאֲמִינוּ בִשְׁבָחָיו כָּל שׁוֹמְעָיו. וְיוֹסִיפוּ בוֹ אַהֲבָה כָל יוֹדְעָיו. וְכַאֲשֶׁר יֵיטִיב אַחֲרִיתוֹ בּוֹרְאוֹ. וּבָנָיו יֵשְׁבוּ עַל כִּסְאוֹ. יִהְיֶה הַסֵּפֶר הַזֶּה לָהֶם. עֵד נֶאֱמָן עַל גְּדֻלַּת אֲבִיהֶם. כִּי זֶה הַסֵּפֶר יִפְקַח לַעִוְרִים עֵינָיִם. וְיַרְנִין לָאִלְּמִים שְׂפָתָיִם. לְשַׁבֵּחַ פְּעֻלּוֹתָיו. וּלְסַפֵּר תְּהִלּוֹתָיו. וְלָשִׂיחַ בְנִפְלְאוֹתָיו. וּשְׁנֵי אֶחָיו הַשַּׂר הַנִּכְבָּד רַבִּי יוֹסֵף וְהַשַּׂר הַנֶּחְמָד רַבִּי עֶזְרָא יְהִי בְעֶזְרָם רוֹכֵב שָׁמַיִם. שְׁנֵיהֶם לְבֵית הַמִּשְׂרָה עַמּוּדִים שְׁנַיִם. כְּאִלּוּ הַתּוֹרָה אָרוֹן וְהֵם שְׁנֵי כְרוּבִים פּוֹרְשֵׂי כְנָפַיִם. אוֹ הַמִּשְׂרָה צוּרָה וְהֵם לָהּ שְׁתֵּי עֵינַיִם. גַּם הֵם לָבְשׁוּ לְבוּשׁ הָעֲנָוָה. וּפָשְׁטוּ כְסוּת הַגַּאֲוָה. וְרָחֲקוּ מִכָּל דִּבָּה וּמַעֲוָה.

וְאָסְפוּ כָּל מִדָּה נָאוָה. עַד יָצְאוּ כְּכֶסֶף צָרוּף תַּאֲוָה לָעֵינַיִם. מְזֻקָּק שִׁבְעָתַיִם. הָאֵל יָאִיר נֵרָם. וְיַזְרִיחַ אוֹרָם. וּבַמָּקוֹם הַזֶּה אֶחְתֹּם דְּבָרַי. וְאֲסַיֵּם אֲמָרַי. וְהָאֵל יוֹדִיעֵנוּ אֹרַח חַיִּים. כִּי מִמֶּנּוּ תּוֹצְאוֹת חַיִּים:
