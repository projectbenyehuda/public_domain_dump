אמרו כי איש היה

אָדָם וּבֵיתו


– – – אָמְרוּ כִּי אִישׁ הָיָה בְּאֶרֶץ הַבָּשָׁן / וּשְׁמוֹ חַמְדָּן בֶּן דִּישָׁן / וְשֵׁם אִשְׁתּוֹ מֵי בְאֵל בַּת מֵי זָהָב / מֵאֶרֶץ הַחֲוִילָה אֲשֶׁר שָׁם הַזָּהָב / וּפִלַגְשׁוֹ וּשְׁמָהּ רְאוּמָה / בַּת אָהֳלִיבָמָה / וְהָיָה הָאִישׁ תָּם וְיָשָׁר יְרֵא ה׳ וְחָפֵץ בְּמִצְווֹתָיו / הַיּוֹרְדִים עַל פִּי מִדּוֹתָיו / רַק בִּדְבַר הַחֲפָצִים וְהַקִּנְיָנִים / זָהָב וְרֹב פְּנִינִים / דָבַָק נַפְשׁוֹ לְאַהֲבָה / יוֹם וָלַיְלָה לֹא יִשְׁבּוֹת מִן הַמַּדְהֵבָה / וַיִּוָּלְדוּ לוֹ שְׁלֹשִׁים בָּנִים וּשְׁלֹשִׁים בָּנוֹת / וַיִּגְדַּל הָאִישׁ מְאֹד וַיֶּעֱשַׁר מִכָּל הָאָדָם / וַיִּפְרֹץ מִכָּל בְּנֵי קֶדֶם / וַיַּעַשׂ לוֹ שִׁדָּה וְשִׁדּוֹת וְתַעֲנוּגוֹת בְּנֵי אָדָם / וְהָלְכוּ בָנָיו וְעָשׂוּ לָהֶם מִשְׁתֶּה אִישׁ יוֹמוֹ וְקָרְאוּ לִשְׁלֹשִׁים אַחְיוֹתֵיהֶם / לֶאֱכֹל וְלִשְׁתּוֹת עִמָּהֶם / וַיְהִי כִּי הִקִּיפוּ יְמֵי הַמִּשְׁתֶּה / וַיִּשְׁלַח חַמְדָּן וַיְקַדְּשֵׁם וְהִשְׁכִּים בַּבֹּקֶר / וְהֶעֱלָה עוֹלוֹת מִסְפַּר כֻּלָּם מִן הַצֹּאן וּמִן הַבָּקָר / וְעִם כָּל זֶה הָיָה כִילַי אוֹבֵר בְּמָמוֹנוֹ / כִּי יִתֵּן בַּכִּיס עֵינוֹ / מִלַּחְמוֹ לֹא נָתַן לַדָּל / וּמֵעָנִי וְאֶבְיוֹן מַתְּנַת יָדוֹ חָדַל / גַּם הָיָה חוֹמֵד לְהַעֲכִיר שְׁאֵרוֹ / וּלְהוֹנוֹת אֶת חֲבֵירוֹ / הֵן יַעְשֹׁק נְהָרוֹת וְלֹא יַחְפֹּז / וְאִם יִתֵּן לוֹ אִישׁ כָּל הוֹן בֵּיתוֹ בּוֹז יָבוֹז / כִּי אָמַר מָה אֶעֱשֶׂה לְבָנַי וְלִבְנוֹתַי / אֲשֶׁר רְבָבָה כְצֶמַח הַשָּׂדֶה נְתָנָם הָאֵל בְּבֵיתִי וּבְחוֹמוֹתַי / לַדָּבָר הַזֶּה יִסְלַח לִי אֱלֹהֵי קֶדֶם מְעוֹנָה / כִּי יִקַּח מִנְחָתִי מִיָּדִי אֲשֶׁר הֵבֵאתִי לוֹ בִתְחִנָּה / וַאֲקַדֵּם פָּנָיו בִּתְשׁוּרָה וַאֲרוּחָה / כִּי אָמַר אֲכַפְּרָה פָנָיו בַּמִּנְחָה.


וַיְהִי הַיּוֹם וְחַמְדָּן בְּבֵית בְּנוֹ הַבְּכוֹר / וְהוּא שׁוֹתֶה שִׁכּוֹר / וְאֵשׁ יָצְאָה מִלִּפְנֵי ה׳ וַתִּקָּחֵהוּ בְצִיצִית רֹאשׁוֹ / וַתִּקְצַר נַפְשׁוֹ / וְלֹא קָמָה בוֹ עוֹד רוּחוֹ / וְהִנֵּה הַצָּרַעַת זָרְחָה בְמִצְחוֹ / וַיִּפֹּל אַרְצָה מְלֹא קוֹמָתוֹ / וַיִּקְחוּ עֲבָדָיו אוֹתוֹ / וַיִּשָּׂאוּהוּ אֶל בֵּיתוֹ / וַיַּשְׁכִּיבוּ אוֹתוֹ בַחֲדַר מִטָּתוֹ / וַיִּגְדַּל עָלָיו הַכְּאֵב מְאֹד עַד כִּי חָשַׁךְ תָּאֳרוֹ / וְצָפַד עוֹרוֹ וַיֵּאָכֵל חֲצִי בְשָׂרוֹ.


וַיִּשְׁלַח וַיִּקְרָא אֶת כָּל אוֹהֲבָיו מִסָּבִיב עַד אֵלוֹן צַעֲנַנִּים / וְאֶת הָרוֹפְאִים אֲשֶׁר עַשְׁתְּרוֹת קַרְנַיִם / וַיְחַל פְּנֵיהֶם אוּלַי יוּכְלוּ לְכַלְכֵּל מַחֲלֵהוּ / וְעַל עֶרֶש דְּוַי יִסְעָדוּהוּ / וִיחַלְּצוּהוּ מֵחֶבְלוֹ / וְשָׁב וְרָפָא לוֹ / וַיִּפְנוּ הָאֲנָשִׁים לְהַבִּיט אֶל מַכַּת מַכֵּהוּ / הֵמָּה רָאוּ כֵן תָּמָהוּ / עַד כִּי נוֹסְדוּ יָחַד / וַיַּעֲנוּ כֻלָּם פֶּה אֶחָד: ׳עֲצָתְךָ לֹא תָקוּם וְלֹא תִהְיֶה / וְעַתָּה צַו לְבֵיתְךָ כִּי מֵת אַתָּה וְלֹא תִחְיֶה׳ / וַיִּשְׁמְעוּ אִשְׁתּוֹ וְרֹב בָּנָיו כִּי אֶפֶס עָצוּר וְעוֹזֵר / וַיִּשְׂאוּ אֶת קוֹלָם וַיִּבְכּוּ בִּבְכִי יַעְזֵר / וַיַּעֲנוּ וַיֹּאמְרוּ: ׳יָשִׂים נַפְשֵׁנוּ תַּחַת נַפְשְׁךָ, צוּרִי וּמִשְׂגַּבִּי / מִי יִתֵּן מוּתֵנוּ תַּחְתְּךָ אָבִי אָבִי / אָנָה בְמוֹתְךָ נֵלֵךְ וְאָנָה נִפְנֶה / עַל מִי נָטַשְׁתָּ מְעַט הַצֹּאן הָהֵנָה / וְאֵיפֹה נוֹלִיךְ אֶת חֶרְפָּתֵנוּ / נִשְׁכְּבָה בְּבָשְׁתֵּנוּ וְנָלִינָה בְחֶרְפָּתֵנוּ׳ / וַיִּזְעֲקוּ מָרָה בְּצִירָם וְחִילָם / וַתִּבָּקַע הָאָרֶץ לְקוֹלָם.


וַיְהִי עַד כֹּה וְעַד כֹּה וַיִּקֶר מִקְרֵהוּ / וַיָּבֹא הֵלֶךְ אֶל מַחֲנֵהוּ / אִישׁ מֵאֶרֶץ הַתֵּימָן / צַדִּיק וְנֶאֱמָן / אָב הָרַחֲמָן / וּשְׁמוֹ אֵיתָן הָאֶזְרָחִי / בֶּן בִּלְדַּד הַשּׁוּחִי / וַיִּוָּעֵד לִרְאוֹתוֹ / וּלְשַׁחֵר אוֹדוֹתוֹ / וַיִּשְׂמַח חַמְדָּן לִקְרָאתוֹ / וַיֹּאמֶר אֵלָיו: ׳מֵאַיִן תָּבוֹא אֶזְרָחִי / הֲשָׁלוֹם אַתָּה אָחִי / וּמַה מַּעֲשֶׂיךָ וּפָעֳלֶיךָ / לְמִי אַתָּה וְאָנָה תֵלֵךְ?׳ / וַיֹּאמַר: ׳אֲנִי רוֹפֵא אָמָּן / וְצִיר נֶאֱמָן / לְרַפּאוֹת צָרַעַת נַעֲמָן / וְכָל מַכַּת הַשְּׁחִין לַח וְיָבֵשׁ / אֶהְיֶה חוֹבֵשׁ / וְרַק אֲנִי לְבַדִּי נוֹתַרְתִּי מִיֶּתֶר הָרוֹפְאִים אֲשֶׁר בְּגִלְעָד / הַסְּמוּכִים לָעַד / וּבְלֹא בֶצַע הַצָּרַעַת מִמְּצוֹרָע אֶאֱסֹף / חִנָּם אֵין כָּסֶף / אֶחֱבֹשׁ אֶת חֵילָם / וְזֶה שְׁמִי לְעוֹלָם.


וַיְהִי כִשְׁמֹעַ חַמְדָּן אֶת דְּבָרָיו / וַיַּחֲזֵק בִּכְנַף מְעִילוֹ / בָּכָה וְהִתְחַנֵּן לוֹ / וַיֹּאמַר: ׳אַל נָא תַעֲבֹר מֵעַל עַבְדֶּךָ / כִּי לָמָּה אָמוּת נֶגְדֶּךָ / יָבוֹאוּ נָא דְּבָרֶיךָ לְהוֹשִׁיעֵנִי / יְחַיֵּנִי מִיּוֹמָיִם, בַּיּוֹם הַשְּׁלִישִׁי יְקִימֵנִי / וְאִם אֶת ה׳ אֱלֹהֶיךָ הֶאֱמַרְתָּ / כֵּן תַּעֲשֶׂה כַּאֲשֶׁר דִּבַּרְתָּ׳ / וְאִשְׁתּוֹ וּבְנֵי בֵיתוֹ יָצְאוּ נִצָּבִים עָלָיו / וְכָל אוֹהֲבָיו מִסָּבִיב שָׁתוּ עָלָיו / וַיִּפְּלוּ לְרַגְלָיו / לְחַלּוֹת אֶת פָּנָיו / וַיֹּאמְרוּ: ׳אָנָּא, עֶבֶד צוּר יִשְׂרָאֵל וְגוֹאֲלוֹ / רְפָא נָא לוֹ׳ / וַיּוֹסֶף חַמְדָּן שְׂאֵת מְשָׁלוֹ / וַיִּתֵּן בִּבְכִי קוֹלוֹ: / ׳לֹא עַל נַפְשִׁי אֶשָּׂא קִינָה אַף לֹא עַל עֲלוּמֵינוּ / יָשַׁבְנוּ גַּם בָּכִינוּ / אַךְ עַל זֶה דָּוֶה לִבֵּנוּ / וְעַל אֵלֶּה חָשְׁכוּ עֵינֵינוּ / עַל מַרְבִּית בֵּיתִי / אֲשֶׁר טִפַּחְתִּי וְרִבִּיתִי / מִנִּי רֶחֶם נְשׂוּאִים / אֵיפוֹא יִהְיוּ רוֹעִים / וְאִשְׁתִּי הָעֲנֻגָּה וְרַכָּה / אֲשֶׁר לֹא נִסְּתָה לְהַצִּיג עַל הָאָרֶץ כַּף רַגְלָהּ / מִי יָקוּם לָהּ / אֵיכָה תֵשֵׁב בְּדוּדָה / שְׁכוּלָה וְגַלְמוּדָה / לָכֵן כְּאֶפְעֶה אֶשְׁאַף וְאֶבֹּלָה / וְאֵרֵד אָבֵל שְׁאוֹלָה׳.


וַיַּעַן אֵיתָן הָאֶזְרָחִי: ׳אֱלֹהֵי קֶדֶם מְעוֹנָה / יִהְיֶה לִי לְמַשְׁעֵן וּמַשְׁעֵנָה / אַחַת דִּבַּרְתִּי וְלֹא אֶשְׁנֶה / וְאַתָּה הִתְיַצֵּב לִתְשׁוּעַת ה׳ וְאוֹתוֹתָיו חֲזֵה / לְמָחָר יִהְיֶה הָאוֹת הַזֶּה / וְאֵין לִי אִתְּךָ לֹא כֶסֶף וְזָהָב / וְלֹא אֶפְנֶה אֶל שׁוּם רַהַב / אַךְ אֶשְׁאַל מִמְּךָ שְׁאֵלָה קְטַנָּה / אוֹתָהּ אֲבַקֵּשׁ כִּי תִּתֵּנָה / וּבָהּ עַד מְהֵרָה תִמְצָא אֲרוּכָה לְחוֹלָךְ / אֲשֶׁר עוֹלֵל לָךְ׳ / וַיַּעַן חַמְדָּן: ׳הִנֵּה דָּבָר גָּדוֹל דִּבֶּר אֵלַי אֶעֱשֶׂה / וְכָל אֲשֶׁר תֹּאמַר אֶתֵּן / אַף כִּי דָּבָר קָטָן׳.


וַיַּעַן אֵיתָן: ׳זֶה הַדָּבָר אֲשֶׁר תַּעֲשֶׂה וְיִקְחוּ אֵלֶיךָ / מִמְּתֵי אֹהָלֶיךָ / אִשְׁתְּךָ וּבְנֵי בֵיתֶךָ / הַמְרַנְּנִים יְשׁוּעָתֶךָ / מִן הַדָּם אֲשֶׁר בְּאֶצְבַּע קְטַנָּה שֶׁבְּיָדָם / וְהֶחָבִיב חָבִיב קוֹדֵם / וְאֶת הַדָּם תַּמְצִיא כְמַר מִדְּלִי / וְנָתֹן אוֹתוֹ בְתוֹךְ מַיִם חַיִּים אֶל כֶּלִי / וּרְחַץ בְּשָׂרְךָ בַמַּיִם שֶׁבַע פְּעָמִים וְאַחַר תִּטְהָר / כִּבְרוֹשׁ וְתִדְהָר / אַךְ לֹא בְחֶרֶב וְלֹא בַחֲנִית / וְלֹא בְשׁוּם דָּבָר מַמְאִיר אֶת הַדָּם תִּמְצֶה / כִּי לֹא יֵרָצֶה / וְלֹא יַצְלִיחַ לִמְלֶאכֶת הָרְפוּאוֹת / וְלֹא יֵחָשֵׁב לְמוֹשָׁעוֹת / כִּי אִם צְלִי אֵשׁ וַחֲשַׁשׁ לֶהָבָה / וְשָׁלְחוּ אִישׁ אֶת אֶצְבָּעוֹ בָהּ / עַד כִּי יֵעָשֶׂה הַבָּשָׂר צַלֶּקֶת אֲדַמְדָּם / וְגִידָיו יְעַלְעוּ דָם / זֹאת חֲקַרְנוּהוּ שְׁמַעֲנוּהוּ / מִמַּיִם רַבִּים מְשִׁיתִיהוּ / וּמִמְּצוּלוֹת חַכְמֵי הַטֶּבַע דְּלִיתִיהוּ / וְאִם אֶת הַדָּבָר הַזֶּה תַּעֲשֶׂה מַחֲלָתְךָ אֲכַלְכְּלֶנָּה / אָנֹכִי אֶעֶרְבֶנָּה / מִיָּדִי תְּבַקְשֶׁנָּה / וְאַתָּה דַע לָךְ׳.


וַיְהִי כְּכַלּוֹתוֹ לְדַבֵּר וַיְמַהֵר חַמְדָּן וַיִּקְרָא אֶת נָשָׁיו / וְאֶת בָּנָיו / וַיָּבֹאוּ כֻלָּם וַיִּשְׁתַּחֲווּ אַפַּיִם לְפָנָיו / וַיְסַפֵּר לָהֶם אֶת כָּל הַדְּבָרִים הָאֵלֶּה וְאֵיךְ מָצָא רִפְאוּת תְּעָלָה / וַיַּעַן וַיֹּאמַר: ׳אַתֶּם רְאִיתֶם אֶת אֲשֶׁר עָשִׂיתִי לָכֶם מֵעוֹדִי / וְנָתַתִּי עֲלֵיכֶם מֵהוֹדִי / וּפָרַשְׂתִּי עֲלֵיכֶם עֲנַן כְּבוֹדִי / וָאֶשָּׂא אֶתְכֶם עַל כַּנְפֵי נְשָׁרִים / וָאַנְוֶה לְחָיֵיכֶם בַּתּוֹרִים / וָאַלְבִּישׁ אֶתְכֶם שְׁנִי עֲדָנִים / וּמִלֵּאתִי אֶת יְדֵיכֶם מַטְמוֹנִים / וְעַתָּה בַּצַּר לִי אָנָּא שַׁחֲדוּ בַעֲדִי כְּבוֹדְכֶם / מִסַּת נִדְבַת יְדֵיכֶם / פְּדָעוּנִי מֵרֶדֶת שַׁחַת / וּמִפְּנִי שְׁמִי נִחַת׳ / וַיַּעֲנוּ כֻלָּם בְּדֶמַע: ׳אֶת כָּל אֲשֶׁר דִּבֶּר אֲדֹנִי נַעֲשֶׂה וְנִשְׁמַע / הִנְנוּ נְכוֹנִים לִהְיוֹת קְרוּצִים / נְתָחִים עֲרוּכִים עַל הָאֵשׁ וְהָעֵצִים / כָּלִיל נִהְיֶה אַזְכָּרָה לְרוּחַ אַפֵּינוּ / עוֹלָה אִשֵּׁה רֵיחַ נִיחוֹחַ לַאדֹנֵנוּ׳ / וַיֹּאמֶר אֵיתָן אֶל אִשְׁתּוֹ: ׳גְּשִׁי הֵנָּה / וְשִׁלְחִי אֶצְבַּע קְטַנָּה׳ / וַתִּתֵּן אֶת יָדָהּ, וַיִּקַּח בְּמֶלְקָחַיִם רִצְפָּה / וַיַּדְבֵּק אֶל כַּפָּהּ / וַיְהִי טֶרֶם נָשְׁקָה אֶת בְּשָׂרָהּ / לֶאֱכֹל אֶת עוֹרָהּ / עָלָה עָשָׁן בְּאַפָּהּ / כִּי חָרָה לָהּ / וַתִּשָּׂא אֶת קוֹלָהּ / כְּדַבְּרָהּ רְתֵת נָשְׂאָה וַתֶּאֱשַׁם בְּבַעְלָהּ / וַתֹּאמֶר: ׳הַלְמַעֲנוֹ אֶסְבּוֹל צַעֲרִי / לְהַכְאִיב אֶת בְּשָׂרִי / וְכִי יִרְאֶה שַׁחַת מָה אָחוּשׁ / הַאִם כֹּחַ אֲבָנִים כֹּחִי וּבְשָׂרִי כָּחוּשׁ / הַעוֹד אִישׁ אֵין בָּאָרֶץ / לָבוֹא עָלַי כְּדֶרֶךְ כָּל הָאָרֶץ / וְכִי תַמּוּ נִכְרָתוּ רֵעִים אֲהוּבִים / בַּחוּרֵי עֲגָבִים / וַתַּלְבֵּשׁ צַוָּארָהּ רַעְמָה / וַתֵּפֶן וַתֵּלֶךְ בְּחֵמָה / וַיִּקְרָא אֶת בְּנוֹ הַגָּדוֹל הַנָּצוּר לוֹ כְאִישׁוֹן / וַיַּעַשׂ לוֹ כַּמִּשְׁפָּט הָרִאשׁוֹן / וְהָיָה כִּי יֵחַם לוֹ וַיִּקְפֹּץ יָדוֹ אֵלָיו / וַיַּעֲבֹר עַל פָּנָיו / וַיִּשָּׂא אֶת רַגְלָיו בִמְרוּצָה / וַיָּנָס וַיֵּצֵא הַחוּצָה / וְהַנִּשְׁאָרִים נָסוּ לְקוֹלוֹ, וַיָּבֹא חֶדֶר בְּחֶדֶר לְהִמָּלֵט / בְּמַלְבֵּן וּבְמֶלֶט / וַיִּשָּׁאֵר אַךְ חַמְדָּן לְבַדּוֹ / וְעֵינָיו רָאוּ כִידוֹ / וַיִּתְהַפֵּךְ בְּצִירָיו וַחֲבָלָיו / וְאֵימַת מָוֶת נָפְלָה עָלָיו.


וַיַּעַן אֵיתָן הָאֶזְרָחִי: ׳הַהֵיטֵב חָרָה לְךָ אָחִי?׳ / וַיֹּאמַר: ׳הֵיטֵב חָרָה לִי עַד מָוֶת וְאֵיכָכָה אוּכַל וְאֶרְאֶה יְלָדַי / וּמַעֲשֵׂי יָדַי / בְּקִרְבָּם יַעֲרִיצוּ אָרְבָּם / הָפַךְ לִשְׂנֹא אוֹתִי לִבָּם / רוּחִי זָרָה לְאִשְׁתִּי / וְחַנּוֹתִי לִבְנֵי בִטְנִי / וַיֹּאמֶר אֶחָד מִן הַנְּעָרִים הַנִּצָּבִים עָלָיו: ׳שָׁמַעְתִּי אוֹמְרִים הַעוֹד לָנוּ / חֵלֶק וְנַחֲלָה בְּבֵית אָבִינוּ / לְכַבְּדוֹ וּלְבָרֵךְ בִּשְׁמוֹ / הֲלֹא צָבָא לֶאֱנוֹשׁ עַל הָאָרֶץ וּכְשָׂכִיר יִרְצֶה יוֹמוֹ / הֵן מָלְאוּ יָמָיו / וְכָלוּ רַחֲמָיו / גַּם אַהֲבָתוֹ / גַּם שִׂנְאָתוֹ / כִּי לֹא יִקַּח הַכֹּל בְּמוֹתוֹ / הוּא יֵלֵךְ לְמִיתָה וְיָנוּחַ עַל מִשְׁכָּבוֹ בְשָׁלוֹם / וַאֲנַחְנוּ נִשָּׁאֵר הֲלוֹם / כֻּלָּנוּ חַיִּים / עַד לֹא שָׁמַיִם׳ / וַיִּשְׁמַע חַמְדָּן וַיָּמָת לִבּוֹ בְּקִרְבּוֹ / וְהוּא הָיָה לְאֶבֶן וְכָל רוּחַ אֵין בּוֹ. – – –
