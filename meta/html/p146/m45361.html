אָהַב לִבִּי

חתימה: א״ב אלשבזי


אָהַב לִבִּי לָבוֹא / בְּיוֹשֶׁר אָרְחוֹת נְגוֹהִים

וְאֶתְבּוֹנֵן בְּשׁוּבוֹ / לְעוֹרֵר כּוֹחוֹת שׁוֹהִים

וְאֵיךְ חוֹטֵא יֵחָבֵא / וְיֵשׁ שׁוֹמֵר וּגְבוֹהִים

אֵימָתַי אָבוֹא / וְאֵרָאֶה פְנֵי אֱלֹהִים:


בַּדִּין הַכֹּל עוֹמְדִים / וְכָל מַעֲשֵׂיהֶם נִשְׁקָל

וְקַצְווֹת אֶרֶץ רוֹעֲדִים / וְאֵין שָׁם מִשְׁפָּט מְעֻקָּל

וְהָרְשָׁעִים הַמּוֹרְדִים / יִהְיֶה דָּתָם מְקֻלְקָל

וּפְעָמַי מִתְנוֹדְדִים / צָמְאָה נַפְשִׁי לֵאלֹהִים:


גַּל עֵינַי יָהּ וְאֶחֱזֶה / בְּתוֹרָתְךָ נִפְלָאוֹת

וְעַל בֶּן אֲמָתְךָ תַּזֶּה / מֵי טוֹהַר יִהְיֶה אוֹת

כִּי בְּהֶבְלוֹ נִתְבַּזֶּה / בִּתְכוּנוֹת נִטְמָאוֹת

גַּם אֶת זֶה לְעֻמַּת זֶה / עָשָׂה הָאֱלֹהִים:


דְּרָכַי בְּחֵן יִתְיַשְּׁרוּ / בְּעֶזְרַת שׁוֹכֵן עֲרָבוֹת

וְאִם צְעָדַי נִתְאַחֲרוּ / בֵּין בְּנֵי עַוְלָה וְרִיבוֹת

וִימֵי חַיַּי נִקְשְׁרוּ / בְּחֶסֶד אֱלֹהַי סְבִיבוֹת

וּבֵין שְׂרִידִים שֶׁנִּבְחֲרוּ / חָנֵּנִי אֱלֹהִים:


הַשְׁלֵךְ יְהָבְךָ בֶּן אִישׁ / עַל הַזָּן לְכָל בִּרְיָה

כִּי הוּא מַעְשִׁיר וּמוֹרִישׁ / וְרוּחוֹ לַכֹּל מְחַיָּה

וְגוּפוֹת עֲרֻמִּים מַלְבִּישׁ / וְעוֹנֶה עֲשׁוּקִים בַּשִּׁבְיָה

וְלָדַעַת אֲנִי מַחֲרִישׁ / יִשְׁעִי עַל אֱלֹהִים:


וַעַד טוֹב וְחֵן יַמְצִיא / לַעֲדַת קוֹדֶשׁ צוֹפִים

וּמִבֵּית כְּלָאִים יוֹצִיא / עֲנִיִּים הַנִּרְדָּפִים

לְמוֹעֵד מוֹעֲדִים וָחֵצִי / יְחַיֶּה רוּחוֹת עֲיֵפִים

מֵעָפָר הִתְנַפְּצִי / מַחֲנֵה אֱלֹהִים:


זַרְזִיף אֶרֶץ תֶּהֱמֶה / לְקַבֵּל שֶׁפַע טוֹבָה

וּשְׁנֵי יְצָרִים תַּגְמֶה / לְרַפֵּא נֶפֶשׁ עֲצֵבָה

וְשֵׂכֶל כְּמוֹ אוֹר דּוֹמֶה / וְאֶל הַשְּׁלֹשָׁה לֹא בָא

לִהְיוֹת בְחֶטְאָם הוֹמֶה / כִּי יְרֵא אֱלֹהִים:


חַשְׁרַת מֵימֵי חֶסֶד / מְכַבֶּה שְׁבִיב לֶהָבוֹת

מְקָרֵר לְאַבְנֵי מוּסַּד / זְבוּל אֶמְצָעִי סְבִיבוֹת

אֶבֶן שְׁתִיָּה יָסַד / בְּצִיּוֹן בִּזְכוּת אָבוֹת

וְכָל הֹוֶה וְכָל נִפְסָד / מִפְעֲלוֹת אֱלֹהִים:


טַעֲמוּ וּרְאוּ בְּבִינָה / כִּי טוֹב יְוֹיָ וְנֶאֱמָן

לְשַׁלֵּם לְנֶפֶשׁ נְכוֹנָה / שָׂכָר לָעֲתִיד מְזֻמָּן

יוֹם תַּקְבִּיל אוֹר שְׁכִינָה / וּשְׁחָקִים שׁוֹחֲקִים מָן

וְאִישׁ צַדִּיק יָלִינָה / וְיֵרָאֶה פְנֵי אֱלֹהִים:


יִזְכּוּ נְפָשׁוֹת נְקִיּוֹת / לַחֲזוֹת גִּנַּת בִּיתָן

רֹאשׁ גַּלְגַּלִּים וְחַיּוֹת / יֵטִיבוּ בְאַחֲרִיתָן

כִּי שָׁם סוֹד הַהֲוִיּוֹת / חֲקוּקִים בִּמְחִצָּתָן

בַּעֲלֵי הַזְּכִיּוֹת / יִרְאֵי אֱלֹהִים:


כּוֹבֵשׁ עֲווֹנוֹת סְלַח נָא / לַעֲווֹן עֲנִיֵּי עַמָּךְ

כִּי מֵחַסְדְּךָ נִהְנָה / הַכֹּל עָשִׁיר וְהַמָּךְ

וְעוֹלָם בְּחֶסֶד נִבְנָה / בְּיַד יְמִינֶךָ נִסְמָךְ

יְוֹיָ עֲשֵׂה לְמַעַן שְׁמָךְ / כְּחַסְדֶּךָ אֱלֹהִים:


לִתְשׁוּעַת אֵל חַכּוּ / זֶרַע יְשׁוּרוּן וְגִילוּ

אֲבָל רַחֲצוּ הִזַּכּוּ / וְכוֹחוֹתֵיכֶם הָחִילוּ

בָּמֳתֵי תִקְוָה דִּרְכוּ / וַעֲצַת שְׂמֹאל הַזִּילוּ

בְּמַקְהֵלוֹת / בָּרְכוּ אֱלֹהִים:


מַה טּוֹב תְּשׁוּבַת מוֹדֶה / עַל עֲווֹנוֹ וּזְדוֹנוֹ

נַפְשׁוֹ מִשְּׁאוֹל פּוֹדֶה / וּמֵסִיר הִרְהוּר יְגוֹנוֹ

וַעֲדִי טָהוֹר יַעֲדֶה / לַעֲשׂוֹת אֶת רְצוֹן קוֹנוֹ

וּמִתַּחְבּוּלוֹת רוֹדֶה / טוֹב לְיִשְׂרָאֵל אֱלֹהִים:


נֶפֶשׁ צַדִּיק תַּעֲלֶה / בְּאַהֲבָה לְמַעְיַן גַּנִּים

וְגּוּף מִיָּגוֹן תִּדְלֶה / וְכָל מַחְשְׁבוֹתָיו תַּשְׁלִים

זַכֵּה יָהּ דָּל נִקְלֶה / אַתָּה מָעוֹז לְדַלִּים

רַפֵּא לְלֵבָב חוֹלֶה / לְהִתְהַלֵּךְ לִפְנֵי אֱלֹהִים:


סְגֻלַּת נְדִיבִים הוֹמִים / עֲנִיִּים לְבַקֵּשׁ פְּרוּסָה

זֶרַע קוֹדֶשׁ שְׁלֵמִים / לְמִי שֶׁאָמַר וְעָשָׂה

וּבְיֵחוּדוֹ מְרוֹמְמִים / וְנַפְשָׁם בִּשְׁמוֹ חוֹסָה

הוֹדַעְתָּ בָעַמִּים / עֻזְּךָ אֱלֹהִים:


עוּרוּ חֶבְרַת קְדוֹשִׁים / לָבוֹא בֵית אֱלֹהֵינוּ

וּנְעָרִים וִישִׁישִׁים / מִשְׁפָּט לַבֹּקֶר דִּינוּ

וְאָז נֵחָלֵץ חוּשִׁים / וְנִקְרָא לְעֶזְרָתֵנוּ

רָם עַל גַּלְגַּל וְאִישִׁים / יְוֹיָ אֱלֹהִים:


פִּתְחֵי רַחֲמִים פְּתוּחִים / לְקַבֵּל תְּשׁוּבָה גְּמוּרָה

וּמִשְׁקְלֵי צֶדֶק נְכוֹחִים / לִשְׁקוֹל זְכוּת וַעֲבֵרָה

וְרָשָׁע עָלָיו שְׁלוּחִים / וְהוּא נִדּוֹן בַּחֲמוּרָה

וּמַלְאֲכֵי דִין מוֹכִיחִים / נוֹכַח פְּנֵי אֱלֹהִים:


צְדָקָה עֲשֵׂה עִמָּנוּ / נוֹרָא עַל כָּל מְלָכִים

הֱיֵה עֶזְרָתָה לָנוּ / לְהַצְלִיחַ כָּל דְּרָכִים

כִּי כָּל פְּעֻלּוֹתֵינוּ / לְפָנֶיךָ נֶעֱרָכִים

בִּזְכוּת תּוֹרָתֵנוּ / יִתְעַשֵּׁת הָאֱלֹהִים:


קוֹרְאֵי דִּבְרֵי תְּחִנּוֹת / יַעֲנֶה נוֹרָא תְהִלּוֹת

וּנְפָשׁוֹת מִתְעַנּוֹת / יְטַהֵר מִכָּל מְעִילוֹת

בְּחֶסֶד הָאֵל אֲפוּנוֹת / נַפְשׁוֹת בְּנֵי תָם סְגוּלוֹת

וּבְיַד צוּרָם אֱמוּנוֹת / לִפְנֵי הָאֱלֹהִים:


רַצִּי עָלַי יְחִידָה / אֵל חַי מִכָּל אֲשָׁמִים

וּבְשָׁעַת הַפְּרִידָה / נַקִּי זֻהְמַת גְּלָמִים

הַבִּיעִי שִׁיר וְחִידָה / לְהוֹדוֹת צוּר עוֹלָמִים

כִּי בַעֲלוֹתֵךְ עֲתִידָה / לְמִזְבַּח הָאֱלֹהִים:


שֶׁפַע נְדָבוֹת יוֹרִיד / רַחוּם וְחַנּוּן בְּרָצוֹן

עַל בֵּן אֲשֶׁר הוּא יָדִיד / וְאָנוּס בְּיַד בְּנֵי לָצוֹן

וְאָז יִשְׁכַּב וְאֵין מַחֲרִיד / וְאוֹיֵב נִדְחַף חִיצוֹן

וּפִי בִּתְהִלּוֹת יַגִּיד / הַרְנִינוּ לֵאלֹהִים:


תְּשׁוּעַת יְוֹיָ תָּבוֹא / לְקַבֵּץ עֲנִיֵּי עַמּוֹ

וְיָשׁוּב בֵּן לְמוֹשָׁבוֹ / וְיִפְרוֹשׂ סֻכַּת שְׁלוֹמוֹ

וְיַשְׁפִּיעַ עֲלֵיהֶם בְּטוּבוֹ / וִיקַנֵּא לְמַעַן שְׁמוֹ

וְהַחַי יָשִׁיב אֶל לִבּוֹ / מִפְעֲלוֹת אֱלֹהִים:


אִמְרוֹת יְוֹיָ טְהוֹרוֹת / בְּסוֹד חֻקִּים וּמִשְׁפָּטִים

עֵין יִשְׂרָאֵל מְאִירוֹת / מֵחֹשֶׁךְ וַעֲלָטִים

מְיֻסָּדִים לַדּוֹרוֹת / בִּכְלָלִים וּפְרָטִים

וְנַפְשׁוֹת צַדִּיקִים צְרוּרוֹת / בְּיֶשַׁע אֱלֹהִים:


לְהֵיכַל קוֹדֶשׁ זְמוּנִים / זֶרַע יַעֲקֹב בְּשׁוּבָם

רוּחַ וְנֶפֶשׁ אֱמוּנִים / כִּי שָׁם נְאוֹתָם וְטוּבָם

וְאֶעֱרוֹךְ תַּחֲנוּנִים / לְזַכּוֹת נַפְשִׁי סְבִיבָם

כִּי יְוֹיָ בָּחַר בָּם / אֶל בֵּית אֱלֹהִים:


שׁוֹמְרֵי מִצְווֹת זוֹכִים / אֶל הַקּוֹדֶשׁ פְּנִימָה

רְצוֹן קוֹנֵיהֶם חוֹכִים / לְעֵדֶן פְּאַת קָדִימָה

וְגוּפוֹתָן הַנְּמוֹכִים / לָהֶם חֵן יָשִׂימָה

כִּי בַחֲיוֹתָם סְמוּכִים / רֶכֶב אֱלֹהִים:


בְּמַעְגְּלֵי צֶדֶק יַנְחֵם / לָשׁוּב לְנַחֲלַת אָבוֹתָם

כִּי שָׁם מְחוֹז מִבְטָחֵם / יָבִינוּ לְאַחֲרִיתָם

רוּחַ אֱלֹהִים תִּקָּחֵם / לִזְכּוֹר לָהֶם בְּרִיתָם

כִּי בְסוֹד הַקּוֹר וְהַחוֹם / נָחָם הָאֱלֹהִים:


זוֹכוֹת נְפָשׁוֹת יְקָרוֹת / תִּהְיֶה לָהֶן רְוָחָה

בְּשֵׁשׁ כְּנָפַיִם מְאִירוֹת / לַעֲלוֹת בְּעֵת הַצְלָחָה

אֲבַג יְתַץ כַּנְפֵי צוּרוֹת / בָּהֶן נֶפֶשׁ תִּפְרְחָה

עַל מִי חֶסֶד נוֹהֲרוֹת / עַד הַר הָאֱלֹהִים:


יְזַכֶּה צוּרִי לְנַפְשִׁי / בֵּין עֲנִיֵּי קְהַל עֲדָתוֹ

וְיִמְחוֹל חֲטָאַי בְּדָרְשִׁי / לִרְאוֹת הֲדַר שְׁכִינָתוֹ

יְחַזֵּק רִפְיִי וְחָלְשִׁי / בִּזְכוּת חֵן תּוֹרָתוֹ

וְיָאִיר מִקָּו שִׁשִּׁי / בְּרֵאשִׁית בָּרָא אֱלֹהִים:
