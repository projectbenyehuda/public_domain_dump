שָׁמַעְתִּי דִּבְרֵי נְבוּאוֹת

רצון הלוי – שירת ישראל בתימן


חתימה: שלם יוסף


שָׁמַעְתִּי דִּבְרֵי נְבוּאוֹת / מִפִּי חֶזְיוֹן נְבִיאֵנוּ

נָשְׂאוּ עָלַי מַשָּׂאוֹת / כִּי יֵשׁ קֵץ לְגָלוּתֵינוּ

רָאוּ בְּחֶזְיוֹנֵי מַרְאוֹת / וְשָׁמְעָה שֶׁמֶץ אָזְנֵינוּ

וְדִבְרֵיהֶם לִי לְעֵד וּלְאוֹת / מְחַכִּים לִישׁוּעָתֵנוּ

יַרְעֵם אֵל בְּקוֹלוֹ נִפְלָאוֹת / אֱלֹהֵי קֶדֶם מַלְכֵּנוּ

בְּעִיר אֱלֹהִים צְבָאוֹת / בְּעִיר אֱלֹהֵינוּ:


לָמָּה זֶה תֶּהֱמִי נַפְשִׁי / הָעִירִי יְשֵׁנֵי חֶבְרוֹן /

דִּגְלֵי שְׁבָטַיִךְ הַמְשִׁי / וּלְכִי בְּיוֹשֶׁר וּבְכִשְׁרוֹן

וְעֵת רָצוֹן צוּרֵךְ דִּרְשִׁי / כִּי יֵשׁ לַעֲדָתֵךְ זִכְרוֹן

קוּמִי עֲלִי אֶל הַר קָדְשִׁי / וְנִמְצָא שָׁם חֵלֶק יִתְרוֹן

וּמֹשֶׁה וּשְׁמוּאֵל וְאַהֲרֹן / יַחְדָּיו יֵלְכוּ לְפָנֵינוּ:


מַלְכֵי עֲרָב חָבְרוּ עָלַי / וּמִי זֶה יָקוּם לְעֶזְרָתִי

הִזְכִּירוּנִי מַעֲלָלַי / לוּלֵי אֱלֹהֵי יְשׁוּעָתִי

עָמַד וְקִנֵּא בִּגְלָלַי / כִּי בִּקְּשׁוּ לְהָמִיר דָּתִי

אוֹדֶה צוּרִי בְּמַהֲלָלַי / חֶלְקִי יְסוֹד חֵשֶׁק מְנָתִי

הָאוֹמֵר שׁוּבוּ אֵלַי / יָשׁוּב יְרַחֲמֵנוּ:


יְחַפֵּשׂ אָדָם דְּרָכָיו / וְדַלְתֵי תְּשׁוּבָה נִפְתָּחִים

וְיָכִינָה כָּל מַהֲלָכָיו / אָז יֵלֵךְ לוֹ בָּרְוָחִים

וְיִשְׁלַח צוּרִי מַלְאָכָיו / לְרַפֵּא לְבָבוֹת נֶאֱנָחִים

וְיָדִין אוֹיְבִים בְּמַעֲרָכָיו / כִּי הֵפֵרוּ בְּרִית אַחִים

מִמִּגְדַל עֹז צוּרִי לְחוֹכָיו / עֶזְרֵנוּ וּמָגִנֵּנוּ:


וְאִם טָבַעְתִּי בִּמְצוּלוֹת / אֶקְרָא לֶאֱלֹהַי יִשְׁעִי

מַשְׁוֶה רַגְלַי כָּאַיָּלוֹת / כִּי הוּא מְצוּדָתִי וְסַלְעִי

חַסְדּוֹ יָחִישׁ לְהִגָּלוֹת / וְקָרוֹב לִשְׁמֹעַ שַׁוְעִי

וּבְּיוֹם שְׁבָטַי נִדְגָּלוֹת / מִתּוֹךְ יָם דְּלֵה טָבְעִי

בִּמְשִׁיחוֹ יְנַהֵל עָלוֹת / כְּקֶדֶם יְחַדֵּשׁ יָמֵינוּ:


סְלִיחָה בְּנֵי אִישׁ תָּם בַּקְּשׁוּ / אוּלַי הֶעָווֹן יְכֻפַּר

יוֹם זֶה אֶת יְוֹיָ דִּרְשׁוּ / כִּי בְּרִית אָבוֹת לֹא תוּפָר

יוֹמָם וָלַיְלָה הִתְקוֹשְׁשׁוּ / הִתְנַעֲרוּ מִגּוּשׁ עָפָר

רוּחֲכֶם בְּמֵי חֶסֶד טִמְשׁוּ / קוֹל תְּקִיעַת שׁוֹפָר

יִרְאוּ גוֹיִם וְיֵבוֹשׁוּ / כִּי בוֹ יִשְׂמַח לִבֵּנוּ:


פִּתְאֹם יָבוֹא עֵת יְשׁוּעָה / וְנַעֲלֶה בְּרִנָּה הַר צִיּוֹן

נִשְׁמַע קוֹל שׁוֹפָר תְּרוּעָה / וּמֶלֶךְ יוֹשֵׁב בְּאַפִּרְיוֹן

בְּמִשְׁפָּט וּצְדָקָה זְרוּעָה / בְּמִגְדַּל עוֹז וְגֵיא חִזָּיוֹן

וְשָׁם מַעְתִּיקֵי הַשְּׁמוּעָה / הַיּוֹדְעִים דַּעַת עֶלְיוֹן

כִּי לַיוֹיָ הַתְּשׁוּעָה / וְהוּא יָבוּס קָמֵינוּ:
