ספר תחכמוני - שַׁעַר עֶשְׂרִים וַחֲמִשָׁה

כַּצַּיָּד הַגִּבּוֹר וְהֶחָזָק. רָצוֹא וָשׁוֹב כְּמַרְאֵה הַבָּזָק

נְאֻם הֵימָן הָאֶזְרָחִי: הָיִיתִי נוֹסֵעַ מֵאַשְׁקְלוֹן. לְעֵמֶק אַיָּלוֹן. וַיְהִי כִּי בָאנוּ אֶל הַמָּלוֹן. וְיָשַׁבְנוּ תַחַת הָאַלּוֹן. הֵחֵל כָּל אִישׁ מֵאַנְשֵׁי הַחֲבוּרָה לְסַפֵּר מֵאֲשֶׁר רָאוּ עֵינַיו. וְשָׁמְעוּ אָזְנָיו. וְשָׁם נִקְרָה אִישׁ זָקֵן וּמַשְׂכִּיל. וּמַרְאֵה עֵינַיו עַל שֵׂכֶל רַעְיוֹנָיו הוֹלֵךְ רָכִיל. וַיֹּאמֶר לָנוּ: אַגִּידָה לָכֶם אֵת אֲשֶׁר רָאוּ עֵינַי בַּמִּדְבָּר הַזֶּה. כִּי זֶה שָׁנִים רַבּוֹת עָבַרְתִּי עַל הַמָּקוֹם הַזֶּה. וָאֶשָּׂא עֵינַי וָאֵרֶא וְהִנֵּה אִישׁ גִּבּוֹר. גְּאוֹן אַדִּירִים יִשְׁבֹּר. וּמִגַּעֲרָתוֹ נָמַסּוּ הֶהָרִים. מִשַּׁאֲגָתוֹ יֵחַת לֵב כְּפִירִים. וְתַחְתָּיו סוּס דּוֹהֵר. כְּבָרָק מְמַהֵר. וְכַשֶּׁמֶשׁ זוֹהֵר. וּמִקּוֹל מִצְהֲלוֹתָיו. וְשַׁאֲטַת פַּרְסוֹתָיו. יֶחֶרְדוּ הַלְּבָבוֹת. וְנִבְהֲלוּ הַמַּחֲשָׂבוֹת. וּבְעֵת יִמַּסּוּ אֲסוּרָיו. וְיָכִין לָרוּץ אַשּׁוּרָיו. יִבְקַע הַיָּם וּמִשְׁבָּרָיו. וְיִרְקַע הַמִּיׁשֹור וּמִדְבָּרָיו. וְיָשִׁיב כֹּל אֲשֶׁר לְפָנָיו אַחֲרָיו. עָשָׂה מִן הָרוּחוֹת רַגְלַיִם. וּמִן הַבְּרָקִים כְּנָפַיִם. וּמִן הַכּוֹכָבִים עֵינַיִם. וְכִמְעַט לְחֹם לִבּוֹ יִמַּס גּוּפוֹ וִיהִי לְמָיִם. וְהַגִּבּוֹר רוֹכֵב עָלָיו אַמִּיץ וְחָזָק. וְרָץ וָשָׁב כְּמַרְאֵה הַבָּזָק. וְעַל יָדוֹ נֵץ מִתְנוֹצֵץ. כְּרוּחַ יְרוֹצֵץ. וְיֵצֵא חוֹצֵץ. אֶבְרָתוֹ רְקוּמָה מִמִּשְׁבְּצוֹת זָהָב. וְנַפְשׁוֹ חֲצוּבָה מִמְדוּרוֹת לַהַב. וְכָל צַיִד אֲשֶׁר יִרְאֶנּוּ. בְּעֵינָיו יִקָּחֶנּוּ. וּבְעוּפוֹ אַחַר הָעֳפָרִים. וְחַיּוֹת הַיְּעָרִים. וַיֵּרֶד הָעַיִט עַל הַפְּגָרִים. יִקְשְׁרֵם בְּחַבְלֵי אֲיֻמָּתוֹ.

וְיִכְרְעוּ כֻלָּם לְרַעַם גְּבוּרָתוֹ. וְהַנְּשָרִים יְסַפְּרוּ כֹחוֹ אֲשֶׁר גָּבַר. וּבַעַל כְּנָפַיִם יַגִּיד דָּבָר. וְהָיוּ סְבִיב הַגִּבּוֹר כְּלָבִים. פְּנֵיהֶם פְּנֵי לְהָבִים. אֲרֻכִּים וְלֹא רְחָבִים. דַּקִּים וְלֹא עָבִים. כְּאִלּוּ מִמַּחְצַב אֵשׁ חֲצוּבִים. יַשִּׂיגוּ כָל בּוֹרֵחַ וְלוּ בָא בֶעָבִים. וְיוֹרִידוּהוּ וְאִלּוּ עָלָה בְרֹאשׁ הַכּוֹכָבִים. לֹא יָנוּס לָהֶם נָס. עַד יְשִׁיתוּהוּ לָמַס. וּלְרַגְלָם לְמִרְמָס. וְלֵב הָאֲרִי מִפְּנַיהֶם יִמַּס. וָאֶשָׂא עֵינַי וָאֵרֶא הַגִּבּוֹר עַל סוּסֹו בַחֲמַת כֹּחוֹ יְרִיצוֹ. וּבְיָדוֹ נִצּוֹ. וּבְאַשְׁפָּתוֹ חִצּוֹ. יְפוֹרֵר בְּפַרְסוֹת סוּסוֹ הַסֶּלַע הֶחָזָק. וְיִשְׁחַק צוּר חַלָּמִישׁ וְיִטְחַן עַד אֲשֶׁר דָּק. וְהוּא שׁוֹאֵג בּקוֹל רָעָם וּתְרוּעוֹת. מְדַלֵּג עַל הֶהָרִים מְקַפֵּץ עַל הַגְּבָעוֹת. וְרוֹדֵף וְקוֹרֵא. כַּאֲשֶׁר יִרְדֹּף הַקּוֹרֵא. לְמַעַן יְעוֹרֵר הַצְּבָאִים. מִן הַמַּחֲבוֹאִים. וְיוֹצִיא הַחַיּוֹת מֵחוֹרֵיהֶם. וְהַשְּׁפַנִּים מִנִּקְרוֹת צוּרֵיהֶם. יוֹרִיד הַיְּעֵלִים מֵהַרְרֵיהֶם. וּבְסֶלַע הָהָר הָיְתָה מְעָרָה גְדוֹלָה. וּמְחִלָּה וּמִתּוֹכָהּ מְסִלָּה. וְאֹרַח סְלוּלָה. וּבְתוֹכָהּ כַּמָּה מִינֵי צְבָאִים נֶחְבָּאִים. וּמַחֲנֵה עֳפָרִים נִסְתָּרִים. וַהֲמוֹן אַיָּלִים קְטַנִּים עִם גְּדוֹלִים. וְאִמּוֹת עִם עוֹלָלִים. שׁוֹכְבִים בְּחֵיק אִמּוֹתָם. יוֹנְקִים שָׁדַיִם בִּלְשׁוֹנוֹתָם. וְהָאֵם רוֹבֶצֶת עֲלֵיהֶם. וּמִתְעַלֶּסֶת עִמָּהֶם. וְהֵם פַּעַם בְּחֵיקָהּ יְשֵׁנִים. וּבְמִרְעֶה טוֹב דְּשֵׁנִים. וַּפַעם יְרַקְּדוּ לְפָנֶיהָ בְשִׂמְחָתָם. בְּהִשְׁתַּפֵּךְ נַפְשָׁם אֶל חֵיק אִמּוֹתָם. וְהִיא עִמָּהֶם מְרַקֶּדֶת. וְעוֹלָה וְיוֹרֶדֶת. וְלֹא יָדְעָה כִּי יוֹם אֵידָהּ קָרוֹב.

וְשִׁמְשָׁהּ פָּנָה לַעֲרֹב. וּבְעוֹדָהּ עִמָּם מִשְׁתַּקְשֶׁקֶת וּמִתְרַפֶּקֶת. וְגוּרֶיהָ מְחַבֶּקֶת וּמְנַשֶּׁקֶת. וְהִנֵּה הַגִּבּוֹר מִהֵר עַל רִכְבּוֹ. וּבָא מִבּוֹא. וְרָץ מְהֵרָה. כְּרוּחַ סְעָרָה. אֶל פִּי הַמְּעָרָה. וְהִתְאַמֵּץ עַל סוּסוֹ וְהֵרִיצוֹ. וְדָרַךְ עַל קַשְׁתּוֹ חִצּוֹ. וַיָּשֶׂם אֶל הַמְּעָרָה מְגַמַּת חֶפְצוֹ. וְהֵכִין לְהִלָּחֵם כֵּלָיו. וְקָרָא הַכְּלָבִים אֵלָיו. וְהַנֵּץ הִתְעוֹרֵר עֲלֵיהֶם בַּחֲמָתוֹ. וְכִמְעַט לְרֹב תַּאֲוָתוֹ. יִגְזֹר אֶל הַצְּבָאִים חֲבָלָיו. וְיִשְׁבֹּר לָרוּץ לָהֶם כְּבָלָיו. וְכַאֲשֶׁר עָמַד הַגִּבּוֹר עַל פִּי הַמְּעָרָה. וְשָׁאַג בְּקוֹל הַגְּעָרָה. וְהִבְעִיר הָאָרֶץ כַּמְּדוּרָה. בָּרְחוּ הָעֳפָרִים מִמְּעוֹנוֹתָם. וְהַיְּעֵלִים מֵחֵיק אִמּוֹתָם. וְנֶחְבְּאוּ יֵלְדֵי הַצְּבָאִים. לְשַׁאֲגָתוֹ וְלֹא נוֹדַע אַיָּם. כִּי הוּא יִשְׁאַג וְיֶחֶרְדוּ בָנִים מִיָּם.

וְהַגִּבּוֹר בָּא אֲלֵיהֶם. וְשָׁלַח הַנֵּץ לַשָּׁמַיִם לְהוֹרִיד הַמָּוֶת עֲלֵיהֶם. וְהוּא עָבַר לִפְנֵיהֶם. וְהַכְּלָבִים מַקִּיפִים מִכָּל צַד אֶתְהֶם. וְהוּא מְרוֹצֵץ בְּאֵשׁ פְּלָדוֹת וּבְרֶק חֲנַית לְעֵינֵיהֶם. וְהָעֳפָרִים נִבְהֲלוּ וְנִתְחַלְחָלוּ. וְרָגְזוּ וְחָלוּ. וּבְזָכְרָם רַחֲמֵי יוֹלְדֵיהֶם. הָיוּ חוֹשְׁבִים לָשׁוּב אֲלֵיהֶם. וַיִּמְצְאוּ הַצַּר הַצּוֹרֵר עֲלֵיהֶם. מַפְרִיד בֵּינֵיהֶם. גַּם הָאִמּוֹת. נָפְלוּ עֲלֵיהֶם אֵימוֹת. וּקְצָתָם נֶחְבְּאוּ בִמְעֵי אֲדָמוֹת. וּקְצָתָם עָלוּ לִגְבָעוֹת רָמוֹת. וְכִזְכֹּר בְּנֵיהֶם. תִּתְעַטֵּף נַפְשָׁם עֲלֵיהֶם. וּמַה נִכְמְרוּ רַחֲמַי לָעֳפָרִים. הַמְפֻזָּרִים עַל רָאשֵׁי הֶהָרִים. וְלֹא חָדַל הַגִּבּוֹר לְדַלֵּג עַל הַשְּׁפָאִים. וְלִרְדֹּף אַחֲרֵי הַצְּבָאִים. וְלִשְׁאֹג עֲלֵיהֶם כִּלְבָאִים. וְכָל אֲשֶׁר הָיוּ רוֹצִים לִבְרֹחַ מֵרוֹדְפֵיהֶם. יָבוֹא הַנֵּץ מִכַּל מָקוֹם לְעֵינֵיהֶם. עַד יְשִׁיבֵם לַאֲחוֹרֵיהֶם. וְכַאֲשֶׁר נָקְטָה נַפְשָׁם. וּבָאָה שִׁמְשָׁם. נָמֵס לִבָּם לְקוֹל הַשּׁוֹאֲגִים. וְעָיְפָה נַפְשָׁם לַהוֹרְגִים. וְהַכְּלָבִים הִקִּיפוּ מִכָּל צַד סְבִיבָם. זֶה אוֹחֵז זְנָבָם. וְזֶה יָגוּד עֲקֵבָם. וְזֶה נוֹשֵׁךְ. וְזֶה טוֹרֵף. וְזֶה חוֹטֵף. עַד הִדְבִּיק הַמָּוֶת לִקְצָתָם. וְנִלְכְּדוּ בִשְׁחִיתוֹתָם. וְהַכְּלָבִים בְּאַכְזְרִיּוּתָם. קָרְעוּ כָל בְּשָׂרָם בְּמַלְתְּעוֹתָם. וְהוֹצִיאוּ בְנֵי מֵעֵיהֶם. וְלָקְקוּ בִלְשׁוֹנוֹתָם דְּמֵיהֶם. עַל לֹא חָמָס בְּכַפֵּיהֶם וְאֵין עָוֶל בִּידֵיהֶם. יִתְבָּרֵך הָאֵל אֲדוֹן עוֹלָם. וּמַנְהִיג בַּמִּשְׁפָּט יְצוּרָיו כֻּלָּם. וּמַסְתִּיר מִבְּנֵי אִישׁ סוֹדוֹתָיו. וְסִבּוֹת מִפְעֲלוֹתָיו. הַנּוֹתֵן קְצָת בְּרוּאָיו טֶרֶף לִקְצָתָם. וְהֵמֵּכִין מִיתַת אֵלֶּה סִבָּה לְטוֹבַת אֵלֶּה וּלְמִחְיָתָם. וּבָרָא הַגְּדוֹלִים לְנֵזֶק הַצְּעִירִים. וְשָׁלַח הַחֲזָקִים לְצַעַר הַדַּלִים הָעֲכוּרִים. וְנָתַן זֶה מִחְיָה לָזֶה. וּבְאַחֲרִיתָם זֶה כֵּן מוֹת זֶה. וְכֻלָּם בִּשְׁאוֹל יִשְׁתָווּ וְיִרְקָבוּ. וְיַחַד עַל עָפָר יִשְׁכָּבוּ.

וַיִּשָׂא מְשָלוֹ:

בְּנֵי תֵבֵל חֲיוֹתָם הוּא בְּמוֹתָם

וְהֵם מָזוֹן קְצָתָם אֶל קְצָתָם.

וְזֶה יִטְרֹף לְזֶה. גִּבֹּור וְחַלָּשׁ

וְכֵן הַצּוּר בְּרָאָם מִהְיוֹתָם.

וְהַחַלָּשׁ יְהִי נִרְדָּף וּמֻזָּק

וְהָעַזִּים לְהָרַע מַחְשְׁבוֹתָם.

וְהֶחָזָק יִהְי מַזִּיק וּבַסּוֹף

יְהִי מֻזָּק וְסָף אוֹנוֹ עֲדֵי תָם.

וְהָאוֹכֵל וְהֵנֶּאְכָל יְסוּפוּן

וְיִשְׁתָּווּ עֲלֵי עָפָר בְּמוֹתָם.

יְבֹרַךְ אֵל אֲשֶׁר שָׂגְבוּ פְלָאָיו

וְאֵין מֵבִין צְפוּנֵי מִפְעֲלוֹתָם.


אָמַר הַמַּגִּיד: וּכְשָׁמְעִי צָחוֹת מִלָּיו. וְחֹזֶק מְשָׁלָיו. תָּמַהּ לִבִּי וְרַעְיוֹנִי. וְאָמַרְתִּי אֵין זֶה כִּי אִם חֶבֶר הַקֵּינִי. וְהוּא חוֹשֵׁב אֲשֶׁר חָשַׁבְתִּי. וְהִכִּיר כִּי אוֹתוֹ הִכַּרְתִּי. וְרָץ לִקְרָאתִי. וְקִדֵּם בְּשָׁלוֹם אוֹתִי. וְהִתְעַלַּסְנוּ כָל הַיָּמִים אֲשֶׁר נִצְמַדְנוּ. עַד אֲשֶׁר מֵחֵיק הַחֶבְרָה נִפְרַדְנוּ. וּלְכָל צַד נָדַדְנוּ.
