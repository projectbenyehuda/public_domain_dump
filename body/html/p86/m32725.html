ספר תחכמוני - שַׁעַר אַרְבָּעָה עָשָׂר

בְּסֵדֶר תְּפִלָּה יְקָרָה. לֹא יִשָּׁקֵל כֶּסֶף מְחִירָהּ

נְאֻם הֵימָן הָאֶזְרָחִי: עָבַרְתִּי בַיָּם לְאֶרֶץ עַזָּה. אַחֲרֵי יְגִיעָה קָשָׁה וְעַזָּה. וְכַאֲשֶׁר בָּאתִי נְאוֹתֶיהָ. וְהִתְהַלַּכְתִּי בִנְתִיבוֹתֶיהָ. רָאִיתִי בִּקְצָת הַיָּמִים אַרְמוֹן. מָלֵא הָמוֹן. כִּמְלאֹת רִמּוֹן. וּבְתוֹךְ הֶהָמוֹן זָקֵן. בְּפִיו הַמְּלִיצוֹת יְתַקֵּן. וְיֶאֱסֹף פְּזוּרֵי הַתְּבוּנוֹת כֶּאֱסֹף בֵּיצִים מִקֵּן. וּלְפָנָיו תַּלְמִידִים. בְּמִינֵי הַחָכְמוֹת לוֹמְדִים. וָאֶשְׁמַע אֶחָד מֵהֶם אוֹמֵר לוֹ: אִם טוֹב בְּעֵינֵי אֲדוֹנֵנוּ יְחַבֵּר לָנוּ מִפְּנִינֵי רַעְיוֹנָיו. וּבְדֹלַח הֶגְיוֹנָיו. תְּפִלָּה אוֹ תְחִנָּה. תִּהְיֶה לָנוּ לְסוֹחֵרָה וְצִנָּה. וְתִבְקַע שַׁעֲרֵי מְעוֹנָה. וְתַגִּיעַ לְכִסֵּא הַשְּׁכִינָה. וַיַּעַן וַיֹּאמֶר לוֹ: הַט אָזְנְךָ וּשְׁמַע שְׁאֵלָתְךָ אֲשֶׁר שָׁאַלְתָּ. וּתְפִלָּתְךָ אֲשֶׁר בִּקַּשְׁתָּ. וְלֹא הִתְעַכֵּב כִּי אִם רֶגַע כְּפִי אֲשֶׁר הֵעִיר יְשֵׁנֵי זְמָמָיו. וְדִבֶּר אֶל סֶלַע לְשׁוֹנוֹ וְנָתַן מֵימָיו.

וַיַּעַן וַיֹּאמֶר לָהֶם: שִׁמְעוּ אֵלַי רוֹדְפֵי צֶדֶק הִנֵּה לָכֶם תְּפִלָּה. הִיא פְּאֵר הַתְּהִלָּה. וְנֵזֶר הַגְּדֻלָּה. לוּ הִתְפַּלֵּל בָּהּ אִישׁ בְּכַוָּנַת דַּעְתּוֹ. יִשְׁמַע הַבּוֹרֵא תְּפִלָתוֹ. וְזֹאת תְּחִלָתָהּ: אָנָּא אֲדוֹן עוֹלָם. וּבוֹרֵא הַבְּרוּאִים כֻּלָּם. מְעוֹן הַנְּשָמוֹת. וִיסוֹד הָעוֹלָמוֹת. בָּאתִי לְהִשְׁתַּחֲווֹת לְפָנֶיךָ כִּי אַתָּה הָאָדוֹן הַמְיֻחָד. הָאֶחָד וְלָא כְכָל אֶחָד. הַטּוֹב הָאֲמִתִּי. הַנֹּעַם הַנִּצְחִי. הָאוֹר הַפְּנִימִי. הַזֹּהַר הַנִּרְאֶה לְעֵין הָרַעְיוֹן. הַכָּבוֹד הָעֶלְיוֹן. יְסוֹד כָּל יְסוֹד. וְסוֹד כָּל סוֹד. וְעִקַּר כָּל עִקָּר. וְעִלַּת כָּל עִלָּה. וּמַעְיַן הַנְּכוֹחָה. וּמוֹצָא הָאֱמוּנָה. וִיסוֹד הַנְּכוֹנָה. מֵנִיעַ הַבְּרוּאִים מֵאִתּוֹ אֵלָיו. וּמַסִּיעַ הַמְּצוּאִים מֵרְשׁוּתוֹ [לִרְשׁוּת בְּעָלָיו]. הַקָּרוֹב כְּפִי רָחֳקוֹ. הָרָחוֹק לְפִי קָרְבוֹ. הַנִּמְצָא לְעַצְמוֹ. וְאֵין לְאַחֵר עִמּוֹ. הֶחָכָם וְלֹא בְחָכְמָה נוֹסֶפֶת. הַגִּבּוֹר וְלֹא בִגְבוּרָה נְתוּנָה. הַחַי וְלֹא בְחַיִּים קְנוּיִים מִבַּלְעָדָיו. הַמּוֹרֶה לִנְתִיב צֶדֶק. הַמְנַהֵג לְדֶרֶךְ הַיֹּשֶׁר. הַנִּסְתָּר מֵעֵינֵי הַנְּשָׁמוֹת הָעֶלְיוֹנוֹת. הַנִּמְצָא לְכָל דַּכָּא וּשְׁפַל רוּחַ. הַמֵּיטִיב וְטוֹבָתוֹ נְדָבָה. בְּלִי חוֹבָה. הַמִּתְחַסֵּד וְחַסְדּוֹ לֹא בְאֹנֶס רַק בְּאַהֲבָה. הַגִּבּוֹר וּלְרֹב גְּבוּרָתוֹ. יִמְחֹל בַּעֲלוֹת חֲמָתוֹ. הַנָּדִיב וּמֵרֹב נִדְבָתוֹ. יִתְנַדֵּב לְחוֹטְאִים בִּמְחִילָתוֹ. הַבּוֹרֵא וְלֹא נִבְרָא. הַמַּמְצִיא וְלֹא הֻמְצָא. הַשַּׁלִּיט וְאֵין שׁוֹלֵט בּוֹ. הַמַּשִּׂיג בְּלִי מֻשָּׂג. הָעוֹשֶׂה בְּלִי שֻׁתָּף. הַגּוֹזֵר בְּלִי עוֹזֵר. הַמּוֹלֵךְ בְּלִי יוֹעֵץ. הַשּׁוֹפֵט וְהוּא בַעַל הַחוֹב. אֵלֶיךָ אֶקְרָא הָאוֹר הָעֶלְיוֹן.

הַנִּשְׂגָּב בְּחַדְרֵי חֶבְיוֹן. אֲשֶׁר אֵלָיו כַּוָּנַת כָּל מִתְכַּוֵּן. וְחֵשֶׁק כָּל נֶפֶשׁ וְתַאֲוַת כָּל רַעְיוֹן. וְאֶשְׁאַל מִמְּךָ עֶזְרָתְךָ וְקִרְבָתְךָ בְּחַיֵּי הַסּוֹד אֲשֶׁר בּוֹ נִשְׂגַּבְתָּ. וּבְאֵין מָקוֹם נִצַּבְתָּ. וְרָחַקְתָּ וְקָרַבְתָּ. וּבָרָאתָ הַכֹּל מֵאַיִן. וְאֶתְחַנֵּן אֵלֶיךָ בִּכְבוֹד אֱלֹהוּתְךָ הַנִּצְחִית. וּבִמְצִיאוּתְךָ הַתְּמִידִית. וּגְדֻלָּתְךָ הַפְּנִימִית. הַגַּאֲוָה [הָעוֹלָמִית]. וּבִגְבוּרָתְךָ הַקַּיֶּימֶת. וּבְאַחְדוּתְךָ הַנֶּעְלֶמֶת. לְשַׂגְּבֵנִי וּלְחַלְּצֵנִי מִשּׁוּחַת הַתַּאֲוָה. וּלְהוֹצִיאֵנִי מִגַּלֵּי יָם הַמַּעֲוָה. וּלְעוֹדְדֵנִי בְּרוּחַ עֶלְיוֹנָה זַכָּה וּבָרָה. וּלְאַמְּצֵנִי בְּעֵצָה יְשָׁרָה. וּלְאַזְּרֵנִי בְשֵׂכֶל הַטָּהוֹר וְהַטּוֹב בְּלִי שִׁמְצָה. וּלְקַדְּשֵׁנִי בְּדֵעָה צְלוּלָה. וַאֲשֶׁר מֵרוּחַךָ אֲצוּלָה. וּלְחְּזֵקִני בְּנֶפֶש יוֹדַעַת עֵרֶךְ הָאֱמֶת וְלִהְיוֹת פְּרִי לִבְרִיאָתִי. וְלֹא תְהֵא לָרִיק יְצִירָתִי. וְאֶל אֵצֵא מִן הָעוֹלָם כַּאֶׁשר בָּאתִי. עָרֹם מִבִּגְדֵי ישֶׁר. וְעֶרְיָה מִמְּעִילֵי כשֶׁר.

אָנָּא הָאֵל הַבּוֹרְאִי מֵאַיִן וְהַיּוֹצְרִי מֵאֶפֶס. אַתָּה הוּא ה' לְבַדְּךָ הַמַּגְבִּיר עָלַי חַסְדּוֹ בְּטֶרֶם הֱיוֹתִי. הַמַּגְדִיל עָלַי טוּבוֹ עַד לֹא חָיִיתִי. וּמִזְּמַנִּים קַדְמוֹנִים. וְשָׁנִים בְּלִי נִמְנִים. הֶעֲלַנִי עַל מַחְשַׁבְתּוֹ בְּטֶרֶם נִמְצֵאתִי. וּפְקָדַנִי בִּפְקֻדָּתוֹ וְעוֹד לֹא נִבְרֵאתִי. וּזְכָרַנִי לְטוֹבָה בְּטֶרֶם נִקְרֵאתִי. וְגָזַר וְהוּא בִּשְׁמֵי עָרֶץ. לִבְרֹא אוֹתִי בָאָרֶץ. וְצִוָּה עַל הַגַּלְגַּלִּים וְהִתְגַּלְגָּלוּ. וְדִבֶּר אֶל כֹּחוֹת יְצִירוֹת וְנִפְעָלוּ. וְקָרָא אֶל הַשָּׁמַיִם מֵעָל וְאֶל הָאָרֶץ מִתַּחַת וְהֶחְבִּיר מִשְּׁנֵיהֶם נְשָׁמָה עֶלְיוֹנָה. עִם גּוּפָה שְׁפָלָה. וּבְטֶרֶם בְּרָאַנִי אָצַל רוּחֹו הַנְּדִיבָה. וְכֹחוֹ הַטּוֹבָה. עַל אֲבוֹת אֲבוֹתַי הַקַּדְמוֹנִים. וְהִגִּיעַ הַטּוּב הַהוּא עַד אָבִי וְאִמִּי. וְצִוָּה עַל הָאֶפֶס וְנִבְקַע. וְעַל הָאַיִן וְנִרְקַע. וּמִתּוֹכוֹ הוֹצִיאַנִי. וּמִלֵּב הָאֶפֶס הִמְצִיאַנִי. וְאָצַל עָלַי רוּחוֹ וְנִשְׁמָתוֹ. וְהֶעֱנִיקַנִי טוֹבָתוֹ. וּבְבֶטֶן אִמִּי הֱכִינַנִי. וּלְאוֹר עוֹלָם הוֹצִיאני. וְנָפַח בִּי רוּחַ חַיִּים. וְהִשְׁקַנִי מִתְּעָלוֹת הַשָׁדַיִם שְׁמָנִים מְמֻחָיִם. וְנָתן לִי חֵן בְּעֵינֵי אֲבוֹתַי לִסְבֹּל הֵם כָּל נֵזֶק וּלְהֵיטִיב לִי. וְלָשֵׂאת כָּל צַעַר וּלְהַרְוִיחַ לִי. וְכַאֲשֶׁר גָּדַלְתִּי בְּמַעֲדַנִּים כִּלְכְּלַנִי וְגִדְּלַנִי. וְעַל מֵי מְנוּחוֹת נִהֲלַנִי. וְדֶרֶךְ הַטּוֹב לִמְּדַנִי. וְתוֹרָתוֹ הִשְׂכִּילַנִי. וְאוֹר הָעוֹלָם הַבָּא בְּעֵין לִבִּי הֶרְאַנִי. וְכַמָּה זְמַנִּים הָלַכְתִּי שׁוֹבָב בְּדֶרֶךְ לִבִּי וְלֹא גְמָלַנִי. וְהֶאֱרִיךְ אַפּוֹ לְחַטָּאתִי וּבְיִסּוּרֵי אַהֲבָה יִסְּרַנִי. וְכַמָּה פְעָמִים שְׁכַחְתִּיו וְלֹא שְׁכֵחַנִי. וְלֹא זְכַרְתִּיו וּזְכָרַנִי. וַעֲזַבְתִּיו וְלֹא עֲזָבַנִי. וְדָבְקוּ בִי חֲסָדָיו וְטוֹב פְּעָלָיו. כְּאִלּוּ הֵם חוֹבָה עָלָיו. וּבְכַמָּה צָרוֹת נִסַּנִי. וּמִכֻּלָּם גְּאָלַנִי. וּמַה יוּכַל עֶבֶד נִמְאָס כָּמוֹנִי. וְעֶצֶב נִבְזֶה וְנִקְלֶה כְמוֹתִי. לְהָשִׁיב גְּמוּל בִּשְׁבָחָיו הַגְּרוּעִים עַל אַחַת מֵאֵלֶּה הֶטּוֹבוֹת הָרַבּוֹת.

יְהִי רָצוֹן וְרַחֲמִים מִלְפָנֶיךָ מְעוֹן הָרָצוֹן. וּמְכוֹן הָאֱמֱת. לְהֵיטִיב אַחֲרִיתִי מֵרֵאשִׁיתִי. וְאַל תֶּרֶף יָדְךָ מֵעַבְדֶּךָ. וְאַל תַּהֲרֹס הַטּוֹבָה אֲשֶׁר בָּנִיתָ. וְאַל תַּעֱקֹר הַחֶסֶד אֲשֶׁר נָטַעְתָּ. וְכַאֲשֶׁר הַחִלּוֹתָ כֵּן הַתְמֵד חֲסָדֶיךָ עִמִּי. עַד אֲתַקֵּן הַמְעֻוָּת וַאֲיַשֵּׁר הַמְקֻלְקָל. וַאֲבָרֵר סִיג לְבָבִי. וַאֲטַהֵר טֻמְאַת יְחִידָתִי. עַד אֲשִׁיבֶנָּה אֵלֶיךָ נְקִיָּה כַּאֲשֶׁר נְתַתָּהּ בִּי. וְעָזְרֵנִי לְחַלֵּץ נַפְשִׁי מֵחֶשְׁבּוֹנְךָ. וּלְהַצִּיל יְחִידָתִי מֵעָנְשְׁךָ. וּלְמַלֵּט רוּחִי מִצַּעַר דִּינְךָ. כִּי יָדַעְתִּי כִּי כֹל תּוּכַל לְהוֹרוֹת הַחַטָּאִים. וּלְטַהֵר בְּצִדְקָתְךָ הַטְּמֵאִים. וּלְהָשִׁיב אֵלֶיךָ הַפּוֹשְׁעִים. וּלְלַמֵּד דְּרָכֶיךָ לְתוֹעִים.

אָנָא אֱלֹהַי כַּמָּה חֲסָדִים עִם עַבְדְּךָ עָשִׂיתָ. כִּי בְלֹא אֹנֶס וּבְלֹא הֶכְרַח אוֹתִי בָרָאתָ. וְגוִּפי הֶחֱיֵיתָ. וְגֵוִי כָרִיתָ. וְנַפְשִׁי רִבִּיתָ. וְעַצְמִי שִׁוִּיתָ. וְלִבִּי כָּאֹהֶל נָטִיתָ. וּבְתוֹכוֹ אָהָלְךָ בָנִיתָ וְחָנִיתָ. וְגָלְמִי רָאִיתָ. וְדַרְכִּי פִנִּיתָ. וְטֶרֶם קְרָאתִיךָ עָנִיתָ. וְאִלּוּ אֶעֱמֹד אֶלֶף אַלְפֵי שָׁנִים מִשְׁתַּחֲוֶה. וּמוֹדֶה וּמְשַׁבֵּחַ וּמְהַלֵּל. לֹא יָכֹלְתְּי לְהָשִׁיב גְּמוּל הַטּוֹבָה אֲשֶׁר גְּמַלְתַּנִי. אַף כִּי עֲזַבְתִּיךָ וְהִכְעַסְתִּיךָ. וְהִתְמַכַּרְתִּי לְהָרֵעַ וְלֹא יְרֵאתִיךָ.וְהֵעַזְתִּי פָּנַי וְלֹא נִכְלַמְתִּי. וְהָלַכְתִּי בִּשְׁרִירוּת לִבִּי וְלֹא נִחַמְתִּי. וְיָדַעְתִּי וְהֶאֱמַנְתִּי. כִּי לוּלֵא שֶׁאַתָּה רַחוּם וְחַנּוּן אֶרֶךְ אַפַּיִם לָרְשָׁעִים. כְּדַאי הָיִיתִי לַעֲשׂוֹת בִּי שְׁפָטִים עַל רֹב חַטְּאוֹתַי. וּמִעוּט צִדְקוֹתַי. אֲבָל אַתָּה רוֹצֶה לְהַרְאוֹת יִתְרוֹן מַעֲשֶׂיךָ עַל מַעֲשֵׂה בָשָׂר וָדָם. כִּי כְשֶׁיֶּחֱטָא לִפְנֵי הַמֶּלֶךְ עַבְדּוֹ. יִגְמְלֵהוּ כְרִשְׁעוֹ וּמִרְדּוֹ. וְיִמְנַע מִמֶּנּוּ חַסְדּוֹ. וְאַתָּה לֹא כֵן בְּמַעֲשֶׂיךָ. רַק תֵּיטִיב לָרְשָׁעִים וְתִגְמֹל חֶסד לַחַיָּבִים. לְמַעַן לָא יִשָּׁאֵר לָהֶם טַעֲנָה. וְאַל יְהִי לָהֶם פִּתְחוֹן פֶּה בְאַחֲרִיתָם. כִּי יוֹדֵעַ אַתָּה. כִּי אִם תַּאֲרִיךְ אַפְּךָ לַחוֹטֵא כָּל יְמֵי חֲיוֹתוֹ. לֹא יִבְרַח מִיָּדְךָ בְּמוֹתוֹ. רַק בְּצֵאת נִשְׁמָתוֹ. לְיָדְךָ תְשׁוּבָתוֹ. וְאָז תּוּכַל לָקַחַת נִקְמָתְךָ עַל רֹעַ פְּעֻלָּתוֹ.

יְהִי רָצוֹן מִלְפָנֶיךָ יְסוֹד הָרָצוֹן וְעִקַּר הַטּוֹב הַנָּכוֹן. לִהְיוֹת כַּוָּנַת כָּל מַעֲשַׂי אֵלֶיךָ. וְכָל פְּעֻלָּתִי לִשְׁמֶךָ. וְכָל חֶפְצִי לִכְבוֹדָךְ. וְכָל מַחְשְׁבוֹתַי לָךְ. וְכָל רַעְיוֹנַי לְסוֹדָךְ. וְכָל מַעֲיָנַי בָּךְ. וְכָל תְּשׁוּקָתִי אֵלֶיךָ. וְכָל מִבְטָחִי עָלֶיךָ. וְכָל גְּדֻלָּתִי לְהִכָּנַע לְפָנֶיךָ. וְכָל הֲנָאָתִי לְהִתְעַנּוֹת בַּעֲבוֹדָתֶיךָ. וְאַל תָּשִׂים חֶלְקִי עִם רוֹדְפֵי תַאֲוַת הָעוֹלָם הַזֶּה וְטוֹבוֹתָיו. וְהַמִּתְעַדְּנִים בְּתַעֲנוּגוֹתָיו. וְהַחוֹשְׁבִים עֲבוֹדָתְךָ עֹל בַּרְזֶל עַל צַוָּארָם וְתוֹרָתְךָ דִּבְרֵי לַעַג לָהֶם.

אָנָּא מְקוֹר הַחַיִּים וּמַעְיַן הָרָצוֹן. יְהִי רָצוֹן מִלְּפָנֶיךָ לְהַצִּילֵנִי מֵאֱמוּנָתָם הָרָעָה. וְכַוָּנָתָם הַסְּכָלָה וְהַגְּרוּעָה. וְהַטֵּה לִבִּי לְיִרְאָתְךָ. וְהָבֵא בְרַעְיוֹנִי לְפַחֵד מִיּוֹם דִּינְךָ. וְלִזְכֹּר בְּכָל עֵת עָנְשְׁךָ. וְלִהְיוֹת נֶגֶד עֵינַי פַּחְדְּךָ כְטוֹטָפוֹת כַּל הַיָּמִים אֲשֶׁר אֶחְיֶה. וּשְׁמַע תְּפִלָּתִי בְּשַׁוְּעִי אֵלֶיךָ. בָּרוּך אַתָּה ה' שׁוֹמֵעַ תְּפִלָּה. יִהְיוּ לְרָצוֹן אִמְרֵי פִי וְהֶגְיוֹן לִבִּי לְפָנֶיךָ ה' צוּרִי וְגוֹאֲלִי.

וַיֹּאמֶר לָהֶם: וְהִנֵּה זֹאת תְּפִלָּה אַחֶרֶת יְפֵה-פִיָּה. אֵין לָהּ בְּתֵבֵל שְׁנִיָּה. וְלָהּ תְהִלָּה דוּמִיָּה. לְחֹזֶק עִנְיָנָהּ. וּלְתֹקֶף בִּנְיָנָהּ. וַיַּעַן וַיֹּאמַר: אֱלֹהַי בּשְׁתִּי וְנִכְלַמְתִּי. וּבְזָכְרִי אֲשֶׁר אָשַׁמְתִּי. יָד לְפֶה שַׂמְתִּי. וְעַל אֲלֻמּוֹת עֲוֹנוֹת אֲשֶׁר אִלַּמְתִּי. נָתַתִּי פָנַי אַרְצָה וְנֶאֱלַמְתִּי. וְרַעְיוֹנַי עָטוּ עַל שָׂפָם. כִּי הַיֵּצֶר שְׁאָפָם. וּבְכוּר הַיָּגוֹן צְרָפָם. וְלֹא שְׂרָפָם. מְעִיל הֲדָרִי קָרַע. וְעָלַי כָּאֲרִי כָּרַע. דְּמֵי חֲלָלִים לוֹעֵט. וּבְרֶשֶׁת הַתַּעֲנוּגִים בּוֹעֵט. וּבִזְדוֹנִי רְבִיד עֻזִּי לַגַּיְא הֻגַּר. וּכְבוֹדִי עַל צוּר חַרְבּוֹ חֻגַּר. וּבְנֵי רְשָׁפָיו לִרְגָעִים יִרְדְּפוּנִי. וּבְאֶפֶס יָד יִרְדּוּנִי. לְרֹב בַּלְהוֹת וּבַהֲלוֹת הַזְּמַן שָׁבוּ לִי מְתוּקָיו כְּמָרָיו. וּזְמָרָיו כִּנְמֵרָיו. וְנִפְתָּה לִבִּי בִגְאוֹנוֹ וְגֵאוֹ. וְלֹא בָן כִּי יֶלֶד יוֹם יָמִיר שִׂיאוֹ בְּגֵאוֹ. וְאִם הַזְּמַן אֵלָיו לֵב לֹא יָשִׁית. גְּאוֹנוֹ כְרֶגַע יַעַזְבוֹ. וְעַל כֵּן יֵקַד יְקֹד לַהֲבוֹ. יִשָּׂא תֵבֵל בְּאַנְחוֹתָיו. וּמַר שִׂיחוֹתָיו. וְיַשְׁקִיעַ גִּבְעוֹת עוֹלָם בְּנִבְכֵי נַהֲרוֹתָיו. וְגַלֵּי תְהוֹמוֹתָיו. עַד בִּקַּע יְאוֹרִים בַּצּוּרוֹת בַּצּוֹרוֹת. וְהֵשִׁיב עָבוֹת זָבוֹת עֳצוּרוֹת. וּנְהָרוֹת חִבֵּשׁ מִבְּכִי. וּבְחֻבּוֹ לְחוֹבוֹ צָרַב תּוֹכִי. וּבְתוֹכוֹ חַיַּת אַנְחָתוֹ חָנְתָה. וְיַד קַשְׁתָּהּ. עָלָיו קָשְׁתָה. וְאַשְׁפָּתָהּ עָלָיו רָנְתָה. הָעִירוּ וְהָקִיצוּ מַעֲמִיקֵי סָרָה. וְשׁוֹתֵי עֲסִיס הַתַּאֲוָה לְשָׁכְרַה. הֲלוּמֵי יֵין תַּעֲנוּגִים. וּמִנְּתִיב ישֶׁר נְסוֹגִים וְסוּגִים. וּבְתַאֲוַת תֵּבֵל נְגוּעִים וְנוּגִים. הוֹמִים כַּיּוֹנִים. וְנַפְשָׁם יוֹנִים. גְּדוּד מְרִיבָה הֵעַזְתֶּם. וּכְשָׁמִיר פְּנֵיכֶם הֵעַזְתֶּם. וּמִן הַגְּדֶרֶת הֲגִינָה לַזְתֶּם. שַׂמְתֶּם בְּרֹעַ תַּאֲוַתְכֶם. בִּנְאוֹת הַחֶסֶד תַּאֲוַתְכֶם. וּמֵאַנְתֶּם לִהְיוֹת הַחֶסֶד תַּאֲוַתְכֶם. עוּרִי נֶפֶשׁ הוֹמִיָּה. וְהִתְפַּתְּחִי מוֹסְרֵי צַוָּארֵךְ שְׁבִיָּה. בְּבוֹר הַתַּאֲוָה אֲסוּרָה. גּוֹלָה וְסוּרָה. הֲלוּמַת שְׂרוּקִים וְסוּרָה. הִתְלוֹנְנִי וְהִתְאוֹנְנִי. וְעַל חֶטְאֵךְ קוֹנְנִי. וְשִׁיתִי דִמְעֵךְ כַּיָּם וְעֵינַיִךְ אֳנִי. וְעַל אַחֲרִיתֵךְ אֲנִי. בָּנַי הָבִינוּ וְהַשְׂכִּילוּ. הָעָםָ אֲשֶׁר לְהַר הַגַּאֲוָה יַעְפִּילוּ. כִּי הַיּוֹם יָרוּמוּ. מָחָר יִשַּׁחוּ לִשְׁאוֹל וְיָרֻמּוּ. וַאֲשֶׁר בְּחַיָּתָם יָנוּמוּ. בְּמוֹתָם לְמַעֲמַד הַדִּין יָקוּמוּ. וְעָדֵיהֶם בְּכוֹר מָוֶת בְּגַבֵּי מָגִנָּיו יָרוּץ. וְתָקְפָם יָרוּץ. בְּמוֹרַג חָרוּץ. בְּכִלָּיוֹן חָרוּץ. וְלֹא יוֹעִיל לֶחָרוּץ עַל חָרוּץ. אַלְלַי לִי כִּי עֲוֹנִי כֵּלָיו עָדָה וְעָדָה. עַל נְאוֹת גִּילִי וְנָצוּ. עַל עֲלִיצוֹתַי וְנָצוּ. וְעָלַי חֲבָלָיו יָדָה. וּבְנֵי יְגוֹנִי תִרְגֵּל וְרָדָה. ה' מָה אָדָם וַתֵּדָעֵהוּ. וְהַמָּוֶת יִרְעֵהוּ. וְשִׁבְטוֹ בְּכָל עֵת יְרֹעֵהוּ. וְנַפְשׁוֹ הָעֲנֻגָּה נוּגָה. וְנִשְׁמָתוֹ הַנְּסוּכָה סוּגָה. וּשְׂרִידָיו עֲיֵפִים וְנִרְפִּים. וּגְדוּדָיו חַלָּשִׁים וְנִכְפִּים. וּמְאוֹרָיו לֹא יִגְּהוּ. וּמְזוֹרָיו לֹא יִגְהוּ. וְיַלְדֵי יוֹם בְּאֵידוֹ וּפִידוֹ יִנְהוּ.

יְהִי רָצוֹן וְרַחֲמִים מִלְּפָנֶיךָ מְעוֹן הָרָצוֹן. הַפְּנִימִי וְהַחִיצוֹן. לְאַזְּרֵנִי וּלְעָזְרֵנִי. וּלְעוֹדְדֵנִי וּלְסַעֲדֵנִי. בְּרוּחַ נְקִיָּה בָּרָה וּבְהִירָה. בִּמְלַאכְתְּךָ מְהִירָה. וּבְאוֹר אֲמִתְּךָ מַזְהִירָה. וּלְגָבְהֵי שְׁחָקֶיךָ תַּרְקִיעַ. וְלֹא תַרְגִּיעַ. עַד הַיֵּצֶר הָרָע תַּכְנִיעַ. וְאוּלַי גְּדוּדֵי הַתַּאֲוֹת אֲשֶׁר יִרְדְּפוּהָ. עַד יִגֳּפוּהָ. רְתוּקוֹת מְצוּקוֹת נִתְּקָה. וְגַלֵּי דְמָעֶיהָ שִׁתְּקָה. וְרֹאשׁ יְגוֹנֶיהָ מָחֲקָה. וּמֵהַרְרֵי טֶרֶף כַּאֲרִי זִנְּקָה. אַך מַה לַעֲשׂוֹת וְיִצְרִי הֲדִמַּנִי. וַהֲשִׁימַּנִי עַד הֲתִמַּנִי. שָׂמַנִי לוֹ לְמִפְגָּע. וּבְמַחֲלָתוֹ לִבִּי מְנֻגָּע וּמְיֻגָּע. וְאָרַחְתִּי אֶת צֶמֶד כָּל חֶמֶד לְחֶבְרָה. וְהִתְרָעִיתִי אֶת בַּעֲלֵי אוֹן קְצוּצֵי אֶבְרָה. וְנָשִׁיתִי אֲשֶׁר בְּרוּחוֹ שָׁמַיִם שִׁפְרָה. וְתְּהִי הַתַּאֲוָה לְיַלְדֵי עֲוֹנִי מְיַלֶּדֶת עַד נֶהְפְּכָה לְפוּעָה שִׁפְרָה. וִיקוֹד אַנְחוֹתַי בְּשִׁבְטוֹ יְרֹעֵנִי. וְהִכָּה בְּמַטֵּהוּ יָם עֵינִי. עַד נָמַס צְחִיחַ צֹר לְזִרְמֵי עֲנָנִי. אֲשֶׁר לֹא נֶעְצַר וְהִסְתַּבְּכוּ בְגַנֵּי יְגוֹנִי. פְּרָחָיו וְנִצָּנָיו. וְהִתְאַבְּכוּ בְלִבִּי גֵאוּת עֲשָׁנָיו. וּבְלַהֲבֵי צָרֶבֶת הָאֲנָחוֹת הַמַּרְתִּיחוֹת. וְשַׁלְהֶבֶת הַיְגוֹנוֹת הַקּוֹדְחוֹת. הֵגִיף עֵינִי דַלְתֵי מְרוֹמָיו. וַיִּסָּכְרוּ עֵינוֹת תְּהוֹמָיו. וְהָגָה בְרוּחוֹ הַקָּשָׁה. וְהָפַךְ יָם לְיַבָּשָׁה.

אָנָּא אוֹר הָעוֹלָם הָאוֹר הָעֶלְיוֹן. הָאֵר עֵינֵי שִׂכְלִי. מִמַּאֲפֵל סִכְלִי. וְהָסֵר עֲלוּמַי מִנֶּגֶד עֵינֶיךָ. וְהַעֲבֵר מָסַךְ הַחֵטְא הַמַּבְדִּיל בֵּינִי וּבֵינֶךָ. וְהָפֵץ עֲנַן זַעְמֶךָ. הַמַּסְתִּיר מִמֶּנִּי פָּנֶיךָ.

אָנָּא אֱלֹהַי. בְּטֶרֶם מָדַדְתָּ בַּשַּׁעַל. מְמַדֵּי תַחַת וּמֵעַל. הַנְחֵת וְתַעַל. אֲרוּכָה וְתַעַל. לְשׁוּבִי מִמַּעַל. וְשָׁלַחְתָּ מָזוֹר. לִגְהוֹת מָזוֹר. אֲשֶׁר לֹא חֻבַּש וְלֹא זוֹר. אֶל לֵב אָחוֹר נָזוֹר. וְעַל כֵּן בְּךָ בָטַחְתִּי. וְכַפַּי לְךָ שִׁטַּחְתִּי. לְרַפֵּא הַהֶרֶס. וְלִקְשֹׁר הַקֶּרֶס. וּלְהָאִיר חָשְׁכִּי כְּאוֹר חֶרֶס. אֲהָהּ עַל יְמֵי חֹרֶף. אֲשֶׁר הָפְכוּ עֹרֶף. וְאַחֲרֵי קָצוּ בִי וְחָרְפוּ. נָקְטוּ בִּי וְחֵרְפוּ. וְכַצֵּל חָלָפוּ. וּמַחְלְפוֹתָם אוֹתִי חָלָפוּ. וּגְדוּדֵי עֲוֹנַי בְּאוּרָם קָלוּנִי וְצָלוּנִי. וְאַחֲרֵי סַלּוֹנַי סִלּוּנִי וְסָלוּנִי. וּבְנִכְלֵיהֶם נִכְּלוּנִי. וּמִלֵּדָה וּמִבֶּטֶן שִׁכְּלוּנִי. וּמִדַּעַת סִכְּלוּנִי. וּבְיָדָם הָרָמָה רִצּוּנִי וְרַצּוּנִי. וּבְמַעֲרֶכֶת הַיָּגוֹן הֱרִצוּנִי. וּמִבּוֹר שָׁאוֹן לֹא הֱרִיצוּנִי. וּמִבַּיִת וּמִחוּץ הִקְצוּנִי. וּבַחֲרָבוֹת הִקְצוּנִי וְקַצּוּנִי. עַד שִׁבְּרוּנִי וְהִדְבִּירוּנִי. וְהַיֵּצֶר הַנַּעֲוֶה. אֲשֶׁר לְהַבְלֵי הָעוֹלָם מִתְאַוֶּה. וּבִמְעוֹן הַסִּכְלוּת מְעוֹנוֹ חָג וּמִתְאַוֶּה. פִּתּוּנִי חֲלָקָיו וֶהֱסִיתוּנִי חֲלָקָיו. וְעִנְּגוּנִי מְרוּקָיו. וּנְגָעוּנִי מְרוּקָיו. וְהִתְגָּעֲשׁוּ גַלֵּי מְצוּקָיו. וְרָעֲשׁוּ בִי מְצוּקָיו. וְכִסּוּנִי וְנָטוּ חִצַּי מֵעַל אֲפִיקָיו. כַּאֲרִי עָלַי נָעַר. וְאַבִּירֵי שְׂשׁוֹנִי נִעַר. וּגְדוּדָיו מִבְצְרֵי שִׂיחַי יְקִיצוּן. וִישֵׁנֵי תֹהוּ יְקִיצוּן. וְגָזַל הַזְּמַן מֵעָלַי מַעֲטֵה הוֹדוֹ. וְיַד נִגְעוֹ לְרִגְעוֹ עָלַי הוֹדוֹ. וְצוּקוֹתָיו אֲפָפוּנִי. וְכַנַּחַל גְּרָפוּנִי. וַאֲכָלוּנִי וַיְכַלּוּנִי. וּמִיַּד יְגוֹנִי לֹא גְאָלוּנִי. וּבַחשֶׁךְ גְּאַלוּנִי. עַד גָבְרוּ עָלַי צְלָלַי. וְזָרַק עָלָיו וּמְלֶאכֶת שִׂכְלִי נְמִבְזָה וְעָיְפוּ נַהֲלָאָיו וְעָלָיו עֹל וְהָפַךְ הַזְּמַן שַׁלְוִי לְפִיד. וְהִבְעִיר בְּקֶרֶב קְרָבַי לַפִּיד. וּבִעוּתָיו נָשׁוּנִי. וּבְלִי טוֹבָה נָשׁוּנִי. לָכֵן עוּרִי נֶפֶשׁ הוֹמִיָּה טְרוּפָה כָאֳנִיָּה. בְּנִבְכֵי יָם אֲנִיָּה. לְמָתַי הַבְלֵי תֵבֵל יְסִיתוּךְ. וְאַל יְסִיתוּךְ וְיַשְׁחִיתוּךְ. עַד יְחִתּוּךְ. וּמִזְּמַן אָרוּךְ. לִהְיוֹת טֶרֶף לְשִׁנֵּיהֶם כָּרוּךְ וּבָרוּךְ. נִפְגָּע לְיֵצֶר סָמוּךְ שָׂמוּךְ. וּבְבוֹר הַדְּמִי הֲדִמּוּךְ וְדִמּוּךְ. עַל כֵּן לְמַדִּיחֵךְ אֲשֶׁר יַדִּיחֵךְ. הַזִּי נִצְחֵךְ. עַל הַמְּרוֹרִים. בְּקוֹל תַּמְרוּרִים. וּמֵעַל הַיֵּצֶר שְׂטִי. וּמִדְּרָכָיו נְטִי. וְסוּת רֶשַׁע פִּשְׁטִי. וְעַל עִיר הָאֱמוּנָה פִשְטִי. אוּלַי תְּיַשְּׁרִי רְכָסֶיהָ. וּתְרַפְּאִי רְסִיסֶיהָ. וְתִבְנִי הֲרוּסֶיהָ. וְתַעַזְבִי מְעוֹנוֹת הַתַּאֲוָה הַנּוֹשָׁבוֹת. וְתַעַזְבִי טִירוֹת הַצְּדָקוֹת הַחֲרֵבוֹת. וְאַל תִּבְטְחִי בְתֵבֵל אֲשֶׁר גּוֹיֶיהָ מְשַׁכֶּלֶת. כְּסֻכָּה נוֹפֶלֶת. וּכְאֵלָה נוֹבֶלֶת. וְהַבְעִירִי זִיקֵךְ. וּפִתְחִי זִיקֵךְ. אֲשֶׁר יְצִיקֵךְ. עַד כְּלִי רֵק יַצִּיגֵךְ. וְשׁוּבִי עַד סוֹד הַיְסוֹד. אֲשֶׁר בִּנְיָנֵךְ עָלָיו יַעֲמֹד. כִּי שְׁמוֹ מָקוֹר. וּרְצוֹנוֹ מָקוֹר. וּסְלִיחָתוֹ מָקוֹר. וְהָאֵל יוֹרֵךְ עַד תִּזְכִּי לִרְאוֹת פָּנָיו. וְתִמְצְאִי חֵן בְּעֵינָיו. וְכֵן יְהִי רָצוֹן אָמֵן.

אָמַר הַמַּגִּיד: וָאֶשְׁאַל עַל שְׁלוֹמוֹ. וְעַל שְׁמוֹ. לְאֶחָד מִתַּלְמִידָיו. לְמַעַן אֶעֱמֹד עַל סוֹדָיו. וּכְשָׁמְעוֹ מִלָּי. וַיֹּאמֶר אֵלָי:

אֲנִי חֶבֶר מְחַבֵּר הַתְּהִלּוֹת

וּבִלְבָבִי לְכָל חָכְמָה מְסִלּוֹת.

אֲחַבֵּר לַסְּכָלִים מַהֲתַלּוֹת

וְאֶעְרֹךְ לַחֲסִידִים נִיב תְּפִלּוֹת.
