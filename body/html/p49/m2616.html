טוֹבוֹת לְאֵל

וְזִמֵּן הָאֵל יִתְעַלֶּה הַתִּגְבֹּרֶת וְהַנִּצּוּחַ בַּמִּלְחָמָה הַנִּזְכֶּרֶת, אֶלָּא שֶׂאָבִי, יוֹקִירֵהוּ אֵל, שָׁב מִמֶּנָה חוֹלֶה וְחָלָה זְמַן אָרֹךְ, וְכַאֲשֶׁר חָלָה אָמַר זֹאת:


שִׁירָה

טוֹבוֹת לְאֵל עָלַי וְנִסִּים מְכֻפָּלִים, / רַבּוּ כְּמוֹ אַרְבֶּה וְגָדְלוּ כְּמִגְדָּלִים.

בַּמָּה אֲקַדֵּם אֵל עֲלֵיהֶם, וּמָה אָשִׁיב / עַל כָּל אֲשֶׁר גָּמַל לְנַפְשִׁי בְתַגְמוּלִים

אִם אֶתְּנָה לוֹ כָּל אֲשֶׁר לִי – הֲלֹא לָאֵל / אֶרֶץ וְקִנְיָנִים וּבָתִים וְאֹהָלִים;

אוֹ אַעֲלֶה נַפְשִׁי לְשַׁי לוֹ – בְּיָדוֹ כָּל / נֶפֶשׁ, וְהוּא שַׁלִּיט בְּלִבּוֹת וּבִכְסָלִים;

אוֹ אִם אֱהִי מַרְבֶּה דְבָרַי בְּהִלּוּלָיו – / יִכְלוּ, וְהִלּוּלָיו עֲצוּמִים וְלֹא כָלִים;

אִם אֶחֱסֹם אֶת פִּי וְאֶדֹּם – עֲלֵי מַה זֶּה / שָׂם פֶּה לְבֶן אָדָם, וְלָשׁוֹן וּמִלּוּלִים?

אוֹ אֵיךְ הֱבִינַנִי אֱלֹהִים, וְחַנַּנִי / דַּעַת, וְאֶדְמֶה אֶל פְּתָאִים וְלַעְצֵלִים

אֵין לַעֲשׂוֹת כִּי אִם לְסַפֵּר כְּפִי יָכְלִי גָּדְלוֹ, וְלַרְצוֹתוֹ בְּפֵרוּשׁ וּבִכְלָלִים.

אוֹדֶה לְצוּר, לֹא סָךְ בְּיוֹם צַר וְתוֹכֵחָה / בֵּינִי וּבֵין יִשְׁעוֹ בְּמַסְגֵּר וּמַנְעוּלִים.

כַּמָּה הֲנִיעוֹתָ בְחֵילָךְ וְהוֹרַדְתָּ / שָׁרִים וְגִבּוֹרִים בְּלָחֶם מְגֻדָּלִים,

רָצוּ לְלָכְדֵנִי בְחֵמָה וְחֹזֶק יָד – / אָכֵן הֲשִׁיבוֹתָם בְּבוּשָׁה מְחֻלָּלִים.

גַּם בֶּאֱמֹר אַחַי בְּחָלְיִי: אֲהָהּ, כִּי זֶה / חָשׁוּב כְּמֵתִים לֹא יְחֻשַּׁב בְּתוֹךְ חוֹלִים!

חַסְתָּ וְצִוִּיתָ לְהַצִּיל יְחִידָתִי / מִכָּל חֳלִי הֻפְלָא בְצִירִים וּבַחְבָלִים.

שָׁנָה – תְּכַתֵּת אֶת מְשַׂנְאַי, וְאַחֶרֶת – / תַּצִּיל יְדִידָך מֵחֳלָאִים וּמֵחִילִים,

וּבְעֵת יְקוֹשׁוּן לִי שְׁתֵּיהֶן בְּעוֹד שָׁנָה – / תָּקִים פְּעָמַי עַל דְּרָכִים מְסֻקָּלִים.

נִקְווּ פְלִשְׁתִּים וַעֲרָבִים וְהִקִּיפוּ / עָלַי לְהִלָּחֵם בְּסוּסִים וּבַחְיָלִים,

חָשְׁבוּ וְאָמְרוּ: אֵין תְּקוּמָה לְגִרְגָּשִׁים / לִפְנֵי מְדָנִים וֶאֱמוֹרִים וְנִבְדָּלִים!

תָּעוּ בְחֶשְׁבּוֹנָם, לְמַעַן כְּתָבַנִי / אֵל חַי בְּסוֹד אִישִׁים גְּאוּלִים וְנִצָּלִים!

אֵלֶּה בְרֶכֶב אוֹ בְסוּסִים – וְהוּא יַעַשׂ / לִי כַּעֲשׂוֹתוֹ לַחֲסִידִים וְלִגְדוֹלִים!

בָּאוּ בְשִׂמְחָתָם, כְּנוֹצְחִים בְּיוֹם חַלֵּק / שָׁלָל, אֲבָל שָׁבוּ עֲצֵבִים, מְחֻבָּלִים;

חָנוּ חֲצִי הַיּוֹם בְּטוּחִים בְּכֹחָם, אַךְ / נָסְעוּ חֲצִי לַיִל נְגוּפִים וְנִכְשָׁלִים;

יָצְאוּ בְדֶרֶךְ חַד שְׁלֵמִים, וְנָסוּ עַל / שִׁבְעָה דְרָכִים נֶהֱלָמִים, מְחֻלָּלִים;

עָלוּ לְאוֹר הַיּוֹם, כְּגִבּוֹר, בְּמַסְלוּלִים, / פָּנוּ בְעֶרֶב יוֹם, כְּגַנָּב, בְּמִשְׁעוֹלִים,

הָלְכוּ בְחֶרְפָּתָם וְנָשׂוּ כְלִמָּתָם, / הָיוּ בְּפִי אִישִׁים לְחִידוֹת וְלִמְשָׁלִים.

שַׁבְתִּי בְלֵב עָלֵץ, וְאוּלָם פְּקָדַנִי / צוּרִי בְתַחְלוּאִים בְּטוֹבָה מְסֻבָּלִים,

אֵחַם, בְּעֵת בּוֹאָם, כְּתַנּוּר בְּגֶחָלִים, / אֶשְׁטֹף, בְּעֵת שׁוּבָם, כְּנָהָר וְכִנְחָלִים,

וּכְמוֹ אֲרִי שִׁמְשׁוֹן אֲנִי עַז, וְדַם גּוּפִי / מָתוֹק, וְתַחְלוּאַי בְּדָמִי מְכֻלְכָּלִים,

מֵעֵת לְעֵת בָּאִים, כְּכָרוֹז אֲשֶׁר בָּא אֶל / שָׁעוֹת מְזֻמָּנוֹת וְעִתִּים מְשֻׁקָּלִים.

אֶצְרַח בְּרַעַד, אַחֲרָיו אֵשׁ, וְאַחְרֵיהֶם / זֵעָה, בְּמֵימֶיהָ עֲצָמַי מְטֻבָּלִים,

כָּעָב אֲשֶׁר מַרְעִים וּבוֹרֵק, וְסוֹף דָּבָר / מֵימָיו עֲלֵי אֶרֶץ נְטוּפִים וְנִזָּלִים,

אֵשֵׁב – וְאֶשָּׁעֵן כְּזָקֵן, וְעֵת מִשְׁכָּב – / אֶשֹּׁם וְאֶתְנוֹדֵד בְּעֶרֶשׂ כְּתַעְלוּלִים,

כּוֹשֵׁל בְּעֵת קִימָה, מְתָאֵב אֱלֵי נוּמָה, / צָמֵא אֱלֵי מַיִם, מְתָעֵב לְאֹכָלִים.

כַּמָּה סְבִיבַי מִקְּרוֹבִים, כְּאֵבֵי לֵב / עָלַי, לְעֵת מַרְפֵּא מְצַפִּים וּמוֹחִילִים,

שָׁטִים לְעֵת עֵרֵב וּבֹקֶר בְּכָל רוֹפֵא / אֻמָּן, וְכָל שִׁקּוּי מְבִיאִים וּמוֹבִילִים

שָׁבִים אֱלֵי צוּרָם בְּסֵתֶר וּמַפִּילִים / שֶׁוַע, וּבִגְדֵיהֶם בְּדָמִים מְגוֹלָלִים.

רָאוּ, בְּהֵאָרֵךְ יְמֵי מַחֲלָה, כִּי לֹא / עָלְתָה אֲרוּכָה לִי, וְאָמְרוּ מְבֹהָלִים:

הַאֵין תְּעָלָה לוֹ – וְהוּא יֵשְׁתְּ תָּמִיד כָּל / סַמִּים, לְתַחְלוּאִים מְרַפִּים וּמוֹעִילִים!

הַאֵין בְּכָל אֵלֶּה, אֲשֶׁר יִשְׁפְּכוּ שִׂיחָם / בַּעְדוֹ, אֲשֶׁר שִׂיחָיו בְּשַׁחַק מְקֻבָּלִים?

עָבְרוּ עֲלֵי נַפְשִׁי שְׁלֹשָׁה יְרָחִים, עַד / הָיוּ כְמוֹ אַיִן יְצוּרַי, וְכִצְלָלִים,

שָׁלְמוּ – וְשָׁלַח לִי רְפוּאָה, כְּמוֹ מָטָר / שָׁלַח בְּהֵעָצֵר שְׁחָקִים, וְכִטְלָלִים.

יֵשׁ בָּם בְּרָכָה, כִּי יְמָרֵק אֱלֹהִים בָּם / חוֹבִים, אֲבָל אִישִׁים קְרָאוּם מְקֻלָּלִים.

אַחְבִּיר לְאֵל, גָּמַל כְּחַסְדּוֹ וְרַב טוּבוֹ / עָלַי, וְלֹא גָמַל כְּמַעְלִי וּמַעְלָלִים,

שֶׁבַח, כְּיַיִן לַזְּקֵנִים דְּבָרָיו, אוֹ / שֶׁמֶן לְחָתָן, אוֹ כְחָלָב לְעוֹלָלִים:

אֵלַי, נְקַלּוֹתִי וְשַׁחְתִּי, וְכֵן אָנִי / בֶּן שָׁח וּמוּשָׁחִים וְשָׁפָל וּמָשְׁפָּלִים,

מִכֹּל אֲשֶׁר תִּפְעַל לְעֶבֶד, אֲשֶׁר פֹּעַל / יָדָיו מְעֻוָּת, וַאֲמָרָיו מְעֻקָּלִים,

מַה זֶּה אֲדַבֵּר עוֹד וְאֹמַר – וְהוּא עָשָׂה / טוֹבָה וְלֹא הֵשִׁיב, כְּרִשְׁעָם, לְנִפְתָּלִים.

הִנֵּה לְךָ אֶרֶץ, וְיֵשׁ בָּה אֲשֶׁר יִבְזֶה / דָּתָךְ וְיִסְגֹּד לֶאֱלִילִים וְלִפְסִילִים,

תּוּכַל לְצַוּוֹתָהּ – וְתִפְתַּח, כְּיוֹם קֹרַח, / פִּיהָ וְתִבְלָעֵם כְּמֵתִים וְכִנְפָלִים.

אִם תַּאֲרִיךְ אַפְּךָ לְזָרִים – הֲתִקְצֹף עַל / זֵדִים אֲשֶׁר צָצוּ בְעַמָּךְ, וְעַוָּלִים?

אִם תַּעֲמִידֵנִי בְמִשְׁפָּט – צְדָקָה אֵין / עִמִּי, וְאַתְּ צַדִּיק בְּדָבָר וּמִפְעָלִים.

אוֹ אִם תְּחַפֵּש אֶת חֲטָאַי – תְּהִי מוֹצֵא / רָעוֹת כְּנִטְפֵי יָם וְכַחוֹל וְכִכְסִילִים.

אִם בָּם אֱהִי נִכְשָׁל לְמָחָר – אֲנִי שַׂמְתִּים, / אֶמֶשׁ וְהַיּוֹם, לִי לְמוֹקֵשׁ וּמִכְשׁוֹלִים.

אִם תַּעֲבִירֵם – כֵּן שְׁמָךְ: מַעֲבִיר חַטַּאת / רָעִים, וְחוֹמֵל עַל סְכָלִים וּמַשְׂכִּילִים

אִם אֶעֱשֶׂה עָוֶל בְּבֵיתִי, וְכַמִּשְׁפָּט / אַעַשׂ בְּשִׁבְתִּי חוּץ בְּנָתִיב וּבִשְׁבִילִים –

אֶהִי כְאִישִׁים רוֹחֲצִים אֶת פְּנֵי גִלְדָּם / מִחוּץ, וּמִבַּיִת – לְבָבוֹת מְגֹאָלִים.

אֶדְרשׁ רְצוֹן הָאֵל בְּלִבִּי וּבִלְשׁוֹנִי, / אָכִין פְעֻלָּתִי בְסֵתֶר וּמַקְהֵלִים.

אֶתֵּן בְּנֵי אֵלִים לְמֶלֶךְ, אֲשֶׁר אֵין לוֹ / עֵרֶךְ וְדוֹמֶה בֶּאֱלֹהִים וּבָאֵלִים,

לֹא אֶהֱיֶה חָבֵר לְעַוָּל, וְלֹא אָשׁוּב / לָעַד לְכִסְלָה כַּעֲוִילִים וְכִכְסִילִים,

לֹא אֶעֱמֹד בִּנְתִיב נְסוֹגִים, וְלֹא אֵלֵּךְ / עִם רַע, וְלֹא אֵשֵׁב לְהָלִיץ כְּנוֹאָלִים,

כִּי יוֹם לְאֵל מוּכָן לְמִשְׁפָּט, וּבוֹ יַעַל / כָּל מַעֲשֶׂה אִם טוֹב וְאִם רָע בְּמִשְׁקָלִים!
