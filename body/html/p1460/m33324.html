וַיְהִי כַאֲשֶׁר כָּלוּ

מתוך ספר המבקש – פלאי הרפואה


וַיְהִי כַאֲשֶׁר כָּלוּ תוֹכְחוֹת הַמְבַקֵּשׁ עִם הֶחָרָשׁ / לְהִתְוַכֵּחַ עִם הָרוֹפֵא דָּרוֹשׁ דָּרַשׁ. / וְהָאֵל מַצְלִיחַ בְּיָדוֹ כֹּל אֲשֶׁר הוּא עוֹשֶׂה. / וַיָּבוֹא בְתוֹךְ הַבָּאִים אֶל בֵּיתוֹ / אִישׁ אִישׁ לְבַקֵּשׁ רְפוּאָה מִמַּחֲלָתוֹ – / וְהוּא יוֹשֵׁב עַל כִּסְאוֹ בְּכָבוֹד וּבְתִפְאֶרֶת / וַיַּעַט כַּמְעִיל הַהוֹד וְהָאַדֶּרֶת / לָבַשׁ לְמַעַן כַּחֵשׁ אַדֶּרֶת. / וַיּוֹצֵא כָל אֶחָד מֵהֶם כְּלִי שִׁתְנוֹ / וַיִּקָּחֵהוּ וְעָלָיו הֵשִׂים עֵינוֹ, / וַיִּסְתַּכֵּל בְּמַרְאֵהוּ וְעֵינוֹ / וַיָּנַע אוֹתָם פַּעֲמַיִם – / וַיֶּהֱמוּ וַיֶּחְמְרוּ הַמַּיִם. / וַיַּבֵּט אֶל הֶעָנָן אֲשֶׁר עַל פְּנֵיהֶם / וּבְמַרְאֵה שִׁמְרֵיהֶם וְתַבְנִיּוֹתֵיהֶם, / וַיֹּאמֶר אֶל הָאֶחָד ״זֶה הַשֶּׁתֶן מוֹרֶה עַל חֳלִי עָצוּם וְכָבֵד / וְעַל חֻלְשַׁת הָאִצְטוֹמְכָא וְסִתּוּם הַכָּבֵד / וְאִם לֹא יֵרָפֵא בְרֵאשִׁיתוֹ – אַחֲרִיתוֹ יִהְיֶה עֲדֵי אוֹבֵד. / וְזֶה הָאַחֵר מוֹרֶה עַל חֲמִימוּת גָּדוֹל וְחָזָק / וְזֶה מוֹרֶה עַל קְרִירוּת וּבְכָל דָּבָר קַר בְּעָלָיו נִזָּק״. / וַיִּגְזֹר עַל הָאֶחָד שֶׁיִּחְיֶה מֵחָלְיוֹ / וְעַל הָאַחֵר שֶׁיָּמוּת וְרוֹאָיו יֹאמְרוּ ״אַיּוֹ?״ / וַיֹּאמֶר כִּי הָאֶחָד בְּזֵעָה תִהְיֶה רְפוּאָתוֹ, / וּלְאַחֵר אָמַר כִּי תִהְיֶה בְהִלּוּךְ מֵעָיו אֲרוּכָתוֹ, / וּלְאַחֵר בְּהָקִיא הַלֵּחוּת אֲשֶׁר בְּקֵבָתוֹ.

כָּזֶה וְכַיּוֹצֵא בוֹ אָמַר לַמְבַקְשִׁים מִמֶּנּוּ רְפוּאוֹת. / וַיֹּאמֶר הַמְבַקֵּשׁ בְּלִבּוֹ: אֵין אֵלּוּ אֶלָּא דִבְרֵי נְבוּאוֹת, / וְאִם בְּכֹחַ זֹאת הַמְּלָאכָה לָדַעַת כָּל זֶה עַל אֲמִתָּתוֹ – / זוֹ הִיא בֶאֱמֶת חָכְמַת הָאָדָם הָאֲמִתִּית וְתַכְלִית הַצְלָחָתוֹ, / וְהִנְנִי אוֹחֵז בָּהּ וְלֹא אֵרֶף מִמֶּנָּה / אֲבַקְשֶׁנָּה כַכָּסֶף וְכַמַּטְמוֹנִים אַחְפְּשֶׂנָּה. / וַיֹּאמַר: אֲנַסֶּה עוֹד דִּבְרֵי הָרוֹפֵא, / הָרוֹאֶה הַנֶּעְלָם וְהָעֲתִידוֹת צוֹפֶה – / וּמִי יוֹדֵעַ הֶחָכָם הוּא אוֹ סָכָל, וְאִם זָר מַעֲשֵׂהוּ / וּדְבַר פִּיהוּ – אוּלַי מִשְׁגֶּה הוּא! / וַיֵּלֶךְ אִתּוֹ בְּלֶכְתּוֹ לִרְאוֹת הַחוֹלִים / וַיָּבוֹאוּ לְבֵית אִישׁ גָּדוֹל וְרַב פְּעָלִים / וְהָאִישׁ תָּם יוֹשֵׁב אֹהָלִים. / וְלוֹ בַת לְמַעֲצֵבָה שׁוֹכֶבֶת / וְעַל עֶרֶשׂ דְּוַי יוֹשֶׁבֶת, / וְנַפְשָׁהּ מֵהַחֳלִי דוֹאֶבֶת. / וְהִיא לְבַדָּהּ נוֹלְדָה לְאָבִיהָ מֵאֵין תְּאוֹם וּתְאוֹמָה, / עַל כֵּן הָיְתָה רַכָּה וַעֲנֻגָּה הָעַלְמָה, דַּדֵּי בְתוּלֶיהָ לֹא עִשּׂוּ / וְכַפּוֹת רַגְלֶיהָ הַצֵּג עַל הָאָרֶץ לֹא נִסּוּ, / כִּי הָאָב שָׁמַר זֹאת הַבַּת כְּאִשׁוֹן בַּת עַיִן / וְכָל מַחֲמַדִּים נֶחְשְׁבוּ נֶגְדָּהּ לְאַיִן. / וְהִיא בְתוּלָה וְאִישׁ לֹא יְדָעָהּ וְעֹל הַבַּעַל לֹא עָלָה עָלֶיהָ. / וַיֵּט הָרוֹפֵא יַד יְמִינוֹ אֵלֶיהָ / וַיְּמַשֵּׁשׁ אֶת כָּל כֵּלֶיהָ / וַיּוֹסֶף לְגַלּוֹת זְרוֹעַ הַשְּׂמָאלִית וְעָלֶיהָ מִשֵּׁשׁ בְּאֶצְבְּעוֹתָיו – / וַיִּשְׁתּוֹמֵם כְּשָׁעָה חֲדָה חוֹשֵׁב מַחְשְׁבוֹתָיו. / וְהַמְבַקֵּשׁ הָיָה שׂוֹבֵר עַל כָּל מַעֲשָׂיו / וְאֵשׁ הַקִּנְאָה בּוֹ בוֹעֵרָה / וְכִמְעַט תֹּאכְלֵהוּ – כֶּאֱכֹל קַשׁ לְשׁוֹן אֵשׁ – קִנְאָתוֹ לַנָּעֲרָה. / אָז אָמַר וַיִּשָּׂא מְשָׁלוֹ / כִּי קִנֵּא בָרוֹפֵא וּבְפָעֳלוֹ:


אִם תֶּאֱהַב לַעְמֹד בְּסוֹד אִישִׁים / יָקְרוּ וְסִתְרֵיהֶם הֱיוֹת צוֹפֶה

וּרְאוֹת צְבִיַּת חֵן אֲשֶׁר מֵעֵין / כָּל אִישׁ מְסֻתֶּרֶת – הֱיֵה רוֹפֵא!


אַחַר כֵּן אָמַר לַאֲבִי הַנַּעֲרָה ״דְּבַר סֵתֶר לִי אֵלֶיךָ, / צַוֵּה לְהוֹצִיא כָּל אִישׁ מֵעָלֶיךָ!״ / וַיֹּאמַר ״הִסְתַּכַּלְתִּי בָחֳלִי בִתְּךָ וְלִי נִגְלֹה נִגְלָה / וְהִיא מֵחֳלִי הַחֵשֶׁק חוֹלָה, /וּמֵאַהֲבָה מְסֻתֶּרֶת תּוֹכַחַת מְגֻלָּה״ / וְשֵׁם הָאִישׁ הֶחָשׁוּק לוֹ גִּלָּה. / ״וְעַתָּה רְאֵה בִתְּךָ הַיָּפָה בַנָּשִׁים / אֲשֶׁר בְּיָפְיָהּ תָּצוּד הַנְּפָשִׁים / וְתִשְׁלַח רוּחַ הַתַּאֲוָה בִישִׁישִׁים / וּתְסַכֵּל יוֹעֵץ וַחֲכַם חֲרָשִׁים! / וְאִם אֵינְךָ שׁוֹמֵעַ לִי / וְלֹא תַקְשִׁיב לְקוֹלִי – / מְהֵרָה תִהְיֶה זוֹ הָאַיָּלָה שְׁלוּחָה / לָאַשְׁמַנִּים כַּמֵּתִים וְיוֹרֶדֶת לְשׁוּחָה״. / וַיֹּאמֶר אֲבִי הַנַּעֲרָה ״אַתָּה מוֹצִיא דִבָּה עַל בִּתִּי הַיְשָׁרָה, / וְאַף עַל פִּי שֶׁהָיִיתָ יוֹדֵעַ יְדִיעָה בְרוּרָה – / הָיָה לְךָ לִהְיוֹת קָלוֹן כּוֹסֶה!״ / וַתַּעַן הַנַּעֲרָה ״אָבִי, הָרוֹפֵא דִּבֶּר אֵלֶיךָ – הֲלֹא תַעֲשֶׂה?״ / וַיִּתְמַהּ הַמְבַקֵּשׁ עַל דְּבָרָיו וְאָמַר: גַּם זֶה פֶלֶא בְעֵינַי / וְלֹא יוּכַל לָדַעַת זֶה כִּי אִם הַנְּבִיאִים מָצְאוּ חָזוֹן מֵיְיָ.
