יְהוּדָה הַלֵּוִי

(מִשִׁירֵי הֵיינֶע)


יְרוּשָׁלַיִם! תִּדְבַּק לְשׁוֹנִי

לְחִכִּי, אִם לֹא אֶזְכְּרֵכִי!

וְתִיבַשׁ יְמִינִי אִם לֹא בְשִֹמְחָתִי

עַל זִכְרוֹנִי אֶעֱלֵךְ, אִם אֶשְׁכָּחֵכִי!


חוּשַׁי יַבִּיעוּ, עַצְמוֹתַי תֹאמַרְנָה

דִבְרֵי הַתְּהִלִּים, מִלָּה בְמִלָּה;

יֵעָרֵב לְאָזְני רַחַשׁ אֲנָשִׁים,

בְּהִשְׁתַּפֵּךְ נַפְשָׁם בְּשִׁירֵי תְפִלָּה!


לְפָנַי יִתְנוֹסֵס גַּן הָעֵדֶן,

תַּעֲלוֹזְנָה כִלְיוֹתַי, יָשִֹישֹ גֵּוִי!

בְּהַדְרַת זְקָנָם הַלָּבָן יְשִׁישִׁים

מִתְהַלְכִים, בְּתוֹכָם יְהוּדָה הַלֵּוִי!


כִּצְלִלים הָרוּחוֹת עָפוּ, חָלָפוּ,

נָסוּ וְאֵינָם, הַלַּיְלָה יָפוּחַ;

בַּתְּמוּנוֹת לְפָנַי אַחַת יָדַעְתִּי,

תְּמוּנַת יְהוּדָה, אִישׁ הָרוּחַ!


הִכַּרְתִיו בְּמִצְחוֹ הֶהָדָר, הָרָחָב;

בְּמַבָּט עֵינָיו הֶעָמוֹק, הַחוֹדֵר;

בְּחִוְרַת לְחָיָיו וּבְגַעֲגוּעֵי הַיָּגוֹן

הָעוֹטְרִים עַל פָּנָיו כְעָנָן-קוֹדֵר!


וְעַל כֻּלָּם הִכַּרְתִּיו בַּשְֹחוֹק,

הַלָּן כְּטַל עַל מֶתֶק שִֹפְתוֹתָיו!

בַּחֵן הַשָּׁפוּךְ עַל פִּתְחֵי פִּיהוּ

נוֹדַע הַמְשׁוֹרֵר, נָתַן אוֹתוֹתָיו!


יָמִים בָּאִים, דּוֹרוֹת הוֹלְכִים,

לְמֵאוֹת עָבְרוּ, חָלְפוּ הַשָּׁנִים,

לְמִן הַיּוֹם יְהוּדָה הַלֵּוִי

עַל פְּנֵי תֵבֵל חָשַֹף פָּנִים!


בִּמְדִינַת קַשְֹטִיל, בְּעִיר תּוֹלָדָה,

הֵנִיעָה אִמּוֹ עֲרִיסָתוֹ!

לְרַגְלֵי גַלֵּי נְהַר תָּגָא,

בְּשִׁיר הִמְתִּיקָה לוֹ שְׁנָתוֹ!


וְאָבִיו אָצַל עָלָיו מֵרוּחוֹ;

מִשְׁחָר יַלְדּוּת הַבֵּן, הוֹרָה

הָאָב לִיהוּדָה דַעַת וְהָבֵן

פֵּרוּשֵׁי מִלּוֹת מִפְּסוּקֵי הַתּוֹרָה!


לְבֶן-חֶמֶד הַלָּז לִמֵּד הָאָב

אֵת אֲשֶׁר קָרָא בַּמִּקְרָאוֹת,

פִּתּוּחֵי נִקּוּד וּטְעָמִים,

עַל הָאוֹתִיּוֹת הַמְרֻבָּעוֹת!


מֵעוֹלַם הַיַּלְדּוּת מְקוֹרָם!

מִימֵי בְרֵאשִׁית הַנְעִימִים!

תֹּאַר הַתֵּבוֹת וְנוֹעַם הַמִּלּוֹת,

לְרוּחַ הַיְלָדִים מַתְאִימִים!


מֵעַל הַסֵּפֶר, זֶה הַיָּשָׁן,

אוֹצַר נֶחְמָד מֵעַתִּיקֵי יָמִים,

קָרָא וְשָׁנָה רַךְ הַנַּעַר

מִשְׁנָתוֹ, בְּנִגּוּן הַטְעָמִים!


“מַהְפַּךְ, פַּשְׁטָא, מֻנַּח אֶתְנַחְתָּא”,

בְּרִנָּה יַטְעִים, וְהַנֶּפֶש מִתְפָּעֶלֶת;

“מֵרְכָא, טִפְּחָא, זָקֵף גָדוֹל”,

כְּצִפּוֹר יַטִּיף, יַאֲרִיךְ הַ“שַׁלְשֶׁלֶת”!


גַם הַתַּרְגוּם הָאֲרַמִּי,

שֶׁל הַבַּלְשָׁן בֶּן אוֹנְקוֹלוֹס

עַל כָּל פַּרְשָׁה שֶׁבַּתּוֹרָה

יִקְרָא, יָשִׁיר בְּקוֹלֵי קוֹלוֹת!


(כִּי קְרוֹבוֹת הֵן הַשָֹּפוֹת,

הָאֲרַמִּית וּלְשׁוֹן הַנְּבִיאִים,

כִּלְשׁוֹן צַח הָאַשְׁכְּנָזִים

והַפְּסוֹלֶת שֶׁל הַשְׁוַבִּיאִים.)


מִנְעָרָיו מַתְמִיד זֶה הַנָּעַר;

רָחוֹק מֵחֲבֵרִים יָשַׁב גַלְמוּד

עַל הַתּוֹרָה וְעַל הָעֲבוֹדָה,

הֲגִיגוֹ אַךְ בְּיַם הַ“תַּלְמוּד”!


וַיַּעַשֹ חַיִל בַּלִּמּוּדִים,

מֵאָב לַבֵּן זֹאת הַבְּרָכָה;

כִּי מִנְעוּרָיו הוֹלִיךְ אוֹתוֹ

מֵחַיִל לְחַיִל אֶל הַ“הֲלָכָה”!


הוֹלִיכוֹ אָבִיו לְבָתֵּי מִדְרַשׁ

הַמְפוֹרְסָמִים; אֶל חֲברוּתָא

שֶׁל הַגְאוֹנִים הַנּוֹדָעִים

בְּבָבֶל וּבְפוּמְפְּדוּתָא!


שָׁם לָמַד הַבְחֵן יָשָׁר

מַה בֵּין הִגָּיוֹן וּמְשַׁל אֵפֶר;

וַאֲשֶׁר יָדַע, מָצָא, חִדֵּשׁ,

כָּתַב,רָשַׁם בּכוּזְרִי הַסֵּפֶר!


עַל הָאֲדָמָה לִבְנֵי הָאָדָם

שֶׁמֶשׁ וְיָרֵחַ, מתְּנוֹת שָׁמָיִם!

מָאוֹר גָּדוֹל לַיּוֹם, וּמָאוֹר

קָטֹן לַלַּיְלָה, מְאֹרוֹת שְׁתָּיִם!


וּמִשְׁנֵה-אוֹר גַּם לַ“תַּלְמוּד”!

הַ“הֲלָכָה”, אוֹרָה חָדָא;

הֲכִי נִכְבָּד, הוּא הַשֵּׁנִי,

הַמָּאוֹר שֶׁל הַ“הֲגָדָה”!


בִּשְֹדֵה הַ“הֲלָכָה” הַכֹּל דָּשִׁין,

בָּהּ יַד מְרִיבֵי שַׁעַר חָלָה;

וְדִבְרֵי הַ“הֲגָדָה” בּנַחַת

נִשְׁמָעִין, לְפַרְדֵס נִמְשָׁלָה!


כְּגַן הַנֶּאֱדָר, גַּן הַפְּלָאוֹת,

גַּן “זְמִירַמִּית” הַגְבִירָה,

הִפְרִיחָה לְיִף תִּפְאָרָה

בְּבָבֶל, עִיר הַבִּירָה!


וַתְּהִי לִזְמִירַמִּית הַמַּלְכָּה

אוֹמֶנֶת יְלָדִים צִפּוֹרָה;

בְּשִֹיחַת צִפֳּרִים הַמַּלְכָּה חָשְׁקָה,

מִפִּיהֶם לָמְדָה תוֹרָה!


וַתִּבֶז בְּלִבָּהּ בְּנֵי הָאָדָם,

הַגָּרִים בִּשְׁפַל הָאֲדָמָה;

וַתַּעַשֹ לָהּ גַן בָּאֲוִיר,

כַּצִּפֳּרִים גָּרֵי רָמָה!


עַל עַמוּדִים גְבוֹהִים יֵרָאוּ

תַּפּוּחֵי חַכְלִיל זְהַב פַּרְוָיִם;

אִילָנֵי דֶקֶל, עֲרוּגֵי פְּרָחִים,

דְיוֹקְנֵי שַׁיִשׁ, מַעַיְנֵי מָיִם!


בְּחָכְמָה הַגְשָׁרִים כּוֹנָנוּ,

מַרְאֵיהֶם כִּזְרוֹעוֹת הֶעָנָף

נְטֻיוֹת בָּאֲוֵיר עַל בְּלִימָה,

וַעֲלֵיהֶן יְקַנֵּן עוֹף-כָּנָף!


מֵהֶם גְדוֹלִים, בַעֲלֵי מַחֲשָׁבוֹת,

מְקוּשָׁטֵי נוֹצוֹת, פְּאֵר גְוָנִים,

וּמֵהֶם קְטַנִּים, רַכֵּי אָפְרֹחִים,

צִפֳּרִים מְזַמְרִים שִׁירֵי רְנָנִים!


נֵרְד וְקִנָּמוֹן, רֵיחַ בְּשָֹמִים,

יַעֲלוּ בְאַף כָּל הַנְשָׁמָה;

מָהוּל בְּאֵדֵי עָשׁ וּבְאָשָׁה,

וְכָל מִינֵי צַחְנַת הָאֲדָמָה!


הַ“הֲגָדָה” הִיא הַפַּרְדֵס;

לַגִּנָּה הַזֹאת הַמְבוֹרָכָה,

רַךְ הַנַּעַר אָץ, בְּאַוָּתוֹ

לְהִמָּלֵט מָאֲבַק הַ“הֲלָכָה”!


לְהִמָּלֵט מֵרִיב לְשׁוֹנוֹת,

מִפִּלְפּוּלֵי הֶבֶל, בִקֵּשׁ עֵצָה;

לְבַל יְאוּנֶה שֶׁבְּיוֹם מוֹעֵד

הַתַּרְנְגוֹלֶת תַּפְלִיט בֵּיצָה!


וְכִי נִכְשַׁל בְּקוּשְׁיָא חוּמְרָא,

בְּחָרוּל מִקִּמְשׁוֹנֵי הַ“הֲלָכָה”;

אֶל פַּרְדֵס הַ“הֲגָדָה” נָס

הַנַּעַר, וְהָיְתָה לוֹ רְוָחָה!


וַהֲרִיחוֹ הַגַּן בְּרֵיחַ צִיצִים,

רֵיחַ שׁוֹשַׁנִּים, אֲבַק פְּרָחִים;

לְקוֹל-יָהּ מִתְהַלֵּךְ בַּגָּן

לָחַשׁ, הַנַּעַר, לְשִׁיר הַמַּלְאָכִים!


לְסִפּוּרֵי נִפְלָאוֹת גְדוֹלוֹת

לָחַשׁ לְמִסְתְּרֵי הָאֱמוּנָה;

כָּל רָז וְסָתוּם לָדַעַת אִוָּה,

מֵאַיִן מְקוֹר הַתְּבוּנָה!


וַיִּכְאַב לֵב הַנַּעַר בְּקִרְבּוֹ,

וַיִּמָלֵא רִגְשׁוֹת נוֹרָאוֹת;

עָרְגָה נַפְשׁוֹ הַנְדִיבָה

לְמֶתֶק גַעֲגוּעֵי הַפְּלָאוֹת!


לְסוֹד שִֹיחַ שַֹרְפֵי-יָהּ,

רָזִין דְרָזִין הַטְמוּנוֹת;

לְרוּחַ שִׁיר פִּיוּט הַנַּעֲלֶה,

הִשְׁתוֹקֵק, לְאוֹצְרוֹתָיו הַצְפוּנוֹת!


וְנִגְלָה לְפָנָיו כָּל הֲדַר

גְאוֹן הַשִּׁירָה הַנְעִימָה!

וַיָּבֶן כִּי רוּחַ אֱלוֹהַּ

פּוֹעֵם בִּלְבָבוֹ פְּנִימָה!


וַיֵּצֵא שֵׁם לִיהוּדָה

הַלֵּוִי, כִּי רוּחוֹ כּבִּירָה

לֹא רַק בִּידִיעַת הַתּוֹרָה,

כִּי גַם בְּחָכְמַת הַשִּׁירָה!


כֵּן, מְשׁוֹרֵר גָּדוֹל בְּדוֹרוֹ

הָיָה, עֲמוּד הָאֵשׁ לְהָאִיר

חֶשְׁכַּת עַמּוֹ בְגָלוּתוֹ,

חַיִּים חֲדָשִׁים בָּם לְהָעִיר!


(כְּעַמוּד הָאֵשׁ וְעוֹז הַשִּׁירָה

שֶׁלִפְנֵי בְנֵי יִשְֹרָאֵל זָרָחוּ,

לַנְחוֹתָם הַדֶּרֶךְ בַּמִּדְבָּר,

עֵת מִמִּצְרַיִם בָּרָחוּ!)


כֻּלָּה טְהוֹרָה וּתְמִימָה,

יָפָה, נִשְֹגָבָה שִׁירָתוֹ!

כִּבְיוֹם בְּרָאוֹ הַיּוֹצֵר,

וַיִּפַּח בּוֹ נִשְׁמָתוֹ!


וְהָאֱלֹהִים מָשַׁךְ חֶסֶד עָלָיו,

וַיִּשָׁקֵהוּ בְּיוֹם בְּרוֹאוֹ!

וּבְכָל שִׁיר תְּצַלְצֵל הַנְשִׁיקָה,

תְּחַיֶה עוֹלָם וּמְלוֹאוֹ!


אִישׁ, אִישׁ לֹא יֶחְטָא

בִּשְֹפָתָיו אוֹ בָּרוּחַ,

אִם מֵאֱלהִים עָלָיו

חוּט שֶׁל חֶסֶד מָתוּחַ!


וַאֲשֶׁר חוֹנַן בְּחֶסֶד אֵל,

מִבֶּטֶן אֵם בְּצֵאתוֹ,

הוּא הַמּוֹלֵךְ עַל כֹּל,

הָעוֹלָם יָחִיל מִשְֹאֵתוֹ!


לְיָהּ לְבַדּוֹ יוֹדֶה הַמְשׁוֹרֵר;

הֲמוֹן הָעָם לֹא יְבִינֵהוּ!

חָלִילָה לוֹ מִשְׁפּוֹט אוֹתוֹ,

אִם גַם הָמֵת יְמִיתֵהוּ!
