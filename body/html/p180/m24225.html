יַשְׁמִיעַ שְׁעִיַּת

יַשְׁמִיעַ שְׁעִיַּת מְגִנָּתִי / בְּשָׁפְכִי תְחִנָּתִי

שִׁמְעָה יְיָ צֶדֶק הַקְשִׁיבָה רִנָּתִי


יַשְׁמִיעַ לֶקַח סְגֻלָּה / בְּלַהֲגִי בִתְפִלָּה

לִשְׁמֹעַ אֶל הָרִנָּה וְאֶל הַתְּפִלָּה


יַשְׁמִיעַ מְעֹז יְסוֹדָהּ / בְּמַלְּלִי בְסוֹדָהּ

מִי הִשְׁמִיעַ זֹאת מִקֶּדֶם מֵאָז הִגִּידָהּ


יַשְׁמִיעַ הוֹד קְהִלָּתִי / בַּהֲכִינִי תְהִלָּתִי

הִרְחַבְתָּ לִי חָנֵּנִי וּשְׁמַע תְּפִלָּתִי


יַשְׁמִיעַ הֲלוּמֵי מִגְמָע / בְּהִתְעַטְּפָם בְּמִדְמָע

הַטֵּה אֱלֹהַי אָזְנְךָ וּשְׁמַע


יַשְׁמִיעַ צָהֳלַת רוֹשׁ גּוֹמֵעַ / בְּצַפְצְפוֹ דוֹמֵעַ

צָעֲקוּ וַיְיָ שָׁמֵעַ


יַשְׁמִיעַ עִנְיַן [חוֹזֶה] / בְּעָרְכִי מַחֲזֶה

עוֹד יִשָּׁמַע בַּמָּקוֹם הַזֶּה


יַשְׁמִיעַ יֹשֶׁר קְ[הָ]לִי / בְּיָסְדִי פִלּוּלִי

יִשְׁמַע מֵהֵיכָלוֹ קוֹלִי


יַשְׁמִיעַ רָם לְהַרְגִּיעֵם / בְּרוּצִי לְהַשְׁמִיעֵם

רְצוֹן יְרֵאָיו יַעֲשֶׂה וְאֶת שַׁוְעָתָם יִשְׁמַע וְיוֹשִׁיעֵם


יַשְׁמִיעַ בִּרְכַּת אֱמוּנַי / בְּבַטְּאִי בְהֶגְיוֹנָי

בָּרוּךְ יְיָ כִּי שָׁמַע קוֹל תַּחֲנוּנָי


יַשְׁמִיעַ רִנַּת וַיִּגְלֹם / בְּרַחֲשִׁי לְעֵילוֹם

רַגְלֵי מְבַשֵּׂר מַשְׁמִיעַ שָׁלוֹם


יַשְׁמִיעַ יָצִיץ מִנִּי חוֹר / בְּיַשְּׁרִי מִבְחוֹר

יַאֲזִין זֹאת יַקְשִׁיב וְיִשְׁמַע לְאָחוֹר


יַשְׁמִיעַ ה צְפִירָתִי / בַּהֲגִיגִי בְדִבְרָתִי

הַט אָזְנְךָ לִי שְׁמַע אִמְרָתִי


יַשְׁמִיעַ ו  קָה / בְּו  די מְעֻשָּׁקָה

וְלֹא יִשָּׁמַע [בָּהּ] עוֹד קוֹל בְּכִי וְקוֹל זְעָקָה


יַשְׁמִיעַ דּ  [תי / בְּד ]   תִי

דִּי                  תִי


יַשְׁמִיעַ ה          בּה
