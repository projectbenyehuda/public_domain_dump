וַיְהִי שָׁלוּחַ

וּבְכֵן וַיְהִי דְבַר יְיָ אֶל יוֹנָה בֶּן אֲמִתַּי לֵאמֹר


וַיְהִי שָׁלוּחַ אֹמֶר נִגְזָר לִשְׁמֹר

אֶל יוֹנָה בֶּן אֲמִתַּי לֵאמֹר


שַׁלִּיט [אֲנִי אֵל חַי וְקַיָּם

מ       אֵדַע אַיָּם

ה]     בְּקַרְקַע הַיָּם קָדוֹשׁ


קוּם לֵךְ אֶל נִינְוֵה לִקְרֹא שָׁם חֲרוֹנָי

כִּי עָלְתָה רָעָתָם לְפָנָי


וַיָּקָם מְשׁוֹרֵר רֶדֶת בָּאֳנִי נְכוֹנָי

לָבוֹא עִמָּהֶם תַּרְשִׁישָׁה מִלִּפְנֵי יְיָ


שַׁדַּי לְךָ מַעֲרָב וּמִזְרָח

הִנְנִי בְיָדֶךָ מַר אֶצְרָח

וְאָנָה מִפָּנֶיךָ אֶבְרָח קָדוֹשׁ


ייְיָ הֵפִיק רוּחַ וְסַעַר עוֹבֵר

וְהָאֳנִיָּה חִשְּׁבָה לְהִשָּׁבֵר


וַיִּירְאוּ בְכֵן הַמַּלָּחִים וַיִּזְעֲקוּ בְסוֹדָם

וְיוֹנָה יָרַד אֶל יַרְכְּתֵי הַסְּפִינָה וַיִּשְׁכַּב וַיֵּרָדַם


וַיִּקְרַב רַב הַחוֹבֵל לֵאמֹר קָרָא בְּקוֹל כָּבֵד

אוּלַי יִתְעַשֵּׁת הָאֱלֹהִים לָנוּ וְלֹא נֹאבֵד


וַיֹּאמְרוּ יוֹעֲצִים לְהַפִּיל גּוֹרָל לְמָ[נָה]

וַיִּפֹּל הַגּוֹרָל עַל יוֹנָה


וַיֹּאמְרוּ הֲלוּ[מִים הַגִּידָה] לָנוּ עָתָּה

מָה אַרְצֶךָ וְאֵי מִזֶּה עַם אָתָּה


וַיֹּאמֶר וּדְעוּ כִּי אֲנִי יָרֵא דָר בְּסוֹד קְדֻשָּׁה

אֲשֶׁר עָשָׂה אֶת הַיָּם וְאֶת הַיַּבָּשָׁה


וַיִּירְאוּ דְבָרוֹ בְּדַבְּרוֹ אֲלֵיהֶם

כִּי מִלִּפְנֵי יְיָ הוּא בוֹרֵחַ כִּי הִגִּיד לָהֶם


וַיֹּאמְרוּ הוֹמִים מַה נַּעֲשֶׂה וְיִשְׁתֹּק גּוֹעֵר

כִּי הַיָּם הוֹלֵךְ וְסוֹעֵר


וַיֹּאמֶר גְּשׁוּ וַהֲטִילוּנִי עִם כְּלֵיכֶם

כִּי בְשֶׁלִּי הַסַּעַר הַגָּדוֹל הַזֶּה עֲלֵיכֶם


וַיַּחְתְּרוּ בְמָשׁוֹט וְלֹא יָכְלוּ חוֹבְלֵיהֶם

כִּי הַיָּם הוֹלֵךְ וְסוֹעֵר עֲלֵיהֶם


וַיִּקְרְאוּ יָהּ אַל נֹאבַד בְּנֶפֶשׁ כִּ[סִּיתָ]

כִּי אַתָּה יְיָ כַּאֲשֶׁר חָפַצְתָּ עָשִׂיתָ


וַיִּשְׂאוּ ר לְהָטִיל נַפְשׁוֹ בְכַפּוֹ

וַיַּעֲמֹד הַיָּם מִזַּעְפּוֹ


וַיִּירְאוּ יִרְאָה גְדוֹלָה הַגְּבָרִים

וַיִּזְבְּחוּ זֶבַח לַיְיָ וַיִּדְּרוּ נְדָרִים


וַיְמַן לִבְלֹעַ צִיר דָּג [בַּמְּצוּ]לוֹת

וַיְהִי יוֹנָה בִּמְעֵי הַדָּג שְׁלֹשָׁה יָמִים וּשְׁלֹשָׁה לֵילוֹת.
