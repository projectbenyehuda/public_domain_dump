אֱנוֹשׁ חָצִיר

אֱנוֹשׁ חָצִיר יִיבַשׁ / וּבַמָּוֶת נִכְבָּשׁ / וְגוּשׁ רִמָּה יִלְבַשׁ / וְיוּבַל לִקְבָרִים

וְלַמָּוֶת נִצָּב / וְאֵלָיו הוּא נֶחְצָב / וְסוֹפוֹ הַנִּקְצָב / לְעַפְרוֹת מִדְבָּרִים

וְהַחַי הַנִִּגְבָּר / עֲלֵי מַה יִּתְגַּבָּר? / וּמָחָר לַמִּדְבָּר / יְהוּשַׂם מִדָּרִים

וְלֹא יוֹעִיל הוֹנוֹ / וְחֵילוֹ וַהְמוֹנוֹ / וְעוֹד כִּמְעַט אֵינוֹ / בְּבַיִת וּמְדוֹרִים

וְלֹא יוֹעִיל קָרוֹב / וְלֹא מָמוֹן לָרֹב / וְהַמָּוֶת יֶאְרֹב / לְחָטְפוֹ כִנְמֵרִים

וְלֹא יוֹעִיל מַטְמוֹן / וְלֹא יַעְזֹר הָמוֹן / וְלֹא יֶתֶר אָמוֹן / בְּיוֹם מָוֶת עוֹזְרִים

וְגִבּוֹר הַנֶאְזָר / לְיוֹם מָוֶת נֶחְזָר / וְלַקָּרוֹב יֻכְזַר / וְיִהְיוּ לוֹ זָרִים

וְהַחַי הַנִּמְעָד / וְלַקֶּבֶר מוּעָד / וְלֹא יִמְצָא מִסְעָד / לְנַפְשׁוֹ מִבּוֹרִים

וְזֶה הוּא הַמִּקְרֶה / לְכָל אָדָם יִקְרֶה / כְּבָרַחַת נִזְרֶה / יְהוּבַל מִן זוֹרִים

וְאָדָם הַגִּבּוֹר / עֲדֵי מָתַי יִגְבֹּר / וּבֶעָפָר לִקְבֹּר / וְיוּסַר כַּגֵּרִים

וְהוֹדִיעָם גֻּדְלוֹ / כְּבָר אֵין הוֹעִילוֹ / וְלֹא יוּכַל גָּאְלוֹ / בְּיוֹם אֵיד וּשְׁבָרִים

וְלֶעָפָר נִשְׁבָּר / וְיוֹם מָחָר נִקְבָּר / וְיַעְבֹר הַמַּעְבָּר / וְיִהְיֶה מַעְבָרִים

וְיִשָּׁכַח זִכְרוֹ / בְּדוֹרוֹ וּמְדוֹרוֹ / וְרֵעוֹ וּשְׁאֵרוֹ / בְּלִי לִשְׁמוֹ זוֹכְרִים

וְיֵאָמַר אַיּוֹ / בְּעֵת יִקַּח דַּיּוֹ / וְלִשְׁכֹּן אֶת עִיּוֹ / בְּרִגְבֵי מִסְתָּרִים

וְיוּסַר הָאַבִּיר / אֲשֶׁר הָיָה כַבִּיר / וְעַל כָּל מַכְבִּיר / בְּחַיָּיו לִדְבָרִים

וְתֹאבַד תּוֹחֶלֶת / וְתָסוּר מֶמְשֶׁלֶת / לְעוֹלָם אִוֶּלֶת / וְדִבְרֵי נַעְתָּרִים

וְשֶׁהָיָה כוֹכָב ־ / וְהַיּוֹם הִנֵּה כָב / וְעוֹד אֵינוֹ נֶעְכָּב / וְנֶעְתָּק לַעְפָרִים

וְהָעָם הָרוֹהִים / בְּפַחְדּוֹ וּתְמֵהִים / עֲלֵי יוֹמוֹ נוֹהִים / וְהוֹמִים נִשְׁבָּרִים

וְסָרוּ הַבִּינוֹת / וּבָטְלוּ חִשְּׁבוֹנוֹת / וְאָבְדוּ עֶשְׁתּוֹנוֹת / אֲשֶׁר הָיָה מַעְרִים

וּבָשַׁל הַקָּצִיר / כְּגֶפֶן הַבָּצִיר / וְנִמְשָׁלִים חָצִיר / בְּמַגָּל נִקְצָרִים

וְחִילִים עִם צִירִים / אֲחָזוּם וּמְצָרִים / וְאֵינָם נֶעְצָרִים / וְגַם לֹא נִבְצָרִים

וְגִבּוֹר הַבָּצוּר / כְּלוּ הָיָה נָצוּר / וְכָאֶשְׁכּוֹל בָּצוּר / וְלָזֶה נוֹצָרִים

וְהַכֹּל הוּא בוֹלֶה / וְיָמוּת גַּם יִכְלֶה / וְיָסוּר לַכֶּלֶא / וְיֵשֵׁב דּוֹר דּוֹרִים

וְיִשְׁתֶּה תַרְעֵלוֹת / וְיִפֹּל בִּנְפִילוֹת / וְלֹא יוּכַל לַעְלוֹת / בְּיוֹם אֵיד וּמְזוֹרִים

וְיֵלֵךְ לִתְעָלוֹת / רְגָבִים וּמְחִילוֹת / וּמָזוֹר וּתְעָלוֹת / לְחָלְיוֹ נֶעְדָּרִים

מְהֵרָה יִשְׁתּוֹלֵל / כְּגֶפֶן יִתְעוֹלֵל / כְּדֶשֶׁא יִתְמוֹלֵל / וְיִשְׁכַּב בַּחוֹרִים

וְאַיֵּה הָעַמִּים / וְאֵי מִשְׁפְּחוֹת קְדוּמִים / וְאַיֵּה הָאֵימִים / וְאַיֵּה גִבּוֹרִים?

וְלֹא מָצְאוּ פִדְיוֹם / בְּיוֹם מָוֶת אָיוֹם / וּמִמּוֹקְשֵׁי הַיּוֹם / כְּלֵיהֶם נֶעְדָּרִים

יְדִידִי, מַה תִּכְאַב / וְעַל מַה זֶּה תִדְאַב / וְתִבְכֶּה עַל הָאָב / בְּמִסְפַּד תַּמְרוּרִים

וְתִשָּׂא בִילָלָה / בְּיוֹמָם וָלַיְלָה / וְגַם קוֹל כַּחוֹלָה / וְדִמְעָתָךְ תַּזְרִים

וְעַד מָתַי תָּחִיל / בְּמוֹ צִירִים וְחִיל? / וְדֹם לְאֵל הוֹחִיל / אֱלֹהֵי הַיְצוּרִים

יְנַחֵם לִבֶּךָ / וְיַרְבֶּה טוּבֶךָ / וְיָרִים אָבִיךָ / בְּחֶבְרַת הַיְשָׁרִים

וְיַגִּיהַּ צָהְרוֹ / וְיַדְלִיק אֶת נֵרוֹ / וְאָרְחוֹ יַזְהִירוֹ / כְּנֹגַהּ הַמְּאוֹרִים.
