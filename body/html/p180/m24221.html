יְמַלֵּל אֹמֶץ

וּבְכֵן יְמַלֵּל


יְמַלֵּל אֹמֶץ אֱלֹהוּתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֲבָאֵר אֲבָרֵךְ / שִׁמְךָ וּבוֹ אֶחְתֹּם

יְמַלֵּל גֹּדֶל גְּבוּרָתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֶדְגֹּל אֶדְרֹשׁ / שִׁמְךָ וּבוֹ אֶחְתֹּם

יְמַלֵּל הוֹד הַדְרָתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אוֹקִיר אוֹדִיעַ / שִׁמְךָ וּבוֹ אֶחְתֹּם

יְמַלֵּל זֹהַר זִמְרָתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֲחַקֵּר אֲחַסֵּן / שִׁמְךָ וּבוֹ אֶחְתֹּם

יְמַלֵּל טֶכֶס טָהֳרָתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֲיַשֵּׁר אֲיַקֵּר / שִׁמְךָ וּבוֹ אֶחְתֹּם

יְמַלֵּל כֶּתֶר כִתְרָתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֲלַהֵג אֲלַמֵּד / שִׁמְךָ וּבוֹ אֶחְתֹּם

יְמַלֵּל מְעֹז מֶמְשַׁלְתְּךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֲנַצַּח אֲנַשֵּׂא / שִׁמְךָ וּבוֹ אֶחְתֹּם

[יְמַלֵּל ס ס לָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אע אע / שִׁמְךָ וּבוֹ אֶחְתֹּם]

יְמַלֵּל פְּאֵר פְּעֻלָּתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֲצַדֵּק אֲצַפֶּה / שִׁמְךָ וּבוֹ אֶחְתֹּם

יְמַלֵּל קֹשְׁטְ קִרְבָתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֲרַחֵשׁ אֲרוֹמֵם / שִׁמְךָ וּבוֹ אֶחְתֹּם

יְמַלֵּל שֶׁנֶן שִׁירָתְךָ / הַכֹּל וְלֹא יִתֹּם / וַאֲנִי אֲתַכֵּן אֲתַנֶּה / שִׁמְךָ וּבוֹ אֶחְתֹּם.
