יוֹנֵי גֵאָיוֹת

יוֹנֵי גֵאָיוֹת / הוֹמוֹת בְּפַחְדָּן,

גָּלָה מַחְמַדָּן / וְנִקְלָה נִכְבַּדָּן.

יִתְרָן וְחֶמְדָּן – / לְיִתְרָן וְחֶמְדָּן,

לְהֶרֶג וְאַבְדָּן / הֻצְּבוּ לְבַדָּן.

זֶה אֶלֶף שָׁנִים / צְנִינִים בְּצִדָּן,

כִּי לְדִינָן אֵין דָּן / וְיִסְפְּרוּ כָל חֶלְדָּן

עִדָּנִין וְעִדָּן / וּפְלַג עִדָּן.


הָחְשְׁכוּ נִשְׁפֵיהֶם / וְדֹעֲכוּ רִשְׁפֵּיהֶם

וְרָפוּ כַנְפֵיהֶם – / וְקַלּוּ רוֹדְפֵיהֶם,

וְנֹעֲרוּ כַפֵּיהֶם / מִכְּבוֹד כַּסְפֵּיהֶם

וּבַאֲשֶׁר אַפְאֵיהֶם / תַּנִּים שׁוֹאֲפֵיהֶם.

עֲרָב לִנְשִׂיאֵיהֶם, / אֱדוֹם לְאַלּוּפֵיהֶם,

וּמֶשֶׁךְ וּפְלֶשֶׁת / נָתְנוּ עַל פִּיהֶם –

וְנִבְדְּלוּ הֵם לְבַדָּן / וְאֵין עוֹזֵר כְּנֶגְדָּן.


וּנְבָיוֹת וְכוּשִׁים / רָדוּ בַחֲמוּשִׁים

וְהִנִּיחוּם רְטוּשִׁים, / בְּכָל פִּנָּה נְטוּשִׁים.

גְּדוֹלֵיהֶם דְּרוּשִׁים / וּבָתֵיהֶם בְּלוּשִׁים

וְנָשִׁים וַאֲנָשִׁים / בְּכָל רֶגַע עֲנוּשִׁים.

וּמֵרֹב הַכְּחָשִׁים / דְּלָקוּם נְחָשִׁים

וְהִנִּיחוּם חֲבוּשִׁים / וּבוּשָׁה מְלֻבָּשִׁים –

נְתוּנָה עַל מַדָּן / כְּמוֹ עוֹרָן וְעִדָּן.


דִּבְרֵי הַיָּמִים / נִפְתְּרוּ לַחֲכָמִים –

וְקִצִּים נַעֲלָמִים / סְתוּמִים וַחֲתוּמִים,

וְאֵין נְבִיאִים וְחוֹזִים / וְאֵין אוּרִים וְתֻמִּים,

וְצֶדֶק עוֹלָמִים / נֶחֱלַף בִּקְסָמִים.

וּבְנוֹת מְלָכִים / שְׁפָחוֹת לָעַמִּים,

פָּרְקוּ הַנְּזָמִים / וְזָרְקוּ הַבְּשָׂמִים –

וְעַבְדָּן יִתְעַדָּן / בְּקִנְמוֹנָן וְנִרְדָּן.


הַעֲבֵר טַבַּעְתָּם / מֵעַל יַד גַּעְתָּם

וּמֵעַל רֹאשׁ קֵדָר – / אֶת פְּאֵר מִגְבַּעְתָּם,

וְאוֹהֲבֵי מַפַּלְתָּם – / הַשְׁקֵם קֻבַּעְתָּם.

וְגַלֵּה קֵץ נֶחְתָּם / וְהָאֵר אוֹר נֶעְתָּם

וְדִגְלְךָ הָרֵם עַל / דִּגְלֵי מִשְׁמַעְתָּם,

וּמַחֲנֵה כֹהֲנֶיךָ / תְּנָה בְּתוֹךְ אַרְבַּעְתָּם –

וְיַצִּיעוּ מַרְבַדָּן / מִבְּאֵר צֶבַע וְעַד דָּן.
