[האשה שלא עמדה במבחן]

אָמְרוּ כִי מֶלֶךְ אֶחָד מִמַּלְכֵי הָעֲרָב הָיָה חָכָם וְנָבוֹן וּבַעַל מִשְׁפָּט, וְהָיוּ חֲכָמָיו וּנְבוֹנָיו / יוֹשְׁבִים תָּמִיד לְפָנָיו. / וְיוֹם אֶחָד הִתְחִילוּ לְסַפֵּר בְּשֶׁבַח הַנָּשִׁים וּגְדֻלָּתָן / וְרֹב חָכְמָתָן וְסָכְלְתָנוּתָן. / וַיֹּאמֶר לָהֶם הַמֶּלֶךְ "קַצְּרוּ דִבְרֵיכֶם / וְעִצְרוּ מִלֵּיכֶם, / כִּי מֵעוֹלָם לֹא נִרְאֲתָה וְלֹא נִשְׁמְעָה / אִשָּׁה טוֹבָה וּכְשֵרָה, בַּעֲלַת שֵׂכֶל וְדֵעָה, / וְכָל אַהֲבָתָן לְפִי הֲנָאָתָן וְצֹרֶךְ שְׁעָתָן / וְאֵינָן מוֹשְׁלוֹת בְּתַאֲוָתָן / וְחוֹטְאוֹת בְּנַפְשׁוֹתָן ". / וַיֹּאמְרוּ לוֹ חֲכָמָיו “אַל יֹאמַר הַמֶּלֶךְ כֵּן, כִּי יֵשׁ נָשִׁים חֲכָמוֹת וּנְבוֹנוֹת, / כְּשֵׁרוֹת וְנֶאֱמָנוֹת, / אוֹהֲבוֹת וּמְכַבְּדוֹת בַּעֲלֵיהֶן, / מְכַסּוֹת עֶרְוַת בָּתֵּיהֶן / וּמְפַרְנְסוֹת בְּנֵיהֶן וּבְנוֹתֵיהֶן, / וְכָל מוּם וְדִבָּה אֵין בָּהֶן”.

וַיֹּאמֶר הַמֶּלֶךְ “אֲנִי אֶתֵּן לָכֶם אוֹת עַל דְּבָרַי / וּמוֹפֵת עַל מַאֲמָרַי: / הִנֵּה הָעִיר הַזֹּאת גְּדוֹלָה מְאֹד, שׁוֹטֲטוּ וּרְאוּ אִם תִּמְצְאוּ בָהּ אִשָּׁה אַחַת בְּכָל הַטּוֹבוֹת אֲשֶׁר אֲמַרְתֶּם / וְהַמִּדּוֹת אֲשֶׁר זְכַרְתֶּם”. / וַיְבַקְשׁוּ וַיִּמְצְאוּ בָהּ אִשָּׁה אַחַת בַּעֲלַת שֵׁם, צְנוּעָה וּכְשֵׁרָה וַחֲכָמָה / יָפָה כַלְּבָנָה בָּרָה כַּחַמָּה, / וּבַעֲלָהּ אִישׁ עָשִׁיר גָּדוֹל וְסוֹחֵר / מְבַקֵּשׁ רָצוֹן וְטוֹב שׁוֹחֵר. / וַיָּשִׁיבוּ אֶל הַמֶּלֶךְ דָּבָר וַיֹּאמְרוּ לוֹ “הִנֵּה מָצָאנוּ אִשָּׁה חֲכָמָה / וּמַשְׂכֶּלֶת בְּלִי מוּם וּמִרְמָה / וּבְלִי מֶרֶד וָמַעַל, / וְהִיא בְּעוּלַת בַּעַל”. / וַיִּשְׁלַח הַמֶּלֶךְ לִקְרֹא לְאִישָׁהּ, וַיָּבוֹא לְפָנָיו וַיְכַבְּדֵהוּ / וַיַּרְחִיב לוֹ מָקוֹם וַיּוֹשִׁיבֵהוּ. / וַיֹּאמֶר לוֹ “דְּבַר סֵתֶר לִי אֵלֶיךָ”. / וַיֹּאמֶר לוֹ “דַּבֵּר אֱלֵי עַבְדֶּךָ, / כִּי יִשְׁמַע לְקוֹל מִלֶּיך ָ”. / וַיֹּאמֶר לוֹ הַמֶּלֶךְ “יֵשׁ לִי בַּת טוֹבָה וַחֲמוּדָה / וְהִיא לִי רַק יְחִידָה, / וְלֹא אֶחְפֹּץ לְתִתָּהּ לְמֶלֶךְ אוֹ לְשַׂר, / כִּי חִשְׁקָהּ בְּלִבִּי נֶאֱסָר, / רַק אֲבַקֵּשׁ אִישׁ טוֹב וְנֶאֱמָן / וְזַךְ מִכָּל חֶלְאַת הַזְּמָן / יֶאֱהָבֶנָּה וִיכַבְּדֶנָּה / וְעַל שִׁכְמוֹ יִשָּׂאֶנָּה / וַעֲטָרוֹת יַעַנְדֶנָּה / וּבְסֵבֶר פָּנִים יְקַבְּלֶנָּה / וּבְחֶסֶד וּבְרַחֲמִים יְכַלְכְּלֶנָּה. / וַאֲנִי שָׁמַעְתִּי עָלֶיךָ וְעַל רֹב מַהֲלָלֶיך ָ / וְטוּב מִדּוֹתֶיך ָ וּמַעֲלָלֶיךָ, / וְאֶרְצֶה לְתִתָּהּ לְךָ לְאִשָּׁה – / אֲבָל לֹא אוּכַל לְתִתָּהּ לְבַעַל אִשָּׁה. / הֲרֹג אֶת אִשְׁתְּךָ הַלַּיְלָה וּמָחָר אֶתְּנֶנָּה לְךָ לְאִשָּׁה!” / וַיֹּאמֶר לוֹ הַסּוֹחֵר “אֲדוֹנִי הַמֶּלֶךְ, מִי אֲנִי וּמַה חַיַּי, כִּי אֶהְיֶה חָתָן לַמֶּלֶךְ – וְאֵינֶנִּי רָאוּי לִרְעוֹת צֹאנֶךָ?” וַיֹּאמֶר לוֹ “אֲנִי אֶבְחַר בְּךָ, כִּי אַתָּה מַאֲוַיִּי וְחֶפְצִי, / וּבִלְעָדֶיךָ לֹא יָרִים אִישׁ אֶת יָדוֹ וְאֶת רַגְלוֹ בְּכָל אַרְצִי”. / וַיֹּאמֶר לוֹ “אֲדוֹנִי הַמֶּלֶךְ, אֵיךְ אוּכָל לַהֲרֹג אֶת אִשְׁתִּי? / וְהִיא חֲמֵשׁ עֶשְׂרֵה שָׁנָה אִתִּי, / מִפִּתִּי תֹאכַל וְתִשְׁתֶּה מִכּוֹסִי, / וְהִיא שִׂמְחַת לִבִּי וּמְשׂוֹשִׂי, / תֶּאֱהָבֵנִי וּתְכַבְּדֵנִי / וְתִשְׁמְרֵנִי וְתַעַבְדֵנִי, / וּכְבוֹדִי בְּכָל יוֹם בְּעֵינֵיהָ יוֹסִיף וְלֹא יִגָּרַע, / וְנִשְׁמַרְתִּי מִכָּל דָּבָר רַע”. / וַיֹּאמֶר לוֹ הַמֶּלֶךְ “שְׁמַע בְּקוֹלִי וַהֲרֹג אוֹתָהּ,/ כִּי כָבוֹד גָּדוֹל יִהְיֶה לְךָ בְמוֹתָהּ: / כִּי עַל כָּל שָׂרֵי מַלְכוּתִי אָרִים רֹאשְׁךָ / וּמָשַלְתָּ בְּכֹל אֲשֶׁר תְּאַוֶּה נַפְשֶׁךָ”. / וַיֹּאמֶר הָאִישׁ “אֲנַסֶּה וְאֶחֱזֶה / אִם אוּכַל לַעֲשׂוֹת אֶת הַדָּבָר הָרַע הַזֶּה”.

וַיֵּצֵא מֵעִם הַמֶּלֶךְ סָר וְזָעֵף / וְדַם כְּבֵדוֹ מִנְּחִירוֹ רוֹעֵף. / וַיָּבוֹא אֶל בֵּיתוֹ / וּבִרְאוֹתוֹ אֶת אִשְׁתּוֹ / גָּדְלָה אַנְחָתוֹ / וְנִכְפְּלָה דַאֲגָתוֹ. / וַתֹּאמֶר לוֹ אִשְׁתּוֹ “אֲדוֹנִי הַטּוֹב וְהַנָּעִים, / מַדּוּעַ פָּנֶיךָ רָעִים?” / וַיִּתְאַפַּק וַיֹּאמֶר אֵלֶיהָ “צוּרִי יִשְׁמְרֵך ְ מִמָּוֶת וְקָטֹב, / אֵין לִי דָבָר וְאֵין בְּלִבִּי רַק טוֹב”. / וַיְהִי בַלַּיְלָה, וְהִיא יְשֵׁנָה עַל מִטָּתָה ּ, / קָם בְּתִמְהוֹן לֵב לַהֲמִיתָהּ. / וַיִּקַּח הַחֶרֶב בִּימִינוֹ וְהַנֵּר בִּשְׂמֹאלוֹ וַיָּסַר אֶת הַכְּסוּת מֵעָלֶיהָ – / וַיִּרְאֶנָּה יְשֵׁנָה וּשְׁנֵי בָנֶיהָ שׁוֹכְבִים בֵּין שָׁדֶיהָ, וַיַּחְמֹל עָלֶיהָ / וַיֹּאמַר “אוֹי לִי, אֵיךְ אֶהֶרְגֶנָּה / וְחֶרְפָּתִי אָנָה אוֹלִיכֶנָּה? / וּמִי יְגַדֵּל אֶת בָּנַי / אֲשֶׁר הֵם כְּאִישׁוֹן עֵינַי? / אֵין זֶה כִּי אִם רֹב פְּשָעַי וַעֲוֹנַי!” / וַיָּשֶׁב הַחֶרֶב אֶל נְדָנָה ּ, / וְנַפְשׁוֹ מִתְמוֹגֶגֶת בִּיגוֹנָהּ, / וַיֹּאמֶר בְּלִבּוֹ – וְעֵינָיו בּוֹכוֹת: / הִנֵּה אִשְׁתִּי טוֹבָה מִכָּל הַמַּמְלָכוֹת. / אֲרוּרִים יִהְיוּ כָל הַמְּלָכִים / כִּי הֵם אַחַר תַּאֲוָתָם הוֹלְכִים / וְלִבּוֹת בְּנֵי אָדָם בְּהַבְלֵיהֶם מוֹשְׁכִים / וּבְיֵין שִׂמְחָתָם מֵי אֲנָחָה נוֹסְכִים! / וַיַּעַל עַל הַמִּטָּה ויִּשְׁכַּב עִמָּהּ וַיְנַשְּׁקָהּ, / וַיָּשֶׂם שְׂמֹאלוֹ תַּחַת לְרֹאשָׁהּ וַיְחַבְּקָהּ.

וַיְהִי בַּבֹּקֶר וַיָּקָם וַיֵּלֵךְ לַחֲנוּתוֹ / לַעֲשׂוֹת מְלַאכְתּוֹ, / וְהַמֶּלֶךְ מְיַחֵל אוֹתוֹ. / וַיִּתְמַהְמַהּ, וַיִּשְׁלַח הַמֶּלֶךְ לַמַּעֲנֵהוּ / לָבוֹא לְפָנָיו וּלְמַהֲרֵהו ּ / וְלָדַעַת מִקְרֵהוּ. / וְכִרְאוֹתוֹ שְׁלִיחַ הַמֶּלֶךְ גָּדַל כְּאֵבוֹ / וּפָחַד וְרָהָה לְבָבוֹ. / וַיָּבוֹא לִפְנֵי הַמֶּלֶךְ, וַיֹּאמַר לוֹ “מֶה עָשִׂיתָ – הָרַגְתָּ אוֹתָהּ?” / וַיֹּאמַר “לֹא אֲדוֹנִי, כִּי גָדְלָה עָלַי אַהֲבָתָהּ וְחֶמְלָתָהּ”. / וַיְסַפֶּר לוֹ אֵת כָּל אֲשֶׁר קָרָהו ּ בְּקַחְתּו ֹ אֶת הַחֶרֶב לְהַכּוֹתָהּ. / וַיִּגְעַר בּוֹ הַמֶּלֶךְ וַיֹּאמֶר לוֹ " צֵא מִנֶּגֶד עֵינַי / אַל תּוֹסֶף רְאוֹת פָּנַי, / כִּי אֵינְךָ מִן הָאֲנָשִׁים / וְלִבְּךָ כְּלֵב הַנָּשִׁים“. / וַיֵּצֵא מִלְפָנָיו, וַתֵּרֶב שִׂמְחָתוֹ / לְזַעַף הַמֶּלֶךְ וְגַעֲרָתוֹ. וַיְהִי מִמָּחֳרָת לְעֵת עֶרֶב שָׁלַח הַמֶּלֶךְ אֶחָד מֵעֲבָדָיו לְהָבִיא אֶת אֵשֶׁת הָאִישׁ בַּסֵּתֶר, / וַיֵּלֶךְ וַיְבִיאֶהָ אֵלָיו. וְכִרְאוֹת הַמֶּלֶךְ אוֹתָהּ / תָּמַהּ עַל הוֹד תְּמוּנָתָהּ / וַהֲדַר תָּאֳרָהּ וּדְמוּתָהּ, / וַיֹּאמֶר אֵלֶיהָ: הִנֵּה שָׁמַעְתִּי עַל חָכְמָתֵךְ / וְעַל יְפִי יִפְעָתֵךְ, / וּמִיּוֹם שָׁמַעְתִּי שִׁמְעֵךְ קָדְחָה בְלִבִּי אֵשׁ אַהֲבָתֵךְ / וְנִמְשְׁכָה נַפְשִׁי בְּחַבְלֵי תַאֲוָתֵך ְ, / וְעַתָּה חָפַצְתִּי לְקַחְתֵּךְ לִי לְאִשָּׁה / וְהַמַּלְכוּת תִּהְיֶה לָךְ לְמוֹרָשָׁה. / אָמְנָם לֹא אוּכַל קַחַת אֵשֶׁת אִישׁ, פֶּן יִהְיֶה לְבוּשָׁה. / לְכִי הַלַּיְלָה וְהִרְגִי אֶת אִישֵׁךְ – / וְאַחֲרֵי-כֵן אֶקָּחֵךְ וְאֶעֱשֶׂה כָּל אֲשֶׁר תְּאַוֶּה נַפְשֵׁךְ”. / וַתַּעַן הָאִשָּׁה בְּטוּב-לֵב וּבְשִׂמְחָה / “אֶעֱשֶׂה מִצְוַת הַמֶּלֶךְ וְאֶהְיֶה לוֹ לְשִׁפְחָה”. / וַיֹּאמֶר לָהּ הַמֶּלֶךְ “אֲבָל תִּהְיִי אִשְׁתִּי וְכָל נָשַׁי תִּהְיֶינָה אַמְהוֹתַיִך ְ / וּפִילַגְשַׁי – שִׁפְחוֹתַיִךְ”. / וַיִּתֶּן לָהּ הַמֶּלֶךְ חֶרֶב בְּדִיל לְטוּשָׁה, / כִּי יָדַע חֶסְרוֹן דַּעַת הָאִשָּׁה, / וַיֹּאמֶר לָהּ “הִנֵּה חַרְבִּי חֶרֶב חַדָּה מְאֹד - / הַכִּי אוֹתוֹ פַעַם אַחַת וְלֹא תִשְׁנִי לוֹ / וְלֹא יוּכַל עוֹד קוּם לוֹ”.

וַתִּקַּח הַחֶרֶב וַתֵּלֶךְ לְבֵיתָהּ / וְהִיא מִתְרַפֶּקֶת בְּשִׂמְחָתָהּ, / וְטָבְחָה טִבְחָה, אַף מָסְכָה יֵינָהּ / וַתָּבֵא מַטְעַמִּים וַתַּעֲרֹךְ שֻׁלְחָנָהּ. / וַיָּבוֹא הָאִישׁ אֶל בֵּיתוֹ / לֶאֱכֹל וְלִשְׁתּוֹת עִם בָּנָיו וְעִם אִשְׁתּוֹ, / וַתַּשְׁקֵהוּ יַיִן בַּלַּיְלָה הַהוּא וַתְּשַׁכְּרֵהוּ / וַתִּשָּׂאֵהו ּ אֶל הַמִּטָּה וַתַּשְׁכִּיבֵהוּ. / וַתָּקָם בְּעוֹד לַיְלָה, וְהוּא בְשִׁכְרוּתוֹ / יָשֵׁן עַל מִטָּתו ֹ, / וַתִּקַּח הַחֶרֶב וַתַּךְ עַל רֹאשׁוֹ / וַתַּחְשֹׁב כִּי בַרְזֶל בָּאָה נַפְשׁוֹ – / וַתִּכָּפֵל חֶרֶב הַבְּדִיל וַיֵּעוֹר מִשְּׁנָתוֹ / וַיִּנָּעֵר מְעַט מִשִּׁכְרוּתוֹ. / וַיִּצְעַק וַיֹּאמֶר לְאִשְׁתּוֹ “מִי מִשְּׁנָתִי הֶעִירַנִי / וְעַל רֹאשִׁי מִי הִכַּנִי?” / וְכִרְאוֹת הָאִשָּׁה כִּי הֵקִיץ וְלֹא הִזִּיקַתְהו ּ חַרְבָּהּ, / חָרְדָה חֲרָדָה גְדוֹלָה וַיֵּצֵא לִבָּהּ / וּמֵרֹב פַּחַד וְאֵימָה / כִּמְעַט לֹא נוֹתְרָה בָּהּ נְשָׁמָה. / וַתֹּאמֶר לוֹ “אֲהוּבִי, יְדִיד נַפְשִׁי, שְׁכַב וְשׁוּב לְנוּמֶךָ – / אוּלַי מַכֶּה אוֹתְךָ רָאִיתָ בַחֲלוֹמֶךָ”. / וַיָּשָׁב אֶל שְׁנָתוֹ וְלֹא יָכֹל לִפְקֹחַ עֵינוֹ, / כִּי עוֹד לֹא הֵקִיץ מִיֵּינוֹ. / וַתֵּרֶב דַּאֲגַת הָאִשָּׁה עַד אֵין חֵקֶר / וַתִּתְחַלְחַל עַד אוֹר הַבֹּקֶר. / וַתַּשְׁכֵּם לַעֲשׂוֹת מַעֲשֶׂיהָ וּלְתַקֵּן אֶת בֵּיתָהּ / לְכַסּוֹת עֶרְוָתָהּ מֵהַמַּעֲשֶׂה הָרַע אֲשֶׁר עָשְׂתָה / וְלַעֲשׂוֹת לְבַעְלָהּ מַטְעַמִּים / כִּי כֵן מִשְׁפָּטָהּ כָּל הַיָּמִים.

וַיְהִי הַמֶּלֶךְ מְיַחֵל עֵת בּוֹאָהּ, / וְלֹא בָאָה עַד אֲשֶׁר צִוָּה לַהֲבִיאָהּ. / וַיְהִי כְבוֹאָהּ וַיֹּאמֶר לָהּ הַמֶּלֶךְ “עָשִׂית אֶת הַדָּבָר אֲשֶׁר הָיָה בֵּינִי וּבֵינַיִך ְ / אוֹ חָסוּ עָלָיו עֵינַיִךְ?” / וַתֹּאמַר “אֲדוֹנִי הַמֶּלֶךְ, אֲנִי עָשִׂיתִי מִצְוָתֶך ָ – / אָכֵן אַתָּה שַׂמְתָּ לְאַל מְגַמָּתֶךָ / וְהֵפַרְתָּ עֲצָתְךָ הַטּוֹבָה וַתְּסַכְּלֶהָ, / כִּי חַרְבְּךָ הֵנַפְתִּי עָלָיו – וַתְּחַלֲלֶהָ.” / וַתְּסַפֵּר לוֹ אֵת אֲשֶׁר עָשְׂתָה / וְאֶת מְעֻוָּתָה ּ אֲשֶׁר עִוְּתָה. / וַיִּשְׁלַח הַמֶּלֶךְ לִקְרֹא לְאִישָׁהּ וַיֹּאמֶר לוֹ “סַפֵּר לַחֲכָמִים הַיּוֹשְבִים לְפָנַי אֵת כָּל אֲשֶׁר קָרְך ָ עִם אִשְׁתֶּךָ, אַל תְּכַחֵד מֵהֶם דָּבָר!” וַיְּסַפֵּר לָהֶם אֵת כָּל הַמּוֹצְאוֹת אוֹתוֹ עִם אִשְׁתּוֹ וְאַחֲרֵי כֵן צִוָּה הַמֶּלֶךְ אֶת הָאִשָּׁה לְהַגִּיד לָהֶם אֵת כָּל אֲשֶׁר קָרָה לָהּ עִם בַּעֲלָהּ וַתַּעַשׂ כֵּן. וַיַּעַן הַמֶּלֶךְ לַחֲכָמָיו "הֲלֹא זֶה דְבָרִי אֲשֶׁר דִּבַּרְתִּי אֲלֵיכֶם / בְּאָמְרִי לָכֶם "עִצְרוּ מִלֵּיכֶם ". / וַיַּעֲנוּ “כֵּן” וְהֵם מֵחָכְמָתוֹ תְמֵהִים / “מַה נֹּאמַר – וַאדוֹנִי חָכָם כְּמַלְאַךְ הָאֱלֹהִים!”
