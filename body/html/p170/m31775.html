נֹפֶת שְׂפָתַיִם

אל אבו־איוב [שלמה] בן אלמעלם ואחיו


נֹפֶת שְׂפָתַיִם וְיֵין שִׁנָּיִם / אוֹ צִיץ לְחָיַיִם וּמֹר אַפָּיִם,

אוֹ נָשְׁבוּ רוּחוֹת נְעוּרִים מִפְּאַת / מִכְתַּב שְׁלֹמֹה תַּאֲוַת עֵינָיִם,

מִכְתָּב כְּזֹהַר יוֹם בְּמַרְאֵהוּ אֲבָל / פָּרַשׂ דְּיוֹ עָלָיו כְּסוּת עַרְבָּיִם.

סֵפֶר כְּעֵין סַפִּיר בְּנֹפֶךְ פֻּתְּחוּ / טוּרָיו כְּמוֹ בָּבוֹת בְּעַפְעַפָּיִם,

אוֹ מִתְּכֵלֶת רָקְמוּ אוֹתוֹ יְדֵי / הָעֵט עֲלֵי חוּר מַעֲשֵׂה מִצְרָיִם.

חַלָּשׁ, וְחוֹגְרֵי הַחֲרָבוֹת יַחֲלֹשׁ / בִּקְרָב לְפִי שִׁנָּיו כְּאִישׁ בֵּינָיִם,

רַגְלִי, וְיָרִיחַ לְרָחוֹק מִלְחֲמוֹת / רֶכֶב וְלָזֹאת הוּא חָגוּר מָתְנָיִם,

דּוֹבֵר בְּלִי לָשׁוֹן וְאוֹחֵז מִבְּלִי / אֶצְבַּע וְרָץ כִּצְבִי בְּלִי רַגְלָיִם,

אִלֵּם, וְצַחוֹת מַאֲמָרָיו יִשְׁמְעוּ / שׁוֹכְנִים בְּאִיֵּי יָם וּבִסְפַרְוָיִם.

יָפִיק זְמַם כָּל יוֹדְעֵי סֵפֶר וְשֹׁד / לִכְסִיל בְּלִי סֵפֶר וּפִיק בִּרְכָּיִם.

כַּחֵץ, וְעֵת יוֹרֶה שְׁלֹמֹה מַאֲפֵל / כֶּסֶל יְלֻבַּשׁ מַעֲטֵה קַרְנָיִם.

וּבְצַוְארֵי סִפְרוֹ עֲנָקִים יַעֲרֹךְ / לֹא יַעֲרֹךְ אוֹתָם זְהַב פַּרְוָיִם,

סֵפֶר כְּרִצְפַּת שֵׁשׁ וְהַדִּבֵּר כְּמוֹ / רִצְפָּה וְשִׁנֵּי עֵט כְּמֶלְקָחָיִם,

מִלִּים וְעִנְיָנִים רְאֵה בָם וֶאֱמֹר: / מֹר מִכְּפוֹר הוּצָא וְאֵשׁ מִמָּיִם.

לִקְרַאת נְחִירִים יָצְאוּ סַמָּיו וּמֹר / עוֹבֵר בְּמָשְׁכוֹ נָטְפוּ כַּפָּיִם.

לוּ יֶחֱזוּ זֹהַר דְּבָרָיו כּוֹכְבֵי / רוּם כָּרְעוּ וַיִּקְדוּ אַפָּיִם,

אוֹ יִגְבְּהוּ שִׁירִים כְּנֶשֶׁר, הֵם לְבַד / כִּדְבַר נְבוּאוֹת קָרְעוּ שָׁמָיִם.

לֹא שָׂבְעָה עֵין הָאֱנוֹשׁ מִמַּחֲזֵה / הוֹדָם וּבָם לֹא נִמְלְאוּ אָזְנָיִם.

שׁוֹאֵל בְּעַד חָכְמַת שְׁלֹמֹה, סוּר אֱלֵי / יַם בִּין מְאֹד גָּדוֹל רְחַב יָדָיִם,

אֹדֶם גְּבוּלוֹ שָׂם וּמֵימָיו יִגְרְשׁוּ / בֹּשֶׂם וְטַעְמוֹ צוּף לְמַלְקוֹחָיִם.

אִם עָרְכוּ עִמּוֹ בְּנֵי חַיִל קְרָב / אֹמֶר וְשִׁיר – לֹא מָצְאוּ יָדָיִם,

אוֹ רוֹצְצוּ סוּסֵי בְּרָקִים אַחֲרָיו – / לָאוּ וְדַלְיוּ עַד מְאֹד שׁוֹקָיִם,

יִכְבַּד בְּיוֹם תָּבוֹר כְּעָלֶה יַעֲלֶה / לוּ יִהְיֶה עִמּוֹ בְּכַף מֹאזְנָיִם,

שֶׁמֶשׁ בְּלוֹיֵי מַחְלְצוֹתָיו תַּחֲמֹד / כִּי לָבְשָׁה מֵהֶם כְּסוּת עֵינָיִם,

הוּא הַיְּסוֹד לַבִּין וּמוֹלַדְתּוֹ, כְּמוֹ / אֶחָד יְסוֹד חֶשְׁבּוֹן לְרִבּוֹתָיִם.

הַצּוֹרְרִים אֹתוֹ וְשִׂיאוֹ חָמְדוּ / הֵם צוֹרְרִים רוּחַ בְּתוֹךְ חָפְנָיִם,

יוֹם הַנְּדֹד תִּקַּח אֲפֵלָה, יֶחְשְׁכוּ / כָּל כּוֹכְבֵי נִשְׁפּוֹ כְּמוֹ כִירָיִם,

גַּם לֵיל פְּרִידָה אַל יְשַׁחֵר יוֹם וְאַל / יִרְאֶה לְשַׁחַר אוֹר וְעַפְעַפָּיִם.

יוֹם אֵד קְרָבִים יַעֲלֶה עַד נַחֲלֵי / רֹאשׁ נָזְלוּ אַךְ עָבְרוּ אַפְסָיִם,

מַיִם אֲשֶׁר גָּבְרוּ וְכִמְעַט יִשְׁטְפוּ / לוּלֵי בְּעֵינַי נֶחֱצוּ לִשְׁנָיִם,

לֹא דָלְלוּ, כִּי מֵי אֲהָבִים שָׁאֲבוּ / מִלֵּב וְדָלוּ אֶת דְּמֵי מֵעָיִם.

לֵב יַעֲזֹב אֶת רַב שְׁלֹמֹה אַחֲרָיו / אֵיךְ יַעֲמֹד אוֹ יִחְיֶה יוֹמָיִם?

לִבּוֹת בְּדַם הָאַחֲוָה רָווּ וְגַם / מֵאֵם יְדִידוּת יָנְקוּ שָׁדָיִם.

הַתֵּר בְּמִכְתַּב יָדְךָ נִסְגָּר בְּכוֹר / אַהַב וְעֹל פֵּרוּד וּבִנְחֻשְׁתָּיִם.

נַפְשִׁי אֲמָתְךָ אַל תְּיַסֵּר, בַּעֲבוּר / כִּי לָקְחָה מִיַּד נְדֹד כִפְלָיִם.

הִנֵּה לְךָ שִׁירַת יְדִידוּת תַּעֲלֹץ / בִּשְׁמָךְ כְּשִׁיר מָחוֹל וּבִמְצִלְתָּיִם,

תִּשַּׁח לְךָ, אָכֵן יְאַו שֶׁמֶשׁ נְשׂוֹא / אוֹתָהּ עֲלֵי חֹצֶן וּבִכְתֵפָיִם,

תּוֹדֶה לְחָכְמָתְךָ, וְאִם הִיא מִמְּעִיל / בִּין עַל שְׁחָקִים סָחֲבָה שׁוּלָיִם;

תִּקְרָא: אֲנִי שִׁפְחָה לְשָׁרֵת מַהֲלָל / ״עָב תַּעֲרֹף אוֹ יִזְּלוּ עֵינָיִם״.

תֵּרוֹם עֲדֵי גָּבְהֵי שְׁחָקִים יִשְׁפְּלוּ / תִּשְׂרֶה עֲדֵי יִמַּק צְבָא שָׁמָיִם.
