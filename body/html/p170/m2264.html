[עֵינֵי סְלָעִים דָּֽמְעוּ דָמֹעַ]

עֵינֵי סְלָעִים דָֽמְעוּ דָמֹעַ / עָלַי וְכִמְעַט נִקְרְעוּ קָרֹעַ

וַיְאַמְּצוּ לִבָּם בְּנֵי אִמִּי בְּיוֹם / לִבּוֹת מְשַׂנְאַי נִבְקְעוּ בָקֹעַ

וַיְנַצְּלוּ בִידֵי פְתַיּוּתָם צְבִי / עֶדְיָם וְהֵם לֹא־יָֽדְעוּ יָדֹעַ

עַם לֹא־יְדַעְתִּים מִהֲרוּ עָדַי בְּשׁוּר / כִּי בִי יְלָדַי פָּּֽשְׁעוּ פָשֹׁעַ

וַיְדַבְּרוּ עַל־לֵב וְסָרוּ לַחֲבֹשׁ / מַכִּים אֲשֶׁר הֵם פָּֽצְעוּ פָצֹעַ

אֵיךְ יַעֲמֹד לִבִּי וְיֶלֶד הַזְּמָן / אֹתִי בְּשֵׁבֶט הַנְּדֹד יָרֹעַ

בֹּקֶר אֲכֹל רֹתֶם וְלָעֶרֶב שְׁתוֹת / מֵי־רֹאשׁ לְרַוּוֹת זֶה וְזֶה לִשְׂבֹּעַ

אֶל מַעֲמַקֵּי הַֽיְגוֹנִים יוֹם לְיוֹם / יָרֹד וְאֶל־הָרֵי תְלָאוֹת שׁוֹעַ

חָלַק לְעֵינַי הַנְּדֹד בֶּכֶה וְאֶל / קִרבִּי הֲמוֹת יַמִּים וְלַלֵּב נוֹעַ

בֵּינוֹת זְאֵבִים שָֽׁכְנוּ עֶרֶב עֲדֵי / שֵׁם הָאֱנוֹשׁ לֹא־שָֽׁמְעוּ שָׁמֹעַ

פָּגֹשׁ בְּדֹב שַׁכּוּל וְחֶבְרָה עִם־אֲרִי / טוֹב מִפְּגֹשׁ אֹתָם וּבָם לִפְגֹּעַ

שָׂמִים לְחֹשֶׁךְ אוֹר וְכִילַי אֵיךְ יְהִי / נִכָּר בְּעֵינֵיהֶם לְנֶגֶד שׁוֹעַ

שַׂחְתִּי לְלִבִּי צֵא וְאוּלַי תִּֽהְיֶה / מוֹצֵא בְאַלְפֵי הַזְּמָן מַרְגּוֹעַ

וַיַּעֲנֶה נוֹזְלִים לְחֶסֶד חָֽרְבוּ / כִּי כַל־רְבִיבָיו נִמְנְעוּ מָנֹעַ

וּנְאוֹת יְדִידֹת יָֽבְשׁוּ כִּי־נֶעְצְרוּ / גִשְׁמֵי אֱמֶת עַד־עָֽרְכוּ גִלְבֹּעַ

הָהּ לַשְּׂרִידִים מִיְּדִידַי נָדֲדוּ / כִּי רֹב הֲמוֹנָם שָֽׁלְמוּ לִגְוֹעַ

הֵם אָהֳלֵי חֶסֶד אֲשֶׁר יוֹם צָעֲנוּ / בֵּין לֵב וְכָבֵד נִתְקְעוּ תָקֹעַ

שָׁלוֹם עֲלֵיהֶם כִּי מְאֹד הִתְרַחֲקוּ / אַךְ אֶל־לְבָבִי נָֽגְעוּ נָגֹעַ

כָּזֹה וְכָזֶה בִּעֲתוּ נַפְשִׁי בְּנֵי / יָמִים וְאַךְ הִיא מֵאֲנָה לִנְגֹּעַ

עוֹלָם מְאֹד צַר בִּי עֲדֵי כִי־נֶחְשְׁבָה / תֵבֵל כְּצָמִיד וַאֲנִי אֶזְרוֹעַ.
