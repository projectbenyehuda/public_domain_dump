[נִבְטַח בְּהָפִיק זְמָן כִּמְעַט זְמָמֵינוּ]

נִבְטַח בְּהָפִיק זְמָן כִּמְעַט זְמָמֵינוּ / וַנַּאֲמִין בּוֹ וְאָכֵן הוּא יְרַמֵּנוּ

יַשִּׁיא לְבָבוֹת בְּדָבָר רֵק וְנִפָּת וְלֹא / נָבִין הֲכִי חִישׁ בְּכַף־קֶלַע יְשִׂימֵנוּ

עַיִן נְעַצֵּם יְמֵי חֶבְרָה וְחִצֵּי נְדֹד / יוֹרֶה וְנִכְחָם לְמַטָּרָה יְקִימֵנוּ

נַחְנוּ נְקַבֵּץ פְּזוּרֵי לֵב וְנָפְרִיד קְשׁוּ־ / רֵי לֵב וְנִפְרֹשׂ לְלָכְדָם אֶת־חֲרָמֵינוּ

נָשִׂים טְפָחוֹת יְמֵי חֶבְרָה וּפֵרוּד עֲדֵי / יָשׁוּב בְּעוֹלָם מְקוֹם צָפוֹן דְּרוֹמֵנוּ

מִי־זֶה אֲשֶׁר הֶאֱמִין בָּנוּ יְשֻׁלָּם וְאִם / יִרְגַּע הֲלֹא יֶחֱרַד מִקּוֹל רְעָמֵינוּ

נַתְאִים לְיַחֵד וְגַם נַחְבִּיר לְהַפְרִיד וְשִׂיא / אִישׁ אִם־נְרוֹמֵם לְהַשְׁכִּינוֹ תְּהוֹמֵנוּ

אֵלֶּה קְצוֹת מַעֲלָלֵינוּ וְכָל־מִי יְאַו / בִּלְתָּם יְאַוֶּה קְדֹחַ אֵשׁ בְּמֵימֵינוּ

הָהּ יָֽזְמוּ לַעֲשׂוֹת רָעָה וְלֹא בֹשֲׁשׁוּ / עַד־הֶעֱלִימָם מְאוֹר אַנְשֵׁי שְׁלוֹמֵנוּ

שֶׁמֶשׁ וְסַהַר אֲשֶׁר בָּאוּ בְּפֶתַע עֲדֵי / נָדוּ לְנוּדָם שְׁנֵי אוֹרִים בְּשָׁמֵינוּ

רֹעֵי אֱמוּנָה וְתַמֵּי־בִין וְחֶסֶד וְהֵם / מֵאָז תְּרוּמָה לְאֵל מִכָּל־יְקוּמֵינוּ

צִבְיֵי עֳפָרִים וּמִשִּׂכְלָם הֲלֹא אָֽלְפוּ / עָרְמָה פְתָאיִם וּבוֹ חָכְמוּ חֲכָמֵינוּ

חָכְמַת יְשִׁישִׁים וְהוֹד טַעַם חֲכָמִים וְיָ־ /שַׁבְנוּ סְכָלִים וְהֵמָּה מַעֲרִימֵינוּ

כּוֹכְבֵי אֲדָמָה וְאַךְ מָשְׁלָם בְּלַיִל וְיוֹם / מֵאֵין חֲלִיפוֹת כְּמוֹ כוֹכְבֵי מְרוֹמֵנוּ

לָבְשׁוּ לְבוּשׁ תֹּם וּבִבְלוֹיֵ חֲלִיצוֹת אֲשֶׁר / חָלְצוּ עֲלֵי הַזְּמָן גָּאוּ תְמִימֵינוּ

נֻצַּל עֲדִי בַעֲלֵי הַשֵּׁם וְהַתֹּם עֲדֵי / הָיוּ עֲלֵי־אַף בְּנֵי הַיּוֹם נְזָמֵינוּ

הָיוּ אֲבֹתָם לְרֹאשׁ אַךְ יָֽסְפוּ הֵם יְקָר / עַד כִּי־נְתָנָם זְמָן צֶפֶת לְאֻמֵּנוּ

דֹּרְשֵׁי בְּעַד שֹֽׁבְרֵי שֶׁבֶר תְּעוּדָה לְבַב / אָבוּן וְיוֹסֵף בְּיוֹם חֹסֶר אֲסָמֵינוּ

וּלְשֹׁאֲלִים עַל־מְטַר הַשְׂכֵּל לְפִיהֶם בְּיוֹם / עֹצֶר תְּפוּשׁוּן וּבוֹ נוֹזְלֵי גְשָׁמֵינוּ

וּלְחֹֽפְשֵׂי מִדְבְּרֵי מוּסָר תְּרוּצוּן אֱלֵי / מִלִּים אֲשֶׁר יִטְּפוּ מֶתֶק טְעָמֵינוּ

וּלְזֹעֲפִים מִן־יְגוֹן תֵּבֵל פְּנֵיהֶם רְאוּ / כִּי בָם זְמָן שָׂם לְבַד צַהְלַת עֲגוּמֵינוּ

יִשְׁגּוּ וְאָכֵן בְּרֹךְ לָשׁוֹן יְרַצּוּ וְיוֹם / נֶחְטָא בְּטֶרֶם דְּרֹשׁ יִשְׂאוּ אֲשָמֵינוּ

מִיּוֹם אֲשֶׁר נֵסְתְּרוּ גִּלּוּ דְמָעוֹת צְפוּן / לִבּוֹת וְחָזוּ בְלִי־עַיִן עֲלֻמֵנוּ

נִבְכֶּה בְדֶמַע עֲלֵי מִפְקַד פְּנֵיהֶם וְעֵת / נִזְכֹּר נְדֹדָם אֲזַי נִמְסֹךְ בְּדָמֵנוּ

עָרְבוּ שְׂמָחוֹת וְהֵן שֵׁכָר לְשֹׁתָיו מְאֹד / יֵמַר כְּאִלּוּ בְרֹאשׁ הֻשְׁקוּ כְרָמֵינוּ

הִלְבִּין נְדֹדָם שְׁחוֹר שֵׂעָר וְהִקְדִּיר מְאוֹר / פָּנִים וְשִׁנָּה הֲדָרֵנוּ וְצַלְמֵנוּ

מָרוּ נְפָשִׁים עֲדֵי מָתַק לְנֶגְדָּם בְּיוֹם / רָתְמוּ לְרִכְבֵי נְדֹד שֹׁרֶשׁ רְתָמֵינוּ

נִשְׂבַּע נְדֻדִים עֲדֵי־נֶשֶׁף וְנִרְעֶה פְנֵי / שַׁחַק וְלֹא־נֶחֱזֶה כָהֵם בְּרוֹמֵנוּ

לוּלֵי אֱשֶׁר נָדֲדָה שֵׁנָה נְקַוֶּה רְאוֹת / תַּבְנִית פְּנֵיהֶם בְּלַיִל בַּחֲלוֹמֵנוּ

הָבָה וְנִבְכֶּה עֲלֵי־אֹהֶל יְדִידֹת אֲשֶׁר / צָעַן וְנַרְוֶה בְּדַם כָּבֵד תְּלָמֵינוּ

אֵיךְ נֶחֱרַד מִזְּמָן לָעַד וְחָלוּ מְגוּ־ / רֵינוּ בְּיוֹם נָדֲדוּ יַחַד וְאֵימֵינוּ

הִפְרִיד אֲהוּבִים וְלֹא־קֶשֶׁר אֲהָבִים וְאֵיךְ / יוּכַל לְשַׁנּוֹת זְמָן מֵיטַב כְּתָמֵינוּ

כֹּפֶר אֲנַחְנוּ לְעָפְרֵי חֵן רְפוּאָה בְּנִיב / פִּיהֶם לְמַכְאוֹב וְשִׁקּוּי לַעֲצָמֵינוּ

וּפְדוּת לְיָמִים אֲשֶׁר הָיוּ אֲדֹנִים לְכָל־ / שָׁנִים אֲשֶׁר חָֽלְפוּ מֵאָז וְיָמֵינוּ

הָיוּ יְמוֹת הַזְּמָן חֹלֶשׁ וְזֹקֶן וְאַךְ / אֵלֶּה לְבַדָּם יְמֵי־נֹעַם עֲלוּמֵינוּ

לֹא נִכְּרוּ מִשְּׁחוֹר בִּלְתָּם וְעִמָּם אֲפַל / לֵילוֹת מְאֹד צַח כְּעֵין גִּזְרַת לְשָׁמֵינוּ

נָשׁוּט סְבִיב אָהֳלָם תָּמִיד וְנִמְצָא בְמִד־ / רַךְ נַעֲלֵיהֶם נְעִים רֵיחַ בְּשָׂמֵינוּ

נָנוּד לְעִתּוֹת בְּמַשָּׁאוֹן וְנֶחְשֶׁה פְנֵי / צָרִים לְבַל־יֹאֽמְרוּ הֶאָח וְקָמֵינוּ

אַחַי בְּחֵי אַהֲבָה הִשָּֽׁמְרוּ מִשְּׁכֹן / זָרִים יְמֵי־עַד בְּלִבַּתְכֶם מְקוֹמֵנוּ

כִּי אֹהֲבִים לֹא נְחַמֵּד בִּלְעֲדֵיכֶם וְעַד / לֹא־יִמְעֲדוּ מִמְּכוֹן חֵשֶׁק פְּעָמֵינוּ

אָכֵן צְרִי מִכְתְּבֵי יֶדְכֶם בְּיוֹם תִּשְׁלְחוּ / מִנִּי חֳלִי תַאֲוָה הוּא יַחֲלִימֵנוּ

נוֹחִיל וְאוּלַי בְּסוֹד יָמִים לְהַפְלִיא וְעוֹד / מִבֵּין תְּהֹמוֹת נְדֹד יָשׁוּב יְרִימֵנוּ

וּשְׂאוּ כְּפִי אַהֲבָה שָׁלוֹם וְאִם־יֶחֱנַן / הָאֵל נְפָשִׁים נְשַׁו נִדְרֵי שְׁלָמֵינוּ.
