אֶל אֱלֹהֵי נַפְשִׁי אֲנַהֵל

סימן: א’ב (משולש), משה קטן (משולש)

אֶל אֱלֹהֵי נַפשִׁי אֲנַהֵל –

אוּלַי אוֹר סְלִיחָתוֹ יָהֵל,

אָכֵן מִפָּנָיו אֶבָּהֵל

אֶתְבּוֹנֵן וְאֶפְחַד מִמֶּנּוּ: / איו' כג, טז


בְּפַחַד קַמְתִּי וְנָפְלוּ פָנָי

בְּאֵד קְרָבַי נָזֹלּוּ עֵינַי

בְּאֵימָה בָאתִי לִפְנֵי אֲדֹנָי

גָּדוֹל וְנוֹרָא מְאֹד וּמִי יְכִילֶנּוּ: / יוא' ב, יא


גְּלוּת צְפוּן לִבִּי אוֹאֵל

גֵּר אָנֹכִי עַל דֶּלֶת שׁוֹאֵל

גָּר מִפַּחְדּוֹ, כִּי יָקוּם אֵל

וְכִי יִפְקֹד מָה אֲשִׁיבֶנּוּ: / איו', לא, י


דַּלּוֹתִי לְיוֹם אֱהִי בוֹ נִשְׁפָּט

דַּלּוּ עֵינַי לְיוֹשֵׁב עַל הַמִּשְׁפָּט

דּוֹמֵם אֶהְיֶה בְעָמְדִי לְמִשְׁפָּט

כִּי לֹא אִישׁ כָּמוֹנִי אֶעֱנֶנּוּ: / שם ט, ל


הָאֵל, עֲוֹן נְעוּרַי אַל תִּשְׁמְרֶנּוּ

הֶבֶל כָּל אָדָם וְאֵיךְ תִּבְחָנֶנּוּ –

הֶרֶף, מָה אֱנוֹשׁ כִּי תִזְכְּרֶנּוּ

וּבֶן אָדָם כִּי תִפְקְדֶנּוּ: / תה' ה, ה


וּלְאַט לְשִׁכּוֹר בְּעוֹלָם בְּלִי יָיִן

וְיִסְכַּל כִּי עָשְׁרוֹ כְאָיִן

וּזְכוֹר כִּי נִבְהָל לַהוֹן אִישׁ רַע עָיִן

וְלֹא יֵדַע כִּי חֶסֶר יְבֹאֶנּוּ: / מש' כה, כב


זְהָבוֹ כְעָפָר יִהְיֶה וְהוֹנוֹ,

זוֹחֵל עַל מַעֲשֵׂהוּ וְקִנְיָנוֹ

זְמָן יֹרִשֶׁנּוּ מִבִּטְנוֹ

חַיִל בָּלַע וַיְקִאֶנּוּ: / איו' כ, טו


חָסְנוֹ יִדַּל וְעֻזּוֹ יְשֻׁנֶּא

חֵילוֹ יֹאבַד וְכָל אֲשֶׁר הוּא קוֹנֶה

חֲמוּדוֹ יִקְרָא וְלֹא יֵעָנֶה

מִצָּרָתוֹ לֹא יַצִּילֶנּוּ: / ישע' מו, ז


טֹרַף עֹשֶׁר, כְּרָקָב יְחֻבַּל

טְחוּן פָּנִים לַקֶּבֶר יוּבַל

טוּבוֹ יִמְאַס וִיהִי כְנָבָל

וּבַחֲצִי יָמָיו יַעַזְבֶנּוּ: / ירמ' יז, יא


יִצְבֹּר כְּחוֹל יַמִּים כֶּסֶף

יֶלֶד יוֹם אוֹתוֹ יְשַׁסֵּף

יוֹם תְּבֹאֵהוּ שׁוֹאָה וְלֹא יֵאָסֵף

עֵינָיו פָּקַח וְאֵינֶנּוּ: / איו' כז, יט


כְּבוֹדוֹ יָעוּף כַּקַּשׁ לְפָנָיו

כִּידוֹ יִרְאוּ עֵינָיו

כְּפֶתַע, קוֹל פְּחָדִים בְּאָזְנָיו

בַּשָּׁלוֹם שׁוֹדֵד יְבוֹאֶנּוּ: / שם טו, כא


לְתֵבֵל אֵל תִּצְעַד צָעֹד

לְיָפְיָהּ אַל תָּרוּץ לַחֲמֹד

לֵב הַכְנִיעַ וְתִתְמֹךְ כָּבוֹד

וְגַאֲוַת אָדָם תַּשְׁפִּילֶנּוּ: / מש' כט, כג


מְאַס צוּף פִּיהָ וְשִׂיחָהּ

מִהְיוֹת נִפְתֶּה בְרוּחָהּ –

מִי פֶתִי הִטַּתּוּ בְּרֹב לִקְחָהּ

בְּחֵלֶק שׂפָתֶיהָ תַּדִּיחֶנּוּ: / מש' ז, כא


נוֹצַר אִישׁ לֶעָמָל חָיָה,

נַפְשׁוֹ תֹאבַד וְאַחֲרִיתוֹ צִיָּה,

נִפְלָאוֹת יִדְרֹשׁ וְרָחוֹק מַה שֶּׁהָיָה

וְעָמֹק עָמֹק מִי יִמְצָאֶנּוּ: / קה' ז, כד


סוֹד יוֹצְרוֹ אֵיךְ יִהְיֶה מֵעִיר

סַף לִפְנֵי שֶׁמֶשׁ כְּעַל דֶּשֶׁא שָׂעִיר? –

סָכָל לֹא יָדַע לָלֶכֶת אֶל עִיר

עֲמַל הַכְּסִילִים תְּיַגְּעֶנּוּ: / שמ י, מד


עֵרוֹם מִסּוּת הַשֵׂכֶל וּבַעַר

עִוֵּר יְמַשֵּׁשׁ לִמְצֹא שַׁעַר

אִוֶּלֶת קְשׁוּרָה בְּלֶב-נָעַר

שֵׁבֶט מוּסָר יַרְחִיקֶנָּה מִמֶּנּוּ: / מש' כב, טו


פַּחֲדּוּ מִיּוֹם יַכְבִּיד שִׁבְרוֹ

פְּנוּ אֶל צֶדֶק מַאֲמָרוֹ –

פִּתְאֹם יוֹם יָרֵחַ לֹא יָהֵל אוֹרוֹ

וְשֶׁמֶשׁ בֶּעָנָן אֲכַסֶּנּוּ: / יחז' לב, ז


צְעַד אֶל יֹשֶׁר נְקִי כַפָּיִם,

צְבֹר חֶסֶד טְהָר-יָדָיִם,

שַׁלַּח לַחְמְךָ עַל פְּנֵי הַמָּיִם

כִּי בְרֹב הַיָּמִים תִּמְצָאֶנּוּ: / קה' יא, א


קַנֵּא לְיוֹנָה קְצוּצַת אֵבֶר

קוֹנְנָה בְּפַח יָגוֹן וָשֶׁבֶר

קְהַל בָּנֶיהָ יָשִׂישׂ אָל קֶבֶר

הַמְּחַכִּים לַמָּוֶת וְאֵינֶנּוּ: / איו' ג' כא-כב


שִׁרְיָן תִּלְבַּשׁ לְעֶזְרַת אֲנוּשִׁים

שְׁלַח יָדְךָ לַהֲמוֹן יְקוּשִׁים

שׁוֹפֵט, קַנֵּא לְמוֹשֵׁל בַּאֲנָשִׁים

צַדִּקִים וְטֹבִים מִמֶּנּוּ: / מל"א ב, לב


תַּמּוּ חֲסִידַי וְגָבַר צָרִי –

תָּעִיתִי וְהוּא בֵּין אַחִים יַפְרִיא

תָּמַהְתִּי כִּי כָרַע שָׁכַב כַּאֲרִי

וּכְלָבִיא מִי יְקִימֶנּוּ: / במד' כד, ט


מִכְמַרְתּוֹ תִהְיֶה פָעֳלוֹ

מַהֵר יֶאֳרֶה פְּרִי מַעֲלָלוֹ

מַשְׂטִין יְפַתֵּהוּ וְאוּלַי נוּכְלָה לוֹ

וְנִקְחָה נִקְמָתֵנוּ מִמֶּנּוּ: / ירמ' כ, י


שִׁקְרוֹ לְנֶגֶד פָּנָיו הִצִּיבוֹ –

שָׁת בַּשָּׁמַיִם פִּיהוּ וְנִיבוֹ,

שָׂח: מַה נּוֹעִיל כִּי נִפְגַּע בּוֹ

וּמַה שַּׂדַּי כִּי נַעַבְדֶנּוּ: / איו' כא, טו


הֲמוֹן פְּסִילָיו אוֹתוֹ יַפִּילוּ

הֶמְיַת אֲמָרָיו פְּעָמָיו יַכְשִׁילוּ –

הַיּוֹם כְּמָרָיו עָלָיו יָגִילוּ

עַל כְּבוֹדוֹ כִּי גָּלָה מִמֶּנּוּ: / הו' י, ה


קִרְיַת חָנָה וּפִנּוֹת מִקְדָּשׁ

קָדוֹשׁ אֲשֶׁר בִּצְדָקָה נִקְדָּשׁ –

קֹרָא לָךְ שֵׁם חָדָשׁ

אֲשֶׁר פִּי אֲדֹנָי יִקָּבֶנּוּ: / ישע' סב, ב


טוּבוֹ יַרְאֵהוּ מַאֲוַיָו

טְהוֹר פָּעֳלוֹ יְחַוֶּה בְּחַיָּיו

טוֹב אֲדֹנָי לְקֹוָו

לְנֶפֶשׁ תִּדְרְשֶׁנּוּ: / אי' ג, כה


נוֹרָא, סְעַד יְמִין נוֹקָשׁ

נָאוֹר, הַיְשַׁר לְפָנָיו דֶּרֶךְ נֶעֱקָשׁ

נְבוּאַת בָּעֵת הַהִיא יְבֻקַּשׁ

אֶת עֲוֹן יִשְׂרָאֵל וְאֵינֶנּוּ: / ירמ' נ, כ
