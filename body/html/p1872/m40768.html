וִכּוּחַ רֵכָב וּבַעֲנָה

Immanuel Frances

]נולד בלִיווֹרְנוֹ בשנת 1618, חי איזה זמן בפֵֿירֶנְצֵי ובאַלְגַ֗זִיר, מת בלִיווֹרְנוֹ אחרי שנת 1710]


רכב: סוּרָה אֵלַי בַּעֲנָה אָחִי.

בענה: הֶרֶף מִמֶּנִּי כִּי הַיּוֹם מְרִי שִׂחִי / כִּי לֹא נְתָנַנִי הַזְּמָן הָשֵׁב רוּחִי.

רכב: אֵיךְ נִבְהֲלוּ רַעְיוֹנֶיךָ / וְלָמָּה נָפְלוּ פָנֶיךָ / וְתַחֲרֹק שִׁנֶּיךָ/ וּמַה־יִּרְמְזוּן עֵינֶיךָ.

בענה: הִנְנִי נִשְׁבַּע בְּחַיֵּי רֹאשִׁי / כִּי כִמְעַט אֶבְחַר מַחֲנַק נַפְשִׁי.

רכב: גָּרְסָה נַפְשִׁי לְתַאֲבָה / לָדַעַת הַסִּבָּה / הַגִּידָה־לִי הַמַּחֲלָה / אוּלַי אֶמְצָא לָהּ תְּעָלָה.

בענה: תְּשֻׁאוֹת חֵן חֵן לְךָ רֵכָב רֵעִי / אַלּוּפִי וּמְיֻדָּעַי / אָמְנָם דַּע כִּי אֲנוּשָׁה מַכָּתִי / וְאֵין מַרְפֵּא לְמַחֲלָתִי / כִּי אֵיכָכָה אוּכַל / וְלִבִּי אֵשׁ קִנְאָה לֹא תֹּאכַל / בִּרְאוֹתִי ישֵׁב בְּסֵתֶר הֵיכָל / כָּל־טִפֵּשׁ וְכָל־סָכָל / אֲשֶׁר בְּשֵׁם רַבָּנִים יְקֹרָאוּ / וּמִפְּנֵיהֶם זְקֵנִים עָמְדוּ וּנְעָרִים נֶחְבָּאוּ / וַאֲנִי הַדַּל וְהַמָּךְ / בְּהֵמוֹת הַבַּיִת הָיִיתִי עִמָּךְ / לֹא יַבִּיטוּ וְלֹא יִרְאוּ בִי / וְלֹא הִגַּעְתִּי לְהִקָּרֵא רַבִּי / עַל־זֶה אַשְׂחֶה בְּכָל־לַיְלָה מִטָּתִי / וְאַמְסֶה עַרְשִׂי בְּדִמְעָתִי / וְאֶקְרָא לַמָּוֶת לְהָפִיק מַאֲוַיָּי / כִּי טוֹב מוֹתִי מֵחַיָּי.

רכב: צְחוֹק עָשִׂיתָ לִי בֶּאֱמֶת וֶאֱמוּנָה / וְעַתָּה שְׁמַע בְּקוֹלִי אֲאַלֶּפְךָ בִּינָה / וּלְמַרְבֶּה לִתְקוּפַת הַשָּׁנָה / תִּקָּרֵא הָרַב כמהר"ר בַּעֲנָה.

בענה: רֵכָב מֵרִים רֹאשִׁי וּכְבוֹדִי / אִם הַחֶסֶד הַזֶּה תַּעֲשֶׂה עִמָּדִי / אֶהְיֶה לְךָ מֵאָח דָּבֵק / וּבַעֲפַר רַגְלֵיךָ אֶתְאַבֵּק.

רכב: הַסְכֶּת־נָא אֵלָי / וַעֲשֵׂה אָזְנְךָ כַּאֲפַרְכֶּסֶת לְמִלָּי / שִׂים הַסַּרְבָּל עַל־הַכְּתֵפַיִם / שֶׁיִּהְיֶה אָרֹךְ עַד הָרַגְלַיִם / וּבָתֵּי עֵינַיִם קְשׁוּרִים לְאָזְנַיִם / וְכוֹבָע גְּדוֹל הַכְּנָפַיִם / וְאַבְנֵט חָגוּר עַל־הַמָּתְנַיִם / וּבָתֵּי יָדַיִם / וּבְשִׁבְתְּךָ בְּבֵית הַכְּנֶסֶת בְשִׁפְלוּת עֵינַיִם / תָּנִיעַ הַשְּׂפָתַיִם / וּבְבִרְכַּת “בּוֹנֶה יְרוּשָׁלַיִם” / תִּשָּׂא עֵינֶיךָ לַשָּׁמַיִם / וְתוֹרִיד דְּמָעוֹת כַּמַּיִם / וּבְבִרְכַּת הוֹדָאָה תִּקֹּד עַד הַקַּרְסֻלַּיִם / וּבְ“עוֹשֵׂה שָׁלוֹם” תִּשְׁתַּחֲוֶה שִׁבְעָתַיִם / וְאִם תּוּכַל גַּדֵל זְקָנְךָ שֶׁיַּגִּיעַ לְבִרְכַּיִם / כִּי מִי יִרְאֶה אִישׁ זְקָנוֹ מַגִּיעַ לַטַּבּוּר / גַּם כִּי יִהְיֶה רֵק וָבוּר / וְלֹא יֹאמַר שֶׁהוּא רָאוּי לִהְיוֹת רַב מְמֻנֶּה עַל־הַצִּבּוּר.

בענה: מַה־נִּמְלְצוּ אִמְרוֹתֶיךָ לְחִכִּי / וְאוּלָם רֵכָב מַלְכִּי / הַיְשַׁר לְפָנַי דַּרְכִּי / כִּי אַתָּה תַּגִּיהַּ חָשְׁכִּי.

רכב: גַּם בְּדֶרֶךְ אַחֶרֶת תִּשָּׂא עַל־חֲבֵרֶיךָ דָּגֶל / וְהוּא בִדְרָשׁ נָכוֹן לְמוֹעֲדֵי רָגֶל. בענה: סִכְלוּתִי אוֹדִיעֲךָ וְאִוַּלְתִּי לֹא כִסִּיתִי / כִּי מֵעוֹלָם בִּדְרָשׁ לֹא נִסִּיתִי.

רכב: יֵשׁ דֹּרֵשׁ טוֹב לְעַמּוֹ / וּמִי יוֹדֵעַ אִם הוּא קֶמַח שֶׁטְּחָנוֹ בְּעַצְמוֹ / וְלֹא תִשְׁמַע לְקוֹל מְלַחֲשִׁים / שֶׁיֹּאמְרוּ כִּי מָה הֵם דּוֹרְשִׁים / בִּפְשָׁטִים נָאִים וּפֵרוּשִׁים / מִלִבָּם הֵם מְחֻדָּשִׁים / כִּי גָבְרוּ וְלֹא לֶאֱמוּנָה / וּבְפִיהֶם אֵין נְכוֹנָה / כִּי רָבוּ כְּמוֹ רָבוּ סִפְרֵי הַדְּרָשׁוֹת / יְשָׁנוֹת גַּם חֲדָשׁוֹת / וְהַדַּרְשָׁנִים יַעֲשׂוּ מֵהֶם / דְּרוּשִׁים לְכָל־חֶפְצֵיהֶם / גַּם יֵשׁ דַּרְשָׁן בִּזְמַנֵּנוּ / שֶׁרָאִינוּ בְעֵינֵינוּ / וְשָׁמַעְנוּ בְאָזְנֵינוּ / אֲשֶׁר הָיָה בְּעֵינֵי הָעָם לְאוֹת / וַיְסַפְּרוּ מִדְּרָשׁוֹתָיו נִפְלָאוֹת / וְלֹא שָׁמַעְתִּי מִפִּיו כִּי אִם גִּמַּטְרִיָּאוֹת לְמֵאוֹת / וְנוֹטָרִיקוֹן וְרָאשֵׁי תֵיבוֹת / לַאֲלָפִים וְלִרְבָבוֹת / וְעַל־כַּמָּה מַאֲמָרִים וּפְסוּקִים / שֶׁקָּשִׁים לְהָלְמָם וְכִרְאִי מוּצָק חֲזָקִים / יֹאמַר פֵּרוּשִׁים דְּחוּקִים / וּמֵהָאֱמֶת רְחוֹקִים / כָּל־הָעָם מִקָּצֶה וְעַד־קָצֶה / כָּל־דִּבּוּר אֲשֶׁר מִפִּיו יוֹצֵא / בְּצָמָא שׁוֹתֶה וּמוֹצֶה / וְקָרָא זֶה אֶל־זֶה וְאֹמֵר / הַפֵּרוּשׁ הַזֶּה כְּמִין חֹמֶר / וְכָל־הַדָּרוּשׁ בְּעֵינַי כִּנְתִינָתוֹ מִסִּינַי.

בענה: בֹּא אֶל־הַפְּרָטִים כֹּה יוֹשִׁיעֲךָ הַיּוֹצֵר / וּדְבַר־מָה לְךָ לֹא יִבָּצֵר / וּבְמָקוֹם שֶׁאָמְרוּ לְהַאֲרִיךְ אֵינְךָ רַשָּׁאִי לְקַצֵּר / וְדָבָר מִמֶּנִּי לֹא תְּכַסֶּה / וְאֵדְעָה מָה אֶעֱשֶׂה / בַּעֲלוֹתִי עַל־הַדּוּכָן אוֹ בְּשִׁבְתִּי עַל־הַכִּסֵּא.

רכב: תִּתְעַטֵּף בְּצִיצִת בַּעֲלוֹתְךָ עַל־הַדּוּכָן / אוֹ בְּשִׁבְתְּךָ עַל־הַכִּסֵּא לַדּוֹרְשִׁים מוּכָן / וְקֹדֶם תַּתְחִיל הַדְּרָשָׁה תִּירַק / וְהַחֹטֶם מִן הַזָּהֳמָא יוּרַק / וְקַנֵּחַ פִּיךָ כְּאִלּוּ שָׁתִיתָ מָרָק / וּזְקָנְךָ בְּאֶצְבְּעוֹתֶיךָ תִסְרָק. / אַחַר זֶה דְּרשׁ דְּרוּשְׁךָ בְּלִי בוּשָׁה / בְּעַזּוּת פָּנִים וּבְמֵצַח נְחוּשָׁה / וּבְסוֹף דְּבָרֶיךָ שִׂים פָּנֶיךָ נֶגֶד הָעָם / וְהַגֵּד לָהֶם חַטָּאתָם וּפִשְׁעָם / בֶּאֱמֹר “אֵיכָה זְהַב הַמּוּסָר יוּעַם / בְּטוּבוֹ לַאֲנָשִׁים יִטְעַם וְלַמּוֹכִיחִים יִנְעָם / רַבּוּ חַטֹּאת הַדּוֹר מֵחַטֹּאת יָרָבְעָם / וְאוּלָם אִי לָאו דְּמִסְתְּפֵינָא שֶׁיִּקְרְאוּנִי אַכְזָר מִבִּלְעָם / אֶזְעֹם מַה־שֶּׁהוּא לֹא זָעַם / אֶרְאֶה הַדּוֹר וְהוּא פָּרוּץ / וְכָל־אֶחָד וְאֶחָד לָרַע יָרוּץ / זָרִיז וְחָרוּץ / לֶאֱסֹף פָּז וְחָרוּץ / וּלְכַנֵּס הַחָכְמָה מִתְרַשְּׁלִים וּמִתְעַצְּלִים / כְּאִלּוּ הֵם נִכְשָׁלִים וְנֶחְשָׁלִים / וְאֵין מַכְלִים / וְגַם עַל־פַּרְנְסֵי הַקָּהָל אֶשְׁפֹּךְ שִׂיחַ / וּכְלַפֵּי שָׂרֵי הָעָם דְּבָרִים אַטִּיחַ / וְלוּ יֵשׁ־בֵּינֵינוּ מוֹכִיחַ / יָרִיעַ אַף־יַצְרִיחַ / עַד־מָתַי עַמִּי תּוֹעֶה / כַּצֹאן אֲשֶׁר אֵין־לָהֶם רֹעֶה. / קַדְּשׁוּ צוֹם קִרְאוּ עֲצָרָה / וְשׁוּבוּ לְאֵל בִּמְהֵרָה / פֶּן־יְשַׁלַּח בָּכֶם אֶת־הַמְּאֵרָה / עֶבְרָה וְזַעַם וְצָרָה / וְאַל־נָא תַקְשׁוּ עָרְפְּכֶם / וְהָסִירוּ אֶת־אֱלֹהֵי הַנֵּכָר אֲשֶׁר בְּתֹכֲכֶם / הוּא יִצְרְכֶם הָרָע הַצּוֹרֵר אֶתְכֶם / וְדִרְשׁוּ יְיָ בְּכָל־לְבַבְכֶם / אוּלַי יֵחֱנַנְכֶם וְיִשְׁלַח גּוֹאֵל וִיקַבֵּץ נִדַּחֲכֶם / כַּאֲשֶׁר דִּבֶּר לָכֶם”. / וְתֹאמַר הַכֹּל בְּרָמָה / וְתִסְפֹּק עַל־הַקְּרָשִׁים בְּחֵמָה / וּתְהִי לְךָ לְחָכְמָה / וְאִם אֵינְךָ יוֹדֵעַ עַד־מָה / וְכָל־זֶה תַּעֲשֵׂה בִנְקַלָּה / וּתְהִי לְךָ לִתְהִלָּה / וְכָל־הָעָם מִזָּקֵן וְעַד־נַעַר / יֹאמְרוּ “אִלּוּ לֹא הָיָה אָדָם גָּדוֹל, לֹא הָיָה מוֹכִיחַ בַּשָּׁעַר.”

בענה: אֵין לִי עֵסֶק בַּנִּסְתָּרוֹת אֲפִילּוּ כְּגַרְגִּיר חַרְדָּל / וְאֶרְאֶה כָּל־דַּרְשָׁן מִדַּבֵּר בָּם לֹא יֶחְדָּל / וְאִם לֹא אֶעֱשֶׂה כֵּן אֲנִי הַדַּל / הַקָּטוֹן שֶׁבָּהֶם מִמֶּנִּי יִגְדָּל.

רכב: דַּע כִּי רַבִּים מֵהֶם בְּהֵמָה הֵמָּה לָהֶם / וְגַם אַתָּה עֲשֵׂה כְמַעֲשֵׂיהֶם / וְדַבֶּר בָּם כַּאֲשֶׁר בִּלְבָבֶךָ / וֶאֱמֹר כִּי נָהִירִין לָךְ שְׁבִילֵי הַסּוֹדוֹת כִּשְׁבִילֵי עִיר מוֹשָׁבֶךָ / וְאִם לֹא אֵפוֹ מִי יַכְזִיבֶךָ / וּקְרָא בְגָרוֹן “שִׁמְעוּ עַמִּים כֻּלָּם / כִּי יֵשׁ־לִי סוֹד נִסְתָּר וְנֶעְלָם / אֲשֶׁר שָׁמַעְתִּי בִּמְּדִינַת עֵילָם / מִפִּי הַמְקֻבָּל הַגָּדוֹל ר' חִירָה מֵעִיר עֲדֻלָּם / שֶׁחִבֵּר סֵפֶר “חֲמוֹר עוֹלֶה בַסֻּלָּם” / מִי־הִקְשָׁה אֵלָיו וַיִּשְׁלָם / וּמִפְּנֵי שֶׁהוּא מִכִּבְשׁוֹנוֹ שֶׁל־עוֹלָם / הֲלֹא הוּא כָּמֻס עִמָּדִי / וְיִהְיֶה לִי לְבַדִּי / עַד אֶמְצָא אֵיזֶה יְחִידִי/ רָאוּי לָבֹא בְסוֹדִי”.

בענה: וְאִם יָבוֹא לְפָנַי דְּבַר אִסּוּר וְהֶתֵּר.

רכב: אִם תּוּכַל מֵהֶם הִסָּתֵר / וְאִם לֹא תּוּכַל לְהִתְעַלֵּם הַחֲרֵשׁ וְהִתְאַפֵּק / וֶאֱמֹר כִּי אַתָּה מְסֻפָּק / וְדַעְתְּךָ בַּדָּבָר נְבוּכָה / וְהַמַּחְמִיר תָּבוֹא עָלָיו בְּרָכָה.

בענה: וּבְלֶכְתִּי אֶל־הַיְשִׁיבָה / אֵיךְ אֶתְנַהֵג בָּהּ / כִּי לֹא אֵדַע הַוָּיוֹת אַבַּיֵי וְרָבָא / וַאֲנִי אָנָה אֲנִי־בָא.

רכב: יֵשׁ לַמָּקוֹם הַרְבֵּה דְרָכִים / וַאֲנִי אֶתֶּן־לְךָ בָּהֶם מַהֲלָכִים / אֱמֹר כִּי שָׁלְחוּ לְךָ שְׁאֵלוֹת מִתָּם / וְאַתָּה עָסוּק בִּתְשׁוּבוֹתָם / אוֹ אֱמֹר כִּי מֵרֹב עֲסָקֶיךָ לֹא נָתַתָּ לְעֵינֶיךָ שֵׁנָה / וּשְׁמֵעְתָּא בַעְיָא צַלּוּתָא כְּיוֹמָא דְּאִסְתָּנָא.

בענה: בְּךָ יְבָרֵךְ יִשְׂרָאֵל / כִּי הָיִיתָ לִּי לְאָב וְגוֹאֵל.

רכב: גַּם זֹאת עֲשֵׂה אִם בַּיְשִׁיבָה תֵשֵׁב / דַּעַת בַּעֲלֵי תְּרִיסִין הַקְשֵׁב / וּדְבַר־מָה קֹדֶם אַל־תָּשֵׁב / וְסִכְלוּתְךָ לַעֲנָוָה תֵחָשֵׁב / וְזֹאת הַדֶּרֶךְ אֲשֶׁר תָּבֹר לְעוֹלָם / נְטֵה אַחַר סִבְרַת הַגָּדוֹל שֶׁבְּכֻלָּם / וְלֹא תָשׁוּב דַּךְ נִכְלָם / וֶאֱמֹר “יֶחְדְּלוּן אַחִים וְרֵעִים / כִּי דִבְרֵי חֲכָמִים בְּנַחַת נִשְׁמָעִים / וּבֶאֱמֶת כִּי דִבְרֵי הָרַב פְּלוֹנִי בְּזֹאת הַמֵּימְרָא / סְבָרָא נְכוֹנָה וִישָׁרָה / וּמִמֶּנָּה אֵין לְהָמִישׁ”.

בענה: וְאִם יָשִׂימוּ הַחוֹלְקִים פְּנֵיהֶם כַּחַלָּמִישׁ / וְיֹאמְרוּ לִי בְּאַף וּבְזַעַם / “קוֹלְךָ כְּקוֹל הָרַעַם / אֱמָר־לָנוּ טַעַם / וְלֹא תֵלֵךְ עִמָּנוּ כְּפַעַם בְּפָעַם”.

רכב: עֲנֵה לָהֶם כְּבָר אָמַרְתִּי דַי וְהוֹתֵר / וְלֹא בָאתִי כָאן לְקַנְטֵר / כִּי כָל־יָמַי בֵּין חֲכָמִים עָמַדְתִּי / וּבְדַלְתוֹת מִדְרְשֵׁיהֶם שָׁקַדְתִּי / וּלְעָמְקָהּ שֶׁל־הֲלָכָה יָרַדְתִּי / וְלֹא עָמַדְתִּי עַל־הַנִּצּוּחַ / בְּדִבְרֵי הֶבֶל וְרַעְיוֹן רוּחַ / וּזְכוּרַנִי כִּי מִנְּעוּרַי שָׁלשׁ פְּעָמִים הוֹדֵיתִי לְדִבְרֵי חֲבֵרַי / וְגַם הֵמָּה פְּעָמִים רַבּוֹת הוֹדוּ לִדְבָרָי / וּלְמַעַן כְּבוֹדִי לֹא יִתְבַּזֶּה / לֹא אוֹסִיף עוֹד דַּבֵּר בַּדָּבָר הַזֶּה / וְלָכֵן בַּעֲנָה לֵךְ וֶאֱכֹל לַחְמְךָ בְּשִׂמְחָה / וְנָסוּ יָגוֹן וַאֲנָחָה / וּמֻבְטָח אֲנִי כִּי לַמּוֹעֵד הַזֶּה כָּעֵת חַיָּה / תִּקָּרֵא רַב בְּסִיַּעְתָּא דִּשְׁמַיָּא.

בענה: מִי־יִתֵּן וְהָיָה / וְאוּלָם בְּהִתְהַלְּכִי עֲצָתְךָ תַנְחֵנִי / וַהֲקִיצוֹתִי וְהִיא תְשִׂיחֵנִי / וְאֶמְצָא חֵן בְּעֵינֶיךָ כִּי נִחַמְתַּנִי / וְעַד־הֲלֹם הֲבִיאֹתַנִי / וְעַתָּה שִׁמְעָה שִׁירָתִי / וְהַקְשִׁיבָה רִנָּתִי:

יָדַי מְמִישׁוֹת

יוּשָׂם לְרַבָּן / רָאוּי לְדָרְבָן / אִישִׁים רְאוּיִם /

אֶל־מַחֲרֵשׁוֹת

אָסְפוּ בְחָפְנָם / רוּחַ וְגָרְנָם / תֶּבֶן וְאֵין כָּל־ /

בָּר בַּמְּדוּשׁוֹת

צֶאנָה זְמִירוֹת / פִּי בַחֲצוֹצְרוֹת / צֶאנָה מְהִירוֹת /

לָרוּץ וְחָשׁוֹת

צֶאנָה בְדִיצָה / שִׂפְתֵי מְלִיצָה / מִשִּׁיר וְרִנָּה /

וּשְׂחוֹק גְּדוּשׁוֹת

מִי יַאֲמִין כִּי / יַסְפִּיק לְצָרְכִּי / אַרְבַּע הֲגָדוֹת /

וּשְׁתֵּי דְרָשׁוֹת

פִּלְפּוּל גְּמָרָא / שִׁקּוּל סְבָרָא / נֶחְשָׁב לְהֶבֶל /

וּלְמַעֲקַשּׁוֹת

חָכְמוֹת עֲמֻקּוֹת / מִפֹּה מְחוּקוֹת / בַּחוּץ נְטוּשׁוֹת /

בַּטִיט רְטוּשׁוֹת

נַפְשִׁי עֲרֻמָּה / מִבִּין וְחָכְמָה / יֵשׁ יֵשׁ כְּמוֹתֵךְ /

הָדָר לְבוּשׁוֹת

יֵשׁ יֵשׁ מְנוּיוֹת / בִּכְלַל רְאוּיוֹת / לִפְאֵר וּמִמֵּךְ /

רָעוֹת וְטִפְּשׁוֹת

נֶפֶשׁ מְקֻבָּל / לֹא עוֹד תְּחֻבַּל / לֹא יֶחֱטָאוּ /

עַל־הַנְּפָשׁוֹת

כֻּלָּם בְּרִיאִים / גַּסִּים וְנָאִים / מֵרְאוֹת פְּנֵיהֶם /

אֵינָם כְּבוּשׁוֹת

כָּל־מַחְשְׁבוֹת רַב / הִנָּם לְהוֹן רַב / רַגְלָיו בְּרֶשֶׁת /

חֶמְדָּה יְקוּשׁוֹת

נִדְמוּ לְבִלְעָם / כָּל־הוֹן בְּבָלְעָם / וּשְׁמָם כְּמשֶׁה /

לִמְשֹׁךְ וְלִמְשׁוֹת

כָּל־בּוּר וְסָכָל / יָשַׂר וְיוּכַל/ לֹא עֵת לְשׁוֹנִי /

לִשְׁתֹּק וְלַחְשׁוֹת

הִנְנִי הֲגַם כִּי / שָׁפָל בְּעֶרְכִּי / אָדָם וְלֹא קוֹף /

לִמְאֹס וְלִנְשׁוֹת

מָתַי אֲבֹרַר / לִהְיוֹת כְּמַהְרַ"ר / הוֹד רַב כְּשֵׁעוּר /

שָׁלשׁ עֲדָשׁוֹת

אִמְרוֹת תְּחִנָּה / נָא תִֽהְיֶנָה / הַיּוֹם בְּלִבּוֹת /

מוֹרִים חֲרוּשׁוֹת

עֵינֵי גְאוֹנִים / אָזְנֵי נְבוֹנִים / הָיוּ עֲדֶנָה /

עִוְרוֹת וְחֵרְשׁוֹת

אָמְנָם בְּאֶהְיֶה / אֶבְטַח יְחַיֶּה / עַצְמוֹת כְּבוֹדִי /

אַף כִּי יְבֵשׁוֹת

אוֹחִיל לְחַסְדּוֹ / כִּי הוּא לְבַדּוֹ / פּוֹעֵל גְּבוּרוֹת /

עוֹשֶׂה חֲדָשׁוֹת.


רכב: יְשֻׂשׂוּם מִדְבָּר וְצִיָּה לְמִדְבָּרֶךָ / וְתָגֵל עֲרָבָה לַעֲרֵבוּת זְמִירֶךָ / וְעַתָּה לֵךְ לְשָׁלוֹם

לְאֹהָלֶךָ.

בענה: יָאֵר ה' פָּנָיו אֵלֶיךָ / וִיהִי שָׁלוֹם בְּחֵילֶךָ.
