שֶׁמֶץ שָׁמַעְתִּי

חתימה: שלם בירב יוסף משתא אלשבזי מבני פרחיה


שֶׁמֶץ שָׁמַעְתִּי בְּאָזְנַי

כִּי חָבְרוּ כָּל מַשְׂטִינַי

מֵאַיִן יָבוֹא עֶזְרִי עֶזְרִי מֵיְוֹיָ:


לוּלֵי יְדִידִי בְּרִית שָׁמַר

וְצִוָּה לִמְשִׁיחוֹ וְאָמַר

יְרֵה וַיּוֹר וַיֹּאמֶר

חֵץ תְּשׁוּעָה לַיוֹיָ:


מְחֵה יָהּ גְּזֵרַת צָר קָשֶׁה

וּזְכוּת תּוֹרָה אַל תְּנַשֶּׁה

וְזֹאת הַתּוֹרָה אֲשֶׁר שָׂם מֹשֶׁה

עֶבֶד יְוֹיָ:


בֶּן יְשׁוּרוּן יְמַלֵּא מְקוֹמוֹ

יָגִיל וְיִשְׂמַח בְּעוֹלָמוֹ

כְּמוֹ וַיַּעֲמֹד שְׁלֹמֹה

לִפְנֵי מִזְבַּח יְוֹיָ:


יוֹנָה לְבָנַיִךְ הַשְׁרִי

עֲבוּר אַבְרָהָם הָעִבְרִי

אָגִיל וְלֹא יֵבוֹשׁ שִׂבְרִי

בַּצַּר לִי אֶקְרָא יְוֹיָ:


רָשָׁע וַהֲמוֹנוֹ יֻדַּשׁ

וּמַלְכוּת דָּוִד תִּתְחַדַּשׁ

שְׂאוּ יְדֵיכֶם קֹדֶשׁ

וּבָרְכוּ אֶת יְוֹיָ:


בַּת נָדִיב לְשִׂיחֵךְ שִׁפְכִי

וְעַל בָּמוֹת אוֹיְבַי דִּרְכִי

לְחַדֵּשׁ כַּנֶּשֶׁר נְעוּרָיְכִי

עוֹשֶׂה צְדָקוֹת יְוֹיָ:


יוֹם פֶּרֶא מַבְאִישׁ גָּזַר

סִלַּק עֲטָרָה וְהַנֵּזֶר

וְאַבְנֵי אוֹהָלִי פִּזַּר

וָאֶתְחַנֵּן אֶל יְוֹיָ:


וְצוּרִי רַבִּים נִפְלְאוֹתָיו

מַזְכִּיר לְעַמּוֹ זְכוּת אֲבוֹתָיו

לִבְנֵי יִשְׂרָאֵל עֲלִילוֹתָיו

רַחוּם וְחַנּוּן יְוֹיָ:


סוֹמֵךְ נוֹפְלִים וּמְחַיֵּם

חַסְדּוֹ לְעַמּוֹ הוּא יְקַיֵּם

אַתֶּם נִצָּבִים הַיּוֹם

כֻּלְּכֶם לִפְנֵי יְוֹיָ


פּוֹדֶה עֲבָדָיו לְהַעֲלוֹתָם

לִנְוֵה אֶרֶץ נַחֲלָתָם

וְהַמַּטִּים עֲקַלְקַלּוֹתָם

יוֹלִיכֵם יְוֹיָ:


מִתֵּימָן מְאֹד נִבְלַמְנוּ

וּמֵעִיר אֶל עִיר נָדַדְנוּ

וַאֲנַחְנוּ נִמְלַטְנוּ

עָזְרֵנוּ בְּשֵׁם יְוֹיָ:


שָׁמְמוּ בָּתֵי כְּנָסִיּוֹת

וּמִקְדָּשׁ בָּנוּי לְתַלְפִּיּוֹת

עוּרִי חֶרֶב פִּיפִיּוֹת

חֶרֶב לַיוֹיָ:


תְּכַלִּי פֶּרֶא אֱוִיל עוֹיֵן

וְיִלְבַּשׁ קְלָלָה כַּשִּׁרְיוֹן

וְעָלוּ מוֹשִׁיעִים בְּהַר צִיּוֹן

הָהָר חָמַד יְוֹיָ:


אֶהֱמֶה בְּקוֹלִי וְאָשִׂיחַ

וְרוּחִי לְגוּפִי תַּקְשִׁיחַ

מְחַכֶּה לְבִיאַת מָשִׁיחַ

לִפְנֵי בּוֹא יוֹם יְוֹיָ:


אוֹיֵב הֲדַר הוֹדוֹ הָלַךְ

וְנִזְרוֹ לָאָרֶץ הוּשְׁלַךְ

וַיְגָרְשֵׁהוּ וַיֵּלַךְ

אָבָרְכָה יְוֹיָ:


לְעַמּוֹ יְקַבֵּץ הַנִּתְעָב

בְּעַמּוּד אֵשׁ וּבְעַמּוּד עָב

וּלְחַיּוֹתָם בָּרָעָב

נַפְשֵׁנוּ חִכְּתָה לַיוֹיָ:


שָׁוְא רָשָׁע גּוֹאֶה בְּחָזְקוֹ

וְתוֹבֵעַ עַם עָנִי לְעָשְׁקוֹ

כִּי בָהֵמָּה שָׁמֵן חֶלְקוֹ

קֹדֶשׁ יִשְׂרָאֵל לַיוֹיָ:


בֶּן אָמָה בְּתַרְמִית אָתָה

מַיִם זְדוֹנִים שָׁתָה

וְאֵשׁ זָרָה לְחֵילוֹ חָתָה

וַתִּבְעַר בָּם אֵשׁ יְוֹיָ:


זָהַת דַּעְתּוֹ בְּמַשְׂטֵמָה

לְהַטִּיל עָלֵינוּ אֵימָה

הַבִּיטוּ אָרְחוֹת תֵּימָא

נַפְשֵׁנוּ חִכְּתָה לַיוֹיָ:


יְשׁוּעַת אֵל שַׁדַּי תִּרְאוּ

מִמַּעֲמַקִּים בְּקוֹל קִרְאוּ

עַתָּה טַעֲמוּ וּרְאוּ

כִּי טוֹב יְוֹיָ:


מִשִּׁינָה יְדִידִי יִיקַץ

יָעִיר גְּאֻלָּה בַּאֲתַּתְקַץ

כִּי פֶּרֶא מְנָאֵץ שִׁקַץ

הַבּוֹטְחִים בַּיְוֹיָ:


בּוֹאוּ נָא אֶל נַחֲלַתְכֶם

זֶרַע יַעֲקֹב בְּהַדְרַתְכֶם

אָהַבְתִּי אֶתְכֶם

אָמַר יְוֹיָ:


נַפְשִׁי הִתְחַטְּאִי בְּאֵזוֹב

גּוֹאֲלִי חָזָק לֹא יַעֲזוֹב

כָּל הָאָדָם כּוֹזֵב

מָה אָשִׁיב לַיְוֹיָ:


יוֹצְרִי תָמִיד אֲנִי שׁוֹאֵל

יְקַבֵּץ אֶת גּוֹלַת אֲרִיאֵל

וְאָמַרְתָּ הָרֵי יִשְׂרָאֵל

שִׁמְעוּ דְּבַר יְוֹיָ:


פִּתְאוֹם יָבוֹא יְדִיד נַפְשִׁי

יַעֲלֵנִי אֶל הַר קָדְשִׁי

וְשֶׁמֶשׁ צְדָקָה שִׁמְשִׁי

אוּלַי יֶחֱנַן יְוֹיָ:


רֶגֶל מְשִׁיחִי בְּחֵן צוֹעֵד

לַאֲיֻמָּתִי הוּא מִסָּעֵד

וְגוֹיֵךְ לֹא תַכְשְׁלִי עוֹד

נְאֻם יְוֹיָ:


חֲיִי חֲבַצֶּלֶת בְּחֶסֶד אֵל

וְהַעֲלִי גֵּזַע אִיתִיאֵל

לָכֵן הָרֵי יִשְׂרָאֵל

שִׁמְעוּ דְּבַר יְוֹיָ:


יִשְׂמְחוּ כָּל בַּעֲלֵי מוּסָר

כִּי טוּב יְוֹיָ לֹא יֶחְסָר

וְרָאוּ כָּל בָּשָׂר

כִּי אֲנִי יְוֹיָ:


הַדְרַת שְׁבָטֵינוּ תְּבוֹרַךְ

נִזְכֶּה לְיוֹם יָבוֹא חַדְרַךְ

אַשְׁרֵי תְמִימֵי דָרֶךְ

הַהוֹלְכִים בְּתוֹרַת יְוֹיָ:
