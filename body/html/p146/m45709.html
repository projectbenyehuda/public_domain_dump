אֵל הַנּוֹרָא

חתימה: אני שלם שבזי משתא


אֵל הַנּוֹרָא וְקָדוֹשׁ שְׁמוֹ / עוֹשֶׂה אֶרֶץ וְשָׁמַיִם

וְהוּא אֵל חַי מְקוֹם עוֹלָמוֹ / שׁוֹקֵל הַכֹּל בְּמֹאזְנַיִם

וּבְכוֹחַ שֵׁם תַּעֲצוּמוֹ / יָסַד אֶרֶץ עֲלֵי מָיִם

וּמֵאֵין יֵשׁ בָּרָא יְקוּמוֹ / זָכָר וּנְקֵבָה שְׁנָיִם

וְאָדָם תַּכְלִית חוֹתָמוֹ / וְלִבּוֹ כְּמוֹ קָו בִּנְתָיִם

וְזֶרַע יַעֲקֹב סוֹד טַעְמוֹ / אִישׁ תָּם וּנְקִי כַפַּיִם

בָּחַר בְּזַרְעוֹ וֶהֱקִימוֹ / וְהוֹצִיאוֹ מִמִּצְרַיִם

כִּי נִחַם יְוֹיָ עַמּוֹ / גָּאַל יְרוּשָׁלַיִם:


נִאֵץ רָשָׁע וְהִרְשִׁיעַ / וְהֵצַר לָהֶם בְּמַלְכוּתוֹ

תָּקַע אָהֳלוֹ וְהִצִּיעַ / בַּעֲרָב לְהַרְאוֹת גְּדֻלָּתוֹ

הִכְרִיז לְהָרַע וְהוֹדִיעַ / מְגִמַּת לְבָבוֹ וְתַחְרוּתוֹ

לְהַחֲלִישׁ שְׂרִידַי לְהַכְנִיעַ / וְהִחְרִיב אָהֲֳלִי בְּאַחְרִיתוֹ

וְחִשַּׁב שְׁאֵין לִי מוֹשִׁיעַ / מְפַתֵּנִי לַחְמוֹד דָּתוֹ

זִלְזַל יְקָרִי וְהִגְרִיעַ / וְקָרַב עֵת קֵץ נְפִילָתוֹ

אוֹהֵב לְכֶסֶף וּמַטְמִיעַ / גּוֹאֶה וּמַקְשֶׁה גְּזֵרָתוֹ

מַגִּיד מִדָּן וּמַשְׁמִיעַ / אָוֶן מֵהַר אֶפְרַיִם:


יִשְׁרֵי לֵב הַנֶּאֱמָנִים / שִׁיתוּ לִבְּכֶם לְחֵילָה

הָעֲנִיִּים וְהָאֶבְיוֹנִים / כֻּלְּכֶם יוֹשְׁבֵי הַגּוֹלָה

תַּעֲלוּ לְצִיּוֹן בִּרְנָנִים / וְיָאִיר צוּרִי הָאֲפֵלָה

וְיֹאבְדוּ גּוֹיִם חַמָּנִים / יִשְׁתּוּ כּוֹס הַתַּרְעֵלָה

בְּבוֹא מְטַטְרוֹן שַׂר הַפָּנִים / וְיִדְלֶה נֶפֶשׁ צוֹלֵלָה

וְכֵלִים מִכֵּלִים שׁוֹנִים / תִּלְבַּשׁ הָעִיר הָאֲבֵלָה

יַאְזִין צוּרִי תַּחֲנוּנִים / וְיִרְצֶה תְפִלָּתִי כְּעוֹלָה

לֹא הִפְנוּ אָבוֹת אֶל בָּנִים / מֵרִפְיוֹן יָדַיִם:


שׁוּב יְשׁוּרוּן מֵחֲטָאֶיךָ / כִּי זֶה עִתָּךְ עֵת דּוֹדִים

כִּי אֵל שַׁדַּי בְּעֶזְרֶךָ / לַעֲלוֹת הַר צִיּוֹן גְּדוּדִים

לֵךְ עַמִּי בּוֹא בַּחֲדָרֶיךָ / עַד יַעֲבָר-זַעַם זֵדִים

וְהָיָה שַׁדַּי בְּצָרֶיךָ / בְּשׁוּחָה יִהְיוּ נִלְכָּדִים

בְּטַח בַּיְוֹיָ רוֹפְאֶךָ / אוֹהֵב צְדָקָה וַחֲסָדִים

יִתֵּן לְךָ כִּלְבָבֶךָ / בַּעֲבוּר אָבוֹת נִכְבָּדִים

וְיָדַעְתָּ אֶת נָוֶךָ / יַעֲלוּ דְּגָלֵינוּ יְדִידִים

וְזָרַח בַּחֹשֶׁךְ אוֹרֶךָ / וַאֲפֵלָתְךָ כַּצָּהֳרַיִם:


לָבְשׁוּ מְשַׂנְאֵינוּ פַּחְדָּם / מִיּוֹם בָּאָה בְּשׂוֹרָתִי

וְעָמַד פֶּרֶא וְהֵזִידָם / לְהָצֵר לִי בְּיוֹם גָּלוּתִי

וְגָזַר לִפְשׁוֹט אֶת יָדָם / לְהַחְרִיב אֶת בֵּית תְּפִלָּתִי

יַחַד יָבוֹא יוֹם אֵידָם / וְתִתְעוֹרֵר עֵת יְשׁוּעָתִי

וְיָקִיצוּ כָּל הַנִּרְדָּם / לַעֲלוֹת אֶל בֵּית מְנוּחָתִי

וְיִרְאוּ צִיּוֹן בְּעָמְדָם / אֲשֶׁר שָׁם הוּא חֵשֶׁק מְנָתִי

וְיַשִּׂיגוּ אֶת כָּל סוֹדָם / וְיִשְׁתּוֹמְמוּ בְּנֵי אֲמָתִי

וְהִנֵּה אֵין הָאָדָם / וְאֵין עוֹף הַשָּׁמַיִם:


מִתּוֹךְ עָרִים הַפְּרוּזוֹת / הִקָּבְצוּ הַנִּפְזָרִים

בִּשְׁנַת הַיּוֹבֵל הַזֹּאת / נַעֲלֶה לְעָרֵי מִבְצָרִים

יִזְכּוּ שְׁבָטֵינוּ לַחֲזוֹת / חֶמְדַּת יְוֹיָ מְרוֹם הָרִים

תָּגִיל נְפָשׁוֹת הַנְּמִבְזוֹת / בֵּין בַּעֲלֵי רִיב הַצָּרִים

וְעַנְנֵי כָבוֹד מִתְפַּרְשְׁזוֹת / עַל בְּנֵי צִיּוֹן הַיְקָרִים

וְעָרֵי אֱדוֹם וּבָצְרָה בְזוּזוֹת / וּמִגְדְּלֵי פֶרֶא מֻבְעָרִים

וְצִנּוֹרֵי אֵשׁ בָּם נוֹתְזוֹת / תְּכַלֶּה זְקֵנִים וּנְעָרִים

מֵאֵת יְוֹיָ הָיְתָה זֹאת / שִׂמְחוּ אֶת יְרוּשָׁלָיִם:


שׁוּב עַמִּי לִגְדּוֹר פֶרֶץ / לְחַלּוֹת פְּנֵי צוּרִי וְאָבִי

וּלְהִנָּצֵל מִקֶּרֶץ / תִּרְמוֹס עַל שַׁחַל וְלָבִיא

וְלֹא גּוּר אַרְיֵה וְלֹא שַׁחַץ / יֶהֱמֶה בְּאַרְצִי וּמוֹשָׁבִי

חֲזֵה דָּת אֵל בְּעִיּוּן נִמְרָץ / כִּי דָרַךְ מְאוֹר כּוֹכָבִי

וְעָמַד מְשִׁיחִי בֶּן פֶּרֶץ / וּבָא אֵלִיָּה הַנָּבִיא

וְעֵינֵי פֶרֶא אֲשֶׁר קָרַץ / תֶּחְשַׁךְ בְּשׁוּב נֵר מַעְרָבִי

שִׁמְעִי וְהַאֲזִינִי אֶרֶץ / וְאִמְרִי לְשׁוּלַמִּית שׁוּבִי

יִתְיַצְּבוּ מַלְכֵי אֶרֶץ / בְּקוֹל צְעָקָה מֵחֲרוֹנִים:


בְּנֵי אָמָה בָּא עֵת זְמַנָּם / יֶהֱמוּ יַחְדָּו בְּהֵיכְלֵיהֶם

כִּי יֵצְרוּ צַעֲדֵי אוֹנָם / רוּחַ שְׂעָרָה תִּרְדְּפֵיהֶם

שָׂרִים יִפְּלוּ מִכַּנָּם / וְאֵשׁ תְּלַהֵט סְבִיבוֹתֵיהֶם

נָפַל שָׂרָם וְאוֹפַנָּם / פַּחַד וְאֵימָה תִּפּוֹל עֲלֵיהֶם

וְעַמִּי הַקּוֹרְאִים לְקוֹנָם / יְוֹיָ צְבָאוֹת יַעְזְרֵיהֶם

יִזְכּוּ וְיֵצֵא לָאוֹר דִּינָם / וְחֵן תּוֹרָה יָאִיר פְּנֵיהֶם

וְאוֹיְבַי יִתְגַלֶּה קְלוֹנָם / וְיִנָּגְפוּ תּוֹךְ אָהֳלֵיהֶם

לְמִי פְּצָעִים חִנָּם / לְמִי חַכְלִילוּת עֵינָיִם:


זַמְּרִי יְחִידָה בְּאַשְׁמוֹרֶת / פִּצְחִי בְּרִנָּה וְקוֹל תּוֹדָה

בַּשְּׂרִי אֲיֻמָּה שְׁחַרחוֹרֶת / כִּי בָא פֶרֶץ בֶּן יְהוּדָה

דֶּרֶךְ גְּאוּלִים מְיֻשֶּׁרֶת / רוּץ נָא וּמַהֵר צִיר תְּעוּדָה

עוּרָה מִגְּבוּל צַמֶּרֶת / וּפְרוֹשׂ לְבֶן אָמָה מְצוּדָה

בַּטֵּל גְּזֵרָה אֲשֶׁר חָרַת / רַגְלוֹ בְּפַח מוֹקֵשׁ לְכוּדָה

תָּשׁוּב לְמַלְכִּי הָעֲטֶרֶת / מִפָּז וּמִזָּהָב חֲמוּדָה

עַל הַר עֲלִי לָךְ מְבַשֶּׂרֶת / צִיּוֹן אֲשֶׁר בְּחֵן אֲפוּדָה

אֶבֶן יְקָרָה לְתִפְאֶרֶת / וְהַזָּהָב זְהַב פַּרְוָיִם:


יָבוֹא נָא מְשִׁיחַ צִדְקִי / וִיקַבֵּץ אֶת נְדוּחֵינוּ

הוּא כָּל תַּאֲוָתִי וְחֶשְׁקִי / יִהְיֶה שָׁלוֹם בְּעִירֵנוּ

אֶחֱסֶה בְּצֵל יוֹצְרִי חֶלְקִי / יַעֲלֶה יְוֹיָ בְּרֹאשֵׁינוּ

עַל יְרֵכֵךְ בַּת אֱדוֹם סִפְקִי / כִּי בָאָה יְשׁוּעָתֵנוּ

כִּי אַתְּ בְּאֵשׁ יַעֲקֹב תִּלְקִי / וּבְאֵשׁ יְהוֹסֵף יְדִידֵנוּ

רַעְיָה לְרַגְלֵי צְבִי נִשְׁקִי / יָחִישׁ וְיִקְשׁוֹר דְּגָלֵינוּ

אֶשְׂמַח וִיפַתְּחָה שַׂקִּי / יֹאמַר: עֲלוּ עִם מְשִׁיחֵנוּ

וְאֶשְׁבּוֹר עָלָיו חֻקִּי / וְאָשִׂים בְּרִיחַ וּדְלָתָיִם:


מִתּוּגַת פֶּרֶא נְמִבְזֶה / בְּתֵימָן יוֹם בָּא נַחְבֵּאתִי

נַמְתִּי: הַאֶחְיֶה מֵחֳלִי זֶה / כִּי מֵהֲמוֹנוֹ יָרֵאתִי

וְנִעֵר שִׂכְלִי הַלָּזֶה / וְקֵרַר דַּעְתִּי אֲנִי וּבֵיתִי

וְאָמַר לִי: קוּם עֲלֵה מִזֶּה / כִּי בְאֶרֶץ עֲרָב נִטְמֵאתִי

פִּתְאוֹם יָבוֹא דְּבַר חוֹזֶה / יְרַפֵּא לִבִּי בַּחֲלוֹתִי

וְעַתָּה רוּץ מַה לְּךָ בָּזֶה / וּלְמֵי יְשׁוּעָה צָמֵאתִי

חֲזִיר וּפֶרֶא יַחַד תַּרְזֶה / וַאֲנִי בְּדִינָם הֶחֱשֵׁיתִי

וְלַנָּשִׂיא מִזֶּה וּמִזֶּה / שׁוֹדֵד בַּצָּהֳרַיִם:


שׁוֹמְרֵי דָּת אֵל אַשְׁרֵיהֶם / שְׂרִידִים אֲשֶׁר יְוֹיָ קוֹרֵא

רוּחַ אֱלֹהִים תְּנִיחֵיהֶם / הוּא הַיּוֹצֵר וְהַבּוֹרֵא

חֶסֶד אֵל הוּא מְסוֹבְבֵיהֶם / וְצַדִּיק מַגִּין עֲלֵי דוֹרוֹ

צַד הַשְּׂמֹאל לֹא יְנִידֵיהֶם / כִּי הוּא כַּתַּנִּין בְּתוֹךְ יְאוֹרוֹ

כִּי לֹא יָנוּם שׁוֹמְרֵיהֶם / מֵַשְׁקִיט לְפֶתֶן בְּגֵיא חוֹרוֹ

וּבְנֵי יְשׁוּרוּן יְחַיֵּהֶם / יָאִיר פְּנֵיהֶם בְזִיו אוֹרוֹ

עוּרָה יְדִידִי אֱמוֹר לָהֶם: / וְהַעֲמַדְתָּ לָהֶם כָּל מָתְנָיִם:


תִּפְאֶרֶת בָּנִים אֲבוֹתָם / בְּיֵחוּד צוּרִי יִתְהַלְלוּ

כִּי הֵם עוֹמְדִים בִּזְכוּתָם / וּבִזְכוּת […]
