אֲיֻמָּה לִבְשִׁי עֻזֵּךְ

חתימה: אלשבזי משתא


אֲיֻמָּה לִבְשִׁי עֻזֵּךְ / בַּת נָדִיב הַצְּנוּעָה

וַעֲלִי אֶל הַר הַמּוֹר / וְשִׁמְעִי שׁוֹפַר תְּרוּעָה

וְאִמְרִי לְדוֹדֵךְ יִזְכּוֹר / זְכוּת אָבוֹת וּשְׁבוּעָה

וּשְׁנֵים עָשָׂר שְׁבָטִים / וְכָל מַעְתִּיקֵי שְׁמוּעָה

לְעַם עָנִי יוֹשִׁיעַ / בְּעֵת רָצוֹן הַיְּדוּעָה

עָזִי וְזִמְרַת יָהּ / וַיְהִי לִי לִישׁוּעָה:


לִבְשִׁי עֻזֵּךְ צִיּוֹן / בַּחֲצוֹת לַיִל עוּרִי

וְעוֹרְרִי בַּעְלֵי חֶזְיוֹן / וְאֵלַי בְּאַהֲבָה סוּרִי

וּבְשֵׂכֶל הָרַעְיוֹן / אַשְׂכִּיל בְּיוֹפִי צוּרִי

אָשִׁיר וְאֶשְׂמַח עַל יֵין / וְאֶנְשֶׁה עוֹל תַּמְרוּרִי

יְהֶא לַעֲדָתִי רִתְיוֹן / יְקַנֵּא יָהּ בַּעֲבוּרִי

וִיזַכֶּה אֶת נַפְשִׁי / לַחֲזוֹת גַּנָּה נְטוּעָה:


שַׁאֲלוּ חֶבְרַת קְדוֹשִׁים / אֶת שְׁלוֹם יְרוּשָׁלַיִם

מֵרֹאשׁ הַשָּׁלִישִׁים / אֲשֶׁר בֵּין אֵשׁ וּמַיִם

וּרְאוּ מַעֲלוֹת שְׁלוֹשִׁים / וּמַסֶּכְתוֹת כִּפְלַיִם

אָז נֵחָלֵץ חוּשִׁים / וּשְׁכִינָה בִּנְתַּיִם

אֲלֵיכֶם אֶקְרָא אִישִׁים / בַּעֲלֵי שֵׁשׁ כְּנָפַיִם

אֲשֶׁר בְּפֶתַח עֵינַיִם / לָכֶם לִשְׁכָּה יְדוּעָה:


בּוֹאוּ יְדִידַי הַיּוֹם / נְחַלֶּה פְּנֵי צוּרֵנוּ

הַגּוֹזֵר וּמְקַיֵּם / וְגוֹלֶה מַצְפּוּנֵנוּ

אוּלַי נִמְצְאָה פִדְיוֹם / לְהַחֲיוֹת נַפְשׁוֹתֵינוּ

כִּי בִרְצוֹנוֹ יְחַיֵּם / וִיכַפֵּר בַּעֲדֵנוּ

אֵל חַי נוֹרָא וְאָיוֹם / יַאֲזִין תְּפִלָּתֵנוּ

וְיַשְׁלִיךְ אֶת אוֹיְבֵינוּ / אֶל אֶרֶץ לֹא זְרוּעָה:


זְרִיזֵי הַלֵּב עִמְדוּ / בְּפֶתַח יָם כִּנֶּרֶת

וְלֵאלֹהֵיכֶם עִבְדוּ / כְּמוֹ שׁוֹמְרֵי מִשְׁמֶרֶת

וְדַלְתֵי תְּשׁוּבָה שִׁקְדוּ / בְּנֶפֶשׁ מִתְיַשֶּׁרֶת

וְתוֹרַת מֹשֶׁה לִמְדוּ / וְלָכֶם חֵן מַזְכֶּרֶת

וְאֶת נָוֵיכֶם פִּקְדוּ / בְּדָת מַלְכוּת וְתִפְאֶרֶת

וְיוֹם כִּפּוּר מְטַהֶרֶת / חָלְיֵי נֶפֶשׁ נְגוּעָה:


יוֹם עֲווֹן עֵגֶל הוּסַר / מִשְׁמֶרֶת קָבַעְתִּי

סוֹדוֹ לְמֹשֶׁה נִמְסַר / וּבִזְכוּתוֹ טָהַרְתִּי

וְלוֹ בִּמְחִילָה בִּשַּׂר / וְאָמַר לוֹ: סָלַחְתִּי.

וּמִצְווֹת שֵׁית מְאָה וְחַד סַר / שְׁתַּיִם זוּ שָׁמָעְתִּי

וּמַעְבַּר יָם גִּינוֹסַר / וְדָרוֹמָה יָרַשְׁתִּי

וּבְיוֹם זֶה חָמַדְתִּי / מִן הַבִּינָה שְׁפוּעָה:


מֵחֶשְׁקַת דּוֹד יָפֶה / לֹא טָעַמְתִּי שֵׁינָה

וַעֲנַן יְקָרוֹ חוֹפֶה / לְמַעְלָה מִצִּיּוֹנָה

וּשְׂמֹאל מַבְאִישׁ רָפֶה / וְנַפְשִׁי בְּתִקְוָה חוֹנָה

וְלִבִּי רְצוֹן אֵל צוֹפֶה / אֶל דַּעַת עֶלְיוֹנָה

וּבְתוֹרָה שֶׁבְּעַל פֶּה / מְנָתִי בָּרִאשׁוֹנָה

וְאִם בִּמְלָאכָה פּוֹנָה / לָהּ מִשְׁמֶרֶת קְבוּעָה:


שָׁרְשִׁי וְצֶמַח גִּזְעִי / בְּחַסְדּוֹ הָאֵל יַגְבִּיר

יְוֹיָ אוֹרִי וְיִשְׁעִי / וּפָנָיו לְעַבְדּוֹ יַסְבִּיר

הוּא מְצוּדָתִי וְסַלְעִי / חֲרוֹן אַף מֶנִּי יַעְבִיר

וְיִדְלֶה מִיָּם טָבְעִי / וְרוּחִי בְשִׂכְלִי יַחְבִּיר

וְאֶזְכֶּה וְיִירַשׁ זַרְעִי / חֶלְקוֹ בְּצִיּוֹן וּדְבִיר

וּגְאוֹן אוֹיְבִים יַשְׁבִּיר / וְיָחִישׁ לְעַמּוֹ תְּשׁוּעָה:


תִּכָּתְבוּן עַם יְוֹיָ / יוֹם זֶה בְסֵפֶר מְחִילָה

עֲבוּר מֹשֶׁה וּזְקֵנָי / נֶחֱזֶה שִׂמְחָה וְגִילָה

בְּמִנְחַת כִּפּוּר עֲווֹנִי / הַסְּמוּכָה לַנְּעִילָה

רַצִּי נָא עַל הֲמוֹנָי / בַּת נָדִיב הָאֲצִילָה

אוּלַי יִשָּׂא פָנַי / וְלִי חַסְדּוֹ יַקְבִּילָה

קְדוֹשָׁיו יַצִּילָה / בְּגַלְגַּל חָשׁ בִּתְנוּעָה:


אַהֲבַת קוֹנִי תַּנְהִיר / שְׁבָטִים הַנִּפְזָרִים

בְּאוֹר זָרוּעַ בָּהִיר / קֶרֶן מְשִׁיחוֹ יָרִים

וְכוֹכָבוֹ הַמַּזְהִיר / וְקוֹל מְבַשֵּׂר בְֶּהָרִים

וְכֹהֵן גָּדוֹל מָהִיר / בְּקָרְבַּן אֵלִים וּפָרִים

וְיַשְׁפִּילָה לֵץ יָהִיר / וְיִכָּלְמוּ כָּל זָרִים

רְכַב מַלְכִּי בְּמֵישָׁרִים / מֶרְכְּבוֹתֶיךָ יְשׁוּעָה:


נוסח אחר של שתי החרוזות הקודמות:


תִּכָּתְבוּן עַם יְוֹיָ / יוֹם זֶה בְסֵפֶר מְחִילָה

עֲבוּר מֹשֶׁה וּזְקֵנָי / נֶחֱזֶה בָּהֶם יוֹם וָלַיְלָה

וְנִתְבָּרֵךְ בְּפִי כֹהֲנָי / שַׁחְרִית וּמוּסָף וּנְעִילָה

וְנַגִּיד תְּהִלּוֹת יְיֹיָ / בְּעֶלֶז חֵן וְצַהְלָה

וּבִזְכוּת חָזָה סֻלָּם / יְקַבֵּץ עַמּוֹ וְיוֹשִׁיעָה:


אֱלֹהֵינוּ וֵאלֹהֵי אֲבוֹתֵינוּ / זְכוֹר חַסְדָּךְ לְבָנֶיךָ

שָׁמְרֵנוּ בְּבוֹאֵנוּ וּבְצֵאתֵנוּ / וְרַחֵם עַל שְׁאֵרִית צֹאנֶךָ

וְזוּנֵנוּ וּפַרְנְסֵנוּ /לְמַעַן זֶה שֶׁחִלָּה פָנֶיךָ

וְאֶרֶץ קָדְשְׁךָ הַעֲלֵנוּ / לִבְנוֹת מִשְׁכַּן מְעוֹנֶךָ

וְתִפְרַח כַּחֲבַצֶּלֶת אַרְצֵנוּ/ וּמֵהַרְרֵי אֵל בְּשִׁיר תּוֹדָה נַבִּיעָה:
