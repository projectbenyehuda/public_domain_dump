אַשְׁרֵי נֶפֶשׁ

חתימה: אלשבזי


אַשְׁרֵי נֶפֶשׁ מְחַכָּה / וְתָעִיר גּוּף דָּל נִרְדָּם

לְזֶרַע יַעֲקֹב מְזַכָּה / וְשַׁעֲרֵי תְשׁוּבָה לְנֶגְדָּם

וּבַת נָדִיב הַמַּלְכָּה / מַצְלַחַת יְמֵי חֶלְדָּם

לְוִדּוּי דְּבָרִים

יְחַוּוּן יְשָׁרִים

בְּפִתְחֵי שְׁעָרִים

מַגִּידִים אֶת סוֹדָם

הַזְכִּירֵנִי

לְשִׁיר הֶגְיוֹנִי

שׁוֹכֵן מְעוֹנִי

כִּי אֲנִי בָשָׂר וָדָם

תָּשֵׁב אֱנוֹשׁ עַד דַּכָּא / וְתֹאמַר שׁוּבוּ בְנֵי אָדָם:


אָדָם לְנֶפֶשׁ קָרָא / מֶלֶךְ נוֹרָא מִכָּל שָׂר

וְגוּף מִיּוֹם אֲשֶׁר בָּרָא / קְרָאוֹ בְתוֹרָתוֹ בָשָׂר

וְשֵׂכֶל אֱנוּשִׁי בוֹ שָׂרָה / וְדַעַת וְדִבּוּר לֹא יֶחְסָר

יְדִידִי מַהֶר

וְעַיֵּן זוֹהַר

בְּחֵן וּבְטוֹהַר

בְּפִסּוּק מָלֵא לֹא יֶחְסָר

וְתִמְצָא טַעַם

בְּטוֹב וּבְנוֹעַם

כְּפַעַם בְּפַעַם

בְּפִסּוּק עַל בְּשַׂר אָדָם שָׂר

וְאַשְׁרֵי נְכוֹנֵי דַעַת / לֵידַע סוֹד תַּלְמוּדָם:


לֵידַע תְּחִלָּה בְנַפְשׁוֹ / כִּי הוּא מִמְּקוֹר חַיִּים

וְאִם הֶחְטִיאָה לְעָנְשׁוֹ / בְּהַבְלֵי דָת אֱמוֹרִיִּים

בְיִסּוּרִין יַכְתִּישׁוֹ / וְיִגְלֶה לְיַמִּים וְאִיִּים

וְאִם שָׁב לַחְזוֹר

אַחַר מָזוֹר

שִׂכְלוֹ עוֹזֵר

בִּהְיוֹת מַעֲשָׂיו רְצוּיִים

שְׁבָטַי הִכּוֹנוּ

וְאַל תִּתְמַסְכֵּנוּ

וְלָבֶטַח תִּשְׁכּוֹנוּ

וְאַתֶּם תִּהְיוּ נְקִיִּים

וְעוֹזְבֵי יְוֹיָ יִכְלוּ כִּי קָרוֹב יוֹם אֵידָם:


שֵׁם טוֹב לַצַּדִּיקִים / עוֹשֵׂי רְצוֹן קוֹנֵיהֶם

מִכָּל עָווֹן מְנֻקִּים / וּשְׁכִינַת אֵל עֲלֵיהֶם

לוֹמְדֵי מִצְווֹת וְחֻקִּים / שֵׂכֶל טוֹב לְכָל עוֹשֵׂיהֶם

וְגוֹמְלִים חֲסָדִים

לַעֲנִיִּים מְרוּדִים

מַקְדִּישִׁים וּמְיַחֲדִים

לַיוֹיָ אֱלֹהֵיהֵֶם

וַאֲנִי בְתוֹכָם

עֲפַר רַגְלֵי חָכָם

עָנָו הַמַּדְרִיכָם

יְהֶא חֶלְקִי עִמָּהֶם

וְהַמַּשְׂכִּילִים יַזְהִירוּ / וּשְׂמֹאל מַבְאִישׁ לֹא יְנִידֵם:


בִּינוּ זֹאת כָּל הָעַמִּים / כִּי אֲנִי לְצוּרִי בָחַרְתִּי

גּוֹאֲלִי מִכָּל הַקָּמִים / חַשְׁתִּי וְלֹא הִתְמַהְמָתִי

הֶעֱבִירַנִי בְתוֹךְ יַמִּים / וּמִיַד אוֹיֵב נִמְלַטְתִּי

וּמִי יְפַתֵּנִי

אוֹ יְסִיתֵנִי

לְהַבְאִישֵׁנִי

בְּדָתוֹ אֲשֶׁר בּוֹ מָאַסְתִּי

וְהֵיךְ דָּת מֹשֶׁה

אֲדוֹנִי אֶנְשֶׁה

וְהוּא שֶׁהֻרְשָׁה

בְּסִינַי וַאֲנִי שָׁמַעְתִּי

תּוֹרָה בְּקוֹלוֹת וּבְרָקִים / וְכָל מְלָכִים רָפוּ יָדָם:


זְדוֹנִי יִמְחוֹל וְשִׁגְיוֹנִי / צוּרִי אֲשֶׁר לוֹ חָטָאתִי

בְּמַעֲשֵׂה יָדַי וּלְשׁוֹנִי / נְבָלָה וְשֶׁקֶר טָפַלְתִּי

וְאָבַד כַּסְפִּי וְכָל הוֹנִי / וּבָנַי וּבֵיתִי עָזַבְתִּי

וּבְדֶרֶךְ נֵס

אֲנִי מִתְפַּרְנֵס

וּפֶרֶא אוֹנֵס

מִגְּזֵרוֹתָיו בָּרַחְתִּי

מַחֲזִיק בְּשֶׁקֶר

וְזִלְזַל יוֹקֶר

לְהוֹנִי הֶפְקֵר

וּבְמַלְכוּתוֹ נִכְלַמְתִּי

אַךְ בַּיוֹיָ בָּטַחְתִּי / יִגְעַר בְּאוֹיְבַי וְיַפְרִידָם:


יְצַוֶּה חַסְדּוֹ רָם וְנִשָּׂא / בִּזְכוּת עֲנִיִּים נִפְרָדִים

וִירַחֵם עָם הַמְנֻסָּה / יָקִים דְּבַר חוֹזִים וְצוֹפִים

וּבְצִלּוֹ תָלִין הֲדַסָּה / תְּקַבֵּץ פְּזוּרִים נִכְסָפִים

נוֹרָא עֲלִילָה

יְמַהֵר מְחִילָה

לַאֲיֻמָּה אֲצִילָה

נוֹצֵר חֶסֶד לַאֲלָפִים

וְיָחִישׁ מְשִׁיחוֹ

כְּצִיץ יַזְרִיחוֹ

וּלְבֵית מְנוּחוֹ

נַעֲלֶה כִשְׂרָפִים

יְקַבֵּץ פְּזוּרֵי עַמּוֹ / אֶל בֵּית וַעֲדָם:
