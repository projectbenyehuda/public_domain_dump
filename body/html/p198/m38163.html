וַיֵּלֶךְ הָאוֹפָן : מקאמה

בִּגְנוּת הָאִשָּׁה וְהַנִּשּׂוּאִין


וַיֵּלֶךְ הָאוֹפָן כַּאֲשֶׁר כִּלָּה נְאוּמוֹ / וְתַחְכְּמֹנִי שָׁב לִמְקוֹמוֹ / וַיִּקְרָא לְזֶרַח בְּנוֹ / וַיְשִׂימֵהוּ עַל יַד יְמִינוֹ / וַיֹּאמֶר לוֹ: הִנֵּה נָא זָקַנְתִּי / לֹא יָדַעְתִּי יוֹם מוֹתִי / וְעַתָּה, בְנִי, שְׁמַע בְּקוֹלִי וְקַבֵּל עֲצָתִי, וְהַחֲזֵק בַּמּוּסָר אַל תֶּרֶף יָדֶךָ / שְׁמַע לְאָבִיךָ זֶה יְלָדֶךָ / אִם תִּהְיֶה צַוָּאָתִי עַל לוּחַ לִבְּךָ חֲרוּשָׁה: אַל תִּקַּח לְךָ אִשָּׁה / שָׁלוֹם בְּפִיהָ, הַוּוֹת בְּקִרְבָּהּ / לֹא יִנָּקֶה כָּל הַנֹּגֵעַ בָּהּ / שֵׁשׁ וָמֶשִׁי תַּלְבִּישֶׁנָּה / וּמַעֲדַנִּים תַּשְׂבִּיעֶנָּה / וְאִישׁ אַחֵר יִשְׁכָּבֶנָּה / בּוּר הַנָּשׂוּי / לַמָּוֶת הוּא עָשׂוּי / וְהַמַּשְׂכִּיל – מִמֶּנָּה נַפְשׁוֹ יִשְׁמֹר / וְנָפַל שָׁמָּה שׁוֹר אוֹ חֲמוֹר / כָּל הַנּוֹשֵׂא אִשָּׁה / מַכָּתוֹ אֲנוּשָׁה / אִם יִתְּנוּ לוֹ זָהָב מְלוֹא בֵיתוֹ / לְשַׁמָּה וּלְחֶרְפָּה תִּהְיֶה אַחֲרִיתוֹ / כִּי לְרָעָתוֹ וּלְאֵידוֹ / צְרוֹר הַכֶּסֶף לָקַח בְּיָדוֹ / בְּנִי, אִם חֲכַם לִבֶּךָ, כַּסְפָּן וּזְהָבָן מִמְּךָ רָחוֹק יִהְיֶה / וְשׂוֹנֵא מַתָּנוֹת יִחְיֶה / לְכָל מִקְרֶה רַע יוּשַׁת כֹּפֶר / וְכֹפֶר מִקְרֵה הַנָּשׂוּי לֹא יִמַּד וְלֹא יִסָּפֵר.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אִם בַּצָּפוֹן אִשָּׁה תִשְׁכֹּן / שִׂים פָּנֶיךָ נֶגֶד דָּרוֹם אִם תִּשְׁמַע בֵּין עֵדָה שִׁמְעָהּ / מִן הָעֵדָה הַהִיא תֵּרוֹם

אִם תִּשְׁלַח יָד אֶל מַלְבּוּשָׁךְ / הִנָּצֵל מִמֶּנָּה עָרוֹם.


בְּנִי, אֲנִי הַיּוֹם יְלִדְתִּיךָ / וּבְמַעְגְּלֵי יֹשֶׁר הִדְרַכְתִּיךָ / וְשֵׂכֶל טוֹב הוֹדַעְתִּיךָ / אִמַּצְתִּיךָ עֲזַרְתִּיךָ / אַף תְּמַכְתִּיךָ / פֶּן תִּלָּכֵד בְּפַח יְקוּשָׁה / וְאֵיבָה אָשִׁית בֵּינְךָ וּבֵין הָאִשָּׁה / אַל תַּחְמֹד יָפְיָהּ וְאַל יַשִּׂיאוּךָ תַּעֲנוּגֶיהָ / כִּי רַבִּים חֲלָלִים הִפִּילָה וַעֲצוּמִים כָּל הֲרוּגֶיהָ. / בְּנִי, אַל תִּתֵּן לַאֲחֵרִים הוֹדֶךָ / פֶּן יִהְיֶה לְךָ לְמוֹקֵשׁ וְקִיקָלוֹן עַל כְּבוֹדֶךָ / אִם תִּהְיֶה נִשָּׂא וְגֵאֶה / שַׁעֲרֵי מָוֶת תִּרְאֶה / חַיִּים לִבְאֵר שַׁחַת תּוֹרִידֶךָ / וְאֵין לְאֵל יָדֶךָ / וְהִיא תִשָּׁאֵר אַלְמָנָה בַּעֲבוּר תִּהְיֶה לִי לְעֵדָה / כִּי לֹא אַלְמָן יִשְׂרָאֵל וִיהוּדָה. – – –


בְּנִי, יְגִיעַ כַּפֶּיךָ וְכָל מַחֲמַדֶּיךָ / יִהְיוּ לְךָ לְבַדֶּךָ / וְלֹא תִהְיֶה רוֹעֶה זוֹנוֹת / וְאַל יִהְיוּ לְךָ בָנִים וּבָנוֹת / כִּי אֶת אֲשֶׁר מָאַסְתָּ לָשִׁית עִם כַּלְבֵי צֹאנֶךָ / יִתְחַתְּנוּ בְךָ מֵאֵין רְצוֹנֶךָ / וְיָבֹזוּ זָרִים יְגִיעֶךָ / וְכָל קִנְיָנֶיךָ / אֲשֶׁר לְמָצוֹק אוֹתְךָ יָבִיאוּ / מֵחֲלָצֶיךָ יֵצֵאוּ.


בְּנִי, הִשָּׁמֵר לְךָ פֶּן יִהְיֶה עִם לְבָבְךָ דְּבַר בְּלִיַּעַל / וְאַחֲרֵי מוֹתִי בְּתוֹכַחְתִּי תִגְעַל / כִּי תֹאמַר: ׳הֲלֹא בִּישִׁישִׁים / חָכְמָה וְדַעַת קְדוֹשִׁים / וַיִּשְׂאוּ לָהֶם נָשִׁים׳ / אַל תַּעֲשֶׂה כְמַעֲשֵׂיהֶם לָשִׂים נֶשֶׁף חִשְׁקְךָ לַחֲרָדָה / אִם זוֹנֶה יִשְׂרָאֵל אַל יֶאְשַׁם יְהוּדָה / וְאַל תָּמִית בַּנֹּעַר נַפְשֶׁךָ / וּבַחֲצִי יָמֶיךָ אַל תַּחְשִׁיךְ שִׁמְשֶׁךָ / הֶרֶף מִן הַנָּשׂוּי וְרַָעָתוֹ / כִּי לֹא תוּכַל שֵׂאתוֹ / כִּי יִקַּח אִישׁ אִשָּׁה / מִכָּל מַחֲמַד נַפְשׁוֹ יְבֵשָׁה / יָמָיו מוּעָטִים וְרָעִים / וּדְבָרָיו אֵינָם נִשְׁמָעִים / פָּנָיו לֹא יַחְשֹׂךְ מִכְּלִמּוֹת וָרוֹק / כָּל עוֹבֵר עָלָיו יִשֹּׁם וְיִשְׁרֹק / יֵצְרוּ צַעֲדֵי אוֹנוֹ / וְיִבַשׁ מְקוֹרוֹ וְיֶחֱרַב מַעְיָנוֹ / מַחֲלָתוֹ תִּגְדָּל / וּמֵהֱיוֹת אֶבְיוֹן לֹא יֶחְדָּל / כָּל יָמָיו מַכְאֹבִים וּשְׁנָתוֹ נִהְיְתָה עָלָיו / הָמֵק בְּשָׂרוֹ וְהוּא עֹמֵד עַל רַגְלָיו / וּמַעֲדַנֵּי יְמֵי נְעוּרָיו יִכְלוּ וְיִבְלוּ / וְהַיָּמִים הָרִאשׁוֹנִים יִפְּלוּ / וְהָאִישׁ הַפָּנוּי / לְמִגְדַּל עֹז הוּא בָנוּי / הוֹד וְהָדָר לְבוּשׁוֹ / וְנֵזֶר אֱלֹהָיו עַל רֹאשׁוֹ / מֵרֵיחַ מַיִם יַפְרִיחַ / וְכָל אֲשֶׁר יַעֲשֶׂה יַצְלִיחַ / כָּל יָמָיו שָׁלֵיו שַׁלְאֲנָן / וּמִתְעָרֶה כְּאֶזְרָח רַעֲנָן / קַל מְהֵרָה יָבוֹא כֹּל אֲשֶׁר יְקַוֶּה / וְאֵינֶנּוּ חָסֵר לְנַפְשׁוֹ מִכֹּל אֲשֶׁר יִתְאַוֶּה / יְצַו ה׳ אִתּוֹ אֶת הַבְּרָכָה / וְעוֹד לוֹ אַךְ הַמְּלוּכָה.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

פְּדוּת לַאֲסִירִים יֵשׁ / וְשַׁחַר כְּתֹם לַיִל

וּמַרְפֵּא לְמַכְאֹבִים / וְיֵשׁ קֵץ לְכָל תִּכְלָה

וְרוֹן אַחֲרֵי יָגוֹן / וְתַנְחוּם לְכָל אֵבֶל

לְצָרָה וְלִמְצוּקָה / רְוָחָה וְהַצָּלָה

אֲבָל לַאֲסִיר אִשָּׁה / בְּכוּ נָא לְמִפְקָדוֹ

הֲכִי קֵץ פְּדוּת נַפְשׁוֹ / נִסְתַּם וְלֹא נִגְלָה.


בְּנִי, אַל תִּתֵּן לַנָּשִׁים חֵילֶךָ / וְאַל תַּשְׁכֵּן עַוְלָה בְּאֹהָלֶיךָ / פָּגוֹשׁ דֹּב שַׁכּוּל וְחֶבְרַת זְאֵבֵי יְעָרִים / וְלֹא עִם אִשָּׁה בְחַדְרֵי חֲדָרִים / חַכְמֵי קֶדֶם הִפְטִירוּ בְשָׂפָה: / ׳מִן הַחֶדֶר תָּבֹא סוּפָה׳ / בֵּין שָׁמִיר וָשַׁיִת / וְלֹא עִם אִשָּׁה בְּיַרְכְּתֵי הַבַּיִת / בֵּין תַּנּוּר וְכִירַיִם / וְלֹא בֵּין שְׁנֵי שָׁדַיִּם / בֵּין קוֹצִים כְּסוּחִים וַחֲרוּלִים / וְלֹא בֵין קְוֻצּוֹת תַּלְתַּלִּים / בֵּין קִמְּשֹׁנִים / וְלֹא בֵין שַׂהֲרוֹנִים / סוּף לְרֹאשׁ חָבוּשׁ / וְעָרוֹם מִכָּל לְבוּשׁ / וְלֹא כְסוּת וְצֵדָה / עִם בּוֹגֵדָה / קוֹל נְהִי וְקוֹל יְלָלָה / וְלֹא קוֹל חָתָן וְקוֹל כַּלָּה / אֵבֶל יְעֵנִים / וְלֹא מִצְהֲלוֹת חֻפּוֹת חֲתָנִים / אֶל בֵּית מִשְׁתֵּיהֶם לִבְּךָ אַל יְפֻתֶּה / טוֹב לָלֶכֶת אֶל בֵּית אֵבֶל, מִלֶּכֶת אֶל בֵּית מִשְׁתֶּה / מְשׂוֹשׂ חָתָן וְחֻפָּתוֹ / לְבָשְׁתּוֹ וּלְחֶרְפָּתוֹ / וְלֹא יִרְאֶה בְטוֹבָתוֹ / מִלּוּשׁ בָּצֵק עַד חֻמְצָתוֹ / אוֹיְבֵי אִישׁ נְשֵׁי בֵיתוֹ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

יֵעָצְבוּ הַנֹּשְׂאִים נָשִׁים וּמִ/טּוּב הַזְּמָן יִשָּׁכְחוּ שָׁכוֹחַ

כַּמָּה אֲנָשִׁים גָּבְרוּ, עֵת נָשְׂאוּ / נָשִׁים רְאִיתִים הוֹלְכִים שָׁחוֹחַ

לוּ הַכְּלָבִים נָשְׂאוּ יִסָּכְרוּ / פִּיהֶם עֲדֵי לֹא יוּכְלוּ לִנְבֹּחַ.


בְּנִי, אַל יְפַתּוּךָ בַּעֲלֵי נָשִׁים בְּפִיהֶם / כִּי תֹעֵי לֵבָב הֵם / סָר מֵעֲלֵיהֶם הוֹדָם וְצִלָּם / צֹאנֶה וַאֲלָפִים כֻּלָּם / דַּרְכֵיהֶם וּמַעַלְלֵיהֶם לֹא תִּלְמֹד / הֵמָּה יֹאבֵדוּ וְאַתָּה תַעֲמֹד / הַנִּשּׂוּי כַּמָּה מְרוֹרוֹת הִשְׂבִּיעַ / וּנְעָרִים עַד שְׁאוֹל הִגִּיעַ / וּבַחוּרֵי יִשְׂרָאֵל הִכְרִיעַ / יְשַׁוְּעוּ וְאֵין מוֹשִׁיעַ / מִן הָעֲשִׁירִים הֵנִיס רְבָבָה / וַיַּכֵּם וַיִּרְדְּפֵם עַד חוֹבָה / נַעַר הָיִיתִי / וְגַם זָקַנְתִּי / וְלֹא רָאִיתִי נָשׂוּי מַצְלִיחַ דַּרְכּוֹ / וַה׳ הִגִּיהַּ חָשְׁכּוֹ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אִם יִתְּנוּ בָךְ בַּעֲלֵי נָשִׁים / עַיִן פְּתַח יָד וֶאֱמֹר חָמֵשׁ

גּוּר מִפְּנֵי אִשָּׁה וְלֹא תַחְמֹד / יָפְיָהּ אֲשֶׁר יִמַּס כְּחֹם שֶׁמֶשׁ

אִם תַּעֲלֶה עַל לִבְּךָ הַיּוֹם / הָשַׁח גְּאוֹן תִּפְאַרְתְּךָ אֶמֶשׁ

לוּ דָמְתָה קוֹמָתְךָ תָּמָר / חִישׁ תַּהֲלֹךְ כָּפוּף כְּמוֹ חֶרְמֵשׁ.


בְּנִי, הַט אָזְנְךָ וּשְׁמַע דִּבְרֵי חֲכָמִים / לְמַעַן יִיטַב לָךְ וְהַאֲרַכְתָּ יָמִים / כָּל הַמַּרְבֶּה עִם הָאִשָּׁה שִׂיחָה / סוֹפוֹ לְגֵיהִנָּם פְּתוּחָה / בְּעוֹדְךָ רַךְ תְּקָרְבֶךָּ / וְכִכְלוֹת כֹּחֲךָ תַּעַזְבֶךָּ / כָּל נֵתַח טוֹב תֹּאכַל מִן הַמַּטְעַמִּים / וְתָגִישׁ לְפָנֶיךָ הָעֲצָמִים / תַּרְמִית הַנָּשִׁים וְנִכְלֵיהֶן / מִי יַגִּיעַ אֲלֵיהֶן? / כַּמָּה גְדוֹלִים הִשְׁפִּילוּ / וְכַמָּה יְגוֹנוֹת הִנְחִילוּ / כַּמָּה גִבּוֹרִים כָּשְׁלוּ בָהֶן וְאֵין מְקִימָם / וְכָשַׁל גַּם יְהוּדָה עִמָּם.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

כָּל הַיְּקוּם הֻכּוּ בְעִוָּרוֹן / וּנְבוֹן דְּבָרִים תַּם וְסַף דֵּעוֹ

כִּי יֶחֱזוּ גֶּבֶר בְּיַם נִשּׂוּי / טָבַע וְיָבֹא אַחֲרָיו רֵעוֹ.


בְּנִי, אַל תַּחְמֹד יָפְיָהּ בִּלְבָבֶךָ / וּבִעַרְתָּ הָרָע מִקִּרְבֶּךָ / כִּי הַנָּשִׁים אֵינָן יָפוֹת / כִּי אִם לְרַקָּחוֹת וּלְטַבָּחוֹת וּלְאוֹפוֹת / בָּרוּךְ הַשֵּׁם שֶׁלֹּא שָׂם חֶלְקֵנוּ כָהֶן / וּמַעֲשֵׂינוּ כְמַעֲשֵׂיהֶן / כְּמוֹהֶן יִהְיוּ בוֹעֲלֵיהֶן / וְכָל הַבּוֹטֵחַ בָּהֶן / בָּאֵשׁ יִשְׂרְפוּ אוֹתוֹ וְאֶתְהֶן / וַאֲנִי מוֹדֶה לַאֲשֶׁר עָשָׂה אֶת הַיָּם וְאֶת הַיַּבָּשָׁה / שֶׁלֹּא עָשָׂנִי אִשָּׁה / טוֹב לִפְנֵי אֱלֹהִים יִמָּלֵט מִקָּרְבָה / וְחוֹטֶא יִלָּכֶד בָּהּ / אֵלֶּה תּוֹלְדוֹת הַָנָשׂוּי וְקוֹרוֹתָיו / וְלֹא זָכַרְתִּי לְךָ חֲצִי תְּלָאוֹתָיו. – – –
