משל קיד: תִּנְשֶׁמֶת וְרֹכְלִים

מָלֵא דֹפִי וְשִׁמְצָה וְהוֹלֵלוּת / נֶחְשָׁב לִבְנֵי אָדָם כְּרֻם זֻּלּוּת

תִּנְשֶׁמֶת נֹבֶרֶת בְּשִׁנֶּיהָ / אַחֲרֵי הַשָּׁרָשִׁים וְהָעֲשָׂבִים אֲשֶׁר תַּחְתֶּיהָ / וַתַּשְׁלֵךְ אֶת הֶעָפָר לְמַעְלָה הֵמֶנָּה / לָלֶכֶת אָנֶה וָאָנָה / לְמַעַן רַבּוֹת כָּהֵנָּה וְכָהֵנָּה / וַתְּמַלֵּא הַשָּׂדֶה תִּלֵּי תִלִּים / וַיָּבֹאוּ שָׁמָּה הָרֹכְלִים / לְבַקֵּשׁ שָׁרָשִׁים לִרְפוּאָה / וַיַּחְפְּרוּ אַחֲרֵיהֶם עַד אֲשֶׁר יָצְאָה / מִמְּקוֹמָהּ לִקְרָאתָם הַתִּנְשֶׁמֶת / וַיֹאמְרוּ: עַל פִּי שְׁנַיִם עֵדִים יוּמַת הַמֵּת / וַיִּרְדְּפוּ אַחֲרֶיהָ עַד הַתֵּל / וַתִּמָּלֵט בְּתוֹכוֹ וּבָהֶם תְּהַתֵּל / וַתִּקְרָא: מַדּוּעַ דְּלַקְתֶּם אַחֲרַי / כִּי יָדַעְתִּי בֵּיתִי סְתָרָי / לְהַצִּיל מִמָּוֶת נַפְשִׁי וּלְחַיּוֹתִי בָּרָעָב / אֶתְהַלֵּךְ בְּאַרְצוֹת הַחַיִּים וְאַל אֶהְיֶה לְלָעַב / וְלֹא תוֹסִיפוּ לִרְאוֹתִי עוֹד עַד עוֹלָם / וְהַשָּׁרָשִׁים אֲשֶׁר אַתֶּם מְבַקְשִׁים עָקַרְתִּי כֻלָּם / כִּי בְלִבִּי יוֹם נָקָם / וְאַתֶּם תָּשׁוּבוּ רֵיקָם / בֹּשִׁים וַחֲפוּרִים וְנִכְלָמִים / בֵּין הָרֹפְאִים וּבֵין הָעַמִּים / וְרֹפְאֵי אֱלִיל כֻּלְּכֶם / אַתֶּם בְּמַעַלְלֵיכֶם וּבְנִכְלֵיכֶם / סוּרוּ מִפָּנַי מְהֵרָה / וְאַל תּוֹפַע עֲלֵיכֶם נְהָרָה / וַעֲלֵיכֶם כָּל אָדָם יִשְׁרֹק / וּמִפְּנֵיכֶם לֹא אֶחְשֹׂךְ רֹק / שְׁפָלִים וְנִבְזִים מִכָּל אָדָם / אֲשֶׁר הֵמִיר בְּקָלוֹן כְּבוֹדָם / וַיִּפְנוּ מִשָּׁם הָאֲנָשִׁים / אֶל מָקוֹם אַחֵר אַחַר הַשָּׁרָשִׁים / וַתִּשְׁלַח אַחֲרֵיהֶם בִּתָּהּ / לְמַעַן עֲשׂוֹתָהּ מְזִמָּתָהּ / לַעֲקֹר הַשָּׁרָשִׁים מִן הָאָרֶץ / וַתֹּאמַר: יִשְּׁכֶנּוּ נָחָשׁ גֹּדֵר פָּרֶץ / אֲשֶׁר יֵרְדוּ הָאֲנָשִׁים לְחַיֵּינוּ / לָקַחַת טֶרֶף וּמָזוֹן לְנַפְשָׁם מִמֶּנּוּ / וַתֵּלֶךְ בִּתָּהּ אַחַר עֲצָתָהּ / וַתַּעַשׂ כְּכָל אֲשֶׁר עָלֶיהָ אִמָּהּ צִוְּתָה / וַתָּבֹא עַד מְקוֹם תַּחֲנוֹתָם / וְלֹא שָׁבָה עַד כְּלוֹתָם / הַשָּׁרָשִׁים אֲשֶׁר בָּאָרֶץ כֻּלָּם / וַתְּשִׂימֵם כָּלָה עַד עוֹלָם / וְלֹא נִמְצָא שֹׁרֶשׁ אֶחָד / וַיֵּלְכוּ מִשָּׁם הָאֲנָשִׁים בְּיַחַד / וְהַתִּנְשֶׁמֶת אַחֲרֵיהֶם רָצְתָה / לְהַגִּיד לָהֶם כִּי זֹאת עָשְׂתָה / וַתֹּאמַר: לָמָּה יְגַעְתֶּם לָתֹהוּ / מִי הוּא זֶה וְאֵיזֶהוּ / אֲשֶׁר מְלָאוֹ לִבּוֹ לַעֲשׂוֹת כֵּן / לֹא נִמְצָא חָכָם בָּכֶם / אֲנִי עִוֶּרֶת וְאַתֶּם פִּקְחִים / מִמֶּנִּי תִּקְחוּ בְשָׂמִים וּמֶרְקָחִים / לִי הָעֵצָה וְהַחָכְמָה / לְהַכִּיר כָּל שָׁרְשֵׁי אֲדָמָה / אֵיזֶה מֵהֶם יָפִים לְאִסְפְּלָנִיּוֹת / מִכָּל מִינֵי גְוָנִיּוֹת / לָבָן אָדֹם יָרוֹק שָׁחוֹר / וְאַתֶּם תֵּלְכוּ אָחוֹר / יוֹמָם וָלַיְלָה אַתֶּם נִלְאִים / וְשֹׁרֶשׁ אֵין אַתֶּם מוֹצְאִים / וְהָאֲנָשִׁים לֹא יָדְעוּ מְאוּמָה / תַּחַת הָאָרֶץ אֵיזֶהוּ מְקוֹמָהּ / כִּי לֹא לַגִּבּוֹרִים הַמִּלְחָמָה / וְלֹא לַקַּלִּים הַמֵּרוֹץ לָבֹא שָׁמָּה / וַיָּבֹאוּ הָאֲנָשִׁים הָאֵלֶּה בַמַּחֲנֶה / אֵין קוֹל קֹשֵׁב וְאֵין עֹנֶה.

וְהַמָּשָׁל בְּאִישׁ הַמִּתְפָּאֵר בְּרֹב חָכְמָתוֹ / בְּיֹפִי בְּנוֹי וּבְקוֹמָתוֹ / לֵאמֹר בָּאָרֶץ מִי כָמוֹהוּ / מִי הוּא זֶה וְאֵיזֶה הוּא / וְלִבּוֹ רֵיק מִתְּבוּנָה / מִכָּל אַנְשֵׁי אֲמָנָה / כִּי אִם מָלֵא דֹפִי וְשִׁמְצָה וְהוֹלֵלוּת וְסִכְלוּת / וְנֶחְשָׁב לִבְנֵי אָדָם כְּרֻם זֻּלּוּת / וּמֵחָכְמָתוֹ יֵלֵךְ שְׁחוֹחַ / תַּחַת חִטָּה יֵצֵא חוֹחַ / וּבְכָל עֶצֶב יֵשׁ מוֹתָר וּמִישׁוֹר / וּדְבַר שְׂפָתַיִם אַךְ לְמַחְסוֹר / וְאָמַר קֹהֶלֶת בְּמִשְׁלֵי הַנְּבוּאָה / תֵּחַת גְּעָרָה בְמֵבִין מֵהַכּוֹת כְּסִיל מֵאָה.
