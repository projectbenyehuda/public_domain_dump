משל קב: גָּמָל וִיעֵל־סֶלַע

הַבִּלְתִּי מִסְתַּפֵּק בְּדֵי סִפְקוֹ / יֹאבַד גַּם לֶחֶם חֻקּוֹ


גָּמָל רָאָה יַעֲלֵי סֶלַע אַחַר הַמִּדְבָּר / יַחְלְמוּ בְנֵיהֶם יִרְבּוּ בַבָּר / עַל יְדֵיהֶם יִרְבְּצוּ אֵילִים וְקַרְנַיִם בְּרֹאשָׁם / וְכָל חַיַּת הַשָּׂדֶה יְשַׂחֲקוּ שָׁם / וְהַגָּמָל בָּא הַמַּחֲנַיִם / וַיָּבֹא עַד הָאַיִל בַּעַל הַקְּרָנַיִם / וַיִּתְפָּאֵר בְּצַלְמוֹ וּבִדְמוּתוֹ / בִּגְבוּרָתוֹ וּבְגֹבַהּ קוֹמָתוֹ / וְכִי כָּל הֹלְכֵי עַל אַרְבַּע אֲשֶׁר לְפָנָיו / הִנָּם כַּחֲגָבִים בְּעֵינָיו / וַיַּעַן הָאַיִל: אָמְנָם נֶחְמַדְּתָּ וְנָעַמְתָּ / וְאִם הָיוּ לְךָ קַרְנַיִם כָּמוֹנִי מְאֹד עָצָמְתָּ / וַיָּשֶׂם הַגָּמָל אֶת הַדְּבָרִים אֶל לִבּוֹ / וַיֵּלֶךְ אֶל בְּעָלָיו הַמֹּשֵׁל בּוֹ / וַיֹאמַר: אִם הַגְּדֻלָּה וְהַגְּבוּרָה נָפְלָה בְגוֹרָלִי / כָּל זֶה אֵינֶנּוּ שֹׁוֶה לִי / בִּרְאוֹתִי חַיּוֹת וְקַרְנַיִם בְּרָאשֵׁיהֶם / אֲשֶׁר קָטָנִּי עָבָה מִמָּתְנֵיהֶם / וַיְנַגְּחוּ יָמָּה צָפוֹנָה וָנֶגְבָּה / וְעַל גֹּבַהּ קוֹמָתִי הוֹצִיאוּ דִבָּה / וְרֶסֶן מִפָּנַי שִׁלֵּחוּ / לֵאמֹר: כִּי אֹתִי יְנַגֵּחוּ / וְעַתָּה אֲדֹנִי שְׁמַע בְּקוֹלִי / אָזְנַיִם כְּרֹת לִי / חַסְדְּךָ הָאַחֲרוֹן מִן הָרִאשׁוֹן הַשְׂגִּיבָה / וְקַרְנַיִם בְּמִצְחִי לִי הָבָה / וַיִּפְצַר בּוֹ עַד בֹּשׁ / וַיֹּאמֶר לוֹ אֲדֹנָיו: הֲגַם אֹתִי תִכְבֹּשׁ / וּמַדּוּעַ לִשְׁאוֹל הִקְשֵׁתָ / טוֹב אֲשֶׁר בְּמִשְׁאָלְךָ הֶחֱשֵׁיתָ / וַיִּתְמַרְמַר אֵלָיו אֲדֹנָיו / וַיִּכְרֹת עַד הַחֵצִי אָזְנָיו / וְלֹא הֵשִׁיב יָדוֹ וְלֹא חָמַל / עַל כֵּן אָזְנַיִם קְטַנּוֹת לַגָּמָל.


וְהַמָּשָׁל בַּאֲשֶׁר לֹא דַי לוֹ בְּסִפְקוֹ / וְיִשְׁאַל טֶרֶף יוֹתֵר מֵחֻקּוֹ / וְאֹמֵר לִשְׁכֵנָיו כִּי מֵאֲשֶׁר בְּיָדוֹ לֹא יִחְיֶה / שָׁרְשׁוֹ כַמָּק יִהְיֶה / וּפִרְחוֹ כָאָבָק יַעֲלֶה / וּכְרֶגַע הוֹנוֹ יִכְלֶה / כִּי חֶרֶב לַזְּמָן מְצֻמֶּדֶת בְּתַעֲרָהּ / לְהַשְׁמִיד וּלְכַבּוֹת גְּדֻלָּתוֹ אֲשֶׁר נִשְׁאָרָה / וַאֲנִי מָצָאתִי בְּסֵפֶר חָכְמַת הַזִּכָּרוֹן / הַתַּאֲוָה שֻׁתֶּפֶת הָעִוָּרוֹן / וְקֹהֶלֶת כָּתַב בִּמְלִיצַת מִשְׁלֵי חָכְמָתוֹ / דְּבַשׁ מָצָאתָ אֱכֹל דַּיֶּךָ פֶּן תִּשְׂבָּעֶנּוּ וַהֲקֵאֹתוֹ.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


גָּמָל עֵת שָׁאַל קַרְנַיִם / וַיִּכְרְתוּ מֶנּוּ אָזְנַיִם

כֵּן יִקְרֶה אֶל חֹמֵד תָּמִיד / אִם לוֹ הוֹן רָב וּמַחֲנָיִם.
