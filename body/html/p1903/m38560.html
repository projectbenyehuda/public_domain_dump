משל קא: סֹחֵר וְלִסְטִים וּפָרָשׁ

הַגִּזְבָרִים וְהַשֹּׁפְטִים נָצוּ גַּם נָעוּ / חֲדָשִׁים מִקָּרוֹב בָּאוּ

סֹחֵר אָרַח עִם אֱוִילִים לְחֶבְרָה / וַיְהִי בַצָּהֳרַיִם בְּבוֹאָם הַיַּעֲרָה / וַיִּרְאוּ עָלָיו בִּגְדֵי חֲמֻדוֹת / פֹּחֲזִים אַנְשֵׁי בֹּגְדוֹת / כְּרֵיקִים בֹּקְקִים בְּקָקוּהוּ / וְהִכּוּהוּ שְׁלָלוּהוּ פְּצָעוּהוּ / וַיִּשָּׁאֵר הָלוּם וּמְדֻכֶּה / נָגוּעַ פָּצוּעַ וּמֻכֶּה / וּמִיֶּתֶר קַשְׁתָּם נֶאֱסַר וְחָבוּשׁ / עָרוֹם יָצָא מִבְּלִי לְבוּשׁ / וְלָלֶכֶת לְהִתְרַפֵּא מִן הַמַּכִּים רָצָה / וְלֹא הִתְחַזֵּק וְנָפַל אַרְצָה / וַיִּצֶּק אֶל חֵיק הָאָרֶץ דָּמוֹ / וְאֵין אֶחָד מִתְחַזֵּק עִמּוֹ / וְהַזְּבוּבִים נֶאֶסְפוּ עַל צְלָעָיו וּזְרוֹעָיו / וַיַּחֲנוּ סָבִיב עַל כָּל פְּצָעָיו / לָלֹק וְלִלְחֹךְ דְּמֵי הַגְּזָרִים / כַּאֲשֶׁר תַּעֲשֶׂינָה הַדְּבוֹרִים / וַיֹּאכְלוּ וַיִּשְׂבְּעוּ וַיּוֹתִירוּ / וּמֵרֹב שָׂבְעָם עָלָיו נָחוּ וַיִּשָּׁאֲרוּ / וְהִנֵּה דֶרֶךְ שָׁם פָּרָשׁ רֹכֵב / וַיִּפֶן וַיַּרְא אֶת הָאִישׁ שֹׁכֵב / וְכִי יָדָיו מֵאֲחֹרֵי גֵווֹ אֲסוּרוֹת / וְכִי יוּבַל לְיוֹם עֲבָרוֹת / בַּאֲשֶׁר עֲרֻמָּה כָּל גְּוִיָּתוֹ / וְהַזְּבוּב לְמִינוֹ שָׁם תַּחֲנוֹתוֹ / וַיִּשְׁכַּב עַל עֲפַר הָאָרֶץ כְּרֵסוֹ / וַיַּחְמֹל עָלָיו וַיֵּרֶד מֵעַל סוּסוֹ / וַיְנַתְּקֵהוּ מִמּוֹסְרוֹתָיו / וּמֵעַנְפֵי הַיַּעַר וּפֹארֹתָיו / וַיַּשֵּׁב הַדְּבוֹרִים וַיְנִיסֵם יַחַד / וְלֹא נִשְׁאַר בָּהֶם עַד אֶחָד / וַתִּשָּׁאֵר גְּוִיַּת הַפָּצוּעַ מְפֹרָצֶת / וַיְקַלֵּל אֶת הַפָּרָשׁ קְלָלָה נִמְרֶצֶת / וּכְאוֹב מֵאֶרֶץ קוֹלוֹ מְחָרֵף וּמְגַדֵּף / וַיֹּאמֶר לוֹ: לָמָּה אֶת הַזְּבוּבִים תְּרַדֵּף / יַעְכָּרְךָ עֹכְרִי, לָמָּה עֲכַרְתַּנִי / מֶה עָשִׂיתִי לְךָ וּמָה הֶלְאֵיתַנִי / אַךְ מֵרְשָׁעִים יֵצֵא רֶשַׁע / אָנוּשׁ חִצִּי בְּלִי פָשַׁע / כִּי הִקִּיפוּ עָלַי אֹרְבִים וְשַׁדּוּנִי / הֲלָמוּנִי הִכּוּנִי פְּצָעוּנִי / אוֹי וַאֲבוֹי הִגְרִימוּנִי שׁוֹלֲלַי / נָשְׂאוּ אֶת רְדִידִי מֵעָלַי / וְהִנֵּה קַמְתָּ אַחֲרֵיהֶם לְהַכְאִיב לְבָבִי / יָסַפְתָּ מַכְאוֹב עַל מַכְאוֹבִי / וַיַּעַן הַפָּרָשׁ: אַךְ מִצָּרוֹתֶיךָ פְּדִיתִיךָ / מֶה עָשִׂיתִי לְךָ וּמָה הֶלְאֵיתִיךָ / אֲשֶׁר כִּכְסִיל תּוֹצִיא עָלַי דִּבָּה / כִּי אֶעֱרוֹץ הָמוֹן רַבָּה / אֲשֶׁר גְּוִיָּתְךָ מְכַלִּים וְכוֹרְתִים / וַתִּקְרַב לַשַּׁחַת נַפְשֶׁךָ וְחַיָּתְךָ לַמְמִתִּים / אֲשֶׁר הָיוּ עַל בְּשָׂרְךָ הָעֲרֻמָּה / וְהִדַּחְתִּיו אֶל אֶרֶץ צִיָּה וּשְׁמָמָה / וָאָסִיר מֵעַל שִׁכְמְךָ סֻבֳּלָם / לֹא תוֹסִיף לִרְאֹתָם עוֹד עַד עוֹלָם / וְלֹא תוֹסִיף עוֹד לְדַאֲבָה / לָמָּה שִׁלַּמְתָּ רָעָה תַּחַת טוֹבָה / וְתַחַת בִּרְכָתִי קְלָלָה / וְתַחַת תְּהִלָּתִי תָהֳלָה / וַיִּקֶשׁ דְּבַר הַסֹּחֵר עַל הַפָּרָשׁ / וַיַּעַן לוֹ: נֹקֵם נְקָמוֹת אַל יֶחֱרַשׁ / הֱקִלֹּתַנִי וַעֲכַרְתַּנִי בִזְרוֹעֲךָ הַנְּטוּיָה / וּבְיָדְךָ אֲשֶׁר הוֹיָה / לְרַדֵּף הַזְּבוּבִים מֵעִמִּי / אֲשֶׁר כְּבָר שָׂבְעוּ מִדָּמִי / וּגְוִיַָּתִי כַּלְּבוּשׁ כִּסּוּ / וְאַחֲרֵי אֲשֶׁר תְּנִיסֵם וְנָסוּ / בָּאוּ נָכְרִים אֲשֶׁר דָּמִי יִשְׁאָבוּן / וְעֹרְקַי לֹא יִשְׁכָּבוּן / וְטוֹב לִי לִסְבֹּל זְבוּב וּדְבוֹרָה / אֲשֶׁר אָכְלוּ לְשָׂבְעָה וְשָׁתוּ לְשָׁכְרָה / מִלִּסְבֹּל אֲשֶׁר עֲדֶן לֹא שָׂבָעוּ / עַל כֵּן דְּבָרַי לָעוּ.


וְהַמָּשָׁל בְּגִזְבָּרִים וּמֶלְצָרִים / אֲשֶׁר הָעָם מַחֲלִיפִים וּמְמִירִים / אַחֲרֵי אֲשֶׁר הֵרִיקוּ הַכֶּסֶף מִכִּיסֵנוּ / הָאֲנָשִׁים הָאֵלֶּה שְׁלֵמִים הֵם אִתָּנוּ / כִּי הִשְׂבַּעְנוּ מַעֲדַנִּים כְּרֵסָם / וּמִלֵּאנוּ לָרְוָיָה כוֹסָם וְכִיסָם / וּמִכַּסְפֵּנוּ קִצְפָּם נִסְתָּם / וַיַּעֲמִידוּ אֲחֵרִים תַּחְתָּם / אֲשֶׁר לִגְזֹל לֹא נִלְאוּ / חֲדָשִׁים מִקָּרוֹב בָּאוּ / עַל כֵּן כְּאֶבֶן שֹׁאֶבֶת יִשְׁאָבוּ / כְּפִירִים רָשׁוּ וְרָעֵבוּ / אֲשֶׁר אַחַר הַבֶּצַע יַטּוּ / רָאשֶׁיהָ בְשֹׁחַד יִשְׁפֹּטוּ / וְאֶל מַכִּירֵיהֶם יִתְנַכָּרוּ / וּמִחְיָה לְאִישׁ לֹא יַשְׁאִירוּ / וְאֵין אִישׁ תַּחַת יָדָם חָפְשִׁי / וִיצַו גַּם אֶת הַשֵּׁנִי גַּם אֶת הַשְּׁלִישִׁי / לָכֵן אֲמָרַי הַאֲזִינוּ וַעֲצָתִי שְׂאוּ / שֹׁפֵט יָשָׁן מִפְּנֵי חָדָשׁ אַל תּוֹצִיאוּ / הַזְּמָן אֲשֶׁר עַל אִישׁ גֹּבֵר / פַּעֲמַיִם שָׁלוֹשׁ עִם גָּבֶר / אֲשֶׁר יִפְעַל בַּחֲבֵרָיו / עַל כֵּן אִישׁ בִּימֵי מְגוּרָיו / יֶאֱהַב הַשֹּׁפֵט אֲשֶׁר כְּבָר הִכִּיר אֹתוֹ / כִּי הֶחָדָשׁ יִקַּח מִמְּךָ עֶמְדָּתוֹ / וּלְעָזְרוֹ בְמֶמְשַׁלְתּוֹ הֱיֵה מֵעִיר / פֶּן יִהְיֶה מוּסָר מֵעִיר.
