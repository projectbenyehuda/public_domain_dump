משל י: עַכְבַּר־בַּיִת וְעַכְבַּר־יַעַר

מְבַקֵּשׁ גְּדוֹלוֹת יִשָּׁאֵר מִכָּאן וּמִכָּאן קֵרֵחַ / וְיִהְיֶה מַר צוֹרֵח

עַכְבַּר־הַבַּיִת נָסַע מִמְּקוֹם מוֹלַדְתּוֹ / לִרְאוֹת בְּעִיר אַחֶרֶת מִשְׁפַּחְתּוֹ / וַיֵּלֶךְ כָּל הַיּוֹם עַד כִּי פָנָה לַעֲרֹב / וַיַּרְא וְהִנֵּה יַעַר אֶצְלוֹ קָרוֹב / וּבוֹ עַכְבְּרֵי יַעַר מְשַׁעְשְׁעִים בְּעֵדֶן נַפְשָׁם / וַיִּפְגַּע בַּמָּקוֹם וַיָּלֶן שָׁם / עִם עַכְבַּר בָּר אֲשֶׁר שָׂמַח בּוֹ / וְזֶה לָזֶה הִגִּיד כָּל לִבּוֹ / וַיַּרְא אֶת מַאֲכָלָיו וְאֶת מַשְׁקָיו / וְיָדוֹ חִלַּקְתָּה לָהֶם בַּקָּו / הָעֲשָׂבִים וְכָל הַנִּלְוִים עִמָּם / וְשֹׁרֶשׁ רְתָמִים לַחְמָם / וַיֹּאמַר אֶל עַכְבָּר הַיַּעֲרִי / רַגְלְךָ תֹאחֵז בַּאֲשׁוּרִי / וּמָחָר אֲנִיעֲךָ עִמִּי / וְאַשְׁקְךָ מִשִּׁקּוּיִי וְתֹאכַל לַחְמִי / וְטוֹב לְךָ יוֹם בַּחֲצֵרִי מֵאֶלֶף הֵנָּה בַיַּעַר / בְּמָצוֹר וּבְמָצוֹק וּבְרוּחַ סֹעָה מִסָּעַר / וּבְחֹסֶר כָּל מַאֲכָל / וּבְעֵת הַסְּתָו אֵיךְ תּוּכַל / לִמְצֹא מִחְיָה וְהָעֲשָׂבִים יָבְשׁוּ / וְיֹשְׁבֵיהֶם קִצְרֵי יָד חַתּוּ וָבֹשׁוּ / וְאַתָּה מִן הַחֲרַכִּים מֵצִיץ / יָבֵשׁ חָצִיר נָבֵל צִיץ / בֹּא אִתִּי וְשָׂא רֹאשֶׁךָ / וְאֶתֵּן מַעֲדָנִים לְנַפְשֶׁךָ / וַיְסִיתֵהוּ לָבֹא מִן הַיַּעַר הָעִירָה / מְקוֹם חִטָּה וּשְׂעוֹרָה / וְזֶרַע בַּמְּגוּרָה / וַיְהִי כְבוֹאָם אֶל בֵּית לֶחֶם / וַיֹּאמֶר עַכְבַּר הַבַּיִת: עָלֶיךָ אֲרַחֵם / הִנֵּה כָּל בֵּיתִי לְפָנֶיךָ / וֶאֱכֹל כַּטּוֹב בְּעֵינֶיךָ / וְכַאֲשֶׁר הֵמָּה אוֹכְלִים וְחוֹגְגִים / קֶמַח וְלֶחֶם בָּשָׂר וְדָגִים / וַיָּבֹא שָׁם הָאָדָם כְּדַרְכּוֹ / לָקַחַת לֶחֶם לַעֲרֹךְ עֶרְכּוֹ / וַיָּנָס עַכְבַּר הַבַּיִת וַיִּתְחַבֵּא בְּסֻכּוֹ / וְעַכְבַּר הַיַּעַר נֶחְפַּז אַחֲרָיו / וְלֹא פָחַז כָּזֹאת מִנְּעוּרָיו / וּבְעֵת מִשָּׁם הָאָדָם נִפְרַד / עַכְבַּר הַבַּיִת מִנְּקִיקָיו יָרָד / וַיֵּצְאוּ אֵלֶּה וָאֵלֶּה מִמִּגְרָשָׁם / מִן הַחוֹרִים אֲשֶׁר הִתְחַבְּאוּ שָׁם / וְעַכְבַּר הַבַּיִת בְּדָגָן וּבְקֶמַח שָׁלַח שֵׁן / וְאָכַל וְשָׂבַע וְדָשֵׁן / וְהַיַּעֲרִי מֵרָחוֹק עָמַד / לִרְאוֹת מַה יֵּעָשֶׂה לַאֲשֶׁר אִתּוֹ נִצְמָד / כִּי נִבְעַת מִן הָאָדָם אֲשֶׁר רָאָה / סָמַר בְּשָׂרוֹ וְנַפְשׁוֹ נְכֵאָה / עַל הַדֶּרֶךְ אֲשֶׁר בָּא בוֹ / וַיִּנָּחֵם בְּמַחֲשָׁבוֹ / וּכְחָשְׁבוֹ בְּלִבּוֹ כָּכָה / עַכְבַּר הַבַּיִת נָס מִן הַמַּעֲרָכָה / וַיִּתְחַבֵּא בְּבֵית סְתָרָיו / וַיֹּאמֶר לוֹ הַיַּעֲרִי, אַשְׁרָיו / אֲשֶׁר מִזֶּה בְּשָׁלוֹם יִפָּרֵד / וְלֹא יִהְיֶה עוֹד יָרֵא וְחָרֵד / כִּי מֵעֵת אֲשֶׁר עָזַבְתִּי מְלוֹנִי / עֲווֹן עֲקֵבַי יְסֻבֵּנִי / רְאֵה בֵיתְךָ וּמְקוֹמְךָ / כִּי לֹא אוֹסִיף לָגוּר עִמְּךָ / וְהַפַּחַד בְּרִיתֵנוּ הֵפַר / בֹּא בַּצּוּר וְהִטָּמֵן בֶּעָפָר / מִפְּנֵי הָאָדָם פֶּן יְגֹרְךָ בְחֶרְמוֹ / וְהָאֹרֵב מֵגִיחַ מִמְּקוֹמוֹ / כָּל חַיֶּיךָ כַעַס וָזַעַם / וְלֶחֶם סְתָרִים יִנְעָם / וְלָמָּה הוֹצֵאתָ עַל מְקוֹם הַיַּעַר דִּבָּה / טוֹב פַּת חֲרֵבָה וְשַׁלְוָה בָהּ / מִבַּיִת מָלֵא זִבְחֵי רִיב / וּלְכָל אָדָם אֲשֶׁר יַקְרִיב / תָּנוּס בְּחָפְזָה וְתִבָּהֵל / לְהִתְחַבֵּא מֵאֹהֶל אֶל אֹהֶל / וּבִמְקוֹמִי אֶמְצָא שִׁיר וָרוֹן / הֲדַר הַכַּרְמֶל וְהַשָּׁרוֹן.

וְהַמָּשָׁל לִמְבַקֵּשׁ גְּדוֹלוֹת וְנִפְלָאוֹת / יַנִּיחַ הַטּוֹב וְהַנָּאוֹת / וְלֹא יִסְתַּפֵּק בְּחֶלְקוֹ / וְלֶחֶם חֻקּוֹ / יְבַקֵּשׁ לַעֲלוֹת בְּמַעֲלוֹת נוֹסָפוֹת / וְלָרֹב יֵלֵךְ לַחֲרָפוֹת / בָּא זֶה וְאִבֵּד אֶת זֶה / וְגַם אֶת הַשֵּׁנִי יִבְזֶה / אָז יֹאמַר בְּפֶה וּבְלָשׁוֹן / אֵלְכָה וְאָשׁוּבָה אֶל דַּרְכִּי הָרִאשׁוֹן / כִּי טוֹב לִי אָז מֵעָתָּה / וְלִפְעָמִים יֵרֵד מַטָּה מָטָּה / וְיִהְיֶה מַר צוֹרֵחַ / וְיִשָּׁאֵר מִכָּאן וּמִכָּאן קֵרֵחַ.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


טוֹב פַּת חֲרֵבָה לִי וְשַׁלְוָה בָהּ / מִבַּרְבּוּר אָבוּס וּבִמְרִיבָה

טוֹב לִי בְאַהְבָה כַּף מְלֹא קֶמַח / מֵעֶשְרִים כַּפּוֹת וּבָם אֵיבָה

טוֹב לִי מְעַט יָרָק בְּבִטָּחוֹן / מִצְּבִי מְקוֹם רִיבָה וּמַדְהֵבָה

שָׁלֵו בְּבֵיתָךְ שֵׁב וְאַל תָּתוּר / בֵּית זָר בְּזַעַף אַף וְלֶהָבָה.
