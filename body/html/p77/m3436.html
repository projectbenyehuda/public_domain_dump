[רוח זמירות]

כתובת: שירים שעשה המשורר הנכבד דון וידאל בנבנשת למשורר אנשלמה דאיפייארה במקום אלקניס


רוּחַ זְמִירוֹת נוֹסְסָה עָלֶיךָ / רוּחִי וְעַל מִי זֶה תְשַׁו כִּסְלֶךָ

הַיְחָבְרְךָ יוּבָל אֲבִי כָל תּוֹפְשֵׂי / כִנּוֹר וְאֶת כֵּלָיו יְהַנְחִילֶךָ

אוֹ אִם בְּנֵי אָסָף יְקִיצוּן מִשְׁנַת / עוֹלָם וְיַעַמְדוּ לְגוֹרָלֶךָ

אוֹ אֶל בְּנֵי קֹרַח לְחֶבְרָה תֶאֱרַח / וִידֵי יְדוּתוּן כּוֹנְנוּ כֵלֶיךָ

אוֹ יַד בְּנוֹ יִשַׁי בְכָל זֹאת הָיְתָה / עִמָּךְ וְדַרְדַּע יַעֲרֹךְ מִלֶּיךָ

אוֹ שִׁירְךָ אָמוֹן בְּחֵיק הֵימָן וּמִן / זִמְרַת זְמִירָיו מָן לְהַאְכִילֶךָ

אוֹ תוֹךְ נְוֵה אֵיתָן תְּשַׁו מוֹשָׁבְךָ / אֵיתָן וְטַל אוֹרוֹ יְהִי טַלֶּךָ

אוֹ אַהֲרֹן הָיָה נְבִיאֶךָ וְהוּא / הֵישִׁיר דְּרָכֶיךָ וּמַסְלוּלֶיךָ

אוֹ שֵׁם וְעֵבֵר קָבְעוּ מִדְרָשׁ בְּתוֹךְ / בֵּיתָךְ וְיִשָׂשׁכָר בְּאָהָלֶךָ

לֵב נַעֲנָה עַל מִי תְשַו מִבְטַחֲךָ / בִּשְׂאֵת זְמִירֶךָ וּבִמְשָׁלֶךָ

מִי אִתְּךָ כִּי תַעֲלֶה לִקְרַאת כְּרוּב/ מִמְשַׁח בְּתֻפֶּיךָ וּבִנְבָלֶיךָ

אוֹ מִי יְסִירָךְ לֵב כְּבַד פֶּה לֶאֱחֹז / בִּכְנַף חֲלִילֶךָ וְתוֹלָלֶיךָ

הַאִם לְנֹטַע הַזְּמִירוֹת תַּאֲמִיר / בִּנְטַע צְנִינֶךָ וְנַהְלוּלֶךָ

אוֹ אִם לְבוֹרֵא נִיב שְׂפָתַיִם שְׂפַת / כִּנּוֹר עֲרֵלָה נָהֲגָה דִגְלֶךָ

בִּנְעִים זְמִירוֹתָיו הֲיַשְׁגִּיחַ בְּבַת / קוֹל עִלְּגֵי שָׂפָה בְנֵי גִילֶךָ

בֵּין שַׁעֲטוֹת פַּרְסוֹת הֲמוֹנֵי מַרְכְּבוֹת/ שִׁירָיו הֲיִשָּׁמַע שְׁפַל קוֹלֶךָ

הַיְכַלְכְּלוּ קִירוֹת לְבָבָךְ נָעֳמוֹ/ בַהְמוֹן פְּרָטֶיךָ וּבִכְלָלֶיךָ

אוֹ עֵטְךָ יִמְשֹׁל בְּגֵאוּת יָם וְעַל / רֻכְסֵי זְמִיר יָרוּץ כְּאַיָּלֶךָ

שֵׁב שַׁח עֲרֵל שָׂפָה וְאַל תַּחְשֹׁב עֲלוֹת / אֶל בֵּין כְּרוּבֶיךָ וְחַשְׁמָלֶיךָ

וּבְאֶשְׁכְּלוֹת הַשִּׁיר רְעֵה עַל יָדְךָ / וּבְחֵיק נְגִינוֹת שֵׁב עֲדֵי גָדְלֶךָ

כַּזֶּה שְׂעִפַּי יֶהֱמוּ עֵת נוֹעֲצוּ / לַעְלוֹת שְׁמֵי הוֹדָךְ וּבִזְבוּלֶךָ

וַיִּבְחֲרוּ כִמְעַט דֳּמִי כִי יָרְאוּ / גַעַת בְּחוֹתָמָךְ וּבִפְתִילֶיךָ

כִּמְעַט לְשׁוֹן שִׁירִי לְחכִּי דָבְקָה / כִּי יָרְאָה מִלַּעֲמֹד אֶצְלֶךָ

וָאֶשְׁמְרָה עַד כֹּה לְפִי מַחְסוֹם עֲדֵי / נֶחְשַׁל לְבָבִי מִסְּבֹל עֻלֶּךָ

כִּי עַל פְּנֵי חוּצוֹת יְפוּצוּן מַעְיְנוֹת / שִׁכְלָךְ לְהַשְׁקוֹת כָּל בְּנֵי חֶדְלֶךָ

בָּאוּ עֲדֵי נֶפֶשׁ הֲמוֹן מֵי נוֹזְלֵי / שִׁירָךְ וְאֵיכָה תַעֲצֹר נוֹזְלֶיךָ

וַיְהִי לְבָבִי בִי סְחַרְחַר וַעֲלֵי / קֹטֶב זְמִיר יִסֹּב בגללך

וַעְצֹר בְּמִלִּין אֵיךְ יְכַלְכֵּל אִם תְּמוֹל / עָרְלַת לְשׁוֹנוֹ מָל בְּמִלּוּלֶיךָ

יִדְמֶה לְבָבָךְ יָם וּמִשָּׁם אֶמְשְׁכָה / נַחְלֵי זְמִיר וִיסוֹבְבוּ אֵלֶיךָ

עַל כֵּן לְבָבִי יַעֲלֹז כִּי הוּא כְבֵן / יוֹנֵק שְׁדֵי שִירָךְ וְכִגְמוּלֶךָ

לֵב נָד עֲלֹז וַחְדֵה רְאֵה נָפַל בְּתוֹךְ / חֶבֶל נְבִיאֶיךָ מְנַת חֶבְלֶךָ

שָׁלוֹם לְךָ אַל תֶּחֱרַד כִּי יַד אֲבִי / כָל נוֹגְנִים אִתָּךְ לְהַשְׂכִּילֶךָ

וּמְשׁוֹל מְשָׁלֶיךָ וְחוּד חִידָתְךָ / וַעְשֵׂה כְצִבְאוֹת שִׁיר בְּמִשְׁאָלֶיךָ

כִּי מִפְּאַת יִקְרַת שְׁלֹמֹה הַגְּבִיר / רוּחַ זְמִירוֹת נוֹסְסָה עֲלֶיךָ


סימן

מִי נָחֲךָ לִבִּי וּמִי יִשְׁבֶּךָּ / כִּי שׁוֹחֲרֶיךָ יֹאמְרוּ אַיֶּךָּ

לִמְצוֹא מְנוּחָה בֵית שְׁלֹמֹה רוּץ וּבֵין / שִׁירוֹ דְבָשׁ תִּמְצָא אֱכוֹל דַּיֵּךָּ.
