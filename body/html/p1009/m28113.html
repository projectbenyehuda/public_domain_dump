אֶרְאֶה שְׂרִיגֵי הַזְּמָן

השיר מעוצב בתבנית הקצידה הערבית: פתיחה – בריאה וטבע; גוף השיר – גאולה; חתימה – תהילה וידידות.


אֶרְאֶה שְׂרִיגֵי הַזְּמָן צוֹמְחִים / אַבִּיט עֲצֵי עֶדְנוֹ וְהֵם פּוֹרְחִים

אַשְׁקִיף בְּאֶשְׁנַבִּי בְּחוּשׁ שִׂכְלִי / אֶפְנֶה לְכוֹכָבָיו וְהֵם זוֹרְחִים

רוּחַ תְּרַחֵף עַל עֲרוּגוֹתָיו / מִמָּר דְּרוֹר עוֹבֵר מְרֻקָּחִים

הֵצִיץ וּפָרַח כָּל פְּנֵי תֵבֵל / הִבְשִׁיל פְּרִי רִמּוֹן וְתַפּוּחִים

חָנְטוּ פְרִי טוֹב כָּל עֲצֵי יַעַר / עַרְמוֹן וְעֵץ אֶרֶז וְהַשִּׂיחִים

וָאֶשְׁאֲלָה אֵיכָה לְכָל מַשְׂכִּיל / גַּם עַל נְתִיבוֹת עוֹבְרֵי אֳרָחִים

לֹא נִרְאֲתָה כָזֹאת לְעוֹלָמִים / מִדֵּי זְמָן שָׁנִים וְלִירָחִים

כִּי הָאֲדָמָה טָמְנָה טוּבָהּ / עַד בּוֹא מְבַשֵּׂר טוֹב לְנִדָּחִים

וַיֹּאמְרוּ לִי בֶּאֱמֶת כִּי קָם / הַיּוֹם מְבַשֵּׂר טוֹב לְנֶאֱנָחִים

הַעוֹרְקִים צִיָּה יְבֻשָּׂרוּ / עַל אִישׁ אֱמוּנוֹת יִהֲיוּ בוֹטְחִים

עוּרוּ יְשֵׁנֵי לֵב וְאַל תִּרְהוּ / הִתְעַתְּדוּ נָא לֶאֱמֶת נָחִים

קַמְתִּי כְשָׁמְעִי לֶאֱסֹף כַּיּוֹם / בֵּיצִים עֲזוּבוֹת לַהֲמוֹן דּוֹלְחִים

אַשְׁרֵי לְמִי יִחְיֶה בְּשׂוּמוֹ אֵל / תַּמּוּ עֲוֹנוֹתָיו וְנִסְלָחִים

אַל גִּיל בְּנֵי אָמָה וְרַב לָכֶם / חוֹזִים לְמַשְׂאוֹת שָׁוְא וּמַדּוּחִים

הָרִים שְׂאוּ שָׁלוֹם לְשׁוֹכְנֵיכֶם / הָעָם אֲשֶׁר הֵם בֶּאֱמֶת אַחִים

הָעָם אֲשֶׁר גָּלוּ בְסוֹד יָמִין / הֵמָּה לְקֵץ יָמִין מְשֻׁלָּחִים

בִּשְׁנַת הֲתַ״ח שַׁלְוָה וְאֵל שַׁדַּי / נִתַּק לְכָל סוּגַר וְלַחַחִים

נִתַּק לְמוֹסֵרוֹת בְּנֵי אָמָה / הַבּוֹגְדִים הַטּוֹמְנִים פַּחִים

וּצְאוּ לְבָתֵיכֶם מְלֵאִים טוֹב / וּשְׂאוּ לְעוֹלָלִים בְּטִפּוּחִים

כִּי יִשְׁמְעֵאלִים שָׁכְנוּ דוּמָה / הֵם מִמְּשׂוֹשׂ תֵּבֵל מְנֻסָּחִים

הַשּׁוֹמְרִים הֶבֶל וְשָׁוְא תָּמִיד / הֵמָּה כְּלָבִים בֶּאֱמֶת נוֹבְחִים

הָרוֹצְצִים דַּלִּים רְאוּ וּדְעוּ / כִּי שָׁוְא וְתָפֵל בֶּאֱמֶת טָחִים

הָעוֹשְׁקִים אֶבְיוֹן בְּכוּר עֹנִי / עַתָּה בְּעוֹלָמָם וּמַצְלִיחִים

בּוֹרֵא לְעוֹלָמִים עֲמֹד בַּדִּין / עַל גּוֹנְבֵי הַשֶּׂה וְגַם טוֹבְחִים

וִישַׁלְּמוּ נִשְׁיָם כְּמוֹ נַמְתָּ / אַרְבַּע כְּדִין תּוֹרָה וּמֻכְרָחִים

עַיִן פְּקַח הַבֵּט לְמַעֲשֵׂיהֶם / הַשּׁוֹכְבִים מִטּוֹת וּמֻסְרָחִים

עוּרָה נְקֹם נִקְמַת עֲבָדֶיךָ / כַּמָּה בְּיַד צָרִים מְנֻתָּחִים

הַקּוֹרְאִים קוּמוּ צְאוּ סוּרוּ / אוֹתָם בְּיַד מַשְׁחִית הֲלֹא נִדְחִים

הִתְבַּשְּׂרוּ עַם אֵל בְּמַפַּלְתָּם / פָּנִים יְהֵי לוּן כַּבְּדֹלָחִים

פָּנִים אֲשֶׁר קָדְרוּ וְהִחְשִׁיכוּ / עַתָּה יְהוּ זַכִּים וְגַם צַחִים

וּפְנֵי הֲמוֹן מוֹנַי אֲשֶׁר גָּבְרוּ / כָּעֵת יְהוּ כּוֹרְעִים וְגַם שׁוֹחִים

אַשְׁרֵי אֱנוֹשׁ יֵיטִיב לְמַעֲשֵׂהוּ / לִפְנֵי אֱלֹהִים חַי, כְּצֹאן מֵחִים

אַל יֹאמְרוּ רֵעַי שְׁכַחְתָּנוּ / הֵמָּה בְּתוֹךְ לִבִּי מְפֻתָּחִים

קֵץ מַעֲשַׂי רֵאשִׁית לְמַחֲשָׁבִי / הַדְרַת פְּנֵי הַדּוֹר נְזִיר אַחִים

הָרָב סְעַדְיָהוּ בְּנוֹ יוֹסֵף / יִהְיוּ כְמֹץ אוֹיְבָיו וְגַם נִמְחִים

גַּם לַאֲהוּב לִבִּי זְכַרְיָהוּ / אָדוֹן בְּנוֹ יוֹסֵף יְלוּד קָרְחִים

גַּם לַעֲדַת אֵל אֶכְרְעָה עַד כִּי / יִרְצוּ לְמַהְלָלִי כְּנִיחוֹחִים

וּדְעוּ אֲדוֹנַי כִּי לְזֶה הַקֵּץ / סוֹדוֹת עֲמֻקִּים בֶּאֱמֶת צוֹרְחִים

נִכְתַּב בְּדָנִיאֵל בְּבֵאוּר טוֹב / אִם לֹא עֲוֹנוֹת יִהֲיוּ מוֹחִים

לָכֵן וּמִי אָנִי אֲאַלְּפְכֶם / כָּל הַלְּשׁוֹנוֹת נִבְרְאוּ לַחִים

אֵל חַי לְיִשְׁעֵנוּ וְאִם אֵלֶּה / קָמִים בְּמוֹ רֶכֶב וּבִרְמָחִים

נִדְגֹל בְּשֵׁם אַדִּיר אֲשֶׁר צִוָּה / אֶרֶץ וְשָׁמַיִם וְנִמְתָּחִים

אַשְׁרַי וְאַחְלַי גַּם וּמִי יִתֵּן / אֶרְאֶה שְׂרִיגֵי הַזְּמָן צוֹמְחִים.
