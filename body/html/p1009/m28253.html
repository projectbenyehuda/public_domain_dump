אֶל אֵל אַדִּיר

אֶל אֵל אַדִּיר בִּמְרוֹמֵי עָרֶץ / אֵל אֱלֹהִים יי דִּבַּר וַיִּקְרָא אָרֶץ

אַזְכִּירָה עֲבוֹדַת דּוּמָה וּמַשָּׂא / אֵלֶיךָ יי נַפְשִׁי אֶשָּׂא

בִּימֵי בְנֵי קֵדָר גָּבְרוּ וְגָאוּ / בַּחוּרִים טְחוֹן נָשָׂאוּ

גָּדְעוּ קוֹמָתִי בְּחֹמֶר וּבִלְבֵנִים / גּוֹי עַז פָּנִים

דְּחָפוּנִי בְאַמַּת בִּצְעָם / דָּתֵיהֶם שׁוֹנוֹת מִכָּל עָם

הֲדָפוּנִי בְּבֵית הַכֶּלֶא / הֲלַמֵּתִים תַּעֲשֶׂה פֶּלֶא

וְגַם שָׂמוּ רַגְלַי בַּסַּד / וּלְךָ יי חָסֶד

זָמְמוּ עַל בָּנֶיךָ לְהַכְחִידָם / זֶה סֵפֶר תּוֹלְדוֹת אָדָם

חָשְׁבוּ לְהַעֲבִידָם בְּמֶרֶץ / חֶסֶד יי מָלְאָה הָאָרֶץ

טָפְלוּ שֶׁקֶר וְהִתְעִיבוּ עֲלִילָה / טָבַעְתִּי בִיוֵן מְצוּלָה

יַחַד קִוּוּ לְאַבְּדֵנִי / יְהִי נָא חַסְדְּךָ לְנַחֲמֵנִי

כָּל יָמַי יָרֵא וְזוֹחֵל / כָּלוּ עֵינַי מִיַּחֵל

לָעַד הֶעֱבִידוּנִי בְּפֶרֶךְ / לֹא אֲלֵיכֶם כָּל עוֹבְרֵי דֶרֶךְ

מַעֲלִי הִמְשִׁילַנִי לְאָבָק דָּק / מַה נְדַבֵּר וּמַה נִּצְטַדָּק

נָפַלְתִּי בְּיַד מְעַנָּי / נִפְּלָה נָא בְּיַד יי

סְבָבוּנִי עָנְיִי וְצִירִי / סָמַר מִפַּחְדְּךָ בְּשָׂרִי

עָבַרְתִּי עַל מִצְוָתָךְ קוֹנִי / עֲוֹן עֲקֵבַי יְסֻבֵּנִי

פָּשַׁעְתִּי וְהִנֵּה נָפְלוּ פָנָי / פַּלְגֵּי מַיִם יָרְדוּ עֵינָי

צְרָפַנִי עֲוֹנִי בְּכוּר עוֹנִי / צַר וּמָצוֹק מְצָאוּנִי

קוֹמְמִיּוּת שִׂבַּרְתִּי וְאַרְגִּיעָה / קִדַּמְתִּי בַנֶּשֶׁף וָאֲשַׁוֵּעָה

רַחוּם הַצִּילֵנִי וְאִמָּלְטָה / רָאִיתִי בוֹגְדִים וָאֶתְקוֹטְטָה

שׁוֹקֵד וּמוֹצִיא עוֹלָמוֹ לְיֵשׁ מֵאָיִן / שָׁמְרֵנִי כְאִישׁוֹן בַּת עָיִן

תּוֹכֵן נְשָׁמוֹת בְּגַלְגַּלֵּי מַעְלָה / תַּבַּעְנָה שְׂפָתַי לְךָ תְּהִלָּה

אָדוֹן נִקְרֵאתָ צוּר עוֹלָמִים / אֵל מֶלֶךְ יוֹשֵׁב עַל כִּסֵּא רַחֲמִים

בָּאתִי לְחַלּוֹת פָּנֶיךָ עַל רֹב עֲוֹנָי / בַּצַּר לִי אֶקְרָא יי

נִכְפַּפְתִּי וְאַתָּה זוֹקֵף כְּפוּפִים / נוֹצֵר חֶסֶד לָאֲלָפִים

רַחֵם יְחִידָתִי בְּצֵאתָהּ וּבְבוֹאֶיהָ / רוֹקֵעַ הָאָרֶץ וְצֶאֱצָאֶיהָ

בַּקֵּשׁ עַבְדָּךְ צוֹעֵק מִצָּרוֹתָיו / בּוֹנֶה בַשָּׁמַיִם מַעֲלוֹתָיו

נִין הָאָמָה כַלֵּה עִם מְתֵי הַשֶּּׁקֶר / נַפְשִׁי לַיי מִשּׁוֹמְרִים לַבֹּקֶר

חִשַּׁבְתִּי דְּרָכַי בִּלְבָבִי וְרַחְשִׁי / חֶלְקִי יי אָמְרָה נַפְשִׁי

לוֹבֵשׁ צְדָקוֹת שִׂיחִי אַל תֶּשִׁי / לָמָה יי תִּזְנַח נַפְשִׁי

קוּמִי וְשִׁבְתִּי שַׂמְתִּיךְ לְנֶגֶד עֵינָי / קָרָאתִי שִׁמְךָ יי.
