קִינַת סוֹפְדִים (עַל־אָסוֹן שֶׁקָּרָה בְמַנְטוֹבָה)

Jacob Raffael Saraval

[נולד בוֵנֶצְיָה בשנת 1707 ומת במַנְטוֹבָה בשנת 1782]


קִינַת סוֹפְדִים (עַל־אָסוֹן שֶׁקָּרָה בְמַנְטוֹבָה)

– – ͝ – – – – – – – –

תִּצַּל וְתִבָּהֵל אֹזֶן שׁוֹמַעַת

מֵאֵין הֲפוּגוֹת אַל־תִּדֹּם בַּת־עָיִן

רוּחַ אֱנוֹשׁ תִּכְהֶה וּכְרוּחַ אָיִן

כָּל־רֹאשׁ לְקָרְחָה כָּל־זָקָן לִגְרַעַת

לַמְקוֹֽנְנוֹת קִרְאוּ נֶהִי תִשֶּׂאנָה

אֶל־הַחֲכָמוֹת שִׁלְחוּ וּתְבֹאֶינָה.


הֶהָֽיְתָה כָזֹאת שָׁאֹל יִשְׁאָלוּ

דּוֹר אַחֲרוֹן בָּנִים כִּי יִוָּלֵדוּ

מַשְׂאוֹת עֲיֵפוֹת מִמֶּנּוּ כָבֵדוּ

בָּנִים וְאָב וָאֵם יַחַד אֻמְלָלוּ

יִמַּס לְבַב אִישׁ תִּרְפֶּינָה יָדָיִם

תַּחַת פְּתִיגִיל שַׂק יַחְגֹּר מָתְנָיִם.


שִׁשִּׁי בְשַׁבָּת בִּכְנִיסַת הַקֹּדֶשׁ

שָׁבַת מְשׂוֹשׂ שַׁבָּת עַד־כִּמְעַט רָגַע

הַשִּׁיר לְצִיר נֶהְפַּךְ עֹנֶג אֶל־נָגַע.

הֶגֶ֒ה סְפַר זִכְרוֹ בִּימוֹת הַחֹדֶשׁ

יֶרַח שְׁלִישִׁי חֻלַּל יִפְעָתֵהוּ

הַלְק֒וֹת פְּרַט שָׁנָה מַר זִכְרוֹנֵהוּ.


בֵּיתָה חֲתוּנָה מַקְהֵלוֹת נוֹעָדוּ

שָׂשׂוֹן סְאוֹן אָסוֹן פֶּתַע נִפְגָּשׁוּ

רִצְפָּה וְקוֹרוֹת מִקִּירוֹת נִתָּשׁוּ

אֵין מָעֳמָד קַרְסֻלַּיִם מָעָדוּ

עַד דּוֹר שְׁלִישִׁי נִסָּעוּ נִגָּלוּ

שָׁם נָֽפְלוּ דֹּחוּ קוּם לֹא־יָכֹלוּ.


בֵּין קוֹל הֲמוֹן שִׁירִים צִירִים נָבֹכוּ

פֶּה פָּֽתְחָה, דִּמוּ, אֶרֶץ לִבְלֹעַ

קוֹל אוֹֽמְרִים “הַאִם תִּמְנוּ לִגְוֹעַ”

חִישׁ בּוֹֽרְחִים וּכְלַפִּידִים יַהְלֹכוּ

אֶל־קוֹל פְּחָדִים נָס מֵרֶדֶת שַׁחַת

בַּדֵּי שְׁאוֹל יָרַד עַד־עָפָר נָחַת.


קוֹל קוֹֽרְאִים מִן־הַמֵּצַר עֶזְרָתָה

קוֹל שׁוֹֽפְכִים נַפְשָׁם אֶל־שׁוֹכֵן עָרֶץ

קוֹל נוֹהֲמִים כָּאוֹב מִתַּחְתִּית אָרֶץ

קוֹל צוֹעֲקִים “מִי לָנוּ לִישׁוּעָתָה”

כֹּל דּוֹבֲבִים דָּוִים "נוֹדֶה פִשְׁעֵנוּ

דַּיַּן אֱמֶת עַל־כָּל־הַבָּא עָלֵינוּ".


נִבְהָל שְׁמֹעַ הִתְאַזָּר לָלֶכֶת

הַצֵּל נְפָשׁוֹת נִלְקָחִים לַמָּוֶת

בַּאְשֶׁר חֲלָלִים שָׁם אֹפֶל צַלְמָוֶת

גִּבּוֹר וְאַמִּיץ לֵב קָרַב לִמְלֶאכֶת.

מִי מָֽצְאוּ מֻכֶּה אוֹ מִי קָרוּעַ

מִי מֵת וּמִי עֻלְפֶה אוֹ מִי פָצוּעַ.


מִי עֻוְּרוּ עֵינָיו אֶל־מִי יָצָאוּ

מִי כָלְאָה רוּחוֹ גָרוֹן נֶחְנָקֶת

מִי נִשְׁבְּרָה צַלְעוֹ מִי הַמַּפְרָקֶת

הֵילֵל בְּרֹאשׁ אֶרֶז נָפַל הִכָּהוּ

מִי נִגְדְּעָה יָדָם אוֹ מִי רַגְלֵימוֹ

הֵכִין כְּלֵי מַשְׁחִית מָוֶת אֵלֵימוֹ.


“לִבִּי לְמוֹ־אָב” הַבֵּן מַר צוֹרֵחַ

הָאָב לְמוּל צוֹעֵק “אָן אֶמְצָאֵהוּ”

הָאֵם “בְּנִי אַיּוֹ בַּעְלִי אַיֵּהוּ”

תִּצְעַק וְתִתְאוֹנֵן מַר תִּתְיַפֵּחַ.

פִּתְאֹם שְׁלָשְׁתָּם נִשְׁמָתָם פָּרָחָה

מִמְקוֹם אֲשֶׁר נַפְשָׁם לֶעָפָר שָׁחָה.


בִּכְ֒יָה לְאוֹת מִסְפֵּד מִסְפַּר הַגָּבֶר

בָּחוּר וְזָקֵן שָׂב וָטָף אִתָּמוֹ

אֶבְ֒כֶּה נְשֵׁי עַמִּי כִּי סָפוּ לָמוֹ

הַ֒ס לָךְ לְהַזְכִּיר נִתָּנוּ לַקָּבֶר.

אַב רַחֲמִים חַסְדָּךְ עַל־בֵּ֒ן אִמָּצְתָּ

כֵּן מִשְׁפָּטֵי צִדְקָךְ אַתָּה חָרָצְתָּ.


דָּת נִתְּנָה מֵאֵת שָׂרַי הַפֶּלֶךְ

הוֹצִיא פְגָרִים פֶּן־תִּבְאַשׁ הָאָרֶץ

כִּי יַעֲלֶה בָאְשָׁם הַרְבּוֹת הַפָּרֶץ

נֶעְזַב שְׁבוּת שַׁבָּת עַל־מִצְוַת מֶלֶךְ.

הוֹי לֵיל יְהִי גַלְמוּד יֶחְשַׁךְ נִשְׁפֵּהוּ

הוֹי לוֹ לְדוֹר מִקְרֶה כָזֹאת יִקְרֵהוּ.


עַל־שֵׁשׁ עֲגָלוֹת לִקְבָרוֹת הוּבָלוּ

לֹא מָצְאוּ גַם שָׁם שׁוּבָה וָנָחַת

הָרוֹפְאִים טֶרֶם יוֹרִידוּם שָׁחַת

בָּאוּ כְדָת לִרְאוֹת גּוּפוֹת חֻבָּלוּ.

אִם יַעֲמֹד לֵב אִישׁ בִּרְאוֹתוֹ אֵלֶּה

אִם תֶּחֱזַק יָדוֹ רוּחוֹ לֹא תֵלֶא?


אֶל־רַעֲשַׁת אוֹפָן סוֹבֵב בַּדֶּרֶךְ

נָשִׁים בְּעַד הַחַלּוֹנִים תִּקְרֶאנָה

אִשָּׁה רְעוּתָהּ קִינָה הוֹי תִּשֶּׂאנָה

זֹאת צָעֲקָה “אֶל־מַכָּתִי אֵין עֵרֶךְ”

תָּשִׁיב אֲמָרִים הִיא "מַכְאוֹבִי נֶצַח

בַּעְלִי בְּנִי אוֹיָה לֹא אֶשְׁכַּח נֶצַח."


אָמְנָם עֲדֶן מֵתִים לֹא הוּנַח לָמוֹ

כִּי יוֹם הֲנָחָה הוּא חַיִּים יַרְגִּיעוּ

גַּם יוֹם אֲנָחָה הוּא עַד־כִּי יַבְקִיעוּ

אוֹרָם כְּסִילֵי־רוּם בִּמְסִלּוֹתָמוֹ.

אָז אָזְרוּ עֻזָּם מָתְנֵי כָל־גָּבֶר

בָּעֵת מְצֹא מֵתִים מָקוֹם שָׁם קָבֶר.


אַחַי רְאוּ קוֹרוֹת זָרוֹת כִּי רָבוּ

אִם יוּֽכְלוּ הָכִיל אַנְשֵׁי הֶחָיִל

אִם נִסְפְּדָה סָפֹד יוֹמָם וָלָיִל

לֹא נוּֽכְלָה סַפֵּר כַּמָּה נִרְחָבוּ.

יָחִיל בְּקִרְבּוֹ אִישׁ לֹא יִסְבֹּל סֵבֶל

כִּי יֵשׁ דְּבָרִים הַרְבֵּה מַרְבִּים הֶבֶל.


אֵבֶל עֲלֵי אִישִׁים אֵין עֶזֶר לָמוֹ

כִּי לֻקְּחָה מֵהֶם תִּפְאֶרֶת בָּיִת

אֵבֶל עֲלֵי בָּנִים כִּשְׁתִילֵי־זָיִת

הָיוּ וְאֵין סוֹמֵךְ תּוֹמֵךְ אוֹתָמוֹ

אֵבֶל עֲלֵי־בַת עִם־אִמָּהּ נִפְרָצָה

כַּלָּה כְּמוֹת שֶׁהִיא כָּלָה נֶחְרָצָה.


אֵבֶל עֲלֵי אִמָּהוֹת כִּי אָבָדוּ

בֵּן מַחֲמַד בִּטְנָם לָמוֹ מִשְׁעָנֶת

אֵבֶל עֲלֵי אִשָּׁה מַר מִתְאוֹנָנֶת

אִישָׁהּ שְׁנֵי בָנִים מֶנָּה נִפְקָדוּ

אֵבֶל עֲלֵי הָרָה כִּי בָא לָהּ חֵבֶל

אָז נָֽפְחָה נַפְשָׁהּ רַבָּתִּי אֵבֶל.


אֵבֶל עֲלֵי הַמְלַמֵּד אָדָם דָּעַת

מַשְׂכִּיל לְהֵיטִיב תַּלְמִידִים בַּדָּרֶךְ

אֵבֶל עֲלֵי רוֹעִים אֵין לָמוֹ עֵרֶךְ

אֶל־בֵּית כְּנִיסָתָם בָּאָה מִגְרָעַת

אֵבֶל עֲלֵי־צִיר שָׁר לִפְנֵי קוֹנֵהוּ

אֶל־שֽׁוֹלְחָיו נֶאְמָן תָּם מִפְעָלֵהוּ.


אֵבֶל עֲלֵי אִשָּׁה פִּתְאֹם אָבָדָה

בַּעַל נְעוּרִים אָב וָאֵם אָח יָחַד

אֵבֶל עֲלֵי־אִישׁ נָס מִקּוֹל הַפָּחַד

נָפַל בְּפַחַת זוֹ רַבִּים לָכָדָה

אֵבֶל עֲלֵי יָקָר נִשְׁבַּר כַּנֵּבֶל

אֵבֵל אֲבֵלִים הִנֵּה הַכֹּל אֵבֶל.


נַפְשׁוֹת חֲסִידֵי אֵל קוֹלִי תִשְׁמַעְנָה

מִיתָה מְכַפֶּרֶת אַשְׁמָה וָרֶשַׁע

אֶל־הַמְּנוּחָה בָאתֶם מַעְיַן יֶשַׁע

שָׁם נַפְשְׁכֶם רַב טוּב צָפוּן תִּשְׂבַּעְנָה.

נַחְנוּ נְחַלֶּה חַנּוּן וִיחָנֵּנוּ

וּזְכוּתְכֶם הָגֵן יָגֵן עָלֵינוּ.


הַדּוֹר רְאוּ אַתֶּם מִלַּי הַקְשִׁיבוּ

לֹא אֶל־בְּנֵי עִירִי בִּלְבַד אַבִּיעַ

אֶל־שׁוֹבֲבִים שָׁבִים דֵּעִי אוֹדִיעַ

אֶת־דַּרְכְּכֶם עִזְבוּ שׁוּבוּ הֵיטִיבוּ

רֵעַ לְרֵעוֹ לֹא יָצוּדוּ חֵרֶם

הַחוֹקְקִים חֻקִּים יָסִירוּ חֵרֶם.


הַתַּחֲרוּת הָסֵר שִׂנְאַת הָרֵעַ

לַשָּׁוְא בְּרִית כָּרֹת הִשָּׁבַע שֶׁקֶר

הַצְמֵד לְשׁוֹן מִרְמָה וּמְשַׁלֵּשׁ דֶּקֶר

הִתְרַחֲצוּ חוֹטְאִים חִדְלוּ הָרֵעַ.

יָתוֹם וְאַלְמָנָה צֶדֶק תִּשְׁפֹּטוּ

שׁוּבוּ בְּנוּ בָתִּים עוֹד לֹא יִמּוֹטוּ.


הָהּ אֵל נְקָמוֹת עַד־עַתָּה הוֹפָעְתָּ

אֶל־תַּעֲרָהּ הָשֵׁב חֶרֶב נוֹקָמֶת

יוֹנָֽתְךָ הוֹמָה נֶגְדָּךְ נוֹהָמֶת

“זָכְרָה בְּרִית אָבוֹת כַּאְשֶׁר נִשְׁבָּעְתָּ”

אִם נָֽשְׂאָה עַד־כֹּה מַשְׂאַת אֵימֶיךָ

אַל־תֶּחֱשֶׁה מַהֵר חִישׁ נִחוּמֶיךָ.


יָהּ עַמְּךָ קוֹרֵא “בָּרוּךְ” בִּשְׁמֶךָ

דַּיַּן אֱמֶת עוֹשֶׂה מִשְׁפָּט וָצֶדֶק

הַחְזֵק בְּמָעֻזּוֹ מַחְזִיק כָּל־בֶּדֶק

תַּעַל תְּחִנָּתוֹ לִמְעוֹן שָׁמֶיךָ

שַׁכַּח עֲמָלוֹ שׁוּבָה וּתְחַיֵּנוּ

הַשְׁלֵךְ מְצוּלוֹת־יָם כָּל־חַטֹּאתֵינוּ.


סוּרָה שְׁבָה וּשְׁכֹן אֵל חַי קִרְבֵּנוּ

רַחֵם פְּלֵיטַת צֹאן טִבְחָה נֶחְשָׁבָה

אֵל נָא רְפָא־נָא לָהּ עָדֶיךָ שָׁבָה

וּמְחַל עֲוֹן אָבוֹת יַשֵּׁר דַּרְכֵּנוּ.

אָנָּא סְלַח יָשׁוּב עַם זוּ יִסָּרְתָּ

לָשׂוּשׂ בְּהֵיטִבְךָ עַם זוּ יָצָרְתָּ.


מֶלֶךְ בְּיָפְיוֹ תֶּחְזֶינָה עֵינֵינוּ

נֹגַהּ כְּאוֹר תִּהְיֶה אוֹר שִׁבְעָתָיִם

טַלֵּי תְחִיָּה יַרְעִיפוּ שָׁמָיִם

וִירַנְּנוּ מֵתִים “חַיִּים עוֹדֵינוּ”

בָּנוּ רָצָה הָאֵל בִּרְצוֹן עַמֵּךָ

הוֹאֵל וּפָקְדֵנוּ בִּישׁוּעָתֶךָ.

(1781)
