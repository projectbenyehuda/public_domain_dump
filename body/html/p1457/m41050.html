ובכן יי קנני ראשית דרכו


אֶרְאֶלֵּי עֲרָפֶל / וְשׁוֹכְנֵי שֵׁפֶל / עַד אַיִן לֹא הָיוּ

אֹפֶל מַחְשָׁךְ / וְאוֹר וָחֹשֶׁךְ / טֶרֶם יִהְיוּ

אַלְפַּיִם שָׁנִים / קֹדֶם רִאשׁוֹנִים / שִׂפְתוֹתָיו לִי חִוּוּ

אֲנִי וָאַתְּ נֵשֵׁב / בקרה צֵל מוֹשָׁב / אֲשֶׁר לָנוּ שִׁוּוּ

אֲדַקְדֵּק אֲמָרַיִךְ / אֲבָאֵר דְּבָרַיִךְ / כִּי מְאֹד נָאווּ

אוֹר פָּנָיו / וּדְבַר אָפְנָיו / רִקְחֵי עֲסִיסִי רִוּוּ

אוֹת עִם אוֹת כתבתי / וּבַפַּרְגּוֹד שֶׁלּוֹ יָשַׁבְתִּי / וּפֶקֶק דְּבָרַי אֶצְבְּעוֹתָיו טָווּ

מְגִלָּה / בְּרוּכָה גְלוּלָה / יָדָיו לִי אִוּוּ

אֹמֶץ הִשְׁמִיעַנִי / בַּדּוֹרוֹת אֲשֶׁר שִׁעְשְׁעַנִי / במה בְּרוּאִים יִהְיוּ

אַרְקָה אַנְטִיעַ / דּוֹק אַרְקִיעַ / וְשׁוֹכְנֵיהֶם עֲבוּרֵךְ יִהְיוּ


וְשִׁבְעָה קֹדֶם כֹּל

עֲבוּרָם בָּרָא כל

יִשּׁוּב הָעוֹלָם וְכִלְכּוּל כֹּל

בְּמַעְמַד תְּמִימָה הֵשִׁיב קָדוֹשׁ


בָּנָה כָל בִּנְיָן / הֵכִין כָּל עִנְיָן / וְהֵעִיץ אוֹתָהּ

בְּאוֹרָהּ נִטְלַל / הֲדָרָהּ גָּלַל / עִמּוֹ בִּהְיוֹתָהּ

בִּגְלִילוֹת יָדוֹ רִשַּׁם / וְנָם תֵּצֵא מִשָּׁם / עַד יוֹם עִתָּהּ

בְּפִיו הָג שִׁנּוּנָהּ / הִטָּה חֶסֶד עַל לְשׁוֹנָהּ / לְפִי רוֹב מִדָּתָהּ

בְּכִתְרוֹ עֲרָכָהּ / בְּנִזְרוֹ הֶאֱרִיכָהּ / כְּרֹחַב תַּבְנִיתָהּ

בִּכְתֻבָּה מִתְפָּאֵר / הֵטֵיב מְבָאֵר / זַיִן מִלְחַמְתָּהּ

בְּאֶלֶף דּוֹר בירק / וְאַחֲרָיו דִּקְדַּק / לִשְׂמוֹחַ בְּחֻפָּתָהּ

בֵּית הוֹנוֹ אֲצָרָהּ / עֲצָרָהּ וּנְצָרָהּ / לְעֵת מְלִיאָתָהּ

בָּהּ לְהַשְׂכִּיל כָּל מַשְׂכִּיל / וְהֶאֱרִיכָהּ מְכַלְכֵּל / כְּיַד גְּדֻלָּתָהּ

בְּמַחְמַד גְּנָזָיו / בְּפִרְשׁוּז מַפְרִיזָיו / עָשׂ יְשִׁיבָתָהּ

בְּמַעְמַד תְּמִימָה הֵשִׁיב קָדוֹשׁ


גָּשׁ לְפַתּוֹתָהּ רַךְ / בטרם מַעֲנֶה רַךְ / בְּרוֹךְ לְשׁוֹן תְּשׁוּבָה

גּוֹשִׁי עַד הֲלוֹם / וְנַעֲשֶׂה שָׁלוֹם / חוֹלַת אַהֲבָה

גִּהֵךְ יִפָּקֵק / נֵרֵךְ יֵחָקֵק / לְמַרְבֵּה יְשִׁיבָה

גַּדְתִּי לָצוּד צֵידֵךְ / וַאֲנִי בְּצִדֵּךְ / נוֹטֶה מְסִבָּה

גָּדֵר אֲנִי לְחֶמְדָּתֵךְ / מַפְתֵּחַ אֲנִי לְבֵיתֵךְ / וְצָהֳרֵךְ לֹא יִכְבֶּה

גְּרוֹנֵךְ מֻטָּל / בְּסִיחֵךְ אוֹרִיד טַל / לַיּוֹצֵא וְלַבָּא

גִּילִי מְאֹד בַּבְּרוּאִים / כִּי תַחְתַּי נֹרָאִים / מִי יַעֲבֹר וּמִי יוּבָא

גָּדְלִי הוּא גָדְלֵךְ / גּוֹרָלִי הוּא גּוֹרָלֵךְ / שָׁוִים בִּנְתִיבָה

גֶּפֶן רִקְחִי / בְּעוֹלָמִי הִלָּקְחִי / וקבל כְתֻבָּה

דָּבָר אוֹתְךָ אָשִׁיב / אָשִׂיחַ וְתַקְשִׁיב / וּרְצוֹנְךָ אֶתְאַוֶּה

דֶּרֶךְ שִׁבְתְּךָ אֶחְפֹּץ / מִמְּקוֹמְךָ לֹא אֶקְפֹּץ / וְהִיא לִי כָּל הַתַּאֲוָה

דַּי תָּשִׁית וְיֶהִי / בְּמַאֲמָרֶיךָ עוֹלָם יֶהִי / עֵת תְּצַוֶּה

דִּמְיוֹנִים תַּעֲנִיק / דַּדִּים תֵּנִיק / תַּשְׁקֶה וְתִרְוֶה

דִּינְךָ אֱמֶת / חוֹתָמְךָ אֱמֶת / קָטוֹן וְגָדוֹל נֶגְדְּךָ שָׁוֶה

דַּעַת תִּמְצָא / עֲנָוָה תִרְצֶה / כְּמוֹצָא גַן רָוֶה

דַּלְתוֹתֶיךָ מימינת טֹהַר / קְדֻשָּׁה וַעֲנָוָה וְזֹהַר / וּלְךָ מִי יִשְׁוֶה

דּוֹפֵק עֲלִיּוֹת / דּוֹבֵק תַּחְתִּיּוֹת / לְדָבָר מְצֻוֶּה

דַּרְכְּךָ תְמִימָה / מְסִלָּתְךָ אֵימָה / אֶשְׁוֶה וְתִשְׁוֶה

הַחֶסֶד לָךְ אֶקְשֹׁר / לְאוֹרַח מִישׁוֹר / פִּקְחִי עֵינֵךְ

הַבְּרִית בָּךְ לַכֹּל / טֶרֶם נִבְרָא כֹל / הִיא בֵינִי וּבֵינֵךְ

הַחֲצוּבִים יֵחָצֵבוּ / הַקְּצוּבִים יִקָּצֵבוּ / מִשִּׂיחַ לְשׁוֹנֵךְ

הַכָּבוֹד מִמֵּךְ / הָעֹשֶׁר עִמֵּךְ / הַחַיִּים הוֹנֵךְ

הַשּׁוֹמֵעַ יִשְׁמָע / וְיָבִין וְיִשְׁמַע / רָז הֶגְיוֹנֵךְ

הָרוֹצֶה לִטֹּל / יָבוֹא וְיִטּוֹל / לִמְבַקְשֵׁי הוֹנֵךְ

וְגָדוֹל הוֹדְךָ / וְעָצוּם כְּבוֹדְךָ / בְּזֶמֶר וְהַלֵּל

פָּקַדְתִּי לְהִקְהוּל / הַנּוֹלָד מָהוּל / סְפָרֶיהָ לְהַאֲהִילָן

פֵּרַשׁ בְּמַקְלוֹ ונקפה / וְהַסֻּלָּם צָפָה / בַּמָּקוֹם אֲשֶׁר לָן

פָּנָה וְרָאָה / מַלְאֲכֵי מַרְאָה / עוֹלִים לְמַעְלָן

פֶּלֶס דְּיוֹקְנוֹ / בְּכִסֵּא קוֹנוֹ / צוּר הִשְׂכִּילָן

פָּגַע בִּמְקוֹם לִינָה / וְחָן עַל גַּבּוֹ שְׁכִינָה / וּמַלְאָכִין כֻּלָּן

פָּרַח לְהָעֲמִיד הָאָרֶץ / וְשָׁכַן בִּצְבִי אֶרֶץ / וּבִקְדֻשָּׁתָהּ הוּלַן

פְּרָחִים שְׁנֵים עָשָׂר פָּרְחוּ לוֹ / בִּמְסִבּוֹת אָהֳלוֹ / לִמְסַכֵּךְ חוולן


וְשִׁבְעָה קֹדֶם כֹּל

עֲבוּרָם בָּרָא כל

יִשּׁוּב הָעוֹלָם וְכִלְכּוּל כֹּל

בְּמַעְמַד תְּמִימָה הֵשִׁיב קָדוֹשׁ


צַח וְאָדוֹם אַתָּה / בְּתוֹךְ נְקִיִּים רַדְתָּה / מִי יֹאמַר לְךָ מַה תַּעֲשֶׂה

צוֹפֶה סְתָרִים / חוֹפֵשׂ חֲדָרִים / רָז הַמַּעֲשֶׂה

צִדְקְךָ כְּנוֹגַהּ נָטוּי / נְהוֹרָה כִּתְרָךְ עָטוּי / אוֹמֵר וְעוֹשֶׂה

צוֹפוֹת עֵינֶיךָ רְחוֹקִים / דְּלוֹת סֵתֶר מֶרְחַקִּים / רָז הַמַּעֲשֶׂה

צַדִּיק אֲשֶׁר הָקְדַּשׁ / מֵרֶחֶם מְקֻדָּשׁ / אַרְבַּע אֲחָיוֹת מְנַשֵּׂא

צֵדָה בִּרְמִיּוֹת הִשְׂכִּיל / וּלְאָבִיו אוֹתָהּ הֶאֱכִיל / בְּרָכָה לִישָׂא

צָעַק הוֹרוֹ בְּמַר / מִי אֵפֹא אָמַר / אֲשֶׁר מֶנִּי כִּסָּה

צָרְחָה בַּת קוֹל / הוּא הַקּוֹל קוֹל / אוֹי אוֹי כִּי לוּלֵי עמסה

צָהַל וְתָבַע / נָשִׁים אַרְבַּע / בְּחֻפַּת כְּנִיסָה

צִוָּה יְשׁוּעָתוֹ / וְהֵכִין מִבֵּיתוֹ / אֲשֶׁר אוֹתִי יִשָּׂא


וְשִׁבְעָה קֹדֶם כֹּל

קֵץ קָרַב מָלֵא וְשָׁלֵם / רְחָבָה קוּמִי לְהִשְׁתַּלֵּם / בְּיַד שָׁלִיחַ וְסוֹפֵר

קָם נֶאֱמַן בֵּיתִי / מַנְהִיג עֲדָתִי / עֲנָוָה מְסַפֵּר

קָפַץ לִרְעוֹת צֹאן / וְהָלַךְ בְּרָצוֹן / אֲמָרַי לְסַפֵּר

קְרוּאִים בְּדִי זָהָב גָּאוּ / קְדוֹשִׁים טִמְּאוּ / וְנָם סְלַח ואכפר

קָבַע תַּחַן סְלִיחָה / וְשָׁפַךְ סִיחָה / וַאֲנִי אֲכַפֵּר

קָד וְהוֹדָה / מְחֵינִי נָא / אִם לֹא תְכַפֵּר

קְרָאתִיו לִמְעוֹנָה / וְשָׁמַע וְעָנָה / וַאֲנִי אֲסַפֵּר

קוֹלוֹ בִּשְׁנֵים עָשָׂר מִיל / עַל שְׁנֵים עָשָׂר מִיל / תַּלְמוּד מְסַפֵּר

קִנְאָה לֹא אָחַז / קְלַסְתּוֹרִי חָז / תְּפִלִּים תּוֹפֵר

קְהַל בַּת רַבִּים / הֶגְיוֹנֵךְ יוֹצֵא כְּשַׁלְהָבִים / בָּךְ לְהִתְכַּפֵּר

רֹאשׁ הַנְּבִיאִים יָבוֹא / בְּלִבּוֹ אָבוֹא / וַאֲקַבְּלֶנּוּ בְּשִׂמְחָה

רֹאשׁוֹ אֲנַשֵּׁק / גּוּפוֹ אֲחַבֵּק / רוּחוֹ לְרוּחִי נָחָה

רוֹם יָרוּם בִּי / יִשָּׂאֵנִי אוֹהֲבִי / בִּכְתֻבָּה פְּתוּחָה

רֵעָיו יְשַׁעַשְׁעוּנִי / בְּבָתֵיהֶם יִטָּעוּנִי / כַּמְּזוּזָה הַמְּתוּחָה

שָׁמַיִם מוֹר נָטָפוּ / הָרִים עָסִיס הִטִּיפוּ / בִּקְטֹרֶת כְּלוּלָה

שערים גַּן עֵדֶן / כָּל בְּשָׂמִים עִדֵּן / וְהָעוֹלָם נִתְמַלָּא

שָׁמָיו הִטָּה / וְלָבַשׁ וְעָטָה / וַאֲפַרְסְמוֹן מִלָּא

שָׁחוּ הָרִים / מְעוּכִים וּדְקוּרִים / וְעָשָׁן עָלָה

שׁוֹרְרוּ אֵלִים / מַלְאָכִים קַלִּים / רִנָּה וְצָהֳלָה

תֵּבֵל וְכָל אֲשֶׁר בָּהּ גָּעֲשָׁה / וְנִתְפַּלְּצָה וְנִתְפַּלְּשָׁה / בְּשָׁמְְעָהּ אָנֹכִי יְיָי
