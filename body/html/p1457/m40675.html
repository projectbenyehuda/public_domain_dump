אַלּוּפִים מְסֻבָּלִים

קדושה לשבועות


אַלּוּפִים מְסֻבָּלִים בַּעֲבֵי נָגְהֶיךָ

אִזַּנְתָּ דִּבּוּר עִבְרִית כְּמֵהֶיךָ

אָנֹכִי יְיָ אֱלֹהֶיךָ


בְּתַחְתִּית הַר לָהֶם בְּהַגְבִּילְךָ

בַּאֵר הוֹרֵיתָם בְּצֶרַח קוֹלְךָ

בַּעַל לֹא יִהְיֶה לְךָ


גְּמוּלֵי מֵחָלָב בְּיָמִין מְנוֹסָסָה

גְּדֵילֵי מֵחֲלַמִּישׁ פִּנּוּק עֲלִיסָה

גָּנוּז לֹא תִשָּׂא


אֲמָרוֹת טְהוֹרוֹת לְזוּ מְזֻקָּקוֹת

עִנְיְנֵי זַךְ בְּאֶצְבַּע חֲקוּקוֹת

רְצוּפִים מִפַּז וּמִדְּבַשׁ מְתוּקוֹת

קָדוֹשׁ


דִּבְּרוֹת מְשַׁבֵּר סְלָעִים אָיוֹם

דִּגְלוֹ בְנַחַת הִשְׁמִיעַ אָיוֹם

דָּרוּר בִּקְדֻשּׁוֹת זָכוֹר אֶת יוֹם


הִמְתִּיק פִּרְיוֹ לְחִכִּי בְּזֶבֶד

הָרִים הִפְרִיק תֵּקַע כָּבֵד

הוֹרִים בִּיקָר וּמוֹרָא לְכַבֵּד


וְרִיחֲפֵי עַתִּיק יָמִים וְצַח

וְצִחְצַח בְּצִחְצוּחַ צְרוּפָה לְהִצְטַחְצֵחַ

וְהִזְהִיר יְצוּר לֹא תִּרְצָח


זוֹרוּ רְשָׁעִים שָׂטֵי כָזָב בְּשַׁאַף

זָרָה מֵחִבּוּק בְּשֶׁגֶג לִשְׁאַף

זָרִים גְּנוּבִים לֹא תִּנְאָף


חוֹמְסִים מְגַנְּבֵי דֵעוֹת מֵעֲנוֹב

חֲדוֹל מֵחִבּוּרָם וּבְחֶלְקָךְ תִּתְנוֹב

חֶמְדַּת עָמֵל לֹא תִּגְנוֹב


טִכְּסָךְ מְרַחֵף כְּנֶשֶׁר בְּיוֹקֶר

טוֹר מַעֲנֶה פֶּן תִּדָּקֵר

טַעַם לֹא תַעֲנֶה בְרֵעֲךָ עֵד שֶׁקֶר


יָשָׁר בְּגֹרָלָךְ לַקֵּץ תַּעֲמֹד

יְקָרָה מִפְּנִינִים אִם תִּתְמוֹד

יְפַת תּוֹאַר מִזּוּגָהּ לֹא תַחְמוֹד


כִּתְרֵי עֲנָנִים לַאֲגַנּוֹ אָסַר

כְּלוּלִים הֱיוֹת בְּשִׁנּוּן מוּסָר

כַּנּוֹגַהּ לְיַפּוֹת שְׁבָטִים שְׁנֵים עָשָׂר


לְעֵת רַד לְחֶמֶד מִמְּעוֹנוֹת

לִמְאֹד חֻלְחֲלוּ כָּל פִּנּוֹת

לַהֶבֶת כְּנִרְאַת אַרְבַּע דִּמְיוֹנוֹת


מַאְדֶּמֶת מַשְׁחֶרֶת הָחְצְבָה לֶהָבָה

מַלְבֶּנֶת מוֹרֶקֶת בָּהָר קְצוּבָה

מִפְעֲלוֹת נוֹרָאִי דָּגוּל מֵרְבָבָה


נָאוֹר בְּהִגָּלוֹתוֹ נֶגֶד מַקְהֵלוֹת

נְטוּיָה עַל תֹּהוּ הוּשָׂמָה מְחִלּוֹת

נִבְהֲלָה בְשָׁמְעָהּ שִׁבְעָה קוֹלוֹת


סוֹעֵר משוך הַמּוּכָן לְיוֹמַיִם

סְפוּנִים פָּעָנְחוּ מֵאֵימַת דָּר שָׁמַיִם

שְּׂאֵת קוֹל יְיָ עַל הַמָּיִם


עִזּוּז בֵּיאֵר עִנְיְנֵי וִכּוּחַ

עוֹז וְתַעֲצוּמוֹת בְּלִי שְׁכוֹחַ

עֹצֶם קוֹל יְיָ בַּכֹּחַ


פָּארָן וְשֵׂעִיר אָפְלוּ מֵהָדָר

פֶּלִיא בְּעַד עָנָן כְּנִתְאַדַּר

פְּעוֹת קוֹל יְיָ בֶּהָדָר


צֻבְּאוּ כְעֵדֶר גִּנַּת אֱגוֹזִים

צוֹלְעִים חִיל וָפַחַד אֲחוּזִים

צָרַח קוֹל יְיָ שׁוֹבֵר אֲרָזִים


קוֹלוֹת מִקּוֹלוֹת נִרְאוּ קְצוּבוֹת

קַלּוֹת וַחֲמוּרוֹת בְּגִלּוּי מַקְצִיבוֹת

קוֹל יְיָ חוֹצֵב לַהֲבוֹת


רַעֲנָנָה בִּבְרִית אֵיתָנֶיהָ הִסְבַּר

רְטֻבָּה בְּזִנּוּק חִטּוּבֶיהָ לְהִתְגַּבַּר

רַעַשׁ קוֹל יְיָ יָחִיל מִדְבָּר


שִׁכְּנָה כְּעֵדֶר קְוֻצֵּי תַּלְתַּלּוֹת

שֶׁבָּהּ יִתְפָּאֵר יוֹשֵׁב תְּהִלּוֹת

שָׁאַג קוֹל יְיָ יְחוֹלֵל אַיָּלוֹת


תִּרְגֵּל לְבוּבָתוֹ זָבוֹד

תֹּקֶף עֲנָנָיו אִפְּדָהּ בְּכָבוֹד

תָּמִיד בְּהֵיכָלוֹ כֻּלּוֹ אוֹמֵר כָבוֹד

קָדוֹשׁ
