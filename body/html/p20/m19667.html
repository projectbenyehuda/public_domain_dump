יֶשַׁע בַּשְּׂרוּ אֹם גּוֹלָה

פיוט לחנוכה


יֶשַׁע בַּשְּׂרוּ אֹם גּוֹלָה / כִּי בָא יוֹם גְּאוּלִים,

וִיהִי לָהּ לְשֵׁם וּתְהִלָּה / אֵל שׁוֹכֵן זְבוּלִים.


אַהְבַת אַב הֲמוֹנִים הֵעִיר / הַיּוֹם לַעֲדָתוֹ –

וַיִּשְׁפֹּךְ עֲלֵי אִישׁ שָׂעִיר / כָּאֵשׁ אֶת חֲמָתוֹ

וַיַּאְבִיד שְׂרִידָיו מֵעִיר / וַיַּשִּׁים נְוָתוֹ;

וַיַּמְצִיא צֳרִי וּתְעָלָה / לַעְנִיִּים אֲמֵלִים,

הַחוֹכִים פְּדוּת וּגְאֻלָּה / וּלְיֶשַׁע מְיַחְלִים.


בֹּאוּ לַהֲגוֹת רֹב טוֹבָה / עָשָׂה מִלְּפָנִים

וּתְנוּ לוֹ תְּהִלָּה עָרְבָה / וַחְמוּדֵי רְנָנִים,

כִּי הִשְׁמִיד בְּחֵמָה רַבָּה / מַלְכוּת הַיְּוָנִים –

יוֹם בָּאוּ לְהָשִׂים תָּהְלָה / בִּילָדִים אֲצִילִים

וּלְתִתָּן מְעִי מַפָּלָה / מִקְדָּשׁ עִם גְּבוּלִים.


רָם לֵב צַר בְּבוֹאוֹ לִפְנִים / וַיִּגְדַּל גְּאוֹנוֹ,

וּלְכַלּוֹת בְּנֵי אֵיתָנִים / הִשִּׂיאוֹ זְדוֹנוֹ,

אוּלָם שָׁב בְּבֹשֶׁת פָּנִים / וַיִּכְלֶה הֲמוֹנוֹ,

מָתְנָם אָחֲזָה חַלְחָלָה – / כָּל עוֹבְדֵי אֱלִילִים –

וּלְדָךְ נִתְּנָה הַצָּלָה / וּגְבוּרָה לְכוֹשְׁלִים.


הַכֹּהֲנִים בְּבוֹאָם רָאוּ / פַּךְ אַחַת חֲתוּמָה,

שֶׁמֶן בָּהּ לְהַדְלִיק מָצְאוּ / יוֹם אֶחָד, וְהֵמָּה

הִדְלִיקוּ שְׁמוֹנָה מָלְאוּ / בִּבְרָכָה עֲצוּמָה.

אָז הוֹדוּ לְשׁוֹכֵן מַעְלָה, / כִּי מַעְשָׂיו גְּדוֹלִים,

כִּי רִפְאוּת לְמַחְלָה הֶעְלָה / וּצְרִי לַעֲמֵלִים.


מַשְׂכִּילִים שְׁמוֹנַת יָמִים / הָאֵלְֶּה לְדוֹרִים

קָבְעוּ, לִהְיוֹתָם שׂוּמִים / יוֹם שִׂמְחָה וְשִׁירִים,

גַּם נֵרוֹת יְהוּא מוּקָמִים / עַל פִּתְחֵי שְׁעָרִים:

זִכָּרוֹן לְנֵס הַנִּפְלָא; / לָכֵן עַל מְחוֹלִים

שִׁירוּ, וַעֲשׂוּ יוֹם צָהְלָה / וּתְנוּ מַהֲלָלִים.


׳בַּת צִיּוֹן, בְּכָל לֵב עִלְזִי / וּשְׁתִי כוֹס יְשׁוּעוֹת!

הִנֵּה לָךְ אֲגַלֶּה רָזִי / וַאְקַיֵּם שְׁבוּעוֹת.

נִשְׁבַּעְתִּי עֲלֵי יַד חוֹזִי / לֶאְסֹף מִתְּנוּעוֹת

הָעָם, שֶׁלְּקַחְתִּיו נַחְלָה / לִי, וּבְכָל שְׁבִילִים

אֶקְרָא: סַקְּלוּ הַמְסִלָּה / חִישׁ לַעְבֹר גְּאוּלִים!׳
