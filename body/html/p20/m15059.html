אַשְׁרֵי הָאִישׁ יָרֵא

אַשְׁרֵי הָאִישׁ יָרֵא / מִקּוֹל הַקּוֹרֵא, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

בַּמֵֶּה אִישׁ חָשׁוּב? / יֵלֵךְ וְלֹא יָשׁוּב, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.

רוּחוֹ נוֹסְסָה בוֹ – / וַיִּגְבַּהּ לִבּוֹ, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

הוֹלֵךְ לַקֶּבֶר – / וּפִתְאֹם יִשָּׁבֵר, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.

מַהֲלָכָיו גְּבֹהִים / כְּמַלְאַךְ הָאֱלֹהִים, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

בְּמוֹתוֹ – כַּבְּהֵמָה / לַהֶבֶל דָּמָה, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.

רוֹחֵץ אֶת פָּנָיו / טָהוֹר בְּעֵינָיו, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

מִי יִגַּשׁ אֵלָיו – / וְטֻמְאָתוֹ עָלָיו, / בְּשׁוּבוֹ אֶל הָאֲדָמָה?

אֶתְמוֹל קוֹל הֵרִים / בְּשִׂמְחָה וּבְשִׁירִים, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

יוֹדְעוֹ בוֹ יֶהְגֶּה / קִינִים וָהֶגֶה, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.

רוֹדֵף טוּב רֵיחַ / וְשֶׁמֶן רוֹקֵחַ, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

בְּלוֹיָיו נִקְרָעִים / וְיָרֻם תּוֹלָעִים, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.

נִמְשְׁכָה מַכָּתוֹ – / יַרְבֶּה כְּבוֹד בֵּיתוֹ, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

עָבְרוּ יְמֵי מְהוּמָה – / וְאֵין בְּיָדוֹ מְאוּמָה, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.

זְרוֹעַ תִּפְאֶרֶת / וְלָשׁוֹן מְדַבֶּרֶת, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

רַד אֶל בּוֹר עָמַק – / וּלְשׁוֹנוֹ תִּמַּק, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.

הוֹד וְהָדָר לְפָנָיו, / יִרְאֶה בְעֵינָיו, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

נָפַל לְמַשּׁוּאוֹת / וְחָשְׁכוּ הָרוֹאוֹת, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.

חָמַד סוֹד רֵעִים / וּמַעֲשֵׂה תַעְתּוּעִים, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

מַה יְּבַקֵּשׁ בְּתֵבֵל? / כִּי הַכֹּל הֶבֶל, / בְּשׁוּבוֹ אֶל אֲדָמָה.

תְּסוֹבְבוֹ מִלָּתְךָ / וְיַגִּיד תְּהִלָּתְךָ, / בְּעוֹד יֵשׁ בּוֹ נְשָׁמָה;

וְעַל עַצְמוֹ תִגְמֹר – / אַךְ אֶת נַפְשׁוֹ שְׁמֹר, / בְּשׁוּבוֹ אֶל הָאֲדָמָה.
