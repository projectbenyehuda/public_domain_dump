בְּשֵׁם אֵל אֲשֶׁר אָמַר

בְּשֵׁם אֵל אֲשֶׁר אָמַר / וְהָעֹז לְאִמְרוֹתָיו,

וְצִוָּה בְּלִי מוֹרֶה / וּמוֹרֶה לְמִצְוֹתָיו,

וּפָעַל וְעָשָׂה כֹל / אֲשֶׁר אִוְּתָה נַפְשׁוֹ,

וְהֵחֵל בְּלִי עֵזֶר / וְכִלָּה פְעֻלּוֹתָיו.


אֲהַלְלָה לְמֶמְשַׁלְתּוֹ / וְאוֹדֶה לְמַהֲלָלוֹ,

וְאֶקֹד לְכֵס גָּדְלוֹ / לְזִכְרוֹ וְלִשְׁמוֹתָיו;

וּמִי זֶה אֲשֶׁר יִצְרוֹ / מְאַחֵז יְקַר צוּרוֹ,

וּמִי זֶה אֲשֶׁר יַעֲמֹד / בְּסוֹדוֹ וְסִתְרוֹתָיו;

וְאֵיךְ מַה יְקַו לִסְפּוֹר / וְסֹפֵר גְּאוֹן הֶחָי,

וְחֶסְרוֹן הֲיִמָּנֶה / לְהַוֹּת יְתֵרוֹתָיו,

לְרִאשׁוֹן בְּלִי רֵאשִׁית / וְאַחֲרוֹן בְּלִי אַחֲרִית,

וְאֵיךְ לוֹ יְהִי תַכְלִית / וְאָפְסוּ תְהִלּוֹתָיו;

וְאֶחָד, וְאַךְ לֹא הוּא / כְּאֶחָד אֲשֶׁר נָתַן

יְסוֹד הָאֱנוֹשׁ לַחְשֹׁב / וְלִכְלוֹל סְפוֹרוֹתָיו;

וְאֶחָד, וְלֹא יִרְבֶּה / וְיִמְעַט וְיֵחָלֵק

כְּאֶחָד אֲשֶׁר הוּא רֹאשׁ / אֲלָפָיו וְרִבְבוֹתָיו;

אֲבָל הוּא אֲשֶׁר נוֹדַע / בְּרֵאשִׁית עֲלִילוֹתָיו,

בְּכֹחוֹ וְחָכְמָתוֹ / וְעֹז תַּעֲצוּמוֹתָיו;

וְרָחוֹק וְהוּא קָרוֹב / וְקָרוֹב וְהוּא רָחוֹק,

וְנִמְצָא בְכָל מָקוֹם / וְשָׂגְבוּ מְקוֹמוֹתָיו;

וְנִסְתָּר וְהוּא רֹאֶה / וְנֶעְלָם וְיֵרָאֶה

בְּעֵינַי גְּבוּרוֹתָיו / וְעַל יַד פְּלִיאוֹתָיו;

וְאֵיךְ לֹא אֱהִי צוֹפֶה / כְבוֹדוֹ בְּעֵין לִבִּי,

וְדֵעָה אֲשֶׁר שָׂם בִּי / וְנֶפֶשׁ מְעִידוֹתָיו;

וְנַעֲלָה בְּלִי מָעוֹן / וְנִשָּׂא בְּלִי מָכוֹן,

אֲבָל הוּא מְקוֹם חֶלְדּוֹ / וְסוֹבֵל יְצִירוֹתָיו;

וּבִבְלִי כְרוּב יָעוּף / וְעָבִים לְסֵתֶר לוֹ,

יְהַלֵּךְ עֲלֵי כַנְפֵי / עֲנָנָיו וְרוּחוֹתָיו;

וְעָשָׂה מְשָׁרְתָיו אֵשׁ / וְלַהַט, וּמַלְאָכָיו

וְרוּחוֹת, וְאַף קָרָה / בְמַיִם עֲלִיוֹתָיו;

וְלַיְלָה כְּיוֹם יָאִיר / לְפָנָיו, וְאִם נָתַן

עֲרָפֶל חֲתֻלָּתוֹ / וְחֹשֶׁךְ סְבִיבוֹתָיו;

וּמֵבִיא בְּלֵב נוֹצָר / כְּבוֹדוֹ וְאִם נִבְצָר

וְאֶרֶץ וְשָׁמַיִם / מְלֵאִים תְּהִלּוֹתָיו;

וְהוּא כֹל וּבוֹ הַכֹּל / וְנִמְצָא בְּקֶרֶב כֹּל,

וְאִם גָּבְהוּ מִכֹּל / דְּרָכָיו וּמִדּוֹתָיו;

וְכִרְחוֹק פְּאַת מִזְרָח / לְיַד מַעֲרָב, רָחֲקוּ

מְאֹד מַחְשְׁבוֹת אִישׁ / מֵעֱזוּז עֶשְׁתּוֹנוֹתָיו;

הֲכִי נִמְצָא בִזְמַן / וּמָקוֹם, וְהוּא מֵאֵין

זְמַנִּים וּמִשְׁכְּנוֹת / מְצִיאוֹת מְזִמּוֹתָיו;

וְעָלָה בְמַחֲשָׁבָה / תְחִלָּה בְּרֹאשׁ חֹמֶר

וְצוּרָה וְהַחֵפֶץ / וְאֵלָיו תְּשׁוּקוֹתָיו;

וְהוּא רֹאשׁ לְכָל חֹמֶר / וְשֹׁרֶשׁ לְכָל צוּרָה,

וְאֵין לַחֲמָרָיו קֵץ / וּמִסְפָּר לְצוּרוֹתָיו;

וְזָמַם עֲשׂוֹת הַיֵּשׁ / בְּלִימָה וְנִתְכָּנוּ

עֲלִילוֹת לְמַחֲשַׁבְתּוֹ / וְסִבּוֹת לַעֲצוֹתָיו,

וְנָטָה כְדֹק שַׁחַק / לְסַפֵּר בְּלִי אֹמֶר

כְּבוֹדוֹ וּמַלְכוּתוֹ / וְחַוֹּת עֲלִילוֹתָיו;

מְרֻצָּע כְּמוֹ עֶבֶד / לְהַשִּׂיג רְצוֹן קוֹנוֹ,

וְאֵלָיו לְהִתְרַצּוֹת / בְּפַחְדּוֹ וְאֵימוֹתָיו;

מְתָחוֹ כְּמוֹ אֹהֶל / אֲבָל קָם בְּרוּחַ פִּיו,

וְהֵבִיא בְלִי יָד אֶת / קְרָסָיו בְּלֻלְאֹתָיו;

וְעָשׁ עִם כְּסִיל לְפִנּוֹת / בְּצָפוֹן וּבַדָּרוֹם

קְצוֹתָיו, וְעַל גַּבָּם / הֲלִיכָיו וְעִקְבוֹתָיו.

וְשִׁבְעָה רְקִיעִים / הוּבְאוּ זֶה בְזֶה לִהְיוֹת

מְגוּרוֹת לְכוֹכָבִים / וְלִשְׁנֵי מְאוֹרוֹתָיו;

בְּלֵיל שַׁבְּתַאי גַּם חוֹל / לְצֶדֶק וְחוּג מַאֲדִים

וְשֶׁמֶשׁ, וְאֶת אָפְלָם / יְכַסֶּה בְּאוֹרוֹתָיו;

וְנֹגַהּ וְחוּג כּוֹכָב / וְסַהַר אֲשֶׁר יַזְהִיר

בְּרָחְקוֹ, וְאִם יִקְרַב / יְלֻבַּשׁ בְּאַפְלוֹתָיו;

גְּלִילִים וְכוֹכְבֵיהֶם / יְרוּצוּן, וּבַשְּׁמִינִי,

חֲקֻקִים בְּאֵין מֵרוּץ / גְּדוּדֵי חֲיָלוֹתָיו;

וְעֹלִים בְּמַתְכֻּנְתָּם / וְרָצִים בְּמִשְׁמַרְתָּם

עֲלֵי קַו וְהָעִגּוּל / נְכוֹנוֹת מְסִבּוֹתָיו;

וְהֵמָּה בְמַסְמֵרִים / נְטוּעִים כְּנֹגַהּ,

יְרוּצוּן לְאֹרֶךְ עַל / שְׁבִילֵי תְהִלּוֹתָיו;

וְהוּא בִגְבוּרָתוֹ / מְסוֹבֵב בְּמַזָּלוֹת,

לְכַלּוֹת חֲדָשׁוֹתָיו / וְחִדֵּשׁ יְשָׁנוֹתָיו;

וְהוּא אֵל גְּדָל כֹּחַ / אֲשֶׁר הוּא לְכֹל סוֹבֵל

וּבוֹ נֶחְשְׁלוּ זִמּוֹת / אֲחִי בִין וְחָכְמוֹתָיו;

וְהוּא אֵל אֲשֶׁר יִסֹּב / גְּלִילָיו בְּיוֹם פַּעַם

לְמַעֲרָב וְלַמִּזְרָח / וְעָלָיו תְּשׁוּבוֹתָיו;

וְיַחַד שְׂרָפָיו הֵם / יְרוּצוּן וְיַקְדִּישׁוּ,

וְאִם לֹא יְבִינוּן אֶת / מְקוֹמוֹ וּמוֹשְׁבוֹתָיו;

יְצָרָם כְּסַפִּירִים / מְאִירִים, וְלֹא זַכּוּ

בְעֵינֵי שְׁכִינָתוֹ / וְזֹהַר מְגוּרוֹתָיו;

וְאָז נִתְכְּנוּ אִישִׁים / וְצוּרוֹת בְּעוֹלָמוֹ,

בְּעֹצֶם מְרוּצָתוֹ / וְכֹחַ תְּנוּעוֹתָיו.

וְשֶׁמֶשׁ בְּמַזָּלוֹת / וְהֵמָּה שְׁנֵי עָשָׂר

יְסוֹבֵב יְמֵי שָׁנָה / לְכוֹנֵן תְּקוּפוֹתָיו;

וְהֵם קֹר וְחֹם, וּזְמַן / לְקַיִץ וְחֹרֶף לֹא

יְסוּפוּן וְיִשְׁתַּנֶּה / זְמַן בַּחֲלִיפוֹתָיו;

וְשֵׁשׁ יוֹרְדִים מֵהֶם / וְשֵׁשׁ יַעֲלֶה בָהֶם,

יְמֵי עַד לְקוֹמָם, חוּץ יְמוֹתָיו וְלֵילוֹתָיו;

שְׁלֹשִׁים בְּכָל מַזָּל / לְהַשְׁלִים שְׁלֹשׁ מֵאוֹת

וְשִׁשִּׁים, יְהַלֵּךְ עַל / חֲלָקָיו וּמַעֲלוֹתָיו;

וְיִשְׁאַף וְיִזְרַח אֶל / מְקוֹמוֹ בְּרֹאשׁ טָלֶה,

וְיִסֹּב עֲלֵי בָתָיו / כְּמֶלֶךְ בְּטִירוֹתָיו;

וְיָבֹא בְצֵל אֶרֶץ / וְאָז יֶחְשַׁךְ אוֹרוֹ

בְּצִלּוֹ, לְמַעַן אֵין / חֲשֵׁכִים תְּמוּרוֹתָיו;

וּבֵינָם כְּחָתָן / מֵחֶדְרוֹ יְהִי יוֹצֵא

לְהָאִיר עֲלֵי טַבּוּר / וְיָרוּץ בְּאֹרְחוֹתָיו;

וְסַהַר בְכָל מַזָּל / בְּיוֹמוֹ יְהִי חֹנֶה,

לְהַשְׁלִים יְמֵי חֹדֶשׁ / בְּחָדְשׁוֹ וְלִשְׁנוֹתָיו;

וְיִתְכַּס וְיִתְגַּלֶּה / וְיִשַּׁח וְיִתְעַלֶּה,

וְיֶחְסַר וְיִמָּלֵא / לְהָאִיר כְּחֻקּוֹתָיו;

וְיַגְבִּיר בְּהִמָּלְאוֹ / גְאוֹן יָם וְשִׁפְעוֹתָיו,

וְגֶרֶשׁ יְרָחִים גַם / לְהַבְשִׁיל תְּנוּבוֹתָיו;

וְכִימָה תְּקַשֵּׁר / מַעֲדָנוֹת וְתַקְשִׁיחַ,

וְנֶגְדָּהּ יְפַתֵּחַ / כְּסִילָה וּמוֹשְׁכוֹתָיו;

כְּסִילִים לְמַזָּלוֹת / נְשָׁמוֹת, וְהֵם גּוּפוֹת,

לְהַרְאוֹת זְרוֹעַ אֵל / בְּעוֹלָם וְאוֹתוֹתָיו;

לְהַחֲיוֹת לְהָמִית גַּם / לְהַשְׁפִּיל וְגַם רוֹמֵם,

לְרוֹשֵׁשׁ וְלַעֲשִׁיר כֹּל / יְמוֹתָיו מְרוּצוֹתָיו;

וְלִבְנוֹת וְגַם לַהֲרוֹס / וְלַעֲקֹר וְלִנְטוֹעַ,

וּמַגִּיד לְכָל מֶלֶךְ / יְקָרוֹ, וְכִסְאוֹתָיו;

וְעֹז הַתְּלִי לַעֲזוֹר / וְלַכְשִׁיל דְּמוּת תַּנִּין

זְנָבוֹ תְּבוּסָתוֹ, / וְהָעֵז מְרַאֲשֹׁתָיו;

וְעָשָׂה רְקִיעַ / מַבְדִּיל הוּא לְבֵין מַיִם,

וְלִהְיוֹת מְקוֹם קָרוֹב / לְהַשְׁקוֹת לְאַדְמוֹתָיו;

וְרֵאשִׁית לְכָל נִבְרָא /יְסוֹדוֹת אֲשֶׁר בָּרָא,

וּמַה נִּפְלָאָה חָכְמָה / בְּאַרְבַּע יְסוֹדוֹתָיו;

וְהֵם אֵשׁ וְהָרוּחַ / וּמַיִם וְהֶעָפָר,

בְּהִתְחַבְּרָם עוֹלָם / אֲזַי בָּם וְתֹלְדוֹתָיו;

וְיָסַד בְּכָל אֶחָד / שְׁתַיִם בְּמִפְעָלָיו

לְיוֹבְשׁוֹ וְאֶת חֻמּוֹ / וְקֹרוֹ וְלֵחוֹתָיו;

הֲכִי הוּא אֲשֶׁר יַצְמִיד / מְשַׂנְּאִים וְהוּא יַפְרִיד

אֲהוּבִים, וְאֵין בִּלְתּוֹ / מְחַבֵּר פְּזוּרוֹתָיו;

וְכֹבֶד וְנָקֵל בָּם / וּפֹעֵל וְנִפְעָל בָּם,

וְעֹלֶה וְיוֹרֵד בָּם / רְצוֹן אֵל וְדָתוֹתָיו.

וְהֵם נִקְרְאוּ אִמּוֹת / הֲכִי יָלְדוּ בָנוֹת,

אֲשֶׁר חָק אֱלֹהַּ בָּם / הֲמוֹן הַיְקוּם וּמְתָיו

וְאֶרֶץ תְּלוּיָה עַל / בְּלִימָה בְּאִמְרָתוֹ,

קְשׁוּרָה לְבַל תִּזַּח / בְּחַבְלֵי אֱמוּנוֹתָיו

כְּכַדּוּר, וְהָרוּחוֹת / מְפִיחוֹת וְהֵן אַרְבַּע

וְנוֹשְׁבוֹת בְּכָל רֶגַע / בְּעוֹלָם לְרוּחוֹתָיו

בְּצִדָּם, וְאַרְבָּעָה / וְעֶשְׂרִים אֲלָפִים מִיל

אֱמֶת מַהֲלַךְ עוֹלָם / אֲפָסָיו וּמִדּוֹתָיו;

וְנִגְזַר עֲלֵי כִפּוֹת / סְפִירוֹת וְהֵם תִּשְׁעָה,

וּבָהֶם שְׁבַע עֶשְׂרֵה / אֲלָפִים מְדִינוֹתָיו.

וּבָהֶם חַי אֲשֶׁר יִצְעַד / וְיִשְׂחֶה וְיִתְעוֹפֵף,

וְחִפָּה בְּבֵין אֶרֶץ / וְתוֹךְ מַהֲמֹרוֹתָיו;

בְּזָהָב וְגַם כֶּסֶף / וּנְחֹשֶׁת וְעֹפֶרֶת,

וּבְדִיל וּבַבַּרְזֶל / וּבִשְׁאָר יְצוּקוֹתָיו;

וְעוֹד הֻתְּכוּ בִשְׁאוֹל / בְּאֶרֶץ הֲמוֹן מִינֵי

פְּנִינִים וְכָל אֶבֶן / יְצוּקָה מְשׂוּכּוֹתָיו.

וְזֶרַע אֲשְֶׁר נִזְרַע / וְכָל עֵץ אֲשֶׁר נִטַּע

וְכָל דִּשְׁנֵי אֶרֶץ / וְשִׂיחָיו וְעִשְׂבוֹתָיו;

וְלָבָן בְּעֵינֵיהֶם / וְשָׁחוֹר בְּצִיצֵיהֶם,

וְיָרוֹק בְּמַרְאֵיהֶם / וְאָדֹם בְּצִמְחוֹתָיו;

וּמָתוֹק מַר מֵהֶם / וְחָמוּץ בַּעֲלֵיהֶם;

וְתָפֵל וּמָלוּחַ / בְּטַעַם לְאוֹרוֹתָיו,

וּמֵהֶם כְּמוֹ בֹשֶׂם / וּבָם עוֹבְרִים רוֹשָׁם

וּמַבְאִישׁ פְּנֵי רִבּוֹא / חֲלָלָיו וְרִבּוֹתָיו;

וְחָצִיר וְעֵשֶׂב / לַבְּהֵמָה וְלַחַיָּה,

וּמֵהֶם לְמוֹתֵת חַי / וּבָהֶם רְפוּאוֹתָיו;

וְהִשְׁוָה אֲדָמָה / לַעֲבֹדָה וְצִוָּה עַל

עֲרָבָה בְּעוֹלָם / וַעֲמָקָיו וְהָרוֹתָיו;

וְעָפָר בְּשָׁלִישׁ כָּל / וּמַיִם בְּיָד מָדַד,

וְרוֹמֵשׂ בְּגוּשׁ אֶרֶץ / וְשֹׁרֶשׁ בְּטִירוֹתָיו;

וְאָמַר לְיָם עַד פֹּה / תְהִי בוֹ וְלֹא תוֹסִיף,

וְהִתְגַּעֲשׁוּ מֵימָיו / וְהַחוּץ גְּבֻלוֹתָיו;

וְהֶעֱלָה נְשִׂיאִים / מִקְּצוֹת הָאֲרָצוֹת גַּם

בְּרָקִים לְמָטָר שָׂם / וְרוּחַ בְּאוֹצְרֹתָיו;

וּבָקַע עֲיָנוֹת גַּם / תְהֹמוֹת לְהַשְׁקוֹת כֹּל,

וְחַיְתוֹ שְׂדֵי יַעַר / בְּעָרָיו וְחֻצּוֹתָיו;

וְהַכֹּל בְּשֵׁם עוֹלָם / קְרָאוֹ, אֲבָל גָּדוֹל

לְאָרְכּוֹ וְגַם רָחְבּוֹ / גְּבֹהָיו וְתַחְתִיתָיו;

וְזֹאת יָצְאָה מֵאֵל / נְדָבָה בְּלִי צֹרֶךְ,

אֲבָל כִּי לְהוֹדִיעַ / לְאָדָם גְּבוּרוֹתָיו;

וְלִחְיוֹת לְאוֹת לָעַד / הֲכִי מַעֲשָׂיו הֵמָּה,

וְהוּא בִלְבַד קַדְמוֹן / וְכֻלָּם חַדְשׁוֹתָיו;

וְיוֹם בְּרֹאת אָדָם / בְּצַלְמוֹ וּבִדְמוּתוֹ

בְּמַעֲשָׂיו לְהַמְשִׁילוֹ / וּמִינֵי בְרִיאוֹתָיו,

וְעָשָׂה כְנֶגְדּוֹ / מִצְּלָעָיו לְעֵזֶר לוֹ,

לְהַשְׁאִיר כְּתַבְנִיתוֹ / וְדִמְיוֹן הֲוַיּוֹתָיו,

כְּחָלָב לְהַתִּיכוֹ / בְּשִׁלְיָא לְהַקְפִּיאוֹ,

וּבָשָׂר לְהַלְבִּישׁוֹ / בְּגִידָיו וְעַצְמוֹתָיו;

וְאָחוֹר וְקֶדֶם צָר / וְתַחַת וְרָם יָצַר,

לְהֵימִין לְהַשְׂמִיל חָק / וְאֵלֶּה תְּכוּנוֹתָיו;

וְגוּפוֹ בְּעֵת עָצֹם / וְחָזָק יְתֵרוֹתָיו;

וְיִקְרַם עֲלֵיהֶם עוֹר / וּבָשָׂר בְּשַׂלְמוֹתָיו;

בְּבֶטֶן יְצָרָהוּ / וְדַם אֵם לְהֵינִיקוֹ,

וְטוּבוֹ עֲנָקוֹ / בַּאֲפֵלוֹ וְחַנּוֹתָיו;

וְאֶל קֵץ זְמַן יֵצֵא / לְעוֹלָם לְהַבְטִיחוֹ,

לְאַט עַל שְׁדֵי אִמּוֹ / וְהַחֹק לְאִמּוֹתָיו;

וּבָאֵר בְּהַרְגָּשׁוֹת / חֲמִשָּׁה תְּבוּנוֹתָיו,

אֲשֶׁר הֵן עֲדִי גוּפוֹ / וְחֶלְיַת חֲמֻדוֹתָיו;

יְצָרוֹ יְפֵה מַרְאֶה / וְתֹאַר וְהֵכִין אֶת

צְלָעָיו וּמַפְרַקְתּוֹ / וְעָרַךְ שְׂדֵרוֹתָיו;

וְעֵינָיו לְהַשְׁגִּיחַ / וּפִיהוּ לְהַצְרִיחַ,

וְאַפָּיו לְהָרִיחַ / בְּנַחֲרַת נְטִישׁוֹתָיו;

וְלָשׁוּךְ אֲשֶׁר הוּטַל / וְהוּעַד, וְכָעַיִט

יְצַפְצֵף לְבָאֵר / מֶחְקְרֵי לֵב וְחִידוֹתָיו;

וְיָלִיץ בְּעַד סוֹדוֹ / לְשׁוֹאֵל, וְרַעְיוֹנָיו

יְבָאֵר וְיוֹדִיעַ / סְעִפֵּי עֲלוּמוֹתָיו;

וְיָדָיו לְהָנִיעַ / וְרַגְלָיו לְהַסִּיעַ,

וְאֹזֶן לְהַגִּיעַ / לְלִבּוֹ שְׁמוּעוֹתָיו;

בְּגוּפוֹ לַהֲמִישׁוֹ / בְּכַפָּיו לְאָחֲזֵהוּ,

בְּשׁוֹקָיו לְהַצִּיעוֹ / וְיֹשֶׁר הֲלִיכוֹתָיו;

וְיָצַר בְּרֹאשׁוֹ / מַזְכֶּרֶת וּמִשְׁמֶרֶת,

וְנָתַן לְשֶׂכְוִי בִּין / וְחָכְמָה בְּטֻחוֹתָיו;

בְּקֶרֶב אֱנוֹשׁ הוּשַׂם / כְּמֶלֶךְ עֲלֵי כִסְאוֹ,

וְכָל הָאֲבָרִים / לַעֲבָדָיו וְאַמְהֹתָיו;

וְכִיס מַר עֲלֵי יָמִין / וּבוֹ רֹאשׁ לְבַל יָצוּק,

וּבִשְׂמֹאל לְחֹם טָחוּל / בְּאוֹרוֹ וּמוֹקְדֹתָיו

לְקָרֵר, לְהַרְבּוֹת אֶת / יְמוֹתָיו בְּמִפְעָלָיו

וְיֶאֱמַץ בְּהִתְיָעֵץ / בְּמָתְנָיו וְכִלְיוֹתָיו;

וְקֵבָה לְהַחֲיוֹתוֹ / וְכָבֵד לְהַפְרוֹתוֹ,

וְקָדְְקֹד לְהוֹרוֹתוֹ / סְעִפֵּי תְבוּנוֹתָיו;

לְדָם יַהֲפֹךְ כֹּבֶד / אֲכָלָיו וְשֶׁל פִּגְרוֹ

יְשַׁלַּח אֲרֻחוֹתָיו / בְּכָל עֵת וּמַשְׂאוֹתָיו;

וְעַל לֵב, כְּנַף רֵיאָה / יְחוֹפֵף לְרוּחוֹתָיו,

לְהוֹצִיא עֲשַׁן אַפּוֹ / וְהֵבִיא לְקוֹרוֹתָיו;

וְרוּחַ עֲלֵי גָרוֹן / שְׁבִילוֹ, לְתִתּוֹ קוֹל

וְחִתַּךְ בְּחֵךְ וּבְשֵׁן / וְלָשׁוֹן בְּשִׂפְתוֹתָיו;

נְתִיבוֹת שְׁתַיִם גַּם / שְׁלֹשִׁים פְּלָאוֹתָיו

בְּחָכְמָה בְּיָהּ חָקַק / לְמֵבִין עֲלוּמוֹתָיו;

לְסֵפֶר וּלְסַפֵּר / וְלִסְפּוֹר, וְאוֹתִיּוֹת

שְׁתַיִם וְעֶשְׂרִים הֵם / וְעֶשֶׂר סְפִירוֹתָיו;

וְהִשְׁלִים מְלֶאכֶת כָּל / נְקָבָיו וְגַם תֻּפָּיו

מְסִלּוֹת לְמֵי רַגְלָיו / וְדֶרֶךְ לְחֶלְאוֹתָיו;

וּמֵהֶם נְתִיבוֹת חָק / לְכָל תּוֹצְאוֹת הַגּוּף

בְּרִירוֹ וְדִמְעוֹתָיו / וְרֻקּוֹ וְנִיעוֹתָיו;

מְעַט מֵאֱלֹהִים / חִסְּרָהוּ, הֲכִי הֵם לְעַד

בְּחַיִּים, וְהוּא עָפָר / וְיָשׁוּב לְעַפְרוֹתָיו;

וְכֵן עִטְּרָהוּ הוֹד / בְּמַדַּע אֲשֶׁר נִמְשַׁךְ,

אֲבָל מִמְּקוֹר צוּרוֹ / וּמֵעֵין יְשׁוּעוֹתָיו,

מְדַבֵּר בְּאֵין לָשׁוֹן / וּמַבִּיט בְּלִי אִישׁוֹן,

וְחוֹשֵׁב בְּאֵין רַעְיוֹן / וְצוֹפֶה עֲתִידוֹתָיו;

מְהַלֵּךְ בְּאֵין רֶגֶל / מְרוֹמוֹת וְתַחְתִּיּוֹת,

וְצוֹפֶה בְלִי עַיִן / לְעוֹלָם וְכַנְפוֹתָיו;

וְנֶפֶשׁ טְהוֹרָה / נִגְזְרָה מִכְּבוֹדֵהוּ,

וְתֶחְסַר מְעַט מִבֹּא / לְקַצְוֵי תְהִלּוֹתָיו;

וְתִשְׁאַף לְהִתְיַמֵּר / בְּמַדַּע אֲשֶׁר קָדַם,

וְכָבוֹד אֲשֶׁר נִגְזַר / לְכַָל אָדָם בְּדֹרֹתָיו;

הֲכִי הִיא בְּגוּף נָבָל / אֲסוּרָה בְּלִי כָבָל,

וְחַיָּה וְלֹא תִבֹּל / בְּשַׁחַק וְצִבְאוֹתָיו;

וְזֶה הוּא בְּשֵׁם עוֹלָם / מְכֻנֶּה אֲבָל קִנְיָן

וּמִי הוּא אֲשֶׁר בָּא עַד / תְּכוּנַת יְצִירוֹתָיו;

וְכִלָּה אֱלֹהִים כָּל / מְלַאכְתּוֹ וְאָז רָאָה

הֲכִי טוֹב מְאֹד הַכֹּל / וְיָפֶה בְּעִתּוֹתָיו;

וְאָדָם אֲשֶׁר יֻלַּד / לְעָמָל, כְּפִי יְכָלְתוֹ

יְחַוֶּה גְּאוֹן צוּרוֹ / וְיַטִּיף גְּדֻלּוֹתָיו;

וְנֶאֱחָז בְּפַח תֵּבֵל / וְנִשְׁקָע בְּרֹב הֶבֶל,

וְאָכֵן פְּלִיאוֹתָיו / לְהַלֵּל מְעִידוֹתָיו

וְאִם בֵּין צְלָעָיו צַר / לְנַפְשׁוֹ מְאֹד יֵצַר,

יְמִינוֹ בְּטוֹב תִּקְצַר / וְתֶאֱרַךְ בְּרָעוֹתָיו;

וְיַפְרִשׁ כְּצִפְעוֹנִי / וְיִקְשֹׁר וְלֹא יָנִיא,

וְיִשֹּׁךְ וְאִם יִשַּׁק / פְּצָעִים נְשִׁיקוֹתָיו;

בְּתֹאֲנָה יְסִיתֵהוּ, / וְלִבְכּוֹת, יִחַדֵּהוּ

בְּעוֹלָם, יְפַתֵּהוּ / בְּחֶמְדַת שְׂכִיּוֹתָיו;

וְיַשִּׁיא אֱמוּנָתוֹ / וְיִיגַע לְפַתּוֹתוֹ,

בְּהוֹנוֹ וְגַם בִּשְׁלַל / צִבְעוֹ וְרִקְמוֹתָיו;

אֱמֶת גָּדְלוּ מַעֲשֵׂי / אֲדֹנָי וְקִנְיָנָיו,

אֲשֶׁר נַעֲשׂוּ כֻלָּם / בְּצִדְקוֹ וְחָכְמוֹתָיו;

וְאֵלֶּה מְעַט מִזְעָר / גְּאוֹנוֹ וּמַלְכוּתוֹ,

וְאֵלֶּה קְצוֹת דַּרְכּוֹ / וּפִלְאֵי מְסִלּוֹתָיו;

אֱנוֹשׁ מֵת אֱלֹהַי חַי / יְזַמֵּר בְּעוֹד נֻפַּח

בְּאַפָּיו, וּמֶה אָדָם / וּמָה הֵם זְמִרוֹתָיו;

וְיָעִיר לְבַב שִׂכְלוֹ / לְהַמְעִיט קְצַת חֶבְלוֹ,

וְיַמְעִיט יְמֵי הֶבְלוֹ / כְּלִילָיו תְּמוּנוֹתָיו;

וְיוֹדוּ לְשֵׁם, אִמְרוֹ / בְּפִיהוּ לְהַגְבִּירוֹ,

לְהַשְׂכִּיל לְשׁוֹן קֹדֶשׁ / וְחֶזְיוֹן נְבוּאוֹתָיו;

וּמְדַבֵּר יְהוּדִית / הַמְדֻבָּר עֲלֵי אָפְנָיו,

וְעוֹד מַאֲמַר מֵעֲרַב / וּמַעֲנָיו וְצַחוֹתָיו,

וְצַעֲדוֹ בְּמוֹעֵצָה / בְּחָכְמָה וּבַמְּלִיצָה,

וּמוּסַר לְשׁוֹן יָוָן / מְשָׁלָיו וְחִידוֹתָיו;

עֲדֵי יָבְשׁוּ מֵימֵי / נְעוּרָיו, וְהִזְקִינוּ

קְצִירוֹ עֲצֵי גִילוֹ / וְשֵׁרְשׁוּ שְׁלֻחוֹתָיו;

וְעָבַר זְמַן יַלְדוּת / כְּבָרָק וְהַשַּׁחֲרוּת,

וְנִדְמוּ יְמֵי עֶדְנָה / לְהֶבֶל חֲלֹמֹתָיו;

וּמָשׁוּ לְפִי שָׁעָה / וְחָשׁוּ יְמֵי רָעָה,

לְהַפְרוֹת מְצוּקוֹתָיו / לְהַסְפּוֹת אַנְחוֹתָיו;

וְהָפַךְ זְמַן הוֹדוֹ / לְמַשְׁחִית, וְעֹז פָּנָיו

יְשֻׁגֶּא, וְכָבָה נֵר / סְעִפָּיו וְרוֹאוֹתָיו;

וְיֵלֵךְ כְּפוּף קוֹמָה / כְּאַגְמוֹן, וְצָדוּ אוֹן

צְעָדָיו, וּפְעָמָיו / מִדְּרוֹךְ אֶת מְסִלּוֹתָיו;

וְשַׁחוּ בְּנוֹת שִׁירוֹ / וְגָדַל כְּיָם שִׁבְרוֹ

וּמִתְנוֹדְדִים תַּחַת / זְמִרָיו וְשִׁירוֹתָיו;

וְקוֹל יַעֲבֹר נָחוּץ / אֲבָל אֶל חֲצַר מָוֶת

מְגוּרֵי מְיֻדָּעָיו / וְהוֹרָיו וְהוֹרוֹתָיו;

וְנוֹשֶׁה יְבַקֵּשׁ / הַנְּפָשִׁים וּפִקְדּוֹנוֹ

יְבַקֵּשׁ, וּמִי יוּכַל / תְּפוֹס אֶת חֲבוּלוֹתָיו;

וּמִדְּבִיר יְנִדֻהוּ / לְמִדְבָּר יְרִיצֻהוּ,

וְחֶמְדּוֹת מְעוֹנֹתָיו / בְּזָדוֹן מְקִיאוֹתָיו;

הֲכִי סוֹף אֱנוֹשׁ לָשׁוּב / לְעַפְרוֹ וּבָא מוֹעֵד

לְכָל חַי, וּבַשַּׁחַת / יְכוֹנֵן מְלוֹנֹתָיו;

יְשׁוּבוּן יְסוֹדוֹתָן / כְּמֵרֹאשׁ לְקַדְמוּתָן,

וְרוּחוֹ לְהֵיכְלֵי אֵל / וְצִלּוֹ נְעִימוֹתָיו;

וְנוֹדַע בְּעוֹלַם / הַפְּגָרִים, אֲשֶׁר זָרַע

וְיֶאֱסֹף בְּעוֹלַם / הַנְּפָשִׁים תְּבוּאוֹתָיו;

וְיוּבָא בְמִשְׁפָּט עַל / גְּלוּיָו וְנֶעֱלָמָיו,

וְשָׁמָּה יְשַׁלֶּם לוֹ / אֱלֹהִים פְּעֻלּוֹתָיו.


אֱלֹהַי אֲחָזַתְנִי / רְעָדָה בְּהַזְכִּירִי

פְשָׁעַי וְחַם לִבִּי / וְעָצְמוּ מְרֹרֹתָיו;

הֲכִי מִימֵי נֹעַר / לְשַׁלֵּם חֲטָאִים סָר

וְהִזְקִין עֲדֵי לָאָה / נְשֹׂא אֶת אֲלֻמּוֹתָיו;

וְהֵזִיד בְּחַטָּאתוֹ / וְשָׁגָה בְּצִדְקָתוֹ,

וְהִמְעִיט תְּשׁוּבָתוֹ / וְהִרְבָּה מְשׁוּבֹתָיו;

וְהֵעֵז כְּצוּר מִצְחוֹ / וְהִקְשָׁה כְצוּר מוֹחוֹ,

עֲדֵי סָף זְמַן כֹּחוֹ / וְסָר צֵל בְּחוּרֹתָיו;

וְעָפָה שְׁנוֹת חַיָּיו / וְחֶפְצוֹ וּמַאֲוַיָּיו,

וְרַבּוּ כְּחוֹל הַיָּם / חֳלָיָיו בְּחַטֹּאתָיו;

וְעָשַׁשׁ כְּעָשׁ עַצְמוֹ / וְסָר מֵאֱנוֹשׁ טַעְמוֹ,

וּפָרַץ זְמַן כַּרְמוֹ / וְנִשְׁחֲתוּ זְמִירוֹתָיו;

וְשָׁלַח גְּאוֹן יָדוֹ / בְּגוּפוֹ וּבִכְבוֹדוֹ,

וְקָדַר מְאוֹר פָּנָיו / וְהִלְבִּין קְוֻצּוֹתָיו;

וְהֵמַר בְּמַר רוּחוֹ / עֲדֵי מָתְקוֹ נִכְחוֹ,

לְפִי טֹעֲמִים שֹׁרֶשׁ / רְתָמָיו וְלַעֲנוֹתָיו;

וְאִם דָּם בְּפִיו יוֹרִיד / וּמָטָר נְדוֹד יָרִיד

וְיִסְתֹּם בְּעָב צִדְקוֹ / וְיָסִיר תְּפִלּוֹתָיו;

וְאֵיךְ זֵד, פְּנֵי יוֹצְרוֹ / יְחַלֶּה וְיִתְחַנֵּן,

מְטֻנָּף בְּדַם סִכְלוֹ / וּבִדְמֵי זְדוֹנֹתָיו.

וְאָדוֹן הֲיִתְחַקֶּה / עֲלֵי צַעֲדֵי חוֹטֵא,

אֲשֶׁר לַעֲשׂוֹת הִסְכִּיל / וְהִרְבָּה תְּנוּאוֹתָיו;

וְיִפְקֹד עֲלֵי רִמָּה / וְיִדְרֹשׁ לְתוֹלֵעָה,

וְהוּא יַעֲשֶׂה בָשְׁתּוֹ / וְיִלְבַּשׁ כְּלִמּוֹתָיו;

בְּרַחֲמָיו יְהִי פֹנֶה / לְעַבְדּוֹ וְאַל יִבְזֶה

עֲנוֹתוֹ בְּגֵרוּתוֹ / וְצוּקָיו וְקוֹרוֹתָיו;

וְנוּדוֹ יְהִי סוֹפֵר / וְיָשִׂים בְּנֹאדוֹ אֶת

דְּמָעָיו, וְיַעֲבֹר עַל / הֲמוֹנֵי עֲוֹנוֹתָיו;

וְאַל נָא יְדִינֵהוּ / בְּמַעֲלוֹ כְּמִפְעָלָיו,

וְיִמְחֶה זְדוֹנוֹתָיו / בְּמֵימֵי סְלִיחוֹתָיו;

וְחֶלְקוֹ יְשַׁו עִם / הַחֲפֵצִים בְּיִרְאָתוֹ,

וְנוֹצְרֵי תְעוּדוֹתָיו / וְהֹלְכִים בְּתוֹרֹתָיו;

וְיִזְכֶּה בְּיוֹם יִקְרָא / לְעַמּוֹ פְּקַח-קוֹחַ,

וְיִקְבֹּץ נְפוּצוֹתָיו / וְיוֹם הַחֲיוֹת מֵתָיו.
