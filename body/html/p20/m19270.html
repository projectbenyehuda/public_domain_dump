אֱמוּנַת אֱלֹהַי אוֹדִיעָה

מי כמוך


אֱמוּנַת אֱלֹהַי אוֹדִיעָה / וְחִידוֹת קֶדֶם אַבִּיעָה,

גְּבוּרַת בְּרֵאשִׁית אַשְׁמִיעָה, / הִקָּבְצוּ וְשִׁמְעוּ בְּנֵי יַעֲקֹב.

בְּטֶרֶם הֱיוֹת אוֹרָה / וְשָׁמַיִם וָאָרֶץ בָּרָא –

אָמוֹן אֶצְלוֹ תּוֹרָה / וַיַּעֲמִידֶהָ לְיַעֲקֹב.

רִבָּה בְאַרְצוֹ דְּשָׁאָיו / לְהַחְיוֹת נִבְרָאָיו,

עֲלֵיהֶם יְבָרְכוּ יְרֵאָיו / וְהִקְדִּישׁוּ אֶת קְדוֹשׁ יַעֲקֹב.

הֵכִין בַּמְּרוֹמִים צְבָאָם, / לְהָאִיר לָאָדָם בְּרָאָם,

וְהֵם יַעַבְדוּם בְּחֶטְאָם – / לֹא כְאֵלֶּה חֵלֶק יַעֲקֹב.

מִינֵי חַיּוֹת הוֹלִידָם, / בַּיָּם וּבַיַּבָּשָׁה יְסָדָם,

וַיֹּאמֶר: נַעֲשֶׂה אָדָם, / לְהָקִים אֶת שִׁבְטֵי יַעֲקֹב.

בְּשִׁשָּׁה הָיוּ צִבְאוֹת / מַעְלָה וּמַטָּה נִבְרָאוֹת,

שַׁבָּת לַייָ, לְאוֹת / בֵּינוֹ וּבֵין יַעֲקֹב.

נִתְּנָה לְאָדָם מִצַּלְעוֹ / עֵזֶר בְּגַן שַׁעֲשׁוּעוֹ –

הוּא הַמּוּכָן לְזַרְעוֹ, / הַנִּקְרָאִים בְּשֵׁם יַעֲקֹב.

עָרוּם הִשִּׁיא הַשֵּׁא / הָאִשָּׁה בְּדָבָר קָשֶׁה,

כַּכָּתוּב בְּתוֹרַת מֹשֶׁה, / מוֹרָשָׁה קְהִלַּת יַעֲקֹב.

זְעוּמִים מֵעֵדֶן גֹּרְשׁוּ, / וְאֵלֶּה הַדְּבָרִים פֹּרְשׁוּ,

מַכּוֹת עֲלֵיהֶם הִתְחַדְּשׁוּ – / כֹּה אָמַר יְיָ אֶל בֵּית יַעֲקֹב.

רוֹעֶה – מַעֲשֵׂהוּ זָר, / זַרְעוֹ אִכָּר גָּזַר,

וַיֶּאֱרֹב לוֹ כְּאַכְזָר, / כִּי כָל אָח עָקוֹב יַעֲקֹב.

הֶעֱוָה הַיְקוּם יָשְׁרוֹ / וּמַבּוּל נִמְחָה יִצְרוֹ,

לוּלֵי תָמִים בְּדוֹרוֹ – / מִי יָקוּם יַעֲקֹב?

מִגְדָּל בָּנוּ מוֹרִים / וּמִשָּׁם הָיוּ נִפְזָרִים,

אַחַר וְנוֹלְדוּ הוֹרִים, / אַבְרָהָם יִצְחָק וְיַעֲקֹב.

נֶעֱקָד נָטַע שְׁתִילִים – / הַגָּדוֹל אִישׁ הֲתֻלִּים,

וְהַקָּטָן יוֹשֵׁב אֹהָלִים; / מַה טֹּבוּ אֹהָלֶיךָ, יַעֲקֹב!

מָכַר הָרַב בְּכוֹרָתוֹ / לְצָעִיר לָקַח בִּרְכָּתוֹ,

וַיֹּאמֶר, כְּהַשְׁמִיעוֹ צַעֲקָתוֹ: / ׳הֲכִי קָרָא שְׁמוֹ יַעֲקֹב!׳

דִּבְרֵי סִבּוֹת הִתְעַתְּדוּ, / עֲלֵיהֶם בִּנְיָנִים יֻסְּדוּ,

וּשְׁנֵים עָשָׂר יֻלְּדוּ – / אֵלֶּה תֹּלְדוֹת יַעֲקֹב.

נִמְכַּר יֶלֶד שַׁעֲשֻׁעִים / וַיִּמְלֹךְ כְּאַחַד הַשּׁוֹעִים,

וַיֵּרְדוּ מִצְרַיְמָה שִׁבְעִים / יֹצְאֵי יֶרֶךְ יַעֲקֹב.

תְּמִימָה שִׁוְּעָה בְעָנְיָהּ / וּבְכָל מִצְרַיִם הִכּוּ לֶחְיָהּ

וַיֹּאמְרוּ: ׳לֹא יִרְאֶה יָהּ / וְלֹא יָבִין אֱלֹהֵי יַעֲקֹב.

תֹּקֶף צָרָה וְצוּקָה / וּבְכָל מָתְנַיִם מוּעָקָה,

וּבְרִית אָבוֹת צָעֲקָה: / ׳מִי נָתַן לִמְשִׁסָּה יַעֲקֹב?׳

טוֹב מֵאַרְצוֹ בָּרַח, / בְּזִקְנָתוֹ כְּשֶׁמֶשׁ זָרַח,

מִמַּטֵּה לֵוִי פָּרַח – / דָּרַךְ כּוֹכָב מִיַּעֲקֹב.

יְהִירִים בְּמַכּוֹת נִצְמָתוּ / וְכִמְעַט תַּמּוּ נִכְרָתוּ,

רֵאשִׁית אוֹנִים הוּמָתוּ – / וַתְּחִי רוּחַ יַעֲקֹב.

לַיְלָה יָצְאוּ חָפְשִׁים / בְּיַד רָמָה, חֲמוּשִׁים,

וּמֶרְכָּבוֹת עִם פָּרַָשִׁים / רָדְפוּ אַחֲרֵי בְּנֵי יַעֲקֹב. הֲדוּרִים בַּיָּם בְּבוֹאָם – / וְהִנֵּה הָאוֹיֵב רָאָם;

׳אָרִיק חַרְבִּי, נָאַם, / וְאַהַרְגָה אֶת יַעֲקֹב!׳

וְאֵל בַּיַּבָּשָׁה הֶעֱבִירָם – / וְסוֹרְרִים בַּיָּם נִעֲרָם,

אָז יָשִׁיר מֹשֶׁה בְּקוֹל רָם: / ׳גָּאַל יְיָ עַבְדּוֹ יַעֲקֹב!׳
