מַסְעֵי יְהוּדָה פרק ראשון

מסעי יהודה – חמישה פרקי מסע מחורזים לאלחריזי

יוצאים לאור על פי כתבי יד עבריים בידי יוסף יהלום ועל פי כתבי יד בערבית־יהודית בידי יהושע בלאו

מכון בן צבי, יד יצחק בן צבי והאוניברסיטה העברית ירושלים תשס״ג.


נְאֻם הֵימָן הָאֶזְרָחִי הַגֵּד הִגִּיד לִי חֶבֶר הַקֵּינִי וַיֹּאמַר לִי הָיִיתִי בְּאֶרֶץ תֵּבֵץ / חֲלִיצוֹת הַמְּלִיצוֹת אֲשַׁבֵּץ / וּנְפוּצוֹת הַתְּבוּנוֹת אֲקַבֵּץ / וַיְהִי הַיּוֹם וַאֲנִי בְּשַׁעַר הָעִיר / וַאֵרֶא תַחַת אֵלָה עֲבֻתָּה קִבּוּץ הָמוֹן רָב / וַאֲסַפְסֻף נֶעֱרָב / וְכַאֲשֶׁר קָרַבְתִּי בְּלֵב מִשְׁתָאֶה / לִרְאוֹת הַמַּרְאֶה / מָצָאתִי חֶבְרַת מַשְׂכִּילִים / מִבְּנֵי הַפַּחוֹת וְהָאֲצִילִים / בְּדִבְרֵי מְלִיצוֹת מְדַבְּרִים / וְשִׁיר נְעִימוֹת מְחַבְּרִים / וּמִמַּחְצַב לִבָּם פְּנִינִים חוֹצְבִים/ וּמִמַּעְיַן שִׂכְלָם מַיִם חַיִּים שׁוֹאֲבִים / וְזֶה לָזֶה רוֹצֶה לְנַסּוֹת / לְמַעַן יִלְמְדוּ מְלֶאכֶת הַשִּׁיר לַעֲשׂוֹת / וְכַאֲשֶׁר רָאִיתִי כִּי בִמְלֶאכֶת הַשִּׁיר בָּאוּ / וְלַעֲרֹך מִלְחַמְתּוֹ נִצְבָּאוּ / יָדַעְתִּי כִּי אֲנִי גִּבּוֹרָם / וִיוֹשֵׁב עַל כִּסֵּא כְבוֹדָם / וְלִי יִתְּנוּ יָדָם.

וָאַעַן וָאֹמַר אַנְשֵׁי לֵבָב שִׁמְעוּ לִי / וְהַקְשִׁיבוּ לִמְלִיצוֹת שִׂכְלִי / כִּי אֲנִי הָאִישׁ אֲשֶׁר בְּנִיבוֹ הַרֲרֵי עַד יַרְעִישׁ / וּבְשִׁירוֹ הַפְּנִינִים יוֹבִישׁ / וְנַחֲלֵי אֵיתָן בְּשִׂכְלוֹ יוֹבִישׁ / וְיוֹצִיא מַיִם מִצּוּר הַחַלָּמִישׁ / וְהָרוֹצֶה לְנַסּוֹת עָדָיו יַגִּישׁ / וְהַמַּאֲמִין לֹא יָחִישׁ / אָמַר לִי רָאשׁ הַקְּרוּאִים בַּמֶּה כֹּחֲךָ זֶה / וּבַמֶּה נִכָּר שִׂכְלְךָ וְיֵחָזֶה / הַבְּשִׁירוֹת / אוֹ בְּאִגְּרוֹת / וּבַחֲרוּזִים נֶחֱמָדִים / אוֹ בְּמִלִּים מְפוֹרָדִים / אָמַרְתִּי דְעוּ כִּי לִבִּי בְּכָל קֶלַע יִקְלַע / וְשִׂכְלִי בְּכָל תּוּשִׁיָּה יִתְגַּלַּע / וּבְכָל אֲשֶׁר תְּנַסּוּנִי / תִּמְצָאוּנִי / אָמְרוּ לִי אִם כֵּן הַשְׁקֵנוּ מִן הַמַּעְיָן / וְהַשְׁמִיעֵנוּ מִשִּׁירוֹתֶיךָ אֲשֶׁר חִבַּרְתָּ בְכָל עִנְיָן / אָמַרְתִּי לָהֶם כְּדִבְרֵיכֶם אֶעֱשֶׂה / וְסוֹדִי מִכֶּם לֹא אֲכַסֶּה. וּתְחִלַּת מַה שֶׁאֶזְכּוֹר לָכֶם שִׁירֵי תְשׁוּבָה / עֲשִׂיתִים לְבַקֵּשׁ סְלִיחָה עַל הַמְּשׁוּבָה

          אֱלֹהַי, יַעֲלֶה זִכְרִי לְטוֹבָה / וְאַל תִּרְחַק, הֲכִי צָרָה קְרוֹבָה.

          וְהַבֶּט נָא לְגֹדֶל הַתְּלָאָה / וְאַל תַּבֵּט לְגֹדֶל הַמְּשׁוּבָה.

          בְּמֵי דִמְעִי מְחֵה פִשְׁעִי וְאַל נָא / תְּהִי חַטַּאת יְהוּדָה עוֹד כְּתוּבָה.


וְכָתַבְתִּי אֵלֶּה הַשִּׁירִים עַל הֵיכַל בֵּית הַכְּנֶסֶת / יָפָה מִתְנוֹסֶסֶת:

          שַׁדַּי, שְׁעֵה קוֹל צוֹעֲקִים אֵלֶיךָ / מִשְׁתַּחֲוִים לִפְנֵי הֲדוֹם רַגְלֶיךָ.

          רַמְתָּ וְעָצַמְתָּ וְנֶעְלַמְתָּ בְּאֵין /מָקוֹם וְנִגְלֵיתָ בְּמִפְעָלֶיךָ.

          עַל כֵּן עֲבָדֶיךָ יְדֵיהֶם קָצְרוּ / לִבְנוֹת לְךָ נָוֶה כְּפִי גָדְלֶךָ.

          אוּלָם בְּכֹחָם כּוֹנְנוּ מִקְדָּשׁ מְעַט / לָשׂוּם מְגַמָּתָם לְהֵיכָלֶךָ.

          לִבָּךְ וְעֵינֶיךָ יְהִי שָׁמָּה, הֲכִי / לִבָּם וְעֵינָם נָשְׂאוּ אֵלֶיךָ.


וְאֵלֶּה עֲשִׂיתִים עַל עִיר יְרוּשָׁלָיִם / בְּבוֹאִי אֵלֶיהָ מִמִּצְרָיִם:

          שָׁלוֹם לְעִיר שָׁלֵם אֲשֶׁר יָפְיָהּ / נֶעְלָם לְרוֹאִים אוֹמְרִים אַיָּהּ.

          כָּל עֵת אֲחוֹנֵן אֶת עֲפָרָה כִּי / הִיא מַחֲלַת נַפְשִׁי וְהִיא צָרְיָהּ.

          עֵינִי בְעֵת כִּי רָאֲתָה אוֹרָהּ / לֹא נֶעְצְרוּ בָהּ עַנְנֵי בִכְיָהּ.

          הָאֵל אֲשֶׁר אִוָּה לְמוֹשָׁב לוֹ / יִרְאֶה בְּרֹב עָנְיָהּ וְשָׁב שִׁבְיָהּ.

          בּוֹכָה לְנוּד דּוֹדָהּ בְּחֶמְלָתוֹ / יִמְחֶה דְמָעוֹת מֵעֲלֵי לֶחְיָהּ.


וְעוֹד אָמַרְתִּי / וְעַל זֶה הָעִנְיָן חִבַּרְתִּי:

          לְצִיּוֹן שָׁאֲפָה נַפְשִׁי וְרוּחִי / הֱיוֹת בָּהּ כָּל יְמֵי חַיַּי מְנוּחִי.

          בְּחִשְׁקָהּ יִנְעֲמוּ לִי הַתְּלָאוֹת / וְאֶתֶּן־בָּהּ לְמַכִּי רֹאשׁ וְלֶחִי

          וְלֹא אֶתְאַו לְבִלְתָּהּ כָּל יְמוֹתַי / עֲדֵי אֶגְוַע וְנָס כֹּחִי וְלֵחִי.

          אֲחוֹנֵן אֶת אֲבָנֶיהָ בְּכָל עֵת / עֲדֵי יַשְׁאִיר עֲפָרָהּ אוֹת בְּמִצְחִי.

          וְאֵיךְ אֶכְסוֹף לְטוּב רִקְחֵי בְשָׂמִים / וְהִיא בָשְׂמִי וְטוֹב רִקְחִי וְרֵיחִי.

          עֲפָרָה אֶעֱשֶׂה מַרְפֵּא לְנִגְעִי / וְאֶמְרַח עַל שְׁחִין לִבִּי וְיֶחִי.


וְאֵלֶּה עָשִׂיתִי בְחָלְיִי / כְּשֶׁהֵפִיק הָאֵל צָרְיִי:

          אַל נָא אֱלֹהַי יֶחֱרֶה אַפָּךְ / עַל דַל אֲשֶׁר יֶחְסֶה בְּתוֹךְ צִלָּךְ.

          אַל תַּעֲלֵנִי בַחֲצִי יָמַי / מוֹתִי הֲיוֹעִיל לָךְ וְיִסְכֹּן לָךְ.

          לֹא אֶשְׁאֲלָה עשֶׁר וְתֶן־לִי הַ/נֶּפֶשׁ וְאָכֵן הָרְכוּשׁ קַח לָךְ.


וְעָשִׂיתִי עַל אִישׁ מֵחַסִידֵי עַמּוֹ / נִפְטַר לְעוֹלָמוֹ:

          מִפַּחְדְּךָ נַפְשָׁךְ בְּצוֹם עִנִּיתָ / עַד כִּי לְחוּט הַשַּׂעֲרָה דָמִיתָ.

          יָגְעוּ בְנֵי נָבָל וְיָגַעְתָּ לְטוֹב / כִּי יָרְדוּ הֵמָּה וְאַתְּ עָלִיתָ.

          אַשְׁרֵי יְחִידָתָךְ אֲשֶׁר שַׂמְתָּ בְכֵס / עֶלְיוֹן וְאוֹתָהּ מִשְּׁאוֹל פָּדִיתָ.


וְאֵלֶּה עָשִיתִי בַיָּם / בְּהִנָּצְלִי מִשְּׁאוֹן גַּלָּיו וְדָכְיָם:

          בַּמָּה אֲקַדֵּם אֵל וּמִי אָנִי / כִּי עַד הֲלֹם חַסְדּוֹ הֱבִיאָנִי.

          מִקְּצֵה אֲדָמָה וַאֲצִילֶיהָ / לָבֹא לְהַר קָדְשׁוֹ קְרָאַנִי.

          וּבְיָם בְּעָבְרִי בִּשְּׂרַנִי כִּי / תַּעֲבֹר בְּמַיִם אִתְּךָ אָנִי.


וְאֵלֶּה שָׁלַחְתִּי לְקֶבֶר אֲדוֹנֵנוּ יְחֶזְקֵאל / מִכֹּבֶד חֶטְאִי לְהָקֵל:

          לְבַקֵּר קִבְרְךָ עֵינַי אֲגַמִּים / וְדִמְעָתִי מְגוֹלָלָה בְדָמִים.

          נְבִיא הָאֵל יְחֶזְקֵאל הַמְּכֻבָּד / בְּפִי צִבְאוֹת מְרוֹמִים וַהֲדוֹמִים.

          קְרָבַי נִכְסְפוּ לִלְחֹךְ עֲפָרָךְ / וְלִלְקֹט מֵעֲפַר קִבְרָךְ לְשָׁמִים.

          וְלִשְׁאֹף מֶרְקְחוֹת בֹּשֶׂם בְּרֵיחָךְ / מְקֻטָּרִים בְּכָל רָאשֵׁי בְשָׂמִים.

          וְאָז אָשִׁיר לְאֵל חַיַּי וְאוֹמַר / חֲבָלִים נָפְלוּ לִי בַנְּעִימִים.


וְאֵלֶּה עָשִׂיתִי לְקֶבֶר אֲדוֹנֵנוּ עֶזְרָא הַסּוֹפֵר / לִמְצֹא כַּפָּרָה:

          לְאוֹר עוֹלָם אֲשֶׁר הֶחְפִּיר מְאוֹרוֹת / וְסָבִיב עוֹמְדִים לוֹ כַמְּנוֹרוֹת.

          וְעָדָיו שׁוֹמְעֵי שִׁמְעוֹ יְבוֹאוּן / וְנַפְשׁוֹתָם יְרִימוּן לוֹ תְשׁוּרוֹת.

          וְאֵיךְ יוּכַל לְכַחֵשׁ נִפְלְאוֹתָיו / וּבוֹ הָעוֹבְרִים יִרְאוּ גְבוּרוֹת.

          כְּבוֹד עֶלְיוֹן סְבִיב קִבְרוֹ יְנוֹצֵץ / וְיָפִיץ אוֹר לְעֵינַיִם סְגוּרוֹת.

          כְּבוֹד עֶזְרָא קְדוֹשׁ אֵל הַמְכֻבָּד / אֲשֶׁר הוּא לַיְּקוּם עֶזְרָה בְצָרוֹת.


וְעָשִׂיתִי לְאִישׁ עַז וְקָשֶׁה / מֵשִׁיב עַל חִבּוּר רַבֵּנוּ משֶׁה ז״ל

          כָּל מֵשִׁיב עַל דִּבְרֵי משֶׁה / שָׁבַר עוֹל גַּם מוּסָר נִתַּק.

          תֵּאָלַמְנָה שִׂפְתֵי שֶׁקֶר / הַדּוֹבְרוֹת עַל צַדִּיק עָתָק.


וְעוֹד עָשִׂיתִי בְּעִנְיָנוֹ / עַל הַמִּתְפָּאֵר עָלָיו בְּהֶגְיוֹנוֹ

          וְהַחַלָּשׁ בְּעֵת יִהְיֶה בְּבֵיתוֹ / יְגַלֶּה אֶת גְבוּרָתוֹ לְאִשְתּוֹ.

          וְעֵת אֶל הַקְּרָב יִגַּשׁ וְיִקְרַב / אֲזַי תִּתְגַּל כְּלִמָּתוֹ וּבָשְׁתּוֹ.


וְאֵלֶּה עֲשִׂיתִים / וּלְרַבֵּנוּ משֶׁה מִסְּפָרַד שְׁלַחְתִּים:

          מֵאִתְּךָ הַשַּׂר תְּהִלָּתֵנוּ / וּבְךָ וּמִמֶּךָ כָּל גְּדֻלָּתֵנוּ.

          מַלְאַךְ אֱלֹהִים אַתְּ וְנִבְרֵאתָ בְּצֶ/לֶם אֵל וְאִם אַתָּה כְתַבְנִיתֵנוּ

          בִּגְלָלְךָ אָמַר אֱלֹהִים נַעֲשֶׂה / אָדָם בְּצַלְמֵנוּ וְכִדְמוּתֵנוּ.


וְעָשִׂיתִי בְעִנְיָן אַחֵר הֶפְכָּם / עַל אֲנָשִׁים הִשְׁחִיתוּ דַרְכָּם:

          הוֹי עַל אֲנָשִׁים לֹא לְעֵזֶר נִבְרְאוּ / עַל כֵּן בְּזָכְרָם יִפְּלוּ פָנֵינוּ.

          הֵם כַּשְּׁוָרִים מִבְּלִי לֵבָב וְאֵיךְ / יֵדְעוּ דְבַר חָכְמָה וְיִתְבּוֹנֵנוּ.

          לֹא בַעֲדָם אָמַר אֱלֹהִים נַעֲשֶׂה / אָדָם בְּצַלְמֵנוּ וְכִדְמוּתֵנוּ.


וְאֵלֶּה עָשִׂיתִי לְרַ׳ מְנַחֵם בַּר יִצְחָק דַּיַּן מִצְרַיִם / אֲשֶׁר אֵין כָּמוֹהוּ נְדִיב יָדַיִם וּנְקִי כַפָּיִם.

          יְמִינָךְ מִנְּהַר פִּישׁוֹן רְחָבָה / וְלָךְ בָּרָא אֱלֹהִים הַנְּדָבָה.

          לְךָ נֶפֶשׁ אֲשֶׁר תָּפִיץ יְאוֹרִים / וְהִיא מִלַּהֲבֵי הָאֵשׁ חֲצוּבָה.

          וְלוּ נָגְעָה יְמִין כָּל קוֹפְצֵי יָד / וְכָל כִּילַי בְּיָדָךְ הַנְּדִיבָה.

          אֲזַי שָׁבוּ בְנֵי דוֹרָךְ נְדִיבִים / וְלֹא נִשְׁאַר אֱנוֹשׁ כִּילַי בְּצוֹבָה.


וְאֵלֶּה עָשִׂיתִי לְחָכָם אַבְרָהָם בֶּן רַבֵּנוּ מֹשֶׁה בֶּן מַיְמוֹן זַ"ל.

          עֲשִׂיתִים בִּירוּשָׁלַיִם. וּשְׁלַחְתִּים לְמִצְרַיִם:

          אִם הַיְקָר אָדוֹן לְכָל אָדוֹן / הוּא בֶאֱמֶת עֶבֶד לְאַבְרָהָם.

          מִיּוֹם כְּבוֹדוֹ נֶעֱלַם מִנִּי / אֵלֵךְ מְשׁוֹטֵט בַּעֲדוֹ נִדְהָם.

          כָּמְהוּ שְׂעִפַּי לוֹ בְּעָבְרִי אֶ/רֶץ נוֹד וְאִתִּי יַעֲבֹר כִּמְהָם.


וְעָשִׂיתִי לְאִישׁ אֲשֶׁר הָיְתָה לוֹ בַּת בְּתוּלָה יְקָרָה / וְנִמְצֵאת הָרָה:

          וְיוֹנַת חֵן מְקַנֶּנֶּת בְּחֵיק אָב / מְמַלֵּא יוֹם לְיוֹם טַרְפָּהּ וּמַסְפִּיק.

          חֲשָׁבוּהָ כְּהֵא רָפָה אֲבָל עֵת / בְּחָנוּהָ מְצָאוּהָ בְּמַפִּיק.


וְעָשִׂיתִי לְאִישׁ מִן הַנְּבָלִים. וְרֹאשׁ הַנּוֹכְלִים:

          רְאוּ כִילַי אֲשֶׁר יָדוֹ בְּתוּלָה / לְכָל־שׁוֹאֵל וְלַכִּילוּת בְּעוּלָה.

          וּבוֹ פֶלֶא וְאוֹת כִּי הוּא בְאֶחָד / וְיֵחָשֵׁב כְּסַנְהֶדְרִי גְדוֹלָה.


וְעָשִׂיתִי לְאִישׁ בַּעַל הַמִּדּוֹת וְהַמַּעֲשִׂים / רַ׳ שְׁמוּאֵל בֶּן הָרֹאשׁ ר׳ נִסִּים

          הַשַּׂר אֲשֶׁר לוֹ בֵּין צְבָא רוּם מַחֲנֶה / לִבִּי לְפֵרוּדָךְ מְדֻכָּא נַעֲנֶה.

          אֶקְרָא וְלֹא יַעְנֶה מְשׂוֹשִׂי בִּלְתְּךָ / אַךְ הַבְּכִי טֶרֶם קְרָאתִיו יַעֲנֶה.

          אַתָּה מְאוֹר עוֹלָם וְתִשְׁכֹּן מַחֲשָׁךְ / כֵּן הַשְּׁכִינָה בַעֲרָפֶל תַּחֲנֶה.

          אֵשׁ הַסְּנֶה בָּךְ נִרְאֲתָה הַשָּׂר הֲכִי / אַתָּה כְּמַלְאַךְ אַךְ בְּהֶסְנֶה הַסְּנֶה.


וְעָשִׂיתִי לְאִישׁ אֲשֶׁר יָדוֹ סְגוּרָה / וְעַב נִדְבָתוֹ עֲצוּרָה:

          יָדָךְ לְשׁוֹאֵל סְגוּרָה / בְּתוּלָה וְאִישׁ לֹא יְדָעָהּ.

          לוּ אִמְּךָ הָיְתָה כֵן / לֹא יִהְיֶה מוּם בְּזַרְעָהּ.


וְעָשִׂיתִי עוֹד עָלָיו / וּשְׁלַחְתִּים אֵלָיו:

       רְאוּ כִילַי אֲשֶׁר יָדוֹ כְסֶלַע / וְלִבּוֹ לַעֲצֹר אוֹתָהּ יְאַמֵּץ.

       אֲנָשִׁים נוֹקְדִים יָדוֹ בְּפַתָּח / וְאַךְ הוּא מַחֲלִיף אוֹתָהּ בְּקָמַץ.


וְעָשִׂיתִי לְנַעַר נָשָׂא אִשָּׁה זְקֵנָה / וְלָהּ עשֶׁר וְאֵין קֵצֶה לַתְּכוּנָה:

       נִפְלֵאת גְּבִירַת חֵן בְּעִנְיָנַיִךְ / מֵרֹב כְּבוֹדֵךְ יִבְעֲלוּךְ בָּנָיִךְ.

       לוּ תֶחֱזִי צוּרַת דְּמוּתֵךְ תּוֹךְ רְאִי / מוּצָק לְפַחְדֵּךְ נָפְלוּ פָנָיִךְ.

       יָפְיֵךְ מְעַט כִּימֵי שְׁנוֹת בַּעֲלֵךְ וְעָשְׁ/רֵךְ רַב לְאֵין קֵצֶה כְּמוֹ שָׁנָיִךְ.


וְעָשִׂיתִי לְאִישׁ נָבָל אֲשֶׁר קָפַץ מֶנִּי יָדוֹ / וְלָמַד הַכִּילוּת מִיּוֹלְדוֹ:

       אִם בֶּן נָבָל קָפַץ מֶנִּי / יָדוֹ מִצְוַת יוֹלְדוֹ הֵקִים.

       נִשְׁבַּע הָאֵל בַּל יִהְיֶה עַל / נִכְבָּדִים חֶסֶד הָרֵקִים.

       כִּי לֹא יָנוּחַ שֵׁבֶט רֶ/שַׁע עַל גּוֹרַל הַצַּדִּיקִים.


וְעָשִׂיתִי לְאִישׁ נָדִיב בִּלְשׁוֹנוֹ / וְכִילַי בְּהוֹנוֹ:

       אִם בַּעֲלֵי מִשְׂרָה כְּמוֹ עַיִן / אַתָּה אֱמֶת לָהֶם כְּמוֹ אִישׁוֹן.

       וּבְעֵת יְרוּצוּן עַל נְתִיב הַיְקָר / אַתָּה לְבַדָּךְ תַּעֲבֹר רִאשׁוֹן.

       וַאֲשֶׁר יְמַלְאוּן הֵם בְּמַעְשֵׂה יָד / אַתָּה תְמַלֵּא מִדְּבַר לָשׁוֹן.


וְעָשִיתִי לְאִישׁ מְשַׁלֵּחַ מְדָנִים / רְאִיתִיו וְהוּא מִתְפַּלֵּל בַּעֲנָוָה וּבְתַחֲנוּנִים:

       רְאוּ זָקֵן עוֹלֶה לְהַפִּיל תַּחְנוּנִים / בְּלֵב יִקְבֹּץ אָוֶן וּמָלֵא תַזְנוּנִים.

       בְּלָבָן מִתְעַטֵּף כְּשׁוֹבַךְ הַיּוֹנִים / מְלֻבָּן מִבַּחוּץ מְטֻנָּף מִבִּפְנִים.


וְעָשִׂיתִי לְאִישׁ יָצָא לוֹ שֵׁם נְדָבָה / וְיָדוֹ מִטַּל נְדָבוֹת חֲרֵבָה:

       אֶתְמַהּ לְעָבֵי טוּבְךָ אֵיךְ נִכְלְאוּ / דּוֹדִי וְנוֹזְלֵי יָדְךָ אֵיךְ קָפְאוּ.

       אַתָּה כְּמוֹ מָחוֹל אֲשֶׁר לוֹ קוֹל מְאֹד / גָּדוֹל. וּבוֹקְעָיו רִיק וְרוּחַ מָצְאוּ.


וְעָשִׂיתִי לְאִישׁ מֵאַנְשֵׁי כַלְנַה. הִלַּלְתִּיו וּבָרַח לְחָרָן מִמֶּנִּי וְנִסְתָּר מֵעֵינַי:

       יְלִיד זֻלּוּת. אֲחִי נַבְלוּת וְכִילוּת / אֲשֶׁר כָּל הַנְּבָלוֹת נִקְבְּצוּ בוֹ.

       וְהִלַּלְתִּי שְׁמוֹ בִשְׁתֵּי לְשׁוֹנוֹת / וְנִכְשַׁלְתִּי בְּמַאְמַר פִּי וְנִיבוֹ.

       וְאוּלָם עֵת דְּרַשְׁתִּיו לֹא מְצָאתִיו / וְאָמְרוּ כִּי לְחָרָן שָׂם נְתִיבוֹ.

       וְאִם נֶחְבָּא וְנִסְתַּר מִלְּפָנַי / שְׁמוֹ הָרַע בְּפִי עֵטִי עֲזָבוֹ.

       כְּמוֹ עַכְבָּר אֲשֶׁר נֶחְבָּא בְּתוֹךְ חוֹר / וְעָזַב חוּץ בְּפִי חָתוּל זְנָבוֹ.


וְעָשִׂיתִי בְכַלְנַה עַל אִישׁ מִן הַנְּבָלִים הַשְּׁוָרִים / עוֹשֶׂה שִׁירִים.

;וְחוֹצֵב בּוֹרוֹת נִשְׁבָּרִים / וּמֵימָיו מָרִים.

       קְלָלוֹת הוּא מְחַבֵּר שִׁיר כְּצוֹאָה / וְהַצּוֹאָה בְּפִיו יַחְשֹׁב כְּנֹפֶת.

       וְהֶגְיוֹנוֹ יְגַלֶּה אֶת קְלוֹנוֹ / וְעֶרְוָתוֹ בְּשִׁירוֹ נֶחֱשֶׂפֶת.

       וְאָבִיו הוּא מְשֻׁעְבָּד לַנְּבָלוֹת / וְאִמּוֹ לַחֲרָפוֹת נֶחֱרֶפֶת.

       וְנֶאְסְפוּ לְעוֹלָמָם שְׁנֵיהֶם / וְחֶרְפָּתָם לְאוֹת לֹא נֶאֱסֶפֶת.

       וְהַבֵּן עֵד עֲלֵי טֻמְאַת אֲבוֹתָיו / כְּרֵיחַ רָע עֲלֵי בֵית הַטְּנֹפֶת.


וְשָׁאַלְתִּי מִמֶּנּוּ סֵפֶר / כֹּל מְשָׁלָיו מִשְׁלֵי אֵפֶר / וְתֵכֶף אֲשֶׁר שְׁלָחוֹ /

בָּא אֵלַי וּלְקָחוֹ / וְעָשִׂיתִי עָלָיו:

       קְלָלוֹת בָּא וְהִשְׁאִיל לִי כְּתָבוֹ / בְּיוֹם אֶחָד. וּבַשֵּׁנִי גְזָלוֹ.

       חֲשַׁבְתִּיהוּ בְּעֵת תִּתּוֹ וְקַחְתּוֹ / כְּכֶלֶב שָׁב לְקִיאוֹ וַאֲכָלוֹ.


וְעָשִׂיתִי לְאִישׁ כִּילַי אֲשֶׁר הִבְטִיחַ אוֹתִי בְנִדְבָתוֹ וְנָדַר נְדָרִים /

וַיּוֹצֵא פֶרַח וַיָּצֵץ צִיץ וַיִּגְמֹל שְׁקָרִים.

       בְּהִתְנַדֵּב בְּפִיו כִּילַי / וְיַחְשֹׁב חֵטְא אֲשֶׁר בָּטָא.

       יְהִי קוֹפֵץ שְׁתֵּי יָדָיו / וְנִסְלַח לוֹ אֲשֶׁר חָטָא.


וְעוֹד עָלָיו עָשִׂיתִי / וְזֶה חָזִיתִי:

       כִּילַי הֲתַעְלִים מוּמְךָ / הִנֵּה בְיָדָךְ עֵדְךָ.

       כִּילוּת בְּךָ דָבַק עֲדֵי / הָיָה לְאוֹת עַל יָדְךָ.


וְעָשִׂיתִי עַל הַיַּיִן / הַנֶּחְמָד לְמַרְאֵה עַיִן:

       וְכָל חָדָשׁ מְאֹד יֶעְרַב וְיִנְעַם / וְעֵת יִבְלֶה יְקוּצוּן בּוֹ נְפָשִׁים.

       וְיַיִן כֹּל אֲשֶׁר יִבְלֶה וְנוֹשַׁן / מְשַׂמַּח אֱלֹהִים וַאֲנָשִׁים.


וְעָשִׂיתִי בַחֵשֶׁק בִּימֵי הַנְּעוּרִים / וְתַעֲנוּג הַבַּחוּרִים:

       צְבִי הַחֵן דְמוּת יוֹסֵף בְּיָפְיוֹ / וְעַל רֹאשׁוֹ עֲטֶרֶת הוֹד כְּחֵלֶם.

       וְנַפְשִׁי הִיא לְכַף רַגְלוֹ כְּחִירִק / וְהוּא גֵאֶה וְרָם עָלַי כְּחֹלֶם.

       וְנִגַּשְׁתִּי לְגַן פָּנָיו כְּגַנָּב / לְלַקֵּט מִגְדְּנֵי פִיהוּ וָאוֹכְלֵם.

       וְעָלַי זָרְחָה שֶׁמֶשׁ לְחָיָיו / וְדָמִים לִי וְהֵן רוּחִי אֲשַׁלֵּם.


וּבְעִנְיַן הַדְּמָעוֹת / הַיּוֹרְדוֹת מֵאֵשׁ הַצְּלָעוֹת / אָמַרְתִּי:

       אַסְתִּיר צְפוּן חֵשֶׁק בְּלִבִּי אַךְ / דִּמְעִי כְּנֵס נִגְלָה בְרֹאשׁ דֶּגֶל.

       לִבִּי עֲלֵי נַחַל אֲמָנָה הוּא / שׁוֹכֵן אֲבָל דִּמְעִי בְעֵין רוֹגֵל.


וְאָמַרְתִּי עַל הָעִנְיָן / לְהָקִיר נוֹזְלֵי שִׁיר מִן הַמַּעְיָן:

       הָהּ נִגְלָה סוֹד חִשְׁקִי, אַחַר / הָיָה בִכְלוּא לִבִּי לָחוּץ.

       דִּמְעִי רָגַל אֶל עֵינַי וַ/יַּגֵּד לִשְׁנֵי אֶחָיו בַּחוּץ.


וְעָשִׂיתִי בְּאִשָּׁה אֲשֶׁר לִקְרֹא תוֹרָה יוֹדַעַת / וְכָל אִישׁ לְמִשַּׁכב זָכָר יוֹדָעַת:

       שְׁאֵלוּנִי בְּעַד אִשָּׁה חֲכָמָה / עֲנִיתִים לָמְדָה חָכְמָה לְבָשְׁתָּהּ.

       מְזִמָּתָהּ לְזִמָּתָהּ לְמוֹפֵת / וְסִמָּן נֶאֱמָן דַּעְתָּה לְדַעְתָּהּ.


וְעוֹד עָשִׂיתִי עַל עִנְיָנֶיהָ / לְגַלּוֹת תַּזְנוּנֶיהָ / וְהָיָה שְׁמָהּ טְהוֹרָה:

       וְאִשָּׁה חֲכָמָה לְהָרַע עֲרוּמָה / נְתִיב חֵטְא תְּסַקֵּל וְישֶׁר תְּעַקֵּל.

       וְקָרְאוּ שְׁמָהּ יוֹלְדֶיהָ טְהוֹרָה / וְכֵן הוּא. וְאָכֵן כְּטָהְרַת יְחֶזְקֵאל.


וְחִבַּרְתִּי סֵפֶר גּוֹרָלוֹת / לְאִישׁ מֵחֲשׁוּבֵי הַקְּהִלּוֹת / וְכָתַבְתִּי אֵלּוּ הַשִּׁירִים:

       שַׂר הַיְּקָר אַתָּה נְסִיכֵנוּ / וּנְשִׂיא אֱלֹהִים אַתְּ בְּתוֹכֵנוּ.

       הִנֵּה לְךָ סֵפֶר אֲשֶׁר מִלָּיו / יֹאמְרוּ: לְכוּ נִכְרַע לְמַלְכֵּנוּ.

       קוֹרְאִים לְךָ קוּמָה מְשׁוֹל בָּנוּ / כִּי שִׂכְלְךָ יָאִיר חֲשֻׁכֵּנוּ.

    דִּבְרֵי אֱמֶת תִּמְצָא בְּפִינוּ יוֹם / גּוֹרָלְךָ תַּפִּיל בְּתוֹכֵנוּ.


וְעָשִׂיתִי לְנָשִׂיא כִּילַי רְאִיתִיו דּוֹאֵג עַל הָעָב הַנֶּעֱצֶרֶת / בִּשְׁנַת בַּצֹרֶת:

          אַל תֶּחֱרַד עַל הַנְּשִיאִים שַׂר יְקָר / אִם נוֹזְלִים לֹא יָצְקוּ עָלֶיךָ.

          כִּי מִמְּךָ לָמְדוּ קְפֹץ יָד כִּי שְׁמָם / כִּשְׁמָךְ וּמִפְעָלָם כְּמִפְעָלֶךָ.


וְעָשִׂיתִי עַל הֶחָכָם רַ׳ יוֹסֵף הַמַּעֲרָבִי / בְּנָסְעִי לְצוֹבָה מֵאֶרֶץ צְּבִי:

          בְּךָ הוֹבִישׁ אֱלֹהִים יָם דְּמָעוֹת / בְּךָ אָמַר לְצוּלָתָם חֳרָבִי.

          בְּלִבָּךְ בֵּית זְבוּל בָּנָה לְשִׁבְתּוֹ / לְשׁוֹן קֹדֶשׁ וְיַאהֵל שָׁם עֲרָבִי.

          וְהָיִיתָ גְבִיר וָרָב בְּמַעְרָב / וּבַמִּזְרָח מְשָׁחָךְ אֵל לְנָבִיא.

          וְאָמְנָם כִּי אֲרַם צוֹבָה כְּמִקְדָּשׁ / וְאַתָּה בּוֹ כְּמוֹ נֵר מַעֲרָבִי.


וְעָשִׂיתִי עַל קְּהִלָּה / מוֹאֶסֶת הַתְּהִלָּה:

          שַּׁדַּי שְׁלַח צוֹעִים לְרוֹשֵׁשׁ כָּל / טוֹבַת בְּנֵי כַלְנֵה וְכַלֶּנָּה.

          עָשְׁרָהּ וְכָל חֵילָהּ לְזָרִים תֵּן / וַחְשׂף יְפִי סִתְרָהּ וְגַלֶּנָּה.

          אִם כַּסְּפִינָה רָחֲבָה קַצֵּר / אוֹתָהּ וְאֶל אַמָּה תְּכַלֶּנָּה.


וְעָשִׂיתִי עַל בְּנֵי עֲדִינָה / אֲשֶׁר הָיוּ בִימֵי קֶדֶם פְּאֵר כָּל מְדִינָה / וְעַתָּה נוֹתְרוּ לְמָשָׁל וְלִשְׁנִינָה:

          אַנְשֵׁי עֲדִינָה נִבְזִים שְׁפָלִים / מִבְחַר הֲמוֹנָם דֹּמֶן בְּאַשְׁפָּה.

          לֹא אַחֲרִישׁ עַד אָשִׁית לְלִבָּם / לִהְיוֹת לְחִצֵּי שִׁירַי כְּאַשְׁפָּה.

          הֵן אִם תְּמוֹל בָּם הָיוּ אֲצִילִים / הַיּוֹם יְקָרָם אָבַד וְנִסְפָּה.

          הַיּוֹם בְּשַׁחַת וּתְמוֹל בְּשַׁחַק / הַיּוֹם בְּאַשְׁפָּה וּתְמוֹל בְּחֻפָּה.

          הֵם לֹא לְעֵזֶר וָלֹא לְהוֹעִיל / כִּי אִם לְבשֶׁת וְגַּם כֵּן לְחֶרְפָּה.


וְעָשִׂיתִי לְאִישׁ כִּילַי מִכֹּל אֲשֶׁר לְפָנָיו / וְהוּא נֶחְשַׁב נָדִיב בְּעֵינָיו:

          וְרָאִיתִי אֱנוֹשׁ כִּילַי וְלִפְנֵי / נְדִיבִים הוּא בְעֵינָיו רֹאשׁ לְכֻלָּם.

          וְלֹא יַעְלֶה לְמַעְלוֹת הַנְּדָבָה / עֲדֵי יוּכַל חֲמוֹר לַעְלוֹת בְּסֻלָּם.


וְעָשִׂיתִי לְאִישׁ מִנְּדִיבֵי סְפָרַד כָּמוֹהוּ בִנְדָבָה לֹא נִהְיָה / וּשְׁמוֹ ר׳ חִיָּיא:

          שְׂאוּ שָׁלוֹם לִגְבִיר / יְקָר עִם תֹּם הֶחְבִּיר / וְזִכְרוֹ קוֹל יַעְבִיר / נְדִיב הַדּוֹר חִיָּיא.

          לְמֵי חָכְמוֹת צָמֵא / וְלַעֲשׂוֹת טוֹב הוֹמֶה / עֲרֵמוֹת הַיְקָר מֵ/עֲפַר שִׁפְלוּת חִיָּה.

          וְלוּ מִנִּדְבָתוֹ / יְאַו אִישׁ נִשְׁמָתוֹ / יְחַיֶּה זוּלָתוֹ / וְנַפְשׁוֹ לֹא חִיָּה.


וְעָשִׂיתִי לַאֲנָשִׁים הִלַּלְתִּים בְּשִׁירִים / וְלֹא הֵבִינוּ דַרְכָּם / וְלֹא יָדְעוּ עֶרְכָּם:

          תְּשׁוּרַת שִׁיר שְׁלַחְתִּיהָ לְאִישִׁים / חֲסֵרֵי בִין וְשֵׂכֶל עִם תְּמִימוּת.

          וְחָשְׁבוּ אֶת תְּהִלוֹתִַי לְקָלוֹן / וְהָרַע יַחְשְׁבוּ ישֶׁר וְתַמּוּת.

          וְכֻלָּם בַּנְּבָלוֹת הֵם שְׂמֵחִים / וְאָכֵן יִמְאֲסוּ דִבְרֵי נְעִימוּת.

          וְדִמְיוֹנָם כְּרֶמֶשׂ הַגְּלָלִים / אֲשֶׁר בַּמּוֹר יְהִי אָבוּד וְצָמוּת.

          תְּנָה אוֹתוֹ בְתוֹךְ צוֹאָה וְיִחְיֶה / וְשֶׁמֶן רֹאשׁ מְשַח רֹאשׁוֹ וְיָמוּת.


וְעָשִׂיתִי לְאִישׁ נָדִיב שָׁלַח לִי מִנְחָה / וּבַיּוֹם הַשֵּׁנִי כָּפַל לִי הָאֲרוּחָה:

          נְדִיב הַלֵּב אֲשֶׁר הֶעְשִׁיר בְּנֵי אִישׁ / וְלֹא נִמְצָא בְיָמָיו דַּל וּמִסְכֵּן.

          שְׁתֵּי יָדָיו שְׁתֵּי צָרוֹת מְקַנְּאוֹת / וְזֹאת תֵּיטִיב וְהַשֵּׁנִית תְּתַכֵּן.

          בְּעֵת תִּגְמֹל נְדָבוֹת יַד יְמִינוֹ / תְּקַנֵּא הַשְּׁנִיָּה לַעֲשׂוֹת כֵּן.


וְעָשִׂיתִי בְעִנְיַן הַחֵשֶׁק / בְּעֹפֶר אֲשֶׁר עֵינָיו כְּלֵי נֶשֶׁק:

          צְבִי הַחֵן אֲשֶׁר שָׁכֵן בְּלִבִּי / שְׂרָפָהוּ יְקוֹד צַלְעִי וּמֵעַי.

          וְלוּלֵי כִי הֲרָגַנִי בְעֵינָיו / וְהִרְבָּה עַד מְאֹד חִנָּם פְּצָעַי.

          נְתַתִּיהוּ כְּבַת עֵינִי וְשַׂמְתִּיו / כְּעֵץ שָׁתוּל עַל פַּלְגֵי דְמָעַי.

          וְעַל רִשְעוֹ נְתַתִּיהוּ בְלִבִּי / עֲדֵי שָׁכַן חֲרֵרִים בֵּין צְלָעַי.


וְעָשִׂיתִי לְאִישׁ הָיְתָה לוֹ שִׁפְחָה / וְהוּא וְאָבִיו הָיוּ מִתְעוֹלְלִים בָּהּ / זֶה הוֹלֵךְ וְזֶה בָּא:

          שֹׁמּוּ מְתֵי צֶדֶק וְיוֹשְׁבֵי שָׁעֲרָה / כִּי גָבְרָה זִמָּה וּפִיהָ פָעֲרָה.

          אֵיךְ מוֹסְדוֹת שַׁחַק לְזֹאת לֹא יִרְגְּזוּן / כִּי אִישׁ וְאָבִיו יֵלְכוּ לַנַּעֲרָה.


וְעָשִׂיתִי בְאִישׁ רָאִיתִי בְרֹאשׁוֹ מִצְנֶפֶת עֲצוּמָה / כְּפִי מָאתַיִם אַמָּה:

          וּמִצְנֶפֶת בְּרֹאשׁ נָבָל קְשׁוּרָה / חֲשַׁבְתִּיהָ כְלִי מָלֵא כְלִמָּה.

          מְאֹד גָּדְלָה וְהִיא עוֹמְדָה עֲלֵי רִיק / כְּמוֹ תֵבֵל תְּלוּיָה עַל בְּלִימָה.


וְעָשִׂיתִי עַל אִישׁ מְרוֹחַ אָשֶׁךְ / וְנָשָׂא אִשָּׁה יָפָה זִיוָהּ יָאִיר בַּחֹשֶׁךְ:

          חֲבֹק דּוֹדִי שְׁדֵי כַלָּה נְעִימָה / וְלֹא יֶחְסַר לְךָ תַּעֲנוּג וְכָל טוֹב.

          עֲלֵה עִמָּהּ עֲלֵי עֶרֶש אֲהָבִים / עֲדֵי תִהְיֶה כְגַן רָוֶה וְרָטֹב.

          רְאֵה בֵיצָתְךָ עִמָּהּ בְּרוּכָה / וְהִיא בֵיצָה אֲשֶׁר נוֹלְדָה בְיוֹם טוֹב.


וְעָשִׂיתִי עַל שֵׂיבָה. אֲשֶׁר בִּימֵי נְעוּרִים זָרְקָה בִי / וּכְנֶגַע פָּרְחָה בְלִבִּי:

          מוֹדֶה אֲנִי מַעֲשֵׂה מְתֵי עוֹלָם / הַלּוֹבְשִׁים לָבָן בְּיוֹם אֶבְלָם.

          כֵּן שַׂעֲרוֹתַי לָבְשׁוּ לָבָן / עַל נוֹד יְמֵי נֹעַר וְסוּר צִלָּם.


וְעוֹד בָּעִנְיָן הַזֶּה עָשִׂיתִי / וְלָלֶכֶת בְּדַרְכָּהּ נִסִּיתִי:

          דָּאוּ כְרֶגַע עוֹרְבֵי קַדְרוּת אֲשֶׁר / קֶדֶם בְּשַׁעַר שַׂעֲרִי שָׁקָדוּ.

          דָּאוּ וְתוֹךְ קִירוֹת לְבָבִי קִנְּנוּ / מִיּוֹם אֲשֶׁר מִשַּׂעֲרִי נָדָדוּ.


וְהֶעְתַּקְתִּי בִּסְפָרַד בִּלְשׁוֹן הַקֹּדֶשׁ סֵפֶר מוֹרֵה הַנְּבוּכִים / לְאֶחָד מִן הַנְּסִיכִים /

          וְהָיָה לִרְאוֹתוֹ כוֹסֵף / וּשְׁמוֹ יוֹסֵף / וּשְׁלַחְתִּיהוּ לוֹ עִם אֵלּוּ הַשִּׁירִים:

          שָׁלוֹם לְךָ עוֹרֵךְ קְרָב חָכְמָה / כִּי בָהּ זְמָמָיו יוֹם וְלֵיל נִצִּים.

          עַד קִנְּאוּ בוֹ יוֹצְאֵי צָבָא / וַיִּשְׂטְמוּהוּ בַעֲלֵי חִצִּים.

          אִם תֶּאֱהַב לַעְלֹז בְּגַן חָכְמָה / הִנֵּה עֲרוּגָה מָלְאָה צִצִּים.

          אוֹ לִבְּךָ תַקְרִיב לְאֵל קָרְבָּן / הִנֵּה לְךָ הָאֵשׁ וְהָעֵצִים.


וְעָשִׂיתִי בְעִיר כַּלְנֵה לְאִישׁ לָקַח הַבּוּז תְּרוּמִיָּה / וּשְׁמוֹ יְחִזְקִיָּה:

          אִמְרוּ לְרֹאשׁ הָעָם יְחִזְקִיָּה / יָחִיד בְּרָאָךְ רַב עֲלִילִיָּה.

          כִּמְעַט וְאָבְדוּ הַנְּבָלוֹת, אַךְ / לָהֶם שְׁלָחָךְ יוֹצְרָך מִחְיָה.

          אֵינְךָ יְחִזְקִיָּה, וְחָלִילָה / כִּי אִם מְנַשֶּׁה בֶּן יְחִזְקִיָּה.


וְעָשִׂיתִי בְּצָרְפַת לְאִישׁ חָנֵף מִכָּל עַמּוֹ / יִצְחָק הַלֵּוִי שְׁמוֹ:

          אֱמוּנָה אֵין בִּלְתָּהּ / בְּרוּם שַׁחַק שִׁבְתָּהּ / שְׁבִי נַפְשִׁי אִתָּהּ / וְעִמָּהּ הִלָּוִי

          יְקָר וָהוֹד יֵשׁ בָּהּ / וְהָאוֹת עַל טוּבָהּ / אֲשֶׁר אֵין חֵלֶק בָּהּ / לְיִצְחָק הַלֵּוִי.


וְעָשִׂיתִי בְּאַרְצוֹת מַעֲרָב לְאִישׁ טִפְסָר / לְלַמְּדוֹ דֶרֶךְ מוּסָר.

          דּוֹרֵשׁ קְנוֹת מוּסָר קְחָה כָל יוֹם רְאִי / מוּצָק לְהַבִּיט מַחֲזֵה צַלְמֶךָ.

          כִּי אִם יְהִי נָאֶה הֱיֵה נִשְׁמָר לְבַל / תַּחְסִיר בְּחֶסְרוֹן פָּעֳלָךְ תֻּמֶּךָ.

          אוֹ אִם יְהִי בוֹ מוּם הֱיֵה נִשְׁמָר לְבַל / מוּם חֶטְאֲךָ תּוֹסִיף עֲלֵי מוּמֶךָ.

          אִם תֵּט לְנִיבִי אוֹיְבֶיךָ תַחֲלשׁ / וּבְרֹב גְּאוֹנָךְ תַּהֲרֹס קָמֶיךָ.


וְעָשִׂיתִי בְּטוּלַיְטוּלָה עַל הַנָּשִׂיא ר׳ יוֹסֵף בֶּן שׁוֹשָׁן כְּשֶׁנִּפְטַר לְבֵית עוֹלָמוֹ:

          שָׁנָה אֲשֶׁר בָּהּ יְהוֹסֵף הַגְּבִיר נֶאֱסַף / תֹּאבַד הֲכִי אַחֲרָיו צֹאן אוֹבְדוֹת נֶהִי.

          הָיְתָה שְׁנַת אֵיד שְׁנַת זַעַם וְצָרָה שְׁנַת / יָגוֹן וְהָאֵל שְׁמָהּ קָרָא שְׁנַת נֶהִי.


וְקֹדֶם מוֹתוֹ בְּחֹדֶשׁ אֶחָד אוֹ שְׁנַיִם / שָׁמַעְנוּ כִּי יֵשׁ שֶׁבֶר בְּמִצְרַיִם

          וְכִי משֶׁה עָלָה לַשָּׁמַיִם. וָאַעַן וָאֹמַר:

          שֶׁבֶר מֵעַל שֶׁבֶר נִקְרָא / לִבּוֹת שׁוֹמְעָיו כָּאְרִי שׁוֹסֵף.

          כִּי אַחְרֵי מוֹת משֶׁה גָוַע / הַשַּׂר יוֹסֵף וַיֵּאָסֵף.

          לִשְׁכֹּן בִּמְחִיצָתוֹ בָא וַ/יִּקַח משֶׁה עַצְמוֹת יוֹסֵף.


וְעָשִׂיתִי בְּעִנְיְנֵי תֵבֵל / אֲשֶׁר הֵם נוֹבְלִים כְּצִיץ נוֹבֵל:

          וְרוֹאֶה בִלְבָבוֹ / זְמַן רַע וּנְתִיבוֹ / בְּנִקְרַת צוּר יָבֹא / וְנֶחְבָּא בִּמְחִלָה.

          וְתֵבֵל כַּמִּסְחָר / וְכָל אִישׁ בָּהּ יִתְחָר / וּמַה זֶּה בָּהּ אֶבְחַר / וְהִיא רָעָה כֻלָּה

          וְסֻכַּת טוֹבָתָהּ / אֲנִי פוֹסֵל אוֹתָהּ / לְמַעַן חַמָּתָהּ / מְרֻבָּה מִצִּלָּהּ.


וְעָשִׂיתִי לְאִישׁ קִדַּמְתִּיו בִּתְהִלָּתִי / וְלֹא הִכִּיר עֵרֶךְ גְּדֻלָּתִי:

          לְבָבִי קַח לְךָ מוּסָר וְשׁוּר כִּי / תְּהִלָּתְךָ נְבָלִים בּוֹסְסוּהָ.

          חֲשׂוֹף שִׁירָתְךָ לִפְנֵי חֲכָמִים / וְלֹא לִפְנֵי סְכָלִים יַחְמְסוּהָ,

          וְאִם תַּשְׁלִיךְ לְךָ אֶבֶן יְקָרָה / לְרַגְלֵי הַכְּלָבִים יִרְמְֶסוּהָ.


וְעָשִׂיתִי לְאַנְשֵׁי הַדּוֹר לִזְכּוֹר עִנְיְנֵיהֶם / וְרוֹעַ מַעֲשֵׂיהֶם:

          הָהּ כִּי בְנֵי הַדּוֹר אֲרָיוֹת טוֹרְפִים / כִּי לַעֲשׂוֹת נֶזֶק יְדֵיהֶם פּוֹשְׁטִין,

          לָכֵן שְׁמוֹר אֶת נַפְשְׁךָ מֵהֶם הֲכִי / כָּל הַיְּקוּם לִסְטִין וְהַכֹּל שׁוֹחֲטִין.


וְעָשִׂיתִי בְחֹדֶשׁ תַּעֲנִית יִשְׁמְעֵאלִים /

          כְּשֶׁשְּׁמַעְתִּים מַשְׁכִּימִים בַּעֲלוֹת הַשַּׁחַר

          וְאוֹכְלִים:

          הָאֵל עִנֵּג נַפְשׁוֹת עַמּוֹ / בַּמַּאְכָלִים. כִּי לֹא עִנָּה.

          לֹא יַרְעִיב אֵל נֶפֶשׁ צַדִּיק / רַק יַשְׂבִּיעַ נֶפֶשׁ נַעְנָה.

          כִּי לֹא צִוָּה עַנּוֹת אָדָם / נַפְשׁוֹ רַק אַחַת בַּשָּׁנָה.

          שָׁוְא לָכֶם כָּל מַשְׁכִּימֵי קוּם / כֵּן יִתֵּן לִידִידוֹ שֵׁנָא.


וְעָשִׂיתִי עַל אֶחָד מִן הַנְּבָלִים / גֵּאֶה בְהוֹנוֹ עַד כּוֹכְבֵי הַכְּסִילִים:

          בִּרְאוֹתְךָ נָבָל אֲשֶׁר / עָלָה וְשָׂם בִּזְבוּל דְּבִיר,

          פֶּתַע יְהִי נִשְׁבָּר וְשַׁח / כִּי אַחֲרֵי דַרְגָּא תְּבִיר.


וְעָשִׂיתִי עַל אִישׁ נָבָל וְכִילַי / וְאֵין בְּפִיהוּ רַק ׳לוּ׳ וְ׳אוּלַי׳ /

          וְיָמֻת בַּצָּמָא וְהוּא יוֹשֵׁב עַל אוּבַל אוּלַי:

          שְׁאֵלוּנִי עַל הַנְּדָבָה וּשְׁלֹמֹה / עֲנִיתִים הִיא וָהוּא כְּאֵשׁ עִם נוֹזְלֵי עָב

          לְשׁוֹן כּוֹסוֹ דָּבֵק לְחִכּוֹ בַּצָּמָא / וְשִׂפְתֵי שֻׁלְחָנוֹ עֲטוּפִים בָּרָעָב.

          לְהָסֵךְ מֵי רַגְלָיו יְפַחֵד פֶּן יִצְמָא / וְכֵן יִירָא לִבְדּוֹק נְקָבָיו פֶּן יִרְעָב.


וְעָשִׂיתִי עַל עִנְיַן הַדְּמָעוֹת / אֲשֶׁר מִן הַלֵּב נוֹבְעוֹת:

          כָּל הַדְּמָעוֹת יִזְּלוּ גַּם יַחְדְּלוּ / עִתִּים לְבַד דִּמְעוֹת עֲשׁוּקֵי אַהַב,

          כִּי אֵיד בְּכִי עֵת מִכְּבֵדָם יַעֲלֶה / יִהְיֶה לְמַיִם בַּעֲלוֹת הַלַּהַב.


וְעָשִׂיתִי עוֹד עַל הָעִנְיָן / וְלִשְׁאוֹב מִן הַמַּעְיָן:

          אִלּוּ כְּפִי אֵידִי דְמָעַי יִזְּלוּ / לֹא דָרְכָה רֶגֶל אֱנוֹשׁ יַבֶּשֶׁת.

          אָמְנָם דְּעוּ כִּי לֹא לְמֵי מַבּוּל לְבַד / כִּי גַּם לְדִמְעִי נִרְאֲתָה הַקֶּשֶׁת.


וְעָשִׂיתִי עַל הַנָּשִׂיא הַגָּדוֹל רַבֵּנוּ שֵׁשֶׁת ז״ל אֲשֶׁר הָיָה רֹאשׁ נְדִיבֵי עוֹלָם /

          וְאָסַף הַחֲמוּדוֹת כֻּלָּם:

          אִם תִּשְׁאֲלוּ עַל הַגְּבִיר שֵׁשֶּׁת

          / הוּא בַזְּמַן נִשְׁאַר לְרֹאשׁ פִּנָּה.

          נָגַע עֲדֵי זִקְנָה וְנִדְבָתוֹ / לֹא בָלְתָה. אַךְ יָסְפָה עֶדְנָה.


וְעָשִׂיתִי בְעִנְיַן הָאֲהוּבִים / הַנְּעִימִים וְהַנֶּאֱהָבִים:

          מֶרְחַב אֲרָצוֹת צַר וְלֹא / יָכִיל שְׁנַיִם אוֹיְבִים.

          כַּאֲשֶׁר מְקוֹם זֶרֶת מְאֹד / רָחָב לְאֶלֶף אוֹהֲבִים.


וְעָשִׂיתִי עַל קְהַל לוּנֵיל / יִשְׁמְרֵם הָאֵל:

          שְׂפַת יֶתֶר הֲתַשִּׂיג מַהֲלַל עָם / אֲשֶׁר שִׂכְלוֹ בְּיִרְאַת אֵל יְרִיחוֹ.

          וְשׁוֹלֵחַ דְּבַר חָכְמָה אֲלֵיהֶם / וְעוֹרֵךְ אֶל מְאוֹר שִׁמְשָׁם יְרֵחוֹ,

          וְשׁוֹלֵחַ רְסִיסֵי טַל לְיַם סוּף / וּמֵבִיא שַׁי תְמָרִים אֶל יְרִיחוֹ.


וּבִהְיוֹתִי בְלוּנִיל בִּקֵּשׁ מִמֶּנִּי ר' יְהוֹנָתָן הַכֹּהֵן ז"ל לְהַעְתִּיק לוֹ סֵדֶר זְרָעִים.

          לְר' משֶׁה אַבִּיר הָרוֹעִים. וְהֶעְתַּקְתִּיו וְאֵלָיו שְׁלַחְתִּיו עִם אֵלֶּה הַשִּׁירִים:

          יְצַו אֵל שְׁלוֹמִים / לְחֶבְרַת תְּמִימִים / חֲכָמִים נְעִימִים / נְבוֹנִים יְדוּעִים.

          חֲשֵׁכָה כְאוֹרָה / יְשִׁיבוּן בְּתוֹרָה / וְהִיא כַמְּנוֹרָה / וְהֵם לָהּ גְּבִיעִים.

          אֲקַדֵּם גְּבוֹהִים / וְאוֹרֵי נְגוֹהִים / וְכֹהֲנֵי אֱלֹהִים. בְּמִנְחַת זְרָעִים.


וְכָתַבְתִּי לְיוֹסֵף מְקוֹר הַתּוּשִׁיָּה / בֶּן הַשַּׂר הַמִּצְרִי ר' עוֹבַדְיָה ז"ל:

          הַשַּׂר, לְךָ חָכְמָה וּבִין יִתְיַחֲשׂוּ / וּבְתוֹךְ לְבָבָךְ נִמְצְאוּ יוֹם חֻפְּשׂוּ.

          וּבְיוֹם בְּרֹא הָאֵל חֲמוּדוֹת הַזְּמַן / וַיְשׁוֹטְטוּ לִמְצֹא בְּצֵל מִי יֶחֱסוּ

          אָמַר אֱלֹהִים: מַהֲרוּ וּלְכוּ אֱלֵי / יוֹסֵף אֲשֶׁר יֹאמַר אֲלֵיכֶם תַּעֲשׂוּ.


וּבְעִנְיַן הַחֵשֶׁק עָשִׂיתִי אֵלֶּה הַשִּׁירִים / בִּימֵי הַנְּעוּרִים:

          צְבִי הַחֵן אֲשֶׁר לֶחְיוֹ / בְּסַפִּירִים מְעֻלֶּפֶת,

          וְעֵינָיו שׁוֹלְפִים עָלַי / בְּרַק חֶרֶב מְעוֹפֶפֶת.

          שְׂפָתָיו צוּף וּפָנָיו גַּן / וְשִׁנָּיו דַּר וּפִיו נֹפֶת.

          בְּגַן לֶחְיוֹ מְשׂוֹשׂ עֵינִי / בְּעֵת צוֹפָה וְנִשְׁקֶפֶת.

          אֲבָל נַפְשִׁי בְּאֵשׁ חִשְׁקוֹ / צְרוּבָה הִיא וְנִשְׂרֶפֶת.

          שְׁתֵּי עֵינַי בְּגַן עֵדֶן / אֲבָל לִבִּי בְּתוֹךְ תֹּפֶת.


וְעָשִׂיתִי בְּעִיר כַּלְנֵה עַל אִישׁ אֲשֶׁר בָּרוּךְ שְׁמוֹ / וּנְבָלָה עִמּוֹ:

          נָבָל אֲשֶׁר קִנֵּא לְנַבְלוּת מוֹלְדוֹ / עַד בַּנְּבָלָה נִמְשְׁלוּ כַּבֵּן כְּהוֹר.

          נִקְרָא בְרָכוֹת אַךְ קְלָלוֹת הוּא שְׁמוֹ / כַּאְשֶׁר לְעִוֵּר יִקְרְאוּ סַגִּיא נְהוֹר.


וְעָשִׂיתִי בַעֲדִינָה עַל אִישׁ מִן הָאֲצִילִים / אֲשֶׁר הָיָה מִתְעַלֵּל בְּתַעֲלוּלִים:

          עֲדִינָה בָּהּ אֲנָשִׁים כַּבְּהֵמוֹת / וְנִרְבָּעִים לְפָרִיצִים נְבָלִים.

          וְהֵמָה בּוֹעֲלִים כִּי הַבְּהֵמָה / וְהַכֵּלִים כְּרַגְלֵי הַבְּעָלִים.



וְכָתַבְתִּי לְאִישׁ מֵאֲצִילֵי הַדּוֹר וּגְבִירָיו / לְמַעַן אֶחֱרוּ לִי פַעֲמֵי סְפָרָיו:

          שְׂאוּ שָׁלוֹם לְדוֹד אִטֵּר וְאִלֵּם / שְׁפַל רוּחַ וְעָלַי גַּאֲוָתוֹ.

          לְאִטֵּר מִכְּתֹב סֵפֶר לְדוֹדוֹ / וְאִלֵּם יַחֲשֹךְ לוֹ נִיב שְׂפָתוֹ.


וְעָשִיתִי לְאִישׁ נָשָׂא אִשָּׁה קְדוֹשָׁה / וְהָיָה מִתְחַבֵּר לְכָל אִשָּׁה קְדֵשָׁה:

          חוֹשֵׁק בְּנוֹת נֵכָר וְעוֹזֵב רַעֲיַת / חֵיקוֹ. עֲבֵרוֹתָיו גְּשׁוּ לִשְמֹעַ.

          עִבּוּר בְּנוֹת קֹדֶשׁ בְּזָרוֹת יַחֲלִיף / כִּי הַדְּחוּיוֹת יַחֲשֹׁב לִקְבֹּעַ.


וְעָשִׂיתִי בְּעִנְיַן דִּמְעוֹת הָעֵינַיִם / הַנִּגְרָשׁוֹת כִּיאוֹר מִצְרַיִם:

          זְמַן רָע הֶחֱשִׁיךְ עֵינַי יְגוֹנָךְ / וְרָמָה יָדְךָ בַּל יֶחֱזָיוּן.

          שְׂעִפַּי, עַד לְמָתַי הַיְגוֹנִים / בְּכַנְפֵיכֶם בְּכָל עֵת יֶחֱסָיוּן.

          וְשִׂמְחָתִי בְּבַת עֵינִי תְּמִיתוּן / לְרֹב בִּכְיִי – אֲבָל הַבַּת תְּחַיּוּן.


וְעָשִׂיתִי בְחֵרוּף הַנְּבָלִים / וְגִדּוּף הַנּוֹכְלִים:

          הֵם כַּבְּהֵמוֹת מִבְּלִי זָנָב אֲבָל / לָהֶם זְנָבוֹת רַק בְּעָרְפָּם נִקְשְׁרוּ.

          אוֹת עַל קְפוֹץ יָדָם בְּעֵת תַּחְפֹּץ רְאוֹת / שׁוּר מֵאֲחוֹרֵיהֶם יְדֵיהֶם אֻסְּרוּ

          אִם תֶּאֱסֹף מִנְיָן לְךָ מֵהֶם תְּפִ/לָּתָךְ אֲסוּרָה כִּי עֲשָׂרָה יַחְסְרוּ.


וְעָשִׂיתִי שִׁיר מוּסָר וְתוּשִׁיָּה. בְּעִנְיְנֵי הַמִּחְיָה:

          אַל תִּהְיֶה עָצֵל לְבַקֵּשׁ לָךְ / מִחְיָה וְשִׂים קִנָּךְ מְלוֹן אוֹרְחִים.

          כִּי לֹא תְהִי מוֹצֵא מְנוּחָה בִמְ/נוּחָה וּבַעְלִיּוֹת מְרֻוָּחִים.

          הַבֵּט לְשׁוֹן קֹדֶשׁ וְשׁוּר כִּי לֹא / יִתְקַבְּצוּ תוֹכוֹ שְׁנֵי נָחִים.


וְעָשִׂיתִי לְאִישׁ מַרְבֶּה לִקְנוֹת סְפָרִים / וְהוּא מִן הַכְּסִילִים וְהַבּוֹעֲרִים:

          רְאוּ פֶתִי אֲשֶׁר נוֹשַׁן בְּסִכְלוֹ / וְרָעוֹתָיו חֲדָשִׁים לַבְּקָרִים.

          וְהוּא מַרְבֶּה קְנוֹת סִפְרֵי תְעוּדוֹת / וְיִדְמֶה בָּם חֲמוֹר נוֹשֵׂא סְפָרִים.


וְעָשִׂיתִי עַל אִישׁ הִלַּלְתִּיו בְּשִׁירַי / וְהִכִּיר מְלִיצוֹת דְּבָרַי:

          הַשַּׂר, חֲלִיצוֹת מִמְּלִיצוֹתַי לְבַשׁ / יַעַן צְרִי צִירִי לְכָל צִיר יֶחֱבָשׁ.

          רוֹדֶה דְבַשׁ מִלַּי יְהִי מַכִּיר אֱמֶת / כִּי מִגְּוִיַּת הָאֲרִי רָדָה דְבָשׁ.


וְעָשִׂיתִי בְדַמֶּשֶׂק עַל הַדַּיָּן / לְסִבָּה וּלְעִנְיָן:

          אֱנוֹשׁ נָטַע בְּכַרְמוֹ עֵץ פְּרִי גַּם / עֲבָדָהוּ בְּיוֹם וָלֵיל וְהִשְׁקָה.

          וְעֵת הֵחֵל נְתֹן פִּרְיוֹ בְּעִתּוֹ / וְאֶל כָּל חֵךְ אֲכִילָתוֹ מְתוּקָה,

          נְתָשָׁהוּ וְתַם לָרִיק יְגִיעוֹ / וְכָכָה מַעֲשֵׂה ר׳ צְדָקָה.

          תְּמוֹל בָּנָה דְבִיר מוּסָר וְהַיּוֹם / הֲרָסָהוּ וְשָׁב שׁוּחָה עֲמֻקָּה.

          אֲקַו מִשְׁפָּט וְהֵן מִשְׂפָּח וּמַפָּח / וְלִצְדָקָה וְהִנֵּה קוֹל צְעָקָה.

          וְלָא יָדַע הֲכִי צוּר שָׂם בְּתוֹךְ פִּי / עֲלֵי כָל אוֹיְבֵי חֶרֶב מְרוּקָה.

          וְעֵת כִּי אֶשְׁלְפֶנָּה. הָעֲנָקִים / אֲנִי מוֹסֵר בְּזַעַם לַעֲלוּקָה.

          אֲנִי אַחְרִישׁ וְעֵת שִׁירִי יְדַבֵּר / זְמַן יַרְעִישׁ בְּיָדוֹ הַחֲזָקָה.

          וּמִי נִגַּשׁ לְמִלְחַמְתִּי וְנִמְלַט / וּמִי שָׁלַח בְּהוֹדִי יָד וְנִקָּה?


וְעָשִׂיתִי בְּמִצְרַיִם לַדַּיָּן ר' מְנַחֵם עַל עִנְיַן ר׳ הִלֵּל אֲשֶׁר בָּטַחְתִּי בוֹ. וְנִכְשַׁלְתִּי בְנִיבוֹ:

          קָרָאתִי בַּצַּר לִי פֶּסֶל / וָאֶבְטַח עַל עֶצֶב נִבְזֶה.

          הָיָה כֶסֶף סִיגִים וּצְרַפְ/תִּיהוּ אוּלַי סוֹדִי אֶחְזֶה.

          וָאַשְׁלִיכֵהוּ לִי בָּאֵשׁ / וַיֵּצֵא הָעֵגֶל הַזֶּה.


וּפַעַם אַחַת הֱצִיקוּנִי בְּלֵיל שַׁבָּת פַּרְעוֹשִׁים / נוֹשְׁכִים כִּנְחָשִׁים. וָאַעַן וָאֹמַר:

          לִבִּי לְפַרְעוֹשׁ חִלֵּל יְצוּעִי / וּרְווֹת בְּדָמִי כָּל תַּעֲנוּגוֹ.

          לֹא אֶשְׁקְטָה עַד אָשִׁיב גְּמוּלוֹ / וּבְיַד עֲווֹנוֹ כַּיּוֹם אֲמוּגוֹ.

          מוֹעֵד וְשַׁבָּת לֹא יַעְצְרֵנִי / לַהְפֹךְ לְאֵבֶל חָדְשׁוֹ וְחַגּוֹ.

          אָמְרוּ חֲבֵרַי: הֵן הַחֲכָמִים / אָסְרוּ בְּיוֹם הַשַּׁבָּת לְהָרְגוֹ.

          וָאוֹמְרָה: הֵם אָמְרוּ תְחִלָּה / הַבָּא לְהָרְגָךְ הַשְׁכֵּם לְהָרְגוֹ.


וְשָׁלַחְתִּי מִדַּמֶּשֶׂק לַחֲמָת אֶל הָרוֹפֵא ר׳ משֶׁה בַר צְדָקָה:

          גְּבִיר הַהוֹד, גְּדוֹר פִּרְצָךְ וְגִדְרָךְ / וְשׁוּב לִרְעוֹת שְׁאָר צֹאנָךְ וְעֶדְרָךְ.

          וְאַל תִּשְׁכַּח יְדִידִים לָךְ יְקַוּוּ / וְיוֹחִילוּ דְבַר פִּיךָ וְנִדְרָךְ.

          הֲכִי אַתָּה כְּרוּב מִמְשַׁח וְלָכֵן / לְעִירָךְ שׁוּב וְשִׂים הַדְרָךְ בְּחַדְרָךְ.


וְעָשִׂיתִי עַל הַנְּשָׁמָה. וְהַנֶּפֶשׁ הַחֲכָמָה:

          חוֹנָה וְנִסְגֶּרֶת בְּבוֹר גְּוִיָהּ / מִיּוֹם חֲנוֹתָהּ נִצְּלָה עֶדְיָהּ.

          רִמָּה הֲתֵחָבֵר אֱלֵי כִימָה / מִי יַעֲרִיךְ דָּפְיָהּ לְהוֹד יָפְיָהּ.

          יוֹם פָּשְׁטָה הַגּוּף אֲזַי לָבְשָׁה / עֶדְיָהּ וְהֵסִירָה כְּסוּת שִׁבְיָהּ.

          זוֹרְחָה כְּשֶׁמֶשׁ עָלְתָה מִבֵּית / כֶּלֶא נְוֵה חשֶׁךְ וּמַאְפֵּלְיָה.

          יוֹצְרָה בְּשַׁחַק הִלְּלָהּ כִּי כֵן / כֹּל הַנְּשָׁמָה הִיא תְהַלֵּל יָהּ.


וְעָשִׂיתִי עַל אֹרֶךְ יְמֵי הַגָּלוּת / וְרֹב שְׁנוֹת הַדַּלּוּת וְהַשִּׁפְלוּת:

          חָסִין גְּבוּרוֹת יָדְךָ גַלֵּם / בַּיּוֹם וְאֵיךְ תּוּכַל לְהִתְעַלֵּם.

          רָצוּץ בְּבוֹר גָּלוּת יְשׁוּעָתָךְ / יַחְלֹם כְּמוֹ עֶבֶד פְּדוּת חוֹלֵם.

          יִשְׁאַל בְּעַד הַקֵּץ וְאֵין עוֹנֶה / יִדְמֶה לְעִוֵּר יִשְׁאֲלָה אִלֵּם.

          זֶה לוֹ שְׁנוֹת אֶלֶף יְקַו אוּלַי / תִּפְקֹד עֲווֹן צָרָיו וְלֹא שָׁלֵם.

          יָהּ שׁוּב וְיִשְׂרָאֵל לְעִיר הַרְאֵל / תָּשִׁיב וְיָבֹא יַעֲקֹב שָׁלֵם.


וְעָשִׂיתִי בְּעִנְיַן בִּקּוּר בֵּית הַתְּפִלָּה / לַעֲרֹךְ לָאֵל תְּהִלָּה:

          חֲצַר בֵּית קָדְשְׁךָ קַמְתִּי לְבַקֵּר / לְחַוּוֹת נָעֳמָךְ עֶרֶב וּבֹקֶר.

          רְאֵה עָצְבִּי וְשׁוּר מַצְפּוּן לְבָבִי / אֲשֶׁר יִרְאֶה צְפוּנֵי כֹל וְחוֹקֵר.

          יְקַו לִבִּי וְכָל שִׂבְרוֹ בְךָ שָׂם / וְיֵדַע בֶּאֱמֶת כִּי לֹא תְשַׁקֵּר.

          זְרֵה אוֹיְבַי כְּמוֹ דָוֶה וְעֵינֵי / מְקַנְאַי מֵחֲזוֹת טוּבִי תְנַקֵּר.

          יְמִינָךְ שִׂים עֲלֵי גֶפֶן נְטַעְתָּהּ / וְאוֹתָהּ אַל תְּהִי נוֹתֵשׁ וְעוֹקֵר.


וְעָשִׂיתִי לְאִישׁ מִן הַנּוֹכְלִים / וְהוּא מִזֶּרַע אֲצִילִים:

          כִּילוּת בְּתֵבֵל שׁוֹטְטָה לֹא שָׁקְטָה / עַד כִּי בְּלִבָּךְ מָצְאָה מַרְגּוֹעַ,

          לוּ נָגְעָה יָדָךְ בְּעָבִים נִפְזְרוּ / אוֹ בַּעֲיָנוֹת מֵאֲנוּ לִנְבֹּעַ,

          אוֹ בַעֲשָׂבִים יָבְשׁוּ עַד נִשְׂרְפוּ / אוֹ עַל דְּגֵי יָם תַּמְנוּ לִגְוֹעַ,

          אוֹ דָרְכָה רַגְלָךְ בְּחֶרְמוֹן קִנְּאוּ / עַל רוֹב יְבוֹשֶׁת הַרֲרֵי גִלְבּוֹעַ.


וְעָשִׂיתִי לְאִישׁ מִן הַנְּדִיבִים / וּמִגֶּזַע הַחֲשׁוּבִים:

          אָמַר אֱלֹהִים לַנְּדָבָה שַׁחֲרִי / הַשַּׂר לְמַעַן בָּךְ מְאֹד רַב גִּילוֹ

          עִזְבִי בְנֵי תֵבֵל וְלוּ בִלְבַד לְכִי / כִּי הוּא אֲדוֹנַיִךְ וְהִשְׁתַּחֲוִי לוֹ.


וְעָשִׂיתִי עַל אִישִׁים יַחֲלִיקוּ לְשׁוֹנָם / וְקֶבֶר פָּתוּחַ גְּרוֹנָם:

          וְיֵשׁ אִישִׁים מְתוּקֵי נִיב וְשָׂפָה / וּבִלְשׁוֹנָם יְפַתּוּנִי וְנִיבָם.

          יְגַלּוּ לִי פְנֵי רָצוֹן וְאַהְבָה / וְיָדַעְתִּי הֲכִי אָרְבָּם בְּלִבָּם.

          וְאֵיך יוּכְלוּן צְפֹן רוֹעַ לְבָבָם / וְהַכָּרַת פְּנֵיהֶם עָנְתָה בָם.


וְעָשִׂיתִי עַל עִנְיַן הַנְּדוֹד / הַמַּבְעִיר בְּלִבִּי כִידוֹד:

          יְדִידַי יִדּוֹדוּן וְיָרֵעוּ רֵעַי / וְלִנְדוֹדָם אֶמְצָא יְקוֹד אֵשׁ בִּצְלָעַי.

          וּמִי יִתֵּן קָרְבוּ חֲשׁוּקַי כִּמְצוּקַי / וְדוֹדַי כִּנְדוֹדַי וְרֵעַי כִּדְמָעַי.


וְעָשִׂיתִי עַל עִנְיַן הַפְּרִידָה / אֲשֶׁר בָּהּ רַגְלִי לְכוּדָה:

          לְמָתַי נָע וְנָד אֶהְיֶה כְקַיִן / הֲלוּם יָגוֹן וְלֹא שֵׁכָר וְיַיִן.

          אֲקַו יֶשַׁע וְלֹא אֶחֱזֶה וְאוּלָם / מְצוּקִים אֶחֱזֶה עַיִן בְּעַיִן.

          כְּמוֹ עִוֵּר בְּמַחְשַׁכֵּי כְּלָאִים / חֲצוֹת לֵיל אֲקַו לָאוֹר וְאַיִן.


וְעָשִׂיתִי עַל אֲנָשִׁים גֵּאִים / וְהֵם רָעִים מִכָּל הַבְּרוּאִים:

          יַכְרֵת אֱלֹהִים בַּעֲלֵי גָאוֹן / נִבְזִים וְעַד כּוֹכְבֵי זְבוּל גֵּאִים.

          לוֹבְשִׁים בְּגָדִים הֵם חֲדָשִׁים אַךְ / הַמַּעֲשִׂים בָּלִים מְטֻלָּאִים.

          חוֹטְאִים וְחָלִילָה הֱיוֹת כָּהֶם / אַנְשֵׁי סְדוֹם רָעִים וְחַטָּאִים.


וּבְעִנְיַן הַחֵשֶׁק עָשִׂיתִי / וּבֵית אַהֲבָה בָנִיתִי:

          צְבִי הַחֵן מְחָצַנִי בְּיָפְיוֹ / וְהִבְעִיר בִּי יְקוֹד חִשְׁקוֹ וְעָצְבּוֹ.

          וּבַמִדָּה אֲשֶׁר מָדַד גְּמַלְתִּיו / וְהֶאְכַלְתִּיו פְּרִי חֶטְאוֹ וְחוֹבוֹ.

          בְּהַרְאוֹתִי רְאִי מוּצָק לְעֵינָיו / הֲשִׁיבוֹתִי יְקוֹד חִשְׁקִי לְלִבּוֹ.


וְעַל זֶה הָעִנְיָן. בָּנִיתִי לוֹ בִּנְיָן:

          עֵינֵי צְבִיָּה דָרְכוּ לִי קֶשֶׁת / עַל כֵּן שְׂעִפַּי יָרְאוּ מִגֶּשֶת.

          נַפְשִׁי כְמוֹ צִפּוֹר בְּרִשְׁתָּהּ נִלְכְּדָה / אַךְ אַהֲבָתָהּ הָיְתָה הָרֶשֶׁת.

          עֵינָהּ דְּמוּת קֶשֶׁת אֲבָל פָּרְשָׂה / עָנָן וְעָלָיו נִרְאֲתָה הַקֶּשֶׁת.


          וְעוֹד בְּעִנְיַן הַחֵשֶׁק / בִּצְבִי הַלְּבָבוֹת עוֹשֵׁק:

          צְבִי הַחֵן עֲשָׁקַנִי בְּיָפְיוֹ / וְהִשְׁבִּית כָּל רְנָנַי גַּם שְׂשׂוֹנַי,

          וְאֵין לִי מִלְּחָיָיו הַנְּעִימִים / לְבַד הָאוֹר אֲשֶׁר אֶרְאֶה בְעֵינַי.

          יְרִיבוּנִי עֲלֵי חִשְׁקוֹ מְרִיבִים / לְהוֹסִיף אֵיד וְיָגוֹן עַל יְגוֹנַי,

          וְנִשְׁבַּעְתִּי לְבַל אֶשְׁמַע בְּקוֹלָם / אֲבָל כִּי אֶסְגְּרָה עֵינַי וְאָזְנַי.

          וְהוּא נַפְשִׁי וְאֵיךְ אֶמְאַס בְּנַפְשִׁי / הֲלֹא אֶחְטָא לְצוּר שׁוֹכֵן מְעוֹנַי.

          וּמָה אֶקֹּב וְלֹא קַבֹּה אֱלֹהִים / וּמָה אֶזְעוֹם וְלֹא זָעַם יְיָ?


וְעוֹד עַל הָעִנְיָן / הֲכִינוֹתִי זֶה הַבִּנְיָן:

          סָבַב חֹשֵׁק עַל יַד הַיְאוֹר / וּכְאֵב חִשְקוֹ פִצְעֵי זַיִן.

          וַיַּרְא עַלְמָה הִיא רוֹחֶצֶת / מַלְבּוּשׁ דּוֹדוֹ עַל הָעַיִן.

          וּנְשָׁקָהוּ לֹא לִכְבוֹדוֹ / רַק לִכְבוֹד חַכְלִילִי עַיִן.

          כֵּן יֵשׁ אָדָם סוֹבֵא יִשְׁתֶּה / הַחֹמֶץ לִכְבוֹד הַיַּיִן.


וְעָשִׂיתִי עַל נְשִׂיאִים הֵם / נְדִיבִים בְּפִיהֶם / וְקוֹפְצִים יְדֵיהֶם:

          הָהּ רָאֲתָה עֵינִי אֲנָשִׁים הֵם כְּמוֹ / הַכַּרְמְלִי נָבָל וְכַאְחֵי קַיִן.

          וּשְׁמָם נְשִׂיאִים אַךְ בְּנִדְבָתָם נְשִׂי/אִים הֵם וְגַם רוּחַ וְגֶשֶׁם אָיִן.


וְעָשִׂיתִי בְּאַשּׁוּר עַל אִשָּׁה רָעָה וְשַׁלֶּטֶת / וְהִיא כָאֵשׁ מִתְלַהֶטֶת /

          וְלָהּ בֵּן פָּרִיץ דּוֹמֶה לְאִמּוֹ / בְּכַעְסוֹ וּבְזַעְמוֹ:

          אֵשֶׁת כְּסִילוּת צוּר בְּשַׁחַת יִצְפְּנָהּ / חִישׁ יִתְּנֶנָּה תוֹךְ שְׁאוֹל וַיִּשְׁכְּנָהּ.

          תִּדְמֶה עֲתַלְיָה רַק בְּנָהּ יִדְמֶה אֲחַזְ/יָהּ. בּוֹרְאָהּ לֹא יַעְזְרָהּ וִיְחוֹנְנָהּ.

          שָׁרְשָׁהּ יְקַצֵּץ עִם וְלָדָהּ וַעֲתַלְ/יָה אֵם אֲחַזְיָה רָאֲתָה כִּי מֵת בְּנָהּ.


וְעָשִׂיתִי עַל אִישׁ מִן הַנְּבָלִים שְׁמוֹ [צְדָקָה בֶּן עֲתִירָה] / כְּשֶׁמֵּתָה אִמּוֹ וְנִפְטְרָה:

          וְאָמְרוּ לִי סְפוֹד עַל אֵם צְדָקָה / עֲנִיתִים לֹא עֲדֵי יִהְיֶה כְּאִמּוֹ.

          אֲקוֹנֵן לוֹ בְּעֵת יִהְיֶה שְׁכֵנָהּ / וְיָשִׂים מוּל מְקוֹמָהּ אֶת מְקוֹמוֹ.

          וְאוֹתוֹ תִּהְיֶה נוֹשֵׂאת בְּקִבְרָהּ / לְאַט כַּאֲשֶׁר נְשָׂאָהּ הוּא בְּשִׁכְמוֹ.


וְעָשִׂיתִי בְּאַשּׁוּר עַל עִנְיַן הַקְּהִלוֹת אֲשֶׁר חָזִיתִי / וּבְאַרְצוֹת מִזְרָח רָאִיתִי:

          שִׁמְעוּ בְנֵי שֵׂכֶל פְּלִילִיּוֹת / בָּם תִּלְמְדוּ דֵעוֹת וְתוּשִׁיּוֹת.

          בָּאתִי לְנֹא אָמוֹן וְחָנִיתִי / בֵּינָם כְּמוֹ הַדָּג בְּתוֹךְ צִיּוֹת.

          וָאֶמְצְאָה אִישִׁים בְּמִצְרַיִם / אֵין בָּם לְאַנְשֵׁי הַיְּקָר מִחְיוֹת.

          וָאֶעֱבֹר מִשָּׁם לְדַמֶּשֶׂק / אֶל עַם אֲשֶׁר תָּעוּ כְמוֹ שֵׂיוֹת.

          וָאֶשְׁכְּחָה אֵידָם בְּעַם צוֹבָה / כִּי הֵם כְּחֶרֶב לָהּ שְׁתֵּי פֵיוֹת.

          וָאֶמְאֲסָה כַלְנֵה הֲכִי לַעְצוֹר / הוֹנָם יְדֵיהֶם תּוֹקְפוֹת הוֹיוֹת.

          אוּלָם בְּאַשׁוּר נִשְׁכְּחוּ כֻלָּם / כִּי הֵם בְּהֵמוֹת אוֹ כְמוֹ חַיּוֹת.

          אִם אֵרְדָה כֵן כָּל אֲשֶׁר אֵלֵךְ / אֶמְצָא בְנֵי תֵבֵל בְּתַחְתִּיּוֹת.


וְעָשִׂיתִי לְאִישׁ נֶאֱמָן חֲשַׁבְתִּיו / וְנֶהְפַּךְ לְאוֹיֵב בְּעֵת בְּחַנְתִּיו:

          יְדִידוּתָךְ חֲשַׁבְתִּיהָ כְמוֹ מָן / בְּצִנְצֶנֶת יְקָר נִצְפַּן וְנִכְמָס.

          בְּחַנְתִּיהָ וְהַמִּבְחָן כְּשֶׁמֶשׁ / אֲשֶׁר יָצָא וְחַם עָלָיו וְנָמָס.


וְעָשִׂיתִי בִימֵי הַנֹּעַר / בְּעִנְיַן רֹב הַשַּׂעַר:

          אִישִׁים יְרִיבוּנִי עַל רֹב שַׂעֲרִי / אָמְנָם כְּסִילוּתָם שְׁחִין פּוֹרֵחַ.

          לֹא זָכְרוּ כִּי לוּ אֱלִישָׁע כֵּן יְהִי / לֹא קָרְאוּ אַחֲרָיו עֲלֵה קֵרֵחַ.


וְעָשִׂיתִי עַל צִפּוֹר נִלְכַּד. בְּפַח צַיָּד:

          רְאוּ צִפּוֹר אֲשֶׁר נָפַל בְּתוֹךְ פַּח / וְלֹא יוֹעִיל בְּעֵת צַר לוֹ כְנָפוֹ.

          חֲשַׁבְתִּיהוּ כְּבֶן אָדָם בְּעוֹלָם / וְלֹא יוֹעִיל כְּבוֹדוֹ לוֹ בְסוֹפוֹ.

          וְעָשִׂיתִי עַל חֶרֶב מְרוּטָא / לְטֶבַח מְעֻטָּה.

          חֶרֶב בְּיַד בַּעְלֵי קְרָב תָעוּף / וּכְמוֹ לְשׁוֹן אֵשׁ תִּהֲלֹךְ תּוֹכָם.

          רֵעַי שְׁאֵלוּנִי לְמִי תִדְמֶה / וָאוֹמְרָה תִּדְמֶה לְשׁוֹן חָכָם.


וְעָשִׂיתִי לְלַמֵּד לִבְנֵי אִישׁ דַּרְכֵי מוּסָר / וּלְהִתְנַהֵג בַּחֲסִידוּת עִם כָּל בָּשָׂר:

          אַתָּה תְכַבֵּד מַעֲלָתֶךָ / כִּי מַעֲלָתְךָ לֹא תְכַבֶּדְךָ.

          כַּאֲשֶׁר תְּהִי עוֹשֶׂה תְּהִי מוֹצֵא / וּשְׁאַל לְאָבִיךָ וְיַגֵּדְךָ.

          אִם תַּעֲשֶׂה טוֹבָה תְּרוֹמֶמְךָ / אוֹ תַעֲשֶׂה רָעָה תְּאַבֶּדְךָ.


וְעָשִׂיתִי עַל אִישׁ נָבָל קוֹפֵץ יָדָיו. בְּנִדְבוֹתָיו וּבַחֲסָדָיו:

          אִם אִישׁ נָבָל יִצְבֹּר עשֶׁר / אֶל אוֹיְבָיו חִישׁ יוֹרִישֶׁנּוּ.

          חַיִל בָּלַע וַיְקִיאֶנּוּ / אֵל מִבִּטְנוֹ יוֹרִישֶׁנּוּ.


וְעָשִׂיתִי עַל מוֹנֵעַ טוּבוֹ / מִן הַבּוֹטְחִים בּוֹ:

          אִם יִצְבֹּר עשֶׁר אִישׁ כִּילַי / וִיהִי עֵץ שָׁתוּל עַל יוּבַל.

          בַּחְצִי יָמָיו יַעַזְבֶנּוּ / וּבְאַחְרִיתוֹ יִהְיֶה נָבָל.


וְעָשִׂיתִי עַל אִישׁ גְּבַהּ לֵבָב / וְהוֹלֵךְ שׁוֹבָב:

          בִּרְאוֹתָךְ אִישׁ אֵין בּוֹ ישֶׁר / רַק בָּעשֶׁר נַפְשׁוֹ יָקְרָה.

          אַל תָּשִׁית לֵב אֶל מַעְלָתוֹ / כִּי מַעְלָתוֹ מַעְלַת גֵּרָה.


וְעָשִׂיתִי עַל מְנוֹרוֹת / בָּהֶם שְׁנֵי נֵרוֹת / וְאוֹרוֹתָם נִפְרָדוֹת וּמְחֻבָּרוֹת:

          עוּרָה יְדִידִי וְאַל תִּישַׁן וְקוּמָה שְׂמַח / בִּשְׁנֵי מְאוֹרוֹת בְּלַיִל זֶה בְזֶה דָבְקוּ.

          קוּמָה רְאֵה נֵר אֲשֶׁר אוֹרוֹ כְּלֶחְיָךְ וְכִצְ/לָעַי פְּתִילָיו וְכֵן דַּלּוּ וְכֵן דָּלְקוּ.

          פַּעַם יְנוּדוּן בְּלִי רָצוֹן וְיִפָּרְדוּ / רֶגַע וּפַעַם יְרוּצוּן חִישׁ וְיֵאָבְקוּ.

          יִדְמוּ לְחוֹשֵׁק וְחָשׁוּק רָחֲקוּ בַנְּדוֹד / וּבְעֵת אֲשֶׁר נִפְגְּשׁוּ רָצוּ וְהִתְחַבְּקוּ.


וְעָשִׂיתִי עַל הַנָּשִׂיא ר׳ מָכִיר / יוֹצְרוֹ אוֹתוֹ לְטוֹב יַכִּיר:

          יוֹם נִבְרֵאתָ אָמַר הָאֵל / הֵן עַבְדִּי אֶתְמָךְ בּוֹ לָעַד.

          בִּדְבַר פִּיהוּ אָפִיק מַרְפֵּא / וּבְמַאְמָרָיו לִבּוֹת אֶסְעָד.

          וּפְרִי עֵדֶן וּצְרִי חַיִּים / בּוֹ לִדְוֵה לֵב יִהְיֶה נוֹעָד.

          כָּל אִישׁ לָקַח חֶלְקוֹ וּלְמָ/כִיר נָתַתִּי אֶת הַגִּילְעַד.


וְעָשִׂיתִי עַל אִישׁ רָאִיתִי בְּבֵית הַכְּנֶסֶת יָרוֹק יָרַק / וְרֻקּוֹ בִּפְנֵי רֵעָיו זָרַק:

          רָאוּי עַל כָּל בָּאֵי קֹדֶשׁ / מֵימֵי טֹהַר עָלָיו לִזְרֹק.

          בִּגְלַל נִבְזִים בּוֹזִים אוֹתוֹ / כִּי מִפָּנָיו לֹא חָשְׂכוּ רוֹק.


וְעָשִׂיתִי עַל סֵפֶר הַמַּדָּע / עֵת בְּאֶרֶץ סְפָרַד נִגְלָה וְנוֹדַע:

          סֵפֶר כְּעֵץ שָׁתוּל בְּמֵי מַדָּע וּבִין / כִּי בוֹ חֲפָצָיו יִמְצְאָה כָל דּוֹרֵשׁ

          סִפְרֵי גְאוֹנֵינוּ וְאִם רַבּוּ מְאֹד / הֵמָּה עֲנָפִים לוֹ וְהוּא הַשֹּׁרֶשׁ.


וְעָשִׂיתִי עַל אִישׁ בִּירוּשָׁלַיִם יָצְאוּ עָלָיו דִּבּוֹת / וְרָעוֹת רַבּוֹת:

          הָפְקַד בְּהַר צִיּוֹן אֱנוֹשׁ טָמֵא לְרֹאשׁ / קֹדֶשׁ וְלוֹ נֶגַע אֲשֶׁר פָּשָׂה בְעוֹר.

          יִפְעַר לְכָל עוֹבֵר וְשָׁב עַצְמוֹ לְזֹאת / כָּל יוֹדְעָיו קָרְאוּ שְׁמוֹ רֹאשׁ הַפְּעוֹר.


וְעָשִׂיתִי עַל אִישׁ עוֹזֵב אִשְׁתּוֹ / בְּעַד שִׁפְחָתוֹ / וְיָלְדָה לוֹ בַת דּוֹמָה לְצוּרָתוֹ / וְהָיָה שְׁמוֹ:

          מִדּוֹת אֱנוֹשׁ חָנֵף בְּלִיַּעַל / הִתְקַבְּצוּ שִׁמְעוּ בְנֵי יַעֲקֹב.

          חָשַׁק בְּשִׁפְחָתוֹ מְקוֹם אִשְׁתּוֹ / וַיַּהֲפֹךְ מִישׁוֹר אֱלֵי עָקוֹב.

          עַד יָלְדָה לוֹ בַת אֲשֶׁר דָּמְתָה / אֵלָיו בְּמֵצַח עַז וְלֵב עָקוֹב.

          נִשְׁבַּע שְׁמוֹ הָרַע לְבַל יַעְזֹב / אוֹתָהּ הֲכִי חָפֵץ בְּבַת יַעֲקֹב.


וְעָשִׂיתִי עַל אִישׁ הוֹלֵךְ בְּמֵישָׁרִים / וּמַרְבֶּה חֲסָדִים בְּמִסְתָּרִים:

          יְרוֹצֵץ בְּיוֹמוֹ עֲשׂוֹת טוֹב וְלֵילוֹ / לְשַׁחֵר פְּנֵי אֵל יְעוֹרֵר שְׁחָרִים.

          וְיַרְבֶּה חֲסָדָיו בְּסֵתֶר לְדַעְתּוֹ / אֱמֶת כִּי בְיַד אֵל מְגִלַּת סְפָרִים.


וְאֵלֶּה עָשִׂיתִי בִימֵי נַעֲרוּתִי / כְּשֶׁנִּפְטְרָה הוֹרָתִי ת"מ

          לְבָבִי, אֵיךְ זְמַן פָּרַץ גְּדֵרָךְ / וְהֵסִיר מֵעֲלֵי רֹאשָׁךְ פְּאֵרָךְ!

          בְּמוֹת אֵם הַיְּקָר תַּמָּה כְבוּדָה / אֲשֶׁר הָיְתָה כְּמוֹ שֶׁמֶשׁ בְּדוֹרָךְ.

          תְּאֻשַּׁר בֵּין תְּמִימוֹת לַעֲדֵי עַד / וּמִנָּשִׁים בְּעַד אֹהֶל תְּבֹרָךְ.


וְאֵלֶּה עָשִׁיתִי בְּאִישׁ כְּסִיל / מִן הַכְּסִילוּת אַלְפַּיִם בַּת יָכִיל:

          דּוֹדִי נִקַּדְתִּי לָךְ שִׁירִי / יַעַן שִׂכְלָךְ נֶחְשַׁל חָשׁוֹל.

          וּנְתִיבוֹתָיו לָךְ פִּנִּיתִי / אוּלַי לִבָּךְ עָלָיו יִמְשֹׁל.

          כִּי זָכַרְתִּי מִצְוַת אֵל לִפְ/נֵי עִוֵּר לֹא תִתֵּן מִכְשׁוֹל.


וְאֵלֶּה עָשִׂיתִי לְאִישׁ חֲסַר דַּעַת / וְהוּא כָבֵד בַּעַל נֶפֶשׁ מִשְׁתַּגַּעַת:

          חֲסַר דַּעַת וְקַל רוּחַ בְּדַעְתּוֹ / וְהוּא כָבֵד כְּמוֹ תָבוֹר בֶּהָרִים.

          וְלוּ נָפְלָה כְבֵדָתוֹ בְיַם סוּף / אֲזַי שָׁקַע וְשָׁב אֶרֶץ חֲרֵרִים.

          וְלוּ הָיָה בְרַגְלָיו קַל כְּרֹאשׁוֹ / יְהִי מַשִּׂיג בְּרוּצוֹ הָעֳפָרִים.


וְאֵלֶּה עָשִיתִי לְאִישׁ עוֹשֶׂה תַּפּוּחֵי כֶסֶף וַעֲטָרָה / לְסֵפֶר תּוֹרָה:

          שׁוּר תַפּוּחֵי כֶסֶף יִדְמוּ / טַבָּעוֹת עַל בַּדֵּי אָרוֹן.

          מִן הַקֹּדֶשׁ נִשְׁמַע קוֹלָם / כַּפַּעְמוֹנִים בִּמְעִיל אַהְרֹן.

          כָּזֹאת נִדְבַת הַשַּׂר יוֹסֵף / כָּל הַמַּשְׂכִּיל אוֹתָהּ יָרוֹן.

          יוֹדוּ חַסְדּוֹ אֶל כָּל יָבֹא / לִבְנֵי יִשְׂרָאֵל זִכָּרוֹן.


וְאֵלֶּה עָשִׁיתִי עַל הַיַּיִן / אֲשֶׁר הוּא מַחְמַד כָּל עַיִן:

          וּבֶן שׂוֹרֵק חֲשַׁבְתִּיו נֶאֱמָן לִי / וְהוּא יַפְרִישׁ כְּמוֹ נָחָשׁ וְיַשִּׂיא.

          אֲהַבְתִּיהוּ וָּבַגד בִּי בְעָרְמָה / וְיִמְשֹׁךְ אֶת לְבָבִי לוֹ וְנַפְשִׁי.

          אֲנִי שַׂמְתִּיו הֱיוֹת שָׁמוּר בְּקִרְבִּי / וְהוּא עָלָה הֱיוֹת שׁוֹמֵר לְרֹאשִׁי.


וְאֵלֶּה עָשִׂיתִי לְהִתְעַלֵּס בַּגַּנִּים / וְלִרְווֹת בַּת הַגְּפָנִים:

          דּוֹדִי שְׁתֵה עִמִּי בְּצֵל גִּנָּה / אָשִׂים הֲדַס נֶגְדָּךְ וְשׁוֹשַׁנָּה.

          אֶשְׂמַח בְּבַת כֶּרֶם אֲשֶׁר גִּילָהּ / חָדָשׁ, וְהִיא תִּבְלֶה וְנוֹשָׁנָה.

          הֵן אִם בְּעֵין הָאֵל כְּמוֹ אֶלֶף / שָׁנִים, וְהַשָּׁנָה כְּמוֹ עוֹנָה.

          מִי יִתְּנֵנִי אֶהְיֶה בַכּוֹס / עַד כִּי בְעֵין הָאֵל יְהִי שָנָה.


וְעוֹד עָשִׂיתִי בְעִנְיַן הַתְּשׁוּקָה / וְהַחֶבְרָה הַחֲשׁוּקָה:

          יַעְלָה הֱבִיאַתְנִי לְבֵית אַהְבָה / בָּרָה וְתַמָּה כַּאֲבִיגַיִל.

          וּבְסוּר צְעִיפָהּ גִּלְּתָה תֹאַר / הֶחְפִּיר בְּיָפְיוֹ בַת אֲבִיחַיִל.

          זָרַח בְּלֵיל אוֹרָהּ וְחָרְדוּ כֹּל / וַיִּרְקְדוּ הָרִים כְּמוֹ אַיִל.

          שַׂחְתִּי אֲהָהּ נִגְלוּ צְפוּנֵינוּ / וַתַּחֲשׂף יָדָהּ כְּבַת חַיִל.

          וַתַּהֲפֹךְ עָלַי עֲנַן שֵׂעָר / שָׁחוֹר – וְהַיּוֹם הָפְכָה לַיִל.


וְאִישׁ מִבְּנֵי עֲדִינָה עָשָׂה שִׁירֵי זִמָּה / מְלֵאִים טֻמְאָה:

          לוּ שָׁר בְּנוֹ עַמְרָם פְּנֵי דוֹדִי / מִתְאַדְּמִים הָעֵת שְׁתוֹת שֵׁכָר,

          וּלְפִי קְוֻצּוֹתָיו וְהוֹד יָפְיוֹ / לֹא חַק בְּתוֹרָתוֹ וְאֶת זָכָר.


וּכְשָׁמְעִי שִׁירוֹ קִבַּצְתִּי עֲשָׂרָה אֲנָשִׁים בַּעֲלֵי שֵׂכֶל וּתְבוּנָה / וְישֶׁר וֶאֱמוּנָה /

          וְהֶחֱרַמְנוּהוּ בְּשִׁירִים כְּשִׁירָיו / עַל קַו חֲרוּזָיו וּדְבָרָיו.


וְאָמַרְתִּי בַּתְּחִלָּה:

          שַׁדַּי שְׁלֹף חֶרֶב נְקָמָה עַל / טָמֵא אֲשֶׁר תּוֹרָתְךָ עָכָר.

          הַפֶּה אֲשֶׁר נִפְתַּח לְחָרֵף אֵל / חִישׁ בַּחֲרָבוֹת יִהְיֶה נִסְכָּר.

          מוֹכֵר קְדֻשַּׁת אֵל בְּעַד טֻמְאָה / מַהֵר לְיַד הוֹרֵג יְהִי נִמְכָּר.


וַיַּעַן הַשֵּׁנִי וַיֹּאמַר:

          אָנָּא אֲדוֹן עוֹלָם שְׁפוֹךְ זַעְמָךְ / עַל אִישׁ בְּיֵין זִמָּה וְחֵטְא שָׁכָר.

          הַמְטֵר מְטַר גָּפְרִית וְאֵשׁ עָלָיו / עַד כִּי בְיַד מָוֶת יְהִי נֶעְכָּר.

          רוּחוֹ לְאֵשׁ תֹּפֶת שְׁלַח מִנְחָה / וּלְפִי שְׁאוֹל גּוּפוֹ שְׁלַח אֶשְׁכָּר.


וַיַּעַן הַשְּׁלִישִׁי וַיֹּאמַר:

          שַׁדַּי מְכֹר מַהֵר לְיַד מָוֶת / הָאִישׁ אֲשֶׁר דָּתָךְ בְּחֵטְא מָכַר.

          תּוֹרָתְךָ תִּצְעַק בְּפִיו תָּמִיד / אוֹתִי פְּצֵה מִיַּד בְּנֵי נֵכָר.

          יוֹם יַעֲלֶה זִכְרוֹן בְּנֵי תֵבֵל / לָטוֹב, יְהִי לָרַע שְׁמוֹ נִזְכָּר.


וַיַּעַן הָרְבִיעִי וַיֹּאמַר:

          הוֹי עַל אֲנָשִׁים עוֹזְבִים תּוֹרַת / הָאֵל וְהֵם עוֹבְדִים לְאֵל נֵכָר.

          כִּי כָל גְּדוֹלֵיהֶם מְטֻנָּפִים / בִּמְקוֹם נְקֵבָה בּוֹעֲלִים זָכָר.

          בַּאֲשֶׁר בְּיֵין חֵטְא שָׁכְרוּ, חֶרֶב / אוֹיְבִים בְּדָמָם יוֹם וְלֵיל תִּשְׁכַּר.


וַיַּעַן הַחֲמִישִׁי וַיֹּאמַר:

          עוּרָה בְּנוֹ עַמְרָם רְאֵה דָתָךְ / כִּי חִלְּלוּהָ סוֹבְאֵי שֵׁכָר.

          יִתְעַלְּלוּ תָמִיד בְּתַעְלוּלִים / וְיִתְּנוּ נָדָן לְכָל זָכָר.

          צוּר יַעֲקֹר שָׁרְשָׁם וּבִבְשׂוֹרַת / מוֹתָם לְכָל אֶרֶץ יְשֻׁלַּח כָּר.


וַיַּעַן הַשִּׁשִּׁי וַיֹּאמַר:

          הָאִישׁ אֲשֶׁר תּוֹרַת בְּנוֹ עַמְרָם / חִלֵּל בְּנַבְלוּת פִּיו אֲשֶׁר זָכַר.

          פּוֹעֵר לְכָל עוֹבֵר וָשָׁב עַצְמוֹ / נִרְבָּע עֲלֵי אַרְבַּע לְכָל זָכָר.

          הוֹרֵג יְבַקֵּעַ אֶת בְּשָׂרוֹ כַּאְ/דָמָה תְבֻקַּע בַּחֲרשׁ אִכָּר.


וַיַּעַן הַשְּׁבִיעִי וַיֹּאמַר:

          הָאִישׁ אֲשֶׁר דָּת יוֹצְרוֹ חִלֵּל / וַיִּשְׁכְּחָהּ יוֹם דִּין וְלֹא זָכַר.

          וַיַּעֲנֶה עַזּוֹת לְמוּל צוּרוֹ / וַיִּפְעֲרָה פִיהוּ וְלֹא סָכַר.

          יַחְשִׁיךְ אֱלֹהִים מִשְּׁחוֹר פָּנָיו / עַד כִּי לְרוֹאָיו לֹא יְהִי נִכָּר.


וַיַּעַן הַשְּׁמִינִי וַיֹּאמַר:

          הָאִישׁ אֲשֶׁר הֵעֵז לְאֵל פָּנָיו / עַד כִּי בְּנוֹ עַמְרָם לְרַע זָכַר.

          חָשַׁב עֲכוֹר תּוֹרַת אֱלֹהֵינוּ / בַּשָּׁוְא וְנַפְשׁוֹ בֶּאֱמֶת עָכַר.

          יַעְקֹר אֱלֹהִים כָּל בְּנֵי בֵיתוֹ / עַד עִם נְקֵבָה יַאֲבִיד זָכָר.


וַיַּעַן הַתְּשִׁיעִי וַיֹּאמַר:

          הַבֵּט בְּנוֹ עַמְרָם הֲכִי חִלֵּל / דָּתָךְ אֲבִי טֻמְאָה וּבֶן נֵכָר.

          יִתְגַּל בְּמַחְבוֹאָיו וְאוּלָם תּוֹךְ / קָהָל יְהִי נִסְתָּר וְהִתְנַכַּר.

          מַהֵר אֱלֹהִים יַאֲבִידוֹ וַחְ/רוֹן אַל יְהִי אֶל פָּעֳלוֹ שָׂכָר.


וַיַּעַן הָעֲשִׂירִי וַיֹּאמַר:

          הוֹי עַל קְהִלָּה שַׂר קְהִלָתָם / הֵזִיד לְגַדֵּף אֵל וְהִתְמַכַּר.

          תָּעָה בְיֵין זִמָּה וְלֹא יַיִן / רָוָה בְּסִבְאוֹת חֵטְא וְלֹא שֵׁכָר.

          אֵשׁ מִשְּׁמֵי מָרוֹם בְּאַף תִּשְׂרֹף / גּוּפוֹ כְּמוֹ חֵלֶב וְכִיקַר כָּר.


וְאֵלֶּה עָשִׂיתִי בְּעִנְיַן הַסִּפּוּק / כִּי הוּא לָאֱמוּנָה חִזּוּק:

          אִם בַּעֲבוֹדַת אֵל תְּכַל יָמֶיךָ / תִּהְיֶה פְּאֵר עַל רֹאשׁ בְּנֵי עַמֶּךָ.

          שִׂים חֶלְקְךָ פַּת עִם כְּלִי מַיִם וּבַ/רֵךְ לַחְמְךָ יוֹצְרָךְ וְגַם מֵימֶיךָ.

          הָקֵם בְּלֵב מִזְבַּחֲךָ אַךְ חֶלְבְּךָ / הַקְרֵב וְנַסֵּךְ בַּבְּכִי דָמֶיךָ.

          וּזְכוֹר הֲכִי תֵבֵל כְּמוֹץ עוֹבֵר וְכִי / כַּצֵּל וְכַצֶּלֶם דְּמוּת צַלְמֶךָ.

          וַעְרֹךְ לְרָעֵב פַּת חֲסָדֶיךָ / וּקְרָא עֲנִיֵּי עַם לְמַטְעַמֶּיךָ.

          וּבְנֵה לְךָ בַיִת אֲשֶׁר תִּמְצָא בְיוֹם / מִבֵּיתְךָ תִּסַּע לְעוֹלָמֶךָ.

          אִם תַּעֲשֶׂה כֵן אוֹיְבֶךָ תַּחְלשׁ / וּבְרֹב גְּאוֹנְךָ תַּהֲרֹס קָמֶיךָ.


וְאֵלֶּה עָשִׂיתִי בְּעִנְיַן הַתְּשׁוּבָה / מִן הַמְּשׁוּבָה:

          אַזִּיל עֲנַן דִּמְעָה עֲלֵי פִשְׁעִי עֲדֵי / נֶגֶד דְּמָעַי נִשְׁכְּחוּ מֵי נֹחַ.

          מַה לָּךְ לְבָבִי תַחֲשֹׁק תֵּבֵל אֲשֶׁר / בָּהּ כָּל חֲכַם לֵב יִזְנְחָהּ זָנוֹחַ?

          רוּץ לַעֲבוֹדַת אֵל וְשׁוּבָה לְאֱלֹ/הֵינוּ הֲכִי יַרְבֶּה מְאֹד לִסְלוֹחַ.

          שֵׂיבָה הֲלֹא תִשְׁלֹף לְךָ חַרְבָּה וְגַרְ/זֶן הַפְּרִידָה תַּחֲשֹׁב לִנְדּוֹחַ.

          קַוֵּה בְכַל עֵת אֶל אֱלֹהֶיךָ הֲכִי / קֹוֵי אֲדֹנָי יַחֲלִיפוּ כֹחַ.


וְאֵלֶּה עָשִׂיתִי לְהוֹכִיחַ הַכְּסִילִים / וּלְהַזְהִיר הַנּוֹאָלִים:

          יָשֵׁן לְמָתַי עֵינְךָ עָצַמְתָּ / אוֹ כִי בְחֵיק הַתַּאֲוָה נִרְדַּמְתָּ?

          גּוֹיִם אֲשֶׁר לֹא יָדְעוּ בוֹרֵא דְרָ/שׁוּהוּ – וְאַתְּ שׁוֹגֶה וְלֹא נְכִלַמְתָּ.

          רָאוּ מְאוֹר צִדְקוֹ וְאַתָּה מֵרְאוֹת / אוֹר הָאֱמֶת מַסְוֶה בְעֵינֶיךָ שַׂמְתָּ!

          כַּמָּה קְרָאוּךָ בְלִי פֶה קוֹרְאוֹת / יָמִים. וְאַתְּ נִרְדָּם וְעוֹד לֹא קַמְתָּ.

          רָצוּ חֲבֵרֶיךָ חֲצוֹת לֵיל וְהִשְׁ/כִּימוּ לְשַׁחֵר אֵל, וְלֹא הִשְׁכַּמְתָּ.

          תִּישַׁן בְּחֵיק תֵּבֵל וְטוֹבָתָהּ חֲלוֹם / מָה הַחֲלוֹם הַזֶּה אֲשֶׁר חָלַמְתָּ.


וְעָשִׂיתִי תוֹדָה עַל חַסְדֵי הַבּוֹרֵא / וּמִי כָמוֹהוּ מוֹרֶה:

          אֲדוֹן עוֹלָם גְּמַלְתַּנִי חֲסָדִים / וְעַל חַסְדָּךְ גְּמַלְתִּיךָ מְרָדִים.

          וְרַעְיוֹנַי לְרַחְמֶיךָ בְכָל עֵת / מְלֻמָּדִים וְדָתָךְ לֹא לְמֵדִים.

          וְלֹא רָחְקוּ חֲסָדֶיךָ בְעוֹדִי / בְּעֵת שִׁבְתִּי וְעֵת אַרְחִיק נְדוֹדִים.

          בְּיוֹם בּוֹ אָרְבוּ צָרוֹת לְנַפְשִׁי / וְחִכִּיתִי כְּחַכֵּה אִישׁ גְּדוּדִים.

          גְּאַלְתַּנִי וְהִרְחַקְתָּ מְגוּרֵי / וְשִׂמַּחְתָּ סְעִפֵּי הַחֲרֵדִים.

          וּבִהְיוֹתִי בְתוֹךְ צִיָּה וּמִדְבָּר / וְהָאָרֶץ כְּמוֹ גַחֲלֵי יְקוֹדִים.

          וְדָבַק בָּהּ לְשׁוֹן כָּל אִישׁ לְחִכּוֹ / וְאֵין מָנוֹס וְאֵין מַעְמָד לְעוֹמְדִים.

          וְעֵת רָגְזוּ קְרָבַי בִּי וְהָמוּ / וּמֵעַי בַּכְּאֵב עוֹלִים וְיוֹרְדִים.

          עֲזַרְתַּנִי וְהָיוּ לָךְ שְׁבָחַי / כְּמוּסָפִים וְרַחְמֶיךָ תְמִידִים.

          וְעַל כָּל־זֶה שְׂעִפַּי כּוֹרְעִים לָךְ / וְהֵם מִשְׁתַּחֲוִים נֶגְדָּךְ וּמוֹדִים.

          זְכוֹר צִדְקָךְ אֲדוֹנֵי הָאֲדוֹנִים / וְאַל תִּזְכֹּר עֲווֹן עֶבֶד עֲבָדִים.

          וְדִבְרֵי תַחֲנוּנִים חַבְּרֵם עִם / חֲסָדֶיךָ וְהָיוּ לַאֲחָדִים.

          בְּמֵי דִמְעִי עֲצֵי תִקְוָה אֲרַוֶּה / עֲדֵי עֵץ חַסְדְּךָ יִתֵּן מְגָדִים.


וְאֵלֶּה עָשִׂיתִי עַל הָרָב רַבֵּנוּ מֹשֶׁה / אֲשֶׁר מִמַּהֲלָלוֹ כָּל פֶּה לֹא יֶחֱשֶׁה:

          יוֹם עָלוּ כָּל חַכְמֵי תֵבֵל / אֶל מַעְלוֹת רָמִים וּגְבוֹהִים

          עָלוּ לִמְרוֹם הַשֵׂכֶל אַךְ / מֹשֶׁה עָלָה אֶל אֱלֹהִים.


וְאֵלֶּה עָשִׂיתִי עַל עֲדַת נְבָלִים / וְאֵין בָּהֶם אֲצִילִים:

          בְּגוֹי נָבָל כְּבוֹדִי שָׁב מְשֻׁסֶּה / וּמַעְלָתִי בְּרַגְלֵיהֶם רְמוּסָה.

          אֲשֶׁר יָדָם בְּתוּלָה לַנְּדָבָה / וְלַכִּילוּת בְּעוּלָה גַּם כְּנוּסָה.

          בְּמוּמָם נִטְמְאָה תֵבֵל וְלוּלֵא / בְּרִית מַבּוּל – שְׁטָפָם יָם וְכִסָּה.

          וְנַפְשָׁם הִיא קְרוּצָה מִנְּבָלוֹת / וּמִכָּל מוּם וְקַלָּסָה עֲמוּסָה.

          וְצִיר מָוֶת בְּעֵת יָבוֹא לְהָרְגָם – / יְבַקֵּשׁ מִבְּאוֹשׁ נַפְשָׁם מְנוּסָה.

          וְיוֹרִידָהּ לְשַׁחַת עַל קְצָת עֵץ / לְטֻמְאָתָהּ וְהִיא עָלָיו מְאוּסָה.

          וְלוּ יִלְמַד בְּשַׁבָּת יִמְצְאָה שָׁם / הֲכִי נַפְשָׁם מְטַמְּאָה הִיא בְּמַשָּׂא.


וְאֵלֶּה עָשִׂיתִי לְאִישׁ בְּיָדוֹ נְדָבוֹת / וּפָנָיו חֶדְוַת לְבָבוֹת:

          תְּהִלּוֹתַי שְׁלַחְתִּים אֶל יְדֵי שַׂר / מְרִיקִים מֵי נְדָבוֹת מִהְיוֹתָם.

          וְאֶפְחַד אִם יְשַׁו יָדָיו עֲלֵיהֶם / הֲכִי יִמְחַק וְלֹא יוּכַל קְרֹאתָם.


וְאֵלֶּה עָשִׂיתִי עַל בַּת הַגְּפָנִים / מְגָרֶשֶׁת הַיְגוֹנִים:

          אֱמֶת בַּת הַגְּפָנִים הִיא צְרִי צִיר / וְכָל לָשׁוֹן תְּהִלָּתָהּ תְּסַפֵּר.

          בְּדָמָהּ יִמְצְאָה דָמִי מְנוּחָה / וְהַדָּם הוּא וּבַנֶּפֶשׁ יְכַפֵּר.


וְאֵלֶּה עָשִׂיתִי בְּמוּסַר הַמַּשְׂכִּילִים / לְהַזְהִירָם מֵעֲלוֹת לְמַעֲלוֹת הַגְּדוֹלִים:

          הַמַּעֲלוֹת שָׁלשׁ, וְאִם בָּם תַּחֲשֹׁק / מִן הַשְּׁפָלָה נוּס וְהָעֶלְיוֹנָה.

          כִּי הַקְּצָווֹת הֵם מְקוֹם פַּחַד וְהָ/אֶמְצַע מְקוֹם שָׁלוֹם וְהַתִּיכוֹנָה.



וְאֵלֶּה עָשִׂיתִי לְאִישׁ נָדִיב אֵין עֲרוֹךְ אֵלָיו / בְּנֹעַם מִפְעָלָיו:

          שׁוֹאֵל בְּעַד נִדְבַת בְּחִיר אֵל הַגְּבִיר – / צוּרֵי סְלָעִים דָּשְׁאוּ מִנּוֹזְלָיו.

          מוֹשֵׁל אֲשֶׁר מָשַׁל בְּרֹאשׁ שָׂרִים אֲבָל / נִכְנַע לְכָל דּוֹרְשָׁיו עֲדֵי הֵם מוֹשְׁלָיו.

          מַה נּוֹאֲלוּ הַחוֹשְׁדִים כִּי יִגְמְלֵם / חֶסֶד, וְלֹא יָדְעוּ הֲכִי הֵם גּוֹמְלָיו.

          נַפְשׁוֹ לְאֵל מָכַר וְעֵינָיו אֶל יְדֵי / שׁוֹאֵל כְּעֵין עֶבֶד אֶל יַד גּוֹאֲלָיו.

          יִתֵּן וְיִתְנַדֵּב כְּנִדְבַת עַנֲנֵי / שַׁחַק וְיִתְחַנֵּן תְּחִנַּת שׁוֹאֲלָיו.


וְאֵלֶּה עָשִׂיתִי עַל אַנְשֵׁי מִזְרָח בְּעֵת אוֹתָם נִסִּיתִי / וּבְצִלָּם חָסִיתִי:

          יְגַדַּע אֵל בְּאַף קֶרֶן זְמַנִּי / אֲשֶׁר פִּזֵּר חֲפָצַי עֵת אֲסַפְתִּים!

          נְדִיבָיו נִבְרְאוּ טֶרֶם הֱיוֹתִי / וְעֵת צֵאתִי לְאוֹר עוֹלָם חֲסַרְתִּים.

          כְּאִלּוּ שָׁמְעוּ שִׁמְעִי וּבָרְחוּ / וְעַל כֵּן עֵת דְּרַשְׁתִּים לֹא מְצָאתִים.


וְעוֹד עָשִיתִי עַל הָעִנְיָן הַזֶּה. וְזָכַרְתִּי / שְׁלשֶׁת גִּבּוֹרֵי הַשִּׁיר אֲשֶׁר הָיוּ בִּסְפָרַד:

          אֲבוֹת הַשִּׁיר שְׁלֹמֹה עִם יְהוּדָה / וּמשֶׁה זָרְחָה שִׁמְשָׁם בְּמַעְרָב.

          הֲכִי מָצְאוּ בְּדוֹרוֹתָם נְדִיבִים / וּמָכְרוּ אֶת פְּנִינֵיהֶם בְּהוֹן רָב.

          וְנִבְרֵאתִי אֲנִי עֵת אָבְדוּ כָל / נְדִיבֵי לֵב וּבָא שִׁמְשָׁם וְעָרַב.

          וְהֵמָּה מָצְאוּ נַחְלֵי עֲדָנִים / וּמָצָאתִי אֲנִי חֹרֶב וְשָׁרָב.


וְאֵלֶּה עָשִׂיתִי לַשַּׂר רַ' שְׁמוּאֵל בֶּן הָרֹאשׁ רַ' נִסִּים בִּסְפָרַד, חִבַּרְתִּיו עַל שְׁמוֹ /

          וְעַל שֵׁם־בְּנוֹ יְהִי אֱלֹהָיו עִמּוֹ:

          לְבָבִי בָחַן כָּל בְּנֵי שֵׁם וּגְדֻלָּה / וּבָחַר לוֹ הַשַּׂר שְׁמוּאֵל לִסְגֻלָּה.

          וְשָׂם אוֹתוֹ וּבְנוֹ לְנֵס גַּם לִתְהִלָּה / וּבַגָּדוֹל הֵחֵל וּבַקּטֹן כִּלָּה.


וְעוֹד עָשִׂיתִי לִבְנוֹ יוֹצְרוֹ יִצְּרֵהוּ / וְיִשְׁמְרֵהוּ וִיחַיֵּהוּ:

          צְבִי חֵן כְּלִיל יֹפִי פְּלָאוֹת בְּתָאֳרוֹ: / לְחָיָיו כְּמוֹ שֶׁלֶג וְזִיקוֹת יְאַזְּרוּ.

          בְּמִצְחוֹ מְאוֹר שַׁחַר וְלַיִל יְקֻבְּצוּ / וְעַל לֶחֱיוֹ לַהַב וּבָרָד יְחֻבְּרוּ.

          וְשִׁנָּיו שְׁנֵי טוּרֵי בְדֹלַח כְּחוּט עֲנָק / וּבָם טַלְלֵי אוֹרוֹת רְפָאִים יְעוֹרְרוּ.

          שְׁמוֹ קָרְאוּ מַזְהִיר וְלֶחְיוֹ כְמוֹ שְׁמוֹ / הֲכִי יֵלְכוּ עִוְרִים לְאוֹרוֹ וְיִנְהֲרוּ.

          יְהִי רֹאשׁ לְבֵית אָבוֹת כְּיוֹלְדָיו וְיִחְיֶה / עֲדֵי יֶחֱזוּ הוֹרָיו אֲשֶׁר בּוֹ יְשַׂבְּרוּ.


וְעָשִׂיתִי בְדַמֶּשֶׂק כְּשֶׁנָּסַע ר׳ משֶׁה הָרוֹפֵא לְאֶרֶץ חֲמָת:

          שְׁאֵלוּנִי, הֲלֹא תִשְׂמַח בְּחֶבְרַת / בְּנֵי מוּסָר, מְתֵי שֵׂכֶל וְתֻמָּה?

          עֲנִיתִים: בַּגְּבִיר משֶׁה מְשׂוֹשִׂי / וּבִלְתּוֹ הַנְּשָׁמָה הִיא נְשַׁמָּה.

          וְהוּא נֶפֶשׁ וְאַנְשֵׁי הַזְּמַן גּוּף / וּמַה מּוֹעִיל לְגוּף מֵאֵין נְשָׁמָה?


וְעָשִׂיתִי בְּעִנְיַן הַחֵשֶׁק וְהָאַהֲבָה:

          יְקַצֵּץ אֵל צְעִיפִים מַעֲלִימִים / פְּנֵי עָפְרָה וְיִקְרַע הָרְדִידִים.

          הֲלֹא דַי לִי בְצִיר חֵשֶׁק וּפֵרוּד / עֲדֵי כִּי יִבְגְדוּ בִי הַבְּגָדִים.



וְעָשִׂיתִי בְעִנְיַן הַצְּדָקוֹת / אֲשֶׁר יַצִּילוּ מִן הַמְּצוּקוֹת:

          זְרַע בֶּן אִישׁ לְךָ זֶרַע צְדָקָה / וּמָחָר תִּמְצְאָה מֵאָה שְׁעָרִים.

          וְאִם יִסְגֹּר זְמַן שַׁעְרֵי רְצוֹנָךְ / בְּיַד הָאֵל לְךָ כַמָּה שְׁעָרִים.


          וְעָשִׂיתִי בְּעִנְיַן הָעוֹלָם אֲשֶׁר הוּא כְּצֵל עוֹבֵר / וְשׁוֹכְנָיו שֶׁבֶר לִפְנֵי קֶבֶר:

          אֲשֶׁר יִבְטַח בְּתַהְפּוּכוֹת זְמַנּוֹ / יְהִי נִכְשָׁל וְיִתְחַלְחַל בְּצִירוֹ.

          הֲכִי חַיֵּי אֱנוֹשׁ כֶּרֶם וְרַק הַ/זְּמַן בּוֹצֵר וְהַמָּוֶת בְּצִירוֹ.


וּבְעִנְיַן הַמָּוֶת / וּשְׁכוֹן אֶרֶץ וְצַלְמָוֶת:

          בֶּן אִישׁ הֲתֵעָדֵן בְּתַעֲנוּג הַזְּמַן / וּבְשָׂרְךָ דָשֵׁן מְאֹד וְגַם שָׁמֵן.

          הוֹנָךְ כְּלִי שֶׁמֶן וְיִשָּׁבֵר בְּמוֹ/תַךְ הַכְּלִי, וְיַעֲמֹד הַשָּׁמֶן.


וְעָשִׂיתִי לְאִישׁ מִן הָאֲצִילִים / שׁוֹכֵן בֵּין הַנְּבָלִים:

          הַשַּׂר אֲשֶׁר בֵּין כּוֹכְבֵי רוּם יַחֲנֶה / אַךְ שׁוֹאֲלוֹ טֶרֶם קְרָאוֹ יַעֲנֶה.

          שַׁח לֵב וְאִם נִלְוֶה לְעַם מִשְׂתָּרְרִים / נָדִיב וְאִם בֵּין קוֹפְצֵי יָד יַחֲנֶה.

          אֶתְמַהּ לְשִׁכְנוֹ בָם וְלֹא אֶתְמַהּ הֲכִי / כֵּן הַשְּׁכִינָה שָׁכְנָה עַל הַסְּנֶה.


וְעוֹד עָשִׂיתִי לוֹ לְשֶׁבַחַ מַעֲלָתוֹ וְגָדְלוֹ:

          לְבָבִי לְיִצְחָק בְּחִיר אֵל מְכַרְתִּיו / בְּכֶסֶף יְדִידוּת וּבִמְחִיר אֲהָבִים.

          דְּבִיר הַיְּקָר הוּא וְלִבּוֹ אֲרוֹן דָּת / וְיָדָיו בְּתִתָּם שְׁנֵי הַכְּרוּבִים.

          וְלוּחוֹת לְבָבוֹ כְּלוּחוּת בְּרִית הֵם / וְאַךְ הֵם בְּאֶצְבַּע אֱלֹהִים כְּתוּבִים.

          וְהוּא רָם בְּעֵין כֹּל וְשָׁפָל בְּעֵינָיו / וְיָחִיד וְרַבִּים הֲלָכָה כְּרַבִּים.


וְעוֹד עָשִׂיתִי בְּמִבְחַן הַיְדִידִים:

          נַסֵּה יְדִידֶיךָ בְּעֵת מָצוֹר / וּטְעַם בְּחֵךְ שִׂכְלָךְ פְּרִי דָתָם.

          אִם לֹא יְהִי לִבָּם לְךָ נֶאְמָן / טוֹב מִיְּדִידוּתָם פְּרִידָתָם.


וְעוֹד עֲשִׂיתִי בְּמַהֲלַל אֲצִילִים / אַנְשֵׁי חַיִל רַבֵּי פְעָלִים:

          אִישִׁים אֲשֶׁר אֶל אוֹר כְּבוֹדָם יֵלְכוּ / כָּל־יוֹשְׁבֵי תֵבֵל וְשׁוֹכְנֵי אֶרֶץ.

          כִּי בָם שְׂמָחוֹת חִדְּשָׁה תֵבֵל וּבָם / חֶסֶד אֲדֹנָי מָלְאָה הָאָרֶץ.

          בַּבִּין וּבַמּוּסָר וּמִדּוֹת כָּל יְקָר – / לָהֶם לְבַדָּם נִתְּנָה הָאָרֶץ.

          יִגְאוּ וְאוֹיְבֵיהֶם עֲפָרָם יִלְחֲכוּ / עַד כִּי לְשׁוֹנָם תִּהֲלַךְ בָּאָרֶץ.

רוֹאִים גְּדֻלָּתָם וְשִׂכְלָם יֹאמְרוּ / אַךְ יֵשׁ אֱלֹהִים שׁוֹפְטִים בָּאָרֶץ.


וְעוֹד עָשִׂיתִי בְעִנְיַן הַסִּפּוּק:

          הֲלֹא דַי לִי בְטַרְפִּי מִזְּמַנִי / וְאַל אָמִיר בְּקָלוֹן אֶת גְּאוֹנִי.

          לְבִלְתִּי יֹאמְרוּ עָלַי בְּנֵי אִישׁ / רְאוּ חֶסֶד פְּלוֹנִי עַל פְּלוֹנִי.

          בְּעֵת לֹא אֶהֱיֶה נִצְרָךְ לְהוֹנוֹ / חֲשַׁבְתִּיו כַּאֲשֶׁר הוּא יַחְשְׁבֵנִי.

          אֲהַבְתִּיו אִם יְבַקֵּשׁ אַהֲבָתִי / וְאָפֵר הַבְּרִית אִם יַעְקְבֵנִי.


וְעוֹד עַל עִנְיַן הַחֶבְרָה:

          יְמֵי חֶבְרָה מְאֹד מָתְקוּ לְחִכִּי / וְלֵילוֹת הַנְּדוֹד רָעִים וּמָרִים.

          שְׂמָחוֹת בָּרְחוּ מִנִּי וְדָמוּ / נְשָׁמוֹת מָאֲסוּ לִשְׁכֹּן פְּגָרִים.

          יְגוֹנוֹתַי בְּלֵיל פֵּרוּד אֲרֻכִּים / וְשִמְחוֹתַי כְּלֵיל חֶבְרָה קְצָרִים.


וְעוֹד עַל הָעִנְיָן:

          עֲדֵי מָתַי תְעַנֵּנִי צְבִי חֵן / וְנַפְשִׁי לָךְ כְּמוֹ אָמָה מְכוּרָה.

          וּמִי יִתֵּן וְאֶמְצָא סוֹף לְקִצְפָּךְ / וּמַגֵּפַת נְדוֹדָךְ נֶעֱצָרָה.

          מְקוֹרוֹת לַבְּכִי שַׂמְתָּ בְּעֵינִי / וְהִבְעַרְתָּ לְבָבִי תּוֹךְ מְדוּרָה.

          וְרָאוּי לָךְ לְשַׁלֵּם אֶת לְבָבִי / וְכֵן הַדִּין לְמַבְעִיר הַבְּעֵרָה.


וְעוֹד עַל הָעִנְיָן:

          אִשֵּׁי לְבָבוֹת הֲלֹא אֶשְׁלַח בְּעַד אֶשְלַח / בִּשְׂפַת צְבִי חֵן אֲשֶׁר גּוּפוֹ בְּדֹלַח לַח.

          אוֹחִיל תְּשׁוּבַת שְׁלוּחִי לוֹ וְלֹא יִצְלַח / הוּא יֶחֱטַא לוֹ וְאֶתְחַטָּא וְלֹא יִסְלַח.


וְעוֹד עַל הָעִנְיָן:

          שָׂרִיג אֲשֶׁר מֵעֲצֵי גַן הַיֳּפִי נִקְצַב / דַּר לֶחֱיוֹ מִמְּקוֹם מֶחְצַב יְקָר חֻצָּב.

          כִּנְצִיב בְּדֹלַח עֲלֵי קוֹמַת כְּפוֹר נִצָּב / בִּדְבַשׁ שְׂפָתוֹ מְשׂוֹשׂ לֵבָב אֱנוֹשׁ נֶעְצָב.


וְעוֹד עַל הָעִנְיָן:

          לוּ מֵעֲסִיס חִכְּךָ עֹפֶר מְעַט אֶשְאָב / אַחְלִיץ עֲצָמַי אֲשֶׁר בָּלוּ וְלֹא אֶדְאָב.

          אֶשְׁתַּחֲוֶה נֶגְדְּךָ כָל יוֹם וְלָךְ אֶתְאָב / אֶדְמֶה כְּעֶבֶד לְךָ אוֹ בֵן יְכַבֵּד אָב.


וְעוֹד עַל הָעִנְיָן:

          נִשְׁבַּע צְבִי עַד יְמִיתֵנִי לְבַל יִשְׁקֹט / נַפְשִׁי עֲלֵי זֹאת בְּחַיַּי נָקְטָה נָקֹט.

          אוֹחִיל רְצוֹנוֹ וְכִסְלִי יוֹם לְיוֹם יָקֹט / אֶרְאֶה עֲלֵי לֶחֱיוֹ הַמָּן וְלֹא אֶלְקֹט.


וְעוֹד עַל הָעִנְיָן:

          עֹפֶר בְּפִיהוּ שְׁנֵי טוּרֵי עֲנַק לֶשֶׁם / יִצְחַק לְבִכְיִּי כְּמוֹ קֶשֶׁת בְּיוֹם גֶּשֶׁם.

          גוּפוֹ כְשֶׁלֶג וְלוֹ מִן הַכְּפוֹר גֶּשֶׁם / בּוֹ יִקְטְלֵנִי וְלֹא יָדַע וְהוּא אָשֵׁם.


וְעוֹד עַל הָעִנְיָן:

          עָפְרַת אֲהָבִים בְּתוֹךְ בָּנוֹת

          מְאֻשֶּׁרֶת / נַפְשִׁי אֲהָהּ בַּעֲבוֹת חִשְׁקָהּ מְקֻשֶּׁרֶת.

          חִשְׁקִי צְפַנְתִּיו – וְדִמְעָתִי כְּכִנֶּרֶת / לֹא תַעֲזֹב לֶאֱנוֹשׁ אַהְבָה

          מְסֻתֶּרֶת.




וְאֵלֶּה עָשִׂיתִי לְאֶחָד מִן הַנְּשִׂיאִים הָאֲצִילִים / וְהוּא רֹאשׁ לַנְּבָלִים:

          וְשִׁבַּחְתִּי לְאִישׁ נָבָל בְּאַשּׁוּר / בְּשִׁיר כָּזָב אֲשֶׁר אֵין בּוֹ נְכוֹנָה.

          וְנָדַר לִי נְדָרִים הֵם כְּשִׁירַי / מְלֵאִים שָׁוְא וְאֵין בָּהֶם אֱמוּנָה.

          וְעוֹד לֹא רָאֲתָה עֵינִי לְשׁוֹן שָׁוְא / כְּמוֹתוֹ מִסְּפָרַד עַד עֲדִינָה.

          וְעֵת רֵעָיו שְׁאֵלוּהוּ וְאָמְרוּ / הֲתִקְפֹּץ יָד מְלֻמֶּדֶת נְתִינָה.

          הֱשִׁיבָמוֹ אֲנִי הַיּוֹם קְצַר יָד / וְנַפְשִׁי הִיא בְּחֹסֶר כֹּל בְּחוּנָה.

          וְלִי צֹרֶךְ לְהוֹצִיאָה בְּכָל יוֹם / אֲשֶׁר בִּשְׁטָר וְחוֹב עָלַי נְתוּנָה.

          וְלִי נַעַר יְשָׁרְתֵנִי כְּחֶפְצִי / וְעַרְשֵׂנוּ בְכָל עֵת רַעֲנָנָה.

          וְעָלַי חוֹב לְהַשְׁלִים לוֹ צְרָכָיו / וְלָתֵת לוֹ שְׁאֵר וּכְסוּת וְעוֹנָה.


וְעָשִׂיתִי עַל הַנָּשִׂיא ר׳ יוֹסֵף בֶּן שׁוֹשָׁן זַ"ל כְּשֶׁנִּפְטַר:

          יִכְלוּ זְמַנִּים וְלֹא יִכְלוּ יְגוֹנֵינוּ / הַאִם לְבָבוֹת כְּצוּר אוֹ יָם בְּעֵינֵינוּ?

          שָׂרֵי זְמַן נִבְהֲלוּ עַל מוֹת גְּבִירָם אֲשֶׁר / מָשְׁחוּ לְמֶלֶךְ וְאָזְלַת יַד גְּאוֹנֵינוּ.

          בּוֹשֵׁשׁ בְּחִיר אֵל וְלִזְמַן יֹאמְרוּ קוּם עֲשֵׂה / לָנוּ אֱלֹהִים אֲשֶׁר יֵלְכוּ לְפָנֵינוּ.


וְעָשִיתִי לְאִישׁ מִן הַחֲשׁוּבִים / וּמִבְחַר הַנְּדִיבִים / וּשְׁמוֹ יְרֻחָם:

          אֵיךְ לֵב חֲסַר לֵב יְרֻחָם / אֵיךְ מִיְּגוֹנוֹ יְנֻחָם.

          בָּטַח בְּתִקְוָה מְאוֹרָהּ / קָדַר וּפָעַל בְּפֶחָם.

          כָּל בַּעֲלֵי הַנְּדָבוֹת / הֵם רִיק וְצִיצִים וּפִרְחָים

          אַךְ יֵשׁ בְּתֵבֵל נְדִיב לֵב / הִשְׁפִּיל נְדִיבִים וְהִשְׁחָם.

          תַּאְוַת נְדִיבִים בְּתִתָּם / מַהְלַל מְבַקְשִׁים בְּלָקְחָם.

          אוּלָם לְכָל שׁוֹאֲלָיו הוּא / מוֹדֶה וְעוֹזֵב יְרֻחָם.


וְעָשִׂיתִי לְאִישׁ יוֹדֵעַ נַגֵּן בְּכִנּוֹר:

          שְׂאוּ נָא אֶת שְׁלוֹמִי אֶל יְשַׁעְיָה / אֲשֶׁר לָקַח תְּעוּדָה לוֹ לְרַעְיָה.

          מְעוֹרֵר מֵיתְרֵי כִנּוֹר לְשׁוֹרֵר / כְּמוֹ לֵוִי בְּמִשְׁמֶרֶת לְוִיָּה.

          וְכַיּוֹנֵק בְּחֵיק אִמּוֹ מְצַחֵק / וְיָרִים קוֹל בְּרִנָּה לֹא בְּבִכְיָה.

          וְלוּ עַל מֵת יְנַגֵּן עוֹרְרָהוּ / וְעָלָיו צָלְחָה רוּחַ וְחָיָה.


וְעָשִׂיתִי אֵלֶּה בְּבֵית הַכְּנֶסֶת אֲדוֹנֵנוּ עֶזְרָא / וְהִנָּם חֲקוּקִים עַל קִירוֹת הַבִּירָה:

          יְצַו הָאֵל לְעֶזְרָא רֹב שְׁלוֹמוֹת / צְבִי צִבְאוֹת מְרוֹמוֹת וַהֲדוֹמוֹת.

          לְזִיו אוֹרוֹ פְּאֵר כָּל הַמְּאוֹרִים / וְלִמְקוֹמוֹ פְאֵר כָּל הַמְּקוֹמוֹת.

          מְקוֹמוֹ הוּא מְקוֹם מַחְנֵה אֱלֹהִים / וְאֵלָיו נִפְתְּחוּ שַׁעְרֵי מְרוֹמוֹת.

          תְּהִי נַפְשׁוֹ בְכִסֵּא אֵל צְרוּרָה / וְיַחְלִיץ טַל יְשׁוּעָה הָעֲצָמוֹת,

          וְהָאֵל מִשְּנָתוֹ חִישׁ יְעִירוֹ / לְקֵץ יָמִין וְיִחְיֶה אַחֲרֵי מוֹת.

          זְכוּתוֹ יַעֲמֹד עָלָיו לְמֵלִיץ / לְהַשְׁקִיט אֶת שְׁאוֹן יָם הַתְּהוֹמוֹת.

          וְיָפִיק אֶל יְהוּדָה בֶּן שְׁלֹמֹה / בְּיָדוֹ עֹז וּבִימִינוֹ נְעִימוֹת.
