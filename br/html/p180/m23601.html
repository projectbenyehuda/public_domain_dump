לאט עלי

ולו עוד שירה למני שהתמיד בידידותו


לְאַט עָלַי פְּרִידָתְךָ, כְּאֵבִי, / וְאַל תֹּאמַר לְצוּלַת יַם: חֳרָבִי!

וְאִם אַגִּיד לְךָ דִבְרֵי מְרִיבוֹת ־ / שְׁפֹט, כִּי עִמְּךָ דִינִי וְרִיבִי

תְּנָה עֵד אִם הֲרֵעוֹתִי תְּחִלָּה / וְתָרֹם יָדְךָ עַל יַד נְדִיבִי

יְדִידִי, אִם עֲקַבְתִּיךָ לְעוֹלָם / בְּפָנִים אוֹ בְלֵב ־ יָגוּר עֲקֵבִי!

הֲתַחַת אַהֲבָתִי תִּשְׂטְמֵנִי / וּבָחְרִי אַף תְּפַלֵּס אֶת נְתִיבִי

וְקִנֵּא לִבְּךָ תַּכְלִית קְנָאוֹת / וּמָשַׁכְתָּ זְנַב שִׂנְאָה בְגַבִּי

וְלוּ הוּשַׂם תְּבוּנָה בַבְּהֵמוֹת / אֲזַי אָמַר חֲמוֹר: כִּרְתּוּ זְנָבִי!

וְתָשִׂים לִבְּךָ אוֹצַר קְנָאוֹת / וְלֹא שָׁמַר לְךָ אַהַב בְּקִרְבִּי

וְתַעְבֹר מִבְּלִי שָׁלוֹם וְתִפְנֶה / וּמַה בֶּצַע בְּבָרָק יַעֲבָר בִּי

מְאֹד נִכְאַב לְבָבִי עַל פְּנוֹתָךְ / וְלָמָּה זֶּה יְהִי נֶצַח כְּאֵבִי

וְשָׁכַחְתִּי בְּרֹב יָמִים דְּמוּתָךְ / כְּאִלּוּ בַחֲלוֹם רָאָךְ לְבָבִי

אֲשְֶׁר נִשְׁאַר כְּלֵב חוֹלַת אֲהָבִים / בְּעֵת יִשְּׂאוּ גְמָלָיו הָעֲרָבִי

לְךָ אֶצְמָא וְאֶתְעַלֶּה בְּךָ, לֹא, / אֲבָל עַל זִכְרְךָ קוּמִי וְשִׁכְבִי

וְאֵיךְ תּוּכַל לְהִתְאַפֵּק ־ וְהִנֵּה / פְרָת אֵין דַּי לְכַבּוֹת אֵשׁ שְׁבִיבִי

וְאִם תִּטְעַם לְךָ מֵרֹאשׁ שְׂפָתַי ־ / הֲלֹא עַל חִכְּךָ יָזוּב חֲלָבִי

וְנֶאֶמְנוּ לְךָ פִּצְעֵי אֲהוּבָךְ / וְהַמָּשָׁל מְשַׁל חָכָם וְנָבִיא

וְאִם לֹא אַעֲלֶה נַפְשִׁי ־ לְמָה זֶּה / הֲכִינוֹתִי אֲנִי שַׁחְרִי וְעַרְבִּי

וְאִם אָרוּץ, יְדִידִי, בַקְּרָבוֹת ־ / דְּרֹךְ תַּחַת אֲבַק סוּסִי וְרִכְבִּי

וְאִם תּוּכַל ־ עֲמֹד בַּמַּעֲרָכָה / וְאַל יַבְרִיחֲךָ קַשְׁתִּי וְחַרְבִּי

וְאִם תִּקְצַר לְהָכִיל אֶת דְּבָרַי / וְתֹאמַר כִּי יְשַׁדֶּדְךָ זְאֵבִי

אֲנִי מִנּוּדְךָ אֶחְרַץ לְשׁוֹנִי / וְקָחֶנּוּ לְשָׁלָל עוֹדב וְלִשְׁבִי

וְאִם אָרִיב לְךָ מֵרֹב יְדִידוֹת / הֲלֹא עַל בִּלְתְּךָ אֶשְׁאַג כְּלָבִיא

וְדַע כִּי אַפְּךָ, אָחִי, אֲכַפֵּר / וְאָשׁוּב אַחֲרֵי מִרְיִי וְשׁוּבִי

וְאִם תִּסֹּב עֲלֵי צִיר הַכְּסִילוּת / וְלֹא תִשָּׂא פְנֵי מַשְׂאַת כְּתָבִי

אֲנִי אֶסֹּב לְהָפֵר אַהֲבָתָךְ / וְאָסִיר מִמְּךָ עֵינַי וְלִבִּי.
