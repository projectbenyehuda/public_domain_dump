ספר תחכמוני - שַׁעַר חֲמִשִּׁים

בְּשִׁירִים נִפְרָדִים. מִזָּהָב וּמִפָּז רַב נֶחֱמָדִים


נְאֻם הֵימָן הָאֶזְרָחִי: הַגֵּד הִגִּיד לִי חֶבֶר הַקֵּינִי וַיֹּאמַר: הָיִיתִ בְּאֶרֶץ תֵּבֵץ. עִם יוֹשְׁבֵי יַעְבֵּץ. חֲלִיצוֹת הַמְּלִיצוֹת מְשַׁבֵּץ. וּנְפוּצוֹת הַתְּבוּנוֹת מְקַבֵּץ. וַיְהִי הַיּוֹם וַאֲנִי בְּשַׁעַר הָעִיר וַאֵרָא תַחַת אֵלָה עֲבֻתָּה קִבּוּץ הָמוֹן רָב. וַאֲסַפְסֻף נֶעֱרָב. וְכַאֲשֶׁר קָרַבְתִּי בְּלֵב מִשְׁתָאֶה. לִרְאוֹת הַמַּרְאֶה. מָצָאתִי חֶבְרַת מַשְׂכִּילִים. מִבְּנֵי הַפַּחוֹת וְהָאֲצִילִים. בְּדִבְרֵי מְלִיצוֹת מְדַבְּרִים. וְשִׁירֵי נְעִימוֹת מְחַבְּרִים. וּמִמֶּחְצַב לִבָּם פְּנִינִים חוֹצְבִים. וּמִמַּעְיַן שִׂכְלָם מַיִם חַיִּים שׁוֹאֲבִים. וְזֶה לָזֶה רוֹצֶה לְנַסּוֹת. לְמַעַן יִלְמְדוּ מְלֶאכֶת הַשִּׁיר לַעֲשׂוֹת. וְכַאֲשֶׁר רָאִיתִי כִּי בִמְלֶאכֶת הַשִּׁיר בָּאוּ. וְלַעֲרֹך מִלְחַמְתּוֹ נִצְבָּאוּ. יָדַעְתִּי כִּי אֲנִי נְגִידָם. וִיוֹשֵׁב עַל כִּסֵּא כְבוֹדָם. וְלִי יִתְּנוּ יָדָם. וָאַעַן וָאֹמַר: אַנְשֵׁי לֵבָב שִׁמְעוּ לִי. וְהַקְשִׁיבוּ לִמְלִיצוֹת שִׂכְלִי. אֲנִי הָאִישׁ. אֲשֶׁר בְּנִיבוֹ הַרֲרֵי עַד יַרְעִישׁ. וּבְשִׁירוֹ הַפְּנִינִים יוֹבִישׁ. וְנַחֲלֵי אֵיתָן בְּשִׂכְלוֹ יוֹבִישׁ. וְיוֹצִיא מַיִם מִצּוּר הַחַלָּמִישׁ. וְהָרוֹצֶה לְנַסּוֹת שִירָיו עָדָיו יַגִּישׁ. וְהַמַּאֲמִין לֹא יָחִישׁ.

אָמְרוּ לוֹ רָאשֵׁי הַקְּרוּאִים: בַּמֶּה כֹּחֲךָ זֶה. וּבַמֶּה נַכִּיר שִׂכְלְךָ וְנֶחֱזֶה. הַבְּשִׁירוֹת. אִם בְּאִגְּרוֹת. אוֹ בַחֲרוּזִים נֶחֱמָדִים. אוֹ בְּמִלִּים מְפֹרָדִים. אָמַרְתִּי: דְעוּ כִּי לִבִּי בְּכָל קֶלַע יִקְלַע. וְשִׂכְלִי בְּכָל תּוּשִׁיָּה יִתְגַּלַּע. וּבְכָל אֲשֶׁר תְּנַסּוּנִי. תִּמְצָאוּנִי. אָמְרוּ לִי: אִם כֵּן הַשְׁקֵנוּ מִן הַמַּעְיָן. וְהַשְׁמִיעֵנוּ מִשִּׁירוֹתֶיךָ אֲשֶׁר חִבַּרְתָּ בְכָל עִנְיָן. אָמַרְתִּי לָהֶם: כְּדִבְרֵיכֶם אֶעֱשֶׂה. וְסוֹדִי מִכֶּם לֹא אֲכַסֶּה. וּתְחִלָּה אֶזְכֹּר לָכֶם שִׁירֵי תְשׁוּבָה. עֲשִׂיתִים לְבַקֵּשׁ סְלִיחָה עַל הַמְּשׁוּבָה.

וָאֶשָּׂא מְשָׁלִי:

אֱלֹהַי, יַעֲלֶה זִכְרִי לְטוֹבָה

וְאַל תִּרְחַק, הֲכִי צָרָה קְרוֹבָה.

וְהַבֶּט-נָא לְגֹדֶל הַתְּלָאָה

וְאַל תַּבֵּט לְגֹדֶל הַמְּשׁוּבָה.

בְּמֵי דִמְעִי מְחֵה פִשְׁעִי וְאַל נָא

תְּהִי חַטַּאת יְהוּדָה עוֹד כְּתוּבָה.

וְכָתַבְתִּי אֵלֶּה הַשִּׁירִים עַל הֵיכַל בֵּית הַכְּנֶסֶת. יָפֶה מִתְנוֹסֶסֶת:

שַׁדַּי, שְׁעֵה קוֹל צוֹעֲקִים אֵלֶיךָ

מִשְׁתַּחֲוִים לִפְנֵי הֲדוֹם רַגְלֶיךָ.

רַמְתָּ וְעָצַמְתָּ וְנֶעְלַמְתָּ בְּאֵין

מָקוֹם וְנִגְלֵיתָ בְּמִפְעָלֶיךָ.

עַל כֵּן עֲבָדֶיךָ יְדֵיהֶם קָצְרוּ

לִבְנוֹת לְךָ נָוֶה כְּפִי גָדְלֶךָ.

אוּלָם בְּכֹחָם כּוֹנְנוּ מִקְדָּשׁ מְעַט

לָשׂוּם מְגַמָּתָם לְהֵיכָלֶךָ.

לִבָּךְ וְעֵינֶיךָ יְהוּ שָׁמָּה, הֲכִי

לִבָּם וְעֵינָם נָשְׂאוּ אֵלֶיךָ.

וְאֵלֶּה עֲשִׂיתִים עַל עִיר יְרוּשָׁלָיִם. בְּבוֹאִי אֵלֶיהָ מִמִּצְרָיִם:

שָׁלוֹם לְעִיר שָׁלֵם אֲשֶׁר יָפְיָה

נֶעְלָם לְרוֹאִים יֹאמְרוּ אַיָּהּ.

כָּל-יוֹם אֲחוֹנֵן אֶת עֲפָרָה כִּי

הִיא מַחֲלַת נַפְשִׁי וְהִיא צָרְיָהּ.

עֵינִי בְעֵת כִּי רָאֲתָה אוֹרָהּ

לֹא נֶעְצְרוּ בָהּ עַנְנֵי בִכְיָהּ.

הָאֵל אֲשֶׁר אִוָּה לְמוֹשָׁב לוֹ

יִרְאֶה בְּרֹב עָנְיָהּ וְשָׁב שִׁבְיָהּ.

בּוֹכָה לְנוּד דּוֹדָהּ בְּשׁוּבוֹ לָהּ

יִמְחֶה דְמָעוֹת מֵעֲלֵי לֶחְיָהּ.

וְעוֹד אָמַרְתִּי. וְעַל זֶה הָעִנְיָן חִבַּרְתִּי:

לְצִיּוֹן שָׁאֲפָה נַפְשִׁי וְרוּחִי

הֱיוֹת בָּהּ כָּל יְמֵי חַיַּי מְנוּחִי.

בְּחִשְׁקָהּ יִנְעֲמוּ לִי הַתְּלָאוֹת

וְאֶתֶּן-בָּהּ לְמַכֶּה רֹאשׁ וְלֶחִי

וְלֹא אֶתְאַו לְבִלְתָּהּ כָּל-יְמוֹתַי

עֲדֵי אֶגְוַע וְנָס כֹּחִי וְלֵחִי.

אֲחוֹנֵן אֶת עֲפָרֶיהָ בָּכָל-עֵת

עֲדֵי יַשְׁאִיר עֲפָרָהּ אוֹת בְּמִצְחִי.

וְאֵיךְ אֶכְסֹף לְטוּב רִקְחֵי בְשָׂמִים

וְהִיא בָשְׂמִי וְטוֹב רִקְחִי וְרֵיחִי.

עֲפָרָה אֶעֱשֶׂה מַרְפֵּא לְנִגְעִי

וְאֶמְרַח עַל שְׁחִין לִבִּי וְיֶחִי.

וְאֵלֶּה עָשִׂיתִי בְחָלְיִי. כְּשֶׁהֵפִיק אֱלֹהִים צָרְיִי:

אַל נָא אֱלֹהַי יֶחֱרֶה אַפָּךְ

עַל דַל אֲשֶׁר יֶחְסֶה בְּתוֹךְ צִלָּךְ.

אַל תַּעֲלֵנִי בַחֲצִי יָמַי

מוֹתִי הֲיוֹעִילָךְ וְיִסְכָּן-לָךְ.

לֹא אֶשְׁאֲלָה עשֶׁר וְתֶן-לִי הַ-

נֶּפֶשׁ וְאָכֵן הָרְכֻשׁ קַח לָךְ.

וְעָשִׂיתִי עַל אִישׁ מֵחַסִידֵי עַמּוֹ. נִפְטַר לְבֵית עוֹלָמוֹ:

מִפַּחְדְּךָ נַפְשָׁךְ בְּצוֹם עִנִּיתָ

עַד כִּי לְחוּט הַשַּׂעֲרָה דָמִיתָ.

יָגְעוּ בְנֵי נָבָל וְיָגַעְתָּ לְטוֹב

כִּי יָרְדוּ הֵמָּה וְאַתְּ עָלִיתָ.

אַשְׁרֵי יְחִידָתָךְ אֲשֶׁר שָׁכְנָה בְכֵס

עֶלְיוֹן וְאוֹתָהּ מִשְּׁאוֹל פָּדִיתָ.

וְאֵלֶּה עָשִיתִי בַיָּם. בְּהִנָּצְלִי מִשְּׁאוֹן גַּלָּיו וְדָכְיָם:

בַּמָּה אֲקַדֵּם אֵל וּמִי אָנִי

כִּי עַד הֲלֹם חַסְדּוֹ הֱבִיאָנִי.

מְקֵצה אֲדָמָה וַאֲצִילֶיהָ

לָבֹא לְהַר קָדְשׁוֹ קְרָאַנִי.

בַּיָּם בְּעָבְרִי בִּשְּׂרַנִי כִּי

תַּעֲבֹר בְּמַיִם אִתְּךָ אָנִי.

וְאֵלֶּה שָׁלַחְתִּי לְקֶבֶר אֲדוֹנֵנוּ יְחֶזְקֵאל. מִכֹּבֶד חֶטְאִי לְהָקֵל:

לְבַקֵּר קִבְרְךַ עֵינַי אֲגַמִּים

וְדִמְעָתִי מְגוֹלָלָה בְדָמִים.

נְבִיא הָאֵל יְחֶזְקֵאל הַמְכֻבָּד

בְּפִי צִבְאוֹת מְרוֹמִים וַהֲדוֹמִים.

קְרָבַי נִכְסְפוּ לִלְחֹךְ עֲפָרָךְ

וְלִלְקֹט מֵעֲפַר קִבְרָךְ לְשָׁמִים.

וְלִשְׁאֹף מֶרְקְחוֹת בֹּשֶׂם בְּרוּחָךְ

מְקֻטָּרִים בְּכָל רָאשֵׁי בְשָׂמִים.

וְאָז אָשִׁיר לְאֵל חַיַּי וְאֹמַר

חֲבָלִים נָפְלוּ לִי בַנְּעִימִים.

וְאֵלֶּה עָשִׂיתִי לְקֶבֶר אֲדוֹנֵנוּ עֶזְרָא הַסּוֹפֵר. לִמְצֹא כֹפֶר:

לְאוֹר עוֹלָם אֲשֶׁר הֶחְפִּיר מְאוֹרוֹת

וְסָבִיב עוֹמְדִים לוֹ כַמְּנוֹרוֹת.

וְעָדָיו שׁוֹמְעֵי שִׁמְעוֹ יְבוֹאוּן

וְנַפְשׁוֹתָם יְרִימוּן לוֹ תְשׁוּרוֹת.

וְאֵיךְ יוּכְלוּ לְכַחֵשׁ נִפְלְאוֹתָיו

וּבוֹ הָעוֹבְרִים יִרְאוּ גְבוּרוֹת.

כְּבוֹד עֶלְיוֹן סְבִיב קִבְרוֹ יְנוֹצֵץ

וְיָפִיץ אוֹר לְעֵינַיִם סְגוּרוֹת.

כְּבוֹד עֶזְרָא קְדוֹשׁ אֵל הַמְכֻבָּד

אֲשֶׁר הוּא לַיְקוּם עֶזְרָה בְצָרוֹת.

וְעָשִׂיתִי לְאִישׁ עַז וְקַשֶׁה. מֵשִׁיב עַל חִבּוּר רַבִּנוּ משֶׁה:

כָּל-מֵשִׁיב עַל דִּבְרֵי משֶׁה

שָׁבַר עֹל גַּם מוּסָר נִתַּק.

תֵּאָלַמְנָה שִׂפְתֵי שֶׁקֶר

הַדּוֹבְרוֹת עַל צַדִּיק עָתָק.

וְעוֹד עָשִׂיתִי בְּעִנְיָנָיו עַל הַמִּתְפָּאֵר עָלָיו:

וְהַחַלָּשׁ בְּעֵת יִהְיֶה בְּבֵיתוֹ

יְגַלֶּה גְבוּרָתוֹ לְאִשְתּוֹ.

וְעֵת אֶל-הַקְּרָב יִגַּשׁ וְיִקְרַב –

אֲזַי תִּתְגַּל כְּלִמָּתוֹ וּבָשְׁתּוֹ.

וְאֵלֶּה עֲשִׂיתִים. וּלְרַבֵּנוּ משֶׁה מִסְּפָרַד שְׁלַחְתִּים:

מֵאִתְּךָ הַשַּׂר תְּהִלָּתֵנוּ

וּבְרוֹמְמֶךָ כָּל-גְּדֻלָּתֵנוּ.

גָּבְהוּ דְרָכֶיךָ וְעָמָקוּ וְיִתְ-

רוֹן פָּעֳלָךְ עַל פְּעֻלָּתֵנוּ.

מַלְאַךְ אֱלֹהִים אַתְּ וְנִבְרֵאתָ בְּצֶ-

לֶם אֵל וְאִם אַתָּה כְתַבְנִיתֵנוּ

בִּגְלָלְךָ אָמַר אֱלֹהִים נַעֲשֶׂה

אָדָם בְּצַלְמֵנוּ וְכִדְמוּתֵנוּ.

וְעָשִׂיתִי בְעִנְיָן אַחֵר הֶפְכָּם. עַל אֲנָשִׁים אֲשֶׁר הִשְׁחִיתוּ דַרְכָּם:

הוֹי עַל אֲנָשִׁים לֹא לְעֵזֶר נִבְרְאוּ

עַל כֵּן בְּזִכְרָם יִפְּלוּ פָנֵינוּ.

הֵם כַּשְּׁוָרִים מִבְּלִי לֵבָב וְאֵיךְ

יֵדְעוּ דְבַר חָכְמָה וְיִתְבּוֹנֵנוּ.

לֹא בַעֲדָם אָמַר אֱלֹהִים נַעֲשֶׂה

אָדָם בְּצַלְמֵנוּ וְכִדְמוּתֵנוּ.

וְאֵלֶּה עָשִׂיתִי לְרַבִּי מְנַחֵם בַּר יִצְחָק דַּיַּן מִצְרַיִם. אֲשֶׁר אֵין כָּמוֹהוּ נְדִיב יָדַיִם וּנְקִי כַפָּיִם. יְבָרְכֵהוּ אֵל שׁוֹכֵן שָׁמָיִם:

יְמִינָךְ מִנְּהַר פִּישׁוֹן רְחָבָה

וְלָךְ בָּרָא אֱלֹהִים הַנְּדָבָה.

לְךָ נֶפֶשׁ אֲשֶׁר תָּפִיץ יְאוֹרִים

וְהִיא מִלַּהֲבֵי הָאֵשׁ חֲצוּבָה.

וְלוּ נָגְעָה יְמִין כָּל-קוֹפְצֵי יָד

וְכָל-כִּילַי בְּיָדָךְ הַנְּדִיבָה.

אֲזַי שָׁבוּ בְנֵי דוֹרָךְ נְדִיבִים

וְלֹא נִשְׁאַר אֱנוֹשׁ כִּילַי בְּצוֹבָה.

וְאֵלֶּה עָשִׂיתִי עַל רַבִּי אַבְרָהָם מַיְמוֹן זַ"ל. עֲשִׂיתִים בִּירוּשָׁלַיִם. וּשְׁלַחְתִּים לְמִצְרַיִם:

אִם הַיְקָר אָדוֹן לְכָל-אָדוֹן

הוּא בֶאֱמֶת עֶבֶד לְאַבְרָהָם.

מִיּוֹם כְּבוֹדוֹ נֶעֱלַם מִנִּי

אֵלֵךְ מְשׁוֹטֵט בַּעֲדוֹ נִדְהָם.

כָּמְהוּ שְׂעִפַּי לוֹ בְּעָבְרִי אֶ-

רֶץ נוֹד וְאִתִּי יַעֲבֹר כִּמְהָם.

וְעָשִׂיתִי לְאִישׁ אֲשֶׁר הָיְתָה לוֹ בַּת בְּתוּלָה יְקָרָה. וְנִמְצְּאָה הָרָה:

וְיוֹנַת חֵן מְקַנֶּנֶּת בְּחֵיק אָב

מְמַלֵּא יוֹם לְיוֹם טַרְפָּהּ וּמַסְפִּיק.

חֲשָׁבוּהָ בְהֵא רָפָה אֲבָל עֵת

בְּחָנוּהָ מְצָאוּהָ בְּמַפִּיק.

וְעָשִׂיתִי לְאִישׁ מִן הַנְּבָלִים. וְרֹאשׁ הַנּוֹכְלִים:

רְאוּ כִילַי אֲשֶׁר יָדוֹ בְתוּלָה

לְכָל-שׁוֹאֵל וְלַכִּילוּת בְּעוּלָה.

וּבוֹ פֶלֶא וְאוֹת כִּי הוּא בְאֶחָד

וְיֵחָשֵׁב כְּסַנְהֶדְרִי גְדוֹלָה.

וְעָשִׂיתִי עַל הָרֹאשׁ רַבִּי שְׁמוּאֵל בַּר נִסִּים וְשָׁלַחְתִּי אֵלָיו:

הַשַּׂר אֲשֶׁר לוֹ בֵּין צְבָא רוּם מַחֲנֶה

לִבִּי לְפֵרוּדָךְ מְדֻכֶּה נַעֲנֶה.

אַתָּה מְאוֹר עוֹלָם וְתִשְׁכֹּן מַחֲשָׁךְ

כֵּן הַשְּׁכִינָה בַעֲרָפֶל תַּחֲנֶה.

אֶקְרָא וְלֹא יַעְנֶה מְשׂוֹשִׂי בִּלְתְּךָ

אַךְ הַבְּכִי טֶרֶם קְרָאתִיו יַעֲנֶה.

אֵשׁ הַסְּנֶה בָּךְ נִרְאֲתָה הַשָּׂר. הֲכִי

אַתָּה כְּמַלְאַךְ הָאֱלֹהִים בַּסְּנֶה.

וְעָשִׂיתִי לְאִישׁ אֲשֶׁר יָדוֹ סְגוּרָה. וְעַב נִדְבָתוֹ עֲצוּרָה:

יָדָךְ לְשׁוֹאֵל סְגוּרָה

בְּתוּלָה וְאִישׁ לֹא יְדָעָהּ.

לוּ אִמְּךָ הָיְתָה כֵן

לֹא יִהְיֶה מוּם בְּזַרְעָהּ.

וְעוֹד עָשִׂיתִי עָלָיו. וּשְׁלַחְתִּים אֵלָיו:

רְאוּ כִּילַי אֲשֶׁר יָדוֹ כְסֶלַע

וְלִבּוֹ לַעֲצֹר אוֹתָהּ יְאַמֵּץ.

אֲנָשִׁים נוֹקְדִים יָדוֹ בְּפַתָּח

וְאַךְ הוּא מַחֲלִיף פַּתָּח בְּקָמַץ.

וְעָשִׂיתִי לְנַעַר נָשָׂא אִשָּׁה זְקֵנָה. וְלָהּ עשֶׁר וְאֵין קֵצֶה לַתְּכוּנָה:

נִפְלֵאת גְּבִירַת חֵן בְּעִנְיָנַיִךְ

מֵרֹב כְּבוֹדֵךְ יִבְעֲלוּךְ בָּנָיִךְ.

לוּ תֶחֱזִי צוּרַת דְּמוּתֵךְ תּוֹךְ רְאִי

מוּצָק. לְפַחְדֵּךְ נָפְלוּ פָנָיִךְ.

יָפְיֵךְ מְעַט כִּימֵי שְׁנוֹת בַּעֲלֵךְ וְעָשְׁ-

רֵךְ רַב לְאֵין קֵצֶה כְּמוֹ שָׁנָיִךְ.

וְעָשִׂיתִי לְאִישׁ נָבָל אֲשֶׁר קָפַץ מִמִני יָדוֹ. וְלָמַד הַכִּילוּת מִיּוֹלְדוֹ:

אִם בֶּן נָבָל קָפַץ מֶנִּי

יָדוֹ מִצְוַת יוֹלְדוֹ הֵקִים.

נִשְׁבַּע הָאֵל בַּל יִהְיֶה עַל

נִכְבָּדִים חֶסֶד הָרֵיקִים.

כִּי לֹא יָנוּחַ שֵׁבֶט הָ-

רֶשַׁע עַל גּוֹרַל צַדִּיקִים.

וְעָשִׂיתִי לְאִישׁ נָדִיב בִּלְשׁוֹנוֹ. וְכִילַי בְּהוֹנוֹ:

אִם בַּעֲלֵי מִשְׂרָה כְּמוֹ עַיִן

אַתָּה אֱמֶת לָהֶם כְּמוֹ אִישׁוֹן.

וּבְעֵת יְרוּצוּן עַל נְתִיב הַיְקָר

אַתָּה לְבַדָּךְ תַּעֲבֹר רִאשׁוֹן.

וַאֲשֶׁר יְמַלְאוּן הֵם בְּמַעֲשֵׂה יָד

אַתָּה תְמַלֵּא מִדְּבַר לָשׁוֹן.

וְעָשִיתִי לְאִישׁ מְשַׁלֵּחַ מְדָנִים. רְאִיתִיו וְהוּא מִתְפַּלֵּל בַּעֲנָוָה וְתַחֲנוּנִים:

רְאוּ זָקֵן עוֹלֶה לְהַפִּיל תַּחֲנוּנִים

בְּלֵב יֵקְבָּץ-אוֹן וּמָלֵא תַזְנוּנִים.

בְּלָבָן מִתְעַטֵּף כְּשׁוֹבַךְ הַיּוֹנִים

מְלֻבָּן מִבַּחוּץ מְטֻנָּף מִבִּפְנִים.

וְעָשִׂיתִי לְאִישׁ יָצָא לוֹ שֵׁם בְּנָדָבה. וְיָדוֹ מִטַּל נְדָבָה חֲרֵבָה:

אֶתְמַהּ לְעָבֵי טוּבְךָ אֵיךְ נִכְלְאוּ

דּוֹדֵי וְנוֹזְלֵי יָדְךָ אֵיךְ קָפְאוּ.

אַתָּה כְּמוֹ מָחוֹל אֲשֶׁר לוֹ קוֹל מְאֹד

גָּדוֹל. וּבוֹקְעָיו רִיק וְרוּחַ מָצְאוּ.

וְעָשִׂיתִי לְאִישׁ מֵאַנְשֵׁי כַלְנַה. הִלַּלְתִּיו וּבָרַח לְחָרָן מִמֶּנִּי וְנִסְתָּר מֵעֵינַי:

יְלִיד זֻלּוּת. אֲחִי נַבְלוּת וְכִילוּת

אֲשֶׁר כָּל הַנְּבָלוֹת נִקְבְּצוּ בוֹ.

וְהִלַּלְתִּי שְׁמוֹ בִשְׁתֵּי לְשׁוֹנוֹת

וְנִכְשַׁלְתִּי בְּמַאְמַר פִּי וְנִיבוֹ.

וְאוָּלם עֵת דְּרַשְׁתִּיו לֹא מְצָאתִיו

וְאָמְרוּ כִּי לְחָרָן שָׂם נְתִיבוֹ.

וְאִם נֶחְבָּא וְנִסְתַּר מִלְּפָנַי –

שְׁמוֹ הָרַע בְּפִי עֵטִי עֲזָבוֹ.

כְּמוֹ עַכְבָּר אֲשֶׁר נְכִנַס בְּתוֹךְ חוֹר

וְעָזַב חוּץ בְּפִי חָתוּל זְנָבוֹ.

וְעָשִׂיתִי בְכַלְנַה עַל אִישׁ מִן הַנְּבָלִים הַשְּׁוָרִים. עוֹשֶׂה שִׁירִים. וְחוֹצֵב בּוֹרוֹת נִשְׁבָּרִים. וּמֵימָיו מָרִים. וּשְׁמוֹ בְּרָכוֹת:

קְלָלוֹת הוּא. מְחַבֵּר שִׁיר כְּצוֹאָה

וְהַצּוֹאָה בְּפִיו יַחְשֹׁב כְּנֹפֶת.

וְהֶגְיוֹנוֹ יְגַלֶּה אֶת קְלוֹנוֹ

וְעֶרְוָתוֹ בְּשִׁירוֹ נֶחֱשֶׂפֶת.

וְאָבִיו הוּא מְשֻׁעְבָּד לַנְּבָלוֹת

וְאִמּוֹ לְחֲרָפוֹת נֶחֱרֶפֶת.

וְנֶאְסְפוּ לְעוֹלָמָם שְׁנֵיהֶם

וְחֶרְפָּתָם לְאוֹת לֹא נֶאֱסֶפֶת.

וְעֵד הַבֵּן עֲלֵי טֻמְאַת אֲבוֹתָיו

כְּרֵיחַ רָע עֲלֵי בֵית הַטְּנֹפֶת.

וְשָׁאַלְתִּי מִמֶּנּוּ סֵפֶר. כֹּל מְשָׁלָיו מִשְׁלֵי אֵפֶר. וְתֵכֶף אֲשֶׁר שְׁלָחוֹ. בָּא אֵלַי וּלְקָחוֹ. וְעָשִׂיתִי עָלָיו:

קְלָלוֹת בָּא וְהִשְׁאִיל לִי כְתָבוֹ

בְּיוֹם אֶחָד. וּבַשֵּׁנִי גְזָלוֹ.

חֲשַׁבְתִּיהוּ בְּעֵת תִּתּוֹ וְקַחְתּוֹ

כְּכֶלֶב שָׁב לְקִיאוֹ וַאֲכָלוֹ.

וְעָשִׂיתִי לְאִישׁ כִּילַי אֲשֶׁר הִבְטִיחַ אוֹתִי בְנִדְבָתוֹ וְנָדַר נְדָרִים. וַיּוֹצֵא פֶרַח וַיָּצֵץ צִיץ וַיִּגְמֹל שְׁקָרִים.

בְּהִתְנַדֵּב בְּפִיו כִּילַי

וְיַחְשֹׁב חֵטְא אֲשֶׁר בָּטָא.

יְהִי קוֹפֵץ שְׁתֵּי יָדָיו

וְנִסְלַח לוֹ אֲשֶׁר חָטָא.

וְעוֹד עָלָיו עָשִׂיתִי. וְזֶה חָזִיתִי:

כִּילַי הֲתַעְלִים מוּמְךָ

הִנֵּה בְיָדָךְ עֵדְךָ.

כִּילוּת בְּךָ דָבַק עֲדֵי

הָיָה לְאוֹת עַל יָדְךָ.

וְעָשִׂיתִי עַל הַיַּיִן. הַנֶּחְמָד לְמַרְאֵה עַיִן:

וְכָל-חָדָשׁ מְאֹד יֶעְרַב וְיִנְעַם

וְעֵת יִבְלֶה יְקוּצוּן בּוֹ נְפָשִׁים.

וְיַיִן כֹּל אֲשֶׁר יִבְלֶה וְנוֹשָׁן

מְשַׂמַּח אֱלֹהִים וַאֲנָשִׁים.

עוֹד עַל הַיַּיִן:

וּבֶן שׂוֹרֵק חֲשַׁבְתִּיו נֶאֱמָן לִי

וְהוּא יַפְרִישׁ כְּמוֹ נָחָשׁ וְיַשִּׂיא.

אֲהַבְתִּיהוּ וָּבַגד בִּי בְעָרְמָה

וְיִמְשֹׁךְ אֶת לְבָבִי לוֹ וְנַפְשִׁי.

אֲנִי שַׂמְתִּיו הֱיוֹת שָׁמוּר בְּקִרְבִּי

וְהוּא עָלָה הֱיוֹת שׁוֹמֵר לְרֹאשִׁי.

וְעָשִׂיתִי בַחֵשֶׁק בִּימֵי הַנְּעוּרִים. וְתַעֲנוּג הַבַּחוּרִים:

צְבִי חֵן הוּא דְמוּת יוֹסֵף בְּיָפְיוֹ

וְעַל רֹאשׁוֹ עֲטֶרֶת הוֹד כְּחֵלֶם.

וְנַפְשִׁי הִיא לְכַף רַגְלוֹ כְּחִירִק

וְהוּא גֵאֶה וְרָם עָלַי כְּחֹלֶם.

וְנִגְנַבְתִּי לְגַן פָּנָיו כְּגַנָּב

לְלַקֵּט מִגְדְּנֵי פִיו וָאֹכְלֵם.

וְעָלַי זָרְחָה שֶׁמֶשׁ לְחָיָיו

וְדָמִים לִי וְהוֹן רוּחִי אֲשַׁלֵּם.

וְעַל זֶה הָעִנְיָן. בָּנִיתִי לוֹ בִּנְיָן:

עֵינֵי צְבִיָּה דָרְכיּ לִי קֶשֶׁת

עַל כֵּן שְׂעִפַּי יָרְאוּ מִגֶּשֶת.

נַפְשִׁי כְמוֹ צִפּוֹר בְּרִשְׁתָּהּ נִלְכְּדָה

אַךְ אַהֲבָתָהּ הָיְתָה הָרֶשֶׁת.

עֵינָהּ דְּמוּת קֶשֶׁת אַבָל פָּרְשָׂה

עָנָן וְעָלָיו נִרְאֲתָה הַקֶּשֶׁת.

עוֹד עַל הַחֵשֶׁק:

סוֹבֵב חוֹשֵׁק עַל יַד הַיְאֹר

וּכְאֵב חִשְקוֹ פִצְעֵי זָיִן.

וַיַּרְא עַלְמָה הִיא רוֹחֶצֶת

מַלְבּוּשׁ דּוֹדָהּ עַל הָעָיִן.

וּנְשָׁקָהוּ לֹא לִכְבוֹדוֹ

רַק לִכְבוֹד חַכְלִילִי עָיִן.

כֵּן יֵשׁ אָדָם סוֹבֵא יִשְׁתֶּה

הַחֹמֶץ לִכְבוֹד הַיָּיִן.

וּבְעִנְיַן הַדְּמָעוֹת. הַיּוֹרְדוֹת מֵאֵשׁ הַצְּלָעוֹת. אָמַרְתִּי:

אַסְתִּיר צְפוּן חֵשֶׁק בְּלִבִּי אַךְ

דִּמְעִי כְּנֵס נִגְלָה בְרֹאשׁ דֶּגֶל.

לִבִּי עֲלֵי נַחַל אֲמָנָה הוּא

שׁוֹכֵן אֲבָל דִּמְעִי בְעֵין רוֹגֵל.

וְאָמַרְתִּי עַל הָעִנְיָן. לְהָקִיר נוֹזְלֵי שִׁיר מִן הַמַּעְיָן:

הָהּ נִגְלָה סוֹד חִשְׁקִי, אַחַר

הָיָה בִכְלוּא לִבִּי לָחוּץ.

דִּמְעִי רָגַל אֶל עֵינַי וַ-

יַּגֵּד לִשְׁנֵי אֶחָיו בַּחוּץ.

וְעָשִׂיתִי עַל אִשָּׁה יְוָנִית אֲשֶׁר לִקְרֹא תוֹרָה יוֹדַעַת. וְכָל אִישׁ לְמִשַּׁכב זָכָר יוֹדָעַת:

שְׁאֵלוּנִי בְּעַד אִשָּׁה חֲכָמָה

עֲנִיתִים לָמְדָה חָכְמָה לְבָשְׁתָּהּ.

מְזִמָּתָהּ לְזִמָּתָהּ לְמוֹפֵת

וְסִמָּן נֶאֱמָן דַּעְתָּה לְדַעְתָּהּ.

וְעוֹד עָשִׂיתִי עַל עִנְיָנֶיהָ. לְגַלּוֹת תַּזְנוּנֶיהָ. וְהָיָה שְׁמָהּ טְהוֹרָה:

וְאִשָּׁה חֲכָמָה לְהָרַע עֲרוּמָה

נְתִיב חֵטְא תְּסַקֵּל וְישֶׁר תְּעַקֵּל.

וְקָרְאוּ שְׁמָהּ יוֹלְדֶיהָ טְהוֹרָה

וְכֵן הִיא. וְאָכֵן כְּטָהְרַת יְחֶזְקֵאל.

וְחִבַּרְתִּי סֵפֶר גּוֹרָלוֹת. לְאִישׁ מֵחֲשׁוּבֵי הַקְּהִלּוֹת. וְכָתַבְתִּי אֵלֶּה הַשִּׁירִים:

שַׂר הַיְקָר אַתָּה נְסִיכֵנוּ

וּנְשִׂיא אֱלֹהִים אַתְּ בְּתוֹכֵנוּ.

הִנֵּה לְךָ סֵפֶר אֲשֶׁר מִלָּיו

יֹאמְרוּ: לְכוּ נִכְרַע לְמַלְכֵּנוּ.

קוֹרְאִים לְךָ קוּמָה מְשָׁל-בָּנוּ

כִּי שִׂכְלְךָ יָאִיר לְחָשְׁכֵּנוּ.

דִּבְרֵי אֱמֶת תִּמְצָא בְּפִינוּ יוֹם

גּוֹרָלְךָ תַּפִּיל בְּתוֹכֵנוּ.

וְעָשִׂיתִי לְנָשִׂיא כִילַי רְאִיתִיו דּוֹאֵג עַל הָעָב הַנֶּעֱצֶרֶת. בִּשְׁנַת הַבַּצֹרֶת:

אַל תֶּחֱרַד עַל הַנְּשִיאִים שַׂר-יְקָר

אִם נוֹזְלִים לֹא יָצְקוּ עָלֶיךָ.

כִּי מִמְּךָ לָמְדוּ קְפֹץ יָד כִּי שְׁמָם

כִּשְׁמָךְ וּמִפְעָלָם כְּמִפְעָלֶיךָ.

וְעָשִׂיתִי עַל הֶחָכָם רַבִּי יוֹסֵף הַמַּעֲרָבִי. בְּנָסְעִי לְצוֹבָה מֵאֶרֶץ הַצְּבִי:

בְּךָ הוֹבִישׁ אֱלֹהִים יָם דְּמָעוֹת

בְּךָ אָמַר לְצוּלָתָם חֳרָבִי.

בְּלִבָּךְ בֵּית זְבוּל בָּנָה לְשִׁבְתּוֹ

לְשׁוֹן קֹדֶשׁ וְיַהֵל שָׁם עֲרָבִי.

וְהָיִיתָ גְבִיר וָרָב בְּמַעְרָב

וּבַמִּזְרָח מְשָׁחָךְ אֵל לְנָבִיא.

וְאָמְנָם כִּי אֲרַם צוֹבָא כְּמִקְדָּשׁ

וְאַתָּה בָהּ כְּמוֹ נֵר מַעֲרָבִי.

וְעָשִׂיתִי עַל הַקְּהִילָּה. הַמּוֹאֶסֶת הַתְּהִלָּה:

שַּׁדַּי שְׁלַח צוֹעִים לְרוֹשֵׁשׁ כָּל-

טוֹבַת בְּנֵי כַלְנֵה וְכַלֶּנָּה.

עָשְׁרָהּ וְכָל-חֵילָהּ לְזָרִים תֵּן

וַחֲשׂף יְפִי סִתְרָהּ וְגַלֶּנָּה.

אִם כַּסְּפִינָה רָחֲבָה. קַצֵּר

אוֹתָהּ וְאֶל אַמָּה תְּכַלֶּנָּה.

וְעָשִׂיתִי עַל בְּנֵי עֲדִינָה. אֲשֶׁר הָיוּ בִימֵי קֶדֶם פְּאֵר כָּל מְדִינָה. וְעַתָּה נוֹתְרוּ לְמָשָׁל וְלִשְׁנִינָה:

אַנְשֵׁי עֲדִינָה נִבְזִים שְׁפָלִים

מִבְחַר הֲמוֹנָם דֹּמֶן וְאַשְׁפָּה.

לֹא אַחֲרִישׁ עַד אָשִׁית לְבָבָם

לִהְיוֹת לְחִצֵּי שִׁירַי כְּאַשְׁפָּה.

הֵן אִם תְּמוֹל בָּם הָיוּ אֲצִילִים

הַיּוֹם יְקָרָם אָבַד וְנִסְפָּה.

הַיּוֹם בְּשַׁחַת וּתְמוֹל בְּשַׁחַק

הַיּוֹם בְּאַשְׁפָּה וּתְמוֹל בְּחֻפָּה.

הֵם לֹא לְעֵזֶר גַם לֹא לְהוֹעִיל

כִּי אִם לְבשֶׁת גַּם כֵּן לְחֶרְפָּה.

וְעָשִׂיתִי לְאִישׁ כִּילַי מִכֹּל אֲשֶׁר לְפָנָיו. וְהוּא נֶחְשַׁב נָדִיב בְּעֵינָיו:

וְרָאִיתִי אֱנוֹשׁ כִּילַי וְלִפְנֵי

נְדִיבִים הוּא בְעֵינָיו רֹאשׁ לְכֻלָּם.

וְלֹא יַעְלֶה לְמַעְלוֹת הַנְּדָבָה

עֲדֵי יוּכַל חֲמוֹר לַעְלוֹת בְּסֻלָּם.

וְעָשִׂיתִי עַל אֲנָשִׁים הִלַּלְתִּים בְּשִׁירִים. וְלֹא הֵבִינוּ דַרְכָּם. וְלֹא יָדְעוּ עֶרְכָּם:

תְּשׁוּרַת שִׁיר שְׁלַחְתִּיהָ לְאִישִׁים

חֲסֵרֵי בִין וְשֵׂכֶל עִם תְּמִימוּת.

וְחָשְׁבוּ אֶת תְּהִילָתִי לְקָלוֹן

וְהָרַע יַחְשְׁבוּ ישֶׁר וְתַמּוּת.

וְכֻלָּם בַּנְּבָלוֹת הֵם שְׂמֵחִים

וְאָכֵן יִמְאֲסוּ דִבְרֵי נְעִימוּת.

וְדִמְיוֹנָם כְּרֶמֶשׂ הַגְּלָלִים

אֲשֶׁר בַּמּוֹר יְהִי אָבוּד וְצָמוּת.

תְּנָה אוֹתוֹ בְתוֹךְ צוֹאָה וְיִחְיֶה

וְשֶׁמֶן רֹאשׁ מְשַח אוֹתוֹ וְיָמוּת.

וְעָשִׂיתִי לְאִישׁ נָדִיב שָׁלַח לִי מִנְחָה. וּבַיּוֹם הַשֵּׁנִי כָפַל לִי הָאֲרֻחָה:

נְדִיב הַלֵּב אֲשֶׁר הֶעְשִׁיר בְּנֵי אִישׁ

וְלֹא נִמְצָא בְיָמָיו דַּל וּמִסְכֵּן.

שְׁתֵּי יָדָיו שְׁתֵּי צָרוֹת מְקַנְּאוֹת

וְזֹאת תֵּיטִיב וְהַשֵּׁנִית תְּתַכֵּן.

בְּעֵת תִּגְמֹל נְדָבוֹת יַד יְמִינוֹ

תְּקַנֵּא הַשְּׁנִיָּה לַעֲשׂוֹת כֵּן.

וְעָשִׂיתִי לְאִישׁ מִנְּדִיבֵי סְפָרַד כָּמוֹהוּ בִנְדָבָה לֹא נִהְיָה. שְׁמוֹ רַבִּי חִיָּא:

שְׂאוּ שָׁלוֹם לִגְבִיר. יְקָר עִם תֹּם הֶחְבִּיר

וְזִכְרוֹ קוֹל יַעְבִיר נְדִיב הַדּוֹר חִיָּא.

לְמֵי חָכְמוֹת צָמֵא וְלַעֲשׂוֹת טוֹב הוֹמֶה

עֲרֵמוֹת הַיְקָר מֵעֲפַר שִׁפְלוּת חִיָּה.

וְלוּ מִנִּדְבָתוֹ יְאַו אִישׁ נִשְׁמָתוֹ

יְחַיֶּה זוּלָתוֹ וְנַפְשׁוֹ לֹא חִיָּה.

וְעָשִׂיתִי בְעִנְיַן הַחֵשֶׁק. בְּעֹפֶר אֲשֶׁר עֵינָיו כְּלֵי נֶשֶׁק:

צְבִי הַחֵן אֲשֶׁר שָׁכֵן בְּלִבִּי

שְׂרָפָהוּ יְקוֹד צַלְעִי וּמֵעַי.

וְלוּלֵי כִי הֲרָגַנִי בְעֵינָיו

וְהִרְבָּה עַד מְאֹד חִנָּם פְּצָעַי.

נְתַתִּיהוּ כְּבַת עֵינִי וְשַׂמְתִּיו

כְּעֵץ שָׁתוּל עַל פַּלְגֵי דְמָעַי.

וְעַל רִשְעוֹ נְתַתִּיהוּ בְלִבִּי

עֲדֵי שָׁכַן חֲרֵרִים בֵּין צְלָעַי.

וְעָשִׂיתִי עַל אִישׁ הָיְתָה לוֹ שִׁפְחָה. וְהוּא וְאָבִיו הָיוּ מִתְעוֹלְלִים בָּהּ. זֶה הוֹלֵךְ וְזֶה בָּא:

שֹׁמוּ מְתֵי צֶדֶק וְיוֹשְׁבֵי שָׁעֲרָה

כִּי גָבְרָה זִמָּה וּפִיהָ פָעֲרָה.

אֵיךְ מוֹסְדוֹת שַׁחַק לְזֹאת לֹא יִרְגְּזןּ

כִּי אִישׁ וְאָבִיו יֵלְכוּ לַנַּעֲרָה.

וְעָשִׂיתִי בְאִישׁ רָאִיתִי בְרֹאשׁוֹ מִצְנֶפֶת עֲצוּמָה. כְּפִי מָאתַיִם אַמָּה:

וּמִצְנֶפֶת בְּרֹאשׁ נָבָל קְשׁוּרָה

חֲשַׁבְתִּיהָ כְלִי מָלֵא כְלִמָּה.

מְאֹד גָּדְלָה וְהִיא עוֹמְדָה עֲלֵי רִיק

כְּמוֹ תֵבֵל תְּלוּיָה עַל בְּלִימָה.

וְעָשִׂיתִי עַל אִישׁ מְרוּחַ אָשֶׁךְ. וְנָשָׂא אִשָּׁה יָפָה זִיוָהּ יָאִיר בַּחֹשֶׁךְ:

חֲבֹק דּוֹדִי שְׁדֵי כַלָּה נְעִימָה

וְלֹא יֶחְסַר לְךָ כָּל תַּעֲנוּג טוֹב.

עֲלֵה עִמָּהּ עֲלֵי עֶרֶש אֲהָבִים

עֲדֵי תִהְיֶה כְגַן רָוֶה וְרָטֹב.

רְאֵה בֵיצָתְךָ עִמָּהּ גְרוּרָה

וְהִיא בֵיצָה אֲשֶׁר נוֹלְדָה בְיוֹם טוֹב.

וְעָשִׂיתִי עַל הַשֵּׂיבָה. אֲשֶׁר בִּימֵי נְעוּרִים זָרְקָה בִי. וּכְנֶגַע פָּרְחָה בְלִבִּי:

מוֹדֶה אֲנִי מַעֲשֵׂה מְתֵי עוֹלָם

הַלּוֹבְשִׁים לָבָן בְּיוֹם אֶבְלָם.

כֵּן שַׂעֲרוֹתַי לָבְשׁוּ לָבָן

עַל נוֹד יְמֵי נֹעַר וְסוּר צִלָּם.

וְעוֹד בָּעִנְיָן הַזֶּה עָשִׂיתִי. וְלָלֶכֶת בְּדַרְכָּהּ נִסִּיתִי:

דָּאוּ כְרֶגַע עוֹרְבֵי קַדְרוּת אֲשֶׁר

קֶדֶם בְּשַׁעַר שַׂעֲרִי שָׁקָדוּ.

בָּאוּ וְתוֹךְ קִירוֹת לְבָבִי קִנְּנוּ

מִיּוֹם אֲשֶׁר מִשַּׂעֲרִי נָדָדוּ.

וְעָשִׂיתִי בִימֵי הַנֹּעַר. בְּעִנְיַן רֹב הַשַּׂעַר:

אִישִׁים יְרִיבוּנִי עַל רֹב שְׂעָרִי

אָמְנָם כְּסִילוּתָם שְׁחִין פּוֹרֵחַ.

כִּי לוּ אֱלִישָׁע אָז יְהִי כֵן הֵן אֲזַי

לֹא קָרְאוּ אַחֲרָיו עֲלֵה קֵרֵחַ.

וְהֶעְתַּקְתִּי בִּסְפָרַד לִלְשׁוֹן הַקֹּדֶשׁ סֵפֶר מוֹרֵה הַנְּבוּכִים. לְאֶחָד מִן הַנְּסִיכִים. וְהָיָה לִרְאוֹתוֹ כוֹסֵף. וּשְׁמוֹ יוֹסֵף. וּשְׁלַחְתִּיהוּ לוֹ עִם אֵלֶּה הַשִּׁירִים:

שָׁלוֹם לְךָ עוֹרֵךְ קְרָב חָכְמָה

כִּי בָהּ זְמָמָיו יוֹם וְלֵיל נִצִּים.

עַד קִנְּאוּ בוֹ יוֹצְאֵי צָבָא

וַיִּשְׂטְמוּהוּ בַעֲלֵי חִצִּים.

אִם תֶּאֱהַב לַעְלֹז בְּגַן חָכְמָה

הִנֵּה עֲרוּגָה מָלְאָה צִצִּים.

אוֹ לִבְּךָ תַקְרִיב לְאֵל קָרְבָּן

הִנֵּה לְךָ הָאֵשׁ וְהָעֵצִים.

וְעָשִׂיתִי בְעִיר כַּלְנֵה לְאִישׁ לָקַח הַבּוּז תְּרוּמִיָּה. וּשְׁמוֹ חִזְקִיָּה:

אִמְרוּ לְרֹאשׁ הָעָם יְחִזְקִיָּה

יָחִיד בְּרָאָךְ רַב עֲלִילִיָּה.

כִּמְעַט וְיֹאבְדוּ הַנְּבָלוֹת. אַךְ

לָהֶם שְׁלָחָךְ יוֹצְרָך מִחְיָה.

אֵינְךָ יְחִזְקִיָּה. וְחָלִילָה

כִּי אִם מְנַשֶּׁה בֶּן יְחִזְקִיָּה.

וְעָשִׂיתִי בְּאַרְצוֹת מַעֲרָב לְאִישׁ טִפְסָר. לְלַמְּדוֹ דֶרֶךְ מוּסָר.

דּוֹרֵשׁ קְנוֹת מוּסָר קְחָה כָל-יוֹם רְאִי

מוּצָק לְהַבִּיט מַחֲזֵה צַלְמֶךָ.

כִּי אִם יְהִי נָאֶה הֱיֵה נִשְׁמָר לְבַל

תַּחְסִיר בְּחֶסְרוֹן פָּעֳלָךְ תֻּמֶּךָ.

אוֹ אִם יְהִי בוֹ מוּם הֱיֵה נִשְׁמָר לְבַל

מוּם חֶטְאֲךָ תּוֹסִיף עֲלֵי מוּמֶיךָ.

אִם תֵּט לְנִיבִי אוֹיְבֶיךָ תַחְלשׁ

וּבְרֹב גְּאוֹנָךְ תַּהֲרֹס קָמֶיךָ.

וְעָשִׂיתִי בְצָרְפַת לְאִישׁ חָנֵף מִכָּל עַמּוֹ. וְיִצְחָק הַלֵּוִי שְׁמוֹ:

אֱמוּנָה אֵין בִּלְתָּהּ. בְּרוּם שַׁחַק שִׁבְתָּהּ

שְׁבִי נַפְשִׁי אִתָּהּ. וְעִמָּהּ הִלָּוִי.

יְקָר וָהוֹד יֶשׁ-בָּהּ. וְהָאוֹת עַל טוּבָהּ

אֲשֶׁר אֵין חֵלֶק בָּהּ. לְיִצְחָק הַלֵּוִי.

וְעָשִׂיתִי בְּטֻלִיטוּלָה עַל הַנָּשִׂיא רַבִּי יוֹסֵף בֶּן שׁוֹשָׁן כְּשֶׁנִּפְטַר לְבֵית עוֹלָמוֹ:

שָׁנָה אֲשֶׁר בָּהּ יְהוֹסֵף הַגְּבִיר

נֶאְסַף הֲכִי אַחֲרָיו צֹאן אוֹבְדוֹת נֶהִי.

הָיְתָה שְׁנַת אֵיד שְׁנַת זַעַם וְעֵת

יָגוֹן וְהָאֵל שְׁמָהּ קָרָא שְׁנַת נֶהִי.

וְקֹדֶם מוֹתוֹ חֹדֶשׁ אֶחָד אוֹ שְׁנַיִם. שָׁמַעְנוּ כִּי יֵשׁ שֶׁבֶר בְּמִצְרַיִם. וְכִי משֶׁה עָלָה לַשָּׁמַיִם. וָאַעַן וָאֹמַר:

שֶׁבֶר מֵעַל שֶׁבֶר נִקְרָא

לִבּוֹת שׁוֹמְעָיו כָּאְרִי שִׁסֵּף.

כִּי אַחְרֵי מוֹת משֶׁה גָוַע

הַשַּׂר יוֹסֵף וַיֵּאָסֵף.

לִשְׁכֹּן בִּמְחִיצָתוֹ בָא וַ-

יִּקַח משֶׁה עַצְמוֹת יוֹסֵף.

וְעָשִׂיתִי בְּעִנְיְנֵי תֵבֵל. אֲשֶׁר הֵם נוֹבְלִים כְּמוֹ ציץ נוֹבֵל:

וְרוֹאֶה בִלְבָבוֹ. זְמַן רַע וּנְתִיבוֹ

בְּנִקְרַת צוּר יָבֹא. וְנֶחְבָּא בִּמְחִלָה.

וְתֵבֵל כָּמִּסְחָר. וְכָל אִישׁ בָּהּ יִתְחָר.

וּמַה זֶּה בָהּ אֶבְחַר. וְהִיא רָעָה חוֹלָה

וְסֻכַּת טוֹבָתָהּ. אֲנִי פוֹסֵל אוֹתָהּ

לְמַעַן חַמָּתָהּ. מְרֻבָּה מִצִּלַּהּ.

וְעָשִׂיתִי עַל אֶחָד מִן הַנְּבָלִים. גֵּאֶה בְהוֹנוֹ עַד כּוֹכְבֵי הַכְּסִילִים:

בִּרְאוֹתְךָ נָבָל אֲשֶׁר

עָלָה וְשָׂם בִּזְבוּל דְּבִיר.

פֶּתַע יְהִי נִשְׁבָּר וְשַׁח

כִּי אַחֲרֵי דַרְגָּא תְּבִיר.

וְעָשִׂיתִי שִׁיר מוּסָר וְתוּשִׁיָּה. בְּעִנְיְנֵי הַמִּחְיָה:

אַל תִּהְיֶה עָצֵל לְבַקֵּשׁ לָךְ

מִחְיָה וְשִׂים קִנָּךְ מְלוֹן אוֹרְחִים.

כּי לֹא תְהִי מוֹצֵא מְנוּחָה בִמְ-

נוּחָה וּבַעְלִיּוֹת מְרֻוָּחִים.

הַבֵּט לְשׁוֹן קֹדֶשׁ וְשׁוּר כִּי לֹא

יִתְקַבְּצוּ תוֹכוֹ שְׁנֵי נָחִים.

וּפַעַם הֱקִיצוֹתִי בְּלֵיל שַׁבָּת מִפַּרְעוֹשִׁים. נוֹשְׁכִים כִּנְחָשִׁים. וָאַעַן וָאֹמַר:

לִבִּי לְפַרְעוֹשׁ חִלֵּל יְצוּעִי

לִרְווֹת בְּדָמַי כָּל-תַּעֲנוּגוֹ.

לֹא אֶשְׁקְטָה עַד אָשִׁיב גְּמוּלוֹ

וּבְיַד עֲוֹנוֹ אֲמוּגוֹ כַּיּוֹם.

מוֹעֵד וְשַׁבָּת לֹא-יַעֲצָר-בִּי

לַהְפֹךְ לְאֵבֶל חָדְשׁוֹ וְחַגּוֹ.

אָמְרוּ חֲבֵרַי: הֵן הַחֲכָמִים

אָסְרוּ בְּיוֹם הַשַּׁבָּת לְהָרְגוֹ.

וָאוֹמְרָה: הֵם אָמְרוּ תְחִלָּה

הַבָּא לְהָרְגָךְ הַשְׁכֵּם לְהָרְגוֹ.

וְעָשִׂיתִי בְחֹדֶשׁ תֲּעִנית יִשְׁמְעֵאלִים.

כְּשֶׁשְּׁמַעְתִּים מַשְׁכִּימִים בַּעֲלוֹת הַשַּׁחַר וְאוֹכְלִים:

הָאֵל עִנֵּג נְפָשׁוֹת עַמּוֹ

בַּמַּאְכָלִים. כִּי לֹא עִנָּה.

לֹא יַרְעִיב אֵל נֶפֶשׁ צַדִּיק

רַק יַשְׂבִּיעַ נֶפֶשׁ נַעְנָה.

כִּי לֹא צִוָּה עַנּוֹת אָדָם

נַפְשׁוֹ רַק אַחַת בַּשָּׁנָה.

שָׁוְא לָכֶם כָּל-מַשְׁכִּימֵי קוּם

כֵּן יִתֵּן לִידִידוֹ שֵׁנָה.

וְעָשִׂיתִי עַל הַנָּשִׂיא הַגָּדוֹל רַבֵּנוּ שֵׁשֶׁת אֲשֶׁר הָיָה רֹאשׁ נְדִיבֵי עוֹלָם, וְאָסַף הַחֲמוּדוֹת כֻּלָּם:

אִם תִּשְׁאֲלוּ עַל הַגְּבִיר שֵׁשֶּׁת

הוּא בַזְּמַן נִשְׁאַר לְרֹאשׁ פִּנָּה.

נָגַע עֲדֵי זִקְנָה וְנִדְבָתוֹ

לֹא בָלְתָה. אַךְ יָסְפָה עֶדְנָה.

וְעָשִׂיתִי בְעִנְיַן הָאוֹהֲבִים: הַנְּעִימִים וְהַנֶּאֱהָבִים:

מֶרְחַב אֲרָצוֹת צַר וְלֹא

יָכִיל שְׁנַיִם אוֹיְבִים.

כַּאֲשֶׁר מְקוֹם זֶרֶת מְאֹד

רָחָב לְאֶלֶף אוֹהַבִים.

וְעָשִׂיתִי עַל קְהַל לוּנִיאֵל. יִשְׁמְרֶנּוּ הָאֵל:

שְׂפַת יֶתֶר הֲתַשִּׂיג מַהֲלַל עָם

אֲשֶׁר שִׂכְלוֹ בְּיִרְאַת אֵל יְרִיחוֹ.

וְשׁוֹלֵחַ דְּבַר חָכְמָה אֲלֵיהֶם

וְעוֹרֵךְ אֶל מְאוֹר שִׁמְשָׁם יְרֵחוֹ.

וְשׁוֹלֵחַ רְסִיסֵי טַל לְיַם-סוּף

וּמֵבִיא שַׁי תְמָרִים אֶל יְרִיחוֹ.

וּבִהְיוֹתִי בְלוּנִיל בִּקֵּשׁ מִמֶּנִּי ר' יְהוֹנָתָן הַכֹּהֵן זַ"ל לְהַעְתִּיק לוֹ סֵדֶר זְרָעִים. לְר' משֶׁה אַבִּיר הָרוֹעִים. וְהֶעְתַּקְתִּיו וְאֵלָיו שְׁלַחְתִּיו עִם אֵלֶּה הַשִּׁירִים:

יְצַו אֵל שְׁלוֹמִים. לְחֶבְרַת תְּמִימִים.

חֲכָמִים נְעִימִים. נְבוֹנִים יְדוּעִים.

חֲשֵׁכָה כְאוֹרָה. יְשִׁיבוּן בְּתוֹרָה.

וְהִיא כַמְּנוֹרָה. וְהֵם לָהּ גְּבִיעִים.

אֲקַדֵּם גְּבוֹהִים. וְאוֹרֵי נְגוֹהִים.

וְכֹהֲנֵי אֱלֹהִים. בְּמִנְחַת זְרָעִים.

וְכָתַבְתִּי לַרַבִּי יוֹסֵף מְקוֹר הַתּוּשִׁיָּה. בֶּן הַשַּׂר הַמִּצְרִי ר' עוֹבַדְיָה זַצַ"ל:

הַשַּׂר, לְךָ חָכְמָה וּבִין יִתְיַחֲשׂוּ

וּבְתוֹךְ לְבָבָךְ נִמְצְאוּ יוֹם חֻפְּשׂוּ.

וּבְיוֹם בְּרוֹא הָאֵל חֲמוּדוֹת הַזְּמַן

וַיְשׁוֹטְטוּ לִמְצֹא בְּצֵלִ מי יֶחֱסוּ

אָמַר אֱלֹהִים: מַהֲרוּ וּלְכוּ אֱלֵי

יוֹסֵף אֲשֶׁר יֹאמַר אֲלֵיכֶם תַּעֲשׂוּ.

וּבְעִנְיַן הַחֵשֶׁק עָשִׂיתִי אֵלֶּה הַשִּׁירִים. בִּימֵי הַנְּעוּרִים:

צְבִי הַחֵן אֲשֶׁר לֶחְיוֹ

בְּסַפִּירִים מְעֻלֶּפֶת.

וְעֵינָיו שׁוֹלְפִים עָלַי

בְּרַק חֶרֶב מְעוֹפֶפֶת.

שְׂפָתָיו צוּף וּפָנָיו גַּן

וְשִׁנָּיו דַּר וּפִיו נֹפֶת.

בְּגַן לֶחְיוֹ מְשׂוֹשׂ עֵינִי

בְּעֵת צוֹפָה וְנִשְׁקֶפֶת.

אֲבָל נַפְשִׁי בְּאֵשׁ חִשְׁקוֹ

צְרוּבָה הִיא וְנִשְׂרֶפֶת.

שְׁתֵּי עֵינַי בְּגַן עֵדֶן

אֲבָל לִבִּי בְּתוֹךְ תֹּפֶת.

וְעָשִׂיתִי בְּעִיר כַּלְנֵה עַל אִישׁ אֲשֶׁר בְּרָכוֹת שְׁמוֹ. וּנְבָלָה עִמּוֹ:

נָבָל אֲשֶׁר קִנֵּא לְנַבְלוּת יוֹלְדוֹ

עַד בַּנְּבָלָה נִמְשְׁלוּ כַּבֵּן כְּהוֹר.

נִקְרָא בְרָכוֹת אַךְ קְלָלוֹת הוּא שְׁמוֹ

כַּאְשֶׁר לְעִוֵּר יִקְרְאוּ סַגִּי נָהוֹר.

וְעָשִׂיתִי בַעֲדִינָה עַל אִישׁ מִן הָאֲצִילִים. אֲשֶׁר הָיָה מִתְעוֹלֵל בְּתַעֲלוּלִים:

עֲדִינָה בָּהּ אֲנָשִׁים כַּבְּהֵמוֹת

וְנִבְעָלִים לְפָרִיצִים נְבָלִים.

וְכָהֵם בּוֹעֲלִים כִּי הַבְּהֵמָה

וְהַכֵּלִים כְּרַגְלֵי הַבְּעָלִים.

וְכָתַבְתִּי לְאִישׁ מֵאֲצִילֵי הַדּוֹר וּגְבִירָיו. לְמַעַן אֵחֲרוּ לִי פַעְמֵי סְפָרָיו:

שְׂאוּ שָׁלוֹם לְדוֹד אִטֵּר וְאִלֵּם

שְׁפַל רוּחַ וְעָלַי גַּאֲוָתוֹ.

לְאִטֵּר מִכְּתֹב סֵפֶר לְדוֹדוֹ

וְאִלֵּם יַחֲשָׂךְ-לוֹ ניב שְׂפָתוֹ.

וְעָשִיתִי לְאִישׁ נָשָׂא אִשָּׁה קְדוֹשָׁה. וְהָיָה מִתְחַבֵּר לְכָל אִשָּׁה קְדֵשָׁה:

חוֹשֵׁק בְּנוֹת נֵכָר וְעוֹזֵב רַעֲיַת

חֵיקוֹ. עֲבֵרוֹתָיו גְּשׁוּ לִשְמֹעַ.

עִבּוּר בְּנוֹת קֹדֶשׁ בְּזָרוֹת יַחֲלִיף

כִּי הַדְּחוּיוֹת יַחֲשֹׁב לִקְבֹּעַ.

וְעָשִׂיתִי בְּעִנְיַן דִּמְעוֹת הָעֵינַיִם. הַנִּגְרָשׁוֹת כִּיאוֹר מִצְרַיִם:

זְמַן רָע הֶחֱשִׁיךְ עֵינֵי יְגוֹנָךְ

וְרָמָה יָדְךָ בַּל יֶחֱזָיוּן.

שְׂעִפַּי, עַד לְמָתַי הַיְגוֹנִים

בְּכַנְפֵיכֶם בְּכָל עֵת יֶחֱסָיוּן.

וְשִׂמְחָתִי בְּבַת עֵינִי תְּמִיתוּן

לְרֹב בִּכְיִי – אֲבָל הַבַּת תְּחַיּוּן.

וְשָׁלַחְתִּי מִדַּמֶּשֶׂק לַחֲמָת אֶל הָרוֹפֵא רַבִּי משֶׁה בַר צְדָקָה:

גְּבִיר הַדּוֹר, גְּדוֹר פִּרְצָךְ וְגִדְרָךְ

וְשׁוּב לִרְעוֹת שְׁאָר צֹאנָךְ וְעֶדְרָךְ.

וְאַל תִּשְׁכַּח יְדִידִים לָךְ יְקַוּוּ

וְיוֹחִילוּ דְבַר פִּיךָ וְנִדְרָךְ.

הַכִי אַתָּה כְּרוּב מִמְשַׁח וְלָכֵן

לְעִירָךְ שׁוּב וְשִׂים הַדְרָךְ בְּחַדְרָךְ.

וְעָשִׂיתִי עַל הַנְּשָׁמָה. וְעַל הַנֶּפֶשׁ הַחֲכָמָה:

חוֹנָה וְנִסְגֶּרֶת בְּבוֹר גֶּוְיָהּ

מִיּוֹם חֲנוֹתָהּ נִצְּלָה עֶדְיָהּ.

רָמָה וְתִתְחַבֵּר אֱלֵי רִמָּה

מִי יַעֲרֹךְ דָּפְיָהּ לְרֹב יָפְיָהּ.

יוֹם פָּשְׁטָה הַגּוּף אֲזַי לָבְשָׁה

עֶדְיָהּ וְהֵסִירָה כְּסוּת שִׁבְיָהּ.

זוֹרְחָה כְּשֶׁמֶשׁ עָלְתָה מִבֵּית

כֶּלֶא נְוֵה חשֶׁךְ וּמַאְפֵּלְיָה.

יוֹצְרָה בְּשַׁחַק הִלְּלָהּ כִּי כֵן

כֹּל הַנְּשָׁמָה הִיא תְהַלֵּל יָהּ.

וְעָשִׂיתִי עַל אֹרֶךְ הַגָּלוּת. וְרֹב שְׁנוֹת הַדַּלּוּת וְהַשִּׁפְלוּת:

חָסִין גְּבוּרַת יָדְךָ גַלֵּה

בַּיּוֹם וְאֵיךְ תּוּכַל לְהִתְעַלֵּם.

רָצוּץ בְּבוֹר גָּלוּת יְשׁוּעָתָךְ

יַחְלֹם כְּמוֹ עֶבֶד פְּדוּת חוֹלֵם.

יִשְׁאַל בְּעַד הַקֵּץ וְאֵין עוֹנֶה

יִדְמֶה לְעִוֵּר יִשְׁאֲלָה אִלֵּם.

זֶה לוֹ שְׁנַת אֶלֶף יְקַו אוּלַי

תִּפְקֹד עֲוֹן צָרָיו וְלֹא שָׁלֵם.

יָהּ שׁוּב וְהַהַרְאֵל לְיִשְׂרָאֵל

תָּשִׁיב וְיָבֹא יַעֲקֹב שָׁלֵם.

וְעָשִׂיתִי בְעִנְיַן בִּקּוּר בֵּית תְּפִלָּה. לַעֲרֹךְ לָאֵל תְּהִלָּה:

חֲצַר בֵּית קָדְשְׁךָ קַמְתִּי לְבַקֵּר

לְחַוּוֹת נָעֳמָךְ עֶרֶב וּבֹקֶר.

רְאֵה עָצְבִּי וְשׁוּר מַצְפּוּן לְבָבִי

אֲשֶׁר יִרְאֶה צְפוּן כָּל-לֵב וְחוֹקֵר.

יְקַו לִבִּי וְכָל-שִׂבְרִי בְךָ שָׂם

וְיֵדַע בֶּאֱמֶת כִּי לֹא תְשַׁקֵּר.

זְרֵה אוֹיְבַי כְּמוֹ דָוֶה וְעֵינַי

מְקַנְאַי מֵחֲווֹת טוּבִי תְנַקֵּר.

יְמִינָךְ שִׂים עֲלֵי גֶפֶן נְטַעְתָּהּ

וְאוֹתָהּ אַל תְּהִי נוֹתֵשׁ וְעוֹקֵר.

וְעָשִׂיתִי עַל אִישִׁים יַחֲלִיקוּ לְשׁוֹנָם. וְקֶבֶר פָּתוּחַ גְּרוֹנָם:

וְיֵשׁ אִישִׁים מְתוּקֵי נִיב וְשָׂפָה

וּבִלְשׁוֹנָם יְפַתּוּנִי וְנִיבָם.

יְגַלּוּ לִי פְנֵי רָצוֹן וְאַהְבָה

וְיָדַעְתִּי הֲכִי אָרְבָּם בְּלִבָּם.

וְאֵיך יוּכְלוּן צְפֹן רֹעַ לְבָבָם

וְהַכַּרַת פְּנֵיהֶם עָנְתָה בָם.

וְעָשִׂיתִי עַל עִנְיַן הַפְּרִידָה. אֲשֶׁר בָּהּ רַגְלִי לְכוּדָה:

לְמָתַי נָע וְנָד אֶהְיֶה כְקַיִן

הֲלוּם יָגוֹן וְלֹא שֵׁכָר וְיַיִן.

אֲקַו יֶשַׁע וְלֹא אֶחֱזֶה וְאוּלָם

מְצוּקִים אֶחֱזֶה עַיִן בְּעָיִן.

כְּמוֹ עִוֵּר בְּמַחְשַׁכֵּי כְלָאִים

חֲצוֹת לֵיל אֲקַו לָאוֹר וְאָיִן.

וְעָשִׂיתִי עַל אֲנָשִׁים גֵּאִים. וְהֵם רָעִים מִכָּל הַבְּרוּאִים:

יַכְרֵת אֱלֹהִים בַּעֲלֵי גָאוֹן

נִבְזִים וְעַד כּוֹכְבֵי זְבוּל גֵּאִים.

לוֹבְשִׁים בְּגָדִים הֵם חֲדָשִׁים אַךְ

הַמַּעֲשִׂים בָּלִים מִטֻלָּאִים.

יִדְמוּ לְרוֹאֵיהֶם אֱלֵי בִנְיָן

נָאֶה וְשָׁתוֹתָיו מְדֻכָּאִים.

רָעִים וְחָלִילָה הֱיוֹת כָּהֶם

אַנְשֵׁי סְדוֹם רָעִים וְחַטָּאִים.

וְעָשִׂיתִי עַל נְדִיבִים בְּפִיהֶם. וְקוֹפְצִים יְדֵיהֶם:

הָהּ רָאֲתָה עֵינִי אֲנָשִׁים הֵם כְּמוֹ

הַכַּרְמְלִי נָבָל וְכַאְחֵי קַיִן.

וּשְׁמָם נְשִׂיאִים אַךְ בְּנִדְבָתָם נְשִׂי-

אִים הֵם וְגַם רוּחַ וְגֶשֶׁם אָיִן.

וְעָשִׂיתִי עַל אִישׁ אֲשֶׁר נֶאֱמָן חֲשַׁבְתִּיו. וְנֶהְפַּךְ לְאוֹיֵב בְּעֵת בְּחַנְתִּיו:

יְדִידוּתָךְ חֲשַׁבְתִּיהָ כְמוֹ מָן

בְּצִנְצֶנֶת יְקָר נִצְפַּן וְנִכְמָס.

בְּחַנְתִּיהָ וְהַמִּבְחָן כְּשֶׁמֶשׁ

אֲשֶׁר יָצָא וְחַם עָלָיו וְנָמָס.

וְעָשִׂיתִי לְאִישׁ מַרְבֶּה לִקְנוֹת סְפָרִים. וְהוּא מִן הַכְּסִילִים הַבּוֹעֲרִים:

רְאוּ פֶתִי אֲשֶׁר נוֹשַׁן בְּסִכְלוֹ

וְרָעוֹתָיו חֲדָשִׁים לַבְּקָרִים.

וְהוּא מַרְבֶּה קְנוֹת סִפְרֵי תְעוּדוֹת

וְיִדְמֶה בָם חֲמוֹר נוֹשֵׂא סְפָרִים.

וְעָשִׂיתִי לְאִישׁ הִלַּלְתִּיו בְּשִׁירַי. וְהִכִּיר מְלִיצַת דְּבָרַי:

הַשַּׂר, חֲלִיצוֹת מִמְּלִיצוֹתַי לְבַשׁ

יַעַן צְרִי צִירִי לְכָל צִיר יֶחֱבָשׁ.

רוֹדֶה דְבַשׁ מִלַּי יְהִי מַכִּיר. אֱמֶת

כִּי מִגְּוִיַּת הָאֲרִי רָדָה דְבָשׁ.

וְעָשִׂיתִי בְדַמֶּשֶׂק עַל רַבִּי צְדָקָה הַדַּיָּן. לְסִבָּה וּלְעִנְיָן:

אֱנוֹשׁ נָטַע בְּכַרְמוֹ עֵץ פְּרִי גַן

עֲבָדָהוּ בְּיוֹם וָלֵיל וְהִשְׁקָה.

וְעֵת הֵחֵל נְתֹן פִּרְיוֹ בְּעִתּוֹ

וְאֶל כָּל-חֵךְ אֲכִילָתוֹ מְתוּקָה.

נְתָשָׁהוּ וְתַם לָרִיק יְגִיעוֹ

וְכָכָה מַעֲשֵׂה רַבִּי צְדָקָה.

תְּמוֹל בָּנָה דְבִיר מוּסָר וְהַיּוֹם

הֲרָסָהוּ וְשָׁב שׁוּחָה עֲמֻקָּה.

אֲקַו מִשְׁפָּט וְהֵן מִשְׂפָּח וּמַפָּח

וְלִצְדָקָה וְהִנֵּה קוֹל צְעָקָה.

וְלָא יָדַע הֲכִי צוּר שָׂם בְּתוֹךְ פִּי

עֲלֵי כָל-אוֹיְבֵי חֶרֶב מְרוּקָה.

וְעֵת כִּי אֶשְׁלְפֶנָּה. הָעֲנָקִים

אֲנִי מוֹרִיד בְּזַעַם לַעֲלוּקָה.

אֲנִי אַחְרִישׁ וְעֵת שִׁירִי יְדַבֵּר

זְמַן יַרְעִישׁ בְּיָדוֹ הַחֲזָקָה.

וּמִי נִגַּשׁ לְמִלְחַמְתִּי וְנִמְלַט

וּמִי שָׁלַח בְּהוֹדִי יָד וְנִקָּה?

וְעָשִׂיתִי בְּמִצְרַיִם לַדַּיָּן ר' מְנַחֵם זַ"ל עַל עִנְיַן אִישׁ אֲשֶׁר בָּטַחְתִּי בוֹ. וְנִכְשַׁלְתִּי בְנִיבוֹ:

קָרָאתִי בַּצַּר לִי פֶסֶל

וָאֶבְטַח עַל עֶצֶב נִבְזֶה.

הָיָה כֶסֶף סִיגִים וּצְרַפְ-

תִּיהוּ אוּלַי סוֹדוֹ אֶחְזֶה.

וָאַשְׁלִיכֵהוּ אֶל הָאֵשׁ –

וַיֵּצֵא הָעֵגֶל הַזֶּה.

וּבְעִנְיַן הַחֵשֶׁק עָשִׂיתִי. וּבֵית אַהֲבָה בָנִיתִי:

צְבִי הַחֵן מְחָצַנִי בְּיָפְיוֹ

וְהִבְעִיר בִּי יְקוֹד חִשְׁקוֹ וְעָצְבּוֹ.

וּבַמִדָּה אֲשֶׁר מָדַד גְּמַלְתִּיו

וְהֶאְכַלְתִּיו פְּרִי חֶטְאוֹ וְחוֹבוֹ.

בְּהַרְאוֹתִי רְאִי מוּצָק לְעֵינָיו

הֲשִׁיבוֹתִי יְקוֹד חִשְׁקִי לְלִבּוֹ.

וְעָשִׂיתִי בְּאַשּׁוּר עַל אִשָּׁה רָעָה וְשַׁלֶּטֶת.

וְהִיא כָאֵשׁ מִתְלַהֶטֶת. וְלָהּ בֵּן פָּרִיץ דּוֹמֶה לְאִמּוֹ.

בְּכַעְסוֹ וּבְזַעְמוֹ:

אֵשֶׁת כְּסִילוּת צוּר בְּשַׁחַת יִצְפְּנָהּ

חִישׁ יִתְּנֶנָּה תוֹךְ שְׁאוֹל וְיִשְׁבְּרָהּ.

תִּדְמֶה עֲתַלְיָה רַק בְּנָהּ יִדְמֶה אֲחַז-

יָהּ. בּוֹרְאָהּ לֹא יַעְזְרָהּ וִיְחַלְּלָהּ.

וְעָשִׂיתִי עַל צִפֹּר נִלְכַּד. בְּפַח צַיָּד:

רְאוּ צִפֹּר אֲשֶׁר עָלָה בְתוֹךְ פַּח

וְלֹא יוֹעִיל בְּעֵת צַר לוֹ כְנָפוֹ.

חֲשַׁבְתִּיהוּ כְּבֶן אָדָם בְּעוֹלָם

וְלֹא יוֹעִיל כְּבוֹדוֹ לוֹ בְסוֹפוֹ.

וְעָשִׂיתִי עַל חֶרֶב מְרוּטָה:

חֶרֶב בְּיַד בַּעְלֵי קְרָב תָעוּף

וּכְמוֹ לְשׁוֹן אֵשׁ תִּהֲלֵךְ תּוֹכָם.

רֵעַי שְׁאֵלוּנִי לְמִי תִדְמֶה

וָאוֹמְרָה תִּדְמֶה לְשׁוֹן חָכָם.

וְעָשִׂיתִי לְלַמֵּד בְּנֵי אָדָם מוּסָר. וּלְהִתְנַהֵג בַּחֲסִידוּת עִם כָּל בָּשָׂר:

אַתָּה תְכַבֵּד מַעֲלָתֶךָ

כִּי מַעֲלָתָךְ לֹא תְכַבֶּדְךָ.

כַּאֲשֶׁר תְּהִי עוֹשֶׂה תְּהִי מוֹצֵא

וּשְׁאַל לְאָבִיךָ וְיַגֵּדְךָ.

אִם תַּעֲשֶׂה טוֹבָה תְּרוֹמִמְךָ

אוֹ תַעֲשֶׂה רָעָה תְּאַבֶּדְךָ.

עַל אִישׁ נָבָל קוֹפֵץ יָדָיו. בְּנִדְבוֹתָיו וּבַחֲסָדָיו:

אִם אִישׁ נָבָל יִצְבֹּר עשֶׁר

אֶל אוֹיְבָיו חִישׁ יוֹרִישֶׁנּוּ.

חַיִל בָּלַע וַיְקִיאֶנּוּ

אֵל מִבִּטְנוֹ יוֹרִישֶׁנּוּ.

וְעָשִׂיתִי עַל אִישׁ גְּבַהּ לֵבָב. הוֹלֵךְ שׁוֹבָב:

בִּרְאוֹתָךְ אִישׁ אֵין ישֶׁר בּוֹ

רַק בָּעשֶׁר נַפְשׁוֹ יָקְרָה.

אַל תָּשֶׁת לֵב אֶל גַּאְוָתוֹ

כִּי מַעְלָתוֹ מַעְלַת גֵּרָה.

וְעָשִׂיתִי עַל אִישׁ כָּל עֵת יָרֹק בְּבֵית הַכְּנֶסֶת:

רָאוּי עַל כָּל-בָּאֵי קֹדֶשׁ

מֵימֵי טֹהַר עָלָיו לִזְרֹק.

בִּגְלַל נִבְזִים בּוֹזִים אוֹתוֹ

כִּי מִפָּנָיו לֹא חָשְׂכוּ רֹק.

וְאֵלֶּה עָשִיתִי לְאִישׁ עָשָׂה פִּתּוּחֵי כֶסֶף וַעֲטָרָה. לְסֵפֶר תּוֹרָה:

תַפּוּחֵי זָהָב יִדְמוּ טַ-

בָּעוֹת עַל בַּדֵּי הָאָרוֹן.

מִן הַקֹּדֶשׁ נִשְׁמַע קוֹלָם

כַּפַּעְמוֹנִים בִּמְעִיל אַהְרֹן.

כָּזֹאת נִדְבַת הַשַּׂר יוֹסֵף

כָּל הַמַּשְׂכִּיל אוֹתָהּ יָרֹן.

יוֹדוּ חַסְדּוֹ אֶל כָּל-יָבוֹא

לִבְנֵי יִשְׂרָאֵל זִכָּרוֹן.

וְעָשִׂיתִי לְהִתְעַלֵּס בַּגַּנִּים. וְלִרְווֹת בַּת הַגְּפָנִים:

דּוֹדִי שְׁתֵה עִמִּי בְּצֵל גִּנָּה

אָשִׂים הֲדַס נֶגְדָּךְ וְשׁוֹשַׁנָּה.

אֶשְׂמַח בְּבַת כֶּרֶם אֲשֶׁר גִּילָהּ

חָדָשׁ. וְהִיא בָלָה וְנוֹשָׁנָה.

הֵן יוֹם בְּעֵין הָאֵל כְּמוֹ אֶלֶף

שָׁנִים. וְהַשָּׁנָה כְּמוֹ עוֹנָה.

מִי יִתְּנֵנִי אֶהְיֶה בַכּוֹס

עַד כִּי בְעֵין הָאֵל יְהִי שָנָה.

וְעוֹד עָשִׂיתִי בְעִנְיַן הַתְּשׁוּקָה. וְהַחֶבְרָה הַחֲשׁוּקָה:

יַעְלָה הֱבִיאַתְנִי לְבֵית אַהְבָה

בָּרָה וְתַמָּה כַּאֲבִיגַיִל.

וּבְסוּר צְעִיפָהּ גִּלְּתָה תֹאַר

הֶחְפִּיר בְּיָפְיוֹ בַת אֲבִיחַיִל.

זָרַח בְּלֵיל אוֹרָהּ וְחָרְדוּ כֹּל

וַיִּרְקְדוּ הָרִים כְּמוֹ אַיִל.

שַׂחְתִּי אֲהָהּ נִגְלוּ צְפוּנֶיהָ

וַתַּחֲשׂף יָדָהּ כְּבַת חַיִל.

וַתַּהֲפֹךְ עָלַי עֲנַן שֵׂעָר

שָׁחוֹר – וְהַיּוֹם הָפְכָה לַיִל.

עַל סֵפֶר הַמַּדָּע:

סֵפֶר כְּעֵץ שָׁתוּל בְּמֵי מַדָּע וּבִין

כִּי בוֹ חֲפָצָיו יִמְצְאָה כָל-דּוֹרֵשׁ.

סִפְרֵי גְאוֹנֵינוּ וְאִם רַבּוּ מְאֹד

הֵמָּה עֲנָפִים לוֹ וְהוּא הַשֹּׁרֶשׁ.

וְעָשִׂיתִי עַל אִישׁ בִּירוּשָׁלַיִם יָצְאוּ עָלָיו דִּבּוֹת רָעוֹת:

הָפְקַד בְּהַר צִיּוֹן אֱנוֹשׁ טָמֵא לְרֹאשׁ

קֹדֶשׁ וְלוֹ נֶגַע אֲשֶׁר פָּשָׂה בְעוֹר.

יִפְעַר לְכָל-עוֹבֵר וְשָׁב עַצְמוֹ. לְזֹאת

כָּל-יוֹדְעָיו קָרְאוּ שְׁמוֹ רֹאשׁ הַפְּעוֹר.

וְעָשִׂיתִי עַל אִישׁ עוֹזֵב אִשְׁתּוֹ. בְּעַד שִׁפְחָתוֹ. וְיָלְדָה לוֹ בַת דּוֹמָה לְצוּרָתוֹ. וְהָיָה שְׁמוֹ יַעֲקֹב:

מִדּוֹת אֱנוֹשׁ חָנֵף בְּלִיַּעַל

הִתְקַבְּצוּ שִׁמְעוּ בְנֵי יַעֲקֹב.

חָשַׁק בְּשִׁפְחָתוֹ מְקוֹם אִשְׁתּוֹ

וַיַּהֲפֹךְ מִישׁוֹר אֱלֵי עָקֹב.

עַד יָלְדָה לוֹ בַת אֲשֶׁר דָּמְתָה

אֵלָיו בְּמֵצַח עַז וְלֵב עָקֹב.

נִשְׁבַּע שְׁמוֹ הָרַע לְבַל יַעְזֹב

אוֹתָהּ וְלֹא חָפֵץ בְּבַת יַעֲקֹב.

וְאֵלֶּה עָשִׁיתִי לְאִישׁ כְּסִיל. מִן הַכְּסִילוּת אַלְפַּיִם בַּת יָכִיל:

דּוֹדִי נִקַּדְתִּי לָךְ שִׁירִי

יַעַן שִׂכְלָךְ נֶחְשַׁל חָשׁוֹל.

וּנְתִיבוֹתָיו לָךְ פִּנִּיתִי

אוּלַי לִבָּךְ עָלָיו יִמְשֹׁל.

כִּי זָכַרְתִּי מִצְוַת אֵל לִפְ-

נֵי עִוֵּר לֹא תִתֵּן מִכְשׁוֹל.

וְאֵלֶּה עָשִׂיתִי לְאִישׁ חֲסַר דַּעַת. וְהוּא כָבֵד בֶּעֶל נֶפֶשׁ מִשְׁתַּגַּעַת:

חֲסַר דַּעַת וְקַל רוּחַ בְּדַעִתּוֹ

וְהוּא כָבֵד כְּמוֹ תָבוֹר בֶּהָרִים.

וְלוּ נָפְלָה כְבֵדָתוֹ בְיַם-סוּף

אֲזַי שָׁקַע וְשָׁב אֶרֶץ חֲרֵרִים.

וְלוּ הָיָה בְרַגְלָיו קַל כְּרֹאשׁוֹ

יְהִי מַשִּׂיג בְּרוּצוֹ הָעֳפָרִים.

לְאִישׁ מִבְּנֵי עֲדִינָה עָשָׂה שִׁיר מָלֵא זִמָּה וְטֻמְאָה:

לוּ שָׁר בְּנוֹ עַמְרָם פְּנֵי דוֹדִי

מִתְאַדְּמִים הָעֵת שְׁתוֹת שֵׁכָר

וִיפִי קְוֻצּוֹתָיו וְהוֹד יָפְיוֹ

לֹא חַק בְּתוֹרָתוֹ וְאֶת זָכָר.

וּכְשָׁמְעִי שִׁירוֹ קִבַּצְתִּי עֲשָׂרָה אֲנָשִׁים

בַּעֲלֵי שֵׂכֶל וּתְבוּנָה. וְישֶׁר וֶאֱמוּנָה. וְהֶחֱרַמְנוּהוּ

בְּשִׁירִים כְּשִׁירָיו. עַל קַו חֲרוּזָיו וּדְבָרָיו.

אָמַרְתִּי אֲנִי בַתְּחִלָּה:

שַׁדַּי שְׁלֹף חֶרֶב נְקָמָה עַל

טָמֵא אֲשֶׁר תּוֹרָתְךָ עָכָר.

הַפֶּה אֲשֶׁר נִפְתַּח לְחָרֵף אֵל

חִישׁ בַּחֲרָבוֹת יִהְיֶה נִסְכָּר.

מוֹכֵר קְדֻשַּׁת אֵל בְּעַד טֻמְאָה

מַהֵר לְיַד הוֹרֵג יְהִי נִמְכָּר.

וַיַּעַן הַשֵּׁנִי:

אָנָּא אֲדוֹן עוֹלָם שְׁפוֹךְ זַעְמָךְ

עַל אִישׁ בְּיֵין זִמָּה וְחֵטְא שָׁכָר.

הַמְטֵר מְטַר גָּפְרִית וְאֵשׁ עָלָיו

עַד כִּי בְיַד מָוֶת יְהִי נֶעְכָּר.

רוּחֹו לְאֵשׁ תֹּפֶת שְׁלַח מִנְחָה

וּלְפִי שְׁאוֹל גּוּפוֹ שְׁלַח אֶשְׁכָּר.

וַיַּעַן הַשְּׁלִישִׁי:

שַׁדַּי מְכֹר מַהֵר לְיַד מָוֶת

הָאִישׁ אֲשֶׁר דָּתָךְ בְּחֵטְא מָכָר.

תּוֹרָתְךָ תִּצְעַק בְּפִיו תָּמִיד

אוֹתִי פְצֵה מִיַּד בְּנֵי נֵכָר.

יוֹם יַעֲלֶה זִכְרוֹן בְּנֵי תֵבֵל

לְטוֹב, יְהִי לָרַע שְׁמוֹ נִזְכָּר.

וַיַּעַן הָרְבִיעִי:

הוֹי עַל אֲנָשִׁים עוֹזְבִים תּוֹרַת

הָאֵל וְהֵם עוֹבְדִים לְאֵל נֵכָר.

כִּי כָל-גְּדוֹלֵיהֶם מְטֻנָּפִים

בִּמְקוֹם נְקֵבָה בּוֹעֲלִים זָכָר.

בַּאֲשֶׁר בְּיֵין חֵטְא יִשְׁכְּרוּ, חֶרֶב

אוֹיְבִים בְּדָמָם יוֹם וְלֵיל תִּשְׁכָּר.

וַיַּעַן הַחֲמִישִׁי:

עוּרָה בְנוֹ עַמְרָם רְאֵה דָתָךְ

כִּי חִלְּלוּהָ סוֹבְאֵי שֵׁכָר.

יִתְעַלְּלוּ תָמִיד בְּתַעְלוּלִים

וְיִתְּנוּ נָדָן לְכָל-זָכָר.

צוּר יַעֲקֹר שָׁרְשָׁם וּבִבְשׂוֹרַת

מוֹתָם לְכָל-אֶרֶץ יְשֻׁלַּח כָּר.

וַיַּעַן הַשִּׁשִּׁי:

הָאִישׁ אֲשֶׁר תּוֹרַת בּנֹו עַמְרָם

חִלֵּל בְּנַבְלוּת פִּיו אֲשֶׁר זָכַר.

פּוֹעֵר לְכָל-עוֹבֵר וָשָׁב עַצְמוֹ

נִרְבָּע עֲלֵי אַרְבַּע לְכָל זָכָר.

הוֹרֵג יְבַקֵּעַ אֶת בְּשָׂרוֹ כַּאְ-

דָמָה תְבֻקַּע בַּחֲרשׁ אִכָּר.

וַיַּעַן הַשְּׁבִיעִי:

הָאִישׁ אֲשֶׁר דָּת יוֹצְרוֹ חִלֵּל

וַיִּשְׁכְּחָהּ יוֹם דִּין וְלֹא זָכָר.

וַיַּעֲנֶה עַזּוֹת לְמוּל צוּרוֹ

וַיִּפְעֲרָה פִיהוּ וְלֹא סָכָר.

יַחְשִׁיךְ אֱלֹהִים מִשְּׁחוֹר פָּנָיו

עַד כִּי לְרוֹאָיו לֹא יְהִי נִכָּר.

וַיַּעַן הַשְּׁמִינִי:

הָאִישׁ אֲשֶׁר הֵעֵז לְאֵל פָּנָיו

עַד כִּי בְנוֹ עַמְרָם לְרַע זָכָר.

חָשַׁב עֲכֹר תּוֹרַת אֱלֹהֵינוּ

בַּשָּׁוְא וְנַפְשׁוֹ בֶּאֱמֶת עָכָר.

יַעְקֹר אֱלֹהִים כָּל-בְּנִי בֵיתוֹ

עַד עִם נְקֵבָה יַאֲבִיד זָכָר.

וַיַּעַן הַתְּשִׁיעִי:

הַבֵּט בְּנוֹ עַמְרָם אֲשֶׁר חִלֵּל

דָּתָךְ אֲבִי טֻמְאָה וּבֶן נֵכָר.

יִתְגַּל בְּמַחְבוֹאָיו וְאוּלָם תּוֹךְ

קָהָל יְהִי נִסְתָּר וְיִתְנַכָּר.

מַהֵר אֱלֹהִים יַאֲבִידוֹ וַחְ-

רוֹן אַל יְהִי אֶל-פָּעֳלוֹ שָׂכָר.

וַיַּעַן הָעֲשִׂירִי:

הוֹי עַל קְהִלָּה שַׂר קְהִלָתָם

הֵזִיד לְגַדֵּף אֵל וְהִתְמַכָּר.

תָּעָה בְיֵין זִמָּה וְלֹא יַיִן

רָוָה בְּסִבְאוֹת חֵטְא וְלֹא שֵׁכָר.

אֵשׁ מִשְּׁמֵי מָרוֹם בְּאַף תִּשְׂרֹף

גּוּפוֹ כְמוֹ חֵלֶב וְכִיקַר כָּר.

וְאֵלֶּה עָשִׂיתִי בְּעִנְיַן הַתְּשׁוּבָה מִן הַמְּשׁוּבָה:

אַזִּיל עֲנַן דִּמְעָה עֲלֵי פִשְׁעִי עֲדֵי

נֶגֶד דְּמָעַי נִשְׁכְּחוּ מֵי נֹחַ.

מַה-לָךְ לְבָבִי תֶחֱשַׁק תֵּבֵל אֲשֶׁר

בָּהּ כָּל-חֲכַם לֵב זָנְחָה זָנֹחַ?

רוּץ לַעֲבֹדַת אֵל וְשׁוּבָה אֶל אֱלֹ-

הֵינוּ הֲכִי יַרְבֶּה מְאֹד לִסְלֹחַ.

שׁוּבָה הֲלֹא תִשְׁלֹף לְךָ חַרְבָּה וְגַרְ-

זֶן הַפְּרִידָה תַּחֲשֹׁב לִנְדֹּחַ.

קַוֵּה בְכַל-עֵת אֶל אֱלֹהֶיךָ הֲכִי

קֹוֵי אֲדֹנָי יַחֲלִיפוּ כֹחַ.

וְעָשִׂיתִי לְהוֹכִיחַ הַסְּכָלִים. וּלְהַזְהִיר הַנּוֹאָלִים:

יָשֵׁן לְמָתַי עֵינְךָ עָצַמְתָּ

אוֹ כִי בְחֵיק הַתַּאֲוָה נִרְדַּמְתָּ?

גּוֹיִם אֲשֶׁר לֹא יָדְעוּ בוֹרֵא דְרָ-

שׁוּהוּ – וְאַתְּ שׂוֹנֵא וְלֹא נְכִלַמְתָּ.

הֵאִיר מְאוֹר צִדְקוֹ וְאַתָּה מֵרְאוֹת

אוֹר הָאֱמֶת מַסְוֶה בְעֵינֶיךָ שַׂמְתָּ!

כָּמָּה קְרָאוּךָ בְלִי פֶה קוֹרוֹת

יָמִים. וְאַתְּ נִרְדָּם וְעוֹד לֹא קַמְתָּ.

רָצוּ חֲבֵרֶיךָ חֲצוֹת לֵיל. וְהִשְׁ-

כִּימוּ לְשַׁחֵר אֵל. וְלֹא הִשְׁכַּמְתָּ.

תִּישַׁן בְּחֵיק תֵּבֵל וְטוֹבוֹת תַּחֲלֹם

מָה הַחֲלוֹם הַזֶּה אֲשֶׁר חָלַמְתָּ.

וְאֵלֶּה עָשִׂיתִי בְּעִנְיַן הַסִּפּוּק. כִּי הוּא לָאֱמוּנָה חִזּוּק:

אִם בַּעֲבוֹדַת אֵל תְּכַל יָמֶיךָ

תִּהְיֶה פְּאֵר עַל רֹאשׁ בְּנֵי עַמֶּךָ.

שִׂים חֶלְקְךָ פַּת עִם כְּלִי מַיִם וּבַ-

רֵךְ לַחְמְךָ יוֹצְרָךְ וְגַם מֵימֶיךָ.

הָקֵם בְּלֵב מִזְבַּחֲךָ אַךְ חֶלְבְּךָ

הַקְרֵב וְנַסֵּךְ בַּבְּכִי דָמֶיךָ.

וּזְכֹר הֲכִי תֵבֵל כְּמֹץ עוֹבֵר וְכִי

כַּצֵּל וְכַצֶּלֶם דְּמוּת צַלְמֶךָ.

וַעְרֹךְ לְרָעֵב פַּת חֲסָדֶיךָ וְאָז

תִּקְרָא עֲנִיֵּי עַם לְמַטְעַמֶּיךָ.

וּבְנֵה לְךָ בַיִת אֲשֶׁר תִּמְצָא בְיוֹם

מִבֵּיתְךָ תִּסַּע לְעוֹלָמֶךָ.

אִם תַּעֲשֶׂה כֵן אוֹיְבֶךָ תַּחְלשׁ

וּבְרֹב גְּאוֹנְךָ תַּהֲרֹס קָמֶיךָ.

וְעָשִׂיתִי תוֹדָה עַל חַסְדֵי הַבּוֹרֵא. וּמִי כָמוֹהוּ מוֹרֶה:

אֲדוֹן עוֹלָם גְּמַלְתַּנִי חֲסָדִים

וְעַל חַסְדָּךְ גְּמַלְתִּיךָ מְרָדִים.

וְרַעְיוֹנַי לְרַחִמֶיךָ בְכָל-עֵת

מְלֻמָּדִים וְדָתָךְ לֹא לְמֵדִים.

וְלֹא רָחְקוּ חֲסָדֶיךָ בְעוֹדִי

בְּעֵת שִׁבְתִּי וְעֵת אַרְחִיק נְדוֹדִים.

בְּיוֹם בּוֹ אָרְבוּ צָרוֹת לְנַפְשִׁי

וְחִכִּיתִים כְּחַכֵּי אִישׁ גְּדוּדִים.

גְּאַלְתַּנִי וְהִרְחַקְתָּ מְגוּרֵי

וְשִׂמַּחְתָּ שְׂעִפַּי הַחֲרֵדִים.

וּבִהְיוֹתִי בְתוֹךְ צִיָּה וּמִדְבָּר

וְהָאָרֶץ כְּמוֹ נַחְלֵי יְקוֹדִים.

וְדָבַק בָּהּ לְשׁוֹן כָּל-אִישׁ לְחִכּוֹ

וְאֵין מָנוֹס וְאֵין מַעְמָד לְעוֹמְדִים.

וְעֵת רָגְזוּ קְרָבַי בִּי וְהָמוּ

וּמֵעַי בַּכְּאֵב עוֹלִים וְיוֹרְדִים.

עֲזַרְתַּנִי וְהָיוּ לָךְ שְׁבָחַי

כְּמוּסָפִים וְרַחְמֶיךָ תְמִידִים.

וְעַל כָּל-זֶה שְׂעִפַּי כּוֹרְעִים לָךְ

וְהֵם מִשְׁתַּחֲוִים נֶגְדָּךְ וּמוֹדִים.

זְכֹר צִדְקָךְ אֲדוֹנֵי הָאֲדוֹנִים

וְאַל תִּזְכֹּר עֲוֹן עֶבֶד עֲבָדִים.

וְדִבְרֵי תַחֲנוּנִים חַבְּרֵם עִם

חֲסָדֶיךָ וְהָיוּ לַאֲחָדִים.

בְּמֵי דִמְעִי עֲצֵי תִקְוָה אֲרַוֶּה

עֲדֵי עֵץ חַסְדְּךָ יִתֵּן מְגָדִים.

וְעָשִׂיתִי עַל בַּת הַגְּפָנִים. מְגָרֶשֶׁת הַיְגוֹנִים:

אֱמֶת בַּת הַגְּפָנִים הִיא צְרִי צִיר

וְכָל-לָשׁוֹן תְּהִלָּתָהּ תְּסַפֵּר.

בְּדָמָהּ יִמְצְאָה דָמִי מְנוּחָה

וְהַדָּם הוּא וּבַנֶּפֶשׁ יְכַפֵּר.

וְאֵלֶּה עָשִׂיתִי לַשַּׂר רַ' שְׁמוּאֵל בֶּן הָרֹאשׁ

רַ' נִסִּים בִּסְפָרַד, חִבַּרְתִּיו עַל שְׁמוֹ. וְעַל שֵׁם-בְּנוֹ יְהִי אֱלֹהָיו עִמּוֹ:

לְבָבִי בָחַן כָּל-בְּנֵי שֵׁם וּגְדֻלָּה

וּבָחַר לוֹ הַשַּׂר שְׁמוּאֵל לִסְגֻלָּה.

וְשָׂם אוֹתוֹ וּבְנוֹ לְנֵס גַּם לִתְהִלָּה

וּבַגָּדוֹל הֵחֵל וּבַקּטֹן כִּלָּה.

וְעוֹד עָשִׂיתִי לִבְנוֹ. יוֹצְרוֹ יִצְּרֵהוּ וְיִשְׁמְרֵהוּ וִיחַיֵּהוּ:

צְבִי חֵן כְּלִיל יֹפִי פְּלָאוֹת בְּתָאֳרוֹ:

לְחָיָיו כְּמוֹ שֶׁלֶג וְזִיקוֹת יְאַזְּרוּ.

בְּמִצְחוֹ מְאוֹר שַׁחַר וְלַיִל יְקֻבְּצוּ

וְעַל לֶחֱיוֹ לַהַב וּבָרָד יְחֻבְּרוּ.

וְשִׁנָּיו שְׁנֵי טוּרֵי בְדֹלַח כְּחוּט עֲנָק

וּבָם טַלְלֵי אוֹרוֹת רְפָאִים יְעוֹרְרוּ.

שְׁמוֹ יִקְרְאוּ מַזְהִיר וְלֶחְיוֹ כְמוֹ שְׁמוֹ

הֲכִי יֵלְכוּ עִוְרִים לְאוֹרוֹ וְיִנְהֲרוּ.

יְהִי רֹאשׁ לְבֵית אָבוֹת כְּיוֹלְדָיו וְיִחְיֶה

עֲדֵי יֶחֱזוּ הוֹרָיו אֲשֶׁר בּוֹ יְשַׂבְּרוּ.

וְעָשִׂיתִי בְּעִנְיַן הָאַהֲבָה. הַחֵשֶׁק וְהַחִבָּה:

יְקַצֵּץ אֵל צְעִיפִים מַעֲלִימִים

פְּנֵי עָפְרָה וְיִקְרַע הָרְדִידִים.

הֲלֹא דַי לִי בְצִיר חֵשֶׁק וּפֵרוּד

עֲדֵי כִּי יִבְגְדוּ בִי הַבְּגָדִים.

עַל הָרַמְבַּ“ם זַ”ל:

יוֹם עָלוּ כָּל חַכְמֵי תֵבֵל

אֶל מַעְלוֹת רָמִים וּגְבֹהִים.

עָלוּ לִמְרוֹם הַשֵּׂכֶל אַךְ

משֶׁה עָלָה אֶל הָאְלֹהִים.

וְעָשִׂיתִי בְעִנְיַן הַצְּדָקוֹת. אֲשֶׁר יַצִּילוּ מִן הַמְּצוּקוֹת:

זְרַע בֶּן אִישׁ לְךָ זֶרַע צְדָקָה

וּמָחָר תִּמְצְאָה מֵאָה שְׁעָרִים.

וְאִם יִסְגֹּר זְמַן שַׁעְרֵי רְצוֹנָךְ

בְּיַד הָאֵל לְךָ כַמָּה שְׁעָרִים.

וְעָשִׂיתִי בְּעִנְיַן הָעוֹלָם:

אֲשֶׁר יִבְטַח בְּתַהְפֻּכוֹת זְמַנּוֹ

יְהִי נִכְשָׁל וְיִתְחַלְחַל בְּצִירוֹ.

הֲכִי חַיֵּי אֱנוֹשׁ כֶּרֶם וְרַק הַ-

זְּמַן בּוֹצֵר וְהַמָּוֶת בְּצִירוֹ.

וּבְעִנְיַן הַמָּוֶת. וְשׁוֹכֵן אֶרֶץ וְצַלְמָוֶת:

בֶּן-אִישׁ הֲתִתְעַדֵּן בְּתַעֲנוּג הַזְּמַן

וּבְשָׂרְךָ דָשֵׁן מְאֹד וְגַם שָׁמֵן.

הוֹנָךְ כְּלִי שֶׁמֶן וְיִשָּׁבֵר כְּמוֹ

תַךְ הַכְּלִי. וְיַעֲמֹד הַשָּׁמֶן.

וְעָשִיתִי לְאִישׁ מִן הַחֲשׁוּבִים. הַיְקָרִים וְהַנְּדִיבִים. וּשְׁמוֹ יִרֹחָם. מִן הַשָּׁמַיִם יְנֻחָם:

אֵיךְ לֵב חֲסַר לֵב יְרֻחָם

אֵיך מִיְּגוֹנָיו יְנֻחָם.

בָּטַח בְּתִקְוָה מְאוֹרָה

קָדַר וּפָעַל בְּפֶחָם.

אַךְ יֵשׁ בְּתֵבֵל נְדִיב לֵב

הִשְׁפִּיל נְדִיבִים וְהִשְׁחָם.

תַּאְוַת נְדִיבִים בְּתִתָּם

מַהְלַל מְבַקְשִׁים בְּלָקְחָם.

אוּלָם לְכָל שׁוֹאֲלָיו הוּא

מוֹדֶה וְעוֹזֵב יְרֻחָם.

וְעָשִׂיתִי אֵלֶּה בְּבֵית הַכְּנֶסֶת אֲדוֹנֵנוּ עֶזְרָא. וְהִנָּם חֲקוּקִים עַל קִירוֹת הַבִּירָה:

יְצַו הָאֵל לְעֶזְרָא רֹב שְׁלוֹמוֹת

צְבִי צִבְאוֹת מְרוֹמוֹת וַהֲדוֹמוֹת.

לְזִיו אוֹרוֹ פְּאֵר עַל הַמְּרוֹמִים

וְלִמְקוֹמוֹ פְאֵר כָּל-הַמְּקוֹמוֹת.

מְקוֹמוֹ הוּא מְקוֹם מַחְנֵה אֱלֹהִים

וְאֵלָיו נִפְתְּחוּ שַׁעְרֵי מְרוֹמוֹת.

תְּהִי נַפְשׁוֹ בְכִסֵּא אֵל צְרוּרָה

וְיַחְלִיץ טַל יְשׁוּעָה הָעֲצָמוֹת

וְהָאֵל מִשְּנָתוֹ חִישׁ יְעִירוֹ

לְקֵץ יָמִים וְיִחְיֶה אַחֲרֵי מוֹת.

זְכוּתוֹ יַעֲמֹד עָלָיו לְמֵלִיץ

לְהַשְׁקִיט אֶת שְׁאוֹן יָם הַמְּהוּמֹות.

וְיָפִיק אֶל יְהוּדָה בַּר שְׁלֹמֹה

בְּיָדוֹ עֹז וּבִימִינוֹ נְעִימוֹת.

וְעָשִׂיתִי לְאִישׁ מִן הָאֲצִילִים. שׁוֹכֵן בֵּין הַנְּבָלִים:

הַשַּׂר אֲשֶׁר בֵּין כּוֹכְבֵי רוּם יַחֲנֶה

אַךְ שׁוֹאֲלוֹ טֶרֶם קְרָאוֹ יַעֲנֶה.

שַׁח לֵב וְאִם נִלְוֶה לְעַם מִשְׂתָּרְרִים

נָדִיב וְאִם בֵּין קוֹפְצֵי יָד יַחֲנֶה.

אֶתְמַהּ לְשִׁכְנוֹ בָם וְלֹא אֶתְמַהּ הֲכִי

כֵּן הַשְּׁכִינָה שָׁכְנָה עַל הַסְּנֶה.

וְעוֹד עָשִׂיתִי לוֹ. לְשַׁבֵּחַ מַעֲלָתוֹ וְגָדְלוֹ:

לְבָבִי לְיִצְחָק בְּחִיר אֵל מְכַרְתִּיו

בְּכֶסֶף יְדִידוּת וּבִמְחִיר אֲהָבִים.

דְּבִיר הַיְקָר הוּא וְלִבּוֹ אֲרוֹן דָּת

וְיָדָיו בְּתִתָּם שְׁנֵי הַכְּרוּבִים.

וְלוּחוֹת לְבָבוֹ כְּלוּחוּת בְּרִית הֵם

וְאַךְ הֵם בְּאֶצְבַּע אֱלֹהִים כְּתוּבִים.

וְהוּא רָם בְּעֵין כֹּל וְשָׁפָל בְּעֵינָיו

וְיָחִיד וְרַבִּים הֲלָכָה כְּרַבִּים.

וְעָשִׂיתִי בְעִנְיַן הַסִּפּוּק:

הֲלֹא דַי לִי בְטַרְפִּי מִזְּמַנִי

וְאַל אָמִיר בְּקָלוֹן אֶת גְּאוֹנִי.

לְבִלְתִּי יֹאמְרוּ עָלַי בְּנֵי אִישׁ

רְאוּ חֶסֶד פְּלוֹנִי עַל פְּלוֹנִי.

בְּעֵת לֹא אֶהֱיֶה נִצְרָךְ לְהוֹנוֹ

חֲשַׁבְתִּיו כַּאֲשֶׁר הוּא יַחְשְׁבֵנִי.

אֲהַבְתִּיו אִם יְבַקֵּשׁ אַהֲבָתִי

וְאָפֵר הַבְּרִית אִם יַעְקְבֵנִי.

וְעָשִׂיתִי עַל עִנְיַן הַחֶבְרָה:

יְמֵי חֶבְרָה כְצוּף מָתְקוּ לְחִכִּי

וְלֵילוֹת הַנְּדוֹד רָעִים וּמָרִים.

שְׂמָחוֹת בָּרְחוּ מִנִּי וְדָמוּ

נְשָׁמוֹת מָאֲסוּ לִשְׁכֹּן פְּגָרִים.

יְגוֹנוֹתַי בְּלֵיל פֵּרוּד אֲרֻכִּים

וְשִמְחוֹתַי בְּלֵיל חֶבְרָה קְצָרִים.

עוֹד עַל הָעִנְיָן:

עֹפֶר בְּפִיהוּ שְׁנֵי טוּרֵי עֲנַק לֶשֶׁם

יִצְחַק לְבִכְיִּי כְּמוֹ קֶשֶׁת בְּיוֹם גֶּשֶׁם.

גוּפוֹ כְשֶׁלֶג וְלוֹ מִן הַכְּפוֹר גֶּשֶׁם

בּוֹ יִקְטְלֵנִי וְלֹא יָדַע וְהוּא אָשֵׁם.

עוֹד עַל הָעִנְיָן:

עֲדֵי מוֹתִי תְעַנֵּנִי צְבִי חֵן

וְנַפְשִׁי לָךְ כְּמוֹ אָמָה מְכוּרָה.

וּמִי יִתֵּן וְאֶמְצָא סוֹף לְקִצְפָּךְ

וּמַגֵּפַת נְדוֹדָךְ נֶעֱצֶרֶה.

וְאֵשׁ חִשְׁקָךְ יְדִיד תִּשְלֵח בְּקִרְבִּי

וְהִבְעַרְתָּ לְבָבִי תּוֹךְ מְדוּרָה.

וְרָאוּי לָךְ לְשַׁלֵּם אֶת לְבָבִי

וְכֵן הַדִּין לְמַבְעִיר הַבְּעֵרָה.

וְאֵלֶּה עָשִׂיתִי לְאֶחָד מִן הַנְּשִׂיאִים הָאֲצִילִים. וְהוּא רֹאשׁ לַנְּבָלִים:

וְשִׁבַּחְתִּי לְאִישׁ נָבָל בְּאַשׁוּר

בְּשִׁיר כָּזָב אֲשֶׁר אֵין בּוֹ נְכוֹנָה.

וְנָדַר לִי נְדָרִים הֵם כְּשִׁירִים

מְלֵאִים שָׁוְא וְאֵין בָּהֶם אֱמוּנָה.

וְעוֹד לֹא רָאֲתָה עֵינִי לְשׁוֹן שָׁוְא

כְּמוֹתוֹ מִסְּפָרַד עַד עֲדִינָה.

וְעֵת רֵעָיו שְׁאֵלוּהוּ וְאָמְרוּ

הֲתִקְפֹּץ יָד מְלֻמֶּדֶת נְתִינָה.

הֱשִׁיבָמוֹ אֲנִי הַיּוֹם קְצַר יָד

וְנַפְשִׁי הִיא בְּחֹסֶר כֹּל בְּחוּנָה.

וְלִי צֹרֶךְ לְהוֹצָאָה בְּכָל-יוֹם

אֲשֶׁר בִּשְׁטָר וְחוֹב עָלַי נְתוּנָה.

וְלִי נַעַר יְשָׁרְתֵנִי כְּחֶפְצִי

וְעַרְשֵׂנוּ בְכָל עֵת רַעֲנָנָה.

וְעָלַי חוֹב לְהַשְׁלִים לוֹ צְרָכָיו

וְלָתֵת לוֹ שְׁאֵר וּכְסוּת וְעוֹנָה.

אַחַר הַדְּבָרִים הָאֵלֶּה יָצָאנוּ מִמִּצְרַיִם. לִרְאוֹת בְּטוּב יְרוּשָׁלַיִם. וַנַּעֲבֹר דֶּרֶך הַמִּדְבָּרָה. אֶרֶץ צִיָּה וְצָרָה. וּבְחַרְבוֹנֵי קַיִץ עַצְמִי חָרָה. וְקָרָאתִי שֵׁם הַמָּקוֹם הַהוּא תֵּבְעֵרָה. כִּי שָׁם בְּנֵי-יָמִים מַקְרִיבִים אֵשׁ זָרָה. וַתְּהִי נַפְשָׁם בְרִשְׁפֵיהֶם שְׂרוּפָה. וְתַנִּים וּבְנוֹת יַעֲנָה יִבְכּוּ אֶת הַשְּׂרֵפָה. וַנֵּלֶךְ בֵּמִּדְבָּר הַגָּדוֹל וְהַנּוֹרָא. הוּא מְקוֹם דַּאֲבוֹן נֶפֶשׁ וְכִלְיוֹן עֵינַיִם. וְנָחָשׁ וְשָׂרָף וְעַקְרָב וְצִמָּאוֹן אֲשֶׁר אֵין מַיִם. אֲבָל שַׂמְנוּהוּ בְזֵעַת אַפֵּינוּ. וְדִמְעַת עַפְעֵּפינוּ. מְקוֹם נְהָרִים גְּדוֹלִים רַחֲבֵי יָדַיִם.

וַיְהִי בְּבוֹאֵנוּ עַד עַזָּה. אַחַר יְגִיעָה קָשָׁה וְעַזָּה. עָרַכְנוּ תְהִלָּה לָאֵל בְּתֹף וּמָחוֹל. עַל כִּי עָבַרְנוּ מַעַבר הַחוֹל. וְיָצָאנוּ מִן הַקֹּדֶשׁ אֶל הַחֹל. וְלֹא שָׁקַטְנוּ לָרוּץ כְּנַחַל שׁוֹטֵף. אוֹ בָרָק חוֹטֵף. וַיַּסְתִּירֵנוּ בְּכַף הָאֲפֵלָה. וַיּוֹרֵנוּ הַנְּתִיבוֹת בְּאִישׁוֹן לַיְלָה. עַד נִפְתְּחוּ לָנוּ שַׁעֲרֵי שָׁמַיִם. הִיא עִיר ה' יְרוּשָׁלַיִם. וַנִּקֹד וַנִּשְתַּחֲוֶה לְאֵל עֶלְיוֹן אֲשֶׁר פְּקָדָנוּ בִּפְקֻדַּת רַחִמים. וַנַּחֲנֶה שָׁם חֹדֶשׁ יָמִים. וּבְכָל אֵלֶּה הַיָּמִים. חֲבָלִים נָפְלוּ לִי בַּנְּעִימִים. יָמִים כְּאִלּוּ פְנֵיהֶם מִפְּנִינִים נֶחְצְבוּ. וּמֵעֲצֵי חַיִּים נִקְצְבוּ. וּמִכּוֹכְבֵי שַׁחַק גֻּנְּבוּ. וְהָיִינוּ בְּכָל יוֹם מִתְהַלְכִים עַל קְבָרֶיהָ וְצִיּוּנֶיהָ. וְלִבְכּוֹת עַל צִיּוֹן הַמִּתְאַבֶּלֶת עַל בָּנֶיהָ. עַד כָּרוּ יִבְלֵי הַדְּמָעוֹת שׁוּחוֹת בְּפָנֶיהָ. וּלְקוֹנֵן עַל הֲרִיסוֹת אַרְמְנוֹנֶיהָ. וּשְׁאֵרִית בִּנְיָנֶיהָ. וּלְסוֹבֵב עַל קִבְרֵי הַצַּדִּיקִים אֲשֶׁר סְבִיבוֹתֶיהָ. וְלִסְפֹּד עַל הָאֵם וְעַל בְּנוֹתֶיהָ. וּבִקְצָת הַיָּמִים מְשָׁכַתְנוּ יַד הַתְּשׁוּקָה בַעֲבוֹתִים. לַעֲלוֹת עַל הַר הַזֵּיתִים. וְלַעֲרֹךְ תְּפִלּוֹת. לְעוֹשֵׂה נִפְלָאוֹת. וּלְהִשְׁתַּחֲוֹת לַמֶּלֶךְ ה' צְבָאוֹת. וְהָיִינוּ בְעַיִן דּוֹמַעַת. וְנֶפֶשׁ נִכְנַעַת. מַשְׁקִיפִים עַל מְקוֹם הָעֲזָרָה. אֲשֶׁר נֶחֱלַף לַעֲבוֹדָה זָרָה. וְהָיִינוּ מַסְתִּירִים פָּנֵינוּ מֵהַבִּיט לִנְוֵה הַכָּבוֹד הַגָּדוֹל אֲשֶׁר הָיָה שָׁם אָהֳלֹה בַתְּחִלָּה. אֲשֶׁר מִקֶּדֶם שָׁכְנָה בוֹ שְׁכִינָה. וְעַתָּה שָׁכְנָה עָלָיו עֲנָנָה. וְלֹא נִשְׁאַר מִן הַבִּנְיָנִים כִּי אִם אוֹתוֹת מַכְאִיבִים לְבַב זוֹכְרֵיהֶם. וְיַדְאִיבוּ בִבְכִי עֵינֵי רוֹאֵיהֶם. וּבִמְלֹאת הַיָּמִים הָאֵלֶּה שַׂמְנוּ מְגַמָּתֵנוּ. מִדְבַּרָה דַמָּשֶׂק. וְקָרָאנוּ שֶׁם-הַזְּמַן עֵשֶׂק. כִּי הִתְעַשְּׂקוּ עִמּוֹ. וּכְשֶׁהִגַּעְנוּ עָדֶיהָ. רָאִינוּ עִיר יְפֵה-פִיָּה. בְּנוּיָה לְתַלְפִּיָּה. דְּרָכֶיהָ דַּרְכֵי נֹעַם. וְכֻלָּהּ יָפָה וְסָרַת טָעַם. דְּשֵׁנָה כְּגֶפֶן פּוֹרִיָּה. בְּשִׂפְתֵי שׁוֹשַׁנֶּיהָ. וְעֵינֵי מְעוֹנֶיהָ. שׂוֹחֶקֶת וּבוֹכִיָּה. וְכָל נְאוֹתֶיהָ עֲרוּגוֹת הֲדַסִּים. וְכָל יְעָרֶיהָ גַנּוֹת וּפַרְדֵּסִים. כַּמַּיִם לַיָּם מְכַסִּים. וְכָל דְּרָכֶיהָ נְחָלִים נִטָּיוּ. וּבִלְחִי הֲרָרֶיהָ אֵשׁ וָשֶׁלֶג יִשְׁלָיוּ. וּמִצּוּף שְׂפָתֶיהָ צְמֵאִים יִרְוָיוּ. וּבְכָל מְסִלּוֹת. תְּעָלוֹת. וְעַל הַדְּרָכִים. לַמַּיִם מַהְלְכִים. וְעַל הַנְּתִיבוֹת. מַרְאוֹת הַשּׁוֹאֲבוֹת. וְרֵיחָהּ כְּרֵיחַ לְבָנוֹן. וְסָבִיב לָהּ שְׂנִיר וְחֶרְמוֹן. וּבַקַּיִץ תַּשְׁלַג כְּצַלְמוֹן. וּבְתוֹךְ הָעִיר מִבְּנֵי הַגּוֹלָה. קְהִלָּה גְדוֹלָה. וּבְתוֹכָם הָרוֹפֵא הַגָּדוֹל רַבִּי משֶׁה בַר צְדָקָה יָרֹם הוֹדוֹ. הוּא לְבַדּוֹ. בֵּין בְּנֵי עִירוֹ כִּבְּדַנִי. וּבְדִבְרֵי פִיו עוֹדְדַנִי. וְזֶבֶד טוֹב זְבָדַנִי. אֵין בָּם אֱנוֹשׁ יַעֲטֶה סוּת הַמַּהֲלָלִים כְּמַדּוֹ. יִירְאוּ קְרֹב לַנְּדָבָה וְנִגַּשׁ משֶׁה לְבַדּוֹ. וְכִמְעַט אָבְדָה נַפְשִׁי בְעָנְיָהּ. וְלֹא מָצָאתִי צֳרִי לְחָלְיָהּ. לוּלֵי אֲשֶׁר מָצָאתִי שָׁם אִישׁ מִבְּנֵי עֲדִינָה הוּא עֶזְרָא עָלָה מִבָּבֶל. מְצָאתִיו בְּיָם הַנְּדוֹד לְרַב חוֹבֵל. וְהִשְׁכִּיחַנִי בְּמֶתֶק חֶבְרָתוֹ הַחֲמוּדָה. לַעֲנַת הַפְּרִידָה. וְעֵת קָצָה נַפְשֵׁנוּ בַחַיִּים [נֶאֱנַחְנוּ. וְנִפְרַדְנוּ מִן הַשַּׂר וּבֵיתוֹ זָנַחְנוּ]. וְנוֹסְעִים אֲנַחְנוּ לִנְוֵה הַחֲמוּדוֹת. וּמְקוֹם הַמִּדּוֹת הַנִּכְבָּדוֹת. הִיא אֲרַם צוֹבָה אֲשֶׁר שָׁם גַּלְגַּל הַגְּדֻלָּה. וּנְקֻדַּת הָעֲגֻלָּה. הוּא הַשַּׂר הַגָּדוֹל מוֹפֵת הַזְּמַנִּים. וּרְאָיָה לָאַחֲרוֹנִים. עַל הָרִאשׁוֹנִים. מְנַהֵל הָעֵדֶר. וְרֹאשׁ הַסֵּדֶר. יוֹסֵף הוּא הַשַּׁלִּיט עַל הָאָרֶץ. הוּא הַמַּשְׁבִּיר לְכָל עַם הָאָרֶץ.

וָאֶשָּׂא מְשָׁלִי:

שְּׂאוּ שָׁלוֹם לִגְבִיר פְּזוּרֵי בִין אוֹסֵף

תְּהִלָּתוֹ מִפָּז יְקָרָה וָכֶסֶף.

זְמַן לִסְפֹּר מִקְצָת חֲמוּדוֹתָיו כּוֹסֵף

וּבִקֶּשׁ לוֹ מֵלִיץ הֱיוֹת סוֹדוֹ חוֹשֵׂף.

וְאֶת יְהוּדָה שָׁלַח לְפָנָיו אֶל יוֹסֵף. הוּא שֶׁמֶשׁ הַתְּהִלּוֹת עָלְתָה לַמִּזְרָח מִמַּעֲרָב. וַיְהִי שָׁם לְגוֹי גָּדוֹל עָצוּם וָרָב. בָּנָה אוֹצָרוֹת הַחֲמוּדוֹת. וַיְמַלֵּא אוֹתָם בַּר הַתְּעוּדוֹת. וַיִּצְבֹּר יוֹסֵף בָּר. וּבָא לַעֲרֹךְ שֻׁלְחָן בַּמִּדְבָּר. לָדַעַת לָעוּת אֶת יָעֵף דָּבָר. הִרְוָה צְמֵאִים. וַיְרַפֵּא נְכָאִים. וְהֵשִׁיב לֵב אָבוֹת עַל בָּנִים. וּמָחָה ה' דִּמְעָה מֵעַל כָּל פָּנִים. חִדֵּשׁ זְמַן אַחֲרֵי בְלוֹתוֹ נְעוּרִים. וְרָחַץ בְּמַיִם טְהוֹרִים. וּפָשַׁט אֶת בְּגָדָיו וְלָבַשׁ בְּגָדִים אֲחֵרִים. וּבְיוֹם טָהֳרָתוֹ לָקַח לוֹ שְׁנֵי תֹרִים. תּוֹר הַתּוֹרָה. וְתוֹר הַמִּשְרָה. אֲשֶׁר בּוֹ נִבְחָרִים. וּבִשְׁנֵיהֶם נָאווּ לְחָיָיו בַּתּוֹרִים. צַוָּארָיו בַּחֲרוּזֵי הַשִּׁירִים.

וָאֶשָּׂא מְשָׁלִי:

הַשָּׂר, בְּתוֹךְ פִּיהוּ לְשׁוֹן זָהָב וְאַ-

דֶּרֶת תְּהִלָּתוֹ מְאֹד נֶאְדֶּרֶת.

בָּזַז חֲמוּדוֹת בִּין וְגָנַב לֵב מְתֵי

שֵׂכֶל וְלֹא נִמְצָא בְתוֹךְ מַחְתֶּרֶת.

וּבְעֵת פְּלִילִים חִפְּשׂוּ תוֹךְ אָהֳלוֹ

מָצְאוּ לְשׁוֹן זָהָב וְהָאֶדֶּרֶת.

אַשְׁרֵי הָאִישׁ אֲשֶׁר יְלָדָהוּ. וּצְרוֹר הַמּוֹר אֲשֶׁר צְפָנָהוּ. גָּבַר עַל בְּנֵי דוֹרוֹ. וְכַבֵּן כְּהוֹרוֹ. וְכַשֹּׁרֶשׁ אֲמִירוֹ. וְאֵין לְמִי אַמְשִׁיל אֶחָד מֵהֶם וְאַשְׁוֶה. כִּי מִמִּדוֹת בְּנֵי הַדּוֹר יוֹסֵף אֵינֶנּוּ וְשִׁמְעוֹן אֵינֶנּוּ. הַרְאִיתֶם אֲשֶׁר בָּחַר בּוֹ ה‘. זֶה סִינַי מִפְּנֵי ה’. אֶבֶן יְקָרָה לֹא יָדַע אֱנוֹשׁ עֶרְכָּהּ. אֱלֹהִים הֵבִין דַּרְכָּהּ.

וָאֶשָּׂא מְשָׁלִי:

הָיָה בְאֶרֶץ מַעֲרָב אָדָם בְּתוֹךְ

קִבּוּץ וְאֵין אִישׁ סוֹד יְקָרוֹ חוֹשֵׂף.

עַד כִּי יְדֵי הַיְקָר לְדָרְשׁוֹ מִהֲרוּ

וַיִּמְשְׁכוּ וַיַּעֲלוּ אֶת יוֹסֵף.

הוּא כוֹלֵל הַמִּדּוֹת הַחֲמוּדוֹת. וּבְסוֹד הַתְּעוּדוֹת. לוֹ עֶשֶׂר יָדוֹת. וּבְסוֹד הָרְפוּאוֹת. יְגַלֶּה נִפְלָאוֹת. וּבְחָכְמַת הַמִּדּוֹת. בָּנָה לוֹ בֵּית מִדּוֹת. וּבִשְׁאֵלוֹת הַמִּנְיָן. מֵשִׁיב כָּעִנְיָן. וּבִנְתִיבוֹת הַגַּלְגַּל. יְפַלֵּס כָּל מַעְגָּל. וְכָל רָזֵי הַתּוֹרָה. יוֹצִיאֵם לְאוֹרָה. וְסוֹד הַדִּקְדּוּק. עַל לִבּוֹ חָקוּק. וְיִפְתַּח בִּגְמָרָא. כָּל דֶּלֶת סְגוּרָה. וּבְסוֹד הַמִּשְנָה. מִפִּיו דַּעַת וּתְבוּנָה. וְחָכְמָתוֹ כְבוּדָּה בַת מֶלֶךְ. וּבְבוֹאָהּ לִפְנֵי הַמֶּלֶךְ. תְּשׁוּאוֹת חֵן חֵן לָהּ. וִיבַהֵל אֶת תַּמְרוּקֶיהָ וְאֶת מָנוֹתֶיהָ לָתֵת לָהּ. וַאֲרֻחָתוֹ אֳרֻחַת תָּמִיד נִתְּנָה לָהּ. וּמְנוֹרַת שִׂכְלוֹ בְּאֹהֶל מוֹעֵד הֶעִמיד. לְהַעֲלוֹת נֵר תָּמִיד.
