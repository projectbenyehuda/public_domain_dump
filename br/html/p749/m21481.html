הרופא

וְיֵשׁ שֶׁיִּתְפָּאֵר בְּחָכְמַת הָרְפוּאוֹת, / יְסַפֵּר עַל עַצְמוֹ לַעֲשׂוֹת נִפְלָאוֹת / וְכִי עִמּוֹ סוֹדוֹת בְּקִירוֹת לִבּוֹ חֲרוּתִים / לְהַחֲזִיר נְשָׁמוֹת לִפְגָרִים מֵתִים. / יֹאמַר ״יֶשׁ לִי סֵפֶר רְפוּאוֹת הַגָּנוּז / וְסִפְרֵי אַבּוּקְרָט וְגָלֵינוּס / וְסִפְרֵי שְׁאָר הַחֲכָמִים וְסוֹדוֹתֵיהֶם / הַבָּאִים אַחֲרֵיהֶם, / הַכֹּל בַּחֲדָרַי הֲכִינוֹתִי / וְהַכֹּל קָרָאתִי וַהֲבִינוֹתִי. / כַּמָּה מְלָאכוֹת נֶעֱרָכוֹת עָשִׂיתִי בַזְּמָן / יָמָּה וְקֵדְמָה צָפוֹן וְתֵימָן! / רִפֵּאתִי אֲנָשִׁים מִשְּׁחִין חִזְקִיָּהוּ וְצָרַעַת נַעֲמָן״ – וְאוֹתוֹ הַמְּזֻיָּף מִכָּל אֵלֶּה יָתוֹם וְאַלְמָן. / וַהֲמוֹן הַסְּכָלִים בִּרְאוֹתָם טַבַּעַת בְּיָדוֹ וְעָלָיו בֶּגֶד אַרְגָּמָן / יַחְשְׁבוּהוּ רוֹפֵא מֻמְחֶה וְנֶאֱמָן, / יֹאמְרוּ שֶׁאַרְבַּע עֶשְׂרֵה שָׁנָה נִטְמַן / בְּבֵית עֵבֶר הַגִּלְעָדִי רֹאשׁ רוֹפְאֵי גִלְעָד, / מִפְּרִי שְׂפָתָיו לְבַב אֱנוֹשׁ יִסְעַד / הַרְבֵּה לָמַד מֵרַבּוֹתָיו וְחָכְמָה גְדוֹלָה קָנָה, / מַפְלִיא לַעֲשׂוֹת אַחַר הַיֵּאוּשׁ בְּמַה שֶּׁאֵין בּוֹ סַכָּנָה – / סַכָּנַת עֲלוּקָה – הַב הַב / דִּינְרֵי זָהָב – / תְּנָה אֶת הַמָּנָה!

שִׁמְעוּ, עַמִּים כֻּלָּם, כַּמָּה יִתְנַהֵג זֶה בֶּאֱמוּנָה בִמְלַאכְתּוֹ! / כְּשֶׁיִּקְרַב לַחוֹלֶה בְּיוֹם מַפַּלְתּוֹ / יֹאמַר שֶׁהוּא מְסֻכָּן מִמֻּרְסָא גְדוֹלָה בַמֹּחַ תַּחַת גֻּלְגָּלְתּוֹ / ״בִּינוֹתִי יָפֶה עִנְיָנוֹ מִתּוֹךְ דָּפְקֵי הַיָּדַיִם / וְעָבִיט שֶׁל מֵימֵי רַגְלַיִם, / אוֹתוֹתֵינוּ אוֹתוֹת יְדוּעִים לֹא מֵאֶתְמוֹל וּמֵאֶמֶשׁ – / בָּרוּר לִי הַדָּבָר כַּשֶּׁמֶשׁ! / וְעַתָּה הָבוּ שְׂכָרִי חֲמִשִּׁים כֶּסֶף, / וְאִם לֹא – וְאֵדְעָה מֶה חָדֵל אָנִי, / מָה אֲנִי בְשָׂכָר אַתֶּם גַּם אַתֶּם, כִּי לֹא בְחִנָּם מְלַמְּדֵנִי הוֹרַנִי. / עֲשׂוּ אַתֶּם רִאשׁוֹנָה פֵּרָעוֹן חָשׁוּב / וְהַנִּשְׁאָר – אַחַר אֲשֶׁר לִבְרִיאוּתוֹ יָשׁוּב. / בְּזֹאת אֵאוֹת לָכֶם לְהִתְעַסֵּק בּוֹ, / אֲשִׁיבֶנּוּ רַעֲנָן בְּאִבּוֹ / וַהֲסִירוֹתִי מַחֲלָה מִקִּרְבּוֹ.״ / כְּלָלוֹ שֶׁל דָּבָר: אֵין לוֹ בֹשֶׁת / לְבַקֵּשׁ שָׂכָר גָּדוֹל עַל הַמַּשְׁקֶה וְעַל הַתַּחְבֹּשֶׁת, / בַּחֲלַקְלַקּוֹת יַבְטִיחַ לַעֲשׂוֹת מִרְקַחַת / לְהַתִּיךְ הַמֻּרְסָא וּלְהָסִיר הַקַּדַּחַת – / וְהָעֲנִיִּים גּוֹאֲלֵי הַחוֹלֶה אֲשֶׁר בִּרְפוּאָתוֹ יִרְצוּ / כֶּסֶף לֹא יַחְשֹׁבוּ וְזָהָב לֹא יַחְפְּצוּ, / כְּכָל הַיּוֹצֵא מִפִּיו יַעֲשׂוּ, / מַה שֶּׁגָּמַר אֹמֶר – / יָמֹדוּ בָעֹמֶר.

וְהָרוֹפֵא אַחַר שֶׁמָּעוֹת מָשַׁךְ וְהַכֹּל בְּכִיסוֹ שָׂם / יִקְרָא אֶל הַבַּשָּׂם / לֵאמֹר ״כְּתֹב זֹאת זִכָּרוֹן בַּסֵּפֶר: / קַח לְךָ סַמִּים – נָטָף וּשְׁחֵלֶת הַבָּאִים מֵאֶרֶץ חֵפֶר, / וְשָׁחַקְתָּ הָדֵק כֶּעָפָר וָאֵפֶר, / תְּעָרֵב עִמָּהֶם קִמּוֹשׁ וָחוֹחַ / וַעֲפַר תִּחוּחַ, / שְׂרַף עָרְלָה וְשֹׁרֶשׁ רְתָמִים, / עֲלֵי זַיִת וְדַם צִפֹּרֶת כְּרָמִים, / קְלִפַּת צְנוֹנוֹת / וּמִיץ חָלָב מֵעֶשֶׂר אֲתוֹנוֹת, / פֶּרַח שׁוֹשָׁן וְזֶרַע מְרֵעִים וַאֲבַק לְשׁוֹן הָרַע – / לֹא תוֹסִיף עַל אֵלֶּה וְלֹא תִגְרַע! – / תְּרָדִין לִרְפוּאָה זֶרַע וְיָרָק / וְתָשִׂים הַכֹּל בְּמִזְרָק. / תְּחַבֵּר אֲלֵיהֶם שִׁבּוּשׁ הַסְּבָרָא / וּמְעַט קוֹל הֲבָרָה, / וְתַעֲבִיר הַכֹּל בַּכְּבָרָה, / בַּד בְּבַד יִהְיֶה – / בִּטְחוֹנִי בְקוֹנִי בְּזֹאת הַהַרְכָּבָה מֵחֳלִי זֶה חָיֹה יִחְיֶה. / עֲשֵׂה מִמֶּנּוּ רְטִיָּה / וְהַנִּיחָהּ לוֹ כְּנֶגֶד הַלֵּב מְקוֹם שֶׁהַנְּשָׁמָה תְלוּיָה – / מִיָּד יָסוּר כְּאֵבוֹ וְעַיִן בְּעַיִן תֶּחֱזֶה, / לְמָחָר יִהְיֶה הָאוֹת הַזֶּה! / וְהַמִּרְקַחַת שֶׁעָשִׂיתִי לְצִירוֹ – מִלִּבִּי יְסַדְתִּיו, / אֲנִי הַיּוֹם יְלִדְתִּיו. / זֶה תָאֳרוֹ: קַח אֵזוֹב יָוָן וְאֵזוֹב כֹּחֲלִי / שֶׁיֵּשׁ בּוֹ סְגֻלָּה לְכָל חֳלִי, / שְׂרוֹךְ הַמִּנְעָל / וְשֵׁן שֶׁל שׁוּעָל / יִתְדוֹת הָאֹהֶל וְסִכְּתָא בְגֻדָּא – / מִכָּל אֶחָד אֲגֻדָּה. / מִיץ אֶבֶן שַׁיִשׁ, / צִפָּרְנֵי לַיִשׁ, / רֹאשׁ שִׁבֹּלֶת / וַחֲלֵב תַּרְנְגֹלֶת, / עֲצֵי הַכְּלוּב, / דִּדְלוּז וְדִדְלוּב / וּמַסְמֵר הַצָּלוּב, / אַבּוּב רָעֲיָא / זַרְקָא תְּבִיר וְגַעְיָא, / כְּמוֹ שֶׁכָּתוּב בַּסֵּפֶר שֶׁגָּנַז חִזְקִיָּה – / מִכָּל אֶחָד אוֹקִיָּא. / קְלִפֵּי רִמּוֹנִים / וְעָקְצֵי תְאֵנִים, / אֶבֶן תְּקוּמָה / וְדִמְדּוּמֵי חַמָּה, / בְּרַק חֲנִית / וְסֶלַע שֶׁעַל גַבֵּי הַצִּנִּית, / קִדָּה וְכַמּוֹן / וּמֵי רִמּוֹן – / מִכָּל אֶחָד דַּרְכְּמוֹן. / קִלּוּף חִוָּר דִּי עַל חֻטְרָא, / צֵל דָּפְנוֹת עֲנַן בֹּקֶר זִקָּא דְבָתַר מִטְרָא – / מִכָּל אֶחָד חֲצִי לִטְרָא. / הָדֵק הֵיטֵב הַסַּמָּנִים הַקָּשִׁים מִפֹּה וּמִפֹּה / וְעָרֵב בָּהֶם כָּרָאוּי דְּבַשׁ צִרְעָה שֶׁהוּסַר קִצְפּוֹ, / עֲשֵׂה מִמֶּנּוּ מִרְקַחַת מֻפְלָא כְּמוֹ שֶׁתּוּכַל / וְהֵבֵאתָ לַחוֹלֶה וְאָכַל. / וְאִישׁ אֲשֶׁר יִרְקַח כָּמוֹהוּ לְאָכְלָה / לֹא יֵדַע שְׁכוֹל וְלָא מַלִּיף תִּכְלָא. / וְאוּלָם הַמַּשְׁקֶה / אֲשֶׁר בּוֹ מִכָּל רַע יִנָּקֶה – / גְּרוֹגְרוֹת וְצִמּוּקִים / מֵחַרְצַנֵּיהֶם מְנֻקִּים / עֲסִיס רִמּוֹנֵי בָדָן וּקְלִפּוֹתָיו / וַעֲלֵי כְרוּב וְרוֹבְיָא כְּדֵי שֶׁיִּרְבּוּ זְכִיּוֹתָיו / חֲלֵב הַמִּנְעוֹל וְדָמוֹ / וְדָגִים קְטַנִּים מֵהַיָּם שֶׁעָשָׂה שְׁלֹמֹה, / לַגְלוֹנוֹת הַבָּאוֹת מֵאֶרֶץ תֵּימָא, / יוֹתֶרֶת הַכָּבֵד וּמוֹרִנֵּי בְהֵמָה. / מִכָּל אֵלֶּה חֵלֶק בְּחֵלֶק, שָׁוֶה בְשָׁוֶה, / וּתְבַשֵּׁל הַכֹּל עַד שֶׁיִּהְיֶה מְעַט עָבֶה. / יִשְׁתֶּה אוֹתוֹ בָעֶרֶב מָזוּג בְּמֵי אַפְסַיִם, מֵי שָׂחוּ – / מֵרֵיחָם יַפְרִיחַ שָׂדֶה, יִשְׂגֶּה אָחוּ, / וְיִתֵּן אַחֲרָיו שֵׁנָה לְעֵינוֹ / וְיָשׁוּב לִפְנוֹת בֹּקֶר לְאֵיתָנוֹ״.

וְהָיָה כְּכַלּוֹת הָרוֹפֵא דְבָרָיו / וְתַחְבּוּלוֹתָיו וּשְׁקָרָיו, / יָשׁוּב לְבֵיתוֹ שָׂמֵחַ, יִשְׁתֶּה וְיֹאכֵל – / וְהָאֲנָשִׁים קְרוֹבֵי הַחוֹלֶה יְמַהֲרוּ אֶל בֵּית הָרוֹכֵל / לְהָבִיא הָרְפוּאוֹת / וְכָל אוֹתָן הַנֻּסְחָאוֹת. / וְלַתּוֹעֶלֶת אֲשֶׁר יְקַוּוּ / יַעֲשׂוּ כֹל מַה שֶּׁנִּצְטַוּוּ. / בַּלֵּב יַחְשְׁבוּ, בַּיָּד יִקָּחוּ – / לֹא יַעַבְרוּ מִמִּצְווֹתָיו וְלֹא יִשְׁכָּחוּ. / כָּל אֶחָד וְאֶחָד בִּזְרִיזוּת יַעַבְדֶנּוּ – / וְעִם כָּל זֶה בַּחֲצִי הַלַּיְלָה הַמָּוֶת יִתְפְּשֶׂנּוּ, / רוּחַ עָבְרָה בּוֹ וְאֵינֶנּוּ! / וְהָרוֹפֵא עַל זֶה לֹא יַקְפִּיד אַחַר שֶׁנִּפְרַע, / לֹא יִבְכֶּה לוֹ וְקָרוֹעַ לֹא יִקְרַע, / לֹא יַעְטֶה עַל שָׂפָם וְרֹאשׁוֹ לֹא יִפְרַע / לֹא יִנְהַג אֶבְלוֹ לֹא שִׁבְעָה וְלֹא שְׁלֹשִׁים, / אַף עַל פִּי שֶׁלָּקַח מִמְּמוֹנוֹ קֹדֶם כָּל הַיּוֹרְשִׁים!

כֵּן דֶּרֶךְ רוֹפְאֵי זְמַנֵּנוּ: עֵינֵיהֶם בַּמָּמוֹן וְלֹא יָחוּשׁוּ לְמַה שֶּׁיִהְיֶה, / אִם יָמוּת גֶּבֶר הֲיִחְיֶה, / לֹא יָחוּשׁוּ אִם בְּעִתָּם יָמוּתוּ כַמָּה עֲנִיִּים / וְאִם תִּהְיֶה נַפְשָׁם צְרוּרָה בִצְרוֹר נָקוּב אוֹ בִצְרוֹר הַחַיִּים. / וְרָעָה מִזֹּאת – יִצְעַק חָמָס זֶה הַמְזֻיָּף עַל הַקְּרוֹבִים וְהַיְדִידִים / לֵאמֹר ״אַתֶּם הֲמִתֶּם אֶת הָאִישׁ בַּדְּבָרִים הַמִּתְנַגְּדִים! / אוֹתוֹ בְיֶדְכֶם שִׁוִּיתִי – / וְאַתֶּם מְרִיתֶם פִּי בְּכָל אֲשֶׁר צִוִּיתִי. / כֻּלְְכֶם פְּשַׁעְתֶּם בּוֹ בְּשׂוּמְכֶם מִצְוָתִי לְאַיִן / וַתַּשְׁקוּ אֶת הַחוֹלֶה יַיִן״. / סוֹף דָּבָר – כֻּלָּם טוֹפְלֵי שֶׁקֶר, רוֹפְאֵי אֱלִיל, יְיָ לֹא יִירָאוּ, / יְדֵיהֶם דָּמִים מָלֵאוּ. / עַל כֵּן אָמַרְתִּי: רַבּוֹתֵינוּ הַקְּדוֹשִׁים לֹא אָמְרוּ בְחִנָּם / ״טוֹב שֶׁבָּרוֹפְאִים לְגֵיהִנָּם!״
