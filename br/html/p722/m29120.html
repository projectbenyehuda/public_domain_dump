אֲשֶׁר חָפֵץ קְנוֹת חָכְמָה

[הלכות שחיטה]


אֲשֶׁר חָפֵץ קְנוֹת חָכְמָה וּבִינָה / גְּשָׁה אַשְׂכִּילְךָ דָּת הָאֱמוּנָה:

גְּשָׁה אַשְׁמִיעֲךָ דִּבְרֵי נְגִידִים / וְאַָדְרִיכָךְ עֲלֵי דֶרֶךְ נְכוֹנָה:

פְּנֵה אֵלַי וְאַל תָּקוֹץ בְּמִלַּי / וְעֵין לִבָּךְ לְהַשְׂכִּילָם פְּקַח נָא:

הֲלוֹךְ אַחַר עֲבוֹדַת אֵל וְדָתוֹ / שְׁמוֹר מִסּוּר לְךָ אָנֶה וְאָנָה:

וְיִרְאָתוֹ בְּפָנֶיךָ לְעֵדָה / וְאַהְבָתוֹ בְּלִבָּךְ שִׂים צְפוּנָה:

יְרָא אוֹתוֹ לְבַל תֵּט מֵאֲמָרָיו / וְאָהְבֵהוּ לְמַעַן תַּעֲשֶׂנָּה:

וְאַל תֵּט אַחַר אַוַּת לְבָבָךְ / לְבַל תַּעְמוֹד עֲלֵי יָמִין לְשִׂטְנָה:

וְאַל תֵּט אַחֲרֵי אֹכֶל וּמִשְׁתֶּה / וְאַל תֵּט אַחֲרֵי דִבְרֵי רְגִינָה:

וְלֹא תַבְרֶה בְּבָשָׂר כִּי לְתַאְוָה / לְהָקִים נַפְשְׁךָ רַק לֹא לְעֶדְנָה:

וְהִזָּהֵר בְּתוֹרַת הַשְּׁחִיטָה / כְּמִשְׁפָּטָהּ בְּעֵדוּת נֶאֱמָנָה:

וְנִשְׁחָט כַּהֲלָכָה תֹאכְלֵהוּ / אֲשֶׁר בּוֹ נִקְבְּצוּ אָבוֹת שְׁמֹנָה:

וְלֹא תֹאכַל בְּהֵמָה הַמְצַפָּה / לְיַד מָוֶת הֱיוֹתָהּ לוֹ לְמִקְנָה:

וְהִיא הִיא הַנְּבֵלָה בַּחֲיוֹתָה / מְשַׁמֶּרֶת יְמֵי מָוֶת וּמוֹנָה:

וְהִזָּהֵר בְּמִנְיַן הַנְּבֵלוֹת / אֲשֶׁר מָנוּ לְךָ אַנְשֵׁי תְבוּנָה:

וְהֵם שׁוֹחֵט בְּכוֹבַע אוֹ בְמַשְׂעִיר / וּבַמָּקוֹם אֲשֶׁר תַּרְבֵּץ מְכֻנָּה:

וְהַשּׁוֹחֵט בְּהַקָּנֶה בְּמָקוֹם / אֲשֶׁר יֵצֵא לְחוּץ מֵאֵין רְצוֹנָה:

וְשֶׁלֹּא נִבְדְקָה אַחַר שְׁחִיטָה / וְהַשּׁוֹחֵט חֲצִי וֶשֶׁט וְקָנָה:

וְהַשּׁוֹחֵט בְּפָגוּם אוֹ מְחֻבָּר / וְהַשּׁוֹחֵט בְּבָתֵי הַשְּׁכִינָה:

וְהַשּׁוֹחֵט בְּיוֹם שַׁבָּת וְכִפּוּר / וְהַשּׁוֹחֵט לְחַמָּה אוֹ לְבָנָה:

וְהַשּׁוֹחֵט לְשֵׁם קֹדֶשׁ אֲשֶׁר בָּא / בְּנֵדֶר מִבְּלִי חֵטְא אוֹ תְאוּנָה:

וְכֵן אִם שָׁחֲטָה סַכִּין בְּנָפְלָהּ / וְאַפִלּוּ שְׁחִיטָתָהּ כְּתָקְנָהּ:

וְהַשּׁוֹהֶה וְהַדּוֹרֵס וּמַגְרִים / וְעִקּוּר עִם שְׁחִיטָה הַטְּמוּנָה:

וְיִשְׂרָאֵל אֲשֶׁר יִשְׁחַט לְבַדּוֹ / בְּלִי שֵׁנִי וְלֹא לָמַד וְשָׁנָה:

וְהַחֵרֵשׁ וְהַשּׁוֹטֶה וְקָטָן / וְהַשִּׁכּוֹר שְׁאֵין דַּעְתּוֹ נְכוֹנָה:

וּמִינִים אוֹ מְשֻׁמָּדִים וְכֵן כָּל / אֲשֶׁר סָר מֵעֲבוֹדַת אֵל וּפָנָה:

וְגַם הֵן אֶסְפְּרָה לָךְ הַטְּרֵפוֹת / וְאַעַרְכֵם כְּשׁוֹשַׁנִּים בְּגַנָּה:

וְאֵלוּ הֶן: דְּרוּסָה אוֹ נְקוּבָה / בְּהַתַּרְבֵּץ וּבַלֵּב אוֹ בְקָנָה:

וְעוֹר מוֹחַ וְהַמָּרָה וְכָרֵס / קְנֵי כָבֵד וְקֵבָה הַנְּתוּנָה:

וּבֵית כּוֹסוֹת וּמֵעַיִם וְהַמְסָס / וּמֵעַיִם בְּהָפְכָם מִמְּכוֹנָה:

וְחֶסְרוֹן הַמְּנֻקָּבִים וְיִתְרָם / וְרֵאָה הַנְּקוּבָה אוֹ מְשֻׁנָּה:

וְהָרֵאָה וְהַמּוֹחַ בְּהִמּוֹס / וְהַסִּמְפּוֹן בְּהִמּוֹק תֶּחֱזֶה נָא:

וְכֵן סִמְפּוֹן בְּהִנָּקֵב, וְכֵן כִּיס / אֲשֶׁר מָרָה שְׁחוֹרָה בוֹ שְׁכֵנָה:

וְהִנָּקֵב קְנֵי רֵאָה לְמַטָּה / וְכֵן רֵאָה חֲלָקָה כַגְּבִינָה:

וְהַנִּמְצֵאת נְפוּחָה אוֹ אֲטוּמָה / וְכֵן אִם צָמְקָה מֵרֹב מְגִנָּה:

וְכֵן אִם חָסְרָה מִקְצָת, וְיָבֵשׁ / קְצָת גּוּפָהּ וְנִפְרָךְ כַּלְּבֵנָה:

וְכֵן מַיִם סְרוּחִים אוֹ עֲכוּרִים / וְאִם רֵיחָם בְּלִי נָמָר וְשָׁנָה:

וְכֵן לִיחָה סְרוּחָה אִם תְּהִי בָהּ / וְכֵן אִם נִסְמְכָה אוּנָּה לְאוּנָּה:

וְהָאוּנּוֹת בְּחֶסְרוֹן אוֹ בְיִתְרוֹן / וְחִלּוּפָם אֲשֶׁר לֹא כַתְּבוּנָה:

וְהָרֶגֶל בְּחָסְרָהּ אוֹ בְיִתְרָהּ / וְצֵאת יָרֵךְ אֲשֶׁר בַּכַּף אֱמוּנָה:

וְכֵן אִם נִטְּלָה צֻמָּה וְכָבֵד / וְהַלֶּחִי וְכִלְיָה הַקְּטַנָּה:

וּמֵימֶיהָ סְרוּחִים אוֹ עֲכוּרִים / וְלִיחָה בָהּ וּמַכָּה נֶאֱמָנָה:

וְחוּט שִׁדְרָה בְּהִפָּסֵק וּמוֹחוֹ / בְּהִתְמַסְמֵס, וּפָשַׁט עוֹר בְּצִנָּה:

וְסִימָנִין בְּהִדַּלְדֵּל, וּוֶשֶׁט / בְּהֵהָפֵךְ אֲדַמְדֶּמֶת לְבָנָה:

וְדִינֵי הַנְּפוּלָה אוֹ קְרוּעָה / בְּרֹב בָּשָׂר אֲשֶׁר חוֹפֶה לְבִטְנָהּ:

וְכֵן אִם נִשְׁבְּרוּ רֹב הַצְּלָעוֹת / וְנֶעְקָרוּ לְחוּץ וַתִּגְבְּהֶינָה:

וְכֵן אִם נֶעְקְרָה חֻלְיָה, וְצֵלַע / בְּחֻלְיָתָהּ אֲשֶׁר הִיא בָהּ סְפוּנָה:

וְגֻלְגֹּלֶת בְּהֵחָבֵס בְּרֻבָּהּ / וְכֵן אִם חָסְרָה סֶלַע הֲגוּנָה:

הֲרֵי שִׁבְעִים טְרֵפוֹת הֶם מְנוּיִים / שְׁמוֹר אוֹתָם וְסוּרָה מִתְּלוּנָה:

וְאַל תָּקֵל בְּדִינֵי אֵל, וְשָׁמְרֵם / כְּבַת עַיִן, וְקוֹלָן הַאֲזִינָה:

לְמַעַן אִם לְמִשְׁמַעְתּוֹ תְהִי סָר / יְהִי חַסְדּוֹ לְךָ מָגֵן וְצִנָּה:

תְּנָה נַפְשָׁךְ לְיוֹצֶרְךָ לְמָנָה / וְהוּא יִתֵּן לְךָ חַסְדּוֹ לְמָנָה:

וְהַלֵּךְ בַּעֲנָוָה עַל שְׁבִילוֹ / וּמַאְמַר פִּיו וּמִצְווֹתָיו שְׁמָר־נָא:

וְהַעְלֵה לִבְּךָ מִיַּם חֲטָאתוֹ / וְעוֹרֵר נַפְשְׁךָ מִבּוֹא בְשִׁינָה:

דְבֹק לָךְ בַּאֲשֶׁר אָהֵב לְדָבְקָה / וְסוּרָה מֵאֲשֶׁר יִזְנַח וְיִשְׂנָא:

הֲכִי אִם תַּעֲשֶׂה כָזֹאת, תְּפַלֵּט / יְחִידָתָךְ וְתַמְצִיאָה חֲנִינָה:

וּבֹשֶׁת יִלְבְּשׁוּ כָּל שׂוֹנְאֶיךָ / וְיִשָּׁמַע עֲלֵי פִיךָ רְנָנָה:

וְתִזְכֶּה לַחֲזוֹת יֶשַׁע, וְתִהְיֶה / יְחִידָתָךְ בְּמַעְדָּנִים דְּשֵׁנָה:

בְּעֶת־בּוֹא קֵץ גְּאוּלָה הָעֲתִידָה / לְעוֹרֵר אַהֲבַת אָבוֹת יְשֵׁנָה:

וְיָחֹן בַּת שְׁחַרְחֹרֶת עֲנִיָּה / וְיִשְׁבּוֹר עֹז גְּבֶרֶת הָעֲדִינָה:

וְשִׁמְשָׁהּ יַעֲרִיב לַצָּהֳרַיִם / וְיַשְׁכִּין עַל מְאוֹר יוֹמָהּ עֲנָנָה:

וְכִפָּתָהּ וְאַגְמוֹנָהּ יְקוֹסֵס / וְלֹא תִהְיֶה לְעוֹלָם רַעֲנָנָה:

בְּבוֹא קִצָּהּ וְאֵין עוֹזֵר לְנַפְשָׁהּ / וְיִשָּׁבֵר בְּפֶתַע רֹאשׁ גְּאוֹנָהּ:

וְרֹאשׁ שָׂעִיר בְּתַעַר הַשְּׂכִירָה / יְגַלַּח גַּם יְכַס פָּנָיו בְּבָשְׁנָה:

לְמַעַן גָּבְהָה קוֹמַת אֲרָזָיו / בְּהִמָּשֵׁךְ שְרָרוּתוֹ עֲדֶנָּה:

אֱלֹהִים רִיב לְאוֹיְבֵנוּ וְהָרֵק / לְמוּלָם חֵץ וּפִי חֶרֶב שְׁנוּנָה:

וְשׁוֹבֵב יַעֲקֹב אֶל מִשְׁכְּנוֹתָיו / וְיִתְעַלֵּס בְּאַהְבָה נַעֲמָנָה:

וְהַזְרִיחַ מְאוֹר שִׁמְשׁוֹ וְתִמְשׁוֹךְ / רְצוֹנָךְ אֶל דְּבִיר קָדְשָׁךְ וְנִבְנָה:

לְמַעַן כִּי יְסֻפַּר חַסְדְּךָ אֵל / לְדוֹר וָדוֹר וּמִשָּׁנָה לְשָׁנָה:
