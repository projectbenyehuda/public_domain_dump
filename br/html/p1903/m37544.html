משל ב: עַכְבָּר עִם צְפַרְדֵּעַ וָנֶשֶׁר

בּוֹר כָּרָה וַיַּחְפְּרֵהוּ / וִישֻׁלַּם לוֹ כִּפְרִי מַעַלְלֵהוּ

עַכְבָּר יָשַׁב עַל מִפְתַּן בֵּית הָרֵחַיִם / וַיְמַלֵּא קֶמַח שְׂפָתַיִם וּמַלְקוֹחַיִם / וְהִנֵּה הַשֶּׁמֶשׁ זָרְחָה עָלָיו / וּבְכַפּוֹת יָדָיו וְרַגְלָיו / יִשְׁתַּעֲשַׁע וּמְמַשֵּׁשׁ בִּשְׂפָמוֹ / כְּמִשְׁפַּט אָבִיו וּבְנֵי עַמּוֹ / וּבִהְיוֹתוֹ שָׁם בְּלִי פַחַד וְאֵימָה / צְפַרְדֵּעַ עוֹבֶרֶת לְפִי תֻמָּהּ / וַתַּעֲמֹד לַחֲקֹר כָּל עִנְיָנוֹ / וְאִם אָדוֹן בְּבֵיתוֹ הִנּוֹ / וַיַּעַן לָהּ הָעַכְבָּר / לִי הַבַּיִת וְלִי הַבָּר / וְלִי הַזֶּרַע בַּמְּגוּרָה / לְאָכְלָה וְלֹא לִסְחוֹרָה / וְהִנֵּה בֵּיתִי סִתְרִי / וָאֲמַלֵּא טֶרֶף חוֹרִי / וּבְנוֹת הָרֵחַיִם וְהַשֶּׁכֶב / סְבִיבוֹתָם אֲנִי שׁוֹכֵב / וְיַסְפִּיקוּ לִי רֹב מַכֹּלֶת / שְׁנֵיהֶם מְלֵאִים סֹלֶת / כֹּה מִשְׁפָּטִי כָּל הַיָּמִים / וּכְשָׁמְעִי הֲמוֹן עַמִּים / וְאֶתְחַבֵּא בְנִקְרוֹתַי / וּמִשָּׁם תּוֹצְאוֹתַי / בִּרְאוֹתִי כִּי אֵין גֶּבֶר / וְאֵלֵךְ לִשְׁבֹּר שֶׁבֶר / וַתֵּרֶא הַצְּפַרְדֵּעַ לֹבֶן עַל כָּל אֲבָרָיו / וְעַל יָדָיו וְעַל חֶלְקַת צַוָּארָיו / וַתֹּאמֶר לוֹ: מַדּוּעַ לָבָן לִלְבוּשֶׁךָ / הַאִם שֵׂיבָה זָרְקָה עַל רֹאשֶׁךָ / וַיֹּאמַר: עִם לָבָן גַּרְתִּי / וּמִסֹּלֶת חִטִּים טָעַמְתִּי / וְהַקֶּמַח עָלַי יָרַד / מְפֻזָּר וּמְפֹרָד / כִּי לֹא נִסְגְּרוּ בְּפָנַי דַּלְתּוֹת / הַקְּמָחִים וְהַסְּלָתוֹת / וַתֹּאמֶר לוֹ הַצְּפַרְדֵּעַ בְּמִרְמָה / כָּל חַיֶּיךָ פַּחַד וְאֵימָה / מַה תִּתְהַלֵּל בְּסֹלֶת וָמֶגֶד / וְחַיֶּיךָ תְּלוּאִים לְךָ מִנֶּגֶד מִכָּל הַבָּא אֶל הָאֹהֶל / תִּגַּע עָדֶיךָ וְתִבָּהֵל / וּמִבְּנֵי אִישׁ שָׁם מַקְהֵלוֹת / וְרַב מִהְיוֹת קוֹלוֹת / וְכֹל בְּךָ עֹצֵר וְחֹבֵשׁ / וּמַאֲכָלְךָ קָשֶׁה וְיָבֵשׁ / וּבְגָדֶיךָ תָּמִיד מְעֻפָּרִים / אַךְ בְּגָדַי נְקִיִּים וְנִנְעָרִים / וּבְמִשְׁכְּנוֹתַי אֵין אֹרֵב / וּמַאֲכָלִי דָּשֵׁן וְעָרֵב / סְבִיבוֹתֶיךָ פַּחִים / וּמִשְׁכְּנוֹתַי מִבְטַחִים / שְׁמַע אֲמָרַי וּמִלַּי / וְכָל מַחְסֹרְךָ עָלַי / בֹּא אִתִּי בִמְרוּצָה / בְּסֵתֶר קָנֶה וּבִצָּה / וַתַּדִּיחֶנּוּ בְחֵלֶק שְׂפָתֶיהָ / וַתֵּלֶךְ לְפָנָיו מֹשֶׁכֶת יַרְכוֹתֶיהָ / וְהָעַכְבָּר לֹא נִזְהָר / עַד בּוֹאָם אֶל שְׂפַת הַנָּהָר / וַיֹּאמֶר לַצְּפַרְדֵּעַ: וְאָנָה / בֵּיתֵךְ אֲשֶׁר אַתְּ בּוֹ חֹנָה / וַתַּעַן לוֹ: מֵעֵבֶר לַנָּהָר הַזֶּה / וְאַתָּה מִכָּל אֲשֶׁר בְּאָהֳלִי תֶּחֱזֶה / הַטּוֹב וְהַיָּשָׁר בְּעֵינֶיךָ / וְשִׂים אֶל הַנָּהָר פָּנֶיךָ / לַעֲבֹר מֵי מְנוּחוֹת / כַּאֲשֶׁר יְפָרֵשׂ הַשֹּׂחֶה לִשְׂחוֹת / וְעֵין הָעַכְבָּר עַל נַפְשׁוֹ חָסָה / וַיּוֹאֶל לָלֶכֶת כִּי לֹא נִסָּה / וַיֹּאמֶר לָהּ: מִשְׁכְּנוֹתַיִךְ אֲשֶׁר אָמַרְתְּ אַיָּם / מִי יַעֲבֹר לָנוּ אֶל עֵבֶר הַיָּם / לַעֲבֹר בַּיָּם לֹא נִסִּיתִי / טוֹב לִי לָשֶׁבֶת בְּבֵיתִי / וְלִהְיוֹת שָׁם כָּל יְמֵי חֶלְדִּי / כִּי יָרַט הַדֶּרֶךְ לְנֶגְדִּי / וַתֹּאמֶר לוֹ: אֲדֹנִי שְׁמָעֵנִי / כִּי תַּעֲבֹר בַּמַּיִם אִתְּךָ אָנִי / וּנְהָרוֹת לֹא יִשְׁטְפוּךָ / וְתַרְחִיק שׂוֹנְאֶיךָ אֲשֶׁר רְדָפוּךָ / אֶת הַחוּט הַזֶּה בְּרַגְלַי וּבְרַגְלֶיךָ תִּקְשֹׁר / וְיַעַמְדוּ רַגְלֶיךָ בְמִישׁוֹר / וְאֵלֵךְ עִמְּךָ לִנְחוֹתְךָ הַדָּרֶךְ / בְּצָמִיד פָּתִיל אֲשֶׁר תִּקְשֹׁר לִי בַיָּרֵךְ / וּבְמַעֲשֶׂה וּבִשְׁמוּעָה / לֹא תְאֻנֶּה אֵלֶיךָ רָעָה / וְאַתָּה בֵּין אַחִים תַּפְרִיא / עַתָּה תִרְאֶה הֲיִקְרְךָ דְבָרִי / וַיֹּאמַר: אִם כִּדְבָרֵךְ בְּרִית לָךְ אֶכְרֹת / גַּלִּי שׁוֹק עִבְרִי נְהָרוֹת / הִטַּתּוּ בְרָב לִקְחָהּ / וְרוּחַ חֲבוּרָה עֲלֵיהֶם נָחָה / וַיֵּלְכוּ צְמָדִים הַצְּפַרְדֵּעַ וְהָעַכְבָּר / בְּצָמִיד פָּתִיל אֲשֶׁר בְּרַגְלֵיהֶם חֻבָּר / וּבְהַגִּיעָם לְעִמְקֵי תְהוֹם וּמְצוּלָה / וַתַּחְשֹׁב הַצְּפַרְדֵּעַ לְהַטְבִּיעוֹ וּלְהוֹרִידוֹ שְׁאוֹלָה / וּבְהִתְחַזֵּק הָעַכְבָּר עַל פְּנֵי הַמַּיִם / דָּרַךְ הַנֶּשֶׁר בַּשָּׁמַיִם / וַיַּרְא אוֹתָם זֶה בָזֶה מְגֹלָל / וַיַּעַט אֶל הַשָּׁלָל / אַכְזָרִי הוּא וְלֹא יְרַחֵם / יִפְרֹשׂ כְּנָפָיו יִקָּחֵם / וּבִפְתִיל חָזָק אֲשֶׁר יַחְדָּו נִצְמָדוּ / יִתְלַכְּדוּ לֹא יִתְפָּרָדוּ / וַיְהִי הַנֶּשֶׁר חָשׁ לֶאֱכוֹל / וַיַּרְא כִּי אֶל הָעַכְבָּר לֹא יָכוֹל / כִּי צָפַד עוֹר חָזָק עַל גַּבּוֹ / וְשֵׂעָר שָׁחוֹר צָמַח בּוֹ / וַיָּשֶׁת אֶל הַצְּפַרְדֵּעַ פָּנָיו / נֶחְמֶדֶת לְמַרְאֶה וְטוֹבָה לְמַאֲכָל בְּעֵינָיו / וַיִּבְלָעֶנָּה בְּפַעַם אַחַת וְלֹא שָׁנָה / וַיִּפֶן הָעַכְבָּר הֵנָּה אָנֶה וְאָנָה / וַיַּרְא כִּי מוֹסְרוֹתָיו נִתָּקוּ / לָנוּס מַהֵר רַגְלָיו הִתְחַזָּקוּ / וַיִּמָּלֵט רַק הוּא לְבַדּוֹ / וְהַצְּפַרְדֵּעַ אֲשֶׁר נִקְשְׁרָה בְצִדּוֹ / נָפְלָה בַּפַּח אֲשֶׁר יָקְשָׁה / וְהוּשַׁב גְּמוּלָהּ בְּרֹאשָׁהּ.

וְהַמָּשָׁל אֶרְאֶה בְּאִישׁ / אֲשֶׁר יַחְפִּיר וְיַבְאִישׁ / לַחֲרֹשׁ עַל רֵעֵהוּ רָעָה / לַעֲשׂוֹת חֹנֶף לְדַבֵּר תּוֹעָה / וְכָל מְזִמּוֹתָיו לְעָנְשׁוֹ / גְּמוּלוֹ יָשׁוּב בְּרֹאשׁוֹ / כַּאֲשֶׁר זָמַם לַעֲשׂוֹת לוֹ / וְיִמָּלֵט נָקִי שָׁלֵו בְּאָהֳלוֹ / וְאָמַר שְׁלֹמֹה בְּחָכְמָתוֹ / מְרַדֵּף רָעָה לְמוֹתוֹ.

וָאֶשָּׂא מְשָׁלִי וָאֹמַר:


פְּתִיל מִרְמָה אֲשֶׁר חִבַּר / צְפַרְדֵּעַ וְהָעַכְבָּר

הֲלֹא נִתַּק וְשָׁב שָׁלֵם / אֲשֶׁר מִלֵּא אֲסָמָיו בָּר

וְלֹא נֶאְכַל וְלֹא נִקְבָּר / וְלֹא נִשְׁבָּה וְלֹא נִשְׁבָּר

וּמֵתָה הִמְתִיקַתְהוּ / וְלֹא שָׁבָה מְקוֹם מִדְבָּר

וְכֵן יִהְיֶה גְּמוּל מֵסִית / בְּפֶה חָלָק יְהִי נִדְבָּר

בְּלִבּוֹ מִרְמוֹת וָתֹךְ / וְלֹא יוֹעִיל לְסוֹד עָבָר

וְחֹרֵשׁ רָע לְרֵעוֹ מוֹ/קְשׁוֹ יִמְצָא וְיִתְעַבָּר

תְּמִימִים יִשְׁכְּנוּ אָרֶץ / וְדֹרְשֵׁי טוֹב וְנוֹשְׁקֵי בָר.
