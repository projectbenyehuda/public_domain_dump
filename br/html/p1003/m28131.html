הגאון מוילנה

הגאון מוילנה / ישראל שף

ורשה: האור, [1906].



א. הַגָּאוֹן בִּימֵי יַלְדוּתוֹ.


בִּשְׁנַת חֲמֵשֶׁת אֲלָפִים אַרְבַּע מֵאוֹת וּשְׁמוֹנִים

לִבְרִיאַת הָעוֹלָם, בַּיּוֹם הָרִאשׁוֹן לְחַג הַפֶּסַח, בַּיּוֹם

הַשְּׁלִישִׁי לְשָׁבוּעַ, נוֹלַד לְהַרָב הַצַּדִּיק רַבִּי שְׁלֹמֹה

זַלְמָן מִוִּילְנָה בֵן וַיִּקְרָא אֶת שְׁמוֹ אֵלִיָּהוּ.

וַיְהִי הַיֶּלֶד אֵלִיָּהוּ יְפֵה תוֹאַר וִיפֵה מַרְאֶה.

וּבִהְיוֹת הַיֶּלֶד בֶּן שָׁנָה כְבָר רָאוּ אֲבוֹתָיו,

כִּי הַיֶּלֶד אֵלִיָּהוּ יִהְיֶה לְגָדוֹל בְּיִשְׂרָאֵל; חָכְמָתוֹ

הָרַבָּה הֵאִירָה אֶת פָּנָיו וְעֵינָיו הַשְּׁחוֹרוֹת וְהַגְּדוֹלוֹת

הֵאִירוּ כְאוֹר הַשָּׁמֶשׁ.

כַּאֲשֶׁר מָלְאוּ לְהַיָּלֶד שָׁלשׁ שָׁנִים מָסַר אוֹתוֹ

אָבִיו לְאַחַד הַמְּלַמְּדִים לְלַמֵּד אוֹתוֹ תוֹרָה, וּבִהְיוֹתוֹ בֶן

שֵׁשׁ שָׁנִים כְּבָר יָדַע אֶת כָּל הַתּוֹרָה כֻלָּהּ, וְלֹא נִמְצָא

עוֹד בְּכָל הָעִיר אִישׁ אֲשֶׁר יִגְדַל מִמֶּנּוּ בְתוֹרָה, וְלֹא יָסֵף

עוֹד אָבִיו לִשְׁלוֹחַ אוֹתוֹ אֶל הַחֵדֶר, וַיִּלְמַד הַנַּעַר בֶּן

שֵׁשׁ הַשָּׁנִים אֶת הַתּוֹרָה בְעַצְמוֹ. וְכַאֲשֶׁר מָלְאוּ

לְאֵלִיָּהוּ שֵׁשׁ שָׁנִים וָחֵצִי, לִמֵּד אוֹתוֹ אָבִיו רַבִּי שְׁלֹמֹה

דְרָשָׁה, וַיִּדְרוֹשׁ הַנַּעַר אֵלִיָּהוּ לִפְנֵי קָהָל גָּדוֹל

בְּאֶחָד מִבָּתֵּי הַמִּדְרָשׁ הַגְּדוֹלִים אֲשֶׁר בְּוִילְנָה,

וַיִּשְׁתּוֹמְמוּ הַשּׁוֹמְעִים הָרַבִּים בְּשָׁמְעָם אֶת דִּבְרַי

הַדּוֹרֵשׁ הַקָּטָן הַזֶּה, הַמְּבָרֵר וּמְבָאֵר אֶת דִּבְרַי הַתּוֹרָה

אֲשֶׁר לִמַּד אוֹתוֹ אָבִיו, וַיִּתְפַּלְּאוּ עָלָיו מְאֹד.

בֵּין הַשּׁוֹמְעִים הָרַבִּים נִמְצָא גַּם הַגָּאוֹן

הַגָּדוֹל רַבִּי הֶשִׁיל, שֶׁהָיָה בַּיָּמִים הָהֵם הָרַב בְּעִיר

וִילְנָה, וּבְיוֹם הַשַּׁבָּת הַבָּאָה אַחֲרֵי הַסְּעוּדָה

הַשְּׁלִישִׁית שָׁלַח הָרַב לִקְרוֹא אֵלָיו אֶת הַנַּעַר הַקָּטָן

אֵלִיָּהוּ, וּבְבוֹאוֹ לְפָנָיו נִסָּה אוֹתוֹ בִּשְׁאֵלוֹת שׁוֹנוֹת,

וְעַל כֻּלָּן הֵשִׁיב אֵלִיָּהוּ תְּשׁוּבוֹת בְּרוּרוֹת וּנְכוֹנוֹת.

וַיּוֹסֶף הָרַב וְכָל הָעוֹמְדִים שָׁמָּה לְהִתְפַּלֵּא וּלְהִשְׁתּוֹמֵם

עַל הַנַּעַר הַנִפְלָא וְהַנֶּחְמָד הַזֶּה, וַיְחַבְּקוּהוּ

וַיְנַשְּׁקוּהוּ מֵרוֹב חִבָּתָם אֵלָיו וַיֹּאמְרוּ כֻּלָּם פֶּה אֶחָד:

אֵין נַעַר חָכָם וְנָבוֹן כָּמוֹהוּ בְּכָל יִשְׂרָאֵל.


ב. הַגָּאוֹן בִּימֵי נַעֲרוּתוֹ.


וְהַנַּעַר אֵלִיָּהוּ הוֹלֵךְ וְגָדוֹל בַּתּוֹרָה מְאֹד.

בַּיָּמִים הָהֵם וְלַנַּעַר אֵלִיָּהוּ מָלְאוּ שְׁמוֹנֶה שָׁנִים, וַיִּתְאַסְּפוּ הַרְבֵּה לוֹמְדִים מוּפְלָגִים1 לִלְמוֹד יַחַד אֶת דִּבְרֵי “מִשְׁנֵה הַתּוֹרָה” אֲשֶׁר לְרַבֵּנוּ מֹשֶׁה בֶּן מַיְמוּן הַנּוֹדָע בְּשֵׁם “רַמְבַּ”ם",

וַיִּקַּשׁ מֵהֶם לְהָבִין עַל נָכוֹן אֶת דִּבְרַי הָרַמְבַּ"ם

בְּעִנְיָנֵי “קִדּוּשׁ הַחוֹדֶשׁ”, כִּי הַלִּמּוּד הַזֶּה דּוֹרֵשׁ

יְדִיעָה רַבָּה בְּחָכְמַת הַתְּכוּנָה2 וְדַעַת מַהֲלַךְ הַכּוֹכָבִים וְסִבּוּב הָאָרֶץ וְכָל

הַגַּלְגַּלִים. וּלְמַעַן יָקֵל לָהֶם לְהָבִין אֶת הָעִנְיָן הַהוּא

עָשׂוּ לָהֶם הַלּוֹמְדִים כַּדּוּרִים וַיְצַיְּרוּ עֲלֵיהֶם אֶת מַהֲלַךְ

הַגַּלְגַּלִים, בְּאָמְרָם לְהָבִין עַל יְדֵי הַצִּיּוּרִים הָאֵלֶּה אֶת

כַּוֶּנַת דִּבְרֵי הָרַמְבַּ"ם. וּבְיוֹם רֹאשׁ חֹדֶשׁ נֶאֶסְפוּ

הַלּוֹמְדִים הָאֵלֶּה, שֶׁרֻבָּם הָיוּ כְבָר בָּאִים בַּיָּמִים מִבְּנֵי

חֲמִשִּׁים שָׁנָה וָמַעֲלָה, אֶל מָקוֹם אֶחָד וַיָּבִיאוּ אִישׁ אִישׁ

כַּדּוּרוֹ בְיָדוֹ; אוּלָם גַּם בִּהְיוֹתָם כֻּלָּם יַחַד לֹא יָכְלוּ

עוֹד לִמְצוֹא אֶת הַכַּוָּנָה הַנְּכוֹנָה, כִּי הַנַּגָּר שֶׁעָשָׂה

לָהֶם אֶת הַכַּדּוּרִים שָׁגָה בִמְלַאכְתּוֹ וְלֹא עָשָׂה אֶת

הַכַּדּוּרִים, כְּפִי שֶׁהֵם צְרִיכִים לִהְיוֹת בֶּאֱמֶת, וְלָכֵן שָׁגוּ

גַּם הַלּוֹמְדִים הָהֵם וְלֹא הֵבִינוּ גַּם עַתָּה אֶת דִּבְרַי

הָרַמְבַּ"ם וְלֹא יָכְלוּ לִמְצוֹא אֶת הָאֱמֶת וַיִּחִר לָהֶם

מְאֹד.

וְהַנַּעַר אֵלִיָּהוּ, הַנַּעַר בֶּן שְׁמוֹנֶה שָׁנִים,

בְּשׁוּבוֹ מִבֵּית הַמִּדְרָשׁ וּמִתַּלְמוּדוֹ וְלֹא מָצָא עוֹד אֶת

סְעֻדַּת הַצָּהֳרַיִם עוֹמֶדֶת לְפָנָיו עַל

הַשֻּׁלְחָן, הָיָה עוֹסֵק, עַד אֲשֶׁר יַגִּישׁוּ לְפָנָיו לֶאֱכוֹל,

בְּחָכְמַת הַתְּכוּנָה, עַל פִּי סֵפֶר אֶחָד אֲשֶׁר מָצָא בְאוֹצָר

הַסְּפָרִים אֲשֶׁר לְאָבִיו, כִּי לֹא יָכוֹל הַנַּעַר הַמַּתְמִיד הַזֶּה

לָשֶׁבֶת בָּטֵל אַף רֶגַע אֶחָד; וּבְמֶשֶׁךְ יָמִים לֹא רַבִּים גָּדְלָה

חָכְמָתוֹ וִידִיעָתוֹ בְּחָכְמַת הַתְּכוּנָה עַד מְאֹד, וַיֵּדַע אֶת

כָּל דַּרְכֵּי הַכּוֹכָבִים וּמְסִלּוֹתֵיהֶם וְאֶת נְתִיבוֹת כָּל צְבָא

הַשָּׁמַיִם וּמַהֲלָכָם. וַיְהִי כַּאֲשֶׁר נוֹדַע לְהַנַּעֵר אֵלִיָּהוּ,

כִּי מִתְקַשִּׁים הַלּוֹמְדִים בַּהֲבָנַת דִּבְרֵי הָרַמְבַּ"ם,

וַיִּקְרַב אֲלֵיהֶם וַיַּבֵּט עַל הַכַּדּוּרִים אֲשֶׁר בִּידֵיהֶם,

וַיֵּדַע תֵּכֶף, כִּי בְשֶׁל הַשְּׁגִיאָה שֶׁשָׁגָה הַנַּגָּר

בִּמְלַאכְתּוֹ, הַלּוֹמְדִים הָאֵלֶּה מִתְקַשִּׁים. וַיִּשָּׂא אֶת

קוֹלוֹ וַיֹּאמַר:

– שִׁמְעוּ אֵלַי, רַבּוֹתַי, לֹא לְמַעַן הִתְגָּאוֹת

בְּחָכְמָתִי כִּי רַבָּה, אֲנִי אוֹמֵר לָכֶם אֶת דְּבָרַי, כִּי אִם

לְמַעַן חָכְמַת תּוֹרָתֵנוּ הַקְּדוֹשָׁה. רוֹאֶה אֲנִי, כִּי לֹא מֵחֹסֶר

יְדִיעַתְכֶם אֶת דִּבְרֵי הָרַמְבַּ"ם אַתֶּם מִתְקַשִּׁים בָּהֶם, כִּי

אִם מִפְּנֵי שְׁגִיאַת הַנַּגָּר, שֶׁשָׁגָה בִּמְלֶאכֶת הַכַּדּוּרִים

אַתֶּם טוֹעִים; וְעַתָּה הַבִּיטוּ וּרְאוּ אֶת הַכַּדּוּר אֲשֶׁר

בְּיָדִי וּרְאִיתֶם כִּי כָל דִּבְרֵי רַבֵּנוּ משֶׁה בֶּן מַיְמוּן אֱמֶת

הֵם! – וּבְדַבְּרוֹ אֶת הַדְּבָרִים הָאֵלֶּה הוֹצִיא מִכִּיסוֹ כַדוּר

אֶחָד שֶׁעָשָׂה לוֹ בְעַצְמוֹ, וַיְבָאֵר לִפְנֵיהֶם, עַל פִּי כַדּוּרוֹ,

אֶת כָּל דִּבְרֵי הָרַמְבַּ"ם, וַיִּשְׁתּוֹמְמוּ הַמּוּפְלָגִים הָאֵלֶּה

וַיֹּאמְרוּ: לֹא בֶן אָדָם הוּא הַנַּעַר הַזֶּה, כִּי אִם מַלְאַךְ

מִמָּרוֹם שֶׁכָּל דִּבְרֵי הַתּוֹרָה גְּלוּיִים וִידוּעִים

לְפָנָיו.

וְרַבִּי שְׁלֹמֹה זַלְמָן, אֲבִי הַנַּעַר, הָיָה כָל

יָמָיו עָנִי מְדֻכָּא וְלֹא יָכוֹל לְכַלְכֵּל אֶת בְּנֵי בֵיתוֹ

וּלְפַרְנֵס אוֹתָם בְּלֶחֶם. וַיִּקַּח אִישׁ עָשִׁיר אֶחָד מִתּוֹשְׁבֵי

וִילְנָה, וּשְׁמוֹ רַבִּי אַבְרָהָם קַצֶּנֶלְנְבּוֹגֶן, אֶת הַנַּעַר

אֵלִיָּהוּ אֵלָיו וַיְבִיאֵהוּ לָעִיר קֵידָן הַקְּרוֹבָה לְוִילְנָה,

וְיִתְּנֻהוּ עַל יַד אָבִיו הַגָּאוֹן רַבִּי דָוִד הָרַב בְּעִיר הַהִיא,

וַיִּלְמוֹד הַגָּאוֹן שָׁם תּוֹרָה, וּבִהְיוֹתוֹ כְבֶן תֵּשַׁע שָׁנִים

יָדַע כִּמְעַט עַל פֶּה אֶת כָּל סִפְרֵי הַתּוֹרָה, הַמִּשְׁנֶה

וְהַגְּמָרָא, וְעוֹד הַרְבֵּה סְפָרִים אֲחֵרִים.

וַיְהִי הַיּוֹם, יוֹם שְׁמִינִי הָעֲצֶרֶת בָּעֶרֶב,

וַיִּזְכּוֹר הַנַּעַר אֵלִיָּהוּ, וְהוּא אָז בֶּן אַחַת עֶשְׂרֵה שָׁנָה,

כִּי נָדַר נֵדֶר בַּשָּׁנָה הַהִיא לִלְמוֹד אֶת שְׁתֵּי

הַמְּסָכוֹת3 “זֶבָחִים וּמְנָחוֹת”,

הַמְּסָכוֹת הַיּוֹתֵר קָשׁוֹת שֶׁבְּכָל הַתַּלְמוּד, בְּמֶשֶׁךְ

הַשָּׁנָה וּלְסַיֵּם4 אוֹתָם בְּיוֹם שִׂמְחַת

וְכָל חוֹבְבֵי תוֹרָה וְהַצְּמֵאִים לִדְבַר אֲדֹנָי

בָאוּ מִכָּל הֶעָרִים הַקְּרוֹבוֹת וְהָרְחוֹקוֹת אֶל בֵּית מִשְׁכַּן

הַגָּאוֹן לִשְׁמוֹעַ תּוֹרָה מִפִּיו. וְתַלְמִידִים רַבִּים לָמְדוּ

תּוֹרָה מִמֶּנּוּ, וַיִּהְיוּ גַם הֵם לִגְאוֹנִים גְּדוֹלִים מְאוֹרֵי

יִשְׂרָאֵל. וּמֵהֶם הַיּוֹתֵר נִכְבָּדִים הָאַחִים: רַבִּי חַיִּים

וְרַבִּי שְׁלֹמֹה זַלְמָן מֵעִיר וַלּוֹזִ’ין, הוּא הַגָּאוֹן רַבִּי

חַיִּים, אֲשֶׁר יָסַד אֶת הַיְּשִׁיבָה הַנּוֹדָעָה בְּשֵׁם "יְשִׁיבַת

עֵץ חַיִּים" בְּוַלּוֹזִ’ין, אֲשֶׁר כִּשְׁתֵּי מֵאוֹת שָׁנָה הָיְתָה

אַכְסַנְיָה שֶׁל תּוֹרָה, וּמִמֶּנָּה יָצְאוּ גְאוֹנֵי יִשְׂרָאֵל

וַחֲכָמָיו עַד הַיּוֹם הַזֶּה.

וּסְפָרִים רַבִּים כָּתַב הַגָּאוֹן עַל כָּל דִּבְרֵי

הַתּוֹרָה שֶׁבִּכְתָב וְשֶׁבַּעַל פֶּה וַיָּאֶר בִּדְבָרָיו וּבִסְפָרָיו

אֶת עֵינֵי כָּל חַכְמֵי יִשְׂרָאֵל.

פְּעָמִים אֲחָדוֹת חָלָה הַגָּאוֹן בְּמַחֲלוֹת שׁוֹנוֹת,

אַךְ הוּא לֹא דָרַשׁ בְּרוֹפְאִים, כִּי אִם צִוָּה לְהָבִיא לְפָנָיו אֶת

הַמַּגִּיד הַמְּפוּרְסָם רַבִּי יַעֲקֹב מִדֻּבְּנָה, הַנּוֹדָע לְשֵׁם

בִּמְשָׁלָיו הַנְּעִימִים וְהַמְּחֻכָּמִים, וְהוּא בִּדַּח אֶת לֵב הַגָּאוֹן בִּדְבָרָיו וַיָּבֵא בָּזֶה מַרְפֵּא

לְמַחֲלָתוֹ.

וּבִשְׁנַת תקנ"ח בִּמְלֹאוֹת לְהַגָּאוֹן שִׁבְעִים

וּשְׁמוֹנֶה שָׁנָה חָלָה בְּעֶרֶב יוֹם הַכִּפּוּרִים מַחֲלָה קַלָה, אַךְ

לִבּוֹ נִבָּא לוֹ כִּי לֹא יָקוּם הַפַּעַם מֵחָלְיוֹ זֶה. וַיִּקְרָא

לְבָנָיו וְלִבְנוֹתָיו וְלִנְכָדָיו, וַיִּשְׁלַח יָדוֹ וַיָּשֶּׂם עַל

רֹאשׁ בְּנוֹ הַגָּדוֹל רַבִּי אַבְרָהָם וַיְבָרֵךְ אוֹתוֹ וַיִּשָּׂא

קוֹלוֹ וַיֵבְךְּ, וְכָל הָעוֹמְדִים שָׁמָּה רָעֲדוּ וְנִבְהָלוּ.

וּמִמָּחֳרָת יוֹם הַכִּפּוּרִים גָּבְרָה הַמַּחֲלָה וַתֵּלֶךְ הָלֹךְ

וְקָשָׁה, וְשִׂמְחַת חַג הַסֻּכּוֹת נֶהֶפְּכָה לְעִיר וִילְנָה לְאֵבֶל,

כִּי גָוַע וַיֵּאָסֵף אֶל עַמָּיו גְּאוֹן הָעִיר וַהֲדָרָה רַבֵּנוּ

אֵלִיָּהוּ, בִּהְיוֹתוֹ בֶן שִׁבְעִים וּשְׁמוֹנֶה שָׁנָה.

וּבְכָל עָרֵי יִשְׂרָאֵל, בְּהַגִּיעַ אֲלֵיהֶם

הַשְּׁמוּעָה, כִּי עָלָה אֵלִיָּהוּ הַשָּׁמָיְמָה, הִסְפִּידוּ אוֹתוֹ

וַיְבַכּוּ כָל יִשְׂרָאֵל אֶת נִשְׁמָתוֹ, אֲשֶׁר עָלְתָה בִקְדוּשָׁה

וּבְטָהֳרָה אֶל הָאֱלֹהִים.
