ויהיה מלך אחד אוהב חכמה

וַיְהִי מֶלֶךְ אֶחָד אוֹהֵב חָכְמָה, וְהָיָה מִתְנַכֵּר וְלוֹבֵשׁ בְּגָדִים מִבְּגָדִים שׁוֹנִים / וְהוֹלֵךְ אֶל מְקוֹם הִתְקַבֵּץ הַחֲכָמִים / וְיוֹשֵׁב בְּמוֹשָׁב שָׁפָל מִכֻּלָּם וְאֵין מַכִּיר לוֹ וְלֹא יוֹדֵעַ / וַיָּבֹא לַיְלָה אֶחָד מִתְנַכֵּר כְּדָתוֹ, וַיִּמְצָא בְּבֵית תְּפִלָּתָם שְׁלֹשָׁה אֲנָשִׁים / יוֹשְׁבִים מִתְלַחֲשִׁים / וַיִּתֵּן לָהֶם שָׁלוֹם וַיֵּשֵׁב עִמָּהֶם / וַיֹּאמְרוּ לוֹ: ׳מִי אַתָּה וּמֵאֵיפֹה / וּמַה לְּךָ פֹה?׳ אָמַר לָהֶם: ׳גֵּר אָנֹכִי מֵאֶרֶץ רְחוֹקָה / וָאִיקַץ בְּנֶפֶשׁ נֶעֱצֶבֶת דְּחוּקָה / וָאָבֹא אוּלַי אֲדַבְּרָה וְיִרְוַח לִי / וְאֶנְשֶׁה מְעַט מֵעֲמָלִי׳. / וַיָּבֹא בִּכְלַל דִּבְרֵיהֶם / וַיֵּשֶׁב כְּאֶחָד מֵהֶם. וַיְדַבְּרוּ הַשְּׁלֹשָׁה בַמֶּלֶךְ וַיְסַפְּרוּ כִי הוּא עַוָּל וְחוֹמֵס / וְכִי הָעֹשֶׁר לְנַפְשׁוֹ וְלִמְשָׁרְתָיו בִּלְבָד כּוֹמֵס / וְלֹא יַעַבְרוּ לוֹ הַחֲסָדִים עַל הַנֶּפֶשׁ / וְלֹא יִמְצָא מִצְטָרֵךְ נֶגְדּוֹ נוֹפֶשׁ / וַיֹּאמְרוּ אִישׁ אֶל אָחִיו: ׳נִקְרָא עַל הָאַכְזָר / כִּי אֵין אִתָּנוּ זָר / אוּלַי יִכָּרֵת / וְיֹאבַד בְּאָבְדוֹ עוֹבֵד וּמְשָׁרֵת׳ / וַיֵּאֹתוּ הָאֲנָשִׁים אֶל צַעֲקַת אֶחָד מֵהֶם וְהַשְּׁלֹשָׁה יַעֲנוּ אָמֵן.


וַיָּקָם הַגָּדוֹל שֶׁבָּהֶם וַיִּפְרֹשׂ אֶת כַּפָּיו וַיִּקְרָא וַיֹּאמַר: ׳אָנָּא הַמְרַחֵם, רְאֵה כִי נִמְלְאָה הָאָרֶץ חָמָס בְּפֶשַׁע מַלְכָּהּ עוֹשֵׁק וְהָרֵעַ / כִּי קָצַר הַמַּצָּע מֵהִשְׂתָּרֵעַ / מַהֵר הֲסִירֵהוּ / אוֹ תְמִיתֵהוּ׳ / וַיַּעֲנוּ הַשְּׁנַיִם: ׳אָמֵן!׳ מִתְאַוִּים שׁוֹמְעִים / וַיִּדֹּם הַמֶּלֶךְ וּפָנָיו זֹעֲפִים / וַיֹּאמְרוּ לוֹ: ׳לָמָּה שָׁתַקְתָּ / וּמַה לְּךָ כִּי נִזְעַקְתָּ?׳ / אָמַר לָהֶם: ׳הוֹסִיפוּ בַקְּרִיאָה שְׁלֹשָׁה עִנְיָנִים / וְאֶעֱנֶה אָמֵן כַּאֲשֶׁר אַתֶּם עוֹנִים׳ / אָמְרוּ לוֹ: ׳וּמַה הֵם?׳ וַיֹּאמֶר לָהֶם, וְאֵשׁ הַחֵמָה בּוֹעֶרֶת בְּאַפּוֹ / וּמַרְאֵהוּ כְּמַרְאֵה אַרְיֵה רָעַב עַל טַרְפּוֹ: / כָּכָה יֹאמַר הַקּוֹרֵא ׳הֲסִירֵהוּ בְמַזָּל טוֹב עַל הָעָם אִם תְּסִירֵהוּ / אוֹ תְמִירֵהוּ בְטוֹב מִמֶּנּוּ אִם תְּמִירֵהוּ / אוֹ תַחְזִירֵהוּ בִתְשׁוּבָה שְׁלֵמָה עַד שֶׁיִּרְצֵהוּ / אֲשֶׁר יְנַאֲצֵהוּ׳.


וַיַּכִּירוּ הָאֲנָשִׁים בְּהִתְמַרְמְרוֹ כִּי הוּא הַמֶּלֶךְ, וַיִּתְמְהוּ הָאֲנָשִׁים אִישׁ אֶל אָחִיו, וַיִּפְלוּ עַל פְּנֵיהֶם וַיֹּאמְרוּ: ׳חָטָאנוּ בַה׳ וָבָךְ, הִנֶּנּוּ לְךָ לַעֲבָדִים׳ / וַיֹּאמֶר לָהֶם: ׳שָׁלוֹם לָכֶם אַל תִּירְאוּ מִמֶּנִּי, אֲבָל אֲשֵׁמִים אַתֶּם לַה׳ כִּי עֲזַבְתֶּם מִצְוַת הַתּוֹכַחַת / וַתְּמַהֲרוּ לִקְרֹא בְרִדְתִּי אֶל שְׁאוֹל שַׁחַת׳ / וַיִּקְרָא הַמֶּלֶךְ אֶחָד מִסָּרִיסָיו, וַיְצַוֵּהוּ לְהָבִיא שָׁם שְׁתֵּי מֵאוֹת זְהוּבִים, וַיִּתְנֵם לִשְׁלֹשָׁה הָאֲנָשִׁים הַקּוֹרְאִים עָלָיו, וַיֹּאמֶר אֲלֵיהֶם: ׳אַל תּוֹסִיפוּ לַחֲטֹא! אַשְּׁרוּ הֶחָמוֹץ וְהוֹכִיחוּ הַתּוֹעֶה וְלַמְּדוּ הַפֶּתִי וְהַגְדִּילוּ עַל הָרָשָׁע אוּלַי יָשׁוּב בְּעוֹדוֹ. / לִפְנֵי עָמְדוֹ / לָתֵת חֶשְׁבּוֹן לִפְנֵי מֶלֶךְ מַלְכֵי הַמְּלָכִים הַקָּדוֹשׁ בָּרוּךְ הוּא הָאֶחָד / אֲשֶׁר לֹא יִשָּׂא פָנִים וְלֹא יִקַּח שֹׁחַד. – – –
