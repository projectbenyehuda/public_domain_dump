נְזִיר אַחִים וּמִבְחַר הַחֲבֵרִים

נְזִיר אַחִים וּמִבְחַר הַחֲבֵרִים / וְאִישׁ מַשְׂכִּיל וְגִבּוֹר בַּגְּבָרִים

קְחָה מִלִּין אֲשֶׁר אָרִיד בְּשִׂיחִי / וְאָהִימָה בְרָאשֵׁי הַדְּבָרִים

וְהַקְשֵׁב אָזְנְךָ הֵיטֵב לְמַעַן / בְּדָבְרֶךָ תְּהִי מַשְׂכִּיל וּמַעֲרִים

וְתֵדַע אֵיךְ תְּנַהֵג הַכְּפִירִים / וְאֵיךְ תִּרְעֶה וְתַרְבִּיץ הָעֲדָרִים

וְאֵיךְ תָּשִׁיב לְדִבְרֵי הַכְּסִילִים / וְאֵיךְ תַּעַן לְמִלַּי הַיְּקָרִים

וְדַע כִּי בָז לְרֵעֵהוּ חֲסַר לֵב / וְתוֹכַחְתּוֹ חֲדָשִׁים לַבְּקָרִים

וְאֵיךְ תִּבְזֶה דְּבַר הַשֵּׁם וְתָפֵר / לְמִצְווֹתָיו כְּכוֹכְבֵי רוּם מְאִירִים

וְקָפַצְתָּ וְאִמַּצְתָּ לְבָבָךְ / בְּאִמְרָתָךְ הֲלֹא הֵמָּה שְׁקָרִים

וְאִם אָנִי אֲסִיר הַגּוּף בְּסֹהַר / הֲלֹא לִבִּי כְנוֹדֵד בַּיְּעָרִים

וְלֹא תָבִין לְדִבְרֵי הַשְּׂרִידִים / אֲשֶׁר כָּתְבוּ בְטוּרֵיהֶם סְדָרִים

לְכָל תִּינוֹק אֲשֶׁר חִדַּשׁ בְּמִלָּה / הֲלֹא הַכֹּל בְּסִינַי נֶאֱמָרִים

וְכָל שֶׁכֵּן לְדִבְרֵי בִין בְּכָל סוֹד / לְקַדְמוֹנִים בְּפִי הַכֹּל הֲדוּרִים

וְהִסְמַכְתִּים בְּפָסוּקִים בְּלִקְחִי / בְּסַרְעַפַּי בְּחַדְרֵי הַחֲדָרִים

וְלֹא אֶכְתֹּב לְחִדּוּשַׁי לְבַד כִּי / צְרַפְתִּימוֹ בְשִׂכְלִי נֶחֱקָרִים

וְשָׁנִיתִי וְשִׁלַּשְׁתִּי לְחָקְרָם / עֲדֵי כִי אֶמְצְאָה אוֹתָם גְּמוּרִים

וְאִם לֹא אֶשְׁמְרָה מַחְסוֹם לְמוֹ פִי / וְהִנַּחְתִּי דְבָרִים הַסְּגוּרִים

וְזֹאת הִיא לִי יְרוּשָׁה מֵאֲבוֹתַי / וְאִם אָנִי וְרַעְיוֹנַי אֲסוּרִים

וְאָנֹכִי אֲנִי הַדַּל בְּאַלְפִּי / וְאַל תַּחְשֹׁב לְמִשְׁפַּחְתִּי צְעִירִים

שְׁאַל אָב לָךְ וְיַגֵּדְךָ שְׁבָחָם / וְאִם עַתָּה בְּקִבְרֵיהֶם קְבוּרִים

עֲדֶנָּה הֵם בְּצִדְקָתָם לְחַיִּים / וְלֹא מֵתוּ וְלִבְרָכָה זְכוּרִים

וְעַתָּה הֵן אֲנִי נִשְׁאָר לְבַדִּי / לְהַגִּיד לָךְ לְשָׁרְשֵׁי הַגְּבָרִים

בְּזִיתַנִי וּבָזִיתָ דְּבָרַי / וְהִסְכַּלְתָּ עֲשׂוֹ כַיּוֹם לְדוֹרִים

בְּטֶרֶם תֵּדְעָה אֵיכָה חֲקוּקִים / וְאֵיפֹה הֵם כְּפִקְחִים אוֹ כְעִוְרִים

וּפָרַצְתָּ גְּבוּל אַחְוָה לְעוֹלָם / וְאֵיךְ תִּבְנֶה לְבָתִים הַסְּתוּרִים

חֲקֹר וּדְרֹשׁ בְּבָתֵּי הַתְּבוּנוֹת / וְתִרְאֶה מַעְגְּלֵי צֶדֶק יְשָׁרִים

הֲלֹא תִטְעַם דְּבַשׁ יַעְרָם וְתִרְאֶה / הֲטוֹבִים הֵם הֲמָרִים אוֹ תְּמָרִים

וְרַעְיוֹנַי בְּךָ אֶתְמוֹל חֲזָקִים / וְהַיּוֹם נֶהְפְּכוּ מִכָּל עֲבָרִים

בְּכָל מָקוֹם תְּהַפֵּךְ אֶת דְּבָרַי / וְכַאֲשֶׁר תַּעֲשֶׂינָה הַדְּבוֹרִים

וְאֵי חָכְמָה וּמוּסָר וַעֲנָוָה / וְאֵי שֵׂכֶל וְאֵי יִרְאָה לְמוֹרִים

וְהָיִיתִי מְצַפֶּה לְךָ לְהָקִים / לְבִרְכֵּי כּוֹשְׁלִים אוֹ נֶעֱצָרִים

וְתֹאמַר מִשְּׁמִי כָל מַעֲנֶה טוֹב / וְיִהְיֶה לִי בְּךָ חַיֵּי בְשָׂרִים

וְאָכֵן דַּעְתְּךָ נִגְלֵית לְלִבִּי / הֲכִי תַעֲקֹר לְשִׂיחַי הַצְּבוּרִים

וְאֵיךְ תִּגְעַר בְּרָב מִמָּךְ בְּשָׁנִים / וְגַעֲרָתָךְ לְזֵדִים הָאֲרוּרִים

וְתִשְׁתֶּה מֵי בְאֵר מַיִם צְלוּלִים / וְתַשְׁלִיךְ בָּהּ אֲבָנִים וַעֲפָרִים

לְזֹאת יִתַּר לְבָבִי מִמְּקוֹמוֹ / וְלֹא יִשְׂמַח וְלֹא יֶהְגֶּה בְשִׁירִים

וְעַל כֵּן אֵשְׁבָה בָדָד וְאֶדֹּם / וְיִבָּקְעוּ דְמָעַי כַיְּאוֹרִים

וְשׂוֹנְאַי יִלְבְּשׁוּ בֹשֶׁת לְמַדָּם / יְדִידַי יִנְחֲלוּ כָבוֹד לְהָרִים

וְלִפְנֵי אֵל אֲנִי אֶכְרַע וְאוֹדֶה / שְׁהוּא עוֹנֶה וְהוּא תוֹמֵךְ אֲשׁוּרִים

וְיַבִּיט מַחְשְׁבוֹת לִבִּי בְּעֵינָיו / וְיִפְדֵּנִי וְיִקְרָא לִי דְרוֹרִים

וְהוּא תוֹלֶה אֲדָמָה עַל בְּלִימָה / אֲיַחֵל לוֹ מְחַיֶּה הַפְּגָרִים.
