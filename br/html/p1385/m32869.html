ויכוח העט והחרב. החרב נוחלת מפלה ושומעת דברי מוסר ופילוסופיה מפי העט.

נְאֻם לְמוּאֵל בֶּן־אִיתִיאֵל. / וַיְהִי דוֹר הַסִּכְלוּת גָּבַר וְרַבָּה / וְהַשִּׁיר אָבַד וְנֶחְבָּא / וּמֵת אוֹ נִשְׁבָּר אוֹ נִשְׁבָּה. / וְהִרְבָּה תֵּבֵל לְהוֹלִיד כָּל בֵּן מְנַבֵּל / בַּעֲבוּר טֻמְאָה לְחַבֵּל. / וְאֵין יִתְרוֹן לְבַעַל הַלָּשׁוֹן / כַּמִּשְׁפָּט הָרִאשׁוֹן. / וְהֵסִיר הָמָן מַלְבּוּשָׁיו / הַחֲמוּדִים וְהַיְקָרִים / וּפָשַׁט אֶת בְּגָדָיו / וְלָבַשׁ בְּגָדִים אֲחֵרִים.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

פְּתַיּוּת פָּשְׁתָה בַּחוּץ / כְּנֶגַע פָּשְׁתָה עֲקֵבָהּ

וְזָנְתָה אֶת בְּנוֹת סִכְלוּת / וְעָגְבָה עַל חֲסַר לִבָּהּ

וְסָכָל זָע וְקִנֵּא בִי / לְהָכִיל הוֹנוֹת הִרְבָּה

וְדָקַר בַּחֲנִית בִּינָה / קֳבָתָהּ בַּחֶדֶר קִבָּה

אֲזַי עָלְתָה חֲמַת פֶּתִי / וּצְבָאוֹת בֶּעָרִים רַבָּה

וְרָדַף אַחֲרֵי שֵׂכֶל / וְהִרְדִּיפוֹ עֲדֵי נֶחְבָּא

וְלֹא נוֹדַע הֲמֵת אָמְנָם / נִשְׁבַּר וְאִם נִשְׁבָּה

וְיֹאמַר אֶל נְבִיא שֵׂכֶל / חֲדָל לְךָ מַה לּוֹ נִבָּא

עֲדֵי מִבִּין וּמִשֵּׂכֶל / אֲדָמָה נִשְׁאֲרָה חֲרֵבָה.


וְהַשְּׂכָלִים כָּל בֶּגֶד יִפְרְמוּ / וְכָל מַעְיַן מַיִם יִסְתְּמוּ. / וַיְהִי כֵן הַשֵּׂכֶל מִפָּנֶיהָ נָחַת / מַעְיָן נִרְפַּשׁ וּמָקוֹר מָשְׁחָת.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

רַבּוּ גָדְלוּ בֹּעֲרִים בָּעָם / וַיִּגְבְּרוּ בָּם סֹבְאֵי יָיִן

וַיִּסְתְּמוּ כָל מַעְיְנֵי בִינָה / מִלְּאָם וְלֹא נוֹדַע מְקוֹם עַיִן

עַד כִּי פְתַיּוּת בָּרְחוֹב תִּקָּרֵא / נוֹתַר חֲכַם־לֵב אָמְרוּ אַיִן

דֹּמִי תְבוּנָה כִּי יְשׁוּעָתֵךְ / תִּשְׁמַח נְאֻם גֶּבֶר שְׁתֻם עָיִן.


וַיְהִי בִּשְׁנַת שָׁלֹשׁ וְתִשְׁעִים / בְהָתֵם פֹּשְׁעִים / וְרוּחַ לָבְשָׁה אֶת הַזְּמָן וְאֶת יְלָדָיו / וְקָדַשׁ הוּא וּבְגָדָיו / וַיִּמְלֹךְ מֶלֶךְ וּשְׁמוֹ מֵלִיץ־הַחוֹזֶה. / וַיְהִי בְּעֵינֵי הַנְּבָלִים נִבְזֶה. / וּבְנֵי בְּלִיַּעַל אָמְרוּ: מַה יּוֹשִׁיעֵנוּ זֶה! וַתִּצְלַח עָלָיו רוּחַ צַח שְׁפָאִים / וַיָּקָם וַיֹּאבַד כָּל זֶרַע הַפְּתָאִים / וַיְהִי לַחָכְמָה לְמוֹשִׁיעַ / וּבְכָל אֲשֶׁר יִפְנֶה יַרְשִׁיעַ / וַיָּשָׁב הַנִּסְחַף סוֹחֵף / וְהַנִּדְחַף דּוֹחֵף / וְהָעָם הַנָּס אֶל הַמִּדְבָּר נֶהְפַּךְ אֶל־הָרוֹדֵף.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

פְּתַיּוּת הֶאֱבִיד מֵלִיץ / לְפָנָיו כָּל כְּסִיל נִבְעַת

וְלִבּוֹת בֹּעֲרִים כָּל יוֹם / יְמִין מֵלִיץ מְשַׁסַּעַת

עֲדֵי תָמוּ מְתֵי סִכְלוּת / וְנָדַמּוּ מִבְּלִי דָּעַת.


וַיִּבֶן לַתְּבוּנוֹת / עָרֵי מִסְכְּנוֹת / וַיְחַדֵּשׁ הֵיכְלֵי הַחָכְמָה וְאֶת מִגְרָשֶׁיהָ / בְּנֵי הָאֱמוּנָה הֵקִים וַאֲנָשֶׁיהָ / וַיִּבֶן אֶת סִדְרֵי הַתְּבוּנָה וְאֶת מִדְרָשֶׁיהָ / וַיֵּצְאוּ מִנֶּחְבְּאֵי הַחָכְמָה וְאֶת נִגְרְשָׁם / מִן הַחוֹרִים אֲשֶׁר נִתְחַבְּאוּ שָׁם. / וַיֵּשֵׁב הַשֵּׂכֶל עַל כִּסְאוֹ וְעַל כַּנּוֹ. / וַיִּטַּע אָהֳלֵי אַפַּדְנוֹ / וַיְהִי כַאֲשֶׁר הֵכִינוּ הַשֵּׂכֶל וְהַחָכְמָה לִבּוֹתָם / וַיִּפְתְּחוּ לַצְּמֵאִים מַעְיָנֹתָם / וַיֵּצְאוּ מַיִם חַיִּים מִמְּקוֹרָם / וַתֵּשְׁתְּ הָעֵדָה וּבְעִירָם / כִּי הַפְּתָאִים / גַּם הֵם יַשְׁקוּ לַצְּבָאִים / יִשְׁבְּרוּ פְרָאִים צְמָאָם.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

מְלִיצִים פָּצְחוּ רִינָה לְעֵינֹת / תְּבוּנָה נִפְתְּחוּ לִשְׁתּוֹת צְמֵאִים

עֲדֵי כִי בוֹעֲרִים שָׁתוּ וְלָעוּ / וְהָיוּ תַאֲווֹת לִבָּם מְבִיאִים

וְעַל כֵּן הָיְתָה לָנוּ לְמָשָׁל / הֲגַם חַבְרֵי פְתָאִים בַּנְּבִיאִים.


וַתִּשְׁמַע נַפְשִׁי לַשֵּׂכֶל תְּבוּנָתוֹ / וַתֵּצֵא לְנַסּוֹתוֹ / לִרְאוֹת הַחָכְמָה וּכְבוֹד עָשְׁרָהּ / לְעָבְדָהּ וּלְשָׁמְרָהּ. / וְהִנֵּה אִשָּׁה מֵרֹאשׁ אֲמָנָה תָּשׁוּר / עַל הָעַיִן בְּדֶרֶךְ שׁוּר / וַתָּרָץ נַפְשִׁי לִקְרָאתָהּ, כִּי לְמִיְּמֵי שִׂכְלָהּ תִּכְסֹף וְתִתְאַב / וַתֵּרֵד הָעַיְנָה וַתִּשְׁאָב. / וַיִּתְעוֹרֵר גַּם לְבָבִי וַיֵּלֵךְ בְחוּצוֹת הַתְּבוּנָה וְהָרְחוֹבוֹת, / לְחַזֵּק בֶּדֶק הֲרִסּוֹתֶיהָ הַחֲרֵבוֹת. / לְעֵת עֶרֶב לְעֵת צֵאת הַשֹּׁאֲבֹת / כְּחֹק כָּל לֵב חָכָם אֲשֶׁר יִמְצָא כָּל דָּבָר נֶעְלָם וְנִפְלָא / בְּנֶשֶׁף בְּעָרָב יוֹם בְּאִישׁוֹן לַיְלָה וַאֲפֵלָה / הֲלֹא זֶה הַלְּבָבוֹת, יִלְטֹשׁ וְסִיגֵיהֶם יְרֻטְּשׁוּ. / וַיְּעוֹפְפוּ מַחְשְׁבוֹתָם כְּמוֹ חֶרֶב / וְחַדּוּ מִזּאֵבֵי

עֶרֶב.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

לְעֵת עֶרֶב לְעֵת צֵאת שׁוֹאֲבֵי מֵי / תְבוּנָה מִמְּקוֹר הַמַּחֲשָׁבוֹת

בְּהִתְאַחֵד לְבָבוֹת בַּמְּלִיצָה / וְחַדּוּ מִזְּאֵבֵי הַעֲרָבוֹת

בְּהִגָּלוֹת פְּלִיאוֹת הַמְּזִמָּה / אֲלֵיהֶם אַךְ בְּעֵין לֵב לֹא בְּבָבוֹת

וְלִבּוֹת בֹּעֲרִים הוֹזִים וְשׁוֹכְבִים / בְּשׁוּט לִבּוֹת בְּנֵי הַתַּעֲרֻבוֹת

יְשׁוּטוּן לַחֲזוֹת נֹעַם קְדוֹשִׁים / וְלִדְרֹשׁ לַאֲשֶׁר דּוֹרֵשׁ לְבָבוֹת

וְאִם תִּרְאֶה סְכָלִים יִדְרְשׁוּ אֵל / תְּדַמֶּה יִדְרֹשׁוּ כִּי אִם בָּאוֹבוֹת.


וְעַד הַיָּמִים הָהֵם חָכְמָה לֹא נִבְעֲלָה / וְעִם אִישׁ לֹא נִתְפְּשָׂה / בְּתוּלָה אֲשֶׁר לֹא אֹרְשָׂה וּלְבָבִי יִשְׁתָּאֶה וְיִתְמַהּ / וְהִנֵּה חָכְמָה יוֹצֵאת וְכַדָּהּ עַל שִׁכְמָהּ / וַיָּרָץ אַחֲרֶיהָ כַּאֲשֶׁר הַמַּחְשָׁב מְצָאָהּ / וּלְחַדְרֵי שִׂכְלוֹ הֱבִיאָהּ / וַיַחֲזֶק בָּהּ בְּצֵאתָהּ מִמְּקוֹמָהּ / וּתְפָשָׂהּ וְשָׁכַב עִמָּה.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

לִבִּי שָׁכַב אֶת חָכְמָה כִּי / מָצָא אוֹתָהּ בִּשְׂדֵה מַחְשָׁב

צָעֲקָה וְאֵין מוֹשִׁיעַ לָהּ מֶ/נְהוּ וְגַם אֵין אֹמֶר הַשָּׁב

אָמְרוּ זֹנָה אָמַר אָמָה אֶתְ/כַּנֶּה לִי גַּם כַּסְפָּהּ מוּשָׁב

אָז עִנְּתָה אֶת עַמִּי אֶזְרָח / אַךְ זוּלָתְךָ שָׂכִיר תּוֹשָׁב

חָכָם יִבְעַל בַּעֲלַת חֵן כִּי / עַל כִּסֵּא חֵן לִבּוֹ יָשַׁב

וּכְסִיל יִבְעַל בַּת־אֵל נֵכָר / כִּי לֵב סָכָל בְמַדְמֵנָה שָׁב.


וַתֵּלֵךְ הַחָכְמָה לְבֵית מִדְרָשָׁהּ / וַיֵּלֵךְ עִמָּהּ אִישָׁהּ / וּבְלֶכֶת לְבָבִי עִם הַחָכְמָה רַעְיָתוֹ / וְהוּא כְחָתָן יֹצֵא מְחֻפָּתוֹ / מִתְהַלֵּךְ בֵּין גַּנּוֹתֶיהָ וּפַרְדְסֶיהָ / וְשׁושַׁנֶּיהָ וַהֲדָסֶיהָ / אוֹרָהּ וּמוֹרָהּ, וּמֵרִיחַ טוּבֵי רֵיחֶיהָ / וּמִשְׁתַּעְשֵׁעַ בְּרִימוֹנֶיהָ וּבְתַפּוּחֶיהָ / וְכַפְתּוֹרֶיהָ וּפְרָחֶיהָ. / וַיִּשָּׂא עֵינָיו בֵּין הַהֲדַסִּים וְהַצִּיצִים / וְהִנֵּה שְׁנֵי אֲנָשִׁים עִבְרִים נִצִּים. / וְהָאֶחָד גִּבּוֹר חַיִל. פְּנֵי אַרְיֵה פָּנָיו. / חֲגוֹר חֶרֶב מְצֻמֶּדֶת עַל־מָתְנָיו / יָרִיעַ אַף יַצְרִיחַ / וְכָל רוֹאָיו מִמֶּנּוּ יַבְרִיחַ / וּשְׁאוֹנוֹ כִּשְׁאוֹן מֵימֵי תְהוֹם / יִשְׁאַָג כַּכְּפִירִים וְיִנְהֹם.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

הֲיֵשׁ גִּבּוֹר וּבָחוּר יַעֲרֹךְ לִי / וּמִי הוּא [זֶה] אֲשֶׁר יַעֲמֹד לְפָנַי

הֲלֹא יִבָּהֲלוּן שְׁנֵי מְלָכִים / וְחֵילֵיהֶם בְּצֵאתָם מִנְּדָנַי.


וּבְיָד הָאֶחָד קָנֶה / מִקּוֹלוֹ לֹא יֵחַת וּמֵהֲמוֹנוֹ לֹא יֵעָנֶה.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

אֱוִיל שִׁית יַד לְפֶה גַּם הַחֲרֵשׁ כִּי / אֲמַתֵּתְךָ בְּפִי עֵטִי וְקָנַי

וּמִקֶּדֶם הֲלֹא הֻגַּד לְךָ כִּי / בְּחֶרֶב לֹא יְהוֹשִׁיעַ.


וַיִּגַּשׁ בַּעַל הַחֶרֶב אֶל בַּעַל הָעֵטִים / וַיְדַבֵּר אִתּוֹ מִשְׁפָּטִים / וַיֹּאמַר: מֶה הָעֵט אֲשֶׁר יִתְפָּאֵר בֵּין הַקָּנִים / וְהִנֵּה עָלָה כֻּלּוֹ כִמְשׁוֹנִים / וְהֲלֹא נוֹלָד וְנִמְצָא בְּסֵתֶר קָנֶה וּבִצָּה. וַיִּבְאַשׁ רֵיחוֹ וּמֵחוֹ / וַיָּנֹס לֵחֹה / בְּיוֹם הִלָּקְחוֹ / וַיָּשוּב יְרַקְרַק וְנָקָל / וְהוּא לֹא פָּנִים קִלְקַל / נְפוּחוֹת גּוּפוֹת כַּנֹּאדִים / שְׁדוּפוֹת קָדִים / וְלוּלֵי הַחֲרָבוֹת / לֹא הָיוּ לַקָּנִים לְבָבוֹת / כִּי יָבוֹאוּ מִקְצָתָן / וּבָרָא אוֹתָם בְּחַרְבוֹתָם / וַיִּפְתְּחוּ פִיהֶם לָשׂוּחַ / וַתַּנַּח עֲלֵיהֶם הָרוּחַ / לִכְתֹּב כְּהַיּוֹם עֲווֹנוֹת וְשִׁגְיוֹנוֹת / עַל סֵפֶר הַזִּכְרוֹנוֹת וּמִכְתְּבֵי עָמָל כָּתְבוּ / וְעַל סִפְרְךָ כֻּלָּם יִכָּתֵבוּ / וְאֵיךְ אֶל הַחֶרֶב יֹאמַר סָכָל / וְהוּא אֶל הֶהָרִים אֻכַּל / וַאֲנִי בְּרָאתִיו אַחַר הֱיוֹת נִקְלֶה וְנִבְזֶה / לָקוּם עָלַי לְאֹרֵב כְּהַיּוֹם הַזֶּה.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

הֲלַחֶרֶב כְּסִיל תֹּאמַר / וּמִי פֶתִי כְעֵט סָכָל

הֲלֹא הָעֵט וְהַקָּנֶה / לְחֶרֶב נָתְנוּ מַאֲכָל

הַקַּשׁ הַיָּבֵשׁ וּמוֹץ נִדָּף / לְחֶרֶב מֹרָטָּה יֹאכַל

וְאֵיךְ בָּעֵט יְכֹלֶת כִּי / לְהָרִים בָּדָיו יוּכַל.


וַיַּעַן הָעֵט וַיֹּאמַר: בְּדַעַת וּמְלִיצָה / מַה זֶּה מַחְשִׁיךְ עֵצָה / וּמִי יְקַלֵּל וְיָעֹז פָּנָיו / דּוֹר חֲרָבוֹת שִׁנָּיו? וְאֵין דֵּעָה וְאֵין חָכְמָה / פְּתַיּוּת בַּל יָדְעָה מָה! / וְהֵם אִם יִקְרְאוּ הָעֵטִים, קָנִים אַחַר אֲשֶׁר יָבֹאוּ / עַל שֵׁם אֲחִיהֶם קֹרָאוּ / וּבְסִכְלוּתְךָ חָשַׁבְתָּ נָבוּב כַּמְּשׁוֹבֵב / וְאִישׁ נָבוּב יִלָּבֵב. / וַתָּקָם עַל הַכֵּלִים / כַּאֲשֶׁר הֵם נִמּוֹלִים / הֲלֹא נֶחְמָד סִפְרוֹ וּכְתִיבָתוֹ / בְּהִמּוֹלוֹ בְּשַׂר עָרְלָתוֹ. / וַתֶּחֱרַף שָׁרְשִׁי אֲשֶׁר עַל יְאֹרִים יַצְמִיחַ / מֵרֵיחַ מַיִם יַפְרִיחַ / וְאִם הִגַּדְתִּי לָךְ מוֹלַדְתֵּךְ. / אֶפֶס כִּי לֹא תִהְיֶה תִּפְאַרְתֵּךְ. / זְכֹר מוֹלַדְתֵּךְ וְאַל תִּשְׁכַּח: / בַּרְזֶל מֵעָפָר יֻקַּח.


וָאֶשָּׂא מְשָׁלִי וָאֹמַר:

רְדִי וּשְׁבִי אֱלֵי עָפָר וְאַל תֵּ/לְכָה רוּמָה וְאֵיךְ חֶרֶב תְּרוֹמָם

וְהִיא עָפָר אֲבָל נִתַּךְ עֲפָרָהּ / כְּחֹם הַשֶּׁמֶשׁ עַל עָפָר תְּחֻמָּם.


וַיְהִי כִשְׁמֹעַ הַחֶרֶב וַתָּקָם בַּחֲרוֹנָהּ / וַתֵּצֵא מִנְּדָנָהּ וַתֹּאמַר: / אוֹי לָעֵטִים לְמַעַנִי בָּהֶם. / בְּעוֹפְפִי חֶרֶב עַל פְּנֵיהֶם. / הִכָּבֵד וְשֵׁב בְּבֵיתֶךָ / פֶּן בַּחֶרֶב אֵצֵא לִקְרָאתֶךָ / וְאַל תִּתְהַלֵּל בְּהַשְׂכֵּל וְדֵעָה. / וְלָמָּה תִתְגָרֶה בְּרָעָה / וְאִם אֶפְתַּח פֶּה אַחֲרִישׁ בַּדֶּיךָ / וְחִין עֶרְכְּךָ וּלְשׁוֹנְךָ אַדְבִּיק אֶל חִכְּךָ / הֱיוֹתָהּ אֵלַי דִמִּיתָ הֱיוֹת אֶהְיֶה כָמוֹךָ הִנֵּה הַחֶרֶב אַדֶּרֶת מְלָכִים / וּמְשָׁרֵת נְסִיכִים. / וְעֹזֶרֶת נְבוּכִים / וְנֶפֶשׁ מַחֲזִיקָה תְּחַיֶּה / וְעֵזֶר מִצָּרָיו תִּהְיֶה / וְכָל אִישׁ אֲשֶׁר יַחְגִּירֵנִי לְצִדּוֹ. / אֶעֱשֶׂה לוֹ עִֵזֶר כְּנֶגְדּוֹ / וְאוֹיְבַי מַחֲזִיקִים, לוּ יִשְׁמְעוּ, / וְאוֹתִי יַעְזֹרוּ / וְאִם לֹא יִשְׁמְעוּ / בַּשֶּׁלַח יַעַבֹרוּ.


וַתִשָּׂא מְשָׁלָהּ וַתֹּאמַר:

אֲנִי אֶצְלַח לְכָל דִּבְרֵי גְבוּרוֹת / וּמִי יַעְמֹד פְּנֵי פָנַי וְחָיָה

אֲנִי יֶשַׁע בְּיַד רָאשִׁים וְרָשִׁים / וְשֻׁלַּחְתִּי לְנֶחְשָׁלִים לְמִחְיָה

אֲנִי יוֹמָם לְצַד אִישִׁים וְלַיִל / לְרָאשֵׁיהֶם אֲנִי לוּטָה וְלִוְיָה

וְאָחוֹת יִקְרְאוּ לִי הַסְּגָנִים / וְהַשָּׂרִים לְקָחוּנִי לְרַעְיָה

וְלִי מִשְׂרָה וְלִי תִּכּוֹן מְלוּכָה / וּבִי גָדֵר הֲרִיסוֹתֶיהָ בְנוּיָה

וְלוּלֵי אֶאֱבֹד חַיִל מְזִידִים / וְאִישׁ מַכֶּה בְּאֵיבָה בַּצְּדִיָּה

אֲזַי מָלְאוּ פְנֵי תֵבֵל מְתֵי שָׁוְא / וְשֹׁפְכֵי דָם וְאֵין קֵץ לַגְּוִיָּה.

וּמַה לְּעֵט מְאֹד נִקְלֶה וְנִבְזֶה / אֲשֶׁר עֲלוֹת לוֹ הָעֲלִיָּה

וְיִתְפָּאֵר וְיִתְהַלֵּל בְּחָכְמָה / וְהוּא מִסְכֵּן וְחָכְמָתוֹ בְּזוּיָה.


וַיַּעַן הָעֵט וַיֹּאמַר: דַּי לָךְ כִּי תַעֲלִי, כִּי תָרֹמִי / הֵאָסְפִי אֶל תַּעְרֵךְ הֵרָגְעִי וָדֹמִי / וּמַה תִּתְהַלְּלִי בַּהֶרֶג וּבָרֶצַח / הָאוֹיֵב תַּמּוּ חֳרָבוֹת לָנֶצַח / וְכִי תְשִׂימוּן אֶרֶץ נוֹשֶׁבֶת חֲרֵבָה? / עֲמַדְתֶּם עַל חַרְבְּכֶם וַעֲשִׂיתֶם תּוֹעֵבָה. / וְאֵין בָּם מוֹעִיל, כִּי לְהַרְבּוֹת חֲלָלִים / לְמַעַן לָמוּג לֵב וּלְהַרְבּוֹת מִכְשׁוֹלִים / וּמִפִּיהָ דַּם נְקִיִּים הִזִּילָה / כִּי רַבִּים חֲלָלִים הִפִּילָה / וְנָקִי וְצַדִּיק לֹא הֶחֱיִית / מַכְשֶׁלֶת גּוֹיֶיךָ הָיִית / וּבְעֵת אֲשֶׁר פִּיהוּ עַל אוֹיְבָיו יִפְתַּח / כְּקַשׁ נֶחְשְׁבוּ לוֹ תוֹתָח / וּכְמוֹץ נִדָּף יֶהֶדֳפֶנּוּ בְנִיבוֹ / יִתֵּן בֶּעָפָר חַרְבּוֹ / וְקַשְׁתּוֹ יִשָּׁבֵר / וְחַיָּלִים יְגַבֵּר / וְיַשְׁמִידוּ / גְּדוּדִים וְשַׁלִּיטָם / בִּסְעִיף סֶלַע עֵיטָם וְאֶפְתְּחָה בְּמָשָׁל פִּי / אֶדְרְכֵם בְּאַפִּי / עַד אֲשֶׁר יְכַחֲשׁוּ לוֹ כִּי הִכְנַעְתִּים / וְכִתְּתוּ חַרְבוֹתָם לְאִתִּים וּלְפִי יְיַחֵלוּ לְהוֹרוֹת / אֵלֶּה הַחֻקִּים וְהַתּוֹרוֹת / וְהַשּׁוֹאֵל יִשְׁאַל בְּעֵטוֹ / וְלֹא יַרְשִׁיעֶנּוּ בְּהִשָּׁפְטוֹ / וּפִי יוֹדִיעָךְ מַה שֶּׁהָיָה לְפָנִים / בַּיָּמִים הַקַּדְמוֹנִים / וְדִבְרֵי הַיָּמִים. לְשָׂרֵינוּ, לִמְלָכֵינוּ / לְכֹהֲנֵינוּ אֲשֶׁר הָיוּ לְפָנֵינוּ / עַד תַּחְשֹׁב כִּי עִם כָּל אֶחָד תִּתְחַבֵּר / וּפִיהוּ אֶל פִּיךָ יְדַבֵּר / וְיוֹדִיעָךְ אֶת צִדְקָתוֹ וֶאֱמוּנָתוֹ / וְרִשְׁעוֹ וְחַטָּאתוֹ / וּמִפִּי תִקַּח וְלֶקַח וְעָרְמָה / וְיַגֶּד־לְךָ תַּעֲלוּמוֹת וְחָכְמָה.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֲנִי גֶבֶר וְלִי תֵיאוֹת גְּבוּרָה / בְּרוֹק פִּי אוֹיְבִי יֻתַּן שְׁמָמָה

וְאִם הָעֵט קְצַר קוֹמָה חֲנִית פִּיו / הֲלֹא תַגַּע עֲדֵי קַצְוֵי אֲדָמָה

אֲנִי חָכְמָה וְשָׁכַנְתִּי בְּעָרְמָה / וּפִי יַגִּיד לְךָ סוֹד תַּעֲלוּמָה

וְאַגִּידָךְ אֲשֶׁר הָיָה לְפָנִים / אֲבָל מַה שֶּׁיְהִי לֹא אֵדְעָה מַה.


וְכַאֲשֶׁר שָׁמְעָה הַחֶרֶב מַעֲנָיו, / וַתִּקֹּד וַתִּשְׁתַחוּ לְפָנָיו. / וַתֹּאמַר: תְּנָה לִי קְצָת חָכְמָתְךָ וּפָרַשְׂתָּ כְּנָפֶיךָ עַל אֲמָתְךָ / וְכַאֲשֶׁר הִתְחַנְּנָה לוֹ וָאִלַּצְתָּהוּ. / וַיַּגֵּד לָהּ כִּי הֱצִיקַתְהוּ. / וַיַעַן הָעֵט וַיֹּאמַר: דְּעִי כִי לֹא הַחֶרֶב גְּבוּרָה בַמִּלְחָמָה / וְאֵין לְבַעַל הָעֵט תְּבוּנָה וְחָכְמָה וּכְאֵין כָּל תְּבוּנָתוֹ אֲשֶׁר בְּלִבּוֹ תָּר / וּבִינַת נְבוֹנָיו תָּתַר / וּכְאֶפֶס חֲכַָמַי וּנְבוֹנַי / אֵין חָכְמָה וְעֵצָה נֶגֶד יְיָ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

פְּתַח סִפְרִי וְתִמְצָא בוֹ תְבוּנוֹת / וְהוּא יוֹדִיעֲךָ דַּעַת מְזִמָּה

עֲדֵי תָבִין אֲשֶׁר אֵין בָּךְ יְכֹלֶת / לְהוֹסִיף אוֹ לְגָרַע בַּמְּאוּמָה

וְאָז תֵּדַע וְתַשְׂכִּיל כִּי כְּמִקְרֶה / בְּנֵי אָדָם כְּמִקְרֶה הַבְּהֵמָה.

וְשַׁעַר מָוְתָה הַכֹּל יְבֹאוּן / וְלֹא יֵדַע אֱנוֹשׁ מַה בַּפְּנִימָה

וְלֹא יַעְמֹד לְבַד נוֹטֶה שְׁחָקִים / עֲלֵי תֹּהוּ וְאֶרֶץ עַל בְּלִימָה.


וַתִּתְמַהּ הַחֶרֶב וַתֹּאמַר: וּמִי הוּא זֶה הָאָדוֹן הַנּוֹרָא אֲשֶׁר יִמְשֹׁל בְּכָל נִבְרָא? וַיֹּאמַר: הוּא יְיָ אֱלֹהֵי הָאֱלֹהִים / וַאֲדוֹנֵי הָאֲדֹנִים רִאשׁוֹן לָרִאשׁוֹנִים / וַאַחֲרוֹן לָאַחֲרֹנִים / אֶחָד לְכָל הַאֲחָדִים / אֲשֶׁר הַכֹּל לוֹ מוֹדִים / אֲשֶׁר תְּבוּנָה נַעֲלָמָה הִמְצִיא / וְתַעֲלוּמֶיהָ אוֹר הוֹצִיא / וַתִּקַּח מִתְּבוּנָה הַפֶּרֶט / וַיֵצֵר אוֹתוֹ בַּחֶרֶט.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

שְׁמוֹנָה הֵם אֲבוֹת כָּל הָאֲחָדִים / וְהָאֶחָד מְיֻחָד בֶּאֱמוּנָה

אֲשֶׁא אֵין לוֹ דְבַר מַרְבִּית וְהוּא אֵל / אֱמֶת כִּי הוּא לְמִקֶּדֶם מְעוֹנָה

וְלֹא נִמְצָא בְמָקוֹם כִּי בְמַחְשָׁב / חֲכַם לֵב גַּם בְּמַצְרֵף הַתְּבוּנָה

וְהוּא אֵל חַי וְקַיָּם אֵין דְּמוּת לוֹ / וְאֵין עֵרֶךְ וְצוּרַת כָּל תְּמוּנָה

וְהוּא הִמְצִיא תְבוּנָה נַעֲלָמָה / וְכָל נִמְצָא לְאֵל שַׁדַַּי לְמִקְנָה

וְהוּא יָדַע לְכָל נִבְרָא עֲרָכָיו, / וּמֵאוֹתוֹ נְתוּנָה כָּל תְּבוּנָה

וְהוּא רֹאשׁ כָּל וְגַם הוּא אַחֲרִית כָּל / וְהוּא סִבָּה לְכָל סִבָּה הֲגוּנָה.


וּבִלְתּוֹ אֵין מְאַחֵר בַּאֲחָדִים. וְזוּלָתוֹ הֲלֹא שְׁלֹשׁ מֵאוֹת לָךְ אַחַת לְכָל הַבְּרִיאָה. בָּרוּחַ גַּם בַּגֶּשֶׁם הִיא נְתוּנָה / וְכָל גֶּשֶׁם בְּצוּרָה גַּם תְּבוּנָה / וְרוּחַ לָךְ אוֹת בָּאֱמוּנָה / וְכָל אֲשֶׁר נִקְרָא אֶחָד בִּלְתּוֹ וְאֵינוֹ צוֹמֵת / כִּי שֶׁקֶר הוּא מֵת / וַיְיָ אֱלֹהִים אֱמֶת.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

שַׁדַּי מִקְרָא בֶאֱמֶת אֶחָד / לֹא נֶחֱלַק לִסְפֹּר וְלֹא יֵחַד

כִּי כָל אֲשֶׁר נִקְרָא שְׁמוֹ אֶחָד / בִּלְתִּי יְחֻלַּק אַחֲרֵי נִצְמָד

אֶחָד אֲשֶׁר לֹא נֶחֱלַק יַעְמֹד / לָעַד וְאֶחָד נֶחְלַק נִשְׁמָד.


צַיִד לֹא צוֹד / וְלֹא יַחְמֹד / כִּי הַיְקוּם בִּלְבַד יַחְכִּים בְּרוּאָיו לַעֲבֹד / אוֹתוֹ מְלַמְּדָם אָכֵן בְּלִי סוֹד / הָאֱלֹהִים נֶעְלָם מִבְּלִי כִּי מִי בְסוֹד יְיָ נֶעְלַם עֲמֹד / וְהָאֶחָד אֲשֶׁר אַבִּירֵי אֲדָמָה יִסְלֶה, / כַּקָּטָן כַּגָּדוֹל יִכְלֶה / כִּי הִנֵּה קְדוֹשִׁי עֶלְיוֹן לֹא נֶאֶמְנוּ בִּכְבוֹדָם / אַף שֹׁכְנֵי בָּתֵי חֹמֶר אֲשֶׁר בֶּעָפָר יְסוֹדָם / וְכָל יוֹשְׁבֵי תַחְתִּית הַכֹּל יִמְעַד / וְצִדְקָתוֹ עוֹמֶדֶת לָעַד.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

אֶרֶץ וְשָׁמַיִם וְשֹׁכְנֵיהֶם / הַכֹּל אֱלֵי אֵל נַעֲרָץ נֶעְבַד

תַּחְתִּים וְעֶלְיוֹנִים בְּעֵין יוֹצְרָם / שֶׁקֶר כְּמוֹ נִקְלֶה כְּמוֹ נִכְבַּד

אֵין מַעֲמַד לָהֶם אֲבָל יַעְמֹד / הָאֵל אֱמֶת כִּי לֹא אֱמֶת תֹּאבַד

כִּי לֹא אֱמֶת תִּכּוֹן בְּנִבְרָא רַק / הָאֵל יְיָ הוּא אֱמֶת בִּלְבַד.


אֲבָל מַלְאֲכָיו הָעֶלְיוֹנִים / וּקְדוֹשָׁיו הַנֶּאֱמָנִים / וְכָל אֲשֶׁר אוֹרוּ מֵאוֹר כְּבוֹדוֹ יְסוֹדוֹ קָמּוּ עָמְדוּ. אַךְ בְּמַאֲמַר מֶלֶךְ הַכָּבוֹד. / כִּי הוּא אָמַר וַיֶּהִי, הוּא צִוָּה וַיַּעֲמֹד / עַל כֵּן מֵאוֹר יְקָרוֹ נִגְזָר הֶעֱמִיד אוֹתוֹ לִפְנֵי יְיָ לְשָׁרְתוֹ.


וַיִּשָּׂא מְשָׁלוֹ וַיֹּאמַר:

וְלֹא יִכְלוּ בְפֹעַל כִּי בְכֹחַ / וְהַכֹּחַ הֲלֹא קָרוֹב לְפֹעַל

וּמוֹצִיא תַעֲלוּמָה אוֹר הֲלֹא הוּא / מְסִבּוֹתָם הֲלֹא עָשָׂה וּפָעַל.


כִּי כָל נִבְרָא מוּכָן לְתַכְלִית / וְהַבּוֹרֵא יָכוֹל לְהַשְׁחִית / כִּי הָאָמְנָם לוֹ חֵפֶץ פִּתְאוֹם / יִפְקֹד יְיָ עַל צְבָא מָרוֹם / וַיִּמַּסּוּ בְמִיסֵי וְנָגוֹלוּ כְסֵפֶר שָׁמַיִם / וְעַל כֵּן בָּרָא שָׁמַיִם מוּסַבִּים / לְמַעַן יַעַמְדוּ יָמִים רַבִּים / וְכָל אֲשֶׁר נָתַן בֵּין הָאָרֶץ וּבֵין הַשָּׁמַיִם / יִכְלֶה וְיִשְׁקַע / וְכָל אֲשֶׁר בָּאָרֶץ יִגְוַע.


וַיְחַפְּשׂוּ לִבּוֹת בְּנֵי הָאָדָם / וְשֹׁטְטוּ בְּחוּצוֹת הַמַּדָּע כְּפִי תוֹלְדוֹת יְסוֹדָם / וַיַּחְקְרוּ וַיִּלְמְדוּ / אֶת אֲשֶׁר יַעֲמֹדוּ / וְהִשְׁתַּקְשְׁקוּ בְּחוּצוֹת הַתְּבוּנָה / לִרְאוֹת כָּל הַתְּמוּנָה / וּפָחֲדוּ וְרָהוּ / כִּי לֹא יָדְעוּ מַה הוּא / וְכַאֲשֶׁר הִתְאַחֲדָה עֲצָתָם / וְהִתְעוֹרְרָה מִשְּׁנָתָם חָכְמָתָם / וְנִצְרַף הַמַּחְשָׁב / וְלֵב הַשֵּׂכֶל רָץ וְשָׁב / וַיַּעַל כָּל רַעְיוֹן / לָדַעַת דַּעַת עֶלְיוֹן / עַד שֶׁנִּמְצָא בְּכָל נֵתַח / וַיִּלְאוּ לִמְצֹא הַפֶּתַח / כִּי לֹא הִֵשִּׂיגָה יַד הַתְּבוּנָה / לִרְאוֹת תְּכוּנָה / וְהִנֵּה מַלְאָכָיו לֹא רָאוּ כִּי אִם כְּבוֹדוֹ / וְלֹא יָדְעוּ סוֹדוֹ / כִּי אִם הוֹדוֹ / אֲשֶׁר יְסוֹד טָהֹר וְנַעֲלֶה יְסוֹדָם / אַף כִּי לִבּוֹת בְּנֵי אָדָם / אֲבָל רְאִינוּהוּ כְּעֵין הַלְּבָבוֹת / וּבְמַצְרֵף מַחֲשָׁבוֹת / וּמְצָאנוּהוּ בִּקְרָבוֹת / כִּי הוּא אֶחָד לְכָל אֲחָדִים אֲשֶׁר עָשָׂה נֶפֶשׁ חַיָּה וְצוֹמַחַת / וְנֶפֶשׁ חֲכָמָה אֲשֶׁר בָּאָדָם פּוֹרַחַת / וַיִּצְמַח וַיְהִי לְגֶפֶן פּוֹרַחַת / וֵיִּתֵּן עֲצָמוֹת וְגִידִים שְׁלֹשָׁה גְבִיעִים מְשֻׁקָּדִים / וְהוּא הָאֶחָד אֲשֶׁר הוֹדוֹ שָׁמַיִם כִּסָּה / וְהוּא עָשָׂה כָּל אֲשֶׁר נַעֲשָׂה. וַתֶּחֶרַד הַחֶרֶב וַתִּלְבַּשׁ רְגָזוֹת / כְּשָׁמְעָהּ הַחָכְמָה הַזֹּאת. / וַתֹּאמַר לָעֵט: הֵיטַבְתָּ מְאֹד לְהַחְכִּימֵנִי / צַדִּיק אַתָּה מִמֶּנִּי / וּמַה לַּחֲרָבוֹת אֶל הָעֵטִים / הַחָכְמָה תָּעֹז לֶחָכָם מֵעֲשָׂרָה שַׁלִּיטִים / כִּי מַה יִּמְתַּק שִׂיחֶךָ וּמַה יֶּעֱרַב / טוֹבָה חָכְמָה מִכְּלֵי קְרָב.
