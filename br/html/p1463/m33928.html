אַזְכִּיר גְּבוּרוֹת אֱלוֹהַּ

אַזְכִּיר גְּבוּרוֹת / אֱלוֹהַּ נֶאְדָּר / יָחִיד וְאֵין עוֹד / אֶפֶס וְאֵין שֵׁנִי,

אַחֲרָיו אֵין בַּחֶלֶד / לְפָנָיו אֵין בַּשַּׁחַק / אֵין בִּלְתּוֹ קֶדֶם / וְאֵין זוּלָתוֹ בְּעֵקֶב.


אָדוֹן לַחֲשׁוֹב / אֱלֹהִים לַעֲשׂוֹת / נִמְלָךְ וְאֵין נֶעְדָּר / שָׂח וְאֵין מְאַחֵר

אוֹמֵר וְעוֹשֶׂה / יוֹעֵץ וּמֵקִים / אַמִּיץ לָשֵׂאת / וְגִבּוֹר לִסְבּוֹל.


אֲשֶׁר לוֹ רְנָנוֹת / מִפִּי יְצוּרָיו / מִמַּעְלָה וָמַטָּה / יִשָּׂא תְהִלָּה

אֵל אֶחָד בָּאָרֶץ / וְקָדוֹשׁ בַּשָּׁמַיִם / מִמַּיִם רַבִּים / אַדִּיר בַּמָּרוֹם.


אֶדֶר מִתְּהוֹמוֹת / שֶׁבַח מִמְּאֹרוֹת / אֹמֶר מִיָּמִים / וְזֶמֶר מִלֵּילוֹת

אֵשׁ תּוֹדִיעַ שְׁמוֹ / עֲצֵי יַעַר יְרַנֵּנוּ / בְּהֵמָה תְּלַמֵּד / עֱזוּז נוֹרְאוֹתָיו.


אָמוֹן שַׂחֲקוֹ / דָּת שַׁעֲשׁוּעָיו / הִיא הֶגְיוֹנוֹ / עַד עֲמוֹד סְגֻלָּה

אָז קֶדֶם לְאֶלֶף דּוֹר / עָלְתָה בְּמַחֲשָׁבֶת / וּמִמֶּנָּה תְּכוּנַת / כָּל מַלְאֲכוֹת הַתַּבְנִית.


בַּמָּרוֹם הֵכִין / כִּסֵּא הֲדָרוֹ / פַּרְשֵׁז עֲנָנוֹ / וְדוֹק נָטָה לְאֹהֶל

בַּל יִצְעַן / וּבַל יִסַּע יְתֵדוֹתָיו / עַד בּוֹא קִצּוֹ / וְיִתְחַדֵּשׁ בְּאוֹמֶר.


בֵּירַר עַל־מַיִם / עַמּוּדֵי חָלֶד / וְשִׁנֵּס מָתְנֶיהָ / בְּתוֹהוּ וּבִסְעָרָה

בַּל תִּמּוֹט / וּבַל יִמְעֲדוּ מְכוֹנֶיהָ / עַד תִּבְלֶה כַבֶּגֶד / וְתוּמַר כְּמֵאָז.


בּוֹהוּ וַאֲפֵלָה / כִּסּוּ פְנֵי חָלֶד / וַתַּבְהֵק אוֹר / מֵאוֹר פְּנֵי מֶלֶךְ

בִּיהֵק נוֹגַהּ לְיוֹם / לָצֵאת אָדָם לְפוֹעַל / וְשָׁת חֹשֶׁךְ לְאִישׁוֹן / לִרְמֹס חַיְתוֹ יָעַר.


בְּרַחֵף רוּחַ / בֵּין שְׁחָקִים לְמַּיִם / וְשָׁת תִּקְרָה בַּתָּוֶךְ / שְׂאֵת חֲצִי מְדוּדִים

בָּם יִוָּדַע / פִּלְאוֹ לְפָעֳלוֹ / בְּטוּב פְּרִי גֶשֶׁם / פֶּלֶג מָלֵא מָיִם.


בְּהַבְדִּילוֹ / בֵּין מַיִם לָמָיִם / הִסִּיק עֲלוּקָה / תָּפְתֶּה לִבְלִי חֹק

בְּאֵשׁ לוּ נֻפָּח / וְלֹא נִכְבֶּה נֶצַח / וְשֵׁטֶף מַיִם רַבִּים / לֹא יַגִּיעֶנּוּ.


גֵּרַשׁ לַמִּקְוֶה / מַיִם אַדִּירִים / וְשָׂם לָהֶם חוֹל / דְּלָתַיִם וּבְרִיחַ

גָּדַר בַּעֲדָם / בַּל יְכַסּוּ אָרֶץ / עֵד נֶאֱמָן בַּיִת / יַעֲשֵׂם גְּזָרִים.


גִּילָּה פְּנֵי אֲדָמָה / לַעֲבוֹדַת אֱנוֹשׁ / וְהִדְשִׁיאָה שִׂיחַ / וְעֵשֶׂב מַזְרִיעַ זֶרַע

גַּם נָטַע בְּעֵדֶן / גַּן רַעֲנָנָה / לְחֻפַּת כְּבֻדַּת / אַדִּיר כָּל חֵפֶץ.


גָּמַר לְהַעֲלִים / אוֹר שִׁבְעָתַיִם / וְעָרַךְ נֵרוֹת / לִמְשֹׁל בַּיּוֹם וּבַלָּיְלָה

גְּבוּל לֹא יַשִּׂיגוּ / וְלָלֶכֶת לֹא יְאַחֵרוּ / עַד מְשָׁרֵת אֹהֶל / יַדְמִימֵם בְּעֵמֶק.


גָּאוּ מִמַּיִם / בְּרִיחֵי תְהוֹמוֹת / אֲפִיקֵי מָגֵן / שְׁנוּנֵי חִצִּים

גָּנַז מֵהֶם / לְמִשְׁתֵּה נְצָחִים / וְהֵכִין בָּם כֶּלֶא / לְצִיר הַבּוֹרֵחַ.

גְּבֵיהֵי עוּף / צָצוּ מֵרְקַק מָיִם / לְאוֹכְלֵי שֻׁלְחַן מֶלֶךְ / וְחֵיל צִבְאוֹתָיו

גָּזַר לְתוֹעֶבֶת / עוֹף לֹא טָהוֹר / עַד בּוֹא תִשְׁבִּי / וְעוֹרְבִים כִּלְכְּלוּהוּ.


דָּגוּ מֵחֶלֶד / מֻקְרָנִים לַזֶּבַח / חַיּוֹת מַאֲכָל / עַד בְּהֵמָה וָרָמֶשׂ

דִּשֵּׁן בְּהֵמוֹת / בְּיבוּל הָרֵי אֶלֶף / כִּי בְיוֹם זִבְחוֹ / יַגֵּשׁ חַרְבּוֹ.


דָּץ יוֹצֵר / וְשָׂמַח בְּמַעֲשָׂיו / בְּהַבִּיטוֹ / כִּי טוֹב פָּעֳלוֹ

דֶּשֶׁא מְנָחוֹת / וְאָכְלֵי מְגָדִים / וְשֻׁלְחָן עָרוּךְ / וְאֵין מִי יְדֻשָּׁן.


דִּבֶּר בְּלִבּוֹ / מִי יָסוּר הֵנָּה / לַטִּבּוּחַ הַטָּבוּחַ / וְיַיִן הַמָּסוּךְ

דְּבָרִי אִם יַעַשׂ / יִהְיֶה כַאלֹהִים / וְאִם יָמִיר אֹמֶר / לְחָמְרוֹ אֲשִׁיבֶנּוּ.


דָּרַשׁ וְהֵבִין / וְיָדַע מְתֵי שָׁוְא / כִּי לֹא יָבִין / לָלִין בִּיקָר

דִּמָּהוּ בְצֶלֶם / חֲקָקוֹ בְתַבְנִית / לִהְיוֹת בְּדוֹק וָחֶלֶד / פַּחַד מוֹרָאוֹ.


דָּגוּל בְּתוֹךְ עֵדֶן / יִפָּה יְצוּעוֹ / בְּפָז וְאֶבֶן יְקָרָה / חָק מְסֻכָּתוֹ

דַּעַת קִשְּׁטוֹ / וְרוּחַ חֲנָנוֹ / לְהַשְׂכִּיל מִקֶּדֶם / קֵץ אוֹתִיּוֹת.


הִמְשִׁילוֹ / כְּצַדִּיק וּמוֹשֵׁל / לְכַלְכֵּל שֵׁמוֹת / לָאֵל וּמִפְעָלָיו

הִזְהִירוֹ / עַל עֵץ הַדַּעַת / פֶּן יִנָּקְשׁוּ / בְּמַאֲכַל תַּאֲוָה.


הִרְדִּימוֹ / בְּנֹעַם שֵׁינָה / וְהֵרִים מִמֶּנּוּ עֶצֶם / וְכוֹנֵן עַלְמָה

הֶעְדָּהּ כְּכַלָּה / עֲדִי קִשּׁוּרִים / וְהִדְבִּיקָם יַחַד / הֱיוֹת שְׁאֵר אֶחָד.


הִשִּׂיא פוֹתָה / זוֹחֵל עָפָר / לְהָנִיא אֹמֶר / וְלָשׁוּב לֶעָפָר

הֵמִיתָה אִישׁ / בְּתַאֲוַת הָעֵץ / וְהִפִּילָה רַבִּים / וְהָרְגָה עֲצוּמִים.


הֻכָּה נָחָשׁ בְּטַרְפּוֹ / וּבִפְעָמָיו לָנֶצַח / וְהוּא יְשֻׁלַּח צִיר / לְפוֹרְצֵי גָדֵר

הָעָרְמוּ וְכֻסּוּ / בַּעֲלֵה נוֹבֶלֶת / תְּמוּר לְבוּשׁ עָנָן / חֲתֻלַּת עֲרָפֶל.


הֵרְיוֹן עֶצֶב / נִתַּן לְתוֹלָדוֹת / וְהוּא לְעָמָל וְעִצָּבוֹן / לְבָאֵי עוֹלָם.

הוּמַר תָּאֳרָם / וְהוּסַר פַּחְדָּם / לְהֵרָדוֹת מִכֹּל / וְלֹא לִרְדּוֹת בַּכֹּל.


הוּרְמוּ מֵעֵדֶן / וְגוֹרְשׁוּ מִגָּן / וְהוּכַן נְקִיק סֶלַע / בֵּית מָלוֹן לָמוֹ.

הֻפְתָּה יְצִיר פֶּתִי / בְּשִׂיחַ פִּתְיוֹן פֶּתֶן / וְגֵיא בָּם קֻלָּלָה / כְּאֵם עַל בָּנִים.


וְיוֹצֵר הֵחִישׁ / תַּאֲוַת יְצוּרוֹ / וְהִרְגִּיל עֶזְרוֹ / לְרֶבַע תְּשׁוּקָתוֹ

וְיִחֲמָה וְהִזְרִיעָה / וְחִבְּלָה וְיָלָדָה / וּבִכְּרָה מְשַׂדֵּד / תַּלְמֵי אֲדָמָה.


וְהוֹסִיפָה שֵׁנִית / וְהִכרִיכָה וְשִׁלֵּחָה / עֲוִיל תּוֹמֵךְ שֵׁבֶט / רוֹעֶה עֲדָרִים

וּבִתְשׁוּרַת שַׁי / קִדְּמוּ פְנֵי מֶלֶךְ / זֶה בִּפְרִי אֶרֶץ / וְזֶה בְּפִטְרֵי צֹאן.


וְהִבִּיט קָדוֹשׁ / לַלֵּב וְלֹא לַָעֵינַיִם / וְשָׁע עֵרֶךְ / נִיחֹחֵי צָעִיר

וְשִׂגֵּב אָח / קוֹדֵר לְמָרוֹם / בִּרְצוֹתוֹ שַׁי / עֵרֶךְ נִיחוֹחוֹ.


וְהִקְדִּיר פְּנֵי רַב / וְגַם חָרָה לוֹ / כִּי מָאַס בְּשַׁי / יֶתֶר מַאֲכָלוֹ

וַיִּפְעַל זֶה / כְּמִשְׁפַּט הוֹרֵהוּ, / כִּי לֹא נָצַר / מִצְוַת מְצַוֵּהוּ.


וְטָרַף אַף / וְשָׁמַר עֶבְרָה / וְחִבֵּל מֵאֶרֶץ / אָח וּמִשְׁפְּחוֹתָיו

וְהִצְרִיחַ זְעָקָה / דָּם הַשָּׁפוּךְ / וְנֶעֱנַשׁ בְּנָע וָנָד / רֹאשׁ לְשׁוֹפְכֵי דָם.


וְנִתְוַדָּה עַל פֶּשַׁע / וְנִתַּן לוֹ אוֹת / פֶּן יִשָּׁפֵךְ / דָּמוֹ בָאָדָם

וְהֵחֵל תְּשׁוּבָה / לְבָאֵי עוֹלָם / כִּי כָל מוֹדֶה וְעוֹזֵב / בִּתְשׁוּבָה יְרוּחָם.


זֶה חָבַשׁ עֶצֶבוֹ / יְצִיר קַדְמוֹנִי / כִּי הֵחֵל שְׁתוֹת כּוֹס / מָסוּךְ לְהוֹדוֹת

זָכַר לָתֵת לֹו / זֶרַע לִזְרֹעַ / לְהַחֲלִיף לוֹ / פְּרִי הָאֻמְלָל.


זַרְעוֹ הִשְׁרִישׁ / שֹׁרֶשׁ פּוֹרֶה רוֹשׁ / וְשִׂגְשֵׂג נִטְעוֹ / עֲשׂוֹת פְּרִי בְאוּשִׁים

זָמְמוּ וְהֵחֵלוּ / דּוֹר עֲקַלְקַלּוֹת / לְהָמִיר בְּשֵׁם אֱלִיל / שֵׁם אֵל עוֹלָם.


זַעַף מַיִם עֲצוּרִים / עֲלֵיהֶם קָרָא / פָּרְצוּ גְבוּלָם / וַיַּהַפְכוּ אָרֶץ

זֵכֶר תִּקְוַת אֱנוֹשׁ / הֶאֱבִיד מִן הָאָרֶץ / וְשָׁטַף יְסוֹדֵי / סְפִיחֵי עֲפַר אָרֶץ.


זַלְזַלִּים אֲשֶׁר / בַּעֲצֵי יָעַר / עַד דּוֹר עֲשִׂירִי / לֹא עָשׂוּ פֶרִי

זַךְ וְתָמִים / בְּקִרְבָּם נִמְצָא / וְרֵיחַ מִפְעָלָיו / כְּרֶקַח בְּדוֹרוֹתָיו.


זָדוּ וּמָרְדוּ / בְּשִׁבְעַת יְמֵי נוּגָה / וּמֵרוֹב כָּל טוּב / אָמְרוּ לָאֵל סוּר

זוֹעֲמוּ בְּמֵי מָעְלָה / וְדֻכְּאוּ בְּמֵי מָטָּה / וְרִצָּהוּ צַדִּיק בְּשַׁי / מֵעֲבוֹר מֵי נוֹחַ.


חֻדְּשׁוּ פְנֵי חֶלֶד / וְשָׁבוּ לְקַדְמָתָן / מִמְּתֵי מִסְפָּר / יוֹצְאֵי מִמַּסְגֵּר

חָשְׁבוּ וְנוֹעָצוּ / אָז דּוֹר חֲמִישִׁי / לָרוּם וְלָשׂוּם / קֵן בַּמָּרוֹם.


חַי לֹא בִצֵּר / מֵהֶם מְזִמָּה / מֵלִיץ לַלֵּצִים / חֶפְצָם הִשְׁלִים

חִלֵּק שְׂפַת לְשׁוֹן / בַּעֲלֵי מַחֲלֹקֶת / וְכֻתְּתוּ זֶה בָזֶה / וְנִלְחֲמוּ לְמַשְׁחִית.


חָלְפוּ כַּעֲבוֹר / סוּפָה וְאֵינֵמוֹ / וְהוּקַם צַדִּיק / יְסוֹד עוֹלָם

חָשַׁב לְהָבִין / סוֹד מִפְעָלוֹת / הַמִּתְנַהֲגוֹת / בְּשׁוֹטֵר וּמוֹשֵׁל.


חֲזוֹתוֹ מְרוּצַת / שׁוֹאֵף וְ[זוֹרֵחַ / שָׂשׂ כְּגִבּוֹ]ר בְּצֵאתוֹ / וְחַלָּשׁ בְּבוֹאוֹ

חַלּוֹנֵי שַׁחַק / אֲשֶׁר בְּקֵדְמָה וְיָמָּה / אֲשֶׁר בָּם תְּנַתֵּר / לְבָנָה יוֹם יוֹם.


חִצֵּי בְּרָקִים / וְדוֹהֵר כּוֹכָבִים / רָצִים וְשָׁבִים / וְאִישׁ לֹא נֶעְדָּר

חָכַם עוג לֵב / וְשָׂכַל מֵאֵלָיו / וְשָׂח: אָדוֹן לָאֵלֶּה / אַחֲרָיו אָרוּצָה.


טֶרֶם הִקְדִּים אֵל / בֵּירַר לוֹ דָרֶךְ / לֶךְ־לְךָ מִמָּוֶת / לְדֶרֶךְ הַחַיִּים

טָהוֹר בְּמַעֲשָׂיו / דִּבֶּר גְּבוּרוֹת: / נָא אֱלֹהֵי הַמִּשְׁפָּט / שְׁפָט־נָא בְרַחֲמִים.


טָרַף רוֹזְנִים / בְּלֹא מָגֵן וָחָרֶב / שִׂגְּבוֹ מֵאֵשׁ / עֲזָרוֹ בְמִלְחָמֶת

טָפַשׁ דּוֹרוֹ / וְצוּר לֹא הִכִּירוֹ / וְהוֹדִיעָם בְּאֵשֶׁל / שֵׁם אֵל עוֹלָם.


טֶבַע חוֹתָם בְּרִית / שָׂם בִּשְׁאֵרוֹ / בָּהּ תִּפֶן פּוֹעֶרֶת / וְאָז תִּקְפֹּץ פִּיהָ

טַרְפֵי צֶמַח זְקוּנִים / הֵנֵצּוּ מִמֶּנּוּ / בְּהִפָּתַח מָקוֹר / לַעֲרוּגַת הַבֹּשֶׂם.


טוֹרֵחַ בּוֹחֵן מְנַסֵּיהוּ / עָמַס עַל־שֶׁכֶם / וְגַם מָשַׁל בְּיֵצֶר / תֵּת שֵׂיוֹ לִטְבֹּחַ

טֶנֶא בִכּוּרִים / לְמִנְחָה הוֹבִיל / אָב לֹא חָמַל / וּבֵן לֹא אֵחַר.


טוֹבֵחַ בְּעַד שֶׂה / בְּ[עֵ]ת [תָּפַשׂ] חָרֶב / מַלְאֲכֵי שָׁ[לוֹם] / מַר צָעֲקוּ בֶכִי

טוֹב וְרַחוּם שָׂח: / אַל תַּשְׁחֵת נָעַר / נִרְצָה פָעָלְכֶם / כְּזוֹבֵחַ וְנִזְבָּח.


טָלֶה הוּחָן / וְהוּשַׁת כָּפְרוֹ אַיִל / וְיוּחַק שָׂרוּף / לְהַצְדִיק עֲמוּסָיו.


יֻסַּד אִישׁ תָּם / בְּפִתְחֵי תְעוּדָה / טֶרֶם תֵּאָמֵר / שִׁנֵּן אֲמָרֶיהָ

יָהּ מֵעוֹלָם / בּוֹ נִתְפָּאָר / וּבִשְׁמוֹ לְעוֹלָם / יוֹשֵׁב תְּהִלּוֹתָיו.


יוֹדְעוֹ נִצָּב / עָלָיו בַּמָּלוֹן / וְשָׂח: אֲנִי שׁוֹמְרֶךָ / צֵל עַל יְמִינֶךָ

יוֹרְדִים וְעוֹלִים / קְדוֹשִׁים לַמַּעֲנֵהוּ / לְהַכִּיר תָּאֳרוֹ / חָקוּק בַּמְּרוֹמִים.


יוֹצֵר חִסְּרוֹ / מְעַט מֵאֱלֹהִים / לְיַחֵם בְּמַקְלוֹת / כַּחֹמֶר בְּיַד הַיּוֹצֵר

יִרְאַת חָזוֹן / אֲרַמִּי בִּעֲתַתּוּ / אַל תִּגַּע בְּבֵן / עָמוּס מִנִּי בָטֶן.


יִדּוֹד אֵשׁ לוֹהֵט / עִמּוֹ בְּהֵאָבְקוֹ / לוֹ כִּי לֹא יָכוֹל / נִתְחַנֵּן בְּבֶכִי

יֶשַׁע הִפְלִיא לוֹ / מֵחֲגוּר חָרֶב / שׂוֹטֵם וְשׁוֹמֵר / אֵיבַת עוֹלָם.


יָרַשׁ תַּאֲוַת / בִּרְכוֹת הוֹרָיו / שְׁנֵים עָשָׂר שֵׁבֶט / כְּשׁוֹטְרֵי רָקִיעַ

יוּחַד שְׁלִישִׁי / לִרְאוֹת פְּנֵי מֶלֶךְ / לְשׁוֹרֵר לְשָׁרֵת / לָבוֹא [חֲדָ]רָיו.


כְּעֵץ שָׁתוּל / עַל פַּלְגֵי מַיִם / [הִפְ]רִיחַ מַטֵּהוּ / שְׁלֹשֶׁת בַּדֵּי קֹדֶשׁ:

כְּפוּפֵי סֵבֶל / [לְאֹ]הֶל וּקְרָשָׁיו, / תּוֹקְעֵי יָתֵד / פּוֹרְשֵׂי יְרִיעוֹת,


כָּתֵף נְטוּיָה / [כְּ]עֶגְלָה מְלֻמֶּדֶת / לְעוֹל עֲבוֹדַת קֹדֶשׁ / אֲדוֹן כָּל הָאָרֶץ.

כְּגֶפֶן אַדֶּרֶת / יְפַת פְּרִי וְעָנָף / הִצְמִיחַ עַמְרָם / מִשֹּׁרֶשׁ לֵוִי,


כִּי שִׁלְּחָה שְׁלֹשֶׁת / שָׂרִיגֵי חֶמֶד: / מְכַהֵן וְרוֹעֶה / וְאִשָּׁה נְבִיאָה.

כְּגֶשֶׁת עֵת דּוֹרִים / פִּרְחוֹ הוּקָם / לִשְׁבֹּר מוֹסְרוֹת צֹעַן / וְלִפְרֹץ גֶּדֶר שׁוֹעֵל.


כֻּסָּה בֶעָנָן / וְנִתְקַדֵּשׁ שָׁבוּעַ / וְנִצָּב בַּתָּוֶךְ / בְּיוֹם תֵּת אֹמֶר

כֹּחַ אַדִּירִים / לְפָנָיו הֻכְנָע / וּמִמּוֹרָאוֹ / לֹא עָמַד אִישׁ.


כִּלְכֵּל צֹאן קֹדֶשׁ / שְׁאֵר בִּישִׁימוֹן / מִלֶּחֶם שְׁחָקִים / עַד בּוֹאָם לָאָרֶץ.

כָּרְתָה בֹּאָהּ לָעָם / עַלְמָה מְתוֹפֶפֶת / גָּוְעָה וְנֶאֶסָפָה / וְלֹא הָיָה מָיִם.


לִוּוּיֵי עַנְנֵי הוֹד / יְדִידִים נֶחֱנָנוּ / עַל יְדֵי מְכַהֵן / בְּשָׁלוֹם וּבְמִישׁוֹר

לוֹ וּלְזַרְעוֹ / בְּרִית מֶלַח נֶחְקָקָה / לְבַל תַּשְׁבִּית בְּרִית / מֶלַח הַנִּיחֹחַ.


לְקִנְאַת חֲמַת גֶּבֶר / גְּמוּל גָּמַל אֵל / וְחִדֵּשׁ חֻקּוֹ / לְדוֹרוֹת עוֹלָם

לְזָר לֹא יָאֲתָה / נַחֲלַת כְּבוֹדָם / וְלֹא יִירַשׁ אֱנוֹשׁ / מַתְּנַת חֶלְקָם.


לְצַלְמָוֶת הוֹרִידוּ / עֲדַת אַנְשֵׁי שֵׁ[ם / וְהוֹשִׁיבוּ] עֻזִּיָהוּ / בְּבֵית הַחָפְשִׁית

לֹא יְקַנְאוּ בְחֶבֶל / [אֲחֻזַּת] אַחִים / כִּי אֱלֹהִים נִקְרָא / מְנָת גּוֹרָלָם.


לֶחֶם הָאִ[שִּׁים] / לָהֶם הוּכָן / בְּהִתְקָרְבָם לָאֵל / בְּעֵת רִחוּק עֵ[גֶל]

לִמְּדָם מְחוֹקֵק / סִדְרֵי עֲבוֹדָה / כִּי בְּשִׁבְתָּם פֶּתַ[ח] / עֲלֵיהֶם הֻפְקָד


לִרְחֹץ וְלָסוּךְ / לְקַדֵּשׁ יָד וָרָ[גֶל] / לִלְבֹּשׁ בַּדִּים / וְלַחֲגֹר בְּמֵזַח

לְמַלֵּא יָדָם / יָמִ[ים] שִׁבְעָה. / וַיְשִׂימֶהָ לְחוֹק / לְדוֹרוֹ עוֹלָם,


מִשְׁפְּחוֹת בְּרוּרִים / גּוֹרָל יָטִילוּ / לְגַדֵּל סֶגֶן / לַעֲלוֹתוֹ בֶּעָשׂוֹר

מֵעֶרֶשׂ יְצוּעוֹ / שְׁאֵרוֹ יַבְדִּילוֹ / פֶּן יִטְמָא שָׁבוּעַ / בְּשִׁגְגוּת דָּוָה.


מְקוֹם עֲצֶרֶת סְגָנִים / בֵּית מוֹשָׁבוֹ / הֲלֹא שָׁם יָלִין / כָּל יְמֵי שָׁנָה

מַעֲלֶה מִנְחָה / וְנוֹתֵן סַמִּים / וְעוֹרֵךְ נֵרוֹת / וְראשׁוֹן בְּרֹאשׁ וָרָגֶל.


מִמְ[כַּר] שַׁבָּתוֹת / וְחֵלֶק מִשְׁמָרוֹת / חֶלְקוֹ בְרֹאשׁ / כְּנֶגֶד כֻּלָּם

מַחְכִּימֵי פֶתִי / יָעָמְתוּ לוֹ / לְשַׁנְּנוֹ לְחַנְּכוֹ / בְּחֻקֵּי עָשׂוֹ[ר].


מְבִיאִים לוֹ / [בְּעֶרֶב י]וֹם סְלִיחָה / פָּרוֹ וְאֵילוֹ / וְזִבְחֵי קְהָלוֹ

מַה נָּאָה / עֲבוֹדָתֶךָ [יֹאמֵרוּ], / חֲזַק וֶאֱמָץ / כִּי כָל אֵלֶּה פָּעָלֶךָ.


מַאֲכָל יַמְעִיטוּ / מֶנּוּ [לְעֵת] עֶרֶב / פֶּן יֵרָדֵם וְיִ[קֶּר] / מִקְרֵה לַיְלָה

מְחַנְּכָיו יְחַבְּרוּהוּ / לְזִקְנֵי שִׁבְטוֹ / וּבְבֵית [רוֹקְחֵי רֶקַח / בַּ]שֵּׁם יַשְׁבִּיעוּהוּ.


נֶעֱגַם בְּבֶכִי / כִּי הֻרְשַׁע כְּפֶתִ[י] / וְהֵמָּה יִבְכּוּ / פֶּן צַדִּיק יַרְשִׁיעוּ

נְעִימוֹת מִדְרָשׁ / וּמִקְרָא יְשׂוֹחֵחוּ / אַף בְּכִתְבֵי קֹדֶשׁ / הֵם יַעֲסִיקוּהוּ.


נַפְשׁוֹ יְשׁוֹבֵבוּ / בְּשִׂיחַ מַלְכֵי קֶדֶם / אִם בַּעַר הוּא / וְלֹא לָמַד לֶקַח

נָטָה לָנוּם / שְׁנָתוֹ יַפְרִידוּ / בְּנֹעַם שִׁיר צְרָדָה / בְּפֶה וְלֹא בְּכִינוֹר.


נוֹתְנִים שָׁאוֹן / עִיר הוֹמִיָּה / לְהַשְׂבִּיעוֹ / נְדוּדִים עֲדֵי נָשֶׁף

נוֹצְרֵי חֻקִּים / וּבוֹזְזֵי מִצְווֹת / לְהָרִים דֶּשֶׁן / חֲצוֹת לַיְלָה יַקְדִּימוּ.


נֶהְדְּפוּ רָצִים / בְּמַעֲלוֹת מִזְבֵּחַ / וְשָׁקְדוּ בְפַיִס / לְהַשְּׁבִּית מְדָנִים

נוֹעֲרוּ וְטָבְלוּ / וְעִנְּדוּ עֲטָרָה / וְהֵפִיסוּ תְחִלָּה / עַל־תְּרוּמַת דָּשֶׁן.


נִצְבְּעוּ שֵׁנִית / עַל־מַעֲשֵׂה כֶבֶשׂ / וְעַל־דִּישׁוּן מִזְבֵּחַ / פְּנִימִי וְנֵרוֹת.

עַל שְׂפַת שׁוּלָיו / רִמּוֹנֵי צְבָעִים / וּפַעֲמוֹנֵי פָז / סָבִיב בַּתָּוֶךְ

עֵת יַשִּׁיקוּ / קוֹל זֶה בָּזֶה / יְכֻפַּר בָּזֹאת / מַכֵּה רֵעֵהוּ בַּסָּתֶר.


פְּעֻלַּת חוֹשֵׁב / אֲרִיגַת הָאֵפוֹד / מָשְׁזָר שֵׁשׁ וְשָׁנִי / וּתְכֵלֶת וְאַרְגָּמָן

פָּז מְסֻבּוֹת / בְּמִלְּאוֹתָם / וּמַרְאֵה דְמוּתָ[ם] / כְּמַרְאֵה עֵינַיִם /

פֶּרֶץ דְּבָקִים / אֲשֶׁר בֵּין אֵפוֹד לְחֹשֶׁ[ן] / יְכַסֶּה בְשַׁרְשְׁרוֹת / עֲבוֹתוֹת הַזָּהָב.


פֶּשַׁע הוֹפְכֵי מִשְׁפָּט / יִמְחֶה בְּחֹשֶׁן הַמִּשְׁפָּט / וְיַצְדִּיקֵנוּ בַמִּשְׁ[פָּט] / אֱלֹהֵי הַמִּשְׁפָּט

צִיץ פָז מֻקָּף / צִצֵי פְנִינִים / דִּבֵּק וְחִ[בֵּר] בְּחוּט / לְנֵזֶר הַקְּדֻשָּׁה

צוּר עוֹלָמִים / הָאֵל הַנַּעֲרָץ / חָקוּק עַל־צִיץ / בְּמִכְתַּב חוֹתָם

צוֹאַת טֻמְאַת / עֲווֹן דָּם וָנֶסֶךְ / יְרַצֶּה הַצִּיץ / כְּנִיחוֹחַ טָהוֹר.


צִיץ נוֹבֵל / תּוֹלַעַת וְלֹא אִישׁ / נֶחְשָׁב לְשָׁרֵת / לְצוּר עוֹלָמִים

צוֹפֶה פְּנֵי מֶלֶךְ / וּבָא בַחֲדַר קֹדֶשׁ / לָבוּשׁ בַּדִּים / וְלֹא בְכֶתֶר אוֹפִיר.


צָו עוֹנֶשׁ מִיתָה / לְבָאֵי עֲזָרָה / לְקַדֵּשׁ יָדַיִם וְרַגְלַיִם / בְּבוֹאָם לְשָׁרֵת

צְעִירָיו יַגִּישׁוּ לוֹ / אֶת־תְּמִיד הַשָּׁחַר / קְרָצוֹ וְקִבֵּל / דָּמוֹ וְזָרַק

קִדַּם בְּצֵאתוֹ / לְבֵית פַּרְוָה בַּקֹּדֶשׁ / וְסָכוּ מְחִיצַת שֵׁשׁ / בֵּינוֹ לְבֵין עָם

קָרַב אֵצֶל / פַּר חַטָּאתוֹ / וְהוּא עוֹמֵד / בֵּין אוּלָם וְלַמִּזְבֵּחַ

קָם תֵּת יָדוֹ / עַל־רֹאשׁוֹ בְּסֵבֶל / לְהִתְוַדּוֹת בְּעַד / עַצְמוֹ וְנָוֵיהוּ.


סוֹמֵךְ שְׁתֵּי יָדָיו עָלָיו וּמִתְוַדֶּה, וְכָךְ הָיָה אוֹ׳ אָנָּא הַשֵּׁם חָטָאתִי עָוִיתִי פָּשַׁעְתִּי לְפָנֶיךָ אֲנִי וּבֵיתִי אָנָּא הַשֵּׁם כַּפֶּר־נָא עַל הַחֲטָאִים עַל־הָעֲווֹנוֹת וְעַל הַפְּשָׁעִים שֶׁחָטָאתִי שֶׁעָוִיתִי שֶׁפָּשַׁעְתִּי לְפָנֶיךָ אֲנִי וּבֵיתִי. ככת׳ בְּתוֹרַת מֹשֶׁה עַבְדֶּךָ לֵאמֹר כִּי בַיּוֹם הַזֶּה יְכַפֵּר עֲלֵיכֶם [לְטַהֵר אֶתְכֶם] מִכֹּל חַטֹּאתֵיכֶם לִפְנֵי הַשֵּׁם. וְהַכֹּהֲנִים וְהָעָם, הָעוֹמְדִים בַּעֲזָרָה וְהַמְשַׁמְּ[שִׁים] בַּהֵיכָל, בִּזְמַן שֶׁשּׁוֹמְעִים אֶת־הַשֵּׁם הַמְפוֹרָשׁ הַמְפוֹאָר שֶׁיּוֹצֵא מִפִּי כֹּהֵן גָּדוֹל בִּקְדֻשָּׁה וּבְטָהֳרָה הָיוּ כוֹרְעִים וּמִשְׁתַּחֲוִים וְנוֹפְלִים עַל־פְּנֵיהֶם ואו׳ בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד. אַף הוּא מִתְכַוֵּן לִגְמוֹר אֶת הַשֵּׁם כְּנֶגֶד הַמְבָרְכִים וְאוֹמֵר לָהֶם: תִּטְהָרוּ.

קָפַץ וּנְתָנָם / עַל שְׁנֵי הַשְּׂעִירִים / וְהִצְרִיחַ בְּקוֹל מַר / זֶה לְשֵׁם יהוה חַטָּאת

קָשַׁר שָׁנִי / בְּרֹאשׁ שָעִיר הַמִּשְׁתַּלֵּחַ / פָּנָיו לַמִּדְבָּר / וְזֶה לְבֵית הַשְּׁחִיטָה.


וְכָךְ הָיָה אוֹמֵר אָנָּא הַשֵּׁם יהוה חָטָאתִי עָוִיתִי פָּשַׁעְתִּי לְפָנֶיךָ אֲנִי וּבֵיתִי וּבְנֵי אַהֲרֹן עַם קְדוֹשֶׁךָ אָנָּא בַּשֵּׁם יהוה כַּפֶּר־נָא לַחֲטָאִים וְלַעֲווֹנוֹת וְלַפְּשָׁעִים שֶׁחָטָאתִי שֶׁעָוִיתִי שֶׁפָּשַׁעְתִּי לְפָנֶיךָ אֲנִי וּבֵיתִי וּבְנֵי אַהֲרֹן עַם קְדוֹשֶׁיךָ כַּכָּתוּב בְּתוֹרַת מֹשֶׁה עַבְדֶּךָ כִּי בַיּוֹם הַזֶּה יְכַפֵּר עֲלֵיכֶם לְטַהֵר אֶתְכֶם מִכֹּל חַטֹּאתֵיכֶם לִפְנֵי יהוה. וְהַכֹּהֲנִים וְהָעָם הָעוֹמְדִים בַּעֲזָרָה בְּשָׁעַת שֶׁהָיוּ שׁוֹמְעִים אֶת־הַשֵּׁם הַמְפוֹרָשׁ יוֹצֵא מִפִּי כֹּהֵן גָּדוֹל בִּקְדֻשָּׁה וּבְטָהֳרָה הָיוּ כוֹרְעִים וּמִשְׁתַּחֲוִים וְנוֹפְלִים עַל פְּנֵיהֶם וְאוֹמְרִים בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד. וְאַף הוּא הָיָה מִתְכַוֵּן לִגְמוֹר אֶת־הַשֵּׁם אַחַר הַמְבָרְכִים וְאוֹמֵר לָהֶם: תִּטְהָרוּ.


רְצָחוֹ כַדָּת / וְקִבֵּל דָּם בְּכֶלִי / וְאַחַר יְמָרְסֶנּוּ / בְּמַעֲלוֹת הָאוּלָם

רָמַס כֶּבֶשׁ / וְעָלָה לְרֹאשׁ / וּבְיָדוֹ מַחְתַּת / פָּז הָאָדוֹם.


רְשָׁפִים מִלְּאָהּ / מִפְּנֵי מַעֲרָב / יָרַד וְשָׂמָהּ / בְּרֹבֶד המרוצפת

רֵעָיו יוֹצִיאוּ לוֹ / אֵת הַמַּחְתָּה / וּכְלִי בֵית צְבִירַת / דַּקַּת הַדָּק.


רוֹקַח מְלֹא חָפְנָיו / נָתַן לְתוֹךְ כָּף / נְטָלָהּ בִּשְׂמֹאל / וּמַחְתָּה בְיָמִין

רִכַּב כֹּבֶד / יָמִין עַל שְׂמֹאל / וְהִשְׂמִאיל בַּהֵיכָל / בְעֵקֶב הַפָּרֹכֶת.


רִגֵּל בְּקִרְבָּם / מִנֶּגֶד לְצָפוֹן / וּבְהַגִּיעוֹ שָׁב / עַד חֲצִי הַבָּיִת

רָאשֵׁי הַבַּדִּים / כְּמַרְאֵה שָׁדַיִם / לוֹחֲצִים בַּפָּרֹכֶת / וְאֵין דֶּרֶךְ לַעֲבוֹר.


רָב כֹּחַ יוֹדֵעַ / וְיִדְחֶה בַּפָּרוֹכֶת / וְיָבֹא בְּלֹא פֶסַע / בֵּין שְׁנֵי הַבַּדִּים

רֹאשׁ עַפְרוֹת תֵּבֵל / אֶבֶן הַשְּׁתִיָּה / מָקוֹם מוּכָן שָׁם / שָׂם מַחְתַּת פָּז.


שָׁפַךְ בָּהּ קְטֹרֶת / וְתִמְרוֹת עָשָׁן / וְהִצְדִּיד פָּנָיו / וְיָצָא כְדַרְכּוֹ

שִׁוַּע בְּעַד עָם / בְּקֹצֶר בַּהֵיכָל / אֲשֶׁר לֹא יָשׂוֹר / רַב עַל צָעִיר.


שׁוֹבֵב לוֹ דָּם / הַמְּמָרֵס בּוֹ / וּבָא וְנִתְיַצֵּב / בִּמְקוֹם עָמְדוֹ

שָׁלַח אֶצְבַּע מְיֻמֶּנֶת / וְטָבַל וְהִצְלִיף / אַחַת לְמַעְלָה / וְשֶׁבַע לְמַטָּה בְמִסְפָּר.


וְכָךְ הָיָה מוֹנֶה אַחַת, אַחַת וְאַחַת אַחַת וּשְׁתַּיִם אַחַת וְשָׁלוֹשׁ אַחַת וְאַרְבַּע אַחַת וְחָמֵשׁ אַחַת וָשֵׁשׁ אַחַת וָשֶׁבַע, יָצָא וְהִנִּיחוֹ עַל כַּן זָהָב שֶׁהָיָה בַּהֵיכָל.


שָׁב וְשָׂמוֹ בַכָּן / וְזָבַח שְׂעִיר עָם / וְקִבֵּל דָּמוֹ / וּבָא וְהִצְלִיף כְּדַרְכּוֹ

שִׁוָּהוּ בְכַנּוֹ / וְנָטַל דַּם פָּר / וְכֵן יַזֶּה בָאֹהֶל / מוּל אָרוֹן חוּצָה.


וְכָךְ הָיָה מוֹנֶה אַחַת, אַחַת וְאַחַת אַחַת וּשְׁתַּיִם אַחַת וְשָׁלוֹשׁ אַחַת וְאַרְבַּע אַחַת וְחָמֵשׁ אַחַת וָשֵׁשׁ אַחַת וָשֶׁבַע, יָצָא וְהִנִּיחוֹ עַל כַּן הַשֵּׁנִי שֶׁהָיָה בַּהֵיכָל.


תּוֹדָה נָתְנוּ שָׁב וַיְמִירֶנּוּ / בְּדַם הַשָּׂעִיר / וַיִּפְעַל בּוֹ / כְּמִשְׁפַּט דַּם הַפָּר

שָׁלְמוּ כַדָּת / סִדְרֵי הַזָּיוֹת פְּנִימָה / בְּלָלָם וְיָצָא / לְחַטֵּא מִזְבַּח הַקְּטֹרֶת.


שֶׁיַּתְחִיל מִקֶּרֶן / צָפוֹנָה מִזְרָחָה / וּבִמְקוֹם יִגְמוֹר / שָׁם יַזֶּה שֶׁבַע

שׁוֹפֵךְ שְׁאָר בַּחוּץ / בִּיסוֹד פְּנֵי פֶתַח / כִּי בָם יְכַפֵּר / עַל עֶזְרַת הַקֹּדֶשׁ.


וְאַחַר כָּךְ בָּא לוֹ אֵצֶל שָׂעִיר הַמִּשְׁתַּלֵּחַ וְסוֹמֵךְ שְׁתֵּי יָדָיו עָלָיו וְנִתְוַדָּה בַּעֲדֵינוּ סוֹמֵךְ שְׁתֵּי יָדָיו עָלָיו וּמִתְוַדֶּה בַּעֲדֵינוּ וְכָךְ הָיָה אוֹ׳ אָנָּא הַשֵּׁם חָטְאוּ עָווּ פָּשְׁעוּ לְפָנֶיךָ עַמְּךָ בֵּית יִשְׂרָאֵל אָנָּא הַשֵּׁם כַּפֶּר־נָא סְלַח נָא לַחֲטָאִים לַעֲווֹנוֹת וְלַפְּשָׁעִים שֶׁחָטְאוּ שֶׁעָווּ לְפָנֶיךָ עַמְּךָ בֵּית יִשְׂרָאֵל ככת׳ בְּתוֹרַת מֹשֶׁה עַבְדֶּךָ כִּי בַיּוֹם הַזֶּה יְכַפֵּר עֲלֵיכֶם לְטַהֵר אֶתְכֶם מִכֹּל חַטֹּאתֵיכֶם לִפְנֵי הַשֵּׁם תִּטְהָרוּ. וְהַכֹּהֲנִים וְהָעָם, הָעוֹמְדִים בַּעֲזָרָה וְהַמְשַׁמְּשִׁים בַּהֵיכָל, בִּזְמַן שֶׁשּׁוֹמְעִים אֶת הַשֵּׁם הַמְפוֹרָשׁ הַמְפוֹאָר שֶׁיּוֹצֵא מִפִּי כֹּהֵן גָּדוֹל בִּקְדֻשָּׁה וּבְטָהֳרָה הָיוּ כוֹרְעִים וּמִשְׁתַּחֲוִים וְנוֹפְלִים עַל פְּנֵיהֶם ואו׳ בָּרוּךְ שֵׁם כְּבוֹד מַלְכוּתוֹ לְעוֹלָם וָעֶד אַף הוּא מִתְכַוֵּן לִגְמוֹר אֶת הַשֵּׁם כְּנֶגֶד הַמְבָרְכִים וְאוֹמֵר לָהֶם: תִּטְהָרוּ.


תִּכֵּן אִישׁ / מוּעָד לְשַׁלְּחוֹ / וְעָלָה וְהִקְטִיר / חֶלְבֵי פָר וְשָׂעִיר

תְּקָפָם בְּמוֹטוֹת / חוּץ לְמַחֲנֵה עִיר / וּשְׂרָפוּם כָּלִיל / עוֹר וּבָשָׂר וָפֶרֶשׁ.


תָּאַר שָׂעִיר / לְרֹאשׁ הַמִּדְבָּר / מְשָׁכוּהוּ לְשַׁנֵּן / בְּחֻקֵּי עָשׂוֹר

תְּחִלָּה יְקַדֵּשׁ / יָדָיו וְרַגְלָיו / וְיִפְשׁוֹט כְּלֵי בָד / וְיַעַט בִּגְדֵי עָשְׁרוֹ.


תַּם הֶגְיוֹנוֹ / וּפָשַׁט וְטָבַל / וְעָטָה שְׁמוֹנָה / וְשָׁב וְנִתְקַדָּשׁ

תְּשׁוּרַת כְּבָשִׂים / וְאֵלִים הִכְלִיל / וְקִדֵּשׁ יָדָיו / וְרַגְלָיו וּפָשַׁט.


תִּרְגֵּל פְּעָמָיו / לְבֵית הַטְּבִילָה / וְטָבַל וְסִפֵּג / וְעָטָה כְלֵי בָד

תַּלְמוּדוֹ יַעַשׂ / כְּדָבְרוֹ, וְקִדַּשׁ / וְיוֹצִיא מִפְּנִימָה / כַּף וּמַחְתַּת פָּז.


תֹּכֶן עֲבוֹדַת / כְּלֵי בַד הִשְׁלִים / וְקִדַּשׁ וּפָשַׁט / וְהִנִּיחָם נֶצַח

תָּכַף טְבִילָה / לָבַשׁ כְּלֵי פָז / וְשָׁב וְקִדֵּשׁ / יָדָיו וְרַגְלָיו.


תְּמִיד הָעַרְבַּיִם / יַעֲשֶׂה כַסֵּדֶר / וְהִקְטִיר קְטֹרֶת / בֵּין נְתָחִים לָנָסֶךְ

תַּכְלִית מַעֲשִׂים / נֵרוֹת הִבְעִיר / פֵּירַשׂ כַּפַּיִם / וְקִדַּשׁ וּפָשַׁט.


תִּלְבֹּשֶׁת שְׁאֵרוֹ / חֲנִיכָיו יָרִיצוּ / לְבָשָׁם בְּכָבוֹד / וְיָצָא בְמָשׂוֹשׂ

תָּאֳרוֹ הֻקְרָן / כְּצֵאת הַשֶּׁמֶשׁ בִּגְבוּרָתוֹ / מְשַׁלֵּחַ לְשׁוֹלְחָיו / צְדָקָה וּמַרְפֵּא.


תִּקְוַת קָהָל / לְצֵאת אִישׁ מָהִיר / מְבַשֵּׂר יְשׁוּעוֹת / וּמוֹדִיעַ סְלוחוֹת

תְּהִלָּה נָתַן / לַשֵּׁם אֶל פְּנֵי עָם / בְּחָמֵשׁ טְבִילוֹת / וְעֶשֶׂר קְדֻשּׁוֹת.


תְּחִלָּה הִשְׁמִיעָם: / לְכוּ שְׁתוּ בְּלֶב טוֹב / אֵל נוֹשֵׂא עָווֹן / וְעוֹבֵר עַל פֶּשַׁע

תֵּן לָנוּ מוֹפֵת / בְּאָזְנָיו יֹאמֵרוּ, / בַּמֶּה נֵדַע / כִּי כֻפַּר עֲווֹנֵנוּ.


תִּתּוֹ צִיר שָׂעִיר / וְהוּא יְבַשֵּׂר סְלִיחָה, / וְהִלְבִּין כַּשֶּׁלֶג / אֹדֶם הַשָּׁנִי

תּוֹדָה נָתְנוּ / כִּי טוֹב הִשְׁמִיעָם / עַל צִיר אֱמוּנִים / הַמְשֻׁלָּח לַמִּדְבָּר.


תְּמִימִים יְשָׁרִים / לְנָוֵהוּ יְלַוּוּהוּ / וְשִׂמְחָה יַעֲשׂוּ / בְּצֵאתוֹ בְּלֹא פָגַע.

אַשְׁרֵי הָעָם / שֶׁכָּכָה לּוֹ / אַשְׁרֵי הָעָם / שֶׁיְיָ אֱלֹהָיו.
