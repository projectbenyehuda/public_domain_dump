כְּאִישׁ נִדְהָם

כתובת על השיר: משלי אהבים על הפלגת חשק המשכילים לחכמה ולתבונה.



כְּאִישׁ נִדְהָם וְנָד אָנֶה וְאָנָה / וְכַשּׁוֹכֵב בְּרֹאשׁ חִבֵּל סְפִינָה

בְּלֶב יָם סוֹעֲרָה מִן הַסְּעָרָה / וְלֹא יִשְׁקֹט הֲמוֹן גַּלֵּי שְׁאוֹנָהּ –

לְבָבִי יַהֲלֹךְ תֶּמֶס וְנָמוֹג / וְנַפְשִׁי בִי מְנַהֶמֶת כְּיוֹנָה,

עֲדֵי כִי יַחֲלֹף רוּחַ יְדִידוֹת – / צְבִיַּת חֵן אֲהוּבָה נַעֲמָנָה,

עֲדֵי רוּחַ אֲהָבֶיהָ תְהִי לִי / כְּמוֹ מָגֵן מְגִנָּה מִמְּגִנָּה,

בְּאוֹר לֶחְיָהּ אֲשֶׁר בּוֹשׁוּ לְנֶגְדָּהּ / פְּנֵי חַמָּה וְחָפְרָה הַלְּבָנָה,

בְּקֶסֶם עַל שְׂפָתֶיהָ וְעֵינָהּ / בְּצוּף וּדְבַשׁ אֲשֶׁר תַּחַת לְשׁוֹנָהּ.

יְרוֹת חִצֵּי אֲהָבִים לַאֲהוּבִים / בְּעַפְעַפָּהּ דְּמוּת קֶשֶׁת נְתוּנָה,

תְּלַבֵּב לֵב חֲשׁוּקֶיהָ וְתִקְשֹׁר / יְמִין עֻזָּם, וְחוּט שַׂעְרָהּ – אֲטוּנָהּ

עֲנָקִים לֹא יְיַפּוּ צַוְארוֹנָהּ / אֲבָל יַנְעִים רְבִידֶיהָ גְּרוֹנָהּ,

בְּלֶחְיָהּ שָׁלְחָה זֹהַר וְטֹהַר / וְאוֹר בָּהִיר וְצַח מֵאֵין עֲנָנָה.

יְפַת עַיִן, יְפַת תֹּאַר וּמַרְאֶה, / דְּמוּת מַרְאֵה חֲבַצֶּלֶת שְׁרוֹנָה,

וְשָׁדֶיהָ – דְּמוּת לוּחוֹת כְּתוּבִים / וּבָם מִצְוַת שְׁאֵר וּכְסוּת וְעוֹנָה.

יְאַו חוֹשְׁקָהּ נְתֹן פָּנָיו לְמִדְרַךְ / פְּעָמֶיהָ, וְלִבּוֹ – בֵּית מְלוֹנָהּ.

בְּתַבְנִיתָהּ וְקוֹמָתָהּ כְּמוֹתָהּ, / מְרֵעַי, לֹא רְאִיתֶם כָּל תְּמוּנָה!

״לְאֶנְקָתִי, צְבִיַּת חֵן, פְּנִי נָא / וְלֹא אָמוּת כְּמוֹת יַלְדֵי פְנִנָּה

וְנַחְמִי נָא לְנַפְשִׁי הָעֲנֻגָּה / וְצִיר לִבִּי וּמַכְאוֹב תִּרְפְּאִי נָא.

רְאִי נָא כִּי בְכָל לִבִּי דְרַשְׁתִּיךְ / וְחִבַּבְתִּיךְ כְּאֶלְקָנָה לְחַנָּה,

רְאִי נָא אֶת יְגוֹן רוּחִי וְשִׂיחִי / וְהַקְשִׁיבִי וְשִׁמְעִי זֹאת, עֲדִינָה!

עֲדִינָה, צִיץ צְבִי עֶדְיֵךְ עֲדִי נָא / וְאֶל עֵבֶר פְּנֵי דוֹדֵךְ עֲדִי נָא,

שְׂאִי נָא לִי שְׁלוֹם חֵן וַחֲנִינָה / וְתוֹךְ חַדְרֵי סְגוֹר לִבִּי חֲנִי נָא,

וּפִתְרִי סוֹד חֲזוֹן לַיִל חֲלַמְתִּיו / בְּשָׁכְבֵנוּ בְּעֶרֶשׂ רַעֲנָנָה:

נְשַׁקְתִּיךְ אָז וְחִבַּקְתִּיךְ – אֲבָל לֹא / אֲעַנֵּךְ, רַעֲיָה, כִּשְׁכֶם לְדִינָה.

וּמִי יִתֵּן בְּלַעְתִּנִּי, אֲבָל לֹא / תְקִיאִינִי כְּהָקִיא דָג לְיוֹנָה!

וְאִם אוּלַי מְקַנְאִים קִנְּאוּ בִי / וְתַחַת אַהֲבָה כָתְבוּ בְשִׂטְנָה –

אֲחוֹתִי, הַחֲרִישִׁי, אַל תְּגַלִּי, / צְפוּן חִבָּה בְּתוֹךְ חֻבֵּךְ טְמוּנָה

וְאֶל דּוֹדֵךְ תְּנִי דֹדֵךְ בְּעֵינֵךְ / וְרַב שָׁלוֹם כְּפַרְפַּר וַאֲבָנָה!״

הֱשִׁיבַתְנִי ״יְדִיד, מַה זֶּה תְּבַכֶּה / בְמִסְתָּרִים וְתָרִים קוֹל תְּלֻנָּה?

רְאֵה שֵׂיבָה בְךָ זָרְקָה, וְדַלָּה / שְׁחַרְחֹרֶת אֲשֶׁר אֲשֶׁר שָׁבָה לְבָנָה,

וְטַל יַלְדוּתְךָ חָלַף וְעָבַר / וְחֵשֶׁק בַּחֲרוּתְךָ סָר וּפָנָה.

הֲתִשְׁאַל אַחֲרֵי זֹקֶן נְעוּרִים / וְאִם אַחַר בְּלוֹת תָּשׁוּב לְעֶדְנָה?

וְגֶבֶר אֵין אֱיָל בּוֹ אֵיךְ יְמַלֵּא / שְׂחוֹק פִּיהוּ וּבִשְׂפָתָיו נְגִינָה?

בְּהֵיכְלֵי שֵׁן וְחַדְרֵי יַעֲלוֹת חֵן / הֲיָבוֹא שָׂב וּבִיאָה זוֹ מְשֻׁנָּה!

וּמַה יּוֹסִיף לְאוֹר עִוֵּר מְאוֹר נֵר? / וּמַה יִתֵּן לְסָרִיס בֵּית חֲתֻנָּה?

עֲזֹב, זָקֵן, עֲזֹב שִׁירֵי עֲגָבִים / וּפֶן תִּהְיֶה לְמָשָׁל גַּם שְׁנִינָה,

זְכֹר, זָקֵן, יְמוֹת עוֹלָם וְעַד אָן / יְחִידָתְךָ בְּחֵיק חֵשֶׁק אֱמוּנָה?

זְכֹר, יָשִׁישׁ, גְּוִיָּתְךָ כְּתַרְשִׁישׁ / לְאֵיד וּלְמוֹעֲדֵי רֶגֶל נְכוֹנָה.

בְּטֶרֶם הַזְּמָן יִבְגֹּד בְּךָ, קוּם / וְתִבְגֹּד בּוֹ כְּמוֹ רֵכָב וּבַעְנָה,

נְתֹץ חַיִץ אֲשֶׁר בָּנוּ יְדֵי חֵטְא / בְּיִרְאַת חֵטְא וּבִתְשׁוּבָה הֲגוּנָה,

לְהִתְיַחֵד וְהִתְבּוֹדֵד בְּנֵה לָךְ / לְהִתְפַּלֵּל עֲלִיַּת קִיר קְטַנָּה.

שְׁאַל, נוֹאָל, מְחוֹת חַטַּאת נְעוּרִים / שְׁאַל וּדְרֹשׁ לְאֵל שׁוֹכֵן מְעוֹנָה,

וְדַי לָךְ כִּי תְבַקֵּשׁ לַעֲווֹנְךָ / נְתֹן כֹּפֶר בְּכָל רֶגַע וְעוֹנָה.

עֲזָב נָא לָעֲלָמִים חֵן עֲלָמוֹת / לְזַמֵּר עַל עֲלָמוֹת שִׁיר וְרִנָּה!

הֲיִשְׁמַע שָׂב בְּקוֹל שָׁרִים וְשָׁרוֹת / וְאִם תָּבוֹא בְּפִי יָשִׁישׁ רְנָנָה?

וּמֶתֶק צוּף לְחִכּוֹ – רוֹשׁ וְלַעֲנָה, / וְרֵיחַ מֹר בְּאַפּוֹ הוּא כְּצַחְנָה!

יְגוֹנָיו עֵת זְקוּנָיו לוֹ לְמָחוֹל / וְשִׁיר מִזְמוֹר עֲלוּמָיו לוֹ לְקִינָה״.

עֲנִיתִיהָ ״כְּלִילַת חֵן וְיֹפִי / וְשֵׂכֶל טוֹב בְּלֶקַח טוֹב וּבִינָה,

וּמַה נֵּחַנְתְּ וּמַה נָּעַמְתְּ וְנָאַמְתְּ / נְאוּם – כָּל מַעֲשֵׂהוּ בֶאֱמוּנָה!

יְכָלְתִנִּי, נְצַחְתִּנִּי בְּמוֹפְתֵי / שְׂפַת יֶתֶר, שְׂאֵת סִבָּה וְטַעְנָה,

וְשׁוּב שִׂכְלֵךְ בְּחָכְמָתֵךְ וְדַעְתֵּךְ / עֲצַת שָׁלוֹם חֲקָרָהּ גַּם הֱכִינָהּ

לְדַעַת סוֹד בְּקִיר לִבִּי כְמַסְתִּיר / וּמַחְשָׁבָה בְּמוֹרָשָׁיו צְפוּנָה.

אֲנִי הָאִישׁ אֲשֶׁר חֶרְפַּת נְעוּרָיו / סְבָבַתְהוּ כְנָחָשׁ וַעֲכָנָה,

עֲווֹן בִּצְעָהּ וְרֹב רִשְׁעָהּ וּפִשְׁעָהּ / אֲפָפוּהוּ – וְנָשָׂא אֶת עֲווֹנָהּ,

וְגָדַר הַזְּמָן דַּרְכִּי עֲדֵי כִי / הֱעִירַנִי, דְּבַר שָׁנָה בְּשָׁנָה,

עֲדֵי שַׂבְתִּי וְזָקַנְתִּי וְשַׁבְתִּי / וְנִחַמְתִּי לְשֹׁד נַפְשִׁי וְאוֹנָהּ

וְנִחַמְתִּי וְנִכְלַמְתִּי וּבֹשְׁתִּי / וְלָבַשְׁתִּי כְסוּת בּוּשָׁה וּבָשְׁנָה.

וְיָדַעְתִּי אֲנִי נֶפֶשׁ בְּהֶמְתִּי / בְּמַשְׂאוֹת שָׁוְא וּמַדּוּחִים טְעוּנָה,

וְהִלְבַּשְׁתִּי לְרוּחִי סוּת חֲרִיצוּת, / בְּטֶרֶם אֶל שְׁאוֹל יֵרֵד הֲמוֹנָהּ,

וּבִינוֹתִי בְּסוֹד מֶחְקָר וְעִקַּר / יְקָר נִזְכָּר בְּעֵדוּת נֶאֱמָנָה –

יְקָר נִזְכָּר בְּפִי כַמָּה חֲכָמִים / מְחֻכָּמִים בְּסוֹד מִקְרָא וּמִשְׁנָה.

יְעָצוּנִי סְעִפַּי: צֹר תְּעוּדוֹת, / נְצֹר סוֹדוֹת יְסוֹדוֹת הָאֱמוּנָה,

וּבָם נוֹחַ לְכָל נֶפֶשׁ וְנֹפֶשׁ, / וּבָם תִּזְכֶּה בְּמִשְׁפָּטָהּ וְדִינָהּ!

וְהִתְקַדֵּשׁ – בְּקוּם שַׁדַּי – לְמַדַּי, / בְּיוֹם מִשְׁפָּט לְבַקֵּשׁ חִשְּׁבוֹנָהּ,

וּבָם תִּבְחַר בְּכָל מִדּוֹת כְּבֻדּוֹת / וְעָזוֹב תַּעֲזֹב מִדָּה מְגֻנָּה.

וְלִי מַה יָּקְרוּ רֵעַי – וְאֶחְשֹׁק / הֱיוֹת דָּתִי לְבַדָּהּ לִי לְמָנָה,

וְכָל יִשְׁעִי וְכָל חֶפְצִי לְחֶפְצָהּ / לְהִתְמַכֵּר וְלִהְיוֹת לָהּ לְמִקְנָה.

וְאוּלָם בִּי זְמִיר אַלּוּף נְעוּרַי / וְנָחַנִי עֲדֵי שֵׂיבָה וְזִקְנָה

וְהִפְקִיד הַזְּמָן עֵטִי וּמִנָּה / לְהָשִׁיב הוֹד עֲטֶרֶת שִׁיר לְיָשְׁנָהּ.

בְּמִשְׁפַּט הַזְּמִיר כִּלְכֵּל דְּבָרָיו / וּמַאְמָרָיו בְּשִׁיר כּוֹרְתִים אֲמָנָה,

מְשֹׁל מָשָׁל בְּכָל הֶגְיוֹן לְשׁוֹנוֹ – / בְּרִית כָּרַת, תְּנַאי קִיֵּם וְהִתְנָה,

וְאוֹתוֹתָיו וְתֵבוֹתָיו בְּפֶלֶס / וּמֹאזְנֵי הַזְּמִיר שָׁקַל וּמָנָה.

בְּכֵן כִנָּה תְעוּדָתִי כְעַלְמָה / יְפֵהפִיָּה בְּעֶדְיֵי שַׂהֲרוֹנָהּ,

וְחֻקִּים הַחֲקוּקִים בָּהּ – כְּעָגִיל / עֲלֵי אָזְנָהּ וְחוֹתָם עַל יְמִינָהּ,

כְּעַלְמַת חֵן תְּשַׂמַּח לֵב יְדִידָהּ / בְּקָרְצָהּ לוֹ בְּאִישׁוֹנָהּ וְעֵינָהּ,

וּפַעַם חֵץ תְּשׁוּעָה תּוֹר בְּעֵינָהּ / וּפַעַם מַדְקְרוֹת חֶרֶב שְׁנוּנָה –

וְכֵן דָּת אֵל תְּכַבֵּד אֶת דְּבֵקָהּ / וְתִשְׁפֹּךְ עַל רְחֵקָהּ בּוּז קְלוֹנָהּ,

וְלֹא יִגְוַע אֲשֶׁר יַשִּׂיג רְצוֹנָהּ / וְלֹא יִחְיֶה כְּרֶגַע בַּחֲרוֹנָהּ.

וְנִכְסַפְתִּי לְהִתְרַפֵּק בְּדוֹדָה / כְּחוֹשֵׁק יַעֲלַת חֵן וַחֲנִינָה,

וְאֶחְפֹּץ כָּל אֲשֶׁר תִּבְחַר וְתַחְפֹּץ / וְאֶנְאַץ כֹּל אֲשֶׁר תִּנְאַץ וְתִשְׂנָא.

אֱלֹהַי, קַח לְךָ רוּחִי בְקִרְבִּי / נְתוּנָה לָךְ בְּמַתָּנָה וְאֶתְנָה!

תְּנָה לִי יוֹם קְרָב חֶטְאָהּ מְנָתְךָ / וְחֶמְלָתְךָ כְּלִי נִשְׁקָהּ וְזֵינָהּ,

וְלֹא יַבְדִּיל עֲווֹן סִכְלָהּ וְכִסְלָהּ / וּמַעַל מָעֲלָה בֵּינְךָ לְבֵינָהּ.

כְּמוֹ כָלְתָה לְךָ וּלְדָתְךָ – כֵּן / יְכַלֶּה חַסְדְּךָ עֹצֶם זְדוֹנָהּ,

כְּמוֹ חָשְׁקָה בְךָ – תַּחְשֹׁק, הֲשִׁיבָהּ / לְאֵיתָנָהּ אֲשֶׁר הָיְתָה בְאָמְנָה,

וְיוֹם תִּשְׁפֹּךְ לְךָ שִׂיחַ לְשׁוֹנָהּ – / עֲנֵה טֶרֶם קְרֹא ״אַחְלַי וְאָנָּא״

וְהֵעָתֵר לְקוֹל זַעְקָהּ כְּמֵאָז / לְבֶן-יִשַׁי בְּגֹרֶן הָאֲרַוְנָה,

וְאַל תִּגְמֹל כְּמִפְעָלָהּ, וְתֹאכַל / פְּרִי יָדָהּ כְּמוֹ נוֹצֵר תְּאֵנָה.

וְאִם אוּלַי פְּשָׁעִים הַנְּטוּעִים / בְּמֶזֶג כָּל יְצוּר אוֹ בַתְּכוּנָה

וְאוּלַי מִפְּנֵי הֶרְגֵּל עֲבֵרָה / וְהַמִּנְהָג כְּמִנְהַג הַמְּדִינָה

עֲכָרוּהָ מְכָרוּהָ בְּיַד צַר / בְּיַד יַלְדּוּת וּמָסְכוּ רוֹשׁ בְּיֵינָהּ

וּמַאְכָלָהּ חֲסַר מֶלַח וְתַבְלִין / וְלֹא נִתַּן בְּמִנְחָתָהּ לְבוֹנָה

וְתַבְשִׁילָהּ כְּמִצְטַמֵּק וְרַע לוֹ / עֲדֵי שֻׁלַּח בְּמִשְׁמַנָּהּ רְזוֹנָהּ

וְהָעוֹלָם בְּמַצְרֵפוֹ צְרָפָהּ / וְתוֹךְ מִבְחָן בְּכוּר עֹנִי בְּחָנָהּ –

בְּכָל זֹאת לֹא הֱנִיאוּהָ בְנֵי יוֹם / וְלֹא רִמָּה זְּמָן אוֹתָהּ וְהוֹנָה,

וְלֹא הִפִּיל וְהִשְׁפִּיל רוּם גְּאוֹנָהּ / וְלֹא עִנָּה עֳנִי כֹחָהּ וְאוֹנָהּ,

וְתֵבֵל לֹא תְסִיתֶהָ בְּשִׂפְקָהּ / בְּמַמְתַקָּהּ וְצוּף חִכָּהּ וְהוֹנָהּ –

אֲבָל דָּבְקָה בְצוּר חִזְקָהּ וְחִשְׂקָהּ, / וּמִשְׁעָנָהּ וּבִטְחוֹנָהּ – בְּקוֹנָהּ.

הֱיֵה לָהּ בֵּית מְצוּדוֹת מִמְּצוּדוֹת / לְרַגְלֶיהָ זְמָן הֵכִין וְאִנָּה,

וּמִשּׁוֹאוֹת וּמַשּׁוּאוֹת פְּלָאוֹת / וְנוֹרָאוֹת בְּצֶדֶק תַּעֲנֶנָּה,

וּמִשָּׁאוֹן וּמַשָּׁאוֹן וְגָלוּת / וּבוֹר דַּלּוּת וְשִׁפְלוּת תַּעֲלֶנָּה,

וְעֵינֶיהָ מְשִׁיחַ אֵל וְגוֹאֵל / לְיִשְׂרָאֵל מְהֵרָה תֶחֱזֶינָה,

וְתַעַל בַּעֲלוֹת אוֹבְדִים שְׁדוּדִים / לְהַר הַמֹּר וְגִבְעַת הַלְּבוֹנָה.

וְאַחְרִיתָהּ תְּהִי עִם דּוֹר יְשָׁרִים / מְתִים מֵתוּ וְהֵם חַיִּים עֲדֶנָה,

תְּשִׂימֶהָ בְאַפַּדְנָהּ וְכַנָּהּ / וְשָׂא נִסָּהּ בְּפַרְדֵּסָהּ וְגַנָּה

וְשִׂים שִׂיאָהּ עֲלֵי גַבְהוּת נְשִׂיאָהּ / וְתֵן כִּסְאָהּ בְּכֵס זִקְנֵי כְהֻנָּה.

וְעֵת תָּשִׁיב נְשָׁמוֹת אַחֲרֵי מוֹת / לְתוֹךְ גֵּוָה בְּגֵו עָפָר יְשֵׁנָה –

תְּקִיצֵנִי לְקֵץ יָמִין, וְתַעְטִיר / רְצוֹנָהּ לִי כְּסוֹחֵרָה וְצִנָּה.

וְלִי יֶעְרַב בְּיוֹם שָׁרָב כְּעָב טַל / וְכִמְדוּרָה – בְּיוֹם קָרַָה וְצִנָּה.

וְאוּלָם הֵן בְּעוֹדִי חַי אֲשַׁחֵר / פְּנֵי חַסְדְּךָ וְאַפִּיל לָךְ תְּחִִנָּה

לְהַסְפִּיק מַחֲסוֹר נַפְשִׁי בְּטוּבְךָ / וְלֹא יֶחְסַר מְזוֹנָהּ כָּל זְמַנָּהּ,

וְשִׁפְעַת חַסְדְּךָ תָּכִין דְּגָנָהּ / וְשֵׂיבָתִי תְכַלְכֵּל תּוֹךְ מְכוֹנָהּ

וְגֵוִי לֹא יְמֵי חֶדְלוֹ יְהַלֵּךְ / כְּאִישׁ נִדְהָם וְנָד אָנֶה וְאָנָה.
