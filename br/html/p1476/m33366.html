אָמְרוּ כִּי הָיָה אִישׁ אֶחָד בְּאֶרֶץ תּוֹגַרְמָה

מתוך השער השלישי של משל הקדמוני


אָמְרוּ כִּי הָיָה אִישׁ אֶחָד בְּאֶרֶץ תּוֹגַרְמָה /

הָיָה עוֹבֵד אֲדָמָה, / וַיְהִי שׁוֹכֵן בִּכְפָרִים / וּבָתֵּי

הַחֲצֵרִים, / וְהָיָה מִתְפַּתֶּה וּמִתְהוֹלֵל בַּעֲמָלוֹ / אָדָם אֲשֶׁר

לֹא יָדַע בֵּין יְמִינוֹ לִשְׂמֹאלוֹ, / וּבֵינוֹ וּבֵין עִיר

הַסּוֹחֲרִים הָרָמִים / מַהֲלָךְ שְׁלֹשֶׁת יָמִים. / וְשָׁנָה אַחַת

גִּדַּל בְּבֵיתוֹ זְרָעִים / לַעֲבוֹדַת הַמֶּשִׁי יְדוּעִים / וַיָּרֻם

תּוֹלָעִים, / וַתְּהִי לְחֶפְצוֹ הָעֲבוֹדָה נִמְשָׁכָה / וַתִּשְׁלַם

כָּל הַמְּלָאכָה.

וַיָּקָם יוֹם אֶחָד לָלֶכֶת בְּחָפְזָה / אֶל הָעִיר

הָעַלִּיזָה / לִמְכֹּר הַמֶּשִׁי הַגָּדֵל בְּהֵיכָלוֹ / לְהַרְבּוֹת

חֵילוֹ. / וַיְמַהֵר לְהָפִיק רְצוֹנוֹ, / וַיַּחֲבֹשׁ אֶת אֲתוֹנוֹ, /

וַיָּשִׂים אֶת הַמֶּשִׁי בָּאַמְתַּחַת / בְּשׁוּבָה וָנַחַת, / וַיִּתֵּן

אֲבָנִים בַּצַּד הָאַחֵר / וּלְמִשְׁקַל הַמֶּשִׁי אוֹתָם בּוֹחֵר, /

וַיִּקַּח צֵידָה לִמְזוֹנוֹ / וַיַעֲמֹס בַּשָּׁוֶה עַל אֲתוֹנוֹ. /

וַיְצַו לְבֵיתוֹ וַיֵּלֶךְ / דֶּרֶךְ הַמֶּלֶךְ. / וַיְהִי הוּא הוֹלֵךְ

לְבַדּוֹ – / וְהִנֵּה אִישׁ אֶחָד קָרֵב לְנֶגְדּוֹ, / וְהוּא עָנִי

מִסְכֵּן, / בְּנַפְשׁוֹ מִסְתַּכֵּן. / וַיִּירָא מְאֹד הָעוֹבֵד / כִּי

רָאָה אוֹתוֹ מִבְּלִי לְבוּשׁ אוֹבֵד / וַיֹּאמֶר אֶל הָאִישׁ ״הֲשָׁלוֹם?

/ גֶּשׁ נָא הֲלֹם!״ / וַיֹּאמֶר לוֹ הָאִישׁ ״אַל תִּירָא מִפָּנַי / כִּי

שָׁלוֹם לְךָ וְאֵין דָּבָר, חַי יְיָ!״


וַיֹּאמֶר הָעוֹבֵד ״לְאֵיזֶה מָקוֹם מְגַמָּתֶךָ? /

אוּלַי אוּכַל לָלֶכֶת בְּחֶבְרָתֶךָ״. / וַיֹּאמֶר ״אֶל הַסְּחוֹרָה

מְגַמָּתִי / וּלְפִי מִשְׁקַל הַמֶּשִׁי תַּחֲנוֹתִי״. / וַיִּשְׂמַח

הָעוֹבֵד עַל הַבְּשׂוֹרָה / וְאָרַח עִמּוֹ לְחֶבְרָה. / וַיֹּאמֶר לוֹ

״הִנְנִי בְחֶבְרָתְךָ בוֹטֵחַ / וּבְאַהֲבָתְךָ שָׂמֵחַ. / וְעַתָּה

הוֹדִיעֵנִי מְלַאכְתְּךָ וְעִנְיָנֶיךָ, / אִם נָא מָצָאתִי חֵן

בְּעֵינֶיךָ״. / וַיֹּאמֶר הָאִישׁ ״מִנְּעוּרַי אֲנִי שׁוֹכֵן רָמִים /

עִיר גְּדוֹלָה שֶׁל חֲכָמִים / וְשָׁם הֲכִינוֹתִי מְנוּחָתִי

וְשִׁבְתִּי, / וַאֲנִי סוֹפֵר מָהִיר בִּמְלַאכְתִּי, / וְיֶשׁ לִי

בְתוֹגַרְמָה אָח וְקָרוֹב / וְרֵעִים לָרֹב: / וְהִנְנִי שׁוֹחֵר

לִקְרָאתָם / לִדְרֹשׁ שְׁלוֹמָם וְטוֹבָתָם״. / וַיֹּאמֶר ״הוֹדִיעֵנִי

דָבָר בִּמְלַאכְתְּךָ הָרָמָה / וְיַעֲבֹר עָלַי מָה…

וַיֹּאמֶר הָעוֹבֵד ״רְאֵה דְבָרֶיךָ טוֹבִים וּנְכוֹחִים

בְּמִלּוּלָךְ / וְשׁוֹמֵעַ אֵין לָךְ! / כִּי הֶעָמָל הִפְלִיאַנִי

הַפְלֵא וָפֶלֶא / וְלֹא נִסִּיתִי לָלֶכֶת בָּאֵלֶּה, / וְאֵינִי מַכִּיר

בְּדִבּוּרְךָ לְבַד חִתּוּךְ שְׂפָתַיִם / פַּעַם וּשְׁתַיִם״. / וַיְהִי

הֵם הוֹלְכִים הָלוֹךְ וְדַבֵּר / וְכָל אֶחָד בַּחֲבֵרוֹ מִתְחַבֵּר – /

וַתִּצְנַח הָאָתוֹן וַתִּכְרַע / וַיֶּחֱלַשׁ כֹּחָהּ וַיִּגָּרַע, / כִּי

לֹא יָכְלָה לִסְבֹּל הַמַּשָּׂא. / וְעֵין הָעוֹבֵד עָלֶיהָ חָסָה /

וַיִּצְעַק וַיֶּהֱמֶה לְקוֹלָהּ / וַיָשֶׁת לִבּוֹ לְחִילָהּ.


וַיֹּאמֶר הַסּוֹפֵר ״לָמָּה תִּשּׁוֹמֵם, / חָרֵד

וְהוֹמֵם? / קְרַב אֶל הָאָתוֹן וְהָסֵר הַמַּשָּׂא אֲשֶׁר עָלֶיהָ – /

וְתָנוּחַ מְעַט מִזְעָר מֵחֲבָלֶיהָ״. / וַיִּגַּשׁ הָעוֹבֵד לְהָסִיר

הַמַּשָּׂא מֵעַל אֲתוֹנוֹ / וְהַסּוֹפֵר עֲזָרוֹ כְּפִי כֹחוֹ וְאוֹנוֹ. /

וְכִרְאוֹת בְּאַמְתַּחְתּוֹ אֶבֶן מַעֲמָסָה / וַיֹּאמֶר לוֹ ״מָה

הַמַּשָּׂא?״ / וַיֹּאמַר ״אַמְתַּחְתִּי הָאֶחָד מָלֵא מֵהַמֶּשִׁי הַקַּל

– / וְהָאֲבָנִים לְנֶגְדּוֹ לְמִדָּה וּלְמִשְׁקָל״. / וַיֹּאמֶר

הַסּוֹפֵר ״לָמָּה מַחְשְׁבוֹתֶיךָ חֲפֵצוֹת / בְּאַבְנֵי גִּיר

מְנֻפָּצוֹת?״ / וַיֹּאמֶר הָעוֹבֵד ״וְאֵיךְ יִנָּשֵׂא הַמֶּשִׁי לְבַדּוֹ

/ וְאֵין מִשְׁקָל לְנֶגְדּוֹ?״ / וַיֹּאמַר ״בְּעִנְיַן הַמַּשָּׂא

אֲלַמֶּדְךָ דְּבָרִים / שְׁמַע בְּקוֹלִי אִיעָצְךָ מֵישָׁרִים״. /

וַיְצַו לְהָרִים הָאֲבָנִים מִן הָאַמְתַּחַת / וּלְהַצִּיל נֶפֶשׁ

בְּהֶמְתּוֹ מִנִּי שַׁחַת / וּלְחַלֵּק הַמֶּשִׁי בְּאַמְתְּחוֹתָיו /

מִשְּׁתֵי רוּחוֹתָיו. / וַיַּעַשׂ הָעוֹבֵד כְּפִי מַאֲמָרוֹ וּרְצוֹנוֹ /

וַיֵּקַל הַמַּשָּׂא מֵעַל אֲתוֹנוֹ.

וַיָּקוּמוּ וַיֵּלְכוּ שְׁנֵיהֶם חֲבֵרִים / וְהָאָתוֹן

לִפְנֵיהֶם מַעֲבִירִים. / וְהַסּוֹפֵר הִשְׁלִים חִידָתוֹ הַנִּפְלָאָה /

לְהַעֲבִיר מֵהֶם יָגוֹן וּתְלָאָה, / וְהָעוֹבֵד תָּמַהּ מְאֹד

מֵחִדּוּדוֹ / וְגֹדֶל תְּבוּנָתוֹ וְהוֹדוֹ / וַיֹּאמַר ״אַחֲרֵי אֲשֶׁר

הוֹד תְּבוּנַת לְשׁוֹנְךָ גָדַל / מַדּוּעַ אַתָּה כָּכָה דַּל?״ /

וַיֹּאמֶר הַסּוֹפֵר ״וַהֲלֹא חֲכָמִים עָשׂוּ תַעֲנִיּוֹת וּצְדָקוֹת /

עַל הַסּוֹפְרִים וְעַל מְלַמְּדֵי תִינוֹקוֹת / שֶׁלֹּא יִתְעַשְּׁרוּ

בִמְלַאכְתָּם, / וְכֵן נִכְתַּב וְנֶחְתַּם! / וַעֲלֵיהֶם אָמְרוּ: נָאָה

עֲנִיּוּת לְיִשְׂרָאֵל וְצַעֲרָא / כְּוַרְדָא סוּמְקָא לְסוּסְיָא

חִוְרָא״. / וַיֹּאמֶר הָעוֹבֵד ״הוֹאִיל וְזֹאת הִיא מַחֲלָתֶךָ – /

הָעֲבוֹדָה! לֹא אֲקַבֵּל עֲצָתֶךָ / וְלֹא אֶתְחַבֵּר עוֹד לְחֶבְרָתֶךָ.

/ וְעַתָּה הִפָּרֵד נָא מֵעָלַי / אַל תּוֹסֶף עוֹד דַּבֵּר אֵלַי /

וְלֵךְ לְשָׁלוֹם דֶּרֶךְ מְגַמָּתְךָ עֲרוּכָה – / וַאֲנִי אֶתְנַהֲלָה

לְאִטִּי לְרֶגֶל הַמְּלָאכָה״. / וַיֵּלֶךְ הָאִישׁ לְבַדּוֹ אֹרַח

נְתִיבוֹ וַיִּתְפָּרְדוּ אִישׁ מֵעַל אוֹהֲבוֹ.

וַיָּשָׁב הָעוֹבֵד וַיִּקַּח אֶת הָאֲבָנִים אֲשֶׁר

הִנִּיחַ / וְנֶגֶד עֲצַת הַסּוֹפֵר דְּבָרִים הִטִּיחַ / לְלַקֵּט אוֹתָם

בְּמִסְפָּר / מַעֲרֵמַת הֶעָפָר / וַיִּתֵּן אוֹתָם בְּאַמְתַּחְתּוֹ /

אֲשֶׁר נִמְצָא אִתּוֹ / וּבְמִשְׁקַל הַמֶּשִׁי אוֹתָם סִלָּה / כַּאֲשֶׁר

הָיָה בַתְּחִלָּה. / וַיָּקָם וַיֵּלֶךְ לְבַדּוֹ מְעַט מְעַט / לְרֶגֶל

הָאָתוֹן לְאַט, / וַיַּעַשׂ בַּדֶּרֶךְ שְׁלֹשָׁה יָמִים / וּבָרָק

מִמְּרוֹרָתוֹ יַהֲלֹךְ עָלָיו אֵימִים. / וַיַּגִּיעַ אֶל הָעִיר יָגֵעַ,

/ עֹנִי וְעָמָל שָׂבֵעַ. / וַיֵּלֶךְ בַּלַּיְלָה הַהוּא בִמְלוֹנוֹ /

וַיַּאֲכֵל לַאֲתוֹנוֹ.

וַיְהִי בַבֹּקֶר וַיָּבוֹא הַסּוֹחֵר, וַיִּקְנֶה

הַמֶּשִׁי מִיָּדוֹ / וַיִּשְׁקֹל הַכֶּסֶף בַּעֲדוֹ / וַיֹּאמַר ״יֵשׁ

עוֹד בְּאַמְתַּחְתְּךָ דָּבָר לְמָכְרוֹ? / וַאֲנִי אֶתֵּן שְׂכָרוֹ /

וְאַרְבֶּה כֶּסֶף שִׁבְרוֹ״. / וַיֹּאמַר ״אֵין בּוֹ לְבַד דָּבָר נָקַל –

/ אַבְנֵי מִשְׁקָל / הֵבֵאתִי לְיַשֵּׁר הָמַּשָּׂא, / אֶבֶן מַעֲמָסָה״.

/ וַיִּגַּשׁ הַסּוֹחֵר לִרְאוֹתָם / וְהִנֵּה הֵם שׁוֹנוֹת בְּמַרְאִיתָם!

/ וַיָּבֶן כִּי הֵם טוֹבוֹת / לְחַדֵּד וְלִמְרֹט הַחֲרָבוֹת. /

וְהַמֶּלֶךְ צִוָּה עַל כָּל בְּנֵי הָעִיר / יַחַד יָשִׁישׁ וְצָעִיר /

לְתַקֵּן כָּל כְּלֵי מִלְחָמָה וּלְחַדְּדָם / לְהַרְאוֹת הֲדָרָם

וְהוֹדָם / וּלְתַקֵּן כָּל נֶפֶשׁ אֲשֶׁר לְבוּשׁ שִׁרְיוֹן עוֹטָה /

חֶרֶב חֲדוּדַָה וְגַם מֹרָטָּה, / וְכָל הָאֲנָשִׁים אֲשֶׁר לֹא יַרְאוּהָ

בִזְמָן קָבוּעַ / חֶרְפָּתוֹ וְעָנְשׁוֹ יָדוּעַ, / כִּי כָל חֹשֶׁךְ

לִצְפוּנָיו טָמוּן / וּבָתֵיהֶם יִתְשָׂמוּן. / וּבִרְאוֹת הָאִישׁ

הָאֲבָנִים הַסְּפוּרוֹת / אֲשֶׁר הָיוּ מִזְּמָן רַב נֶעְדָּרוֹת /

וּבְעֵת הַצֹּרֶךְ הֵמָּה נִסְתָּרוֹת, / אֲבָנִים גְּדוֹלוֹת אֲבָנִים

יְקָרוֹת ־ ־ / דִּבֶּר אֶל הָעוֹבֵד דְּבָרִים טוֹבִים / אֶל לֵב כָּל

מַשְׂכִּיל קְרוֹבִים / וַיֹּאמַר ״נָקְבָה אֵלַי בְּאֵלֶּה הָאֲבָנִים

מְחִירֶךָ / וַאֲנִי אֶתֵּן אֶת שְׂכָרֶךָ״. / וַיְהִי בְעֵינֵי הָעוֹבֵד

לִשְׂחוֹק / וַיָּבִיא עֵצוֹת מֵרָחוֹק / וַיֹּאמַר ״אֲנִי אֶמְכְּרֵם

בַּעֲבוּר נַעֲלַיִם / וּבִמְחִיר הַמֶּשִׁי כִּפְלַיִם״. / וַיֹּאמֶר

הַסּוֹחֵר ״אֲשַׁלֵּם לְךָ כָּל אֲשֶׁר תִּשְׁאַל / וְאַל תָּשִׂים

מִלָּתְךָ לְאַל״. / וַיִּקַּח הָאֲבָנִים וַיּוֹצִיאֵם לַחוּץ, / כִּי

הָיָה דְבַר הַמֶּלֶךְ נָחוּץ, / וַיִּתֵּן לָעוֹבֵד כְּסוּתוֹ וּמִנְעָלוֹ

/ וְכֶסֶף הִרְבָּה לוֹ. / וַיִּשְׂמַח הָעוֹבֵד בְּסִכְלוֹ וַיִּתְהַלָּל

/ כִּי הוֹלִיךְ יוֹעֲצִים שׁוֹלָל / וְסוֹפְרִים יְהוֹלָל / וַיֹּאמַר

״זֶה תַגְמוּל כָּל עָמֵל וְכָל רוֹעֵץ / אֲשֶׁר לֹא שָׁמַע לְקוֹל

יוֹעֵץ״. / וַיָּשָׁב לִבּוֹ שָׂמֵחַ / וּבְהוֹלְלוּתוֹ מִשְׁתַּבֵּחַ /

וַיְהִי לוֹ מִקְנֵה צֹאן וּמִקְנֵה בָקָר / שִׂמְחָה וְשָׂשׂוֹן

וִיקָר.
