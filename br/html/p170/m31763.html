עַיִן אֲשֶׁר מֵהַבְּכִי נִמֶּקֶת

על מות בת אחיו

עַיִן אֲשֶׁר מֵהַבְּכִי נִמֶּקֶת / מָתַי תְּהִי נָחָה וּמִתְאַפֶּקֶת,

מִכְלַל חֲפָצֶיהָ זְמָן יַרְחִיק אֲבָל / יַגִּישׁ מְצוֹקֶיהָ וְהִיא מוֹצֶקֶת,

מֵאֵין דְּלִי תִשְׁאַב וְתַשְׁקֶה פַאֲתֵי / פָנַי וְדָם עִם הַבְּכִי זוֹרֶקֶת.

אִמְרוּ: הֲבֵין מֵעַי אֲפִיקֵי יָם, וְאִם / קִרְבִּי מְקוֹר מַיִם וְרֹאשִׁי שֹׁקֶת?

אוֹ אֵיךְ צְלָעַי יִשְׂרְפוּ מֵי נַחֲלֵי / עֵינַי וְדִמְעָתִי כְאֵשׁ דּוֹלֶקֶת?

יָדִי עֲלֵי אַנְחַת לְבָבִי כָּבְדָה / עַד רָפְתָה מִשֵּׂאת וְהִיא מַחְזֶקֶת,

אֶאְנֹק וְאֵאָנַח בְּשִׁבְרוֹן לֵב עֲדֵי / אֶצְרַח: אֲהָהּ! כִּי נִשְׁבְּרָה מִפְרֶקֶת.

אֶבְכֶּה וְאֶחָנֵק בְּדִמְעָתִי עֲדֵי / כִמְעַט תְּהִי בוֹ מִלְּבַד נֶחְנֶקֶת.

תֵּבֵל עֲדֵי מַה תַּעֲשֶׂה תֶּבֶל, וְעַד / אָנָה לְשַׁחֵת הַיְקוּם מַעְמֶקֶת?

תּוֹלִיד וְתַשְׁמִיד, עַד יְדַמֶּה הָאֱנוֹשׁ / כִּי הִיא מְהוֹלֶלֶת וְכִמְצַחֶקֶת.

תִּתֵּן וְתָבֹז, עַד מְלִיצִים בָּעֲרוּ: / אִם הִיא מְכַזֶּבֶת וְאִם מַצְדֶקֶת?

חִצֵּי נְדֹד תָּבֵר בְּהִתְחַבֵּר וְצוּר / שֶׁלַח לְהָמִית בַּחֲיוֹת מוֹרֶקֶת,

לִבָּהּ בְּהַתְאִים הַיְּגוֹנִים יַעֲלֹז, / נַפְשָׁהּ בְּהַפְרִיד אֹהֲבִים חוֹשֶׁקֶת.

שָׁלוֹם וּמִלְחָמָה חֲלִיפוֹת הַיְקוּם / מֵאָז, וְאַךְ הִיא לַקְּרָב נוֹשֶׁקֶת,

תֹּאכַל וְלֹא תִשְׂבַּע וְאֶל הַשּׁוֹדְדִים / עַיִן כְּצָר תִּלְטֹשׁ וְשֵׁן חוֹרֶקֶת,

תַּמְתִּיק לְשׁוֹן שֶׁקֶר אֱמוּנִים לַחֲתוֹף, / שָׂפָה לְהַרְגִּיז שׁוֹקְטִים מַחְלֶקֶת.

אַל תַּאֲמֵן בָּהּ, כִּי בְכֶתֶם הָאֱלִיל / נֶחְפֵּת וְכֶסֶף הַמְּרִי נֶחְשֶׁקֶת.

לוּ הָדְרָה שָׂרִים וְאָז לִבּוֹת בְּנֵי / אָבִי בְתוּגָה לֹא תְהִי דוֹחֶקֶת,

אוֹ נִכְּרוּ שׁוֹעִים לְפָנֶיהָ – יְדֵי / אָחִי לְבַדּוֹ מִיְרוֹא נוֹשֶׁקֶת,

אוֹ נָשְׂאָה פָנִים לְמַשְׂכִּיל – מַחֲמַד / עֵינָיו בְּחַיָּיו לֹא תְהִי עוֹשֶׁקֶת,

אוֹ תַאֲרִיךְ אַפָּהּ לְתָם – בֵּינוֹ לְבֵין / אֵשֶׁת נְעוּרָיו לֹא תְהִי חוֹלֶקֶת.

הִיא הָאֱמוּנָה תוֹךְ נְוֵה צֶדֶק וְשַׁד / הַתֹּם עֲלֵי חֵיק הָאֱמֶת יוֹנֶקֶת,

הַמַּעֲטִירָה, וַעֲנַק יֹפִי וְחֵן / אֶל גַּרְגְּרוֹת יַחַשׂ וְתֹם מַעְנֶקֶת,

גִּבְעַת לְבוֹנָה גָבְהָה, אֵיכָה עֲלֵי / כִתְפוֹת אֲנָשִׁים הָיְתָה נֶעְתֶּקֶת?

אֵיךְ עֻפְּרָה עָפְרָה בְּגוּשׁ עָפָר, וְעִם / בַּדֵּי שְׁאוֹל אֵיךְ הָיְתָה נֶאְבֶּקֶת?

בָּהּ נָגְעָה הַיָּד אֲשֶׁר כֶּתֶם וְסִיג / אִישִׁים בְּעָפָר חָמְדָה לָצֶקֶת,

פִּטְדָה בְמַרְגֵּמָה פְתָאִים זָרְקוּ, / וּבְגַל אֲבָנִים אִבְּדוּ בָרֶקֶת,

בֵּין עָם אֲשֶׁר הָיוּ וְנֶעְתָּקוּ אֱלֵי / צִיָּה קְרוֹבָה הִיא וּמִתְרַחֶקֶת,

דֶּרֶךְ רְחוֹקָה רָחֲקוּ וְלֹא יִיעֲפוּ / תוֹכָהּ וְרַגְלָם לֹא תְהִי בָצֶקֶת.

אַחִים – וְהֵפֵרוּ בְלִי שִׂנְאָה בְרִית / אָחוֹת וְכַחוּט הָיְתָה נִתֶּקֶת,

גֵּרִים אֲשֶׁר לֹא יִקְרְאוּ עוֹזֵר, אֲבָל / גָּרִים אֲשֶׁר לֹא יַעֲנוּ צוֹעֶקֶת.

לֹא יִירְאוּ מִשֹּׁד אֲשֶׁר יָבֹא, הֲכִי / בָם לֹא תְהִי עוֹד יַד אֱנוֹשׁ דּוֹבֶקֶת.

יָשְׁנוּ שְׁנַת עוֹלָם וְיַזְהִירוּ אֲשֶׁר / עֵינָם פְּקוּחָה בַזְּמָן עוֹתֶקֶת.

הִיא נֶאֱלָמָה, אַךְ תְּיַסֵּר כָּל מְתֵי / לֵבָב, וְתוֹכִיחַ – וְהִיא שׁוֹתֶקֶת.

בָּנוּ מְסֻכָּתָהּ בְּתוֹכָם וַעֲלֵי / לִבּוֹת וּבָבוֹת הָיְתָה נֶחְקֶקֶת.

שֶׁמֶשׁ אֲחוֹתָהּ בּוֹגְדָה הִיא יוֹם עֲלֵי / חוּג הָאֲדָמָה תִהְיֶה נֹסֶקֶת,

כֵּן בָּהּ רְעוּתָהּ הַלְּבָנָהּ כִחֲשָׁה / אִם אַחֲרֶיהָ תַּעֲלֶה בוֹרֶקֶת,

עַיִשׁ תְּחַלֵּל הַבְּרִית אִם לֹא עֲלֵי / שָׂרַת בְּנוֹתֶיהָ תְּהִי נוֹאֶקֶת.

[עַב הַשְּׁחָקִים] אִם בְּכִיתָהּ תַּחֲשֹׂךְ / וּשְׂפַת שְׁחָרִים אִם תְּהִי שׂוֹחֶקֶת.

עָמְדָה עֲלֵי מַשְׁבֵּר וְהִמְלִיטָה, אֲבָל / זָכָר אֲשֶׁר לֹא תִהְיֶה חוֹבֶקֶת.

חֶבְלֵי שְׁאוֹל סַבּוּ פְעָמֶיהָ וְהִיא / נוֹטָה עֲלֵי דוֹדָהּ וּמִתְרַפֶּקֶת:

זָכְרָה בְרִית נֹעַר וְאַט עַל שַׁעֲרֵי / קֶבֶר תְּהִי יַד אַהֲבָה דוֹפֶקֶת,

וּשְׁמֹר אֱמוּנָתִי לְהוֹמוֹת אַחֲרֵי / אֵם נוֹסְעָה מֵהֶם וּמִתְפָּרֶקֶת,

וּכְתֹב אֱלֵי דוֹדִי, הֲכִי אֵשׁ הַנְּדוֹד / הָיְתָה בְנַפְשׁוֹ בַעֲדִי נִשֶּׂקֶת,

הַנֵּר אֲשֶׁר נִסְגַּר בְּבוֹר נֵּרוּת וְכַף / מָצוֹק עֲלֵי יֶרֶךְ נְדֹד סוֹפֶקֶת,

יִתְאַב שְׁתוֹת כּוֹסוֹת עֲרֵבִים, וְכוֹס / אָסוֹן לְכָמוֹהָ תְּהִי נִמְתֶּקֶת.

אָחִי, לְךָ נֶפֶשׁ בְּיַלְדֵי הַזְּמָן / תַּלְעִיג וּבִפְנֵי יוֹלְדָם יוֹרֶקֶת,

וּלְךָ תְעוּדָה כֹאֲבִים תִּשְׁלַח – וְאֵיךְ / עֹל תַּאֲלָתְךָ לֹא תְהִי פוֹרֶקֶת?

הִיא יִסְּרָה רַבִּים וְהִשְׂכִּילָה עֲדַת / כֶּסֶל אֲשֶׁר מֵהָאֱמֶת עוֹרֶקֶת,

רִיבָה בְדִמְעָה עַד בְּלִי דַי תַּעֲרֹף / וּגְעַר בְּנֶפֶשׁ לַאֲלוֹת נִזְעֶקֶת.

גֶּבֶר עֲלֵי מַה זֶּה יְקוֹנֵן עַח אֲבֹד / נֶאְהָב – וְנַפְשׁוֹ אַחֲרָיו דּוֹלֶקֶת,

אִם יַעֲלֶה שַׁחַק וְיַצִּיעַ שְׁאוֹל – / אוֹתוֹ יְמִין מָוֶת תְּהִי מַדְבֶּקֶת.

לוּ הַבְּכִי יוֹעִיל קְבוּרָתָהּ בְּמֵי / רֹאשִׁי מְצוּקִים עָצְמוּ מוֹחֶקֶת

וּלְשׁוֹן תְּקוּמָה רוּחֲכֶם תָּשִׁיב בְּקוֹל / נֹחַם וְלַנֵּר תִּהְיֶה שׁוֹרֶקֶת.

נֹכַח אֲדֹנָי יִהְיֶה דֶרֶךְ אֲשֶׁר / אֶל גַּן־עֲדָנִים עוֹבְרָה חוֹמֶקֶת,

עָנָן עֲלֵי קִבְרָהּ יְסוֹכֵךְ כָּל יְמֵי / אֶרֶץ וְהָעָב לוֹ תְהִי מוֹצֶקֶת.
