הֲדָמִי תִדְרְשִׁי מִפִּי לְבָאִים

הֲדָמִי תִדְרְשִׁי מִפִּי לְבָאִים / וְנוּמִי תִשְׁאֲלִי מִידֵי צְבָאִים

וְדַם חשֵׁק הֲיִגָּאֵל וְשֵׁנָה / הֲתֶעְרַב לוֹ וּמַכְאֹבָיו מְנִיאִים

כְּאִלּוּ כָחֲלוּ עֵינָיו בְּרֶשֶׁף / וְאִישׁוֹנָיו בְּקוֹץ מַכְאִיב מְלֵאִים

וְעַפְעַפָּי לְהִתְעָרֵב רְחוֹקִים / כְּאִלּוּ הֵם בְּגַבּוֹתָם תְּלוּאִים

5 וְהַלַּיִל בְּיָם אָפְלוּ מְנֻדָּח / בְּלִי רַעַשׁ וְגַלָּיו לֹא נְשֻׂאִים

וּמִיָּם רָחֲבוּ יָדָיו בְּעֵינַי / וְאֵין לוֹ חוֹף וְלֹא עֵבֶר לְבָאִים

וְסַהַר יַהֲלֹךְ יָקָר כְּרֹעֶה / יְנַהֵל אַט בְּכַר נִרְחָב טְלָאִים

וְאִם שֶׁמֶשׁ שְׁלָחָהו מְאַסֵּף / לְמַחְנוֹתָיו וְשָׂמוֹ שַׂר צְבָאִים

וְלֹא אָבִין הֲיֶאְרַךְ אוֹ הֲיִקְצַר / וְאֵיךְ יָבִין אֱנוֹשׁ יִשָּׂא חֳלָאִים

10 אֱהִי כֹפֶר לְלֵב הִסְתִּיר אֲהָבָיו / וְחָשַׁב כָּל מְתֵי תֵבֵל מְשַׂנְאִים

וְעַל עֵינַי חֲמָסִי לֹא לְלִבִּי / וְהַדִּמְעָה לְשׁוֹן צֶדֶק לְרֹאִים

כְּאִלּוּ נוֹצְרוּ עֵינַי נְבָלִים / לְהַשְׁכִּיב אֶת לְחָיַי אוֹ נְשִׂיאִים

לְנוּד רֵעִים יְצָאֻנִי וְאֵינָם / וְאִם הֵמָּה בְּרַעְיוֹנַי מְצוּאִים

וּבָגְדוּ בִי יְצֻרֵיהֶם עֲדֵי כִי / בְנוּמִי מִקְּרוֹב אֵלַי יְרֵאִים

15 אֲהָהּ לִזְמָן אֲשֶׁר יָזַם לְהַפְרִיד / אֲחֵי לִבִּי וְתוּגוֹתָיו לְהַתְאִים

יְמֵי חַיַּי בְּצוּף אַהַב שְׂבֵעִים / וּמִיֵּין הַנְּעוּרִים רַק סְבוּאִים

בְּאֶרֶץ נָעֲמָה מִכָּל אֲרָצוֹת / וְגָרֶיהָ כְּפִי חֶפְצָם בְּרוּאִים

וְחֶלְקָה מִכְּבֵדִי בָהּ חֲלָקִים / וְגִזְרָה כָל גְּזָרַי בָּהּ בְּלוֹאִים

עֲזַבְתִּימוֹ הֲכִי מֵתוּ מְתֵיהֶם / וְהֵם בִּלְתָּם כְּמוֹ בָתֵּי כְלָאִים

20 וְהָאִישִׁים נְשָׁמוֹת לַמְּכוֹנִים / וְאִם יִפָּקְדוּ פָּקְדוּ רְפָאִים

לְגָרִים אֶתְאֲבָה לֹא לַמְּגוּרִים / וְלִמְתָאֵי חֲסָדִים לֹא לְתָאִים

וְלַבָּנִים אֲבָל לֹא לַלְּבֵנִים / וְלַבָּאִים הֲכִי לֹא לַמְּבוֹאִים

הֱדִיחַנִי זְמַנִּי מֵהֲמוֹנָם / וְהִפְקִיד בִּי שְׁכֹן מִדְבַּר פְּרָאִים

פְּרָאִים אַךְ לְפַת שֵׂכֶל רְעֵבִים / וְאֶל מֵי הָאֱמוּנָה הֵם צְמֵאִים

25 כְּמַחְרִשִׁים וְחָרָשִׁים לְהַשְׁחִית / וְחַטָּאִים וּבָרִים מַחְטִיאִים

וְהֵם מִתְחַכְּמִים אַךְ לֹא חֲכָמִים / וּמִתְנַבְּאִים וְלֹא בַחְזוֹן נְבִיאִים

וְרוּח אַהֲבָתָם לֹא לְהָבַר / וְלֹא לִזְרוֹת כְּרוּח צַח שְׁפָאִים

וְאַךְ שָׁם הַזְּמָן אִנָּה לְיָדִי / שְׁתִיל בֹּשֶׂם בְּבֵין גִנַּת דְּשָׁאִים

וְשׁוֹשָׁן יַעֲלֶה בֵּין קוֹץ וְרָמִים / הֲדַסִּים בֵּין עֲצֵי יַעַר וְגֹאִים

30 וְהָאֹדֶם יְהִי בִימְעֵי אֲדָמָה / וְשָׁם כֶּסֶף דְּדָן וּזְהַב שְׁבָאִים

בְּמַרְאֵהוּ לְבַד שׂבַע שְׂמָחוֹת / וּבִדְבָרָיו תְּרוּפַת מַחֲלֻאִים

וְלוֹ יֶתֶר שְׂאֵת עַל כָּל יְדֻעִים / וְיוֹדוּ לוֹ הֲכִי הֵם אַךְ נְכָאִים

וְמַדּוֹת מַחֲמַדָּיו מֹר לְאַפִּי / וַּדְוֵיהֶם מְטֻנָּפִים וְצוֹאִים

וְיָם חַסְדּוֹ כְּצוּף יִמְתָּק וּמֵימֵי / עֲלִילוֹתָם כְּמוֹ בִצֹּאת גְּבָאִים

35 וְנֹפֶת יִגְרְשׁוּ מֵימָיו וּבֹשֶׂם / וְהֵם רֶפֶשׁ וְטִיט יָוֵן מְקִיאִים

שְׁאֵר הַשְׂכֵּל וְלֹא בָשָׂר וְהֵמָּה / אֲחֵי בָשָׂר בְּלִי אַהַב קְרֻאִים

קְחָה שִׁירִים תְּסַכֵּל בָּם חֲכָמִים / בְּעֵינֵיהֶם וְתַחְכִּים הַפְּתָאִים

חֲרָזִים יַחְמְדוּ נָגְהָם מְאוֹרִים / וְאַנְשֵׁי הַדְּבָרִים בָּם מְקַנְאִים

וְאִם עִמִּי מְלָאָם לֵב לְרוֹצֵץ / אֲנִי הַסּוּס וְהֵמָּה הַמְּרִיאִים

40 וְיִהְיוּ אִם יְשׁוֹרֵרוּ לְפָנַי / כְּכוֹס מִדְבָּר וְכַסִּיס בֵּין עֳפָאִים

וְאִם שִׁירִי לְכָל שִׁירִים כְּצֶפֶת / אֲנִי פֶלֶא בְּעֵינֵי הַפְּלָאִים.
