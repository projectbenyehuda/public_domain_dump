אַבְקַת בְּשָׂמִים

אל יוסף


אַבְקַת בְּשָׂמִים מַעֲשֵׂה רוֹקֵחַ / אוֹ מִפְּאַת אָח מָר־דְּרוֹר נוֹפֵחַ?

וּמְאוֹר פְּנֵי שֶׁמֶשׁ וְאִם זֹהַר כְּתָב / יָדוֹ בְּאֹפֶל הַנְּדוֹד זוֹרֵחַ?

פְּקַד אֲשֶׁר נֶהְפַּךְ בְּבוֹר פֵּרוּד עֲדֵי / קָרָא זְמָן מוֹעֵד לְהִפָּתֵחַ,

מוֹעֵד אֲשֶׁר נָשָׂא לְבָבוֹ בַעֲדוֹ / אֶל אֵל וְכַפָּיו יוֹם לְיוֹם שׁוֹטֵחַ.

נָדָה תְמוּנָתוֹ בְּלַיִל יֶאֱסֹף / אוֹרוֹ וְאֹהֶל מַאֲפֵל מוֹתֵחַ,

רָעָה גְדוּדֵי רוֹם וְרָעָה כוֹכְבֵי / נֶשֶׁף כְּשׁוֹשַׁנִּים בְּגַן פּוֹרֵחַ,

אַךְ חָשְׁכוּ וַיִּכְבְּדוּ מֵרוּץ עֲדֵי / דָמוּ לְעַם עִוֵּר וְגוֹי פִּסֵּחַ.

אֶשֹּׁם לְבָרָק מִפְּאַת אַרְצוֹ בְּעֵת / יִבְרֹק בְּאֵשׁ מֵעַי וְיִתְלַקֵּחַ,

וְאֶשְׁאֲפָה רוּחַ כְּתַנִּים, כִּי נְעִים / זִכְרוֹ לְאַפִּי מָר־דְּרוֹר שׁוֹלֵחַ.

אֶפְעֶה וְאֶדְמֶה אֶל נְשׁוּךְ אֶפְעֶה עֲלֵי / עֶרֶשׂ דְוָי מֵהַכְּאֵב צוֹרֵחַ.

יִרְגַּז לְבָבִי עַד יְהִי פוּךְ מַאֲפֵל / נוֹתֵן וְשֹׁהַם אוֹר יְהִי לוֹקֵחַ.

יִלָּחֲמוּ צִירַי וּמֵי עֵינַי עֲדֵי / בֹקֶר, וְאַךְ חֵיל הַבְּכִי רוֹצֵחַ.

שֶׁפַע לְיַמִּים בֵַּין צְלָעַי יֶהֱמוּ, / אָכֵן יְקוֹד הַתַּאֲוָה קוֹדֵחַ.

כָּלוּ בְהֶבֶל כָּל־יְמוֹתַי בִּלְעֲדֵי / יוֹסֵף וְכָל־שָׁנַי כְּצֵל בּוֹרֵחַ,

עַד קָצְרָה נַפְשִׁי נְשׂוֹא מוּסַר נְדֹד / אָחִי וְרַצְתִּי אֲחָרָיו שָׂמֵחַ,

רַצְתִּי לְהִנָּחֵם בְּצוּפֵי מַעֲנָיו / וּבְנַחֲלַת דּוֹדָיו לְהִסְתַּפֵּחַ,

כִּי מַעֲרָב לֹא יֶעֱרַב בִּלְתּוֹ וְהֵן / מִזְרָח בְּלִי פָנָיו כְּבֵית מַרְזֵחַ.

כַּמָּה הֱנִאוּנִי מְרִיבִים בֶּאֱמֹר: / אַט, פֶּן תְּהִי דֹאֵב וְתֵאָנֵחַ,

אֵיךְ מַאֲמִין תִּהְיֶה בְּאִישׁ הֵפֶךְ וְעַל / קָנֶה מְאֹד רָצוּץ תְּהִי בוֹטֵחַ?

רוֹפְאֵי אֱלִיל אַתֶּם וּמִדְבַּרְכֶם כְּמוֹ / רוּחַ אֲשֶׁר יַעֲבֹר וְיִשְׁתַּכֵּחַ.

אֵלֵךְ וְאֶרְאֶנּוּ בְטֶרֶם יַד זְמָן / מִבְנֵה בְחוּרוֹתַי יְהִי נוֹסֵחַ.

נַמְתִּיק בְּיַחַד סוֹד וְשֶׁמֶן מִדְבְּרֵי / פִיו עַל שְׁחִין לִבִּי יְהִי מוֹרֵחַ.

אֶשְׂמַח בְּשׁוּרִי אֶת־פְּנֵי בָנָיו אֲשֶׁר / לִבִּי לְמַרְאֵיהֶם כְּעוֹף פּוֹרֵחַ.

רִכְבִּי אֲשֶׁר קָרַב מְפַחֵד מִקְּרֹב / וּבְמֵי קְרָבוֹת לִהְיוֹת צוֹלֵחַ,

לִקְרַאת פְּעָמָיו בֶּן־עֲצָתִי אִם יְהִי / מוֹצִיא – עֲלֵי נַפְשִׁי יְהִי פוֹסֵחַ.

עָלָיו כְּאַחְוָתִי שְׁלוֹמִים וַעֲלֵי / הַתָּם יְהוֹסֵף עַד בְּלִי יָרֵחַ,

הַדּוֹד אֲשֶׁר הִרְחִיק נְדֹד, אַךְ מֶרְקְחֵי / דּוֹדָיו לְרָחוֹק יִתְּנוּ לִי רֵיחַ.

קִשְׁרֵי יְדִידוּת אוֹ בְרִית הָאַהֲבָה / תִּשְׁכַּח יְמִינִי אִם אֱהִי שׁוֹכֵחַ.

חֵי נַפְשְׁכֶם אִם אֶשְׁקְטָה רֶגַע עֲדֵי / עֵינַי עֲלֵיכֶם אֶהְיֶה פוֹקֵחַ.

אֶשָּׂא אֲזַי לִזְמָן, וְאִם הִסְכִּיל עֲשֹׂה / עִמִּי לְחֵטְא בָנָיו אֱהִי סוֹלֵחַ.

הִנֵּה תְּשׁוּרַת שִׁיר אֲשֶׁר אֶרֶץ עֲלֵי / שַׁחַק בְּהוֹדָהּ רָץ לְהִשְׁתַּבֵּחַ,

הוֹבִישׁ מְאוֹר שֶׁמֶשׁ וְרֵיחוֹ רָם עֲלֵי / אַבְקַת בְּשָׂמִים מַעֲשֵׂה רוֹקֵחַ.
