מִי כָמוֹךָ אֶפֶס יֵשׁ מִבַּלְעֲדֶיךָ

מי כמוך


מִי כָמוֹךָ? / אֶפֶס יֵשׁ מִבַּלְעֲדֶיךָ; / אוֹדְךָ בְמַעֲשֵׂי יָדֶיךָ –

אֲרוֹמִמְךָ, אֱלוֹהַי הַמֶּלֶךְ.

מִי כָמוֹךָ? / בְּמָתְחֲךָ אֹהֶל מֵישָׁרִים, / הוֹאַלְתָּ לִפְתֹּחַ סְפָרִים –

וַיִּהְיוּ נִקְרָאִים לִפְנֵי הַמֶּלֶךְ.

מִי כָמוֹךָ? / רַחֲמֶיךָ וּשְׁפָטֶיךָ נִצָּבוּ / וּבְרוּאֶיךָ כַּצֹּאן נֶחְשָׁבוּ,

עוֹבְרִים לִפְנֵי הַמֶּלֶךְ.

מִי כָמוֹךָ? הַיּוֹדֵעַ צְפוּן חֻבָּם, / אָנָה יִסָּתְרוּ, וְלִבָּם –

הוּא מְסַפֵּר לַמֶּלֶךְ?

מִי כָמוֹךָ? / מַקְשִׁיב לְקוֹל מְשַׁוֵּעַ, / עָנִי אֲנִי וְגֹוֵעַ,

מַפִּיל אֲנִי תְחִנָּתִי לִפְנֵי הַמֶּלֶךְ.


אֶחָד לְבַדּוֹ נַעֲלָה / כִּסֵּא הַכָּבוֹד הַנִּפְלָא,

אַבְנֵי קֹדֶשׁ מְמֻלָּא – / אַפִּרְיוֹן עָשָׂה לוֹ הַמֶּלֶךְ.

בָּנָה מוֹסְדֵי עוֹלָם / וּבְפִתּוּחַ קֹדֶשׁ נְעָלָם,

בְּאוֹתִיּוֹת שְׁמוֹ כְּלָלָם / וְנֶחְתַּם בְּטַבַּעַת הַמֶּלֶךְ.

גָּזַר אֹמֶר לִהְיוֹת / בְּגֻלּוֹת עִלִּיּוֹת וְתַחְתִּיּוֹת

אוֹר מִתְרוֹצֵץ, לִחְיוֹת / בְּאוֹר פְּנֵי הַמֶּלֶךְ.

דָּבָר שָׁלַח וְהִרְקִיעַ, וּמִקְוֵה / יַמִּים כּוֹנֵן וַיְצַוֶּה,

וּמוּצָק הֶעֱמִיד בְּשָׁוֶה – / הוּא עֵמֶק הַמֶּלֶךְ.

הִדְשִׁיאָה בְכֹחוֹ אָז חֳרָבָה / וּמָלְאוּ פְנֵי תֵבֵל תְנוּבָה –

וְעוֹד אוֹרֵחַ לֹא בָא / אֶל שֻׁלְחַן הַמֶּלֶךְ.

וַיִּתְרוֹצְצוּ סָבִיב עֶלְיוֹנִים, / שִׁבְעָה פְּנִימִים חִיצוֹנִים –

הֵמָּה הַשָּׂרִים הָרִאשׁוֹנִים / הַמְשָׁרְתִים אֶת פְּנֵי הַמֶּלֶךְ.

זְמִירוֹת רָצִים לְקוֹנֵיהֶם, / בְּלִי יִסַּבּוּ פְּנֵיהֶם,

וְהַשֶּׁמֶשׁ הוֹלֵךְ לִפְנֵיהֶם – / הַשָּׁלִישׁ אֲשֶׁר לַמֶּלֶךְ.

חֻקּוֹת הֲלִיכָתָהּ נוֹדְעָה: / בַּבֹּקֶר יָצְאָה וְנָסְעָה,

וְלָעֶרֶב עֵת נָגְעָה / וַתָּבֹא וַתִּשְׁתַּחוּ לַמֶּלֶךְ.

טוֹבוֹת אוֹצְרוֹתָיו בְּיָדוֹ, / יַעַל אוֹר וְיוֹרִידוֹ,

וְיִתְּנֵהוּ לְיָרֵחַ פְּקִידוֹ – / וְהוּא יִתֵּן מַעֲדַנֵּי מֶלֶךְ.

יִמְלָא וְיֶחְסָר לִזְמָן, / וּפַעַם יֵרָאֶה – וְנִטְמַן,

בַּשַּׁחַק לְעֵד נֶאֱמָן, / אֹמֵר: אָנִי – מַעֲשַׂי לַמֶּלֶךְ.

כִּמְלֹאת חַיּוֹת שָׁמַיִם / שָׁרְצוּ חַיּוֹת בַּמַּיִם,

לַעֲשׂוֹת לְתוּשִׁיָּה כִּפְלַיִם, / כִּי כֵן יִסַּד הַמֶּלֶךְ.

לְהוֹצִיא מֵאֶרֶץ בְּהֵמוֹת / קָרָא, וַיַּרְא בְּחָכְמוֹת,

כִּי הַנִּבְרָא בָאֲדָמוֹת / רָב כְּיַד הַמֶּלֶךְ.

מָדַד רוּחוֹת עוֹלָמוֹ / וַיִּבְרָא אָדָם בְּצַלְמוֹ,

עַל כָּל הַיְקוּם הֱרִימוֹ, / וַיֹּאמְרוּ: יְחִי הַמֶּלֶךְ.

נוֹצַר הַכֹּל בְּעִתּוֹ / וַיַּעֲמִידֵהוּ עַל מַתְכֻּנְתּוֹ,

וַיְכַל כָּל מְלַאכְתּוֹ / בַּיּוֹם הַשְּׁבִיעִי, כְּטוֹב לֵב הַמֶּלֶךְ.

סוֹד בָּאָדָם נָצַר, / בִּגְלָלוֹ בַכֹּל עָצַר,

אָכֵן הוֹשִׁיבוֹ בַחֲצַר / גִּנַּת בִּיתַן הַמֶּלֶךְ.

עֵצָה הִגְדִּיל וְהִפְלִיאָהּ / וּמִצַּלְעוֹ רַעְיָה הוֹצִיאָהּ,

עָדָיו כַּכַּלָּה הֱבִיאָהּ – / כְּבוּדָּה בַת מֶלֶךְ.

׳פְּרֵה וּרְבֵה בְּחֶדְלְךָ! / רְאֵה, הָעֲשָׂבִים לְאָכְלְךָ

וְהַחַיּוֹת לַעֲבָדִים לְךָ – / וְאַתָּה הֹוֶה לָהֶם לְמֶלֶךְ!׳

צִוָּם לִמְנֹעַ יְדֵיהֶם / מֵעֵץ הַדַּעַת – וְהֵם

הֶאֱמִינוּ בַנָּחָשׁ, וּשְׁנֵיהֶם / שָׁמְעוּ בְּצַוֹּת הַמֶּלֶךְ.

קַל מְהֵרָה יָרְאוּ / מִקּוֹל מִתְהַלֵּךְ – וְנֶחְבְּאוּ,

וּמִגַּן עֵדֶן יָצְאוּ / מְבֹהָלִים וּדְחוּפִים בִּדְבַר הַמֶּלֶךְ.

רָעוֹת חָשַׁב הֲפַכְפָּךְ / וְיָדוֹ כְּאַכְזָר הָפַךְ,

וְדָם אָחִיו שָׁפָךְ – / וְהוּא צָעַק אֶל הַמֶּלֶךְ.

׳שֹׁרֶשׁ פּוֹרֶה שֵׁרַשְׁתָּ / וְנַחַל נוֹבֵעַ יִבַּשְׁתָּ;

הֲרָצַחְתָּ וְגַם יָרָשְׁתָּ? / וְכָל דָּבָר לֹא יִכָּחֵד מִן הַמֶּלֶךְ!׳

תְּמִים דֵּעוֹת הוֹרָה / אוֹתוֹ בְּדֶרֶךְ יְשָׁרָה

הוֹדָה חֶטְאוֹ, וְקָרָא / כִּי קָרוֹב הַמֶּלֶךְ.

אֶרֶץ טְהוֹרָה הֶחְנִיפוּ / וּנְבִיאִים לָהֶם הִטִּיפוּ,

אָז לַמְרוֹת הוֹסִיפוּ / בְּהִשָּׁמַע דְּבַר הַמֶּלֶךְ.

בִּרְאוֹתוֹ הַדּוֹר לֹא שָׁב / מִשְּׁרִירוּת לִבּוֹ חָשַׁב –

יְיָ לַמַּבּוּל יָשָׁב, / וַיֵּשֵׁב יְיָ מֶלֶךְ.

גָּאַל נֹחַ בְּמוֹעֵצָה, / כִּי חֵן בְּעֵינָיו מָצָא,

וּמִתֵּבָה לְיָמִים יָצָא / כְּשֹׁךְ חֲמַת הַמֶּלֶךְ.

דִּבֶּר בְּעִתִּים בָּאוֹת: / ׳לֹא אַשְׁחִית עוֹד הַבְּרִיאוֹת!׳

וְקֶשֶׁת בֶּעָנָן לְאוֹת / בַּיּוֹם הַהוּא נָתַן הַמֶּלֶךְ.

הִתְחַבְּרוּ עָלָיו, בְּהִתְיָעֲצָם / לִבְנוֹת מִגְדָּל כְּחֶפְצָם,

לְאַרְבַּע רוּחוֹת לַהֲפִיצָם / הַדָּבָר יָצָא מִפִּי הַמֶּלֶךְ.

וְאָב הֵטִיב בְּמַעֲלָלָיו, / הַכָּבוֹד חוֹפֵף עָלָיו,

וּמַלְאָךְ נִרְאָה אֵלָיו: / ׳הִנֵּה חָפֵץ בְּךָ הַמֶּלֶךְ!׳

זָנַח אֱלִילֵי הוֹרוֹ / וַיְּגַדֵּף מֶלֶךְ דּוֹרוֹ –

נִשְׁמַע בַּגּוֹיִם דְּבָרוֹ / וַיְדַבְּרוּ הַכַּשְׂדִּים לַמֶּלֶךְ:

׳חֻקּוֹתֶיךָ לֹא שָׁמַר / וְלַעֲבֹד אֱלֹהָיו אָמַר,

בִּכְבוֹד שְׁמוֹ יִתְיַמָּר – / וְזֶה אֲשֶׁר הֵרִים יַד בַּמֶּלֶךְ.׳

׳טָעָה לִבְּךָ וַיִּתְגַּבֵּר – / לְמִי בָאֵשׁ תְּשַׂבֵּר?

מַדּוּעַ אַתָּה עוֹבֵר / אֵת מִצְוַת הַמֶּלֶךְ?׳

יָדַעְתִּי, וְעֵינַי צוֹפִיּוֹת / הֲלִיכוֹת רַב עֲלִילִיּוֹת,

וּמָחָר נְפָשׁוֹת וּגְוִיּוֹת / יָבֹאוּ לְמִשְׁפָּט אֶל הַמֶּלֶךְ.

כִּשְׁמוֹ כֵּן תְּהִלָּתוֹ, / מְשַׁלֵּם לַכֹּל כִּפְעֻלָּתוֹ,

וְנֶפֶשׁ תַּעֲבֹד זוּלָתוֹ – / לֹא תָבוֹא עוֹד אֶל הַמֶּלֶךְ.

לְאוּר כַּשְׂדִּים הוֹרִידוּהוּ, / וְכַאֲשֶׁר נִצַּל – הוֹדוּהוּ:

׳אֱלֹהֶיךָ אֱמֶת, נַעַבְדֵהוּ!׳ / וְגַם כָּכָה אָמַר הַמֶּלֶךְ:

׳מֵעֲבוֹדָה נָכְרִיָּה צֵאוּ, / אַךְ הָאֱלֹהִים תִּירָאוּ,

וּבְצֵאת נַפְשְׁכֶם – תָּבוֹאוּ / בֵּית הַמַּלְכוּת אֲשֶׁר לַמֶּלֶךְ!׳

נִסָּהוּ בִּבְנוֹ יְחִידוֹ / וּבְהַר הַמּוֹרִיָּה עֲקָדוֹ,

וְלַמַּאֲכֶלֶת שָׁלַח יָדוֹ, / כִּי כֵן צִוָּה לוֹ הַמֶּלֶךְ.

סָעֲרוּ בַשָּׁמַיִם תּוֹרוֹת, / וּסְבִיבוֹת הַכֵּס מִדּוֹת

הַדִּין וְהָרַחֲמִים עוֹמְדוֹת, / וַתְּדַבֵּרְנָה לִפְנֵי הַמֶּלֶךְ.

עֶלְיוֹנִים הָיוּ לַאֲגֻדָּה / בַּעֲמֹד מֵלִיץ חֶמְדָּה,

וְטָלֶה בְאַיִל נִפְדָּה / עַל יַד אִישׁ מִשָּׂרֵי הַמֶּלֶךְ.

׳פְּדֵה בָנַי בְּשַׁוְּעָם, / בָּהָר הַנִּבְחָר בְּפָגְעָם,

וְהַיּוֹם הַזֶּה לְזַרְעָם / יִזְכָּר נָא הַמֶּלֶךְ!׳

צָמְחוּ תְּאוֹמִים וְתָמִים / בִּשְׁנֵים עָשָׂר שְׁלֵמִים

לָבַשׂ מַלְבּוּשׁ מְרוֹמִים, / אֲשֶׁר לָבַשׁ בּוֹ הַמֶּלֶךְ.

קָדוֹשׁ הִפְלִיא מַעֲשִׂים – / וּבַעַל כְּתֹנֶת הַפַּסִּים

נִמְכַּר לְעֶבֶד, וְתַחְפְּנֵסִים אֹתוֹ מָשְׁחוּ לְמֶלֶךְ.

רָעָב קָרָא בְהַגִּיעַ / קֵץ, לְאַב הָמוֹן הִבִּיעַ,

וּבֵן לְאָבִיו הוֹדִיעַ / כִּי הֱכִינוֹ יְיָ לְמֶלֶךְ.

שָׁבָה רוּחוֹ בְּהִתְבַּשְּׂרוֹ / וּמִצְרַיְמָה יָרְדוּ בַעֲבוּרוֹ –

וַיָּמָת יוֹסֵף וְדוֹרוֹ / וַיָּקָם מֶלֶךְ.

׳תִּרְאוּ יוֹעֲצִים וְתֶחֱשׁוּ, / טֶרֶם אַרְצֵנוּ יָרְשׁוּ –

וְעַתָּה יִקְשְׁרוּ וִיבַקְּשׁוּ / לִשְׁלֹחַ יַד בַּמֶּלֶךְ.

אֲשֶׁר יוֹלִידוּ – נַשְׁמִידֵם, / אוּלַי מִגּוֹי נַכְחִידֵם,

וְהַחַיִּים – בְּפֶרֶךְ נַעֲבִידֵם!׳ / וַיֶּחֱזַק דְּבַר הַמֶּלֶךְ.

בָּנִים צָעֲקוּ מֵעֲבוֹדָה – / וְרִגְשַׁת מָרוֹם נוֹעֲדָה,

וּבְרִית הוֹרִים עָמְדָה / וַתְּדַבֵּר לִפְנֵי הַמֶּלֶךְ.

׳רוֹעֶה, שְׁלַחְתִּיךָ מִמְּרוֹמִי / לְהוֹצִיא מִמִּצְרַיִם עַמִּי,

הִתְחַזֵּק בְּאוֹתוֹת שְׁמִי / וּבָאתָ אֶל הַמֶּלֶךְ!׳

מָתְנָיו שִׁנֵּס עָנָיו / וְהוֹדִיעַ לְאָחִיו עִנְיָנָיו,

וַיִּנְאֲמוּ לְתַנִּין וּקְצִינָיו: / ׳כֹּה אָמַר הַמֶּלֶךְ:

בָּנַי בְּאַרְצְךָ בְּעַבְדּוּת – / שָׁלַחְתִּי אֲלֵיהֶם פְּדוּת!׳

וַיַעַן בֶּן הַמַּרְדוּת: / ׳מִי הוּא זֶה מֶלֶךְ?׳

׳רַבָּה תִפְלֶצֶת בְּקִרְבְּךָ / וְהִשִּׂיאֲךָ זְדוֹן לִבְּךָ,

וְגַם אֵין תְּבוּנָה בְךָ – / מִי אַתָּה קָרָאתָ אֶל הַמֶּלֶךְ?

מִי כָמֹהוּ מוֹרֶה / וְיֵשׁ מֵאַיִן בּוֹרֵא,

וְדוֹרוֹת מֵרֹאשׁ קוֹרֵא? / יְיָ צְבָאוֹת הוּא מֶלֶךְ!׳

אָמְרוּ לְמֹשֶׁה מֻכִּים: / ׳בְּעֵינֵינוּ שַמְתָּם לְשִׂכִּים;

וּלְמִי אָנוּ מְחַכִּים – / וְשֹׁמֵעַ אֵין לְךָ מֵאֵת הַמֶּלֶךְ!׳

יָשׁוּב אֵלָיו חוֹזֶה: / ׳דְּבָרֶיךָ תַּנִּין בּוֹזֶה,

לָמָּה יְקַלֵּל הַכֶּלֶב הַמֵּת הַזֶּה / אֶת אֲדֹנִי הַמֶּלֶךְ?׳

׳רָאֹה תִרְאֶה בַסּוֹף, / מִי אֲפַזֵּר וְאֶאֱסֹף;

יָמִין מְחוֹלֶלֶת אֶחְשֹׂף – / בַּיּוֹם הַהוּא יֹאבַד לֵב הַמֶּלֶךְ!׳

׳עָרִיץ, שַׁלַּח בָּנַי, / פֶּן תִּסָּפֶה בַחֲרוֹנַי;

מָאַסְתָּ אֶת דְּבַר יְיָ – / וַיִּמְאָסְךָ מִמֶּלֶךְ!׳

זוּ מַטּוֹת רְשָׁעִים / בְּמַטֵּה אַהֲרֹן נִבְלָעִים,

יַכֵּם בְּדַם, וּצְפַרדְּעִים / תְּבֹאֶינָה בְּהֵיכַל מֶלֶךְ.

׳רְאֵה, אֲשַׁלַּח הֲמוֹנֶיךָ – / וְהָסֵר מֵעָלַי חֲרוֹנֶךָ!׳

וַיֹּאמֶר: ׳אֵצֵא מִפָּנֶיךָ / וּמָחָר אֶעֱשֶׂה כִּדְבַר הַמֶּלֶךְ.׳

הִקְשָׁה לִבּוֹ חוֹטֵא – / וְרוּחַ יְיָ תְּבַטֵּא:

׳הַךְ הֶעָפָר בַּמַּטֶּה / וּבַבֹּקֶר אֱמֹר לַמֶּלֶךְ.׳

קִמְּטָם בְּעָרוֹב וְדֶבֶר / וּשְׁחִין אֲבַעְבֻּעוֹת וְשֶׁבֶר,

אַרְבֶּה מֵבִישׁ מִשֵּׂבֶר / לְהִנָּקֵם בְּאֹיְבֵי הַמֶּלֶךְ.

טָבְעוּ בַּחֹשֶׁךְ זֵידוֹנִים – / וְעוֹד הֵעֵזוּ פָנִים

עַד מוּת רֵאשִׁית אוֹנִים; / בַּלַּיְלָה הַהוּא נָדְדָה שְׁנַת הַמֶּלֶךְ.

נִקְהֲלוּ וְיַחַד בְּמַר צָרְחוּ / ׳צֹאנְכֶם וּבְקַרְכֶם קְחוּ –

וְקוּמוּ צְאוּ וְזִבְחוּ / כַּטּוֹב בְּעֵינֵיכֶם, בְּשֵׁם הַמֶּלֶךְ.׳

חֲמוּשִׁים עָלוּ מִמְּצָרֵיהֶם – / וְהֵם מְקַבְּרִים בְּכוֹרֵיהֶם,

וּמֶרְכָּבוֹת וְשָׁלִישִׁים אַחֲרֵיהֶם / יָצְאוּ דְחוּפִים בִּדְבַר הַמֶּלֶךְ.

זְרוֹעַ רָמָה הִתְעוֹרְרָה / וּבַיָּם עֲבָדִים נִעֲרָה,

וּבַיַּבָּשָׁה עָבְרָה הָעֲבָרָה / לַעֲבִיר אֶת בֵּית הַמֶּלֶךְ.

קוֹל נָשְׂאוּ כֻלָּם: / ׳אָשִׁירָה לַייָ בְּמַהֲלָלָם,

׳יְיָ יִמְלֹךְ לְעֹלָם / וְהָיָה יְיָ לְמֶלֶךְ!׳
