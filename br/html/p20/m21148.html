מִנְּדוֹד אָמוֹן

מִנְּדוֹד אָמוֹן בְּחֵיק

מִשְׂרָה, יֵאָבְקוּ

עִמִּי תְּלָאוֹת

ולכנהם אלסבק

נָדֵד וְשָׂבְעָה נְדוּדִים עֵינִי אַחֲרָיו,

גַּם כָּהֲתָה מִבְּכוֹת מִפְקַד יְמֵי זָהֳרָיו

שֶׁמֶשׁ לְרוֹאָיו וְכֵן שֶׁמֶשׁ לִמְשַׁחֲרָיו.

הָה לְבַב עָזַב בְּצוּק

פֵּרוּד, יִנָּתְקוּ

חוּטָיו, ומתל גנאה

אלטיר יכפק.

דַּי לִי בְּצִבְאוֹת זְמַן מֶמֶר אֲשֶׁר פָּשְׁטוּ

בָּזוּ כְבוֹדִי, וְדִין עָוֶל בִּי שָׁפְטוּ

אַל תַּאֲמֵן בּוֹ, וְתַהְפֻּכָיו לֹא שָׁקְטוּ,

אִם לְךָ יוֹם יַחֲלִיק

לָשׁוֹן, וְאִם מָתְקוּ

מִלָּיו, גדא פיגר

סהמא בה תרשׁק.

אַךְ יִצְחָק הוּא גְבִיר עַמּוֹ וְאוֹת הַזְּמָן,

יָתֵד לְחָכְמוֹת תְּקוּעוֹת בְּמָקוֹם נֶאֱמָן,

שֶׁבּוֹ מְרוֹרִים אֲשֶׁר אֶשְׂבַּע מְתוּקִים כְּמָן;

יוֹם אֲשֶׁר כָּזַב אֲפִיק

צֶדֶק, וְגַם נֶעְתְּקוּ

מֵימָיו וְיָבְשׁוּ, פקד

אצחי בה יפהק,

דֶּגֶל חֲסָדִים מְאַסֵּף אַתְּ לְכָל מַחֲנוֹת,

חָכְמָה וְשֵׂכֶל וּבִין, וּבְקִרְבְּךָ שׁוכְנוֹת,

כִּי מִנְּעֻרִים לְךָ נֹשְׂאוֹת כְּמוֹ אוֹמְנוֹת,

אַל תִּהְיוּ יוֹגְעִים לְרִיק

שָׂרִים, וְהִתְאַפְּקוּ

הדא אלדי עאלם

אלעלא בה ישפק.

אֶפְדֶּה בְנַפְשִׁי צְבִיַּת חֵן אֲשֶׁר אֵחֲרָה

יוֹם שׁוּב דְּלָתַי, אֲשֶׁר הִיא שָׁלְחָה יַעֲרָה

רֵיקָם, וְלֹא מָצְאָה צַיִד אֲשֶׁר שִׁבְּרָה

אלגזאל שק אלחריק

ואלסלאק תרהק

מא חזני אלא חריד

אדא לם ילחק.
