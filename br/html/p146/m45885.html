לֵךְ עִמִּי בֹּא בַּחֲדָרֶיךָ

רצון הלוי – שירת ישראל בתימן

חתימה: לאלשבזי


לֵךְ עִמִּי בֹּא בַּחֲדָרֶיךָ / וְחַפֵּשׂ מִתְּמוֹל דְּרָכֶיךָ

יְרָא אֶת יְוֹיָ בִּלְבָבֶךָ / אוּלַי יְרַחֵם עָלֶיךָ

שׁוּבָה וְיָשׁוּב אֵלֶיךָ / וִיְרַפֵּא אֶת מְשׁוּבָתֶךָ

לְמַעַן תִּצְדַּק בְּדָבְרֶךָ / תִּזְכֶּה בְּשָׁפְטֶךָ:


אֱהוֹב צוּרֶךָ וַעֲשֵׂה רְצוֹנוֹ / וְהַרְחֵק מִלִּבָּךְ זְדוֹנוֹ

וְסוּרָה מֵהִרְהוּר עֲווֹנוֹ / הַגּוֹרֵם עָלָיו יְגוֹנוֹ

טוֹב עֶבֶד אוֹהֵב אֲדוֹנוֹ / וְתוֹרַת חֶסֶד עַל לְשׁוֹנוֹ

בְּלֶכְתְּךָ תִּשְׁמוֹר צַעֲדֶךָ / וַהַקִיצוֹתָ הִיא תְּשִׂיחֶךָ:


לַיִל תַּעֲמוֹד בְּאַשְׁמוֹרֶת / בְּרוֹב תַּחֲנוּנִים וַעֲתֶרֶת

וְנַפְשְׁךָ לְטוּב אֵל נוֹהֶרֶת / מִצְוֹת וְחוּקִים שׁוֹמֶרֶת

וְאוֹרַח חַיִּים בּוֹחֶרֶת / וּמִכָּל פְּגָעִים נִזְהֶרֶת

וְיָפוּצוּ אוֹיְבֶיךָ / וְיָנוּסוּ מְשַׂנְאֶיךָ:


שׁוּבָה וְתִכּוֹן בְּלֵב תָּמִים / מִכָּל עֲווֹן וַאֲשָׁמִים

וְלֹא תִירָא פַּחַד אֵמִים / וְהַרְגָּשַׁת אוֹיְבִים וְקָמִים

סוּר מֵרָע וּמֵאִישׁ דָּמִים / וְקַח נָא מוּסָרֵי חֲכָמִים /

וְהָיָה שַׁדַּי בְּצָרֶיךָ / יִזְכּוֹר כָּל מִנְחוֹתֶיךָ:


בְּעֶזְרַת אֵל חַי מַלְכֵּנוּ / יִשְׂמַח וְיָגֵל לִבֵּנוּ

עֶזְרֵנוּ וּמָגִנֵּנוּ / יָאֵר פָּנָיו אִתָּנוּ

בְּסֵפֶר חַיִּים יִכְתְּבֵנוּ / וּמִכָּל צָרָה יְמַלְּטֵנוּ

וְיַאֲזִין בְּרָצוֹן שַׁוְעֵנוּ / יִמְחֶה אֶת כָּל פְּשָׁעֵינוּ

אֱלֹהִים בְּרֹב חַסְדֶּךָ / עֲנֵנִי בְּאֶמֶת יִשְׁעֶךָ:


זְכוּת לַמְּדוּ עָלַי אָבוֹת / לִפְנֵי צוּר שׁוֹכֵן עֲרָבוֹת

יְשַׂמַּח נְפָשׁוֹת נֶעֱצָבוֹת / אֲשֶׁר בְּיוֹם הַדִּין נִצָּבוֹת

בְּסֵפֶר חַיִּים נִכְתָּבוֹת / יַעֲלוּ גַּלְגַּל וּמֶרְכָּבוֹת

נְטֵה לִבִּי לִיְמִינֶךָ / לְהֵיטִיבְךָ בְּאַחֲרִיתֶךָ:


יוֹצֵר כֹּל בְּחָכְמָה וָדַעַת / יְרַחֵם אֲיוּמָה נִכְנַעַת

כִּי בְּשֵׁם קָדְשׁוֹ נוֹשַׁעַת / בְּגָלוּתָהּ מִתְיַגַּעַת

בְּעֵמֶק עָכוֹר טוֹבַעַת / וּמֵעִיר אֶל עִיר נוֹסַעַת לְ

קוֹל נִחוּמִים שׁוֹמַעַת / בְּחֵן תּוֹרָה מִשְׁתַּעֲשַׁעַת

צוּרִי רְצֵה עַל עַמֶּךָ / יִכְבְּשׁוּ רַחֲמֶיךָ אֶת כַּעֲסֶךָ:
